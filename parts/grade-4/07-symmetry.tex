\chapter{Symmetry}\label{ch:g4:symmetry}

Butterflies, faces, snowflakes: nature loves figures whose two halves
match when folded. This chapter explores symmetry with folds and grid
paper --- constructions with the set square wait for
\cref{ch:g6:symmetry}.

\section{Folding and matching}

\begin{definition}[Axis of symmetry]\label{def:g4:symmetry:def}
A line is an \emph{axis of symmetry}\index{axis of symmetry} of a
figure when folding the figure along that line makes the two halves
match \emph{exactly}. A figure can have no axis, one axis, or
several.
\end{definition}

\end{omfigure}

\begin{example}[Testing with tracing paper]\label{ex:g4:symmetry:trace}
Trace the figure, flip the tracing paper over along the candidate
axis, and lay it back: if the tracing covers the figure exactly, the
axis is genuine. Eyes alone are often fooled --- a rectangle's
diagonal \emph{looks} like an axis, but the fold test says no (try
it!).
\end{example}

\section{Completing figures on a grid}

\begin{method}[Symmetric of a figure across a grid line]\label{met:g4:symmetry:grid}
The axis is a vertical (or horizontal) line of the grid.
\begin{enumerate}
  \item Take each vertex of the figure in turn;
  \item count its distance to the axis in squares;
  \item place the image vertex at the \emph{same} distance on the
        \emph{other} side, on the same row (or column);
  \item join the image vertices in the same order, and check: the
        image is the mirror twin of the figure.
\end{enumerate}
\end{method}

  \draw[omThm, very thick] (6,-0.4) -- (6,5.4);

\begin{example}[Same shape, same size, flipped]\label{ex:g4:symmetry:properties}
The image has exactly the same lengths and the same angles as the
original: symmetry copies the figure --- but flips it over, like a
left hand and a right hand. If the original letter is F, its mirror
image is a backwards F, not another F.
\end{example}

\section{Axes of the usual shapes}

\begin{example}[Counting axes]\label{ex:g4:symmetry:shapes}
By folding:
\begin{itemize}
  \item a square: $4$ axes (two through the middles of opposite
        sides, two along the diagonals);
  \item a rectangle: $2$ axes (middles of opposite sides only!);
  \item an isosceles triangle: $1$; an equilateral triangle: $3$;
  \item a circle: every line through its center --- more axes than
        one can count.
\end{itemize}
\end{example}

\begin{example}[Symmetry in the alphabet]\label{ex:g4:symmetry:letters}
A, M, T, U, V have a vertical axis; B, C, D, E have a horizontal
one; H, I, O, X have both. Words made of the right letters can be
symmetric too: MUM has a vertical axis; OXO works both ways.
\end{example}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:g4:symmetry:1}
Trace and fold: which of these have an axis of symmetry --- a
square; a (non-square) rectangle; a scalene triangle (all sides
different); a circle?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:symmetry:2}
Draw the axes of symmetry of: a square; a rectangle; an equilateral
triangle. How many for each?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:symmetry:3}
Use the tracing-paper test of \cref{ex:g4:symmetry:trace} to show
that the diagonal of a $6 \times 4$ rectangle is \emph{not} an axis
of symmetry.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:symmetry:4}
On grid paper, draw a vertical axis and the letter L (three squares
tall, two wide) at $2$ squares from the axis. Construct its mirror
image.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:symmetry:5}
On grid paper, draw a horizontal axis and a small flag (a $1 \times
1$ triangle on a $3$-square mast) above it. Complete the figure by
symmetry below the axis.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:symmetry:6}
A figure touches its axis of symmetry at a point $P$. Where is the
image of $P$? Explain with the folding picture.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:symmetry:7}
Sort the capital letters A, B, E, F, H, N, O, T: vertical axis?
horizontal axis? both? neither?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:symmetry:8}
A triangle has sides $5$ cm, $5$ cm and $3$ cm. Does it have an
axis of symmetry? Which sides does the fold exchange?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g4:symmetry:9}
On grid paper, draw a figure of your own that has \emph{exactly two}
axes of symmetry, and draw both axes. (Hint: think of a rectangle,
or of a cross.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g4:symmetry:10}
Half of a symmetric drawing is given: the left half of a house
(a square wall and half a triangular roof), with a vertical axis
along its right edge. Describe or draw the complete house. How do
the lengths of the completed right half compare to the left half?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g4:symmetry:11}
Aya claims: ``every triangle with two equal sides has an axis of
symmetry, and every triangle with an axis of symmetry has two equal
sides.'' Test her double claim on drawings: an isosceles triangle,
an equilateral one, a scalene one. Does the folding support her?
\end{exercise}
