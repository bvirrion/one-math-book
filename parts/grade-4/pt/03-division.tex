\chapter{Divisão}\label{ch:g4:division}

Grau 3 dividido com as mesas e pequenos restos
(\cref{ch:g3:division}); este capítulo instala o texto completo escrito
método --- divisão longa --- que lida com qualquer número, grupo de dígitos por
grupo de dígitos.

\section{O método de divisão longa}

\begin{method}[Divisão longa por um número de um dígito]\label{met:g4:division:method}
Para dividir $749$ por $6$:
\begin{enumerate}
  \item pegue o dígito mais à esquerda, $7$ (centenas): $6$ vai $1$ hora
        em $7$; escrever $1$ no \omterm{def:g3:division:remainder}{quociente}, subtrair
        $6$, \omterm{def:g3:division:remainder}{restante} $1$;
  \item \emph{derrubar} o próximo dígito, $4$: agora divida $14$
        (dezenas): $6 \times 2 = 12$; escrever $2$, \omterm{def:g3:division:remainder}{restante} $2$;
  \item derrubar o $9$: dividir $29$: $6 \times 4 = 24$; escrever
        $4$, \omterm{def:g3:division:remainder}{restante} $5$;
  \item nenhum dígito \omterm{def:g3:division:remainder}{restante}: o \omterm{def:g3:division:remainder}{quociente} é $124$, o \omterm{def:g3:division:remainder}{restante} $5$.
\end{enumerate}
Verificar: $6 \times 124 + 5 = 744 + 5 = 749$, e $5 < 6$.
\end{method}

\begin{example}[Um zero no quociente]\label{ex:g4:division:zero}
Dividir $618$ por $3$: centenas, $6 \div 3 = 2$; dezenas, abaixe o
$1$: $3$ vai $0$ vezes em $1$ --- \emph{escreva o} $0$! ---
\omterm{def:g3:division:remainder}{restante} $1$; unidades, derrubar o $8$: $18 \div 3 = 6$. \omterm{def:g3:division:remainder}{Quociente}
$206$, \omterm{def:g3:division:remainder}{restante} $0$. Esquecendo o meio \omterm{def:g1:counting:zero}{zero} (escrita $26$)
é o erro clássico; o cheque
$3 \times 26 = 78 \neq 618$ pega imediatamente.
\end{example}

\begin{example}[Estimando o tamanho do quociente]\label{ex:g4:division:size}
Antes de dividir $749$ por $6$, enquadre a resposta:
$6 \times 100 = 600$ e $6 \times 200 = 1\,200$, então o \omterm{def:g3:division:remainder}{quociente} é
entre $100$ e $200$ --- terá três dígitos. Esta linha única
estimativa evita a maioria dos erros de dígitos mal colocados.
\end{example}

\section{Escolhendo o que a pergunta pergunta}

\begin{method}[Quociente, resto ou quociente mais um]\label{met:g4:division:interpret}
Como em \cref{met:g3:division:interpret}, a história decide:
\begin{enumerate}
  \item ``quantos completos \dots'' ou ``quantos cada'': o \omterm{def:g3:division:remainder}{quociente};
  \item ``quantos restam'': o \omterm{def:g3:division:remainder}{restante};
  \item ``quantos contêineres são necessários para todos'': o \omterm{def:g3:division:remainder}{quociente}, mais
        um se o \omterm{def:g3:division:remainder}{restante} não é \omterm{def:g1:counting:zero}{zero}.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g4:division:problem}
$530$ os livros devem ser embalados em caixas de $8$.
\[
530 = 8 \times 66 + 2 .
\]
Quantos \emph{completo} caixas? $66$. Quantos livros não estão completos
caixa? $2$. Quantas caixas embalar \emph{todos} os livros? $67$.
Três perguntas, uma divisão.
\end{example}

\begin{example}[Compartilhamento justo de dinheiro]\label{ex:g4:division:money}
Quatro amigos dividem o custo de um $92$ presentear igualmente:
$92 \div 4 = 23$ exatamente (\omterm{def:g3:division:remainder}{restante} $0$). Cada um paga $23$. Quando o
\omterm{def:g3:division:remainder}{restante} é \omterm{def:g1:counting:zero}{zero}, a divisão ``fica empatada'' e a partilha
é perfeitamente justo.
\end{example}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g4:division:1}
Enquadre cada \omterm{def:g3:division:remainder}{quociente} como em \cref{ex:g4:division:size} (entre os quais
dois ``redondo'' números?), então calcule a divisão longa:
$96 \div 4$; \quad $87 \div 5$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:division:2}
Calcule as divisões longas e verifique cada uma delas:
$672 \div 4$; \quad $925 \div 7$; \quad $804 \div 6$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:division:3}
Cuidado com o \omterm{def:g1:counting:zero}{zero} (ver \cref{ex:g4:division:zero}):
$816 \div 4$; \quad $420 \div 7$; \quad $2\,512 \div 5$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:division:4}
Escreva a igualdade de verificação ($ a = b \times q + r $) para: $85 \div 9$;
\quad $1\,000 \div 3$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:division:5}
$156$ os ovos são colocados em caixas de $6$. Quantas caixas estão preenchidas?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:division:6}
Uma fita de $250$ cm é cortado em pedaços de $8$ cm. Quantas peças,
e que comprimento sobra?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:division:7}
$375$ estudantes vão a um show em ônibus de $52$ assentos. Quantos ônibus
são necessários? (Qual caso de
\cref{met:g4:division:interpret} é isto? Dividir por $52$ usando
múltiplos: $52 \times 7 = 364$.)
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:division:8}
Três amigos compartilham $234$ bolinhas de gude igualmente. Quantas bolinhas de gude cada?
Verifique com uma multiplicação.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:division:9}
Encontre o dividendo: a divisão por $7$ dá \omterm{def:g3:division:remainder}{quociente} $58$ e
\omterm{def:g3:division:remainder}{restante} $3$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g4:division:10}
Uma prateleira de bibliotecário $438$ livros, $9$ por prateleira. As prateleiras entram
estantes de $6$ prateleiras. Quantas prateleiras são necessárias? Quantos
estantes devem ser compradas? (Duas divisões, ambas com
pergunta ``mais um''.)
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g4:division:11}
Numa divisão por $8$, o \omterm{def:g3:division:remainder}{quociente} é igual ao \omterm{def:g3:division:remainder}{restante}. O que são
os possíveis dividendos? Liste todos eles. (O \omterm{def:g3:division:remainder}{restante} deve ficar
abaixo $8$.)
\end{exercise}
