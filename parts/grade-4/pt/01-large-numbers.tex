\chapter{Grandes números}\label{ch:g4:numbers}

Milhares eram o teto de \cref{ch:g3:numbers}; agora o teto
sobe para \emph{milhões}. O truque que torna legíveis números enormes
é cortar seus dígitos em grupos de três --- o aulas.

\section{Classes de três dígitos}

\begin{definition}[Classes]\label{def:g4:numbers:classes}
Começando pela direita, os dígitos de um número são agrupados em
\emph{aulas} de três: a classe de \emph{unidades}, a classe de
\emph{milhares}, a classe de \emph{milhões}. Cada aula tem seu
próprias unidades, dezenas e centenas. Por exemplo
\[
6\,309\,452 = 6 \text{ millions} + 309 \text{ thousands}
+ 452 \text{ units},
\]
leia ``seis milhões trezentos e nove mil e quatrocentos
cinquenta e dois''.
\end{definition}

\begin{example}[Escrevendo com as aulas]\label{ex:g4:numbers:write}
``Doze milhões e quarenta mil e quinhentos'' tem $12$ no
classe de milhões, $040$ na classe dos milhares, $500$ nas unidades
aula:
\[
12\,040\,500 .
\]
O \omterm{def:g1:counting:zero}{zeros} de $040$ são essenciais: cada classe (exceto a mais à esquerda)
deve mostrar seus três dígitos.
\end{example}

\begin{example}[Quantos milhares?]\label{ex:g4:numbers:contained}
O número $6\,309\,452$ \emph{contém} $6\,309$ milhares
(tudo o que resta da classe de unidades): $6\,309\,452 = 6\,309 \times
1\,000 + 452$. Perguntar “quantos milhares” não é pedir um
dígito, mas para um prefixo inteiro do número.
\end{example}

\section{Comparando e arredondando}

\begin{method}[Comparando grandes números]\label{met:g4:numbers:compare}
Exatamente como em \cref{met:g3:numbers:compare}, com o aulas como um
atalho: mais dígitos ganham; mesmo comprimento, compare a partir da esquerda ---
classe por classe e depois dígito por dígito.
\end{method}

\begin{example}\label{ex:g4:numbers:compare}
$2\,145\,890$ e $2\,148\,540$: milhões iguais ($2$), milhares
aula $145 < 148$, então
\[
2\,145\,890 < 2\,148\,540 .
\]
\end{example}

\begin{definition}[Arredondamento]\label{def:g4:numbers:round}
Para \emph{arredondar}\index{arredondar} um número para o milhar mais próximo
(ou dez, cem, milhões\,\dots), substitua-o pelo múltiplo mais próximo
de $1\,000$. Se o número estiver exatamente na \omterm{def:g3:division:half}{metade}, arredonde. Em
prática: observe o dígito logo abaixo do local de arredondamento --- $5$ ou
mais rodadas, $4$ ou menos arredondado para baixo.
\end{definition}

\begin{example}\label{ex:g4:numbers:round}
$37\,614$ \omterm{def:g4:numbers:round}{arredondado} para o milhar mais próximo: o algarismo das centenas é
$6$, então redondo acima: $38\,000$. \omterm{def:g4:numbers:round}{Arredondado} para a centena mais próxima:
$37\,600$. \omterm{def:g4:numbers:round}{Arredondado} para os dez mil mais próximos: $40\,000$.
\end{example}

\begin{example}[Ordens de magnitude no mundo]\label{ex:g4:numbers:world}
Arredondamento é como os grandes números são realmente usados: um país de
$67\,842\,591$ habitantes é ``sobre $68$ milhão''; uma cidade de
$214\,376$ pessoas é `` sobre $200\,000$''. O número exato muda
todos os dias de qualquer maneira --- o \omterm{def:g4:numbers:round}{arredondado} carrega o útil
informação.
\end{example}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g4:numbers:1}
Escreva em algarismos, com o aulas separados: ``três milhões e cinco
cento e vinte mil''; ``sete milhões e oito''; ``duzentos
quarenta mil e sessenta''.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:numbers:2}
Escreva em palavras: $5\,204\,300$; \quad $10\,050\,000$; \quad
$999\,999$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:numbers:3}
Em $8\,472\,156$: qual dígito representa as dezenas de milhares? As centenas
da classe dos milhões? Quantos \emph{milhares} faz o número
contém (ver \cref{ex:g4:numbers:contained})?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:numbers:4}
Copie e complete com $<$ ou $>$:
\[
980\,000 \;?\; 1\,020\,000, \qquad
3\,456\,000 \;?\; 3\,465\,000, \qquad
7\,090\,050 \;?\; 7\,090\,500 .
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:numbers:5}
Ordene do menor para o maior: $2\,300\,000$; \ $980\,500$; \
$2\,030\,000$; \ $2\,299\,999$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:numbers:6}
Redondo $63\,782$: até a dezena mais próxima; para a centena mais próxima; para o
milhar mais próximo; para os dez mil mais próximos.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:numbers:7}
\omterm{def:g4:numbers:round}{Arredonde} para o milhão mais próximo: $4\,721\,000$; \quad $6\,499\,999$;
\quad $9\,500\,000$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:numbers:8}
Conte por etapas: três etapas de $1\,000$ de $998\,500$; dois passos
de $100\,000$ de $950\,000$. O que acontece com o aulas como você
ultrapassar um milhão?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:numbers:9}
Um estádio comporta $81\,338$ espectadores. Um jornal escreve “mais do que
$80\,000$ fãs''; um rádio diz ``sobre $81\,000$''. São ambos
certo? Para qual lugar cada um redondo?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g4:numbers:10}
Usando cada um dos dígitos $1$ para $7$ exatamente uma vez, escreva o maior
número possível de sete dígitos, o menor e o mais próximo de
$4\,000\,000$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g4:numbers:11}
Qual é o maior número inteiro que rodadas para $50\,000$ quando
arredondamento para o milhar mais próximo? E o menor?
\end{exercise}
