\chapter{Linhas e Polígonos}\label{ch:g4:geometry}

Com uma régua, um esquadro e um compasso, a geometria se torna uma arte.
Este capítulo ensina as duas posições especiais das linhas ---
\omterm{def:g4:geometry:def}{perpendicular} e \omterm{def:g4:geometry:def}{paralelo} --- no nível da ferramenta e faz um tour
o \omterm{def:g4:geometry:polygon}{polígono} família. (A teoria dessas noções surge
\cref{ch:g6:lines}.)

\section{Perpendiculares e paralelos}

\begin{definition}[Perpendicular, paralelo]\label{def:g4:geometry:def}
Duas linhas são \emph{perpendicular}\index{perpendicular} quando
eles cruzam em um \omterm{def:g3:shapes:rightangle}{ângulo reto} (verificado com o esquadro, marcado com
um pequeno quadrado). Duas linhas são \emph{paralelo}\index{paralelo}
quando eles correm na mesma direção e nunca se cruzam --- como os dois
trilhos de uma via, sempre à mesma distância um do outro.
\end{definition}

\begin{method}[Desenhando com o esquadro]\label{met:g4:geometry:draw}
\emph{\omterm{def:g4:geometry:def}{Perpendicular} para uma linha $d $ através de um ponto $ P $}: coloque uma borda
do \omterm{def:g3:shapes:rightangle}{ângulo reto} junto $ d $, deslize até que a outra borda alcance $ P $,
empate.

\emph{\omterm{def:g4:geometry:def}{Paralelo} para $ d $ através $ P $}: coloque a borda do esquadro ao longo
$ d $, pressione a régua contra outra borda como um trilho, deslize o conjunto
quadrado ao longo da régua até $ P $, empate.
\end{method}

\begin{example}[Identificando-os em números]\label{ex:g4:geometry:spot}
Em um retângulo, cada lado é \omterm{def:g4:geometry:def}{perpendicular} aos seus dois vizinhos e
\omterm{def:g4:geometry:def}{paralelo} para o lado oposto. O papel quadriculado é uma rede pronta de
\omterm{def:g4:geometry:def}{paralelos} e \omterm{def:g4:geometry:def}{perpendiculares} --- é isso que faz com que seja assim
confortável para desenhar.
\end{example}

\section{Polígonos}

\begin{definition}[Polígono]\label{def:g4:geometry:polygon}
UM \emph{polígono}\index{polígono} é uma figura fechada cuja borda é
feito de segmentos retos, seu \emph{lados}; os pontos de canto são
o \emph{vértices}\index{vértices}. Os polígonos são nomeados pelo número de lados:
\emph{triângulo} ($3$), \emph{quadrilátero}\index{quadrilátero} ($4$),
\emph{Pentágono}\index{Pentágono} ($5$), \emph{hexágono}\index{hexágono} ($6$),
\emph{octógono}\index{octógono}
($8$).
\end{definition}

\begin{example}[Descrevendo um polígono]\label{ex:g4:geometry:describe}
Um jogo útil: descreva uma figura com tanta precisão que um colega possa
redesenhar sem ver. ``Um \omterm{def:g4:geometry:polygon}{quadrilátero} $ ABCD $ com
$ AB = 5$ cm, o ângulo em $ A $ certo, $ AD = 3$ cm\,\dots'' ---
vocabulário é o que faz a mensagem funcionar.
\end{example}

\section{O círculo e a bússola}

\begin{definition}[Círculo]\label{def:g4:geometry:circle}
O \emph{círculo} do centro $ O $ e \omterm{def:g3:shapes:shapes}{raio} $3$ cm é o conjunto de todos
aponta exatamente $3$ cm de $ O $. A bússola desenha: pico em
$ O $, abrindo $3$ cm, vire. UM \emph{diâmetro}\index{diâmetro} é um segmento através
o centro une dois pontos do círculo; mede o dobro
\omterm{def:g3:shapes:shapes}{raio}.
\end{definition}

\begin{example}[A bússola como guardião da distância]\label{ex:g4:geometry:compass}
A bússola faz mais do que desenhar círculos: ela \emph{transportes
distâncias}. Para marcar todos os pontos em $4$ cm de um ponto $ A $, desenhe o
círculo do centro $ A $, \omterm{def:g3:shapes:shapes}{raio} $4$ cm; comparar dois segmentos sem
uma régua, abra o compasso em um e teste no outro.
\end{example}

\begin{method}[Reproduzindo uma figura em uma grade]\label{met:g4:geometry:reproduce}
\begin{enumerate}
  \item Escolha um começo vértice e sua posição na grade em branco;
  \item viajar ao longo da figura vértice por vértice, contando a grade
        passos (``$3$ certo, $2$ acima'');
  \item junte-se ao \omterm{def:g4:geometry:polygon}{vértices} em ordem e compare com o modelo:
        mesmos comprimentos, mesmas direções.
\end{enumerate}
\end{method}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g4:geometry:1}
Desenhe uma linha $ d $ e um ponto $ P $ não nisso. Construa o
\omterm{def:g4:geometry:def}{perpendicular} para $ d $ através $ P $, então o \omterm{def:g4:geometry:def}{paralelo} para $ d $ através
$ P $ (com o método régua-trilho). Marque o \omterm{def:g3:shapes:rightangle}{ângulo reto}.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:geometry:2}
Observe as letras maiúsculas E, H, N, T, Z. Em cada uma, encontre um \omterm{def:g3:numbers:evenodd}{par} de
\omterm{def:g4:geometry:def}{paralelo} golpes e um \omterm{def:g3:numbers:evenodd}{par} de \omterm{def:g4:geometry:def}{perpendicular} golpes quando eles
existir.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:geometry:3}
Nomeie cada um \omterm{def:g4:geometry:polygon}{polígono}: $6$ lados; $4$ lados; $3$ lados; $8$ lados;
$5$ lados. Quantos \emph{\omterm{def:g4:geometry:polygon}{vértices}} cada um tem?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:geometry:4}
Desenhe qualquer \omterm{def:g4:geometry:polygon}{Pentágono} (não necessariamente regular) e rotule seu \omterm{def:g4:geometry:polygon}{vértices}
$ A $, $ B $, $ C $, $ D $, $ E $. Quantos lados ele tem? Desenhe seu
diagonais de $ A $ (segmentos juntando-se $ A $ para não vizinho \omterm{def:g4:geometry:polygon}{vértices}):
quantos são?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:geometry:5}
Desenhe um círculo de centro $ O $ e \omterm{def:g3:shapes:shapes}{raio} $4$ cm; um \omterm{def:g4:geometry:circle}{diâmetro} $[AB]$;
e um ponto $ M $ do círculo não ligado $[AB]$. Dê os comprimentos $ OA $,
$ OM $ e $ AB $ sem medir.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:geometry:6}
Com a bússola como guardião da distância
(\cref{ex:g4:geometry:compass}), compare os lados de um triângulo
desenhado por um colega e decida se tem dois lados iguais.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:geometry:7}
Em papel quadriculado, reproduza esse caminho e feche-o em uma \omterm{def:g4:geometry:polygon}{polígono}:
comece em um ponto $ A $; ir $4$ certo; $2$ acima; $3$ esquerda; $2$ acima;
então de volta para $ A $ em linha reta. Qual é o nome do \omterm{def:g4:geometry:polygon}{polígono}
você consegue (contar seus lados)?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:geometry:8}
Verdadeiro ou falso? Explique com um desenho.
``Dois linhas \omterm{def:g4:geometry:def}{perpendiculares} sempre cruze.
``Dois linhas paralelas nunca cruze.
``Uma linha pode ser \omterm{def:g4:geometry:def}{paralelo} para uma linha e \omterm{def:g4:geometry:def}{perpendicular} para
outro.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g4:geometry:9}
Desenhe uma linha $ d $, então uma linha $ e $ \omterm{def:g4:geometry:def}{perpendicular} para $ d $, então uma linha
$ f $ \omterm{def:g4:geometry:def}{perpendicular} para $ e $. Olhe para $ d $ e $ f $: o que você percebe?
(Esta observação se torna um teorema em \cref{ch:g6:lines}.)
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g4:geometry:10}
Descreva a figura abaixo em palavras para um colega de classe, sem mostrar
isto: um quadrado de lado $4$ cm, com um círculo de \omterm{def:g3:shapes:shapes}{raio} $2$ cm centrado
no meio do quadrado (tocando os quatro lados). Escreva o
descrição como instruções numeradas.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g4:geometry:11}
Quantas diagonais tem um \omterm{def:g4:geometry:polygon}{quadrilátero} ter? UM \omterm{def:g4:geometry:polygon}{Pentágono}? UM \omterm{def:g4:geometry:polygon}{hexágono}?
Desenhe os três casos, conte cuidadosamente e descreva o padrão que você
veja.
\end{exercise}
