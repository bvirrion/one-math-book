\chapter{Medidas e Perímetro}\label{ch:g4:measure}

Grau 3 medido com réguas, balanças e relógios
(\cref{ch:g3:measure}); este capítulo adiciona a tabela de unidades --- o
máquina que converte tudo --- e duas maneiras de medir um apartamento
forma: o comprimento de sua borda (\omterm{def:g4:measure:perimeter}{perímetro}) e a sala dentro dele
(\omterm{def:g4:measure:area}{área}, contado em quadrados por enquanto).

\section{A tabela de unidades}

\begin{method}[Convertendo com a mesa]\label{met:g4:measure:table}
Para comprimentos, cada coluna vale dez das seguintes:
\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{c|c|c|c|c|c|c}
km & hm & dam & m & dm & cm & mm \\
\hline
 & & & $3$ & $4$ & $0$ & \\
\end{tabular}
\end{center}
\begin{enumerate}
  \item Escreva o número com um dígito por coluna, o dígito das unidades
        da medida na coluna da sua unidade (aqui $3.4$ eu: o $3$
        sob m);
  \item leia o número na nova unidade: $3.4$ eu $= 340$ cm (preencher
        colunas vazias com \omterm{def:g1:counting:zero}{zeros});
  \item \omterm{def:g2:measure:units}{massas} (kg\,\dots\,g) e \omterm{def:g2:measure:units}{capacidades} (L\,\dots\,cL) trabalhar o
        da mesma maneira.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g4:measure:convert}
$5$ quilômetros $= 5\,000$ m; \quad $270$ cm $= 2.7$ m (o $2$ terras sob
m); \quad $3$ kg $250$ g $= 3\,250$ g; \quad
$2.5$ eu $= 250$ cl.
\end{example}

\section{Perímetro}

\begin{definition}[Perímetro]\label{def:g4:measure:perimeter}
O \emph{perímetro}\index{perímetro} de um \omterm{def:g4:geometry:polygon}{polígono} é o total
comprimento de sua borda: some os comprimentos de todos os seus lados (no mesmo
unidade!). Atalhos para os clássicos:
\[
\text{rectangle: } P = 2 \times (L + w),
\qquad
\text{square: } P = 4 \times c .
\]
\end{definition}

\begin{example}\label{ex:g4:measure:perimeter}
Um jardim retangular $12$ m por $7$ eu:
\[
P = 2 \times (12 + 7) = 2 \times 19 = 38 \text{ m}
\]
de cerca. Uma moldura quadrada de lado $18$ cm:
$ P = 4 \times 18 = 72$ cm.
\end{example}

\section{Área: contando quadrados}

\begin{definition}[Área por contagem]\label{def:g4:measure:area}
O \emph{área}\index{área} de uma figura desenhada em uma grade é o número
de quadrados da grade que cobre. Dois meios quadrados contam como um. Área
responde ``quanta superfície'', \omterm{def:g4:measure:perimeter}{perímetro} responde ``quanto tempo dura o
fronteira'' --- duas perguntas diferentes!
\end{definition}

\begin{omfigure}
\begin{tikzpicture}[scale=0.55]
  \draw[gray!40, thin] (0,0) grid (13,4);
  \draw[very thick, omDef, fill=omDef!15] (1,1) rectangle (5,3);
  \node[font=\small, omDef] at (3,2) {$8$ squares};
  \draw[very thick, omThm, fill=omThm!15]
    (7,1) -- (11,1) -- (11,2) -- (9,2) -- (9,3) -- (7,3) -- cycle;
  \node[font=\small, omThm] at (8.4,1.5) {$6$ squares};
\end{tikzpicture}

{\small Duas figuras na grade: \omterm{def:g4:measure:area}{áreas} $8$ e $6$ quadrados. Agora ande
as fronteiras e conte as unidades: $12$ para ambos! Mesmo \omterm{def:g4:measure:perimeter}{perímetro},
diferente \omterm{def:g4:measure:area}{áreas}.}
\end{omfigure}

\begin{example}[Meio-quadrados]\label{ex:g4:measure:half}
Um triângulo retângulo desenhado na grade com pernas $4$ e $2$ capas $3$
quadrados completos e $2$ meios-quadrados\,\dots\ conte com cuidado: é \omterm{def:g4:measure:area}{área}
é \omterm{def:g3:division:half}{metade} do $4 \times 2$ retângulo, ou seja,\ $4$ quadrados. Cortando um
retângulo ao longo de sua diagonal sempre dá dois triângulos de iguais
\omterm{def:g4:measure:area}{área}.
\end{example}

\begin{example}[Mesma área, perímetros diferentes]\label{ex:g4:measure:samearea}
UM $4 \times 4$ quadrado e um $8 \times 2$ retângulo ambas as tampas $16$
quadrados --- iguais \omterm{def:g4:measure:area}{área}. Fronteiras: $16$ unidades para o quadrado, $20$ para
o retângulo. Nenhuma medida determina a outra: um longo e fino
a forma tem muitas bordas para pouca superfície.
\end{example}

\section{Durações}

\begin{method}[Adicionando tempos ao longo da hora]\label{met:g4:measure:time}
Para adicionar ou subtrair durações, trabalhe em saltos durante as horas completas
(lembre-se $1$ h $= 60$ min):
\[
9\text{ h }40 + 35 \text{ min}:
\quad 9\text{ h }40 \xrightarrow{+20} 10\text{ h }00
\xrightarrow{+15} 10\text{ h }15 .
\]
Nunca adicione minutos passados $60$ sem converter: $40 + 35 = 75$ min
$= 1$ h $15$ min.
\end{method}

\begin{example}\label{ex:g4:measure:timetable}
O trem sai às $8{:}47$ e chega em $11{:}05$. Duração:
$8{:}47 \to 9{:}00$ é $13$ min; $9{:}00 \to 11{:}00$ é $2$ h;
$11{:}00 \to 11{:}05$ é $5$ min. Total: $2$ h $18$ min.
\end{example}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g4:measure:1}
Converta com a tabela de unidades: $7$ m em cm; $3.2$ km em m; $85$ mm em
cm; $4\,500$ m em km.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:measure:2}
Converter: $2$ kg em g; $1\,800$ g em kg e g; $3.5$ L em cL;
$25$ cL em L.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:measure:3}
Calcule o \omterm{def:g4:measure:perimeter}{perímetro}: um retângulo $9$ cm $\times $ $5$ cm; um quadrado
de lado $7.5$ cm; um triângulo com lados $6$ cm, $8$ centímetros e $10$ cm.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:measure:4}
Um retângulo tem \omterm{def:g4:measure:perimeter}{perímetro} $26$ cm e comprimento $8$ cm. Encontre sua largura
(primeiro encontre o comprimento $+$ largura).
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:measure:5}
Em papel quadriculado, desenhe uma figura em forma de L cobrindo exatamente $10$
quadrados e conte as unidades de seus \omterm{def:g4:measure:perimeter}{perímetro}.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:measure:6}
Desenhe dois retângulos diferentes de \omterm{def:g4:measure:area}{área} $12$ quadrados. Calcule o
\omterm{def:g4:measure:perimeter}{perímetro} de cada um. Qual é ``mais redondo'', qual é ``mais fino''?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:measure:7}
Um triângulo retângulo é desenhado na grade com pernas $6$ e $3$. O que é
é \omterm{def:g4:measure:area}{área} em quadrados? (Usar \cref{ex:g4:measure:half}.)
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:measure:8}
Calcule as durações: de $10{:}20$ para $12{:}45$; de $7{:}35$ para
$9{:}10$. Um filme dura $1$ h $50$ e começa em $20{:}30$: quando é que
acabou?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:measure:9}
Kim corre $3$ voltas de um campo retangular $60$ m por $45$ m. Quantos
metros ela corre?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g4:measure:10}
Um fazendeiro quer cercar um campo quadrado de lado $85$ m, deixando um
portão de $4$ estou sem cerca. Quantos metros de cerca ele deve comprar?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g4:measure:11}
Verdadeiro ou falso, com exemplos de grade: ``se duas figuras têm o mesmo
\omterm{def:g4:measure:perimeter}{perímetro}, eles têm o mesmo \omterm{def:g4:measure:area}{área}''; ``se duas figuras têm o mesmo
\omterm{def:g4:measure:area}{área}, eles têm o mesmo \omterm{def:g4:measure:perimeter}{perímetro}''.
(\Cref{ex:g4:measure:samearea} e a figura do capítulo ajudará.)
\end{exercise}
