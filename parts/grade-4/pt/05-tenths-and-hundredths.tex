\chapter{Décimos e Centésimos}\label{ch:g4:decimals}

Entre $3$ e $4$ os números inteiros são silenciosos --- mas um corredor pode
ser $3$ metros e um pouco à frente. Cortando a unidade em dez, depois um
cem, dá o \emph{decimal frações}, e um maravilhoso
abreviatura para eles: o \omterm{def:g4:decimals:point}{ponto decimal}. (Os números decimais recebem seus
capítulo completo em \cref{ch:g5:decimals}.)

\section{Frações decimais}

\begin{definition}[Décimos e centésimos]\label{def:g4:decimals:def}
Cortar uma unidade em $10$ partes iguais dão \emph{décimos},
$\frac{1}{10}$; cortando cada décimo em $10$ novamente dá
\emph{centésimos}, $\frac{1}{100}$. Então
\[
1 = \frac{10}{10} = \frac{100}{100},
\qquad
\frac{1}{10} = \frac{10}{100}.
\]
\end{definition}

\begin{example}\label{ex:g4:decimals:count}
Sombreamento $3$ colunas e $7$ células extras dos tons de cem quadrados
\[
\frac{3}{10} + \frac{7}{100} = \frac{30}{100} + \frac{7}{100}
= \frac{37}{100}
\]
do quadrado --- trinta e sete centésimos.
\end{example}

\section{O ponto decimal}

\begin{definition}[Escrita decimal]\label{def:g4:decimals:point}
O número $2 + \frac{3}{10} + \frac{7}{100}$ está escrito
\[
2.37
\]
--- o \emph{ponto decimal}\index{ponto decimal} separa a parte inteira ($2$) do
parte decimal; o primeiro dígito após o ponto conta os décimos, o
segundo os centésimos. Leitura: ``dois vírgula três sete'', ou ``dois
e trinta e sete centésimos''.
\end{definition}

\begin{example}\label{ex:g4:decimals:convert}
\[
\frac{7}{10} = 0.7, \qquad
\frac{43}{100} = 0.43, \qquad
\frac{5}{100} = 0.05, \qquad
3 + \frac{4}{10} = 3.4 .
\]
Cuidado com $\frac{5}{100}$: cinco \emph{centésimos} preciso de um $0$ em
a décima posição --- $0.05$, não $0.5$.
\end{example}

\begin{example}[O dinheiro é centésimos]\label{ex:g4:decimals:money}
Um centavo é um centésimo de um euro (ou dólar, ou libra): um preço
de $4.85$ significa $4$ moedas inteiras e $85$ centésimos. O dinheiro é o
\omterm{def:g4:decimals:point}{ponto decimal} você já usa todos os dias: $10$ centavos $= 0.10$; dois
cinquenta euros $= 2.50$; e $0.05$ é uma moeda de cinco centavos, não de cinquenta!
\end{example}

\section{Comparando com décimos}

\begin{method}[Comparando decimais (primeiro contato)]\label{met:g4:decimals:compare}
\begin{enumerate}
  \item Compare as partes inteiras primeiro: $3.2 > 2.9$.
  \item Mesmas partes inteiras: compare os décimos e depois os centésimos:
        $2.37 < 2.5$ porque $3$ décimos $< 5$ décimos.
  \item Escrever ambos com duas casas decimais ajuda: $2.5 = 2.50$, e
        $37 < 50$ centésimos.
\end{enumerate}
A armadilha: $2.37$ tem mais dígitos do que $2.5$, mas é
\emph{menor} --- mais dígitos não significa maior!
\end{method}

\begin{example}\label{ex:g4:decimals:compare}
Ordem $0.4$, $0.35$ e $0.09$: com centésimos, $0.40$, $0.35$,
$0.09$; então
\[
0.09 < 0.35 < 0.4 .
\]
\end{example}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g4:decimals:1}
Em um quadrado de cem, quanto fica sombreado se alguém sombrear: $5$ colunas;
$2$ colunas e $3$ células; $45$ células? Responda com frações.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:decimals:2}
Escreva com um \omterm{def:g4:decimals:point}{ponto decimal}:
$\frac{9}{10}$; \quad $\frac{27}{100}$; \quad $\frac{3}{100}$; \quad
$5 + \frac{6}{10}$; \quad $\frac{60}{100}$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:decimals:3}
Escreva como um \omterm{def:g4:fractions:def}{fração} (décimos ou centésimos):
$0.3$; \quad $0.51$; \quad $0.07$; \quad $1.9$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:decimals:4}
Em $6.58$: o que significa o dígito $5$ contar? O dígito $8$? Decompor
o número como em \cref{def:g4:decimals:point}.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:decimals:5}
Coloque em uma reta numérica graduada em décimos de $4$ para $5$:
$4.2$; \ $4.75$ (entre quais duas marcas?); \ $4.9$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:decimals:6}
Copie e complete com $<$, $>$ ou $=$:
\[
0.6 \;?\; 0.60, \qquad
2.8 \;?\; 2.75, \qquad
0.09 \;?\; 0.1, \qquad
5.3 \;?\; 5.13 .
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:decimals:7}
Ordene do menor para o maior: $1.05$; \ $1.5$; \ $0.95$; \ $1.45$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:decimals:8}
Escrever em vírgula decimal: três euros e cinco cêntimos; doze euros
e cinquenta centavos; noventa e nove centavos. Então peça os três preços.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:decimals:9}
Quais números decimais com um dígito decimal estão estritamente entre
$3$ e $4$? Liste todos eles. Quantos são?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g4:decimals:10}
Tom diz: ``$0.25$ é maior que $0.5$ porque $25$ é maior que
$5$''. Mostre a ele seu erro com o centésimo quadrado (quantas células
cada tonalidade faz?).
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g4:decimals:11}
Os três saltos de um corredor medidos $3.6$ m, $3.58$ eu e $3.65$ m.
Ordene os saltos do mais curto para o mais longo. Quais dois diferem por
exatamente $\frac{5}{100}$ de um metro?
\end{exercise}
