\chapter{Simetria}\label{ch:g4:symmetry}

Borboletas, \omterm{def:g2:shapes:solids}{rostos}, flocos de neve: a natureza adora figuras cujas duas \omterm{def:g3:division:half}{metades}
combinar quando dobrado. Este capítulo explora a simetria com dobras e grade
papel --- construções com o esquadro aguarde
\cref{ch:g6:symmetry}.

\section{Dobrável e combinando}

\begin{definition}[Eixo de simetria]\label{def:g4:symmetry:def}
Uma linha é um \emph{eixo de simetria}\index{eixo de simetria} de um
figura ao dobrar a figura ao longo dessa linha faz com que as duas \omterm{def:g3:division:half}{metades}
combinar \emph{exatamente}. Uma figura não pode ter nenhum eixo, um eixo ou
vários.
\end{definition}

\begin{example}[Testando com papel vegetal]\label{ex:g4:symmetry:trace}
Trace a figura, vire o papel vegetal ao longo do candidato
eixo e recolocá-lo: se o traçado cobrir exatamente a figura, o
eixo é genuíno. Somente os olhos são frequentemente enganados --- um retângulo
diagonal \emph{visual} como um eixo, mas o teste de dobra diz não (tente
isso!).
\end{example}

\section{Completando figuras em uma grade}

\begin{method}[Simétrico de uma figura através de uma linha de grade]\label{met:g4:symmetry:grid}
O eixo é uma linha vertical (ou horizontal) da grade.
\begin{enumerate}
  \item Pegue cada um vértice da figura por sua vez;
  \item conte sua distância ao eixo em quadrados;
  \item coloque a imagem vértice no \emph{mesmo} distância no
        \emph{outro} lado, na mesma linha (ou coluna);
  \item junte-se à imagem \omterm{def:g4:geometry:polygon}{vértices} na mesma ordem, e verifique: o
        a imagem é o gêmeo espelhado da figura.
\end{enumerate}
\end{method}

\begin{example}[Mesma forma, mesmo tamanho, invertido]\label{ex:g4:symmetry:properties}
A imagem tem exatamente os mesmos comprimentos e os mesmos ângulos que o
original: a simetria copia a figura --- mas a vira, como um
mão esquerda e uma mão direita. Se a letra original for F, seu espelho
a imagem é um F invertido, não outro F.
\end{example}

\section{Eixos das formas usuais}

\begin{example}[Eixos de contagem]\label{ex:g4:symmetry:shapes}
Ao dobrar:
\begin{itemize}
  \item um quadrado: $4$ eixos (dois no meio de eixos opostos
        lados, dois nas diagonais);
  \item um retângulo: $2$ eixos (apenas meios de lados opostos!);
  \item um triângulo isósceles: $1$; um triângulo equilátero: $3$;
  \item um círculo: cada linha que passa pelo seu centro --- mais eixos do que
        pode-se contar.
\end{itemize}
\end{example}

\begin{example}[Simetria no alfabeto]\label{ex:g4:symmetry:letters}
A, M, T, U, V têm eixo vertical; B, C, D, E têm uma horizontal
um; H, I, O, X têm ambos. Palavras feitas com as letras certas podem ser
simétrico também: MUM tem um eixo vertical; OXO funciona nos dois sentidos.
\end{example}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g4:symmetry:1}
Trace e dobre: quais destes têm um \omterm{def:g4:symmetry:def}{eixo de simetria} --- um
quadrado; um retângulo (não quadrado); um triângulo escaleno (todos os lados
diferente); um círculo?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:symmetry:2}
Desenhe os eixos de simetria de: um quadrado; um retângulo; um equilátero
triângulo. Quantos para cada?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:symmetry:3}
Use o teste do papel vegetal de \cref{ex:g4:symmetry:trace} mostrar
que a diagonal de um $6 \times 4$ retângulo é \emph{não} um \omterm{def:g4:symmetry:def}{eixo
de simetria}.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:symmetry:4}
Em papel quadriculado, desenhe um eixo vertical e a letra L (três quadrados
alto, dois de largura) em $2$ quadrados do eixo. Construa seu espelho
imagem.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:symmetry:5}
Em papel quadriculado, desenhe um eixo horizontal e uma pequena bandeira (uma $1 \times
1$ triângulo em um $3$-mastro quadrado) acima dele. Complete a figura por
simetria abaixo do eixo.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:symmetry:6}
Uma figura toca seu \omterm{def:g4:symmetry:def}{eixo de simetria} em um ponto $ P $. Onde está o
imagem de $ P $? Explique com a imagem dobrável.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:symmetry:7}
Classifique as letras maiúsculas A, B, E, F, H, N, O, T: eixo vertical?
eixo horizontal? ambos? nenhum?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:symmetry:8}
Um triângulo tem lados $5$ cm, $5$ centímetros e $3$ cm. Será que tem um
\omterm{def:g4:symmetry:def}{eixo de simetria}? Quais lados a dobra troca?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g4:symmetry:9}
Em papel quadriculado, desenhe uma figura sua que tenha \emph{exatamente dois}
eixos de simetria e desenhe ambos os eixos. (Dica: pense em um retângulo,
ou de uma cruz.)
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g4:symmetry:10}
\omterm{def:g3:division:half}{Metade} de um desenho simétrico é dado: a esquerda \omterm{def:g3:division:half}{metade} de uma casa
(uma parede quadrada e \omterm{def:g3:division:half}{metade} telhado triangular), com eixo vertical
ao longo de sua borda direita. Descreva ou desenhe a casa completa. Como fazer
os comprimentos do direito concluído \omterm{def:g3:division:half}{metade} comparar com a esquerda \omterm{def:g3:division:half}{metade}?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g4:symmetry:11}
Aya afirma: “todo triângulo com dois lados iguais tem um \omterm{def:g4:symmetry:def}{eixo de
simetria}, e todo triângulo com um \omterm{def:g4:symmetry:def}{eixo de simetria} tem dois iguais
lados.'' Teste sua dupla reivindicação em desenhos: um triângulo isósceles,
um equilátero, um escaleno. A dobradura suporta ela?
\end{exercise}
