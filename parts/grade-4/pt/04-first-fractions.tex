\chapter{Primeiras Frações}\label{ch:g4:fractions}

\omterm{def:g3:division:half}{Metade} uma maçã, uma \omterm{def:g3:division:half}{trimestre} de uma hora, três \omterm{def:g3:division:half}{trimestres} da aula:
frações nomeie os pedaços das coisas. Este capítulo apresenta
escrevendo $\frac{a}{b}$, lê frações em imagens e retas numéricas,
e compara os mais simples. (Frações crescer em
\cref{ch:g6:fractions}.)

\section{Nomeando as peças}

\begin{definition}[Fração]\label{def:g4:fractions:def}
Corte um todo em partes iguais e pegue alguns --- isso é um
\emph{fração}\index{fração}:
\[
\frac{3}{4}
\quad
\begin{array}{l}
3 \text{: how many parts are taken (the \emph{numerador}\index{numerador});}\\
4 \text{: how many equal parts the whole is cut into (the
\emph{denominador}\index{denominador}).}
\end{array}
\]
$\frac12$ é um \emph{\omterm{def:g3:division:half}{metade}}, $\frac13$ um \emph{terceiro}, $\frac14$ um
\emph{\omterm{def:g3:division:half}{trimestre}}, $\frac{1}{10}$ um \emph{décimo}.
\end{definition}

\begin{example}[Frações de uma coleção]\label{ex:g4:fractions:collection}
$\frac13$ de $12$ bolinhas de gude: compartilhe o $12$ bolinhas de gude em $3$ igual
grupos de $4$; um grupo é $\frac13$, então $\frac13$ de $12$ é $4$
--- e $\frac23$ de $12$ são dois grupos, $8$.
\end{example}

\section{Frações na reta numérica}

\begin{method}[Colocação $\frac{a}{b}$]\label{met:g4:fractions:line}
Corte cada unidade da linha em $ b $ passos iguais; de $0$, andar $ a $
passos. Se $ a $ é maior que $ b $, a caminhada passa $1$: frações pode
seja maior que um todo!
\end{method}

\begin{example}[Números inteiros escondidos em frações]\label{ex:g4:fractions:whole}
$\frac44 = 1$ (quatro \omterm{def:g3:division:half}{trimestres} faça o todo); $\frac82 = 4$;
$\frac{12}{3} = 4$. E $\frac74$ é $1 + \frac34$: um todo e
três \omterm{def:g3:division:half}{trimestres} mais.
\end{example}

\section{Comparando e adicionando frações simples}

\begin{proposition}[Mesmo denominador]\label{prop:g4:fractions:compare}
Com o mesmo \omterm{def:g4:fractions:def}{denominador}, as peças têm o mesmo tamanho --- então é só
conte-os:
\[
\frac{2}{8} < \frac{5}{8},
\qquad
\frac{3}{8} + \frac{4}{8} = \frac{7}{8}.
\]
Comparar com um todo também é fácil: $\frac{5}{8} < 1 < \frac{9}{8}$
(menos, então mais, do que $8$ oitavos).
\end{proposition}

\begin{proof}[Por que, em uma foto]
Sombreamento $3$ oitavas e depois $4$ mais oitavos dos mesmos tons de barra
$7$ oitavos no total.
\end{proof}

\begin{example}[Denominadores diferentes, mesmo ponto]\label{ex:g4:fractions:equal}
Na reta numérica, $\frac12$, $\frac24$ e $\frac{5}{10}$ toda a terra
no mesmo ponto: são três nomes do mesmo número. Corte
cada \omterm{def:g3:division:half}{metade} em duas marcas \omterm{def:g3:division:half}{trimestres}; em cinco décimos. (O geral
regra está em \cref{ch:g6:fractions}.)
\end{example}

\begin{example}[Quartos de hora]\label{ex:g4:fractions:hour}
Uma hora já $60$ minutos, então um \omterm{def:g3:division:half}{trimestre} de uma hora é
$60 \div 4 = 15$ minutos e três \omterm{def:g3:division:half}{trimestres} de uma hora é
$45$ minutos. ``Duas e meia'' é $\frac12$ hora depois das duas
horas.
\end{example}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g4:fractions:1}
Escreva o \omterm{def:g4:fractions:def}{fração} mostrado: uma pizza cortada em $6$ com $5$ fatias \omterm{def:g3:division:remainder}{restantes};
uma barra de chocolate de $8$ quadrados com $3$ comido (\omterm{def:g4:fractions:def}{fração} comido); um
bandeira dividida em $3$ faixas verticais iguais, $1$ colorido.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:fractions:2}
Desenhe uma barra cortada em $5$ partes iguais e sombra $\frac35$. Então sombreie
$\frac{2}{5}$ de outra barra igual. Qual sombreamento é maior?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:fractions:3}
Calcular: $\frac12$ de $18$; \quad $\frac14$ de $20$; \quad
$\frac13$ de $21$; \quad $\frac34$ de $20$ (três grupos de um
\omterm{def:g3:division:half}{trimestre}).
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:fractions:4}
Coloque em uma reta numérica cortada em terços: $\frac13$; \ $\frac33$; \
$\frac53$; \ $2$. Qual \omterm{def:g4:fractions:def}{fração} pousa exatamente em $2$?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:fractions:5}
Copie e complete com $<$, $>$ ou $=$:
\[
\frac38 \;?\; \frac68, \qquad
\frac55 \;?\; 1, \qquad
\frac74 \;?\; 1, \qquad
\frac12 \;?\; \frac24 .
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:fractions:6}
Calcular: $\frac26 + \frac36$; \quad $\frac58 - \frac28$; \quad
$\frac14 + \frac24$; \quad $1 - \frac13$ (quantos terços perfazem
$1$?).
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:fractions:7}
Escreva como um número inteiro mais um \omterm{def:g4:fractions:def}{fração} menor que $1$:
$\frac54$; \quad $\frac73$; \quad $\frac{9}{2}$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:fractions:8}
Quantos minutos são: \omterm{def:g3:division:half}{metade} uma hora; um \omterm{def:g3:division:half}{trimestre} de uma hora; três
\omterm{def:g3:division:half}{trimestres} de uma hora; $\frac13$ de uma hora?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:fractions:9}
Em uma aula de $24$ estudantes, $\frac14$ use óculos. Quantos
os alunos usam óculos? Quantos não o fazem?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g4:fractions:10}
Nina comeu $\frac38$ de uma pizza e Sam comeu $\frac48$ do mesmo
pizza. O que \omterm{def:g4:fractions:def}{fração} eles comeram juntos? O que \omterm{def:g4:fractions:def}{fração} resta?
Quem comeu mais?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g4:fractions:11}
Verdadeiro ou falso, com uma imagem ou exemplo: ``$\frac12$ de um pequeno
pizza é menor que $\frac14$ de uma pizza bem grande é possível''.
O que se deve sempre saber antes de comparar dois frações de
alguma coisa?
\end{exercise}
