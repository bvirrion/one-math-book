\chapter{Multiplicando números maiores}\label{ch:g4:mult}

Grau 3 multiplicado por um dígito (\cref{ch:g3:mult}); este capítulo
multiplica por números de dois dígitos --- e explica o misterioso
``linha deslocada'' do método coluna, que nada mais é do que uma
divisão em dezenas e unidades.

\section{Multiplicando por dezenas}

\begin{proposition}[Vezes 10, 20, 300\,\dots]\label{prop:g4:mult:tens}
Multiplicando por $10$ anexa um \omterm{def:g1:counting:zero}{zero}, por $100$ dois \omterm{def:g1:counting:zero}{zeros}. Para multiplicar
por $20$, multiplique por $2$, então por $10$:
\[
36 \times 20 = 36 \times 2 \times 10 = 72 \times 10 = 720 .
\]
Da mesma maneira $14 \times 300 = 14 \times 3 \times 100 = 4\,200$.
\end{proposition}
\admitted

\section{O método da coluna com dois dígitos}

\begin{example}[Por que a linha mudada?]\label{ex:g4:mult:why}
Para calcular $23 \times 12$, dividir $12$ em $10 + 2$:
\[
23 \times 12 = 23 \times 2 + 23 \times 10 = 46 + 230 = 276 .
\]
O método da coluna escreve exatamente esses dois \omterm{def:g2:mult:def}{produtos}, um sob o
outro --- o segundo deslocado para a esquerda porque é um número de
\emph{dezenas}:
\[
\begin{array}{r}
2\,3 \\
\times\ 1\,2 \\
\hline
4\,6 \\
2\,3\,\cdot \\
\hline
2\,7\,6
\end{array}
\]
(o ponto marca o local das unidades da linha deslocada, geralmente à esquerda
em branco).
\end{example}

\begin{method}[Multiplicação de coluna por um número de dois dígitos]\label{met:g4:mult:column}
Para calcular $457 \times 36$:
\begin{enumerate}
  \item primeira linha: $457 \times 6 = 2\,742$;
  \item segunda linha: $457 \times 3$ \emph{dezenas} $= 1\,371$ dezenas ---
        escrever $1\,371$ mudou um lugar para a esquerda;
  \item adicione as duas linhas: $2\,742 + 13\,710 = 16\,452$;
  \item estimativa para verificar: $457 \times 36 \approx 500 \times 36
        = 18\,000$? Melhorar: $450 \times 36$ é sobre $16\,000$ ---
        plausível.
\end{enumerate}
\[
\begin{array}{r}
4\,5\,7 \\
\times\ \ \,3\,6 \\
\hline
2\,7\,4\,2 \\
1\,3\,7\,1\,\cdot \\
\hline
1\,6\,4\,5\,2
\end{array}
\]
\end{method}

\begin{example}[Truques mentais]\label{ex:g4:mult:tricks}
\begin{itemize}
  \item \emph{Divida o caminho amigável}: $18 \times 11 = 18 \times 10
        + 18 = 198$.
  \item \emph{Use quase cem}: $25 \times 99 = 25 \times 100 -
        25 = 2\,475$.
  \item \emph{Dobrar e \omterm{def:g3:division:half}{metade}}: $16 \times 35 = 8 \times 70 =
        560$ (dividir um fator pela \omterm{def:g3:division:half}{metade} e dobrar o outro mantém o
        \omterm{def:g2:mult:def}{produto}).
\end{itemize}
\end{example}

\section{Multiplicação em problemas}

\begin{example}\label{ex:g4:mult:problem}
Uma escola ordena $28$ caixas de $145$ folhas de papel.
\begin{enumerate}
  \item Operação: $145 \times 28$.
  \item Estimativa: $150 \times 30 = 4\,500$, um pouco demais em ambos
        conta.
  \item Colunas: $145 \times 8 = 1\,160$; $145 \times 2$ dezenas
        $= 2\,900$; total $4\,060$.
  \item Sentença: a escola recebe $4\,060$ folhas.
\end{enumerate}
\end{example}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g4:mult:1}
Calcule mentalmente: $47 \times 10$; \quad $23 \times 100$; \quad
$36 \times 20$; \quad $15 \times 300$; \quad $42 \times 50$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:mult:2}
Calcular $34 \times 21$ dividindo como em \cref{ex:g4:mult:why}
($34 \times 21 = 34 \times 20 + 34$) e faça check-in das colunas.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:mult:3}
Calcule em colunas: $63 \times 24$; \quad $85 \times 47$; \quad
$126 \times 53$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:mult:4}
Calcule em colunas, com uma estimativa primeiro: $208 \times 34$; \quad
$517 \times 62$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:mult:5}
Calcule com um truque de \cref{ex:g4:mult:tricks}:
$45 \times 11$; \quad $32 \times 99$; \quad $24 \times 45$ (duplo
e \omterm{def:g3:division:half}{metade} --- duas vezes, se quiser).
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:mult:6}
Um cinema tem $26$ linhas de $18$ assentos. Quantos assentos ao todo?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:mult:7}
Um caminhão carrega $32$ paletes de $48$ caixas. Quantas caixas? Dê
primeiro uma estimativa e depois a resposta exata.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:mult:8}
Encontre o erro: Ana calcula $57 \times 23$ como
$57 \times 3 = 171$ e $57 \times 2 = 114$, então adiciona
$171 + 114 = 285$. A resposta correta é $1\,311$. O que ela
esqueceu?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g4:mult:9}
Um jardineiro planta $14$ linhas de $35$ tulipas e $12$ linhas de $28$
narcisos. Quantas flores ao todo? (Dois \omterm{def:g2:mult:def}{produtos}, então um \omterm{def:g1:addition:def}{soma}.)
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g4:mult:10}
Todos os dias, uma padaria usa $67$ kg de farinha. Sobre quanta farinha é
isso em um $30$-dia mês --- estimativa com números \omterm{def:g4:numbers:round}{arredondados}, então
calcular exatamente.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g4:mult:11}
Explique, com uma imagem retangular como a deste capítulo, por que
$15 \times 12$ pode ser calculado como $15 \times 10 + 15 \times 2$.
Então invente outra divisão de $15 \times 12$ isso também funciona.
\end{exercise}
