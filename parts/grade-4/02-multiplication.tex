\chapter{Multiplying Larger Numbers}\label{ch:g4:mult}

Grade 3 multiplied by one digit (\cref{ch:g3:mult}); this chapter
multiplies by two-digit numbers --- and explains the mysterious
``shifted line'' of the column method, which is nothing but a
splitting into tens and units.

\section{Multiplying by tens}

\begin{proposition}[Times 10, 20, 300\,\dots]\label{prop:g4:mult:tens}
Multiplying by $10$ appends one \omterm{def:g1:counting:zero}{zero}, by $100$ two \omterm{def:g1:counting:zero}{zeros}. To multiply
by $20$, multiply by $2$, then by $10$:
\[
36 \times 20 = 36 \times 2 \times 10 = 72 \times 10 = 720 .
\]
Likewise $14 \times 300 = 14 \times 3 \times 100 = 4\,200$.
\end{proposition}
\admitted

\section{The column method with two digits}

\begin{example}[Why the shifted line?]\label{ex:g4:mult:why}
To compute $23 \times 12$, split $12$ into $10 + 2$:
\[
23 \times 12 = 23 \times 2 + 23 \times 10 = 46 + 230 = 276 .
\]
The column method writes exactly these two \omterm{def:g2:mult:def}{products}, one under the
other --- the second shifted left because it is a number of
\emph{tens}:
\[
\begin{array}{r}
2\,3 \\
\times\ 1\,2 \\
\hline
4\,6 \\
2\,3\,\cdot \\
\hline
2\,7\,6
\end{array}
\]
(the dot marks the units place of the shifted line, often left
blank).
\end{example}

\begin{method}[Column multiplication by a two-digit number]\label{met:g4:mult:column}
To compute $457 \times 36$:
\begin{enumerate}
  \item first line: $457 \times 6 = 2\,742$;
  \item second line: $457 \times 3$ \emph{tens} $= 1\,371$ tens ---
        write $1\,371$ shifted one place left;
  \item add the two lines: $2\,742 + 13\,710 = 16\,452$;
  \item estimate to check: $457 \times 36 \approx 500 \times 36
        = 18\,000$? Better: $450 \times 36$ is about $16\,000$ ---
        plausible.
\end{enumerate}
\[
\begin{array}{r}
4\,5\,7 \\
\times\ \ \,3\,6 \\
\hline
2\,7\,4\,2 \\
1\,3\,7\,1\,\cdot \\
\hline
1\,6\,4\,5\,2
\end{array}
\]
\end{method}

\begin{example}[Mental tricks]\label{ex:g4:mult:tricks}
\begin{itemize}
  \item \emph{Split the friendly way}: $18 \times 11 = 18 \times 10
        + 18 = 198$.
  \item \emph{Use a near-hundred}: $25 \times 99 = 25 \times 100 -
        25 = 2\,475$.
  \item \emph{Double and halve}: $16 \times 35 = 8 \times 70 =
        560$ (halving one factor and doubling the other keeps the
        \omterm{def:g2:mult:def}{product}).
\end{itemize}
\end{example}

\section{Multiplication in problems}

\begin{example}\label{ex:g4:mult:problem}
A school orders $28$ boxes of $145$ sheets of paper.
\begin{enumerate}
  \item Operation: $145 \times 28$.
  \item Estimate: $150 \times 30 = 4\,500$, a bit too much on both
        counts.
  \item Columns: $145 \times 8 = 1\,160$; $145 \times 2$ tens
        $= 2\,900$; total $4\,060$.
  \item Sentence: the school receives $4\,060$ sheets.
\end{enumerate}
\end{example}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:g4:mult:1}
Compute mentally: $47 \times 10$; \quad $23 \times 100$; \quad
$36 \times 20$; \quad $15 \times 300$; \quad $42 \times 50$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:mult:2}
Compute $34 \times 21$ by splitting as in \cref{ex:g4:mult:why}
($34 \times 21 = 34 \times 20 + 34$), then check in columns.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:mult:3}
Compute in columns: $63 \times 24$; \quad $85 \times 47$; \quad
$126 \times 53$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:mult:4}
Compute in columns, with an estimate first: $208 \times 34$; \quad
$517 \times 62$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:mult:5}
Compute with a trick from \cref{ex:g4:mult:tricks}:
$45 \times 11$; \quad $32 \times 99$; \quad $24 \times 45$ (double
and halve --- twice if you like).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:mult:6}
A cinema has $26$ rows of $18$ seats. How many seats in all?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:mult:7}
A truck carries $32$ pallets of $48$ crates. How many crates? Give
an estimate first, then the exact answer.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:mult:8}
Find the mistake: Ana computes $57 \times 23$ as
$57 \times 3 = 171$ and $57 \times 2 = 114$, then adds
$171 + 114 = 285$. The correct answer is $1\,311$. What did she
forget?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g4:mult:9}
A gardener plants $14$ rows of $35$ tulips and $12$ rows of $28$
daffodils. How many flowers in all? (Two \omterm{def:g2:mult:def}{products}, then a \omterm{def:g1:addition:def}{sum}.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g4:mult:10}
Every day, a bakery uses $67$ kg of flour. About how much flour is
that in a $30$-day month --- estimate with rounded numbers, then
compute exactly.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g4:mult:11}
Explain, with a rectangle picture like the one of this chapter, why
$15 \times 12$ can be computed as $15 \times 10 + 15 \times 2$.
Then invent another splitting of $15 \times 12$ that also works.
\end{exercise}
