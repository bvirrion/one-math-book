\section*{Capítulo \ref{ch:g4:geometry} --- Lines and Polygons}

\begin{solution}{exo:g4:geometry:1}
Construcciones con \cref{met:g4:geometry:draw}; la pequeña plaza
marca el \omterm{def:g3:shapes:rightangle}{ángulo recto} entre $d$ y \omterm{def:g4:geometry:def}{perpendicular}.
\end{solution}

\begin{solution}{exo:g4:geometry:2}
E: los tres trazos horizontales son \omterm{def:g4:geometry:def}{paralelo}; cada uno es \omterm{def:g4:geometry:def}{perpendicular}
al vertical. H: los dos trazos verticales son \omterm{def:g4:geometry:def}{paralelo}, el
la barra transversal es \omterm{def:g4:geometry:def}{perpendicular} para ambos. N: los dos trazos verticales son
\omterm{def:g4:geometry:def}{paralelo}; la diagonal no es ninguna de las dos. T: los dos trazos son
\omterm{def:g4:geometry:def}{perpendicular}. Z: los trazos superior e inferior son \omterm{def:g4:geometry:def}{paralelo}; el
la diagonal no es ninguna de las dos cosas.
\end{solution}

\begin{solution}{exo:g4:geometry:3}
\omterm{def:g4:geometry:polygon}{Hexágono}; \omterm{def:g4:geometry:polygon}{cuadrilátero}; triángulo; \omterm{def:g4:geometry:polygon}{octágono}; \omterm{def:g4:geometry:polygon}{pentágono}. Un \omterm{def:g4:geometry:polygon}{polígono} tiene
tantos \omterm{def:g4:geometry:polygon}{vértices} como lados: $6$, $4$, $3$, $8$, $5$.
\end{solution}

\begin{solution}{exo:g4:geometry:4}
Un \omterm{def:g4:geometry:polygon}{pentágono} tiene lados $5$. De $A$, los no vecinos son $C$ y
$D$: dos diagonales, $[AC]$ y $[AD]$.
\end{solution}

\begin{solution}{exo:g4:geometry:5}
$OA = 4$ cm (\omterm{def:g3:shapes:shapes}{radio}); $OM = 4$ cm (cada punto del círculo);
$AB = 8$ cm (un \omterm{def:g4:geometry:circle}{diámetro} es el doble de \omterm{def:g3:shapes:shapes}{radio}).
\end{solution}

\begin{solution}{exo:g4:geometry:6}
Abra la brújula a lo largo de un lado, luego lleve esa abertura a los dos
otros lados: si el tramo de púa y lápiz cabe en un segundo lado
exactamente, esos dos lados son iguales.
\end{solution}

\begin{solution}{exo:g4:geometry:7}
El camino se cierra en un \omterm{def:g4:geometry:polygon}{polígono} con lados $5$ --- un \omterm{def:g4:geometry:polygon}{pentágono} (no un
uno normal!).
\end{solution}

\begin{solution}{exo:g4:geometry:8}
``\omterm{def:g4:geometry:def}{rectas perpendiculares} siempre cruza'': verdadero --- se cruzan en el
\omterm{def:g3:shapes:rightangle}{ángulo recto} en sí.

``\omterm{def:g4:geometry:def}{Líneas paralelas} nunca cruza'': verdadero --- esa es su definición.

``Una línea puede ser \omterm{def:g4:geometry:def}{paralelo} a una línea y \omterm{def:g4:geometry:def}{perpendicular} a
another'': true --- en papel cuadriculado, una línea horizontal es \omterm{def:g4:geometry:def}{paralelo} para
cada línea horizontal y \omterm{def:g4:geometry:def}{perpendicular} a cada línea vertical.
\end{solution}

\begin{solution}{exo:g4:geometry:9}
Las líneas $d$ y $f$ parecen \omterm{def:g4:geometry:def}{paralelo} --- y siempre son: dos
las líneas \omterm{def:g4:geometry:def}{perpendicular} a la misma línea son \omterm{def:g4:geometry:def}{paralelo}
(\cref{prop:g6:lines:facts} hace esto oficial).
\end{solution}

\begin{solution}{exo:g4:geometry:10}
Un programa correcto: ``1. Dibuja un cuadrado $ABCD$ de lado $4$ cm.
2. Marcar el punto $O$ donde se cruzan sus diagonales. 3. Dibuja el círculo
del centro $O$ y \omterm{def:g3:shapes:shapes}{radio} $2$ cm.'' El círculo toca cada lado en
su medio.
\end{solution}

\begin{solution}{exo:g4:geometry:11}
\omterm{def:g4:geometry:polygon}{Cuadrilátero}: $2$ diagonales. \omterm{def:g4:geometry:polygon}{Pentágono}: $5$. \omterm{def:g4:geometry:polygon}{Hexágono}: $9$. Cada nuevo
\omterm{def:g4:geometry:polygon}{vértice} agrega más y más diagonales --- los conteos crecen más rápido y
más rápido ($+3$, luego $+4$, \dots).
\end{solution}
