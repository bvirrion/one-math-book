\section*{Capítulo \ref{ch:g4:division} --- Division}

\begin{solution}{exo:g4:division:1}
$96 \div 4$: entre $4 \times 20 = 80$ y $4 \times 30 = 120$, entonces
el \omterm{def:g3:division:remainder}{cociente} tiene dos dígitos; la división larga da $24$, \omterm{def:g3:division:remainder}{resto}
$0$.

$87 \div 5$: entre $5 \times 10 = 50$ y $5 \times 20 = 100$;
\omterm{def:g3:division:remainder}{cociente} $17$, \omterm{def:g3:division:remainder}{resto} $2$ ($87 = 5 \times 17 + 2$).
\end{solution}

\begin{solution}{exo:g4:division:2}
$672 \div 4 = 168$, \omterm{def:g3:division:remainder}{resto} $0$ (comprobar: $4 \times 168 = 672$).

$925 \div 7 = 132$, \omterm{def:g3:division:remainder}{resto} $1$ (verificar:
$7 \times 132 + 1 = 925$).

$804 \div 6 = 134$, \omterm{def:g3:division:remainder}{resto} $0$ (comprobar: $6 \times 134 = 804$).
\end{solution}

\begin{solution}{exo:g4:division:3}
$816 \div 4 = 204$ (el paso de decenas es $1 \div 4$: \omterm{def:g1:counting:zero}{cero} veces ---
escriba el $0$).

$420 \div 7 = 60$.

$2\,512 \div 5 = 502$, \omterm{def:g3:division:remainder}{resto} $2$ (verificar:
$5 \times 502 + 2 = 2\,512$).
\end{solution}

\begin{solution}{exo:g4:division:4}
$85 = 9 \times 9 + 4$ (\omterm{def:g3:division:remainder}{cociente} $9$, \omterm{def:g3:division:remainder}{resto} $4$).

$1\,000 = 3 \times 333 + 1$.
\end{solution}

\begin{solution}{exo:g4:division:5}
$156 \div 6 = 26$: se llenan veintiséis casillas, no sobra ninguna.
\end{solution}

\begin{solution}{exo:g4:division:6}
$250 = 8 \times 31 + 2$: treinta y una piezas y $2$ cm de cinta
izquierda.
\end{solution}

\begin{solution}{exo:g4:division:7}
$375 = 52 \times 7 + 11$: siete autobuses transportan estudiantes $364$, $11$
permanecen --- el caso ``más uno'': se necesitan autobuses $8$.
\end{solution}

\begin{solution}{exo:g4:division:8}
$234 \div 3 = 78$: cada uno recibe canicas $78$. Comprobar:
$3 \times 78 = 234$.
\end{solution}

\begin{solution}{exo:g4:division:9}
Dividendo $= 7 \times 58 + 3 = 406 + 3 = 409$.
\end{solution}

\begin{solution}{exo:g4:division:10}
Estantes: $438 = 9 \times 48 + 6$: $48$ estantes completos y libros $6$
más --- Se necesitan estantes $49$. Librerías:
$49 = 6 \times 8 + 1$: ocho librerías completas y un estante extra ---
Se deben comprar librerías $9$.
\end{solution}

\begin{solution}{exo:g4:division:11}
\omterm{def:g3:division:remainder}{Cociente} $=$ \omterm{def:g3:division:remainder}{resto} $= r$ con $r < 8$, entonces $r$ es $0, 1, \dots,
7$ y el dividendo es $8r + r = 9r$: los posibles dividendos son
$0$, $9$, $18$, $27$, $36$, $45$, $54$, $63$.
\end{solution}
