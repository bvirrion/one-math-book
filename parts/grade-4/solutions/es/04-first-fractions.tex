\section*{Capítulo \ref{ch:g4:fractions} --- First Fractions}

\begin{solution}{exo:g4:fractions:1}
Pizza: Queda $\frac56$. Chocolate: $\frac38$ comido. Bandera: $\frac13$
coloreado.
\end{solution}

\begin{solution}{exo:g4:fractions:2}
$\frac35$ sombrea tres partes, $\frac25$ sólo dos: $\frac35$ es el
sombreado más grande.
\end{solution}

\begin{solution}{exo:g4:fractions:3}
\omterm{def:g3:division:half}{Medio} de $18$: $9$. \omterm{def:g3:division:half}{Cuarto} de $20$: $5$. Tercero de $21$: $7$.
$\frac34$ de $20$: tres \omterm{def:g3:division:half}{cuarteles}, $3 \times 5 = 15$.
\end{solution}

\begin{solution}{exo:g4:fractions:4}
$\frac13$: un paso; $\frac33$: en $1$; $\frac53$: dos pasos pasados
$1$. El \omterm{def:g4:fractions:def}{fracción} que aterriza exactamente en $2$ es $\frac63$.
\end{solution}

\begin{solution}{exo:g4:fractions:5}
$\frac38 < \frac68$; \quad $\frac55 = 1$; \quad $\frac74 > 1$;
\quad $\frac12 = \frac24$.
\end{solution}

\begin{solution}{exo:g4:fractions:6}
$\frac26 + \frac36 = \frac56$; \quad
$\frac58 - \frac28 = \frac38$; \quad
$\frac14 + \frac24 = \frac34$; \quad
$1 = \frac33$, so $1 - \frac13 = \frac23$.
\end{solution}

\begin{solution}{exo:g4:fractions:7}
$\frac54 = 1 + \frac14$; \quad
$\frac73 = 2 + \frac13$; \quad
$\frac92 = 4 + \frac12$.
\end{solution}

\begin{solution}{exo:g4:fractions:8}
\omterm{def:g3:division:half}{Medio} una hora: $30$ min. \omterm{def:g3:division:half}{Cuarto}: $15$ mín. Tres \omterm{def:g3:division:half}{cuarteles}: $45$
mín. Tercero: $20$ mín.
\end{solution}

\begin{solution}{exo:g4:fractions:9}
$\frac14$ de $24$ es $6$ estudiantes con gafas; $24 - 6 = 18$
sin.
\end{solution}

\begin{solution}{exo:g4:fractions:10}
Juntos: $\frac38 + \frac48 = \frac78$. Izquierda: $\frac18$. Sam comió
más ($4$ octavos contra $3$).
\end{solution}

\begin{solution}{exo:g4:fractions:11}
Posible: \omterm{def:g3:division:half}{medio} de una pizza pequeña puede ser menos comida que una
\omterm{def:g3:division:half}{cuarto} de uno gigante. \omterm{def:g4:fractions:def}{fracciones} comparar piezas \emph{del mismo
entero}; antes de comparar $\frac12$ y $\frac14$ ``de algo'',
hay que saber que los dos algo son iguales.
\end{solution}
