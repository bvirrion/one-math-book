\section*{अध्याय \ref{ch:g4:numbers} --- बड़ी संख्या}

\begin{solution}{exo:g4:numbers:1}
$3\,520\,000$; \quad $7\,000\,008$; \quad $240\,060$.
\end{solution}

\begin{solution}{exo:g4:numbers:2}
$5\,204\,300$: पांच लाख दो सौ चार हजार तीन सौ.

$10\,050\,000$: दस लाख पचास हजार.

$999\,999$: नौ सौ निन्यानवे हजार नौ सौ
निन्यानबे.
\end{solution}

\begin{solution}{exo:g4:numbers:3}
हजारों की संख्या में: $7$. सैकड़ों लाखों वर्ग: कोई नहीं है
सात अंकों की संख्या में सैकड़ों-लाखों का अंक --- लाखों
की कक्षा $8\,472\,156$ केवल अंक शामिल है $8$ (की इकाइयां
लाखों)। हजारों शामिल: $8\,472$ (तब से
$8\,472\,156 = 8\,472 \times 1\,000 + 156$).
\end{solution}

\begin{solution}{exo:g4:numbers:4}

$7\,090\,050 < 7\,090\,500$.
\end{solution}

\begin{solution}{exo:g4:numbers:6}

$64\,000$. दस हजार तक: $60\,000$.
\end{solution}

\begin{solution}{exo:g4:numbers:7}
$5\,000\,000$; \quad $6\,000\,000$ (द $499\,999$ कम पड़ जाता है
\omterm{def:g3:division:half}{आधा} एक लाख); \quad $10\,000\,000$ (बिल्कुल \omterm{def:g3:division:half}{आधा} राउंड ऊपर)।
\end{solution}

\begin{solution}{exo:g4:numbers:8}
$998\,500 \to 999\,500 \to 1\,000\,500 \to 1\,001\,500$: दूसरा
कदम दस लाख को पार कर जाता है, और बाईं ओर एक नई कक्षा रोशन हो जाती है।

$950\,000 \to 1\,050\,000 \to 1\,150\,000$: उसी चौराहे पर
पहला कदम.
\end{solution}

\begin{solution}{exo:g4:numbers:9}
दोनों सही हैं. ``इससे भी अधिक $80\,000$'' राउंड दस तक नीचे
हजार ($81\,338 \to 80\,000$); ``के बारे में $81\,000$'' राउंड को
निकटतम हजार ($81\,338 \to 81\,000$).
\end{solution}

\begin{solution}{exo:g4:numbers:10}
सबसे बड़ा: $7\,654\,321$. सबसे छोटा: $1\,234\,567$. से नजदीकी
$4\,000\,000$: ठीक नीचे सबसे अच्छा काम करता है --- $3\,765\,421$ है
$234\,579$ दूर, जबकि ठीक ऊपर, $4\,123\,567$, केवल है
$123\,567$ दूर: निकटतम है $4\,123\,567$.
\end{solution}

\begin{solution}{exo:g4:numbers:11}
\omterm{def:g4:numbers:round}{गोल}ाई निकटतम हजार देता है $50\,000$ सभी नंबरों के लिए
से $49\,500$ (आधे रास्ते राउंड तक $50\,499$. सबसे बड़ा:
$50\,499$; सबसे छोटा: $49\,500$.
\end{solution}
