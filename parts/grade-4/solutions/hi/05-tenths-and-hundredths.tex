\section*{अध्याय \ref{ch:g4:decimals} --- दसवाँ और सौवाँ}

\begin{solution}{exo:g4:decimals:1}
$5$ कॉलम: $\frac{5}{10}$ (अर्थात्\ $\frac{50}{100}$). $2$ कॉलम
और $3$ कोशिकाएँ: $\frac{23}{100}$. $45$ कोशिकाएँ: $\frac{45}{100}$.
\end{solution}

\begin{solution}{exo:g4:decimals:2}
$0.9$; \quad $0.27$; \quad $0.03$; \quad $5.6$; \quad
$0.60 = 0.6$.
\end{solution}

\begin{solution}{exo:g4:decimals:3}
$0.3 = \frac{3}{10}$; \quad $0.51 = \frac{51}{100}$; \quad
$0.07 = \frac{7}{100}$; \quad $1.9 = 1 + \frac{9}{10} =
\frac{19}{10}$.
\end{solution}

\begin{solution}{exo:g4:decimals:4}
 $5$ दसवाँ भाग गिनता है, द $8$ सौवां:
\[
6.58 = 6 + \frac{5}{10} + \frac{8}{100}.
\]
\end{solution}

\begin{solution}{exo:g4:decimals:5}
$4.2$: के बाद दूसरे निशान पर $4$. $4.75$: निशानों के बीच
$4.7$ और $4.8$, आधे रास्ते. $4.9$: एक अंक पहले $5$.
\end{solution}

\begin{solution}{exo:g4:decimals:6}
$0.6 = 0.60$; \quad $2.8 > 2.75$ (तुलना करना $2.80$); \quad
$0.09 < 0.1$ ($9$ सौवें विरुद्ध $10$); \quad
$5.3 > 5.13$.
\end{solution}

\begin{solution}{exo:g4:decimals:7}
$0.95 < 1.05 < 1.45 < 1.5$.
\end{solution}

\begin{solution}{exo:g4:decimals:8}

$0.99 < 3.05 < 12.50$.
\end{solution}

\begin{solution}{exo:g4:decimals:10}
सौ वर्ग पर, $0.25$ रंगों $25$ कोशिकाएं लेकिन $0.5 = 0.50$
रंगों $50$ कोशिकाएँ: $0.5$ बड़ा वाला है. टॉम ने तुलना की $25$ और
$5$ पूर्ण संख्याओं के रूप में; उसे तुलना करनी चाहिए $25$ और $50$ सौवां.
\end{solution}

\begin{solution}{exo:g4:decimals:11}

\end{solution}
