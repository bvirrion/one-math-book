\section*{अध्याय \ref{ch:g4:fractions} --- प्रथम अंश}

\begin{solution}{exo:g4:fractions:1}
पिज़्ज़ा: $\frac56$ बाएं। चॉकलेट: $\frac38$ खाया. झंडा: $\frac13$
रंगीन.
\end{solution}

\begin{solution}{exo:g4:fractions:2}
$\frac35$ शेड्स तीन भाग, $\frac25$ सिर्फ दो: $\frac35$ है
बड़ी छायांकन.
\end{solution}

\begin{solution}{exo:g4:fractions:3}
\omterm{def:g3:division:half}{आधा} का $18$: $9$. \omterm{def:g3:division:half}{तिमाही} का $20$: $5$. का तीसरा $21$: $7$.
$\frac34$ का $20$: तीन क्वार्टरों, $3 \times 5 = 15$.
\end{solution}

\begin{solution}{exo:g4:fractions:4}
$\frac13$: एक कदम; $\frac33$: पर $1$; $\frac53$: दो कदम आगे
$1$. \omterm{def:g4:fractions:def}{अंश} बिल्कुल उतरना $2$ है $\frac63$.
\end{solution}

\begin{solution}{exo:g4:fractions:5}
$\frac38 < \frac68$; \quad $\frac55 = 1$; \quad $\frac74 > 1$;
\quad $\frac12 = \frac24$.
\end{solution}

\begin{solution}{exo:g4:fractions:6}
$\frac26 + \frac36 = \frac56$; \quad
$\frac58 - \frac28 = \frac38$; \quad
$\frac14 + \frac24 = \frac34$; \quad
$1 = \frac33$, इसलिए $1 - \frac13 = \frac23$.
\end{solution}

\begin{solution}{exo:g4:fractions:7}
$\frac54 = 1 + \frac14$; \quad
$\frac73 = 2 + \frac13$; \quad
$\frac92 = 4 + \frac12$.
\end{solution}

\begin{solution}{exo:g4:fractions:8}
\omterm{def:g3:division:half}{आधा} एक घंटा: $30$ मि. \omterm{def:g3:division:half}{तिमाही}: $15$ मि. तीन क्वार्टरों: $45$
मि. तीसरा: $20$ मि.
\end{solution}

\begin{solution}{exo:g4:fractions:9}
$\frac14$ का $24$ है $6$ चश्मे वाले छात्र; $24 - 6 = 18$
बिना।
\end{solution}

\begin{solution}{exo:g4:fractions:10}
एक साथ: $\frac38 + \frac48 = \frac78$. बाएं: $\frac18$. सैम ने खाया
अधिक ($4$ आठवें विरुद्ध $3$).
\end{solution}

\begin{solution}{exo:g4:fractions:11}
वास्तव में संभव: \omterm{def:g3:division:half}{आधा} एक छोटे पिज्जा की तुलना में कम भोजन हो सकता है
\omterm{def:g3:division:half}{तिमाही} एक विशालकाय का. भिन्न भागों की तुलना करें \emph{उसी का
संपूर्ण}; तुलना करने से पहले $\frac12$ और $\frac14$ ``किसी चीज़ का'',
किसी को पता होना चाहिए कि दोनों चीज़ें समान हैं।
\end{solution}
