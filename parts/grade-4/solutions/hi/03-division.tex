\section*{अध्याय \ref{ch:g4:division} --- विभाजन}

\begin{solution}{exo:g4:division:1}
$96 \div 4$: बीच में $4 \times 20 = 80$ और $4 \times 30 = 120$, इतना \omterm{def:g3:division:remainder}{भागफल} दो अंक हैं; लंबा विभाजन देता है $24$, \omterm{def:g3:division:remainder}{शेष}
$0$.

$87 \div 5$: बीच में $5 \times 10 = 50$ और $5 \times 20 = 100$;
\omterm{def:g3:division:remainder}{भागफल} $17$, \omterm{def:g3:division:remainder}{शेष} $2$ ($87 = 5 \times 17 + 2$).
\end{solution}

\begin{solution}{exo:g4:division:2}
$672 \div 4 = 168$, \omterm{def:g3:division:remainder}{शेष} $0$ (जाँच करना: $4 \times 168 = 672$).

$925 \div 7 = 132$, \omterm{def:g3:division:remainder}{शेष} $1$ (जाँच करना:
$7 \times 132 + 1 = 925$).

$804 \div 6 = 134$, \omterm{def:g3:division:remainder}{शेष} $0$ (जाँच करना: $6 \times 134 = 804$).
\end{solution}

\begin{solution}{exo:g4:division:3}
$816 \div 4 = 204$ (दहाई कदम है $1 \div 4$: \omterm{def:g1:counting:zero}{शून्य} समय ---
लिखो $0$).

$420 \div 7 = 60$.

$2\,512 \div 5 = 502$, \omterm{def:g3:division:remainder}{शेष} $2$ (जाँच करना:
$5 \times 502 + 2 = 2\,512$).
\end{solution}

\begin{solution}{exo:g4:division:4}
$85 = 9 \times 9 + 4$ (\omterm{def:g3:division:remainder}{भागफल} $9$, \omterm{def:g3:division:remainder}{शेष} $4$).

$1\,000 = 3 \times 333 + 1$.
\end{solution}

\begin{solution}{exo:g4:division:5}
$156 \div 6 = 26$: छब्बीस डिब्बे भरे हुए हैं, कोई भी नहीं बचा।
\end{solution}

\begin{solution}{exo:g4:division:6}
$250 = 8 \times 31 + 2$: इकतीस टुकड़े, और $2$ रिबन का सेमी
बायां.
\end{solution}

\begin{solution}{exo:g4:division:7}
$375 = 52 \times 7 + 11$: सात बसें ले जाती हैं $364$ छात्र, $11$
बने रहें --- ``प्लस वन'' मामला: $8$ बसों की जरूरत है.
\end{solution}

\begin{solution}{exo:g4:division:8}
$234 \div 3 = 78$: प्रत्येक को मिलता है $78$ पत्थर. जाँच करना:
$3 \times 78 = 234$.
\end{solution}

\begin{solution}{exo:g4:division:9}
लाभांश $= 7 \times 58 + 3 = 406 + 3 = 409$.
\end{solution}

\begin{solution}{exo:g4:division:10}
अलमारियाँ: $438 = 9 \times 48 + 6$: $48$ पूर्ण अलमारियाँ और $6$ किताबें
और अधिक --- $49$ अलमारियों की जरूरत है. किताबों की अलमारी:
$49 = 6 \times 8 + 1$: आठ पूर्ण बुककेस और एक अतिरिक्त शेल्फ ---
$9$ किताबों की अलमारियाँ अवश्य खरीदी जानी चाहिए।
\end{solution}

\begin{solution}{exo:g4:division:11}
\omterm{def:g3:division:remainder}{भागफल} $=$ \omterm{def:g3:division:remainder}{शेष} $= r $ साथ $ r < 8$, इसलिए $ r $ है $0, 1, \dots,
7$ और लाभांश है $8r + r = 9r $: संभावित लाभांश हैं
$0$, $9$, $18$, $27$, $36$, $45$, $54$, $63$.
\end{solution}
