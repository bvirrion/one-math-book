\section*{अध्याय \ref{ch:g4:mult} --- बड़ी संख्याओं को गुणा करना}

\begin{solution}{exo:g4:mult:1}
$470$; \quad $2\,300$; \quad $720$; \quad $4\,500$; \quad
$2\,100$.
\end{solution}

\begin{solution}{exo:g4:mult:2}
$34 \times 21 = 34 \times 20 + 34 = 680 + 34 = 714$; स्तंभ
वही पंक्तियाँ दें: $34$ और $680$.
\end{solution}

\begin{solution}{exo:g4:mult:3}
$63 \times 24 = 1\,512$; \quad
$85 \times 47 = 3\,995$; \quad
$126 \times 53 = 6\,678$.
\end{solution}

\begin{solution}{exo:g4:mult:4}
$208 \times 34$: अनुमान लगाना $200 \times 34 = 6\,800$; बिल्कुल
$7\,072$.

$517 \times 62$: अनुमान लगाना $500 \times 60 = 30\,000$; बिल्कुल
$32\,054$.
\end{solution}

\begin{solution}{exo:g4:mult:5}
$45 \times 11 = 450 + 45 = 495$.

$32 \times 99 = 3\,200 - 32 = 3\,168$.

$24 \times 45 = 12 \times 90 = 1\,080$ (या एक बार और:
$6 \times 180 = 1\,080$).
\end{solution}

\begin{solution}{exo:g4:mult:6}
$26 \times 18 = 468$ सीटें.
\end{solution}

\begin{solution}{exo:g4:mult:7}
अनुमान लगाना: $30 \times 50 = 1\,500$. बिल्कुल:
$32 \times 48 = 1\,536$ टोकरे.
\end{solution}

\begin{solution}{exo:g4:mult:8}
वह भूल गयी कि $2$ का $23$ गिनता \emph{दसियों}: दूसरी पंक्ति
होना ही चाहिए $57 \times 20 = 1\,140$, नहीं $114$. सही कुल:
$171 + 1\,140 = 1\,311$.
\end{solution}

\begin{solution}{exo:g4:mult:9}
ट्यूलिप: $14 \times 35 = 490$. डैफोडील्स: $12 \times 28 = 336$.
कुल: $490 + 336 = 826$ फूल.
\end{solution}

\begin{solution}{exo:g4:mult:10}
अनुमान लगाना: $70 \times 30 = 2\,100$ किग्रा. बिल्कुल:
$67 \times 30 = 2\,010$ किग्रा.
\end{solution}

\begin{solution}{exo:g4:mult:11}
ए $15 \times 12$ इकाई वर्गों के आयत को ऊर्ध्वाधर द्वारा काटा जा सकता है
ए में पंक्तिबद्ध करें $15 \times 10$ भाग और ए $15 \times 2$ भाग: गिनती
प्रत्येक भाग देता है $150 + 30 = 180$. एक और कट भी ठीक वैसे ही काम करता है,
उदाहरण के लिए $15 = 10 + 5$: $10 \times 12 + 5 \times 12 = 120 + 60
= 180$. आयत को किसी भी प्रकार से काटने पर कुल संख्या निकल जाती है
वर्ग अपरिवर्तित.
\end{solution}
