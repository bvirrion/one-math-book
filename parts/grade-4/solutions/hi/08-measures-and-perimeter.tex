\section*{अध्याय \ref{ch:g4:measure} --- माप और परिधि}

\begin{solution}{exo:g4:measure:1}
$7$ एम $= 700$ सेमी; \quad $3.2$ किमी $= 3\,200$ एम; \quad
$85$ मिमी $= 8.5$ सेमी; \quad $4\,500$ एम $= 4.5$ किमी.
\end{solution}

\begin{solution}{exo:g4:measure:2}
$2$ किग्रा $= 2\,000$ जी; \quad $1\,800$ जी $= 1$ किग्रा $800$ जी; \quad
$3.5$ एल $= 350$ सीएल; \quad $25$ क्लोरीन $= 0.25$ एल
\end{solution}

\begin{solution}{exo:g4:measure:3}
आयत: $2 \times (9 + 5) = 28$ सेमी। वर्ग:
$4 \times 7.5 = 30$ सेमी। त्रिभुज: $6 + 8 + 10 = 24$ सेमी।
\end{solution}

\begin{solution}{exo:g4:measure:4}
लंबाई $+$ चौड़ाई $= 26 \div 2 = 13$ सेमी, तो चौड़ाई है
$13 - 8 = 5$ सेमी।
\end{solution}

\begin{solution}{exo:g4:measure:5}
कई एल-आकार काम करते हैं; एक के लिए $10$-स्क्वायर एल ए से बना है $4 \times 2$
आयत प्लस ए $2 \times 1$ पैर, सीमा पर चलना देता है $16$
इकाइयाँ। (कोई भी सही आंकड़ा और सावधानीपूर्वक गिनती सही है;)
\omterm{def:g4:measure:perimeter}{परिधि} चुने गए एल पर निर्भर करता है।)
\end{solution}

\begin{solution}{exo:g4:measure:6}
$3 \times 4$: \omterm{def:g4:measure:perimeter}{परिधि} $14$. $2 \times 6$: \omterm{def:g4:measure:perimeter}{परिधि} $16$.
($1 \times 12$: \omterm{def:g4:measure:perimeter}{परिधि} $26$.) एक वर्ग के जितना करीब, उतना
छोटा \omterm{def:g4:measure:perimeter}{परिधि} उसी के लिए \omterm{def:g4:measure:area}{क्षेत्र}.
\end{solution}

\begin{solution}{exo:g4:measure:7}
\omterm{def:g3:division:half}{आधा} की $6 \times 3$ आयत: $18 \div 2 = 9$ वर्ग।
\end{solution}

\begin{solution}{exo:g4:measure:8}
$10{:}20 \to 12{:}45$: $2$ एच $25$ मि. $7{:}35 \to 9{:}10$:
$1$ एच $35$ मि. पतली परत: $20{:}30 + 1$ एच $50 = 22{:}20$.
\end{solution}

\begin{solution}{exo:g4:measure:9}
एक गोद: $2 \times (60 + 45) = 210$ एम। तीन गोद:
$3 \times 210 = 630$ एम।
\end{solution}

\begin{solution}{exo:g4:measure:10}
\omterm{def:g4:measure:perimeter}{परिधि}: $4 \times 85 = 340$ एम; माइनस द गेट:
$340 - 4 = 336$ बाड़ का मी.
\end{solution}

\begin{solution}{exo:g4:measure:11}
दोनों दावे हैं \emph{असत्य}. वही \omterm{def:g4:measure:perimeter}{परिधि}, अलग \omterm{def:g4:measure:area}{क्षेत्र}ों: द
$4 \times 2$ आयत और अध्याय की आकृति का एल-आकार (दोनों
\omterm{def:g4:measure:perimeter}{परिधि} $12$; \omterm{def:g4:measure:area}{क्षेत्र}ों $8$ और $6$). वही \omterm{def:g4:measure:area}{क्षेत्र}, अलग
\omterm{def:g4:measure:perimeter}{परिधि}: द $4 \times 4$ वर्ग और $8 \times 2$ आयत
(दोनों \omterm{def:g4:measure:area}{क्षेत्र} $16$; \omterm{def:g4:measure:perimeter}{परिधि} $16$ और $20$).
\end{solution}
