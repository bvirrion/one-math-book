\section*{Chapter \ref{ch:g4:measure} --- Measures and Perimeter}

\begin{solution}{exo:g4:measure:1}
$7$ m $= 700$ cm; \quad $3.2$ km $= 3\,200$ m; \quad
$85$ mm $= 8.5$ cm; \quad $4\,500$ m $= 4.5$ km.
\end{solution}

\begin{solution}{exo:g4:measure:2}
$2$ kg $= 2\,000$ g; \quad $1\,800$ g $= 1$ kg $800$ g; \quad
$3.5$ L $= 350$ cL; \quad $25$ cL $= 0.25$ L.
\end{solution}

\begin{solution}{exo:g4:measure:3}
Rectangle: $2 \times (9 + 5) = 28$ cm. Square:
$4 \times 7.5 = 30$ cm. Triangle: $6 + 8 + 10 = 24$ cm.
\end{solution}

\begin{solution}{exo:g4:measure:4}
Length $+$ width $= 26 \div 2 = 13$ cm, so the width is
$13 - 8 = 5$ cm.
\end{solution}

\begin{solution}{exo:g4:measure:5}
Many L-shapes work; for a $10$-square L made of a $4 \times 2$
rectangle plus a $2 \times 1$ foot, walking the border gives $16$
units. (Any correct figure and careful count is right; the
\omterm{def:g4:measure:perimeter}{perimeter} depends on the L chosen.)
\end{solution}

\begin{solution}{exo:g4:measure:6}
$3 \times 4$: \omterm{def:g4:measure:perimeter}{perimeter} $14$. $2 \times 6$: \omterm{def:g4:measure:perimeter}{perimeter} $16$.
($1 \times 12$: \omterm{def:g4:measure:perimeter}{perimeter} $26$.) The closer to a square, the
smaller the \omterm{def:g4:measure:perimeter}{perimeter} for the same \omterm{def:g4:measure:area}{area}.
\end{solution}

\begin{solution}{exo:g4:measure:7}
\omterm{def:g3:division:half}{Half} of the $6 \times 3$ rectangle: $18 \div 2 = 9$ squares.
\end{solution}

\begin{solution}{exo:g4:measure:8}
$10{:}20 \to 12{:}45$: $2$ h $25$ min. $7{:}35 \to 9{:}10$:
$1$ h $35$ min. Film: $20{:}30 + 1$ h $50 = 22{:}20$.
\end{solution}

\begin{solution}{exo:g4:measure:9}
One lap: $2 \times (60 + 45) = 210$ m. Three laps:
$3 \times 210 = 630$ m.
\end{solution}

\begin{solution}{exo:g4:measure:10}
\omterm{def:g4:measure:perimeter}{Perimeter}: $4 \times 85 = 340$ m; minus the gate:
$340 - 4 = 336$ m of fence.
\end{solution}

\begin{solution}{exo:g4:measure:11}
Both claims are \emph{false}. Same \omterm{def:g4:measure:perimeter}{perimeter}, different \omterm{def:g4:measure:area}{areas}: the
$4 \times 2$ rectangle and the L-shape of the chapter's figure (both
\omterm{def:g4:measure:perimeter}{perimeter} $12$; \omterm{def:g4:measure:area}{areas} $8$ and $6$). Same \omterm{def:g4:measure:area}{area}, different
\omterm{def:g4:measure:perimeter}{perimeters}: the $4 \times 4$ square and the $8 \times 2$ rectangle
(both \omterm{def:g4:measure:area}{area} $16$; \omterm{def:g4:measure:perimeter}{perimeters} $16$ and $20$).
\end{solution}
