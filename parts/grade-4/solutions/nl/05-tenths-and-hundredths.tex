\section*{Hoofdstuk \ref{ch:g4:decimals} --- Tienden en honderdsten}

\begin{solution}{exo:g4:decimals:1}
$5$ kolommen: $\frac{5}{10}$ (d.w.z.\ $\frac{50}{100}$). $2$
kolommen en $3$ vakjes: $\frac{23}{100}$. $45$ vakjes:
$\frac{45}{100}$.
\end{solution}

\begin{solution}{exo:g4:decimals:2}
$0.9$; \quad $0.27$; \quad $0.03$; \quad $5.6$; \quad
$0.60 = 0.6$.
\end{solution}

\begin{solution}{exo:g4:decimals:3}
$0.3 = \frac{3}{10}$; \quad $0.51 = \frac{51}{100}$; \quad
$0.07 = \frac{7}{100}$; \quad $1.9 = 1 + \frac{9}{10} =
\frac{19}{10}$.
\end{solution}

\begin{solution}{exo:g4:decimals:4}
De $5$ telt de tienden, de $8$ de honderdsten:
\[
6.58 = 6 + \frac{5}{10} + \frac{8}{100}.
\]
\end{solution}

\begin{solution}{exo:g4:decimals:5}
$4.2$: op de tweede streep na $4$. $4.75$: tussen de strepen $4.7$
en $4.8$, halverwege. $4.9$: \'{e}\'{e}n streep v\'{o}\'{o}r $5$.
\end{solution}

\begin{solution}{exo:g4:decimals:6}
$0.6 = 0.60$; \quad $2.8 > 2.75$ (vergelijk $2.80$); \quad
$0.09 < 0.1$ ($9$ honderdsten tegen $10$); \quad
$5.3 > 5.13$.
\end{solution}

\begin{solution}{exo:g4:decimals:7}
$0.95 < 1.05 < 1.45 < 1.5$.
\end{solution}

\begin{solution}{exo:g4:decimals:8}
$3.05$; \quad $12.50$; \quad $0.99$. Volgorde:
$0.99 < 3.05 < 12.50$.
\end{solution}

\begin{solution}{exo:g4:decimals:9}
$3.1$, $3.2$, $3.3$, $3.4$, $3.5$, $3.6$, $3.7$, $3.8$, $3.9$ ---
negen getallen.
\end{solution}

\begin{solution}{exo:g4:decimals:10}
Op het honderdvierkant arceert $0.25$ $25$ vakjes maar
$0.5 = 0.50$ arceert $50$ vakjes: $0.5$ is de grotere. Tom
vergeleek $25$ en $5$ als hele getallen; hij moet $25$ en $50$
honderdsten vergelijken.
\end{solution}

\begin{solution}{exo:g4:decimals:11}
Volgorde: $3.58 < 3.6 < 3.65$. De sprongen $3.6$ en $3.65$
verschillen $0.05 = \frac{5}{100}$ meter (vijf centimeter).
\end{solution}
