\section*{Hoofdstuk \ref{ch:g4:measure} --- Maten en omtrek}

\begin{solution}{exo:g4:measure:1}
$7$ m $= 700$ cm; \quad $3.2$ km $= 3\,200$ m; \quad
$85$ mm $= 8.5$ cm; \quad $4\,500$ m $= 4.5$ km.
\end{solution}

\begin{solution}{exo:g4:measure:2}
$2$ kg $= 2\,000$ g; \quad $1\,800$ g $= 1$ kg $800$ g; \quad
$3.5$ L $= 350$ cL; \quad $25$ cL $= 0.25$ L.
\end{solution}

\begin{solution}{exo:g4:measure:3}
Rechthoek: $2 \times (9 + 5) = 28$ cm. Vierkant:
$4 \times 7.5 = 30$ cm. Driehoek: $6 + 8 + 10 = 24$ cm.
\end{solution}

\begin{solution}{exo:g4:measure:4}
Lengte $+$ breedte $= 26 \div 2 = 13$ cm, dus de breedte is
$13 - 8 = 5$ cm.
\end{solution}

\begin{solution}{exo:g4:measure:5}
Veel L-vormen werken; voor een L van $10$ vierkantjes gemaakt van
een $4 \times 2$-rechthoek plus een $2 \times 1$-voet, geeft de rand
lopen $16$ eenheden. (Elke correcte figuur en zorgvuldige telling is
goed; de \omterm{def:g4:measure:perimeter}{omtrek} hangt af van de gekozen L.)
\end{solution}

\begin{solution}{exo:g4:measure:6}
$3 \times 4$: \omterm{def:g4:measure:perimeter}{omtrek} $14$. $2 \times 6$: \omterm{def:g4:measure:perimeter}{omtrek} $16$.
($1 \times 12$: \omterm{def:g4:measure:perimeter}{omtrek} $26$.) Hoe dichter bij een vierkant, hoe
kleiner de \omterm{def:g4:measure:perimeter}{omtrek} voor dezelfde \omterm{def:g4:measure:area}{oppervlakte}.
\end{solution}

\begin{solution}{exo:g4:measure:7}
\omterm{def:g3:division:half}{Helft} van de $6 \times 3$-rechthoek: $18 \div 2 = 9$ vierkantjes.
\end{solution}

\begin{solution}{exo:g4:measure:8}
$10{:}20 \to 12{:}45$: $2$ u $25$ min. $7{:}35 \to 9{:}10$:
$1$ u $35$ min. Film: $20{:}30 + 1$ u $50 = 22{:}20$.
\end{solution}

\begin{solution}{exo:g4:measure:9}
E\'{e}n rondje: $2 \times (60 + 45) = 210$ m. Drie rondjes:
$3 \times 210 = 630$ m.
\end{solution}

\begin{solution}{exo:g4:measure:10}
\omterm{def:g4:measure:perimeter}{Omtrek}: $4 \times 85 = 340$ m; min de poort:
$340 - 4 = 336$ m hek.
\end{solution}

\begin{solution}{exo:g4:measure:11}
Beide beweringen zijn \emph{niet waar}. Zelfde \omterm{def:g4:measure:perimeter}{omtrek}, verschillende
\omterm{def:g4:measure:area}{oppervlaktes}: de $4 \times 2$-rechthoek en de L-vorm van de figuur
van dit hoofdstuk (beide \omterm{def:g4:measure:perimeter}{omtrek} $12$; \omterm{def:g4:measure:area}{oppervlaktes} $8$ en $6$).
Zelfde \omterm{def:g4:measure:area}{oppervlakte}, verschillende omtrekken: het $4 \times 4$-
vierkant en de $8 \times 2$-rechthoek (beide \omterm{def:g4:measure:area}{oppervlakte} $16$;
omtrekken $16$ en $20$).
\end{solution}
