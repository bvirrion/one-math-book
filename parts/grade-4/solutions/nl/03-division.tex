\section*{Hoofdstuk \ref{ch:g4:division} --- Delen}

\begin{solution}{exo:g4:division:1}
$96 \div 4$: tussen $4 \times 20 = 80$ en $4 \times 30 = 120$, dus
het \omterm{def:g3:division:remainder}{quoti\"ent} heeft twee cijfers; staartdeling geeft $24$, \omterm{def:g3:division:remainder}{rest}
$0$.

$87 \div 5$: tussen $5 \times 10 = 50$ en $5 \times 20 = 100$;
\omterm{def:g3:division:remainder}{quoti\"ent} $17$, \omterm{def:g3:division:remainder}{rest} $2$ ($87 = 5 \times 17 + 2$).
\end{solution}

\begin{solution}{exo:g4:division:2}
$672 \div 4 = 168$, \omterm{def:g3:division:remainder}{rest} $0$ (controle: $4 \times 168 = 672$).

$925 \div 7 = 132$, \omterm{def:g3:division:remainder}{rest} $1$ (controle:
$7 \times 132 + 1 = 925$).

$804 \div 6 = 134$, \omterm{def:g3:division:remainder}{rest} $0$ (controle: $6 \times 134 = 804$).
\end{solution}

\begin{solution}{exo:g4:division:3}
$816 \div 4 = 204$ (de tientallenstap is $1 \div 4$: \omterm{def:g1:counting:zero}{nul} keer ---
schrijf de $0$).

$420 \div 7 = 60$.

$2\,512 \div 5 = 502$, \omterm{def:g3:division:remainder}{rest} $2$ (controle:
$5 \times 502 + 2 = 2\,512$).
\end{solution}

\begin{solution}{exo:g4:division:4}
$85 = 9 \times 9 + 4$ (\omterm{def:g3:division:remainder}{quoti\"ent} $9$, \omterm{def:g3:division:remainder}{rest} $4$).

$1\,000 = 3 \times 333 + 1$.
\end{solution}

\begin{solution}{exo:g4:division:5}
$156 \div 6 = 26$: zesentwintig dozen worden gevuld, geen over.
\end{solution}

\begin{solution}{exo:g4:division:6}
$250 = 8 \times 31 + 2$: eenendertig stukken, en $2$ cm lint over.
\end{solution}

\begin{solution}{exo:g4:division:7}
$375 = 52 \times 7 + 11$: zeven bussen vervoeren $364$ leerlingen,
$11$ blijven over --- het ``plus \'{e}\'{e}n''-geval: $8$ bussen zijn
nodig.
\end{solution}

\begin{solution}{exo:g4:division:8}
$234 \div 3 = 78$: elk krijgt $78$ knikkers. Controle:
$3 \times 78 = 234$.
\end{solution}

\begin{solution}{exo:g4:division:9}
Deeltal $= 7 \times 58 + 3 = 406 + 3 = 409$.
\end{solution}

\begin{solution}{exo:g4:division:10}
Planken: $438 = 9 \times 48 + 6$: $48$ volle planken en $6$ boeken
meer --- $49$ planken zijn nodig. Kasten:
$49 = 6 \times 8 + 1$: acht volle kasten en \'{e}\'{e}n extra plank ---
$9$ kasten moeten gekocht worden.
\end{solution}

\begin{solution}{exo:g4:division:11}
\omterm{def:g3:division:remainder}{Quoti\"ent} $=$ \omterm{def:g3:division:remainder}{rest} $= r$ met $r < 8$, dus $r$ is $0, 1, \dots, 7$
en het deeltal is $8r + r = 9r$: de mogelijke deeltallen zijn
$0$, $9$, $18$, $27$, $36$, $45$, $54$, $63$.
\end{solution}
