\section*{Hoofdstuk \ref{ch:g4:numbers} --- Grote getallen}

\begin{solution}{exo:g4:numbers:1}
$3\,520\,000$; \quad $7\,000\,008$; \quad $240\,060$.
\end{solution}

\begin{solution}{exo:g4:numbers:2}
$5\,204\,300$: vijf miljoen tweehonderdvierduizend driehonderd.

$10\,050\,000$: tien miljoen vijftigduizend.

$999\,999$: negenhonderdnegenennegentigduizend
negenhonderdnegenennegentig.
\end{solution}

\begin{solution}{exo:g4:numbers:3}
Tientallen van de duizendtallen: $7$. Honderdtallen van de
miljoenenklasse: er is geen honderden-van-miljoenen-cijfer in een
zevencijferig getal --- de miljoenenklasse van $8\,472\,156$ bevat
alleen het cijfer $8$ (eenheden van miljoenen). Duizendtallen
bevat: $8\,472$ (want
$8\,472\,156 = 8\,472 \times 1\,000 + 156$).
\end{solution}

\begin{solution}{exo:g4:numbers:4}
$980\,000 < 1\,020\,000$; \quad
$3\,456\,000 < 3\,465\,000$ (duizendtallenklasse $456 < 465$); \quad
$7\,090\,050 < 7\,090\,500$.
\end{solution}

\begin{solution}{exo:g4:numbers:5}
$980\,500 < 2\,030\,000 < 2\,299\,999 < 2\,300\,000$.
\end{solution}

\begin{solution}{exo:g4:numbers:6}
Op het tiental: $63\,780$. Op het honderdtal: $63\,800$. Op het
duizendtal: $64\,000$. Op het tienduizendtal: $60\,000$.
\end{solution}

\begin{solution}{exo:g4:numbers:7}
$5\,000\,000$; \quad $6\,000\,000$ (de $499\,999$ haalt de helft
van een miljoen niet); \quad $10\,000\,000$ (precies halfway rondt
naar boven af).
\end{solution}

\begin{solution}{exo:g4:numbers:8}
$998\,500 \to 999\,500 \to 1\,000\,500 \to 1\,001\,500$: de tweede
stap steekt de miljoen over, en een nieuwe klasse gaat links aan.

$950\,000 \to 1\,050\,000 \to 1\,150\,000$: dezelfde oversteek bij
de eerste stap.
\end{solution}

\begin{solution}{exo:g4:numbers:9}
Beide hebben gelijk. ``Meer dan $80\,000$'' rondt naar beneden af
op het tienduizendtal ($81\,338 \to 80\,000$); ``ongeveer
$81\,000$'' rondt af op het dichtstbijzijnde duizendtal
($81\,338 \to 81\,000$).
\end{solution}

\begin{solution}{exo:g4:numbers:10}
Grootste: $7\,654\,321$. Kleinste: $1\,234\,567$. Dichtst bij
$4\,000\,000$: net eronder werkt minder goed --- $3\,765\,421$ is
$234\,579$ weg, terwijl net erboven, $4\,123\,567$, maar
$123\,567$ weg is: het dichtstbijzijnde is $4\,123\,567$.
\end{solution}

\begin{solution}{exo:g4:numbers:11}
Afronden op het dichtstbijzijnde duizendtal geeft $50\,000$ voor
alle getallen van $49\,500$ (halfway rondt naar boven) tot
$50\,499$. Grootste: $50\,499$; kleinste: $49\,500$.
\end{solution}
