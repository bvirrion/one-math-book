\section*{Chapitre \ref{ch:g4:division} --- La division}

\begin{solution}{exo:g4:division:1}
$96 \div 4$~: entre $4 \times 20 = 80$ et $4 \times 30 = 120$, donc
le quotient a deux chiffres~; la division posée donne $24$, reste
$0$.

$87 \div 5$~: entre $5 \times 10 = 50$ et $5 \times 20 = 100$~;
quotient $17$, reste $2$ ($87 = 5 \times 17 + 2$).
\end{solution}

\begin{solution}{exo:g4:division:2}
$672 \div 4 = 168$, reste $0$ (vérif.~: $4 \times 168 = 672$).

$925 \div 7 = 132$, reste $1$ (vérif.~:
$7 \times 132 + 1 = 925$).

$804 \div 6 = 134$, reste $0$ (vérif.~: $6 \times 134 = 804$).
\end{solution}

\begin{solution}{exo:g4:division:3}
$816 \div 4 = 204$ (l'étape des dizaines est $1 \div 4$~: zéro
fois --- écrire le $0$).

$420 \div 7 = 60$.

$2\,512 \div 5 = 502$, reste $2$ (vérif.~:
$5 \times 502 + 2 = 2\,512$).
\end{solution}

\begin{solution}{exo:g4:division:4}
$85 = 9 \times 9 + 4$ (quotient $9$, reste $4$).

$1\,000 = 3 \times 333 + 1$.
\end{solution}

\begin{solution}{exo:g4:division:5}
$156 \div 6 = 26$~: vingt-six boîtes sont remplies, rien ne reste.
\end{solution}

\begin{solution}{exo:g4:division:6}
$250 = 8 \times 31 + 2$~: trente et un morceaux, et $2$~cm de ruban
restant.
\end{solution}

\begin{solution}{exo:g4:division:7}
$375 = 52 \times 7 + 11$~: sept bus transportent $364$ élèves, $11$
restent --- le cas «~plus un~»~: $8$ bus sont nécessaires.
\end{solution}

\begin{solution}{exo:g4:division:8}
$234 \div 3 = 78$~: chacun obtient $78$ billes. Vérif.~:
$3 \times 78 = 234$.
\end{solution}

\begin{solution}{exo:g4:division:9}
Dividende $= 7 \times 58 + 3 = 406 + 3 = 409$.
\end{solution}

\begin{solution}{exo:g4:division:10}
Étagères~: $438 = 9 \times 48 + 6$~: $48$ étagères pleines et
$6$ livres de plus --- $49$ étagères sont nécessaires.
Bibliothèques~: $49 = 6 \times 8 + 1$~: huit bibliothèques pleines
et une étagère en plus --- $9$ bibliothèques doivent être achetées.
\end{solution}

\begin{solution}{exo:g4:division:11}
Quotient $=$ reste $= r$ avec $r < 8$, donc $r$ vaut
$0, 1, \dots, 7$ et le dividende est $8r + r = 9r$~: les
dividendes possibles sont $0$, $9$, $18$, $27$, $36$, $45$, $54$,
$63$.
\end{solution}
