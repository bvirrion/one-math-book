\section*{Chapitre \ref{ch:g4:fractions} --- Premières fractions}

\begin{solution}{exo:g4:fractions:1}
Pizza~: $\frac56$ restants. Chocolat~: $\frac38$ mangés. Drapeau~:
$\frac13$ coloré.
\end{solution}

\begin{solution}{exo:g4:fractions:2}
$\frac35$ colore trois parts, $\frac25$ seulement deux~:
$\frac35$ est le plus grand coloriage.
\end{solution}

\begin{solution}{exo:g4:fractions:3}
Moitié de $18$~: $9$. Quart de $20$~: $5$. Tiers de $21$~: $7$.
$\frac34$ de $20$~: trois quarts, $3 \times 5 = 15$.
\end{solution}

\begin{solution}{exo:g4:fractions:4}
$\frac13$~: un pas~; $\frac33$~: sur $1$~; $\frac53$~: deux pas
après $1$. La fraction qui tombe exactement sur $2$ est
$\frac63$.
\end{solution}

\begin{solution}{exo:g4:fractions:5}
$\frac38 < \frac68$~; \quad $\frac55 = 1$~; \quad $\frac74 > 1$~;
\quad $\frac12 = \frac24$.
\end{solution}

\begin{solution}{exo:g4:fractions:6}
$\frac26 + \frac36 = \frac56$~; \quad
$\frac58 - \frac28 = \frac38$~; \quad
$\frac14 + \frac24 = \frac34$~; \quad
$1 = \frac33$, donc $1 - \frac13 = \frac23$.
\end{solution}

\begin{solution}{exo:g4:fractions:7}
$\frac54 = 1 + \frac14$~; \quad
$\frac73 = 2 + \frac13$~; \quad
$\frac92 = 4 + \frac12$.
\end{solution}

\begin{solution}{exo:g4:fractions:8}
Demi-heure~: $30$~min. Quart~: $15$~min. Trois quarts~: $45$~min.
Tiers~: $20$~min.
\end{solution}

\begin{solution}{exo:g4:fractions:9}
$\frac14$ de $24$ fait $6$ élèves avec lunettes~; $24 - 6 = 18$
sans.
\end{solution}

\begin{solution}{exo:g4:fractions:10}
Ensemble~: $\frac38 + \frac48 = \frac78$. Reste~: $\frac18$. Sam a
mangé plus ($4$ huitièmes contre $3$).
\end{solution}

\begin{solution}{exo:g4:fractions:11}
C'est bien possible~: la moitié d'une petite pizza peut être moins
de nourriture qu'un quart d'une géante. Les fractions comparent des
parts \emph{du même tout}~; avant de comparer $\frac12$ et
$\frac14$ «~de quelque chose~», il faut savoir que les deux
«~quelque chose~» sont égaux.
\end{solution}
