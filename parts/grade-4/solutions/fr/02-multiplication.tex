\section*{Chapitre \ref{ch:g4:mult} --- Multiplier de plus grands nombres}

\begin{solution}{exo:g4:mult:1}
$470$~; \quad $2\,300$~; \quad $720$~; \quad $4\,500$~; \quad
$2\,100$.
\end{solution}

\begin{solution}{exo:g4:mult:2}
$34 \times 21 = 34 \times 20 + 34 = 680 + 34 = 714$~; la posée
donne les mêmes lignes~: $34$ et $680$.
\end{solution}

\begin{solution}{exo:g4:mult:3}
$63 \times 24 = 1\,512$~; \quad
$85 \times 47 = 3\,995$~; \quad
$126 \times 53 = 6\,678$.
\end{solution}

\begin{solution}{exo:g4:mult:4}
$208 \times 34$~: estimation $200 \times 34 = 6\,800$~; exact
$7\,072$.

$517 \times 62$~: estimation $500 \times 60 = 30\,000$~; exact
$32\,054$.
\end{solution}

\begin{solution}{exo:g4:mult:5}
$45 \times 11 = 450 + 45 = 495$.

$32 \times 99 = 3\,200 - 32 = 3\,168$.

$24 \times 45 = 12 \times 90 = 1\,080$ (ou encore une fois~:
$6 \times 180 = 1\,080$).
\end{solution}

\begin{solution}{exo:g4:mult:6}
$26 \times 18 = 468$ sièges.
\end{solution}

\begin{solution}{exo:g4:mult:7}
Estimation~: $30 \times 50 = 1\,500$. Exact~:
$32 \times 48 = 1\,536$ caisses.
\end{solution}

\begin{solution}{exo:g4:mult:8}
Elle a oublié que le $2$ de $23$ compte des \emph{dizaines}~: la
deuxième ligne doit être $57 \times 20 = 1\,140$, et non $114$.
Total correct~: $171 + 1\,140 = 1\,311$.
\end{solution}

\begin{solution}{exo:g4:mult:9}
Tulipes~: $14 \times 35 = 490$. Jonquilles~: $12 \times 28 = 336$.
Total~: $490 + 336 = 826$ fleurs.
\end{solution}

\begin{solution}{exo:g4:mult:10}
Estimation~: $70 \times 30 = 2\,100$~kg. Exact~:
$67 \times 30 = 2\,010$~kg.
\end{solution}

\begin{solution}{exo:g4:mult:11}
Un rectangle $15 \times 12$ de carreaux unitaires peut être coupé
par une ligne verticale en une partie $15 \times 10$ et une partie
$15 \times 2$~: compter chaque partie donne $150 + 30 = 180$. Une
autre découpe fonctionne aussi bien, par exemple
$15 = 10 + 5$~: $10 \times 12 + 5 \times 12 = 120 + 60
= 180$. Toute façon de découper le rectangle laisse le nombre total
de carreaux inchangé.
\end{solution}
