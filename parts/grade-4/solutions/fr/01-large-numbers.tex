\section*{Chapitre \ref{ch:g4:numbers} --- Les grands nombres}

\begin{solution}{exo:g4:numbers:1}
$3\,520\,000$~; \quad $7\,000\,008$~; \quad $240\,060$.
\end{solution}

\begin{solution}{exo:g4:numbers:2}
$5\,204\,300$~: cinq millions deux cent quatre mille trois cents.

$10\,050\,000$~: dix millions cinquante mille.

$999\,999$~: neuf cent quatre-vingt-dix-neuf mille neuf cent
quatre-vingt-dix-neuf.
\end{solution}

\begin{solution}{exo:g4:numbers:3}
Dizaines de milliers~: $7$. Centaines de la classe des millions~: il
n'y a pas de chiffre des centaines de millions dans un nombre à
sept chiffres --- la classe des millions de $8\,472\,156$ ne
contient que le chiffre $8$ (unités de millions). Milliers
contenus~: $8\,472$ (car
$8\,472\,156 = 8\,472 \times 1\,000 + 156$).
\end{solution}

\begin{solution}{exo:g4:numbers:4}
$980\,000 < 1\,020\,000$~; \quad
$3\,456\,000 < 3\,465\,000$ (classe des milliers $456 < 465$)~; \quad
$7\,090\,050 < 7\,090\,500$.
\end{solution}

\begin{solution}{exo:g4:numbers:5}
$980\,500 < 2\,030\,000 < 2\,299\,999 < 2\,300\,000$.
\end{solution}

\begin{solution}{exo:g4:numbers:6}
À la dizaine~: $63\,780$. À la centaine~: $63\,800$. Au millier~:
$64\,000$. À la dizaine de milliers~: $60\,000$.
\end{solution}

\begin{solution}{exo:g4:numbers:7}
$5\,000\,000$~; \quad $6\,000\,000$ (les $499\,999$ n'atteignent pas
la \omterm{def:g3:division:half}{moitié} d'un million)~; \quad $10\,000\,000$ (exactement à
mi-chemin, on arrondit vers le haut).
\end{solution}

\begin{solution}{exo:g4:numbers:8}
$998\,500 \to 999\,500 \to 1\,000\,500 \to 1\,001\,500$~: le second
pas franchit le million, et une nouvelle classe s'allume à gauche.

$950\,000 \to 1\,050\,000 \to 1\,150\,000$~: même franchissement dès
le premier pas.
\end{solution}

\begin{solution}{exo:g4:numbers:9}
Les deux ont raison. «~Plus de $80\,000$~» arrondit vers le bas à la
dizaine de milliers ($81\,338 \to 80\,000$)~; «~environ $81\,000$~»
arrondit au millier le plus proche ($81\,338 \to 81\,000$).
\end{solution}

\begin{solution}{exo:g4:numbers:10}
Le plus grand~: $7\,654\,321$. Le plus petit~: $1\,234\,567$. Le
plus proche de $4\,000\,000$~: juste en dessous marche moins bien ---
$3\,765\,421$ est à $234\,579$, tandis que juste au-dessus,
$4\,123\,567$, n'est qu'à $123\,567$~: le plus proche est
$4\,123\,567$.
\end{solution}

\begin{solution}{exo:g4:numbers:11}
L'\omterm{def:g4:numbers:round}{arrondi} au millier le plus proche donne $50\,000$ pour tous les
nombres de $49\,500$ (mi-chemin \omterm{def:g4:numbers:round}{arrondi} vers le haut) à $50\,499$.
Le plus grand~: $50\,499$~; le plus petit~: $49\,500$.
\end{solution}
