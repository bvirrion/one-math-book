\section*{Chapitre \ref{ch:g4:decimals} --- Dixièmes et centièmes}

\begin{solution}{exo:g4:decimals:1}
$5$ colonnes~: $\frac{5}{10}$ (c'est-à-dire $\frac{50}{100}$).
$2$ colonnes et $3$ cases~: $\frac{23}{100}$. $45$ cases~:
$\frac{45}{100}$.
\end{solution}

\begin{solution}{exo:g4:decimals:2}
$0.9$~; \quad $0.27$~; \quad $0.03$~; \quad $5.6$~; \quad
$0.60 = 0.6$.
\end{solution}

\begin{solution}{exo:g4:decimals:3}
$0.3 = \frac{3}{10}$~; \quad $0.51 = \frac{51}{100}$~; \quad
$0.07 = \frac{7}{100}$~; \quad $1.9 = 1 + \frac{9}{10} =
\frac{19}{10}$.
\end{solution}

\begin{solution}{exo:g4:decimals:4}
Le $5$ compte les dixièmes, le $8$ les centièmes~:
\[
6.58 = 6 + \frac{5}{10} + \frac{8}{100}.
\]
\end{solution}

\begin{solution}{exo:g4:decimals:5}
$4.2$~: sur la deuxième marque après $4$. $4.75$~: entre les
marques $4.7$ et $4.8$, à mi-chemin. $4.9$~: une marque avant $5$.
\end{solution}

\begin{solution}{exo:g4:decimals:6}
$0.6 = 0.60$~; \quad $2.8 > 2.75$ (comparer $2.80$)~; \quad
$0.09 < 0.1$ ($9$ centièmes contre $10$)~; \quad
$5.3 > 5.13$.
\end{solution}

\begin{solution}{exo:g4:decimals:7}
$0.95 < 1.05 < 1.45 < 1.5$.
\end{solution}

\begin{solution}{exo:g4:decimals:8}
$3.05$~; \quad $12.50$~; \quad $0.99$. Ordre~:
$0.99 < 3.05 < 12.50$.
\end{solution}

\begin{solution}{exo:g4:decimals:9}
$3.1$, $3.2$, $3.3$, $3.4$, $3.5$, $3.6$, $3.7$, $3.8$, $3.9$ ---
neuf nombres.
\end{solution}

\begin{solution}{exo:g4:decimals:10}
Sur le carré de cent, $0.25$ colore $25$ cases mais $0.5 = 0.50$
colore $50$ cases~: $0.5$ est le plus grand. Tom a comparé $25$ et
$5$ comme des entiers~; il devait comparer $25$ et $50$ centièmes.
\end{solution}

\begin{solution}{exo:g4:decimals:11}
Ordre~: $3.58 < 3.6 < 3.65$. Les sauts $3.6$ et $3.65$ diffèrent de
$0.05 = \frac{5}{100}$ de mètre (cinq centimètres).
\end{solution}
