\section*{Chapitre \ref{ch:g4:geometry} --- Droites et polygones}

\begin{solution}{exo:g4:geometry:1}
Constructions avec \cref{met:g4:geometry:draw}~; le petit carré
marque l'angle droit entre $d$ et la perpendiculaire.
\end{solution}

\begin{solution}{exo:g4:geometry:2}
E~: les trois traits horizontaux sont parallèles~; chacun est
perpendiculaire au vertical. H~: les deux traits verticaux sont
parallèles, la barre transversale est perpendiculaire aux deux.
N~: les deux traits verticaux sont parallèles~; la diagonale
n'est ni parallèle ni perpendiculaire. T~: les deux traits sont
perpendiculaires. Z~: les traits du haut et du bas sont
parallèles~; la diagonale n'est ni l'un ni l'autre.
\end{solution}

\begin{solution}{exo:g4:geometry:3}
Hexagone~; quadrilatère~; triangle~; octogone~; pentagone. Un
polygone a autant de sommets que de côtés~: $6$, $4$, $3$, $8$,
$5$.
\end{solution}

\begin{solution}{exo:g4:geometry:4}
Un pentagone a $5$ côtés. Depuis $A$, les non-voisins sont $C$ et
$D$~: deux diagonales, $[AC]$ et $[AD]$.
\end{solution}

\begin{solution}{exo:g4:geometry:5}
$OA = 4$~cm (rayon)~; $OM = 4$~cm (tout point du cercle)~;
$AB = 8$~cm (un diamètre vaut deux fois le rayon).
\end{solution}

\begin{solution}{exo:g4:geometry:6}
Ouvrir le compas le long d'un côté, puis reporter cette ouverture
sur les deux autres côtés~: si l'écart pointe-crayon s'ajuste
exactement à un second côté, ces deux côtés sont égaux.
\end{solution}

\begin{solution}{exo:g4:geometry:7}
Le chemin se ferme en un polygone à $5$ côtés --- un pentagone
(pas régulier~!).
\end{solution}

\begin{solution}{exo:g4:geometry:8}
«~Les droites perpendiculaires se croisent toujours~»~: vrai ---
elles se croisent à l'angle droit lui-même.

«~Les droites parallèles ne se croisent jamais~»~: vrai --- c'est
leur définition.

«~Une droite peut être parallèle à une droite et perpendiculaire à
une autre~»~: vrai --- sur papier quadrillé, une droite horizontale
est parallèle à toute droite horizontale et perpendiculaire à
toute droite verticale.
\end{solution}

\begin{solution}{exo:g4:geometry:9}
Les droites $d$ et $f$ semblent parallèles --- et elles le sont
toujours~: deux droites perpendiculaires à une même droite sont
parallèles (\cref{prop:g6:lines:facts} le rend officiel).
\end{solution}

\begin{solution}{exo:g4:geometry:10}
Un programme correct~: «~1. Tracer un carré $ABCD$ de côté $4$~cm.
2. Marquer le point $O$ où ses diagonales se croisent. 3. Tracer le
cercle de centre $O$ et de rayon $2$~cm.~» Le cercle touche chaque
côté en son milieu.
\end{solution}

\begin{solution}{exo:g4:geometry:11}
Quadrilatère~: $2$ diagonales. Pentagone~: $5$. Hexagone~: $9$.
Chaque nouveau sommet ajoute de plus en plus de diagonales --- les
comptes croissent de plus en plus vite ($+3$, puis $+4$, \dots).
\end{solution}
