\section*{Chapter \ref{ch:g4:division} --- Division}

\begin{solution}{exo:g4:division:1}
$96 \div 4$: between $4 \times 20 = 80$ and $4 \times 30 = 120$, so
the \omterm{def:g3:division:remainder}{quotient} has two digits; long division gives $24$, \omterm{def:g3:division:remainder}{remainder}
$0$.

$87 \div 5$: between $5 \times 10 = 50$ and $5 \times 20 = 100$;
\omterm{def:g3:division:remainder}{quotient} $17$, \omterm{def:g3:division:remainder}{remainder} $2$ ($87 = 5 \times 17 + 2$).
\end{solution}

\begin{solution}{exo:g4:division:2}
$672 \div 4 = 168$, \omterm{def:g3:division:remainder}{remainder} $0$ (check: $4 \times 168 = 672$).

$925 \div 7 = 132$, \omterm{def:g3:division:remainder}{remainder} $1$ (check:
$7 \times 132 + 1 = 925$).

$804 \div 6 = 134$, \omterm{def:g3:division:remainder}{remainder} $0$ (check: $6 \times 134 = 804$).
\end{solution}

\begin{solution}{exo:g4:division:3}
$816 \div 4 = 204$ (the tens step is $1 \div 4$: \omterm{def:g1:counting:zero}{zero} times ---
write the $0$).

$420 \div 7 = 60$.

$2\,512 \div 5 = 502$, \omterm{def:g3:division:remainder}{remainder} $2$ (check:
$5 \times 502 + 2 = 2\,512$).
\end{solution}

\begin{solution}{exo:g4:division:4}
$85 = 9 \times 9 + 4$ (\omterm{def:g3:division:remainder}{quotient} $9$, \omterm{def:g3:division:remainder}{remainder} $4$).

$1\,000 = 3 \times 333 + 1$.
\end{solution}

\begin{solution}{exo:g4:division:5}
$156 \div 6 = 26$: twenty-six boxes are filled, none left over.
\end{solution}

\begin{solution}{exo:g4:division:6}
$250 = 8 \times 31 + 2$: thirty-one pieces, and $2$ cm of ribbon
left.
\end{solution}

\begin{solution}{exo:g4:division:7}
$375 = 52 \times 7 + 11$: seven buses carry $364$ students, $11$
remain --- the ``plus one'' case: $8$ buses are needed.
\end{solution}

\begin{solution}{exo:g4:division:8}
$234 \div 3 = 78$: each gets $78$ marbles. Check:
$3 \times 78 = 234$.
\end{solution}

\begin{solution}{exo:g4:division:9}
Dividend $= 7 \times 58 + 3 = 406 + 3 = 409$.
\end{solution}

\begin{solution}{exo:g4:division:10}
Shelves: $438 = 9 \times 48 + 6$: $48$ full shelves and $6$ books
more --- $49$ shelves are needed. Bookcases:
$49 = 6 \times 8 + 1$: eight full bookcases and one extra shelf ---
$9$ bookcases must be bought.
\end{solution}

\begin{solution}{exo:g4:division:11}
\omterm{def:g3:division:remainder}{Quotient} $=$ \omterm{def:g3:division:remainder}{remainder} $= r$ with $r < 8$, so $r$ is $0, 1, \dots,
7$ and the dividend is $8r + r = 9r$: the possible dividends are
$0$, $9$, $18$, $27$, $36$, $45$, $54$, $63$.
\end{solution}
