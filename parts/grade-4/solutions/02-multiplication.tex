\section*{Chapter \ref{ch:g4:mult} --- Multiplying Larger Numbers}

\begin{solution}{exo:g4:mult:1}
$470$; \quad $2\,300$; \quad $720$; \quad $4\,500$; \quad
$2\,100$.
\end{solution}

\begin{solution}{exo:g4:mult:2}
$34 \times 21 = 34 \times 20 + 34 = 680 + 34 = 714$; the columns
give the same lines: $34$ and $680$.
\end{solution}

\begin{solution}{exo:g4:mult:3}
$63 \times 24 = 1\,512$; \quad
$85 \times 47 = 3\,995$; \quad
$126 \times 53 = 6\,678$.
\end{solution}

\begin{solution}{exo:g4:mult:4}
$208 \times 34$: estimate $200 \times 34 = 6\,800$; exactly
$7\,072$.

$517 \times 62$: estimate $500 \times 60 = 30\,000$; exactly
$32\,054$.
\end{solution}

\begin{solution}{exo:g4:mult:5}
$45 \times 11 = 450 + 45 = 495$.

$32 \times 99 = 3\,200 - 32 = 3\,168$.

$24 \times 45 = 12 \times 90 = 1\,080$ (or once more:
$6 \times 180 = 1\,080$).
\end{solution}

\begin{solution}{exo:g4:mult:6}
$26 \times 18 = 468$ seats.
\end{solution}

\begin{solution}{exo:g4:mult:7}
Estimate: $30 \times 50 = 1\,500$. Exactly:
$32 \times 48 = 1\,536$ crates.
\end{solution}

\begin{solution}{exo:g4:mult:8}
She forgot that the $2$ of $23$ counts \emph{tens}: the second line
must be $57 \times 20 = 1\,140$, not $114$. Correct total:
$171 + 1\,140 = 1\,311$.
\end{solution}

\begin{solution}{exo:g4:mult:9}
Tulips: $14 \times 35 = 490$. Daffodils: $12 \times 28 = 336$.
Total: $490 + 336 = 826$ flowers.
\end{solution}

\begin{solution}{exo:g4:mult:10}
Estimate: $70 \times 30 = 2\,100$ kg. Exactly:
$67 \times 30 = 2\,010$ kg.
\end{solution}

\begin{solution}{exo:g4:mult:11}
A $15 \times 12$ rectangle of unit squares can be cut by a vertical
line into a $15 \times 10$ part and a $15 \times 2$ part: counting
each part gives $150 + 30 = 180$. Another cut works just as well,
for instance $15 = 10 + 5$: $10 \times 12 + 5 \times 12 = 120 + 60
= 180$. Any way of cutting the rectangle leaves the total number of
squares unchanged.
\end{solution}
