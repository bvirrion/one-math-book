\section*{Capítulo \ref{ch:g4:geometry} --- Linhas e Polígonos}

\begin{solution}{exo:g4:geometry:1}
Construções com \cref{met:g4:geometry:draw}; a pequena praça
marca o \omterm{def:g3:shapes:rightangle}{ângulo reto} entre $d $ e o \omterm{def:g4:geometry:def}{perpendicular}.
\end{solution}

\begin{solution}{exo:g4:geometry:2}
E: os três traços horizontais são \omterm{def:g4:geometry:def}{paralelo}; cada um é \omterm{def:g4:geometry:def}{perpendicular}
para o vertical. H: os dois traços verticais são \omterm{def:g4:geometry:def}{paralelo}, o
barra transversal é \omterm{def:g4:geometry:def}{perpendicular} para ambos. N: os dois traços verticais são
\omterm{def:g4:geometry:def}{paralelo}; a diagonal não é nenhuma das duas. T: os dois golpes são
\omterm{def:g4:geometry:def}{perpendicular}. Z: os traços superior e inferior são \omterm{def:g4:geometry:def}{paralelo}; o
diagonal não é nenhum dos dois.
\end{solution}

\begin{solution}{exo:g4:geometry:3}
\omterm{def:g4:geometry:polygon}{Hexágono}; \omterm{def:g4:geometry:polygon}{quadrilátero}; triângulo; \omterm{def:g4:geometry:polygon}{octógono}; \omterm{def:g4:geometry:polygon}{Pentágono}. UM \omterm{def:g4:geometry:polygon}{polígono} tem
tantos \omterm{def:g4:geometry:polygon}{vértices} como lados: $6$, $4$, $3$, $8$, $5$.
\end{solution}

\begin{solution}{exo:g4:geometry:4}
UM \omterm{def:g4:geometry:polygon}{Pentágono} tem $5$ lados. De $ A $, os não-vizinhos são $ C $ e
$ D $: duas diagonais, $[AC]$ e $[AD]$.
\end{solution}

\begin{solution}{exo:g4:geometry:5}
$ OA = 4$ cm (\omterm{def:g3:shapes:shapes}{raio}); $ OM = 4$ cm (cada ponto do círculo);
$ AB = 8$ centímetros (um \omterm{def:g4:geometry:circle}{diâmetro} é o dobro \omterm{def:g3:shapes:shapes}{raio}).
\end{solution}

\begin{solution}{exo:g4:geometry:6}
Abra a bússola de um lado e depois leve essa abertura para os dois
outros lados: se a extensão do prego e do lápis se encaixar em um segundo lado
exatamente, esses dois lados são iguais.
\end{solution}

\begin{solution}{exo:g4:geometry:7}
O caminho se fecha em um \omterm{def:g4:geometry:polygon}{polígono} com $5$ lados --- um \omterm{def:g4:geometry:polygon}{Pentágono} (não um
normal!).
\end{solution}

\begin{solution}{exo:g4:geometry:8}
``Linhas \omterm{def:g4:geometry:def}{perpendiculares} sempre cruzam'': verdadeiro --- eles cruzam no
\omterm{def:g3:shapes:rightangle}{ângulo reto} em si.

``Linhas paralelas nunca cruze '': verdadeiro --- essa é a definição deles.

``Uma linha pode ser \omterm{def:g4:geometry:def}{paralelo} para uma linha e \omterm{def:g4:geometry:def}{perpendicular} para
outro'': verdadeiro --- em papel quadriculado, uma linha horizontal é \omterm{def:g4:geometry:def}{paralelo} para
cada linha horizontal e \omterm{def:g4:geometry:def}{perpendicular} para cada vertical.
\end{solution}

\begin{solution}{exo:g4:geometry:9}
As linhas $ d $ e $ f $ olhar \omterm{def:g4:geometry:def}{paralelo} --- e sempre são: dois
linhas \omterm{def:g4:geometry:def}{perpendicular} para a mesma linha são \omterm{def:g4:geometry:def}{paralelo}
(\cref{prop:g6:lines:facts} torna isso oficial).
\end{solution}

\begin{solution}{exo:g4:geometry:10}
Um programa correto: ``1. Desenhe um quadrado $ ABCD $ de lado $4$ cm.
2. Marque o ponto $ O $ onde suas diagonais se cruzam. 3. Desenhe o círculo
do centro $ O $ e \omterm{def:g3:shapes:shapes}{raio} $2$ cm.'' O círculo toca cada lado em
seu meio.
\end{solution}

\begin{solution}{exo:g4:geometry:11}
\omterm{def:g4:geometry:polygon}{Quadrilátero}: $2$ diagonais. \omterm{def:g4:geometry:polygon}{Pentágono}: $5$. \omterm{def:g4:geometry:polygon}{Hexágono}: $9$. Cada novo
vértice adiciona mais e mais diagonais --- as contagens crescem mais rápido e
mais rápido ($+3$, então $+4$, \dots).
\end{solution}
