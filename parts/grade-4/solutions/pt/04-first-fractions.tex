\section*{Capítulo \ref{ch:g4:fractions} --- Primeiras Frações}

\begin{solution}{exo:g4:fractions:1}
Pizza: $\frac56$ esquerda. Chocolate: $\frac38$ comido. Bandeira: $\frac13$
colorido.
\end{solution}

\begin{solution}{exo:g4:fractions:2}
$\frac35$ sombreia três partes, $\frac25$ apenas dois: $\frac35$ é o
sombreamento maior.
\end{solution}

\begin{solution}{exo:g4:fractions:3}
\omterm{def:g3:division:half}{Metade} de $18$: $9$. \omterm{def:g3:division:half}{Trimestre} de $20$: $5$. Terceiro de $21$: $7$.
$\frac34$ de $20$: três \omterm{def:g3:division:half}{trimestres}, $3 \times 5 = 15$.
\end{solution}

\begin{solution}{exo:g4:fractions:4}
$\frac13$: um passo; $\frac33$: sobre $1$; $\frac53$: dois passos depois
$1$. O \omterm{def:g4:fractions:def}{fração} pousando exatamente em $2$ é $\frac63$.
\end{solution}

\begin{solution}{exo:g4:fractions:5}
$\frac38 < \frac68$; \quad $\frac55 = 1$; \quad $\frac74 > 1$;
\quad $\frac12 = \frac24$.
\end{solution}

\begin{solution}{exo:g4:fractions:6}
$\frac26 + \frac36 = \frac56$; \quad
$\frac58 - \frac28 = \frac38$; \quad
$\frac14 + \frac24 = \frac34$; \quad
$1 = \frac33$, então $1 - \frac13 = \frac23$.
\end{solution}

\begin{solution}{exo:g4:fractions:7}
$\frac54 = 1 + \frac14$; \quad
$\frac73 = 2 + \frac13$; \quad
$\frac92 = 4 + \frac12$.
\end{solution}

\begin{solution}{exo:g4:fractions:8}
\omterm{def:g3:division:half}{Metade} uma hora: $30$ min. \omterm{def:g3:division:half}{Trimestre}: $15$ min. Três \omterm{def:g3:division:half}{trimestres}: $45$
min. Terceiro: $20$ min.
\end{solution}

\begin{solution}{exo:g4:fractions:9}
$\frac14$ de $24$ é $6$ estudantes com óculos; $24 - 6 = 18$
sem.
\end{solution}

\begin{solution}{exo:g4:fractions:10}
Junto: $\frac38 + \frac48 = \frac78$. Esquerda: $\frac18$. Sam comeu
mais ($4$ oitavos contra $3$).
\end{solution}

\begin{solution}{exo:g4:fractions:11}
De fato possível: \omterm{def:g3:division:half}{metade} de uma pizza pequena pode ser menos comida do que uma
\omterm{def:g3:division:half}{trimestre} de um gigante. Frações comparar peças \emph{do mesmo
inteiro}; antes de comparar $\frac12$ e $\frac14$ ``de alguma coisa'',
é preciso saber que as duas coisas são iguais.
\end{solution}
