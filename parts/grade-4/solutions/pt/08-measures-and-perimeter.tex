\section*{Capítulo \ref{ch:g4:measure} --- Medidas e Perímetro}

\begin{solution}{exo:g4:measure:1}
$7$ eu $= 700$ cm; \quad $3.2$ quilômetros $= 3\,200$ m; \quad
$85$ milímetros $= 8.5$ cm; \quad $4\,500$ eu $= 4.5$ km.
\end{solution}

\begin{solution}{exo:g4:measure:2}
$2$ kg $= 2\,000$ g; \quad $1\,800$ g $= 1$ kg $800$ g; \quad
$3.5$ eu $= 350$ cl; \quad $25$ cL $= 0.25$ eu.
\end{solution}

\begin{solution}{exo:g4:measure:3}
Retângulo: $2 \times (9 + 5) = 28$ cm. Quadrado:
$4 \times 7.5 = 30$ cm. Triângulo: $6 + 8 + 10 = 24$ cm.
\end{solution}

\begin{solution}{exo:g4:measure:4}
Comprimento $+$ largura $= 26 \div 2 = 13$ cm, então a largura é
$13 - 8 = 5$ cm.
\end{solution}

\begin{solution}{exo:g4:measure:5}
Muitos formatos de L funcionam; por um $10$-quadrado L feito de um $4 \times 2$
retângulo mais um $2 \times 1$ pé, andando pela fronteira dá $16$
unidades. (Qualquer número correto e contagem cuidadosa estão corretos; o
\omterm{def:g4:measure:perimeter}{perímetro} depende do L escolhido.)
\end{solution}

\begin{solution}{exo:g4:measure:6}
$3 \times 4$: \omterm{def:g4:measure:perimeter}{perímetro} $14$. $2 \times 6$: \omterm{def:g4:measure:perimeter}{perímetro} $16$.
($1 \times 12$: \omterm{def:g4:measure:perimeter}{perímetro} $26$.) Quanto mais próximo de um quadrado, mais
menor o \omterm{def:g4:measure:perimeter}{perímetro} para o mesmo \omterm{def:g4:measure:area}{área}.
\end{solution}

\begin{solution}{exo:g4:measure:7}
\omterm{def:g3:division:half}{Metade} do $6 \times 3$ retângulo: $18 \div 2 = 9$ quadrados.
\end{solution}

\begin{solution}{exo:g4:measure:8}
$10{:}20 \to 12{:}45$: $2$ h $25$ min. $7{:}35 \to 9{:}10$:
$1$ h $35$ min. Filme: $20{:}30 + 1$ h $50 = 22{:}20$.
\end{solution}

\begin{solution}{exo:g4:measure:9}
Uma volta: $2 \times (60 + 45) = 210$ m. Três voltas:
$3 \times 210 = 630$ m.
\end{solution}

\begin{solution}{exo:g4:measure:10}
\omterm{def:g4:measure:perimeter}{Perímetro}: $4 \times 85 = 340$ m; menos o portão:
$340 - 4 = 336$ m de cerca.
\end{solution}

\begin{solution}{exo:g4:measure:11}
Ambas as reivindicações são \emph{falso}. Mesmo \omterm{def:g4:measure:perimeter}{perímetro}, diferente \omterm{def:g4:measure:area}{áreas}: o
$4 \times 2$ retângulo e a forma de L da figura do capítulo (ambos
\omterm{def:g4:measure:perimeter}{perímetro} $12$; \omterm{def:g4:measure:area}{áreas} $8$ e $6$). Mesmo \omterm{def:g4:measure:area}{área}, diferente
\omterm{def:g4:measure:perimeter}{perímetros}: o $4 \times 4$ quadrado e o $8 \times 2$ retângulo
(ambos \omterm{def:g4:measure:area}{área} $16$; \omterm{def:g4:measure:perimeter}{perímetros} $16$ e $20$).
\end{solution}
