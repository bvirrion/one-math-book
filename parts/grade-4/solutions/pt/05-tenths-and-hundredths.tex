\section*{Capítulo \ref{ch:g4:decimals} --- Décimos e Centésimos}

\begin{solution}{exo:g4:decimals:1}
$5$ colunas: $\frac{5}{10}$ (ou seja,\ $\frac{50}{100}$). $2$ colunas
e $3$ células: $\frac{23}{100}$. $45$ células: $\frac{45}{100}$.
\end{solution}

\begin{solution}{exo:g4:decimals:2}
$0.9$; \quad $0.27$; \quad $0.03$; \quad $5.6$; \quad
$0.60 = 0.6$.
\end{solution}

\begin{solution}{exo:g4:decimals:3}
$0.3 = \frac{3}{10}$; \quad $0.51 = \frac{51}{100}$; \quad
$0.07 = \frac{7}{100}$; \quad $1.9 = 1 + \frac{9}{10} =
\frac{19}{10}$.
\end{solution}

\begin{solution}{exo:g4:decimals:4}
O $5$ conta os décimos, o $8$ os centésimos:
\[
6.58 = 6 + \frac{5}{10} + \frac{8}{100}.
\]
\end{solution}

\begin{solution}{exo:g4:decimals:5}
$4.2$: na segunda marca depois $4$. $4.75$: entre as marcas
$4.7$ e $4.8$, no meio do caminho. $4.9$: uma marca antes $5$.
\end{solution}

\begin{solution}{exo:g4:decimals:6}
$0.6 = 0.60$; \quad $2.8 > 2.75$ (comparar $2.80$); \quad
$0.09 < 0.1$ ($9$ centésimos contra $10$); \quad
$5.3 > 5.13$.
\end{solution}

\begin{solution}{exo:g4:decimals:7}
$0.95 < 1.05 < 1.45 < 1.5$.
\end{solution}

\begin{solution}{exo:g4:decimals:8}
$3.05$; \quad $12.50$; \quad $0.99$. Ordem:
$0.99 < 3.05 < 12.50$.
\end{solution}

\begin{solution}{exo:g4:decimals:9}
$3.1$, $3.2$, $3.3$, $3.4$, $3.5$, $3.6$, $3.7$, $3.8$, $3.9$ ---
nove números.
\end{solution}

\begin{solution}{exo:g4:decimals:10}
Na centena quadrada, $0.25$ tons $25$ células, mas $0.5 = 0.50$
tons $50$ células: $0.5$ é o maior. Tom comparou $25$ e
$5$ como números inteiros; ele deveria comparar $25$ e $50$ centésimos.
\end{solution}

\begin{solution}{exo:g4:decimals:11}
Ordem: $3.58 < 3.6 < 3.65$. Os saltos $3.6$ e $3.65$ diferir por
$0.05 = \frac{5}{100}$ de um metro (cinco centímetros).
\end{solution}
