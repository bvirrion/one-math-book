\section*{Capítulo \ref{ch:g4:division} --- Divisão}

\begin{solution}{exo:g4:division:1}
$96 \div 4$: entre $4 \times 20 = 80$ e $4 \times 30 = 120$, então
o \omterm{def:g3:division:remainder}{quociente} tem dois dígitos; divisão longa dá $24$, \omterm{def:g3:division:remainder}{restante}
$0$.

$87 \div 5$: entre $5 \times 10 = 50$ e $5 \times 20 = 100$;
\omterm{def:g3:division:remainder}{quociente} $17$, \omterm{def:g3:division:remainder}{restante} $2$ ($87 = 5 \times 17 + 2$).
\end{solution}

\begin{solution}{exo:g4:division:2}
$672 \div 4 = 168$, \omterm{def:g3:division:remainder}{restante} $0$ (verificar: $4 \times 168 = 672$).

$925 \div 7 = 132$, \omterm{def:g3:division:remainder}{restante} $1$ (verificar:
$7 \times 132 + 1 = 925$).

$804 \div 6 = 134$, \omterm{def:g3:division:remainder}{restante} $0$ (verificar: $6 \times 134 = 804$).
\end{solution}

\begin{solution}{exo:g4:division:3}
$816 \div 4 = 204$ (o passo das dezenas é $1 \div 4$: \omterm{def:g1:counting:zero}{zero} vezes ---
escreva o $0$).

$420 \div 7 = 60$.

$2\,512 \div 5 = 502$, \omterm{def:g3:division:remainder}{restante} $2$ (verificar:
$5 \times 502 + 2 = 2\,512$).
\end{solution}

\begin{solution}{exo:g4:division:4}
$85 = 9 \times 9 + 4$ (\omterm{def:g3:division:remainder}{quociente} $9$, \omterm{def:g3:division:remainder}{restante} $4$).

$1\,000 = 3 \times 333 + 1$.
\end{solution}

\begin{solution}{exo:g4:division:5}
$156 \div 6 = 26$: vinte e seis caixas estão preenchidas, sem sobrar nenhuma.
\end{solution}

\begin{solution}{exo:g4:division:6}
$250 = 8 \times 31 + 2$: trinta e uma peças, e $2$ cm de fita
esquerda.
\end{solution}

\begin{solution}{exo:g4:division:7}
$375 = 52 \times 7 + 11$: sete ônibus transportam $364$ estudantes, $11$
permanecer --- o caso ``mais um'': $8$ são necessários ônibus.
\end{solution}

\begin{solution}{exo:g4:division:8}
$234 \div 3 = 78$: cada um recebe $78$ bolinhas de gude. Verificar:
$3 \times 78 = 234$.
\end{solution}

\begin{solution}{exo:g4:division:9}
Dividendo $= 7 \times 58 + 3 = 406 + 3 = 409$.
\end{solution}

\begin{solution}{exo:g4:division:10}
Prateleiras: $438 = 9 \times 48 + 6$: $48$ prateleiras cheias e $6$ livros
mais --- $49$ são necessárias prateleiras. Estantes:
$49 = 6 \times 8 + 1$: oito estantes completas e uma prateleira extra ---
$9$ estantes devem ser compradas.
\end{solution}

\begin{solution}{exo:g4:division:11}
\omterm{def:g3:division:remainder}{Quociente} $=$ \omterm{def:g3:division:remainder}{restante} $= r $ com $ r < 8$, então $ r $ é $0, 1, \dots,
7$ e o dividendo é $8r + r = 9r $: os possíveis dividendos são
$0$, $9$, $18$, $27$, $36$, $45$, $54$, $63$.
\end{solution}
