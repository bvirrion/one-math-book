\section*{Chapter \ref{ch:g4:fractions} --- First Fractions}

\begin{solution}{exo:g4:fractions:1}
Pizza: $\frac56$ left. Chocolate: $\frac38$ eaten. Flag: $\frac13$
colored.
\end{solution}

\begin{solution}{exo:g4:fractions:2}
$\frac35$ shades three parts, $\frac25$ only two: $\frac35$ is the
bigger shading.
\end{solution}

\begin{solution}{exo:g4:fractions:3}
\omterm{def:g3:division:half}{Half} of $18$: $9$. \omterm{def:g3:division:half}{Quarter} of $20$: $5$. Third of $21$: $7$.
$\frac34$ of $20$: three \omterm{def:g3:division:half}{quarters}, $3 \times 5 = 15$.
\end{solution}

\begin{solution}{exo:g4:fractions:4}
$\frac13$: one step; $\frac33$: on $1$; $\frac53$: two steps past
$1$. The \omterm{def:g4:fractions:def}{fraction} landing exactly on $2$ is $\frac63$.
\end{solution}

\begin{solution}{exo:g4:fractions:5}
$\frac38 < \frac68$; \quad $\frac55 = 1$; \quad $\frac74 > 1$;
\quad $\frac12 = \frac24$.
\end{solution}

\begin{solution}{exo:g4:fractions:6}
$\frac26 + \frac36 = \frac56$; \quad
$\frac58 - \frac28 = \frac38$; \quad
$\frac14 + \frac24 = \frac34$; \quad
$1 = \frac33$, so $1 - \frac13 = \frac23$.
\end{solution}

\begin{solution}{exo:g4:fractions:7}
$\frac54 = 1 + \frac14$; \quad
$\frac73 = 2 + \frac13$; \quad
$\frac92 = 4 + \frac12$.
\end{solution}

\begin{solution}{exo:g4:fractions:8}
\omterm{def:g3:division:half}{Half} an hour: $30$ min. \omterm{def:g3:division:half}{Quarter}: $15$ min. Three \omterm{def:g3:division:half}{quarters}: $45$
min. Third: $20$ min.
\end{solution}

\begin{solution}{exo:g4:fractions:9}
$\frac14$ of $24$ is $6$ students with glasses; $24 - 6 = 18$
without.
\end{solution}

\begin{solution}{exo:g4:fractions:10}
Together: $\frac38 + \frac48 = \frac78$. Left: $\frac18$. Sam ate
more ($4$ eighths against $3$).
\end{solution}

\begin{solution}{exo:g4:fractions:11}
Possible indeed: \omterm{def:g3:division:half}{half} of a small pizza can be less food than a
\omterm{def:g3:division:half}{quarter} of a giant one. \omterm{def:g4:fractions:def}{Fractions} compare parts \emph{of the same
whole}; before comparing $\frac12$ and $\frac14$ ``of something'',
one must know that the two somethings are equal.
\end{solution}
