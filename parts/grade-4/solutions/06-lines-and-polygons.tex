\section*{Chapter \ref{ch:g4:geometry} --- Lines and Polygons}

\begin{solution}{exo:g4:geometry:1}
Constructions with \cref{met:g4:geometry:draw}; the little square
marks the \omterm{def:g3:shapes:rightangle}{right angle} between $d$ and the \omterm{def:g4:geometry:def}{perpendicular}.
\end{solution}

\begin{solution}{exo:g4:geometry:2}
E: the three horizontal strokes are \omterm{def:g4:geometry:def}{parallel}; each is \omterm{def:g4:geometry:def}{perpendicular}
to the vertical one. H: the two vertical strokes are \omterm{def:g4:geometry:def}{parallel}, the
crossbar is \omterm{def:g4:geometry:def}{perpendicular} to both. N: the two vertical strokes are
\omterm{def:g4:geometry:def}{parallel}; the diagonal is neither. T: the two strokes are
\omterm{def:g4:geometry:def}{perpendicular}. Z: the top and bottom strokes are \omterm{def:g4:geometry:def}{parallel}; the
diagonal is neither.
\end{solution}

\begin{solution}{exo:g4:geometry:3}
\omterm{def:g4:geometry:polygon}{Hexagon}; \omterm{def:g4:geometry:polygon}{quadrilateral}; triangle; \omterm{def:g4:geometry:polygon}{octagon}; \omterm{def:g4:geometry:polygon}{pentagon}. A \omterm{def:g4:geometry:polygon}{polygon} has
as many \omterm{def:g4:geometry:polygon}{vertices} as sides: $6$, $4$, $3$, $8$, $5$.
\end{solution}

\begin{solution}{exo:g4:geometry:4}
A \omterm{def:g4:geometry:polygon}{pentagon} has $5$ sides. From $A$, the non-neighbors are $C$ and
$D$: two diagonals, $[AC]$ and $[AD]$.
\end{solution}

\begin{solution}{exo:g4:geometry:5}
$OA = 4$ cm (\omterm{def:g3:shapes:shapes}{radius}); $OM = 4$ cm (every point of the circle);
$AB = 8$ cm (a \omterm{def:g4:geometry:circle}{diameter} is twice the \omterm{def:g3:shapes:shapes}{radius}).
\end{solution}

\begin{solution}{exo:g4:geometry:6}
Open the compass along one side, then carry that opening to the two
other sides: if the spike-and-pencil span fits a second side
exactly, those two sides are equal.
\end{solution}

\begin{solution}{exo:g4:geometry:7}
The path closes into a \omterm{def:g4:geometry:polygon}{polygon} with $5$ sides --- a \omterm{def:g4:geometry:polygon}{pentagon} (not a
regular one!).
\end{solution}

\begin{solution}{exo:g4:geometry:8}
``\omterm{def:g4:geometry:def}{Perpendicular lines} always cross'': true --- they cross at the
\omterm{def:g3:shapes:rightangle}{right angle} itself.

``\omterm{def:g4:geometry:def}{Parallel lines} never cross'': true --- that is their definition.

``A line can be \omterm{def:g4:geometry:def}{parallel} to one line and \omterm{def:g4:geometry:def}{perpendicular} to
another'': true --- on grid paper, a horizontal line is \omterm{def:g4:geometry:def}{parallel} to
every horizontal line and \omterm{def:g4:geometry:def}{perpendicular} to every vertical one.
\end{solution}

\begin{solution}{exo:g4:geometry:9}
The lines $d$ and $f$ look \omterm{def:g4:geometry:def}{parallel} --- and they always are: two
lines \omterm{def:g4:geometry:def}{perpendicular} to the same line are \omterm{def:g4:geometry:def}{parallel}
(\cref{prop:g6:lines:facts} makes this official).
\end{solution}

\begin{solution}{exo:g4:geometry:10}
One correct program: ``1. Draw a square $ABCD$ of side $4$ cm.
2. Mark the point $O$ where its diagonals cross. 3. Draw the circle
of center $O$ and \omterm{def:g3:shapes:shapes}{radius} $2$ cm.'' The circle touches each side at
its middle.
\end{solution}

\begin{solution}{exo:g4:geometry:11}
\omterm{def:g4:geometry:polygon}{Quadrilateral}: $2$ diagonals. \omterm{def:g4:geometry:polygon}{Pentagon}: $5$. \omterm{def:g4:geometry:polygon}{Hexagon}: $9$. Each new
\omterm{def:g4:geometry:polygon}{vertex} adds more and more diagonals --- the counts grow faster and
faster ($+3$, then $+4$, \dots).
\end{solution}
