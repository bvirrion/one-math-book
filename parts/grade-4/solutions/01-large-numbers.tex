\section*{Chapter \ref{ch:g4:numbers} --- Large Numbers}

\begin{solution}{exo:g4:numbers:1}
$3\,520\,000$; \quad $7\,000\,008$; \quad $240\,060$.
\end{solution}

\begin{solution}{exo:g4:numbers:2}
$5\,204\,300$: five million two hundred four thousand three hundred.

$10\,050\,000$: ten million fifty thousand.

$999\,999$: nine hundred ninety-nine thousand nine hundred
ninety-nine.
\end{solution}

\begin{solution}{exo:g4:numbers:3}
Tens of thousands: $7$. Hundreds of the millions class: there is no
hundreds-of-millions digit in a seven-digit number --- the millions
class of $8\,472\,156$ contains only the digit $8$ (units of
millions). Thousands contained: $8\,472$ (since
$8\,472\,156 = 8\,472 \times 1\,000 + 156$).
\end{solution}

\begin{solution}{exo:g4:numbers:4}
$980\,000 < 1\,020\,000$; \quad
$3\,456\,000 < 3\,465\,000$ (thousands class $456 < 465$); \quad
$7\,090\,050 < 7\,090\,500$.
\end{solution}

\begin{solution}{exo:g4:numbers:5}
$980\,500 < 2\,030\,000 < 2\,299\,999 < 2\,300\,000$.
\end{solution}

\begin{solution}{exo:g4:numbers:6}
To the ten: $63\,780$. To the hundred: $63\,800$. To the thousand:
$64\,000$. To the ten thousand: $60\,000$.
\end{solution}

\begin{solution}{exo:g4:numbers:7}
$5\,000\,000$; \quad $6\,000\,000$ (the $499\,999$ falls short of
half a million); \quad $10\,000\,000$ (exactly halfway rounds up).
\end{solution}

\begin{solution}{exo:g4:numbers:8}
$998\,500 \to 999\,500 \to 1\,000\,500 \to 1\,001\,500$: the second
step crosses the million, and a new class lights up on the left.

$950\,000 \to 1\,050\,000 \to 1\,150\,000$: same crossing at the
first step.
\end{solution}

\begin{solution}{exo:g4:numbers:9}
Both are right. ``More than $80\,000$'' rounds down to the ten
thousand ($81\,338 \to 80\,000$); ``about $81\,000$'' rounds to the
nearest thousand ($81\,338 \to 81\,000$).
\end{solution}

\begin{solution}{exo:g4:numbers:10}
Biggest: $7\,654\,321$. Smallest: $1\,234\,567$. Closest to
$4\,000\,000$: just below works best --- $3\,765\,421$ is
$234\,579$ away, while just above, $4\,123\,567$, is only
$123\,567$ away: the closest is $4\,123\,567$.
\end{solution}

\begin{solution}{exo:g4:numbers:11}
Rounding to the nearest thousand gives $50\,000$ for all numbers
from $49\,500$ (halfway rounds up) to $50\,499$. Biggest:
$50\,499$; smallest: $49\,500$.
\end{solution}
