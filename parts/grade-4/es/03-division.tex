\section{El método de división larga}

\begin{method}[División larga por un número de un dígito]\label{met:g4:division:method}
Para dividir $749$ por $6$:
\begin{enumerate}
  \item toma el dígito más a la izquierda, $7$ (centenas): $6$ va a $1$ hora
        en $7$; escribe $1$ en el \omterm{def:g3:division:remainder}{cociente}, resta
        $6$, \omterm{def:g3:division:remainder}{resto} $1$;
  \item \emph{reducir} el siguiente dígito, $4$: ahora divide $14$
        (decenas): $6 \times 2 = 12$; escriba $2$, \omterm{def:g3:division:remainder}{resto} $2$;
  \item bajar el $9$: dividir $29$: $6 \times 4 = 24$; escribir
        $4$, \omterm{def:g3:division:remainder}{resto} $5$;
  \item no queda ningún dígito: el \omterm{def:g3:division:remainder}{cociente} es $124$, el \omterm{def:g3:division:remainder}{resto} $5$.
\end{enumerate}
Verifique: $6 \times 124 + 5 = 744 + 5 = 749$ y $5 < 6$.
\end{method}

\begin{example}[Un cero en el consciente]\label{ex:g4:division:zero}
Divida $618$ por $3$: centenas, $6 \div 3 = 2$; decenas, baja el
$1$: $3$ pasa $0$ veces a $1$ --- ¡\emph{escribe el} $0$! ---
\omterm{def:g3:division:remainder}{resto}$1$; unidades, baje el $8$: $18 \div 3 = 6$. \omterm{def:g3:division:remainder}{Cociente}
$206$, \omterm{def:g3:division:remainder}{resto} $0$. Olvidando el medio \omterm{def:g1:counting:zero}{cero} (escribiendo $26$)
es el error clásico; el cheque
$3 \times 26 = 78 \neq 618$ lo detecta inmediatamente.
\end{example}

\begin{example}[Estimando el tamaño del cociente]\label{ex:g4:division:size}
Antes de dividir $749$ por $6$, encuadre la respuesta:
$6 \times 100 = 600$ y $6 \times 200 = 1\,200$, por lo que \omterm{def:g3:division:remainder}{cociente} es
entre $100$ y $200$ --- tendrá tres dígitos. Esta línea
La estimación evita la mayoría de errores de dígitos extraviados.
\end{example}

\section{Elegir lo que pregunta la pregunta}

\begin{method}[Conciente, resto o conciente más uno]\label{met:g4:division:interpret}
Como en \cref{met:g3:division:interpret}, la historia decide:
\begin{enumerate}
  \item ``cuantos \dots completos'' o ``cuantos cada uno'': el \omterm{def:g3:division:remainder}{cociente};
  \item ``cuantos quedan'': el \omterm{def:g3:division:remainder}{resto};
  \item ``cuántos contenedores se necesitan para todos'': el \omterm{def:g3:division:remainder}{cociente}, más
        uno si el \omterm{def:g3:division:remainder}{resto} no es \omterm{def:g1:counting:zero}{cero}.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g4:division:problem}
Los libros $530$ deben empaquetarse en cajas de $8$.
\[
530 = 8 \times 66 + 2 .
\]
¿Cuántas cajas \emph{lleno}? $66$. ¿Cuántos libros no están completos?
caja? $2$. ¿Cuántas cajas para empacar \emph{todo} los libros? $67$.
Tres preguntas, una división.
\end{example}

\begin{example}[Reparto justo del dinero]\label{ex:g4:division:money}
Cuatro amigos comparten equitativamente el coste de un regalo $92$:
$92 \div 4 = 23$ exactamente (\omterm{def:g3:division:remainder}{resto} $0$). Cada uno paga $23$. cuando el
\omterm{def:g3:division:remainder}{resto} es \omterm{def:g1:counting:zero}{cero}, la división ``sale pareja'' y el reparto
es perfectamente justo.
\end{example}

\section{Ceremonias}

\begin{exercise}[$\star$]\label{exo:g4:division:1}
Enmarque cada \omterm{def:g3:division:remainder}{cociente} como en \cref{ex:g4:division:size} (entre los cuales
dos números ``\omterm{def:g4:numbers:round}{redondo}''?), luego calcula la división larga:
$96 \div 4$; \quad $87 \div 5$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:division:2}
Calcula las divisiones largas y verifica cada una:
$672 \div 4$; \quad$925 \div 7$; \quad $804 \div 6$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:division:3}
Cuidado con el \omterm{def:g1:counting:zero}{cero} (ver \cref{ex:g4:division:zero}):
$816 \div 4$; \quad$420 \div 7$; \quad $2\,512 \div 5$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:division:4}
Escriba la igualdad de verificación ($a = b \times q + r$) para: $85 \div 9$;
\quad $1\,000 \div 3$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:division:5}
Los huevos $156$ se colocan en cajas de $6$. ¿Cuántas cajas están llenas?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:division:6}
Una cinta de $250$ cm se corta en trozos de $8$ cm. cuantas piezas
y que largo sobra?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:division:7}
Estudiantes $375$ van a un espectáculo en autobuses de plazas $52$. cuantos autobuses
son necesarios? (¿Qué caso de
\cref{met:g4:division:interpret} ¿es esto? Dividir por $52$ usando
múltiplos: $52 \times 7 = 364$.)
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:division:8}
Tres amigos comparten canicas $234$ a partes iguales. ¿Cuántas canicas cada una?
Compruébalo con una multiplicación.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:division:9}
Encuentre el dividendo: la división entre $7$ da \omterm{def:g3:division:remainder}{cociente} $58$ y
\omterm{def:g3:division:remainder}{resto} $3$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g4:division:10}
Un bibliotecario almacena libros $438$, $9$ por estante. Los estantes entran
librerías de estantes $6$. ¿Cuántos estantes se necesitan? cuantos
¿Hay que comprar librerías? (Dos divisiones, ambas con un
pregunta "más uno".)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g4:division:11}
En una división por $8$, el \omterm{def:g3:division:remainder}{cociente} es igual al \omterm{def:g3:division:remainder}{resto}. que son
¿Los posibles dividendos? Enumérelos todos. (El \omterm{def:g3:division:remainder}{resto} debe permanecer
debajo de $8$.)
\end{exercise}
