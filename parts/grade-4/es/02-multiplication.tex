\chapter{Multiplicar números más grandes}\label{ch:g4:mult}

Grado 3 multiplicado por un dígito (\cref{ch:g3:mult}); este capitulo
multiplica por números de dos dígitos --- y explica el misterioso
``línea desplazada'' del método de la columna, que no es más que una
dividir en decenas y unidades.

\section{Multiplicar por decenas}

\begin{proposition}[Veces 10, 20, 300\,\dots]\label{prop:g4:mult:tens}
Al multiplicar por $10$ se agrega un \omterm{def:g1:counting:zero}{cero}, por $100$ dos \omterm{def:g1:counting:zero}{ceros}. multiplicar
por $20$, multiplicar por $2$, luego por $10$:
\[
36 \times 20 = 36 \times 2 \times 10 = 72 \times 10 = 720 .
\]
Asimismo $14 \times 300 = 14 \times 3 \times 100 = 4\,200$.
\end{proposition}
\admitted

\section{El método de la columna con dos dígitos.}

\begin{example}[¿Por qué la línea desplazada?]\label{ex:g4:mult:why}
Para calcular $23 \times 12$, divida $12$ en $10 + 2$:
\[
23 \times 12 = 23 \times 2 + 23 \times 10 = 46 + 230 = 276 .
\]
El método de la columna escribe exactamente estos dos \omterm{def:g2:mult:def}{productos}, uno debajo del
otro --- el segundo se desplazó hacia la izquierda porque es un número de
\emph{decenas}:
\[
\begin{array}{r}
2\,3 \\
\times\ 1\,2 \\
\hline
4\,6 \\
2\,3\,\cdot \\
\hline
2\,7\,6
\end{array}
\]
(el punto marca el lugar de las unidades de la línea desplazada, a menudo hacia la izquierda
en blanco).
\end{example}

\begin{method}[Multiplicación de columna por un número de dos dígitos]\label{met:g4:mult:column}
Para calcular $457 \times 36$:
\begin{enumerate}
  \item primera línea: $457 \times 6 = 2\,742$;
  \item segunda línea: $457 \times 3$ \emph{decenas} $= 1\,371$ decenas ---
        escriba $1\,371$ desplazado un lugar a la izquierda;
  \item agregue las dos líneas: $2\,742 + 13\,710 = 16\,452$;
  \item estimación a comprobar: $457 \times 36 \approx 500 \times 36
        = 18\,000$? Mejor: $450 \times 36$ se trata de $16\,000$ ---
        plausible.
\end{enumerate}
\[
\begin{array}{r}
4\,5\,7 \\
\times\ \ \,3\,6 \\
\hline
2\,7\,4\,2 \\
1\,3\,7\,1\,\cdot \\
\hline
1\,6\,4\,5\,2
\end{array}
\]
\end{method}

\begin{example}[trucos mentales]\label{ex:g4:mult:tricks}
\begin{itemize}
  \item \emph{Dividir el camino amistoso}: $18 \times 11 = 18 \times 10
        + 18 = 198$.
  \item \emph{Utilice casi cien}: $25 \times 99 = 25 \times 100 -
        25 = 2\,475$.
  \item \emph{Duplicar y reducir a la mitad}: $16 \times 35 = 8 \times 70 =
        560$ (reducir a la mitad un factor y duplicar el otro mantiene el
        \omterm{def:g2:mult:def}{producto}).
\end{itemize}
\end{example}

\section{Multiplicación en problemas}

\begin{example}\label{ex:g4:mult:problem}
Una escuela pide $28$ cajas de $145$ hojas de papel.
\begin{enumerate}
  \item Operación: $145 \times 28$.
  \item Estimación: $150 \times 30 = 4\,500$, demasiado en ambos
        cuenta.
  \item Columnas: $145 \times 8 = 1\,160$; $145 \times 2$ decenas
        $= 2\,900$; total $4\,060$.
  \item Sentencia: el colegio recibe hojas $4\,060$.
\end{enumerate}
\end{example}

\section{Ceremonias}

\begin{exercise}[$\star$]\label{exo:g4:mult:1}
Calcular mentalmente: $47 \times 10$; \quad$23 \times 100$; \quad
$36 \times 20$; \quad$15 \times 300$; \quad$42 \times 50$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:mult:2}
Calcule $34 \times 21$ dividiendo como en \cref{ex:g4:mult:why}
($34 \times 21 = 34 \times 20 + 34$), luego verifique las columnas.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:mult:3}
Calcular en columnas: $63 \times 24$; \quad$85 \times 47$; \quad
$126 \times 53$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:mult:4}
Calcule en columnas, con una estimación primero: $208 \times 34$; \quad
$517 \times 62$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:mult:5}
Calcula con un truco de \cref{ex:g4:mult:tricks}:
$45 \times 11$; \quad$32 \times 99$; \quad $24 \times 45$ (doble
y reducir a la mitad --- dos veces si lo desea).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:mult:6}
Un cine tiene $26$ filas de $18$ asientos. ¿Cuantos asientos en total?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:mult:7}
Un camión transporta paletas $32$ de cajas $48$. ¿Cuántas cajas? dar
Primero una estimación y luego la respuesta exacta.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:mult:8}
Encuentra el error: Ana calcula $57 \times 23$ como
$57 \times 3 = 171$ y $57 \times 2 = 114$, luego agrega
$171 + 114 = 285$. La respuesta correcta es $1\,311$. ¿Qué hizo ella?
olvidar?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g4:mult:9}
Un jardinero planta $14$ hileras de tulipanes $35$ y $12$ hileras de $28$
narcisos. ¿Cuántas flores en total? (Dos \omterm{def:g2:mult:def}{productos}, luego un \omterm{def:g1:addition:def}{suma}).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g4:mult:10}
Cada día una panadería utiliza $67$ kg de harina. aproximadamente cuanta harina es
que en un mes de $30$ días --- estimación con números redondeados, entonces
calcular exactamente.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g4:mult:11}
Explica, con un dibujo rectangular como el de este capítulo, por qué
$15 \times 12$ se puede calcular como $15 \times 10 + 15 \times 2$.
Luego invente otra división de $15 \times 12$ que también funcione.
\end{exercise}
