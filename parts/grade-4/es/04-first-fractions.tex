\section{Nombrando las piezas}

\begin{definition}[fracción]\label{def:g4:fractions:def}
Cortar un todo en partes iguales y tomar un poco --- eso es un
\emph{fracción}\index{fracción}:
\[
\frac{3}{4}
\quad
\begin{array}{l}
3 \text{: how many parts are taken (the \emph{numerador}\index{numerador});}\\
4 \text{: how many equal parts the whole is cut into (the
\emph{denominador}\index{denominador}).}
\end{array}
\]
$\frac12$ es un \emph{\omterm{def:g3:division:half}{medio}}, $\frac13$ un \emph{tercero}, $\frac14$ un
\emph{\omterm{def:g3:division:half}{cuarto}}, $\frac{1}{10}$ y \emph{décimo}.
\end{definition}

\begin{example}[Fracciones de una colección]\label{ex:g4:fractions:collection}
$\frac13$ de canicas $12$: comparte las canicas $12$ en $3$ iguales
grupos de $4$; un grupo es $\frac13$, por lo que $\frac13$ de $12$ es $4$
--- y $\frac23$ de $12$ son dos grupos, $8$.
\end{example}

\section{Fracciones en la recta numérica}

\begin{method}[Colocando $\frac{a}{b}$]\label{met:g4:fractions:line}
Corte cada unidad de la línea en $b$ pasos iguales; desde $0$, camina $a$
pasos. Si $a$ es mayor que $b$, el recorrido pasa por $1$: \omterm{def:g4:fractions:def}{fracciones} puede
ser más grande que un todo!
\end{method}

\begin{example}[Numeros enteros escondidos en fracciones]\label{ex:g4:fractions:whole}
$\frac44 = 1$ (cuatro \omterm{def:g3:division:half}{cuarteles} forman el conjunto); $\frac82 = 4$;
$\frac{12}{3} = 4$. Y $\frac74$ es $1 + \frac34$: un todo y
tres \omterm{def:g3:division:half}{cuarteles} más.
\end{example}

\section{Comparar y sumar fracciones simples}

\begin{proposition}[Mismo denominador]\label{prop:g4:fractions:compare}
Con el mismo \omterm{def:g4:fractions:def}{denominador}, las piezas tienen el mismo tamaño --- así que solo
cuéntalos:
\[
\frac{2}{8} < \frac{5}{8},
\qquad
\frac{3}{8} + \frac{4}{8} = \frac{7}{8}.
\]
Comparar con un todo también es fácil: $\frac{5}{8} < 1 < \frac{9}{8}$
(menos, luego más, de $8$ octavos).
\end{proposition}

\begin{proof}[Por qué, en una imagen]
Sombreando $3$ octavos y luego $4$ octavos más de las mismas sombras de barra
$7$ octavos en total.
\end{proof}

\begin{example}[Diferentes denominadores, mismo punto.]\label{ex:g4:fractions:equal}
En la recta numérica, $\frac12$, $\frac24$ y $\frac{5}{10}$ aterrizan
sobre el mismo punto: son tres nombres del mismo número. corte
cada \omterm{def:g3:division:half}{medio} dividido en dos forma \omterm{def:g3:division:half}{cuarteles}; en cinco, décimas. (el general
La regla está en \cref{ch:g6:fractions}.)
\end{example}

\begin{example}[cuartos de hora]\label{ex:g4:fractions:hour}
Una hora tiene $60$ minutos, por lo que \omterm{def:g3:division:half}{cuarto} de una hora es
$60 \div 4 = 15$ minutos, y tres \omterm{def:g3:division:half}{cuarteles} de hora son
$45$ minutos. ``Las dos y media'' es $\frac12$ hora después de las dos
en punto.
\end{example}

\section{Ceremonias}

\begin{exercise}[$\star$]\label{exo:g4:fractions:1}
Escribe el \omterm{def:g4:fractions:def}{fracción} que se muestra: una pizza cortada en $6$ con porciones $5$ restantes;
una barra de chocolate de $8$ cuadrados con $3$ comido (\omterm{def:g4:fractions:def}{fracción} comido); un
Bandera dividida en $3$ bandas verticales iguales, $1$ de color.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:fractions:2}
Dibuja una barra cortada en $5$ partes iguales y sombrea $\frac35$. Luego sombrea
$\frac{2}{5}$ de otra barra igual. ¿Qué sombreado es más grande?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:fractions:3}
Calcular: $\frac12$ de $18$; \quad $\frac14$ de $20$; \quad
$\frac13$ de $21$; \quad $\frac34$ de $20$ (tres grupos de un
\omterm{def:g3:division:half}{cuarto}).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:fractions:4}
Colocar en una recta numérica cortada en tercios: $\frac13$; $\frac33$; \
$\frac53$; $2$. ¿Qué \omterm{def:g4:fractions:def}{fracción} aterriza exactamente en $2$?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:fractions:5}
Copiar y completar con $<$, $>$ o $=$:
\[
\frac38 \;?\; \frac68, \qquad
\frac55 \;?\; 1, \qquad
\frac74 \;?\; 1, \qquad
\frac12 \;?\; \frac24 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:fractions:6}
Calcular: $\frac26 + \frac36$; \quad$\frac58 - \frac28$; \quad
$\frac14 + \frac24$; \quad $1 - \frac13$ (cuántos tercios forman
$1$?).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:fractions:7}
Escribe como un número entero más un \omterm{def:g4:fractions:def}{fracción} menor que $1$:
$\frac54$; \quad$\frac73$; \quad $\frac{9}{2}$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:fractions:8}
¿Cuántos minutos son: \omterm{def:g3:division:half}{medio} una hora; un \omterm{def:g3:division:half}{cuarto} de una hora; tres
\omterm{def:g3:division:half}{cuarteles} de una hora; $\frac13$ de una hora?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:fractions:9}
En una clase de estudiantes $24$, $\frac14$ usa anteojos. cuantos
¿Los estudiantes usan gafas? ¿Cuantos no lo hacen?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g4:fractions:10}
Nina comió $\frac38$ de una pizza y Sam comió $\frac48$ de la misma.
pizza. ¿Qué \omterm{def:g4:fractions:def}{fracción} comieron juntos? ¿Qué \omterm{def:g4:fractions:def}{fracción} queda?
¿Quién comió más?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g4:fractions:11}
Verdadero o falso, con una imagen o un ejemplo: ``$\frac12$ de un pequeño
la pizza es menor que $\frac14$ de una pizza muy grande es posible''.
¿Qué hay que saber siempre antes de comparar dos \omterm{def:g4:fractions:def}{fracciones} de
algo?
\end{exercise}
