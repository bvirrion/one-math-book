\section{Plegado y combinación}

\begin{definition}[Eje de simetria]\label{def:g4:symmetry:def}
Una línea es un \emph{eje de simetría}\index{eje de simetría} de un
figura al doblar la figura a lo largo de esa línea se forman las dos mitades
coincide con \emph{exactamente}. Una figura puede no tener eje, tener un solo eje o
varios.
\end{definition}

\section{Completar figuras en una cuadrícula}

\begin{method}[Simétrica de una figura a través de una línea de cuadrícula]\label{met:g4:symmetry:grid}
El eje es una línea vertical (u horizontal) de la cuadrícula.
\begin{enumerate}
  \item Tome cada \omterm{def:g4:geometry:polygon}{vértice} de la figura por turno;
  \item cuenta su distancia al eje en cuadrados;
  \item coloque la imagen \omterm{def:g4:geometry:polygon}{vértice} a la distancia \emph{mismo} en el
        \emph{otro} lado, en la misma fila (o columna);
  \item une la imagen \omterm{def:g4:geometry:polygon}{vértices} en el mismo orden y verifica: el
        La imagen es el espejo gemelo de la figura.
\end{enumerate}
\end{method}

\section{Ejes de las formas habituales.}

\begin{example}[Contador de ejes]\label{ex:g4:symmetry:shapes}
Al doblar:
\begin{itemize}
  \item un cuadrado: $4$ ejes (dos por el medio de ejes opuestos
        lados, dos a lo largo de las diagonales);
  \item un rectángulo: $2$ ejes (¡solo el centro de los lados opuestos!);
  \item un triángulo isósceles: $1$; un triángulo equilátero: $3$;
  \item un círculo: cada línea que pasa por su centro --- más ejes que
        uno puede contar.
\end{itemize}
\end{example}

\begin{example}[Simetría en el alfabeto]\label{ex:g4:symmetry:letters}
A, M, T, U, V tienen un eje vertical; B, C, D, E tienen una dirección horizontal.
uno; H, I, O, X tienen ambos. Las palabras formadas con las letras correctas pueden ser
simétrico también: MUM tiene un eje vertical; OXO funciona en ambos sentidos.
\end{example}

\section{Ceremonias}

\begin{exercise}[$\star$]\label{exo:g4:symmetry:1}
Trazar y doblar: ¿cuáles de estos tienen un \omterm{def:g4:symmetry:def}{eje de simetría} --- a
cuadrado; un rectángulo (no cuadrado); un triángulo escaleno (todos los lados
diferente); un circulo?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:symmetry:2}
Dibuja los ejes de simetría de: un cuadrado; un rectángulo; un equilátero
triángulo. ¿Cuantos para cada uno?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:symmetry:3}
Utilice la prueba de papel de calco de \cref{ex:g4:symmetry:trace} para mostrar
que la diagonal de un rectángulo $6 \times 4$ es \emph{no} y \omterm{def:g4:symmetry:def}{eje
de simetria}.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:symmetry:4}
En papel cuadriculado, dibuja un eje vertical y la letra L (tres cuadrados
de alto, dos de ancho) en $2$ cuadrados desde el eje. Construye su espejo
imagen.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:symmetry:5}
En papel cuadriculado, dibuja un eje horizontal y una pequeña bandera (un triángulo $1 \times
1$ en un mástil cuadrado $3$) encima. Completa la figura con
simetría debajo del eje.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:symmetry:6}
Una figura toca su \omterm{def:g4:symmetry:def}{eje de simetría} en un punto $P$. donde esta el
imagen de $P$? Explica con el dibujo plegable.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:symmetry:7}
Ordenar las letras mayúsculas A, B, E, F, H, N, O, T: ¿eje vertical?
eje horizontal? ¿ambos? ¿ni?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:symmetry:8}
Un triángulo tiene lados $5$ cm, $5$ cm y $3$ cm. ¿Tiene un
\omterm{def:g4:symmetry:def}{eje de simetría}? ¿Qué lados intercambia el pliegue?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g4:symmetry:9}

ejes de simetría y dibuja ambos ejes. (Pista: piensa en un rectángulo,

\end{exercise}
