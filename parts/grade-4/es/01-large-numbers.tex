\chapter{Grandes números}\label{ch:g4:numbers}

Miles eran el tope de \cref{ch:g3:numbers}; ahora el techo
sube a \emph{millones}. El truco que hace legibles los números enormes
es cortar sus dígitos en grupos de tres --- el \omterm{def:g4:numbers:classes}{clases}.

\section{Clases de tres dígitos}

\begin{definition}[clases]\label{def:g4:numbers:classes}
Empezando por la derecha, los dígitos de un número se agrupan en
\emph{clases} de tres: la clase de \emph{unidades}, la clase de
\emph{miles}, la clase de \emph{millones}. Cada clase tiene su
unidades propias, decenas y centenas. por ejemplo
\[
6\,309\,452 = 6 \text{ millions} + 309 \text{ thousands}
+ 452 \text{ units},
\]
léase "seis millones trescientos nueve mil cuatrocientos
cincuenta y dos''.
\end{definition}

\begin{example}[Escribiendo con las clases]\label{ex:g4:numbers:write}
``Doce millones cuarenta mil quinientos'' tiene $12$ en el
clase millones, $040$ en la clase miles, $500$ en las unidades
clase:
\[
12\,040\,500 .
\]
Los \omterm{def:g1:counting:zero}{ceros} de $040$ son esenciales: cada clase (excepto la más a la izquierda)
debe mostrar sus tres dígitos.
\end{example}

\begin{example}[¿Cuántas millas?]\label{ex:g4:numbers:contained}
El número $6\,309\,452$ \emph{contiene} $6\,309$ miles
(todo lo que queda de la clase de unidades): $6\,309\,452 = 6\,309 \times
1\,000 + 452$. Preguntar "cuántos miles" no es pedir uno
dígito, sino para un prefijo completo del número.
\end{example}

\section{Comparar y redondear}

\begin{method}[Comparando números grandes]\label{met:g4:numbers:compare}
Exactamente como en \cref{met:g3:numbers:compare}, con el \omterm{def:g4:numbers:classes}{clases} como
atajo: más dígitos ganan; misma longitud, comparar desde la izquierda ---
clase por clase, luego dígito por dígito.
\end{method}

\begin{example}\label{ex:g4:numbers:compare}
$2\,145\,890$ y $2\,148\,540$: millones iguales ($2$), miles
clase $145 < 148$, entonces
\[
2\,145\,890 < 2\,148\,540 .
\]
\end{example}

\begin{definition}[redondeo]\label{def:g4:numbers:round}
A \emph{redondo}\index{redondeo} un número al millar más cercano
(o diez, cien, millones\,\dots), reemplácelo por el múltiplo más cercano
de $1\,000$. Si el número está exactamente a la mitad, redondea hacia arriba. en
práctica: mire el dígito justo debajo del lugar de redondeo --- $5$ o
más redondeos hacia arriba, $4$ o menos redondeos hacia abajo.
\end{definition}

\begin{example}\label{ex:g4:numbers:round}
$37\,614$ redondeado al millar más cercano: el dígito de las centenas es
$6$, entonces \omterm{def:g4:numbers:round}{redondo} arriba: $38\,000$. Redondeado a la centena más cercana:
$37\,600$. Redondeado a la decena de millar más cercana: $40\,000$.
\end{example}

\begin{example}[Órdenes de magnitud en el mundo]\label{ex:g4:numbers:world}
\omterm{def:g4:numbers:round}{redondeo} así es como se utilizan realmente los números grandes: un país de
$67\,842\,591$ habitantes son ``aproximadamente $68$ millones''; una ciudad de
$214\,376$ personas son "aproximadamente $200\,000$". La cifra exacta cambia.
todos los días de todos modos --- el redondeado lleva lo útil
información.
\end{example}

\section{Ceremonias}

\begin{exercise}[$\star$]\label{exo:g4:numbers:1}
Escribe en dígitos, con el \omterm{def:g4:numbers:classes}{clases} separado: ``tres millones cinco
ciento veinte mil''; ``siete millones ocho''; ``doscientos
cuarenta mil sesenta''.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:numbers:2}
Escriba en palabras: $5\,204\,300$; \quad$10\,050\,000$; \quad
$999\,999$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:numbers:3}
En $8\,472\,156$: ¿qué dígito son las decenas de miles? los cientos
de la clase de los millones? ¿A cuántos \emph{miles} equivale el número?
contiene (ver \cref{ex:g4:numbers:contained})?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:numbers:4}
Copiar y completar con $<$ o $>$:
\[
980\,000 \;?\; 1\,020\,000, \qquad
3\,456\,000 \;?\; 3\,465\,000, \qquad
7\,090\,050 \;?\; 7\,090\,500 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:numbers:5}
Ordenar de menor a mayor: $2\,300\,000$; $980\,500$; \
$2\,030\,000$; $2\,299\,999$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:numbers:6}
Ronda $63\,782$: a la decena más cercana; a la centena más cercana; al
mil más cercano; a la decena de millar más cercana.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:numbers:7}
Redondear al millón más cercano: $4\,721\,000$; \quad$6\,499\,999$;
\quad $9\,500\,000$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:numbers:8}
Cuente por pasos: tres pasos de $1\,000$ desde $998\,500$; dos pasos
de $100\,000$ de $950\,000$. ¿Qué le sucede al \omterm{def:g4:numbers:classes}{clases} cuando
cruzar un millón?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:numbers:9}
Un estadio tiene capacidad para $81\,338$ espectadores. Un periódico escribe "más de
$80\,000$ fans''; una radio dice "sobre $81\,000$". son ambos
¿verdad? ¿A qué lugar llegó cada \omterm{def:g4:numbers:round}{redondo}?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g4:numbers:10}
Usando cada uno de los dígitos $1$ a $7$ exactamente una vez, escriba el mayor
posible número de siete cifras, el más pequeño y el más cercano a
$4\,000\,000$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g4:numbers:11}
¿Cuál es el número entero mayor que \omterm{def:g4:numbers:round}{rondas} a $50\,000$ cuando
\omterm{def:g4:numbers:round}{redondeo} al millar más cercano? ¿Y el más pequeño?
\end{exercise}
