\section{Perímetro}

\begin{definition}[Perímetro]\label{def:g4:measure:perimeter}
El \emph{perímetro}\index{perímetro} de un \omterm{def:g4:geometry:polygon}{polígono} es el total
longitud de su borde: suma las longitudes de todos sus lados (en la misma
unidad!). Atajos para los clásicos:
\[
\text{rectangle: } P = 2 \times (L + w),
\qquad
\text{square: } P = 4 \times c .
\]
\end{definition}

\begin{example}\label{ex:g4:measure:perimeter}
Un jardín rectangular $12$ m por $7$ m:
\[
P = 2 \times (12 + 7) = 2 \times 19 = 38 \text{ m}
\]
de valla. Un marco de fotos cuadrado de lado $18$ cm:
$P = 4 \times 18 = 72$cm.
\end{example}

\section{Área: contar cuadrados}

\begin{definition}[Área por conteo]\label{def:g4:measure:area}
El \emph{área}\index{área} de una figura dibujada en una cuadrícula es el número
de cuadrados de la cuadrícula que cubre. Dos medios cuadrados cuentan como uno. Área
responde ``cuánta superficie'', \omterm{def:g4:measure:perimeter}{perímetro} responde ``¿cuánto mide el
frontera '' --- ¡dos preguntas diferentes!
\end{definition}

\section{Duraciones}

\begin{method}[Agregando tiempos a lo largo de la hora]\label{met:g4:measure:time}
Para sumar o restar duraciones, trabaje en saltos durante las horas completas
(recuerde $1$ h $= 60$ min):
\[
9\text{ h }40 + 35 \text{ min}:
\quad 9\text{ h }40 \xrightarrow{+20} 10\text{ h }00
\xrightarrow{+15} 10\text{ h }15 .
\]
Nunca agregue minutos después de $60$ sin convertir: $40 + 35 = 75$ min
$= 1$ h $15$ mín.
\end{method}

\begin{example}\label{ex:g4:measure:timetable}
El tren sale en $8{:}47$ y llega a $11{:}05$. Duración:
$8{:}47 \to 9{:}00$ es $13$ mínimo; $9{:}00 \to 11{:}00$ es $2$h;
$11{:}00 \to 11{:}05$ es $5$ mín. Total: $2$ h $18$ mín.
\end{example}
