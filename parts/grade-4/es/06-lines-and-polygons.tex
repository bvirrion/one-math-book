\chapter{Líneas y polígonos}\label{ch:g4:geometry}

Con una regla, una escuadra y un compás, la geometría se convierte en un oficio.
Este capítulo enseña las dos posiciones especiales de las líneas ---
\omterm{def:g4:geometry:def}{perpendicular} y \omterm{def:g4:geometry:def}{paralelo} --- a nivel de herramienta, y realiza un recorrido por
la familia \omterm{def:g4:geometry:polygon}{polígono}. (La teoría de estas nociones entra en juego
\cref{ch:g6:lines}.)

\section{Perpendicular y paralela}

\begin{definition}[Perpendicular, paralela]\label{def:g4:geometry:def}
Dos líneas son \emph{perpendicular}\index{lineas perpendiculares} cuando
se cruzan en \omterm{def:g3:shapes:rightangle}{ángulo recto} (comprobado con la escuadra, marcado con
un cuadrito). Dos líneas son \emph{paralelo}\index{lineas paralelas}
cuando corren en la misma dirección y nunca se cruzan --- como los dos
carriles de una vía, siempre a la misma distancia entre sí.
\end{definition}

\begin{method}[Dibujo con la escuadra]\label{met:g4:geometry:draw}
\emph{\omterm{def:g4:geometry:def}{Perpendicular} to a line $d$ through a point $P$}: colocar un borde
del \omterm{def:g3:shapes:rightangle}{ángulo recto} a lo largo de $d$, deslice hasta que el otro borde llegue a $P$,
dibujar.

\emph{\omterm{def:g4:geometry:def}{Paralelo} to $d$ through $P$}: coloque el borde de la escuadra a lo largo
$d$, presione la regla contra otro borde a modo de riel, deslice el conjunto
cuadrado a lo largo de la regla hasta $P$, dibuje.
\end{method}

\begin{example}[Observándolos en cifras]\label{ex:g4:geometry:spot}
En un rectángulo, cada lado es \omterm{def:g4:geometry:def}{perpendicular} con respecto a sus dos vecinos y
\omterm{def:g4:geometry:def}{paralelo} al lado opuesto. El papel cuadriculado es una red ya preparada de
\omterm{def:g4:geometry:def}{paralelas} y \omterm{def:g4:geometry:def}{perpendiculares} --- eso es lo que lo hace así
cómodo para dibujar.
\end{example}

\section{Polígonos}

\begin{definition}[Polígono]\label{def:g4:geometry:polygon}
Un \emph{polígono}\index{polígono} es una figura cerrada cuyo borde es
hecho de segmentos rectos, su \emph{lados}; los puntos de las esquinas son
el \emph{vértices}\index{vértice}. Los polígonos se nombran por su número de lados:
\emph{triángulo} ($3$), \emph{cuadrilátero}\index{cuadrilátero} ($4$),
\emph{pentágono}\index{pentágono} ($5$), \emph{hexágono}\index{hexágono} ($6$),
\emph{octágono}\index{octágono}
($8$).
\end{definition}

\begin{example}[Describiendo un polígono]\label{ex:g4:geometry:describe}
Un juego útil: describe una figura con tanta precisión que un compañero pueda
Vuelve a dibujarlo sin verlo. ``A \omterm{def:g4:geometry:polygon}{cuadrilátero} $ABCD$ con
$AB = 5$ cm, el ángulo en $A$ a la derecha, $AD = 3$ cm\,\dots'' ---
El vocabulario es lo que hace que el mensaje funcione.
\end{example}

\section{El círculo y el compás}

\begin{definition}[circulo]\label{def:g4:geometry:circle}
El \emph{círculo} de centro $O$ y \omterm{def:g3:shapes:shapes}{radio} $3$ cm es el conjunto de todos
apunta exactamente a $3$ cm de $O$. La brújula lo dibuja: punta en
$O$, apertura $3$ cm, giro. Un \emph{diámetro}\index{diámetro} es un segmento que pasa por
el centro que une dos puntos del círculo; mide el doble
\omterm{def:g3:shapes:shapes}{radio}.
\end{definition}

\begin{example}[La brújula como distanciador]\label{ex:g4:geometry:compass}
La brújula hace más que dibujar círculos: \emph{transporta
distancias}. Para marcar todos los puntos a $4$ cm desde un punto $A$, dibuje el
círculo de centro $A$, \omterm{def:g3:shapes:shapes}{radio} $4$ cm; comparar dos segmentos sin
una regla, abre el compás de uno y pruébalo en el otro.
\end{example}

\begin{method}[Reproduciendo una figura en una cuadrícula]\label{met:g4:geometry:reproduce}
\begin{enumerate}
  \item Elija un \omterm{def:g4:geometry:polygon}{vértice} inicial y su posición en la cuadrícula en blanco;
  \item recorre la figura \omterm{def:g4:geometry:polygon}{vértice} por \omterm{def:g4:geometry:polygon}{vértice}, contando la cuadrícula
        pasos (``$3$ derecha, $2$ arriba'');
  \item únete al \omterm{def:g4:geometry:polygon}{vértices} en orden y compáralo con el modelo:
        mismas longitudes, mismas direcciones.
\end{enumerate}
\end{method}

\section{Ceremonias}

\begin{exercise}[$\star$]\label{exo:g4:geometry:1}
Dibuja una línea $d$ y un punto $P$ que no esté en ella. Construir el
\omterm{def:g4:geometry:def}{perpendicular} a $d$ a $P$, luego \omterm{def:g4:geometry:def}{paralelo} a $d$ a
$P$ (con el método regla-carril). Marque el \omterm{def:g3:shapes:rightangle}{ángulo recto}.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:geometry:2}
Observa las letras mayúsculas E, H, N, T, Z. En cada una, encuentra un par de
\omterm{def:g4:geometry:def}{paralelo} trazos y un par de trazos \omterm{def:g4:geometry:def}{perpendicular} cuando
existir.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:geometry:3}
Nombre cada lado \omterm{def:g4:geometry:polygon}{polígono}: $6$; $4$ lados; $3$ lados; $8$ lados;
$5$ lados. ¿Cuantos \emph{\omterm{def:g4:geometry:polygon}{vértices}} tiene cada uno?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:geometry:4}
Dibuja cualquier \omterm{def:g4:geometry:polygon}{pentágono} (no necesariamente regular) y etiquétalo como \omterm{def:g4:geometry:polygon}{vértices}.
$A$, $B$, $C$, $D$, $E$. ¿Cuantos lados tiene? Dibuja su
diagonales de $A$ (segmentos que unen $A$ al no vecino \omterm{def:g4:geometry:polygon}{vértices}):
cuantos hay?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:geometry:5}
Dibuja un círculo de centro $O$ y \omterm{def:g3:shapes:shapes}{radio} $4$ cm; a \omterm{def:g4:geometry:circle}{diámetro} $[AB]$;
y un punto $M$ del círculo que no está en $[AB]$. Indique las longitudes $OA$,
$OM$ y $AB$ sin medir.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:geometry:6}
Con la brújula como distanciador
(\cref{ex:g4:geometry:compass}), compara los lados de un triángulo
dibujado por un compañero y decide si tiene dos lados iguales.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:geometry:7}
En papel cuadriculado, reproduzca este camino y ciérrelo en un \omterm{def:g4:geometry:polygon}{polígono}:
comenzar en un punto $A$; vaya $4$ a la derecha; $2$ arriba; $3$ izquierda; $2$ arriba;
luego regresa a $A$ en línea recta. ¿Cómo se llama el \omterm{def:g4:geometry:polygon}{polígono}?
obtienes (cuenta sus lados)?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:geometry:8}
¿Verdadero o falso? Explica con un dibujo.
``Dos \omterm{def:g4:geometry:def}{lineas perpendiculares} siempre se cruzan.''
``Dos \omterm{def:g4:geometry:def}{lineas paralelas} nunca se cruzan.''
``Una línea puede ser \omterm{def:g4:geometry:def}{paralelo} a una línea y \omterm{def:g4:geometry:def}{perpendicular} a
otro."
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g4:geometry:9}
Dibuja una línea $d$, luego una línea $e$ \omterm{def:g4:geometry:def}{perpendicular} a $d$, luego una línea
$f$ \omterm{def:g4:geometry:def}{perpendicular} a $e$. Mira $d$ y $f$: ¿qué notas?
(Esta observación se convierte en un teorema en \cref{ch:g6:lines}).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g4:geometry:10}
Describe la siguiente figura con palabras para un compañero de clase, sin mostrar
it: un cuadrado de lado $4$ cm, con un círculo de \omterm{def:g3:shapes:shapes}{radio} $2$ cm centrado
en el medio del cuadrado (tocando los cuatro lados). Escribe el
Descripción como instrucciones numeradas.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g4:geometry:11}
¿Cuántas diagonales tiene un \omterm{def:g4:geometry:polygon}{cuadrilátero}? ¿A \omterm{def:g4:geometry:polygon}{pentágono}? ¿A \omterm{def:g4:geometry:polygon}{hexágono}?
Dibuja los tres casos, cuenta con cuidado y describe el patrón que
ver.
\end{exercise}
