\chapter{Mesures et périmètre}\label{ch:g4:measure}

L'année~3 mesurait avec règles, balances et horloges
(\cref{ch:g3:measure})~; ce chapitre ajoute le tableau des unités ---
la machine qui convertit tout --- et deux façons de mesurer une
figure plane~: la longueur de son bord (périmètre) et l'espace
à l'intérieur (aire, comptée en carreaux pour l'instant).

\section{Le tableau des unités}

\begin{method}[Convertir avec le tableau]\label{met:g4:measure:table}
Pour les longueurs, chaque colonne vaut dix de la suivante~:
\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{c|c|c|c|c|c|c}
km & hm & dam & m & dm & cm & mm \\
\hline
 & & & $3$ & $4$ & $0$ & \\
\end{tabular}
\end{center}
\begin{enumerate}
  \item Écrire le nombre avec un chiffre par colonne, le chiffre
        des unités de la mesure dans la colonne de son unité (ici
        $3.4$~m~: le $3$ sous m)~;
  \item lire le nombre dans la nouvelle unité~: $3.4$~m $= 340$~cm
        (remplir les colonnes vides avec des zéros)~;
  \item les masses (kg\,\dots\,g) et les capacités (L\,\dots\,cL)
        fonctionnent de la même façon.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g4:measure:convert}
$5$~km $= 5\,000$~m~; \quad $270$~cm $= 2.7$~m (le $2$ se place sous
m)~; \quad $3$~kg $250$~g $= 3\,250$~g~; \quad
$2.5$~L $= 250$~cL.
\end{example}

\section{Périmètre}

\begin{definition}[Périmètre]\label{def:g4:measure:perimeter}
Le \emph{périmètre}\index{périmètre} d'un polygone est la longueur
totale de son bord~: on additionne les longueurs de tous ses côtés
(dans la même unité~!). Raccourcis pour les classiques~:
\[
\text{rectangle~: } P = 2 \times (L + \ell),
\qquad
\text{carré~: } P = 4 \times c .
\]
\end{definition}

\begin{example}\label{ex:g4:measure:perimeter}
Un jardin rectangulaire de $12$~m sur $7$~m~:
\[
P = 2 \times (12 + 7) = 2 \times 19 = 38 \text{ m}
\]
de clôture. Un cadre photo carré de côté $18$~cm~:
$P = 4 \times 18 = 72$~cm.
\end{example}

\section{Aire~: compter les carreaux}

\begin{definition}[Aire par comptage]\label{def:g4:measure:area}
L'\emph{aire}\index{aire} d'une figure dessinée sur un quadrillage
est le nombre de carreaux qu'elle couvre. Deux demi-carreaux
comptent pour un. L'aire répond à «~combien de surface~», le
périmètre à «~quelle longueur a le bord~» --- deux questions
différentes~!
\end{definition}

\begin{omfigure}
\begin{tikzpicture}[scale=0.55]
  \draw[gray!40, thin] (0,0) grid (13,4);
  \draw[very thick, omDef, fill=omDef!15] (1,1) rectangle (5,3);
  \node[font=\small, omDef] at (3,2) {$8$ carreaux};
  \draw[very thick, omThm, fill=omThm!15]
    (7,1) -- (11,1) -- (11,2) -- (9,2) -- (9,3) -- (7,3) -- cycle;
  \node[font=\small, omThm] at (8.4,1.5) {$6$ carreaux};
\end{tikzpicture}

{\small Deux figures sur le quadrillage~: aires $8$ et $6$ carreaux.
Maintenant parcourir les bords et compter les unités~: $12$ pour
les deux~! Même périmètre, aires différentes.}
\end{omfigure}

\begin{example}[Demi-carreaux]\label{ex:g4:measure:half}
Un triangle rectangle dessiné sur le quadrillage avec des côtés de
l'angle droit $4$ et $2$ couvre $3$ carreaux pleins et $2$
demi-carreaux\,\dots\ compter soigneusement~: son aire est la moitié
du rectangle $4 \times 2$, c'est-à-dire $4$ carreaux. Couper un
rectangle le long de sa diagonale donne toujours deux triangles de
même aire.
\end{example}

\begin{example}[Même aire, périmètres différents]\label{ex:g4:measure:samearea}
Un carré $4 \times 4$ et un rectangle $8 \times 2$ couvrent tous
deux $16$ carreaux --- même aire. Bords~: $16$ unités pour le
carré, $20$ pour le rectangle. Ni l'une ni l'autre mesure ne
détermine l'autre~: une forme longue et fine a beaucoup de bord
pour peu de surface.
\end{example}

\section{Durées}

\begin{method}[Additionner des temps en passant l'heure]\label{met:g4:measure:time}
Pour additionner ou soustraire des durées, on travaille par
sauts à travers les heures pleines (rappel $1$~h $= 60$~min)~:
\[
9\text{ h }40 + 35 \text{ min}:
\quad 9\text{ h }40 \xrightarrow{+20} 10\text{ h }00
\xrightarrow{+15} 10\text{ h }15 .
\]
Ne jamais additionner les minutes au-delà de $60$ sans convertir~:
$40 + 35 = 75$~min $= 1$~h $15$~min.
\end{method}

\begin{example}\label{ex:g4:measure:timetable}
Le train part à $8{:}47$ et arrive à $11{:}05$. Durée~:
$8{:}47 \to 9{:}00$ fait $13$~min~; $9{:}00 \to 11{:}00$ fait
$2$~h~; $11{:}00 \to 11{:}05$ fait $5$~min. Total~: $2$~h $18$~min.
\end{example}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:g4:measure:1}
Convertir avec le tableau des unités~: $7$~m en cm~; $3.2$~km en
m~; $85$~mm en cm~; $4\,500$~m en km.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:measure:2}
Convertir~: $2$~kg en g~; $1\,800$~g en kg et g~; $3.5$~L en cL~;
$25$~cL en L.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:measure:3}
Calculer le périmètre~: un rectangle $9$~cm $\times$ $5$~cm~; un
carré de côté $7.5$~cm~; un triangle de côtés $6$~cm, $8$~cm et
$10$~cm.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:measure:4}
Un rectangle a un périmètre de $26$~cm et une longueur de $8$~cm.
Trouver sa largeur (trouver d'abord longueur $+$ largeur).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:measure:5}
Sur papier quadrillé, dessiner une figure en L couvrant exactement
$10$ carreaux, et compter les unités de son périmètre.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:measure:6}
Dessiner deux rectangles différents d'aire $12$ carreaux. Calculer
le périmètre de chacun. Lequel est «~plus carré~», lequel est
«~plus allongé~»~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:measure:7}
Un triangle rectangle est dessiné sur le quadrillage avec des côtés
de l'angle droit $6$ et $3$. Quelle est son aire en carreaux~?
(Utiliser \cref{ex:g4:measure:half}.)
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:measure:8}
Calculer les durées~: de $10{:}20$ à $12{:}45$~; de $7{:}35$ à
$9{:}10$. Un film dure $1$~h $50$ et commence à $20{:}30$~: à
quelle heure se termine-t-il~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:measure:9}
Kim court $3$ tours d'un terrain rectangulaire de $60$~m sur
$45$~m. Combien de mètres court-elle~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g4:measure:10}
Un fermier veut clôturer un champ carré de côté $85$~m, en laissant
un portail de $4$~m sans clôture. Combien de mètres de clôture
doit-il acheter~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g4:measure:11}
Vrai ou faux, avec des exemples sur le quadrillage~: «~si deux
figures ont le même périmètre, elles ont la même aire~»~; «~si
deux figures ont la même aire, elles ont le même périmètre~».
(\Cref{ex:g4:measure:samearea} et la figure du chapitre aideront.)
\end{exercise}
