\chapter{Droites et polygones}\label{ch:g4:geometry}

Avec une règle, une équerre et un compas, la géométrie devient un
métier. Ce chapitre enseigne les deux positions spéciales des
droites --- \omterm{def:g4:geometry:def}{perpendiculaires} et \omterm{def:g4:geometry:def}{parallèles} --- au niveau des
outils, et fait un tour de la famille des \omterm{def:g4:geometry:polygon}{polygones}. (La théorie de
ces notions arrive dans \cref{ch:g6:lines}.)

\section{Perpendiculaires et parallèles}

\begin{definition}[Perpendiculaires, parallèles]\label{def:g4:geometry:def}
Deux droites sont \emph{perpendiculaires}\index{droites perpendiculaires}\index{perpendiculaire}
lorsqu'elles se croisent en formant un \omterm{def:g3:shapes:rightangle}{angle droit} (vérifié avec
l'équerre, marqué d'un petit carré). Deux droites sont
\emph{parallèles}\index{droites parallèles}\index{parallèle} lorsqu'elles vont dans
la même direction et ne se croisent jamais --- comme les deux rails
d'une voie, toujours à la même distance l'un de l'autre.
\end{definition}

\begin{method}[Tracer avec l'équerre]\label{met:g4:geometry:draw}
\emph{\omterm{def:g4:geometry:def}{Perpendiculaire} à une droite $d$ passant par un point $P$}~:
poser un côté de l'\omterm{def:g3:shapes:rightangle}{angle droit} le long de $d$, glisser jusqu'à ce
que l'autre côté atteigne $P$, tracer.

\emph{\omterm{def:g4:geometry:def}{Parallèle} à $d$ passant par $P$}~: poser un côté de l'équerre
le long de $d$, appuyer la règle contre un autre côté comme un
rail, faire glisser l'équerre le long de la règle jusqu'à $P$,
tracer.
\end{method}

\begin{example}[Les repérer dans les figures]\label{ex:g4:geometry:spot}
Dans un rectangle, chaque côté est \omterm{def:g4:geometry:def}{perpendiculaire} à ses deux
voisins et \omterm{def:g4:geometry:def}{parallèle} au côté opposé. Le papier quadrillé est un
réseau tout prêt de \omterm{def:g4:geometry:def}{parallèles} et de \omterm{def:g4:geometry:def}{perpendiculaires} --- c'est ce
qui le rend si confortable pour dessiner.
\end{example}

\section{Polygones}

\begin{definition}[Polygone]\label{def:g4:geometry:polygon}
Un \emph{polygone}\index{polygone} est une figure fermée dont le
bord est formé de segments de droite, ses \emph{côtés}~; les points
de coin sont les \emph{sommets}\index{sommet}. Les polygones se nomment d'après
leur nombre de côtés~: \emph{triangle} ($3$),
\emph{quadrilatère}\index{quadrilatère} ($4$),
\emph{pentagone}\index{pentagone} ($5$),
\emph{hexagone}\index{hexagone} ($6$),
\emph{octogone}\index{octogone} ($8$).
\end{definition}

\begin{example}[Décrire un polygone]\label{ex:g4:geometry:describe}
Un jeu utile~: décrire une figure si précisément qu'un camarade
puisse la redessiner sans la voir. «~Un \omterm{def:g4:geometry:polygon}{quadrilatère} $ABCD$ avec
$AB = 5$~cm, l'angle en $A$ droit, $AD = 3$~cm\,\dots~» --- le
vocabulaire est ce qui fait fonctionner le message.
\end{example}

\section{Le cercle et le compas}

\begin{definition}[Cercle]\label{def:g4:geometry:circle}
Le \emph{cercle} de centre $O$ et de \omterm{def:g3:shapes:shapes}{rayon} $3$~cm est l'ensemble de
tous les points situés exactement à $3$~cm de $O$. Le compas le
trace~: pointe sur $O$, ouverture $3$~cm, on tourne. Un
\emph{diamètre}\index{diamètre} est un segment passant par le centre et joignant
deux points du cercle~; il mesure deux fois le \omterm{def:g3:shapes:shapes}{rayon}.
\end{definition}

\begin{example}[Le compas comme garde-distance]\label{ex:g4:geometry:compass}
Le compas fait plus que tracer des cercles~: il \emph{transporte
des distances}. Pour marquer tous les points à $4$~cm d'un point
$A$, on trace le cercle de centre $A$, \omterm{def:g3:shapes:shapes}{rayon} $4$~cm~; pour comparer
deux segments sans règle, on ouvre le compas sur l'un et on le
teste sur l'autre.
\end{example}

\begin{method}[Reproduire une figure sur un quadrillage]\label{met:g4:geometry:reproduce}
\begin{enumerate}
  \item Choisir un \omterm{def:g4:geometry:polygon}{sommet} de départ et sa position sur le
        quadrillage vierge~;
  \item parcourir la figure \omterm{def:g4:geometry:polygon}{sommet} par \omterm{def:g4:geometry:polygon}{sommet}, en comptant les pas
        du quadrillage («~$3$ à droite, $2$ en haut~»)~;
  \item joindre les \omterm{def:g4:geometry:polygon}{sommets} dans l'ordre, et comparer au modèle~:
        mêmes longueurs, mêmes directions.
\end{enumerate}
\end{method}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:g4:geometry:1}
Tracer une droite $d$ et un point $P$ hors de $d$. Construire la
\omterm{def:g4:geometry:def}{perpendiculaire} à $d$ passant par $P$, puis la \omterm{def:g4:geometry:def}{parallèle} à $d$
passant par $P$ (avec la méthode règle-rail). Marquer l'\omterm{def:g3:shapes:rightangle}{angle
droit}.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:geometry:2}
Regarder les lettres capitales E, H, N, T, Z. Dans chacune, trouver
une paire de traits \omterm{def:g4:geometry:def}{parallèles} et une paire de traits
\omterm{def:g4:geometry:def}{perpendiculaires} lorsqu'elles existent.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:geometry:3}
Nommer chaque \omterm{def:g4:geometry:polygon}{polygone}~: $6$ côtés~; $4$ côtés~; $3$ côtés~;
$8$ côtés~; $5$ côtés. Combien de \emph{\omterm{def:g4:geometry:polygon}{sommets}} chacun a-t-il~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:geometry:4}
Dessiner un \omterm{def:g4:geometry:polygon}{pentagone} quelconque (pas nécessairement régulier) et
étiqueter ses \omterm{def:g4:geometry:polygon}{sommets} $A$, $B$, $C$, $D$, $E$. Combien de côtés
a-t-il~? Tracer ses diagonales issues de $A$ (segments joignant $A$
à des \omterm{def:g4:geometry:polygon}{sommets} non voisins)~: combien y en a-t-il~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:geometry:5}
Tracer un cercle de centre $O$ et de \omterm{def:g3:shapes:shapes}{rayon} $4$~cm~; un \omterm{def:g4:geometry:circle}{diamètre}
$[AB]$~; et un point $M$ du cercle non situé sur $[AB]$. Donner les
longueurs $OA$, $OM$ et $AB$ sans mesurer.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:geometry:6}
Avec le compas comme garde-distance
(\cref{ex:g4:geometry:compass}), comparer les côtés d'un triangle
dessiné par un camarade et décider s'il a deux côtés égaux.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:geometry:7}
Sur papier quadrillé, reproduire ce chemin et le fermer en
\omterm{def:g4:geometry:polygon}{polygone}~: partir d'un point $A$~; aller $4$ à droite~; $2$ en
haut~; $3$ à gauche~; $2$ en haut~; puis revenir en $A$ en ligne
droite. Quel est le nom du \omterm{def:g4:geometry:polygon}{polygone} obtenu (compter ses côtés)~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:geometry:8}
Vrai ou faux~? Expliquer avec un dessin.
«~Deux \omterm{def:g4:geometry:def}{droites perpendiculaires} se croisent toujours.~»
«~Deux \omterm{def:g4:geometry:def}{droites parallèles} ne se croisent jamais.~»
«~Une droite peut être \omterm{def:g4:geometry:def}{parallèle} à une droite et \omterm{def:g4:geometry:def}{perpendiculaire} à
une autre.~»
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g4:geometry:9}
Tracer une droite $d$, puis une droite $e$ \omterm{def:g4:geometry:def}{perpendiculaire} à $d$,
puis une droite $f$ \omterm{def:g4:geometry:def}{perpendiculaire} à $e$. Regarder $d$ et $f$~:
que remarque-t-on~? (Cette observation devient un théorème dans
\cref{ch:g6:lines}.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g4:geometry:10}
Décrire la figure ci-dessous en mots pour un camarade, sans la
montrer~: un carré de côté $4$~cm, avec un cercle de \omterm{def:g3:shapes:shapes}{rayon} $2$~cm
centré au milieu du carré (touchant les quatre côtés). Écrire la
description sous forme d'instructions numérotées.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g4:geometry:11}
Combien de diagonales a un \omterm{def:g4:geometry:polygon}{quadrilatère}~? Un \omterm{def:g4:geometry:polygon}{pentagone}~? Un
\omterm{def:g4:geometry:polygon}{hexagone}~? Dessiner les trois cas, compter soigneusement, et
décrire le motif observé.
\end{exercise}
