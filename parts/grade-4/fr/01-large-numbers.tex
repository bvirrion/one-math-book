\chapter{Les grands nombres}\label{ch:g4:numbers}

Les milliers étaient le plafond de \cref{ch:g3:numbers}~; maintenant
le plafond monte jusqu'aux \emph{millions}. L'astuce qui rend les
grands nombres lisibles, c'est de découper leurs chiffres en groupes
de trois --- les \omterm{def:g4:numbers:classes}{classes}.

\section{Classes de trois chiffres}

\begin{definition}[Classes]\label{def:g4:numbers:classes}
En partant de la droite, les chiffres d'un nombre sont regroupés en
\emph{classes} de trois~: la classe des \emph{unités}, la classe des
\emph{milliers}, la classe des \emph{millions}. Chaque classe a ses
propres unités, dizaines et centaines. Par exemple
\[
6\,309\,452 = 6 \text{ millions} + 309 \text{ milliers}
+ 452 \text{ unités},
\]
qui se lit «~six millions trois cent neuf mille quatre cent
cinquante-deux~».
\end{definition}

\begin{example}[Écrire avec les classes]\label{ex:g4:numbers:write}
«~Douze millions quarante mille cinq cents~» a $12$ dans la
classe des millions, $040$ dans celle des milliers, $500$ dans celle
des unités~:
\[
12\,040\,500 .
\]
Les \omterm{def:g1:counting:zero}{zéros} de $040$ sont indispensables~: chaque classe (sauf la plus
à gauche) doit afficher ses trois chiffres.
\end{example}

\begin{example}[Combien de milliers~?]\label{ex:g4:numbers:contained}
Le nombre $6\,309\,452$ \emph{contient} $6\,309$ milliers
(tout ce qui est à gauche de la classe des unités)~:
$6\,309\,452 = 6\,309 \times 1\,000 + 452$. Demander «~combien de
milliers~» n'est pas demander un seul chiffre, mais tout un préfixe
du nombre.
\end{example}

\section{Comparer et arrondir}

\begin{method}[Comparer de grands nombres]\label{met:g4:numbers:compare}
Exactement comme dans \cref{met:g3:numbers:compare}, avec les \omterm{def:g4:numbers:classes}{classes}
en raccourci~: plus de chiffres gagne~; même longueur, on compare de
la gauche --- classe par classe, puis chiffre par chiffre.
\end{method}

\begin{example}\label{ex:g4:numbers:compare}
$2\,145\,890$ et $2\,148\,540$~: millions égaux ($2$), classe des
milliers $145 < 148$, donc
\[
2\,145\,890 < 2\,148\,540 .
\]
\end{example}

\begin{definition}[Arrondi]\label{def:g4:numbers:round}
Pour \emph{arrondir}\index{arrondi} un nombre au millier le plus
proche (ou à la dizaine, à la centaine, au million\,\dots), on le
remplace par le multiple de $1\,000$ le plus proche. Si le nombre est
exactement à mi-chemin, on arrondit vers le haut. En pratique~: on
regarde le chiffre juste en dessous de la place d'arrondi --- $5$ ou
plus arrondit vers le haut, $4$ ou moins vers le bas.
\end{definition}

\begin{example}\label{ex:g4:numbers:round}
$37\,614$ \omterm{def:g4:numbers:round}{arrondi} au millier le plus proche~: le chiffre des
centaines est $6$, donc on arrondit vers le haut~: $38\,000$. \omterm{def:g4:numbers:round}{Arrondi}
à la centaine la plus proche~: $37\,600$. \omterm{def:g4:numbers:round}{Arrondi} à la dizaine de
milliers la plus proche~: $40\,000$.
\end{example}

\begin{example}[Ordres de grandeur dans le monde]\label{ex:g4:numbers:world}
L'\omterm{def:g4:numbers:round}{arrondi} est la façon dont on utilise vraiment les grands
nombres~: un pays de $67\,842\,591$ habitants, c'est «~environ
$68$~millions~»~; une ville de $214\,376$ personnes, c'est «~environ
$200\,000$~». Le chiffre exact change chaque jour de toute façon ---
l'\omterm{def:g4:numbers:round}{arrondi} porte l'information utile.
\end{example}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:g4:numbers:1}
Écrire en chiffres, avec les \omterm{def:g4:numbers:classes}{classes} séparées~: «~trois millions cinq
cent vingt mille~»~; «~sept millions huit~»~; «~deux cent quarante
mille soixante~».
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:numbers:2}
Écrire en lettres~: $5\,204\,300$~; \quad $10\,050\,000$~; \quad
$999\,999$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:numbers:3}
Dans $8\,472\,156$~: quel chiffre est celui des dizaines de
milliers~? Celui des centaines de la classe des millions~? Combien
de \emph{milliers} le nombre contient-il (voir
\cref{ex:g4:numbers:contained})~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:numbers:4}
Recopier et compléter avec $<$ ou $>$~:
\[
980\,000 \;?\; 1\,020\,000, \qquad
3\,456\,000 \;?\; 3\,465\,000, \qquad
7\,090\,050 \;?\; 7\,090\,500 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:numbers:5}
Ranger du plus petit au plus grand~: $2\,300\,000$~; \ $980\,500$~; \
$2\,030\,000$~; \ $2\,299\,999$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:numbers:6}
Arrondir $63\,782$~: à la dizaine la plus proche~; à la centaine la
plus proche~; au millier le plus proche~; à la dizaine de milliers
la plus proche.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:numbers:7}
Arrondir au million le plus proche~: $4\,721\,000$~; \quad
$6\,499\,999$~; \quad $9\,500\,000$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:numbers:8}
Compter par pas~: trois pas de $1\,000$ à partir de $998\,500$~;
deux pas de $100\,000$ à partir de $950\,000$. Que se passe-t-il
pour les \omterm{def:g4:numbers:classes}{classes} quand on franchit un million~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:numbers:9}
Un stade accueille $81\,338$ spectateurs. Un journal écrit «~plus de
$80\,000$ fans~»~; une radio dit «~environ $81\,000$~». Les deux ont-ils
raison~? À quelle place chacun a-t-il \omterm{def:g4:numbers:round}{arrondi}~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g4:numbers:10}
En utilisant chaque chiffre de $1$ à $7$ exactement une fois, écrire
le plus grand nombre à sept chiffres possible, le plus petit, et
celui le plus proche de $4\,000\,000$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g4:numbers:11}
Quel est le plus grand nombre entier qui s'arrondit à $50\,000$
lorsqu'on arrondit au millier le plus proche~? Et le plus petit~?
\end{exercise}
