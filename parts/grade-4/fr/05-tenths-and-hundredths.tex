\chapter{Dixièmes et centièmes}\label{ch:g4:decimals}

Entre $3$ et $4$, les nombres entiers se taisent --- mais un
coureur peut être à $3$~mètres et un peu devant. Découper l'unité
en dix, puis en cent, donne les \emph{fractions décimales}, et un
merveilleux raccourci pour elles~: la virgule. (Les nombres décimaux
ont leur chapitre complet dans \cref{ch:g5:decimals}.)

\section{Fractions décimales}

\begin{definition}[Dixièmes et centièmes]\label{def:g4:decimals:def}
Découper une unité en $10$ parts égales donne des \emph{dixièmes},
$\frac{1}{10}$~; découper chaque dixième encore en $10$ donne des
\emph{centièmes}, $\frac{1}{100}$. Ainsi
\[
1 = \frac{10}{10} = \frac{100}{100},
\qquad
\frac{1}{10} = \frac{10}{100}.
\]
\end{definition}

\begin{example}\label{ex:g4:decimals:count}
Colorier $3$ colonnes et $7$ cases supplémentaires du carré de cent
colore
\[
\frac{3}{10} + \frac{7}{100} = \frac{30}{100} + \frac{7}{100}
= \frac{37}{100}
\]
du carré --- trente-sept centièmes.
\end{example}

\section{La virgule}

\begin{definition}[Écriture décimale]\label{def:g4:decimals:point}
Le nombre $2 + \frac{3}{10} + \frac{7}{100}$ s'écrit
\[
2.37
\]
--- la \emph{virgule} sépare la partie entière ($2$) de la partie
décimale~; le premier chiffre après la virgule compte les dixièmes,
le second les centièmes. Lecture~: «~deux virgule trois sept~», ou
«~deux et trente-sept centièmes~».
\end{definition}

\begin{example}\label{ex:g4:decimals:convert}
\[
\frac{7}{10} = 0.7, \qquad
\frac{43}{100} = 0.43, \qquad
\frac{5}{100} = 0.05, \qquad
3 + \frac{4}{10} = 3.4 .
\]
Attention avec $\frac{5}{100}$~: cinq \emph{centièmes} demandent un
$0$ à la place des dixièmes --- $0.05$, et non $0.5$.
\end{example}

\begin{example}[L'argent, ce sont des centièmes]\label{ex:g4:decimals:money}
Un centime est un centième d'un euro (ou d'un dollar, d'une
livre)~: un prix de $4.85$ signifie $4$ pièces entières et
$85$ centièmes. L'argent est la virgule qu'on utilise déjà tous les
jours~: $10$~centimes $= 0.10$~; deux euros cinquante $= 2.50$~; et
$0.05$ est une pièce de cinq centimes, pas cinquante~!
\end{example}

\section{Comparer avec des dixièmes}

\begin{method}[Comparer des décimaux (premier contact)]\label{met:g4:decimals:compare}
\begin{enumerate}
  \item Comparer d'abord les parties entières~: $3.2 > 2.9$.
  \item Mêmes parties entières~: comparer les dixièmes, puis les
        centièmes~: $2.37 < 2.5$ car $3$ dixièmes $< 5$ dixièmes.
  \item Écrire les deux avec deux décimales aide~: $2.5 = 2.50$, et
        $37 < 50$ centièmes.
\end{enumerate}
Le piège~: $2.37$ a plus de chiffres que $2.5$, mais est
\emph{plus petit} --- plus de chiffres ne veut pas dire plus grand~!
\end{method}

\begin{example}\label{ex:g4:decimals:compare}
Ranger $0.4$, $0.35$ et $0.09$~: avec des centièmes, $0.40$,
$0.35$, $0.09$~; donc
\[
0.09 < 0.35 < 0.4 .
\]
\end{example}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:g4:decimals:1}
Sur un carré de cent, quelle portion est coloriée si l'on colorie~:
$5$ colonnes~; $2$ colonnes et $3$ cases~; $45$ cases~? Répondre
avec des fractions.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:decimals:2}
Écrire avec une virgule~:
$\frac{9}{10}$~; \quad $\frac{27}{100}$~; \quad $\frac{3}{100}$~; \quad
$5 + \frac{6}{10}$~; \quad $\frac{60}{100}$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:decimals:3}
Écrire sous forme de fraction (dixièmes ou centièmes)~:
$0.3$~; \quad $0.51$~; \quad $0.07$~; \quad $1.9$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:decimals:4}
Dans $6.58$~: que compte le chiffre $5$~? Le chiffre $8$~?
Décomposer le nombre comme dans \cref{def:g4:decimals:point}.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:decimals:5}
Placer sur une droite graduée en dixièmes de $4$ à $5$~:
$4.2$~; \ $4.75$ (entre quelles deux marques~?)~; \ $4.9$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:decimals:6}
Recopier et compléter avec $<$, $>$ ou $=$~:
\[
0.6 \;?\; 0.60, \qquad
2.8 \;?\; 2.75, \qquad
0.09 \;?\; 0.1, \qquad
5.3 \;?\; 5.13 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:decimals:7}
Ranger du plus petit au plus grand~: $1.05$~; \ $1.5$~; \ $0.95$~; \
$1.45$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:decimals:8}
Écrire en écriture décimale~: trois euros et cinq centimes~; douze
euros et cinquante centimes~; quatre-vingt-dix-neuf centimes. Puis
ranger les trois prix.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:decimals:9}
Quels nombres décimaux à un chiffre après la virgule se situent
strictement entre $3$ et $4$~? Les lister tous. Combien y en
a-t-il~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g4:decimals:10}
Tom dit~: «~$0.25$ est plus grand que $0.5$ parce que $25$ est plus
grand que $5$~». Lui montrer son erreur avec le carré de cent
(combien de cases chaque nombre colore-t-il~?).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g4:decimals:11}
Les trois sauts d'un athlète mesurent $3.6$~m, $3.58$~m et
$3.65$~m. Ranger les sauts du plus court au plus long. Lesquels
diffèrent d'exactement $\frac{5}{100}$ de mètre~?
\end{exercise}
