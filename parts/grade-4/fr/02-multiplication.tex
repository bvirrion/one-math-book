\chapter{Multiplier de plus grands nombres}\label{ch:g4:mult}

L'année~3 multipliait par un chiffre (\cref{ch:g3:mult})~; ce
chapitre multiplie par des nombres à deux chiffres --- et explique la
mystérieuse «~ligne décalée~» de la multiplication posée, qui n'est
rien d'autre qu'un découpage en dizaines et unités.

\section{Multiplier par des dizaines}

\begin{proposition}[Fois 10, 20, 300\,\dots]\label{prop:g4:mult:tens}
Multiplier par $10$ ajoute un zéro, par $100$ deux zéros. Pour
multiplier par $20$, on multiplie par $2$, puis par $10$~:
\[
36 \times 20 = 36 \times 2 \times 10 = 72 \times 10 = 720 .
\]
De même $14 \times 300 = 14 \times 3 \times 100 = 4\,200$.
\end{proposition}
\admitted

\section{La multiplication posée à deux chiffres}

\begin{example}[Pourquoi la ligne décalée~?]\label{ex:g4:mult:why}
Pour calculer $23 \times 12$, on découpe $12$ en $10 + 2$~:
\[
23 \times 12 = 23 \times 2 + 23 \times 10 = 46 + 230 = 276 .
\]
La multiplication posée écrit exactement ces deux produits, l'un
sous l'autre --- le second décalé à gauche parce que c'est un nombre
de \emph{dizaines}~:
\[
\begin{array}{r}
2\,3 \\
\times\ 1\,2 \\
\hline
4\,6 \\
2\,3\,\cdot \\
\hline
2\,7\,6
\end{array}
\]
(le point marque la place des unités de la ligne décalée, souvent
laissée vide).
\end{example}

\begin{method}[Multiplication posée par un nombre à deux chiffres]\label{met:g4:mult:column}
Pour calculer $457 \times 36$~:
\begin{enumerate}
  \item première ligne~: $457 \times 6 = 2\,742$~;
  \item deuxième ligne~: $457 \times 3$ \emph{dizaines} $= 1\,371$
        dizaines --- on écrit $1\,371$ décalé d'une place à gauche~;
  \item on additionne les deux lignes~:
        $2\,742 + 13\,710 = 16\,452$~;
  \item on estime pour vérifier~: $457 \times 36 \approx 500 \times 36
        = 18\,000$~? Mieux~: $450 \times 36$ fait environ $16\,000$ ---
        plausible.
\end{enumerate}
\[
\begin{array}{r}
4\,5\,7 \\
\times\ \ \,3\,6 \\
\hline
2\,7\,4\,2 \\
1\,3\,7\,1\,\cdot \\
\hline
1\,6\,4\,5\,2
\end{array}
\]
\end{method}

\begin{example}[Astuces de calcul mental]\label{ex:g4:mult:tricks}
\begin{itemize}
  \item \emph{Découper le plus simple}~: $18 \times 11 = 18 \times 10
        + 18 = 198$.
  \item \emph{Utiliser une centaine proche}~: $25 \times 99 = 25 \times 100 -
        25 = 2\,475$.
  \item \emph{Doubler et diviser par deux}~: $16 \times 35 = 8 \times 70 =
        560$ (diviser un facteur par deux et doubler l'autre
        conserve le produit).
\end{itemize}
\end{example}

\section{La multiplication dans les problèmes}

\begin{example}\label{ex:g4:mult:problem}
Une école commande $28$ boîtes de $145$ feuilles de papier.
\begin{enumerate}
  \item Opération~: $145 \times 28$.
  \item Estimation~: $150 \times 30 = 4\,500$, un peu trop des deux
        côtés.
  \item Posée~: $145 \times 8 = 1\,160$~; $145 \times 2$ dizaines
        $= 2\,900$~; total $4\,060$.
  \item Phrase~: l'école reçoit $4\,060$ feuilles.
\end{enumerate}
\end{example}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:g4:mult:1}
Calculer mentalement~: $47 \times 10$~; \quad $23 \times 100$~; \quad
$36 \times 20$~; \quad $15 \times 300$~; \quad $42 \times 50$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:mult:2}
Calculer $34 \times 21$ en découpant comme dans
\cref{ex:g4:mult:why} ($34 \times 21 = 34 \times 20 + 34$), puis
vérifier en posant.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:mult:3}
Calculer en posant~: $63 \times 24$~; \quad $85 \times 47$~; \quad
$126 \times 53$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:mult:4}
Calculer en posant, avec une estimation d'abord~: $208 \times 34$~;
\quad $517 \times 62$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:mult:5}
Calculer avec une astuce de \cref{ex:g4:mult:tricks}~:
$45 \times 11$~; \quad $32 \times 99$~; \quad $24 \times 45$ (doubler
et diviser par deux --- deux fois si on veut).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:mult:6}
Un cinéma a $26$ rangées de $18$ sièges. Combien de sièges en tout~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:mult:7}
Un camion transporte $32$ palettes de $48$ caisses. Combien de
caisses~? Donner d'abord une estimation, puis la réponse exacte.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:mult:8}
Trouver l'erreur~: Ana calcule $57 \times 23$ comme
$57 \times 3 = 171$ et $57 \times 2 = 114$, puis additionne
$171 + 114 = 285$. La bonne réponse est $1\,311$. Qu'a-t-elle
oublié~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g4:mult:9}
Un jardinier plante $14$ rangées de $35$ tulipes et $12$ rangées de
$28$ jonquilles. Combien de fleurs en tout~? (Deux produits, puis
une somme.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g4:mult:10}
Chaque jour, une boulangerie utilise $67$~kg de farine. Environ
combien de farine pour un mois de $30$~jours --- estimer avec des
nombres arrondis, puis calculer exactement.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g4:mult:11}
Expliquer, avec un dessin de rectangle comme celui de ce chapitre,
pourquoi $15 \times 12$ peut se calculer comme
$15 \times 10 + 15 \times 2$. Puis inventer un autre découpage de
$15 \times 12$ qui fonctionne aussi.
\end{exercise}
