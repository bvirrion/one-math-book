\chapter{La division}\label{ch:g4:division}

L'année~3 divisait avec les tables et de petits restes
(\cref{ch:g3:division})~; ce chapitre installe la méthode écrite
complète --- la division posée --- qui traite n'importe quel nombre,
groupe de chiffres par groupe de chiffres.

\section{La méthode de la division posée}

\begin{method}[Division posée par un nombre à un chiffre]\label{met:g4:division:method}
Pour diviser $749$ par $6$~:
\begin{enumerate}
  \item on prend le chiffre le plus à gauche, $7$ (centaines)~: $6$
        rentre $1$~fois dans $7$~; on écrit $1$ au quotient, on
        soustrait $6$, reste $1$~;
  \item on \emph{abaisse} le chiffre suivant, $4$~: on divise
        maintenant $14$ (dizaines)~: $6 \times 2 = 12$~; on écrit
        $2$, reste $2$~;
  \item on abaisse le $9$~: on divise $29$~: $6 \times 4 = 24$~; on
        écrit $4$, reste $5$~;
  \item plus de chiffre~: le quotient est $124$, le reste $5$.
\end{enumerate}
Vérification~: $6 \times 124 + 5 = 744 + 5 = 749$, et $5 < 6$.
\end{method}

\begin{example}[Un zéro au quotient]\label{ex:g4:division:zero}
Diviser $618$ par $3$~: centaines, $6 \div 3 = 2$~; dizaines, on
abaisse le $1$~: $3$ rentre $0$~fois dans $1$ --- \emph{on écrit
le}~$0$~! --- reste $1$~; unités, on abaisse le $8$~:
$18 \div 3 = 6$. Quotient $206$, reste $0$. Oublier le zéro du
milieu (écrire $26$) est l'erreur classique~; la vérification
$3 \times 26 = 78 \neq 618$ la rattrape aussitôt.
\end{example}

\begin{example}[Estimer la taille du quotient]\label{ex:g4:division:size}
Avant de diviser $749$ par $6$, on encadre la réponse~:
$6 \times 100 = 600$ et $6 \times 200 = 1\,200$, donc le quotient
est entre $100$ et $200$ --- il aura trois chiffres. Cette
estimation d'une ligne évite la plupart des erreurs de place.
\end{example}

\section{Choisir ce que la question demande}

\begin{method}[Quotient, reste, ou quotient plus un]\label{met:g4:division:interpret}
Comme dans \cref{met:g3:division:interpret}, c'est l'histoire qui
décide~:
\begin{enumerate}
  \item «~combien de \dots complets~» ou «~combien chacun~»~: le
        quotient~;
  \item «~combien il reste~»~: le reste~;
  \item «~combien de contenants pour tout~»~: le quotient, plus un
        si le reste n'est pas zéro.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g4:division:problem}
$530$ livres doivent être emballés dans des cartons de $8$.
\[
530 = 8 \times 66 + 2 .
\]
Combien de cartons \emph{pleins}~? $66$. Combien de livres ne sont
pas dans un carton plein~? $2$. Combien de cartons pour emballer
\emph{tous} les livres~? $67$. Trois questions, une division.
\end{example}

\begin{example}[Partage équitable d'argent]\label{ex:g4:division:money}
Quatre amis partagent équitablement le coût d'un cadeau de $92$~:
$92 \div 4 = 23$ exactement (reste $0$). Chacun paie $23$. Quand le
reste est zéro, la division «~tombe juste~» et le partage est
parfaitement équitable.
\end{example}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:g4:division:1}
Encadrer chaque quotient comme dans \cref{ex:g4:division:size}
(entre quels deux nombres «~ronds~»~?), puis calculer la division
posée~: $96 \div 4$~; \quad $87 \div 5$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:division:2}
Calculer les divisions posées et vérifier chacune~:
$672 \div 4$~; \quad $925 \div 7$~; \quad $804 \div 6$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:division:3}
Attention au zéro (voir \cref{ex:g4:division:zero})~:
$816 \div 4$~; \quad $420 \div 7$~; \quad $2\,512 \div 5$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:division:4}
Écrire l'égalité de vérification ($a = b \times q + r$) pour~:
$85 \div 9$~; \quad $1\,000 \div 3$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:division:5}
$156$ œufs sont mis dans des boîtes de $6$. Combien de boîtes sont
remplies~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:division:6}
Un ruban de $250$~cm est découpé en morceaux de $8$~cm. Combien de
morceaux, et quelle longueur reste-t-il~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:division:7}
$375$ élèves vont à un spectacle en bus de $52$ places. Combien de
bus faut-il~? (Quel cas de \cref{met:g4:division:interpret} est-ce~?
Diviser par $52$ en utilisant les multiples~:
$52 \times 7 = 364$.)
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:division:8}
Trois amis se partagent équitablement $234$ billes. Combien de
billes chacun~? Vérifier avec une multiplication.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:division:9}
Trouver le dividende~: la division par $7$ donne le quotient $58$
et le reste $3$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g4:division:10}
Un bibliothécaire range $438$ livres, $9$ par étagère. Les étagères
viennent par bibliothèques de $6$ étagères. Combien d'étagères
faut-il~? Combien de bibliothèques doit-on acheter~? (Deux
divisions, toutes deux avec une question «~plus un~».)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g4:division:11}
Dans une division par $8$, le quotient est égal au reste. Quels
sont les dividendes possibles~? Les lister tous. (Le reste doit
rester strictement inférieur à $8$.)
\end{exercise}
