\chapter{Premières fractions}\label{ch:g4:fractions}

La moitié d'une pomme, un quart d'heure, les trois quarts de la
classe~: les fractions nomment les morceaux des choses. Ce chapitre
introduit l'écriture $\frac{a}{b}$, lit les fractions sur des
images et des droites graduées, et compare les plus simples.
(Les fractions grandissent dans \cref{ch:g6:fractions}.)

\section{Nommer les morceaux}

\begin{definition}[Fraction]\label{def:g4:fractions:def}
On découpe un tout en parts égales et on en prend certaines~:
\[
\frac{3}{4}
\quad
\begin{array}{l}
3 \text{~: combien de parts sont prises (le \emph{numérateur})~;}\\
4 \text{~: en combien de parts égales le tout est découpé (le
\emph{dénominateur}).}
\end{array}
\]
$\frac12$ est une \emph{moitié}, $\frac13$ un \emph{tiers},
$\frac14$ un \emph{quart}, $\frac{1}{10}$ un \emph{dixième}.
\end{definition}

\begin{example}[Fractions d'une collection]\label{ex:g4:fractions:collection}
$\frac13$ de $12$ billes~: on partage les $12$ billes en $3$ groupes
égaux de $4$~; un groupe est $\frac13$, donc $\frac13$ de $12$
est $4$ --- et $\frac23$ de $12$ est deux groupes, $8$.
\end{example}

\section{Fractions sur la droite graduée}

\begin{method}[Placer $\frac{a}{b}$]\label{met:g4:fractions:line}
On découpe chaque unité de la droite en $b$ pas égaux~; depuis $0$,
on avance de $a$ pas. Si $a$ est plus grand que $b$, le parcours
dépasse $1$~: les fractions peuvent être plus grandes qu'un
tout~!
\end{method}

\begin{example}[Entiers cachés dans les fractions]\label{ex:g4:fractions:whole}
$\frac44 = 1$ (quatre quarts font le tout)~; $\frac82 = 4$~;
$\frac{12}{3} = 4$. Et $\frac74$ est $1 + \frac34$~: un tout et
trois quarts de plus.
\end{example}

\section{Comparer et additionner des fractions simples}

\begin{proposition}[Même dénominateur]\label{prop:g4:fractions:compare}
Avec le même dénominateur, les parts ont la même taille --- on les
compte donc simplement~:
\[
\frac{2}{8} < \frac{5}{8},
\qquad
\frac{3}{8} + \frac{4}{8} = \frac{7}{8}.
\]
Comparer à un tout est aussi facile~: $\frac{5}{8} < 1 < \frac{9}{8}$
(moins, puis plus, que $8$ huitièmes).
\end{proposition}

\begin{proof}[Pourquoi, sur un dessin]
Colorier $3$ huitièmes puis $4$ huitièmes de plus de la même barre
colore $7$ huitièmes en tout.
\end{proof}

\begin{example}[Dénominateurs différents, même point]\label{ex:g4:fractions:equal}
Sur la droite graduée, $\frac12$, $\frac24$ et $\frac{5}{10}$
tombent tous au même point~: ce sont trois noms du même nombre.
Découper chaque moitié en deux donne des quarts~; en cinq, des
dixièmes. (La règle générale est dans \cref{ch:g6:fractions}.)
\end{example}

\begin{example}[Quarts d'heure]\label{ex:g4:fractions:hour}
Une heure a $60$ minutes, donc un quart d'heure est
$60 \div 4 = 15$ minutes, et trois quarts d'heure sont $45$
minutes. «~Deux heures et demie~» est $\frac12$ heure après deux
heures.
\end{example}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:g4:fractions:1}
Écrire la fraction montrée~: une pizza coupée en $6$ avec $5$ parts
restantes~; une tablette de chocolat de $8$ carrés avec $3$ mangés
(fraction mangée)~; un drapeau divisé en $3$ bandes verticales
égales, $1$ colorée.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:fractions:2}
Dessiner une barre découpée en $5$ parts égales et colorier
$\frac35$. Puis colorier $\frac{2}{5}$ d'une autre barre égale.
Quel coloriage est le plus grand~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:fractions:3}
Calculer~: $\frac12$ de $18$~; \quad $\frac14$ de $20$~; \quad
$\frac13$ de $21$~; \quad $\frac34$ de $20$ (trois groupes d'un
quart).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:fractions:4}
Placer sur une droite graduée découpée en tiers~: $\frac13$~; \
$\frac33$~; \ $\frac53$~; \ $2$. Quelle fraction tombe exactement
sur $2$~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:fractions:5}
Recopier et compléter avec $<$, $>$ ou $=$~:
\[
\frac38 \;?\; \frac68, \qquad
\frac55 \;?\; 1, \qquad
\frac74 \;?\; 1, \qquad
\frac12 \;?\; \frac24 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:fractions:6}
Calculer~: $\frac26 + \frac36$~; \quad $\frac58 - \frac28$~; \quad
$\frac14 + \frac24$~; \quad $1 - \frac13$ (combien de tiers font
$1$~?).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:fractions:7}
Écrire comme un entier plus une fraction plus petite que $1$~:
$\frac54$~; \quad $\frac73$~; \quad $\frac{9}{2}$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:fractions:8}
Combien de minutes font~: une demi-heure~; un quart d'heure~; trois
quarts d'heure~; $\frac13$ d'heure~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:fractions:9}
Dans une classe de $24$ élèves, $\frac14$ portent des lunettes.
Combien d'élèves portent des lunettes~? Combien n'en portent
pas~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g4:fractions:10}
Nina a mangé $\frac38$ d'une pizza et Sam $\frac48$ de la même
pizza. Quelle fraction ont-ils mangée ensemble~? Quelle fraction
reste-t-il~? Qui a mangé le plus~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g4:fractions:11}
Vrai ou faux, avec un dessin ou un exemple~: «~$\frac12$ d'une
petite pizza plus petit que $\frac14$ d'une très grande pizza est
possible~». Que doit-on toujours savoir avant de comparer deux
fractions de quelque chose~?
\end{exercise}
