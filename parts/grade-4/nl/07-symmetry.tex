\chapter{Symmetrie}\label{ch:g4:symmetry}

Vlinders, gezichten, sneeuwvlokken: de natuur houdt van figuren
waarvan de twee helften bij vouwen overeenkomen. Dit hoofdstuk
verkent symmetrie met vouwen en ruitjespapier --- constructies met de
geodriehoek wachten in \cref{ch:g6:symmetry}.

\section{Vouwen en laten kloppen}

\begin{definition}[Symmetrieas]\label{def:g4:symmetry:def}
Een lijn is een \emph{symmetrieas}\index{symmetrieas} van een figuur
als het vouwen van de figuur langs die lijn de twee helften
\emph{precies} op elkaar laat vallen. Een figuur kan geen as, \'{e}\'{e}n
as, of meerdere assen hebben.
\end{definition}

\begin{example}[Testen met calqueerpapier]\label{ex:g4:symmetry:trace}
Trek de figuur over, klap het calqueerpapier om langs de kandidaat-as,
en leg het terug: als de calque de figuur precies bedekt, is de as
echt. Het oog alleen wordt vaak bedrogen --- de diagonaal van een
rechthoek \emph{lijkt} een as, maar de vouwtest zegt nee (probeer
het!).
\end{example}

\section{Figuren afmaken op een raster}

\begin{method}[Spiegelbeeld van een figuur over een rasterlijn]\label{met:g4:symmetry:grid}
De as is een verticale (of horizontale) lijn van het raster.
\begin{enumerate}
  \item Neem elk hoekpunt van de figuur om de beurt;
  \item tel zijn afstand tot de as in vakjes;
  \item plaats het beeldhoekpunt op \emph{dezelfde} afstand aan de
        \emph{andere} kant, op dezelfde rij (of kolom);
  \item verbind de beeldhoekpunten in dezelfde volgorde, en
        controleer: het beeld is de spiegeltweeling van de figuur.
\end{enumerate}
\end{method}

\begin{example}[Zelfde vorm, zelfde grootte, omgeklapt]\label{ex:g4:symmetry:properties}
Het beeld heeft precies dezelfde lengtes en dezelfde hoeken als het
origineel: symmetrie kopieert de figuur --- maar klapt hem om, als
een linkerhand en een rechterhand. Als de originele letter F is, is
zijn spiegelbeeld een achterstevoren F, geen andere F.
\end{example}

\section{Assen van de gebruikelijke vormen}

\begin{example}[Assen tellen]\label{ex:g4:symmetry:shapes}
Door te vouwen:
\begin{itemize}
  \item een vierkant: $4$ assen (twee door de middens van overstaande
        zijden, twee langs de diagonalen);
  \item een rechthoek: $2$ assen (alleen middens van overstaande
        zijden!);
  \item een gelijkbenige driehoek: $1$; een gelijkzijdige
        driehoek: $3$;
  \item een cirkel: elke lijn door het middelpunt --- meer assen
        dan je kunt tellen.
\end{itemize}
\end{example}

\begin{example}[Symmetrie in het alfabet]\label{ex:g4:symmetry:letters}
A, M, T, U, V hebben een verticale as; B, C, D, E hebben een
horizontale; H, I, O, X hebben beide. Woorden van de juiste letters
kunnen ook symmetrisch zijn: MUM heeft een verticale as; OXO werkt
beide kanten op.
\end{example}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g4:symmetry:1}
Trek over en vouw: welke van deze hebben een symmetrieas --- een
vierkant; een (niet-vierkante) rechthoek; een ongelijkzijdige
driehoek (alle zijden verschillend); een cirkel?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:symmetry:2}
Teken de symmetrieassen van: een vierkant; een rechthoek; een
gelijkzijdige driehoek. Hoeveel voor elk?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:symmetry:3}
Gebruik de calqueerpapertest van \cref{ex:g4:symmetry:trace} om te
tonen dat de diagonaal van een $6 \times 4$-rechthoek \emph{geen}
symmetrieas is.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:symmetry:4}
Teken op ruitjespapier een verticale as en de letter L (drie vakjes
hoog, twee breed) op $2$ vakjes van de as. Construeer het
spiegelbeeld.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:symmetry:5}
Teken op ruitjespapier een horizontale as en een klein vlaggetje
(een $1 \times 1$-driehoek op een mast van $3$ vakjes) erboven.
Maak de figuur af door symmetrie onder de as.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:symmetry:6}
Een figuur raakt zijn symmetrieas in een punt $P$. Waar is het
beeld van $P$? Leg uit met het vouwplaatje.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:symmetry:7}
Sorteer de hoofdletters A, B, E, F, H, N, O, T: verticale as?
horizontale as? beide? geen van beide?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:symmetry:8}
Een driehoek heeft zijden $5$ cm, $5$ cm en $3$ cm. Heeft hij een
symmetrieas? Welke zijden wisselt de vouw om?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g4:symmetry:9}
Teken op ruitjespapier een figuur van jezelf met \emph{precies twee}
symmetrieassen, en teken beide assen. (Hint: denk aan een
rechthoek, of aan een kruis.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g4:symmetry:10}
De helft van een symmetrische tekening is gegeven: de linkerhelft van
een huis (een vierkante muur en half een driehoekig dak), met een
verticale as langs de rechterrand. Beschrijf of teken het complete
huis. Hoe verhouden de lengtes van de voltooide rechterhelft zich tot
de linkerhelft?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g4:symmetry:11}
Aya beweert: ``elke driehoek met twee gelijke zijden heeft een
symmetrieas, en elke driehoek met een symmetrieas heeft twee gelijke
zijden.'' Test haar dubbele bewering op tekeningen: een
gelijkbenige driehoek, een gelijkzijdige, een ongelijkzijdige.
Ondersteunt het vouwen haar?
\end{exercise}
