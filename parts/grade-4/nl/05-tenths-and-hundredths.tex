\chapter{Tienden en honderdsten}\label{ch:g4:decimals}

Tussen $3$ en $4$ zwijgen de hele getallen --- maar een loper kan $3$
meter en een beetje voor liggen. De eenheid in tien snijden, dan in
honderd, geeft de \emph{decimale \omterm{def:g4:fractions:def}{breuken}}, en een prachtige
verkorting: de \omterm{def:g4:decimals:point}{decimale punt}. (Decimale getallen krijgen hun volle
hoofdstuk in \cref{ch:g5:decimals}.)

\section{Decimale breuken}

\begin{definition}[Tienden en honderdsten]\label{def:g4:decimals:def}
Een eenheid in $10$ gelijke delen snijden geeft \emph{tienden},
$\frac{1}{10}$; elk tiende opnieuw in $10$ snijden geeft
\emph{honderdsten}, $\frac{1}{100}$. Dus
\[
1 = \frac{10}{10} = \frac{100}{100},
\qquad
\frac{1}{10} = \frac{10}{100}.
\]
\end{definition}

\begin{example}\label{ex:g4:decimals:count}
$3$ kolommen en $7$ extra vakjes van het honderdvierkant arceren
\[
\frac{3}{10} + \frac{7}{100} = \frac{30}{100} + \frac{7}{100}
= \frac{37}{100}
\]
van het vierkant --- zevenendertig honderdsten.
\end{example}

\section{De decimale punt}

\begin{definition}[Decimale notatie]\label{def:g4:decimals:point}
Het getal $2 + \frac{3}{10} + \frac{7}{100}$ wordt geschreven
\[
2.37
\]
--- de \emph{decimale punt}\index{decimale punt} scheidt het hele deel ($2$) van het
decimale deel; het eerste cijfer na de punt telt de tienden, het
tweede de honderdsten. Lezen: ``twee komma drie zeven'', of ``twee
en zevenendertig honderdsten''.
\end{definition}

\begin{example}\label{ex:g4:decimals:convert}
\[
\frac{7}{10} = 0.7, \qquad
\frac{43}{100} = 0.43, \qquad
\frac{5}{100} = 0.05, \qquad
3 + \frac{4}{10} = 3.4 .
\]
Let op $\frac{5}{100}$: vijf \emph{honderdsten} hebben een $0$ bij
de tienden nodig --- $0.05$, niet $0.5$.
\end{example}

\begin{example}[Geld is honderdsten]\label{ex:g4:decimals:money}
E\'{e}n cent is \'{e}\'{e}n honderdste van \'{e}\'{e}n euro (of dollar, of
pond): een prijs van $4.85$ betekent $4$ hele munten en $85$
honderdsten. Geld is de \omterm{def:g4:decimals:point}{decimale punt} die je al elke dag gebruikt:
$10$ cent $= 0.10$; twee euro vijftig $= 2.50$; en $0.05$ is een
vijfcentstuk, niet vijftig!
\end{example}

\section{Vergelijken met tienden}

\begin{method}[Decimalen vergelijken (eerste kennismaking)]\label{met:g4:decimals:compare}
\begin{enumerate}
  \item Vergelijk eerst de hele delen: $3.2 > 2.9$.
  \item Zelfde hele delen: vergelijk de tienden, dan de
        honderdsten: $2.37 < 2.5$ omdat $3$ tienden $< 5$ tienden.
  \item Beide met twee decimalen schrijven helpt: $2.5 = 2.50$, en
        $37 < 50$ honderdsten.
\end{enumerate}
De valkuil: $2.37$ heeft meer cijfers dan $2.5$, maar is
\emph{kleiner} --- meer cijfers betekent niet groter!
\end{method}

\begin{example}\label{ex:g4:decimals:compare}
Zet $0.4$, $0.35$ en $0.09$ op volgorde: met honderdsten, $0.40$,
$0.35$, $0.09$; dus
\[
0.09 < 0.35 < 0.4 .
\]
\end{example}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g4:decimals:1}
Op een honderdvierkant, hoeveel is gearceerd als men arceert: $5$
kolommen; $2$ kolommen en $3$ vakjes; $45$ vakjes? Antwoord met
\omterm{def:g4:fractions:def}{breuken}.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:decimals:2}
Schrijf met een \omterm{def:g4:decimals:point}{decimale punt}:
$\frac{9}{10}$; \quad $\frac{27}{100}$; \quad $\frac{3}{100}$; \quad
$5 + \frac{6}{10}$; \quad $\frac{60}{100}$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:decimals:3}
Schrijf als \omterm{def:g4:fractions:def}{breuk} (tienden of honderdsten):
$0.3$; \quad $0.51$; \quad $0.07$; \quad $1.9$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:decimals:4}
In $6.58$: wat telt het cijfer $5$? Het cijfer $8$? Ontbind het
getal zoals in \cref{def:g4:decimals:point}.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:decimals:5}
Plaats op een getallenlijn in tienden van $4$ tot $5$:
$4.2$; \ $4.75$ (tussen welke twee strepen?); \ $4.9$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:decimals:6}
Schrijf over en vul in met $<$, $>$ of $=$:
\[
0.6 \;?\; 0.60, \qquad
2.8 \;?\; 2.75, \qquad
0.09 \;?\; 0.1, \qquad
5.3 \;?\; 5.13 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:decimals:7}
Zet op volgorde van klein naar groot: $1.05$; \ $1.5$; \ $0.95$; \
$1.45$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:decimals:8}
Schrijf in decimale notatie: drie euro en vijf cent; twaalf euro en
vijftig cent; negenennegentig cent. Zet daarna de drie prijzen op
volgorde.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:decimals:9}
Welke decimale getallen met \'{e}\'{e}n decimaal cijfer liggen strikt
tussen $3$ en $4$? Noem ze allemaal. Hoeveel zijn er?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g4:decimals:10}
Tom zegt: ``$0.25$ is groter dan $0.5$ omdat $25$ groter is dan
$5$''. Laat hem zijn fout zien met het honderdvierkant (hoeveel
vakjes arceert elk?).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g4:decimals:11}
Drie sprongen van een loper maten $3.6$ m, $3.58$ m en $3.65$ m.
Zet de sprongen op volgorde van kort naar lang. Welke twee
verschillen precies $\frac{5}{100}$ meter?
\end{exercise}
