\chapter{Maten en omtrek}\label{ch:g4:measure}

In jaar~3 maten we met linialen, weegschalen en klokken
(\cref{ch:g3:measure}); dit hoofdstuk voegt de eenhedentabel toe ---
de machine die alles omzet --- en twee manieren om een platte vorm te
meten: de lengte van zijn rand (\omterm{def:g4:measure:perimeter}{omtrek}) en de ruimte erin
(\omterm{def:g4:measure:area}{oppervlakte}, voorlopig geteld in vierkantjes).

\section{De eenhedentabel}

\begin{method}[Omzetten met de tabel]\label{met:g4:measure:table}
Voor lengtes is elke kolom tien van de volgende waard:
\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{c|c|c|c|c|c|c}
km & hm & dam & m & dm & cm & mm \\
\hline
 & & & $3$ & $4$ & $0$ & \\
\end{tabular}
\end{center}
\begin{enumerate}
  \item Schrijf het getal met \'{e}\'{e}n cijfer per kolom, het
        eenhedencijfer van de maat in de kolom van zijn eenheid
        (hier $3.4$ m: de $3$ onder m);
  \item lees het getal in de nieuwe eenheid: $3.4$ m $= 340$ cm
        (vul lege kolommen met nullen);
  \item \omterm{def:g2:measure:units}{massa}'s (kg\,\dots\,g) en \omterm{def:g2:measure:units}{inhouden} (L\,\dots\,cL) werken
        op dezelfde manier.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g4:measure:convert}
$5$ km $= 5\,000$ m; \quad $270$ cm $= 2.7$ m (de $2$ landt onder
m); \quad $3$ kg $250$ g $= 3\,250$ g; \quad
$2.5$ L $= 250$ cL.
\end{example}

\section{Omtrek}

\begin{definition}[Omtrek]\label{def:g4:measure:perimeter}
De \emph{omtrek}\index{omtrek} van een \omterm{def:g4:geometry:polygon}{veelhoek} is de totale lengte
van zijn rand: tel de lengtes van al zijn zijden op (in dezelfde
eenheid!). Snelwegen voor de klassiekers:
\[
\text{rechthoek: } P = 2 \times (L + b),
\qquad
\text{vierkant: } P = 4 \times z .
\]
\end{definition}

\begin{example}\label{ex:g4:measure:perimeter}
Een rechthoekige tuin $12$ m bij $7$ m:
\[
P = 2 \times (12 + 7) = 2 \times 19 = 38 \text{ m}
\]
hek. Een vierkant fotolijstje van zijde $18$ cm:
$P = 4 \times 18 = 72$ cm.
\end{example}

\section{Oppervlakte: vierkantjes tellen}

\begin{definition}[Oppervlakte door te tellen]\label{def:g4:measure:area}
De \emph{oppervlakte}\index{oppervlakte} van een figuur getekend op
een raster is het aantal rastervierkantjes dat hij bedekt. Twee
half-vierkantjes tellen als \'{e}\'{e}n. Oppervlakte beantwoordt ``hoeveel
vlak'', \omterm{def:g4:measure:perimeter}{omtrek} beantwoordt ``hoe lang is de rand'' --- twee
verschillende vragen!
\end{definition}

\begin{tikzpicture}[scale=0.55]

  \draw[very thick, omDef, fill=omDef!15] (1,1) rectangle (5,3);
  \node[font=\small, omDef] at (3,2) {$8$ vierkantjes};
  \draw[very thick, omThm, fill=omThm!15]
    (7,1) -- (11,1) -- (11,2) -- (9,2) -- (9,3) -- (7,3) -- cycle;
  \node[font=\small, omThm] at (8.4,1.5) {$6$ vierkantjes};
\end{tikzpicture}

{\small Twee figuren op het raster: \omterm{def:g4:measure:area}{oppervlaktes} $8$ en $6$
vierkantjes. Loop nu de randen en tel de eenheden: $12$ voor beide!
Zelfde \omterm{def:g4:measure:perimeter}{omtrek}, verschillende \omterm{def:g4:measure:area}{oppervlaktes}.}
\end{omfigure}

\begin{example}[Half-vierkantjes]\label{ex:g4:measure:half}
Een rechthoekige driehoek op het raster met benen $4$ en $2$ bedekt
$3$ volle vierkantjes en $2$ half-vierkantjes\,\dots\ tel zorgvuldig:
zijn \omterm{def:g4:measure:area}{oppervlakte} is de \omterm{def:g3:division:half}{helft} van de $4 \times 2$-rechthoek, d.w.z.\
$4$ vierkantjes. Een rechthoek langs de diagonaal snijden geeft
altijd twee driehoeken van gelijke \omterm{def:g4:measure:area}{oppervlakte}.
\end{example}

\begin{example}[Zelfde oppervlakte, verschillende omtrekken]\label{ex:g4:measure:samearea}
Een $4 \times 4$-vierkant en een $8 \times 2$-rechthoek bedekken
beide $16$ vierkantjes --- dezelfde \omterm{def:g4:measure:area}{oppervlakte}. Randen: $16$
eenheden voor het vierkant, $20$ voor de rechthoek. Geen van beide
maten bepaalt de andere: een lange dunne vorm heeft veel rand voor
weinig vlak.
\end{example}

\section{Duur}

\begin{method}[Tijden optellen over het uur]\label{met:g4:measure:time}
Om duren op te tellen of af te trekken, werk in sprongen door de
hele uren (onthou $1$ u $= 60$ min):
\[
9\text{ u }40 + 35 \text{ min}:
\quad 9\text{ u }40 \xrightarrow{+20} 10\text{ u }00
\xrightarrow{+15} 10\text{ u }15 .
\]
Tel nooit minuten voorbij $60$ zonder om te zetten: $40 + 35 = 75$
min $= 1$ u $15$ min.
\end{method}

\begin{example}\label{ex:g4:measure:timetable}
De trein vertrekt om $8{:}47$ en komt aan om $11{:}05$. Duur:
$8{:}47 \to 9{:}00$ is $13$ min; $9{:}00 \to 11{:}00$ is $2$ u;
$11{:}00 \to 11{:}05$ is $5$ min. Totaal: $2$ u $18$ min.
\end{example}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g4:measure:1}
Zet om met de eenhedentabel: $7$ m in cm; $3.2$ km in m; $85$ mm in
cm; $4\,500$ m in km.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:measure:2}
Zet om: $2$ kg in g; $1\,800$ g in kg en g; $3.5$ L in cL;
$25$ cL in L.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:measure:3}
Bereken de \omterm{def:g4:measure:perimeter}{omtrek}: een rechthoek $9$ cm $\times$ $5$ cm; een
vierkant van zijde $7.5$ cm; een driehoek met zijden $6$ cm, $8$ cm
en $10$ cm.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:measure:4}
Een rechthoek heeft \omterm{def:g4:measure:perimeter}{omtrek} $26$ cm en lengte $8$ cm. Vind de
breedte (vind eerst lengte $+$ breedte).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:measure:5}
Teken op ruitjespapier een L-vormige figuur die precies $10$
vierkantjes bedekt, en tel de eenheden van zijn \omterm{def:g4:measure:perimeter}{omtrek}.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:measure:6}
Teken twee verschillende rechthoeken van \omterm{def:g4:measure:area}{oppervlakte} $12$
vierkantjes. Bereken de \omterm{def:g4:measure:perimeter}{omtrek} van elk. Welke is ``ronder'', welke
is ``dunner''?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:measure:7}
Een rechthoekige driehoek is op het raster getekend met benen $6$ en
$3$. Wat is zijn \omterm{def:g4:measure:area}{oppervlakte} in vierkantjes? (Gebruik
\cref{ex:g4:measure:half}.)
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:measure:8}
Bereken de duren: van $10{:}20$ tot $12{:}45$; van $7{:}35$ tot
$9{:}10$. Een film duurt $1$ u $50$ en begint om $20{:}30$: hoe
laat eindigt hij?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:measure:9}
Kim loopt $3$ rondjes van een rechthoekig veld $60$ m bij $45$ m.
Hoeveel meter loopt ze?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g4:measure:10}
Een boer wil een vierkant veld van zijde $85$ m omheinen, en een
poort van $4$ m zonder hek laten. Hoeveel meter hek moet hij kopen?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g4:measure:11}
Waar of niet waar, met rastervoorbeelden: ``als twee figuren dezelfde
\omterm{def:g4:measure:perimeter}{omtrek} hebben, hebben ze dezelfde \omterm{def:g4:measure:area}{oppervlakte}''; ``als twee
figuren dezelfde \omterm{def:g4:measure:area}{oppervlakte} hebben, hebben ze dezelfde \omterm{def:g4:measure:perimeter}{omtrek}''.
(\Cref{ex:g4:measure:samearea} en de figuur van dit hoofdstuk
helpen.)
\end{exercise}
