\chapter{Lijnen en veelhoeken}\label{ch:g4:geometry}

Met een liniaal, een geodriehoek en een passer wordt meetkunde een
ambacht. Dit hoofdstuk leert de twee speciale posities van lijnen ---
loodrecht en evenwijdig --- op het niveau van het gereedschap, en
maakt een tour door de veelhoekenfamilie. (De theorie van deze
begrippen komt in \cref{ch:g6:lines}.)

\section{Loodrecht en evenwijdig}

\begin{definition}[Loodrecht, evenwijdig]\label{def:g4:geometry:def}
Twee lijnen zijn \emph{loodrecht}\index{loodrechte lijnen} als ze
elkaar kruisen onder een rechte hoek (gecontroleerd met de
geodriehoek, gemarkeerd met een klein vierkantje). Twee lijnen zijn
\emph{evenwijdig}\index{evenwijdige lijnen} als ze in dezelfde
richting lopen en elkaar nooit kruisen --- als de twee rails van een
spoor, altijd op dezelfde afstand van elkaar.
\end{definition}

\begin{method}[Tekenen met de geodriehoek]\label{met:g4:geometry:draw}
\emph{Loodrecht op een lijn $d$ door een punt $P$}: leg \'{e}\'{e}n zijde
van de rechte hoek langs $d$, schuif tot de andere zijde $P$ bereikt,
teken.

\emph{Evenwijdig aan $d$ door $P$}: leg de zijde van de geodriehoek
langs $d$, druk de liniaal tegen een andere zijde als rail, schuif de
geodriehoek langs de liniaal tot $P$, teken.
\end{method}

\begin{example}[Ze herkennen in figuren]\label{ex:g4:geometry:spot}
In een rechthoek is elke zijde loodrecht op zijn twee buren en
evenwijdig aan de overstaande zijde. Ruitjespapier is een kant-en-klaar
netwerk van evenwijdige en loodrechte lijnen --- daarom tekent men er
zo comfortabel op.
\end{example}

\section{Veelhoeken}

\begin{definition}[Veelhoek]\label{def:g4:geometry:polygon}
Een \emph{veelhoek}\index{veelhoek} is een gesloten figuur waarvan de
rand uit rechte segmenten bestaat, zijn \emph{zijden}; de
hoekpunten zijn de \emph{hoeken} (of \emph{vertices}). Veelhoeken
worden benoemd naar hun aantal zijden: \emph{driehoek} ($3$),
\emph{vierhoek} ($4$), \emph{vijfhoek} ($5$), \emph{zeshoek}
($6$), \emph{achthoek} ($8$).
\end{definition}

\begin{example}[Een veelhoek beschrijven]\label{ex:g4:geometry:describe}
Een nuttig spel: beschrijf een figuur zo precies dat een klasgenoot
hem kan natekenen zonder hem te zien. ``Een vierhoek $ABCD$ met
$AB = 5$ cm, de hoek in $A$ recht, $AD = 3$ cm\,\dots'' ---
woordenschat is wat de boodschap doet werken.
\end{example}

\section{De cirkel en de passer}

\begin{definition}[Cirkel]\label{def:g4:geometry:circle}
De \emph{cirkel} met middelpunt $O$ en straal $3$ cm is de
verzameling van alle punten op precies $3$ cm van $O$. De passer
tekent hem: punt op $O$, opening $3$ cm, draaien. Een
\emph{diameter} is een segment door het middelpunt dat twee punten
van de cirkel verbindt; hij meet tweemaal de straal.
\end{definition}

\begin{example}[De passer als afstandsbehouder]\label{ex:g4:geometry:compass}
De passer doet meer dan cirkels tekenen: hij \emph{verplaatst
afstanden}. Om alle punten op $4$ cm van een punt $A$ te markeren,
teken de cirkel met middelpunt $A$, straal $4$ cm; om twee segmenten
zonder liniaal te vergelijken, open de passer op de ene en test hem
op de andere.
\end{example}

\begin{method}[Een figuur op een raster natekenen]\label{met:g4:geometry:reproduce}
\begin{enumerate}
  \item Kies een startpunt en zijn positie op het lege raster;
  \item loop de figuur hoekpunt voor hoekpunt af, tel de
        rasterstappen (``$3$ rechts, $2$ omhoog'');
  \item verbind de hoekpunten op volgorde, en vergelijk met het
        model: dezelfde lengtes, dezelfde richtingen.
\end{enumerate}
\end{method}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g4:geometry:1}
Teken een lijn $d$ en een punt $P$ niet op $d$. Construeer de
loodlijn op $d$ door $P$, dan de evenwijdige aan $d$ door $P$ (met
de liniaal-railmethode). Markeer de rechte hoek.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:geometry:2}
Kijk naar de hoofdletters E, H, N, T, Z. Zoek in elk een paar
evenwijdige streken en een paar loodrechte streken wanneer die
bestaan.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:geometry:3}
Noem elke veelhoek: $6$ zijden; $4$ zijden; $3$ zijden; $8$ zijden;
$5$ zijden. Hoeveel \emph{hoekpunten} heeft elk?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:geometry:4}
Teken een willekeurige vijfhoek (niet per se regelmatig) en benoem
zijn hoekpunten $A$, $B$, $C$, $D$, $E$. Hoeveel zijden heeft hij?
Teken zijn diagonalen vanuit $A$ (segmenten van $A$ naar
niet-buurhoekpunten): hoeveel zijn er?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:geometry:5}
Teken een cirkel met middelpunt $O$ en straal $4$ cm; een diameter
$[AB]$; en een punt $M$ van de cirkel niet op $[AB]$. Geef de
lengtes $OA$, $OM$ en $AB$ zonder te meten.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:geometry:6}
Met de passer als afstandsbehouder
(\cref{ex:g4:geometry:compass}), vergelijk de zijden van een
driehoek getekend door een klasgenoot en beslis of hij twee gelijke
zijden heeft.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:geometry:7}
Nateken op ruitjespapier dit pad en sluit het tot een veelhoek:
begin bij een punt $A$; ga $4$ rechts; $2$ omhoog; $3$ links; $2$
omhoog; dan terug naar $A$ in een rechte lijn. Hoe heet de veelhoek
die je krijgt (tel de zijden)?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:geometry:8}
Waar of niet waar? Leg uit met een tekening.
``Twee loodrechte lijnen kruisen elkaar altijd.''
``Twee evenwijdige lijnen kruisen elkaar nooit.''
``Een lijn kan evenwijdig zijn aan \'{e}\'{e}n lijn en loodrecht op een
andere.''
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g4:geometry:9}
Teken een lijn $d$, dan een lijn $e$ loodrecht op $d$, dan een lijn
$f$ loodrecht op $e$. Kijk naar $d$ en $f$: wat val je op? (Deze
waarneming wordt een stelling in \cref{ch:g6:lines}.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g4:geometry:10}
Beschrijf de figuur hieronder in woorden voor een klasgenoot, zonder
hem te laten zien: een vierkant van zijde $4$ cm, met een cirkel van
straal $2$ cm gecentreerd in het midden van het vierkant (raakt de
vier zijden). Schrijf de beschrijving als genummerde instructies.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g4:geometry:11}
Hoeveel diagonalen heeft een vierhoek? Een vijfhoek? Een zeshoek?
Teken de drie gevallen, tel zorgvuldig, en beschrijf het patroon dat
je ziet.
\end{exercise}
