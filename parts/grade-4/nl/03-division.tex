\chapter{Delen}\label{ch:g4:division}

In jaar~3 deelden we met de tafels en kleine \omterm{def:g3:division:remainder}{resten}
(\cref{ch:g3:division}); dit hoofdstuk installeert de volledige
schriftelijke methode --- staartdelen --- die elk getal aankan,
cijfergroep voor cijfergroep.

\section{De staartdelingsmethode}

\begin{method}[Staartdelen door een eencijferig getal]\label{met:g4:division:method}
Om $749$ te delen door $6$:
\begin{enumerate}
  \item neem het linkerste cijfer, $7$ (honderdtallen): $6$ past $1$
        keer in $7$; schrijf $1$ in het \omterm{def:g3:division:remainder}{quoti\"ent}, trek $6$ af,
        \omterm{def:g3:division:remainder}{rest} $1$;
  \item \emph{haal} het volgende cijfer, $4$, \emph{naar beneden}:
        deel nu $14$ (tientallen): $6 \times 2 = 12$; schrijf $2$,
        \omterm{def:g3:division:remainder}{rest} $2$;
  \item haal de $9$ naar beneden: deel $29$: $6 \times 4 = 24$;
        schrijf $4$, \omterm{def:g3:division:remainder}{rest} $5$;
  \item geen cijfer meer over: het \omterm{def:g3:division:remainder}{quoti\"ent} is $124$, de \omterm{def:g3:division:remainder}{rest} $5$.
\end{enumerate}
Controle: $6 \times 124 + 5 = 744 + 5 = 749$, en $5 < 6$.
\end{method}

\begin{example}[Een nul in het quoti\"ent]\label{ex:g4:division:zero}
Deel $618$ door $3$: honderdtallen, $6 \div 3 = 2$; tientallen,
haal de $1$ naar beneden: $3$ past $0$ keer in $1$ ---
\emph{schrijf de} $0$! --- \omterm{def:g3:division:remainder}{rest} $1$; eenheden, haal de $8$ naar
beneden: $18 \div 3 = 6$. \omterm{def:g3:division:remainder}{Quoti\"ent} $206$, \omterm{def:g3:division:remainder}{rest} $0$. De middelste
\omterm{def:g1:counting:zero}{nul} vergeten (schrijven $26$) is de klassieke fout; de controle
$3 \times 26 = 78 \neq 618$ vangt hem meteen.
\end{example}

\begin{example}[De grootte van het quoti\"ent schatten]\label{ex:g4:division:size}
V\'{o}\'{o}r $749$ te delen door $6$, kader het antwoord:
$6 \times 100 = 600$ en $6 \times 200 = 1\,200$, dus het \omterm{def:g3:division:remainder}{quoti\"ent}
ligt tussen $100$ en $200$ --- het heeft drie cijfers. Deze
\'{e}\'{e}nregel-schatting voorkomt de meeste fouten met verkeerde
plaatsen.
\end{example}

\section{Kiezen wat de vraag vraagt}

\begin{method}[Quoti\"ent, rest, of quoti\"ent plus \'{e}\'{e}n]\label{met:g4:division:interpret}
Zoals in \cref{met:g3:division:interpret}, beslist het verhaal:
\begin{enumerate}
  \item ``hoeveel volle \dots'' of ``hoeveel elk'': het \omterm{def:g3:division:remainder}{quoti\"ent};
  \item ``hoeveel over'': de \omterm{def:g3:division:remainder}{rest};
  \item ``hoeveel containers nodig voor alles'': het \omterm{def:g3:division:remainder}{quoti\"ent},
        plus \'{e}\'{e}n als de \omterm{def:g3:division:remainder}{rest} niet \omterm{def:g1:counting:zero}{nul} is.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g4:division:problem}
$530$ boeken moeten in dozen van $8$ gepakt worden.
\[
530 = 8 \times 66 + 2 .
\]
Hoeveel \emph{volle} dozen? $66$. Hoeveel boeken zitten niet in een
volle doos? $2$. Hoeveel dozen om \emph{alle} boeken te pakken?
$67$. Drie vragen, \'{e}\'{e}n deling.
\end{example}

\begin{example}[Eerlijk delen van geld]\label{ex:g4:division:money}
Vier vrienden delen de kosten van een cadeau van $92$ even:
$92 \div 4 = 23$ precies (\omterm{def:g3:division:remainder}{rest} $0$). Elk betaalt $23$. Als de \omterm{def:g3:division:remainder}{rest}
\omterm{def:g1:counting:zero}{nul} is, ``komt de deling precies uit'' en is het delen perfect
eerlijk.
\end{example}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g4:division:1}
Kader elk \omterm{def:g3:division:remainder}{quoti\"ent} zoals in \cref{ex:g4:division:size} (tussen
welke twee ``ronde'' getallen?), reken dan de staartdeling uit:
$96 \div 4$; \quad $87 \div 5$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:division:2}
Reken de staartdelingen uit en controleer elk:
$672 \div 4$; \quad $925 \div 7$; \quad $804 \div 6$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:division:3}
Let op de \omterm{def:g1:counting:zero}{nul} (zie \cref{ex:g4:division:zero}):
$816 \div 4$; \quad $420 \div 7$; \quad $2\,512 \div 5$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:division:4}
Schrijf de controle-gelijkheid ($a = b \times q + r$) voor:
$85 \div 9$; \quad $1\,000 \div 3$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:division:5}
$156$ eieren worden in dozen van $6$ gedaan. Hoeveel dozen worden
gevuld?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:division:6}
Een lint van $250$ cm wordt in stukken van $8$ cm geknipt. Hoeveel
stukken, en welke lengte blijft over?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:division:7}
$375$ leerlingen gaan naar een voorstelling in bussen van $52$
zitplaatsen. Hoeveel bussen zijn nodig? (Welk geval van
\cref{met:g4:division:interpret} is dit? Deel door $52$ met
veelvouden: $52 \times 7 = 364$.)
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:division:8}
Drie vrienden delen $234$ knikkers even. Hoeveel knikkers elk?
Controleer met een vermenigvuldiging.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:division:9}
Vind het deeltal: de deling door $7$ geeft \omterm{def:g3:division:remainder}{quoti\"ent} $58$ en \omterm{def:g3:division:remainder}{rest}
$3$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g4:division:10}
Een bibliothecaris zet $438$ boeken in de kast, $9$ per plank.
Planken komen in kasten van $6$ planken. Hoeveel planken zijn nodig?
Hoeveel kasten moeten gekocht worden? (Twee delingen, beide met een
``plus \'{e}\'{e}n''-vraag.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g4:division:11}
Bij een deling door $8$ is het \omterm{def:g3:division:remainder}{quoti\"ent} gelijk aan de \omterm{def:g3:division:remainder}{rest}. Wat
zijn de mogelijke deeltallen? Noem ze allemaal. (De \omterm{def:g3:division:remainder}{rest} moet onder
$8$ blijven.)
\end{exercise}
