\chapter{Grote getallen}\label{ch:g4:numbers}

Duizendtallen waren het plafond van \cref{ch:g3:numbers}; nu stijgt
het plafond tot \emph{miljoenen}. De truc die enorme getallen
leesbaar maakt, is hun cijfers in groepen van drie te knippen --- de
klassen.

\section{Klassen van drie cijfers}

\begin{definition}[Klassen]\label{def:g4:numbers:classes}
Vanaf rechts worden de cijfers van een getal gegroepeerd in
\emph{klassen} van drie: de klasse van de \emph{eenheden}, de klasse
van de \emph{duizendtallen}, de klasse van de \emph{miljoenen}. Elke
klasse heeft haar eigen eenheden, tientallen en honderdtallen.
Bijvoorbeeld
\[
6\,309\,452 = 6 \text{ miljoenen} + 309 \text{ duizendtallen}
+ 452 \text{ eenheden},
\]
gelezen als ``zes miljoen driehonderdnegenduizend
vierhonderdtwee\"{e}nvijftig''.
\end{definition}

\begin{example}[Schrijven met de klassen]\label{ex:g4:numbers:write}
``Twaalf miljoen veertigduizend vijfhonderd'' heeft $12$ in de
miljoenenklasse, $040$ in de duizendtallenklasse, $500$ in de
eenhedenklasse:
\[
12\,040\,500 .
\]
De nullen van $040$ zijn essentieel: elke klasse (behalve de
linkerste) moet haar drie cijfers tonen.
\end{example}

\begin{example}[Hoeveel duizendtallen?]\label{ex:g4:numbers:contained}
Het getal $6\,309\,452$ \emph{bevat} $6\,309$ duizendtallen (alles
links van de eenhedenklasse): $6\,309\,452 = 6\,309 \times
1\,000 + 452$. Vragen ``hoeveel duizendtallen'' vraagt niet om \'{e}\'{e}n
cijfer, maar om een heel voorvoegsel van het getal.
\end{example}

\section{Vergelijken en afronden}

\begin{method}[Grote getallen vergelijken]\label{met:g4:numbers:compare}
Precies zoals in \cref{met:g3:numbers:compare}, met de klassen als
snelweg: meer cijfers wint; zelfde lengte, vergelijk van links ---
klasse voor klasse, dan cijfer voor cijfer.
\end{method}

\begin{example}\label{ex:g4:numbers:compare}
$2\,145\,890$ en $2\,148\,540$: miljoenen gelijk ($2$),
duizendtallenklasse $145 < 148$, dus
\[
2\,145\,890 < 2\,148\,540 .
\]
\end{example}

\begin{definition}[Afronden]\label{def:g4:numbers:round}
Om een getal te \emph{afronden}\index{afronden} op het dichtstbijzijnde
duizendtal (of tiental, honderdtal, miljoen\,\dots), vervang het door
het dichtstbijzijnde veelvoud van $1\,000$. Als het getal precies
halverwege ligt, rond naar boven af. In de praktijk: kijk naar het
cijfer net onder de afrondplaats --- $5$ of meer rondt naar boven
af, $4$ of minder naar beneden.
\end{definition}

\begin{example}\label{ex:g4:numbers:round}
$37\,614$ afgerond op het dichtstbijzijnde duizendtal: het
honderdtallencijfer is $6$, dus naar boven: $38\,000$. Afgerond op
het dichtstbijzijnde honderdtal: $37\,600$. Afgerond op het
dichtstbijzijnde tienduizendtal: $40\,000$.
\end{example}

\begin{example}[Ordes van grootte in de wereld]\label{ex:g4:numbers:world}
Afronden is hoe grote getallen echt gebruikt worden: een land van
$67\,842\,591$ inwoners is ``ongeveer $68$ miljoen''; een stad van
$214\,376$ mensen is ``ongeveer $200\,000$''. Het exacte cijfer
verandert toch elke dag --- het afgeronde draagt de nuttige
informatie.
\end{example}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g4:numbers:1}
Schrijf in cijfers, met de klassen gescheiden: ``drie miljoen
vijfhonderdtwintigduizend''; ``zeven miljoen acht''; ``tweehonderdveertigduizend zestig''.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:numbers:2}
Schrijf in woorden: $5\,204\,300$; \quad $10\,050\,000$; \quad
$999\,999$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:numbers:3}
In $8\,472\,156$: welk cijfer is de tientallen van de duizendtallen?
De honderdtallen van de miljoenenklasse? Hoeveel \emph{duizendtallen}
bevat het getal (zie \cref{ex:g4:numbers:contained})?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:numbers:4}
Schrijf over en vul in met $<$ of $>$:
\[
980\,000 \;?\; 1\,020\,000, \qquad
3\,456\,000 \;?\; 3\,465\,000, \qquad
7\,090\,050 \;?\; 7\,090\,500 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:numbers:5}
Zet op volgorde van klein naar groot: $2\,300\,000$; \ $980\,500$; \
$2\,030\,000$; \ $2\,299\,999$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:numbers:6}
Rond $63\,782$ af: op het dichtstbijzijnde tiental; op het
dichtstbijzijnde honderdtal; op het dichtstbijzijnde duizendtal; op
het dichtstbijzijnde tienduizendtal.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:numbers:7}
Rond af op het dichtstbijzijnde miljoen: $4\,721\,000$; \quad
$6\,499\,999$; \quad $9\,500\,000$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:numbers:8}
Tel met stappen: drie stappen van $1\,000$ vanaf $998\,500$; twee
stappen van $100\,000$ vanaf $950\,000$. Wat gebeurt er met de
klassen als je een miljoen oversteekt?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:numbers:9}
Een stadion houdt $81\,338$ toeschouwers. Een krant schrijft ``meer
dan $80\,000$ fans''; een radio zegt ``ongeveer $81\,000$''.
Hebben beide gelijk? Op welke plaats heeft elk afgerond?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g4:numbers:10}
Met elk van de cijfers $1$ tot $7$ precies \'{e}\'{e}n keer, schrijf het
grootst mogelijke zevencijferige getal, het kleinste, en het getal
dat het dichtst bij $4\,000\,000$ ligt.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g4:numbers:11}
Wat is het grootste gehele getal dat afrondt op $50\,000$ bij
afronden op het dichtstbijzijnde duizendtal? En het kleinste?
\end{exercise}
