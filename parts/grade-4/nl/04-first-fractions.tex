\chapter{Eerste breuken}\label{ch:g4:fractions}

Een half appeltje, een kwartier, driekwart van de klas: \omterm{def:g4:fractions:def}{breuken}
noemen de stukken van dingen. Dit hoofdstuk introduceert de notatie
$\frac{a}{b}$, leest \omterm{def:g4:fractions:def}{breuken} op plaatjes en getallenlijnen, en
vergelijkt de eenvoudigste. (\omterm{def:g4:fractions:def}{Breuken} groeien verder in
\cref{ch:g6:fractions}.)

\section{De stukken benoemen}

\begin{definition}[Breuk]\label{def:g4:fractions:def}
Snijd een geheel in gelijke delen en neem er enkele --- dat is een
\emph{breuk}\index{breuk}:
\[
\frac{3}{4}
\quad
\begin{array}{l}
3 \text{: hoeveel delen genomen worden (de \emph{teller}\index{teller});}\\
4 \text{: in hoeveel gelijke delen het geheel is gesneden (de
\emph{noemer}\index{noemer}).}
\end{array}
\]
$\frac12$ is een \emph{half}, $\frac13$ een \emph{derde}, $\frac14$
een \emph{\omterm{def:g3:division:half}{kwart}}, $\frac{1}{10}$ een \emph{tiende}.
\end{definition}

\begin{example}[Breuken van een verzameling]\label{ex:g4:fractions:collection}
$\frac13$ van $12$ knikkers: verdeel de $12$ knikkers in $3$ gelijke
groepen van $4$; \'{e}\'{e}n groep is $\frac13$, dus $\frac13$ van $12$ is
$4$ --- en $\frac23$ van $12$ is twee groepen, $8$.
\end{example}

\section{Breuken op de getallenlijn}

\begin{method}[$\frac{a}{b}$ plaatsen]\label{met:g4:fractions:line}
Snijd elke eenheid van de lijn in $b$ gelijke stappen; vanaf $0$,
loop $a$ stappen. Als $a$ groter is dan $b$, gaat de wandeling voorbij
$1$: \omterm{def:g4:fractions:def}{breuken} kunnen groter zijn dan \'{e}\'{e}n geheel!
\end{method}

\begin{example}[Hele getallen verborgen in breuken]\label{ex:g4:fractions:whole}
$\frac44 = 1$ (vier \omterm{def:g3:division:half}{kwarten} maken het geheel); $\frac82 = 4$;
$\frac{12}{3} = 4$. En $\frac74$ is $1 + \frac34$: \'{e}\'{e}n geheel en
drie \omterm{def:g3:division:half}{kwarten} erbij.
\end{example}

\section{Eenvoudige breuken vergelijken en optellen}

\begin{proposition}[Zelfde noemer]\label{prop:g4:fractions:compare}
Met dezelfde \omterm{def:g4:fractions:def}{noemer} hebben de stukken dezelfde grootte --- dus tel
ze gewoon:
\[
\frac{2}{8} < \frac{5}{8},
\qquad
\frac{3}{8} + \frac{4}{8} = \frac{7}{8}.
\]
Vergelijken met \'{e}\'{e}n geheel is ook makkelijk:
$\frac{5}{8} < 1 < \frac{9}{8}$ (minder, dan meer, dan $8$ achtsten).
\end{proposition}

\begin{proof}[Waarom, op een plaatje]
$3$ achtsten arceren en dan $4$ achtsten meer van dezelfde balk
arceert in totaal $7$ achtsten.
\end{proof}

\begin{example}[Verschillende noemers, zelfde punt]\label{ex:g4:fractions:equal}
Op de getallenlijn landen $\frac12$, $\frac24$ en $\frac{5}{10}$
allemaal op hetzelfde punt: het zijn drie namen van hetzelfde getal.
Elke \omterm{def:g3:division:half}{helft} in twee snijden maakt \omterm{def:g3:division:half}{kwarten}; in vijf, tienden. (De
algemene regel staat in \cref{ch:g6:fractions}.)
\end{example}

\begin{example}[Kwarten van een uur]\label{ex:g4:fractions:hour}
Een uur heeft $60$ minuten, dus een kwartier is
$60 \div 4 = 15$ minuten, en driekwart uur is $45$ minuten. ``Half
drie'' is $\frac12$ uur na twee uur.
\end{example}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g4:fractions:1}
Schrijf de getoonde \omterm{def:g4:fractions:def}{breuk}: een pizza in $6$ met $5$ stukken over; een
chocoladereep van $8$ vakjes met $3$ opgegeten (\omterm{def:g4:fractions:def}{breuk} opgegeten);
een vlag in $3$ gelijke verticale banen, $1$ gekleurd.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:fractions:2}
Teken een balk in $5$ gelijke delen en arceer $\frac35$. Arceer dan
$\frac{2}{5}$ van een andere gelijke balk. Welke arcering is groter?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:fractions:3}
Reken uit: $\frac12$ van $18$; \quad $\frac14$ van $20$; \quad
$\frac13$ van $21$; \quad $\frac34$ van $20$ (drie groepen van een
\omterm{def:g3:division:half}{kwart}).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:fractions:4}
Plaats op een getallenlijn in derden: $\frac13$; \ $\frac33$; \
$\frac53$; \ $2$. Welke \omterm{def:g4:fractions:def}{breuk} landt precies op $2$?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:fractions:5}
Schrijf over en vul in met $<$, $>$ of $=$:
\[
\frac38 \;?\; \frac68, \qquad
\frac55 \;?\; 1, \qquad
\frac74 \;?\; 1, \qquad
\frac12 \;?\; \frac24 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:fractions:6}
Reken uit: $\frac26 + \frac36$; \quad $\frac58 - \frac28$; \quad
$\frac14 + \frac24$; \quad $1 - \frac13$ (hoeveel derden maken
$1$?).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:fractions:7}
Schrijf als een geheel getal plus een \omterm{def:g4:fractions:def}{breuk} kleiner dan $1$:
$\frac54$; \quad $\frac73$; \quad $\frac{9}{2}$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:fractions:8}
Hoeveel minuten zijn: een half uur; een kwartier; driekwart uur;
$\frac13$ van een uur?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:fractions:9}
In een klas van $24$ leerlingen draagt $\frac14$ een bril. Hoeveel
leerlingen dragen een bril? Hoeveel niet?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g4:fractions:10}
Nina at $\frac38$ van een pizza en Sam at $\frac48$ van dezelfde
pizza. Welke \omterm{def:g4:fractions:def}{breuk} aten ze samen? Welke \omterm{def:g4:fractions:def}{breuk} is over? Wie at meer?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g4:fractions:11}
Waar of niet waar, met een plaatje of een voorbeeld: ``$\frac12$ van
een kleine pizza is minder dan $\frac14$ van een heel grote pizza is
mogelijk''. Wat moet men altijd weten v\'{o}\'{o}r men twee \omterm{def:g4:fractions:def}{breuken} van
iets vergelijkt?
\end{exercise}
