\chapter{माप और परिधि}\label{ch:g4:measure}

ग्रेड 3 को रूलर, तराजू और घड़ियों से मापा जाता है
(\cref{ch:g3:measure}); यह अध्याय इकाई तालिका --- जोड़ता है
मशीन जो सब कुछ बदल देती है --- और एक फ्लैट को मापने के दो तरीके
आकार: इसकी सीमा की लंबाई (\omterm{def:g4:measure:perimeter}{परिधि}) और उसके अंदर का कमरा
(\omterm{def:g4:measure:area}{क्षेत्र}, अभी के लिए वर्गों में गिना जाता है)।

\section{इकाई तालिका}

\begin{method}[तालिका के साथ कनवर्ट करना]\label{met:g4:measure:table}
लंबाई के लिए, प्रत्येक स्तंभ अगले में से दस के बराबर है:
\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{c|c|c|c|c|c|c}
km & hm & dam & m & dm & cm & mm \\
\hline
 & & & $3$ & $4$ & $0$ & \\
\end{tabular}
\end{center}
\begin{enumerate}
  \item संख्या को प्रति कॉलम एक अंक, इकाई अंक के साथ लिखें
        इसकी इकाई के कॉलम में माप का (यहां)। $3.4$ एम: द $3$
        एम के तहत);
  \item नई इकाई में संख्या पढ़ें: $3.4$ एम $= 340$ सेमी (भरें)
        के साथ खाली कॉलम \omterm{def:g1:counting:zero}{शून्य});
  \item \omterm{def:g2:measure:units}{द्रव्यमान} (किलो\,\dots\,जी) और क्षमताएं (एल\,\dots\,सीएल) काम करें
        उसी तरह.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g4:measure:convert}
$5$ किमी $= 5\,000$ एम; \quad $270$ सेमी $= 2.7$ मी (द $2$ के \omterm{ex:g1:subtraction:difference}{अंतर}्गत भूमि
एम); \quad $3$ किग्रा $250$ जी $= 3\,250$ जी; \quad
$2.5$ एल $= 250$ सीएल.
\end{example}

\section{परिधि}

\begin{definition}[परिधि]\label{def:g4:measure:perimeter}
 \emph{परिधि}\index{परिधि} एक का \omterm{def:g4:geometry:polygon}{बहुभुज} कुल है
इसकी सीमा की लंबाई: इसकी सभी भुजाओं की लंबाई \omterm{def:g1:addition:def}{जोड़}ें (उसी में)।
इकाई!) क्लासिक्स के लिए शॉर्टकट:
\[
\text{rectangle: } P = 2 \times (L + w),
\qquad
\text{square: } P = 4 \times c .
\]
\end{definition}

\begin{example}\label{ex:g4:measure:perimeter}
एक आयताकार उद्यान $12$ मी द्वारा $7$ एम:
\[
P = 2 \times (12 + 7) = 2 \times 19 = 38 \text{ m}
\]
अपराध। किनारे का एक चौकोर फोटो फ्रेम $18$ सेमी:
$ P = 4 \times 18 = 72$ सेमी।
\end{example}

\section{क्षेत्रफल: वर्ग गिनती}

\begin{definition}[गिनती के अनुसार क्षेत्रफल]\label{def:g4:measure:area}
 \emph{क्षेत्र}\index{क्षेत्र} ग्रिड पर खींची गई आकृति की संख्या होती है
इसके द्वारा कवर किए गए ग्रिड वर्ग। दो आधे वर्ग एक के रूप में गिने जाते हैं। क्षेत्र
उत्तर ``कितनी सतह'', \omterm{def:g4:measure:perimeter}{परिधि} उत्तर ``कितना समय है
सीमा'' --- दो अलग-अलग प्रश्न!
\end{definition}

\begin{omfigure}
\begin{tikzpicture}[scale=0.55]
  \draw[gray!40, thin] (0,0) grid (13,4);
  \draw[very thick, omDef, fill=omDef!15] (1,1) rectangle (5,3);
  \node[font=\small, omDef] at (3,2) {$8$चौकों};
  \draw[very thick, omThm, fill=omThm!15]
    (7,1) -- (11,1) -- (11,2) -- (9,2) -- (9,3) -- (7,3) -- cycle;
  \node[font=\small, omThm] at (8.4,1.5) {$6$चौकों};
\end{tikzpicture}

{\small ग्रिड पर दो आंकड़े: \omterm{def:g4:measure:area}{क्षेत्र}ों $8$ और $6$ वर्ग। अब चलो
सीमाएँ और इकाइयाँ गिनें: $12$ दोनों के लिए! वही \omterm{def:g4:measure:perimeter}{परिधि},
अलग \omterm{def:g4:measure:area}{क्षेत्र}ों.}
\end{omfigure}

\begin{example}[अर्ध-वर्ग]\label{ex:g4:measure:half}
ग्रिड पर पैरों के साथ एक \omterm{def:g3:shapes:rightangle}{समकोण} त्रिभुज बनाया गया है $4$ और $2$ कवर $3$
पूर्ण वर्ग और $2$ \omterm{def:g3:division:half}{आधा} वर्ग\,\dots\ ध्यान से गिनें: यह \omterm{def:g4:measure:area}{क्षेत्र}
है \omterm{def:g3:division:half}{आधा} की $4 \times 2$ आयत, यानी\ $4$ वर्ग। ए काटना
अपने विकर्ण के अनुदिश आयत हमेशा बराबर के दो त्रिभुज देता है
\omterm{def:g4:measure:area}{क्षेत्र}.
\end{example}

\begin{example}[एक ही क्षेत्रफल, अलग-अलग परिधि]\label{ex:g4:measure:samearea}
ए $4 \times 4$ वर्ग और एक $8 \times 2$ दोनों आवरणों को आयत करें $16$
वर्ग --- वही \omterm{def:g4:measure:area}{क्षेत्र}. सीमाएँ: $16$ वर्ग के लिए इकाइयाँ, $20$ के लिए
आयत. कोई भी माप दूसरे को निर्धारित नहीं करता: एक लंबा पतला
आकार में छोटी सतह के लिए बहुत सारे बॉर्डर होते हैं।
\end{example}

\section{अवधियां}

\begin{method}[पूरे घंटे में समय जोड़ना]\label{met:g4:measure:time}
अवधियों को जोड़ने या घटाने के लिए, पूरे घंटों में हॉप्स में काम करें
(याद करें $1$ एच $= 60$ मिनट):
\[
9\text{ h }40 + 35 \text{ min}:
\quad 9\text{ h }40 \xrightarrow{+20} 10\text{ h }00
\xrightarrow{+15} 10\text{ h }15 .
\]
कभी भी पिछले मिनट न \omterm{def:g1:addition:def}{जोड़}ें $60$ परिवर्तित किए बिना: $40 + 35 = 75$ मिन
$= 1$ एच $15$ मि.
\end{method}

\begin{example}\label{ex:g4:measure:timetable}
ट्रेन रवाना हो जाती है $8{:}47$ और पहुँच जाता है $11{:}05$. अवधि:
$8{:}47 \to 9{:}00$ है $13$ मिनट; $9{:}00 \to 11{:}00$ है $2$ एच;
$11{:}00 \to 11{:}05$ है $5$ मि. कुल: $2$ एच $18$ मि.
\end{example}

\section{अभ्यास}

\begin{exercise}[$\star $]\label{exo:g4:measure:1}
इकाई तालिका के साथ कनवर्ट करें: $7$ सेमी में मी; $3.2$ मी में किमी; $85$ मिमी में
सेमी; $4\,500$ मी किमी में.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:measure:2}
बदलना: $2$ जी में किलो; $1\,800$ जी किलो और जी में; $3.5$ एल सीएल में;
$25$ एल में सीएल.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:measure:3}
की गणना करें \omterm{def:g4:measure:perimeter}{परिधि}: एक आयत $9$ सेमी $\times $ $5$ सेमी; एक वर्ग
पक्ष का $7.5$ सेमी; भुजाओं वाला एक त्रिभुज $6$ सेमी, $8$ सेमी और $10$ सेमी।
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:measure:4}
एक आयत है \omterm{def:g4:measure:perimeter}{परिधि} $26$ सेमी और लंबाई $8$ सेमी। इसकी चौड़ाई ज्ञात कीजिये
(पहले लंबाई ज्ञात करें $+$ चौड़ाई)।
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:measure:5}
ग्रिड पेपर पर, बिल्कुल कवर करते हुए एक एल-आकार की आकृति बनाएं $10$
वर्ग, और इसकी इकाइयाँ गिनें \omterm{def:g4:measure:perimeter}{परिधि}.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:measure:6}
के दो अलग-अलग आयत बनाएं \omterm{def:g4:measure:area}{क्षेत्र} $12$ वर्ग। की गणना करें
\omterm{def:g4:measure:perimeter}{परिधि} प्रत्येक का. कौन सा ``राउंडर'' है, कौन सा ``पतला'' है?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:measure:7}
ग्रिड पर पैरों से एक \omterm{def:g3:shapes:rightangle}{समकोण} त्रिभुज बनाया गया है $6$ और $3$. यह क्या है \omterm{def:g4:measure:area}{क्षेत्र} चौकों में? (उपयोग \cref{ex:g4:measure:half}.)
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:measure:8}
अवधि की गणना करें: से $10{:}20$ को $12{:}45$; से $7{:}35$ को
$9{:}10$. एक फिल्म चलती है $1$ एच $50$ और शुरू होता है $20{:}30$: कब करता है
यह ख़त्म?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:measure:9}
किम दौड़ता है $3$ एक आयताकार मैदान की गोद $60$ मी द्वारा $45$ एम। कितने
क्या वह \omterm{def:g4:fractions:def}{मीटर} दौड़ती है?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g4:measure:10}
एक किसान किनारे के एक वर्गाकार खेत की बाड़ लगाना चाहता है $85$ मी, ए को छोड़कर
का द्वार $4$ मैं बाड़ के बिना. उसे कितने \omterm{def:g4:fractions:def}{मीटर} की बाड़ खरीदनी होगी?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g4:measure:11}
सही या गलत, ग्रिड उदाहरणों के साथ: ``यदि दो आंकड़े समान हैं
\omterm{def:g4:measure:perimeter}{परिधि}, उनके पास समान है \omterm{def:g4:measure:area}{क्षेत्र}''; ``यदि दो अंक समान हैं
\omterm{def:g4:measure:area}{क्षेत्र}, उनके पास समान है \omterm{def:g4:measure:perimeter}{परिधि}''.
(\Cref{ex:g4:measure:samearea} और अध्याय का चित्र मदद करेगा।)
\end{exercise}
