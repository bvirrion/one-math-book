\chapter{रेखाएँ और बहुभुज}\label{ch:g4:geometry}

एक रूलर, एक सेट स्क्वायर और एक कम्पास के साथ, ज्यामिति एक शिल्प बन जाती है।
यह अध्याय रेखाओं की दो वि\omterm{def:g3:division:remainder}{शेष} स्थितियाँ सिखाता है ---
\omterm{def:g4:geometry:def}{सीधा} और \omterm{def:g4:geometry:def}{समानांतर} --- उपकरण स्तर पर, और का दौरा लेता है
द \omterm{def:g4:geometry:polygon}{बहुभुज} परिवार। (इन धारणाओं का सिद्धांत आता है
\cref{ch:g6:lines}.)

\section{लंबवत और समानांतर}

\begin{definition}[लम्बवत्, समानान्तर]\label{def:g4:geometry:def}
दो पंक्तियाँ हैं \emph{सीधा}\index{सीधा} कब
वे एक पर पार करते हैं \omterm{def:g3:shapes:rightangle}{समकोण} (सेट वर्ग से जांच की गई, चिह्नित किया गया
थोड़ा चौकोर)। दो पंक्तियाँ हैं \emph{समानांतर}\index{समानांतर}
जब वे एक ही दिशा में दौड़ते हैं और कभी भी पार नहीं करते --- दोनों की तरह
एक ट्रैक की पटरियाँ, हमेशा एक दूसरे से समान दूरी पर।
\end{definition}

\begin{method}[सेट स्क्वायर के साथ ड्राइंग]\label{met:g4:geometry:draw}
\emph{\omterm{def:g4:geometry:def}{सीधा} एक पंक्ति के लिए $d $ एक बिंदु के माध्यम से $ P $}: एक किनारा रखें
का \omterm{def:g3:shapes:rightangle}{समकोण} साथ में $ d $, तब तक स्लाइड करें जब तक कि दूसरा किनारा न पहुंच जाए $ P $,
खींचना।

\emph{\omterm{def:g4:geometry:def}{समानांतर} को $ d $ के माध्यम से $ P $}: सेट स्क्वायर के किनारे को साथ रखें
$ d $, रूलर को रेल की तरह दूसरे किनारे पर दबाएं, सेट को स्लाइड करें
तक रूलर के अनुदिश वर्गाकार $ P $, खींचना।
\end{method}

\begin{example}[उन्हें आंकड़ों में देखिए]\label{ex:g4:geometry:spot}
एक आयत में, प्रत्येक भुजा होती है \omterm{def:g4:geometry:def}{सीधा} अपने दो पड़ोसियों और
\omterm{def:g4:geometry:def}{समानांतर} विपरीत दिशा में. ग्रिड पेपर एक रेडीमेड नेटवर्क है
समानताएं और लंबवत --- यही बात इसे ऐसा बनाती है
खींचने में आरामदायक.
\end{example}

\section{बहुभुज}

\begin{definition}[बहुभुज]\label{def:g4:geometry:polygon}
ए \emph{बहुभुज}\index{बहुभुज} एक बंद आकृति है जिसकी सीमा है
सीधे खंडों से बना है, इसका \emph{दोनों पक्ष}; कोने बिंदु हैं
द \emph{कोने}\index{कोने}. बहुभुजों का नाम उनकी भुजाओं की संख्या के आधार पर रखा गया है:
\emph{\omterm{def:g3:shapes:shapes}{त्रिकोण}} ($3$), \emph{चतुष्कोष}\index{चतुष्कोष} ($4$),
\emph{पंचकोण}\index{पंचकोण} ($5$), \emph{षट्भुज}\index{षट्भुज} ($6$),
\emph{अष्टकोना}\index{अष्टकोना}
($8$).
\end{definition}

\emph{नियमित}: समान भुजाएँ और समान \omterm{def:g4:geometry:polygon}{कोने})। एक वृत्त एक नहीं है
\omterm{def:g4:geometry:polygon}{बहुभुज}: इसकी सीमा खंडों से नहीं बनी है।}

\begin{example}[बहुभुज का वर्णन]\label{ex:g4:geometry:describe}
एक उपयोगी खेल: किसी आकृति का इतनी सटीकता से वर्णन करें जितना कोई सहपाठी कर सके
इसे देखे बिना इसे दोबारा बनाएं। ``ए \omterm{def:g4:geometry:polygon}{चतुष्कोष} $ ABCD $ साथ
$ AB = 5$ सेमी, पर कोण $ A $ सही, $ AD = 3$ सेमी\,\dots'' ---
शब्दावली वह है जो संदेश को कार्यान्वित करती है।
\end{example}

\section{वृत्त और कम्पास}

\begin{definition}[घेरा]\label{def:g4:geometry:circle}
 \emph{\omterm{def:g3:shapes:shapes}{घेरा}} केंद्र का $ O $ और \omterm{def:g3:shapes:shapes}{RADIUS} $3$ सेमी सभी का समुच्चय है
बिलकुल ठीक इंगित करता है $3$ सेमी से $ O $. कम्पास इसे खींचता है: स्पाइक ऑन
$ O $, खोलना $3$ सेमी, बारी. ए \emph{व्यास}\index{व्यास} के माध्यम से एक खंड है
वृत्त के दो बिंदुओं को जोड़ने वाला केंद्र; इसका माप दोगुना है
\omterm{def:g3:shapes:shapes}{RADIUS}.
\end{definition}

\begin{example}[दूरी-रक्षक के रूप में कम्पास]\label{ex:g4:geometry:compass}
कम्पास वृत्त खींचने से कहीं अधिक कार्य करता है: यह \emph{परिवहन
दूरियाँ}. सभी बिन्दुओं को अंकित करना $4$ एक बिंदु से सेमी $ A $, ड्रा करें
केंद्र का \omterm{def:g3:shapes:shapes}{घेरा} $ A $, \omterm{def:g3:shapes:shapes}{RADIUS} $4$ सेमी; बिना दो खंडों की तुलना करना
एक रूलर, एक पर कम्पास खोलें और दूसरे पर इसका परीक्षण करें।
\end{example}

\begin{method}[ग्रिड पर एक आकृति का पुनरुत्पादन]\label{met:g4:geometry:reproduce}
\begin{enumerate}
  \item एक आरंभ चुनें शिखर और रिक्त ग्रिड पर इसकी स्थिति;
  \item आकृति के साथ यात्रा करें शिखर द्वारा शिखर, ग्रिड की गिनती
        चरण (``$3$ सही, $2$ ऊपर'');
  \item शामिल होना \omterm{def:g4:geometry:polygon}{कोने} क्रम में, और मॉडल के साथ तुलना करें:
        समान लंबाई, समान दिशाएँ।
\end{enumerate}
\end{method}

\section{अभ्यास}

\begin{exercise}[$\star $]\label{exo:g4:geometry:1}
एक रेखा खींचो $ d $ और एक बिंदु $ P $ उस पर नहीं. का निर्माण करें
\omterm{def:g4:geometry:def}{सीधा} को $ d $ के माध्यम से $ P $, फिर \omterm{def:g4:geometry:def}{समानांतर} को $ d $ के माध्यम से
$ P $ (रूलर-रेल विधि के साथ)। चिन्हित करें \omterm{def:g3:shapes:rightangle}{समकोण}.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:geometry:2}
बड़े अक्षरों E, H, N, T, Z को देखें। प्रत्येक में, एक \omterm{def:g1:addition:def}{जोड़}ी खोजें
\omterm{def:g4:geometry:def}{समानांतर} स्ट्रोक और की एक \omterm{def:g1:addition:def}{जोड़}ी \omterm{def:g4:geometry:def}{सीधा} स्ट्रोक जब वे
मौजूद.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:geometry:3}
प्रत्येक का नाम बताएं \omterm{def:g4:geometry:polygon}{बहुभुज}: $6$ भुजाएँ; $4$ भुजाएँ; $3$ भुजाएँ; $8$ भुजाएँ;
$5$ पक्ष. कितने \emph{\omterm{def:g4:geometry:polygon}{कोने}} क्या हर एक के पास है?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:geometry:4}
कोई भी ड्रा करें \omterm{def:g4:geometry:polygon}{पंचकोण} (जरूरी नहीं कि नियमित हो) और इसे लेबल करें \omterm{def:g4:geometry:polygon}{कोने}
$ A $, $ B $, $ C $, $ D $, $ E $. इसकी कितनी भुजाएँ हैं? इसका चित्र बनाओ
से विकर्ण $ A $ (खंड जुड़ रहे हैं $ A $ गैर-पड़ोसी को \omterm{def:g4:geometry:polygon}{कोने}):
कितने हैं?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:geometry:5}
केंद्र का एक वृत्त बनाएं $ O $ और \omterm{def:g3:shapes:shapes}{RADIUS} $4$ सेमी; ए \omterm{def:g4:geometry:circle}{व्यास} $[AB]$;
और एक बिंदु $ M $ वृत्त का चालू नहीं है $[AB]$. लंबाई बतायें $ OA $,
$ OM $ और $ AB $ बिना मापे.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:geometry:6}
दूरी-रक्षक के रूप में कम्पास के साथ
(\cref{ex:g4:geometry:compass}), एक त्रिभुज की भुजाओं की तुलना करें
एक सहपाठी द्वारा तैयार किया गया और तय करें कि क्या इसकी दो बराबर भुजाएँ हैं।
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:geometry:7}
ग्रिड पेपर पर, इस पथ को पुन: उत्पन्न करें और इसे एक में बंद करें \omterm{def:g4:geometry:polygon}{बहुभुज}:
एक बिंदु से प्रारंभ करें $ A $; जाना $4$ सही; $2$ ऊपर; $3$ बाएं; $2$ ऊपर;
फिर वापस $ A $ एक सीधी रेखा में। का नाम क्या है \omterm{def:g4:geometry:polygon}{बहुभुज}
तुम्हें मिलता है (उसकी भुजाएँ गिनें)?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:geometry:8}
सच या झूठ? चित्र सहित समझाइये।
``दो लंबवत रेखाएँ ''हमेशा पार करो.''
``दो \omterm{def:g4:geometry:def}{समानांतर} रेखाएँ कभी पार मत करना.''
``एक लाइन हो सकती है \omterm{def:g4:geometry:def}{समानांतर} एक पंक्ति में और \omterm{def:g4:geometry:def}{सीधा} दूसरे करने के लिए।''
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g4:geometry:9}
एक रेखा खींचो $ d $, फिर एक पंक्ति $ e $ \omterm{def:g4:geometry:def}{सीधा} को $ d $, फिर एक पंक्ति
$ f $ \omterm{def:g4:geometry:def}{सीधा} को $ e $. देखो $ d $ और $ f $: आपको क्या लगा?
(यह अवलोकन एक प्रमेय बन जाता है \cref{ch:g6:lines}.)
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g4:geometry:10}
बिना दिखाए किसी सहपाठी के लिए नीचे दिए गए चित्र का शब्दों में वर्णन करें
यह: भुजा का एक वर्ग $4$ सेमी, के एक वृत्त के साथ \omterm{def:g3:shapes:shapes}{RADIUS} $2$ सेमी केन्द्रित
वर्ग के मध्य में (चारों भुजाओं को छूते हुए)। लिखें
क्रमांकित निर्देशों के रूप में विवरण।
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g4:geometry:11}
A में कितने विकर्ण होते हैं \omterm{def:g4:geometry:polygon}{चतुष्कोष} पास होना? ए \omterm{def:g4:geometry:polygon}{पंचकोण}? ए \omterm{def:g4:geometry:polygon}{षट्भुज}?
तीन केस बनाएं, ध्यान से गिनें और अपने पैटर्न का वर्णन करें
देखिये.
\end{exercise}
