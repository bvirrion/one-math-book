\chapter{प्रथम अंश}\label{ch:g4:fractions}

\omterm{def:g3:division:half}{आधा} एक सेब, ए \omterm{def:g3:division:half}{तिमाही} एक घंटे का, तीन क्वार्टरों कक्षा का:
\omterm{def:g4:fractions:def}{अंश}ों चीज़ों के टुकड़ों के नाम बताइए. यह अध्याय परिचय देता है
लेखन $\frac{a}{b}$, पढ़ता है \omterm{def:g4:fractions:def}{अंश}ों चित्रों और संख्या रेखाओं पर,
और सबसे सरल की तुलना करता है। (भिन्न में बड़ा होना
\cref{ch:g6:fractions}.)

\section{टुकड़ों का नामकरण}

\begin{definition}[अंश]\label{def:g4:fractions:def}
पूरे को बराबर भागों में काटें और कुछ लें---यह एक है
\emph{अंश}\index{अंश}:
\[
\frac{3}{4}
\quad
\begin{array}{l}
3 \text{: how many parts are taken (the \emph{मीटर}\index{मीटर});}\\
4 \text{: how many equal parts the whole is cut into (the
\emph{भाजक}\index{भाजक}).}
\end{array}
\]
$\frac12$ एक है \emph{\omterm{def:g3:division:half}{आधा}}, $\frac13$ ए \emph{तीसरा}, $\frac14$ ए
\emph{\omterm{def:g3:division:half}{तिमाही}}, $\frac{1}{10}$ ए \emph{दसवां}.
\end{definition}

\begin{example}[एक संग्रह के अंश]\label{ex:g4:fractions:collection}
$\frac13$ का $12$ मार्बल्स: साझा करें $12$ में पत्थर $3$ बराबर
के समूह $4$; एक समूह है $\frac13$, इसलिए $\frac13$ का $12$ है $4$
--- और $\frac23$ का $12$ दो समूह हैं, $8$.
\end{example}

\section{संख्या रेखा पर भिन्न}

\begin{method}[लगाना $\frac{a}{b}$]\label{met:g4:fractions:line}
पंक्ति की प्रत्येक इकाई को इसमें काटें $ b $ समान कदम; से $0$, टहलना $ a $
कदम. अगर $ a $ से बड़ा है $ b $, चलना गुजरता है $1$: \omterm{def:g4:fractions:def}{अंश}ों कर सकते हैं
एक पूरे से बड़ा हो!
\end{method}

\begin{example}[पूर्ण संख्याएँ भिन्नों में छिपी हुई हैं]\label{ex:g4:fractions:whole}
$\frac44 = 1$ (चार क्वार्टरों पूरा बनाओ); $\frac82 = 4$;
$\frac{12}{3} = 4$. और $\frac74$ है $1 + \frac34$: एक संपूर्ण और
तीन क्वार्टरों अधिक।
\end{example}

\section{सरल भिन्नों की तुलना करना और उन्हें जोड़ना}

\begin{proposition}[वही भाजक]\label{prop:g4:fractions:compare}
उसी के साथ \omterm{def:g4:fractions:def}{भाजक}, टुकड़ों का आकार समान है --- तो बस
उन्हें गिनें:
\[
\frac{2}{8} < \frac{5}{8},
\qquad
\frac{3}{8} + \frac{4}{8} = \frac{7}{8}.
\]
एक संपूर्ण से तुलना करना भी आसान है: $\frac{5}{8} < 1 < \frac{9}{8}$
(कम, फिर अधिक, से $8$ आठवाँ भाग)।
\end{proposition}

\begin{proof}[क्यों, एक तस्वीर पर]
छायांकन $3$ आठवीं और फिर $4$ एक ही बार शेड्स के अधिक आठवें भाग
$7$ कुल आठवाँ।
\end{proof}

\begin{example}[अलग-अलग भाजक, एक ही बिंदु]\label{ex:g4:fractions:equal}
संख्या रेखा पर, $\frac12$, $\frac24$ और $\frac{5}{10}$ सारी भूमि
एक ही बात पर: वे एक ही संख्या के तीन नाम हैं। काटना
प्रत्येक \omterm{def:g3:division:half}{आधा} दो रूपों में क्वार्टरों; पाँच, दसवें में। (जनरल
नियम में है \cref{ch:g6:fractions}.)
\end{example}

\begin{example}[एक घंटे का चौथाई]\label{ex:g4:fractions:hour}
एक घंटा हो गया है $60$ मिनट, तो ए \omterm{def:g3:division:half}{तिमाही} एक घंटे का है
$60 \div 4 = 15$ मिनट, और तीन क्वार्टरों एक घंटे का है
$45$ मिनट। ``ढाई बजकर दो मिनट'' है $\frac12$ दो घंटे बाद
बजे.
\end{example}

\section{अभ्यास}

\begin{exercise}[$\star $]\label{exo:g4:fractions:1}
लिखना \omterm{def:g4:fractions:def}{अंश} दिखाया गया: एक पिज़्ज़ा काटा गया $6$ साथ $5$ बचे हुए टुकड़े;
का एक चॉकलेट बार $8$ के साथ वर्ग $3$ खाया (\omterm{def:g4:fractions:def}{अंश} खाया); ए
झंडे को विभाजित किया गया $3$ समान ऊर्ध्वाधर बैंड, $1$ रंगीन.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:fractions:2}
एक कटी हुई पट्टी बनाएं $5$ समान भाग और छाया $\frac35$. फिर छाया दें
$\frac{2}{5}$ एक और बराबर बार का. कौन सा शेडिंग बड़ा है?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:fractions:3}
गणना करें: $\frac12$ का $18$; \quad $\frac14$ का $20$; \quad
$\frac13$ का $21$; \quad $\frac34$ का $20$ (ए के तीन समूह
\omterm{def:g3:division:half}{तिमाही}).
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:fractions:4}
तिहाई में काटी गई संख्या रेखा पर रखें: $\frac13$; \ $\frac33$; \
$\frac53$; \ $2$. कौन \omterm{def:g4:fractions:def}{अंश} बिलकुल पर उतरता है $2$?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:fractions:5}
कॉपी करें और पूरा करें $<$, $>$ या $=$:
\[
\frac38 \;?\; \frac68, \qquad
\frac55 \;?\; 1, \qquad
\frac74 \;?\; 1, \qquad
\frac12 \;?\; \frac24 .
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:fractions:6}
गणना करें: $\frac26 + \frac36$; \quad $\frac58 - \frac28$; \quad
$\frac14 + \frac24$; \quad $1 - \frac13$ (कितने तिहाई बनते हैं
$1$?).
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:fractions:7}
पूर्ण संख्या प्लस ए के रूप में लिखें \omterm{def:g4:fractions:def}{अंश} तुलना में छोटा $1$:
$\frac54$; \quad $\frac73$; \quad $\frac{9}{2}$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:fractions:8}
कितने मिनट होते हैं: \omterm{def:g3:division:half}{आधा} एक घंटा; ए \omterm{def:g3:division:half}{तिमाही} एक घंटे का; तीन
क्वार्टरों एक घंटे का; $\frac13$ एक घंटे का?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:fractions:9}
की एक कक्षा में $24$ छात्र, $\frac14$ चश्मा पहनें. कितने
छात्र चश्मा पहनते हैं? कितने नहीं करते?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g4:fractions:10}
नीना ने खाया $\frac38$ एक पिज़्ज़ा और सैम ने खाया $\frac48$ उसी का
पिज़्ज़ा. क्या \omterm{def:g4:fractions:def}{अंश} क्या उन्होंने एक साथ खाना खाया? क्या \omterm{def:g4:fractions:def}{अंश} \omterm{def:g3:division:remainder}{शेष} है?
किसने ज्यादा खाया?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g4:fractions:11}
सही या ग़लत, चित्र या उदाहरण के साथ: ``$\frac12$ एक छोटे से
पिज़्ज़ा से कम है $\frac14$ ''बहुत बड़ा पिज़्ज़ा संभव है।''
दो की तुलना करने से पहले हमेशा क्या जानना चाहिए? \omterm{def:g4:fractions:def}{अंश}ों का
कुछ?
\end{exercise}
