\chapter{समरूपता}\label{ch:g4:symmetry}

तितलियाँ, \omterm{def:g2:shapes:solids}{चेहरे}, बर्फ के टुकड़े: प्रकृति उन आकृतियों को पसंद करती है जिनके दो हिस्से होते हैं
मोड़ने पर मिलान करें। यह अध्याय सिलवटों और ग्रिड के साथ समरूपता का पता लगाता है
कागज --- सेट वर्ग के साथ निर्माण के लिए प्रतीक्षा करें
\cref{ch:g6:symmetry}.

\section{तह करना और मिलान करना}

\begin{definition}[समरूपता की धुरी]\label{def:g4:symmetry:def}
एक पंक्ति एक है \emph{समरूपता की धुरी}\index{समरूपता की धुरी} एक का
आकृति को उस रेखा के अनुदिश मोड़ने पर दो भाग बन जाते हैं
मिलान \emph{बिल्कुल}. किसी आकृति में कोई अक्ष, एक अक्ष या नहीं हो सकता
अनेक.
\end{definition}

  \end{scope}
\end{tikzpicture}

{\small धराशायी रेखा के साथ मोड़ें: आधे भाग मेल खाते हैं। आखिरी
त्रिभुज में कोई तह नहीं है जो काम करता हो --- नहीं \omterm{def:g4:symmetry:def}{समरूपता की धुरी}.}
\end{omfigure}

\begin{example}[ट्रेसिंग पेपर से परीक्षण]\label{ex:g4:symmetry:trace}
आकृति को ट्रेस करें, ट्रेसिंग पेपर को उम्मीदवार के साथ पलटें
अक्ष, और इसे वापस रखें: यदि ट्रेसिंग आकृति को बिल्कुल कवर करती है, तो
अक्ष वास्तविक है. आंखें ही अक्सर धोखा खा जाती हैं --- एक आयत की
विकर्ण \emph{दिखता है} एक अक्ष की तरह, लेकिन तह परीक्षण कहता है कि नहीं (कोशिश करें)।
यह!)
\end{example}

\section{ग्रिड पर आंकड़े पूरा करना}

\begin{method}[ग्रिड रेखा के पार किसी आकृति का सममित होना]\label{met:g4:symmetry:grid}
अक्ष ग्रिड की एक ऊर्ध्वाधर (या क्षैतिज) रेखा है।
\begin{enumerate}
  \item प्रत्येक ले लो शिखर बदले में आकृति का;
  \item अक्ष से इसकी दूरी को वर्गों में गिनें;
  \item छवि लगाएं शिखर पर \emph{वही} पर दूरी
        \emph{अन्य} किनारे, एक ही पंक्ति (या स्तंभ) पर;
  \item छवि से जुड़ें \omterm{def:g4:geometry:polygon}{कोने} उसी क्रम में, और जाँच करें:
        छवि आकृति का दर्पण जुड़वां है।
\end{enumerate}
\end{method}

\end{tikzpicture}

{\small लाल अक्ष पर पूर्णता: प्रत्येक शिखर उसी पर कूद पड़ता है
दूसरी ओर दूरी. \omterm{def:g4:geometry:polygon}{कोने} अक्ष को छूना बना रहेगा
डाल.}
\end{omfigure}

\begin{example}[एक ही आकार, एक ही आकार, पलटा हुआ]\label{ex:g4:symmetry:properties}
छवि की लंबाई और कोण बिल्कुल समान हैं
मूल: समरूपता आकृति की प्रतिलिपि बनाती है --- लेकिन इसे पलट देती है, जैसे कि
बायां हाथ और दाहिना हाथ। यदि मूल अक्षर F है, तो इसका दर्पण है
प्रतिबिम्ब पीछे की ओर F है, अन्य F नहीं।
\end{example}

\section{सामान्य आकृतियों की कुल्हाड़ियाँ}

\begin{example}[कुल्हाड़ियाँ गिनना]\label{ex:g4:symmetry:shapes}
मोड़कर:
\begin{itemize}
  \item एक वर्ग: $4$ कुल्हाड़ियाँ (विपरीत के मध्य से होकर दो
        भुजाएँ, विकर्णों के अनुदिश दो);
  \item एक आयत: $2$ अक्ष (केवल विपरीत भुजाओं के मध्य!);
  \item एक समद्विबाहु त्रिभुज: $1$; एक समबाहु त्रिभुज: $3$;
  \item एक वृत्त: प्रत्येक रेखा अपने केंद्र से होकर गुजरती है --- से अधिक अक्ष
        कोई गिन सकता है.
\end{itemize}
\end{example}

\begin{example}[वर्णमाला में समरूपता]\label{ex:g4:symmetry:letters}
ए, एम, टी, यू, वी की एक ऊर्ध्वाधर धुरी है; बी, सी, डी, ई में एक क्षैतिज रेखा है
एक; H, I, O, X में दोनों हैं। सही अक्षरों से बने शब्द हो सकते हैं
सममित भी: MUM में एक ऊर्ध्वाधर अक्ष है; ओएक्सओ दोनों तरह से काम करता है।
\end{example}

\section{अभ्यास}

\begin{exercise}[$\star $]\label{exo:g4:symmetry:1}
ट्रेस और फोल्ड: इनमें से किसमें एक है \omterm{def:g4:symmetry:def}{समरूपता की धुरी} --- ए
वर्गाकार; एक (गैर-वर्ग) आयत; एक विषमबाहु त्रिभुज (सभी भुजाएँ
भिन्न); एक चक्र?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:symmetry:2}
सममिति के अक्ष खींचिए: एक वर्ग; एक आयत; एक समबाहु
\omterm{def:g3:shapes:shapes}{त्रिकोण}. प्रत्येक के लिए कितने?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:symmetry:3}
के ट्रेसिंग-पेपर परीक्षण का उपयोग करें \cref{ex:g4:symmetry:trace} दिखाने के लिए
कि a का विकर्ण $6 \times 4$ आयत है \emph{नहीं} एक अक्ष
समरूपता का.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:symmetry:4}
ग्रिड पेपर पर, एक ऊर्ध्वाधर अक्ष और अक्षर L (तीन वर्ग) बनाएं
लंबा, दो चौड़ा) पर $2$ अक्ष से वर्ग. इसके दर्पण का निर्माण करें
छवि.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:symmetry:5}
ग्रिड पेपर पर, एक क्षैतिज अक्ष और एक छोटा झंडा बनाएं (ए)। $1 \times
1$ ए पर \omterm{def:g3:shapes:shapes}{त्रिकोण} $3$-वर्ग मस्तूल) इसके ऊपर। द्वारा चित्र पूरा करें
अक्ष के नीचे समरूपता.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:symmetry:6}
एक आंकड़ा इसे छूता है \omterm{def:g4:symmetry:def}{समरूपता की धुरी} एक बिंदु पर $ P $. कहाँ है
की छवि $ P $? मोड़ने योग्य चित्र द्वारा समझाइये।
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:symmetry:7}
बड़े अक्षरों A, B, E, F, H, N, O, T को क्रमबद्ध करें: ऊर्ध्वाधर अक्ष?
क्षैतिज अक्ष? दोनों? कोई भी नहीं?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:symmetry:8}
एक त्रिभुज की भुजाएँ होती हैं $5$ सेमी, $5$ सेमी और $3$ सेमी। क्या इसमें कोई है?
\omterm{def:g4:symmetry:def}{समरूपता की धुरी}? तह किस तरफ विनिमय करती है?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g4:symmetry:9}
ग्रिड पेपर पर अपनी खुद की एक आकृति बनाएं \emph{बिल्कुल दो}
सममिति अक्ष, और दोनों अक्ष खींचिए। (संकेत: एक आयत के बारे में सोचें,
या एक क्रॉस का।)
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g4:symmetry:10}
\omterm{def:g3:division:half}{आधा} एक सममित रेखाचित्र दिया गया है: बाएँ \omterm{def:g3:division:half}{आधा} एक घर का
(एक चौकोर दीवार और \omterm{def:g3:division:half}{आधा} एक \omterm{def:g3:shapes:shapes}{त्रिकोण}ीय छत), एक ऊर्ध्वाधर अक्ष के साथ
इसके दाहिने किनारे पर. पूरे घर का वर्णन करें या चित्र बनाएं। कैसे करें?
पूर्ण अधिकार की लंबाई \omterm{def:g3:division:half}{आधा} बाईं ओर से तुलना करें \omterm{def:g3:division:half}{आधा}?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g4:symmetry:11}
अया का दावा है: ``दो समान भुजाओं वाले प्रत्येक त्रिभुज में एक होता है की धुरी
समरूपता, और प्रत्येक त्रिभुज एक के साथ \omterm{def:g4:symmetry:def}{समरूपता की धुरी} दो बराबर हैं
भुजाएँ।'' रेखाचित्रों पर उसके दोहरे दावे का परीक्षण करें: एक समद्विबाहु त्रिभुज,
एक समबाहु, एक विषमकोण। क्या तह उसका समर्थन करती है?
\end{exercise}
