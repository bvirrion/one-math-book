\chapter{बड़ी संख्याओं को गुणा करना}\label{ch:g4:mult}

ग्रेड 3 को एक अंक से गुणा किया गया (\cref{ch:g3:mult}); यह अध्याय
दो अंकों की संख्याओं से गुणा करता है --- और रहस्यमयी व्याख्या करता है
कॉलम विधि की ``शिफ्ट की गई लाइन'', जो और कुछ नहीं बल्कि एक है
दहाई और इकाई में विभाजित करना।

\section{दहाई से गुणा करना}

\begin{proposition}[टाइम्स 10, 20, 300\,\dots]\label{prop:g4:mult:tens}
से गुणा करना $10$ एक जोड़ता है \omterm{def:g1:counting:zero}{शून्य}, द्वारा $100$ दो \omterm{def:g1:counting:zero}{शून्य}. गुणा करना
द्वारा $20$, गुणा करके $2$, फिर द्वारा $10$:
\[
36 \times 20 = 36 \times 2 \times 10 = 72 \times 10 = 720 .
\]
वैसे ही $14 \times 300 = 14 \times 3 \times 100 = 4\,200$.
\end{proposition}
\admitted

\section{दो अंकों वाली स्तंभ विधि}

\begin{example}[स्थानांतरित लाइन क्यों?]\label{ex:g4:mult:why}
गणना करना $23 \times 12$, विभाजित करना $12$ में $10 + 2$:
\[
23 \times 12 = 23 \times 2 + 23 \times 10 = 46 + 230 = 276 .
\]
कॉलम विधि बिल्कुल इन दोनों को लिखती है \omterm{def:g2:mult:def}{उत्पाद}ों, के \omterm{ex:g1:subtraction:difference}{अंतर}्गत एक
अन्य --- दूसरा बाईं ओर स्थानांतरित हो गया क्योंकि यह एक संख्या है
\emph{दसियों}:
\[
\begin{array}{r}
2\,3 \\
\times\ 1\,2 \\
\hline
4\,6 \\
2\,3\,\cdot \\
\hline
2\,7\,6
\end{array}
\]
(बिंदु स्थानांतरित रेखा के इकाई स्थान को चिह्नित करता है, अक्सर बाईं ओर
रिक्त).
\end{example}

\begin{method}[दो अंकों की संख्या से कॉलम का गुणन]\label{met:g4:mult:column}
गणना करना $457 \times 36$:
\begin{enumerate}
  \item पहली पंक्ति: $457 \times 6 = 2\,742$;
  \item दूसरी पंक्ति: $457 \times 3$ \emph{दसियों} $= 1\,371$ दहाई ---
        लिखो $1\,371$ एक स्थान छोड़ दिया गया;
  \item दो पंक्तियाँ \omterm{def:g1:addition:def}{जोड़}ें: $2\,742 + 13\,710 = 16\,452$;
  \item जांचने का अनुमान: $457 \times 36 \approx 500 \times 36
        = 18\,000$? बेहतर: $450 \times 36$ के बारे में है $16\,000$ ---
        प्रशंसनीय.
\end{enumerate}
\[
\begin{array}{r}
4\,5\,7 \\
\times\ \ \,3\,6 \\
\hline
2\,7\,4\,2 \\
1\,3\,7\,1\,\cdot \\
\hline
1\,6\,4\,5\,2
\end{array}
\]
\end{method}

\begin{example}[मानसिक युक्तियाँ]\label{ex:g4:mult:tricks}
\begin{itemize}
  \item \emph{मैत्रीपूर्ण तरीके से विभाजित करें}: $18 \times 11 = 18 \times 10
        + 18 = 198$.
  \item \emph{लगभग सौ का प्रयोग करें}: $25 \times 99 = 25 \times 100 -
        25 = 2\,475$.
  \item \emph{दोगुना और \omterm{def:g3:division:half}{आधा}}: $16 \times 35 = 8 \times 70 =
        560$ (एक कारक को \omterm{def:g3:division:half}{आधा} करने और दूसरे को दोगुना करने से स्थिति बनी रहती है
        \omterm{def:g2:mult:def}{उत्पाद}).
\end{itemize}
\end{example}

\section{समस्याओं में गुणा}

\begin{example}\label{ex:g4:mult:problem}
एक स्कूल आदेश देता है $28$ के बक्से $145$ कागज के पत्र।
\begin{enumerate}
  \item संचालन: $145 \times 28$.
  \item अनुमान लगाना: $150 \times 30 = 4\,500$, दोनों पर थोड़ा ज्यादा
        मायने रखता है.
  \item कॉलम: $145 \times 8 = 1\,160$; $145 \times 2$ दसियों
        $= 2\,900$; कुल $4\,060$.
  \item वाक्य: स्कूल को प्राप्त होता है $4\,060$ चादरें.
\end{enumerate}
\end{example}

\section{अभ्यास}

\begin{exercise}[$\star $]\label{exo:g4:mult:1}
मानसिक रूप से गणना करें: $47 \times 10$; \quad $23 \times 100$; \quad
$36 \times 20$; \quad $15 \times 300$; \quad $42 \times 50$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:mult:2}
गणना करना $34 \times 21$ के रूप में विभाजित करके \cref{ex:g4:mult:why}
($34 \times 21 = 34 \times 20 + 34$), फिर कॉलम में जाँच करें।
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:mult:3}
कॉलम में गणना करें: $63 \times 24$; \quad $85 \times 47$; \quad
$126 \times 53$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:mult:4}
पहले अनुमान लगाकर, कॉलम में गणना करें: $208 \times 34$; \quad
$517 \times 62$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:mult:5}
से एक ट्रिक से गणना करें \cref{ex:g4:mult:tricks}:
$45 \times 11$; \quad $32 \times 99$; \quad $24 \times 45$ (डबल
और यदि आप चाहें तो आधा--- दो बार करें)।
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:mult:6}
एक सिनेमा है $26$ की पंक्तियाँ $18$ सीटें. कुल कितनी सीटें?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:mult:7}
एक ट्रक ले जाता है $32$ के फूस $48$ टोकरे. कितने टोकरे? देना
पहले अनुमान, फिर सटीक उत्तर।
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:mult:8}
गलती ढूंढें: एना गणना करती है $57 \times 23$ जैसा
$57 \times 3 = 171$ और $57 \times 2 = 114$, फिर जोड़ता है
$171 + 114 = 285$. सही उत्तर है $1\,311$. उसने क्या किया?
भूल गए?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g4:mult:9}
एक माली पौधे लगाता है $14$ की पंक्तियाँ $35$ ट्यूलिप और $12$ की पंक्तियाँ $28$
डैफोडील्स कुल कितने फूल? (दो \omterm{def:g2:mult:def}{उत्पाद}ों, फिर ए \omterm{def:g1:addition:def}{जोड़}.)
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g4:mult:10}
हर दिन, एक बेकरी का उपयोग करता है $67$ किलो आटा. आटा कितना है इसके बारे में
वह एक में $30$-दिन महीना --- तो फिर, पूर्ण संख्याओं के साथ अनुमान लगाएं
सटीक गणना करें.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g4:mult:11}
इस अध्याय की तरह एक आयताकार चित्र के साथ समझाएँ, क्यों
$15 \times 12$ के रूप में गणना की जा सकती है $15 \times 10 + 15 \times 2$.
फिर एक और विभाजन का आविष्कार करें $15 \times 12$ वह भी काम करता है.
\end{exercise}
