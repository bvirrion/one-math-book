\chapter{विभाजन}\label{ch:g4:division}

ग्रेड 3 को टेबल और छोटे से विभाजित किया गया है \omterm{def:g3:division:remainder}{शेष}
(\cref{ch:g3:division}); यह अध्याय पूर्ण लिखित स्थापित करता है
विधि --- दीर्घ विभाजन --- जो किसी भी संख्या, अंक समूह को संभालती है
अंक समूह.

\section{दीर्घ विभाजन विधि}

\begin{method}[एक अंक वाली संख्या से लंबा विभाजन]\label{met:g4:division:method}
अलग करना $749$ द्वारा $6$:
\begin{enumerate}
  \item सबसे बायां अंक लें, $7$ (सैकड़ों): $6$ जाता है $1$ समय
        में $7$; लिखना $1$ में \omterm{def:g3:division:remainder}{भागफल}, घटाना
        $6$, \omterm{def:g3:division:remainder}{शेष} $1$;
  \item \emph{उतार देना} अगला अंक, $4$: अब बांटो $14$
        (दसियों): $6 \times 2 = 12$; लिखना $2$, \omterm{def:g3:division:remainder}{शेष} $2$;
  \item नीचे लाओ $9$: विभाजित करना $29$: $6 \times 4 = 24$; लिखना
        $4$, \omterm{def:g3:division:remainder}{शेष} $5$;
  \item कोई अंक नहीं बचा: द \omterm{def:g3:division:remainder}{भागफल} है $124$, द \omterm{def:g3:division:remainder}{शेष} $5$.
\end{enumerate}
जाँच करना: $6 \times 124 + 5 = 744 + 5 = 749$, और $5 < 6$.
\end{method}

\begin{example}[भागफल में शून्य]\label{ex:g4:division:zero}
विभाजित करना $618$ द्वारा $3$: सैकड़ों, $6 \div 3 = 2$; दसियों, नीचे लाओ
$1$: $3$ जाता है $0$ बार में $1$ --- \emph{लिखना} $0$! ---
\omterm{def:g3:division:remainder}{शेष} $1$; इकाइयाँ, नीचे लाएँ $8$: $18 \div 3 = 6$. \omterm{def:g3:division:remainder}{भागफल}
$206$, \omterm{def:g3:division:remainder}{शेष} $0$. बीच को भूल जाना \omterm{def:g1:counting:zero}{शून्य} (लिखना $26$)
क्लासिक गलती है; चेक
$3 \times 26 = 78 \neq 618$ तुरंत पकड़ लेता है.
\end{example}

\begin{example}[भागफल के आकार का अनुमान लगाना]\label{ex:g4:division:size}
बंटवारा करने से पहले $749$ द्वारा $6$, उत्तर फ्रेम करें:
$6 \times 100 = 600$ और $6 \times 200 = 1\,200$, इतना \omterm{def:g3:division:remainder}{भागफल} के बीच है $100$ और $200$ --- इसमें तीन अंक होंगे. यह एक पंक्ति
अनुमान अधिकांश गलत अंकों वाली त्रुटियों को रोकता है।
\end{example}

\section{यह चुनना कि प्रश्न क्या पूछता है}

\begin{method}[भागफल, शेषफल, या भागफल प्लस एक]\label{met:g4:division:interpret}
के रूप में \cref{met:g3:division:interpret}, कहानी तय करती है:
\begin{enumerate}
  \item ``कितने भरे \dots'' या ``प्रत्येक कितने'': \omterm{def:g3:division:remainder}{भागफल};
  \item ``कितने बचे'': द \omterm{def:g3:division:remainder}{शेष};
  \item ``सभी के लिए कितने कंटेनरों की आवश्यकता है'': \omterm{def:g3:division:remainder}{भागफल}, प्लस
        एक अगर \omterm{def:g3:division:remainder}{शेष} क्या नहीं है \omterm{def:g1:counting:zero}{शून्य}.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g4:division:problem}
$530$ पुस्तकों को बक्सों में पैक किया जाना चाहिए $8$.
\[
530 = 8 \times 66 + 2 .
\]
कितने \emph{भरा हुआ} बक्से? $66$. कितनी किताबें पूरी नहीं हैं
बक्सा? $2$. कितने डिब्बे पैक करने हैं \emph{सभी} किताबें? $67$.
तीन प्रश्न, एक प्रभाग।
\end{example}

\begin{example}[पैसे का उचित बंटवारा]\label{ex:g4:division:money}
चार मित्र एक की लागत साझा करते हैं $92$ समान रूप से उपहार:
$92 \div 4 = 23$ बिल्कुल (\omterm{def:g3:division:remainder}{शेष} $0$). प्रत्येक भुगतान करता है $23$. जब
\omterm{def:g3:division:remainder}{शेष} है \omterm{def:g1:counting:zero}{शून्य}, विभाजन ``समता से बाहर आता है'' और साझाकरण
बिल्कुल निष्पक्ष है.
\end{example}

\section{अभ्यास}

\begin{exercise}[$\star $]\label{exo:g4:division:1}
प्रत्येक को फ़्रेम करें \omterm{def:g3:division:remainder}{भागफल} के रूप में \cref{ex:g4:division:size} (जिसके बीच
दो ``\omterm{def:g4:numbers:round}{गोल}'' संख्याएं?), फिर लंबे विभाजन की गणना करें:
$96 \div 4$; \quad $87 \div 5$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:division:2}
लंबे विभाजनों की गणना करें और प्रत्येक की जाँच करें:
$672 \div 4$; \quad $925 \div 7$; \quad $804 \div 6$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:division:3}
के साथ सावधान \omterm{def:g1:counting:zero}{शून्य} (देखना \cref{ex:g4:division:zero}):
$816 \div 4$; \quad $420 \div 7$; \quad $2\,512 \div 5$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:division:4}
चेक समानता लिखें ($ a = b \times q + r $) के लिए: $85 \div 9$;
\quad $1\,000 \div 3$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:division:5}
$156$ के बक्सों में अंडे डाल दिए जाते हैं $6$. कितने डिब्बे भरे हैं?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:division:6}
का एक रिबन $250$ सेमी के टुकड़ों में काटा जाता है $8$ सेमी। कितने टुकड़े,
और कितनी लंबाई बची है?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:division:7}
$375$ की बसों में एक शो में जाते छात्र $52$ सीटें. कितनी बसें
आवश्यक हैं? (कौन सा मामला
\cref{met:g4:division:interpret} क्या यह? द्वारा विभाजित करें $52$ का उपयोग कर रहे हैं
एकाधिक: $52 \times 7 = 364$.)
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:division:8}
तीन दोस्त शेयर करते हैं $234$ समान रूप से कंचे. प्रत्येक में कितने कंचे?
गुणा करके जांचें.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g4:division:9}
लाभांश ज्ञात करें: द्वारा विभाजन $7$ देता है \omterm{def:g3:division:remainder}{भागफल} $58$ और
\omterm{def:g3:division:remainder}{शेष} $3$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g4:division:10}
एक लाइब्रेरियन शेल्फ $438$ किताबें, $9$ प्रति शेल्फ. अलमारियाँ आती हैं
की किताबों की अलमारी $6$ अलमारियाँ। कितनी अलमारियों की आवश्यकता है? कितने
किताबों की अलमारियाँ खरीदी जानी चाहिए? (दो डिवीजन, दोनों ए के साथ
``प्लस वन'' प्रश्न।)
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g4:division:11}
द्वारा एक प्रभाग में $8$, द \omterm{def:g3:division:remainder}{भागफल} के बराबर है \omterm{def:g3:division:remainder}{शेष}. क्या हैं?
संभावित लाभांश? उन सभी की सूची बनाएं. (द \omterm{def:g3:division:remainder}{शेष} रहना चाहिए
नीचे $8$.)
\end{exercise}
