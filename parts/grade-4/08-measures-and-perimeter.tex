\chapter{Measures and Perimeter}\label{ch:g4:measure}

Grade 3 measured with rulers, scales and clocks
(\cref{ch:g3:measure}); this chapter adds the units table --- the
machine that converts everything --- and two ways of measuring a flat
shape: the length of its border (perimeter) and the room inside it
(area, counted in squares for now).

\section{The units table}

\begin{method}[Converting with the table]\label{met:g4:measure:table}
For lengths, each column is worth ten of the next:
\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{c|c|c|c|c|c|c}
km & hm & dam & m & dm & cm & mm \\
\hline
 & & & $3$ & $4$ & $0$ & \\
\end{tabular}
\end{center}
\begin{enumerate}
  \item Write the number with one digit per column, the units digit
        of the measure in its unit's column (here $3.4$ m: the $3$
        under m);
  \item read the number in the new unit: $3.4$ m $= 340$ cm (fill
        empty columns with zeros);
  \item masses (kg\,\dots\,g) and capacities (L\,\dots\,cL) work the
        same way.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g4:measure:convert}
$5$ km $= 5\,000$ m; \quad $270$ cm $= 2.7$ m (the $2$ lands under
m); \quad $3$ kg $250$ g $= 3\,250$ g; \quad
$2.5$ L $= 250$ cL.
\end{example}

\section{Perimeter}

\begin{definition}[Perimeter]\label{def:g4:measure:perimeter}
The \emph{perimeter}\index{perimeter} of a polygon is the total
length of its border: add the lengths of all its sides (in the same
unit!). Shortcuts for the classics:
\[
\text{rectangle: } P = 2 \times (L + w),
\qquad
\text{square: } P = 4 \times c .
\]
\end{definition}

\begin{example}\label{ex:g4:measure:perimeter}
A rectangular garden $12$ m by $7$ m:
\[
P = 2 \times (12 + 7) = 2 \times 19 = 38 \text{ m}
\]
of fence. A square photo frame of side $18$ cm:
$P = 4 \times 18 = 72$ cm.
\end{example}

\section{Area: counting squares}

\begin{definition}[Area by counting]\label{def:g4:measure:area}
The \emph{area}\index{area} of a figure drawn on a grid is the number
of grid squares it covers. Two half-squares count as one. Area
answers ``how much surface'', perimeter answers ``how long is the
border'' --- two different questions!
\end{definition}

\begin{omfigure}
\begin{tikzpicture}[scale=0.55]
  \draw[gray!40, thin] (0,0) grid (13,4);
  \draw[very thick, omDef, fill=omDef!15] (1,1) rectangle (5,3);
  \node[font=\small, omDef] at (3,2) {$8$ squares};
  \draw[very thick, omThm, fill=omThm!15]
    (7,1) -- (11,1) -- (11,2) -- (9,2) -- (9,3) -- (7,3) -- cycle;
  \node[font=\small, omThm] at (8.4,1.5) {$6$ squares};
\end{tikzpicture}

{\small Two figures on the grid: areas $8$ and $6$ squares. Now walk
the borders and count the units: $12$ for both! Same perimeter,
different areas.}
\end{omfigure}

\begin{example}[Half-squares]\label{ex:g4:measure:half}
A right triangle drawn on the grid with legs $4$ and $2$ covers $3$
full squares and $2$ half-squares\,\dots\ count carefully: its area
is half of the $4 \times 2$ rectangle, i.e.\ $4$ squares. Cutting a
rectangle along its diagonal always gives two triangles of equal
area.
\end{example}

\begin{example}[Same area, different perimeters]\label{ex:g4:measure:samearea}
A $4 \times 4$ square and an $8 \times 2$ rectangle both cover $16$
squares --- same area. Borders: $16$ units for the square, $20$ for
the rectangle. Neither measure determines the other: a long thin
shape has lots of border for little surface.
\end{example}

\section{Durations}

\begin{method}[Adding times across the hour]\label{met:g4:measure:time}
To add or subtract durations, work in hops through the full hours
(recall $1$ h $= 60$ min):
\[
9\text{ h }40 + 35 \text{ min}:
\quad 9\text{ h }40 \xrightarrow{+20} 10\text{ h }00
\xrightarrow{+15} 10\text{ h }15 .
\]
Never add minutes past $60$ without converting: $40 + 35 = 75$ min
$= 1$ h $15$ min.
\end{method}

\begin{example}\label{ex:g4:measure:timetable}
The train leaves at $8{:}47$ and arrives at $11{:}05$. Duration:
$8{:}47 \to 9{:}00$ is $13$ min; $9{:}00 \to 11{:}00$ is $2$ h;
$11{:}00 \to 11{:}05$ is $5$ min. Total: $2$ h $18$ min.
\end{example}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:g4:measure:1}
Convert with the units table: $7$ m in cm; $3.2$ km in m; $85$ mm in
cm; $4\,500$ m in km.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:measure:2}
Convert: $2$ kg in g; $1\,800$ g in kg and g; $3.5$ L in cL;
$25$ cL in L.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:measure:3}
Compute the perimeter: a rectangle $9$ cm $\times$ $5$ cm; a square
of side $7.5$ cm; a triangle with sides $6$ cm, $8$ cm and $10$ cm.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:measure:4}
A rectangle has perimeter $26$ cm and length $8$ cm. Find its width
(first find length $+$ width).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:measure:5}
On grid paper, draw an L-shaped figure covering exactly $10$
squares, and count the units of its perimeter.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:measure:6}
Draw two different rectangles of area $12$ squares. Compute the
perimeter of each. Which is ``rounder'', which is ``thinner''?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:measure:7}
A right triangle is drawn on the grid with legs $6$ and $3$. What is
its area in squares? (Use \cref{ex:g4:measure:half}.)
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:measure:8}
Compute the durations: from $10{:}20$ to $12{:}45$; from $7{:}35$ to
$9{:}10$. A film lasts $1$ h $50$ and starts at $20{:}30$: when does
it end?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g4:measure:9}
Kim runs $3$ laps of a rectangular field $60$ m by $45$ m. How many
meters does she run?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g4:measure:10}
A farmer wants to fence a square field of side $85$ m, leaving a
gate of $4$ m without fence. How many meters of fence must he buy?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g4:measure:11}
True or false, with grid examples: ``if two figures have the same
perimeter, they have the same area''; ``if two figures have the same
area, they have the same perimeter''.
(\Cref{ex:g4:measure:samearea} and the chapter's figure will help.)
\end{exercise}
