\chapter{Decimal Numbers}\label{ch:g5:decimals}

Tenths and hundredths appeared in \cref{ch:g4:decimals}; adding one
more level, thousandths, completes the decimal system. This chapter
consolidates reading, decomposing and --- above all --- comparing
decimal numbers without falling into their famous traps.

\section{Down to thousandths}

\begin{definition}[Decimal places]\label{def:g5:decimals:places}
After the \omterm{def:g4:decimals:point}{decimal point} come the tenths, the hundredths, the
thousandths --- each worth ten times less than the one before:
\[
4.362 = 4 + \frac{3}{10} + \frac{6}{100} + \frac{2}{1000}
= \frac{4\,362}{1000}.
\]
\end{definition}

\begin{example}[Many readings, one number]\label{ex:g5:decimals:readings}
$4.362$ can be read ``four point three six two'', or ``four and
$362$ thousandths'', or decomposed as $4 + 0.3 + 0.06 + 0.002$. And
padding with final \omterm{def:g1:counting:zero}{zeros} changes nothing: $4.362 = 4.3620$. But
$4.362 \neq 4.0362$: a \omterm{def:g1:counting:zero}{zero} \emph{inside} pushes every digit to a
smaller place.
\end{example}

\begin{example}[Where is it on the line?]\label{ex:g5:decimals:line}
To place $4.362$: between $4$ and $5$; zooming, between $4.3$ and
$4.4$; zooming again, between $4.36$ and $4.37$, closer to $4.36$.
Each decimal digit is one more level of zoom.
\end{example}

\section{Comparing decimals}

\begin{method}[Comparing two decimals]\label{met:g5:decimals:compare}
\begin{enumerate}
  \item Compare the whole parts: $6.1 > 5.987$.
  \item Equal whole parts: compare the decimal parts digit by digit
        from the point --- tenths, then hundredths, then
        thousandths;
  \item padding with final \omterm{def:g1:counting:zero}{zeros} makes the comparison fair:
        $2.5$ vs $2.48$ becomes $2.50$ vs $2.48$, and $50 > 48$
        hundredths.
\end{enumerate}
The trap, one last time: \emph{a longer decimal part does not mean a
bigger number} --- $2.48$ has more digits than $2.5$ and is smaller.
\end{method}

\begin{example}\label{ex:g5:decimals:order}
Order $7.3$; $7.09$; $7.31$; $7.299$. Pad to three decimals:
$7.300$; $7.090$; $7.310$; $7.299$. Then
\[
7.09 < 7.299 < 7.3 < 7.31 .
\]
Note $7.299 < 7.3$ even though $299$ looks big: $299$ thousandths
against $300$ thousandths.
\end{example}

\begin{example}[Squeezing between two decimals]\label{ex:g5:decimals:between}
Is there a number between $5.7$ and $5.8$? Yes, plenty: $5.75$,
$5.71$, $5.799$\,\dots\ Between $5.79$ and $5.8$? Again plenty:
$5.795$, for one. Between two different decimals, one can
\emph{always} squeeze another --- there is no ``next'' decimal
number.
\end{example}

\section{Rounding decimals}

\begin{definition}[Rounding]\label{def:g5:decimals:round}
To \omterm{def:g4:numbers:round}{round} to the unit (or tenth, or hundredth), look at the next
digit: $5$ or more \omterm{def:g4:numbers:round}{rounds} up, otherwise \omterm{def:g4:numbers:round}{round} down. So $4.362$
\omterm{def:g4:numbers:round}{rounds} to $4$ (unit), to $4.4$ (tenth), to $4.36$ (hundredth).
\end{definition}

\begin{example}[Money rounds to cents]\label{ex:g5:decimals:money}
A price computed as $7.4983$ is displayed as $7.50$: real-life
amounts are rounded to the hundredth. Note the cascade: $7.4983 \to
7.50$, where the $49$ became $50$ --- \omterm{def:g4:numbers:round}{rounding} can ripple through
several digits.
\end{example}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:g5:decimals:1}
Write as a decimal number: $5 + \frac{2}{10} + \frac{7}{1000}$;
\quad $\frac{85}{100}$; \quad $\frac{4\,507}{1000}$; \quad ``twelve
and nine thousandths''.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:decimals:2}
In $27.418$: what does the $4$ count? The $8$? Decompose the number
as in \cref{def:g5:decimals:places}.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:decimals:3}
True or false? Explain with places.
$3.60 = 3.6$; \quad $0.5 = 0.05$; \quad $2.070 = 2.07$; \quad
$1.3 = 1.03$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:decimals:4}
Copy and complete with $<$, $>$ or $=$:
\[
4.8 \;?\; 4.79, \qquad
0.302 \;?\; 0.32, \qquad
6.5 \;?\; 6.500, \qquad
9.09 \;?\; 9.1 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:decimals:5}
Order from smallest to biggest: $3.14$; \ $3.4$; \ $3.041$; \
$3.104$; \ $3.41$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:decimals:6}
Between which two whole numbers does each number lie? Between which
two numbers with one decimal? \quad $6.83$; \quad $0.492$; \quad
$12.06$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:decimals:7}
Round $8.276$: to the unit; to the tenth; to the hundredth. Round
$3.95$ to the tenth.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:decimals:8}
Give a number strictly between: $2.6$ and $2.7$; \quad $0.41$ and
$0.42$; \quad $5.99$ and $6$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:decimals:9}
The heights of four plants are $0.85$ m, $1.2$ m, $0.9$ m and
$1.02$ m. Order them from shortest to tallest.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g5:decimals:10}
Lucie says: ``$7.12 > 7.9$ because $12 > 9$''. Correct her mistake,
comparing tenths first.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g5:decimals:11}
Find all the numbers with exactly two decimal digits that \omterm{def:g4:numbers:round}{round} to
$6.4$ when rounded to the tenth. What are the smallest and the
biggest?
\end{exercise}
