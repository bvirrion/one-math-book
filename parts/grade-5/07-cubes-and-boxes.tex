\chapter{Cubes and Boxes}\label{ch:g5:solids}

Leaving the flat page: \omterm{def:g5:solids:def}{solids} are the shapes of the real world ---
dice, cereal boxes, cans, balls. This chapter learns to name them,
count their \omterm{def:g5:solids:def}{faces} and \omterm{def:g5:solids:def}{edges}, unfold them into \omterm{def:g5:solids:net}{nets}, and measure
their \omterm{def:g5:solids:volume}{volume} by counting little \omterm{def:g5:solids:def}{cubes}. (\omterm{def:g5:solids:volume}{Volume} formulas come in
\cref{ch:g6:measure}.)

\section{The solid family}

\begin{definition}[Solids and their vocabulary]\label{def:g5:solids:def}
A \emph{solid}\index{solid} occupies space. Flat sides are
\emph{faces}\index{face}; faces meet along \emph{edges}\index{edge};
edges meet at \emph{vertices}\index{vertex}. The family
portraits:
\begin{itemize}
  \item the \emph{cube}\index{cube}: $6$ square faces, $12$ edges,
        $8$ vertices;
  \item the \emph{box}\index{box} (rectangular prism): $6$
        rectangular faces, $12$ edges, $8$ vertices;
  \item the \emph{cylinder}\index{cylinder}: two disk faces and one rolled face
        (a can);
  \item the \emph{pyramid}\index{pyramid}: a \omterm{def:g4:geometry:polygon}{polygon} base and triangles meeting at
        an apex;
  \item the \emph{ball} (sphere): one perfectly round surface, no
        face, no edge.
\end{itemize}
\end{definition}

\begin{example}[Counting on the cube]\label{ex:g5:solids:count}
Check on a die: $6$ \omterm{def:g5:solids:def}{faces} (the numbers $1$ to $6$), $8$ \omterm{def:g5:solids:def}{vertices}
(the corners), $12$ \omterm{def:g5:solids:def}{edges} (run a finger along them: $4$ on top, $4$
at the bottom, $4$ vertical). And
$6 + 8 = 12 + 2$ --- the same curious pattern holds for the \omterm{def:g5:solids:def}{box} and
the \omterm{def:g5:solids:def}{pyramid} (\cref{ex:g8:solids:count} returns to it).
\end{example}

\section{Nets}

\begin{definition}[Net]\label{def:g5:solids:net}
A \emph{net}\index{net} of a \omterm{def:g5:solids:def}{solid} is a flat drawing of all its
\omterm{def:g5:solids:def}{faces}, attached along \omterm{def:g5:solids:def}{edges}, that folds up into the \omterm{def:g5:solids:def}{solid} --- the
\omterm{def:g5:solids:def}{solid} unwrapped like a cardboard \omterm{def:g5:solids:def}{box} flattened for recycling.
\end{definition}

\begin{example}[Reading a net]\label{ex:g5:solids:read}
On a die, opposite \omterm{def:g5:solids:def}{faces} add up to $7$. On a \omterm{def:g5:solids:net}{net}, opposite \omterm{def:g5:solids:def}{faces} are
the ones separated by exactly one square in a row (or around a
corner) --- not the neighbors! Marking the pairs on the \omterm{def:g5:solids:net}{net} before
folding is a great exercise in seeing space flat.
\end{example}

\section{Volume: counting cubes}

\begin{definition}[Volume by counting]\label{def:g5:solids:volume}
The \emph{volume}\index{volume} of a \omterm{def:g5:solids:def}{solid} built from unit \omterm{def:g5:solids:def}{cubes} is
the number of \omterm{def:g5:solids:def}{cubes} it contains. The unit can be the centimeter \omterm{def:g5:solids:def}{cube}
(cm$^3$) or the meter \omterm{def:g5:solids:def}{cube} (m$^3$).
\end{definition}

\begin{example}[Counting a box layer by layer]\label{ex:g5:solids:layers}
A \omterm{def:g5:solids:def}{box} $4$ \omterm{def:g5:solids:def}{cubes} long, $3$ wide, $2$ high: the bottom layer holds
$4 \times 3 = 12$ \omterm{def:g5:solids:def}{cubes}, and there are $2$ layers:
\[
12 \times 2 = 24 \text{ cubes} .
\]
Counting by layers always works --- and it quietly proves the
formula $V = L \times w \times h$ of \cref{ch:g6:measure}.
\end{example}

\begin{example}[Same cubes, different shapes]\label{ex:g5:solids:rearrange}
Eight unit \omterm{def:g5:solids:def}{cubes} can build a $2 \times 2 \times 2$ \omterm{def:g5:solids:def}{cube}, a
$8 \times 1 \times 1$ stick, or a $4 \times 2 \times 1$ slab: three
different \omterm{def:g5:solids:def}{solids}, one \omterm{def:g5:solids:volume}{volume} ($8$ \omterm{def:g5:solids:def}{cubes}). \omterm{def:g5:solids:volume}{Volume} counts the
material, not the shape.
\end{example}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:g5:solids:1}
Name a real object shaped like: a \omterm{def:g5:solids:def}{cube}; a \omterm{def:g5:solids:def}{box}; a \omterm{def:g5:solids:def}{cylinder}; a ball;
a \omterm{def:g5:solids:def}{pyramid}.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:solids:2}
How many \omterm{def:g5:solids:def}{faces}, \omterm{def:g5:solids:def}{edges} and \omterm{def:g5:solids:def}{vertices} has a \omterm{def:g5:solids:def}{box}? Check the pattern
\omterm{def:g5:solids:def}{faces} $+$ \omterm{def:g5:solids:def}{vertices} $=$ \omterm{def:g5:solids:def}{edges} $+ 2$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:solids:3}
Which \omterm{def:g5:solids:def}{faces} of a \omterm{def:g5:solids:def}{box} are identical? (Group the six \omterm{def:g5:solids:def}{faces} in pairs.)
What is special about the \omterm{def:g5:solids:def}{cube}'s \omterm{def:g5:solids:def}{faces}?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:solids:4}
Draw a \omterm{def:g5:solids:net}{net} of a \omterm{def:g5:solids:def}{cube} of side $2$ cm (use the cross-shaped model of
the chapter), cut it out and fold it.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:solids:5}
On the cross-shaped \omterm{def:g5:solids:net}{net} of a die, place the numbers $1$ to $6$ so
that opposite \omterm{def:g5:solids:def}{faces} sum to $7$. (Use \cref{ex:g5:solids:read} to
find the opposite pairs first.)
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:solids:6}
Count the unit \omterm{def:g5:solids:def}{cubes}: a \omterm{def:g5:solids:def}{box} $5$ long, $2$ wide, $3$ high; a \omterm{def:g5:solids:def}{cube} of
side $3$; an L-shaped stack made of a $3 \times 2 \times 1$ slab
with a $1 \times 2 \times 1$ slab on top.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:solids:7}
A \omterm{def:g5:solids:def}{box} holds exactly $12$ unit \omterm{def:g5:solids:def}{cubes} in one layer. Give two possible
layer shapes ($L \times w$), and for each, the \omterm{def:g5:solids:def}{box}'s \omterm{def:g5:solids:volume}{volume} if it
has $3$ such layers.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:solids:8}
With $27$ unit \omterm{def:g5:solids:def}{cubes}, what \omterm{def:g5:solids:def}{cube} can you build? With $64$? (Think of
the layer counting backwards.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g5:solids:9}
Sugar \omterm{def:g5:solids:def}{cubes} of side $1$ cm come in a \omterm{def:g5:solids:def}{box} $10$ cm long, $5$ cm wide,
$4$ cm high (interior sizes). How many sugar \omterm{def:g5:solids:def}{cubes} fit? If a
family uses $6$ \omterm{def:g5:solids:def}{cubes} a day, for how many days does one \omterm{def:g5:solids:def}{box} last
(\omterm{def:g4:numbers:round}{round} sensibly)?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g5:solids:10}
A $3 \times 3 \times 3$ \omterm{def:g5:solids:def}{cube} is painted red on the outside, then cut
into $27$ unit \omterm{def:g5:solids:def}{cubes}. How many small \omterm{def:g5:solids:def}{cubes} have paint on exactly
$3$ \omterm{def:g5:solids:def}{faces}? On $2$? On $1$? On none? (Check: the four counts must add
up to $27$.)
\end{exercise}
