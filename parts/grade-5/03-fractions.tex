\chapter{Fractions as Numbers}\label{ch:g5:fractions}

A \omterm{def:g4:fractions:def}{fraction} is not just a picture of a pie: it is a genuine
\emph{number}, with a place on the number line --- sometimes the very
same place as a decimal number. This chapter connects the two
worlds. (The full rules of \omterm{def:g4:fractions:def}{fraction} arithmetic come in
\cref{ch:g6:fractions} and beyond.)

\section{Fractions on the line, again}

\begin{example}[Placing and reading]\label{ex:g5:fractions:place}
On a line cut in fifths, $\frac{7}{5}$ is $7$ steps from $0$: past
$1$ (which is $\frac55$), at $1 + \frac25$. Every \omterm{def:g4:fractions:def}{fraction} lives
somewhere on the line; bigger \omterm{def:g4:fractions:def}{numerator} (same \omterm{def:g4:fractions:def}{denominator}) means
further right.
\end{example}

\begin{proposition}[Same point, many names]\label{prop:g5:fractions:equal}
Cutting every part in two (or three\,\dots) does not move the
point:
\[
\frac{1}{2} = \frac{2}{4} = \frac{3}{6} = \frac{5}{10},
\qquad
\frac{2}{3} = \frac{4}{6} = \frac{20}{30}.
\]
Multiplying (or dividing) the \omterm{def:g4:fractions:def}{numerator} and the \omterm{def:g4:fractions:def}{denominator} by the
same number gives another name of the same \omterm{def:g4:fractions:def}{fraction}.
\end{proposition}
\admitted

\section{Fractions and decimals}

\begin{proposition}[Fractions with decimal names]\label{prop:g5:fractions:decimal}
\omterm{def:g4:fractions:def}{Fractions} of tenths, hundredths, thousandths \emph{are} decimal
numbers:
\[
\frac{3}{10} = 0.3, \qquad \frac{27}{100} = 0.27 .
\]
Some other \omterm{def:g4:fractions:def}{fractions} also have decimal names, found by switching to
tenths or hundredths:
\[
\frac{1}{2} = \frac{5}{10} = 0.5, \qquad
\frac{1}{4} = \frac{25}{100} = 0.25, \qquad
\frac{3}{4} = 0.75, \qquad
\frac{1}{5} = \frac{2}{10} = 0.2 .
\]
But not all: $\frac13 = 0.333\dots$ never ends --- the \omterm{def:g4:fractions:def}{fraction} is
its only exact name.
\end{proposition}
\admitted

\begin{example}[Choosing the handier name]\label{ex:g5:fractions:choose}
To pay three \omterm{def:g3:division:half}{quarters} of $12$: the \omterm{def:g4:fractions:def}{fraction} name is handier,
$\frac34$ of $12$ is $9$. To compare $\frac34$ and $0.8$: the
decimal name is handier, $0.75 < 0.8$. Being bilingual pays off.
\end{example}

\section{Comparing fractions}

\begin{method}[Three comparison tools]\label{met:g5:fractions:compare}
\begin{enumerate}
  \item \emph{Same \omterm{def:g4:fractions:def}{denominator}}: more parts wins,
        $\frac57 > \frac37$;
  \item \emph{same \omterm{def:g4:fractions:def}{numerator}}: bigger parts win, so \emph{smaller}
        \omterm{def:g4:fractions:def}{denominator} wins: $\frac35 > \frac38$ (fifths are bigger
        than eighths);
  \item \emph{landmarks}: compare each to $\frac12$ or to $1$:
        $\frac38 < \frac12 < \frac59$, so $\frac38 < \frac59$.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g5:fractions:compare}
Who ate more pizza: Ali with $\frac58$ or Bea with $\frac54$? Bea's
\omterm{def:g4:fractions:def}{fraction} is bigger than $1$ (five \omterm{def:g3:division:half}{quarters} is more than one whole
pizza), Ali's is smaller than $1$: Bea ate more --- more than a
whole pizza, in fact.
\end{example}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:g5:fractions:1}
Draw a number line cut in fifths from $0$ to $2$, and place:
$\frac25$; \ $\frac55$; \ $\frac85$; \ $\frac{10}{5}$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:fractions:2}
Complete the equal names:
$\frac12 = \frac{?}{8}$; \quad
$\frac34 = \frac{?}{8}$; \quad
$\frac{2}{5} = \frac{4}{?}$; \quad
$\frac{6}{9} = \frac{2}{?}$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:fractions:3}
Which \omterm{def:g4:fractions:def}{fractions} name whole numbers? Give the whole number when so:
$\frac82$; \quad $\frac{9}{4}$; \quad $\frac{15}{3}$; \quad
$\frac{20}{5}$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:fractions:4}
Write as decimals: $\frac{7}{10}$; \quad $\frac{41}{100}$; \quad
$\frac12$; \quad $\frac34$; \quad $\frac15$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:fractions:5}
Write as \omterm{def:g4:fractions:def}{fractions} (tenths or hundredths): $0.9$; \quad $0.13$;
\quad $2.5$; \quad $0.75$ (then give its \omterm{def:g3:division:half}{quarter} name).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:fractions:6}
Compare, naming the tool used
(\cref{met:g5:fractions:compare}):
\[
\frac47 \;?\; \frac67, \qquad
\frac35 \;?\; \frac37, \qquad
\frac25 \;?\; \frac{7}{12}, \qquad
\frac98 \;?\; \frac{11}{12} .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:fractions:7}
Order with the decimal names: $\frac12$; \ $0.4$; \ $\frac34$; \
$0.6$; \ $\frac15$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:fractions:8}
A \omterm{def:g3:division:half}{quarter} of the students of a class of $28$ play an instrument, and
\omterm{def:g3:division:half}{half} play a sport. How many students is each? Which group is bigger?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:fractions:9}
Match each \omterm{def:g4:fractions:def}{fraction} with a time: $\frac14$ h, $\frac12$ h,
$\frac34$ h, $\frac{1}{60}$ h --- to $30$ min, $45$ min, $1$ min,
$15$ min.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g5:fractions:10}
Tom drank $\frac35$ of his $50$ cL bottle; Ana drank $\frac12$ of
her $60$ cL bottle. Who drank more? (Compute both amounts in cL ---
comparing the bare \omterm{def:g4:fractions:def}{fractions} is not enough here, and say why.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g5:fractions:11}
Give a \omterm{def:g4:fractions:def}{fraction} strictly between $\frac12$ and $1$; then one
strictly between $\frac12$ and $\frac34$. (Decimal names may
help.)
\end{exercise}
