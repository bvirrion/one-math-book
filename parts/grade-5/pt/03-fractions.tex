\chapter{Frações como Números}\label{ch:g5:fractions}

UM \omterm{def:g4:fractions:def}{fração} não é apenas a imagem de uma torta: é um verdadeiro
\emph{número}, com um lugar na reta numérica --- às vezes o próprio
mesma casa que um número decimal. Este capítulo conecta os dois
mundos. (As regras completas de \omterm{def:g4:fractions:def}{fração} aritmética entra
\cref{ch:g6:fractions} e além.)

\section{Frações na linha, novamente}

\begin{example}[Colocação e leitura]\label{ex:g5:fractions:place}
Numa linha cortada em quintos, $\frac{7}{5}$ é $7$ passos de $0$: passado
$1$ (que é $\frac55$), no $1 + \frac25$. Todo \omterm{def:g4:fractions:def}{fração} vidas
em algum lugar na linha; maior \omterm{def:g4:fractions:def}{numerador} (mesmo \omterm{def:g4:fractions:def}{denominador}) significa
mais à direita.
\end{example}

\begin{proposition}[Mesmo ponto, muitos nomes]\label{prop:g5:fractions:equal}
Cortar cada parte em duas (ou três\,\dots) não move o
ponto:
\[
\frac{1}{2} = \frac{2}{4} = \frac{3}{6} = \frac{5}{10},
\qquad
\frac{2}{3} = \frac{4}{6} = \frac{20}{30}.
\]
Multiplicando (ou dividindo) o \omterm{def:g4:fractions:def}{numerador} e o \omterm{def:g4:fractions:def}{denominador} pelo
mesmo número dá outro nome do mesmo \omterm{def:g4:fractions:def}{fração}.
\end{proposition}
\admitted

\section{Frações e decimais}

\begin{proposition}[Frações com nomes decimais]\label{prop:g5:fractions:decimal}
Frações de décimos, centésimos, milésimos \emph{são} decimal
números:
\[
\frac{3}{10} = 0.3, \qquad \frac{27}{100} = 0.27 .
\]
Algum outro frações também têm nomes decimais, encontrados mudando para
décimos ou centésimos:
\[
\frac{1}{2} = \frac{5}{10} = 0.5, \qquad
\frac{1}{4} = \frac{25}{100} = 0.25, \qquad
\frac{3}{4} = 0.75, \qquad
\frac{1}{5} = \frac{2}{10} = 0.2 .
\]
Mas nem todos: $\frac13 = 0.333\dots $ nunca acaba --- o \omterm{def:g4:fractions:def}{fração} é
seu único nome exato.
\end{proposition}
\admitted

\begin{example}[Escolhendo o nome mais prático]\label{ex:g5:fractions:choose}
Para pagar três \omterm{def:g3:division:half}{trimestres} de $12$: o \omterm{def:g4:fractions:def}{fração} o nome é mais prático,
$\frac34$ de $12$ é $9$. Para comparar $\frac34$ e $0.8$: o
o nome decimal é mais prático, $0.75 < 0.8$. Ser bilíngue compensa.
\end{example}

\section{Comparando frações}

\begin{method}[Três ferramentas de comparação]\label{met:g5:fractions:compare}
\begin{enumerate}
  \item \emph{Mesmo \omterm{def:g4:fractions:def}{denominador}}: mais peças ganham,
        $\frac57 > \frac37$;
  \item \emph{mesmo \omterm{def:g4:fractions:def}{numerador}}: peças maiores ganham, então \emph{menor}
        \omterm{def:g4:fractions:def}{denominador} ganha: $\frac35 > \frac38$ (quintos são maiores
        que oitavos);
  \item \emph{pontos de referência}: compare cada um com $\frac12$ ou para $1$:
        $\frac38 < \frac12 < \frac59$, então $\frac38 < \frac59$.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g5:fractions:compare}
Quem comeu mais pizza: Ali com $\frac58$ ou Bia com $\frac54$? Bia
\omterm{def:g4:fractions:def}{fração} é maior que $1$ (cinco \omterm{def:g3:division:half}{trimestres} é mais que um todo
pizza), Ali's é menor que $1$: Bea comeu mais --- mais do que um
pizza inteira, na verdade.
\end{example}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g5:fractions:1}
Desenhe uma reta numérica cortada em quintos de $0$ para $2$ e coloque:
$\frac25$; \ $\frac55$; \ $\frac85$; \ $\frac{10}{5}$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:fractions:2}
Complete os nomes iguais:
$\frac12 = \frac{?}{8}$; \quad
$\frac34 = \frac{?}{8}$; \quad
$\frac{2}{5} = \frac{4}{?}$; \quad
$\frac{6}{9} = \frac{2}{?}$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:fractions:3}
Qual frações nomear números inteiros? Dê o número inteiro quando assim:
$\frac82$; \quad $\frac{9}{4}$; \quad $\frac{15}{3}$; \quad
$\frac{20}{5}$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:fractions:4}
Escreva como decimais: $\frac{7}{10}$; \quad $\frac{41}{100}$; \quad
$\frac12$; \quad $\frac34$; \quad $\frac15$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:fractions:5}
Escreva como frações (décimos ou centésimos): $0.9$; \quad $0.13$;
\quad $2.5$; \quad $0.75$ (então dê o seu \omterm{def:g3:division:half}{trimestre} nome).
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:fractions:6}
Compare, nomeando a ferramenta usada
(\cref{met:g5:fractions:compare}):
\[
\frac47 \;?\; \frac67, \qquad
\frac35 \;?\; \frac37, \qquad
\frac25 \;?\; \frac{7}{12}, \qquad
\frac98 \;?\; \frac{11}{12} .
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:fractions:7}
Ordene com os nomes decimais: $\frac12$; \ $0.4$; \ $\frac34$; \
$0.6$; \ $\frac15$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:fractions:8}
UM \omterm{def:g3:division:half}{trimestre} dos alunos de uma turma de $28$ tocar um instrumento e
\omterm{def:g3:division:half}{metade} praticar um esporte. Quantos alunos tem cada um? Qual grupo é maior?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:fractions:9}
Combine cada um \omterm{def:g4:fractions:def}{fração} com um tempo: $\frac14$ h, $\frac12$ h,
$\frac34$ h, $\frac{1}{60}$ h --- para $30$ min, $45$ min, $1$ min,
$15$ min.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g5:fractions:10}
Tom bebeu $\frac35$ dele $50$ garrafa cL; Ana bebeu $\frac12$ de
ela $60$ garrafa cL. Quem bebeu mais? (Calcule ambos os valores em cL ---
comparando o nu frações não é suficiente aqui e diga por quê.)
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g5:fractions:11}
Dê um \omterm{def:g4:fractions:def}{fração} estritamente entre $\frac12$ e $1$; então um
estritamente entre $\frac12$ e $\frac34$. (Nomes decimais podem
ajuda.)
\end{exercise}
