\chapter{Números decimais}\label{ch:g5:decimals}

Décimos e centésimos apareceram em \cref{ch:g4:decimals}; adicionando um
mais nível, milésimos, completa o sistema decimal. Este capítulo
consolida a leitura, a decomposição e --- sobretudo --- a comparação
números decimais sem cair em suas famosas armadilhas.

\section{Até milésimos}

\begin{definition}[Casas decimais]\label{def:g5:decimals:places}
Depois do \omterm{def:g4:decimals:point}{ponto decimal} vêm os décimos, os centésimos, o
milésimos --- cada um valendo dez vezes menos que o anterior:
\[
4.362 = 4 + \frac{3}{10} + \frac{6}{100} + \frac{2}{1000}
= \frac{4\,362}{1000}.
\]
\end{definition}

\begin{example}[Muitas leituras, um número]\label{ex:g5:decimals:readings}
$4.362$ pode ser lido ``quatro vírgula três seis dois'', ou ``quatro e
$362$ milésimos'', ou decomposto como $4 + 0.3 + 0.06 + 0.002$. E
preenchimento com final \omterm{def:g1:counting:zero}{zeros} não muda nada: $4.362 = 4.3620$. Mas
$4.362 \neq 4.0362$: um \omterm{def:g1:counting:zero}{zero} \emph{dentro} empurra cada dígito para um
lugar menor.
\end{example}

\begin{example}[Onde está na linha?]\label{ex:g5:decimals:line}
Para colocar $4.362$: entre $4$ e $5$; zoom, entre $4.3$ e
$4.4$; ampliando novamente, entre $4.36$ e $4.37$, mais perto de $4.36$.
Cada dígito decimal é mais um nível de zoom.
\end{example}

\section{Comparando decimais}

\begin{method}[Comparando duas casas decimais]\label{met:g5:decimals:compare}
\begin{enumerate}
  \item Compare as partes inteiras: $6.1 > 5.987$.
  \item Partes inteiras iguais: compare as partes decimais dígito por dígito
        do ponto --- décimos, depois centésimos, então
        milésimos;
  \item preenchimento com final \omterm{def:g1:counting:zero}{zeros} torna a comparação justa:
        $2.5$ contra $2.48$ torna-se $2.50$ contra $2.48$, e $50 > 48$
        centésimos.
\end{enumerate}
A armadilha, uma última vez: \emph{uma parte decimal mais longa não significa um
número maior} --- $2.48$ tem mais dígitos do que $2.5$ e é menor.
\end{method}

\begin{example}\label{ex:g5:decimals:order}
Ordem $7.3$; $7.09$; $7.31$; $7.299$. Preencha com três casas decimais:
$7.300$; $7.090$; $7.310$; $7.299$. Então
\[
7.09 < 7.299 < 7.3 < 7.31 .
\]
Observação $7.299 < 7.3$ embora $299$ parece grande: $299$ milésimos
contra $300$ milésimos.
\end{example}

\begin{example}[Espremendo entre duas casas decimais]\label{ex:g5:decimals:between}
Existe um número entre $5.7$ e $5.8$? Sim, bastante: $5.75$,
$5.71$, $5.799$\,\dots\ Entre $5.79$ e $5.8$? Novamente bastante:
$5.795$, por exemplo. Entre duas casas decimais diferentes, pode-se
\emph{sempre} aperte outro --- não há ``próximo'' decimal
número.
\end{example}

\section{Arredondando decimais}

\begin{definition}[Arredondamento]\label{def:g5:decimals:round}
Para redondo para a unidade (ou décimo, ou centésimo), veja o próximo
dígito: $5$ ou mais rodadas para cima, caso contrário redondo abaixo. Então $4.362$
rodadas para $4$ (unidade), para $4.4$ (décimo), para $4.36$ (centésimo).
\end{definition}

\begin{example}[O dinheiro é arredondado para centavos]\label{ex:g5:decimals:money}
Um preço calculado como $7.4983$ é exibido como $7.50$: vida real
os valores são \omterm{def:g4:numbers:round}{arredondados} para o centésimo. Observe a cascata: $7.4983 \to
7.50$, onde o $49$ tornou-se $50$ --- arredondamento pode ondular
vários dígitos.
\end{example}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g5:decimals:1}
Escreva como um número decimal: $5 + \frac{2}{10} + \frac{7}{1000}$;
\quad $\frac{85}{100}$; \quad $\frac{4\,507}{1000}$; \quad ``doze
e nove milésimos''.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:decimals:2}
Em $27.418$: o que o $4$ contar? O $8$? Decomponha o número
como em \cref{def:g5:decimals:places}.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:decimals:3}
Verdadeiro ou falso? Explique com lugares.
$3.60 = 3.6$; \quad $0.5 = 0.05$; \quad $2.070 = 2.07$; \quad
$1.3 = 1.03$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:decimals:4}
Copie e complete com $<$, $>$ ou $=$:
\[
4.8 \;?\; 4.79, \qquad
0.302 \;?\; 0.32, \qquad
6.5 \;?\; 6.500, \qquad
9.09 \;?\; 9.1 .
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:decimals:5}
Ordene do menor para o maior: $3.14$; \ $3.4$; \ $3.041$; \
$3.104$; \ $3.41$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:decimals:6}
Entre quais dois números inteiros está cada número? Entre os quais
dois números com uma casa decimal? \quad $6.83$; \quad $0.492$; \quad
$12.06$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:decimals:7}
Redondo $8.276$: para a unidade; ao décimo; ao centésimo. Redondo
$3.95$ ao décimo.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:decimals:8}
Dê um número estritamente entre: $2.6$ e $2.7$; \quad $0.41$ e
$0.42$; \quad $5.99$ e $6$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:decimals:9}
As alturas de quatro plantas são $0.85$ m, $1.2$ m, $0.9$ eu e
$1.02$ m. Ordene-os do mais baixo para o mais alto.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g5:decimals:10}
Lúcia diz: ``$7.12 > 7.9$ porque $12 > 9$''. Corrija o erro dela,
comparando os décimos primeiro.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g5:decimals:11}
Encontre todos os números com exatamente duas casas decimais que redondo para
$6.4$ quando \omterm{def:g4:numbers:round}{arredondado} para o décimo. Quais são os menores e os
maior?
\end{exercise}
