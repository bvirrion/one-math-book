\chapter{Computando com Decimais}\label{ch:g5:decimalops}

Os números decimais somam, subtraem e multiplicam quase como números inteiros
--- toda a arte é saber para onde vai a questão. Este capítulo
cobre os métodos de coluna e a magia de multiplicar ou dividir por
$10$, $100$, $1000$. (Multiplicar duas casas decimais espera por
\cref{ch:g6:decimals}.)

\section{Adição e subtração}

\begin{method}[Colunas com um ponto]\label{met:g5:decimalops:addsub}
\begin{enumerate}
  \item Escreva os números com \emph{pontos decimais alinhado}
        (então as unidades estão abaixo das unidades, décimos abaixo dos décimos);
  \item preencha a parte decimal mais curta com \omterm{def:g1:counting:zero}{zeros};
  \item adicionar ou subtrair como acontece com números inteiros;
  \item deixe cair o ponto do resultado diretamente para baixo.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g5:decimalops:addsub}
$27.5 + 3.86$ (almofada $27.5 = 27.50$):
\[
\begin{array}{r}
2\,7.5\,0 \\
+\ \ \ 3.8\,6 \\
\hline
3\,1.3\,6
\end{array}
\qquad\qquad
\begin{array}{r}
1\,2.4\,0 \\
-\ \ \ 5.6\,5 \\
\hline
\ \ \,6.7\,5
\end{array}
\]
Para a subtração ($12.4 - 5.65$): preencher, pedir emprestado como de costume e verificar
adicionando de volta: $6.75 + 5.65 = 12.4$.
\end{example}

\begin{example}[Complementos mentais]\label{ex:g5:decimalops:complement}
O que deve ser adicionado $7.6$ alcançar $10$? Escalar: $7.6 \to 8$ é
$0.4$, então $8 \to 10$ é $2$: o complemento é $2.4$. Útil para
mudança: pagar $10$ por um $7.60$ item, obter $2.40$ voltar.
\end{example}

\section{Dez vezes maior, dez vezes menor}

\begin{proposition}[Vezes e dividido por 10, 100, 1000]\label{prop:g5:decimalops:shift}
Multiplicando por $10$ faz com que cada dígito valha dez vezes mais: o
dígitos deslizam uma casa para a esquerda --- que se parece com o \emph{ponto
movendo um lugar certo}. A divisão faz o oposto:
\[
3.47 \times 10 = 34.7, \qquad
3.47 \times 100 = 347, \qquad
52.1 \div 10 = 5.21, \qquad
6 \div 100 = 0.06 .
\]
\end{proposition}
\admitted

\begin{example}[Conversões novamente]\label{ex:g5:decimalops:convert}
As unidades de medida são potências de dez separadas, então as conversões são apenas
essas mudanças: $2.35$ eu $= 235$ cm ($\times 100$); $470$ g
$= 0.47$ kg ($\div 1000$); $1.5$ eu $= 150$ cl.
\end{example}

\section{Multiplicando um número decimal por um número inteiro}

\begin{method}[Decimais vezes inteiras]\label{met:g5:decimalops:mult}
Calcule como se não houvesse sentido e depois dê o resultado tantos
dígitos decimais como fator decimal:
\[
4.35 \times 6: \quad 435 \times 6 = 2\,610
\ \to\ 4.35 \times 6 = 26.10 = 26.1 .
\]
Faça uma estimativa para colocar o ponto com confiança: $4.35$ está perto de
$4$, e $4 \times 6 = 24$ --- então $26.1$, não $2.61$ nem $261$.
\end{method}

\begin{example}[Adição repetida ainda funciona]\label{ex:g5:decimalops:mult}
$0.75 \times 4 = 0.75 + 0.75 + 0.75 + 0.75 = 3$: quatro \omterm{def:g3:division:half}{trimestres} fazer
um todo, em roupas decimais. Multiplicar por um número inteiro ainda é
``tantas vezes''.
\end{example}

\begin{example}[Compras]\label{ex:g5:decimalops:shopping}
Três cadernos em $2.45$ cada um e uma caneta em $1.20$:
\begin{enumerate}
  \item cadernos: $2.45 \times 3 = 7.35$;
  \item total: $7.35 + 1.20 = 8.55$;
  \item mudar de $10$: $10 - 8.55 = 1.45$.
\end{enumerate}
\end{example}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g5:decimalops:1}
Calcule em colunas: $45.7 + 8.93$; \quad $16.25 + 3.75$; \quad
$0.86 + 12.4$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:decimalops:2}
Calcule em colunas e verifique: $9.4 - 2.72$; \quad $20 - 13.45$;
\quad $6.03 - 5.9$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:decimalops:3}
Complementos mentais para $10$ (como em
\cref{ex:g5:decimalops:complement}): $6.2$; \quad $3.75$; \quad
$9.99$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:decimalops:4}
Calcule sem colunas:
\[
7.24 \times 10, \qquad
0.58 \times 100, \qquad
34.5 \div 10, \qquad
7 \div 100, \qquad
0.3 \times 1000 .
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:decimalops:5}
Converter: $4.05$ m em cm; \quad $325$ cm em m; \quad $0.6$ kg em g;
\quad $85$ cL em L.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:decimalops:6}
Estime primeiro e depois calcule: $6.2 \times 4$; \quad
$3.45 \times 8$; \quad $12.5 \times 6$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:decimalops:7}
Uma volta de uma pista mede $0.4$ km. Quanto tempo dura $7$ voltas? E
$25$ voltas?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:decimalops:8}
Léa compra $2$ baguetes em $1.15$ cada um e um bolo em $12.60$. Ela
paga com um $20$ conta. Calcule seu total e sua mudança.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:decimalops:9}
Uma garrafa contém $1.5$ L. Quantos litros tem um pacote de $6$ garrafas?
Uma família bebe $0.75$ L por dia: por quantos dias uma embalagem
último? (Quantas vezes $0.75$ encaixar $9$?)
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g5:decimalops:10}
Milo calcula $5.6 \times 3 = 15.18$, raciocinando ``$5 \times 3 = 15$
e $6 \times 3 = 18$''. Use uma estimativa para mostrar que a resposta deve ser
errado, então calcule corretamente.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g5:decimalops:11}
As quatro voltas de um corredor duraram $65.4$ é, $64.9$ é, $65.0$ areia
$66.1$ S.
\begin{enumerate}
  \item Calcule o tempo total das quatro voltas.
  \item O recorde de quatro voltas é $260$ S. Ela venceu? Por como
        muito?
\end{enumerate}
\end{exercise}
