\chapter{Cubos e Caixas}\label{ch:g5:solids}

Saindo da página plana: \omterm{def:g5:solids:def}{sólidos} são as formas do mundo real ---
dados, \omterm{def:g5:solids:def}{caixas} de cereais, latas, bolas. Este capítulo aprende a nomeá-los,
conte seus \omterm{def:g5:solids:def}{rostos} e \omterm{def:g5:solids:def}{bordas}, desdobre-os em redese medir
seu \omterm{def:g5:solids:volume}{volume} contando pouco \omterm{def:g5:solids:def}{cubos}. (\omterm{def:g5:solids:volume}{Volume} entram as fórmulas
\cref{ch:g6:measure}.)

\section{A família sólida}

\begin{definition}[Sólidos e seu vocabulário]\label{def:g5:solids:def}
UM \emph{sólido}\index{sólido} ocupa espaço. Os lados planos são
\emph{rostos}\index{rostos}; rostos se encontram \emph{bordas}\index{bordas};
bordas se encontram em \emph{vértices}\index{vértices}. A família
retratos:
\begin{itemize}
  \item o \emph{cubo}\index{cubo}: $6$ rostos quadrados, $12$ bordas,
        $8$ vértices;
  \item o \emph{caixa}\index{caixa} (prisma retangular): $6$
        faces retangulares, $12$ bordas, $8$ vértices;
  \item o \emph{cilindro}\index{cilindro}: duas faces de disco e uma face laminada
        (uma lata);
  \item o \emph{pirâmide}\index{pirâmide}: um \omterm{def:g4:geometry:polygon}{polígono} base e triângulos se encontrando em
        um ápice;
  \item o \emph{bola} (esfera): uma superfície perfeitamente redonda, não
        cara, sem borda.
\end{itemize}
\end{definition}

\begin{example}[Contando com o cubo]\label{ex:g5:solids:count}
Verifique em um dado: $6$ \omterm{def:g5:solids:def}{rostos} (os números $1$ para $6$), $8$ \omterm{def:g5:solids:def}{vértices}
(os cantos), $12$ \omterm{def:g5:solids:def}{bordas} (passe um dedo sobre eles: $4$ no topo, $4$
na parte inferior, $4$ vertical). E
$6 + 8 = 12 + 2$ --- o mesmo padrão curioso se aplica ao \omterm{def:g5:solids:def}{caixa} e
o \omterm{def:g5:solids:def}{pirâmide} (\cref{ex:g8:solids:count} retorna a ele).
\end{example}

\section{Redes}

\begin{definition}[Líquido]\label{def:g5:solids:net}
UM \emph{líquido}\index{líquido} de um \omterm{def:g5:solids:def}{sólido} é um desenho plano de todos os seus
\omterm{def:g5:solids:def}{rostos}, anexado junto \omterm{def:g5:solids:def}{bordas}, que se dobra no \omterm{def:g5:solids:def}{sólido} --- o
\omterm{def:g5:solids:def}{sólido} desembrulhado como um papelão \omterm{def:g5:solids:def}{caixa} achatado para reciclagem.
\end{definition}

\begin{example}[Lendo uma rede]\label{ex:g5:solids:read}
Em um dado, oposto \omterm{def:g5:solids:def}{rostos} somar a $7$. Em um \omterm{def:g5:solids:net}{líquido}, oposto \omterm{def:g5:solids:def}{rostos} são
aqueles separados por exatamente um quadrado consecutivo (ou em torno de um
canto) --- não os vizinhos! Marcando os \omterm{def:g3:numbers:evenodd}{pares} no \omterm{def:g5:solids:net}{líquido} antes
dobrar é um ótimo exercício para ver o espaço plano.
\end{example}

\section{Volume: contando cubos}

\begin{definition}[Volume por contagem]\label{def:g5:solids:volume}
O \emph{volume}\index{volume} de um \omterm{def:g5:solids:def}{sólido} construído a partir da unidade \omterm{def:g5:solids:def}{cubos} é
o número de \omterm{def:g5:solids:def}{cubos} ele contém. A unidade pode ser o centímetro \omterm{def:g5:solids:def}{cubo}
(cm $^3$) ou o medidor \omterm{def:g5:solids:def}{cubo} (m $^3$).
\end{definition}

\begin{example}[Contando uma caixa camada por camada]\label{ex:g5:solids:layers}
UM \omterm{def:g5:solids:def}{caixa} $4$ \omterm{def:g5:solids:def}{cubos} longo, $3$ largo, $2$ alto: a camada inferior contém
$4 \times 3 = 12$ \omterm{def:g5:solids:def}{cubos}, e há $2$ camadas:
\[
12 \times 2 = 24 \text{ cubes} .
\]
Contar por camadas sempre funciona --- e prova silenciosamente a
fórmula $ V = L \times w \times h $ de \cref{ch:g6:measure}.
\end{example}

\begin{example}[Mesmos cubos, formas diferentes]\label{ex:g5:solids:rearrange}
Oito unidades \omterm{def:g5:solids:def}{cubos} pode construir um $2 \times 2 \times 2$ \omterm{def:g5:solids:def}{cubo}, um
$8 \times 1 \times 1$ bastão, ou um $4 \times 2 \times 1$ laje: três
diferente \omterm{def:g5:solids:def}{sólidos}, um \omterm{def:g5:solids:volume}{volume} ($8$ \omterm{def:g5:solids:def}{cubos}). \omterm{def:g5:solids:volume}{Volume} conta o
material, não a forma.
\end{example}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g5:solids:1}
Nomeie um objeto real com a forma de: um \omterm{def:g5:solids:def}{cubo}; um \omterm{def:g5:solids:def}{caixa}; um \omterm{def:g5:solids:def}{cilindro}; uma bola;
um \omterm{def:g5:solids:def}{pirâmide}.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:solids:2}
Quantos \omterm{def:g5:solids:def}{rostos}, \omterm{def:g5:solids:def}{bordas} e \omterm{def:g5:solids:def}{vértices} tem um \omterm{def:g5:solids:def}{caixa}? Verifique o padrão
\omterm{def:g5:solids:def}{rostos} $+$ \omterm{def:g5:solids:def}{vértices} $=$ \omterm{def:g5:solids:def}{bordas} $+ 2$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:solids:3}
Qual \omterm{def:g5:solids:def}{rostos} de um \omterm{def:g5:solids:def}{caixa} são idênticos? (Agrupe os seis \omterm{def:g5:solids:def}{rostos} em \omterm{def:g3:numbers:evenodd}{pares}.)
O que há de especial no cubode \omterm{def:g5:solids:def}{rostos}?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:solids:4}
Desenhe um \omterm{def:g5:solids:net}{líquido} de um \omterm{def:g5:solids:def}{cubo} de lado $2$ cm (use o modelo em forma de cruz de
capítulo), recorte e dobre.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:solids:5}
Na forma de cruz \omterm{def:g5:solids:net}{líquido} de um dado, coloque os números $1$ para $6$ então
aquele oposto \omterm{def:g5:solids:def}{rostos} \omterm{def:g1:addition:def}{soma} para $7$. (Usar \cref{ex:g5:solids:read} para
encontre os \omterm{def:g3:numbers:evenodd}{pares} opostos primeiro.)
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:solids:6}
Conte a unidade \omterm{def:g5:solids:def}{cubos}: um \omterm{def:g5:solids:def}{caixa} $5$ longo, $2$ largo, $3$ alto; um \omterm{def:g5:solids:def}{cubo} de
lado $3$; uma pilha em forma de L feita de $3 \times 2 \times 1$ laje
com um $1 \times 2 \times 1$ laje por cima.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:solids:7}
UM \omterm{def:g5:solids:def}{caixa} segura exatamente $12$ unidade \omterm{def:g5:solids:def}{cubos} em uma camada. Dê duas possíveis
formas de camada ($ L \times w $), e para cada um, o caixade \omterm{def:g5:solids:volume}{volume} se for
tem $3$ tais camadas.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:solids:8}
Com $27$ unidade \omterm{def:g5:solids:def}{cubos}, o que \omterm{def:g5:solids:def}{cubo} você pode construir? Com $64$? (Pense em
a camada contando regressivamente.)
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g5:solids:9}
Açúcar \omterm{def:g5:solids:def}{cubos} de lado $1$ cm vem em um \omterm{def:g5:solids:def}{caixa} $10$ cm de comprimento, $5$ cm de largura,
$4$ cm de altura (tamanhos interiores). Quanto açúcar \omterm{def:g5:solids:def}{cubos} ajustar? Se um
família usa $6$ \omterm{def:g5:solids:def}{cubos} por dia, por quantos dias alguém \omterm{def:g5:solids:def}{caixa} último
(redondo sensatamente)?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g5:solids:10}
UM $3 \times 3 \times 3$ \omterm{def:g5:solids:def}{cubo} é pintado de vermelho por fora e depois cortado
em $27$ unidade \omterm{def:g5:solids:def}{cubos}. Quantos pequenos \omterm{def:g5:solids:def}{cubos} tem tinta exatamente
$3$ \omterm{def:g5:solids:def}{rostos}? Sobre $2$? Sobre $1$? Em nenhum? (Verifique: as quatro contagens devem somar
até $27$.)
\end{exercise}
