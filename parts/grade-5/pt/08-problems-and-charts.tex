\chapter{Problemas e gráficos}\label{ch:g5:data}

O último capítulo do ano coloca tudo para funcionar: várias etapas
problemas com dinheiro, quantidades e durações, e os gráficos que
apresentar informações rapidamente. A ideia estrela é ``por um'' ---
encontrar o preço de um único item --- que crescerá em
proporcionalidade em \cref{ch:g6:propdata}.

\section{Problemas de várias etapas}

\begin{method}[Domando um longo problema]\label{met:g5:data:steps}
\begin{enumerate}
  \item Leia até o fim; diga qual é a pergunta final;
  \item planeje de trás para frente: para responder, o que preciso primeiro?
  \item resolva as etapas em ordem, escrevendo uma operação e uma
        frase curta por etapa;
  \item verifique a resposta final contra o bom senso e contra uma
        estimativa.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g5:data:multistep}
``Uma escola ordena $4$ \omterm{def:g5:solids:def}{caixas} de $25$ livros em $6$ cada um e paga
$180$ para entrega em cima. Qual é o custo total?
\begin{enumerate}
  \item Livros: $4 \times 25 = 100$ livros.
  \item Custo do livro: $100 \times 6 = 600$.
  \item Total: $600 + 180 = 780$.
\end{enumerate}
Verificação da estimativa: cerca de cem livros em $6$ é sobre $600$, mais
sobre $200$: sobre $800$ --- a resposta $780$ é plausível.
\end{example}

\begin{example}[Por um]\label{ex:g5:data:perone}
``$5$ custo de canecas idênticas $12.50$; o que fazer $8$ custo das canecas?
Encontre o preço de \emph{um} caneca primeiro:
\[
12.50 \div 5 = 2.50,
\qquad\text{then}\qquad
8 \times 2.50 = 20 .
\]
Através de um, tudo se torna acessível – este padrão de duas etapas
resolve toda uma família de problemas (receitas, velocidades, preços) e é
a semente de \cref{ch:g6:propdata}.
\end{example}

\begin{example}[Mudança e orçamento]\label{ex:g5:data:budget}
Zoe tem $30$. Ela compra um livro em $12.40$ e duas canetas em $2.30$
cada. Ela também pode pagar um $13$ jogo de tabuleiro?
\begin{enumerate}
  \item Canetas: $2 \times 2.30 = 4.60$.
  \item Gastei até agora: $12.40 + 4.60 = 17$.
  \item \omterm{def:g3:division:remainder}{Restante}: $30 - 17 = 13$ --- exatamente o suficiente para o jogo,
        sem mais nada.
\end{enumerate}
\end{example}

\section{Lendo gráficos}

\begin{method}[Lendo qualquer gráfico]\label{met:g5:data:read}
\begin{enumerate}
  \item Leia o título: o que está sendo contado?
  \item Leia os eixos ou a chave: o que significa uma graduação, uma barra,
        um símbolo representa?
  \item Só então responda às perguntas --- apontando para o gráfico com um
        a régua ajuda a ler as alturas com exatidão.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g5:data:chart}
Precipitação em uma cidade, mês a mês:

\begin{omfigure}
\begin{tikzpicture}
\begin{axis}[
  width=8.8cm, height=5.0cm,
  ybar, bar width=13pt,
  symbolic x coords={Jan,Feb,Mar,Apr,May,Jun},
  xtick=data,
  ymin=0, ymax=90,
  ytick={20,40,60,80},
  ylabel={rain (mm)},
  tick label style={font=\small},
  label style={font=\small}]
  \addplot[fill=omDef!30, draw=omDef] coordinates
    {(Jan,60) (Feb,45) (Mar,55) (Apr,70) (May,80) (Jun,30)};
\end{axis}
\end{tikzpicture}

{\small Um gráfico de barras de precipitação. Uma formatura $= 20$ mm.}
\end{omfigure}

Leituras: o mês mais chuvoso é maio ($80$ milímetros); Junho tem \omterm{def:g3:division:half}{metade} de
Chuva de abril ($30$ contra $70$? não --- \omterm{def:g3:division:half}{metade} de $70$ é $35$, então
\emph{não exatamente} \omterm{def:g3:division:half}{metade}: ler com precisão importa!); total para o
seis meses: $60 + 45 + 55 + 70 + 80 + 30 = 340$ mm.
\end{example}

\begin{example}[Uma quantidade variável]\label{ex:g5:data:line}
A altura de uma planta, medida todas as semanas:

\begin{omfigure}
\begin{tikzpicture}
\begin{axis}[omaxis,
  width=8.4cm, height=5.0cm,
  xmin=0, xmax=6.5, ymin=0, ymax=24,
  xlabel={week}, ylabel={height (cm)},
  xtick={1,2,3,4,5,6}, ytick={5,10,15,20}]
  \addplot[omDef, thick, mark=*, mark size=1.6pt] coordinates
    {(1,3) (2,6) (3,10) (4,15) (5,19) (6,21)};
\end{axis}
\end{tikzpicture}

{\small Unir os pontos de medição mostra o \emph{crescimento}: o
quanto mais íngreme o segmento, mais rápido a planta cresceu naquela semana.}
\end{omfigure}

A planta cresceu mais rápido entre semanas $3$ e $4$ ($+5$ centímetros) e
desacelerou no final ($+2$ cm na semana $6$).
\end{example}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g5:data:1}
Resolva em etapas: ``Um clube aluga um ônibus por $240$ e compra $32$ museu
ingressos em $7$ cada. Qual é o custo total da viagem?''
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:data:2}
Resolva com ``por um'': ``$3$ custo de kg de maçãs $7.50$. O que fazer
$5$ custo em kg?''
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:data:3}
Resolva: ``Seis amigos compartilham igualmente o $87$ custo de um piquenique. Como
quanto cada um paga?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:data:4}
Sam tem $25$. Ele compra uma camiseta em $13.90$ e um boné em $8.50$. Como
quanto dinheiro ele deixou?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:data:5}
Usando o gráfico de precipitação de \cref{ex:g5:data:chart}: quais meses
tenho mais do que $50$ mm? Quanta chuva caiu mais em maio do que em
Junho?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:data:6}
Usando o gráfico da planta de \cref{ex:g5:data:line}: qual era a altura do
plantar na semana $2$? Entre quais semanas cresceu exatamente
$4$ cm?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:data:7}
Desenhe um gráfico de barras para o número de gols que uma equipe marcou:
Setembro $4$, outubro $7$, novembro $3$, dezembro $6$. Escolha o
formatura e dê o total.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:data:8}
Um pacote de $8$ custos de iogurtes $3.20$; vendido sozinho, custa um iogurte
$0.50$. Calcule o preço “por um” no pacote e quanto custa
economiza por iogurte comprando o pacote.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g5:data:9}
``Um cinema vendido $148$ ingressos em $9.50$ no sábado e $95$ ingressos
pelo mesmo preço no domingo. Quanto mais dinheiro o sábado trouxe
do que domingo?'' Resolva de duas maneiras: computando as receitas de ambos os dias, ou
computando com o \emph{\omterm{ex:g1:subtraction:difference}{diferença}} em ingressos --- e verifique isso
as respostas concordam.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g5:data:10}
Uma receita para $4$ necessidades das pessoas $300$ g de arroz. Aline cozinha para
$10$ pessoas. Quanto arroz ela precisa? (Por uma pessoa primeiro ---
decimais permitidos.)
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g5:data:11}
Invente um problema de três etapas usando dinheiro cuja resposta final seja
$5.50$, escreva sua solução completa no estilo de
\cref{met:g5:data:steps}e teste-o em um colega de classe.
\end{exercise}
