\chapter{Perímetro e Área}\label{ch:g5:areas}

Quadrados de grade contados da 4ª série (\cref{ch:g4:measure}); este capítulo
ganha o primeiro \omterm{def:g4:measure:area}{área} \emph{fórmula} --- comprimento vezes largura para o
retângulo --- e aprende as unidades cm $^2$ e eu $^2$. As fórmulas para
triângulos e outras formas amadurecem em \cref{ch:g6:measure} e
\cref{ch:g7:areas}.

\section{Duas medidas diferentes}

\begin{example}[Fronteira vs superfície]\label{ex:g5:areas:contrast}
Um jardineiro precisa \emph{\omterm{def:g4:measure:perimeter}{perímetro}} para comprar a cerca e \emph{\omterm{def:g4:measure:area}{área}}
para comprar a semente de grama. Os dois não se sucedem: alongamento
um retângulo mais fino e mais longo pode manter sua \omterm{def:g4:measure:area}{área} enquanto é \omterm{def:g4:measure:perimeter}{perímetro}
cresce --- compare um $6 \times 2$ e um $4 \times 3$ retângulo
(mesmo \omterm{def:g4:measure:area}{área} $12$, \omterm{def:g4:measure:perimeter}{perímetros} $16$ e $14$).
\end{example}

\section{A fórmula do retângulo}

\begin{proposition}[Área de um retângulo]\label{prop:g5:areas:rectangle}
Um retângulo de comprimento $ L $ e largura $ w $ (na mesma unidade) tem \omterm{def:g4:measure:area}{área}
\[
A = L \times w ,
\]
em unidades quadradas: cm $^2$ (quadrados de lado $1$ centímetros), m $^2$ (quadrados de
lado $1$ m), \dots
\end{proposition}

\begin{proof}[Por que, contando]
Cubra o retângulo com quadrados unitários: $ w $ linhas de $ L $ quadrados cada,
então $ L \times w $ quadrados no total --- exatamente a multiplicação
retângulo de \cref{ch:g3:mult}.
\end{proof}

\begin{omfigure}
\begin{tikzpicture}[scale=0.62]
  \draw[gray!50, thin] (0,0) grid (6,3);
  \draw[very thick, omDef] (0,0) rectangle (6,3);
  \fill[omThm!30] (0,2) rectangle (1,3);
  \draw[omThm, thick] (0,2) rectangle (1,3);
  \node[right, font=\small, omThm] at (1.15,2.5) {$1$ cm $^2$};
  \node[below, font=\small] at (3,-0.15) {$6$ cm};
  \node[left, font=\small] at (0,1.5) {$3$ cm};
  \node[font=\small] at (3.9,1.2) {$6 \times 3 = 18$ cm $^2$};
\end{tikzpicture}

{\small Três linhas de quadrados de seis centímetros: a fórmula
$ L \times w $ é a contagem, escrita de uma vez por todas.}
\end{omfigure}

\begin{example}\label{ex:g5:areas:rectangle}
Medidas de um tapete $2.5$ m por $2$ eu: \omterm{def:g4:measure:area}{área} $2.5 \times 2 = 5$ eu $^2$.
Um selo mede $3$ cm por $2.4$ cm: \omterm{def:g4:measure:area}{área} $3 \times 2.4 = 7.2$
cm $^2$. Mesma fórmula, qualquer unidade --- desde que ambos os lados usem o
\emph{mesmo} um.
\end{example}

\begin{example}[Unidades quadradas são convertidas em 100]\label{ex:g5:areas:units}
$1$ eu $= 100$ centímetros, mas $1$ eu $^2 = 100 \times 100 = 10\,000$ cm $^2$:
um metro quadrado é um $100$-por-$100$ grade de centímetros quadrados. \omterm{def:g4:measure:area}{Área}
as unidades saltam em centenas, não em dezenas.
\end{example}

\section{Figuras compostas}

\begin{method}[Cortar, adicionar, subtrair]\label{met:g5:areas:composite}
Para uma figura feita de retângulos (um L, um T, uma moldura):
\begin{enumerate}
  \item corte-o em retângulos ou complete-o em um retângulo grande;
  \item calcular cada retângulo \omterm{def:g4:measure:area}{área} com a fórmula;
  \item adicione as peças --- ou subtraia o buraco do grande
        retângulo;
  \item verifique com uma contagem aproximada de quadrados da grade.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g5:areas:composite}
Uma sala em forma de L: uma $6$ eu $\times $ $4$ m retângulo com um
$2$ eu $\times $ $2$ m canto faltando.
\[
\text{Area} = 6 \times 4 - 2 \times 2 = 24 - 4 = 20 \text{ m}^2 .
\]
Ou corte o L em um $6 \times 2$ tira e um $4 \times 2$ tira:
$12 + 8 = 20$ eu $^2$ --- duas estradas, uma resposta.
\end{example}

\begin{example}[Meio retângulo]\label{ex:g5:areas:halfrect}
Cortar um retângulo ao longo de uma diagonal dá dois triângulos iguais
\omterm{def:g4:measure:area}{área}: cada um é \emph{\omterm{def:g3:division:half}{metade}} o retângulo. Um triângulo retângulo com pernas
$6$ centímetros e $4$ cm, portanto, tem \omterm{def:g4:measure:area}{área} $\frac{6 \times 4}{2} = 12$
cm $^2$ --- uma imagem que vale a pena lembrar
\cref{ch:g6:measure}, onde se torna uma fórmula.
\end{example}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g5:areas:1}
Calcule o \omterm{def:g4:measure:area}{área} e o \omterm{def:g4:measure:perimeter}{perímetro} de um retângulo $8$ cm $\times $
$5$ cm. Qual resposta está em cm, qual está em cm $^2$?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:areas:2}
Calcule o \omterm{def:g4:measure:area}{áreas}: um quadrado de lado $9$ cm; um retângulo
$12$ eu $\times $ $7$ m; um retângulo $4.5$ cm $\times $ $6$ cm.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:areas:3}
Um retângulo tem \omterm{def:g4:measure:area}{área} $63$ cm $^2$ e comprimento $9$ cm. Encontre sua largura,
então é \omterm{def:g4:measure:perimeter}{perímetro}.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:areas:4}
Desenhe dois retângulos diferentes com \omterm{def:g4:measure:area}{área} $24$ quadrados em papel quadriculado,
e calcular ambos \omterm{def:g4:measure:perimeter}{perímetros}. Mesmo \omterm{def:g4:measure:area}{área} --- mesmo \omterm{def:g4:measure:perimeter}{perímetro}?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:areas:5}
Converter: $3$ eu $^2$ em cm $^2$; \quad $50\,000$ cm $^2$ em m $^2$.
(Lembrar \cref{ex:g5:areas:units}: às centenas!)
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:areas:6}
Uma figura em forma de T é feita de $8 \times 2$ barra horizontal no topo
um $2 \times 5$ barra vertical (em cm). Calcule seu \omterm{def:g4:measure:area}{área} adicionando dois
retângulos.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:areas:7}
Um porta-retratos: um $30$ cm $\times $ $20$ cm retângulo com um
$24$ cm $\times $ $14$ janela retangular de cm recortada. O que \omterm{def:g4:measure:area}{área} de
madeira a moldura usa?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:areas:8}
Calcule o \omterm{def:g4:measure:area}{área} de um triângulo retângulo com pernas $8$ centímetros e $6$ cm
(\omterm{def:g3:division:half}{metade} um retângulo, \cref{ex:g5:areas:halfrect}).
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:areas:9}
Uma horta retangular mede $7$ m por $4$ m. Custos de sementes
$2$ por metro quadrado e cerca $3$ por metro. Calcule o custo de
a semente, depois o custo da cerca.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g5:areas:10}
Um piso de corredor é $12$ sou longo e $2$ m de largura e deve ser coberto
com ladrilhos quadrados de lado $50$ cm. Quantas peças são necessárias?
(Converta primeiro ou conte as peças em cada direção.)
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g5:areas:11}
Dobre os lados de um $4$ cm $\times $ $3$ cm retângulo. O que acontece
para o seu \omterm{def:g4:measure:perimeter}{perímetro}? Para o seu \omterm{def:g4:measure:area}{área}? (Calcule antes e depois ---
as duas respostas são diferentes!)
\end{exercise}
