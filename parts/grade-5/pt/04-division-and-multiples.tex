\chapter{Divisão e Múltiplos}\label{ch:g5:division}

Divisão longa aprendida em \cref{ch:g4:division} recebe duas atualizações:
dois dígitos \omterm{def:g5:division:multiple}{divisores}, e \omterm{def:g3:division:remainder}{quocientes} que continuam além do ponto ---
$3 \div 4$ finalmente obtém uma resposta, $0.75$. O capítulo termina com
\omterm{def:g5:division:multiple}{múltiplos}, \omterm{def:g5:division:multiple}{divisores} e os primeiros atalhos de divisibilidade.

\section{Dividindo por um número de dois algarismos}

\begin{method}[Divisores de dois dígitos]\label{met:g5:division:twodigits}
O mesmo método de antes, com uma habilidade extra: adivinhar quantas vezes
o \omterm{def:g5:division:multiple}{divisor} cabe. Escreva uma pequena tabela de \omterm{def:g5:division:multiple}{múltiplos} do \omterm{def:g5:division:multiple}{divisor}
primeiro. Para $986 \div 23$ (\omterm{def:g5:division:multiple}{múltiplos} de 23: 23, 46, 69, 92, 115,
138, 161, 184, 207):
\begin{enumerate}
  \item $98$ dezenas: $23 \times 4 = 92$ cabe, $23 \times 5 = 115$ faz
        não: dígito $4$, \omterm{def:g3:division:remainder}{restante} $6$;
  \item derrubar o $6$: $66$: $23 \times 2 = 46$ cabe: dígito
        $2$, \omterm{def:g3:division:remainder}{restante} $20$;
  \item \omterm{def:g3:division:remainder}{quociente} $42$, \omterm{def:g3:division:remainder}{restante} $20$. Verificar:
        $23 \times 42 + 20 = 966 + 20 = 986$.
\end{enumerate}
\end{method}

\section{Quocientes decimais}

\begin{example}[A divisão que se recusa a parar no resto]\label{ex:g5:division:decimal}
Compartilhar $3$ pizzas entre $4$ crianças: $3 \div 4$ tem \omterm{def:g3:division:remainder}{quociente} $0$ e
\omterm{def:g3:division:remainder}{restante} $3$ --- inútil! Continue a divisão além do ponto:
\begin{enumerate}
  \item $3$ unidades $= 30$ décimos; $30 \div 4$: $7$ décimos, \omterm{def:g3:division:remainder}{restante}
        $2$ décimos;
  \item $2$ décimos $= 20$ centésimos; $20 \div 4 = 5$ centésimos,
        \omterm{def:g3:division:remainder}{restante} $0$.
\end{enumerate}
Então $3 \div 4 = 0.75$: cada criança recebe $0.75$ pizza --- três
\omterm{def:g3:division:half}{trimestres}, como o \omterm{def:g4:fractions:def}{fração} a língua já conhecia
(\cref{prop:g5:fractions:decimal}).
\end{example}

\begin{method}[Continuando uma divisão além do ponto]\label{met:g5:division:continue}
Quando as unidades estiverem esgotadas e um \omterm{def:g3:division:remainder}{restante} resta:
\begin{enumerate}
  \item escreva o \omterm{def:g4:decimals:point}{ponto decimal} no \omterm{def:g3:division:remainder}{quociente};
  \item anexar um \omterm{def:g1:counting:zero}{zero} para o \omterm{def:g3:division:remainder}{restante} (as unidades tornam-se décimos, décimos
        tornar-se centésimos\,\dots) e continue dividindo;
  \item pare quando o \omterm{def:g3:division:remainder}{restante} é $0$ --- ou quando os dígitos começam
        repetindo para sempre, como $1 \div 3 = 0.333\dots $: então dê um
        valor \omterm{def:g4:numbers:round}{arredondado}.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g5:division:continue}
$27 \div 6$: \omterm{def:g3:division:remainder}{quociente} $4$, \omterm{def:g3:division:remainder}{restante} $3$; apontar; $30$ décimos
$\div\ 6 = 5$: então $27 \div 6 = 4.5$. \quad
$22 \div 8$: $2$, \omterm{def:g3:division:remainder}{restante} $6$; então $60 \div 8 = 7$ R $4$;
então $40 \div 8 = 5$: então $22 \div 8 = 2.75$.
\end{example}

\section{Múltiplos e divisores}

\begin{definition}[Múltiplo, divisor]\label{def:g5:division:multiple}
O \emph{múltiplos}\index{múltiplos} de $6$ sua tabela continua
para sempre: $0, 6, 12, 18, 24, \dots $ Quando $24$ é um múltiplo de $6$,
um também diz que $6$ é um \emph{divisor}\index{divisor} de $24$,
ou isso $24$ é \emph{divisível} por $6$: a divisão
$24 \div 6$ não deixa \omterm{def:g3:division:remainder}{restante}.
\end{definition}

\begin{proposition}[Atalhos de divisibilidade]\label{prop:g5:division:criteria}
Sem divisão, um número é divisível:
\begin{itemize}
  \item por $2$ quando seu último dígito é $0$, $2$, $4$, $6$ ou $8$
        (o números \omterm{def:g3:numbers:evenodd}{pares});
  \item por $5$ quando seu último dígito é $0$ ou $5$;
  \item por $10$ quando seu último dígito é $0$;
  \item por $3$ quando o \emph{\omterm{def:g1:addition:def}{soma} dos seus dígitos} é divisível por
        $3$ (por exemplo\ $741$: $7 + 4 + 1 = 12$, divisível --- e de fato
        $741 = 3 \times 247$).
\end{itemize}
\end{proposition}
\admitted

\begin{example}\label{ex:g5:division:criteria}
É $534$ divisível por $2$? Termina em $4$: sim. Por $5$? Não. Por $3$?
$5 + 3 + 4 = 12$: sim. Então $534$ é divisível por $6$ também (por $2$
\emph{e} por $3$) --- verificar: $534 = 6 \times 89$.
\end{example}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g5:division:1}
Calcule as divisões longas (tabela de \omterm{def:g5:division:multiple}{múltiplos} primeiro):
$851 \div 23$; \quad $704 \div 32$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:division:2}
Calcule o decimal exato \omterm{def:g3:division:remainder}{quocientes}: $9 \div 2$; \quad $27 \div 4$;
\quad $33 \div 5$; \quad $21 \div 8$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:division:3}
Quatro amigos compartilham um $54$ conta do restaurante igualmente. Quanto custa
cada um paga? (Continue além do ponto.)
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:division:4}
Calcular $10 \div 3$, continuando três dígitos além do ponto. O que fazer
você observa? Forneça o valor \omterm{def:g4:numbers:round}{arredondado} às centésimas.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:division:5}
Lista: o \omterm{def:g5:division:multiple}{múltiplos} de $7$ até $70$; o \omterm{def:g5:division:multiple}{divisores} de $18$; o
\omterm{def:g5:division:multiple}{divisores} de $24$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:division:6}
Divisível por $2$? por $5$? por $10$? por $3$? Teste cada atalho em:
$470$; \quad $735$; \quad $8\,001$; \quad $1\,314$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:division:7}
Encontre todos os números entre $60$ e $80$ que são divisíveis por
$3$ \emph{e} por $5$. (Qual divisibilidade única faz isso
combinar?)
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:division:8}
Uma fita de $7.2$ m é cortado em $6$ peças iguais. Quanto tempo dura cada
peça? (Divida um decimal por um inteiro: divida os metros, depois o
décimos.)
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:division:9}
$1\,000$ bolinhas de gude são embaladas em sacos de $24$. Quantos sacos cheios e
quantas bolas de gude restam? (Dois dígitos \omterm{def:g5:division:multiple}{divisor}.)
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g5:division:10}
Pode $5$ amigos compartilham $7$ barras de chocolate igualmente, sem sobrar nada?
Dê cada ação como um decimal. A mesma pergunta para $7$ amigos compartilhando
$5$ barras --- o que acontece de diferente?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g5:division:11}
Usando o \omterm{def:g1:addition:def}{soma} de dígitos atalho, encontre o menor dígito $ d $ isso faz
$52d $ (um número de três dígitos terminado em $ d $) divisível por $3$. Então
encontre todos esses dígitos $ d$.
\end{exercise}
