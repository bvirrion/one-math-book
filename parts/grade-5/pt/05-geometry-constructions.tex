\chapter{Construções Geométricas}\label{ch:g5:geometry}

Régua, esquadro, compasso: com estas três ferramentas e um pouco de
método, qualquer figura descrita em palavras pode ser construída com exatidão. Isto
construções de capítulo \omterm{def:g4:geometry:def}{perpendiculares}, \omterm{def:g4:geometry:def}{paralelos}, triângulos e o
especial \omterm{def:g4:geometry:polygon}{quadriláteros} --- artesanato primeiro, teoria depois
(\cref{ch:g6:lines,ch:g6:shapes}).

\section{Perpendiculares e paralelas através de um ponto}

\begin{method}[As duas construções básicas]\label{met:g5:geometry:basic}
\emph{\omterm{def:g4:geometry:def}{Perpendicular} para $d $ através $ A $}: deslize o esquadro ao longo
$ d $ até que sua outra aresta em \omterm{def:g3:shapes:rightangle}{ângulo reto} alcance $ A $; empate; marque o
\omterm{def:g3:shapes:rightangle}{ângulo reto}.

\emph{\omterm{def:g4:geometry:def}{Paralelo} para $ d $ através $ A $}: pressione uma régua contra o aparelho
quadrado como um trilho colocado ao longo $ d $; deslize o esquadro para cima até $ A $;
empate. (Verifique: a lacuna entre as duas linhas é a mesma
em todos os lugares.)
\end{method}

\begin{example}[Encadeamento de construções]\label{ex:g5:geometry:chain}
Muitas figuras são cadeias desses dois movimentos. Um retângulo $ ABCD $ com
$ AB = 5$ centímetros e $ BC = 3$ cm: desenhar $[AB]$; \omterm{def:g4:geometry:def}{perpendicular} no $ A $;
\omterm{def:g4:geometry:def}{perpendicular} no $ B $; marca $ D $ e $ C $ no $3$ cm do mesmo lado;
juntar. Cada etapa é uma construção básica.
\end{example}

\section{Triângulos com a bússola}

\begin{method}[Triângulo de três lados]\label{met:g5:geometry:triangle}
Para construir um triângulo com lados $6$ cm, $5$ cm, $4$ cm:
\begin{enumerate}
  \item desenhe o lado mais longo, $[AB]$ com $ AB = 6$ cm;
  \item arco do centro $ A $ e \omterm{def:g3:shapes:shapes}{raio} $5$ cm;
  \item arco do centro $ B $ e \omterm{def:g3:shapes:shapes}{raio} $4$ cm;
  \item sua travessia é a terceira vértice $ C $; juntar.
\end{enumerate}
A bússola garante os comprimentos: cada ponto do primeiro arco é
exatamente $5$ cm de $ A $.
\end{method}

\begin{example}[Isósceles e equilátero de graça]\label{ex:g5:geometry:special}
Mesma abertura da bússola para ambos os arcos $\Rightarrow $ triângulo isósceles.
Abertura igual à base $\Rightarrow $ equilátero. A bússola
não mede: é \emph{cópias} comprimentos, o que é ainda melhor.
\end{example}

\section{Os quadriláteros especiais, por construção}

\begin{method}[Construindo a família]\label{met:g5:geometry:quadrilaterals}
\begin{enumerate}
  \item \emph{Retângulo}: dois \omterm{def:g3:numbers:evenodd}{pares} de \omterm{def:g4:geometry:def}{paralelos} atravessando em certo
        ângulos --- ou a cadeia de \cref{ex:g5:geometry:chain};
  \item \emph{quadrado}: um retângulo cujos quatro lados usam o mesmo
        comprimento --- carregue-o com a bússola;
  \item \emph{losango}: quatro lados iguais --- desenhe um lado e depois dois
        arcos de bússola a partir de suas extremidades com a mesma abertura fixam o
        outros dois \omterm{def:g4:geometry:polygon}{vértices} (um losango são dois triângulos isósceles
        colado ao longo de uma diagonal).
\end{enumerate}
Depois de construir, sempre \emph{verificar} com as ferramentas: esquadro
os cantos, bússola nas laterais.
\end{method}

\begin{example}[Reconhecendo a partir de uma descrição]\label{ex:g5:geometry:recognize}
``Um \omterm{def:g4:geometry:polygon}{quadrilátero} com quatro $4$ cm lados e um \omterm{def:g3:shapes:rightangle}{ângulo reto}'' ---
construí-lo: os quatro lados iguais forçam uma forma de losango, e pressionando
um canto para um \omterm{def:g3:shapes:rightangle}{ângulo reto} endireita todos os outros: o resultado
é um quadrado. Algumas descrições descrevem secretamente uma experiência mais especial
forma do que anunciam. (Por que o \omterm{def:g3:shapes:rightangle}{ângulo reto} se espalha para todos os quatro
cantos é explicado em \cref{ch:g7:central}.)
\end{example}

\section{Círculos em construções}

\begin{example}[O círculo como ferramenta]\label{ex:g5:geometry:circle}
`` Encontre todos os pontos em $3$ cm de $ A $'': o círculo do centro $ A $,
\omterm{def:g3:shapes:shapes}{raio} $3$ cm. `` Encontre todos os pontos em $3$ cm de $ A $ \emph{e}
$4$ cm de $ B $'': desenhe os dois círculos; suas travessias (duas, uma ou
\omterm{def:g1:counting:zero}{zero} pontos) são as respostas. Mapas do tesouro e triângulo
construções usam exatamente essa ideia.
\end{example}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g5:geometry:1}
Desenhe uma linha $ d $ e um ponto $ A $ por volta de $3$ cm dele. Construir
o \omterm{def:g4:geometry:def}{perpendicular} para $ d $ através $ A $, então o \omterm{def:g4:geometry:def}{paralelo} para $ d $
através $ A $. Marque o ângulos retos.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:geometry:2}
Construa um retângulo $ ABCD $ com $ AB = 7$ centímetros e $ BC = 4$ cm,
seguindo \cref{ex:g5:geometry:chain}. Verifique o quarto ângulo com
o esquadro: se a construção for exata, está certa
automaticamente.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:geometry:3}
Construa um triângulo com lados $7$ cm, $6$ centímetros e $3$ cm
(\cref{met:g5:geometry:triangle}).
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:geometry:4}
Construa um triângulo isósceles com base $5$ cm e lados iguais de
$7$ cm; então um triângulo equilátero de lado $6$ cm.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:geometry:5}
Construa um quadrado de lado $5$ cm. Quais ferramentas garantem o certo
ângulos? Quais garantem os lados iguais?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:geometry:6}
Construa um losango de lado $4$ cm que não seja um quadrado (escolha o seu
abertura própria para a primeira curva). Verifique os quatro lados com o
bússola.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:geometry:7}
Desenhe dois pontos $ A $ e $ B $ com $ AB = 5$ cm. Construa todos os pontos
que estão em $4$ cm de $ A $ e em $3$ cm de $ B $. Quantos são
aí?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:geometry:8}
Escreva um programa de construção (etapas numeradas, pontos nomeados) para um
retângulo $6$ cm $\times $ $2$ cm encimado por um triângulo isósceles
com lados iguais $4$ cm (formato de lápis). Peça a um colega que execute
isso.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g5:geometry:9}
Tente construir um triângulo com lados $8$ cm, $3$ centímetros e $4$ cm.
O que acontece com os arcos? Explique em uma frase por que isso
triângulo não pode existir.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g5:geometry:10}
Construa um losango cujas diagonais medem $6$ centímetros e $4$ cm,
sabendo que as diagonais de um losango se cruzam no meio em um
\omterm{def:g3:shapes:rightangle}{ângulo reto}: desenhe as diagonais \emph{primeiro}, então junte seus quatro
termina.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g5:geometry:11}
Uma cabra está amarrada a uma estaca $ A $ por um $4$ m corda, em um campo plano com
uma cerca reta passando $3$ estou de $ A$. Desenhe um plano (1 cm por
metro): qual região a cabra pode alcançar? Onde a cerca limita
isso?
\end{exercise}
