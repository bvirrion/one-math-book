\chapter{Computing with Decimals}\label{ch:g5:decimalops}

Decimal numbers add, subtract and multiply almost like whole numbers
--- the whole art is knowing where the point goes. This chapter
covers the column methods and the magic of multiplying or dividing by
$10$, $100$, $1000$. (Multiplying two decimals together waits for
\cref{ch:g6:decimals}.)

\section{Adding and subtracting}

\begin{method}[Columns with a point]\label{met:g5:decimalops:addsub}
\begin{enumerate}
  \item Write the numbers with the \emph{\omterm{def:g4:decimals:point}{decimal points} aligned}
        (then units are under units, tenths under tenths);
  \item pad the shorter decimal part with \omterm{def:g1:counting:zero}{zeros};
  \item add or subtract as with whole numbers;
  \item drop the point of the result straight down.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g5:decimalops:addsub}
$27.5 + 3.86$ (pad $27.5 = 27.50$):
\[
\begin{array}{r}
2\,7.5\,0 \\
+\ \ \ 3.8\,6 \\
\hline
3\,1.3\,6
\end{array}
\qquad\qquad
\begin{array}{r}
1\,2.4\,0 \\
-\ \ \ 5.6\,5 \\
\hline
\ \ \,6.7\,5
\end{array}
\]
For the subtraction ($12.4 - 5.65$): pad, borrow as usual, and check
by adding back: $6.75 + 5.65 = 12.4$.
\end{example}

\begin{example}[Mental complements]\label{ex:g5:decimalops:complement}
What must be added to $7.6$ to reach $10$? Climb: $7.6 \to 8$ is
$0.4$, then $8 \to 10$ is $2$: the complement is $2.4$. Handy for
change: pay $10$ for a $7.60$ item, get $2.40$ back.
\end{example}

\section{Ten times bigger, ten times smaller}

\begin{proposition}[Times and divided by 10, 100, 1000]\label{prop:g5:decimalops:shift}
Multiplying by $10$ makes each digit worth ten times more: the
digits slide one place left --- which looks like the \emph{point
moving one place right}. Dividing does the opposite:
\[
3.47 \times 10 = 34.7, \qquad
3.47 \times 100 = 347, \qquad
52.1 \div 10 = 5.21, \qquad
6 \div 100 = 0.06 .
\]
\end{proposition}
\admitted

\begin{example}[Conversions again]\label{ex:g5:decimalops:convert}
Units of measure are powers of ten apart, so conversions are just
these shifts: $2.35$ m $= 235$ cm ($\times 100$); $470$ g
$= 0.47$ kg ($\div 1000$); $1.5$ L $= 150$ cL.
\end{example}

\section{Multiplying a decimal by a whole number}

\begin{method}[Decimal times whole]\label{met:g5:decimalops:mult}
Compute as if there were no point, then give the result as many
decimal digits as the decimal factor:
\[
4.35 \times 6: \quad 435 \times 6 = 2\,610
\ \to\ 4.35 \times 6 = 26.10 = 26.1 .
\]
Estimate to place the point with confidence: $4.35$ is close to
$4$, and $4 \times 6 = 24$ --- so $26.1$, not $2.61$ nor $261$.
\end{method}

\begin{example}[Repeated addition still works]\label{ex:g5:decimalops:mult}
$0.75 \times 4 = 0.75 + 0.75 + 0.75 + 0.75 = 3$: four \omterm{def:g3:division:half}{quarters} make
a whole, in decimal clothing. Multiplying by a whole number is still
``so many times''.
\end{example}

\begin{example}[Shopping]\label{ex:g5:decimalops:shopping}
Three notebooks at $2.45$ each and a pen at $1.20$:
\begin{enumerate}
  \item notebooks: $2.45 \times 3 = 7.35$;
  \item total: $7.35 + 1.20 = 8.55$;
  \item change from $10$: $10 - 8.55 = 1.45$.
\end{enumerate}
\end{example}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:g5:decimalops:1}
Compute in columns: $45.7 + 8.93$; \quad $16.25 + 3.75$; \quad
$0.86 + 12.4$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:decimalops:2}
Compute in columns and check: $9.4 - 2.72$; \quad $20 - 13.45$;
\quad $6.03 - 5.9$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:decimalops:3}
Mental complements to $10$ (as in
\cref{ex:g5:decimalops:complement}): $6.2$; \quad $3.75$; \quad
$9.99$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:decimalops:4}
Compute without columns:
\[
7.24 \times 10, \qquad
0.58 \times 100, \qquad
34.5 \div 10, \qquad
7 \div 100, \qquad
0.3 \times 1000 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:decimalops:5}
Convert: $4.05$ m in cm; \quad $325$ cm in m; \quad $0.6$ kg in g;
\quad $85$ cL in L.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:decimalops:6}
Estimate first, then compute: $6.2 \times 4$; \quad
$3.45 \times 8$; \quad $12.5 \times 6$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:decimalops:7}
One lap of a track measures $0.4$ km. How long are $7$ laps? And
$25$ laps?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:decimalops:8}
Lea buys $2$ baguettes at $1.15$ each and a cake at $12.60$. She
pays with a $20$ bill. Compute her total and her change.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:decimalops:9}
A bottle holds $1.5$ L. How many liters in a pack of $6$ bottles?
A family drinks $0.75$ L per day: for how many days does one pack
last? (How many times does $0.75$ fit in $9$?)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g5:decimalops:10}
Milo computes $5.6 \times 3 = 15.18$, reasoning ``$5 \times 3 = 15$
and $6 \times 3 = 18$''. Use an estimate to show the answer must be
wrong, then compute correctly.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g5:decimalops:11}
A runner's four laps took $65.4$ s, $64.9$ s, $65.0$ s and
$66.1$ s.
\begin{enumerate}
  \item Compute the total time of the four laps.
  \item The record for four laps is $260$ s. Did she beat it? By how
        much?
\end{enumerate}
\end{exercise}
