\chapter{क्यूब्स और बक्से}\label{ch:g5:solids}

समतल पृष्ठ छोड़ना: एसएनएफ वास्तविक दुनिया के आकार हैं ---
पासा, अनाज के डिब्बे, डिब्बे, गेंदें। यह अध्याय उन्हें नाम देना सिखाता है,
उनकी गिनती करें \omterm{def:g5:solids:def}{चेहरे} और \omterm{def:g5:solids:def}{किनारों}, उन्हें प्रकट करें \omterm{def:g5:solids:net}{जाल}, और मापें
उनके \omterm{def:g5:solids:volume}{आयतन} थोड़ी गिनती करके क्यूब्स. (\omterm{def:g5:solids:volume}{आयतन} सूत्र आते हैं
\cref{ch:g6:measure}.)

\section{ठोस परिवार}

\begin{definition}[ठोस और उनकी शब्दावली]\label{def:g5:solids:def}
ए \emph{ठोस}\index{ठोस} जगह घेरता है। समतल भुजाएँ हैं
\emph{चेहरे}\index{चेहरे}; चेहरे साथ मिलते हैं \emph{किनारों}\index{किनारों};
किनारे मिलते हैं \emph{कोने}\index{कोने}. परिवार
चित्र:
\begin{itemize}
  \item  \emph{घनक्षेत्र}\index{घनक्षेत्र}: $6$ चौकोर चेहरे, $12$ किनारे,
        $8$ शिखर;
  \item  \emph{डिब्बा}\index{डिब्बा} (आयताकार आयता): $6$
        आयताकार चेहरे, $12$ किनारे, $8$ शिखर;
  \item  \emph{सिलेंडर}\index{सिलेंडर}: दो डिस्क फलक और एक लुढ़का हुआ फलक
        (एक कैन);
  \item  \emph{पिरामिड}\index{पिरामिड}: ए \omterm{def:g4:geometry:polygon}{बहुभुज} आधार और त्रिभुज मिलते हैं
        एक शीर्ष;
  \item  \emph{गेंद} (\omterm{def:g4:numbers:round}{गोल}ाकार): एक बिल्कुल \omterm{def:g4:numbers:round}{गोल} सतह, नहीं
        चेहरा, कोई किनारा नहीं.
\end{itemize}
\end{definition}

\begin{example}[घन पर गिनती]\label{ex:g5:solids:count}
पासे पर जाँचें: $6$ \omterm{def:g5:solids:def}{चेहरे} (संख्या $1$ को $6$), $8$ \omterm{def:g5:solids:def}{कोने}
(\omterm{def:g5:solids:def}{कोने}), $12$ \omterm{def:g5:solids:def}{किनारों} (उन पर एक उंगली चलाएं: $4$ शीर्ष पर, $4$
तल पर, $4$ खड़ा)। और
$6 + 8 = 12 + 2$ --- वही जिज्ञासु पैटर्न इसके लिए भी लागू होता है \omterm{def:g5:solids:def}{डिब्बा} और यह \omterm{def:g5:solids:def}{पिरामिड} (\cref{ex:g8:solids:count} उस पर वापस लौटता है)।
\end{example}

\section{जाल}

\begin{definition}[जाल]\label{def:g5:solids:net}
ए \emph{जाल}\index{जाल} एक का \omterm{def:g5:solids:def}{ठोस} यह सभी का एक सपाट चित्रण है
\omterm{def:g5:solids:def}{चेहरे}, साथ में संलग्न \omterm{def:g5:solids:def}{किनारों}, जो अंदर की ओर मुड़ जाता है \omterm{def:g5:solids:def}{ठोस} --- द
\omterm{def:g5:solids:def}{ठोस} कार्डबोर्ड की तरह खुला हुआ \omterm{def:g5:solids:def}{डिब्बा} पुनर्चक्रण के लिए चपटा किया गया।
\end{definition}

\begin{example}[नेट पढ़ना]\label{ex:g5:solids:read}
पासे पर, विपरीत \omterm{def:g5:solids:def}{चेहरे} तक \omterm{def:g1:addition:def}{जोड़}ा गया $7$. एक पर \omterm{def:g5:solids:net}{जाल}, विलोम \omterm{def:g5:solids:def}{चेहरे} हैं
जो एक पंक्ति में बिल्कुल एक वर्ग से अलग होते हैं (या एक के आसपास)।
कोना) --- पड़ोसी नहीं! पर \omterm{def:g1:addition:def}{जोड़}ियों को अंकित करना \omterm{def:g5:solids:net}{जाल} पहले
स्थान को सपाट देखने के लिए फोल्डिंग एक बेहतरीन व्यायाम है।
\end{example}

\section{आयतन: घनों की गिनती}

\begin{definition}[गिनती के अनुसार आयतन]\label{def:g5:solids:volume}
 \emph{आयतन}\index{आयतन} एक का \omterm{def:g5:solids:def}{ठोस} इकाई से निर्मित क्यूब्स है
की संख्या क्यूब्स इसमें है। इकाई सेंटी\omterm{def:g4:fractions:def}{मीटर} हो सकती है \omterm{def:g5:solids:def}{घनक्षेत्र}
(सेमी $^3$) या \omterm{def:g4:fractions:def}{मीटर} \omterm{def:g5:solids:def}{घनक्षेत्र} (एम $^3$).
\end{definition}

\begin{example}[एक बॉक्स को परत दर परत गिनना]\label{ex:g5:solids:layers}
ए \omterm{def:g5:solids:def}{डिब्बा} $4$ क्यूब्स लंबा, $3$ चौड़ा, $2$ उच्च: निचली परत धारण करती है
$4 \times 3 = 12$ क्यूब्स, और वहाँ हैं $2$ परतें:
\[
12 \times 2 = 24 \text{ cubes} .
\]
परतों द्वारा गिनती हमेशा काम करती है --- और यह चुपचाप साबित करती है
सूत्र $ V = L \times w \times h $ का \cref{ch:g6:measure}.
\end{example}

      \foreach \z in {0,1,2} {

    }
  }
  \node[below, font=\small] at (2.6,-0.4)
    {$4 \times 3$प्रति परत घन,$2$परतें:$24$क्यूब्स};
\end{tikzpicture}

{\small ए $4 \times 3 \times 2$ \omterm{def:g5:solids:def}{डिब्बा} इकाई का क्यूब्स, परत दर परत गिना गया
परत.}
\end{omfigure}

\begin{example}[एक जैसे घन, अलग-अलग आकार]\label{ex:g5:solids:rearrange}
आठ इकाई क्यूब्स का निर्माण कर सकते हैं $2 \times 2 \times 2$ \omterm{def:g5:solids:def}{घनक्षेत्र}, ए
$8 \times 1 \times 1$ छड़ी, या ए $4 \times 2 \times 1$ स्लैब: तीन
अलग एसएनएफ, एक \omterm{def:g5:solids:volume}{आयतन} ($8$ क्यूब्स). \omterm{def:g5:solids:volume}{आयतन} गिनती करता है
सामग्री, आकार नहीं.
\end{example}

\section{अभ्यास}

\begin{exercise}[$\star $]\label{exo:g5:solids:1}
एक वास्तविक वस्तु का नाम बताइए जिसका आकार इस प्रकार है: a \omterm{def:g5:solids:def}{घनक्षेत्र}; ए \omterm{def:g5:solids:def}{डिब्बा}; ए \omterm{def:g5:solids:def}{सिलेंडर}; एक गेंद;
ए \omterm{def:g5:solids:def}{पिरामिड}.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:solids:2}
कितने \omterm{def:g5:solids:def}{चेहरे}, \omterm{def:g5:solids:def}{किनारों} और \omterm{def:g5:solids:def}{कोने} एक \omterm{def:g5:solids:def}{डिब्बा}? पैटर्न की जाँच करें
\omterm{def:g5:solids:def}{चेहरे} $+$ \omterm{def:g5:solids:def}{कोने} $=$ \omterm{def:g5:solids:def}{किनारों} $+ 2$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:solids:3}
कौन \omterm{def:g5:solids:def}{चेहरे} एक का \omterm{def:g5:solids:def}{डिब्बा} समान हैं? (छह का समूह बनाएं \omterm{def:g5:solids:def}{चेहरे} \omterm{def:g1:addition:def}{जोड़}े में.)
इसमें क्या खास है \omterm{def:g5:solids:def}{घनक्षेत्र}'एस \omterm{def:g5:solids:def}{चेहरे}?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:solids:4}
चित्रित करो \omterm{def:g5:solids:net}{जाल} एक का \omterm{def:g5:solids:def}{घनक्षेत्र} इस पक्ष की $2$ सेमी (क्रॉस-आकार वाले मॉडल का उपयोग करें)
अध्याय), इसे काटें और मोड़ें।
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:solids:5}
क्रॉस के आकार पर \omterm{def:g5:solids:net}{जाल} एक पासे में, संख्याएं रखें $1$ को $6$ तो
वह विपरीत \omterm{def:g5:solids:def}{चेहरे} का योग $7$. (उपयोग \cref{ex:g5:solids:read} को
पहले विपरीत युग्म खोजें।)
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:solids:6}
इकाई की गणना करें क्यूब्स: ए \omterm{def:g5:solids:def}{डिब्बा} $5$ लंबा, $2$ चौड़ा, $3$ उच्च; ए \omterm{def:g5:solids:def}{घनक्षेत्र} इस पक्ष की $3$; एक एल-आकार का ढेर जो बना है $3 \times 2 \times 1$ स्लैब
एक के साथ $1 \times 2 \times 1$ शीर्ष पर स्लैब.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:solids:7}
ए \omterm{def:g5:solids:def}{डिब्बा} बिल्कुल सही रखता है $12$ इकाई क्यूब्स एक परत में. दो संभव बताइये
परत आकार ($ L \times w $), और प्रत्येक के लिए, \omterm{def:g5:solids:def}{डिब्बा}'एस \omterm{def:g5:solids:volume}{आयतन} अगर यह
है $3$ ऐसी परतें.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:solids:8}
साथ $27$ इकाई क्यूब्स, क्या \omterm{def:g5:solids:def}{घनक्षेत्र} क्या आप निर्माण कर सकते हैं? साथ $64$? (सोचो
परत पीछे की ओर गिनती करती है।)
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g5:solids:9}
चीनी क्यूब्स इस पक्ष की $1$ सेमी ए में आओ \omterm{def:g5:solids:def}{डिब्बा} $10$ सेमी लंबा, $5$ सेमी चौड़ा,
$4$ सेमी ऊँचा (आंतरिक आकार)। कितनी चीनी क्यूब्स उपयुक्त? यदि ए
पारिवारिक उपयोग $6$ क्यूब्स एक दिन, कितने दिनों तक करता है \omterm{def:g5:solids:def}{डिब्बा} अंतिम
(\omterm{def:g4:numbers:round}{गोल} समझदारी से)?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g5:solids:10}
ए $3 \times 3 \times 3$ \omterm{def:g5:solids:def}{घनक्षेत्र} इसे बाहर से लाल रंग से रंगा जाता है, फिर काटा जाता है
में $27$ इकाई क्यूब्स. कितने छोटे क्यूब्स बिल्कुल पेंट लगा हुआ है
$3$ \omterm{def:g5:solids:def}{चेहरे}? पर $2$? पर $1$? किसी पर नहीं? (जांचें: चार गिनती अवश्य \omterm{def:g1:addition:def}{जोड़}ी जानी चाहिए
तक $27$.)
\end{exercise}
