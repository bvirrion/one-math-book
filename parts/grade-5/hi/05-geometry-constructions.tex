\chapter{ज्यामितीय निर्माण}\label{ch:g5:geometry}

रूलर, सेट स्क्वायर, कम्पास: इन तीन उपकरणों और थोड़े से के साथ
विधि से शब्दों में वर्णित कोई भी आकृति हूबहू बनाई जा सकती है। यह
अध्याय निर्माण लंबवत, समानताएं, \omterm{def:g3:shapes:shapes}{त्रिकोण} और
वि\omterm{def:g3:division:remainder}{शेष} चतुर्भुज --- शिल्प पहले, सिद्धांत बाद में
(\cref{ch:g6:lines,ch:g6:shapes}).

\section{एक बिंदु से गुजरने वाले लम्ब और समांतर रेखाएँ}

\begin{method}[दो बुनियादी निर्माण]\label{met:g5:geometry:basic}
\emph{\omterm{def:g4:geometry:def}{सीधा} को $d $ के माध्यम से $ A $}: सेट स्क्वायर को साथ में स्लाइड करें
$ d $ जब तक कि उसका दूसरा \omterm{def:g3:shapes:rightangle}{समकोण} किनारा न पहुँच जाए $ A $; खींचना; चिन्हित करें
\omterm{def:g3:shapes:rightangle}{समकोण}.

\emph{\omterm{def:g4:geometry:def}{समानांतर} को $ d $ के माध्यम से $ A $}: सेट के सामने रूलर दबाएँ
किनारे पर बिछाई गई रेल की तरह वर्गाकार $ d $; सेट स्क्वायर को ऊपर की ओर स्लाइड करें $ A $;
खींचना। (जाँचें: दोनों पंक्तियों के बीच का \omterm{ex:g1:subtraction:difference}{अंतर} समान है
हर जगह।)
\end{method}

\begin{example}[जंजीर निर्माण]\label{ex:g5:geometry:chain}
कई आकृतियाँ इन दो चालों की श्रृंखलाएँ हैं। एक आयत $ ABCD $ साथ
$ AB = 5$ सेमी और $ BC = 3$ सेमी: ड्रा $[AB]$; \omterm{def:g4:geometry:def}{सीधा} पर $ A $;
\omterm{def:g4:geometry:def}{सीधा} पर $ B $; निशान $ D $ और $ C $ पर $3$ एक ही तरफ सेमी;
जोड़ना। प्रत्येक चरण एक बुनियादी निर्माण है।
\end{example}

\section{कम्पास के साथ त्रिकोण}

\begin{method}[तीन ओर से त्रिभुज]\label{met:g5:geometry:triangle}
भुजाओं वाला एक त्रिभुज बनाना $6$ सेमी, $5$ सेमी, $4$ सेमी:
\begin{enumerate}
  \item सबसे लंबी भुजा खींचिए, $[AB]$ साथ $ AB = 6$ सेमी;
  \item केंद्र का चाप $ A $ और \omterm{def:g3:shapes:shapes}{RADIUS} $5$ सेमी;
  \item केंद्र का चाप $ B $ और \omterm{def:g3:shapes:shapes}{RADIUS} $4$ सेमी;
  \item उनका क्रॉसिंग तीसरा है शिखर $ C $; जोड़ना।
\end{enumerate}
कम्पास लंबाई की गारंटी देता है: पहले चाप का प्रत्येक बिंदु है
बिल्कुल $5$ सेमी से $ A $.
\end{method}

\begin{example}[समद्विबाहु और समबाहु निःशुल्क]\label{ex:g5:geometry:special}
दोनों चापों के लिए समान कम्पास उद्घाटन $\Rightarrow $ समद्विबाहु त्रिभुज.
आधार के बराबर खुलना $\Rightarrow $ समबाहु. कम्पास
माप नहीं करता: यह \emph{प्रतियां} लम्बाई, जो और भी बेहतर है।
\end{example}

\section{निर्माण द्वारा विशेष चतुर्भुज}

\begin{method}[परिवार का निर्माण]\label{met:g5:geometry:quadrilaterals}
\begin{enumerate}
  \item \emph{आयत}: के दो \omterm{def:g1:addition:def}{जोड़}े समानताएं पर पार करना सही कोण --- या की श्रृंखला \cref{ex:g5:geometry:chain};
  \item \emph{वर्ग}: एक आयत जिसकी चारों भुजाएँ समान होती हैं
        लंबाई --- इसे कंपास के साथ ले जाएं;
  \item \emph{विषमकोण}: चार समान भुजाएँ --- एक भुजा खींचें, फिर दो
        कम्पास आर्क को इसके सिरों से एक ही उद्घाटन के साथ ठीक करें
        अन्य दो \omterm{def:g4:geometry:polygon}{कोने} (एक समचतुर्भुज दो समद्विबाहु त्रिभुज हैं
        एक विकर्ण के साथ चिपका हुआ)।
\end{enumerate}
निर्माण के बाद, हमेशा \emph{जाँच करना} टूल के साथ: स्क्वायर ऑन सेट करें
\omterm{def:g4:geometry:polygon}{कोने}, किनारों पर कम्पास।
\end{method}

\begin{example}[विवरण से पहचानना]\label{ex:g5:geometry:recognize}
``ए \omterm{def:g4:geometry:polygon}{चतुष्कोष} चार के साथ $4$ सेमी भुजाएँ और एक \omterm{def:g3:shapes:rightangle}{समकोण}'' ---
इसे बनाएँ: चार समान भुजाएँ एक समचतुर्भुज आकार को बल देती हैं, और दबाती हैं
एक \omterm{def:g4:geometry:polygon}{कोने} से एक \omterm{def:g3:shapes:rightangle}{समकोण} अन्य सभी को \omterm{def:g4:geometry:def}{सीधा} करता है: परिणाम
एक वर्ग है. कुछ विवरण गुप्त रूप से अधिक वि\omterm{def:g3:division:remainder}{शेष} का वर्णन करते हैं
जितना वे घोषणा करते हैं उससे कहीं अधिक आकार देते हैं। (क्यों \omterm{def:g3:shapes:rightangle}{समकोण} चारों में फैल जाता है
कोनों की व्याख्या की गई है \cref{ch:g7:central}.)
\end{example}

\section{निर्माणों में वृत्त}

\begin{example}[एक उपकरण के रूप में वृत्त]\label{ex:g5:geometry:circle}
``सभी बिंदु खोजें $3$ सेमी से $ A $'': केंद्र का वृत्त $ A $,
\omterm{def:g3:shapes:shapes}{RADIUS} $3$ सेमी। ``सभी बिंदु खोजें $3$ सेमी से $ A $ \emph{और}
$4$ सेमी से $ B $'': दोनों वृत्त बनाएं; उनकी क्रॉसिंग (दो, एक या)
\omterm{def:g1:counting:zero}{शून्य} अंक) उत्तर हैं। ख़ज़ाने के नक्शे और \omterm{def:g3:shapes:shapes}{त्रिकोण}
निर्माण बिल्कुल इसी विचार का उपयोग करते हैं।
\end{example}

\section{अभ्यास}

\begin{exercise}[$\star $]\label{exo:g5:geometry:1}
एक रेखा खींचो $ d $ और एक बिंदु $ A $ लगभग $3$ इससे सेमी. निर्माण
द \omterm{def:g4:geometry:def}{सीधा} को $ d $ के माध्यम से $ A $, फिर \omterm{def:g4:geometry:def}{समानांतर} को $ d $
के माध्यम से $ A $. चिन्हित करें सही कोण.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:geometry:2}
एक आयत का निर्माण करें $ ABCD $ साथ $ AB = 7$ सेमी और $ BC = 4$ सेमी,
निम्नलिखित \cref{ex:g5:geometry:chain}. चौथे कोण की जाँच करें
सेट वर्ग: यदि निर्माण सटीक है, तो यह सही है
स्वचालित रूप से.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:geometry:3}
भुजाओं वाला एक त्रिभुज बनाइये $7$ सेमी, $6$ सेमी और $3$ सेमी
(\cref{met:g5:geometry:triangle}).
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:geometry:4}
आधार सहित एक समद्विबाहु त्रिभुज की रचना कीजिए $5$ सेमी और बराबर भुजाएँ
$7$ सेमी; फिर भुजा का एक समबाहु त्रिभुज $6$ सेमी।
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:geometry:5}
भुजा का एक वर्ग बनाइये $5$ सेमी। कौन से उपकरण इसकी गारंटी देते हैं सही कोण? कौन समान भुजाओं की गारंटी देता है?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:geometry:6}
भुजा का एक समचतुर्भुज बनाइये $4$ सेमी जो एक वर्ग नहीं है (अपना चुनें)।
पहले \omterm{def:g4:geometry:polygon}{कोने} के लिए स्वयं का उद्घाटन)। से चारों भुजाओं की जाँच करें
दिशा सूचक यंत्र।
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:geometry:7}
दो बिंदु बनाएं $ A $ और $ B $ साथ $ AB = 5$ सेमी। सभी बिंदुओं का निर्माण करें
वह पर हैं $4$ सेमी से $ A $ और कम से $3$ सेमी से $ B $. कितने हैं?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:geometry:8}
के लिए एक निर्माण कार्यक्रम (क्रमांकित चरण, नामित बिंदु) लिखें
आयत $6$ सेमी $\times $ $2$ सेमी के शीर्ष पर एक समद्विबाहु त्रिभुज है
समान भुजाओं के साथ $4$ सेमी (एक पेंसिल आकार)। एक सहपाठी को निष्पादित करें
यह.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g5:geometry:9}
भुजाओं वाला एक त्रिभुज बनाने का प्रयास करें $8$ सेमी, $3$ सेमी और $4$ सेमी।
चापों से क्या होता है? ऐसा क्यों है, एक वाक्य में स्पष्ट करें
त्रिभुज का अस्तित्व नहीं हो सकता.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g5:geometry:10}
एक समचतुर्भुज की रचना कीजिए जिसके विकर्ण मापें $6$ सेमी और $4$ सेमी,
यह जानते हुए कि एक समचतुर्भुज के विकर्ण अपने मध्य में एक बिंदु पर प्रतिच्छेद करते हैं
\omterm{def:g3:shapes:rightangle}{समकोण}: विकर्ण खींचिए \emph{पहला}, फिर उनके चार में शामिल हो जाएं
समाप्त होता है.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g5:geometry:11}
एक बकरी को खूंटी से बांधा गया है $ A $ ए द्वारा $4$ मी रस्सी, एक समतल मैदान में
एक सीधी बाड़ गुजर रही है $3$ मैं वहां से हूँ $ A$. एक योजना बनाएं (प्रति सेमी 1)।
\omterm{def:g4:fractions:def}{मीटर}): बकरी किस \omterm{def:g4:measure:area}{क्षेत्र} तक पहुंच सकती है? बाड़ कहाँ तक सीमित है
यह?
\end{exercise}
