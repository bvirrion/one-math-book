\chapter{संख्याओं के रूप में भिन्न}\label{ch:g5:fractions}

ए \omterm{def:g4:fractions:def}{अंश} यह सिर्फ एक पाई की तस्वीर नहीं है: यह असली है
\emph{संख्या}, संख्या रेखा पर एक स्थान के साथ --- कभी-कभी बहुत ही
दशमलव संख्या के समान स्थान। यह अध्याय दोनों को जोड़ता है
संसार. (के पूर्ण नियम \omterm{def:g4:fractions:def}{अंश} अंकगणित अंदर आओ
\cref{ch:g6:fractions} और इसके बाद में।)

\section{भिन्न, फिर से रेखा पर}

\begin{example}[रखना और पढ़ना]\label{ex:g5:fractions:place}
पंचम में कटी हुई एक पंक्ति पर, $\frac{7}{5}$ है $7$ से कदम $0$: अतीत
$1$ (जो है $\frac55$), पर $1 + \frac25$. प्रत्येक \omterm{def:g4:fractions:def}{अंश} रहता है
कहीं लाइन पर; बड़ा \omterm{def:g4:fractions:def}{मीटर} (वही \omterm{def:g4:fractions:def}{भाजक}) का मतलब है
आगे सही.
\end{example}

\begin{proposition}[बात एक, नाम अनेक]\label{prop:g5:fractions:equal}
प्रत्येक भाग को दो (या तीन) भागों में काटना\,\dots) को स्थानांतरित नहीं करता है
बिंदु:
\[
\frac{1}{2} = \frac{2}{4} = \frac{3}{6} = \frac{5}{10},
\qquad
\frac{2}{3} = \frac{4}{6} = \frac{20}{30}.
\]
को गुणा करना (या भाग देना)। \omterm{def:g4:fractions:def}{मीटर} और यह \omterm{def:g4:fractions:def}{भाजक} द्वारा
वही संख्या उसी का दूसरा नाम देती है \omterm{def:g4:fractions:def}{अंश}.
\end{proposition}
\admitted

\section{भिन्न और दशमलव}

\begin{proposition}[दशमलव नामों के साथ भिन्न]\label{prop:g5:fractions:decimal}
भिन्न दसवाँ, सौवाँ, हज़ारवाँ \emph{हैं} दशमलव संख्याएं:
\[
\frac{3}{10} = 0.3, \qquad \frac{27}{100} = 0.27 .
\]
किसी और \omterm{def:g4:fractions:def}{अंश}ों इसमें दशमलव नाम भी होते हैं, जिन्हें स्विच करने पर पता चलता है
दसवाँ या सौवाँ:
\[
\frac{1}{2} = \frac{5}{10} = 0.5, \qquad
\frac{1}{4} = \frac{25}{100} = 0.25, \qquad
\frac{3}{4} = 0.75, \qquad
\frac{1}{5} = \frac{2}{10} = 0.2 .
\]
लेकिन सभी नहीं: $\frac13 = 0.333\dots $ कभी ख़त्म नहीं होता --- \omterm{def:g4:fractions:def}{अंश} है
यह एकमात्र सटीक नाम है.
\end{proposition}
\admitted

\begin{example}[सुविधाजनक नाम चुनना]\label{ex:g5:fractions:choose}
तीन का भुगतान करने के लिए क्वार्टरों का $12$: द \omterm{def:g4:fractions:def}{अंश} नाम आसान है,
$\frac34$ का $12$ है $9$. तुलना करने के लिए $\frac34$ और $0.8$: द
दशमलव नाम आसान है, $0.75 < 0.8$. द्विभाषी होने से लाभ मिलता है।
\end{example}

\section{भिन्नों की तुलना करना}

\begin{method}[तीन तुलना उपकरण]\label{met:g5:fractions:compare}
\begin{enumerate}
  \item \emph{वही \omterm{def:g4:fractions:def}{भाजक}}: अधिक भाग जीतता है,
        $\frac57 > \frac37$;
  \item \emph{वही \omterm{def:g4:fractions:def}{मीटर}}: बड़े हिस्से जीतते हैं, इसलिए \emph{छोटे}
        \omterm{def:g4:fractions:def}{भाजक} जीतता है: $\frac35 > \frac38$ (पांचवें बड़े हैं
        आठवें से);
  \item \emph{लैंडमार्क्स}: प्रत्येक की तुलना करें $\frac12$ या करने के लिए $1$:
        $\frac38 < \frac12 < \frac59$, इसलिए $\frac38 < \frac59$.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g5:fractions:compare}
किसने अधिक पिज़्ज़ा खाया: अली के साथ $\frac58$ या बी के साथ $\frac54$? बिया का
\omterm{def:g4:fractions:def}{अंश} से बड़ा है $1$ (पाँच क्वार्टरों एक से अधिक पूर्ण है
पिज़्ज़ा), अली से छोटा है $1$: बी ने अधिक खाया --- ए से अधिक
वास्तव में, संपूर्ण पिज़्ज़ा।
\end{example}

\section{अभ्यास}

\begin{exercise}[$\star $]\label{exo:g5:fractions:1}
से पाँचवें भाग में कटी हुई एक संख्या रेखा खींचिए $0$ को $2$, और स्थान:
$\frac25$; \ $\frac55$; \ $\frac85$; \ $\frac{10}{5}$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:fractions:2}
समान नामों को पूरा करें:
$\frac12 = \frac{?}{8}$; \quad
$\frac34 = \frac{?}{8}$; \quad
$\frac{2}{5} = \frac{4}{?}$; \quad
$\frac{6}{9} = \frac{2}{?}$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:fractions:3}
कौन \omterm{def:g4:fractions:def}{अंश}ों पूर्ण संख्याओं के नाम बताएं? ऐसा होने पर पूरी संख्या दें:
$\frac82$; \quad $\frac{9}{4}$; \quad $\frac{15}{3}$; \quad
$\frac{20}{5}$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:fractions:4}
दशमलव के रूप में लिखें: $\frac{7}{10}$; \quad $\frac{41}{100}$; \quad
$\frac12$; \quad $\frac34$; \quad $\frac15$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:fractions:5}
के रूप में लिखें \omterm{def:g4:fractions:def}{अंश}ों (दसवां या सौवां): $0.9$; \quad $0.13$;
\quad $2.5$; \quad $0.75$ (तो उसे दे दो \omterm{def:g3:division:half}{तिमाही} नाम)।
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:fractions:6}
प्रयुक्त उपकरण का नामकरण करते हुए तुलना करें
(\cref{met:g5:fractions:compare}):
\[
\frac47 \;?\; \frac67, \qquad
\frac35 \;?\; \frac37, \qquad
\frac25 \;?\; \frac{7}{12}, \qquad
\frac98 \;?\; \frac{11}{12} .
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:fractions:7}
दशमलव नामों के साथ आदेश दें: $\frac12$; \ $0.4$; \ $\frac34$; \
$0.6$; \ $\frac15$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:fractions:8}
ए \omterm{def:g3:division:half}{तिमाही} की एक कक्षा के छात्रों की $28$ कोई वाद्य यंत्र बजाओ, और
\omterm{def:g3:division:half}{आधा} कोई खेल खेलें। प्रत्येक में कितने छात्र हैं? कौन सा समूह बड़ा है?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:fractions:9}
प्रत्येक को मिलाएं \omterm{def:g4:fractions:def}{अंश} एक समय के साथ: $\frac14$ एच, $\frac12$ एच,
$\frac34$ एच, $\frac{1}{60}$ ज --- को $30$ मिनट, $45$ मिनट, $1$ मिनट,
$15$ मि.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g5:fractions:10}
टॉम ने पी लिया $\frac35$ के बारे में उनकी $50$ सीएल बोतल; एना ने पी लिया $\frac12$ का
उसे $60$ सीएल बोतल. किसने अधिक पीया? (सीएल में दोनों राशियों की गणना करें ---
नंगे की तुलना करना \omterm{def:g4:fractions:def}{अंश}ों यहां पर्याप्त नहीं है, और बताएं क्यों।)
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g5:fractions:11}
दे दो \omterm{def:g4:fractions:def}{अंश} सख्ती से बीच में $\frac12$ और $1$; फिर एक
सख्ती से बीच में $\frac12$ और $\frac34$. (दशमलव नाम हो सकते हैं
मदद.)
\end{exercise}
