\chapter{परिधि और क्षेत्रफल}\label{ch:g5:areas}

ग्रेड 4 में गिने गए ग्रिड वर्ग (\cref{ch:g4:measure}); यह अध्याय
सबसे पहले कमाता है \omterm{def:g4:measure:area}{क्षेत्र} \emph{FORMULA} --- के लिए लंबाई गुणा चौड़ाई
आयत --- और इकाई सेमी सीखता है $^2$ और एम $^2$. के लिए सूत्र
\omterm{def:g3:shapes:shapes}{त्रिकोण} और अन्य आकृतियाँ पकती हैं \cref{ch:g6:measure} और
\cref{ch:g7:areas}.

\section{दो अलग-अलग उपाय}

\begin{example}[सीमा बनाम सतह]\label{ex:g5:areas:contrast}
एक माली की जरूरत है \emph{\omterm{def:g4:measure:perimeter}{परिधि}} बाड़ खरीदने के लिए और \emph{\omterm{def:g4:measure:area}{क्षेत्र}}
घास के बीज खरीदने के लिए. दोनों एक दूसरे का अनुसरण नहीं करते: स्ट्रेचिंग
एक आयत पतला और लंबा इसे रख सकता है \omterm{def:g4:measure:area}{क्षेत्र} जबकि यह \omterm{def:g4:measure:perimeter}{परिधि}
बढ़ता है --- तुलना करें ए $6 \times 2$ और ए $4 \times 3$ आयत
(वही \omterm{def:g4:measure:area}{क्षेत्र} $12$, \omterm{def:g4:measure:perimeter}{परिधि} $16$ और $14$).
\end{example}

\section{आयत सूत्र}

\begin{proposition}[एक आयत का क्षेत्रफल]\label{prop:g5:areas:rectangle}
लंबाई का एक आयत $ L $ और चौड़ाई $ w $ (उसी इकाई में) है \omterm{def:g4:measure:area}{क्षेत्र}
\[
A = L \times w ,
\]
वर्ग इकाइयों में: सेमी$^2$ (पक्ष के वर्ग $1$ सेमी), मी $^2$ (के वर्ग)
ओर $1$ एम), \dots
\end{proposition}

\begin{proof}[क्यों, गिनकर]
आयत को इकाई वर्गों से ढकें: $ w $ की पंक्तियाँ $ L $ प्रत्येक वर्ग,
तो $ L \times w $ कुल मिलाकर वर्ग --- बिलकुल गुणन
का आयत \cref{ch:g3:mult}.
\end{proof}

\end{omfigure}

\begin{example}\label{ex:g5:areas:rectangle}
एक गलीचा मापता है $2.5$ मी द्वारा $2$ एम: \omterm{def:g4:measure:area}{क्षेत्र} $2.5 \times 2 = 5$ एम $^2$.
एक स्टांप मापता है $3$ सेमी द्वारा $2.4$ सेमी: \omterm{def:g4:measure:area}{क्षेत्र} $3 \times 2.4 = 7.2$
सेमी $^2$. समान सूत्र, कोई भी इकाई --- जब तक दोनों पक्ष इसका उपयोग करते हैं
\emph{वही} एक।
\end{example}

\begin{example}[वर्ग इकाइयाँ 100 से परिवर्तित होती हैं]\label{ex:g5:areas:units}
$1$ एम $= 100$ सेमी, लेकिन $1$ एम $^2 = 100 \times 100 = 10\,000$ सेमी $^2$:
एक वर्ग \omterm{def:g4:fractions:def}{मीटर} एक है $100$-द्वारा-$100$ वर्ग सेंटी\omterm{def:g4:fractions:def}{मीटर} का ग्रिड. \omterm{def:g4:measure:area}{क्षेत्र}
इकाइयाँ सैकड़ों से आगे बढ़ती हैं, दसियों से नहीं।
\end{example}

\section{समग्र आंकड़े}

\begin{method}[काटना, जोड़ना, घटाना]\label{met:g5:areas:composite}
आयतों (एक एल, एक टी, एक फ्रेम) से बनी आकृति के लिए:
\begin{enumerate}
  \item इसे आयतों में काटें, या एक बड़े आयत में पूरा करें;
  \item प्रत्येक आयताकार की गणना करें \omterm{def:g4:measure:area}{क्षेत्र} सूत्र के साथ;
  \item टुकड़े \omterm{def:g1:addition:def}{जोड़}ें --- या बड़े में से छेद घटाएँ
        आयत;
  \item ग्रिड वर्गों की अनुमानित गिनती के विरुद्ध जाँच करें।
\end{enumerate}
\end{method}

\begin{example}\label{ex:g5:areas:composite}
एल आकार का कमरा: ए $6$ एम $\times $ $4$ ए के साथ एम आयत
$2$ एम $\times $ $2$ मी कोना गायब.
\[
\text{Area} = 6 \times 4 - 2 \times 2 = 24 - 4 = 20 \text{ m}^2 .
\]
या L को a में काटें $6 \times 2$ पट्टी और ए $4 \times 2$ पट्टी:
$12 + 8 = 20$ एम $^2$ --- दो रास्ते, एक जवाब.
\end{example}

\begin{example}[आधा आयत]\label{ex:g5:areas:halfrect}
एक आयत को एक विकर्ण के अनुदिश काटने पर समान दो त्रिभुज प्राप्त होते हैं
\omterm{def:g4:measure:area}{क्षेत्र}: प्रत्येक है \emph{\omterm{def:g3:division:half}{आधा}} आयत. पैरों वाला एक \omterm{def:g3:shapes:rightangle}{समकोण} त्रिभुज
$6$ सेमी और $4$ सेमी इसलिए है \omterm{def:g4:measure:area}{क्षेत्र} $\frac{6 \times 4}{2} = 12$
सेमी $^2$ --- याद रखने लायक तस्वीर
\cref{ch:g6:measure}, जहां यह एक सूत्र बन जाता है।
\end{example}

\section{अभ्यास}

\begin{exercise}[$\star $]\label{exo:g5:areas:1}
की गणना करें \omterm{def:g4:measure:area}{क्षेत्र} और यह \omterm{def:g4:measure:perimeter}{परिधि} एक आयत का $8$ सेमी $\times $
$5$ सेमी। कौन सा उत्तर सेमी में है, कौन सा सेमी में $^2$?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:areas:2}
की गणना करें \omterm{def:g4:measure:area}{क्षेत्र}ों: भुजा का एक वर्ग $9$ सेमी; एक आयत
$12$ एम $\times $ $7$ एम; एक आयत $4.5$ सेमी $\times $ $6$ सेमी।
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:areas:3}
एक आयत है \omterm{def:g4:measure:area}{क्षेत्र} $63$ सेमी $^2$ और लंबाई $9$ सेमी। इसकी चौड़ाई ज्ञात कीजिए,
फिर यह \omterm{def:g4:measure:perimeter}{परिधि}.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:areas:4}
के साथ दो अलग-अलग आयत बनाएं \omterm{def:g4:measure:area}{क्षेत्र} $24$ ग्रिड पेपर पर वर्ग,
और दोनों की गणना करें \omterm{def:g4:measure:perimeter}{परिधि}. वही \omterm{def:g4:measure:area}{क्षेत्र} --- वही \omterm{def:g4:measure:perimeter}{परिधि}?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:areas:5}
बदलना: $3$ एम $^2$ सेमी में $^2$; \quad $50\,000$ सेमी $^2$ एम में $^2$.
(याद करना \cref{ex:g5:areas:units}: सैकड़ों द्वारा!)
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:areas:6}
एक T-आकार की आकृति किससे बनी है? $8 \times 2$ शीर्ष पर क्षैतिज पट्टी
ए $2 \times 5$ ऊर्ध्वाधर पट्टी (सेमी में)। इसकी गणना करें \omterm{def:g4:measure:area}{क्षेत्र} दो जोड़कर
आयत.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:areas:7}
एक चित्र फ़्रेम: ए $30$ सेमी $\times $ $20$ ए के साथ सेमी आयत
$24$ सेमी $\times $ $14$ सेमी आयताकार खिड़की काट दी गई। क्या \omterm{def:g4:measure:area}{क्षेत्र} का
फ्रेम में लकड़ी का उपयोग होता है?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:areas:8}
की गणना करें \omterm{def:g4:measure:area}{क्षेत्र} पैरों के साथ एक \omterm{def:g3:shapes:rightangle}{समकोण} त्रिभुज का $8$ सेमी और $6$ सेमी
(\omterm{def:g3:division:half}{आधा} एक आयत, \cref{ex:g5:areas:halfrect}).
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:areas:9}
एक आयताकार सब्जी पैच का माप $7$ मी द्वारा $4$ एम। बीज की लागत
$2$ प्रति वर्ग \omterm{def:g4:fractions:def}{मीटर}, और बाड़ लगाना $3$ प्रति \omterm{def:g4:fractions:def}{मीटर}. की लागत की गणना करें
बीज, फिर बाड़ की लागत।
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g5:areas:10}
एक गलियारा फर्श है $12$ मी लंबा और $2$ मी चौड़ा है, और इसे ढका जाना चाहिए
किनारे की चौकोर टाइलों के साथ $50$ सेमी। कितनी टाइल्स की आवश्यकता है?
(पहले कनवर्ट करें, या प्रत्येक दिशा में टाइल्स गिनें।)
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g5:areas:11}
ए की भुजाओं को दोगुना करें $4$ सेमी $\times $ $3$ सेमी आयत. क्या होता है?
इसके लिए \omterm{def:g4:measure:perimeter}{परिधि}? इसके लिए \omterm{def:g4:measure:area}{क्षेत्र}? (पहले और बाद दोनों की गणना करें ---
दोनों उत्तर भिन्न हैं!)
\end{exercise}
