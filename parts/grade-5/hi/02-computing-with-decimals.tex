\chapter{दशमलव के साथ गणना}\label{ch:g5:decimalops}

दशमलव संख्याओं को लगभग पूर्ण संख्याओं की तरह ही \omterm{def:g1:addition:def}{जोड़}ा, घटाया और गुणा किया जाता है
--- पूरी कला यह जानने में है कि बात कहां जा रही है। यह अध्याय
इसमें कॉलम विधियों और गुणा करने या विभाजित करने के जादू को शामिल किया गया है
$10$, $100$, $1000$. (दो दशमलव को एक साथ गुणा करने पर प्रतीक्षा होती है
\cref{ch:g6:decimals}.)

\section{जोड़ना और घटाना}

\begin{method}[एक बिंदु के साथ कॉलम]\label{met:g5:decimalops:addsub}
\begin{enumerate}
  \item के साथ संख्याएँ लिखें \emph{दशमलव अंक गठबंधन}
        (तब इकाइयाँ इकाइयों के \omterm{ex:g1:subtraction:difference}{अंतर}्गत होती हैं, दशमांश दशमांश के \omterm{ex:g1:subtraction:difference}{अंतर}्गत);
  \item छोटे दशमलव भाग को पैड करें \omterm{def:g1:counting:zero}{शून्य};
  \item पूर्ण संख्याओं की तरह जोड़ना या घटाना;
  \item परिणाम के बिंदु को सीधे नीचे छोड़ें।
\end{enumerate}
\end{method}

\begin{example}\label{ex:g5:decimalops:addsub}
$27.5 + 3.86$ (पैड $27.5 = 27.50$):
\[
\begin{array}{r}
2\,7.5\,0 \\
+\ \ \ 3.8\,6 \\
\hline
3\,1.3\,6
\end{array}
\qquad\qquad
\begin{array}{r}
1\,2.4\,0 \\
-\ \ \ 5.6\,5 \\
\hline
\ \ \,6.7\,5
\end{array}
\]
घटाव के लिए ($12.4 - 5.65$): पैड, हमेशा की तरह उधार लें और जांचें
वापस जोड़कर: $6.75 + 5.65 = 12.4$.
\end{example}

\begin{example}[मानसिक पूरक]\label{ex:g5:decimalops:complement}
इसमें क्या \omterm{def:g1:addition:def}{जोड़}ा जाना चाहिए $7.6$ तक पहुँचने $10$? चढ़ना: $7.6 \to 8$ है
$0.4$, तब $8 \to 10$ है $2$: पूरक है $2.4$. के लिए सुविधाजनक
परिवर्तन: भुगतान करें $10$ एक के लिए $7.60$ आइटम, प्राप्त करें $2.40$ पीछे।
\end{example}

\section{दस गुना बड़ा, दस गुना छोटा}

\begin{proposition}[गुणा और 10, 100, 1000 से विभाजित]\label{prop:g5:decimalops:shift}
से गुणा करना $10$ प्रत्येक अंक का मूल्य दस गुना अधिक बनाता है:
अंक बाईं ओर एक स्थान खिसकते हैं --- जो दिखता है \emph{बिंदु
एक जगह दाहिनी ओर घूमना}. विभाजन विपरीत कार्य करता है:
\[
3.47 \times 10 = 34.7, \qquad
3.47 \times 100 = 347, \qquad
52.1 \div 10 = 5.21, \qquad
6 \div 100 = 0.06 .
\]
\end{proposition}
\admitted

\begin{example}[फिर से रूपांतरण]\label{ex:g5:decimalops:convert}
माप की इकाइयाँ दस अलग-अलग घातें हैं, इसलिए रूपांतरण उचित हैं
ये बदलाव: $2.35$ एम $= 235$ सेमी ($\times 100$); $470$ जी
$= 0.47$ किग्रा ($\div 1000$); $1.5$ एल $= 150$ सीएल.
\end{example}

\section{दशमलव को पूर्ण संख्या से गुणा करना}

\begin{method}[पूरे दशमलव समय]\label{met:g5:decimalops:mult}
ऐसे गिनें जैसे कोई मतलब ही न हो, फिर उतने ही परिणाम दें
दशमलव गुणक के रूप में दशमलव अंक:
\[
4.35 \times 6: \quad 435 \times 6 = 2\,610
\ \to\ 4.35 \times 6 = 26.10 = 26.1 .
\]
बात को आत्मविश्वास के साथ रखने का अनुमान लगाएं: $4.35$ के करीब है
$4$, और $4 \times 6 = 24$ --- इसलिए $26.1$, नहीं $2.61$ और न $261$.
\end{method}

\begin{example}[बार-बार जोड़ना अभी भी काम करता है]\label{ex:g5:decimalops:mult}
$0.75 \times 4 = 0.75 + 0.75 + 0.75 + 0.75 = 3$: चार क्वार्टरों बनाओ
एक संपूर्ण, दशमलव परिधान में। किसी पूर्ण संख्या से गुणा करना अभी भी बाकी है
``इतनी बार''।
\end{example}

\begin{example}[खरीदारी]\label{ex:g5:decimalops:shopping}
तीन नोटबुक पर $2.45$ प्रत्येक और एक कलम पर $1.20$:
\begin{enumerate}
  \item नोटबुक: $2.45 \times 3 = 7.35$;
  \item कुल: $7.35 + 1.20 = 8.55$;
  \item से बदलें $10$: $10 - 8.55 = 1.45$.
\end{enumerate}
\end{example}

\section{अभ्यास}

\begin{exercise}[$\star $]\label{exo:g5:decimalops:1}
कॉलम में गणना करें: $45.7 + 8.93$; \quad $16.25 + 3.75$; \quad
$0.86 + 12.4$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:decimalops:2}
कॉलम में गणना करें और जांचें: $9.4 - 2.72$; \quad $20 - 13.45$;
\quad $6.03 - 5.9$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:decimalops:3}
मानसिक रूप से पूरक है $10$ (के रूप में
\cref{ex:g5:decimalops:complement}): $6.2$; \quad $3.75$; \quad
$9.99$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:decimalops:4}
कॉलम के बिना गणना करें:
\[
7.24 \times 10, \qquad
0.58 \times 100, \qquad
34.5 \div 10, \qquad
7 \div 100, \qquad
0.3 \times 1000 .
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:decimalops:5}
बदलना: $4.05$ सेमी में मी; \quad $325$ मी में सेमी; \quad $0.6$ जी में किलो;
\quad $85$ एल में सीएल.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:decimalops:6}
पहले अनुमान लगाएं, फिर गणना करें: $6.2 \times 4$; \quad
$3.45 \times 8$; \quad $12.5 \times 6$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:decimalops:7}
ट्रैक का एक चक्कर मापता है $0.4$ किमी. कितने लंबे हैं $7$ गोद? और
$25$ गोद?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:decimalops:8}
ली खरीदता है $2$ baguettes पर $1.15$ प्रत्येक और एक केक पर $12.60$. वह
के साथ भुगतान करता है $20$ बिल। उसके कुल और उसके परिवर्तन की गणना करें।
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:decimalops:9}
एक बोतल रखती है $1.5$ एल. के एक पैक में कितने लीटर $6$ बोतलें?
एक परिवार शराब पीता है $0.75$ एल प्रति दिन: एक पैक कितने दिनों के लिए होता है
आख़िरी? (कितनी बार करता है $0.75$ में फिट $9$?)
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g5:decimalops:10}
मिलो गणना करता है $5.6 \times 3 = 15.18$, तर्क ``$5 \times 3 = 15$
और $6 \times 3 = 18$''. यह दिखाने के लिए अनुमान का उपयोग करें कि उत्तर अवश्य होना चाहिए
गलत है, तो सही गणना करें।
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g5:decimalops:11}
एक धावक ने चार चक्कर लगाये $65.4$ एस, $64.9$ एस, $65.0$ रेत
$66.1$ एस।
\begin{enumerate}
  \item चार चक्करों के कुल समय की गणना करें।
  \item चार लैप्स का रिकॉर्ड है $260$ एस। क्या उसने इसे हराया? से कितना?
\end{enumerate}
\end{exercise}
