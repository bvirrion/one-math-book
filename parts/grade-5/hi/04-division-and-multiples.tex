\chapter{विभाजन और गुणज}\label{ch:g5:division}

लांग डिवीजन में सीखा \cref{ch:g4:division} दो अपग्रेड मिलते हैं:
दो अंकीय वि\omterm{def:g5:division:multiple}{भाजक}, और \omterm{def:g3:division:remainder}{भागफल} जो बिंदु के बाद भी जारी है ---
$3 \div 4$ अंततः उत्तर मिलता है, $0.75$. अध्याय समाप्त होता है
\omterm{def:g5:division:multiple}{गुणकों}, वि\omterm{def:g5:division:multiple}{भाजक} और पहला विभाज्यता शॉर्टकट।

\section{दो अंकों की संख्या से भाग देना}

\begin{method}[दो अंकीय विभाजक]\label{met:g5:division:twodigits}
पहले जैसी ही विधि, एक अतिरिक्त कौशल के साथ: अनुमान लगाना कि कितनी बार
द \omterm{def:g5:division:multiple}{भाजक} फिट बैठता है. की एक छोटी तालिका लिखें \omterm{def:g5:division:multiple}{गुणकों} की \omterm{def:g5:division:multiple}{भाजक}
पहला। के लिए $986 \div 23$ (\omterm{def:g5:division:multiple}{गुणकों} 23 में से: 23, 46, 69, 92, 115,
138, 161, 184, 207):
\begin{enumerate}
  \item $98$ दहाई: $23 \times 4 = 92$ फिट बैठता है, $23 \times 5 = 115$ करता है
        नहीं: अंक $4$, \omterm{def:g3:division:remainder}{शेष} $6$;
  \item नीचे लाओ $6$: $66$: $23 \times 2 = 46$ फिट बैठता है: अंक
        $2$, \omterm{def:g3:division:remainder}{शेष} $20$;
  \item \omterm{def:g3:division:remainder}{भागफल} $42$, \omterm{def:g3:division:remainder}{शेष} $20$. जाँच करना:
        $23 \times 42 + 20 = 966 + 20 = 986$.
\end{enumerate}
\end{method}

\section{दशमलव भागफल}

\begin{example}[वह विभाजन जो शेष पर रुकने से इंकार करता है]\label{ex:g5:division:decimal}
शेयर करना $3$ पिज्जा के बीच $4$ बच्चे: $3 \div 4$ है \omterm{def:g3:division:remainder}{भागफल} $0$ और
\omterm{def:g3:division:remainder}{शेष} $3$ --- बेकार! बिंदु से आगे विभाजन जारी रखें:
\begin{enumerate}
  \item $3$ इकाइयां $= 30$ दसवाँ; $30 \div 4$: $7$ दसवाँ भाग, \omterm{def:g3:division:remainder}{शेष}
        $2$ दसवाँ;
  \item $2$ दसवां $= 20$ सौवां; $20 \div 4 = 5$ सौवां,
        \omterm{def:g3:division:remainder}{शेष} $0$.
\end{enumerate}
इसलिए $3 \div 4 = 0.75$: प्रत्येक बच्चे को मिलता है $0.75$ पिज़्ज़ा --- तीन
क्वार्टरों, के रूप में \omterm{def:g4:fractions:def}{अंश} भाषा पहले से ही पता थी
(\cref{prop:g5:fractions:decimal}).
\end{example}

\begin{method}[बिंदु से आगे विभाजन जारी रखना]\label{met:g5:division:continue}
जब इकाइयाँ समाप्त हो जाती हैं और a \omterm{def:g3:division:remainder}{शेष} \omterm{def:g3:division:remainder}{शेष} है:
\begin{enumerate}
  \item लिखना \omterm{def:g4:decimals:point}{दशमलव बिंदु} में \omterm{def:g3:division:remainder}{भागफल};
  \item ए \omterm{def:g1:addition:def}{जोड़}ें \omterm{def:g1:counting:zero}{शून्य} तक \omterm{def:g3:division:remainder}{शेष} (इकाइयां दसवीं, दसवीं हो जाती हैं
        सौवां हो जाओ\,\dots) और बांटते रहो;
  \item रुकें जब \omterm{def:g3:division:remainder}{शेष} है $0$ --- या जब अंक शुरू होते हैं
        हमेशा के लिए दोहराना, जैसे $1 \div 3 = 0.333\dots $: तो एक दे दो
        \omterm{def:g4:numbers:round}{गोल}ाकार मूल्य.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g5:division:continue}
$27 \div 6$: \omterm{def:g3:division:remainder}{भागफल} $4$, \omterm{def:g3:division:remainder}{शेष} $3$; बिंदु; $30$ दसवां
$\div\ 6 = 5$: इसलिए $27 \div 6 = 4.5$. \quad
$22 \div 8$: $2$, \omterm{def:g3:division:remainder}{शेष} $6$; तब $60 \div 8 = 7$ आर $4$;
तब $40 \div 8 = 5$: इसलिए $22 \div 8 = 2.75$.
\end{example}

\section{गुणज और भाजक}

\begin{definition}[एकाधिक, भाजक]\label{def:g5:division:multiple}
 \emph{गुणकों}\index{गुणकों} का $6$ क्या इसकी तालिका जारी है?
हमेशा के लिए: $0, 6, 12, 18, 24, \dots $ कब $24$ का गुणज है $6$,
एक यह भी कहता है $6$ एक है \emph{भाजक}\index{भाजक} का $24$,
या वह $24$ है \emph{भाज्य} द्वारा $6$: प्रखंड
$24 \div 6$ पत्ते नं \omterm{def:g3:division:remainder}{शेष}.
\end{definition}

\begin{proposition}[विभाज्यता शॉर्टकट]\label{prop:g5:division:criteria}
बिना विभाजित किये कोई संख्या विभाज्य होती है:
\begin{itemize}
  \item द्वारा $2$ जब इसका अंतिम अंक है $0$, $2$, $4$, $6$ या $8$
        (द सम संख्या);
  \item द्वारा $5$ जब इसका अंतिम अंक है $0$ या $5$;
  \item द्वारा $10$ जब इसका अंतिम अंक है $0$;
  \item द्वारा $3$ जब \emph{\omterm{def:g1:addition:def}{जोड़} इसके अंकों का} से विभाज्य है
        $3$ (उदा.\ $741$: $7 + 4 + 1 = 12$, विभाज्य --- और वास्तव में
        $741 = 3 \times 247$).
\end{itemize}
\end{proposition}
\admitted

\begin{example}\label{ex:g5:division:criteria}
है $534$ द्वारा विभाज्य $2$? में समाप्त होता है $4$: हाँ। द्वारा $5$? नहीं. द्वारा $3$?
$5 + 3 + 4 = 12$: हाँ। इसलिए $534$ से विभाज्य है $6$ साथ ही (द्वारा $2$
\emph{और} द्वारा $3$) --- जाँच करना: $534 = 6 \times 89$.
\end{example}

\section{अभ्यास}

\begin{exercise}[$\star $]\label{exo:g5:division:1}
लंबे विभाजनों की गणना करें (तालिका)। \omterm{def:g5:division:multiple}{गुणकों} पहला):
$851 \div 23$; \quad $704 \div 32$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:division:2}
सटीक दशमलव की गणना करें \omterm{def:g3:division:remainder}{भागफल}: $9 \div 2$; \quad $27 \div 4$;
\quad $33 \div 5$; \quad $21 \div 8$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:division:3}
चार दोस्त एक साझा करते हैं $54$ रेस्टोरेंट का बिल बराबर। कितना करता है
प्रत्येक भुगतान? (बिंदु से आगे जारी रखें।)
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:division:4}
गणना करना $10 \div 3$, बिंदु से तीन अंकों को आगे बढ़ाते हुए। क्या करें?
क्या आप निरीक्षण करते हैं? मान को सौवें तक पूर्णांकित करें।
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:division:5}
लिस्ट \omterm{def:g5:division:multiple}{गुणकों} का $7$ तक $70$; वि\omterm{def:g5:division:multiple}{भाजक} का $18$;
वि\omterm{def:g5:division:multiple}{भाजक} का $24$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:division:6}
द्वारा विभाज्य $2$? द्वारा $5$? द्वारा $10$? द्वारा $3$? प्रत्येक शॉर्टकट का परीक्षण करें:
$470$; \quad $735$; \quad $8\,001$; \quad $1\,314$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:division:7}
के बीच की सभी संख्याएँ ज्ञात कीजिए $60$ और $80$ जिनसे विभाज्य हैं
$3$ \emph{और} द्वारा $5$. (कौन सी एकल विभाज्यता ऐसा करती है
गठबंधन?)
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:division:8}
का एक रिबन $7.2$ मी में काटा गया है $6$ बराबर टुकड़े. प्रत्येक कितना लंबा है
टुकड़ा? (दशमलव को पूर्ण से विभाजित करें: \omterm{def:g4:fractions:def}{मीटर} साझा करें, फिर
दसवाँ भाग।)
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g5:division:9}
$1\,000$ मार्बल्स को बैगों में पैक किया जाता है $24$. कितने भरे बैग, और
कितने कंचे बचे हैं? (दो अंकीय \omterm{def:g5:division:multiple}{भाजक}.)
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g5:division:10}
कर सकना $5$ मित्र साझा करें $7$ चॉकलेट बार समान रूप से, कुछ भी नहीं बचा?
प्रत्येक शेयर को दशमलव के रूप में दें। के लिए वही प्रश्न $7$ मित्र साझा कर रहे हैं
$5$ बार --- क्या अलग होता है?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g5:division:11}
का उपयोग अंक-योग शॉर्टकट, सबसे छोटा अंक ढूंढें $ d $ कि बनाता है
$52d $ (अंत में समाप्त होने वाली तीन अंकों की संख्या $ d $) द्वारा विभाज्य $3$. फिर
ऐसे सभी अंक खोजें $ d$.
\end{exercise}
