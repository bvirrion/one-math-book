\chapter{Division et multiples}\label{ch:g5:division}

La division posée apprise dans \cref{ch:g4:division} reçoit deux
mises à niveau~: des diviseurs à deux chiffres, et des quotients
qui continuent après la virgule --- $3 \div 4$ obtient enfin une
réponse, $0.75$. Le chapitre se termine par les multiples, les
diviseurs et les premiers critères de divisibilité.

\section{Diviser par un nombre à deux chiffres}

\begin{method}[Diviseurs à deux chiffres]\label{met:g5:division:twodigits}
Même méthode qu'avant, avec une compétence en plus~: deviner combien
de fois le diviseur rentre. Écrire d'abord une petite table des
multiples du diviseur. Pour $986 \div 23$ (multiples de 23~: 23,
46, 69, 92, 115, 138, 161, 184, 207)~:
\begin{enumerate}
  \item $98$ dizaines~: $23 \times 4 = 92$ rentre, $23 \times 5 = 115$
        ne rentre pas~: chiffre $4$, reste $6$~;
  \item on abaisse le $6$~: $66$~: $23 \times 2 = 46$ rentre~: chiffre
        $2$, reste $20$~;
  \item quotient $42$, reste $20$. Vérification~:
        $23 \times 42 + 20 = 966 + 20 = 986$.
\end{enumerate}
\end{method}

\section{Quotients décimaux}

\begin{example}[La division qui refuse de s'arrêter au reste]\label{ex:g5:division:decimal}
Partager $3$ pizzas entre $4$ enfants~: $3 \div 4$ a pour quotient
$0$ et reste $3$ --- inutile~! On continue la division après la
virgule~:
\begin{enumerate}
  \item $3$ unités $= 30$ dixièmes~; $30 \div 4$~: $7$ dixièmes,
        reste $2$ dixièmes~;
  \item $2$ dixièmes $= 20$ centièmes~; $20 \div 4 = 5$ centièmes,
        reste $0$.
\end{enumerate}
Donc $3 \div 4 = 0.75$~: chaque enfant obtient $0.75$ pizza ---
trois quarts, comme le langage des fractions le savait déjà
(\cref{prop:g5:fractions:decimal}).
\end{example}

\begin{method}[Continuer une division après la virgule]\label{met:g5:division:continue}
Quand les unités sont épuisées et qu'il reste un reste~:
\begin{enumerate}
  \item écrire la virgule au quotient~;
  \item ajouter un zéro au reste (les unités deviennent des
        dixièmes, les dixièmes des centièmes\,\dots) et continuer à
        diviser~;
  \item s'arrêter quand le reste est $0$ --- ou quand les chiffres
        recommencent à jamais, comme $1 \div 3 = 0.333\dots$~: alors
        donner une valeur arrondie.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g5:division:continue}
$27 \div 6$~: quotient $4$, reste $3$~; virgule~; $30$ dixièmes
$\div\ 6 = 5$~: donc $27 \div 6 = 4.5$. \quad
$22 \div 8$~: $2$, reste $6$~; puis $60 \div 8 = 7$ r $4$~;
puis $40 \div 8 = 5$~: donc $22 \div 8 = 2.75$.
\end{example}

\section{Multiples et diviseurs}

\begin{definition}[Multiple, diviseur]\label{def:g5:division:multiple}
Les \emph{multiples}\index{multiple} de $6$ sont sa table
prolongée à l'infini~: $0, 6, 12, 18, 24, \dots$ Quand $24$ est un
multiple de $6$, on dit aussi que $6$ est un
\emph{diviseur}\index{diviseur} de $24$, ou que $24$ est
\emph{divisible} par $6$~: la division $24 \div 6$ ne laisse pas de
reste.
\end{definition}

\begin{proposition}[Critères de divisibilité]\label{prop:g5:division:criteria}
Sans diviser, un nombre est divisible~:
\begin{itemize}
  \item par $2$ quand son dernier chiffre est $0$, $2$, $4$, $6$
        ou $8$ (les nombres pairs)~;
  \item par $5$ quand son dernier chiffre est $0$ ou $5$~;
  \item par $10$ quand son dernier chiffre est $0$~;
  \item par $3$ quand la \emph{somme de ses chiffres} est
        divisible par $3$ (ex.~: $741$~: $7 + 4 + 1 = 12$,
        divisible --- et en effet $741 = 3 \times 247$).
\end{itemize}
\end{proposition}
\admitted

\begin{example}\label{ex:g5:division:criteria}
$534$ est-il divisible par $2$~? Se termine par $4$~: oui. Par
$5$~? Non. Par $3$~? $5 + 3 + 4 = 12$~: oui. Donc $534$ est aussi
divisible par $6$ (par $2$ \emph{et} par $3$) --- vérif.~:
$534 = 6 \times 89$.
\end{example}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:g5:division:1}
Calculer les divisions posées (table des multiples d'abord)~:
$851 \div 23$~; \quad $704 \div 32$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:division:2}
Calculer les quotients décimaux exacts~: $9 \div 2$~; \quad
$27 \div 4$~; \quad $33 \div 5$~; \quad $21 \div 8$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:division:3}
Quatre amis partagent équitablement une addition de restaurant de
$54$. Combien chacun paie-t-il~? (Continuer après la virgule.)
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:division:4}
Calculer $10 \div 3$, en continuant trois chiffres après la
virgule. Que remarque-t-on~? Donner la valeur arrondie au
centième.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:division:5}
Lister~: les multiples de $7$ jusqu'à $70$~; les diviseurs de
$18$~; les diviseurs de $24$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:division:6}
Divisible par $2$~? par $5$~? par $10$~? par $3$~? Tester chaque
critère sur~: $470$~; \quad $735$~; \quad $8\,001$~; \quad
$1\,314$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:division:7}
Trouver tous les nombres entre $60$ et $80$ qui sont divisibles par
$3$ \emph{et} par $5$. (Quelle seule divisibilité cela
combine-t-il~?)
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:division:8}
Un ruban de $7.2$~m est découpé en $6$ morceaux égaux. Quelle
longueur a chaque morceau~? (Diviser un décimal par un entier~:
partager les mètres, puis les dixièmes.)
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:division:9}
$1\,000$ billes sont emballées par sacs de $24$. Combien de sacs
pleins, et combien de billes restent~? (Diviseur à deux chiffres.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g5:division:10}
$5$ amis peuvent-ils se partager équitablement $7$ barres de
chocolat, sans rien laisser~? Donner chaque part en décimal. Même
question pour $7$ amis partageant $5$ barres --- qu'est-ce qui
change~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g5:division:11}
En utilisant le critère de la somme des chiffres, trouver le plus
petit chiffre $d$ qui rend $52d$ (un nombre à trois chiffres se
terminant par $d$) divisible par $3$. Puis trouver tous ces
chiffres $d$.
\end{exercise}
