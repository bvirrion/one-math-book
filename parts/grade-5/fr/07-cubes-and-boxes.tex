\chapter{Cubes et pavés}\label{ch:g5:solids}

Quitter la page plane~: les \omterm{def:g5:solids:def}{solides} sont les formes du monde réel ---
dés, boîtes de céréales, boîtes de conserve, balles. Ce chapitre
apprend à les nommer, compter leurs \omterm{def:g5:solids:def}{faces} et \omterm{def:g5:solids:def}{arêtes}, les déplier en
\omterm{def:g5:solids:net}{patrons}, et mesurer leur \omterm{def:g5:solids:volume}{volume} en comptant de petits \omterm{def:g5:solids:def}{cubes}. (Les
formules de \omterm{def:g5:solids:volume}{volume} arrivent dans \cref{ch:g6:measure}.)

\section{La famille des solides}

\begin{definition}[Solides et leur vocabulaire]\label{def:g5:solids:def}
Un \emph{solide}\index{solide} occupe de l'espace. Les côtés plats sont
des \emph{faces}\index{face}~; les faces se rejoignent le long
d'\emph{arêtes}\index{arête}~; les arêtes se rejoignent en des
\emph{sommets}\index{sommet}. Portraits de
famille~:
\begin{itemize}
  \item le \emph{cube}\index{cube}~: $6$ faces carrées, $12$ arêtes,
        $8$ sommets~;
  \item le \emph{pavé}\index{pavé} (prisme droit à base
        rectangulaire)~: $6$ faces rectangulaires, $12$ arêtes,
        $8$ sommets~;
  \item le \emph{cylindre}\index{cylindre}~: deux faces disques et une face
        enroulée (une boîte de conserve)~;
  \item la \emph{pyramide}\index{pyramide}~: une base polygonale et des triangles se
        rejoignant en un sommet~;
  \item la \emph{boule} (sphère)~: une surface parfaitement ronde,
        pas de face, pas d'arête.
\end{itemize}
\end{definition}

\begin{example}[Compter sur le cube]\label{ex:g5:solids:count}
Vérifier sur un dé~: $6$ \omterm{def:g5:solids:def}{faces} (les nombres $1$ à $6$), $8$
\omterm{def:g5:solids:def}{sommets} (les coins), $12$ \omterm{def:g5:solids:def}{arêtes} (passer un doigt le long~: $4$ en
haut, $4$ en bas, $4$ verticales). Et
$6 + 8 = 12 + 2$ --- le même motif curieux vaut pour le \omterm{def:g5:solids:def}{pavé} et la
\omterm{def:g5:solids:def}{pyramide} (\cref{ex:g8:solids:count} y reviendra).
\end{example}

\section{Patrons}

\begin{definition}[Patron]\label{def:g5:solids:net}
Un \emph{patron}\index{patron} d'un \omterm{def:g5:solids:def}{solide} est un dessin plat de
toutes ses \omterm{def:g5:solids:def}{faces}, attachées le long d'\omterm{def:g5:solids:def}{arêtes}, qui se plie pour
former le \omterm{def:g5:solids:def}{solide} --- le \omterm{def:g5:solids:def}{solide} déplié comme une boîte en carton
aplatie pour le recyclage.
\end{definition}

\begin{example}[Lire un patron]\label{ex:g5:solids:read}
Sur un dé, les \omterm{def:g5:solids:def}{faces} opposées additionnent $7$. Sur un \omterm{def:g5:solids:net}{patron}, les
\omterm{def:g5:solids:def}{faces} opposées sont celles séparées par exactement un carré dans
une rangée (ou autour d'un coin) --- pas les voisines~! Marquer
les paires sur le \omterm{def:g5:solids:net}{patron} avant de plier est un excellent exercice
pour voir l'espace à plat.
\end{example}

\section{Volume~: compter les cubes}

\begin{definition}[Volume par comptage]\label{def:g5:solids:volume}
Le \emph{volume}\index{volume} d'un \omterm{def:g5:solids:def}{solide} construit avec des \omterm{def:g5:solids:def}{cubes}
unitaires est le nombre de \omterm{def:g5:solids:def}{cubes} qu'il contient. L'unité peut être
le centimètre cube (cm$^3$) ou le mètre cube (m$^3$).
\end{definition}

\begin{example}[Compter un pavé couche par couche]\label{ex:g5:solids:layers}
Un \omterm{def:g5:solids:def}{pavé} de $4$ \omterm{def:g5:solids:def}{cubes} de long, $3$ de large, $2$ de haut~: la couche
du bas contient $4 \times 3 = 12$ \omterm{def:g5:solids:def}{cubes}, et il y a $2$ couches~:
\[
12 \times 2 = 24 \text{ cubes} .
\]
Compter par couches marche toujours --- et prouve discrètement la
formule $V = L \times \ell \times h$ de \cref{ch:g6:measure}.
\end{example}

\begin{example}[Mêmes cubes, formes différentes]\label{ex:g5:solids:rearrange}
Huit \omterm{def:g5:solids:def}{cubes} unitaires peuvent construire un \omterm{def:g5:solids:def}{cube} $2 \times 2 \times 2$,
un bâton $8 \times 1 \times 1$, ou une plaque $4 \times 2 \times 1$~:
trois \omterm{def:g5:solids:def}{solides} différents, un \omterm{def:g5:solids:volume}{volume} ($8$ \omterm{def:g5:solids:def}{cubes}). Le \omterm{def:g5:solids:volume}{volume} compte
la matière, pas la forme.
\end{example}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:g5:solids:1}
Nommer un objet réel en forme de~: \omterm{def:g5:solids:def}{cube}~; \omterm{def:g5:solids:def}{pavé}~; \omterm{def:g5:solids:def}{cylindre}~; boule~;
\omterm{def:g5:solids:def}{pyramide}.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:solids:2}
Combien de \omterm{def:g5:solids:def}{faces}, d'\omterm{def:g5:solids:def}{arêtes} et de \omterm{def:g5:solids:def}{sommets} a un \omterm{def:g5:solids:def}{pavé}~? Vérifier le
motif \omterm{def:g5:solids:def}{faces} $+$ \omterm{def:g5:solids:def}{sommets} $=$ \omterm{def:g5:solids:def}{arêtes} $+ 2$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:solids:3}
Quelles \omterm{def:g5:solids:def}{faces} d'un \omterm{def:g5:solids:def}{pavé} sont identiques~? (Regrouper les six \omterm{def:g5:solids:def}{faces}
par paires.) Qu'y a-t-il de spécial aux \omterm{def:g5:solids:def}{faces} du \omterm{def:g5:solids:def}{cube}~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:solids:4}
Dessiner un \omterm{def:g5:solids:net}{patron} d'un \omterm{def:g5:solids:def}{cube} de côté $2$~cm (utiliser le modèle en
croix du chapitre), le découper et le plier.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:solids:5}
Sur le \omterm{def:g5:solids:net}{patron} en croix d'un dé, placer les nombres $1$ à $6$ de
sorte que les \omterm{def:g5:solids:def}{faces} opposées additionnent $7$. (Utiliser
\cref{ex:g5:solids:read} pour trouver d'abord les paires
opposées.)
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:solids:6}
Compter les \omterm{def:g5:solids:def}{cubes} unitaires~: un \omterm{def:g5:solids:def}{pavé} de $5$ de long, $2$ de large,
$3$ de haut~; un \omterm{def:g5:solids:def}{cube} de côté $3$~; un empilement en L fait d'une
plaque $3 \times 2 \times 1$ avec une plaque $1 \times 2 \times 1$
dessus.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:solids:7}
Un \omterm{def:g5:solids:def}{pavé} contient exactement $12$ \omterm{def:g5:solids:def}{cubes} unitaires en une couche.
Donner deux formes de couche possibles ($L \times \ell$), et pour
chacune, le \omterm{def:g5:solids:volume}{volume} du \omterm{def:g5:solids:def}{pavé} s'il a $3$ telles couches.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:solids:8}
Avec $27$ \omterm{def:g5:solids:def}{cubes} unitaires, quel \omterm{def:g5:solids:def}{cube} peut-on construire~? Avec
$64$~? (Penser au comptage par couches à l'envers.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g5:solids:9}
Des morceaux de sucre de côté $1$~cm viennent dans une boîte de
$10$~cm de long, $5$~cm de large, $4$~cm de haut (dimensions
intérieures). Combien de morceaux de sucre tiennent~? Si une
famille en utilise $6$ par jour, pendant combien de jours une boîte
suffit-elle (\omterm{def:g4:numbers:round}{arrondir} sensément)~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g5:solids:10}
Un \omterm{def:g5:solids:def}{cube} $3 \times 3 \times 3$ est peint en rouge à l'extérieur,
puis découpé en $27$ \omterm{def:g5:solids:def}{cubes} unitaires. Combien de petits \omterm{def:g5:solids:def}{cubes} ont
de la peinture sur exactement $3$ \omterm{def:g5:solids:def}{faces}~? Sur $2$~? Sur $1$~? Sur
aucune~? (Vérifier~: les quatre comptes doivent additionner $27$.)
\end{exercise}
