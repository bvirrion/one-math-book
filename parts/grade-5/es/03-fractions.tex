\chapter{Fracciones como números}\label{ch:g5:fractions}

Un \omterm{def:g4:fractions:def}{fracción} no es sólo la imagen de un pastel: es un auténtico
\emph{número}, con un lugar en la recta numérica --- a veces el mismo
mismo lugar que un número decimal. Este capítulo conecta los dos
mundos. (Las reglas completas de la aritmética \omterm{def:g4:fractions:def}{fracción} vienen en
\cref{ch:g6:fractions} y posteriores).

\section{Fracciones en la recta, otra vez}

\begin{example}[Colocación y lectura]\label{ex:g5:fractions:place}
En una línea cortada en quintas, $\frac{7}{5}$ es $7$ pasos de $0$: pasado
$1$ (que es $\frac55$), en $1 + \frac25$. Cada \omterm{def:g4:fractions:def}{fracción} vive
en algún lugar de la línea; \omterm{def:g4:fractions:def}{numerador} más grande (mismo \omterm{def:g4:fractions:def}{denominador}) significa
más a la derecha.
\end{example}

\begin{proposition}[Mismo punto, muchos nombres.]\label{prop:g5:fractions:equal}
Cortar cada pieza en dos (o tres\,\dots) no mueve el
punto:
\[
\frac{1}{2} = \frac{2}{4} = \frac{3}{6} = \frac{5}{10},
\qquad
\frac{2}{3} = \frac{4}{6} = \frac{20}{30}.
\]
Multiplicando (o dividiendo) el \omterm{def:g4:fractions:def}{numerador} y el \omterm{def:g4:fractions:def}{denominador} por el
El mismo número da otro nombre del mismo \omterm{def:g4:fractions:def}{fracción}.
\end{proposition}
\admitted

\section{fracciones y decimales}

\begin{proposition}[Fracciones con nombres decimales]\label{prop:g5:fractions:decimal}
\omterm{def:g4:fractions:def}{fracciones} de décimas, centésimas, milésimas \emph{son} decimal
números:
\[
\frac{3}{10} = 0.3, \qquad \frac{27}{100} = 0.27 .
\]
Algunos otros \omterm{def:g4:fractions:def}{fracciones} también tienen nombres decimales, que se encuentran cambiando a
décimas o centésimas:
\[
\frac{1}{2} = \frac{5}{10} = 0.5, \qquad
\frac{1}{4} = \frac{25}{100} = 0.25, \qquad
\frac{3}{4} = 0.75, \qquad
\frac{1}{5} = \frac{2}{10} = 0.2 .
\]
Pero no todo: $\frac13 = 0.333\dots$ nunca termina --- el \omterm{def:g4:fractions:def}{fracción} es
su único nombre exacto.
\end{proposition}
\admitted

\begin{example}[Elegir el nombre más práctico]\label{ex:g5:fractions:choose}
Para pagar tres \omterm{def:g3:division:half}{cuarteles} de $12$: el nombre \omterm{def:g4:fractions:def}{fracción} es más práctico,
$\frac34$ de $12$ es $9$. Para comparar $\frac34$ y $0.8$: el
El nombre decimal es más práctico, $0.75 < 0.8$. Ser bilingüe vale la pena.
\end{example}

\section{Comparar fracciones}

\begin{method}[Tres herramientas de comparación]\label{met:g5:fractions:compare}
\begin{enumerate}
  \item \emph{Same \omterm{def:g4:fractions:def}{denominador}}: más piezas ganan,
        $\frac57 > \frac37$;
  \item \emph{same \omterm{def:g4:fractions:def}{numerador}}: las piezas más grandes ganan, por lo que \emph{menor}
        \omterm{def:g4:fractions:def}{denominador} gana: $\frac35 > \frac38$ (las quintas son más grandes
        de octavos);
  \item \emph{puntos de referencia}: compare cada uno con $\frac12$ o $1$:
        $\frac38 < \frac12 < \frac59$, entonces $\frac38 < \frac59$.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g5:fractions:compare}
¿Quién comió más pizza: Ali con $\frac58$ o Bea con $\frac54$? bea
\omterm{def:g4:fractions:def}{fracción} es más grande que $1$ (cinco \omterm{def:g3:division:half}{cuarteles} es más que un entero
pizza), Ali's es más pequeño que $1$: Bea comió más --- más de un
pizza entera, de hecho.
\end{example}

\section{Ceremonias}

\begin{exercise}[$\star$]\label{exo:g5:fractions:1}
Dibuja una recta numérica cortada en quintas desde $0$ a $2$ y coloca:
$\frac25$; $\frac55$; $\frac85$; $\frac{10}{5}$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:fractions:2}
Completa los nombres iguales:
$\frac12 = \frac{?}{8}$; \quad
$\frac34 = \frac{?}{8}$; \quad
$\frac{2}{5} = \frac{4}{?}$; \quad
$\frac{6}{9} = \frac{2}{?}$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:fractions:3}
¿Qué \omterm{def:g4:fractions:def}{fracciones} nombra números enteros? Da el número entero cuando sea así:
$\frac82$; \quad$\frac{9}{4}$; \quad$\frac{15}{3}$; \quad
$\frac{20}{5}$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:fractions:4}
Escribe como decimales: $\frac{7}{10}$; \quad$\frac{41}{100}$; \quad
$\frac12$; \quad$\frac34$; \quad $\frac15$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:fractions:5}
Escribe como \omterm{def:g4:fractions:def}{fracciones} (décimas o centésimas): $0.9$; \quad$0.13$;
\quad $2.5$; \quad $0.75$ (luego proporcione su nombre \omterm{def:g3:division:half}{cuarto}).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:fractions:6}
Comparar, nombrando la herramienta utilizada
(\cref{met:g5:fractions:compare}):
\[
\frac47 \;?\; \frac67, \qquad
\frac35 \;?\; \frac37, \qquad
\frac25 \;?\; \frac{7}{12}, \qquad
\frac98 \;?\; \frac{11}{12} .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:fractions:7}
Orden con los nombres decimales: $\frac12$; $0.4$; $\frac34$; \
$0.6$; $\frac15$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:fractions:8}
Un \omterm{def:g3:division:half}{cuarto} de los alumnos de una clase de $28$ toca un instrumento, y
\omterm{def:g3:division:half}{medio} practicar un deporte. ¿Cuántos estudiantes tiene cada uno? ¿Qué grupo es más grande?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:fractions:9}
Relacione cada \omterm{def:g4:fractions:def}{fracción} con una hora: $\frac14$ h, $\frac12$ h,
$\frac34$ h, $\frac{1}{60}$ h --- a $30$ min, $45$ min, $1$ min,
$15$ mín.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g5:fractions:10}
Tom bebió $\frac35$ de su botella de $50$ cL; Ana bebió $\frac12$ de
su botella $60$ cL. ¿Quién bebió más? (Calcule ambas cantidades en cL ---
comparar el \omterm{def:g4:fractions:def}{fracciones} desnudo no es suficiente aquí, y diga por qué).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g5:fractions:11}
Dé un \omterm{def:g4:fractions:def}{fracción} estrictamente entre $\frac12$ y $1$; entonces uno
estrictamente entre $\frac12$ y $\frac34$. (Los nombres decimales pueden
ayuda.)
\end{exercise}
