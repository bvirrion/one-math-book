\chapter{Números decimales}\label{ch:g5:decimals}

Las décimas y centésimas aparecieron en \cref{ch:g4:decimals}; agregando uno
más nivel, milésimas, completa el sistema decimal. este capitulo
consolida la lectura, la descomposición y --- sobre todo --- la comparación
números decimales sin caer en sus famosas trampas.

\section{Hasta milésimas}

\begin{definition}[Lugares decimales]\label{def:g5:decimals:places}
Después del \omterm{def:g4:decimals:point}{coma decimal} vienen las décimas, las centésimas, las
milésimas --- cada una vale diez veces menos que la anterior:
\[
4.362 = 4 + \frac{3}{10} + \frac{6}{100} + \frac{2}{1000}
= \frac{4\,362}{1000}.
\]
\end{definition}

\begin{example}[Muchas lecturas, un número]\label{ex:g5:decimals:readings}
$4.362$ se puede leer ``cuatro punto tres seis dos'', o ``cuatro y
$362$ milésimas'', o descompuesto como $4 + 0.3 + 0.06 + 0.002$. y
el relleno con \omterm{def:g1:counting:zero}{ceros} final no cambia nada: $4.362 = 4.3620$. pero
$4.362 \neq 4.0362$: a \omterm{def:g1:counting:zero}{cero} \emph{adentro} empuja cada dígito a un
lugar más pequeño.
\end{example}

\begin{example}[¿Dónde está en línea?]\label{ex:g5:decimals:line}
Para colocar $4.362$: entre $4$ y $5$; haciendo zoom, entre $4.3$ y
$4.4$; haciendo zoom nuevamente, entre $4.36$ y $4.37$, más cerca de $4.36$.
Cada dígito decimal es un nivel más de zoom.
\end{example}

\section{Comparando decimales}

\begin{method}[Comparando dos decimales]\label{met:g5:decimals:compare}
\begin{enumerate}
  \item Comparar las piezas enteras: $6.1 > 5.987$.
  \item Partes enteras iguales: compara las partes decimales dígito a dígito
        desde el punto --- décimas, luego centésimas, luego
        milésimas;
  \item El relleno  con \omterm{def:g1:counting:zero}{ceros} final hace que la comparación sea justa:
        $2.5$ vs $2.48$ se convierte en $2.50$ vs $2.48$ y $50 > 48$
        centésimas.
\end{enumerate}
La trampa, una última vez: \emph{una parte decimal más larga no significa una
numero mas grande} --- $2.48$ tiene más dígitos que $2.5$ y es más pequeño.
\end{method}

\begin{example}\label{ex:g5:decimals:order}
Orden $7.3$; $7.09$; $7.31$; $7.299$. Rellenar con tres decimales:
$7.300$; $7.090$; $7.310$; $7.299$. entonces
\[
7.09 < 7.299 < 7.3 < 7.31 .
\]
Nota $7.299 < 7.3$ aunque $299$ parece grande: $299$ milésimas
contra $300$ milésimas.
\end{example}

\begin{example}[Apretando entre dos decimales]\label{ex:g5:decimals:between}
¿Existe un número entre $5.7$ y $5.8$? Sí, muchos: $5.75$,
$5.71$, $5.799$\,\dots\ ¿Entre $5.79$ y $5.8$? De nuevo mucho:
$5.795$, por ejemplo. Entre dos decimales diferentes, se puede
\emph{siempre} aprieta otro --- no hay un decimal ``siguiente''
número.
\end{example}

\section{Redondear decimales}

\begin{definition}[redondeo]\label{def:g5:decimals:round}
Para \omterm{def:g4:numbers:round}{redondo} a la unidad (o décimo, o centésimo), mira la siguiente
dígito: $5$ o más \omterm{def:g4:numbers:round}{rondas} hacia arriba, de lo contrario \omterm{def:g4:numbers:round}{redondo} hacia abajo. Entonces $4.362$
\omterm{def:g4:numbers:round}{rondas} a $4$ (unidad), a $4.4$ (décimo), a $4.36$ (centésimo).
\end{definition}

\begin{example}[Redondeos de dinero a centavos]\label{ex:g5:decimals:money}
Un precio calculado como $7.4983$ se muestra como $7.50$: vida real
las cantidades se redondean a la centésima. Tenga en cuenta la cascada: $7.4983 \to
7.50$, donde $49$ se convirtió en $50$ --- \omterm{def:g4:numbers:round}{redondeo} puede extenderse
varios dígitos.
\end{example}

\section{Ceremonias}

\begin{exercise}[$\star$]\label{exo:g5:decimals:1}
Escribe como número decimal: $5 + \frac{2}{10} + \frac{7}{1000}$;
\quad$\frac{85}{100}$; \quad$\frac{4\,507}{1000}$; \quad ``doce
y nueve milésimas''.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:decimals:2}
En $27.418$: ¿qué cuenta el $4$? ¿El $8$? descomponer el numero
como en \cref{def:g5:decimals:places}.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:decimals:3}
¿Verdadero o falso? Explica con lugares.
$3.60 = 3.6$; \quad$0.5 = 0.05$; \quad $2.070 = 2.07$; \quad
$1.3 = 1.03$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:decimals:4}
Copiar y completar con $<$, $>$ o $=$:
\[
4.8 \;?\; 4.79, \qquad
0.302 \;?\; 0.32, \qquad
6.5 \;?\; 6.500, \qquad
9.09 \;?\; 9.1 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:decimals:5}
Ordenar de menor a mayor: $3.14$; $3.4$; $3.041$; \
$3.104$; $3.41$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:decimals:6}
¿Entre qué dos números enteros se encuentra cada número? entre los cuales
¿Dos números con un decimal? \quad$6.83$; \quad$0.492$; \quad
$12.06$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:decimals:7}
Ronda $8.276$: a la unidad; al décimo; a la centésima. redondo
$3.95$ al décimo.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:decimals:8}
Indique un número estrictamente entre: $2.6$ y $2.7$; \quad $0.41$ y
$0.42$; \quad $5.99$ y $6$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:decimals:9}
Las alturas de cuatro plantas son $0.85$ m, $1.2$ m, $0.9$ m y
$1.02$m. Ordénalos del más bajo al más alto.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g5:decimals:10}
Lucie dice: ``$7.12 > 7.9$ porque $12 > 9$''. Corrija su error,
comparando décimas primero.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g5:decimals:11}
Encuentre todos los números con exactamente dos dígitos decimales que \omterm{def:g4:numbers:round}{redondo}
$6.4$ cuando se redondea a la décima. ¿Cuáles son los más pequeños y los
¿más grande?
\end{exercise}
