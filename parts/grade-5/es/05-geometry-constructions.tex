\chapter{Construcciones Geométricas}\label{ch:g5:geometry}

Regla, escuadra, compás: con estas tres herramientas y un poco de
método, cualquier figura descrita con palabras se puede construir exactamente. esto
construcciones del capítulo \omterm{def:g4:geometry:def}{perpendiculares}, \omterm{def:g4:geometry:def}{paralelas}, triángulos y el
especial \omterm{def:g4:geometry:polygon}{cuadriláteros} --- artesanía primero, teoría después
(\cref{ch:g6:lines,ch:g6:shapes}).

\section{Perpendiculares y paralelas que pasan por un punto.}

\begin{method}[Las dos construcciones básicas]\label{met:g5:geometry:basic}
\emph{\omterm{def:g4:geometry:def}{Perpendicular} to $d$ through $A$}: deslizar la escuadra a lo largo
$d$ hasta que su otro borde en ángulo recto llegue a $A$; dibujar; marcar el
\omterm{def:g3:shapes:rightangle}{ángulo recto}.

\emph{\omterm{def:g4:geometry:def}{Paralelo} to $d$ through $A$}: presionar una regla contra el conjunto
cuadrado como un riel colocado a lo largo de $d$; deslice la escuadra hasta $A$;
dibujar. (Compruebe: el espacio entre las dos líneas es el mismo
en todas partes.)
\end{method}

\begin{example}[Encadenamiento de construcciones]\label{ex:g5:geometry:chain}
Muchas figuras son cadenas de estos dos movimientos. Un rectángulo $ABCD$ con
$AB = 5$ cm y $BC = 3$ cm: dibujar $[AB]$; \omterm{def:g4:geometry:def}{perpendicular} en $A$;
\omterm{def:g4:geometry:def}{perpendicular} en $B$; marcar $D$ y $C$ en $3$ cm en el mismo lado;
unirse. Cada paso es una construcción básica.
\end{example}

\section{Triángulos con el compás}

\begin{method}[Triangulo de tres lados]\label{met:g5:geometry:triangle}
Para construir un triángulo con lados $6$ cm, $5$ cm, $4$ cm:
\begin{enumerate}
  \item dibuja el lado más largo, $[AB]$ con $AB = 6$ cm;
  \item arco de centro $A$ y \omterm{def:g3:shapes:shapes}{radio} $5$ cm;
  \item arco de centro $B$ y \omterm{def:g3:shapes:shapes}{radio} $4$ cm;
  \item su cruce es el tercero \omterm{def:g4:geometry:polygon}{vértice} $C$; unirse.
\end{enumerate}
La brújula garantiza las longitudes: cada punto del primer arco es
exactamente a $5$ cm de $A$.
\end{method}

\begin{example}[Isósceles y equiláteros gratis]\label{ex:g5:geometry:special}
Misma apertura del compás para ambos arcos $\Rightarrow$ triángulo isósceles.
Apertura igual a la base $\Rightarrow$ equilátera. la brújula
no mide: mide \emph{copias}, lo cual es aún mejor.
\end{example}

\section{Los cuadriláteros especiales, por construcción.}

\begin{method}[Construyendo la familia]\label{met:g5:geometry:quadrilaterals}
\begin{enumerate}
  \item \emph{Rectángulo}: dos pares de \omterm{def:g4:geometry:def}{paralelas} cruzando en \omterm{def:g3:shapes:rightangle}{correcto
        ángulos} --- o la cadena de \cref{ex:g5:geometry:chain};
  \item \emph{cuadrado}: un rectángulo cuyos cuatro lados usan el mismo
        longitud --- llévalo con la brújula;
  \item \emph{rombo}: cuatro lados iguales --- dibuja un lado, luego dos
        arcos de compás desde sus extremos con la misma abertura fijan el
        otros dos \omterm{def:g4:geometry:polygon}{vértices} (un rombo son dos triángulos isósceles
        pegado a lo largo de una diagonal).
\end{enumerate}
Después de construir, siempre \emph{controlar} con las herramientas: coloque la escuadra en
las esquinas, compás en los lados.
\end{method}

\begin{example}[Reconocer por una descripción]\label{ex:g5:geometry:recognize}
``A \omterm{def:g4:geometry:polygon}{cuadrilátero} con cuatro lados $4$ cm y un \omterm{def:g3:shapes:rightangle}{ángulo recto}'' ---
construirlo: los cuatro lados iguales fuerzan una forma de rombo, y presionando
una esquina de un \omterm{def:g3:shapes:rightangle}{ángulo recto} endereza todas las demás: el resultado
es un cuadrado. Algunas descripciones describen en secreto algo más especial.
forma de la que anuncian. (¿Por qué el \omterm{def:g3:shapes:rightangle}{ángulo recto} se propaga a los cuatro
Las esquinas se explican en \cref{ch:g7:central}.)
\end{example}

\section{Círculos en construcciones.}

\begin{example}[El círculo como herramienta]\label{ex:g5:geometry:circle}
``Encontrar todos los puntos en $3$ cm desde $A$'': el círculo del centro $A$,
\omterm{def:g3:shapes:shapes}{radio} $3$ cm. ``Encontrar todos los puntos en $3$ cm desde $A$ \emph{y}
$4$ cm de $B$'': dibujar ambos círculos; sus cruces (dos, uno o
\omterm{def:g1:counting:zero}{cero} puntos) son las respuestas. Mapas del tesoro y triángulo.
Las construcciones utilizan exactamente esta idea.
\end{example}

\section{Ceremonias}

\begin{exercise}[$\star$]\label{exo:g5:geometry:1}
Dibuja una línea $d$ y un punto $A$ a aproximadamente $3$ cm de ella. construir
del \omterm{def:g4:geometry:def}{perpendicular} al $d$ al $A$, luego del \omterm{def:g4:geometry:def}{paralelo} al $d$
hasta $A$. Marque el \omterm{def:g3:shapes:rightangle}{ángulos rectos}.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:geometry:2}
Construya un rectángulo $ABCD$ con $AB = 7$ cm y $BC = 4$ cm,
siguiente \cref{ex:g5:geometry:chain}. Comprueba el cuarto ángulo con
La escuadra: si la construcción es exacta, es correcta
automáticamente.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:geometry:3}
Construye un triángulo con lados $7$ cm, $6$ cm y $3$ cm.
(\cref{met:g5:geometry:triangle}).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:geometry:4}
Construye un triángulo isósceles con base $5$ cm y lados iguales de
$7$cm; luego un triángulo equilátero de lado $6$ cm.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:geometry:5}
Construye un cuadrado de lado $5$ cm. ¿Qué herramientas garantizan el \omterm{def:g3:shapes:rightangle}{correcto
ángulos}? ¿Cuáles garantizan los lados iguales?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:geometry:6}
Construye un rombo de lado $4$ cm que no sea un cuadrado (elige tu
propia apertura para la primera curva). Verifique los cuatro lados con el
brújula.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:geometry:7}
Dibuja dos puntos $A$ y $B$ con $AB = 5$ cm. Construir todos los puntos
que están en $4$ cm de $A$ y en $3$ cm de $B$. cuantos son
ahí?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:geometry:8}
Escriba un programa de construcción (pasos numerados, puntos con nombre) para un
rectángulo $6$ cm $\times$ $2$ cm rematado por un triángulo isósceles
con lados iguales $4$ cm (forma de lápiz). Haz que un compañero ejecute
eso.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g5:geometry:9}
Intenta construir un triángulo con lados $8$ cm, $3$ cm y $4$ cm.
¿Qué pasa con los arcos? Explica en una frase por qué esto
El triángulo no puede existir.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g5:geometry:10}
Construya un rombo cuyas diagonales midan $6$ cm y $4$ cm,
sabiendo que las diagonales de un rombo se cruzan en sus centros en un
\omterm{def:g3:shapes:rightangle}{ángulo recto}: dibuja las diagonales \emph{primero}, luego une sus cuatro
termina.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g5:geometry:11}
Una cabra está atada a una estaca $A$ mediante una cuerda de $4$ m, en un terreno llano con
una valla recta que pasa por $3$ m desde $A$. Dibujar un plano (1 cm por
metro): ¿a qué región puede llegar la cabra? ¿Dónde limita la valla?
eso?
\end{exercise}
