\chapter{Cubos y Cajas}\label{ch:g5:solids}

Saliendo de la página plana: \omterm{def:g5:solids:def}{sólidos} son las formas del mundo real ---
dados, cajas de cereales, latas, pelotas. Este capítulo aprende a nombrarlos,
cuente sus \omterm{def:g5:solids:def}{caras} y \omterm{def:g5:solids:def}{bordes}, desdóblelos en \omterm{def:g5:solids:net}{redes} y mida
su \omterm{def:g5:solids:volume}{volumen} contando el poco \omterm{def:g5:solids:def}{cubos}. (Las fórmulas \omterm{def:g5:solids:volume}{Volumen} vienen en
\cref{ch:g6:measure}.)

\section{la familia solida}

\begin{definition}[Sólidos y su vocabulario]\label{def:g5:solids:def}
Un \emph{sólido}\index{sólido} ocupa espacio. Los lados planos son
\emph{caras}\index{rostro}; los rostros se encuentran a lo largo de \emph{bordes}\index{borde};
los bordes se encuentran en \emph{vértices}\index{vértice}. la familia
retratos:
\begin{itemize}
  \item o \emph{cubo}\index{cubo}: $6$ caras cuadradas, $12$ aristas,
        $8$ vértices;
  \item o \emph{caja}\index{caja} (prisma rectangular): $6$
        caras rectangulares, aristas $12$, vértices $8$;
  \item o \emph{cilindro}\index{cilindro}: dos caras de disco y una cara enrollada
        (una lata);
  \item el \emph{pirámide}\index{pirámide}: una base \omterm{def:g4:geometry:polygon}{polígono} y triángulos que se encuentran en
        un ápice;
  \item o \emph{pelota} (esfera): una superficie perfectamente redonda, sin
        cara, sin filo.
\end{itemize}
\end{definition}

\begin{example}[Contando con el cubo]\label{ex:g5:solids:count}
Verifique en un troquel: $6$ \omterm{def:g5:solids:def}{caras} (los números $1$ a $6$), $8$ \omterm{def:g5:solids:def}{vértices}
(las esquinas), $12$ \omterm{def:g5:solids:def}{bordes} (pasa un dedo por ellas: $4$ arriba, $4$
en la parte inferior, $4$ vertical). y
$6 + 8 = 12 + 2$ --- el mismo patrón curioso se aplica a \omterm{def:g5:solids:def}{caja} y
el \omterm{def:g5:solids:def}{pirámide} (\cref{ex:g8:solids:count} vuelve a él).
\end{example}

\section{Redes}

\begin{definition}[neto]\label{def:g5:solids:net}
Un \emph{neto}\index{neto} de un \omterm{def:g5:solids:def}{sólido} es un dibujo plano de todos sus
\omterm{def:g5:solids:def}{caras}, adjunto a lo largo de \omterm{def:g5:solids:def}{bordes}, que se pliega en \omterm{def:g5:solids:def}{sólido} --- el
\omterm{def:g5:solids:def}{sólido} desenvuelto como un cartón \omterm{def:g5:solids:def}{caja} aplanado para su reciclaje.
\end{definition}

\begin{example}[Leyendo una roja]\label{ex:g5:solids:read}
En un dado, lo opuesto a \omterm{def:g5:solids:def}{caras} suman $7$. En un \omterm{def:g5:solids:net}{neto}, frente a \omterm{def:g5:solids:def}{caras} están
los que están separados exactamente por un cuadrado en una fila (o alrededor de un
esquina) --- ¡no los vecinos! Marcando los pares en el \omterm{def:g5:solids:net}{neto} antes
Plegar es un gran ejercicio para ver el espacio plano.
\end{example}

\section{Volumen: contar cubos}

\begin{definition}[volumen contando]\label{def:g5:solids:volume}
El \emph{volumen}\index{volumen} de un \omterm{def:g5:solids:def}{sólido} construido a partir de la unidad \omterm{def:g5:solids:def}{cubos} es
el número de \omterm{def:g5:solids:def}{cubos} que contiene. La unidad puede ser el centímetro \omterm{def:g5:solids:def}{cubo}
(cm$^3$) o el medidor \omterm{def:g5:solids:def}{cubo} (m$^3$).
\end{definition}

\begin{example}[Contando un cuadro capa por capa]\label{ex:g5:solids:layers}
A \omterm{def:g5:solids:def}{caja} $4$ \omterm{def:g5:solids:def}{cubos} largo, $3$ ancho, $2$ alto: la capa inferior aguanta
$4 \times 3 = 12$ \omterm{def:g5:solids:def}{cubos}, y hay capas $2$:
\[
12 \times 2 = 24 \text{ cubes} .
\]
Contar por capas siempre funciona, y demuestra silenciosamente la
fórmula $V = L \times w \times h$ de \cref{ch:g6:measure}.
\end{example}

\begin{example}[Mismos cubos, diferentes formas.]\label{ex:g5:solids:rearrange}
Ocho unidades \omterm{def:g5:solids:def}{cubos} pueden construir un $2 \times 2 \times 2$ \omterm{def:g5:solids:def}{cubo}, un
palo $8 \times 1 \times 1$, o una losa $4 \times 2 \times 1$: tres
diferente \omterm{def:g5:solids:def}{sólidos}, uno \omterm{def:g5:solids:volume}{volumen} ($8$ \omterm{def:g5:solids:def}{cubos}). \omterm{def:g5:solids:volume}{Volumen} cuenta el
material, no la forma.
\end{example}

\section{Ceremonias}

\begin{exercise}[$\star$]\label{exo:g5:solids:1}
Nombra un objeto real con la forma: \omterm{def:g5:solids:def}{cubo}; un \omterm{def:g5:solids:def}{caja}; un \omterm{def:g5:solids:def}{cilindro}; una pelota;
a \omterm{def:g5:solids:def}{pirámide}.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:solids:2}
¿Cuántos \omterm{def:g5:solids:def}{caras}, \omterm{def:g5:solids:def}{bordes} y \omterm{def:g5:solids:def}{vértices} tiene un \omterm{def:g5:solids:def}{caja}? Comprueba el patrón
\omterm{def:g5:solids:def}{caras} $+$ \omterm{def:g5:solids:def}{vértices} $=$ \omterm{def:g5:solids:def}{bordes} $+ 2$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:solids:3}
¿Qué \omterm{def:g5:solids:def}{caras} de un \omterm{def:g5:solids:def}{caja} son idénticos? (Agrupe los seis \omterm{def:g5:solids:def}{caras} en pares).
¿Qué tiene de especial el \omterm{def:g5:solids:def}{caras} del \omterm{def:g5:solids:def}{cubo}?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:solids:4}
Dibuje un \omterm{def:g5:solids:net}{neto} de un \omterm{def:g5:solids:def}{cubo} de lado $2$ cm (use el modelo en forma de cruz de
el capítulo), recórtalo y dóblalo.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:solids:5}
En el \omterm{def:g5:solids:net}{neto} en forma de cruz de un dado, coloque los números $1$ a $6$ para que
el opuesto \omterm{def:g5:solids:def}{caras} suma $7$. (Utilice \cref{ex:g5:solids:read} para
encuentre primero los pares opuestos.)
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:solids:6}
Cuente la unidad \omterm{def:g5:solids:def}{cubos}: a \omterm{def:g5:solids:def}{caja} $5$ de largo, $2$ de ancho, $3$ de alto; un \omterm{def:g5:solids:def}{cubo} de
lado $3$; una pila en forma de L hecha de una losa $3 \times 2 \times 1$
con una losa $1 \times 2 \times 1$ encima.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:solids:7}
Un \omterm{def:g5:solids:def}{caja} contiene exactamente la unidad $12$ \omterm{def:g5:solids:def}{cubos} en una capa. Da dos posibles
formas de capa ($L \times w$), y para cada una, el \omterm{def:g5:solids:volume}{volumen} del \omterm{def:g5:solids:def}{caja} si
tiene $3$ tales capas.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:solids:8}
Con la unidad $27$ \omterm{def:g5:solids:def}{cubos}, ¿qué \omterm{def:g5:solids:def}{cubo} puedes construir? ¿Con $64$? (Piensa en
la capa cuenta hacia atrás.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g5:solids:9}
Azúcar \omterm{def:g5:solids:def}{cubos} de lado $1$ cm vienen en un \omterm{def:g5:solids:def}{caja} $10$ cm de largo, $5$ cm de ancho,
$4$ cm de alto (medidas interiores). ¿Cuántos azúcar \omterm{def:g5:solids:def}{cubos} caben? si un
la familia usa $6$ \omterm{def:g5:solids:def}{cubos} al día, ¿cuántos días dura un \omterm{def:g5:solids:def}{caja}?
(\omterm{def:g4:numbers:round}{redondo} con sensatez)?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g5:solids:10}
Se pinta un $3 \times 3 \times 3$ \omterm{def:g5:solids:def}{cubo} de rojo por fuera y luego se corta
en la unidad $27$ \omterm{def:g5:solids:def}{cubos}. ¿Cuántos \omterm{def:g5:solids:def}{cubos} pequeños tienen pintura exactamente?
$3$\omterm{def:g5:solids:def}{caras}? ¿En $2$? ¿En $1$? ¿En ninguno? (Compruebe: las cuatro cuentas deben sumar
hasta $27$.)
\end{exercise}
