\chapter{Perímetro y Área}\label{ch:g5:areas}

Cuadrículas contadas de grado 4 (\cref{ch:g4:measure}); este capitulo
gana el primer \omterm{def:g4:measure:area}{área} \emph{fórmula} --- largo por ancho para el
rectángulo --- y aprende las unidades cm$^2$ y m$^2$. Las fórmulas para
triángulos y otras formas maduran en \cref{ch:g6:measure} y
\cref{ch:g7:areas}.

\section{Dos medidas diferentes}

\begin{example}[Frontera vs superficie]\label{ex:g5:areas:contrast}
Un jardinero necesita \emph{\omterm{def:g4:measure:perimeter}{perímetro}} para comprar la cerca y \emph{\omterm{def:g4:measure:area}{área}}
para comprar la semilla de pasto. Los dos no se suceden: estiramientos
un rectángulo más delgado y más largo puede mantener su \omterm{def:g4:measure:area}{área} mientras que su \omterm{def:g4:measure:perimeter}{perímetro}
crece --- compare un rectángulo $6 \times 2$ y $4 \times 3$
(mismo \omterm{def:g4:measure:area}{área} $12$, \omterm{def:g4:measure:perimeter}{perímetros} $16$ y $14$).
\end{example}

\section{La fórmula del rectángulo}

\begin{proposition}[Área de un rectángulo]\label{prop:g5:areas:rectangle}
Un rectángulo de largo $L$ y ancho $w$ (en la misma unidad) tiene \omterm{def:g4:measure:area}{área}
\[
A = L \times w ,
\]
en unidades cuadradas: cm$^2$ (cuadrados de lado $1$ cm), m$^2$ (cuadrados de
lado $1$ m), \dots
\end{proposition}

\begin{proof}[Por qué, contando]
Cubra el rectángulo con cuadrados unitarios: $w$ filas de $L$ cuadrados cada una,
entonces $L \times w$ cuadrados en total --- exactamente la multiplicación
rectángulo de \cref{ch:g3:mult}.
\end{proof}

\begin{omfigure}
\begin{tikzpicture}[scale=0.62]
  \draw[gray!50, thin] (0,0) grid (6,3);
  \draw[very thick, omDef] (0,0) rectangle (6,3);
  \fill[omThm!30] (0,2) rectangle (1,3);
  \draw[omThm, thick] (0,2) rectangle (1,3);
  \node[right, font=\small, omThm] at (1.15,2.5) {$1$ cm$^2$};
  \node[below, font=\small] at (3,-0.15) {$6$ cm};
  \node[left, font=\small] at (0,1.5) {$3$ cm};
  \node[font=\small] at (3.9,1.2) {$6 \times 3 = 18$ cm$^2$};
\end{tikzpicture}

{\small Tres filas de cuadrados de seis centímetros: la fórmula
$L \times w$ es el conteo, escrito de una vez por todas.}
\end{omfigure}

\begin{example}\label{ex:g5:areas:rectangle}
Una alfombra mide $2.5$ m por $2$ m: \omterm{def:g4:measure:area}{área} $2.5 \times 2 = 5$ m$^2$.
Un sello mide $3$ cm por $2.4$ cm: \omterm{def:g4:measure:area}{área} $3 \times 2.4 = 7.2$
cm$^2$. Misma fórmula, cualquier unidad --- siempre y cuando ambos lados usen la
\emph{mismo} uno.
\end{example}

\begin{example}[Conversión de unidades cuadradas por 100]\label{ex:g5:areas:units}
$1$ m $= 100$ cm, pero $1$ m$^2 = 100 \times 100 = 10\,000$ cm$^2$:
un metro cuadrado es una cuadrícula de centímetros cuadrados $100$ por $100$. \omterm{def:g4:measure:area}{Área}
las unidades saltan en centenas, no en decenas.
\end{example}

\section{Figuras compuestas}

\begin{method}[Recortar, sumar, reiniciar]\label{met:g5:areas:composite}
Para una figura hecha de rectángulos (una L, una T, un marco):
\begin{enumerate}
  \item córtalo en rectángulos o complétalo en un rectángulo grande;
  \item calcula cada \omterm{def:g4:measure:area}{área} rectangular con la fórmula;
  \item suma las piezas --- o resta el agujero del grande
        rectángulo;
  \item comparar con un recuento aproximado de cuadrados de la cuadrícula.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g5:areas:composite}
Una habitación en forma de L: un rectángulo de $6$ m $\times$ $4$ m con un
$2$ m $\times$ $2$ m falta esquina.
\[
\text{Area} = 6 \times 4 - 2 \times 2 = 24 - 4 = 20 \text{ m}^2 .
\]
O cortar la L en una tira $6 \times 2$ y una tira $4 \times 2$:
$12 + 8 = 20$ m$^2$ --- dos caminos, una respuesta.
\end{example}

\begin{example}[Medio rectángulo]\label{ex:g5:areas:halfrect}
Al cortar un rectángulo a lo largo de una diagonal se obtienen dos triángulos iguales.
\omterm{def:g4:measure:area}{área}: cada uno es \emph{\omterm{def:g3:division:half}{medio}} el rectángulo. Un triángulo rectángulo con catetos.
$6$ cm y $4$ cm por lo tanto tiene \omterm{def:g4:measure:area}{área} $\frac{6 \times 4}{2} = 12$
cm$^2$ --- una imagen que vale la pena recordar
\cref{ch:g6:measure}, donde se convierte en fórmula.
\end{example}

\section{Ceremonias}

\begin{exercise}[$\star$]\label{exo:g5:areas:1}
Calcular el \omterm{def:g4:measure:area}{área} y el \omterm{def:g4:measure:perimeter}{perímetro} de un rectángulo $8$ cm $\times$
$5$cm. ¿Qué respuesta está en cm y cuál en cm$^2$?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:areas:2}
Calcular el \omterm{def:g4:measure:area}{áreas}: un cuadrado de lado $9$ cm; un rectángulo
$12$ m $\times$ $7$ m; un rectángulo $4.5$ cm $\times$ $6$ cm.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:areas:3}
Un rectángulo tiene \omterm{def:g4:measure:area}{área} $63$ cm$^2$ y longitud $9$ cm. Encuentra su ancho,
entonces es \omterm{def:g4:measure:perimeter}{perímetro}.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:areas:4}
Dibuja dos rectángulos diferentes con cuadrados \omterm{def:g4:measure:area}{área} $24$ en papel cuadriculado,
y calcule ambos \omterm{def:g4:measure:perimeter}{perímetros}. Mismo \omterm{def:g4:measure:area}{área} --- ¿mismo \omterm{def:g4:measure:perimeter}{perímetro}?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:areas:5}
Convertir: $3$ m$^2$ en cm$^2$; \quad $50\,000$ cm$^2$ en m$^2$.
(Recuerde \cref{ex:g5:areas:units}: ¡por cientos!)
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:areas:6}
Una figura en forma de T está hecha de una barra horizontal $8 \times 2$ encima de
una barra vertical $2 \times 5$ (en cm). Calcule su \omterm{def:g4:measure:area}{área} sumando dos
rectángulos.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:areas:7}
Un marco de imagen: un rectángulo de $30$ cm $\times$ $20$ cm con un
$24$ cm $\times$ $14$ cm ventana rectangular recortada. ¿Qué \omterm{def:g4:measure:area}{área} de
¿Qué madera utiliza el marco?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:areas:8}
Calcular el \omterm{def:g4:measure:area}{área} de un triángulo rectángulo con catetos $8$ cm y $6$ cm.
(\omterm{def:g3:division:half}{medio} un rectángulo, \cref{ex:g5:areas:halfrect}).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:areas:9}
Un huerto rectangular mide $7$ m por $4$ m. Costos de semillas
$2$ por metro cuadrado y cercado $3$ por metro. Calcular el costo de
la semilla, luego el costo de la cerca.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g5:areas:10}
El suelo de un pasillo mide $12$ m de largo y $2$ m de ancho y debe cubrirse
con tejas cuadradas de lado $50$ cm. ¿Cuántas fichas se necesitan?
(Convierta primero o cuente las fichas en cada dirección).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g5:areas:11}
Duplica los lados de un rectángulo de $4$ cm $\times$ $3$ cm. que pasa
a su \omterm{def:g4:measure:perimeter}{perímetro}? ¿A su \omterm{def:g4:measure:area}{área}? (Calcule tanto antes como después ---
¡las dos respuestas difieren!)
\end{exercise}
