\chapter{Problemas y gráficos}\label{ch:g5:data}

El último capítulo del año pone todo a funcionar: multipaso
problemas con el dinero, cantidades y duraciones, y los gráficos que
presentar información de un vistazo. La idea estrella es ``por uno'' ---
encontrar el precio de un solo artículo --- que crecerá hasta convertirse en
proporcionalidad en \cref{ch:g6:propdata}.

\section{Problemas de varios pasos}

\begin{method}[Domesticando un problema largo]\label{met:g5:data:steps}
\begin{enumerate}
  \item Leer hasta el final; diga cuál es la pregunta final;
  \item planifica al revés: para responderlo, ¿qué necesito primero?
  \item resuelve los pasos en orden, escribiendo una operación y una
        frase corta por paso;
  \item compare la respuesta final con el sentido común y con una
        estimación.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g5:data:multistep}
``Una escuela pide $4$ cajas de libros $25$ a $6$ cada una y paga
$180$ para entrega en la parte superior. ¿Cuál es el costo total?
\begin{enumerate}
  \item Libros: $4 \times 25 = 100$ libros.
  \item Costo del libro: $100 \times 6 = 600$.
  \item Total: $600 + 180 = 780$.
\end{enumerate}
Verificación de estimación: alrededor de cien libros en $6$ son aproximadamente $600$, más
acerca de $200$: acerca de $800$ --- la respuesta $780$ es plausible.
\end{example}

\begin{example}[Por uno]\label{ex:g5:data:perone}
``$5$ tazas idénticas cuestan $12.50$; ¿Cuánto cuestan las tazas $8$?
Encuentre primero el precio de la taza \emph{uno}:
\[
12.50 \div 5 = 2.50,
\qquad\text{then}\qquad
8 \times 2.50 = 20 .
\]
A través de uno, todo se vuelve accesible: este patrón de dos pasos
Resuelve toda una familia de problemas (recetas, velocidades, precios) y es
la semilla de \cref{ch:g6:propdata}.
\end{example}

\begin{example}[Cambio y presupuesto]\label{ex:g5:data:budget}
Zoe tiene $30$. Compra un libro en $12.40$ y dos bolígrafos en $2.30$.
cada uno. ¿También puede permitirse un juego de mesa $13$?
\begin{enumerate}
  \item Bolígrafos: $2 \times 2.30 = 4.60$.
  \item Pasado hasta el momento: $12.40 + 4.60 = 17$.
  \item Restante: $30 - 17 = 13$ --- exactamente lo suficiente para el juego,
        sin nada más.
\end{enumerate}
\end{example}

\section{Tablas de lectura}

\begin{method}[Leyendo cualquier gráfico]\label{met:g5:data:read}
\begin{enumerate}
  \item Leer el título: ¿qué se cuenta?
  \item Leer los ejes o la clave: ¿qué significa una graduación, una barra,
        ¿Qué representa un símbolo?
  \item Sólo entonces responda las preguntas --- señalando el gráfico con un
        La regla ayuda a leer las alturas exactamente.
\end{enumerate}
\end{method}

\section{Ceremonias}

\begin{exercise}[$\star$]\label{exo:g5:data:5}

\begin{exercise}[$\star$]\label{exo:g5:data:7}

\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:data:8}

$0.50$. Calcule el precio "por unidad" del paquete y cuánto
Ahorra por yogur comprando el pack.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g5:data:9}
``Un cine vendió entradas $148$ a $9.50$ el sábado y entradas $95$
al mismo precio el domingo. ¿Cuánto dinero más trajo el sábado?
que el domingo?" Resuélvelo de dos maneras: calculando las ganancias de ambos días, o
computando con el \emph{\omterm{ex:g1:subtraction:difference}{diferencia}} en tickets --- y verifica que
las respuestas concuerdan.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g5:data:10}
Una receta para personas $4$ necesita $300$ g de arroz. Aline cocina para
$10$ personas. ¿Cuánto arroz necesita? (Por una persona primero ---
Se permiten decimales.)
\end{exercise}
