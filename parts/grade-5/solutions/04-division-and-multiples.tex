\section*{Chapter \ref{ch:g5:division} --- Division and Multiples}

\begin{solution}{exo:g5:division:1}
$851 \div 23$: multiples of $23$: $23, 46, 69, 92, 115, 138, 161,
184, 207$. Then $85 \div 23$: $3$ times ($69$), remainder $16$;
bring down the $1$: $161 \div 23 = 7$ exactly. Quotient $37$,
remainder $0$ ($23 \times 37 = 851$).

$704 \div 32$: $70 \div 32$: $2$ times ($64$), remainder $6$; bring
down the $4$: $64 \div 32 = 2$. Quotient $22$, remainder $0$.
\end{solution}

\begin{solution}{exo:g5:division:2}
$9 \div 2 = 4.5$; \quad $27 \div 4 = 6.75$; \quad
$33 \div 5 = 6.6$; \quad $21 \div 8 = 2.625$.
\end{solution}

\begin{solution}{exo:g5:division:3}
$54 \div 4 = 13.5$: each pays $13.50$.
\end{solution}

\begin{solution}{exo:g5:division:4}
$10 \div 3 = 3.333\dots$: the digit $3$ repeats forever --- the
division never stops. Rounded to the hundredth: $3.33$.
\end{solution}

\begin{solution}{exo:g5:division:5}
Multiples of $7$ up to $70$: $7, 14, 21, 28, 35, 42, 49, 56, 63,
70$. Divisors of $18$: $1, 2, 3, 6, 9, 18$. Divisors of $24$:
$1, 2, 3, 4, 6, 8, 12, 24$.
\end{solution}

\begin{solution}{exo:g5:division:6}
$470$: by $2$ (ends in $0$), by $5$, by $10$; digit sum $11$: not by
$3$.

$735$: ends in $5$: by $5$, not by $2$ nor $10$; digit sum $15$: by
$3$.

$8\,001$: odd; digit sum $9$: by $3$ only.

$1\,314$: even; digit sum $9$: by $2$ and by $3$ (hence by $6$),
not by $5$.
\end{solution}

\begin{solution}{exo:g5:division:7}
Divisible by $3$ and $5$ means divisible by $15$: between $60$ and
$80$, the numbers $60$, $75$.
\end{solution}

\begin{solution}{exo:g5:division:8}
$7.2 \div 6 = 1.2$: each piece measures $1.2$ m (check:
$1.2 \times 6 = 7.2$).
\end{solution}

\begin{solution}{exo:g5:division:9}
$1\,000 = 24 \times 41 + 16$: forty-one full bags, $16$ marbles
left.
\end{solution}

\begin{solution}{exo:g5:division:10}
$7 \div 5 = 1.4$: each of the $5$ friends gets $1.4$ bars ---
possible by cutting bars into tenths. For $5 \div 7$: the division
gives $0.714285\dots$, which never stops --- no exact decimal share
exists; only the fraction $\frac57$ names it exactly.
\end{solution}

\begin{solution}{exo:g5:division:11}
Digit sum: $5 + 2 + d = 7 + d$. Divisible by $3$ when $7 + d$ is
$9$, $12$ or $15$: $d = 2$, $5$ or $8$. The smallest is $d = 2$
(giving $522 = 3 \times 174$).
\end{solution}
