\begin{solution}{exo:g5:decimalops:1}
$45.7 + 8.93 = 54.63$; \quad
$16.25 + 3.75 = 20$; \quad
$0.86 + 12.4 = 13.26$.
\end{solution}

\begin{solution}{exo:g5:decimalops:2}
$9.4 - 2.72 = 6.68$ (comprobar: $6.68 + 2.72 = 9.4$).

$20 - 13.45 = 6.55$ (comprobar: $6.55 + 13.45 = 20$).

$6.03 - 5.9 = 0.13$.
\end{solution}

\end{solution}

\begin{solution}{exo:g5:decimalops:5}

\begin{solution}{exo:g5:decimalops:6}
$6.2 \times 4$: estimación $24$; exactamente $24.8$.

$3.45 \times 8$: estimación $28$ ($3.5 \times 8$); exactamente $27.6$.

$12.5 \times 6$: estimación $75$; exactamente $75$.
\end{solution}

\begin{solution}{exo:g5:decimalops:7}
$7$ vueltas: $0.4 \times 7 = 2.8$ km. $25$ vueltas:
$0.4 \times 25 = 10$ kilómetros.
\end{solution}

\begin{solution}{exo:g5:decimalops:8}
Baguettes: $1.15 \times 2 = 2.30$. Total: $2.30 + 12.60 = 14.90$.
Cambio: $20 - 14.90 = 5.10$.
\end{solution}

\begin{solution}{exo:g5:decimalops:9}
Pack: $1.5 \times 6 = 9$ L. Días: $0.75 + 0.75 = 1.5$ L cada dos
días, por lo que $9$ L últimos $12$ días (doce porciones de $0.75$:
$12 \times 0.75 = 9$).
\end{solution}

\begin{solution}{exo:g5:decimalops:10}
Estimación: $5.6$ es mayor que $5.5$ y $5.5 \times 3 = 16.5$ --- entonces
el \omterm{def:g2:mult:def}{producto} debe ser \emph{más} que $16.5$, y $15.18$ es demasiado
pequeño. Correcto: $56 \times 3 = 168$, un dígito decimal, entonces
$5.6 \times 3 = 16.8$. Milo pegó sus dos parciales \omterm{def:g2:mult:def}{productos} ($15$ y
$18$) uno al lado del otro en lugar de agregarlos con sus lugares:
$15 + 1.8 = 16.8$.
\end{solution}
