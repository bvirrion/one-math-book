\section*{Capítulo \ref{ch:g5:solids} --- Cubes and Boxes}

\begin{solution}{exo:g5:solids:1}
Por ejemplo: un dado (\omterm{def:g5:solids:def}{cubo}); un cereal \omterm{def:g5:solids:def}{caja} (\omterm{def:g5:solids:def}{caja}); una lata (\omterm{def:g5:solids:def}{cilindro}); un
fútbol (pelota); ciertos tejados o los monumentos de Egipto
(\omterm{def:g5:solids:def}{pirámide}).
\end{solution}

\begin{solution}{exo:g5:solids:2}
Un \omterm{def:g5:solids:def}{caja} tiene $6$ \omterm{def:g5:solids:def}{caras}, $12$ \omterm{def:g5:solids:def}{bordes}, $8$ \omterm{def:g5:solids:def}{vértices} --- y
$6 + 8 = 12 + 2$. \checkmark
\end{solution}

\begin{solution}{exo:g5:solids:3}
Los seis \omterm{def:g5:solids:def}{caras} vienen en tres pares de \omterm{def:g5:solids:def}{caras} opuestos idénticos.
(arriba--abajo, adelante--atrás, izquierda-derecha). En un \omterm{def:g5:solids:def}{cubo}, los seis \omterm{def:g5:solids:def}{caras}
son cuadrados idénticos.
\end{solution}

\begin{solution}{exo:g5:solids:4}
En forma de cruz \omterm{def:g5:solids:net}{neto}: cuatro cuadrados de $2$ cm seguidos, uno arriba y otro
debajo del segundo cuadrado. Doblar el anillo de cuatro forma los lados;
los dos cuadrados adicionales cierran la parte superior e inferior.
\end{solution}

\begin{solution}{exo:g5:solids:5}
En la cruz \omterm{def:g5:solids:net}{neto} (cuatro cuadrados seguidos, uno arriba, uno debajo de la
segundo), los pares opuestos son: los cuadrados $1$ y $3$ de la fila;
cuadrados $2$ y $4$ de la fila; los cuadrados superior e inferior. uno
etiquetado correcto: fila $= 1, 2, 6, 5$; arriba $= 3$; abajo $= 4$
--- entonces $1{+}6$, $2{+}5$, $3{+}4$ hacen $7$.
\end{solution}

\begin{solution}{exo:g5:solids:6}
\omterm{def:g5:solids:def}{Caja}: $5 \times 2 = 10$ por capa, $3$ capas: $30$ \omterm{def:g5:solids:def}{cubos}. \omterm{def:g5:solids:def}{Cubo}:
$3 \times 3 \times 3 = 27$. Pila L: $3 \times 2 \times 1 = 6$ plus
$1 \times 2 \times 1 = 2$: $8$ \omterm{def:g5:solids:def}{cubos}.
\end{solution}

\begin{solution}{exo:g5:solids:7}
Capas de $12$: por ejemplo $4 \times 3$ o $6 \times 2$ (o
$12 \times 1$). Con capas $3$, el \omterm{def:g5:solids:volume}{volumen} es
$12 \times 3 = 36$ \omterm{def:g5:solids:def}{cubos} en todos los casos.
\end{solution}

\begin{solution}{exo:g5:solids:8}
$27 = 3 \times 3 \times 3$: un \omterm{def:g5:solids:def}{cubo} del lado $3$.
$64 = 4 \times 4 \times 4$: un \omterm{def:g5:solids:def}{cubo} del lado $4$.
\end{solution}

\begin{solution}{exo:g5:solids:9}
El \omterm{def:g5:solids:def}{caja} contiene azúcar $10 \times 5 \times 4 = 200$ \omterm{def:g5:solids:def}{cubos}. En $6$ por
día: $200 = 6 \times 33 + 2$, por lo que \omterm{def:g5:solids:def}{caja} dura $33$ días completos (y
$2$ \omterm{def:g5:solids:def}{cubos} quedan para la mañana del día $34$).
\end{solution}

\begin{solution}{exo:g5:solids:10}
Pintar en $3$ \omterm{def:g5:solids:def}{caras}: la esquina $8$ \omterm{def:g5:solids:def}{cubos}. En $2$ \omterm{def:g5:solids:def}{caras}: los medios
del $12$ \omterm{def:g5:solids:def}{bordes}: $12$ \omterm{def:g5:solids:def}{cubos}. En $1$ \omterm{def:g5:solids:def}{rostro}: los centros del $6$
\omterm{def:g5:solids:def}{caras}: $6$ \omterm{def:g5:solids:def}{cubos}. En ninguno: el único centro oculto \omterm{def:g5:solids:def}{cubo}: $1$.
Verificar: $8 + 12 + 6 + 1 = 27$. \checkmark
\end{solution}
