\section*{Capítulo \ref{ch:g5:data} --- Problems and Charts}

\begin{solution}{exo:g5:data:1}
Entradas: $32 \times 7 = 224$. Total: $240 + 224 = 464$.
\end{solution}

\begin{solution}{exo:g5:data:2}
Por un kg: $7.50 \div 3 = 2.50$. Cinco kilos:
$5 \times 2.50 = 12.50$.
\end{solution}

\begin{solution}{exo:g5:data:3}
$87 \div 6 = 14.5$: cada uno paga $14.50$.
\end{solution}

\begin{solution}{exo:g5:data:4}
Gastado: $13.90 + 8.50 = 22.40$. Izquierda: $25 - 22.40 = 2.60$.
\end{solution}

\begin{solution}{exo:g5:data:5}
Más de $50$ mm: enero ($60$), marzo ($55$), abril ($70$), mayo
($80$). Mayo menos junio: $80 - 30 = 50$ mm.
\end{solution}

\begin{solution}{exo:g5:data:6}
Semana $2$: $6$ cm. Crecimiento de $4$ cm: entre las semanas $2$ y $3$
($6 \to 10$), y entre las semanas $4$ y $5$ ($15 \to 19$).
\end{solution}

\begin{solution}{exo:g5:data:7}
Cuatro barras de alturas $4$, $7$, $3$, $6$ (graduación cada $1$ o
$2$ goles). Total: $4 + 7 + 3 + 6 = 20$ goles.
\end{solution}

\begin{solution}{exo:g5:data:8}
Paquete por uno: $3.20 \div 8 = 0.40$. Ahorro por yogur:
$0.50 - 0.40 = 0.10$ (diez céntimos).
\end{solution}

\begin{solution}{exo:g5:data:9}
Camino 1: Sábado $148 \times 9.50 = 1\,406$; domingo
$95 \times 9.50 = 902.50$; \omterm{ex:g1:subtraction:difference}{diferencia}$503.50$.

Camino 2: $148 - 95 = 53$ boletos más, cada uno con un valor de $9.50$:
$53 \times 9.50 = 503.50$. Misma respuesta --- el \omterm{ex:g1:subtraction:difference}{diferencia} del
la toma es la toma del \omterm{ex:g1:subtraction:difference}{diferencia}.
\end{solution}

\begin{solution}{exo:g5:data:10}
Por persona: $300 \div 4 = 75$ g. Por diez:
$10 \times 75 = 750$ g de arroz.
\end{solution}

\begin{solution}{exo:g5:data:11}
Muchos problemas correctos --- por ejemplo: ``Milo compra jugos $3$ en
$1.20$ cada uno y un sándwich en $3.40$. Paga con un $12.50$
vale. ¿Cuánto queda del bono?'' Solución: jugos
$3 \times 1.20 = 3.60$; total gastado $3.60 + 3.40 = 7$; izquierda
$12.50 - 7 = 5.50$.
\end{solution}
