\section*{Capítulo \ref{ch:g5:division} --- Division and Multiples}

\begin{solution}{exo:g5:division:1}
$851 \div 23$: \omterm{def:g5:division:multiple}{múltiplos} de $23$: $23, 46, 69, 92, 115, 138, 161,
184, 207$. Luego $85 \div 23$: $3$ veces ($69$), \omterm{def:g3:division:remainder}{resto} $16$;
baje el $1$: $161 \div 23 = 7$ exactamente. \omterm{def:g3:division:remainder}{Cociente}$37$,
\omterm{def:g3:division:remainder}{resto} $0$ ($23 \times 37 = 851$).

$704 \div 32$: $70 \div 32$: $2$ veces ($64$), \omterm{def:g3:division:remainder}{resto} $6$; traer
abajo el $4$: $64 \div 32 = 2$. \omterm{def:g3:division:remainder}{Cociente} $22$, \omterm{def:g3:division:remainder}{resto} $0$.
\end{solution}

\begin{solution}{exo:g5:division:2}
$9 \div 2 = 4.5$; \quad $27 \div 4 = 6.75$; \quad
$33 \div 5 = 6.6$; \quad $21 \div 8 = 2.625$.
\end{solution}

\begin{solution}{exo:g5:division:3}
$54 \div 4 = 13.5$: cada uno paga $13.50$.
\end{solution}

\begin{solution}{exo:g5:division:5}
\omterm{def:g5:division:multiple}{Múltiplos} of $7$ up to $70$: $7, 14, 21, 28, 35, 42, 49, 56, 63,

\end{solution}

\begin{solution}{exo:g5:division:6}
$470$: por $2$ (termina en $0$), por $5$, por $10$; dígito \omterm{def:g1:addition:def}{suma} $11$: no por
$3$.

$735$: termina en $5$: por $5$, no por $2$ ni $10$; dígito \omterm{def:g1:addition:def}{suma} $15$: por
$3$.

$8\,001$: \omterm{def:g3:numbers:evenodd}{extraño}; dígito \omterm{def:g1:addition:def}{suma} $9$: por $3$ únicamente.

$1\,314$: \omterm{def:g3:numbers:evenodd}{incluso}; dígito \omterm{def:g1:addition:def}{suma} $9$: por $2$ y por $3$ (por lo tanto, por $6$),
no por $5$.
\end{solution}

\begin{solution}{exo:g5:division:7}
Divisible por $3$ y $5$ significa divisible por $15$: entre $60$ y
$80$, los números $60$, $75$.
\end{solution}

\begin{solution}{exo:g5:division:8}
$7.2 \div 6 = 1.2$: cada pieza mide $1.2$ m (consultar:
$1.2 \times 6 = 7.2$).
\end{solution}

\begin{solution}{exo:g5:division:9}
$1\,000 = 24 \times 41 + 16$: cuarenta y una bolsas llenas, canicas $16$
izquierda.
\end{solution}

\begin{solution}{exo:g5:division:10}
$7 \div 5 = 1.4$: cada uno de los amigos $5$ recibe barras $1.4$ ---
posible cortando barras en décimas. Para $5 \div 7$: la división
da $0.714285\dots$, que nunca se detiene --- sin porcentaje decimal exacto
existe; sólo el \omterm{def:g4:fractions:def}{fracción} $\frac57$ lo nombra exactamente.
\end{solution}

\begin{solution}{exo:g5:division:11}
Dígito \omterm{def:g1:addition:def}{suma}: $5 + 2 + d = 7 + d$. Divisible por $3$ cuando $7 + d$ es
$9$, $12$ o $15$: $d = 2$, $5$ o $8$. El más pequeño es $d = 2$
(dando $522 = 3 \times 174$).
\end{solution}
