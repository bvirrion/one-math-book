\section*{Chapter \ref{ch:g5:solids} --- Cubes and Boxes}

\begin{solution}{exo:g5:solids:1}
For instance: a die (\omterm{def:g5:solids:def}{cube}); a cereal \omterm{def:g5:solids:def}{box} (\omterm{def:g5:solids:def}{box}); a can (\omterm{def:g5:solids:def}{cylinder}); a
football (ball); certain roofs or the monuments of Egypt
(\omterm{def:g5:solids:def}{pyramid}).
\end{solution}

\begin{solution}{exo:g5:solids:2}
A \omterm{def:g5:solids:def}{box} has $6$ \omterm{def:g5:solids:def}{faces}, $12$ \omterm{def:g5:solids:def}{edges}, $8$ \omterm{def:g5:solids:def}{vertices} --- and
$6 + 8 = 12 + 2$. \checkmark
\end{solution}

\begin{solution}{exo:g5:solids:3}
The six \omterm{def:g5:solids:def}{faces} come in three pairs of identical opposite \omterm{def:g5:solids:def}{faces}
(top--bottom, front--back, left--right). On a \omterm{def:g5:solids:def}{cube}, all six \omterm{def:g5:solids:def}{faces}
are identical squares.
\end{solution}

\begin{solution}{exo:g5:solids:4}
Cross-shaped \omterm{def:g5:solids:net}{net}: four $2$ cm squares in a row, one above and one
below the second square. Folding the ring of four makes the sides;
the two extra squares close the top and bottom.
\end{solution}

\begin{solution}{exo:g5:solids:5}
On the cross \omterm{def:g5:solids:net}{net} (four squares in a row, one above, one below the
second), the opposite pairs are: squares $1$ and $3$ of the row;
squares $2$ and $4$ of the row; the top and bottom squares. One
correct labelling: row $= 1, 2, 6, 5$; top $= 3$; bottom $= 4$
--- then $1{+}6$, $2{+}5$, $3{+}4$ all make $7$.
\end{solution}

\begin{solution}{exo:g5:solids:6}
\omterm{def:g5:solids:def}{Box}: $5 \times 2 = 10$ per layer, $3$ layers: $30$ \omterm{def:g5:solids:def}{cubes}. \omterm{def:g5:solids:def}{Cube}:
$3 \times 3 \times 3 = 27$. L-stack: $3 \times 2 \times 1 = 6$ plus
$1 \times 2 \times 1 = 2$: $8$ \omterm{def:g5:solids:def}{cubes}.
\end{solution}

\begin{solution}{exo:g5:solids:7}
Layers of $12$: for instance $4 \times 3$ or $6 \times 2$ (or
$12 \times 1$). With $3$ layers, the \omterm{def:g5:solids:volume}{volume} is
$12 \times 3 = 36$ \omterm{def:g5:solids:def}{cubes} in every case.
\end{solution}

\begin{solution}{exo:g5:solids:8}
$27 = 3 \times 3 \times 3$: a \omterm{def:g5:solids:def}{cube} of side $3$.
$64 = 4 \times 4 \times 4$: a \omterm{def:g5:solids:def}{cube} of side $4$.
\end{solution}

\begin{solution}{exo:g5:solids:9}
The \omterm{def:g5:solids:def}{box} holds $10 \times 5 \times 4 = 200$ sugar \omterm{def:g5:solids:def}{cubes}. At $6$ per
day: $200 = 6 \times 33 + 2$, so the \omterm{def:g5:solids:def}{box} lasts $33$ full days (and
$2$ \omterm{def:g5:solids:def}{cubes} remain for the morning of day $34$).
\end{solution}

\begin{solution}{exo:g5:solids:10}
Paint on $3$ \omterm{def:g5:solids:def}{faces}: the $8$ corner \omterm{def:g5:solids:def}{cubes}. On $2$ \omterm{def:g5:solids:def}{faces}: the middles
of the $12$ \omterm{def:g5:solids:def}{edges}: $12$ \omterm{def:g5:solids:def}{cubes}. On $1$ \omterm{def:g5:solids:def}{face}: the centers of the $6$
\omterm{def:g5:solids:def}{faces}: $6$ \omterm{def:g5:solids:def}{cubes}. On none: the single hidden center \omterm{def:g5:solids:def}{cube}: $1$.
Check: $8 + 12 + 6 + 1 = 27$. \checkmark
\end{solution}
