\section*{Chapter \ref{ch:g5:areas} --- Perimeter and Area}

\begin{solution}{exo:g5:areas:1}
\omterm{def:g4:measure:area}{Area}: $8 \times 5 = 40$ cm$^2$. \omterm{def:g4:measure:perimeter}{Perimeter}:
$2 \times (8 + 5) = 26$ cm.
\end{solution}

\begin{solution}{exo:g5:areas:2}
$9 \times 9 = 81$ cm$^2$; \quad $12 \times 7 = 84$ m$^2$; \quad
$4.5 \times 6 = 27$ cm$^2$.
\end{solution}

\begin{solution}{exo:g5:areas:3}
Width: $63 \div 9 = 7$ cm. \omterm{def:g4:measure:perimeter}{Perimeter}: $2 \times (9 + 7) = 32$ cm.
\end{solution}

\begin{solution}{exo:g5:areas:4}
For instance $6 \times 4$ (\omterm{def:g4:measure:perimeter}{perimeter} $20$) and $8 \times 3$
(\omterm{def:g4:measure:perimeter}{perimeter} $22$): same \omterm{def:g4:measure:area}{area} $24$, different \omterm{def:g4:measure:perimeter}{perimeters}.
\end{solution}

\begin{solution}{exo:g5:areas:5}
$3$ m$^2 = 30\,000$ cm$^2$; \quad $50\,000$ cm$^2 = 5$ m$^2$.
\end{solution}

\begin{solution}{exo:g5:areas:6}
Bar: $8 \times 2 = 16$ cm$^2$; stem: $2 \times 5 = 10$ cm$^2$;
total $26$ cm$^2$.
\end{solution}

\begin{solution}{exo:g5:areas:7}
$30 \times 20 - 24 \times 14 = 600 - 336 = 264$ cm$^2$ of wood.
\end{solution}

\begin{solution}{exo:g5:areas:8}
\omterm{def:g3:division:half}{Half} the $8 \times 6$ rectangle: $48 \div 2 = 24$ cm$^2$.
\end{solution}

\begin{solution}{exo:g5:areas:9}
Seed: \omterm{def:g4:measure:area}{area} $7 \times 4 = 28$ m$^2$, cost $28 \times 2 = 56$.
Fence: \omterm{def:g4:measure:perimeter}{perimeter} $2 \times (7 + 4) = 22$ m, cost
$22 \times 3 = 66$.
\end{solution}

\begin{solution}{exo:g5:areas:10}
Two tiles of $50$ cm make a meter: along the $12$ m length,
$24$ tiles; along the $2$ m width, $4$ tiles. Total:
$24 \times 4 = 96$ tiles.
\end{solution}

\begin{solution}{exo:g5:areas:11}
Before: \omterm{def:g4:measure:perimeter}{perimeter} $2 \times (4 + 3) = 14$ cm, \omterm{def:g4:measure:area}{area} $12$ cm$^2$.
After (sides $8$ and $6$): \omterm{def:g4:measure:perimeter}{perimeter} $28$ cm, \omterm{def:g4:measure:area}{area} $48$ cm$^2$.
The \omterm{def:g4:measure:perimeter}{perimeter} \emph{doubled}; the \omterm{def:g4:measure:area}{area} was multiplied by
\emph{four} ($2 \times 2$): lengths and \omterm{def:g4:measure:area}{areas} do not scale the same
way.
\end{solution}
