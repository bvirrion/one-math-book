\section*{Chapitre \ref{ch:g5:geometry} --- Constructions géométriques}

\begin{solution}{exo:g5:geometry:1}
Les deux constructions suivent \cref{met:g5:geometry:basic}~; la
\omterm{def:g4:geometry:def}{perpendiculaire} et la \omterm{def:g4:geometry:def}{parallèle} par $A$ sont elles-mêmes
\omterm{def:g4:geometry:def}{perpendiculaires} l'une à l'autre.
\end{solution}

\begin{solution}{exo:g5:geometry:2}
Chaîne de \omterm{def:g4:geometry:def}{perpendiculaires} comme dans \cref{ex:g5:geometry:chain}.
Si les trois angles construits sont exactement droits et les
longueurs exactes, le quatrième angle n'a d'autre choix que d'être
droit aussi --- un rectangle bien construit se vérifie tout seul.
\end{solution}

\begin{solution}{exo:g5:geometry:3}
Base $[AB]$ de $7$~cm~; arcs de \omterm{def:g3:shapes:shapes}{rayons} $6$~cm (depuis $A$) et
$3$~cm (depuis $B$)~; le croisement est $C$.
\end{solution}

\begin{solution}{exo:g5:geometry:4}
Isocèle~: base $5$~cm, les deux arcs avec la même ouverture de
$7$~cm. Équilatéral~: base $6$~cm, les deux arcs avec l'ouverture
de $6$~cm.
\end{solution}

\begin{solution}{exo:g5:geometry:5}
L'équerre garantit les \omterm{def:g3:shapes:rightangle}{angles droits}~; le compas (ou la règle)
reporte le côté de $5$~cm sur les quatre côtés.
\end{solution}

\begin{solution}{exo:g5:geometry:6}
Tracer un côté $[AB]$ de $4$~cm, choisir une direction pour $[AD]$
(pas \omterm{def:g4:geometry:def}{perpendiculaire}, pour éviter le carré), $AD = 4$~cm~; puis des
arcs de $4$~cm depuis $B$ et depuis $D$ se croisent en $C$. Le
compas confirme les quatre côtés égaux.
\end{solution}

\begin{solution}{exo:g5:geometry:7}
Le cercle de centre $A$, \omterm{def:g3:shapes:shapes}{rayon} $4$~cm, et le cercle de centre $B$,
\omterm{def:g3:shapes:shapes}{rayon} $3$~cm, se croisent en \emph{deux} points (un au-dessus, un
au-dessous de la droite $(AB)$), car $4 + 3 > 5$ et les cercles se
chevauchent.
\end{solution}

\begin{solution}{exo:g5:geometry:8}
Un programme correct~: «~1. Tracer un rectangle $ABCD$ avec
$AB = 6$~cm (bas) et $BC = 2$~cm. 2. Tracer deux arcs de \omterm{def:g3:shapes:shapes}{rayon}
$4$~cm centrés en $D$ et en $C$ (les coins du haut)~; nommer $S$
leur croisement au-dessus du rectangle. 3. Tracer $[DS]$ et
$[CS]$.~» (La pointe du crayon est le triangle isocèle $DSC$.)
\end{solution}

\begin{solution}{exo:g5:geometry:9}
Les arcs de \omterm{def:g3:shapes:shapes}{rayons} $3$~cm et $4$~cm, tracés depuis les deux
extrémités de la base de $8$~cm, ne se rencontrent jamais~: même
bout à bout, $3 + 4 = 7$~cm ne peut pas couvrir $8$~cm. Deux côtés
ensemble doivent toujours être plus longs que le troisième ---
sinon le triangle ne peut pas se fermer.
\end{solution}

\begin{solution}{exo:g5:geometry:10}
Tracer un segment de $6$~cm et son milieu $O$~; tracer la
\omterm{def:g4:geometry:def}{perpendiculaire} en $O$~; marquer dessus deux points à $2$~cm de $O$
(un de chaque côté)~; joindre les quatre extrémités dans l'ordre. Le
compas confirme les quatre côtés égaux.
\end{solution}

\begin{solution}{exo:g5:geometry:11}
Sur le plan, la chèvre atteint le disque de centre $A$ et de \omterm{def:g3:shapes:shapes}{rayon}
$4$~cm, coupé par la ligne de clôture à $3$~cm de $A$~: sa région
est le disque moins la tranche au-delà de la clôture. Le long de la
clôture, elle peut brouter sur un segment (où le cercle croise la
ligne de clôture)~; au-delà, rien.
\end{solution}
