\section*{Chapitre \ref{ch:g5:decimals} --- Les nombres décimaux}

\begin{solution}{exo:g5:decimals:1}
$5.207$~; \quad $0.85$~; \quad $4.507$~; \quad $12.009$.
\end{solution}

\begin{solution}{exo:g5:decimals:2}
Le $4$ compte les dixièmes, le $8$ les millièmes~:
\[
27.418 = 27 + \frac{4}{10} + \frac{1}{100} + \frac{8}{1000}.
\]
\end{solution}

\begin{solution}{exo:g5:decimals:3}
$3.60 = 3.6$~: vrai (un zéro final n'ajoute rien).

$0.5 = 0.05$~: faux --- $5$ dixièmes contre $5$ centièmes.

$2.070 = 2.07$~: vrai.

$1.3 = 1.03$~: faux --- $3$ dixièmes contre $3$ centièmes.
\end{solution}

\begin{solution}{exo:g5:decimals:4}
$4.8 > 4.79$~; \quad $0.302 < 0.32$ (dixièmes égaux, centièmes
$0 < 2$)~; \quad $6.5 = 6.500$~; \quad $9.09 < 9.1$.
\end{solution}

\begin{solution}{exo:g5:decimals:5}
$3.041 < 3.104 < 3.14 < 3.4 < 3.41$.
\end{solution}

\begin{solution}{exo:g5:decimals:6}
$6.83$~: entre $6$ et $7$~; entre $6.8$ et $6.9$.

$0.492$~: entre $0$ et $1$~; entre $0.4$ et $0.5$.

$12.06$~: entre $12$ et $13$~; entre $12.0$ et $12.1$.
\end{solution}

\begin{solution}{exo:g5:decimals:7}
$8.276$~: à l'unité $8$~; au dixième $8.3$~; au centième $8.28$.
Et $3.95$ au dixième~: le chiffre des centièmes est $5$, on
arrondit vers le haut~: $4.0$.
\end{solution}

\begin{solution}{exo:g5:decimals:8}
Par exemple $2.65$~; \quad $0.415$~; \quad $5.995$. (Infiniment de
réponses à chaque fois.)
\end{solution}

\begin{solution}{exo:g5:decimals:9}
$0.85 < 0.9 < 1.02 < 1.2$ (en mètres).
\end{solution}

\begin{solution}{exo:g5:decimals:10}
Comparer d'abord les dixièmes~: $7.12$ a $1$ dixième, $7.9$ a $9$
dixièmes, donc $7.12 < 7.9$. Lucie a comparé les parties décimales
comme des entiers ($12$ contre $9$), oubliant que $12$ centièmes
est bien moins que $90$ centièmes.
\end{solution}

\begin{solution}{exo:g5:decimals:11}
L'arrondi au dixième donne $6.4$ pour les nombres à deux décimales
de $6.35$ (mi-chemin arrondi vers le haut) à $6.44$~: les nombres
$6.35$, $6.36$, \dots, $6.44$. Le plus petit~: $6.35$~; le plus
grand~: $6.44$.
\end{solution}
