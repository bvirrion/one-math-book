\section*{Chapitre \ref{ch:g5:areas} --- Périmètre et aire}

\begin{solution}{exo:g5:areas:1}
\omterm{def:g4:measure:area}{Aire}~: $8 \times 5 = 40$~cm$^2$. \omterm{def:g4:measure:perimeter}{Périmètre}~:
$2 \times (8 + 5) = 26$~cm.
\end{solution}

\begin{solution}{exo:g5:areas:2}
$9 \times 9 = 81$~cm$^2$~; \quad $12 \times 7 = 84$~m$^2$~; \quad
$4.5 \times 6 = 27$~cm$^2$.
\end{solution}

\begin{solution}{exo:g5:areas:3}
Largeur~: $63 \div 9 = 7$~cm. \omterm{def:g4:measure:perimeter}{Périmètre}~: $2 \times (9 + 7) = 32$~cm.
\end{solution}

\begin{solution}{exo:g5:areas:4}
Par exemple $6 \times 4$ (\omterm{def:g4:measure:perimeter}{périmètre} $20$) et $8 \times 3$
(\omterm{def:g4:measure:perimeter}{périmètre} $22$)~: même \omterm{def:g4:measure:area}{aire} $24$, \omterm{def:g4:measure:perimeter}{périmètres} différents.
\end{solution}

\begin{solution}{exo:g5:areas:5}
$3$~m$^2 = 30\,000$~cm$^2$~; \quad $50\,000$~cm$^2 = 5$~m$^2$.
\end{solution}

\begin{solution}{exo:g5:areas:6}
Barre~: $8 \times 2 = 16$~cm$^2$~; tige~: $2 \times 5 = 10$~cm$^2$~;
total $26$~cm$^2$.
\end{solution}

\begin{solution}{exo:g5:areas:7}
$30 \times 20 - 24 \times 14 = 600 - 336 = 264$~cm$^2$ de bois.
\end{solution}

\begin{solution}{exo:g5:areas:8}
\omterm{def:g3:division:half}{Moitié} du rectangle $8 \times 6$~: $48 \div 2 = 24$~cm$^2$.
\end{solution}

\begin{solution}{exo:g5:areas:9}
Graines~: \omterm{def:g4:measure:area}{aire} $7 \times 4 = 28$~m$^2$, coût $28 \times 2 = 56$.
Clôture~: \omterm{def:g4:measure:perimeter}{périmètre} $2 \times (7 + 4) = 22$~m, coût
$22 \times 3 = 66$.
\end{solution}

\begin{solution}{exo:g5:areas:10}
Deux carreaux de $50$~cm font un mètre~: le long des $12$~m,
$24$ carreaux~; le long des $2$~m, $4$ carreaux. Total~:
$24 \times 4 = 96$ carreaux.
\end{solution}

\begin{solution}{exo:g5:areas:11}
Avant~: \omterm{def:g4:measure:perimeter}{périmètre} $2 \times (4 + 3) = 14$~cm, \omterm{def:g4:measure:area}{aire} $12$~cm$^2$.
Après (côtés $8$ et $6$)~: \omterm{def:g4:measure:perimeter}{périmètre} $28$~cm, \omterm{def:g4:measure:area}{aire} $48$~cm$^2$.
Le \omterm{def:g4:measure:perimeter}{périmètre} a \emph{doublé}~; l'\omterm{def:g4:measure:area}{aire} a été multipliée par
\emph{quatre} ($2 \times 2$)~: longueurs et \omterm{def:g4:measure:area}{aires} ne se mettent pas
à l'échelle de la même façon.
\end{solution}
