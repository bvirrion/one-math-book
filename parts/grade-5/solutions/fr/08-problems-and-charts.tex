\section*{Chapitre \ref{ch:g5:data} --- Problèmes et graphiques}

\begin{solution}{exo:g5:data:1}
Billets~: $32 \times 7 = 224$. Total~: $240 + 224 = 464$.
\end{solution}

\begin{solution}{exo:g5:data:2}
Par kg~: $7.50 \div 3 = 2.50$. Cinq kg~:
$5 \times 2.50 = 12.50$.
\end{solution}

\begin{solution}{exo:g5:data:3}
$87 \div 6 = 14.5$~: chacun paie $14.50$.
\end{solution}

\begin{solution}{exo:g5:data:4}
Dépensé~: $13.90 + 8.50 = 22.40$. \omterm{def:g3:division:remainder}{Reste}~: $25 - 22.40 = 2.60$.
\end{solution}

\begin{solution}{exo:g5:data:5}
Plus de $50$~mm~: janvier ($60$), mars ($55$), avril ($70$), mai
($80$). Mai moins juin~: $80 - 30 = 50$~mm.
\end{solution}

\begin{solution}{exo:g5:data:6}
Semaine $2$~: $6$~cm. Croissance de $4$~cm~: entre les semaines $2$
et $3$ ($6 \to 10$), et entre les semaines $4$ et $5$
($15 \to 19$).
\end{solution}

\begin{solution}{exo:g5:data:7}
Quatre barres de hauteurs $4$, $7$, $3$, $6$ (graduation tous les
$1$ ou $2$ buts). Total~: $4 + 7 + 3 + 6 = 20$ buts.
\end{solution}

\begin{solution}{exo:g5:data:8}
Pack par un~: $3.20 \div 8 = 0.40$. Économie par yaourt~:
$0.50 - 0.40 = 0.10$ (dix centimes).
\end{solution}

\begin{solution}{exo:g5:data:9}
Façon 1~: samedi $148 \times 9.50 = 1\,406$~; dimanche
$95 \times 9.50 = 902.50$~; \omterm{ex:g1:subtraction:difference}{différence} $503.50$.

Façon 2~: $148 - 95 = 53$ billets de plus, chacun valant $9.50$~:
$53 \times 9.50 = 503.50$. Même réponse --- la \omterm{ex:g1:subtraction:difference}{différence} des
recettes est la recette de la \omterm{ex:g1:subtraction:difference}{différence}.
\end{solution}

\begin{solution}{exo:g5:data:10}
Par personne~: $300 \div 4 = 75$~g. Pour dix~:
$10 \times 75 = 750$~g de riz.
\end{solution}

\begin{solution}{exo:g5:data:11}
Beaucoup de problèmes corrects --- par exemple~: «~Milo achète
$3$ jus à $1.20$ chacun et un sandwich à $3.40$. Il paie avec un
bon de $12.50$. Combien reste-t-il du bon~?~» Solution~: jus
$3 \times 1.20 = 3.60$~; total dépensé $3.60 + 3.40 = 7$~; \omterm{def:g3:division:remainder}{reste}
$12.50 - 7 = 5.50$.
\end{solution}
