\section*{Chapitre \ref{ch:g5:solids} --- Cubes et pavés}

\begin{solution}{exo:g5:solids:1}
Par exemple~: un dé (cube)~; une boîte de céréales (pavé)~; une
boîte de conserve (cylindre)~; un ballon de football (boule)~;
certains toits ou les monuments d'Égypte (pyramide).
\end{solution}

\begin{solution}{exo:g5:solids:2}
Un pavé a $6$ faces, $12$ arêtes, $8$ sommets --- et
$6 + 8 = 12 + 2$. \checkmark
\end{solution}

\begin{solution}{exo:g5:solids:3}
Les six faces viennent en trois paires de faces opposées
identiques (haut--bas, avant--arrière, gauche--droite). Sur un
cube, les six faces sont des carrés identiques.
\end{solution}

\begin{solution}{exo:g5:solids:4}
Patron en croix~: quatre carrés de $2$~cm en rangée, un au-dessus
et un en dessous du deuxième carré. Plier l'anneau de quatre donne
les côtés~; les deux carrés en plus ferment le haut et le bas.
\end{solution}

\begin{solution}{exo:g5:solids:5}
Sur le patron en croix (quatre carrés en rangée, un au-dessus, un
en dessous du deuxième), les paires opposées sont~: les carrés $1$
et $3$ de la rangée~; les carrés $2$ et $4$ de la rangée~; les
carrés du haut et du bas. Un étiquetage correct~: rangée $= 1, 2,
6, 5$~; haut $= 3$~; bas $= 4$ --- alors $1{+}6$, $2{+}5$, $3{+}4$
font tous $7$.
\end{solution}

\begin{solution}{exo:g5:solids:6}
Pavé~: $5 \times 2 = 10$ par couche, $3$ couches~: $30$ cubes.
Cube~: $3 \times 3 \times 3 = 27$. Empilement en L~:
$3 \times 2 \times 1 = 6$ plus $1 \times 2 \times 1 = 2$~: $8$
cubes.
\end{solution}

\begin{solution}{exo:g5:solids:7}
Couches de $12$~: par exemple $4 \times 3$ ou $6 \times 2$ (ou
$12 \times 1$). Avec $3$ couches, le volume est
$12 \times 3 = 36$ cubes dans tous les cas.
\end{solution}

\begin{solution}{exo:g5:solids:8}
$27 = 3 \times 3 \times 3$~: un cube de côté $3$.
$64 = 4 \times 4 \times 4$~: un cube de côté $4$.
\end{solution}

\begin{solution}{exo:g5:solids:9}
La boîte contient $10 \times 5 \times 4 = 200$ morceaux de sucre.
À $6$ par jour~: $200 = 6 \times 33 + 2$, donc la boîte dure
$33$ jours complets (et $2$ morceaux restent pour le matin du jour
$34$).
\end{solution}

\begin{solution}{exo:g5:solids:10}
Peinture sur $3$ faces~: les $8$ cubes des coins. Sur $2$ faces~:
les milieux des $12$ arêtes~: $12$ cubes. Sur $1$ face~: les
centres des $6$ faces~: $6$ cubes. Sur aucune~: le seul cube
caché au centre~: $1$. Vérif.~: $8 + 12 + 6 + 1 = 27$. \checkmark
\end{solution}
