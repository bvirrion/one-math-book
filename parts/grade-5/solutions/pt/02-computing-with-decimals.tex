\begin{solution}{exo:g5:decimalops:1}
$45.7 + 8.93 = 54.63$; \quad
$16.25 + 3.75 = 20$; \quad
$0.86 + 12.4 = 13.26$.
\end{solution}

\begin{solution}{exo:g5:decimalops:2}
$9.4 - 2.72 = 6.68$ (verificar: $6.68 + 2.72 = 9.4$).

$20 - 13.45 = 6.55$ (verificar: $6.55 + 13.45 = 20$).

$6.03 - 5.9 = 0.13$.
\end{solution}

\begin{solution}{exo:g5:decimalops:3}

$9.99 \to 10$: $0.01$.
\end{solution}

\begin{solution}{exo:g5:decimalops:4}

\begin{solution}{exo:g5:decimalops:5}
$4.05$ eu $= 405$ cm; \quad $325$ cm $= 3.25$ m; \quad
$0.6$ kg $= 600$ g; \quad $85$ cL $= 0.85$ eu.
\end{solution}

\begin{solution}{exo:g5:decimalops:6}
$6.2 \times 4$: estimativa $24$; exatamente $24.8$.

$3.45 \times 8$: estimativa $28$ ($3.5 \times 8$); exatamente $27.6$.

$12.5 \times 6$: estimativa $75$; exatamente $75$.
\end{solution}

\begin{solution}{exo:g5:decimalops:7}
$7$ voltas: $0.4 \times 7 = 2.8$ km. $25$ voltas:
$0.4 \times 25 = 10$ km.
\end{solution}

\begin{solution}{exo:g5:decimalops:8}
Baguetes: $1.15 \times 2 = 2.30$. Total: $2.30 + 12.60 = 14.90$.
Mudar: $20 - 14.90 = 5.10$.
\end{solution}

\begin{solution}{exo:g5:decimalops:9}
Pacote: $1.5 \times 6 = 9$ L. Dias: $0.75 + 0.75 = 1.5$ L a cada dois
dias, então $9$ eu último $12$ dias (doze porções de $0.75$:
$12 \times 0.75 = 9$).
\end{solution}

\begin{solution}{exo:g5:decimalops:10}
Estimativa: $5.6$ é mais do que $5.5$, e $5.5 \times 3 = 16.5$ --- então
o \omterm{def:g2:mult:def}{produto} deve ser \emph{mais} que $16.5$, e $15.18$ é muito
pequeno. Corretamente: $56 \times 3 = 168$, um dígito decimal, então
$5.6 \times 3 = 16.8$. Milo colou seus dois parciais \omterm{def:g2:mult:def}{produtos} ($15$ e
$18$) lado a lado em vez de adicioná-los aos seus lugares:
$15 + 1.8 = 16.8$.
\end{solution}
