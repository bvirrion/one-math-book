\section*{Capítulo \ref{ch:g5:solids} --- Cubos e Caixas}

\begin{solution}{exo:g5:solids:1}
Por exemplo: um dado (\omterm{def:g5:solids:def}{cubo}); um cereal \omterm{def:g5:solids:def}{caixa} (\omterm{def:g5:solids:def}{caixa}); uma lata (\omterm{def:g5:solids:def}{cilindro}); um
futebol (bola); certos telhados ou os monumentos do Egito
(\omterm{def:g5:solids:def}{pirâmide}).
\end{solution}

\begin{solution}{exo:g5:solids:2}
UM \omterm{def:g5:solids:def}{caixa} tem $6$ \omterm{def:g5:solids:def}{rostos}, $12$ \omterm{def:g5:solids:def}{bordas}, $8$ \omterm{def:g5:solids:def}{vértices} --- e
$6 + 8 = 12 + 2$. \checkmark
\end{solution}

\begin{solution}{exo:g5:solids:3}
Os seis \omterm{def:g5:solids:def}{rostos} vêm em três \omterm{def:g3:numbers:evenodd}{pares} de opostos idênticos \omterm{def:g5:solids:def}{rostos}
(cima-baixo, frente-trás, esquerda-direita). Em um \omterm{def:g5:solids:def}{cubo}, todos os seis \omterm{def:g5:solids:def}{rostos}
são quadrados idênticos.
\end{solution}

\begin{solution}{exo:g5:solids:4}
Em forma de cruz \omterm{def:g5:solids:net}{líquido}: quatro $2$ cm quadrados seguidos, um acima e um
abaixo do segundo quadrado. Dobrar o anel de quatro faz os lados;
os dois quadrados extras fecham a parte superior e inferior.
\end{solution}

\begin{solution}{exo:g5:solids:5}
Na cruz \omterm{def:g5:solids:net}{líquido} (quatro quadrados seguidos, um acima, um abaixo do
segundo), os \omterm{def:g3:numbers:evenodd}{pares} opostos são: quadrados $1$ e $3$ da linha;
quadrados $2$ e $4$ da linha; os quadrados superior e inferior. Um
rotulagem correta: linha $= 1, 2, 6, 5$; principal $= 3$; fundo $= 4$
--- então $1{+}6$, $2{+}5$, $3{+}4$ todos fazem $7$.
\end{solution}

\begin{solution}{exo:g5:solids:6}
\omterm{def:g5:solids:def}{Caixa}: $5 \times 2 = 10$ por camada, $3$ camadas: $30$ \omterm{def:g5:solids:def}{cubos}. \omterm{def:g5:solids:def}{Cubo}:
$3 \times 3 \times 3 = 27$. Pilha L: $3 \times 2 \times 1 = 6$ mais
$1 \times 2 \times 1 = 2$: $8$ \omterm{def:g5:solids:def}{cubos}.
\end{solution}

\begin{solution}{exo:g5:solids:7}
Camadas de $12$: por exemplo $4 \times 3$ ou $6 \times 2$ (ou
$12 \times 1$). Com $3$ camadas, o \omterm{def:g5:solids:volume}{volume} é
$12 \times 3 = 36$ \omterm{def:g5:solids:def}{cubos} em todos os casos.
\end{solution}

\begin{solution}{exo:g5:solids:8}
$27 = 3 \times 3 \times 3$: um \omterm{def:g5:solids:def}{cubo} de lado $3$.
$64 = 4 \times 4 \times 4$: um \omterm{def:g5:solids:def}{cubo} de lado $4$.
\end{solution}

\begin{solution}{exo:g5:solids:9}
O \omterm{def:g5:solids:def}{caixa} detém $10 \times 5 \times 4 = 200$ açúcar \omterm{def:g5:solids:def}{cubos}. No $6$ por
dia: $200 = 6 \times 33 + 2$, então o \omterm{def:g5:solids:def}{caixa} dura $33$ dias inteiros (e
$2$ \omterm{def:g5:solids:def}{cubos} permanecer para a manhã do dia $34$).
\end{solution}

\begin{solution}{exo:g5:solids:10}
Pintar $3$ \omterm{def:g5:solids:def}{rostos}: o $8$ canto \omterm{def:g5:solids:def}{cubos}. Sobre $2$ \omterm{def:g5:solids:def}{rostos}: os meios
do $12$ \omterm{def:g5:solids:def}{bordas}: $12$ \omterm{def:g5:solids:def}{cubos}. Sobre $1$ face: os centros do $6$
\omterm{def:g5:solids:def}{rostos}: $6$ \omterm{def:g5:solids:def}{cubos}. Em nenhum: o único centro oculto \omterm{def:g5:solids:def}{cubo}: $1$.
Verificar: $8 + 12 + 6 + 1 = 27$. \checkmark
\end{solution}
