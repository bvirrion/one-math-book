\section*{Capítulo \ref{ch:g5:fractions} --- Frações como Números}

\begin{solution}{exo:g5:fractions:1}
$\frac25$: dois passos de $0$; $\frac55$: sobre $1$; $\frac85$: três
passos além $1$; $\frac{10}{5}$: sobre $2$.
\end{solution}

\begin{solution}{exo:g5:fractions:2}
$\frac12 = \frac48$; \quad $\frac34 = \frac68$; \quad
$\frac25 = \frac{4}{10}$; \quad $\frac69 = \frac23$.
\end{solution}

\begin{solution}{exo:g5:fractions:3}
$\frac82 = 4$; \quad $\frac94$ não está inteiro ($9$ não é um múltiplo
de $4$); \quad $\frac{15}{3} = 5$; \quad $\frac{20}{5} = 4$.
\end{solution}

\begin{solution}{exo:g5:fractions:4}
$0.7$; \quad $0.41$; \quad $0.5$; \quad $0.75$; \quad $0.2$.
\end{solution}

\begin{solution}{exo:g5:fractions:5}
$0.9 = \frac{9}{10}$; \quad $0.13 = \frac{13}{100}$; \quad
$2.5 = \frac{25}{10}$; \quad $0.75 = \frac{75}{100} = \frac34$.
\end{solution}

\begin{solution}{exo:g5:fractions:6}
$\frac47 < \frac67$ (mesmo \omterm{def:g4:fractions:def}{denominador}). \quad
$\frac35 > \frac37$ (mesmo \omterm{def:g4:fractions:def}{numerador}: quintos são maiores). \quad
$\frac25 < \frac12 < \frac{7}{12}$, então $\frac25 < \frac{7}{12}$
(ponto de referência $\frac12$). \quad
$\frac98 > 1 > \frac{11}{12}$ (ponto de referência $1$).
\end{solution}

\begin{solution}{exo:g5:fractions:7}
Decimais: $0.5$; $0.4$; $0.75$; $0.6$; $0.2$. Ordem:
\[
\frac15 < 0.4 < \frac12 < 0.6 < \frac34 .
\]
\end{solution}

\begin{solution}{exo:g5:fractions:8}
Instrumento: $\frac14$ de $28 = 7$. Esporte: $\frac12$ de $28 = 14$.
O grupo esportivo é maior.
\end{solution}

\begin{solution}{exo:g5:fractions:9}
$\frac14$ h $= 15$ min; $\frac12$ h $= 30$ min; $\frac34$ h $= 45$
min; $\frac{1}{60}$ h $= 1$ min.
\end{solution}

\begin{solution}{exo:g5:fractions:10}
Tom: $\frac35$ de $50 = 30$ cl. Anna: $\frac12$ de $60 = 30$ cl. Eles
bebi a mesma quantidade! O nu frações ($\frac35 > \frac12$)
comparar ações de \emph{diferente} garrafas --- apenas o real
os valores podem ser comparados.
\end{solution}

\begin{solution}{exo:g5:fractions:11}
Entre $\frac12$ e $1$: por exemplo $\frac34$ (ou seja,\ $0.75$).
Entre $\frac12 = 0.5$ e $\frac34 = 0.75$: por exemplo $0.6 =
\frac35$.
\end{solution}
