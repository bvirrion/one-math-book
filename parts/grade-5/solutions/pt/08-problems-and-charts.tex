\section*{Capítulo \ref{ch:g5:data} --- Problemas e gráficos}

\begin{solution}{exo:g5:data:1}
Ingressos: $32 \times 7 = 224$. Total: $240 + 224 = 464$.
\end{solution}

\begin{solution}{exo:g5:data:2}
Por um kg: $7.50 \div 3 = 2.50$. Cincokg:
$5 \times 2.50 = 12.50$.
\end{solution}

\begin{solution}{exo:g5:data:3}
$87 \div 6 = 14.5$: cada um paga $14.50$.
\end{solution}

\begin{solution}{exo:g5:data:4}
Gasto: $13.90 + 8.50 = 22.40$. Esquerda: $25 - 22.40 = 2.60$.
\end{solution}

\begin{solution}{exo:g5:data:5}
Mais do que $50$ mm: janeiro ($60$), Marchar ($55$), abril ($70$), Poderia
($80$). Maio menos junho: $80 - 30 = 50$ mm.
\end{solution}

\begin{solution}{exo:g5:data:6}
Semana $2$: $6$ cm. Crescimento de $4$ cm: entre semanas $2$ e $3$
($6 \to 10$) e entre semanas $4$ e $5$ ($15 \to 19$).
\end{solution}

\begin{solution}{exo:g5:data:7}
Quatro barras de altura $4$, $7$, $3$, $6$ (formatura a cada $1$ ou
$2$ metas). Total: $4 + 7 + 3 + 6 = 20$ metas.
\end{solution}

\begin{solution}{exo:g5:data:8}
Pacote por um: $3.20 \div 8 = 0.40$. Economizando por iogurte:
$0.50 - 0.40 = 0.10$ (dez centavos).
\end{solution}

\begin{solution}{exo:g5:data:9}
Caminho 1: sábado $148 \times 9.50 = 1\,406$; Domingo
$95 \times 9.50 = 902.50$; \omterm{ex:g1:subtraction:difference}{diferença} $503.50$.

Caminho 2: $148 - 95 = 53$ mais ingressos, cada um valendo $9.50$:
$53 \times 9.50 = 503.50$. Mesma resposta --- a \omterm{ex:g1:subtraction:difference}{diferença} do
takes é a tomada do \omterm{ex:g1:subtraction:difference}{diferença}.
\end{solution}

\begin{solution}{exo:g5:data:10}
Por pessoa: $300 \div 4 = 75$ g. Por dez:
$10 \times 75 = 750$ g de arroz.
\end{solution}

\begin{solution}{exo:g5:data:11}
Muitos problemas corretos --- por exemplo: ``Milo compra $3$ sucos em
$1.20$ cada um e um sanduíche em $3.40$. Ele paga com um $12.50$
comprovante. Quanto resta do voucher?'' Solução: sucos
$3 \times 1.20 = 3.60$; total gasto $3.60 + 3.40 = 7$; esquerda
$12.50 - 7 = 5.50$.
\end{solution}
