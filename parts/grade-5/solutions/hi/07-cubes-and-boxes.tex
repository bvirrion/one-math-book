\section*{अध्याय \ref{ch:g5:solids} --- क्यूब्स और बक्से}

\begin{solution}{exo:g5:solids:1}
उदाहरण के लिए: एक पासा (\omterm{def:g5:solids:def}{घनक्षेत्र}); एक अनाज \omterm{def:g5:solids:def}{डिब्बा} (\omterm{def:g5:solids:def}{डिब्बा}); एक कर सकते हैं (\omterm{def:g5:solids:def}{सिलेंडर}); ए
फ़ुटबॉल (गेंद); मिस्र की कुछ छतें या स्मारक
(\omterm{def:g5:solids:def}{पिरामिड}).
\end{solution}

\begin{solution}{exo:g5:solids:2}
ए \omterm{def:g5:solids:def}{डिब्बा} है $6$ \omterm{def:g5:solids:def}{चेहरे}, $12$ \omterm{def:g5:solids:def}{किनारों}, $8$ \omterm{def:g5:solids:def}{कोने} --- और
$6 + 8 = 12 + 2$. \checkmark
\end{solution}

\begin{solution}{exo:g5:solids:3}
छह \omterm{def:g5:solids:def}{चेहरे} समान विपरीत के तीन \omterm{def:g1:addition:def}{जोड़}े में आते हैं \omterm{def:g5:solids:def}{चेहरे}
(ऊपर-नीचे, आगे-पीछे, बाएँ-दाएँ)। एक पर \omterm{def:g5:solids:def}{घनक्षेत्र}, सभी छह \omterm{def:g5:solids:def}{चेहरे}
समरूप वर्ग हैं.
\end{solution}

\begin{solution}{exo:g5:solids:4}
क्रॉस के आकार \omterm{def:g5:solids:net}{जाल}: चार $2$ एक पंक्ति में सेमी वर्ग, एक ऊपर और एक
दूसरे वर्ग के नीचे. चार की अंगूठी को मोड़ने से भुजाएँ बनती हैं;
दो अतिरिक्त वर्ग ऊपर और नीचे को बंद करते हैं।
\end{solution}

\begin{solution}{exo:g5:solids:5}
एक दोगला \omterm{def:g5:solids:net}{जाल} (एक पंक्ति में चार वर्ग, एक ऊपर, एक नीचे
दूसरा), विपरीत \omterm{def:g1:addition:def}{जोड़}े हैं: वर्ग $1$ और $3$ पंक्ति का;
चौकों $2$ और $4$ पंक्ति का; ऊपर और नीचे के वर्ग. एक
सही लेबलिंग: पंक्ति $= 1, 2, 6, 5$; शीर्ष $= 3$; तल $= 4$
--- तब $1{+}6$, $2{+}5$, $3{+}4$ सभी बनाते हैं $7$.
\end{solution}

\begin{solution}{exo:g5:solids:6}
\omterm{def:g5:solids:def}{डिब्बा}: $5 \times 2 = 10$ प्रति परत, $3$ परतें: $30$ क्यूब्स. \omterm{def:g5:solids:def}{घनक्षेत्र}:
$3 \times 3 \times 3 = 27$. एल-स्टैक: $3 \times 2 \times 1 = 6$ प्लस
$1 \times 2 \times 1 = 2$: $8$ क्यूब्स.
\end{solution}

\begin{solution}{exo:g5:solids:7}
की परतें $12$: उदाहरण के लिए $4 \times 3$ या $6 \times 2$ (या
$12 \times 1$). साथ $3$ परतें, \omterm{def:g5:solids:volume}{आयतन} है
$12 \times 3 = 36$ क्यूब्स हर मामले में.
\end{solution}

\begin{solution}{exo:g5:solids:8}
$27 = 3 \times 3 \times 3$: ए \omterm{def:g5:solids:def}{घनक्षेत्र} इस पक्ष की $3$.
$64 = 4 \times 4 \times 4$: ए \omterm{def:g5:solids:def}{घनक्षेत्र} इस पक्ष की $4$.
\end{solution}

\begin{solution}{exo:g5:solids:9}
 \omterm{def:g5:solids:def}{डिब्बा} रखती है $10 \times 5 \times 4 = 200$ चीनी क्यूब्स. पर $6$ प्रति दिन: $200 = 6 \times 33 + 2$, इतना \omterm{def:g5:solids:def}{डिब्बा} रहता है $33$ पूरे दिन (और
$2$ क्यूब्स दिन की सुबह तक रहो $34$).
\end{solution}

\begin{solution}{exo:g5:solids:10}
पर पैंट $3$ \omterm{def:g5:solids:def}{चेहरे}: द $8$ कोना क्यूब्स. पर $2$ \omterm{def:g5:solids:def}{चेहरे}: मध्य
का $12$ \omterm{def:g5:solids:def}{किनारों}: $12$ क्यूब्स. पर $1$ चेहरा: के केंद्र $6$
\omterm{def:g5:solids:def}{चेहरे}: $6$ क्यूब्स. किसी पर नहीं: एकल छिपा हुआ केंद्र \omterm{def:g5:solids:def}{घनक्षेत्र}: $1$.
जाँच करना: $8 + 12 + 6 + 1 = 27$. \checkmark
\end{solution}
