\section*{अध्याय \ref{ch:g5:areas} --- परिधि और क्षेत्रफल}

\begin{solution}{exo:g5:areas:1}
\omterm{def:g4:measure:area}{क्षेत्र}: $8 \times 5 = 40$ सेमी $^2$. \omterm{def:g4:measure:perimeter}{परिधि}:
$2 \times (8 + 5) = 26$ सेमी।
\end{solution}

\begin{solution}{exo:g5:areas:2}
$9 \times 9 = 81$ सेमी $^2$; \quad $12 \times 7 = 84$ एम $^2$; \quad
$4.5 \times 6 = 27$ सेमी $^2$.
\end{solution}

\begin{solution}{exo:g5:areas:3}
चौड़ाई: $63 \div 9 = 7$ सेमी। \omterm{def:g4:measure:perimeter}{परिधि}: $2 \times (9 + 7) = 32$ सेमी।
\end{solution}

\begin{solution}{exo:g5:areas:4}
उदाहरण के लिए $6 \times 4$ (\omterm{def:g4:measure:perimeter}{परिधि} $20$) और $8 \times 3$
(\omterm{def:g4:measure:perimeter}{परिधि} $22$): वही \omterm{def:g4:measure:area}{क्षेत्र} $24$, अलग \omterm{def:g4:measure:perimeter}{परिधि}.
\end{solution}

\begin{solution}{exo:g5:areas:5}
$3$ एम $^2 = 30\,000$ सेमी $^2$; \quad $50\,000$ सेमी $^2 = 5$ एम $^2$.
\end{solution}

\begin{solution}{exo:g5:areas:6}
छड़: $8 \times 2 = 16$ सेमी $^2$; तना: $2 \times 5 = 10$ सेमी $^2$;
कुल $26$ सेमी $^2$.
\end{solution}

\begin{solution}{exo:g5:areas:7}
$30 \times 20 - 24 \times 14 = 600 - 336 = 264$ सेमी $^2$ लकड़ी का.
\end{solution}

\begin{solution}{exo:g5:areas:8}
\omterm{def:g3:division:half}{आधा}  $8 \times 6$ आयत: $48 \div 2 = 24$ सेमी $^2$.
\end{solution}

\begin{solution}{exo:g5:areas:9}
बीज: \omterm{def:g4:measure:area}{क्षेत्र} $7 \times 4 = 28$ एम $^2$, लागत $28 \times 2 = 56$.
बाड़: \omterm{def:g4:measure:perimeter}{परिधि} $2 \times (7 + 4) = 22$ मी, लागत
$22 \times 3 = 66$.
\end{solution}

\begin{solution}{exo:g5:areas:10}
की दो टाइलें $50$ सेमी एक \omterm{def:g4:fractions:def}{मीटर} बनाओ: के साथ $12$ मी लंबाई,
$24$ टाइल्स; साथ $2$ मी चौड़ाई, $4$ टाइल्स. कुल:
$24 \times 4 = 96$ टाइल्स.
\end{solution}

\begin{solution}{exo:g5:areas:11}
पहले: \omterm{def:g4:measure:perimeter}{परिधि} $2 \times (4 + 3) = 14$ सेमी, \omterm{def:g4:measure:area}{क्षेत्र} $12$ सेमी $^2$.
(पक्षों के बाद) $8$ और $6$): \omterm{def:g4:measure:perimeter}{परिधि} $28$ सेमी, \omterm{def:g4:measure:area}{क्षेत्र} $48$ सेमी $^2$.
 \omterm{def:g4:measure:perimeter}{परिधि} \emph{दोगुनी}; \omterm{def:g4:measure:area}{क्षेत्र} से गुणा किया गया
\emph{चार} ($2 \times 2$): लंबाई और \omterm{def:g4:measure:area}{क्षेत्र}ों समान माप न करें
रास्ता.
\end{solution}
