\section*{अध्याय \ref{ch:g5:fractions} --- संख्याओं के रूप में भिन्न}

\begin{solution}{exo:g5:fractions:1}
$\frac25$: दो कदम आगे $0$; $\frac55$: पर $1$; $\frac85$: तीन
कदम आगे $1$; $\frac{10}{5}$: पर $2$.
\end{solution}

\begin{solution}{exo:g5:fractions:2}
$\frac12 = \frac48$; \quad $\frac34 = \frac68$; \quad
$\frac25 = \frac{4}{10}$; \quad $\frac69 = \frac23$.
\end{solution}

\begin{solution}{exo:g5:fractions:3}
$\frac82 = 4$; \quad $\frac94$ संपूर्ण नहीं है ($9$ एकाधिक नहीं है
का $4$); \quad $\frac{15}{3} = 5$; \quad $\frac{20}{5} = 4$.
\end{solution}

\begin{solution}{exo:g5:fractions:4}
$0.7$; \quad $0.41$; \quad $0.5$; \quad $0.75$; \quad $0.2$.
\end{solution}

\begin{solution}{exo:g5:fractions:5}
$0.9 = \frac{9}{10}$; \quad $0.13 = \frac{13}{100}$; \quad
$2.5 = \frac{25}{10}$; \quad $0.75 = \frac{75}{100} = \frac34$.
\end{solution}

\begin{solution}{exo:g5:fractions:6}
$\frac47 < \frac67$ (वही \omterm{def:g4:fractions:def}{भाजक}). \quad
$\frac35 > \frac37$ (वही \omterm{def:g4:fractions:def}{मीटर}: पाँचवें बड़े हैं)। \quad
$\frac25 < \frac12 < \frac{7}{12}$, इसलिए $\frac25 < \frac{7}{12}$
(सीमाचिह्न $\frac12$). \quad
$\frac98 > 1 > \frac{11}{12}$ (सीमाचिह्न $1$).
\end{solution}

\begin{solution}{exo:g5:fractions:7}
दशमलव: $0.5$; $0.4$; $0.75$; $0.6$; $0.2$. आदेश देना:
\[
\frac15 < 0.4 < \frac12 < 0.6 < \frac34 .
\]
\end{solution}

\begin{solution}{exo:g5:fractions:8}
यंत्र: $\frac14$ का $28 = 7$. खेल: $\frac12$ का $28 = 14$.
खेल समूह बड़ा है.
\end{solution}

\begin{solution}{exo:g5:fractions:9}
$\frac14$ एच $= 15$ मिनट; $\frac12$ एच $= 30$ मिनट; $\frac34$ एच $= 45$
मिनट; $\frac{1}{60}$ एच $= 1$ मि.
\end{solution}

\begin{solution}{exo:g5:fractions:10}
टॉम: $\frac35$ का $50 = 30$ सीएल. एना: $\frac12$ का $60 = 30$ सीएल. वे
उतनी ही मात्रा में पिया! दी बेयर \omterm{def:g4:fractions:def}{अंश}ों ($\frac35 > \frac12$)
के शेयरों की तुलना करें \emph{अलग} बोतलें --- केवल वास्तविक
मात्राओं की तुलना की जा सकती है।
\end{solution}

\begin{solution}{exo:g5:fractions:11}
बीच में $\frac12$ और $1$: उदाहरण के लिए $\frac34$ (अर्थात्\ $0.75$).
बीच में $\frac12 = 0.5$ और $\frac34 = 0.75$: उदाहरण के लिए $0.6 =
\frac35$.
\end{solution}
