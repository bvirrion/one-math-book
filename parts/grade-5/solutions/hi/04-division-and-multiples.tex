\section*{अध्याय \ref{ch:g5:division} --- विभाजन और गुणज}

\begin{solution}{exo:g5:division:1}
$851 \div 23$: \omterm{def:g5:division:multiple}{गुणकों} का $23$: $23, 46, 69, 92, 115, 138, 161,
184, 207$. तब $85 \div 23$: $3$ समय ($69$), \omterm{def:g3:division:remainder}{शेष} $16$;
नीचे लाओ $1$: $161 \div 23 = 7$ बिल्कुल। \omterm{def:g3:division:remainder}{भागफल} $37$,
\omterm{def:g3:division:remainder}{शेष} $0$ ($23 \times 37 = 851$).

$704 \div 32$: $70 \div 32$: $2$ समय ($64$), \omterm{def:g3:division:remainder}{शेष} $6$; लाओ
नीचे $4$: $64 \div 32 = 2$. \omterm{def:g3:division:remainder}{भागफल} $22$, \omterm{def:g3:division:remainder}{शेष} $0$.
\end{solution}

\begin{solution}{exo:g5:division:2}
$9 \div 2 = 4.5$; \quad $27 \div 4 = 6.75$; \quad
$33 \div 5 = 6.6$; \quad $21 \div 8 = 2.625$.
\end{solution}

\begin{solution}{exo:g5:division:3}
$54 \div 4 = 13.5$: प्रत्येक भुगतान करता है $13.50$.
\end{solution}

\begin{solution}{exo:g5:division:4}
$10 \div 3 = 3.333\dots $: अंक $3$ हमेशा के लिए दोहराता है ---
विभाजन कभी नहीं रुकता. सौवें तक पूर्णांकित: $3.33$.
\end{solution}

\begin{solution}{exo:g5:division:5}
मल्टीपल्स का $7$ तक $70$: $7, 14, 21, 28, 35, 42, 49, 56, 63,
70$. वि\omterm{def:g5:division:multiple}{भाजक} का $18$: $1, 2, 3, 6, 9, 18$. वि\omterm{def:g5:division:multiple}{भाजक} का $24$:
$1, 2, 3, 4, 6, 8, 12, 24$.
\end{solution}

\begin{solution}{exo:g5:division:6}
$470$: द्वारा $2$ (में समाप्त होता है $0$), द्वारा $5$, द्वारा $10$; अंक \omterm{def:g1:addition:def}{जोड़} $11$: द्वारा नहीं
$3$.

$735$: में समाप्त होता है $5$: द्वारा $5$, द्वारा नहीं $2$ और न $10$; अंक \omterm{def:g1:addition:def}{जोड़} $15$: द्वारा
$3$.

$8\,001$: \omterm{def:g3:numbers:evenodd}{विषम}; अंक \omterm{def:g1:addition:def}{जोड़} $9$: द्वारा $3$ केवल।

$1\,314$: \omterm{def:g3:numbers:evenodd}{यहां तक की}; अंक \omterm{def:g1:addition:def}{जोड़} $9$: द्वारा $2$ और तक $3$ (इसलिए द्वारा $6$),
द्वारा नहीं $5$.
\end{solution}

\begin{solution}{exo:g5:division:7}
द्वारा विभाज्य $3$ और $5$ का अर्थ है विभाज्य $15$: बीच में $60$ और
$80$, संख्या $60$, $75$.
\end{solution}

\begin{solution}{exo:g5:division:8}
$7.2 \div 6 = 1.2$: प्रत्येक टुकड़े का माप $1.2$ मी (जांचें:
$1.2 \times 6 = 7.2$).
\end{solution}

\begin{solution}{exo:g5:division:9}
$1\,000 = 24 \times 41 + 16$: इकतालीस पूर्ण बैग, $16$ पत्थर
बायां.
\end{solution}

\begin{solution}{exo:g5:division:10}
$7 \div 5 = 1.4$: हरेक $5$ दोस्तों मिलता है $1.4$ बार ---
सलाखों को दसवें हिस्से में काटने से संभव है। के लिए $5 \div 7$: विभाजन
देता है $0.714285\dots $, जो कभी नहीं रुकता --- कोई सटीक दशमलव हिस्सा नहीं
मौजूद है; केवल \omterm{def:g4:fractions:def}{अंश} $\frac57$ इसे सटीक नाम दें।
\end{solution}

\begin{solution}{exo:g5:division:11}
अंक \omterm{def:g1:addition:def}{जोड़}: $5 + 2 + d = 7 + d $. द्वारा विभाज्य $3$ कब $7 + d $ है
$9$, $12$ या $15$: $ d = 2$, $5$ या $8$. सबसे छोटा है $ d = 2$
(दे रहा है $522 = 3 \times 174$).
\end{solution}
