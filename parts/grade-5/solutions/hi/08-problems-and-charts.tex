\section*{अध्याय \ref{ch:g5:data} --- समस्याएँ और चार्ट}

\begin{solution}{exo:g5:data:1}
टिकट: $32 \times 7 = 224$. कुल: $240 + 224 = 464$.
\end{solution}

\begin{solution}{exo:g5:data:2}
प्रति एक किलो: $7.50 \div 3 = 2.50$. पांच किलो:
$5 \times 2.50 = 12.50$.
\end{solution}

\begin{solution}{exo:g5:data:3}
$87 \div 6 = 14.5$: प्रत्येक भुगतान करता है $14.50$.
\end{solution}

\begin{solution}{exo:g5:data:4}
खर्च किया गया: $13.90 + 8.50 = 22.40$. बाएं: $25 - 22.40 = 2.60$.
\end{solution}

\begin{solution}{exo:g5:data:5}
इससे अधिक $50$ मिमी: जनवरी ($60$), मार्च ($55$), अप्रैल ($70$), मई
($80$). मई माइनस जून: $80 - 30 = 50$ मिमी.
\end{solution}

\begin{solution}{exo:g5:data:6}
सप्ताह $2$: $6$ सेमी। का विकास $4$ सेमी: सप्ताहों के बीच $2$ और $3$
($6 \to 10$), और सप्ताहों के बीच $4$ और $5$ ($15 \to 19$).
\end{solution}

\begin{solution}{exo:g5:data:7}
ऊँचाई की चार पट्टियाँ $4$, $7$, $3$, $6$ (स्नातक प्रत्येक $1$ या
$2$ लक्ष्य)। कुल: $4 + 7 + 3 + 6 = 20$ लक्ष्य।
\end{solution}

\begin{solution}{exo:g5:data:8}
प्रति पैक: $3.20 \div 8 = 0.40$. प्रति दही बचत:
$0.50 - 0.40 = 0.10$ (दस सेंट).
\end{solution}

\begin{solution}{exo:g5:data:9}
तरीका 1: शनिवार $148 \times 9.50 = 1\,406$; रविवार
$95 \times 9.50 = 902.50$; \omterm{ex:g1:subtraction:difference}{अंतर} $503.50$.

रास्ता 2: $148 - 95 = 53$ अधिक टिकट, प्रत्येक मूल्य $9.50$:
$53 \times 9.50 = 503.50$. वही उत्तर--- \omterm{ex:g1:subtraction:difference}{अंतर} का
लेना का लेना है \omterm{ex:g1:subtraction:difference}{अंतर}.
\end{solution}

\begin{solution}{exo:g5:data:10}
प्रति व्यक्ति: $300 \div 4 = 75$ जी। दस के लिए:
$10 \times 75 = 750$ चावल का ग्राम.
\end{solution}

\begin{solution}{exo:g5:data:11}
कई सही समस्याएं --- उदाहरण के लिए: ``मिलो खरीदता है $3$ जूस पर
$1.20$ प्रत्येक और एक सैंडविच $3.40$. वह एक के साथ भुगतान करता है $12.50$
वाउचर. वाउचर का कितना हिस्सा बचा है?'' समाधान: जूस
$3 \times 1.20 = 3.60$; कुल खर्च $3.60 + 3.40 = 7$; बाएं
$12.50 - 7 = 5.50$.
\end{solution}
