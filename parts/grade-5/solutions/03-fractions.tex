\section*{Chapter \ref{ch:g5:fractions} --- Fractions as Numbers}

\begin{solution}{exo:g5:fractions:1}
$\frac25$: two steps from $0$; $\frac55$: on $1$; $\frac85$: three
steps past $1$; $\frac{10}{5}$: on $2$.
\end{solution}

\begin{solution}{exo:g5:fractions:2}
$\frac12 = \frac48$; \quad $\frac34 = \frac68$; \quad
$\frac25 = \frac{4}{10}$; \quad $\frac69 = \frac23$.
\end{solution}

\begin{solution}{exo:g5:fractions:3}
$\frac82 = 4$; \quad $\frac94$ is not whole ($9$ is not a multiple
of $4$); \quad $\frac{15}{3} = 5$; \quad $\frac{20}{5} = 4$.
\end{solution}

\begin{solution}{exo:g5:fractions:4}
$0.7$; \quad $0.41$; \quad $0.5$; \quad $0.75$; \quad $0.2$.
\end{solution}

\begin{solution}{exo:g5:fractions:5}
$0.9 = \frac{9}{10}$; \quad $0.13 = \frac{13}{100}$; \quad
$2.5 = \frac{25}{10}$; \quad $0.75 = \frac{75}{100} = \frac34$.
\end{solution}

\begin{solution}{exo:g5:fractions:6}
$\frac47 < \frac67$ (same denominator). \quad
$\frac35 > \frac37$ (same numerator: fifths are bigger). \quad
$\frac25 < \frac12 < \frac{7}{12}$, so $\frac25 < \frac{7}{12}$
(landmark $\frac12$). \quad
$\frac98 > 1 > \frac{11}{12}$ (landmark $1$).
\end{solution}

\begin{solution}{exo:g5:fractions:7}
Decimals: $0.5$; $0.4$; $0.75$; $0.6$; $0.2$. Order:
\[
\frac15 < 0.4 < \frac12 < 0.6 < \frac34 .
\]
\end{solution}

\begin{solution}{exo:g5:fractions:8}
Instrument: $\frac14$ of $28 = 7$. Sport: $\frac12$ of $28 = 14$.
The sport group is bigger.
\end{solution}

\begin{solution}{exo:g5:fractions:9}
$\frac14$ h $= 15$ min; $\frac12$ h $= 30$ min; $\frac34$ h $= 45$
min; $\frac{1}{60}$ h $= 1$ min.
\end{solution}

\begin{solution}{exo:g5:fractions:10}
Tom: $\frac35$ of $50 = 30$ cL. Ana: $\frac12$ of $60 = 30$ cL. They
drank the same amount! The bare fractions ($\frac35 > \frac12$)
compare shares of \emph{different} bottles --- only the actual
amounts can be compared.
\end{solution}

\begin{solution}{exo:g5:fractions:11}
Between $\frac12$ and $1$: for instance $\frac34$ (i.e.\ $0.75$).
Between $\frac12 = 0.5$ and $\frac34 = 0.75$: for instance $0.6 =
\frac35$.
\end{solution}
