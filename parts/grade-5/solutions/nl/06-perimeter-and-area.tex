\section*{Hoofdstuk \ref{ch:g5:areas} --- Omtrek en oppervlakte}

\begin{solution}{exo:g5:areas:1}
Oppervlakte: $8 \times 5 = 40$ cm$^2$. Omtrek:
$2 \times (8 + 5) = 26$ cm.
\end{solution}

\begin{solution}{exo:g5:areas:2}
$9 \times 9 = 81$ cm$^2$; \quad $12 \times 7 = 84$ m$^2$; \quad
$4.5 \times 6 = 27$ cm$^2$.
\end{solution}

\begin{solution}{exo:g5:areas:3}
Breedte: $63 \div 9 = 7$ cm. Omtrek: $2 \times (9 + 7) = 32$ cm.
\end{solution}

\begin{solution}{exo:g5:areas:4}
Bijvoorbeeld $6 \times 4$ (omtrek $20$) en $8 \times 3$ (omtrek
$22$): zelfde oppervlakte $24$, verschillende omtrekken.
\end{solution}

\begin{solution}{exo:g5:areas:5}
$3$ m$^2 = 30\,000$ cm$^2$; \quad $50\,000$ cm$^2 = 5$ m$^2$.
\end{solution}

\begin{solution}{exo:g5:areas:6}
Balk: $8 \times 2 = 16$ cm$^2$; steel: $2 \times 5 = 10$ cm$^2$;
totaal $26$ cm$^2$.
\end{solution}

\begin{solution}{exo:g5:areas:7}
$30 \times 20 - 24 \times 14 = 600 - 336 = 264$ cm$^2$ hout.
\end{solution}

\begin{solution}{exo:g5:areas:8}
Helft van de $8 \times 6$-rechthoek: $48 \div 2 = 24$ cm$^2$.
\end{solution}

\begin{solution}{exo:g5:areas:9}
Zaad: oppervlakte $7 \times 4 = 28$ m$^2$, kost $28 \times 2 = 56$.
Hek: omtrek $2 \times (7 + 4) = 22$ m, kost $22 \times 3 = 66$.
\end{solution}

\begin{solution}{exo:g5:areas:10}
Twee tegels van $50$ cm maken een meter: langs de lengte van $12$ m,
$24$ tegels; langs de breedte van $2$ m, $4$ tegels. Totaal:
$24 \times 4 = 96$ tegels.
\end{solution}

\begin{solution}{exo:g5:areas:11}
V\'{o}\'{o}r: omtrek $2 \times (4 + 3) = 14$ cm, oppervlakte $12$ cm$^2$.
Na (zijden $8$ en $6$): omtrek $28$ cm, oppervlakte $48$ cm$^2$.
De omtrek is \emph{verdubbeld}; de oppervlakte is vermenigvuldigd
met \emph{vier} ($2 \times 2$): lengtes en oppervlaktes schalen niet
op dezelfde manier.
\end{solution}
