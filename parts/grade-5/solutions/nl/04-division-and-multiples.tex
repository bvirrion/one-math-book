\section*{Hoofdstuk \ref{ch:g5:division} --- Delen en veelvouden}

\begin{solution}{exo:g5:division:1}
$851 \div 23$: veelvouden van $23$: $23, 46, 69, 92, 115, 138, 161,
184, 207$. Dan $85 \div 23$: $3$ keer ($69$), rest $16$; haal de
$1$ naar beneden: $161 \div 23 = 7$ precies. Quoti\"{e}nt $37$, rest
$0$ ($23 \times 37 = 851$).

$704 \div 32$: $70 \div 32$: $2$ keer ($64$), rest $6$; haal de $4$
naar beneden: $64 \div 32 = 2$. Quoti\"{e}nt $22$, rest $0$.
\end{solution}

\begin{solution}{exo:g5:division:2}
$9 \div 2 = 4.5$; \quad $27 \div 4 = 6.75$; \quad
$33 \div 5 = 6.6$; \quad $21 \div 8 = 2.625$.
\end{solution}

\begin{solution}{exo:g5:division:3}
$54 \div 4 = 13.5$: elk betaalt $13.50$.
\end{solution}

\begin{solution}{exo:g5:division:4}
$10 \div 3 = 3.333\dots$: het cijfer $3$ herhaalt eeuwig --- de
deling stopt nooit. Afgerond op de honderdste: $3.33$.
\end{solution}

\begin{solution}{exo:g5:division:5}
Veelvouden van $7$ tot $70$: $7, 14, 21, 28, 35, 42, 49, 56, 63,
70$. Delers van $18$: $1, 2, 3, 6, 9, 18$. Delers van $24$:
$1, 2, 3, 4, 6, 8, 12, 24$.
\end{solution}

\begin{solution}{exo:g5:division:6}
$470$: door $2$ (eindigt op $0$), door $5$, door $10$; cijfersom
$11$: niet door $3$.

$735$: eindigt op $5$: door $5$, niet door $2$ noch $10$;
cijfersom $15$: door $3$.

$8\,001$: oneven; cijfersom $9$: alleen door $3$.

$1\,314$: even; cijfersom $9$: door $2$ en door $3$ (dus door
$6$), niet door $5$.
\end{solution}

\begin{solution}{exo:g5:division:7}
Deelbaar door $3$ en $5$ betekent deelbaar door $15$: tussen $60$
en $80$, de getallen $60$, $75$.
\end{solution}

\begin{solution}{exo:g5:division:8}
$7.2 \div 6 = 1.2$: elk stuk meet $1.2$ m (controle:
$1.2 \times 6 = 7.2$).
\end{solution}

\begin{solution}{exo:g5:division:9}
$1\,000 = 24 \times 41 + 16$: eenenveertig volle zakjes, $16$
knikkers over.
\end{solution}

\begin{solution}{exo:g5:division:10}
$7 \div 5 = 1.4$: elk van de $5$ vrienden krijgt $1.4$ reep ---
mogelijk door repen in tienden te snijden. Voor $5 \div 7$: de
deling geeft $0.714285\dots$, die nooit stopt --- er bestaat geen
exacte decimale portie; alleen de breuk $\frac57$ noemt hem exact.
\end{solution}

\begin{solution}{exo:g5:division:11}
Cijfersom: $5 + 2 + d = 7 + d$. Deelbaar door $3$ als $7 + d$
$9$, $12$ of $15$ is: $d = 2$, $5$ of $8$. De kleinste is $d = 2$
(geeft $522 = 3 \times 174$).
\end{solution}
