\section*{Hoofdstuk \ref{ch:g5:solids} --- Kubussen en balken}

\begin{solution}{exo:g5:solids:1}
Bijvoorbeeld: een dobbelsteen (\omterm{def:g5:solids:def}{kubus}); een pak ontbijtgranen
(\omterm{def:g5:solids:def}{balk}); een blikje (\omterm{def:g5:solids:def}{cilinder}); een voetbal (bol); bepaalde daken of
de monumenten van Egypte (\omterm{def:g5:solids:def}{piramide}).
\end{solution}

\begin{solution}{exo:g5:solids:2}
Een \omterm{def:g5:solids:def}{balk} heeft $6$ \omterm{def:g5:solids:def}{vlakken}, $12$ \omterm{def:g5:solids:def}{ribben}, $8$ \omterm{def:g5:solids:def}{hoekpunten} --- en
$6 + 8 = 12 + 2$. \checkmark
\end{solution}

\begin{solution}{exo:g5:solids:3}
De zes \omterm{def:g5:solids:def}{vlakken} komen in drie paren van identieke overstaande \omterm{def:g5:solids:def}{vlakken}
(boven--onder, voor--achter, links--rechts). Op een \omterm{def:g5:solids:def}{kubus} zijn alle
zes \omterm{def:g5:solids:def}{vlakken} identieke vierkanten.
\end{solution}

\begin{solution}{exo:g5:solids:4}
Kruisvormig \omterm{def:g5:solids:net}{net}: vier vierkantjes van $2$ cm op een rij, \'{e}\'{e}n
erboven en \'{e}\'{e}n eronder bij het tweede vierkantje. De ring van
vier vouwen maakt de zijkanten; de twee extra vierkantjes sluiten
boven- en onderkant.
\end{solution}

\begin{solution}{exo:g5:solids:5}
Op het kruisnet (vier vierkantjes op een rij, \'{e}\'{e}n erboven,
\'{e}\'{e}n eronder bij het tweede) zijn de overstaande paren: vierkantjes
$1$ en $3$ van de rij; vierkantjes $2$ en $4$ van de rij; de
boven- en onder-vierkantjes. Een correcte labeling: rij $= 1, 2, 6,
5$; boven $= 3$; onder $= 4$ --- dan maken $1{+}6$, $2{+}5$,
$3{+}4$ allemaal $7$.
\end{solution}

\begin{solution}{exo:g5:solids:6}
\omterm{def:g5:solids:def}{Balk}: $5 \times 2 = 10$ per laag, $3$ lagen: $30$ kubusjes. \omterm{def:g5:solids:def}{Kubus}:
$3 \times 3 \times 3 = 27$. L-stapel: $3 \times 2 \times 1 = 6$ plus
$1 \times 2 \times 1 = 2$: $8$ kubusjes.
\end{solution}

\begin{solution}{exo:g5:solids:7}
Lagen van $12$: bijvoorbeeld $4 \times 3$ of $6 \times 2$ (of
$12 \times 1$). Met $3$ lagen is het \omterm{def:g5:solids:volume}{volume} $12 \times 3 = 36$
kubusjes in elk geval.
\end{solution}

\begin{solution}{exo:g5:solids:8}
$27 = 3 \times 3 \times 3$: een \omterm{def:g5:solids:def}{kubus} van zijde $3$.
$64 = 4 \times 4 \times 4$: een \omterm{def:g5:solids:def}{kubus} van zijde $4$.
\end{solution}

\begin{solution}{exo:g5:solids:9}
De doos houdt $10 \times 5 \times 4 = 200$ suikerklontjes. Bij $6$
per dag: $200 = 6 \times 33 + 2$, dus de doos houdt $33$ volle
dagen (en $2$ klontjes blijven over voor de ochtend van dag $34$).
\end{solution}

\begin{solution}{exo:g5:solids:10}
Verf op $3$ \omterm{def:g5:solids:def}{vlakken}: de $8$ hoekkubusjes. Op $2$ \omterm{def:g5:solids:def}{vlakken}: de middens
van de $12$ \omterm{def:g5:solids:def}{ribben}: $12$ kubusjes. Op $1$ \omterm{def:g5:solids:def}{vlak}: de centra van de $6$
\omterm{def:g5:solids:def}{vlakken}: $6$ kubusjes. Op geen: de enkele verborgen middelste
\omterm{def:g5:solids:def}{kubus}: $1$. Controle: $8 + 12 + 6 + 1 = 27$. \checkmark
\end{solution}
