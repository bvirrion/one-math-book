\section*{Hoofdstuk \ref{ch:g5:decimals} --- Decimale getallen}

\begin{solution}{exo:g5:decimals:1}
$5.207$; \quad $0.85$; \quad $4.507$; \quad $12.009$.
\end{solution}

\begin{solution}{exo:g5:decimals:2}
De $4$ telt de tienden, de $8$ de duizendsten:
\[
27.418 = 27 + \frac{4}{10} + \frac{1}{100} + \frac{8}{1000}.
\]
\end{solution}

\begin{solution}{exo:g5:decimals:3}
$3.60 = 3.6$: waar (een eindnul voegt niets toe).

$0.5 = 0.05$: niet waar --- $5$ tienden tegen $5$ honderdsten.

$2.070 = 2.07$: waar.

$1.3 = 1.03$: niet waar --- $3$ tienden tegen $3$ honderdsten.
\end{solution}

\begin{solution}{exo:g5:decimals:4}
$4.8 > 4.79$; \quad $0.302 < 0.32$ (tienden gelijk, honderdsten
$0 < 2$); \quad $6.5 = 6.500$; \quad $9.09 < 9.1$.
\end{solution}

\begin{solution}{exo:g5:decimals:5}
$3.041 < 3.104 < 3.14 < 3.4 < 3.41$.
\end{solution}

\begin{solution}{exo:g5:decimals:6}
$6.83$: tussen $6$ en $7$; tussen $6.8$ en $6.9$.

$0.492$: tussen $0$ en $1$; tussen $0.4$ en $0.5$.

$12.06$: tussen $12$ en $13$; tussen $12.0$ en $12.1$.
\end{solution}

\begin{solution}{exo:g5:decimals:7}
$8.276$: op de eenheid $8$; op de tiende $8.3$; op de honderdste
$8.28$. En $3.95$ op de tiende: het honderdstencijfer is $5$, naar
boven \omterm{def:g4:numbers:round}{afronden}: $4.0$.
\end{solution}

\begin{solution}{exo:g5:decimals:8}
Bijvoorbeeld $2.65$; \quad $0.415$; \quad $5.995$. (Oneindig veel
antwoorden telkens.)
\end{solution}

\begin{solution}{exo:g5:decimals:9}
$0.85 < 0.9 < 1.02 < 1.2$ (in meters).
\end{solution}

\begin{solution}{exo:g5:decimals:10}
Vergelijk eerst de tienden: $7.12$ heeft $1$ tiende, $7.9$ heeft
$9$ tienden, dus $7.12 < 7.9$. Lucie vergeleek de decimale delen als
hele getallen ($12$ vs $9$), en vergat dat $12$ honderdsten veel
minder is dan $90$ honderdsten.
\end{solution}

\begin{solution}{exo:g5:decimals:11}
Afronden op de tiende geeft $6.4$ voor de tweedecimale getallen van
$6.35$ (halfway rondt naar boven) tot $6.44$: de getallen $6.35$,
$6.36$, \dots, $6.44$. Kleinste: $6.35$; grootste: $6.44$.
\end{solution}
