\section*{Hoofdstuk \ref{ch:g5:decimalops} --- Rekenen met decimalen}

\begin{solution}{exo:g5:decimalops:1}
$45.7 + 8.93 = 54.63$; \quad
$16.25 + 3.75 = 20$; \quad
$0.86 + 12.4 = 13.26$.
\end{solution}

\begin{solution}{exo:g5:decimalops:2}
$9.4 - 2.72 = 6.68$ (controle: $6.68 + 2.72 = 9.4$).

$20 - 13.45 = 6.55$ (controle: $6.55 + 13.45 = 20$).

$6.03 - 5.9 = 0.13$.
\end{solution}

\begin{solution}{exo:g5:decimalops:3}
$6.2 \to 10$: $3.8$. \quad $3.75 \to 10$: $6.25$. \quad
$9.99 \to 10$: $0.01$.
\end{solution}

\begin{solution}{exo:g5:decimalops:4}
$72.4$; \quad $58$; \quad $3.45$; \quad $0.07$; \quad $300$.
\end{solution}

\begin{solution}{exo:g5:decimalops:5}
$4.05$ m $= 405$ cm; \quad $325$ cm $= 3.25$ m; \quad
$0.6$ kg $= 600$ g; \quad $85$ cL $= 0.85$ L.
\end{solution}

\begin{solution}{exo:g5:decimalops:6}
$6.2 \times 4$: schatting $24$; precies $24.8$.

$3.45 \times 8$: schatting $28$ ($3.5 \times 8$); precies $27.6$.

$12.5 \times 6$: schatting $75$; precies $75$.
\end{solution}

\begin{solution}{exo:g5:decimalops:7}
$7$ rondjes: $0.4 \times 7 = 2.8$ km. $25$ rondjes:
$0.4 \times 25 = 10$ km.
\end{solution}

\begin{solution}{exo:g5:decimalops:8}
Stokbroden: $1.15 \times 2 = 2.30$. Totaal: $2.30 + 12.60 = 14.90$.
Wisselgeld: $20 - 14.90 = 5.10$.
\end{solution}

\begin{solution}{exo:g5:decimalops:9}
Pak: $1.5 \times 6 = 9$ L. Dagen: $0.75 + 0.75 = 1.5$ L elke twee
dagen, dus $9$ L houdt $12$ dagen (twaalf porties van $0.75$:
$12 \times 0.75 = 9$).
\end{solution}

\begin{solution}{exo:g5:decimalops:10}
Schatting: $5.6$ is meer dan $5.5$, en $5.5 \times 3 = 16.5$ --- dus
het \omterm{def:g2:mult:def}{product} moet \emph{meer} dan $16.5$ zijn, en $15.18$ is te
klein. Correct: $56 \times 3 = 168$, \'{e}\'{e}n decimaal cijfer, dus
$5.6 \times 3 = 16.8$. Milo plakte zijn twee deelproducten ($15$ en
$18$) naast elkaar in plaats van ze met hun plaatsen op te tellen:
$15 + 1.8 = 16.8$.
\end{solution}

\begin{solution}{exo:g5:decimalops:11}
\emph{1.} $65.4 + 64.9 + 65.0 + 66.1 = 261.4$ s.

\emph{2.} $261.4 > 260$: ze heeft het record niet gebroken; ze was
$1.4$ s eroverheen.
\end{solution}
