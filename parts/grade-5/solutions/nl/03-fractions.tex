\section*{Hoofdstuk \ref{ch:g5:fractions} --- Breuken als getallen}

\begin{solution}{exo:g5:fractions:1}
$\frac25$: twee stappen vanaf $0$; $\frac55$: op $1$; $\frac85$:
drie stappen voorbij $1$; $\frac{10}{5}$: op $2$.
\end{solution}

\begin{solution}{exo:g5:fractions:2}
$\frac12 = \frac48$; \quad $\frac34 = \frac68$; \quad
$\frac25 = \frac{4}{10}$; \quad $\frac69 = \frac23$.
\end{solution}

\begin{solution}{exo:g5:fractions:3}
$\frac82 = 4$; \quad $\frac94$ is niet heel ($9$ is geen veelvoud
van $4$); \quad $\frac{15}{3} = 5$; \quad $\frac{20}{5} = 4$.
\end{solution}

\begin{solution}{exo:g5:fractions:4}
$0.7$; \quad $0.41$; \quad $0.5$; \quad $0.75$; \quad $0.2$.
\end{solution}

\begin{solution}{exo:g5:fractions:5}
$0.9 = \frac{9}{10}$; \quad $0.13 = \frac{13}{100}$; \quad
$2.5 = \frac{25}{10}$; \quad $0.75 = \frac{75}{100} = \frac34$.
\end{solution}

\begin{solution}{exo:g5:fractions:6}
$\frac47 < \frac67$ (zelfde noemer). \quad
$\frac35 > \frac37$ (zelfde teller: vijfden zijn groter). \quad
$\frac25 < \frac12 < \frac{7}{12}$, dus $\frac25 < \frac{7}{12}$
(ori\"{e}ntatiepunt $\frac12$). \quad
$\frac98 > 1 > \frac{11}{12}$ (ori\"{e}ntatiepunt $1$).
\end{solution}

\begin{solution}{exo:g5:fractions:7}
Decimalen: $0.5$; $0.4$; $0.75$; $0.6$; $0.2$. Volgorde:
\[
\frac15 < 0.4 < \frac12 < 0.6 < \frac34 .
\]
\end{solution}

\begin{solution}{exo:g5:fractions:8}
Instrument: $\frac14$ van $28 = 7$. Sport: $\frac12$ van $28 = 14$.
De sportgroep is groter.
\end{solution}

\begin{solution}{exo:g5:fractions:9}
$\frac14$ u $= 15$ min; $\frac12$ u $= 30$ min; $\frac34$ u $= 45$
min; $\frac{1}{60}$ u $= 1$ min.
\end{solution}

\begin{solution}{exo:g5:fractions:10}
Tom: $\frac35$ van $50 = 30$ cL. Ana: $\frac12$ van $60 = 30$ cL. Ze
dronken dezelfde hoeveelheid! De kale breuken ($\frac35 > \frac12$)
vergelijken aandelen van \emph{verschillende} flessen --- alleen de
werkelijke hoeveelheden kunnen vergeleken worden.
\end{solution}

\begin{solution}{exo:g5:fractions:11}
Tussen $\frac12$ en $1$: bijvoorbeeld $\frac34$ (d.w.z.\ $0.75$).
Tussen $\frac12 = 0.5$ en $\frac34 = 0.75$: bijvoorbeeld $0.6 =
\frac35$.
\end{solution}
