\chapter{Kubussen en balken}\label{ch:g5:solids}

Het platte blad verlaten: lichamen zijn de vormen van de echte wereld
--- dobbelstenen, pakken ontbijtgranen, blikjes, ballen. Dit
hoofdstuk leert ze te noemen, hun \omterm{def:g5:solids:def}{vlakken} en randen te tellen, ze
uit te vouwen tot netten, en hun \omterm{def:g5:solids:volume}{volume} te meten door kleine
kubusjes te tellen. (Volumeformules komen in \cref{ch:g6:measure}.)

\section{De lichamenfamilie}

\begin{definition}[Lichamen en hun woordenschat]\label{def:g5:solids:def}
Een \emph{lichaam}\index{lichaam} neemt ruimte in. Platte kanten zijn
\emph{vlakken}\index{vlak}; vlakken ontmoeten elkaar langs
\emph{ribben}\index{ribbe}; ribben ontmoeten elkaar in
\emph{hoekpunten}\index{hoekpunt}. De familieportretten:
\begin{itemize}
  \item de \emph{kubus}\index{kubus}: $6$ vierkante vlakken, $12$
        ribben, $8$ hoekpunten;
  \item de \emph{balk}\index{balk} (rechthoekig prisma): $6$
        rechthoekige vlakken, $12$ ribben, $8$ hoekpunten;
  \item de \emph{cilinder}\index{cilinder}: twee schijfvormige vlakken en \'{e}\'{e}n
        opgerold vlak (een blikje);
  \item de \emph{piramide}\index{piramide}: een veelhoekige basis en driehoeken die
        in een top samenkomen;
  \item de \emph{bol}: \'{e}\'{e}n perfect ronde \omterm{def:g4:measure:area}{oppervlakte}, geen
        vlak, geen ribbe.
\end{itemize}
\end{definition}

\begin{example}[Tellen op de kubus]\label{ex:g5:solids:count}
Controleer op een dobbelsteen: $6$ \omterm{def:g5:solids:def}{vlakken} (de getallen $1$ tot
$6$), $8$ \omterm{def:g5:solids:def}{hoekpunten} (de hoeken), $12$ \omterm{def:g5:solids:def}{ribben} (strijk er met een
vinger langs: $4$ boven, $4$ onder, $4$ verticaal). En
$6 + 8 = 12 + 2$ --- hetzelfde curieuze patroon geldt voor de \omterm{def:g5:solids:def}{balk}
en de \omterm{def:g5:solids:def}{piramide} (\cref{ex:g8:solids:count} keert ernaar terug).
\end{example}

\section{Netten}

\begin{definition}[Net]\label{def:g5:solids:net}
Een \emph{net}\index{net} van een \omterm{def:g5:solids:def}{lichaam} is een platte tekening van
al zijn \omterm{def:g5:solids:def}{vlakken}, verbonden langs \omterm{def:g5:solids:def}{ribben}, die opvouwt tot het \omterm{def:g5:solids:def}{lichaam}
--- het \omterm{def:g5:solids:def}{lichaam} uitgepakt als een kartonnen doos platgelegd voor
recycling.
\end{definition}

\begin{example}[Een net lezen]\label{ex:g5:solids:read}
Op een dobbelsteen tellen overstaande \omterm{def:g5:solids:def}{vlakken} op tot $7$. Op een \omterm{def:g5:solids:net}{net}
zijn overstaande \omterm{def:g5:solids:def}{vlakken} die gescheiden door precies \'{e}\'{e}n
vierkantje in een rij (of om een hoek) --- niet de buren! De paren
op het \omterm{def:g5:solids:net}{net} markeren v\'{o}\'{o}r het vouwen is een geweldige oefening in
ruimte plat zien.
\end{example}

\section{Volume: kubusjes tellen}

\begin{definition}[Volume door te tellen]\label{def:g5:solids:volume}
Het \emph{volume}\index{volume} van een \omterm{def:g5:solids:def}{lichaam} gebouwd uit
eenheidskubusjes is het aantal kubusjes dat het bevat. De eenheid
kan de centimeterkubus (cm$^3$) of de meterkubus (m$^3$) zijn.
\end{definition}

\begin{example}[Een balk laag voor laag tellen]\label{ex:g5:solids:layers}
Een \omterm{def:g5:solids:def}{balk} $4$ kubusjes lang, $3$ breed, $2$ hoog: de onderste laag
houdt $4 \times 3 = 12$ kubusjes, en er zijn $2$ lagen:
\[
12 \times 2 = 24 \text{ kubusjes} .
\]
Tellen per laag werkt altijd --- en bewijst stilletjes de formule
$V = L \times b \times h$ van \cref{ch:g6:measure}.
\end{example}

\begin{example}[Zelfde kubusjes, andere vormen]\label{ex:g5:solids:rearrange}
Acht eenheidskubusjes kunnen een $2 \times 2 \times 2$-kubus bouwen,
een $8 \times 1 \times 1$-stok, of een $4 \times 2 \times 1$-plaat:
drie verschillende lichamen, \'{e}\'{e}n \omterm{def:g5:solids:volume}{volume} ($8$ kubusjes). \omterm{def:g5:solids:volume}{Volume}
telt het materiaal, niet de vorm.
\end{example}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g5:solids:1}
Noem een echt voorwerp in de vorm van: een \omterm{def:g5:solids:def}{kubus}; een \omterm{def:g5:solids:def}{balk}; een
\omterm{def:g5:solids:def}{cilinder}; een bol; een \omterm{def:g5:solids:def}{piramide}.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:solids:2}
Hoeveel \omterm{def:g5:solids:def}{vlakken}, \omterm{def:g5:solids:def}{ribben} en \omterm{def:g5:solids:def}{hoekpunten} heeft een \omterm{def:g5:solids:def}{balk}? Controleer
het patroon \omterm{def:g5:solids:def}{vlakken} $+$ \omterm{def:g5:solids:def}{hoekpunten} $=$ \omterm{def:g5:solids:def}{ribben} $+ 2$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:solids:3}
Welke \omterm{def:g5:solids:def}{vlakken} van een \omterm{def:g5:solids:def}{balk} zijn identiek? (Groepeer de zes \omterm{def:g5:solids:def}{vlakken}
in paren.) Wat is speciaal aan de \omterm{def:g5:solids:def}{vlakken} van de \omterm{def:g5:solids:def}{kubus}?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:solids:4}
Teken een \omterm{def:g5:solids:net}{net} van een \omterm{def:g5:solids:def}{kubus} van zijde $2$ cm (gebruik het
kruisvormige model van dit hoofdstuk), knip het uit en vouw het.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:solids:5}
Op het kruisvormige \omterm{def:g5:solids:net}{net} van een dobbelsteen, plaats de getallen $1$
tot $6$ zodat overstaande \omterm{def:g5:solids:def}{vlakken} optellen tot $7$. (Gebruik
\cref{ex:g5:solids:read} om eerst de overstaande paren te vinden.)
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:solids:6}
Tel de eenheidskubusjes: een \omterm{def:g5:solids:def}{balk} $5$ lang, $2$ breed, $3$ hoog;
een \omterm{def:g5:solids:def}{kubus} van zijde $3$; een L-vormige stapel van een plaat
$3 \times 2 \times 1$ met een plaat $1 \times 2 \times 1$
erbovenop.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:solids:7}
Een \omterm{def:g5:solids:def}{balk} houdt precies $12$ eenheidskubusjes in \'{e}\'{e}n laag. Geef
twee mogelijke laagvormen ($L \times b$), en voor elk het \omterm{def:g5:solids:volume}{volume} van
de \omterm{def:g5:solids:def}{balk} als hij $3$ zulke lagen heeft.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:solids:8}
Met $27$ eenheidskubusjes, welke \omterm{def:g5:solids:def}{kubus} kun je bouwen? Met $64$?
(Denk aan het tellen per laag achterstevoren.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g5:solids:9}
Suikerklontjes van zijde $1$ cm komen in een doos $10$ cm lang,
$5$ cm breed, $4$ cm hoog (binnenmaten). Hoeveel suikerklontjes
passen erin? Als een gezin $6$ klontjes per dag gebruikt, voor
hoeveel dagen houdt \'{e}\'{e}n doos (rond zinvol af)?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g5:solids:10}
Een $3 \times 3 \times 3$-kubus wordt aan de buitenkant rood
geschilderd, dan in $27$ eenheidskubusjes gesneden. Hoeveel kleine
kubusjes hebben verf op precies $3$ \omterm{def:g5:solids:def}{vlakken}? Op $2$? Op $1$? Op
geen? (Controle: de vier tellingen moeten optellen tot $27$.)
\end{exercise}
