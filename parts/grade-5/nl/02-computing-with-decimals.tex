\chapter{Rekenen met decimalen}\label{ch:g5:decimalops}

Decimale getallen optellen, aftrekken en vermenigvuldigen bijna als
hele getallen --- de hele kunst is te weten waar de punt komt. Dit
hoofdstuk behandelt de kolommethodes en de magie van vermenigvuldigen
of delen door $10$, $100$, $1000$. (Twee decimalen met elkaar
vermenigvuldigen wacht in \cref{ch:g6:decimals}.)

\section{Optellen en aftrekken}

\begin{method}[Kolommen met een punt]\label{met:g5:decimalops:addsub}
\begin{enumerate}
  \item Schrijf de getallen met de \emph{decimale punten uitgelijnd}
        (dan staan eenheden onder eenheden, tienden onder tienden);
  \item vul het kortere decimale deel aan met nullen;
  \item tel op of trek af als bij hele getallen;
  \item laat de punt van het resultaat recht omlaag vallen.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g5:decimalops:addsub}
$27.5 + 3.86$ (vul aan $27.5 = 27.50$):
\[
\begin{array}{r}
2\,7.5\,0 \\
+\ \ \ 3.8\,6 \\
\hline
3\,1.3\,6
\end{array}
\qquad\qquad
\begin{array}{r}
1\,2.4\,0 \\
-\ \ \ 5.6\,5 \\
\hline
\ \ \,6.7\,5
\end{array}
\]
Voor de aftrekking ($12.4 - 5.65$): vul aan, leen zoals gewoonlijk,
en controleer door terug te optellen: $6.75 + 5.65 = 12.4$.
\end{example}

\begin{example}[Mentale aanvullingen]\label{ex:g5:decimalops:complement}
Wat moet bij $7.6$ om $10$ te bereiken? Klim: $7.6 \to 8$ is $0.4$,
dan $8 \to 10$ is $2$: de aanvulling is $2.4$. Handig voor
wisselgeld: betaal $10$ voor een artikel van $7.60$, krijg $2.40$
terug.
\end{example}

\section{Tien keer groter, tien keer kleiner}

\begin{proposition}[Keer en gedeeld door 10, 100, 1000]\label{prop:g5:decimalops:shift}
Vermenigvuldigen met $10$ maakt elk cijfer tien keer meer waard: de
cijfers schuiven \'{e}\'{e}n plaats naar links --- wat eruitziet alsof de
\emph{punt \'{e}\'{e}n plaats naar rechts beweegt}. Delen doet het
omgekeerde:
\[
3.47 \times 10 = 34.7, \qquad
3.47 \times 100 = 347, \qquad
52.1 \div 10 = 5.21, \qquad
6 \div 100 = 0.06 .
\]
\end{proposition}
\admitted

\begin{example}[Omzetten opnieuw]\label{ex:g5:decimalops:convert}
Meeteenheden liggen machten van tien uit elkaar, dus omzettingen
zijn precies deze verschuivingen: $2.35$ m $= 235$ cm ($\times 100$);
$470$ g $= 0.47$ kg ($\div 1000$); $1.5$ L $= 150$ cL.
\end{example}

\section{Een decimaal vermenigvuldigen met een heel getal}

\begin{method}[Decimaal keer heel]\label{met:g5:decimalops:mult}
Reken alsof er geen punt is, geef het resultaat dan evenveel
decimale cijfers als de decimale factor:
\[
4.35 \times 6: \quad 435 \times 6 = 2\,610
\ \to\ 4.35 \times 6 = 26.10 = 26.1 .
\]
Schat om de punt met vertrouwen te plaatsen: $4.35$ ligt dicht bij
$4$, en $4 \times 6 = 24$ --- dus $26.1$, niet $2.61$ noch $261$.
\end{method}

\begin{example}[Herhaald optellen werkt nog steeds]\label{ex:g5:decimalops:mult}
$0.75 \times 4 = 0.75 + 0.75 + 0.75 + 0.75 = 3$: vier kwarten
maken een geheel, in decimale kleding. Vermenigvuldigen met een heel
getal is nog steeds ``zoveel keer''.
\end{example}

\begin{example}[Boodschappen]\label{ex:g5:decimalops:shopping}
Drie schriften \`{a} $2.45$ en een pen \`{a} $1.20$:
\begin{enumerate}
  \item schriften: $2.45 \times 3 = 7.35$;
  \item totaal: $7.35 + 1.20 = 8.55$;
  \item wisselgeld van $10$: $10 - 8.55 = 1.45$.
\end{enumerate}
\end{example}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g5:decimalops:1}
Reken kolomsgewijs uit: $45.7 + 8.93$; \quad $16.25 + 3.75$; \quad
$0.86 + 12.4$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:decimalops:2}
Reken kolomsgewijs uit en controleer: $9.4 - 2.72$; \quad
$20 - 13.45$; \quad $6.03 - 5.9$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:decimalops:3}
Mentale aanvullingen tot $10$ (zoals in
\cref{ex:g5:decimalops:complement}): $6.2$; \quad $3.75$; \quad
$9.99$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:decimalops:4}
Reken uit zonder kolommen:
\[
7.24 \times 10, \qquad
0.58 \times 100, \qquad
34.5 \div 10, \qquad
7 \div 100, \qquad
0.3 \times 1000 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:decimalops:5}
Zet om: $4.05$ m in cm; \quad $325$ cm in m; \quad $0.6$ kg in g;
\quad $85$ cL in L.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:decimalops:6}
Schat eerst, reken dan: $6.2 \times 4$; \quad
$3.45 \times 8$; \quad $12.5 \times 6$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:decimalops:7}
E\'{e}n rondje van een baan meet $0.4$ km. Hoe lang zijn $7$
rondjes? En $25$ rondjes?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:decimalops:8}
Lea koopt $2$ stokbroden \`{a} $1.15$ en een cake \`{a} $12.60$.
Ze betaalt met een biljet van $20$. Bereken haar totaal en haar
wisselgeld.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:decimalops:9}
Een fles houdt $1.5$ L. Hoeveel liter in een pak van $6$ flessen?
Een gezin drinkt $0.75$ L per dag: voor hoeveel dagen houdt \'{e}\'{e}n
pak? (Hoe vaak past $0.75$ in $9$?)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g5:decimalops:10}
Milo rekent $5.6 \times 3 = 15.18$, redenerend ``$5 \times 3 = 15$
en $6 \times 3 = 18$''. Gebruik een schatting om te tonen dat het
antwoord fout moet zijn, reken dan correct.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g5:decimalops:11}
Vier rondjes van een loper duurden $65.4$ s, $64.9$ s, $65.0$ s en
$66.1$ s.
\begin{enumerate}
  \item Bereken de totale tijd van de vier rondjes.
  \item Het record voor vier rondjes is $260$ s. Heeft ze het
        gebroken? Met hoeveel?
\end{enumerate}
\end{exercise}
