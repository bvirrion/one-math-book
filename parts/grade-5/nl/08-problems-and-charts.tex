\chapter{Problemen en diagrammen}\label{ch:g5:data}

Het laatste hoofdstuk van het jaar zet alles aan het werk:
meertrapsproblemen met geld, hoeveelheden en duren, en de diagrammen
die informatie in \'{e}\'{e}n oogopslag presenteren. Het steridee is
``per stuk'' --- de prijs van \'{e}\'{e}n artikel vinden --- dat zal
uitgroeien tot evenredigheid in \cref{ch:g6:propdata}.

\section{Meertrapsproblemen}

\begin{method}[Een lang probleem temmen]\label{met:g5:data:steps}
\begin{enumerate}
  \item Lees tot het einde; zeg wat de eindvraag is;
  \item plan achteruit: om die te beantwoorden, wat heb ik eerst
        nodig?
  \item los de stappen op volgorde op, schrijf \'{e}\'{e}n bewerking en
        \'{e}\'{e}n korte zin per stap;
  \item controleer het eindantwoord tegen gezond verstand en tegen
        een schatting.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g5:data:multistep}
``Een school bestelt $4$ dozen van $25$ boeken \`{a} $6$ elk, en
betaalt $180$ voor levering erbovenop. Wat is de totale kost?''
\begin{enumerate}
  \item Boeken: $4 \times 25 = 100$ boeken.
  \item Boekenkost: $100 \times 6 = 600$.
  \item Totaal: $600 + 180 = 780$.
\end{enumerate}
Schattingscontrole: ongeveer honderd boeken \`{a} $6$ is ongeveer
$600$, plus ongeveer $200$: ongeveer $800$ --- het antwoord $780$ is
aannemelijk.
\end{example}

\begin{example}[Per stuk]\label{ex:g5:data:perone}
``$5$ identieke mokken kosten $12.50$; wat kosten $8$ mokken?''
Vind eerst de prijs van \emph{\'{e}\'{e}n} mok:
\[
12.50 \div 5 = 2.50,
\qquad\text{dan}\qquad
8 \times 2.50 = 20 .
\]
Via \'{e}\'{e}n wordt alles bereikbaar --- dit tweestappenpatroon lost
een hele familie problemen op (recepten, snelheden, prijzen), en is
het zaad van \cref{ch:g6:propdata}.
\end{example}

\begin{example}[Wisselgeld en budget]\label{ex:g5:data:budget}
Zoe heeft $30$. Ze koopt een boek \`{a} $12.40$ en twee pennen \`{a}
$2.30$ elk. Kan ze ook een bordspel van $13$ betalen?
\begin{enumerate}
  \item Pennen: $2 \times 2.30 = 4.60$.
  \item Tot nu uitgegeven: $12.40 + 4.60 = 17$.
  \item Over: $30 - 17 = 13$ --- precies genoeg voor het spel, met
        niets over.
\end{enumerate}
\end{example}

\section{Diagrammen lezen}

\begin{method}[Elk diagram lezen]\label{met:g5:data:read}
\begin{enumerate}
  \item Lees de titel: wat wordt geteld?
  \item Lees de assen of de legenda: waar staat \'{e}\'{e}n streepje,
        \'{e}\'{e}n staaf, \'{e}\'{e}n symbool voor?
  \item Beantwoord pas daarna vragen --- met een liniaal naar het
        diagram wijzen helpt hoogtes precies af te lezen.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g5:data:chart}
Neerslag in een stad, maand voor maand:

\begin{omfigure}
\begin{tikzpicture}
\begin{axis}[
  width=8.8cm, height=5.0cm,
  ybar, bar width=13pt,
  symbolic x coords={jan,feb,mrt,apr,mei,jun},
  xtick=data,
  ymin=0, ymax=90,
  ytick={20,40,60,80},
  ylabel={regen (mm)},
  tick label style={font=\small},
  label style={font=\small}]
  \addplot[fill=omDef!30, draw=omDef] coordinates
    {(jan,60) (feb,45) (mrt,55) (apr,70) (mei,80) (jun,30)};
\end{axis}
\end{tikzpicture}

{\small Een staafdiagram van neerslag. E\'{e}n streepje $= 20$ mm.}
\end{omfigure}

Aflezingen: de natste maand is mei ($80$ mm); juni kreeg de helft van
aprils regen ($30$ tegen $70$? nee --- de helft van $70$ is $35$, dus
\emph{niet helemaal} de helft: precies lezen doet ertoe!); totaal
voor de zes maanden: $60 + 45 + 55 + 70 + 80 + 30 = 340$ mm.
\end{example}

\begin{example}[Een veranderende hoeveelheid]\label{ex:g5:data:line}
De hoogte van een plant, elke week gemeten:

\begin{omfigure}
\begin{tikzpicture}
\begin{axis}[omaxis,
  width=8.4cm, height=5.0cm,
  xmin=0, xmax=6.5, ymin=0, ymax=24,
  xlabel={week}, ylabel={hoogte (cm)},
  xtick={1,...,6}, ytick={5,10,15,20}]
  \addplot[omDef, thick, mark=*, mark size=1.6pt] coordinates
    {(1,3) (2,6) (3,10) (4,15) (5,19) (6,21)};
\end{axis}
\end{tikzpicture}

{\small De meetpunten verbinden toont de \emph{groei}: hoe steiler
het segment, hoe sneller de plant die week groeide.}
\end{omfigure}

De plant groeide het snelst tussen weken $3$ en $4$ ($+5$ cm) en
vertraagde aan het eind ($+2$ cm in week $6$).
\end{example}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g5:data:1}
Los in stappen op: ``Een club huurt een bus voor $240$ en koopt $32$
museumkaartjes \`{a} $7$ elk. Wat is de totale kost van de
uitstap?''
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:data:2}
Los op met ``per stuk'': ``$3$ kg appels kosten $7.50$. Wat kosten
$5$ kg?''
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:data:3}
Los op: ``Zes vrienden delen even de kosten van $87$ van een picnic.
Hoeveel betaalt elk?''
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:data:4}
Sam heeft $25$. Hij koopt een T-shirt \`{a} $13.90$ en een pet \`{a}
$8.50$. Hoeveel geld heeft hij over?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:data:5}
Met het neerslagdiagram van \cref{ex:g5:data:chart}: welke maanden
kregen meer dan $50$ mm? Hoeveel meer regen viel in mei dan in
juni?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:data:6}
Met het plantdiagram van \cref{ex:g5:data:line}: hoe hoog was de
plant in week $2$? Tussen welke weken groeide ze precies $4$ cm?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:data:7}
Teken een staafdiagram voor het aantal doelpunten van een team:
september $4$, oktober $7$, november $3$, december $6$. Kies de
schaal, en geef het totaal.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:data:8}
Een pak van $8$ yoghurts kost $3.20$; los verkocht kost \'{e}\'{e}n
yoghurt $0.50$. Bereken de ``per stuk''-prijs in het pak, en
hoeveel men per yoghurt bespaart door het pak te kopen.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g5:data:9}
``Een bioscoop verkocht $148$ kaartjes \`{a} $9.50$ op zaterdag en
$95$ kaartjes tegen dezelfde prijs op zondag. Hoeveel meer geld
bracht zaterdag op dan zondag?'' Los het op twee manieren op: beide
dagen omzetten berekenen, of rekenen met het \emph{verschil} in
kaartjes --- en controleer dat de antwoorden kloppen.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g5:data:10}
Een recept voor $4$ personen heeft $300$ g rijst nodig. Aline kookt
voor $10$ personen. Hoeveel rijst heeft ze nodig? (Eerst per
persoon --- decimalen toegestaan.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g5:data:11}
Bedenk een probleem in drie stappen met geld waarvan het
eindantwoord $5.50$ is, schrijf de volledige oplossing in de stijl
van \cref{met:g5:data:steps}, en test het bij een klasgenoot.
\end{exercise}
