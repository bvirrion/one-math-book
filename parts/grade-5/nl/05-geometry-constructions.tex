\chapter{Meetkundige constructies}\label{ch:g5:geometry}

Liniaal, geodriehoek, passer: met deze drie hulpmiddelen en een beetje
methode kan elke in woorden beschreven figuur precies gebouwd worden.
Dit hoofdstuk construeert loodlijnen, evenwijdigen, driehoeken en de
speciale vierhoeken --- eerst ambacht, later theorie
(\cref{ch:g6:lines,ch:g6:shapes}).

\section{Loodlijnen en evenwijdigen door een punt}

\begin{method}[De twee basisconstructies]\label{met:g5:geometry:basic}
\emph{Loodrecht op $d$ door $A$}: schuif de geodriehoek langs $d$ tot
zijn andere recht-hoek-zijde $A$ bereikt; teken; markeer de rechte
hoek.

\emph{Evenwijdig aan $d$ door $A$}: druk een liniaal tegen de
geodriehoek als rail langs $d$; schuif de geodriehoek tot $A$; teken.
(Controle: de tussenruimte tussen de twee lijnen is overal
hetzelfde.)
\end{method}

\begin{example}[Constructies ketenen]\label{ex:g5:geometry:chain}
Veel figuren zijn ketens van deze twee zetten. Een rechthoek $ABCD$
met $AB = 5$ cm en $BC = 3$ cm: teken $[AB]$; loodrecht in $A$;
loodrecht in $B$; markeer $D$ en $C$ op $3$ cm aan dezelfde kant;
verbind. Elke stap is \'{e}\'{e}n basisconstructie.
\end{example}

\section{Driehoeken met de passer}

\begin{method}[Driehoek uit drie zijden]\label{met:g5:geometry:triangle}
Om een driehoek met zijden $6$ cm, $5$ cm, $4$ cm te bouwen:
\begin{enumerate}
  \item teken de langste zijde, $[AB]$ met $AB = 6$ cm;
  \item boog met middelpunt $A$ en straal $5$ cm;
  \item boog met middelpunt $B$ en straal $4$ cm;
  \item hun snijpunt is het derde hoekpunt $C$; verbind.
\end{enumerate}
De passer garandeert de lengtes: elk punt van de eerste boog ligt op
precies $5$ cm van $A$.
\end{method}

\begin{example}[Gelijkbenig en gelijkzijdig gratis]\label{ex:g5:geometry:special}
Zelfde passeropening voor beide bogen $\Rightarrow$ gelijkbenige
driehoek. Opening gelijk aan de basis $\Rightarrow$ gelijkzijdig. De
passer meet niet: hij \emph{kopieert} lengtes, wat nog beter is.
\end{example}

\section{De speciale vierhoeken, door constructie}

\begin{method}[De familie bouwen]\label{met:g5:geometry:quadrilaterals}
\begin{enumerate}
  \item \emph{Rechthoek}: twee paren evenwijdigen die loodrecht
        kruisen --- of de keten van \cref{ex:g5:geometry:chain};
  \item \emph{vierkant}: een rechthoek waarvan de vier zijden
        dezelfde lengte gebruiken --- draag die over met de passer;
  \item \emph{ruit}: vier gelijke zijden --- teken \'{e}\'{e}n zijde, dan
        leggen twee passerbogen vanaf de uiteinden met dezelfde
        opening de andere twee hoekpunten vast (een ruit is twee
        gelijkbenige driehoeken geplakt langs een diagonaal).
\end{enumerate}
Na het bouwen, \emph{controleer} altijd met de hulpmiddelen:
geodriehoek op de hoeken, passer op de zijden.
\end{method}

\begin{example}[Herkennen uit een beschrijving]\label{ex:g5:geometry:recognize}
``Een vierhoek met vier zijden van $4$ cm en \'{e}\'{e}n rechte hoek'' ---
bouw hem: de vier gelijke zijden forceren een ruitvorm, en \'{e}\'{e}n
hoek recht drukken richt alle andere: het resultaat is een
vierkant. Sommige beschrijvingen beschrijven stiekem een specialere
vorm dan ze aankondigen. (Waarom de rechte hoek naar alle vier
hoeken uitspreidt, wordt uitgelegd in \cref{ch:g7:central}.)
\end{example}

\section{Cirkels in constructies}

\begin{example}[De cirkel als hulpmiddel]\label{ex:g5:geometry:circle}
``Zoek alle punten op $3$ cm van $A$'': de cirkel met middelpunt $A$,
straal $3$ cm. ``Zoek alle punten op $3$ cm van $A$ \emph{en}
$4$ cm van $B$'': teken beide cirkels; hun snijpunten (twee, \'{e}\'{e}n
of nul punten) zijn de antwoorden. Schatkaarten en
driehoeksconstructies gebruiken precies dit idee.
\end{example}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g5:geometry:1}
Teken een lijn $d$ en een punt $A$ op ongeveer $3$ cm ervan.
Construeer de loodlijn op $d$ door $A$, dan de evenwijdige aan $d$
door $A$. Markeer de rechte hoeken.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:geometry:2}
Construeer een rechthoek $ABCD$ met $AB = 7$ cm en $BC = 4$ cm,
volgens \cref{ex:g5:geometry:chain}. Controleer de vierde hoek met
de geodriehoek: als de constructie exact is, is hij automatisch
recht.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:geometry:3}
Construeer een driehoek met zijden $7$ cm, $6$ cm en $3$ cm
(\cref{met:g5:geometry:triangle}).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:geometry:4}
Construeer een gelijkbenige driehoek met basis $5$ cm en gelijke
zijden van $7$ cm; dan een gelijkzijdige driehoek van zijde $6$ cm.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:geometry:5}
Construeer een vierkant van zijde $5$ cm. Welke hulpmiddelen
garanderen de rechte hoeken? Welke garanderen de gelijke zijden?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:geometry:6}
Construeer een ruit van zijde $4$ cm die geen vierkant is (kies je
eigen opening voor de eerste hoek). Controleer de vier zijden met de
passer.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:geometry:7}
Teken twee punten $A$ en $B$ met $AB = 5$ cm. Construeer alle
punten die op $4$ cm van $A$ en op $3$ cm van $B$ liggen. Hoeveel
zijn er?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:geometry:8}
Schrijf een constructieprogramma (genummerde stappen, benoemde
punten) voor een rechthoek $6$ cm $\times$ $2$ cm met daarop een
gelijkbenige driehoek met gelijke zijden $4$ cm (een potloodvorm).
Laat een klasgenoot het uitvoeren.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g5:geometry:9}
Probeer een driehoek te construeren met zijden $8$ cm, $3$ cm en
$4$ cm. Wat gebeurt er met de bogen? Leg in \'{e}\'{e}n zin uit waarom
deze driehoek niet kan bestaan.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g5:geometry:10}
Construeer een ruit waarvan de diagonalen $6$ cm en $4$ cm meten,
wetend dat de diagonalen van een ruit elkaar in hun middens
kruisen onder een rechte hoek: teken de diagonalen \emph{eerst},
verbind dan hun vier uiteinden.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g5:geometry:11}
Een geit is vastgebonden aan een paal $A$ met een touw van $4$ m, in
een vlak veld met een recht hek dat op $3$ m van $A$ passeert. Teken
een plattegrond (1 cm per meter): welk gebied kan de geit bereiken?
Waar beperkt het hek haar?
\end{exercise}
