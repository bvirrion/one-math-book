\chapter{Omtrek en oppervlakte}\label{ch:g5:areas}

In jaar~4 telden we rastervierkantjes (\cref{ch:g4:measure}); dit
hoofdstuk verdient de eerste oppervlakte\emph{formule} --- lengte
keer breedte voor de rechthoek --- en leert de eenheden cm$^2$ en
m$^2$. De formules voor driehoeken en andere vormen rijpen in
\cref{ch:g6:measure} en \cref{ch:g7:areas}.

\section{Twee verschillende maten}

\begin{example}[Rand vs vlak]\label{ex:g5:areas:contrast}
Een tuinman heeft \emph{omtrek} nodig om het hek te kopen en
\emph{oppervlakte} om het graszaad te kopen. De twee volgen elkaar
niet: een rechthoek dunner en langer trekken kan de oppervlakte
houden terwijl de omtrek groeit --- vergelijk een $6 \times 2$ en
een $4 \times 3$ rechthoek (zelfde oppervlakte $12$, omtrekken $16$
en $14$).
\end{example}

\section{De rechthoekformule}

\begin{proposition}[Oppervlakte van een rechthoek]\label{prop:g5:areas:rectangle}
Een rechthoek van lengte $L$ en breedte $b$ (in dezelfde eenheid)
heeft oppervlakte
\[
A = L \times b ,
\]
in vierkante eenheden: cm$^2$ (vierkantjes van zijde $1$ cm), m$^2$
(vierkantjes van zijde $1$ m), \dots
\end{proposition}

\begin{proof}[Waarom, door te tellen]
Bedek de rechthoek met eenheidsvierkantjes: $b$ rijen van $L$
vierkantjes elk, dus $L \times b$ vierkantjes in totaal --- precies
de vermenigvuldigingsrechthoek van \cref{ch:g3:mult}.
\end{proof}

\begin{example}\label{ex:g5:areas:rectangle}
Een kleed meet $2.5$ m bij $2$ m: oppervlakte $2.5 \times 2 = 5$
m$^2$. Een postzegel meet $3$ cm bij $2.4$ cm: oppervlakte
$3 \times 2.4 = 7.2$ cm$^2$. Zelfde formule, elke eenheid --- zolang
beide zijden dezelfde gebruiken.
\end{example}

\begin{example}[Vierkante eenheden zetten om met 100]\label{ex:g5:areas:units}
$1$ m $= 100$ cm, maar $1$ m$^2 = 100 \times 100 = 10\,000$ cm$^2$:
een vierkante meter is een $100$-bij-$100$-raster van
centimetervierkantjes. Oppervlakte-eenheden springen met honderden,
niet tientallen.
\end{example}

\section{Samengestelde figuren}

\begin{method}[Snijden, optellen, aftrekken]\label{met:g5:areas:composite}
Voor een figuur van rechthoeken (een L, een T, een kader):
\begin{enumerate}
  \item snijd hem in rechthoeken, of maak hem compleet tot een grote
        rechthoek;
  \item bereken elke rechthoekoppervlakte met de formule;
  \item tel de stukken op --- of trek het gat af van de grote
        rechthoek;
  \item controleer met een ruwe telling van rastervierkantjes.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g5:areas:composite}
Een L-vormige kamer: een rechthoek $6$ m $\times$ $4$ m met een
hoek $2$ m $\times$ $2$ m ontbrekend.
\[
\text{Oppervlakte} = 6 \times 4 - 2 \times 2 = 24 - 4 = 20 \text{ m}^2 .
\]
Of snijd de L in een strook $6 \times 2$ en een strook $4 \times 2$:
$12 + 8 = 20$ m$^2$ --- twee wegen, \'{e}\'{e}n antwoord.
\end{example}

\begin{example}[Helft van een rechthoek]\label{ex:g5:areas:halfrect}
Een rechthoek langs een diagonaal snijden geeft twee driehoeken van
dezelfde oppervlakte: elk is de \emph{helft} van de rechthoek. Een
rechthoekige driehoek met benen $6$ cm en $4$ cm heeft dus
oppervlakte $\frac{6 \times 4}{2} = 12$ cm$^2$ --- een plaatje om
te onthouden voor \cref{ch:g6:measure}, waar het een formule
wordt.
\end{example}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g5:areas:1}
Bereken de oppervlakte en de omtrek van een rechthoek $8$ cm
$\times$ $5$ cm. Welk antwoord is in cm, welk in cm$^2$?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:areas:2}
Bereken de oppervlaktes: een vierkant van zijde $9$ cm; een
rechthoek $12$ m $\times$ $7$ m; een rechthoek $4.5$ cm $\times$
$6$ cm.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:areas:3}
Een rechthoek heeft oppervlakte $63$ cm$^2$ en lengte $9$ cm. Vind
de breedte, dan de omtrek.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:areas:4}
Teken twee verschillende rechthoeken met oppervlakte $24$
vierkantjes op ruitjespapier, en bereken beide omtrekken. Zelfde
oppervlakte --- zelfde omtrek?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:areas:5}
Zet om: $3$ m$^2$ in cm$^2$; \quad $50\,000$ cm$^2$ in m$^2$.
(Onthou \cref{ex:g5:areas:units}: met honderden!)
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:areas:6}
Een T-vormige figuur is gemaakt van een horizontale balk $8 \times 2$
bovenop een verticale balk $2 \times 5$ (in cm). Bereken de
oppervlakte door twee rechthoeken op te tellen.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:areas:7}
Een fotolijst: een rechthoek $30$ cm $\times$ $20$ cm met een
rechthoekig venster $24$ cm $\times$ $14$ cm uitgesneden. Welke
oppervlakte hout gebruikt de lijst?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:areas:8}
Bereken de oppervlakte van een rechthoekige driehoek met benen
$8$ cm en $6$ cm (helft van een rechthoek,
\cref{ex:g5:areas:halfrect}).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:areas:9}
Een rechthoekig groentebed meet $7$ m bij $4$ m. Zaad kost $2$ per
vierkante meter, en hek $3$ per meter. Bereken de kosten van het
zaad, dan de kosten van het hek.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g5:areas:10}
Een gangvloer is $12$ m lang en $2$ m breed, en moet bedekt worden
met vierkante tegels van zijde $50$ cm. Hoeveel tegels zijn nodig?
(Zet eerst om, of tel tegels in elke richting.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g5:areas:11}
Verdubbel de zijden van een rechthoek $4$ cm $\times$ $3$ cm. Wat
gebeurt er met de omtrek? Met de oppervlakte? (Bereken beide v\'{o}\'{o}r
en na --- de twee antwoorden verschillen!)
\end{exercise}
