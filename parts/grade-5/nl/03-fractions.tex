\chapter{Breuken als getallen}\label{ch:g5:fractions}

Een \omterm{def:g4:fractions:def}{breuk} is niet alleen een taartplaatje: het is een echt
\emph{getal}, met een plaats op de getallenlijn --- \omterm{def:g1:addition:def}{soms} precies
dezelfde plaats als een decimaal getal. Dit hoofdstuk verbindt de
twee werelden. (De volle regels van breukrekenen komen in
\cref{ch:g6:fractions} en verder.)

\section{Breuken op de lijn, opnieuw}

\begin{example}[Plaatsen en lezen]\label{ex:g5:fractions:place}
Op een lijn in vijfden gesneden is $\frac{7}{5}$ $7$ stappen vanaf
$0$: voorbij $1$ (dat is $\frac55$), bij $1 + \frac25$. Elke \omterm{def:g4:fractions:def}{breuk}
leeft ergens op de lijn; grotere \omterm{def:g4:fractions:def}{teller} (zelfde \omterm{def:g4:fractions:def}{noemer}) betekent
verder naar rechts.
\end{example}

\begin{proposition}[Zelfde punt, veel namen]\label{prop:g5:fractions:equal}
Elk deel in twee (of drie\,\dots) snijden verplaatst het punt niet:
\[
\frac{1}{2} = \frac{2}{4} = \frac{3}{6} = \frac{5}{10},
\qquad
\frac{2}{3} = \frac{4}{6} = \frac{20}{30}.
\]
\omterm{def:g4:fractions:def}{Teller} en \omterm{def:g4:fractions:def}{noemer} met hetzelfde getal vermenigvuldigen (of delen)
geeft een andere naam van dezelfde \omterm{def:g4:fractions:def}{breuk}.
\end{proposition}
\admitted

\section{Breuken en decimalen}

\begin{proposition}[Breuken met decimale namen]\label{prop:g5:fractions:decimal}
\omterm{def:g4:fractions:def}{Breuken} van tienden, honderdsten, duizendsten \emph{zijn} decimale
getallen:
\[
\frac{3}{10} = 0.3, \qquad \frac{27}{100} = 0.27 .
\]
Sommige andere \omterm{def:g4:fractions:def}{breuken} hebben ook decimale namen, gevonden door naar
tienden of honderdsten over te stappen:
\[
\frac{1}{2} = \frac{5}{10} = 0.5, \qquad
\frac{1}{4} = \frac{25}{100} = 0.25, \qquad
\frac{3}{4} = 0.75, \qquad
\frac{1}{5} = \frac{2}{10} = 0.2 .
\]
Maar niet alle: $\frac13 = 0.333\dots$ stopt nooit --- de \omterm{def:g4:fractions:def}{breuk} is
haar enige exacte naam.
\end{proposition}
\admitted

\begin{example}[De handigere naam kiezen]\label{ex:g5:fractions:choose}
Om driekwart van $12$ te betalen: de breuknaam is handiger,
$\frac34$ van $12$ is $9$. Om $\frac34$ en $0.8$ te vergelijken: de
decimale naam is handiger, $0.75 < 0.8$. Tweetalig zijn loont.
\end{example}

\section{Breuken vergelijken}

\begin{method}[Drie vergelijkingsgereedschappen]\label{met:g5:fractions:compare}
\begin{enumerate}
  \item \emph{Zelfde \omterm{def:g4:fractions:def}{noemer}}: meer delen wint,
        $\frac57 > \frac37$;
  \item \emph{zelfde \omterm{def:g4:fractions:def}{teller}}: grotere delen winnen, dus
        \emph{kleinere} \omterm{def:g4:fractions:def}{noemer} wint: $\frac35 > \frac38$ (vijfden
        zijn groter dan achtsten);
  \item \emph{ori\"{e}ntatiepunten}: vergelijk elk met $\frac12$ of
        met $1$: $\frac38 < \frac12 < \frac59$, dus
        $\frac38 < \frac59$.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g5:fractions:compare}
Wie at meer pizza: Ali met $\frac58$ of Bea met $\frac54$? Bea's
\omterm{def:g4:fractions:def}{breuk} is groter dan $1$ (vijf \omterm{def:g3:division:half}{kwarten} is meer dan \'{e}\'{e}n hele pizza),
Ali's is kleiner dan $1$: Bea at meer --- meer dan een hele pizza
zelfs.
\end{example}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g5:fractions:1}
Teken een getallenlijn in vijfden van $0$ tot $2$, en plaats:
$\frac25$; \ $\frac55$; \ $\frac85$; \ $\frac{10}{5}$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:fractions:2}
Vul de gelijke namen aan:
$\frac12 = \frac{?}{8}$; \quad
$\frac34 = \frac{?}{8}$; \quad
$\frac{2}{5} = \frac{4}{?}$; \quad
$\frac{6}{9} = \frac{2}{?}$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:fractions:3}
Welke \omterm{def:g4:fractions:def}{breuken} noemen hele getallen? Geef het hele getal als dat zo
is: $\frac82$; \quad $\frac{9}{4}$; \quad $\frac{15}{3}$; \quad
$\frac{20}{5}$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:fractions:4}
Schrijf als decimalen: $\frac{7}{10}$; \quad $\frac{41}{100}$; \quad
$\frac12$; \quad $\frac34$; \quad $\frac15$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:fractions:5}
Schrijf als \omterm{def:g4:fractions:def}{breuken} (tienden of honderdsten): $0.9$; \quad $0.13$;
\quad $2.5$; \quad $0.75$ (geef dan zijn kwartnaam).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:fractions:6}
Vergelijk, noem het gebruikte gereedschap
(\cref{met:g5:fractions:compare}):
\[
\frac47 \;?\; \frac67, \qquad
\frac35 \;?\; \frac37, \qquad
\frac25 \;?\; \frac{7}{12}, \qquad
\frac98 \;?\; \frac{11}{12} .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:fractions:7}
Zet op volgorde met de decimale namen: $\frac12$; \ $0.4$; \
$\frac34$; \ $0.6$; \ $\frac15$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:fractions:8}
Een \omterm{def:g3:division:half}{kwart} van de leerlingen van een klas van $28$ speelt een
instrument, en de \omterm{def:g3:division:half}{helft} speelt een sport. Hoeveel leerlingen is elk?
Welke groep is groter?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:fractions:9}
Koppel elke \omterm{def:g4:fractions:def}{breuk} aan een tijd: $\frac14$ u, $\frac12$ u,
$\frac34$ u, $\frac{1}{60}$ u --- aan $30$ min, $45$ min, $1$ min,
$15$ min.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g5:fractions:10}
Tom dronk $\frac35$ van zijn fles van $50$ cL; Ana dronk $\frac12$
van haar fles van $60$ cL. Wie dronk meer? (Bereken beide
hoeveelheden in cL --- de kale \omterm{def:g4:fractions:def}{breuken} vergelijken is hier niet
genoeg, en zeg waarom.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g5:fractions:11}
Geef een \omterm{def:g4:fractions:def}{breuk} strikt tussen $\frac12$ en $1$; dan \'{e}\'{e}n strikt
tussen $\frac12$ en $\frac34$. (Decimale namen kunnen helpen.)
\end{exercise}
