\chapter{Delen en veelvouden}\label{ch:g5:division}

Staartdelen geleerd in \cref{ch:g4:division} krijgt twee upgrades:
tweecijferige \omterm{def:g5:division:multiple}{delers}, en \omterm{def:g3:division:remainder}{quoti\"enten} die voorbij de punt doorgaan
--- $3 \div 4$ krijgt eindelijk een antwoord, $0.75$. Het hoofdstuk
eindigt met \omterm{def:g5:division:multiple}{veelvouden}, \omterm{def:g5:division:multiple}{delers} en de eerste deelbaarheidssnelwegen.

\section{Delen door een tweecijferig getal}

\begin{method}[Tweecijferige delers]\label{met:g5:division:twodigits}
Zelfde methode als eerder, met \'{e}\'{e}n extra vaardigheid: raden hoe
vaak de \omterm{def:g5:division:multiple}{deler} past. Schrijf eerst een kleine tabel van \omterm{def:g5:division:multiple}{veelvouden}
van de \omterm{def:g5:division:multiple}{deler}. Voor $986 \div 23$ (\omterm{def:g5:division:multiple}{veelvouden} van 23: 23, 46, 69, 92,
115, 138, 161, 184, 207):
\begin{enumerate}
  \item $98$ tientallen: $23 \times 4 = 92$ past, $23 \times 5 = 115$
        niet: cijfer $4$, \omterm{def:g3:division:remainder}{rest} $6$;
  \item haal de $6$ naar beneden: $66$: $23 \times 2 = 46$ past:
        cijfer $2$, \omterm{def:g3:division:remainder}{rest} $20$;
  \item \omterm{def:g3:division:remainder}{quoti\"ent} $42$, \omterm{def:g3:division:remainder}{rest} $20$. Controle:
        $23 \times 42 + 20 = 966 + 20 = 986$.
\end{enumerate}
\end{method}

\section{Decimale quoti\"enten}

\begin{example}[De deling die weigert te stoppen bij de rest]\label{ex:g5:division:decimal}
Verdeel $3$ pizza's onder $4$ kinderen: $3 \div 4$ heeft
\omterm{def:g3:division:remainder}{quoti\"ent} $0$ en \omterm{def:g3:division:remainder}{rest} $3$ --- nutteloos! Ga door met de deling
voorbij de punt:
\begin{enumerate}
  \item $3$ eenheden $= 30$ tienden; $30 \div 4$: $7$ tienden, \omterm{def:g3:division:remainder}{rest}
        $2$ tienden;
  \item $2$ tienden $= 20$ honderdsten; $20 \div 4 = 5$
        honderdsten, \omterm{def:g3:division:remainder}{rest} $0$.
\end{enumerate}
Dus $3 \div 4 = 0.75$: elk kind krijgt $0.75$ pizza --- drie
\omterm{def:g3:division:half}{kwarten}, zoals de breukentaal al wist
(\cref{prop:g5:fractions:decimal}).
\end{example}

\begin{method}[Een deling voorbij de punt voortzetten]\label{met:g5:division:continue}
Als de eenheden op zijn en er een \omterm{def:g3:division:remainder}{rest} overblijft:
\begin{enumerate}
  \item schrijf de \omterm{def:g4:decimals:point}{decimale punt} in het \omterm{def:g3:division:remainder}{quoti\"ent};
  \item voeg een \omterm{def:g1:counting:zero}{nul} toe aan de \omterm{def:g3:division:remainder}{rest} (eenheden worden tienden,
        tienden worden honderdsten\,\dots) en blijf delen;
  \item stop als de \omterm{def:g3:division:remainder}{rest} $0$ is --- of als de cijfers eeuwig
        herhalen, zoals $1 \div 3 = 0.333\dots$: geef dan een
        afgeronde waarde.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g5:division:continue}
$27 \div 6$: \omterm{def:g3:division:remainder}{quoti\"ent} $4$, \omterm{def:g3:division:remainder}{rest} $3$; punt; $30$ tienden
$\div\ 6 = 5$: dus $27 \div 6 = 4.5$. \quad
$22 \div 8$: $2$, \omterm{def:g3:division:remainder}{rest} $6$; dan $60 \div 8 = 7$ \omterm{def:g3:division:remainder}{rest} $4$;
dan $40 \div 8 = 5$: dus $22 \div 8 = 2.75$.
\end{example}

\section{Veelvouden en delers}

\begin{definition}[Veelvoud, deler]\label{def:g5:division:multiple}
De \emph{veelvouden}\index{veelvoud} van $6$ zijn zijn tafel voor
eeuwig doorgezet: $0, 6, 12, 18, 24, \dots$ Als $24$ een veelvoud
van $6$ is, zegt men ook dat $6$ een \emph{deler}\index{deler} van
$24$ is, of dat $24$ \emph{deelbaar} is door $6$: de deling
$24 \div 6$ laat geen \omterm{def:g3:division:remainder}{rest}.
\end{definition}

\begin{proposition}[Deelbaarheidssnelwegen]\label{prop:g5:division:criteria}
Zonder te delen is een getal deelbaar:
\begin{itemize}
  \item door $2$ als het laatste cijfer $0$, $2$, $4$, $6$ of $8$ is
        (de \omterm{def:g3:numbers:evenodd}{even} getallen);
  \item door $5$ als het laatste cijfer $0$ of $5$ is;
  \item door $10$ als het laatste cijfer $0$ is;
  \item door $3$ als de \emph{\omterm{def:g1:addition:def}{som} van zijn cijfers} deelbaar is door
        $3$ (bijv.\ $741$: $7 + 4 + 1 = 12$, deelbaar --- en
        inderdaad $741 = 3 \times 247$).
\end{itemize}
\end{proposition}
\admitted

\begin{example}\label{ex:g5:division:criteria}
Is $534$ deelbaar door $2$? Eindigt op $4$: ja. Door $5$? Nee. Door
$3$? $5 + 3 + 4 = 12$: ja. Dus $534$ is ook deelbaar door $6$ (door
$2$ \emph{en} door $3$) --- controle: $534 = 6 \times 89$.
\end{example}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g5:division:1}
Reken de staartdelingen uit (eerst tabel van \omterm{def:g5:division:multiple}{veelvouden}):
$851 \div 23$; \quad $704 \div 32$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:division:2}
Reken de exacte decimale \omterm{def:g3:division:remainder}{quoti\"enten} uit: $9 \div 2$; \quad
$27 \div 4$; \quad $33 \div 5$; \quad $21 \div 8$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:division:3}
Vier vrienden delen een restaurantrekening van $54$ even. Hoeveel
betaalt elk? (Ga door voorbij de punt.)
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:division:4}
Reken $10 \div 3$ uit, ga drie cijfers voorbij de punt door. Wat
val je op? Geef de waarde afgerond op de honderdste.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:division:5}
Noem: de \omterm{def:g5:division:multiple}{veelvouden} van $7$ tot $70$; de \omterm{def:g5:division:multiple}{delers} van $18$; de \omterm{def:g5:division:multiple}{delers}
van $24$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:division:6}
Deelbaar door $2$? door $5$? door $10$? door $3$? Test elke
snelweg op: $470$; \quad $735$; \quad $8\,001$; \quad $1\,314$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:division:7}
Zoek alle getallen tussen $60$ en $80$ die deelbaar zijn door $3$
\emph{en} door $5$. (Welke enkele deelbaarheid combineert dat?)
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:division:8}
Een lint van $7.2$ m wordt in $6$ gelijke stukken geknipt. Hoe lang
is elk stuk? (Deel een decimaal door een heel getal: verdeel de
meters, dan de tienden.)
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g5:division:9}
$1\,000$ knikkers worden in zakjes van $24$ gepakt. Hoeveel volle
zakjes, en hoeveel knikkers blijven over? (Tweecijferige \omterm{def:g5:division:multiple}{deler}.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g5:division:10}
Kunnen $5$ vrienden $7$ chocoladerepen even delen, zonder iets over?
Geef elk aandeel als decimaal. Zelfde vraag voor $7$ vrienden die
$5$ repen delen --- wat gaat anders?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g5:division:11}
Met de cijfersomsnelweg, vind het kleinste cijfer $d$ dat $52d$ (een
driedijferig getal eindigend op $d$) deelbaar door $3$ maakt. Zoek
dan alle zulke cijfers $d$.
\end{exercise}
