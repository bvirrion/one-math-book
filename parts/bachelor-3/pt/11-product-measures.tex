\chapter{Medidas de Produto, Fubini, Mudança de Variáveis}
\label{ch:b3:product}

A teoria unidimensional de Lebesgue torna-se multidimensional
cálculo através de dois teoremas. \emph{Tonelli--Fubini} diz que
integrais sobre produtos são integrais iteradas --- fatiar é
legítimo, em qualquer ordem, sob hipóteses pode-se realmente
verifique. O \emph{fórmula de mudança de variáveis} transportes
integrais ao longo $\mathcal C^1$ difeomorfismos, com o
Determinante Jacobiano como taxa de câmbio por volume; nós provamos
completamente, a partir do caso linear onde explica
qual é o determinante \emph{é}. Cascata de aplicativos: a camada
fórmula de bolo, convolução, \omterm{ex:b3:product:polar}{coordenadas polares}, o volume do
$ n $-ball --- e, no problema do fim de semana, a fórmula de Stirling
com uma análise de erro honesta.

\section{Produto $\sigma $-álgebras e medidas de produto}

\begin{definition}\label{def:b3:product:sigma}
Para \omterm{def:b3:lebesgue:measurable}{mensurável} espaços $(X, \mathcal A)$, $(Y, \mathcal B)$, o
\emph{produto $\sigma $-álgebra}\index{produto $\sigma $-álgebra} $\mathcal A \otimes \mathcal B $ sobre $ X \times
Y $ é gerado pelo \emph{retângulos} $ A \times B $ ($ A \in
\mathcal A $, $ B \in \mathcal B $) --- um $\pi $-sistema. Para $ E
\subseteq X\times Y $ e $ x \in X $, o \emph{seção} é $ E_x
= \{y : (x,y) \in E\}$; para uma função $ f $ no produto,
$ f_x = f(x, \cdot)$.
\end{definition}

\begin{proposition}\label{prop:b3:product:sections}
(a) Se $ E \in \mathcal A\otimes\mathcal B $, cada seção $ E_x
\in \mathcal B $ (e simetricamente); se $ f $ é $\mathcal
A\otimes\mathcal B $- \omterm{def:b3:lebesgue:measurable}{mensurável}, cada $ f_x $ é $\mathcal
B $-mensurável.
(b) $\mathcal B(\R^m)\otimes\mathcal B(\R^n) = \mathcal
B(\R^{m+n})$.
\end{proposition}

\begin{proof}
(a) Bons conjuntos: $\{E : E_x \in \mathcal B\ \forall x\}$ é um
$\sigma $-álgebra (seções comutam com complementos e
uniões contáveis) contendo os retângulos. Para $ f $:
$(f_x)^{-1}(B) = (f^{-1}(B))_x $.
(b) ($\subseteq $) Retângulos de conjuntos de Borel: basta que
aberto $\times $ caixas abertas são do Borel $\R^{m+n}$ (eles estão abertos)
e que os retângulos gerais do Borel são limites --- bons conjuntos
novamente: $\{A : A\times\R^n \in \mathcal B(\R^{m+n})\}$ é um
$\sigma $-álgebra contendo as aberturas; cruze dois deles.
($\supseteq $) Todo conjunto aberto de $\R^{m+n}$ é uma união contável
de caixas abertas racionais $ U\times V $: contido no produto
$\sigma $-álgebra.
\end{proof}

\begin{theorem}[Medida do produto]\label{thm:b3:product:existence}
Deixar $(X, \mathcal A, \mu)$ e $(Y, \mathcal B, \nu)$ ser
\emph{$\sigma $-finito}. Para cada $ E \in \mathcal
A\otimes\mathcal B $, a função $ x \mapsto \nu(E_x)$ é
\omterm{def:b3:lebesgue:measurable}{mensurável}, e\index{medida do produto}
\[
(\mu\otimes\nu)(E) = \int_X \nu(E_x)\,\dd\mu(x)
\]
define o único \omterm{def:b3:measure:measure}{medir} sobre $\mathcal A\otimes\mathcal B $
com $(\mu\otimes\nu)(A\times B) = \mu(A)\nu(B)$. Isso é
$\sigma $-finito e simétrico: o mesmo \omterm{def:b3:measure:measure}{medir} é obtido por
integrando $ x $-seções contra $\nu $.
\end{theorem}

\begin{proof}
\emph{Mensurabilidade de $ x \mapsto \nu(E_x)$.} Primeiro deixe $\nu $
seja finito. A aula $\mathcal D $ de $ E $ para o qual o mapa é
\omterm{def:b3:lebesgue:measurable}{mensurável} contém os retângulos ($\nu((A\times B)_x) =
\nu(B)\mathbf 1_A(x)$) e é um $\lambda $-sistema: para $ E
\subseteq F $ em $\mathcal D $, $\nu((F\setminus E)_x) =
\nu(F_x) - \nu(E_x)$ (finitude); para $ E_n \uparrow E $,
$\nu((E_n)_x) \uparrow \nu(E_x)$ (continuidade de baixo) e
limites monótonos de funções mensuráveis são \omterm{def:b3:lebesgue:measurable}{mensurável}.
Os retângulos formam um $\pi $-sistema: Dynkin
(\cref{thm:b3:measure:dynkin}) dá $\mathcal D = \mathcal
A\otimes\mathcal B $. Se $\nu $ é $\sigma $-finito, escreva $ Y =
\bigcup Y_k $, $ Y_k \uparrow $, $\nu(Y_k) < \infty $: $\nu(E_x) =
\lim_k\nu_k(E_x)$ com $\nu_k = \nu(\cdot\cap Y_k)$ finito.

\emph{\omterm{def:b3:measure:measure}{Medir}.} $\sigma $-aditividade de $ E \mapsto
\int\nu(E_x)\dd\mu $ segue de
\cref{cor:b3:lebesgue:additivity} (seções de conjuntos disjuntos
são disjuntos). Em retângulos dá $\mu(A)\nu(B)$.
\emph{Singularidade}: dois candidatos concordam com a $\pi $-sistema de
retângulos; $\sigma $-finitude fornece retângulos $ X_k\times
Y_k \uparrow X\times Y $ de finito \omterm{def:b3:measure:measure}{medir}:
\cref{thm:b3:measure:uniqueness}. Simetria: o
construção de outra ordem também é uma \omterm{def:b3:measure:measure}{medir} concordando em
retângulos --- únicos, portanto iguais.
\end{proof}

\begin{definition}\label{def:b3:product:lebesgue}
\emph{Medida de Lebesgue em $\R^d $}\index{Medida de Lebesgue em $\R^d $} é $\lambda_d = \lambda\otimes\cdots\otimes\lambda $
($ d $ fatores; a associatividade da construção é verificada em
caixas e propagadas por exclusividade). É o Borel único
\omterm{def:b3:measure:measure}{medir} dando cada caixa $\prod\intoc{a_i}{b_i}$ seu volume
$\prod(b_i - a_i)$; é invariável à tradução (traduz
concordar com as caixas), $\sigma $-finito, e completo depois
Carathéodoro \omterm{thm:b3:complete:completion}{conclusão} --- nós escrevemos $\lambda_d $ para o
concluído \omterm{def:b3:measure:measure}{medir} e integre adequadamente.
\end{definition}

\section{Tonelli e Fubini}

\begin{theorem}[Tonelli]\label{thm:b3:product:tonelli}
$\mu, \nu $ $\sigma $-finito, $ f \colon X\times Y \to [0,
+\infty]$ \omterm{def:b3:lebesgue:measurable}{mensurável}. Então $ x \mapsto \int_Y f_x\,\dd\nu $ é
\omterm{def:b3:lebesgue:measurable}{mensurável} e\index{Teorema de Tonelli}
\[
\int_{X\times Y}f\,\dd(\mu\otimes\nu)
= \int_X\Bigl(\int_Y f(x,y)\,\dd\nu(y)\Bigr)\dd\mu(x)
= \int_Y\Bigl(\int_X f(x,y)\,\dd\mu(x)\Bigr)\dd\nu(y).
\]
\end{theorem}

\begin{proof}
A máquina padrão. Para $ f = \mathbf 1_E $ isso é
\cref{thm:b3:product:existence} (e sua forma simétrica). Por
linearidade que vale para \omterm{def:b3:groups:simple}{simples} $ f \geq 0$. Para geral $ f \geq
0$: pegue \omterm{def:b3:groups:simple}{simples} $ s_n \nearrow f $
(\cref{thm:b3:lebesgue:approximation}); então $\int_Y(s_n)_x
\dd\nu \nearrow \int_Y f_x\dd\nu $ para cada $ x $ (MCT em $ Y $),
então os membros da esquerda convergem por MCT em $ X $, enquanto
$\int s_n\,\dd(\mu\otimes\nu) \nearrow \int f $ pelo MCT no
produto.
\end{proof}

\begin{theorem}[Fubini]\label{thm:b3:product:fubini}
$\mu, \nu $ $\sigma $-finito, $ f \in L^1(\mu\otimes\nu)$. Então
para $\mu $-a.e.\ $ x $ a seção $ f_x $ é $\nu $-integrável, o
função definida por a.e. $ x \mapsto \int f_x\dd\nu $ é
\omterm{def:b3:lebesgue:l1}{integrável}, e as duas integrais iteradas são iguais
$\int f\,\dd(\mu\otimes\nu)$.\index{Teorema de Fubini}
\end{theorem}

\begin{proof}
Tonelli se candidatou a $\abs f $ mostra $\varphi(x) = \int\abs{f_x}
\dd\nu $ tem integral finita, portanto é finito a.e.: $ f_x \in
L^1(\nu)$ para a.e.\ $ x $. Dividir $ f = f^+ - f^-$ (caso real;
complexo por componentes): Tonelli calcula cada iteração
integral de $ f^\pm $ como $\int f^\pm\dd(\mu\otimes\nu) <
\infty $, e a diferença definida por a.e. integra-se ao
diferença. Simetricamente para a outra ordem.
\end{proof}

\begin{method}\label{met:b3:product:fubini}
Para trocar duas integrais (ou uma integral e uma soma, ou duas
somas): se o integrando for \emph{não negativo}, intercâmbio
livremente (Tonelli). Caso contrário, aplique primeiro Tonelli $\abs f $
na ordem que for mais fácil de estimar; se o resultado for
finito, Fubini legitima o intercâmbio. Nunca pule o
$\abs f $ verificar: \cref{exo:b3:product:4}O integrando de tem dois
integrais iteradas com \emph{valores diferentes}.
\end{method}

\begin{proposition}[Bolo de camadas]\label{prop:b3:product:layercake}
Para $ f \geq 0$ \omterm{def:b3:lebesgue:measurable}{mensurável} sobre $(X, \mathcal A, \mu)$
$\sigma $-finito:\index{fórmula de bolo de camadas}
\[
\int_X f\,\dd\mu = \int_0^{+\infty}\mu(\{f > t\})\,\dd t,
\qquad
\int_X f^p\,\dd\mu = p\int_0^{+\infty}t^{p-1}\mu(\{f >
t\})\,\dd t \quad (p \geq 1).
\]
\end{proposition}

\begin{proof}
Aplique Tonelli em $\mathbf 1_{\{(x,t) : 0 < t < f(x)\}}$ sobre $ X
\times \intoo0{+\infty}$ (\omterm{def:b3:lebesgue:measurable}{mensurável}: isso é $\{(x,t): f(x) - t
> 0\}\cap\{t > 0\}$, uma combinação do tipo Borel do
\omterm{def:b3:lebesgue:measurable}{mensurável} $(x,t)\mapsto f(x) - t $): integrando em $ t $ primeiro
dá $\int f\,\dd\mu $; em $ x $ primeiro, $\int_0^\infty\mu(f >
t)\dd t $. Para $ f^p $: substituto $ t = s^p $ em $\int\mu(f^p > t)
\dd t $, ou seja, aplique a primeira fórmula a $ f^p $ e mudar
variáveis na integral unidimensional ($\{f^p > s^p\} =
\{f > s\}$).
\end{proof}

\begin{theorem}[Convolução ativada $ L^1$]\label{thm:b3:product:convolution}
Para $ f, g \in L^1(\R^d, \lambda_d)$, a integral
\[
(f * g)(x) = \int_{\R^d} f(x - y)\,g(y)\,\dd y
\]
converge absolutamente para a.e.\ $ x $, define $ f * g \in
L^1(\R^d)$ com $\norm{f*g}_1 \leq \norm f_1\norm g_1$, e
$*$ é comutativo e associativo.\index{convolução}
\end{theorem}

\begin{proof}
$(x, y) \mapsto f(x-y)g(y)$ é \omterm{def:b3:lebesgue:measurable}{mensurável} ($ (x,y)\mapsto x -
y $ é \omterm{def:b3:topology:continuity}{contínuo}; compor e multiplicar). Tonelli:
\[
\int\!\!\int \abs{f(x-y)}\abs{g(y)}\,\dd y\,\dd x
= \int\abs{g(y)}\Bigl(\int\abs{f(x - y)}\dd x\Bigr)\dd y
= \norm f_1\norm g_1 < \infty
\]
(invariância de tradução de $\lambda_d $ na integral interna).
Portanto a integral dupla é finita; Fubini dá a.e.\ absoluto
convergência e o limite da norma $\norm{f*g}_1 \leq \norm
f_1\norm g_1$. Comutatividade: substituto $ y \mapsto x - y $
(invariância de tradução e reflexão --- reflexão
a invariância se mantém nas caixas e, portanto, em todos os lugares por exclusividade).
Associatividade: Tonelli--Fubini em integral tripla.
\end{proof}

\section{Mudança de variáveis}

\begin{theorem}[Mudança linear de variáveis]
\label{thm:b3:product:linearchange}
Para $ T \in GL_d(\R)$ e $ A \in \mathcal B(\R^d)$:
$\lambda_d(T(A)) = \abs{\det T}\,\lambda_d(A)$; consequentemente
$\int f(y)\dd y = \abs{\det T}\int f(Tx)\,\dd x $ para $ f \geq 0$
ou \omterm{def:b3:lebesgue:l1}{integrável}.\index{mudança linear de variáveis}
\end{theorem}

\begin{proof}
O \omterm{def:b3:measure:measure}{medir} $\mu_T(A) = \lambda_d(T(A))$ é um Borel \omterm{def:b3:measure:measure}{medir}
(\omterm{def:b3:topology:continuity}{homeomorfismos} preservar conjuntos de Borel,
\cref{pb:b3:measure:1}), invariante à tradução
($ T(A + x) = T(A) + Tx $), finito na caixa unitária: pelo
caracterização de Medida Lebesgue
(\cref{exo:b3:measure:6}, cuja prova funciona literalmente em
$\R^d $ com cubos diádicos), $\mu_T = c(T)\lambda_d $ com $ c(T)
= \lambda_d(T(\intco01^d))$. O mapa $ T \mapsto c(T)$ é
multiplicativo ($ c(ST) = c(S)c(T)$, compondo), então
é suficiente para calcular $ c $ em geradores de $ GL_d $: elementar
matrizes. Diagonal $\operatorname{diag}(a, 1, \dots, 1)$:
mapeia o cubo unitário para uma caixa de volume $\abs a $: $ c = \abs a =
\abs\det $. Transposição de coordenadas: permuta o cubo: $ c
= 1 = \abs\det $. Transvecção $ T(x) = x + \alpha
x_2e_1$: a imagem do cubo unitário é um prisma cisalhado; por
Tonelli é \omterm{def:b3:measure:measure}{medir} é $\int\lambda_1(\text{section})\dd x_2
\cdots \dd x_d = 1$, cada $ x_1$-seção sendo um intervalo de
comprimento $1$: $ c = 1 = \abs{\det}$. Toda matriz invertível é uma
produto destes (eliminação gaussiana), e ambos $ c $ e
$\abs\det $ são multiplicativos: $ c(T) = \abs{\det T}$. O
fórmula integral segue pela máquina padrão (indicadores,
\omterm{def:b3:groups:simple}{simples}, MCT).
\end{proof}

\begin{theorem}[Mudança de variáveis]\label{thm:b3:product:changeofvar}
Deixar $ U, V \subseteq \R^d $ estar aberto e $\Phi \colon U \to V $ um
$\mathcal C^1$ difeomorfismo. Para cada \omterm{def:b3:lebesgue:measurable}{mensurável} $ f \colon V
\to [0, +\infty]$ (ou $ f \in L^1(V)$):\index{mudança de
fórmula de variáveis}
\[
\int_V f(y)\,\dd y = \int_U
f\bigl(\Phi(x)\bigr)\,\abs{\det D\Phi(x)}\,\dd x .
\]
\end{theorem}

\begin{proof}
Escrever $ J(x) = \abs{\det D\Phi(x)}$. O cerne da prova é
a desigualdade
\[
\lambda_d\bigl(\Phi(A)\bigr) \leq \int_A J\,\dd\lambda_d
\qquad\text{for every Borel } A \subseteq U;
\tag{$*$}
\]
Etapa 4 abaixo das atualizações $(*)$ --- aplicado a ambos $\Phi $ e
$\Phi^{-1}$ --- à igualdade do teorema. Observe que
$\Phi^{-1}$ é em si um $\mathcal C^1$ difeomorfismo com
Jacobiano $\abs{\det D\Phi^{-1}(y)} = J(\Phi^{-1}y)^{-1}$
(regra da cadeia em $\Phi\circ\Phi^{-1} = \mathrm{id}$).

\emph{Etapa 1: $(*)$ para cubos com fator de distorção.} Consertar um
cubo fechado $ Q \subseteq U $ de \omterm{ex:b3:groups:actions}{centro} $ x_0$ e lado $2r $
(bola \omterm{def:b3:groups:normal}{sup-normal}). Reivindicação: para cada $\varepsilon > 0$, se $\Phi $
é diferenciável em $ Q $ com
$\norm{D\Phi(x) - D\Phi(x_0)} \leq \varepsilon $ sobre $ Q $
(norma do operador para a norma sup), então
\[
\Phi(Q) \subseteq \Phi(x_0) + D\Phi(x_0)\Bigl(\,\bigl(1 +
\varepsilon\norm{D\Phi(x_0)^{-1}}\bigr)\,(Q - x_0)\Bigr),
\]
porque para $ x \in Q $, a desigualdade de valor médio aplicada a
$\Phi(x) - \Phi(x_0) - D\Phi(x_0)(x - x_0)$ dá
$\norm{\Phi(x) - \Phi(x_0) - D\Phi(x_0)(x-x_0)}_\infty \leq
\varepsilon\norm{x - x_0}_\infty \leq \varepsilon r $, e
$ D\Phi(x_0)^{-1}$ puxa esse defeito para um
$\varepsilon\norm{D\Phi(x_0)^{-1}}\,r $-ampliação do cubo.
Por \cref{thm:b3:product:linearchange},
\[
\lambda_d(\Phi(Q)) \leq \abs{\det D\Phi(x_0)}\,
\bigl(1 + \varepsilon\,C\bigr)^{d}\,\lambda_d(Q),
\qquad C = \sup_{Q}\norm{D\Phi(\cdot)^{-1}} .
\]

\emph{Etapa 2: $(*)$ para compactar cubos, por subdivisão.} Deixar $ Q
\subseteq U $ seja um compactar cubo e $\varepsilon > 0$. Sobre $ Q $,
$ D\Phi $ é uniformemente \omterm{def:b3:topology:continuity}{contínuo} e $\norm{D\Phi^{-1}}$
limitado (compactação); subdividir $ Q $ em $2^{kd}$ subcubos
$ Q_i $ pequeno o suficiente para que a oscilação $ D\Phi $ em cada um está
$\leq \varepsilon $. Etapa 1 em cada subcubo (\omterm{ex:b3:groups:actions}{centro} $ x_i $):
\[
\lambda_d(\Phi(Q)) \leq \sum_i\lambda_d(\Phi(Q_i))
\leq (1 + C\varepsilon)^d \sum_i \abs{\det
D\Phi(x_i)}\,\lambda_d(Q_i)
\leq (1 + C\varepsilon)^d\Bigl(\int_Q J + \varepsilon'\Bigr),
\]
o último passo porque $\sum_i\abs{\det D\Phi(x_i)}\mathbf
1_{Q_i} \to J $ uniformemente ligado $ Q $ (continuidade de $\det D\Phi $)
--- Comparação da soma de Riemann. Deixar $\varepsilon \to 0$: $(*)$
vale para compactar cubos.

\emph{Etapa 3: $(*)$ para todos Borel $ A $.} A função definida $ A
\mapsto \lambda_d(\Phi(A))$, em subconjuntos Borel de $ U $, é um
\omterm{def:b3:measure:measure}{medir} ($\Phi $ é uma bijeção em $ V $ preservando conjuntos de Borel
e disjunção contável), e assim é $ A \mapsto \int_AJ $.
Cada subconjunto aberto de $ U $ é uma união contável de
diádico quase disjunto compactar cubos (padrão diádico
decomposição: pegue os cubos diádicos \omterm{def:b3:rings:primemaximal}{máximos} contidos no aberto
definir) e ambos medidas são aditivos entre eles (limites
de cubos são $\lambda_d $-nulo, e seus $\Phi $-imagens são
nulo pela Etapa 2 aplicado a coberturas cúbicas finas das faces):
$(*)$ passa de cubos para conjuntos abertos. General Borel $ A $:
escapamento $ U $ por compactos $ K_m \uparrow U $ com $ K_m \subseteq
\mathring K_{m+1}$ e corrigir $ m $; sobre $\mathring K_{m+1}$, $ J $
é limitado por alguns $ M_m $. Por regularidade externa de $\lambda_d $
(prova como em \cref{thm:b3:measure:regularity}, com caixas),
escolher conjuntos abertos $ O_n $ com $ A \cap K_m \subseteq O_n
\subseteq \mathring K_{m+1}$ e $\lambda_d\bigl(O_n \setminus
(A\cap K_m)\bigr) \to 0$. Então
\[
\lambda_d\bigl(\Phi(A\cap K_m)\bigr) \leq
\lambda_d\bigl(\Phi(O_n)\bigr) \leq \int_{O_n}J
\leq \int_{A\cap K_m}J + M_m\,\lambda_d\bigl(O_n\setminus(A\cap
K_m)\bigr) \xrightarrow[n\to\infty]{} \int_{A\cap K_m}J .
\]
Deixar $ m \to \infty $: continuidade de baixo para a esquerda, MCT em
a direita. Isso estabelece $(*)$.

\emph{Etapa 4: igualdade e fórmula integral.} Primeiro
estender $(*)$ de conjuntos a integrais: para cada \omterm{def:b3:lebesgue:measurable}{mensurável} $ g
\geq 0$ sobre $ V $,
\[
\int_V g(y)\,\dd y \leq \int_U g(\Phi(x))\,J(x)\,\dd x .
\tag{$**$}
\]
Na verdade, para $ g = \mathbf 1_B $ isso é $(*)$ com $ A =
\Phi^{-1}(B)$; linearidade estende-o ao \omterm{def:b3:groups:simple}{simples} $ g $ e MCT para
tudo $ g \geq 0$ (a máquina padrão). Agora aplique $(**)$ duas vezes:
primeiro a $ g $, então --- para o difeomorfismo $\Phi^{-1}$ ---
para a função $ x \mapsto g(\Phi(x))J(x)$:
\[
\int_Vg \leq \int_U g(\Phi(x))J(x)\dd x
\leq \int_V g(y)\,J(\Phi^{-1}y)\,\abs{\det
D\Phi^{-1}(y)}\,\dd y = \int_V g ,
\]
desde $ J(\Phi^{-1}y)\abs{\det D\Phi^{-1}(y)} = \abs{\det\bigl(
D\Phi(\Phi^{-1}y)\,D\Phi^{-1}(y)\bigr)} = 1$ (regra da cadeia em
$\Phi\circ\Phi^{-1} = \mathrm{id}$). Todas as desigualdades são
igualdades: a fórmula vale para $ g \geq 0$, e para $ L^1$
funções por decomposição.
\end{proof}

\begin{example}[Coordenadas polares; o Gaussiano novamente]
\label{ex:b3:product:polar}
$\Phi(r, \theta) = (r\cos\theta, r\sin\theta)$ é um $\mathcal
C^1$ difeomorfismo de $\intoo0{+\infty}\times\intoo0{2\pi}$
para $\R^2$ menos uma meia linha (conjunto nulo), com $\det D\Phi =
r $:\index{coordenadas polares}
\[
\int_{\R^2}f(x, y)\,\dd x\,\dd y
= \int_0^{2\pi}\!\!\int_0^{+\infty}
f(r\cos\theta, r\sin\theta)\,r\,\dd r\,\dd\theta .
\]
Para $ f = \eu^{-x^2-y^2}$, Tonelli e esta fórmula dão
\[
G^2 = \Bigl(\int_\R \eu^{-x^2}\dd x\Bigr)^2
= \int_{\R^2}\eu^{-x^2-y^2}
= 2\pi\int_0^\infty r\eu^{-r^2}\dd r = \pi:
\]
a prova clássica de duas linhas de $ G = \sqrt\pi $, agora totalmente
justificado (compare a prova do parâmetro de
\cref{pb:b3:lebesgue:1}).
\end{example}

\begin{theorem}[Volume da bola unitária]
\label{thm:b3:product:ballvolume}
Deixar $ v_d = \lambda_d(B(0,1))$ em $\R^d $. Então\index{volume de
a bola unitária}
\[
v_d = \frac{\pi^{d/2}}{\Gamma\bigl(\frac d2 + 1\bigr)} :
\qquad
v_1 = 2,\quad v_2 = \pi,\quad v_3 = \tfrac{4\pi}3,\quad
v_4 = \tfrac{\pi^2}2,\ \dots
\]
\end{theorem}

\begin{proof}
Calcular $ I = \int_{\R^d}\eu^{-\norm x_2^2}\dd\lambda_d $ duas vezes.
Por Tonelli considera: $ I = G^d = \pi^{d/2}$. Pelo bolo de camadas
fórmula (\cref{prop:b3:product:layercake}) com $ f = \eu^{-
\norm x^2}$, cujos conjuntos de níveis são bolas: $\{f > t\} =
B\bigl(0, \sqrt{-\ln t}\bigr)$ para $0 < t < 1$, de \omterm{def:b3:measure:measure}{medir}
$ v_d(-\ln t)^{d/2}$ (dilatação por $\rho $ escalas $\lambda_d $ por
$\rho^d $: \cref{thm:b3:product:linearchange}), então
\[
I = \int_0^1 v_d\,(-\ln t)^{d/2}\,\dd t
\overset{t = \eu^{-s}}{=}
v_d\int_0^\infty s^{d/2}\eu^{-s}\,\dd s
= v_d\,\Gamma\Bigl(\frac d2 + 1\Bigr).
\]
Igualar. (Os valores: $\Gamma(\frac32) = \frac{\sqrt\pi}2$,
$\Gamma(2) = 1$, etc.) Nota $ v_d \to 0$ como $ d \to \infty $ ---
o problema do fim de semana quantifica a rapidez, via Stirling.
\end{proof}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:b3:product:1}
Deixar $\mu $ estar contando \omterm{def:b3:measure:measure}{medir} sobre $(\intcc01, \mathcal
B(\intcc01))$ (não $\sigma $-finito) e $\lambda $ Lebesgue
\omterm{def:b3:measure:measure}{medir}, e deixe $\Delta = \{(x,x)\}$ seja a diagonal em
$\intcc01^2$. Mostre isso $\Delta $ é \omterm{def:b3:lebesgue:measurable}{mensurável}, e calcule o
duas integrais iteradas de $\mathbf 1_\Delta $ contra
$\lambda $ e $\mu $: eles diferem. Qual hipótese de
\cref{thm:b3:product:tonelli} falha?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b3:product:2}
Justifique a troca e deduza novamente a integral de Dirichlet: para
$ A > 0$,
\[
\int_0^A\frac{\sin x}x\,\dd x
= \int_0^A\!\!\int_0^{+\infty}\eu^{-xy}\sin x\,\dd y\,\dd x
= \int_0^{+\infty}\!\!\int_0^A \eu^{-xy}\sin x\,\dd x\,\dd y,
\]
calcule a integral interna na forma fechada e deixe $ A \to
+\infty $ (dominar o $ y $-integral) para obter
$\int_0^\infty\frac{\sin x}x\dd x = \frac\pi2$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:product:3}
(a) Prove que para $ f \geq 0$ \omterm{def:b3:lebesgue:measurable}{mensurável} e $\mu $ finito:
$\sum_{n\geq1}\mu(\{f \geq n\}) \leq \int f\,\dd\mu \leq
\mu(X) + \sum_{n\geq1}\mu(\{f\geq n\})$: integrabilidade é
somabilidade da cauda medidas.
(b) Deduza que $ f \in L^1(\mu)$ ($\mu $ finito) se
$\sum_n\mu(\abs f \geq n) < \infty $.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:product:4}
Para $ f(x, y) = \dfrac{x^2 - y^2}{(x^2 + y^2)^2}$ sobre
$\intoo01^2$, mostrar
\[
\int_0^1\!\!\int_0^1 f\,\dd y\,\dd x = \frac\pi4,
\qquad
\int_0^1\!\!\int_0^1 f\,\dd x\,\dd y = -\frac\pi4
\]
\emph{(observação $ f = \partial_y\bigl(\frac{y}{x^2+y^2}\bigr)$)},
e verifique diretamente que $\int\!\int\abs f = +\infty $: Fubini
integrabilidade hipótese não é decorativa.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:product:5}
(a) Calcular $\mathbf 1_{\intcc01} * \mathbf 1_{\intcc01}$
explicitamente (uma função de tenda), e $(\mathbf 1 * \mathbf 1 *
\mathbf 1)$ forma geral.
(b) Mostrar $\operatorname{supp}(f * g) \subseteq
\overline{\operatorname{supp}f + \operatorname{supp}g}$.
(c) Mostre que se $ f \in L^1$ e $ g $ é limitado e \omterm{def:b3:topology:continuity}{contínuo},
$ f * g $ é \omterm{def:b3:topology:continuity}{contínuo}. \emph{(DCT via continuidade de
tradução no limite $ g $.)}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:product:6}
(a) Mostre que o simplex $\Delta_d = \{x \in \intco0\infty^d
: x_1 + \dots + x_d \leq 1\}$ tem volume $\frac1{d!}$
\emph{(indução e Fubini)}.
(b) Recuperar $ v_2 = \pi $, $ v_3 = \frac{4\pi}3$ de
\cref{thm:b3:product:ballvolume}e mostrar
$\lambda_d(\text{ellipsoid with semi-axes } a_i) =
v_d\prod a_i $.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:product:7}
Para qual $ s > 0$ os seguintes são finitos? Justifique com polar
coordenadas:
\[
\int_{B(0,1)\subseteq\R^2}\frac{\dd x\,\dd y}{(x^2 +
y^2)^{s}},
\qquad
\int_{\R^2\setminus B(0,1)}\frac{\dd x\,\dd y}{(x^2 +
y^2)^{s}} .
\]
Generalizar para $\R^d $ (os limites $ s < d/2$ e $ s > d/2$).
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b3:product:8}
(Beta--Gama) Para $ p, q > 0$, deixar $ B(p, q) = \int_0^1t^{p-1}(1
- t)^{q-1}\dd t $. A partir de $\Gamma(p)\Gamma(q)$ como um
integral dupla, substituto $(x, y) = (uv,\, u(1 - v))$ (um
difeomorfismo do quadrante aberto em
$\intoo0\infty\times\intoo01$; calcule seu Jacobiano $= u $) e
concluir\index{Função beta}
\[
B(p, q) = \frac{\Gamma(p)\,\Gamma(q)}{\Gamma(p + q)} .
\]
Deduzir $\int_0^{\pi/2}\sin^{2p-1}\theta\cos^{2q-1}\theta\,
\dd\theta = \frac12B(p,q)$ e o valor dos Wallis
integrais $ W_n = \int_0^{\pi/2}\sin^n $.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:product:9}
(Fórmula de transferência) Deixe $ T \colon (X, \mathcal A, \mu) \to (Y,
\mathcal B)$ ser \omterm{def:b3:lebesgue:measurable}{mensurável} e $ T_*\mu(B) = \mu(T^{-1}(B))$
o \emph{medida pushforward}\index{medida pushforward}.
Mostre isso para cada \omterm{def:b3:lebesgue:measurable}{mensurável} $ g \geq 0$ sobre $ Y $:
\[
\int_Y g\,\dd(T_*\mu) = \int_X g\circ T\,\dd\mu
\]
(máquina padrão). Então compare com
\cref{thm:b3:product:linearchange}: que informações extras fazem
a fórmula de mudança de variáveis conduz que o resumo
fórmula de transferência não? \emph{(A fórmula de transferência nunca
identifica $ T_*\mu $; o teorema da mudança de variáveis computa
$ T_*\lambda_d $ explicitamente como um \omterm{ex:b3:lebesgue:gamma}{densidade} \omterm{def:b3:measure:measure}{medir}.)}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b3:product:10}
(momentos gaussianos) Usando \omterm{ex:b3:product:polar}{coordenadas polares} e Fubini, calculam
para o peso gaussiano padrão em $\R^d $:
\[
\int_{\R^d}\norm x_2^2\;\eu^{-\norm x_2^2}\,\dd x
\qquad\text{and}\qquad
\int_{\R^d}x_1^2\,\eu^{-\norm x^2_2}\,\dd x,
\]
verifique a consistência ($\norm x^2 = \sum x_i^2$) e deduza
o segundo momento do \omterm{def:b3:measure:measure}{medir}
$\pi^{-d/2}\eu^{-\norm x^2}\dd x $.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:product:11}
(Gráfico e hipógrafo) Deixe $ f \colon \R^d \to \intco0\infty $
ser \omterm{def:b3:lebesgue:measurable}{mensurável}.
(a) Mostre que \emph{hipógrafo} $ H = \{(x, y) \in
\R^d\times\R : 0 < y < f(x)\}$ é \omterm{def:b3:lebesgue:measurable}{mensurável} em
$\R^{d+1}$ com
\[
\lambda_{d+1}(H) = \int_{\R^d}f\,\dd\lambda_d :
\]
``a integral é a área sob o gráfico'', finalmente uma
teorema. \emph{(Seções; Tonelli.)}
(b) Mostre que o gráfico $\{(x, f(x)) : x \in \R^d\}$ é um
conjunto nulo de $\R^{d+1}$.
(c) Deduza uma prova de duas linhas de que a esfera $ S^{d-1}$ é
Lebesgue-nulo em $\R^d $.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:product:12}
(Uma famosa integral dupla) Usando a série geométrica e
Tonelli em $\intoo01^2$, provar
\[
\int_0^1\!\!\int_0^1\frac{\dd x\,\dd y}{1 - xy}
= \sum_{n\geq1}\frac1{n^2} = \zeta(2),
\qquad
\int_0^1\!\!\int_0^1\frac{\dd x\,\dd y}{1 + xy}
= \sum_{n\geq1}\frac{(-1)^{n-1}}{n^2} = \frac{\zeta(2)}2 .
\]
(A identidade da segunda série: divisão de índices pares e ímpares.)
Com $\zeta(2) = \frac{\pi^2}6$ (\cref{ch:b3:spectral}), dois
integrais de aparência inocente são avaliadas como $\frac{\pi^2}6$ e
$\frac{\pi^2}{12}$; onde exatamente está a positividade de Tonelli
hipótese faz seu trabalho?
\end{exercise}

\section{Problema: fórmula de Stirling}

\begin{problem}[{Problema de fim de semana --- $ n! \sim
\sqrt{2\pi n}\,(n/\eu)^n $, por convergência dominada}]
\label{pb:b3:product:1}
A fórmula de Stirling rege todas as contagens assintóticas neste livro
--- volumes de bola, coeficientes binomiais, limite central
forma local do teorema. Nós provamos isso a partir do $\Gamma $ integrante
(\cref{ex:b3:lebesgue:gamma}) com o \emph{Método Laplace},
em sua forma de convergência dominada mais limpa, então colete
dividendos.\index{Fórmula de Stirling}

\medskip
\noindent\textbf{Part I --- The formula.} Para $ t > 0$,
$\Gamma(t + 1) = \int_0^\infty x^{t}\eu^{-x}\dd x $.
\begin{enumerate}
  \item Substituir $ x = t + \sqrt t\,u $ e mostrar
        \[
        \frac{\Gamma(t+1)}{t^{t}\eu^{-t}\sqrt t}
        = \int_{-\sqrt t}^{+\infty}
        \exp\Bigl(t\ln\Bigl(1 + \frac u{\sqrt
        t}\Bigr) - \sqrt t\,u\Bigr)\,\dd u
        \;=\;\int_\R g_t(u)\,\dd u,
        \]
        onde $ g_t(u) = \exp\bigl(t\ln(1 + u/\sqrt t) -
        \sqrt t\,u\bigr)\mathbf 1_{u > -\sqrt t}$.
  \item Mostre o limite pontual: para cada fixo $ u $, $ g_t(u)
        \to \eu^{-u^2/2}$ como $ t \to +\infty $ \emph{(expandir
        $\ln(1 + h)$ para segunda ordem)}.
  \item Dominação. Deixar $\varphi(h) = \ln(1 + h) - h $, para que
        $ g_t(u) = \exp\bigl(t\,\varphi(u/\sqrt t)\bigr)$ para
        $ u > -\sqrt t $. Prove os dois limites
        \[
        \varphi(h) \leq -\frac{h^2}4 \quad (-1 < h \leq 1),
        \qquad
        \varphi(h) \leq -c\,h \quad (h \geq 1),\ \ c = 1 -
        \ln 2 > 0
        \]
        \emph{(estudar $\varphi(h) + \frac{h^2}4$ e $\varphi(h)
        + ch $: calcule as derivadas e verifique o sinal
        cada intervalo)}. Deduza, por $ t \geq 1$:
        \[
        g_t(u) \leq \eu^{-u^2/4}\ \ (\abs u \leq \sqrt t),
        \qquad
        g_t(u) \leq \eu^{-cu}\ \ (u \geq \sqrt t),
        \]
        para que $ g_t(u) \leq \eu^{-u^2/4} +
        \eu^{-cu}\,\mathbf 1_{u > 0}$: um \omterm{def:b3:lebesgue:l1}{integrável} dominador
        independente de $ t \geq 1$.
  \item Conclua com o teorema da convergência dominada e o
        Integral gaussiana (\cref{ex:b3:product:polar}):
        \[
        \Gamma(t + 1) \;\sim\;
        \sqrt{2\pi t}\;\Bigl(\frac t\eu\Bigr)^{t}
        \qquad (t \to +\infty),
        \]
        e em particular $ n! \sim \sqrt{2\pi
        n}\,(n/\eu)^n $.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Dividends.}
\begin{enumerate}[resume]
  \item (Wallis) De \cref{exo:b3:product:8}, $ W_{2n} =
        \frac\pi2\binom{2n}n4^{-n}$ fórmulas de tipo: derivar
        $\binom{2n}{n} \sim \frac{4^n}{\sqrt{\pi n}}$ de
        Stirling e compare com a recursão $ W_{n} =
        \frac{n-1}nW_{n-2}$.
  \item (Colapso dos volumes das bolas) Mostrar
        \[
        v_d = \frac{\pi^{d/2}}{\Gamma(\frac d2 + 1)}
        \;\sim\; \frac{1}{\sqrt{\pi d}}
        \Bigl(\frac{2\pi\eu}{d}\Bigr)^{d/2},
        \]
        então $ v_d \to 0$ mais rápido que qualquer sequência geométrica;
        encontre a dimensão maximizando $ v_d $ (numericamente: $ d =
        5$).
  \item (Concentração do binômio --- uma prévia de
        \cref{ch:b3:clt}) Usando Stirling, mostre o local
        estimativa, para $ k = n/2 + s\sqrt n/2$ com $ s $ fixo
        e $ n $ até:
        \[
        2^{-n}\binom{n}{k} \;\sim\;
        \sqrt{\frac{2}{\pi n}}\;\eu^{-s^2/2},
        \]
        o perfil Gaussiano discreto: de Moivre--Laplace em
        embrião.
  \item Onde exatamente a prova da Parte I usou: (i) MCT ou
        DCT; (ii) a integral gaussiana; (iii) a invariância
        propriedades de Medida Lebesgue? Uma frase cada.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The error term: Stirling with
bars.} Definir $ d_n = \ln n! - \bigl(n + \tfrac12\bigr)\ln n + n -
\ln\sqrt{2\pi}$, então a Parte I diz $ d_n \to 0$.
\begin{enumerate}[resume]
  \item Mostrar $ d_n - d_{n+1} = \bigl(n +
        \tfrac12\bigr)\ln\bigl(1 + \tfrac1n\bigr) - 1$.
  \item Com $ t = \frac1{2n+1}$, verifique $\frac{n+1}n =
        \frac{1+t}{1-t}$ e expandir:
        \[
        d_n - d_{n+1} = \frac{t^2}3 + \frac{t^4}5 +
        \frac{t^6}7 + \cdots,
        \]
        e deduzir os limites bilaterais
        \[
        \frac1{3(2n+1)^2} \;<\; d_n - d_{n+1} \;<\;
        \frac1{12n} - \frac1{12(n+1)} .
        \]
  \item Telescópio (usando $ d_m \to 0$) e verifique o agradável
        \omterm{def:b3:galois:algebraic}{algébrico} identidade $\frac1{3(2m+1)^2} >
        \frac1{12m+1} - \frac1{12(m+1)+1}$ para $ m \geq 1$, para
        obter o bracketing clássico
        \[
        \sqrt{2\pi n}\Bigl(\frac n\eu\Bigr)^n
        \eu^{1/(12n+1)} \;<\; n! \;<\;
        \sqrt{2\pi n}\Bigl(\frac n\eu\Bigr)^n\eu^{1/(12n)} .
        \]
  \item Duas consequências: (a) o erro relativo da teoria de Stirling
        fórmula é $< 10^{-6}$ assim que $ n \geq 83\,334$;
        (b) estimativa $100!$ com quatro algarismos significativos à mão
        do colchete ($100! \approx 9.3326\cdot
        10^{157}$), e maravilhe-se brevemente com a precisão de um
        fórmula assintótica em um valor muito finito $ n $.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The Wallis route: Stirling
without the Gaussian.} Historicamente a constante $\sqrt{2\pi}$
veio de Wallis, não de Gauss; esta parte prova novamente Stirling
independentemente das Partes I-II, e assim reprova a
Integral gaussiana. Deixar $ W_n = \int_0^{\pi/2}\sin^n\theta\,
\dd\theta $.
\begin{enumerate}[resume]
  \item Estabelecer $ W_n = \frac{n-1}nW_{n-2}$ (integrar por
        partes), as formas fechadas
        \[
        W_{2n} = \frac\pi2\binom{2n}n4^{-n}, \qquad
        W_{2n+1} = \frac{4^n}{(2n+1)\binom{2n}n},
        \]
        e a identidade $ W_nW_{n-1} = \frac\pi{2n}$.
  \item Da monotonicidade $(W_n)$ deduzir
        $ W_{2n}/W_{2n+1} \to 1$, então
        \[
        W_{2n} \sim \frac12\sqrt{\frac\pi n}
        \qquad\text{and}\qquad
        \binom{2n}n4^{-n}\sqrt n \longrightarrow
        \frac1{\sqrt\pi} :
        \]
        Teorema de Wallis, obtido sem Stirling.
  \item Mostre, apenas pela telescopia da Parte III (sem valor
        da constante necessária), que $ e_n = \ln n! - (n +
        \frac12)\ln n + n $ converge para algum limite $\ell $;
        equivalentemente $ n! \sim K\,n^{n+1/2}\eu^{-n}$ com $ K =
        \eu^\ell > 0$ ainda não identificado.
  \item Insira este assintótico em $\binom{2n}n4^{-n}\sqrt n $
        e identificar, através da questão 14, a única possível
        valor: $ K = \sqrt{2\pi}$. Monte a lógica: Peças
        III--IV juntos dão um completo segunda prova de
        Stirling --- e, portanto, executando a substituição da Parte I
        de trás para frente, uma avaliação independente de
        $\int_\R\eu^{-u^2/2}\dd u = \sqrt{2\pi}$. Dois pilares,
        qualquer um dos quais apoia o outro.
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Last dividends.}
\begin{enumerate}[resume]
  \item (O perfil local completo) Para números inteiros $\abs j \leq
        K\sqrt n $ ($ K $ fixo), mostrar
        \[
        \frac{\binom{2n}{n+j}}{\binom{2n}{n}} =
        \prod_{i=1}^{\abs j}\frac{n - i + 1}{n + i}
        = \exp\Bigl(-\frac{j^2}n +
        O\Bigl(\frac1{\sqrt n}\Bigr)\Bigr),
        \]
        uniformemente em $ j $ \emph{(pegue logaritmos e use $\ln
        \frac{1-x}{1+y} = -(x + y) + O(x^2 + y^2)$)}. Isto é
        a versão bilateral da questão 7 e a exata
        estimativa citada em \cref{ch:b3:clt}problema de fim de semana.
  \item (Uma prévia de Poisson) Mostre com Stirling que
        $\eu^{-n}\dfrac{n^n}{n!} \sim \dfrac1{\sqrt{2\pi n}}$:
        o modo de uma lei de Poisson de grande média $ n $ carrega
        massa $\approx (2\pi n)^{-1/2}$, exatamente como o \omterm{ex:b3:groups:actions}{centro}
        o teorema do limite irá prever.
  \item (Proporções gama) Para $ a \in \intoo01$, provar
        $\dfrac{\Gamma(n + a)}{\Gamma(n)\,n^a} \to 1$ usando
        os limites da inclinação log-convexidade de
        \cref{pb:b3:lebesgue:1} (pergunta 14 lá), e
        estender a todos os reais $ a > 0$ pelo funcional
        equação. (Isso é o que ``$\Gamma(t+1) \sim $
        Stirling'' significa entre os inteiros.)
  \item (Bolas, encore) De $ v_d = \pi^{d/2}/\Gamma(\frac d2
        + 1)$: tabular $ v_1, \dots, v_7$ exatamente, verifique
        unimodalidade via $\frac{v_d}{v_{d-2}} = \frac{2\pi}d $
        (aumentando enquanto $ d < 2\pi $, diminuindo depois) e
        comprovar a marcante identidade geradora
        \[
        \sum_{k\geq0}v_{2k}\,x^{2k} = \eu^{\pi x^2} :
        \]
        todos os volumes de esferas unitárias de dimensões pares embalados em um
        exponencial.
  \item (Entropia assintótica) Para fixo $\alpha \in \intoo01$
        com $\alpha n \in \N $, deduza de Stirling
        \[
        \binom{n}{\alpha n} \;\sim\;
        \frac{\eu^{n\,H(\alpha)}}
        {\sqrt{2\pi\,\alpha(1-\alpha)\,n}},
        \qquad
        H(\alpha) = -\alpha\ln\alpha -
        (1-\alpha)\ln(1-\alpha) :
        \]
        a taxa de crescimento exponencial dos coeficientes binomiais
        é o \emph{entropia} $ H $ --- verifique isso $\alpha =
        \frac12$ recupera a questão 5, e que $ H(\alpha) <
        \ln2$ para $\alpha \neq \frac12$ (então descentralizado
        binômios são exponencialmente insignificantes em $2^n $).
  \item (Áreas de superfície) A área da esfera unitária $ S^{d-1}$
        é $ s_{d-1} = d\,v_d $ (provado como
        \cref{exo:b3:forms:6} nas formas diferenciais
        capítulo; aqui, tome isso como a definição). Tabular
        $ s_0, \dots, s_6$, localize o \omterm{def:b3:rings:primemaximal}{máximo} ($ d - 1 =
        6$, $ s_6 = \frac{16\pi^3}{15} \approx 33.07$) e
        mostrar $ s_{d-1} \to 0$ supergeometricamente também ---
        esferas de alta dimensão são, por todos os euclidianos
        critério, extremamente pequeno.
  \item (O primeiro termo de correção) Deduza do
        colchetes da questão 11 que $ d_n = \frac1{12n} +
        O\bigl(\frac1{n^2}\bigr)$, por isso
        \[
        n! = \sqrt{2\pi n}\,\Bigl(\frac
        n\eu\Bigr)^{n}\Bigl(1 + \frac1{12n} +
        O\Bigl(\frac1{n^2}\Bigr)\Bigr).
        \]
        Verifique em $ n = 10$: a fórmula \omterm{def:b3:groups:simple}{simples} dá
        $3\,598\,696$ (erro relativo $8.3\cdot10^{-3}$),
        o corrigido $3\,628\,685$ contra $10! =
        3\,628\,800$ (erro relativo $3.2\cdot10^{-5}$) ---
        um termo da série compra dois dígitos e meio.
  \item (A mediana de $\Gamma $) Mostre que
        \[
        \frac{1}{\Gamma(t+1)}
        \int_0^{t} x^{t}\eu^{-x}\,\dd x
        \;\longrightarrow\; \frac12
        \qquad (t \to +\infty) :
        \]
        assintoticamente, exatamente metade da massa do
        $\Gamma $ integrando fica abaixo de seu modo $ x = t $.
        \emph{(Execute a substituição da Parte I no truncado
        integral; o dominador da questão 3 já está em
        lugar.)}
  \item (Entropia, não assintoticamente) Para $\alpha \in
        \intoc0{\frac12}$ comprovar o limite, válido para
        \emph{todo} $ n \geq 1$:
        \[
        \sum_{k=0}^{\lfloor\alpha n\rfloor}\binom nk
        \;\leq\; \eu^{n\,H(\alpha)} ,
        \]
        comparando a soma com $\sum_k\binom
        nk\lambda^{k-\alpha n}$ para a inclinação $\lambda =
        \frac{\alpha}{1-\alpha} \leq 1$. Verifique se isso
        escolha de $\lambda $ é \omterm{def:b3:rings:ideal}{ideal} e se reconcilia com
        questão 21: a taxa exponencial $ H(\alpha)$ do
        declaração assintótica é obtida por uma linha
        desigualdade sem nenhuma assintótica.
\end{enumerate}
\end{problem}
