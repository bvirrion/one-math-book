\section{DefiniĂ§ĂŁo; HĂ¶lder e Minkowski}

\begin{definition}\label{def:b3:lp:space}
Para $1 \leq p < \infty $, $\mathcal L^p(\mu)$ Ă© o conjunto de
\omterm{def:b3:lebesgue:measurable}{mensurĂˇvel} $ f $ com $\norm f_p = \bigl(\int\abs
f^p\dd\mu\bigr)^{1/p} < \infty $, e $\mathcal L^\infty(\mu)$
o conjunto de $ f $ limitado fora de um conjunto nulo, com $\norm
f_\infty $ o \emph{comida essencial} --- o mĂ­nimo $ M $ com
$\abs f \leq M $ a.e.\ (o inf Ă© alcanĂ§ado: cruza o nulo
conjuntos para $ M + \frac1n $). Desde $\norm f_p = 0$ apenas forĂ§as $ f
= 0$ \emph{a.e.} (\cref{exo:b3:lebesgue:5}), definimos
\[
L^p(\mu) = \mathcal L^p(\mu)/\{f = 0 \text{ a.e.}\} :
\]
elementos sĂŁo classes de funĂ§Ăµes mĂłdulo conjuntos nulos, e
$\norm\cdot_p $ Ă© uma norma genuĂ­na sobre
$ L^p $.\index{EspaĂ§o Lp@$ L^p $ espaĂ§o}
\end{definition}

\begin{theorem}[Hdesigualdade dos idosos]\label{thm:b3:lp:holder}
Deixar $1 \leq p, q \leq \infty $ com $\frac1p + \frac1q = 1$
(expoentes conjugados). Para \omterm{def:b3:lebesgue:measurable}{mensurĂˇvel} $ f, g $:\index{HĂ¶lder's
inequality}
\[
\norm{fg}_1 \leq \norm f_p\,\norm g_q ,
\]
com igualdade (para $1 < p < \infty $, normas finitas, $ f,g\ne0$)
se $\abs f^p $ e $\abs g^q $ sĂŁo proporcionais a.e.
\end{theorem}

\begin{proof}
Os casos $\{p, q\} = \{1, \infty\}$ sĂŁo diretos ($\abs{fg} \leq
\norm g_\infty\abs f $ e.e.). Deixar $1 < p < \infty $; normalizar
$\norm f_p = \norm g_q = 1$ (homogeneidade; zero ou infinito
as normas sĂŁo triviais). Desigualdade de Young $ ab \leq \frac{a^p}p +
\frac{b^q}q $ ($ a, b \geq 0$; concavidade de $\ln $, como em
\cref{pb:b3:banach:1}) dĂˇ pontualmente $\abs{f g} \leq
\frac{\abs f^p}p + \frac{\abs g^q}q $; integrar: $\norm{fg}_1
\leq \frac1p + \frac1q = 1$. A igualdade forĂ§a a.e.\ igualdade em
Jovem, ou seja\ $\abs f^p = \abs g^q $ a.e.\ (apĂłs o
normalizaĂ§ĂŁo; desfazendo, proporcionalidade).
\end{proof}

\begin{theorem}[Desigualdade de Minkowski]\label{thm:b3:lp:minkowski}
Para $1 \leq p \leq \infty $: $\norm{f + g}_p \leq \norm f_p +
\norm g_p $.\index{Desigualdade de Minkowski}
\end{theorem}

\begin{proof}
$ p = 1, \infty $: desigualdade pontual/a.e.\ triĂ˘ngulo. Para $1 <
p < \infty $, suponha $\norm{f+g}_p < \infty $ (caso contrĂˇrio, use
$\abs{f+g}^p \leq 2^{p-1}(\abs f^p + \abs g^p)$, da convexidade
de $ t^p $, para ver que o lado esquerdo Ă© finito quando o direito Ă©).
EntĂŁo
\[
\norm{f{+}g}_p^p \leq \int\abs f\,\abs{f{+}g}^{p-1} +
\int\abs g\,\abs{f{+}g}^{p-1}
\leq \bigl(\norm f_p + \norm g_p\bigr)\,
\bigl\|\abs{f{+}g}^{p-1}\bigr\|_q
\]
por HĂ¶lder, e $\norm{\abs{f+g}^{p-1}}_q =
\norm{f+g}_p^{p/q}$ desde $(p-1)q = p $; dividir por
$\norm{f+g}_p^{p/q}$ (se diferente de zero; caso contrĂˇrio, trivial) e use $ p -
\frac pq = 1$.
\end{proof}

\section{Completude e seus companheiros}

\begin{theorem}[Riesz--Fischer]\label{thm:b3:lp:complete}
Para $1 \leq p \leq \infty $, $ L^p(\mu)$ Ă© um espaĂ§o de Banach.
AlĂ©m disso, toda sequĂŞncia convergindo em $ L^p $ tem uma subsequĂŞncia
convergente \emph{\omterm{def:b3:lebesgue:l1}{quase em todos os lugares}} (com um $ L^p $ dominador
no caso $ p < \infty $).\index{Teorema de Riesz-Fischer}
\end{theorem}

\begin{proof}
$ p = \infty $: um $\norm\cdot_\infty $-SequĂŞncia de Cauchy Ă©, fora de um
Ăşnico conjunto nulo (uniĂŁo de muitos contĂˇveis), uniformemente Cauchy:
converge uniformemente para fora dele; feito. Deixar $ p < \infty $. Por
\cref{exo:b3:complete:1}(b) basta somar absolutamente
sĂ©rie convergente: deixe $\sum\norm{f_k}_p = M < \infty $. Definir
$ G_n = \sum_{k \leq n}\abs{f_k}$ e $ G = \sum_k\abs{f_k}$
(pontualmente em $[0,\infty]$): por Minkowski $\norm{G_n}_p \leq
M $ e MCT ($ G_n^p \nearrow G^p $) dĂˇ $\int G^p \leq M^p $:
$ G < \infty $ ou seja, entĂŁo a sĂ©rie $\sum f_k(x)$ converge
absolutamente para a.e.\ $ x $; chame a soma $ S(x)$ (qualquer valor em
o conjunto nulo). EntĂŁo $\abs{S - \sum_{k\leq n}f_k}^p \leq (2G)^p
\in L^1$, e DCT dĂˇ $\norm{S - \sum_{k\leq n}f_k}_p \to 0$:
a sĂ©rie converge em $ L^p $.

A declaraĂ§ĂŁo de subsequĂŞncia: se $ f_n \to f $ em $ L^p $, escolha $ n_k $
com $\norm{f_{n_{k+1}} - f_{n_k}}_p \leq 2^{-k}$; a sĂ©rie
$\sum(f_{n_{k+1}} - f_{n_k})$ enquadra-se no anterior
argumento: converge absolutamente, ou seja, dominado por um $ G \in
L^p $, entĂŁo $ f_{n_k} \to f_{n_1} + \sum(\cdots) $ a.e., e isso
a.e.-limit deve ser (um representante de) $ f $ (ambos sĂŁo $ L^p $
limites). O dominador: $\abs{f_{n_k}} \leq \abs{f_{n_1}} + G $.
\end{proof}

\begin{remark}\label{rem:b3:lp:modes}
$ L^p $ convergĂŞncia nĂŁo implica a.e.\ convergĂŞncia (o
\emph{mĂˇquina de escrever} sequĂŞncia, \cref{exo:b3:lp:3}), nem
inversamente (escapando de solavancos): os dois modos estĂŁo vinculados apenas
atravĂ©s de subsequĂŞncias e dominaĂ§ĂŁo. Mantendo os contra-exemplos
de \cref{exo:b3:lp:3} em mente Ă© a melhor vacina.
\end{remark}

\section{Teoremas de densidade}

\begin{theorem}\label{thm:b3:lp:density}
Deixar $1 \leq p < \infty $.
\begin{enumerate}
  \item FunĂ§Ăµes \omterm{def:b3:groups:simple}{simples} (com medida finita suportes) sĂŁo
        \omterm{def:b3:topology:interior}{denso} em $ L^p(\mu)$.
  \item Em $ L^p(\R^d)$, o \omterm{def:b3:topology:continuity}{contĂ­nuo} suportado de forma compacta
        funĂ§Ăµes $\mathcal C_c(\R^d)$ sĂŁo \omterm{def:b3:topology:interior}{densos}.
  \item A traduĂ§ĂŁo Ă© \omterm{def:b3:topology:continuity}{contĂ­nuo} sobre $ L^p(\R^d)$: escrita
        $\tau_hf = f(\cdot - h)$, $\norm{\tau_hf - f}_p \to 0$
        como $ h \to 0$.
\end{enumerate}
Nenhum dos trĂŞs vale para $ p = \infty $.
\end{theorem}

\begin{proof}
(1) Para $ f \geq 0$: o diĂˇdico $ s_n \nearrow f $ de
\cref{thm:b3:lebesgue:approximation} satisfazer $\abs{f - s_n}^p
\leq f^p \in L^1$: DCT. (Cada $ s_n \leq f $ reside em $ L^p $ e
seus conjuntos de nĂ­veis tĂŞm finito \omterm{def:b3:measure:measure}{medir} onde estĂˇ o valor
positivo: $\mu(s_n \geq c) \leq c^{-p}\int f^p $.) Dividir
geral $ f $ em quatro partes nĂŁo negativas.

(2) Por (1) Ă© suficiente aproximar $\mathbf 1_A $, $ A $ Borel
com $\lambda_d(A) < \infty $. Regularidade (prova como em
\cref{thm:b3:measure:regularity}) dĂˇ compactar $ K \subseteq A
\subseteq U $ abrir com $\lambda_d(U\setminus K) <
\varepsilon $; FunĂ§ĂŁo de Urysohn
\[
\varphi(x) = \frac{d(x, \R^d\setminus U)}{d(x, \R^d\setminus U)
+ d(x, K)}
\]
Ă© \omterm{def:b3:topology:continuity}{contĂ­nuo}, $1$ sobre $ K $, $0$ fora $ U $, e pode ser tomado
suportado de forma compacta (encolher $ U $ para uma abertura limitada primeiro). EntĂŁo
$\norm{\mathbf 1_A - \varphi}_p^p \leq \lambda_d(U\setminus K)
< \varepsilon $.

(3) Para $ g \in \mathcal C_c $: uniforme continuidade dĂˇ
$\norm{\tau_hg - g}_\infty \to 0$, com suportes fixos
compactar para $\abs h \leq 1$: $\norm{\tau_hg - g}_p \to 0$. Para
geral $ f $: escolha $ g \in \mathcal C_c $ com $\norm{f - g}_p <
\varepsilon $; entĂŁo $\norm{\tau_hf - f}_p \leq 2\norm{f - g}_p +
\norm{\tau_hg - g}_p $ (invariĂ˘ncia de traduĂ§ĂŁo da norma).

Para $ p = \infty $: aproximaĂ§ĂŁo uniforme de $\mathbf
1_{\intoo0\infty}$ por \omterm{def:b3:topology:continuity}{contĂ­nuo} funĂ§Ăµes Ă© impossĂ­vel
(pular) e $\norm{\tau_h\mathbf 1_{\intoo0\infty} - \mathbf
1_{\intoo0\infty}}_\infty = 1$ para $ h \neq 0$.
\end{proof}

\section{ConvoluĂ§ĂŁo e regularizaĂ§ĂŁo}

\begin{theorem}[Desigualdade de Young]\label{thm:b3:lp:young}
Deixar $1 \leq p \leq \infty $, $ f \in L^1(\R^d)$, $ g \in
L^p(\R^d)$. EntĂŁo $ f * g $ Ă© definido, ou seja, pertence a $ L^p $,
e\index{Desigualdade de Young}
\[
\norm{f * g}_p \leq \norm f_1\,\norm g_p .
\]
\end{theorem}

\begin{proof}
$ p = \infty $: ligaĂ§ĂŁo direta. $ p = 1$:
\cref{thm:b3:product:convolution}. Deixar $1 < p < \infty $, $ q $
conjugado. Dividir $\abs{f(y)} = \abs{f(y)}^{1/q}\cdot
\abs{f(y)}^{1/p}$ e aplique HĂ¶lder:
\[
\int\abs{f(y)}\,\abs{g(x{-}y)}\,\dd y
\leq \Bigl(\int\abs f\Bigr)^{1/q}
\Bigl(\int\abs{f(y)}\,\abs{g(x - y)}^p\,\dd y\Bigr)^{1/p} .
\]
Levante para o $ p $-th poder e integrar em $ x $; Tonelli no
segundo fator dĂˇ $\norm f_1^{p/q}\cdot\norm f_1\norm g_p^p $,
ou seja\ $\norm{f*g}_p^p \leq \norm f_1^{1 + p/q}\norm g_p^p =
(\norm f_1\norm g_p)^p $ --- e a finitude do Tonelli
integral justifica a.e.\ convergĂŞncia absoluta como em
\cref{thm:b3:product:convolution}.
\end{proof}

\begin{definition}[Molificadores]\label{def:b3:lp:mollifier}
A funĂ§ĂŁo\index{calmante}\index{funĂ§ĂŁo de colisĂŁo}
\[
\rho(x) = \begin{cases}
c\,\exp\Bigl(-\dfrac{1}{1 - \norm x^2}\Bigr) & \norm x < 1,\\
0 & \norm x \geq 1,
\end{cases}
\]
com $ c $ normalizando $\int\rho = 1$, Ă© $\mathcal C^\infty $ sobre
$\R^d $: o ponto Ă© que $ t \mapsto \eu^{-1/t}\mathbf
1_{t>0}$ Ă© $\mathcal C^\infty $ sobre $\R $, todas as suas derivadas em
$0^+$ ser $0$ (cada derivada Ă© $ P(1/t)\eu^{-1/t}$ por um
polinĂ´mio $ P $, o que tende a $0$; induĂ§ĂŁo). Para
$\varepsilon > 0$ definir $\rho_\varepsilon(x) =
\varepsilon^{-d}\rho(x/\varepsilon)$: suportado em $\bar B(0,
\varepsilon)$, ainda de integral $1$.
\end{definition}

\begin{theorem}[RegularizaĂ§ĂŁo]\label{thm:b3:lp:regularization}
Deixar $1 \leq p < \infty $ e $ f \in L^p(\R^d)$. EntĂŁo:
\begin{enumerate}
  \item $ f * \rho_\varepsilon \in \mathcal C^\infty(\R^d)$,
        com $\partial^\alpha(f * \rho_\varepsilon) = f *
        \partial^\alpha\rho_\varepsilon $;
  \item $\norm{f * \rho_\varepsilon - f}_p \to 0$ como
        $\varepsilon \to 0$;
  \item consequentemente $\mathcal C^\infty_c(\R^d)$ Ă© \omterm{def:b3:topology:interior}{denso} em
        $ L^p(\R^d)$.
\end{enumerate}
\end{theorem}

\begin{proof}
(1) DiferenciaĂ§ĂŁo sob a integral
(\cref{thm:b3:lebesgue:paramdiff}) em $ x $: para $ x $ em uma bola
$ B $, $\abs{\partial_{x_i}\rho_\varepsilon(x - y)} \leq
C_\varepsilon\,\mathbf 1_{K}(y)$ com $ K $ compactar ($ y $ dentro de
$\varepsilon $ de $ B $), e $\abs f\,\mathbf 1_K \in L^1$
(HĂ¶lder contra $\mathbf 1_K $): o teorema se aplica;
iterar para derivadas mais altas.

(2) Desde $\int\rho_\varepsilon = 1$:
\[
(f * \rho_\varepsilon)(x) - f(x)
= \int \bigl(f(x - y) - f(x)\bigr)\rho_\varepsilon(y)\,\dd y ,
\]
e Minkowski \emph{integrante} desigualdade --- ou diretamente:
HĂ¶lder/Jensen com a probabilidade \omterm{def:b3:measure:measure}{medir}
$\rho_\varepsilon\dd y $ e Tonelli ---
\[
\norm{f*\rho_\varepsilon - f}_p^p
\leq \int\Bigl(\int\abs{f(x-y) -
f(x)}^p\dd x\Bigr)\rho_\varepsilon(y)\,\dd y
= \int \norm{\tau_yf - f}_p^p\;\rho_\varepsilon(y)\,\dd y
\]
(o passo intermediĂˇrio: aplicar a desigualdade de Jensen,
\cref{exo:b3:lp:10}, para o \omterm{def:b3:topology:interior}{interior} $ y $-integral, depois Tonelli).
O integrando Ă© suportado em $\norm y \leq \varepsilon $ e
tende a $0$ ali uniformemente como $\varepsilon \to 0$
(\cref{thm:b3:lp:density}(3)): toda a expressĂŁo tende a
$0$.

(3) Aproximado $ f $ por $ g \in \mathcal C_c $
(\cref{thm:b3:lp:density}(2)), entĂŁo $ g $ por $ g *
\rho_\varepsilon \in \mathcal C_c^\infty $ (compactar suporte:
soma dos apoios).
\end{proof}

\begin{example}[Acalmador $\abs x $, com taxas]
\label{ex:b3:lp:mollifyexample}
Pegar $ f(x) = \abs x $ sobre $\R $ (localmente $ L^1$; o teorema
aplica-se a todas as janelas limitadas) e um simĂ©trico \omterm{def:b3:lp:mollifier}{calmante}
$\rho_\varepsilon $. EntĂŁo
\[

= \int\abs{x - y}\,\rho_\varepsilon(y)\,\dd y
\]

para $\abs x \geq \varepsilon $, $\abs{x - y}$ Ă© linear em $ x $
no apoio de $\rho_\varepsilon $, entĂŁo $ f_\varepsilon(x) =
\abs x $ \emph{exatamente} (simetria mata a correĂ§ĂŁo).
Aproximar $0$, suavizando custos com precisĂŁo
\[

y\,\rho_\varepsilon(y)\,\dd y \leq \varepsilon,
\qquad

\]
o erro de aproximaĂ§ĂŁo estĂˇ confinado ao
$\varepsilon $-vizinhanĂ§a da singularidade e Ă© de sua

\end{example}

\begin{corollary}[Lema fundamental do cĂˇlculo de
variaĂ§Ăµes]\label{cor:b3:lp:fundlemma}
Deixar $ f \in L^1_{\mathrm{loc}}(\R^d)$ (\omterm{def:b3:lebesgue:l1}{integrĂˇvel} sobre compactos)
com $\int f\varphi = 0$ para cada $\varphi \in \mathcal
C^\infty_c(\R^d)$. EntĂŁo $ f = 0$ a.e.\index{lema fundamental de
o cĂˇlculo das variaĂ§Ăµes}
\end{corollary}

\begin{proof}
Consertar uma bola $ B = B(0, R)$ e deixe $ g = f\mathbf 1_{B(0, R+1)}
\in L^1$. Para $ x \in B $ e $\varepsilon < 1$: $(g *
\rho_\varepsilon)(x) = \int f(y)\rho_\varepsilon(x - y)\dd y =
0$, sendo a funĂ§ĂŁo de teste $ y \mapsto \rho_\varepsilon(x-y)
\in \mathcal C_c^\infty $. Mas $ g * \rho_\varepsilon \to g $ em
$ L^1$ (\cref{thm:b3:lp:regularization}): $ g = 0$ a.e.\ em $ B $;
escape $\R^d $.
\end{proof}

\section{O \texorpdfstring{$ L^p $}{Lp} geografia}

\begin{proposition}\label{prop:b3:lp:inclusions}
(a) Se $\mu(X) < \infty $ e $1 \leq p \leq q \leq \infty $,
entĂŁo $ L^q \subseteq L^p $ com $\norm f_p \leq
\mu(X)^{\frac1p - \frac1q}\,\norm f_q $.
(b) Ligado $\R^d $ (infinito \omterm{def:b3:measure:measure}{medir}) nĂŁo hĂˇ inclusĂµes:
para $ p \neq q $ existem funĂ§Ăµes em $ L^p\setminus L^q $.
(c) (InterpolaĂ§ĂŁo) Se $ p < r < q $ e $\alpha \in \intoo01$
Ă© definido por $\frac1r = \frac\alpha p + \frac{1 - \alpha}q $,
entĂŁo
\[
\norm f_r \leq \norm f_p^{\alpha}\,\norm f_q^{1 - \alpha} ;
\]
em particular $ L^p \cap L^q \subseteq L^r $.
\end{proposition}

\begin{proof}
(a)HĂ¶lder com expoentes $\frac qp $ e seu conjugado:
$\int\abs f^p\cdot 1 \leq \norm{\abs f^p}_{q/p}\,\norm
1_{(q/p)'} = \norm f_q^p\,\mu(X)^{1 - p/q}$ ($ q = \infty $
diretamente). (b) Perto $0$ e perto $\infty $, poderes $ x^{-\alpha}$
calibrar: \cref{exo:b3:lp:2}. (c) Escreva $\abs f^r = \abs
f^{r\alpha}\abs f^{r(1-\alpha)}$ e aplique HĂ¶lder com o
par conjugado $\frac p{r\alpha}$, $\frac q{r(1-\alpha)}$
(conjugado precisamente pela definiĂ§ĂŁo de $\alpha $):
$\int\abs f^r \leq \norm f_p^{r\alpha}\norm f_q^{r(1 -
\alpha)}$.
\end{proof}

\begin{method}\label{met:b3:lp:toolkit}
O $ L^p $ kit de ferramentas, conforme usado abaixo: para provar uma
identidade ou desigualdade para todos $ f \in L^p $ --- prove isso em um
classe densa ($\mathcal C_c^\infty $ atravĂ©s de
\cref{thm:b3:lp:regularization}) e estender por continuidade
(\cref{thm:b3:complete:extension}, ambos os lados sendo
$ L^p $-contĂ­nuo); provar $ f = 0$, teste contra $\mathcal
C_c^\infty $ (\cref{cor:b3:lp:fundlemma}); para ganhar suavidade,
convolver; para expoentes comerciais, HĂ¶lder e interpolaĂ§ĂŁo. O
Teoria de Fourier de \cref{ch:b3:fouriertransform} Ă© um longo
aplicaĂ§ĂŁo deste mĂ©todo.
\end{method}

\section{Problema: desigualdade de Hardy}

\begin{problem}[{Problema de fim de semana --- Desigualdade de Hardy e
sua constante acentuada}]\label{pb:b3:lp:1}
Para $ f \in L^p(\intoo0{+\infty})$, $1 < p < \infty $, definir
o \emph{Operador resistente}
\[
(Hf)(x) = \frac1x\int_0^x f(t)\,\dd t .
\]
A desigualdade de Hardy (1920) afirma\index{Desigualdade de Hardy}
\[
\norm{Hf}_p \;\leq\; \frac{p}{p-1}\,\norm f_p ,
\]
e a constante $\frac p{p-1}$ Ă© Ăłtimo e nĂŁo alcanĂ§ado.
Este problema prova tudo e depois se estende Ă s sĂ©ries.

\medskip
\noindent\textbf{Part I --- The inequality.} Suponha primeiro $ f
\geq 0$ \omterm{def:b3:topology:continuity}{contĂ­nuo} com compactar suporte em
$\intoo0{+\infty}$, e deixe $ F(x) = \int_0^xf $.
\begin{enumerate}
  \item Mostre isso $ Hf \in L^p $: aproximar $0$, $ F $ desaparece em um
        \omterm{def:b3:topology:topology}{vizinhanĂ§a} de $0$; aproximar $\infty $, $ F $ Ă© limitado, entĂŁo
        $(Hf)(x) = O(1/x)$, e $ x \mapsto \frac1x $ pertence a
        $ L^p(\intoo1{+\infty})$ para $ p > 1$.
  \item Integrar por partes para mostrar
        \[
        \int_0^\infty \Bigl(\frac Fx\Bigr)^p\dd x
        = \frac{p}{p-1}\int_0^\infty\Bigl(\frac
        Fx\Bigr)^{p-1}f(x)\,\dd x .
        \]
        \emph{(Diferenciar $ x^{1-p}F^p $; termos de limite
        desaparecer --- justificar ambas as extremidades.)}
  \item Aplicar HĂ¶lder para o lado direito e deduzir
        $\norm{Hf}_p \leq \frac p{p-1}\norm f_p $ para tal $ f $.
  \item Estender a todos $ L^p $: para $ f \geq 0$, construir
        $ f_n $ \omterm{def:b3:topology:continuity}{contĂ­nuo} com compactar suporte em
        $\intoo0{+\infty}$, $0 \leq f_n \nearrow f $ a.e.\
        \emph{(truncar e depois aproximar monotonicamente ---
        justificar a construĂ§ĂŁo)}; entĂŁo $ Hf_n \nearrow Hf $
        pontualmente (MCT dentro da mĂ©dia) e MCT passa o
        desigualdade ao limite. Para assinado ou complexo $ f $,
        concluir com $\abs{Hf} \leq H\abs f $.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Optimality.}
\begin{enumerate}[resume]
  \item Para $ A > 1$ deixar $ f_A(t) = t^{-1/p}\,\mathbf
        1_{\intcc1A}(t)$. Calcular $\norm{f_A}_p^p = \ln A $
        e, por $1 \leq x \leq A $,
        \[
        (Hf_A)(x) = \frac p{p-1}\;x^{-1/p}\,
        \bigl(1 - x^{-(1 - 1/p)}\bigr).
        \]
  \item Deduzir $\liminf_{A\to\infty} \norm{Hf_A}_p/
        \norm{f_A}_p \geq \frac{p}{p-1}$, e concluir que
        a constante Ă© Ăłtima.
  \item Mostre essa igualdade $\norm{Hf}_p = \frac
        p{p-1}\norm f_p $ com $ f \neq 0$ Ă© impossĂ­vel.
        \emph{(Acompanhe o caso de igualdade de HĂ¶lder em questĂŁo
        3: forĂ§aria $ f = cx^{-1/p}$ comportamento do tipo, que
        nĂŁo estĂˇ em $ L^p $.)}
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The discrete inequality.}
\begin{enumerate}[resume]
  \item Para um nĂŁo crescente $ g \geq 0$ sobre $(0,\infty)$ e
        $ a_n = g(n)$, comparar $\sum a_n^p $ e $\int g^p $,
        e $ H $-mĂ©dias em conformidade, para deduzir da Parte I
        \emph{Desigualdade discreta de Hardy}: para $ a_n \geq 0$,
        \[
        \sum_{n\geq1}\Bigl(\frac{a_1 + \dots +
        a_n}{n}\Bigr)^{p}
        \;\leq\;
        \Bigl(\frac{p}{p-1}\Bigr)^{p}\,\sum_{n\geq1}a_n^p
        \]
        --- prove primeiro para nĂŁo crescente $(a_n)$ atravĂ©s do
        comparaĂ§ĂŁo acima, entĂŁo reduza o caso geral ao
        nĂŁo crescente por rearranjo \emph{(admita, com
        uma justificativa de uma linha, essa classificaĂ§ĂŁo $(a_n)$ em
        ordem decrescente sĂł pode aumentar o lado esquerdo enquanto
        consertando a direita)}.
  \item Deduza: se $\sum a_n^p < \infty $ entĂŁo o CesĂ ro
        meio de $(a_n)$ estĂŁo de novo $\ell^p $ --- e dĂŞ um
