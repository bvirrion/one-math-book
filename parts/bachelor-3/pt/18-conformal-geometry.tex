\chapter{Mapas Conformes e o Teorema de Mapeamento de Riemann}
\label{ch:b3:conformal}

UM \omterm{def:b3:holomorphic:holo}{holomórfico} bijeção entre dois domínios transporta todos
análise complexa de um para o outro: tais mapas ---
\emph{conforme}, porque preservam ângulos --- são os
isomorfismos do \omterm{def:b3:holomorphic:holo}{holomórfico} mundo. Este capítulo classifica
onde a classificação é possível (o disco, o plano:
O lema de Schwarz é a chave, uma desigualdade de surpreendente
potência), constrói a teoria da compacidade de \omterm{def:b3:holomorphic:holo}{holomórfico} famílias
(Montel), e prova o teorema de existência mais profundo do
assunto: \emph{todo} \omterm{def:b3:conformal:simplyconnected}{simplesmente conectado} subdomínio adequado de
$\C $, por mais irregular que seja seu limite, é conformemente equivalente
para o disco da unidade. Fechamos com funções harmônicas e o
Kernel de Poisson, resolvendo o problema de Dirichlet no disco ---
a recompensa analítica da geometria conforme. Por todo,
$\mathbb D = D(0,1)$ e $\mathbb H = \{\operatorname{Im}z >
0\}$.

\section{Mapas conformes; transformações de Möbius}

\begin{definition}\label{def:b3:conformal:conformal}
UM \emph{mapa conforme}\index{mapa conforme} (ou
biholomorfismo) entre conjuntos abertos é um \omterm{def:b3:holomorphic:holo}{holomórfico} bijeção;
seu inverso é automaticamente \omterm{def:b3:holomorphic:holo}{holomórfico}
(\cref{cor:b3:residues:openmapping}). Dois domínios são
\emph{conformemente equivalente} se tal mapa existir;
$\operatorname{Aut}(\Omega)$ denota o grupo de conformes
automapas. Onde $ f' \neq 0$ --- em todos os lugares, para injetivo $ f $
(\cref{cor:b3:residues:openmapping}prova) --- o
diferencial é multiplicação por $ f'(z) \neq 0$: um
similaridade, então os mapas conformes preservam os ângulos entre as curvas,
incluindo orientação.
\end{definition}

\begin{example}[transformações de Möbius]
\label{ex:b3:conformal:mobius}
Para $\bigl(\begin{smallmatrix}a & b\\ c &
d\end{smallmatrix}\bigr) \in GL_2(\C)$, o \emph{transformação de Möbius}\index{transformação de Möbius} $ z \mapsto
\frac{az + b}{cz + d}$ é conforme de
$\C\setminus\{-d/c\}$ para $\C\setminus\{a/c\}$ (inverso de
do mesmo tipo, da matriz inversa; composição
corresponde ao produto da matriz). O \emph{Cayley
mapa}\index{Cayley
mapa}
\[
\varphi(z) = \frac{z - \iu}{z + \iu}
\]
mapas $\mathbb H $ conformemente para $\mathbb D $: de fato $\abs{z
- \iu} < \abs{z + \iu}$ exatamente quando $ z $ está mais perto de $\iu $
do que $-\iu $, ou seja\ $\operatorname{Im}z > 0$; o inverso é
$ w \mapsto \iu\frac{1 + w}{1 - w}$. Mmapas obius enviam o
círculos e linhas familiares para si mesmo (\cref{exo:b3:conformal:1}).
\end{example}

\begin{example}[O mapa de Joukowski]\label{ex:b3:conformal:joukowski}
Além de Möbius, o mais útil \omterm{def:b3:conformal:conformal}{mapa conforme} de clássico
matemática aplicada é
\[
J(z) = \frac12\Bigl(z + \frac1z\Bigr) .
\]
No exterior $\Omega = \{\abs z > 1\}$ do disco unitário,
$ J $ é injetivo: $ J(z) = J(w)$ dá $(z - w)(1 -
\frac1{zw}) = 0$ e $\abs{zw} > 1$. Sua derivada $ J'(z) =
\frac12(1 - z^{-2})$ desaparece apenas em $ z = \pm1$, no
limite: $ J $ é uma equivalência conforme de $\Omega $ para
sua imagem, que é $\C\setminus\intcc{-1}1$ --- a unidade
o próprio círculo é dobrado dois para um no segmento
($ J(\eu^{\iu\theta}) = \cos\theta $). Assim, o exterior de um
\emph{segmento}, um plano de fenda sem limite suave, é
conformemente o exterior de um disco: os cantos não são obstáculo
para equivalência conforme, apenas para suavidade de contorno. Imagens
de círculos através $\pm1$ mas descentralizado são em forma de aerofólio
curvas e compondo $ J $ com Mmapas obius transportaram o
fluir passando por um cilindro --- computável manualmente --- para o fluxo
além de uma ala: durante a primeira metade do século XX,
este exemplo \emph{era} aerodinâmica. É também a porta para
Chebyshev: $ J $ conjugados $ z \mapsto z^n $ para o Chebyshev
polinômio $ T_n $ (\cref{pb:b3:hilbert:1}, Parte V), uma vez que
$ J(z^n) = \cos(n\theta)$ quando $ z = \eu^{\iu\theta}$.
\end{example}

\section{Lema de Schwarz e grupos de automorfismo}

\begin{theorem}[Lema de Schwarz]\label{thm:b3:conformal:schwarz}
Deixar $ f \colon \mathbb D \to \mathbb D $ ser \omterm{def:b3:holomorphic:holo}{holomórfico} com
$ f(0) = 0$. Então\index{Lema de Schwarz}
\[
\abs{f(z)} \leq \abs z \ \ (z \in \mathbb D)
\qquad\text{and}\qquad \abs{f'(0)} \leq 1 ,
\]
e se $\abs{f(z_0)} = \abs{z_0}$ por um $ z_0 \neq 0$, ou
$\abs{f'(0)} = 1$, então $ f(z) = \eu^{\iu\theta}z $ é um
rotação.
\end{theorem}

\begin{proof}
$ g(z) = f(z)/z $ estende holomorficamente para $\mathbb D $ (o
singularidade em $0$ é removível: $ g $ é delimitado perto $0$,
\cref{thm:b3:residues:riemanncw}; seu valor em $0$ é
$ f'(0)$). Sobre $\abs z = r < 1$: $\abs g \leq \frac1r $, então por
o princípio \omterm{def:b3:rings:primemaximal}{máximo} (\cref{thm:b3:holomorphic:maximum})
$\abs g \leq \frac1r $ sobre $\bar D(0,r)$; deixar $ r \to 1$: $\abs
g \leq 1$ sobre $\mathbb D $, que são ambas desigualdades. Igualdade
em um \omterm{def:b3:topology:interior}{interior} ponto faz $\abs g $ atingir um \omterm{def:b3:topology:interior}{interior}
\omterm{def:b3:rings:primemaximal}{máximo}: $ g $ constante de módulo $1$.
\end{proof}

\begin{theorem}[Automorfismos do disco]
\label{thm:b3:conformal:autdisc}
Para $ a \in \mathbb D $, o \emph{Blaschke
fator}\index{Blaschke
fator}
\[
\varphi_a(z) = \frac{z - a}{1 - \bar a z}
\]
é um automorfismo de $\mathbb D $ trocando $ a $ e $0$, com
$\varphi_a^{-1} = \varphi_{-a}$. Todo automorfismo de
$\mathbb D $ é $\eu^{\iu\theta}\varphi_a $ para único $\theta
\in \R/2\pi\Z $, $ a \in \mathbb D $.
\end{theorem}

\begin{proof}
Sobre $\abs z = 1$: $\abs{1 - \bar az} = \abs{\bar z}\abs{1 -
\bar az} = \abs{\bar z - \bar a\abs z^2} = \abs{\bar z - \bar
a} = \abs{z - a}$, então $\abs{\varphi_a} = 1$ lá; pelo
princípio \omterm{def:b3:rings:primemaximal}{máximo} $\varphi_a(\mathbb D) \subseteq \bar{\mathbb
D}$, e a abertura coloca a imagem em $\mathbb D $. O \omterm{def:b3:galois:algebraic}{algébrico}
identidade $\varphi_{-a}\circ\varphi_a = \mathrm{id}$ (direto
computação) mostra bijetividade. Agora deixe $ f \in
\operatorname{Aut}(\mathbb D)$ e $ a = f^{-1}(0)$: $ g =
f\circ\varphi_{-a}$ é uma fixação de automorfismo $0$. Preto
aplicado a $ g $ e para $ g^{-1}$: $\abs{g(z)} \leq \abs z $ e
$\abs{g^{-1}(w)} \leq \abs w $, então $\abs{g(z)} = \abs z $:
rotação, $ g = \eu^{\iu\theta}\,\mathrm{id}$, ou seja\ $ f =
\eu^{\iu\theta}\varphi_a $. Singularidade: $ a = f^{-1}(0)$ e
$\theta $ de $ f'$ avaliação de tipo (ou de $ f(0) =
-\eu^{\iu\theta}a $ e mais um valor).
\end{proof}

\begin{theorem}[Automorfismos do plano]
\label{thm:b3:conformal:autplane}
$\operatorname{Aut}(\C) = \{z \mapsto az + b : a \in \C^*,\ b
\in \C\}$. Consequentemente $\C $ e $\mathbb D $ não são
conformemente equivalente.
\end{theorem}

\begin{proof}
Deixar $ f \in \operatorname{Aut}(\C)$ e considere $ g(z) =
f(1/z)$ sobre $\C^*$: um função holomórfica com um isolado
singularidade no $0$. Se fosse essencial,
Casorati-Weierstrass (\cref{thm:b3:residues:riemanncw}) iria
fazer $ g\bigl(D(0,\varepsilon)\setminus\{0\}\bigr)$ \omterm{def:b3:topology:interior}{denso},
enquanto $ f(D(0, 1))$ é aberto e disjunto dele ($ f $
injetivo: os dois conjuntos são imagens de conjuntos disjuntos) ---
impossível para um conjunto \omterm{def:b3:topology:interior}{denso} e um não vazio conjunto aberto. Então $0$ é
um poste ou removível para $ g $, ou seja\ $\abs{f(z)}$ tem no \omterm{def:b3:rings:primemaximal}{máximo}
crescimento polinomial: $ f $ é um polinômio
(\cref{exo:b3:holomorphic:4}(um)). \omterm{def:b3:galois:extension}{Grau} de forças de injetividade
$1$: um polinômio de \omterm{def:b3:galois:extension}{grau} superior tem uma raiz múltipla de
$ f - c $ em algum lugar ($ f'$ desaparece) ou vários
pré-imagens (d'Alembert--Gauss, \cref{pb:b3:holomorphic:1});
de qualquer forma, a injetividade falha. Finalmente, um conformado $\C \to
\mathbb D $ seria uma função inteira limitada: constante
(Liouville) --- sem equivalência.
\end{proof}

\section{Teorema de Montel}

\begin{theorem}[Montel]\label{thm:b3:conformal:montel}
Deixar $\mathcal F \subseteq \mathcal H(\Omega)$ ser \emph{localmente
limitado}: cada ponto tem um \omterm{def:b3:topology:topology}{vizinhança} em que $\sup_{f\in
\mathcal F}\sup\abs f < \infty $. Então cada sequência de
$\mathcal F $ tem uma subsequência convergindo uniformemente em todos
compactar subconjuntos de $\Omega $ (para um \omterm{def:b3:holomorphic:holo}{holomórfico}
limite).\index{Teorema de Montel}
\end{theorem}

\begin{proof}
Local equicontinuidade: se $\abs f \leq M $ sobre $ D(a, 2r)
\subseteq \Omega $ para todos $ f \in \mathcal F $, o Cauchy
fórmula dá, para $ z, z' \in D(a, r)$,
\[
\abs{f(z) - f(z')} = \frac{\abs{z - z'}}{2\pi}
\Bigl|\int_{C_{2r}}\frac{f(w)\,\dd w}{(w-z)(w-z')}\Bigr|
\leq \frac{\abs{z - z'}\,2\pi\cdot2r\,M}{2\pi\,r^2}
= \frac{2M}{r}\,\abs{z - z'} :
\]
um limite uniforme de Lipschitz. Escape $\Omega $ por compactos $ K_m $;
cada $ K_m $ é coberto por um número finito desses discos, então
$\mathcal F $ é uniformemente limitado e \omterm{def:b3:complete:equicontinuous}{equicontínuo} sobre
$ K_m $: Arzelà--Ascoli (\cref{thm:b3:complete:ascoli})
extrai uma subsequência convergente uniformemente em $ K_m $;
diagonalizar sobre $ m $. O limite é \omterm{def:b3:holomorphic:holo}{holomórfico} por
\cref{thm:b3:holomorphic:weierstrassconv}.
\end{proof}

\section{O teorema do mapeamento de Riemann}

\begin{definition}\label{def:b3:conformal:simplyconnected}
Um aberto \omterm{def:b3:topology:connected}{conectado} $\Omega \subseteq \C $ é \emph{simplesmente
conectado}\index{simplesmente
conectado} (no homológico
sentido, suficiente para todos os nossos propósitos) se
$\operatorname{Ind}_\gamma(w) = 0$ para cada ciclo $\gamma $ em
$\Omega $ e cada $ w \notin \Omega $ --- ``nenhum ciclo de
$\Omega $ envolve um buraco''. Pelo teorema global de Cauchy
(\cref{thm:b3:residues:globalcauchy}) e
\cref{prop:b3:holomorphic:primitive}, em tal $\Omega $
\emph{todo função holomórfica tem um primitivo}; daí cada
\omterm{def:b3:modules:free}{livre} de zero $ f \in \mathcal H(\Omega)$ tem um \omterm{def:b3:holomorphic:holo}{holomórfico}
logaritmo ($\exp\circ $(primitivo de $ f'/f $), ajustado por um
constante, como $(f\eu^{-L})' = 0$) e \omterm{def:b3:holomorphic:holo}{holomórfico} $ n $-ésimas raízes
$\eu^{L/n}$.\index{logaritmo holomórfico}
\end{definition}

\begin{theorem}[Teorema de mapeamento de Riemann]
\label{thm:b3:conformal:rmt}
Todo \omterm{def:b3:conformal:simplyconnected}{simplesmente conectado} abrir $\Omega \subsetneq \C $, $\Omega
\neq \varnothing $, é conformemente equivalente para $\mathbb D $;
dado $ z_0 \in \Omega $, existe um conformal único $ f \colon
\Omega \to \mathbb D $ com $ f(z_0) = 0$ e $ f'(z_0) >
0$.\index{Teorema de mapeamento de Riemann}
\end{theorem}

\begin{proof}
\emph{Passo 0: a família não está vazia.} Escolha $ b \notin \Omega $:
$ z - b $ é \omterm{def:b3:modules:free}{livre} de zero em $\Omega $, então tem um \omterm{def:b3:holomorphic:holo}{holomórfico}
raiz quadrada $ h $ ($ h^2 = z - b $). $ h $ é injetivo ($ h(z) =
h(z')$ quadrados para $ z = z'$) e se $ w \in h(\Omega)$ então $-w
\notin h(\Omega)$ ($ h(z) = -h(z')$ também quadrados para $ z = z'$,
dando $ w = -w = 0$, impossível como $ h $ é \omterm{def:b3:modules:free}{livre} de zero). Desde
$ h(\Omega)$ está aberto, contém um disco $ D(h(z_0), \rho)$;
então $ D(-h(z_0), \rho) \cap h(\Omega) = \varnothing $, ou seja\
$\abs{h(z) + h(z_0)} \geq \rho $ para cada $ z \in \Omega $.
Por isso
\[
g(z) = \frac{\rho}{2\,\bigl(h(z) + h(z_0)\bigr)}
\]
é \omterm{def:b3:holomorphic:holo}{holomórfico}, injetivo (um Mmapa obius composto com o
injetivo $ h $), com $\abs g \leq \frac12 < 1$. Compondo com um Fator Blaschke
(\cref{thm:b3:conformal:autdisc}) para mover $ g(z_0)$ para $0$, o
família
\[
\mathcal F = \{f \colon \Omega \to \mathbb D \text{
holomorphic, injective, } f(z_0) = 0\}
\]
não está vazio.

\emph{Etapa 1: um elemento extremo.} Deixar $ s =
\sup_{\mathcal F}\abs{f'(z_0)} \in \intoc0{+\infty}$ ($> 0$:
membros são injetivos, então $ f'(z_0) \neq 0$). Pegar $ f_n \in
\mathcal F $ com $\abs{f_n'(z_0)} \to s $: a família é
limitado por $1$, então Montel (\cref{thm:b3:conformal:montel})
extratos $ f_n \to f $ uniformemente ligado compactos; $ f $ é
\omterm{def:b3:holomorphic:holo}{holomórfico}, $ f(z_0) = 0$, $\abs{f'(z_0)} = s $
(\cref{thm:b3:holomorphic:weierstrassconv} para o
derivados), em particular $ f $ é inconstante; $ f $ é
injetivo de Hurwitz (\cref{exo:b3:residues:8}(banda
$ f(\Omega) \subseteq \bar{\mathbb D}$, por isso $\subseteq
\mathbb D $ (mapeamento aberto). Então $ f \in \mathcal F $ atinge o
supremo: $ s < \infty $.

\emph{Etapa 2: o mapa extremo está ativado.} Suponha $ a \in
\mathbb D\setminus f(\Omega)$. O transporte Blaschke
$\varphi_a\circ f $ é \omterm{def:b3:modules:free}{livre} de zero no \omterm{def:b3:conformal:simplyconnected}{simplesmente conectado}
$\Omega $: tem um \omterm{def:b3:holomorphic:holo}{holomórfico} raiz quadrada $ F $ (com
$ F(\Omega) \subseteq \mathbb D $, como $\abs F^2 =
\abs{\varphi_a\circ f} < 1$), injetivo (quadrados
distinguir). Normalizar: $ G = \varphi_{F(z_0)}\circ F \in
\mathcal F $. Desfazendo: $ f = \varphi_{-a}\circ s_2 \circ
\varphi_{-F(z_0)}\circ G $ onde $ s_2(w) = w^2$; o mapa $\Psi
= \varphi_{-a}\circ s_2\circ\varphi_{-F(z_0)} \colon \mathbb D
\to \mathbb D $ é \omterm{def:b3:holomorphic:holo}{holomórfico} com $\Psi(0) = f(z_0) = 0$ e
é \emph{não} uma rotação (não é injetiva: $ s_2$ não é).
Lema de Schwarz (caso estrito): $\abs{\Psi'(0)} < 1$, e o
regra da cadeia $ f = \Psi\circ G $ dá $\abs{f'(z_0)} =
\abs{\Psi'(0)}\,\abs{G'(z_0)} < \abs{G'(z_0)}$ ---
contradizendo a maximalidade (nota $ G \in \mathcal F $). Por isso $ f $
é sobre: uma equivalência conforme.

\emph{Etapa 3: normalização e exclusividade.} Multiplicar $ f $ por
$\eu^{-\iu\arg f'(z_0)}$ fazer $ f'(z_0) > 0$ (isto permanece em
$\mathcal F $). Se $ f_1, f_2$ ambos trabalham, $\psi =
f_2\circ f_1^{-1} \in \operatorname{Aut}(\mathbb D)$ correções
$0$ com $\psi'(0) = f_2'(z_0)/f_1'(z_0) > 0$; por
\cref{thm:b3:conformal:autdisc} $\psi $ é uma rotação
$\eu^{\iu\theta}$ com $\eu^{\iu\theta} > 0$: $\psi =
\mathrm{id}$.
\end{proof}

\begin{remark}\label{rem:b3:conformal:rmtscope}
O teorema é uma declaração de existência pura de surpreendente
escopo: um quadrado, um semiplano, o complemento de uma fenda, o
região entre dois círculos tangentes, um domínio de limite fractal
--- todos conformemente idêntico a $\mathbb D $. O que não
give: qualquer fórmula (mapas explícitos são a exceção:
\cref{exo:b3:conformal:5}), comportamento de limite (uma teoria mais profunda
--- Carathteorema de eodory --- lida com isso), ou unicidade
de extensão para $\C $ ou multiplique \omterm{def:b3:topology:connected}{conectado} domínios: o
anel $\{1 < \abs z < 2\}$ é \emph{não} conformemente um
disco perfurado e anéis de diferentes proporções de raio são
inequivalente (um fato genuinamente mais difícil).
\end{remark}

\section{Funções harmônicas e o kernel de Poisson}

\begin{proposition}\label{prop:b3:conformal:harmonicholo}
Deixar $\Omega $ ser \omterm{def:b3:conformal:simplyconnected}{simplesmente conectado} e $ u \colon \Omega \to \R $
harmônico\index{função harmônica} ($\mathcal C^2$ com
$\Delta u = u_{xx} + u_{yy} = 0$). Então $ u =
\operatorname{Re}F $ por um \omterm{def:b3:holomorphic:holo}{holomórfico} $ F $, único até um
constante imaginária. Consequentemente $ u $ é $\mathcal C^\infty $,
satisfaz a propriedade do valor médio e obedece ao \omterm{def:b3:rings:primemaximal}{máximo}
princípio (sem estrito \omterm{def:b3:topology:interior}{interior} extremo, a menos que constante).
\end{proposition}

\begin{proof}
$ g = u_x - \iu u_y $ satisfaz o Equações de Cauchy-Riemann
($ P = u_x $, $ Q = -u_y $: $ P_x = u_{xx} = -u_{yy} = Q_y $ e
$ P_y = u_{xy} = u_{yx} = -Q_x $) com \omterm{def:b3:topology:continuity}{contínuo} parciais: $ g $
é \omterm{def:b3:holomorphic:holo}{holomórfico} (\cref{prop:b3:holomorphic:cauchyriemann}; o
$\R $-diferenciabilidade segue de $\mathcal C^1$). Deixar $ F_0$
ser um primitivo (conectividade \omterm{def:b3:groups:simple}{simples},
\cref{def:b3:conformal:simplyconnected}); então
$\operatorname{Re}F_0$ tem gradiente $(u_x, u_y)$
($ F_0' = g $ descompacta exatamente isso via Cauchy--Riemann para
$ F_0$), então $ u - \operatorname{Re}F_0$ é constante ($\Omega $
\omterm{def:b3:topology:connected}{conectado}): ajustar $ F = F_0 + c $. A transferência de propriedades
de \cref{thm:b3:holomorphic:maximum} e
\cref{exo:b3:holomorphic:10} (para o princípio \omterm{def:b3:rings:primemaximal}{máximo}
aplicado a $ u $ em si, use $\eu^{F}$ como lá).
\end{proof}

\begin{theorem}[Fórmula de Poisson; Problema de Dirichlet no
disco]\label{thm:b3:conformal:poisson}
Para $0 \leq r < 1$ definir o \emph{Poisson
núcleo}\index{Poisson
núcleo}
\[
P_r(\theta) = \sum_{n\in\Z}r^{\abs n}\eu^{\iu n\theta}
= \frac{1 - r^2}{1 - 2r\cos\theta + r^2} \;>\; 0 .
\]
Deixar $ g \colon \partial\mathbb D \to \R $ ser contínuoe
definir, para $ z = r\eu^{\iu\varphi} \in \mathbb D $,
\[
u(z) = \frac1{2\pi}\int_0^{2\pi}
P_r(\varphi - t)\,g(\eu^{\iu t})\,\dd t .
\]
Então $ u $ é harmônico em $\mathbb D $ e estende continuamente
para $\bar{\mathbb D}$ com valores limite $ g $: o único tal
\omterm{prop:b3:conformal:harmonicholo}{função harmônica}.\index{Problema de Dirichlet}
\end{theorem}

\begin{proof}
\emph{Identidades do kernel}: somando duas séries geométricas,
\[
\sum_{n\in\Z}r^{\abs n}\eu^{\iu n\theta}
= \operatorname{Re}\frac{1 + r\eu^{\iu\theta}}{1 -
r\eu^{\iu\theta}}
= \frac{1 - r^2}{\abs{1 - r\eu^{\iu\theta}}^2},
\]
qual é o quociente exibido; a positividade é clara e
$\frac1{2\pi}\int_0^{2\pi}P_r = 1$ (apenas $ n = 0$ sobrevive).

\emph{Harmonicidade}: com $ z = r\eu^{\iu\varphi}$,
\[
u(z) = \operatorname{Re}\biggl[\frac1{2\pi}\int_0^{2\pi}
\frac{\eu^{\iu t} + z}{\eu^{\iu t} - z}\,
g(\eu^{\iu t})\,\dd t\biggr],
\]
(o kernel entre colchetes tem parte real $ P_r(\varphi - t)$:
calcular), e o colchete é \omterm{def:b3:holomorphic:holo}{holomórfico} em $ z $ sobre $\mathbb D $
(\cref{exo:b3:holomorphic:7}): $ u $ é a parte real de um
função holomórfica, portanto harmônico.

\emph{Valores limite}: $ P_r(\cdot)$ é um aproximado
identidade como $ r \to 1^-$: massa $1$, e para $\delta \leq
\abs\theta \leq \pi $, $ P_r(\theta) \leq \frac{1 - r^2}{1 -
2r\cos\delta + r^2} \to 0$ uniformemente. A divisão padrão
(continuidade de $ g $ aproximar $\eu^{\iu\varphi_0}$, limitação
em outro lugar) dá $ u(r\eu^{\iu\varphi}) \to
g(\eu^{\iu\varphi_0})$ como $ r\eu^{\iu\varphi} \to
\eu^{\iu\varphi_0}$, uniformemente no ponto limite: o
extensão é \omterm{def:b3:topology:continuity}{contínuo}. \emph{Singularidade}: a diferença de
duas soluções é harmônica em $\mathbb D $, \omterm{def:b3:topology:continuity}{contínuo} no
\omterm{def:b3:topology:interior}{encerramento}, zero no limite: pelo princípio do \omterm{def:b3:rings:primemaximal}{máximo}
(aplicado a $\pm $ a diferença), ela desaparece.
\end{proof}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:b3:conformal:1}
(a) Verifique se o Mapa de Cayley $\varphi(z) = \frac{z -
\iu}{z + \iu}$ é uma bijeção $\mathbb H \to \mathbb D $ com
o inverso declarado e calcule as imagens de $\iu $, $0$,
$1$, $\infty $ (limite).
(b) Mostre que $ z \mapsto 1/z $ mapeia círculos e linhas para
círculos e linhas. \emph{(Escreva sua equação comum
$\alpha\abs z^2 + \bar\beta z + \beta\bar z + \gamma = 0$,
$\alpha, \gamma \in \R $.)} Deduza o mesmo para todo Möbius
mapas.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b3:conformal:2}
Deixar $ f \colon \mathbb D \to \mathbb D $ ser \omterm{def:b3:holomorphic:holo}{holomórfico}.
(a) Se $ f(0) = 0$ e $ f(a) = a $ para alguns $ a \neq 0$, mostrar $ f
= \mathrm{id}$.
(b) Se $ f $ é um automorfismo com dois pontos fixos distintos
em $\mathbb D $, mostrar $ f = \mathrm{id}$
\emph{(conjugado por um Fator Blaschke reduzir para (a))}.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:conformal:3}
(Schwarz - Escolha) Para \omterm{def:b3:holomorphic:holo}{holomórfico} $ f\colon \mathbb D \to
\mathbb D $, provar
\[
\frac{\abs{f'(z)}}{1 - \abs{f(z)}^2} \;\leq\;
\frac{1}{1 - \abs z^2}
\qquad (z \in \mathbb D),
\]
com igualdade (em um ponto, portanto em todos os lugares) se $ f \in
\operatorname{Aut}(\mathbb D)$.
\emph{(Aplique Schwarz para $\varphi_{f(z)}\circ
f\circ\varphi_{-z}$.)} Interpretação: \omterm{def:b3:holomorphic:holo}{holomórfico} automapas
contraia a métrica hiperbólica.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:conformal:4}
Encontre equivalências conformes explícitas:
(a) a tira $\{0 < \operatorname{Im}z < \pi\} \to \mathbb H $;
(b) o quadrante $\{\operatorname{Re}z > 0,
\operatorname{Im}z > 0\} \to \mathbb H $;
(c) o meio disco $\mathbb D\cap\mathbb H \to $ um quadrante,
então $\to \mathbb H $;
(d) $\mathbb D \to \mathbb D $ enviando $\frac12$ para $0$ com
derivada positiva ali.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:conformal:5}
(a) Mostre que não \omterm{def:b3:conformal:conformal}{mapa conforme} $\C \to \mathbb D $ ou $\C \to
\mathbb H $ existe e nenhum $\mathbb D \to \C $.
(b) Quais dos seguintes são conformemente equivalente para
$\mathbb D $? Justificar através \cref{thm:b3:conformal:rmt} ou um
obstrução: um quadrado; $\C\setminus\intoc{-\infty}0$;
$\mathbb D\setminus\{0\}$; $\{1 < \abs z < 2\}$.
\emph{(Para os dois últimos: uma imagem conforme do perfurado
o disco se estenderia sobre o furo por
\cref{thm:b3:residues:riemanncw}(1) --- desenvolva isso.)}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:conformal:6}
Deixar $\mathcal F = \{f \in \mathcal H(\mathbb D) : f(0) = 1,\
\operatorname{Re}f > 0\}$.
(a) Mostre que $\mathcal F $ é localmente limitado.
\emph{(Componha com o mapa tipo Cayley $ w \mapsto \frac{w -
1}{w + 1}$ enviando o semiplano direito para $\mathbb D $ e
aplique Schwarz.)}
(b) Deduza o \emph{Herglotz com destino}: $\abs{f(z)} \leq
\frac{1 + \abs z}{1 - \abs z}$ para $ f \in \mathcal F $, com
possibilidades de igualdade.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b3:conformal:7}
Onde está a prova de \cref{thm:b3:conformal:rmt} usar cada
hipótese? Trace: (i) conectividade \omterm{def:b3:groups:simple}{simples} (duas vezes); (ii)
$\Omega \neq \C $; (iii) conectividade. Mostre então que
teorema \emph{falhar} para $\Omega = \C $ e para o anel,
identificar qual etapa da prova quebra em cada caso.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:conformal:8}
Resolva o problema de Dirichlet em $\mathbb D $ para o limite
dados: (a) $ g(\eu^{\iu t}) = \cos t $; (b) $ g(\eu^{\iu t}) =
\cos^2 t $; (c) $ g = \mathbf 1_{\text{upper semicircle}}$ ---
para (c) calcular $ u(0)$ e interpretar através do valor médio
propriedade. \emph{(Expandir $ g $ em série de Fourier e uso
$ P_r $ série de: $ u(r\eu^{\iu\varphi}) = \sum_n c_n(g)
r^{\abs n}\eu^{\iu n\varphi}$.)}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b3:conformal:9}
(Harnack) Deixe $ u \geq 0$ ser harmônico em $\mathbb D $. Provar,
para $\abs z = r < 1$:
\[
\frac{1 - r}{1 + r}\,u(0) \;\leq\; u(z) \;\leq\;
\frac{1 + r}{1 - r}\,u(0)
\]
\emph{(vinculado o Kernel de Poisson entre $\frac{1-r}{1+r}$ e
$\frac{1+r}{1-r}$; aplique o \omterm{def:b3:representations:rep}{representação} ligeiramente
discos menores e passar até o limite)}. Deduzir: um harmônico
função sobre $\C $ limitado abaixo é constante.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:conformal:10}
Usando invariância conforme de harmonicidade ($ u\circ f $ é
harmônico quando $ u $ é harmônico e $ f $ \omterm{def:b3:holomorphic:holo}{holomórfico} --- provar
através \cref{prop:b3:conformal:harmonicholo} localmente), resolva
o problema de Dirichlet no semiplano superior com limite
dados $\mathbf 1_{\intoo{-\infty}0}$: mostre isso
\[
u(x + \iu y) = \frac1\pi\,\arg(x + \iu y)
\qquad (\arg \in \intoo0\pi \text{ on } \mathbb H)
\]
é harmônico em $\mathbb H $ (parte imaginária de um \omterm{def:b3:holomorphic:holo}{holomórfico}
logaritmo) com os limites exigidos em cada $ x \neq
0$, e transportá-lo para o disco de Cayley para derivar novamente
\cref{exo:b3:conformal:8}(c).
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:conformal:11}
(Pontos fixos e iteração no disco) Deixe $ f\colon\mathbb
D \to \mathbb D $ ser \omterm{def:b3:holomorphic:holo}{holomórfico}.
(a) Mostre que se $ f $ tem \emph{dois} pontos fixos distintos,
então $ f = \mathrm{id}$ \emph{(mova um para $0$ por um
automorfismo e aplicar o caso de igualdade de Schwarz)}.
(b) Suponha $ f(0) = 0$ e $ f $ não é uma rotação. Mostre isso
as iterações $ f^{\circ n} \to 0$ uniformemente em cada compactar
$\bar D(0, r)$, $ r < 1$ \emph{(Schwarz dá $\abs{f(z)}
\leq c_r\abs z $ sobre $\bar D(0,r)$ com $ c_r < 1$ --- justificar
esta constante estrita através do princípio \omterm{def:b3:rings:primemaximal}{máximo} aplicado a
$ f(z)/z $)}.
(c) Ilustrar com $ f(z) = \frac{z^2 + z}2$: pontos fixos,
e a taxa de convergência do \omterm{def:b3:groups:action}{órbita} de $ z_0 =
\frac12$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:conformal:12}
(Conjugados harmônicos, concretamente) Deixe $ u(x, y) = x^3 - 3xy^2
+ 2y $.
(a) Verifique se $ u $ é harmônico em $\R^2$ e encontre tudo
conjugados harmônicos $ v $ (ou seja,\ $ u + \iu v $ \omterm{def:b3:holomorphic:holo}{holomórfico}) por
integrando o Equações de Cauchy-Riemann; identificar $ f(z) =
u + \iu v $ como um polinômio em $ z $.
(b) Mostre que em um \emph{em forma de estrela} conjunto aberto, todo
\omterm{prop:b3:conformal:harmonicholo}{função harmônica} admite um conjugado harmônico, único até
uma constante aditiva \emph{(o $1$-forma $-u_y\,\dd x +
u_x\,\dd y $ está fechado; \cref{thm:b3:holomorphic:cauchy}de
maquinaria primitiva, ou \cref{ch:b3:forms}de Poincaré
lema)}.
(c) Dê o contra-exemplo padrão em
$\C^*$: $ u = \ln\abs z $ não tem conjugado global --- relacionar
para a forma angular e o número do enrolamento
(\cref{ch:b3:forms}).
\end{exercise}

\section{Problema: o teorema da área e o quarto de Koebe
teorema}

\begin{problem}[{Problema de fim de semana --- quanto deve um
capa de mapa univalente?}]\label{pb:b3:conformal:1}
UM \emph{univalente} função é uma \omterm{def:b3:holomorphic:holo}{holomórfico} injeção. O
funções univalentes normalizadas no disco,
\[
\mathcal S = \bigl\{f \in \mathcal H(\mathbb D) \text{
injective},\ f(z) = z + a_2z^2 + a_3z^3 + \cdots\bigr\},
\]
são rigidamente restringidos: provamos a desigualdade de Bieberbach
$\abs{a_2} \leq 2$ e deduzir o \emph{Bairro Koebe
teorema}: a imagem de qualquer $ f \in \mathcal S $ contém o
disco $ D(0, \frac14)$ --- a constante universal nítida de
geometria conforme.\index{Teorema do quarto de Koebe}\index{área
teorema}

\medskip
\noindent\textbf{Part I --- The area theorem.} Deixar $ g(w) = w
+ b_0 + \frac{b_1}w + \frac{b_2}{w^2} + \cdots $ ser \omterm{def:b3:holomorphic:holo}{holomórfico}
e \emph{injetivo} sobre $\{\abs w > 1\}$.
\begin{enumerate}
  \item Para $\rho > 1$, deixar $ A_\rho $ seja a área (Lebesgue
        \omterm{def:b3:measure:measure}{medir}) do compactar definir $ K_\rho = \C\setminus
        g(\{\abs w > \rho\})$, a região delimitada pelo
        curva suave de Jordan $ g(C_\rho)$. Usando a área
        fórmula do teorema de Green--Riemann do volume do Ano 2
        --- a área fechada é $\frac1{2\iu}
        \oint\bar\zeta\,\dd\zeta $ ao longo do positivamente
        fronteira orientada --- mostre que
        \[
        A_\rho = \frac{1}{2\iu}\int_{C_\rho}
        \overline{g(w)}\,g'(w)\,\dd w
        = \pi\Bigl(\rho^2 -
        \sum_{n\geq1}n\,\abs{b_n}^2\rho^{-2n}\Bigr) :
        \]
        substitua o \omterm{thm:b3:residues:laurent}{Série Laurent} de $\bar g $ e $ g'$ sobre
        $ C_\rho $ e integrar termo a termo (\omterm{def:b3:groups:normal}{normal}
        convergência; apenas os produtos de frequência zero
        sobreviver).
  \item Deixar $\rho \to 1^+$ e concluir o \emph{área
        teorema}:
        \[
        \sum_{n\geq1}n\,\abs{b_n}^2 \;\leq\; 1 .
        \]
        Em particular $\abs{b_1} \leq 1$. Quando é $\abs{b_1}
        = 1$?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Bieberbach's $\abs{a_2} \leq
2$.} Deixar $ f = z + a_2z^2 + \cdots \in \mathcal S $.
\begin{enumerate}[resume]
  \item Mostre isso $ f(z^2)/z^2$ é \omterm{def:b3:holomorphic:holo}{holomórfico} e sem zero
        ligado $\mathbb D $, e admite uma \omterm{def:b3:holomorphic:holo}{holomórfico} raiz quadrada
        $\varphi $ com $\varphi(0) = 1$; definir $ h(z) =
        z\varphi(z^2)$, para que $ h(z)^2 = f(z^2)$. Mostre isso
        $ h $ é um \emph{chance} função univalente em $\mathbb
        D $ com expansão $ h(z) = z + \frac{a_2}2z^3 +
        \cdots $. \emph{(Injetividade: $ h(z)^2 = h(z')^2$
        forças $ z^2 = z'^2$; use estranheza para terminar.)}
  \item Aplique o teorema da área para $ g(w) = 1/h(1/w) = w -
        \frac{a_2}{2w} + \cdots $ sobre $\{\abs w > 1\}$
        (verifique a univalência e a expansão) e conclua
        $\abs{a_2} \leq 2$.
  \item Mostre que o \emph{Função Koebe}
        \[
        k(z) = \frac{z}{(1 - z)^2} = \sum_{n\geq1}n\,z^n
        \]
        pertence a $\mathcal S $, tem $ a_2 = 2$ e mapas
        $\mathbb D $ para $\C\setminus\intoc{-\infty}{-\frac14}$
        \emph{(escrever $ k = \frac14\bigl[\bigl(\frac{1+z}{1 -
        z}\bigr)^2 - 1\bigr]$ e acompanhe as imagens)}: todos
        as desigualdades que virão são acentuadas.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The quarter theorem.}
\begin{enumerate}[resume]
  \item Deixar $ f \in \mathcal S $ e $ c \notin f(\mathbb D)$.
        Mostre isso
        \[
        F(z) = \frac{c\,f(z)}{c - f(z)}
        \]
        pertence a $\mathcal S $, e calcule seu segundo
        coeficiente: $ A_2 = a_2 + \frac1c $.
  \item Aplique Bieberbach a ambos $ f $ e $ F $: concluir
        $\abs{\frac1c} \leq \abs{A_2} + \abs{a_2} \leq 4$,
        ou seja\ $\abs c \geq \frac14$. \emph{Cada valor omitido
        tem módulo $\geq \frac14$}: $ f(\mathbb D) \supseteq
        D(0, \frac14)$ --- Teorema do quarto de Koebe. Verifique
        nitidez na função Koebe.
  \item Deduza uma estimativa quantitativa do mapa de Riemann: se
        $\varphi \colon \Omega \to \mathbb D $ é o Riemann
        mapa de \cref{thm:b3:conformal:rmt} no $ z_0$, então
        \[
        \frac{d\bigl(z_0, \partial\Omega\bigr)}{4}
        \;\leq\; \frac1{\varphi'(z_0)} \;\leq\;
        4\,d\bigl(z_0, \partial\Omega\bigr)
        \]
        --- prove pelo menos a desigualdade à esquerda aplicando
        Koebe para $\varphi^{-1}$ adequadamente normalizado, e o
        direito de Schwarz aplicado a $\varphi $ no disco
        $ D(z_0, d) \subseteq \Omega $.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Perspective.}
\begin{enumerate}[resume]
  \item Bieberbach conjecturou (1916) $\abs{a_n} \leq n $ para
        tudo $ n $, com igualdade apenas para rotações do
        Função Koebe; de Branges provou isso em 1985. Verifique
        a conjectura manual para a função de Koebe e sua
        rotações $\eu^{-\iu\theta}k(\eu^{\iu\theta}z)$. Então
        empurre a expansão da questão 4 um termo adiante: escrever
        $ h(z) = z + \frac{a_2}2z^3 + c_5z^5 + \cdots $, mostrar
        $ c_5 = \frac{a_3}2 - \frac{a_2^2}8$ e
        \[
        g(w) = w - \frac{a_2}{2}\,w^{-1} +
        \Bigl(\frac{3a_2^2}8 - \frac{a_3}2\Bigr)w^{-3} +
        \cdots ,
        \]
        então o teorema da área produz a desigualdade refinada
        $\bigl|\frac{a_2}2\bigr|^2 + 3\bigl|\frac{3a_2^2}8 -
        \frac{a_3}2\bigr|^2 \leq 1$. Confira no Koebe
        função ($ a_2 = 2$, $ a_3 = 3$).
\end{enumerate}

\medskip
\noindent\textbf{Part V --- The distortion theorem.}
A desigualdade de Bieberbach, transportada pelo disco por
automorfismos, controles $ f'$ em todos os lugares. Consertar $ f \in \mathcal
S $.
\begin{enumerate}[resume]
  \item (transformada de Koebe) Para $ z_0 \in \mathbb D $ deixar
        $\varphi(z) = \frac{z + z_0}{1 + \bar z_0z}$, um disco
        automorfismo (\cref{thm:b3:conformal:autdisc}) com
        $\varphi(0) = z_0$. Mostre isso
        \[
        F(z) = \frac{f(\varphi(z)) - f(z_0)}
        {f'(z_0)\,\bigl(1 - \abs{z_0}^2\bigr)}
        \]
        pertence a $\mathcal S $ \emph{(univalência é
        herdado; calcular $\varphi'(0) = 1 - \abs{z_0}^2$
        e verifique a normalização; lembrar $ f' \neq 0$ para
        injetivo $ f $,
        \cref{def:b3:conformal:conformal})}.
  \item Calcule o segundo coeficiente $ A_2 = \frac12F''(0)$
        de $ F $:
        \[
        A_2 = \frac12\Bigl[\bigl(1 - \abs{z_0}^2\bigr)
        \frac{f''(z_0)}{f'(z_0)} - 2\bar z_0\Bigr] ,
        \]
        e deduzir de Bieberbach (questão 4), para $ z =
        r\eu^{\iu\theta}$, o \emph{fundamental
        desigualdade}:
        \[
        \Bigl|\,z\,\frac{f''(z)}{f'(z)} -
        \frac{2r^2}{1 - r^2}\Bigr|
        \;\leq\; \frac{4r}{1 - r^2} .
        \]
  \item Extraia a parte real:
        \[
        \frac{2r^2 - 4r}{1 - r^2} \;\leq\;
        \operatorname{Re}\Bigl(z\,\frac{f''(z)}{f'(z)}\Bigr)
        \;\leq\; \frac{2r^2 + 4r}{1 - r^2} .
        \]
  \item Mostre isso $\frac{\dd}{\dd t}\log\bigl|
        f'(t\eu^{\iu\theta})\bigr| = \frac1t
        \operatorname{Re}\bigl(z\frac{f''(z)}{f'(z)}\bigr)$
        no $ z = t\eu^{\iu\theta}$ \emph{(para um não-desaparecido
        $\mathcal C^1$ função $ g $ de uma variável real,
        $\frac{\dd}{\dd t}\log\abs g =
        \operatorname{Re}(g'/g)$)}e integre a pergunta
        limites de 12 ao longo do raio para obter o
        \emph{teorema da distorção}:
        \[
        \frac{1 - r}{(1 + r)^3} \;\leq\; \abs{f'(z)}
        \;\leq\; \frac{1 + r}{(1 - r)^3},
        \qquad \abs z = r .
        \]
  \item Deduza o \emph{teorema do crescimento}:
        \[
        \frac{r}{(1 + r)^2} \;\leq\; \abs{f(z)} \;\leq\;
        \frac{r}{(1 - r)^2},
        \qquad \abs z = r
        \]
        \emph{(limite superior: integrar $ f'$ no segmento
        $[0, z]$; limite inferior: se $\abs{f(z)} < \frac14$,
        o segmento $[0, f(z)]$ reside em $ f(\mathbb D)$ por
        questão 7; puxe-o de volta $ f^{-1}$ ---
        \omterm{def:b3:holomorphic:holo}{holomórfico} por
        \cref{cor:b3:residues:openmapping} --- e vinculado
        $\abs{f(z)} = \int_\gamma\abs{f'(\zeta)}
        \,\abs{\dd\zeta} \geq \int_0^r\frac{1 - t}{(1 +
        t)^3}\,\dd t $, usando $\abs{\dd\zeta} \geq
        \dd\abs\zeta $)}.
  \item Verifique se a função de Koebe alcança igualdade em
        todos os quatro limites, em $ z = r $ para os superiores e
        $ z = -r $ para os inferiores: $ k $ é simultaneamente
        o mais expansivo e, no antípoda, o mais
        membro contratante da $\mathcal S $.
\end{enumerate}

\medskip
\noindent\textbf{Part VI --- Extremal rigidity.} Em cada
desigualdade até agora, a igualdade identifica a função Koebe
para rotação. Nós provamos isso e depois colhemos.
\begin{enumerate}[resume]
  \item Suponha $ f \in \mathcal S $ tem $\abs{a_2} = 2$.
        Persiga a igualdade por meio das questões 2 a 4: a área
        forças do teorema $ g(w) = w + b_0 + \eu^{\iu\alpha}/w $;
        estranheza de $ h $ faz $ g $ estranho, então $ b_0 = 0$; inverter
        encontrar $ h $, então $ f $, e concluir que
        \[
        f(z) = \eu^{-\iu\theta}k\bigl(\eu^{\iu\theta}z\bigr)
        \quad\text{with } \eu^{\iu\theta} =
        -\eu^{\iu\alpha} :
        \]
        as rotações da função Koebe são as únicas
        membros de $\mathcal S $ com $\abs{a_2} = 2$.
  \item Mostre que se $ f \in \mathcal S $ omite um valor $ c $
        com $\abs c = \frac14$ exatamente, então $ f $ é um
        rotação da função Koebe e identificar $ c =
        -\eu^{-\iu\theta}/4$ \emph{(traçar igualdade através
        cadeia da questão 7 $4 = \abs{1/c} = \abs{A_2 - a_2}
        \leq \abs{A_2} + \abs{a_2} \leq 4$)}: o trimestre
        a constante do teorema é alcançada apenas pelo extremo
        família.
  \item (Coeficientes baratos) Combine o crescimento
        teorema com o \omterm{thm:b3:holomorphic:analytic}{Estimativas de Cauchy}
        (\cref{thm:b3:holomorphic:analytic}) no círculo
        $\abs z = 1 - \frac1n $ provar
        \[
        \abs{a_n} \;\leq\; \eu\,n^2 \qquad (n \geq 2) .
        \]
        (De Branges, 1985: $\abs{a_n} \leq n $; o fator
        $\eu n $ é o preço das ferramentas elementares.)
  \item (Cobertura de subdiscos) Mostre que para cada $0 < r <
        1$,
        \[
        f\bigl(D(0, r)\bigr) \supseteq
        D\Bigl(0, \frac{r}{(1 + r)^2}\Bigr),
        \]
        afiado para a função Koebe e recuperar o
        teorema do quarto como $ r \to 1^-$. \emph{(Limite
        pontos da imagem aberta $ f(D(0,r))$ deite-se
        $ f(\partial D(0,r))$, portanto, tem módulo $\geq
        \frac{r}{(1+r)^2}$ pela questão 14; um segmento de
        $0$ para um ponto perdido de módulo menor teria
        para cruzar esse limite.)}
  \item (Koebe em todos os pontos) Vamos $ f $ ser univalente em
        $\mathbb D $, não necessariamente normalizado, e $ z_0
        \in \mathbb D $. Provar
        \[
        \tfrac14\bigl(1 - \abs{z_0}^2\bigr)\abs{f'(z_0)}
        \;\leq\;
        d\bigl(f(z_0), \partial f(\mathbb D)\bigr)
        \;\leq\;
        \bigl(1 - \abs{z_0}^2\bigr)\abs{f'(z_0)}
        \]
        \emph{(esquerda: teorema do quarto aplicado ao Koebe
        transformação da questão 10; à direita: Schwarz
        (\cref{thm:b3:conformal:schwarz}) aplicado a
        $\psi^{-1}\circ\hat g $, onde $\hat g(w) =
        f^{-1}\bigl(f(z_0) + dw\bigr)$, $ d $ a distância,
        e $\psi $ um automorfismo de disco enviando $0$ para
        $ z_0$)}. Por que é $\partial f(\mathbb D)$ não vazio?
  \item Verifique a questão 20 em $ f = k $ no $ z_0 = r \in
        \intoo01$: calcular $ d\bigl(k(r), \partial k(\mathbb
        D)\bigr) = \frac{(1+r)^2}{4(1-r)^2}$ e verifique isso
        a desigualdade à esquerda é uma \emph{igualdade}: o Koebe
        função satura seu próprio teorema em cada ponto de
        $\intoo01$.
  \item (A moral) Em dez linhas: que princípio único
        está subjacente ao teorema da área, e como Bieberbach,
        o teorema do quarto, distorção, crescimento e
        cobrindo todo o fluxo dele? Compare com o
        Schwarz - Escolha o mundo de
        \cref{thm:b3:conformal:schwarz}: em ambos, um
        \omterm{def:b3:topology:interior}{interior} a desigualdade enrijece toda a geometria,
        e os extremos são únicos até a rotação.

\end{enumerate}
\medskip
\noindent\textbf{Part VII --- Compactness, inverses, and a
reality check.}
\begin{enumerate}[resume]
  \item Mostre que a classe $\mathcal S $ é \emph{compactar}
        para convergência localmente uniforme: é localmente
        limitado pelo teorema do crescimento, portanto \omterm{def:b3:groups:normal}{normal}
        (\cref{thm:b3:conformal:montel}); e um local
        limite uniforme de membros de $\mathcal S $ está novamente em
        $\mathcal S $ \emph{(as normalizações passam para o
        limite pela convergência de derivadas de Weierstrass;
        a injetividade sobrevive por Hurwitz,
        \cref{exo:b3:residues:8}, sendo o limite
        inconstante)}. Por que isso importa para extremos
        problemas como o de Bieberbach?
  \item Para $ f \in \mathcal S $, deixar $ g = f^{-1}$, definido
        perto $0$. Mostrar $ g(w) = w - a_2w^2 + O(w^3)$, então o
        o segundo coeficiente do inverso obedece ao mesmo sustenido
        vinculado $\abs{A_2} = \abs{a_2} \leq 2$, com igualdade
        exatamente para as funções giradas de Koebe.
  \item Determinar para qual $ a \in \C $ o polinômio $ f(z)
        = z + az^2$ pertence a $\mathcal S $: mostrar $ f $ é
        injetivo em $\mathbb D $ se $\abs a \leq \frac12$
        \emph{(fator $ f(z_1) - f(z_2)$)}. Conclua: para
        polinômios de \omterm{def:b3:galois:extension}{grau} dois, o limite do coeficiente verdadeiro é
        $\frac12$, quatro vezes menor que o de Bieberbach $2$
        --- os extremos de $\mathcal S$ são genuinamente
        objetos transcendentais, e nenhum polinômio vem
        perto.
\end{enumerate}
\end{problem}
