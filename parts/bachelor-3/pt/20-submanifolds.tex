\chapter{Subvariedades de \texorpdfstring{$\R^n $}{Rn}}
\label{ch:b3:submanifolds}

Esferas, toros, grupos de rotação: os habitats naturais de
geometria e mecânica não são espaços vetoriais, mas \emph{curvo
conjuntos que parecem planos de perto}. Este capítulo fornece essa frase
um significado preciso --- subvariedades de $\R^n $ --- e o
cálculo para trabalhar neles. A \omterm{def:b3:topology:basis}{base} é o inverso
teorema da função, aqui provado pelo ponto fixo de Banach;
todo o resto é mudança de coordenadas: os quatro equivalentes
descrições de uma subvariedade (endireitamento local, nível
conjuntos, gráficos, parametrizações), espaços tangentese
otimização restrita por \omterm{thm:b3:submanifolds:lagrange}{Multiplicadores de Lagrange} --- que,
como uma demonstração de despedida, reprova o teorema espectral
para matrizes simétricas em três linhas de geometria. O
problema de fim de semana constrói o grupo de rotação $ SO(3)$ e seu
capa dupla quaterniônica: álgebra (\cref{ch:b3:groups}de
$ Q_8$ crescido) conhecendo a geometria.

\section{O teorema da função inversa}

\begin{theorem}[Teorema da função inversa]
\label{thm:b3:submanifolds:ift}
Deixar $ U \subseteq \R^n $ estar aberto, $ f \colon U \to \R^n $ de
aula $\mathcal C^1$, e $ a \in U $ com $ Df(a)$ invertível.
Então há conjuntos abertos $ V \ni a $, $ W \ni f(a)$ tal que
$ f \colon V \to W $ é uma bijeção com $\mathcal C^1$ inverso,
e\index{teorema da função inversa}
\[
D(f^{-1})(y) = \bigl(Df(f^{-1}(y))\bigr)^{-1}
\qquad (y \in W).
\]
Se $ f $ é $\mathcal C^k $, assim é $ f^{-1}$.
\end{theorem}

\begin{proof}
Normalizar: substituindo $ f $ por $ x \mapsto Df(a)^{-1}\bigl(f(a +
x) - f(a)\bigr)$, podemos assumir $ a = 0$, $ f(0) = 0$, $ Df(0) =
I $ (a declaração geral segue compondo com o
bijeções afins). Escrever $ f(x) = x + g(x)$: $ Dg(0) = 0$ e
por continuidade de $ Dg $ escolher $ r > 0$ com
$\vertiii{Dg(x)} \leq \frac12$ sobre $\bar B(0, r)$; a média
desigualdade de valor dá $\norm{g(x) - g(x')} \leq \frac12
\norm{x - x'}$ lá.

\emph{Bijetividade em um \omterm{def:b3:topology:topology}{vizinhança}.} Para $ y \in B(0,
\frac r2)$, resolvendo $ f(x) = y $ significa encontrar um ponto fixo de
$\Phi_y(x) = y - g(x)$; $\Phi_y $ mapas $\bar B(0, r)$ em
em si ($\norm{\Phi_y(x)} \leq \norm y + \frac12\norm x \leq
r $) e é $\frac12$-Lipschitz: Banach
(\cref{thm:b3:complete:banach}) fornece uma solução única
$ x = \varphi(y) \in \bar B(0,r)$. Além disso $ f $ é injetivo
ligado $\bar B(0,r)$:
\[
\norm{f(x) - f(x')} \geq \norm{x - x'} - \norm{g(x) - g(x')}
\geq \tfrac12\norm{x - x'} .
\tag{$*$}
\]
Definir $ W = B(0, \frac r2)$ e $ V = f^{-1}(W)\cap B(0, r)$:
abrir (continuidade), com $ f \colon V \to W $ bijetivo.

\emph{Continuidade e diferenciabilidade do inverso.} $(*)$
diz $\varphi = f^{-1}$ é $2$-Lipschitz. Consertar $ y_0 = f(x_0)
\in W $; invertibilidade de $ A = Df(x_0)$ (sua distância até $ I $
é $\leq \frac12$: Neumann,
\cref{prop:b3:banach:neumann}) e diferenciabilidade de $ f $
dar, por $ y = f(x)$ aproximar $ y_0$:
\[
\varphi(y) - \varphi(y_0) - A^{-1}(y - y_0)
= -A^{-1}\bigl(f(x) - f(x_0) - A(x - x_0)\bigr)
= -A^{-1}\,o(\norm{x - x_0}) = o(\norm{y - y_0}),
\]
usando $(*)$ converter $\norm{x - x_0} \leq 2\norm{y - y_0}$:
$\varphi $ é diferenciável em $ y_0$ com o inverso
diferencial. Continuidade de $ y \mapsto D\varphi(y) =
Df(\varphi(y))^{-1}$: composição de mapas \omterm{def:b3:topology:continuity}{contínuos}
(a inversão é \omterm{def:b3:topology:continuity}{contínuo},
\cref{prop:b3:banach:neumann}): $\varphi \in \mathcal C^1$;
inicializar a mesma fórmula dá $\mathcal C^k $.
\end{proof}

\begin{theorem}[Teorema da função implícita]
\label{thm:b3:submanifolds:implicit}
Deixar $ F \colon U \subseteq \R^p\times\R^q \to \R^q $ ser
$\mathcal C^1$ aproximar $(a, b)$, $ F(a,b) = 0$, e suponha que
diferencial parcial $ D_yF(a,b) \in \mathcal L(\R^q)$ é
invertível. Então há bairros $ A \ni a $, $ B \ni b $
e um $\mathcal C^1$ mapa $\psi \colon A \to B $ com
\[
\bigl\{(x, y)\in A\times B : F(x,y) = 0\bigr\}
= \{(x, \psi(x)) : x \in A\},
\]
e $ D\psi(x) = -D_yF(x, \psi(x))^{-1}\,D_xF(x,
\psi(x))$.\index{teorema da função implícita}
\end{theorem}

\begin{proof}
Aplicar \cref{thm:b3:submanifolds:ift} para $\Theta(x, y) = (x,
F(x,y))$: seu diferencial em $(a,b)$, bloco triangular com
blocos diagonais invertíveis $ I $ e $ D_yF $, é invertível. O
inverso local tem a forma $\Theta^{-1}(x, z) = (x, h(x,
z))$; definir $\psi(x) = h(x, 0)$: então $ F(x, y) = 0$ se
$\Theta(x,y) = (x, 0)$ se $ y = \psi(x)$, localmente. O
fórmula: diferenciar $ F(x, \psi(x)) = 0$ pela cadeia
regra.
\end{proof}

\section{Subvariedades: quatro definições}

\begin{theorem}[Caracterizações equivalentes]
\label{thm:b3:submanifolds:characterizations}
Deixar $ M \subseteq \R^n $, $ d \in \{0, \dots, n\}$, e $ k \geq
1$. Os seguintes são equivalentes, para cada ponto $ a \in M $
(e $ M $ é um \emph{$ d $ subvariedade -dimensional da classe
$\mathcal C^k $}\index{$ d $ subvariedade -dimensional da classe
$\mathcal C^k $} se eles segurarem a cada $ a
\in M $):
\begin{enumerate}
  \item (\emph{Alisamento}) Existe um $\mathcal C^k $
        difeomorfismo $\Phi $ de um aberto $\Omega \ni a $
        para um aberto $\Omega' \subseteq \R^n $ com
        \[
        \Phi(M\cap\Omega) = \Omega' \cap
        \bigl(\R^d\times\{0\}\bigr).
        \]
  \item (\emph{Nível definido}) Existe um $\mathcal C^k $
        \emph{submersão} $ F \colon \Omega \to \R^{n-d}$
        (ou seja,\ $ DF(x)$ sobrejetiva) em um aberto $\Omega \ni a $
        com $ M\cap\Omega = F^{-1}(0)$.
  \item (\emph{Gráfico}) Até permutar coordenadas, $ M $ é
        localmente o gráfico de um $\mathcal C^k $ mapa $\psi
        \colon A \subseteq \R^d \to \R^{n-d}$.
  \item (\emph{Parametrização}) Existe um $\mathcal C^k $
        \emph{imersão} $\varphi \colon A \subseteq \R^d \to
        \R^n $ ($ D\varphi(u)$ injetivo) com $ A $ abrir,
        $\varphi $ um \omterm{def:b3:topology:continuity}{homeomorfismo} de $ A $ para $ M \cap
        \Omega $ para alguns abertos $\Omega \ni a $.
\end{enumerate}
\end{theorem}

\begin{proof}
(1)$\Rightarrow $(2): $ F = (\Phi_{d+1}, \dots, \Phi_n)$ (último
coordenadas de $\Phi $): uma submersão ($ D\Phi $ invertível).
(2)$\Rightarrow $(3): $ DF(a)$ sobrejetivo: alguns $ q \times q $
menor do Jacobiano é invertível ($ q = n - d $); depois
permutando coordenadas, $ D_yF(a)$ é invertível e o
teorema da função implícita (\cref{thm:b3:submanifolds:implicit})
expressa $ M $ localmente como um gráfico $ y = \psi(x)$.
(3)$\Rightarrow $(4): $\varphi(x) = (x, \psi(x))$: um
imersão (diferencial $\bigl(\begin{smallmatrix}I\\
D\psi\end{smallmatrix}\bigr)$ injetivo), um \omterm{def:b3:topology:continuity}{homeomorfismo} para
o gráfico (inverso: a projeção, \omterm{def:b3:topology:continuity}{contínuo}).
(4)$\Rightarrow $(1): deixe $\varphi(u_0) = a $; completo
$\operatorname{im}D\varphi(u_0)$ por um suplemento $ E $
($\dim E = n - d $) e definir $\Theta(u, v) = \varphi(u) + v $
sobre $ A\times E $: $ D\Theta(u_0, 0)$ é bijetivo (imagem
contém $\operatorname{im}D\varphi(u_0)$ e $ E $), então
$\Theta $ é um difeomorfismo local
(\cref{thm:b3:submanifolds:ift}); é inverso $\Phi $
endireita: perto $ a $, pontos de $ M $ são exatamente os
$\varphi(u) = \Theta(u, 0)$ --- para isso, o \omterm{def:b3:topology:continuity}{homeomorfismo}
hipótese em (4) garante que $ M\cap\Omega $, para pequenos
$\Omega $, não contém outras planilhas ($\varphi(u') + v = m \in
M $ perto de $ a $ com pequeno $ v \neq 0$ deve ser excluído:
$ m = \varphi(u'')$ para alguns $ u''$ aproximar $ u_0$ pelo
\omterm{def:b3:topology:continuity}{homeomorfismo} propriedade e injetividade local de $\Theta $
forças $ v = 0$). Então $\Phi(M\cap\Omega) = (A\times\{0\})
\cap\Phi(\Omega)$ até encolher.
\end{proof}

\begin{example}\label{ex:b3:submanifolds:examples}
A esfera $ S^{n-1} = \{\norm x^2 = 1\}$: conjunto de níveis do
submersão $ F(x) = \norm x_2^2 - 1$ sobre
$\R^n\setminus\{0\}$ ($ DF(x) = 2x^{\mathsf T} \ne 0$):
um $\mathcal C^\infty $ subvariedade de dimensão $ n - 1$. O
toro em $\R^3$: conjunto de níveis de $\bigl(\sqrt{x^2 + y^2} -
R\bigr)^2 + z^2 - r^2$ ($0 < r < R $). O cone $\{x^2 + y^2 =
z^2\}$ é \emph{não} uma subvariedade em $0$
(\cref{exo:b3:submanifolds:1}). Grupos de matrizes: $ SL_n $ e
$ O_n $ são subvariedades de $ M_n(\R)$
(\cref{exo:b3:submanifolds:5,exo:b3:submanifolds:6}) --- o
ponto de partida da teoria de Lie.
\end{example}

\section{Espaços tangentes}

\begin{definition}\label{def:b3:submanifolds:tangent}
Deixar $ M $ seja um $ d $-subvariedade e $ a \in M $. O \emph{tangente
espaço}\index{tangente
espaço} $ T_aM $ é o conjunto da velocidade
vetores $\gamma'(0)$ de $\mathcal C^1$ curvas $\gamma \colon
\intoo{-\varepsilon}\varepsilon \to M $ com $\gamma(0) = a $.
\end{definition}

\begin{proposition}\label{prop:b3:submanifolds:tangent}
$ T_aM $ é um $ d $ subespaço vetorial tridimensional de $\R^n $, e:
\begin{enumerate}
  \item se $ M = F^{-1}(0)$ localmente com $ F $ uma submersão:
        $ T_aM = \ker DF(a)$;
  \item se $ M $ é parametrizado pela imersão $\varphi $
        ($\varphi(u_0) = a $): $ T_aM =
        \operatorname{im}D\varphi(u_0)$.
\end{enumerate}
\end{proposition}

\begin{proof}
Curvas em $ M $ satisfazer $ F(\gamma(t)) = 0$; a regra da cadeia em
$0$ dá $ DF(a)\gamma'(0) = 0$: $ T_aM \subseteq \ker DF(a)$.
Por outro lado, endireitar (\cref{thm:b3:submanifolds:%
characterizations}(1)) transporta linhas de
$\R^d\times\{0\}$ para curvas de $ M $: todo vetor de a
$ d $ o subespaço -dimensional é realizado; comparando dimensões
($\dim\ker DF(a) = n - (n - d) = d $) força a igualdade em (1),
e o mesmo argumento de transporte fornece (2) ($ D\varphi(u_0)$
aplicado a linhas retas em $ A $; dimensões novamente).
\end{proof}

\begin{theorem}[Multiplicadores de Lagrange]
\label{thm:b3:submanifolds:lagrange}
Deixar $ M = F^{-1}(0)$ com $ F = (F_1, \dots, F_q) \colon \Omega
\to \R^q $ um $\mathcal C^1$ submersão e deixe $ f \colon
\Omega \to \R $ ser $\mathcal C^1$. Se a restrição
$ f\restriction_M $ tem um extremo local em $ a \in M $, então
existem reais únicos $\lambda_1, \dots, \lambda_q $
(\emph{\omterm{thm:b3:submanifolds:lagrange}{Multiplicadores de Lagrange}}) com\index{Multiplicadores de Lagrange}
\[
\nabla f(a) = \lambda_1\nabla F_1(a) + \dots +
\lambda_q\nabla F_q(a) .
\]
\end{theorem}

\begin{proof}
Para cada curva $\gamma $ em $ M $ através $ a $: $ t \mapsto
f(\gamma(t))$ tem um extremo local em $0$, então $0 =
\frac{\dd}{\dd t}f(\gamma(t))\big|_0 = \langle\nabla f(a),
\gamma'(0)\rangle $: $\nabla f(a) \perp T_aM = \ker DF(a)$
(\cref{prop:b3:submanifolds:tangent}). Agora $\ker DF(a)^\perp
= \operatorname{im}DF(a)^{\mathsf T}$ (a classificação/ortogonalidade
identidade do Ano 2, ou \cref{exo:b3:hilbert:8} em finito
dimensão: $(\ker T)^\perp = \operatorname{im}T^*$), que é
abrangido pelos gradientes $\nabla F_i(a)$ --- independente, como
$ DF(a)$ é sobrejetivo: os multiplicadores existem e são únicos.
\end{proof}

\begin{example}[O teorema espectral, geometricamente]
\label{ex:b3:submanifolds:rayleigh}
Deixar $ A $ seja um verdadeiro simétrico $ n\times n $ matriz e maximizar
$ f(x) = \langle Ax, x\rangle $ na esfera $ S^{n-1}$
(compactar: o \omterm{def:b3:rings:primemaximal}{máximo} é atingido, em algum $ v_1$). Lagrange
com $ F(x) = \norm x^2 - 1$: $\nabla f = 2Ax $ e $\nabla F =
2x $ dar $ Av_1 = \lambda_1v_1$ --- um autovetor, com
$\lambda_1 = \max_{S^{n-1}}\langle Ax, x\rangle $. Restringir
$ A $ para $ v_1^\perp $ (invariante: $\langle Av, v_1\rangle =
\langle v, Av_1\rangle = \lambda_1\langle v, v_1\rangle = 0$)
e iterar: uma \omterm{def:b3:topology:basis}{base} ortonormal de autovetores. O
teorema espectral do ano 2, reprovado por otimização pura
--- e a sombra de dimensão infinita do mesmo argumento
comprovado \cref{lem:b3:spectral:existence}.
\end{example}

\begin{method}\label{met:b3:submanifolds:method}
Para provar que um conjunto é uma subvariedade: exiba-o localmente como
$ F^{-1}(0)$ com $ DF $ sobrejetivo \emph{no set} (o mais
rota comum) ou como um gráfico. Para calcular sua dimensão e
espaço tangente: $ d = n - q $ e $ T_a = \ker DF(a)$. Para
otimizar: Lagrange --- sempre verifique a compactação (ou
coercividade) primeiro, então existe um extremo para o qual o teorema
pode ser aplicado e lembre-se de que a equação do multiplicador é apenas
\emph{necessário}: colete todos os pontos críticos e compare
valores. Para grupos de matrizes, diferencie as curvas no
identidade para identificar espaços tangentes.
\end{method}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:b3:submanifolds:1}
(a) Verifique se os seguintes são $\mathcal C^\infty $
subvariedades e dê suas dimensões: $ S^{n-1}$; o
hiperbolóide $\{x^2 + y^2 - z^2 = 1\}$; o toro de
\cref{ex:b3:submanifolds:examples}.
(b) Mostre que o cone $ C = \{x^2 + y^2 = z^2\} \subseteq
\R^3$ não é um $2$-subvariedade em $0$: determinar o número
de componentes \omterm{def:b3:topology:connected}{conectados} de
$\bigl(C\setminus\{0\}\bigr)\cap B(0,\varepsilon)$ e
compare com o número um endireitamento
(\cref{thm:b3:submanifolds:characterizations}(1)) seria
força para um plano menos um ponto.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b3:submanifolds:2}
Calcule o espaços tangentes:
(a) $ T_aS^{n-1}$ para qualquer $ a $ (responder: $ a^\perp $);
(b) o plano tangente ao toro de
\cref{ex:b3:submanifolds:examples} em um ponto arbitrário de
o equador externo $\{z = 0,\ x^2 + y^2 = (R + r)^2\}$;
(c) a linha tangente à hélice $\varphi(t) = (\cos t,
\sin t, t)$ no $\varphi(t_0)$, verificando
\cref{prop:b3:submanifolds:tangent}(2).
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:submanifolds:3}
Deixar $ f(x, y) = (x^2 - y^2,\ 2xy)$ (ou seja,\ $ z \mapsto z^2$).
(a) Em quais pontos \cref{thm:b3:submanifolds:ift} aplicar?
(b) Mostre que $ f $ é localmente, mas não globalmente, invertível em
$\R^2\setminus\{0\}$, e exibir explicitamente os dois locais
inversos definidos em um \omterm{def:b3:topology:topology}{vizinhança} de $(1, 0)$ (os dois
ramos de raiz quadrada).
(c) Mesma discussão para o \omterm{ex:b3:product:polar}{coordenadas polares} mapa $(r,
\theta) \mapsto (r\cos\theta, r\sin\theta)$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:submanifolds:4}
(Folium) Deixe $ F(x,y) = x^3 + y^3 - 3xy $ e $\mathcal C =
F^{-1}(0)$.
(a) Mostre que perto de cada ponto de $\mathcal C $ além do
origem, $\mathcal C $ é um $1$-subvariedade, localmente um gráfico em
$ x $ ou em $ y $ (qual, onde?).
(b) Calcule a reta tangente em $(\frac32, \frac32)$.
(c) O que acontece na origem? (Dois ramos se cruzam:
exibir dois $\mathcal C^1$ curvas em $\mathcal C $ através $0$
com velocidades independentes, e concluímos que não
existe endireitamento.)
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:submanifolds:5}
Deixar $ F(M) = M^{\mathsf T}M $ de $ M_n(\R)$ para o espaço
$ S_n $ de matrizes simétricas.
(a) Mostrar $ DF(M)(H) = M^{\mathsf T}H + H^{\mathsf T}M $ e
isso $ DF(M)$ é sobrejetivo em $ S_n $ a cada $ M \in O_n $
\emph{(dado $ S \in S_n $, tentar $ H = \frac12MS $)}.
(b) Conclua que $ O_n = F^{-1}(I)$ é um compactar
$\mathcal C^\infty $ subvariedade de dimensão
$\frac{n(n-1)}2$, com $ T_IO_n = \{H : H^{\mathsf T} = -H\}$,
as matrizes antissimétricas.
(c) Mostre que $\eu^{tH} \in O_n $ para cada antisimétrico $ H $:
as direções tangentes integram-se às curvas do grupo.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:submanifolds:6}
(a) Mostre que $\det \colon M_n(\R) \to \R $ tem diferencial
$ D\det(M)(H) = \operatorname{tr}\bigl(\operatorname{com}(M)
^{\mathsf T}H\bigr)$, diferente de zero em cada $ M \in SL_n $.
(b) Conclua que $ SL_n(\R)$ é uma subvariedade de dimensão
$ n^2 - 1$ com $ T_ISL_n = \{H : \operatorname{tr}H = 0\}$.
(c) É $ GL_n(\R)$ uma subvariedade? De que dimensão?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:submanifolds:7}
Por \omterm{thm:b3:submanifolds:lagrange}{Multiplicadores de Lagrange}:
(a) encontre os extremos de $ f(x,y) = xy $ no círculo $ x^2 +
y^2 = 1$;
(b) mostre que entre todos os vetores de probabilidade $(p_1, \dots,
p_n)$ (positivo, somando $1$), a entropia $-\sum
p_i\ln p_i $ é maximizado exatamente na distribuição uniforme;
(c) encontre o ponto da elipse $\{x^2/4 + y^2 = 1\}$
mais próximo de $(1, 0)$, e verifique a equação do multiplicador
geometricamente (alinhamento \omterm{def:b3:groups:normal}{normal}).
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b3:submanifolds:8}
Escreva \cref{ex:b3:submanifolds:rayleigh} na íntegra: provar
por indução que uma matriz simétrica real admite um
\omterm{def:b3:topology:basis}{base} ortonormal de autovetores, com
$\lambda_1 \geq \dots \geq \lambda_n $ o sucessivo
\omterm{def:b3:rings:primemaximal}{Máximos} restritos do quociente de Rayleigh. Então deduza o
Fórmulas de Courant-Fischer de \cref{exo:b3:spectral:8} em
dimensão finita diretamente desta construção.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b3:submanifolds:9}
(Desigualdade de Hadamard) Para $ M \in GL_n(\R)$ com colunas
$ c_1, \dots, c_n $:
\[
\abs{\det M} \;\leq\; \prod_{i=1}^n\norm{c_i}_2 ,
\]
com igualdade se as colunas forem ortogonais. \emph{(Reduzir
para colunas de norma $1$ por escala; maximizar $\det $ no
compactar produto de esferas $(S^{n-1})^n $; em um maximizador,
Lagrange em cada coluna separadamente dá $\nabla_{c_i}\det =
\lambda_ic_i $, e $\nabla_{c_i}\det $ é o $ i $-ésima coluna de
$\operatorname{com}(M)$: deduzir $ M^{\mathsf T}M $ diagonal,
daí $= I $, por isso $\det = \pm1$.)} Leitura geométrica: a
O volume de um paralelepípedo é no \omterm{def:b3:rings:primemaximal}{máximo} o produto de sua aresta
comprimentos.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:submanifolds:10}
Perto de qual de seus pontos está o círculo $ S^1$ um gráfico $ y =
\psi(x)$? Um gráfico $ x = \chi(y)$? Verifique o gráfico
caracterização (\cref{thm:b3:submanifolds:%
characterizations}(3)) explicitamente em $(1, 0)$ e explique
em uma frase porque \emph{alguns} permutação de coordenadas é
sempre suficiente, mas ninguém sempre funciona.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:submanifolds:11}
(O grupo ortogonal como uma subvariedade, quantitativamente)
(a) Mostre que $ O_n = \{M : M^{\mathsf T}M = I\}$ é compactar:
limitado (cada coluna é um vetor unitário, então $\norm M \leq
\sqrt n $ para a norma da matriz euclidiana) e fechada.
(b) Mostre que espaço tangente no $ I $ é o espaço de
matrizes antissimétricas, de dimensão $\frac{n(n-1)}2$ e
em um geral $ A \in O_n $: $ T_AO_n = \{AK : K^{\mathsf T} =
-K\}$.
(c) Deduza que o mapa $ t \mapsto A\exp(tK)$ é, para cada
antisimétrico $ K $, uma curva em $ O_n $ através $ A $ com
velocidade $ AK $ \emph{(verificar $\exp(tK) \in O_n $ usando
$\exp(X)^{\mathsf T} = \exp(X^{\mathsf T})$ e
$\exp(-X)\exp(X) = I $)}: todo vetor tangente é realizado por
uma curva explícita, sem necessidade de teorema de função implícita.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:submanifolds:12}
(Pontos críticos da distância) Deixe $ M \subseteq \R^n $ ser
uma subvariedade e $ p \notin M $. Mostre que se $ x_0 \in M $
minimiza a distância até $ p $ (tal ponto existe quando $ M $
está fechado e não vazio --- por quê?), então
\[
p - x_0 \;\perp\; T_{x_0}M
\]
\emph{(diferenciar $ t \mapsto \norm{\gamma(t) - p}^2$
ao longo de curvas em $ M $)}. Deduzir: o ponto mais próximo em um
esfera está no raio que passa pelo \omterm{ex:b3:groups:actions}{centro}; e use o
condição para calcular a distância de $ p = (2, 0)$ para o
parábola $ y = x^2$ (reduza para uma cúbica e resolva
numericamente para três dígitos).
\end{exercise}

\section{Problema: \texorpdfstring{$ SO(3)$}{SO(3)} e o
quatérnios}

\begin{problem}[{Problema de fim de semana --- rotações, o grupo
$ S^3$ e a capa dupla}]\label{pb:b3:submanifolds:1}
Os quaterniões $\mathbb H = \{t + x\mathrm i + y\mathrm j +
z\mathrm k\}$ --- a álgebra cujo grupo de unidades contém
\cref{pb:b3:groups:1}de $ Q_8$ --- parametrizar
rotações tridimensionais duas vezes: o mapa ``conjuga
por uma unidade quaternion'' é um morfismo sobrejetivo $ S^3 \to
SO(3)$ com núcleo $\{\pm1\}$. Nós construímos tudo.
Lembre-se/defina: multiplicação é $\R $-bilinear com
$\mathrm i^2 = \mathrm j^2 = \mathrm k^2 =
\mathrm{ijk} = -1$; o conjugado de $ q = t + x\mathrm i +
y\mathrm j + z\mathrm k $ é $\bar q = t - x\mathrm i -
y\mathrm j - z\mathrm k $; $ N(q) = q\bar q = t^2 + x^2 + y^2
+ z^2$.

\medskip
\noindent\textbf{Part I --- The algebra $\mathbb H $ and the
group $ S^3$.}
\begin{enumerate}
  \item Verifique isso $\mathbb H $ é um associativo
        $\R $-álgebra com \omterm{ex:b3:groups:actions}{centro} $\R $, que $\overline{pq} =
        \bar q\,\bar p $, e isso $ N(pq) = N(p)N(q)$
        \emph{(uma rota limpa: represente $ q $ como o
        $2\times2$ matriz complexa
        $\bigl(\begin{smallmatrix}\alpha & \beta\\
        -\bar\beta & \bar\alpha\end{smallmatrix}\bigr)$, $ q =
        \alpha + \beta\mathrm j $ e usar $\det $)}.
  \item Deduza que cada $ q \neq 0$ é invertível ($ q^{-1} =
        \bar q/N(q)$): $\mathbb H $ é um (não comutativo)
        campo, e $ S^3 = \{N(q) = 1\}$ é um grupo ---
        e um compactar $3$-subvariedade de $\R^4$
        (\cref{ex:b3:submanifolds:examples}).
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The rotation morphism.} Identificar
$\R^3$ com o \emph{quaterniões puros} $ P = \{x\mathrm i +
y\mathrm j + z\mathrm k\}$, e para $ q \in S^3$ definir
$\rho_q(v) = q\,v\,\bar q $.
\begin{enumerate}[resume]
  \item Mostre isso $\rho_q $ mapas $ P $ para $ P $ \emph{(puro
        quatérnios são aqueles com $\bar v = -v $)}, é
        $\R $-linear, preserva a norma, e isso $\rho
        \colon q \mapsto \rho_q $ é um morfismo de grupo $ S^3
        \to O(3)$.
  \item Calcule o kernel: $\rho_q = \mathrm{id}$ se $ q $
        viaja com $\mathrm i, \mathrm j, \mathrm k $ se
        $ q \in \R\cap S^3 = \{\pm1\}$.
  \item Escrever $ q = \cos\frac\theta2 + \sin\frac\theta2\,u $
        com $ u \in P $, $ N(u) = 1$ (por que isso é sempre
        possível para $ q \in S^3$?). Mostre isso $\rho_q $ correções
        $ u $ e, no avião $ u^\perp\cap P $, atua como o
        rotação do ângulo $\theta $ \emph{(calcular
        $\rho_q(w)$ para $ w \perp u $ usando $ uw = -wu $ para
        unidades puras ortogonais --- prove esta identidade de
        a tabuada de multiplicação, ou de $ uw + wu =
        -2\langle u, w\rangle $)}.
  \item Concluir: $\operatorname{im}\rho \subseteq SO(3)$
        (cada $\rho_q $ é uma rotação com eixo e ângulo como
        calculado --- determinante $+1$ por continuidade de $ q
        \mapsto \det\rho_q $ no \omterm{def:b3:topology:connected}{conectado} $ S^3$, ou
        diretamente) e $\rho $ é \emph{para} $ SO(3)$: cada
        rotação de $\R^3$ tem um eixo (prove: um eixo real
        $3\times3$ matriz ortogonal com $\det = 1$ tem
        autovalor $1$ --- considere a característica
        polinômio) e é, portanto, algum $\rho_q $. Resumo:
        \[
        SO(3) \;\cong\; S^3/\{\pm 1\} .
        \]
\end{enumerate}

\medskip
\noindent\textbf{Part III --- $ SO(3)$ as a submanifold;
Rodrigues.}
\begin{enumerate}[resume]
  \item Mostre isso $ SO(3)$ é um compactar $3$-dimensional
        subvariedade de $ M_3(\R)$ com $ T_ISO(3) =$
        matrizes antisimétricas
        (\cref{exo:b3:submanifolds:5}; o determinante
        condição seleciona uma união de componentes).
  \item Para a matriz antissimétrica $ A_u $ associado com
        $ u \in \R^3$ ($ A_uv = u\wedge v $, a cruz
        produto), provar \emph{Fórmula de Rodrigues}:
        \[
        \eu^{\theta A_u} = I + \sin\theta\,A_u + (1 -
        \cos\theta)\,A_u^2 \qquad (\norm u = 1)
        \]
        \emph{(de $ A_u^3 = -A_u $: dividir o exponencial
        série junto $ A_u $ poderes)}e identificá-lo como o
        rotação do eixo $ u $ e ângulo $\theta $. Deduza isso
        $\exp $ mapeia as matrizes antissimétricas \emph{para}
        $ SO(3)$.
  \item Relacione as duas parametrizações: mostre que $ t
        \mapsto \rho_{q(t)}$ com $ q(t) = \cos\frac t2 +
        \sin\frac t2\,u $ é um grupo de um parâmetro de
        rotações cuja derivada em $ t = 0$ é $ A_u $
        --- o quaterniônico e exponenciais de matriz diga
        a mesma história na metade e na velocidade máxima, respectivamente.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The double cover, felt.}
\begin{enumerate}[resume]
  \item Mostre que o caminho $ q(t) = \cos\frac t2 + \sin\frac
        t2\,\mathrm k $, $ t \in \intcc0{2\pi}$, é um loop em
        $ SO(3)$ (sua imagem $\rho_{q(t)}$ retorna ao
        identidade) cuja elevação quaterniônica é \emph{não} um
        ciclo: $ q(2\pi) = -q(0)$. Continuando a $ t = 4\pi $
        fecha o elevador. Explique em um pequeno parágrafo o que
        isto diz: um $2\pi $ rotação não é continuamente
        reversível enquanto um $4\pi $ rotação é (a correia
        truque), porque $ SO(3)$ Os loops de são detectados em seu
        capa dupla $ S^3$.
  \item Deduza também o dividendo prático: composição de
        rotações = multiplicação de quatérnios
        ($4$ multiplicações de dados em vez de $9$,
        sem desvio da ortogonalidade) --- verifique no
        composição de dois quartos de volta sobre $\mathrm i $
        e $\mathrm j $: calcule o eixo e o ângulo do
        produto.
\end{enumerate}

\medskip
\noindent\textbf{Part V --- The explicit matrix:
Euler--Rodrigues.} Escrever $ q = a + b\mathrm i + c\mathrm j +
d\mathrm k \in S^3$, para que $ a^2 + b^2 + c^2 + d^2 = 1$.
\begin{enumerate}[resume]
  \item Calcular $\rho_q(\mathrm i)$ na íntegra a partir do
        tabuada de multiplicação; então obtenha $\rho_q(\mathrm
        j)$ e $\rho_q(\mathrm k)$ pelo cíclico
        substituição $\mathrm i \to \mathrm j \to \mathrm k
        \to \mathrm i $, $(b, c, d) \to (c, d, b)$
        \emph{(justifique: andar de bicicleta $\mathrm i, \mathrm j,
        \mathrm k $ se estende a um automorfismo de $\mathbb
        H $, porque as relações definidoras são ciclicamente
        simétrico)}. Conclua que a matriz de $\rho_q $
        na \omterm{def:b3:topology:basis}{base} $(\mathrm i, \mathrm j, \mathrm k)$ é
        o \emph{Matriz de Euler-Rodrigues}
        \[
        R_q = \begin{pmatrix}
        a^2 + b^2 - c^2 - d^2 & 2(bc - ad) & 2(bd + ac)\\
        2(bc + ad) & a^2 - b^2 + c^2 - d^2 & 2(cd - ab)\\
        2(bd - ac) & 2(cd + ab) & a^2 - b^2 - c^2 + d^2
        \end{pmatrix}.
        \]
  \item (Lendo uma rotação para trás) Mostre que
        \[
        \operatorname{tr}R_q = 4a^2 - 1 = 1 + 2\cos\theta,
        \qquad
        \tfrac12\bigl(R_q - R_q^{\mathsf T}\bigr) =
        \sin\theta\,A_u,
        \]
        na notação das questões 5 e 8. Deduza um
        recuperação de algoritmo $\pm q $ de uma matriz de rotação
        $ R $: o ângulo do traço; o eixo do
        parte antisimétrica quando $0 < \theta < \pi $; e,
        quando $\theta = \pi $, provar e usar a identidade $ R
        + I = 2\,uu^{\mathsf T}$.
  \item Avaliar $ R_q $ para o produto da questão 11 $ q =
        \frac12(1 + \mathrm i + \mathrm j + \mathrm k)$: um
        matriz de permutação aparece. Identifique a rotação
        e reconciliar com o eixo e o ângulo encontrados em
        questão 11.
\end{enumerate}

\medskip
\noindent\textbf{Part VI --- Inside $ S^3$: $ SU(2)$,
\omterm{ex:b3:groups:actions}{aulas de conjugação}, exponentials.}
\begin{enumerate}[resume]
  \item Mostre que a matriz \omterm{def:b3:representations:rep}{representação} da pergunta 1
        (chame-o $\Phi $) restringe-se a um isomorfismo de grupo
        de $ S^3$ para o \emph{grupo unitário especial}
        \[
        SU(2) = \bigl\{U \in M_2(\C) : U^*U = I,\ \det U =
        1\bigr\}
        \]
        \emph{(para sobrejetividade, escreva as equações
        $ U^{-1} = U^*$ e $\det U = 1$ para um geral
        $2\times2$ matriz complexa)}.
  \item Mostre que a parte real é uma conjugação invariante
        ligado $ S^3$ --- $\operatorname{Re}(pq\bar p) =
        \operatorname{Re}q $ para todos $ p \in S^3$ --- e,
        inversamente, que dois quatérnios unitários com o mesmo
        parte real são conjugados em $ S^3$ \emph{(reduzir para
        moving one unit pure axis onto another, which Part
        II fornece)}. Descreva o \omterm{ex:b3:groups:actions}{aulas de conjugação} de
        $ S^3$ geometricamente; traduzir para $ SU(2)$ (nível
        conjuntos do traço); e projeto de $\rho $: dois
        rotações são conjugadas em $ SO(3)$ se e somente se
        eles têm o mesmo ângulo $\theta \in \intcc0\pi $.
  \item Definir $\exp $ sobre $\mathbb H $ pelo exponencial
        série; verifique a convergência absoluta, usando
        $\abs{pq} = \abs p\,\abs q $ para $\abs q =
        \sqrt{N(q)}$. Mostre, para uma unidade pura $ u $ e $\theta
        \in \R $,
        \[
        \exp(\theta u) = \cos\theta + \sin\theta\,u ,
        \]
        deduzir isso $\exp $ mapeia o hiperplano $ P $ para
        $ S^3$ e verifique isso $\rho_{\exp(su)} =
        \eu^{2sA_u}$: o fenômeno do meio ângulo da questão
        9 novamente.
  \item Para quatérnios puros $ v, w $ provar a regra do produto
        $ vw = -\langle v, w\rangle + v\wedge w $, daí o
        \omterm{def:b3:groups:derived}{comutador} identidade $ vw - wv = 2\,v\wedge w $; provar
        também $[A_v, A_w] = A_{v\wedge w}$ para as matrizes
        da questão 8. Conclua que a derivada de
        $\rho $ no $1$ ao longo das curvas $ t \mapsto \exp(tv)$
        é o isomorfismo linear $ v \mapsto 2A_v $ de $ P $
        nas matrizes antissimétricas, e que
        transporta o quatérnio \omterm{def:b3:groups:derived}{comutador} para a matriz
        \omterm{def:b3:groups:derived}{comutador}.
\end{enumerate}

\medskip
\noindent\textbf{Part VII --- Global structure.}
\begin{enumerate}[resume]
  \item (Não \omterm{def:b3:topology:continuity}{contínuo} seção) Suponha $ s \colon SO(3) \to
        S^3$ é \omterm{def:b3:topology:continuity}{contínuo} com $\rho \circ s =
        \operatorname{id}$. Para o circuito $ R(t) =
        \rho_{q(t)}$ da questão 10, conjunto $\varepsilon(t) =
        s(R(t))\,q(t)^{-1}$ para $ t \in \intcc0{2\pi}$. Mostrar
        isso $\varepsilon $ é \omterm{def:b3:topology:continuity}{contínuo} com valores em
        $\{\pm1\}$, e deriva uma contradição: não há
        \omterm{def:b3:topology:continuity}{contínuo} escolha global de uma unidade quaternion
        representando cada rotação.
  \item (O modelo da bola) Vamos $\bar B \subseteq \R^3$ seja o
        bola fechada de raio $\pi $ e $ E(v) = \eu^{A_v}$,
        com $ E(0) = I $. Mostre isso $ E $ mapas $\bar B $ para
        $ SO(3)$, é injetivo na bola aberta e na
        esfera limite identifica exatamente antípodas:
        $ E(\pi u) = E(-\pi u) = 2uu^{\mathsf T} - I $, com
        nenhuma outra coincidência. Por isso $ SO(3)$ é a bola
        com pontos de fronteira antípodas colados --- o
        espaço projetivo $\mathbb{RP}^3$ --- e um diâmetro
        torna-se o loop não contratível da questão 10.
  \item Mostre isso $\rho_p\rho_q\rho_p^{-1} = \rho_{pq\bar
        p}$; que as involuções de $ SO(3)$ (o $ R \neq
        I $ com $ R^2 = I $) são exatamente as meias voltas
        $\rho_w $ com $ w $ uma unidade quatérnio puro; e isso
        o \omterm{ex:b3:groups:actions}{centro} de $ SO(3)$ é trivial.
  \item Mostre que toda rotação é um produto de dois
        meias voltas: para $ q = \cos\frac\theta2 +
        \sin\frac\theta2\,u $, escolha uma unidade pura $ w \perp
        u $, verifique isso $ w' = qw $ é novamente uma unidade pura
        quatérnio e verifique $\rho_q =
        \rho_{w'}\rho_w $. Onde estão os dois eixos e
        que ângulo eles fazem?
  \item Conclua o topológico resumo: $ SO(3)$ é
        compactar e \omterm{def:b3:topology:pathconnected}{conectado por caminho} (dê duas provas:
        \omterm{def:b3:topology:continuity}{contínuo} imagem de $ S^3$ sob $\rho $; imagem de
        $\exp $), enquanto $ O(3)$ tem exatamente dois \omterm{def:b3:topology:connected}{conectado}
        componentes, cada homeomórfico para $ SO(3)$.

  \item (Uma composição, três maneiras) Vamos $ R_1$ seja a rotação
        por $\frac\pi2$ sobre o $ z $-eixo e $ R_2$ o
        rotação por $\frac\pi2$ sobre o $ x $-eixo. Calcular
        o eixo e o ângulo de $ R_2R_1$: (i) multiplicando
        os dois $3\times3$ matrizes e usando
        parte traço/antisimétrica (Parte V); (ii) por
        multiplicando os quatérnios unitários correspondentes
        $ q_2q_1$. Verifique se as duas respostas concordam: ângulo
        $\frac{2\pi}3$, eixo $\frac1{\sqrt3}(1, -1, 1)$.
  \item (O Transformada de Cayley) Para $ K $ antisimétrico, mostrar
        isso $ I + K $ é invertível e
        \[
        C(K) = (I - K)(I + K)^{-1} \in SO(n),
        \]
        com $-1$ nunca um autovalor de $ C(K)$; mostre isso
        $ K \mapsto C(K)$ é uma bijeção de antisimétrica
        matrizes em $\{R \in SO(n) : -1 \notin
        \operatorname{Sp}R\}$, com inverso $ R \mapsto (I -
        R)(I + R)^{-1}$. (Um gráfico racional de $ SO(n)$,
        companheiro do transcendental $\exp$ de
        \cref{exo:b3:submanifolds:11}.)
\end{enumerate}
\end{problem}
