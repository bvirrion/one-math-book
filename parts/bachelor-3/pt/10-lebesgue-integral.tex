\chapter{A Integral de Lebesgue}\label{ch:b3:lebesgue}

A integral de Riemann fatia o \emph{domínio} em pequeno
intervalos; As fatias de Lebesgue \emph{faixa}: integrar
$f $, \omterm{def:b3:measure:measure}{medir} os conjuntos $\{f > t\}$. A mudança parece inocente
e é revolucionário. Limites e integrais, sempre brigando
na teoria de Riemann (é necessária convergência uniforme!), são
reconciliado por três teoremas de convergência --- monótono
convergência, Fatou, convergência dominada --- cujas hipóteses
são quase embaraçosamente fracos. Este capítulo constrói o
integral sobre um arbitrário \omterm{def:b3:measure:measure}{medir} o espaço $(X, \mathcal A,
\mu)$, prova os três teoremas, estabelece o exato
relação com a integral de Riemann (uma função limitada é
Riemann \omterm{def:b3:lebesgue:l1}{integrável} se for \omterm{def:b3:topology:continuity}{contínuo} \omterm{def:b3:lebesgue:l1}{quase em todos os lugares}) e
industrializa a diferenciação de parâmetros dependentes
integrais --- a técnica que o problema do fim de semana usa para
calcular $\int_0^\infty\frac{\sin x}x\,\dd x $ e
$\int_\R \eu^{-x^2}\dd x $.

\section{Funções mensuráveis}

\begin{definition}\label{def:b3:lebesgue:measurable}
Deixar $(X, \mathcal A)$, $(Y, \mathcal B)$ ser espaços mensuráveis.
$ f \colon X \to Y $ é \emph{mensurável}\index{mensurável} se $ f^{-1}(B) \in \mathcal A $ para cada $ B \in
\mathcal B $. De verdade (ou $[-\infty,+\infty]$-valorizado)
funções, $ Y = \R $ carrega seu Borel $\sigma $-álgebra, e isso
basta verificar $ f^{-1}(\intoo t{+\infty}) = \{f > t\} \in
\mathcal A $ para todos $ t \in \R $: os bons conjuntos $\{B :
f^{-1}(B) \in \mathcal A\}$ formar um $\sigma $-álgebra
(pré-imagens comutam com operações definidas) contendo o
gerando raios (\cref{def:b3:measure:borel},
\cref{met:b3:measure:goodsets}).
\end{definition}

\begin{proposition}\label{prop:b3:lebesgue:stability}
(a) Composições de \omterm{def:b3:lebesgue:measurable}{mensurável} mapas são \omterm{def:b3:lebesgue:measurable}{mensurável}; \omterm{def:b3:topology:continuity}{contínuo}
mapas são \omterm{def:b3:lebesgue:measurable}{Borel-mensurável}.
(b) Se $ f, g \colon X \to \R $ são \omterm{def:b3:lebesgue:measurable}{mensurável}, assim são $ f + g $,
$ fg $, $\max(f,g)$, $\abs f $, $\lambda f $.
(c) Se $(f_n)$ são \omterm{def:b3:lebesgue:measurable}{mensurável} com valores em
$[-\infty, +\infty]$, então $\sup_nf_n $, $\inf_nf_n $,
$\limsup f_n $, $\liminf f_n $ são \omterm{def:b3:lebesgue:measurable}{mensurável}; se $ f_n \to f $
pontualmente, $ f $ é \omterm{def:b3:lebesgue:measurable}{mensurável}.
\end{proposition}

\begin{proof}
(um) $(g\circ f)^{-1}(B) = f^{-1}(g^{-1}(B))$; continuidade dá
mensurabilidade através da geração conjuntos abertos
(\cref{pb:b3:measure:1}, questão 10, de forma geral).
(b) $(f, g) \colon X \to \R^2$ é \omterm{def:b3:lebesgue:measurable}{mensurável} para o Borel
$\sigma $-álgebra de $\R^2$ --- marque as caixas abertas, que
gerar ($\R^2$ as aberturas de são uniões contáveis de racionais
caixas): $(f,g)^{-1}(U\times V) = f^{-1}(U)\cap g^{-1}(V)$ ---
e $+, \times, \max $ são \omterm{def:b3:topology:continuity}{contínuo} $\R^2 \to \R $: compor.
(c) $\{\sup f_n > t\} = \bigcup_n\{f_n > t\}$; $\inf = -\sup(-)$;
$\limsup = \inf_N\sup_{n \geq N}$; um limite pontual é próprio
$\limsup $.
\end{proof}

\begin{definition}\label{def:b3:lebesgue:simple}
UM \emph{função simples}\index{função simples} é um \omterm{def:b3:lebesgue:measurable}{mensurável}
função com um número finito de valores: $ s = \sum_{i=1}^n
c_i\,\mathbf 1_{A_i}$, $ A_i \in \mathcal A $ disjunto, $ c_i \geq
0$ (para a teoria não negativa). Sua integral é
\[
\int s\,\dd\mu = \sum_i c_i\,\mu(A_i) \in [0, +\infty]
\]
(convenção $0\cdot\infty = 0$); o valor não depende
o \omterm{def:b3:representations:rep}{representação} (refinar duas partições).
\end{definition}

\begin{theorem}[Aproximação por funções simples]
\label{thm:b3:lebesgue:approximation}
Todo \omterm{def:b3:lebesgue:measurable}{mensurável} $ f \colon X \to [0, +\infty]$ é o ponto
limite de um \emph{aumentando} sequência de funções \omterm{def:b3:groups:simple}{simples}:
\[
s_n = \sum_{k=1}^{n2^n} \frac{k-1}{2^n}\,
\mathbf 1_{\{\frac{k-1}{2^n} \leq f < \frac k{2^n}\}}
+ n\,\mathbf 1_{\{f \geq n\}} \nearrow f .
\]
\end{theorem}

\begin{proof}
Cada $ s_n $ é \omterm{def:b3:groups:simple}{simples} (os conjuntos são pré-imagens dos conjuntos do Borel).
Monotonicidade: passando de $ n $ para $ n+1$ divide cada diádico
nível em dois e nunca diminui o valor atribuído (um ponto
com $\frac{k-1}{2^n} \leq f(x) < \frac k{2^n}$ obtém qualquer um
$\frac{2k-2}{2^{n+1}}$ ou $\frac{2k-1}{2^{n+1}}$, ambos $\geq
\frac{k-1}{2^n}$; o boné $ n $ sobe também). Convergência: se
$ f(x) < \infty $, para $ n > f(x)$ nós temos $ f(x) - s_n(x) \leq
2^{-n}$; se $ f(x) = \infty $, $ s_n(x) = n \to \infty $.
\end{proof}

\section{Os teoremas da integral e da convergência}

\begin{definition}\label{def:b3:lebesgue:integral}
Para \omterm{def:b3:lebesgue:measurable}{mensurável} $ f \geq 0$:
\[
\int f \,\dd\mu = \sup\Bigl\{\int s\,\dd\mu : s \text{ simple},
\ 0 \leq s \leq f\Bigr\} \in [0, +\infty].
\]
É monótono em $ f $ por construção, e amplia o \omterm{def:b3:groups:simple}{simples}
caso (para \omterm{def:b3:groups:simple}{simples} $ f $, o sup é alcançado em $ f $: comparação de
integrais \omterm{def:b3:groups:simple}{simples} através de refinamentos comuns).
\end{definition}

\begin{theorem}[Convergência monótona, Beppo Levi]
\label{thm:b3:lebesgue:mct}
Se $0 \leq f_n \nearrow f $ pontualmente (\omterm{def:b3:lebesgue:measurable}{mensurável}), então
\[
\int f_n\,\dd\mu \nearrow \int f\,\dd\mu .
\]
\index{teorema da convergência monótona}
\end{theorem}

\begin{proof}
$ f $ é \omterm{def:b3:lebesgue:measurable}{mensurável} (\cref{prop:b3:lebesgue:stability}(c)) e
$\int f_n $ aumenta para alguns $ L \leq \int f $ (monotonicidade).
Por outro lado, conserte um \omterm{def:b3:groups:simple}{simples} $ s = \sum c_i\mathbf 1_{A_i} \leq f $
e $\theta \in (0,1)$; os conjuntos $ E_n = \{f_n \geq \theta s\}$
são \omterm{def:b3:lebesgue:measurable}{mensurável} e aumentar para $ X $ (onde $ s(x) > 0$: $ f(x)
\geq s(x) > \theta s(x)$, então eventualmente $ f_n(x) \geq \theta
s(x)$; onde $ s(x) = 0$: trivialmente). Então
\[
\int f_n \geq \int_{E_n}\theta s\,\dd\mu
= \theta\sum_i c_i\,\mu(A_i \cap E_n)
\xrightarrow[n\to\infty]{} \theta\sum_ic_i\,\mu(A_i)
= \theta\int s
\]
por continuidade de baixo (\cref{prop:b3:measure:basics}(c)).
Então $ L \geq \theta\int s $ para todos $\theta < 1$ e tudo \omterm{def:b3:groups:simple}{simples} $ s
\leq f $: $ L \geq \int f $.
\end{proof}

\begin{corollary}\label{cor:b3:lebesgue:additivity}
Para \omterm{def:b3:lebesgue:measurable}{mensurável} $ f, g \geq 0$ e $ c \geq 0$: $\int(f + g) =
\int f + \int g $ e $\int cf = c\int f $; para uma série de
não negativo funções mensuráveis, $\int\sum_nf_n =
\sum_n\int f_n $.
\end{corollary}

\begin{proof}
Para funções \omterm{def:b3:groups:simple}{simples}, aditividade é um cálculo em um comum
refinamento. Em geral tome $ s_n \nearrow f $, $ t_n \nearrow g $
(\cref{thm:b3:lebesgue:approximation}): $ s_n + t_n \nearrow f +
g $, e o MCT passa a aditividade até o limite. A série
declaração é o MCT aplicado aos valores parciais.
\end{proof}

\begin{theorem}[Lema de Fatou]\label{thm:b3:lebesgue:fatou}
Para \omterm{def:b3:lebesgue:measurable}{mensurável} $ f_n \geq 0$:
\[
\int \liminf_n f_n \,\dd\mu \;\leq\; \liminf_n \int
f_n\,\dd\mu .
\]
\index{Lema de Fatou}
\end{theorem}

\begin{proof}
Deixar $ g_N = \inf_{n\geq N}f_n $: \omterm{def:b3:lebesgue:measurable}{mensurável}, $0 \leq g_N \nearrow
\liminf f_n $, e $ g_N \leq f_n $ para cada $ n \geq N $, então
$\int g_N \leq \inf_{n \geq N}\int f_n $. Aplicar MCT à esquerda
lado: $\int\liminf f_n = \lim_N\int g_N \leq
\lim_N\inf_{n\geq N}\int f_n = \liminf\int f_n $.
\end{proof}

\begin{definition}\label{def:b3:lebesgue:l1}
UM \omterm{def:b3:lebesgue:measurable}{mensurável} $ f \colon X \to \R $ (ou $\C $) é
\emph{integrável}\index{integrável} se $\int\abs
f\,\dd\mu < \infty $; então $\int f = \int f^+ - \int f^-$
(partes positivas e negativas; partes reais e imaginárias no
caso complexo). A integral é linear em funções integráveis
(decompor e recombinar partes positivas; o caso complexo
reduz ao real) e satisfaz $\abs{\int f} \leq
\int\abs f $ (caso real: $\pm\int f = \int(\pm f) \leq
\int\abs f $; caso complexo: multiplicar por uma constante unimodular
para tornar a integral real). Uma propriedade contém \emph{quase
em todos os lugares} (e.e.)\index{quase em todos os lugares} se falhar apenas em
um $\mu $-conjunto nulo; modificando $ f $ em um conjunto nulo muda não
integral (a diferença é dominada por $\infty\cdot\mathbf
1_N $, de integral $0$).
\end{definition}

\begin{theorem}[Convergência dominada]\label{thm:b3:lebesgue:dct}
Deixar $ f_n \to f $ ou seja, com $\abs{f_n} \leq g $ a.e.\ por um valor fixo
\emph{\omterm{def:b3:lebesgue:l1}{integrável}} $ g $. Então $ f $ é \omterm{def:b3:lebesgue:l1}{integrável} e
\[
\int f_n\,\dd\mu \longrightarrow \int f\,\dd\mu,
\qquad\text{indeed}\quad \int\abs{f_n - f}\,\dd\mu \to 0 .
\]
\index{teorema da convergência dominada}
\end{theorem}

\begin{proof}
Descarte um conjunto nulo para tornar as hipóteses pontuais. $\abs f
\leq g $: $ f $ é \omterm{def:b3:lebesgue:l1}{integrável}. As funções $ h_n = 2g - \abs{f_n
- f} \geq 0$ satisfazer $\liminf h_n = 2g $; Fatou dá
\[
\int 2g \leq \liminf\int\bigl(2g - \abs{f_n - f}\bigr)
= \int 2g - \limsup\int\abs{f_n - f},
\]
então $\limsup\int\abs{f_n - f} \leq 0$ (a subtração é legal:
$\int 2g < \infty $). Finalmente $\abs{\int f_n - \int f} \leq
\int\abs{f_n - f} \to 0$.
\end{proof}

\begin{method}\label{met:b3:lebesgue:convergence}
Diante de $\lim_n\int f_n $: tente, em ordem --- (1) é o
sequência monótona (ou uma série de termos não negativos)? MCT, não
integrabilidade necessário. (2) Existe um único \omterm{def:b3:lebesgue:l1}{integrável}
dominador $ g \geq \abs{f_n}$, encontrado por limites brutos
(``$\sup_n $'' as estimativas)? DCT. (3) Sem dominação, não
monotonicidade? Fatou ainda limita um lado, e a igualdade pode
falhar genuinamente: o solavanco que escapa $ f_n = n\mathbf
1_{\intoo0{1/n}}$ tem $\int f_n = 1$ mas $ f_n \to 0$ a.e.
A dominação é exatamente o que proíbe a massa de escapar para
infinito, vertical ou horizontalmente.
\end{method}

\section{Riemann versus Lebesgue}

\begin{theorem}[Critério de Lebesgue]
\label{thm:b3:lebesgue:riemann}
Deixar $ f \colon \intcc ab \to \R $ ser \emph{limitado}. Então $ f $ é
Riemann \omterm{def:b3:lebesgue:l1}{integrável} se $ f $ é \omterm{def:b3:topology:continuity}{contínuo} $\lambda $-quase
em todos os lugares; nesse caso $ f $ é \omterm{def:b3:lebesgue:l1}{Lebesgue-integrável} e os dois
integrais coincidem.\index{Integral de Riemann x Lebesgue}
\end{theorem}

\begin{proof}
Para uma subdivisão $\sigma = (a = x_0 < \dots < x_N = b)$, deixar
$ L_\sigma $ e $ U_\sigma $ sejam as funções degrau iguais, em cada
$\intoo{x_{i-1}}{x_i}$, para $ m_i = \inf_{[x_{i-1}, x_i]}f $ e
$ M_i = \sup $; as somas de Darboux são suas integrais (Riemann e
Lebesgue concorda com funções escalonadas, ambas dando $\sum
m_i\Delta x_i $). Pegue uma sequência de subdivisões $\sigma_n $,
cada um refinando o último, de malha $\to 0$, com somas de Darboux
convergindo para as integrais de Darboux inferior e superior de $ f $.
Os refinamentos fazem $ L_{\sigma_n}$ não decrescente e
$ U_{\sigma_n}$ não crescente pontualmente fora do conjunto contável
$ D $ de todos os pontos de divisão; chame os limites $\ell $ e $ u $
(\omterm{def:b3:lebesgue:measurable}{mensurável}, \cref{prop:b3:lebesgue:stability}). Para $ x \notin
D $, escrita $ I_n(x)$ para o aberto $\sigma_n $-intervalo
contendo $ x $: $\ell(x) = \sup_n\inf_{I_n(x)}f $ e $ u(x) =
\inf_n\sup_{I_n(x)}f $; já que as malhas encolhem para $0$, estes
são o inferior e o superior \emph{envelopes} de $ f $ no $ x $
--- $ u(x) - \ell(x)$ é a oscilação de $ f $ no $ x $ --- para que \emph{$\ell(x) = u(x)$ se $ f $ é \omterm{def:b3:topology:continuity}{contínuo} no $ x $}. Por MCT/DCT (intervalo limitado e finito):
\[
\int_{\intcc ab}\ell\,\dd\lambda = \lim_n\int L_{\sigma_n}
= \underline{\int}f,
\qquad
\int_{\intcc ab}u\,\dd\lambda = \overline{\int}f .
\]
$ f $ Riemann \omterm{def:b3:lebesgue:l1}{integrável} $\iff $ $\underline\int f =
\overline\int f $ $\iff $ $\int(u - \ell) = 0$ $\iff $ $ u = \ell $
a.e.\ ($ u - \ell \geq 0$; \cref{exo:b3:lebesgue:5}) $\iff $ $ f $
\omterm{def:b3:topology:continuity}{contínuo} a.e. Nesse caso $\ell \leq f \leq u $ com $\ell =
u $ a.e.: $ f $ é igual ao \omterm{def:b3:lebesgue:measurable}{mensurável} $\ell $ a.e., portanto é
\omterm{def:b3:lebesgue:measurable}{Lebesgue-mensurável} (completude de $\lambda $) com
$\int f\,\dd\lambda = \int\ell\,\dd\lambda = \underline\int f =
\int_a^bf $.
\end{proof}

\begin{example}\label{ex:b3:lebesgue:riemannexamples}
$\mathbf 1_\Q $ não está em lugar nenhum \omterm{def:b3:topology:continuity}{contínuo}: não Riemann \omterm{def:b3:lebesgue:l1}{integrável}
--- mas Lebesgue-trivial: $\int\mathbf 1_\Q\,\dd\lambda =
\lambda(\Q) = 0$. Função de Thomas (\,$\frac1q $ em racionais
$\frac pq $, $0$ em outro lugar) é \omterm{def:b3:topology:continuity}{contínuo} exatamente no
irracionais: Riemann \omterm{def:b3:lebesgue:l1}{integrável} com integral $0$. E
\emph{impróprio} Integrais de Riemann são uma noção diferente:
$\int_0^\infty\frac{\sin x}x\,\dd x $ converge como um limite de
$\int_0^A $ (o problema do fim de semana calcula $= \frac\pi2$),
mas $\frac{\sin x}x \notin L^1(\intoo0{+\infty})$: o absoluto
integral diverge como a série harmônica
(\cref{exo:b3:lebesgue:6}). A teoria de Lebesgue é negociada
convergência condicional para teoremas de limite robustos.
\end{example}

\section{Integrais com parâmetros}

Por todo, $(X, \mathcal A, \mu)$ é um \omterm{def:b3:measure:measure}{medir} o espaço, $ T $ um
espaço métrico (o parâmetro), e $ f \colon T \times X \to \C $
com $ f(t, \cdot)$ \omterm{def:b3:lebesgue:l1}{integrável} para cada $ t $; definir $ F(t) = \int_X
f(t, x)\,\dd\mu(x)$.

\begin{theorem}[Continuidade]\label{thm:b3:lebesgue:paramcont}
Suponha: $ t \mapsto f(t,x)$ é \omterm{def:b3:topology:continuity}{contínuo} no $ t_0$ para a.e.\
$ x $, e há um \omterm{def:b3:lebesgue:l1}{integrável} $ g $ com $\abs{f(t,x)} \leq
g(x)$ para todos $ t $ em um \omterm{def:b3:topology:topology}{vizinhança} de $ t_0$ e a.e.\ $ x $.
Então $ F $ é \omterm{def:b3:topology:continuity}{contínuo} no $ t_0$.\index{integrais de parâmetro}
\end{theorem}

\begin{proof}
Para qualquer sequência $ t_n \to t_0$: $ f(t_n, \cdot) \to f(t_0,
\cdot)$ ou seja, dominado por $ g $: DCT dá $ F(t_n) \to F(t_0)$;
sequencial continuidade é suficiente em espaços métricos
(\cref{rem:b3:topology:sequences}).
\end{proof}

\begin{theorem}[Diferenciação sob a integral]
\label{thm:b3:lebesgue:paramdiff}
Deixar $ T $ seja um intervalo aberto de $\R $. Suponha: para a.e.\ $ x $,
$ t \mapsto f(t,x)$ é diferenciável em $ T $, com
\[
\Bigl|\frac{\partial f}{\partial t}(t, x)\Bigr| \leq g(x)
\quad \text{for all } t \in T,\ \text{a.e. } x,
\]
$ g $ \omterm{def:b3:lebesgue:l1}{integrável}. Então $ F $ é diferenciável em $ T $ com $ F'(t)
= \int_X \frac{\partial f}{\partial t}(t, x)\,\dd\mu(x)$.
\end{theorem}

\begin{proof}
Consertar $ t $ e $ h_n \to 0$: os quocientes de diferença
\[
\varphi_n(x) = \frac{f(t + h_n, x) - f(t, x)}{h_n}
\longrightarrow \frac{\partial f}{\partial t}(t,x)
\quad\text{a.e.},
\]
e os limites da desigualdade do valor médio $\abs{\varphi_n(x)} \leq
\sup_{s}\abs{\partial_tf(s,x)} \leq g(x)$: DCT se aplica, e
$\frac{F(t + h_n) - F(t)}{h_n} = \int\varphi_n \to
\int\partial_t f(t, \cdot)$.
\end{proof}

\begin{example}[A função Gama]\label{ex:b3:lebesgue:gamma}
Para $ t > 0$ deixar\index{Função gama}
\[
\Gamma(t) = \int_0^{+\infty} x^{t-1}\eu^{-x}\,\dd x .
\]
A integral converge: perto $0$, $ x^{t-1}$ é \omterm{def:b3:lebesgue:l1}{integrável} ($ t >
0$); no infinito, $ x^{t-1}\eu^{-x} \leq C\eu^{-x/2}$.
Integração por partes (em $[\varepsilon, A]$, então limita via
MCT) fornece a equação funcional $\Gamma(t + 1) =
t\,\Gamma(t)$, de onde $\Gamma(n+1) = n!$: o fatorial
interpolado. Em cada $\intcc ab \subseteq \intoo0{+\infty}$,
$\partial_t\bigl(x^{t-1}\eu^{-x}\bigr) = \ln
x\cdot x^{t-1}\eu^{-x}$ é dominado por $\abs{\ln x}(x^{a-1} +
x^{b-1})\eu^{-x}$, \omterm{def:b3:lebesgue:l1}{integrável}: $\Gamma $ é $\mathcal C^1$ e
por indução $\mathcal C^\infty $, com $\Gamma^{(k)}(t) =
\int_0^\infty(\ln x)^kx^{t-1}\eu^{-x}\dd x $. O valor
$\Gamma(\frac12) = \sqrt\pi $ é a integral gaussiana em
disfarce (\cref{pb:b3:lebesgue:1}).
\end{example}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:b3:lebesgue:1}
(a) Mostre que uma função monótona $\R \to \R $ é
\omterm{def:b3:lebesgue:measurable}{Borel-mensurável}, e que uma derivada (de um em todos os lugares
função diferenciável) é \omterm{def:b3:lebesgue:measurable}{Borel-mensurável}.
(b) Mostre que $ f \colon X \to \R $ é \omterm{def:b3:lebesgue:measurable}{mensurável} se $\{f > q\}
\in \mathcal A $ para cada \emph{racional} $ q $.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b3:lebesgue:2}
Calcule, com plena justificativa:
\[
\lim_{n\to\infty}\int_0^{+\infty}
\frac{\cos x}{(1 + x/n)^{n}}\,\dd x,
\qquad
\lim_{n\to\infty}\int_0^1 \frac{n\,x^{n-1}}{1 + x}\,\dd x .
\]
\emph{(Para o segundo: substitua $ u = x^n $ antes de dominar.)}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:lebesgue:3}
(a) Mostre a desigualdade estrita no lema de Fatou.
(b) Anexo $ f_n \to 0$ pontualmente com $\int f_n = 1$ em três
caminhos: escapar em altura, em largura, até o infinito. Qual solteiro
hipótese de DCT cada uma viola?
(c) Mostre que no lema de Fatou não se pode substituir $\liminf $ por
$\limsup $ em ambos os lados.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:lebesgue:4}
(a) Mostrar $\displaystyle\int_0^{+\infty}\frac{x}{\eu^x -
1}\,\dd x = \sum_{n\geq1}\frac1{n^2} = \frac{\pi^2}6$
\emph{(expandir $\frac1{\eu^x - 1}$ em uma série geométrica e
integrar termo por termo --- qual teorema permite isso?)}.
(b) (Sonho do segundo ano) Mostrar
$\displaystyle\int_0^1 x^{-x}\,\dd x = \sum_{n\geq1}n^{-n}$.
\emph{(Escrever $ x^{-x} = \eu^{-x\ln x} = \sum_k\frac{(-x\ln
x)^k}{k!}$ e calcular $\int_0^1(-x\ln x)^k\dd x $ por
substituindo $ x = \eu^{-u/(k+1)}$, reconhecendo $\Gamma $.)}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:lebesgue:5}
(a) Mostre que $ f \geq 0$ \omterm{def:b3:lebesgue:measurable}{mensurável} com $\int f\,\dd\mu = 0$
satisfaz $ f = 0$ a.e. \emph{(Considerar $\{f \geq 1/n\}$ e
Desigualdade de Markov: $\mu(\{f \geq a\}) \leq \frac1a\int f $ ---
prove.)}
(b) Mostre que \omterm{def:b3:lebesgue:l1}{integrável} $ f $ é finito a.e.
(c) Mostre que se $\int_A f\,\dd\mu = 0$ para \emph{todo}

\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:lebesgue:6}

\begin{exercise}[$\star\star $]\label{exo:b3:lebesgue:7}
Justifique isso $ F(t) = \int_0^{+\infty}\eu^{-x^2}\cos(tx)\,\dd x $

\begin{exercise}[$\star\star\star $]\label{exo:b3:lebesgue:8}

justificando o intercâmbio através da teoria não negativa
(\cref{cor:b3:lebesgue:additivity} em \omterm{def:b3:topology:continuity}{contínuo} formulário ---

\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:lebesgue:9}
Deixar $ f \geq 0$ ser \omterm{def:b3:lebesgue:measurable}{mensurável} sobre $(X, \mathcal A, \mu)$. Mostrar

(\emph{densidade}\index{densidade} $ f $ em relação a
$\mu $) e que $\int g\,\dd\nu = \int gf\,\dd\mu $ para todos

\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b3:lebesgue:10}

x^2}\dd x $ para cada $ t $ --- mas isso diferenciador

(b) Admitir $ f'(t) = \frac\pi2\eu^{-t}$ para $ t > 0$ (provado
em \cref{ch:b3:residues}), o que é
$\int_0^\infty\frac{x\sin(tx)}{1+x^2}\dd x $ para $ t > 0$ e
por que sua fórmula confirma a falha em (a)?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:lebesgue:11}
(Scheffĺema de e) Deixe $ f_n, f \geq 0$ ser \omterm{def:b3:lebesgue:l1}{integrável} com
$ f_n \to f $ a.e.\ e $\int f_n \to \int f $.
(a) Mostre que $\int\abs{f_n - f} \to 0$. \emph{(Aplicar
convergência dominada para $ g_n = (f - f_n)^+ \leq f $ e
escrever $\int\abs{f_n - f} = 2\int g_n - \int(f - f_n)$.)}
(b) Mostre por exemplo que a hipótese $\int f_n \to \int
f $ não pode ser derrubado (um impacto deslizante ou de concentração) e
que a \omterm{thm:b3:complete:completion}{conclusão} falha por assinado $ f_n $ sem
controle de valor absoluto: $ f_n = n\mathbf 1_{\intoc0{1/n}} -
n\mathbf 1_{\intoc{-1/n}0}$ tem $ f_n \to 0$ a.e., $\int f_n
= 0 \to 0$, ainda $\int\abs{f_n} = 2$.
(c) Aplicação (\omterm{ex:b3:lebesgue:gamma}{densidades}): se \omterm{ex:b3:lebesgue:gamma}{densidades} de probabilidade $ p_n
\to p $ ou seja, então automaticamente $\int\abs{p_n - p} \to 0$:
convergência pontual de \omterm{ex:b3:lebesgue:gamma}{densidades} é $ L^1$ convergência ---
uma atualização de convergência gratuitamente.
\end{exercise}

\section{Problema: duas integrais celebradas}

\begin{problem}[{Problema de fim de semana --- a integral gaussiana e
Integral de Dirichlet, apenas por parâmetros}]
\label{pb:b3:lebesgue:1}
Análise aplicada à regra de duas integrais:
\[
G = \int_{-\infty}^{+\infty}\eu^{-x^2}\dd x = \sqrt\pi,
\qquad
D = \int_0^{+\infty}\frac{\sin x}{x}\,\dd x = \frac\pi2
\]
(o segundo como uma integral imprópria,
\cref{ex:b3:lebesgue:riemannexamples}). Provamos ambos usando
apenas as ferramentas deste capítulo.

\medskip
\noindent\textbf{Part I --- The Gaussian.} Para $ t \geq 0$ definir
\[
A(t) = \Bigl(\int_0^t\eu^{-x^2}\dd x\Bigr)^{2},
\qquad
B(t) = \int_0^1\frac{\eu^{-t^2(1 + x^2)}}{1 + x^2}\,\dd x .
\]
\begin{enumerate}
  \item Justifique isso $ A $ e $ B $ são $\mathcal C^1$ sobre
        $\intoo0{+\infty}$ e calcular $ A'$ e $ B'$; mostrar
        $ A'(t) + B'(t) = 0$. \emph{(Em $ B'$, substitua $ u =
        tx $.)}
  \item Calcular $ A(0) + B(0)$ e $\lim_{t\to+\infty}(A + B)(t)$
        --- justifique o limite sob a integral em $ B $.
  \item Concluir $\int_0^\infty \eu^{-x^2}\dd x =
        \frac{\sqrt\pi}2$, por isso $ G = \sqrt\pi $, e deduza
        $\Gamma(\tfrac12) = \sqrt\pi $ \emph{(substituir $ x =
        u^2$ em $\Gamma(\frac12)$)}.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Dirichlet's integral.} Para $ t
\geq 0$ definir
\[
F(t) = \int_0^{+\infty}\eu^{-tx}\,\frac{\sin x}{x}\,\dd x .
\]
\begin{enumerate}[resume]
  \item Mostre que a definição integral $ F(t)$ converge para
        cada $ t > 0$ como uma integral de Lebesgue, e para $ t = 0$
        como uma integral imprópria; mostre isso $ D =
        \lim_{A\to\infty}\int_0^A\frac{\sin x}x\dd x $ existe
        \emph{(integrar por partes em $[\pi, A]$)}.
  \item Mostre isso $ F $ é $\mathcal C^1$ sobre
        $\intoo0{+\infty}$ com
        \[
        F'(t) = -\int_0^{+\infty}\eu^{-tx}\sin x\,\dd x
        = -\frac{1}{1 + t^2}
        \]
        \emph{(dominação em $[t_0, \infty)$ para cada $ t_0 >
        0$; a última integral por duas integrações por partes ou
        exponenciais complexas)}.
  \item Mostrar $ F(t) \to 0$ como $ t \to +\infty $, e deduza $ F(t)
        = \frac\pi2 - \arctan t $ sobre $\intoo0{+\infty}$.
  \item O ponto delicado: $ D = \lim_{t\to0^+}F(t)$. Prove
        pelo controle uniforme da cauda: para $0 \leq t \leq 1$
        e $ A \geq \pi $, integre por partes para mostrar
        \[
        \Bigl|\int_A^{+\infty}\eu^{-tx}\frac{\sin
        x}x\,\dd x\Bigr| \leq \frac{C}{A}
        \]
        com $ C $ independente de $ t $ \emph{(diferenciar
        $\frac{\eu^{-tx}}x $ e vinculado $\abs{\cos}$ por $1$;
        observação $ t\eu^{-tx} \leq 1/x\cdot(tx\eu^{-tx})$ com
        $\sup_{u\geq0}u\eu^{-u} < 1$)}; então divida $ F(t) - D $
        em $[0, A]$ (onde a DCT se aplica como $ t \to 0$) e
        $[A, \infty)$.
  \item Concluir: $ D = \frac\pi2$.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Dividends.}
\begin{enumerate}[resume]
  \item Calcular $\int_0^{+\infty}\frac{\sin^2x}{x^2}\,\dd x $
        \emph{(integrar por partes e reduzir a $ D $ através de $\sin
        2x = 2\sin x\cos x $)}.
  \item Calcular $\int_0^{+\infty}\frac{1 -
        \cos x}{x^2}\,\dd x $ e verifique a consistência do
        dois resultados.
  \item Para $ a > 0$, calcular
        $\int_0^{+\infty}\frac{\sin(ax)}x\dd x $ e
        $\int_{-\infty}^{+\infty}\eu^{-ax^2}\dd x $ e registre
        as regras de escalonamento (elas serão os burros de carga de
        \cref{ch:b3:fouriertransform}).
  \item Explique exatamente por que $ D $ não poderia ter sido tratado
        pela DCT diretamente em $ t = 0$ (não \omterm{def:b3:lebesgue:l1}{integrável} dominador ligado
        $[0,1]\times[0,\infty)$), e por que a divisão da cauda de
        a questão 7 é o substituto honesto --- este padrão
        (``uniforme integrabilidade de coroa'') se repete ao longo
        análise.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The \omterm{ex:b3:lebesgue:gamma}{Função gama} according to
Bohr and Mollerup.} A função $\Gamma $
(\cref{ex:b3:lebesgue:gamma}) satisfaz $\Gamma(1) = 1$ e
$\Gamma(x+1) = x\Gamma(x)$ --- mas o mesmo acontece com infinitas outras
funções (multiplique por qualquer $1$-oscilação periódica). Uma convexidade
pinos de condição $\Gamma $ para baixo de forma única, e é mais profundo
as identidades então caem mecanicamente. Uma função positiva $ f $
em um intervalo é \emph{log-convexo} se $\log f $ é convexo.
\begin{enumerate}[resume]
  \item Mostre que log-convexo implica convexo, que os produtos de
        funções log-convexas e suas composições com afins
        os mapas são log-convexos e --- por meio da função de duas funções
        Hdesigualdade mais antiga $\int\abs{uv} \leq
        \bigl(\int\abs u^p\bigr)^{1/p}\bigl(\int\abs
        v^q\bigr)^{1/q}$, provado diretamente de Young
        desigualdade --- que $\Gamma $ é log-convexo em
        $\intoo0\infty $.
  \item (Lema da inclinação) Deixe $ g $ ser convexo em $\intoo0\infty $
        com $ g(n+1) - g(n) = \log n $ para cada número inteiro $ n
        \geq 1$. Para $ x \in \intoc01$ e $ n \geq 2$, comparar
        as encostas de $ g $ sobre $[n-1, n]$, $[n, n+x]$ e
        $[n, n+1]$, e deduza
        \[
        x\log(n-1) \;\leq\; g(n + x) - g(n) \;\leq\; x\log n .
        \]
  \item (Bohr--Mollerup) Deixe $ f > 0$ satisfazer $ f(1) = 1$,
        $ f(x+1) = xf(x)$, e $\log f $ convexo. Desenrolando o
        recursão em $ f(n + x) = x(x+1)\cdots(x + n -
        1)\,f(x)$ e $ f(n) = (n-1)!$, deduza da questão 14
        isso para $ x \in \intoc01$
        \[
        f(x) = \lim_{n\to\infty}
        \frac{n!\,n^x}{x(x+1)\cdots(x+n)} :
        \]
        $ f $ é único, portanto $ f = \Gamma $, e \emph{Gauss
        fórmula limite} detém (estender a todos $ x > 0$ pelo
        recursão).
  \item Defina o \emph{Função beta} $ B(x, y) =
        \int_0^1t^{x-1}(1-t)^{y-1}\,\dd t $ ($ x, y > 0$). Provar
        convergência, a recursão $ B(x+1, y) =
        \frac{x}{x+y}\,B(x, y)$ (integrar por partes), e
        $ B(1, y) = \frac1y $.
  \item Mostre isso $ x \mapsto B(x, y)$ é log-convexo (Hölder
        novamente) e aplique Bohr-Mollerup para
        \[
        f(x) = \frac{B(x, y)\,\Gamma(x + y)}{\Gamma(y)}
        \]
        para concluir \emph{Fórmula de Euler}: $ B(x, y) =
        \dfrac{\Gamma(x)\Gamma(y)}{\Gamma(x+y)}$ --- sem duplo
        integrais em qualquer lugar.
  \item Calcular $ B(\frac12, \frac12)$ diretamente (substituir $ t
        = \sin^2\theta $) e deduzir $\Gamma(\frac12) =
        \sqrt\pi $: a integral gaussiana da Parte I, recuperada
        por pura convexidade. Compare as duas provas em uma
        sentença cada.
  \item (Duplicação da legenda) Mostre que
        \[
        g(x) = \frac{2^{x-1}}{\sqrt\pi}\,
        \Gamma\Bigl(\frac x2\Bigr)
        \Gamma\Bigl(\frac{x+1}2\Bigr)
        \]
        satisfaz as três hipóteses de Bohr-Mollerup, e
        concluir $ g = \Gamma $, ou seja\ $\Gamma(2z) =
        \frac{2^{2z-1}}{\sqrt\pi}\,\Gamma(z)\,\Gamma(z +
        \tfrac12)$ para todos $ z > 0$.
  \item Deduza a forma fechada $\Gamma\bigl(n + \tfrac12\bigr)
        = \dfrac{(2n)!}{4^n\,n!}\sqrt\pi $, e provar, pelo
        lema da inclinação aplicado a $\log\Gamma $ em torno de grande
        inteiros, os assintóticos
        \[
        \frac{\Gamma(n + \frac12)}{\Gamma(n)\,\sqrt n}
        \longrightarrow 1 .
        \]
  \item Combine as duas últimas perguntas na central
        assintótica binomial
        \[
        \binom{2n}{n} \sim \frac{4^n}{\sqrt{\pi n}},
        \]
        e verifique numericamente para $ n = 10$ ($\binom{20}{10} =
        184756$, contra $4^{10}/\sqrt{10\pi} \approx
        187079$: razão $\approx 0.988$).
  \item (Síntese) A constante $\sqrt\pi $ agora apareceu como
        a integral gaussiana (Parte I), como $ B(\frac12,
        \frac12)$ (questão 18), e duplicação interna
        (questão 19); a estimativa binomial central
        antecipa tanto Stirling (\cref{ch:b3:product}de
        problema de fim de semana) e de Moivre-Laplace. Mapeie o
        conexões: quais afirmações são equivalentes a quais,
        e o que cada técnica faz --- diferenciação sob
        a integral versus convexidade --- contribui para que o
        outro não pode?
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Three more dividends.}
\begin{enumerate}[resume]
  \item (Wallis, por Beta) Para $ p > -1$, substitua $ t =
        \sin^2\theta $ mostrar
        \[
        W_p = \int_0^{\pi/2}\sin^p\theta\,\dd\theta
        = \frac12\,B\Bigl(\frac{p+1}2, \frac12\Bigr),
        \]
        e deduzir da recursão Beta (questão 16) que
        $ W_{n+2} = \frac{n+1}{n+2}\,W_n $ para números inteiros $ n \geq
        0$. Calcular $ W_{2n}$ e $ W_{2n+1}$ de forma fechada,
        mostrar $ W_{2n+1}/W_{2n} \to 1$ apertando e conclua
        com \emph{Produto de Wallis}
        \[
        \frac\pi2 = \lim_{n\to\infty}
        \prod_{k=1}^{n}\frac{4k^2}{4k^2 - 1} .
        \]
  \item (O gaussiano encontra uma frequência) Para $ b \in \R $ definir
        \[
        \Phi(b) = \int_{-\infty}^{+\infty}
        \eu^{-x^2}\cos(2bx)\,\dd x .
        \]
        Mostre isso $\Phi $ é $\mathcal C^1$ sobre $\R $, que um
        integração por partes produz a equação diferencial
        $\Phi'(b) = -2b\,\Phi(b)$, e concluir
        \[
        \Phi(b) = \sqrt\pi\,\eu^{-b^2} :
        \]
        o Gaussiano se reproduz sob esta transformação ---
        a identidade única na qual
        \cref{ch:b3:fouriertransform} será executado.
  \item (integral de Frullani) Para $0 < a < b $, mostre que
        \[
        \int_0^{+\infty}
        \frac{\eu^{-ax} - \eu^{-bx}}{x}\,\dd x
        = \log\frac ba ,
        \]
        diferenciando no parâmetro $ a $ (justifique o
        dominação em cada $\intco{a_0}{+\infty}$, $ a_0 > 0$,
        e identifique a constante deixando $ a \to b $).
        Onde exatamente o integrando precisa de seu removível
        singularidade em $ x = 0$?
\end{enumerate}
\end{problem}
