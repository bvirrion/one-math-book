\chapter{Extensões de Campo e Teoria de Galois}\label{ch:b3:galois}

Toda equação pode ser resolvida por radicais, como a fórmula quadrática
e as fórmulas cúbicas de Cardano sugerem? É possível trissectar um ângulo
com \omterm{cor:b3:galois:impossible}{régua e compasso}? Ambas as questões, abertas há séculos, são
respondido --- negativamente --- por uma única ideia de Évaristo \omterm{def:b3:galois:galois}{Galois}:
anexe a cada polinômio um finito \emph{grupo} de simetrias de
suas raízes e leia a resposta do grupo. Este capítulo constrói
o dicionário: extensões de campo e \omterm{def:b3:galois:extension}{graus}, divisão de campos e
fechamentos \omterm{def:b3:galois:algebraic}{algébricos}, campos finitos (uma teoria completa --- e o
ciclicidade prometida de $\mathbb F_q^\times $), separabilidade, então
o \omterm{thm:b3:galois:fundamental}{Correspondência de Galois} em si, com provas completas. Colhemos:
a impossibilidade das construções clássicas, a estrutura de
campos ciclotômicos e a insolubilidade do quíntico por
radicais --- \cref{ch:b3:groups}a simplicidade de $ A_5$ impressionante
seu alvo.

\section{Extensões, grau, algebricidade}

\begin{definition}\label{def:b3:galois:extension}
UM \emph{extensão de campo}\index{extensão de campo} $ L/K $ é um campo
$ L $ contendo $ K $ como um subcampo; $ L $ é então um $ K $-espaço vetorial,
e o \emph{grau}\index{grau} $[L:K]$ é
sua dimensão. A extensão é \emph{finito} se $[L:K] <
\infty $. O \emph{característica} de um campo é o gerador
$\geq 0$ do núcleo de $\Z \to K $, $ n \mapsto n\cdot 1$: isso é
$0$ ou um primo $ p $; correspondentemente $ K $ contém um menor
subcampo (\emph{campo principal}) isomórfico a $\Q $ ou para $\mathbb
F_p = \Z/p\Z $.
\end{definition}

\begin{theorem}[Lei da torre]\label{thm:b3:galois:tower}
Se $ K \subseteq L \subseteq M $, então $[M:K] = [M:L]\,[L:K]$: se
$(e_i)$ é uma base de $ L $ sobre $ K $ e $(f_j)$ uma base de $ M $
sobre $ L $, então $(e_if_j)$ é uma base de $ M $ sobre
$ K $.\index{lei da torre}
\end{theorem}

\begin{proof}
Gerando: $ x \in M $ escreve $ x = \sum_j \lambda_jf_j $
($\lambda_j \in L $), cada $\lambda_j = \sum_i \mu_{ij}e_i $
($\mu_{ij} \in K $): $ x = \sum_{i,j}\mu_{ij}e_if_j $. Independente:
$\sum_{i,j}\mu_{ij}e_if_j = 0$ reescreve $\sum_j (\sum_i
\mu_{ij}e_i)f_j = 0$; as somas internas estão em $ L $, então desapareça
($ f_j $ independente sobre $ L $); então tudo $\mu_{ij} = 0$ ($ e_i $
independente sobre $ K $).
\end{proof}

\begin{definition}\label{def:b3:galois:algebraic}
Deixar $ L/K $ e $\alpha \in L $. Se algum diferente de zero $ P \in K[X]$ tem
$ P(\alpha) = 0$, $\alpha $ é \emph{algébrico}\index{algébrico} sobre $ K $; o gerador mônico $\pi_\alpha $ do \omterm{def:b3:rings:ideal}{ideal}
$\{P : P(\alpha) = 0\}$ de $ K[X]$ é o seu \emph{mínimo
polinômio}\index{mínimo
polinômio}, um
irredutível polinômio ($\pi = QR $ com $ Q(\alpha) = 0$ forças $ R $
constante pela minimalidade do \omterm{def:b3:galois:extension}{grau}). De outra forma $\alpha $ é
\emph{transcendental}. Nós escrevemos $ K(\alpha)$ para o menor
subcampo de $ L $ contendo $ K $ e $\alpha $, e $ K[\alpha]$ para
o menor subanel.
\end{definition}

\begin{theorem}\label{thm:b3:galois:simple}
Se $\alpha $ é \omterm{def:b3:galois:algebraic}{algébrico} sobre $ K $ com $ d = \deg\pi_\alpha $,
então
\[
K(\alpha) = K[\alpha] \cong K[X]/(\pi_\alpha),
\qquad [K(\alpha) : K] = d,
\]
com base $1, \alpha, \dots, \alpha^{d-1}$. Por outro lado, se
$[L:K] < \infty $, todo $\alpha \in L $ é \omterm{def:b3:galois:algebraic}{algébrico} de \omterm{def:b3:galois:extension}{grau}
$\deg \pi_\alpha $ dividindo $[L:K]$.
\end{theorem}

\begin{proof}
Avaliação $ K[X] \to L $, $ P \mapsto P(\alpha)$, tem imagem
$ K[\alpha]$ e núcleo $(\pi_\alpha)$; desde $\pi_\alpha $ é
irredutível, $ K[X]/(\pi_\alpha)$ é um \emph{campo}
(\cref{prop:b3:rings:primemaximal}: no \omterm{def:b3:rings:pidufd}{PID} $ K[X]$,
irredutível gera um \omterm{def:b3:rings:ideal}{ideal} \omterm{def:b3:rings:primemaximal}{máximo}), então $ K[\alpha]$ é um campo
contendo $ K $ e $\alpha $: é igual $ K(\alpha)$. As aulas
de $1, X, \dots, X^{d-1}$ formam uma base do quociente (euclidiano
divisão), de onde a base e o \omterm{def:b3:galois:extension}{grau}. Inversamente se $[L:K]
= n < \infty $: $1, \alpha, \dots, \alpha^n $ são dependentes, dando
um polinômio aniquilador; então $[K(\alpha):K] =
\deg\pi_\alpha $ divide $ n $ pelo \omterm{thm:b3:galois:tower}{lei da torre}.
\end{proof}

\begin{corollary}\label{cor:b3:galois:algclosed}
Se $\alpha, \beta $ são \omterm{def:b3:galois:algebraic}{algébrico} sobre $ K $, assim são $\alpha \pm
\beta $, $\alpha\beta $, $\alpha/\beta $ ($\beta \ne 0$): o
elementos de $ L $ \omterm{def:b3:galois:algebraic}{algébrico} sobre $ K $ formar um subcampo de $ L $.
Além disso, a algebricidade é transitiva: \omterm{def:b3:galois:algebraic}{algébrico} sobre \omterm{def:b3:galois:algebraic}{algébrico} é
\omterm{def:b3:galois:algebraic}{algébrico}.
\end{corollary}

\begin{proof}
$ K(\alpha, \beta) = (K(\alpha))(\beta)$ é finito
$ K(\alpha)$ ($\beta $ \omterm{def:b3:galois:algebraic}{algébrico} sobre $ K \subseteq K(\alpha)$) e
$ K(\alpha)/K $ é finito: pelo \omterm{thm:b3:galois:tower}{lei da torre} $[K(\alpha,\beta):K] <
\infty $, e cada elemento de $ K(\alpha, \beta)$ --- incluindo
os quatro listados --- é \omterm{def:b3:galois:algebraic}{algébrico} (\cref{thm:b3:galois:simple}).
Transitividade: se $\beta $ é \omterm{def:b3:galois:algebraic}{algébrico} sobre $ L $ e $ L/K $ é
\omterm{def:b3:galois:algebraic}{algébrico}, os coeficientes $ c_0, \dots, c_{m-1}$ de
$\pi_{\beta/L}$ gerar uma extensão finita $ F = K(c_0, \dots,
c_{m-1})$ de $ K $ (repetido \omterm{thm:b3:galois:tower}{lei da torre}), e $ F(\beta)/F $ é
finito: $[F(\beta):K] < \infty $, então $\beta $ é \omterm{def:b3:galois:algebraic}{algébrico} sobre
$ K $.
\end{proof}

\begin{example}\label{ex:b3:galois:degrees}
$[\Q(\sqrt2):\Q] = 2$, $[\Q(\sqrt[3]2):\Q] = 3$ ($ X^3 - 2$ é
irredutível: \omterm{thm:b3:rings:criteria}{Eisenstein}), $[\Q(\zeta_p):\Q] = p - 1$ para
$\zeta_p = \eu^{2\iu\pi/p}$ ($\Phi_p $ é irredutível,
\cref{ex:b3:rings:eisenstein}). O \omterm{thm:b3:galois:tower}{lei da torre} já é um
arma: $\sqrt[3]2 \notin \Q(\sqrt2)$, desde $3 \nmid 2$.
\end{example}

\section{Divisão de campos; fechamento algébrico}

\begin{theorem}[Divisão de campos]\label{thm:b3:galois:splitting}
Deixar $ P \in K[X]$ ser inconstante. Existe um \emph{divisão
campo}\index{divisão
campo} de $ P $ sobre $ K $: uma extensão $ L =
K(\alpha_1, \dots, \alpha_n)$ gerado por raízes de $ P $ em que
$ P $ se divide em fatores lineares. É único até
$ K $-isomorfismo, e $[L:K] \leq (\deg P)!$.
\end{theorem}

\begin{proof}
\emph{Existência}, por indução em $\deg P $: escolha um irredutível
fator $ Q $ de $ P $; o campo $ K_1 = K[X]/(Q)$ contém a raiz
$\alpha_1 = \bar X $ de $ Q $, portanto de $ P $; escrever $ P = (X -
\alpha_1)P_1$ sobre $ K_1$ e aplique indução a $ P_1$ sobre $ K_1$;
\omterm{def:b3:galois:extension}{graus} multiplique para no \omterm{def:b3:rings:primemaximal}{máximo} $ n(n-1)\cdots = n!$.

\emph{Singularidade} segue do mais forte \emph{isomorfismo
lema de extensão}: deixar $\sigma \colon K \to K'$ seja um isomorfismo,
$ P \in K[X]$, $ P^\sigma $ o polinômio com coeficientes mapeados,
$ L, L'$ divisão de campos de $ P, P^\sigma $; então $\sigma $ estende
para um isomorfismo $ L \to L'$. Indução ativada $[L:K]$: se $ P $ divisões
em $ K $, então $ L = K $, e $ L' = K'$ ($ P^\sigma $ se divide em $ K'$,
e $ L'$ é gerado por suas raízes). Caso contrário, escolha uma raiz
$\alpha \in L \setminus K $ de um irredutível fator $ Q $ de $ P $
com $\deg Q \geq 2$; $ Q^\sigma $ é um irredutível fator de
$ P^\sigma $, com raiz $\beta \in L'$; então
\[
K(\alpha) \cong K[X]/(Q) \xrightarrow{\ \sigma\ }
K'[X]/(Q^\sigma) \cong K'(\beta)
\]
estende $\sigma $ com $\alpha \mapsto \beta $. Agora $ L $ é um
campo de divisão de $ P $ sobre $ K(\alpha)$, e $ L'$ de $ P^\sigma $
sobre $ K'(\beta)$, com $[L : K(\alpha)] < [L:K]$: indução
se estende ainda mais $ L \to L'$.
\end{proof}

\begin{definition}\label{def:b3:galois:closure}
Um campo $\Omega $ é \emph{algebricamente fechado}\index{algebricamente fechado} se todo polinômio não constante de $\Omega[X]$ tem
uma raiz em $\Omega $ (portanto, divisões). Um \emph{algébrico
fechamento}\index{algébrico
fechamento} de $ K $ é um \omterm{def:b3:galois:algebraic}{algébrico}
extensão $\bar K/K $ com $\bar K $ algebricamente fechado.
\end{definition}

\begin{theorem}[Steinitz]\label{thm:b3:galois:steinitz}
Cada campo $ K $ tem um fechamento \omterm{def:b3:galois:algebraic}{algébrico}, único até
$ K $-isomorfismo.
\end{theorem}

\begin{proof}
\emph{Existência (construção de Artin).} Deixar $ R = K[(X_f)_f]$ ser
o anel polinomial com uma variável $ X_f $ por não constante
Mônica $ f \in K[X]$, e $ I $ o \omterm{def:b3:rings:ideal}{ideal} gerado por todos
$ f(X_f)$. $ I $ é apropriado: uma relação $1 = \sum_{i=1}^r g_i\,
f_i(X_{f_i})$ envolve um número finito de polinômios; em um comum
campo de divisão $ E $ de $ f_1\cdots f_r $ escolha raízes $\alpha_i $ de
$ f_i $ e avaliar $ X_{f_i} \mapsto \alpha_i $ (outras variáveis
$\mapsto 0$): $1 = 0$, absurdo. Deixar $\mathfrak m \supseteq I $ ser
\omterm{def:b3:rings:primemaximal}{máximo} (\cref{thm:b3:rings:krull}; Zorn) e $ K_1 = R/\mathfrak
m $: um \omterm{def:b3:galois:extension}{extensão de campo} de $ K $ em que cada constante $ f \in
K[X]$ tem uma raiz, ou seja, $\bar X_f $, e qual é \omterm{def:b3:galois:algebraic}{algébrico} sobre
$ K $ (é gerado pelo $\bar X_f $, cada \omterm{def:b3:galois:algebraic}{algébrico}).
Iterar: $ K \subseteq K_1 \subseteq K_2 \subseteq \cdots $, onde
$ K_{n+1}$ faz para $ K_n $ o que $ K_1$ fez para $ K $, e deixe $\Omega =
\bigcup_n K_n $, um campo. Qualquer não constante $ g \in \Omega[X]$ tem
seus finitos coeficientes em alguns $ K_n $; um irredutível
fator de $ g $ sobre $ K_n $ tem uma raiz em $ K_{n+1} \subseteq
\Omega $: $\Omega $ é \omterm{def:b3:galois:closure}{algebricamente fechado}, e \omterm{def:b3:galois:algebraic}{algébrico} sobre
$ K $ (cada $ K_n $ é, por transitividade,
\cref{cor:b3:galois:algclosed}): $\Omega $ é um \omterm{def:b3:galois:closure}{algébrico
fechamento}.

\emph{Singularidade.} Deixar $\Omega, \Omega'$ ser dois \omterm{def:b3:galois:closure}{algébrico
fechamentos}. Considere o conjunto de pares $(E, \tau)$ onde $ K
\subseteq E \subseteq \Omega $ e $\tau\colon E \to \Omega'$ é um
$ K $-incorporação, ordenada por extensão; não está vazio ($(K,
\mathrm{id})$) e indutivo (união de uma cadeia), então Zorn dá um
\omterm{def:b3:rings:primemaximal}{máximo} $(E_0, \tau_0)$. Se $ E_0 \neq \Omega $, escolha $\alpha \in
\Omega\setminus E_0$: $\pi_{\alpha/E_0}$ mapeia para um polinômio
acabou $\tau_0(E_0)$ tendo uma raiz $\beta $ no \omterm{def:b3:galois:closure}{algebricamente
fechado} $\Omega'$, e $\tau_0$ estende-se a $ E_0(\alpha) \to
\Omega'$ ($\alpha \mapsto \beta $), contradizendo a maximidade. Então
há um $ K $-incorporação $\tau \colon \Omega \to \Omega'$; é
imagem, isomórfica a $\Omega $, é \omterm{def:b3:galois:closure}{algebricamente fechado}, e
$\Omega'$ é \omterm{def:b3:galois:algebraic}{algébrico} sobre isso: para $ x \in \Omega'$,
$\pi_{x/\tau(\Omega)}$ se divide $\tau(\Omega)$, então $ x \in
\tau(\Omega)$. Por isso $\tau $ está em: um isomorfismo.
\end{proof}

\begin{remark}\label{rem:b3:galois:closureexamples}
Para $ K = \Q $ pode-se evitar a maquinaria transfinita: o
\omterm{def:b3:galois:algebraic}{algébrico} números $\bar\Q = \{z \in \C : z \text{ algebraic over
} \Q\}$ formar um fechamento \omterm{def:b3:galois:algebraic}{algébrico} --- um subcampo de $\C $ por
\cref{cor:b3:galois:algclosed}, \omterm{def:b3:galois:closure}{algebricamente fechado} porque $\C $
é (d'Alembert--Gauss, provado por análise complexa em
\cref{ch:b3:holomorphic}) e raízes de polinômios sobre $\bar\Q $
são \omterm{def:b3:galois:algebraic}{algébrico} sobre $\Q $ pela transitividade.
\end{remark}

\section{Campos finitos}

\begin{theorem}\label{thm:b3:galois:finitefields}
Deixar $ p $ ser primo, $ n \geq 1$, $ q = p^n $.
\begin{enumerate}
  \item UM \omterm{thm:b3:galois:finitefields}{campo finito} tem cardinalidade como potência principal, e para
        cada $ q $ há exatamente um campo $\mathbb F_q $ com
        $ q $ elementos até o isomorfismo: o campo de divisão de
        $ X^q - X $ sobre $\mathbb F_p $.\index{campo finito}
  \item O \emph{Frobenius}\index{Frobenius} $ F
        \colon x \mapsto x^p $ é um automorfismo de $\mathbb
        F_q $, e o grupo de automorfismo de $\mathbb F_q $ é
        cíclico de ordem $ n $, gerado por $ F $.
  \item $\mathbb F_{p^m}$ incorpora em $\mathbb F_{p^n}$ se $ m \mid
        n $.
\end{enumerate}
\end{theorem}

\begin{proof}
(1) Um \omterm{thm:b3:galois:finitefields}{campo finito} $ E $ tem característica $ p > 0$ e é um
dimensão finita $\mathbb F_p $-espaço vetorial: $\abs E = p^n $.
Seu grupo multiplicativo tem ordem $ q - 1$, então cada $ x \in E $
satisfaz $ x^q = x $: $ E $ consiste em $ q $ raízes de $ X^q - X $,
daí é um campo de divisão disso acabou $\mathbb F_p $ ---
determinando $ E $ até o isomorfismo
(\cref{thm:b3:galois:splitting}). Por outro lado, em um \omterm{thm:b3:galois:splitting}{divisão
campo} $ L $ de $ X^q - X $, o conjunto $ E $ de suas raízes é um
\emph{subcampo}: $(x + y)^q = x^q + y^q $ iterando o
sonho de calouro $(a+b)^p = a^p + b^p $ ($ p \mid \binom pk $), e
$(xy)^q = x^qy^q $, $(x^{-1})^q = (x^q)^{-1}$; tem exatamente $ q $
elementos desde $ X^q - X $ é \omterm{def:b3:galois:separable}{separável}: sua derivada é
$ qX^{q-1} - 1 = -1$ (como $ p \mid q $), coprime a ele, então não
raízes repetidas. Por isso $ L = E $ tem $ q $ elementos.

(2) $ F $ é um morfismo de campo (sonho de calouro), injetivo
(campos), portanto bijetivo no finito $\mathbb F_q $. $ F^n =
\mathrm{id}$ ($ x^q = x $), e nenhuma potência menor é a identidade:
$ F^m = \mathrm{id}$ significa tudo $ q $ elementos são raízes de
$ X^{p^m} - X $, forçando $ p^m \geq q $. Então $\langle F\rangle $ é
cíclico de ordem $ n $; e não há outros automorfismos, por
o limite $\abs{\operatorname{Aut}} \leq [\,\mathbb F_q :
\mathbb F_p\,] = n $ provado abaixo
(\cref{prop:b3:galois:embeddings} com $ L = \mathbb F_q $, $ K =
\mathbb F_p $: automorfismos fixam o corpo principal).

(3) Se $\mathbb F_{p^m} \subseteq \mathbb F_{p^n}$, o \omterm{thm:b3:galois:tower}{lei da torre}
dá $ p^n = (p^m)^d $: $ m \mid n $. Inversamente se $ m \mid n $, então
$ p^m - 1 \mid p^n - 1$ (soma geométrica), então $ X^{p^m} - X $ divide
$ X^{p^n} - X $ (mesmo argumento em expoentes: $ X^{a} - 1 \mid X^{b}
- 1$ quando $ a \mid b $), e as raízes do primeiro dentro
$\mathbb F_{p^n}$ formar o subcampo obrigatório, de cardinalidade
$ p^m $ (separabilidade como em (1)).
\end{proof}

\begin{theorem}[Ciclicidade]\label{thm:b3:galois:cyclic}
Cada subgrupo finito do grupo multiplicativo de um corpo é
cíclico. Em particular $\mathbb F_q^\times \cong
\Z/(q-1)\Z $.\index{grupo multiplicativo de um corpo finito}
\end{theorem}

\begin{proof}
Deixar $ G \leq K^\times $ seja finito. Pelo teorema da estrutura
(\cref{cor:b3:modules:abelian}), $ G \cong \Z/d_1 \times \dots
\times \Z/d_s $ com $ d_1 \mid \dots \mid d_s $. Todo $ x \in G $
então satisfaz $ x^{d_s} = 1$; mas $ X^{d_s} - 1$ tem no \omterm{def:b3:rings:primemaximal}{máximo}
$ d_s $ raízes no campo $ K $: $\abs G = d_1\cdots d_s \leq d_s $,
forçando $ s = 1$: $ G $ é cíclico.
\end{proof}

\begin{example}\label{ex:b3:galois:f8}
$\mathbb F_8 = \mathbb F_2[X]/(X^3 + X + 1)$: a cúbica não tem
enraizar $\mathbb F_2$, portanto é irredutível. Escrita $\omega =
\bar X $: $\mathbb F_8^\times $ é cíclico de ordem $7$, então
\emph{todo} elemento $\neq 0, 1$ gera. Os subcampos de
$\mathbb F_{p^{12}}$ formam a rede divisora de $12$:
$\mathbb F_p, \mathbb F_{p^2}, \mathbb F_{p^3}, \mathbb F_{p^4},
\mathbb F_{p^6}, \mathbb F_{p^{12}}$ --- um primeiro, completo
instância do \omterm{thm:b3:galois:fundamental}{Correspondência de Galois}.
\end{example}

\section{Separabilidade e incorporações}

\begin{definition}\label{def:b3:galois:separable}
Um polinômio $ P \in K[X]$ é \emph{separável}\index{separável} se não tiver raiz repetida em um campo de divisão ---
equivalentemente $\gcd(P, P') = 1$ (uma raiz repetida é comum
raiz; inversamente, ao longo do campo de divisão, uma raiz comum é
repetido; e o mdc não muda sob \omterm{def:b3:galois:extension}{extensão de campo},
\cref{cor:b3:modules:descent}argumento). Um elemento \omterm{def:b3:galois:algebraic}{algébrico}
é separável se for polinômio mínimo é; uma extensão $ L/K $ é
separável se todos os seus elementos o forem.
\end{definition}

\begin{proposition}\label{prop:b3:galois:perfect}
Um \emph{irredutível} $ P \in K[X]$ é \omterm{def:b3:galois:separable}{separável} a menos que $ P' = 0$,
quais forças $\operatorname{char} K = p > 0$ e $ P \in K[X^p]$.
Consequentemente cada \omterm{def:b3:galois:algebraic}{algébrico} extensão de um campo de
característica $0$, e de um \omterm{thm:b3:galois:finitefields}{campo finito}, é \omterm{def:b3:galois:separable}{separável} (tal
campos são chamados \emph{perfeito}\index{perfeito}).
\end{proposition}

\begin{proof}
$\gcd(P, P')$ divide $ P $; se não for $1$, irredutibilidade
forças $\gcd = P $ (até uma constante), então $ P \mid P'$ com $\deg
P' < \deg P $: $ P' = 0$. Escrita $ P = \sum a_kX^k $: $ ka_k = 0$
para todos $ k $, então em característica $0$, $ P $ é constante (excluído);
em característica $ p $, $ a_k = 0$ a menos que $ p \mid k $: $ P =
Q(X^p)$. Mais de um \omterm{thm:b3:galois:finitefields}{campo finito}, todo elemento é um $ p $-º poder
(\omterm{thm:b3:galois:finitefields}{Frobenius} está em), então $ Q(X^p) = \sum b_k^p X^{pk} = (\sum
b_kX^k)^p $ não é irredutível: $ P' = 0$ não pode acontecer por
irredutível $ P $ lá também.
\end{proof}

\begin{proposition}[Contando incorporações]\label{prop:b3:galois:embeddings}
Deixar $ L = K(\alpha_1, \dots, \alpha_r)$ ser finito $ K $, e
$\sigma \colon K \to \Omega $ uma incorporação em um algebricamente
campo fechado. Então o número de extensões de $\sigma $ para $ L $
é no \omterm{def:b3:rings:primemaximal}{máximo} $[L:K]$, com igualdade se $ L/K $ é \omterm{def:b3:galois:separable}{separável}. Em particular $\abs{\operatorname{Aut}_K(L)} \leq [L:K]$.
\end{proposition}

\begin{proof}
Indução ativada $[L:K]$ através de passos \omterm{def:b3:groups:simple}{simples}. Para $ L = K(\alpha)$: um
extensão $\tau $ é determinado por $\tau(\alpha)$, que deve ser um
enraizar $\Omega $ de $\pi_\alpha^\sigma $; inversamente cada um desses
root fornece uma extensão ($ K(\alpha) \cong K[X]/(\pi_\alpha)$).
O número de extensões é o número de \emph{distinto} raízes
de $\pi_\alpha^\sigma $ em $\Omega $: no \omterm{def:b3:rings:primemaximal}{máximo} $\deg\pi_\alpha =
[K(\alpha):K]$, com igualdade se $\pi_\alpha $ é \omterm{def:b3:galois:separable}{separável}
(separabilidade de $\pi^\sigma $ e $\pi $ concordo: mdc com o
derivada é preservada por $\sigma $). Em geral, fator $ L =
K(\alpha_1)(\alpha_2, \dots)$: extensões de $\sigma $ para
$ K(\alpha_1)$ número $\leq [K(\alpha_1):K]$, e cada um se estende em
$\leq [L : K(\alpha_1)]$ formas por indução; multiplicar (torre
lei). No \omterm{def:b3:galois:separable}{separável} caso ambas as contagens sejam iguais: \omterm{def:b3:galois:algebraic}{mínimo
polinômios} sobre o campo maior $ K(\alpha_1)$ divida aqueles
acabou $ K $, portanto permanecem \omterm{def:b3:galois:separable}{separável}.
\end{proof}

\begin{theorem}[Elemento primitivo]\label{thm:b3:galois:primitive}
Cada finito \emph{\omterm{def:b3:galois:separable}{separável}} extensão é \omterm{def:b3:groups:simple}{simples}: $ L =
K(\gamma)$ para alguns $\gamma $.\index{teorema do elemento primitivo}
\end{theorem}

\begin{proof}
Se $ K $ é finito, então é $ L $, e um gerador $\gamma $ do
grupo cíclico $ L^\times $ (\cref{thm:b3:galois:cyclic}) faz isso.
Deixar $ K $ ser infinito; por indução basta tratar $ L =
K(\alpha, \beta)$. Deixar $ n = [L:K]$; por
\cref{prop:b3:galois:embeddings} há $ n $ distinto
$ K $-incorporações $\sigma_1, \dots, \sigma_n \colon L \to \Omega $
($\Omega $ um fechamento \omterm{def:b3:galois:algebraic}{algébrico}). O polinômio
\[
D(T)=\prod_{i<j}\bigl[\bigl(\sigma_i(\alpha)-\sigma_j(\alpha)
\bigr) + T\bigl(\sigma_i(\beta) - \sigma_j(\beta)\bigr)\bigr]
\]
não é identicamente zero: um fator desaparece de forma idêntica apenas se
$\sigma_i, \sigma_j $ concordo em ambos $\alpha $ e $\beta $, daí em diante
$ L $ --- excluído para $ i \neq j $. Como $ K $ é infinito, escolha $ c \in
K $ com $ D(c) \ne 0$: então o $ n $ elementos $\sigma_i(\alpha +
c\beta)$ são pares distintos, então $\gamma = \alpha + c\beta $
tem pelo menos $ n $ conjugados distintos em $\Omega $, ou seja\
$\deg \pi_\gamma \geq n $: $[K(\gamma):K] \geq n = [L:K]$ forças
$ L = K(\gamma)$.
\end{proof}

\section{A correspondência de Galois}

\begin{definition}\label{def:b3:galois:galois}
Uma extensão finita $ L/K $ é \emph{Galois}\index{Galois}
se for o campo de divisão de um \emph{\omterm{def:b3:galois:separable}{separável}} polinômio
acabou $ K $. Isso é \emph{Grupo Galois}\index{Grupo Galois} é
$\operatorname{Gal}(L/K) = \operatorname{Aut}_K(L)$, o grupo de
automorfismos de campo de $ L $ consertando $ K $ pontualmente.
\end{definition}

\begin{proposition}\label{prop:b3:galois:galoisorder}
Se $ L/K $ é \omterm{def:b3:galois:galois}{Galois}, então $\abs{\operatorname{Gal}(L/K)} =
[L:K]$; além disso $ L/F $ é \omterm{def:b3:galois:galois}{Galois} para cada campo intermediário $ K
\subseteq F \subseteq L $, e cada $ F $-incorporação $ L \to \Omega
\supseteq L $ tem imagem $ L $ (\emph{normalidade}).
\end{proposition}

\begin{proof}
Deixar $ L $ dividir o \omterm{def:b3:galois:separable}{separável} $ P $ sobre $ K $ e consertar um \omterm{def:b3:galois:closure}{algébrico
fechamento} $\Omega \supseteq L $. $ L/K $ é \omterm{def:b3:galois:separable}{separável}: é gerado
por raízes de $ P $; separabilidade de cada elemento segue do
caso de igualdade abaixo, mas vamos argumentar diretamente ---
\cref{prop:b3:galois:embeddings} aplicado aos geradores (raízes
do \omterm{def:b3:galois:separable}{separável} $ P $, cujo polinômios mínimos dividir $ P $)
rende exatamente $[L:K]$ extensões de $ K \hookrightarrow \Omega $
(na etapa indutiva, o polinômio mínimo de uma raiz de $ P $
sobre um campo intermediário ainda divide $ P $, portanto é
\omterm{def:b3:galois:separable}{separável}). Cada uma dessas incorporações $\tau \colon L \to \Omega $
permuta as raízes de $ P $ ($\tau $ fixa os coeficientes), e
$ L $ é gerado por eles: $\tau(L) = L $. Daí incorporações $=$
automorfismos: $\abs{\operatorname{Gal}(L/K)} = [L:K]$. Para
intermediário $ F $: $ L $ também é o campo de divisão de $ P $ sobre
$ F $, e $ P $ restos \omterm{def:b3:galois:separable}{separável}: $ L/F $ é \omterm{def:b3:galois:galois}{Galois}; o mesmo
argumento dá normalidade $ F $.
\end{proof}

\begin{lemma}[Artin]\label{lem:b3:galois:artin}
Deixar $ G $ ser um grupo finito de automorfismos de um corpo $ L $ e
$ K = L^G = \{x : \sigma(x) = x\ \forall\sigma \in G\}$ está fixo
campo. Então $[L : L^G] \leq \abs G $.
\end{lemma}

\begin{proof}
Deixar $ n = \abs G $, $ G = \{\sigma_1, \dots, \sigma_n\}$ e
suponha $ x_1, \dots, x_{n+1} \in L $ são linearmente independentes sobre
$ K $. O sistema linear homogêneo de $ n $ equações em $ n+1$
incógnitas $(c_j)$ sobre $ L $,
\[
\sum_{j=1}^{n+1} c_j\,\sigma_i(x_j) = 0
\qquad (i = 1, \dots, n),
\]
tem uma solução diferente de zero; escolha um com o \emph{menor} diferente de zero
entradas, digamos $ c_1, \dots, c_r \neq 0$ (renumeração), $ r \geq 2$
(um único $ c_j\sigma_i(x_j) = 0$ é impossível), normalizado
$ c_r = 1$. Nem todos $ c_j $ deitar em $ K $: a equação para $\sigma_i =
\mathrm{id}$ contradiria a independência; dizer $ c_1 \notin K $,
então $\tau(c_1) \ne c_1$ para alguns $\tau \in G $. Aplicar $\tau $ para todos
equações: desde $\tau\sigma_i $ atropela $ G $, o vetor
$(\tau(c_j))_j $ é outra solução; subtraindo, $(c_j -
\tau(c_j))_j $ é uma solução com menos entradas diferentes de zero (o
$ r $-ésima entrada $1 - 1 = 0$ desaparece, o primeiro não) e não
zero: contradição. Então qualquer $ n+1$ os elementos são dependentes: $[L:K]
\leq n $.
\end{proof}

\begin{theorem}[Teorema fundamental da teoria de Galois]
\label{thm:b3:galois:fundamental}
Deixar $ L/K $ seja um Extensão \omterm{def:b3:galois:galois}{Galois} com grupo $ G =
\operatorname{Gal}(L/K)$.\index{Correspondência de Galois}
\begin{enumerate}
  \item $ L^G = K $.
  \item Os mapas $ H \mapsto L^H $ e $ F \mapsto
        \operatorname{Gal}(L/F)$ são mutuamente inversos,
        bijeções de reversão de inclusão entre subgrupos de $ G $
        e campos intermediários $ K \subseteq F \subseteq L $;
        além disso $[L : L^H] = \abs H $ e $[L^H : K] = [G : H]$.
  \item $ H \trianglelefteq G $ se $ L^H/K $ é Galoise então
        restrição induz $\operatorname{Gal}(L^H/K) \cong
        G/H $.
\end{enumerate}
\end{theorem}

\begin{proof}
(1) Claramente $ K \subseteq L^G $. Por outro lado, deixe $\alpha \in L
\setminus K $; nós exibimos $\sigma \in G $ com $\sigma(\alpha) \ne
\alpha $. O polinômio mínimo $\pi_\alpha $ sobre $ K $ tem \omterm{def:b3:galois:extension}{grau}
$\geq 2$ e é \omterm{def:b3:galois:separable}{separável} ($ L/K $ \omterm{def:b3:galois:separable}{separável},
\cref{prop:b3:galois:galoisorder}), então tem outra raiz
$\beta \neq \alpha $ em um fechamento \omterm{def:b3:galois:algebraic}{algébrico} $\Omega \supseteq L $.
Estenda o $ K $-incorporação $ K(\alpha) \to \Omega $, $\alpha \mapsto
\beta $, para uma incorporação $\tau\colon L \to \Omega $
(\cref{prop:b3:galois:embeddings}); pela normalidade
(\cref{prop:b3:galois:galoisorder}) $\tau(L) = L $, então $\tau \in
G $, $\beta = \tau(\alpha) \in L $, e $\tau(\alpha) \neq
\alpha $.

(2) Para um subgrupo $ H $: $ L/L^H $ é \omterm{def:b3:galois:galois}{Galois}
(\cref{prop:b3:galois:galoisorder}), e
$\operatorname{Gal}(L/L^H) \supseteq H $ trivialmente, então $[L:L^H] =
\abs{\operatorname{Gal}(L/L^H)} \geq \abs H $; Lema de Artin
dá $[L:L^H] \leq \abs H $: igualdade e
$\operatorname{Gal}(L/L^H) = H $. Para um campo intermediário $ F $:
$ L/F $ \omterm{def:b3:galois:galois}{Galois} dá $ L^{\operatorname{Gal}(L/F)} = F $ por (1)
aplicado a $ L/F $. Os dois mapas são mutuamente inversos; eles
inclusões reversas evidentemente. \omterm{def:b3:galois:extension}{Graus}: $[L:L^H] = \abs H $ apenas
comprovado, e $[L^H:K] = [L:K]/[L:L^H] = \abs G/\abs H $.

(3) Para $\sigma \in G $ e $ H \leq G $: $\sigma(L^H) =
L^{\sigma H\sigma^{-1}}$ (verificação direta). Pela bijeção,
$\sigma(L^H) = L^H $ para todos $\sigma $ se $ H \trianglelefteq G $.
Agora se $ H \trianglelefteq G $, definir $ F = L^H $: todo $\sigma \in
G $ restringe-se a um automorfismo de $ F $, dando um morfismo $\rho
\colon G \to \operatorname{Aut}_K(F)$ com núcleo $\{\sigma :
\sigma\restriction_F = \mathrm{id}\} = \operatorname{Gal}(L/F) =
H $. Então $ G/H $ incorpora em $\operatorname{Aut}_K(F)$, de onde
$\abs{\operatorname{Aut}_K(F)} \geq [G:H] = [F:K]$; o inverso
a desigualdade sempre vale (\cref{prop:b3:galois:embeddings}):
$\abs{\operatorname{Aut}_K(F)} = [F:K]$ e $\rho $ está em. Isso
resta ver $ F/K $ é \omterm{def:b3:galois:galois}{Galois}: $ F $ é \omterm{def:b3:galois:separable}{separável} sobre $ K $
(dentro do \omterm{def:b3:galois:separable}{separável} $ L/K $), e $ F = K(\gamma)$
(\cref{thm:b3:galois:primitive}); o polinômio
$\prod_{\sigma \in G/H}\bigl(X - \sigma(\gamma)\bigr)$ (produto
sobre as imagens distintas, que se encontram em $ F $: $\sigma(F) =
L^{\sigma H\sigma^{-1}} = L^H = F $ pela normalidade de $ H $) tem
coeficientes
corrigido por $ G $, portanto em $ K $ por (1): é um polinômio \omterm{def:b3:galois:separable}{separável}
de $ K[X]$ dividido por $ F $, e suas raízes geram $ F $: $ F/K $ é
\omterm{def:b3:galois:galois}{Galois}. Por outro lado, se $ F = L^H $ com $ F/K $ \omterm{def:b3:galois:galois}{Galois}, normalidade de
$ F $ (\cref{prop:b3:galois:galoisorder}, aplicado a incorporações
$ F \to \Omega $ restrito de elementos de $ G $) dá
$\sigma(F) = F $ para todos $\sigma \in G $, ou seja\ $ H
\trianglelefteq G $.
\end{proof}

\section{Extensões ciclotômicas}

\begin{definition}\label{def:b3:galois:cyclotomic}
Deixar $ n \geq 1$ e $\zeta_n = \eu^{2\iu\pi/n}$. O $ n $-º
\emph{polinômio ciclotômico}\index{polinômio ciclotômico} é
$\Phi_n = \prod_{\gcd(k,n)=1,\ 1 \le k \le n} \bigl(X -
\zeta_n^k\bigr)$, de \omterm{def:b3:galois:extension}{grau} $\varphi(n)$; agrupando as raízes de
$ X^n - 1$ por ordem exata, $ X^n - 1 = \prod_{d \mid n}\Phi_d $,
o que mostra indutivamente que $\Phi_n \in \Z[X]$ (\omterm{def:b3:rings:pidufd}{Euclidiano}
divisão de polinômios inteiros mônicos).
\end{definition}

\begin{theorem}\label{thm:b3:galois:cyclotomicirred}
$\Phi_n $ é irredutível sobre $\Q $; por isso $[\Q(\zeta_n) : \Q] =
\varphi(n)$ e
\[
\operatorname{Gal}\bigl(\Q(\zeta_n)/\Q\bigr) \;\cong\;
(\Z/n\Z)^\times,
\qquad \sigma_a(\zeta_n) = \zeta_n^a .
\]
A extensão $\Q(\zeta_n)/\Q $ é assim \omterm{def:b3:galois:galois}{Galois} com \emph{abeliano}
grupo.
\end{theorem}

\begin{proof}
Deixar $ f = \pi_{\zeta_n}$, então $\Phi_n = fg $ com $ f, g \in \Z[X]$
mônico (lema de Gauss \cref{lem:b3:rings:gauss}: conteúdo
multiplique, todos os polinômios mônicos). \emph{Afirmação: se $\zeta $ é um
raiz de $ f $ e $ p \nmid n $ é primo, então $\zeta^p $ é uma raiz
de $ f $.} De outra forma $\zeta^p $ é uma raiz de $ g $ (é um primitivo
$ n $-ésima raiz da unidade), então $\zeta $ é uma raiz de $ g(X^p)$, e $ f
\mid g(X^p)$ em $\Z[X]$ (polinômio mínimo, depois Gauss novamente).
Reduzir mod $ p $: $\bar g(X^p) = \bar g(X)^p $ (\omterm{thm:b3:galois:finitefields}{Frobenius} sobre
$\mathbb F_p[X]$: coeficiente $ a^p = a $ e calouro
sonho), então $\bar f \mid \bar g^{\,p}$: $\bar f $ e $\bar g $
compartilhe um irredutível fator, e $\bar\Phi_n = \bar f\bar g $ tem
um fator repetido. Então o mesmo acontece $ X^n - \bar 1$; mas é
derivada $\bar nX^{n-1}$ é primo dele ($ p \nmid n $, e
$0$ não é uma raiz): contradição.

Toda raiz primitiva $\zeta_n^k $ ($\gcd(k, n) = 1$) é obtido
de $\zeta_n $ por sucessivas potências principais que não dividem $ n $
(fator $ k $): a afirmação se propaga, então toda raiz primitiva é uma
raiz de $ f $: $ f = \Phi_n $, irredutível. Consequentemente
$[\Q(\zeta_n):\Q] = \varphi(n)$, e $\Q(\zeta_n)$ é o
campo de divisão do \omterm{def:b3:galois:separable}{separável} $ X^n - 1$ (todas as raízes são potências
de $\zeta_n $): \omterm{def:b3:galois:galois}{Galois}. Um automorfismo $\sigma $ envia $\zeta_n $
para outra raiz primitiva $\zeta_n^{a(\sigma)}$, e $\sigma
\mapsto a(\sigma)$ é um morfismo injetivo em
$(\Z/n\Z)^\times $; ambos os grupos têm ordem $\varphi(n)$:
isomorfismo.
\end{proof}

\section{Régua e compasso}

\begin{definition}\label{def:b3:galois:constructible}
Identifique o avião com $\C $; começar de $\{0, 1\}$. Um ponto é
\emph{interpretável}\index{interpretável} se for
obtido por um número finito de interseções de linhas através de dois
pontos e círculos já construídos centrados em um ponto construído
ponto com raio a uma distância de dois pontos construídos.
\end{definition}

\begin{theorem}[Wantzel]\label{thm:b3:galois:wantzel}
$ z \in \C $ é construtível se houver uma torre $\Q = F_0
\subseteq F_1 \subseteq \dots \subseteq F_r $ com $[F_{i+1} :
F_i] = 2$ e $ z \in F_r $. Em particular, um número construtível
é \omterm{def:b3:galois:algebraic}{algébrico} de \omterm{def:b3:galois:extension}{grau} um poder de $2$ sobre
$\Q $.\index{Teorema de Wantzel}
\end{theorem}

\begin{proof}
($\Rightarrow $) As coordenadas da intersecção de duas linhas
através de pontos com coordenadas em um subcampo $ F \subseteq \R $
resolver um sistema linear sobre $ F $: eles ficam em $ F $. Linha--círculo
e círculo--círculo interseções levam, depois de eliminar o
parte linear (subtrair as duas equações do círculo dá uma reta),
para uma equação quadrática sobre $ F $: as novas coordenadas estão em $ F $
ou em $ F(\sqrt d)$ para alguns $ d \in F $, $ d > 0$. Por indução,
cada ponto construído tem coordenadas em uma torre de quadrática
extensões de $\Q $; e $ z = x + \iu y $ está em uma torre quadrática
também (adjunto $\iu $: mais um passo quadrático). O \omterm{def:b3:galois:extension}{grau}
conseqüência: $[\Q(z):\Q]$ divide $[F_r : \Q] = 2^r $ (torre
lei).

($\Leftarrow $) O números construtíveis formar um campo: somas e
diferenças por paralelogramos (os paralelos são construtível:
baixar e levantar perpendiculares duas vezes --- o clássico
perpendicular a um ponto usa um círculo e dois arcos);
produtos e quocientes pelas configurações de interceptação de Thales
(dados comprimentos $ a, b $ construir $ ab $ e $ a/b $ com semelhante
triângulos em dois raios). E o campo está fechado sob o quadrado
raízes: para $ a > 0$, o círculo de diâmetro $1 + a $ e o
perpendiculares no ponto de junção encontram-se na altura $\sqrt a $
(relação altitude-média geométrica em um triângulo retângulo); por um
complexo $ w = \rho\eu^{\iu\theta}$, construir $\sqrt\rho $ e
bissectar $\theta $ (a bissecção do ângulo é uma construção de bússola).
As partes reais e imaginárias dos membros de uma torre quadrática são
portanto construtível por indução na torre: cada passo
une raízes de uma quadrática, expressável por operações de campo
e uma raiz quadrada de um número já construído (o
fórmula quadrática; em característica $0$).
\end{proof}

\begin{corollary}\label{cor:b3:galois:impossible}
Os três problemas clássicos são insolúveis por governante e
bússola:\index{régua e compasso}
\begin{enumerate}
  \item \emph{Duplicação do cubo}: $\sqrt[3]2$ tem \omterm{def:b3:galois:extension}{grau}
        $3$, não um poder de $2$.
  \item \emph{Trissecção do ângulo}: trissecção $60^\circ $
        requer $\cos 20^\circ $, uma raiz do irredutível $8X^3
        - 6X - 1$: \omterm{def:b3:galois:extension}{grau} $3$.
  \item \emph{Quadratura do círculo}: $\sqrt\pi $ é transcendental
        ($\pi $ é --- o teorema de Lindemann, admitido aqui: é
        a prova pertence a um curso de teoria da transcendência).
\end{enumerate}
Além disso, o \omterm{def:b3:groups:normal}{normal} $ n $-gon é construtível se
$\varphi(n)$ é um poder de $2$ (Gauss--Wantzel; o ``se'' usa
método do problema do fim de semana, o ``somente se'' é
\cref{thm:b3:galois:wantzel} aplicado a $\zeta_n $, de \omterm{def:b3:galois:extension}{grau}
$\varphi(n)$). Para $ n = 7$: $\varphi(7) = 6$: o \omterm{def:b3:groups:normal}{normal}
heptágono é impossível; para $ n = 17$: $\varphi(17) = 16 = 2^4$:
construtível --- o problema do fim de semana o constrói.
\end{corollary}

\begin{proof}
(1) $ X^3 - 2$ é irredutível (\omterm{thm:b3:rings:criteria}{Eisenstein}). (2) De $\cos
3\theta = 4\cos^3\theta - 3\cos\theta $ com $3\theta =
60^\circ $: $8c^3 - 6c = 1$ para $ c = \cos 20^\circ $; o cúbico
$8X^3 - 6X - 1$ não tem raiz racional (candidatos $\pm1, \pm\frac
12, \pm\frac14, \pm\frac18$ falhar), portanto é irredutível: \omterm{def:b3:galois:extension}{grau}
$3$. Um general $60^\circ $ ângulo é construtível, então um trissetor
construiria $ c $. (3) Se $\sqrt\pi $ eram construtível isso
seria \omterm{def:b3:galois:algebraic}{algébrico}, portanto também $\pi $. O $ n $ declaração -gon: o
\omterm{def:b3:galois:extension}{grau} de $\zeta_n $ é $\varphi(n)$
(\cref{thm:b3:galois:cyclotomicirred}); a necessidade decorre de
Querzel; por suficiência, o \omterm{def:b3:galois:galois}{Grupo Galois}, abeliano de ordem
$2^m $, admite uma cadeia de índices $2$ subgrupos (um finito $2$-grupo
faz: \cref{exo:b3:groups:10}), cujos campos fixos formam um
torre quadrática terminando em $\Q(\zeta_n)$
(\cref{thm:b3:galois:fundamental}); concluir por
\cref{thm:b3:galois:wantzel}.
\end{proof}

\section{Solubilidade por radicais}

\begin{definition}\label{def:b3:galois:radical}
Uma extensão $ L/K $ (característica $0$ ao longo desta seção)
é \emph{radical} se houver uma torre $ K = F_0 \subseteq \dots
\subseteq F_r = L $ com $ F_{i+1} = F_i(\alpha_i)$,
$\alpha_i^{n_i} \in F_i $: cada etapa é adjacente a uma $ n_i $-ésima raiz. Um
polinômio $ P \in K[X]$ é \emph{solucionável por
radicais}\index{solucionável por
radicais} se for campo de divisão é
contida em alguma extensão radical de $ K $.
\end{definition}

\begin{lemma}\label{lem:b3:galois:cyclicsteps}
Deixar $ K $ contém um primitivo $ n $-ésima raiz da unidade $\zeta $, ou seja\
$\zeta $ de ordem $ n $ em $ K^\times $, e $ a \in K^\times $. Então
$ K(\sqrt[n]a)/K $ é \omterm{def:b3:galois:galois}{Galois} com \emph{cíclico} grupo. Por outro lado
--- não é necessário abaixo --- toda extensão cíclica de \omterm{def:b3:galois:extension}{grau} $ n $ é
deste formulário. Além disso $ K(\zeta_n)/K $ é \omterm{def:b3:galois:galois}{Galois} com
\emph{abeliano} grupo, para qualquer $ K $ de característica $0$.
\end{lemma}

\begin{proof}
$ X^n - a $ é \omterm{def:b3:galois:separable}{separável} ($\gcd $ com $ nX^{n-1}$: $ a \neq 0$) e
se divide em $ K(\alpha)$, $\alpha^n = a $: suas raízes são o
$\zeta^k\alpha \in K(\alpha)$. Então $ K(\alpha)/K $ é \omterm{def:b3:galois:galois}{Galois}; o
mapa $\sigma \mapsto \sigma(\alpha)/\alpha \in \mu_n = \langle
\zeta\rangle $ é um morfismo injetivo ($\sigma\tau(\alpha) =
\sigma(\tau(\alpha)/\alpha \cdot \alpha) =
\tau(\alpha)/\alpha\cdot\sigma(\alpha)$, o quociente está em
$ K $), em um grupo cíclico: $\operatorname{Gal}$ é cíclico. O
o inverso é a teoria de Kummer, da qual não precisaremos (veja o
observação abaixo). Para $ K(\zeta_n)$: divide o \omterm{def:b3:galois:separable}{separável} $ X^n -
1$, e $\sigma \mapsto a(\sigma)$ com $\sigma(\zeta_n) =
\zeta_n^{a(\sigma)}$ incorpora o grupo no abeliano
$(\Z/n\Z)^\times $ como em \cref{thm:b3:galois:cyclotomicirred}
(a injetividade só precisa $\zeta_n $ para gerar as raízes de
unidade envolvida).
\end{proof}

\begin{theorem}[Galois]\label{thm:b3:galois:solvable}
Deixar $ K $ ser de característica $0$ e $ P \in K[X]$ com \omterm{thm:b3:galois:splitting}{divisão
campo} $ L $. Se $ P $ é \omterm{def:b3:galois:radical}{solucionável por radicais}, então
$\operatorname{Gal}(L/K)$ é um grupo \omterm{def:b3:groups:derived}{solucionável}. (O inverso é
também é verdade; não precisaremos disso.)\index{Solubilidade de Galois
teorema}
\end{theorem}

\begin{proof}
Etapa 1: \emph{ampliar o \omterm{def:b3:galois:radical}{radical} torre para um \omterm{def:b3:galois:galois}{Galois} um.} Deixar $ L
\subseteq M $ com $ M/K $ \omterm{def:b3:galois:radical}{radical}, com expoentes raiz $ n_1, \dots,
n_r $ e $ n = n_1\cdots n_r $. Primeiro contíguo $\zeta_n $: a torre
$ K \subseteq K(\zeta_n) \subseteq M(\zeta_n)$ ainda está \omterm{def:b3:galois:radical}{radical}
($\zeta_n $ é uma raiz de unidade: uma \omterm{def:b3:galois:radical}{radical} etapa, $\zeta_n^n = 1$),
e seus passos além do primeiro acontecem em campos que contêm o
precisava de raízes de unidade. A seguir, substitua $ M(\zeta_n)$ pelo
composto $ N $ de tudo $\sigma(M(\zeta_n))$, $\sigma $ variando
o (finitamente muitos) $ K $-incorporações de $ M(\zeta_n)$ em um fixo
fechamento \omterm{def:b3:galois:algebraic}{algébrico}: $ N $ é o campo de divisão do produto de
polinômios mínimos de um grupo gerador (característica $0$:
finito e \omterm{def:b3:galois:separable}{separável}), por isso $ N/K $ é \omterm{def:b3:galois:galois}{Galois}; e $ N $ é
\omterm{def:b3:galois:radical}{radical} sobre $ K $: cada $\sigma(M(\zeta_n))$ é \omterm{def:b3:galois:radical}{radical} sobre $ K $
(aplicar $\sigma $ para um \omterm{def:b3:galois:radical}{radical} torre) e um composto de \omterm{def:b3:galois:radical}{radical}
extensões é \omterm{def:b3:galois:radical}{radical} (concatenar as torres: se $ F'/K $ é
\omterm{def:b3:galois:radical}{radical} com torre adjacente $\beta_j $, então $ F''(\beta_j)$-tipo
passos permanecem \omterm{def:b3:galois:radical}{radical} sobre qualquer base maior).

Etapa 2: \emph{ler solvabilidade fora do \omterm{def:b3:galois:galois}{Galois} torre.} Então assuma
$ L \subseteq N $, $ N/K $ \omterm{def:b3:galois:galois}{Galois} e \omterm{def:b3:galois:radical}{radical} com torre $ K
\subseteq K(\zeta_n) = E_0 \subseteq E_1 \subseteq \dots
\subseteq E_s = N $, cada $ E_{i+1} = E_i(\sqrt[n_i]{a_i})$ com
$\zeta_{n_i} \in E_0 \subseteq E_i $. Deixar $ G =
\operatorname{Gal}(N/K)$ e $ G_i = \operatorname{Gal}(N/E_i)$:
uma cadeia decrescente $ G \supseteq G_0 \supseteq G_1 \supseteq
\dots \supseteq G_s = \{e\}$. Cada $ E_{i+1}/E_i $ é \omterm{def:b3:galois:galois}{Galois} com
grupo cíclico (\cref{lem:b3:galois:cyclicsteps}), então pelo
teorema fundamental aplicado ao Extensão \omterm{def:b3:galois:galois}{Galois} $ N/E_i $
(\cref{thm:b3:galois:fundamental}(3), com grupo ambiente $ G_i $):
$ G_{i+1} \trianglelefteq G_i $ com $ G_i/G_{i+1} \cong
\operatorname{Gal}(E_{i+1}/E_i)$ cíclico. De forma similar $ E_0/K $ é
\omterm{def:b3:galois:galois}{Galois} com grupo abeliano $ G/G_0$
(\cref{lem:b3:galois:cyclicsteps}). A cadeia exibe $ G $ como
\omterm{def:b3:groups:derived}{solucionável} (\cref{prop:b3:groups:derived}). Finalmente
$\operatorname{Gal}(L/K)$ é um \emph{quociente} de $ G $: $ L/K $ é
\omterm{def:b3:galois:galois}{Galois} ($ P $ \omterm{def:b3:galois:separable}{separável} em característica $0$) e restrição $ G
\to \operatorname{Gal}(L/K)$ está em
(\cref{thm:b3:galois:fundamental}(3) com $ H =
\operatorname{Gal}(N/L)$); quocientes de grupos solucionáveis são
\omterm{def:b3:groups:derived}{solucionável}.
\end{proof}

\begin{corollary}[Insolubilidade do quíntico]
\label{cor:b3:galois:quintic}
Existem polinômios de \omterm{def:b3:galois:extension}{grau} $5$ sobre $\Q $ que não são
\omterm{def:b3:galois:radical}{solucionável por radicais}: por exemplo $ X^5 - 4X + 2$, cujo \omterm{def:b3:galois:galois}{Galois}
grupo é $ S_5$ (\cref{exo:b3:galois:11}), um insolúvel grupo
(\cref{cor:b3:groups:snnotsolvable}). Nenhuma fórmula geral em
radicais pode existir para \omterm{def:b3:galois:extension}{grau} $\geq 5$.\index{quintico,
insolubilidade}
\end{corollary}

\begin{remark}\label{rem:b3:galois:converse}
O inverso de \cref{thm:b3:galois:solvable} --- um \omterm{def:b3:groups:derived}{solucionável}
\omterm{def:b3:galois:galois}{Grupo Galois} implica solvabilidade por radicais --- é provado por
descendo o série derivada e mostrando cada extensão cíclica
(com raízes de unidade suficientes) é \omterm{def:b3:galois:radical}{radical}, através \emph{Lagrange
resolventes}; isso explica por que \omterm{def:b3:galois:extension}{graus} $2, 3, 4$ tem fórmulas:
$ S_2, S_3, S_4$ são \omterm{def:b3:groups:derived}{solucionável}
(\cref{ex:b3:groups:solvableexamples}). Deixamos admitido em
este nível; um tratamento completo pertence a um curso de mestrado, mas
\cref{exo:b3:galois:8} torna concreto para o cúbico.
\end{remark}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:b3:galois:1}
Mostrar $[\Q(\sqrt2, \sqrt3):\Q] = 4$, que $\Q(\sqrt2 + \sqrt3) =
\Q(\sqrt2, \sqrt3)$, e calcule o polinômio mínimo de
$\sqrt2 + \sqrt3$ sobre $\Q $.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b3:galois:2}
Deixar $\alpha = \sqrt[3]2$. Mostre isso $\Q(\alpha)/\Q $ não é
\omterm{def:b3:groups:normal}{normal} (exibir uma incorporação $\Q(\alpha) \to \C $ cuja imagem é
não $\Q(\alpha)$), determine o campo de divisão $ L $ de $ X^3 -
2$ e $[L:\Q]$ e verifique $\operatorname{Aut}_\Q(\Q(\alpha)) =
\{\mathrm{id}\}$: para não \omterm{def:b3:galois:galois}{Galois} extensões, o automorfismo
grupo pode ser muito menor do que o \omterm{def:b3:galois:extension}{grau}.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b3:galois:3}
Construir $\mathbb F_9$ como $\mathbb F_3[X]/(X^2+1)$ e encontre um
gerador de $\mathbb F_9^\times $. Liste o mônico irredutível
polinômios de \omterm{def:b3:galois:extension}{graus} $1, 2, 3$ sobre $\mathbb F_2$ e verifique
$ X^8 - X = X(X+1)(X^3+X+1)(X^3+X^2+1)$ sobre $\mathbb F_2$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:galois:4}
(a) Encontre todas as raízes primitivas módulo $7$ e módulo $11$ (ou seja,\
geradores de $\mathbb F_7^\times $, $\mathbb F_{11}^\times $).
(b) Mostre que para $ p $ chance, $ x \in \mathbb F_p^\times $ é um
quadrado se e sef $ x^{(p-1)/2} = 1$ (critério de Euler) e recuperar
o critério para $-1$ de \cref{pb:b3:rings:1}.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:galois:5}
Mostre isso $\mathbb F_{p^m} \cap \mathbb F_{p^n} = \mathbb
F_{p^{\gcd(m,n)}}$ e $\mathbb F_{p^m}\mathbb F_{p^n} = \mathbb
F_{p^{\operatorname{lcm}(m,n)}}$ dentro de um fixo \omterm{def:b3:galois:closure}{algébrico
fechamento} $\bar{\mathbb F}_p $ e descreva
$\operatorname{Gal}(\mathbb F_{p^n}/\mathbb F_{p^m})$ para $ m
\mid n $.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:galois:6}
Deixar $ I_d(q)$ seja o número de mônico irredutível polinômios de
\omterm{def:b3:galois:extension}{grau} $ d $ sobre $\mathbb F_q $. Provar
\[
X^{q^n} - X \;=\; \prod_{d \mid n}\ \prod_{P \text{ irred.
monic, } \deg P = d} P ,
\qquad\text{hence}\qquad
q^n = \sum_{d \mid n} d\, I_d(q).
\]
Deduzir $ I_1, I_2, I_3, I_4$ explicitamente, e $ I_d(q) \geq 1$ para
cada $ d $ (então extensões $\mathbb F_{q^d}/\mathbb F_q $ existe como
quocientes $\mathbb F_q[X]/(P)$ para todos $ d $).
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:galois:7}
Determinar $\operatorname{Gal}(\Q(\sqrt2,\sqrt3)/\Q)$ e o
rede completa de campos intermediários. A mesma pergunta para
campo de divisão de $(X^2-2)(X^2-3)(X^2-6)$ --- o que você
percebeu?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:galois:8}
(A cúbica, resolvida por seu grupo) Deixe $ P = X^3 + pX + q \in
\Q[X]$ ser irredutível com raízes $ x_1, x_2, x_3$ e \omterm{thm:b3:galois:splitting}{divisão
campo} $ L $. Deixar $\delta = (x_1 - x_2)(x_1 - x_3)(x_2 - x_3)$ e
$\Delta = \delta^2 = -4p^3 - 27q^2$ (admita este clássico
identidade ou verificá-la expandindo funções simétricas).
(a) Mostrar $\operatorname{Gal}(L/\Q) \cong A_3$ ou $ S_3$, de acordo
para saber se $\Delta $ é ou não é um quadrado em $\Q $.
(b) Com $ j = \zeta_3$, defina os resolventes de Lagrange $ u = x_1
+ jx_2 + j^2x_3$ e $ v = x_1 + j^2x_2 + jx_3$. Mostrar $ u^3 + v^3
= -27q $ e $ uv = -3p $, e resolva para $ u^3, v^3$: Cardano
as fórmulas desaparecem. Onde foi solvabilidade de $ S_3$ se acostumar?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:galois:9}
Em $\Q(\zeta_5)$: mostre que o subcampo quadrático único é
$\Q(\sqrt5)$, através das somas de Gauss $\eta_0 = \zeta_5 + \zeta_5^4$,
$\eta_1 = \zeta_5^2 + \zeta_5^3$: calcular $\eta_0 + \eta_1$ e
$\eta_0\eta_1$, e deduza $\cos\frac{2\pi}5 = \frac{\sqrt5 -
1}4$. Conclua que o pentágono regular é construtível.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b3:galois:10}
Deixar $ K = \mathbb F_p(S, T)$ (funções racionais em dois
indeterminados) e $ L = K(S^{1/p}, T^{1/p})$.
(a) Mostrar $[L:K] = p^2$ e isso $\alpha^p \in K $ para cada
$\alpha \in L $.
(b) Deduza que $ L/K $ é \emph{não} \omterm{def:b3:groups:simple}{simples}: nenhum elemento primitivo
existe --- a inseparabilidade é fatal para
\cref{thm:b3:galois:primitive}.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b3:galois:11}
Deixar $ P = X^5 - 4X + 2$ e $ G $ isso é \omterm{def:b3:galois:galois}{Grupo Galois} sobre $\Q $,
agindo sobre o $5$ raízes.
(a) Mostrar $ P $ é irredutível, e deduza $5 \mid \abs G $; concluir
isso $ G $ contém um $5$-ciclo (Cauchy,
\cref{thm:b3:groups:cauchy}).
(b) Mostre, estudando as variações de $ x \mapsto x^5 - 4x +
2$, que $ P $ tem exatamente $3$ raízes reais; deduzir esse complexo
conjugação se restringe a uma transposição em $ G $.
(c) Mostre que um subgrupo de $ S_5$ contendo uma transposição e
um $5$-ciclo é $ S_5$ \emph{(conjugue a transposição por
poderes do ciclo)}. Concluir $ G \cong S_5$ e, com
\cref{thm:b3:galois:solvable}, que $ P $ não é \omterm{def:b3:galois:radical}{solucionável por
radicais}.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b3:galois:12}
(O diédrico quártico) Deixe $\alpha = \sqrt[4]2$ e $ L =
\Q(\alpha, \iu)$, o campo de divisão de $ X^4 - 2$ sobre $\Q $.
(a) Mostrar $[L : \Q] = 8$ e isso $ G =
\operatorname{Gal}(L/\Q)$ é gerado por $\sigma\colon
\alpha \mapsto \iu\alpha,\ \iu \mapsto \iu $ e complexo
conjugação $\tau $, com $\sigma^4 = \tau^2 = e $ e
$\tau\sigma\tau = \sigma^{-1}$: $ G \cong D_4$.
(b) Liste a rede de subgrupos de $ D_4$ (dez subgrupos) e
combine cada um com seu campo fixo; verificar em particular que
$\Q(\sqrt2)$, $\Q(\iu)$, $\Q(\iu\sqrt2)$ são os três
subcampos quadráticos e localize $\Q(\alpha)$,
$\Q(\iu\alpha)$, $\Q(\sqrt2, \iu)$.
(c) Quais campos intermediários são \omterm{def:b3:galois:galois}{Galois} sobre $\Q $? Partida
sua resposta contra o subgrupos normais de $ D_4$ e
explique por que $\Q(\alpha)/\Q $ falha enquanto $\Q(\sqrt2)/\Q $
consegue.
\end{exercise}

\section{Problema: Gauss e o 17-gon regular}

\begin{problem}[{Problema de fim de semana --- construtibilidade do
17 anos}]\label{pb:b3:galois:1}
Em 30 de março de 1796, Gauss, de dezenove anos, mostrou que o
\omterm{def:b3:groups:normal}{normal} $17$-gon é construtível --- o primeiro progresso no
questão desde a antiguidade. Reconstruímos seu cálculo com o
ferramentas deste capítulo. Definir $\zeta = \eu^{2\iu\pi/17}$, $ L =
\Q(\zeta)$, $ G = \operatorname{Gal}(L/\Q)$.

\medskip
\noindent\textbf{Part I --- The group and its filtration.}
\begin{enumerate}
  \item Justificar: $[L:\Q] = 16$, $ G \cong (\Z/17\Z)^\times $,
        cíclico de ordem $16$. Verifique isso $3$ é um gerador de
        $(\Z/17\Z)^\times $ \emph{(calcular as potências de $3$
        módulo $17$: $3, 9, 10, 13, 5, 15, 11, 16, \dots $)}.
  \item Deixar $\sigma \in G $ com $\sigma(\zeta) = \zeta^3$, e
        $ H_k = \langle \sigma^{2^k}\rangle $ para $ k = 0, \dots,
        4$. Mostre isso $ G = H_0 \supset H_1 \supset H_2 \supset
        H_3 \supset H_4 = \{e\}$ com cada índice $[H_k :
        H_{k+1}] = 2$, e que os campos fixos $\Q = L_0
        \subset L_1 \subset L_2 \subset L_3 \subset L_4 = L $
        formam uma torre de extensões quadráticas.
  \item Concluir \emph{a priori}, usando
        \cref{thm:b3:galois:wantzel}, que $\zeta $ --- daí o
        $17$-gon --- é construtível. O resto do problema
        torna a torre explícita.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The periods of length 8.}
Defina o \emph{Períodos de Gauss}
\[
\eta_0 = \sum_{k \text{ even}} \zeta^{3^k \bmod 17}
= \zeta^{1} + \zeta^{9} + \zeta^{13} + \zeta^{15} + \zeta^{16} +
\zeta^{8} + \zeta^{4} + \zeta^{2},
\qquad
\eta_1 = \sum_{k \text{ odd}} \zeta^{3^k \bmod 17}.
\]
\begin{enumerate}[resume]
  \item Mostre isso $\eta_0, \eta_1$ são fixados por $ H_1$ e trocado
        por $\sigma $; deduzir $\eta_0, \eta_1 \in L_1$ e isso
        eles são as duas raízes de uma quadrática sobre $\Q $.
  \item Calcular $\eta_0 + \eta_1 = -1$. Mostrar $\eta_0\eta_1 = -4$
        \emph{(cada produto $\zeta^a\zeta^b $ é algum $\zeta^c $,
        $ c \neq 0$; conte quantas vezes cada $ c $ ocorre, ou
        argumentar que o produto é um número inteiro racional fixado por
        $ G $, igual à soma de todos $64$ produtos e uso
        que cada resíduo diferente de zero aparece com a mesma frequência)}.
  \item Deduzir $\eta_0 = \frac{-1 + \sqrt{17}}2$, $\eta_1 =
        \frac{-1-\sqrt{17}}2$ (identifique qual é qual
        numericamente: $\eta_0 \approx 1.56$), e $ L_1 =
        \Q(\sqrt{17})$.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Periods of length 4 and 2.}
Definir
\[
\beta_0 = \zeta + \zeta^{13} + \zeta^{16} + \zeta^{4},\quad
\beta_1 = \zeta^3 + \zeta^5 + \zeta^{14} + \zeta^{12},\quad
\beta_2 = \zeta^9 + \zeta^{15} + \zeta^{8} + \zeta^{2},\quad
\beta_3 = \zeta^{10} + \zeta^{11} + \zeta^{7} + \zeta^{6}.
\]
\begin{enumerate}[resume]
  \item Mostrar $\beta_0 + \beta_2 = \eta_0$, $\beta_1 + \beta_3 =
        \eta_1$, e isso $\beta_0, \beta_2$ são fixados por $ H_2$,
        trocado por $\sigma^2$.
  \item Calcular $\beta_0\beta_2 = -1$ e $\beta_1\beta_3 = -1$
        \emph{(expandir: os dezesseis expoentes obtidos cobrem
        $1, \dots, 16$ exatamente uma vez)}.
  \item Deduzir $\beta_0 = \frac{\eta_0 + \sqrt{\eta_0^2 + 4}}2$
        (verifique o sinal numericamente: $\beta_0 \approx 2.05$) e
        a fórmula análoga para $\beta_1$; por isso $ L_2 =
        \Q(\beta_0)$, quadrático sobre $ L_1$.
  \item Deixar $\gamma_0 = \zeta + \zeta^{16} =
        2\cos\frac{2\pi}{17}$ e $\gamma_1 = \zeta^{13} +
        \zeta^4$. Mostrar $\gamma_0 + \gamma_1 = \beta_0$ e
        $\gamma_0\gamma_1 = \beta_1$, para que $\gamma_0 =
        \frac{\beta_0 + \sqrt{\beta_0^2 - 4\beta_1}}2$.
  \item Monte a cadeia de fórmulas expressando
        $\cos\frac{2\pi}{17}$ por raízes quadradas aninhadas e dê um
        verificação decimal ($\cos\frac{2\pi}{17} \approx 0.93247$).
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Epilogue.}
\begin{enumerate}[resume]
  \item Onde exatamente o argumento usou isso $17$ é um
        \emph{Fermat prime} ($17 = 2^{2^2} + 1$)? Mostre isso para
        um primo $ p $, o regular $ p $-gon é construtível se $ p
        = 2^{2^t} + 1$ para alguns $ t $ \emph{(se $ p - 1 = 2^m $,
        mostrar $ m $ deve ser em si um poder de $2$)}.
  \item Deduza a lista completa de construtível regular
        $ n $-gons para $ n \leq 20$, usando Gauss--Wantzel
        critério de \cref{cor:b3:galois:impossible}.
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Gauss sums and quadratic
reciprocity.} Os períodos da Parte II escondem um tesouro. Por um
primo ímpar $ p $, o \emph{Símbolo de legenda} $\bigl(\frac
ap\bigr)$ é $+1$ se $ a $ é um mod quadrado diferente de zero $ p $, $-1$ se
não é, $0$ se $ p \mid a $; \cref{exo:b3:galois:4}(b)
(critério de Euler) dá $\bigl(\frac ap\bigr) \equiv
a^{(p-1)/2} \pmod p $, de onde multiplicatividade. Escrever $\zeta =
\eu^{2\iu\pi/p}$, $ p^* = (-1)^{(p-1)/2}p $, e defina o
\emph{Soma de Gauss}
\[
g \;=\; \sum_{a=1}^{p-1}\Bigl(\frac ap\Bigr)\zeta^a .
\]
\begin{enumerate}[resume]
  \item Mostrar $\sum_{a=1}^{p-1}\bigl(\frac ap\bigr) = 0$
        (tantos quadrados quanto não quadrados) e prove a
        forma alternativa $ g = \sum_{a=0}^{p-1}\zeta^{a^2}$
        \emph{(cada quadrado diferente de zero é atingido duas vezes, e
        $\sum_{a}\zeta^a = 0$)}. Para $ p = 17$: relacionar $ g $ para
        os períodos da Parte II --- mostram $ g = \eta_0 - \eta_1$
        \emph{(o mod de quadrados $17$ são exatamente iguais
        potências do gerador $3$)}.
  \item Provar $ g^2 = p^*$: expandir
        \[
        g^2 = \sum_{a,b\neq0}\Bigl(\frac{ab}p\Bigr)
        \zeta^{a+b}
        = \sum_{c}\ \sum_{a \neq 0}\Bigl(\frac{a(c -
        a)}p\Bigr)\zeta^{c}
        \]
        (definir $ b = c - a $), substitua $ c - a = at $ avaliar
        a soma interna como $\bigl(\frac{-1}p\bigr)(p - 1)$ para
        $ c = 0$ e $-\bigl(\frac{-1}p\bigr)$ caso contrário, e
        conclua com a questão 14. Verifique numericamente: para $ p =
        17$, $(\eta_0 - \eta_1)^2 = 17$ (Parte II).
  \item Deduzir $\sqrt{p^*} \in \Q(\zeta_p)$, e concluir
        que o \emph{exclusivo} subcampo quadrático de
        $\Q(\zeta_p)$ é $\Q(\sqrt{p^*})$ --- único porque
        $\operatorname{Gal}(\Q(\zeta_p)/\Q)$ é cíclico
        (\cref{thm:b3:galois:cyclotomicirred}) e um cíclico
        grupo tem exatamente um subgrupo de índice $2$. (Cada
        campo quadrático incorporado em algum campo ciclotômico ---
        este é o primeiro caso do Kronecker-Weber
        teorema, cuja forma geral está muito à frente.)
  \item Agora deixe $ q \neq p $ ser outro primo ímpar. Trabalhando em
        o anel $\Z[\zeta]$ módulo $ q $, provar
        \[
        g^q \equiv \Bigl(\frac qp\Bigr)\,g \pmod{q\Z[\zeta]}
        \]
        \emph{(sonho de calouro: $(x + y)^q \equiv x^q + y^q $
        moda $ q $ em qualquer anel comutativo; então $ g^q \equiv
        \sum_a\bigl(\frac ap\bigr)^q\zeta^{aq}$, reindexar $ b =
        aq $ e retire $\bigl(\frac{q^{-1}}p\bigr) =
        \bigl(\frac qp\bigr)$)}.
  \item Por outro lado, $ g^q = g\,(g^2)^{(q-1)/2} =
        g\,(p^*)^{(q-1)/2}$; usando o mod critério de Euler $ q $,
        deduzir $ g^q \equiv \bigl(\frac{p^*}q\bigr)g
        \pmod{q\Z[\zeta]}$, então --- multiplicando os dois
        expressões para $ g^q $ por $ g $ e usando $ g^2 = p^*$,
        mod invertível $ q $ --- concluir
        \[
        \Bigl(\frac qp\Bigr) = \Bigl(\frac{p^*}q\Bigr) .
        \]
        \emph{(Por que uma congruência entre os inteiros
        $\pm p^*$ módulo $ q\Z[\zeta]$ implica sua igualdade?
        Cruzar com $\Z $.)}
  \item Desdobrar $\bigl(\frac{p^*}q\bigr) =
        \bigl(\frac{-1}q\bigr)^{(p-1)/2}\bigl(\frac pq\bigr)$
        e $\bigl(\frac{-1}q\bigr) = (-1)^{(q-1)/2}$ para
        obter o \emph{lei da reciprocidade quadrática}:
        \[
        \Bigl(\frac pq\Bigr)\Bigl(\frac qp\Bigr)
        = (-1)^{\frac{p-1}2\cdot\frac{q-1}2} .
        \]
        Verifique em $(p, q) = (17, 3)$ listando os quadrados
        moda $17$ e mod $3$, e use-o para decidir em três
        linhas se $ x^2 \equiv 219 \pmod{383}$ é \omterm{def:b3:groups:derived}{solucionável}
        ($383$ é primo, $219 = 3\cdot73$).
\end{enumerate}

\medskip
\noindent\textbf{Part VI --- Counting irredutível polynomials:
the prime number theorem of $\mathbb F_q[X]$.} Consertar um poder principal
$ q $ e deixe $ N_q(n)$ seja o número de \emph{Mônica irredutível}
polinômios de \omterm{def:b3:galois:extension}{grau} $ n $ sobre $\mathbb F_q $; lembrar de
\cref{exo:b3:galois:6} a fatoração de $ X^{q^n} - X $ e
a identidade $ q^n = \sum_{d\mid n}d\,N_q(d)$, que agora
inverter, reinterpretar e explorar.
\begin{enumerate}[resume]
  \item (Palavras) Chame uma palavra $ w \in \mathbb F_q^n $
        \emph{primitivo} se não for um poder $ u^{n/d} =
        u\cdots u $ de uma palavra estritamente mais curta $ u $, e deixe
        $ A(d)$ seja o número de palavras primitivas de comprimento $ d $.
        Mostre que cada palavra de comprimento $ n $ é exclusivamente um poder
        de uma palavra primitiva de algum comprimento $ d \mid n $, para que
        $ q^n = \sum_{d \mid n}A(d)$; comparando com
        \cref{exo:b3:galois:6}, concluir $ A(d) = d\,N_q(d)$
        para cada $ d $, e explique essa coincidência por um
        bijeção explícita: um elemento $\alpha \in
        \mathbb F_{q^d}$ de \omterm{def:b3:galois:extension}{grau} $ d $ tem \omterm{thm:b3:galois:finitefields}{Frobenius} \omterm{def:b3:groups:action}{órbita}
        $(\alpha, \alpha^q, \dots, \alpha^{q^{d-1}})$ de
        exatamente $ d $ elementos distintos e elementos de \omterm{def:b3:galois:extension}{grau}
        $ d $ corresponder $ d $-para-um para irredutíveis de \omterm{def:b3:galois:extension}{grau}
        $ d $.
  \item Prove o \emph{Mfórmula de inversão de obius}: se $ f(n) =
        \sum_{d\mid n}g(d)$ para todos $ n $, então $ g(n) =
        \sum_{d\mid n}\mu(d)\,f(n/d)$, onde $\mu $ é o
        Mfunção Óbius ($\mu(m) = (-1)^{\#\text{prime
        factors}}$ se $ m $ é \omterm{def:b3:modules:free}{livre} de quadrados, $0$ de outra forma)
        \emph{(lema chave: $\sum_{d \mid m}\mu(d) = 0$ para $ m >
        1$ --- emparelhar os divisores com e sem um primo fixo
        fator)}. Deduzir
        \[
        N_q(n) = \frac1n\sum_{d \mid n}\mu(d)\,q^{n/d} .
        \]
  \item Mostrar $ N_q(n) \geq \frac1n\bigl(q^n - 2q^{n/2}\bigr) > 0$
        para cada $ n \geq 1$: uma nova prova de que $\mathbb F_{q^n}$
        existe para todos $ n $. Interprete o termo principal: um aleatório
        polinômio mônico de \omterm{def:b3:galois:extension}{grau} $ n $ é irredutível com
        probabilidade $\sim \frac1n $ --- o \omterm{prop:b3:galois:perfect}{perfeito} análogo de
        o teorema dos números primos, com $\log x $ negociado por $ n $;
        verificar numericamente para $ q = 2$, $ n \leq 4$
        (\cref{exo:b3:galois:6} lista as contagens).
  \item Prove o companheiro multiplicativo da questão 21:
        \[
        \prod_{\substack{\pi \text{ monic irred.}\\ \deg\pi =
        n}}\pi \;=\; \prod_{d \mid n}
        \bigl(X^{q^d} - X\bigr)^{\mu(n/d)}
        \]
        \emph{(Mïnversão de obius no grupo abeliano de
        funções racionais diferentes de zero)}; verifique-o manualmente para $ q
        = 2$, $ n = 2$: $(X^4 - X)/(X^2 - X) = X^2 + X + 1$.
\end{enumerate}

\medskip
\noindent\textbf{Part VII --- Two codas.}
\begin{enumerate}[resume]
  \item (O segundo suplemento) O método da Parte V também calcula
        $\bigl(\frac2q\bigr)$. Deixar $\omega = \eu^{2\iu\pi/8}$
        e $ g = \omega + \omega^{-1}$. Mostrar $ g^2 = 2$
        \emph{($\omega^2 = \iu $)}; então, para um primo ímpar $ q $,
        provar em $\Z[\omega]$ módulo $ q $ que
        \[
        g^q \equiv \omega^q + \omega^{-q} \pmod{q\Z[\omega]},
        \]
        e que o lado direito é igual $ g $ se $ q \equiv \pm1
        \pmod 8$ e $-g $ se $ q \equiv \pm3 \pmod 8$. Comparando
        com $ g^q = g\,(g^2)^{(q-1)/2} \equiv
        \bigl(\frac2q\bigr)g $ como na questão 18, conclua
        \[
        \Bigl(\frac2q\Bigr) = (-1)^{(q^2-1)/8},
        \]
        verificando isso $(q^2 - 1)/8$ é exatamente quando $ q
        \equiv \pm1 \pmod 8$. Verificar: $2$ é um mod quadrado $7$
        e mod $17$ ($3^2$ e $6^2$), não mod $3$ nem mod
        $5$.
  \item (A função zeta de $\mathbb F_q[X]$) Prove o
        identidade de séries de potências formais em $ t $:
        \[
        \prod_{n \geq 1}\bigl(1 - t^n\bigr)^{-N_q(n)}
        = \frac1{1 - qt}
        \]
        \emph{(fatoração única em monic irredutíveis:
        expanda cada fator como uma série geométrica e conte
        polinômios mônicos de \omterm{def:b3:galois:extension}{grau} $ n $)}. Recuperar a identidade
        $ q^m = \sum_{d \mid m}d\,N_q(d)$ de
        \cref{exo:b3:galois:6} tomando logaritmos. Verifique o
        coeficiente de $ t^2$ à mão para $ q = 2$ e usar
        fórmula da questão 21 para calcular $ N_2(6) = 9$,
        verificando $2^6 = 1\cdot2 + 2\cdot1 + 3\cdot2 +
        6\cdot9$.
\end{enumerate}
\end{problem}
