\chapter{Espaços de Banach e os Teoremas Fundamentais}
\label{ch:b3:banach}

A análise funcional estuda espaços normados de dimensão infinita
através dos operadores e funcionais que vivem neles. É
descoberta fundadora é que \emph{completude}, através de Baire
teorema, força uma uniformidade forte: famílias pontualmente limitadas de
os operadores são limitados por normas (Banach--Steinhaus), \omterm{def:b3:topology:continuity}{contínuo}
bijeções têm \omterm{def:b3:topology:continuity}{contínuo} inversos (mapeamento aberto) e gráficos
detectar continuidade (gráfico fechado). O outro pilar,
\emph{Hahn--Banach}, não precisa completude de todo --- apenas
Lema de Zorn --- e garante que espaços duplos são ricos o suficiente
para ver cada vetor. Este capítulo prova todos os quatro teoremas e
testa-os nos espaços de sequência clássicos $\ell^p $, em
concreto \omterm{def:b3:banach:operator}{dual} cálculos, e de uma forma genuinamente surpreendente
aplicação: existem \omterm{def:b3:topology:continuity}{contínuo} $2\pi $-funções periódicas
cuja série de Fourier \emph{diverge} em um ponto --- resolvendo em
a negativa é uma questão que o Ano 2 deixou em aberto.

Por todo, $ E, F $ são espaços normatizados acima $ K = \R $ ou $\C $;
\emph{Espaço Banach}\index{Espaço Banach} significa completo normatizado
espaço.

\section{Operadores limitados; espaços de sequência}

\begin{definition}\label{def:b3:banach:operator}
$\mathcal L(E, F)$ denota o espaço de \emph{limitado}
(= \omterm{def:b3:topology:continuity}{contínuo}, Ano 2) mapas lineares com o \emph{operador
norma}\index{operador
norma} $\vertiii T = \sup_{\norm x \leq
1}\norm{Tx}$; é submultiplicativo: $\vertiii{ST} \leq
\vertiii S\,\vertiii T $. O \emph{dual}\index{dual} é
$ E' = \mathcal L(E, K)$.
\end{definition}

\begin{proposition}\label{prop:b3:banach:llcomplete}
Se $ F $ é um espaço de Banach, então é $\mathcal L(E, F)$; em particular $ E'$ é sempre um espaço Banach.
\end{proposition}

\begin{proof}
Deixar $(T_n)$ seja Cauchy para $\vertiii\cdot $. Para cada $ x $,
$\norm{T_nx - T_mx} \leq \vertiii{T_n - T_m}\norm x $: $(T_nx)$
está Cauchy em $ F $, convergente; chame o limite $ Tx $. $ T $ é
linear (limites de identidades lineares); passando ao limite em
$\norm{T_nx - T_mx} \leq \varepsilon\norm x $ ($ n, m \geq N $)
dá $\norm{T_nx - Tx} \leq \varepsilon\norm x $: $ T_n \to T $
em norma do operador, e $\vertiii T \leq \vertiii{T_N} +
\varepsilon < \infty $.
\end{proof}

\begin{definition}\label{def:b3:banach:lp}
Os espaços de sequência clássica\index{sequence spaces $\ell^p $}
(sobre $ K $, indexado por $\N $):
\[
\ell^p = \Bigl\{x = (x_n) : \norm x_p = \Bigl(\sum_n
\abs{x_n}^p\Bigr)^{1/p} < \infty\Bigr\}\quad (1 \leq p <
\infty),
\]
\[
\ell^\infty = \{x : \norm x_\infty = \sup\abs{x_n} < \infty\},
\]
e $ c_0 = \{x : x_n \to 0\}$ com $\norm\cdot_\infty $. Que
$\norm\cdot_p $ é uma norma que segue da desigualdade de Minkowski,
provado no caso discreto exatamente como em \cref{ch:b3:lp} (ou
somando a desigualdade de dimensão finita do Ano 2). Todos
são espaços de Banach, e $ c_0$ é um subespaço fechado de
$\ell^\infty $ (\cref{exo:b3:banach:3}).
\end{definition}

\begin{proposition}[Série Neumann]\label{prop:b3:banach:neumann}
Deixar $ E $ seja Banach e $ T \in \mathcal L(E) = \mathcal L(E,E)$
com $\vertiii T < 1$. Então $ I - T $ é invertível em $\mathcal
L(E)$, com $(I - T)^{-1} = \sum_{n\geq0}T^n $ (convergente em
norma do operador). Consequentemente, o conjunto de operadores invertíveis é
aberta e a inversão é \omterm{def:b3:topology:continuity}{contínuo} nisso.\index{Série Neumann}
\end{proposition}

\begin{proof}
A série converge absolutamente ($\vertiii{T^n} \leq \vertiii
T^n $, geométrico) no Banach $\mathcal L(E)$
(\cref{prop:b3:banach:llcomplete};
\cref{exo:b3:complete:1}(b)). Telescópico, $(I - T)\sum_{n \leq
N}T^n = I - T^{N+1} \to I $, e da mesma forma do outro lado.
Para abertura: se $ S $ é invertível e $\vertiii{H} <
1/\vertiii{S^{-1}}$, então $ S + H = S(I + S^{-1}H)$ com
$\vertiii{S^{-1}H} < 1$: invertível. Continuidade de inversão:
a expressão da série dá $\vertiii{(S+H)^{-1} - S^{-1}} =
O(\vertiii H)$ localmente.
\end{proof}

\begin{example}[Uma equação de Volterra, resolvida por Neumann]
\label{ex:b3:banach:volterraexample}
Sobre $ E = \mathcal C(\intcc01)$, considere a equação integral
\[
u(x) = 1 + \lambda\int_0^xu(t)\,\dd t,
\qquad\text{i.e.}\qquad u = \mathbf 1 + \lambda Tu,
\quad (Tu)(x) = \int_0^xu .
\]
Aqui $\vertiii T \leq 1$, então para $\abs\lambda < 1$ o Neumann
série aplica-se diretamente: $ u = (I - \lambda T)^{-1}\mathbf 1 =
\sum_n\lambda^nT^n\mathbf 1$. Computação, $ T^n\mathbf 1 =
\frac{x^n}{n!}$, então
\[
u(x) = \sum_{n\geq0}\frac{(\lambda x)^n}{n!} =
\eu^{\lambda x} ,
\]
como a diferenciação confirma. Melhorar: $\vertiii{T^n} \leq
\frac1{n!}$ (o kernel iterado diminui fatorialmente), então
$\sum\lambda^nT^n $ converge para \emph{todo} $\lambda $ ---
o operador $ I - \lambda T $ é invertível para todos $\lambda
\in \C $, embora $\vertiii{\lambda T} \geq 1$ eventualmente:
o que importa é a decadência espectral dos poderes, não o
primeira norma. Os operadores de Volterra têm esse decaimento fatorial
construído em (\cref{exo:b3:complete:4}(a) explorado exatamente
isso), e é por isso que os problemas de valor inicial nunca sofrem o
fenômenos de ressonância de problemas de valor de contorno
(\cref{ch:b3:spectral}).
\end{example}

\section{Hahn--Banach}

\begin{theorem}[Hahn - Banach, forma analítica]
\label{thm:b3:banach:hahnbanach}
Deixar $ E $ seja um \emph{real} espaço vetorial, $ p \colon E \to \R $
\emph{sublinhado} ($ p(x + y) \leq p(x) + p(y)$ e $ p(tx) =
tp(x)$ para $ t \geq 0$), $ F \subseteq E $ um subespaço e $ f
\colon F \to \R $ linear com $ f \leq p $ sobre $ F $. Então $ f $ estende
para um linear $\tilde f \colon E \to \R $ com $\tilde f \leq p $
sobre $ E $.\index{Teorema de Hahn-Banach}
\end{theorem}

\begin{proof}
\emph{Extensão de uma etapa.} Deixar $ x_0 \notin F $; nós estendemos $ f $ para
$ F \oplus \R x_0$ escolhendo $\alpha = \tilde f(x_0)$
corretamente: precisamos, para todos $ y \in F $, $ t > 0$,
\[
f(y) + t\alpha \leq p(y + tx_0)
\quad\text{and}\quad
f(y) - t\alpha \leq p(y - tx_0),
\]
que depois de dividir por $ t $ (sublinearidade) reduzir para
\[
\sup_{v \in F}\ \bigl[f(v) - p(v - x_0)\bigr]
\;\leq\; \alpha \;\leq\;
\inf_{u \in F}\ \bigl[p(u + x_0) - f(u)\bigr].
\]
Tal $\alpha $ existe se todo membro à esquerda for $\leq $ cada
membro direito: de fato $ f(v) + f(u) = f(u + v) \leq p(u + v) \leq
p(u + x_0) + p(v - x_0)$, ou seja\ $ f(v) - p(v - x_0) \leq p(u +
x_0) - f(u)$.

\emph{Zorn.} Encomende as extensões de $ f $ dominado por $ p $
(pares: subespaço, funcional) por extensão; uma corrente tem o
união como limite superior; um elemento \omterm{def:b3:rings:primemaximal}{máximo} deve ser definido em todos
de $ E $, caso contrário, a extensão de uma etapa contradiz a maximidade.
\end{proof}

\begin{corollary}\label{cor:b3:banach:hbnormed}
Deixar $ E $ ser um espaço normado ($ K = \R $ ou $\C $).
\begin{enumerate}
  \item Todo $ f \in F'$ ($ F $ um subespaço) se estende a $\tilde f
        \in E'$ com $\norm{\tilde f}_{E'} = \norm f_{F'}$.
  \item Para cada $ x \neq 0$ há $ f \in E'$ com $\norm f
        = 1$ e $ f(x) = \norm x $. Em particular $ E'$ separa
        os pontos de $ E $, e $\norm x = \sup_{\norm f \leq
        1}\abs{f(x)}$.
  \item Para um subespaço fechado $ F $ e $ x \notin F $, há $ f
        \in E'$ desaparecendo em $ F $ com $ f(x) = d(x, F)$ e
        $\norm f \leq 1$.
\end{enumerate}
\end{corollary}

\begin{proof}
(1) Caso real: aplicar \cref{thm:b3:banach:hahnbanach} com $ p(x)
= \norm f_{F'}\,\norm x $ (sublinear); a extensão satisfaz
$\pm\tilde f(x) = \tilde f(\pm x) \leq p(x)$, então
$\norm{\tilde f} \leq \norm f $, e $\geq $ é restrição.
Caso complexo: deixe $ u = \operatorname{Re}f $, um verdadeiro funcional
com $\abs u \leq \norm f\norm\cdot $; observação $ f(x) = u(x) -
\iu\,u(\iu x)$ (verifique as partes reais e imaginárias:
$\operatorname{Im}f(x) = -\operatorname{Re}f(\iu x)$). Estender
$ u $ real-linearmente com o mesmo limite, e defina $\tilde f(x) =
\tilde u(x) - \iu\tilde u(\iu x)$: $\C $-linear (verificação direta em
multiplicação por $\iu $), estende $ f $; norma: para dado $ x $
escrever $\tilde f(x) = r\eu^{\iu\theta}$, então $\abs{\tilde f(x)}
= \tilde f(\eu^{-\iu\theta}x) = \tilde u(\eu^{-\iu\theta}x)
\leq \norm f\,\norm x $.

(2) Ligado $ F = Kx $ definir $ f(tx) = t\norm x $: norma $1$ sobre $ F $;
estender por (1). A fórmula da dualidade: $\leq $ é claro, $\geq $ por
isso $ f $.

(3) Ligado $ F \oplus Kx $ definir $ f(y + tx) = t\,d(x, F)$; então para
$ t \neq 0$, $\norm{y + tx} = \abs t\,\norm{x + y/t} \geq \abs
t\,d(x, F) = \abs{f(y + tx)}$: $\norm f \leq 1$ no
subespaço; estender por (1).
\end{proof}

\begin{remark}\label{rem:b3:banach:bidual}
Por (2), o mapa canônico $ J \colon E \to E''$, $ J(x)(f) =
f(x)$, é uma isometria (\cref{exo:b3:banach:10}): cada norma
o espaço fica dentro de seu bidual. Espaços com $ J $ sobrejetivos são
chamado \emph{reflexivo}\index{reflexivo}; o fim de semana
problema mostra $\ell^p $ ($1 < p < \infty $) é \omterm{rem:b3:banach:bidual}{reflexivo} enquanto
$\ell^1$ não é.
\end{remark}

\section{A trilogia Baire}

\begin{theorem}[Banach-Steinhaus, limites uniformes]
\label{thm:b3:banach:banachsteinhaus}
Deixar $ E $ seja um \emph{Banach} espaço, $ F $ normatizado, e $(T_i)_{i\in
I} \subseteq \mathcal L(E, F)$ uma família com $\sup_i
\norm{T_ix} < \infty $ para cada $ x \in E $. Então $\sup_i
\vertiii{T_i} < \infty $.\index{Teorema de Banach-Steinhaus}
\end{theorem}

\begin{proof}
Os conjuntos $ F_n = \{x : \sup_i\norm{T_ix} \leq n\}$ estão fechados
(interseções de pré-imagens de bolas fechadas) e cobertura $ E $.
Baire (\cref{thm:b3:complete:baire}) dá $ n_0$ e uma bola
$ B(x_0, r) \subseteq F_{n_0}$. Para $\norm z < r $: $\norm{T_iz}
\leq \norm{T_i(x_0 + z)} + \norm{T_ix_0} \leq 2n_0$, então
$\vertiii{T_i} \leq 2n_0/r $ para cada $ i $.
\end{proof}

\begin{corollary}\label{cor:b3:banach:bslimits}
Se $ E $ é Banach e $ T_n \in \mathcal L(E,F)$ convergir
\emph{pontualmente} ($ T_nx \to Tx $ para cada $ x $), então $\sup_n
\vertiii{T_n} < \infty $, $ T \in \mathcal L(E, F)$, e
$\vertiii T \leq \liminf \vertiii{T_n}$.
\end{corollary}

\begin{proof}
As sequências convergentes são limitadas: limitação pontual;
Banach-Steinhaus limita as normas por alguns $ M $; então $\norm{Tx}
= \lim\norm{T_nx} \leq M\norm x $ ($ T $ é linear como um ponto
limite), e o limite mais nítido passando para $\liminf $ em
$\norm{T_nx} \leq \vertiii{T_n}\norm x $.
\end{proof}

\begin{theorem}[Série Divergente de Fourier]
\label{thm:b3:banach:fourierdiverge}
Existem \omterm{def:b3:topology:continuity}{contínuo} $2\pi $-funções periódicas $ f $ cujo
A série de Fourier diverge em $0$: $\sup_N\abs{S_N(f)(0)} =
\infty $. Na verdade tal $ f $ formam um subconjunto \omterm{def:b3:topology:interior}{denso} de $\mathcal
C(S^1)$.\index{Série de Fourier, divergência}
\end{theorem}

\begin{proof}
Trabalhar em $ E = (\mathcal C(S^1), \norm\cdot_\infty)$, um Banach
espaço, com as funcionalidades $\Lambda_N(f) = S_N(f)(0) =
\frac1{2\pi}\int_{-\pi}^{\pi}f(t)\,D_N(t)\,\dd t $ (Dirichlet
kernel, Ano 2). Cada $\Lambda_N $ é \omterm{def:b3:topology:continuity}{contínuo} com
\[
\norm{\Lambda_N} = \frac1{2\pi}\int_{-\pi}^\pi\abs{D_N(t)}\,\dd
t \;=\; L_N .
\]
($\leq $ é claro; $\geq $: pegar $ f $ \omterm{def:b3:topology:continuity}{contínuo}, $\norm f_\infty
\leq 1$, aproximando $\operatorname{sign}D_N $ --- o sinal
tem um número finito de saltos; suavizando cada salto em um intervalo de
comprimento $\varepsilon $ altera a integral por $ O(N\varepsilon)$.)
O \emph{Constantes de Lebesgue} $ L_N $ tendem ao infinito:
\[
L_N = \frac1{2\pi}\int_{-\pi}^{\pi}
\frac{\abs{\sin\bigl((N{+}\tfrac12)t\bigr)}}{\abs{\sin(t/2)}}
\,\dd t
\geq \frac{2}{\pi}\int_0^{\pi}
\frac{\abs{\sin\bigl((N{+}\tfrac12)t\bigr)}}{t}\,\dd t
= \frac{2}{\pi}\int_0^{(N+\frac12)\pi}\frac{\abs{\sin
u}}{u}\,\dd u
\]
(usando $\abs{\sin(t/2)} \leq t/2$ sobre $[0, \pi]$, então
substituindo $ u = (N + \tfrac12)t $). Cortando em arcos:
\[
\int_{(k-1)\pi}^{k\pi}\frac{\abs{\sin u}}u\,\dd u
\geq \frac1{k\pi}\int_{(k-1)\pi}^{k\pi}\abs{\sin u}\,\dd u =
\frac{2}{k\pi},
\qquad\text{so}\qquad
L_N \geq \frac{4}{\pi^2}\sum_{k=1}^N\frac1k
\xrightarrow[N\to\infty]{} \infty .
\]
Se cada \omterm{def:b3:topology:continuity}{contínuo} $ f $ tive $\sup_N
\abs{\Lambda_N(f)} < \infty $, Banach--Steinhaus forçaria
$\sup_N\norm{\Lambda_N} < \infty $: contradição. Então, alguns $ f $
--- na verdade um não \omterm{rem:b3:complete:meagre}{escasso}, conjunto \omterm{def:b3:topology:interior}{denso} de $ f $ (o complemento de
$\bigcup_M\{f: \sup_N\abs{\Lambda_Nf}\leq M\}$, um contável
união de conjuntos fechados que, não tendo \omterm{def:b3:topology:interior}{interior} pelo acima
aplicado em qualquer bola, é \omterm{rem:b3:complete:meagre}{escasso}) --- tem $\sup_N\abs{S_Nf(0)} =
\infty $.
\end{proof}

\begin{theorem}[Mapeamento aberto]\label{thm:b3:banach:openmapping}
Deixar $ E, F $ ser espaços de Banach e $ T \in \mathcal L(E, F)$
\emph{sobrejetivo}. Então $ T $ está aberto: $ T(B_E(0,1)) \supseteq
B_F(0, c)$ para alguns $ c > 0$. Consequentemente, um limite bijetivo
operador entre espaços Banach tem um limite
inverso.\index{teorema do mapeamento aberto}
\end{theorem}

\begin{proof}
Escrever $ B = B_E(0,1)$. A sobrejetividade dá $ F = \bigcup_n
\overline{T(nB)} = \bigcup_n n\,\overline{T(B)}$; Baire
(\cref{thm:b3:complete:baire}) dá \omterm{def:b3:topology:interior}{interior} para
$\overline{T(B)}$: alguns $ B_F(y_0, 4c) \subseteq
\overline{T(B)}$. Centralizar novamente no $0$: para $\norm y < 4c $, ambos
$ y_0 + y $ e $ y_0$ são limites de imagens $ Tu_k $, $ Tv_k $ com
$ u_k, v_k \in B $, então $ y = \lim T(u_k - v_k)$ com $ u_k - v_k
\in 2B $: $ B_F(0, 4c) \subseteq \overline{T(2B)}$, ou seja\
$ B_F(0, 2c) \subseteq \overline{T(B)}$.

\emph{Removendo o \omterm{def:b3:topology:interior}{encerramento}} (aqui completude de $ E $ entra):
deixe $\norm y < c $. Escolha $ x_1 \in \frac12 B $ com $\norm{y -
Tx_1} < c/2$ ($ B_F(0,2c) \subseteq \overline{T(B)}$ dimensionado por
$\frac12$); indutivamente $ x_k \in 2^{-k}B $ com $\norm{y -
T(x_1 + \dots + x_k)} < c\,2^{-k}$. A série $\sum x_k $
converge absolutamente no Banach $ E $, para $ x \in B $ (norma $<
\sum 2^{-k} = 1$), e $ Tx = y $ por continuidade: $ B_F(0, c)
\subseteq T(B)$. Abertura de $ T $ em aberturas arbitrárias segue por
tradução e dimensionamento; como corolário, abertura de $ T $
significa $ T^{-1}$ é \omterm{def:b3:topology:continuity}{contínuo}.
\end{proof}

\begin{corollary}[Normas equivalentes]\label{cor:b3:banach:equivnorms}
Se um espaço vetorial é completo para duas normas comparáveis
($\norm\cdot_a \leq C\norm\cdot_b $), as normas são equivalentes.
\end{corollary}

\begin{proof}
A identidade $(E, \norm\cdot_b) \to (E, \norm\cdot_a)$ é
limitado e bijetivo entre espaços de Banach: seu inverso é
limitado.
\end{proof}

\begin{theorem}[Gráfico fechado]\label{thm:b3:banach:closedgraph}
Deixar $ E, F $ seja Banach e $ T \colon E \to F $ linear. Se o
gráfico $\Gamma = \{(x, Tx)\}$ está fechado em $ E \times F $ (ou seja,\
$ x_n \to x $ e $ Tx_n \to y $ implicar $ y = Tx $), então $ T $ é
limitado.\index{teorema do gráfico fechado}
\end{theorem}

\begin{proof}
$ E \times F $ com $\norm{(x,y)} = \norm x + \norm y $ é Banach;
$\Gamma $, um subespaço fechado, é Banach. A projeção
$\pi_E\colon \Gamma \to E $ é limitado e bijetivo, então é
inverso $ x \mapsto (x, Tx)$ é limitado
(\cref{thm:b3:banach:openmapping}): $\norm{Tx} \leq \norm{(x,
Tx)} \leq C\norm x $.
\end{proof}

\begin{method}\label{met:b3:banach:trilogy}
Quando chegar a qual teorema. \emph{Hahn--Banach}: para
produzir um funcional com comportamento prescrito (normando um
vetor, desaparecendo em um subespaço, estendendo-se de um subespaço) ---
não completude necessário. \emph{Banach-Steinhaus}: para converter
informações pontuais em limites uniformes --- normalmente para mostrar
uma operação limite é \omterm{def:b3:topology:continuity}{contínuo}, ou (contrapositivo) para provar
\emph{divergência} para algum elemento, como para a série de Fourier.
\emph{Mapeamento aberto/gráfico fechado}: obter continuidade de graça
de \omterm{def:b3:galois:algebraic}{algébrico} bijetividade ou de uma \omterm{def:b3:topology:interior}{encerramento} propriedade do
gráfico --- uso típico: comparando dois completo normas, ou provando
automático continuidade. Todos os três teoremas de Baire exigem
completude \emph{da fonte}; contra-exemplos caso contrário
(\cref{exo:b3:banach:7}).
\end{method}

\section{Espaços duplos, concretamente}

\begin{theorem}\label{thm:b3:banach:duals}
Isometricamente: $(c_0)' \cong \ell^1$ e $(\ell^1)' \cong
\ell^\infty $, através do emparelhamento $\langle x, y\rangle = \sum_n
x_ny_n $.\index{duais de espaços de sequência}
\end{theorem}

\begin{proof}
Nós provamos $(c_0)' \cong \ell^1$; a segunda identificação é
\cref{exo:b3:banach:5}. Para $ y \in \ell^1$ associado $\Lambda_y(x)
= \sum x_ny_n $ ($ x \in c_0$): absolutamente convergente, com
$\abs{\Lambda_y(x)} \leq \norm x_\infty\norm y_1$, então
$\norm{\Lambda_y} \leq \norm y_1$. Por outro lado, deixe $\Lambda \in
(c_0)'$; definir $ y_n = \Lambda(e_n)$ ($ e_n $ as sequências unitárias).
Para qualquer $ N $, teste $ x^{(N)} = \sum_{n \leq N}
\operatorname{sign}(\overline{y_n})\,e_n \in c_0$ (norma $\leq
1$; no caso complexo, use fatores unimodulares $\bar y_n/\abs
{y_n}$): $\Lambda(x^{(N)}) = \sum_{n\leq N}\abs{y_n} \leq
\norm\Lambda $. Então $ y \in \ell^1$ com $\norm y_1 \leq
\norm\Lambda $. Finalmente $\Lambda = \Lambda_y $: ambos concordam com o
$ e_n $, portanto, em sequências finitas, densas em $ c_0$ (truncamento:
$\norm{x - \sum_{n \leq N}x_ne_n}_\infty = \sup_{n > N}\abs{x_n}
\to 0$ precisamente porque $ x_n \to 0$); \omterm{def:b3:topology:continuity}{contínuo} funcionais
concordando em um conjunto \omterm{def:b3:topology:interior}{denso} são iguais. A correspondência é linear,
bijetivo e isométrico ($\norm{\Lambda_y} = \norm y_1$ de
as duas desigualdades).
\end{proof}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:b3:banach:1}
Calcule o normas do operador:
(a) as mudanças $ S(x_1, x_2, \dots) = (0, x_1, x_2, \dots)$ e
$ S^*(x_1, x_2, \dots) = (x_2, x_3, \dots)$ sobre $\ell^2$;
(b) o operador de multiplicação $ M_a x = (a_nx_n)$ sobre $\ell^2$,
para $ a \in \ell^\infty $;
(c) o funcional $\Lambda(f) = \int_0^{1/2}f -
\int_{1/2}^1f $ sobre $\mathcal C(\intcc01)$ --- mostrar
$\norm\Lambda = 1$ e que a norma é \emph{não alcançado}.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b3:banach:2}
Deixar $ E $ seja Banach, $ T \in \mathcal L(E)$ invertível e $ S $
com $\vertiii{S - T} < 1/\vertiii{T^{-1}}$. Mostre isso $ S $ é
invertível e estimado $\vertiii{S^{-1} - T^{-1}}$. Aplicação:
se um sistema linear $ Tx = b $ é \omterm{def:b3:groups:derived}{solucionável} com $ T $ invertível, um
perturbação suficientemente pequena de $ T $ mantém-no único
\omterm{def:b3:groups:derived}{solucionável}, com um limite quantitativo na mudança de solução.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:banach:3}
(a) Prove que $\ell^1$, $\ell^\infty $ e $ c_0$ são Banach
espaços, e que $ c_0$ é o \omterm{def:b3:topology:interior}{encerramento} em $\ell^\infty $ do
espaço de sequências finitas.
(b) Mostrar $\ell^p \subseteq \ell^q $ com $\norm\cdot_q \leq
\norm\cdot_p $ para $1 \leq p \leq q \leq \infty $, e que o
a inclusão é estrita.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:banach:4}
Deixar $ F \subseteq E $ ser um subespaço fechado e $ x \notin F $.
Usando \cref{cor:b3:banach:hbnormed}, prove a fórmula da dualidade
\[
d(x, F) = \max\bigl\{\abs{f(x)} : f \in E',\ \norm f \leq 1,\
f\restriction_F = 0\bigr\}
\]
(nota: um \emph{\omterm{def:b3:rings:primemaximal}{máximo}}). Deduza isso $ F = \bigcap\{\ker f : f
\in E',\ f\restriction_F = 0\}$: subespaços fechados são exatamente
as interseções de núcleos de funcionais.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:banach:5}
Provar $(\ell^1)' \cong \ell^\infty $ isometricamente, seguindo o
esquema de \cref{thm:b3:banach:duals} (sequências finitas são densas
em $\ell^1$).
Onde o argumento termina $(\ell^\infty)'$?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:banach:6}
Deixar $ E, F, G $ ser normatizado com $ E $ Banach, e $ B \colon E \times
F \to G $ bilinear, \omterm{def:b3:topology:continuity}{contínuo} em cada variável separadamente. Mostrar
isso $ B $ é (em conjunto) \omterm{def:b3:topology:continuity}{contínuo}: $\norm{B(x,y)} \leq
C\norm x\norm y $. \emph{(Aplique Banach-Steinhaus à família
$(B(\cdot, y))_{\norm y \leq 1}$.)}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:banach:7}
(a) Ligado $ E = \mathcal C(\intcc01)$, comparar $\norm\cdot_\infty $
e $\norm\cdot_1$: a identidade $(E, \norm\cdot_\infty) \to
(E, \norm\cdot_1)$ é limitado e bijetivo, mas seu inverso é
ilimitado. Qual hipótese de
\cref{cor:b3:banach:equivnorms} falha?
(b) Mostre um mapa linear descontínuo de um subespaço \omterm{def:b3:topology:interior}{denso} de
$\ell^2$ para $ K $ (por exemplo,\ em sequências finitas) e explique por que
isso não contradiz o teorema do gráfico fechado.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b3:banach:8}
(Hellinger--Toeplitz) Deixe $ T \colon \ell^2 \to \ell^2$ ser linear
(em todos os lugares definidos) e \emph{simétrico}: $\langle Tx,
y\rangle = \langle x, Ty\rangle $ para todos $ x, y $, onde $\langle
x, y \rangle = \sum x_n\bar y_n $. Mostre isso $ T $ é limitado.
\emph{(Gráfico fechado: se $ x_k \to x $ e $ Tx_k \to z $, teste
contra a arbitrariedade $ y $.)} Moral: operadores simétricos ilimitados
--- os hamiltonianos da mecânica quântica --- nunca podem ser
definido em todo o espaço.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b3:banach:9}
(Teorema de Polya sobre quadratura) Para cada $ n $, deixar $\Lambda_n(f)
= \sum_{i=0}^{n} w_{i,n}f(x_{i,n})$ ser uma regra de quadratura em
$\mathcal C(\intcc01)$ ($ x_{i,n} \in \intcc01$, $ w_{i,n} \in
\R $). Mostre isso $\Lambda_n(f) \to \int_0^1f $ para \emph{todo}
\omterm{def:b3:topology:continuity}{contínuo} $ f $ se e somente se: (i) $\Lambda_n(P) \to \int_0^1P $
para cada polinômio $ P $ e (ii) $\sup_n\sum_i\abs{w_{i,n}} <
\infty $. \emph{(Calcular $\norm{\Lambda_n}$; usar
Banach  Steinhaus e Weierstrass.)} Verifique essas regras com
\emph{positivo} pesos exatos em constantes satisfazem (ii)
automaticamente.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:banach:10}
Mostre isso $ J \colon E \to E''$, $ J(x)(f) = f(x)$, é uma linear
isometria (use \cref{cor:b3:banach:hbnormed}(2)), e que é
sobrejetivo quando $\dim E < \infty $. Mostre também que se $ E'$ é
\omterm{def:b3:galois:separable}{separável} então também é $ E $. \emph{(Escolha $ x_n $ quase normalizando um
sequência densa de $ E'$ e mostre que seu vão fechado é $ E $, através
\cref{cor:b3:banach:hbnormed}(3).)}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:banach:11}
(Espaços quocientes) Deixe $ E $ ser um espaço Banach e $ F \subseteq
E $ um \emph{fechado} subespaço. Sobre $ E/F $ definir
\[
\norm{\bar x} \;=\; d(x, F) = \inf_{y\in F}\norm{x - y} .
\]
(a) Mostre que este é um bem definido \emph{norma} sobre $ E/F $ (onde
o fechamento de $ F $ entra?) e que a projeção $\pi
\colon E \to E/F $ tem $\vertiii\pi \leq 1$ e mapeia o espaço aberto
esfera unitária na esfera unitária aberta.
(b) Mostre que $ E/F $ é completo. \emph{(Use a série
critério de \cref{exo:b3:complete:1}(b): aulas dadas
$\bar x_k $ com $\sum\norm{\bar x_k} < \infty $, levante cada um para
$ x_k \in E $ com $\norm{x_k} \leq \norm{\bar x_k} + 2^{-k}$
e soma em $ E $.)}
(c) Calcular: para $ E = c $ (sequências convergentes) e $ F =
c_0$, mostrar $ c/c_0 \cong K $ isometricamente via $\bar x \mapsto
\lim_nx_n $.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:banach:12}
(Projeções limitadas e subespaços complementados) Deixe $ E $ seja um
Espaço Banach e $ P \colon E \to E $ linear com $ P^2 = P $
(um \omterm{def:b3:galois:algebraic}{algébrico} projeção), $ V = \operatorname{im}P $, $ W =
\ker P $.
(a) Suponha $ P $ é limitado. Mostre isso $ V $ e $ W $ estão fechados
e $ E = V \oplus W $ com a decomposição $ x = Px + (x -
Px)$.
(b) Por outro lado, suponha $ E = V \oplus W $ com \emph{ambos} $ V,
W $ fechado e deixe $ P $ seja a projeção em $ V $ junto $ W $.
Mostre isso $ P $ é limitado. \emph{(Gráfico fechado: se $ x_n \to x $
e $ Px_n \to z $, então $ z \in V $, $ x_n - Px_n \to x - z \in
W $, e a singularidade da decomposição identifica $ z =
Px $.)}
(c) Deduza o \emph{equivalência}: um subespaço $ V $ admite um
projeção limitada se for fechado e tiver um fechado
\omterm{def:b3:galois:algebraic}{algébrico} complemento --- e observe (sem prova) que
existem subespaços fechados sem esta propriedade ($ c_0$ dentro
$\ell^\infty $ é o exemplo clássico): espaços de Hilbert,
onde $ V^\perp $ sempre funciona
(\cref{ch:b3:hilbert}), são a exceção, não a regra.
\end{exercise}

\section{Problema: a dualidade do \texorpdfstring{$\ell^p $}%
{lp} espaços}

\begin{problem}[{Problema de fim de semana --- $(\ell^p)' = \ell^q $,
reflexividade e a estranheza de $\ell^\infty $}]
\label{pb:b3:banach:1}
Consertar $1 < p < \infty $ e deixe $ q $ seja o expoente conjugado,
$\frac1p + \frac1q = 1$. O emparelhamento é $\langle x,
y\rangle = \sum_n x_ny_n $.

\medskip
\noindent\textbf{Part I --- Hölder and Minkowski for
sequences.}
\begin{enumerate}
  \item (Desigualdade de Young) Para $ a, b \geq 0$ mostrar $ ab \leq
        \frac{a^p}p + \frac{b^q}q $, usando a concavidade de $\log $ ou
        estudando $ t \mapsto \frac{t^p}p + \frac1q - t $.
  \item (Hölder) Deduza: $\abs{\langle x, y\rangle} \leq
        \norm x_p\norm y_q $ para $ x \in \ell^p $, $ y \in \ell^q $;
        identificar o caso de igualdade.
  \item (Minkowski) Deduza a desigualdade triangular para
        $\norm\cdot_p $. \emph{(Escrever $\abs{x_n + y_n}^p \leq
        \abs{x_n}\,\abs{x_n{+}y_n}^{p-1} +
        \abs{y_n}\,\abs{x_n{+}y_n}^{p-1}$ e aplique Hölder para
        cada termo.)}
  \item Prove isso $\ell^p $ é completo e que o finito
        as sequências são densas nele.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The duality $(\ell^p)' =
\ell^q $.}
\begin{enumerate}[resume]
  \item Para $ y \in \ell^q $, mostre que $\Lambda_y(x) = \langle
        x, y\rangle $ define $\Lambda_y \in (\ell^p)'$ com
        $\norm{\Lambda_y} \leq \norm y_q $ e, testando em
        $ x_n = \abs{y_n}^{q-1}\operatorname{sign}(y_n)$
        (adequadamente truncado e normalizado), que
        $\norm{\Lambda_y} = \norm y_q $.
  \item Por outro lado, dado $\Lambda \in (\ell^p)'$, definir $ y_n =
        \Lambda(e_n)$; mostrar $ y \in \ell^q $ com $\norm y_q \leq
        \norm\Lambda $ \emph{(teste em truncamentos como na questão
        5 e deixe o comprimento do truncamento aumentar)}, e concluir
        $\Lambda = \Lambda_y $: o mapa $ y \mapsto \Lambda_y $ é
        um isomorfismo isométrico $\ell^q \to (\ell^p)'$.
  \item Deduza isso $\ell^p $ é \emph{\omterm{rem:b3:banach:bidual}{reflexivo}} para $1 < p <
        \infty $: compondo as duas dualidades, cada elemento de
        $(\ell^p)''$ vem de $\ell^p $; verifique cuidadosamente que
        o composto é o canônico $ J $.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- $\ell^1$ and $\ell^\infty $ are
different animals.}
\begin{enumerate}[resume]
  \item Mostre isso $\ell^p $ ($1 \leq p < \infty $) e $ c_0$ são
        \omterm{def:b3:galois:separable}{separável}, mas $\ell^\infty $ não é. \emph{(O
        incontáveis sequências de indicadores de subconjuntos de $\N $
        estão em pares à distância $1$.)}
  \item Deduzir de \cref{exo:b3:banach:10} que $(\ell^1)'
        \cong \ell^\infty $ mas $(\ell^\infty)' \not\cong
        \ell^1$: $\ell^1$ é \emph{não} \omterm{rem:b3:banach:bidual}{reflexivo}.
        \emph{(Se $(\ell^\infty)'$ eram $\ell^1$, seria
        \omterm{def:b3:galois:separable}{separável}, forçando $\ell^\infty $ \omterm{def:b3:galois:separable}{separável}.)}
  \item (Um limite de Banach, explicitamente) Em $\ell^\infty_\R $, deixar
        $ p(x) = \limsup_n \frac{x_1 + \dots + x_n}{n}$. Mostrar
        isso $ p $ é sublinear, e isso no subespaço $ c $ de
        sequências convergentes, $\mathrm{LIM}(x) = \lim x $
        satisfaz $\mathrm{LIM} \leq p $. Estender por Hahn--Banach
        para $\mathrm{LIM} \colon \ell^\infty_\R \to \R $ e mostrar:
        $\mathrm{LIM}$ é positivo ($ x \geq 0 \Rightarrow
        \mathrm{LIM}(x) \geq 0$), invariante ao deslocamento
        ($\mathrm{LIM}(x_2, x_3, \dots) = \mathrm{LIM}(x)$),
        estende o limite e satisfaz $\liminf x \leq
        \mathrm{LIM}(x)\leq \limsup x $.
  \item Mostre que tal $\mathrm{LIM}$, visto em
        $(\ell^\infty)'$, é \emph{não} da forma $\Lambda_y $
        para qualquer $ y \in \ell^1$; concluir novamente $(\ell^\infty)'
        \neq \ell^1$. \emph{(Avalie nas sequências de unidades
        $ e_n $, então na sequência constante $1$.)}
  \item Avaliar $\mathrm{LIM}$ sobre $(0,1,0,1,\dots)$ e mostrar
        que não há mudança invariante \emph{multiplicativo} extensão
        do limite pode existir \emph{(considerar $ x =
        (0,1,0,1,\dots)$ e $ x\cdot Sx $ onde $ S $ é o
        mudança)}.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Epilogue: why reflexivity
matters.}
\begin{enumerate}[resume]
  \item Usando \cref{cor:b3:banach:bslimits} e a questão 6,
        Mostre que toda sequência limitada de $\ell^p $ ($1 < p <
        \infty $) tem uma subsequência $(x^{(k)})$ que converge
        \emph{fracamente}: $\Lambda(x^{(k)})$ converge para cada
        $\Lambda \in (\ell^p)'$. \emph{(Extração diagonal em
        as muitas coordenadas contáveis; identificar o limite fraco
        em $\ell^p $ usando limites uniformes de normas e
        Hölder.)} Mostrar por exemplo ($ e_n $ em $\ell^1$, contra
        elementos bem escolhidos $\ell^\infty $) que isso falha
        em $\ell^1$: a compactação fraca é um privilégio de
        espaços \omterm{rem:b3:banach:bidual}{reflexivos}.
\end{enumerate}

\medskip
\noindent\textbf{Part V --- The weak \omterm{def:b3:topology:topology}{topologia} at work, and
Schur's surprise.} Escrever $ x^{(k)} \rightharpoonup x $ em um
espaço normado $ E $ (\emph{convergência fraca}) quando
$\Lambda(x^{(k)}) \to \Lambda(x)$ para cada $\Lambda \in E'$.
\begin{enumerate}[resume]
  \item Completo o censo: mostrar $(c_0)' \cong \ell^1$
        isometricamente, pelo esquema das questões 5 a 6 (o que
        substitui as sequências de teste?). Monte a corrente
        $ c_0 \to \ell^1 \to \ell^\infty \to \dots $ de
        sucessivo duplas e marque onde a reflexividade falha.
  \item Mostre que toda sequência fracamente convergente de um Banach
        o espaço é limitado: veja o $ x^{(k)}$ através do
        incorporação canônica $ J $ como funcionais em $ E'$ e
        aplicar Banach-Steinhaus
        (\cref{thm:b3:banach:banachsteinhaus}) --- em que
        Espaço Banach e por que é completude disponível aí?
  \item Mostre isso em $\ell^p $, $1 < p < \infty $: $ x^{(k)}
        \rightharpoonup x $ se $\sup_k\norm{x^{(k)}}_p <
        \infty $ e $ x^{(k)}_n \to x_n $ para cada coordenada
        $ n $ \emph{(uma direção usa Banach--Steinhaus através
        a incorporação canônica; para o outro, aproximado $ y
        \in \ell^q $ por sequências finitas)}. Deduzir $ e_k
        \rightharpoonup 0$ em $\ell^2$ enquanto $\norm{e_k}_2 =
        1$: limites fracos podem perder massa.
  \item Mostre que a norma é semicontínua fracamente inferior:
        $ x^{(k)} \rightharpoonup x $ implica $\norm x \leq
        \liminf_k\,\norm{x^{(k)}}$ \emph{(escolha uma norma
        funcional para $ x $, \cref{cor:b3:banach:hbnormed})}.
  \item (Radon--Riesz em $\ell^2$) Mostre isso em $\ell^2$, fraco
        convergência junto com convergência de normas implica
        convergência normativa \emph{(expandir $\norm{x^{(k)} -
        x}_2^2$)}. Dê um contra-exemplo para a mesma afirmação
        sem a hipótese da norma.
  \item (Schur, etapa 1) Deixe $ x^{(k)} \rightharpoonup 0$ em
        $\ell^1$ e suponha, por contradição,
        $\norm{x^{(k)}}_1 \geq \delta > 0$ ao longo de um
        subsequência. Mostre primeiro que $ x^{(k)}_n \to 0$ para
        cada $ n $ (quais funcionais?), então construa
        índices recursivamente $ k_1 < k_2 < \cdots $ e inteiros
        $0 = N_0 < N_1 < N_2 < \cdots $ tal que a massa de
        $ x^{(k_j)}$ concentra-se no bloco $ B_j =
        \intoc{N_{j-1}}{N_j}$:
        \[
        \sum_{n \in B_j}\bigl|x^{(k_j)}_n\bigr| \geq
        \norm{x^{(k_j)}}_1 - \frac\delta{10} .
        \]
  \item (Schur, etapa 2) Definir $ y \in \ell^\infty $ por $ y_n =
        \operatorname{sign}\bigl(x^{(k_j)}_n\bigr)$ para $ n \in
        B_j $. Mostrar $\langle x^{(k_j)}, y\rangle \geq
        \norm{x^{(k_j)}}_1 - \frac{2\delta}{10} \geq
        \frac{8\delta}{10}$ e derivar uma contradição com
        $ x^{(k)} \rightharpoonup 0$. Concluir \emph{Schur
        teorema}: em $\ell^1$, sequências fracamente convergentes
        convergem na norma.
  \item Deduza isso $(e_k)$ não tem fracamente convergente
        subsequência em $\ell^1$ (seu único limite de candidatos é
        $0$, coordenadas --- então aplique Schur), recuperando
        a falha da questão 13 em termos de compacidade fraca; e resolver
        o aparente paradoxo: em $\ell^1$ fraco e \omterm{def:b3:groups:normal}{normal}
        convergência de \emph{sequências} coincidem, mas o
        fraco e \omterm{def:b3:groups:normal}{normal} \emph{\omterm{def:b3:topology:topology}{topologias}} diferem e limitados
        conjuntos ainda não conseguem ser sequencialmente fracos compactar ---
        nenhuma contradição, apenas o fracasso da reflexividade.
  \item (Tabela síntese) Para $ E \in \{c_0,\ \ell^1,\ \ell^p\
        (1{<}p{<}\infty),\ \ell^\infty\}$, tabular: o \omterm{def:b3:banach:operator}{dual};
        separabilidade; reflexividade; se sequências limitadas
        admitir subsequências fracamente convergentes; e um
        propriedade de assinatura de cada espaço, justificada em um
        linha deste problema.
\end{enumerate}

\medskip
\noindent\textbf{Part VI --- Complements: nearest points,
averaged convergence, the value of a Banach limit.}
\begin{enumerate}[resume]
  \item (Pontos mais próximos: um dividendo de reflexividade) Deixe $ F $ ser
        um subespaço fechado de $\ell^p $ ($1 < p < \infty $) e
        $ x \in \ell^p $. Mostre isso $ d = \operatorname{dist}(x,
        F)$ é \emph{alcançado}: extrair de uma minimização
        sequência uma subsequência fracamente convergente (questão
        13), mantenha o limite fraco dentro $ F $ construindo, através
        Hahn--Banach, um desaparecimento funcional em $ F $ mas não em
        um ponto fora dele e conclua com a questão 17.
        Então mostre que o privilégio não é universal: em $ c_0$,
        para $\Lambda(x) = \sum_n2^{-n}x_n $, provar
        $\norm\Lambda = 1$ não é alcançado na bola unitária,
        estabeleça a fórmula da distância
        $\operatorname{dist}(x, \ker\Lambda) =
        \abs{\Lambda(x)}$, e deduza que não $ x \notin
        \ker\Lambda $ tem um ponto mais próximo no fechado
        hiperplano $\ker\Lambda $.
  \item (Banach--Saks em $\ell^2$) Deixar $ x^{(k)}
        \rightharpoonup 0$ em $\ell^2_{\R}$ com
        $\norm{x^{(k)}}_2 \leq C $. Construa uma subsequência
        $(y_j)$ com $\abs{\langle y_i, y_j\rangle} \leq
        \frac1j $ para todos $ i < j $, e deduza
        \[
        \Bigl\lVert\frac{y_1 + \dots + y_m}m\Bigr\rVert_2^2
        \leq \frac{C^2 + 2}m \longrightarrow 0 :
        \]
        após a extração, o Cesàro significa convergir em
        \emph{norma}. Verifique $(e_k)$, cujas médias têm norma
        $\frac1{\sqrt m}$: convergência fraca, inútil para o
        a própria sequência (questão 16), torna-se norma
        convergência para médias.
  \item (O valor de um limite de Banach) Deixe $ L $ seja qualquer Banach
        limite (questão 10) e $ A_mx = \frac1m(x + Sx + \dots
        + S^{m-1}x)$. Mostrar $ L(A_mx) = L(x)$ e $\liminf A_mx
        \leq L(x) \leq \limsup A_mx $; deduzir isso \emph{todos}
        Os limites de Banach concordam com sequências periódicas, com valor
        a média durante um período --- $\frac13$ sobre $(1, 0, 0, 1,
        0, 0, \dots)$, consistente com a pergunta 12
        $\frac12$. Então mostre que o acordo falha em geral: por
        a sequência de blocos $ x $ igual a $1$ sobre
        $\intoc{3^{j-1}}{3^j}$ até mesmo $ j $ e $0$ em outro lugar,
        mostre que o Cesàro significa oscilar entre $\leq
        \frac13$ e $\geq \frac23$, e construir dois Banach
