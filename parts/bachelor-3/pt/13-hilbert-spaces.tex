\chapter{Espaços Hilbert}\label{ch:b3:hilbert}

UM \omterm{def:b3:hilbert:inner}{Espaço Hilbert} é um espaço de Banach cuja norma vem de um
\omterm{def:b3:hilbert:inner}{produto interno} --- e essa única estrutura extra restaura, em
dimensão infinita, quase toda a geometria euclidiana:
existem projeções ortogonais, cada \omterm{def:b3:topology:continuity}{contínuo} funcional é um
\omterm{def:b3:hilbert:inner}{produto interno} contra um vetor fixo (Riesz) e ortonormal
\omterm{def:b3:topology:basis}{bases} expandem cada vetor em uma série convergente com
Escrituração Pitagórica (\omterm{thm:b3:hilbert:parseval}{Parseval}). O clímax do capítulo é um
dívida honrada: o sistema trigonométrico é uma \omterm{def:b3:topology:basis}{base} ortonormal
de $L^2$, então Parsevala identidade de vale para \emph{todo}
quadrado \omterm{def:b3:lebesgue:l1}{integrável} função --- a declaração Ano 2 só poderia
provar por partes $\mathcal C^1$ funções. Terminamos com
Lax--Milgram, o lema burro de carga da abordagem variacional
para equações diferenciais.

Por todo, $ H $ é um espaço vetorial sobre $ K = \R $ ou $\C $.

\section{Produtos internos; o teorema da projeção}

\begin{definition}\label{def:b3:hilbert:inner}
Um \emph{produto interno}\index{produto interno} é um mapa $\langle
\cdot,\cdot\rangle \colon H\times H \to K $, linear no
segunda variável, com $\langle y, x\rangle =
\overline{\langle x, y\rangle}$ e $\langle x, x\rangle > 0$
para $ x \neq 0$. Induz a norma $\norm x = \langle x,
x\rangle^{1/2}$, o \emph{Desigualdade de Cauchy-Schwarz}
$\abs{\langle x, y\rangle} \leq \norm x\norm y $ (Ano 2
prova --- o discriminante --- permanece inalterado), e o
\emph{lei do paralelogramo}
\[
\norm{x + y}^2 + \norm{x - y}^2 = 2\norm x^2 + 2\norm y^2 .
\]
UM \emph{Espaço Hilbert}\index{Espaço Hilbert} é um
espaço do produto interno completo para esta norma. Exemplos: $\ell^2$
(\cref{pb:b3:banach:1}) e, o fundamental, $ L^2(\mu)$
com $\langle f, g\rangle = \int\bar fg\,\dd\mu $ ---
completo por Riesz--Fischer (\cref{thm:b3:lp:complete}); o
o produto interno é finito por Cauchy--Schwarz ($= $ Hölder em
$ p = q = 2$).
\end{definition}

\begin{theorem}[Projeção em um conjunto convexo fechado]
\label{thm:b3:hilbert:projection}
Deixar $ C \neq \varnothing $ ser um fechado \emph{convexo} subconjunto de
o \omterm{def:b3:hilbert:inner}{Espaço Hilbert} $ H $ e $ x \in H $. Existe um único $ p_C(x)
\in C $ com\index{teorema da projeção}
\[
\norm{x - p_C(x)} = d(x, C),
\]
caracterizado por: $\operatorname{Re}\langle x - p_C(x),\ c -
p_C(x)\rangle \leq 0$ para todos $ c \in C $. O mapa $ p_C $ é
$1$-Lipschitz.
\end{theorem}

\begin{proof}
Deixar $ d = d(x, C)$ e $(c_n) \subseteq C $ com $\norm{x - c_n}
\to d $. Paralelogramo ativado $ x - c_n $ e $ x - c_m $:
\[
\norm{c_n - c_m}^2 = 2\norm{x - c_n}^2 + 2\norm{x - c_m}^2 -
4\,\bigl\|x - \tfrac{c_n + c_m}2\bigr\|^2
\leq 2\norm{x{-}c_n}^2 + 2\norm{x{-}c_m}^2 - 4d^2
\]
(a convexidade coloca o ponto médio em $ C $): o lado direito tende a
$0$, então $(c_n)$ é Cauchy e seu limite $ p \in C $ (fechado)
atinge $ d $. Singularidade: dois minimizadores dão, pelo mesmo
identidade, $\norm{p - p'}^2 \leq 2d^2 + 2d^2 - 4d^2 = 0$.

Caracterização: para $ c \in C $, $ t \in \intoc01$, o vetor
$ p + t(c - p) \in C $, então
\[
d^2 \leq \norm{x - p - t(c-p)}^2
= d^2 - 2t\operatorname{Re}\langle x - p, c - p\rangle +
t^2\norm{c-p}^2 ;
\]
dividir por $ t \to 0^+$: $\operatorname{Re}\langle x - p, c -
p\rangle \leq 0$. Por outro lado, esta desigualdade dá $\norm{x -
c}^2 = \norm{x - p}^2 - 2\operatorname{Re}\langle x - p, c -
p\rangle + \norm{p - c}^2 \geq \norm{x-p}^2$. Lipschitz: para
$ x, y $ com projeções $ p, q $, adicione as duas variações
desigualdades (com $ c = q $, resp.\ $ c = p $):
$\operatorname{Re}\langle x - y - (p - q), p - q\rangle \geq
0$, então $\norm{p - q}^2 \leq \operatorname{Re}\langle x - y, p -
q\rangle \leq \norm{x - y}\norm{p - q}$.
\end{proof}

\begin{theorem}[Decomposição ortogonal]
\label{thm:b3:hilbert:decomposition}
Deixar $ F $ seja um \emph{subespaço fechado} de $ H $. Então $ p_F $ é
linear, $ x - p_F(x) \perp F $ para todos $ x $, e\index{ortogonal
complemento}
\[
H = F \oplus F^\perp,
\qquad F^\perp = \{y : \langle y, f\rangle = 0\ \forall f\in
F\},
\qquad (F^\perp)^\perp = F .
\]
Para um subespaço geral, $(F^\perp)^\perp = \bar F $; em particular $ F $ é \omterm{def:b3:topology:interior}{denso} se $ F^\perp = \{0\}$.
\end{theorem}

\begin{proof}
Para um subespaço, a caracterização variacional com $ c =
p_F(x) \pm f $ ($ f \in F $, ambos os sinais, e $\iu f $ no
caso complexo) forças $\langle x - p_F(x), f\rangle = 0$: o
residual é ortogonal a $ F $. Decomposição $ x = p_F(x) + (x
- p_F(x))$ com $ F \cap F^\perp = \{0\}$ ($\langle y, y
\rangle = 0$); linearidade de $ p_F $ decorre da singularidade de
tais decomposições (ambos os lados lineares nelas). $(F^\perp)
^\perp \supseteq F $ sempre; inversamente se $ x \perp F^\perp $,
escrever $ x = f + g $: $ g = x - f \in F^\perp $ e $\langle g,
g\rangle = \langle x, g\rangle - \langle f, g\rangle = 0$: $ x
= f \in F $. Para um subespaço geral $ F $: $ F^\perp = \bar
F^{\,\perp}$ (continuidade do \omterm{def:b3:hilbert:inner}{produto interno}), então
$(F^\perp)^\perp = \bar F $ pelo caso encerrado; \omterm{ex:b3:lebesgue:gamma}{densidade} se
$\bar F = H $ se $ F^\perp = 0$.
\end{proof}

\begin{theorem}[Representação Riesz]\label{thm:b3:hilbert:riesz}
Para cada \omterm{def:b3:topology:continuity}{contínuo} funcional linear $\varphi \in H'$ lá
é um único $ a \in H $ com\index{Teorema da representação de Riesz}
\[
\varphi(x) = \langle a, x\rangle \quad (x \in H),
\qquad \norm\varphi_{H'} = \norm a .
\]
\end{theorem}

\begin{proof}
Se $\varphi = 0$: $ a = 0$. De outra forma $ F = \ker\varphi $ é um
subespaço próprio fechado; escolha $ u \in F^\perp $, $\norm u = 1$
(\cref{thm:b3:hilbert:decomposition}: $ F^\perp \neq 0$ desde
$ F \neq H $). Para qualquer $ x $, o vetor $\varphi(x)u -
\varphi(u)x \in \ker\varphi $, por isso $\perp u $:
\[
0 = \langle u, \varphi(x)u - \varphi(u)x\rangle
= \varphi(x) - \varphi(u)\langle u, x\rangle :
\qquad \varphi(x) = \langle
\overline{\varphi(u)}\,u,\ x\rangle .
\]
Então $ a = \overline{\varphi(u)}u $ funciona. Singularidade: $\langle a
- a', x\rangle = 0$ para todos $ x $, teste $ x = a - a'$. Normas:
$\abs{\varphi(x)} \leq \norm a\norm x $ (Cauchy-Schwarz) com
igualdade em $ x = a $.
\end{proof}

\begin{example}[Uma projeção, calculada até o final]
\label{ex:b3:hilbert:projectionexample}
Em $ H = L^2(\intcc01)$, qual é a melhor aproximação de
$ f(x) = x^2$ por uma função afim? O subespaço $ F =
\operatorname{Vect}(1, x)$ é fechado (dimensional finito),
e $ p_F(f) = a + bx $ é caracterizado pela ortogonalidade de
o resíduo para $1$ e para $ x $:
\[
\int_0^1(x^2 - a - bx)\,\dd x = 0,
\qquad
\int_0^1x\,(x^2 - a - bx)\,\dd x = 0,
\]
ou seja\ $\frac13 = a + \frac b2$ e $\frac14 = \frac a2 +
\frac b3$: $ a = -\frac16$, $ b = 1$. Então $ p_F(x^2) = x -
\frac16$, e o erro é
\[
d(f, F)^2 = \int_0^1\Bigl(x^2 - x + \frac16\Bigr)^2\dd x =
\frac1{180},
\qquad d(f, F) = \frac1{6\sqrt5} .
\]
Duas observações que valem a pena internalizar. Primeiro, o cálculo é
nada além de um $2\times2$ sistema linear --- o \emph{normais
equações}; para a \omterm{def:b3:topology:basis}{base} monomial sua matriz
$\bigl(\frac1{i+j+1}\bigr)$ é o notoriamente
matriz de Hilbert mal condicionada e ortogonalizando primeiro
(polinômios de legenda, \cref{pb:b3:hilbert:1}) é a cura.
Em segundo lugar, o melhor \emph{uniforme} aproximação de $ x^2$ por
funções afins são diferentes ($ x - \frac18$, por
equioscilação): cada norma tem sua própria geometria, e somente
o hilbertiano responde com um sistema linear.
\end{example}

\section{Bases ortonormais}

\begin{definition}\label{def:b3:hilbert:onb}
Uma família $(e_i)_{i\in I}$ é \emph{ortonormal} se $\langle
e_i, e_j\rangle = \delta_{ij}$, e um \emph{Hilbert
básico}\index{Hilbert
básico} (\omterm{def:b3:topology:basis}{base} ortonormal) se além disso
suas combinações lineares finitas são densas em $ H $ (a família é
\emph{total}). Tratamos o caso contável $ I = \N $, que por
Gram-Schmidt cobre todos \emph{\omterm{def:b3:galois:separable}{separável}} $ H $
(\cref{prop:b3:hilbert:gramschmidt}).
\end{definition}

\begin{theorem}[Bessel, Parseval]\label{thm:b3:hilbert:parseval}
Deixar $(e_n)_{n\in\N}$ ser ortonormal em $ H $, e $ c_n(x) =
\langle e_n, x\rangle $.
\begin{enumerate}
  \item (Bessel) $\sum_n\abs{c_n(x)}^2 \leq \norm x^2$ e
        a série $\sum_nc_n(x)e_n $ converge em $ H $, com
        soma $ p_F(x)$, $ F = \overline{\operatorname{Vect}}(e_n)$.
  \item Os seguintes são equivalentes: (i) $(e_n)$ é um Hilbert
        \omterm{def:b3:topology:basis}{base}; (ii) $ x = \sum_nc_n(x)e_n $ para cada $ x $; (iii)
        \emph{Parseval}\index{Parseval}: $\norm x^2
        = \sum_n\abs{c_n(x)}^2$ para cada $ x $; (iv) o único
        vetor ortogonal a todos $ e_n $ é $0$.
  \item Se $(e_n)$ é um \omterm{def:b3:topology:basis}{Base} de Hilbert, $ x \mapsto (c_n(x))_n $
        é um isomorfismo isométrico $ H \to \ell^2$
        (\emph{todo} \omterm{def:b3:galois:separable}{separável} de dimensão infinita Hilbert
        espaço ``é'' $\ell^2$), e $\langle x, y\rangle =
        \sum_n\overline{c_n(x)}c_n(y)$.
\end{enumerate}
\end{theorem}

\begin{proof}
(1) Para finito $ N $: $ x - \sum_{n\leq N}c_ne_n \perp e_k $ ($ k
\leq N $), então Pitágoras dá $\norm x^2 = \sum_{n\leq
N}\abs{c_n}^2 + \norm{x - \sum_{n\leq N}c_ne_n}^2$: Bessel.
As somas parciais $ S_N = \sum_{n\leq N}c_ne_n $ são Cauchy:
$\norm{S_N - S_M}^2 = \sum_{M<n\leq N}\abs{c_n}^2$, cauda de um
série convergente; o limite está em $ F $, e $ x - \lim S_N
\perp $ cada $ e_k $ (continuidade), por isso $\perp F $: por exclusividade
da decomposição ortogonal, $\lim S_N = p_F(x)$.

(2) (eu)$\Rightarrow $(ii): $ F = H $, então $ p_F = \mathrm{id}$.
(ii)$\Rightarrow $(iii): Pitágoras no limite ($\norm{S_N}^2
= \sum_{n \leq N}\abs{c_n}^2 \to \norm x^2$).
(iii)$\Rightarrow $(4): $ x \perp $ todos $ e_n $ dá $\norm x^2 =
0$. (4)$\Rightarrow $(eu): $ F^\perp = \{0\}$
(ortogonalidade para todos $ e_n $ é ortogonalidade a $ F $), então $ F $
é \omterm{def:b3:topology:interior}{denso} por \cref{thm:b3:hilbert:decomposition}; mas $ F $, um
\omterm{def:b3:topology:interior}{encerramento}, já está fechado: $ F = H $.

(3) O mapa é linear, isométrico por (iii) (portanto injetivo),
e sobrejetivo: dado $(c_n) \in \ell^2$, a série $\sum
c_ne_n $ converge (Cauchy como em (1)) para uma pré-imagem. O \omterm{def:b3:topology:interior}{interior}
produto fórmula é a polarização de (iii), ou um limite direto
computação.
\end{proof}

\begin{proposition}[Gram-Schmidt]\label{prop:b3:hilbert:gramschmidt}
Deixar $(x_n)$ ser uma sequência linearmente independente. Configuração
indutivamente $\tilde e_n = x_n - \sum_{k<n}\langle e_k,
x_n\rangle e_k $ e $ e_n = \tilde e_n/\norm{\tilde e_n}$
produz uma ortonormal $(e_n)$ com os mesmos vãos finitos:
$\operatorname{Vect}(e_1, \dots, e_n) = \operatorname{Vect}
(x_1, \dots, x_n)$. Conseqüentemente, todo \omterm{def:b3:galois:separable}{separável} \omterm{def:b3:hilbert:inner}{Espaço Hilbert}
(um com um subconjunto \omterm{def:b3:topology:interior}{denso} contável) tem um Hilbert
\omterm{def:b3:topology:basis}{base}.\index{Ortonormalização de Gram-Schmidt}
\end{proposition}

\begin{proof}
Indução: $\tilde e_n \perp e_k $ ($ k < n $) por construção,
e $\tilde e_n \neq 0$ pela independência; os vãos coincidem em
cada estágio (mudança triangular de \omterm{def:b3:topology:basis}{base}). Para um \omterm{def:b3:galois:separable}{separável} $ H $:
de uma sequência densa extraia um linearmente independente
subfamília com extensão densa (descarte cada vetor na extensão de
seus antecessores - a extensão permanece inalterada), ortonormalizar:
o resultado é total.
\end{proof}

\begin{theorem}[O sistema trigonométrico; Parseval finalmente]
\label{thm:b3:hilbert:fourier}
Em $ L^2(\intcc{-\pi}\pi)$ com $\langle f, g\rangle =
\frac1{2\pi}\int_{-\pi}^\pi \bar fg $, a família $ e_n(t) =
\eu^{\iu nt}$, $ n \in \Z $, é um \omterm{def:b3:topology:basis}{Base} de Hilbert. Conseqüentemente,
para \emph{todo} $ f \in L^2$ --- em particular cada pedaço
\omterm{def:b3:topology:continuity}{contínuo} $2\pi $-periódico $ f $ --- com $ c_n(f) =
\frac1{2\pi}\int_{-\pi}^{\pi}f(t)\eu^{-\iu nt}\dd t $:
\[
f = \sum_{n\in\Z}c_n(f)\,\eu^{\iu nt} \ \ \text{in } L^2,
\qquad
\frac1{2\pi}\int_{-\pi}^{\pi}\abs f^2 =
\sum_{n\in\Z}\abs{c_n(f)}^2 .
\]
Isto prova, em geral, que \omterm{thm:b3:hilbert:parseval}{Parseval} identidade que
Ano 2 admitido.\index{Identidade de Parseval (geral)}
\end{theorem}

\begin{proof}
A ortonormalidade é um cálculo direto (Ano 2). Totalidade: deixe
$ f \in L^2$ ser $\perp $ todos $ e_n $, ou seja,\ todos Fourier
os coeficientes desaparecem. \omterm{def:b3:topology:continuity}{Contínuo} $2\pi $-funções periódicas são
\omterm{def:b3:topology:interior}{denso} em $ L^2(\intcc{-\pi}\pi)$: de fato $\mathcal
C_c(\intoo{-\pi}\pi)$ é \omterm{def:b3:topology:interior}{denso}
(\cref{thm:b3:lp:density}(2)) e tais funções se estendem
periodicamente e continuamente. Polinômios trigonométricos são
$\norm\cdot_\infty $-denso entre \omterm{def:b3:topology:continuity}{contínuo} funções periódicas
(Stone - Weierstrass, \cref{cor:b3:complete:weierstrass}(c)),
e $\norm\cdot_2 \leq \norm\cdot_\infty $: trigonométrico
polinômios são \omterm{def:b3:topology:interior}{densos} em $ L^2$. Mas $ f \perp $ cada
polinômio trigonométrico, portanto $ f \perp $ um subespaço \omterm{def:b3:topology:interior}{denso}: $ f
\in (\text{dense})^\perp = \{0\}$
(\cref{thm:b3:hilbert:decomposition}). Critério (iv) de
\cref{thm:b3:hilbert:parseval} conclui; (ii) e (iii)
descompacte para o display (reindexando o contável $\Z $; o
série dupla converge incondicionalmente --- a parcial
somas sobre qualquer família exaustiva convergem, pelo $\ell^2$ cauda
argumento).
\end{proof}

\begin{theorem}[Lax - Milgram]\label{thm:b3:hilbert:laxmilgram}
Deixar $ H $ seja um verdadeiro \omterm{def:b3:hilbert:inner}{Espaço Hilbert} e $ a \colon H\times H \to
\R $ bilinear, \emph{\omterm{def:b3:topology:continuity}{contínuo}} ($\abs{a(u,v)} \leq M\norm
u\norm v $) e \emph{coercitivo} ($ a(u, u) \geq \alpha\norm u^2$,
$\alpha > 0$). Então para cada $\varphi \in H'$ há um
único $ u \in H $ com\index{Teorema de Lax-Milgram}
\[
a(u, v) = \varphi(v) \qquad \text{for all } v \in H .
\]
\end{theorem}

\begin{proof}
Para fixo $ u $, $ v \mapsto a(u, v)$ é \omterm{def:b3:topology:continuity}{contínuo} linear: Riesz
dá um único $ Au \in H $ com $ a(u,v) = \langle Au,
v\rangle $; $ A $ é linear com $\norm{Au} \leq M\norm u $
(singularidade dos representantes, então vinculados). Coercividade:
$\alpha\norm u^2 \leq a(u,u) = \langle Au, u\rangle \leq
\norm{Au}\norm u $, então $\norm{Au} \geq \alpha\norm u $: $ A $ é
injetivo com alcance fechado (uma sequência de imagens de Cauchy $ Au_n $
forças $ u_n $ Cauchy). O intervalo é \omterm{def:b3:topology:interior}{denso}: $ w \perp
\operatorname{im}A $ dá $0 = \langle Aw, w\rangle \geq
\alpha\norm w^2$. Fechado e \omterm{def:b3:topology:interior}{denso}: $ A $ é bijetivo. Dado
$\varphi $, deixar $ f $ representá-lo (Riesz) e $ u = A^{-1}f $:
$ a(u, v) = \langle f, v\rangle = \varphi(v)$, exclusivamente
($ a(u - u', \cdot) = 0$ e coercividade).
\end{proof}

\begin{remark}\label{rem:b3:hilbert:variational}
Quando $ a $ é simétrico, a solução de Lax--Milgram é a única
minimizador do \emph{energia} $ J(v) = \frac12a(v,v) -
\varphi(v)$ (\cref{exo:b3:hilbert:9}): existência de soluções
para problemas variacionais de uma só vez. Aplicado a adequado
espaços funcionais (os espaços de Sobolev de um curso posterior), este
resolve problemas de valor limite para equações diferenciais ---
o ponto de entrada moderno para equações diferenciais parciais.
\end{remark}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:b3:hilbert:1}
(a) Prove as identidades de polarização (reais: $4\langle x,
y\rangle = \norm{x+y}^2 - \norm{x-y}^2$; complexo: o
versão de quatro termos).
(b) Mostre que $\norm\cdot_1$ sobre $ L^1(\intcc01)$ e
$\norm\cdot_\infty $ sobre $\mathcal C(\intcc01)$ violar o
lei do paralelogramo: essas normas não vêm do nada \omterm{def:b3:hilbert:inner}{produto interno}.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b3:hilbert:2}
Em $ H = L^2(\intcc01)$ (verdadeiro):
(a) calcular a projeção de $ f $ no subespaço de
funções constantes e interpretação;
(b) calcular a projeção em $\{g : g = 0 \text{ a.e.\ on }
\intcc0{1/2}\}$;
(c) calcular $ d\bigl(x \mapsto x,\ \operatorname{Vect}(\mathbf
1)\bigr)$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:hilbert:3}
(a) Mostre que para um subespaço $ F $: $ F $ \omterm{def:b3:topology:interior}{denso} $\iff $ $ F^\perp =
\{0\}$, e dê um exemplo em $\ell^2$ de um \emph{apropriado}
subespaço \omterm{def:b3:topology:interior}{denso} (então $ F^\perp = 0$ sem $ F = H $: o
o teorema da decomposição realmente precisa $ F $ fechado).
(b) Mostre que se $ x_n \to x $ e $ y_n \to y $ em norma, então
$\langle x_n, y_n\rangle \to \langle x, y\rangle $ e localize
dois lugares onde o capítulo usou isso continuidade.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:hilbert:4}
Aplicar Gram-Schmidt a $1, x, x^2$ em $ L^2(\intcc{-1}1)$
(Medida Lebesgue): obtenha os três primeiros normalizados
\emph{Polinômios de Legendre}e verifique se eles correspondem
$\sqrt{n + \frac12}\,P_n $ para os polinômios de Rodrigues $ P_n $
de \cref{pb:b3:hilbert:1}.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:hilbert:5}
Aplicar \omterm{thm:b3:hilbert:parseval}{Parseval} (\cref{thm:b3:hilbert:fourier}) para $ f(t) = t $
e $ f(t) = t^2$ sobre $\intcc{-\pi}\pi $ --- agora legitimamente para
estes (\omterm{def:b3:topology:continuity}{contínuo}, mas anteriormente a identidade precisava
por partes-$\mathcal C^1$ cuidado no envoltório
descontinuidade): recuperar
\[
\sum_{n\geq1}\frac1{n^2} = \frac{\pi^2}6,
\qquad
\sum_{n\geq1}\frac1{n^4} = \frac{\pi^4}{90} .
\]
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:hilbert:6}
(a) Encontrar $ a \in L^2(\intcc01)$ com $\int_0^{1/2}f =
\langle a, f\rangle $ para todos $ f $; calcular $\norm\varphi $ para
isso funcional.
(b) Mostre que a avaliação $ f \mapsto f(\frac12)$, definido
no subespaço $\mathcal C(\intcc01) \subseteq
L^2(\intcc01)$, é \emph{não} \omterm{def:b3:topology:continuity}{contínuo} para
$\norm\cdot_2$: não existe representante Riesz (a avaliação é
não é um $ L^2$ noção).
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b3:hilbert:7}
Deixar $ H $ ser \omterm{def:b3:galois:separable}{separável} com \omterm{def:b3:topology:basis}{Base} de Hilbert $(e_n)$, e $(x_k)$ um
sequência limitada.
(a) Mostre que alguma subsequência converge \emph{fracamente}: há $ x $ com $\langle y, x_{k_j}\rangle \to \langle y,
x\rangle $ para cada $ y \in H $. \emph{(Extração diagonal em
os coeficientes $\langle e_n, x_k\rangle $; montar $ x $ através de
Bessel e a limitação uniforme das normas.)}
(b) Mostrar $ e_n \rightharpoonup 0$ mas $\norm{e_n} = 1$: fraco
limites podem perder a norma. Mostrar $\norm x \leq
\liminf\norm{x_{k_j}}$ em (a).
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:hilbert:8}
(Adjuntos) Para $ T \in \mathcal L(H)$, mostre que existe um único
$ T^* \in \mathcal L(H)$ com $\langle Tx, y\rangle = \langle
x, T^*y\rangle $ (Riesz), e $\vertiii{T^*} = \vertiii T $.
Calcule o adjunto do deslocamento $ S $ sobre $\ell^2$, e provar
$\ker T^* = (\operatorname{im}T)^\perp $ --- deduzir
$\overline{\operatorname{im}T} = (\ker T^*)^\perp $.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:hilbert:9}
Deixar $ a $ seja como em Lax - Milgram e além disso \emph{simétrico}.
Mostre isso $ u $ resolve $ a(u, \cdot) = \varphi $ se $ u $ minimiza
$ J(v) = \frac12a(v, v) - \varphi(v)$, e que o mínimo é
alcançado exatamente em um ponto. \emph{(Completo a praça:
$ J(u + w) - J(u) = \frac12a(w,w) \geq
\frac\alpha2\norm w^2$.)} Aplicação: derivar novamente o
teorema de projeção para subespaços fechados de Lax-Milgram.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b3:hilbert:10}
(O sistema Haar) Ligado $\intcc01$, deixar $ h_{0} = \mathbf 1$ e
para $ n = 2^j + k $ ($ j \geq 0$, $0 \leq k < 2^j $):
\[
h_n = 2^{j/2}\Bigl(\mathbf 1_{[k2^{-j},\,(k +
\frac12)2^{-j})} - \mathbf 1_{[(k+\frac12)2^{-j},\,(k+1)2^{-j})}
\Bigr).
\]
Mostre isso $(h_n)_{n\geq0}$ é ortonormal em $ L^2(\intcc01)$,
e total. \emph{(Ortogonalidade: apoios disjuntos ou aninhados;
totalidade: vãos finitos contêm todas as funções degrau diádicas,
que são \omterm{def:b3:topology:interior}{densos} --- via \cref{thm:b3:lp:density}(1) e diádico
aproximação de intervalos.)} O sistema Haar é o ancestral
de wavelets.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:hilbert:11}
(Projeções ortogonais, caracterizadas) Deixe $ H $ seja um Hilbert
espaço e $ P \in \mathcal L(H)$ com $ P^2 = P $, $ P \neq 0$.
Mostre a equivalência de:
(eu) $ P $ é a projeção ortogonal sobre
$\operatorname{im}P $;
(ii) $ P = P^*$ (\cref{exo:b3:hilbert:8});
(iii) $\vertiii P = 1$.
\emph{(Para (iii) $\Rightarrow $ (i): se algum $ x \in
(\ker P)^\perp $ tive $ Px \neq x $, considere $ x + t(Px - x)$
--- ou diretamente: para $ u \in \operatorname{im}P $ e $ v \in
\ker P $, expandir $\norm{P(u + tv)}^2 \leq \norm{u + tv}^2$
para todos $ t \in \R $ e concluir $\langle u, v\rangle = 0$.)}
Exiba uma projeção não ortogonal em $\R^2$ e calcular
sua norma.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b3:hilbert:12}
(Teorema ergódico de Von Neumann) Deixe $ U \in \mathcal L(H)$ ser
\emph{unitário} ($ U^*U = UU^* = I $), $ F = \ker(U - I)$ o
espaço fixo, $ P $ a projeção ortogonal sobre $ F $, e
$ A_n = \frac1n\sum_{k=0}^{n-1}U^k $.
(a) Mostrar $\ker(U - I) = \ker(U^* - I)$ \emph{(de
$\norm{Ux - x}^2 = 2\norm x^2 - 2\operatorname{Re}\langle
Ux, x\rangle $ e unitariedade)}, e deduza
$\overline{\operatorname{im}(U - I)} = F^\perp $.
(b) Mostre que $ A_nx \to x $ para $ x \in F $, e $ A_nx \to 0$
para $ x \in \operatorname{im}(U - I)$ \emph{(telescópio)},
então para $ x \in \overline{\operatorname{im}(U - I)}$
(uniforme vinculado $\vertiii{A_n} \leq 1$).
(c) Concluir: $ A_nx \to Px $ para \emph{todo} $ x \in H $ ---
as médias temporais convergem para a projeção nos invariantes.
(d) Soletre para $ H = L^2(\R/\Z)$ e $ Uf = f(\cdot +
\alpha)$ com $\alpha $ irracional: identificar $ F $ (usar
Série de Fourier, \cref{thm:b3:hilbert:fourier}) e deduzir
isso $\frac1n\sum_{k<n}f(x + k\alpha) \to \int_0^1f $ em
$ L^2$: o $ L^2$ equidistribuição de rotações irracionais.
\end{exercise}

\section{Problema: polinômios ortogonais}

\begin{problem}[{Problema de fim de semana --- Legendre, Hermite e
Quadratura de Gauss}]\label{pb:b3:hilbert:1}
Deixar $ I \subseteq \R $ seja um intervalo e $ w > 0$ um \omterm{def:b3:topology:continuity}{contínuo}
\emph{peso} no \omterm{def:b3:topology:interior}{interior} de $ I $ tal que $\int_I
\abs t^nw(t)\dd t < \infty $ para todos $ n $; trabalhar em $ H = L^2(I,
w\,\dd\lambda)$ com $\langle f, g\rangle = \int_I \bar
fg\,w $. Gram-Schmidt aplicou-se a $1, t, t^2, \dots $ produz
o \emph{polinômios ortogonais} $(p_n)$ para $ w $ (mônico
normalização: $ p_n = t^n + \cdots $).\index{ortogonal
polinômios}

\medskip
\noindent\textbf{Part I --- General theory.}
\begin{enumerate}
  \item Mostre isso $ p_n $ é ortogonal a todo polinômio de
        \omterm{def:b3:galois:extension}{grau} $< n $, e isso $(p_0, \dots, p_n)$ é uma \omterm{def:b3:topology:basis}{base}
        de $\R_n[t]$.
  \item (Recorrência de três termos) Mostre que existem reais $ a_n,
        b_n $ com
        \[
        p_{n+1}(t) = (t - a_n)\,p_n(t) - b_n\,p_{n-1}(t),
        \qquad b_n = \frac{\norm{p_n}^2}{\norm{p_{n-1}}^2} >
        0 .
        \]
        \emph{(Expandir $ t\,p_n $ na \omterm{def:b3:topology:basis}{base} $(p_k)_{k \leq
        n+1}$ e coeficientes de eliminação por ortogonalidade, usando
        $\langle tp_n, p_k\rangle = \langle p_n,
        tp_k\rangle $.)}
  \item (Raízes) Mostre que $ p_n $ tem $ n $ \emph{distinto} raízes,
        tudo \omterm{def:b3:topology:interior}{interior} para $ I $. \emph{(Deixar $ t_1 < \dots < t_m $ ser
        o \omterm{def:b3:topology:interior}{interior} assinar alterações de $ p_n $; se $ m < n $,
        teste $ p_n $ contra $\prod_{i\leq m}(t - t_i)$ e
        contradiz a ortogonalidade.)}
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Legendre ($ I = \intcc{-1}1$, $ w =
1$).} Definir $ P_n(t) = \frac{1}{2^nn!}\,\frac{\dd^n}{\dd
t^n}\bigl[(t^2 - 1)^n\bigr]$ (Rodrigues).
\begin{enumerate}[resume]
  \item Mostrar $\deg P_n = n $ com coeficiente líder
        $\frac{(2n)!}{2^n(n!)^2}$, e, integrando por partes
        $ n $ vezes, isso $\langle P_n, Q\rangle = 0$ para cada
        polinômio $ Q $ de \omterm{def:b3:galois:extension}{grau} $< n $: o $ P_n $ são (até
        normalização) o polinômios ortogonais para $ w = 1$.
  \item Calcular $\norm{P_n}_2^2 = \frac{2}{2n+1}$
        \emph{(integrar por partes $ n $ vezes contra si mesmo e
        reduzir para uma integral Beta/Wallis,
        \cref{exo:b3:product:8})}.
  \item Mostre que os polinômios de Legendre normalizados formam um
        \omterm{def:b3:topology:basis}{Base} de Hilbert de $ L^2(\intcc{-1}1)$
        \emph{(Weierstrass, \cref{cor:b3:complete:weierstrass},
        mais \omterm{ex:b3:lebesgue:gamma}{densidade} de $\mathcal C $ em $ L^2$)}e expandir
        $ f(t) = \abs t $ até o \omterm{def:b3:galois:extension}{grau} $2$: calcule o melhor
        quadrático $ L^2$-aproximação de $\abs t $.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Hermite ($ I = \R $, $ w(t) =
\eu^{-t^2}$).} Definir $ H_n(t) =
(-1)^n\eu^{t^2}\frac{\dd^n}{\dd t^n}\eu^{-t^2}$.
\begin{enumerate}[resume]
  \item Mostre isso $ H_n $ é um polinômio de \omterm{def:b3:galois:extension}{grau} $ n $ com
        coeficiente líder $2^n $, que $ H_{n+1} = 2tH_n -
        H_n'$, e isso $\langle H_m, H_n\rangle_w =
        \delta_{mn}\,2^nn!\sqrt\pi $ \emph{(partes novamente)}.
  \item Mostre que a família Hermite está totalmente em $ L^2(\R,
        \eu^{-t^2}\dd t)$, admitindo um resultado de
        \cref{ch:b3:fouriertransform}: se $ g \in L^1(\R)$ tem
        $\int g(t)\eu^{-\iu\xi t}\dd t = 0$ para todos $\xi $,
        então $ g = 0$ a.e. \emph{(Para $ f \perp $ todos $ H_n $,
        ou seja\ $\perp $ todos os polinômios: mostrar $ z \mapsto \int
        f(t)\eu^{-t^2}\eu^{-\iu zt}\dd t $ está bem definido,
        expandir o exponencial como uma série, justificar o
        intercâmbio por dominação, e concluímos que o
        Transformada de Fourier de $ f\eu^{-t^2}$ desaparece.)}
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Gauss quadrature.} Consertar $ n $, deixar
$ t_1 < \dots < t_n $ sejam as raízes de $ p_n $ (Parte I) e definir
os pesos $ w_i = \int_I \ell_i(t)\,w(t)\dd t $ onde
$\ell_i $ são os polinômios da \omterm{def:b3:topology:basis}{base} de interpolação de Lagrange em
o $ t_i $.
\begin{enumerate}[resume]
  \item Mostre que a regra da quadratura $ Q(f) = \sum_iw_if(t_i)$
        é exato em todos os polinômios de \omterm{def:b3:galois:extension}{grau} $\leq n - 1$
        (interpolação), e de fato --- o milagre --- em
        todos os polinômios de \omterm{def:b3:galois:extension}{grau} $\leq 2n - 1$: escrever $ P =
        qp_n + r $ e use ortogonalidade no quociente $ q $.
  \item Mostre que os pesos são positivos
        \emph{(aplicar a regra a $\ell_i^2$, de \omterm{def:b3:galois:extension}{grau} $2n -
        2$)}, e deduza do teorema de Polya
        (\cref{exo:b3:banach:9}) que a quadratura de Gauss
        converge: $ Q_n(f) \to \int_I fw $ para cada \omterm{def:b3:topology:continuity}{contínuo}
        $ f $ em um compactar $ I $.
  \item Para $ n = 2$, $ I = \intcc{-1}1$, $ w = 1$: calcular o
        nós $\pm\frac1{\sqrt3}$ e pesos $1, 1$ e
        verificar a exatidão em $1, t, t^2, t^3$ à mão. Comparar
        com a regra do trapézio nas mesmas duas avaliações
        pontos.
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Chebyshev: the polynomials that
oscillate best.} Agora $ I = \intcc{-1}1$ e $ w(t) =
\frac1{\sqrt{1 - t^2}}$.
\begin{enumerate}[resume]
  \item Mostre isso $ T_n(\cos\theta) = \cos n\theta $ define um
        polinômio $ T_n $ de \omterm{def:b3:galois:extension}{grau} $ n $ (estabelecer $ T_{n+1} =
        2t\,T_n - T_{n-1}$ de uma identidade trigonométrica),
        com coeficiente líder $2^{n-1}$ para $ n \geq 1$;
        e que a substituição $ t = \cos\theta $ dá
        \[
        \langle T_m, T_n\rangle_w =
        \int_0^\pi\cos m\theta\,\cos n\theta\,\dd\theta
        = 0 \ (m \neq n), \qquad
        \norm{T_0}_w^2 = \pi,\ \ \norm{T_n}_w^2 = \frac\pi2 :
        \]
        o $ T_n $ são os polinômios ortogonais para isso
        peso e expansões de Chebyshev \emph{são} Fourier
        série cosseno disfarçada.
  \item Localize explicitamente o $ n $ raízes $ t_k =
        \cos\frac{(2k-1)\pi}{2n}$ e o $ n + 1$ extremos
        $ s_j = \cos\frac{j\pi}n $ de $ T_n $ sobre $\intcc{-1}1$,
        onde $ T_n(s_j) = (-1)^j $: o gráfico
        \emph{equioscila} entre $\pm1$.
  \item (Minimax) Mostre que entre todos \emph{Mônica}
        polinômios de \omterm{def:b3:galois:extension}{grau} $ n $, o polinômio
        $2^{1-n}T_n $ tem a menor norma suplente em
        $\intcc{-1}1$, nomeadamente $2^{1-n}$ --- e é o
        minimizador exclusivo. \emph{(Se um mônico $ P $ tive
        $\sup\abs P < 2^{1-n}$, a diferença $2^{1-n}T_n -
        P $, de \omterm{def:b3:galois:extension}{grau} $\leq n-1$, alternaria no sinal em
        o $ n+1$ pontos de equioscilação.)}
  \item Aplicação à interpolação: para nós $ t_1 < \dots
        < t_n $ em $\intcc{-1}1$, o erro de Lagrange
        interpolação de um $\mathcal C^n $ função envolve
        $\omega(t) = \prod_i(t - t_i)$. Mostre que escolher
        Raízes de Chebyshev à medida que os nós são minimizados
        $\sup_{\intcc{-1}1}\abs\omega $, e dê o
        limite resultante $\norm{f -
        L_nf}_\infty \leq \frac{\norm{f^{(n)}}_\infty}
        {2^{n-1}\,n!}$ --- compare com nós igualmente espaçados
        (declarar o fenômeno de Runge como um conto de advertência).
  \item Verificar $\abs{T_n'(\pm1)} = n^2$ \emph{(diferenciar
        $ T_n(\cos\theta) = \cos n\theta $ e tomar limites
        $\theta \to 0, \pi $)}: polinômios limitados por $1$ sobre
        $\intcc{-1}1$ pode ter derivada tão grande quanto $ n^2$
        no limite (a desigualdade de Markov não diz maior ---
        apenas declaração). Onde no intervalo está o
        derivativo vinculado apenas $ O(n)$?
  \item (Quadratura Chebyshev--Gauss) Mostre que o Gauss
        regra para o peso $ w $ no $ n $ Raízes de Chebyshev
        tem \emph{igual} pesos $ w_i = \frac\pi n $
        \emph{(exatidão em $ T_0, \dots, T_{n-1}$ mais o
        somas trigonométricas
        $\sum_{k=1}^n\cos\bigl(j\tfrac{(2k-1)\pi}{2n}\bigr) =
        0$ para $1 \leq j \leq n - 1$)}: o mais uniforme de
        todas as quadraturas. Escreva para $ n = 3$.
\end{enumerate}

\medskip
\noindent\textbf{Part VI --- Christoffel--Darboux,
interlacing, and the Jacobi matrix.} De volta a um peso geral;
$ h_k = \norm{p_k}^2$ (mônico $ p_k $), $ b_k = h_k/h_{k-1}$.
\begin{enumerate}[resume]
  \item (Mínima norma) Mostre que entre todos \emph{Mônica}
        polinômios de \omterm{def:b3:galois:extension}{grau} $ n $, o ortogonal $ p_n $ é
        o único do mínimo $ L^2(w)$-norma --- identificar
        a minimização como uma projeção ortogonal sobre
        $\R_{n-1}[t]$ (\cref{thm:b3:hilbert:projection} ou
        a projeção de dimensão finita do Ano 2). O
        propriedade minimax da questão 14 é a mesma afirmação
        com $ L^\infty $ no lugar de $ L^2$: mesmo herói, dois
        normas.
  \item (Christoffel--Darboux) Prove, por indução em $ n $
        usando a recorrência de três termos, a identidade
        \[
        \sum_{k=0}^{n}\frac{p_k(x)\,p_k(y)}{h_k}
        = \frac{p_{n+1}(x)\,p_n(y) -
        p_n(x)\,p_{n+1}(y)}{h_n\,(x - y)}
        \qquad (x \neq y),
        \]
        e sua forma confluente ($ y \to x $):
        $\sum_{k\leq n}\frac{p_k(x)^2}{h_k} =
        \frac{p_{n+1}'(x)p_n(x) - p_n'(x)p_{n+1}(x)}{h_n}$.
  \item Deduza isso $ p_n $ e $ p_{n+1}$ não têm raiz comum,
        e isso em cada raiz $ x_0$ de $ p_{n+1}$:
        $ p_n(x_0)\,p_{n+1}'(x_0) > 0$. Conclua o
        \emph{entrelaçamento} de raízes: entre dois consecutivos
        raízes de $ p_{n+1}$ reside exatamente em uma raiz de $ p_n $.
  \item (Matriz de Jacobi) Deixe $ J_n $ seja o $ n\times n $
        matriz tridiagonal simétrica com diagonal $ a_0,
        \dots, a_{n-1}$ e entradas fora da diagonal $\sqrt{b_1},
        \dots, \sqrt{b_{n-1}}$. Mostre por indução que
        $\det(tI_n - J_n) = p_n(t)$, então as raízes de $ p_n $
        são os autovalores de uma matriz simétrica real ---
        provando novamente em uma linha que eles são reais, e (com
        o entrelaçamento acima) amarrando polinômios ortogonais
        para o mundo espectral de \cref{ch:b3:spectral}.
  \item (Síntese) Monte o dicionário para os três
        famílias clássicas (Legendre, Hermite, Chebyshev):
        intervalo, peso, fórmula definidora, três termos
        recorrência, norma e habitat natural de cada
        (quadratura e aproximação em compacta; Gaussiana
        análise; métodos minimax e cosseno de Fourier). Um
        frase sobre o que a teoria geral (Partes I, VI)
        deu que nenhum cálculo individual poderia.
\end{enumerate}

\medskip
\noindent\textbf{Part VII --- The error term, and the kernel
behind the weights.} Aqui $ I $ é compactar e $ f \in \mathcal
C^{2n}(I)$.
\begin{enumerate}[resume]
  \item (Fórmula de erro de Gauss) Deixe $ Hf $ seja o eremita
        interpolante de \omterm{def:b3:galois:extension}{grau} $\leq 2n - 1$ correspondência $ f $ e
        $ f'$ nos nós $ t_1, \dots, t_n $ (provar a sua
        existência e o erro pontual
        \[
        f(t) - Hf(t) =
        \frac{f^{(2n)}(\xi_t)}{(2n)!}\;p_n(t)^2
        \]
        pelo argumento usual da função auxiliar). Deduza, por
        integrando esta identidade contra $ w $ e apertando
        entre os extremos de $ f^{(2n)}$, que
        \[
        \int_I f\,w - Q_n(f)
        = \frac{f^{(2n)}(\xi)}{(2n)!}\,h_n
        \qquad\text{for some } \xi \in I,
        \]
        com $ h_n = \norm{p_n}^2$ como na Parte VI: Gauss
        quadratura erra por um $2n $-ésima derivada, ponderada por
        a norma quadrada do polinômio ortogonal mônico.
  \item (Os pesos são valores de Christoffel) Usando o
        reproduzindo kernel $ K_n(x, y) = \sum_{k=0}^{n-1}
        \frac{p_k(x)p_k(y)}{h_k}$ de $\R_{n-1}[t]$ e o
        exatidão de $ Q_n $ até o \omterm{def:b3:galois:extension}{grau} $2n - 2$, provar
        \[
        w_i \;=\;
        \Bigl(\,\sum_{k=0}^{n-1}
        \frac{p_k(t_i)^2}{h_k}\Bigr)^{\!-1} :
        \]
        cada peso é o valor em seu nó do
        \emph{Função Christoffel} --- positividade do
        pesos (questão 10) novamente, agora com um valor exato
        fórmula. Verifique se ele se recupera $ w_1 = w_2 = 1$ para $ n =
        2$, $ I = \intcc{-1}1$, $ w = 1$.
  \item (Tudo verifica uma integral) Para o Chebyshev
        peso e $ n = 3$ nós, calcule ambos os lados de
        \[
        \int_{-1}^{1}\frac{t^6}{\sqrt{1 - t^2}}\,\dd t
        = \frac{5\pi}{16},
        \qquad
        Q_3(t^6) = \frac{9\pi}{32},
        \]
        então o erro de quadratura é exatamente $\frac{\pi}{32}$;
        então verifique se a fórmula de erro da questão 23
        prevê precisamente este valor (aqui $ f^{(6)} = 6!$ é
        constante, e $ h_3 = \norm{2^{-2}T_3}_w^2 =
        \frac\pi{32}$): teoria e computação concordam com o
        último dígito.
\end{enumerate}
\end{problem}
