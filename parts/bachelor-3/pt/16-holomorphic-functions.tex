\chapter{Funções Holomórficas}\label{ch:b3:holomorphic}

A diferenciabilidade complexa parece uma pequena variação na
teoria real --- um limite, um quociente. Em vez disso, é um
universo diferente. Uma função diferenciável \emph{uma vez} em um
subconjunto aberto de $\C $ é automaticamente infinitamente diferenciável,
analítico, determinado em todo um domínio por seus valores próximos a um
ponto único e limitado por rígidos princípios globais
(Liouville, módulo \omterm{def:b3:rings:primemaximal}{máximo}). Tudo isso flui de um
milagre, teorema de Cauchy: a integral de um \omterm{def:b3:holomorphic:holo}{holomórfico}
função ao longo de um fechado caminho em um domínio em forma de estrela desaparece.
Este capítulo prova o milagre (argumento de Goursat, sem
regularidade assumida além da diferenciabilidade), colhe sua
consequências clássicas e estabelece um teorema que este livro
usado a crédito desde \cref{ch:b3:galois}: todo não constante
polinômio complexo tem uma raiz.

Por todo, $\Omega \subseteq \C $ está aberto e $ D(a, r)$
denota o disco aberto.

\section{Diferenciabilidade complexa}

\begin{definition}\label{def:b3:holomorphic:holo}
$ f \colon \Omega \to \C $ é \emph{holomórfico}\index{holomórfico} sobre $\Omega $ se para cada $ z_0 \in \Omega $
\[
f'(z_0) = \lim_{h\to0}\frac{f(z_0 + h) - f(z_0)}{h}
\]
existe ($ h \in \C^*$). Somas, produtos, quocientes (que não desaparecem
denominadores), composições de funções holomórficas são
holomórfico, com as fórmulas usuais (as provas do Ano 1--2 são
literalmente: eles usam apenas operações de campo e limites).
$\mathcal H(\Omega)$ denota o conjunto de funções holomórficas
ligado $\Omega $.
\end{definition}

\begin{proposition}[Cauchy-Riemann]
\label{prop:b3:holomorphic:cauchyriemann}
Escrever $ f(x + \iu y) = P(x,y) + \iu Q(x,y)$. Então $ f $ é
\omterm{def:b3:holomorphic:holo}{holomórfico} no $ z_0$ se $ f $ é $\R $- diferenciável em $ z_0$
(como um mapa de duas variáveis reais) e\index{Cauchy-Riemann
equações}
\[
\frac{\partial P}{\partial x} = \frac{\partial Q}{\partial y},
\qquad
\frac{\partial P}{\partial y} = -\frac{\partial Q}{\partial x}
\qquad \text{at } z_0 ;
\]
equivalentemente, o diferencial real é a multiplicação pelo
número complexo $ f'(z_0)$.
\end{proposition}

\begin{proof}
$\C $-diferenciabilidade diz $ f(z_0 + h) = f(z_0) + ch +
o(\abs h)$ com $ c = f'(z_0)$: um $\R $-diferencial linear
que é multiplicação por $ c = a + \iu b $, ou seja,\ tem matriz
$\bigl(\begin{smallmatrix} a & -b\\ b & a
\end{smallmatrix}\bigr)$ na \omterm{def:b3:topology:basis}{base} $(1, \iu)$ --- exatamente
as relações exibidas para as parciais. Por outro lado, tal
diferencial é $\C $-linear e o $ o(\abs h)$ definições
combinar.
\end{proof}

\begin{example}\label{ex:b3:holomorphic:examples}
Polinômios em $ z $, funções racionais fora de seus pólos, e
--- pelo teorema de diferenciação termo a termo do ano 2 para potência
série, cuja prova funciona de forma idêntica em $\C $ --- cada soma
de uma série de potências $\sum a_n(z - a)^n $ dentro do seu disco de
convergência: \omterm{def:b3:holomorphic:holo}{holomórfico}, com derivada $\sum na_n(z -
a)^{n-1}$ (mesmo raio). Em particular $\exp z = \sum z^n/n!$
é inteiro (\omterm{def:b3:holomorphic:holo}{holomórfico} sobre $\C $) com $\exp' = \exp $. Por outro lado $ z \mapsto \bar z $, $\abs z $, $\operatorname{Re}z $
não estão em lugar nenhum \omterm{def:b3:holomorphic:holo}{holomórfico} (Cauchy--Riemann falha em todos os lugares):
holomorfia é a rigidez que preserva a orientação e o ângulo, não
suavidade.
\end{example}

\section{Integrais de contorno}

\begin{definition}\label{def:b3:holomorphic:contour}
UM \emph{caminho} é um pedaço $\mathcal C^1$ mapa $\gamma \colon
\intcc ab \to \C $; isso é \emph{fechado} se $\gamma(a) =
\gamma(b)$. Para \omterm{def:b3:topology:continuity}{contínuo} $ f $ na imagem de
$\gamma $:\index{integral de contorno}
\[
\int_\gamma f(z)\,\dd z =
\int_a^b f(\gamma(t))\,\gamma'(t)\,\dd t,
\qquad
\Bigl|\int_\gamma f\,\dd z\Bigr| \leq
\sup_{\gamma}\abs f\cdot\operatorname{length}(\gamma)
\]
(o \emph{Desigualdade de ML}; comprimento $= \int_a^b\abs{\gamma'}$).
A integral é invariante sob aumento $\mathcal C^1$
reparametrização e mudança de sinal sob inversão de orientação.
\end{definition}

\begin{proposition}[Primitivos]\label{prop:b3:holomorphic:primitive}
Para \omterm{def:b3:topology:continuity}{contínuo} $ f $ sobre $\Omega $, os seguintes são equivalentes:
(eu) $ f $ tem um \emph{primitivo} $ F \in \mathcal H(\Omega)$ ($ F'
= f $); (ii) $\int_\gamma f\,\dd z = 0$ para cada fechado caminho
$\gamma $ em $\Omega $. Nesse caso $\int_\gamma f\,\dd z =
F(\gamma(b)) - F(\gamma(a))$ para cada caminho.
\end{proposition}

\begin{proof}
(eu)$\Rightarrow $: $\frac{\dd}{\dd t}F(\gamma(t)) =
F'(\gamma(t))\gamma'(t)$ (regra da cadeia, válida por partes), então o
telescópios integrais até a diferença final; fechado caminhos
dar $0$. (ii)$\Rightarrow $(eu): corrigir $ z_*$ em um \omterm{def:b3:topology:connected}{conectado}
componente, definir $ F(z) = \int_{\gamma_z}f\,\dd z $ ao longo de qualquer
caminho de $ z_*$ para $ z $ (bem definido: dois caminhos diferem por um
fechado caminho); para $ h $ pequeno, tirando o segmento de $ z $ para
$ z + h $,
\[
\frac{F(z + h) - F(z)}{h} - f(z)
= \frac1h\int_{[z, z+h]}\bigl(f(w) - f(z)\bigr)\dd w
\longrightarrow 0
\]
pela desigualdade ML e continuidade de $ f $ no $ z $.
\end{proof}

\begin{definition}[Número do enrolamento]\label{def:b3:holomorphic:index}
Para um fechado caminho $\gamma $ e $ z \notin
\operatorname{im}\gamma $, o \emph{índice}\index{índice}\index{índice de um caminho} é
\[
\operatorname{Ind}_\gamma(z) = \frac1{2\iu\pi}
\int_\gamma\frac{\dd w}{w - z} .
\]
É um \emph{inteiro}: contexto $\varphi(t) =
\int_a^t\frac{\gamma'(s)}{\gamma(s) - z}\dd s $, a função
$(\gamma(t) - z)\eu^{-\varphi(t)}$ tem derivada zero
(por partes), portanto, é constante; no $ t = b $, $\eu^{\varphi(b)} =
\frac{\gamma(b) - z}{\gamma(a) - z} = 1$, então $\varphi(b) \in
2\iu\pi\Z $. Em função de $ z $, o índice é \omterm{def:b3:topology:continuity}{contínuo} sobre
$\C\setminus\operatorname{im}\gamma $ (convergência dominada),
portanto constante em cada \omterm{def:b3:topology:components}{componente conectado}, e $0$ no
componente ilimitado (ML: a integral tende a $0$ como $ z \to
\infty $). Para o círculo $\gamma(t) = a + r\eu^{\iu t}$, $ t
\in \intcc0{2\pi}$: $\operatorname{Ind}_\gamma(z) = 1$ para $ z
\in D(a,r)$ (calcular em $ z = a $: $\frac1{2\iu\pi}
\int_0^{2\pi}\frac{r\iu\eu^{\iu t}}{r\eu^{\iu t}}\dd t = 1$;
a constância faz o resto).
\end{definition}

\section{Teorema de Cauchy}

\begin{theorem}[Goursat]\label{thm:b3:holomorphic:goursat}
Deixar $ f \in \mathcal H(\Omega)$ e $ T \subseteq \Omega $ um
triângulo sólido fechado. Então $\int_{\partial T}f\,\dd z = 0$
(limite executado uma vez, qualquer orientação).\index{Teorema de Goursat}
\end{theorem}

\begin{proof}
Deixar $ I(T) = \int_{\partial T}f\,\dd z $. Unindo os pontos médios
dos lados se divide $ T $ em quatro triângulos de meio tamanho $ T^{(1)},
\dots, T^{(4)}$, e as arestas internas se cancelam em pares: $ I(T) =
\sum_iI(T^{(i)})$. Escolher $ T_1$ entre eles com
$\abs{I(T_1)} \geq \frac14\abs{I(T)}$ e iterar: um aninhado
sequência $ T \supseteq T_1 \supseteq T_2 \supseteq\cdots $ com
\[
\abs{I(T_n)} \geq 4^{-n}\abs{I(T)},
\qquad
\operatorname{diam}T_n = 2^{-n}\operatorname{diam}T,
\quad
\operatorname{length}(\partial T_n) =
2^{-n}\operatorname{length}(\partial T).
\]
A interseção $\bigcap T_n $ é um único ponto $ z_0$
(aninhado compactos com diâmetros desaparecendo,
\cref{thm:b3:topology:compactprops}(3)). Diferenciabilidade em
$ z_0$: dado $\varepsilon $, para $ n $ grande, em $ T_n $,
\[
f(z) = f(z_0) + f'(z_0)(z - z_0) + R(z),
\qquad \abs{R(z)} \leq \varepsilon\abs{z - z_0} \leq
\varepsilon\operatorname{diam}T_n .
\]
A parte afim tem uma primitiva: sua integral na parte fechada
$\partial T_n $ desaparece (\cref{prop:b3:holomorphic:primitive}),
saindo
\[
\abs{I(T_n)} = \Bigl|\int_{\partial T_n}R\Bigr|
\leq \varepsilon\operatorname{diam}(T_n)\,
\operatorname{length}(\partial T_n)
= \varepsilon\,4^{-n}\operatorname{diam}(T)
\operatorname{length}(\partial T) .
\]
Comparando com $\abs{I(T_n)} \geq 4^{-n}\abs{I(T)}$:
$\abs{I(T)} \leq \varepsilon\cdot\text{const}$ para cada
$\varepsilon $: $ I(T) = 0$.
\end{proof}

\begin{theorem}[Teorema de Cauchy, versão em forma de estrela]
\label{thm:b3:holomorphic:cauchy}
Deixar $\Omega $ ser \emph{em forma de estrela} sobre $ c $ (cada segmento
$[c, z]$, $ z \in \Omega $, reside em $\Omega $) --- por exemplo\ convexo.
Todo $ f \in \mathcal H(\Omega)$ tem um primitivo em $\Omega $;
consequentemente $\int_\gamma f\,\dd z = 0$ para \emph{todo}
fechado caminho $\gamma $ em $\Omega $.\index{Teorema de Cauchy}
\end{theorem}

\begin{proof}
Definir $ F(z) = \int_{[c,z]}f\,\dd w $. Para $ z, z + h \in
\Omega $ com $[z, z+h] \subseteq \Omega $ (verdadeiro para pequenos $ h $),
o triângulo com vértices $ c, z, z+h $ reside em $\Omega $
(formato de estrela: cada um de seus pontos está em um segmento $[c,
w]$ com $ w \in [z, z+h] \subseteq \Omega $): Goursat dá
\[
F(z + h) - F(z) = \int_{[z, z+h]}f\,\dd w,
\]
e o cálculo do quociente de diferença de
\cref{prop:b3:holomorphic:primitive} rendimentos $ F' = f $. O
desaparecimento de caminho fechado integrais seguem do mesmo
proposição.
\end{proof}

\begin{theorem}[Fórmula integral de Cauchy]
\label{thm:b3:holomorphic:formula}
Deixar $ f \in \mathcal H(\Omega)$, $\bar D(a, r) \subseteq
\Omega $, e $ C_r $ o círculo $\partial D(a,r)$ execute uma vez
sentido anti-horário. Então para cada $ z \in D(a, r)$:\index{Cauchy
fórmula integral}
\[
f(z) = \frac1{2\iu\pi}\int_{C_r}\frac{f(w)}{w - z}\,\dd w .
\]
\end{theorem}

\begin{proof}
Consertar $ z $ e definir em $\Omega $
\[
g(w) = \begin{cases}
\dfrac{f(w) - f(z)}{w - z} & w \neq z,\\[4pt]
f'(z) & w = z :
\end{cases}
\]
$ g $ é \omterm{def:b3:topology:continuity}{contínuo} sobre $\Omega $ e \omterm{def:b3:holomorphic:holo}{holomórfico} desligado $ z $. Goursat
vale para $ g $ em cada triângulo $ T \subseteq \Omega'$, onde
$\Omega'$ é um disco ligeiramente maior que $\bar D(a,r)$ dentro
$\Omega $: se $ z \notin T $, diretamente; se $ z \in T $, dividir $ T $
em pequenos triângulos tendo $ z $ como um vértice mais triângulos
evitando $ z $; em um triângulo com vértice $ z $ o limite de ML dá
$\abs{\int_{\partial T'}g} \leq \sup_{T'}\abs
g\cdot\operatorname{length} \to 0$ à medida que o triângulo encolhe,
e as peças restantes desaparecem por Goursat --- daí
$\int_{\partial T}g = 0$ em todos os casos. A prova de
\cref{thm:b3:holomorphic:cauchy} usei apenas este triângulo
propriedade: $ g $ tem um primitivo no convexo $\Omega'$, então
$\int_{C_r}g = 0$, ou seja
\[
\frac1{2\iu\pi}\int_{C_r}\frac{f(w)}{w - z}\dd w
= f(z)\,\frac{1}{2\iu\pi}\int_{C_r}\frac{\dd w}{w - z}
= f(z)\operatorname{Ind}_{C_r}(z) = f(z) .
\]
\end{proof}

\section{Analiticidade e sua cascata}

\begin{theorem}[Holomórfico $=$ analítico]
\label{thm:b3:holomorphic:analytic}
Deixar $ f \in \mathcal H(\Omega)$ e $ D(a, R) \subseteq \Omega $.
Então\index{analiticidade de funções holomórficas}
\[
f(z) = \sum_{n\geq0}c_n\,(z - a)^n \quad \text{on } D(a, R),
\qquad
c_n = \frac1{2\iu\pi}\int_{C_r}\frac{f(w)}{(w -
a)^{n+1}}\,\dd w \ \ (0 < r < R),
\]
sendo os coeficientes independentes de $ r $. Consequentemente $ f $ é
infinitamente $\C $- diferenciável, $ c_n = f^{(n)}(a)/n!$, e o
\emph{Estimativas de Cauchy}\index{Estimativas de Cauchy} segurar:
\[
\abs{c_n} \;\leq\; \frac{\sup_{\abs{w - a} =
r}\abs{f(w)}}{r^{n}} .
\]
\end{theorem}

\begin{proof}
Para $\abs{z - a} < r $: expanda o kernel Cauchy no
série geométrica
\[
\frac1{w - z} = \frac1{(w - a)\bigl(1 - \frac{z - a}{w -
a}\bigr)} = \sum_{n\geq0}\frac{(z - a)^n}{(w - a)^{n+1}},
\]
normalmente convergente em $ w $ sobre $ C_r $ ($\abs{\frac{z-a}{w-a}} =
\frac{\abs{z-a}}r < 1$): integrar termo por termo contra
$\frac{f(w)}{2\iu\pi}$ (a convergência uniforme justifica a
intercâmbio) e aplicar \cref{thm:b3:holomorphic:formula}. Um
série de potências é infinitamente diferenciável com $ c_n =
f^{(n)}(a)/n!$ (Ano 2), que também mostra o $ c_n $ não
depender de $ r $. As estimativas: limitam a integral do coeficiente
por M.L.
\end{proof}

\begin{example}[Singularidades ditam raios]
\label{ex:b3:holomorphic:radius}
Por que o real inocente funciona $\frac1{1 + x^2}$ tenha um
Série de Taylor em $ x = 3$ convergindo apenas para $\abs{x - 3} <
\sqrt{10}$, quando nada dá errado na linha real? Porque
o teorema acima faz com que o raio de convergência em $ a $
igual à distância de $ a $ para o ponto mais próximo onde
holomorfia falha. Aqui $ f(z) = \frac1{1 + z^2}$ é
\omterm{def:b3:holomorphic:holo}{holomórfico} exatamente em $\C\setminus\{\pm\iu\}$, então o
expansão em $ a = 3$ converge para o maior disco evitando
$\pm\iu $, de raio $\abs{3 - \iu} = \sqrt{10}$ --- e
não pode convergir para um maior, pois a soma se estenderia
$ f $ holomorficamente para um \omterm{def:b3:topology:topology}{vizinhança} de $\pm\iu $, onde
$\abs f \to \infty $. A verdadeira teoria vê o misterioso
raio $\sqrt{10}$; o plano complexo vê dois pólos.
Esta é a regra prática: \emph{para encontrar um raio de
convergência, localize as singularidades} --- por exemplo\ o Taylor
série de $\tan $ no $0$ tem raio $\frac\pi2$ (mais próximo
zeros de $\cos $) e a função geradora de Bernoulli
$\frac z{\eu^z - 1}$ (\cref{pb:b3:holomorphic:1}, Parte VI)
tem raio $2\pi $ (zeros diferentes de zero mais próximos de $\eu^z - 1$:
$\pm2\iu\pi $).
\end{example}

\begin{corollary}[Liouville; d'Alembert - Gauss]
\label{cor:b3:holomorphic:liouville}
Uma função inteira limitada é constante. Consequentemente cada
polinômio não constante sobre $\C $ tem uma raiz: \emph{$\C $ é
\omterm{def:b3:galois:closure}{algebricamente fechado}}.\index{Teorema de Liouville}\index{fundamental
teorema da álgebra}
\end{corollary}

\begin{proof}
Se $\abs f \leq M $ sobre $\C $: para cada $ a $ e $ r $, $\abs{c_1(a)}
= \abs{f'(a)} \leq M/r \to 0$: $ f' \equiv 0$, e $ f $ é
constante (no \omterm{def:b3:topology:connected}{conectado} $\C $: derivada zero implica
localmente constante --- integrar ao longo dos segmentos). Se $ P $ não tinha
raiz, $1/P $ seria inteiro e limitado ($\abs{P(z)} \to
\infty $ como $\abs z \to \infty $: o termo principal domina, então
$\abs{1/P}$ é pequeno fora de um disco grande e \omterm{def:b3:topology:continuity}{contínuo} no compactar disco): constante --- absurdo para não constante $ P $.
(O problema do fim de semana fornece uma segunda prova elementar e a
\omterm{def:b3:galois:algebraic}{algébrico} consequências.)
\end{proof}

\begin{theorem}[Zeros são isolados; teorema da identidade]
\label{thm:b3:holomorphic:identity}
Deixar $\Omega $ ser \emph{\omterm{def:b3:topology:connected}{conectado}} e $ f \in \mathcal
H(\Omega)$, $ f \not\equiv 0$. Então cada zero $ a $ de $ f $ tem
finito \emph{ordem}: $ f(z) = (z - a)^m\,g(z)$ com $ g \in
\mathcal H(\Omega)$, $ g(a) \neq 0$, e os zeros de $ f $ tem
nenhum ponto de acumulação em $\Omega $. Consequentemente, se dois
funções holomórficas sobre $\Omega $ concordar em um conjunto com um
ponto de acumulação em $\Omega $, eles concordam
em todos os lugares.\index{teorema da identidade}\index{zeros isolados}
\end{theorem}

\begin{proof}
Deixar $ Z $ seja o conjunto de pontos onde todas as derivadas de $ f $
desaparecer. $ Z $ é fechado (intersecção de conjuntos fechados) e aberto:
se tudo $ c_n = 0$ no $ a $, a expansão em série de potências faz $ f
\equiv 0$ em um disco ao redor $ a $. Conectividade: $ Z =
\varnothing $ ou $ Z = \Omega $; este último é excluído por $ f
\not\equiv 0$. Então, em zero $ a $, algum coeficiente é diferente de zero:
deixe $ m $ seja mínimo com $ c_m \neq 0$; então $ f(z) = (z -
a)^m\sum_{n\geq m}c_n(z-a)^{n-m}$ em um disco, e a soma
define $ g $ \omterm{def:b3:holomorphic:holo}{holomórfico} aproximar $ a $ com $ g(a) = c_m \neq 0$;
estender $ g = f/(z-a)^m $ desligado $ a $ (\omterm{def:b3:holomorphic:holo}{holomórfico} lá). Desde
$ g(a) \neq 0$ e $ g $ é \omterm{def:b3:topology:continuity}{contínuo}, $ f $ não tem outro zero em
um \omterm{def:b3:topology:topology}{vizinhança} de $ a $: zeros são isolados e um conjunto de
pontos isolados não tem ponto de acumulação em $\Omega $ (um
ponto de acumulação de zeros é um zero --- continuidade --- e
não ficaria isolado). Identidade: aplicar à diferença,
cujo conjunto zero tem um ponto de acumulação, forçando-o para o
$ Z = \Omega $ filial.
\end{proof}

\begin{theorem}[Valor médio e módulo máximo]
\label{thm:b3:holomorphic:maximum}
Deixar $ f \in \mathcal H(\Omega)$.
\begin{enumerate}
  \item (Valor médio) Para $\bar D(a, r) \subseteq \Omega $:
        $ f(a) = \frac1{2\pi}\int_0^{2\pi}f(a +
        r\eu^{\iu t})\,\dd t $.
  \item (Princípio \omterm{def:b3:rings:primemaximal}{máximo}) Se $\Omega $ é \omterm{def:b3:topology:connected}{conectado} e
        $\abs f $ atinge um \omterm{def:b3:rings:primemaximal}{máximo} local em algum ponto
        $\Omega $, então $ f $ é constante. Consequentemente, para
        $\Omega $ limitado e $ f $ \omterm{def:b3:topology:continuity}{contínuo} sobre
        $\bar\Omega $: $\sup_{\bar\Omega}\abs f =
        \sup_{\partial\Omega}\abs f $.\index{princípio máximo}
\end{enumerate}
\end{theorem}

\begin{proof}
(1) é a fórmula de Cauchy no \omterm{ex:b3:groups:actions}{centro}: parametrizar $ C_r $.
(2) Diga $\abs f \leq \abs{f(a)}$ sobre $\bar D(a, \rho)
\subseteq \Omega $. Se $ f(a) = 0$, $ f \equiv 0$ aproximar $ a $.
Caso contrário, por $0 < r \leq \rho $, o valor médio dá
\[
\abs{f(a)} \leq \frac1{2\pi}\int_0^{2\pi}\abs{f(a +
r\eu^{\iu t})}\,\dd t \leq \abs{f(a)} :
\]
o \omterm{def:b3:topology:continuity}{contínuo} integrando não negativo $\abs{f(a)} - \abs{f(a +
r\eu^{\iu t})}$ tem média zero, portanto desaparece: $\abs f $ é
\emph{constante} $= \abs{f(a)} \ne 0$ no disco. UM \omterm{def:b3:holomorphic:holo}{holomórfico}
função do módulo constante diferente de zero em um disco é constante:
diferenciando $ P^2 + Q^2 = \text{const}$ dá $ PP_x + QQ_x
= 0$ e $ PP_y + QQ_y = 0$; substituindo o Cauchy--Riemann
relações $ P_y = -Q_x $, $ Q_y = P_x $ na segunda equação
produz o sistema linear
\[
P\,P_x + Q\,Q_x = 0, \qquad -P\,Q_x + Q\,P_x = 0,
\]
cujo determinante é $ P^2 + Q^2 \neq 0$: $ P_x = Q_x = 0$, então
$ f' = P_x + \iu Q_x = 0$ no disco: $ f $ constante aí. O
o teorema da identidade espalha constância para todos $\Omega $. A forma limite: $\abs f $
atinge seu sup no compactar $\bar\Omega $; um \omterm{def:b3:topology:interior}{interior}
\omterm{def:b3:rings:primemaximal}{máximo} faz $ f $ constante, e o sup é alcançado no
limite em todos os casos.
\end{proof}

\begin{theorem}[Teorema da convergência de Weierstrass]
\label{thm:b3:holomorphic:weierstrassconv}
Se $ f_n \in \mathcal H(\Omega)$ convergir para $ f $ uniformemente ligado
cada compactar subconjunto de $\Omega $, então $ f \in \mathcal
H(\Omega)$ e $ f_n^{(k)} \to f^{(k)}$ uniformemente ligado compactos,
para cada $ k $.\index{Teorema da convergência de Weierstrass}
\end{theorem}

\begin{proof}
$ f $ é \omterm{def:b3:topology:continuity}{contínuo}. Para qualquer triângulo fechado $ T \subseteq
\Omega $: $\int_{\partial T}f = \lim\int_{\partial T}f_n = 0$
(convergência uniforme no compactar $\partial T $; Goursat para
$ f_n $). Pelo argumento de \cref{thm:b3:holomorphic:cauchy},
$ f $ tem primitivas locais (os discos são convexos; apenas o triângulo
propriedade foi usada), ou seja,\ $ f = F'$ localmente com $ F $
\omterm{def:b3:holomorphic:holo}{holomórfico}; $ F $ é analítico
(\cref{thm:b3:holomorphic:analytic}), portanto, também é $ f = F'$:
\omterm{def:b3:holomorphic:holo}{holomórfico}. (Isso é \emph{Morera's
teorema}\index{Morera's
teorema}: \omterm{def:b3:topology:continuity}{contínuo} com desaparecimento
integrais de triângulo implicam \omterm{def:b3:holomorphic:holo}{holomórfico}.) Derivados: para
$\bar D(a, 2r) \subseteq \Omega $ e $ z \in \bar D(a, r)$,
Fórmula de Cauchy para derivadas (diferenciar
\cref{thm:b3:holomorphic:formula} sob a integral, ou use
a fórmula do coeficiente) dá
\[
\abs{f_n'(z) - f'(z)} =
\Bigl|\frac{1}{2\iu\pi}\int_{C_{2r}}\frac{f_n(w) - f(w)}{(w -
z)^2}\,\dd w\Bigr|
\leq \frac{2r\,\sup_{C_{2r}}\abs{f_n - f}}{r^2} \to 0
\]
uniformemente ligado $\bar D(a,r)$; cobrir um compactar por um número finito
tais discos, e iterar para maior $ k $.
\end{proof}

\begin{method}\label{met:b3:holomorphic:toolkit}
O kit de ferramentas diário. Para provar uma função \omterm{def:b3:holomorphic:holo}{holomórfico}: exibi-lo
como uma série de potências, uma composição, um limite localmente uniforme
(\cref{thm:b3:holomorphic:weierstrassconv}) ou uma integral
com \omterm{def:b3:holomorphic:holo}{holomórfico} parâmetro (\cref{exo:b3:holomorphic:7} ---
diferenciar sob $\int $ ou aplique Morera-Fubini). Para provar
identidades: prove-as em um segmento ou subdomínio e invoque
o teorema da identidade. Para vincular: \omterm{thm:b3:holomorphic:analytic}{Estimativas de Cauchy} no
maior círculo disponível. Para provar constância/inexistência:
Liouville ou o princípio do \omterm{def:b3:rings:primemaximal}{máximo}. Sempre saiba \emph{onde}
sua função é \omterm{def:b3:holomorphic:holo}{holomórfico} e \emph{qual} os discos cabem
$\Omega $.
\end{method}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:b3:holomorphic:1}
(a) Em quais pontos estão $ z \mapsto \bar z $, $\abs z^2$,
$\operatorname{Re}z $ complexo diferenciável? \omterm{def:b3:holomorphic:holo}{Holomórfico} em um
conjunto aberto?
(b) Mostre que $ P(x, y) = x^2 - y^2$ é a parte real de um
função holomórfica sobre $\C $, encontrado explicitamente e determine
\emph{todos} deles.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b3:holomorphic:2}
Calcule a partir das definições: $\int_{C}z^n\,\dd z $ para todos $ n
\in \Z $, $ C $ o círculo unitário; $\int_\gamma\bar z\,\dd z $ junto
o segmento $[0, 1+\iu]$ e ao longo dos dois segmentos caminho
através $1$: concluir que $\bar z $ não tem primitivo em nenhum
\omterm{def:b3:topology:topology}{vizinhança} destes caminhos.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:holomorphic:3}
(a) Mostre que o logaritmo principal $\log z = \ln\abs z +
\iu\arg z $ ($\arg \in \intoo{-\pi}\pi $) é \omterm{def:b3:holomorphic:holo}{holomórfico} sobre
$\C\setminus\intoc{-\infty}0$ com derivada $\frac1z $
\emph{(primitivo de $\frac1z $ no plano de corte em forma de estrela:
\cref{thm:b3:holomorphic:cauchy}; corrigir a constante)}.
(b) Mostre que não \omterm{def:b3:topology:continuity}{contínuo} logaritmo existe em
$\C^*$ \emph{(sua obstrução \omterm{def:b3:modules:free}{livre} de derivativos: o \omterm{def:b3:holomorphic:index}{índice} de
o círculo unitário)}.
(c) Expandir $\log(1 + z)$ em série de potências em $ D(0,1)$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:holomorphic:4}
(a) Deixe $ f $ estar inteiro com $\abs{f(z)} \leq C(1 + \abs
z)^{n}$. Mostre isso $ f $ é um polinômio de \omterm{def:b3:galois:extension}{grau} $\leq n $
\emph{(\omterm{thm:b3:holomorphic:analytic}{Estimativas de Cauchy} em círculos grandes)}.
(b) Deixe $ f $ estar inteiro com $\operatorname{Re}f $ delimitado acima.
Mostre isso $ f $ é constante \emph{(considerar $\eu^{f}$)}.
(c) Deduza o pequeno Picard para mapas afins: um conjunto inteiro
a função que omite um semiplano é constante.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:holomorphic:5}
(a) Deixe $ f $ ser \omterm{def:b3:holomorphic:holo}{holomórfico} em um \omterm{def:b3:topology:connected}{conectado} $\Omega \ni 0$ com
$ f(\frac1n) = \frac1{n^2}$ para todos os grandes $ n $. Determinar $ f $.
(b) Algum \omterm{def:b3:holomorphic:holo}{holomórfico} $ f $ sobre $\C^*$ satisfazer $ f(\frac1n) =
\frac{(-1)^n}{n}$ para todos $ n \geq 1$? Justificar.
(c) Apresentar dois distintos funções holomórficas sobre
$ D(0,1)\sqcup D(3,1)$ concordando em $ D(0,1)$: onde fica o
uso do teorema da identidade conectividade?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:holomorphic:6}
Deixar $ f $ ser \omterm{def:b3:holomorphic:holo}{holomórfico} no disco da unidade aberta $\mathbb D $,
\omterm{def:b3:topology:continuity}{contínuo} sobre $\bar{\mathbb D}$, com $\abs f \equiv 1$ no
círculo limite.
(a) Se $ f $ não tem zero em $\mathbb D $, mostrar $ f $ é constante
\emph{(aplicar o princípio do \omterm{def:b3:rings:primemaximal}{máximo} a $ f $ e para $1/f $)}.
(b) Dê um exemplo com zero onde $ f $ não é constante.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:holomorphic:7}
(Holomorfia sob a integral) Deixe $\mu $ ser um finito \omterm{def:b3:measure:measure}{medir}
em um espaço $ X $ e $ g \colon X\times\Omega \to \C $ com:
$ g(x, \cdot) \in \mathcal H(\Omega)$ para cada $ x $, $ g $
\omterm{def:b3:lebesgue:measurable}{mensurável} em $ x $, e $\abs g \leq h(x)$ com $ h $ \omterm{def:b3:lebesgue:l1}{integrável},
localmente uniformemente em $ z $. Mostrar $ G(z) = \int_Xg(x, z)\dd\mu(x)$
é \omterm{def:b3:holomorphic:holo}{holomórfico} sobre $\Omega $. \emph{(Morera: integrais triangulares
desaparecer por Fubini e Goursat; continuidade por dominado
convergência. Então aplique em $\Gamma(z) =
\int_0^\infty t^{z-1}\eu^{-t}\dd t $ sobre $\{\operatorname{Re}z >
0\}$.)}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b3:holomorphic:8}
(Gauss--Lucas) Deixe $ P \in \C[X]$ ser inconstante. Mostre isso
cada raiz de $ P'$ encontra-se na casca convexa das raízes de
$ P $. \emph{(Escrever $\frac{P'}{P} = \sum_k\frac{m_k}{z - a_k}$
em uma raiz $ z $ de $ P'$ isso não é uma raiz de $ P $, pegue
conjugados e leia uma combinação convexa.)} Ilustrar em
$ P = z^3 - 1$.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b3:holomorphic:9}
Deixar $ f $ ser inteiro e duplamente periódico: $ f(z + 1) = f(z + \iu)
= f(z)$ para todos $ z $. Mostre isso $ f $ é constante. \emph{(Vinculado
$ f $ no compactar quadrado fundamental, então em todos os lugares;
Liouville.)} Moral: funções elípticas não constantes devem ter
pólos --- o tema de \cref{ch:b3:residues}.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:holomorphic:10}
(a) Mostre que $ P = \operatorname{Re}f $ de um \omterm{def:b3:holomorphic:holo}{holomórfico} $ f $
satisfaz a propriedade do valor médio $ P(a) =
\frac1{2\pi}\int_0^{2\pi}P(a + r\eu^{\iu t})\dd t $ e é
\emph{harmônico}: $\partial^2_{xx}P + \partial^2_{yy}P = 0$
(diferenciar Cauchy--Riemann; usar
\cref{thm:b3:holomorphic:analytic} para a suavidade necessária).
(b) Deduza o princípio do \omterm{def:b3:rings:primemaximal}{máximo} para partes reais de \omterm{def:b3:holomorphic:holo}{holomórfico}
funções em domínios limitados.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b3:holomorphic:11}
(Reflexão de Schwarz) Vamos $\Omega^+ = \{z : \abs z < 1,\
\operatorname{Im}z > 0\}$, $ I = \intoo{-1}1$, e $ f $
\omterm{def:b3:holomorphic:holo}{holomórfico} sobre $\Omega^+$, \omterm{def:b3:topology:continuity}{contínuo} sobre $\Omega^+\cup I $,
\emph{com valor real em $ I $}. Definir
\[
F(z) = \begin{cases} f(z) & z \in \Omega^+\cup I,\\
\overline{f(\bar z)} & \bar z \in \Omega^+ . \end{cases}
\]
(a) Mostre que $ F $ está bem definido e \omterm{def:b3:topology:continuity}{contínuo} sobre
$\Omega = \Omega^+\cup I\cup\Omega^-$, e \omterm{def:b3:holomorphic:holo}{holomórfico} sobre
$\Omega^\pm $ \emph{(para $\Omega^-$: verifique Cauchy--Riemann
para $\overline{f(\bar z)}$ ou expandir $ f $ no poder local
séries e coeficientes conjugados)}.
(b) Mostre que $ F $ é \omterm{def:b3:holomorphic:holo}{holomórfico} em todos $\Omega $ por
Critério de Morera: $\int_{\partial T}F = 0$ para cada
triângulo $ T \subseteq \Omega $ \emph{(dividir triângulos em $ I $
e empurre seus lados horizontais para fora do eixo
$\varepsilon $, usando uniforme continuidade)}.
(c) Deduzir: a função holomórfica no disco, real em um
diâmetro, satisfaz $ f(\bar z) = \overline{f(z)}$; e um
inconstante função holomórfica não pode ser valorizado em termos reais
qualquer subconjunto aberto não vazio de seu (\omterm{def:b3:topology:connected}{conectado}) domínio.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:holomorphic:12}
(A complexa equação pitagórica) Encontre todos os pares de
\emph{inteiro} funções com $ f^2 + g^2 = 1$.
(a) Mostre que $ h = f + \iu g $ é inteiro e \omterm{def:b3:modules:free}{livre} de zero, e
que toda função inteira \omterm{def:b3:modules:free}{livre} de zero é $\eu^{\varphi}$ para
alguns inteiros $\varphi $ \emph{($ h'/h $ é inteiro, portanto tem um
primitivo em forma de estrela $\C $; ajuste a constante e
mostrar $ h\eu^{-\varphi}$ é constante)}.
(b) Concluir $ f = \cos\varphi $, $ g = \sin\varphi $ com
$\varphi $ inteiro e verifique o inverso. Quais são os
soluções completas de $ f^2 + g^2 = 0$?
\end{exercise}

\section{Problema: o teorema fundamental da álgebra, duas vezes}

\begin{problem}[{Problema de fim de semana --- $\C $ é algebricamente
fechado: prova de d'Alembert, prova de Liouville e o
colheita}]\label{pb:b3:holomorphic:1}
Deixar $ P(z) = z^n + a_{n-1}z^{n-1} + \dots + a_0$, $ n \geq 1$.
Provamos isso duas vezes $ P $ tem uma raiz, então colete qual álgebra
está esperando desde \cref{ch:b3:galois}.

\medskip
\noindent\textbf{Part I --- Coercivity and the minimum.}
\begin{enumerate}
  \item Mostre isso $\abs{P(z)} \to +\infty $ como $\abs z \to
        \infty $: precisamente, $\abs{P(z)} \geq \frac12\abs z^n $
        para $\abs z \geq R_0$ adequado.
  \item Deduza isso $\abs P $ atinge um mínimo global em $\C $:
        há $ z_0$ com $\abs{P(z_0)} = \inf_\C\abs P $
        \emph{(compactação de um grande disco fechado,
        \cref{cor:b3:topology:heineborel})}.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- d'Alembert's descent.} Suponha,
por contradição, $ P(z_0) \neq 0$.
\begin{enumerate}[resume]
  \item Expandir $ Q(h) = P(z_0 + h)/P(z_0)$ como um polinômio em
        $ h $: $ Q(h) = 1 + c_kh^k + h^{k+1}S(h)$ com $ c_k \neq
        0$, $ k \geq 1$, $ S $ um polinômio.
  \item Escolha a direção da descida: para pequenos $ t > 0$,
        definir $ h = t\,\omega $ onde $\omega^k = -1/c_k $ (por que
        faz tal $\omega $ existe? --- \emph{provar} o
        existência de $ k $-ésimas raízes de qualquer número complexo via
        forma polar, independentemente do teorema ser
        comprovado). Mostrar
        \[
        \abs{Q(t\omega)} \leq 1 - t^k + C\,t^{k+1}
        \]
        para $ t $ pequeno, com uma constante explícita $ C $.
  \item Concluir $\abs{Q(t\omega)} < 1$ para pequeno $ t $ ---
        contrariando a minimalidade $\abs{P(z_0)}$. Por isso
        $ P(z_0) = 0$: \emph{todo complexo não constante
        polinômio tem raiz} (d'Alembert--Argand).
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Liouville's one-liner,
in full.}
\begin{enumerate}[resume]
  \item Escreva cuidadosamente a prova de
        \cref{cor:b3:holomorphic:liouville}: se $ P $ não tem
        raiz, verifique se $1/P $ é inteiro, limitado (quantificar,
        usando a questão 1), portanto constante, e conclua.
        Compare as duas provas: quais ingredientes cada uma
        usar? (A compactação aparece em ambos --- onde?)
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The harvest.}
\begin{enumerate}[resume]
  \item Mostre que cada $ P \in \C[X]$ de \omterm{def:b3:galois:extension}{grau} $ n $ divisões:
        $ P = c\prod_{i}(X - \alpha_i)^{m_i}$ com $\sum m_i =
        n $ \emph{(indução, divisão euclidiana por $(X -
        \alpha)$)}.
  \item Mostre que o irredutível polinômios de $\R[X]$ são
        os lineares e os quadráticos com negativo
        discriminante \emph{(par de raízes conjugadas)}; deduzir
        que todo polinômio real de \omterm{def:b3:galois:extension}{grau} ímpar tem um real
        raiz e forneça uma segunda prova teórica de ordem de
        esse último fato (teorema do valor intermediário) ---
        verificando se eles concordam $ X^3 - X - 1$.
  \item Deduza as dívidas que este livro pode agora pagar: (i) cada
        endomorfismo de uma dimensão finita diferente de zero
        $\C $-o espaço vetorial tem um autovalor, então todo complexo
        matriz tem um \omterm{thm:b3:modules:jordan}{Forma da Jordânia}
        (\cref{thm:b3:modules:jordan}); (ii) o campo
        $\bar\Q $ de \omterm{def:b3:galois:algebraic}{algébrico} números usados em
        \cref{rem:b3:galois:closureexamples} é de fato um
        fechamento \omterm{def:b3:galois:algebraic}{algébrico} de $\Q $.
  \item (Final) Identifique onde cada prova quebraria em um
        campo como $\Q(\iu)$: quais etapas usam a existência de
        $ k $-ésimas raízes (questão 4), e que usam compacidade
        ou completude (questões 2 e 6)? Conclua em cinco
        linhas: o teorema é genuinamente \emph{analítico} ---
        toda prova em algum lugar invoca o completude ou
        conectividade de $\R $ --- mesmo que sua declaração seja
        puramente \omterm{def:b3:galois:algebraic}{algébrico}.
\end{enumerate}

\medskip
\noindent\textbf{Part V --- The rigidity ladder of entire
functions.} Liouville é o primeiro degrau de uma escada; nós escalamos
isso.
\begin{enumerate}[resume]
  \item (\omterm{thm:b3:holomorphic:analytic}{Estimativas de Cauchy}) Da fórmula de Cauchy no
        círculo de raio $ r $ em volta $ a $, provar
        \[
        \bigl|f^{(n)}(a)\bigr| \leq
        \frac{n!\,\sup_{\abs{z-a}=r}\abs f}{r^n} ,
        \]
        e recuperar Liouville como o caso $ n = 1$, $ r \to
        \infty $.
  \item (Crescimento polinomial) Mostre que um todo $ f $ com
        $\abs{f(z)} \leq A + B\abs z^m $ para todos $ z $ é um
        polinômio de \omterm{def:b3:galois:extension}{grau} $\leq m $ \emph{(mate o Taylor
        coeficientes além $ m $ com a pergunta 11)}.
  \item (Parte real limitada) Mostre que um inteiro $ f $ com
        $\operatorname{Re}f $ limitado acima é constante
        \emph{(aplique Liouville a $\eu^{f}$)}.
  \item (Dupla periodicidade) Deixe $ f $ estar inteiro com $ f(z + 1)
        = f(z)$ e $ f(z + \iu) = f(z)$ para todos $ z $. Mostrar $ f $
        é constante. Concluir: uma ``elíptica'' não constante
        função deve ter singularidades --- o histórico
        os pólos da razão entram em análises complexas.
  \item (Intervalo \omterm{def:b3:topology:interior}{denso}) Mostre que o intervalo de um valor não constante
        toda a função é densa em $\C $: se $ f(\C)$ sente falta de um
        disco $ D(a, r)$, então $\frac1{f - a}$ é inteiro e
        limitado. (Picard provou que o alcance erra no \omterm{def:b3:rings:primemaximal}{máximo}
        \emph{um ponto}; \omterm{ex:b3:lebesgue:gamma}{densidade} é o nível de nossas ferramentas
        alcance.)
  \item (Apropriado $\Rightarrow $ polinômio) Suponha $ f $ é
        inteiro e $\abs{f(z)} \to \infty $ como $\abs z \to
        \infty $. Mostre: os zeros de $ f $ são finitos em número
        ($ z_1, \dots, z_p $, com multiplicidades $ m_i $); o
        quociente $ g = f/\prod(z - z_i)^{m_i}$ é inteiro e
        \omterm{def:b3:modules:free}{livre} de zero; $1/g $ tem crescimento polinomial, portanto
        (questão 12) é um polinômio, necessariamente constante
        (\omterm{def:b3:modules:free}{livre} de zero); concluir que $ f $ é um polinômio. Então
        entre funções inteiras, os polinômios são exatamente os
        apropriados --- $\eu^z $ falha na adequação ao longo
        $\R_-$.
\end{enumerate}

\medskip
\noindent\textbf{Part VI --- Harmonic shadows and a mean
value of Gauss.}
\begin{enumerate}[resume]
  \item Deixar $ f = u + \iu v $ ser \omterm{def:b3:holomorphic:holo}{holomórfico} em um conjunto aberto.
        Verifique isso $ u = \operatorname{Re}f $ satisfaz o
        propriedade de valor médio
        \[
        u(a) = \frac1{2\pi}\int_0^{2\pi}
        u\bigl(a + r\eu^{\iu\theta}\bigr)\,\dd\theta
        \]
        (parte real da fórmula de Cauchy) e deduza o
        princípio \omterm{def:b3:rings:primemaximal}{máximo} para $ u $ em um domínio limitado, com
        o mesmo conectividade prova quanto a $\abs f $.
  \item (Valor médio de Gauss) Para $ a \in \C $ e $ r > 0$ com
        $\abs a \neq r $, provar
        \[
        \frac1{2\pi}\int_0^{2\pi}
        \log\bigl|a - r\eu^{\iu\theta}\bigr|\,\dd\theta
        = \log\max\bigl(\abs a, r\bigr)
        \]
        \emph{(se $\abs a > r $: $ z \mapsto \log\abs{a - z}$
        é a parte real de um \omterm{def:b3:holomorphic:holo}{holomórfico} logaritmo em um
        \omterm{def:b3:topology:topology}{vizinhança} do disco fechado --- por que alguém
        existe? --- então a questão 17 se aplica; se $\abs a < r $:
        fator $\abs{a - r\eu^{\iu\theta}} = r\,\abs{1 -
        \frac ar\eu^{-\iu\theta}}$ e reutilizar o primeiro
        caso)}.
  \item (Mahler \omterm{def:b3:measure:measure}{medir}) Para $ P = c\prod_{i=1}^n(X -
        \alpha_i) \in \C[X]$, deduzir \emph{Fórmula de Jensen
        para polinômios}:
        \[
        \frac1{2\pi}\int_0^{2\pi}\log\bigl|P(\eu^{\iu\theta})
        \bigr|\,\dd\theta
        = \log\Bigl(\abs c\prod_{i}\max(1,
        \abs{\alpha_i})\Bigr) :
        \]
        a média geométrica de $\abs P $ no círculo unitário
        lê as raízes fora do disco. Verifique em $ P =
        X^2 - X $ e assim por diante $ P = 2X - 1$.
  \item (Números de Bernoulli) Defina os coeficientes $ B_n $ por
        $\frac{z}{\eu^z - 1} = \sum_{n\geq0}\frac{B_n}{n!}
        z^n $ aproximar $0$ (por que o lado esquerdo é analítico em
        $0$?). Derive a recorrência
        $\sum_{k=0}^{n}\binom{n+1}kB_k = 0$ ($ n \geq 1$) de
        $(\eu^z - 1)\cdot\frac z{\eu^z-1} = z $, calcular $ B_0,
        \dots, B_6$ e mostrar $ B_{2k+1} = 0$ para $ k \geq 1$
        \emph{(a função $\frac z{\eu^z-1} + \frac z2$ é
        mesmo)}. Esses números custarão cada $\zeta(2k)$ em
        \cref{ch:b3:residues}.
  \item (Realidade) Mostre que uma função inteira tomando real
        valores em $\R $ satisfaz $ f(\bar z) =
        \overline{f(z)}$ em todos os lugares \emph{(compare Taylor
        coeficientes em $0$, ou aplique o teorema da identidade
        para $ z \mapsto \overline{f(\bar z)}$)}; deduzir novamente
        que raízes não reais de polinômios reais vêm
        pares conjugados (emparelhamento da questão 8, reprovado
        analiticamente).
  \item (A moral) Monte a escada de rigidez: limitada
        $\Rightarrow $ constante; limitado polinomialmente
        $\Rightarrow $ polinomial; apropriado $\Rightarrow $
        polinomial; faltando um disco $\Rightarrow $ constante;
        duplamente periódico $\Rightarrow $ constante. Contraste em um
        parágrafo curto com real $\mathcal C^\infty $
        funções (funções de colisão,
        \cref{thm:b3:lp:regularization}): por que
        holomorfia, uma condição puramente local, impõem
        lei e ordem?
\end{enumerate}

\medskip
\noindent\textbf{Part VII --- Last harvest.}
\begin{enumerate}[resume]
  \item (Desigualdade de Landau) Para $ P = \sum_{k=0}^na_kX^k $,
        provar o valor médio $\frac1{2\pi}\int_0^{2\pi}
        \abs{P(\eu^{\iu\theta})}^2\dd\theta =
        \sum_k\abs{a_k}^2$ (ortogonalidade do
        $\eu^{\iu k\theta}$), então, usando o limite pontual
        $\log t \leq t - 1$ comparar os meios de
        $\log\abs P^2$ e $\abs P^2$, deduza da pergunta
        19 que
        \[
        \abs c\prod_{i}\max\bigl(1, \abs{\alpha_i}\bigr)
        \;\leq\;
        \Bigl(\sum_{k=0}^{n}\abs{a_k}^2\Bigr)^{1/2} :
        \]
        o produto das raízes fora do disco unitário é
        controlado pelo $\ell^2$ tamanho dos coeficientes
        (lidar com raízes \emph{sobre} o círculo aplicando o
        desigualdade para $ P(rX)$ e deixando $ r \to 1$). Verifique
        está ligado $ X^2 - X $.
  \item (Os números de Bernoulli crescem fatorialmente) Mostre que
        raio de convergência de $\sum\frac{B_n}{n!}z^n $ é
        \emph{exatamente} $2\pi $: pelo menos $2\pi $ porque
        $ z/(\eu^z - 1)$ estende holomorficamente para
        $ D(0, 2\pi)$, no \omterm{def:b3:rings:primemaximal}{máximo} $2\pi $ porque a soma seria
        caso contrário, fique limitado perto $2\pi\iu $, onde
        $\abs{z/(\eu^z-1)} \to \infty $. Deduzir
        \[
        \limsup_{k\to\infty}
        \Bigl(\frac{\abs{B_{2k}}}{(2k)!}\Bigr)^{1/2k}
        = \frac1{2\pi} :
        \]
        Os números de Bernoulli crescem fatorialmente. Admitindo $ B_{12}
        = -\frac{691}{2730}$ (a recorrência da questão 20,
        correr mais), comparar $\abs{B_{12}} \approx 0.25311$
        com a previsão mais nítida $2\,(2k)!/(2\pi)^{2k}
        \approx 0.25305$ no $ k = 6$ --- quatro dígitos correspondentes
        de uma lei assintótica que \cref{ch:b3:residues} vontade
        provar exatamente, via $\zeta(2k)$.
  \item (As raízes se movem continuamente) Deixar $(P_j)$ ser mônico de
        \omterm{def:b3:galois:extension}{grau} $ n $ com coeficientes convergindo para aqueles de
        (mônico) $ P $. Prove o \emph{Cauchy ligado}: cada
        raiz de um mônico $ Q = X^n + \sum_{k<n}q_kX^k $
        satisfaz $\abs\alpha \leq 1 + \max_k\abs{q_k}$;
        deduzir que as raízes do $ P_j $ fique em um fixo
        compactar conjunto, e, extraindo convergente
        subsequências de vetores raiz e passando para o
        limite na fatoração da questão 7, que o
        multiconjuntos raiz de $ P_j $ convergem para o de $ P $.
        Mostre finalmente isso continuidade é o melhor que se pode
        diga: para $ P_\varepsilon = X^2 - 2X + 1 +
        \varepsilon $, uma perturbação de tamanho $\varepsilon =
        10^{-4}$ move a raiz dupla $1$ por $10^{-2}$ ---
        Hëxpoente mais antigo $\frac1m $ em um $ m$- raiz dobrada,
        nunca Lipschitz: numericamente, raízes múltiplas custam
        metade dos dígitos.
\end{enumerate}
\end{problem}
