\chapter{Representações de grupos finitos}\label{ch:b3:representations}

Para entender um grupo abstrato, faça-o atuar em um espaço vetorial
e usar álgebra linear --- autovalores, traços, produtos internos ---
na ação. Este programa, \emph{\omterm{def:b3:representations:rep}{representação} teoria}, é
surpreendentemente eficaz para grupos finitos ao longo $\C $: todo
\omterm{def:b3:representations:rep}{representação} se divide em \omterm{def:b3:representations:rep}{irredutível} aqueles (Maschke), os
\omterm{def:b3:representations:rep}{irredutível} aqueles são presos por seus \emph{personagens}
(vestígios) e o personagens satisfazer relações de ortogonalidade que
tornar os cálculos mecânicos. O capítulo constrói esse cálculo
e suas primeiras obras-primas --- tabelas de caracteres do pequeno
grupos --- e o problema do fim de semana produz um teorema muito além
o alcance da teoria dos grupos \omterm{def:b3:groups:simple}{simples}: Burnside $ p^aq^b $ teorema.
Por todo, $ G $ é um grupo finito e todos os espaços vetoriais são
de dimensão finita $\C $.

\section{Representações, Maschke, Schur}

\begin{definition}\label{def:b3:representations:rep}
UM \emph{representação}\index{representação} de $ G $ é um
morfismo $\rho \colon G \to GL(V)$ por um $\C $-espaço vetorial $ V $;
$\dim V $ é o seu \emph{\omterm{def:b3:galois:extension}{grau}}. Um subespaço $ W \subseteq V $ é
\emph{invariante} se $\rho(g)W \subseteq W $ para todos $ g $;
restrição faz $ W $ um \emph{sub-representação}. $\rho $ é
\emph{irredutível}\index{irredutível} se $ V \neq
0$ e seus únicos subespaços invariantes são $0$ e $ V $. UM
\emph{morfismo} entre $(\rho, V)$ e $(\sigma, W)$ é linear
$ f\colon V \to W $ com $ f\rho(g) = \sigma(g)f $ para todos $ g $
(equivariância); bijetivo $ f $ são \emph{isomorfismos}.
\end{definition}

\begin{example}\label{ex:b3:representations:examples}
(a) \omterm{def:b3:galois:extension}{Grau} $1$: morfismos $ G \to \C^\times $.
(b) O \emph{representação regular}\index{representação regular}: $ V = \C^G $ com base $(e_h)_{h \in G}$,
$\rho(g)e_h = e_{gh}$; \omterm{def:b3:galois:extension}{grau} $\abs G $.
(c) Uma ação de permutação de $ G $ em um conjunto finito $ X $ dá o
\emph{permutação \omterm{def:b3:representations:rep}{representação}} sobre $\C^X $: $\rho(g)e_x =
e_{g\cdot x}$.
(d) $ S_n $ atua em $\C^n $ permutando coordenadas; o hiperplano
$\{\sum x_i = 0\}$ é invariável: o \emph{padrão
\omterm{def:b3:representations:rep}{representação}}, de \omterm{def:b3:galois:extension}{grau} $ n - 1$.
\end{example}

\begin{theorem}[Máscara]\label{thm:b3:representations:maschke}
Todo subespaço invariante $ W $ de um \omterm{def:b3:representations:rep}{representação} $(V, \rho)$
admite um complemento invariante. Consequentemente cada \omterm{def:b3:representations:rep}{representação}
é uma soma direta de \omterm{def:b3:representations:rep}{irredutível} uns
(\emph{semisimplicidade}).\index{Teorema de Maschke}
\end{theorem}

\begin{proof}
Deixar $ p \colon V \to V $ ser \emph{qualquer} projeção com imagem $ W $
(escolha qualquer complemento). Média do grupo:
\[
\tilde p = \frac1{\abs G}\sum_{g \in G} \rho(g)\,p\,\rho(g)^{-1}.
\]
Cada termo mapeia $ V $ em $ W $ ($ W $ invariante) e corrige $ W $
pontualmente: para $ w \in W $, $\rho(g)^{-1}w \in W $, $ p $ corrige isso,
e $\rho(g)$ traz de volta --- então $\tilde p $ é novamente um
projeção em $ W $. É equivalente: para $ h \in G $,
$\rho(h)\tilde p\rho(h)^{-1}$ reindexa a mesma soma. Por isso $\ker
\tilde p $ é um complemento invariante de $ W $. Iterando no
somas (dimensão finita) se decompõem $ V $ em irredutíveis.
\end{proof}

\begin{theorem}[Lema de Schur]\label{thm:b3:representations:schur}
Deixar $ f \colon V \to W $ seja um morfismo de \emph{\omterm{def:b3:representations:rep}{irredutível}}
representações. Então $ f = 0$ ou $ f $ é um isomorfismo; e se
$(V,\rho) = (W,\sigma)$, então $ f = \lambda\,\mathrm{id}$ para alguns
$\lambda \in \C $. Por isso
$\dim\operatorname{Hom}_G(V, W)$ é $1$ se $ V \cong W $, $0$
de outra forma.\index{Lema de Schur}
\end{theorem}

\begin{proof}
$\ker f $ e $\operatorname{im} f $ são invariantes (equivariância),
então cada um é $0$ ou tudo: ou $ f = 0$, ou $ f $ é injetivo
com imagem completa. Se $ V = W $: $ f $ tem um autovalor $\lambda $ ($\C $
\omterm{def:b3:galois:closure}{algebricamente fechado}); $ f - \lambda\,\mathrm{id}$ é um
morfismo não injetivo $ V \to V $, por isso $0$. Para a dimensão
contar quando $ V \cong W $: corrigindo um isomorfismo $ u $, qualquer morfismo
$ f $ dá o endomorfismo $ u^{-1}f = \lambda\,\mathrm{id}$: $ f =
\lambda u $.
\end{proof}

\section{Personagens e ortogonalidade}

\begin{definition}\label{def:b3:representations:character}
O \emph{personagem}\index{personagem} de $(V, \rho)$ é
$\chi_\rho(g) = \operatorname{tr}\rho(g)$. Isso satisfaz
$\chi_\rho(e) = \dim V $, $\chi_\rho(hgh^{-1}) = \chi_\rho(g)$
(os traços são invariantes à conjugação): os caracteres são \emph{aula
funções}\index{aula
funções} --- elementos do espaço
$\mathcal{CF}(G)$ de funções constantes em \omterm{ex:b3:groups:actions}{aulas de conjugação},
equipado com o produto interno Hermitiano
\[
\langle \varphi, \psi\rangle = \frac1{\abs G}\sum_{g \in G}
\overline{\varphi(g)}\,\psi(g).
\]
\end{definition}

\begin{proposition}\label{prop:b3:representations:charbasics}
$\rho(g)$ é diagonalizável com raízes de autovalores da unidade;
$\chi_\rho(g^{-1}) = \overline{\chi_\rho(g)}$, e
$\abs{\chi_\rho(g)} \leq \chi_\rho(e)$ com igualdade se
$\rho(g)$ é um escalar. Personagens adicione somas diretas: $\chi_{V
\oplus W} = \chi_V + \chi_W $.
\end{proposition}

\begin{proof}
$\rho(g)^N = \mathrm{id}$ para $ N = \abs G $ (Lagrange): $\rho(g)$
aniquila $ X^N - 1$, dividido com raízes \omterm{def:b3:groups:simple}{simples}, então é
diagonalizável com autovalores $\lambda_i \in \mu_N $. Então
$\chi(g^{-1}) = \sum \lambda_i^{-1} = \sum\bar\lambda_i =
\overline{\chi(g)}$, e $\abs{\chi(g)} = \abs{\sum\lambda_i}
\leq \dim V $, com igualdade na desigualdade triangular se todos
$\lambda_i $ são iguais, ou seja,\ $\rho(g) = \lambda\,\mathrm{id}$.
Aditividade em somas diretas: traços de bloco.
\end{proof}

\begin{lemma}\label{lem:b3:representations:homtrace}
Deixar $(V, \rho)$, $(W, \sigma)$ ser representações. A média
operador ligado $\operatorname{Hom}(W, V)$,
\[
c(A) = \frac1{\abs G}\sum_{g}\rho(g)\,A\,\sigma(g)^{-1},
\]
é uma projeção sobre $\operatorname{Hom}_G(W, V)$, e para o
mapa $\Phi_g \colon A \mapsto \rho(g)A\sigma(g)^{-1}$ um tem
$\operatorname{tr}\Phi_g =
\chi_\rho(g)\,\overline{\chi_\sigma(g)}$.
\end{lemma}

\begin{proof}
$ c(A)$ é equivariante (reindexar a soma como em Maschke), e $ c $
corrige mapas equivariantes (cada termo é igual $ A $): $ c $ é um
projeção com imagem $\operatorname{Hom}_G(W, V)$. Rastreamento: em
bases, $\Phi_g(A) = BAC $ com $ B = \rho(g)$, $ C = \sigma(g)^{-1}$;
com base $(E_{kl})$ de matrizes, $ BE_{kl}C = \sum_{m,n}
b_{mk}c_{ln}E_{mn}$, então o coeficiente de $ E_{kl}$ em
$\Phi_g(E_{kl})$ é $ b_{kk}c_{ll}$: $\operatorname{tr}\Phi_g =
\sum_{k,l}b_{kk}c_{ll} = \operatorname{tr}(B)
\operatorname{tr}(C) = \chi_\rho(g)\chi_\sigma(g^{-1})$, e
$\chi_\sigma(g^{-1}) = \overline{\chi_\sigma(g)}$.
\end{proof}

\begin{theorem}[Primeiras relações de ortogonalidade]
\label{thm:b3:representations:orthogonality}
Deixar $\rho, \sigma $ ser \emph{\omterm{def:b3:representations:rep}{irredutível}}. Então\index{ortogonalidade
de personagens}
\[
\langle\chi_\sigma, \chi_\rho\rangle =
\begin{cases}
1 & \text{if } \rho \cong \sigma,\\
0 & \text{otherwise:}
\end{cases}
\]
o \omterm{def:b3:representations:rep}{irredutível} personagens formar uma família ortonormal em
$\mathcal{CF}(G)$.
\end{theorem}

\begin{proof}
O traço de uma projeção é a dimensão da sua imagem:
\[
\dim\operatorname{Hom}_G(W, V) = \operatorname{tr} c
= \frac1{\abs G}\sum_g \operatorname{tr}\Phi_g
= \frac1{\abs G}\sum_g \chi_\rho(g)\overline{\chi_\sigma(g)}
= \langle \chi_\sigma, \chi_\rho\rangle,
\]
e o lema de Schur avalia o lado esquerdo para $\delta_{\rho \cong
\sigma}$.
\end{proof}

\begin{corollary}\label{cor:b3:representations:multiplicity}
Decompor $ V \cong \bigoplus_i V_i^{\oplus m_i}$ em distinto
irredutíveis ($ V_i \not\cong V_j $). Então $ m_i = \langle
\chi_{V_i}, \chi_V\rangle $: as multiplicidades --- daí o
\omterm{def:b3:representations:rep}{representação} até o isomorfismo --- são determinados pelo
\omterm{def:b3:representations:character}{personagem}. Além disso $\langle \chi_V, \chi_V\rangle = \sum_i
m_i^2$; em particular $ V $ é \omterm{def:b3:representations:rep}{irredutível} se $\langle\chi_V,
\chi_V\rangle = 1$.
\end{corollary}

\begin{proof}
$\chi_V = \sum m_i\chi_{V_i}$
(\cref{prop:b3:representations:charbasics}); pegue produtos internos
com cada um $\chi_{V_i}$ e usar ortonormalidade. Dois
representações com igual personagens têm multiplicidades iguais,
portanto, são isomórficos.
\end{proof}

\begin{theorem}[A representação regular]
\label{thm:b3:representations:regular}
Deixar $\chi_1, \dots, \chi_r $ seja o distinto \omterm{def:b3:representations:rep}{irredutível}
personagens, de \omterm{def:b3:galois:extension}{graus} $ n_i = \chi_i(e)$. O \omterm{def:b3:groups:normal}{normal}
\omterm{def:b3:representations:rep}{representação} se decompõe com multiplicidades $ m_i = n_i $;
consequentemente
\[
\sum_{i=1}^{r} n_i^2 = \abs G,
\qquad
\sum_i n_i\chi_i(g) = 0 \quad (g \neq e).
\]
\end{theorem}

\begin{proof}
O \omterm{def:b3:groups:normal}{normal} \omterm{def:b3:representations:character}{personagem}: $\chi_{\mathrm{reg}}(g) =
\#\{h : gh = h\}$, que é $\abs G $ para $ g = e $ e $0$
de outra forma. Então $ m_i = \langle \chi_i,\chi_{\mathrm{reg}}\rangle =
\frac1{\abs G}\overline{\chi_i(e)}\,\abs G = n_i $. Avaliando
$\chi_{\mathrm{reg}} = \sum n_i\chi_i $ no $ e $ e em $ g \neq e $
fornece as duas identidades exibidas.
\end{proof}

\begin{theorem}\label{thm:b3:representations:numberirr}
O \omterm{def:b3:representations:rep}{irredutível} personagens formar uma ortonormal \emph{base} de
$\mathcal{CF}(G)$: o número de representações irredutíveis
é igual ao número de \omterm{ex:b3:groups:actions}{aulas de conjugação} de $ G $.
\end{theorem}

\begin{proof}
Resta apenas a completude: deixe $ f \in \mathcal{CF}(G)$ ser
ortogonal a cada $\chi_i $; nós mostramos $ f = 0$. Por um
\omterm{def:b3:representations:rep}{representação} $(V,\rho)$, definir $ T_{f,\rho} = \frac{1}{\abs G}
\sum_g \overline{f(g)}\,\rho(g)$. É equivalente: para $ h \in
G $,
\[
\rho(h)T_{f,\rho}\rho(h)^{-1} = \frac1{\abs G}\sum_g
\overline{f(g)}\rho(hgh^{-1})
= \frac1{\abs G}\sum_{g'}\overline{f(h^{-1}g'h)}\rho(g') =
T_{f,\rho}
\]
($ f $ é um função de classe). Se $\rho $ é \omterm{def:b3:representations:rep}{irredutível} de \omterm{def:b3:galois:extension}{grau}
$ n $, Schur dá $ T_{f,\rho} = \lambda\,\mathrm{id}$ com
\[
\lambda = \frac{\operatorname{tr}T_{f,\rho}}{n}
= \frac{1}{n\abs G}\sum_g \overline{f(g)}\,\chi_\rho(g)
= \frac1n\,\langle f, \chi_\rho\rangle = 0 .
\]
Então $ T_{f,\rho} = 0$ em cada \omterm{def:b3:representations:rep}{irredutível}, portanto (somas diretas) em
\emph{todo} \omterm{def:b3:representations:rep}{representação} --- em particular no \omterm{def:b3:groups:normal}{normal}.
Aplique-o ao vetor base $ e_e $: $0 = T_{f,\mathrm{reg}}e_e =
\frac1{\abs G}\sum_g\overline{f(g)}e_g $, forçando cada
$\overline{f(g)} = 0$. Assim, a família ortonormal
$(\chi_i)$ vãos $\mathcal{CF}(G)$, cuja dimensão é a
número de \omterm{ex:b3:groups:actions}{aulas de conjugação}.
\end{proof}

\begin{corollary}[Ortogonalidade da coluna]
\label{cor:b3:representations:column}
Para $ g, h \in G $:
\[
\sum_{i=1}^{r}\overline{\chi_i(g)}\,\chi_i(h) =
\begin{cases}
\abs{Z_G(g)} & \text{if } g, h \text{ are conjugate},\\
0 & \text{otherwise.}
\end{cases}
\]
\end{corollary}

\begin{proof}
Deixar $ g_1, \dots, g_r $ representam as aulas, $ c_j $ a aula
tamanhos. O $ r \times r $ matriz $ U_{ij} =
\sqrt{c_j/\abs G}\;\chi_i(g_j)$ tem linhas ortonormais
(\cref{thm:b3:representations:orthogonality} escrito em classe:
$\sum_j \frac{c_j}{\abs G}\chi_i(g_j)\overline{\chi_{i'}(g_j)} =
\delta_{ii'}$), ou seja\ $ UU^* = I $; uma matriz quadrada com $ UU^* =
I $ também tem $ U^*U = I $: as colunas são ortonormais, que são descompactadas para
a identidade exibida ($\abs G/c_j = \abs{Z_G(g_j)}$, o
relação \omterm{def:b3:groups:action}{órbita-estabilizador} para conjugação).
\end{proof}

\begin{proposition}[Personagens unidimensionais; elevação]
\label{prop:b3:representations:onedim}
(um) $ G $ é abeliano se todos os seus representações irredutíveis tem
\omterm{def:b3:galois:extension}{grau} $1$; o número de diplomas $1$ personagens de qualquer $ G $ é
$[G : D(G)]$ (eles são os personagens da abelianização).
(b) Se $ N \trianglelefteq G $, o representações irredutíveis de
$ G/N $ levantar (compor com $ G \to G/N $) exatamente para aqueles
representações irredutíveis de $ G $ cujo kernel contém $ N $.
\end{proposition}

\begin{proof}
(a) Se $ G $ é abeliano, cada classe é um singleton: $ r = \abs G $,
e $\sum n_i^2 = \abs G $ força tudo $ n_i = 1$; inversamente se
tudo $ n_i = 1$, o \omterm{ex:b3:representations:examples}{representação regular} é uma soma de
unidimensionais, então $\rho_{\mathrm{reg}}(G)$ é
simultaneamente diagonalizável, portanto comutativa, e
$\rho_{\mathrm{reg}}$ é fiel: $ G $ abeliano. Grau-$1$
representações são morfismos $ G \to \C^\times $ com abeliano
alvo: eles levam em consideração $ G^{\mathrm{ab}} = G/D(G)$
(\cref{exo:b3:groups:9}) e distinto personagens do abeliano
$ G^{\mathrm{ab}}$ número $\abs{G^{\mathrm{ab}}}$ (tem isso
muitas turmas, todos os \omterm{def:b3:galois:extension}{graus} $1$). (b) Composição com o
a projeção preserva a irredutibilidade (subespaços invariantes
correspondem) e um \omterm{def:b3:representations:rep}{representação} trivial em $ N $ fatores através
o quociente (\cref{thm:b3:groups:firstiso}).
\end{proof}

\section{Tabelas de caracteres}

\begin{definition}\label{def:b3:representations:table}
O \emph{tabela de caracteres}\index{tabela de caracteres} de $ G $ é o
$ r \times r $ matriz $\bigl(\chi_i(g_j)\bigr)$: linhas indexadas por
\omterm{def:b3:representations:rep}{irredutível} personagens, colunas por \omterm{ex:b3:groups:actions}{aulas de conjugação} (com seus
tamanhos exibidos). As linhas são ortonormais para o produto ponderado,
colunas ortogonais (\cref{cor:b3:representations:column}): o
tabela está severamente sobredeterminada, o que a torna
computável.
\end{definition}

\begin{example}[A mesa de $ S_3$]\label{ex:b3:representations:s3}
Aulas: $ e $ (tamanho 1), transposições (3), $3$-ciclos (2); então $ r
= 3$ irredutíveis, de \omterm{def:b3:galois:extension}{graus} $ n_i $ com $\sum n_i^2 = 6$: $1, 1,
2$. \omterm{def:b3:galois:extension}{Grau} $1$: trivial $\mathbf 1$ e assinatura $\varepsilon $.
A última linha segue da ortogonalidade da coluna (ou de
$\chi_{\mathrm{std}} = \chi_{\mathrm{perm}} - \mathbf 1$):
\[
\begin{array}{c|ccc}
S_3 & e & (1\,2)\ [3] & (1\,2\,3)\ [2]\\
\hline
\mathbf 1 & 1 & 1 & 1\\
\varepsilon & 1 & -1 & 1\\
\chi_{\mathrm{std}} & 2 & 0 & -1
\end{array}
\]
Verificar: $\langle\chi_{\mathrm{std}},\chi_{\mathrm{std}}\rangle =
\frac{1}{6}(4 + 0 + 2) = 1$: \omterm{def:b3:representations:rep}{irredutível}.
\end{example}

\begin{example}[A mesa de $ S_4$]\label{ex:b3:representations:s4}
Aulas: $ e $ [1], transposições [6], transposições duplas [3],
$3$-ciclos [8], $4$-ciclos [6]: cinco irredutíveis, $\sum n_i^2 =
24$ com dois graus-$1$ de ($\mathbf 1, \varepsilon $; $[S_4 :
D(S_4)] = [S_4 : A_4] = 2$): \omterm{def:b3:galois:extension}{graus} $1, 1, 2, 3, 3$. O
grau-$2$ levanta de $ S_4/V \cong S_3$
(\cref{prop:b3:representations:onedim}(b), $ V $ o grupo Klein);
\omterm{def:b3:galois:extension}{grau} $3$: o padrão \omterm{def:b3:representations:rep}{representação} e sua reviravolta por
$\varepsilon $:
\[
\begin{array}{c|ccccc}
S_4 & e\,[1] & (1\,2)\,[6] & (1\,2)(3\,4)\,[3] &
(1\,2\,3)\,[8] & (1\,2\,3\,4)\,[6]\\
\hline
\mathbf 1 & 1 & 1 & 1 & 1 & 1\\
\varepsilon & 1 & -1 & 1 & 1 & -1\\
\chi_2 & 2 & 0 & 2 & -1 & 0\\
\chi_{\mathrm{std}} & 3 & 1 & -1 & 0 & -1\\
\varepsilon\chi_{\mathrm{std}} & 3 & -1 & -1 & 0 & 1
\end{array}
\]
($\chi_{\mathrm{std}}(g) = \operatorname{fix}(g) - 1$;
$\chi_2$ avalia o $ S_3$-tabela na imagem de cada turma
moda $ V $.) Todas as verificações de ortogonalidade de linhas e colunas são aprovadas ---
executar dois deles é \cref{exo:b3:representations:3}é o aquecimento.
\end{example}

\begin{method}\label{met:b3:representations:table}
Para construir um \omterm{def:b3:representations:table}{tabela de caracteres}: (1) lista \omterm{ex:b3:groups:actions}{aulas de conjugação} e tamanhos;
(2) contar graus-$1$ personagens através de $ G/D(G)$ e escreva-os; (3)
encontre os \omterm{def:b3:galois:extension}{graus} restantes de $\sum n_i^2 = \abs G $ (pequeno
combinatória inteira); (4) obter barato irredutíveis: levante de
quocientes, subtrair $\mathbf 1$ da permutação personagens
(verificar $\langle\chi,\chi\rangle = 1$), multiplique conhecido personagens
por grau-$1$ uns; (5) terminar linhas desconhecidas por coluna
ortogonalidade --- cada coluna é ortogonal às colunas
já preenchido e coluna $ e $ carrega os \omterm{def:b3:galois:extension}{graus}. Verifique
tudo com uma varredura completa de ortogonalidade.
\end{method}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:b3:representations:1}
(a) Mostre que \omterm{def:b3:representations:rep}{irredutível} personagens de $\Z/n\Z $ são os
$\chi_k(\bar m) = \eu^{2\iu\pi km/n}$, $ k = 0, \dots, n-1$ e
escreva o \omterm{def:b3:representations:table}{tabela de caracteres} de $\Z/4\Z $.
(b) Verifique ambas as relações de ortogonalidade nele --- e reconheça
a matriz: onde este livro já viu isso antes?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b3:representations:2}
Deixar $ G $ agir em um conjunto finito $ X $ e $\chi $ seja o \omterm{def:b3:representations:character}{personagem} de
a permutação \omterm{def:b3:representations:rep}{representação} $\C^X $.
(a) Mostrar $\chi(g) = \abs{\operatorname{Fix}_X(g)}$ e
$\langle \mathbf 1, \chi\rangle = \#\{\text{orbits}\}$ ---
Lema de contagem de Burnside (\cref{exo:b3:groups:5}) é um \omterm{def:b3:representations:character}{personagem}
computação.
(b) Suponha que a ação seja transitiva, então $\chi = \mathbf 1 +
\psi $. Mostre isso $\psi $ é \omterm{def:b3:representations:rep}{irredutível} se a ação for
\emph{$2$-transitivo} (transitivo em pares ordenados de distintos
pontos). \emph{(Calcular $\langle\chi,\chi\rangle $ como o número
de \omterm{def:b3:groups:action}{órbitas} sobre $ X \times X $.)}
(c) Conclua que o padrão \omterm{def:b3:representations:rep}{representação} de $ S_n $ ($ n \geq

\section{Problema: Burnside \texorpdfstring{$ p^aq^b $}{paqb}
teorema}

\begin{problem}[{Problema de fim de semana --- solubilidade de grupos de
ordem $ p^aq^b $}]\label{pb:b3:representations:1}
Burnside provou em 1904 que todo grupo cuja ordem tenha no \omterm{def:b3:rings:primemaximal}{máximo}
dois fatores principais é \omterm{def:b3:groups:derived}{solucionável} --- uma declaração sobre resumo
grupos cujas únicas provas conhecidas durante meio século passaram por
\omterm{def:b3:representations:character}{personagem} teoria. Este problema constrói a prova completa,
montagem \cref{ch:b3:groups} (solvabilidade),
\cref{ch:b3:modules} (finitamente gerado $\Z $-módulos) e isso
capítulo. Por todo, $\chi_1, \dots, \chi_r $ são os \omterm{def:b3:representations:rep}{irredutível}
personagens de $ G $, $ n_i = \chi_i(e)$.

\medskip
\noindent\textbf{Part I --- \omterm{def:b3:galois:algebraic}{Algébrico} integers.} Um
\emph{\omterm{def:b3:galois:algebraic}{algébrico} inteiro} é uma raiz de \emph{Mônica} polinômio
de $\Z[X]$.
\begin{enumerate}
  \item Mostre isso $\alpha $ é um \omterm{def:b3:galois:algebraic}{algébrico} inteiro se o anel
        $\Z[\alpha]$ é um finitamente gerado $\Z $-módulo.
  \item Deduza que o \omterm{def:b3:galois:algebraic}{algébrico} inteiros formam um subanel de
        $\C $. \emph{(Se $\Z[\alpha], \Z[\beta]$ são finitamente
        gerado, assim é $\Z[\alpha, \beta]$ e submódulos de
        finitamente gerado $\Z $-módulos são finitamente gerado,
        por \cref{thm:b3:modules:submodule} e uma apresentação
        argumento --- ou diretamente: um submódulo de $\Z^n $ é grátis
        de classificação $\leq n $.)}
  \item Mostre que um \emph{racional} \omterm{def:b3:galois:algebraic}{algébrico} inteiro é um
        inteiro. \emph{(Teorema da raiz racional.)}
  \item Mostre que cada \omterm{def:b3:representations:character}{personagem} valor $\chi(g)$ é um
        \omterm{def:b3:galois:algebraic}{algébrico} inteiro.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The class-sum relations.} Consertar um
\omterm{def:b3:representations:rep}{irredutível} $(V, \rho)$ de \omterm{def:b3:galois:extension}{grau} $ n $ e \omterm{def:b3:representations:character}{personagem} $\chi $. Por um
aula de conjugação $ C $, deixar $ S_C = \sum_{g \in C}\rho(g) \in
\mathcal L(V)$.
\begin{enumerate}[resume]
  \item Mostre isso $ S_C $ é equivariante, portanto $ S_C =
        \omega_C\,\mathrm{id}$ com
        \[
        \omega_C = \frac{\abs C\,\chi(g_C)}{n}
        \qquad (g_C \in C \text{ any representative}).
        \]
  \item Mostre isso $ S_CS_{C'} = \sum_{C''}
        a_{CC'C''}\,S_{C''}$ onde $ a_{CC'C''} \in \N $ conta,
        para um fixo $ z \in C''$, os pares $(x, y) \in C \times
        C'$ com $ xy = z $. Deduza que o $\omega_C $ satisfazer
        $\omega_C\,\omega_{C'} = \sum_{C''}
        a_{CC'C''}\,\omega_{C''}$.
  \item Conclua que cada $\omega_C $ é um \omterm{def:b3:galois:algebraic}{algébrico} inteiro.
        \emph{(O $\Z $-módulo gerado por $1$ e o
        $\omega_C $ é um finitamente gerado anel; aplicar pergunta
        Critério de 1 --- mostrar mais precisamente $ M = \Z\text{-span}
        (1, (\omega_C)_C)$ satisfaz $\omega_{C_0} M \subseteq
        M $ e usar um determinante/truque de Cayley-Hamilton, ou
        argumento do subanel da questão 2.)}
  \item Deduzir \emph{\omterm{thm:b3:galois:finitefields}{Frobenius} divisibilidade}: $ n_i $ divide
        $\abs G $ para cada \omterm{def:b3:representations:rep}{irredutível} \omterm{def:b3:galois:extension}{grau}.
        \emph{(Calcular $\frac{\abs G}{n} = \frac{\abs G}{n}
        \langle\chi,\chi\rangle = \sum_C \omega_C\,
        \overline{\chi(g_C)}$: um \omterm{def:b3:galois:algebraic}{algébrico} inteiro que é
        racional.)}
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Burnside's simplicity criterion.}
\begin{enumerate}[resume]
  \item Deixar $\chi $ ser \omterm{def:b3:representations:rep}{irredutível} de \omterm{def:b3:galois:extension}{grau} $ n $ e $ C $ uma aula
        com $\gcd(\abs C, n) = 1$. Usando Bézout e perguntas
        4-7, mostre que $\frac{\chi(g_C)}{n}$ é um \omterm{def:b3:galois:algebraic}{algébrico}
        inteiro.
  \item Suponha, além disso $0 < \abs{\chi(g_C)} < n $. Mostre que isso é
        impossível: o \omterm{def:b3:galois:algebraic}{algébrico} inteiro $\alpha =
        \chi(g_C)/n $ tem todo o seu \emph{conjugados} --- o
        números $\alpha_\sigma = \frac1n\sigma(\chi(g_C))$ para
        $\sigma \in \operatorname{Gal}(\Q(\zeta_{\abs G})/\Q)$,
        cada uma em média $ n $ raízes da unidade --- do módulo
        $\leq 1$,
        então o produto $ N = \prod_\sigma\alpha_\sigma $ é um
        racional \omterm{def:b3:galois:algebraic}{algébrico} inteiro com $0 < \abs N < 1$ ---
        justifique cada afirmação, citando
        \cref{thm:b3:galois:cyclotomicirred} para o \omterm{def:b3:galois:galois}{Galois}
        grupo e o fato de que $\sigma $ permuta raízes de
        unidade. Conclua: ou $\chi(g_C) = 0$ ou $\rho(g_C)$
        é escalar (\cref{prop:b3:representations:charbasics}).
  \item (Critério de Burnside) Deixe $ C \neq \{e\}$ seja um
        aula de conjugação de \emph{poder principal} tamanho $ p^k > 1$,
        e suponha $ G $ é \omterm{def:b3:groups:simple}{simples} não-abeliano. Coluna
        ortogonalidade na coluna de $ C $ contra a coluna de
        $ e $ dá $1 + \sum_{i \geq 2} n_i\chi_i(g_C) = 0$. Mostrar
        que alguns não triviais $\chi_i $ com $ p \nmid n_i $ tem
        $\chi_i(g_C) \neq 0$ \emph{(de outra forma $\frac1p $ seria
        um \omterm{def:b3:galois:algebraic}{algébrico} inteiro)}; pela pergunta 10, $\rho_i(g_C)$
        é escalar; derivar uma contradição com a simplicidade
        \emph{(o conjunto de $ g $ com $\rho_i(g)$ escalar é um
        subgrupo \omterm{def:b3:groups:normal}{normal}; use a fidelidade de
        \cref{exo:b3:representations:6})}. Concluir: \emph{não
        grupo não-abeliano \omterm{def:b3:groups:simple}{simples} tem um aula de conjugação de primo
        tamanho de potência $> 1$.}
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The theorem.}
\begin{enumerate}[resume]
  \item Deixar $\abs G = p^aq^b $ com $ a + b \geq 1$. Se $ G $ é
        \omterm{def:b3:groups:simple}{simples}, mostre que é abeliano: escolha $ z \neq e $ no
        \omterm{ex:b3:groups:actions}{centro} de um Sylow $ q $-subgrupo
        (\cref{thm:b3:groups:pfixed}) e considere o tamanho de
        é aula de conjugação $[G : Z_G(z)]$, um poder de $ p $
        (por que?); aplique a questão 11.
  \item Conclua por indução em $\abs G $: \emph{cada grupo de
        ordem $ p^aq^b $ é \omterm{def:b3:groups:derived}{solucionável}} (Burnside). Por que o
        quebra de argumento para três primos --- e deve, dado
        $\abs{A_5} = 2^2\cdot3\cdot5$?
\end{enumerate}

\medskip
\noindent\textbf{Part V --- The \omterm{def:b3:representations:table}{tabela de caracteres} of $ A_5$.} O
menor grupo que o teorema de Burnside não pode tocar merece sua completa
retrato; tudo abaixo usa apenas este capítulo mais
\cref{exo:b3:groups:11}.
\begin{enumerate}[resume]
  \item Lembrar de \cref{exo:b3:groups:11} os cinco conjugação
        aulas de $ A_5$: $\{e\}$, o $15$ duplo
        transposições, o $20$ três ciclos e duas turmas
        de $12$ cinco ciclos cada, representados por $ c =
        (1\,2\,3\,4\,5)$ e $ c^2$. Explique por que os cinco ciclos
        dividido em dois $ A_5$-classes embora formem uma única
        $ S_5$-aula.
  \item Mostre que o \omterm{def:b3:representations:rep}{irredutível} \omterm{def:b3:galois:extension}{graus} de $ A_5$ são exatamente
        $1, 3, 3, 4, 5$: usar $\sum_in_i^2 = 60$ com $ r = 5$
        aulas e o fato de que $ A_5$ é \omterm{prop:b3:galois:perfect}{perfeito} ($ D(A_5) =
        A_5$), então o trivial \omterm{def:b3:representations:character}{personagem} é o único linear;
        em seguida, elimine todos os outros multisets \emph{(escrever $59$
        como uma soma de quatro quadrados de inteiros $\geq 2$: verifique
        acontece de apenas uma maneira com todos os resumos plausíveis
        \omterm{def:b3:galois:extension}{graus})}.
  \item Deixar $\pi $ seja a permutação \omterm{def:b3:representations:character}{personagem} de $ A_5$ sobre
        $\{1, \dots, 5\}$: $\pi(g) = \#\operatorname{Fix}(g)$,
        com valores $5, 1, 2, 0, 0$ nas cinco aulas. Calcular
        $\langle\pi, \mathbf 1\rangle $ e $\langle\pi,
        \pi\rangle $, e deduza que $\chi_4 = \pi - \mathbf 1$
        é \omterm{def:b3:representations:rep}{irredutível} de \omterm{def:b3:galois:extension}{grau} $4$, com valores $4, 0, 1, -1,
        -1$.
  \item Mesmo jogo no $10$ pares não ordenados $\{i, j\}$: o
        contagens de ponto fixo são $10, 2, 1, 0, 0$. Calcular
        $\langle\pi_{10}, \pi_{10}\rangle $, $\langle\pi_{10},
        \mathbf 1\rangle $ e $\langle\pi_{10}, \chi_4\rangle $,
        deduzir a decomposição $\pi_{10} = \mathbf 1 + \chi_4
        + \chi_5$, e obter o \omterm{def:b3:representations:rep}{irredutível} $\chi_5$ de \omterm{def:b3:galois:extension}{grau}
        $5$ com valores $5, 1, -1, 0, 0$.
  \item Os dois restantes irredutíveis $\chi_2, \chi_3$ tem
        \omterm{def:b3:galois:extension}{grau} $3$. Ortogonalidade da coluna (cada não identidade
        coluna contra a coluna de identidade, e cada coluna com
        em si) determina seus valores fora dos cinco ciclos:
        mostrar $\chi_2(g) = \chi_3(g) = -1$ em dobro
        transposições e $0$ em três ciclos.
  \item Nas aulas de cinco ciclos, definir $ x = \chi_2(c)$ e $ y =
        \chi_2(c^2)$; simetria permite tomar $\chi_3(c) = y $,
        $\chi_3(c^2) = x $. Da coluna de $ c $ emparelhado com
        a coluna de identidade e com a coluna de $ c^2$, derivar
        $ x + y = 1$ e $ xy = -1$ (e verifique o valor $ x^2 +
        y^2 = 3$ dado pela coluna de $ c $ consigo mesmo), daí
        \[
        \{x, y\} = \Bigl\{\frac{1 + \sqrt5}2,\ \frac{1 -
        \sqrt5}2\Bigr\} :
        \]
        a proporção áurea e seu conjugado. Monte o
        completo \omterm{def:b3:representations:table}{tabela de caracteres} de $ A_5$.
  \item Execute as verificações: a norma da linha de $\chi_2$ é $1$ (usar
        $\varphi^2 + \bar\varphi^2 = 3$), $\langle\chi_2,
        \chi_3\rangle = 0$, e \omterm{thm:b3:galois:finitefields}{Frobenius} divisibilidade (questão
        8) para todos os cinco \omterm{def:b3:galois:extension}{graus}. Onde na mesa você
        \emph{ver} uma diferença com $ S_5$, todos cujos
        \omterm{def:b3:representations:character}{personagem} valores são inteiros racionais?
  \item Deduza apenas da tabela que $ A_5$ é \omterm{def:b3:groups:simple}{simples}: um
        subgrupo \omterm{def:b3:groups:normal}{normal} é uma união de \omterm{ex:b3:groups:actions}{aulas de conjugação}
        contendo $ e $ cuja cardinalidade divide $60$ --- verificar
        que nenhuma subsoma adequada de $1 + 15 + 20 + 12 + 12$
        contendo o termo $1$ divide $60$. Verifique com
        critério da questão 11: verificar se nenhuma classe de $ A_5$
        tem tamanho de potência principal $> 1$.
  \item (Coda icosaédrica) $ A_5$ é o grupo de rotação do
        icosaedro, e o grau-$3$ representações são os
        duas ações geométricas em $\R^3$. Verifique o rastreamento
        identidade: uma rotação por ângulo $\theta $ tem rastro $1 +
        2\cos\theta $, e $1 + 2\cos\frac{2\pi}5 =
        \frac{1+\sqrt5}2 = \varphi $. Explique sem qualquer
        cálculo por que o outro grau-$3$ \omterm{def:b3:representations:character}{personagem} deve
        carrega o valor conjugado: o \omterm{def:b3:galois:galois}{Grupo Galois} de
        $\Q(\sqrt5)/\Q $ atua em geral \omterm{def:b3:representations:table}{tabela de caracteres}
        (entrada), permutando o \omterm{def:b3:representations:rep}{irredutível} personagens.
\end{enumerate}

\noindent\textbf{Part VI --- Complements: a central bound and
the tensor square.}
\begin{enumerate}[resume]
  \item (Mais nítido que uma divisibilidade) Vamos $\chi $ ser \omterm{def:b3:representations:rep}{irredutível}
        de \omterm{def:b3:galois:extension}{grau} $ n $. Mostre isso $\abs{\chi(z)} = n $ para cada
        $ z \in Z(G)$ \emph{(Lema de Schur: $\rho(z)$ é um
        escalar, de ordem finita)}, e deduza de
        $\langle\chi, \chi\rangle = 1$ o limite
        \[
        n^2 \leq [G : Z(G)] .
        \]
        Mostre que os grupos não-abelianos de ordem $8$ ter
        \omterm{def:b3:representations:rep}{irredutível} \omterm{def:b3:galois:extension}{graus} $1, 1, 1, 1, 2$ \emph{(cinco
        \omterm{ex:b3:groups:actions}{aulas de conjugação}; escrever $8$ como uma soma de cinco quadrados)}
        e alcançar a igualdade $4 = [G : Z(G)]$; verifique o limite
        ligado $ A_5$, cujo \omterm{ex:b3:groups:actions}{centro} é trivial.
  \item (Tensor quadrado de $\chi_2$) Para $ g $ de ordem finita,
        $\rho(g)$ é diagonalizável com raiz da unidade
        autovalores; deduzir o \omterm{def:b3:representations:character}{personagem} fórmulas
        \[
        \chi_{\operatorname{Sym}^2 V}(g)
        = \frac{\chi(g)^2 + \chi(g^2)}2,
        \qquad
        \chi_{\Lambda^2 V}(g)
        = \frac{\chi(g)^2 - \chi(g^2)}2 .
        \]
        Aplique-os a $\chi_2$ de $ A_5$ \emph{(observação $ g^2$ corre
        através da aula de $ c^2$ quando $ g $ passa por isso
        de $ c $ e inversamente)}: mostrar $\Lambda^2\chi_2 =
        \chi_2$ e $\operatorname{Sym}^2\chi_2 = \mathbf 1 +
        \chi_5$, por isso
        \[
        \chi_2\otimes\chi_2 = \mathbf 1 + \chi_2 + \chi_5 .
        \]
        Interpretar $\Lambda^2\chi_2 = \chi_2$ geometricamente através
        o produto vetorial em $\R^3$.
  \item (Auditoria final da tabela) Verifique numericamente: coluna
        ortogonalidade entre as duas colunas de cinco ciclos
        ($\varphi\bar\varphi = -1$), o valor $5 =
        \abs{Z_{A_5}(c)}$ para a coluna de $ c $ contra si mesmo,
        e o desaparecimento do regular \omterm{def:b3:representations:character}{personagem} $\sum_i
        n_i\chi_i(g) = 0$ em cada uma das quatro não-identidades
        colunas da tabela.
\end{enumerate}
\end{problem}
