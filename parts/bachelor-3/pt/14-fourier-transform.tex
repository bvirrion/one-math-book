\chapter{A Transformada de Fourier}\label{ch:b3:fouriertransform}

A série de Fourier decompõe sinais periódicos em sinais discretos
harmônicos; o Fourier \emph{transformar} faz o mesmo para
sinais em toda a linha, com um \omterm{def:b3:topology:continuity}{contínuo} de frequências. Isso
converte diferenciação em multiplicação, convolução em
produtos e Gaussianos em Gaussianos --- as razões pelas quais
resolve equações diferenciais, impulsiona o processamento de sinais e
provará o teorema do limite central em \cref{ch:b3:clt}. Isto
capítulo desenvolve o $L^1$ teoria (Riemann--Lebesgue,
inversão, injetividade), o \omterm{def:b3:fouriertransform:schwartz}{Classe Schwartz} onde o
transformar é um \omterm{prop:b3:galois:perfect}{perfeito} bijeção e a $ L^2$ teoria
(Plancherel: a transformada é, até uma constante, uma unidade
operador), com duas aplicações exemplares: a equação do calor,
resolvido de ponta a ponta no problema do fim de semana, e o Poisson
fórmula de soma. Convenção:
\[
\hat f(\xi) = \int_\R f(x)\,\eu^{-\iu\xi x}\,\dd x .
\]

\section{A transformação em \texorpdfstring{$ L^1$}{L1}}

\begin{proposition}\label{prop:b3:fouriertransform:basic}
Para $ f \in L^1(\R)$: $\hat f $ é bem definido, limitado
($\norm{\hat f}_\infty \leq \norm f_1$), \omterm{def:b3:topology:continuity}{contínuo}, e:
\begin{enumerate}
  \item $\widehat{\tau_af}(\xi) = \eu^{-\iu a\xi}\hat f(\xi)$
        e $\widehat{\eu^{\iu ax}f}(\xi) = \hat f(\xi - a)$;
  \item $\widehat{f(\cdot/\lambda)}(\xi) = \lambda\hat
        f(\lambda\xi)$ para $\lambda > 0$;
  \item se $ xf \in L^1$, então $\hat f $ é $\mathcal C^1$ com
        $(\hat f)'(\xi) = \widehat{(-\iu x)f}(\xi)$;
  \item se $ f \in \mathcal C^1$ com $ f' \in L^1$ (e $ f \to
        0$ no $\pm\infty $, automático aqui), então
        $\widehat{f'}(\xi) = \iu\xi\hat f(\xi)$;
  \item $\widehat{f * g} = \hat f\,\hat g $ para $ f, g \in L^1$.
\end{enumerate}
\end{proposition}

\begin{proof}
Limite: $\abs{\hat f} \leq \int\abs f $. Continuidade: DCT
com dominador $\abs f $ (\cref{thm:b3:lebesgue:paramcont}).
(1), (2): substituições
(\cref{thm:b3:product:linearchange}). (3): diferenciação
sob o integral, dominador $\abs{xf}$
(\cref{thm:b3:lebesgue:paramdiff}). (4): primeiro, $ f(x) = f(0)
+ \int_0^xf'$ tem um limite em $\pm\infty $ ($ f' \in L^1$), que
deve ser $0$ ($ f \in L^1$); depois integre por partes em $[-A,
A]$ e deixe $ A \to \infty $. (5): Fubini, legítimo desde
$(x,y)\mapsto f(x - y)g(y)\eu^{-\iu\xi x}$ é absolutamente
\omterm{def:b3:lebesgue:l1}{integrável} (\cref{thm:b3:product:convolution}):
\[
\widehat{f*g}(\xi) = \iint f(x - y)g(y)\eu^{-\iu\xi(x - y)}
\eu^{-\iu\xi y}\dd x\,\dd y = \hat f(\xi)\,\hat g(\xi).
\]
\end{proof}

\begin{example}\label{ex:b3:fouriertransform:gaussian}
O Gaussiano: para $ a > 0$,\index{Gaussiana, transformada de Fourier}
\[
\widehat{\eu^{-ax^2}}(\xi) =
\sqrt{\frac\pi a}\;\eu^{-\xi^2/4a} :
\]
por \cref{exo:b3:lebesgue:7} (a ODE $ F' = -\frac\xi{2}F $
truque, redimensionado), ou por (3): $ g = \widehat{\eu^{-ax^2}}$
satisfaz $ g'(\xi) = -\frac{\xi}{2a}g(\xi)$ (integrar por
peças), $ g(0) = \sqrt{\pi/a}$. Gaussianas são pontos fixos de
a transformação até o dimensionamento --- a razão profunda pela qual eles governam o
teorema do limite central.
\end{example}

\begin{theorem}[Riemann--Lebesgue]
\label{thm:b3:fouriertransform:riemannlebesgue}
Para $ f \in L^1(\R)$: $\hat f(\xi) \to 0$ como $\abs\xi \to
\infty $. Por isso $\widehat{\phantom f} \colon L^1 \to \mathcal
C_0(\R)$ (\omterm{def:b3:topology:continuity}{contínuo} funções desaparecendo em
infinito).\index{Lema de Riemann-Lebesgue}
\end{theorem}

\begin{proof}
Para um indicador de um intervalo, $\widehat{\mathbf 1_{\intcc
ab}}(\xi) = \frac{\eu^{-\iu a\xi} - \eu^{-\iu b\xi}}{\iu\xi}
\to 0$; portanto, para funções degrau. Funções degrau são densas em
$ L^1$ (\cref{thm:b3:lp:density}(1) mais aproximação de
medida finita conjuntos por uniões finitas de intervalos;
\cref{exo:b3:measure:7}) e a transformação é
$\norm\cdot_\infty $-$\norm\cdot_1$ \omterm{def:b3:topology:continuity}{contínuo}: para $\norm{f -
s}_1 < \varepsilon $, $\limsup_{\abs\xi\to\infty}\abs{\hat
f(\xi)} \leq \varepsilon $.
\end{proof}

\section{Inversão e injetividade}

\begin{lemma}[Fórmula de multiplicação]
\label{lem:b3:fouriertransform:multiplication}
Para $ f, g \in L^1(\R)$: $\displaystyle\int \hat f\,g =
\int f\,\hat g $.
\end{lemma}

\begin{proof}
Ambos os lados iguais $\iint f(x)g(\xi)\eu^{-\iu x\xi}\dd
x\,\dd\xi $ (Tonelli--Fubini: a integral dupla do
valor absoluto é $\norm f_1\norm g_1$).
\end{proof}

\begin{theorem}[Inversão]\label{thm:b3:fouriertransform:inversion}
Deixar $ f \in L^1(\R)$.\index{Teorema da inversão de Fourier}
\begin{enumerate}
  \item (Somabilidade gaussiana) Para cada $ x $,
        \[
        (f * g_\varepsilon)(x) =
        \frac1{2\pi}\int_\R \hat f(\xi)\,
        \eu^{-\varepsilon\xi^2}\,\eu^{\iu x\xi}\,\dd\xi,
        \qquad\text{where } g_\varepsilon(y) =
        \frac{1}{2\sqrt{\pi\varepsilon}}\,
        \eu^{-y^2/4\varepsilon},
        \]
        e $ f * g_\varepsilon \to f $ em $ L^1$ como $\varepsilon
        \to 0$.
  \item Se além disso $\hat f \in L^1$, então para quase todos
        $ x $
        \[
        f(x) = \frac{1}{2\pi}\int_\R \hat
        f(\xi)\,\eu^{\iu x\xi}\,\dd\xi ,
        \]
        e $ f $ tem um \omterm{def:b3:topology:continuity}{contínuo} representante.
  \item (Injetividade) Se $\hat f = 0$ então $ f = 0$ a.e.
\end{enumerate}
\end{theorem}

\begin{proof}
(1) Correção $ x $ e aplicar
\cref{lem:b3:fouriertransform:multiplication} para $ f $ e
$ g(\xi) = \frac1{2\pi}\eu^{-\varepsilon\xi^2}\eu^{\iu x\xi}$:
por \cref{ex:b3:fouriertransform:gaussian} (com a modulação
regra),
\[
\hat g(y) = \frac1{2\pi}\sqrt{\frac\pi\varepsilon}\,
\eu^{-(y - x)^2/4\varepsilon} = g_\varepsilon(x - y),
\]
então $\frac1{2\pi}\int\hat f(\xi)\eu^{-\varepsilon\xi^2}
\eu^{\iu x\xi}\dd\xi = \int f(y)g_\varepsilon(x - y)\dd y =
(f*g_\varepsilon)(x)$. O $ g_\varepsilon $ são um
identidade aproximada: $ g_\varepsilon \geq 0$,
$\int g_\varepsilon = 1$ (integral gaussiana), concentrando-se em
$0$; a prova de \cref{thm:b3:lp:regularization}(2) aplica-se
literalmente (apenas $\int g_\varepsilon = 1$ e concentração foram
usado: para a cauda, $\int_{\abs y > \delta}g_\varepsilon \to
0$): $\norm{f * g_\varepsilon - f}_1 \to 0$.

(2) Se $\hat f \in L^1$: o lado direito de (1) converge, por
DCT (dominador $\abs{\hat f}$), para $\frac1{2\pi}\int\hat
f(\xi)\eu^{\iu x\xi}\dd\xi $ para \emph{todo} $ x $, e isso
função limite é \omterm{def:b3:topology:continuity}{contínuo} (DCT novamente). Por outro lado $ f
* g_\varepsilon \to f $ em $ L^1$, então ao longo de uma subsequência a.e.\
(\cref{thm:b3:lp:complete}): os dois limites concordam, a.e.

(3) $\hat f = 0$ faz o lado direito de (1) desaparecer: $ f *
g_\varepsilon = 0$ para todos $\varepsilon $, e $ f *
g_\varepsilon \to f $ em $ L^1$: $ f = 0$ a.e.
\end{proof}

\section{A classe Schwartz}

\begin{definition}\label{def:b3:fouriertransform:schwartz}
O \emph{Classe Schwartz}\index{Classe Schwartz} $\mathcal
S(\R)$ consiste no $\mathcal C^\infty $ funções $ f $ com
$\sup_x\abs{x^m f^{(n)}(x)} < \infty $ para todos $ m, n \geq 0$
(todas as derivadas decaem mais rápido que qualquer potência). Exemplos:
$\eu^{-ax^2}$, $\mathcal C_c^\infty $. Claramente $\mathcal S
\subseteq L^p $ para cada $ p $ (vinculado por $ C(1 + x^2)^{-1}$), e
$\mathcal S $ é estável sob derivadas, multiplicação por
polinômios e produtos.
\end{definition}

\begin{theorem}\label{thm:b3:fouriertransform:schwartz}
Os mapas da transformada de Fourier $\mathcal S(\R)$ bijetivamente em
em si, com inverso $\check g(x) = \frac1{2\pi}\int
g(\xi)\eu^{\iu x\xi}\dd\xi $.
\end{theorem}

\begin{proof}
Deixar $ f \in \mathcal S $. Iterando
\cref{prop:b3:fouriertransform:basic}(3), $(\hat f)^{(n)} =
\widehat{(-\iu x)^nf}$ (cada $ x^kf \in L^1$); iterando (4)
com $ h = (-\iu x)^nf \in \mathcal S $ (todos cujos
derivados são \omterm{def:b3:lebesgue:l1}{integrável}), $(\iu\xi)^m\hat h =
\widehat{h^{(m)}}$. Combinando,
\[
\abs{\xi^m\,(\hat f)^{(n)}(\xi)}
= \bigl|\widehat{\,h^{(m)}}(\xi)\bigr|
\leq \bigl\|\bigl((-\iu x)^nf\bigr)^{(m)}\bigr\|_1 < \infty
\]
uniformemente em $\xi $: $\hat f \in \mathcal S $. Desde $\hat f \in L^1$, inversão
(\cref{thm:b3:fouriertransform:inversion}(2)) detém
em todos os lugares (ambos os lados \omterm{def:b3:topology:continuity}{contínuo}): $\check{\hat f} = f $ e
simetricamente $\widehat{\check g} = g $ (a transformação de verificação
é $ g \mapsto \frac1{2\pi}\hat g(-\cdot)$, novamente preservando
$\mathcal S $): bijeção.
\end{proof}

\section{Plancherel e \texorpdfstring{$ L^2$}{L2}}

\begin{theorem}[Plancherel]\label{thm:b3:fouriertransform:plancherel}
Para $ f \in L^1 \cap L^2(\R)$:\index{Teorema de Plancherel}
\[
\norm{\hat f}_2^2 = 2\pi\,\norm f_2^2 .
\]
Consequentemente $\widehat{\phantom f}$ se estende exclusivamente a um
\omterm{def:b3:topology:continuity}{contínuo} mapa linear $\mathcal F \colon L^2(\R) \to L^2(\R)$
com $\norm{\mathcal Ff}_2 = \sqrt{2\pi}\norm f_2$; $\mathcal
F $ é bijetivo, com $\mathcal F^{-1} =
\frac1{2\pi}\,\mathcal F\circ\sigma $ onde $\sigma f =
f(-\cdot)$, e preserva produtos internos até o fator
$2\pi $.
\end{theorem}

\begin{proof}
Deixar $ f \in L^1\cap L^2$ e $ h = f * \tilde f $ com
$\tilde f(x) = \overline{f(-x)}$. Então $ h \in L^1$
(\cref{thm:b3:product:convolution}), $ h $ é \omterm{def:b3:topology:continuity}{contínuo} e
limitado (\cref{exo:b3:lp:6}: $ f, \tilde f \in L^2$), $ h(0) =
\int f\bar f = \norm f_2^2$, e $\hat h = \hat
f\,\widehat{\tilde f} = \hat f\,\overline{\hat f} =
\abs{\hat f}^2 \geq 0$ (calcular $\widehat{\tilde f} =
\overline{\hat f}$). Aplicar
\cref{thm:b3:fouriertransform:inversion}(1) para $ h $ no $ x = 0$:
\[
(h * g_\varepsilon)(0) = \frac1{2\pi}\int \hat
h(\xi)\,\eu^{-\varepsilon\xi^2}\dd\xi .
\]
Como $\varepsilon \to 0$: o lado esquerdo tende a $ h(0)$ ($ h $
\omterm{def:b3:topology:continuity}{contínuo} limitado: $(h*g_\varepsilon)(0) - h(0) = \int(h(-y)
- h(0))g_\varepsilon(y)\dd y \to 0$ dividindo pequeno/grande
$ y $); o lado direito aumenta para $\frac1{2\pi}\int\hat h $ por
MCT ($\hat h \geq 0$). Por isso $\frac1{2\pi}\int\abs{\hat f}^2 =
\norm f_2^2$, finito ou não \emph{a priori} --- e finito,
comprovando a adesão e a identidade.

Extensão: $ L^1\cap L^2 \supseteq \mathcal C_c $ é \omterm{def:b3:topology:interior}{denso} em
$ L^2$ (\cref{thm:b3:lp:density}); a transformação é
$\sqrt{2\pi}$-isométrico lá, então se estende exclusivamente a um
isometria constante $\mathcal F $ sobre $ L^2$
(\cref{thm:b3:complete:extension}). Inversão para $\mathcal S $
(\cref{thm:b3:fouriertransform:schwartz}) transferências pelo
mesmo \omterm{ex:b3:lebesgue:gamma}{densidade} (ambos os lados $ L^2$-contínuo): $\mathcal
F\bigl(\frac1{2\pi}\mathcal F(\sigma f)\bigr) = f $ sobre
$\mathcal S $, daí em diante $ L^2$: bijetividade. Produtos internos:
polarização da identidade da norma.
\end{proof}

\begin{theorem}[Somatório de Poisson]
\label{thm:b3:fouriertransform:poisson}
Deixar $ f \in \mathcal S(\R)$ (\omterm{def:b3:topology:continuity}{contínuo} $ f $ com $\abs{f} +
\abs{\hat f} \leq C(1 + \abs\cdot)^{-2}$
é suficiente). Então\index{Fórmula de soma de Poisson}
\[
\sum_{n\in\Z} f(n) \;=\; \sum_{k\in\Z}\hat f(2\pi k) .
\]
\end{theorem}

\begin{proof}
Deixar $ F(x) = \sum_{n\in\Z}f(x + n)$: a série converge
normalmente ligado compactos (decadência de $ f $), então $ F $ é contínuoe
é $1$-periódico. Seus coeficientes de Fourier (período $1$:
$ c_k(F) = \int_0^1F(t)\eu^{-2\iu\pi kt}\dd t $):
\[
c_k(F) = \sum_n\int_0^1 f(t + n)\,\eu^{-2\iu\pi kt}\dd t
= \int_\R f(t)\,\eu^{-2\iu\pi kt}\dd t = \hat f(2\pi k)
\]
(a convergência \omterm{def:b3:groups:normal}{normal} justifica o intercâmbio; a fase é
$1$-periódico). A série $\sum_k\abs{c_k(F)}$ converge
(decadência de $\hat f $), então a série de Fourier de $ F $ converge
normalmente; sua soma é um \omterm{def:b3:topology:continuity}{contínuo} funcionar com o mesmo
Coeficientes de Fourier como $ F $, portanto é igual $ F $ (injetividade em
o círculo: a diferença tem coeficientes zero, e
\cref{thm:b3:hilbert:fourier} dá zero em $ L^2$, portanto
em todos os lugares por continuidade). Avaliar em $ x = 0$.
\end{proof}

\begin{example}[A identidade teta]
\label{ex:b3:fouriertransform:theta}
Aplicando Poisson a $ f(x) = \eu^{-\pi tx^2}$ ($ t > 0$), cujo
transformar é $\hat f(\xi) = t^{-1/2}\eu^{-\xi^2/4\pi t}$
(\cref{ex:b3:fouriertransform:gaussian} com $ a = \pi t $):
\[
\sum_{n\in\Z}\eu^{-\pi n^2t}
= \frac1{\sqrt t}\sum_{k\in\Z}\eu^{-\pi k^2/t} :
\]
a equação funcional da função teta de Jacobi, chave para o
equação funcional de Riemann $\zeta $ --- e um espetacular
acelerador numérico: para $ t $ pequeno, o lado esquerdo converge
lentamente, o lado direito extremamente rápido.
\end{example}

\begin{method}\label{met:b3:fouriertransform:method}
Faixas de trabalho: $ L^1$ --- transformar definido pontualmente,
necessidades de inversão $\hat f \in L^1$; $\mathcal S $ --- tudo
é legal, prove aqui primeiro; $ L^2$ --- transformação definida por
\omterm{ex:b3:lebesgue:gamma}{densidade} (não pela integral!), \omterm{prop:b3:galois:perfect}{perfeito} simetria, \omterm{thm:b3:hilbert:parseval}{Parseval}
escrituração contábil. Para calcular uma transformação: reduza para a tabela
(indicador, exponencial, gaussiano) pelas regras de
\cref{prop:b3:fouriertransform:basic}; para provar uma identidade:
estabelecê-lo em $\mathcal S $ (ou $\mathcal C_c^\infty $) e
estender por \omterm{ex:b3:lebesgue:gamma}{densidade} e continuidade
(\cref{met:b3:lp:toolkit}); para resolver uma EDP ou EDO linear com
coeficientes constantes: transformar, dividir, inverter.
\end{method}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:b3:fouriertransform:1}
Calcule as transformadas de Fourier de: $\mathbf 1_{\intcc{-a}a}$;
$\eu^{-a\abs x}$ ($ a > 0$); a função da tenda $\max(0, 1 -
\abs x)$; $\frac1{x^2 + a^2}$ \emph{(use inversão no
segundo)}. Registre a tabela emergente.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b3:fouriertransform:2}
Deixar $ f \in L^1$. Expresse em termos de $\hat f $ as transformações
de: $ f(x - a)$, $ f(x)\cos(bx)$, $ f(ax + b)$,
$\overline{f(-x)}$, $(f * f)(x)$. Verifique cada regra no
Gaussiano.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:fouriertransform:3}
(a) Mostre que $\mathbf 1_{\intcc{-1}1} * \mathbf
1_{\intcc{-1}1}$ tem transformação $\bigl(\frac{2\sin\xi}\xi
\bigr)^2$, e deduza
$\int_\R\bigl(\frac{\sin\xi}\xi\bigr)^2\dd\xi = \pi $ por
Plancherel --- ou por inversão em $0$. Comparar
\cref{pb:b3:lebesgue:1}.
(b) Calcular $\int_\R\frac{\dd x}{(x^2+1)^2}$ via Plancherel
aplicado a $\eu^{-\abs x}$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:fouriertransform:4}
(\omterm{pb:b3:fouriertransform:1}{Núcleo de calor} álgebra) Com $ g_t(x) =
\frac1{2\sqrt{\pi t}}\eu^{-x^2/4t}$:
(a) verificar $\hat g_t(\xi) = \eu^{-t\xi^2}$;
(b) deduzir a lei do semigrupo $ g_t * g_s = g_{t+s}$ sem qualquer
computação integral;
(c) mostrar $\norm{g_t}_1 = 1$ e $\norm{g_t}_2^2 =
(8\pi t)^{-1/2}$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:fouriertransform:5}
Mostre que se $ f \in L^1$ é uniforme e real, $\hat f $ é par
e real; se $ f $ é estranho e real, $\hat f $ é estranho e puramente
imaginário. O que faz $\hat f(0)$ calcular? Deduza isso $ f \geq
0$ forças $\norm{\hat f}_\infty = \hat f(0) = \int f $ e
interpretar para \omterm{ex:b3:lebesgue:gamma}{densidades} de probabilidade (\cref{ch:b3:clt}: um
função característica tem módulo $\leq 1$, alcançado em
$0$).
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b3:fouriertransform:6}
(Não-surjetividade) Mostre que $\widehat{\phantom f}\colon L^1
\to \mathcal C_0$ é injetivo e \omterm{def:b3:topology:continuity}{contínuo}, mas \emph{não}
sobrejetivo, em três etapas.
(i) Injetividade (\cref{thm:b3:fouriertransform:inversion}) e
continuidade ($\norm{\hat f}_\infty \leq \norm f_1$), e
$\mathcal C_0$ é um espaço Banach (fechado em
$\norm\cdot_\infty $).
(ii) Se o mapa fosse sobrejetivo, seria bijetivo, e
o teorema do mapeamento aberto (\cref{thm:b3:banach:openmapping})
daria uma constante $ C $ com $\norm f_1 \leq C\norm{\hat
f}_\infty $ para todos $ f \in L^1$.
(iii) Contradiga isso com $ f_n(x) = \frac{\sin
x}{x}\cdot\frac{\sin(x/n)}{x/n}$: sua transformação é (até
constantes) a convolução $\mathbf 1_{\intcc{-1}1} *
\mathbf 1_{\intcc{-1/n}{1/n}}$-tipo trapézio --- mostrar
$\norm{\hat f_n}_\infty \leq \pi $ uniformemente, enquanto
$\norm{f_n}_1 \geq c\ln n $ contando os arcos de
$\frac{\abs{\sin x}}x $ sobre $[1, n]$ (onde o segundo fator é
delimitado abaixo), como em \cref{thm:b3:banach:fourierdiverge}.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:fouriertransform:7}
(Suavidade $\leftrightarrow $ dicionário de decaimento) Prove: $ f \in
L^1$ com $\hat f(\xi) = O(\abs\xi^{-k-1-\delta})$ para alguns
$\delta > 0$ implica $ f $ tem um $\mathcal C^k $ representante.
Por outro lado $ f \in \mathcal C^k_c $ implica $\hat f(\xi) =
O(\abs\xi^{-k})$. Ilustre ambas as direções na tenda
função.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b3:fouriertransform:8}
(Desigualdade de Heisenberg) Para $ f \in \mathcal S(\R)$ real com
$\norm f_2 = 1$, provar
\[
\Bigl(\int x^2f(x)^2\dd x\Bigr)\cdot
\Bigl(\frac1{2\pi}\int \xi^2\abs{\hat f(\xi)}^2\dd\xi\Bigr)
\;\geq\; \frac14 ,
\]
com igualdade para gaussianos. \emph{(Escrever $1 = \int f^2 =
-\int x\,(f^2)'$ por partes, unidas por Cauchy-Schwarz, e
converter $\norm{f'}_2$ por Plancherel.)} Interpretação: uma
sinal e seu espectro não podem ser concentrados.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:fouriertransform:9}
Justificar \cref{ex:b3:fouriertransform:theta} em detalhes

\begin{exercise}[$\star\star $]\label{exo:b3:fouriertransform:10}
(Funções limitadas por banda) Deixe $ f \in L^2(\R)$ com $\mathcal Ff $
suportado em $\intcc{-\pi}\pi $. Mostre isso $ f $ tem um
representante estendendo cada um dos valores que é recuperável
a partir de amostras: prove a \emph{Interpolação de Shannon} no
inteiros,
\[
f(x) = \sum_{n\in\Z} f(n)\,
\frac{\sin\bigl(\pi(x - n)\bigr)}{\pi(x - n)}
\quad\text{in } L^2,
\]
expandindo $\mathcal Ff $ na \omterm{def:b3:topology:basis}{base} de Fourier de
$ L^2(\intcc{-\pi}\pi)$ (\cref{thm:b3:hilbert:fourier}) e
transformando de volta termo por termo.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:fouriertransform:11}
(A transformação como um operador de ordem quatro) Em $\mathcal
S(\R)$, deixar $\mathcal F f = \hat f $.
(a) Usando a fórmula de inversão, mostre $(\mathcal F^2f)(x) =
2\pi\,f(-x)$, e deduza $\mathcal F^4 = (2\pi)^2\,
\mathrm{id}$.
(b) Deduza que todo autovalor de $\mathcal F $ sobre
$\mathcal S $ pertence a $\{\pm\sqrt{2\pi},
\pm\iu\sqrt{2\pi}\}$, e exibem uma função própria para
$+\sqrt{2\pi}$ \emph{(cuja função deste capítulo é
proporcional à sua própria transformação?)}.
(c) Mostre que funções pares satisfazem $\mathcal F^2f = 2\pi
f $ e estranhos $\mathcal F^2f = -2\pi f $; produzir um
função própria para o valor próprio $-\iu\sqrt{2\pi}$ de
$ x\eu^{-x^2/2}$ calculando sua transformada (diferenciar
a transformada de Gauss).
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:fouriertransform:12}
(Autocorrelação e lema de Wiener) Para $ f \in L^2(\R)$
definir $\tilde f(x) = \overline{f(-x)}$ e o
\emph{autocorrelação} $ A_f = f * \tilde f $.
(a) Mostre que $ A_f $ é um limite \omterm{def:b3:topology:continuity}{contínuo} função com
$ A_f(0) = \norm f_2^2 \geq \abs{A_f(x)}$ para todos $ x $
(\cref{exo:b3:lp:6} e Cauchy-Schwarz).
(b) Mostre, primeiro para $ f \in L^1\cap L^2$, que
$\widehat{A_f} = \abs{\hat f\,}^2 \geq 0$: o
autocorrelação tem transformada não negativa --- espectros de
autocorrelações são espectros de potência.
(c) Deduzir a identidade $\int_\R\abs{\hat
f(\xi)}^2\eu^{\iu x\xi}\,\dd\xi = 2\pi A_f(x)$ (inversão;
justificar sua aplicabilidade quando $\hat f \in L^2$ tem
$\abs{\hat f}^2 \in L^1$) e avaliá-lo para $ f = \mathbf
1_{\intcc{-1/2}{1/2}}$ no $ x = 0$: recuperar
$\int_\R\bigl(\frac{\sin u}u\bigr)^2\dd u = \pi $.
\end{exercise}

\section{Problema: a equação do calor na reta}

\begin{problem}[{Problema de fim de semana --- $\partial_tu =
\partial^2_{xx}u $, resolvido de ponta a ponta}]
\label{pb:b3:fouriertransform:1}
O calor se espalha; a equação $\partial_tu = \partial_{xx}^2u $
diz que é \omterm{ex:b3:lebesgue:gamma}{densidade} se difunde a uma taxa dada pelo local
curvatura do perfil de temperatura. Nós resolvemos o Cauchy
problema em $\R $ --- dado $ f $, encontrar $ u(t, x)$ para $ t > 0$ com
$ u(0, \cdot) = f $ --- provar que a solução é notável
propriedades e veja por que o tempo não pode ser revertido. Por todo,
$ g_t(x) = \frac1{2\sqrt{\pi t}}\eu^{-x^2/4t}$ é o
\emph{núcleo de calor}\index{núcleo de calor} e $ u(t, \cdot) = g_t *
f $.

\medskip
\noindent\textbf{Part I --- Deriving the kernel.} Trabalho
formalmente primeiro: suponha $ u(t, \cdot) \in \mathcal S $ resolve
a equação, e deixe $\hat u(t, \xi)$ seja a transformação em
$ x $.
\begin{enumerate}
  \item Mostrar (formalmente) $\partial_t\hat u = -\xi^2\hat u $,
        por isso $\hat u(t,\xi) = \eu^{-t\xi^2}\hat f(\xi)$ e
        reconhecer $ u(t) = g_t * f $
        (\cref{exo:b3:fouriertransform:4}). Isto motiva o
        \emph{definição} de $ u $; tudo agora está provado
        diretamente, para $ f \in \mathcal C_b(\R)$ (limitado
        \omterm{def:b3:topology:continuity}{contínuo}) ou $ f \in L^p $.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Verification.}
\begin{enumerate}[resume]
  \item Mostre isso para $ t > 0$, $ u(t, x) = \int
        g_t(x-y)f(y)\dd y $ está bem definido para $ f \in \mathcal
        C_b $, e isso $ u $ é $\mathcal C^\infty $ em $(t, x)$
        sobre $\intoo0\infty\times\R $ \emph{(diferenciar em
        a integral; dominar as derivadas gaussianas localmente
        uniformemente em $(t,x)$)}.
  \item Verificar $\partial_tg_t = \partial^2_{xx}g_t $ por direto
        cálculo e dedução $\partial_tu =
        \partial^2_{xx}u $ para $ t > 0$.
  \item (Condição inicial) Mostre que para $ f \in \mathcal
        C_b $, $ u(t, x) \to f(x)$ como $ t \to 0^+$, uniformemente em
        compactos \emph{(identidade aproximada: divisão $\abs y
        \leq \delta $, $\abs y > \delta $)}; para $ f \in L^p $
        ($ p < \infty $), mostrar $\norm{u(t) - f}_p \to 0$.
  \item (Suavização instantânea) \omterm{thm:b3:complete:completion}{Conclusão}: mesmo para limites meramente limitados
        \omterm{def:b3:topology:continuity}{contínuo} $ f $, a solução é $\mathcal C^\infty $
        para cada $ t > 0$ --- o calor apaga instantaneamente
        rugosidade. Calcular $ u(t, \cdot)$ explicitamente para $ f =
        \mathbf 1_{\intoo0\infty}$ (uma função de erro) e
        esboce seu perfil para três valores de $ t $.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Structural properties.}
\begin{enumerate}[resume]
  \item (Positividade e comparação) Se $ f \geq 0$ então $ u > 0$
        para todos $ t > 0$ (estritamente, a menos que $ f = 0$ e.e.); se
        $ f_1 \leq f_2$ então $ u_1 \leq u_2$. Um ponto frio aquece
        instantaneamente: comente.
  \item (Conservação) Para $ f \in L^1$: $\int u(t, x)\dd x =
        \int f $ para todos $ t $ \emph{(Tonelli)} --- o calor total é
        conservado.
  \item (Dissipação) Para $ f \in L^1\cap L^2$, mostrar através
        Plancherel que $ t \mapsto \norm{u(t)}_2$ é
        não crescente, estritamente a menos que $ f = 0$ e calcular
        seu limite como $ t \to \infty $. Mostrar além disso
        $\norm{u(t)}_\infty \leq \frac{\norm
        f_1}{2\sqrt{\pi t}} \to 0$: o calor se espalha e morre.
  \item (Singularidade, $ L^2$ aula) Deixe $ u $ ser uma solução com
        $ u(t) \in L^2$ para todos $ t $, $ u \in \mathcal
        C^1(\intoo0\infty, L^2)$ no sentido natural e
        $ u(t) \to f $ em $ L^2$ como $ t\to0$; admitindo que o
        transform converte-o em $\partial_t\hat u =
        -\xi^2\hat u $ pontualmente a.e.\ em $\xi $ para a.e.\ $ t $
        \emph{(justificado por testes contra $\mathcal
        C_c^\infty $ em $\xi $ --- descreva isso)}, mostrar
        $\hat u(t,\xi) = \eu^{-t\xi^2}\hat f(\xi)$, portanto
        singularidade nesta classe.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The arrow of time.}
\begin{enumerate}[resume]
  \item Mostre que o \emph{para trás} problema está mal colocado:
        para que a solução exista no momento $-s $ ($ s > 0$) com
        dados $ f $ de vez em quando $0$ --- ou seja,\ para $ f = g_s * h $ para
        tem uma solução $ h \in L^2$ --- é necessário que
        $\eu^{s\xi^2}\hat f(\xi) \in L^2$: uma decadência extrema
        condição em $\hat f $. Exibir uma suavidade explícita $ f
        \in L^2$ para o qual não existe solução retroativa em qualquer
        intervalo de tempo: pegue a função com $\hat f(\xi) =
        \eu^{-\abs\xi}$ --- identificar $ f $
        (\cref{exo:b3:fouriertransform:1}) e mostrar
        $\eu^{s\xi^2}\eu^{-\abs\xi} \notin L^2$ para cada $ s >
        0$.
  \item (Suavização vs.\ informação) Explique resumidamente
        parágrafo, usando as questões 5, 9 e 10, por que o calor
        semigrupo $(f \mapsto g_t * f)_{t\geq0}$ é injetivo
        mas não sobrejetivo em $ L^2$, e por que isso expressa
        a irreversibilidade da difusão.
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Shannon's sampling theorem.} Um
função $ f \in L^2(\R)$ é \emph{banda limitada} para $\Omega $
se $\hat f = 0$ a.e.\ fora $\intcc{-\Omega}\Omega $; escrever
$ PW_\Omega $ (Espaço Paley-Wiener) para estes.
\begin{enumerate}[resume]
  \item Mostre que cada $ f \in PW_\Omega $ concorda a.e. com
        o $\mathcal C^\infty $ função
        $\frac1{2\pi}\int_{-\Omega}^{\Omega}\hat
        f(\xi)\eu^{\iu x\xi}\,\dd\xi $ (justificar suavidade e
        a identificação), com todas as derivadas limitadas:
        a limitação de banda é uma forma extrema de regularidade.
        De agora em diante $ f $ denota este representante.
  \item Expandir $\hat f \in L^2(\intcc{-\Omega}\Omega)$ no
        \omterm{def:b3:topology:basis}{base} de Fourier desse intervalo e identifique o
        coeficientes como \emph{amostras} de $ f $:
        \[
        \hat f(\xi) = \frac\pi\Omega\sum_{n\in\Z}
        f\Bigl(\frac{n\pi}\Omega\Bigr)\,
        \eu^{-\iu n\pi\xi/\Omega}
        \quad\text{in } L^2(\intcc{-\Omega}\Omega) .
        \]
  \item Deduza o \emph{teorema da amostragem}: para $ f \in
        PW_\Omega $,
        \[
        f(x) = \sum_{n\in\Z}f\Bigl(\frac{n\pi}
        \Omega\Bigr)\,\operatorname{sinc}(\Omega x - n\pi),
        \qquad \operatorname{sinc}t = \frac{\sin t}t,
        \]
        com convergência em $ L^2(\R)$ e uniformemente em $\R $
        \emph{(injete a série de perguntas 13 no
        fórmula de inversão e calcule o elementar
        integrante)}: um sinal com banda limitada é inteiramente
        determinado por seus valores em uma grade de passo
        $\pi/\Omega $ --- a taxa de Nyquist.
  \item Mostre que as funções $ x \mapsto
        \operatorname{sinc}(\Omega x - n\pi)$, $ n \in \Z $,
        formar uma família ortogonal em $ L^2(\R)$ com constante
        norma $\sqrt{\pi/\Omega}$, e deduza a energia
        identidade $\norm f_2^2 =
        \frac\pi\Omega\sum_n\abs{f(n\pi/\Omega)}^2$.
  \item (Aliasing) Exibir um diferente de zero $ g \in PW_{2\Omega}$
        desaparecendo em cada ponto de amostragem $\frac{n\pi}\Omega $
        \emph{(considerar $ g(x) =
        \sin(\Omega x)\operatorname{sinc}(\Omega x)$ e
        verifique sua banda)}: amostragem abaixo da taxa de Nyquist
        perde informações --- dois sinais diferentes podem compartilhar
        todas as amostras: o efeito estroboscópico da roda de carroça,
        matematizado.
  \item (\omterm{def:b3:galois:extension}{Graus} de liberdade) Utilizando as questões 14 a 15, justifique
        a regra de engenharia: uma banda de sinal limitada a
        $\Omega $ cuja energia é essencialmente transportada por um
        janela de tempo de duração $ T $ é descrito por
        aproximadamente $\frac{\Omega T}\pi $ amostras reais ---
        tornar "essencialmente" preciso através da energia
        identidade e a cauda $\sum_{\abs{n\pi/\Omega} >
        T/2}$.
  \item (Verificações de consistência) Verifique o teorema da amostragem por
        entregar a dois membros do $ PW_\Omega $: (a) $ f =
        \operatorname{sinc}(\Omega\,\cdot)$, cujas amostras
        são $\delta_{n0}$; (b) $ f(x) = \cos(\omega x)
        \operatorname{sinc}(\varepsilon x)$-tipo banda estreita
        sinais --- mais precisamente, mostram que para $ f \in
        PW_{\Omega'}$ com $\Omega' < \Omega $, o
        $\Omega $ A série -rate também reconstrói $ f $
        (a superamostragem é inofensiva), incorporando $ PW_
        {\Omega'} \subseteq PW_\Omega $.
\end{enumerate}

\medskip
\noindent\textbf{Part VI --- Uncertainty, twice more.}
Desigualdade de Heisenberg (\cref{exo:b3:fouriertransform:8})
limites \emph{como} concentrado $ f $ e $\hat f $ pode em conjunto
ser; aqui está seu irmão tudo ou nada e seu exato
saturação.
\begin{enumerate}[resume]
  \item Deixar $ f \in L^1$ com $\operatorname{supp}f \subseteq
        \intcc{-A}A $. Mostre isso $\hat f $ é a soma de um
        série de potências convergentes em todos os lugares:
        \[
        \hat f(\xi) = \sum_{k\geq0}\frac{(-\iu\xi)^k}{k!}
        \,m_k, \qquad m_k = \int_{-A}^{A}x^kf(x)\,\dd x,
        \quad \abs{m_k} \leq A^k\norm f_1
        \]
        \emph{(expandir $\eu^{-\iu\xi x}$ e justifique o
        intercâmbio por convergência \omterm{def:b3:groups:normal}{normal})}: a transformação é
        \emph{real-analítico}, com raio infinito de
        convergência em todos os pontos.
  \item Deduzir a dicotomia de suporte: uma análise real-analítica
        função desaparecendo em um intervalo aberto não vazio
        desaparece de forma idêntica \emph{(o conjunto onde todos
        derivadas desaparecem são abertas e fechadas --- resolver
        o argumento de Taylor)}; concluir que não há nada diferente de zero $ f $
        tem ambos $ f $ e $\hat f $ suportado de forma compacta e
        isso $ PW_\Omega $ não contém nenhum diferente de zero compactamente
        função suportada --- últimos sinais com banda limitada
        para sempre, e sinais limitados no tempo vazam para todos
        frequências.
  \item (Saturação de Heisenberg) Na família Gaussiana $ f
        = \eu^{-ax^2}$, calcule ambos os fatores de concentração
        e verifique se o produto normalizado
        $\bigl(\int x^2\abs f^2\bigr)\bigl(\frac1{2\pi}\int
        \xi^2\abs{\hat f}^2\bigr)\big/\norm f_2^4$ é igual
        $\frac14$ para \emph{todo} $ a $ --- a igualdade
        família de \cref{exo:b3:fouriertransform:8} no
        carne; explique por meio de um argumento de escala por que o produto
        deve ser constante ao longo da família.
  \item Explique a leitura física (posição/momento
        \omterm{ex:b3:lebesgue:gamma}{densidades} de um estado quântico; $\hbar $ no
        normalização dá $\sigma_x\sigma_p \geq
        \frac\hbar2$) e conecte-se ao longo do capítulo:
        suavização instantânea (Parte II), irreversibilidade (Parte
        IV), amostragem (Parte V), Heisenberg e o suporte
        dicotomia são cinco expressões de uma lei --- a
        comportamento de $\hat f $ no infinito legisla o que $ f $
        pode fazer em qualquer lugar.
\end{enumerate}

\medskip
\noindent\textbf{Part VII --- The kernel's algebra, and one
\omterm{def:b3:groups:derived}{solucionável} example.}
\begin{enumerate}[resume]
  \item (Semigrupo) Prove o \emph{Chapman-Kolmogorov
        identidade} $ g_t * g_s = g_{t+s}$ para $ t, s > 0$ (através
        o teorema da convolução e a injetividade do
        transformar em $ L^1$) e deduza $ u(t + s) = g_s *
        u(t)$: evoluindo com o tempo $ t + s $ está evoluindo para $ t $,
        então para $ s $. Afie a dissipação da questão 8:
        escrevendo $\norm{u(t)}_2^2 = \frac1{2\pi}\int
        \eu^{-2t\xi^2}\abs{\hat f(\xi)}^2\dd\xi $, mostrar por
        Cauchy--Schwarz que
        \[
        t \longmapsto \ln\,\norm{u(t)}_2
        \quad\text{is convex on } \intoo0{+\infty} :
        \]
        o $ L^2$ a energia não apenas diminui, ela
        diminui de forma log-convexa.
  \item (Para onde vai o calor) Vamos $ f \geq 0$, $ f \in L^1$,
        com $\int x^2f(x)\dd x < \infty $. Mostre isso para todos
        $ t > 0$
        \[
        \int_\R x\,u(t, x)\,\dd x = \int_\R x f(x)\,\dd x,
        \qquad
        \int_\R x^2u(t, x)\,\dd x
        = \int_\R x^2f(x)\,\dd x + 2t\int_\R f :
        \]
        o \omterm{ex:b3:groups:actions}{centro} de calor nunca se move, e a variação
        cresce \emph{linearmente} no tempo --- o difusivo
        dimensionamento $ x \sim \sqrt{2t}$, para ser relido quando
        O movimento browniano aparece em \cref{ch:b3:probability}.
        \emph{(Calcule os dois primeiros momentos de $ g_t $ e usar
        Tonelli sobre a convolução.)}
  \item (O Gaussiano, resolvido de ponta a ponta) Para $ f(x) =
        \eu^{-x^2}$, estabeleça o formulário fechado
        \[
        u(t, x) = \frac1{\sqrt{1 + 4t}}\,
        \exp\Bigl(-\frac{x^2}{1 + 4t}\Bigr),
        \]
        e verifique nele, manualmente: a equação $\partial_tu
        = \partial^2_{xx}u $; conservação $\int u(t) =
        \sqrt\pi $; a lei da dissipação $\norm{u(t)}_2 =
        (\pi/2)^{1/4}(1 + 4t)^{-1/4}$ (compare seu $ t^{-1/4}$
        decair com o $ t^{-1/2}$ decadência sup-norma da questão
        8); e o crescimento exato da variância da questão 24. Em
        $ t = 6$: o pico caiu para $\frac15$ do seu
        altura inicial enquanto o perfil é cinco vezes mais largo
        --- mesmo calor, espalhe.
\end{enumerate}
\end{problem}
