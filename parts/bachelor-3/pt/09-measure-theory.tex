\chapter{Teoria da Medida}\label{ch:b3:measure}

Quanto tempo dura um subconjunto de $\R $? A resposta ingênua --- atribuir a
cada conjunto tem um comprimento invariante à tradução estendendo-se ao de
intervalos --- é \emph{impossível}: Construção de Vitali, em
final deste capítulo, produz um conjunto sem consistência
comprimento. \omterm{def:b3:measure:measure}{Medir} teoria é o retiro disciplinado: restringir
atenção a uma rica classe de \emph{mensurável} conjuntos, nos quais um
o comprimento aditivo contável existe e é único. As recompensas são
imenso --- integral de Lebesgue (\cref{ch:b3:lebesgue}), o
$ L^p $ espaços de análise funcional, e todo o moderno
probabilidade (\cref{ch:b3:probability}) são construídos sobre os três
teoremas provados aqui: lema da unicidade de Dynkin,
Carathteorema da extensão de eodory e a existência de
Medida Lebesgue.

\section{$\sigma $-álgebras}

\begin{definition}\label{def:b3:measure:sigmaalgebra}
UM \emph{$\sigma $-álgebra}\index{$\sigma $-álgebra}
em um conjunto $ X $ é uma família $\mathcal A $ de subconjuntos contendo
$\varnothing $, estável sob complemento e sob
\emph{contável} sindicatos (daí interseções contáveis, conjunto
diferenças e contém $ X $). O par $(X, \mathcal A)$
é um \emph{espaço mensurável}; membros de $\mathcal A $ são
\emph{conjuntos mensuráveis}. Para qualquer família $\mathcal E $ de subconjuntos,
$\sigma(\mathcal E)$ denota o menor $\sigma $-álgebra
contendo $\mathcal E $ (a intersecção de todos eles ---
um cruzamento de $\sigma $-álgebras é uma delas).
\end{definition}

\begin{definition}\label{def:b3:measure:borel}
O \emph{Borel $\sigma $-álgebra}\index{Borel $\sigma $-álgebra} de um topológico o espaço é
$\mathcal B(X) = \sigma(\{\text{open sets}\})$. Sobre $\R $:
$\mathcal B(\R)$ também é gerado pelos intervalos abertos, por
os intervalos fechados, pelos raios $(-\infty, a]$, e por raios
com pontos finais racionais (\cref{exo:b3:measure:1}) --- cada
família gera as aberturas por operações contáveis, por exemplo\
cada conjunto aberto de $\R $ é uma união contável de intervalos abertos
com dados racionais.
\end{definition}

\begin{definition}\label{def:b3:measure:dynkin}
UM \emph{$\pi $-sistema}\index{$\pi $-sistema} é uma família
estável sob interseções finitas. UM
\emph{$\lambda $-sistema}\index{$\lambda $-sistema}
(classe Dynkin) é uma família $\mathcal D $ com: $ X \in \mathcal
D $; $ A, B \in \mathcal D $, $ A \subseteq B $ $\Rightarrow $ $ B
\setminus A \in \mathcal D $; e $ A_n \uparrow A $, $ A_n \in
\mathcal D $ $\Rightarrow $ $ A \in \mathcal D $.
\end{definition}

\begin{theorem}[Dynkin's $\pi $--$\lambda $ lema]
\label{thm:b3:measure:dynkin}
Se um $\lambda $-sistema $\mathcal D $ contém um $\pi $-sistema
$\mathcal P $, então $\mathcal D \supseteq
\sigma(\mathcal P)$.\index{Lema de Dynkin}
\end{theorem}

\begin{proof}
Deixar $\mathcal D_0$ seja o menor $\lambda $-sistema contendo
$\mathcal P $ (intersecção de tudo isso); basta mostrar
$\mathcal D_0$ é um $\sigma $-álgebra, desde então
$\sigma(\mathcal P) \subseteq \mathcal D_0 \subseteq \mathcal
D $. UM $\lambda $-sistema estável sob interseções finitas
\emph{é} um $\sigma $-álgebra: complementos ($ X \setminus A = X
\setminus A $ com $ A \subseteq X $), uniões finitas ($ A \cup B =
X \setminus ((X\setminus A)\cap(X\setminus B))$) e contável
sindicatos via $\bigcup_{k \leq n}A_k \uparrow \bigcup_kA_k $. Então nós
provar $\mathcal D_0$ é um $\pi $-sistema, em duas etapas. Deixar
\[
\mathcal D_1 = \{A \in \mathcal D_0 : A \cap P \in \mathcal
D_0 \ \forall P \in \mathcal P\}.
\]
$\mathcal D_1$ é um $\lambda $-sistema (todos os três axiomas são
verificado cruzando com $ P $: por exemplo\ $(B\setminus A)\cap P
= (B \cap P)\setminus(A \cap P)$, uma diferença adequada dentro
$\mathcal D_0$) e contém $\mathcal P $ ($\pi $-sistema):
$\mathcal D_1 = \mathcal D_0$. Agora deixe
\[
\mathcal D_2 = \{A \in \mathcal D_0 : A \cap D \in \mathcal
D_0\ \forall D \in \mathcal D_0\}.
\]
Pela etapa anterior, $\mathcal D_2 \supseteq \mathcal P $; e
$\mathcal D_2$ é um $\lambda $-sistema pela mesma verificação:
$\mathcal D_2 = \mathcal D_0$, que diz exatamente isso
$\mathcal D_0$ é estável nas interseções.
\end{proof}

\section{Medidas}

\begin{definition}\label{def:b3:measure:measure}
UM \emph{medir}\index{medir} sobre $(X, \mathcal A)$ é um mapa
$\mu \colon \mathcal A \to [0, +\infty]$ com $\mu(\varnothing)
= 0$ aquilo é \emph{$\sigma $-aditivo}: para disjunção de pares
$(A_n)_{n\in\N}$,
\[
\mu\Bigl(\bigsqcup_n A_n\Bigr) = \sum_n \mu(A_n).
\]
$(X, \mathcal A, \mu)$ é um \emph{medir o espaço}; $\mu $ é
\emph{finito} se $\mu(X) < \infty $, um \emph{probabilidade
medir} se $\mu(X) = 1$, \emph{$\sigma $-finito} se $ X $ é um
união contável de conjuntos de medida finita. Exemplos: contando
medir em $(\N, \mathcal P(\N))$; a massa de Dirac $\delta_a(A)
= \mathbf 1_{a \in A}$; e, o objeto deste capítulo,
Medida Lebesgue.
\end{definition}

\begin{proposition}\label{prop:b3:measure:basics}
Deixar $\mu $ seja um \omterm{def:b3:measure:measure}{medir}. (a) Monotonicidade: $ A \subseteq B
\Rightarrow \mu(A) \leq \mu(B)$. (b) Subaditividade contável:
$\mu(\bigcup A_n) \leq \sum\mu(A_n)$. (c) Continuidade de
abaixo: $ A_n \uparrow A \Rightarrow \mu(A_n) \to \mu(A)$. (d)
Continuidade de cima: $ A_n \downarrow A $ \emph{com
$\mu(A_1) < \infty $} $\Rightarrow \mu(A_n) \to \mu(A)$.
\end{proposition}

\begin{proof}
(um) $ B = A \sqcup (B\setminus A)$. (b) Desconjuntar: $ B_n = A_n
\setminus \bigcup_{k<n}A_k $ são disjuntos do mesmo sindicato,
e $\mu(B_n) \leq \mu(A_n)$. (c) $ A = \bigsqcup_n (A_n
\setminus A_{n-1})$ ($ A_0 = \varnothing $): as somas parciais de
$\sum\mu(A_n\setminus A_{n-1})$ são $\mu(A_n)$. (d) Aplicar (c)
para $ A_1 \setminus A_n \uparrow A_1 \setminus A $ e subtrair
de $\mu(A_1)$ --- finitude faz a subtração
legítimo. Contra-exemplo sem ele: $ A_n = [n, \infty)$ para
Medida Lebesgue: $ A_n \downarrow \varnothing $ mas $\mu(A_n) =
\infty $.
\end{proof}

\begin{theorem}[Singularidade]\label{thm:b3:measure:uniqueness}
Deixar $\mu, \nu $ ser medidas sobre $\sigma(\mathcal P)$, $\mathcal
P $ um $\pi $-sistema, com $\mu = \nu $ sobre $\mathcal P $. Se houver
são conjuntos $ P_k \in \mathcal P $ com $ P_k \uparrow X $ e
$\mu(P_k) < \infty $, então $\mu = \nu $ em todos
$\sigma(\mathcal P)$.
\end{theorem}

\begin{proof}
Consertar $ k $ e considere o finito medidas $\mu_k(A) = \mu(A \cap
P_k)$ e $\nu_k(A) = \nu(A \cap P_k)$ sobre $\sigma(\mathcal
P)$: eles concordam $\mathcal P $, desde $ P \cap P_k \in \mathcal
P $ ($\pi $-sistema), e eles atribuem $ X $ o mesmo valor finito
$\mu(P_k)$. A aula $\mathcal D = \{A : \mu_k(A) =
\nu_k(A)\}$ é um $\lambda $-sistema: $ X \in \mathcal D $; adequado
diferenças por subtração (valores finitos); limites crescentes
por continuidade de baixo (\cref{prop:b3:measure:basics}(c)).
Ele contém o $\pi $-sistema $\mathcal P $, então Dynkin
(\cref{thm:b3:measure:dynkin}) dá $\mathcal D \supseteq
\sigma(\mathcal P)$: $\mu_k = \nu_k $ em todos os lugares. Finalmente, para
qualquer $ A \in \sigma(\mathcal P)$, continuidade de baixo ao longo $ A
\cap P_k \uparrow A $ dá $\mu(A) = \lim_k\mu_k(A) =
\lim_k\nu_k(A) = \nu(A)$.
\end{proof}

\section{Medidas externas e Carathteorema de eodory}

\begin{definition}\label{def:b3:measure:outer}
Um \emph{medida externa}\index{medida externa} sobre $ X $ é um mapa
$\mu^* \colon \mathcal P(X) \to [0, \infty]$ com
$\mu^*(\varnothing) = 0$, monótono e subaditivo contável.
Um conjunto $ A $ é \emph{$\mu^*$-mensurável} (Carathéodory) se
divide cada conjunto aditivamente:
\[
\mu^*(E) = \mu^*(E \cap A) + \mu^*(E \setminus A)
\qquad \text{for every } E \subseteq X
\]
($\leq $ sempre vale por subaditividade; o conteúdo é $\geq $).
\end{definition}

\begin{theorem}[Carathéodoro]\label{thm:b3:measure:caratheodory}
O $\mu^*$-conjuntos mensuráveis formam um $\sigma $-álgebra $\mathcal
M $, e $\mu^*\restriction_{\mathcal M}$ é um \omterm{def:b3:measure:measure}{medir}.
Além disso, cada conjunto com $\mu^*(N) = 0$ pertence a $\mathcal M $
(o \omterm{def:b3:measure:measure}{medir} é \emph{completo}).\index{Carathéodory's
theorem}
\end{theorem}

\begin{proof}
$\mathcal M $ contém $\varnothing $ e é estável sob
complemento (a condição definidora é simétrica em $ A $,
$ X\setminus A $). \emph{Uniões finitas}: deixar $ A, B \in \mathcal
M $ e $ E $ arbitrário; divisão $ E $ por $ A $, então cada peça por
$ B $:
\[
\mu^*(E) = \mu^*(E\cap A\cap B) + \mu^*(E\cap A\setminus B) +
\mu^*(E\cap B\setminus A) + \mu^*(E\setminus(A\cup B)).
\]
As três primeiras peças cobrem $ E \cap (A \cup B)$, então
subaditividade dá $\mu^*(E) \geq \mu^*(E\cap(A\cup B)) +
\mu^*(E\setminus(A\cup B))$: $ A \cup B \in \mathcal M $. Por
indução, uniões finitas; com complementos, desarticulação finita
manipulações estão disponíveis.

\emph{Aditividade ativada $\mathcal M $}: para disjunto $ A, B \in
\mathcal M $ e qualquer $ E $: $\mu^*(E\cap(A\sqcup B)) = \mu^*(E\cap
A) + \mu^*(E \cap B)$ (dividido por $ A $); por indução,
\[
\mu^*\Bigl(E \cap \bigsqcup_{k\leq n}A_k\Bigr)
= \sum_{k\leq n}\mu^*(E\cap A_k).
\tag{$*$}
\]

\emph{Sindicatos contáveis}: deixar $(A_k) \subseteq \mathcal M $ ser
disjunto (basta, por desjunção dentro da álgebra
$\mathcal M $), $ A = \bigsqcup A_k $, $ E $ arbitrário. Usando
$\bigsqcup_{k\leq n}A_k \in \mathcal M $ e monotonicidade:
\[
\mu^*(E) = \mu^*\Bigl(E\cap\bigsqcup_{k\leq n}A_k\Bigr) +
\mu^*\Bigl(E\setminus\bigsqcup_{k\leq n}A_k\Bigr)
\geq \sum_{k \leq n}\mu^*(E\cap A_k) + \mu^*(E\setminus A)
\]
por ($*$). Deixar $ n \to \infty $ e usar subaditividade contável
para trás:
\[
\mu^*(E) \geq \sum_{k}\mu^*(E\cap A_k) + \mu^*(E\setminus A)
\geq \mu^*(E \cap A) + \mu^*(E\setminus A) \geq \mu^*(E):
\]
todas as desigualdades são igualdades. Isto prova ambos $ A \in
\mathcal M $ e, tomando $ E = A $, aditividade contável de
$\mu^*$ sobre $\mathcal M $.

\emph{Conjuntos nulos}: se $\mu^*(N) = 0$, então para qualquer $ E $:
$\mu^*(E \cap N) + \mu^*(E\setminus N) \leq 0 + \mu^*(E)$:
$ N \in \mathcal M $.
\end{proof}

\section{Medida de Lebesgue em \texorpdfstring{$\R $}{R}}

\begin{definition}\label{def:b3:measure:lebesgueouter}
O \emph{Medida externa de Lebesgue}\index{Medida externa de Lebesgue} de $ A
\subseteq \R $ é
\[
\lambda^*(A) = \inf\Bigl\{\sum_{n} (b_n - a_n) :
A \subseteq \bigcup_n \intoo{a_n}{b_n}\Bigr\}
\]
(capas contáveis por intervalos abertos).
\end{definition}

\begin{lemma}\label{lem:b3:measure:outerbasics}
$\lambda^*$ é um \omterm{def:b3:measure:outer}{medida externa}, invariante nas traduções,
e $\lambda^*(I) = \ell(I)$ (o comprimento) para cada intervalo
$ I $.
\end{lemma}

\begin{proof}
\omterm{def:b3:measure:outer}{Medida externa}: $\varnothing $ é coberto por arbitrariamente pequenos
intervalos; a monotonicidade é clara; subaditividade: dadas coberturas
de cada $ A_n $ dentro de $\varepsilon 2^{-n}$ do ínfimo,
suas capas sindicais $\bigcup A_n $ com comprimento total $\leq \sum
\lambda^*(A_n) + \varepsilon $. Invariância de tradução:
traduzir as capas.

Comprimento: é suficiente para tratar $ I = \intcc ab $ (outros tipos
diferem por pontos finais, que têm \omterm{def:b3:measure:outer}{medida externa} $0$: cobrir por
pequenos intervalos; então aperte $\intcc{a+\varepsilon}{b -
\varepsilon} \subseteq \intoo ab $ comparações de tipo).
$\lambda^*(\intcc ab) \leq b - a $: cobrir por
$\intoo{a-\varepsilon}{b+\varepsilon}$. Por outro lado, deixe $\intcc
ab \subseteq \bigcup_n\intoo{a_n}{b_n}$: por
\emph{compacidade} (Borel-Lebesgue,
\cref{thm:b3:topology:metriccompact}), um número finito de intervalos
basta, digamos $ I_1, \dots, I_N $. Nós mostramos $\sum_{k\leq N}(b_k -
a_k) \geq b - a $ por indução em $ N $: escolher $ I_{k_1} \ni a $; se
$ b_{k_1} > b $ feito ($ b_{k_1} - a_{k_1} > b - a $); senão o
segmento $\intcc{b_{k_1}}b $ é coberto pelo restante $ N - 1$
intervalos, e a indução dá $\sum_{k \neq k_1}(b_k - a_k)
\geq b - b_{k_1}$, enquanto $ b_{k_1} - a_{k_1} > b_{k_1} - a $:
resumir.
\end{proof}

\begin{theorem}[Medida Lebesgue]\label{thm:b3:measure:lebesgue}
Cada conjunto de Borel $\R $ é $\lambda^*$-mensurável. O
restrição $\lambda $ de $\lambda^*$ para o $\sigma $-álgebra
$\mathcal L = \mathcal M_{\lambda^*} \supseteq \mathcal B(\R)$
(o \emph{Lebesgue $\sigma $-álgebra}) é o único \omterm{def:b3:measure:measure}{medir} sobre
$\mathcal B(\R)$ atribuindo a cada intervalo sua duração; é
invariante à tradução e $\sigma $-finito.
\end{theorem}

\begin{proof}
Por \cref{thm:b3:measure:caratheodory} basta mostrar cada
raio $ A = \intoo{-\infty}c $ é $\lambda^*$-mensurável
(os raios geram $\mathcal B $,
\cref{def:b3:measure:borel}). Deixar $ E \subseteq \R $ com
$\lambda^*(E) < \infty $ e $\bigcup I_n \supseteq E $ uma capa
com $\sum\ell(I_n) \leq \lambda^*(E) + \varepsilon $. Cada
$ I_n $ divide-se em dois intervalos $ I_n' = I_n \cap A $ e
$ I_n'' = I_n\setminus A $ (um intervalo menos um raio é um
intervalo) com $\ell(I_n') + \ell(I_n'') = \ell(I_n)$; o
$ I_n'$ cobrir $ E \cap A $ e o $ I_n''$ cobrir $ E \setminus A $
(ampliar cada um para um intervalo aberto de comprimento $\ell +
\varepsilon2^{-n}$ permanecer dentro da definição), então
\[
\lambda^*(E\cap A) + \lambda^*(E\setminus A)
\leq \sum_n\bigl(\ell(I_n') + \ell(I_n'')\bigr) + 2\varepsilon
\leq \lambda^*(E) + 3\varepsilon .
\]
Singularidade: dois medidas concordando com o comprimento do
$\pi $-sistema de intervalos $\intoc ab $ (finito sobre eles) concordar em
$\sigma(\text{intervals}) = \mathcal B $ por
\cref{thm:b3:measure:uniqueness} com $ P_k = \intoc{-k}k $.
$\sigma $-finitude: $\R = \bigcup(-k, k]$.
\end{proof}

\begin{theorem}[Regularidade]\label{thm:b3:measure:regularity}
Para cada $ A \in \mathcal L $:\index{regularidade de Lebesgue
medir}
\[
\lambda(A) = \inf\{\lambda(U) : U \supseteq A \text{ open}\}
= \sup\{\lambda(K) : K \subseteq A \text{ compact}\}.
\]
\end{theorem}

\begin{proof}
\emph{Exterior}: uma capa $\bigcup I_n $ com $\sum\ell(I_n) \leq
\lambda(A) + \varepsilon $ é um conjunto aberto $ U \supseteq A $ com
$\lambda(U) \leq \lambda(A) + \varepsilon $ (subaditividade); se
$\lambda(A) = \infty $ a afirmação é trivial. \emph{Interno}:
primeiro deixe $ A $ ser limitado, $ A \subseteq [-M, M]$. Escolha um aberto
$ U \supseteq ([-M,M]\setminus A)$ com $\lambda(U) \leq
\lambda([-M,M]\setminus A) + \varepsilon $; então $ K = [-M,
M]\setminus U $ é compactar, $ K \subseteq A $, e
\[
\lambda(K) = \lambda([-M,M]) - \lambda([-M,M]\cap U)
\geq \lambda([-M,M]) - \bigl(\lambda([-M,M]) - \lambda(A) +
\varepsilon\bigr) = \lambda(A) - \varepsilon .
\]
Para geral $ A $: $\lambda(A) = \lim_M\lambda(A\cap[-M,M])$
(continuidade de baixo) e aplique o caso delimitado dentro.
\end{proof}

\begin{example}\label{ex:b3:measure:cantor}
O conjunto Cantor (\cref{exo:b3:topology:10}) tem $\lambda(C) = 0$:
$ C \subseteq C_n $, uma união de $2^n $ intervalos de comprimento
$3^{-n}$, então $\lambda(C) \leq (2/3)^n \to 0$. Um incontável
conjunto nulo --- cardinalidade não vê \omterm{def:b3:measure:measure}{medir}. Por outro lado,
\emph{conjuntos Cantor gordos} (\cref{exo:b3:measure:5}) não estão em lugar nenhum
\omterm{def:b3:topology:interior}{denso} com positivo \omterm{def:b3:measure:measure}{medir}: \omterm{def:b3:topology:topology}{topologia} não vê \omterm{def:b3:measure:measure}{medir}
qualquer. O problema do fim de semana leva essa interação ao seu
\omterm{thm:b3:complete:completion}{conclusão} surpreendente: existem conjuntos mensuráveis de Lebesgue que
não são Borel.
\end{example}

\begin{theorem}[Vitali]\label{thm:b3:measure:vitali}
Não há \omterm{def:b3:measure:measure}{medir} sobre \emph{todos} subconjuntos de $\R $ isso é
invariante à tradução e atribui a cada intervalo seu comprimento.
Em particular $\mathcal L \neq \mathcal P(\R)$: não mensurável
conjuntos existem.\index{Conjunto não mensurável de Vitali}
\end{theorem}

\begin{proof}
Suponha $\mu $ eram um. Sobre $\intcc01$, considere a equivalência
$ x \sim y \iff x - y \in \Q $; pelo \emph{axioma da escolha}
escolha um representante em $\intcc01$ por aula: um conjunto $ V $. Para
$ q \in \Q\cap\intcc{-1}1$, a tradução $ V + q $ são pares
disjunto (dois pontos de $ V $ diferir por um racional seria
representantes equivalentes, mas distintos) e
\[
\intcc01 \subseteq \bigsqcup_{q \in \Q\cap\intcc{-1}1}(V + q)
\subseteq \intcc{-1}2 :
\]
a primeira inclusão desde cada $ x \in \intcc01$ difere de
seu representante $ v $ por um racional $ q = x - v \in
\intcc{-1}1$. Monotonicidade e $\sigma $-aditividade dar
\[
1 \leq \sum_{q}\mu(V + q) \leq 3,
\qquad\text{with } \mu(V + q) = \mu(V) \text{ for all } q .
\]
Uma soma infinita da constante $\mu(V)$ é $0$ ou $\infty $:
ambos os limites não podem ser mantidos. Então não existe tal $\mu $ existe --- e $ V
\notin \mathcal L $, desde $\lambda $ sobre $\mathcal L $ tem todo o
propriedades utilizadas.
\end{proof}

\begin{remark}\label{rem:b3:measure:banachtarski}
Em $\R^3$ o fracasso é mais dramático: o Banach--Tarski
paradoxo decompõe uma bola em cinco pedaços que se remontam, por
rotações e translações, em \emph{dois} bolas do mesmo
raio --- então nem mesmo um invariante de rotação finitamente aditivo
volume em todos os subconjuntos de $\R^3$ existe. As peças são, de
claro, não mensurável. Mensurabilidade não é burocrática
cautela; é o limite da coerência.
\end{remark}

\begin{method}\label{met:b3:measure:goodsets}
O \emph{princípio dos bons conjuntos}: para provar que todos os conjuntos de
$\sigma(\mathcal E)$ tem uma propriedade, mostre que os conjuntos bons
formar um $\sigma $-álgebra (ou um $\lambda $-sistema, se a propriedade
é teórico da medida e $\mathcal E $ é um $\pi $-sistema ---
então Dynkin) contendo $\mathcal E $. Quase todas as provas de
este capítulo e o próximo são um exemplo. Para provar dois medidas
iguais: verifique-os em uma geração $\pi $-sistema mais
$\sigma $-finitude (\cref{thm:b3:measure:uniqueness}). Para
construir um \omterm{def:b3:measure:measure}{medir}: construir um \omterm{def:b3:measure:outer}{medida externa} por capas e orçamento
Carathéodoro.
\end{method}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:b3:measure:1}
(a) Mostre que $\{A \subseteq X : A $ ou $ X\setminus A $ é
contável $\}$ é um $\sigma $-álgebra: aquela gerada por
solteiros.
(b) Mostre que $\mathcal B(\R)$ é gerado por cada um dos seguintes: aberto
intervalos; intervalos fechados; raios $\intoc{-\infty}a $; raios
com $ a \in \Q $.
(c) A família de \emph{uniões disjuntas finitas de intervalos}
$\intoc ab $ um $\sigma $-álgebra? Uma álgebra (estável sob
complemento e uniões finitas)?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b3:measure:2}
(a) Prove inclusão-exclusão para um período finito \omterm{def:b3:measure:measure}{medir}: $\mu(A\cup
B) = \mu(A) + \mu(B) - \mu(A\cap B)$, e o conjunto de três
versão.
(b) Dê um exemplo mostrando que continuidade de cima
(\cref{prop:b3:measure:basics}(d)) falha sem o
suposição de finitude.
(c) Mostre que um conjunto contável tem Medida Lebesgue zero.
Deduzir $\lambda(\Q) = 0$ e $\lambda(\intcc01\setminus\Q) =
1$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:measure:3}
Deixar $\mu, \nu $ ser probabilidade medidas sobre $\mathcal B(\R)$
com $\mu(\intoc{-\infty}t) = \nu(\intoc{-\infty}t)$ para todos
$ t \in \R $. Mostrar $\mu = \nu $. (Isso faz com que
\emph{função de distribuição} $ F(t) = \mu(\intoc{-\infty}t)$ um
completo invariante --- a \omterm{def:b3:topology:basis}{base} de
\cref{ch:b3:probability}.)
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:measure:4}
(Borel-Cantelli, \omterm{def:b3:measure:measure}{medir} versão) Deixe $(A_n)$ ser mensurável
com $\sum_n\mu(A_n) < \infty $, e $\limsup A_n =
\bigcap_N\bigcup_{n\geq N}A_n $ (os pontos pertencentes a
infinitamente muitos $ A_n $). Mostrar $\mu(\limsup A_n) = 0$.
Aplicação: para quase todos $ x \in \intcc01$, apenas finitamente
muitos $ n $ satisfazer $\abs{x - p/q_n} \leq 4^{-n}$ para o
$ n $-º racional $ p/q_n $ de uma enumeração de
$\Q\cap\intcc01$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:measure:5}
(Conjunto Fat Cantor) Repita a construção do Cantor em $\intcc01$,
mas no passo $ n $ retire de cada um dos $2^{n-1}$ intervalos a
\emph{centrado} intervalo aberto de comprimento $4^{-n}$ apenas. Mostrar
que o resultante $ K = \bigcap K_n $ é compactar, tem vazio
\omterm{def:b3:topology:interior}{interior} (nenhum intervalo sobrevive), e
\[
\lambda(K) = 1 - \sum_{n\geq1}2^{n-1}4^{-n} = \tfrac12 :
\]
um lugar nenhum conjunto \omterm{def:b3:topology:interior}{denso} de \omterm{def:b3:measure:measure}{medir} $\frac12$. Deduza um
\emph{\omterm{rem:b3:complete:meagre}{escasso}} subconjunto de $\intcc01$ de cheio \omterm{def:b3:measure:measure}{medir} $1$, e um
subconjunto \omterm{def:b3:topology:interior}{denso} aberto de \omterm{def:b3:measure:measure}{medir} $< \varepsilon $.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:measure:6}
Deixar $\mu $ seja um \omterm{def:b3:measure:measure}{medir} sobre $\mathcal B(\R)$, invariante sob
traduções, com $ c = \mu(\intoc01) < \infty $. Mostrar $\mu =
c\,\lambda $ sobre $\mathcal B(\R)$. \emph{(Calcular $\mu $ em intervalos diádicos dividindo
$\intoc01$ em $2^n $ traduz, então invoca
\cref{thm:b3:measure:uniqueness}.)}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:measure:7}
(Aproximação) Deixe $ A \in \mathcal L $ com $\lambda(A) <
\infty $ e $\varepsilon > 0$. Mostre que existe um \emph{finito}
união de intervalos $ B $ com $\lambda(A\,\triangle\,B) <
\varepsilon $ ($\triangle $ = diferença simétrica).
\emph{(Regularidade: apertar $ K \subseteq A \subseteq U $ e usar
a estrutura do aberto $ U $ como uma união contável de
intervalos, além de compacidade de $ K $.)}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b3:measure:8}
(Steinhaus) Vamos $ A \in \mathcal L $ com $\lambda(A) > 0$. Mostrar
isso $ A - A = \{x - y : x, y \in A\}$ contém um intervalo
ao redor $0$.
\emph{(Reduzir para $\lambda(A) < \infty $; por
\cref{exo:b3:measure:7}regularidade de estilo, encontre um intervalo $ I $
com $\lambda(A \cap I) > \frac34\ell(I)$; então para $\abs t <
\frac12\ell(I)$, os conjuntos $ A\cap I $ e $(A\cap I) + t $ ambos
sente-se em um intervalo de comprimento $\frac32\ell(I)$ e tem total
\omterm{def:b3:measure:measure}{medir} $> \frac32\ell(I)$: eles devem se cruzar.)}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b3:measure:9}
Mostre que cada $ A \in \mathcal L $ com $\lambda(A) > 0$
contém um subconjunto não mensurável. \emph{(Intersectar $ A $ com o
traduz $ V + q $ do conjunto de Vitali: se tudo $ A \cap (V+q)$ eram
mensuráveis, cada um seria nulo pelo argumento de
\cref{thm:b3:measure:vitali} --- Steinhaus
(\cref{exo:b3:measure:8}) ajuda: um conjunto mensurável de resultados positivos
\omterm{def:b3:measure:measure}{medir} dentro $ V + q $ daria $(V+q) - (V+q) \supseteq $
um intervalo, contradizendo que este conjunto de diferenças atende $\Q $
apenas em $0$; conclua com subaditividade.)}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:measure:10}
Mostre isso $ A \subseteq \R $ com $\lambda^*(A) < \infty $ é
Lebesgue-mensurável iff para cada $\varepsilon > 0$ há um
aberto $ U \supseteq A $ com $\lambda^*(U \setminus A) <
\varepsilon $, se houver um $ G_\delta $ definir $ G \supseteq A $
com $\lambda^*(G\setminus A) = 0$. (Portanto, os conjuntos de Lebesgue são
Borel define conjuntos nulos de módulo.)
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:measure:11}
(Continuidade ao longo dos limites monótonos e sua nitidez)
(a) Mostre que para conjuntos mensuráveis, $\mu(\liminf A_n) \leq
\liminf\mu(A_n)$ (Fatou para conjuntos), e que se
$\mu\bigl(\bigcup A_n\bigr) < \infty $, também
$\limsup\mu(A_n) \leq \mu(\limsup A_n)$.
(b) Exibição, para Medida Lebesgue sobre $\R $, uma sequência com
$\mu(A_n) = 1$ para todos $ n $ ainda $\mu(\limsup A_n) = 0$: o
hipótese de finitude na segunda desigualdade não é
decorativo.
(c) Deduza: se $\sum\mu(A_n) < \infty $ então
$\mu(\limsup A_n) = 0$ (Borel--Cantelli novamente), e se o
$ A_n $ aumentar ou diminuir (com $\mu(A_1) < \infty $ no
caso decrescente), $\mu(\lim A_n) = \lim\mu(A_n)$.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b3:measure:12}
(Teorema de Egorov) Deixe $\mu(X) < \infty $ e $ f_n \to f $
pontualmente, todos mensuráveis (com valor real). Para $ k, N \geq 1$
definir
\[
E_{k,N} = \bigcap_{n \geq N}\Bigl\{x : \abs{f_n(x) - f(x)}
\leq \tfrac1k\Bigr\} .
\]
(a) Mostre que para fixo $ k $, $ E_{k,N} \nearrow X $ como $ N \to
\infty $, e deduza $ N_k $ com $\mu(X \setminus E_{k,N_k})
\leq \varepsilon2^{-k}$.
(b) Concluir \emph{Teorema de Egorov}: para cada $\varepsilon
> 0$ existe um mensurável $ A $ com $\mu(X\setminus A) \leq
\varepsilon $ tal que $ f_n \to f $ \emph{uniformemente ligado $ A $}
--- convergência pontual é a convergência uniforme de um
conjunto arbitrariamente pequeno.
(c) Mostre que o teorema falha em $(\R, \lambda)$: o movimento
solavancos $ f_n = \mathbf 1_{\intcc n{n+1}}$ convergem pontualmente
para $0$ mas uniformemente em nenhum complemento de um medida finita
definir. Onde (a) usou $\mu(X) < \infty $?
\end{exercise}

\section{Problema: a escada Cantor-Vitali e uma
conjunto mensurável que não é Borel}

{\small A escada Cantor-Vitali: constante em cada vão da
o conjunto Cantor, ainda subindo de $0$ para $1$ continuamente. É

\begin{problem}[{Problema de fim de semana --- a escada do diabo,
e $\mathcal B(\R) \subsetneq \mathcal L $}]
\label{pb:b3:measure:1}
Nós construímos o \emph{Função Cantor - Vitali} (do diabo
escada), use-a para transportar \omterm{def:b3:measure:measure}{medir} patologicamente e
conclua com um teorema que nenhum argumento suave oferece: há
existem conjuntos mensuráveis de Lebesgue que não são Borel. Notação:
$ C $ é o conjunto de Cantor, $ C_n $ isso é $ n $-ª etapa ($2^n $ intervalos
de comprimento $3^{-n}$) e cada $ x \in C $ tem dígitos ternários
$ x = \sum 2b_n3^{-n}$, $ b_n \in \{0,1\}$
(\cref{exo:b3:topology:10}).

\medskip
\noindent\textbf{Part I --- The staircase.} Definir $ c_0(x) = x $
e $ c_{n+1}$ de $ c_n $ por
\[
c_{n+1}(x) = \begin{cases}
\tfrac12\,c_n(3x) & 0 \leq x \leq \tfrac13,\\[2pt]
\tfrac12 & \tfrac13 \leq x \leq \tfrac23,\\[2pt]
\tfrac12 + \tfrac12\,c_n(3x - 2) & \tfrac23 \leq x \leq 1.
\end{cases}
\]
\begin{enumerate}
  \item Mostre que cada $ c_n $ é \omterm{def:b3:topology:continuity}{contínuo}, não decrescente, com
        $ c_n(0) = 0$, $ c_n(1) = 1$, e isso
        $\norm{c_{n+1} - c_n}_\infty \leq
        \tfrac12\norm{c_n - c_{n-1}}_\infty $.
  \item Deduza isso $(c_n)$ converge uniformemente para um \omterm{def:b3:topology:continuity}{contínuo}
        não decrescente $ c $ com $ c(0) = 0$, $ c(1) = 1$
        (\emph{o} Função Cantor--Vitali), que satisfaz
        as mesmas relações auto-semelhantes que o $ c_{n+1}$ acima.
  \item Mostre isso $ c $ é constante em cada \omterm{def:b3:topology:components}{componente conectado}
        de $\intcc01\setminus C $, e isso para $ x = \sum_n
        2b_n3^{-n} \in C $, $ c(x) = \sum_n b_n2^{-n}$: o
        escada lê dígitos de Cantor em binário
        (\cref{pb:b3:topology:1}função de $ g $, tornou-se monótono
        e globais).
  \item Deduza isso $ c $ é diferenciável, com $ c' = 0$, em
        cada ponto de $\intcc01\setminus C $: $ c' = 0$
        \emph{$\lambda $-quase em todos os lugares}
        (\cref{ex:b3:measure:cantor}). Conclua que o
        teorema fundamental do cálculo falha para $ c $:
        \[
        c(1) - c(0) = 1 \neq 0 = \int_0^1 c'(t)\,\dd t
        \]
        (a integral está sobre o medida completa definir onde
        $ c' = 0$; antecipando \cref{ch:b3:lebesgue}, conjuntos nulos
        não afetam integrais). Qual hipótese do
        $\mathcal C^1$ teorema fundamental é violado?
  \item Mostre isso $ c(C) = \intcc01$: o conjunto nulo $ C $ está mapeado
        \emph{para} um conjunto completo \omterm{def:b3:measure:measure}{medir}.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The crooked \omterm{def:b3:topology:continuity}{homeomorfismo}.} Deixar
$ h(x) = \frac{c(x) + x}{2}$.
\begin{enumerate}[resume]
  \item Mostre isso $ h \colon \intcc01 \to \intcc01$ é um
        \omterm{def:b3:topology:continuity}{homeomorfismo} (estritamente aumentando, \omterm{def:b3:topology:continuity}{contínuo},
        sobrejetivo).
  \item Mostre isso $\lambda\bigl(h(\intcc01\setminus C)\bigr) =
        \tfrac12$: em cada lacuna de comprimento $\ell $, $ h $ atua como um
        mapa afim de declive $\tfrac12$, e as lacunas têm total
        comprimento $1$.
  \item Deduzir $\lambda\bigl(h(C)\bigr) = \tfrac12$: o
        homeomórfico imagem de um conjunto nulo pode ter positivo
        \omterm{def:b3:measure:measure}{medir}. (Onde isso contradiz a intuição ingênua
        sobre ``tamanho''?)
\end{enumerate}

\medskip
\noindent\textbf{Part III --- A measurable set that is not
Borel.}
\begin{enumerate}[resume]
  \item Por \cref{exo:b3:measure:9}, escolha um não mensurável $ W
        \subseteq h(C)$. Mostre isso $ Z = h^{-1}(W) \subseteq C $
        é mensurável por Lebesgue. \emph{(É um subconjunto de um nulo
        conjunto; completude, \cref{thm:b3:measure:caratheodory}.)}
  \item Mostre que a pré-imagem de um conjunto de Borel sob um
        mapa \omterm{def:b3:topology:continuity}{contínuo} é o Borel. \emph{(Princípio de bons conjuntos:
        $\{B : h^{-1}(B) \in \mathcal B\}$ é um
        $\sigma $-álgebra contendo o conjuntos abertos --- cuidado com
        direção do mapa.)}
  \item Conclua isso $ Z $ é \emph{não} Borel: se fosse,
        $ W = (h^{-1})^{-1}(Z)$ seria o Borel (aplique a questão
        10 para o \omterm{def:b3:topology:continuity}{contínuo} $ h^{-1}$), portanto mensurável ---
        contradição. Portanto
        \[
        \boxed{\ \mathcal B(\R) \subsetneq \mathcal L\ }
        \]
        e completude amplia genuinamente o mundo do Borel.
  \item Exiba uma função mensurável de Lebesgue $ g $ e um
        \omterm{def:b3:topology:continuity}{contínuo} função $\varphi $ tal que $ g \circ
        \varphi $ não é mensurável por Lebesgue: mensurabilidade,
        ao contrário continuidade, não compõe. \emph{(Pegar $ g =
        \mathbf 1_Z $ e $\varphi = h^{-1}$, antecipando o
        definição de funções mensuráveis de
        \cref{ch:b3:lebesgue}: as pré-imagens dos conjuntos do Borel são
        Conjuntos de Lebesgue. Onde é preciso ter cuidado com o que
        $\sigma $-álgebra é usada no alvo?)}
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Epilogue.}
\begin{enumerate}[resume]
  \item Classifique as seguintes classes por inclusão estrita e
        justifique cada rigor com um exemplo deste
        capítulo e seu problema: conjuntos contáveis; Borel nulo
        conjuntos; Conjuntos nulos de Lebesgue; Conjuntos Borel; Conjuntos Lebesgue;
        conjuntos arbitrários.
\end{enumerate}

\medskip
\noindent\textbf{Part V --- The Cantor \omterm{def:b3:measure:measure}{medir}: mass on a null
set.} A escada é a função de distribuição de um
notável \omterm{def:b3:measure:measure}{medir}, que agora construímos com o próprio conteúdo deste capítulo
ferramentas.
\begin{enumerate}[resume]
  \item (Lebesgue--Stieltjes, existência) Deixe $ F\colon\R\to\R $
        ser não decrescente, \omterm{def:b3:topology:continuity}{contínuo}, limitado. Em meio aberto
        intervalos definir $\rho\bigl(\intoc ab\bigr) = F(b) -
        F(a)$ e, por $ A \subseteq \R $,
        \[
        \mu_F^*(A) = \inf\Bigl\{\sum_k\bigl(F(b_k) -
        F(a_k)\bigr) : A \subseteq
        \bigcup_k\intoc{a_k}{b_k}\Bigr\} .
        \]
        Mostre isso $\mu_F^*$ é um \omterm{def:b3:measure:outer}{medida externa} e isso
        $\mu_F^*\bigl(\intoc ab\bigr) = F(b) - F(a)$
        \emph{(imite o argumento da compacidade de
        \cref{thm:b3:measure:lebesgue}, ampliando cada
        $\intoc{a_k}{b_k}$ para um intervalo aberto em
        $ F $-custo $\leq \varepsilon2^{-k}$ --- onde está
        continuidade de $ F $ usado?)}.
  \item Mostre que todo conjunto de Borel é
        $\mu_F^*$-mensurável no Carathśentido eodoro
        \emph{(como no caso Lebesgue, basta testar
        meias linhas; segue a prova de
        \cref{thm:b3:measure:caratheodory}aplicação)}, para que $\mu_F = \mu_F^*$ restrito a $\mathcal B(\R)$
        é um \omterm{def:b3:measure:measure}{medir} com $\mu_F(\intoc ab) = F(b) - F(a)$:
        o \emph{Lebesgue--Stieltjes \omterm{def:b3:measure:measure}{medir}} de $ F $.
  \item Aplique isso à escada ($ F = c $ estendido por $0$
        sobre $\R_-$ e $1$ sobre $\intco1\infty $): o
        \emph{Cantor \omterm{def:b3:measure:measure}{medir}} $\mu $. Mostrar $\mu(\R) = 1$, isso
        cada lacuna do conjunto de Cantor é $\mu $-nulo ($ c $ é
        constante lá), e concluir
        \[
        \mu(C) = 1, \qquad \lambda(C) = 0 :
        \]
        $\mu $ e $\lambda $ vivem em operadoras disjuntas ($ C $ e
        seu complemento). Dois medidas nesta posição estão
        chamado \emph{mutuamente singulares}, escrito $\mu \perp
        \lambda $.
  \item Mostre isso $\mu $ não tem átomos: $\mu(\{x\}) = 0$ para
        cada $ x $ \emph{(continuidade de $ c $)}. Um sem átomo
        probabilidade \omterm{def:b3:measure:measure}{medir} transportado por um Lebesgue-nulo compactar:
        comparar com o único medidas visto até agora.
  \item (Lançamento de moeda disfarçado) Por uma palavra $(\varepsilon_1,
        \dots, \varepsilon_m) \in \{0,1\}^m $, deixar
        $ C_{\varepsilon}$ seja o conjunto de $ x \in C $ cujo ternário
        dígitos satisfazem $ b_i(x) = \varepsilon_i $ para $ i \leq m $
        (um dos $2^m $ Cantor pedaços de profundidade $ m $). Mostrar
        $\mu(C_\varepsilon) = 2^{-m}$ \emph{(a escada
        sobe $2^{-m}$ nessa peça: use a Parte I,
        pergunta 3)}. O Cantor \omterm{def:b3:measure:measure}{medir} é a lei de um
        sequência infinita de lançamentos justos de moedas lidos em ternário
        --- \cref{ch:b3:probability} fará isso exato.
  \item Prove a auto-similaridade: para todo Borel $ A $,
        \[
        \mu(A) = \tfrac12\,\mu(3A) +
        \tfrac12\,\mu(3A - 2) ,
        \]
        onde $3A - 2 = \{3x - 2 : x \in A\}$ \emph{(verifique
        nos intervalos de geração $\intoc ab $ através do
        relações auto-similares de $ c $ e, em seguida, invoque a exclusividade,
        \cref{thm:b3:measure:uniqueness})}.
  \item Mostre que a reflexão $ s(x) = 1 - x $ conservas
        $\mu $: $\mu(s(A)) = \mu(A)$ \emph{(através $ c(1 - x) = 1 -
        c(x)$, que segue da simetria do
        construção --- prove)}.
  \item Calcule os dois primeiros momentos de $\mu $, ou seja,\ de um
        ponto aleatório $ X $ com lei $\mu $ (as integrais podem ser
        tratados como limites de somas ao longo da profundidade $ m $ pedaços,
        antecipando \cref{ch:b3:lebesgue}): simetria dá
        $\int x\,\dd\mu = \frac12$, e a auto-similaridade dá
        \[
        \int x^2\,\dd\mu = \frac38,
        \qquad\text{hence}\qquad
        \operatorname{Var}(X) = \frac18 .
        \]
        Compare com a lei uniforme sobre $\intcc01$ (variância
        $\frac1{12}$): a massa de Cantor, empurrada para as bordas,
        spreads \emph{mais}.
  \item Mostre que (topológico) \emph{apoiar} de $\mu $
        --- o menor conjunto fechado de completo \omterm{def:b3:measure:measure}{medir} --- é
        exatamente $ C $.
  \item (Síntese) A escada $ c $ é \omterm{def:b3:topology:continuity}{contínuo} e
        não decrescente, mas falha no teorema fundamental de
        cálculo (Parte I); o \omterm{def:b3:measure:measure}{medir} $\mu_c $ é um
        probabilidade, sem átomo, singular em relação a
        $\lambda $. Explique em um pequeno parágrafo como eles são
        duas faces de um fenômeno e expor o geral
        moral: funções não decrescentes correspondem a medidas
        ($ F \leftrightarrow \mu_F $), diferenciabilidade a.e.\
        corresponde ao ``absolutamente \omterm{def:b3:topology:continuity}{contínuo} parte'', e
        $ c $ é a testemunha padrão de que um \omterm{def:b3:topology:continuity}{contínuo} $ F $ pode
        carregar \emph{não} absolutamente \omterm{def:b3:topology:continuity}{contínuo} parte em tudo.
  \item (O módulo agudo de continuidade) Deixar $ s =
        \frac{\ln 2}{\ln 3}$. Mostre isso $ c $ é Hölder
        \omterm{def:b3:topology:continuity}{contínuo} do expoente $ s $:
        \[
        \abs{c(x) - c(y)} \leq 4\,\abs{x - y}^{s}
        \qquad (x, y \in \intcc01),
        \]
        e que nenhum expoente $ t > s $ pode funcionar, mesmo localmente.
        Deduza a forma teórica da medida: para cada $ x $ e
        cada $ r \in \intoc01$,
        \[
        \mu\bigl(\intcc{x - r}{x + r}\bigr) \leq 8\,r^{s} .
        \]
        \emph{(Compare uma profundidade $ m $ grade triádica com a escala
        de $\abs{x - y}$; a questão 18 dá a subida
        cada peça. O expoente $ s $ é o \omterm{def:b3:topology:hausdorff}{Hausdorff} dimensão
        de $ C $, como dirão os cursos posteriores.)}
  \item (Auto-similaridade caracteriza $\mu $) Prove o
        inverso da questão 19: se $\nu $ é uma probabilidade
        \omterm{def:b3:measure:measure}{medir} sobre $\mathcal B(\R)$ carregado por $\intcc01$ e
        satisfatório
        \[
        \nu(A) = \tfrac12\,\nu(3A) + \tfrac12\,\nu(3A - 2)
        \qquad (A \in \mathcal B(\R)),
        \]
        então $\nu = \mu $. \emph{(Iterar a relação $ m $
        tempos para espalhar $\nu $ sobre o $2^m $ profundidade-$ m $ Cantor
        peças, estimativa $\nu(\intoc ab)$ contra a contagem de
        peças dentro $\intoc ab $, e deixe $ m \to \infty$;
        terminar com \cref{thm:b3:measure:uniqueness}.)}
\end{enumerate}
\end{problem}
