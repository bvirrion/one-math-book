\chapter{Funções características e o limite central
Teorema}\label{ch:b3:clt}

A \omterm{def:b3:probability:space}{lei} dos grandes números diz que as médias convergem; o \omterm{ex:b3:groups:actions}{centro}
teorema do limite diz \emph{como eles flutuam}: o erro,
ampliado por $\sqrt n $, é assintoticamente \omterm{def:b3:clt:gaussianvector}{Gaussiano} ---
seja qual for a \omterm{def:b3:probability:space}{lei} da qual se partiu. Esta universalidade é a
fato mais profundo da probabilidade elementar e sua prova natural
é analítico de Fourier: o \emph{\omterm{def:b3:clt:cf}{função característica}}
(a transformada de Fourier de uma \omterm{def:b3:probability:space}{lei}) converte \omterm{def:b3:probability:independence}{independente} somas
em produtos e \cref{ch:b3:fouriertransform}máquinas de
--- injetividade, \omterm{def:b3:clt:gaussianvector}{Gaussiano} pontos fixos --- converte ponto a ponto
convergência desses produtos em convergência de \omterm{def:b3:probability:space}{leis}
(EUteorema de evy, provado na íntegra). O capítulo termina com
Vetores gaussianos e a derivação honesta da confiança
intervalos usados em todas as estatísticas; o problema do fim de semana
fornece a segunda prova da CLT de Lindeberg, com uma explicação explícita
taxa de erro.

\section{Funções características}

\begin{definition}\label{def:b3:clt:cf}
O \emph{função característica}\index{função característica} de um verdadeiro variável aleatória $ X $ é
\[
\varphi_X(\xi) = \E\bigl[\eu^{\iu\xi X}\bigr]
= \int_\R \eu^{\iu\xi x}\,\dd\P_X(x)
\qquad (\xi \in \R)
\]
(o teorema da transferência calcula isso a partir da \omterm{def:b3:probability:space}{lei}; para um \omterm{ex:b3:lebesgue:gamma}{densidade}
$ f $, $\varphi_X(\xi) = \hat f(-\xi)$ em
\cref{ch:b3:fouriertransform}convenção).
\end{definition}

\begin{proposition}\label{prop:b3:clt:cfbasics}
(um) $\varphi_X(0) = 1$, $\abs{\varphi_X} \leq 1$, e
$\varphi_X $ é uniformemente \omterm{def:b3:topology:continuity}{contínuo};
$\varphi_{aX + b}(\xi) = \eu^{\iu b\xi}\varphi_X(a\xi)$.
(b) Se $ X, Y $ são \emph{\omterm{def:b3:probability:independence}{independente}}:
$\varphi_{X+Y} = \varphi_X\,\varphi_Y $.
(c) Se $\E\abs X^k < \infty $, então $\varphi_X \in \mathcal
C^k $ com $\varphi_X^{(j)}(0) = \iu^j\,\E[X^j]$ para $ j \leq
k $; em particular, para centrado $ X \in L^2$ com variação
$\sigma^2$:
\[
\varphi_X(\xi) = 1 - \frac{\sigma^2\xi^2}{2} +
o(\xi^2) \qquad (\xi \to 0).
\]
(d) \omterm{def:b3:clt:gaussianvector}{Gaussiano}: $ X \sim \mathcal N(m, \sigma^2)$ tem
$\varphi_X(\xi) = \eu^{\iu m\xi - \sigma^2\xi^2/2}$.
\end{proposition}

\begin{proof}
(a) Os limites são imediatos; continuidade: $\abs{\varphi(\xi + h)
- \varphi(\xi)} \leq \E\abs{\eu^{\iu hX} - 1} \to 0$ como $ h
\to 0$ por convergência dominada, uniformemente em $\xi $. O
regra afim é uma substituição. (b) $\eu^{\iu\xi(X+Y)} =
\eu^{\iu\xi X}\eu^{\iu\xi Y}$, e \omterm{def:b3:probability:space}{expectativas} de produtos
de \omterm{def:b3:probability:independence}{independente} fator de variáveis
(\cref{thm:b3:probability:independence}, aplicado a reais e
partes imaginárias). (c) Diferenciação sob o \omterm{def:b3:probability:space}{expectativa},
dominado por $\E\abs X^j $
(\cref{thm:b3:lebesgue:paramdiff}); a expansão de Taylor em
$0$ é então Taylor - Jovem para o $\mathcal C^2$ função
$\varphi $. (d) Para $\mathcal N(0,1)$: o \omterm{def:b3:clt:gaussianvector}{Gaussiano} transformar
(\cref{ex:b3:fouriertransform:gaussian} com $ a = \frac12$)
dá $\int\eu^{\iu\xi x}\frac{\eu^{-x^2/2}}{\sqrt{2\pi}}\dd
x = \eu^{-\xi^2/2}$; o caso geral pela regra afim.
\end{proof}

\begin{theorem}[Injetividade]\label{thm:b3:clt:injectivity}
Se $\varphi_X = \varphi_Y $, então $ X $ e $ Y $ tem o mesmo
\omterm{def:b3:probability:space}{lei}. Mais precisamente, para $ N \sim \mathcal N(0,1)$ \omterm{def:b3:probability:independence}{independente}
de $ X $ e $\varepsilon > 0$, a variável suavizada $ X +
\varepsilon N $ tem o \omterm{ex:b3:lebesgue:gamma}{densidade}
\[
p_\varepsilon(x) = \frac1{2\pi}\int_\R
\varphi_X(-\xi)\,\eu^{-\varepsilon^2\xi^2/2}\,
\eu^{\iu\xi x}\,\dd\xi ,
\]
determinado por $\varphi_X $ sozinho; deixando $\varepsilon \to 0$
recupera a \omterm{def:b3:probability:space}{lei} de $ X $.\index{função característica,
injetividade}
\end{theorem}

\begin{proof}
$ X + \varepsilon N $ tem o \omterm{ex:b3:lebesgue:gamma}{densidade} $ p_\varepsilon(x) =
\E\bigl[g_\varepsilon(x - X)\bigr]$, onde $ g_\varepsilon $ é
o $\mathcal N(0, \varepsilon^2)$ \omterm{ex:b3:lebesgue:gamma}{densidade}: de fato para o Borel
$ B $, independência e Tonelli dão $\P(X + \varepsilon N \in
B) = \int\!\!\int\mathbf 1_B(x + \varepsilon
n)g_1(n)\,\dd n\,\dd\P_X(x) = \int_B\E[g_\varepsilon(t -
X)]\dd t $ (substituto, depois Tonelli novamente). Escrita
$ g_\varepsilon $ pela inversão de Fourier de sua transformada
(\cref{exo:b3:fouriertransform:4}, redimensionado):
$ g_\varepsilon(u) = \frac1{2\pi}\int
\eu^{-\varepsilon^2\xi^2/2}\eu^{\iu\xi u}\dd\xi $ e Fubini
(tudo dominado pelo \omterm{def:b3:clt:gaussianvector}{Gaussiano} fator):
\[
p_\varepsilon(x) = \frac1{2\pi}\int_\R
\varphi_X(-\xi)\,\eu^{-\varepsilon^2\xi^2/2}\,
\eu^{\iu\xi x}\,\dd\xi ,
\]
um funcional de $\varphi_X $ sozinho. Se $\varphi_X =
\varphi_Y $: $ X + \varepsilon N $ e $ Y + \varepsilon N $ tem
\omterm{def:b3:probability:space}{leis} iguais para todos $\varepsilon $; para limitado \omterm{def:b3:topology:continuity}{contínuo}
$ f $, $\E f(X + \varepsilon N) \to \E f(X)$ como $\varepsilon
\to 0$ (convergência dominada, $ X + \varepsilon N \to X $
pontualmente no espaço do produto), então $\E f(X) = \E f(Y)$ para
tudo isso $ f $ --- e isso determina a \omterm{def:b3:probability:space}{lei}: para cada $ t $,
espremer $\mathbf 1_{\intoc{-\infty}t}$ entre os limites
\omterm{def:b3:topology:continuity}{contínuo} rampas $ f_k^\pm $ (igual a $1$ sobre
$\intoc{-\infty}{t \mp \frac1k}$, para $0$ além $ t \pm
\frac1k $, afim entre); passando ao limite em $\E
f_k^-(X) \leq F_X(t) \leq \E f_k^+(X)$ dá $ F_X(t) =
F_Y(t)$ a cada $ t $ onde ambos estão \omterm{def:b3:topology:continuity}{contínuo}, portanto
em todos os lugares por continuidade à direita e \omterm{ex:b3:lebesgue:gamma}{densidade} de comum
continuidade pontos (ambos $ F $ tem muitos saltos contáveis);
funções de distribuição igual forçam \omterm{def:b3:probability:space}{leis} iguais
(\cref{exo:b3:measure:3}, descansando em
\cref{thm:b3:measure:uniqueness}).
\end{proof}

\section{Convergência na distribuição}

\begin{definition}\label{def:b3:clt:cid}
$ X_n $ \emph{converge na distribuição} (ou na \omterm{def:b3:probability:space}{lei}) para $ X $,
escrito $ X_n \Rightarrow X $, se\index{convergência em
distribuição}
\[
\E\bigl[f(X_n)\bigr] \longrightarrow \E\bigl[f(X)\bigr]
\qquad\text{for every bounded continuous } f\colon\R\to\R .
\]
Equivalentemente (\cref{exo:b3:clt:4}): $ F_{X_n}(t) \to F_X(t)$
a cada continuidade apontar $ t $ de $ F_X $. O $ X_n $ não precisa
viver em comum \omterm{def:b3:probability:space}{espaço de probabilidade}: só as \omterm{def:b3:probability:space}{leis} importam.
\end{definition}

\begin{theorem}[Teorema da seleção de Helly]
\label{thm:b3:clt:helly}
Cada sequência $(F_n)$ de funções de distribuição tem um
subsequência convergindo ponto a ponto, em cada continuidade ponto
do limite, para um valor não decrescente \omterm{def:b3:topology:continuity}{contínuo} à direita $ G \colon
\R \to \intcc01$ --- possivelmente com $ G(+\infty) - G(-\infty) <
1$ (a massa pode escapar para o infinito).\index{Teorema de Helly}
\end{theorem}

\begin{proof}
A extração diagonal dá $ F_{n_k}(q) \to \ell(q)$ para cada
racional $ q $ (valores no compactar $\intcc01$). Definir
$ G(t) = \inf\{\ell(q) : q \in \Q, q > t\}$: não decrescente;
\omterm{def:b3:topology:continuity}{contínuo} à direita (um ínfimo sobre o encolhimento racional
bairros da direita). Em um continuidade apontar $ t $ de
$ G $: para racionais $ q_1 < t < q_2$,
\[
\ell(q_1) \leq \liminf F_{n_k}(t) \leq \limsup F_{n_k}(t)
\leq \ell(q_2),
\]
pela monotonicidade de cada um $ F_{n_k}$. Da definição de
$ G $ como um ínfimo e monotonicidade de $\ell $ no
racionais: $ G(s) \leq \ell(q) \leq G(q)$ em qualquer momento $ s < q $.
Tirando $ s < q_1 < t $ dá $\ell(q_1) \geq G(s)$, e
$\ell(q_2) \leq G(q_2)$; deixando $ s \uparrow t $ e $ q_2
\downarrow t $, continuidade de $ G $ no $ t $ aperta tanto o
$\liminf $ e o $\limsup $ para $ G(t)$.
\end{proof}

\begin{lemma}[Estanqueidade da função característica]
\label{lem:b3:clt:tightness}
Para qualquer variável aleatória $ X $ e $ u > 0$:
\[
\P\Bigl(\abs X \geq \frac2u\Bigr) \;\leq\;
\frac1u\int_{-u}^{u}\bigl(1 -
\operatorname{Re}\varphi_X(\xi)\bigr)\,\dd\xi .
\]
\end{lemma}

\begin{proof}
Por Tonelli--Fubini (integrando limitado, região finita em
$\xi $):
\[
\frac1u\int_{-u}^u\bigl(1 -
\operatorname{Re}\varphi_X(\xi)\bigr)\dd\xi
= \E\Bigl[\frac1u\int_{-u}^u(1 - \cos(\xi X))\,\dd\xi\Bigr]
= 2\,\E\Bigl[1 - \frac{\sin(uX)}{uX}\Bigr]
\]
(interprete o colchete como seu limite $0$ no $ X = 0$). O
integrando é não negativo ($\abs{\sin t} \leq \abs t $) e
para $\abs{uX} \geq 2$: $1 - \frac{\sin(uX)}{uX} \geq 1 -
\frac1{\abs{uX}} \geq \frac12$. Mantendo apenas o evento
$\{\abs{uX} \geq 2\}$ dentro do \omterm{def:b3:probability:space}{expectativa} portanto
sai pelo menos $2 \cdot \frac12\,\P(\abs X \geq \frac2u)$,
qual é a reivindicação.
\end{proof}

\begin{theorem}[euteorema da continuidade de evy]
\label{thm:b3:clt:levy}
Deixar $(X_n)$ ser variáveis aleatórias cujo característica
funções convergem pontualmente: $\varphi_{X_n}(\xi) \to
\varphi(\xi)$ para cada $\xi $, onde $\varphi =
\varphi_X $ é o \omterm{def:b3:clt:cf}{função característica} de alguns aleatório
variável $ X $. Então $ X_n \Rightarrow X $.\index{Lévy's
continuity theorem}
\end{theorem}

\begin{proof}
\emph{Aperto.} Consertar $\varepsilon > 0$. Desde $\varphi $ é
\omterm{def:b3:topology:continuity}{contínuo} no $0$ com $\varphi(0) = 1$, escolher $ u > 0$ com
$\frac1u\int_{-u}^u(1 - \operatorname{Re}\varphi) <
\varepsilon $; por convergência dominada (integrando limitado por
$2$ no fixo $[-u,u]$), a mesma integral para
$\varphi_{X_n}$ é $< 2\varepsilon $ para $ n $ grande:
\cref{lem:b3:clt:tightness} dá $\P(\abs{X_n} \geq \frac2u)
\leq 2\varepsilon $ para grande $ n $, e ampliando a constante
lida com um número finito de outros: as \omterm{def:b3:probability:space}{leis} são \emph{apertado}
--- sem fugas em massa.

\emph{Subsequências.} Deixar $(F_{n_k})$ seja qualquer subsequência; por
Helly (\cref{thm:b3:clt:helly}) extrair $ F_{n_{k_j}} \to G $
no continuidade pontos. Forças de estanqueidade $ G(-\infty) = 0$,
$ G(+\infty) = 1$ ($ G(\frac2u) - G(-\frac2u) \geq 1 -
2\varepsilon $ no continuidade pontos): $ G $ é um genuíno
função de distribuição, de alguns variável aleatória $ Y $. Então
$ X_{n_{k_j}} \Rightarrow Y $ (\cref{exo:b3:clt:4},
convergência distributiva de $ F $ de), então
$\varphi_{X_{n_{k_j}}} \to \varphi_Y $ \emph{pontualmente} ($ x
\mapsto \eu^{\iu\xi x}$ é limitado \omterm{def:b3:topology:continuity}{contínuo}, verdadeiro e
partes imaginárias separadamente); comparando com a hipótese:
$\varphi_Y = \varphi = \varphi_X $ e injetividade
(\cref{thm:b3:clt:injectivity}) dá $ Y \sim X $, ou seja\ $ G =
F_X $.

\emph{\omterm{thm:b3:complete:completion}{Conclusão}.} Cada subsequência de $(F_n)$ tem um
sub-subsequência convergindo para \emph{mesmo} $ F_X $ (no seu
continuidade pontos); por isso $ F_n(t) \to F_X(t)$ a cada
continuidade apontar $ t $ (uma sequência real cuja
subsequências têm subsubsequências com o mesmo limite
converge): $ X_n \Rightarrow X $.
\end{proof}

\section{O teorema do limite central}

\begin{theorem}[Teorema do limite central]\label{thm:b3:clt:clt}
Deixar $(X_n)$ ser i.i.d.\ com $\E X_1 = m $ e $\V(X_1) =
\sigma^2 \in \intoo0\infty $. Então\index{limite central
teorema}
\[
\frac{S_n - nm}{\sigma\sqrt n} \;\Longrightarrow\; \mathcal
N(0, 1) :
\qquad
\P\Bigl(a \leq \frac{S_n - nm}{\sigma\sqrt n} \leq
b\Bigr) \longrightarrow
\frac{1}{\sqrt{2\pi}}\int_a^b\eu^{-x^2/2}\,\dd x
\]
para todos $ a < b $.
\end{theorem}

\begin{proof}
\omterm{ex:b3:groups:actions}{Centro} e normalizar: $ Z_i = \frac{X_i - m}{\sigma}$
(i.i.d., significa $0$, variação $1$) e $ T_n =
\frac1{\sqrt n}\sum_{i\leq n}Z_i $. Por independência e o
regra afim (\cref{prop:b3:clt:cfbasics}):
\[
\varphi_{T_n}(\xi) =
\varphi_{Z}\Bigl(\frac{\xi}{\sqrt n}\Bigr)^{n},
\qquad
\varphi_Z(\eta) = 1 - \frac{\eta^2}2 + \eta^2\rho(\eta),\quad
\rho(\eta)\to0 .
\]
Consertar $\xi $ e deixe $ a_n = \varphi_Z(\xi/\sqrt n)$, $ b_n = 1 -
\frac{\xi^2}{2n}$: ambos têm módulo $\leq 1$ para $ n $ grande
($\abs{b_n} \leq 1$ uma vez $\xi^2 \leq 4n $; $\abs{a_n} \leq 1$
sempre). A desigualdade elementar $\abs{a^n - b^n} \leq
n\abs{a - b}$ para $\abs a, \abs b \leq 1$ (telescópio
$ a^n - b^n = \sum a^k(a - b)b^{n-1-k}$) dá
\[
\Bigl|\varphi_{T_n}(\xi) - \Bigl(1 -
\frac{\xi^2}{2n}\Bigr)^{n}\Bigr|
\leq n\,\Bigl|\varphi_Z\Bigl(\frac\xi{\sqrt n}\Bigr) - 1 +
\frac{\xi^2}{2n}\Bigr|
= \xi^2\,\Bigl|\rho\Bigl(\frac{\xi}{\sqrt n}\Bigr)\Bigr|
\longrightarrow 0,
\]
enquanto $\bigl(1 - \frac{\xi^2}{2n}\bigr)^n \to
\eu^{-\xi^2/2}$ (logaritmo real). Então $\varphi_{T_n}(\xi) \to
\eu^{-\xi^2/2} = \varphi_{\mathcal N(0,1)}(\xi)$
(\cref{prop:b3:clt:cfbasics}(d)) para cada $\xi $: euévy
(\cref{thm:b3:clt:levy}) conclui $ T_n \Rightarrow \mathcal
N(0,1)$. As probabilidades de intervalo seguem desde $ F_{\mathcal
N}$ é \omterm{def:b3:topology:continuity}{contínuo} em todos os lugares.
\end{proof}

\begin{example}[Intervalos de confiança, derivados honestamente]
\label{ex:b3:clt:confidence}
Enquete $ n $ \omterm{def:b3:probability:independence}{independente} eleitores; $\hat p_n = S_n/n $ estima o
verdade $ p $, com $\sigma^2 = p(1-p) \leq \frac14$. A CLT
dá, por grande $ n $,
\[
\P\Bigl(\abs{\hat p_n - p} \leq
\frac{z}{2\sqrt n}\Bigr)
\;\geq\; \P\Bigl(\Bigl|\frac{S_n - np}{\sigma\sqrt n}\Bigr|
\leq z\Bigr)
\longrightarrow \Phi(z) - \Phi(-z),
\]
onde $\Phi $ é o padrão \omterm{def:b3:clt:gaussianvector}{Gaussiano} função de distribuição.
Com $ z = 1.96$: confiança assintótica $95\%$ e margem
$\frac{1.96}{2\sqrt n} \leq 3\%$ requer $ n \geq
\bigl(\frac{1.96}{0.06}\bigr)^2 \approx 1068$ --- o número
por trás de cada ``$\pm3$ pontos, $95\%$'' se lê; comparar
Chebyshev $5556$ (\cref{exo:b3:probability:7}). O
$\sqrt n $ é universal: para reduzir pela metade o erro, quadruplicar o
amostra --- a mesma \omterm{def:b3:probability:space}{lei} que fixa Monte Carlocusto
(\cref{exo:b3:clt:7}).
\end{example}

\section{Vetores gaussianos}

\begin{definition}\label{def:b3:clt:gaussianvector}
Um vetor aleatório $ X = (X_1, \dots, X_d)$ é
\emph{Gaussiano}\index{Gaussiano} se todo linear
combinação $\langle t, X\rangle = \sum t_iX_i $ é um
(possivelmente degenerada) variável gaussiana real. Sua \omterm{def:b3:probability:space}{lei} é
determinado pelo vetor médio $ m = (\E X_i)$ e o
\emph{matriz de covariância} $\Sigma = \bigl(\operatorname{Cov}
(X_i, X_j)\bigr)$: na verdade o \omterm{def:b3:clt:cf}{função característica} do
vetor, $\varphi_X(t) = \E\eu^{\iu\langle t, X\rangle}$, é
o valor em $1$ do cf de $\langle t, X\rangle $:
\[
\varphi_X(t) = \exp\Bigl(\iu\langle t, m\rangle -
\tfrac12\,t^{\mathsf T}\Sigma\,t\Bigr),
\]
e $ d $-dimensional funções características são injetivos
(mesma prova de suavização que \cref{thm:b3:clt:injectivity},
Gaussianos coordenados).
\end{definition}

\begin{theorem}\label{thm:b3:clt:gaussianvector}
Deixar $ X $ seja um Vetor gaussiano.
\begin{enumerate}
  \item Cada imagem afim $ AX + b $ é um Vetor gaussiano.
  \item Os componentes $ X_i $ são \emph{\omterm{def:b3:probability:independence}{independente}} se e
        somente se $\Sigma $ é diagonal: para conjuntamente \omterm{def:b3:clt:gaussianvector}{Gaussiano}
        variáveis, não correlacionadas $=$ \omterm{def:b3:probability:independence}{independente}.
  \item Se $\Sigma $ é invertível, $ X $ tem o \omterm{ex:b3:lebesgue:gamma}{densidade}
        $\frac{1}{(2\pi)^{d/2}\sqrt{\det\Sigma}}
        \exp\bigl(-\frac12(x - m)^{\mathsf T}\Sigma^{-1}(x -
        m)\bigr)$.
\end{enumerate}
\end{theorem}

\begin{proof}
(1) Combinações lineares de componentes de $ AX + b $ são afins
funções de combinações lineares de $ X $: \omterm{def:b3:clt:gaussianvector}{Gaussiano} (um afim
imagem de um \omterm{def:b3:clt:gaussianvector}{Gaussiano} variável é \omterm{def:b3:clt:gaussianvector}{Gaussiano}).
(2) Se $\Sigma $ é diagonal, o \omterm{def:b3:clt:cf}{função característica}
fatora: $\varphi_X(t) = \prod_i\exp(\iu t_im_i -
\frac12\Sigma_{ii}t_i^2) = \prod\varphi_{X_i}(t_i)$, que é
o \omterm{def:b3:clt:cf}{função característica} da \omterm{def:b3:probability:space}{lei} do produto
(\cref{thm:b3:probability:independence} leia
$ d $ injetividade -dimensional): os componentes são \omterm{def:b3:probability:independence}{independente}.
O inverso é o desaparecimento das covariâncias de \omterm{def:b3:probability:independence}{independente}
$ L^2$ variáveis.
(3) Diagonalizar $\Sigma = P D P^{\mathsf T}$ ($ P $ ortogonal,
$ D > 0$ diagonal --- \cref{exo:b3:submanifolds:8}); o
vetor $ Y = P^{\mathsf T}(X - m)$ é \omterm{def:b3:clt:gaussianvector}{Gaussiano} com
covariância $ D $: por (2) seus componentes são \omterm{def:b3:probability:independence}{independente}
$\mathcal N(0, d_i)$, então $ Y $ tem o produto \omterm{ex:b3:lebesgue:gamma}{densidade}; empurrar
avançar pela preservação de volume $ x = m + PY $
(\cref{thm:b3:product:linearchange}, $\abs{\det P} = 1$) e
reescreva o expoente invariavelmente.
\end{proof}

\begin{theorem}[CLT Multidimensional]
\label{thm:b3:clt:multiclt}
Deixar $(X_n)$ ser i.i.d.\ quadrado \omterm{def:b3:lebesgue:l1}{integrável} aleatório
\emph{vetores} de $\R^d $ com média $ m $ e matriz de covariância
$\Sigma $. Então $\frac{S_n - nm}{\sqrt n}$ converge em
distribuição para o Vetor gaussiano $\mathcal N(0,
\Sigma)$.\index{teorema do limite central!multidimensional}
\admitted
\end{theorem}

\begin{remark}\label{rem:b3:clt:multiclt}
Quase tudo já está em nossas mãos. Para cada
direção $ t \in \R^d $, a variável real $\langle t,
\frac{S_n - nm}{\sqrt n}\rangle $ é uma soma normalizada de
i.i.d.\ variáveis reais de variância $ t^{\mathsf T}\Sigma t $,
então \cref{thm:b3:clt:clt}o cálculo de dá ponto
convergência do $ d $-dimensional funções características
para $\eu^{-t^{\mathsf T}\Sigma t/2}$, o característica
função de $\mathcal N(0, \Sigma)$
(\cref{def:b3:clt:gaussianvector}). O que não temos
reprovado é Lévy continuidade teorema \emph{em $\R^d $}:
A seleção de Helly e a estimativa de estanqueidade generalizam
rotineiramente (em coordenadas), e este Cramér--Wold
a redução é realizada honestamente em qualquer pós-graduação
curso de probabilidade; nada além dos métodos deste capítulo é
necessário.
\end{remark}

\begin{method}\label{met:b3:clt:method}
Para identificar uma \omterm{def:b3:probability:space}{lei} limite: calcule funções características,
tome o limite pontual, reconheça-o (\omterm{def:b3:clt:gaussianvector}{Gaussiano}
$\eu^{-\sigma^2\xi^2/2}$, Poisson
$\eu^{\lambda(\eu^{\iu\xi}-1)}$, exponencial $\frac{\lambda}
{\lambda - \iu\xi}$, \dots) e invocar Lévy. Os três passos
ritual (independência $\to $ produto; Taylor em $0$ $\to $
limite exponencial; euévy $\to $ convergência jurídica) prova
a CLT, a \omterm{def:b3:probability:space}{lei} de Poisson dos eventos raros
(\cref{exo:b3:clt:5}), e todo teorema clássico do limite de
este curso. Para declarações a.s.\, volte para
\cref{ch:b3:probability}kit de ferramentas: os dois capítulos respondem
perguntas diferentes sobre o mesmo $ S_n $.
\end{method}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:b3:clt:1}
Calcule o funções características: uniforme ligado
$\intcc{-1}1$; exponencial $\mathcal E(\lambda)$; Poisson
$\mathcal P(\lambda)$; binômio $\mathcal B(n, p)$. Deduzir
através de \cref{thm:b3:clt:injectivity} que a soma de \omterm{def:b3:probability:independence}{independente}
Variáveis de Poisson ($\lambda, \mu $) é Poisson $(\lambda +
\mu)$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:clt:2}
(a) Mostre que $\varphi_X $ tem valor real se e somente se $ X $
e $-X $ têm a mesma \omterm{def:b3:probability:space}{lei} (uma \emph{simétrico} variável).
(b) Suponha $\abs{\varphi_X(\xi_0)} = 1$ para alguns $\xi_0 \neq
0$. Mostre isso $ X $ é quase certamente apoiado em uma aritmética
progressão $ a + \frac{2\pi}{\xi_0}\Z $ \emph{(escrever
$\varphi_X(\xi_0) = \eu^{\iu\theta}$ e calcular $\E[1 -
\cos(\xi_0X - \theta)]$)}. Deduza que se $ X $ tem um \omterm{ex:b3:lebesgue:gamma}{densidade},
então $\abs{\varphi_X(\xi)} < 1$ para todos $\xi \neq 0$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:clt:3}
Deixar $ X \sim \mathcal N(m_1, \sigma_1^2)$ e $ Y \sim
\mathcal N(m_2, \sigma_2^2)$ ser \omterm{def:b3:probability:independence}{independente}. Mostrar $ X + Y
\sim \mathcal N(m_1 + m_2, \sigma_1^2 + \sigma_2^2)$ e
mais geralmente que o \omterm{def:b3:clt:gaussianvector}{Gaussiano} família é \omterm{def:b3:ode:stability}{estável} sob
\omterm{def:b3:probability:independence}{independente} somas e mapas afins. Contraste: é a soma de
dois \emph{dependente} \omterm{def:b3:clt:gaussianvector}{Gaussianos} sempre \omterm{def:b3:clt:gaussianvector}{Gaussiano}?
(\cref{exo:b3:clt:9}.)
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:clt:4}
(a) Prove a equivalência em \cref{def:b3:clt:cid}: se
$\E f(X_n) \to \E f(X)$ para todos os limites \omterm{def:b3:topology:continuity}{contínuo} $ f $,
então $ F_{X_n}(t) \to F_X(t)$ no continuidade pontos
\emph{(espremer $\mathbf 1_{\intoc{-\infty}t}$ entre dois
\omterm{def:b3:topology:continuity}{contínuo} rampas de escadas)}; e inversamente
\emph{(aproximadamente um limite \omterm{def:b3:topology:continuity}{contínuo} $ f $ por somas de rampa
funções, ou condição em uma grade fina de continuidade
pontos)} --- o inverso pode ser tratado para $ f $ uniformemente
\omterm{def:b3:topology:continuity}{contínuo} primeiro, depois em geral.
(b) Mostre que $ X_n \Rightarrow c $ (uma constante) implica $ X_n
\to c $ em probabilidade.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:clt:5}
(\omterm{def:b3:probability:space}{Lei} dos eventos raros) Deixe $ X_n \sim \mathcal B(n, p_n)$ com
$ np_n \to \lambda > 0$. Mostrar, através funções características
e \cref{thm:b3:clt:levy}, que $ X_n \Rightarrow \mathcal
P(\lambda)$. Verificação de integridade numérica: compare $\P(X = 0)$ para
$\mathcal B(100, 0.02)$ e $\mathcal P(2)$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:clt:6}
(a) Um dado justo é lançado $ n = 1000$ vezes; aproximar o
probabilidade de que o total exceda $3600$ (significar $3500$,
variação por rolo $\frac{35}{12}$).
(b) Para $ S \sim \mathcal B(100, \frac12)$, aproximado
$\P(45 \leq S \leq 55)$ pela CLT com o continuidade
correção ($\pm\frac12$) e comente a correção
efeito.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:clt:7}
(\omterm{ex:b3:probability:sllnapps}{Monte Carlo} erro) Na configuração de
\cref{pb:b3:probability:1}, questão 11, com $ g \in
L^2(\intcc01^d)$, deixar $\sigma^2 = \V(g(U_1))$ e $ I = \int
g $. Mostrar
\[
\sqrt n\,\Bigl(\frac1n\sum_{k\leq n}g(U_k) - I\Bigr)
\Longrightarrow \mathcal N(0, \sigma^2),
\]
e deduzir o assintótico $95\%$ barra de erro $\pm
1.96\,\sigma/\sqrt n $ --- \omterm{def:b3:probability:independence}{independente} da dimensão $ d $.
Compare com a regra determinística do ponto médio na dimensão
$ d $ (erro $\sim n^{-2/d}$ para $\mathcal C^2$ integrandos):
de qual dimensão a amostragem aleatória vence?
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b3:clt:8}
(Slutsky) Suponha $ X_n \Rightarrow X $ e $ Y_n \to c $ em
probabilidade ($ c $ constante). Mostrar $ X_n + Y_n \Rightarrow X +
c $ e $ Y_nX_n \Rightarrow cX $. \emph{(Trabalhar com
funções características e o limite $\abs{\E\eu^{\iu\xi
(X_n+Y_n)} - \eu^{\iu\xi c}\E\eu^{\iu\xi X_n}} \leq
\E\abs{\eu^{\iu\xi(Y_n - c)} - 1}$, dividir em $\abs{Y_n - c}
\leq \delta $.)} Aplicação: em
\cref{ex:b3:clt:confidence}, justifique a substituição do desconhecido
$\sigma = \sqrt{p(1-p)}$ por $\sqrt{\hat p_n(1 - \hat
p_n)}$.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b3:clt:9}
Deixar $ X \sim \mathcal N(0,1)$ e $\varepsilon $ \omterm{def:b3:probability:independence}{independente}
com $\P(\varepsilon = \pm1) = \frac12$; definir $ Y =
\varepsilon X $.
(a) Mostrar $ Y \sim \mathcal N(0,1)$ e
$\operatorname{Cov}(X, Y) = 0$.
(b) Mostrar $ X $ e $ Y $ são \emph{não} \omterm{def:b3:probability:independence}{independente}, e isso
$(X, Y)$ não é um Vetor gaussiano \emph{(calcular $\P(X + Y =
0)$)}.
(c) Moral: \cref{thm:b3:clt:gaussianvector}(2) requer
Gaussianidade conjunta --- ``não correlacionada \omterm{def:b3:clt:gaussianvector}{Gaussianos}'' sozinho
não prova nada.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:clt:10}
A \omterm{def:b3:probability:space}{lei} Cauchy \omterm{ex:b3:lebesgue:gamma}{densidade} $\frac1{\pi(1 + x^2)}$.
(a) Mostre seu \omterm{def:b3:clt:cf}{função característica} é $\eu^{-\abs\xi}$
(\cref{exo:b3:fouriertransform:1} e inversão).
(b) Mostre que se $ X_1, \dots, X_n $ são i.i.d.\ Cauchy, então
$\frac{S_n}n $ é novamente Cauchy --- o \emph{mesmo} \omterm{def:b3:probability:space}{lei}: a
a média nunca se concentra.
(c) Conciliar com as \omterm{def:b3:probability:space}{leis} dos grandes números e a CLT:
quais hipóteses falham? (Calcular $\E\abs{X_1}$.)
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:clt:11}
(\omterm{def:b3:probability:space}{Leis} estáveis em embrião) Vamos $(X_n)$ ser i.i.d.\ padrão
Cauchy (\cref{exo:b3:clt:10}).
(a) Mostre que para qualquer $ a, b > 0$, $ aX_1 + bX_2$ tem a \omterm{def:b3:probability:space}{lei}
de $(a + b)X_1$: a família Cauchy é \emph{estritamente
\omterm{def:b3:ode:stability}{estável}} do \omterm{def:b3:holomorphic:index}{índice} $1$.
(b) Mostre que \omterm{def:b3:clt:gaussianvector}{Gaussiano} família é estritamente \omterm{def:b3:ode:stability}{estável}
\omterm{def:b3:holomorphic:index}{índice} $2$: $ aX_1 + bX_2 \sim \sqrt{a^2 + b^2}\,X_1$ para
$ X_i $ eu.i.d.\ $\mathcal N(0,1)$.
(c) Explique, via funções características da forma
$\eu^{-c\abs\xi^\alpha}$, por que indexar $\alpha $ estabilidade
força a normalização $ n^{1/\alpha}$ para somas, e o que
isto diz sobre as bacias de atração da CLT: quais
i.i.d.\ somas podem convergir, após normalização afim, para um
\omterm{def:b3:probability:space}{Lei} de Cauchy, em vez de uma \omterm{def:b3:clt:gaussianvector}{Gaussiano}?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:clt:12}
(A função de distribuição empírica) Deixe $(X_n)$ ser i.i.d.\
com função de distribuição $ F $, e $ F_n(t) =
\frac1n\#\{k \leq n : X_k \leq t\}$.
(a) Correção $ t $. Mostre isso $ n F_n(t) \sim \mathcal B(n, F(t))$,
que $ F_n(t) \to F(t)$ como.\ (\cref{thm:b3:probability:slln}),
e isso
\[
\sqrt n\,\bigl(F_n(t) - F(t)\bigr) \Longrightarrow
\mathcal N\bigl(0,\ F(t)(1 - F(t))\bigr) .
\]
(b) Em que $ t $ a variância assintótica é máxima?
Interprete: a mediana é onde uma distribuição empírica está
mais difícil de definir.
(c) Para $ F $ \omterm{def:b3:topology:continuity}{contínuo}, mostre que a \omterm{def:b3:probability:space}{lei} de
$\sup_t\abs{F_n(t) - F(t)}$ não depende de $ F $
\emph{(reduzir para variáveis uniformes via
\cref{exo:b3:probability:1})} --- o \omterm{def:b3:modules:free}{livre} de distribuição
milagre por trás do teste Kolmogorov-Smirnov; sem cálculo
dessa \omterm{def:b3:probability:space}{lei} é perguntado.
\end{exercise}

\section{Problema: A prova de Lindeberg da CLT, com um
taxa}

\begin{problem}[{Problema de fim de semana --- o método de substituição}]
\label{pb:b3:clt:1}
Lindeberg (1922) provou o teorema do limite central por meio de uma ideia
de simplicidade desarmante: \emph{troque as somas uma de cada
hora de \omterm{def:b3:clt:gaussianvector}{Gaussianos}} e controlar cada troca por um Taylor
expansão. O método não necessita de análise de Fourier, produz uma
taxa de erro explícita e hoje fornece provas de universalidade
através da teoria da probabilidade. Deixar $(X_i)$ ser i.i.d., centrado,
$\V(X_1) = 1$, com $\beta = \E\abs{X_1}^3 < \infty $; deixar
$(N_i)$ ser i.i.d.\ $\mathcal N(0,1)$, \omterm{def:b3:probability:independence}{independente} do
$ X_i $ (existência: \cref{thm:b3:probability:existence}). Definir
\[
T_n = \frac{X_1 + \dots + X_n}{\sqrt n},
\qquad
G_n = \frac{N_1 + \dots + N_n}{\sqrt n} \sim \mathcal N(0,1).
\]

\medskip
\noindent\textbf{Part I --- The swapping identity.} Consertar $ f
\in \mathcal C^3_b(\R)$ (três delimitados \omterm{def:b3:topology:continuity}{contínuo}
derivados; $ M_3 = \sup\abs{f'''}$). Para $0 \leq i \leq n $
definir as somas híbridas
\[
H_i = \frac{X_1 + \dots + X_i + N_{i+1} + \dots +
N_n}{\sqrt n},
\]
então $ H_n = T_n $ e $ H_0 = G_n $.
\begin{enumerate}
  \item Escrever $ H_i = W_i + \frac{X_i}{\sqrt n}$ e $ H_{i-1}
        = W_i + \frac{N_i}{\sqrt n}$ com $ W_i =
        \frac{1}{\sqrt n}\bigl(\sum_{j<i}X_j +
        \sum_{j>i}N_j\bigr)$, e observe que $ W_i $ é
        \omterm{def:b3:probability:independence}{independente} do par $(X_i, N_i)$. Justificar.
  \item Taylor com resto integral ou Lagrange: para qualquer
        verdadeiro $ w, h $:
        \[
        \Bigl|f(w + h) - f(w) - f'(w)h -
        \tfrac12f''(w)h^2\Bigr| \leq
        \frac{M_3\,\abs h^3}{6} .
        \]
  \item Aplique a questão 2 duas vezes ($ h = \frac{X_i}{\sqrt n}$ e
        $ h = \frac{N_i}{\sqrt n}$ no $ w = W_i $), pegar
        expectativase usar independência mais a correspondência
        dos dois primeiros momentos $ X_i $ e $ N_i $ mostrar
        \[
        \bigl|\E f(H_i) - \E f(H_{i-1})\bigr|
        \leq \frac{M_3}{6}\cdot
        \frac{\beta + \gamma}{n^{3/2}},
        \qquad \gamma = \E\abs{N_1}^3 =
        \frac{2\sqrt2}{\sqrt\pi} .
        \]
  \item Telescópio acima $ i $ e concluir o \emph{Lindeberg
        vinculado}:
        \[
        \bigl|\E f(T_n) - \E f(G_n)\bigr| \leq
        \frac{M_3\,(\beta + \gamma)}{6\,\sqrt n} .
        \]
\end{enumerate}

\medskip
\noindent\textbf{Part II --- From smooth $ f $ to the CLT.}
\begin{enumerate}[resume]
  \item Mostre isso $\E f(T_n) \to \E f(N)$ para cada $ f \in
        \mathcal C_b^3$ e atualize para todos os limites
        \omterm{def:b3:topology:continuity}{contínuo} $ f $: dado tal $ f $ e $\varepsilon $,
        construir $ f_\varepsilon \in \mathcal C^3_b $ com
        $\norm{f - f_\varepsilon}_\infty \leq \varepsilon $
        em um intervalo grande --- por exemplo\ convolve $ f $ com um
        $\mathcal C^\infty $ ressalto
        (\cref{thm:b3:lp:regularization}) --- e lidar com o
        caudas por aperto ($\V(T_n) = 1$ e Chebyshev).
        Concluir $ T_n \Rightarrow \mathcal N(0, 1)$: o
        teorema do limite central, reprovado.
  \item Onde a prova usou que o $ X_i $ são
        \emph{de forma idêntica} distribuído? Mostre que mal
        fez: declarar e provar a versão para \omterm{def:b3:probability:independence}{independente},
        centrado, não idêntico $ X_i $ com
        $\sum_i\V(X_i) = s_n^2$ e terceiro momentos, obtendo
        o erro $\frac{M_3}{6s_n^3}\sum_i\bigl(\E\abs{X_i}^3
        + \V(X_i)^{3/2}\gamma\bigr)$ --- Lindeberg é verdade
        teorema em sua forma Lyapunov.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Quantitative dividends.}
\begin{enumerate}[resume]
  \item (Funções de distribuição) Deixe $ t \in \R $ e
        aproximado $\mathbf 1_{\intoc{-\infty}t}$ acima e
        abaixo por $\mathcal C^3_b $ rampas de largura $\delta $
        (construa-os, com $ M_3 = O(\delta^{-3})$).
        Combinando com a Parte I, derive o limite de dois termos
        \[
        \sup_{t\in\R}\,\bigl|\P(T_n \leq t) -
        \Phi(t)\bigr| \;\leq\;
        \frac{C_1(\beta + \gamma)}{\delta^3\sqrt n} +
        C_2\,\delta
        \qquad (\text{every } \delta > 0),
        \]
        com constantes explícitas (o $ C_2\delta $ usos de termos
        isso $\Phi $ tem \omterm{ex:b3:lebesgue:gamma}{densidade} limitado por
        $\frac1{\sqrt{2\pi}}$) e otimizar $\delta \sim
        n^{-1/8}$ para obter uma taxa uniforme de ordem
        $ n^{-1/8}$.
        (O \omterm{def:b3:rings:ideal}{ideal} $ n^{-1/2}$ --- Berry--Esseen ---
        precisa de ferramentas mais sofisticadas; o ponto é um \emph{explícito}
        taxa de troca elementar.)
  \item (De Moivre--Laplace, quantificado) Especializar-se em
        $ X_i = 2B_i - 1$ (sinais de moedas justas): compare o
        \omterm{thm:b3:complete:completion}{conclusão} com a estimativa local de
        \cref{pb:b3:product:1}, pergunta 7 --- o que
        cada método dá que o outro não?
  \item (Universalidade) Explique em um parágrafo por que o
        método de substituição mostra mais que a CLT: qualquer
        estatística do formulário $\E f(\text{sum})$ com $ f $
        suave é insensível, na ordem $ n^{-1/2}$, para o
        \emph{\omterm{def:b3:probability:space}{lei} inteira} dos resumos além de seu primeiro
        dois momentos --- o ``princípio da invariância'' que
        está subjacente aos resultados da universalidade moderna (aleatório
        matrizes, polinômios aleatórios), dos quais o CLT é
        a primeira instância.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Smoothing, pushed: better
rates.} A perda de $ n^{-1/2}$ (suave $ f $) para
$ n^{-1/8}$ (funções de distribuição) vieram do carregamento
$ f'''$ na norma sup. Os híbridos podem reparar parte disso: eles
conter \omterm{def:b3:clt:gaussianvector}{Gaussiano} convocações, e \omterm{def:b3:clt:gaussianvector}{Gaussianos} \emph{suave}.
\begin{enumerate}[resume]
  \item (Um oculto \omterm{def:b3:clt:gaussianvector}{Gaussiano}) Para $1 \leq i \leq n - 1$, $ h =
        \frac{X_i}{\sqrt n}$ ou $\frac{N_i}{\sqrt n}$, e
        $\theta \in \intcc01$, escrever $ W_i + \theta h = A +
        Z $ com $ Z = \frac{N_{i+1} + \dots + N_n}{\sqrt
        n}$. Mostre isso $ Z \sim \mathcal N\bigl(0,
        \frac{n-i}n\bigr)$ é \omterm{def:b3:probability:independence}{independente} do par $(A,
        h)$, e deduza, para cada \omterm{def:b3:topology:continuity}{contínuo} $ g \in
        L^1(\R)$,
        \[
        \E\bigl[\abs h^3\,\abs{g(W_i + \theta h)}\bigr]
        \;\leq\; \sqrt{\frac{n}{2\pi(n - i)}}\;
        \norm{g}_{L^1}\;\E\abs h^3 .
        \]
  \item Combine a questão 10 com a forma integral do
        Resto de Taylor,
        \[
        f(w + h) = f(w) + f'(w)h + \tfrac12f''(w)h^2 +
        \int_0^1\frac{(1 - \theta)^2}2\,f'''(w + \theta
        h)\,h^3\,\dd\theta,
        \]
        para refazer as questões 3-4: para $ f \in \mathcal C^3_b $
        com além disso $ f''' \in L^1(\R)$,
        \[
        \bigl|\E f(T_n) - \E f(G_n)\bigr| \leq
        \frac{\beta + \gamma}{3\sqrt{2\pi}}\cdot
        \frac{\norm{f'''}_{L^1}}{\sqrt n} +
        \frac{M_3(\beta + \gamma)}{6\,n^{3/2}}
        \]
        \emph{(a questão 10 trata das trocas $ i \leq n - 1$
        --- usar $\sum_{m=1}^{n-1}m^{-1/2} \leq 2\sqrt n $
        --- e o limite bruto da pergunta 3 trata do último
        um)}. Verifique se as rampas da questão 7 satisfazem
        $\norm{\psi_\delta'''}_{L^1} = K_1\delta^{-2}$
        enquanto $ M_3 = K\delta^{-3}$, alimente-os e
        otimizar $\delta $: o uniforme
        taxa de função de distribuição melhora para
        $ O(n^{-1/6})$.
  \item (Correspondendo mais um momento) Suponha além disso $\E
        X_1^3 = 0$ e $\beta_4 = \E X_1^4 < \infty $.
        Calcular $\E N_1^3$ e $\E N_1^4$, expanda para o quarto
        ordem e provar da mesma forma que
        taxa de função de distribuição torna-se $ O(n^{-1/4})$
        \emph{(agora $\norm{\psi_\delta^{(4)}}_{L^1} =
        K_2\delta^{-3}$ e $ M_4 = K'\delta^{-4}$; escolher
        $\delta = n^{-1/4}$)}.
  \item (A obstrução) Suponha que o primeiro $ k $ momentos de
        $ X_1$ concordo com o \omterm{def:b3:clt:gaussianvector}{Gaussiano} aqueles ($ k = 2$ sempre;
        $ k = 3$ exatamente quando $\E X_1^3 = 0$; $ k \geq 4$
        essencialmente nunca, como $\E N_1^4 = 3$). Verifique isso
        o esquema das perguntas 10 a 12 fornece o
        taxa de função de distribuição
        $ n^{-(k-1)/(2k+2)}$, equilibrando
        $\delta^{-k}n^{-(k-1)/2}$ contra $\delta $ e
        observe que o expoente se aproxima do
        Berry - Valor Esseen $\frac12$ apenas como $ k \to
        \infty $. Explique em poucas frases por que
        método de troca satura: cada troca é cobrada em
        valor absoluto, enquanto a rota de Fourier
        (desigualdade de suavização de Esseen) explora o
        oscilação da \omterm{def:b3:clt:cf}{função característica}
        diferença e alcance $ C\beta n^{-1/2}$ com três
        apenas momentos.
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Two dimensions: the
multidimensional CLT, by swapping.} Agora deixe o $ X_i $ ser
i.i.d.\ centrado aleatoriamente \emph{vetores} de $\R^2$ com
matriz de covariância $\Sigma $ e $\beta' = \E\norm{X_1}^3 <
\infty $ (Norma euclidiana).
\begin{enumerate}[resume]
  \item (Vetores gaussianos, por ordem) Diagonalizar $\Sigma =
        PDP^{\mathsf T}$ (\cref{exo:b3:submanifolds:8}) e
        definir $ C = P\sqrt DP^{\mathsf T}$. Para $ Z = (Z^1,
        Z^2)$ um par de \omterm{def:b3:probability:independence}{independente} padrão \omterm{def:b3:clt:gaussianvector}{Gaussianos}
        (\cref{thm:b3:probability:existence}), mostre que $ N
        = CZ $ é um Vetor gaussiano
        (\cref{def:b3:clt:gaussianvector}) de média $0$,
        covariância $\Sigma $, com $\gamma' = \E\norm N^3 <
        \infty $; e isso $ G_n = \frac{N_1 + \dots +
        N_n}{\sqrt n}$ tem \omterm{def:b3:probability:space}{lei} $\mathcal N(0, \Sigma)$
        \emph{exatamente} para i.i.d.\ cópias $ N_i $.
  \item (Taylor em duas variáveis) Para $ f \colon \R^2 \to
        \R $ de classe $\mathcal C^3$ com $ M_3 =
        \max_{\abs\alpha = 3}\sup\abs{\partial^\alpha f} <
        \infty $, provar
        \[
        \Bigl|f(w + h) - f(w) - \langle\nabla f(w),
        h\rangle - \tfrac12\langle h, D^2f(w)\,h\rangle
        \Bigr| \leq \frac{M_3}6\,\bigl(\abs{h_1} +
        \abs{h_2}\bigr)^3 \leq \frac{\sqrt2\,M_3}3\,
        \norm h^3
        \]
        \emph{(estudar $ t \mapsto f(w + th)$ sobre
        $\intcc01$)}.
  \item (A CLT em $\R^2$) Execute o esquema de substituição em
        os híbridos de vetores $ H_i $: mostre que o primeiro e
        termos de segunda ordem cancelados (médias e covariâncias
        correspondência), telescópio e atualização como na questão 5
        (aperto de $\E\norm{T_n}^2 =
        \operatorname{tr}\Sigma $; apaziguamento agora em
        $\R^2$, \cref{thm:b3:lp:regularization}) para
        concluir: para todo limite \omterm{def:b3:topology:continuity}{contínuo} $ f \colon
        \R^2 \to \R $,
        \[
        \E\,f\Bigl(\frac{X_1 + \dots + X_n}{\sqrt n}\Bigr)
        \longrightarrow \E\,f(N), \qquad N \sim \mathcal
        N(0, \Sigma) :
        \]
        \cref{thm:b3:clt:multiclt} em dimensão $2$, com um
        taxa para suave $ f $ e sem análise de Fourier.
  \item (Cramér--Wold, e uma flutuação conjunta) Deduza
        isso $\langle t, \frac{S_n}{\sqrt n}\rangle
        \Rightarrow \mathcal N(0, t^{\mathsf T}\Sigma t)$
        para cada fixo $ t \in \R^2$. Aplicação: para
        i.i.d.\ real $(\xi_i)$, centrado, $\E\xi_1^2 = 1$,
        $\E\xi_1^6 < \infty $ (de modo que a Parte V se aplica a
        $ V_i = (\xi_i, \xi_i^2 - 1)$), mostrar
        \[
        \frac1{\sqrt n}\Bigl(\sum_{i\leq n}\xi_i,\
        \sum_{i\leq n}(\xi_i^2 - 1)\Bigr) \Longrightarrow
        \mathcal N\Bigl(0, \begin{pmatrix} 1 & \E\xi_1^3\\
        \E\xi_1^3 & \E\xi_1^4 - 1\end{pmatrix}\Bigr) :
        \]
        média empírica e segundo momento empírico
        flutuam conjuntamente de forma gaussiana --- independentemente em
        o limite se e somente se $\E\xi_1^3 = 0$
        (\cref{thm:b3:clt:gaussianvector}).
\end{enumerate}

\medskip
\noindent\textbf{Part VI --- The delta method.}
\begin{enumerate}[resume]
  \item Deixar $(\hat\theta_n)$ ser variáveis aleatórias com
        $\sqrt n(\hat\theta_n - \theta) \Rightarrow
        \mathcal N(0, \sigma^2)$ para um parâmetro real
        $\theta $, e deixe $ g $ ser diferenciável em
        $\theta $. Prove o \emph{método delta}:
        \[
        \sqrt n\bigl(g(\hat\theta_n) - g(\theta)\bigr)
        \Longrightarrow \mathcal N\bigl(0,
        g'(\theta)^2\sigma^2\bigr)
        \]
        \emph{(escrever $ g(x) - g(\theta) = (g'(\theta) +
        \eta(x))(x - \theta)$ com $\eta \to 0$ no
        $\theta $; mostrar $\hat\theta_n \to \theta $, então
        $\eta(\hat\theta_n) \to 0$, em probabilidade; terminar
        com Slutsky, \cref{exo:b3:clt:8}, e
        \cref{exo:b3:clt:4}(b))}.
  \item Aplicações. (a) Para i.i.d.\ real $(\xi_i)$ com
        significa $\mu $ e variação $\sigma^2$, e $\bar X_n =
        \frac1n\sum_{i\leq n}\xi_i $: mostrar $\sqrt n(\bar
        X_n^2 - \mu^2) \Rightarrow \mathcal N(0,
        4\mu^2\sigma^2)$ quando $\mu \neq 0$, e isso para
        $\mu = 0$ a afirmação correta mora em outro
        escala: $ n\bar X_n^2 \Rightarrow \sigma^2N^2$ com
        $ N \sim \mathcal N(0,1)$ (identificar o limite
        função de distribuição). (b) (Variação
        estabilização) Para $\hat p_n $ a frequência de sucesso
        de um $\mathcal B(1, p)$ amostra, $ p \in \intoo01$:
        mostre isso $ g(p) = \arcsin\sqrt p $ satisfaz
        \[
        \sqrt n\,\bigl(g(\hat p_n) - g(p)\bigr)
        \Longrightarrow \mathcal N\Bigl(0, \frac14\Bigr)
        \]
        \emph{qualquer que seja} $ p $ --- uma barra de erro assintótica
        \omterm{def:b3:modules:free}{livre} do parâmetro desconhecido; comparar com
        \cref{ex:b3:clt:confidence}.
\end{enumerate}

\medskip
\noindent\textbf{Part VII --- Poisson, by the same method:
Le Cam's theorem.} A substituição conhece uma segunda universalidade
classe: somas de muitos \omterm{def:b3:probability:independence}{independente} \emph{cru} eventos. Para
\omterm{def:b3:probability:space}{leis} sobre $\N $ a distância certa é \emph{variação total},
\[
d_{\mathrm{TV}}(\mu, \nu) = \sup_{A\subseteq\N}\,
\abs{\mu(A) - \nu(A)} .
\]
\begin{enumerate}[resume]
  \item Mostre isso $ d_{\mathrm{TV}}(\mu, \nu) =
        \frac12\sum_{k\geq0}\abs{\mu(\{k\}) -
        \nu(\{k\})}$, e prove o acoplamento vinculado: para
        \emph{qualquer} par $(X, Y)$ de variáveis aleatórias com
        \omterm{def:b3:probability:space}{leis} $\mu $ e $\nu $ no mesmo espaço,
        $ d_{\mathrm{TV}}(\mu, \nu) \leq \P(X \neq Y)$.
  \item Calcule exatamente, para $ p \in \intoo01$:
        \[
        d_{\mathrm{TV}}\bigl(\mathcal B(1, p), \mathcal
        P(p)\bigr) = p\bigl(1 - \eu^{-p}\bigr) \leq p^2 .
        \]
  \item (Le Cam, trocando) Vamos $ X_i \sim \mathcal B(1,
        p_i)$ e $ Y_i \sim \mathcal P(p_i)$, o $2n $
        variáveis \omterm{def:b3:probability:independence}{independente}; $ S = X_1 + \dots + X_n $ e
        lembrar $ Y_1 + \dots + Y_n \sim \mathcal P(\lambda)$
        com $\lambda = \sum_ip_i $ (\cref{exo:b3:clt:1}).
        Troque uma coordenada por vez no número inteiro
        híbridos $ H_i = Y_1 + \dots + Y_i + X_{i+1} + \dots
        + X_n $: mostrar, para cada $ A \subseteq \N $,
        \[
        \abs{\P(H_{i-1} \in A) - \P(H_i \in A)} \leq
        d_{\mathrm{TV}}\bigl(\mathcal B(1, p_i), \mathcal
        P(p_i)\bigr),
        \]
        e concluir \emph{A desigualdade de Le Cam}:
        \[
        d_{\mathrm{TV}}\bigl(\text{law of } S,\ \mathcal
        P(\lambda)\bigr) \leq \sum_{i=1}^np_i^2 .
        \]
  \item Dividendos. (a) Para $ p_i = \frac\lambda n $: o
        vinculado é $\frac{\lambda^2}n $ --- a \omterm{def:b3:probability:space}{lei} dos raros
        eventos (\cref{exo:b3:clt:5}) atualizado para um
        taxa explícita, uniforme em todos os eventos e válida
        para desigual $ p_i $ também. (b) $500$ letras são
        entregues, cada um se extraviando independentemente com
        probabilidade $\frac1{500}$: limitou o erro do
        Modelo de parâmetro de Poisson $1$, e estimar o
        probabilidade de que nenhuma carta se extravie. (c) Fechar
        o problema: compare as duas classes de universalidade
        conheci aqui --- \omterm{def:b3:clt:gaussianvector}{Gaussiano} (muitos pequenos espalhados
        contribuições; dois momentos correspondidos; Taylor) e
        Poisson (muitas contribuições raras; uma significa
        combinado; um acoplamento \omterm{def:b3:forms:closedexact}{exato} de variação total) --- e
        o único método de substituição por trás de ambos.

  \item (Erro relativo e a transformação de log) Deixe $(X_n)$
        ser i.i.d., positivo, médio $\mu > 0$, variação
        $\sigma^2$, e $\bar X_n $ a média empírica. Mostrar
        pelo método delta que
        \[
        \sqrt n\,\bigl(\ln\bar X_n - \ln\mu\bigr)
        \Longrightarrow
        \mathcal N\Bigl(0,\ \frac{\sigma^2}{\mu^2}\Bigr) :
        \]
        o parâmetro assintótico de $\ln\bar X_n $ é o
        \emph{coeficiente de variação} $\sigma/\mu $ ---
        erro relativo e sem escala. Deduza um $95\%$
        intervalo de confiança para $\mu $ do multiplicativo
        formulário $\bar X_n\cdot\eu^{\pm1.96\,\sigma/(\mu\sqrt
        n)}$ e explique quando é preferível ao
        aditivo.
  \item (O terceiro momento orienta o erro) Para $ X \sim $
        Bernoulli($ p $) centrado, computar $\E\bigl[(X -
        p)^3\bigr] = p(1-p)(1-2p)$. Usando a análise da Parte IV
        (o erro de troca é causado por terceiros momentos), explique
        por que a aproximação \omterm{def:b3:groups:normal}{normal} de $\mathcal B(n, p)$
        é assimétrico para $ p \neq \frac12$ --- ultrapassagem
        de um lado, undershooting do outro --- e por que
        $ p = \frac12$ desfruta da taxa de momento correspondente mais rápida.
        Verifique o sinal da inclinação numericamente em $\mathcal
        B(20, 0.1)$ contra $\mathcal N(2, 1.8)$: comparar
        $\P(S = 0) = 0.9^{20}$ com o \omterm{def:b3:clt:gaussianvector}{Gaussiano} massa de
        $\intoo{-\infty}{0.5}$.
\end{enumerate}
\end{problem}
