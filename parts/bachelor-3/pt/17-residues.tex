\chapter{Série de Laurent e o Teorema do Resíduo}
\label{ch:b3:residues}

O que acontece com um função holomórfica perto de um ponto onde
é \emph{não} definido? A resposta é uma completo tricotomia ---
ponto removível, pólo ou singularidade essencial --- leia um
série de potências bilaterais, a expansão de Laurent. Um coeficiente
dessa expansão, o \emph{resíduo}, controla cada contorno
integrante em torno da singularidade: o resíduo teorema converte
integrais duras definidas em álgebra finita, conta zeros de
funções (princípio do argumento, Rouché) e prova a abertura
teorema do mapeamento. Primeiro atualizamos o teorema de Cauchy de
domínios em forma de estrela para sua forma definitiva e \omterm{def:b3:modules:free}{livre} de homologia ---
O elegante argumento de Dixon --- para que todos os contornos com
o enrolamento número zero em torno do complemento fica disponível.

\section{O teorema global de Cauchy}

UM \emph{ciclo}\index{ciclo} $\Gamma $ é uma soma formal finita de
fechado caminhos $\gamma_1, \dots, \gamma_m $; integrais e
\omterm{def:b3:holomorphic:index}{índices} junto $\Gamma $ são as somas correspondentes, e
$\operatorname{im}\Gamma = \bigcup\operatorname{im}\gamma_j $.

\begin{theorem}[Cauchy, forma global]
\label{thm:b3:residues:globalcauchy}
Deixar $\Omega \subseteq \C $ estar aberto, $ f \in \mathcal H(\Omega)$,
e $\Gamma $ um ciclo em $\Omega $ tal que
\[
\operatorname{Ind}_\Gamma(w) = 0
\qquad\text{for every } w \notin \Omega .
\]
Então, para todos $ z \in \Omega\setminus\operatorname{im}\Gamma $,
\[
\frac1{2\iu\pi}\int_\Gamma\frac{f(w)}{w - z}\,\dd w =
\operatorname{Ind}_\Gamma(z)\,f(z),
\qquad\text{and}\qquad
\int_\Gamma f(w)\,\dd w = 0 .
\]
\index{Teorema de Cauchy (global)}
\end{theorem}

\begin{proof}[Prova (Dixon)]
Definir $ g \colon \Omega\times\Omega \to \C $ por
\[
g(z, w) = \begin{cases}
\dfrac{f(w) - f(z)}{w - z} & w \neq z,\\[4pt]
f'(z) & w = z .
\end{cases}
\]
\emph{$ g $ é \omterm{def:b3:topology:continuity}{contínuo}}: fora da diagonal, claro. Perto de um
ponto diagonal $(a, a)$, expandir $ f $ em uma série de potências em $ a $
(\cref{thm:b3:holomorphic:analytic}): $ f(w) - f(z) =
\sum_{n\geq1}c_n\bigl((w-a)^n - (z-a)^n\bigr)$, e dividindo
cada termo por $ w - z $ (fatoração de $ u^n - v^n $) dá, por
$ z, w \in D(a, r)$,
\[
g(z, w) =
\sum_{n\geq1}c_n\sum_{j=0}^{n-1}(w-a)^{\,j}(z-a)^{\,n-1-j},
\]
válido também na diagonal (cada soma interna torna-se
$ n(z-a)^{n-1}$, somando $ f'(z)$). Para $ r $ pequena a série
converge uniformemente em $ D(a,r)^2$
($\abs{\text{term}} \leq n\abs{c_n}r^{n-1}$, resumido dentro
o raio): a soma é \omterm{def:b3:topology:continuity}{contínuo}.

Definir $ h(z) = \frac1{2\iu\pi}\int_\Gamma g(z, w)\,\dd w $ sobre
$\Omega $: \omterm{def:b3:topology:continuity}{contínuo} (uniforme continuidade de $ g $ sobre compactos),
e \omterm{def:b3:holomorphic:holo}{holomórfico} --- por Morera
(\cref{thm:b3:holomorphic:weierstrassconv}critério): para um
triângulo $ T \subseteq \Omega $, Fubini dá $\int_{\partial
T}h = \frac1{2\iu\pi}\int_\Gamma\bigl(\int_{\partial T}g(z,
w)\dd z\bigr)\dd w = 0$, a integral interna desaparecendo porque
$ z \mapsto g(z, w)$ é \omterm{def:b3:holomorphic:holo}{holomórfico} sobre $\Omega $ (no $ z = w $ o
singularidade é removível: $ g $ é \omterm{def:b3:topology:continuity}{contínuo} lá e
\omterm{def:b3:holomorphic:holo}{holomórfico} em outro lugar --- o argumento de extensão de
\cref{thm:b3:holomorphic:formula}prova).

No conjunto aberto $\Omega' = \{z \notin \operatorname{im}\Gamma
: \operatorname{Ind}_\Gamma(z) = 0\}$, definir $ h_1(z) =
\frac1{2\iu\pi}\int_\Gamma\frac{f(w)}{w - z}\dd w $:
\omterm{def:b3:holomorphic:holo}{holomórfico} sobre $\Omega'$ (Morera ou diferenciação sob o
integrais). Para $ z \in \Omega\cap\Omega'$:
\[
h(z) = \frac1{2\iu\pi}\int_\Gamma\frac{f(w)}{w-z}\dd w
- f(z)\operatorname{Ind}_\Gamma(z) = h_1(z) .
\]
Por hipótese $\Omega\cup\Omega' = \C $ ($ w \notin \Omega
\Rightarrow \operatorname{Ind}_\Gamma(w) = 0$), então $ h $ e
$ h_1$ colar em uma função inteira $ H $. Desde o ilimitado
componente do complemento de $\operatorname{im}\Gamma $ mentiras
em $\Omega'$ e $ h_1(z) \to 0$ como $\abs z \to \infty $ (ML
vinculado), $ H $ é limitado e tende a $0$: Liouville
(\cref{cor:b3:holomorphic:liouville}) dá $ H \equiv 0$. Por isso
$ h \equiv 0$ sobre $\Omega $, que é a fórmula integral.
Aplicando-o, para fixo $ a \in
\Omega\setminus\operatorname{im}\Gamma $, para $\tilde f(w) =
(w - a)f(w)$ no $ z = a $:
\[
\frac1{2\iu\pi}\int_\Gamma f(w)\dd w =
\frac1{2\iu\pi}\int_\Gamma \frac{\tilde f(w)}{w - a}\dd w =
\operatorname{Ind}_\Gamma(a)\,\tilde f(a) = 0 .
\]
\end{proof}

\section{Série de Laurent e singularidades isoladas}

\begin{theorem}[Expansão Laurent]\label{thm:b3:residues:laurent}
Deixar $ f $ ser \omterm{def:b3:holomorphic:holo}{holomórfico} no anel $ A = \{r < \abs{z - a} <
R\}$ ($0 \leq r < R \leq \infty $). Então\index{Série Laurent}
\[
f(z) = \sum_{n\in\Z}c_n\,(z - a)^n
\qquad\text{on } A,
\qquad
c_n = \frac1{2\iu\pi}\int_{C_\rho}\frac{f(w)}{(w -
a)^{n+1}}\,\dd w
\]
para qualquer $ r < \rho < R $ (independente de $\rho $), os dois
meias séries convergindo normalmente compactar subanuais. O
expansão é única.
\end{theorem}

\begin{proof}
Consertar $ r < \rho_1 < \abs{z - a} < \rho_2 < R $ e deixe $\Gamma =
C_{\rho_2} - C_{\rho_1}$ (externo no sentido anti-horário, interno
horário): um ciclo em $ A $ com
$\operatorname{Ind}_\Gamma(w) = 0$ para todos $ w \notin A $
(aponta dentro do pequeno disco: $1 - 1 = 0$; fora do grande
um: $0 - 0$). Por \cref{thm:b3:residues:globalcauchy},
$\operatorname{Ind}_\Gamma(z) = 1 - 0 = 1$ dá
\[
f(z) = \frac1{2\iu\pi}\int_{C_{\rho_2}}\frac{f(w)}{w - z}\dd
w - \frac1{2\iu\pi}\int_{C_{\rho_1}}\frac{f(w)}{w - z}\dd w .
\]
Expanda o primeiro kernel como em
\cref{thm:b3:holomorphic:analytic} (poderes de $\frac{z -
a}{w - a}$, módulo $< 1$): a parte não negativa
$\sum_{n\geq0}c_n(z-a)^n $. No segundo, expanda o outro
maneira: $\frac{-1}{w - z} = \frac1{(z-a)(1 - \frac{w - a}{z -
a})} = \sum_{m\geq0}\frac{(w-a)^m}{(z - a)^{m+1}}$, normalmente
convergente em $ C_{\rho_1}$: a parte negativa
$\sum_{n\leq-1}c_n(z-a)^n $ com os coeficientes declarados (\omterm{def:b3:holomorphic:index}{índice}
$ n = -m-1$). Independência de $\rho $: as integrais do coeficiente
acabou $ C_{\rho}$ e $ C_{\rho'}$ diferir por
$\int_\Gamma $ ao longo de um ciclo de \omterm{def:b3:holomorphic:index}{índice} nulo do \omterm{def:b3:holomorphic:holo}{holomórfico}
$\frac{f(w)}{(w-a)^{n+1}}$ em $ A $: zero, por
\cref{thm:b3:residues:globalcauchy} de novo. Singularidade:
integrar $\sum c_n(z-a)^n $ contra $(z - a)^{-m-1}$ sobre
$ C_\rho $ termo a termo (convergência \omterm{def:b3:groups:normal}{normal}): apenas $ n = m $
sobrevive.
\end{proof}

\begin{definition}\label{def:b3:residues:singularities}
Se $ f $ é \omterm{def:b3:holomorphic:holo}{holomórfico} em um disco perfurado $ D(a, R)\setminus
\{a\}$, expandido por Laurent ($ r = 0$). Três exclusivos
casos:\index{singularidade isolada}
\begin{itemize}
  \item todos $ c_n = 0$ para $ n < 0$: \emph{removível
        singularidade} (a série não negativa se estende $ f $
        holomorficamente para $ a $);
  \item $ c_n \neq 0$ para um número finito, pelo menos um, $ n < 0$:
        um \emph{pólo} de ordem $ m = -\min\{n : c_n \neq 0\}$;
        equivalentemente $ f = g/(z-a)^m $, $ g $ \omterm{def:b3:holomorphic:holo}{holomórfico}, $ g(a)
        \neq 0$; equivalentemente $\abs{f(z)} \to \infty $ como $ z\to
        a $;
  \item infinitos negativos $ c_n \neq 0$:
        \emph{singularidade essencial}.
\end{itemize}
O \emph{tristeza}\index{tristeza} é
$\operatorname{Res}(f, a) = c_{-1}$. Uma função \omterm{def:b3:holomorphic:holo}{holomórfico} sobre
$\Omega $ menos um conjunto de pólos é
\emph{meromorfo}\index{meromorfo} sobre $\Omega $.
\end{definition}

\begin{theorem}[Riemann; Casorati-Weierstrass]
\label{thm:b3:residues:riemanncw}
Deixar $ f $ ser \omterm{def:b3:holomorphic:holo}{holomórfico} sobre $ D(a,R)\setminus\{a\}$.
\begin{enumerate}
  \item (Riemann) Se $ f $ é \emph{limitado} aproximar $ a $, o
        a singularidade é removível.\index{singularidade removível}
  \item (Casorati--Weierstrass) Se $ a $ é essencial, então
        $ f\bigl(D(a,\varepsilon)\setminus\{a\}\bigr)$ é \omterm{def:b3:topology:interior}{denso}
        em $\C $ para cada $\varepsilon $.
        \index{Teorema de Casorati-Weierstrass}
\end{enumerate}
\end{theorem}

\begin{proof}
(1) Para $ n < 0$ e $\rho \to 0$: $\abs{c_n} \leq
M\rho^{-n-1}\cdot\rho\cdot\rho^{-\,n}\dots $ por ML em
$ C_\rho $: $\abs{c_n} \leq \frac{1}{2\pi}\,2\pi\rho\cdot
M\rho^{-(n+1)} = M\rho^{-n} \to 0$ (como $-n > 0$): todos
coeficientes negativos desaparecem.
(2) Se algum valor $ b $ não foram abordados: $\abs{f - b} \geq
\delta $ aproximar $ a $, então $ g = 1/(f - b)$ é \omterm{def:b3:holomorphic:holo}{holomórfico} e
delimitado perto $ a $: removível (1), $ g $ estende com valor $ c $.
Se $ c \neq 0$, $ f = b + 1/g $ é delimitado perto $ a $: removível
--- excluído. Se $ c = 0$, $ g $ tem um zero de ordem finita $ m $
no $ a $ (\cref{thm:b3:holomorphic:identity}; $ g \not\equiv 0$),
e $ f = b + 1/g $ tem um pólo de ordem $ m $: excluído novamente.
\end{proof}

\section{O teorema do resíduo}

\begin{theorem}[Teorema do resíduo]\label{thm:b3:residues:residue}
Deixar $\Omega $ estar aberto, $ S \subseteq \Omega $ finito, $ f \in
\mathcal H(\Omega\setminus S)$, e $\Gamma $ um ciclo em
$\Omega\setminus S $ com $\operatorname{Ind}_\Gamma(w) = 0$
para cada $ w \notin \Omega $. Então\index{teorema dos resíduos}
\[
\frac{1}{2\iu\pi}\int_\Gamma f(z)\,\dd z
= \sum_{a\in S}\operatorname{Ind}_\Gamma(a)\,
\operatorname{Res}(f, a) .
\]
\end{theorem}

\begin{proof}
Para cada $ a \in S $, deixar $ P_a(z) = \sum_{n\leq-1}c_n^{(a)}(z -
a)^n $ ser a parte principal $ f $ no $ a $: uma série
convergindo para $\C\setminus\{a\}$ (seu raio em $1/(z-a)$ é
infinito: a cauda de Laurent converge para todos os pequenos $\abs{z-a}$
portanto, sendo uma série de potências em $(z-a)^{-1}$, em todos os lugares) e
\omterm{def:b3:holomorphic:holo}{holomórfico} lá. Então $ g = f - \sum_{a\in S}P_a $ tem
singularidades removíveis em cada ponto de $ S $ (é Laurent
expansão em $ a $ não tem parte negativa: a outra $ P_{a'}$ são
\omterm{def:b3:holomorphic:holo}{holomórfico} no $ a $), então $ g $ estende holomorficamente para
$\Omega $, e \cref{thm:b3:residues:globalcauchy} dá
$\int_\Gamma g = 0$. Resta integrar cada $ P_a $:
termo a termo (convergência \omterm{def:b3:groups:normal}{normal} no compactar
$\operatorname{im}\Gamma $, o que evita $ a $),
\[
\frac1{2\iu\pi}\int_\Gamma(z - a)^n\,\dd z = 0 \ (n \leq -2:
\text{primitive } \tfrac{(z-a)^{n+1}}{n+1}),
\qquad
\frac1{2\iu\pi}\int_\Gamma\frac{\dd z}{z - a} =
\operatorname{Ind}_\Gamma(a),
\]
então $\frac1{2\iu\pi}\int_\Gamma P_a =
c_{-1}^{(a)}\operatorname{Ind}_\Gamma(a)$. Soma acima $ a $.
\end{proof}

\begin{method}[Computação de resíduos]\label{met:b3:residues:compute}
Pólo \omterm{def:b3:groups:simple}{simples}: $\operatorname{Res}(f, a) = \lim_{z\to a}(z -
a)f(z)$; para $ f = g/h $ com $ g(a) \neq 0$, $ h(a) = 0$, $ h'(a)
\neq 0$: $\operatorname{Res} = g(a)/h'(a)$. Pólo de ordem $ m $:
$\operatorname{Res}(f, a) = \frac1{(m-1)!}\lim_{z\to
a}\bigl((z-a)^mf(z)\bigr)^{(m-1)}$. Singularidades essenciais:
expandir e ler $ c_{-1}$ (por exemplo\ de séries conhecidas). Sempre
verificar quais pólos o contorno realmente circunda, e com
qual \omterm{def:b3:holomorphic:index}{índice}.
\end{method}

\begin{example}[Os quatro tipos integrais clássicos]
\label{ex:b3:residues:integrals}
(um) \emph{Racional acabou $\R $}: para $\int_\R\frac{\dd x}{1 +
x^4}$, feche com um grande semicírculo $ S_R $ na parte superior
meio plano: o integrando é $ O(R^{-4})$ aí, então
$\int_{S_R} \to 0$ (ML), e o resíduo teorema com o
postes $\eu^{\iu\pi/4}, \eu^{3\iu\pi/4}$ (\omterm{def:b3:groups:simple}{simples}, resíduos
$\frac1{4z^3} = \frac{z}{4z^4} = -\frac z4$ em um poste) dá
\[
\int_\R\frac{\dd x}{1 + x^4}
= 2\iu\pi\Bigl(-\frac{\eu^{\iu\pi/4}}4 -
\frac{\eu^{3\iu\pi/4}}4\Bigr)
= \frac{\pi}{\sqrt2} .
\]
(b) \emph{Tipo Fourier}: para $ t \geq 0$,
$\int_\R\frac{\eu^{\iu tx}}{1 + x^2}\dd x =
2\iu\pi\operatorname{Res}\bigl(\tfrac{\eu^{\iu tz}}{1+z^2},
\iu\bigr) = 2\iu\pi\frac{\eu^{-t}}{2\iu} = \pi\eu^{-t}$ ---
o semicírculo superior funciona porque $\abs{\eu^{\iu tz}} =
\eu^{-t\operatorname{Im}z} \leq 1$ lá; pegando peças reais:
$\int_\R\frac{\cos(tx)}{1+x^2}\dd x = \pi\eu^{-\abs t}$,
resolver \cref{exo:b3:lebesgue:10}fórmula admitida.
(c) \emph{Trigonométrico ao longo de um período}: substituto $ z =
\eu^{\iu t}$, $\cos t = \frac{z + z^{-1}}2$, $\dd t =
\frac{\dd z}{\iu z}$: $\int_0^{2\pi}\frac{\dd t}{a + \cos t}$
($ a > 1$) torna-se um resíduo contar dentro do círculo unitário
(\cref{exo:b3:residues:2}).
(d) \emph{Série}: par $ f $ com $\pi\cot(\pi z)$, cujos pólos
são os inteiros com resíduo $1$: o problema do fim de semana resume
$\sum n^{-2}$ e $\sum n^{-4}$ Por aqui.
\end{example}

\section{O princípio do argumento e Rouchteorema de e}

\begin{theorem}[Princípio do argumento]
\label{thm:b3:residues:argument}
Deixar $ f $ ser \omterm{def:b3:residues:singularities}{meromorfo} sobre $\Omega $, com zeros $ z_j $ (pedidos
$ m_j $) e pólos $ p_k $ (pedidos $\mu_k $), e $\gamma $ um fechado
caminho em $\Omega $ evitando todos eles, com
$\operatorname{Ind}_\gamma = 0$ desligado $\Omega $. Então
\index{princípio do argumento}
\[
\frac1{2\iu\pi}\int_\gamma\frac{f'(z)}{f(z)}\,\dd z
= \sum_j m_j\operatorname{Ind}_\gamma(z_j)
- \sum_k \mu_k\operatorname{Ind}_\gamma(p_k)
\]
(um número finito de termos é diferente de zero). Por um \omterm{def:b3:groups:simple}{simples}
contorno anti-horário, a integral conta zeros menos
pólos internos, com multiplicidade --- e é igual ao enrolamento
número da imagem caminho $ f\circ\gamma $ em volta $0$.
\end{theorem}

\begin{proof}
Perto de um zero de ordem $ m $: $ f = (z-a)^mg $, $ g(a) \neq 0$, então
$\frac{f'}f = \frac m{z - a} + \frac{g'}g $ com o segundo
termo \omterm{def:b3:holomorphic:holo}{holomórfico} aproximar $ a $: um poste \omterm{def:b3:groups:simple}{simples} de resíduo $ m $. Perto
um pólo de ordem $\mu $: $ f = (z-a)^{-\mu}g $ dá resíduo
$-\mu $. Em outro lugar $\frac{f'}f $ é \omterm{def:b3:holomorphic:holo}{holomórfico}. (Os zeros e
pólos com \omterm{def:b3:holomorphic:index}{índice} diferente de zero estão em um compactar região circundada por
$\gamma $; pelo teorema da identidade eles são finitos em número
lá, $ f \not\equiv 0$.) Aplicar
\cref{thm:b3:residues:residue}. A última observação:
$\frac1{2\iu\pi}\int_\gamma\frac{f'}f =
\frac1{2\iu\pi}\int_{f\circ\gamma}\frac{\dd w}w =
\operatorname{Ind}_{f\circ\gamma}(0)$ (substituir $ w =
f(\gamma(t))$).
\end{proof}

\begin{theorem}[Rouché]\label{thm:b3:residues:rouche}
Deixar $ f, g $ ser \omterm{def:b3:holomorphic:holo}{holomórfico} sobre $\Omega $, e $\gamma $ um fechado
caminho com $\operatorname{Ind}_\gamma \in \{0,1\}$, zerar
$\Omega $ (um contorno \omterm{def:b3:groups:simple}{simples}). Se\index{Rouché's theorem}
\[
\abs{g(z)} < \abs{f(z)} \qquad \text{on }
\operatorname{im}\gamma,
\]
então $ f $ e $ f + g $ têm o mesmo número de zeros (com
multiplicidade) na região $\{\operatorname{Ind}_\gamma =
1\}$.
\end{theorem}

\begin{proof}
Para $ t \in \intcc01$, $ f_t = f + tg $ não tem zero ativado
$\operatorname{im}\gamma $ ($\abs{f_t} \geq \abs f - \abs g >
0$), então
\[
N(t) = \frac1{2\iu\pi}\int_\gamma
\frac{f_t'(z)}{f_t(z)}\,\dd z
\]
está bem definido; conta os zeros na região delimitada
(\cref{thm:b3:residues:argument}; sem postes). $ N $ é
\omterm{def:b3:topology:continuity}{contínuo} em $ t $ (o integrando é conjuntamente \omterm{def:b3:topology:continuity}{contínuo}, o
denominadores uniformemente limitados abaixo --- dominados
convergência) e valor inteiro: constante. $ N(0) = N(1)$.
\end{proof}

\begin{corollary}[Teorema do mapeamento aberto]
\label{cor:b3:residues:openmapping}
Um inconstante função holomórfica em um \omterm{def:b3:topology:connected}{conectado} conjunto aberto é
um mapa aberto. Em particular (novamente) o princípio \omterm{def:b3:rings:primemaximal}{máximo}
detém, e um \omterm{def:b3:holomorphic:holo}{holomórfico} bijeção tem \omterm{def:b3:holomorphic:holo}{holomórfico}
inverso.\index{teorema do mapeamento aberto (holomórfico)}
\end{corollary}

\begin{proof}
Deixar $ f(a) = b $; $ f - b $ tem um zero de alguma ordem finita $ m
\geq 1$ no $ a $ (teorema da identidade: $ f \not\equiv b $). Escolher
$ r $ com $ f - b $ \omterm{def:b3:modules:free}{livre} de zero ativado $\bar D(a, r)\setminus\{a\}$
(\omterm{thm:b3:holomorphic:identity}{zeros isolados}) e deixe $\delta = \min_{\abs{z - a} =
r}\abs{f(z) - b} > 0$. Para $\abs{w - b} < \delta $: no
círculo, $\abs{(b - w)} < \delta \leq \abs{f - b}$, então
Rouché ($ f - b $ versus constante $ b - w $) diz $ f - w $ tem
exatamente $ m $ zeros em $ D(a, r)$: cada um desses $ w $ é alcançado
--- $ f(D(a,r)) \supseteq D(b, \delta)$: abrir. \omterm{def:b3:rings:primemaximal}{Máximo}
princípio: um \omterm{def:b3:topology:interior}{interior} \omterm{def:b3:rings:primemaximal}{máximo} de $\abs f $ é impossível para
um inconstante $ f $, sua imagem ao redor $ f(a)$ contendo pontos
de módulo maior. Inverso: um \omterm{def:b3:holomorphic:holo}{holomórfico} bijeção $ f $ é
aberto, então $ f^{-1}$ é \omterm{def:b3:topology:continuity}{contínuo}; o zero de $ f - f(a)$ no
$ a $ é \omterm{def:b3:groups:simple}{simples} ($ m \geq 2$ daria $ m $ pré-imagens de proximidades
valores --- distintos, uma vez que $ f'$ desaparece apenas em
pontos isolados, tão próximos $ a $ o $ m $ zeros de $ f - w $ são
\omterm{def:b3:groups:simple}{simples} e distinto para pequenos genéricos $ w $: contradição
com injetividade); então $ f'(a) \neq 0$ e a diferença
quociente de $ f^{-1}$ converge: $\bigl(f^{-1}\bigr)'(b) =
1/f'(a)$.
\end{proof}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:b3:residues:1}
Classifique a singularidade em $0$ e calcular o resíduo:
\[
\frac{\sin z}{z},\qquad
\frac{\eu^z - 1}{z^2},\qquad
\frac{1}{z(z-1)^2},\qquad
\frac{\cos z}{z^3},\qquad
\eu^{1/z},\qquad
\frac1{\sin z} .
\]
Dê também o resíduo do terceiro em $ z = 1$ e do último
em $ z = \pi $.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b3:residues:2}
Para $ a > 1$ calcular, através $ z = \eu^{\iu t}$:
\[
\int_0^{2\pi}\frac{\dd t}{a + \cos t}
= \frac{2\pi}{\sqrt{a^2 - 1}} .
\]
Verifique os comportamentos limite $ a \to 1^+$ e $ a \to \infty $.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:residues:3}
Calcule com contornos de semicírculo, justificando as estimativas do arco:
\[
\int_\R\frac{x^2}{1 + x^6}\,\dd x = \frac\pi3,
\qquad
\int_\R\frac{\dd x}{(1 + x^2)^{2}} = \frac\pi2
\quad\text{(re-deriving \cref{exo:b3:fouriertransform:3})}.
\]
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:residues:4}
Prove, por $ t \geq 0$ e $ a > 0$:
\[
\int_\R\frac{\cos(tx)}{x^2 + a^2}\,\dd x =
\frac{\pi}{a}\,\eu^{-at},
\]
e deduza a transformada de Fourier de $ x \mapsto \eu^{-a\abs
x}$ por inversão --- comparando com
\cref{exo:b3:fouriertransform:1}.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:residues:5}
Expandir $ f(z) = \dfrac1{(z-1)(z-2)}$ em \omterm{thm:b3:residues:laurent}{Série Laurent} em cada
das três regiões $\abs z < 1$, $1 < \abs z < 2$, $\abs z >
2$. Por que as três expansões são diferentes? Explique por que
coeficiente de $ z^{-1}$ na segunda e terceira expansões é
\emph{não} um resíduo de $ f $ no $0$ ($ f $ não tem singularidade
lá) e calcular o valor real resíduos de $ f $, no $1$ e em
$2$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:residues:6}
(a) Mostre que $\eu^{1/z}$ tem uma singularidade essencial em $0$
e verifique Casorati--Weierstrass manualmente: resolva $\eu^{1/z} =
w $ explicitamente para qualquer $ w \neq 0$, exibindo soluções
arbitrariamente perto de $0$.
(b) Mostre que $\abs{\eu^{1/z}}$ é ilimitado em todos
perfurado \omterm{def:b3:topology:topology}{vizinhança} de $0$ ainda $\eu^{1/z}$ não tem pólo:
qual limite falha?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:residues:7}
Conte com Rouché:
(a) os zeros de $ z^7 - 4z^3 + z - 1$ em $\abs z < 1$;
(b) os zeros de $ z^4 + 5z + 1$ em $\abs z < 1$ e em $1 <
\abs z < 2$;
(c) re-provar d'Alembert--Gauss: um \omterm{def:b3:galois:extension}{grau} mônico-$ n $ polinômio
tem $ n $ zeros em algum disco grande \emph{(comparar com
$ z^n $)}.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b3:residues:8}
(Hurwitz) Deixe $ f_n \to f $ uniformemente ligado compactos, $ f_n \in
\mathcal H(\Omega)$, $\Omega $ \omterm{def:b3:topology:connected}{conectado}, $ f \not\equiv 0$.
(a) Mostre que se todos $ f_n $ são \omterm{def:b3:modules:free}{livres} de zero, então é $ f $.
\emph{(Se $ f(a) = 0$: princípio do argumento em um pequeno círculo
ao redor $ a $, e \cref{thm:b3:holomorphic:weierstrassconv} para
passar ao limite em $\int f_n'/f_n $.)}
(b) Mostre que se todos $ f_n $ são injetivos, $ f $ é injetivo ou
constante. \emph{(Aplicar (a) a $ z \mapsto f_n(z) - f_n(w)$ sobre
$\Omega\setminus\{w\}$.)}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b3:residues:9}
Para $ n \geq 2$, integrar $\frac1{1 + z^n}$ além da fronteira
do setor $\{0 \leq \arg z \leq \frac{2\pi}n,\ \abs z
\leq R\}$ e deduzir
\[
\int_0^{+\infty}\frac{\dd x}{1 + x^n} =
\frac{\pi}{n\,\sin(\pi/n)} .
\]
Verificar $ n = 2$ contra $\arctan $, e o limite $ n \to
\infty $.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:residues:10}
Deixar $ f $ seja uma função racional com $\deg(\text{denominator})
\geq \deg(\text{numerator}) + 2$. Mostre que a soma de
\emph{todos} resíduos de $ f $ é zero \emph{(integrar sobre
círculos cada vez maiores)}. Use isto para recalcular a parcial
decomposição fracionária de $\frac1{z(z-1)(z-2)}$ sem linearidade
álgebra.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b3:residues:11}
(O buraco da fechadura: integral de reflexão de Euler) Para $0 < a < 1$,
calcular
\[
I(a) = \int_0^{\infty}\frac{x^{a-1}}{1 + x}\,\dd x =
\frac{\pi}{\sin(\pi a)}
\]
integrando $ f(z) = \frac{z^{a-1}}{1+z} =
\frac{\eu^{(a-1)\log z}}{1 + z}$ (logaritmo cortado ao longo
$\R_+$, $\arg z \in \intoo0{2\pi}$) sobre o buraco da fechadura
contorno: ao longo da parte superior do corte de $\varepsilon $ para
$ R $, em volta $ C_R $, de volta ao corte, ao redor
$ C_\varepsilon $. Justifique: os dois trechos retos são diferentes
pelo fator $\eu^{2\iu\pi(a-1)}$, as contribuições do círculo
desaparecer ($ R^{a-1}\cdot R \to 0$ e
$\varepsilon^{a-1}\cdot\varepsilon \to 0$) e o único
poste $ z = -1$ tem resíduo $\eu^{\iu\pi(a - 1)}$. Deduza também
$\Gamma(a)\Gamma(1 - a) = \frac\pi{\sin\pi a}$ \emph{(escrever
$\Gamma(a)\Gamma(1-a) = B(a, 1-a)$ por
\cref{pb:b3:lebesgue:1} e substitua $ t =
\frac{x}{1+x}$)}.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:residues:12}
(Contando zeros com o princípio do argumento, numericamente)
Deixe $ P(z) = z^4 + 8z + 1$.
(a) Quantos zeros no disco unitário? \emph{(Rouché contra
$8z + 1$.)}
(b) Quantos no anel $1 < \abs z < 3$?
\emph{(Rouché contra $ z^4$ sobre $\abs z = 3$.)} Afiar:
mostre que todo zero tem módulo $< 2.1$.
(c) Quantos estão no semiplano direito? \emph{(Contar com
$\abs z = 2$ primeiro; em seguida, rastreie a imagem do imaginário
eixo: $ P(\iu t) = t^4 + 1 + 8\iu t $ tem parte real positiva
por toda parte, então não há zeros no eixo, e o argumento
variação ao longo dele é computável --- concluir com um grande
meio disco.)}
\end{exercise}

\section{Problema: \texorpdfstring{$\zeta(2k)$}{zeta(2k)} pelo
cotangente}

\begin{problem}[{Problema de fim de semana --- resumindo $\sum n^{-2k}$
com resíduos}]\label{pb:b3:residues:1}
O resíduo série de somas de teoremas: emparelhamento de uma função racional
com $\pi\cot(\pi z)$, cujos pólos ficam nos inteiros, gira
$\sum_{n}f(n)$ em um resíduo contar. Provamos o método e
calcular $\zeta(2) = \frac{\pi^2}{6}$ e $\zeta(4) =
\frac{\pi^4}{90}$ --- os valores encontrados pela série de Fourier em
Ano 2 e rastreamentos por operador em \cref{ch:b3:spectral}, agora por
integração de contorno.

\medskip
\noindent\textbf{Part I --- The cotangent kernel.}
\begin{enumerate}
  \item Mostre isso $\pi\cot(\pi z)$ é \omterm{def:b3:residues:singularities}{meromorfo} sobre $\C $ com
        postes \omterm{def:b3:groups:simple}{simples} exatamente em $ z = n \in \Z $, cada um de
        resíduo $1$ \emph{(calcular
        $\lim_{z\to n}(z-n)\pi\cot\pi z $)}.
  \item Calcule o início da expansão de Laurent em $0$:
        \[
        \pi\cot(\pi z) = \frac1z - \frac{\pi^2}{3}\,z -
        \frac{\pi^4}{45}\,z^3 + O(z^5) ,
        \]
        dividindo a série de potências de $\cos $ por aquele de
        $\sin $ (justifique a divisão: $\frac{\sin\pi z}{\pi z}$
        é \omterm{def:b3:holomorphic:holo}{holomórfico} e diferente de zero próximo $0$, então é
        recíproco é \omterm{def:b3:holomorphic:holo}{holomórfico}; identificar coeficientes
        através de encomenda $3$).
  \item Deixar $ C_N $ seja o limite do quadrado com vértices
        $(\pm1\pm\iu)(N + \frac12)$. Mostre isso
        $\abs{\cot(\pi z)} \leq 2$ sobre $ C_N $ para cada $ N \geq
        1$. \emph{(Nos lados verticais, $\cot(\pi(\pm(N +
        \frac12) + \iu y)) = \mp\tan(\iu\pi y)$, de módulo
        $\abs{\tanh(\pi y)} \leq 1$; nos lados horizontais
        $\abs y = N + \frac12$, vinculado
        $\abs{\cot(\pi(x\pm\iu y))} \leq \coth(\pi y) \leq
        \coth(\pi/2) < 1.1$.)}
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The summation theorem.}
\begin{enumerate}[resume]
  \item Deixar $ f $ seja racional, \omterm{def:b3:holomorphic:holo}{holomórfico} nos números inteiros,
        com $\deg(\text{denom}) \geq \deg(\text{num}) + 2$.
        Usando o resíduo teorema sobre $ C_N $ e o limite de
        questão 3, prove:
        \[
        \lim_{N\to\infty}\ \sum_{n = -N}^{N} f(n)
        = -\sum_{p\ \text{pole of}\ f}
        \operatorname{Res}\bigl(\pi\cot(\pi z)f(z),\,p\bigr).
        \]
  \item Onde o argumento precisa da condição de \omterm{def:b3:galois:extension}{grau}?
        Mostre por exemplo (tome $ f(z) = 1/(z + \frac12)$)
        que para um decaimento mais lento o limite simétrico
        ainda pode existir enquanto a série bilateral diverge
        --- e que a fórmula então calcula o
        \emph{valor principal}.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The values.}
\begin{enumerate}[resume]
  \item Aplique o método a $ f(z) = 1/z^2$: aqui $ f $ tem seu
        poste \emph{no} um número inteiro, então execute o argumento
        diretamente --- integrar $ g(z) = \frac{\pi\cot(\pi
        z)}{z^2}$ sobre $ C_N $, mostre a integral $\to 0$ e
        calcular $\operatorname{Res}(g, 0)$ da pergunta 2.
        Concluir:
        \[
        2\sum_{n\geq1}\frac1{n^2} = \frac{\pi^2}{3},
        \qquad \zeta(2) = \frac{\pi^2}6 .
        \]
  \item O mesmo com $ g(z) = \frac{\pi\cot(\pi z)}{z^4}$:
        calcular $\operatorname{Res}(g, 0)$ e deduzir
        $\zeta(4) = \frac{\pi^4}{90}$.
  \item Explique o padrão geral: para cada $ k \geq 1$,
        $\zeta(2k)$ é $-\frac12$ vezes o coeficiente de
        $ z^{2k-1}$ na expansão Laurent de $\pi\cot(\pi
        z)$ no $0$ --- um múltiplo racional de $\pi^{2k}$.
        Calcular $\zeta(6)$ empurrando a divisão da questão 2
        um passo adiante. O que o método diz sobre
        $\zeta(3)$ --- e por que não diz nada?
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The partial fraction expansion
of the cotangent.}
\begin{enumerate}[resume]
  \item Consertar $ w \in \C\setminus\Z $ e aplique o método de
        Parte II para $ f(z) = \dfrac{1}{(z - w)(z + w)}$ ---
        observando que $\pi\cot(\pi z)f(z)$ agora tem adicional
        postes \omterm{def:b3:groups:simple}{simples} em $\pm w $, cujo resíduos deve aderir ao
        contar. Deduza a expansão da fração parcial
        \[
        \pi\cot(\pi w) = \frac1w +
        \sum_{n\geq1}\frac{2w}{w^2 - n^2},
        \]
        a série convergindo normalmente compactar subconjuntos de
        $\C\setminus\Z $.
  \item Recupere-se desta expansão, expandindo cada termo
        em poderes de $ w $ (justificar a troca), o
        \emph{mesmo} Coeficientes de Laurent como na questão 2 ---
        o círculo se fecha: a fórmula de Euler
        $\sum\frac1{n^2} = \frac{\pi^2}6$ é o coeficiente
        de $ w $ nas duas faces da cotangente. Compare com o
        Prova da série de Fourier (Ano 2) e prova de rastreamento
        (\cref{pb:b3:spectral:1}): três teorias, uma
        número.
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Euler's product for the sine.}
A expansão da questão 9 é a derivada logarítmica de
um produto infinito; agora provamos a fatoração de Euler em 1734
honestamente.
\begin{enumerate}[resume]
  \item Para $ N \geq 1$ definir $ P_N(z) =
        z\prod_{n=1}^{N}\bigl(1 - \frac{z^2}{n^2}\bigr)$.
        Mostre isso $ P_N $ converge, uniformemente em cada disco
        $\bar D(0, R)$, para uma função inteira $ P $ cujos zeros
        são exatamente os números inteiros, todos \omterm{def:b3:groups:simple}{simples}. \emph{(Para $ n
        \geq 2R $ escreva o fator como $\exp\log(1 -
        z^2/n^2)$ com o logaritmo principal de
        \cref{exo:b3:holomorphic:3}, vinculado $\abs{\log(1+u)}
        \leq 2\abs u $ para $\abs u \leq \frac12$ através do
        série e exponenciar a soma normalmente convergente
        de logaritmos; os finitos fatores restantes
        são um polinômio. Conclua com
        \cref{thm:b3:holomorphic:weierstrassconv}.)}
  \item Mostre isso em $\C\setminus\Z $,
        \[
        \frac{P'(z)}{P(z)} = \frac1z +
        \sum_{n\geq1}\frac{2z}{z^2 - n^2} = \pi\cot(\pi z)
        \]
        \emph{(diferencie os produtos finitos, passe para
        o limite usando
        \cref{thm:b3:holomorphic:weierstrassconv} e o
        liberdade zero de $ P $ desligado $\Z $ e citar a pergunta
        9)}.
  \item Mostre isso $ Q = \sin(\pi z)/P(z)$ estende-se a um
        função inteira \omterm{def:b3:modules:free}{livre} de zero com $ Q' = 0$ e
        concluir \emph{Produto de Euler}:
        \[
        \sin(\pi z) = \pi z\prod_{n\geq1}
        \Bigl(1 - \frac{z^2}{n^2}\Bigr)
        \qquad (z \in \C) .
        \]
  \item (Wallis, 1655) Avalie em $ z = \frac12$:
        \[
        \frac\pi2 = \prod_{n\geq1}\frac{4n^2}{4n^2 - 1}
        = \lim_{N\to\infty}
        \frac{2\cdot2\cdot4\cdot4\cdots(2N)(2N)}
        {1\cdot3\cdot3\cdot5\cdots(2N-1)(2N+1)} .
        \]
  \item Para $\abs z < 1$, expanda o logaritmo do
        produto em série dupla (justificar a
        rearranjo) e recuperar $\zeta(2) =
        \frac{\pi^2}6$ combinando o coeficiente de $ z^3$
        em $\sin(\pi z) = \pi z - \frac{\pi^3}6z^3 + \cdots $
        --- a face do produto do número de Euler.
\end{enumerate}

\medskip
\noindent\textbf{Part VI --- Sister kernels.} A cotangente
tem irmãos; cada um avalia sua própria família de séries.
\begin{enumerate}[resume]
  \item Diferencie a expansão da questão 9 termo por termo
        (justificado por
        \cref{thm:b3:holomorphic:weierstrassconv}) para
        obter, normalmente em compactos de $\C\setminus\Z $,
        \[
        \frac{\pi^2}{\sin^2(\pi z)} =
        \sum_{n\in\Z}\frac1{(z - n)^2} .
        \]
  \item Avaliar em $ z = \frac12$:
        $\sum_{m\geq0}\frac1{(2m+1)^2} = \frac{\pi^2}8$;
        recuperar $\zeta(2)$ mais uma vez dividindo o
        inteiros por paridade.
  \item Verifique a identidade da duplicação $\tan\theta =
        \cot\theta - 2\cot(2\theta)$ e deduzir
        \[
        \pi\tan(\pi z) =
        \sum_{m\geq0}\frac{8z}{(2m+1)^2 - 4z^2} ,
        \]
        normalmente ligado compactos evitando $\frac12 + \Z $.
  \item Expandir ao redor $0$ ($\abs z < \frac12$; Fubini
        novamente): com $\lambda(s) =
        \sum_{m\geq0}(2m+1)^{-s}$,
        \[
        \pi\tan(\pi z) =
        \sum_{k\geq0}8\cdot4^k\,\lambda(2k+2)\,z^{2k+1} ;
        \]
        comparar com $\tan u = u + \frac{u^3}3 + O(u^5)$ para
        recuperar $\lambda(2) = \frac{\pi^2}8$ e para conseguir
        $\lambda(4) = \frac{\pi^4}{96}$ e, em seguida, verifique
        $\zeta(4) = \frac{\pi^4}{90}$ através de $\lambda(4) = (1 -
        2^{-4})\,\zeta(4)$.
  \item Verificar $\frac1{\sin\theta} = \cot\frac\theta2 -
        \cot\theta $ e deduzir
        \[
        \frac{\pi}{\sin(\pi z)} = \frac1z +
        \sum_{n\geq1}(-1)^n\,\frac{2z}{z^2 - n^2} .
        \]
        Verifique os sinais contra o resíduos de
        $\pi/\sin(\pi z)$ nos inteiros.
  \item Leia o coeficiente de $ z $: $\eta(2) =
        \sum_{n\geq1}\frac{(-1)^{n-1}}{n^2} =
        \frac{\pi^2}{12}$ e confirme a consistência
        $\eta(2) = (1 - 2^{1-2})\,\zeta(2)$.
  \item (Final) Avalie a expansão da questão 20 em $ z =
        \frac12$ e deduzir \emph{Fórmula de Leibniz}
        \[
        \frac\pi4 = 1 - \frac13 + \frac15 - \frac17 +
        \cdots
        \]
        Encerre com um pequeno parágrafo: um kernel por
        aritmética --- quais preços de kernel qual família de
        série, e por que todos eles são estruturalmente cegos
        para $\zeta(3)$.

\end{enumerate}
\medskip
\noindent\textbf{Part VII --- The full price list: Bernoulli
numbers.}
\begin{enumerate}[resume]
  \item Combine a expansão em frações parciais de $\pi
        z\cot(\pi z)$ com a função geradora do
        Números de Bernoulli ($\frac{w}{\eu^w - 1} =
        \sum_n\frac{B_n}{n!}w^n $,
        \cref{pb:b3:holomorphic:1}, Parte VI): de
        \[
        \pi z\cot(\pi z) = \iu\pi z +
        \frac{2\iu\pi z}{\eu^{2\iu\pi z} - 1}
        \]
        (prove esta identidade primeiro), deduza a forma fechada
        \[
        \zeta(2k) = (-1)^{k+1}\,
        \frac{(2\pi)^{2k}\,B_{2k}}{2\,(2k)!}
        \qquad (k \geq 1).
        \]
  \item Verifique a fórmula contra $ B_2 = \frac16$, $ B_4 =
        -\frac1{30}$, $ B_6 = \frac1{42}$: recuperar $\zeta(2) =
        \frac{\pi^2}6$, $\zeta(4) = \frac{\pi^4}{90}$ e
        calcular $\zeta(6) = \frac{\pi^6}{945}$.
  \item (Recursão de Euler) Expanda ambos os lados de
        $\bigl(z\cot z\bigr)' = \cot z - z(1 + \cot^2z)$ ---
        ou eleve ao quadrado a série cotangente diretamente --- para provar
        \[
        \Bigl(k + \frac12\Bigr)\zeta(2k) =
        \sum_{j=1}^{k-1}\zeta(2j)\,\zeta(2k - 2j)
        \qquad (k \geq 2),
        \]
        e verifique se calcula $\zeta(4)$ de $\zeta(2)$ e
        $\zeta(6)$ de $\zeta(2), \zeta(4)$ --- tudo igual
        valores zeta da semente única $\frac{\pi^2}6$,
        sem nova integração.
\end{enumerate}
\end{problem}
