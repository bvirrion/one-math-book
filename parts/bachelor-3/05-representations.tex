\chapter{Representations of Finite Groups}\label{ch:b3:representations}

To understand an abstract group, make it act on a vector space
and use linear algebra --- eigenvalues, traces, inner products ---
on the action. This program, \emph{\omterm{def:b3:representations:rep}{representation} theory}, is
astonishingly effective for finite groups over $\C$: every
\omterm{def:b3:representations:rep}{representation} splits into \omterm{def:b3:representations:rep}{irreducible} ones (Maschke), the
\omterm{def:b3:representations:rep}{irreducible} ones are pinned down by their \emph{\omterm{def:b3:representations:character}{characters}}
(traces), and the \omterm{def:b3:representations:character}{characters} satisfy orthogonality relations that
make computations mechanical. The chapter builds this calculus
and its first masterpieces --- \omterm{def:b3:representations:table}{character tables} of the small
groups --- and the weekend problem harvests a theorem far beyond
the reach of bare group theory: Burnside's $p^aq^b$ theorem.
Throughout, $G$ is a finite group and all vector spaces are
finite-dimensional over $\C$.

\section{Representations, Maschke, Schur}

\begin{definition}\label{def:b3:representations:rep}
A \emph{representation}\index{representation} of $G$ is a
morphism $\rho \colon G \to GL(V)$ for a $\C$-vector space $V$;
$\dim V$ is its \emph{degree}. A subspace $W \subseteq V$ is
\emph{invariant} if $\rho(g)W \subseteq W$ for all $g$;
restriction makes $W$ a \emph{subrepresentation}. $\rho$ is
\emph{irreducible}\index{irreducible representation} if $V \neq
0$ and its only invariant subspaces are $0$ and $V$. A
\emph{morphism} between $(\rho, V)$ and $(\sigma, W)$ is a linear
$f\colon V \to W$ with $f\rho(g) = \sigma(g)f$ for all $g$
(equivariance); bijective $f$ are \emph{isomorphisms}.
\end{definition}

\begin{example}\label{ex:b3:representations:examples}
(a) Degree $1$: morphisms $G \to \C^\times$.
(b) The \emph{regular representation}\index{regular
representation}: $V = \C^G$ with basis $(e_h)_{h \in G}$,
$\rho(g)e_h = e_{gh}$; degree $\abs G$.
(c) A permutation action of $G$ on a finite set $X$ gives the
\emph{permutation \omterm{def:b3:representations:rep}{representation}} on $\C^X$: $\rho(g)e_x =
e_{g\cdot x}$.
(d) $S_n$ acts on $\C^n$ by permuting coordinates; the hyperplane
$\{\sum x_i = 0\}$ is invariant: the \emph{standard
\omterm{def:b3:representations:rep}{representation}}, of degree $n - 1$.
\end{example}

\begin{theorem}[Maschke]\label{thm:b3:representations:maschke}
Every invariant subspace $W$ of a \omterm{def:b3:representations:rep}{representation} $(V, \rho)$
admits an invariant complement. Consequently every \omterm{def:b3:representations:rep}{representation}
is a direct sum of \omterm{def:b3:representations:rep}{irreducible} ones
(\emph{semisimplicity}).\index{Maschke's theorem}
\end{theorem}

\begin{proof}
Let $p \colon V \to V$ be \emph{any} projection with image $W$
(pick any complement). Average it over the group:
\[
\tilde p = \frac1{\abs G}\sum_{g \in G} \rho(g)\,p\,\rho(g)^{-1}.
\]
Each term maps $V$ into $W$ ($W$ invariant), and fixes $W$
pointwise: for $w \in W$, $\rho(g)^{-1}w \in W$, $p$ fixes it,
and $\rho(g)$ brings it back --- so $\tilde p$ is again a
projection onto $W$. It is equivariant: for $h \in G$,
$\rho(h)\tilde p\rho(h)^{-1}$ reindexes the same sum. Hence $\ker
\tilde p$ is an invariant complement of $W$. Iterating on the
summands (finite dimension) decomposes $V$ into \omterm{def:b3:representations:rep}{irreducibles}.
\end{proof}

\begin{theorem}[Schur's lemma]\label{thm:b3:representations:schur}
Let $f \colon V \to W$ be a morphism of \emph{\omterm{def:b3:representations:rep}{irreducible}}
\omterm{def:b3:representations:rep}{representations}. Then $f = 0$ or $f$ is an isomorphism; and if
$(V,\rho) = (W,\sigma)$, then $f = \lambda\,\mathrm{id}$ for some
$\lambda \in \C$. Hence
$\dim\operatorname{Hom}_G(V, W)$ is $1$ if $V \cong W$, $0$
otherwise.\index{Schur's lemma}
\end{theorem}

\begin{proof}
$\ker f$ and $\operatorname{im} f$ are invariant (equivariance),
so each is $0$ or everything: either $f = 0$, or $f$ is injective
with full image. If $V = W$: $f$ has an eigenvalue $\lambda$ ($\C$
\omterm{def:b3:galois:closure}{algebraically closed}); $f - \lambda\,\mathrm{id}$ is a
non-injective morphism $V \to V$, hence $0$. For the dimension
count when $V \cong W$: fixing one isomorphism $u$, any morphism
$f$ gives the endomorphism $u^{-1}f = \lambda\,\mathrm{id}$: $f =
\lambda u$.
\end{proof}

\section{Characters and orthogonality}

\begin{definition}\label{def:b3:representations:character}
The \emph{character}\index{character} of $(V, \rho)$ is
$\chi_\rho(g) = \operatorname{tr}\rho(g)$. It satisfies
$\chi_\rho(e) = \dim V$, $\chi_\rho(hgh^{-1}) = \chi_\rho(g)$
(traces are conjugation-invariant): characters are \emph{class
functions}\index{class function} --- elements of the space
$\mathcal{CF}(G)$ of functions constant on \omterm{ex:b3:groups:actions}{conjugacy classes},
equipped with the Hermitian inner product
\[
\langle \varphi, \psi\rangle = \frac1{\abs G}\sum_{g \in G}
\overline{\varphi(g)}\,\psi(g).
\]
\end{definition}

\begin{proposition}\label{prop:b3:representations:charbasics}
$\rho(g)$ is diagonalizable with eigenvalues roots of unity;
$\chi_\rho(g^{-1}) = \overline{\chi_\rho(g)}$, and
$\abs{\chi_\rho(g)} \leq \chi_\rho(e)$ with equality iff
$\rho(g)$ is a scalar. \omterm{def:b3:representations:character}{Characters} add on direct sums: $\chi_{V
\oplus W} = \chi_V + \chi_W$.
\end{proposition}

\begin{proof}
$\rho(g)^N = \mathrm{id}$ for $N = \abs G$ (Lagrange): $\rho(g)$
annihilates $X^N - 1$, split with simple roots, so it is
diagonalizable with eigenvalues $\lambda_i \in \mu_N$. Then
$\chi(g^{-1}) = \sum \lambda_i^{-1} = \sum\bar\lambda_i =
\overline{\chi(g)}$, and $\abs{\chi(g)} = \abs{\sum\lambda_i}
\leq \dim V$, with equality in the triangle inequality iff all
$\lambda_i$ are equal, i.e.\ $\rho(g) = \lambda\,\mathrm{id}$.
Additivity on direct sums: block traces.
\end{proof}

\begin{lemma}\label{lem:b3:representations:homtrace}
Let $(V, \rho)$, $(W, \sigma)$ be \omterm{def:b3:representations:rep}{representations}. The averaging
operator on $\operatorname{Hom}(W, V)$,
\[
c(A) = \frac1{\abs G}\sum_{g}\rho(g)\,A\,\sigma(g)^{-1},
\]
is a projection onto $\operatorname{Hom}_G(W, V)$, and for the
map $\Phi_g \colon A \mapsto \rho(g)A\sigma(g)^{-1}$ one has
$\operatorname{tr}\Phi_g =
\chi_\rho(g)\,\overline{\chi_\sigma(g)}$.
\end{lemma}

\begin{proof}
$c(A)$ is equivariant (reindex the sum as in Maschke), and $c$
fixes equivariant maps (each term equals $A$): $c$ is a
projection with image $\operatorname{Hom}_G(W, V)$. Trace: in
bases, $\Phi_g(A) = BAC$ with $B = \rho(g)$, $C = \sigma(g)^{-1}$;
on the basis $(E_{kl})$ of matrices, $BE_{kl}C = \sum_{m,n}
b_{mk}c_{ln}E_{mn}$, so the coefficient of $E_{kl}$ in
$\Phi_g(E_{kl})$ is $b_{kk}c_{ll}$: $\operatorname{tr}\Phi_g =
\sum_{k,l}b_{kk}c_{ll} = \operatorname{tr}(B)
\operatorname{tr}(C) = \chi_\rho(g)\chi_\sigma(g^{-1})$, and
$\chi_\sigma(g^{-1}) = \overline{\chi_\sigma(g)}$.
\end{proof}

\begin{theorem}[First orthogonality relations]
\label{thm:b3:representations:orthogonality}
Let $\rho, \sigma$ be \emph{\omterm{def:b3:representations:rep}{irreducible}}. Then\index{orthogonality
of characters}
\[
\langle\chi_\sigma, \chi_\rho\rangle =
\begin{cases}
1 & \text{if } \rho \cong \sigma,\\
0 & \text{otherwise:}
\end{cases}
\]
the \omterm{def:b3:representations:rep}{irreducible} \omterm{def:b3:representations:character}{characters} form an orthonormal family in
$\mathcal{CF}(G)$.
\end{theorem}

\begin{proof}
The trace of a projection is the dimension of its image:
\[
\dim\operatorname{Hom}_G(W, V) = \operatorname{tr} c
= \frac1{\abs G}\sum_g \operatorname{tr}\Phi_g
= \frac1{\abs G}\sum_g \chi_\rho(g)\overline{\chi_\sigma(g)}
= \langle \chi_\sigma, \chi_\rho\rangle,
\]
and Schur's lemma evaluates the left side to $\delta_{\rho \cong
\sigma}$.
\end{proof}

\begin{corollary}\label{cor:b3:representations:multiplicity}
Decompose $V \cong \bigoplus_i V_i^{\oplus m_i}$ into distinct
\omterm{def:b3:representations:rep}{irreducibles} ($V_i \not\cong V_j$). Then $m_i = \langle
\chi_{V_i}, \chi_V\rangle$: the multiplicities --- hence the
\omterm{def:b3:representations:rep}{representation} up to isomorphism --- are determined by the
\omterm{def:b3:representations:character}{character}. Moreover $\langle \chi_V, \chi_V\rangle = \sum_i
m_i^2$; in particular $V$ is \omterm{def:b3:representations:rep}{irreducible} iff $\langle\chi_V,
\chi_V\rangle = 1$.
\end{corollary}

\begin{proof}
$\chi_V = \sum m_i\chi_{V_i}$
(\cref{prop:b3:representations:charbasics}); take inner products
with each $\chi_{V_i}$ and use orthonormality. Two
\omterm{def:b3:representations:rep}{representations} with equal \omterm{def:b3:representations:character}{characters} have equal multiplicities,
hence are isomorphic.
\end{proof}

\begin{theorem}[The regular representation]
\label{thm:b3:representations:regular}
Let $\chi_1, \dots, \chi_r$ be the distinct \omterm{def:b3:representations:rep}{irreducible}
\omterm{def:b3:representations:character}{characters}, of degrees $n_i = \chi_i(e)$. The \omterm{ex:b3:representations:examples}{regular
representation} decomposes with multiplicities $m_i = n_i$;
consequently
\[
\sum_{i=1}^{r} n_i^2 = \abs G,
\qquad
\sum_i n_i\chi_i(g) = 0 \quad (g \neq e).
\]
\end{theorem}

\begin{proof}
The regular \omterm{def:b3:representations:character}{character}: $\chi_{\mathrm{reg}}(g) =
\#\{h : gh = h\}$, which is $\abs G$ for $g = e$ and $0$
otherwise. So $m_i = \langle \chi_i,\chi_{\mathrm{reg}}\rangle =
\frac1{\abs G}\overline{\chi_i(e)}\,\abs G = n_i$. Evaluating
$\chi_{\mathrm{reg}} = \sum n_i\chi_i$ at $e$ and at $g \neq e$
gives the two displayed identities.
\end{proof}

\begin{theorem}\label{thm:b3:representations:numberirr}
The \omterm{def:b3:representations:rep}{irreducible} \omterm{def:b3:representations:character}{characters} form an orthonormal \emph{basis} of
$\mathcal{CF}(G)$: the number of \omterm{def:b3:representations:rep}{irreducible representations}
equals the number of \omterm{ex:b3:groups:actions}{conjugacy classes} of $G$.
\end{theorem}

\begin{proof}
Only completeness is left: let $f \in \mathcal{CF}(G)$ be
orthogonal to every $\chi_i$; we show $f = 0$. For a
\omterm{def:b3:representations:rep}{representation} $(V,\rho)$, set $T_{f,\rho} = \frac{1}{\abs G}
\sum_g \overline{f(g)}\,\rho(g)$. It is equivariant: for $h \in
G$,
\[
\rho(h)T_{f,\rho}\rho(h)^{-1} = \frac1{\abs G}\sum_g
\overline{f(g)}\rho(hgh^{-1})
= \frac1{\abs G}\sum_{g'}\overline{f(h^{-1}g'h)}\rho(g') =
T_{f,\rho}
\]
($f$ is a \omterm{def:b3:representations:character}{class function}). If $\rho$ is \omterm{def:b3:representations:rep}{irreducible} of degree
$n$, Schur gives $T_{f,\rho} = \lambda\,\mathrm{id}$ with
\[
\lambda = \frac{\operatorname{tr}T_{f,\rho}}{n}
= \frac{1}{n\abs G}\sum_g \overline{f(g)}\,\chi_\rho(g)
= \frac1n\,\langle f, \chi_\rho\rangle = 0 .
\]
So $T_{f,\rho} = 0$ on every \omterm{def:b3:representations:rep}{irreducible}, hence (direct sums) on
\emph{every} \omterm{def:b3:representations:rep}{representation} --- in particular on the regular one.
Apply it to the basis vector $e_e$: $0 = T_{f,\mathrm{reg}}e_e =
\frac1{\abs G}\sum_g\overline{f(g)}e_g$, forcing every
$\overline{f(g)} = 0$. Thus the orthonormal family
$(\chi_i)$ spans $\mathcal{CF}(G)$, whose dimension is the
number of \omterm{ex:b3:groups:actions}{conjugacy classes}.
\end{proof}

\begin{corollary}[Column orthogonality]
\label{cor:b3:representations:column}
For $g, h \in G$:
\[
\sum_{i=1}^{r}\overline{\chi_i(g)}\,\chi_i(h) =
\begin{cases}
\abs{Z_G(g)} & \text{if } g, h \text{ are conjugate},\\
0 & \text{otherwise.}
\end{cases}
\]
\end{corollary}

\begin{proof}
Let $g_1, \dots, g_r$ represent the classes, $c_j$ the class
sizes. The $r \times r$ matrix $U_{ij} =
\sqrt{c_j/\abs G}\;\chi_i(g_j)$ has orthonormal rows
(\cref{thm:b3:representations:orthogonality} written classwise:
$\sum_j \frac{c_j}{\abs G}\chi_i(g_j)\overline{\chi_{i'}(g_j)} =
\delta_{ii'}$), i.e.\ $UU^* = I$; a square matrix with $UU^* =
I$ also has $U^*U = I$: columns are orthonormal, which unpacks to
the displayed identity ($\abs G/c_j = \abs{Z_G(g_j)}$, the
orbit--stabilizer relation for conjugation).
\end{proof}

\begin{proposition}[One-dimensional characters; lifting]
\label{prop:b3:representations:onedim}
(a) $G$ is abelian iff all its \omterm{def:b3:representations:rep}{irreducible representations} have
degree $1$; the number of degree-$1$ \omterm{def:b3:representations:character}{characters} of any $G$ is
$[G : D(G)]$ (they are the \omterm{def:b3:representations:character}{characters} of the abelianization).
(b) If $N \trianglelefteq G$, the \omterm{def:b3:representations:rep}{irreducible representations} of
$G/N$ lift (compose with $G \to G/N$) to exactly those
\omterm{def:b3:representations:rep}{irreducible representations} of $G$ whose kernel contains $N$.
\end{proposition}

\begin{proof}
(a) If $G$ is abelian, each class is a singleton: $r = \abs G$,
and $\sum n_i^2 = \abs G$ forces all $n_i = 1$; conversely if
all $n_i = 1$, the \omterm{ex:b3:representations:examples}{regular representation} is a sum of
one-dimensional ones, so $\rho_{\mathrm{reg}}(G)$ is
simultaneously diagonalizable, hence commutative, and
$\rho_{\mathrm{reg}}$ is faithful: $G$ abelian. Degree-$1$
\omterm{def:b3:representations:rep}{representations} are morphisms $G \to \C^\times$ with abelian
target: they factor through $G^{\mathrm{ab}} = G/D(G)$
(\cref{exo:b3:groups:9}), and distinct \omterm{def:b3:representations:character}{characters} of the abelian
$G^{\mathrm{ab}}$ number $\abs{G^{\mathrm{ab}}}$ (it has that
many classes, all degrees $1$). (b) Composition with the
projection preserves irreducibility (invariant subspaces
correspond), and a \omterm{def:b3:representations:rep}{representation} trivial on $N$ factors through
the quotient (\cref{thm:b3:groups:firstiso}).
\end{proof}

\section{Character tables}

\begin{definition}\label{def:b3:representations:table}
The \emph{character table}\index{character table} of $G$ is the
$r \times r$ matrix $\bigl(\chi_i(g_j)\bigr)$: rows indexed by
\omterm{def:b3:representations:rep}{irreducible} \omterm{def:b3:representations:character}{characters}, columns by \omterm{ex:b3:groups:actions}{conjugacy classes} (with their
sizes displayed). Rows are orthonormal for the weighted product,
columns orthogonal (\cref{cor:b3:representations:column}): the
table is severely overdetermined, which is what makes it
computable.
\end{definition}

\begin{example}[The table of $S_3$]\label{ex:b3:representations:s3}
Classes: $e$ (size 1), transpositions (3), $3$-cycles (2); so $r
= 3$ \omterm{def:b3:representations:rep}{irreducibles}, of degrees $n_i$ with $\sum n_i^2 = 6$: $1, 1,
2$. Degree $1$: trivial $\mathbf 1$ and signature $\varepsilon$.
The last row follows from column orthogonality (or from
$\chi_{\mathrm{std}} = \chi_{\mathrm{perm}} - \mathbf 1$):
\[
\begin{array}{c|ccc}
S_3 & e & (1\,2)\ [3] & (1\,2\,3)\ [2]\\
\hline
\mathbf 1 & 1 & 1 & 1\\
\varepsilon & 1 & -1 & 1\\
\chi_{\mathrm{std}} & 2 & 0 & -1
\end{array}
\]
Check: $\langle\chi_{\mathrm{std}},\chi_{\mathrm{std}}\rangle =
\frac{1}{6}(4 + 0 + 2) = 1$: \omterm{def:b3:representations:rep}{irreducible}.
\end{example}

\begin{example}[The table of $S_4$]\label{ex:b3:representations:s4}
Classes: $e$ [1], transpositions [6], double transpositions [3],
$3$-cycles [8], $4$-cycles [6]: five \omterm{def:b3:representations:rep}{irreducibles}, $\sum n_i^2 =
24$ with two degree-$1$'s ($\mathbf 1, \varepsilon$; $[S_4 :
D(S_4)] = [S_4 : A_4] = 2$): degrees $1, 1, 2, 3, 3$. The
degree-$2$ lifts from $S_4/V \cong S_3$
(\cref{prop:b3:representations:onedim}(b), $V$ the Klein group);
degree $3$: the standard \omterm{def:b3:representations:rep}{representation} and its twist by
$\varepsilon$:
\[
\begin{array}{c|ccccc}
S_4 & e\,[1] & (1\,2)\,[6] & (1\,2)(3\,4)\,[3] &
(1\,2\,3)\,[8] & (1\,2\,3\,4)\,[6]\\
\hline
\mathbf 1 & 1 & 1 & 1 & 1 & 1\\
\varepsilon & 1 & -1 & 1 & 1 & -1\\
\chi_2 & 2 & 0 & 2 & -1 & 0\\
\chi_{\mathrm{std}} & 3 & 1 & -1 & 0 & -1\\
\varepsilon\chi_{\mathrm{std}} & 3 & -1 & -1 & 0 & 1
\end{array}
\]
($\chi_{\mathrm{std}}(g) = \operatorname{fix}(g) - 1$;
$\chi_2$ evaluates the $S_3$-table on the image of each class
mod $V$.) All row and column orthogonality checks pass ---
running two of them is \cref{exo:b3:representations:3}'s warmup.
\end{example}

\begin{method}\label{met:b3:representations:table}
To build a \omterm{def:b3:representations:table}{character table}: (1) list \omterm{ex:b3:groups:actions}{conjugacy classes} and sizes;
(2) count degree-$1$ \omterm{def:b3:representations:character}{characters} via $G/D(G)$ and write them; (3)
find the remaining degrees from $\sum n_i^2 = \abs G$ (small
integer combinatorics); (4) obtain cheap \omterm{def:b3:representations:rep}{irreducibles}: lift from
quotients, subtract $\mathbf 1$ from permutation \omterm{def:b3:representations:character}{characters}
(check $\langle\chi,\chi\rangle = 1$), multiply known \omterm{def:b3:representations:character}{characters}
by degree-$1$ ones; (5) finish unknown rows by column
orthogonality --- each column is orthogonal to the columns
already complete, and column $e$ carries the degrees. Verify
everything with a full orthogonality sweep.
\end{method}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:b3:representations:1}
(a) Show that the \omterm{def:b3:representations:rep}{irreducible} \omterm{def:b3:representations:character}{characters} of $\Z/n\Z$ are the
$\chi_k(\bar m) = \eu^{2\iu\pi km/n}$, $k = 0, \dots, n-1$, and
write the \omterm{def:b3:representations:table}{character table} of $\Z/4\Z$.
(b) Verify both orthogonality relations on it --- and recognize
the matrix: where has this book seen it before?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b3:representations:2}
Let $G$ act on a finite set $X$ and $\chi$ be the \omterm{def:b3:representations:character}{character} of
the permutation \omterm{def:b3:representations:rep}{representation} $\C^X$.
(a) Show $\chi(g) = \abs{\operatorname{Fix}_X(g)}$ and
$\langle \mathbf 1, \chi\rangle = \#\{\text{orbits}\}$ ---
Burnside's counting lemma (\cref{exo:b3:groups:5}) is a \omterm{def:b3:representations:character}{character}
computation.
(b) Suppose the action is transitive, so $\chi = \mathbf 1 +
\psi$. Show that $\psi$ is \omterm{def:b3:representations:rep}{irreducible} iff the action is
\emph{$2$-transitive} (transitive on ordered pairs of distinct
points). \emph{(Compute $\langle\chi,\chi\rangle$ as the number
of \omterm{def:b3:groups:action}{orbits} on $X \times X$.)}
(c) Conclude that the standard \omterm{def:b3:representations:rep}{representation} of $S_n$ ($n \geq
2$) is \omterm{def:b3:representations:rep}{irreducible}.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b3:representations:3}
Rebuild the table of $S_3$ from scratch following
\cref{met:b3:representations:table}, then verify two row and two
column orthogonality relations in the $S_4$ table of
\cref{ex:b3:representations:s4}. Decompose the permutation
\omterm{def:b3:representations:character}{character} of $S_4$ acting on $\{1,2,3,4\}$ and the \omterm{def:b3:representations:character}{character}
$\chi_{\mathrm{std}}^2$ (pointwise square) into \omterm{def:b3:representations:rep}{irreducibles}.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:representations:4}
Compute the \omterm{def:b3:representations:table}{character tables} of $D_4$ and of $Q_8$. Conclude
that two non-isomorphic groups can have identical \omterm{def:b3:representations:table}{character
tables} --- what group-theoretic data does the table nevertheless
capture in this pair (orders of \omterm{ex:b3:groups:actions}{centers}, abelianizations,
number of involutions)? Which of these does it \emph{fail} to
capture?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:representations:5}
\omterm{def:b3:representations:table}{Character table} of $A_4$: classes $e$ [1], double transpositions
[3], and \emph{two} classes of $3$-cycles [4], [4].
(a) Explain the splitting of the $3$-cycles (compare \omterm{ex:b3:groups:actions}{centralizers}
in $S_4$ and $A_4$, as in \cref{exo:b3:groups:11}).
(b) Find the three degree-$1$ \omterm{def:b3:representations:character}{characters} (via $A_4/V \cong
\Z/3\Z$) and the degree-$3$ \omterm{def:b3:representations:character}{character} (restrict
$\chi_{\mathrm{std}}$ from $S_4$), and assemble the table.
(c) Read off the \omterm{def:b3:groups:normal}{normal subgroups} of $A_4$ from the table
(kernels $\{g : \chi(g) = \chi(e)\}$ and their intersections).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:representations:6}
(a) Show that $\ker\chi = \{g : \chi(g) = \chi(e)\}$ is the
kernel of the underlying \omterm{def:b3:representations:rep}{representation}
(\cref{prop:b3:representations:charbasics}, equality case).
(b) Show that every \omterm{def:b3:groups:normal}{normal subgroup} of $G$ is an intersection of
kernels of \omterm{def:b3:representations:rep}{irreducible} \omterm{def:b3:representations:character}{characters}. \emph{(Represent $G/N$
faithfully: its \omterm{ex:b3:representations:examples}{regular representation}.)}
(c) Deduce: $G$ is simple iff $\ker\chi_i = \{e\}$ for every
nontrivial \omterm{def:b3:representations:rep}{irreducible} $\chi_i$ --- simplicity is readable from
the \omterm{def:b3:representations:table}{character table}.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:representations:7}
Degrees of $\abs G = 8$: show that a nonabelian group of order
$8$ has degree pattern $(1,1,1,1,2)$, and that its degree-$2$
\omterm{def:b3:representations:rep}{representation} is faithful. More generally show that a
nonabelian group of order $p^3$ has pattern $(1^{\,p^2},
p, \dots, p)$ with $p^2$ ones and $p - 1$ \omterm{def:b3:representations:character}{characters} of degree
$p$. \emph{(Use $[G : D(G)]$ and $\sum n_i^2 = \abs G$; here
$D(G) = Z(G)$ has order $p$.)}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b3:representations:8}
For finite groups $G, H$: show that the \omterm{def:b3:representations:character}{class functions}
$\chi(g)\psi(h)$ on $G \times H$, for $\chi, \psi$ \omterm{def:b3:representations:rep}{irreducible}
\omterm{def:b3:representations:character}{characters} of $G, H$, are exactly the \omterm{def:b3:representations:rep}{irreducible} \omterm{def:b3:representations:character}{characters} of
$G \times H$. \emph{(Orthonormality is a direct computation;
completeness by counting classes.)} Deduce the \omterm{def:b3:representations:table}{character table}
of $\Z/2\Z \times \Z/2\Z$ and re-derive
\cref{prop:b3:representations:onedim}(a) for finite abelian
groups via the structure theorem.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b3:representations:9}
The \omterm{def:b3:representations:table}{character table} of $A_5$ (classes of sizes $1, 15, 20, 12,
12$ from \cref{exo:b3:groups:11}):
(a) Show the degrees are $1, 3, 3, 4, 5$ \emph{(the only
solution of $\sum n_i^2 = 60$ with $n_1 = 1$ and, using
\cref{exo:b3:representations:6}(c) with simplicity, no other
$n_i = 1$)}.
(b) Construct the degree-$4$ \omterm{def:b3:representations:character}{character} (permutation action on
$5$ points) and the degree-$5$ \omterm{def:b3:representations:character}{character} (action on the six
$5$-Sylow subgroups gives degree $6 = 1 + 5$; check
irreducibility), and complete the two degree-$3$ rows by column
orthogonality: golden-ratio entries $\frac{1\pm\sqrt5}2$ appear
on the $5$-cycle classes.
(c) Verify from the finished table that $A_5$ is simple
(\cref{exo:b3:representations:6}(c)).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:representations:10}
Let $\rho$ be an \omterm{def:b3:representations:rep}{irreducible representation} of degree $n$ and $z
\in Z(G)$. Show $\rho(z) = \lambda_z\,\mathrm{id}$ with
$\lambda \colon Z(G) \to \C^\times$ a morphism (the
\emph{central \omterm{def:b3:representations:character}{character}}), and deduce $\abs{\chi(z)} = n$ for
central $z$. Application: if $G$ has a faithful \omterm{def:b3:representations:rep}{irreducible
representation}, then $Z(G)$ is cyclic.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:representations:11}
(Isotypic projections) Let $(V, \rho)$ be a \omterm{def:b3:representations:rep}{representation} of
$G$ and $\chi_i$ an \omterm{def:b3:representations:rep}{irreducible} \omterm{def:b3:representations:character}{character} of degree $n_i$.
Define
\[
p_i = \frac{n_i}{\abs G}\sum_{g\in G}
\overline{\chi_i(g)}\,\rho(g) \;\in\; \mathcal L(V).
\]
(a) Show that $p_i$ is $G$-equivariant, and compute its
restriction to an \omterm{def:b3:representations:rep}{irreducible} subrepresentation $W \subseteq
V$ of \omterm{def:b3:representations:character}{character} $\chi_j$: it is $\delta_{ij}\,
\mathrm{id}_W$ \emph{(Schur; take traces to identify the
scalar)}.
(b) Deduce that $p_i$ is a projection onto the sum $V_i$ of
all \omterm{def:b3:representations:rep}{irreducible} subrepresentations of \omterm{def:b3:representations:character}{character} $\chi_i$ (the
\emph{isotypic component}), that $\sum_ip_i =
\mathrm{id}_V$, and that the decomposition $V =
\bigoplus_iV_i$ is canonical --- unlike the finer splitting
of each $V_i$ into \omterm{def:b3:representations:rep}{irreducibles}.
(c) For the \omterm{ex:b3:representations:examples}{regular representation} of $S_3$ and the sign
\omterm{def:b3:representations:character}{character} $\varepsilon$, write out $p_\varepsilon$ explicitly
as an element of the group algebra and check $p_\varepsilon^2
= p_\varepsilon$ by hand.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:representations:12}
(Reading a table) The \omterm{def:b3:representations:table}{character table} of a certain group $G$
of order $24$ is partially known: it has $5$ classes, of
sizes $1, 6, 8, 6, 3$, and degrees $1, 1, 2, 3, 3$.
(a) Recover the full table: the two linear \omterm{def:b3:representations:character}{characters}
(one trivial; the other takes value $-1$ exactly on the
classes of sizes $6$ and $6$), then the degree-$2$ \omterm{def:b3:representations:character}{character}
via column orthogonality with the identity column, then the
two degree-$3$ \omterm{def:b3:representations:character}{characters} likewise (one is $\chi_2\chi_4$).
(b) Identify $G$ ($\cong S_4$: compare classes with cycle
types), and extract from the table the \omterm{def:b3:groups:normal}{normal subgroups} via
\cref{exo:b3:representations:6}: kernels of $\chi_2$ (index
$2$: $A_4$) and of $\chi_3$ (the Klein $V_4$), and nothing
else but $\{e\}, G$.
(c) Explain how the table shows $G/V_4 \cong S_3$
\emph{(which \omterm{def:b3:representations:character}{characters} factor through the quotient?)}.
\end{exercise}

\section{Problem: Burnside's \texorpdfstring{$p^aq^b$}{paqb}
theorem}

\begin{problem}[{Weekend problem --- solvability of groups of
order $p^aq^b$}]\label{pb:b3:representations:1}
Burnside proved in 1904 that every group whose order has at most
two prime factors is \omterm{def:b3:groups:derived}{solvable} --- a statement about abstract
groups whose only known proofs for half a century went through
\omterm{def:b3:representations:character}{character} theory. This problem builds the proof in full,
assembling \cref{ch:b3:groups} (\omterm{def:b3:groups:derived}{solvability}),
\cref{ch:b3:modules} (\omterm{def:b3:modules:free}{finitely generated} $\Z$-modules) and this
chapter. Throughout, $\chi_1, \dots, \chi_r$ are the \omterm{def:b3:representations:rep}{irreducible}
\omterm{def:b3:representations:character}{characters} of $G$, $n_i = \chi_i(e)$.

\medskip
\noindent\textbf{Part I --- \omterm{def:b3:galois:algebraic}{Algebraic} integers.} An
\emph{\omterm{def:b3:galois:algebraic}{algebraic} integer} is a root of a \emph{monic} polynomial
of $\Z[X]$.
\begin{enumerate}
  \item Show that $\alpha$ is an \omterm{def:b3:galois:algebraic}{algebraic} integer iff the ring
        $\Z[\alpha]$ is a \omterm{def:b3:modules:free}{finitely generated} $\Z$-module.
  \item Deduce that the \omterm{def:b3:galois:algebraic}{algebraic} integers form a subring of
        $\C$. \emph{(If $\Z[\alpha], \Z[\beta]$ are \omterm{def:b3:modules:free}{finitely
        generated}, so is $\Z[\alpha, \beta]$, and submodules of
        \omterm{def:b3:modules:free}{finitely generated} $\Z$-modules are \omterm{def:b3:modules:free}{finitely generated},
        by \cref{thm:b3:modules:submodule} and a presentation
        argument --- or directly: a submodule of $\Z^n$ is free
        of rank $\leq n$.)}
  \item Show that a \emph{rational} \omterm{def:b3:galois:algebraic}{algebraic} integer is an
        integer. \emph{(Rational root theorem.)}
  \item Show that every \omterm{def:b3:representations:character}{character} value $\chi(g)$ is an
        \omterm{def:b3:galois:algebraic}{algebraic} integer.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The class-sum relations.} Fix an
\omterm{def:b3:representations:rep}{irreducible} $(V, \rho)$ of degree $n$ and \omterm{def:b3:representations:character}{character} $\chi$. For a
\omterm{ex:b3:groups:actions}{conjugacy class} $C$, let $S_C = \sum_{g \in C}\rho(g) \in
\mathcal L(V)$.
\begin{enumerate}[resume]
  \item Show that $S_C$ is equivariant, hence $S_C =
        \omega_C\,\mathrm{id}$ with
        \[
        \omega_C = \frac{\abs C\,\chi(g_C)}{n}
        \qquad (g_C \in C \text{ any representative}).
        \]
  \item Show that $S_CS_{C'} = \sum_{C''}
        a_{CC'C''}\,S_{C''}$ where $a_{CC'C''} \in \N$ counts,
        for a fixed $z \in C''$, the pairs $(x, y) \in C \times
        C'$ with $xy = z$. Deduce that the $\omega_C$ satisfy
        $\omega_C\,\omega_{C'} = \sum_{C''}
        a_{CC'C''}\,\omega_{C''}$.
  \item Conclude that each $\omega_C$ is an \omterm{def:b3:galois:algebraic}{algebraic} integer.
        \emph{(The $\Z$-module generated by $1$ and the
        $\omega_C$ is a \omterm{def:b3:modules:free}{finitely generated} ring; apply question
        1's criterion --- more precisely show $M = \Z\text{-span}
        (1, (\omega_C)_C)$ satisfies $\omega_{C_0} M \subseteq
        M$ and use a determinant/Cayley--Hamilton trick, or
        question 2's subring argument.)}
  \item Deduce \emph{\omterm{thm:b3:galois:finitefields}{Frobenius} divisibility}: $n_i$ divides
        $\abs G$ for every \omterm{def:b3:representations:rep}{irreducible} degree.
        \emph{(Compute $\frac{\abs G}{n} = \frac{\abs G}{n}
        \langle\chi,\chi\rangle = \sum_C \omega_C\,
        \overline{\chi(g_C)}$: an \omterm{def:b3:galois:algebraic}{algebraic} integer that is
        rational.)}
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Burnside's simplicity criterion.}
\begin{enumerate}[resume]
  \item Let $\chi$ be \omterm{def:b3:representations:rep}{irreducible} of degree $n$ and $C$ a class
        with $\gcd(\abs C, n) = 1$. Using B\'ezout and questions
        4--7, show that $\frac{\chi(g_C)}{n}$ is an \omterm{def:b3:galois:algebraic}{algebraic}
        integer.
  \item Suppose moreover $0 < \abs{\chi(g_C)} < n$. Show this is
        impossible: the \omterm{def:b3:galois:algebraic}{algebraic} integer $\alpha =
        \chi(g_C)/n$ has all its \emph{conjugates} --- the
        numbers $\alpha_\sigma = \frac1n\sigma(\chi(g_C))$ for
        $\sigma \in \operatorname{Gal}(\Q(\zeta_{\abs G})/\Q)$,
        each an average of $n$ roots of unity --- of modulus
        $\leq 1$,
        so the product $N = \prod_\sigma\alpha_\sigma$ is a
        rational \omterm{def:b3:galois:algebraic}{algebraic} integer with $0 < \abs N < 1$ ---
        justify each assertion, quoting
        \cref{thm:b3:galois:cyclotomicirred} for the \omterm{def:b3:galois:galois}{Galois
        group} and the fact that $\sigma$ permutes roots of
        unity. Conclude: either $\chi(g_C) = 0$ or $\rho(g_C)$
        is scalar (\cref{prop:b3:representations:charbasics}).
  \item (Burnside's criterion) Let $C \neq \{e\}$ be a
        \omterm{ex:b3:groups:actions}{conjugacy class} of \emph{prime power} size $p^k > 1$,
        and suppose $G$ is simple nonabelian. Column
        orthogonality on the column of $C$ against the column of
        $e$ gives $1 + \sum_{i \geq 2} n_i\chi_i(g_C) = 0$. Show
        that some nontrivial $\chi_i$ with $p \nmid n_i$ has
        $\chi_i(g_C) \neq 0$ \emph{(otherwise $\frac1p$ would be
        an \omterm{def:b3:galois:algebraic}{algebraic} integer)}; by question 10, $\rho_i(g_C)$
        is scalar; derive a contradiction with simplicity
        \emph{(the set of $g$ with $\rho_i(g)$ scalar is a
        \omterm{def:b3:groups:normal}{normal subgroup}; use faithfulness from
        \cref{exo:b3:representations:6})}. Conclude: \emph{no
        simple nonabelian group has a \omterm{ex:b3:groups:actions}{conjugacy class} of prime
        power size $> 1$.}
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The theorem.}
\begin{enumerate}[resume]
  \item Let $\abs G = p^aq^b$ with $a + b \geq 1$. If $G$ is
        simple, show it is abelian: pick $z \neq e$ in the
        \omterm{ex:b3:groups:actions}{center} of a Sylow $q$-subgroup
        (\cref{thm:b3:groups:pfixed}) and consider the size of
        its \omterm{ex:b3:groups:actions}{conjugacy class} $[G : Z_G(z)]$, a power of $p$
        (why?); apply question 11.
  \item Conclude by induction on $\abs G$: \emph{every group of
        order $p^aq^b$ is \omterm{def:b3:groups:derived}{solvable}} (Burnside). Why does the
        argument break for three primes --- and must it, given
        $\abs{A_5} = 2^2\cdot3\cdot5$?
\end{enumerate}

\medskip
\noindent\textbf{Part V --- The \omterm{def:b3:representations:table}{character table} of $A_5$.} The
smallest group Burnside's theorem cannot touch deserves its full
portrait; everything below uses only this chapter plus
\cref{exo:b3:groups:11}.
\begin{enumerate}[resume]
  \item Recall from \cref{exo:b3:groups:11} the five \omterm{ex:b3:groups:actions}{conjugacy
        classes} of $A_5$: $\{e\}$, the $15$ double
        transpositions, the $20$ three-cycles, and two classes
        of $12$ five-cycles each, represented by $c =
        (1\,2\,3\,4\,5)$ and $c^2$. Explain why the five-cycles
        split into two $A_5$-classes although they form a single
        $S_5$-class.
  \item Show that the \omterm{def:b3:representations:rep}{irreducible} degrees of $A_5$ are exactly
        $1, 3, 3, 4, 5$: use $\sum_in_i^2 = 60$ with $r = 5$
        classes, and the fact that $A_5$ is \omterm{prop:b3:galois:perfect}{perfect} ($D(A_5) =
        A_5$), so the trivial \omterm{def:b3:representations:character}{character} is its only linear one;
        then eliminate every other multiset \emph{(write $59$
        as a sum of four squares of integers $\geq 2$: check it
        happens in only one way with all summands plausible
        degrees)}.
  \item Let $\pi$ be the permutation \omterm{def:b3:representations:character}{character} of $A_5$ on
        $\{1, \dots, 5\}$: $\pi(g) = \#\operatorname{Fix}(g)$,
        with values $5, 1, 2, 0, 0$ on the five classes. Compute
        $\langle\pi, \mathbf 1\rangle$ and $\langle\pi,
        \pi\rangle$, and deduce that $\chi_4 = \pi - \mathbf 1$
        is \omterm{def:b3:representations:rep}{irreducible} of degree $4$, with values $4, 0, 1, -1,
        -1$.
  \item Same game on the $10$ unordered pairs $\{i, j\}$: the
        fixed-point counts are $10, 2, 1, 0, 0$. Compute
        $\langle\pi_{10}, \pi_{10}\rangle$, $\langle\pi_{10},
        \mathbf 1\rangle$ and $\langle\pi_{10}, \chi_4\rangle$,
        deduce the decomposition $\pi_{10} = \mathbf 1 + \chi_4
        + \chi_5$, and obtain the \omterm{def:b3:representations:rep}{irreducible} $\chi_5$ of degree
        $5$ with values $5, 1, -1, 0, 0$.
  \item The two remaining \omterm{def:b3:representations:rep}{irreducibles} $\chi_2, \chi_3$ have
        degree $3$. Column orthogonality (each nonidentity
        column against the identity column, and each column with
        itself) determines their values off the five-cycles:
        show $\chi_2(g) = \chi_3(g) = -1$ on double
        transpositions and $0$ on three-cycles.
  \item On the five-cycle classes, set $x = \chi_2(c)$ and $y =
        \chi_2(c^2)$; symmetry lets one take $\chi_3(c) = y$,
        $\chi_3(c^2) = x$. From the column of $c$ paired with
        the identity column and with the column of $c^2$, derive
        $x + y = 1$ and $xy = -1$ (and check the value $x^2 +
        y^2 = 3$ given by the column of $c$ with itself), hence
        \[
        \{x, y\} = \Bigl\{\frac{1 + \sqrt5}2,\ \frac{1 -
        \sqrt5}2\Bigr\} :
        \]
        the golden ratio and its conjugate. Assemble the
        complete \omterm{def:b3:representations:table}{character table} of $A_5$.
  \item Run the checks: the row norm of $\chi_2$ is $1$ (use
        $\varphi^2 + \bar\varphi^2 = 3$), $\langle\chi_2,
        \chi_3\rangle = 0$, and \omterm{thm:b3:galois:finitefields}{Frobenius} divisibility (question
        8) for all five degrees. Where in the table do you
        \emph{see} a difference with $S_5$, all of whose
        \omterm{def:b3:representations:character}{character} values are rational integers?
  \item Deduce from the table alone that $A_5$ is simple: a
        \omterm{def:b3:groups:normal}{normal subgroup} is a union of \omterm{ex:b3:groups:actions}{conjugacy classes}
        containing $e$ whose cardinality divides $60$ --- check
        that no proper sub-sum of $1 + 15 + 20 + 12 + 12$
        containing the term $1$ divides $60$. Cross-check with
        question 11's criterion: verify that no class of $A_5$
        has prime-power size $> 1$.
  \item (Icosahedral coda) $A_5$ is the rotation group of the
        icosahedron, and the degree-$3$ \omterm{def:b3:representations:rep}{representations} are the
        two geometric actions on $\R^3$. Verify the trace
        identity: a rotation by angle $\theta$ has trace $1 +
        2\cos\theta$, and $1 + 2\cos\frac{2\pi}5 =
        \frac{1+\sqrt5}2 = \varphi$. Explain without any
        computation why the other degree-$3$ \omterm{def:b3:representations:character}{character} must
        carry the conjugate value: the \omterm{def:b3:galois:galois}{Galois group} of
        $\Q(\sqrt5)/\Q$ acts on the whole \omterm{def:b3:representations:table}{character table}
        (entrywise), permuting the \omterm{def:b3:representations:rep}{irreducible} \omterm{def:b3:representations:character}{characters}.
\end{enumerate}

\medskip
\noindent\textbf{Part VI --- Complements: a central bound and
the tensor square.}
\begin{enumerate}[resume]
  \item (Sharper than a divisibility) Let $\chi$ be \omterm{def:b3:representations:rep}{irreducible}
        of degree $n$. Show that $\abs{\chi(z)} = n$ for every
        $z \in Z(G)$ \emph{(Schur's lemma: $\rho(z)$ is a
        scalar, of finite order)}, and deduce from
        $\langle\chi, \chi\rangle = 1$ the bound
        \[
        n^2 \leq [G : Z(G)] .
        \]
        Show that the nonabelian groups of order $8$ have
        \omterm{def:b3:representations:rep}{irreducible} degrees $1, 1, 1, 1, 2$ \emph{(five
        \omterm{ex:b3:groups:actions}{conjugacy classes}; write $8$ as a sum of five squares)}
        and attain equality $4 = [G : Z(G)]$; check the bound
        on $A_5$, whose \omterm{ex:b3:groups:actions}{center} is trivial.
  \item (Tensor square of $\chi_2$) For $g$ of finite order,
        $\rho(g)$ is diagonalizable with root-of-unity
        eigenvalues; deduce the \omterm{def:b3:representations:character}{character} formulas
        \[
        \chi_{\operatorname{Sym}^2 V}(g)
        = \frac{\chi(g)^2 + \chi(g^2)}2,
        \qquad
        \chi_{\Lambda^2 V}(g)
        = \frac{\chi(g)^2 - \chi(g^2)}2 .
        \]
        Apply them to $\chi_2$ of $A_5$ \emph{(note $g^2$ runs
        through the class of $c^2$ when $g$ runs through that
        of $c$, and conversely)}: show $\Lambda^2\chi_2 =
        \chi_2$ and $\operatorname{Sym}^2\chi_2 = \mathbf 1 +
        \chi_5$, hence
        \[
        \chi_2\otimes\chi_2 = \mathbf 1 + \chi_2 + \chi_5 .
        \]
        Interpret $\Lambda^2\chi_2 = \chi_2$ geometrically via
        the cross product on $\R^3$.
  \item (Final audit of the table) Verify numerically: column
        orthogonality between the two five-cycle columns
        ($\varphi\bar\varphi = -1$), the value $5 =
        \abs{Z_{A_5}(c)}$ for the column of $c$ against itself,
        and the vanishing of the regular \omterm{def:b3:representations:character}{character} $\sum_i
        n_i\chi_i(g) = 0$ on each of the four nonidentity
        columns of the table.
\end{enumerate}
\end{problem}
