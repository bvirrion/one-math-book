\chapter{The \texorpdfstring{$L^p$}{Lp} Spaces}\label{ch:b3:lp}

The Lebesgue integral was built for analysis; the $L^p$ spaces
are where that analysis lives. They are Banach spaces
(Riesz--Fischer) --- the \omterm{thm:b3:complete:completion}{completions} that
\cref{exo:b3:complete:1} showed the \omterm{def:b3:topology:continuity}{continuous} functions
lacked --- and they support a smoothing technology,
\emph{convolution against \omterm{def:b3:lp:mollifier}{mollifiers}}, that approximates any
$L^p$ function by $\mathcal C^\infty$ ones. This chapter proves
the integral versions of H\"older and Minkowski, \omterm{def:b3:complete:complete}{completeness},
the \omterm{ex:b3:lebesgue:gamma}{density} theorems, and the regularization machine, ending
with the inclusion and interpolation geography of the $L^p$
scale. Throughout, $(X, \mathcal A, \mu)$ is a \omterm{def:b3:measure:measure}{measure space}
and functions are complex-valued; on $\R^d$, the \omterm{def:b3:measure:measure}{measure} is
$\lambda_d$.

\section{Definition; H\"older and Minkowski}

\begin{definition}\label{def:b3:lp:space}
For $1 \leq p < \infty$, $\mathcal L^p(\mu)$ is the set of
\omterm{def:b3:lebesgue:measurable}{measurable} $f$ with $\norm f_p = \bigl(\int\abs
f^p\dd\mu\bigr)^{1/p} < \infty$, and $\mathcal L^\infty(\mu)$
the set of $f$ bounded outside a null set, with $\norm
f_\infty$ the \emph{essential sup} --- the least $M$ with
$\abs f \leq M$ a.e.\ (the inf is attained: intersect the null
sets for $M + \frac1n$). Since $\norm f_p = 0$ only forces $f
= 0$ \emph{a.e.} (\cref{exo:b3:lebesgue:5}), we define
\[
L^p(\mu) = \mathcal L^p(\mu)/\{f = 0 \text{ a.e.}\} :
\]
elements are classes of functions modulo null sets, and
$\norm\cdot_p$ is a genuine norm on
$L^p$.\index{Lp space@$L^p$ space}
\end{definition}

\begin{theorem}[H\"older's inequality]\label{thm:b3:lp:holder}
Let $1 \leq p, q \leq \infty$ with $\frac1p + \frac1q = 1$
(conjugate exponents). For \omterm{def:b3:lebesgue:measurable}{measurable} $f, g$:\index{H\"older's
inequality}
\[
\norm{fg}_1 \leq \norm f_p\,\norm g_q ,
\]
with equality (for $1 < p < \infty$, finite norms, $f,g\ne0$)
iff $\abs f^p$ and $\abs g^q$ are proportional a.e.
\end{theorem}

\begin{proof}
The cases $\{p, q\} = \{1, \infty\}$ are direct ($\abs{fg} \leq
\norm g_\infty\abs f$ a.e.). Let $1 < p < \infty$; normalize
$\norm f_p = \norm g_q = 1$ (homogeneity; zero or infinite
norms are trivial). Young's inequality $ab \leq \frac{a^p}p +
\frac{b^q}q$ ($a, b \geq 0$; concavity of $\ln$, as in
\cref{pb:b3:banach:1}) gives pointwise $\abs{f g} \leq
\frac{\abs f^p}p + \frac{\abs g^q}q$; integrate: $\norm{fg}_1
\leq \frac1p + \frac1q = 1$. Equality forces a.e.\ equality in
Young, i.e.\ $\abs f^p = \abs g^q$ a.e.\ (after the
normalization; undoing it, proportionality).
\end{proof}

\begin{theorem}[Minkowski's inequality]\label{thm:b3:lp:minkowski}
For $1 \leq p \leq \infty$: $\norm{f + g}_p \leq \norm f_p +
\norm g_p$.\index{Minkowski's inequality}
\end{theorem}

\begin{proof}
$p = 1, \infty$: pointwise/a.e.\ triangle inequality. For $1 <
p < \infty$, assume $\norm{f+g}_p < \infty$ (else use
$\abs{f+g}^p \leq 2^{p-1}(\abs f^p + \abs g^p)$, from convexity
of $t^p$, to see the left side is finite when the right is).
Then
\[
\norm{f{+}g}_p^p \leq \int\abs f\,\abs{f{+}g}^{p-1} +
\int\abs g\,\abs{f{+}g}^{p-1}
\leq \bigl(\norm f_p + \norm g_p\bigr)\,
\bigl\|\abs{f{+}g}^{p-1}\bigr\|_q
\]
by H\"older, and $\norm{\abs{f+g}^{p-1}}_q =
\norm{f+g}_p^{p/q}$ since $(p-1)q = p$; divide by
$\norm{f+g}_p^{p/q}$ (if nonzero; else trivial) and use $p -
\frac pq = 1$.
\end{proof}

\section{Completeness and its companions}

\begin{theorem}[Riesz--Fischer]\label{thm:b3:lp:complete}
For $1 \leq p \leq \infty$, $L^p(\mu)$ is a Banach space.
Moreover, every sequence converging in $L^p$ has a subsequence
converging \emph{\omterm{def:b3:lebesgue:l1}{almost everywhere}} (with an $L^p$ dominator
in the case $p < \infty$).\index{Riesz--Fischer theorem}
\end{theorem}

\begin{proof}
$p = \infty$: a $\norm\cdot_\infty$-Cauchy sequence is, off a
single null set (union of countably many), uniformly Cauchy:
it converges uniformly off it; done. Let $p < \infty$. By
\cref{exo:b3:complete:1}(b) it suffices to sum absolutely
convergent series: let $\sum\norm{f_k}_p = M < \infty$. Set
$G_n = \sum_{k \leq n}\abs{f_k}$ and $G = \sum_k\abs{f_k}$
(pointwise in $[0,\infty]$): by Minkowski $\norm{G_n}_p \leq
M$, and MCT ($G_n^p \nearrow G^p$) gives $\int G^p \leq M^p$:
$G < \infty$ a.e., so the series $\sum f_k(x)$ converges
absolutely for a.e.\ $x$; call the sum $S(x)$ (any value on
the null set). Then $\abs{S - \sum_{k\leq n}f_k}^p \leq (2G)^p
\in L^1$, and DCT gives $\norm{S - \sum_{k\leq n}f_k}_p \to 0$:
the series converges in $L^p$.

The subsequence statement: if $f_n \to f$ in $L^p$, pick $n_k$
with $\norm{f_{n_{k+1}} - f_{n_k}}_p \leq 2^{-k}$; the series
$\sum(f_{n_{k+1}} - f_{n_k})$ falls under the previous
argument: it converges absolutely a.e., dominated by a $G \in
L^p$, so $f_{n_k} \to f_{n_1} + \sum(\cdots) $ a.e., and this
a.e.-limit must be (a representative of) $f$ (both are $L^p$
limits). The dominator: $\abs{f_{n_k}} \leq \abs{f_{n_1}} + G$.
\end{proof}

\begin{remark}\label{rem:b3:lp:modes}
$L^p$ convergence does not imply a.e.\ convergence (the
\emph{typewriter} sequence, \cref{exo:b3:lp:3}), nor
conversely (escaping bumps): the two modes are linked only
through subsequences and domination. Keeping the counterexamples
of \cref{exo:b3:lp:3} in mind is the best vaccine.
\end{remark}

\section{Density theorems}

\begin{theorem}\label{thm:b3:lp:density}
Let $1 \leq p < \infty$.
\begin{enumerate}
  \item \omterm{def:b3:lebesgue:simple}{Simple functions} (with \omterm{def:b3:measure:measure}{finite-measure} supports) are
        dense in $L^p(\mu)$.
  \item In $L^p(\R^d)$, the \omterm{def:b3:topology:continuity}{continuous} compactly supported
        functions $\mathcal C_c(\R^d)$ are dense.
  \item Translation is \omterm{def:b3:topology:continuity}{continuous} on $L^p(\R^d)$: writing
        $\tau_hf = f(\cdot - h)$, $\norm{\tau_hf - f}_p \to 0$
        as $h \to 0$.
\end{enumerate}
None of the three holds for $p = \infty$.
\end{theorem}

\begin{proof}
(1) For $f \geq 0$: the dyadic $s_n \nearrow f$ of
\cref{thm:b3:lebesgue:approximation} satisfy $\abs{f - s_n}^p
\leq f^p \in L^1$: DCT. (Each $s_n \leq f$ lies in $L^p$, and
its level sets have finite \omterm{def:b3:measure:measure}{measure} where the value is
positive: $\mu(s_n \geq c) \leq c^{-p}\int f^p$.) Split
general $f$ into four nonnegative parts.

(2) By (1) it suffices to approximate $\mathbf 1_A$, $A$ Borel
with $\lambda_d(A) < \infty$. Regularity (proof as in
\cref{thm:b3:measure:regularity}) gives \omterm{def:b3:topology:compact}{compact} $K \subseteq A
\subseteq U$ open with $\lambda_d(U\setminus K) <
\varepsilon$; Urysohn's function
\[
\varphi(x) = \frac{d(x, \R^d\setminus U)}{d(x, \R^d\setminus U)
+ d(x, K)}
\]
is \omterm{def:b3:topology:continuity}{continuous}, $1$ on $K$, $0$ outside $U$, and can be taken
compactly supported (shrink $U$ to a bounded open first). Then
$\norm{\mathbf 1_A - \varphi}_p^p \leq \lambda_d(U\setminus K)
< \varepsilon$.

(3) For $g \in \mathcal C_c$: uniform \omterm{def:b3:topology:continuity}{continuity} gives
$\norm{\tau_hg - g}_\infty \to 0$, with supports in a fixed
\omterm{def:b3:topology:compact}{compact} for $\abs h \leq 1$: $\norm{\tau_hg - g}_p \to 0$. For
general $f$: pick $g \in \mathcal C_c$ with $\norm{f - g}_p <
\varepsilon$; then $\norm{\tau_hf - f}_p \leq 2\norm{f - g}_p +
\norm{\tau_hg - g}_p$ (translation invariance of the norm).

For $p = \infty$: uniform approximation of $\mathbf
1_{\intoo0\infty}$ by \omterm{def:b3:topology:continuity}{continuous} functions is impossible
(jump), and $\norm{\tau_h\mathbf 1_{\intoo0\infty} - \mathbf
1_{\intoo0\infty}}_\infty = 1$ for $h \neq 0$.
\end{proof}

\section{Convolution and regularization}

\begin{theorem}[Young's inequality]\label{thm:b3:lp:young}
Let $1 \leq p \leq \infty$, $f \in L^1(\R^d)$, $g \in
L^p(\R^d)$. Then $f * g$ is defined a.e., belongs to $L^p$,
and\index{Young's inequality}
\[
\norm{f * g}_p \leq \norm f_1\,\norm g_p .
\]
\end{theorem}

\begin{proof}
$p = \infty$: direct bound. $p = 1$:
\cref{thm:b3:product:convolution}. Let $1 < p < \infty$, $q$
conjugate. Split $\abs{f(y)} = \abs{f(y)}^{1/q}\cdot
\abs{f(y)}^{1/p}$ and apply H\"older:
\[
\int\abs{f(y)}\,\abs{g(x{-}y)}\,\dd y
\leq \Bigl(\int\abs f\Bigr)^{1/q}
\Bigl(\int\abs{f(y)}\,\abs{g(x - y)}^p\,\dd y\Bigr)^{1/p} .
\]
Raise to the $p$-th power and integrate in $x$; Tonelli on the
second factor gives $\norm f_1^{p/q}\cdot\norm f_1\norm g_p^p$,
i.e.\ $\norm{f*g}_p^p \leq \norm f_1^{1 + p/q}\norm g_p^p =
(\norm f_1\norm g_p)^p$ --- and the finiteness of the Tonelli
integral justifies a.e.\ absolute convergence as in
\cref{thm:b3:product:convolution}.
\end{proof}

\begin{definition}[Mollifiers]\label{def:b3:lp:mollifier}
The function\index{mollifier}\index{bump function}
\[
\rho(x) = \begin{cases}
c\,\exp\Bigl(-\dfrac{1}{1 - \norm x^2}\Bigr) & \norm x < 1,\\
0 & \norm x \geq 1,
\end{cases}
\]
with $c$ normalizing $\int\rho = 1$, is $\mathcal C^\infty$ on
$\R^d$: the point is that $t \mapsto \eu^{-1/t}\mathbf
1_{t>0}$ is $\mathcal C^\infty$ on $\R$, all its derivatives at
$0^+$ being $0$ (each derivative is $P(1/t)\eu^{-1/t}$ for a
polynomial $P$, which tends to $0$; induction). For
$\varepsilon > 0$ set $\rho_\varepsilon(x) =
\varepsilon^{-d}\rho(x/\varepsilon)$: supported in $\bar B(0,
\varepsilon)$, still of integral $1$.
\end{definition}

\begin{theorem}[Regularization]\label{thm:b3:lp:regularization}
Let $1 \leq p < \infty$ and $f \in L^p(\R^d)$. Then:
\begin{enumerate}
  \item $f * \rho_\varepsilon \in \mathcal C^\infty(\R^d)$,
        with $\partial^\alpha(f * \rho_\varepsilon) = f *
        \partial^\alpha\rho_\varepsilon$;
  \item $\norm{f * \rho_\varepsilon - f}_p \to 0$ as
        $\varepsilon \to 0$;
  \item consequently $\mathcal C^\infty_c(\R^d)$ is dense in
        $L^p(\R^d)$.
\end{enumerate}
\end{theorem}

\begin{proof}
(1) Differentiation under the integral
(\cref{thm:b3:lebesgue:paramdiff}) in $x$: for $x$ in a ball
$B$, $\abs{\partial_{x_i}\rho_\varepsilon(x - y)} \leq
C_\varepsilon\,\mathbf 1_{K}(y)$ with $K$ \omterm{def:b3:topology:compact}{compact} ($y$ within
$\varepsilon$ of $B$), and $\abs f\,\mathbf 1_K \in L^1$
(H\"older against $\mathbf 1_K$): the theorem applies;
iterate for higher derivatives.

(2) Since $\int\rho_\varepsilon = 1$:
\[
(f * \rho_\varepsilon)(x) - f(x)
= \int \bigl(f(x - y) - f(x)\bigr)\rho_\varepsilon(y)\,\dd y ,
\]
and Minkowski's \emph{integral} inequality --- or directly:
H\"older/Jensen with the probability \omterm{def:b3:measure:measure}{measure}
$\rho_\varepsilon\dd y$ and Tonelli ---
\[
\norm{f*\rho_\varepsilon - f}_p^p
\leq \int\Bigl(\int\abs{f(x-y) -
f(x)}^p\dd x\Bigr)\rho_\varepsilon(y)\,\dd y
= \int \norm{\tau_yf - f}_p^p\;\rho_\varepsilon(y)\,\dd y
\]
(the middle step: apply Jensen's inequality,
\cref{exo:b3:lp:10}, to the inner $y$-integral, then Tonelli).
The integrand is supported in $\norm y \leq \varepsilon$ and
tends to $0$ there uniformly as $\varepsilon \to 0$
(\cref{thm:b3:lp:density}(3)): the whole expression tends to
$0$.

(3) Approximate $f$ by $g \in \mathcal C_c$
(\cref{thm:b3:lp:density}(2)), then $g$ by $g *
\rho_\varepsilon \in \mathcal C_c^\infty$ (\omterm{def:b3:topology:compact}{compact} support:
sum of supports).
\end{proof}

\begin{example}[Mollifying $\abs x$, with rates]
\label{ex:b3:lp:mollifyexample}
Take $f(x) = \abs x$ on $\R$ (locally $L^1$; the theorem
applies on every bounded window) and a symmetric \omterm{def:b3:lp:mollifier}{mollifier}
$\rho_\varepsilon$. Then
\[
f_\varepsilon(x) = (f * \rho_\varepsilon)(x)
= \int\abs{x - y}\,\rho_\varepsilon(y)\,\dd y
\]
is $\mathcal C^\infty$; away from the kink, nothing happens:
for $\abs x \geq \varepsilon$, $\abs{x - y}$ is linear in $x$
on the support of $\rho_\varepsilon$, so $f_\varepsilon(x) =
\abs x$ \emph{exactly} (symmetry kills the correction).
Near $0$, smoothing costs precisely
\[
0 \leq f_\varepsilon(0) = \int\abs
y\,\rho_\varepsilon(y)\,\dd y \leq \varepsilon,
\qquad
\norm{f_\varepsilon - f}_\infty \leq \varepsilon :
\]
the approximation error is confined to the
$\varepsilon$-neighborhood of the singularity and is of its
size. Meanwhile $f_\varepsilon'' \geq 0$ everywhere
($f$ is convex, and convolution against $\rho_\varepsilon
\geq 0$ preserves convexity), with $\int f_\varepsilon'' =
f_\varepsilon'(\infty) - f_\varepsilon'(-\infty) = 2$: the
second derivative is a bump of mass $2$ squeezed into width
$O(\varepsilon)$, so $\norm{f_\varepsilon''}_\infty \gtrsim
\varepsilon^{-1}$. Smoothing is a trade: uniform error
$O(\varepsilon)$ against derivative blow-up
$O(\varepsilon^{-1})$ --- the exact exchange rate that
quantitative analysis (interpolation inequalities,
\cref{pb:b3:lp:1}'s circle of ideas) formalizes.
\end{example}

\begin{corollary}[Fundamental lemma of the calculus of
variations]\label{cor:b3:lp:fundlemma}
Let $f \in L^1_{\mathrm{loc}}(\R^d)$ (\omterm{def:b3:lebesgue:l1}{integrable} on \omterm{def:b3:topology:compact}{compacts})
with $\int f\varphi = 0$ for every $\varphi \in \mathcal
C^\infty_c(\R^d)$. Then $f = 0$ a.e.\index{fundamental lemma of
the calculus of variations}
\end{corollary}

\begin{proof}
Fix a ball $B = B(0, R)$ and let $g = f\mathbf 1_{B(0, R+1)}
\in L^1$. For $x \in B$ and $\varepsilon < 1$: $(g *
\rho_\varepsilon)(x) = \int f(y)\rho_\varepsilon(x - y)\dd y =
0$, the test function being $y \mapsto \rho_\varepsilon(x-y)
\in \mathcal C_c^\infty$. But $g * \rho_\varepsilon \to g$ in
$L^1$ (\cref{thm:b3:lp:regularization}): $g = 0$ a.e.\ on $B$;
exhaust $\R^d$.
\end{proof}

\section{The \texorpdfstring{$L^p$}{Lp} geography}

\begin{proposition}\label{prop:b3:lp:inclusions}
(a) If $\mu(X) < \infty$ and $1 \leq p \leq q \leq \infty$,
then $L^q \subseteq L^p$ with $\norm f_p \leq
\mu(X)^{\frac1p - \frac1q}\,\norm f_q$.
(b) On $\R^d$ (infinite \omterm{def:b3:measure:measure}{measure}) there are no inclusions:
for $p \neq q$ there are functions in $L^p\setminus L^q$.
(c) (Interpolation) If $p < r < q$ and $\alpha \in \intoo01$
is defined by $\frac1r = \frac\alpha p + \frac{1 - \alpha}q$,
then
\[
\norm f_r \leq \norm f_p^{\alpha}\,\norm f_q^{1 - \alpha} ;
\]
in particular $L^p \cap L^q \subseteq L^r$.
\end{proposition}

\begin{proof}
(a) H\"older with exponents $\frac qp$ and its conjugate:
$\int\abs f^p\cdot 1 \leq \norm{\abs f^p}_{q/p}\,\norm
1_{(q/p)'} = \norm f_q^p\,\mu(X)^{1 - p/q}$ ($q = \infty$
directly). (b) Near $0$ and near $\infty$, powers $x^{-\alpha}$
calibrate: \cref{exo:b3:lp:2}. (c) Write $\abs f^r = \abs
f^{r\alpha}\abs f^{r(1-\alpha)}$ and apply H\"older with the
conjugate pair $\frac p{r\alpha}$, $\frac q{r(1-\alpha)}$
(conjugates precisely by the definition of $\alpha$):
$\int\abs f^r \leq \norm f_p^{r\alpha}\norm f_q^{r(1 -
\alpha)}$.
\end{proof}

\begin{method}\label{met:b3:lp:toolkit}
The $L^p$ toolkit, as used everywhere below: to prove an
identity or inequality for all $f \in L^p$ --- prove it on a
dense class ($\mathcal C_c^\infty$ via
\cref{thm:b3:lp:regularization}) and extend by \omterm{def:b3:topology:continuity}{continuity}
(\cref{thm:b3:complete:extension}, both sides being
$L^p$-continuous); to prove $f = 0$, test against $\mathcal
C_c^\infty$ (\cref{cor:b3:lp:fundlemma}); to gain smoothness,
convolve; to trade exponents, H\"older and interpolation. The
Fourier theory of \cref{ch:b3:fouriertransform} is one long
application of this method.
\end{method}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:b3:lp:1}
(a) State and prove the Cauchy--Schwarz inequality in
$L^2(\mu)$ as the case $p = q = 2$ of H\"older.
(b) On a \emph{probability} space, show $p \mapsto \norm f_p$
is nondecreasing.
(c) When is H\"older an equality for $p = 1$, $q = \infty$?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b3:lp:2}
For which $p \in \intco1\infty$ do the following belong to
$L^p$?
\[
x^{-1/2}\mathbf 1_{\intoo01},\qquad
x^{-1/2}\mathbf 1_{\intoo1\infty},\qquad
\frac{1}{x^{1/2}(1 + \abs{\ln x})}\ \text{on } \intoo01,
\qquad
\frac1{1 + \abs x}\ \text{on } \R .
\]
Conclude: on $\intoo01$ small $p$ is easier, on
$\intoo1\infty$ large $p$ is easier, and no $L^p$ contains
another on $\R$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:lp:3}
(The typewriter) Enumerate the dyadic intervals $I_1 =
\intcc01$, $I_2 = \intcc0{\frac12}$, $I_3 = \intcc{\frac12}1$,
$I_4 = \intcc0{\frac14}$, \dots\ and let $f_n = \mathbf
1_{I_n}$.
(a) Show $f_n \to 0$ in every $L^p(\intcc01)$, $p < \infty$,
but $(f_n(x))$ diverges for \emph{every} $x \in \intcc01$.
(b) Exhibit the a.e.-convergent subsequence promised by
\cref{thm:b3:lp:complete}.
(c) Conversely give a sequence converging a.e.\ but not in
$L^1$, and one converging in $L^1$ but in no $L^p$, $p > 1$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:lp:4}
Let $\mu(X) < \infty$ and $f \in L^\infty(\mu)$, $f \neq 0$.
Show that $\norm f_p \to \norm f_\infty$ as $p \to \infty$.
\emph{(Upper bound by (a) of \cref{prop:b3:lp:inclusions};
lower bound by integrating over $\{\abs f > \norm f_\infty -
\varepsilon\}$, of positive \omterm{def:b3:measure:measure}{measure}.)}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:lp:5}
(a) Where exactly does the proof of
\cref{thm:b3:lp:density}(3) use $p < \infty$?
(b) Show that $f \in L^\infty(\R)$ satisfies $\norm{\tau_hf -
f}_\infty \to 0$ iff $f$ has a uniformly \omterm{def:b3:topology:continuity}{continuous}
representative.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:lp:6}
Let $p, q$ be conjugate, $f \in L^p(\R^d)$, $g \in
L^q(\R^d)$. Show that $f * g$ is defined \emph{everywhere},
bounded, with $\norm{f*g}_\infty \leq \norm f_p\norm g_q$, and
\emph{uniformly \omterm{def:b3:topology:continuity}{continuous}}. \emph{(\omterm{def:b3:topology:continuity}{Continuity} of translation
in $L^p$; treat $p \in \{1, \infty\}$ separately --- for $p =
\infty$ use translation \omterm{def:b3:topology:continuity}{continuity} on the $L^1$ factor.)}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:lp:7}
Let $f \in L^1_{\mathrm{loc}}(\intoo ab)$ with
$\int f\varphi' = 0$ for every $\varphi \in \mathcal
C^\infty_c(\intoo ab)$. Show that $f$ is a.e.\ equal to a
constant. \emph{(Fix $\chi \in \mathcal C_c^\infty$ with
$\int\chi = 1$; any $\psi \in \mathcal C_c^\infty$ with
$\int\psi = 0$ is a $\varphi'$; write a general test function
as $\psi + (\int\psi)\chi$ and apply
\cref{cor:b3:lp:fundlemma} to $f - c$ with $c = \int
f\chi$.)}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b3:lp:8}
(Smooth Urysohn) Let $K \subseteq U \subseteq \R^d$, $K$
\omterm{def:b3:topology:compact}{compact}, $U$ open. Construct $\varphi \in \mathcal
C^\infty_c(\R^d)$ with $0 \leq \varphi \leq 1$, $\varphi = 1$
on $K$, $\operatorname{supp}\varphi \subseteq U$.
\emph{(Mollify the indicator of the $\delta$-neighborhood
$K_\delta$ of $K$ with $\rho_{\delta/2}$, for $\delta$ small.)}
Deduce a $\mathcal C^\infty$ partition-of-unity statement for
a \omterm{def:b3:topology:compact}{compact} covered by finitely many \omterm{def:b3:topology:topology}{open sets}.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:lp:9}
Using interpolation (\cref{prop:b3:lp:inclusions}(c)):
(a) show that $L^1(\R)\cap L^\infty(\R) \subseteq L^p(\R)$ for
all $p$, with $\norm f_p \leq \norm f_1^{1/p}\norm
f_\infty^{1 - 1/p}$;
(b) show that $f \mapsto \norm f_p$ is, for fixed $f$,
log-convex in $\frac1p$, and give an example where $f \in L^p$
exactly for $p$ in a given interval $(p_0, p_1)$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:lp:10}
(Jensen) Let $\mu$ be a \emph{probability} \omterm{def:b3:measure:measure}{measure}, $f \in
L^1(\mu)$ real, and $\Phi \colon \R \to \R$ convex. Show
\[
\Phi\Bigl(\int f\,\dd\mu\Bigr) \leq \int \Phi\circ
f\,\dd\mu
\]
\emph{(support line of $\Phi$ at the point $m = \int f$)}.
Deduce the arithmetic--geometric inequality and the
monotonicity of $p \mapsto \norm f_p$ of
\cref{exo:b3:lp:1}(b).\index{Jensen's inequality}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b3:lp:11}
(Young's convolution inequality) Let $1 \leq p, q, r \leq
\infty$ with $\frac1p + \frac1q = 1 + \frac1r$, and $f \in
L^p(\R^d)$, $g \in L^q(\R^d)$.
(a) Prove $\norm{f * g}_r \leq \norm f_p\,\norm g_q$.
\emph{(Write, for conjugate exponents worked out from $p, q,
r$,
\[
\abs{f(y)g(x-y)} = \bigl(\abs f^p\abs g^q\bigr)^{1/r}
\cdot\abs f^{\,p(1/p - 1/r)}\cdot\abs g^{\,q(1/q - 1/r)},
\]
and apply the three-factor H\"older inequality with
exponents $r$, $\frac{pr}{r - p}$, $\frac{qr}{r - q}$; then
integrate in $x$ by Tonelli.)}
(b) Check the three special cases already known: $r =
\infty$ (H\"older, \cref{exo:b3:lp:6}); $q = 1$
($L^p$-stability of convolution by an \omterm{def:b3:lebesgue:l1}{integrable} kernel);
$p = q = 1$ ($L^1$ is a convolution algebra,
\cref{thm:b3:product:convolution}).
(c) Why is there no inequality with $\frac1p + \frac1q < 1 +
\frac1r$? \emph{(Test on dilations $f_\lambda(x) =
f(\lambda x)$ and compare the scalings of both sides.)}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:lp:12}
(Equality cases) (a) In H\"older's inequality
$\int\abs{fg} \leq \norm f_p\norm g_q$ ($1 < p < \infty$),
show that equality holds iff $\abs f^p$ and $\abs g^q$ are
proportional a.e. \emph{(Track the equality case of Young's
inequality $ab \leq \frac{a^p}p + \frac{b^q}q$, which is $a^p
= b^q$.)}
(b) In Minkowski's inequality $\norm{f + g}_p \leq \norm f_p
+ \norm g_p$ ($1 < p < \infty$), show that equality with $f,
g \neq 0$ forces $g = cf$ a.e.\ with $c > 0$.
(c) Contrast with $p = 1$ and $p = \infty$: describe the
(much larger) equality cases there, on examples.
\end{exercise}

\section{Problem: Hardy's inequality}

\begin{problem}[{Weekend problem --- Hardy's inequality and
its sharp constant}]\label{pb:b3:lp:1}
For $f \in L^p(\intoo0{+\infty})$, $1 < p < \infty$, define
the \emph{Hardy operator}
\[
(Hf)(x) = \frac1x\int_0^x f(t)\,\dd t .
\]
Hardy's inequality (1920) asserts\index{Hardy's inequality}
\[
\norm{Hf}_p \;\leq\; \frac{p}{p-1}\,\norm f_p ,
\]
and the constant $\frac p{p-1}$ is optimal and not attained.
This problem proves everything, then extends to series.

\medskip
\noindent\textbf{Part I --- The inequality.} Assume first $f
\geq 0$ \omterm{def:b3:topology:continuity}{continuous} with \omterm{def:b3:topology:compact}{compact} support in
$\intoo0{+\infty}$, and let $F(x) = \int_0^xf$.
\begin{enumerate}
  \item Show that $Hf \in L^p$: near $0$, $F$ vanishes on a
        \omterm{def:b3:topology:topology}{neighborhood} of $0$; near $\infty$, $F$ is bounded, so
        $(Hf)(x) = O(1/x)$, and $x \mapsto \frac1x$ belongs to
        $L^p(\intoo1{+\infty})$ for $p > 1$.
  \item Integrate by parts to show
        \[
        \int_0^\infty \Bigl(\frac Fx\Bigr)^p\dd x
        = \frac{p}{p-1}\int_0^\infty\Bigl(\frac
        Fx\Bigr)^{p-1}f(x)\,\dd x .
        \]
        \emph{(Differentiate $x^{1-p}F^p$; boundary terms
        vanish --- justify both ends.)}
  \item Apply H\"older to the right side and deduce
        $\norm{Hf}_p \leq \frac p{p-1}\norm f_p$ for such $f$.
  \item Extend to all of $L^p$: for $f \geq 0$, construct
        $f_n$ \omterm{def:b3:topology:continuity}{continuous} with \omterm{def:b3:topology:compact}{compact} support in
        $\intoo0{+\infty}$, $0 \leq f_n \nearrow f$ a.e.\
        \emph{(truncate, then approximate monotonically ---
        justify the construction)}; then $Hf_n \nearrow Hf$
        pointwise (MCT inside the average) and MCT passes the
        inequality to the limit. For signed or complex $f$,
        conclude with $\abs{Hf} \leq H\abs f$.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Optimality.}
\begin{enumerate}[resume]
  \item For $A > 1$ let $f_A(t) = t^{-1/p}\,\mathbf
        1_{\intcc1A}(t)$. Compute $\norm{f_A}_p^p = \ln A$
        and, for $1 \leq x \leq A$,
        \[
        (Hf_A)(x) = \frac p{p-1}\;x^{-1/p}\,
        \bigl(1 - x^{-(1 - 1/p)}\bigr).
        \]
  \item Deduce $\liminf_{A\to\infty} \norm{Hf_A}_p/
        \norm{f_A}_p \geq \frac{p}{p-1}$, and conclude that
        the constant is optimal.
  \item Show that equality $\norm{Hf}_p = \frac
        p{p-1}\norm f_p$ with $f \neq 0$ is impossible.
        \emph{(Track the equality case of H\"older in question
        3: it would force $f = cx^{-1/p}$-type behavior, which
        is not in $L^p$.)}
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The discrete inequality.}
\begin{enumerate}[resume]
  \item For a nonincreasing $g \geq 0$ on $(0,\infty)$ and
        $a_n = g(n)$, compare $\sum a_n^p$ and $\int g^p$,
        and $H$-averages accordingly, to deduce from Part I
        \emph{Hardy's discrete inequality}: for $a_n \geq 0$,
        \[
        \sum_{n\geq1}\Bigl(\frac{a_1 + \dots +
        a_n}{n}\Bigr)^{p}
        \;\leq\;
        \Bigl(\frac{p}{p-1}\Bigr)^{p}\,\sum_{n\geq1}a_n^p
        \]
        --- prove it first for nonincreasing $(a_n)$ via the
        comparison above, then reduce the general case to the
        nonincreasing one by rearrangement \emph{(admit, with
        a one-line justification, that sorting $(a_n)$ in
        decreasing order can only increase the left side while
        fixing the right)}.
  \item Deduce: if $\sum a_n^p < \infty$ then the Ces\`aro
        means of $(a_n)$ are again $\ell^p$ --- and give an
        example ($p = 2$) where $(a_n) \in \ell^2$ but $a_n$
        is not summable, yet Hardy still controls the means.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Epilogue.}
\begin{enumerate}[resume]
  \item Show that Hardy's inequality fails for $p = 1$: with
        $f = \mathbf 1_{\intcc01}$, compute $Hf$ and observe
        $Hf \notin L^1$. Where does the proof break?
\end{enumerate}

\medskip
\noindent\textbf{Part V --- The maximal function, and
Lebesgue's differentiation theorem.} Hardy averages from the
origin; Hardy--Littlewood average \emph{around each point}.
For $f \in L^1(\R)$ define
\[
Mf(x) = \sup_{r>0}\ \frac1{2r}\int_{x-r}^{x+r}\abs
f\,\dd\lambda .
\]
\begin{enumerate}[resume]
  \item (Vitali, finite version) Let $B_1, \dots, B_N$ be
        open intervals. Show there is a disjoint subfamily
        $B_{i_1}, \dots, B_{i_k}$ with $\bigcup_jB_j
        \subseteq \bigcup_l3B_{i_l}$, where $3B$ denotes the
        interval with the same \omterm{ex:b3:groups:actions}{center} and triple length
        \emph{(greedy: repeatedly pick the longest interval
        disjoint from those already picked)}.
  \item (Weak type $(1,1)$) Show that for every $t > 0$,
        \[
        \lambda\bigl(\{Mf > t\}\bigr) \;\leq\;
        \frac3t\,\norm f_1 :
        \]
        each $x$ with $Mf(x) > t$ owns a centered interval
        $B_x$ with $\int_{B_x}\abs f > t\,\lambda(B_x)$; take
        a \omterm{def:b3:topology:compact}{compact} $K \subseteq \{Mf > t\}$ (inner regularity),
        cover it by finitely many $B_x$, apply question 11,
        and exhaust.
  \item Compute $M\mathbf 1_{\intcc01}(x)$ for $x > 1$ and
        deduce that $Mf \notin L^1$ for every $f \neq 0$
        ($Mf(x) \geq \frac c{\abs x}$ at infinity): at $p =
        1$, the weak inequality of question 12 is the best
        possible statement.
  \item (Strong type for $p > 1$) For $f \in L^p$: split $f =
        f\,\mathbf 1_{\abs f > t/2} + f\,\mathbf 1_{\abs f
        \leq t/2}$, observe $Mf \leq M\bigl(f\mathbf 1_{\abs
        f > t/2}\bigr) + \frac t2$, and combine question 12
        with the layer-cake formula
        (\cref{prop:b3:product:layercake}) and Tonelli to
        prove
        \[
        \norm{Mf}_p^p \;\leq\;
        \frac{6p\,2^{p-1}}{p-1}\,\norm f_p^p .
        \]
        (The blow-up as $p \downarrow 1$ is question 13's
        failure, quantified.)
  \item (Lebesgue differentiation theorem) Prove: for $f \in
        L^1(\R)$,
        \[
        \frac1{2r}\int_{x-r}^{x+r}f\,\dd\lambda
        \;\longrightarrow\; f(x)
        \qquad (r \to 0)\quad\text{for a.e.\ }x .
        \]
        \emph{(Clear for \omterm{def:b3:topology:continuity}{continuous} $f$. In general write $f
        = g + h$, $g$ \omterm{def:b3:topology:continuity}{continuous} with \omterm{def:b3:topology:compact}{compact} support,
        $\norm h_1 < \varepsilon$
        (\cref{thm:b3:lp:density}); the set where
        $\limsup_{r\to0}$ of the averaged oscillation exceeds
        $\delta$ sits inside $\{Mh > \delta/2\} \cup \{\abs h
        > \delta/2\}$, of \omterm{def:b3:measure:measure}{measure} $O(\varepsilon/\delta)$;
        let $\varepsilon \to 0$, then $\delta \to 0$ along a
        sequence.)}
  \item Deduce: (a) almost every point is a \emph{Lebesgue
        point} of $f$; (b) for $f \in L^1$, the primitive
        $F(x) = \int_0^xf$ is differentiable a.e.\ with $F' =
        f$ a.e.\ --- the integral half of the fundamental
        theorem of calculus in the Lebesgue world, closing
        the circle opened by the staircase
        (\cref{pb:b3:measure:1}), which showed the converse
        half can fail.
  \item (\omterm{ex:b3:lebesgue:gamma}{Density} points) For \omterm{def:b3:lebesgue:measurable}{measurable} $A \subseteq \R$,
        show that almost every $x \in A$ satisfies
        $\frac{\lambda(A\cap\intcc{x-r}{x+r})}{2r} \to 1$:
        \omterm{def:b3:lebesgue:measurable}{measurable} sets are locally full at almost all of
        their points. Sketch, in two lines, how this yields
        yet another proof of Steinhaus' theorem
        (\cref{exo:b3:measure:8}).
\end{enumerate}

\medskip
\noindent\textbf{Part VI --- Variations on the theme of
averaging.}
\begin{enumerate}[resume]
  \item (Weighted Hardy) For $\alpha < p - 1$ and $f \geq 0$,
        show
        \[
        \int_0^\infty\Bigl(\frac{F(x)}x\Bigr)^{p}
        x^{\alpha}\,\dd x \;\leq\;
        \Bigl(\frac{p}{p - 1 - \alpha}\Bigr)^{p}
        \int_0^\infty f(x)^p\,x^{\alpha}\,\dd x
        \]
        by the same integration by parts, and check that the
        borderline $\alpha = p - 1$ is genuinely forbidden
        (adapt question 10's counterexample).
  \item (The adjoint) Let $H^*f(x) =
        \int_x^{\infty}\frac{f(t)}t\,\dd t$. Show $\langle
        Hf, g\rangle = \langle f, H^*g\rangle$ for
        nonnegative $f, g$ (Tonelli), and prove
        $\norm{H^*f}_p \leq p\,\norm f_p$ \emph{(directly by
        parts, or from Hardy on the conjugate exponent by
        duality --- mind which exponent picks up which
        constant)}.
  \item (A Hilbert-type inequality) Deduce that for
        nonnegative $f \in L^p$, $g \in L^q$:
        \[
        \int_0^\infty\!\!\int_0^\infty
        \frac{f(x)\,g(y)}{\max(x,y)}\,\dd x\,\dd y
        \;\leq\; (p + q)\,\norm f_p\,\norm g_q
        \]
        \emph{(split along $y < x$ / $y \geq x$: each half is
        a pairing of one function against a Hardy transform
        of the other)}.
  \item (Optimality, discrete) Show that the constant
        $\bigl(\frac p{p-1}\bigr)^p$ of question 8 is also
        optimal: test on $a_n = n^{-1/p}\,\mathbf 1_{n \leq
        N}$, compare both sides with integrals, and let $N
        \to \infty$ (the discrete mirror of Part II).
  \item (Synthesis) Three averaging operators appeared in
        this problem: Hardy's $H$, the discrete Ces\`aro
        mean, and the maximal operator $M$. State in one
        line each what its boundedness says, observe that all
        three fail exactly at $p = 1$, and explain why it is
        the same failure three times (the harmonic tail
        $\frac1x$).
\end{enumerate}

\medskip
\noindent\textbf{Part VII --- Carleman's inequality, and how
sharp is sharp.}
\begin{enumerate}[resume]
  \item (Carleman's inequality) Let $a_n \geq 0$ with $\sum
        a_n < \infty$. Apply the discrete Hardy inequality of
        question 8 to $b_n = a_n^{1/p}$, use the
        arithmetic--geometric mean inequality, and let $p \to
        \infty$ (show that $p \mapsto
        \bigl(\frac p{p-1}\bigr)^p$ decreases to $\eu$) to
        obtain
        \[
        \sum_{n\geq1}\bigl(a_1a_2\cdots a_n\bigr)^{1/n}
        \;\leq\; \eu\,\sum_{n\geq1}a_n :
        \]
        the geometric means of a summable sequence are
        summable, at cost at most $\eu$.
  \item (The constant $\eu$ is optimal) Test $a_n =
        \frac1n\,\mathbf 1_{n\leq N}$: using the Stirling
        bracketing of \cref{pb:b3:product:1}, show
        $(n!)^{-1/n} = \frac\eu n\bigl(1 +
        O\bigl(\frac{\ln n}n\bigr)\bigr)$, deduce that both
        sides of Carleman grow like $\eu\ln N$, and conclude
        that no constant smaller than $\eu$ can work. (Observe
        the pattern: the optimizers of Hardy and of Carleman
        are both the harmonic-type sequences that just fail to
        be in the space.)
  \item (How slowly is ``sharp'' approached?) Take $p = 2$.
        For $f = \mathbf 1_{\intcc01}$, compute
        $\norm{Hf}_2/\norm f_2 = \sqrt2$, against the bound
        $2$. For the near-optimizers $f_A$ of question 5,
        prove the exact identity
        \[
        \norm{Hf_A}_2^2 = 4\ln A - 8 + \frac{8}{\sqrt A},
        \qquad\text{so}\qquad
        \frac{\norm{Hf_A}_2^2}{\norm{f_A}_2^2}
        = 4 - \frac{8 - 8A^{-1/2}}{\ln A} .
        \]
        Evaluate at $A = \eu^{10}$ (ratio $\approx 1.790$)
        and comment: the supremum $2$ is approached at speed
        $1/\ln A$ only --- an optimal constant can be all but
        invisible numerically.
\end{enumerate}
\end{problem}
