\chapter{The \texorpdfstring{$L^p$}{Lp} Spaces}\label{ch:b3:lp}

The Lebesgue integral was built for analysis; the $L^p$ spaces
are where that analysis lives. They are Banach spaces
(Riesz--Fischer) --- the completions that
\cref{exo:b3:complete:1} showed the continuous functions
lacked --- and they support a smoothing technology,
\emph{convolution against mollifiers}, that approximates any
$L^p$ function by $\mathcal C^\infty$ ones. This chapter proves
the integral versions of H\"older and Minkowski, completeness,
the density theorems, and the regularization machine, ending
with the inclusion and interpolation geography of the $L^p$
scale. Throughout, $(X, \mathcal A, \mu)$ is a measure space
and functions are complex-valued; on $\R^d$, the measure is
$\lambda_d$.

\section{Definition; H\"older and Minkowski}

\begin{definition}\label{def:b3:lp:space}
For $1 \leq p < \infty$, $\mathcal L^p(\mu)$ is the set of
measurable $f$ with $\norm f_p = \bigl(\int\abs
f^p\dd\mu\bigr)^{1/p} < \infty$, and $\mathcal L^\infty(\mu)$
the set of $f$ bounded outside a null set, with $\norm
f_\infty$ the \emph{essential sup} --- the least $M$ with
$\abs f \leq M$ a.e.\ (the inf is attained: intersect the null
sets for $M + \frac1n$). Since $\norm f_p = 0$ only forces $f
= 0$ \emph{a.e.} (\cref{exo:b3:lebesgue:5}), we define
\[
L^p(\mu) = \mathcal L^p(\mu)/\{f = 0 \text{ a.e.}\} :
\]
elements are classes of functions modulo null sets, and
$\norm\cdot_p$ is a genuine norm on
$L^p$.\index{Lp space@$L^p$ space}
\end{definition}

\begin{theorem}[H\"older's inequality]\label{thm:b3:lp:holder}
Let $1 \leq p, q \leq \infty$ with $\frac1p + \frac1q = 1$
(conjugate exponents). For measurable $f, g$:\index{H\"older's
inequality}
\[
\norm{fg}_1 \leq \norm f_p\,\norm g_q ,
\]
with equality (for $1 < p < \infty$, finite norms, $f,g\ne0$)
iff $\abs f^p$ and $\abs g^q$ are proportional a.e.
\end{theorem}

\begin{proof}
The cases $\{p, q\} = \{1, \infty\}$ are direct ($\abs{fg} \leq
\norm g_\infty\abs f$ a.e.). Let $1 < p < \infty$; normalize
$\norm f_p = \norm g_q = 1$ (homogeneity; zero or infinite
norms are trivial). Young's inequality $ab \leq \frac{a^p}p +
\frac{b^q}q$ ($a, b \geq 0$; concavity of $\ln$, as in
\cref{pb:b3:banach:1}) gives pointwise $\abs{f g} \leq
\frac{\abs f^p}p + \frac{\abs g^q}q$; integrate: $\norm{fg}_1
\leq \frac1p + \frac1q = 1$. Equality forces a.e.\ equality in
Young, i.e.\ $\abs f^p = \abs g^q$ a.e.\ (after the
normalization; undoing it, proportionality).
\end{proof}

\begin{theorem}[Minkowski's inequality]\label{thm:b3:lp:minkowski}
For $1 \leq p \leq \infty$: $\norm{f + g}_p \leq \norm f_p +
\norm g_p$.\index{Minkowski's inequality}
\end{theorem}

\begin{proof}
$p = 1, \infty$: pointwise/a.e.\ triangle inequality. For $1 <
p < \infty$, assume $\norm{f+g}_p < \infty$ (else use
$\abs{f+g}^p \leq 2^{p-1}(\abs f^p + \abs g^p)$, from convexity
of $t^p$, to see the left side is finite when the right is).
Then
\[
\norm{f{+}g}_p^p \leq \int\abs f\,\abs{f{+}g}^{p-1} +
\int\abs g\,\abs{f{+}g}^{p-1}
\leq \bigl(\norm f_p + \norm g_p\bigr)\,
\bigl\|\abs{f{+}g}^{p-1}\bigr\|_q
\]
by H\"older, and $\norm{\abs{f+g}^{p-1}}_q =
\norm{f+g}_p^{p/q}$ since $(p-1)q = p$; divide by
$\norm{f+g}_p^{p/q}$ (if nonzero; else trivial) and use $p -
\frac pq = 1$.
\end{proof}

\section{Completeness and its companions}

\begin{theorem}[Riesz--Fischer]\label{thm:b3:lp:complete}
For $1 \leq p \leq \infty$, $L^p(\mu)$ is a Banach space.
Moreover, every sequence converging in $L^p$ has a subsequence
converging \emph{almost everywhere} (with an $L^p$ dominator
in the case $p < \infty$).\index{Riesz--Fischer theorem}
\end{theorem}

\begin{proof}
$p = \infty$: a $\norm\cdot_\infty$-Cauchy sequence is, off a
single null set (union of countably many), uniformly Cauchy:
it converges uniformly off it; done. Let $p < \infty$. By
\cref{exo:b3:complete:1}(b) it suffices to sum absolutely
convergent series: let $\sum\norm{f_k}_p = M < \infty$. Set
$G_n = \sum_{k \leq n}\abs{f_k}$ and $G = \sum_k\abs{f_k}$
(pointwise in $[0,\infty]$): by Minkowski $\norm{G_n}_p \leq
M$, and MCT ($G_n^p \nearrow G^p$) gives $\int G^p \leq M^p$:
$G < \infty$ a.e., so the series $\sum f_k(x)$ converges
absolutely for a.e.\ $x$; call the sum $S(x)$ (any value on
the null set). Then $\abs{S - \sum_{k\leq n}f_k}^p \leq (2G)^p
\in L^1$, and DCT gives $\norm{S - \sum_{k\leq n}f_k}_p \to 0$:
the series converges in $L^p$.

The subsequence statement: if $f_n \to f$ in $L^p$, pick $n_k$
with $\norm{f_{n_{k+1}} - f_{n_k}}_p \leq 2^{-k}$; the series
$\sum(f_{n_{k+1}} - f_{n_k})$ falls under the previous
argument: it converges absolutely a.e., dominated by a $G \in
L^p$, so $f_{n_k} \to f_{n_1} + \sum(\cdots) $ a.e., and this
a.e.-limit must be (a representative of) $f$ (both are $L^p$
limits). The dominator: $\abs{f_{n_k}} \leq \abs{f_{n_1}} + G$.
\end{proof}

\begin{remark}\label{rem:b3:lp:modes}
$L^p$ convergence does not imply a.e.\ convergence (the
\emph{typewriter} sequence, \cref{exo:b3:lp:3}), nor
conversely (escaping bumps): the two modes are linked only
through subsequences and domination. Keeping the counterexamples
of \cref{exo:b3:lp:3} in mind is the best vaccine.
\end{remark}

\section{Density theorems}

\begin{theorem}\label{thm:b3:lp:density}
Let $1 \leq p < \infty$.
\begin{enumerate}
  \item Simple functions (with finite-measure supports) are
        dense in $L^p(\mu)$.
  \item In $L^p(\R^d)$, the continuous compactly supported
        functions $\mathcal C_c(\R^d)$ are dense.
  \item Translation is continuous on $L^p(\R^d)$: writing
        $\tau_hf = f(\cdot - h)$, $\norm{\tau_hf - f}_p \to 0$
        as $h \to 0$.
\end{enumerate}
None of the three holds for $p = \infty$.
\end{theorem}

\begin{proof}
(1) For $f \geq 0$: the dyadic $s_n \nearrow f$ of
\cref{thm:b3:lebesgue:approximation} satisfy $\abs{f - s_n}^p
\leq f^p \in L^1$: DCT. (Each $s_n \leq f$ lies in $L^p$, and
its level sets have finite measure where the value is
positive: $\mu(s_n \geq c) \leq c^{-p}\int f^p$.) Split
general $f$ into four nonnegative parts.

(2) By (1) it suffices to approximate $\mathbf 1_A$, $A$ Borel
with $\lambda_d(A) < \infty$. Regularity (proof as in
\cref{thm:b3:measure:regularity}) gives compact $K \subseteq A
\subseteq U$ open with $\lambda_d(U\setminus K) <
\varepsilon$; Urysohn's function
\[
\varphi(x) = \frac{d(x, \R^d\setminus U)}{d(x, \R^d\setminus U)
+ d(x, K)}
\]
is continuous, $1$ on $K$, $0$ outside $U$, and can be taken
compactly supported (shrink $U$ to a bounded open first). Then
$\norm{\mathbf 1_A - \varphi}_p^p \leq \lambda_d(U\setminus K)
< \varepsilon$.

(3) For $g \in \mathcal C_c$: uniform continuity gives
$\norm{\tau_hg - g}_\infty \to 0$, with supports in a fixed
compact for $\abs h \leq 1$: $\norm{\tau_hg - g}_p \to 0$. For
general $f$: pick $g \in \mathcal C_c$ with $\norm{f - g}_p <
\varepsilon$; then $\norm{\tau_hf - f}_p \leq 2\norm{f - g}_p +
\norm{\tau_hg - g}_p$ (translation invariance of the norm).

For $p = \infty$: uniform approximation of $\mathbf
1_{\intoo0\infty}$ by continuous functions is impossible
(jump), and $\norm{\tau_h\mathbf 1_{\intoo0\infty} - \mathbf
1_{\intoo0\infty}}_\infty = 1$ for $h \neq 0$.
\end{proof}

\section{Convolution and regularization}

\begin{theorem}[Young's inequality]\label{thm:b3:lp:young}
Let $1 \leq p \leq \infty$, $f \in L^1(\R^d)$, $g \in
L^p(\R^d)$. Then $f * g$ is defined a.e., belongs to $L^p$,
and\index{Young's inequality}
\[
\norm{f * g}_p \leq \norm f_1\,\norm g_p .
\]
\end{theorem}

\begin{proof}
$p = \infty$: direct bound. $p = 1$:
\cref{thm:b3:product:convolution}. Let $1 < p < \infty$, $q$
conjugate. Split $\abs{f(y)} = \abs{f(y)}^{1/q}\cdot
\abs{f(y)}^{1/p}$ and apply H\"older:
\[
\int\abs{f(y)}\,\abs{g(x{-}y)}\,\dd y
\leq \Bigl(\int\abs f\Bigr)^{1/q}
\Bigl(\int\abs{f(y)}\,\abs{g(x - y)}^p\,\dd y\Bigr)^{1/p} .
\]
Raise to the $p$-th power and integrate in $x$; Tonelli on the
second factor gives $\norm f_1^{p/q}\cdot\norm f_1\norm g_p^p$,
i.e.\ $\norm{f*g}_p^p \leq \norm f_1^{1 + p/q}\norm g_p^p =
(\norm f_1\norm g_p)^p$ --- and the finiteness of the Tonelli
integral justifies a.e.\ absolute convergence as in
\cref{thm:b3:product:convolution}.
\end{proof}

\begin{definition}[Mollifiers]\label{def:b3:lp:mollifier}
The function\index{mollifier}\index{bump function}
\[
\rho(x) = \begin{cases}
c\,\exp\Bigl(-\dfrac{1}{1 - \norm x^2}\Bigr) & \norm x < 1,\\
0 & \norm x \geq 1,
\end{cases}
\]
with $c$ normalizing $\int\rho = 1$, is $\mathcal C^\infty$ on
$\R^d$: the point is that $t \mapsto \eu^{-1/t}\mathbf
1_{t>0}$ is $\mathcal C^\infty$ on $\R$, all its derivatives at
$0^+$ being $0$ (each derivative is $P(1/t)\eu^{-1/t}$ for a
polynomial $P$, which tends to $0$; induction). For
$\varepsilon > 0$ set $\rho_\varepsilon(x) =
\varepsilon^{-d}\rho(x/\varepsilon)$: supported in $\bar B(0,
\varepsilon)$, still of integral $1$.
\end{definition}

\begin{theorem}[Regularization]\label{thm:b3:lp:regularization}
Let $1 \leq p < \infty$ and $f \in L^p(\R^d)$. Then:
\begin{enumerate}
  \item $f * \rho_\varepsilon \in \mathcal C^\infty(\R^d)$,
        with $\partial^\alpha(f * \rho_\varepsilon) = f *
        \partial^\alpha\rho_\varepsilon$;
  \item $\norm{f * \rho_\varepsilon - f}_p \to 0$ as
        $\varepsilon \to 0$;
  \item consequently $\mathcal C^\infty_c(\R^d)$ is dense in
        $L^p(\R^d)$.
\end{enumerate}
\end{theorem}

\begin{proof}
(1) Differentiation under the integral
(\cref{thm:b3:lebesgue:paramdiff}) in $x$: for $x$ in a ball
$B$, $\abs{\partial_{x_i}\rho_\varepsilon(x - y)} \leq
C_\varepsilon\,\mathbf 1_{K}(y)$ with $K$ compact ($y$ within
$\varepsilon$ of $B$), and $\abs f\,\mathbf 1_K \in L^1$
(H\"older against $\mathbf 1_K$): the theorem applies;
iterate for higher derivatives.

(2) Since $\int\rho_\varepsilon = 1$:
\[
(f * \rho_\varepsilon)(x) - f(x)
= \int \bigl(f(x - y) - f(x)\bigr)\rho_\varepsilon(y)\,\dd y ,
\]
and Minkowski's \emph{integral} inequality --- or directly:
H\"older/Jensen with the probability measure
$\rho_\varepsilon\dd y$ and Tonelli ---
\[
\norm{f*\rho_\varepsilon - f}_p^p
\leq \int\Bigl(\int\abs{f(x-y) -
f(x)}^p\dd x\Bigr)\rho_\varepsilon(y)\,\dd y
= \int \norm{\tau_yf - f}_p^p\;\rho_\varepsilon(y)\,\dd y
\]
(the middle step: apply Jensen's inequality,
\cref{exo:b3:lp:10}, to the inner $y$-integral, then Tonelli).
The integrand is supported in $\norm y \leq \varepsilon$ and
tends to $0$ there uniformly as $\varepsilon \to 0$
(\cref{thm:b3:lp:density}(3)): the whole expression tends to
$0$.

(3) Approximate $f$ by $g \in \mathcal C_c$
(\cref{thm:b3:lp:density}(2)), then $g$ by $g *
\rho_\varepsilon \in \mathcal C_c^\infty$ (compact support:
sum of supports).
\end{proof}

\begin{corollary}[Fundamental lemma of the calculus of
variations]\label{cor:b3:lp:fundlemma}
Let $f \in L^1_{\mathrm{loc}}(\R^d)$ (integrable on compacts)
with $\int f\varphi = 0$ for every $\varphi \in \mathcal
C^\infty_c(\R^d)$. Then $f = 0$ a.e.\index{fundamental lemma of
the calculus of variations}
\end{corollary}

\begin{proof}
Fix a ball $B = B(0, R)$ and let $g = f\mathbf 1_{B(0, R+1)}
\in L^1$. For $x \in B$ and $\varepsilon < 1$: $(g *
\rho_\varepsilon)(x) = \int f(y)\rho_\varepsilon(x - y)\dd y =
0$, the test function being $y \mapsto \rho_\varepsilon(x-y)
\in \mathcal C_c^\infty$. But $g * \rho_\varepsilon \to g$ in
$L^1$ (\cref{thm:b3:lp:regularization}): $g = 0$ a.e.\ on $B$;
exhaust $\R^d$.
\end{proof}

\section{The \texorpdfstring{$L^p$}{Lp} geography}

\begin{proposition}\label{prop:b3:lp:inclusions}
(a) If $\mu(X) < \infty$ and $1 \leq p \leq q \leq \infty$,
then $L^q \subseteq L^p$ with $\norm f_p \leq
\mu(X)^{\frac1p - \frac1q}\,\norm f_q$.
(b) On $\R^d$ (infinite measure) there are no inclusions:
for $p \neq q$ there are functions in $L^p\setminus L^q$.
(c) (Interpolation) If $p < r < q$ and $\alpha \in \intoo01$
is defined by $\frac1r = \frac\alpha p + \frac{1 - \alpha}q$,
then
\[
\norm f_r \leq \norm f_p^{\alpha}\,\norm f_q^{1 - \alpha} ;
\]
in particular $L^p \cap L^q \subseteq L^r$.
\end{proposition}

\begin{proof}
(a) H\"older with exponents $\frac qp$ and its conjugate:
$\int\abs f^p\cdot 1 \leq \norm{\abs f^p}_{q/p}\,\norm
1_{(q/p)'} = \norm f_q^p\,\mu(X)^{1 - p/q}$ ($q = \infty$
directly). (b) Near $0$ and near $\infty$, powers $x^{-\alpha}$
calibrate: \cref{exo:b3:lp:2}. (c) Write $\abs f^r = \abs
f^{r\alpha}\abs f^{r(1-\alpha)}$ and apply H\"older with the
conjugate pair $\frac p{r\alpha}$, $\frac q{r(1-\alpha)}$
(conjugates precisely by the definition of $\alpha$):
$\int\abs f^r \leq \norm f_p^{r\alpha}\norm f_q^{r(1 -
\alpha)}$.
\end{proof}

\begin{method}\label{met:b3:lp:toolkit}
The $L^p$ toolkit, as used everywhere below: to prove an
identity or inequality for all $f \in L^p$ --- prove it on a
dense class ($\mathcal C_c^\infty$ via
\cref{thm:b3:lp:regularization}) and extend by continuity
(\cref{thm:b3:complete:extension}, both sides being
$L^p$-continuous); to prove $f = 0$, test against $\mathcal
C_c^\infty$ (\cref{cor:b3:lp:fundlemma}); to gain smoothness,
convolve; to trade exponents, H\"older and interpolation. The
Fourier theory of \cref{ch:b3:fouriertransform} is one long
application of this method.
\end{method}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:b3:lp:1}
(a) State and prove the Cauchy--Schwarz inequality in
$L^2(\mu)$ as the case $p = q = 2$ of H\"older.
(b) On a \emph{probability} space, show $p \mapsto \norm f_p$
is nondecreasing.
(c) When is H\"older an equality for $p = 1$, $q = \infty$?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b3:lp:2}
For which $p \in \intco1\infty$ do the following belong to
$L^p$?
\[
x^{-1/2}\mathbf 1_{\intoo01},\qquad
x^{-1/2}\mathbf 1_{\intoo1\infty},\qquad
\frac{1}{x^{1/2}(1 + \abs{\ln x})}\ \text{on } \intoo01,
\qquad
\frac1{1 + \abs x}\ \text{on } \R .
\]
Conclude: on $\intoo01$ small $p$ is easier, on
$\intoo1\infty$ large $p$ is easier, and no $L^p$ contains
another on $\R$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:lp:3}
(The typewriter) Enumerate the dyadic intervals $I_1 =
\intcc01$, $I_2 = \intcc0{\frac12}$, $I_3 = \intcc{\frac12}1$,
$I_4 = \intcc0{\frac14}$, \dots\ and let $f_n = \mathbf
1_{I_n}$.
(a) Show $f_n \to 0$ in every $L^p(\intcc01)$, $p < \infty$,

(b) Exhibit the a.e.-convergent subsequence promised by
\cref{thm:b3:lp:complete}.
(c) Conversely give a sequence converging a.e.\ but not in
$L^1$, and one converging in $L^1$ but in no $L^p$, $p > 1$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:lp:4}
Let $\mu(X) < \infty$ and $f \in L^\infty(\mu)$, $f \neq 0$.
Show that $\norm f_p \to \norm f_\infty$ as $p \to \infty$.
\emph{(Upper bound by (a) of \cref{prop:b3:lp:inclusions};

\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:lp:5}
(a) Where exactly does the proof of
\cref{thm:b3:lp:density}(3) use $p < \infty$?
(b) Show that $f \in L^\infty(\R)$ satisfies $\norm{\tau_hf -
f}_\infty \to 0$ iff $f$ has a uniformly continuous

\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:lp:6}
Let $p, q$ be conjugate, $f \in L^p(\R^d)$, $g \in
L^q(\R^d)$. Show that $f * g$ is defined \emph{everywhere},

\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:lp:7}
Let $f \in L^1_{\mathrm{loc}}(\intoo ab)$ with

\end{exercise}

(b) show that $f \mapsto \norm f_p$ is, for fixed $f$,

\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:lp:10}

\section{Problem: Hardy's inequality}

\begin{problem}[{Weekend problem --- Hardy's inequality and
its sharp constant}]\label{pb:b3:lp:1}
For $f \in L^p(\intoo0{+\infty})$, $1 < p < \infty$, define
the \emph{Hardy operator}
\[
(Hf)(x) = \frac1x\int_0^x f(t)\,\dd t .
\]
Hardy's inequality (1920) asserts\index{Hardy's inequality}
\[
\norm{Hf}_p \;\leq\; \frac{p}{p-1}\,\norm f_p ,
\]
and the constant $\frac p{p-1}$ is optimal and not attained.
This problem proves everything, then extends to series.

\medskip
\noindent\textbf{Part I --- The inequality.} Assume first $f
\geq 0$ continuous with compact support in
$\intoo0{+\infty}$, and let $F(x) = \int_0^xf$.
\begin{enumerate}
  \item Show that $Hf \in L^p$: near $0$, $F$ vanishes on a
        neighborhood of $0$; near $\infty$, $F$ is bounded, so
        $(Hf)(x) = O(1/x)$, and $x \mapsto \frac1x$ belongs to
        $L^p(\intoo1{+\infty})$ for $p > 1$.
  \item Integrate by parts to show
        \[
        \int_0^\infty \Bigl(\frac Fx\Bigr)^p\dd x
        = \frac{p}{p-1}\int_0^\infty\Bigl(\frac
        Fx\Bigr)^{p-1}f(x)\,\dd x .
        \]
        \emph{(Differentiate $x^{1-p}F^p$; boundary terms
        vanish --- justify both ends.)}
  \item Apply H\"older to the right side and deduce
        $\norm{Hf}_p \leq \frac p{p-1}\norm f_p$ for such $f$.
  \item Extend to all of $L^p$: for $f \geq 0$, construct
        $f_n$ continuous with compact support in
        $\intoo0{+\infty}$, $0 \leq f_n \nearrow f$ a.e.\
        \emph{(truncate, then approximate monotonically ---
        justify the construction)}; then $Hf_n \nearrow Hf$
        pointwise (MCT inside the average) and MCT passes the
        inequality to the limit. For signed or complex $f$,
        conclude with $\abs{Hf} \leq H\abs f$.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Optimality.}
\begin{enumerate}[resume]
  \item For $A > 1$ let $f_A(t) = t^{-1/p}\,\mathbf
        1_{\intcc1A}(t)$. Compute $\norm{f_A}_p^p = \ln A$
        and, for $1 \leq x \leq A$,
        \[
        (Hf_A)(x) = \frac p{p-1}\;x^{-1/p}\,
        \bigl(1 - x^{-(1 - 1/p)}\bigr).
        \]
  \item Deduce $\liminf_{A\to\infty} \norm{Hf_A}_p/
        \norm{f_A}_p \geq \frac{p}{p-1}$, and conclude that
        the constant is optimal.
  \item Show that equality $\norm{Hf}_p = \frac
        p{p-1}\norm f_p$ with $f \neq 0$ is impossible.
        \emph{(Track the equality case of H\"older in question
        3: it would force $f = cx^{-1/p}$-type behavior, which
        is not in $L^p$.)}
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The discrete inequality.}
\begin{enumerate}[resume]
  \item For a nonincreasing $g \geq 0$ on $(0,\infty)$ and
        $a_n = g(n)$, compare $\sum a_n^p$ and $\int g^p$,
        and $H$-averages accordingly, to deduce from Part I
        \emph{Hardy's discrete inequality}: for $a_n \geq 0$,
        \[
        \sum_{n\geq1}\Bigl(\frac{a_1 + \dots +
        a_n}{n}\Bigr)^{p}
        \;\leq\;
        \Bigl(\frac{p}{p-1}\Bigr)^{p}\,\sum_{n\geq1}a_n^p
        \]
        --- prove it first for nonincreasing $(a_n)$ via the
        comparison above, then reduce the general case to the
        nonincreasing one by rearrangement \emph{(admit, with
        a one-line justification, that sorting $(a_n)$ in
        decreasing order can only increase the left side while
        fixing the right)}.
  \item Deduce: if $\sum a_n^p < \infty$ then the Ces\`aro
        means of $(a_n)$ are again $\ell^p$ --- and give an
        example ($p = 2$) where $(a_n) \in \ell^2$ but $a_n$
        is not summable, yet Hardy still controls the means.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Epilogue.}
\begin{enumerate}[resume]
  \item Show that Hardy's inequality fails for $p = 1$: with
        $f = \mathbf 1_{\intcc01}$, compute $Hf$ and observe
        $Hf \notin L^1$. Where does the proof break?
\end{enumerate}
\end{problem}
