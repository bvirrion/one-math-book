\chapter{Characteristic Functions and the Central Limit
Theorem}\label{ch:b3:clt}

The law of large numbers says averages converge; the central
limit theorem says \emph{how they fluctuate}: the error,
magnified by $\sqrt n$, is asymptotically \omterm{def:b3:clt:gaussianvector}{Gaussian} ---
whatever the law one started from. This universality is the
deepest fact of elementary probability, and its natural proof
is Fourier-analytic: the \emph{\omterm{def:b3:clt:cf}{characteristic function}}
(the Fourier transform of a law) converts \omterm{def:b3:probability:independence}{independent} sums
into products, and \cref{ch:b3:fouriertransform}'s machinery
--- injectivity, \omterm{def:b3:clt:gaussianvector}{Gaussian} fixed points --- converts pointwise
convergence of these products into convergence of laws
(L\'evy's theorem, proved in full). The chapter ends with
\omterm{def:b3:clt:gaussianvector}{Gaussian vectors} and the honest derivation of the confidence
intervals used everywhere in statistics; the weekend problem
gives Lindeberg's second proof of the CLT, with an explicit
error rate.

\section{Characteristic functions}

\begin{definition}\label{def:b3:clt:cf}
The \emph{characteristic function}\index{characteristic
function} of a real \omterm{def:b3:probability:space}{random variable} $X$ is
\[
\varphi_X(\xi) = \E\bigl[\eu^{\iu\xi X}\bigr]
= \int_\R \eu^{\iu\xi x}\,\dd\P_X(x)
\qquad (\xi \in \R)
\]
(the transfer theorem computes it from the law; for a \omterm{ex:b3:lebesgue:gamma}{density}
$f$, $\varphi_X(\xi) = \hat f(-\xi)$ in
\cref{ch:b3:fouriertransform}'s convention).
\end{definition}

\begin{proposition}\label{prop:b3:clt:cfbasics}
(a) $\varphi_X(0) = 1$, $\abs{\varphi_X} \leq 1$, and
$\varphi_X$ is uniformly \omterm{def:b3:topology:continuity}{continuous};
$\varphi_{aX + b}(\xi) = \eu^{\iu b\xi}\varphi_X(a\xi)$.
(b) If $X, Y$ are \emph{\omterm{def:b3:probability:independence}{independent}}:
$\varphi_{X+Y} = \varphi_X\,\varphi_Y$.
(c) If $\E\abs X^k < \infty$, then $\varphi_X \in \mathcal
C^k$ with $\varphi_X^{(j)}(0) = \iu^j\,\E[X^j]$ for $j \leq
k$; in particular, for centered $X \in L^2$ with variance
$\sigma^2$:
\[
\varphi_X(\xi) = 1 - \frac{\sigma^2\xi^2}{2} +
o(\xi^2) \qquad (\xi \to 0).
\]
(d) \omterm{def:b3:clt:gaussianvector}{Gaussian}: $X \sim \mathcal N(m, \sigma^2)$ has
$\varphi_X(\xi) = \eu^{\iu m\xi - \sigma^2\xi^2/2}$.
\end{proposition}

\begin{proof}
(a) Bounds are immediate; \omterm{def:b3:topology:continuity}{continuity}: $\abs{\varphi(\xi + h)
- \varphi(\xi)} \leq \E\abs{\eu^{\iu hX} - 1} \to 0$ as $h
\to 0$ by dominated convergence, uniformly in $\xi$. The
affine rule is a substitution. (b) $\eu^{\iu\xi(X+Y)} =
\eu^{\iu\xi X}\eu^{\iu\xi Y}$, and \omterm{def:b3:probability:space}{expectations} of products
of \omterm{def:b3:probability:independence}{independent} variables factor
(\cref{thm:b3:probability:independence}, applied to real and
imaginary parts). (c) Differentiation under the \omterm{def:b3:probability:space}{expectation},
dominated by $\E\abs X^j$
(\cref{thm:b3:lebesgue:paramdiff}); the Taylor expansion at
$0$ is then Taylor--Young for the $\mathcal C^2$ function
$\varphi$. (d) For $\mathcal N(0,1)$: the \omterm{def:b3:clt:gaussianvector}{Gaussian} transform
(\cref{ex:b3:fouriertransform:gaussian} with $a = \frac12$)
gives $\int\eu^{\iu\xi x}\frac{\eu^{-x^2/2}}{\sqrt{2\pi}}\dd
x = \eu^{-\xi^2/2}$; the general case by the affine rule.
\end{proof}

\begin{theorem}[Injectivity]\label{thm:b3:clt:injectivity}
If $\varphi_X = \varphi_Y$, then $X$ and $Y$ have the same
law. More precisely, for $N \sim \mathcal N(0,1)$ \omterm{def:b3:probability:independence}{independent}
of $X$ and $\varepsilon > 0$, the smoothed variable $X +
\varepsilon N$ has the \omterm{ex:b3:lebesgue:gamma}{density}
\[
p_\varepsilon(x) = \frac1{2\pi}\int_\R
\varphi_X(-\xi)\,\eu^{-\varepsilon^2\xi^2/2}\,
\eu^{\iu\xi x}\,\dd\xi ,
\]
determined by $\varphi_X$ alone; letting $\varepsilon \to 0$
recovers the law of $X$.\index{characteristic function,
injectivity}
\end{theorem}

\begin{proof}
$X + \varepsilon N$ has the \omterm{ex:b3:lebesgue:gamma}{density} $p_\varepsilon(x) =
\E\bigl[g_\varepsilon(x - X)\bigr]$, where $g_\varepsilon$ is
the $\mathcal N(0, \varepsilon^2)$ \omterm{ex:b3:lebesgue:gamma}{density}: indeed for Borel
$B$, \omterm{def:b3:probability:independence}{independence} and Tonelli give $\P(X + \varepsilon N \in
B) = \int\!\!\int\mathbf 1_B(x + \varepsilon
n)g_1(n)\,\dd n\,\dd\P_X(x) = \int_B\E[g_\varepsilon(t -
X)]\dd t$ (substitute, then Tonelli again). Writing
$g_\varepsilon$ by Fourier inversion of its transform
(\cref{exo:b3:fouriertransform:4}, rescaled):
$g_\varepsilon(u) = \frac1{2\pi}\int
\eu^{-\varepsilon^2\xi^2/2}\eu^{\iu\xi u}\dd\xi$, and Fubini
(everything dominated by the \omterm{def:b3:clt:gaussianvector}{Gaussian} factor):
\[
p_\varepsilon(x) = \frac1{2\pi}\int_\R
\varphi_X(-\xi)\,\eu^{-\varepsilon^2\xi^2/2}\,
\eu^{\iu\xi x}\,\dd\xi ,
\]
a functional of $\varphi_X$ alone. If $\varphi_X =
\varphi_Y$: $X + \varepsilon N$ and $Y + \varepsilon N$ have
equal laws for every $\varepsilon$; for bounded \omterm{def:b3:topology:continuity}{continuous}
$f$, $\E f(X + \varepsilon N) \to \E f(X)$ as $\varepsilon
\to 0$ (dominated convergence, $X + \varepsilon N \to X$
pointwise on the product space), so $\E f(X) = \E f(Y)$ for
all such $f$ --- and this determines the law: for each $t$,
squeeze $\mathbf 1_{\intoc{-\infty}t}$ between the bounded
\omterm{def:b3:topology:continuity}{continuous} ramps $f_k^\pm$ (equal to $1$ on
$\intoc{-\infty}{t \mp \frac1k}$, to $0$ beyond $t \pm
\frac1k$, affine between); passing to the limit in $\E
f_k^-(X) \leq F_X(t) \leq \E f_k^+(X)$ gives $F_X(t) =
F_Y(t)$ at every $t$ where both are \omterm{def:b3:topology:continuity}{continuous}, hence
everywhere by \omterm{def:b3:topology:continuity}{right-continuity} and \omterm{ex:b3:lebesgue:gamma}{density} of common
\omterm{def:b3:topology:continuity}{continuity} points (both $F$'s have countably many jumps);
equal distribution functions force equal laws
(\cref{exo:b3:measure:3}, resting on
\cref{thm:b3:measure:uniqueness}).
\end{proof}

\section{Convergence in distribution}

\begin{definition}\label{def:b3:clt:cid}
$X_n$ \emph{converges in distribution} (or in law) to $X$,
written $X_n \Rightarrow X$, if\index{convergence in
distribution}
\[
\E\bigl[f(X_n)\bigr] \longrightarrow \E\bigl[f(X)\bigr]
\qquad\text{for every bounded continuous } f\colon\R\to\R .
\]
Equivalently (\cref{exo:b3:clt:4}): $F_{X_n}(t) \to F_X(t)$
at every \omterm{def:b3:topology:continuity}{continuity} point $t$ of $F_X$. The $X_n$ need not
live on a common \omterm{def:b3:probability:space}{probability space}: only laws matter.
\end{definition}

\begin{theorem}[Helly's selection theorem]
\label{thm:b3:clt:helly}
Every sequence $(F_n)$ of distribution functions has a
subsequence converging pointwise, at every \omterm{def:b3:topology:continuity}{continuity} point
of the limit, to a nondecreasing \omterm{def:b3:topology:continuity}{right-continuous} $G \colon
\R \to \intcc01$ --- possibly with $G(+\infty) - G(-\infty) <
1$ (mass may escape to infinity).\index{Helly's theorem}
\end{theorem}

\begin{proof}
Diagonal extraction gives $F_{n_k}(q) \to \ell(q)$ for every
rational $q$ (values in the \omterm{def:b3:topology:compact}{compact} $\intcc01$). Define
$G(t) = \inf\{\ell(q) : q \in \Q, q > t\}$: nondecreasing;
\omterm{def:b3:topology:continuity}{right-continuous} (an infimum over shrinking rational
\omterm{def:b3:topology:topology}{neighborhoods} from the right). At a \omterm{def:b3:topology:continuity}{continuity} point $t$ of
$G$: for rationals $q_1 < t < q_2$,
\[
\ell(q_1) \leq \liminf F_{n_k}(t) \leq \limsup F_{n_k}(t)
\leq \ell(q_2),
\]
by monotonicity of each $F_{n_k}$. From the definition of
$G$ as an infimum and monotonicity of $\ell$ on the
rationals: $G(s) \leq \ell(q) \leq G(q)$ whenever $s < q$.
Taking $s < q_1 < t$ gives $\ell(q_1) \geq G(s)$, and
$\ell(q_2) \leq G(q_2)$; letting $s \uparrow t$ and $q_2
\downarrow t$, \omterm{def:b3:topology:continuity}{continuity} of $G$ at $t$ squeezes both the
$\liminf$ and the $\limsup$ to $G(t)$.
\end{proof}

\begin{lemma}[Tightness from the characteristic function]
\label{lem:b3:clt:tightness}
For any \omterm{def:b3:probability:space}{random variable} $X$ and $u > 0$:
\[
\P\Bigl(\abs X \geq \frac2u\Bigr) \;\leq\;
\frac1u\int_{-u}^{u}\bigl(1 -
\operatorname{Re}\varphi_X(\xi)\bigr)\,\dd\xi .
\]
\end{lemma}

\begin{proof}
By Tonelli--Fubini (integrand bounded, region finite in
$\xi$):
\[
\frac1u\int_{-u}^u\bigl(1 -
\operatorname{Re}\varphi_X(\xi)\bigr)\dd\xi
= \E\Bigl[\frac1u\int_{-u}^u(1 - \cos(\xi X))\,\dd\xi\Bigr]
= 2\,\E\Bigl[1 - \frac{\sin(uX)}{uX}\Bigr]
\]
(interpret the bracket as its limit $0$ at $X = 0$). The
integrand is nonnegative ($\abs{\sin t} \leq \abs t$), and
for $\abs{uX} \geq 2$: $1 - \frac{\sin(uX)}{uX} \geq 1 -
\frac1{\abs{uX}} \geq \frac12$. Keeping only the event
$\{\abs{uX} \geq 2\}$ inside the \omterm{def:b3:probability:space}{expectation} therefore
leaves at least $2 \cdot \frac12\,\P(\abs X \geq \frac2u)$,
which is the claim.
\end{proof}

\begin{theorem}[L\'evy's continuity theorem]
\label{thm:b3:clt:levy}
Let $(X_n)$ be \omterm{def:b3:probability:space}{random variables} whose \omterm{def:b3:clt:cf}{characteristic
functions} converge pointwise: $\varphi_{X_n}(\xi) \to
\varphi(\xi)$ for every $\xi$, where $\varphi =
\varphi_X$ is the \omterm{def:b3:clt:cf}{characteristic function} of some \omterm{def:b3:probability:space}{random
variable} $X$. Then $X_n \Rightarrow X$.\index{L\'evy's
continuity theorem}
\end{theorem}

\begin{proof}
\emph{Tightness.} Fix $\varepsilon > 0$. Since $\varphi$ is
\omterm{def:b3:topology:continuity}{continuous} at $0$ with $\varphi(0) = 1$, choose $u > 0$ with
$\frac1u\int_{-u}^u(1 - \operatorname{Re}\varphi) <
\varepsilon$; by dominated convergence (integrand bounded by
$2$ on the fixed $[-u,u]$), the same integral for
$\varphi_{X_n}$ is $< 2\varepsilon$ for $n$ large:
\cref{lem:b3:clt:tightness} gives $\P(\abs{X_n} \geq \frac2u)
\leq 2\varepsilon$ for large $n$, and enlarging the constant
handles the finitely many others: the laws are \emph{tight}
--- no mass escapes.

\emph{Subsequences.} Let $(F_{n_k})$ be any subsequence; by
Helly (\cref{thm:b3:clt:helly}) extract $F_{n_{k_j}} \to G$
at \omterm{def:b3:topology:continuity}{continuity} points. Tightness forces $G(-\infty) = 0$,
$G(+\infty) = 1$ ($G(\frac2u) - G(-\frac2u) \geq 1 -
2\varepsilon$ at \omterm{def:b3:topology:continuity}{continuity} points): $G$ is a genuine
distribution function, of some \omterm{def:b3:probability:space}{random variable} $Y$. Then
$X_{n_{k_j}} \Rightarrow Y$ (\cref{exo:b3:clt:4},
distributional convergence from $F$'s), so
$\varphi_{X_{n_{k_j}}} \to \varphi_Y$ \emph{pointwise} ($x
\mapsto \eu^{\iu\xi x}$ is bounded \omterm{def:b3:topology:continuity}{continuous}, real and
imaginary parts separately); comparing with the hypothesis:
$\varphi_Y = \varphi = \varphi_X$, and injectivity
(\cref{thm:b3:clt:injectivity}) gives $Y \sim X$, i.e.\ $G =
F_X$.

\emph{Conclusion.} Every subsequence of $(F_n)$ has a
sub-subsequence converging to the \emph{same} $F_X$ (at its
\omterm{def:b3:topology:continuity}{continuity} points); hence $F_n(t) \to F_X(t)$ at every
\omterm{def:b3:topology:continuity}{continuity} point $t$ (a real sequence all of whose
subsequences have subsubsequences with the same limit
converges): $X_n \Rightarrow X$.
\end{proof}

\section{The central limit theorem}

\begin{theorem}[Central limit theorem]\label{thm:b3:clt:clt}
Let $(X_n)$ be i.i.d.\ with $\E X_1 = m$ and $\V(X_1) =
\sigma^2 \in \intoo0\infty$. Then\index{central limit
theorem}
\[
\frac{S_n - nm}{\sigma\sqrt n} \;\Longrightarrow\; \mathcal
N(0, 1) :
\qquad
\P\Bigl(a \leq \frac{S_n - nm}{\sigma\sqrt n} \leq
b\Bigr) \longrightarrow
\frac{1}{\sqrt{2\pi}}\int_a^b\eu^{-x^2/2}\,\dd x
\]
for all $a < b$.
\end{theorem}

\begin{proof}
\omterm{ex:b3:groups:actions}{Center} and normalize: $Z_i = \frac{X_i - m}{\sigma}$
(i.i.d., mean $0$, variance $1$) and $T_n =
\frac1{\sqrt n}\sum_{i\leq n}Z_i$. By \omterm{def:b3:probability:independence}{independence} and the
affine rule (\cref{prop:b3:clt:cfbasics}):
\[
\varphi_{T_n}(\xi) =
\varphi_{Z}\Bigl(\frac{\xi}{\sqrt n}\Bigr)^{n},
\qquad
\varphi_Z(\eta) = 1 - \frac{\eta^2}2 + \eta^2\rho(\eta),\quad
\rho(\eta)\to0 .
\]
Fix $\xi$ and let $a_n = \varphi_Z(\xi/\sqrt n)$, $b_n = 1 -
\frac{\xi^2}{2n}$: both have modulus $\leq 1$ for $n$ large
($\abs{b_n} \leq 1$ once $\xi^2 \leq 4n$; $\abs{a_n} \leq 1$
always). The elementary inequality $\abs{a^n - b^n} \leq
n\abs{a - b}$ for $\abs a, \abs b \leq 1$ (telescoping
$a^n - b^n = \sum a^k(a - b)b^{n-1-k}$) gives
\[
\Bigl|\varphi_{T_n}(\xi) - \Bigl(1 -
\frac{\xi^2}{2n}\Bigr)^{n}\Bigr|
\leq n\,\Bigl|\varphi_Z\Bigl(\frac\xi{\sqrt n}\Bigr) - 1 +
\frac{\xi^2}{2n}\Bigr|
= \xi^2\,\Bigl|\rho\Bigl(\frac{\xi}{\sqrt n}\Bigr)\Bigr|
\longrightarrow 0,
\]
while $\bigl(1 - \frac{\xi^2}{2n}\bigr)^n \to
\eu^{-\xi^2/2}$ (real logarithm). So $\varphi_{T_n}(\xi) \to
\eu^{-\xi^2/2} = \varphi_{\mathcal N(0,1)}(\xi)$
(\cref{prop:b3:clt:cfbasics}(d)) for every $\xi$: L\'evy
(\cref{thm:b3:clt:levy}) concludes $T_n \Rightarrow \mathcal
N(0,1)$. The interval probabilities follow since $F_{\mathcal
N}$ is \omterm{def:b3:topology:continuity}{continuous} everywhere.
\end{proof}

\begin{example}[Confidence intervals, honestly derived]
\label{ex:b3:clt:confidence}
Poll $n$ \omterm{def:b3:probability:independence}{independent} voters; $\hat p_n = S_n/n$ estimates the
true $p$, with $\sigma^2 = p(1-p) \leq \frac14$. The CLT
gives, for large $n$,
\[
\P\Bigl(\abs{\hat p_n - p} \leq
\frac{z}{2\sqrt n}\Bigr)
\;\geq\; \P\Bigl(\Bigl|\frac{S_n - np}{\sigma\sqrt n}\Bigr|
\leq z\Bigr)
\longrightarrow \Phi(z) - \Phi(-z),
\]
where $\Phi$ is the standard \omterm{def:b3:clt:gaussianvector}{Gaussian} distribution function.
With $z = 1.96$: asymptotic confidence $95\%$, and margin
$\frac{1.96}{2\sqrt n} \leq 3\%$ requires $n \geq
\bigl(\frac{1.96}{0.06}\bigr)^2 \approx 1068$ --- the number
behind every ``$\pm3$ points, $95\%$'' one reads; compare
Chebyshev's $5556$ (\cref{exo:b3:probability:7}). The
$\sqrt n$ is universal: to halve the error, quadruple the
sample --- the same law that fixes \omterm{ex:b3:probability:sllnapps}{Monte Carlo}'s cost
(\cref{exo:b3:clt:7}).
\end{example}

\section{Gaussian vectors}

\begin{definition}\label{def:b3:clt:gaussianvector}
A random vector $X = (X_1, \dots, X_d)$ is
\emph{Gaussian}\index{Gaussian vector} if every linear
combination $\langle t, X\rangle = \sum t_iX_i$ is a
(possibly degenerate) real Gaussian variable. Its law is
determined by the mean vector $m = (\E X_i)$ and the
\emph{covariance matrix} $\Sigma = \bigl(\operatorname{Cov}
(X_i, X_j)\bigr)$: indeed the \omterm{def:b3:clt:cf}{characteristic function} of the
vector, $\varphi_X(t) = \E\eu^{\iu\langle t, X\rangle}$, is
the value at $1$ of the cf of $\langle t, X\rangle$:
\[
\varphi_X(t) = \exp\Bigl(\iu\langle t, m\rangle -
\tfrac12\,t^{\mathsf T}\Sigma\,t\Bigr),
\]
and $d$-dimensional \omterm{def:b3:clt:cf}{characteristic functions} are injective
(same smoothing proof as \cref{thm:b3:clt:injectivity},
coordinatewise Gaussians).
\end{definition}

\begin{theorem}\label{thm:b3:clt:gaussianvector}
Let $X$ be a \omterm{def:b3:clt:gaussianvector}{Gaussian vector}.
\begin{enumerate}
  \item Every affine image $AX + b$ is a \omterm{def:b3:clt:gaussianvector}{Gaussian vector}.
  \item The components $X_i$ are \emph{\omterm{def:b3:probability:independence}{independent}} if and
        only if $\Sigma$ is diagonal: for jointly \omterm{def:b3:clt:gaussianvector}{Gaussian}
        variables, uncorrelated $=$ \omterm{def:b3:probability:independence}{independent}.
  \item If $\Sigma$ is invertible, $X$ has the \omterm{ex:b3:lebesgue:gamma}{density}
        $\frac{1}{(2\pi)^{d/2}\sqrt{\det\Sigma}}
        \exp\bigl(-\frac12(x - m)^{\mathsf T}\Sigma^{-1}(x -
        m)\bigr)$.
\end{enumerate}
\end{theorem}

\begin{proof}
(1) Linear combinations of components of $AX + b$ are affine
functions of linear combinations of $X$: \omterm{def:b3:clt:gaussianvector}{Gaussian} (an affine
image of a \omterm{def:b3:clt:gaussianvector}{Gaussian} variable is \omterm{def:b3:clt:gaussianvector}{Gaussian}).
(2) If $\Sigma$ is diagonal, the \omterm{def:b3:clt:cf}{characteristic function}
factorizes: $\varphi_X(t) = \prod_i\exp(\iu t_im_i -
\frac12\Sigma_{ii}t_i^2) = \prod\varphi_{X_i}(t_i)$, which is
the \omterm{def:b3:clt:cf}{characteristic function} of the product law
(\cref{thm:b3:probability:independence} read through
$d$-dimensional injectivity): the components are \omterm{def:b3:probability:independence}{independent}.
The converse is the vanishing of covariances of \omterm{def:b3:probability:independence}{independent}
$L^2$ variables.
(3) Diagonalize $\Sigma = P D P^{\mathsf T}$ ($P$ orthogonal,
$D > 0$ diagonal --- \cref{exo:b3:submanifolds:8}); the
vector $Y = P^{\mathsf T}(X - m)$ is \omterm{def:b3:clt:gaussianvector}{Gaussian} with
covariance $D$: by (2) its components are \omterm{def:b3:probability:independence}{independent}
$\mathcal N(0, d_i)$, so $Y$ has the product \omterm{ex:b3:lebesgue:gamma}{density}; push
forward by the volume-preserving $x = m + PY$
(\cref{thm:b3:product:linearchange}, $\abs{\det P} = 1$) and
rewrite the exponent invariantly.
\end{proof}

\begin{theorem}[Multidimensional CLT]
\label{thm:b3:clt:multiclt}
Let $(X_n)$ be i.i.d.\ \omterm{def:b3:lebesgue:l1}{square-integrable} random
\emph{vectors} of $\R^d$ with mean $m$ and covariance matrix
$\Sigma$. Then $\frac{S_n - nm}{\sqrt n}$ converges in
distribution to the \omterm{def:b3:clt:gaussianvector}{Gaussian vector} $\mathcal N(0,
\Sigma)$.\index{central limit theorem!multidimensional}
\admitted
\end{theorem}

\begin{remark}\label{rem:b3:clt:multiclt}
Almost everything is already in our hands. For each
direction $t \in \R^d$, the real variable $\langle t,
\frac{S_n - nm}{\sqrt n}\rangle$ is a normalized sum of
i.i.d.\ real variables of variance $t^{\mathsf T}\Sigma t$,
so \cref{thm:b3:clt:clt}'s computation gives pointwise
convergence of the $d$-dimensional \omterm{def:b3:clt:cf}{characteristic functions}
to $\eu^{-t^{\mathsf T}\Sigma t/2}$, the \omterm{def:b3:clt:cf}{characteristic
function} of $\mathcal N(0, \Sigma)$
(\cref{def:b3:clt:gaussianvector}). What we have not
re-proved is L\'evy's \omterm{def:b3:topology:continuity}{continuity} theorem \emph{in $\R^d$}:
Helly's selection and the tightness estimate generalize
routinely (coordinatewise), and this Cram\'er--Wold
reduction is carried out honestly in any graduate
probability course; nothing beyond this chapter's methods is
needed.
\end{remark}

\begin{method}\label{met:b3:clt:method}
To identify a limit law: compute \omterm{def:b3:clt:cf}{characteristic functions},
take the pointwise limit, recognize it (\omterm{def:b3:clt:gaussianvector}{Gaussian}
$\eu^{-\sigma^2\xi^2/2}$, Poisson
$\eu^{\lambda(\eu^{\iu\xi}-1)}$, exponential $\frac{\lambda}
{\lambda - \iu\xi}$, \dots) and invoke L\'evy. The three-step
ritual (\omterm{def:b3:probability:independence}{independence} $\to$ product; Taylor at $0$ $\to$
exponential limit; L\'evy $\to$ convergence in law) proves
the CLT, the Poisson law of rare events
(\cref{exo:b3:clt:5}), and every classical limit theorem of
this course. For a.s.\ statements, go back to
\cref{ch:b3:probability}'s toolkit: the two chapters answer
different questions about the same $S_n$.
\end{method}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:b3:clt:1}
Compute the \omterm{def:b3:clt:cf}{characteristic functions}: uniform on
$\intcc{-1}1$; exponential $\mathcal E(\lambda)$; Poisson
$\mathcal P(\lambda)$; binomial $\mathcal B(n, p)$. Deduce
via \cref{thm:b3:clt:injectivity} that the sum of \omterm{def:b3:probability:independence}{independent}
Poisson variables ($\lambda, \mu$) is Poisson $(\lambda +
\mu)$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:clt:2}
(a) Show that $\varphi_X$ is real-valued if and only if $X$
and $-X$ have the same law (a \emph{symmetric} variable).
(b) Suppose $\abs{\varphi_X(\xi_0)} = 1$ for some $\xi_0 \neq
0$. Show that $X$ is almost surely supported on an arithmetic
progression $a + \frac{2\pi}{\xi_0}\Z$ \emph{(write
$\varphi_X(\xi_0) = \eu^{\iu\theta}$ and compute $\E[1 -
\cos(\xi_0X - \theta)]$)}. Deduce that if $X$ has a \omterm{ex:b3:lebesgue:gamma}{density},
then $\abs{\varphi_X(\xi)} < 1$ for all $\xi \neq 0$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:clt:3}
Let $X \sim \mathcal N(m_1, \sigma_1^2)$ and $Y \sim

\begin{exercise}[$\star\star$]\label{exo:b3:clt:12}

\end{exercise}

\section{Problem: Lindeberg's proof of the CLT, with a
rate}

\begin{problem}[{Weekend problem --- the replacement method}]
\label{pb:b3:clt:1}
Lindeberg (1922) proved the central limit theorem by an idea
of disarming simplicity: \emph{swap the summands one at a
time for \omterm{def:b3:clt:gaussianvector}{Gaussians}} and control each swap by a Taylor
expansion. The method needs no Fourier analysis, produces an
explicit error rate, and today powers universality proofs
across probability theory. Let $(X_i)$ be i.i.d., centered,
$\V(X_1) = 1$, with $\beta = \E\abs{X_1}^3 < \infty$; let
$(N_i)$ be i.i.d.\ $\mathcal N(0,1)$, \omterm{def:b3:probability:independence}{independent} of the
$X_i$ (existence: \cref{thm:b3:probability:existence}). Set
\[
T_n = \frac{X_1 + \dots + X_n}{\sqrt n},
\qquad
G_n = \frac{N_1 + \dots + N_n}{\sqrt n} \sim \mathcal N(0,1).
\]

\medskip
\noindent\textbf{Part I --- The swapping identity.} Fix $f
\in \mathcal C^3_b(\R)$ (three bounded \omterm{def:b3:topology:continuity}{continuous}
derivatives; $M_3 = \sup\abs{f'''}$). For $0 \leq i \leq n$
define the hybrid sums
\[
H_i = \frac{X_1 + \dots + X_i + N_{i+1} + \dots +
N_n}{\sqrt n},
\]
so $H_n = T_n$ and $H_0 = G_n$.
\begin{enumerate}
  \item Write $H_i = W_i + \frac{X_i}{\sqrt n}$ and $H_{i-1}
        = W_i + \frac{N_i}{\sqrt n}$ with $W_i =
        \frac{1}{\sqrt n}\bigl(\sum_{j<i}X_j +
        \sum_{j>i}N_j\bigr)$, and note that $W_i$ is
        \omterm{def:b3:probability:independence}{independent} of the pair $(X_i, N_i)$. Justify.
  \item Taylor with integral or Lagrange remainder: for any
        real $w, h$:
        \[
        \Bigl|f(w + h) - f(w) - f'(w)h -
        \tfrac12f''(w)h^2\Bigr| \leq
        \frac{M_3\,\abs h^3}{6} .
        \]
  \item Apply question 2 twice ($h = \frac{X_i}{\sqrt n}$ and
        $h = \frac{N_i}{\sqrt n}$ at $w = W_i$), take
        \omterm{def:b3:probability:space}{expectations}, and use \omterm{def:b3:probability:independence}{independence} plus the matching
        of the first two moments of $X_i$ and $N_i$ to show
        \[
        \bigl|\E f(H_i) - \E f(H_{i-1})\bigr|
        \leq \frac{M_3}{6}\cdot
        \frac{\beta + \gamma}{n^{3/2}},
        \qquad \gamma = \E\abs{N_1}^3 =
        \frac{2\sqrt2}{\sqrt\pi} .
        \]
  \item Telescope over $i$ and conclude the \emph{Lindeberg
        bound}:
        \[
        \bigl|\E f(T_n) - \E f(G_n)\bigr| \leq
        \frac{M_3\,(\beta + \gamma)}{6\,\sqrt n} .
        \]
\end{enumerate}

\medskip
\noindent\textbf{Part II --- From smooth $f$ to the CLT.}
\begin{enumerate}[resume]
  \item Show that $\E f(T_n) \to \E f(N)$ for every $f \in
        \mathcal C_b^3$, and upgrade to all bounded
        \omterm{def:b3:topology:continuity}{continuous} $f$: given such $f$ and $\varepsilon$,
        construct $f_\varepsilon \in \mathcal C^3_b$ with
        $\norm{f - f_\varepsilon}_\infty \leq \varepsilon$
        on a large interval --- e.g.\ convolve $f$ with a
        $\mathcal C^\infty$ bump
        (\cref{thm:b3:lp:regularization}) --- and handle the
        tails by tightness ($\V(T_n) = 1$ and Chebyshev).
        Conclude $T_n \Rightarrow \mathcal N(0, 1)$: the
        central limit theorem, re-proved.
  \item Where did the proof use that the $X_i$ are
        \emph{identically} distributed? Show that it barely
        did: state and prove the version for \omterm{def:b3:probability:independence}{independent},
        centered, non-identical $X_i$ with
        $\sum_i\V(X_i) = s_n^2$ and third moments, obtaining
        the error $\frac{M_3}{6s_n^3}\sum_i\bigl(\E\abs{X_i}^3
        + \V(X_i)^{3/2}\gamma\bigr)$ --- Lindeberg's true
        theorem in its Lyapunov form.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Quantitative dividends.}
\begin{enumerate}[resume]
  \item (Distribution functions) Let $t \in \R$ and
        approximate $\mathbf 1_{\intoc{-\infty}t}$ above and
        below by $\mathcal C^3_b$ ramps of width $\delta$
        (construct them, with $M_3 = O(\delta^{-3})$).
        Combining with Part I, derive the two-term bound
        \[
        \sup_{t\in\R}\,\bigl|\P(T_n \leq t) -
        \Phi(t)\bigr| \;\leq\;
        \frac{C_1(\beta + \gamma)}{\delta^3\sqrt n} +
        C_2\,\delta
        \qquad (\text{every } \delta > 0),
        \]
        with explicit constants (the $C_2\delta$ term uses
        that $\Phi$ has \omterm{ex:b3:lebesgue:gamma}{density} bounded by
        $\frac1{\sqrt{2\pi}}$), and optimize $\delta \sim
        n^{-1/8}$ to obtain a uniform rate of order
        $n^{-1/8}$.
        (The optimal $n^{-1/2}$ --- Berry--Esseen ---
        needs finer tools; the point is an \emph{explicit}
        rate from elementary swapping.)
  \item (De Moivre--Laplace, quantified) Specialize to
        $X_i = 2B_i - 1$ (signs of fair coins): compare the
        conclusion with the local estimate of
        \cref{pb:b3:product:1}, question 7 --- what does
        each method give that the other does not?
  \item (Universality) Explain in a paragraph why the
        replacement method shows more than the CLT: any
        statistic of the form $\E f(\text{sum})$ with $f$
        smooth is insensitive, at order $n^{-1/2}$, to the
        \emph{entire law} of the summands beyond its first
        two moments --- the ``invariance principle'' that
        underlies modern universality results (random
        matrices, random polynomials), of which the CLT is
        the first instance.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Smoothing, pushed: better
rates.} The loss from $n^{-1/2}$ (smooth $f$) to
$n^{-1/8}$ (distribution functions) came from charging
$f'''$ in sup norm. The hybrids can repair part of it: they
contain \omterm{def:b3:clt:gaussianvector}{Gaussian} summands, and \omterm{def:b3:clt:gaussianvector}{Gaussians} \emph{smooth}.
\begin{enumerate}[resume]
  \item (A hidden \omterm{def:b3:clt:gaussianvector}{Gaussian}) For $1 \leq i \leq n - 1$, $h =
        \frac{X_i}{\sqrt n}$ or $\frac{N_i}{\sqrt n}$, and
        $\theta \in \intcc01$, write $W_i + \theta h = A +
        Z$ with $Z = \frac{N_{i+1} + \dots + N_n}{\sqrt
        n}$. Show that $Z \sim \mathcal N\bigl(0,
        \frac{n-i}n\bigr)$ is \omterm{def:b3:probability:independence}{independent} of the pair $(A,
        h)$, and deduce, for every \omterm{def:b3:topology:continuity}{continuous} $g \in
        L^1(\R)$,
        \[
        \E\bigl[\abs h^3\,\abs{g(W_i + \theta h)}\bigr]
        \;\leq\; \sqrt{\frac{n}{2\pi(n - i)}}\;
        \norm{g}_{L^1}\;\E\abs h^3 .
        \]
  \item Combine question 10 with the integral form of the
        Taylor remainder,
        \[
        f(w + h) = f(w) + f'(w)h + \tfrac12f''(w)h^2 +
        \int_0^1\frac{(1 - \theta)^2}2\,f'''(w + \theta
        h)\,h^3\,\dd\theta,
        \]
        to redo questions 3--4: for $f \in \mathcal C^3_b$
        with moreover $f''' \in L^1(\R)$,
        \[
        \bigl|\E f(T_n) - \E f(G_n)\bigr| \leq
        \frac{\beta + \gamma}{3\sqrt{2\pi}}\cdot
        \frac{\norm{f'''}_{L^1}}{\sqrt n} +
        \frac{M_3(\beta + \gamma)}{6\,n^{3/2}}
        \]
        \emph{(question 10 handles the swaps $i \leq n - 1$
        --- use $\sum_{m=1}^{n-1}m^{-1/2} \leq 2\sqrt n$
        --- and question 3's crude bound handles the last
        one)}. Check that question 7's ramps satisfy
        $\norm{\psi_\delta'''}_{L^1} = K_1\delta^{-2}$
        while $M_3 = K\delta^{-3}$, feed them in, and
        optimize $\delta$: the uniform
        distribution-function rate improves to
        $O(n^{-1/6})$.
  \item (Matching one more moment) Assume in addition $\E
        X_1^3 = 0$ and $\beta_4 = \E X_1^4 < \infty$.
        Compute $\E N_1^3$ and $\E N_1^4$, expand to fourth
        order, and prove along the same lines that the
        distribution-function rate becomes $O(n^{-1/4})$
        \emph{(now $\norm{\psi_\delta^{(4)}}_{L^1} =
        K_2\delta^{-3}$ and $M_4 = K'\delta^{-4}$; choose
        $\delta = n^{-1/4}$)}.
  \item (The obstruction) Suppose the first $k$ moments of
        $X_1$ agree with the \omterm{def:b3:clt:gaussianvector}{Gaussian} ones ($k = 2$ always;
        $k = 3$ exactly when $\E X_1^3 = 0$; $k \geq 4$
        essentially never, as $\E N_1^4 = 3$). Verify that
        the scheme of questions 10--12 delivers the
        distribution-function rate
        $n^{-(k-1)/(2k+2)}$, by balancing
        $\delta^{-k}n^{-(k-1)/2}$ against $\delta$, and
        observe that the exponent approaches the
        Berry--Esseen value $\frac12$ only as $k \to
        \infty$. Explain in a few sentences why the
        swapping method saturates: each swap is charged in
        absolute value, whereas the Fourier route
        (Esseen's smoothing inequality) exploits the
        oscillation of the characteristic-function
        difference and reaches $C\beta n^{-1/2}$ with three
        moments only.
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Two dimensions: the
multidimensional CLT, by swapping.} Now let the $X_i$ be
i.i.d.\ centered random \emph{vectors} of $\R^2$ with
covariance matrix $\Sigma$ and $\beta' = \E\norm{X_1}^3 <
\infty$ (Euclidean norm).
\begin{enumerate}[resume]
  \item (\omterm{def:b3:clt:gaussianvector}{Gaussian vectors}, to order) Diagonalize $\Sigma =
        PDP^{\mathsf T}$ (\cref{exo:b3:submanifolds:8}) and
        set $C = P\sqrt DP^{\mathsf T}$. For $Z = (Z^1,
        Z^2)$ a pair of \omterm{def:b3:probability:independence}{independent} standard \omterm{def:b3:clt:gaussianvector}{Gaussians}
        (\cref{thm:b3:probability:existence}), show that $N
        = CZ$ is a \omterm{def:b3:clt:gaussianvector}{Gaussian vector}
        (\cref{def:b3:clt:gaussianvector}) of mean $0$,
        covariance $\Sigma$, with $\gamma' = \E\norm N^3 <
        \infty$; and that $G_n = \frac{N_1 + \dots +
        N_n}{\sqrt n}$ has law $\mathcal N(0, \Sigma)$
        \emph{exactly} for i.i.d.\ copies $N_i$.
  \item (Taylor in two variables) For $f \colon \R^2 \to
        \R$ of class $\mathcal C^3$ with $M_3 =
        \max_{\abs\alpha = 3}\sup\abs{\partial^\alpha f} <
        \infty$, prove
        \[
        \Bigl|f(w + h) - f(w) - \langle\nabla f(w),
        h\rangle - \tfrac12\langle h, D^2f(w)\,h\rangle
        \Bigr| \leq \frac{M_3}6\,\bigl(\abs{h_1} +
        \abs{h_2}\bigr)^3 \leq \frac{\sqrt2\,M_3}3\,
        \norm h^3
        \]
        \emph{(study $t \mapsto f(w + th)$ on
        $\intcc01$)}.
  \item (The CLT in $\R^2$) Run the replacement scheme on
        the vector hybrids $H_i$: show that the first- and
        second-order terms cancel (means and covariances
        match), telescope, and upgrade as in question 5
        (tightness from $\E\norm{T_n}^2 =
        \operatorname{tr}\Sigma$; mollification now in
        $\R^2$, \cref{thm:b3:lp:regularization}) to
        conclude: for every bounded \omterm{def:b3:topology:continuity}{continuous} $f \colon
        \R^2 \to \R$,
        \[
        \E\,f\Bigl(\frac{X_1 + \dots + X_n}{\sqrt n}\Bigr)
        \longrightarrow \E\,f(N), \qquad N \sim \mathcal
        N(0, \Sigma) :
        \]
        \cref{thm:b3:clt:multiclt} in dimension $2$, with a
        rate for smooth $f$ and no Fourier analysis.
  \item (Cram\'er--Wold, and a joint fluctuation) Deduce
        that $\langle t, \frac{S_n}{\sqrt n}\rangle
        \Rightarrow \mathcal N(0, t^{\mathsf T}\Sigma t)$
        for every fixed $t \in \R^2$. Application: for
        i.i.d.\ real $(\xi_i)$, centered, $\E\xi_1^2 = 1$,
        $\E\xi_1^6 < \infty$ (so that Part V applies to
        $V_i = (\xi_i, \xi_i^2 - 1)$), show
        \[
        \frac1{\sqrt n}\Bigl(\sum_{i\leq n}\xi_i,\
        \sum_{i\leq n}(\xi_i^2 - 1)\Bigr) \Longrightarrow
        \mathcal N\Bigl(0, \begin{pmatrix} 1 & \E\xi_1^3\\
        \E\xi_1^3 & \E\xi_1^4 - 1\end{pmatrix}\Bigr) :
        \]
        empirical mean and empirical second moment
        fluctuate jointly Gaussianly --- independently in
        the limit if and only if $\E\xi_1^3 = 0$
        (\cref{thm:b3:clt:gaussianvector}).
\end{enumerate}

\medskip
\noindent\textbf{Part VI --- The delta method.}
\begin{enumerate}[resume]
  \item Let $(\hat\theta_n)$ be \omterm{def:b3:probability:space}{random variables} with
        $\sqrt n(\hat\theta_n - \theta) \Rightarrow
        \mathcal N(0, \sigma^2)$ for a real parameter
        $\theta$, and let $g$ be differentiable at
        $\theta$. Prove the \emph{delta method}:
        \[
        \sqrt n\bigl(g(\hat\theta_n) - g(\theta)\bigr)
        \Longrightarrow \mathcal N\bigl(0,
        g'(\theta)^2\sigma^2\bigr)
        \]
        \emph{(write $g(x) - g(\theta) = (g'(\theta) +
        \eta(x))(x - \theta)$ with $\eta \to 0$ at
        $\theta$; show $\hat\theta_n \to \theta$, then
        $\eta(\hat\theta_n) \to 0$, in probability; finish
        with Slutsky, \cref{exo:b3:clt:8}, and
        \cref{exo:b3:clt:4}(b))}.
  \item Applications. (a) For i.i.d.\ real $(\xi_i)$ with
        mean $\mu$ and variance $\sigma^2$, and $\bar X_n =
        \frac1n\sum_{i\leq n}\xi_i$: show $\sqrt n(\bar
        X_n^2 - \mu^2) \Rightarrow \mathcal N(0,
        4\mu^2\sigma^2)$ when $\mu \neq 0$, and that for
        $\mu = 0$ the correct statement lives at another
        scale: $n\bar X_n^2 \Rightarrow \sigma^2N^2$ with
        $N \sim \mathcal N(0,1)$ (identify the limit's
        distribution function). (b) (Variance
        stabilization) For $\hat p_n$ the success frequency
        of a $\mathcal B(1, p)$ sample, $p \in \intoo01$:
        show that $g(p) = \arcsin\sqrt p$ satisfies
        \[
        \sqrt n\,\bigl(g(\hat p_n) - g(p)\bigr)
        \Longrightarrow \mathcal N\Bigl(0, \frac14\Bigr)
        \]
        \emph{whatever} $p$ --- an asymptotic error bar
        free of the unknown parameter; compare with
        \cref{ex:b3:clt:confidence}.
\end{enumerate}

\medskip
\noindent\textbf{Part VII --- Poisson, by the same method:
Le Cam's theorem.} Replacement knows a second universality
class: sums of many \omterm{def:b3:probability:independence}{independent} \emph{rare} events. For
laws on $\N$ the right distance is \emph{total variation},
\[
d_{\mathrm{TV}}(\mu, \nu) = \sup_{A\subseteq\N}\,
\abs{\mu(A) - \nu(A)} .
\]
\begin{enumerate}[resume]
  \item Show that $d_{\mathrm{TV}}(\mu, \nu) =
        \frac12\sum_{k\geq0}\abs{\mu(\{k\}) -
        \nu(\{k\})}$, and prove the coupling bound: for
        \emph{any} pair $(X, Y)$ of \omterm{def:b3:probability:space}{random variables} with
        laws $\mu$ and $\nu$ on the same space,
        $d_{\mathrm{TV}}(\mu, \nu) \leq \P(X \neq Y)$.
  \item Compute exactly, for $p \in \intoo01$:
        \[
        d_{\mathrm{TV}}\bigl(\mathcal B(1, p), \mathcal
        P(p)\bigr) = p\bigl(1 - \eu^{-p}\bigr) \leq p^2 .
        \]
  \item (Le Cam, by swapping) Let $X_i \sim \mathcal B(1,
        p_i)$ and $Y_i \sim \mathcal P(p_i)$, the $2n$
        variables \omterm{def:b3:probability:independence}{independent}; $S = X_1 + \dots + X_n$, and
        recall $Y_1 + \dots + Y_n \sim \mathcal P(\lambda)$
        with $\lambda = \sum_ip_i$ (\cref{exo:b3:clt:1}).
        Swap one coordinate at a time in the integer
        hybrids $H_i = Y_1 + \dots + Y_i + X_{i+1} + \dots
        + X_n$: show, for every $A \subseteq \N$,
        \[
        \abs{\P(H_{i-1} \in A) - \P(H_i \in A)} \leq
        d_{\mathrm{TV}}\bigl(\mathcal B(1, p_i), \mathcal
        P(p_i)\bigr),
        \]
        and conclude \emph{Le Cam's inequality}:
        \[
        d_{\mathrm{TV}}\bigl(\text{law of } S,\ \mathcal
        P(\lambda)\bigr) \leq \sum_{i=1}^np_i^2 .
        \]
  \item Dividends. (a) For $p_i = \frac\lambda n$: the
        bound is $\frac{\lambda^2}n$ --- the law of rare
        events (\cref{exo:b3:clt:5}) upgraded to an
        explicit rate, uniform over all events, and valid
        for unequal $p_i$ as well. (b) $500$ letters are
        delivered, each going astray independently with
        probability $\frac1{500}$: bound the error of the
        Poisson model of parameter $1$, and estimate the
        probability that no letter goes astray. (c) Close
        the problem: compare the two universality classes
        met here --- \omterm{def:b3:clt:gaussianvector}{Gaussian} (many small spread-out
        contributions; two moments matched; Taylor) and
        Poisson (many rare contributions; one mean
        matched; an exact total-variation coupling) --- and
        the single replacement method behind both.

  \item (Relative error and the log transform) Let $(X_n)$
        be i.i.d., positive, mean $\mu > 0$, variance
        $\sigma^2$, and $\bar X_n$ the empirical mean. Show
        by the delta method that
        \[
        \sqrt n\,\bigl(\ln\bar X_n - \ln\mu\bigr)
        \Longrightarrow
        \mathcal N\Bigl(0,\ \frac{\sigma^2}{\mu^2}\Bigr) :
        \]
        the asymptotic parameter of $\ln\bar X_n$ is the
        \emph{coefficient of variation} $\sigma/\mu$ ---
        relative, scale-free error. Deduce a $95\%$
        confidence interval for $\mu$ of the multiplicative
        form $\bar X_n\cdot\eu^{\pm1.96\,\sigma/(\mu\sqrt
        n)}$, and explain when it is preferable to the
        additive one.
  \item (The third moment steers the error) For $X \sim$
        Bernoulli($p$) centered, compute $\E\bigl[(X -
        p)^3\bigr] = p(1-p)(1-2p)$. Using Part IV's analysis
        (the swap error is driven by third moments), explain
        why the normal approximation of $\mathcal B(n, p)$
        is asymmetric for $p \neq \frac12$ --- overshooting
        on one side, undershooting on the other --- and why
        $p = \frac12$ enjoys the faster matched-moment rate.
        Verify the sign of the skew numerically on $\mathcal
        B(20, 0.1)$ against $\mathcal N(2, 1.8)$: compare
        $\P(S = 0) = 0.9^{20}$ with the \omterm{def:b3:clt:gaussianvector}{Gaussian} mass of
        $\intoo{-\infty}{0.5}$.
\end{enumerate}
\end{problem}
