\chapter{Holomorphic Functions}\label{ch:b3:holomorphic}

Complex differentiability looks like a small variation on the
real theory --- one limit, one quotient. It is instead a
different universe. A function differentiable \emph{once} on an
open subset of $\C$ is automatically infinitely differentiable,
analytic, determined on a whole domain by its values near a
single point, and constrained by rigid global principles
(Liouville, maximum modulus). All of this flows from one
miracle, Cauchy's theorem: the integral of a holomorphic
function along a closed path in a star-shaped domain vanishes.
This chapter proves the miracle (Goursat's argument, with no
regularity assumed beyond differentiability), harvests its
classical consequences, and settles a theorem this book has
used on credit since \cref{ch:b3:galois}: every nonconstant
complex polynomial has a root.

Throughout, $\Omega \subseteq \C$ is open, and $D(a, r)$
denotes the open disc.

\section{Complex differentiability}

\begin{definition}\label{def:b3:holomorphic:holo}
$f \colon \Omega \to \C$ is \emph{holomorphic}\index{holomorphic
function} on $\Omega$ if for every $z_0 \in \Omega$
\[
f'(z_0) = \lim_{h\to0}\frac{f(z_0 + h) - f(z_0)}{h}
\]
exists ($h \in \C^*$). Sums, products, quotients (nonvanishing
denominators), compositions of holomorphic functions are
holomorphic, with the usual formulas (the Year 1--2 proofs are
verbatim: they only use field operations and limits).
$\mathcal H(\Omega)$ denotes the set of holomorphic functions
on $\Omega$.
\end{definition}

\begin{proposition}[Cauchy--Riemann]
\label{prop:b3:holomorphic:cauchyriemann}
Write $f(x + \iu y) = P(x,y) + \iu Q(x,y)$. Then $f$ is
holomorphic at $z_0$ iff $f$ is $\R$-differentiable at $z_0$
(as a map of two real variables) and\index{Cauchy--Riemann
equations}
\[
\frac{\partial P}{\partial x} = \frac{\partial Q}{\partial y},
\qquad
\frac{\partial P}{\partial y} = -\frac{\partial Q}{\partial x}
\qquad \text{at } z_0 ;
\]
equivalently, the real differential is multiplication by the
complex number $f'(z_0)$.
\end{proposition}

\begin{proof}
$\C$-differentiability says $f(z_0 + h) = f(z_0) + ch +
o(\abs h)$ with $c = f'(z_0)$: an $\R$-linear differential
which is multiplication by $c = a + \iu b$, i.e.\ has matrix
$\bigl(\begin{smallmatrix} a & -b\\ b & a
\end{smallmatrix}\bigr)$ in the basis $(1, \iu)$ --- exactly
the displayed relations for the partials. Conversely such a
differential is $\C$-linear, and the $o(\abs h)$ definitions
match.
\end{proof}

\begin{example}\label{ex:b3:holomorphic:examples}
Polynomials in $z$, rational functions off their poles, and
--- by Year 2's term-by-term differentiation theorem for power
series, whose proof works identically over $\C$ --- every sum
of a power series $\sum a_n(z - a)^n$ inside its disc of
convergence: holomorphic, with derivative $\sum na_n(z -
a)^{n-1}$ (same radius). In particular $\exp z = \sum z^n/n!$
is entire (holomorphic on $\C$) with $\exp' = \exp$. On the
other hand $z \mapsto \bar z$, $\abs z$, $\operatorname{Re}z$
are nowhere holomorphic (Cauchy--Riemann fails everywhere):
holomorphy is orientation-and-angle-preserving rigidity, not
smoothness.
\end{example}

\section{Contour integrals}

\begin{definition}\label{def:b3:holomorphic:contour}
A \emph{path} is a piecewise $\mathcal C^1$ map $\gamma \colon
\intcc ab \to \C$; it is \emph{closed} if $\gamma(a) =
\gamma(b)$. For continuous $f$ on the image of
$\gamma$:\index{contour integral}
\[
\int_\gamma f(z)\,\dd z =
\int_a^b f(\gamma(t))\,\gamma'(t)\,\dd t,
\qquad
\Bigl|\int_\gamma f\,\dd z\Bigr| \leq
\sup_{\gamma}\abs f\cdot\operatorname{length}(\gamma)
\]
(the \emph{ML inequality}; length $= \int_a^b\abs{\gamma'}$).
The integral is invariant under increasing $\mathcal C^1$
reparametrization and changes sign under orientation reversal.
\end{definition}

\begin{proposition}[Primitives]\label{prop:b3:holomorphic:primitive}
For continuous $f$ on $\Omega$, the following are equivalent:
(i) $f$ has a \emph{primitive} $F \in \mathcal H(\Omega)$ ($F'
= f$); (ii) $\int_\gamma f\,\dd z = 0$ for every closed path
$\gamma$ in $\Omega$. In that case $\int_\gamma f\,\dd z =
F(\gamma(b)) - F(\gamma(a))$ for every path.
\end{proposition}

\begin{proof}
(i)$\Rightarrow$: $\frac{\dd}{\dd t}F(\gamma(t)) =
F'(\gamma(t))\gamma'(t)$ (chain rule, valid piecewise), so the
integral telescopes to the endpoint difference; closed paths
give $0$. (ii)$\Rightarrow$(i): fix $z_*$ in a connected
component, define $F(z) = \int_{\gamma_z}f\,\dd z$ along any
path from $z_*$ to $z$ (well defined: two paths differ by a
closed path); for $h$ small, taking the segment from $z$ to
$z + h$,
\[
\frac{F(z + h) - F(z)}{h} - f(z)
= \frac1h\int_{[z, z+h]}\bigl(f(w) - f(z)\bigr)\dd w
\longrightarrow 0
\]
by the ML inequality and continuity of $f$ at $z$.
\end{proof}

\begin{definition}[Winding number]\label{def:b3:holomorphic:index}
For a closed path $\gamma$ and $z \notin
\operatorname{im}\gamma$, the \emph{index}\index{winding
number}\index{index of a path} is
\[
\operatorname{Ind}_\gamma(z) = \frac1{2\iu\pi}
\int_\gamma\frac{\dd w}{w - z} .
\]
It is an \emph{integer}: setting $\varphi(t) =
\int_a^t\frac{\gamma'(s)}{\gamma(s) - z}\dd s$, the function
$(\gamma(t) - z)\eu^{-\varphi(t)}$ has zero derivative
(piecewise), hence is constant; at $t = b$, $\eu^{\varphi(b)} =
\frac{\gamma(b) - z}{\gamma(a) - z} = 1$, so $\varphi(b) \in
2\iu\pi\Z$. As a function of $z$, the index is continuous on
$\C\setminus\operatorname{im}\gamma$ (dominated convergence),
hence constant on each connected component, and $0$ on the
unbounded component (ML: the integral tends to $0$ as $z \to
\infty$). For the circle $\gamma(t) = a + r\eu^{\iu t}$, $t
\in \intcc0{2\pi}$: $\operatorname{Ind}_\gamma(z) = 1$ for $z
\in D(a,r)$ (compute at $z = a$: $\frac1{2\iu\pi}
\int_0^{2\pi}\frac{r\iu\eu^{\iu t}}{r\eu^{\iu t}}\dd t = 1$;
constancy does the rest).
\end{definition}

\section{Cauchy's theorem}

\begin{theorem}[Goursat]\label{thm:b3:holomorphic:goursat}
Let $f \in \mathcal H(\Omega)$ and $T \subseteq \Omega$ a
closed solid triangle. Then $\int_{\partial T}f\,\dd z = 0$
(boundary run once, any orientation).\index{Goursat's theorem}
\end{theorem}

\begin{proof}
Let $I(T) = \int_{\partial T}f\,\dd z$. Joining the midpoints
of the sides splits $T$ into four half-size triangles $T^{(1)},
\dots, T^{(4)}$, and the inner edges cancel in pairs: $I(T) =
\sum_iI(T^{(i)})$. Choose $T_1$ among them with
$\abs{I(T_1)} \geq \frac14\abs{I(T)}$, and iterate: a nested
sequence $T \supseteq T_1 \supseteq T_2 \supseteq\cdots$ with
\[
\abs{I(T_n)} \geq 4^{-n}\abs{I(T)},
\qquad
\operatorname{diam}T_n = 2^{-n}\operatorname{diam}T,
\quad
\operatorname{length}(\partial T_n) =
2^{-n}\operatorname{length}(\partial T).
\]
The intersection $\bigcap T_n$ is a single point $z_0$
(nested compacts with vanishing diameters,
\cref{thm:b3:topology:compactprops}(3)). Differentiability at
$z_0$: given $\varepsilon$, for $n$ large, on $T_n$,
\[
f(z) = f(z_0) + f'(z_0)(z - z_0) + R(z),
\qquad \abs{R(z)} \leq \varepsilon\abs{z - z_0} \leq
\varepsilon\operatorname{diam}T_n .
\]
The affine part has a primitive: its integral on the closed
$\partial T_n$ vanishes (\cref{prop:b3:holomorphic:primitive}),
leaving
\[
\abs{I(T_n)} = \Bigl|\int_{\partial T_n}R\Bigr|
\leq \varepsilon\operatorname{diam}(T_n)\,
\operatorname{length}(\partial T_n)
= \varepsilon\,4^{-n}\operatorname{diam}(T)
\operatorname{length}(\partial T) .
\]
Comparing with $\abs{I(T_n)} \geq 4^{-n}\abs{I(T)}$:
$\abs{I(T)} \leq \varepsilon\cdot\text{const}$ for every
$\varepsilon$: $I(T) = 0$.
\end{proof}

\begin{theorem}[Cauchy's theorem, star-shaped version]
\label{thm:b3:holomorphic:cauchy}
Let $\Omega$ be \emph{star-shaped} about $c$ (every segment
$[c, z]$, $z \in \Omega$, lies in $\Omega$) --- e.g.\ convex.
Every $f \in \mathcal H(\Omega)$ has a primitive on $\Omega$;
consequently $\int_\gamma f\,\dd z = 0$ for \emph{every}
closed path $\gamma$ in $\Omega$.\index{Cauchy's theorem}
\end{theorem}

\begin{proof}
Define $F(z) = \int_{[c,z]}f\,\dd w$. For $z, z + h \in
\Omega$ with $[z, z+h] \subseteq \Omega$ (true for small $h$),
the triangle with vertices $c, z, z+h$ lies in $\Omega$
(star-shapedness: each of its points lies on a segment $[c,
w]$ with $w \in [z, z+h] \subseteq \Omega$): Goursat gives
\[
F(z + h) - F(z) = \int_{[z, z+h]}f\,\dd w,
\]
and the difference-quotient computation of
\cref{prop:b3:holomorphic:primitive} yields $F' = f$. The
vanishing of closed-path integrals follows from the same
proposition.
\end{proof}

\begin{theorem}[Cauchy's integral formula]
\label{thm:b3:holomorphic:formula}
Let $f \in \mathcal H(\Omega)$, $\bar D(a, r) \subseteq
\Omega$, and $C_r$ the circle $\partial D(a,r)$ run once
counterclockwise. Then for every $z \in D(a, r)$:\index{Cauchy
integral formula}
\[
f(z) = \frac1{2\iu\pi}\int_{C_r}\frac{f(w)}{w - z}\,\dd w .
\]
\end{theorem}

\begin{proof}
Fix $z$ and define on $\Omega$
\[
g(w) = \begin{cases}
\dfrac{f(w) - f(z)}{w - z} & w \neq z,\\[4pt]
f'(z) & w = z :
\end{cases}
\]
$g$ is continuous on $\Omega$ and holomorphic off $z$. Goursat
holds for $g$ on every triangle $T \subseteq \Omega'$, where
$\Omega'$ is a disc slightly larger than $\bar D(a,r)$ inside
$\Omega$: if $z \notin T$, directly; if $z \in T$, split $T$
into small triangles having $z$ as a vertex plus triangles
avoiding $z$; on a triangle with vertex $z$ the ML bound gives
$\abs{\int_{\partial T'}g} \leq \sup_{T'}\abs
g\cdot\operatorname{length} \to 0$ as the triangle shrinks,
and the remaining pieces vanish by Goursat --- hence
$\int_{\partial T}g = 0$ in all cases. The proof of
\cref{thm:b3:holomorphic:cauchy} used only this triangle
property: $g$ has a primitive on the convex $\Omega'$, so
$\int_{C_r}g = 0$, i.e.
\[
\frac1{2\iu\pi}\int_{C_r}\frac{f(w)}{w - z}\dd w
= f(z)\,\frac{1}{2\iu\pi}\int_{C_r}\frac{\dd w}{w - z}
= f(z)\operatorname{Ind}_{C_r}(z) = f(z) .
\]
\end{proof}

\section{Analyticity and its cascade}

\begin{theorem}[Holomorphic $=$ analytic]
\label{thm:b3:holomorphic:analytic}
Let $f \in \mathcal H(\Omega)$ and $D(a, R) \subseteq \Omega$.
Then\index{analyticity of holomorphic functions}
\[
f(z) = \sum_{n\geq0}c_n\,(z - a)^n \quad \text{on } D(a, R),
\qquad
c_n = \frac1{2\iu\pi}\int_{C_r}\frac{f(w)}{(w -
a)^{n+1}}\,\dd w \ \ (0 < r < R),
\]
the coefficients being independent of $r$. Consequently $f$ is
infinitely $\C$-differentiable, $c_n = f^{(n)}(a)/n!$, and the
\emph{Cauchy estimates}\index{Cauchy estimates} hold:
\[
\abs{c_n} \;\leq\; \frac{\sup_{\abs{w - a} =
r}\abs{f(w)}}{r^{n}} .
\]
\end{theorem}

\begin{proof}
For $\abs{z - a} < r$: expand the Cauchy kernel in the
geometric series
\[
\frac1{w - z} = \frac1{(w - a)\bigl(1 - \frac{z - a}{w -
a}\bigr)} = \sum_{n\geq0}\frac{(z - a)^n}{(w - a)^{n+1}},
\]
normally convergent in $w$ on $C_r$ ($\abs{\frac{z-a}{w-a}} =
\frac{\abs{z-a}}r < 1$): integrate term by term against
$\frac{f(w)}{2\iu\pi}$ (uniform convergence justifies the
interchange) and apply \cref{thm:b3:holomorphic:formula}. A
power series is infinitely differentiable with $c_n =
f^{(n)}(a)/n!$ (Year 2), which also shows the $c_n$ do not
depend on $r$. The estimates: bound the coefficient integral
by ML.
\end{proof}

\begin{example}[Singularities dictate radii]
\label{ex:b3:holomorphic:radius}
Why does the innocent real function $\frac1{1 + x^2}$ have a
Taylor series at $x = 3$ converging only for $\abs{x - 3} <
\sqrt{10}$, when nothing goes wrong on the real line? Because
the theorem above makes the radius of convergence at $a$
equal to the distance from $a$ to the nearest point where
holomorphy fails. Here $f(z) = \frac1{1 + z^2}$ is
holomorphic exactly on $\C\setminus\{\pm\iu\}$, so the
expansion at $a = 3$ converges on the largest disc avoiding
$\pm\iu$, of radius $\abs{3 - \iu} = \sqrt{10}$ --- and
cannot converge on a larger one, since the sum would extend
$f$ holomorphically to a neighborhood of $\pm\iu$, where
$\abs f \to \infty$. The real theory sees the mysterious
radius $\sqrt{10}$; the complex plane sees two poles.
This is the practical rule: \emph{to find a radius of
convergence, locate the singularities} --- e.g.\ the Taylor
series of $\tan$ at $0$ has radius $\frac\pi2$ (nearest
zeros of $\cos$), and the Bernoulli generating function
$\frac z{\eu^z - 1}$ (\cref{pb:b3:holomorphic:1}, Part VI)
has radius $2\pi$ (nearest nonzero zeros of $\eu^z - 1$:
$\pm2\iu\pi$).
\end{example}

\begin{corollary}[Liouville; d'Alembert--Gauss]
\label{cor:b3:holomorphic:liouville}
A bounded entire function is constant. Consequently every
nonconstant polynomial over $\C$ has a root: \emph{$\C$ is
algebraically closed}.\index{Liouville's theorem}\index{fundamental
theorem of algebra}
\end{corollary}

\begin{proof}
If $\abs f \leq M$ on $\C$: for every $a$ and $r$, $\abs{c_1(a)}
= \abs{f'(a)} \leq M/r \to 0$: $f' \equiv 0$, and $f$ is
constant (on the connected $\C$: zero derivative implies
locally constant --- integrate along segments). If $P$ had no
root, $1/P$ would be entire and bounded ($\abs{P(z)} \to
\infty$ as $\abs z \to \infty$: the leading term dominates, so
$\abs{1/P}$ is small outside a large disc and continuous on
the compact disc): constant --- absurd for nonconstant $P$.
(The weekend problem gives a second, elementary proof and the
algebraic consequences.)
\end{proof}

\begin{theorem}[Zeros are isolated; identity theorem]
\label{thm:b3:holomorphic:identity}
Let $\Omega$ be \emph{connected} and $f \in \mathcal
H(\Omega)$, $f \not\equiv 0$. Then every zero $a$ of $f$ has
finite \emph{order}: $f(z) = (z - a)^m\,g(z)$ with $g \in
\mathcal H(\Omega)$, $g(a) \neq 0$, and the zeros of $f$ have
no accumulation point in $\Omega$. Consequently, if two
holomorphic functions on $\Omega$ agree on a set with an
accumulation point in $\Omega$, they agree
everywhere.\index{identity theorem}\index{isolated zeros}
\end{theorem}

\begin{proof}
Let $Z$ be the set of points where all derivatives of $f$
vanish. $Z$ is closed (intersection of closed sets) and open:
if all $c_n = 0$ at $a$, the power series expansion makes $f
\equiv 0$ on a disc around $a$. Connectedness: $Z =
\varnothing$ or $Z = \Omega$; the latter is excluded by $f
\not\equiv 0$. So at a zero $a$, some coefficient is nonzero:
let $m$ be minimal with $c_m \neq 0$; then $f(z) = (z -
a)^m\sum_{n\geq m}c_n(z-a)^{n-m}$ on a disc, and the sum
defines $g$ holomorphic near $a$ with $g(a) = c_m \neq 0$;
extend $g = f/(z-a)^m$ off $a$ (holomorphic there). Since
$g(a) \neq 0$ and $g$ is continuous, $f$ has no other zero in
a neighborhood of $a$: zeros are isolated, and a set of
isolated points has no accumulation point in $\Omega$ (an
accumulation point of zeros is a zero --- continuity --- and
would not be isolated). Identity: apply to the difference,
whose zero set has an accumulation point, forcing it into the
$Z = \Omega$ branch.
\end{proof}

\begin{theorem}[Mean value and maximum modulus]
\label{thm:b3:holomorphic:maximum}
Let $f \in \mathcal H(\Omega)$.
\begin{enumerate}
  \item (Mean value) For $\bar D(a, r) \subseteq \Omega$:
        $f(a) = \frac1{2\pi}\int_0^{2\pi}f(a +
        r\eu^{\iu t})\,\dd t$.
  \item (Maximum principle) If $\Omega$ is connected and
        $\abs f$ attains a local maximum at some point of
        $\Omega$, then $f$ is constant. Consequently, for
        $\Omega$ bounded and $f$ continuous on
        $\bar\Omega$: $\sup_{\bar\Omega}\abs f =
        \sup_{\partial\Omega}\abs f$.\index{maximum principle}
\end{enumerate}
\end{theorem}

\begin{proof}
(1) is Cauchy's formula at the center: parametrize $C_r$.
(2) Say $\abs f \leq \abs{f(a)}$ on $\bar D(a, \rho)
\subseteq \Omega$. If $f(a) = 0$, $f \equiv 0$ near $a$.
Otherwise, for $0 < r \leq \rho$, the mean value gives
\[
\abs{f(a)} \leq \frac1{2\pi}\int_0^{2\pi}\abs{f(a +
r\eu^{\iu t})}\,\dd t \leq \abs{f(a)} :
\]
the continuous nonnegative integrand $\abs{f(a)} - \abs{f(a +
r\eu^{\iu t})}$ has zero mean, hence vanishes: $\abs f$ is
\emph{constant} $= \abs{f(a)} \ne 0$ on the disc. A holomorphic
function of constant nonzero modulus on a disc is constant:
differentiating $P^2 + Q^2 = \text{const}$ gives $PP_x + QQ_x
= 0$ and $PP_y + QQ_y = 0$; substituting the Cauchy--Riemann
relations $P_y = -Q_x$, $Q_y = P_x$ into the second equation
yields the linear system
\[
P\,P_x + Q\,Q_x = 0, \qquad -P\,Q_x + Q\,P_x = 0,
\]
whose determinant is $P^2 + Q^2 \neq 0$: $P_x = Q_x = 0$, so
$f' = P_x + \iu Q_x = 0$ on the disc: $f$ constant there. The
identity theorem spreads constancy to all of $\Omega$. The boundary form: $\abs f$
attains its sup on the compact $\bar\Omega$; an interior
maximum makes $f$ constant, and the sup is attained on the
boundary in every case.
\end{proof}

\begin{theorem}[Weierstrass convergence theorem]
\label{thm:b3:holomorphic:weierstrassconv}
If $f_n \in \mathcal H(\Omega)$ converge to $f$ uniformly on
every compact subset of $\Omega$, then $f \in \mathcal
H(\Omega)$ and $f_n^{(k)} \to f^{(k)}$ uniformly on compacts,
for every $k$.\index{Weierstrass convergence theorem}
\end{theorem}

\begin{proof}
$f$ is continuous. For any closed triangle $T \subseteq
\Omega$: $\int_{\partial T}f = \lim\int_{\partial T}f_n = 0$
(uniform convergence on the compact $\partial T$; Goursat for
$f_n$). By the argument of \cref{thm:b3:holomorphic:cauchy},
$f$ has local primitives (discs are convex; only the triangle
property was used), i.e.\ $f = F'$ locally with $F$
holomorphic; $F$ is analytic
(\cref{thm:b3:holomorphic:analytic}), hence so is $f = F'$:
holomorphic. (This is \emph{Morera's
theorem}\index{Morera's theorem}: continuous with vanishing
triangle integrals implies holomorphic.) Derivatives: for
$\bar D(a, 2r) \subseteq \Omega$ and $z \in \bar D(a, r)$,
Cauchy's formula for derivatives (differentiate
\cref{thm:b3:holomorphic:formula} under the integral, or use
the coefficient formula) gives
\[
\abs{f_n'(z) - f'(z)} =
\Bigl|\frac{1}{2\iu\pi}\int_{C_{2r}}\frac{f_n(w) - f(w)}{(w -
z)^2}\,\dd w\Bigr|
\leq \frac{2r\,\sup_{C_{2r}}\abs{f_n - f}}{r^2} \to 0
\]
uniformly on $\bar D(a,r)$; cover a compact by finitely many
such discs, and iterate for higher $k$.
\end{proof}

\begin{method}\label{met:b3:holomorphic:toolkit}
The daily toolkit. To prove a function holomorphic: exhibit it
as a power series, a composition, a locally uniform limit
(\cref{thm:b3:holomorphic:weierstrassconv}), or an integral
with holomorphic parameter (\cref{exo:b3:holomorphic:7} ---
differentiate under $\int$ or apply Morera--Fubini). To prove
identities: prove them on a segment or subdomain and invoke
the identity theorem. To bound: Cauchy estimates on the
largest circle available. To prove constancy/nonexistence:
Liouville or the maximum principle. Always know \emph{where}
your function is holomorphic and \emph{which} discs fit in
$\Omega$.
\end{method}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:b3:holomorphic:1}
(a) At which points are $z \mapsto \bar z$, $\abs z^2$,
$\operatorname{Re}z$ complex-differentiable? Holomorphic on an
open set?
(b) Show that $P(x, y) = x^2 - y^2$ is the real part of a
holomorphic function on $\C$, found explicitly, and determine
\emph{all} of them.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b3:holomorphic:2}
Compute from the definitions: $\int_{C}z^n\,\dd z$ for all $n
\in \Z$, $C$ the unit circle; $\int_\gamma\bar z\,\dd z$ along
the segment $[0, 1+\iu]$ and along the two-segment path
through $1$: conclude that $\bar z$ has no primitive on any
neighborhood of these paths.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:holomorphic:3}
(a) Show that the principal logarithm $\log z = \ln\abs z +
\iu\arg z$ ($\arg \in \intoo{-\pi}\pi$) is holomorphic on
$\C\setminus\intoc{-\infty}0$ with derivative $\frac1z$
\emph{(primitive of $\frac1z$ on the star-shaped cut plane:
\cref{thm:b3:holomorphic:cauchy}; fix the constant)}.
(b) Show that no continuous logarithm exists on
$\C^*$ \emph{(its derivative-free obstruction: the index of
the unit circle)}.
(c) Expand $\log(1 + z)$ in power series on $D(0,1)$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:holomorphic:4}
(a) Let $f$ be entire with $\abs{f(z)} \leq C(1 + \abs
z)^{n}$. Show that $f$ is a polynomial of degree $\leq n$
\emph{(Cauchy estimates on large circles)}.
(b) Let $f$ be entire with $\operatorname{Re}f$ bounded above.
Show that $f$ is constant \emph{(consider $\eu^{f}$)}.
(c) Deduce ``little Picard for affine maps'': an entire
function omitting a half-plane is constant.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:holomorphic:5}
(a) Let $f$ be holomorphic on a connected $\Omega \ni 0$ with
$f(\frac1n) = \frac1{n^2}$ for all large $n$. Determine $f$.
(b) Does some holomorphic $f$ on $\C^*$ satisfy $f(\frac1n) =
\frac{(-1)^n}{n}$ for all $n \geq 1$? Justify.
(c) Exhibit two distinct holomorphic functions on
$D(0,1)\sqcup D(3,1)$ agreeing on $D(0,1)$: where does the
identity theorem use connectedness?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:holomorphic:6}
Let $f$ be holomorphic on the open unit disc $\mathbb D$,
continuous on $\bar{\mathbb D}$, with $\abs f \equiv 1$ on the
boundary circle.
(a) If $f$ has no zero in $\mathbb D$, show $f$ is constant
\emph{(apply the maximum principle to $f$ and to $1/f$)}.
(b) Give an example with a zero where $f$ is not constant.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:holomorphic:7}
(Holomorphy under the integral) Let $\mu$ be a finite measure
on a space $X$ and $g \colon X\times\Omega \to \C$ with:
$g(x, \cdot) \in \mathcal H(\Omega)$ for each $x$, $g$
measurable in $x$, and $\abs g \leq h(x)$ with $h$ integrable,
locally uniformly in $z$. Show $G(z) = \int_Xg(x, z)\dd\mu(x)$
is holomorphic on $\Omega$. \emph{(Morera: triangle integrals
vanish by Fubini and Goursat; continuity by dominated
convergence. Then apply to $\Gamma(z) =
\int_0^\infty t^{z-1}\eu^{-t}\dd t$ on $\{\operatorname{Re}z >
0\}$.)}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b3:holomorphic:8}
(Gauss--Lucas) Let $P \in \C[X]$ be nonconstant. Show that
every root of $P'$ lies in the convex hull of the roots of
$P$. \emph{(Write $\frac{P'}{P} = \sum_k\frac{m_k}{z - a_k}$
at a root $z$ of $P'$ that is not a root of $P$, take
conjugates, and read a convex combination.)} Illustrate on
$P = z^3 - 1$.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b3:holomorphic:9}
Let $f$ be entire and doubly periodic: $f(z + 1) = f(z + \iu)
= f(z)$ for all $z$. Show that $f$ is constant. \emph{(Bound
$f$ on the compact fundamental square, then everywhere;
Liouville.)} Moral: nonconstant elliptic functions must have
poles --- the theme of \cref{ch:b3:residues}.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:holomorphic:10}
(a) Show that $P = \operatorname{Re}f$ of a holomorphic $f$
satisfies the mean value property $P(a) =
\frac1{2\pi}\int_0^{2\pi}P(a + r\eu^{\iu t})\dd t$ and is
\emph{harmonic}: $\partial^2_{xx}P + \partial^2_{yy}P = 0$
(differentiate Cauchy--Riemann; use
\cref{thm:b3:holomorphic:analytic} for the needed smoothness).
(b) Deduce the maximum principle for real parts of holomorphic
functions on bounded domains.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b3:holomorphic:11}
(Schwarz reflection) Let $\Omega^+ = \{z : \abs z < 1,\
\operatorname{Im}z > 0\}$, $I = \intoo{-1}1$, and $f$
holomorphic on $\Omega^+$, continuous on $\Omega^+\cup I$,
\emph{real-valued on $I$}. Define
\[
F(z) = \begin{cases} f(z) & z \in \Omega^+\cup I,\\
\overline{f(\bar z)} & \bar z \in \Omega^+ . \end{cases}
\]
(a) Show that $F$ is well defined and continuous on
$\Omega = \Omega^+\cup I\cup\Omega^-$, and holomorphic on
$\Omega^\pm$ \emph{(for $\Omega^-$: check Cauchy--Riemann
for $\overline{f(\bar z)}$, or expand $f$ in local power
series and conjugate coefficients)}.
(b) Show that $F$ is holomorphic on all of $\Omega$ by
Morera's criterion: $\int_{\partial T}F = 0$ for every
triangle $T \subseteq \Omega$ \emph{(split triangles at $I$
and push their horizontal sides off the axis by
$\varepsilon$, using uniform continuity)}.
(c) Deduce: a holomorphic function on the disc, real on a
diameter, satisfies $f(\bar z) = \overline{f(z)}$; and a
nonconstant holomorphic function cannot be real-valued on
any nonempty open subset of its (connected) domain.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:holomorphic:12}
(The complex Pythagorean equation) Find all pairs of
\emph{entire} functions with $f^2 + g^2 = 1$.
(a) Show that $h = f + \iu g$ is entire and zero-free, and
that every zero-free entire function is $\eu^{\varphi}$ for
some entire $\varphi$ \emph{($h'/h$ is entire, hence has a
primitive on the star-shaped $\C$; adjust the constant and
show $h\eu^{-\varphi}$ is constant)}.
(b) Conclude $f = \cos\varphi$, $g = \sin\varphi$ with
$\varphi$ entire, and check the converse. What are the
entire solutions of $f^2 + g^2 = 0$?
\end{exercise}

\section{Problem: the fundamental theorem of algebra, twice}

\begin{problem}[{Weekend problem --- $\C$ is algebraically
closed: d'Alembert's proof, Liouville's proof, and the
harvest}]\label{pb:b3:holomorphic:1}
Let $P(z) = z^n + a_{n-1}z^{n-1} + \dots + a_0$, $n \geq 1$.
We prove twice that $P$ has a root, then collect what algebra
has been waiting for since \cref{ch:b3:galois}.

\medskip
\noindent\textbf{Part I --- Coercivity and the minimum.}
\begin{enumerate}
  \item Show that $\abs{P(z)} \to +\infty$ as $\abs z \to
        \infty$: precisely, $\abs{P(z)} \geq \frac12\abs z^n$
        for $\abs z \geq R_0$ suitable.
  \item Deduce that $\abs P$ attains a global minimum on $\C$:
        there is $z_0$ with $\abs{P(z_0)} = \inf_\C\abs P$
        \emph{(compactness of a large closed disc,
        \cref{cor:b3:topology:heineborel})}.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- d'Alembert's descent.} Suppose,
for contradiction, $P(z_0) \neq 0$.
\begin{enumerate}[resume]
  \item Expand $Q(h) = P(z_0 + h)/P(z_0)$ as a polynomial in
        $h$: $Q(h) = 1 + c_kh^k + h^{k+1}S(h)$ with $c_k \neq
        0$, $k \geq 1$, $S$ a polynomial.
  \item Choose the direction of descent: for small $t > 0$,
        set $h = t\,\omega$ where $\omega^k = -1/c_k$ (why
        does such $\omega$ exist? --- \emph{prove} the
        existence of $k$-th roots of any complex number via
        polar form, independently of the theorem being
        proved). Show
        \[
        \abs{Q(t\omega)} \leq 1 - t^k + C\,t^{k+1}
        \]
        for $t$ small, with an explicit constant $C$.
  \item Conclude $\abs{Q(t\omega)} < 1$ for small $t$ ---
        contradicting the minimality of $\abs{P(z_0)}$. Hence
        $P(z_0) = 0$: \emph{every nonconstant complex
        polynomial has a root} (d'Alembert--Argand).
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Liouville's one-liner,
in full.}
\begin{enumerate}[resume]
  \item Write out carefully the proof of
        \cref{cor:b3:holomorphic:liouville}: if $P$ has no
        root, verify that $1/P$ is entire, bounded (quantify,
        using question 1), hence constant, and conclude.
        Compare the two proofs: which ingredients does each
        use? (Compactness appears in both --- where?)
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The harvest.}
\begin{enumerate}[resume]
  \item Show that every $P \in \C[X]$ of degree $n$ splits:
        $P = c\prod_{i}(X - \alpha_i)^{m_i}$ with $\sum m_i =
        n$ \emph{(induction, Euclidean division by $(X -
        \alpha)$)}.
  \item Show that the irreducible polynomials of $\R[X]$ are
        the linear ones and the quadratics with negative
        discriminant \emph{(pair conjugate roots)}; deduce
        that every real polynomial of odd degree has a real
        root, and give a second, order-theoretic proof of
        that last fact (intermediate value theorem) ---
        checking they agree on $X^3 - X - 1$.
  \item Deduce the debts this book can now repay: (i) every
        endomorphism of a nonzero finite-dimensional
        $\C$-vector space has an eigenvalue, so every complex
        matrix has a Jordan form
        (\cref{thm:b3:modules:jordan}); (ii) the field
        $\bar\Q$ of algebraic numbers used in
        \cref{rem:b3:galois:closureexamples} is indeed an
        algebraic closure of $\Q$.
  \item (Finale) Pinpoint where each proof would break over a
        field like $\Q(\iu)$: which steps use the existence of
        $k$-th roots (question 4), and which use compactness
        or completeness (questions 2 and 6)? Conclude in five
        lines: the theorem is genuinely \emph{analytic} ---
        every proof somewhere invokes the completeness or
        connectedness of $\R$ --- even though its statement is
        purely algebraic.
\end{enumerate}

\medskip
\noindent\textbf{Part V --- The rigidity ladder of entire
functions.} Liouville is the first rung of a ladder; we climb
it.
\begin{enumerate}[resume]
  \item (Cauchy estimates) From the Cauchy formula on the
        circle of radius $r$ around $a$, prove
        \[
        \bigl|f^{(n)}(a)\bigr| \leq
        \frac{n!\,\sup_{\abs{z-a}=r}\abs f}{r^n} ,
        \]
        and recover Liouville as the case $n = 1$, $r \to
        \infty$.
  \item (Polynomial growth) Show that an entire $f$ with
        $\abs{f(z)} \leq A + B\abs z^m$ for all $z$ is a
        polynomial of degree $\leq m$ \emph{(kill the Taylor
        coefficients beyond $m$ with question 11)}.
  \item (Bounded real part) Show that an entire $f$ with
        $\operatorname{Re}f$ bounded above is constant
        \emph{(apply Liouville to $\eu^{f}$)}.
  \item (Double periodicity) Let $f$ be entire with $f(z + 1)
        = f(z)$ and $f(z + \iu) = f(z)$ for all $z$. Show $f$
        is constant. Conclude: a nonconstant ``elliptic''
        function must have singularities --- the historical
        reason poles enter complex analysis.
  \item (Dense range) Show that the range of a nonconstant
        entire function is dense in $\C$: if $f(\C)$ misses a
        disc $D(a, r)$, then $\frac1{f - a}$ is entire and
        bounded. (Picard proved the range misses at most
        \emph{one point}; density is the level our tools
        reach.)
  \item (Proper $\Rightarrow$ polynomial) Suppose $f$ is
        entire and $\abs{f(z)} \to \infty$ as $\abs z \to
        \infty$. Show: the zeros of $f$ are finite in number
        ($z_1, \dots, z_p$, with multiplicities $m_i$); the
        quotient $g = f/\prod(z - z_i)^{m_i}$ is entire and
        zero-free; $1/g$ has polynomial growth, hence
        (question 12) is a polynomial, necessarily constant
        (zero-free); conclude that $f$ is a polynomial. So
        among entire functions, polynomials are exactly the
        proper ones --- $\eu^z$ fails properness along
        $\R_-$.
\end{enumerate}

\medskip
\noindent\textbf{Part VI --- Harmonic shadows and a mean
value of Gauss.}
\begin{enumerate}[resume]
  \item Let $f = u + \iu v$ be holomorphic on an open set.
        Verify that $u = \operatorname{Re}f$ satisfies the
        mean value property
        \[
        u(a) = \frac1{2\pi}\int_0^{2\pi}
        u\bigl(a + r\eu^{\iu\theta}\bigr)\,\dd\theta
        \]
        (real part of the Cauchy formula), and deduce the
        maximum principle for $u$ on a bounded domain, with
        the same connectedness proof as for $\abs f$.
  \item (Gauss's mean value) For $a \in \C$ and $r > 0$ with
        $\abs a \neq r$, prove
        \[
        \frac1{2\pi}\int_0^{2\pi}
        \log\bigl|a - r\eu^{\iu\theta}\bigr|\,\dd\theta
        = \log\max\bigl(\abs a, r\bigr)
        \]
        \emph{(if $\abs a > r$: $z \mapsto \log\abs{a - z}$
        is the real part of a holomorphic logarithm on a
        neighborhood of the closed disc --- why does one
        exist? --- so question 17 applies; if $\abs a < r$:
        factor $\abs{a - r\eu^{\iu\theta}} = r\,\abs{1 -
        \frac ar\eu^{-\iu\theta}}$ and reuse the first
        case)}.
  \item (Mahler measure) For $P = c\prod_{i=1}^n(X -
        \alpha_i) \in \C[X]$, deduce \emph{Jensen's formula
        for polynomials}:
        \[
        \frac1{2\pi}\int_0^{2\pi}\log\bigl|P(\eu^{\iu\theta})
        \bigr|\,\dd\theta
        = \log\Bigl(\abs c\prod_{i}\max(1,
        \abs{\alpha_i})\Bigr) :
        \]
        the geometric mean of $\abs P$ on the unit circle
        reads off the roots outside the disc. Verify on $P =
        X^2 - X$ and on $P = 2X - 1$.
  \item (Bernoulli numbers) Define the coefficients $B_n$ by
        $\frac{z}{\eu^z - 1} = \sum_{n\geq0}\frac{B_n}{n!}
        z^n$ near $0$ (why is the left side analytic at
        $0$?). Derive the recurrence
        $\sum_{k=0}^{n}\binom{n+1}kB_k = 0$ ($n \geq 1$) from
        $(\eu^z - 1)\cdot\frac z{\eu^z-1} = z$, compute $B_0,
        \dots, B_6$, and show $B_{2k+1} = 0$ for $k \geq 1$
        \emph{(the function $\frac z{\eu^z-1} + \frac z2$ is
        even)}. These numbers will price every $\zeta(2k)$ in
        \cref{ch:b3:residues}.
  \item (Reality) Show that an entire function taking real
        values on $\R$ satisfies $f(\bar z) =
        \overline{f(z)}$ everywhere \emph{(compare Taylor
        coefficients at $0$, or apply the identity theorem
        to $z \mapsto \overline{f(\bar z)}$)}; deduce again
        that nonreal roots of real polynomials come in
        conjugate pairs (question 8's pairing, reproved
        analytically).
  \item (The moral) Assemble the rigidity ladder: bounded
        $\Rightarrow$ constant; polynomially bounded
        $\Rightarrow$ polynomial; proper $\Rightarrow$
        polynomial; missing a disc $\Rightarrow$ constant;
        doubly periodic $\Rightarrow$ constant. Contrast in a
        short paragraph with real $\mathcal C^\infty$
        functions (bump functions,
        \cref{thm:b3:lp:regularization}): why does
        holomorphy, a purely local condition, impose global
        law and order?
\end{enumerate}

\medskip
\noindent\textbf{Part VII --- Last harvest.}
\begin{enumerate}[resume]
  \item (Landau's inequality) For $P = \sum_{k=0}^na_kX^k$,
        prove the mean value $\frac1{2\pi}\int_0^{2\pi}
        \abs{P(\eu^{\iu\theta})}^2\dd\theta =
        \sum_k\abs{a_k}^2$ (orthogonality of the
        $\eu^{\iu k\theta}$), then, using the pointwise bound
        $\log t \leq t - 1$ to compare the means of
        $\log\abs P^2$ and $\abs P^2$, deduce from question
        19 that
        \[
        \abs c\prod_{i}\max\bigl(1, \abs{\alpha_i}\bigr)
        \;\leq\;
        \Bigl(\sum_{k=0}^{n}\abs{a_k}^2\Bigr)^{1/2} :
        \]
        the product of the roots outside the unit disc is
        controlled by the $\ell^2$ size of the coefficients
        (handle roots \emph{on} the circle by applying the
        inequality to $P(rX)$ and letting $r \to 1$). Check
        it on $X^2 - X$.
  \item (Bernoulli numbers grow factorially) Show that the
        radius of convergence of $\sum\frac{B_n}{n!}z^n$ is
        \emph{exactly} $2\pi$: at least $2\pi$ because
        $z/(\eu^z - 1)$ extends holomorphically to
        $D(0, 2\pi)$, at most $2\pi$ because the sum would
        otherwise stay bounded near $2\pi\iu$, where
        $\abs{z/(\eu^z-1)} \to \infty$. Deduce
        \[
        \limsup_{k\to\infty}
        \Bigl(\frac{\abs{B_{2k}}}{(2k)!}\Bigr)^{1/2k}
        = \frac1{2\pi} :
        \]
        Bernoulli numbers grow factorially. Admitting $B_{12}
        = -\frac{691}{2730}$ (the recurrence of question 20,
        run further), compare $\abs{B_{12}} \approx 0.25311$
        with the sharper prediction $2\,(2k)!/(2\pi)^{2k}
        \approx 0.25305$ at $k = 6$ --- four matching digits
        of an asymptotic law that \cref{ch:b3:residues} will
        prove exactly, via $\zeta(2k)$.
  \item (Roots move continuously) Let $(P_j)$ be monic of
        degree $n$ with coefficients converging to those of
        (monic) $P$. Prove the \emph{Cauchy bound}: every
        root of a monic $Q = X^n + \sum_{k<n}q_kX^k$
        satisfies $\abs\alpha \leq 1 + \max_k\abs{q_k}$;
        deduce that the roots of the $P_j$ stay in a fixed
        compact set, and, by extracting convergent
        subsequences of root vectors and passing to the
        limit in the factorization of question 7, that the
        root multisets of $P_j$ converge to that of $P$.
        Show finally that continuity is the best one can
        say: for $P_\varepsilon = X^2 - 2X + 1 +
        \varepsilon$, a perturbation of size $\varepsilon =
        10^{-4}$ moves the double root $1$ by $10^{-2}$ ---
        H\"older exponent $\frac1m$ at an $m$-fold root,
        never Lipschitz: numerically, multiple roots cost
        half the digits.
\end{enumerate}
\end{problem}
