\chapter{लॉरेंट श्रृंखला और अवशेष प्रमेय}
\label{ch:b3:residues}

क्या होता है ए होलोमोर्फिक फ़ंक्शन एक बिंदु के निकट जहां यह
है \emph{नहीं} परिभाषित? उत्तर है ए \omterm{def:b3:complete:complete}{पूरा} ट्राइकोटॉमी ---
हटाने योग्य बिंदु, ध्रुव, या आवश्यक विलक्षणता --- पढ़ें ए
दोतरफा शक्ति श्रृंखला, लॉरेंट विस्तार। एक गुणांक
उस विस्तार का, \emph{\omterm{def:b3:residues:singularities}{अवशेष}}, प्रत्येक को नियंत्रित करता है \omterm{def:b3:holomorphic:contour}{समोच्च
अभिन्न} विलक्षणता के आसपास: \omterm{def:b3:residues:singularities}{अवशेष} प्रमेय परिवर्तित करता है
परिमित बीजगणित में कठोर निश्चित समाकलन, शून्य की गिनती करता है
फ़ंक्शंस (तर्क सिद्धांत, रूच\'ई), और खुला साबित होता है
मानचित्रण प्रमेय. हम सबसे पहले कॉची के प्रमेय को अपग्रेड करते हैं
तारे के आकार के डोमेन को उसके निश्चित, समरूपता-मुक्त रूप में ---
डिक्सन का सुंदर तर्क --- ताकि सब कुछ समरूप हो
पूरक के चारों ओर घुमावदार संख्या शून्य उपलब्ध हो जाती है।

\section{वैश्विक कॉची प्रमेय}

ए \emph{चक्र}\index{चक्र} $\Gamma $ का एक सीमित औपचारिक योग है
\omterm{def:b3:topology:interior}{बंद} के रास्ते $\gamma_1, \dots, \gamma_m $; अभिन्न और
सूचकांक साथ में $\Gamma $ संगत योग हैं, और
$\operatorname{im}\Gamma = \bigcup\operatorname{im}\gamma_j $.

\begin{theorem}[कॉची, वैश्विक रूप]
\label{thm:b3:residues:globalcauchy}
होने देना $\Omega \subseteq \C $ खुले रहो, $ f \in \mathcal H(\Omega)$,
और $\Gamma $ में एक चक्र $\Omega $ ऐसा है कि
\[
\operatorname{Ind}_\Gamma(w) = 0
\qquad\text{for every } w \notin \Omega .
\]
फिर, सभी के लिए $ z \in \Omega\setminus\operatorname{im}\Gamma $,
\[
\frac1{2\iu\pi}\int_\Gamma\frac{f(w)}{w - z}\,\dd w =
\operatorname{Ind}_\Gamma(z)\,f(z),
\qquad\text{and}\qquad
\int_\Gamma f(w)\,\dd w = 0 .
\]
\index{कॉची प्रमेय (वैश्विक)}
\end{theorem}

\begin{proof}[प्रमाण (डिक्सन)]
परिभाषित करें $ g \colon \Omega\times\Omega \to \C $ द्वारा
\[
g(z, w) = \begin{cases}
\dfrac{f(w) - f(z)}{w - z} & w \neq z,\\[4pt]
f'(z) & w = z .
\end{cases}
\]
\emph{$ g $ है \omterm{def:b3:topology:continuity}{निरंतर}}: विकर्ण से हटकर, स्पष्ट। ए के पास
विकर्ण बिंदु $(a, a)$, बढ़ाना $ f $ एक शक्ति श्रृंखला में $ a $
(\cref{thm:b3:holomorphic:analytic}): $ f(w) - f(z) =
\sum_{n\geq1}c_n\bigl((w-a)^n - (z-a)^n\bigr)$, और विभाजित करना
प्रत्येक पद द्वारा $ w - z $ (का गुणनखंडीकरण $ u^n - v^n $) देता है, के लिए
$ z, w \in D(a, r)$,
\[
g(z, w) =
\sum_{n\geq1}c_n\sum_{j=0}^{n-1}(w-a)^{\,j}(z-a)^{\,n-1-j},
\]
विकर्ण पर भी मान्य (प्रत्येक आंतरिक योग बन जाता है
$ n(z-a)^{n-1}$, संक्षेप में $ f'(z)$). के लिए $ r $ छोटी श्रृंखला
पर समान रूप से अभिसरित होता है $ D(a,r)^2$
($\abs{\text{term}} \leq n\abs{c_n}r^{n-1}$, अंदर सारांशित
त्रिज्या): योग है \omterm{def:b3:topology:continuity}{निरंतर}.

तय करना $ h(z) = \frac1{2\iu\pi}\int_\Gamma g(z, w)\,\dd w $ पर
$\Omega $: \omterm{def:b3:topology:continuity}{निरंतर} (वर्दी निरंतरता का $ g $ पर कॉम्पैक्ट कैमरों),
और \omterm{def:b3:holomorphic:holo}{होलोमार्फिक} --- मोरेरा द्वारा
(\cref{thm:b3:holomorphic:weierstrassconv}की कसौटी): एक के लिए
त्रिकोण $ T \subseteq \Omega $, फ़ुबिनी देता है $\int_{\partial
T}h = \frac1{2\iu\pi}\int_\Gamma\bigl(\int_{\partial T}g(z,
w)\dd z\bigr)\dd w = 0$, आंतरिक अभिन्न लुप्त हो रहा है क्योंकि
$ z \mapsto g(z, w)$ है \omterm{def:b3:holomorphic:holo}{होलोमार्फिक} पर $\Omega $ (पर $ z = w $ द
विलक्षणता हटाने योग्य है: $ g $ है \omterm{def:b3:topology:continuity}{निरंतर} वहाँ और
\omterm{def:b3:holomorphic:holo}{होलोमार्फिक} अन्यत्र --- का विस्तार तर्क
\cref{thm:b3:holomorphic:formula}का प्रमाण)

पर \omterm{def:b3:topology:topology}{खुला सेट} $\Omega' = \{z \notin \operatorname{im}\Gamma
: \operatorname{Ind}_\Gamma(z) = 0\}$, परिभाषित करना $ h_1(z) =
\frac1{2\iu\pi}\int_\Gamma\frac{f(w)}{w - z}\dd w $:
\omterm{def:b3:holomorphic:holo}{होलोमार्फिक} पर $\Omega'$ (मोरेरा या विभेदन के अंतर्गत
अभिन्न)। के लिए $ z \in \Omega\cap\Omega'$:
\[
h(z) = \frac1{2\iu\pi}\int_\Gamma\frac{f(w)}{w-z}\dd w
- f(z)\operatorname{Ind}_\Gamma(z) = h_1(z) .
\]
परिकल्पना द्वारा $\Omega\cup\Omega' = \C $ ($ w \notin \Omega
\Rightarrow \operatorname{Ind}_\Gamma(w) = 0$), इसलिए $ h $ और
$ h_1$ संपूर्ण फ़ंक्शन को गोंद करें $ H $. असीमित के बाद से
के पूरक का घटक $\operatorname{im}\Gamma $ में निहित है $\Omega'$ और $ h_1(z) \to 0$ जैसा $\abs z \to \infty $ (एमएल
बाध्य), $ H $ बंधा हुआ है और इसकी प्रवृत्ति है $0$: लिउविल
(\cref{cor:b3:holomorphic:liouville}) देता है $ H \equiv 0$. इस प्रकार
$ h \equiv 0$ पर $\Omega $, जो अभिन्न सूत्र है।
इसे लागू करना, निश्चित के लिए $ a \in
\Omega\setminus\operatorname{im}\Gamma $, को $\tilde f(w) =
(w - a)f(w)$ पर $ z = a $:
\[
\frac1{2\iu\pi}\int_\Gamma f(w)\dd w =
\frac1{2\iu\pi}\int_\Gamma \frac{\tilde f(w)}{w - a}\dd w =
\operatorname{Ind}_\Gamma(a)\,\tilde f(a) = 0 .
\]
\end{proof}

\section{लॉरेंट श्रृंखला और पृथक विलक्षणताएँ}

\begin{theorem}[लॉरेंट विस्तार]\label{thm:b3:residues:laurent}
होने देना $ f $ होना \omterm{def:b3:holomorphic:holo}{होलोमार्फिक} वलय पर $ A = \{r < \abs{z - a} <
R\}$ ($0 \leq r < R \leq \infty $). तब\index{लॉरेंट श्रृंखला}
\[
f(z) = \sum_{n\in\Z}c_n\,(z - a)^n
\qquad\text{on } A,
\qquad
c_n = \frac1{2\iu\pi}\int_{C_\rho}\frac{f(w)}{(w -
a)^{n+1}}\,\dd w
\]
किसी के लिए $ r < \rho < R $ (स्वतंत्र) $\rho $), दो
अर्ध-श्रृंखला \omterm{def:b3:groups:normal}{सामान्य} रूप से परिवर्तित हो रही है कॉम्पैक्ट सुबन्नुलि. द
विस्तार अद्वितीय है.
\end{theorem}

\begin{proof}
हल करना $ r < \rho_1 < \abs{z - a} < \rho_2 < R $ और चलो $\Gamma =
C_{\rho_2} - C_{\rho_1}$ (बाहरी वामावर्त, भीतरी
दक्षिणावर्त): एक चक्र में $ A $ साथ
$\operatorname{Ind}_\Gamma(w) = 0$ सभी के लिए $ w \notin A $
(छोटी डिस्क के अंदर बिंदु: $1 - 1 = 0$; बड़े के बाहर
एक: $0 - 0$). द्वारा \cref{thm:b3:residues:globalcauchy},
$\operatorname{Ind}_\Gamma(z) = 1 - 0 = 1$ देता है
\[
f(z) = \frac1{2\iu\pi}\int_{C_{\rho_2}}\frac{f(w)}{w - z}\dd
w - \frac1{2\iu\pi}\int_{C_{\rho_1}}\frac{f(w)}{w - z}\dd w .
\]
पहले कर्नेल को इस प्रकार विस्तृत करें
\cref{thm:b3:holomorphic:analytic} (की शक्तियां $\frac{z -
a}{w - a}$, मापांक $< 1$): गैर-नकारात्मक भाग
$\sum_{n\geq0}c_n(z-a)^n $. दूसरे में दूसरे का विस्तार करें
रास्ता: $\frac{-1}{w - z} = \frac1{(z-a)(1 - \frac{w - a}{z -
a})} = \sum_{m\geq0}\frac{(w-a)^m}{(z - a)^{m+1}}$, सामान्यतः
पर अभिसरण $ C_{\rho_1}$: नकारात्मक भाग
$\sum_{n\leq-1}c_n(z-a)^n $ बताए गए गुणांकों के साथ (सूचकांक)
$ n = -m-1$). की आज़ादी $\rho $: गुणांक अभिन्न
खत्म $ C_{\rho}$ और $ C_{\rho'}$ से भिन्न
$\int_\Gamma $ के एक शून्य-सूचकांक चक्र पर \omterm{def:b3:holomorphic:holo}{होलोमार्फिक}
$\frac{f(w)}{(w-a)^{n+1}}$ में $ A $: शून्य, द्वारा
\cref{thm:b3:residues:globalcauchy} दोबारा। विशिष्टता:
एकीकृत करें $\sum c_n(z-a)^n $ ख़िलाफ़ $(z - a)^{-m-1}$ ऊपर
$ C_\rho $ पद दर पद (\omterm{def:b3:groups:normal}{सामान्य} अभिसरण): केवल $ n = m $
जीवित रहता है.
\end{proof}

\begin{definition}\label{def:b3:residues:singularities}
अगर $ f $ है \omterm{def:b3:holomorphic:holo}{होलोमार्फिक} एक पंचर डिस्क पर $ D(a, R)\setminus
\{a\}$, लॉरेंट द्वारा विस्तार करें ($ r = 0$). तीन विशेष
मामले:\index{पृथक विलक्षणता}
\begin{itemize}
  \item सभी $ c_n = 0$ के लिए $ n < 0$: \emph{\omterm{thm:b3:residues:riemanncw}{हटाने योग्य
        विलक्षणता}} (गैर-नकारात्मक श्रृंखला का विस्तार होता है $ f $
        समग्र रूप से को $ a $);
  \item $ c_n \neq 0$ बहुत से लोगों के लिए, कम से कम एक, $ n < 0$:
        ए \emph{खंभा} आदेश की $ m = -\min\{n : c_n \neq 0\}$;
        इसके तुल्य $ f = g/(z-a)^m $, $ g $ \omterm{def:b3:holomorphic:holo}{होलोमार्फिक}, $ g(a)
        \neq 0$; इसके तुल्य $\abs{f(z)} \to \infty $ जैसा $ z\to
        a $;
  \item असीम रूप से कई नकारात्मक $ c_n \neq 0$:
        \emph{आवश्यक विलक्षणता}.
\end{itemize}
 \emph{अवशेष}\index{अवशेष} है
$\operatorname{Res}(f, a) = c_{-1}$. एक समारोह \omterm{def:b3:holomorphic:holo}{होलोमार्फिक} पर
$\Omega $ माइनस डंडों का एक सेट है
\emph{मेरोमोर्फिक}\index{मेरोमोर्फिक} पर $\Omega $.
\end{definition}

\begin{theorem}[रीमैन; कैसोराती--वीयरस्ट्रैस]
\label{thm:b3:residues:riemanncw}
होने देना $ f $ होना \omterm{def:b3:holomorphic:holo}{होलोमार्फिक} पर $ D(a,R)\setminus\{a\}$.
\begin{enumerate}
  \item (रीमैन) यदि $ f $ है \emph{घिरे} पास में $ a $, द
        विलक्षणता हटाने योग्य है.\index{हटाने योग्य विलक्षणता}
  \item (कैसोराटी--वीयरस्ट्रैस) यदि $ a $ तो यह आवश्यक है
        $ f\bigl(D(a,\varepsilon)\setminus\{a\}\bigr)$ \omterm{def:b3:topology:interior}{घना} है
        में $\C $ हरएक के लिए $\varepsilon $.
        \index{कैसोराती--वीयरस्ट्रैस प्रमेय}
\end{enumerate}
\end{theorem}

\begin{proof}
(1) के लिए $ n < 0$ और $\rho \to 0$: $\abs{c_n} \leq
M\rho^{-n-1}\cdot\rho\cdot\rho^{-\,n}\dots $ एमएल द्वारा
$ C_\rho $: $\abs{c_n} \leq \frac{1}{2\pi}\,2\pi\rho\cdot
M\rho^{-(n+1)} = M\rho^{-n} \to 0$ (जैसा $-n > 0$): सभी
नकारात्मक गुणांक गायब हो जाते हैं।
(2) यदि कुछ मान $ b $ संपर्क नहीं किया गया: $\abs{f - b} \geq
\delta $ पास में $ a $, इसलिए $ g = 1/(f - b)$ है \omterm{def:b3:holomorphic:holo}{होलोमार्फिक} और
के पास घिरा हुआ $ a $: हटाने योग्य (1), $ g $ मूल्य के साथ विस्तारित होता है $ c $.
अगर $ c \neq 0$, $ f = b + 1/g $ के पास घिरा हुआ है $ a $: हटाने योग्य
--- बहिष्कृत. अगर $ c = 0$, $ g $ परिमित क्रम का शून्य है $ m $
पर $ a $ (\cref{thm:b3:holomorphic:identity}; $ g \not\equiv 0$),
और $ f = b + 1/g $ आदेश का एक ध्रुव है $ m $: फिर से बाहर रखा गया.
\end{proof}

\section{अवशेष प्रमेय}

\begin{theorem}[अवशेष प्रमेय]\label{thm:b3:residues:residue}
होने देना $\Omega $ खुले रहो, $ S \subseteq \Omega $ परिमित, $ f \in
\mathcal H(\Omega\setminus S)$, और $\Gamma $ में एक चक्र
$\Omega\setminus S $ साथ $\operatorname{Ind}_\Gamma(w) = 0$
हरएक के लिए $ w \notin \Omega $. तब\index{अवशेष प्रमेय}
\[
\frac{1}{2\iu\pi}\int_\Gamma f(z)\,\dd z
= \sum_{a\in S}\operatorname{Ind}_\Gamma(a)\,
\operatorname{Res}(f, a) .
\]
\end{theorem}

\begin{proof}
प्रत्येक के लिए $ a \in S $, होने देना $ P_a(z) = \sum_{n\leq-1}c_n^{(a)}(z -
a)^n $ का मुख्य भाग बनें $ f $ पर $ a $: एक शृंखला
पर एकत्रित होना $\C\setminus\{a\}$ (इसकी त्रिज्या में $1/(z-a)$ है
अनंत: लॉरेंट पूंछ सभी छोटे के लिए अभिसरण करती है $\abs{z-a}$
इसलिए, यह एक शक्ति श्रृंखला है $(z-a)^{-1}$, हर जगह), और
\omterm{def:b3:holomorphic:holo}{होलोमार्फिक} वहाँ। तब $ g = f - \sum_{a\in S}P_a $ है
के प्रत्येक बिंदु पर हटाने योग्य विलक्षणताएँ $ S $ (यह लॉरेंट है
पर विस्तार $ a $ इसका कोई नकारात्मक हिस्सा नहीं है: दूसरा $ P_{a'}$ हैं
\omterm{def:b3:holomorphic:holo}{होलोमार्फिक} पर $ a $), इसलिए $ g $ का विस्तार समग्र रूप से को
$\Omega $, और \cref{thm:b3:residues:globalcauchy} देता है
$\int_\Gamma g = 0$. यह प्रत्येक को एकीकृत करना बाकी है $ P_a $:
टर्म-दर-टर्म (पर \omterm{def:b3:groups:normal}{सामान्य} अभिसरण कॉम्पैक्ट
$\operatorname{im}\Gamma $, जो टालता है $ a $),
\[
\frac1{2\iu\pi}\int_\Gamma(z - a)^n\,\dd z = 0 \ (n \leq -2:
\text{primitive } \tfrac{(z-a)^{n+1}}{n+1}),
\qquad
\frac1{2\iu\pi}\int_\Gamma\frac{\dd z}{z - a} =
\operatorname{Ind}_\Gamma(a),
\]
इसलिए $\frac1{2\iu\pi}\int_\Gamma P_a =
c_{-1}^{(a)}\operatorname{Ind}_\Gamma(a)$. योग ख़त्म $ a $.
\end{proof}

\begin{method}[अवशेषों की गणना]\label{met:b3:residues:compute}
\omterm{def:b3:groups:simple}{सरल} ध्रुव: $\operatorname{Res}(f, a) = \lim_{z\to a}(z -
a)f(z)$; के लिए $ f = g/h $ साथ $ g(a) \neq 0$, $ h(a) = 0$, $ h'(a)
\neq 0$: $\operatorname{Res} = g(a)/h'(a)$. व्यवस्था का ध्रुव $ m $:
$\operatorname{Res}(f, a) = \frac1{(m-1)!}\lim_{z\to
a}\bigl((z-a)^mf(z)\bigr)^{(m-1)}$. आवश्यक विलक्षणताएँ:
विस्तार करें और पढ़ें $ c_{-1}$ (उदा.\ ज्ञात श्रृंखला से)। हमेशा
सत्यापित करें कि समोच्च वास्तव में किन ध्रुवों को घेरता है, और किसके साथ
कौन सा सूचकांक.
\end{method}

\begin{example}[चार शास्त्रीय अभिन्न प्रकार]
\label{ex:b3:residues:integrals}
(ए) \emph{तर्कसंगत खत्म $\R $}: के लिए $\int_\R\frac{\dd x}{1 +
x^4}$, एक बड़े अर्धवृत्त के साथ \omterm{def:b3:topology:interior}{बंद} करें $ S_R $ ऊपरी में
अर्ध-तल: इंटीग्रैंड है $ O(R^{-4})$ इसलिए
$\int_{S_R} \to 0$ (एमएल), और \omterm{def:b3:residues:singularities}{अवशेष} प्रमेय के साथ
डंडे $\eu^{\iu\pi/4}, \eu^{3\iu\pi/4}$ (\omterm{def:b3:groups:simple}{सरल}, \omterm{def:b3:residues:singularities}{अवशेष}
$\frac1{4z^3} = \frac{z}{4z^4} = -\frac z4$ एक पोल पर) देता है
\[
\int_\R\frac{\dd x}{1 + x^4}
= 2\iu\pi\Bigl(-\frac{\eu^{\iu\pi/4}}4 -
\frac{\eu^{3\iu\pi/4}}4\Bigr)
= \frac{\pi}{\sqrt2} .
\]
(बी) \emph{फूरियर प्रकार}: के लिए $ t \geq 0$,
$\int_\R\frac{\eu^{\iu tx}}{1 + x^2}\dd x =
2\iu\pi\operatorname{Res}\bigl(\tfrac{\eu^{\iu tz}}{1+z^2},
\iu\bigr) = 2\iu\pi\frac{\eu^{-t}}{2\iu} = \pi\eu^{-t}$ ---
ऊपरी अर्धवृत्त काम करता है क्योंकि $\abs{\eu^{\iu tz}} =
\eu^{-t\operatorname{Im}z} \leq 1$ वहाँ; वास्तविक भाग लेना:
$\int_\R\frac{\cos(tx)}{1+x^2}\dd x = \pi\eu^{-\abs t}$,
निपटान \cref{exo:b3:lebesgue:10}का स्वीकृत फार्मूला.
(सी) \emph{एक अवधि में त्रिकोणमिति}: विकल्प $ z =
\eu^{\iu t}$, $\cos t = \frac{z + z^{-1}}2$, $\dd t =
\frac{\dd z}{\iu z}$: $\int_0^{2\pi}\frac{\dd t}{a + \cos t}$
($ a > 1$) बन जाता है \omterm{def:b3:residues:singularities}{अवशेष} यूनिट सर्कल के अंदर गिनती करें
(\cref{exo:b3:residues:2}).
(डी) \emph{शृंखला}: जोड़ा $ f $ साथ $\pi\cot(\pi z)$, जिसके ध्रुव
के साथ पूर्णांक हैं \omterm{def:b3:residues:singularities}{अवशेष} $1$: सप्ताहांत समस्या का योग
$\sum n^{-2}$ और $\sum n^{-4}$ इस तरह।
\end{example}

\section{तर्क सिद्धांत और रोच\'ई का प्रमेय}

\begin{theorem}[तर्क सिद्धांत]
\label{thm:b3:residues:argument}
होने देना $ f $ होना \omterm{def:b3:residues:singularities}{मेरोमोर्फिक} पर $\Omega $, शून्य के साथ $ z_j $ (आदेश
$ m_j $) और डंडे $ p_k $ (आदेश $\mu_k $), और $\gamma $ एक \omterm{def:b3:topology:interior}{बंद}
पथ में $\Omega $ उन सब से परहेज, साथ
$\operatorname{Ind}_\gamma = 0$ \omterm{def:b3:topology:interior}{बंद} $\Omega $. तब
\index{तर्क सिद्धांत}
\[
\frac1{2\iu\pi}\int_\gamma\frac{f'(z)}{f(z)}\,\dd z
= \sum_j m_j\operatorname{Ind}_\gamma(z_j)
- \sum_k \mu_k\operatorname{Ind}_\gamma(p_k)
\]
(निश्चित रूप से कई पद अशून्य हैं)। एक \omterm{def:b3:groups:simple}{सरल} के लिए
वामावर्त समोच्च, अभिन्न शून्य शून्य को गिनता है
अंदर ध्रुव, बहुलता के साथ --- और घुमावदार के बराबर है
छवि की संख्या पथ $ f\circ\gamma $ आस-पास $0$.
\end{theorem}

\begin{proof}
ऑर्डर के शून्य के करीब $ m $: $ f = (z-a)^mg $, $ g(a) \neq 0$, इसलिए
$\frac{f'}f = \frac m{z - a} + \frac{g'}g $ दूसरे के साथ
अवधि \omterm{def:b3:holomorphic:holo}{होलोमार्फिक} पास में $ a $: का एक \omterm{def:b3:groups:simple}{सरल} ध्रुव \omterm{def:b3:residues:singularities}{अवशेष} $ m $. निकट
व्यवस्था का एक खंभा $\mu $: $ f = (z-a)^{-\mu}g $ देता है \omterm{def:b3:residues:singularities}{अवशेष}
$-\mu $. कहीं $\frac{f'}f $ है \omterm{def:b3:holomorphic:holo}{होलोमार्फिक}. (शून्य और
शून्येतर सूचकांक वाले ध्रुव a में स्थित हैं कॉम्पैक्ट से घिरा हुआ क्षेत्र
$\gamma $; पहचान प्रमेय के अनुसार वे संख्या में सीमित हैं
वहाँ, $ f \not\equiv 0$।) आवेदन करना
\cref{thm:b3:residues:residue}. अंतिम टिप्पणी:
$\frac1{2\iu\pi}\int_\gamma\frac{f'}f =
\frac1{2\iu\pi}\int_{f\circ\gamma}\frac{\dd w}w =
\operatorname{Ind}_{f\circ\gamma}(0)$ (विकल्प $ w =
f(\gamma(t))$).
\end{proof}

\begin{theorem}[रॉच\'ई]\label{thm:b3:residues:rouche}
होने देना $ f, g $ होना \omterm{def:b3:holomorphic:holo}{होलोमार्फिक} पर $\Omega $, और $\gamma $ एक \omterm{def:b3:topology:interior}{बंद}
पथ साथ $\operatorname{Ind}_\gamma \in \{0,1\}$, शून्य \omterm{def:b3:topology:interior}{बंद}
$\Omega $ (एक \omterm{def:b3:groups:simple}{सरल} रूपरेखा). अगर\index{Rouché's theorem}
\[
\abs{g(z)} < \abs{f(z)} \qquad \text{on }
\operatorname{im}\gamma,
\]
तब $ f $ और $ f + g $ शून्य की संख्या समान है (के साथ)।
बहुलता) क्षेत्र में $\{\operatorname{Ind}_\gamma =
1\}$.
\end{theorem}

\begin{proof}
के लिए $ t \in \intcc01$, $ f_t = f + tg $ पर कोई शून्य नहीं है
$\operatorname{im}\gamma $ ($\abs{f_t} \geq \abs f - \abs g >
0$), इसलिए
\[
N(t) = \frac1{2\iu\pi}\int_\gamma
\frac{f_t'(z)}{f_t(z)}\,\dd z
\]
अच्छी तरह से परिभाषित है; यह संलग्न क्षेत्र में शून्यों की गिनती करता है
(\cref{thm:b3:residues:argument}; कोई पोल नहीं)। $ N $ है
\omterm{def:b3:topology:continuity}{निरंतर} में $ t $ (इंटीग्रैंड संयुक्त रूप से है \omterm{def:b3:topology:continuity}{निरंतर}, द
हर समान रूप से नीचे बंधे --- प्रभुत्व वाले
अभिसरण) और पूर्णांक-मूल्यवान: स्थिरांक। $ N(0) = N(1)$.
\end{proof}

\begin{corollary}[मैपिंग प्रमेय खोलें]
\label{cor:b3:residues:openmapping}
एक अचर होलोमोर्फिक फ़ंक्शन एक पर \omterm{def:b3:topology:connected}{जुड़े हुए} \omterm{def:b3:topology:topology}{खुला सेट} है
एक खुला नक्शा. विशेष रूप से (फिर से) अधिकतम सिद्धांत
धारण करता है, और ए \omterm{def:b3:holomorphic:holo}{होलोमार्फिक} आपत्ति है \omterm{def:b3:holomorphic:holo}{होलोमार्फिक}
श्लोक में।\index{ओपन मैपिंग प्रमेय (होलोमोर्फिक)}
\end{corollary}

\begin{proof}
होने देना $ f(a) = b $; $ f - b $ किसी परिमित क्रम का शून्य होता है $ m
\geq 1$ पर $ a $ (पहचान प्रमेय: $ f \not\equiv b $). चुनना
$ r $ साथ $ f - b $ शून्\omterm{def:b3:modules:free}{य-मुक्त} पर $\bar D(a, r)\setminus\{a\}$
(\omterm{thm:b3:holomorphic:identity}{पृथक शून्य}) और चलो $\delta = \min_{\abs{z - a} =
r}\abs{f(z) - b} > 0$. के लिए $\abs{w - b} < \delta $: पर
वृत्त, $\abs{(b - w)} < \delta \leq \abs{f - b}$, तो
रूच\'ई ($ f - b $ बनाम स्थिरांक $ b - w $) कहते हैं $ f - w $ है
बिलकुल $ m $ शून्य में $ D(a, r)$: प्रत्येक ऐसा $ w $ प्राप्त होता है
--- $ f(D(a,r)) \supseteq D(b, \delta)$: खुला। अधिकतम
सिद्धांत: एक आंतरिक भाग अधिकतम $\abs f $ के लिए असंभव है
एक अस्थिर $ f $, चारों ओर इसकी छवि $ f(a)$ अंक युक्त
बड़े मापांक का. उलटा: ए \omterm{def:b3:holomorphic:holo}{होलोमार्फिक} द्विभाजन $ f $ है
खुला, तो $ f^{-1}$ है \omterm{def:b3:topology:continuity}{निरंतर}; का शून्य $ f - f(a)$ पर
$ a $ साधारण है ($ m \geq 2$ देगा $ m $ आस-पास की पूर्वछवियाँ
मान --- अलग-अलग, चूँकि $ f'$ पर ही लुप्त हो जाता है
पृथक बिंदु, इतने निकट $ a $  $ m $ का शून्य $ f - w $ हैं
\omterm{def:b3:groups:normal}{सामान्य} छोटे के लिए \omterm{def:b3:groups:simple}{सरल} और विशिष्ट $ w $:विरोधाभास
इंजेक्शन के साथ); तब $ f'(a) \neq 0$ और अंतर
का भागफल $ f^{-1}$ अभिसरण: $\bigl(f^{-1}\bigr)'(b) =
1/f'(a)$.
\end{proof}

\section{अभ्यास}

\begin{exercise}[$\star $]\label{exo:b3:residues:1}
विलक्षणता को यहाँ वर्गीकृत करें $0$ और गणना करें \omterm{def:b3:residues:singularities}{अवशेष}:
\[
\frac{\sin z}{z},\qquad
\frac{\eu^z - 1}{z^2},\qquad
\frac{1}{z(z-1)^2},\qquad
\frac{\cos z}{z^3},\qquad
\eu^{1/z},\qquad
\frac1{\sin z} .
\]
यह भी दें \omterm{def:b3:residues:singularities}{अवशेष} तीसरे का $ z = 1$ और आखिरी का
पर $ z = \pi $.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b3:residues:2}
के लिए $ a > 1$ गणना, के माध्यम से $ z = \eu^{\iu t}$:
\[
\int_0^{2\pi}\frac{\dd t}{a + \cos t}
= \frac{2\pi}{\sqrt{a^2 - 1}} .
\]
सीमा व्यवहार की जाँच करें $ a \to 1^+$ और $ a \to \infty $.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:residues:3}
चाप अनुमानों को उचित ठहराते हुए, अर्धवृत्त आकृतियों के साथ गणना करें:
\[
\int_\R\frac{x^2}{1 + x^6}\,\dd x = \frac\pi3,
\qquad
\int_\R\frac{\dd x}{(1 + x^2)^{2}} = \frac\pi2
\quad\text{(re-deriving \cref{exo:b3:fouriertransform:3})}.
\]
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:residues:4}
साबित करो, के लिए $ t \geq 0$ और $ a > 0$:
\[
\int_\R\frac{\cos(tx)}{x^2 + a^2}\,\dd x =
\frac{\pi}{a}\,\eu^{-at},
\]
और फूरियर रूपांतरण का अनुमान लगाएं $ x \mapsto \eu^{-a\abs
x}$ व्युत्क्रमण द्वारा --- के साथ तुलना करना
\cref{exo:b3:fouriertransform:1}.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:residues:5}
बढ़ाना $ f(z) = \dfrac1{(z-1)(z-2)}$ में \omterm{thm:b3:residues:laurent}{लॉरेंट श्रृंखला} प्रत्येक में
तीन क्षेत्रों में से $\abs z < 1$, $1 < \abs z < 2$, $\abs z >
2$. तीनों विस्तार भिन्न क्यों हैं? स्पष्ट करें कि क्यों
का गुणांक $ z^{-1}$ दूसरे और तीसरे विस्तार में है
\emph{नहीं} ए \omterm{def:b3:residues:singularities}{अवशेष} का $ f $ पर $0$ ($ f $ कोई विलक्षणता नहीं है
वहां), और वास्तविक की गणना करें \omterm{def:b3:residues:singularities}{अवशेष} का $ f $, पर $1$ और कम से
$2$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:residues:6}
(ए) वह दिखाओ $\eu^{1/z}$ में एक आवश्यक विलक्षणता है $0$
और कैसोराती--वीयरस्ट्रैस को हाथ से सत्यापित करें: हल करें $\eu^{1/z} =
w $ स्पष्ट रूप से किसी के लिए $ w \neq 0$, समाधान प्रदर्शित करना
मनमाने ढंग से करीब $0$.
(बी) वह दिखाओ $\abs{\eu^{1/z}}$ प्रत्येक पर असीमित है
छिद्रित \omterm{def:b3:topology:topology}{अड़ोस-पड़ोस} का $0$ अभी तक $\eu^{1/z}$ कोई पोल नहीं है:
कौन सी सीमा विफल रहती है?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:residues:7}
रूच के साथ गिनें\'ई:
(ए) के शून्य $ z^7 - 4z^3 + z - 1$ में $\abs z < 1$;
(बी) के शून्य $ z^4 + 5z + 1$ में $\abs z < 1$ और में $1 <
\abs z < 2$;
(सी) डी'एलेम्बर्ट--गॉस: एक राक्षसी \omterm{def:b3:galois:extension}{डिग्री} को फिर से साबित करें-$ n $ बहुपद

$ z^n $)}.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b3:residues:8}
(हर्विट्ज़) चलो $ f_n \to f $ समान रूप से चालू कॉम्पैक्ट कैमरों, $ f_n \in
\mathcal H(\Omega)$, $\Omega $ \omterm{def:b3:topology:connected}{जुड़े हुए}, $ f \not\equiv 0$.
(ए) यदि सभी हों तो दिखाएँ $ f_n $ शून्\omterm{def:b3:modules:free}{य-मुक्त} हैं, इसलिए हैं $ f $.
\emph{(अगर $ f(a) = 0$: एक छोटे वृत्त पर तर्क सिद्धांत
चारों ओर $ a $, और \cref{thm:b3:holomorphic:weierstrassconv} को
में सीमा तक पास करें $\int f_n'/f_n $.)}
(बी) यदि सभी हों तो दिखाएँ $ f_n $ इंजेक्शन हैं, $ f $ इंजेक्शन है या
स्थिर. \emph{((ए) को लागू करें $ z \mapsto f_n(z) - f_n(w)$ पर
$\Omega\setminus\{w\}$.)}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b3:residues:9}
के लिए $ n \geq 2$, एकीकृत करें $\frac1{1 + z^n}$ सीमा के पार
सेक्टर का $\{0 \leq \arg z \leq \frac{2\pi}n,\ \abs z
\leq R\}$ और निष्कर्ष निकालें
\[
\int_0^{+\infty}\frac{\dd x}{1 + x^n} =
\frac{\pi}{n\,\sin(\pi/n)} .
\]
सत्यापित करें $ n = 2$ ख़िलाफ़ $\arctan $, और सीमा $ n \to
\infty $.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:residues:10}
होने देना $ f $ के साथ एक तर्कसंगत कार्य हो $\deg(\text{denominator})
\geq \deg(\text{numerator}) + 2$. दिखाएँ कि का योग
\emph{सभी} \omterm{def:b3:residues:singularities}{अवशेष} का $ f $ शून्य है \emph{(एकीकृत करें
बड़े और बड़े वृत्त)}. आंशिक की पुनः गणना करने के लिए इसका उपयोग करें
का अंश अपघटन $\frac1{z(z-1)(z-2)}$ बिना किसी रैखिक के
बीजगणित
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b3:residues:11}
(कीहोल: यूलर का प्रतिबिंब अभिन्न) के लिए $0 < a < 1$,
गणना करें
\[
I(a) = \int_0^{\infty}\frac{x^{a-1}}{1 + x}\,\dd x =
\frac{\pi}{\sin(\pi a)}
\]
एकीकृत करके $ f(z) = \frac{z^{a-1}}{1+z} =
\frac{\eu^{(a-1)\log z}}{1 + z}$ (लघुगणक साथ में काटें
$\R_+$, $\arg z \in \intoo0{2\pi}$) कीहोल के ऊपर
समोच्च: कट के शीर्ष से बाहर $\varepsilon $ को
$ R $, आस-पास $ C_R $, कट के नीचे वापस, चारों ओर
$ C_\varepsilon $. औचित्य सिद्ध करें: दोनों सीधे खंड अलग-अलग हैं
कारक द्वारा $\eu^{2\iu\pi(a-1)}$, वृत्त योगदान
गायब हो जाना ($ R^{a-1}\cdot R \to 0$ और
$\varepsilon^{a-1}\cdot\varepsilon \to 0$), और अद्वितीय
खंभा $ z = -1$ है \omterm{def:b3:residues:singularities}{अवशेष} $\eu^{\iu\pi(a - 1)}$. निष्कर्ष भी निकालें
$\Gamma(a)\Gamma(1 - a) = \frac\pi{\sin\pi a}$ \emph{(लिखना
$\Gamma(a)\Gamma(1-a) = B(a, 1-a)$ द्वारा
\cref{pb:b3:lebesgue:1} और स्थानापन्न $ t =
\frac{x}{1+x}$)}.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:residues:12}
(तर्क सिद्धांत के साथ संख्यात्मक रूप से शून्य की गिनती)
चलो $ P(z) = z^4 + 8z + 1$.
(ए) यूनिट डिस्क में कितने शून्य हैं? \emph{(रौच\'ई विरुद्ध
$8z + 1$.)}
(बी) वलय में कितने हैं $1 < \abs z < 3$?
\emph{(रौच\'ई विरुद्ध $ z^4$ पर $\abs z = 3$.)} तेज़ करना:
दिखाएँ कि प्रत्येक शून्य का मापांक होता है $< 2.1$.
(सी) दाएँ आधे तल में कितने? \emph{(गिनो
$\abs z = 2$ पहला; फिर काल्पनिक की छवि को ट्रैक करें
अक्ष: $ P(\iu t) = t^4 + 1 + 8\iu t $ सकारात्मक वास्तविक भाग है
संपूर्ण, इसलिए अक्ष और तर्क पर कोई शून्य नहीं है
इसके साथ भिन्नता गणना योग्य है --- एक बड़े के साथ समाप्त करें
अर्ध-डिस्क।)}
\end{exercise}

\section{संकट: \texorpdfstring{$\zeta(2k)$}{zeta(2k)} द्वारा
कोटैंजेंट}

\begin{problem}[{सप्ताहांत समस्या --- संक्षेप $\sum n^{-2k}$
साथ \omterm{def:b3:residues:singularities}{अवशेष}}]\label{pb:b3:residues:1}
 \omterm{def:b3:residues:singularities}{अवशेष} प्रमेय योग श्रृंखला: एक तर्कसंगत फ़ंक्शन को जोड़ना
के साथ $\pi\cot(\pi z)$, जिसके ध्रुव पूर्णांकों पर बैठते हैं, बदल जाते हैं
$\sum_{n}f(n)$ एक में \omterm{def:b3:residues:singularities}{अवशेष} गिनती करना। हम विधि और सिद्ध करते हैं
गणना करें $\zeta(2) = \frac{\pi^2}{6}$ और $\zeta(4) =
\frac{\pi^4}{90}$ --- फूरियर श्रृंखला द्वारा पाए गए मान
वर्ष 2 और ऑपरेटर ट्रेस इन द्वारा \cref{ch:b3:spectral}, अब तक
समोच्च एकीकरण.

\medskip
\noindent\textbf{Part I --- The cotangent kernel.}
\begin{enumerate}
  \item बताते हैं कि $\pi\cot(\pi z)$ है \omterm{def:b3:residues:singularities}{मेरोमोर्फिक} पर $\C $ के साथ
        बिल्कुल \omterm{def:b3:groups:simple}{सरल} ध्रुव $ z = n \in \Z $, की प्रत्येक
        \omterm{def:b3:residues:singularities}{अवशेष} $1$ \emph{(गणना करें
        $\lim_{z\to n}(z-n)\pi\cot\pi z $)}.
  \item लॉरेंट विस्तार की शुरुआत की गणना करें $0$:
        \[
        \pi\cot(\pi z) = \frac1z - \frac{\pi^2}{3}\,z -
        \frac{\pi^4}{45}\,z^3 + O(z^5) ,
        \]
        की शक्ति श्रृंखला को विभाजित करके $\cos $ उसके द्वारा
        $\sin $ (विभाजन का औचित्य सिद्ध करें: $\frac{\sin\pi z}{\pi z}$
        है \omterm{def:b3:holomorphic:holo}{होलोमार्फिक} और शून्येतर निकट $0$, तो यह है
        पारस्परिक है \omterm{def:b3:holomorphic:holo}{होलोमार्फिक}; गुणांकों की पहचान करें
        आदेश के माध्यम से $3$).
  \item होने देना $ C_N $ शीर्षों के साथ वर्ग की सीमा बनें
        $(\pm1\pm\iu)(N + \frac12)$. बताते हैं कि
        $\abs{\cot(\pi z)} \leq 2$ पर $ C_N $ हरएक के लिए $ N \geq
        1$. \emph{(ऊर्ध्वाधर पक्षों पर, $\cot(\pi(\pm(N +
        \frac12) + \iu y)) = \mp\tan(\iu\pi y)$, मापांक का
        $\abs{\tanh(\pi y)} \leq 1$; क्षैतिज पक्षों पर
        $\abs y = N + \frac12$, अवश्यंभावी
        $\abs{\cot(\pi(x\pm\iu y))} \leq \coth(\pi y) \leq
        \coth(\pi/2) < 1.1$.)}
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The summation theorem.}
\begin{enumerate}[resume]
  \item होने देना $ f $ विवेकपूर्ण, \omterm{def:b3:holomorphic:holo}{होलोमार्फिक} पूर्णांकों पर,
        के साथ $\deg(\text{denom}) \geq \deg(\text{num}) + 2$.
        का उपयोग \omterm{def:b3:residues:singularities}{अवशेष} प्रमेय पर $ C_N $ और की सीमा
        प्रश्न 3, सिद्ध करें:
        \[
        \lim_{N\to\infty}\ \sum_{n = -N}^{N} f(n)
        = -\sum_{p\ \text{pole of}\ f}
        \operatorname{Res}\bigl(\pi\cot(\pi z)f(z),\,p\bigr).
        \]
  \item तर्क को \omterm{def:b3:galois:extension}{डिग्री} शर्त की आवश्यकता कहाँ है?
        उदाहरण द्वारा दिखाएँ (ले $ f(z) = 1/(z + \frac12)$)
        धीमी गति से क्षय के लिए सममित सीमा
        यह अभी भी मौजूद हो सकता है जबकि दो-तरफा श्रृंखला अलग हो जाती है
        --- और वह सूत्र तब गणना करता है
        \emph{प्रमुख मूल्य}.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The values.}
\begin{enumerate}[resume]
  \item विधि को लागू करें $ f(z) = 1/z^2$: यहाँ $ f $ उसके पास है
        खंभा \emph{पर} एक पूर्णांक, इसलिए तर्क चलाएँ
        सीधे --- एकीकृत करें $ g(z) = \frac{\pi\cot(\pi
        z)}{z^2}$ ऊपर $ C_N $, अभिन्न दिखाओ $\to 0$, और
        गणना करें $\operatorname{Res}(g, 0)$ प्रश्न 2 से.
        निष्कर्ष:
        \[
        2\sum_{n\geq1}\frac1{n^2} = \frac{\pi^2}{3},
        \qquad \zeta(2) = \frac{\pi^2}6 .
        \]
  \item के साथ भी वैसा ही $ g(z) = \frac{\pi\cot(\pi z)}{z^4}$:
        गणना करें $\operatorname{Res}(g, 0)$ और निष्कर्ष निकालें
        $\zeta(4) = \frac{\pi^4}{90}$.
  \item \omterm{def:b3:groups:normal}{सामान्य} पैटर्न स्पष्ट करें: प्रत्येक के लिए $ k \geq 1$,
        $\zeta(2k)$ है $-\frac12$ के गुणांक का गुना
        $ z^{2k-1}$ के लॉरेंट विस्तार में $\pi\cot(\pi
        z)$ पर $0$ --- का एक तर्कसंगत गुणज $\pi^{2k}$.
        गणना करना $\zeta(6)$ प्रश्न 2 के विभाजन को आगे बढ़ाकर
        एक कदम आगे. विधि किस बारे में कहती है
        $\zeta(3)$ --- और यह कुछ क्यों नहीं कहता?
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The partial fraction expansion
of the cotangent.}
\begin{enumerate}[resume]
  \item हल करना $ w \in \C\setminus\Z $ और की विधि लागू करें
        भाग II से $ f(z) = \dfrac{1}{(z - w)(z + w)}$ ---
        उस पर ध्यान देना $\pi\cot(\pi z)f(z)$ अब अतिरिक्त है
        साधारण डंडे पर $\pm w $, किसका \omterm{def:b3:residues:singularities}{अवशेष} में शामिल होना चाहिए
        गिनती. आंशिक भिन्न विस्तार का पता लगाएं
        \[
        \pi\cot(\pi w) = \frac1w +
        \sum_{n\geq1}\frac{2w}{w^2 - n^2},
        \]
        श्रृंखला \omterm{def:b3:groups:normal}{सामान्य} रूप से परिवर्तित हो रही है कॉम्पैक्ट के उपसमुच्चय
        $\C\setminus\Z $.
  \item प्रत्येक पद का विस्तार करके, इस विस्तार से उबरें
        की शक्तियों में $ w $ (विनिमय का औचित्य सिद्ध करें), द
        \emph{वही} प्रश्न 2 के अनुसार लॉरेंट गुणांक ---
        वृत्त \omterm{def:b3:topology:interior}{बंद} हो जाता है: यूलर का सूत्र
        $\sum\frac1{n^2} = \frac{\pi^2}6$ गुणांक है
        का $ w $ कोटैंजेंट के दो चेहरों में. से तुलना करें
        फूरियर-श्रृंखला प्रमाण (वर्ष 2) और ट्रेस प्रमाण
        (\cref{pb:b3:spectral:1}): तीन सिद्धांत, एक
        संख्या.
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Euler's product for the sine.}
प्रश्न 9 का विस्तार लघुगणकीय व्युत्पन्न है
एक अनंत उत्पाद; अब हम यूलर के 1734 गुणनखंडन को सिद्ध करते हैं
ईमानदारी से.
\begin{enumerate}[resume]
  \item के लिए $ N \geq 1$ तय करना $ P_N(z) =
        z\prod_{n=1}^{N}\bigl(1 - \frac{z^2}{n^2}\bigr)$.
        बताते हैं कि $ P_N $ प्रत्येक डिस्क पर समान रूप से अभिसरण होता है
        $\bar D(0, R)$, पूरे समारोह के लिए $ P $ जिसका शून्य
        बिल्कुल पूर्णांक हैं, सभी \omterm{def:b3:groups:simple}{सरल} हैं। \emph{(के लिए $ n
        \geq 2R $ गुणनखंड को इस प्रकार लिखें $\exp\log(1 -
        z^2/n^2)$ के प्रमुख लघुगणक के साथ
        \cref{exo:b3:holomorphic:3}, अवश्यंभावी $\abs{\log(1+u)}
        \leq 2\abs u $ के लिए $\abs u \leq \frac12$ के माध्यम से
        श्रृंखला, और \omterm{def:b3:groups:normal}{सामान्य} रूप से अभिसरण योग का घातांक करें
        लघुगणक का; बहुत सारे शेष कारक
        एक बहुपद हैं. के साथ \omterm{thm:b3:complete:completion}{समापन} करें
        \cref{thm:b3:holomorphic:weierstrassconv}.)}
  \item उसे दिखाओ $\C\setminus\Z $,
        \[
        \frac{P'(z)}{P(z)} = \frac1z +
        \sum_{n\geq1}\frac{2z}{z^2 - n^2} = \pi\cot(\pi z)
        \]
        \emph{(परिमित उत्पादों को अलग करें, पास करें
        उपयोग करने की सीमा
        \cref{thm:b3:holomorphic:weierstrassconv} और
        की शून्\omterm{def:b3:modules:free}{य-मुक्त}ि $ P $ \omterm{def:b3:topology:interior}{बंद} $\Z $, और उद्धरण प्रश्न
        9)}.
  \item बताते हैं कि $ Q = \sin(\pi z)/P(z)$ ए तक फैला हुआ है
        शून्\omterm{def:b3:modules:free}{य-मुक्त} संपूर्ण फ़ंक्शन के साथ $ Q' = 0$, और
        निष्कर्ष \emph{यूलर का उत्पाद}:
        \[
        \sin(\pi z) = \pi z\prod_{n\geq1}
        \Bigl(1 - \frac{z^2}{n^2}\Bigr)
        \qquad (z \in \C) .
        \]
  \item (वालिस, 1655) का मूल्यांकन करें $ z = \frac12$:
        \[
        \frac\pi2 = \prod_{n\geq1}\frac{4n^2}{4n^2 - 1}
        = \lim_{N\to\infty}
        \frac{2\cdot2\cdot4\cdot4\cdots(2N)(2N)}
        {1\cdot3\cdot3\cdot5\cdots(2N-1)(2N+1)} .
        \]
  \item के लिए $\abs z < 1$, के लघुगणक का विस्तार करें
        उत्पाद को \omterm{def:b3:banach:operator}{दोहरी} श्रृंखला के रूप में (उचित ठहराएँ)
        पुनर्व्यवस्था) और पुनर्प्राप्त करें $\zeta(2) =
        \frac{\pi^2}6$ के गुणांक का मिलान करके $ z^3$
        में $\sin(\pi z) = \pi z - \frac{\pi^3}6z^3 + \cdots $
        --- यूलर नंबर का उत्पाद का चेहरा।
\end{enumerate}

\medskip
\noindent\textbf{Part VI --- Sister kernels.} कोटैंजेंट
भाई-बहन हैं; प्रत्येक श्रृंखला के अपने स्वयं के परिवार की कीमतें तय करता है।
\begin{enumerate}[resume]
  \item प्रश्न 9 के विस्तार को पद दर पद अलग करें
        (द्वारा उचित ठहराया गया
        \cref{thm:b3:holomorphic:weierstrassconv}) को
        \omterm{def:b3:groups:normal}{सामान्य} रूप से प्राप्त करें कॉम्पैक्ट कैमरों का $\C\setminus\Z $,
        \[
        \frac{\pi^2}{\sin^2(\pi z)} =
        \sum_{n\in\Z}\frac1{(z - n)^2} .
        \]
  \item पर मूल्यांकन करें $ z = \frac12$:
        $\sum_{m\geq0}\frac1{(2m+1)^2} = \frac{\pi^2}8$;
        वापस पाना $\zeta(2)$ एक बार फिर विभाजित करके
        समता द्वारा पूर्णांक.
  \item दोहराव पहचान सत्यापित करें $\tan\theta =
        \cot\theta - 2\cot(2\theta)$ और निष्कर्ष निकालें
        \[
        \pi\tan(\pi z) =
        \sum_{m\geq0}\frac{8z}{(2m+1)^2 - 4z^2} ,
        \]
        \omterm{def:b3:groups:normal}{सामान्य} रूप से चालू कॉम्पैक्ट कैमरों टाल $\frac12 + \Z $.
  \item चारों ओर विस्तार करें $0$ ($\abs z < \frac12$; फ़ुबिनी
        फिर से): साथ $\lambda(s) =
        \sum_{m\geq0}(2m+1)^{-s}$,
        \[
        \pi\tan(\pi z) =
        \sum_{k\geq0}8\cdot4^k\,\lambda(2k+2)\,z^{2k+1} ;
        \]
        साथ तुलना करें $\tan u = u + \frac{u^3}3 + O(u^5)$ ठीक करने के लिए $\lambda(2) = \frac{\pi^2}8$ और पाने के लिए
        $\lambda(4) = \frac{\pi^4}{96}$, फिर क्रॉस-चेक करें
        $\zeta(4) = \frac{\pi^4}{90}$ के जरिए $\lambda(4) = (1 -
        2^{-4})\,\zeta(4)$.
  \item सत्यापित करें $\frac1{\sin\theta} = \cot\frac\theta2 -
        \cot\theta $ और निष्कर्ष निकालें
        \[
        \frac{\pi}{\sin(\pi z)} = \frac1z +
        \sum_{n\geq1}(-1)^n\,\frac{2z}{z^2 - n^2} .
        \]
        के सामने चिह्नों की जाँच करें \omterm{def:b3:residues:singularities}{अवशेष} का
        $\pi/\sin(\pi z)$ पूर्णांकों पर.
  \item का गुणांक पढ़ें $ z $: $\eta(2) =
        \sum_{n\geq1}\frac{(-1)^{n-1}}{n^2} =
        \frac{\pi^2}{12}$, और संगति की पुष्टि करें
        $\eta(2) = (1 - 2^{1-2})\,\zeta(2)$.
  \item (अंतिम) प्रश्न 20 के विस्तार का मूल्यांकन करें $ z =
        \frac12$ और निष्कर्ष निकालें \emph{लीबनिज़ का सूत्र}
        \[
        \frac\pi4 = 1 - \frac13 + \frac15 - \frac17 +
        \cdots
        \]
        एक संक्षिप्त अनुच्छेद के साथ समाप्त करें: प्रति कर्नेल
        अंकगणित --- किस कर्नेल की कीमत किस परिवार से है
        श्रृंखला, और वे सभी सं\omterm{def:b3:galois:constructible}{रचनात्मक} रूप से अंधे क्यों हैं
        को $\zeta(3)$.

\end{enumerate}
\medskip
\noindent\textbf{Part VII --- The full price list: Bernoulli
numbers.}
\begin{enumerate}[resume]
  \item के आंशिक अंश विस्तार को संयोजित करें $\pi
        z\cot(\pi z)$ के जनरेटिंग फ़ंक्शन के साथ
        बर्नौली संख्या ($\frac{w}{\eu^w - 1} =
        \sum_n\frac{B_n}{n!}w^n $,
        \cref{pb:b3:holomorphic:1}, भाग VI): से
        \[
        \pi z\cot(\pi z) = \iu\pi z +
        \frac{2\iu\pi z}{\eu^{2\iu\pi z} - 1}
        \]
        (पहले इस पहचान को सिद्ध करें), \omterm{def:b3:topology:interior}{बंद} फॉर्म निकालें
        \[
        \zeta(2k) = (-1)^{k+1}\,
        \frac{(2\pi)^{2k}\,B_{2k}}{2\,(2k)!}
        \qquad (k \geq 1).
        \]
  \item के विरुद्ध सूत्र सत्यापित करें $ B_2 = \frac16$, $ B_4 =
        -\frac1{30}$, $ B_6 = \frac1{42}$: वापस पाना $\zeta(2) =
        \frac{\pi^2}6$, $\zeta(4) = \frac{\pi^4}{90}$, और
        गणना करें $\zeta(6) = \frac{\pi^6}{945}$.
  \item (यूलर की पुनरावृत्ति) के दोनों पक्षों का विस्तार करें
        $\bigl(z\cot z\bigr)' = \cot z - z(1 + \cot^2z)$ ---
        या सिद्ध करने के लिए कोटैंजेंट श्रृंखला को सीधे वर्गाकार करें
        \[
        \Bigl(k + \frac12\Bigr)\zeta(2k) =
        \sum_{j=1}^{k-1}\zeta(2j)\,\zeta(2k - 2j)
        \qquad (k \geq 2),
        \]
        और इसकी गणना की जाँच करें $\zeta(4)$ से $\zeta(2)$ और
        $\zeta(6)$ से $\zeta(2), \zeta(4)$ --- सब सम
        एकल बीज से जीटा मान $\frac{\pi^2}6$,
        बिना किसी नए एकीकरण के.
\end{enumerate}
\end{problem}
