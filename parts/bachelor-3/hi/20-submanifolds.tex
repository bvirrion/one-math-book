\chapter{\texorpdfstring{$\R^n$}{Rn} के उपबहुविध}
\label{ch:b3:submanifolds}

गोले, तोरी, घूर्णन समूह: के प्राकृतिक आवास
ज्यामिति और यांत्रिकी सदिश स्थान नहीं हैं लेकिन \emph{घुमावदार
ऐसे सेट जो करीब से सपाट दिखते हैं}. यह अध्याय वह वाक्यांश देता है
एक सटीक अर्थ --- के उपमान $\R^n $ --- और
उन पर काम करने के लिए कैलकुलस। बुनियाद उलटा है
फ़ंक्शन प्रमेय, यहां बानाच निश्चित बिंदु द्वारा सिद्ध किया गया है;
बाकी सब कुछ निर्देशांक का परिवर्तन है: चार समकक्ष
एक सबमैनिफोल्ड का विवरण (स्थानीय सीधाकरण, स्तर
सेट, ग्राफ़, पैरामीट्रिज़ेशन), स्पर्शरेखा रिक्त स्थान, और
द्वारा प्रतिबंधित अनुकूलन \omterm{thm:b3:submanifolds:lagrange}{लैग्रेंज गुणक} --- जो,
एक अलग प्रदर्शन के रूप में, वर्णक्रमीय प्रमेय को पुनः सिद्ध करता है
ज्यामिति की तीन रेखाओं में सममित मैट्रिक्स के लिए। द
सप्ताहांत समस्या रोटेशन समूह का निर्माण करती है $ SO(3)$ और उसके
चतुर्धातुक दोहरा आवरण: बीजगणित (\cref{ch:b3:groups}'एस
$ Q_8$ बड़े होकर) ज्यामिति से मिलना।

\section{व्युत्क्रम फलन प्रमेय}

\begin{theorem}[व्युत्क्रम फलन प्रमेय]
\label{thm:b3:submanifolds:ift}
होने देना $ U \subseteq \R^n $ खुले रहो, $ f \colon U \to \R^n $ का
\omterm{def:b3:groups:action}{कक्षा} $\mathcal C^1$, और $ a \in U $ साथ $ Df(a)$ उलटा।
फिर हैं खुले सेट $ V \ni a $, $ W \ni f(a)$ ऐसा है कि
$ f \colon V \to W $ के साथ एक आक्षेप है $\mathcal C^1$ उलटा,
और\index{व्युत्क्रम फलन प्रमेय}
\[
D(f^{-1})(y) = \bigl(Df(f^{-1}(y))\bigr)^{-1}
\qquad (y \in W).
\]
अगर $ f $ है $\mathcal C^k $, ऐसा ही है $ f^{-1}$.
\end{theorem}

\begin{proof}
\omterm{def:b3:groups:normal}{सामान्य}ीकृत करें: प्रतिस्थापित करना $ f $ द्वारा $ x \mapsto Df(a)^{-1}\bigl(f(a +
x) - f(a)\bigr)$, हम मान सकते हैं $ a = 0$, $ f(0) = 0$, $ Df(0) =
I $ (\omterm{def:b3:groups:normal}{सामान्य} कथन के साथ रचना करके अनुसरण किया जाता है
आक्षेपों को एफ़िन करें)। लिखना $ f(x) = x + g(x)$: $ Dg(0) = 0$, और तक निरंतरता का $ Dg $ चुनना $ r > 0$ साथ
$\vertiii{Dg(x)} \leq \frac12$ पर $\bar B(0, r)$; मतलब
मूल्य असमानता देता है $\norm{g(x) - g(x')} \leq \frac12
\norm{x - x'}$ वहाँ।

\emph{ए पर वस्तुनिष्ठता \omterm{def:b3:topology:topology}{अड़ोस-पड़ोस}.} के लिए $ y \in B(0,
\frac r2)$, सुलझाना $ f(x) = y $ का अर्थ है एक निश्चित बिंदु खोजना
$\Phi_y(x) = y - g(x)$; $\Phi_y $ एमएपीएस $\bar B(0, r)$ में
स्वयं ($\norm{\Phi_y(x)} \leq \norm y + \frac12\norm x \leq
r $) और है $\frac12$-लिप्सचिट्ज़: बानाच
(\cref{thm:b3:complete:banach}) एक अनोखा समाधान देता है
$ x = \varphi(y) \in \bar B(0,r)$. इसके अतिरिक्त $ f $ इंजेक्शन है
पर $\bar B(0,r)$:
\[
\norm{f(x) - f(x')} \geq \norm{x - x'} - \norm{g(x) - g(x')}
\geq \tfrac12\norm{x - x'} .
\tag{$*$}
\]
तय करना $ W = B(0, \frac r2)$ और $ V = f^{-1}(W)\cap B(0, r)$:
खुला (निरंतरता), साथ $ f \colon V \to W $ विशेषण।

\emph{निरंतरता और व्युत्क्रम की भिन्नता।} $(*)$
कहते हैं $\varphi = f^{-1}$ है $2$-लिप्सचिट्ज़. हल करना $ y_0 = f(x_0)
\in W $; की व्युत्क्रमणीयता $ A = Df(x_0)$ (इसकी दूरी $ I $
है $\leq \frac12$: \omterm{prop:b3:banach:neumann}{न्यूमैन},
\cref{prop:b3:banach:neumann}) और की भिन्नता $ f $
इसके लिए दें $ y = f(x)$ पास में $ y_0$:
\[
\varphi(y) - \varphi(y_0) - A^{-1}(y - y_0)
= -A^{-1}\bigl(f(x) - f(x_0) - A(x - x_0)\bigr)
= -A^{-1}\,o(\norm{x - x_0}) = o(\norm{y - y_0}),
\]
का उपयोग करते हुए $(*)$ रूपान्तरण करने के लिए $\norm{x - x_0} \leq 2\norm{y - y_0}$:
$\varphi $ पर भिन्न है $ y_0$ उलटे के साथ
अंतर. निरंतरता का $ y \mapsto D\varphi(y) =
Df(\varphi(y))^{-1}$: की रचना \omterm{def:b3:topology:continuity}{निरंतर} मानचित्र
(उलटा है \omterm{def:b3:topology:continuity}{निरंतर},
\cref{prop:b3:banach:neumann}): $\varphi \in \mathcal C^1$;
बूटस्ट्रैपिंग एक ही फार्मूला देता है $\mathcal C^k $.
\end{proof}

\begin{theorem}[निहित कार्य प्रमेय]
\label{thm:b3:submanifolds:implicit}
होने देना $ F \colon U \subseteq \R^p\times\R^q \to \R^q $ होना
$\mathcal C^1$ पास में $(a, b)$, $ F(a,b) = 0$, और मान लीजिए
आंशिक अंतर $ D_yF(a,b) \in \mathcal L(\R^q)$ है
उलटा। फिर हैं पड़ोस $ A \ni a $, $ B \ni b $
और ए $\mathcal C^1$ नक्शा $\psi \colon A \to B $ साथ
\[
\bigl\{(x, y)\in A\times B : F(x,y) = 0\bigr\}
= \{(x, \psi(x)) : x \in A\},
\]
और $ D\psi(x) = -D_yF(x, \psi(x))^{-1}\,D_xF(x,
\psi(x))$.\index{निहित कार्य प्रमेय}
\end{theorem}

\begin{proof}
आवेदन करना \cref{thm:b3:submanifolds:ift} को $\Theta(x, y) = (x,
F(x,y))$: इसका अंतर है $(a,b)$, ब्लॉक-त्रिकोणीय के साथ
उल्टे विकर्ण ब्लॉक $ I $ और $ D_yF $, उलटा है. द
स्थानीय व्युत्क्रम का रूप है $\Theta^{-1}(x, z) = (x, h(x,
z))$; तय करना $\psi(x) = h(x, 0)$: तब $ F(x, y) = 0$ आईएफएफ
$\Theta(x,y) = (x, 0)$ आईएफएफ $ y = \psi(x)$, स्थानीय स्तर पर। द
सूत्र: अंतर करें $ F(x, \psi(x)) = 0$ श्रृंखला द्वारा
नियम.
\end{proof}

\section{सबमैनिफोल्ड्स: चार परिभाषाएँ}

\begin{theorem}[समतुल्य लक्षण वर्णन]
\label{thm:b3:submanifolds:characterizations}
होने देना $ M \subseteq \R^n $, $ d \in \{0, \dots, n\}$, और $ k \geq
1$. प्रत्येक बिंदु के लिए निम्नलिखित समतुल्य हैं $ a \in M $
(और $ M $ एक है \emph{$ d $-वर्ग का आयामी उपमान
$\mathcal C^k $}\index{$ d $-वर्ग का आयामी उपमान
$\mathcal C^k $} यदि वे हर बात पर पकड़ रखते हैं $ a
\in M $):
\begin{enumerate}
  \item (\emph{सीधा}) वहां एक है $\mathcal C^k $
        भिन्नता $\Phi $ एक खुले से $\Omega \ni a $
        एक खुले पर $\Omega' \subseteq \R^n $ साथ
        \[
        \Phi(M\cap\Omega) = \Omega' \cap
        \bigl(\R^d\times\{0\}\bigr).
        \]
  \item (\emph{लेवल सेट}) वहां एक है $\mathcal C^k $
        \emph{डुबकी} $ F \colon \Omega \to \R^{n-d}$
        (अर्थात्\ $ DF(x)$ विशेषण) एक खुले पर $\Omega \ni a $
        साथ $ M\cap\Omega = F^{-1}(0)$.
  \item (\emph{ग्राफ़}) निर्देशांक क्रमपरिवर्तित करने तक, $ M $ है
        स्थानीय स्तर पर ए का ग्राफ $\mathcal C^k $ नक्शा $\psi
        \colon A \subseteq \R^d \to \R^{n-d}$.
  \item (\emph{पैरामीट्रिजेशन}) वहां एक है $\mathcal C^k $
        \emph{विसर्जन} $\varphi \colon A \subseteq \R^d \to
        \R^n $ ($ D\varphi(u)$ इंजेक्शन) के साथ $ A $ खुला,
        $\varphi $ ए \omterm{def:b3:topology:continuity}{होमियोमोर्फिज्म} से $ A $ पर $ M \cap
        \Omega $ कुछ के लिए खुला $\Omega \ni a $.
\end{enumerate}
\end{theorem}

\begin{proof}
(1)$\Rightarrow $(2): $ F = (\Phi_{d+1}, \dots, \Phi_n)$ (अंतिम
के निर्देशांक $\Phi $): एक विसर्जन ($ D\Phi $ उलटा)।
(2)$\Rightarrow $(3): $ DF(a)$ विशेषण: कुछ $ q \times q $
जैकोबियन का नाबालिग उलटा है ($ q = n - d $); के बाद
निर्देशांक क्रमपरिवर्तित करना, $ D_yF(a)$ उलटा है, और
निहित कार्य प्रमेय (\cref{thm:b3:submanifolds:implicit})
अभिव्यक्त करता है $ M $ स्थानीय रूप से एक ग्राफ़ के रूप में $ y = \psi(x)$.
(3)$\Rightarrow $(4): $\varphi(x) = (x, \psi(x))$: एक
विसर्जन (विभेदक) $\bigl(\begin{smallmatrix}I\\
D\psi\end{smallmatrix}\bigr)$ इंजेक्शन), ए \omterm{def:b3:topology:continuity}{होमियोमोर्फिज्म} पर
ग्राफ़ (उलटा: प्रक्षेपण, \omterm{def:b3:topology:continuity}{निरंतर}).
(4)$\Rightarrow $(1): चलो $\varphi(u_0) = a $; \omterm{def:b3:complete:complete}{पूरा}
$\operatorname{im}D\varphi(u_0)$ एक पूरक द्वारा $ E $
($\dim E = n - d $) और परिभाषित करें $\Theta(u, v) = \varphi(u) + v $
पर $ A\times E $: $ D\Theta(u_0, 0)$ विशेषण है (छवि
शामिल है $\operatorname{im}D\varphi(u_0)$ और $ E $), इसलिए
$\Theta $ एक स्थानीय भिन्नता है
(\cref{thm:b3:submanifolds:ift}); यह उलटा है $\Phi $
सीधा करता है: निकट $ a $, के अंक $ M $ बिल्कुल वही हैं
$\varphi(u) = \Theta(u, 0)$ ---इसके लिए, \omterm{def:b3:topology:continuity}{होमियोमोर्फिज्म}
(4) में परिकल्पना इसकी गारंटी देती है $ M\cap\Omega $, छोटे के लिए
$\Omega $, इसमें कोई अन्य शीट नहीं है ($\varphi(u') + v = m \in
M $ के करीब $ a $ छोटे के साथ $ v \neq 0$ बहिष्कृत किया जाना चाहिए:
$ m = \varphi(u'')$ कुछ के लिए $ u''$ पास में $ u_0$ से
\omterm{def:b3:topology:continuity}{होमियोमोर्फिज्म} संपत्ति, और स्थानीय इंजेक्शन $\Theta $
बल $ v = 0$). तब $\Phi(M\cap\Omega) = (A\times\{0\})
\cap\Phi(\Omega)$ सिकुड़ने तक.
\end{proof}

\begin{example}\label{ex:b3:submanifolds:examples}
गोला $ S^{n-1} = \{\norm x^2 = 1\}$: का स्तर सेट
जलमग्न होना $ F(x) = \norm x_2^2 - 1$ पर
$\R^n\setminus\{0\}$ ($ DF(x) = 2x^{\mathsf T} \ne 0$):
ए $\mathcal C^\infty $ आयाम का उपमान $ n - 1$. द
टोरस में $\R^3$: का स्तर सेट $\bigl(\sqrt{x^2 + y^2} -
R\bigr)^2 + z^2 - r^2$ ($0 < r < R $). शंकु $\{x^2 + y^2 =
z^2\}$ है \emph{नहीं} एक सबमैनिफोल्ड पर $0$
(\cref{exo:b3:submanifolds:1}). मैट्रिक्स समूह: $ SL_n $ और
$ O_n $ के उपमान हैं $ M_n(\R)$
(\cref{exo:b3:submanifolds:5,exo:b3:submanifolds:6}) --- द
झूठ सिद्धांत का प्रारंभिक बिंदु.
\end{example}

\section{स्पर्शरेखा रिक्त स्थान}

\begin{definition}\label{def:b3:submanifolds:tangent}
होने देना $ M $ एक हो $ d $-सबमैनिफोल्ड और $ a \in M $. \emph{स्पर्शरेखा
अंतरिक्ष}\index{स्पर्शरेखा
अंतरिक्ष} $ T_aM $ वेग का समुच्चय है
वेक्टर $\gamma'(0)$ का $\mathcal C^1$ घटता $\gamma \colon
\intoo{-\varepsilon}\varepsilon \to M $ साथ $\gamma(0) = a $.
\end{definition}

\begin{proposition}\label{prop:b3:submanifolds:tangent}
$ T_aM $ एक है $ d $-आयामी वेक्टर उप-स्थान $\R^n $, और:
\begin{enumerate}
  \item अगर $ M = F^{-1}(0)$ स्थानीय स्तर पर साथ $ F $ एक विसर्जन:
        $ T_aM = \ker DF(a)$;
  \item अगर $ M $ विसर्जन द्वारा पैरामीट्रिज्ड किया जाता है $\varphi $
        ($\varphi(u_0) = a $): $ T_aM =
        \operatorname{im}D\varphi(u_0)$.
\end{enumerate}
\end{proposition}

\begin{proof}
में वक्र $ M $ संतुष्ट $ F(\gamma(t)) = 0$; श्रृंखला नियम पर
$0$ देता है $ DF(a)\gamma'(0) = 0$: $ T_aM \subseteq \ker DF(a)$.
इसके विपरीत, सीधा करना (\cref{thm:b3:submanifolds:%
characterizations}(1)) लाइनों का परिवहन करता है
$\R^d\times\{0\}$ के घटता के लिए $ M $: ए का प्रत्येक वेक्टर
$ d $-आयामी उपस्थान का एहसास होता है; आयामों की तुलना करना
($\dim\ker DF(a) = n - (n - d) = d $) (1) में समानता को बल देता है,
और वही परिवहन तर्क देता है (2) ($ D\varphi(u_0)$
में सीधी रेखाओं पर लागू किया गया $ A $; आयाम फिर से)।
\end{proof}

\begin{theorem}[लैग्रेंज गुणक]
\label{thm:b3:submanifolds:lagrange}
होने देना $ M = F^{-1}(0)$ साथ $ F = (F_1, \dots, F_q) \colon \Omega
\to \R^q $ ए $\mathcal C^1$ जलमग्न होना, और जाने देना $ f \colon
\Omega \to \R $ होना $\mathcal C^1$. यदि प्रतिबंध
$ f\restriction_M $ पर एक स्थानीय चरम सीमा है $ a \in M $, फिर
अद्वितीय यथार्थ हैं $\lambda_1, \dots, \lambda_q $
(\emph{\omterm{thm:b3:submanifolds:lagrange}{लैग्रेंज गुणक}}) साथ\index{लैग्रेंज गुणक}
\[
\nabla f(a) = \lambda_1\nabla F_1(a) + \dots +
\lambda_q\nabla F_q(a) .
\]
\end{theorem}

\begin{proof}
प्रत्येक वक्र के लिए $\gamma $ में $ M $ के माध्यम से $ a $: $ t \mapsto
f(\gamma(t))$ पर एक स्थानीय चरम सीमा है $0$, इसलिए $0 =
\frac{\dd}{\dd t}f(\gamma(t))\big|_0 = \langle\nabla f(a),
\gamma'(0)\rangle $: $\nabla f(a) \perp T_aM = \ker DF(a)$
(\cref{prop:b3:submanifolds:tangent}). अब $\ker DF(a)^\perp
= \operatorname{im}DF(a)^{\mathsf T}$ (रैंक/ऑर्थोगोनैलिटी
वर्ष 2 की पहचान, या \cref{exo:b3:hilbert:8} परिमित में
आयाम: $(\ker T)^\perp = \operatorname{im}T^*$), जो है
ग्रेडिएंट्स द्वारा फैलाया गया $\nabla F_i(a)$ --- स्वतंत्र, जैसे
$ DF(a)$ विशेषण है: गुणक मौजूद हैं और अद्वितीय हैं।
\end{proof}

\begin{example}[वर्णक्रमीय प्रमेय, ज्यामितीय रूप से]
\label{ex:b3:submanifolds:rayleigh}
होने देना $ A $ एक वास्तविक सममित बनें $ n\times n $ मैट्रिक्स और अधिकतम
$ f(x) = \langle Ax, x\rangle $ गोले पर $ S^{n-1}$
(कॉम्पैक्ट: कुछ पर अधिकतम प्राप्त हो जाता है $ v_1$). लैग्रेंज
के साथ $ F(x) = \norm x^2 - 1$: $\nabla f = 2Ax $ और $\nabla F =
2x $ देना $ Av_1 = \lambda_1v_1$ --- एक eigenvector, के साथ
$\lambda_1 = \max_{S^{n-1}}\langle Ax, x\rangle $. प्रतिबंध लगाना
$ A $ को $ v_1^\perp $ (अपरिवर्तनीय: $\langle Av, v_1\rangle =
\langle v, Av_1\rangle = \lambda_1\langle v, v_1\rangle = 0$)
और पुनरावृत्त: आइजेनवेक्टरों का एक लंबात्मक \omterm{def:b3:topology:basis}{आधार}। द
वर्ष 2 का वर्णक्रमीय प्रमेय, शुद्ध अनुकूलन द्वारा पुनः सिद्ध किया गया
--- और उसी तर्क की अनंत-आयामी छाया
साबित हुआ \cref{lem:b3:spectral:existence}.
\end{example}

\begin{method}\label{met:b3:submanifolds:method}
यह साबित करने के लिए कि एक सेट एक सबमैनिफोल्ड है: इसे स्थानीय रूप से प्रदर्शित करें
$ F^{-1}(0)$ साथ $ DF $ विशेषण \emph{सेट पर} (सबसे ज्यादा
\omterm{def:b3:groups:normal}{सामान्य} मार्ग), या ग्राफ़ के रूप में। इसके आयाम की गणना करने के लिए और
\omterm{def:b3:submanifolds:tangent}{स्पर्शरेखा स्थान}: $ d = n - q $ और $ T_a = \ker DF(a)$. को
इस पर अनुकूलन करें: लैग्रेंज --- हमेशा कॉम्पैक्टनेस की जांच करें (या
ज़बरदस्ती) पहले, तो एक चरम मौजूद है जिसके लिए प्रमेय
लागू कर सकते हैं, और याद रखें कि गुणक समीकरण ही है
\emph{ज़रूरी}: सभी महत्वपूर्ण बिंदु एकत्र करें, फिर तुलना करें
मूल्य. मैट्रिक्स समूहों के लिए, वक्रों को अलग करें
पहचान की पहचान स्पर्शरेखा रिक्त स्थान.
\end{method}

\begin{exercise}[$\star\star\star $]\label{exo:b3:submanifolds:9}
(हैडमार्ड की असमानता) के लिए $ M \in GL_n(\R)$ स्तंभों के साथ
$ c_1, \dots, c_n $:
\[
\abs{\det M} \;\leq\; \prod_{i=1}^n\norm{c_i}_2 ,
\]
समानता के साथ यदि कॉलम ऑर्थोगोनल हैं। \emph{(कम करें
मानक के स्तंभों के लिए $1$ स्केलिंग द्वारा; अधिकतम $\det $ पर
कॉम्पैक्ट गोले का उत्पाद $(S^{n-1})^n $; अधिकतम सीमा पर,
प्रत्येक कॉलम में लैग्रेंज अलग-अलग देता है $\nabla_{c_i}\det =
\lambda_ic_i $, और $\nabla_{c_i}\det $ है $ i $-वें कॉलम का
$\operatorname{com}(M)$: निष्कर्ष निकालें $ M^{\mathsf T}M $ विकर्ण,

एक समान्तर चतुर्भुज का आयतन अधिकतम उसके किनारे का गुणनफल होता है
लंबाई.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:submanifolds:11}

\begin{exercise}[$\star\star $]\label{exo:b3:submanifolds:12}

\section{संकट: \texorpdfstring{$ SO(3)$}{SO(3)} और
चतुर्भुज}

\begin{problem}[{सप्ताहांत समस्या---रोटेशन, समूह
$ S^3$, और दोहरा आवरण}]\label{pb:b3:submanifolds:1}
चतुर्भुज $\mathbb H = \{t + x\mathrm i + y\mathrm j +
z\mathrm k\}$ --- बीजगणित जिसका इकाई समूह शामिल है
\cref{pb:b3:groups:1}'एस $ Q_8$ --- पैरामीट्रिज
त्रि-आयामी घुमाव दो बार: मानचित्र ``संयुग्मित
एक इकाई क्वाटरनियन द्वारा'' एक विशेषण रूपवाद है $ S^3 \to
SO(3)$ कर्नेल के साथ $\{\pm1\}$. हम सब कुछ बनाते हैं.
स्मरण/परिभाषित करें: गुणन है $\R $-बिलिनियर के साथ
$\mathrm i^2 = \mathrm j^2 = \mathrm k^2 =
\mathrm{ijk} = -1$; का संयुग्म $ q = t + x\mathrm i +
y\mathrm j + z\mathrm k $ है $\bar q = t - x\mathrm i -
y\mathrm j - z\mathrm k $; $ N(q) = q\bar q = t^2 + x^2 + y^2
+ z^2$.

\medskip
\noindent\textbf{Part I --- The algebra $\mathbb H $ and the
group $ S^3$.}
\begin{enumerate}
  \item सत्यापित करें कि $\mathbb H $ एक सहयोगी है
        $\R $-बीजगणित के साथ \omterm{ex:b3:groups:actions}{केंद्र} $\R $, वह $\overline{pq} =
        \bar q\,\bar p $, ओर वो $ N(pq) = N(p)N(q)$
        \emph{(एक साफ़ मार्ग: \omterm{def:b3:representations:rep}{प्रतिनिधित्व} करें $ q $ के रूप में
        $2\times2$ जटिल मैट्रिक्स
        $\bigl(\begin{smallmatrix}\alpha & \beta\\
        -\bar\beta & \bar\alpha\end{smallmatrix}\bigr)$, $ q =
        \alpha + \beta\mathrm j $, और उपयोग करें $\det $)}.
  \item यह निष्कर्ष निकालें कि प्रत्येक $ q \neq 0$ उलटा है ($ q^{-1} =
        \bar q/N(q)$): $\mathbb H $ एक (गैरअनुवांशिक) है
        फ़ील्ड, और $ S^3 = \{N(q) = 1\}$ एक समूह है ---
        और ए कॉम्पैक्ट $3$-सबमैनिफोल्ड का $\R^4$
        (\cref{ex:b3:submanifolds:examples}).
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The rotation morphism.} पहचान करना
$\R^3$ साथ \emph{शुद्ध चतुर्भुज} $ P = \{x\mathrm i +
y\mathrm j + z\mathrm k\}$, और के लिए $ q \in S^3$ परिभाषित करना
$\rho_q(v) = q\,v\,\bar q $.
\begin{enumerate}[resume]
  \item बताते हैं कि $\rho_q $ एमएपीएस $ P $ को $ P $ \emph{(शुद्ध
        चतुर्भुज वे हैं जिनके साथ हैं $\bar v = -v $)}, है
        $\R $-रैखिक, \omterm{def:b3:rings:ideal}{आदर्श} को सुरक्षित रखता है, और वह $\rho
        \colon q \mapsto \rho_q $ एक समूह रूपवाद है $ S^3
        \to O(3)$.
  \item कर्नेल की गणना करें: $\rho_q = \mathrm{id}$ आईएफएफ $ q $
        के साथ आवागमन करता है $\mathrm i, \mathrm j, \mathrm k $ आईएफएफ
        $ q \in \R\cap S^3 = \{\pm1\}$.
  \item लिखना $ q = \cos\frac\theta2 + \sin\frac\theta2\,u $
        साथ $ u \in P $, $ N(u) = 1$ (ऐसा हमेशा क्यों होता है
        के लिए संभव है $ q \in S^3$?). बताते हैं कि $\rho_q $ ठीक करता है
        $ u $ और, विमान पर $ u^\perp\cap P $, के रूप में कार्य करता है
        कोण का घूमना $\theta $ \emph{(गणना करें
        $\rho_q(w)$ के लिए $ w \perp u $ का उपयोग करते हुए $ uw = -wu $ के लिए
        ऑर्थोगोनल शुद्ध इकाइयाँ --- इस पहचान को सिद्ध करें
        गुणन सारणी, या से $ uw + wu =
        -2\langle u, w\rangle $)}.
  \item निष्कर्ष: $\operatorname{im}\rho \subseteq SO(3)$
        (प्रत्येक $\rho_q $ अक्ष और कोण के साथ एक घूर्णन है
        परिकलित --- निर्धारक $+1$ द्वारा निरंतरता का $ q
        \mapsto \det\rho_q $ पर \omterm{def:b3:topology:connected}{जुड़े हुए} $ S^3$, या
        सीधे), और $\rho $ है \emph{पर} $ SO(3)$: हर
        का घूर्णन $\R^3$ एक अक्ष है (साबित करें: एक वास्तविक
        $3\times3$ ऑर्थोगोनल मैट्रिक्स के साथ $\det = 1$ है
        eigenvalue $1$ --- विशेषता पर विचार करें
        बहुपद) और इसलिए कुछ है $\rho_q $. सारांश:
        \[
        SO(3) \;\cong\; S^3/\{\pm 1\} .
        \]
\end{enumerate}

\medskip
\noindent\textbf{Part III --- $ SO(3)$ as a submanifold;
Rodrigues.}
\begin{enumerate}[resume]
  \item बताते हैं कि $ SO(3)$ एक है कॉम्पैक्ट $3$-आयामी
        का सबमैनिफोल्ड $ M_3(\R)$ साथ $ T_ISO(3) =$
        एंटीसिमेट्रिक मैट्रिसेस
        (\cref{exo:b3:submanifolds:5}; निर्धारक
        स्थिति घटकों के एक संघ का चयन करती है)।
  \item एंटीसिमेट्रिक मैट्रिक्स के लिए $ A_u $ के साथ जुड़े
        $ u \in \R^3$ ($ A_uv = u\wedge v $, क्रॉस
        उत्पाद), सिद्ध करें \emph{रोड्रिग्स का सूत्र}:
        \[
        \eu^{\theta A_u} = I + \sin\theta\,A_u + (1 -
        \cos\theta)\,A_u^2 \qquad (\norm u = 1)
        \]
        \emph{(से $ A_u^3 = -A_u $: घातांक को विभाजित करें
        साथ श्रृंखला $ A_u $की शक्तियां)}, और इसे इस रूप में पहचानें
        अक्ष का घूमना $ u $ और कोण $\theta $. उसका निष्कर्ष निकालो
        $\exp $ एंटीसिमेट्रिक मैट्रिक्स को मैप करता है \emph{पर}
        $ SO(3)$.
  \item दो पैरामीट्रिज़ेशनों को संबंधित करें: उसे दिखाएँ $ t
        \mapsto \rho_{q(t)}$ साथ $ q(t) = \cos\frac t2 +
        \sin\frac t2\,u $ का एक-पैरामीटर समूह है
        घूर्णन जिसका व्युत्पन्न पर $ t = 0$ है $ A_u $
        --- चतुष्कोणीय और मैट्रिक्स घातांक बताओ
        एक ही कहानी क्रमशः आधी और पूरी गति से।
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The double cover, felt.}
\begin{enumerate}[resume]
  \item दिखाओ कि पथ $ q(t) = \cos\frac t2 + \sin\frac
        t2\,\mathrm k $, $ t \in \intcc0{2\pi}$, एक लूप है
        $ SO(3)$ (इसकी छवि $\rho_{q(t)}$ पर लौटता है
        पहचान) किसकी चतुष्कोणीय लिफ्ट है \emph{नहीं} ए
        लूप: $ q(2\pi) = -q(0)$. जारी है $ t = 4\pi $
        लिफ्ट \omterm{def:b3:topology:interior}{बंद} कर देता है. एक छोटे पैराग्राफ में बताएं क्या
        यह कहता है: ए $2\pi $ रोटेशन नहीं है लगातार
        पूर्ववत् करने योग्य जबकि ए $4\pi $ रोटेशन है (बेल्ट
        चाल), क्योंकि $ SO(3)$इसके लूप्स का पता लगाया जाता है
        दोहरा आवरण $ S^3$.
  \item व्यावहारिक लाभांश भी निकालें: की संरचना
        घूर्णन = चतुर्भुजों का गुणन
        ($4$ गुणन के बजाय डेटा का मूल्य $9$,
        रूढ़िवादिता से कोई विचलन नहीं) --- पर सत्यापित करें
        दो चौथाई-मोड़ों की रचना $\mathrm i $
        और $\mathrm j $: के अक्ष और कोण की गणना करें
        उत्पाद.
\end{enumerate}

\medskip
\noindent\textbf{Part V --- The explicit matrix:
Euler--Rodrigues.} लिखना $ q = a + b\mathrm i + c\mathrm j +
d\mathrm k \in S^3$, ताकि $ a^2 + b^2 + c^2 + d^2 = 1$.
\begin{enumerate}[resume]
  \item गणना करना $\rho_q(\mathrm i)$ से पूर्णतः
        गुणन सारणी; फिर प्राप्त करें $\rho_q(\mathrm
        j)$ और $\rho_q(\mathrm k)$ चक्रीय द्वारा
        प्रतिस्थापन $\mathrm i \to \mathrm j \to \mathrm k
        \to \mathrm i $, $(b, c, d) \to (c, d, b)$
        \emph{(इसे उचित ठहराएं: साइकिल चलाना $\mathrm i, \mathrm j,
        \mathrm k $ की ऑटोमोर्फिज्म तक फैली हुई है $\mathbb
        H $, क्योंकि परिभाषित संबंध चक्रीय हैं
        सममित)}. निष्कर्ष निकालें कि का मैट्रिक्स $\rho_q $
        \omterm{def:b3:topology:basis}{आधार} में $(\mathrm i, \mathrm j, \mathrm k)$ है
        द \emph{यूलर--रोड्रिग्स मैट्रिक्स}
        \[
        R_q = \begin{pmatrix}
        a^2 + b^2 - c^2 - d^2 & 2(bc - ad) & 2(bd + ac)\\
        2(bc + ad) & a^2 - b^2 + c^2 - d^2 & 2(cd - ab)\\
        2(bd - ac) & 2(cd + ab) & a^2 - b^2 - c^2 + d^2
        \end{pmatrix}.
        \]
  \item (पीछे की ओर घुमाकर पढ़ते हुए) वह दिखाओ
        \[
        \operatorname{tr}R_q = 4a^2 - 1 = 1 + 2\cos\theta,
        \qquad
        \tfrac12\bigl(R_q - R_q^{\mathsf T}\bigr) =
        \sin\theta\,A_u,
        \]
        प्रश्न 5 और 8 के अंकन में। निष्कर्ष निकालें
        एल्गोरिथ्म पुनर्प्राप्ति $\pm q $ एक रोटेशन मैट्रिक्स से
        $ R $: निशान से कोण; से धुरी
        एंटीसिमेट्रिक भाग जब $0 < \theta < \pi $; और जब $\theta = \pi $, पहचान साबित करें और उपयोग करें $ R
        + I = 2\,uu^{\mathsf T}$.
  \item मूल्यांकन करना $ R_q $ प्रश्न 11 के उत्पाद के लिए $ q =
        \frac12(1 + \mathrm i + \mathrm j + \mathrm k)$: ए
        क्रमपरिवर्तन मैट्रिक्स प्रकट होता है. घूर्णन को पहचानें
        और इसमें पाए जाने वाले अक्ष और कोण के साथ सामंजस्य स्थापित करें
        प्रश्न 11.
\end{enumerate}

\medskip
\noindent\textbf{Part VI --- Inside $ S^3$: $ SU(2)$,
\omterm{ex:b3:groups:actions}{संयुग्मता वर्ग}, exponentials.}
\begin{enumerate}[resume]
  \item दिखाएँ कि मैट्रिक्स \omterm{def:b3:representations:rep}{प्रतिनिधित्व} प्रश्न 1 का
        (इसे बुलाओ $\Phi $) एक समूह समरूपता तक सीमित है
        से $ S^3$ पर \emph{विशेष एकात्मक समूह}
        \[
        SU(2) = \bigl\{U \in M_2(\C) : U^*U = I,\ \det U =
        1\bigr\}
        \]
        \emph{(सक्रियता के लिए, समीकरण लिखें
        $ U^{-1} = U^*$ और $\det U = 1$ एक जनरल के लिए
        $2\times2$ जटिल मैट्रिक्स)}.
  \item दिखाएँ कि वास्तविक भाग एक संयुग्मन अपरिवर्तनीय है
        पर $ S^3$ --- $\operatorname{Re}(pq\bar p) =
        \operatorname{Re}q $ सभी के लिए $ p \in S^3$ --- और,
        इसके विपरीत, वह दो इकाई चतुर्भुज समान हैं
        वास्तविक भाग संयुग्मित हैं $ S^3$ \emph{(कम करें
        एक इकाई शुद्ध अक्ष को दूसरे पर ले जाना, कौन सा भाग
        द्वितीय प्रदान करता है)}. का वर्णन करें \omterm{ex:b3:groups:actions}{संयुग्मता वर्ग} का
        $ S^3$ ज्यामितीय रूप से; में अनुवाद करें $ SU(2)$ (स्तर
        ट्रेस के सेट); और प्रोजेक्ट द्वारा $\rho $: दो
        घूर्णन संयुग्मित होते हैं $ SO(3)$ यदि और केवल यदि
        उनका कोण समान है $\theta \in \intcc0\pi $.
  \item परिभाषित करना $\exp $ पर $\mathbb H $ घातांक द्वारा
        शृंखला; का उपयोग करके पूर्ण अभिसरण की जाँच करें
        $\abs{pq} = \abs p\,\abs q $ के लिए $\abs q =
        \sqrt{N(q)}$. शुद्ध इकाई के लिए दिखाएँ $ u $ और $\theta
        \in \R $,
        \[
        \exp(\theta u) = \cos\theta + \sin\theta\,u ,
        \]
        उसका निष्कर्ष निकालो $\exp $ हाइपरप्लेन को मैप करता है $ P $ पर
        $ S^3$, और उसकी जांच करें $\rho_{\exp(su)} =
        \eu^{2sA_u}$: प्रश्न की अर्ध-कोण घटना
        9 फिर से.
  \item शुद्ध चतुर्भुज के लिए $ v, w $ उत्पाद नियम सिद्ध करें
        $ vw = -\langle v, w\rangle + v\wedge w $, इसलिए
        \omterm{def:b3:groups:derived}{कम्यूटेटर} पहचान $ vw - wv = 2\,v\wedge w $; साबित करो
        भी $[A_v, A_w] = A_{v\wedge w}$ मैट्रिक्स के लिए
        प्रश्न 8 का निष्कर्ष निकालें कि का व्युत्पन्न
        $\rho $ पर $1$ मोड़ों के साथ $ t \mapsto \exp(tv)$
        रैखिक समरूपता है $ v \mapsto 2A_v $ से $ P $
        एंटीसिमेट्रिक मैट्रिक्स पर, और वह यह
        चतुर्भुज का परिवहन करता है \omterm{def:b3:groups:derived}{कम्यूटेटर} मैट्रिक्स के लिए
        \omterm{def:b3:groups:derived}{कम्यूटेटर}.
\end{enumerate}

\medskip
\noindent\textbf{Part VII --- Global structure.}
\begin{enumerate}[resume]
  \item (नहीं \omterm{def:b3:topology:continuity}{निरंतर} अनुभाग) मान लीजिए $ s \colon SO(3) \to
        S^3$ है \omterm{def:b3:topology:continuity}{निरंतर} साथ $\rho \circ s =
        \operatorname{id}$. लूप के लिए $ R(t) =
        \rho_{q(t)}$ प्रश्न 10 का, सेट $\varepsilon(t) =
        s(R(t))\,q(t)^{-1}$ के लिए $ t \in \intcc0{2\pi}$. बताते हैं कि $\varepsilon $ है \omterm{def:b3:topology:continuity}{निरंतर} मूल्यों के साथ
        $\{\pm1\}$, और एक विरोधाभास निकालें: कोई नहीं है
        \omterm{def:b3:topology:continuity}{निरंतर} एक इकाई क्वाटरनियन की वैश्विक पसंद
        प्रत्येक घूर्णन का \omterm{def:b3:representations:rep}{प्रतिनिधित्व} करना।
  \item (गेंद मॉडल) चलो $\bar B \subseteq \R^3$ हो
        त्रिज्या की \omterm{def:b3:topology:interior}{बंद} गेंद $\pi $ और $ E(v) = \eu^{A_v}$,
        साथ $ E(0) = I $. बताते हैं कि $ E $ एमएपीएस $\bar B $ पर
        $ SO(3)$, खुली गेंद पर इंजेक्शन है, और पर
        सीमा क्षेत्र बिल्कुल एंटीपोड की पहचान करता है:
        $ E(\pi u) = E(-\pi u) = 2uu^{\mathsf T} - I $, के साथ
        कोई अन्य संयोग नहीं. इस प्रकार $ SO(3)$ गेंद है
        एंटीपोडल सीमा बिंदुओं से चिपके हुए ---
        प्रक्षेप्य स्थान $\mathbb{RP}^3$ --- और एक व्यास
        प्रश्न 10 का गैर-अनुबंधीय लूप बन जाता है।
  \item बताते हैं कि $\rho_p\rho_q\rho_p^{-1} = \rho_{pq\bar
        p}$; का समावेश $ SO(3)$ (द $ R \neq
        I $ साथ $ R^2 = I $) बिल्कुल आधे मोड़ हैं
        $\rho_w $ साथ $ w $ एक इकाई शुद्ध चतुर्भुज; और वह
        द \omterm{ex:b3:groups:actions}{केंद्र} का $ SO(3)$ तुच्छ है.
  \item दिखाएँ कि प्रत्येक घूर्णन दो का गुणनफल है
        आधा मोड़: के लिए $ q = \cos\frac\theta2 +
        \sin\frac\theta2\,u $, शुद्ध एक इकाई चुनें $ w \perp
        u $, जाँच करें कि $ w' = qw $ पुनः एक इकाई शुद्ध है
        चतुर्भुज, और सत्यापित करें $\rho_q =
        \rho_{w'}\rho_w $. दो अक्ष कहाँ स्थित हैं, और
        वे कौन सा कोण बनाते हैं?
  \item निष्कर्ष निकालें संस्थानिक सारांश: $ SO(3)$ है
        कॉम्पैक्ट और \omterm{def:b3:topology:pathconnected}{पथ से जुड़ा हुआ} (दो प्रमाण दीजिए:
        \omterm{def:b3:topology:continuity}{निरंतर} की छवि $ S^3$ अंतर्गत $\rho $; की छवि
        $\exp $), जबकि $ O(3)$ बिल्कुल दो हैं \omterm{def:b3:topology:components}{जुड़ा हुआ
        घटक}, प्रत्येक होम्योमॉर्फिक को $ SO(3)$.

  \item (एक रचना, तीन तरीके) चलिए $ R_1$ घूर्णन हो
        द्वारा $\frac\pi2$ के बारे में $ z $-अक्ष और $ R_2$ द
        द्वारा घूर्णन $\frac\pi2$ के बारे में $ x $-अक्ष. गणना करें
        की धुरी और कोण $ R_2R_1$: (i) गुणा करके
        दो $3\times3$ मैट्रिक्स और उपयोग
        ट्रेस/एंटीसिमेट्रिक भाग (भाग V); (ii) द्वारा
        संगत इकाई चतुर्भुजों को गुणा करना
        $ q_2q_1$. जाँचें कि दो उत्तर सहमत हैं: कोण
        $\frac{2\pi}3$, अक्ष $\frac1{\sqrt3}(1, -1, 1)$.
  \item (द केली परिवर्तन) के लिए $ K $ एंटीसिमेट्रिक, दिखाएँ
        वह $ I + K $ उलटा है और
        \[
        C(K) = (I - K)(I + K)^{-1} \in SO(n),
        \]
        साथ $-1$ का कभी भी एक प्रतिध्वनि नहीं $ C(K)$; बताते हैं कि
        $ K \mapsto C(K)$ एंटीसिमेट्रिक से एक आपत्ति है
        मैट्रिक्स पर $\{R \in SO(n) : -1 \notin
        \operatorname{Sp}R\}$, व्युत्क्रम के साथ $ R \mapsto (I -
        R)(I + R)^{-1}$. (का एक तर्कसंगत चार्ट $ SO(n)$,
        पारलौकिक का साथी $\exp$ का
        \cref{exo:b3:submanifolds:11}.)
\end{enumerate}
\end{problem}
