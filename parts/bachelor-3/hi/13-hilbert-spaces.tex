\chapter{हिल्बर्ट स्पेस}\label{ch:b3:hilbert}

ए \omterm{def:b3:hilbert:inner}{हिल्बर्ट स्थान} एक बानाच स्थान है जिसका मानदंड एक से आता है
\omterm{def:b3:hilbert:inner}{अंदरूनी प्रोडक्ट} --- और वह एकल अतिरिक्त संरचना पुनर्स्थापित हो जाती है
अनंत आयाम, लगभग संपूर्ण यूक्लिडियन ज्यामिति:
ओर्थोगोनल प्रक्षेपण मौजूद हैं, हर \omterm{def:b3:topology:continuity}{निरंतर} कार्यात्मक एक है
\omterm{def:b3:hilbert:inner}{अंदरूनी प्रोडक्ट} एक निश्चित वेक्टर (Riesz), और ऑर्थोनॉर्मल के विरुद्ध
\omterm{def:b3:topology:basis}{आधार} एक अभिसरण श्रृंखला में प्रत्येक वेक्टर का विस्तार करते हैं
पायथागॉरियन बहीखाता पद्धति (\omterm{thm:b3:hilbert:parseval}{पारसेवल}). अध्याय का \omterm{thm:b3:rings:crt}{चरमोत्कर्ष} है a
ऋण सम्मान: त्रिकोणमितीय प्रणाली एक लंबात्मक \omterm{def:b3:topology:basis}{आधार} है
का $L^2$, इसलिए पारसेवलकी पहचान रखती है \emph{प्रत्येक}
वर्ग-अभिन्न समारोह--- कथन वर्ष 2 ही हो सका
टुकड़े-टुकड़े करके सिद्ध करें $\mathcal C^1$ कार्य. हम साथ समाप्त करते हैं
लैक्स--मिलग्राम, परिवर्तनशील दृष्टिकोण का वर्कहॉर्स लेम्मा
विभेदक समीकरणों के लिए.

लगातार, $ H $ ऊपर एक सदिश स्थान है $ K = \R $ या $\C $.

\section{आंतरिक उत्पाद; प्रक्षेपण प्रमेय}

\begin{definition}\label{def:b3:hilbert:inner}
एक \emph{अंदरूनी प्रोडक्ट}\index{अंदरूनी प्रोडक्ट} एक नक्शा है $\langle
\cdot,\cdot\rangle \colon H\times H \to K $, में रैखिक
दूसरा चर, के साथ $\langle y, x\rangle =
\overline{\langle x, y\rangle}$ और $\langle x, x\rangle > 0$
के लिए $ x \neq 0$. यह \omterm{def:b3:rings:ideal}{आदर्श} को प्रेरित करता है $\norm x = \langle x,
x\rangle^{1/2}$, द \emph{कॉची-श्वार्ज़ असमानता}
$\abs{\langle x, y\rangle} \leq \norm x\norm y $ (वर्ष 2)
प्रमाण --- विवेचक --- अपरिवर्तित है), और
\emph{समांतर चतुर्भुज नियम}
\[
\norm{x + y}^2 + \norm{x - y}^2 = 2\norm x^2 + 2\norm y^2 .
\]
ए \emph{हिल्बर्ट स्थान}\index{हिल्बर्ट स्थान} एक है
आंतरिक-उत्पाद स्थान \omterm{def:b3:complete:complete}{पूरा} इस मानदंड के लिए. उदाहरण: $\ell^2$
(\cref{pb:b3:banach:1}) और, मौलिक एक, $ L^2(\mu)$
साथ $\langle f, g\rangle = \int\bar fg\,\dd\mu $ ---
\omterm{def:b3:complete:complete}{पूरा} रिज़्ज़--फिशर द्वारा (\cref{thm:b3:lp:complete}); द
आंतरिक उत्पाद कॉची-श्वार्ज़ द्वारा परिमित है ($= $ एच\"अधिक उम्र में
$ p = q = 2$).
\end{definition}

\begin{theorem}[एक बंद उत्तल सेट पर प्रक्षेपण]
\label{thm:b3:hilbert:projection}
होने देना $ C \neq \varnothing $ एक \omterm{def:b3:topology:interior}{बंद} हो \emph{उत्तल} का उपसमुच्चय
द \omterm{def:b3:hilbert:inner}{हिल्बर्ट स्थान} $ H $ और $ x \in H $. एक अनोखा है $ p_C(x)
\in C $ साथ\index{प्रक्षेपण प्रमेय}
\[
\norm{x - p_C(x)} = d(x, C),
\]
दवार जाने जाते है: $\operatorname{Re}\langle x - p_C(x),\ c -
p_C(x)\rangle \leq 0$ सभी के लिए $ c \in C $. वो नक्शा $ p_C $ है
$1$-लिप्सचिट्ज़.
\end{theorem}

\begin{proof}
होने देना $ d = d(x, C)$ और $(c_n) \subseteq C $ साथ $\norm{x - c_n}
\to d $. समांतर चतुर्भुज चालू $ x - c_n $ और $ x - c_m $:
\[
\norm{c_n - c_m}^2 = 2\norm{x - c_n}^2 + 2\norm{x - c_m}^2 -
4\,\bigl\|x - \tfrac{c_n + c_m}2\bigr\|^2
\leq 2\norm{x{-}c_n}^2 + 2\norm{x{-}c_m}^2 - 4d^2
\]
(उत्तलता मध्यबिंदु को अंदर रखती है $ C $): दाहिनी ओर की ओर झुकाव होता है
$0$, इसलिए $(c_n)$ कॉची है, और इसकी सीमा है $ p \in C $ (\omterm{def:b3:topology:interior}{बंद})
प्राप्त करता है $ d $. विशिष्टता: दो मिनिमाइज़र एक ही तरह से देते हैं
पहचान, $\norm{p - p'}^2 \leq 2d^2 + 2d^2 - 4d^2 = 0$.

विशेषता: के लिए $ c \in C $, $ t \in \intoc01$, वेक्टर
$ p + t(c - p) \in C $, इसलिए
\[
d^2 \leq \norm{x - p - t(c-p)}^2
= d^2 - 2t\operatorname{Re}\langle x - p, c - p\rangle +
t^2\norm{c-p}^2 ;
\]
द्वारा विभाजित करें $ t \to 0^+$: $\operatorname{Re}\langle x - p, c -
p\rangle \leq 0$. इसके विपरीत यह असमानता देती है $\norm{x -
c}^2 = \norm{x - p}^2 - 2\operatorname{Re}\langle x - p, c -
p\rangle + \norm{p - c}^2 \geq \norm{x-p}^2$. लिप्सचिट्ज़: के लिए
$ x, y $ अनुमानों के साथ $ p, q $, दो वेरिएबल जोड़ें
असमानताएं (साथ) $ c = q $, सम्मान।\ $ c = p $):
$\operatorname{Re}\langle x - y - (p - q), p - q\rangle \geq
0$, इसलिए $\norm{p - q}^2 \leq \operatorname{Re}\langle x - y, p -
q\rangle \leq \norm{x - y}\norm{p - q}$.
\end{proof}

\begin{theorem}[ऑर्थोगोनल अपघटन]
\label{thm:b3:hilbert:decomposition}
होने देना $ F $ एक हो \emph{\omterm{def:b3:topology:interior}{बंद} उपस्थान} का $ H $. तब $ p_F $ है
रैखिक, $ x - p_F(x) \perp F $ सभी के लिए $ x $, और\index{ओर्थोगोनल
पूरक}
\[
H = F \oplus F^\perp,
\qquad F^\perp = \{y : \langle y, f\rangle = 0\ \forall f\in
F\},
\qquad (F^\perp)^\perp = F .
\]
एक \omterm{def:b3:groups:normal}{सामान्य} उपस्थान के लिए, $(F^\perp)^\perp = \bar F $; विशेष रूप से $ F $ \omterm{def:b3:topology:interior}{घना} है iff $ F^\perp = \{0\}$.
\end{theorem}

\begin{proof}
एक उप-स्थान के लिए, परिवर्तनशील लक्षण वर्णन $ c =
p_F(x) \pm f $ ($ f \in F $, दोनों संकेत, और $\iu f $ में
जटिल मामला) बल $\langle x - p_F(x), f\rangle = 0$: द
अवशेष ओर्थोगोनल है $ F $. सड़न $ x = p_F(x) + (x
- p_F(x))$ साथ $ F \cap F^\perp = \{0\}$ ($\langle y, y
\rangle = 0$); की रैखिकता $ p_F $ की विशिष्टता से अनुसरण करता है
ऐसे विघटन (उनमें दोनों पक्ष रैखिक)। $(F^\perp)
^\perp \supseteq F $ हमेशा; इसके विपरीत यदि $ x \perp F^\perp $,
लिखना $ x = f + g $: $ g = x - f \in F^\perp $ और $\langle g,
g\rangle = \langle x, g\rangle - \langle f, g\rangle = 0$: $ x
= f \in F $. एक \omterm{def:b3:groups:normal}{सामान्य} उपस्थान के लिए $ F $: $ F^\perp = \bar
F^{\,\perp}$ (निरंतरता की \omterm{def:b3:hilbert:inner}{अंदरूनी प्रोडक्ट}), इसलिए
$(F^\perp)^\perp = \bar F $ \omterm{def:b3:topology:interior}{बंद} मामले से; \omterm{ex:b3:lebesgue:gamma}{घनत्व} आईएफएफ
$\bar F = H $ आईएफएफ $ F^\perp = 0$.
\end{proof}

\begin{theorem}[रिज़्ज़ प्रतिनिधित्व]\label{thm:b3:hilbert:riesz}
हरएक के लिए \omterm{def:b3:topology:continuity}{निरंतर} रैखिक कार्यात्मक $\varphi \in H'$ वहाँ
एक अनोखा है $ a \in H $ साथ\index{रिज़्ज़ प्रतिनिधित्व प्रमेय}
\[
\varphi(x) = \langle a, x\rangle \quad (x \in H),
\qquad \norm\varphi_{H'} = \norm a .
\]
\end{theorem}

\begin{proof}
अगर $\varphi = 0$: $ a = 0$. अन्यथा $ F = \ker\varphi $ एक है
\omterm{def:b3:topology:interior}{बंद} उचित उपस्थान; चुनना $ u \in F^\perp $, $\norm u = 1$
(\cref{thm:b3:hilbert:decomposition}: $ F^\perp \neq 0$ तब से
$ F \neq H $). किसी के लिए $ x $, वेक्टर $\varphi(x)u -
\varphi(u)x \in \ker\varphi $, इस तरह $\perp u $:
\[
0 = \langle u, \varphi(x)u - \varphi(u)x\rangle
= \varphi(x) - \varphi(u)\langle u, x\rangle :
\qquad \varphi(x) = \langle
\overline{\varphi(u)}\,u,\ x\rangle .
\]
इसलिए $ a = \overline{\varphi(u)}u $ काम करता है. विशिष्टता: $\langle a
- a', x\rangle = 0$ सभी के लिए $ x $, परीक्षा $ x = a - a'$. मानदंड:
$\abs{\varphi(x)} \leq \norm a\norm x $ (कॉची-श्वार्ज़) के साथ
समानता पर $ x = a $.
\end{proof}

\begin{example}[एक प्रक्षेपण, अंत तक गणना की गई]
\label{ex:b3:hilbert:projectionexample}
में $ H = L^2(\intcc01)$, का सबसे अच्छा अनुमान क्या है
$ f(x) = x^2$ एक एफ़िन फ़ंक्शन द्वारा? उपस्थान $ F =
\operatorname{Vect}(1, x)$ \omterm{def:b3:topology:interior}{बंद} है (परिमित-आयामी),
और $ p_F(f) = a + bx $ की रूढ़िवादिता द्वारा विशेषता है
करने के लिए अवशेष $1$ और को $ x $:
\[
\int_0^1(x^2 - a - bx)\,\dd x = 0,
\qquad
\int_0^1x\,(x^2 - a - bx)\,\dd x = 0,
\]
यानी\ $\frac13 = a + \frac b2$ और $\frac14 = \frac a2 +
\frac b3$: $ a = -\frac16$, $ b = 1$. इसलिए $ p_F(x^2) = x -
\frac16$, और त्रुटि है
\[
d(f, F)^2 = \int_0^1\Bigl(x^2 - x + \frac16\Bigr)^2\dd x =
\frac1{180},
\qquad d(f, F) = \frac1{6\sqrt5} .
\]
दो टिप्पणियाँ आत्मसात करने लायक हैं। सबसे पहले, गणना है
ए के अलावा कुछ नहीं $2\times2$ रैखिक प्रणाली --- \emph{\omterm{def:b3:groups:normal}{सामान्य}
समीकरण}; एकपदी \omterm{def:b3:topology:basis}{आधार} के लिए उनका मैट्रिक्स
$\bigl(\frac1{i+j+1}\bigr)$ कुख्यात है
ख़राब स्थिति वाले हिल्बर्ट मैट्रिक्स, और पहले ऑर्थोगोनलाइज़िंग
(लीजेंडर बहुपद, \cref{pb:b3:hilbert:1}) इलाज है.
दूसरा, सर्वोत्तम \emph{वर्दी} का सन्निकटन $ x^2$ द्वारा
एफ़िन फ़ंक्शंस अलग है ($ x - \frac18$, द्वारा
इक्विओसिलेशन): प्रत्येक मानदंड की अपनी ज्यामिति होती है, और केवल
हिल्बर्टियन एक रैखिक प्रणाली के साथ उत्तर देता है।
\end{example}

\section{लम्बवत् आधार}

\begin{definition}\label{def:b3:hilbert:onb}
एक परिवार $(e_i)_{i\in I}$ है \emph{ऑर्थोनॉर्मल} अगर $\langle
e_i, e_j\rangle = \delta_{ij}$, और ए \emph{हिल्बर्ट
आधार}\index{हिल्बर्ट
आधार} (ऑर्थोनॉर्मल \omterm{def:b3:topology:basis}{आधार}) यदि इसके अलावा
इसके परिमित रैखिक संयोजन सघन हैं $ H $ (परिवार है
\emph{कुल}). हम गणनीय मामले का इलाज करते हैं $ I = \N $, जिसके द्वारा
ग्राम--श्मिट प्रत्येक को कवर करता है \emph{\omterm{def:b3:galois:separable}{वियोज्य}} $ H $
(\cref{prop:b3:hilbert:gramschmidt}).
\end{definition}

\begin{theorem}[बेसेल, पारसेवल]\label{thm:b3:hilbert:parseval}
होने देना $(e_n)_{n\in\N}$ में ऑर्थोनॉर्मल हो $ H $, और $ c_n(x) =
\langle e_n, x\rangle $.
\begin{enumerate}
  \item (बेसेल) $\sum_n\abs{c_n(x)}^2 \leq \norm x^2$, और
        शृंखला $\sum_nc_n(x)e_n $ में एकत्रित हो जाता है $ H $, के साथ
        योग $ p_F(x)$, $ F = \overline{\operatorname{Vect}}(e_n)$.
  \item निम्नलिखित समतुल्य हैं: (i) $(e_n)$ एक है \omterm{def:b3:hilbert:onb}{हिल्बर्ट
        आधार}; (ii) $ x = \sum_nc_n(x)e_n $ हरएक के लिए $ x $; (iii)
        \emph{पारसेवल}\index{पारसेवल}: $\norm x^2
        = \sum_n\abs{c_n(x)}^2$ हरएक के लिए $ x $; (iv) एकमात्र
        सभी के लिए वेक्टर ऑर्थोगोनल $ e_n $ है $0$.
  \item अगर $(e_n)$ एक है \omterm{def:b3:hilbert:onb}{हिल्बर्ट आधार}, $ x \mapsto (c_n(x))_n $
        एक सममितीय समरूपता है $ H \to \ell^2$
        (\emph{प्रत्येक} अनंत-आयामी \omterm{def:b3:galois:separable}{वियोज्य} हिल्बर्ट
        अंतरिक्ष ``है'' $\ell^2$), और $\langle x, y\rangle =
        \sum_n\overline{c_n(x)}c_n(y)$.
\end{enumerate}
\end{theorem}

\begin{proof}
(1) परिमित के लिए $ N $: $ x - \sum_{n\leq N}c_ne_n \perp e_k $ ($ k
\leq N $), तो पाइथागोरस देता है $\norm x^2 = \sum_{n\leq
N}\abs{c_n}^2 + \norm{x - \sum_{n\leq N}c_ne_n}^2$: बेसेल.
आंशिक रकम $ S_N = \sum_{n\leq N}c_ne_n $ कॉची हैं:
$\norm{S_N - S_M}^2 = \sum_{M<n\leq N}\abs{c_n}^2$, ए की पूँछ
अभिसरण श्रृंखला; सीमा निहित है $ F $, और $ x - \lim S_N
\perp $ प्रत्येक $ e_k $ (निरंतरता), इस तरह $\perp F $: विशिष्टता से
ऑर्थोगोनल अपघटन का, $\lim S_N = p_F(x)$.

(2) (मैं)$\Rightarrow $(ii): $ F = H $, इसलिए $ p_F = \mathrm{id}$.
(ii)$\Rightarrow $(iii): सीमा में पाइथागोरस ($\norm{S_N}^2
= \sum_{n \leq N}\abs{c_n}^2 \to \norm x^2$).
(iii)$\Rightarrow $(iv): $ x \perp $ सभी $ e_n $ देता है $\norm x^2 =
0$. (iv)$\Rightarrow $(मैं): $ F^\perp = \{0\}$
(सभी के लिए रूढ़िवादिता $ e_n $ के लिए रूढ़िवादिता है $ F $), इसलिए $ F $
द्वारा \omterm{def:b3:topology:interior}{घना} है \cref{thm:b3:hilbert:decomposition}; लेकिन $ F $, ए
\omterm{def:b3:topology:interior}{बंद}, पहले से ही \omterm{def:b3:topology:interior}{बंद} है: $ F = H $.

(3) नक्शा रैखिक है, (iii) द्वारा सममितीय (इसलिए इंजेक्शन),
और विशेषण: दिया गया $(c_n) \in \ell^2$, श्रृंखला $\sum
c_ne_n $ एक पूर्वछवि में परिवर्तित हो जाता है (कौची जैसा कि (1) में है)। \omterm{def:b3:hilbert:inner}{अंदरूनी प्रोडक्ट} सूत्र (iii), या प्रत्यक्ष सीमा से ध्रुवीकरण है
गणना.
\end{proof}

\begin{proposition}[ग्राम--श्मिट]\label{prop:b3:hilbert:gramschmidt}
होने देना $(x_n)$ एक रैखिक रूप से स्वतंत्र अनुक्रम बनें। सेटिंग
आगमनात्मक रूप से $\tilde e_n = x_n - \sum_{k<n}\langle e_k,
x_n\rangle e_k $ और $ e_n = \tilde e_n/\norm{\tilde e_n}$
एक ऑर्थोनॉर्मल पैदा करता है $(e_n)$ समान परिमित विस्तार के साथ:
$\operatorname{Vect}(e_1, \dots, e_n) = \operatorname{Vect}
(x_1, \dots, x_n)$. फलस्वरूप प्रत्येक \omterm{def:b3:galois:separable}{वियोज्य} \omterm{def:b3:hilbert:inner}{हिल्बर्ट स्थान}
(गणनीय सघन उपसमुच्चय वाला) में एक है \omterm{def:b3:hilbert:onb}{हिल्बर्ट
आधार}.\index{ग्राम--श्मिट ऑर्थोनॉर्मलाइज़ेशन}
\end{proposition}

\begin{proof}
प्रेरण: $\tilde e_n \perp e_k $ ($ k < n $) निर्माण द्वारा,
और $\tilde e_n \neq 0$ स्वतंत्रता से; स्पैन मेल खाते हैं
प्रत्येक चरण (\omterm{def:b3:topology:basis}{आधार} का त्रिकोणीय परिवर्तन)। एक \omterm{def:b3:galois:separable}{वियोज्य} के लिए $ H $:
सघन अनुक्रम से एक रैखिकतः स्वतंत्र निकालें
सघन विस्तार के साथ उपपरिवार (प्रत्येक सदिश को विस्तार में छोड़ें
इसके पूर्ववर्ती - अवधि अपरिवर्तित है), ऑर्थोनॉर्मलाइज़ करें:
परिणाम समग्र है.
\end{proof}

\begin{theorem}[त्रिकोणमितीय प्रणाली; आख़िरकार पारसेवल]
\label{thm:b3:hilbert:fourier}
में $ L^2(\intcc{-\pi}\pi)$ साथ $\langle f, g\rangle =
\frac1{2\pi}\int_{-\pi}^\pi \bar fg $, परिवार $ e_n(t) =
\eu^{\iu nt}$, $ n \in \Z $, एक है \omterm{def:b3:hilbert:onb}{हिल्बर्ट आधार}. नतीजतन,
के लिए \emph{प्रत्येक} $ f \in L^2$ --- विशेष रूप से प्रत्येक टुकड़े के अनुसार
\omterm{def:b3:topology:continuity}{निरंतर} $2\pi $-आवधिक $ f $ --- साथ $ c_n(f) =
\frac1{2\pi}\int_{-\pi}^{\pi}f(t)\eu^{-\iu nt}\dd t $:
\[
f = \sum_{n\in\Z}c_n(f)\,\eu^{\iu nt} \ \ \text{in } L^2,
\qquad
\frac1{2\pi}\int_{-\pi}^{\pi}\abs f^2 =
\sum_{n\in\Z}\abs{c_n(f)}^2 .
\]
यह, पूर्ण व्यापकता में, सिद्ध करता है \omterm{thm:b3:hilbert:parseval}{पारसेवल} वह पहचान
वर्ष 2 में प्रवेश दिया गया।\index{पारसेवल की पहचान (सामान्य)}
\end{theorem}

\begin{proof}
ऑर्थोनोर्मैलिटी एक प्रत्यक्ष गणना है (वर्ष 2)। समग्रता: चलो
$ f \in L^2$ होना $\perp $ सभी $ e_n $, यानी\ सभी फूरियर
गुणांक गायब हो जाते हैं। \omterm{def:b3:topology:continuity}{निरंतर} $2\pi $-आवधिक कार्य हैं
में \omterm{def:b3:topology:interior}{घना} $ L^2(\intcc{-\pi}\pi)$: वास्तव में $\mathcal
C_c(\intoo{-\pi}\pi)$ \omterm{def:b3:topology:interior}{घना} है
(\cref{thm:b3:lp:density}(2)) और ऐसे कार्यों का विस्तार होता है
समय-समय पर और लगातार. त्रिकोणमितीय बहुपद हैं
$\norm\cdot_\infty $-के बीच \omterm{def:b3:topology:interior}{घना} \omterm{def:b3:topology:continuity}{निरंतर} आवधिक कार्य
(स्टोन--वीयरस्ट्रैस, \cref{cor:b3:complete:weierstrass}(सी)),
और $\norm\cdot_2 \leq \norm\cdot_\infty $: त्रिकोणमितीय
बहुपद सघन होते हैं $ L^2$. लेकिन $ f \perp $ प्रत्येक
त्रिकोणमितीय बहुपद, इसलिए $ f \perp $ एक सघन उपस्थान: $ f
\in (\text{dense})^\perp = \{0\}$
(\cref{thm:b3:hilbert:decomposition}). का मानदंड (iv).
\cref{thm:b3:hilbert:parseval} निष्कर्ष; (ii) और (iii)
डिस्प्ले पर अनपैक करें (गणनीय को पुनः अनुक्रमित करना)। $\Z $; द
डबल-एंडेड श्रृंखला बिना शर्त अभिसरण करती है --- आंशिक
किसी भी थका देने वाले परिवार के अभिसरण पर रकम, द्वारा $\ell^2$ पूंछ
तर्क)
\end{proof}

\begin{theorem}[लैक्स--मिलग्राम]\label{thm:b3:hilbert:laxmilgram}
होने देना $ H $ वास्तविक बनो \omterm{def:b3:hilbert:inner}{हिल्बर्ट स्थान} और $ a \colon H\times H \to
\R $ द्विरेखीय, \emph{\omterm{def:b3:topology:continuity}{निरंतर}} ($\abs{a(u,v)} \leq M\norm
u\norm v $) और \emph{बलपूर्वक} ($ a(u, u) \geq \alpha\norm u^2$,
$\alpha > 0$). फिर हर एक के लिए $\varphi \in H'$ वहाँ एक है
अद्वितीय $ u \in H $ साथ\index{लैक्स--मिलग्राम प्रमेय}
\[
a(u, v) = \varphi(v) \qquad \text{for all } v \in H .
\]
\end{theorem}

\begin{proof}
तय के लिए $ u $, $ v \mapsto a(u, v)$ है \omterm{def:b3:topology:continuity}{निरंतर} रैखिक: रिज़्ज़
एक अनोखा देता है $ Au \in H $ साथ $ a(u,v) = \langle Au,
v\rangle $; $ A $ के साथ रैखिक है $\norm{Au} \leq M\norm u $
(प्रतिनिधियों की विशिष्टता, फिर बाध्य)। ज़बरदस्ती:
$\alpha\norm u^2 \leq a(u,u) = \langle Au, u\rangle \leq
\norm{Au}\norm u $, इसलिए $\norm{Au} \geq \alpha\norm u $: $ A $ है
\omterm{def:b3:topology:interior}{बंद} सीमा के साथ इंजेक्शन (एक कॉची छवि अनुक्रम)। $ Au_n $
बल $ u_n $ कॉची)। दायरा सघन है: $ w \perp
\operatorname{im}A $ देता है $0 = \langle Aw, w\rangle \geq
\alpha\norm w^2$. \omterm{def:b3:topology:interior}{बंद} और \omterm{def:b3:topology:interior}{घना}: $ A $ वस्तुनिष्ठ है. दिया गया
$\varphi $, होने देना $ f $ इसका \omterm{def:b3:representations:rep}{प्रतिनिधित्व} करें (रिस्ज़) और $ u = A^{-1}f $:
$ a(u, v) = \langle f, v\rangle = \varphi(v)$, विशिष्ट रूप से
($ a(u - u', \cdot) = 0$ और ज़बरदस्ती)।
\end{proof}

\begin{remark}\label{rem:b3:hilbert:variational}
कब $ a $ सममित है, लैक्स--मिलग्राम का समाधान अद्वितीय है
का न्यूनतमीकरणकर्ता \emph{ऊर्जा} $ J(v) = \frac12a(v,v) -
\varphi(v)$ (\cref{exo:b3:hilbert:9}): समाधान का अस्तित्व
एक झटके में विभिन्न समस्याओं का समाधान। उपयुक्त पर लागू किया गया
फ़ंक्शन स्पेस (बाद के कोर्स के सोबोलेव स्पेस), यह
अंतर समीकरणों के लिए सीमा मान समस्याओं को हल करता है ---
आंशिक अंतर समीकरणों का आधुनिक प्रवेश बिंदु।
\end{remark}

\begin{exercise}[$\star\star $]\label{exo:b3:hilbert:11}
(ऑर्थोगोनल अनुमान, विशेषता) चलो $ H $ एक हो हिल्बर्ट
अंतरिक्ष और $ P \in \mathcal L(H)$ साथ $ P^2 = P $, $ P \neq 0$.
इसकी समतुल्यता दिखाएँ:
(मैं) $ P $ पर ओर्थोगोनल प्रक्षेपण है
$\operatorname{im}P $;
(ii) $ P = P^*$ (\cref{exo:b3:hilbert:8});
(iii) $\vertiii P = 1$.
\emph{(के लिए (iii) $\Rightarrow $ (i): यदि कुछ $ x \in
(\ker P)^\perp $ था $ Px \neq x $, विचार करना $ x + t(Px - x)$
--- या सीधे: के लिए $ u \in \operatorname{im}P $ और $ v \in
\ker P $, बढ़ाना $\norm{P(u + tv)}^2 \leq \norm{u + tv}^2$
सभी के लिए $ t \in \R $ और निष्कर्ष निकालो $\langle u, v\rangle = 0$.)}
पर एक गैर-ऑर्थोगोनल प्रक्षेपण प्रदर्शित करें $\R^2$ और गणना करें
यह \omterm{def:b3:rings:ideal}{आदर्श} है.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b3:hilbert:12}
(वॉन \omterm{prop:b3:banach:neumann}{न्यूमैन} का एर्गोडिक प्रमेय) आइए $ U \in \mathcal L(H)$ होना
\emph{अमली} ($ U^*U = UU^* = I $), $ F = \ker(U - I)$ द
निश्चित स्थान, $ P $ ओर्थोगोनल प्रक्षेपण $ F $, और
$ A_n = \frac1n\sum_{k=0}^{n-1}U^k $.
(एक शो $\ker(U - I) = \ker(U^* - I)$ \emph{(से
$\norm{Ux - x}^2 = 2\norm x^2 - 2\operatorname{Re}\langle
Ux, x\rangle $ और एकात्मकता)}, और निष्कर्ष निकालें
$\overline{\operatorname{im}(U - I)} = F^\perp $.
(बी) वह दिखाओ $ A_nx \to x $ के लिए $ x \in F $, और $ A_nx \to 0$
के लिए $ x \in \operatorname{im}(U - I)$ \emph{(दूरबीन)},
फिर के लिए $ x \in \overline{\operatorname{im}(U - I)}$
(वर्दी बाध्य $\vertiii{A_n} \leq 1$).
(सी) निष्कर्ष: $ A_nx \to Px $ के लिए \emph{प्रत्येक} $ x \in H $ ---
समय का औसत अपरिवर्तनीयों पर प्रक्षेपण में परिवर्तित होता है।
(d) Spell it out for $ H = L^2(\R/\Z)$ और $ Uf = f(\cdot +
\alpha)$ साथ $\alpha $ तर्कहीन: पहचानें $ F $ (उपयोग करें
फूरियर श्रृंखला, \cref{thm:b3:hilbert:fourier}) और निष्कर्ष निकालें
वह $\frac1n\sum_{k<n}f(x + k\alpha) \to \int_0^1f $ में
$ L^2$: द $ L^2$ अपरिमेय घुमावों का समान वितरण।
\end{exercise}

\section{समस्या: ऑर्थोगोनल बहुपद}

\begin{problem}[{सप्ताहांत समस्या --- लीजेंड्रे, हरमाइट, और
गॉस चतुर्भुज}]\label{pb:b3:hilbert:1}
होने देना $ I \subseteq \R $ एक अंतराल हो और $ w > 0$ ए \omterm{def:b3:topology:continuity}{निरंतर}
\emph{वज़न} पर आंतरिक भाग का $ I $ ऐसा है कि $\int_I
\abs t^nw(t)\dd t < \infty $ सभी के लिए $ n $; में काम $ H = L^2(I,
w\,\dd\lambda)$ साथ $\langle f, g\rangle = \int_I \bar
fg\,w $. ग्राम--श्मिट ने आवेदन किया $1, t, t^2, \dots $ पैदा करता है
द \emph{\omterm{pb:b3:hilbert:1}{ऑर्थोगोनल बहुपद}} $(p_n)$ के लिए $ w $ (मोनिक
\omterm{def:b3:groups:normal}{सामान्य}ीकरण: $ p_n = t^n + \cdots $).\index{ओर्थोगोनल
बहुपद}

\medskip
\noindent\textbf{Part I --- General theory.}
\begin{enumerate}
  \item बताते हैं कि $ p_n $ के प्रत्येक बहुपद के लिए लंबकोणीय है
        \omterm{def:b3:galois:extension}{डिग्री} $< n $, ओर वो $(p_0, \dots, p_n)$ एक \omterm{def:b3:topology:basis}{आधार} है
        का $\R_n[t]$.
  \item (तीन-अवधि की पुनरावृत्ति) दिखाएँ कि वास्तविक हैं $ a_n,
        b_n $ साथ
        \[
        p_{n+1}(t) = (t - a_n)\,p_n(t) - b_n\,p_{n-1}(t),
        \qquad b_n = \frac{\norm{p_n}^2}{\norm{p_{n-1}}^2} >
        0 .
        \]
        \emph{(बढ़ाना $ t\,p_n $ \omterm{def:b3:topology:basis}{आधार} में $(p_k)_{k \leq
        n+1}$ और ऑर्थोगोनैलिटी का उपयोग करके गुणांकों को मारें
        $\langle tp_n, p_k\rangle = \langle p_n,
        tp_k\rangle $.)}
  \item (जड़ें) वह दिखाओ $ p_n $ है $ n $ \emph{विशिष्ट} जड़ें,
        सब आंतरिक भाग को $ I $. \emph{(होने देना $ t_1 < \dots < t_m $ हो आंतरिक भाग के संकेत परिवर्तन $ p_n $; अगर $ m < n $,
        परीक्षा $ p_n $ ख़िलाफ़ $\prod_{i\leq m}(t - t_i)$ और
        रूढ़िवादिता का खंडन करें।)}
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Legendre ($ I = \intcc{-1}1$, $ w =
1$).} परिभाषित करना $ P_n(t) = \frac{1}{2^nn!}\,\frac{\dd^n}{\dd
t^n}\bigl[(t^2 - 1)^n\bigr]$ (रॉड्रिग्स)।
\begin{enumerate}[resume]
  \item दिखाओ $\deg P_n = n $ अग्रणी गुणांक के साथ
        $\frac{(2n)!}{2^n(n!)^2}$, और, भागों द्वारा एकीकृत करना
        $ n $ कई बार, वह $\langle P_n, Q\rangle = 0$ प्रत्येक के लिए
        बहुपद $ Q $ \omterm{def:b3:galois:extension}{डिग्री} का $< n $: द $ P_n $ (तक) हैं
        \omterm{def:b3:groups:normal}{सामान्य}ीकरण) द \omterm{pb:b3:hilbert:1}{ऑर्थोगोनल बहुपद} के लिए $ w = 1$.
  \item गणना करना $\norm{P_n}_2^2 = \frac{2}{2n+1}$
        \emph{(भागों द्वारा एकीकृत करें $ n $ कई बार खुद के खिलाफ और
        बीटा/वालिस इंटीग्रल को कम करें,
        \cref{exo:b3:product:8})}.
  \item दिखाएँ कि \omterm{def:b3:groups:normal}{सामान्य}ीकृत लीजेंड्रे बहुपद एक बनाते हैं
        \omterm{def:b3:hilbert:onb}{हिल्बर्ट आधार} का $ L^2(\intcc{-1}1)$
        \emph{(वीयरस्ट्रैस, \cref{cor:b3:complete:weierstrass},
        प्लस \omterm{ex:b3:lebesgue:gamma}{घनत्व} का $\mathcal C $ में $ L^2$)}, और विस्तार करें
        $ f(t) = \abs t $ \omterm{def:b3:galois:extension}{डिग्री} तक $2$: सर्वोत्तम गणना करें
        द्विघात $ L^2$-का अनुमान $\abs t $.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Hermite ($ I = \R $, $ w(t) =
\eu^{-t^2}$).} परिभाषित करना $ H_n(t) =
(-1)^n\eu^{t^2}\frac{\dd^n}{\dd t^n}\eu^{-t^2}$.
\begin{enumerate}[resume]
  \item बताते हैं कि $ H_n $ \omterm{def:b3:galois:extension}{डिग्री} का एक बहुपद है $ n $ के साथ
        अग्रणी गुणांक $2^n $, वह $ H_{n+1} = 2tH_n -
        H_n'$, ओर वो $\langle H_m, H_n\rangle_w =
        \delta_{mn}\,2^nn!\sqrt\pi $ \emph{(भाग फिर से)}.
  \item दिखाएँ कि हर्माइट परिवार कुल है $ L^2(\R,
        \eu^{-t^2}\dd t)$, से एक परिणाम स्वीकार करते हुए
        \cref{ch:b3:fouriertransform}: अगर $ g \in L^1(\R)$ है
        $\int g(t)\eu^{-\iu\xi t}\dd t = 0$ सभी के लिए $\xi $,
        तब $ g = 0$ ए.ई. \emph{(के लिए $ f \perp $ सभी $ H_n $,
        यानी\ $\perp $ सभी बहुपद: दिखाएँ $ z \mapsto \int
        f(t)\eu^{-t^2}\eu^{-\iu zt}\dd t $ अच्छी तरह से परिभाषित है,
        एक श्रृंखला के रूप में घातांक का विस्तार करें, इसका औचित्य सिद्ध करें
        प्रभुत्व द्वारा आदान-प्रदान, और निष्कर्ष निकालें कि
        का फूरियर रूपांतरण $ f\eu^{-t^2}$ गायब हो जाता है।)}
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Gauss quadrature.} हल करना $ n $, होने देना
$ t_1 < \dots < t_n $ की जड़ें बनें $ p_n $ (भाग I), और परिभाषित करें
वजन $ w_i = \int_I \ell_i(t)\,w(t)\dd t $ कहाँ
$\ell_i $ लैग्रेंज इंटरपोलेशन \omterm{def:b3:topology:basis}{आधार} बहुपद हैं
द $ t_i $.
\begin{enumerate}[resume]
  \item चतुर्भुज नियम दिखाएँ $ Q(f) = \sum_iw_if(t_i)$
        \omterm{def:b3:galois:extension}{डिग्री} के सभी बहुपदों पर सटीक है $\leq n - 1$
        (प्रक्षेप), और वास्तव में --- चमत्कार --- पर
        \omterm{def:b3:galois:extension}{डिग्री} के सभी बहुपद $\leq 2n - 1$: लिखना $ P =
        qp_n + r $ और भागफल पर ऑर्थोगोनैलिटी का उपयोग करें $ q $.
  \item दिखाएँ कि वज़न सकारात्मक हैं
        \emph{(नियम लागू करें $\ell_i^2$, \omterm{def:b3:galois:extension}{डिग्री} का $2n -
        2$)}, और पोलिया के प्रमेय से निष्कर्ष निकालें
        (\cref{exo:b3:banach:9}) वह गॉस चतुर्भुज
        अभिसरण: $ Q_n(f) \to \int_I fw $ हरएक के लिए \omterm{def:b3:topology:continuity}{निरंतर}
        $ f $ एक पर कॉम्पैक्ट $ I $.
  \item के लिए $ n = 2$, $ I = \intcc{-1}1$, $ w = 1$: गणना करें
        नोड्स $\pm\frac1{\sqrt3}$ और वजन $1, 1$, और
        पर सटीकता सत्यापित करें $1, t, t^2, t^3$ हाथ से. तुलना करें
        समान दो मूल्यांकन पर समलम्बाकार नियम के साथ
        अंक.
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Chebyshev: the polynomials that
oscillate best.} अब $ I = \intcc{-1}1$ और $ w(t) =
\frac1{\sqrt{1 - t^2}}$.
\begin{enumerate}[resume]
  \item बताते हैं कि $ T_n(\cos\theta) = \cos n\theta $ ए को परिभाषित करता है
        बहुपद $ T_n $ \omterm{def:b3:galois:extension}{डिग्री} का $ n $ (स्थापित करना $ T_{n+1} =
        2t\,T_n - T_{n-1}$ त्रिकोणमितीय पहचान से),
        अग्रणी गुणांक के साथ $2^{n-1}$ के लिए $ n \geq 1$;
        और वह प्रतिस्थापन $ t = \cos\theta $ देता है
        \[
        \langle T_m, T_n\rangle_w =
        \int_0^\pi\cos m\theta\,\cos n\theta\,\dd\theta
        = 0 \ (m \neq n), \qquad
        \norm{T_0}_w^2 = \pi,\ \ \norm{T_n}_w^2 = \frac\pi2 :
        \]
         $ T_n $ हैं \omterm{pb:b3:hilbert:1}{ऑर्थोगोनल बहुपद} इसके लिए
        वजन, और चेबीशेव विस्तार \emph{हैं} फूरियर
        भेस में कोसाइन श्रृंखला।
  \item स्पष्ट रूप से पता लगाएँ $ n $ जड़ों $ t_k =
        \cos\frac{(2k-1)\pi}{2n}$ और यह $ n + 1$ एक्सट्रीमा
        $ s_j = \cos\frac{j\pi}n $ का $ T_n $ पर $\intcc{-1}1$,
        कहाँ $ T_n(s_j) = (-1)^j $: ग्राफ
        \emph{इक्विओसिलेट्स} बीच में $\pm1$.
  \item (मिनीमैक्स) उसे सबके बीच दिखाओ \emph{राक्षसी}
        \omterm{def:b3:galois:extension}{डिग्री} के बहुपद $ n $, बहुपद
        $2^{1-n}T_n $ पर सबसे छोटा सुपर-मानदंड है
        $\intcc{-1}1$, अर्थात् $2^{1-n}$ --- और यह है
        अनोखा मिनिमाइज़र. \emph{(यदि एक भिक्षुक $ P $ था
        $\sup\abs P < 2^{1-n}$, के अंतर $2^{1-n}T_n -
        P $, \omterm{def:b3:galois:extension}{डिग्री} का $\leq n-1$, वैकल्पिक रूप से साइन इन करेगा
        द $ n+1$ समद्विबाहु बिंदु।)}
  \item प्रक्षेप के लिए आवेदन: नोड्स के लिए $ t_1 < \dots
        < t_n $ में $\intcc{-1}1$, लैग्रेंज की त्रुटि
        ए का प्रक्षेप $\mathcal C^n $ फ़ंक्शन शामिल है
        $\omega(t) = \prod_i(t - t_i)$. चुनकर दिखाओ
        नोड्स के रूप में चेबीशेव की जड़ें न्यूनतम हो जाती हैं
        $\sup_{\intcc{-1}1}\abs\omega $, और दे दो
        परिणामी बाध्य $\norm{f -
        L_nf}_\infty \leq \frac{\norm{f^{(n)}}_\infty}
        {2^{n-1}\,n!}$ --- समान दूरी वाले नोड्स के साथ तुलना करें
        (रंज की घटना को सावधान करने वाली कहानी के रूप में बताएं)।
  \item सत्यापित करें $\abs{T_n'(\pm1)} = n^2$ \emph{(अंतर करें
        $ T_n(\cos\theta) = \cos n\theta $ और सीमा ले लो
        $\theta \to 0, \pi $)}: से घिरे बहुपद $1$ पर
        $\intcc{-1}1$ जितना बड़ा व्युत्पन्न हो सकता है $ n^2$
        किनारे पर (मार्कोव की असमानता कहती है कि कोई बड़ी नहीं ---
        केवल कथन)। अंतराल में कहाँ है
        केवल व्युत्पन्न बाध्य $ O(n)$?
  \item (चेबीशेव--गॉस चतुर्भुज) दिखाएँ कि गॉस
        वजन के लिए नियम $ w $ पर $ n $ चेबीशेव जड़ें
        है \emph{बराबर} तौल $ w_i = \frac\pi n $
        \emph{(सटीकता पर $ T_0, \dots, T_{n-1}$ इसके अलावा
        त्रिकोणमितीय योग
        $\sum_{k=1}^n\cos\bigl(j\tfrac{(2k-1)\pi}{2n}\bigr) =
        0$ के लिए $1 \leq j \leq n - 1$)}: का सबसे समान
        सभी चतुर्भुज. इसके लिए इसे लिखें $ n = 3$.
\end{enumerate}

\medskip
\noindent\textbf{Part VI --- Christoffel--Darboux,
interlacing, and the Jacobi matrix.} \omterm{def:b3:groups:normal}{सामान्य} वज़न पर वापस जाएँ;
$ h_k = \norm{p_k}^2$ (मोनिक $ p_k $), $ b_k = h_k/h_{k-1}$.
\begin{enumerate}[resume]
  \item (न्यूनतम मानदंड) उसे सबके बीच दिखाओ \emph{राक्षसी}
        \omterm{def:b3:galois:extension}{डिग्री} के बहुपद $ n $, ओर्थोगोनल $ p_n $ है
        न्यूनतम का अनोखा $ L^2(w)$-मानदंड ---पहचानें
        ओर्थोगोनल प्रक्षेपण के रूप में न्यूनतमकरण
        $\R_{n-1}[t]$ (\cref{thm:b3:hilbert:projection} या
        वर्ष 2 का परिमित-आयामी प्रक्षेपण)। द
        प्रश्न 14 की मिनिमैक्स संपत्ति एक ही कथन है
        के साथ $ L^\infty $ की जगह $ L^2$: एक ही हीरो, दो
        मानदंड.
  \item (क्रिस्टोफ़ेल--डार्बौक्स) प्रेरण द्वारा सिद्ध करें $ n $
        तीन-अवधि की पुनरावृत्ति, पहचान का उपयोग करना
        \[
        \sum_{k=0}^{n}\frac{p_k(x)\,p_k(y)}{h_k}
        = \frac{p_{n+1}(x)\,p_n(y) -
        p_n(x)\,p_{n+1}(y)}{h_n\,(x - y)}
        \qquad (x \neq y),
        \]
        और इसका संगम रूप ($ y \to x $):
        $\sum_{k\leq n}\frac{p_k(x)^2}{h_k} =
        \frac{p_{n+1}'(x)p_n(x) - p_n'(x)p_{n+1}(x)}{h_n}$.
  \item उसका निष्कर्ष निकालो $ p_n $ और $ p_{n+1}$ कोई \omterm{def:b3:groups:normal}{सामान्य} जड़ नहीं है,
        और वह हर जड़ पर $ x_0$ का $ p_{n+1}$:
        $ p_n(x_0)\,p_{n+1}'(x_0) > 0$. निष्कर्ष निकालें
        \emph{इंटरलेसिंग} जड़ों की: दो लगातार के बीच
        की जड़ें $ p_{n+1}$ की बिल्कुल एक जड़ निहित है $ p_n $.
  \item (जैकोबी मैट्रिक्स) चलो $ J_n $ हो $ n\times n $
        विकर्ण के साथ सममित त्रिविकर्ण मैट्रिक्स $ a_0,
        \dots, a_{n-1}$ और ऑफ-विकर्ण प्रविष्टियाँ $\sqrt{b_1},
        \dots, \sqrt{b_{n-1}}$. प्रेरण द्वारा दिखाएँ कि
        $\det(tI_n - J_n) = p_n(t)$, तो की जड़ें $ p_n $
        एक वास्तविक सममित मैट्रिक्स के eigenvalues हैं ---
        एक पंक्ति में पुनः सिद्ध करना कि वे वास्तविक हैं, और (साथ में)
        ऊपर इंटरलेसिंग) बांधना \omterm{pb:b3:hilbert:1}{ऑर्थोगोनल बहुपद}
        की वर्णक्रमीय दुनिया के लिए \cref{ch:b3:spectral}.
  \item (संश्लेषण) तीनों के लिए शब्दकोश इकट्ठा करें
        शास्त्रीय परिवार (लीजेंडर, हर्माइट, चेबीशेव):
        अंतराल, वजन, परिभाषित सूत्र, तीन-अवधि
        पुनरावृत्ति, मानदंड और प्रत्येक का प्राकृतिक आवास
        (कॉम्पेक्टा पर चतुर्भुज और सन्निकटन; गाऊसी
        विश्लेषण; मिनिमैक्स और फूरियर-कोसाइन विधियाँ)। एक
        \omterm{def:b3:groups:normal}{सामान्य} सिद्धांत क्या है पर वाक्य (भाग I, VI)
        दिया कि कोई भी व्यक्तिगत गणना नहीं हो सकती।
\end{enumerate}

\medskip
\noindent\textbf{Part VII --- The error term, and the kernel
behind the weights.} यहाँ $ I $ है कॉम्पैक्ट और $ f \in \mathcal
C^{2n}(I)$.
\begin{enumerate}[resume]
  \item (गॉस त्रुटि सूत्र) चलो $ Hf $ हर्मिट बनो
        \omterm{def:b3:galois:extension}{डिग्री} का इंटरपोलेंट $\leq 2n - 1$ मेल मिलाना $ f $ और
        $ f'$ नोड्स पर $ t_1, \dots, t_n $ (इसे साबित करें
        अस्तित्व और बिन्दुवार त्रुटि
        \[
        f(t) - Hf(t) =
        \frac{f^{(2n)}(\xi_t)}{(2n)!}\;p_n(t)^2
        \]
        \omterm{def:b3:groups:normal}{सामान्य} सहायक-फ़ंक्शन तर्क द्वारा)। निष्कर्ष निकालें, द्वारा
        इस पहचान को इसके विरुद्ध एकीकृत करना $ w $ और निचोड़ना
        की चरम सीमा के बीच $ f^{(2n)}$, वह
        \[
        \int_I f\,w - Q_n(f)
        = \frac{f^{(2n)}(\xi)}{(2n)!}\,h_n
        \qquad\text{for some } \xi \in I,
        \]
        साथ $ h_n = \norm{p_n}^2$ जैसा कि भाग VI में है: गॉस
        चतुर्भुज एक-एक करके गलतियाँ करता है $2n $-वें व्युत्पन्न, द्वारा भारित
        मोनिक \omterm{pb:b3:hilbert:1}{ऑर्थोगोनल बहुपद} का वर्ग मानदंड।
  \item (वजन क्रिस्टोफ़ेल मान हैं) का उपयोग करना
        कर्नेल का पुनरुत्पादन $ K_n(x, y) = \sum_{k=0}^{n-1}
        \frac{p_k(x)p_k(y)}{h_k}$ का $\R_{n-1}[t]$ और
        की सटीकता $ Q_n $ \omterm{def:b3:galois:extension}{डिग्री} तक $2n - 2$, सिद्ध करना
        \[
        w_i \;=\;
        \Bigl(\,\sum_{k=0}^{n-1}
        \frac{p_k(t_i)^2}{h_k}\Bigr)^{\!-1} :
        \]
        प्रत्येक भार उसके नोड पर मान है
        \emph{क्रिस्टोफ़ेल फ़ंक्शन} --- की सकारात्मकता
        वज़न (प्रश्न 10) फिर से, अब सटीक के साथ
        सूत्र. सत्यापित करें कि यह ठीक हो गया है $ w_1 = w_2 = 1$ के लिए $ n =
        2$, $ I = \intcc{-1}1$, $ w = 1$.
  \item (हर चीज़ एक अभिन्न पर जाँचती है) चेबीशेव के लिए
        वजन और $ n = 3$ नोड्स, दोनों पक्षों की गणना करें
        \[
        \int_{-1}^{1}\frac{t^6}{\sqrt{1 - t^2}}\,\dd t
        = \frac{5\pi}{16},
        \qquad
        Q_3(t^6) = \frac{9\pi}{32},
        \]
        अतः चतुर्भुज त्रुटि बिल्कुल सही है $\frac{\pi}{32}$;
        फिर जांचें कि प्रश्न 23 का त्रुटि सूत्र क्या है
        सटीक रूप से इस मान की भविष्यवाणी करता है (यहाँ)। $ f^{(6)} = 6!$ है
        स्थिर, और $ h_3 = \norm{2^{-2}T_3}_w^2 =
        \frac\pi{32}$): सिद्धांत और संगणना से सहमत हैं
        अंतिम अंक.
\end{enumerate}
\end{problem}
