\section{कॉची--लिप्सचिट्ज़}

\begin{definition}\label{def:b3:ode:lipschitz}
$ f $ है \emph{स्थानीय स्तर पर लिप्सचिट्ज़ में $ x $}\index{स्थानीय स्तर पर लिप्सचिट्ज़ में $ x $}
यदि प्रत्येक बिंदु $ U $ एक \omterm{def:b3:topology:topology}{अड़ोस-पड़ोस} $ V $ और एक \omterm{def:b3:ode:stability}{स्थिर}ांक
$ L $ साथ $\norm{f(t, x_1) - f(t, x_2)} \leq L\norm{x_1 -
x_2}$ के लिए $(t, x_1), (t, x_2) \in V $. अगर $ f $ है $\mathcal
C^1$ (या केवल $\partial_xf $ मौजूद है और है \omterm{def:b3:topology:continuity}{निरंतर}), यह
स्थानीय रूप से लिप्सचिट्ज़ में है $ x $: एक पर कॉम्पैक्ट उत्तल
\omterm{def:b3:topology:topology}{अड़ोस-पड़ोस}, औसत मूल्य असमानता के साथ $ L =
\sup\vertiii{\partial_xf}$.
\end{definition}

\begin{theorem}[कॉची--लिप्सचिट्ज़, स्थानीय]
\label{thm:b3:ode:cauchylipschitz}
होने देना $ f $ होना \omterm{def:b3:topology:continuity}{निरंतर} और \omterm{def:b3:ode:lipschitz}{स्थानीय स्तर पर लिप्सचिट्ज़} में $ x $, और $(t_0,
x_0) \in U $. वहाँ है $ T > 0$ जैसे कि कॉची समस्या
\[
x' = f(t, x), \qquad x(t_0) = x_0
\]
बिल्कुल एक समाधान चालू है $[t_0 - T, t_0 +
T]$.\index{कॉची--लिप्सचिट्ज़ प्रमेय}
\end{theorem}

\begin{proof}
चुनना $ a, b > 0$ साथ $ Q = [t_0 - a, t_0 + a]\times\bar
B(x_0, b) \subseteq U $, जिस पर $\norm f \leq M $ और $ f $ है
$ L $-लिप्सचिट्ज़ में $ x $. ए $\mathcal C^1$ फ़ंक्शन एक समाधान है
यदि यह अभिन्न समीकरण को संतुष्ट करता है
\[
x(t) = x_0 + \int_{t_0}^{t}f\bigl(s, x(s)\bigr)\,\dd s
\]
(कैलकुलस का मौलिक प्रमेय, दोनों तरीके)। होने देना $ T =
\min\bigl(a, \frac bM, \frac1{2L}\bigr)$, $ I = [t_0 - T, t_0
+ T]$, और
\[
\mathcal E = \{x \in \mathcal C(I, \R^d) : \norm{x(t) - x_0}
\leq b\ \text{on } I\},
\]
बानाच स्थान का एक \omterm{def:b3:topology:interior}{बंद} उपसमुच्चय $(\mathcal C(I, \R^d),
\norm\cdot_\infty)$: \omterm{def:b3:complete:complete}{पूरा}
(\cref{def:b3:complete:complete}). परिभाषित करना $\Phi(x)(t) = x_0 +
\int_{t_0}^tf(s, x(s))\dd s $: के लिए $ x \in \mathcal E $,
$\norm{\Phi(x)(t) - x_0} \leq M\abs{t - t_0} \leq MT \leq b $
--- $\Phi $ एमएपीएस $\mathcal E $ स्वयं के लिए --- और के लिए $ x, y \in
\mathcal E $:
\[
\norm{\Phi(x)(t) - \Phi(y)(t)} \leq \Bigl|\int_{t_0}^t
L\,\norm{x(s) - y(s)}\,\dd s\Bigr| \leq LT\,\norm{x -
y}_\infty \leq \tfrac12\norm{x - y}_\infty :
\]
एक संकुचन. बानाच निश्चित बिंदु
(\cref{thm:b3:complete:banach}) में एक अद्वितीय निश्चित बिंदु देता है
$\mathcal E $: अस्तित्व, और समाधानों के बीच विशिष्टता
में रहना $\bar B(x_0, b)$ --- लेकिन कोई समाधान $ I $ रहता है
वहाँ ($\norm{x(t) - x_0} \leq M\abs{t - t_0} \leq b $ जब तक
जैसा कि ग्राफ़ में रहता है $ Q $, ए निरंतरता तर्क):
विशिष्टता पर $ I $.
\end{proof}

\begin{lemma}[ग्रा\"दीवार पर]\label{lem:b3:ode:gronwall}
होने देना $ u \colon I \to \intco0\infty $ होना \omterm{def:b3:topology:continuity}{निरंतर}, $ t_0 \in
I $, और मान लीजिए\index{Grönwall's lemma}
\[
u(t) \leq \alpha + \beta\,\Bigl|\int_{t_0}^{t}u(s)\,\dd
s\Bigr| \qquad (t \in I)
\]
साथ $\alpha \geq 0$, $\beta > 0$. तब $ u(t) \leq
\alpha\,\eu^{\beta\abs{t - t_0}}$ पर $ I $.
\end{lemma}

\begin{proof}
के लिए $ t \geq t_0$: होने देना $ v(t) = \alpha +
\beta\int_{t_0}^tu(s)\dd s $, इसलिए $ u \leq v $, $ v' = \beta u
\leq \beta v $, और $(v\eu^{-\beta(t - t_0)})' \leq 0$: $ v(t)
\leq v(t_0)\eu^{\beta(t-t_0)} = \alpha\eu^{\beta(t - t_0)}$.
के लिए $ t \leq t_0$, उसी को लागू करें $\tilde u(t) = u(2t_0 -
t)$.
\end{proof}

\begin{corollary}[विशिष्टता और निरंतर निर्भरता]
\label{cor:b3:ode:uniqueness}
की परिकल्पना के तहत \cref{thm:b3:ode:cauchylipschitz},
के दो समाधान $ x' = f(t,x)$ जो एक बिंदु पर सहमत हैं सहमत हैं
उनकी परिभाषा के \omterm{def:b3:groups:normal}{सामान्य} अंतराल पर। मात्रात्मक रूप से, यदि
$ x, y $ एक क्षेत्र में ग्राफ़ के साथ दो समाधान हैं $ f $ है
$ L $-लिप्सचिट्ज़ में $ x $, तब
\[
\norm{x(t) - y(t)} \leq \norm{x(t_0) -
y(t_0)}\,\eu^{L\abs{t - t_0}} .
\]
\end{corollary}

\begin{proof}
अनुमान: $ u = \norm{x - y}$ संतुष्ट $ u(t) \leq u(t_0)
+ L\abs{\int_{t_0}^tu}$ (अभिन्न समीकरण घटाएं);
ग्रा\"दीवार पर. वैश्विक विशिष्टता: समझौता सेट $\{t : x(t)
= y(t)\}$ \omterm{def:b3:groups:normal}{सामान्य} अंतराल में \omterm{def:b3:topology:interior}{बंद} है, गैररिक्त, और
खुला --- किसी भी समझौते बिंदु के आसपास, कवर करें कॉम्पैक्ट का टुकड़ा
आम ग्राफ़ को सीमित रूप से कई लिप्सचिट्ज़ बक्सों द्वारा लागू करें
अनुमान के साथ $ u(t_1) = 0$ प्रत्येक पर: स्थानीय स्तर पर $ x \equiv y $.
एक अंतराल का एक गैर-रिक्त खुला \omterm{def:b3:topology:interior}{बंद} उपसमुच्चय ही सब कुछ है।
\end{proof}

\section{अधिकतम समाधान}

\begin{theorem}[अधिकतम समाधान; कॉम्पैक्ट से बचें]
\label{thm:b3:ode:maximal}
मान लीजिए $ f $ \omterm{def:b3:topology:continuity}{निरंतर}, \omterm{def:b3:ode:lipschitz}{स्थानीय स्तर पर लिप्सचिट्ज़} में $ x $.
\begin{enumerate}
  \item प्रत्येक कॉची समस्या में एक विशिष्टता होती है \emph{\omterm{def:b3:rings:primemaximal}{अधिक से अधिक}}
        समाधान $ x \colon \intoo{T_-}{T_+} \to \R^d $: हर
        अन्य समाधान के माध्यम से $(t_0, x_0)$ इसका है
        प्रतिबंध. अंतराल खुला है.
  \item (\emph{पलायन}) हरएक के लिए कॉम्पैक्ट $ K \subseteq U $
        वहाँ है $\varepsilon > 0$ ऐसा है कि $(t, x(t)) \notin
        K $ सभी के लिए $ t \in \intoo{T_+ - \varepsilon}{T_+}$ (और
        सममित रूप से $ T_-$): \emph{ए का ग्राफ \omterm{thm:b3:ode:maximal}{अधिकतम
        समाधान} अंततः प्रत्येक को छोड़ देता है कॉम्पैक्ट का उपसमुच्चय
        $ U $}. विशेष रूप से,
        के लिए $ U = \R\times\R^d $ और $ T_+ < +\infty $:
        $\norm{x(t)} \to +\infty $ जैसा $ t \to T_+^-$
        (\emph{झटका}).\index{अधिकतम समाधान}\index{पलायन
        कॉम्पैक्ट से}\index{झटका}
\end{enumerate}
\end{theorem}

\begin{proof}
(1) चलो $\mathcal S $ के माध्यम से सभी समाधानों का समुच्चय बनें
$(t_0, x_0)$; द्वारा \cref{cor:b3:ode:uniqueness} किन्हीं दो पर सहमति है
उनके अंतरालों का प्रतिच्छेदन, इसलिए वे चिपकते हैं: पर $ J =
\bigcup_{y \in \mathcal S}I_y $, परिभाषित करना $ x(t) = y(t)$ किसी के लिए
$ y $ पर परिभाषित $ t $: जाहिर तौर पर एक अच्छी तरह से परिभाषित समाधान
अधिकतम और अद्वितीय. $ J $ खुला है: एक समाधान परिभाषित किया गया है
\omterm{thm:b3:complete:completion}{समापन} बिंदु को लम्बा खींचा जा सकता है
\cref{thm:b3:ode:cauchylipschitz} उस अंतिम बिंदु पर.

(2) मान लीजिए कि दावा विफल हो जाता है $ T_+$: वहाँ हैं $ t_n \to
T_+$ साथ $(t_n, x(t_n)) \in K $; इस बल पर ध्यान दें $ T_+ <
\infty $ या, यदि $ T_+ = \infty $, साबित करने के लिए कुछ भी नहीं है ($ K $
समय में बंधा हुआ है)। तो चलिए $ T_+ < \infty $. सघनता: एकसमान \omterm{def:b3:ode:stability}{स्थिर}ांक $ M, L, a,
b $ एक में सभी कॉची डेटा के लिए काम करें \omterm{def:b3:topology:topology}{अड़ोस-पड़ोस} का $ K $
--- ठोस रूप से, कवर करें $ K $ बहुत सारे बक्सों द्वारा $ Q_i $ जैसा कि
स्थानीय प्रमेय का प्रमाण और चलो $ T^* > 0$ न्यूनतम हो
संगत अस्तित्व समय: किसी भी कॉची डेटाम में $ K $
कम से कम जीवन जीने का एक समाधान लॉन्च करता है $ T^*$ इसके प्रारंभिक से परे
समय. इस पर लागू करना $(t_n, x(t_n))$ साथ $ t_n > T_+ -
T^*/2$ का विस्तार $ x $ अतीत $ T_+$ (एक्सटेंशन इससे सहमत है $ x $
विशिष्टता से, फिर इसे लम्बा खींचता है): विरोधाभास के साथ
अधिकतमता तो ग्राफ निकल जाता है $ K $ निश्चित रूप से पहले
$ T_+$. के लिए $ U = \R\times\R^d $: अगर $\norm{x(t)}\not\to\infty $,
एक क्रम $ t_n \to T_+$ रखता है $(t_n, x(t_n))$ में कॉम्पैक्ट
$[t_0, T_+]\times\bar B(0, R)$: छोड़ा गया।
\end{proof}

\begin{corollary}[रैखिक विकास के अंतर्गत वैश्विक अस्तित्व]
\label{cor:b3:ode:globallinear}
अगर $ U = I\times\R^d $ ($ I $ खुला अंतराल) और $\norm{f(t, x)}
\leq \alpha(t)\norm x + \beta(t)$ साथ $\alpha, \beta $
\omterm{def:b3:topology:continuity}{निरंतर}, फिर हर \omterm{thm:b3:ode:maximal}{अधिकतम समाधान} सभी पर परिभाषित किया गया है
$ I $.
\end{corollary}

\begin{proof}
एक पर कॉम्पैक्ट $[t_0, T] \subseteq I $: $\norm{x(t)} \leq
\norm{x_0} + \int_{t_0}^t(\alpha\norm x + \beta)$, तो द्वारा
ग्रा\"ऑनवॉल (साथ) $\alpha, \beta $ से घिरा $ A, B $ वहाँ)
$\norm{x(t)} \leq (\norm{x_0} + B(T - t_0))\eu^{A(T - t_0)}$:
घिरा हुआ. अगर $ T_+ < \sup I $, ग्राफ एक में रहता है कॉम्पैक्ट का
$ I\times\R^d $ पास में $ T_+$: विरोधाभास बच जाते हैं
(\cref{thm:b3:ode:maximal}).
\end{proof}

\section{रैखिक प्रणाली}

इस पूरे अनुभाग में $ A \colon I \to M_d(\R)$ और $ b
\colon I \to \R^d $ हैं \omterm{def:b3:topology:continuity}{निरंतर}; सिस्टम है $ x' = A(t)x
+ b(t)$ --- रैखिक विकास: सभी \omterm{thm:b3:ode:maximal}{अधिकतम समाधान} सब पर जियो
का $ I $ (\cref{cor:b3:ode:globallinear}).

\begin{theorem}[संरचना]\label{thm:b3:ode:linearstructure}
सजातीय प्रणाली के समाधान $ x' = A(t)x $ फॉर्म ए
$ d $-आयामी वेक्टर अंतरिक्ष $ S_H $; प्रत्येक के लिए $ t_0$, द
मूल्यांकन $ x \mapsto x(t_0)$ एक समरूपता है $ S_H \to
\R^d $. \emph{विश्लेषक}\index{विश्लेषक} $ R(t, s) \in GL_d(\R)$, द्वारा परिभाषित: $ t \mapsto R(t,
s)v $ मूल्य के साथ समाधान है $ v $ पर $ s $, संतुष्ट करता है
\[
R(s,s) = I,\quad R(t, u)R(u, s) = R(t, s),\quad
\partial_tR(t,s) = A(t)R(t,s),
\]
और अमानवीय समस्या का समाधान हो जाता है \emph{डुहामेल का
सूत्र}\index{डुहामेल का
सूत्र}:
\[
x(t) = R(t, t_0)\,x_0 + \int_{t_0}^{t}R(t, s)\,b(s)\,\dd s .
\]
अंततः \emph{रोन्स्कियन}\index{रोन्स्कियन} $ w(t) = \det
R(t, s)$ लिउविले के सूत्र का पालन करता है $ w' =
\operatorname{tr}A(t)\,w $, इसलिए $ w(t) =
\exp\bigl(\int_s^t\operatorname{tr}A\bigr) > 0$.
\end{theorem}

\begin{proof}
समीकरण की रैखिकता समाधान को एक सदिश समष्टि बनाती है;
मूल्यांकन रैखिक है, इंजेक्शन (विशिष्टता: एक समाधान)।
पर लुप्त हो रहा है $ t_0$ है $\equiv 0$) और विशेषण (अस्तित्व):
आयाम $ d $. \omterm{thm:b3:ode:linearstructure}{विश्लेषक} गुण विशिष्टता को पुनः स्थापित करते हैं
(प्रत्येक पहचान के दोनों पक्ष एक ही कॉची समस्या को हल करते हैं);
से उलटापन $ R(s,t)R(t,s) = I $. डुहामेल: अंतर करें
सूत्र --- $ x'(t) = A(t)R(t,t_0)x_0 + R(t,t)b(t) +
\int_{t_0}^tA(t)R(t,s)b(s)\dd s = A(t)x(t) + b(t)$
(अभिन्न के तहत भेदभाव वैध है:
इंटीग्रैंड है $\mathcal C^1$ में $ t $ साथ \omterm{def:b3:topology:continuity}{निरंतर} व्युत्पन्न
में $(t,s)$; या अभिन्न समीकरण द्वारा सत्यापित करें)। लिउविले:
$ w(t + h) = \det\bigl(R(t+h, t)\bigr)w(t)$ और $ R(t + h, t) =
I + hA(t) + o(h)$ (अभिन्न समीकरण से), तो $\det = 1
+ h\operatorname{tr}A(t) + o(h)$ (का विस्तार $\det $ पर
$ I $): $ w'(t) = \operatorname{tr}A(t)\,w(t)$; को एकीकृत करें
अदिश रैखिक ODE.
\end{proof}

\begin{theorem}[मैट्रिक्स घातांक]\label{thm:b3:ode:matrixexp}
के लिए $ A \in M_d(\C)$, श्रृंखला $\eu^{A} =
\sum_{n\geq0}\frac{A^n}{n!}$ अभिसरण (बिल्कुल, किसी में)।
उपगुणक मानदंड), $\eu^{A+B} = \eu^A\eu^B $ जब कभी भी
$ AB = BA $, और $ t \mapsto \eu^{tA}$ है \omterm{thm:b3:ode:linearstructure}{विश्लेषक} का
\omterm{def:b3:topology:continuity}{निरंतर} प्रणाली: $ R(t,s) = \eu^{(t-s)A}$; यह है $\mathcal
C^\infty $ साथ $\frac{\dd}{\dd t}\eu^{tA} = A\eu^{tA}$.
इसके अलावा: यदि $\operatorname{Re}\lambda < -\alpha < 0$ के लिए
प्रत्येक eigenvalue $\lambda $ का $ A $, तब $\vertiii{\eu^{tA}}
\leq C\eu^{-\alpha t}$ के लिए $ t \geq 0$.\index{मैट्रिक्स
घातीय}
\end{theorem}

\begin{proof}
अभिसरण: $\vertiii{A^n/n!} \leq \vertiii A^n/n!$, संक्षेपणीय
(\cref{exo:b3:complete:1}(बी) बानाच बीजगणित में
$ M_d $). आवागमन के लिए $ A, B $: दोनों का कॉची उत्पाद
द्विपद के माध्यम से बिल्कुल अभिसरण श्रृंखला पुनर्व्यवस्थित होती है
प्रमेय (मान्य जब $ AB = BA $), में
$\sum_n\frac{(A+B)^n}{n!}$. भिन्नता, सीधे:
$\frac{\eu^{(t+h)A} - \eu^{tA}}h = \eu^{tA}\frac{\eu^{hA} -
I}{h} \to \eu^{tA}A $ तब से $\norm{\frac{\eu^{hA} - I}h - A}
\leq \sum_{n\geq2}\frac{\abs h^{n-1}\vertiii A^n}{n!} =
O(h)$. इस तरह $ t \mapsto \eu^{(t - s)A}v $ कॉची को हल करता है
समस्या परिभाषित करना $ R(t,s)v $. वर्णक्रमीय सीमा: द्वारा \omterm{thm:b3:modules:jordan}{जॉर्डन फॉर्म}
(\cref{thm:b3:modules:jordan}), $ A = P(D + N)P^{-1}$ साथ $ D $
विकर्ण eigenvalues को ले जाने वाला, $ N $ निलपोटेंट, और $ DN =
ND $. तब $\eu^{tA} = P\,\eu^{tD}\eu^{tN}P^{-1}$ साथ
$\vertiii{\eu^{tD}} \leq \eu^{-(\alpha + \delta)t}$ कुछ के लिए
$\delta > 0$ ($ t \geq 0$) और $\eu^{tN}$ बहुपद में $ t $
(निलपोटेंस): उत्पाद है $\leq C\eu^{-\alpha t}$
(बहुपद द्वारा पीटा गया $\eu^{-\delta t}$).
\end{proof}

\begin{example}[विमान, वर्गीकृत]
\label{ex:b3:ode:planeclassification}
के लिए $ x' = Ax $ साथ $ A \in M_2(\R)$ उलटा, चरण
चित्र निकट $0$ द्वारा तय किया जाता है $\tau = \operatorname{tr}A $
और $\delta = \det A $, eigenvalues के माध्यम से $\lambda_\pm
= \frac{\tau \pm \sqrt{\tau^2 - 4\delta}}2$:
\begin{itemize}
  \item $\delta < 0$: विपरीत चिह्नों के वास्तविक eigenvalues ---
        ए \emph{काठी}; दो प्रक्षेप पथ प्रवेश करते हैं, दो प्रस्थान करते हैं,
        अन्य सभी उड़ जाते हैं। सदैव अस्थिर.
  \item $\delta > 0$, $\tau^2 \geq 4\delta $: वास्तविक स्वदेशी मूल्य
        एक ही चिन्ह का ($= \operatorname{sign}\tau $) --- ए
        \emph{नोड}, \omterm{def:b3:ode:stability}{स्थिर} आईएफएफ $\tau < 0$; प्रक्षेप पथ हैं
        धीमी ईजेनडायरेक्शन के स्पर्शरेखा।
  \item $\delta > 0$, $\tau^2 < 4\delta $, $\tau \neq 0$:
        जटिल संयुग्मी eigenvalues $\frac\tau2 \pm
        \iu\omega $ --- ए \emph{कुंडली} (\omterm{ex:b3:groups:actions}{केंद्र}), \omterm{def:b3:ode:stability}{स्थिर} आईएफएफ
        $\tau < 0$; समाधान हैं
        $\eu^{\tau t/2}\times $ अवधि का घूर्णन
        $\frac{2\pi}\omega $.
  \item $\tau = 0$, $\delta > 0$: पूर्णतः काल्पनिक
        eigenvalues --- ए \emph{\omterm{ex:b3:groups:actions}{केंद्र}}: \omterm{def:b3:topology:interior}{बंद} किया हुआ कक्षाओं
        (दीर्घवृत्त), स्पर्शोन्मुख स्थिरता के बिना स्थिरता,
        बिल्कुल यही सीमा रेखा
        \cref{thm:b3:ode:linearization} के लिए निर्णय नहीं ले सकता
        नॉनलाइनियर सिस्टम (पेंडुलम का निचला संतुलन,
        \cref{pb:b3:ode:1}, यहाँ बैठता है)।
\end{itemize}
सीमा परवलय $\tau^2 = 4\delta $ ले जाता है
पतित नोड्स (जॉर्डन ब्लॉक: एकल के साथ प्रक्षेपवक्र
स्पर्शरेखा दिशा). सब कुछ दो नंबरों से पढ़ा जाता है ---
यही कारण है कि किसी भी तलीय चरण से पहले पहला प्रतिवर्त
पोर्ट्रेट की गणना करना है $\operatorname{tr}$ और $\det $; उदा.\
$ A = \bigl(\begin{smallmatrix}0 & 1\\ -1 &
-c\end{smallmatrix}\bigr)$ (नम थरथरानवाला): $\delta = 1 >
0$, $\tau = -c $: \omterm{def:b3:ode:stability}{स्थिर} के लिए सर्पिल $0 < c < 2$, \omterm{def:b3:ode:stability}{स्थिर} नोड
के लिए $ c \geq 2$ --- अंडरडैम्पिंग बनाम ओवरडैम्पिंग, एक में
नज़र.
\end{example}

\section{प्रवाह, संतुलन, स्थिरता}

अब विचार करें \emph{स्वायत्त} समीकरण $ x' = F(x)$, $ F
\colon \Omega \to \R^d $ \omterm{def:b3:ode:lipschitz}{स्थानीय स्तर पर लिप्सचिट्ज़} खुले में $\Omega
\subseteq \R^d $. लिखना $\varphi_t(x_0) = x(t)$ के लिए
\omterm{thm:b3:ode:maximal}{अधिकतम समाधान} साथ $ x(0) = x_0$ (द
\emph{प्रवाह}\index{प्रवाह}); स्वायत्तता देता है
समूह संपत्ति $\varphi_{t+s} =
\varphi_t\circ\varphi_s $ जहां परिभाषित किया गया है (दोनों पक्ष हल करते हैं
समय पर वही समस्या $ s $).

\begin{definition}\label{def:b3:ode:stability}
एक \emph{संतुलन} एक बिंदु है $\bar x $ साथ $ F(\bar x) =
0$ (इसलिए $\varphi_t(\bar x) = \bar x $). यह है
\emph{स्थिर}\index{स्थिर} यदि प्रत्येक के लिए
$\varepsilon > 0$ वहाँ है $\delta > 0$ ऐसा है कि
$\norm{x_0 - \bar x} < \delta $ तात्पर्य यह है कि समाधान
सभी के लिए मौजूद है $ t \geq 0$ साथ $\norm{\varphi_t(x_0) - \bar
x} < \varepsilon $; \emph{स्पर्शोन्मुख रूप से स्थिर} यदि इसके अलावा
$\varphi_t(x_0) \to \bar x $ सभी के लिए $ x_0$ पास में $\bar x $.
\end{definition}

\begin{theorem}[ल्यपुनोव कार्य करता है]\label{thm:b3:ode:lyapunov}
होने देना $\bar x $ एक संतुलन हो और $ V \colon \mathcal V \to
\R $ ए $\mathcal C^1$ ए पर कार्य करें \omterm{def:b3:topology:topology}{अड़ोस-पड़ोस} का $\bar x $
साथ:\index{ल्यपुनोव समारोह}
\[
V(\bar x) = 0,\qquad V(x) > 0 \text{ for } x \neq \bar x,
\qquad \dot V(x) = \nabla V(x)\cdot F(x) \leq 0 .
\]
तब $\bar x $ है \omterm{def:b3:ode:stability}{स्थिर}. यदि इसके अलावा $\dot V < 0 $ \omterm{def:b3:topology:interior}{बंद} $\bar
x $, तब $\bar x $ स्पर्शोन्मुख है \omterm{def:b3:ode:stability}{स्थिर}.
\end{theorem}

\begin{proof}
एक समाधान के साथ, $\frac{\dd}{\dd t}V(x(t)) = \dot V(x(t))
\leq 0$: $ V $ घट जाती है. दिया गया $\varepsilon $ (काफ़ी छोटा
वह $\bar B(\bar x, \varepsilon) \subseteq \mathcal V $), होने देना
$ m = \min\{V(x) : \norm{x - \bar x} = \varepsilon\} > 0$
(संक्षिप्तता, सकारात्मकता) और चुनें $\delta < \varepsilon $
साथ $ V < m $ पर $ B(\bar x, \delta)$ (निरंतरता). एक समाधान
में शुरू $ B(\bar x, \delta)$ है $ V(x(t)) < m $ सबके लिए
बाद के समय में, इसलिए यह कभी भी गोले तक नहीं पहुंच सकता $\norm{x -
\bar x} = \varepsilon $ (कहाँ $ V \geq m $): यह में रहता है
गेंद --- और फिर सभी के लिए मौजूद है $ t \geq 0$: समाधान
में रहता है कॉम्पैक्ट $\bar B $, इसलिए
\cref{thm:b3:ode:maximal}(2) (कॉम्पैक्ट से बचें) बल
$ T_+ = +\infty $. स्थिरता.

स्पर्शोन्मुख मामला: चलो $ x(t)$ शुरू में $ B(\bar x, \delta)$;
$ V(x(t))$ कुछ तक कम हो जाता है $ c \geq 0$. अगर $ c > 0$: द
प्रक्षेपवक्र में रहता है $ K = \{x \in \bar B(\bar x,
\varepsilon): V(x) \geq c\}$, ए कॉम्पैक्ट वह सेट जिसमें a शामिल नहीं है
\omterm{def:b3:topology:topology}{अड़ोस-पड़ोस} का $\bar x $ ($ V $ है \omterm{def:b3:topology:continuity}{निरंतर} साथ $ V(\bar x)
= 0 < c $). पर $ K $, समारोह $\dot V $ है \omterm{def:b3:topology:continuity}{निरंतर},
इसलिए, सख्ती से नकारात्मक $\mu = \max_K\dot V < 0$
(कॉम्पैक्टनेस); तब $ V(x(t)) \leq V(x(0)) + \mu t \to
-\infty $: बेतुका, $ V \geq 0$. इसलिए
$ c = 0$, और $ x(t) \to \bar x $ (दूरी पर इंगित करता है $\geq
\rho $ से $\bar x $ गेंद के भीतर है $ V \geq m_\rho >
0$).
\end{proof}

\begin{theorem}[रैखिकरण द्वारा स्थिरता]
\label{thm:b3:ode:linearization}
होने देना $ F $ होना $\mathcal C^1$, $ F(\bar x) = 0$, $ A = DF(\bar x)$.
यदि प्रत्येक eigenvalue $ A $ है $\operatorname{Re}\lambda <
0$, तब $\bar x $ स्पर्शोन्मुख है \omterm{def:b3:ode:stability}{स्थिर}.\index{द्वारा स्थिरता
रैखिककरण}
\end{theorem}

\begin{proof}
अनुवाद $\bar x $ को $0$ और लिखा $ F(x) = Ax + g(x)$ साथ
$ g(x) = o(\norm x)$ ($\mathcal C^1$ भिन्नता)। चुनना
$\alpha > 0$ साथ $\vertiii{\eu^{tA}} \leq C\eu^{-\alpha t}$
($ t \geq 0$; \cref{thm:b3:ode:matrixexp}) और $ r > 0$ साथ
$\norm{g(x)} \leq \frac{\alpha}{2C}\norm x $ के लिए $\norm x \leq
r $. डुहामेल के साथ $ b(s) = g(x(s))$:
\[
x(t) = \eu^{tA}x_0 + \int_0^t\eu^{(t-s)A}g(x(s))\,\dd s,
\]
तक वैध है $\norm{x(s)} \leq r $. तब $ u(t) =
\eu^{\alpha t}\norm{x(t)}$ संतुष्ट
\[
u(t) \leq C\norm{x_0} +
\int_0^{t}C\,\frac{\alpha}{2C}\,u(s)\,\dd s ,
\]
तो जीआर\"ऑनवॉल देता है $ u(t) \leq C\norm{x_0}\eu^{\alpha t/2}$,
यानी\ $\norm{x(t)} \leq C\norm{x_0}\eu^{-\alpha t/2}$. अगर
$\norm{x_0} < r/C $, एक प्राथमिकता बंधी रहती है $\norm{x(t)} <
r $ सभी के लिए $ t $ (ए निरंतरता/बूटस्ट्रैप तर्क: का सेट
कई बार कहाँ $\norm x \leq r $ खुला और \omterm{def:b3:topology:interior}{बंद} है
$\intco0{T_+}$ सख्त अनुमान दिया गया है), समाधान है
वैश्विक, और यह अभिसरण होता है $0$ घातीय रूप से: स्पर्शोन्मुख
स्थिरता.
\end{proof}

\begin{method}\label{met:b3:ode:strategy}
एक ODE का सामना करना: (1) अस्तित्व/विशिष्टता --- स्थानीय जाँच करें
लिप्सचिट्ज़ (आमतौर पर $\mathcal C^1$); (2) वैश्विकता---रेखीय
विकास, सीमाबद्धता, या एक अपरिवर्तनीयता कॉम्पैक्ट ए के माध्यम से \omterm{thm:b3:ode:lyapunov}{ल्यपुनोव
समारोह} या प्रथम अभिन्न; ऐसा न करने पर विस्फोट का संदेह है
और अदिश कैरिकेचर पर परीक्षण करें $ x' = x^2$; (3) रैखिक
सिस्टम --- \omterm{thm:b3:ode:linearstructure}{विश्लेषक}, डुहामेल, और \omterm{def:b3:topology:continuity}{निरंतर} के लिए
की eigenstructure गुणांक $ A $; (4) गुणात्मक
प्रश्न --- संतुलन, रैखिकीकरण, और एक की तलाश \omterm{thm:b3:ode:lyapunov}{ल्यपुनोव
समारोह} (ऊर्जा, जब सिस्टम यांत्रिक है) या पहला
अभिन्न जिसका स्तर जाल प्रक्षेप पथ निर्धारित करता है। सप्ताहांत
समस्या पेंडुलम के माध्यम से पूरी विधि से चलती है।
\end{method}

\begin{exercise}[$\star $]\label{exo:b3:ode:1}
स्पष्ट रूप से हल करें और अधिकतम अंतराल निर्धारित करें:
(ए) $ x' = x^2$, $ x(0) = 1$; (बी) $ x' = 1 + x^2$, $ x(0) = 0$;
(सी) $ x' = x(1-x)$, $ x(0) = \frac12$. प्रत्येक उत्तर का मिलान करें
के साथ \cref{thm:b3:ode:maximal}(2) और
\cref{cor:b3:ode:globallinear}.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b3:ode:2}
होने देना $ x, y $ हल करना $ x' = f(t,x)$ साथ $ f $ विश्व स्तर पर
$ L $-लिप्सचिट्ज़ में $ x $ पर $\R\times\R^d $.
(ए) साबित करें $\norm{x(t) - y(t)} \leq \norm{x(0) -
y(0)}\eu^{L\abs t}$, और उदाहरण द्वारा दिखाएँ (रैखिक!) कि
कारक $\eu^{L\abs t}$ प्राप्त होता है.
(बी) निष्कर्ष निकालें कि \omterm{ex:b3:ode:planeclassification}{प्रवाह} नक्शा $ x_0 \mapsto x(t; x_0)$ है
\omterm{def:b3:topology:continuity}{निरंतर}, वास्तव में बंधे हुए सेट पर लिप्सचिट्ज़।
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:ode:3}
दिखाएँ कि निम्नलिखित में से प्रत्येक में सब कुछ है \omterm{thm:b3:ode:maximal}{अधिकतम समाधान}
वैश्विक पर $\R $, सही प्रमेय उद्धृत करते हुए:
(ए) $ x' = \sin(tx)$;
(बी) $ x' = \frac{t\,x}{1 + x^2}$;
(सी) $ x'' + q(t)x = 0$ साथ $ q $ \omterm{def:b3:topology:continuity}{निरंतर} (ए में कनवर्ट करें
प्रथम-क्रम प्रणाली);
(डी) $ x' = A(t)x $ साथ $ A $ \omterm{def:b3:topology:continuity}{निरंतर} और बाध्य --- और दे
जीआर\"ऑनवॉल बाउंड ऑन $\norm{x(t)}$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:ode:4}
(ए) गणना करें $\eu^{tA}$ के लिए
$ A = \bigl(\begin{smallmatrix}0 & -1\\ 1 &
0\end{smallmatrix}\bigr)$,
$\bigl(\begin{smallmatrix}\lambda & 1\\ 0 &
\lambda\end{smallmatrix}\bigr)$, और
$\bigl(\begin{smallmatrix}0 & 1\\ 1 &
0\end{smallmatrix}\bigr)$.
(बी) मजबूर थरथरानवाला को हल करें $ x'' + x = \cos(\omega t)$
डुहामेल द्वारा (सिस्टम फॉर्म), के लिए $\omega \neq 1$ और $\omega =
1$: अनुनाद धर्मनिरपेक्ष शब्द के रूप में प्रकट होता है $ t\sin t $.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:ode:5}
अदिश समीकरण के लिए $ x'' + p(t)x' + q(t)x = 0$:
(ए) दिखाएँ कि \omterm{thm:b3:ode:linearstructure}{रोन्स्कियन} $ w = x_1x_2' - x_1'x_2$ दो का
समाधान संतुष्ट करता है $ w' = -p\,w $ (हाबिल), और वह दो घटाएँ
समाधान के साथ $ w \neq 0$ कहीं न कहीं एक \omterm{def:b3:topology:basis}{आधार} बनता है.
(बी) एक गैर-लुप्तप्राय समाधान दिया गया है $ x_1$, \omterm{def:b3:groups:normal}{सामान्य} खोजें
द्वारा समाधान \emph{आदेश में कमी}: तय करना $ x_2 = x_1\int
\frac{\eu^{-\int p}}{x_1^2}$ और सत्यापित करें. पर लागू $ t^2x'' -
2x = 0$ पर $\intoo0{+\infty}$ साथ $ x_1(t) = t^2$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:ode:6}
(लॉजिस्टिक) के लिए $ x' = x(1 - x)$: सभी संतुलन निर्धारित करें और
उनकी स्थिरता (द्वारा \cref{thm:b3:ode:linearization} और
सीधे); हर समाधान दिखाएं $ x(0) \in \intoo01$ है
बढ़ती हुई, वैश्विक, सीमाओं के साथ $0$ और $1$ पर $\mp\infty $;
और पुष्टि करने के लिए स्पष्ट रूप से हल करें। इसे और अधिक सामान्यतः दिखाएँ
स्केलर स्वायत्त समाधान मोनोटोन हैं, और निष्कर्ष निकालते हैं:
आयाम में कोई गैर \omterm{def:b3:ode:stability}{स्थिर} आवधिक समाधान नहीं $1$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:ode:7}
(प्रथम अभिन्न) चलो $ H \colon \Omega \to \R $ होना $\mathcal
C^1$ और समतल हैमिल्टनियन प्रणाली पर विचार करें $ x' =
\partial_yH $, $ y' = -\partial_xH $.
(ए) वह दिखाओ $ H $ समाधान के साथ \omterm{def:b3:ode:stability}{स्थिर} है।
(बी) के लिए $ H = \frac{y^2}2 + \frac{x^4}4$: सभी समाधान दिखाएं
वैश्विक और परिबद्ध हैं, और वह मूल है \omterm{def:b3:ode:stability}{स्थिर}
(लायपुनोव: $ H $) हालांकि रैखिककरण
($\bigl(\begin{smallmatrix}0&1\\0&0\end{smallmatrix}\bigr)$)
है \emph{नहीं} स्पर्शोन्मुख रूप से \omterm{def:b3:ode:stability}{स्थिर}:रेखीयकरण हो सकता है
अनिर्णायक.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b3:ode:8}
(भीगा हुआ पेंडुलम) $ x'' + cx' + \sin x = 0$, $ c > 0$; प्रणाली:
$ x' = y $, $ y' = -\sin x - cy $.
(एक शो $ V(x, y) = \frac{y^2}2 + 1 - \cos x $ संतुष्ट $\dot
V = -cy^2 \leq 0$: मूल है \omterm{def:b3:ode:stability}{स्थिर}.
(बी) $\dot V $ संपूर्ण अक्ष पर लुप्त हो जाता है $ y = 0$: ल्यपुनोव का
सख्त मानदंड विफल रहता है. फिर भी स्पर्शोन्मुख स्थिरता सिद्ध करें,
रैखिककरण द्वारा (\cref{thm:b3:ode:linearization}): गणना करें
रेखीयकृत मैट्रिक्स के eigenvalues पर $(0,0)$ और जांचें
$\operatorname{Re} < 0$ हरएक के लिए $ c > 0$.
(सी) संतुलन पर क्या होता है $(\pi, 0)$? की गणना करें
रैखिककरण और निष्कर्ष (एक eigenvalue सकारात्मक:
अस्थिरता --- आप अस्थिरता कथन का उपयोग कर सकते हैं
अनौपचारिक रूप से या स्पष्ट रूप से बचने का समाधान तैयार करें
रैखिक प्रणाली)।
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:ode:9}
(विशिष्टता सीमा) के लिए $\alpha \in \intoo01$, वह दिखाओ
समस्या $ x' = \abs x^{\alpha}$, $ x(0) = 0$ असीम रूप से है
कई समाधान (अनुकूलित) \cref{pb:b3:complete:1}, भाग III).
इसके विपरीत दिखाएँ कि के लिए $\alpha = 1$ (अर्थात्\ $ x' = \abs
x $) समाधान के माध्यम से $0$ अद्वितीय है, और पहचानें
सटीक रूप से कौन सी परिकल्पना
\cref{thm:b3:ode:cauchylipschitz} दोनों को अलग करता है
मामले.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b3:ode:10}
(एक प्राथमिक सीमा जाल समाधान) चलो $ F \colon \R^d \to \R^d $
होना \omterm{def:b3:ode:lipschitz}{स्थानीय स्तर पर लिप्सचिट्ज़} साथ $\langle F(x), x\rangle \leq 0$
जब कभी भी $\norm x \geq R $.
(ए) दिखाओ कि \omterm{def:b3:topology:interior}{बंद} गेंद $\bar B(0, R)$ सकारात्मक है
अपरिवर्तनीय: अंदर से शुरू होने वाले समाधान अंदर ही रहते हैं $ t \geq
0$. \emph{(अगर $\norm{x(t_2)} > R $, पिछली बार पर विचार करें
$ t_1 < t_2$ साथ $\norm{x(t_1)} = R $ और अध्ययन
$\frac{\dd}{\dd t}\norm{x(t)}^2$ पर $\intcc{t_1}{t_2}$.)}
(बी) बॉल में डेटा के लिए वैश्विक फॉरवर्ड अस्तित्व का अनुमान लगाएं।
फिर ग्रेडिएंट सिस्टम का इलाज करें $ x' = -\nabla G(x)$, $ G \in
\mathcal C^2$ साथ $ G(x) \to +\infty $ जैसा $\norm x \to
\infty $: दिखाओ $ G $ समाधान के साथ घटता जाता है, प्रत्येक
समाधान (बाउंडेड) सबलेवल सेट में रहता है $\{G \leq
G(x_0)\}$, और वैश्विक अग्रगामी अस्तित्व का \omterm{thm:b3:complete:completion}{समापन} करें।
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:ode:11}
(तुलना करके उड़ाओ) विचार करें $ x' = x^2 + t^2$, $ x(0) =
1$.
(ए) दिखाओ \omterm{thm:b3:ode:maximal}{अधिकतम समाधान} कुछ पर मौजूद है $\intco0{T_+}$
साथ $ T_+ < \infty $: साथ तुलना करें $ y' = y^2$, $ y(0) = 1$
\emph{(आपको जिस तुलनात्मक प्रमेय की आवश्यकता है उसे सिद्ध करें: यदि $ x' \geq
F(x)$ और $ y' = F(y)$ साथ $ x(0) \geq y(0)$, तब $ x \geq
y $ जहां दोनों रहते हैं)}, और निष्कर्ष निकालें $ T_+ \leq 1$.
(बी) बाध्य $ T_+$ नीचे से: चालू $\intcc01$, $ x' \leq x^2 +
1$; सुपरसोल्यूशन से तुलना करें $ z' = z^2 + 1$, $ z(0) =
1$, द्वारा हल किया गया $ z(t) = \tan\bigl(t + \frac\pi4\bigr)$, और
निष्कर्ष $ T_+ \geq \frac\pi4$.
(सी) इकट्ठा करना $\frac\pi4 \leq T_+ \leq 1$ और फ्रेम करें
नैतिक: दाहिनी ओर का सुपरलीनियर विकास क्या है
वैश्विक अस्तित्व को नष्ट कर देता है (\cref{exo:b3:ode:3} होने के नाते
प्रतिबिंदु), सीमा का अभिसरण है
$\int^{\infty}\frac{\dd s}{F(s)}$.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b3:ode:12}
(स्टर्म की तुलना प्रमेय) चलो $ q_1 \leq q_2$ होना
\omterm{def:b3:topology:continuity}{निरंतर} एक अंतराल पर $ I $, और चलो $ u \neq 0$ हल करना $ u''
+ q_1u = 0$, $ v \neq 0$ हल करना $ v'' + q_2v = 0$.
(ए) स्थापित करें \omterm{thm:b3:ode:linearstructure}{रोन्स्कियन} पहचान: साथ $ W = uv' - u'v $,
$ W' = (q_1 - q_2)\,uv $.
(बी) (स्टर्म) दिखाएँ कि दो लगातार शून्य के बीच $ a < b $
का $ u $, दोनों में से एक $ v $ कहीं गायब हो जाता है $\intoo ab $, या
$ q_1 = q_2$ और $ v \propto u $ वहाँ \emph{(मान लीजिए $ u > 0$
पर $\intoo ab $ और $ v > 0$ बहुत; (ए) से एकीकृत करें $ a $ को
$ b $ और सीमा शर्तों के संकेतों का निरीक्षण करें $ W(a),
W(b)$)}.
(सी) निष्कर्ष निकालें: का समाधान $ u'' + qu = 0$ साथ $ q \geq m > 0$
लंबाई के प्रत्येक अंतराल में कम से कम एक बार लुप्त हो जाना
$\pi/\sqrt m $ (साथ तुलना करें $ v'' + mv = 0$); समाधान के साथ
$ q \leq 0$ \omterm{def:b3:rings:primemaximal}{अधिक से अधिक} एक बार गायब हो जाना $\R $. दोनों पर परीक्षण करें
$ q \equiv \pm1$.
\end{exercise}

\section{समस्या: पेंडुलम, पूरी तरह से हल हो गया}

\begin{problem}[{सप्ताहांत समस्या --- दोलन, घूर्णन,
पृथक्करण, और अवधि}]\label{pb:b3:ode:1}
पेंडुलम समीकरण $ x'' = -\sin x $ --- एक प्रणाली के रूप में: $ x' =
y $, $ y' = -\sin x $ पर $\R^2$ --- का ड्रोसोफिला है
गतिशीलता: लिखने में \omterm{def:b3:groups:simple}{सरल}, प्राथमिक रूप से हल करना असंभव
सूत्र, फिर भी गुणात्मक रूप से पूरी तरह समझने योग्य
विधि. होने देना $ E(x, y) = \frac{y^2}2 - \cos x $ (द
\emph{ऊर्जा}).

\medskip
\noindent\textbf{Part I --- Global structure.}
\begin{enumerate}
  \item वो सब दिखाओ \omterm{thm:b3:ode:maximal}{अधिकतम समाधान} वैश्विक हैं
        ($\R $-परिभाषित): उपयोग करें $\dot E = 0$ और
        \cref{thm:b3:ode:maximal}. संतुलन: $(k\pi, 0)$;
        उनके रैखिककरणों को वर्गीकृत करें (सम के लिए केंद्र-प्रकार)।
        $ k $, विषम के लिए काठी $ k $).
  \item दिखाएँ कि प्रक्षेप पथ स्तर में समाहित हैं
        सेट $\{E = E_0\}$, और उनके द्वारा स्केच/वर्णन करें
        का मूल्य $ E_0 \in \intco{-1}{+\infty}$: $ E_0 = -1$
        (संतुलन), $-1 < E_0 < 1$ (चारों ओर \omterm{def:b3:topology:interior}{बंद} मोड़
        $(2k\pi, 0)$), $ E_0 = 1$ (विभाजक के माध्यम से
        $(\pm\pi, 0)$), $ E_0 > 1$ (रेखांकित करें $ x $:
        घूर्णन)।
  \item सिद्ध कीजिए कि निचला संतुलन $(0,0)$ है \omterm{def:b3:ode:stability}{स्थिर}
        लेकिन \emph{नहीं} स्पर्शोन्मुख रूप से \omterm{def:b3:ode:stability}{स्थिर}. \emph{(ल्यपुनोव
        के साथ $ V = E + 1$; गैर-स्पर्शोन्मुख: ऊर्जा संरक्षण
        जाल कक्षाओं से दूर स्तर पर घटता है
        उत्पत्ति।)}
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Oscillations and their period.}
हल करना $-1 < E_0 < 1$ और लिखा $ E_0 = -\cos a $ साथ $ a \in
\intoo0\pi $ (आयाम).
\begin{enumerate}[resume]
  \item दिखाएँ कि समाधान के साथ $ x(0) = a $, $ y(0) = 0$
        दोलन करता है: $ x(t) \in \intcc{-a}a $, और यह \omterm{def:b3:groups:action}{कक्षा} है
        \omterm{def:b3:topology:interior}{बंद} वक्र $ y^2 = 2(\cos x - \cos a)$. औचित्य सिद्ध करें
        कि समाधान आवधिक है: \omterm{def:b3:groups:action}{कक्षा} एक है कॉम्पैक्ट
        संतुलन के बिना वक्र, गति सीमा पर पार किया गया
        नीचे --- इसे एक तर्क बनाएं (समाधान वापस आता है
        परिमित समय में अपने प्रारंभिक बिंदु तक, फिर विशिष्टता तक
        बल आवधिकता)।
  \item अवधि सूत्र स्थापित करें
        \[
        T(a) = 4\int_0^{a}\frac{\dd x}{\sqrt{2(\cos x -
        \cos a)}}
        \]
        \emph{(एक पर तिमाही-कक्षा, $ y = \frac{\dd x}{\dd t} >
        0$ और अलग चर; अनुचित को उचित ठहराओ
        पर अभिसरण $ x = a $)}.
  \item (छोटे दोलन) स्थानापन्न $\sin\frac x2 =
        \sin\frac a2\,\sin\varphi $ और दिखाओ
        \[
        T(a) = 4\int_0^{\pi/2}\frac{\dd\varphi}{\sqrt{1 -
        k^2\sin^2\varphi}},
        \qquad k = \sin\frac a2
        \]
        (ए \omterm{def:b3:complete:complete}{पूरा} अण्डाकार अभिन्न)। प्रभुत्व से निष्कर्ष निकालें
        अभिसरण कि $ T(a) \to 2\pi $ जैसा $ a \to 0^+$: द
        हार्मोनिक सीमा, आयाम से स्वतंत्र ---
        गैलीलियो की अनुमानित समकालिकता, उसकी सटीकता के साथ
        सुधार $ T(a) = 2\pi\bigl(1 + \frac{k^2}4 +
        O(k^4)\bigr)$ (इंटीग्रैंड का विस्तार करें और एकीकृत करें
        पद दर पद, \omterm{def:b3:groups:normal}{सामान्य} अभिसरण द्वारा उचित ठहराते हुए)।
  \item बताते हैं कि $ T(a) \to +\infty $ जैसा $ a \to \pi^-$
        \emph{(नीचे इंटीग्रैंड को पास से बांधें $\varphi =
        \frac\pi2$ कब $ k \to 1$, या मोनोटोन लागू करें
        अभिसरण)}: सेपरेट्रिक्स के पास, द
        पेंडुलम बिना किसी सीमा के धीमा होता है।
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The separatrix.}
\begin{enumerate}[resume]
  \item के लिए $ E_0 = 1$, $ y = 2\cos\frac x2$ ऊपर पर
        शाखा: चर को अलग करें और खोजने के लिए एकीकृत करें
        स्पष्ट समाधान
        \[
        x(t) = 4\arctan\bigl(\eu^{t}\bigr) - \pi
        \]
        (साथ $ x(0) = 0$, $ y(0) = 2$). उसे सीधे सत्यापित करें
        यह पेंडुलम समीकरण को हल करता है, और इसकी गणना करता है
        की सीमाएँ और सीमाएँ $ y(t)$ जैसा $ t \to
        \pm\infty $.
  \item निष्कर्ष: पृथक्करण \omterm{def:b3:groups:action}{कक्षा} काठी को जोड़ता है
        $(-\pi, 0)$ (जैसा $ t \to -\infty $) काठी तक
        $(\pi, 0)$ (जैसा $ t \to +\infty $) लेकिन दोनों तक नहीं पहुंचता है
        परिमित समय में --- विशिष्टता के अनुरूप (क्यों)
        सीमित समय में काठी तक पहुंचना विरोधाभासी होगा
        \cref{cor:b3:ode:uniqueness}?).
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Rotations, and the full
picture.}
\begin{enumerate}[resume]
  \item के लिए $ E_0 > 1$: दिखाओ $ y $ कभी गायब नहीं होता, $ x $ है
        सख्ती से नीरस और वैश्विक के साथ $ x(t) \to
        \pm\infty $, और $ t \mapsto y(t)$ के साथ आवधिक है
        अवधि
        \[
        \tau(E_0) = \int_{-\pi}^{\pi}
        \frac{\dd x}{\sqrt{2(E_0 + \cos x)}} .
        \]
  \item इकट्ठा करो \omterm{def:b3:complete:complete}{पूरा} चरण चित्र (अध्याय का)
        चित्र) प्रत्येक विशेषता के पूर्ण औचित्य के साथ, और
        विधि का दस पंक्तियों का सारांश लिखें: ऊर्जा,
        स्तर सेट, सघनता, विशिष्टता --- प्रत्येक कैसे
        अध्याय का प्रमेय दर्ज किया गया। हम कहाँ थे कभी
        \omterm{def:b3:groups:normal}{सामान्य} समाधान के लिए किसी सूत्र की आवश्यकता है?
\end{enumerate}

\medskip
\noindent\textbf{Part V --- The period function under the
microscope.}
\begin{enumerate}[resume]
  \item वालिस क्षणों को सिद्ध करें
        \[
        W_n = \int_0^{\pi/2}\sin^{2n}\varphi\,\dd\varphi
        = \frac\pi2\cdot\frac{(2n)!}{4^n\,(n!)^2}
        \]
        प्रेरण द्वारा (भागों द्वारा एकीकृत), विस्तार करें
        द्विपद श्रृंखला द्वारा प्रश्न 6 का समाकलन, और
        पूर्णता प्राप्त करने के लिए अवधि-दर-अवधि एकीकरण को उचित ठहराएँ
        शृंखला
        \[
        T(a) = 2\pi\sum_{n\geq0}
        \Bigl(\frac{(2n)!}{4^n(n!)^2}\Bigr)^{\!2}k^{2n}
        = 2\pi\Bigl(1 + \frac{k^2}4 + \frac{9k^4}{64}
        + O(k^6)\Bigr),
        \qquad k = \sin\frac a2 .
        \]
  \item आयाम में कनवर्ट करें:
        \[
        T(a) = 2\pi\Bigl(1 + \frac{a^2}{16} +
        \frac{11\,a^4}{3072} + O(a^6)\Bigr)
        \]
        \emph{(के विस्तार को प्रतिस्थापित करें $\sin\frac a2$
        और इकट्ठा करें)}. समकालिकता क्रम में विफल रहती है $ a^2$, और
        विफलता को अब ऑर्डर करने के लिए परिमाणित किया गया है $ a^4$.
  \item बताते हैं कि $ a \mapsto T(a)$ है \omterm{def:b3:topology:continuity}{निरंतर} और
        सख्ती से बढ़ रहा है $\intoo0\pi $, और निष्कर्ष निकालें
        प्रश्न 6-7 के साथ $ T $ से एक आक्षेप है
        $\intoo0\pi $ पर $\intoo{2\pi}{+\infty}$: हर
        सुपरक्रिटिकल काल का एहसास ठीक एक से होता है
        आयाम.
  \item (क्लॉकमेकर का अंकगणित) एक पेंडुलम पर विनियमित
        लुप्त आयाम रहता है \omterm{def:b3:rings:ideal}{आदर्श} समय; वह दौड़ दिखाओ
        आयाम पर $ a $ यह अंश से पिछड़ जाता है
        $\frac{a^2}{16} + O(a^4)$ का \omterm{def:b3:rings:ideal}{आदर्श} समय, और गणना
        के लिए बहाव $ a = 0.2$ रेड: के बारे में $216$ प्रति सेकंड
        दिन. (ह्यूजेन्स के साइक्लोइडल गाल और छोटे
        पलायन के \omterm{def:b3:topology:continuity}{निरंतर} आयाम दोनों उत्तर हैं
        इस नंबर पर.)
  \item घूर्णन अवधि पर लौटें $\tau $ प्रश्न का
        10: वो दिखाओ $\tau $ पर सख्ती से कमी आ रही है
        $\intoo1{+\infty}$, वह $\tau(E_0) \to +\infty $ जैसा
        $ E_0 \to 1^+$ (मोनोटोन अभिसरण), और वह
        $\sqrt{2E_0}\,\tau(E_0) \to 2\pi $ जैसा $ E_0 \to
        +\infty $ (प्रभुत्व अभिसरण): तेज़ चक्कर है
        कोणीय गति पर स्पर्शोन्मुख रूप से \omterm{def:b3:modules:free}{मुक्त} घूर्णन
        $\sqrt{2E_0}$.
\end{enumerate}

\noindent\textbf{Part VI --- The method exported:
Lotka--Volterra.} पेंडुलम का नुस्खा --- पहला अभिन्न,
कॉम्पैक्ट स्तर वक्र, विशिष्टता --- एक पारिस्थितिकी तंत्र को हल करती है।
हल करना $\alpha, \beta, \gamma, \delta > 0$ और विचार करें, पर
खुला चतुर्थांश $ Q = \intoo0{+\infty}\times\intoo0{+\infty}$,
\[
x' = x\,(\alpha - \beta y), \qquad
y' = y\,(\delta x - \gamma)
\]
($ x $ शिकार करना, $ y $ शिकारी)।
\begin{enumerate}[resume]
  \item बताते हैं कि $ Q $ अपरिवर्तनीय है --- अक्ष संघ हैं
        का कक्षाओं, स्पष्ट रूप से गणना योग्य, कोई समाधान नहीं
        पार हो सकता है (\cref{cor:b3:ode:uniqueness}) --- और
        इसमें अद्वितीय संतुलन है $ Q $ है $(x_*, y_*) =
        \bigl(\frac\gamma\delta, \frac\alpha\beta\bigr)$.
  \item बताते हैं कि
        \[
        H(x, y) = \delta x - \gamma\ln x + \beta y -
        \alpha\ln y
        \]
        पहला अभिन्न अंग है, वह $ H = f(x) + g(y)$ साथ
        $ f, g $ सख्ती से उत्तल और उचित पर
        $\intoo0{+\infty}$ मिनिमा के साथ $ x_*$, $ y_*$, और
        वह सब निष्कर्ष निकालो \omterm{thm:b3:ode:maximal}{अधिकतम समाधान} में $ Q $ वैश्विक हैं.
  \item उसके लिए दिखाओ $ h > h_* = H(x_*, y_*)$ स्तर निर्धारित
        $\{H = h\}\cap Q $ के चारों ओर एक \omterm{def:b3:topology:interior}{बंद} वक्र है
        संतुलन: दो \omterm{def:b3:topology:continuity}{निरंतर} शाखाओं $ y_\pm(x)$ खत्म
        ए कॉम्पैक्ट अंतराल $\intcc{x_-}{x_+} \ni x_*$, चिपका हुआ
        अंतिम बिंदु पर --- पेंडुलम का एनालॉग
        अंडाकार.
  \item सिद्ध करें कि प्रत्येक कोई भी संतुलन नहीं है \omterm{def:b3:groups:action}{कक्षा} में $ Q $ है
        आवधिक: वामावर्त परिसंचरण स्थापित करें
        रेखाओं द्वारा काटे गए चार क्षेत्रों से होकर $ x = x_*$
        और $ y = y_*$, प्रत्येक चाप के क्रॉसिंग समय को सीमित करें
        एक अभिसरण वर्गमूल के साथ एक अभिन्न अंग द्वारा
        विलक्षणता (जैसा कि प्रश्न 5 में है), और इसके करीब
        विशिष्टता (जैसा कि प्रश्न 4 में है)।
  \item (वोल्टेरा का औसत का नियम) यदि $ T $ अवधि है
        ऐसे का \omterm{def:b3:groups:action}{कक्षा}, बताते हैं कि
        \[
        \frac1T\int_0^Tx(t)\,\dd t = \frac\gamma\delta,
        \qquad
        \frac1T\int_0^Ty(t)\,\dd t = \frac\alpha\beta :
        \]
        समय का औसत संतुलन मूल्यों के बराबर होता है,
        आयाम जो भी हो \emph{(एकीकृत करें $(\ln x)' =
        \alpha - \beta y $ एक अवधि में)}.
  \item (मछली पकड़ने का विरोधाभास) दोनों प्रजातियों की कटाई दर पर करें
        $\varepsilon \in \intoo0\alpha $: सिस्टम रखता है
        इसके स्वरूप के साथ $\alpha - \varepsilon $ और $\gamma +
        \varepsilon $ की जगह $\alpha $ और $\gamma $. क्या
        औसत आबादी के साथ क्या होता है? समझाओ
        डी'एंकोना का अवलोकन (1914-1918): जब एड्रियाटिक
        युद्ध के दौरान मछली पकड़ने में कमी आई, का अनुपात
        शिकारी (शार्क) पकड़ में हैं \emph{बढ़ा हुआ} ---
        और क्यों मध्यम मछली पकड़ने से शिकार को फायदा होता है।
  \item दस-पंक्ति नैतिक लिखें: के कौन से प्रमेय
        प्रत्येक चरण में अध्याय शक्ति, क्या प्रतिस्थापित करती है
        पेंडुलम की ऊर्जा, और किसी भी प्रणाली की आवश्यकता क्यों नहीं ---
        या स्वीकार करता है --- एक प्राथमिक बंद-फ़ॉर्म समाधान।

  \item (स्पीड \omterm{def:b3:modules:module}{मॉड्यूल}ेशन) रोटेशन शासन में $ E_0 > 1$,
        बताते हैं कि $ y = x'$ के बीच दोलन करता है
        $\sqrt{2(E_0 - 1)}$ (पर $ x \equiv \pi \bmod 2\pi $)
        और $\sqrt{2(E_0 + 1)}$ (पर $ x \equiv 0$), कि
        समय का औसत $ y $ एक अवधि से अधिक बिल्कुल है
        $\frac{2\pi}{\tau(E_0)}$, और वह \omterm{def:b3:modules:module}{मॉड्यूल}ेशन
        अनुपात $\sqrt{\frac{E_0 + 1}{E_0 - 1}} \to 1$ जैसा $ E_0
        \to \infty $: तेजी से घूमना स्पर्शोन्मुख है
        वर्दी.
  \item (रोटेशन अवधि की एकरसता) वह दिखाओ
        $\tau(E_0)$ है $\mathcal C^1$ और सख्ती से
        पर घट रहा है $\intoo1{+\infty}$
        \emph{(अभिन्न चिह्न के अंतर्गत अंतर करें, साथ
        हर एक पर आधिपत्य $\intco{1 + \delta}\infty $)},
        साथ $\tau \to \infty $ जैसा $ E_0 \to 1^+$ और $\tau
        \to 0$ जैसा $ E_0 \to \infty $. \omterm{def:b3:complete:complete}{पूरा} इकट्ठा करो
        पेंडुलम का द्विभाजन चित्र
        ऊर्जा अक्ष: संतुलन पर $ E_0 = -1$, \omterm{def:b3:modules:free}{मुक्त}ि
        से अवधि बढ़ने के साथ $2\pi $ को $\infty $ पर
        $-1 < E_0 < 1$, सेपरेट्रिक्स पर $ E_0 = 1$, और
        अवधि घटने के साथ घूर्णन $\infty $ को
        $0$ आगे।
\end{enumerate}
\end{problem}
