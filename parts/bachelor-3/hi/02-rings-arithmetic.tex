\chapter{छल्ले और अंकगणित}\label{ch:b3:rings}

साधारण पूर्णांकों को अभाज्य संख्याओं में विशिष्ट रूप से विभाजित किया जाता है; बहुपद भी ऐसा ही करते हैं
एक मैदान के ऊपर. क्या ये दोनों तथ्य एक प्रमेय हैं? यह अध्याय उत्तर देता है
हाँ, और सटीक परिकल्पनाएँ ढूँढता है जो ``अंकगणित'' बनाती हैं
एक क्रमविनिमेय वलय में संभव: श्रृंखला
\[
\text{Euclidean} \;\Longrightarrow\; \text{principal}
\;\Longrightarrow\; \text{factorial (UFD)},
\]
सभी निहितार्थों को सिद्ध किया गया और सभी वार्तालापों का खंडन किया गया। सिद्धांत
फिर इसका परीक्षण किया जाता है कि यह अपना स्थान कहां अर्जित करता है: गाऊसी पूर्णांक
$\Z[\iu]$ (जो फ़र्मेट के दो-वर्ग प्रमेय को क्रैक कर देगा
सप्ताहांत समस्या), कई चरों में बहुपद वलय (गॉस)।
लेम्मा, Eisensteinकी कसौटी), और \omterm{def:b3:rings:noetherian}{नोथेरियन} रिंग्स, समापन
हिल्बर्ट के आधार प्रमेय में. लगातार, \emph{अँगूठी} मतलब
इकाई के साथ क्रमविनिमेय वलय $1 \neq 0$; वर्ष 2 खंड \omterm{def:b3:rings:ideal}{आदर्श}ों
का $\Z $ और $ K[X]$ हमारे दो मार्गदर्शक उदाहरण हैं।

\section{आदर्श, भागफल और समरूपता प्रमेय}

\begin{definition}\label{def:b3:rings:ideal}
एक \emph{आदर्श}\index{आदर्श} $ I $ एक अंगूठी का $ A $ एक योज्य है
उपसमूह ऐसा कि $ AI \subseteq I $. \emph{भागफल
अंगूठी}\index{भागफल
अंगूठी} $ A/I $ है \omterm{thm:b3:groups:quotient}{भागफल समूह} $(A, +)/I $
गुणन के साथ $(a + I)(b + I) = ab + I $: अच्छी तरह से परिभाषित,
बदलने के बाद से $ a $ को $ a + x $, $ b $ को $ b + y $ ($ x, y \in I $)
परिवर्तन $ ab $ द्वारा $ ay + xb + xy \in I $. प्रक्षेपण $\pi \colon A
\to A/I $ कर्नेल के साथ एक विशेषण वलय आकारिकी है $ I $, और गुठली
वलय आकारिकी बिल्कुल आदर्श हैं।
\end{definition}

\begin{theorem}[प्रथम समरूपता प्रमेय]\label{thm:b3:rings:firstiso}
अगर $ f \colon A \to B $ तो फिर, यह एक रिंग आकारिकी है $\bar f\colon
A/\ker f \to \operatorname{im} f $, $ a + \ker f \mapsto f(a)$, एक है
वलय समरूपता. आम तौर पर अधिक $ f $ के माध्यम से कारक $ A/I $ किसी के लिए
\omterm{def:b3:rings:ideal}{आदर्श} $ I \subseteq \ker f $. \omterm{def:b3:rings:ideal}{आदर्श}ों का $ A/I $ हैं $ J/I $ के लिए
$ J \supseteq I $ एक \omterm{def:b3:rings:ideal}{आदर्श} का $ A $ (पत्राचार प्रमेय).
\end{theorem}

\begin{proof}
समूहों के लिए (\cref{thm:b3:groups:firstiso,%
thm:b3:groups:correspondence}), यह देखते हुए कि सभी मानचित्र भी दृष्टि में हैं
उत्पादों का सम्मान करें: $\bar f $ अच्छी तरह से परिभाषित है, पर विशेषण
छवि, और गुणक; पत्राचार $ J \mapsto J/I $,
$\bar J \mapsto \pi^{-1}(\bar J)$ बरकरार रखता है \omterm{def:b3:rings:ideal}{आदर्श}ों दोनों में
दिशा-निर्देश क्योंकि $\pi $ एक विशेषण वलय आकारिकी है।
\end{proof}

\begin{definition}\label{def:b3:rings:primemaximal}
होने देना $ I \subsetneq A $ उचित बनो \omterm{def:b3:rings:ideal}{आदर्श}. $ I $ है
\emph{मुख्य}\index{मुख्य} अगर $ ab \in I \Rightarrow a \in I $
या $ b \in I $; $ I $ है \emph{अधिक से अधिक}\index{अधिक से अधिक} अगर कोई नहीं
\omterm{def:b3:rings:ideal}{आदर्श} सख्ती से बीच में स्थित है $ I $ और $ A $.
\end{definition}

\begin{proposition}\label{prop:b3:rings:primemaximal}
$ I $ प्रधान है $\iff $ $ A/I $ एक अभिन्न डोमेन है; $ I $ है \omterm{def:b3:rings:primemaximal}{अधिक से अधिक}
$\iff $ $ A/I $ एक मैदान है. विशेष रूप से अधिकतम \omterm{def:b3:rings:ideal}{आदर्श} प्रमुख हैं.
\end{proposition}

\begin{proof}
लिखना $\bar a $ में कक्षाओं के लिए $ A/I $. ``$ I $ प्राइम'' अनुवाद करता है
शब्दशः ``$\bar a\bar b = 0 \Rightarrow \bar a = 0$ या $\bar b
= 0$'', और $ A/I \neq 0$ को $ I \neq A $: यही परिभाषा है
एक डोमेन. अधिकतमता के लिए, पत्राचार प्रमेय का उपयोग करें: नहीं \omterm{def:b3:rings:ideal}{आदर्श}
सख्ती से बीच में $ I $ और $ A $ $\iff $ $ A/I $ है कोई \omterm{def:b3:rings:ideal}{आदर्श} के अलावा अन्य
$0$ और खुद $\iff $ $ A/I $ एक फ़ील्ड है --- अंतिम चरण के लिए: a में
एकमात्र फ़ील्ड \omterm{def:b3:rings:ideal}{आदर्श}ों हैं $0$ और सब कुछ (ए \omterm{def:b3:rings:ideal}{आदर्श} युक्त
$ x \ne 0$ रोकना $ x^{-1}x = 1$); इसके विपरीत यदि प्रत्येक अशून्य $ x $
इकाई उत्पन्न करता है \omterm{def:b3:rings:ideal}{आदर्श}, तब $ xy = 1$ कुछ के लिए $ y $. फ़ील्ड हैं
डोमेन, तो अधिकतम \omterm{def:b3:rings:ideal}{आदर्श} प्रमुख हैं.
\end{proof}

\begin{example}\label{ex:b3:rings:primemaximal}
में $\Z $: द प्रमुख \omterm{def:b3:rings:ideal}{आदर्श} हैं $(0)$ और यह $(p)$, $ p $ मुख्य;
\omterm{def:b3:rings:primemaximal}{अधिक से अधिक} वाले हैं $(p)$ ($\Z/p\Z = \mathbb F_p $ एक मैदान है,
$\Z/(0) = \Z $ क्या नहीं है)। में $ K[X, Y]$: $(X) \subsetneq (X, Y)$ हैं
दोनों अभाज्य ($ K[X,Y]/(X) \cong K[Y]$, एक डोमेन;
$ K[X,Y]/(X,Y) \cong K $, एक फ़ील्ड), तो $(X)$ प्रधान है लेकिन नहीं
\omterm{def:b3:rings:primemaximal}{अधिक से अधिक}.
\end{example}

इसकी गारंटी देने के लिए अधिकतम \omterm{def:b3:rings:ideal}{आदर्श} \emph{अस्तित्व} पूर्ण व्यापकता में,
हमें एक सेट-सैद्धांतिक सिद्धांत की आवश्यकता है। आंशिक रूप से ऑर्डर किया गया सेट है
\emph{अधिष्ठापन का} यदि प्रत्येक पूर्णतः आदेशित उपसमुच्चय (\emph{जंजीर})
एक ऊपरी सीमा है.

\begin{theorem}[ज़ोर्न की लेम्मा]\label{thm:b3:rings:zorn}
प्रत्येक गैर-रिक्त आगमनात्मक आंशिक रूप से ऑर्डर किए गए सेट में एक होता है \omterm{def:b3:rings:primemaximal}{अधिक से अधिक}
तत्व।\index{ज़ोर्न की लेम्मा}
\admitted
\end{theorem}

\begin{remark}\label{rem:b3:rings:zorn}
यह कोई साधारण गणित का प्रमेय नहीं बल्कि एक प्रमेय है \emph{स्वयंसिद्ध}:
यह सेट के मूल ज़र्मेलो--फ़्रैन्केल सिद्धांतों के समतुल्य है
सिद्धांत, पसंद के सिद्धांत के अनुसार (``गैर-रिक्त सेटों का प्रत्येक उत्पाद
गैर-रिक्त है''), जिसे हम इस पुस्तक में स्वीकार करते हैं। हम प्रत्येक को ध्वजांकित करते हैं
उपयोग करें. विश्लेषण इसे फिर से लागू करेगा (हैन--बानाच,
\cref{ch:b3:banach}).
\end{remark}

\begin{theorem}[क्रुल]\label{thm:b3:rings:krull}
हर उचित \omterm{def:b3:rings:ideal}{आदर्श} $ I \subsetneq A $ ए में निहित है अधिकतम
\omterm{def:b3:rings:ideal}{आदर्श}.\index{क्रुल का प्रमेय}
\end{theorem}

\begin{proof}
सेट को शामिल करके ऑर्डर करें $\mathcal E $ उचित का \omterm{def:b3:rings:ideal}{आदर्श}ों
युक्त $ I $; यह गैर-रिक्त है ($ I \in \mathcal E $). एक जंजीर
$(J_\lambda)$ में $\mathcal E $ ऊपरी सीमा है $ J = \bigcup
J_\lambda $: एक \omterm{def:b3:rings:ideal}{आदर्श} (कोई $ a, b \in J $ एक आम में झूठ बोलना $ J_\lambda $
समग्रता से), उचित ($1 \notin J_\lambda $ सभी के लिए $\lambda $),
युक्त $ I $. ज़ोर्न का लेम्मा एक उपज देता है \omterm{def:b3:rings:primemaximal}{अधिक से अधिक} का तत्व $\mathcal
E $, जो एक है अधिकतम \omterm{def:b3:rings:ideal}{आदर्श} युक्त $ I $ (एक \omterm{def:b3:rings:ideal}{आदर्श} सख्ती से
इसके ऊपर और उचित में झूठ होगा $\mathcal E $).
\end{proof}

\begin{theorem}[चीनी शेषफल प्रमेय]\label{thm:b3:rings:crt}
होने देना $ I_1, \dots, I_n $ जोड़ीदार हो \emph{चरमोत्कर्ष}\index{चरमोत्कर्ष} \omterm{def:b3:rings:ideal}{आदर्श}ों का $ A $ ($ I_k + I_l = A $ के लिए $ k \neq l $). तब
\[
A\Big/\bigcap_{k=1}^n I_k \;\xrightarrow{\;\sim\;}\;
\prod_{k=1}^n A/I_k,
\qquad
a \longmapsto (a + I_1, \dots, a + I_n),
\]
और इसके अलावा $\bigcap_k I_k = I_1 I_2 \cdots I_n $ (द \omterm{def:b3:rings:ideal}{आदर्श}
उत्पादों द्वारा उत्पन्न)।\index{चीनी शेषफल प्रमेय}
\end{theorem}

\begin{proof}
वो नक्शा $ f(a) = (a + I_k)_k $ कर्नेल के साथ एक रिंग आकारिकी है
$\bigcap I_k $; द्वारा \cref{thm:b3:rings:firstiso} यह पर्याप्त है
प्रत्यक्षता सिद्ध करें. हल करना $ k $; प्रत्येक के लिए $ l \neq k $ लिखना $1 = u_l +
v_l $ साथ $ u_l \in I_k $, $ v_l \in I_l $ (comaximality). तब
\[
e_k = \prod_{l \neq k} v_l = \prod_{l\neq k}(1 - u_l)
\equiv 1 \pmod{I_k},
\qquad e_k \in I_l \ (l \neq k),
\]
इसलिए $ f(e_k) = (0, \dots, 1, \dots, 0)$; एक लक्ष्य दिया गया $(a_k +
I_k)_k $, तत्व $\sum_k a_k e_k $ इसे मैप करें.

उत्पाद बनाम प्रतिच्छेदन: $ I_1\cdots I_n \subseteq \bigcap I_k $
हमेशा। इसके विपरीत, प्रेरण द्वारा यह उपचार के लिए पर्याप्त है $ n = 2$
(एक जाँच करता है $ I_1$ और $ I_2\cdots I_n $ हैं commaximal: गुणा करना
$1 = u_l + v_l $ ऊपर $ l \geq 2$ देता है $1 \in I_1 + I_2\cdots
I_n $). के लिए $ n = 2$: लिखना $1 = u + v $, $ u \in I_1$, $ v \in I_2$;
के लिए $ x \in I_1 \cap I_2$, $ x = xu + xv \in I_2I_1 + I_1I_2 =
I_1I_2$.
\end{proof}

\begin{example}\label{ex:b3:rings:crtz}
में $\Z $ साथ $ I_k = (m_k)$, $ m_k $ जोड़ीवार सहअभाज्य: $\Z/(m_1\cdots
m_n)\Z \cong \prod \Z/m_k\Z $ --- वर्ष 2 खंड का चीनी
शेषफल प्रमेय. इकाइयों तक सीमित: $(\Z/mn\Z)^\times \cong
(\Z/m\Z)^\times \times (\Z/n\Z)^\times $ के लिए $\gcd(m,n)=1$, कहाँ से
यूलर की बहुलता $\varphi $
(\cref{exo:b3:rings:8}).
\end{example}

\section{विभाज्यता: यूक्लिडियन, प्रिंसिपल, फैक्टोरियल}

\begin{definition}\label{def:b3:rings:divisibility}
होने देना $ A $ एक अभिन्न डोमेन बनें, $ a, b \in A $. हम कहते हैं $ a $
\emph{विभाजित} $ b $ ($ a \mid b $) अगर $ b \in (a) = aA $. तत्वों $ a,
b $ हैं \emph{एसोसिएट्स}\index{एसोसिएट्स} अगर $ a = ub $
साथ $ u \in A^\times $ (समान रूप से $(a) = (b)$). एक शून्येतर
गैर इकाई $ p $ है:
\begin{itemize}
  \item \emph{अलघुकरणीय}\index{अलघुकरणीय} अगर $ p = ab $
        बल $ a \in A^\times $ या $ b \in A^\times $;
  \item \emph{मुख्य}\index{मुख्य} अगर $ p \mid ab $ बल $ p
        \mid a $ या $ p \mid b $ (अर्थात्\  \omterm{def:b3:rings:ideal}{आदर्श} $(p)$ प्रधान है)।
\end{itemize}
\end{definition}

\begin{proposition}\label{prop:b3:rings:primeirred}
किसी भी डोमेन में, प्राइम $\Rightarrow $ \omterm{def:b3:rings:divisibility}{अलघुकरणीय}. उलटा है
\omterm{def:b3:groups:normal}{सामान्य} रूप से असत्य: में $\Z[\iu\sqrt 5] = \{a + \iu b\sqrt5 : a, b
\in \Z\}$, तत्व $2$ है \omterm{def:b3:rings:divisibility}{अलघुकरणीय} लेकिन प्राइम नहीं.
\end{proposition}

\begin{proof}
होने देना $ p $ प्रधान हो और $ p = ab $. तब $ p \mid ab $, कहना $ p \mid a $:
$ a = pc $, इसलिए $ p = pcb $, और रद्द करना $ p $ (डोमेन!) देता है $ cb =
1$: $ b \in A^\times $.

में $\Z[\iu\sqrt5]$, मानक का उपयोग करें $ N(x + \iu y\sqrt 5) = x^2 +
5y^2$, जो गुणक है (यह है $\abs z^2$). अगर $2 = ab $
साथ $ a, b $ फिर, गैर-इकाइयाँ $4 = N(a)N(b)$ साथ $ N(a), N(b) \neq
1$ (मानदंड-$1$ तत्व हैं $\pm1$, इकाइयां), तो $ N(a) = 2$:
असंभव, $ x^2 + 5y^2 = 2$ कोई पूर्णांक समाधान नहीं है. इसलिए $2$ है
\omterm{def:b3:rings:divisibility}{अलघुकरणीय}. लेकिन $2 \mid 6 = (1 + \iu\sqrt5)(1 - \iu\sqrt5)$
जबकि $2$ न तो कारक को विभाजित करता है ($\frac12 \pm \frac{\iu\sqrt5}2
\notin \Z[\iu\sqrt5]$): प्रधान नहीं.
\end{proof}

\begin{definition}\label{def:b3:rings:pidufd}
एक अभिन्न डोमेन $ A $ है:
\begin{itemize}
  \item \emph{इयूक्लिडियन}\index{इयूक्लिडियन} अगर कोई नक्शा है
        $\nu \colon A \setminus \{0\} \to \N $ (ए \emph{यूक्लिडियन
        समारोह}) ऐसा कि सभी के लिए $ a, b $ साथ $ b \ne 0$ वहां है $ q, r $ साथ $ a = bq + r $ और ($ r = 0$ या $\nu(r) <
        \nu(b)$);
  \item \emph{प्रधानाचार्य} (ए \emph{पीआईडी}\index{पीआईडी}) यदि प्रत्येक \omterm{def:b3:rings:ideal}{आदर्श} स्वरूप का है $(a)$;
  \item \emph{कारख़ाने का} (ए \emph{यूएफडी}\index{यूएफडी}) यदि प्रत्येक शून्येतर गैरइकाई का एक उत्पाद है
        इरेड्यूसिबल्स, विशिष्ट रूप से ऑर्डर पर निर्भर और \omterm{def:b3:rings:divisibility}{एसोसिएट्स}.
\end{itemize}
\end{definition}

\begin{theorem}\label{thm:b3:rings:euclideanpid}
\omterm{def:b3:rings:pidufd}{इयूक्लिडियन} $\Rightarrow $ प्रधानाचार्य।
\end{theorem}

\begin{proof}
होने देना $ I \neq (0)$ एक हो \omterm{def:b3:rings:ideal}{आदर्श} और $ b \in I \setminus\{0\}$ साथ
$\nu(b)$ न्यूनतम. के लिए $ a \in I $, विभाजित करना: $ a = bq + r $; तब $ r = a
- bq \in I $, और $\nu(r) < \nu(b)$ न्यूनतमता का खंडन करेगा, इसलिए
$ r = 0$ और $ a \in (b)$: $ I = (b)$.
\end{proof}

\begin{example}\label{ex:b3:rings:euclidean}

\begin{lemma}[प्रमुख आदर्शों की चढ़ती शृंखलाएँ]
\label{lem:b3:rings:acc}
में एक \omterm{def:b3:rings:pidufd}{पीआईडी}, हर बढ़ता हुआ क्रम \omterm{def:b3:rings:ideal}{आदर्श}ों $ I_1 \subseteq I_2
\subseteq \cdots $ अंततः स्थिर है.
\end{lemma}

\begin{proof}
$ I = \bigcup_n I_n $ एक \omterm{def:b3:rings:ideal}{आदर्श} (संघ बढ़ रहा है), इसलिए $ I =
(a)$; तत्व $ a $ कुछ में झूठ है $ I_N $, और तब $ I = (a)
\subseteq I_N \subseteq I_n \subseteq I $ के लिए $ n \geq N $.
\end{proof}

\begin{lemma}[बीézout; यूक्लिड की प्रमेयिका]\label{lem:b3:rings:bezout}
होने देना $ A $ एक हो \omterm{def:b3:rings:pidufd}{पीआईडी} और $ a, b \in A $. तब $(a) + (b) = (d)$ कुछ के लिए
$ d $, ए \emph{महत्तम सामान्य भाजक}\index{महत्तम सामान्य भाजक}: $ d \mid a $, $ d \mid b $, और प्रत्येक \omterm{def:b3:groups:normal}{सामान्य} भाजक $ a,
b $ विभाजित $ d $; इसके अतिरिक्त $ d = au + bv $ कुछ के लिए $ u, v $
(बी\'ईज़आउट)। फलस्वरूप प्रत्येक अघुलनशील तत्व एक का \omterm{def:b3:rings:pidufd}{पीआईडी} है
प्रधान.
\end{lemma}

\begin{proof}
$(a) + (b)$ एक \omterm{def:b3:rings:ideal}{आदर्श}, इस तरह $(d)$; $ a, b \in (d)$ देता है $ d \mid
a, b $; और $ d = au + bv \in (a) + (b)$. एक \omterm{def:b3:groups:normal}{सामान्य} भाजक $ c $ का
$ a, b $ विभाजित $ au + bv = d $.

यूक्लिड: चलो $ p $ होना \omterm{def:b3:rings:divisibility}{अलघुकरणीय}, $ p \mid ab $, $ p \nmid a $. एक जीसीडी
$ d $ का $ p $ और $ a $ विभाजित $ p $, इसलिए $ d $ एक इकाई या सहयोगी है
का $ p $ (इरेड्यूसिबिलिटी); सहयोगी को बाहर रखा गया है $ p \nmid a $. इसलिए
$1 = pu + av $, कहाँ से $ b = pub + abv $, और $ p $ दोनों पदों को विभाजित करता है:
$ p \mid b $.
\end{proof}

\begin{theorem}\label{thm:b3:rings:pidufd}
प्रधानाचार्य $\Rightarrow $ भाज्य.
\end{theorem}

\begin{proof}
\emph{अस्तित्व।} मान लीजिए कुछ अशून्य अ-इकाई $ a $ नहीं है
में गुणनखंडन इरेड्यूसिबल्स. तब $ a $ क्या नहीं है \omterm{def:b3:rings:divisibility}{अलघुकरणीय}: $ a =
a_1b_1$ दोनों कारकों के साथ गैर-इकाइयाँ; उनमें से कम से कम एक, कहो
$ a_1$, फिर से कोई गुणनखंड नहीं है (दो गुणनखंडों का एक उत्पाद)।
तत्व गुणनखंडनीय है)। पुनरावृत्त करने पर, हम पाते हैं $ a = a_0, a_1, a_2,
\dots $, प्रत्येक बिना किसी गुणनखंडन के अंतिम का एक उचित भाजक है,
तो $(a_0) \subsetneq (a_1) \subsetneq (a_2) \subsetneq \cdots $
--- समावेशन सख्त हैं क्योंकि $ a_n = a_{n+1}c $ साथ $ c $ ए
नॉनयूनिट का मतलब है $(a_n) = (a_{n+1})$ जबरदस्ती करेगा $ c \in A^\times $
(एक डोमेन में रद्द करें)। यह विरोधाभासी है \cref{lem:b3:rings:acc}.

\emph{विशिष्टता.} होने देना $ p_1 \cdots p_r = q_1 \cdots q_s $ सबके साथ
कारक \omterm{def:b3:rings:divisibility}{अलघुकरणीय}, $ r \leq s $, प्रेरण द्वारा $ r $. प्रधान
(\cref{lem:b3:rings:bezout}) $ p_1$ दाहिनी ओर को विभाजित करता है, इसलिए
कुछ को बाँट देता है $ q_j $; अंकित करना $ j = 1$. जैसा $ q_1$ है \omterm{def:b3:rings:divisibility}{अलघुकरणीय} और
$ p_1$ एक इकाई नहीं है, $ q_1 = u p_1$ साथ $ u \in A^\times $: $ p_1,
q_1$ हैं \omterm{def:b3:rings:divisibility}{एसोसिएट्स}. रद्द करना $ p_1$: $ p_2 \cdots p_r = (u q_2)
q_3\cdots q_s $ और प्रेरण द्वारा निष्कर्ष निकालें ($ r = 1$ बल $ s = 1$:
एक इकाई बार इरेड्यूसिबल्स हो नहीं सकता $1$).
\end{proof}

\begin{remark}\label{rem:b3:rings:ufdgcd}
में एक \omterm{def:b3:rings:pidufd}{यूएफडी}, जीसीडी मौजूद हैं (न्यूनतम घातांक लें)।
गुणनखंडन) और यूक्लिड की प्रमेयिका रखती है --- \omterm{def:b3:rings:divisibility}{अलघुकरणीय} $=$ मुख्य
(\cref{exo:b3:rings:2}) --- लेकिन बीézout विफल हो सकता है: में $\Z[X]$,
$\gcd(2, X) = 1$ अभी तक $1 \neq 2U + XV $ (पर मूल्यांकन करें $ X = 0$:
$1 = 2U(0)$, असंभव)। बी\'ईज़आउट पहचान विशिष्ट हैं
की संपत्ति \omterm{def:b3:rings:pidufd}{पीआईडी}.
\end{remark}

\begin{example}[अद्वितीय गुणनखंडन के बिना एक अंगूठी]
\label{ex:b3:rings:nonufd}
कोई भी निहितार्थ नहीं \omterm{def:b3:rings:pidufd}{इयूक्लिडियन} $\Rightarrow $ \omterm{def:b3:rings:pidufd}{पीआईडी}
$\Rightarrow $ \omterm{def:b3:rings:pidufd}{यूएफडी} एक तुल्यता है, और की विफलता है
आखिरी बार पूरा विवरण देखने लायक है। में
\[
A = \Z[\iu\sqrt5] = \{a + \iu b\sqrt5 : a, b \in \Z\},
\qquad N(a + \iu b\sqrt5) = a^2 + 5b^2,
\]
मानक गुणात्मक है और $ N(z) = 1$ आईएफएफ $ z \in
A^\times = \{\pm1\}$. विचार करना
\[
6 = 2 \cdot 3 = (1 + \iu\sqrt5)(1 - \iu\sqrt5).
\]
सभी चार कारक हैं \omterm{def:b3:rings:divisibility}{अलघुकरणीय}: उनके मानदंड हैं $4, 9, 6,
6$, और एक उचित गुणनखंडन $ z = z_1z_2$ जबरदस्ती करेगा
$ N(z_1) \in \{2, 3\}$ --- लेकिन $ a^2 + 5b^2$ कभी भी बराबर नहीं होता $2$
या $3$ ($ b = 0$ गैर-वर्गों को छोड़ देता है $2, 3$; $\abs b \geq
1$ देता है $\geq 5$). अभी तक $2$ किसी का भी सहयोगी नहीं है $1 \pm
\iu\sqrt5$ (मानदंड $4 \neq 6$): दो वास्तव में भिन्न
का गुणनखंडन $6$ में इरेड्यूसिबल्स. समान रूप से,
\omterm{def:b3:rings:divisibility}{अलघुकरणीय} $\neq $ यहाँ प्रमुख: $2$ उत्पाद को विभाजित करता है $(1 +
\iu\sqrt5)(1 - \iu\sqrt5) = 6$ लेकिन कोई भी कारक (मानदंड)
पुनः)। इस विफलता की आदर्श-सैद्धांतिक मरम्मत ---
फ़ैक्टराइज़िंग \emph{\omterm{def:b3:rings:ideal}{आदर्श}ों} तत्वों के बजाय --- है
बीजगणितीय संख्या सिद्धांत का जन्म; हमारे स्तर पर, उदाहरण
कैलिब्रेट करता है कि कितना खास है \omterm{def:b3:rings:pidufd}{इयूक्लिडियन} के छल्ले $\Z $, $ K[X]$,
$\Z[\iu]$ इस अध्याय के वास्तव में हैं.
\end{example}

\begin{method}\label{met:b3:rings:quotient}
ए की पहचान करने के लिए \omterm{def:b3:rings:ideal}{भागफल अंगूठी} $ A/I $, एक विशेषण रूपवाद की तलाश करें
$ f \colon A \to B $ कर्नेल के साथ $ I $ और आह्वान करें
\cref{thm:b3:rings:firstiso}; कब $ A = C[X]$ एक बहुपद है
अंगूठी, $ f $ आमतौर पर एक मूल्यांकन है. इस प्रकार $\Z[X]/(X^2+1) \cong
\Z[\iu]$ (पर मूल्यांकन करें $\iu $), $ K[X,Y]/(Y - X^2) \cong K[X]$
(मूल्यांकन करना $ Y $ पर $ X^2$), $\R[X]/(X^2+1) \cong \C $. जाहिर करना। $ I $
प्रधान या \omterm{def:b3:rings:primemaximal}{अधिक से अधिक}, दिखाएँ कि भागफल एक डोमेन या फ़ील्ड है
(\cref{prop:b3:rings:primemaximal}).
\end{method}

\section{यूएफडी पर बहुपद: गॉस और ईसेनस्टीन}

इस पूरे अनुभाग में $ A $ एक है \omterm{def:b3:rings:pidufd}{यूएफडी} अंश क्षेत्र के साथ $ K $
(औपचारिक भागफल के क्षेत्र के रूप में निर्मित $ a/b $, $ b \neq 0$,
बिल्कुल वैसा ही $\Q $ से $\Z $; वर्ष 2 के खंड ने ऐसा किया
के लिए निर्माण $\Q $, और यह शब्दशः स्थानांतरित होता है)। हमारा लक्ष्य:
तथ्यात्मकता से गुजरता है $ A $ को $ A[X]$, और इरेड्यूसिबिलिटी खत्म
$ A $ बड़े क्षेत्र पर मूलतः अप्रासंगिकता है $ K $.

\begin{definition}\label{def:b3:rings:content}
 \emph{सामग्री}\index{सामग्री} $ c(P)$ एक का
शून्येतर $ P \in A[X]$ इसके गुणांकों की एक जीसीडी है (ए तक परिभाषित)।
इकाई); $ P $ है \emph{प्राचीन} अगर $ c(P) \in A^\times $. प्रत्येक $ P
\in A[X]$ लिखते हैं $ P = c(P)\,P_1$ साथ $ P_1$ आदिम, और हर
$ P \in K[X]\setminus\{0\}$ लिखते हैं $ P = \lambda P_1$ साथ $\lambda
\in K^\times $ और $ P_1 \in A[X]$ आदिम (स्पष्ट हर,
फिर सामग्री को फैक्टर आउट करें)।
\end{definition}

\begin{lemma}[गॉस]\label{lem:b3:rings:gauss}
के दो आदिम बहुपदों का गुणनफल $ A[X]$ आदिम है;
फलस्वरूप $ c(PQ) = c(P)c(Q)$ इकाइयों तक.\index{गॉस की लेम्मा}
\end{lemma}

\begin{proof}
होने देना $ P, Q $ आदिम बनें और कुछ मान लें \omterm{def:b3:rings:divisibility}{अलघुकरणीय} (= अभाज्य,
\omterm{def:b3:rings:pidufd}{यूएफडी}) $ p $ के सभी गुणांकों को विभाजित करता है $ PQ $. मॉड्यूलो को कम करें $ p $: में
$(A/(p))[X]$, $\bar P \bar Q = 0$. लेकिन $ A/(p)$ एक डोमेन है
($(p)$ प्राइम), तो $(A/(p))[X]$ एक डोमेन है (अग्रणी गुणांक
गुणा करना), मजबूर करना $\bar P = 0$ या $\bar Q = 0$: $ p $ सबको बांट देता है
के गुणांक $ P $ या सभी $ Q $, आदिमता का खंडन। के लिए
परिणाम, लिखो $ P = c(P)P_1$, $ Q = c(Q)Q_1$: $ PQ =
c(P)c(Q) P_1Q_1$ साथ $ P_1Q_1$ प्राचीन।
\end{proof}

\begin{theorem}\label{thm:b3:rings:gaussufd}
होने देना $ A $ एक हो \omterm{def:b3:rings:pidufd}{यूएफडी} अंश क्षेत्र के साथ $ K $.
\begin{enumerate}
  \item एक आदिम $ P \in A[X]$ डिग्री का $\geq 1$ है \omterm{def:b3:rings:divisibility}{अलघुकरणीय}
        में $ A[X]$ यदि यह है \omterm{def:b3:rings:divisibility}{अलघुकरणीय} में $ K[X]$.
  \item $ A[X]$ एक है \omterm{def:b3:rings:pidufd}{यूएफडी}; इसका इरेड्यूसिबल्स हैं इरेड्यूसिबल्स का
        $ A $ और आदिम बहुपद \omterm{def:b3:rings:divisibility}{अलघुकरणीय} ऊपर $ K $. विशेष रूप से $\Z[X]$, और प्रेरण द्वारा $ K[X_1, \dots, X_n]$
        और $\Z[X_1, \dots, X_n]$, हैं \omterm{def:b3:rings:pidufd}{यूएफडी}.
\end{enumerate}
\end{theorem}

\begin{proof}
(1) ($\Leftarrow $) अगर $ P = QR $ में $ A[X]$ साथ $ Q, R $ गैरइकाइयाँ,
तब कोई भी कारक स्थिर नहीं होता (आदिम का एक स्थिर कारक)।
बहुपद एक इकाई है), इसलिए गुणनखंडन उचित है $ K[X]$.
($\Rightarrow $) कल्पना करना $ P = QR $ साथ $ Q, R \in K[X]$ डिग्रियों का
$\geq 1$. लिखना $ Q = \lambda Q_1$, $ R = \mu R_1$ साथ $ Q_1, R_1
\in A[X]$ प्राचीन: $ P = \lambda\mu\, Q_1R_1$, और $ Q_1R_1$ है
गॉस द्वारा आदिम. लेना अंतर्वस्तु, $\lambda\mu \in A^\times $
(दोनों पक्षों में इकाई है \omterm{def:b3:rings:content}{सामग्री}; औपचारिक रूप से, $\lambda\mu = c(P) \in
A^\times $ इकाइयों तक, और विशेष रूप से $\lambda \mu \in A $): $ P
= (\lambda\mu Q_1) R_1$ में एक उचित गुणनखंडन है $ A[X]$.

(2) अस्तित्व: दिया गया $ P \neq 0$ गैर इकाई, कारक $ P = c(P)P_1$,
कारक $ c(P)$ में इरेड्यूसिबल्स का $ A $, और कारक $ P_1$ में
\omterm{def:b3:rings:pidufd}{यूएफडी} $ K[X]$ जैसा $\prod Q_i $ साथ $ Q_i \in K[X]$ \omterm{def:b3:rings:divisibility}{अलघुकरणीय}; लिखना
$ Q_i = \lambda_i R_i $ साथ $ R_i \in A[X]$ आदिम (इसलिए
\omterm{def:b3:rings:divisibility}{अलघुकरणीय} ऊपर $ K $, इसलिए में $ A[X]$ (1)), उत्पाद द्वारा
$\prod \lambda_i $ की एक इकाई है $ A $ पहले की तरह, और $ P_1 = u\prod
R_i $. विशिष्टता: गुणनखंडन के स्थिर भाग की तुलना करें और
बहुपद भाग; स्थिरांक गुणा हो जाते हैं $ c(P)$ (गॉस), अद्वितीय
की तथ्यात्मकता से $ A $; बहुपद भाग दो देते हैं
में गुणनखंडन $ K[X]$ एक ही बहुपद के, इसलिए वे मेल खाते हैं
के स्थिरांक के लिए $ K^\times $ (की तथ्यात्मकता $ K[X]$,
\cref{thm:b3:rings:pidufd}), और आदिम बहुपदों का मिलान
में संबद्ध $ K[X]$ में जुड़े हुए हैं $ A[X]$: अगर $ R = \lambda R'$
साथ $ R, R'$ आदिम और $\lambda \in K^\times $, फिर ले रहा हूँ
अंतर्वस्तु बल $\lambda \in A^\times $.
\end{proof}

\begin{theorem}[अपरिवर्तनीयता मानदंड]\label{thm:b3:rings:criteria}
होने देना $ A $ एक हो \omterm{def:b3:rings:pidufd}{यूएफडी}, $ K $ इसका अंश क्षेत्र, और $ P = a_nX^n + \dots
+ a_0 \in A[X]$ डिग्री का आदिम $ n \geq 1$.
\begin{enumerate}
  \item (\emph{कमी}) अगर $ p \in A $ प्रधान है, $ p \nmid a_n $,
        और कमी $\bar P $ है \omterm{def:b3:rings:divisibility}{अलघुकरणीय} में
        $(A/(p))[X]$, तब $ P $ है \omterm{def:b3:rings:divisibility}{अलघुकरणीय} में $ K[X]$ (इसलिए में
        $ A[X]$).
  \item (\emph{Eisenstein}\index{Eisenstein}) यदि कुछ
        प्रधान $ p $ संतुष्ट $ p \nmid a_n $, $ p \mid a_i $ के लिए $0
        \leq i < n $, और $ p^2 \nmid a_0$, तब $ P $ है \omterm{def:b3:rings:divisibility}{अलघुकरणीय}
        में $ K[X]$ (इसलिए में $ A[X]$).
\end{enumerate}
\end{theorem}

\begin{proof}
द्वारा \cref{thm:b3:rings:gaussufd}(1), एक उचित गुणनखंडन
$ K $ पैदावार $ P = QR $ साथ $ Q, R \in A[X]$, $\deg Q, \deg R \geq 1$
(स्थिरांकों को बाहर रखा गया है: वे इकाइयाँ होंगी या ख़राब होंगी
आदिमता)।

(1) मॉड कम करें $ p $: $\bar P = \bar Q\bar R $ में $(A/(p))[X]$. तब से
$ p \nmid a_n $ और $\deg $ केवल कटौती के तहत गिर सकता है, $\deg \bar
Q = \deg Q \geq 1$ और $\deg\bar R = \deg R \geq 1$ (उनका नेतृत्व
गुणांकों को गुणा किया जाता है $\bar a_n \neq 0$, इसलिए न तो बूँदें):
$\bar P $ कारक ठीक से --- विरोधाभास।

(2) मॉड कम करें $ p $: $\bar Q \bar R = \bar P = \bar a_n X^n $ (सभी
कम गुणांक मर जाते हैं)। डोमेन में $(A/(p))[X]$, द
का गुणनखंडन $ cX^n $ ($ c \ne 0$) स्थिरांक और शुद्ध में हैं
शक्तियां $ c'X^k $: वास्तव में अगर $\bar Q\bar R = \bar a_nX^n $, और कहते हैं
$\bar Q $ डिग्री में शून्येतर गुणांक था $< \deg\bar Q $, ले लो
निम्नतम अशून्य पद: $\operatorname{val}(\bar Q\bar R) =
\operatorname{val}\bar Q + \operatorname{val}\bar R $ (डोमेन),
जो बराबर होना चाहिए $ n = \deg\bar Q + \deg\bar R $, मजबूर करना
$\operatorname{val} = \deg $ दोनों के लिए: दोनों एकपदी हैं। जैसे
ऊपर, डिग्री नहीं गिरती, इसलिए $ Q $ और $ R $ उनका स्थिरांक है
शर्तें $ Q(0), R(0)$ द्वारा विभाज्य $ p $ --- दोनों, चूँकि दोनों
कटौती डिग्री के एकपदी हैं $\geq 1$. तब $ p^2 \mid
Q(0)R(0) = a_0$: विरोधाभास।
\end{proof}

\begin{example}\label{ex:b3:rings:eisenstein}
$ X^n - p $ है \omterm{def:b3:rings:divisibility}{अलघुकरणीय} ऊपर $\Q $ प्रत्येक प्राइम के लिए $ p $ और $ n
\geq 1$ (\omterm{thm:b3:rings:criteria}{Eisenstein} पर $ p $): वहाँ हैं \omterm{def:b3:rings:divisibility}{अलघुकरणीय} के बहुपद
हर डिग्री खत्म $\Q $ --- के साथ बिल्कुल विपरीत $\C $ (डिग्री
$1$, डी'एलेम्बर्ट--गॉस, में साबित हुआ \cref{ch:b3:holomorphic}) और
$\R $ (डिग्री $1, 2$). की युक्ति \emph{स्थानांतरण} मायनों
Eisensteinकी पहुंच: $ p $-वां \emph{साइक्लोटोमिक
बहुपद}\index{साइक्लोटोमिक
बहुपद} $\Phi_p = X^{p-1} +
\dots + X + 1 = \frac{X^p - 1}{X - 1}$ है
\[
\Phi_p(X + 1) = \frac{(X+1)^p - 1}{X}
= X^{p-1} + \binom{p}{1}X^{p-2} + \dots + \binom{p}{p-1},
\]
\omterm{thm:b3:rings:criteria}{Eisenstein} पर $ p $ ($ p \mid \binom pk $ के लिए $0 < k < p $, और
$\binom{p}{p-1} = p \not\equiv 0 \bmod p^2$): $\Phi_p(X+1)$, इस तरह
$\Phi_p $, है \omterm{def:b3:rings:divisibility}{अलघुकरणीय} ऊपर $\Q $. यह का बीजगणितीय हृदय है
द $17$-गॉन कहानी में बताया गया है \cref{ch:b3:galois}.
\end{example}

\begin{method}\label{met:b3:rings:irreducibility}
साबित करना $ P \in \Z[X]$ \omterm{def:b3:rings:divisibility}{अलघुकरणीय} ऊपर $\Q $: (मैं बनाता हूं $ P $
प्राचीन; (ii) प्रयास करें \omterm{thm:b3:rings:criteria}{Eisenstein}, पर $ P(X)$ और पाली में $ P(X \pm
1)$; (iii) विभाजित न करते हुए मॉड्यूलो छोटे अभाज्यों को कम करने का प्रयास करें
अग्रणी गुणांक --- अपरिवर्तनीयता मॉड \emph{एक} $ p $
पर्याप्त, और भी बहुत कुछ $\mathbb F_p $ अपरिवर्तनीयता एक सीमित जाँच है
(कोई भी मूल डिग्री को छोड़कर नहीं है-$1$ कारक; फिर बहुत सारे का परीक्षण करें
प्रत्येक डिग्री के कारक $\leq \deg P/2$); (iv) यदि अन्य सभी विफल हो जाते हैं,
अनिर्धारित गुणांक. सावधान: रिड्यूसिबिलिटी मॉड प्रत्येक $ p $ करता है
\emph{नहीं} कम करने की क्षमता को खत्म करें $\Q $
(\cref{exo:b3:rings:11}).
\end{method}

\section{नोथेरियन रिंग्स}

\begin{definition}\label{def:b3:rings:noetherian}
एक अंगूठी $ A $ है \emph{नोथेरियन}\index{नोथेरियन} यदि प्रत्येक
\omterm{def:b3:rings:ideal}{आदर्श} का $ A $ अंतिम रूप से उत्पन्न होता है।
\end{definition}

\begin{proposition}\label{prop:b3:rings:noethacc}
$ A $ है \omterm{def:b3:rings:noetherian}{नोथेरियन} यदि प्रत्येक बढ़ता क्रम \omterm{def:b3:rings:ideal}{आदर्श}ों है
अंततः स्थिर (\emph{आरोही श्रृंखला की स्थिति}), यदि प्रत्येक
का गैर-रिक्त परिवार \omterm{def:b3:rings:ideal}{आदर्श}ों एक \omterm{def:b3:rings:primemaximal}{अधिक से अधिक} तत्व (समावेश के लिए)।
\end{proposition}

\begin{proof}
\emph{(एफजी $\Rightarrow $ एसीसी)}: एक श्रृंखला के लिए $ I_1 \subseteq I_2
\subseteq \cdots $, संघ $ I $ एक \omterm{def:b3:rings:ideal}{आदर्श}, द्वारा उत्पन्न $ x_1,
\dots, x_r $; सभी $ x_i $ कुछ में झूठ बोलो $ I_N $, इसलिए $ I = I_N = I_n $ के लिए
$ n \geq N $. \emph{(एसीसी $\Rightarrow $ \omterm{def:b3:rings:primemaximal}{अधिक से अधिक} तत्व)}: यदि ए
गैर-रिक्त परिवार $\mathcal F $ नहीं हैं \omterm{def:b3:rings:primemaximal}{अधिक से अधिक} तत्व, चुनें $ I_1 \in
\mathcal F $, फिर आगमनात्मक रूप से $ I_{n+1} \supsetneq I_n $ में
$\mathcal F $ (तब से संभव है $ I_n $ क्या नहीं है \omterm{def:b3:rings:primemaximal}{अधिक से अधिक}): एक अनंत
सख्ती से बढ़ती श्रृंखला। (यह आश्रित के स्वयंसिद्ध का उपयोग करता है
विकल्प, विकल्प का एक कमजोर रूप जिस पर हम उपद्रव नहीं करते।)
\emph{(\omterm{def:b3:rings:primemaximal}{अधिक से अधिक} तत्वों $\Rightarrow $ एफजी)}: एक दिया गया \omterm{def:b3:rings:ideal}{आदर्श} $ I $,
अंतिम रूप से उत्पन्न का परिवार \omterm{def:b3:rings:ideal}{आदर्श}ों में निहित $ I $ है
गैररिक्त ($(0)$); ए \omterm{def:b3:rings:primemaximal}{अधिक से अधिक} तत्व $ J = (x_1, \dots, x_r)$ अवश्य
बराबर $ I $: अन्यथा, जोड़ना $ x \in I \setminus J $ को
जनरेटर से परिवार का एक बड़ा सदस्य पैदा होता है।
\end{proof}

\begin{theorem}[हिल्बर्ट का आधार प्रमेय]\label{thm:b3:rings:hilbert}
अगर $ A $ है \omterm{def:b3:rings:noetherian}{नोथेरियन}, ऐसा ही है $ A[X]$. इसलिए ऐसे हैं $ A[X_1, \dots,
X_n]$, और उनमें से प्रत्येक भागफल।\index{हिल्बर्ट का आधार प्रमेय}
\end{theorem}

\begin{proof}
होने देना $ I $ एक हो \omterm{def:b3:rings:ideal}{आदर्श} का $ A[X]$, और मान लीजिए $ I $ अंतिम रूप से नहीं है
उत्पन्न. एक क्रम बनाएँ: $ f_1 \in I \setminus \{0\}$ का
न्यूनतम डिग्री, और आगमनात्मक रूप से $ f_{k+1} \in I \setminus (f_1,
\dots, f_k)$ न्यूनतम डिग्री का (सेट गैर-रिक्त है)।
धारणा)। डिग्री $ d_k = \deg f_k $ घटते नहीं (द्वारा)।
प्रत्येक विकल्प की न्यूनतमता: $ f_{k+1}$ कदम पर उपलब्ध था
$ k+1$... एकदम सही, $ f_{k+1} \notin (f_1,\dots,f_k) \supseteq
(f_1, \dots, f_{k-1})$, इसलिए $ f_{k+1}$ कदम कदम पर प्रतिस्पर्धा की $ k $ और हार गया
या बंधा हुआ: $ d_{k+1} \geq d_k $). होने देना $ a_k \in A $ अग्रणी बनें
का गुणांक $ f_k $. की शृंखला \omterm{def:b3:rings:ideal}{आदर्श}ों $(a_1) \subseteq (a_1,
a_2) \subseteq \cdots $ स्थिर करता है: $ a_{n+1} \in (a_1, \dots,
a_n)$ कुछ के लिए $ n $, कहना $ a_{n+1} = \sum_{k\leq n} u_k a_k $.
विचार करना
\[
g = f_{n+1} - \sum_{k=1}^{n} u_k X^{\,d_{n+1} - d_k} f_k .
\]
तब $ g \in I \setminus (f_1, \dots, f_n)$ (योग में निहित है
\omterm{def:b3:rings:ideal}{आदर्श}, $ f_{n+1}$ नहीं), फिर भी डिग्री का गुणांक
$ d_{n+1}$ रद्द: $\deg g < d_{n+1}$, का खंडन करते हुए
की न्यूनतमता $ d_{n+1} = \deg f_{n+1}$.

पुनरावृत्त, $ A[X_1, \dots, X_n] = (A[X_1, \dots, X_{n-1}])[X_n]$
है \omterm{def:b3:rings:noetherian}{नोथेरियन}; एक भागफल $ A/I $ है \omterm{def:b3:rings:noetherian}{नोथेरियन} ये इसलिए है क्योंकि \omterm{def:b3:rings:ideal}{आदर्श}ों
$ J/I $ तक उठाना \omterm{def:b3:rings:ideal}{आदर्श}ों का $ A $ (पत्राचार), जहां बहुत सारे
जनरेटर जनरेटर पर प्रोजेक्ट करते हैं।
\end{proof}

\begin{remark}\label{rem:b3:rings:noethmeaning}
\omterm{def:b3:rings:noetherian}{नोथेरियन}िटी बीजगणितीय ज्यामिति का परिमित सिद्धांत है: कोई भी
बहुपद समीकरणों की प्रणाली $ n $ परिवर्तनशील, चाहे कितने भी अनंत हों,
उनमें से बहुत से परिमित के बराबर है --- इसका समाधान सेट कट जाता है
परिमित अनेक बहुपदों से बाहर। \omterm{def:b3:rings:pidufd}{पीआईडी} हैं \omterm{def:b3:rings:noetherian}{नोथेरियन}
(तुच्छ रूप से); $\Z[X_1, X_2, \dots]$ अपरिमित रूप से अनेक चरों में
नहीं है ($(X_1) \subsetneq (X_1, X_2) \subsetneq \cdots $).
गै\omterm{def:b3:rings:noetherian}{र-नोथेरियन} विश्लेषण में वलय भी स्वाभाविक रूप से होते हैं: निरंतर
पर कार्य करता है $\intcc01$ फॉर्म एक (\cref{exo:b3:rings:10}).
\end{remark}

प्रिंसिपल नहीं --- तो $ K[X,Y]$ एक है \omterm{def:b3:rings:pidufd}{यूएफडी}
(\cref{thm:b3:rings:gaussufd}) वह नहीं है \omterm{def:b3:rings:pidufd}{पीआईडी}; (बी) पहचानें
$ K[X, Y]/(Y - X^2)$ और $ K[X,Y]/(XY - 1)$ तर्कसंगत के उप-अंशों के रूप में
कार्य; (सी) है $(Y - X^2)$ मुख्य? \omterm{def:b3:rings:primemaximal}{अधिक से अधिक}?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:rings:7}
\omterm{def:b3:rings:divisibility}{अलघुकरणीय} या ख़त्म नहीं हुआ $\Q $: $ X^5 - 12X^3 + 36X - 12$;\quad
$ X^4 + X + 1$ \emph{(मॉड कम करें $2$)};\quad $ X^4 + 4$;\quad
$\Phi_8 = X^4 + 1$ \emph{(द्वारा शिफ्ट करें $1$)};\quad $ X^3 - X - 1$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:rings:8}
(ए) से \cref{thm:b3:rings:crt}, साबित करें कि यूलर का कार्य है
सहअभाज्य तर्कों पर गुणनात्मक और वह $\varphi(p^k) =

(बी) हल करें: $ x \equiv 2 \pmod 7$, $ x \equiv 5 \pmod{11}$, $ x

\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b3:rings:9}
(नीलरेडिकल) चलो $\operatorname{Nil}(A)$ का सेट हो
शून्यशक्तिशाली तत्व. (ए) वह दिखाओ $\operatorname{Nil}(A)$ एक

\]
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b3:rings:10}
(ए) चलो $ A $ होना \omterm{def:b3:rings:noetherian}{नोथेरियन} और $ f \colon A \to A $ एक विशेषण
वलय आकारिकी. बताते हैं कि $ f $ इंजेक्शन है. \emph{(विचार करना $\ker

(बी) दिखाओ कि अंगूठी $\mathcal C(\intcc01, \R)$ निरंतर का

\text{ on } \intcc0{1/n}\}$.)}
(सी) इसे ए में दिखाएं \omterm{def:b3:rings:noetherian}{नोथेरियन} \emph{कार्यक्षेत्र}, प्रत्येक अशून्य
नॉनयूनिट एक (परिमित) उत्पाद है इरेड्यूसिबल्स --- तो
की गैर-तथ्यात्मकता $\Z[\iu\sqrt 5]$ की विफलता है
\emph{विशिष्टता} केवल।
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b3:rings:11}
होने देना $ P = X^4 + 1$.
(ए) वह दिखाओ $ P $ है \omterm{def:b3:rings:divisibility}{अलघुकरणीय} ऊपर $\Q $
(\cref{exo:b3:rings:7}).
(बी) वह दिखाओ $ P $ रिड्यूसिबल मॉड्यूलो है \emph{प्रत्येक} मुख्य $ p $:
इलाज $ p = 2$; फिर, विषम के लिए $ p $, बताते हैं कि $8 \mid p^2 - 1$ और
अभी के लिए स्वीकार करें (साबित)। \cref{ch:b3:galois}) कि
क्षेत्र का गुणक समूह $ p^2$ तत्व चक्रीय है,
यह निष्कर्ष निकालने के लिए $ P $ दो द्विघात कारकों में विभाजित मॉड $ p $;
उनमें से एक होने पर उन्हें स्पष्ट करें $-1$, $2$, $-2$ एक वर्गाकार मॉड है
$ p $, और दिखाओ कि उनमें से एक हमेशा है।
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:rings:12}
(इडेम्पोटेंट्स स्प्लिट रिंग्स) एक तत्व $ e $ एक क्रमविनिमेय वलय का
$ A $ है \emph{निष्क्रिय} अगर $ e^2 = e $.
(ए) दिखाओ अगर $ e $ निष्क्रिय है, इसलिए है $1 - e $, और वह
नक्शा $ x \mapsto (ex, (1-e)x)$ एक वलय समरूपता है $ A
\cong Ae \times A(1-e)$, कहाँ $ Ae $ इकाई के साथ एक अंगूठी है $ e $.
(बी) एक डोमेन के सभी बेरोजगार खोजें, और $\Z/12\Z $;
समरूपता प्रदर्शित करें $\Z/12\Z \cong \Z/4\Z \times \Z/3\Z $
इसके दो गैर-तुच्छ निष्क्रिय व्यक्तियों का नाम देकर।
(सी) दिखाएँ कि सीआरटी अपघटन $\Z/n\Z $
(\cref{ex:b3:rings:crtz}) बिल्कुल मेल खाता है
बेरोजगार $ e_i \equiv 1 \bmod p_i^{a_i}$, $ e_i \equiv 0$
मॉड्यूलो अन्य प्रमुख शक्तियां: वलय उनके साथ विघटित होते हैं
अनुमानों के साथ-साथ रिक्त स्थान विघटित हो जाते हैं।
\end{exercise}

\section{समस्या: फ़र्मेट का दो-वर्ग प्रमेय}

\begin{problem}[{सप्ताहांत समस्या --- दो वर्गों का योग, के माध्यम से
$\Z[\iu]$}]\label{pb:b3:rings:1}
कौन से पूर्णांक दो वर्गों का योग हैं? फ़र्मेट का उत्तर (1640) है
अंकगणित के रत्नों में से एक; गॉसियन पूर्णांक इसका प्रमाण बन जाते हैं
रिंग सिद्धांत में. लगातार, $ N(x + \iu y) = x^2 + y^2$ दर्शाता है
\omterm{def:b3:rings:ideal}{आदर्श}, $\Z[\iu]$ है \omterm{def:b3:rings:pidufd}{इयूक्लिडियन} (\cref{exo:b3:rings:4}), इसलिए ए
\omterm{def:b3:rings:pidufd}{पीआईडी} और ए \omterm{def:b3:rings:pidufd}{यूएफडी}, और \emph{गॉसियन प्राइम} मतलब अभाज्य (=
\omterm{def:b3:rings:divisibility}{अलघुकरणीय}) का तत्व $\Z[\iu]$.

\medskip
\noindent\textbf{Part I --- Norms and Gaussian primes.}
\begin{enumerate}
  \item सत्यापित करें $ N(zw) = N(z)N(w)$, फिर से निष्कर्ष निकालें $\Z[\iu]^\times =
        \{\pm1, \pm\iu\}$, और सिद्ध करें \emph{ब्रह्मगुप्त
        पहचान}: दो वर्गों के दो योगों का गुणनफल एक योग होता है
        दो वर्गों का.
  \item दिखाओ अगर $ N(z)$ तो, यह एक अभाज्य संख्या है $ z $ एक है
        गॉसियन प्राइम.
  \item दिखाएँ कि प्रत्येक गाऊसी अभाज्य $\pi $ बिल्कुल एक को विभाजित करता है
        अभाज्य संख्या $ p $ \emph{(विचार करना $ N(\pi) = \pi\bar\pi $)},
        और फिर $ N(\pi) \in \{p, p^2\}$.
  \item द्विभाजन निकालें: प्रत्येक अभाज्य के लिए $ p $, दोनों में से एक $ p $
        में प्रमुख रहता है $\Z[\iu]$ (और मानक का कोई गाऊसी अभाज्य नहीं
        $ p $ मौजूद है), या $ p = \pi\bar\pi $ साथ $\pi $ एक गाऊसी
        \omterm{def:b3:rings:ideal}{आदर्श} का प्रधान $ p $ --- और तब $ p = a^2 + b^2$.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Wilson's theorem and $-1$ modulo
$ p $.}
\begin{enumerate}[resume]
  \item सिद्ध करना \emph{विल्सन का प्रमेय}: के लिए $ p $ मुख्य, $(p-1)!
        \equiv -1 \pmod p $. \emph{(प्रत्येक अवशेष को उसके साथ जोड़ दें
        उलटा; कौन से स्व-युग्मित हैं?)}
  \item होने देना $ p $ एक विषम अभाज्य हो और $ m = \frac{p-1}2$. बताते हैं कि
        $(m!)^2 \equiv (-1)^{m+1} \pmod p $ \emph{(में $(p-1)!$,
        प्रत्येक कारक को बदलें $ k > m $ द्वारा $-(p - k)$)}.
  \item निष्कर्ष: $-1$ एक वर्ग मॉड्यूलो है $ p $ आईएफएफ $ p = 2$ या $ p
        \equiv 1 \pmod 4$. \emph{(``केवल यदि'' के लिए: यदि $ x^2
        \equiv -1$, का आदेश क्या है $ x $ में $(\Z/p\Z)^\times $,
        और लैग्रेंज क्या कहता है?)}
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The splitting law.}
\begin{enumerate}[resume]
  \item होने देना $ p \equiv 1 \pmod 4$, और $ x $ साथ $ p \mid x^2 + 1 =
        (x + \iu)(x - \iu)$. बताते हैं कि $ p $ है \emph{नहीं} ए
        गॉसियन अभाज्य, और भाग I के साथ समाप्त करें: $ p = a^2 + b^2$.
  \item होने देना $ p \equiv 3 \pmod 4$. उसे सीधे दिखाओ $ p $ एक नहीं है
        दो वर्गों का योग \emph{(स्क्वायर मॉड $4$)}, और निष्कर्ष निकालें
        वह $ p $ गॉसियन प्राइम रहता है।
  \item निपटारा करना $ p = 2$: गुणनखंडन प्रदर्शित करें $2 =
        -\iu(1+\iu)^2$ और उसे जांचें $1 + \iu $ एक गाऊसी है
        प्रधान. ($2$ अद्वितीय है \emph{डालियां फैला हुआ} अभाज्य: विभाज्य
        एक गाऊसी अभाज्य के वर्ग द्वारा एक इकाई तक।)
  \item तक गाऊसी अभाज्य संख्याओं का वर्गीकरण एकत्रित करें
        इकाइयाँ: $1 + \iu $; पूर्णांक $ p \equiv 3 \pmod 4$; द
        संयुग्मित जोड़े $\pi, \bar\pi $ \omterm{def:b3:rings:ideal}{आदर्श} का $ p \equiv 1 \pmod
        4$. इसे सत्यापित करें $5 = (2+\iu)(2-\iu)$ और पर $3$.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The two-squares theorem.}
\begin{enumerate}[resume]
  \item प्रत्यक्ष आधा सिद्ध करें: यदि गुणनखंडन में $ n =
        \prod p_i^{\alpha_i}$ हर प्राइम $\equiv 3 \pmod 4$
        तब, एक सम घातांक के साथ प्रकट होता है $ n $ दो का योग है
        वर्ग। \emph{(ब्रह्मगुप्त + भाग II--III।)}
  \item विपरीत सिद्ध करें: यदि $ n = a^2 + b^2 = N(a + \iu b)$ और
        $ q \equiv 3 \pmod 4$ विभाजित $ n $, बताते हैं कि $ q $, ए
        गॉसियन अभाज्य, विभाजित करता है $ a + \iu b $ या $ a - \iu b $, वह
        यह वास्तव में दोनों को विभाजित करता है $ a $ और $ b $, और निष्कर्ष निकालो
        प्रेरण पर $ n $ वह प्रतिपादक है $ q $ में $ n $ सम है.
  \item अंतिम प्रमेय बताएं. कौन सा $2025$, $2026$, $2027$
        दो वर्गों का योग क्या है? \emph{($2025 = 81 \cdot 25$;
        $2026 = 2 \cdot 1013$, $1013$ मुख्य; $2027$ मुख्य।)}
  \item (उपसंहार) वह एक प्रधान दिखाएँ $ p \equiv 1 \pmod 4$ एक है
        अनिवार्य रूप से अनोखे तरीके से दो वर्गों का योग: यदि $ p =
        a^2 + b^2 = c^2 + d^2$ (धनात्मक पूर्णांक), फिर $\{a, b\}
        = \{c, d\}$. \emph{(गुणनखंडन की विशिष्टता
        $\Z[\iu]$.)}
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Counting representations: Jacobi's
formula and Leibniz's series.} लिखना $ r_2(n) = \#\{(a, b) \in
\Z^2 : a^2 + b^2 = n\}$ (जोड़े, चिह्न और शून्य का क्रम दिया गया
शामिल), और चलो $\chi $ गैर-तुच्छ चरित्र मॉड बनें $4$:
$\chi(d) = +1$ अगर $ d \equiv 1$, $-1$ अगर $ d \equiv 3 \pmod4$,
$0$ अगर $ d $ यहां तक की।
\begin{enumerate}[resume]
  \item (इसके विपरीत वार्म-अप) कौन से पूर्णांक हैं
        \emph{मतभेद} दो वर्गों का? दिखाओ: $ n = a^2 - b^2$
        साथ $ a, b \in \Z $ आईएफएफ $ n \not\equiv 2 \pmod 4$ ---
        किसी वलय सिद्धांत की आवश्यकता नहीं है, और न ही किसी संरचना की तुलना की जा सकती है
        इसके बाद क्या होता है.
  \item बताते हैं कि $ r_2(n)$ की संख्या है $ z \in \Z[\iu]$ साथ
        $ N(z) = n $. लिखना $ n = 2^{a}\prod_jp_j^{b_j}
        \prod_kq_k^{c_k}$ साथ $ p_j \equiv 1$, $ q_k \equiv 3
        \pmod4$, प्रश्न 11 के वर्गीकरण का उपयोग करें और
        दिखाने के लिए अद्वितीय गुणनखंडन: जैसे $ z $ यदि सभी अस्तित्व में हैं
        $ c_k $ सम हैं, और उस स्थिति में
        \[
        r_2(n) = 4\prod_j\,(b_j + 1) .
        \]
        \emph{(गिनती करना: $ z = u\,(1+\iu)^{a}\prod_j\pi_j^{s_j}
        \bar\pi_j^{\,b_j - s_j}\prod_kq_k^{c_k/2}$ साथ $ u $ ए
        इकाई और $0 \leq s_j \leq b_j $; यह सूची क्यों है?
        संपूर्ण और दोहराव-मुक्त?)}
  \item बताते हैं कि $ d \mapsto \chi(d)$ पूरी तरह से है
        गुणनफल, वह निष्कर्ष निकालो $ n \mapsto
        \sum_{d \mid n}\chi(d)$ गुणक है, और इसकी गणना करें
        प्रधान शक्तियों पर: यह बराबर है $1$ पर $2^a $; $ b + 1$ पर
        $ p^b $ ($ p \equiv 1$); $1$ या $0$ पर $ q^c $ ($ q \equiv 3$)
        के अनुसार $ c $ सम है या विषम.
  \item निष्कर्ष निकालना \emph{जैकोबी का प्रमेय}:
        \[
        r_2(n) = 4\sum_{d \mid n}\chi(d) =
        4\bigl(d_1(n) - d_3(n)\bigr),
        \]
        कहाँ $ d_i(n)$ विभाजकों को गिनता है $\equiv i \pmod 4$.
        पर सत्यापित करें $ n = 3, 5, 9, 25$, और सूचीबद्ध करें $16$
        का प्रतिनिधित्व $65$.
  \item (वृत्त) वह दिखाओ $\sum_{n \leq x}r_2(n)$ है
        जालक बिंदुओं की संख्या $\Z^2$ की बंद डिस्क में
        त्रिज्या $\sqrt x $, और साबित करो
        \[
        \sum_{n\leq x}r_2(n) = \pi x + O(\sqrt x)
        \]
        \emph{(प्रत्येक जाली बिंदु एक इकाई वर्ग का स्वामी है; तुलना करें
        क्षेत्र, चौड़ाई के वलय में रहने वाली त्रुटि
        $ O(1)$)}.
  \item (लीबनिज़, अंकगणितीय रूप से पढ़ें) प्रश्न 19-20 को संयोजित करें:
        \[
        \sum_{d \leq x}\chi(d)\Bigl\lfloor\frac
        xd\Bigr\rfloor = \frac{\pi x}4 + O(\sqrt x),
        \]
        और निष्कर्ष निकालें --- फर्श को सावधानीपूर्वक हटाना ---
        लीबनिज़ की श्रृंखला
        \[
        1 - \frac13 + \frac15 - \frac17 + \dots = \frac\pi4 .
        \]
        विषम व्युत्क्रमों की वैकल्पिक श्रृंखला \emph{है} द
        भाजक का औसत आधिक्य $\equiv 1$ विभाजकों के ऊपर
        $\equiv 3$: अंकगणित द्वारा गणना किया गया विश्लेषण।
  \item (दो वर्गों का योग कितना दुर्लभ है?) दिखाएँ कि नहीं
        पूर्णांक $\equiv 3 \pmod 4$ दो वर्गों (दो) का योग है
        तरीके: वर्ग मॉड $4$, या समता मानदंड
        प्रश्न 17), तो सभी पूर्णांकों का कम से कम एक चौथाई है
        छूट गया; और दिखाओ कि औसत $\frac1x\sum_{n\leq
        x}r_2(n) \to \pi $ प्रश्न 20 के साथ संगत है
        घनत्व वाले निरूपण योग्य पूर्णांक $0$ ---
        असामान्य रूप से कई निरूपणों के साथ पूर्णांक प्रदर्शित करें
        (कई अभाज्य संख्याओं के उत्पाद लीजिए $\equiv 1 \bmod 4$) को
        स्पष्ट करें कि एक लुप्त हो रहा अनुपात अभी भी कैसे ले जा सकता है
        सकारात्मक औसत. (लैंडौ ने सही घनत्व क्षय साबित किया
        जैसे $1/\sqrt{\log x}$; यह हमारे उपकरणों से परे है, लेकिन
        तंत्र अब दिखाई दे रहा है।)
\end{enumerate}

\medskip
\noindent\textbf{Part VI --- Complements: primitive
representations and Pythagoras.}
\begin{enumerate}[resume]
  \item एक अभ्यावेदन को बुलाओ $ n = a^2 + b^2$ \emph{प्राचीन}
        अगर $\gcd(a, b) = 1$. बताते हैं कि $ n \geq 1$ स्वीकार करता है ए
        आदिम प्रतिनिधित्व iff $4 \nmid n $ और कोई प्राइम नहीं
        $ q \equiv 3 \pmod 4$ विभाजित $ n $. \emph{(के लिए
        आवश्यकता, प्रश्न 13 और वर्गों के अवतरण का पुन: उपयोग करें
        मॉड $4$; पर्याप्तता के लिए, निर्माण करें $ z $ से $1 + \iu $
        और यह $\pi_j $ केवल --- कोई संयुग्म नहीं --- और समझाएं
        का एक \omterm{def:b3:groups:normal}{सामान्य} अभाज्य कारक क्यों $ a $ और $ b $ जबरदस्ती करेगा
        दोनों $\pi_j $ और $\bar\pi_j $, या $(1+\iu)^2$, में
        $ z $.)}
  \item (पायथागॉरियन त्रिगुण) चलो $ a^2 + b^2 = c^2$ साथ $ a, b,
        c $ सकारात्मक, $\gcd(a, b) = 1$ और $ b $ यहां तक की। बताते हैं कि
        $ a + \iu b $ और $ a - \iu b $ में सहप्रधान हैं $\Z[\iu]$
        \emph{(एक \omterm{def:b3:groups:normal}{सामान्य} गाऊसी अभाज्य भाजक विभाजित करेगा
        $2a $ और $2b $, और $ c $ अजीब है)}, अद्वितीय से निष्कर्ष निकालें
        गुणनखंडन कि $ a + \iu b = u(m + \iu n)^2$ एक के लिए
        इकाई $ u $, और शास्त्रीय पैरामीट्रिजेशन समाप्त करें:
        अदला-बदली तक $ a $ और $ b $,
        \[
        a = m^2 - n^2, \qquad b = 2mn, \qquad c = m^2 + n^2,
        \]
        साथ $ m > n \geq 1$ विपरीत समताओं का सहअभाज्य।
        वापस पाना $(3, 4, 5)$ और $(21, 20, 29)$ से $(m, n) =
        (2, 1)$ और $(5, 2)$.
  \item (संख्यात्मक सत्यापन) लीजिए $ x = 25$. गणना करना
        $ r_2(n)$ के लिए $1 \leq n \leq 25$ जैकोबी के सूत्र से,
        जांचें कि शून्येतर मान ठीक-ठीक घटित होते हैं $ n = 1,
        2, 4, 5, 8, 9, 10, 13, 16, 17, 18, 20, 25$, ओर वो
        \[
        \sum_{n \leq 25} r_2(n) = 80
        = 4\sum_{d \leq 25}\chi(d)
        \Bigl\lfloor\frac{25}d\Bigr\rfloor .
        \]
        सत्यापित करें कि बंद डिस्क त्रिज्या की है $5$ रोकना $81$
        जाली बिंदु, और तुलना करें $\pi x \approx 78.5$:
        त्रुटि अच्छी तरह से भीतर है $ O(\sqrt x)$ प्रश्न का
        20.
\end{enumerate}
\end{problem}
