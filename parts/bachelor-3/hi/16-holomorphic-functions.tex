\chapter{होलोमोर्फिक फ़ंक्शंस}\label{ch:b3:holomorphic}

जटिल भिन्नता एक छोटे बदलाव की तरह दिखती है
वास्तविक सिद्धांत --- एक सीमा, एक भागफल। इसके बजाय यह एक है
अलग ब्रह्मांड. एक फ़ंक्शन अलग करने योग्य \emph{एक बार} एक पर
का खुला उपसमुच्चय $\C $ स्वचालित रूप से असीम रूप से भिन्न है,
विश्लेषणात्मक, एक संपूर्ण डोमेन पर इसके मानों द्वारा निर्धारित किया जाता है
एकल बिंदु, और कठोर वैश्विक सिद्धांतों द्वारा बाधित
(लिउविले, अधिकतम मापांक)। यह सब एक से प्रवाहित होता है
चमत्कार, कॉची का प्रमेय: ए का अभिन्न अंग होलोमोर्फिक
समारोह एक \omterm{def:b3:topology:interior}{बंद} के साथ पथ एक तारे के आकार के डोमेन में गायब हो जाता है।
यह अध्याय चमत्कार को सिद्ध करता है (गौरसैट का तर्क, संख्या के साथ)।
भिन्नता से परे मानी गई नियमितता), इसकी कटाई करती है
शास्त्रीय परिणाम, और इस पुस्तक के एक प्रमेय को सुलझाता है
तब से क्रेडिट पर उपयोग किया जाता है \cref{ch:b3:galois}: प्रत्येक अचर
सम्मिश्र बहुपद का एक मूल होता है।

लगातार, $\Omega \subseteq \C $ खुला है, और $ D(a, r)$
खुली डिस्क को दर्शाता है.

\section{जटिल भिन्नता}

\begin{definition}\label{def:b3:holomorphic:holo}
$ f \colon \Omega \to \C $ है \emph{होलोमार्फिक}\index{होलोमार्फिक} पर $\Omega $ यदि प्रत्येक के लिए $ z_0 \in \Omega $
\[
f'(z_0) = \lim_{h\to0}\frac{f(z_0 + h) - f(z_0)}{h}
\]
मौजूद है ($ h \in \C^*$). योग, गुणनफल, भागफल (गायब न होना)।
हर), होलोमोर्फिक फ़ंक्शंस की रचनाएँ हैं
होलोमोर्फिक, \omterm{def:b3:groups:normal}{सामान्य} सूत्रों के साथ (वर्ष 1--2 प्रमाण हैं
शब्दशः: वे केवल फ़ील्ड संचालन और सीमाओं का उपयोग करते हैं)।
$\mathcal H(\Omega)$ होलोमोर्फिक फ़ंक्शंस के सेट को दर्शाता है
पर $\Omega $.
\end{definition}

\begin{proposition}[कॉची--रीमैन]
\label{prop:b3:holomorphic:cauchyriemann}
लिखना $ f(x + \iu y) = P(x,y) + \iu Q(x,y)$. तब $ f $ है
\omterm{def:b3:holomorphic:holo}{होलोमार्फिक} पर $ z_0$ आईएफएफ $ f $ है $\R $-विभेदनीय पर $ z_0$
(दो वास्तविक चरों के मानचित्र के रूप में) और\index{कॉची--रीमैन
समीकरण}
\[
\frac{\partial P}{\partial x} = \frac{\partial Q}{\partial y},
\qquad
\frac{\partial P}{\partial y} = -\frac{\partial Q}{\partial x}
\qquad \text{at } z_0 ;
\]
समतुल्य रूप से, वास्तविक अंतर से गुणा है
सम्मिश्र संख्या $ f'(z_0)$.
\end{proposition}

\begin{proof}
$\C $-विभिन्नता कहती है $ f(z_0 + h) = f(z_0) + ch +
o(\abs h)$ साथ $ c = f'(z_0)$: एक $\R $-रेखीय अंतर
जो कि गुणा है $ c = a + \iu b $, यानी\ मैट्रिक्स है
$\bigl(\begin{smallmatrix} a & -b\\ b & a
\end{smallmatrix}\bigr)$ \omterm{def:b3:topology:basis}{आधार} में $(1, \iu)$ --- बिलकुल
आंशिक के लिए प्रदर्शित संबंध। इसके विपरीत एक
अंतर है $\C $-रैखिक, और $ o(\abs h)$ परिभाषाएँ
मिलान.
\end{proof}

\begin{example}\label{ex:b3:holomorphic:examples}
में बहुपद $ z $, तर्कसंगत कार्य उनके ध्रुवों से दूर होते हैं, और
--- शक्ति के लिए वर्ष 2 के पद-दर-अवधि विभेदन प्रमेय द्वारा
श्रृंखला, जिसका प्रमाण समान रूप से कार्य करता है $\C $ --- प्रत्येक राशि
एक शक्ति श्रृंखला का $\sum a_n(z - a)^n $ इसकी डिस्क के अंदर
अभिसरण: \omterm{def:b3:holomorphic:holo}{होलोमार्फिक}, व्युत्पन्न के साथ $\sum na_n(z -
a)^{n-1}$ (समान त्रिज्या). विशेष रूप से $\exp z = \sum z^n/n!$
संपूर्ण है (\omterm{def:b3:holomorphic:holo}{होलोमार्फिक} पर $\C $) साथ $\exp' = \exp $. वहीं दूसरी ओर $ z \mapsto \bar z $, $\abs z $, $\operatorname{Re}z $
कहीं नहीं हैं \omterm{def:b3:holomorphic:holo}{होलोमार्फिक} (कॉची--रीमैन हर जगह विफल रहता है):
होलोमॉर्फी अभिविन्यास-और-कोण-संरक्षण कठोरता है, नहीं
चिकनाई.
\end{example}

\section{समोच्च अभिन्न}

\begin{definition}\label{def:b3:holomorphic:contour}
ए \emph{पथ} एक टुकड़ावार है $\mathcal C^1$ नक्शा $\gamma \colon
\intcc ab \to \C $; यह है \emph{\omterm{def:b3:topology:interior}{बंद} किया हुआ} अगर $\gamma(a) =
\gamma(b)$. के लिए \omterm{def:b3:topology:continuity}{निरंतर} $ f $ की छवि पर
$\gamma $:\index{समोच्च अभिन्न}
\[
\int_\gamma f(z)\,\dd z =
\int_a^b f(\gamma(t))\,\gamma'(t)\,\dd t,
\qquad
\Bigl|\int_\gamma f\,\dd z\Bigr| \leq
\sup_{\gamma}\abs f\cdot\operatorname{length}(\gamma)
\]
(द \emph{एमएल असमानता}; लंबाई $= \int_a^b\abs{\gamma'}$).
बढ़ते हुए अभिन्न अंग अपरिवर्तनीय है $\mathcal C^1$
पुनर्परिवर्तन और अभिविन्यास उत्क्रमण के अंतर्गत चिह्न में परिवर्तन।
\end{definition}

\begin{proposition}[पुरातन]\label{prop:b3:holomorphic:primitive}
के लिए \omterm{def:b3:topology:continuity}{निरंतर} $ f $ पर $\Omega $, निम्नलिखित समतुल्य हैं:
(मैं) $ f $ एक \emph{प्राचीन} $ F \in \mathcal H(\Omega)$ ($ F'
= f $); (ii) $\int_\gamma f\,\dd z = 0$ प्रत्येक \omterm{def:b3:topology:interior}{बंद} के लिए पथ
$\gamma $ में $\Omega $. उस मामले में $\int_\gamma f\,\dd z =
F(\gamma(b)) - F(\gamma(a))$ हरएक के लिए पथ.
\end{proposition}

\begin{proof}
(मैं)$\Rightarrow $: $\frac{\dd}{\dd t}F(\gamma(t)) =
F'(\gamma(t))\gamma'(t)$ (श्रृंखला नियम, टुकड़े-टुकड़े मान्य), तो
\omterm{thm:b3:complete:completion}{समापन} बिंदु अंतर के लिए अभिन्न दूरबीन; \omterm{def:b3:topology:interior}{बंद} किया हुआ के रास्ते
देना $0$. (ii)$\Rightarrow $(मैं): ठीक करो $ z_*$ में एक \omterm{def:b3:topology:components}{जुड़ा हुआ
घटक}, परिभाषित करना $ F(z) = \int_{\gamma_z}f\,\dd z $ किसी के साथ
पथ से $ z_*$ को $ z $ (अच्छी तरह से परिभाषित: दो के रास्ते ए से भिन्न
\omterm{def:b3:topology:interior}{बंद} पथ); के लिए $ h $ छोटा, से खंड ले रहा है $ z $ को
$ z + h $,
\[
\frac{F(z + h) - F(z)}{h} - f(z)
= \frac1h\int_{[z, z+h]}\bigl(f(w) - f(z)\bigr)\dd w
\longrightarrow 0
\]
एमएल असमानता द्वारा और निरंतरता का $ f $ पर $ z $.
\end{proof}

\begin{definition}[घुमावदार संख्या]\label{def:b3:holomorphic:index}
एक \omterm{def:b3:topology:interior}{बंद} के लिए पथ $\gamma $ और $ z \notin
\operatorname{im}\gamma $, द \emph{अनुक्रमणिका}\index{अनुक्रमणिका}\index{किसी पथ का सूचकांक} है
\[
\operatorname{Ind}_\gamma(z) = \frac1{2\iu\pi}
\int_\gamma\frac{\dd w}{w - z} .
\]
यह है एक \emph{पूर्णांक}: सेटिंग $\varphi(t) =
\int_a^t\frac{\gamma'(s)}{\gamma(s) - z}\dd s $, समारोह
$(\gamma(t) - z)\eu^{-\varphi(t)}$ शून्य व्युत्पन्न है
(टुकड़े-टुकड़े), इसलिए स्थिर है; पर $ t = b $, $\eu^{\varphi(b)} =
\frac{\gamma(b) - z}{\gamma(a) - z} = 1$, इसलिए $\varphi(b) \in
2\iu\pi\Z $. के एक समारोह के रूप में $ z $, सूचकांक है \omterm{def:b3:topology:continuity}{निरंतर} पर
$\C\setminus\operatorname{im}\gamma $ (प्रभुत्व अभिसरण),
इसलिए प्रत्येक पर स्थिर \omterm{def:b3:topology:components}{जुड़ा हुआ घटक}, और $0$ पर
असंबद्ध घटक (एमएल: अभिन्न अंग की ओर प्रवृत्त होता है $0$ जैसा $ z \to
\infty $). सर्कल के लिए $\gamma(t) = a + r\eu^{\iu t}$, $ t
\in \intcc0{2\pi}$: $\operatorname{Ind}_\gamma(z) = 1$ के लिए $ z
\in D(a,r)$ (पर गणना करें $ z = a $: $\frac1{2\iu\pi}
\int_0^{2\pi}\frac{r\iu\eu^{\iu t}}{r\eu^{\iu t}}\dd t = 1$;
निरंतरता बाकी काम करती है)।
\end{definition}

\section{कॉची का प्रमेय}

\begin{theorem}[गौरसैट]\label{thm:b3:holomorphic:goursat}
होने देना $ f \in \mathcal H(\Omega)$ और $ T \subseteq \Omega $ ए
\omterm{def:b3:topology:interior}{बंद} ठोस त्रिकोण. तब $\int_{\partial T}f\,\dd z = 0$
(बाउंड्री रन एक बार, कोई भी ओरिएंटेशन)।\index{गौरसैट का प्रमेय}
\end{theorem}

\begin{proof}
होने देना $ I(T) = \int_{\partial T}f\,\dd z $. मध्यबिंदुओं को जोड़ना
पक्षों का विभाजन $ T $ चार आधे आकार के त्रिकोणों में $ T^{(1)},
\dots, T^{(4)}$, और आंतरिक किनारे जोड़े में रद्द हो जाते हैं: $ I(T) =
\sum_iI(T^{(i)})$. चुनना $ T_1$ उनके बीच में
$\abs{I(T_1)} \geq \frac14\abs{I(T)}$, और पुनरावृत्त: एक नेस्टेड
क्रम $ T \supseteq T_1 \supseteq T_2 \supseteq\cdots $ साथ
\[
\abs{I(T_n)} \geq 4^{-n}\abs{I(T)},
\qquad
\operatorname{diam}T_n = 2^{-n}\operatorname{diam}T,
\quad
\operatorname{length}(\partial T_n) =
2^{-n}\operatorname{length}(\partial T).
\]
चौराहा $\bigcap T_n $ एक बिंदु है $ z_0$
(नेस्टेड कॉम्पैक्ट कैमरों लुप्त होते व्यासों के साथ,
\cref{thm:b3:topology:compactprops}(3)). भिन्नता पर
$ z_0$: दिया गया $\varepsilon $, के लिए $ n $ बड़ा, पर $ T_n $,
\[
f(z) = f(z_0) + f'(z_0)(z - z_0) + R(z),
\qquad \abs{R(z)} \leq \varepsilon\abs{z - z_0} \leq
\varepsilon\operatorname{diam}T_n .
\]
एफ़िन भाग में एक आदिम है: \omterm{def:b3:topology:interior}{बंद} पर इसका अभिन्न अंग
$\partial T_n $ गायब हो जाता है (\cref{prop:b3:holomorphic:primitive}),
जा रहा हूँ
\[
\abs{I(T_n)} = \Bigl|\int_{\partial T_n}R\Bigr|
\leq \varepsilon\operatorname{diam}(T_n)\,
\operatorname{length}(\partial T_n)
= \varepsilon\,4^{-n}\operatorname{diam}(T)
\operatorname{length}(\partial T) .
\]
के साथ तुलना करना $\abs{I(T_n)} \geq 4^{-n}\abs{I(T)}$:
$\abs{I(T)} \leq \varepsilon\cdot\text{const}$ हरएक के लिए
$\varepsilon $: $ I(T) = 0$.
\end{proof}

\begin{theorem}[कॉची का प्रमेय, तारे के आकार का संस्करण]
\label{thm:b3:holomorphic:cauchy}
होने देना $\Omega $ होना \emph{तारे के आकार का} के बारे में $ c $ (प्रत्येक खंड
$[c, z]$, $ z \in \Omega $, में निहित है $\Omega $) --- उदा.\ उत्तल.
प्रत्येक $ f \in \mathcal H(\Omega)$ पर एक आदिम है $\Omega $;
फलस्वरूप $\int_\gamma f\,\dd z = 0$ के लिए \emph{प्रत्येक}
\omterm{def:b3:topology:interior}{बंद} किया हुआ पथ $\gamma $ में $\Omega $.\index{कॉची का प्रमेय}
\end{theorem}

\begin{proof}
परिभाषित करना $ F(z) = \int_{[c,z]}f\,\dd w $. के लिए $ z, z + h \in
\Omega $ साथ $[z, z+h] \subseteq \Omega $ (छोटे के लिए सच है $ h $),
शीर्षों वाला त्रिभुज $ c, z, z+h $ में निहित है $\Omega $
(स्टार-आकार: इसका प्रत्येक बिंदु एक खंड पर स्थित है $[c,
w]$ साथ $ w \in [z, z+h] \subseteq \Omega $): गौरसैट देता है
\[
F(z + h) - F(z) = \int_{[z, z+h]}f\,\dd w,
\]
और अंतर-भागफल की गणना
\cref{prop:b3:holomorphic:primitive} पैदावार $ F' = f $. द
का लुप्त हो जाना \omterm{def:b3:topology:interior}{बंद} रास्ता इंटीग्रल उसी से अनुसरण करता है
प्रस्ताव.
\end{proof}

\begin{theorem}[कॉची का अभिन्न सूत्र]
\label{thm:b3:holomorphic:formula}
होने देना $ f \in \mathcal H(\Omega)$, $\bar D(a, r) \subseteq
\Omega $, और $ C_r $ वृत्त $\partial D(a,r)$ एक बार चलाओ
वामावर्त. फिर हर एक के लिए $ z \in D(a, r)$:\index{कॉची
अभिन्न सूत्र}
\[
f(z) = \frac1{2\iu\pi}\int_{C_r}\frac{f(w)}{w - z}\,\dd w .
\]
\end{theorem}

\begin{proof}
हल करना $ z $ और परिभाषित करें $\Omega $
\[
g(w) = \begin{cases}
\dfrac{f(w) - f(z)}{w - z} & w \neq z,\\[4pt]
f'(z) & w = z :
\end{cases}
\]
$ g $ है \omterm{def:b3:topology:continuity}{निरंतर} पर $\Omega $ और \omterm{def:b3:holomorphic:holo}{होलोमार्फिक} \omterm{def:b3:topology:interior}{बंद} $ z $. गौरसैट
के लिए रखता है $ g $ हर त्रिकोण पर $ T \subseteq \Omega'$, कहाँ
$\Omega'$ से थोड़ी बड़ी डिस्क है $\bar D(a,r)$ अंदर
$\Omega $: अगर $ z \notin T $, सीधे; अगर $ z \in T $, विभाजित करना $ T $
छोटे-छोटे त्रिभुजों में $ z $ एक शीर्ष प्लस त्रिकोण के रूप में
टालना $ z $; शीर्ष वाले त्रिभुज पर $ z $ एमएल बाउंड देता है
$\abs{\int_{\partial T'}g} \leq \sup_{T'}\abs
g\cdot\operatorname{length} \to 0$ जैसे ही त्रिभुज सिकुड़ता है,
और बचे हुए टुकड़े गौरसैट द्वारा गायब हो जाते हैं --- इसलिए
$\int_{\partial T}g = 0$ सभी मामलों में. का प्रमाण
\cref{thm:b3:holomorphic:cauchy} केवल इस त्रिभुज का उपयोग किया गया
संपत्ति: $ g $ उत्तल पर एक आदिम है $\Omega'$, इसलिए
$\int_{C_r}g = 0$, यानी
\[
\frac1{2\iu\pi}\int_{C_r}\frac{f(w)}{w - z}\dd w
= f(z)\,\frac{1}{2\iu\pi}\int_{C_r}\frac{\dd w}{w - z}
= f(z)\operatorname{Ind}_{C_r}(z) = f(z) .
\]
\end{proof}

\section{विश्लेषणात्मकता और उसका झरना}

\begin{theorem}[होलोमार्फिक $=$ विश्लेषणात्मक]
\label{thm:b3:holomorphic:analytic}
होने देना $ f \in \mathcal H(\Omega)$ और $ D(a, R) \subseteq \Omega $.
तब\index{होलोमोर्फिक कार्यों की विश्लेषणात्मकता}
\[
f(z) = \sum_{n\geq0}c_n\,(z - a)^n \quad \text{on } D(a, R),
\qquad
c_n = \frac1{2\iu\pi}\int_{C_r}\frac{f(w)}{(w -
a)^{n+1}}\,\dd w \ \ (0 < r < R),
\]
गुणांक स्वतंत्र हैं $ r $. फलस्वरूप $ f $ है
असीम रूप से $\C $-विभेदनीय, $ c_n = f^{(n)}(a)/n!$, और यह
\emph{कॉची अनुमान}\index{कॉची अनुमान} पकड़ना:
\[
\abs{c_n} \;\leq\; \frac{\sup_{\abs{w - a} =
r}\abs{f(w)}}{r^{n}} .
\]
\end{theorem}

\begin{proof}
के लिए $\abs{z - a} < r $: में कॉची कर्नेल का विस्तार करें
ज्यामितीय श्रृंखला
\[
\frac1{w - z} = \frac1{(w - a)\bigl(1 - \frac{z - a}{w -
a}\bigr)} = \sum_{n\geq0}\frac{(z - a)^n}{(w - a)^{n+1}},
\]
\omterm{def:b3:groups:normal}{सामान्य} रूप से अभिसरण $ w $ पर $ C_r $ ($\abs{\frac{z-a}{w-a}} =
\frac{\abs{z-a}}r < 1$): पद दर पद एकीकृत करें
$\frac{f(w)}{2\iu\pi}$ (समान अभिसरण उचित ठहराता है
इंटरचेंज) और लागू करें \cref{thm:b3:holomorphic:formula}. ए
शक्ति श्रृंखला के साथ असीम रूप से भिन्न है $ c_n =
f^{(n)}(a)/n!$ (वर्ष 2), जो यह भी दर्शाता है $ c_n $ मत करो
पर निर्भर है $ r $. अनुमान: गुणांक अभिन्न बाध्य
एमएल द्वारा.
\end{proof}

\begin{example}[विलक्षणताएँ त्रिज्या निर्धारित करती हैं]
\label{ex:b3:holomorphic:radius}
मासूम असली काम क्यों करता है $\frac1{1 + x^2}$ एक है
टेलर श्रृंखला पर $ x = 3$ केवल के लिए अभिसरण $\abs{x - 3} <
\sqrt{10}$, जब वास्तविक लाइन पर कुछ भी गलत नहीं होता है? क्योंकि
उपरोक्त प्रमेय अभिसरण की त्रिज्या बनाता है $ a $
से दूरी के बराबर $ a $ निकटतम बिंदु पर जहां
होलोमॉर्फी विफल रहता है. यहाँ $ f(z) = \frac1{1 + z^2}$ है
\omterm{def:b3:holomorphic:holo}{होलोमार्फिक} बिलकुल चालू $\C\setminus\{\pm\iu\}$, तो
पर विस्तार $ a = 3$ सबसे बड़ी डिस्क से बचते हुए एकत्रित होता है
$\pm\iu $, त्रिज्या का $\abs{3 - \iu} = \sqrt{10}$ --- और
बड़ी राशि पर एकत्रित नहीं किया जा सकता, क्योंकि राशि बढ़ जाएगी
$ f $ समग्र रूप से एक को \omterm{def:b3:topology:topology}{अड़ोस-पड़ोस} का $\pm\iu $, कहाँ
$\abs f \to \infty $. वास्तविक सिद्धांत रहस्यमय को देखता है
त्रिज्या $\sqrt{10}$; जटिल तल दो ध्रुवों को देखता है।
यह व्यावहारिक नियम है: \emph{की त्रिज्या ज्ञात करने के लिए
अभिसरण, विलक्षणताओं का पता लगाएं} --- जैसे\ टेलर
की श्रृंखला $\tan $ पर $0$ त्रिज्या है $\frac\pi2$ (निकटतम
का शून्य $\cos $), और बर्नौली जनरेटिंग फ़ंक्शन
$\frac z{\eu^z - 1}$ (\cref{pb:b3:holomorphic:1}, भाग VI)
त्रिज्या है $2\pi $ (निकटतम अशून्य शून्य) $\eu^z - 1$:
$\pm2\iu\pi $).
\end{example}

\begin{corollary}[लिउविले; डी'अलेम्बर्ट--गॉस]
\label{cor:b3:holomorphic:liouville}
एक परिबद्ध सम्पूर्ण फलन स्थिर होता है। फलस्वरूप प्रत्येक
अचर बहुपद ओवर $\C $ एक जड़ है: \emph{$\C $ है
\omterm{def:b3:galois:closure}{बीजगणितीय रूप से बंद}}.\index{लिउविले का प्रमेय}\index{मौलिक
बीजगणित का प्रमेय}
\end{corollary}

\begin{proof}
अगर $\abs f \leq M $ पर $\C $: हरएक के लिए $ a $ और $ r $, $\abs{c_1(a)}
= \abs{f'(a)} \leq M/r \to 0$: $ f' \equiv 0$, और $ f $ है
स्थिरांक (पर \omterm{def:b3:topology:connected}{जुड़े हुए} $\C $: शून्य व्युत्पन्न का तात्पर्य है
स्थानीय रूप से स्थिर --- खंडों के साथ एकीकृत करें)। अगर $ P $ नहीं था
जड़, $1/P $ संपूर्ण और परिबद्ध होगा ($\abs{P(z)} \to
\infty $ जैसा $\abs z \to \infty $: प्रमुख पद हावी है, इसलिए
$\abs{1/P}$ एक बड़ी डिस्क के बाहर छोटा है और \omterm{def:b3:topology:continuity}{निरंतर} पर कॉम्पैक्ट डिस्क): स्थिर --- अस्थिर के लिए बेतुका $ P $.
(सप्ताहांत समस्या एक दूसरा, प्रारंभिक प्रमाण और देती है
\omterm{def:b3:galois:algebraic}{बीजगणितीय} नतीजे।)
\end{proof}

\begin{theorem}[शून्य पृथक हैं; पहचान प्रमेय]
\label{thm:b3:holomorphic:identity}
होने देना $\Omega $ होना \emph{\omterm{def:b3:topology:connected}{जुड़े हुए}} और $ f \in \mathcal
H(\Omega)$, $ f \not\equiv 0$. फिर हर शून्य $ a $ का $ f $ है
परिमित \emph{आदेश}: $ f(z) = (z - a)^m\,g(z)$ साथ $ g \in
\mathcal H(\Omega)$, $ g(a) \neq 0$, और के शून्य $ f $ है
कोई संचय बिंदु नहीं $\Omega $. नतीजतन, यदि दो
होलोमोर्फिक कार्य पर $\Omega $ एक के साथ एक सेट पर सहमत हों
संचय बिंदु में $\Omega $, वे सहमत हैं
हर जगह.\index{पहचान प्रमेय}\index{पृथक शून्य}
\end{theorem}

\begin{proof}
होने देना $ Z $ उन बिंदुओं का समूह बनें जहां सभी व्युत्पन्न हों $ f $
गायब होना। $ Z $ \omterm{def:b3:topology:interior}{बंद} है (\omterm{def:b3:topology:interior}{बंद} सेटों का प्रतिच्छेदन) और खुला:
यदि सभी $ c_n = 0$ पर $ a $, शक्ति श्रृंखला विस्तार बनाता है $ f
\equiv 0$ चारों ओर एक डिस्क पर $ a $. संयुक्तता: $ Z =
\varnothing $ या $ Z = \Omega $; उत्तरार्द्ध को बाहर रखा गया है $ f
\not\equiv 0$. तो शून्य पर $ a $, कुछ गुणांक शून्येतर है:
चलो $ m $ के साथ न्यूनतम हो $ c_m \neq 0$; तब $ f(z) = (z -
a)^m\sum_{n\geq m}c_n(z-a)^{n-m}$ एक डिस्क पर, और योग
परिभाषित करता है $ g $ \omterm{def:b3:holomorphic:holo}{होलोमार्फिक} पास में $ a $ साथ $ g(a) = c_m \neq 0$;
बढ़ाना $ g = f/(z-a)^m $ \omterm{def:b3:topology:interior}{बंद} $ a $ (\omterm{def:b3:holomorphic:holo}{होलोमार्फिक} वहाँ)। तब से
$ g(a) \neq 0$ और $ g $ है \omterm{def:b3:topology:continuity}{निरंतर}, $ f $ इसमें कोई अन्य शून्य नहीं है
ए \omterm{def:b3:topology:topology}{अड़ोस-पड़ोस} का $ a $: शून्य पृथक हैं, और का एक सेट
पृथक बिंदुओं में कोई संचय बिंदु नहीं है $\Omega $ (ए.एन
शून्य का संचय बिंदु एक शून्य है --- निरंतरता --- और
पृथक नहीं किया जाएगा) पहचान: अंतर पर लागू करें,
जिसके शून्य सेट में एक संचय बिंदु होता है, जो इसे मजबूर करता है
$ Z = \Omega $ शाखा।
\end{proof}

\begin{theorem}[माध्य मान और अधिकतम मापांक]
\label{thm:b3:holomorphic:maximum}
होने देना $ f \in \mathcal H(\Omega)$.
\begin{enumerate}
  \item (माध्य मान) के लिए $\bar D(a, r) \subseteq \Omega $:
        $ f(a) = \frac1{2\pi}\int_0^{2\pi}f(a +
        r\eu^{\iu t})\,\dd t $.
  \item (अधिकतम सिद्धांत) यदि $\Omega $ है \omterm{def:b3:topology:connected}{जुड़े हुए} और
        $\abs f $ किसी बिंदु पर स्थानीय अधिकतम प्राप्त कर लेता है
        $\Omega $, तब $ f $ स्थिर है. नतीजतन, के लिए
        $\Omega $ घिरा हुआ और $ f $ \omterm{def:b3:topology:continuity}{निरंतर} पर
        $\bar\Omega $: $\sup_{\bar\Omega}\abs f =
        \sup_{\partial\Omega}\abs f $.\index{अधिकतम सिद्धांत}
\end{enumerate}
\end{theorem}

\begin{proof}
(1) कॉची का सूत्र है \omterm{ex:b3:groups:actions}{केंद्र}: पैरामीट्रिज $ C_r $.
(2)कहो $\abs f \leq \abs{f(a)}$ पर $\bar D(a, \rho)
\subseteq \Omega $. अगर $ f(a) = 0$, $ f \equiv 0$ पास में $ a $.
अन्यथा, के लिए $0 < r \leq \rho $, माध्य मान देता है
\[
\abs{f(a)} \leq \frac1{2\pi}\int_0^{2\pi}\abs{f(a +
r\eu^{\iu t})}\,\dd t \leq \abs{f(a)} :
\]
 \omterm{def:b3:topology:continuity}{निरंतर} गैरनकारात्मक एकीकृत $\abs{f(a)} - \abs{f(a +
r\eu^{\iu t})}$ शून्य माध्य है, इसलिए गायब हो जाता है: $\abs f $ है
\emph{स्थिर} $= \abs{f(a)} \ne 0$ डिस्क पर. ए होलोमोर्फिक
समारोह एक डिस्क पर स्थिर अशून्य मापांक का मान स्थिर होता है:
विभेद करना $ P^2 + Q^2 = \text{const}$ देता है $ PP_x + QQ_x
= 0$ और $ PP_y + QQ_y = 0$; कॉची--रीमैन को प्रतिस्थापित करना
रिश्ते $ P_y = -Q_x $, $ Q_y = P_x $ दूसरे समीकरण में
रैखिक प्रणाली उत्पन्न होती है
\[
P\,P_x + Q\,Q_x = 0, \qquad -P\,Q_x + Q\,P_x = 0,
\]
जिसका निर्धारक है $ P^2 + Q^2 \neq 0$: $ P_x = Q_x = 0$, इसलिए
$ f' = P_x + \iu Q_x = 0$ डिस्क पर: $ f $ वहाँ स्थिर. द
पहचान प्रमेय सभी में स्थिरता फैलाता है $\Omega $. सीमा प्रपत्र: $\abs f $
पर अपना समर्थन प्राप्त करता है कॉम्पैक्ट $\bar\Omega $; एक आंतरिक भाग
अधिकतम बनाता है $ f $ स्थिर, और समर्थन प्राप्त होता है पर
हर मामले में सीमा.
\end{proof}

\begin{theorem}[वीयरस्ट्रैस अभिसरण प्रमेय]
\label{thm:b3:holomorphic:weierstrassconv}
अगर $ f_n \in \mathcal H(\Omega)$ में जुटना $ f $ समान रूप से चालू
प्रत्येक कॉम्पैक्ट का उपसमुच्चय $\Omega $, तब $ f \in \mathcal
H(\Omega)$ और $ f_n^{(k)} \to f^{(k)}$ समान रूप से चालू कॉम्पैक्ट कैमरों,
हरएक के लिए $ k $.\index{वीयरस्ट्रैस अभिसरण प्रमेय}
\end{theorem}

\begin{proof}
$ f $ है \omterm{def:b3:topology:continuity}{निरंतर}. किसी भी \omterm{def:b3:topology:interior}{बंद} त्रिकोण के लिए $ T \subseteq
\Omega $: $\int_{\partial T}f = \lim\int_{\partial T}f_n = 0$
(पर एकसमान अभिसरण कॉम्पैक्ट $\partial T $; गौरसैट के लिए
$ f_n $). के तर्क से \cref{thm:b3:holomorphic:cauchy},
$ f $ स्थानीय आदिम हैं (डिस्क उत्तल हैं; केवल त्रिकोण
संपत्ति का उपयोग किया गया था), अर्थात्।\ $ f = F'$ स्थानीय स्तर पर साथ $ F $
\omterm{def:b3:holomorphic:holo}{होलोमार्फिक}; $ F $ विश्लेषणात्मक है
(\cref{thm:b3:holomorphic:analytic}), इसलिए ऐसा है $ f = F'$:
\omterm{def:b3:holomorphic:holo}{होलोमार्फिक}. (यह है \emph{मोरेरा का
प्रमेय}\index{मोरेरा का
प्रमेय}: \omterm{def:b3:topology:continuity}{निरंतर} लुप्त होने के साथ
त्रिभुज समाकलन का तात्पर्य है \omterm{def:b3:holomorphic:holo}{होलोमार्फिक}.) डेरिवेटिव: के लिए
$\bar D(a, 2r) \subseteq \Omega $ और $ z \in \bar D(a, r)$,
डेरिवेटिव के लिए कॉची का सूत्र (अंतर करें)।
\cref{thm:b3:holomorphic:formula} अभिन्न, या उपयोग के अंतर्गत
गुणांक सूत्र) देता है
\[
\abs{f_n'(z) - f'(z)} =
\Bigl|\frac{1}{2\iu\pi}\int_{C_{2r}}\frac{f_n(w) - f(w)}{(w -
z)^2}\,\dd w\Bigr|
\leq \frac{2r\,\sup_{C_{2r}}\abs{f_n - f}}{r^2} \to 0
\]
समान रूप से चालू $\bar D(a,r)$; कवर ए कॉम्पैक्ट बहुत से लोगों द्वारा
ऐसी डिस्क, और उच्चतर के लिए पुनरावृत्त $ k $.
\end{proof}

\begin{method}\label{met:b3:holomorphic:toolkit}
दैनिक टूलकिट. किसी फ़ंक्शन को सिद्ध करने के लिए \omterm{def:b3:holomorphic:holo}{होलोमार्फिक}: इसे प्रदर्शित करें
एक शक्ति श्रृंखला, एक संरचना, एक स्थानीय रूप से समान सीमा के रूप में
(\cref{thm:b3:holomorphic:weierstrassconv}), या एक अभिन्न
के साथ \omterm{def:b3:holomorphic:holo}{होलोमार्फिक} पैरामीटर (\cref{exo:b3:holomorphic:7} ---
के अंतर्गत अंतर करें $\int $ या मोरेरा--फ़ुबिनी) लागू करें। सिद्ध करना
पहचान: उन्हें एक खंड या उपडोमेन पर साबित करें और लागू करें
पहचान प्रमेय. बाध्य करने के लिए: \omterm{thm:b3:holomorphic:analytic}{कॉची अनुमान} पर
सबसे बड़ा सर्कल उपलब्ध है. स्थिरता/अस्तित्व सिद्ध करने के लिए:
लिउविल या अधिकतम सिद्धांत. सदैव जानो \emph{कहाँ}
आपका कार्य है \omterm{def:b3:holomorphic:holo}{होलोमार्फिक} और \emph{कौन} डिस्क फिट होती है
$\Omega $.
\end{method}

\begin{exercise}[$\star\star $]\label{exo:b3:holomorphic:4}
(ए) चलो $ f $ संपूर्ण रूप से साथ रहें $\abs{f(z)} \leq C(1 + \abs
z)^{n}$. बताते हैं कि $ f $ \omterm{def:b3:galois:extension}{डिग्री} का एक बहुपद है $\leq n $

(बी) चलो $ f $ संपूर्ण रूप से साथ रहें $\operatorname{Re}f $ ऊपर से घिरा हुआ.
बताते हैं कि $ f $ स्थिर है \emph{(विचार करना $\eu^{f}$)}.
(सी) ``एफ़िन मानचित्रों के लिए छोटा पिकार्ड'' निकालें: एक संपूर्ण

\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:holomorphic:5}
(ए) चलो $ f $ होना \omterm{def:b3:holomorphic:holo}{होलोमार्फिक} एक पर \omterm{def:b3:topology:connected}{जुड़े हुए} $\Omega \ni 0$ साथ

(बी) कुछ करता है \omterm{def:b3:holomorphic:holo}{होलोमार्फिक} $ f $ पर $\C^*$ संतुष्ट $ f(\frac1n) =
\frac{(-1)^n}{n}$ सभी के लिए $ n \geq 1$? औचित्य।
(सी) दो अलग-अलग प्रदर्शन करें होलोमोर्फिक कार्य पर

\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:holomorphic:6}

(बी) शून्य के साथ एक उदाहरण दीजिए $ f $ स्थिर नहीं है.

\begin{exercise}[$\star\star $]\label{exo:b3:holomorphic:7}
(होलोमॉर्फी अभिन्न के अंतर्गत) चलो $\mu $ एक परिमित हो \omterm{def:b3:measure:measure}{उपाय}

\begin{exercise}[$\star\star\star $]\label{exo:b3:holomorphic:8}

\begin{exercise}[$\star\star\star $]\label{exo:b3:holomorphic:11}
(श्वार्ज़ प्रतिबिंब) चलो $\Omega^+ = \{z : \abs z < 1,\
\operatorname{Im}z > 0\}$, $ I = \intoo{-1}1$, और $ f $
\omterm{def:b3:holomorphic:holo}{होलोमार्फिक} पर $\Omega^+$, \omterm{def:b3:topology:continuity}{निरंतर} पर $\Omega^+\cup I $,
\emph{वास्तविक-मूल्यवान पर $ I $}. परिभाषित करना
\[
F(z) = \begin{cases} f(z) & z \in \Omega^+\cup I,\\
\overline{f(\bar z)} & \bar z \in \Omega^+ . \end{cases}
\]
(ए) वह दिखाओ $ F $ अच्छी तरह से परिभाषित है और \omterm{def:b3:topology:continuity}{निरंतर} पर
$\Omega = \Omega^+\cup I\cup\Omega^-$, और \omterm{def:b3:holomorphic:holo}{होलोमार्फिक} पर
$\Omega^\pm $ \emph{(के लिए $\Omega^-$: कॉची--रीमैन की जाँच करें
के लिए $\overline{f(\bar z)}$, या विस्तार करें $ f $ स्थानीय सत्ता में
श्रृंखला और संयुग्म गुणांक)}.
(बी) वह दिखाओ $ F $ है \omterm{def:b3:holomorphic:holo}{होलोमार्फिक} सब पर $\Omega $ द्वारा
मोरेरा की कसौटी: $\int_{\partial T}F = 0$ प्रत्येक के लिए
त्रिकोण $ T \subseteq \Omega $ \emph{(त्रिकोणों को विभाजित करें $ I $
और उनकी क्षैतिज भुजाओं को अक्ष से दूर धकेलें
$\varepsilon $, वर्दी का उपयोग करना निरंतरता)}.
(सी) निष्कर्ष: ए होलोमोर्फिक फ़ंक्शन डिस्क पर, वास्तविक पर ए
व्यास, संतुष्ट करता है $ f(\bar z) = \overline{f(z)}$; और ए
अस्थिर होलोमोर्फिक फ़ंक्शन इसका वास्तविक मूल्यांकन नहीं किया जा सकता
इसका कोई भी गैररिक्त खुला उपसमुच्चय (\omterm{def:b3:topology:connected}{जुड़े हुए}) कार्यक्षेत्र।
\end{exercise}

\section{समस्या: बीजगणित का मौलिक प्रमेय, दो बार}

\begin{problem}[{सप्ताहांत समस्या --- $\C $ बीजगणितीय रूप से है
बंद: डी'एलेम्बर्ट का प्रमाण, लिउविले का प्रमाण, और
फ़सल}]\label{pb:b3:holomorphic:1}
होने देना $ P(z) = z^n + a_{n-1}z^{n-1} + \dots + a_0$, $ n \geq 1$.
हम इसे दो बार साबित करते हैं $ P $ एक जड़ है, तो क्या बीजगणित एकत्रित करें
तब से इंतज़ार कर रहा हूँ \cref{ch:b3:galois}.

\medskip
\noindent\textbf{Part I --- Coercivity and the minimum.}
\begin{enumerate}
  \item बताते हैं कि $\abs{P(z)} \to +\infty $ जैसा $\abs z \to
        \infty $: एकदम सही, $\abs{P(z)} \geq \frac12\abs z^n $
        के लिए $\abs z \geq R_0$ उपयुक्त।
  \item उसका निष्कर्ष निकालो $\abs P $ पर वैश्विक न्यूनतम प्राप्त करता है $\C $:
        वहाँ है $ z_0$ साथ $\abs{P(z_0)} = \inf_\C\abs P $
        \emph{(एक बड़ी \omterm{def:b3:topology:interior}{बंद} डिस्क की सघनता,
        \cref{cor:b3:topology:heineborel})}.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- d'Alembert's descent.} मान लीजिए,
विरोधाभास के लिए, $ P(z_0) \neq 0$.
\begin{enumerate}[resume]
  \item बढ़ाना $ Q(h) = P(z_0 + h)/P(z_0)$ में एक बहुपद के रूप में
        $ h $: $ Q(h) = 1 + c_kh^k + h^{k+1}S(h)$ साथ $ c_k \neq
        0$, $ k \geq 1$, $ S $ एक बहुपद.
  \item वंश की दिशा चुनें: छोटे के लिए $ t > 0$,
        तय करना $ h = t\,\omega $ कहाँ $\omega^k = -1/c_k $ (क्यों?
        ऐसा करता है $\omega $ अस्तित्व? --- \emph{सिद्ध करना} द
        का अस्तित्व $ k $-किसी सम्मिश्र संख्या का -वाँ मूल
        ध्रुवीय रूप, प्रमेय से स्वतंत्र
        सिद्ध)। दिखाओ
        \[
        \abs{Q(t\omega)} \leq 1 - t^k + C\,t^{k+1}
        \]
        के लिए $ t $ छोटा, एक स्पष्ट स्थिरांक के साथ $ C $.
  \item निष्कर्ष निकालना $\abs{Q(t\omega)} < 1$ छोटे के लिए $ t $ ---
        की न्यूनतमता का खंडन करता है $\abs{P(z_0)}$. इस तरह
        $ P(z_0) = 0$: \emph{प्रत्येक अस्थिर सम्मिश्र
        बहुपद का एक मूल होता है} (डी'एलेम्बर्ट--आर्गैंड)।
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Liouville's one-liner,
in full.}
\begin{enumerate}[resume]
  \item का प्रमाण ध्यानपूर्वक लिखिए
        \cref{cor:b3:holomorphic:liouville}: अगर $ P $ नहीं है
        रूट, उसे सत्यापित करें $1/P $ संपूर्ण है, परिबद्ध है (मात्रा निर्धारित करें,
        प्रश्न 1 का उपयोग करते हुए), इसलिए स्थिर, और निष्कर्ष निकालें।
        दो प्रमाणों की तुलना करें: कौन सी \omterm{def:b3:rings:content}{सामग्री} प्रत्येक में काम करती है
        उपयोग करें? (कॉम्पैक्टनेस दोनों में दिखाई देती है --- कहाँ?)
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The harvest.}
\begin{enumerate}[resume]
  \item वह हर दिखाओ $ P \in \C[X]$ \omterm{def:b3:galois:extension}{डिग्री} का $ n $ विभाजन:
        $ P = c\prod_{i}(X - \alpha_i)^{m_i}$ साथ $\sum m_i =
        n $ \emph{(प्रेरण, यूक्लिडियन डिवीजन द्वारा $(X -
        \alpha)$)}.
  \item दिखाओ कि अलघुकरणीय के बहुपद $\R[X]$ हैं
        रैखिक वाले और नकारात्मक वाले द्विघात
        विभेदक \emph{(युग्मित संयुग्मी जड़ें)}; निष्कर्ष निकालना
        कि विषम \omterm{def:b3:galois:extension}{डिग्री} वाले प्रत्येक वास्तविक बहुपद का एक वास्तविक होता है
        जड़, और दूसरा, क्रम-सैद्धांतिक प्रमाण दें
        वह अंतिम तथ्य (मध्यवर्ती मूल्य प्रमेय) ---
        जाँच कर रहे हैं कि वे किस पर सहमत हैं $ X^3 - X - 1$.
  \item उन ऋणों का अनुमान लगाएं जिन्हें यह पुस्तक अब चुका सकती है: (i) प्रत्येक
        एक गैर-शून्य परिमित-आयामी की एंडोमोर्फिज्म
        $\C $-वेक्टर स्पेस का एक eigenvalue होता है, इसलिए हर कॉम्प्लेक्स
        मैट्रिक्स में एक है \omterm{thm:b3:modules:jordan}{जॉर्डन फॉर्म}
        (\cref{thm:b3:modules:jordan}); (ii) मैदान
        $\bar\Q $ का \omterm{def:b3:galois:algebraic}{बीजगणितीय} में प्रयुक्त संख्याएँ
        \cref{rem:b3:galois:closureexamples} वास्तव में एक है
        \omterm{def:b3:galois:algebraic}{बीजगणितीय} \omterm{thm:b3:complete:completion}{समापन} का $\Q $.
  \item (अंतिम) वह बिंदु जहां प्रत्येक प्रमाण एक से अधिक टूटेगा
        फ़ील्ड की तरह $\Q(\iu)$: कौन से चरण अस्तित्व का उपयोग करते हैं
        $ k $-वें जड़ें (प्रश्न 4), और जो सघनता का उपयोग करती हैं
        या पूर्णता (प्रश्न 2 और 6)? पांच में समाप्त करें
        पंक्तियाँ: प्रमेय वास्तव में है \emph{विश्लेषणात्मक} ---
        प्रत्येक प्रमाण कहीं न कहीं इसका आह्वान करता है पूर्णता या
        संयुक्तता का $\R $ ---भले ही इसका बयान है
        विशुद्ध रूप से \omterm{def:b3:galois:algebraic}{बीजगणितीय}.
\end{enumerate}

\medskip
\noindent\textbf{Part V --- The rigidity ladder of entire
functions.} लिउविले सीढ़ी का पहला पायदान है; हम चढ़ते हैं
यह.
\begin{enumerate}[resume]
  \item (\omterm{thm:b3:holomorphic:analytic}{कॉची अनुमान}) कॉची सूत्र से
        त्रिज्या का वृत्त $ r $ आस-पास $ a $, सिद्ध करना
        \[
        \bigl|f^{(n)}(a)\bigr| \leq
        \frac{n!\,\sup_{\abs{z-a}=r}\abs f}{r^n} ,
        \]
        और मामले के अनुसार लिउविले को पुनर्प्राप्त करें $ n = 1$, $ r \to
        \infty $.
  \item (बहुपद वृद्धि) उसे संपूर्ण दिखाएँ $ f $ साथ
        $\abs{f(z)} \leq A + B\abs z^m $ सभी के लिए $ z $ एक है
        \omterm{def:b3:galois:extension}{डिग्री} का बहुपद $\leq m $ \emph{(टेलर को मार डालो
        गुणांक परे $ m $ प्रश्न 11 के साथ)}.
  \item (असली भाग से घिरा हुआ) उसे \omterm{def:b3:complete:complete}{पूरा} दिखाओ $ f $ साथ
        $\operatorname{Re}f $ ऊपर परिबद्ध स्थिर है
        \emph{(लिउविले को लागू करें $\eu^{f}$)}.
  \item (दोगुनी आवधिकता) चलो $ f $ संपूर्ण रूप से साथ रहें $ f(z + 1)
        = f(z)$ और $ f(z + \iu) = f(z)$ सभी के लिए $ z $. दिखाओ $ f $
        स्थिर है. निष्कर्ष: एक अस्थिर ``अण्डाकार''
        फ़ंक्शन में विलक्षणताएं होनी चाहिए --- ऐतिहासिक
        कारण ध्रुव जटिल विश्लेषण में प्रवेश करते हैं।
  \item (सघन परिसर) दिखाएँ कि एक अचर का परिसर
        संपूर्ण कार्य सघन है $\C $: अगर $ f(\C)$ एक याद आती है
        डिस्क $ D(a, r)$, तब $\frac1{f - a}$ संपूर्ण है और
        घिरा हुआ. (पिकार्ड ने साबित कर दिया कि अधिकतम सीमा चूक जाती है
        \emph{एक बिंदु}; \omterm{ex:b3:lebesgue:gamma}{घनत्व} हमारे उपकरण का स्तर है
        पहुंचें.)
  \item (उचित $\Rightarrow $ बहुपद) मान लीजिए $ f $ है
        संपूर्ण और $\abs{f(z)} \to \infty $ जैसा $\abs z \to
        \infty $. दिखाएँ: के शून्य $ f $ संख्या में सीमित हैं
        ($ z_1, \dots, z_p $, बहुलता के साथ $ m_i $); द
        भागफल $ g = f/\prod(z - z_i)^{m_i}$ संपूर्ण है और
        शून्\omterm{def:b3:modules:free}{य-मुक्त}; $1/g $ इसलिए, बहुपद वृद्धि होती है
        (प्रश्न 12) एक बहुपद है, आवश्यकतः स्थिर
        (शून्\omterm{def:b3:modules:free}{य-मुक्त}); समाप्त करें कि $ f $ एक बहुपद है. तो
        संपूर्ण फलनों के बीच, बहुपद बिल्कुल समान होते हैं
        उचित --- $\eu^z $ उचित रूप से विफल रहता है
        $\R_-$.
\end{enumerate}

\medskip
\noindent\textbf{Part VI --- Harmonic shadows and a mean
value of Gauss.}
\begin{enumerate}[resume]
  \item होने देना $ f = u + \iu v $ होना \omterm{def:b3:holomorphic:holo}{होलोमार्फिक} एक पर \omterm{def:b3:topology:topology}{खुला सेट}.
        सत्यापित करें कि $ u = \operatorname{Re}f $ को संतुष्ट करता है
        औसत मूल्य संपत्ति
        \[
        u(a) = \frac1{2\pi}\int_0^{2\pi}
        u\bigl(a + r\eu^{\iu\theta}\bigr)\,\dd\theta
        \]
        (कौची सूत्र का वास्तविक भाग), और निष्कर्ष निकालें
        के लिए अधिकतम सिद्धांत $ u $ एक बंधे हुए डोमेन पर, के साथ
        वही संयुक्तता के लिए सबूत $\abs f $.
  \item (गॉस का माध्य मान) के लिए $ a \in \C $ और $ r > 0$ साथ
        $\abs a \neq r $, सिद्ध करना
        \[
        \frac1{2\pi}\int_0^{2\pi}
        \log\bigl|a - r\eu^{\iu\theta}\bigr|\,\dd\theta
        = \log\max\bigl(\abs a, r\bigr)
        \]
        \emph{(अगर $\abs a > r $: $ z \mapsto \log\abs{a - z}$
        का असली हिस्सा है \omterm{def:b3:holomorphic:holo}{होलोमार्फिक} ए पर लघुगणक
        \omterm{def:b3:topology:topology}{अड़ोस-पड़ोस} \omterm{def:b3:topology:interior}{बंद} डिस्क का --- कोई क्यों करता है
        अस्तित्व में? --- अतः प्रश्न 17 लागू होता है; अगर $\abs a < r $:
        कारक $\abs{a - r\eu^{\iu\theta}} = r\,\abs{1 -
        \frac ar\eu^{-\iu\theta}}$ और पहले का पुन: उपयोग करें
        मामला)}.
  \item (महलर \omterm{def:b3:measure:measure}{उपाय}) के लिए $ P = c\prod_{i=1}^n(X -
        \alpha_i) \in \C[X]$, निष्कर्ष निकालना \emph{जेन्सेन का सूत्र
        बहुपदों के लिए}:
        \[
        \frac1{2\pi}\int_0^{2\pi}\log\bigl|P(\eu^{\iu\theta})
        \bigr|\,\dd\theta
        = \log\Bigl(\abs c\prod_{i}\max(1,
        \abs{\alpha_i})\Bigr) :
        \]
        का ज्यामितीय माध्य $\abs P $ यूनिट सर्कल पर
        डिस्क के बाहर की जड़ों को पढ़ता है। पर सत्यापित करें $ P =
        X^2 - X $ और पर $ P = 2X - 1$.
  \item (बर्नौली संख्याएँ) गुणांकों को परिभाषित करें $ B_n $ द्वारा
        $\frac{z}{\eu^z - 1} = \sum_{n\geq0}\frac{B_n}{n!}
        z^n $ पास में $0$ (बाईं ओर विश्लेषणात्मक क्यों है?
        $0$?). पुनरावृत्ति प्राप्त करें
        $\sum_{k=0}^{n}\binom{n+1}kB_k = 0$ ($ n \geq 1$) से
        $(\eu^z - 1)\cdot\frac z{\eu^z-1} = z $, गणना करें $ B_0,
        \dots, B_6$, और दिखाओ $ B_{2k+1} = 0$ के लिए $ k \geq 1$
        \emph{(फ़ंक्शन $\frac z{\eu^z-1} + \frac z2$ है
        सम)}. इन नंबरों की कीमत प्रत्येक होगी $\zeta(2k)$ में
        \cref{ch:b3:residues}.
  \item (वास्तविकता) दिखाएँ कि एक संपूर्ण कार्य वास्तविक हो रहा है
        मान चालू $\R $ संतुष्ट $ f(\bar z) =
        \overline{f(z)}$ हर जगह \emph{(टेलर से तुलना करें
        गुणांक पर $0$, या पहचान प्रमेय लागू करें
        को $ z \mapsto \overline{f(\bar z)}$)}; फिर से निष्कर्ष निकालें
        वास्तविक बहुपदों की अवास्तविक जड़ें आती हैं
        संयुग्मी युग्म (प्रश्न 8 का युग्म, निरूपित
        विश्लेषणात्मक रूप से)।
  \item (नैतिक) कठोरता सीढ़ी को इकट्ठा करो: घिरा हुआ
        $\Rightarrow $ स्थिर; बहुपद से घिरा हुआ
        $\Rightarrow $ बहुपद; उचित $\Rightarrow $
        बहुपद; एक डिस्क गायब है $\Rightarrow $ स्थिर;
        दोगुना आवधिक $\Rightarrow $ स्थिर। ए में विरोधाभास
        वास्तविक के साथ लघु अनुच्छेद $\mathcal C^\infty $
        कार्य (टक्कर कार्य,
        \cref{thm:b3:lp:regularization}): क्यों करता है
        होलोमॉर्फी, एक विशुद्ध रूप से स्थानीय स्थिति, वैश्विक लागू करें
        कानून और व्यवस्था?
\end{enumerate}

\medskip
\noindent\textbf{Part VII --- Last harvest.}
\begin{enumerate}[resume]
  \item (लैंडौ की असमानता) के लिए $ P = \sum_{k=0}^na_kX^k $,
        माध्य मान सिद्ध करें $\frac1{2\pi}\int_0^{2\pi}
        \abs{P(\eu^{\iu\theta})}^2\dd\theta =
        \sum_k\abs{a_k}^2$ (की ओर्थोगोनैलिटी
        $\eu^{\iu k\theta}$), फिर, बिंदुवार बाउंड का उपयोग करना
        $\log t \leq t - 1$ के साधनों की तुलना करना
        $\log\abs P^2$ और $\abs P^2$, प्रश्न से निष्कर्ष निकालें
        19 वह
        \[
        \abs c\prod_{i}\max\bigl(1, \abs{\alpha_i}\bigr)
        \;\leq\;
        \Bigl(\sum_{k=0}^{n}\abs{a_k}^2\Bigr)^{1/2} :
        \]
        यूनिट डिस्क के बाहर जड़ों का उत्पाद है
        द्वारा नियंत्रित $\ell^2$ गुणांकों का आकार
        (जड़ों को संभालें \emph{पर} लगाकर वृत्त
        असमानता को $ P(rX)$ और देना $ r \to 1$). जांचें
        यह पर $ X^2 - X $.
  \item (बर्नौली संख्याएँ तथ्यात्मक रूप से बढ़ती हैं) दिखाएँ कि
        के अभिसरण की त्रिज्या $\sum\frac{B_n}{n!}z^n $ है
        \emph{बिल्कुल} $2\pi $: कम से कम $2\pi $ क्योंकि
        $ z/(\eu^z - 1)$ का विस्तार समग्र रूप से को
        $ D(0, 2\pi)$, \omterm{def:b3:rings:primemaximal}{अधिक से अधिक} $2\pi $ क्योंकि योग होगा
        अन्यथा पास ही बंधे रहो $2\pi\iu $, कहाँ
        $\abs{z/(\eu^z-1)} \to \infty $. परिणाम निकालना
        \[
        \limsup_{k\to\infty}
        \Bigl(\frac{\abs{B_{2k}}}{(2k)!}\Bigr)^{1/2k}
        = \frac1{2\pi} :
        \]
        बर्नौली की संख्या तथ्यात्मक रूप से बढ़ती है। स्वीकारते $ B_{12}
        = -\frac{691}{2730}$ (प्रश्न 20 की पुनरावृत्ति,
        आगे दौड़ें), तुलना करें $\abs{B_{12}} \approx 0.25311$
        अधिक तीव्र भविष्यवाणी के साथ $2\,(2k)!/(2\pi)^{2k}
        \approx 0.25305$ पर $ k = 6$ --- चार मिलान अंक
        एक स्पर्शोन्मुख कानून का \cref{ch:b3:residues} होगा
        बिल्कुल सही साबित करें, माध्यम से $\zeta(2k)$.
  \item (जड़ें हिलती हैं लगातार) होने देना $(P_j)$ के प्रति उदासीन रहो
        \omterm{def:b3:galois:extension}{डिग्री} $ n $ उन गुणांकों के साथ अभिसरण के साथ
        (मोनिक) $ P $. सिद्ध करो \emph{कॉची बाउंड}: हर
        एक मोनिक की जड़ $ Q = X^n + \sum_{k<n}q_kX^k $
        संतुष्ट $\abs\alpha \leq 1 + \max_k\abs{q_k}$;
        निष्कर्ष निकालें कि की जड़ें $ P_j $ एक निश्चित अवस्था में रहो
        कॉम्पैक्ट सेट, और, अभिसरण निकालकर
        रूट वैक्टर के अनुवर्ती और पासिंग
        प्रश्न 7 के गुणनखंडन में सीमा, कि
        के रूट मल्टीसेट्स $ P_j $ के साथ अभिसरण करें $ P $.
        अंततः वह दिखाओ निरंतरता वह सर्वोत्तम है जो कोई कर सकता है
        कहो: के लिए $ P_\varepsilon = X^2 - 2X + 1 +
        \varepsilon $, आकार में गड़बड़ी $\varepsilon =
        10^{-4}$ डबल रूट को स्थानांतरित करता है $1$ द्वारा $10^{-2}$ ---
        एच\"पुराने प्रतिपादक $\frac1m $ एक भूरा $ m$-गुना जड़,
        लिप्सचिट्ज़ कभी नहीं: संख्यात्मक रूप से, एकाधिक जड़ों की लागत
        आधे अंक.
\end{enumerate}
\end{problem}
