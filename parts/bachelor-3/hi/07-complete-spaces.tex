\chapter{पूर्ण स्थान: बेयर, एस्कोली, स्टोन--वीयरस्ट्रैस}
\label{ch:b3:complete}

संपूर्णता --- प्रत्येक कॉची अनुक्रम अभिसरित होता है --- है
संपत्ति जो विश्लेषण करने देती है \emph{उत्पादन करना} वस्तुएँ: निश्चित बिंदु
संकुचनों का, श्रृंखलाओं का योग, प्राप्त समीकरणों का समाधान
सीमा के रूप में. यह अध्याय तीन महान अस्तित्वों को एकत्रित करता है
मीट्रिक सिद्धांत की मशीनें। \emph{बेयर का प्रमेय} यह दिखाता है
ए \omterm{def:b3:complete:complete}{पूरा} अंतरिक्ष नगण्य का गणनीय संघ नहीं हो सकता
टुकड़े, और वस्तुओं को जोड़ता है (\omterm{def:b3:topology:continuity}{निरंतर} \omterm{thm:b3:complete:nowherediff}{कहीं भी भिन्न नहीं}
कार्य!) शुद्ध कार्डिनैलिटी-शैली के तर्क से बाहर।
\emph{अर्ज़ेल\`ए--एस्कोली} की पहचान करता है कॉम्पैक्ट के उपसमुच्चय
$\mathcal C(K)$ और विश्लेषण की सघनता का कार्य-घोड़ा है ---
सप्ताहांत समस्या इसका उपयोग पीनो के अस्तित्व प्रमेय को सिद्ध करने के लिए करती है
विभेदक समीकरण. \emph{स्टोन - वीयरस्ट्रैस} दिखाता है
बहुपद, और बहुत कुछ, सघन हैं $\mathcal C(K)$:
सन्निकटन एक बन जाता है \omterm{def:b3:galois:algebraic}{बीजगणितीय} सत्यापन. रास्ते में हम
निर्माण पूर्तियों और इसके लिए विस्तार प्रमेय सिद्ध करें
समान रूप से \omterm{def:b3:topology:continuity}{निरंतर} मानचित्र, की दैनिक रोटी
\cref{ch:b3:lp,ch:b3:hilbert,ch:b3:fouriertransform}.

\section{पूर्ण स्थान, पूर्णताएँ, विस्तार}

\begin{definition}\label{def:b3:complete:complete}
एक मीट्रिक स्थान है \emph{पूरा}\index{पूरा} यदि
प्रत्येक कॉची अनुक्रम अभिसरण करता है (वर्ष 2: $\R^n $ तैयार है;
$\mathcal C(\intcc01)$ साथ $\norm\cdot_\infty $ तैयार है)। ए
पूर्ण स्थान का \omterm{def:b3:topology:interior}{बंद} उपसमुच्चय पूर्ण है; एक पूर्ण उपसमुच्चय
किसी भी मीट्रिक स्थान को \omterm{def:b3:topology:interior}{बंद} कर दिया गया है।
\end{definition}

\begin{proof}
दो कथनों के लिए: \omterm{def:b3:topology:interior}{बंद} का एक कॉची अनुक्रम $ F $
में एकत्रित हो जाता है $ X $, और इसकी सीमा, का पालन $ F $, में निहित है $ F $; ए
अभिसरण-में-$ X $ ए का क्रम \omterm{def:b3:complete:complete}{पूरा} $ A $ कॉची है, इसलिए
में एकत्रित हो जाता है $ A $, और सीमाएँ अद्वितीय हैं।
\end{proof}

\begin{theorem}[समान रूप से सतत मानचित्रों का विस्तार]
\label{thm:b3:complete:extension}
होने देना $ D \subseteq X $ \omterm{def:b3:topology:interior}{घना} हो, $ Y $ \omterm{def:b3:complete:complete}{पूरा}, और $ f \colon D \to
Y $ समान रूप से \omterm{def:b3:topology:continuity}{निरंतर}. तब $ f $ का विस्तार \emph{विशिष्ट} एक को
\omterm{def:b3:topology:continuity}{निरंतर} $\bar f \colon X \to Y $, और $\bar f $ समान रूप से है
\omterm{def:b3:topology:continuity}{निरंतर}.\index{विस्तार प्रमेय (समान निरंतरता)}
\end{theorem}

\begin{proof}
विशिष्टता: दो \omterm{def:b3:topology:continuity}{निरंतर} एक्सटेंशन घने पर सहमत हैं $ D $,
इसलिए हर जगह (समझौता सेट) $\{g = h\}$ \omterm{def:b3:topology:interior}{बंद} है:
नीचे \omterm{def:b3:topology:interior}{बंद} विकर्ण की पूर्वछवि $ x\mapsto(g(x), h(x))$).
अस्तित्व: के लिए $ x \in X $ चुनना $ d_n \to x $, $ d_n \in D $. द
क्रम $(f(d_n))$ कॉची है: दिया गया $\varepsilon $, वर्दी
निरंतरता प्रदान $\delta $ साथ $ d(u,v) < \delta \Rightarrow
d(f(u), f(v)) < \varepsilon $, और $(d_n)$ कॉची है. परिभाषित करना
$\bar f(x) = \lim f(d_n)$; सीमा इस पर निर्भर नहीं करती
चयनित अनुक्रम (उनमें से दो को गूंथ लें)। $\bar f $ का विस्तार $ f $
(\omterm{def:b3:topology:continuity}{निरंतर} अनुक्रम) और का मापांक विरासत में मिला है निरंतरता: अगर
$ d(x, x') < \delta $, दोनों के बिंदुओं द्वारा अनुमान लगाया जा रहा है $ D $ पर
दूरी $< \frac{\delta - d(x,x')}2$ देता है $ d(\bar f(x), \bar
f(x')) \leq \varepsilon $ सीमा में --- $\bar f $ समान रूप से है
\omterm{def:b3:topology:continuity}{निरंतर}.
\end{proof}

\begin{theorem}[समापन]\label{thm:b3:complete:completion}
प्रत्येक मीट्रिक स्थान $ X $ के सघन उपसमुच्चय के रूप में सममितीय रूप से एम्बेड होता है
ए पूर्ण मीट्रिक स्थान $\hat X $, आइसोमेट्री फिक्सिंग तक अद्वितीय
$ X $ बिंदुवार: यह \emph{समापन}\index{समापन}.
\end{theorem}

\begin{proof}
\emph{अस्तित्व।} होने देना $\mathcal X $ कॉची का सेट हो
के अनुक्रम $ X $, छद्म दूरी के साथ
\[
D\bigl((x_n), (y_n)\bigr) = \lim_n d(x_n, y_n),
\]
मौजूदा सीमा क्योंकि $\abs{d(x_n, y_n) - d(x_m, y_m)}
\leq d(x_n, x_m) + d(y_n, y_m)$ वास्तविक अनुक्रम बनाता है
कॉची. होने देना $\hat X = \mathcal X/{\sim}$,
अनुक्रमों की पहचान करना $ D $-दूरी $0$; $ D $ ए तक उतरता है
दूरी. एम्बेड $ X $ \omterm{def:b3:topology:continuity}{निरंतर} अनुक्रमों द्वारा: एक आइसोमेट्री, साथ
सघन छवि (एक कॉची अनुक्रम है $ D $- द्वारा अनुमानित
अपनी शर्तों पर निर्मित स्थिरांक: $ D\bigl((x_n), (x_k)_{\rm
const}\bigr) = \lim_n d(x_n, x_k) \to 0$ जैसा $ k \to \infty $ द्वारा
कॉचीनेस)। संपूर्णता का $\hat X $: होने देना $(\xi^k)$ कॉची हो
में $\hat X $; घनत्व चयन द्वारा $ x_k \in X $ साथ $ D(\xi^k, x_k)
\leq 2^{-k}$; तब $(x_k)$ कॉची में है $ X $ (त्रिकोण
के माध्यम से असमानता $\xi $'s), एक बिंदु को परिभाषित करता है $\xi \in \hat
X $, और $ D(\xi^k, \xi) \leq 2^{-k} + D(x_k, \xi) \to 0$ (द
स्थिरांक से दूरी $ x_k $ की \omterm{def:b3:groups:action}{कक्षा} को $(x_j)_j $ है
$\lim_j d(x_k, x_j)$, बड़े के लिए छोटा $ k $).

\emph{विशिष्टता}: दो पूर्तियों $\hat X_1, \hat X_2$ रोकना
$ X $ सघन रूप से; की पहचान $ X $, एक आइसोमेट्री, समान रूप से है
\omterm{def:b3:topology:continuity}{निरंतर}, तो तक विस्तारित है $\hat X_1 \to \hat X_2$
(\cref{thm:b3:complete:extension}), फिर भी ए पर एक आइसोमेट्री \omterm{def:b3:topology:interior}{घना}
सेट इसलिए हर जगह; दूसरी दिशा में सममित रूप से, और
कंपोजिट सघनता को ठीक करते हैं $ X $: वे पहचान हैं.
\end{proof}

\begin{theorem}[बानाच निश्चित बिंदु]\label{thm:b3:complete:banach}
होने देना $ X $ होना \omterm{def:b3:complete:complete}{पूरा}, गैर-रिक्त, और $ f \colon X \to X $ ए
\emph{सिकुड़न}: $ d(f(x), f(y)) \leq k\,d(x,y)$ साथ $ k < 1$.
तब $ f $ एक अद्वितीय निश्चित बिंदु है $ x^*$, और हर \omterm{def:b3:groups:action}{कक्षा}
स्पष्ट दर के साथ इसमें अभिसरण होता है $ d(x_n, x^*) \leq
\frac{k^n}{1-k}\,d(x_1, x_0)$.\index{बानाच निश्चित बिंदु प्रमेय}
\end{theorem}

\begin{proof}
(वर्ष 2 ने इसे साबित कर दिया; हम इसके लिए दो-पंक्ति तर्क को फिर से रिकॉर्ड करते हैं
आत्मनिर्भरता.) द \omterm{def:b3:groups:action}{कक्षा} $ x_{n+1} = f(x_n)$ है $ d(x_{n+1},
x_n) \leq k^nd(x_1, x_0)$, इसलिए कॉची (ज्यामितीय श्रृंखला) है;
इसकी सीमा $ x^*$ निश्चित है (निरंतरता का $ f $), दो के बाद से अद्वितीय
निश्चित अंक संतुष्ट करते हैं $ d \leq k\,d $. दर: ज्यामितीय योग
पूंछ.
\end{proof}

\begin{example}[पहचान बिगाड़ना]
\label{ex:b3:complete:perturbident}
होने देना $ g \colon \R^d \to \R^d $ होना $ k $-लिप्सचिट्ज़ के साथ $ k < 1$.
तब $\varphi = \mathrm{id} + g $ एक है \emph{\omterm{def:b3:topology:continuity}{होमियोमोर्फिज्म} का
$\R^d $ पर $\R^d $}. इंजेक्शन, एक मात्रात्मक के साथ
मापांक:
\[
\norm{\varphi(x) - \varphi(y)} \geq \norm{x - y} -
\norm{g(x) - g(y)} \geq (1 - k)\norm{x - y} .
\]
प्रत्यक्षता निश्चित बिंदु प्रमेय है: हल करना
$\varphi(x) = y $ मतलब $ x = y - g(x)$, और $ x \mapsto y -
g(x)$ एक है $ k $- का संकुचन \omterm{def:b3:complete:complete}{पूरा} $\R^d $ --- ए
अनोखा समाधान $ x = \psi(y)$ प्रत्येक के लिए मौजूद है $ y $. द
प्रदर्शित असमानता उलटा बनाती है $\psi $ लिप्सचिट्ज़ के साथ
स्थिर $\frac1{1-k}$: ए \omterm{def:b3:topology:continuity}{होमियोमोर्फिज्म}, स्पष्ट के साथ
दोनों \omterm{def:b3:modules:module}{मॉड्यूल} पर सीमाएं। यह अहानिकर दिखने वाला कथन है
व्युत्क्रम फलन प्रमेय के अंदर इंजन
(\cref{ch:b3:submanifolds}): एक बिंदु के पास जहां $ Df $ है
उलटा, $ f $ \emph{है} एक उलटा रेखीय मानचित्र प्लस ए
छोटी लिप्सचिट्ज़ गड़बड़ी, और आज का उदाहरण यही करता है
विश्राम. यह संख्यात्मक मजबूती की मात्रा भी निर्धारित करता है: एक प्रणाली
व्युत्क्रम का मार्जिन कम रहने से परेशान
\omterm{def:b3:groups:derived}{व्याख्या करने योग्य}, समाधान \omterm{def:b3:rings:primemaximal}{अधिक से अधिक} गतिमान होने के साथ
$\frac{1}{1-k}$ कई बार गड़बड़ी.
\end{example}

\section{बेयर का प्रमेय}

\begin{theorem}[बेयर]\label{thm:b3:complete:baire}
में एक पूर्ण मीट्रिक स्थान, घने का एक गणनीय चौराहा
खुले सेट \omterm{def:b3:topology:interior}{घना} है. समतुल्य: यदि $ X = \bigcup_{n}F_n $ के साथ
प्रत्येक $ F_n $ \omterm{def:b3:topology:interior}{बंद}, फिर कुछ $ F_n $ शून्य है
आंतरिक भाग.\index{बेयर का प्रमेय}
\end{theorem}

\begin{proof}
होने देना $(U_n)$ \omterm{def:b3:topology:interior}{घना} खुलता हो और $ B_0 = B(x_0, r_0)$ कोई भी खुला
गेंद; हमें एक बिंदु मिलता है $\bigcap U_n $ में $ B_0$. आगमनात्मक रूप से:
$ U_{n}$ \omterm{def:b3:topology:interior}{घना} और खुला होने के कारण यह खुली गेंद से मिलता है $ B_{n-1}$
एक में \omterm{def:b3:topology:topology}{खुला सेट}, जिसमें एक \omterm{def:b3:topology:interior}{बंद} गेंद है $\bar B(x_n, r_n)$
साथ $0 < r_n \leq r_{n-1}/2$ और $\bar B(x_n, r_n) \subseteq
B_{n-1}\cap U_n $. केन्द्रों एक कॉची अनुक्रम बनाएं ($ x_m \in
B_n $ के लिए $ m \geq n $, त्रिज्या $\to 0$); सीमा $ x $ में निहित है
प्रत्येक $\bar B(x_n, r_n)$ (बंदपन), इसलिए हर में $ U_n $ और
में $ B_0$. दूसरे फॉर्म के लिए: यदि नहीं $ F_n $ है आंतरिक भाग, द
$ U_n = X \setminus F_n $ खुले और घने हैं, और एक बिंदु हैं
$\bigcap U_n $ निकल जाता है $\bigcup F_n = X $: बेतुका।
\end{proof}

\begin{remark}\label{rem:b3:complete:meagre}
शब्दावली: एक समुच्चय है \emph{कहीं सघन नहीं} यदि यह है \omterm{def:b3:topology:interior}{बंद} है
खाली आंतरिक भाग, \emph{अल्प}\index{अल्प} (पहली श्रेणी)
यदि यह कहीं का गणनीय संघ है घने सेट. बैरे:
\emph{ए पूर्ण मीट्रिक स्थान क्या नहीं है \omterm{rem:b3:complete:meagre}{अल्प} अपने आप में}, और
ए का पूरक \omterm{rem:b3:complete:meagre}{अल्प} सेट \omterm{def:b3:topology:interior}{घना} है. ``\omterm{rem:b3:complete:meagre}{अल्प}'' की एक धारणा है
मापने के लिए लघुता ओर्थोगोनल (\cref{ch:b3:measure} होगा
उपज \omterm{rem:b3:complete:meagre}{अल्प} सेट पूर्ण माप का), और बेयर तर्क साबित करते हैं
\emph{बहुतायत से अस्तित्व}: बिना किसी एक वस्तु को प्रदर्शित करना
संपत्ति पी, पी-ऑब्जेक्ट्स फॉर्म ए दिखाएं \omterm{rem:b3:complete:meagre}{अल्प} सेट.
\end{remark}

\begin{corollary}\label{cor:b3:complete:bairecors}
(ए) $\R $ बेशुमार है. (बी) $\Q $ गणनीय नहीं है
के खुले उपसमुच्चय का प्रतिच्छेदन $\R $, और एक गैररिक्त पूर्ण
मीट्रिक स्थान अलग-अलग बिंदुओं के बिना गिनती नहीं की जा सकती।
\end{corollary}

\begin{proof}
(ए) $\R = \bigcup_{x}\{x\}$ एक गणनीय समुच्चय पर कुछ बनेगा
सिंगलटन के पास है आंतरिक भाग. (बी) यदि $\Q = \bigcap_n V_n $ साथ $ V_n $
खुला (आवश्यक रूप से \omterm{def:b3:topology:interior}{घना}, जैसे $\supseteq \Q $), फिर सेट
$ V_n $ और पूरक $\R\setminus\{q\}$, $ q \in \Q $, फॉर्म ए
खाली चौराहे के साथ खुलता सघन का गणनीय परिवार ---
बेयर का खंडन। अगर $ X $ है \omterm{def:b3:complete:complete}{पूरा} पृथक बिंदुओं के बिना
और गणनीय, $ X = \bigcup_{x \in X}\{x\}$ इसे एक के रूप में प्रदर्शित करता है
खाली के साथ \omterm{def:b3:topology:interior}{बंद} सेट का गणनीय संघ आंतरिक भाग (कोई पृथक नहीं
अंक): बेयर फिर से।
\end{proof}

\begin{theorem}[वीयरस्ट्रैस के राक्षस मौजूद हैं]
\label{thm:b3:complete:nowherediff}
वहां है \omterm{def:b3:topology:continuity}{निरंतर} पर कार्य करता है $\intcc01$ पर भिन्न
कोई मतलब नहीं. दरअसल, का सेट $ f \in \mathcal C(\intcc01)$
एक बिंदु पर भी (परिमित) व्युत्पन्न होना है \omterm{rem:b3:complete:meagre}{अल्प} में
$(\mathcal C(\intcc01), \norm\cdot_\infty)$.\index{कहीं नहीं
विभेदनीय कार्य}
\end{theorem}

\begin{proof}
के लिए $ n \geq 1$ होने देना
\[
F_n = \Bigl\{f : \exists x \in \intcc01,\
\forall h \neq 0 \text{ with } x + h \in \intcc01,\
\abs{f(x+h) - f(x)} \leq n\abs h \Bigr\}.
\]
अगर $ f $ पर भिन्न है $ x $, तब $ f \in F_n $ कुछ के लिए $ n $:
भागफल $\abs{f(x+h)-f(x)}/\abs h $ के लिए बाध्य है $\abs h
\leq \delta $ (विभिन्नता: इसकी प्रवृत्ति होती है
$\abs{f'(x)}$) और से घिरा हुआ $2\norm f_\infty/\delta $ के लिए
$\abs h \geq \delta $. इसलिए $\bigcup F_n $ शामिल है
सभी कहीं न कहीं भिन्न-भिन्न कार्य करते हैं, और यह दिखाने के लिए पर्याप्त है
प्रत्येक $ F_n $ खाली के साथ \omterm{def:b3:topology:interior}{बंद} है आंतरिक भाग.

\emph{\omterm{def:b3:topology:interior}{बंद} किया हुआ}: होने देना $ f_k \to f $ समान रूप से, $ f_k \in F_n $ के साथ
गवाह $ x_k \to x $ (कॉम्पैक्टनेस, निष्कर्षण के बाद)। के लिए $ h $
साथ $ x + h \in \intcc01$: चुनना $ h_k \to h $ साथ $ x_k + h_k
\in \intcc01$ (उदा.\ $ h_k = h + x - x_k $ क्लिप किया गया); तब
$\abs{f(x + h) - f(x)} = \lim \abs{f_k(x_k + h_k) - f_k(x_k)}
\leq \lim n\abs{h_k} = n\abs h $, एकसमान अभिसरण का उपयोग करते हुए और
निरंतरता का $ f $ प्रासंगिक बिंदुओं पर: $ f \in F_n $.

\emph{खाली आंतरिक भाग}: दिया गया $ f \in F_n $ और $\varepsilon > 0$,
हम देखतें है $ g $ साथ $\norm{g - f}_\infty \leq \varepsilon $ और $ g
\notin F_n $. पहला अनुमान $ f $ अंदर $\varepsilon/2$ ए द्वारा
टुकड़े-टुकड़े एफ़िन $\varphi $ (वर्दी निरंतरता: ए पर प्रक्षेप करें
बारीक ग्रिड), कुछ से घिरी ढलानों का $ M $. एक छोटा चूरा जोड़ें:
$ g = \varphi + \frac\varepsilon2\,s_N $, कहाँ $ s_N(x)$ है
$\frac1N $-आयाम का आवधिक ज़िगज़ैग $1$ और ढलान $\pm 2N $.
प्रत्येक पर $ x $, एक तरफ है $ h $ मनमाने ढंग से छोटा साथ
सॉटूथ ढलान में योगदान दे रहा है $\pm 2N\cdot\frac\varepsilon2 =
\pm\varepsilon N $ ऊपर $[x, x+h]$: का अंतर भागफल
$ g $ से अधिक है $\varepsilon N - M > n $ के लिए $ N $ बड़ा। इसलिए $ g \notin
F_n $, किसी भी समान दूरी पर $\leq \varepsilon $ से $ f $.
निष्कर्ष: $\bigcup F_n $ है \omterm{rem:b3:complete:meagre}{अल्प}; बेयर द्वारा इसका पूरक
--- से बना है कहीं भी भिन्न कार्य नहीं --- में \omterm{def:b3:topology:interior}{घना} है
$\mathcal C(\intcc01)$: ऐसे कार्य मौजूद हैं \emph{पर्याप्त रूप से}.
\end{proof}

\section{अर्ज़ेल\`ए--एस्कोली}

लगातार, $ K $ एक है \emph{कॉम्पैक्ट} मीट्रिक स्थान और $\mathcal
C(K) = \mathcal C(K, \R^d)$, साथ $\norm f_\infty = \sup_K
\norm{f(x)}$: ए \omterm{def:b3:complete:complete}{पूरा} स्थान (समान सीमाएं) \omterm{def:b3:topology:continuity}{निरंतर}
हैं \omterm{def:b3:topology:continuity}{निरंतर} ---वर्ष 2).

\begin{definition}\label{def:b3:complete:equicontinuous}
एक परिवार $\mathcal F \subseteq \mathcal C(K)$ है
\emph{समनिरंतर}\index{समनिरंतर} यदि प्रत्येक के लिए
$\varepsilon > 0$ वहाँ है $\delta > 0$ ऐसा है कि
\[
d(x, y) < \delta \implies \norm{f(x) - f(y)} < \varepsilon
\quad\text{for \emph{सभी} } f \in \mathcal F
\]
(एक $\delta $ पूरे परिवार के लिए --- उदा.\ किसी भी परिवार के पास
\omterm{def:b3:groups:normal}{सामान्य} लिप्सचिट्ज़ स्थिरांक, या एक \omterm{def:b3:groups:normal}{सामान्य} एच\"पुराने मापांक), और
\emph{बिन्दुवार परिबद्ध} अगर $\sup_{f}\norm{f(x)} < \infty $ प्रत्येक के लिए $ x $.
\end{definition}

\begin{theorem}[अर्ज़ेल\`ए--एस्कोली]\label{thm:b3:complete:ascoli}
एक उपसमुच्चय $\mathcal F \subseteq \mathcal C(K)$ अपेक्षाकृत है
कॉम्पैक्ट (है कॉम्पैक्ट \omterm{def:b3:topology:interior}{बंद}) यदि यह है \omterm{def:b3:complete:equicontinuous}{समनिरंतर} और
बिन्दुवार परिबद्ध। विशेष रूप से, प्रत्येक \omterm{def:b3:complete:equicontinuous}{समनिरंतर},
बिन्दुवार परिबद्ध \emph{अनुक्रम} एक समान रूप से अभिसरण है
परिणाम।\index{Arzelà--Ascoli theorem}
\end{theorem}

\begin{proof}
($\Leftarrow $) होने देना $(f_k)$ का एक क्रम हो $\mathcal F $. $ K $
कॉम्पैक्ट मीट्रिक है \emph{\omterm{def:b3:galois:separable}{वियोज्य}}: प्रत्येक के लिए $ n $, निश्चित रूप से अनेक
त्रिज्या की गेंदें $\frac1n $ ढकना $ K $
(\cref{thm:b3:topology:metriccompact}); उनका केन्द्रों फॉर्म ए
गणनीय \omterm{def:b3:topology:interior}{घना} सेट $ D = \{x_1, x_2, \dots\}$. बिंदुवार
बाउंडेडनेस और बोलजानो--वीयरस्ट्रैस, क्रमिक रूप से निकालें
परिणाम एकाग्र हो रहे हैं $ x_1$, फिर भी $ x_2$, आदि, और
ले लो \emph{विकर्ण} परिणाम को $(g_j)$: यह पर एकत्रित होता है
के हर बिंदु $ D $. समनिरंतरता इसे वर्दी में अपग्रेड करें
कॉची: दिया गया $\varepsilon $, लेना $\delta $ जैसे कि
परिभाषा, आवरण $ K $ बहुत सी गेंदों द्वारा $ B(x_i, \delta)$
साथ $ x_i \in D $ ($ i \leq m $), और चुनें $ J $ इतना बड़ा कि
$\norm{g_j(x_i) - g_{j'}(x_i)} < \varepsilon $ के लिए $ j, j' \geq
J $, $ i \leq m $. मनमानी के लिए $ x \in B(x_i, \delta)$:
\[
\norm{g_j(x) - g_{j'}(x)}
\leq \norm{g_j(x) - g_j(x_i)} + \norm{g_j(x_i) - g_{j'}(x_i)}
+ \norm{g_{j'}(x_i) - g_{j'}(x)} < 3\varepsilon .
\]
इसलिए $(g_j)$ समान रूप से कॉची है, और में परिवर्तित हो जाता है \omterm{def:b3:complete:complete}{पूरा}
$\mathcal C(K)$. इस प्रकार प्रत्येक क्रम $\mathcal F $ एक है
अभिसरण अनुवर्ती: $\bar{\mathcal F}$ है (क्रमिक रूप से,
इसलिए द्वारा \cref{thm:b3:topology:metriccompact}) कॉम्पैक्ट.

($\Rightarrow $) अगर $\bar{\mathcal F}$ है कॉम्पैक्ट:बिंदुवार
सीमा स्पष्ट है (मूल्यांकन है \omterm{def:b3:topology:continuity}{निरंतर}). के लिए
समनिरंतरता, ढकना $\mathcal F $ बहुत सी गेंदों द्वारा
$ B(f_i, \varepsilon)$ का $\mathcal C(K)$; प्रत्येक $ f_i $ है
समान रूप से \omterm{def:b3:topology:continuity}{निरंतर} (हेन,
\cref{cor:b3:topology:heineborel}), एक आम देना $\delta $
के लिए $ i \leq m $; फिर के लिए $ f \in B(f_i, \varepsilon)$ और $ d(x,y)
< \delta $: $\norm{f(x)-f(y)} \leq 2\varepsilon +
\norm{f_i(x)-f_i(y)} < 3\varepsilon $.
\end{proof}

\begin{example}\label{ex:b3:complete:ascoliexamples}
की \omterm{def:b3:topology:interior}{बंद} इकाई गेंद $\mathcal C(\intcc01)$ है \emph{नहीं}
कॉम्पैक्ट ($ f_n(x) = x^n $ कोई समान रूप से अभिसरण नहीं है
परिणाम: बिंदुवार सीमा असंतत है), और वास्तव में
$(x^n)$ क्या नहीं है \omterm{def:b3:complete:equicontinuous}{समनिरंतर} पर $1$. इसके विपरीत $\{f : \norm
f_\infty \leq 1,\ \operatorname{Lip}(f) \leq 1\}$ है कॉम्पैक्ट:
घिरा हुआ और $1$-लिप्सचिट्ज़-समविराम, और \omterm{def:b3:topology:interior}{बंद}। एस्कोली
समझाता है \emph{क्यों} अनंत आयाम में सघनता विफल हो जाती है
(रिस्ज़, वर्ष 2) और इसे पुनर्स्थापित करने के लिए क्या जोड़ना है: एक वर्दी
का मापांक निरंतरता.
\end{example}

\section{स्टोन - वीयरस्ट्रैस}

\begin{lemma}[दीनी]\label{lem:b3:complete:dini}
होने देना $ K $ होना कॉम्पैक्ट और $(f_n)$ ए \emph{एक लय} का क्रम
\omterm{def:b3:topology:continuity}{निरंतर} वास्तविक कार्यों का अभिसरण \emph{बिंदुवार} एक को
\omterm{def:b3:topology:continuity}{निरंतर} $ f $. फिर अभिसरण है
वर्दी.\index{दीनी का प्रमेय}
\end{lemma}

\begin{proof}
कहना $ f_n \uparrow f $; होने देना $ g_n = f - f_n \downarrow 0$,
\omterm{def:b3:topology:continuity}{निरंतर}. दिया गया $\varepsilon $, द खुले सेट $ U_n = \{g_n <
\varepsilon\}$ बढ़ाएँ और कवर करें $ K $ (बिंदुवार अभिसरण);
एक परिमित उपकवर निकालें: $ K = U_{n_0}$ कुछ के लिए $ n_0$
(परिवार बढ़ाना), यानी\ $0 \leq g_n < \varepsilon $
हर जगह के लिए $ n \geq n_0$.
\end{proof}

\begin{lemma}\label{lem:b3:complete:sqrt}
का एक क्रम है \emph{बहुआयामी पद} $ u_n $ साथ $ u_n(t) \to
\sqrt t $ समान रूप से चालू $\intcc01$.
\end{lemma}

\begin{proof}
परिभाषित करना $ u_0 = 0$, $ u_{n+1}(t) = u_n(t) + \frac12\bigl(t -
u_n(t)^2\bigr)$: बहुपद. प्रेरण द्वारा $0 \leq u_n(t) \leq
\sqrt t $ पर $\intcc01$: इसके लिए अनुदान देना $ n $,
\[
\sqrt t - u_{n+1}(t) = \bigl(\sqrt t - u_n(t)\bigr)
\Bigl(1 - \tfrac12\bigl(\sqrt t + u_n(t)\bigr)\Bigr) \geq 0,
\]
तब से $\sqrt t + u_n \leq 2$; और $ u_{n+1} \geq u_n \geq 0$.
इसलिए $(u_n(t))$ घटता नहीं, घिरा हुआ है $\sqrt t $: यह
बिंदुवार अभिसरण होता है, और सीमा $\ell(t)$ संतुष्ट $\ell =
\ell + \frac12(t - \ell^2)$: $\ell(t) = \sqrt t $, \omterm{def:b3:topology:continuity}{निरंतर}.
दीनी (\cref{lem:b3:complete:dini}) वर्दी में उन्नयन।
\end{proof}

\begin{theorem}[स्टोन--वीयरस्ट्रैस, वास्तविक संस्करण]
\label{thm:b3:complete:stoneweierstrass}
होने देना $ K $ एक हो कॉम्पैक्ट (\omterm{def:b3:topology:hausdorff}{हॉसडॉर्फ़}) अंतरिक्ष और $\mathcal A \subseteq
\mathcal C(K, \R)$ ए \emph{उपबीजगणित} (राशि के तहत स्थिर,
गुणनफल, अदिश गुणज) वह \emph{स्थिरांक शामिल हैं}
और \emph{अंक अलग करता है} (के लिए $ x \neq y $, कुछ $ f \in
\mathcal A $ है $ f(x) \neq f(y)$). तब $\mathcal A $ में सघन है
$(\mathcal C(K,\R), \norm\cdot_\infty)$.\index{स्टोन - वीयरस्ट्रैस
प्रमेय}
\end{theorem}

\begin{proof}
होने देना $\bar{\mathcal A}$ हो \omterm{def:b3:topology:interior}{बंद}, फिर से एक बीजगणित
(परिबद्ध सेटों पर समान सीमा के उत्पाद अभिसरित होते हैं)।

\emph{स्टेप 1: $\bar{\mathcal A}$ एक जाली है}, यानी\ स्थिर
के अंतर्गत $\max $ और $\min $. तब से $\max(f,g) = \frac{f + g +
\abs{f-g}}2$ और $\min $ इसी तरह, यह पर्याप्त है $ f \in
\bar{\mathcal A} \Rightarrow \abs f \in \bar{\mathcal A}$: साथ
$ M = \norm f_\infty > 0$, $\abs f = M\sqrt{(f/M)^2}$, और
\cref{lem:b3:complete:sqrt} बहुपद देता है $ u_n $ साथ
$ u_n\bigl((f/M)^2\bigr) \to \abs f/M $ समान रूप से; बहुपद में
बीजगणित के सदस्य (स्थिर पद के साथ: स्थिरांक हैं
वहां) अंदर रहो $\bar{\mathcal A}$.

\emph{चरण 2: दो-बिंदु प्रक्षेप।} के लिए $ x \neq y $ और $ a,
b \in \R $, कुछ $ g \in \mathcal A $ है $ g(x) = a $, $ g(y) = b $:
लेना $ h $ पृथक करना $ x, y $ और सेट करें $ g = a + (b -
a)\frac{h - h(x)}{h(y) - h(x)}$.

\emph{चरण 3.} होने देना $ f \in \mathcal C(K)$, $\varepsilon > 0$.
प्रत्येक जोड़ी के लिए $ x, y $ चुनना $ g_{x,y} \in \mathcal A $ साथ
$ g_{x,y}(x) = f(x)$, $ g_{x,y}(y) = f(y)$ (चरण 2; के लिए $ x = y $
स्थिर कार्य लें $ g_{x,x} = f(x)$).
हल करना $ x $: प्रत्येक के लिए $ y $, द \omterm{def:b3:topology:topology}{खुला सेट} $ V_y = \{g_{x,y} < f +
\varepsilon\}$ रोकना $ y $; सघनता अर्क $ y_1, \dots,
y_m $ साथ $ K = \bigcup V_{y_j}$, और $ h_x = \min_j g_{x, y_j}
\in \bar{\mathcal A}$ (चरण 1) संतुष्ट करता है $ h_x < f +
\varepsilon $ हर जगह, $ h_x(x) = f(x)$. अब बदलो $ x $: $ W_x =
\{h_x > f - \varepsilon\}$ खुला है, सम्मिलित है $ x $; निकालना $ x_1,
\dots, x_l $ कवर $ K $, और $ h = \max_i h_{x_i} \in
\bar{\mathcal A}$ संतुष्ट $ f - \varepsilon < h < f +
\varepsilon $: $\norm{f - h}_\infty \leq \varepsilon $. इस तरह
$ f \in \bar{\mathcal A}$.
\end{proof}

\begin{corollary}\label{cor:b3:complete:weierstrass}
(ए) (\emph{विअरस्ट्रास}) बहुपद सघन होते हैं $\mathcal
C(\intcc ab, \R)$; बहुपद में $ n $ चर सघन हैं
$\mathcal C(K, \R)$ के लिए कॉम्पैक्ट $ K \subseteq \R^n $.
(बी) (\emph{जटिल संस्करण}) अगर $\mathcal A \subseteq \mathcal
C(K, \C)$ एक उपबीजगणित है जिसमें स्थिरांकों को अलग किया जाता है
अंक, और \emph{संयुग्मन के तहत स्थिर}, यह सघन है.
(सी) (\emph{त्रिकोणमितीय संस्करण}) त्रिकोणमितीय बहुपद
$\sum_{\abs n \leq N}c_n\eu^{\iu n t}$ के स्थान में सघन हैं
\omterm{def:b3:topology:continuity}{निरंतर} $2\pi $-आवधिक कार्यों के साथ
$\norm\cdot_\infty $.\index{वीयरस्ट्रैस सन्निकटन प्रमेय}
\end{corollary}

\begin{proof}
(ए) बहुपद अचरों के साथ बीजगणित बनाते हैं; समन्वय
के अलग-अलग बिंदु कार्य करता है $\R^n $.
(बी) वास्तविक और काल्पनिक भाग $\frac{f + \bar f}2$,
$\frac{f - \bar f}{2\iu}$ के सदस्यों का $\mathcal A $ फॉर्म ए
वास्तविक बीजगणित $\mathcal A_\R \subseteq \mathcal C(K, \R)$ के साथ
स्थिरांक; यह बिंदुओं को अलग करता है ($ f(x) \ne f(y)$ बल
$\operatorname{Re}f $ या $\operatorname{Im}f $ अलग अलग करना)।
वास्तविक प्रमेय लागू करें और पुनः संयोजित करें।
(सी) देखें $2\pi $-आवधिक कार्यों के रूप में $\mathcal C(S^1, \C)$
($ S^1 = \R/2\pi\Z $, कॉम्पैक्ट:
\cref{exo:b3:topology:5}); द्वारा उत्पन्न बीजगणित
$\eu^{\iu t}, \eu^{-\iu t}$ और स्थिरांक के अंतर्गत स्थिर है

इस पर इंजेक्शन है)। लागू करें (बी)।
\end{proof}

\section{समस्या: पीनो का अस्तित्व प्रमेय}

\begin{problem}[{सप्ताहांत समस्या---के समाधान का अस्तित्व
$ x' = f(t, x)$ लिप्सचिट्ज़ के बिना}]\label{pb:b3:complete:1}
कॉची--लिप्सचिट्ज़ (वर्ष 2; पुनः सिद्ध किया गया \cref{ch:b3:ode})
मांग $ f $ लिप्सचिट्ज़ में $ x $. पीनो (1890): \emph{निरंतरता का
$ f $ पहले से ही अस्तित्व उत्पन्न करता है} --- हालांकि विशिष्टता नहीं. हम
इसे यूलर बहुभुज और एस्कोली से सिद्ध करें। सेटिंग: $ f \colon R
\to \R^d $ \omterm{def:b3:topology:continuity}{निरंतर} आयत पर $ R = \intcc{t_0 - a}{t_0
+ a}\times \bar B(x_0, b) \subseteq \R\times\R^d $, $ M =
\sup_R\norm f $, और
\[
T = \min\bigl(a,\ b/M\bigr)
\qquad (\text{if } M = 0 \text{ the problem is trivial}).
\]

\medskip
\noindent\textbf{Part I --- Euler polygons.} के लिए $ n \geq 1$,
उपविभाजन $[t_0, t_0 + T]$ द्वारा $ t_k = t_0 + kT/n $ ($0 \leq k
\leq n $) और परिभाषित करें $\varphi_n $ टुकड़ों में आत्मीयता से:
$\varphi_n(t_0) = x_0$ और पर $[t_k, t_{k+1}]$,
\[
\varphi_n(t) = \varphi_n(t_k) + (t -
t_k)\,f\bigl(t_k, \varphi_n(t_k)\bigr).
\]
\begin{enumerate}
  \item प्रेरण द्वारा दिखाएँ कि $\varphi_n $ अच्छी तरह से परिभाषित है,
        के साथ $\norm{\varphi_n(t) - x_0} \leq M(t - t_0) \leq b $
        पर $[t_0, t_0 + T]$ ---तो मूल्यांकन अंक बने रहेंगे
        में $ R $. \emph{(यहीं पर $ T \leq b/M $ प्रवेश करता है।)}
  \item प्रत्येक को दिखाओ $\varphi_n $ है $ M $-लिप्सचिट्ज़.
  \item अर्ज़ेल से निष्कर्ष निकालें\`ए--एस्कोली
        (\cref{thm:b3:complete:ascoli}) वह एक परिणाम है
        $(\varphi_{n_j})$ पर समान रूप से अभिसरित होता है $[t_0, t_0 +
        T]$ कुछ करने के लिए $\varphi $, स्वयं $ M $-लिप्सचिट्ज़ के साथ
        $\varphi(t_0) = x_0$.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The limit solves the equation.}
को परिभाषित करो \emph{दोष} $\Delta_n(t) = \varphi_n'(t) -
f\bigl(t, \varphi_n(t)\bigr)$ गैर-ग्रिड बिंदुओं पर.
\begin{enumerate}[resume]
  \item बताते हैं कि $ f $ समान रूप से है \omterm{def:b3:topology:continuity}{निरंतर} पर $ R $, और
        निष्कर्ष: प्रत्येक के लिए $\varepsilon > 0$ वहाँ है $ n_0$ ऐसे
        उस के लिए $ n \geq n_0$ और हर गैर-ग्रिड $ t $,
        $\norm{\Delta_n(t)} \leq \varepsilon $. \emph{(पर $(t_k,
        t_{k+1})$, $\varphi_n'(t) = f(t_k, \varphi_n(t_k))$,
        और $(t, \varphi_n(t))$ दूरी के भीतर है
        $(1 + M)\,T/n $ का $(t_k, \varphi_n(t_k))$.)}
  \item समग्र स्वरूप स्थापित करें: सबके लिए $ t $,
        \[
        \varphi_n(t) = x_0 + \int_{t_0}^{t}
        f\bigl(s, \varphi_n(s)\bigr)\dd s +
        \int_{t_0}^t \Delta_n(s)\,\dd s,
        \]
        मध्य इंटीग्रैंड टुकड़े-टुकड़े किया जा रहा है \omterm{def:b3:topology:continuity}{निरंतर}.
  \item साथ सीमा तक गुजरें $(n_j)$: दिखाओ
        $ f(s, \varphi_{n_j}(s)) \to f(s, \varphi(s))$
        \emph{समान रूप से} (वर्दी निरंतरता का $ f $ फिर से), और
        निष्कर्ष
        \[
        \varphi(t) = x_0 + \int_{t_0}^t
        f\bigl(s, \varphi(s)\bigr)\dd s .
        \]
  \item उसका निष्कर्ष निकालो $\varphi $ है $\mathcal C^1$ पर $[t_0, t_0 +
        T]$ और हल करता है $ x' = f(t, x)$, $ x(t_0) = x_0$; विस्तार
        करने के लिए निर्माण $[t_0 - T, t_0]$ (समय का उलटफेर)।
        \emph{यह पीनो का प्रमेय है.}
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Uniqueness genuinely fails.}
विचार करना $ x' = 2\sqrt{\abs x}$, $ x(0) = 0$, पर $\R $.
\begin{enumerate}[resume]
  \item जाँच करें कि $ f(x) = 2\sqrt{\abs x}$ है \omterm{def:b3:topology:continuity}{निरंतर} लेकिन
        किसी पर लिप्सचिट्ज़ नहीं \omterm{def:b3:topology:topology}{अड़ोस-पड़ोस} का $0$.
  \item सत्यापित करें कि $ x \equiv 0$ और, प्रत्येक के लिए $ c \geq 0$,
        \[
        x_c(t) = \begin{cases} 0 & t \leq c,\\ (t - c)^2 & t
        \geq c,\end{cases}
        \]
        के माध्यम से सभी समाधान हैं $(0,0)$: का एक सातत्य
        विशिष्ट समाधान.
  \item पिकार्ड पुनरावृत्ति तर्क (बानाच निश्चित) कहां है
        बिंदु) इसके लिए टूट जाओ $ f $?
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Limits of the method.}
\begin{enumerate}[resume]
  \item दिखाएँ कि पीनो का प्रमेय अनंत आयाम में विफल रहता है:
        हम शास्त्रीय को स्वीकार करते हैं (या आप विश्वास पर ले सकते हैं)।
        डाइडॉन का उदाहरण\'ई अंतरिक्ष में $ c_0$ शून्य का
        अनुक्रम; इसके बजाय, परिमित-आयामी सिद्ध करें
        घटक जो वहां विफल रहता है: \omterm{def:b3:topology:interior}{बंद} इकाई गेंद
        $ c_0$ (सुपर नॉर्म) नहीं है कॉम्पैक्ट --- एक सीमाबद्ध प्रदर्शन
        अभिसरण अनुवर्ती के बिना अनुक्रम, और व्याख्या करें
        भाग I का कौन सा चरण टूटता है.
  \item संक्षेप में बताएं: कौन सी परिकल्पनाएं अस्तित्व प्रदान करती हैं? अस्तित्व
        और विशिष्टता? दो प्रमेयों को सटीक रूप से बताएं
        (पीनो; कॉची--लिप्सचिट्ज़) अगल-बगल।
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Osgood: uniqueness beyond
Lipschitz.} होने देना $\omega\colon\intoo0\infty\to\intoo0\infty $ होना
\omterm{def:b3:topology:continuity}{निरंतर}, न घटने वाला, साथ
\[
\int_0^1\frac{\dd r}{\omega(r)} = +\infty
\qquad(\text{divergence at } 0),
\]
और मान लीजिए $ f $ संतुष्ट $\norm{f(t, x) - f(t, y)} \leq
\omega\bigl(\norm{x - y}\bigr)$ पर $ R $.
\begin{enumerate}[resume]
  \item जाँच करें कि $\omega(r) = Lr $ अर्हता प्राप्त करता है (लिप्सचिट्ज़), वह
        $\omega(r) = r\log\frac1r $ (द्वारा बढ़ाया गया निरंतरता, के लिए
        $ r $ छोटा) यद्यपि योग्य है \emph{नहीं} $ O(r)$,
        ओर वो $\omega(r) = 2\sqrt r $ नहीं करता. की गणना करें
        प्रत्येक मामले में अभिन्न.
  \item होने देना $ x_1, x_2$ पर समीकरण हल करें $[t_0, t_0 + T]$
        साथ $ x_1(t_0) = x_2(t_0)$, और $\delta(t) =
        \norm{x_1(t) - x_2(t)}$. दिखाओ, अभिन्न रूपों से
        अकेले, उस के लिए $ t_0 \leq s \leq t $:
        \[
        \delta(t) \leq \delta(s) +
        \int_s^t\omega\bigl(\delta(v)\bigr)\dd v .
        \]
  \item (ओस्गुड का प्रमेय) मान लीजिए $\delta(t_1) > 0$ कुछ के लिए
        $ t_1$, और चलो $\tau = \sup\{t \leq t_1 : \delta(t) =
        0\}$. के लिए $ s \in \intoo\tau{t_1}$, तय करना $ u(t) = \delta(s)
        + \int_s^t\omega(\delta(v))\,\dd v $. दिखाओ $\delta \leq
        u $, $ u' = \omega(\delta) \leq \omega(u)$, और निष्कर्ष निकालें
        \[
        \int_{\delta(s)}^{u(t_1)}\frac{\dd r}{\omega(r)}
        \leq t_1 - s \leq t_1 - \tau .
        \]
        होने देना $ s \downarrow \tau $ और इसके साथ विरोधाभास उत्पन्न करें
        अभिन्न का विचलन. निष्कर्ष:
        \emph{एक \omterm{def:b3:groups:normal}{सामान्य} प्रारंभिक स्थिति के माध्यम से समाधान
        संयोग} --- ऑसगूड की स्थिति के तहत विशिष्टता।
  \item परिणाम निकालें: कॉची--लिप्सचिट्ज़ विशिष्टता है
        मामला $\omega(r) = Lr $; समीकरण $ x' =
        x\log\frac1{\abs x}$ (द्वारा बढ़ाया गया $0$ पर $0$) अद्वितीय है
        समाधान हालांकि इसका दाहिना भाग लिप्सचिट्ज़ नहीं है
        $0$; और के लिए $ x' = 2\sqrt{\abs x}$ का अभिसरण
        $\int_0\frac{\dd r}{2\sqrt r}$ बिल्कुल वही है जो ए को देता है
        समाधान छोड़ें $0$ परिमित समय में --- के मान से मेल खाता है
        अभिन्न $\int_0^{h^2}\frac{\dd r}{2\sqrt r} = h $
        के भागने के व्यवहार के साथ $ x_c $.
\end{enumerate}

\medskip
\noindent\textbf{Part VI --- Rates, schemes, funnels.}
\begin{enumerate}[resume]
  \item (इंटीग्रल जीआर\"ऑनवॉल लेम्मा) चलो $ e, \eta \geq 0$ होना
        \omterm{def:b3:topology:continuity}{निरंतर} पर $[t_0, t_0+T]$ साथ $ e(t) \leq \int_{t_0}
        ^t\bigl(L\,e(s) + \eta(s)\bigr)\dd s $ सभी के लिए $ t $. दिखाओ
        \[
        e(t) \leq \frac{\sup\eta}{L}\bigl(\eu^{L(t - t_0)} -
        1\bigr)
        \]
        \emph{(तय करना $ G(t) = \int_{t_0}^t(Le + \eta)$, टिप्पणी $ G'
        \leq LG + \sup\eta $, और अंतर करें
        $\eu^{-Lt}G(t)$)}.
  \item (यूलर एक दर के साथ अभिसरण करता है) अब मान लीजिए $ f $ है
        $ L $-लिप्सचिट्ज़ में $ x $ और $ L'$-लिप्सचिट्ज़ में $ t $ पर $ R $.
        प्रश्न 4 के दोष परिच्छेद का संयोजन (बना हुआ)
        मात्रात्मक: $\norm{\Delta_n} \leq (L' + LM)T/n $) के साथ
        प्रश्न 17, सिद्ध करें
        \[
        \sup_{[t_0, t_0+T]}\norm{\varphi_n - \varphi}
        \;\leq\; \frac{(L' + LM)\,T}{n}\cdot
        \frac{\eu^{LT} - 1}{L} = O\Bigl(\frac1n\Bigr),
        \]
        कहाँ $\varphi $ है \emph{} समाधान: लिप्सचिट्ज़ के साथ
        संपूर्ण अनुक्रम डेटा एक स्पष्ट के साथ अभिसरण करता है
        दर --- किसी अनुवर्ती की आवश्यकता नहीं है। विशिष्टता क्यों होती है
        अनुवर्ती अभिसरण को पूर्ण अभिसरण में अपग्रेड करें
        इस गणना के बिना भी?
  \item (योजना चुनती है) के लिए $ x' = 2\sqrt{\abs x}$, $ x(0) =
        0$: दिखाएँ कि प्रत्येक यूलर बहुभुज समान रूप से शून्य है,
        इसलिए योजना समाधान तक पहुंचती है $ x \equiv 0$;
        लेकिन शुरू हुआ $ x(0) = \varepsilon > 0$ यह एकत्रित हो जाता है
        (जैसे $ n \to \infty $, तब $\varepsilon \to 0$) को $ t
        \mapsto t^2$, ए \emph{अलग} के माध्यम से समाधान
        उत्पत्ति. गैर-विशिष्टता की संवेदनशीलता के रूप में पुनः उभर आती है
        गड़बड़ी के लिए संख्यात्मक योजना.
  \item दिखाओ कि सेट $\mathcal S $ का \emph{सभी} समाधान
        का $ x' = f(t,x)$, $ x(t_0) = x_0$ पर $[t_0, t_0+T]$
        (में मान $\bar B(x_0, b)$) गैर-रिक्त है (भाग II),
        समान रूप से $ M $-लिप्सचिट्ज़, और अंदर \omterm{def:b3:topology:interior}{बंद} हो गया
        $\bigl(\mathcal C([t_0, t_0+T], \R^d),
        \norm\cdot_\infty\bigr)$; एस्कोली ने निष्कर्ष निकाला कि
        $\mathcal S $ है \emph{कॉम्पैक्ट}. (कनेसेर का प्रमेय जोड़ता है
        वह $\mathcal S $ है \omterm{def:b3:topology:connected}{जुड़े हुए}; हम साबित नहीं करेंगे
        यह.)
  \item उदाहरण पर केन्सर की घटना को सत्यापित करें: के लिए $ x' =
        2\sqrt{\abs x}$, $ x(0) = 0$, पर $[0, 1]$, दिखाओ
        $\mathcal S = \{x_c : c \in \intcc0\infty\}$ साथ
        $ x_\infty \equiv 0$ \emph{(किसी भी समाधान के लिए, आइए $ c =
        \sup\{t : x(t) = 0\}$ और एकीकृत करें $(\sqrt x)' = 1$ पर
        $\{x > 0\}$)}, वह $ c \mapsto x_c $ है \omterm{def:b3:topology:continuity}{निरंतर} से
        $\intcc0\infty $ (एक-बिंदु संघनन, अर्थात\
        $ c = \infty $ सीमा के रूप में चिपकाया गया) से $\mathcal C([0,1])$,
        और निष्कर्ष निकालो $\mathcal S $ वास्तव में है कॉम्पैक्ट और
        \omterm{def:b3:topology:connected}{जुड़े हुए} --- एक खंड-आकार की फ़नल।
  \item (पहुंचने योग्य सेट) प्रत्येक के लिए प्रश्न 20 से निष्कर्ष निकालें
        निश्चित $ t $, द \emph{पहुंच योग्य सेट} $\mathcal S(t) =
        \{x(t) : x \in \mathcal S\}$ है कॉम्पैक्ट; इसके लिए गणना करें
        प्रश्न 21 का उदाहरण और इसकी जाँच भी करें
        \omterm{def:b3:topology:connected}{जुड़े हुए}: $\mathcal S(t) = \intcc0{t^2}$ --- प्रत्येक
        किसी उपाय से मध्यवर्ती अवस्था प्राप्त होती है।
\end{enumerate}

\medskip
\noindent\textbf{Part VII --- Complements: dependence,
optimality, and a scheme computed by hand.}
\begin{enumerate}[resume]
  \item (\omterm{def:b3:topology:continuity}{निरंतर} निर्भरता) मान लीजिए $ f $ है $ L $-लिप्सचिट्ज़
        में $ x $ पर $ R $, और चलो $ x, y $ के साथ दो समाधान हो
        प्रारंभिक मान $ x_0, y_0$ पर $ t_0$. प्रमाण को अपनाना
        असमानता के लिए प्रश्न 17 का $ e(t) \leq \norm{x_0
        - y_0} + \int_{t_0}^tL\,e(s)\dd s $, दिखाओ
        \[
        \norm{x(t) - y(t)} \leq \norm{x_0 - y_0}\,
        \eu^{L(t - t_0)},
        \]
        और जाँच करें $ x' = Lx $ कि सीमा प्राप्त हो गई है:
        ग्रा\"ऑनवॉल शार्प है. फिर से विशिष्टता निकालें ($ x_0 =
        y_0$), और वह प्रवाह मानचित्र $ x_0 \mapsto x(t; x_0)$
        लिप्सचिट्ज़ है, स्थिरांक के साथ $\eu^{LT}$, चाहे वह कहीं भी हो
        परिभाषित.
  \item (ओस्गुड इष्टतम है) इसके विपरीत, चलो $\omega $ होना
        \omterm{def:b3:topology:continuity}{निरंतर}, न घटने वाला, सकारात्मक पर
        $\intoo0\infty $, साथ $\omega(r) \to 0$ जैसा $ r \to 0^+$
        लेकिन
        \[
        \int_0^1\frac{\dd r}{\omega(r)} < +\infty,
        \]
        और विस्तार करें $ f(x) = \omega(\abs x)$ द्वारा $ f(0) = 0$.
        बताते हैं कि $\Omega(x) = \int_0^x\frac{\dd r}{\omega(r)}$
        की ओर से बढ़ती हुई आपत्ति है $\intoc0{1}$ पर
        $\intoc0{\Omega(1)}$, कि यह उलटा है $ g $ हल करती है $ g'
        = \omega(g)$ साथ $ g(0^+) = 0$, ओर वो $ g $, विस्तारित
        द्वारा $0$ के लिए $ t \leq 0$, एक है $\mathcal C^1$ का समाधान
        $ x' = f(x)$ के माध्यम से $(0, 0)$ से अलग $ x \equiv
        0$ \emph{(के लिए $ g'(0) = 0$, अवश्यंभावी $\frac{g(t)}t $ द्वारा
        $\omega(g(t))$)}. निष्कर्ष: विचलन परिकल्पना
        प्रश्न 15 का उत्तर सुविधा नहीं बल्कि सटीक है
        विशिष्टता की सीमा; से भाग III पुनर्प्राप्त करें
        $\omega(r) = 2\sqrt r $, $\Omega(x) = \sqrt x $.
  \item (यूलर ने हाथ से गणना की) के लिए $ x' = x $, $ x(0) = 1$ पर
        $[0, 1]$: दिखाएँ कि यूलर बहुभुज संतुष्ट करता है
        $\varphi_n(1) = \bigl(1 + \frac1n\bigr)^n $. सिद्ध करो
        विस्तार
        \[
        \Bigl(1 + \frac1n\Bigr)^{n}
        = \eu\Bigl(1 - \frac1{2n} + O\Bigl(\frac1{n^2}\Bigr)
        \Bigr),
        \]
        तो त्रुटि पर $ t = 1$ है $\frac{\eu}{2n} +
        O(n^{-2})$: प्रश्न 18 $ O(\frac1n)$ दर, के साथ
        सटीक स्थिरांक. संख्यात्मक रूप से जाँच करें $ n = 10$:
        $1.1^{10} = 2.59374$ ख़िलाफ़ $\eu \approx 2.71828$,
        एक गलती $0.12454$ तुलना करने के लिए $\frac{\eu}{20}
        \approx 0.13591$.
\end{enumerate}
\end{problem}
