\chapter{कॉम्पैक्ट ऑपरेटर्स और स्पेक्ट्रल प्रमेय}
\label{ch:b3:spectral}

विकर्णीकरण परिमित-आयामी रैखिक बीजगणित का मुकुट है
गहना: एक सममित मैट्रिक्स का एक लंबात्मक \omterm{def:b3:topology:basis}{आधार} होता है
eigenvectors. अनंत आयाम में यह सीमाबद्ध के लिए विफल रहता है
स्व-सहायक संचालक \omterm{def:b3:groups:normal}{सामान्य} तौर पर --- से गुणा $x $ पर
$ L^2(\intcc01)$ है \emph{नहीं} बिल्कुल eigenvalues
(\cref{exo:b3:spectral:6}) --- लेकिन यह जीवित रहता है, में \omterm{prop:b3:galois:perfect}{उत्तम}
फॉर्म, उन ऑपरेटरों के लिए जो हैं \emph{लगभग सीमित
आयामी}: द \omterm{def:b3:spectral:compact}{कॉम्पैक्ट} वाले. के लिए वर्णक्रमीय प्रमेय
\omterm{def:b3:spectral:compact}{कॉम्पैक्ट} स्व-सहायक संचालक सबसे अधिक उपयोग किया जाने वाला प्रमेय है
अनुप्रयुक्त कार्यात्मक विश्लेषण का: यह अभिन्न को विकर्णित करता है
समीकरण, चलाता है \omterm{thm:b3:spectral:fredholm}{फ्रेडहोम विकल्प}, और (सप्ताहांत
समस्या) साइन \omterm{def:b3:topology:basis}{आधार} का निर्माण करते हुए कंपन स्ट्रिंग को हल करती है
शुद्ध ऑपरेटर सिद्धांत से फूरियर विश्लेषण का --- यूलर के साथ
$\zeta(2) = \frac{\pi^2}6$ एक ट्रेस सूत्र से बाहर गिर रहा है
बिदाई उपहार. लगातार, $ H $ एक है \omterm{def:b3:hilbert:inner}{हिल्बर्ट स्थान} ऊपर $\C $ (या
$\R $; कथन अनुकूल होते हैं), और ऑपरेटर सीमित होते हैं।

\section{कॉम्पैक्ट ऑपरेटर्स}

\begin{definition}\label{def:b3:spectral:compact}
$ T \in \mathcal L(E, F)$ ($ E, F $ बनच) है
\emph{कॉम्पैक्ट}\index{कॉम्पैक्ट} यदि छवि $ T(B)$ का
यूनिट बॉल अपेक्षाकृत कॉम्पैक्ट होती है $ F $ --- समान रूप से,
प्रत्येक बंधा हुआ क्रम $(x_n)$ के साथ एक अनुवर्ती है
$(Tx_{n_k})$ अभिसारी। परिमित-रैंक ऑपरेटर कॉम्पैक्ट होते हैं
(परिमित आयाम में बंधे सेट); एक की पहचान
अनंत-आयामी अंतरिक्ष कभी नहीं होता (रिस्ज़ का प्रमेय, वर्ष 2)।
\end{definition}

\begin{proposition}\label{prop:b3:spectral:ideal}
 \omterm{def:b3:spectral:compact}{कॉम्पैक्ट} ऑपरेटर $\mathcal K(E, F)$ एक \omterm{def:b3:topology:interior}{बंद} उपस्थान बनाएँ
का $\mathcal L(E, F)$, और एक दोतरफा \omterm{def:b3:rings:ideal}{आदर्श}: $ S $ \omterm{def:b3:spectral:compact}{कॉम्पैक्ट}
$\Rightarrow $ $ AS $ और $ SB $ \omterm{def:b3:spectral:compact}{कॉम्पैक्ट} बाध्य के लिए $ A, B $.
इसके अलावा, ए में \omterm{def:b3:hilbert:inner}{हिल्बर्ट स्थान}, प्रत्येक \omterm{def:b3:spectral:compact}{कॉम्पैक्ट} ऑपरेटर एक है
परिमित-रैंक ऑपरेटरों की मानक-सीमा।
\end{proposition}

\begin{proof}
उपस्थान: अनुक्रम लक्षण वर्णन से स्पष्ट। \omterm{def:b3:rings:ideal}{आदर्श}:
बंधे हुए मानचित्र अभिसारी अनुक्रमों को अभिसारी अनुक्रम भेजते हैं और
बंधे से बंधे. बंदपन: चलो $ T_n \to T $ साथ $ T_n $
\omterm{def:b3:spectral:compact}{कॉम्पैक्ट}, और $(x_k)$ से घिरा $1$; एक विकर्ण निष्कर्षण
बनाता है $(T_nx_{k_j})_j $ प्रत्येक के लिए अभिसरण $ n $; तब
$(Tx_{k_j})$ कॉची है, तब से
\[
\norm{Tx_{k_j} - Tx_{k_l}} \leq 2\vertiii{T - T_n} +
\norm{T_nx_{k_j} - T_nx_{k_l}} ,
\]
पहले चुनना $ n $ फिर सूचकांक. में सन्निकटन हिल्बर्ट
रिक्त स्थान: होने देना $ T $ होना \omterm{def:b3:spectral:compact}{कॉम्पैक्ट}, $ K = \overline{T(B)}$ \omterm{def:b3:spectral:compact}{कॉम्पैक्ट};
दिया गया $\varepsilon $, ढकना $ K $ बहुत सी गेंदों द्वारा
$ B(y_i, \varepsilon)$ और चलो $ P $ ऑर्थोगोनल प्रक्षेपण हो
पर $ V = \operatorname{Vect}(y_1, \dots, y_m)$ (\omterm{def:b3:topology:interior}{बंद}:
परिमित-आयामी)। तब $ PT $ परिमित रैंक है, और के लिए
$\norm x \leq 1$: चुनना $ y_i $ साथ $\norm{Tx - y_i} <
\varepsilon $,
\[
\norm{Tx - PTx} \leq \norm{Tx - y_i} + \norm{P(y_i - Tx)}
\leq 2\varepsilon
\]
($ y_i = Py_i $; $\vertiii P \leq 1$): $\vertiii{T - PT} \leq
2\varepsilon $.
\end{proof}

\begin{example}\label{ex:b3:spectral:examples}
(ए) विकर्ण ऑपरेटरों पर $\ell^2$: $ T(x_n) = (\lambda_nx_n)$
है \omterm{def:b3:spectral:compact}{कॉम्पैक्ट} आईएफएफ $\lambda_n \to 0$
(\cref{exo:b3:spectral:2}).
(बी) कर्नेल ऑपरेटर चालू $\mathcal C(\intcc01)$: \omterm{def:b3:spectral:compact}{कॉम्पैक्ट} द्वारा
एस्कोली (\cref{exo:b3:complete:7}).
(सी) \emph{हिल्बर्ट--श्मिट संचालक}\index{हिल्बर्ट--श्मिट संचालक}: के लिए $ k \in L^2(\intcc01^2)$,
\[
(T_kf)(x) = \int_0^1k(x, y)\,f(y)\,\dd y
\]
ए को परिभाषित करता है \omterm{def:b3:spectral:compact}{कॉम्पैक्ट} ऑपरेटर पर $ L^2(\intcc01)$ साथ
$\vertiii{T_k} \leq \norm k_{L^2}$
(\cref{exo:b3:spectral:4}: के \omterm{def:b3:topology:basis}{आधार} विस्तार को छोटा करना
$ k $ प्रदर्श $ T_k $ परिमित-रैंक ऑपरेटरों की एक सीमा के रूप में)।
\end{example}

\section{स्व-सहायक संचालक}

\begin{definition}\label{def:b3:spectral:selfadjoint}
$ T \in \mathcal L(H)$ है \emph{स्वयं सहायक}\index{स्वयं सहायक} अगर $ T = T^*$ (\cref{exo:b3:hilbert:8}), अर्थात\
$\langle Tx, y\rangle = \langle x, Ty\rangle $ सभी के लिए $ x, y $.
तब $\langle x, Tx\rangle \in \R $ हरएक के लिए $ x $ (इसके बराबर
संयुग्मित)।
\end{definition}

\begin{proposition}\label{prop:b3:spectral:sanorm}
के लिए \omterm{def:b3:spectral:selfadjoint}{स्वयं adjoint} $ T $:\index{संख्यात्मक त्रिज्या}
\[
\vertiii T = \sup_{\norm x \leq 1}\ \abs{\langle x,
Tx\rangle} .
\]
के eigenvalues $ T $ वास्तविक हैं, और विशिष्ट के लिए eigenvectors हैं
eigenvalues ऑर्थोगोनल हैं।
\end{proposition}

\begin{proof}
होने देना $ M $ सर्वोच्च बनें; $ M \leq \vertiii T $ द्वारा
कॉची-श्वार्ज़। इसके विपरीत, ध्रुवीकरण-प्रकार की पहचान
\[
\langle x{+}y, T(x{+}y)\rangle - \langle x{-}y,
T(x{-}y)\rangle = 4\operatorname{Re}\langle y, Tx\rangle
\]
(विस्तृत करें; क्रॉस शर्तें $\langle y, Tx\rangle + \langle x,
Ty\rangle = 2\operatorname{Re}\langle y, Tx\rangle $ द्वारा
\omterm{def:b3:spectral:selfadjoint}{आत्म adjointness}) समांतर चतुर्भुज नियम के साथ देता है,
\[
4\operatorname{Re}\langle y, Tx\rangle \leq
M\bigl(\norm{x{+}y}^2 + \norm{x{-}y}^2\bigr) = 2M\bigl(\norm
x^2 + \norm y^2\bigr).
\]
के लिए $\norm x = 1$ साथ $ Tx \neq 0$, लेना $ y = Tx/\norm{Tx}$:
$4\norm{Tx} \leq 4M $. इसलिए $\vertiii T \leq M $. स्वदेशी मान:
$ Tx = \lambda x $, $ x \ne 0$ देता है $\lambda\norm x^2 = \langle
x, Tx\rangle \in \R $. रूढ़िवादिता: $\lambda\langle x,
y\rangle = \langle Tx, y\rangle = \langle x, Ty\rangle =
\mu\langle x, y\rangle $ साथ $\lambda \neq \mu $ असली।
\end{proof}

\section{वर्णक्रमीय प्रमेय}

\begin{lemma}[अत्यधिक स्वदेशी मूल्य का अस्तित्व]
\label{lem:b3:spectral:existence}
होने देना $ T \neq 0$ होना \omterm{def:b3:spectral:compact}{कॉम्पैक्ट} और \omterm{def:b3:spectral:selfadjoint}{स्वयं adjoint}. तब $\vertiii T $
या $-\vertiii T $ का एक प्रतिरूप है $ T $.
\end{lemma}

\begin{proof}
द्वारा \cref{prop:b3:spectral:sanorm}, यूनिट वैक्टर चुनें $ x_n $
साथ $\langle x_n, Tx_n\rangle \to \mu $, कहाँ $\abs\mu =
\vertiii T > 0$ (चिह्न को ठीक करने के लिए बाद में पास करें)। तब
\[
\norm{Tx_n - \mu x_n}^2
= \norm{Tx_n}^2 - 2\mu\langle x_n, Tx_n\rangle + \mu^2
\leq \vertiii T^2 - 2\mu\langle x_n, Tx_n\rangle + \mu^2
\longrightarrow 2\mu^2 - 2\mu\cdot\mu = 0 .
\]
सघनता से, एक अनुवर्ती $ Tx_{n_k} \to y $; तब $\mu
x_{n_k} = Tx_{n_k} - (Tx_{n_k} - \mu x_{n_k}) \to y $, इसलिए
$ x_{n_k} \to x = y/\mu $, एक इकाई वेक्टर, और निरंतरता देता है
$ Tx = \mu x $.
\end{proof}

\begin{theorem}[कॉम्पैक्ट स्व-सहायक के लिए स्पेक्ट्रल प्रमेय
संचालक]\label{thm:b3:spectral:spectral}
होने देना $ T $ एक हो \omterm{def:b3:spectral:compact}{कॉम्पैक्ट} स्व-सहायक संचालिका एक पर \omterm{def:b3:hilbert:inner}{हिल्बर्ट स्थान}
$ H $.\index{वर्णक्रमीय प्रमेय (कॉम्पैक्ट स्व-सहायक)}
\begin{enumerate}
  \item $ H $ एक ऑर्थोनॉर्मल प्रणाली को स्वीकार करता है $(e_n)_{n \in N}$
        ($ N $ के eigenvectors का परिमित या गणनीय)। $ T $, के साथ
        वास्तविक अशून्य eigenvalues $(\lambda_n)$, ऐसा है कि
        \[
        Tx = \sum_{n\in N}\lambda_n\,\langle e_n, x\rangle\,
        e_n \qquad (x \in H),
        \]
        और $ H = \ker T \,\oplus^\perp\,
        \overline{\operatorname{Vect}}(e_n : n \in N)$.
  \item अगर $ N $ अनंत है, $\lambda_n \to 0$; प्रत्येक के लिए
        $\delta > 0$ केवल सीमित रूप से बहुत से $ n $ पास होना
        $\abs{\lambda_n} \geq \delta $, और प्रत्येक eigenspace
        $\ker(T - \lambda)$, $\lambda \neq 0$, है
        परिमित-आयामी.
  \item पूर्ण कर रहा है $(e_n)$ के अलंकारिक \omterm{def:b3:topology:basis}{आधार} पर $\ker T $
        पैदावार, जब $ H $ \omterm{def:b3:galois:separable}{वियोज्य} है, का एक अलौकिक \omterm{def:b3:topology:basis}{आधार}
        $ H $ eigenvectors से बना: $ T $ विकर्णित है.
\end{enumerate}
\end{theorem}

\begin{proof}
(2) प्रथम. यदि अपरिमित रूप से अनेक ऑर्थोनॉर्मल आइजेनवेक्टर हों $ x_k $
था $\abs{\lambda_{(k)}} \geq \delta $: $\norm{Tx_k - Tx_l}^2 =
\lambda_{(k)}^2 + \lambda_{(l)}^2 \geq 2\delta^2$
(ऑर्थोगोनैलिटी, पाइथागोरस): कोई अभिसरण अनुवर्ती नहीं
$(Tx_k)$, की सघनता का खंडन करता है $ T $ सीमा पर
$(x_k)$. यह प्रत्येक के लिए एक सीमित संख्या से घिरा है $\delta $,
बाहर eigenvalues की कुल बहुलता
$\intoo{-\delta}\delta $; गणनीयता और $\lambda_n \to 0$
अनुसरण करना।

(1) चलो $ H_0$ का \omterm{def:b3:topology:interior}{बंद} विस्तार हो \emph{सभी} eigenvectors
गैर-शून्य eigenvalues के साथ, व्यवस्थित ((2) और द्वारा)।
ग्राम--प्रत्येक परिमित-आयामी आइजनस्पेस के भीतर श्मिट,
ईजेनस्पेसेस में ओर्थोगोनैलिटी
\cref{prop:b3:spectral:sanorm}) एक ऑर्थोनॉर्मल प्रणाली में
$(e_n)$ eigenvalues के साथ $\lambda_n \neq 0$. $ T $ एमएपीएस $ H_0$
में $ H_0$, और भी $ H_0^\perp $ में $ H_0^\perp $: के लिए $ y
\perp H_0$ और $ e $ एक आइजनवेक्टर, $\langle e, Ty\rangle =
\langle Te, y\rangle = \lambda\langle e, y\rangle = 0$. द
प्रतिबंध $ T' = T\restriction_{H_0^\perp}$ है \omterm{def:b3:spectral:compact}{कॉम्पैक्ट}
\omterm{def:b3:spectral:selfadjoint}{स्वयं adjoint} पर \omterm{def:b3:hilbert:inner}{हिल्बर्ट स्थान} $ H_0^\perp $; अगर $ T' \neq
0$, \cref{lem:b3:spectral:existence} एक eigenvector उत्पन्न करता है
का $ T $ अंदर शून्येतर eigenvalue के साथ $ H_0^\perp $ ---
असंभव, ऐसे वैक्टर रहते हैं $ H_0$. इसलिए $ T' = 0$:
$ H_0^\perp \subseteq \ker T $. इसके विपरीत $\ker T \perp $ प्रत्येक
$ e_n $ ($\langle e_n, z\rangle = \frac1{\lambda_n}\langle
Te_n, z\rangle = \frac1{\lambda_n}\langle e_n, Tz\rangle =
0$): $\ker T \subseteq H_0^\perp $, कहाँ से $\ker T =
H_0^\perp $ और ऑर्थोगोनल अपघटन। विस्तार:
के लिए $ x = z + \sum_nc_ne_n $ ($ z \in \ker T $, $ c_n = \langle
e_n, x\rangle $; \cref{thm:b3:hilbert:parseval}(1)पर $ H_0$),
निरंतरता का $ T $ देता है $ Tx = \sum_nc_n\lambda_ne_n $.

(3) $\ker T $, एक पृथक्करणीय स्थान का एक \omterm{def:b3:topology:interior}{बंद} उपस्थान है
\omterm{def:b3:galois:separable}{वियोज्य}: इसका एक लंबात्मक \omterm{def:b3:topology:basis}{आधार} है
(\cref{prop:b3:hilbert:gramschmidt}); संघ एक है
का ऑर्थोनॉर्मल \omterm{def:b3:topology:basis}{आधार} $ H $ (1) में अपघटन द्वारा।
\end{proof}

\begin{theorem}[फ्रेडहोम विकल्प]
\label{thm:b3:spectral:fredholm}
होने देना $ T $ होना \omterm{def:b3:spectral:compact}{कॉम्पैक्ट} \omterm{def:b3:spectral:selfadjoint}{स्वयं adjoint} और $\lambda \in \R\setminus
\{0\}$.\index{फ्रेडहोम विकल्प}
\begin{enumerate}
  \item अगर $\lambda $ तो फिर, यह एक स्वदेशी मूल्य नहीं है $ T - \lambda I $
        परिबद्ध व्युत्क्रम के साथ विशेषण है: प्रत्येक के लिए $ f $, द
        समीकरण $ Tx - \lambda x = f $ बिल्कुल एक है
        समाधान, निर्भर करता है लगातार पर $ f $.
  \item अगर $\lambda $ एक eigenvalue है, $ Tx - \lambda x = f $ है
        \omterm{def:b3:groups:derived}{व्याख्या करने योग्य} आईएफएफ $ f \perp \ker(T - \lambda I)$, और
        समाधान उस तक अद्वितीय है (परिमित-आयामी)
        गिरी.
\end{enumerate}
\end{theorem}

\begin{proof}
विघटित $ x = z + \sum c_ne_n $ और $ f = w + \sum d_ne_n $
साथ में \cref{thm:b3:spectral:spectral} ($ z, w \in \ker T $). द
समीकरण पढ़ता है
\[
-\lambda z = w, \qquad (\lambda_n - \lambda)\,c_n = d_n\
(n \in N).
\]
(1) $\lambda \notin \{\lambda_n\}\cup\{0\}$: (2) के द्वारा
वर्णक्रमीय प्रमेय, $\inf_n\abs{\lambda_n - \lambda} = \delta >
0$ (igenvalues केवल पर जमा होते हैं $0 \neq \lambda $). हल करना:
$ z = -w/\lambda $, $ c_n = d_n/(\lambda_n - \lambda)$, साथ
$\sum\abs{c_n}^2 \leq \delta^{-2}\sum\abs{d_n}^2$: एक अनोखा
समाधान के साथ $\norm x \leq C\norm f $.
(2) $\lambda = \lambda_{n}$ के लिए $ n $ एक सीमित सेट में $ F $:
गलने की योग्यता का $(\lambda_n - \lambda)c_n = d_n $ के लिए $ n \in F $
आवश्यक है $ d_n = 0$, यानी\ $ f \perp e_n $ ($ n \in F $), यानी\ $ f
\perp \ker(T - \lambda I)$; $ c_n $, $ n \in F $, तो हैं
निःशुल्क.
\end{proof}

\begin{example}\label{ex:b3:spectral:minxy}
पर $ L^2(\intcc01)$, होने देना $ Tf(x) = \int_0^1\min(x, y)f(y)\dd y $:
ए \omterm{ex:b3:spectral:examples}{हिल्बर्ट--श्मिट ऑपरेटर} वास्तविक सममित कर्नेल के साथ:
\omterm{def:b3:spectral:compact}{कॉम्पैक्ट} और \omterm{def:b3:spectral:selfadjoint}{स्वयं adjoint}. हल $ Tf = \lambda f $: द
संबंध $\bigl(Tf\bigr)(x) = \int_0^xyf(y)\dd y +
x\int_x^1f(y)\dd y $ दिखाता है $ u = Tf $ संतुष्ट $ u'' = -f $ (दो
भेदभाव, के लिए वैध \omterm{def:b3:topology:continuity}{निरंतर} $ f $, और $ Tf $ है
\omterm{def:b3:topology:continuity}{निरंतर} के लिए $ f \in L^2$: प्रभुत्व अभिसरण), साथ $ u(0)
= 0$ और $ u'(1) = 0$. तो eigenfunctions हल करते हैं $\lambda u'' =
-u $, $ u(0) = 0$, $ u'(1) = 0$:
\[
u_n(x) = \sin\Bigl(\bigl(n + \tfrac12\bigr)\pi x\Bigr),
\qquad
\lambda_n = \frac{1}{\bigl(n + \frac12\bigr)^2\pi^2}
\quad (n \geq 0),
\]
और वर्णक्रमीय प्रमेय दावा करता है --- बिना फूरियर सिद्धांत के ---
कि ये ज्याएँ एक लंबात्मक \omterm{def:b3:topology:basis}{आधार} बनाती हैं $ L^2(\intcc01)$
\omterm{def:b3:groups:normal}{सामान्य}ीकरण के बाद (कर्नेल) $ T $ है $0$: $ Tf = 0$
बल, दो भिन्नताओं द्वारा, $ f = 0$ ए.ई.) द
सप्ताहांत समस्या के लिए विचारों का एक ही चक्र चलता है
हिलती हुई डोरी और अर्क $\zeta(2)$ निशान से.
\end{example}

\begin{method}\label{met:b3:spectral:method}
एक अभिन्न या अवकल समीकरण दिया गया है: (1) इसे इस प्रकार पुनर्गठित करें
$(I - \lambda K)u = f $ या $ Ku = \lambda u $ साथ $ K $ एक
अभिन्न ऑपरेटर; (2) सत्यापित करें $ K $ \omterm{def:b3:spectral:compact}{कॉम्पैक्ट} (\omterm{ex:b3:spectral:examples}{हिल्बर्ट--श्मिट}
कर्नेल, या एस्कोली) और, यदि संभव हो, \omterm{def:b3:spectral:selfadjoint}{स्वयं adjoint} (सममित
असली कर्नेल); (3) वर्णक्रमीय प्रमेय के साथ विकर्ण बनाना या
का आह्वान करें \omterm{thm:b3:spectral:fredholm}{फ्रेडहोम विकल्प} के लिए गलने की योग्यता; (4) पढ़कर सुनाना
अस्तित्व, विशिष्टता, स्थिरता और श्रृंखला सूत्र
ईजेनबासिस में समाधान। विभेदक संचालक हैं
असीमित, लेकिन उनका \emph{प्रतिलोम} (ग्रीन ऑपरेटर्स) हैं
\omterm{def:b3:spectral:compact}{कॉम्पैक्ट}: हमेशा पहले उलटा करें.
\end{method}

\begin{exercise}[$\star $]\label{exo:b3:spectral:2}

\begin{exercise}[$\star\star $]\label{exo:b3:spectral:3}

(\cref{prop:b3:spectral:ideal}): अगर $ S $ है \omterm{def:b3:spectral:compact}{कॉम्पैक्ट} और $ A, B $

कुछ के लिए बाध्य $ S $, और $\dim H = \infty $, तब $ T $ क्या नहीं है

\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:spectral:4}
(\omterm{ex:b3:spectral:examples}{हिल्बर्ट--श्मिट}) चलो $ k \in L^2(\intcc01^2)$ और $(e_n)$ ए
\omterm{def:b3:hilbert:onb}{हिल्बर्ट आधार} का $ L^2(\intcc01)$.
(ए) वह दिखाओ $\vertiii{T_k} \leq \norm k_{L^2}$
\emph{(कॉची-श्वार्ज़ में $ y $-चर, फिर टोनेली)}.
(बी) विस्तार करें $ k(x,y) = \sum_{m,n}c_{mn}e_m(x)\overline{e_n(y)}$
में $ L^2$ वर्ग का (उचित ठहराएं कि उत्पाद एक बनाते हैं
\omterm{def:b3:hilbert:onb}{हिल्बर्ट आधार} वहाँ), और दिखाएँ कि योग को छोटा करने से लाभ मिलता है
परिमित-रैंक ऑपरेटरों का अभिसरण $ T_k $ में ऑपरेटर मानदंड:
$ T_k $ है \omterm{def:b3:spectral:compact}{कॉम्पैक्ट}.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:spectral:5}
होने देना $ T $ होना \omterm{def:b3:spectral:selfadjoint}{स्वयं adjoint} साथ $\langle x, Tx\rangle \geq 0$ सभी के लिए $ x $ (\emph{सकारात्मक} ऑपरेटर)।
(ए) दिखाएँ कि eigenvalues हैं $\geq 0$ ओर वो $\vertiii T =
\sup_{\norm x\leq1}\langle x, Tx\rangle $.
(बी) \omterm{def:b3:groups:normal}{सामान्य}ीकृत कॉची-श्वार्ज़ असमानता साबित करें
$\abs{\langle x, Ty\rangle}^2 \leq \langle x, Tx\rangle\langle
y, Ty\rangle $.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:spectral:6}
पर $ L^2(\intcc01)$, होने देना $(Mf)(x) = x\,f(x)$.
(ए) वह दिखाओ $ M $ घिरा है, \omterm{def:b3:spectral:selfadjoint}{स्वयं adjoint}, साथ $\vertiii M =
1$, लेकिन है \emph{नहीं} eigenvalues.
(बी) दिखाओ $ M $ क्या नहीं है \omterm{def:b3:spectral:compact}{कॉम्पैक्ट} \emph{(एक बंधा हुआ अनुक्रम प्रदर्शित करें
जिसकी छवि में कोई अभिसरण अनुवर्ती नहीं है, उदा.\ \omterm{def:b3:groups:normal}{सामान्य}ीकृत
निकट सिकुड़ते अंतराल के संकेतक $1$ --- या आह्वान करें
वर्णक्रमीय प्रमेय)}.
(सी)कहाँ करता है \cref{lem:b3:spectral:existence}का सबूत तोड़ना
के लिए $ M $?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:spectral:7}
(वोल्टेरा) चालू $ L^2(\intcc01)$, होने देना $ Vf(x) = \int_0^xf(y)\dd
y $.
(एक शो $ V $ है \omterm{def:b3:spectral:compact}{कॉम्पैक्ट} (\omterm{ex:b3:spectral:examples}{हिल्बर्ट--श्मिट} कर्नेल के साथ $\mathbf
1_{y < x}$) लेकिन नहीं \omterm{def:b3:spectral:selfadjoint}{स्वयं adjoint}; गणना $ V^*$.
(बी) वह दिखाओ $ V $ इसका कोई शून्येतर eigenvalue नहीं है। \emph{(से $ Vf =
\lambda f $: $ f $ एक \omterm{def:b3:topology:continuity}{निरंतर} प्रतिनिधि, तो है
$\mathcal C^1$, और हल करता है $\lambda f' = f $, $ f(0) = 0$.)}
(सी) निष्कर्ष निकालें कि अकेले कॉम्पैक्टनेस से कोई ईजेनवेक्टर नहीं मिलता है:
\omterm{def:b3:spectral:selfadjoint}{आत्म adjointness} में \cref{thm:b3:spectral:spectral} जरूरी है।
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b3:spectral:8}
(कूरेंट--फिशर) चलो $ T $ होना \omterm{def:b3:spectral:compact}{कॉम्पैक्ट}, \omterm{def:b3:spectral:selfadjoint}{स्वयं adjoint},
\emph{सकारात्मक}, eigenvalues के साथ $\mu_1 \geq \mu_2 \geq \dots
> 0$ (बहुलता, ईजेनवेक्टर द्वारा दोहराया गया $ e_1, e_2,
\dots $). दिखाओ:
\[
\mu_{k} = \max_{\substack{V \subseteq H \\ \dim V = k}}\
\min_{\substack{x \in V\\ \norm x = 1}}\ \langle x, Tx\rangle
= \min_{\substack{W \subseteq H\\ \operatorname{codim}W =
k-1}}\ \max_{\substack{x\in W\\ \norm x = 1}}\ \langle x,
Tx\rangle .
\]
\emph{(परीक्षा $ V = \operatorname{Vect}(e_1,\dots,e_k)$; के लिए
ऊपरी सीमा किसी को काटती है $ V $ साथ
$\operatorname{Vect}(e_k, e_{k+1}, \dots)$-प्रकार के स्थान:
आयाम गणना एक गैर-शून्य प्रतिच्छेदन को बाध्य करती है।)} निष्कर्ष निकालें
वह eigenvalues एकरस रूप से निर्भर करते हैं $ T $ ($ T \leq S
\Rightarrow \mu_k(T) \leq \mu_k(S)$).
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:spectral:9}
का उपयोग करते हुए \cref{thm:b3:spectral:fredholm} के लिए $ Tf(x) =
\int_0^1\min(x,y)f(y)\dd y $ (\cref{ex:b3:spectral:minxy}):
जिसके लिए $\lambda \in \R $ अभिन्न समीकरण करता है
\[
f(x) - \lambda\int_0^1\min(x,y)\,f(y)\,\dd y = g(x)
\]
एक अनोखा समाधान है $ f \in L^2$ हरएक के लिए $ g \in L^2$? क्या
असाधारण मूल्यों पर होता है?
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b3:spectral:10}
होने देना $ S $ पर बदलाव हो $\ell^2$
(\cref{exo:b3:banach:1}).
(ए) वह दिखाओ $ S $ इसका कोई eigenvalues नहीं है, जबकि प्रत्येक $\lambda $
साथ $\abs\lambda < 1$ का एक प्रतिरूप है $ S^*$ (खोजें
eigenvectors स्पष्ट रूप से, ज्यामितीय अनुक्रम)।
(बी) न तो $ S $ और न $ S^*$ है \omterm{def:b3:spectral:compact}{कॉम्पैक्ट}: के माध्यम से सत्यापित करें
\cref{exo:b3:spectral:2}-शैली परीक्षण चालू $(e_n)$.
(सी) टिप्पणी: के लिए गैर आत्म adjoint, गैर \omterm{def:b3:spectral:compact}{कॉम्पैक्ट} ऑपरेटर्स,
eigenvalue परिदृश्य खाली से पूर्ण तक कुछ भी हो सकता है
डिस्क --- जो धारणा जीवित रहती है वह है \emph{स्पेक्ट्रम},
बाद के पाठ्यक्रम में अध्ययन किया गया।
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:spectral:11}
(वर्गमूल) चलो $ T $ होना \omterm{def:b3:spectral:compact}{कॉम्पैक्ट}, \omterm{def:b3:spectral:selfadjoint}{स्वयं adjoint}, सकारात्मक
($\langle x, Tx\rangle \geq 0$) एक पर \omterm{def:b3:hilbert:inner}{हिल्बर्ट स्थान} $ H $, के साथ
वर्णक्रमीय अपघटन $ Tx = \sum_n\mu_n\langle e_n,
x\rangle e_n $ ($\mu_n > 0$).
(ए) परिभाषित करें $ Sx = \sum_n\sqrt{\mu_n}\,\langle e_n, x\rangle
e_n $; दिखाओ $ S $ है \omterm{def:b3:spectral:compact}{कॉम्पैक्ट}, \omterm{def:b3:spectral:selfadjoint}{स्वयं adjoint}, सकारात्मक, साथ $ S^2
= T $.
(बी) विशिष्टता साबित करें: कोई भी \omterm{def:b3:spectral:compact}{कॉम्पैक्ट} सकारात्मक \omterm{def:b3:spectral:selfadjoint}{स्वयं adjoint} $ R $
साथ $ R^2 = T $ के eigenspaces को सुरक्षित रखता है $ T $
\emph{($ RT = R^3 = TR $: $ R $ के साथ आवागमन करता है $ T $, इसलिए
$ R(\ker(T - \mu)) \subseteq \ker(T - \mu)$)}, और पर
$\ker(T - \mu)$, $ R $ एक सकारात्मक ऑपरेटर है जो वर्ग कर रहा है
$\mu\,\mathrm{id}$ एक परिमित-आयामी स्थान पर:
इसे वहां विकर्णित करें और निष्कर्ष निकालें $ R =
\sqrt\mu\,\mathrm{id}$ इसलिए, प्रत्येक eigenspace पर $ R = S $.
(सी) गणना करें $\sqrt G $ स्ट्रिंग ऑपरेटर के लिए $ G $ का
\cref{pb:b3:spectral:1}: किस कर्नेल में eigenvalues हैं
$\frac1{n\pi}$ साइन \omterm{def:b3:topology:basis}{आधार} पर? (अभिव्यक्त करना $\sqrt G $ के रूप में
$ L^2$-गुठली की सीमा; किसी \omterm{def:b3:topology:interior}{बंद} फॉर्म की आवश्यकता नहीं है।)
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b3:spectral:12}
(एकवचन मान अपघटन) चलो $ T \in \mathcal L(H)$ होना
\emph{\omterm{def:b3:spectral:compact}{कॉम्पैक्ट}}, आवश्यक रूप से नहीं \omterm{def:b3:spectral:selfadjoint}{स्वयं adjoint}.
(ए) वह दिखाओ $ T^*T $ है \omterm{def:b3:spectral:compact}{कॉम्पैक्ट}, \omterm{def:b3:spectral:selfadjoint}{स्वयं adjoint}, सकारात्मक; होने देना
$(e_n)$ eigenvectors का एक ऑर्थोनॉर्मल परिवार बनें
$ T^*Te_n = s_n^2e_n $, $ s_n > 0$ (द \emph{एकवचन मान}),
द्वारा पूर्ण की गयी $\ker(T^*T) = \ker T $ (इस समानता को सिद्ध करें)।
(बी) सेट करें $ f_n = \frac{Te_n}{s_n}$; दिखाओ $(f_n)$ है
ऑर्थोनॉर्मल, और स्थापित करें \emph{एसवीडी}:
\[
Tx = \sum_n s_n\,\langle e_n, x\rangle\,f_n
\qquad (x \in H),
\]
में अभिसरण के साथ $ H $.
(सी) निष्कर्ष: $\vertiii T = \max_ns_n $; $ T $ की एक मानक-सीमा है
परिमित-रैंक ऑपरेटर (पुनः सिद्ध करना)।
\cref{prop:b3:spectral:ideal}के लिए बातचीत हिल्बर्ट
रिक्त स्थान); और वोल्टेरा ऑपरेटर के लिए $ V $ का
\cref{exo:b3:spectral:7}, जिसका कोई eigenvalues नहीं है, समझाएं
एसवीडी फिर भी क्यों मौजूद है और इसकी सामग्रियां क्या हैं
(पहचानें $ V^*V $ एक स्ट्रिंग-प्रकार कर्नेल ऑपरेटर के रूप में ---
इसके eigenvalues की गणना स्पष्ट रूप से है
\cref{exo:b3:spectral:9}का क्षेत्र)।
\end{exercise}

\section{समस्या: कंपन करने वाली स्ट्रिंग और
\texorpdfstring{$\zeta(2)$}{zeta(2)}}

\begin{problem}[{सप्ताहांत समस्या --- ग्रीन का संचालक,
साइन आधार, और एक ट्रेस सूत्र}]\label{pb:b3:spectral:1}
हम कंपन स्ट्रिंग की eigenvalue समस्या को हल करते हैं
निश्चित सिरे --- $-u'' = \nu u $, $ u(0) = u(1) = 0$ --- द्वारा
ऑपरेटर सिद्धांत, बिना किसी के साइन ऑर्थोनॉर्मल \omterm{def:b3:topology:basis}{आधार} प्राप्त करें
फूरियर गणना, और मूल्यांकन $\zeta(2)$ दो की तुलना करके
के लिए अभिव्यक्तियाँ \emph{पता लगाना} ग्रीन ऑपरेटर का.
परिभाषित करें, पर $ L^2(\intcc01)$,
\[
(Gf)(x) = \int_0^1 g(x,y)\,f(y)\,\dd y,
\qquad
g(x, y) = \min(x,y)\,\bigl(1 - \max(x,y)\bigr).
\]

\medskip
\noindent\textbf{Part I --- The Green operator.}
\begin{enumerate}
  \item बताते हैं कि $ g $ है \omterm{def:b3:topology:continuity}{निरंतर}, सममित, के साथ $0 \leq g
        \leq \frac14$, ओर वो $ G $ है \omterm{def:b3:spectral:compact}{कॉम्पैक्ट} और
        \omterm{def:b3:spectral:selfadjoint}{स्वयं adjoint} (\cref{ex:b3:spectral:examples}(सी))।
  \item के लिए $ f $ \omterm{def:b3:topology:continuity}{निरंतर}, बताते हैं कि $ u = Gf $ है $\mathcal
        C^2$ साथ
        \[
        -u'' = f, \qquad u(0) = u(1) = 0
        \]
        \emph{(लिखना $ u(x) = (1-x)\int_0^xyf(y)\dd y +
        x\int_x^1(1-y)f(y)\dd y $ और दो बार अंतर करें)}.
        इसके विपरीत, यदि $ u \in \mathcal C^2$ साथ $ u(0) = u(1)
        = 0$, तब $ G(-u'') = u $: $ G $ स्ट्रिंग को उलट देता है
        ऑपरेटर.
  \item दिखाओ $\ker G = \{0\}$ \emph{(अगर $ Gf = 0$ साथ $ f \in
        L^2$: विरुद्ध परीक्षण करें \omterm{def:b3:topology:continuity}{निरंतर} $\varphi $, स्थानांतरण $ G $
        समरूपता/फ़ुबिनी द्वारा पर $\varphi $, और उपयोग करें
        मौलिक प्रमेयिका \cref{cor:b3:lp:fundlemma} --- या
        नियमित करें)}, ओर वो $ G $ एक सकारात्मक संचालिका है:
        $\langle f, Gf\rangle \geq 0$. \emph{(के लिए \omterm{def:b3:topology:continuity}{निरंतर}
        $ f $: $\langle f, Gf\rangle = \int_0^1 (u')^2$ साथ $ u
        = Gf $, भागों द्वारा; द्वारा समाप्त करें \omterm{ex:b3:lebesgue:gamma}{घनत्व}.)}
\end{enumerate}

\noindent\textbf{Part II --- Diagonalization: the sine
basis.}
\begin{enumerate}[resume]
  \item दिखाएँ कि eigenfunctions $ G $ eigenvalue के साथ
        $\lambda \ne 0$ अदिश तक के समाधान हैं
        $-\lambda u'' = u $, $ u(0) = u(1) = 0$ \emph{(ए.एन
        eigenfunction में एक है \omterm{def:b3:topology:continuity}{निरंतर} प्रतिनिधि --- $ Gf $
        है \omterm{def:b3:topology:continuity}{निरंतर} के लिए $ f \in L^2$, क्यों? --- इसलिए है
        $\mathcal C^2$ बूटस्ट्रैपिंग प्रश्न 2 द्वारा)}.
  \item सीमा मान समस्या का समाधान करें: के eigenvalues
        $ G $ हैं $\lambda_n = \frac1{n^2\pi^2}$ ($ n \geq 1$),
        \omterm{def:b3:groups:normal}{सामान्य}ीकृत eigenfunctions के साथ $ e_n(x) =
        \sqrt2\,\sin(n\pi x)$; सीधे तौर पर ऑर्थोनॉर्मलिटी की जाँच करें
        विवेक परीक्षण के रूप में एकीकरण।
  \item से निष्कर्ष निकालें
        \cref{thm:b3:spectral:spectral} और प्रश्न 3 वह
        $\bigl(\sqrt2\sin(n\pi x)\bigr)_{n\geq1}$ एक है
        ऑर्थोनॉर्मल \emph{\omterm{def:b3:topology:basis}{आधार}} का $ L^2(\intcc01)$ --- नहीं
        स्टोन--वीयरस्ट्रैस, फूरियर श्रृंखला की आवश्यकता नहीं है। बढ़ाना
        $ f(x) = x(1-x)$ इस \omterm{def:b3:topology:basis}{आधार} पर लिखिए \omterm{thm:b3:hilbert:parseval}{पारसेवल} के लिए
        यह.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The trace formula and
$\zeta(2)$.}
\begin{enumerate}[resume]
  \item दोनों सर्वसमिकाओं को सिद्ध कीजिए
        \[
        \langle e_n, Ge_n\rangle = \lambda_n
        \quad\text{and}\quad
        \sum_{n\geq1}\lambda_n = \int_0^1 g(x,x)\,\dd x .
        \]
        दूसरे के लिए ( \emph{ट्रेस सूत्र}): बढ़ाना
        $ g(x, \cdot)$, निश्चित के लिए $ x $, \omterm{def:b3:topology:basis}{आधार} में $(e_n)$ ---
        दिखाएँ कि गुणांक हैं $\lambda_ne_n(x)$, ताकि $ g(x, \cdot) = \sum_n\lambda_ne_n(x)\,e_n $ में
        $ L^2$. यहां साइन स्पष्ट हैं: सीधे सत्यापित करें
        वह $\sum_n\lambda_ne_n(x)e_n(y)$ अभिसरण
        \emph{समान रूप से} चौक पर (तुलना करें $\sum
        \frac2{n^2\pi^2}$), तो इसका योग है \omterm{def:b3:topology:continuity}{निरंतर} और, होना
        वही $ L^2$-विस्तार में $ y $ प्रत्येक के लिए $ x $, बराबर है
        $ g(x,y)$ हर जगह. तय करना $ y = x $ और टर्म को एकीकृत करें
        अवधि.
  \item गणना करना $\int_0^1g(x,x)\dd x = \int_0^1x(1-x)\dd x =
        \frac16$, और निष्कर्ष निकालें
        \[
        \sum_{n\geq1}\frac{1}{n^2\pi^2} = \frac16,
        \qquad\text{i.e.}\qquad
        \boxed{\ \zeta(2) = \frac{\pi^2}{6}\ } :
        \]
        एक ऑपरेटर ट्रेस से यूलर का योग।
  \item पुनः प्राप्त करें $\zeta(2)$ तीसरा तरीका: आवेदन करें \omterm{thm:b3:hilbert:parseval}{पारसेवल} में
        स्थिर कार्य के लिए साइन \omterm{def:b3:topology:basis}{आधार} $\mathbf 1$,
        गणना करें $\sum_{n \text{ odd}}\frac1{n^2}$, और
        निष्कर्ष. फिर तंत्र की तुलना करें: किस अर्थ में है
        प्रश्न 7-8 `` का ट्रेस तर्कपारसेवल
        एक ही बार में पूरे कर्नेल पर लागू किया गया''?
\end{enumerate}

\noindent\textbf{Part IV --- The string vibrates.}
\begin{enumerate}[resume]
  \item (चरों का पृथक्करण, संश्लेषित) के लिए $ f \in
        L^2$, परिभाषित करना
        \[
        u(t, x) = \sum_{n\geq1}\;c_n\,
        \cos(n\pi t)\,\sqrt2\sin(n\pi x),
        \qquad c_n = \langle e_n, f\rangle .
        \]
        दिखाएँ कि श्रृंखला में अभिसरण होता है $ L^2(\intcc01)$ प्रत्येक के लिए
        $ t $, वह $ t\mapsto u(t, \cdot)$ है \omterm{def:b3:topology:continuity}{निरंतर} में
        $ L^2$, और वह के लिए $ f $ परिमित अनेक के अंतराल में
        $ e_n $ यह तरंग समीकरण को हल करता है $\partial_t^2u =
        \partial_x^2u $ साथ $ u(0) = f $, $\partial_tu(0) = 0$,
        निश्चित सिरे. स्वदेशी मूल्य $ n^2\pi^2$ चुकता हैं
        आवृत्तियाँ: स्ट्रिंग के हार्मोनिक्स --- समझाएं
        की संगीतमय व्याख्या
        \cref{thm:b3:spectral:spectral} एक पैराग्राफ में.
\end{enumerate}

\noindent\textbf{Part V --- Variational dividends: the power
method, Weyl stability, and a rigorous bound on $\pi $.} होने देना
$ A $ एक हो \omterm{def:b3:spectral:compact}{कॉम्पैक्ट} \omterm{def:b3:spectral:selfadjoint}{स्वयं adjoint} \emph{सकारात्मक} ऑपरेटर के साथ
eigenvalues $\mu_1 \geq \mu_2 \geq \cdots > 0$ और
ऑर्थोनॉर्मल ईजेनवेक्टर $(u_n)$; $ R(x) = \frac{\langle x,
Ax\rangle}{\norm x^2}$. न्यूनतम-अधिकतम सूत्र हैं
\cref{exo:b3:spectral:8}; यहां हम उन्हें खर्च करते हैं.
\begin{enumerate}[resume]
  \item (शक्ति विधि) के लिए $ x \neq 0$ लिखना $ m_p =
        \sum_n\mu_n^p\abs{\langle u_n, x\rangle}^2$. दिखाओ
        $ m_pm_{p+2} \geq m_{p+1}^2$ (कॉची-श्वार्ज़), निष्कर्ष निकालें
        श्रृंखला
        \[
        R(x) \;\leq\; \frac{\langle Ax, Ax\rangle}{\langle x,
        Ax\rangle} \;\leq\; R(Ax) \;\leq\; \mu_1,
        \]
        और साबित करें कि अगर $\langle u_1, x\rangle \neq 0$,
        तब $ R(A^kx) \to \mu_1$: ऑपरेटर को पुनरावृत्त करना
        कोई भी \omterm{def:b3:groups:normal}{सामान्य} वेक्टर शीर्ष eigenvalue की गणना करता है ---
        संख्यात्मक विश्लेषण की शक्ति विधि प्रमाणित।
  \item (वेइल स्थिरता) के लिए \omterm{def:b3:spectral:compact}{कॉम्पैक्ट} \omterm{def:b3:spectral:selfadjoint}{स्वयं adjoint} सकारात्मक
        $ A, B $, से निष्कर्ष निकालें \cref{exo:b3:spectral:8} वह
        \[
        \abs{\mu_n(A) - \mu_n(B)} \;\leq\; \vertiii{A - B}
        \qquad\text{for every } n :
        \]
        \omterm{def:b3:complete:complete}{पूरा} स्पेक्ट्रम है $1$-लिप्सचिट्ज़ में \omterm{def:b3:banach:operator}{ऑपरेटर
        आदर्श} --- बड़े सममित प्रणालियों के eigenvalues हो सकते हैं
        गारंटीशुदा त्रुटि के साथ सन्निकटन से गणना की गई।
  \item रेले बाउंड को लागू करें $ G $ परीक्षण के साथ
        समारोह $ u(x) = x(1-x)$: हल करना $-w'' = u $, $ w(0) =
        w(1) = 0$ पाने के $ Gu = w = \frac{x(1-x)(1 + x -
        x^2)}{12}$, गणना करें
        \[
        \norm u_2^2 = \frac1{30}, \qquad
        \langle u, Gu\rangle = \frac1{12}\Bigl(\frac1{30} +
        \frac1{140}\Bigr) = \frac{17}{5040},
        \qquad R(u) = \frac{17}{168},
        \]
        और निष्कर्ष निकालो \emph{कठिन} अवश्यंभावी $\frac1{\pi^2}
        = \lambda_1 \geq \frac{17}{168}$, यानी\ $\pi \leq
        \sqrt{168/17} < 3.1437$.
  \item प्रश्न 11 की शृंखला का एक चरण, उसी परीक्षण पर
        फ़ंक्शन: उपयोग करना $\int_0^1(x - x^2)^4\dd x =
        \frac1{630}$, गणना करें
        \[
        \norm{Gu}_2^2 = \frac1{144}\Bigl(\frac1{30} +
        \frac2{140} + \frac1{630}\Bigr) = \frac{31}{90720},
        \qquad
        \frac{\langle Gu, Gu\rangle}{\langle u, Gu\rangle} =
        \frac{31}{306},
        \]
        और निष्कर्ष निकालो $\pi \leq \sqrt{306/31} < 3.1419$: दो
        अभिन्न, चार सही अंक. (प्रत्येक आगे
        पुनरावृत्ति मोटे तौर पर सटीकता को वर्गित करती है:
        ईजेनवेक्टर गैप $\lambda_1/\lambda_2 = 4$ चलाती है
        ज्यामितीय अभिसरण।)
\end{enumerate}

\noindent\textbf{Part VI --- The trace of $ G^2$, and
$\zeta(4)$.}
\begin{enumerate}[resume]
  \item बताते हैं कि $\sum_n\lambda_n^2 =
        \iint_{\intcc01^2}g(x,y)^2\,\dd x\,\dd y $
        \emph{(बढ़ाना $ g $ उत्पाद के \omterm{def:b3:topology:basis}{आधार} पर
        $(e_m(x)e_n(y))_{m,n}$ का $ L^2(\intcc01^2)$ --- ए
        \omterm{def:b3:hilbert:onb}{हिल्बर्ट आधार}, सी.एफ.\ \cref{exo:b3:spectral:5} --- और
        लागू करें \omterm{thm:b3:hilbert:parseval}{पारसेवल} चौराहे पर; प्रश्न 7 पहचानता है
        गुणांक)}.
  \item दोहरे अभिन्न अंग की गणना करें:
        \[
        \iint g^2 = 2\int_0^1(1-x)^2\Bigl(\int_0^x
        y^2\,\dd y\Bigr)\dd x
        = \frac23\int_0^1x^3(1-x)^2\,\dd x = \frac1{90} .
        \]
  \item निष्कर्ष निकालना $\zeta(4) = \dfrac{\pi^4}{90}$; समझाओ,
        बिना गणना के, उच्च शक्तियों के निशान कैसे
        $ G^k $ उत्पादन करना $\zeta(2k) \in \pi^{2k}\,\Q $ सभी के लिए
        $ k \geq 1$, और विषम मान क्यों $\zeta(3),
        \zeta(5), \dots $ सं\omterm{def:b3:galois:constructible}{रचनात्मक} रूप से इससे बाहर हैं
        मशीन की पहुंच.
  \item ($\pi $ नीचे से) से $\lambda_1^2 \leq
        \sum_n\lambda_n^2 = \frac1{90}$ परिणाम निकालना $\pi \geq
        90^{1/4} > 3.080$, और प्रश्न 14 के साथ एकत्रित हों
        दो तरफा फैसला
        \[
        3.080 \;<\; \pi \;<\; 3.1419,
        \]
        पूरी तरह से कंपन स्ट्रिंग से प्राप्त किया गया
        अंकगणित यदि कोई उपयोग करता है तो कौन सा पक्ष तेजी से अभिसरण करता है
        उच्चतर निशान $(\operatorname{tr}G^{2k})^{-1/4k}$,
        और क्यों?
\end{enumerate}

\noindent\textbf{Part VII --- Forcing and resonance.} हल करना
$\nu \in \R $ और मजबूर स्ट्रिंग पर विचार करें $-u'' - \nu u =
f $, $ u(0) = u(1) = 0$, साथ $ f \in L^2$ और $ c_n = \langle
e_n, f\rangle $.
\begin{enumerate}[resume]
  \item कल्पना करना $\nu \notin \{n^2\pi^2 : n \geq 1\}$. बताते हैं कि
        \[
        u = \sum_{n\geq1}\frac{c_n}{n^2\pi^2 - \nu}\,e_n
        \]
        में एकत्रित हो जाता है $ L^2$ और समान रूप से चालू $\intcc01$
        \emph{(कॉची-श्वार्ज़ के बीच $(c_n)$ और पूँछ
        का $\sum n^{-4}$, साथ $\norm{e_n}_\infty =
        \sqrt2$)}, और यह संतुष्ट करता है $ u = Gf + \nu Gu $
        --- का स्पष्ट-निर्देशांक रूप फ्रेडहोम
        वैकल्पिक (\cref{thm:b3:spectral:fredholm}), साथ
        विशिष्टता.
  \item कल्पना करना $\nu = m^2\pi^2$. बताते हैं कि $ u = Gf + \nu
        Gu $ एक समाधान है iff $ c_m = 0$, तक अद्वितीय
        के गुणज जोड़ना $ e_m $. शारीरिक पढ़ना: धक्का देना
        बिल्कुल अपनी ही आवृत्ति पर एक झूला।
  \item के लिए $\nu < \pi^2$, दिखाएँ कि समाधान ऑपरेटर
        $ R_\nu\colon f \mapsto u $ पर सीमाबद्ध है $ L^2$ के साथ
        \omterm{def:b3:rings:ideal}{आदर्श} $\frac1{\pi^2 - \nu}$, \omterm{def:b3:spectral:compact}{कॉम्पैक्ट}, \omterm{def:b3:spectral:selfadjoint}{स्वयं adjoint},
        और सकारात्मक: संपूर्ण वर्णक्रमीय विश्लेषण पुनः आरंभ होता है,
        द्वारा स्थानांतरित कर दिया गया $\nu $.
  \item (संश्लेषण) इस समस्या का शब्दकोश संकलित करें:
        eigenvalue $\leftrightarrow $ वर्ग आवृत्ति
        (हार्मोनिक्स); पता लगाना $\leftrightarrow $ $\zeta(2)$;
        \omterm{ex:b3:spectral:examples}{हिल्बर्ट--श्मिट} मानदंड $\leftrightarrow $ $\zeta(4)$;
        \omterm{thm:b3:spectral:fredholm}{फ्रेडहोम विकल्प} $\leftrightarrow $ प्रतिध्वनि;
        न्यूनतम अधिकतम $\leftrightarrow $ परिवर्तनशील सीमाएँ ($\pi <
        3.1437$ एक बहुपद से)। एक अभिन्न ऑपरेटर,
        विश्लेषण के पाँच अध्यायों को भुनाया गया।
\end{enumerate}

\noindent\textbf{Part VIII --- Three last echoes.}
\begin{enumerate}[resume]
  \item ($\zeta(6)$, निःशुल्क) प्रश्न 6 \omterm{thm:b3:hilbert:parseval}{पारसेवल}
        पहचान दी $\sum_{n\text{ odd}}n^{-6} =
        \frac{\pi^6}{960}$. विभाजित करना $\zeta(6)$ विषम और में
        यहां तक कि $ n $ और निष्कर्ष निकालो
        \[
        \zeta(6) = \frac{\pi^6}{945},
        \]
        बिना किसी नए इंटीग्रल के: प्रश्न 17 की मशीन
        (के निशान $ G^3$) समान मूल्य उत्पन्न करेगा
        एक पुनरावृत्त कर्नेल की कीमत पर --- \omterm{thm:b3:hilbert:parseval}{पारसेवल} एक पर
        अच्छी तरह से चुना गया फ़ंक्शन यहां सस्ता मार्ग है।
  \item (जमीनी स्थिति सकारात्मक है) चलो $ A $ एक हो \omterm{def:b3:spectral:compact}{कॉम्पैक्ट}
        \omterm{def:b3:spectral:selfadjoint}{स्वयं adjoint} सकारात्मक ऑपरेटर चालू $ L^2(\intcc01)$
        ए द्वारा दिया गया \omterm{def:b3:topology:continuity}{निरंतर} सममित कर्नेल $ k > 0$ पर
        $\intoo01^2$, सबसे बड़े eigenvalue के साथ $\mu_1$. दिखाएँ:
        (ए) रेले भागफल का कोई भी अधिकतमक एक है
        $\mu_1$-ईजेनफंक्शन; (बी) यदि $ u $ तो फिर एक है
        $\langle\abs u, A\abs u\rangle \geq \langle u,
        Au\rangle $, साथ \emph{कठोर} असमानता यदि $ u $ लेता है
        सकारात्मक के सेट पर दोनों संकेत \omterm{def:b3:measure:measure}{उपाय} --- इस तरह $ u $
        ए.ई. है\ \omterm{def:b3:topology:continuity}{निरंतर} संकेत, और $ u = \mu_1^{-1}Au $
        कभी गायब नहीं होता $\intoo01$; (सी) $\mu_1$ एक है
        \emph{\omterm{def:b3:groups:simple}{सरल}} eigenvalue. प्रत्येक दावे को सत्यापित करें $ G $:
        $ e_1 = \sqrt2\sin(\pi x) > 0$, और प्रत्येक $ e_n $, $ n
        \geq 2$, ओर्थोगोनल होने के नाते $ e_1$, संकेत बदलना होगा
        (यह करता है: $ n - 1$ आंतरिक भाग शून्य).
  \item (स्पेक्ट्रम से दूरी, और कीमत
        प्रतिध्वनि) के लिए $\nu \notin \{n^2\pi^2\}$, वह दिखाओ
        समाधान संचालक $ R_\nu $ प्रश्न 19 का है
        घिरा हुआ, \omterm{def:b3:spectral:selfadjoint}{स्वयं adjoint}, \omterm{def:b3:spectral:compact}{कॉम्पैक्ट}, साथ
        \[
        \vertiii{R_\nu} =
        \frac1{\min_{n\geq1}\,\abs{n^2\pi^2 - \nu}}
        = \frac1{\operatorname{dist}\bigl(\nu,
        \{n^2\pi^2\}\bigr)},
        \]
        निकटतम मोड पर मानक प्राप्त किया जा रहा है। फिर
        प्रश्न 20 के स्विंग को परिमाणित करें: जबरदस्ती के साथ $ f = e_1$
        पर $\nu = (1 - \varepsilon)\pi^2$ का उत्पादन $ u =
        \frac{1}{\varepsilon\pi^2}\,e_1$, द्वारा एक प्रवर्धन
        $\frac1\varepsilon $ स्थैतिक प्रतिक्रिया का $ Ge_1 =
        \frac1{\pi^2}e_1$ --- से एक प्रतिशत नीचे
        मौलिक ($\varepsilon = 10^{-2}$), स्ट्रिंग
        सौ गुना ऊंचे स्वर से उत्तर देता है.
\end{enumerate}
\end{problem}
