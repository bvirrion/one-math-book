\chapter{फूरियर रूपांतरण}\label{ch:b3:fouriertransform}

फूरियर श्रृंखला आवधिक संकेतों को असतत में विघटित करती है
हार्मोनिक्स; फूरियर \emph{परिवर्तन} के लिए भी ऐसा ही करता है
आवृत्तियों की निरंतरता के साथ, पूरी लाइन पर सिग्नल। यह
विभेदन को गुणन में, कनवल्शन में परिवर्तित करता है
उत्पाद, और गाऊसी से गाऊसी --- इसके कारण
विभेदक समीकरणों को हल करता है, सिग्नल प्रोसेसिंग को संचालित करता है, और
में केन्द्रीय सीमा प्रमेय सिद्ध करेंगे \cref{ch:b3:clt}. यह
अध्याय विकसित करता है $L^1$ सिद्धांत (रीमैन--लेब्सग्यू,
व्युत्क्रम, अंतःक्षेपण), द \omterm{def:b3:fouriertransform:schwartz}{श्वार्ट्ज वर्ग} कहाँ
परिवर्तन एक है \omterm{prop:b3:galois:perfect}{उत्तम} आपत्ति, और $ L^2$ सिद्धांत
(प्लान्चेरेल: परिवर्तन, एक स्थिरांक तक, एकात्मक है
ऑपरेटर), दो शोपीस अनुप्रयोगों के साथ: ताप समीकरण,
सप्ताहांत की समस्या और पॉइसन को अंत-से-अंत तक हल किया गया
योग सूत्र. सम्मेलन:
\[
\hat f(\xi) = \int_\R f(x)\,\eu^{-\iu\xi x}\,\dd x .
\]

\section{परिवर्तन चालू \texorpdfstring{$ L^1$}{L1}}

\begin{proposition}\label{prop:b3:fouriertransform:basic}
के लिए $ f \in L^1(\R)$: $\hat f $ अच्छी तरह परिभाषित है, घिरा हुआ है
($\norm{\hat f}_\infty \leq \norm f_1$), \omterm{def:b3:topology:continuity}{निरंतर}, और:
\begin{enumerate}
  \item $\widehat{\tau_af}(\xi) = \eu^{-\iu a\xi}\hat f(\xi)$
        और $\widehat{\eu^{\iu ax}f}(\xi) = \hat f(\xi - a)$;
  \item $\widehat{f(\cdot/\lambda)}(\xi) = \lambda\hat
        f(\lambda\xi)$ के लिए $\lambda > 0$;
  \item अगर $ xf \in L^1$, तब $\hat f $ है $\mathcal C^1$ साथ
        $(\hat f)'(\xi) = \widehat{(-\iu x)f}(\xi)$;
  \item अगर $ f \in \mathcal C^1$ साथ $ f' \in L^1$ (और $ f \to
        0$ पर $\pm\infty $, यहाँ स्वचालित), फिर
        $\widehat{f'}(\xi) = \iu\xi\hat f(\xi)$;
  \item $\widehat{f * g} = \hat f\,\hat g $ के लिए $ f, g \in L^1$.
\end{enumerate}
\end{proposition}

\begin{proof}
सीमाबद्धता: $\abs{\hat f} \leq \int\abs f $. निरंतरता: डीसीटी
प्रभुत्व के साथ $\abs f $ (\cref{thm:b3:lebesgue:paramcont}).
(1), (2): प्रतिस्थापन
(\cref{thm:b3:product:linearchange}). (3): भेदभाव
अभिन्न के अंतर्गत, प्रभुत्वकर्ता $\abs{xf}$
(\cref{thm:b3:lebesgue:paramdiff}). (4): सबसे पहले, $ f(x) = f(0)
+ \int_0^xf'$ पर एक सीमा है $\pm\infty $ ($ f' \in L^1$), जो
होना ही चाहिए $0$ ($ f \in L^1$); फिर भागों द्वारा एकीकृत करें $[-A,
A]$ और चलो $ A \to \infty $. (5): फ़ुबिनी, तब से वैध
$(x,y)\mapsto f(x - y)g(y)\eu^{-\iu\xi x}$ बिल्कुल है
\omterm{def:b3:lebesgue:l1}{समाकलनीय} (\cref{thm:b3:product:convolution}):
\[
\widehat{f*g}(\xi) = \iint f(x - y)g(y)\eu^{-\iu\xi(x - y)}
\eu^{-\iu\xi y}\dd x\,\dd y = \hat f(\xi)\,\hat g(\xi).
\]
\end{proof}

\begin{example}\label{ex:b3:fouriertransform:gaussian}
गाऊसी: के लिए $ a > 0$,\index{गॉसियन, फूरियर रूपांतरण}
\[
\widehat{\eu^{-ax^2}}(\xi) =
\sqrt{\frac\pi a}\;\eu^{-\xi^2/4a} :
\]
द्वारा \cref{exo:b3:lebesgue:7} (ओडीई $ F' = -\frac\xi{2}F $
ट्रिक, पुनर्स्केल), या (3) द्वारा: $ g = \widehat{\eu^{-ax^2}}$
संतुष्ट $ g'(\xi) = -\frac{\xi}{2a}g(\xi)$ (द्वारा एकीकृत करें
भाग), $ g(0) = \sqrt{\pi/a}$. गॉसियन के निश्चित बिंदु हैं
स्केलिंग तक परिवर्तन --- गहरा कारण है कि वे शासन करते हैं
\omterm{ex:b3:groups:actions}{केंद्र}ीय सीमा प्रमेय.
\end{example}

\begin{theorem}[रीमैन--लेब्सग्यू]
\label{thm:b3:fouriertransform:riemannlebesgue}
के लिए $ f \in L^1(\R)$: $\hat f(\xi) \to 0$ जैसा $\abs\xi \to
\infty $. इस प्रकार $\widehat{\phantom f} \colon L^1 \to \mathcal
C_0(\R)$ (\omterm{def:b3:topology:continuity}{निरंतर} कार्य लुप्त हो रहे हैं
अनंत).\index{रीमैन--लेब्सग्यू लेम्मा}
\end{theorem}

\begin{proof}
एक अंतराल के सूचक के लिए, $\widehat{\mathbf 1_{\intcc
ab}}(\xi) = \frac{\eu^{-\iu a\xi} - \eu^{-\iu b\xi}}{\iu\xi}
\to 0$; इसलिए चरणबद्ध कार्यों के लिए। स्टेप फ़ंक्शंस सघन हैं
$ L^1$ (\cref{thm:b3:lp:density}(1) प्लस सन्निकटन
परिमित-माप अंतरालों के परिमित संघों द्वारा सेट,
\cref{exo:b3:measure:7}), और परिवर्तन है
$\norm\cdot_\infty $-$\norm\cdot_1$ \omterm{def:b3:topology:continuity}{निरंतर}: के लिए $\norm{f -
s}_1 < \varepsilon $, $\limsup_{\abs\xi\to\infty}\abs{\hat
f(\xi)} \leq \varepsilon $.
\end{proof}

\section{उलटा और इंजेक्शन}

\begin{lemma}[गुणन सूत्र]
\label{lem:b3:fouriertransform:multiplication}
के लिए $ f, g \in L^1(\R)$: $\displaystyle\int \hat f\,g =
\int f\,\hat g $.
\end{lemma}

\begin{proof}
दोनों पक्ष बराबर $\iint f(x)g(\xi)\eu^{-\iu x\xi}\dd
x\,\dd\xi $ (टोनेली-फ़ुबिनी: का दोहरा अभिन्न अंग
निरपेक्ष मूल्य है $\norm f_1\norm g_1$).
\end{proof}

\begin{theorem}[उलट देना]\label{thm:b3:fouriertransform:inversion}
होने देना $ f \in L^1(\R)$.\index{फूरियर व्युत्क्रम प्रमेय}
\begin{enumerate}
  \item (गाऊसी सारांश) प्रत्येक के लिए $ x $,
        \[
        (f * g_\varepsilon)(x) =
        \frac1{2\pi}\int_\R \hat f(\xi)\,
        \eu^{-\varepsilon\xi^2}\,\eu^{\iu x\xi}\,\dd\xi,
        \qquad\text{where } g_\varepsilon(y) =
        \frac{1}{2\sqrt{\pi\varepsilon}}\,
        \eu^{-y^2/4\varepsilon},
        \]
        और $ f * g_\varepsilon \to f $ में $ L^1$ जैसा $\varepsilon
        \to 0$.
  \item यदि इसके अलावा $\hat f \in L^1$, तो लगभग हर के लिए
        $ x $
        \[
        f(x) = \frac{1}{2\pi}\int_\R \hat
        f(\xi)\,\eu^{\iu x\xi}\,\dd\xi ,
        \]
        और $ f $ एक \omterm{def:b3:topology:continuity}{निरंतर} प्रतिनिधि।
  \item (इंजेक्शन) यदि $\hat f = 0$ तब $ f = 0$ ए.ई.
\end{enumerate}
\end{theorem}

\begin{proof}
(1) ठीक करें $ x $ और आवेदन करें
\cref{lem:b3:fouriertransform:multiplication} को $ f $ और
$ g(\xi) = \frac1{2\pi}\eu^{-\varepsilon\xi^2}\eu^{\iu x\xi}$:
द्वारा \cref{ex:b3:fouriertransform:gaussian} (\omterm{def:b3:modules:module}{मॉड्यूल}ेशन के साथ
नियम),
\[
\hat g(y) = \frac1{2\pi}\sqrt{\frac\pi\varepsilon}\,
\eu^{-(y - x)^2/4\varepsilon} = g_\varepsilon(x - y),
\]
इसलिए $\frac1{2\pi}\int\hat f(\xi)\eu^{-\varepsilon\xi^2}
\eu^{\iu x\xi}\dd\xi = \int f(y)g_\varepsilon(x - y)\dd y =
(f*g_\varepsilon)(x)$. $ g_\varepsilon $ एक हैं
अनुमानित पहचान: $ g_\varepsilon \geq 0$,
$\int g_\varepsilon = 1$ (गॉसियन इंटीग्रल), पर ध्यान \omterm{ex:b3:groups:actions}{केंद्र}ित करना
$0$; का प्रमाण \cref{thm:b3:lp:regularization}(2) लागू होता है
शब्दशः (केवल $\int g_\varepsilon = 1$ और एकाग्रता थी
प्रयुक्त: पूँछ के लिए, $\int_{\abs y > \delta}g_\varepsilon \to
0$): $\norm{f * g_\varepsilon - f}_1 \to 0$.

(2) यदि $\hat f \in L^1$: (1) का दाहिना भाग, द्वारा अभिसरित होता है
डीसीटी (प्रभुत्वकर्ता) $\abs{\hat f}$), को $\frac1{2\pi}\int\hat
f(\xi)\eu^{\iu x\xi}\dd\xi $ के लिए \emph{प्रत्येक} $ x $, और यह
सीमा समारोह है \omterm{def:b3:topology:continuity}{निरंतर} (डीसीटी फिर से)। वहीं दूसरी ओर $ f
* g_\varepsilon \to f $ में $ L^1$, तो एक अनुवर्ती एई के साथ\
(\cref{thm:b3:lp:complete}): दो सीमाएँ सहमत हैं ae

(3) $\hat f = 0$ (1) का दाहिना भाग गायब कर देता है: $ f *
g_\varepsilon = 0$ सभी के लिए $\varepsilon $, और $ f *
g_\varepsilon \to f $ में $ L^1$: $ f = 0$ ए.ई.
\end{proof}

\section{श्वार्ट्ज वर्ग}

\begin{definition}\label{def:b3:fouriertransform:schwartz}
 \emph{श्वार्ट्ज वर्ग}\index{श्वार्ट्ज वर्ग} $\mathcal
S(\R)$ के होते हैं $\mathcal C^\infty $ कार्य $ f $ साथ
$\sup_x\abs{x^m f^{(n)}(x)} < \infty $ सभी के लिए $ m, n \geq 0$
(सभी डेरिवेटिव किसी भी शक्ति से अधिक तेजी से क्षय होते हैं)। उदाहरण:
$\eu^{-ax^2}$, $\mathcal C_c^\infty $. स्पष्ट रूप से $\mathcal S
\subseteq L^p $ हरएक के लिए $ p $ (से बंधा हुआ $ C(1 + x^2)^{-1}$), और
$\mathcal S $ डेरिवेटिव के तहत स्थिर है, से गुणा
बहुपद, और उत्पाद।
\end{definition}

\begin{theorem}\label{thm:b3:fouriertransform:schwartz}
फूरियर मानचित्रों को रूपांतरित करता है $\mathcal S(\R)$ वस्तुपरक रूप से पर
स्वयं, व्युत्क्रम के साथ $\check g(x) = \frac1{2\pi}\int
g(\xi)\eu^{\iu x\xi}\dd\xi $.
\end{theorem}

\begin{proof}
होने देना $ f \in \mathcal S $. बार-बार दोहराना
\cref{prop:b3:fouriertransform:basic}(3), $(\hat f)^{(n)} =
\widehat{(-\iu x)^nf}$ (प्रत्येक $ x^kf \in L^1$); पुनरावर्ती (4)
के साथ $ h = (-\iu x)^nf \in \mathcal S $ (जिनके सभी
व्युत्पन्न हैं \omterm{def:b3:lebesgue:l1}{समाकलनीय}), $(\iu\xi)^m\hat h =
\widehat{h^{(m)}}$. संयोजन,
\[
\abs{\xi^m\,(\hat f)^{(n)}(\xi)}
= \bigl|\widehat{\,h^{(m)}}(\xi)\bigr|
\leq \bigl\|\bigl((-\iu x)^nf\bigr)^{(m)}\bigr\|_1 < \infty
\]
समान रूप से में $\xi $: $\hat f \in \mathcal S $. तब से $\hat f \in L^1$, उलटा
(\cref{thm:b3:fouriertransform:inversion}(2)) धारण करता है
हर जगह (दोनों तरफ) \omterm{def:b3:topology:continuity}{निरंतर}): $\check{\hat f} = f $, और
सममित रूप से $\widehat{\check g} = g $ (चेक परिवर्तन
है $ g \mapsto \frac1{2\pi}\hat g(-\cdot)$, फिर से संरक्षण
$\mathcal S $): आक्षेप.
\end{proof}

\section{प्लांचरेल और \texorpdfstring{$ L^2$}{L2}}

\begin{theorem}[प्लैंचरेल]\label{thm:b3:fouriertransform:plancherel}
के लिए $ f \in L^1 \cap L^2(\R)$:\index{प्लैंचरेल का प्रमेय}
\[
\norm{\hat f}_2^2 = 2\pi\,\norm f_2^2 .
\]
फलस्वरूप $\widehat{\phantom f}$ तक विशिष्ट रूप से विस्तारित है
\omterm{def:b3:topology:continuity}{निरंतर} रेखीय मानचित्र $\mathcal F \colon L^2(\R) \to L^2(\R)$
साथ $\norm{\mathcal Ff}_2 = \sqrt{2\pi}\norm f_2$; $\mathcal
F $ विशेषण है, साथ में $\mathcal F^{-1} =
\frac1{2\pi}\,\mathcal F\circ\sigma $ कहाँ $\sigma f =
f(-\cdot)$, और यह संरक्षित करता है आंतरिक उत्पाद कारक तक
$2\pi $.
\end{theorem}

\begin{proof}
होने देना $ f \in L^1\cap L^2$ और $ h = f * \tilde f $ साथ
$\tilde f(x) = \overline{f(-x)}$. तब $ h \in L^1$
(\cref{thm:b3:product:convolution}), $ h $ है \omterm{def:b3:topology:continuity}{निरंतर} और
घिरा हुआ (\cref{exo:b3:lp:6}: $ f, \tilde f \in L^2$), $ h(0) =
\int f\bar f = \norm f_2^2$, और $\hat h = \hat
f\,\widehat{\tilde f} = \hat f\,\overline{\hat f} =
\abs{\hat f}^2 \geq 0$ (गणना करें $\widehat{\tilde f} =
\overline{\hat f}$). आवेदन करना
\cref{thm:b3:fouriertransform:inversion}(1) को $ h $ पर $ x = 0$:
\[
(h * g_\varepsilon)(0) = \frac1{2\pi}\int \hat
h(\xi)\,\eu^{-\varepsilon\xi^2}\dd\xi .
\]
जैसा $\varepsilon \to 0$: बाईं ओर की ओर झुकाव होता है $ h(0)$ ($ h $
\omterm{def:b3:topology:continuity}{निरंतर} परिबद्ध: $(h*g_\varepsilon)(0) - h(0) = \int(h(-y)
- h(0))g_\varepsilon(y)\dd y \to 0$ छोटा/बड़ा बाँटकर
$ y $); दाहिना भाग बढ़ जाता है $\frac1{2\pi}\int\hat h $ द्वारा
एमसीटी ($\hat h \geq 0$). इस तरह $\frac1{2\pi}\int\abs{\hat f}^2 =
\norm f_2^2$, परिमित है या नहीं \emph{संभवतः} --- और परिमित,
सदस्यता और पहचान दोनों को साबित करना।

विस्तार: $ L^1\cap L^2 \supseteq \mathcal C_c $ में सघन है
$ L^2$ (\cref{thm:b3:lp:density}); परिवर्तन है
$\sqrt{2\pi}$-आइसोमेट्रिक वहां, इसलिए विशिष्ट रूप से एक तक विस्तारित होता है
आइसोमेट्री-अप-टू-स्थिर $\mathcal F $ पर $ L^2$
(\cref{thm:b3:complete:extension}). के लिए उलटा $\mathcal S $
(\cref{thm:b3:fouriertransform:schwartz}) द्वारा स्थानांतरण
वही \omterm{ex:b3:lebesgue:gamma}{घनत्व} (दोनों पक्षों $ L^2$-निरंतर): $\mathcal
F\bigl(\frac1{2\pi}\mathcal F(\sigma f)\bigr) = f $ पर
$\mathcal S $, इसलिए आगे $ L^2$: वस्तुनिष्ठता. आंतरिक उत्पाद:
\omterm{def:b3:rings:ideal}{आदर्श} पहचान से ध्रुवीकरण.
\end{proof}

\begin{theorem}[पॉइसन सारांश]
\label{thm:b3:fouriertransform:poisson}
होने देना $ f \in \mathcal S(\R)$ (\omterm{def:b3:topology:continuity}{निरंतर} $ f $ साथ $\abs{f} +
\abs{\hat f} \leq C(1 + \abs\cdot)^{-2}$
पर्याप्त)। तब\index{पॉइसन योग सूत्र}
\[
\sum_{n\in\Z} f(n) \;=\; \sum_{k\in\Z}\hat f(2\pi k) .
\]
\end{theorem}

\begin{proof}
होने देना $ F(x) = \sum_{n\in\Z}f(x + n)$: शृंखला अभिसरित होती है
\omterm{def:b3:groups:normal}{सामान्य} रूप से चालू कॉम्पैक्ट कैमरों (का क्षय $ f $), इसलिए $ F $ है \omterm{def:b3:topology:continuity}{निरंतर}, और
यह है $1$-आवधिक. इसके फूरियर गुणांक (अवधि) $1$:
$ c_k(F) = \int_0^1F(t)\eu^{-2\iu\pi kt}\dd t $):
\[
c_k(F) = \sum_n\int_0^1 f(t + n)\,\eu^{-2\iu\pi kt}\dd t
= \int_\R f(t)\,\eu^{-2\iu\pi kt}\dd t = \hat f(2\pi k)
\]
(\omterm{def:b3:groups:normal}{सामान्य} अभिसरण इंटरचेंज को उचित ठहराता है; चरण है
$1$-आवधिक)। शृंखला $\sum_k\abs{c_k(F)}$ जुटता है
(का क्षय $\hat f $), इसलिए फूरियर श्रृंखला $ F $ जुटता है
सामान्यतः; इसका योग एक है \omterm{def:b3:topology:continuity}{निरंतर} उसी के साथ कार्य करें
फूरियर गुणांक के रूप में $ F $, इसलिए बराबर है $ F $ (इंजेक्शन चालू है
वृत्त: अंतर में शून्य गुणांक है, और
\cref{thm:b3:hilbert:fourier} शून्य देता है $ L^2$, इसलिए
हर जगह से निरंतरता). पर मूल्यांकन करें $ x = 0$.
\end{proof}

\begin{example}[थीटा पहचान]
\label{ex:b3:fouriertransform:theta}
पॉइसन को लागू करना $ f(x) = \eu^{-\pi tx^2}$ ($ t > 0$), जिसका
परिवर्तन है $\hat f(\xi) = t^{-1/2}\eu^{-\xi^2/4\pi t}$
(\cref{ex:b3:fouriertransform:gaussian} साथ $ a = \pi t $):
\[
\sum_{n\in\Z}\eu^{-\pi n^2t}
= \frac1{\sqrt t}\sum_{k\in\Z}\eu^{-\pi k^2/t} :
\]
जैकोबी के थीटा फ़ंक्शन के कार्यात्मक समीकरण, की कुंजी
रीमैन का कार्यात्मक समीकरण $\zeta $ --- और एक शानदार
संख्यात्मक त्वरक: के लिए $ t $ छोटा, बाईं ओर अभिसरण होता है
धीरे-धीरे, दाहिना भाग अत्यंत तेज़।
\end{example}

\begin{method}\label{met:b3:fouriertransform:method}
कार्य श्रेणियाँ: $ L^1$ --- परिभाषित बिंदुवार रूपांतरित करें,
व्युत्क्रम आवश्यकताएँ $\hat f \in L^1$; $\mathcal S $ --- सब कुछ
कानूनी है, पहले यहां साबित करें; $ L^2$ ---परिवर्तन द्वारा परिभाषित
\omterm{ex:b3:lebesgue:gamma}{घनत्व} (अभिन्न द्वारा नहीं!), \omterm{prop:b3:galois:perfect}{उत्तम} समरूपता, \omterm{thm:b3:hilbert:parseval}{पारसेवल}
बहीखाता। परिवर्तन की गणना करने के लिए: तालिका में कम करें
(सूचक, घातांकीय, गाऊसी) के नियमों द्वारा
\cref{prop:b3:fouriertransform:basic}; पहचान साबित करने के लिए:
इस पर स्थापित करें $\mathcal S $ (या $\mathcal C_c^\infty $) और
द्वारा विस्तार करें \omterm{ex:b3:lebesgue:gamma}{घनत्व} और निरंतरता
(\cref{met:b3:lp:toolkit}); एक रैखिक PDE या ODE को हल करने के लिए
\omterm{def:b3:topology:continuity}{निरंतर} गुणांक: रूपांतरित करना, विभाजित करना, उलटना।
\end{method}

\section{अभ्यास}

\begin{exercise}[$\star $]\label{exo:b3:fouriertransform:1}
फूरियर रूपांतरण की गणना करें: $\mathbf 1_{\intcc{-a}a}$;
$\eu^{-a\abs x}$ ($ a > 0$); तम्बू समारोह $\max(0, 1 -
\abs x)$; $\frac1{x^2 + a^2}$ \emph{(पर व्युत्क्रम का प्रयोग करें
दूसरा)}. उभरती हुई तालिका को रिकॉर्ड करें.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b3:fouriertransform:2}
होने देना $ f \in L^1$. के संदर्भ में व्यक्त करें $\hat f $ परिवर्तन
का: $ f(x - a)$, $ f(x)\cos(bx)$, $ f(ax + b)$,
$\overline{f(-x)}$, $(f * f)(x)$. प्रत्येक नियम को सत्यापित करें
गाऊसी।
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:fouriertransform:3}
(ए) वह दिखाओ $\mathbf 1_{\intcc{-1}1} * \mathbf
1_{\intcc{-1}1}$ परिवर्तन हुआ है $\bigl(\frac{2\sin\xi}\xi
\bigr)^2$, और निष्कर्ष निकालें
$\int_\R\bigl(\frac{\sin\xi}\xi\bigr)^2\dd\xi = \pi $ द्वारा
प्लैंचरेल --- या व्युत्क्रम द्वारा $0$. तुलना करना
\cref{pb:b3:lebesgue:1}.
(बी) गणना करें $\int_\R\frac{\dd x}{(x^2+1)^2}$ प्लैंचरेल के माध्यम से
पर लागू किया गया $\eu^{-\abs x}$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:fouriertransform:4}
(गिरी को गर्म करें बीजगणित) के साथ $ g_t(x) =
\frac1{2\sqrt{\pi t}}\eu^{-x^2/4t}$:
(ए) सत्यापित करें $\hat g_t(\xi) = \eu^{-t\xi^2}$;
(बी) अर्धसमूह कानून निकालें $ g_t * g_s = g_{t+s}$ बिना किसी के
अभिन्न संगणना;
(सी) दिखाओ $\norm{g_t}_1 = 1$ और $\norm{g_t}_2^2 =
(8\pi t)^{-1/2}$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:fouriertransform:5}
दिखाओ अगर $ f \in L^1$ सम और वास्तविक है, $\hat f $ सम है
और असली; अगर $ f $ अजीब और वास्तविक है, $\hat f $ अजीब और विशुद्ध है
काल्पनिक. क्या करता है $\hat f(0)$ गणना करें? उसका निष्कर्ष निकालो $ f \geq
0$ बल $\norm{\hat f}_\infty = \hat f(0) = \int f $, और
संभाव्यता \omterm{ex:b3:lebesgue:gamma}{घनत्व} के लिए व्याख्या करें (\cref{ch:b3:clt}: ए
विशेषता फ़ंक्शन में मापांक होता है $\leq 1$, पर प्राप्त किया गया
$0$).
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b3:fouriertransform:6}
(गैर-प्रक्षेपात्मकता) वह दिखाओ $\widehat{\phantom f}\colon L^1
\to \mathcal C_0$ इंजेक्शन है और \omterm{def:b3:topology:continuity}{निरंतर}, लेकिन \emph{नहीं}
विशेषण, तीन चरणों में।
(i) इंजेक्टिविटी (\cref{thm:b3:fouriertransform:inversion}) और
निरंतरता ($\norm{\hat f}_\infty \leq \norm f_1$), और
$\mathcal C_0$ एक बानाच स्थान है (\omterm{def:b3:topology:interior}{बंद} किया हुआ)।
$\norm\cdot_\infty $).
(ii) यदि मानचित्र विशेषणात्मक होता, तो वह वस्तुनिष्ठ होता, और
ओपन मैपिंग प्रमेय (\cref{thm:b3:banach:openmapping})
एक स्थिरांक देगा $ C $ साथ $\norm f_1 \leq C\norm{\hat
f}_\infty $ सभी के लिए $ f \in L^1$.
(iii) इसका खंडन करें $ f_n(x) = \frac{\sin
x}{x}\cdot\frac{\sin(x/n)}{x/n}$: इसका रूपांतरण (तक) है
स्थिरांक) कनवल्शन $\mathbf 1_{\intcc{-1}1} *
\mathbf 1_{\intcc{-1/n}{1/n}}$-प्रकार समलम्बाकार ---दिखाएँ
$\norm{\hat f_n}_\infty \leq \pi $ समान रूप से, जबकि
$\norm{f_n}_1 \geq c\ln n $ के मेहराबों की गिनती करके
$\frac{\abs{\sin x}}x $ पर $[1, n]$ (जहां दूसरा कारक है
नीचे परिबद्ध), जैसा कि \cref{thm:b3:banach:fourierdiverge}.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:fouriertransform:7}
(चिकनापन $\leftrightarrow $ क्षय शब्दकोश) सिद्ध करें: $ f \in
L^1$ साथ $\hat f(\xi) = O(\abs\xi^{-k-1-\delta})$ कुछ के लिए
$\delta > 0$ तात्पर्य $ f $ एक $\mathcal C^k $ प्रतिनिधि।
इसके विपरीत $ f \in \mathcal C^k_c $ तात्पर्य $\hat f(\xi) =
O(\abs\xi^{-k})$. तम्बू पर दोनों दिशाओं का चित्रण करें
समारोह.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b3:fouriertransform:8}
(हाइजेनबर्ग की असमानता) के लिए $ f \in \mathcal S(\R)$ असली के साथ
$\norm f_2 = 1$, सिद्ध करना
\[
\Bigl(\int x^2f(x)^2\dd x\Bigr)\cdot
\Bigl(\frac1{2\pi}\int \xi^2\abs{\hat f(\xi)}^2\dd\xi\Bigr)
\;\geq\; \frac14 ,
\]
गॉसियनों के लिए समानता के साथ। \emph{(लिखना $1 = \int f^2 =
-\int x\,(f^2)'$ भागों द्वारा, कॉची-श्वार्ज़ द्वारा बाध्य, और
परिवर्तित करें $\norm{f'}_2$ प्लैंचरेल द्वारा।)} व्याख्या: ए
सिग्नल और उसका स्पेक्ट्रम दोनों \omterm{ex:b3:groups:actions}{केंद्र}ित नहीं हो सकते।
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:fouriertransform:9}
औचित्य \cref{ex:b3:fouriertransform:theta} विस्तार से
(गॉसियन के लिए पॉइसन की परिकल्पना), और पहचान का उपयोग करें
मूल्यांकन करना $\sum_{n\in\Z}\eu^{-\pi n^2}$ छह दशमलव तक
तीन पद. की कितनी शर्तें \emph{परिभाषित} शृंखला
क्या उसी सटीकता की आवश्यकता होगी $ t = 10^{-2}$, बनाम
परिवर्तित श्रृंखला?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:fouriertransform:10}
(बैंड-सीमित कार्य) चलो $ f \in L^2(\R)$ साथ $\mathcal Ff $
में समर्थन किया $\intcc{-\pi}\pi $. बताते हैं कि $ f $ एक है
प्रत्येक मान का विस्तार करने वाला प्रतिनिधि पुनर्प्राप्ति योग्य है
नमूनों से: साबित करें \emph{शैनन प्रक्षेप} पर
पूर्णांक,
\[
f(x) = \sum_{n\in\Z} f(n)\,
\frac{\sin\bigl(\pi(x - n)\bigr)}{\pi(x - n)}
\quad\text{in } L^2,
\]
विस्तार करके $\mathcal Ff $ फूरियर \omterm{def:b3:topology:basis}{आधार} में
$ L^2(\intcc{-\pi}\pi)$ (\cref{thm:b3:hilbert:fourier}) और
पद दर पद परिवर्तन करना।
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:fouriertransform:11}
(क्रम चार के संचालक के रूप में परिवर्तन) चालू $\mathcal
S(\R)$, होने देना $\mathcal F f = \hat f $.
(ए) व्युत्क्रम सूत्र का उपयोग करके दिखाएँ $(\mathcal F^2f)(x) =
2\pi\,f(-x)$, और निष्कर्ष निकालें $\mathcal F^4 = (2\pi)^2\,
\mathrm{id}$.
(बी) प्रत्येक eigenvalue का अनुमान लगाएं $\mathcal F $ पर
$\mathcal S $ से संबंधित $\{\pm\sqrt{2\pi},
\pm\iu\sqrt{2\pi}\}$, और इसके लिए एक eigenfunction प्रदर्शित करें
$+\sqrt{2\pi}$ \emph{(इस अध्याय का कौन सा कार्य है
अपने स्वयं के परिवर्तन के समानुपाती?)}.
(सी) दिखाएँ कि सम फलन संतुष्ट करते हैं $\mathcal F^2f = 2\pi
f $ और अजीब $\mathcal F^2f = -2\pi f $; एक उत्पादन करें
eigenvalue के लिए eigenfunction $-\iu\sqrt{2\pi}$ से
$ x\eu^{-x^2/2}$ इसके परिवर्तन (अंतर) की गणना करके
गॉसियन का परिवर्तन)।
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:fouriertransform:12}
(ऑटोसहसंबंध और वीनर की लेम्मा) के लिए $ f \in L^2(\R)$
परिभाषित करना $\tilde f(x) = \overline{f(-x)}$ और यह
\emph{ऑटो सहसंबंध} $ A_f = f * \tilde f $.
(ए) वह दिखाओ $ A_f $ एक परिबद्ध है \omterm{def:b3:topology:continuity}{निरंतर} के साथ कार्य करें
$ A_f(0) = \norm f_2^2 \geq \abs{A_f(x)}$ सभी के लिए $ x $
(\cref{exo:b3:lp:6} और कॉची-श्वार्ज़)।
(बी) दिखाएँ, पहले के लिए $ f \in L^1\cap L^2$, वह
$\widehat{A_f} = \abs{\hat f\,}^2 \geq 0$: द
स्वसहसंबंध में गैर-नकारात्मक परिवर्तन --- का स्पेक्ट्रा है
स्वसहसंबंध पावर स्पेक्ट्रा हैं।
(सी) पहचान निकालें $\int_\R\abs{\hat
f(\xi)}^2\eu^{\iu x\xi}\,\dd\xi = 2\pi A_f(x)$ (उलटा;
जब इसकी प्रयोज्यता का औचित्य सिद्ध करें $\hat f \in L^2$ है
$\abs{\hat f}^2 \in L^1$), और इसके लिए इसका मूल्यांकन करें $ f = \mathbf
1_{\intcc{-1/2}{1/2}}$ पर $ x = 0$: वापस पाना
$\int_\R\bigl(\frac{\sin u}u\bigr)^2\dd u = \pi $.
\end{exercise}

\section{समस्या: रेखा पर ताप समीकरण}

\begin{problem}[{सप्ताहांत समस्या --- $\partial_tu =
\partial^2_{xx}u $, अंत से अंत तक हल किया गया}]
\label{pb:b3:fouriertransform:1}
गर्मी फैलती है; समीकरण $\partial_tu = \partial_{xx}^2u $
यह कहता है \omterm{ex:b3:lebesgue:gamma}{घनत्व} स्थानीय द्वारा दी गई दर पर फैलता है
तापमान प्रोफ़ाइल की वक्रता. हम कॉची को हल करते हैं
समस्या चालू $\R $ --- दिया गया $ f $, खोजो $ u(t, x)$ के लिए $ t > 0$ साथ
$ u(0, \cdot) = f $ --- समाधान को उल्लेखनीय साबित करें
गुण, और देखें कि समय को उलटा क्यों नहीं किया जा सकता। लगातार,
$ g_t(x) = \frac1{2\sqrt{\pi t}}\eu^{-x^2/4t}$ है
\emph{गरम गिरी}\index{गरम गिरी} और $ u(t, \cdot) = g_t *
f $.

\medskip
\noindent\textbf{Part I --- Deriving the kernel.} काम
औपचारिक रूप से पहले: मान लीजिए $ u(t, \cdot) \in \mathcal S $ हल करता है
समीकरण, और चलो $\hat u(t, \xi)$ में परिवर्तन हो
$ x $.
\begin{enumerate}
  \item दिखाएँ (औपचारिक रूप से) $\partial_t\hat u = -\xi^2\hat u $,
        इस तरह $\hat u(t,\xi) = \eu^{-t\xi^2}\hat f(\xi)$, और
        पहचानो $ u(t) = g_t * f $
        (\cref{exo:b3:fouriertransform:4}). यह प्रेरित करता है
        \emph{परिभाषा} का $ u $; अब सब कुछ सिद्ध हो गया है
        सीधे तौर पर, के लिए $ f \in \mathcal C_b(\R)$ (बंधा हुआ
        \omterm{def:b3:topology:continuity}{निरंतर}) या $ f \in L^p $.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Verification.}
\begin{enumerate}[resume]
  \item उसके लिए दिखाओ $ t > 0$, $ u(t, x) = \int
        g_t(x-y)f(y)\dd y $ के लिए अच्छी तरह से परिभाषित है $ f \in \mathcal
        C_b $, ओर वो $ u $ है $\mathcal C^\infty $ में $(t, x)$
        पर $\intoo0\infty\times\R $ \emph{(अंतर्गत अंतर करें
        अभिन्न; स्थानीय स्तर पर गॉसियन डेरिवेटिव पर हावी रहें
        समान रूप से में $(t,x)$)}.
  \item सत्यापित करें $\partial_tg_t = \partial^2_{xx}g_t $ प्रत्यक्ष द्वारा
        गणना, और निष्कर्ष निकालना $\partial_tu =
        \partial^2_{xx}u $ के लिए $ t > 0$.
  \item (प्रारंभिक शर्त) उसके लिए दिखाओ $ f \in \mathcal
        C_b $, $ u(t, x) \to f(x)$ जैसा $ t \to 0^+$, समान रूप से चालू
        कॉम्पैक्ट कैमरों \emph{(अनुमानित पहचान: विभाजन $\abs y
        \leq \delta $, $\abs y > \delta $)}; के लिए $ f \in L^p $
        ($ p < \infty $), दिखाओ $\norm{u(t) - f}_p \to 0$.
  \item (त्वरित स्मूथिंग) निष्कर्ष: यहां तक कि केवल सीमित के लिए भी
        \omterm{def:b3:topology:continuity}{निरंतर} $ f $, समाधान है $\mathcal C^\infty $
        हरएक के लिए $ t > 0$ --- गर्मी तुरंत मिट जाती है
        खुरदरापन. गणना करना $ u(t, \cdot)$ स्पष्ट रूप से के लिए $ f =
        \mathbf 1_{\intoo0\infty}$ (एक त्रुटि फ़ंक्शन) और
        के तीन मानों के लिए इसकी प्रोफ़ाइल का रेखाचित्र बनाएं $ t $.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Structural properties.}
\begin{enumerate}[resume]
  \item (सकारात्मकता और तुलना) यदि $ f \geq 0$ तब $ u > 0$
        सभी के लिए $ t > 0$ (कड़ाई से, जब तक $ f = 0$ ए.ई.); अगर
        $ f_1 \leq f_2$ तब $ u_1 \leq u_2$. एक ठंडा स्थान गर्म होता है
        तुरंत: टिप्पणी करें.
  \item (संरक्षण) के लिए $ f \in L^1$: $\int u(t, x)\dd x =
        \int f $ सभी के लिए $ t $ \emph{(टोनेली)} --- कुल ताप है
        संरक्षित.
  \item (अपव्यय) के लिए $ f \in L^1\cap L^2$, के माध्यम से दिखाएं
        प्लांचरेल वह $ t \mapsto \norm{u(t)}_2$ है
        न बढ़ना, सख्ती से जब तक $ f = 0$, और गणना करें
        इसकी सीमा इस प्रकार है $ t \to \infty $. इसके अलावा दिखाएँ
        $\norm{u(t)}_\infty \leq \frac{\norm
        f_1}{2\sqrt{\pi t}} \to 0$: गर्मी फैलती है और मर जाती है।
  \item (विशिष्टता, $ L^2$ \omterm{def:b3:groups:action}{कक्षा}) चलो $ u $ के साथ समाधान हो
        $ u(t) \in L^2$ सभी के लिए $ t $, $ u \in \mathcal
        C^1(\intoo0\infty, L^2)$ प्राकृतिक अर्थों में और
        $ u(t) \to f $ में $ L^2$ जैसा $ t\to0$; यह स्वीकार करते हुए कि
        परिवर्तन इसे परिवर्तित करता है $\partial_t\hat u =
        -\xi^2\hat u $ बिंदुवार ए.ई.\ में $\xi $ ए.ई. के लिए\ $ t $
        \emph{(विरुद्ध परीक्षण द्वारा उचित ठहराया गया $\mathcal
        C_c^\infty $ में $\xi $ --- इसे रेखांकित करें)}, दिखाओ
        $\hat u(t,\xi) = \eu^{-t\xi^2}\hat f(\xi)$, इसलिए
        इस वर्ग में विशिष्टता.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The arrow of time.}
\begin{enumerate}[resume]
  \item दिखाओ कि \emph{पिछड़ा} समस्या गलत तरीके से प्रस्तुत की गई है:
        ताकि समाधान समय पर मौजूद रहे $-s $ ($ s > 0$) के साथ
        डेटा $ f $ समय पर $0$ --- यानि\ के लिए $ f = g_s * h $ को
        एक समाधान है $ h \in L^2$ ---यह जरूरी है
        $\eu^{s\xi^2}\hat f(\xi) \in L^2$: अत्यधिक क्षय
        शर्त चालू $\hat f $. एक स्पष्ट चिकनी प्रदर्शन करें $ f
        \in L^2$ जिसके लिए कोई भी पिछड़ा समाधान मौजूद नहीं है
        समय अंतराल: फ़ंक्शन को साथ ले जाएं $\hat f(\xi) =
        \eu^{-\abs\xi}$ --- पहचान करना $ f $
        (\cref{exo:b3:fouriertransform:1}) और दिखाओ
        $\eu^{s\xi^2}\eu^{-\abs\xi} \notin L^2$ हरएक के लिए $ s >
        0$.
  \item (स्मूथिंग बनाम.\ जानकारी) संक्षेप में बताएं
        पैराग्राफ, प्रश्न 5, 9 और 10 का उपयोग करते हुए, गर्मी क्यों
        अर्धसमूह $(f \mapsto g_t * f)_{t\geq0}$ इंजेक्शन है
        लेकिन विशेषण पर नहीं $ L^2$, और यह क्यों व्यक्त करता है
        प्रसार की अपरिवर्तनीयता.
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Shannon's sampling theorem.} ए
समारोह $ f \in L^2(\R)$ है \emph{बैंड-सीमित} को $\Omega $
अगर $\hat f = 0$ ए.ई.\ बाहर $\intcc{-\Omega}\Omega $; लिखना
$ PW_\Omega $ (पैले--वीनर स्पेस) इनके लिए।
\begin{enumerate}[resume]
  \item वह हर दिखाओ $ f \in PW_\Omega $ सहमत ए.ई.\ साथ $\mathcal C^\infty $ समारोह
        $\frac1{2\pi}\int_{-\Omega}^{\Omega}\hat
        f(\xi)\eu^{\iu x\xi}\,\dd\xi $ (चिकनापन को उचित ठहराएं और
        पहचान), सभी डेरिवेटिव के साथ सीमाबद्ध:
        बैंड सीमा नियमितता का एक चरम रूप है।
        अभी से $ f $ इस प्रतिनिधि को दर्शाता है.
  \item बढ़ाना $\hat f \in L^2(\intcc{-\Omega}\Omega)$ में
        उस अंतराल का फूरियर \omterm{def:b3:topology:basis}{आधार} और पहचान करें
        गुणांक के रूप में \emph{नमूने} का $ f $:
        \[
        \hat f(\xi) = \frac\pi\Omega\sum_{n\in\Z}
        f\Bigl(\frac{n\pi}\Omega\Bigr)\,
        \eu^{-\iu n\pi\xi/\Omega}
        \quad\text{in } L^2(\intcc{-\Omega}\Omega) .
        \]
  \item निष्कर्ष निकालें \emph{नमूनाकरण प्रमेय}: के लिए $ f \in
        PW_\Omega $,
        \[
        f(x) = \sum_{n\in\Z}f\Bigl(\frac{n\pi}
        \Omega\Bigr)\,\operatorname{sinc}(\Omega x - n\pi),
        \qquad \operatorname{sinc}t = \frac{\sin t}t,
        \]
        में अभिसरण के साथ $ L^2(\R)$ और समान रूप से चालू $\R $
        \emph{(प्रश्न 13 की श्रृंखला को इसमें शामिल करें
        व्युत्क्रम सूत्र और प्राथमिक की गणना करें
        अभिन्न)}: एक बैंड-सीमित सिग्नल पूरी तरह से है
        चरण की ग्रिड पर इसके मानों द्वारा निर्धारित किया जाता है
        $\pi/\Omega $ --- नाइक्विस्ट दर.
  \item दिखाएँ कि कार्य $ x \mapsto
        \operatorname{sinc}(\Omega x - n\pi)$, $ n \in \Z $,
        में एक ऑर्थोगोनल परिवार बनाएं $ L^2(\R)$ स्थिरांक के साथ
        \omterm{def:b3:rings:ideal}{आदर्श} $\sqrt{\pi/\Omega}$, और ऊर्जा कम करें
        पहचान $\norm f_2^2 =
        \frac\pi\Omega\sum_n\abs{f(n\pi/\Omega)}^2$.
  \item (उपनाम) एक अशून्य प्रदर्शित करें $ g \in PW_{2\Omega}$
        प्रत्येक नमूना बिंदु पर गायब होना $\frac{n\pi}\Omega $
        \emph{(विचार करना $ g(x) =
        \sin(\Omega x)\operatorname{sinc}(\Omega x)$ और
        इसका बैंड जांचें)}: नाइक्विस्ट दर से नीचे नमूनाकरण
        जानकारी खो देता है --- दो अलग-अलग सिग्नल साझा कर सकते हैं
        सभी नमूने: स्ट्रोबोस्कोपिक वैगन-व्हील प्रभाव,
        गणित किया गया।
  \item (स्वतंत्रता की \omterm{def:b3:galois:extension}{डिग्री}) प्रश्न 14-15 का प्रयोग करते हुए औचित्य सिद्ध करें
        इंजीनियरिंग नियम: एक सिग्नल बैंड-सीमित
        $\Omega $ जिसकी ऊर्जा मूलतः a द्वारा वहन की जाती है
        लंबाई की समय खिड़की $ T $ द्वारा वर्णित है
        लगभग $\frac{\Omega T}\pi $ वास्तविक नमूने ---
        ऊर्जा के माध्यम से ``अनिवार्य रूप से'' सटीक बनाएं
        पहचान और पूंछ $\sum_{\abs{n\pi/\Omega} >
        T/2}$.
  \item (संगति जांच) नमूना प्रमेय को सत्यापित करें
        के दो सदस्यों पर हाथ $ PW_\Omega $: (ए) $ f =
        \operatorname{sinc}(\Omega\,\cdot)$, जिनके नमूने
        हैं $\delta_{n0}$; (बी) $ f(x) = \cos(\omega x)
        \operatorname{sinc}(\varepsilon x)$-टाइप नैरोबैंड
        सिग्नल --- अधिक सटीक रूप से, इसके लिए दिखाएं $ f \in
        PW_{\Omega'}$ साथ $\Omega' < \Omega $, द
        $\Omega $-दर श्रृंखला भी पुनर्निर्माण करती है $ f $
        (ओवरसैंपलिंग हानिरहित है), एम्बेडिंग द्वारा $ PW_
        {\Omega'} \subseteq PW_\Omega $.
\end{enumerate}

\medskip
\noindent\textbf{Part VI --- Uncertainty, twice more.}
हाइजेनबर्ग की असमानता (\cref{exo:b3:fouriertransform:8})
सीमा \emph{कैसे} \omterm{ex:b3:groups:actions}{केंद्र}ित $ f $ और $\hat f $ संयुक्त रूप से कर सकते हैं
होना; यहाँ इसके सभी-या-कुछ नहीं सहोदर और इसका सटीक विवरण दिया गया है
संतृप्ति.
\begin{enumerate}[resume]
  \item होने देना $ f \in L^1$ साथ $\operatorname{supp}f \subseteq
        \intcc{-A}A $. बताते हैं कि $\hat f $ एक का योग है
        हर जगह-अभिसरण शक्ति श्रृंखला:
        \[
        \hat f(\xi) = \sum_{k\geq0}\frac{(-\iu\xi)^k}{k!}
        \,m_k, \qquad m_k = \int_{-A}^{A}x^kf(x)\,\dd x,
        \quad \abs{m_k} \leq A^k\norm f_1
        \]
        \emph{(बढ़ाना $\eu^{-\iu\xi x}$ और औचित्य सिद्ध करें
        \omterm{def:b3:groups:normal}{सामान्य} अभिसरण द्वारा विनिमय)}:परिवर्तन है
        \emph{वास्तविक विश्लेषणात्मक}, की अनंत त्रिज्या के साथ
        हर बिंदु पर अभिसरण.
  \item समर्थन द्वंद्व का निष्कर्ष निकालें: एक वास्तविक-विश्लेषणात्मक
        गैर-रिक्त खुले अंतराल पर फ़ंक्शन गायब हो रहा है
        समान रूप से गायब हो जाता है \emph{(वह सेट जहां सब कुछ है
        डेरिवेटिव ग़ायब खुला और \omterm{def:b3:topology:interior}{बंद} है --- काम करें
        टेलर तर्क)}; निष्कर्ष निकालें कि कोई शून्येतर नहीं $ f $
        दोनों हैं $ f $ और $\hat f $ सघन रूप से समर्थित, और
        वह $ PW_\Omega $ इसमें सघन रूप से कोई शून्येतर नहीं है
        समर्थित फ़ंक्शन --- बैंड-सीमित सिग्नल अंतिम
        हमेशा के लिए, और समय-सीमित सिग्नल सभी में लीक हो जाते हैं
        आवृत्तियाँ।
  \item (हाइजेनबर्ग की संतृप्ति) गॉसियन परिवार पर $ f
        = \eu^{-ax^2}$, दोनों एकाग्रता कारकों की गणना करें
        और सत्यापित करें कि \omterm{def:b3:groups:normal}{सामान्य}ीकृत उत्पाद
        $\bigl(\int x^2\abs f^2\bigr)\bigl(\frac1{2\pi}\int
        \xi^2\abs{\hat f}^2\bigr)\big/\norm f_2^4$ के बराबर होती है
        $\frac14$ के लिए \emph{प्रत्येक} $ a $ --- समानता
        का परिवार \cref{exo:b3:fouriertransform:8} साक्षात; एक स्केलिंग तर्क द्वारा समझाएं कि उत्पाद क्यों
        परिवार के साथ \omterm{def:b3:topology:continuity}{निरंतर} रहना चाहिए।
  \item भौतिक रीडिंग (स्थिति/संवेग) को स्पष्ट करें
        क्वांटम अवस्था का \omterm{ex:b3:lebesgue:gamma}{घनत्व}; $\hbar $ में
        \omterm{def:b3:groups:normal}{सामान्य}ीकरण देता है $\sigma_x\sigma_p \geq
        \frac\hbar2$), और पूरे अध्याय से जुड़ें:
        त्वरित स्मूथिंग (भाग II), अपरिवर्तनीयता (भाग)।
        IV), नमूनाकरण (भाग V), हाइजेनबर्ग और समर्थन
        द्विभाजन एक नियम की पाँच अभिव्यक्तियाँ हैं ---
        का व्यवहार $\hat f $ अनंत पर विधान क्या है $ f $
        कहीं भी कर सकते हैं.
\end{enumerate}

\medskip
\noindent\textbf{Part VII --- The kernel's algebra, and one
\omterm{def:b3:groups:derived}{व्याख्या करने योग्य} example.}
\begin{enumerate}[resume]
  \item (सेमीग्रुप) सिद्ध करें \emph{चैपमैन--कोलमोगोरोव
        पहचान} $ g_t * g_s = g_{t+s}$ के लिए $ t, s > 0$ (के माध्यम से)
        कनवल्शन प्रमेय और इंजेक्टिविटी
        चालू करो $ L^1$), और निष्कर्ष निकालें $ u(t + s) = g_s *
        u(t)$: समय के अनुसार विकसित हो रहा है $ t + s $ के लिए विकसित हो रहा है $ t $,
        फिर के लिए $ s $. प्रश्न 8 के अपव्यय को तेज करें:
        लेखन $\norm{u(t)}_2^2 = \frac1{2\pi}\int
        \eu^{-2t\xi^2}\abs{\hat f(\xi)}^2\dd\xi $, द्वारा दिखाओ
        कॉची--श्वार्ज़ वह
        \[
        t \longmapsto \ln\,\norm{u(t)}_2
        \quad\text{is convex on } \intoo0{+\infty} :
        \]
         $ L^2$ ऊर्जा केवल घटती ही नहीं, बल्कि घटती भी है
        लॉग-उत्तल तरीके से घटता है।
  \item (गर्मी कहाँ जाती है) चलो $ f \geq 0$, $ f \in L^1$,
        साथ $\int x^2f(x)\dd x < \infty $. वह सभी को दिखाओ
        $ t > 0$
        \[
        \int_\R x\,u(t, x)\,\dd x = \int_\R x f(x)\,\dd x,
        \qquad
        \int_\R x^2u(t, x)\,\dd x
        = \int_\R x^2f(x)\,\dd x + 2t\int_\R f :
        \]
         \omterm{ex:b3:groups:actions}{केंद्र} गर्मी कभी नहीं चलती, और विचरण
        बढ़ता है \emph{रैखिक} समय में--विस्तारकारी
        स्केलिंग $ x \sim \sqrt{2t}$, कब दोबारा पढ़ा जाए
        ब्राउनियन गति प्रकट होती है \cref{ch:b3:probability}.
        \emph{(पहले दो क्षणों की गणना करें $ g_t $ और उपयोग करें
        कनवल्शन पर टोनेली।)}
  \item (गॉसियन, अंत से अंत तक हल किया गया) के लिए $ f(x) =
        \eu^{-x^2}$, \omterm{def:b3:topology:interior}{बंद} फॉर्म स्थापित करें
        \[
        u(t, x) = \frac1{\sqrt{1 + 4t}}\,
        \exp\Bigl(-\frac{x^2}{1 + 4t}\Bigr),
        \]
        और इसे हाथ से सत्यापित करें: समीकरण $\partial_tu
        = \partial^2_{xx}u $; संरक्षण $\int u(t) =
        \sqrt\pi $; अपव्यय कानून $\norm{u(t)}_2 =
        (\pi/2)^{1/4}(1 + 4t)^{-1/4}$ (इसकी तुलना करें $ t^{-1/4}$
        के साथ क्षय $ t^{-1/2}$ प्रश्न का सुपर-मानक क्षय
        8); और प्रश्न 24 की सटीक विचरण वृद्धि
        $ t = 6$: शिखर तक गिर गया है $\frac15$ इसके
        प्रारंभिक ऊँचाई जबकि प्रोफ़ाइल पाँच गुना चौड़ी है
        --- वही गर्मी, फैली हुई।
\end{enumerate}
\end{problem}
