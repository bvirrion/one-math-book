\chapter{समूह सिद्धांत}\label{ch:b3:groups}

वर्ष 2 का खंड बहीखाता उपकरणों के रूप में समूहों से मिला: लैग्रेंज
प्रमेय, चक्रीय समूह, सममित समूह और उसके हस्ताक्षर। यह
अध्याय समूह सिद्धांत को एक में बदल देता है \emph{तरीका}. इंजन है
एक समूह की धारणा \emph{अभिनय} एक सेट पर: गिनती कक्षाओं और तय किया गया
अंक प्राप्त होता है \omterm{cor:b3:groups:classeq}{वर्ग समीकरण}, कॉची प्रमेय और तीन
\omterm{def:b3:groups:sylow}{सिलो} प्रमेय --- का मौलिक स्थानीय-से-वैश्विक सिद्धांत
परिमित समूह सिद्धांत. फिर हम समूहों को इकट्ठा करना सीखते हैं (प्रत्यक्ष और)
\omterm{def:b3:groups:semidirect}{अर्धप्रत्यक्ष उत्पाद}) और उन्हें अलग करना (\omterm{thm:b3:groups:jordanholder}{रचना शृंखला},
हल करने योग्य समूह), और हम प्रमेय को सिद्ध करते हैं कि, में
\cref{ch:b3:galois}, के बारे में तीन सदी पुराना प्रश्न बंद हो जाएगा
बहुपद समीकरण: \omterm{thm:b3:groups:ansimple}{वैकल्पिक समूह} $A_n $ है \omterm{def:b3:groups:simple}{सरल} के लिए $ n
\geq 5$.

\section{भागफल समूह और समरूपता प्रमेय}

लगातार, $ G $ गुणात्मक रूप से लिखा गया एक समूह है, $ e $ इसका
पहचान. वर्ष 2 खंड से स्मरण करें: उपसमूह, सहसमुच्चय $ gH $,
लैग्रेंज प्रमेय ($\abs G = [G:H]\,\abs H $ परिमित के लिए $ G $), द
किसी तत्व का क्रम, चक्रीय समूह और सममित समूह $ S_n $
इसके हस्ताक्षर रूपवाद के साथ $\varepsilon \colon S_n \to \{\pm1\}$.

\begin{definition}\label{def:b3:groups:normal}
एक उपसमूह $ N $ का $ G $ है \emph{सामान्य}\index{सामान्य}
(लिखा हुआ $ N \trianglelefteq G $) कब $ gNg^{-1} = N $ हरएक के लिए $ g
\in G $ --- समान रूप से, जब बाएँ और दाएँ सहसमुच्चय मेल खाते हैं: $ gN =
Ng $ सभी के लिए $ g $.
\end{definition}

\begin{theorem}[भागफल समूह]\label{thm:b3:groups:quotient}
होने देना $ N \trianglelefteq G $. सेट $ G/N $ कोसेट का, के साथ
गुणन $(gN)(hN) = ghN $, एक सुपरिभाषित समूह है
\emph{भागफल समूह}\index{भागफल समूह}, और यह
\emph{विहित प्रक्षेपण} $\pi \colon G \to G/N $, $ g \mapsto gN $,
कर्नेल के साथ एक विशेषण रूपवाद है $ N $. इसके विपरीत, प्रत्येक कर्नेल
एक समूह का रूपवाद है \omterm{def:b3:groups:normal}{सामान्य}: \omterm{def:b3:groups:normal}{सामान्य} उपसमूह बिल्कुल वही हैं
गुठली.
\end{theorem}

\begin{proof}
अच्छी तरह से परिभाषित होना ही संपूर्ण मुद्दा है। अगर $ gN = g'N $ और $ hN = h'N $,
लिखना $ g' = gn $, $ h' = hm $ साथ $ n, m \in N $. तब $ g'h' = gnhm =
gh\,(h^{-1}nh)\,m \in ghN $ तब से $ h^{-1}nh \in N $ सामान्यता से: द
कोसेट का उत्पाद प्रतिनिधियों पर निर्भर नहीं करता है।
साहचर्य, पहचान $ eN = N $ और व्युत्क्रम $(gN)^{-1} = g^{-1}N $
से विरासत में मिले हैं $ G $. स्पष्ट रूप से $\pi $ एक विशेषण रूपवाद है और
$\pi(g) = N \iff g \in N $.

अगर $ f \colon G \to H $ एक रूपवाद है और $ k \in \ker f $, तब
$ f(gkg^{-1}) = f(g)f(k)f(g)^{-1} = e $: गुठली हैं \omterm{def:b3:groups:normal}{सामान्य}.
\end{proof}

\begin{theorem}[सार्वभौमिक संपत्ति; प्रथम समरूपता प्रमेय]
\label{thm:b3:groups:firstiso}
होने देना $ f \colon G \to H $ एक रूपवाद हो और $ N \trianglelefteq G $ साथ
$ N \subseteq \ker f $. एक अद्वितीय रूपवाद है $\bar f \colon G/N
\to H $ साथ $ f = \bar f \circ \pi $. विशेषकर, लेना $ N = \ker
f $:\index{समरूपता प्रमेय}
\[
G/\ker f \;\xrightarrow{\;\sim\;}\; \operatorname{im} f,
\qquad gN \mapsto f(g).
\]
\end{theorem}

\begin{proof}
विशिष्टता: $\bar f(gN)$ होना ही चाहिए $ f(g)$. अस्तित्व: यदि $ gN = g'N $
तब $ g^{-1}g' \in N \subseteq \ker f $, इसलिए $ f(g) = f(g')$ और $\bar
f(gN) = f(g)$ अच्छी तरह से परिभाषित है; यह एक रूपवाद है क्योंकि $ f $ है। के लिए
$ N = \ker f $: $\bar f $ चूँकि, इंजेक्शन है $\bar f(gN) = e $ मतलब
$ g \in \ker f $, यानी\ $ gN = N $; इसकी छवि है $ f $.
\end{proof}

\begin{theorem}[दूसरा और तीसरा समरूपता प्रमेय]
\label{thm:b3:groups:secondiso}
होने देना $ H \leq G $ और $ N \trianglelefteq G $.
\begin{enumerate}
  \item $ HN = \{hn : h \in H,\, n \in N\}$ एक उपसमूह है, $ N
        \trianglelefteq HN $, $ H \cap N \trianglelefteq H $, और
        \[
        H/(H \cap N) \;\cong\; HN/N .
        \]
  \item यदि इसके अलावा $ N \subseteq K \trianglelefteq G $, तब $ K/N
        \trianglelefteq G/N $ और $(G/N)\big/(K/N) \cong G/K $.
\end{enumerate}
\end{theorem}

\begin{proof}
(1) $ HN $ एक उपसमूह है: $(hn)(h'n') = hh'\,(h'^{-1}nh')n' \in HN $
और $(hn)^{-1} = h^{-1}(hn^{-1}h^{-1}) \in HN $, की सामान्यता का उपयोग करते हुए
$ N $. लिखें $ H \hookrightarrow HN \xrightarrow{\pi} HN/N $: यह
रूपवाद विशेषण है ($ hnN = hN $) कर्नेल के साथ $\{h \in H : h \in
N\} = H \cap N $; आवेदन करना \cref{thm:b3:groups:firstiso}.

(2) प्रक्षेपण $ G/N \to G/K $, $ gN \mapsto gK $, अच्छी तरह से परिभाषित है
($ N \subseteq K $), विशेषण, कर्नेल के साथ $ K/N $; आवेदन करना
\cref{thm:b3:groups:firstiso} दोबारा।
\end{proof}

\begin{theorem}[पत्राचार प्रमेय]\label{thm:b3:groups:correspondence}
होने देना $ N \trianglelefteq G $. वो नक्शा $ H \mapsto H/N $ एक आक्षेप है
के उपसमूहों के बीच $ G $ युक्त $ N $ और के उपसमूह $ G/N $,
समावेशन, सूचकांक और सामान्यता को संरक्षित करना (दोनों दिशाओं में)।
\end{theorem}

\begin{proof}
इसका उलटा है $\bar H \mapsto \pi^{-1}(\bar H)$. दोनों मानचित्र भेजते हैं
उपसमूह से उपसमूह, और परस्पर व्युत्क्रम हैं: $\pi^{-1}(H/N) =
HN = H $ तब से $ N \subseteq H $, और $\pi(\pi^{-1}(\bar H)) = \bar H $
की विशेषणात्मकता द्वारा $\pi $. समावेशन स्पष्ट रूप से संरक्षित हैं;
$[G:H] = [G/N : H/N]$ क्योंकि $ gH \mapsto (gN)(H/N)$ एक है
कोसेट रिक्त स्थान के बीच अच्छी तरह से परिभाषित आक्षेप; और $ gHg^{-1} = H $ सभी के लिए $ g $ आईएफएफ $(gN)(H/N)(gN)^{-1} = H/N $ सभी के लिए $ gN $, फिर से
की प्रक्षेपात्मकता $\pi $.
\end{proof}

\begin{example}\label{ex:b3:groups:firstiso}
$\varepsilon \colon S_n \to \{\pm 1\}$ देता है $ S_n/A_n \cong
\{\pm1\}$; $\det \colon GL_n(K) \to K^\times $ देता है
$ GL_n(K)/SL_n(K) \cong K^\times $; $ t \mapsto \eu^{2\iu\pi t}$ देता है
$\R/\Z \cong \mathbb U $, वृत्त समूह. प्रथम समरूपता
प्रमेय बताता है कि भागफल कैसे होते हैं \emph{गणना} व्यवहार में: एक खोजें
दाएँ कर्नेल के साथ आक्षेप।
\end{example}

\begin{method}\label{met:b3:groups:normal}
साबित करना $ N \trianglelefteq G $, सुंदरता के घटते क्रम में:
प्रदर्शन $ N $ एक रूपवाद के कर्नेल के रूप में परिभाषित किया गया है $ G $; जाँच करना
$ gNg^{-1} \subseteq N $ सभी के लिए $ g $ (यह पर्याप्त है: इसे लागू करना
$ g^{-1}$ और संयुग्मन विपरीत समावेशन देता है); सत्यापित करें कि
$ N $ का एक संघ है \omterm{ex:b3:groups:actions}{संयुग्मता वर्ग}; या उस पर ध्यान दें $[G:N] = 2$
(तब $ gN = Ng $ मजबूर है --- \cref{exo:b3:groups:1}).
\end{method}

\section{समूह क्रियाएँ}

\begin{definition}\label{def:b3:groups:action}
एक \emph{कार्रवाई}\index{कार्रवाई} का $ G $ एक सेट पर $ X $ एक है
रूपवाद $\varphi \colon G \to \mathfrak{S}(X)$ के समूह में
का आक्षेप $ X $; एक लिखता है $ g \cdot x $ के लिए $\varphi(g)(x)$.
समतुल्य: एक नक्शा $ G \times X \to X $ साथ $ e \cdot x = x $ और $ g
\cdot (h \cdot x) = (gh) \cdot x $. \emph{कक्षा}\index{कक्षा} का
$ x $ है $\mathcal O_x = \{g \cdot x : g \in G\}$, इसका
\emph{स्टेबलाइजर}\index{स्टेबलाइजर} उपसमूह है $ G_x = \{g :
g\cdot x = x\}$, और $ X^G = \{x : \forall g,\ g \cdot x = x\}$ है
का सेट \emph{निश्चित बिंदु}. क्रिया है \emph{सकर्मक} यदि
बिल्कुल एक ही कक्षा है, \emph{वफादार} अगर $\varphi $ है
इंजेक्शन, \emph{मुक्त} यदि सभी स्टेबलाइजर्स तुच्छ हैं।
\end{definition}

\begin{example}\label{ex:b3:groups:actions}
पाँच \omterm{def:b3:groups:action}{कार्रवाई} सभी परिमित समूह सिद्धांत चलाएँ:
\begin{enumerate}
  \item $ G $ अपने आप पर \emph{वाम अनुवाद} $ g \cdot x = gx $:
        स्वतंत्र और सकर्मक.
  \item $ G $ अपने आप पर \emph{विकार} $ g \cdot x = gxg^{-1}$:
        कक्षाओं हैं \emph{संयुग्मता वर्ग}\index{संयुग्मता वर्ग}, स्थिरिकारी  \emph{केंद्रीकरणकर्ता}\index{केंद्रीकरणकर्ता} $ Z_G(x) = \{g : gx = xg\}$, तय किया गया
        इंगित करता है \emph{केंद्र}\index{केंद्र} $ Z(G)$.
  \item $ G $ कोसेट स्थान पर $ G/H $ द्वारा $ g \cdot xH = gxH $:
        सकर्मक, के साथ \omterm{def:b3:groups:action}{स्टेबलाइजर} कोसेट का $ H $ के बराबर $ H $.
        प्रत्येक सकर्मक \omterm{def:b3:groups:action}{कार्रवाई} इस रूप का है
        (\cref{exo:b3:groups:8}).
  \item $ G $ संयुग्मन द्वारा इसके उपसमूहों के सेट पर: द \omterm{def:b3:groups:action}{स्टेबलाइजर}
        का $ H $ है \emph{नॉर्मलाइज़र}\index{नॉर्मलाइज़र} $ N_G(H) =
        \{g : gHg^{-1} = H\}$, का सबसे बड़ा उपसमूह $ G $ जिसमें
        $ H $ है \omterm{def:b3:groups:normal}{सामान्य}.
  \item $ S_n $ पर $\intint{1}{n}$: सभी उदाहरणों की जननी।
\end{enumerate}
\end{example}

\begin{theorem}[कक्षा--स्थिरकारक]\label{thm:b3:groups:orbitstab}
वो नक्शा $ gG_x \mapsto g \cdot x $ एक सुपरिभाषित आपत्ति है
$ G/G_x \to \mathcal O_x $. विशेष रूप से, के लिए $ G $
परिमित,\index{कक्षा--स्थिरीकरण प्रमेय}
\[
\abs{\mathcal O_x} = [G : G_x] \quad\text{divides } \abs G,
\]
और यह कक्षाओं विभाजन $ X $ (वे की कक्षाएं हैं
समतुल्यता $ x \sim y \iff y \in \mathcal O_x $),
\[
\abs X = \sum_{i} \,[G : G_{x_i}]
\qquad (x_i\colon \text{one point per orbit}).
\]
\end{theorem}

\begin{proof}
अच्छी तरह से परिभाषित और इंजेक्शन: $ gG_x = hG_x \iff h^{-1}g \in G_x \iff
h^{-1}g \cdot x = x \iff g \cdot x = h \cdot x $; श्रृंखला को पढ़ें
दोनों दिशाएँ. प्रयोज्यता की परिभाषा है \omterm{def:b3:groups:action}{कक्षा}. द
गिनती के कथन लैग्रेंज के प्रमेय और से अनुसरण करते हैं
का विभाजन $ X $ में कक्षाओं.
\end{proof}

\begin{corollary}[वर्ग समीकरण]\label{cor:b3:groups:classeq}
एक सीमित समूह के लिए $ G $, एक प्रतिनिधि चुनना $ x_i $ प्रत्येक में
संयुग्मन वर्ग एक से अधिक तत्वों के साथ:\index{वर्ग समीकरण}
\[
\abs G = \abs{Z(G)} + \sum_i \,[G : Z_G(x_i)],
\qquad\text{each } [G : Z_G(x_i)] > 1 \text{ dividing } \abs G.
\]
\end{corollary}

\begin{proof}
आवेदन करना \cref{thm:b3:groups:orbitstab} संयुग्मन के लिए \omterm{def:b3:groups:action}{कार्रवाई}:
सिंगलटन कक्षाओं वास्तव में के तत्व हैं $ Z(G)$.
\end{proof}

\begin{theorem}[के निश्चित बिंदु $ p $-समूह]\label{thm:b3:groups:pfixed}
होने देना $ p $ प्रधान बनो. ए \emph{$ p $-समूह}\index{$ p $-समूह} एक है
परिमित समूह जिसका क्रम एक शक्ति है $ p $. यदि एक $ p $-समूह $ G $ कार्य करता है
एक सीमित सेट पर $ X $, तब
\[
\abs{X^G} \equiv \abs X \pmod p .
\]
परिणाम: एक गैरतुच्छ $ p $-समूह में गैर-तुच्छता है \omterm{ex:b3:groups:actions}{केंद्र}, और
आदेश का प्रत्येक समूह $ p^2$ एबेलियन है.
\end{theorem}

\begin{proof}
प्रत्येक \omterm{def:b3:groups:action}{कक्षा} प्रमुखता है $[G:G_x]$, की एक शक्ति $ p $; यह शक्ति है
$1$ बिल्कुल निश्चित बिंदुओं पर और अन्यथा विभाज्य $ p $. संक्षेप
खत्म कक्षाओं अनुरूपता देता है. के लिए \omterm{ex:b3:groups:actions}{केंद्र}: संयुग्मन
\omterm{def:b3:groups:action}{कार्रवाई} का $ G $ अपने आप पर है $ X^G = Z(G)$, इसलिए $\abs{Z(G)} \equiv
\abs G \equiv 0 \pmod p $, और $ Z(G) \ni e $ बल $\abs{Z(G)} \geq
p $. आदेश $ p^2$: अगर $ Z(G) \neq G $ तब $\abs{Z(G)} = p $ और
$ G/Z(G)$ क्रम का चक्रीय है $ p $, जो मजबूर करता है $ G $ एबेलियन
(\cref{exo:b3:groups:2}) --- विरोधाभास।
\end{proof}

\begin{theorem}[कॉची]\label{thm:b3:groups:cauchy}
यदि कोई प्राइम $ p $ विभाजित $\abs G $, तब $ G $ का एक तत्व शामिल है
आदेश $ p $.\index{कॉची का प्रमेय}
\end{theorem}

\begin{proof}[प्रमाण (मैके)]
चलो $ X = \{(g_1, \dots, g_p) \in G^p : g_1 g_2 \cdots g_p = e\}$.
का चयन $ g_1, \dots, g_{p-1}$ स्वतंत्र रूप से निर्धारित करता है $ g_p $: $\abs X =
\abs G^{p-1}$, द्वारा विभाज्य $ p $. चक्रीय समूह $\Z/p\Z $ पर कार्य करता है
$ X $ चक्रीय बदलाव द्वारा $(g_1, \dots, g_p) \mapsto (g_2, \dots, g_p,
g_1)$ --- यह संरक्षित करता है $ X $, तब से $ g_2 \cdots g_p g_1 =
g_1^{-1}(g_1 \cdots g_p)g_1 = e $. द्वारा \cref{thm:b3:groups:pfixed},
$\abs{X^{\Z/p\Z}} \equiv \abs X \equiv 0 \pmod p $. निश्चित बिंदु हैं
निरंतर टुपल्स $(g, \dots, g)$ साथ $ g^p = e $; टुपल $(e,
\dots, e)$ उनमें से एक है, इसलिए कम से कम हैं $ p $ उनमें से, इसलिए
कम से कम एक $ g \neq e $ साथ $ g^p = e $: इसका क्रम बिल्कुल सही है $ p $.
\end{proof}

\begin{theorem}[केली]\label{thm:b3:groups:cayley}
आदेश का हर समूह $ n $ में एम्बेड करता है $ S_n $.\index{केली का प्रमेय}
\end{theorem}

\begin{proof}
वाम अनुवाद $\varphi \colon G \to \mathfrak S(G) \cong S_n $ है
एक रूपवाद; $\varphi(g) = \mathrm{id}$ बल $ g = ge = e $: यह है
वफादार.
\end{proof}

\begin{method}\label{met:b3:groups:counting}
निश्चित-बिंदु गिनती परिमित समूह की सार्वभौमिक प्रारंभिक चाल है
सिद्धांत. कुछ ऐसा साबित करने के लिए \emph{मौजूद है} (एक \omterm{ex:b3:groups:actions}{केंद्र}ीय तत्व,
व्यवस्था का एक तत्व $ p $, ए \omterm{def:b3:groups:normal}{सामान्य} उपसमूह, एक निश्चित बिंदु), एक बनाओ
एक अच्छी तरह से चुना गया समूह एक अच्छी तरह से चुने गए परिमित सेट पर कार्य करता है, फिर तुलना करता है
$\abs{X^G}$ साथ $\abs X $ सापेक्ष $ p $, या चलो \omterm{def:b3:groups:action}{कक्षा} आकार विभाजित करते हैं
समूह क्रम. के प्रमाण
\cref{thm:b3:groups:pfixed,thm:b3:groups:cauchy} और तीनों का
नीचे दिए गए \omterm{def:b3:groups:sylow}{सिलो} प्रमेय इस एकल विचार पर पाँच भिन्नताएँ हैं।
\end{method}

\section{सिलो प्रमेय}

लैग्रेंज का प्रमेय कहता है कि उपसमूह का क्रम विभाजित होता है $\abs G $;
बातचीत विफल हो जाती है ($ A_4$, आदेश की $12$, आदेश का कोई उपसमूह नहीं है
$6$ --- \cref{exo:b3:groups:1}). \omterm{def:b3:groups:sylow}{सिलो} प्रमेय बचाव करते हैं
प्रमुख शक्तियों के लिए बातचीत, और उनकी गिनती खंड है
उत्पादन के लिए हमारे पास सबसे तेज़ \omterm{def:b3:groups:normal}{सामान्य} उपकरण है \omterm{def:b3:groups:normal}{सामान्य} उपसमूह.

\begin{definition}\label{def:b3:groups:sylow}
लिखना $\abs G = p^a m $ साथ $ p \nmid m $. ए \emph{सिलो
$ p $-उपसमूह}\index{सिलो
$ p $-उपसमूह} का $ G $ क्रम का एक उपसमूह है
$ p^a $ --- ए $ p $-सबसे बड़े कल्पनीय क्रम का उपसमूह। द
सिलो की संख्या $ p $-के उपसमूह $ G $ निरूपित किया जाता है $ n_p $.
\end{definition}

\begin{lemma}\label{lem:b3:groups:binom}
अगर $\abs G = p^a m $ साथ $ p \nmid m $, तब $\dbinom{p^a m}{p^a}
\equiv m \pmod p $.
\end{lemma}

\begin{proof}
में $\mathbb F_p[X]$, नये आदमी का सपना $(1+X)^p = 1 + X^p $ (द
गुणांक $\binom pk $, $0<k<p $, से विभाज्य हैं $ p $: $ p $
के अंश को विभाजित करता है $\frac{p!}{k!(p-k)!}$ लेकिन नहीं
हर) को पुनरावृत्त करता है $(1+X)^{p^a} = 1 + X^{p^a}$, कहाँ से
\[
(1+X)^{p^a m} = \bigl(1 + X^{p^a}\bigr)^m
= \sum_{k=0}^{m} \binom mk X^{k p^a} \quad\text{in } \mathbb
F_p[X].
\]
के गुणांक को पहचानें $ X^{p^a}$: बाईं तरफ $\binom{p^a
m}{p^a} \bmod p $, दायीं तरफ $\binom m1 = m $.
\end{proof}

\begin{theorem}[सिलो I: अस्तित्व]\label{thm:b3:groups:sylow1}
प्रत्येक प्राइम के लिए $ p $, \omterm{def:b3:groups:sylow}{सिलो} $ p $-के उपसमूह $ G $ अस्तित्व।
\end{theorem}

\begin{proof}[प्रमाण (विलैंड्ट)]
चलो $\Omega $ का सेट हो \emph{सबसेट} का $ G $ प्रमुखता का
$ p^a $; $ G $ पर कार्य करता है $\Omega $ बाएँ अनुवाद द्वारा $ g \cdot S = gS $.
द्वारा \cref{lem:b3:groups:binom}, $\abs\Omega = \binom{p^a m}{p^a}
\equiv m \not\equiv 0 \pmod p $, इतना कुछ \omterm{def:b3:groups:action}{कक्षा} $\mathcal O_S $ है
आकार प्राइम टू $ p $ (अगर $ p $ प्रत्येक को विभाजित किया \omterm{def:b3:groups:action}{कक्षा} आकार, यह विभाजित होगा
$\abs\Omega $). होने देना $ H = G_S $ हो \omterm{def:b3:groups:action}{स्टेबलाइजर} ऐसे का $ S $.
तब से $[G : H] = \abs{\mathcal O_S}$ के लिए प्रमुख है $ p $ और $ p^a \mid
\abs G = [G:H]\,\abs H $, हम पाते हैं $ p^a \mid \abs H $. इसके विपरीत, ठीक करें
$ s \in S $: वो नक्शा $ H \to S $, $ h \mapsto hs $, इंजेक्शन है और
में उतरता है $ S $ क्योंकि $ hS = S $; इस तरह $\abs H \leq \abs S = p^a $.
इसलिए $\abs H = p^a $.
\end{proof}

\begin{theorem}[सिलो II: वर्चस्व और संयुग्मता]\label{thm:b3:groups:sylow2}
होने देना $ P $ \omterm{def:b3:groups:sylow}{सिलो} बनो $ p $-उपसमूह तथा $ Q $ कोई $ p $-उपसमूह $ G $.
तब $ Q \subseteq gPg^{-1}$ कुछ के लिए $ g \in G $. विशेषकर सभी
\omterm{def:b3:groups:sylow}{सिलो} $ p $-उपसमूह संयुग्मित हैं, और $ P \trianglelefteq G \iff
n_p = 1$.
\end{theorem}

\begin{proof}
होने देना $ Q $ कोसेट स्पेस पर कार्य करें $ X = G/P $, कार्डिनैलिटी का $ m
\not\equiv 0 \pmod p $. द्वारा \cref{thm:b3:groups:pfixed} पर लागू किया गया
द $ p $-समूह $ Q $, $\abs{X^Q} \equiv m \not\equiv 0 \pmod p $:वहां
एक निश्चित कोसेट है $ gP $, यानी\ $ QgP = gP $, यानी\ $ g^{-1}Qg \subseteq
P $. अगर $ Q $ स्वयं एक है \omterm{def:b3:groups:sylow}{सिलो उपसमूह}, आदेशों की समानता बदल जाती है
$ Q \subseteq gPg^{-1}$ एक समानता में. अंत में $ P
\trianglelefteq G $ यदि यह संयुग्मित है $\{gPg^{-1}\}$ --- जिसके द्वारा
उपरोक्त हैं \emph{सभी} \omterm{def:b3:groups:sylow}{सिलो} $ p $-उपसमूह --- को कम करें
$\{P\}$.
\end{proof}

\begin{theorem}[सिलो III: गिनती]\label{thm:b3:groups:sylow3}
$ n_p \equiv 1 \pmod p $, और $ n_p = [G : N_G(P)]$, जो विभाजित करता है
$ m $.
\end{theorem}

\begin{proof}
होने देना $\mathrm{Syl}_p $ \omterm{def:b3:groups:sylow}{सिलो} का सेट हो $ p $-उपसमूह; $ G $ पर कार्य करता है
यह संयुग्मन द्वारा सकर्मक रूप से (\cref{thm:b3:groups:sylow2}), साथ
\omterm{def:b3:groups:action}{स्टेबलाइजर} का $ P $  \omterm{ex:b3:groups:actions}{नॉर्मलाइज़र} $ N_G(P) \supseteq P $: $ n_p = [G :
N_G(P)]$, और $ m = [G:P] = [G:N_G(P)]\,[N_G(P):P]$ दिखाता है $ n_p \mid
m $.

अब प्रतिबंधित करें \omterm{def:b3:groups:action}{कार्रवाई} को $ P $ और निश्चित अंक गिनें। अगर $ Q \in
\mathrm{Syl}_p $ द्वारा तय किया गया है $ P $, तब $ P \subseteq N_G(Q)$; दोनों
$ P $ और $ Q $ \omterm{def:b3:groups:sylow}{सिलो} हैं $ p $-समूह के उपसमूह $ N_G(Q)$, इसलिए
इसमें संयुग्मित (\cref{thm:b3:groups:sylow2} के लिए आवेदन किया $ N_G(Q)$);
लेकिन $ Q \trianglelefteq N_G(Q)$, इसलिए $ Q $ क्या वहां इसका एकमात्र संयुग्म है:
$ P = Q $. इस प्रकार एकमात्र निश्चित बिंदु है $ P $ स्वयं, और
\cref{thm:b3:groups:pfixed} देता है $ n_p = \abs{\mathrm{Syl}_p}
\equiv \abs{\mathrm{Syl}_p^P} = 1 \pmod p $.
\end{proof}

\begin{method}\label{met:b3:groups:sylowmethod}
दिए गए क्रम के एक समूह का विश्लेषण करना $ n = p^a m $: विभाजकों की सूची बनाएं
का $ m $ के अनुरूप $1$ आधुनिक $ p $ --- ये हैं उम्मीदवार
$ n_p $. यदि एकमात्र उम्मीदवार है $1$, \omterm{def:b3:groups:sylow}{सिलो} $ p $-उपसमूह है
\omterm{def:b3:groups:normal}{सामान्य}. अगर $ n_p > 1$ छोटा होने के लिए मजबूर किया जाता है, संयुग्मन द्वारा कार्य किया जाता है
$\mathrm{Syl}_p $ एक रूपवाद प्राप्त करने के लिए $ G \to S_{n_p}$ छोटे के साथ
गिरी. और तत्वों की गिनती करें: विशिष्ट \omterm{def:b3:groups:sylow}{सिलो} $ p $-के उपसमूह
\emph{मुख्य} आदेश $ p $ तुच्छ रूप से प्रतिच्छेद करते हैं, इसलिए वे ले जाते हैं
$ n_p(p-1)$ बिल्कुल क्रम के तत्व $ p $; ओवरलैपिंग के लिए मायने रखता है
अलग-अलग अभाज्य संख्याएं अक्सर विरोधाभास पैदा करती हैं
(\cref{exo:b3:groups:7}).
\end{method}

\begin{example}\label{ex:b3:groups:pq}
होने देना $\abs G = pq $ साथ $ p < q $ अभाज्य संख्याएँ और $ p \nmid q - 1$. तब
$ n_q \mid p $ और $ n_q \equiv 1 \bmod q $ बल $ n_q = 1$ (जैसा $ p <
q $); $ n_p \mid q $ और $ n_p \equiv 1 \bmod p $ बल $ n_p = 1$ (जैसा $ q
\not\equiv 1 \bmod p $). होने देना $ P, Q $ दो हो \omterm{def:b3:groups:normal}{सामान्य} सिलोज़: $ P
\cap Q = \{e\}$ (कोप्राइम आदेश), तो $\abs{PQ} = pq $
(\cref{exo:b3:groups:4}) और $ G \cong P \times Q \cong \Z/p\Z
\times \Z/q\Z \cong \Z/pq\Z $ द्वारा
\cref{prop:b3:groups:direct} नीचे। आदेश का हर समूह $15$,
$33$, $35$, \dots\ चक्रीय है. बहिष्कृत मामला $ p \mid q - 1$
बिल्कुल एक और समूह तैयार करता है, नॉनबेलियन --- सप्ताहांत देखें
समस्या (\cref{pb:b3:groups:1}).
\end{example}

\begin{example}[संपूर्ण सिलो जनगणना: $ S_4$]
\label{ex:b3:groups:s4sylow}
आइए विधि को आगे बढ़ाएँ $ G = S_4$, $\abs G = 24 = 2^3\cdot3$.
\emph{\omterm{def:b3:groups:sylow}{सिलो} $3$:} $ n_3 \mid 8$, $ n_3 \equiv 1 \bmod 3$, इसलिए $ n_3
\in \{1, 4\}$; तब से $\langle(123)\rangle $ और
$\langle(124)\rangle $ विशिष्ट हैं, $ n_3 = 4$ --- चार
उपसमूह $\langle(abc)\rangle $, प्रत्येक के लिए एक $3$-तत्व उपसमुच्चय
$\{a, b, c\}$, के लिए लेखांकन $8$ तीन चक्र. \omterm{def:b3:groups:sylow}{सिलो} ~ II द्वारा
वे संयुग्मी हैं, और संयुग्मन रूपवाद $ S_4 \to
S_{\mathrm{Syl}_3} \cong S_4$ यहाँ एक समरूपता है (इसका कर्नेल
में निहित है $ N = N_G(\langle(123)\rangle)$ आदेश की $24/4 =
6$, और ए \omterm{def:b3:groups:normal}{सामान्य} उपसमूह का $ S_4$ अंदर $ S_3$-पसंद $ N $ होना ही चाहिए
तुच्छ: इसमें चारों को ठीक करने वाले सम क्रमपरिवर्तन शामिल होंगे
सिलोज़, और केवल $ e $ करता है)। \emph{\omterm{def:b3:groups:sylow}{सिलो} $2$:} $ n_2 \mid 3$, $ n_2
\equiv 1 \bmod 2$: $ n_2 \in \{1, 3\}$. उपसमूह $ D =
\langle(1234), (13)\rangle $ आदेश है $8$ (एक डायहेड्रल $ D_4$:
शीर्षों के साथ वर्ग की समरूपता $1, 2, 3, 4$), क्या नहीं है
\omterm{def:b3:groups:normal}{सामान्य} ($(12)(1234)(12) = (2134)$ एक अलग उत्पन्न करता है
$4$-साइकिल उपसमूह), तो $ n_2 = 3$: की तीन प्रतियाँ $ D_4$
युग्मन के तीन तरीकों के अनुरूप $4$ ए में इंगित करता है
``वर्ग''। जनगणना की नैतिकता पर ध्यान दें: $\abs{S_4} = 24$
साइलो में से किसी एक के लिए \omterm{def:b3:groups:normal}{सामान्य} स्थिति में असफल होने की गुंजाइश छोड़ देता है, और दोनों --- करते हैं
क्रम की तुलना करें $12$, जहां गिनती उनमें से एक को मजबूर करती है \omterm{def:b3:groups:normal}{सामान्य}
(भाग~IV) \cref{pb:b3:groups:1}).
\end{example}

\section{उत्पाद, प्रत्यक्ष और अर्धप्रत्यक्ष}

\begin{proposition}[किसी प्रत्यक्ष उत्पाद को पहचानना]\label{prop:b3:groups:direct}
होने देना $ H, K \trianglelefteq G $ साथ $ H \cap K = \{e\}$ और $ HK = G $.
तब $(h,k) \mapsto hk $ एक समरूपता है $ H \times K \to
G $.\index{प्रत्यक्ष उत्पाद}
\end{proposition}

\begin{proof}
के लिए $ h \in H $, $ k \in K $, द \omterm{def:b3:groups:derived}{कम्यूटेटर} $ hkh^{-1}k^{-1}$ में निहित है
$ K $ (इसे ऐसे पढ़ें $(hkh^{-1})k^{-1}$, की सामान्यता का उपयोग करते हुए $ K $) और में
$ H $ (इसे ऐसे पढ़ें $ h(kh^{-1}k^{-1})$): यह है $ e $, इसलिए $ H $ और $ K $
तत्ववार आवागमन करें और मानचित्र एक रूपवाद है। यह विशेषणात्मक है
तब से $ HK = G $, और इंजेक्शन तब से $ hk = e $ देता है $ h = k^{-1} \in
H \cap K = \{e\}$.
\end{proof}

की सामान्यता \emph{दोनों} कारक वह है जो सबसे अधिक बार विफल होता है: में $ S_3
= \langle (1\,2\,3)\rangle \,\langle(1\,2)\rangle $ दोनों कारक
तुच्छ रूप से प्रतिच्छेद करें और उत्पन्न करें, फिर भी $ S_3 \not\cong \Z/3\Z \times
\Z/2\Z $. सही धारणा तब होती है जब केवल एक ही कारक होता है \omterm{def:b3:groups:normal}{सामान्य}:

\begin{definition}\label{def:b3:groups:semidirect}
होने देना $ H $, $ K $ समूह बनें और $\varphi \colon K \to
\operatorname{Aut}(H)$ एक रूपवाद. \emph{अर्धप्रत्यक्ष
उत्पाद}\index{अर्धप्रत्यक्ष
उत्पाद} $ H \rtimes_\varphi K $ है
सेट $ H \times K $ से सुसज्जित
\[
(h, k)\,(h', k') = \bigl(h\,\varphi(k)(h'),\; kk'\bigr).
\]
\end{definition}

\begin{proposition}\label{prop:b3:groups:semidirect}
$ H \rtimes_\varphi K $ एक समूह है; $ H \times \{e\}$ एक है \omterm{def:b3:groups:normal}{सामान्य}
उपसमूह समरूपी से $ H $, $\{e\} \times K $ एक उपसमूह समरूपी
को $ K $; वे तुच्छ रूप से प्रतिच्छेद करते हैं और उत्पन्न करते हैं। इसके विपरीत, यदि $ G =
NK $ साथ $ N \trianglelefteq G $, $ K \leq G $ और $ N \cap K = \{e\}$,
तब $ G \cong N \rtimes_\varphi K $ के लिए $\varphi(k) = (n \mapsto
knk^{-1})$.
\end{proposition}

\begin{proof}
प्रत्यक्ष सत्यापन: साहचर्यता कम हो जाती है $\varphi(kk') =
\varphi(k)\circ\varphi(k')$ और प्रत्येक $\varphi(k)$ एक रूपवाद होना;
पहचान है $(e,e)$ और $(h,k)^{-1} =
\bigl(\varphi(k^{-1})(h^{-1}), k^{-1}\bigr)$. प्रक्षेपण $(h,k)
\mapsto k $ पर एक रूपवाद है $ K $ कर्नेल के साथ $ H \times \{e\}$,
जो इसलिए है \omterm{def:b3:groups:normal}{सामान्य}. बातचीत के लिए: हर $ g \in G $ लिखते हैं
\emph{विशिष्ट} जैसा $ nk $ साथ $ n \in N $, $ k \in K $ (अस्तित्व: $ G =
NK $; विशिष्टता: $ nk = n'k'$ देता है $ n'^{-1}n = k'k^{-1} \in N \cap
K $), और
\[
(nk)(n'k') = n\,(kn'k^{-1})\;kk'
\]
पता चलता है कि $ nk \mapsto (n, k)$ के कानून का परिवहन करता है $ G $ उस के लिए
का $ N \rtimes_\varphi K $.
\end{proof}

\begin{example}\label{ex:b3:groups:semidirectexamples}
(ए) द \emph{डायहेड्रल समूह}\index{डायहेड्रल समूह} $ D_n $ ($ n \geq
3$) की $2n $ एक नियमित की समरूपता $ n $-गॉन: घूर्णन बनता है
ए \omterm{def:b3:groups:normal}{सामान्य} उपसमूह सूचकांक का $2$, कोई भी प्रतिबिंब उत्पन्न करता है
पूरक, और एक प्रतिबिंब द्वारा घूर्णन को संयुग्मित करना इसे उलट देता है:
$ D_n \cong \Z/n\Z \rtimes_\varphi \Z/2\Z $ साथ $\varphi(1) = (x
\mapsto -x)$.
(बी) एक पंक्ति का एफ़िन समूह, $\{x \mapsto ax + b : a \in
K^\times,\, b \in K\} \cong K \rtimes K^\times $: अनुवाद
\omterm{def:b3:groups:normal}{सामान्य}, समरूपता एक पूरक है।
(सी) $ S_n \cong A_n \rtimes \Z/2\Z $ (पूरक: कोई भी
ट्रांसपोज़िशन)।
(डी) द \omterm{pb:b3:groups:1}{चतुर्भुज समूह} $ Q_8$ है \emph{नहीं} ए \omterm{def:b3:groups:semidirect}{अर्धप्रत्यक्ष उत्पाद}
उचित उपसमूहों की संख्या: प्रत्येक गैर-तुच्छ उपसमूह में शामिल है $-1$
(\cref{pb:b3:groups:1}), इसलिए कोई भी दो उचित उपसमूह प्रतिच्छेद नहीं करते
तुच्छ रूप से।
\end{example}

\section{हल करने योग्य समूह; की सादगी \texorpdfstring{$ A_n $}{An}}

\begin{definition}\label{def:b3:groups:derived}
 \emph{कम्यूटेटर}\index{कम्यूटेटर} का $ x, y \in G $ है $[x,y]
= xyx^{-1}y^{-1}$; \emph{व्युत्पन्न उपसमूह}\index{व्युत्पन्न उपसमूह} $ D(G)$ सभी कम्यूटेटरों द्वारा उत्पन्न उपसमूह है।
\emph{व्युत्पन्न श्रृंखला} है $ D^0(G) = G $, $ D^{i+1}(G) = D(D^i(G))$,
और $ G $ है \emph{व्याख्या करने योग्य}\index{व्याख्या करने योग्य} अगर $ D^n(G) =
\{e\}$ कुछ के लिए $ n $.
\end{definition}

\begin{proposition}\label{prop:b3:groups:derived}
$ D(G)$ है \omterm{def:b3:groups:normal}{सामान्य} (वास्तव में हर ऑटोमोर्फिज्म के तहत स्थिर), $ G/D(G)$
एबेलियन है, और के लिए $ N \trianglelefteq G $: $ G/N $ एबेलियन $\iff
D(G) \subseteq N $. इसके अतिरिक्त $ G $ है \omterm{def:b3:groups:derived}{व्याख्या करने योग्य} यदि कोई शृंखला है
$ G = G_0 \trianglerighteq G_1 \trianglerighteq \dots
\trianglerighteq G_n = \{e\}$ प्रत्येक के साथ $ G_{i+1} \trianglelefteq
G_i $ और प्रत्येक भागफल $ G_i/G_{i+1}$ एबेलियन। उपसमूह और
के भागफल हल करने योग्य समूह हैं \omterm{def:b3:groups:derived}{व्याख्या करने योग्य}; इसके विपरीत, यदि $ N $ और
$ G/N $ हैं \omterm{def:b3:groups:derived}{व्याख्या करने योग्य}, ऐसा ही है $ G $.
\end{proposition}

\begin{proof}
एक ऑटोमोर्फिज्म $\alpha $ एमएपीएस $[x,y]$ को $[\alpha x, \alpha y]$: यह
को क्रमपरिवर्तित करता है commutators, इसलिए उनके द्वारा उत्पन्न उपसमूह को संरक्षित करता है;
संयुग्मन ऑटोमोर्फिज्म हैं, जहां से सामान्यता आती है। में $ G/D(G)$,
$\bar x\bar y \bar x^{-1}\bar y^{-1} = \overline{[x,y]} = \bar e $:
भागफल एबेलियन है। अगर $ G/N $ एबेलियन है तो हर $[x,y] \in
N $, इसलिए $ D(G) \subseteq N $; इसके विपरीत यदि $ D(G) \subseteq N $ तब
$ G/N $, एबेलियन का एक भागफल $ G/D(G)$ तीसरे समरूपता द्वारा
प्रमेय, एबेलियन है।

अगर $ G $ है \omterm{def:b3:groups:derived}{व्याख्या करने योग्य}, द व्युत्पन्न श्रृंखला एक ऐसी शृंखला है. इसके विपरीत,
एक जंजीर दी, $ D^i(G) \subseteq G_i $ प्रेरण द्वारा: $ G_i/G_{i+1}$
एबेलियन देता है $ D(G_i) \subseteq G_{i+1}$, इसलिए $ D^{i+1}(G) = D(D^i G)
\subseteq D(G_i) \subseteq G_{i+1}$; इस तरह $ D^n(G) = \{e\}$.

आनुवंशिकता: $ D^i(H) \subseteq D^i(G)$ के लिए $ H \leq G $ (प्रेरण),
और $ D^i(G/N) = \pi(D^i(G))$ तब से $\pi $ एमएपीएस commutators पर
commutators; यह उपसमूहों और भागफलों के लिए कथन देता है।
विस्तार: यदि $ D^m(G/N) = \{e\}$ तब $ D^m(G) \subseteq N $, और
$ D^n(N) = \{e\}$ देता है $ D^{m+n}(G) = D^n(D^m(G)) \subseteq D^n(N)
= \{e\}$.
\end{proof}

\begin{example}\label{ex:b3:groups:solvableexamples}
एबेलियन समूह हैं \omterm{def:b3:groups:derived}{व्याख्या करने योग्य}. $ p $-समूह हैं \omterm{def:b3:groups:derived}{व्याख्या करने योग्य}, प्रेरण द्वारा
आदेश पर: $ Z(G) \neq \{e\}$ और $ G/Z(G)$ एक छोटा है
$ p $-समूह। $ S_3$ और $ S_4$ हैं \omterm{def:b3:groups:derived}{व्याख्या करने योग्य}: $ S_4 \trianglerighteq A_4
\trianglerighteq V \trianglerighteq \{e\}$, कहाँ $ V = \{e,
(1\,2)(3\,4), (1\,3)(2\,4), (1\,4)(2\,3)\}$ का क्लेन समूह है
दोहरा स्थानान्तरण (\omterm{def:b3:groups:normal}{सामान्य} में $ S_4$: का एक संघ संयुग्मन
कक्षाएं), एबेलियन भागफल के साथ $\Z/2\Z $, $\Z/3\Z $, $ V $. में
\cref{ch:b3:galois}, ``डिग्री का \omterm{def:b3:groups:normal}{सामान्य} समीकरण $ n $ है
\omterm{def:b3:groups:derived}{व्याख्या करने योग्य} कट्टरपंथियों द्वारा'' सचमुच \emph{अर्थ} ``$ S_n $ एक है
हल करने योग्य समूह''. अगली परिभाषा का महत्व कहां से है.
\end{example}

\begin{definition}\label{def:b3:groups:simple}
एक समूह $ G \neq \{e\}$ है \emph{सरल}\index{सरल} यदि यह है
केवल \omterm{def:b3:groups:normal}{सामान्य} उपसमूह हैं $\{e\}$ और $ G $. एक नॉनबेलियन सरल
समूह नहीं है \omterm{def:b3:groups:derived}{व्याख्या करने योग्य}: $ D(G) \trianglelefteq G $ क्या नहीं है $\{e\}$
(अन्य $ G $ एबेलियन), तो $ D(G) = G $ और यह व्युत्पन्न श्रृंखला है
स्थिर. एबेलियन सरल समूह बिल्कुल ऐसे ही हैं $\Z/p\Z $, $ p $
प्राइम (एक एबेलियन समूह सरल है यदि इसमें कोई उचित गैर-तुच्छता नहीं है
उपसमूह, यदि यह लैग्रेंज द्वारा अभाज्य क्रम का चक्रीय है)।
\end{definition}

\begin{lemma}\label{lem:b3:groups:threecycles}
के लिए $ n \geq 3$, $ A_n $ द्वारा उत्पन्न होता है $3$-चक्र; के लिए $ n \geq 5$,
सभी $3$-चक्र संयुग्मी हैं \emph{में $ A_n $}.
\end{lemma}

\begin{proof}
का एक तत्व $ A_n $ की सम संख्या का गुणनफल है
स्थानान्तरण; उन्हें जोड़े और उपयोग करें (दाएँ से बाएँ रचना)
\[
(a\,b)(c\,d) = (a\,c\,b)(a\,c\,d), \qquad
(a\,b)(b\,c) = (a\,b\,c), \qquad
(a\,b)(a\,b) = e
\]
क्रमशः असंयुक्त, अतिव्यापी और समान युग्मों के लिए: प्रत्येक युग्म
स्थानान्तरण का एक उत्पाद है $3$-चक्र।

संयुग्मता: $\sigma(a\,b\,c)\sigma^{-1} = (\sigma a\; \sigma b\;
\sigma c)$, तो कोई दो $3$-चक्र कुछ से संयुग्मित होते हैं $\sigma \in
S_n $. अगर $\sigma $ अजीब है, इसे से बदलें $\sigma' = \sigma (d\,e)$
कहाँ $ d, e $ दो बिंदु बाहर हैं $\{a, b, c\}$ --- वे मौजूद हैं
तब से $ n \geq 5$; तब $\sigma'$ सम है और $\sigma'(a\,b\,c)
\sigma'^{-1} = \sigma(a\,b\,c)\sigma^{-1}$, तब से $(d\,e)$ आवागमन
के साथ $(a\,b\,c)$.
\end{proof}

\begin{theorem}[वैकल्पिक समूह की सरलता]\label{thm:b3:groups:ansimple}
$ A_n $ है \omterm{def:b3:groups:simple}{सरल} के लिए $ n \geq 5$.\index{वैकल्पिक समूह}
\end{theorem}

\begin{proof}
होने देना $ N \trianglelefteq A_n $, $ N \neq \{e\}$. द्वारा
\cref{lem:b3:groups:threecycles} यह दिखाना काफी है $ N $
रोकना \emph{एक} $3$-चक्र: सामान्यता और संयुग्मता
$3$-चक्र में $ A_n $ फिर सब डाल दो $3$-चक्र में $ N $, इसलिए $ N = A_n $.

के लिए $\rho \in S_n $ होने देना $ F(\rho) = \{x : \rho(x) \neq x\}$ यह हो
\emph{सहायता} और $ f(\rho) = \abs{F(\rho)}$. चुनना $\sigma \in N
\setminus \{e\}$ साथ $ f(\sigma)$ \emph{न्यूनतम}. ध्यान दें कि ए
यहां तक कि क्रमपरिवर्तन भी गैरतुच्छ है $ f \geq 3$, ओर वो $ f(\sigma) =
4$ के लिए असंभव है $\sigma \in A_n $ जब तक $\sigma $ एक दोहरा है
स्थानान्तरण (ए $4$-चक्र विषम है)। हम दिखाते हैं $\sigma $ एक है
$3$-चक्र।

\emph{केस ए: $\sigma $ असंयुक्त स्थानान्तरण का एक उत्पाद है},
कहना $\sigma = (a\,b)(c\,d)\cdots $ साथ $ f(\sigma) \geq 4$. चुनना $ e'
\notin \{a, b, c, d\}$ (संभव: $ n \geq 5$), तय करना $\tau =
(c\,d\,e')$ और
\[
\sigma' = \tau\sigma\tau^{-1}\,\sigma^{-1} \in N
\qquad (\tau\sigma\tau^{-1} \in N \text{ by normality}).
\]
तब से $\sigma\tau^{-1}\sigma^{-1} = (\sigma c\;\sigma e'\;\sigma
d) = (d\;\sigma e'\;c)$ (का उपयोग करते हुए $\sigma c = d $, $\sigma d = c $), हम पाते हैं $\sigma' = (c\,d\,e')(d\;\sigma e'\;c)$.

अगर $\sigma e' = e'$ (जो विशेष रूप से कब लागू होता है $ f(\sigma) =
4$, यानी\ $\sigma = (a\,b)(c\,d)$): तब $(d\,e'\,c) =
(c\,d\,e')$ और $\sigma' = (c\,d\,e')^2 = (c\,e'\,d)$, ए
$3$-साइकिल में पड़ा हुआ $ N $, साथ $ f(\sigma') = 3 < 4 \leq f(\sigma)$
--- न्यूनतमता का खंडन।

अगर $\sigma e' \neq e'$: तब $\sigma e' \notin \{a, b, c, d,
e'\}$ ($\sigma $ स्वैप $ a,b $ और $ c,d $, और $ e' \notin \{a,b,c,d\}$
साथ $\sigma $ इंजेक्शन), तो $\sigma $ छह अंक ले जाता है $ a, b,
c, d, e', \sigma e'$: $ f(\sigma) \geq 6$. वहीं दूसरी ओर
$\sigma'$, दो का एक उत्पाद $3$-सपोर्ट वाली साइकिलें $\{c, d,
e', \sigma e'\}$, संतुष्ट करता है $ f(\sigma') \leq 4$; और $\sigma' \neq
e $, तब से $\sigma'(d) = \tau\sigma\tau^{-1}(c) = \tau\sigma(e') =
\sigma e' \neq d $ ($\tau $ ठीक करता है $\sigma e' \notin \{c,d,e'\}$).
इसलिए $\sigma' \in N \setminus\{e\}$ साथ $ f(\sigma') \leq 4 <
f(\sigma)$: न्यूनतमता का खंडन किया गया है।

\emph{केस बी: कुछ चक्र $\sigma $ लंबाई है $\geq 3$}, कहना
$\sigma(a) = b $, $\sigma(b) = c $ साथ $ a, b, c $ विशिष्ट। अगर
$\sigma $ बिलकुल यही है $3$-चक्र, हमारा काम हो गया। अन्यथा
$ f(\sigma) \geq 5$ (मामला $ f(\sigma) = 4$ के साथ $\geq3$-चक्र
अजीब है $4$-चक्र, बहिष्कृत), इसलिए हम चुन सकते हैं $ d, e' \in
F(\sigma) \setminus \{a, b, c\}$. तय करना $\tau = (c\,d\,e')$ और
$\sigma' = \tau\sigma\tau^{-1}\sigma^{-1} \in N $. पहले जैसा
$\sigma' = (c\,d\,e')\,(\sigma c\;\sigma e'\;\sigma d)$ केवल चलता है
के अंक
\[
M = \{c, d, e'\} \cup \{\sigma c, \sigma d, \sigma e'\}
\subseteq F(\sigma)
\]
(स्थानांतरित बिंदुओं की छवियाँ स्थानांतरित की गई हैं: $\sigma(x) \ne x $ तात्पर्य
$\sigma(\sigma x) \neq \sigma x $, $\sigma $ इंजेक्शन होना)। भी
$ b \notin M $: पांच अंक $ a, b, c, d, e'$ विशिष्ट हैं, इसलिए $ b
\notin \{c, d, e'\}$; और $ b \in \{\sigma c, \sigma d, \sigma
e'\}$ जबरदस्ती करेगा $ a \in \{c, d, e'\}$ (आवेदन करना $\sigma^{-1}$, का उपयोग करना
$\sigma a = b $), जो गलत है। इस तरह $\sigma'$ ठीक करता है $ b $, जबकि
$\sigma $ चाल $ b $; और
$ F(\sigma') \subseteq F(\sigma)$. अंत में $\sigma' \neq e $:
$\sigma^{-1}(c) = b $, $\tau^{-1}(b) = b $, $\sigma(b) = c $,
$\tau(c) = d $, इसलिए $\sigma'(c) = d \neq c $. इस प्रकार $\sigma' \in N
\setminus \{e\}$ साथ $ f(\sigma') \leq f(\sigma) - 1$,
न्यूनतमता का खंडन.

दोनों ही स्थितियाँ असंभव हैं, $\sigma $ एक है $3$-चक्र।
\end{proof}

\begin{corollary}\label{cor:b3:groups:snnotsolvable}
के लिए $ n \geq 5$: $ A_n $ और $ S_n $ नहीं हैं \omterm{def:b3:groups:derived}{व्याख्या करने योग्य}, और एकमात्र
\omterm{def:b3:groups:normal}{सामान्य} उपसमूह का $ S_n $ हैं $\{e\}$, $ A_n $ और $ S_n $.
\end{corollary}

\begin{proof}
$ A_n $ नॉनबेलियन है \omterm{def:b3:groups:simple}{सरल}, इसलिए नहीं \omterm{def:b3:groups:derived}{व्याख्या करने योग्य}
(\cref{def:b3:groups:simple}); एक समूह जिसमें a न सुलझ सकने वाला
उपसमूह नहीं है \omterm{def:b3:groups:derived}{व्याख्या करने योग्य} (\cref{prop:b3:groups:derived}). होने देना $ N
\trianglelefteq S_n $: तब $ N \cap A_n \trianglelefteq A_n $ के बराबर होती है
$\{e\}$ या $ A_n $. अगर $ N \cap A_n = A_n $, तब $ A_n \subseteq N $
और $ N \in \{A_n, S_n\}$ सूचकांक द्वारा. अगर $ N \cap A_n = \{e\}$,
पर प्रतिबंध $ N $ प्रक्षेपण का $ S_n \to S_n/A_n \cong
\Z/2\Z $ इंजेक्शन है, इसलिए $\abs N \leq 2$; अगर $ N = \{e, \sigma\}$,
सामान्यता बनाता है संयुग्मन वर्ग का $\sigma $ के बराबर
$\{\sigma\}$, यानी\ $\sigma \in Z(S_n)$. लेकिन $ Z(S_n) = \{e\}$ के लिए
$ n \geq 3$: अगर $\sigma \ne e $ चाल $ a $ को $ b \neq a $, चुनना $ c
\notin \{a, b\}$; तब $(b\,c)\sigma(b\,c)^{-1}$ भेजता है $ a $ को $ c
\ne b $, इसलिए यह भिन्न है $\sigma $. इस तरह $ N = \{e\}$.
\end{proof}

\begin{theorem}[जॉर्डन--एच\"पुराने]\label{thm:b3:groups:jordanholder}
प्रत्येक परिमित समूह $ G \neq \{e\}$ स्वीकार करता है ए \emph{रचना
शृंखला}\index{रचना
शृंखला}
\[
\{e\} = G_0 \trianglelefteq G_1 \trianglelefteq \cdots
\trianglelefteq G_r = G,
\qquad G_{i}/G_{i-1} \text{ simple},
\]
और का मल्टीसेट \emph{रचना कारक} $ G_i/G_{i-1}$, ऊपर
समरूपता के लिए, चुनी गई श्रृंखला पर निर्भर नहीं करता है। एक परिमित
समूह है \omterm{def:b3:groups:derived}{व्याख्या करने योग्य} यदि इसके सभी संघटन कारक चक्रीय हैं
प्रमुख आदेश.\index{Jordan--Hölder theorem}
\end{theorem}

\begin{proof}
\emph{अस्तित्व}: प्रेरण चालू $\abs G $. अगर $ G $ है \omterm{def:b3:groups:simple}{सरल}, लेना
$\{e\} \trianglelefteq G $. अन्यथा अधिकतम उचित चुनें \omterm{def:b3:groups:normal}{सामान्य}
उपसमूह $ N $ (अंततः कई उपसमूह हैं); $ G/N $ है \omterm{def:b3:groups:simple}{सरल}
पत्राचार प्रमेय द्वारा (एक उचित गैर-तुच्छ \omterm{def:b3:groups:normal}{सामान्य} उपसमूह
का $ G/N $ एक तक उठा देगा \omterm{def:b3:groups:normal}{सामान्य} उपसमूह का $ G $ सख्ती से बीच में
$ N $ और $ G $). संलग्न $ N \trianglelefteq G $ एक को \omterm{thm:b3:groups:jordanholder}{रचना शृंखला}
का $ N $.

\emph{विशिष्टता}: प्रेरण चालू $\abs G $, मामला $ G $ \omterm{def:b3:groups:simple}{सरल}
स्पष्ट होना. टेक टू \omterm{thm:b3:groups:jordanholder}{रचना शृंखला}, अंतिम शब्दों के साथ
$ M \trianglelefteq G $ और $ N \trianglelefteq G $ (इसलिए $ G/M $, $ G/N $
हैं \omterm{def:b3:groups:simple}{सरल}). अगर $ M = N $, पर लागू प्रेरण द्वारा निष्कर्ष निकालें $ M $.
अन्यथा $ MN $, \omterm{def:b3:groups:normal}{सामान्य} में $ G $ और सख्ती से युक्त $ M $, बराबर है
$ G $ ($ M $ अधिकतम है \omterm{def:b3:groups:normal}{सामान्य}: कोई \omterm{def:b3:groups:normal}{सामान्य} $ M \subsetneq L
\subsetneq G $ एक उचित गैरतुच्छ के लिए मानचित्र होगा \omterm{def:b3:groups:normal}{सामान्य} उपसमूह की \omterm{def:b3:groups:simple}{सरल} $ G/M $). दूसरा समरूपता प्रमेय देता है
\[
G/M = MN/M \cong N/(M \cap N),
\qquad
G/N = MN/N \cong M/(M \cap N).
\]
तय करना $ K = M \cap N $ ($\trianglelefteq G $) और ठीक करें a \omterm{thm:b3:groups:jordanholder}{रचना
शृंखला} का $ K $. तब $ M $ दो ले जाता है \omterm{thm:b3:groups:jordanholder}{रचना शृंखला}: इसका
मूल एक, और की श्रृंखला $ K $ के बाद $ K \trianglelefteq
M $ (भागफल $ M/K \cong G/N $ है \omterm{def:b3:groups:simple}{सरल}). प्रेरण द्वारा (लागू)
को $ M $), की मूल श्रृंखला के कारक $ M $ हैं
$\{\text{factors of } K\} \cup \{G/N\}$; इसी तरह के लिए $ N $. इसलिए
की दोनों श्रृंखला $ G $ कारक हैं
\[
\{\text{factors of } K\} \;\cup\; \{\,G/N,\; G/M\,\},
\]
वही मल्टीसेट.

गलने की योग्यता: यदि सभी कारक हैं $\Z/p_i\Z $, शृंखला एक शृंखला है
एबेलियन भागफल के साथ, इसलिए $ G $ है \omterm{def:b3:groups:derived}{व्याख्या करने योग्य}
(\cref{prop:b3:groups:derived}). इसके विपरीत, एक रचना कारक
एक का हल करने योग्य समूह है \omterm{def:b3:groups:derived}{व्याख्या करने योग्य} (एक उपसमूह का एक भागफल) और
\omterm{def:b3:groups:simple}{सरल}; ए \omterm{def:b3:groups:derived}{व्याख्या करने योग्य} \omterm{def:b3:groups:simple}{सरल} समूह एबेलियन है ($ D(G) \ne G $ बल
$ D(G) = \{e\}$), इसलिए कुछ $\Z/p\Z $.
\end{proof}

\begin{remark}\label{rem:b3:groups:jhmeaning}
जॉर्डन--एच\"पुराने का कहना है कि प्रत्येक परिमित समूह का निर्माण होता है \omterm{def:b3:groups:simple}{सरल}
समूह, एक अच्छी तरह से परिभाषित भागों की सूची के साथ --- समूहों का एक अंकगणित
जिसमें \omterm{def:b3:groups:simple}{सरल} समूह अभाज्य संख्याएँ हैं, और कहाँ हैं \emph{कैसे
हिस्से चिपके हुए हैं} (विस्तार डेटा, जैसा कि \omterm{def:b3:groups:semidirect}{अर्धप्रत्यक्ष उत्पाद})
मात्र गुणन की जगह लेता है। परिमित का वर्गीकरण
\omterm{def:b3:groups:simple}{सरल} समूह --- चक्रीय $\Z/p\Z $, बारी-बारी से $ A_{n \geq
5}$, झूठ प्रकार के सोलह परिवार, और $26$ छिटपुट समूह --- है
बीसवीं सदी के गणित के स्मारकों में से एक; इसका प्रमाण,
लगभग दस हजार जर्नल पृष्ठों में फैला हुआ, बहुत आगे है
यह कोर्स.
\end{remark}

\section{अभ्यास}

\begin{exercise}[$\star $]\label{exo:b3:groups:1}
(ए) दिखाएँ कि सूचकांक का प्रत्येक उपसमूह $2$ है \omterm{def:b3:groups:normal}{सामान्य}.
(बी) दिखाओ अगर $[G : H] = 2$, तब $ x^2 \in H $ हरएक के लिए $ x \in
G $.
(सी) उसका निष्कर्ष निकालें $ A_4$ आदेश का कोई उपसमूह नहीं है $6$: लैग्रेंज का
बातचीत विफल हो जाती है. \emph{(के वर्ग गिनें $3$-चक्र।)}
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b3:groups:2}
दिखाओ अगर $ G/Z(G)$ तो फिर चक्रीय है $ G $ एबेलियन है. फिर से निष्कर्ष निकालें
वह आदेश का हर समूह $ p^2$ प्रत्येक के लिए एबेलियन और प्रदर्शन है
प्रधान $ p $, आदेश का एक नॉनबेलियन समूह $ p^3$. \emph{(सोचो
इकाई विकर्ण के साथ ऊपरी त्रिकोणीय आव्यूह $\mathbb F_p $.)}
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b3:groups:3}
(ए) वह दिखाओ $\operatorname{Aut}(\Z/n\Z) \cong (\Z/n\Z)^\times $.
(बी) दिखाएँ कि आंतरिक स्वचालितताएँ $\iota_g \colon x \mapsto
gxg^{-1}$ फॉर्म ए \omterm{def:b3:groups:normal}{सामान्य} उपसमूह $\operatorname{Inn}(G)
\trianglelefteq \operatorname{Aut}(G)$, साथ
$\operatorname{Inn}(G) \cong G/Z(G)$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:groups:4}
होने देना $ H, K $ एक परिमित समूह के उपसमूह बनें $ G $.
(ए) सिद्ध करें \emph{उत्पाद सूत्र} $\abs{HK}\,\abs{H\cap K} =
\abs H\, \abs K $, मानचित्र के रेशों को गिनकर $ H \times K
\to HK $, $(h,k) \mapsto hk $.
(बी) वह दिखाओ $ HK $ एक उपसमूह है iff $ HK = KH $ (स्वचालित जब एक
दोनों में से है \omterm{def:b3:groups:normal}{सामान्य}).
(सी) यदि $ H, K \trianglelefteq G $ और $ H \cap K = \{e\}$, बताते हैं कि
$ hk = kh $ सभी के लिए $ h \in H $, $ k \in K $.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:groups:5}
(बर्नसाइड्स काउंटिंग लेम्मा) एक सीमित समूह $ G $ एक परिमित पर कार्य करता है
सेट $ X $. दिखाओ कि संख्या कक्षाओं है \emph{औसत
निश्चित बिंदुओं की संख्या}:
\[
\#\{\text{orbits}\} = \frac{1}{\abs G}\sum_{g \in G}
\abs{\operatorname{Fix}(g)},
\qquad \operatorname{Fix}(g) = \{x \in X : g \cdot x = x\},
\]
सेट की गिनती करके $\{(g,x) : g\cdot x = x\}$ दो तरह से.
आवेदन पत्र: $\Z/p\Z $ ($ p $ prime) के हार पर घूर्णन द्वारा कार्य करता है
$ p $ मोतियों के साथ $ a $ उपलब्ध रंग; फ़र्मेट का छोटा निष्कर्ष निकालें
प्रमेय $ a^p \equiv a \pmod p $.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b3:groups:6}
साइलो प्रमेय का उपयोग करते हुए दिखाएँ कि क्रम का प्रत्येक समूह $15$ है
चक्रीय और वह क्रम का प्रत्येक समूह $45$ एबेलियन है.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:groups:7}
दिखाएँ कि आदेश का कोई समूह नहीं $30$, और कोई भी ऑर्डर नहीं $56$, है
\omterm{def:b3:groups:simple}{सरल}. \emph{(के लिए $30$: अगर $ n_3 \neq 1$ और $ n_5 \neq 1$, गिनती
आदेश के तत्व $3$ और $5$. के लिए $56$: तत्वों की गिनती करें
क्रम का $7$.)}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:groups:8}
(ए) चलो $ H \leq G $ सूचकांक का $ n $. दिखाओ कि \omterm{def:b3:groups:action}{कार्रवाई} का $ G $ पर
$ G/H $ एक रूपवाद उत्पन्न करता है $ G \to S_n $ किसकी गिरी $\bigcap_{g \in
G} gHg^{-1}$ सबसे बड़ा है \omterm{def:b3:groups:normal}{सामान्य} उपसमूह का $ G $ में निहित
$ H $.
(बी) निष्कर्ष निकालें: यदि $ G $ परिमित है और $ p $ है \emph{सबसे छोटा} प्रधान
का भाजक $\abs G $, सूचकांक का प्रत्येक उपसमूह $ p $ है \omterm{def:b3:groups:normal}{सामान्य}.
(सी) दिखाएँ कि प्रत्येक सकर्मक \omterm{def:b3:groups:action}{कार्रवाई} का $ G $ एक सेट पर $ X $ है
के लिए समरूपी \omterm{def:b3:groups:action}{कार्रवाई} कोसेट स्थान पर: एक आक्षेप है $ X
\to G/G_x $ के साथ आवागमन \omterm{def:b3:groups:action}{कार्रवाई}.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:groups:9}
(ए) वह दिखाओ $ D(G)$ सबसे छोटा है \omterm{def:b3:groups:normal}{सामान्य} उपसमूह का $ G $ के साथ
एबेलियन भागफल, और वह प्रत्येक रूपवाद $ G $ एक एबेलियन को
के माध्यम से विशिष्ट रूप से समूह कारक \emph{अबेलियनाइजेशन}
$ G^{\mathrm{ab}} = G/D(G)$.
(बी) गणना करें $ D(S_n)$ और $ S_n^{\mathrm{ab}}$ के लिए $ n \geq 2$, और
$ D(Q_8)$ और $ Q_8^{\mathrm{ab}}$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:groups:10}
होने देना $ G $ एक हो $ p $-समूह तथा $ H \subsetneq G $ एक उचित उपसमूह.
बताते हैं कि $ H \subsetneq N_G(H)$ (``\omterm{def:b3:groups:normal}{सामान्य}ीकरणकर्ता बढ़ो''), और घटाओ
कि प्रत्येक अधिकतम उपसमूह a $ p $-समूह है \omterm{def:b3:groups:normal}{सामान्य} सूचकांक का $ p $.
\emph{(प्रेरण चालू $\abs G $, का उपयोग करना $ Z(G) \neq \{e\}$: इलाज
अलग से $ Z(G) \subseteq H $ और $ Z(G) \not\subseteq H $.)}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b3:groups:11}
(सरलता $ A_5$, हाथ पर)
(ए) दिखाएँ कि \omterm{ex:b3:groups:actions}{संयुग्मता वर्ग} का $ A_5$ प्रमुखताएँ हैं
$1$, $15$, $20$, $12$, $12$. के बंटवारे पर ध्यान दें
$ S_5$-की श्रेणी $5$-चक्र: एक के लिए $5$-चक्र $\sigma $, तुलना करें
\omterm{ex:b3:groups:actions}{केंद्रीकरणकर्ता} का $\sigma $ में $ S_5$ और में $ A_5$.
(बी) उसका निष्कर्ष निकालें $ A_5$ है \omterm{def:b3:groups:simple}{सरल}: ए \omterm{def:b3:groups:normal}{सामान्य} उपसमूह का एक संघ है
\omterm{ex:b3:groups:actions}{संयुग्मता वर्ग}, रोकना $ e $, और कार्डिनैलिटी डिवाइडिंग है
$60$.
(सी) दिखाएँ कि ए \omterm{def:b3:groups:simple}{सरल} समूह आदेश की $60$ आवश्यक रूप से है $ n_5 =
6$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:groups:12}
(\omterm{def:b3:groups:normal}{सामान्य}ीकरणकर्ता का \omterm{def:b3:groups:sylow}{सिलो उपसमूह} स्\omterm{def:b3:groups:normal}{व-सामान्य}ीकरण कर रहे हैं) चलो $ P $
\omterm{def:b3:groups:sylow}{सिलो} बनो $ p $-एक परिमित समूह का उपसमूह $ G $ और $ H =
N_G(P)$.
(ए) वह दिखाओ $ P $ है \emph{अद्वितीय} \omterm{def:b3:groups:sylow}{सिलो} $ p $-उपसमूह
$ H $.
(बी) निष्कर्ष निकालना $ N_G(H) = H $. \emph{(के लिए $ g \in N_G(H)$: $ gPg^{-1}$
एक \omterm{def:b3:groups:sylow}{सिलो} है $ p $-उपसमूह $ H $, इसलिए $ gPg^{-1} = P $.)}
(सी) निष्कर्ष निकालें कि कोई \omterm{def:b3:groups:sylow}{सिलो} नहीं \omterm{ex:b3:groups:actions}{नॉर्मलाइज़र} एक उचित में निहित है
\omterm{def:b3:groups:normal}{सामान्य} उपसमूह का $ G $, और वह अधिकतम उपसमूह युक्त
$ N_G(P)$ स्\omterm{def:b3:groups:normal}{व-सामान्य}ीकरण है।
\end{exercise}

\section{समस्या: क्रम के समूह अधिकतम 15}

\begin{problem}[{सप्ताहान्त समस्या---छोटे का वर्गीकरण
समूह}]\label{pb:b3:groups:1}
इसका उद्देश्य पूर्ण प्रमाणों के साथ पूर्ण वर्गीकरण करना है
आदेश के समूह $\leq 15$ समरूपता तक. आदेश $1, 2, 3, 5,
7, 11, 13$ लैग्रेंज (चक्रीय), और आदेशों द्वारा तय किए जाते हैं $4$ और
$9$ द्वारा \cref{thm:b3:groups:pfixed} साथ ही नीचे दिया गया विश्लेषण
$ p^2$:वहां बाकी है $6, 8, 10, 12, 14, 15$.

\medskip
\noindent\textbf{Part I --- Tools.}
\begin{enumerate}
  \item वह समूह दिखाएँ जिसमें प्रत्येक तत्व संतुष्ट हो $ x^2 =
        e $ एबेलियन है; यह निष्कर्ष निकालें कि ऐसे सीमित समूह में व्यवस्था है
        $2^k $ और समरूपी है $(\Z/2\Z)^k $. \emph{(इसे ऐसे देखें
        ऊपर एक सदिश स्थान $\mathbb F_2$.)}
  \item दिखाएँ कि आदेश का एक समूह $ p^2$ के लिए समरूपी है
        $\Z/p^2\Z $ या $(\Z/p\Z)^2$. एबेलियन समूहों की सूची बनाएं
        आदेश $8$ समरूपता तक: $\Z/8\Z $, $\Z/4\Z \times
        \Z/2\Z $, $(\Z/2\Z)^3$ --- सिद्ध करें कि सूची पूर्ण है और
        निरर्थक \emph{बिना} की संरचना प्रमेय
        \cref{ch:b3:modules} (ए के अधिकतम क्रम से चर्चा करें
        तत्त्व).
  \item होने देना $\varphi, \varphi' \colon K \to \operatorname{Aut}(H)$
        दो हो \omterm{def:b3:groups:action}{कार्रवाई}. दिखाओ अगर $\varphi' = \varphi \circ
        \alpha $ साथ $\alpha \in \operatorname{Aut}(K)$, तब $ H
        \rtimes_{\varphi} K \cong H \rtimes_{\varphi'} K $.
  \item ठानना $\operatorname{Aut}(\Z/n\Z)$ के लिए $ n = 3, 4, 5,
        7$ स्पष्ट रूप से, और दिखाओ $\operatorname{Aut}\bigl((\Z/2\Z)^2
        \bigr) \cong S_3$.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Orders $2p $ ($6$, $10$, $14$) and
$ pq $.}
\begin{enumerate}[resume]
  \item होने देना $\abs G = 2p $ साथ $ p $ एक अजीब प्रधानमंत्री. बताते हैं कि $ G $ एक \omterm{def:b3:groups:normal}{सामान्य} उपसमूह $ N = \langle r \rangle $ आदेश की $ p $ और
        एक तत्व $ s $ आदेश की $2$ बाहर $ N $.
  \item परिणाम निकालना $ G \cong \Z/p\Z \rtimes_\varphi \Z/2\Z $, कहाँ
        $\varphi(1) \in \operatorname{Aut}(\Z/p\Z)$ एक है
        शामिल होना, और निष्कर्ष निकालना: $ G \cong \Z/2p\Z $ या $ G \cong
        D_p $; जांचें कि ये दोनों समरूपी तो नहीं हैं। यह तय हो गया
        आदेश $6$, $10$, $14$.
  \item अधिक सामान्यतः, चलो $\abs G = pq $ साथ $ p < q $ अभाज्य.
        दिखाओ अगर $ p \nmid q-1$ तब $ G $ चक्रीय है
        (\cref{ex:b3:groups:pq}), और वह यदि $ p \mid q - 1$ वहाँ
        है, इसके अलावा $\Z/pq\Z $, \emph{बिल्कुल एक} नॉनबेलियन समूह
        $\Z/q\Z \rtimes \Z/p\Z $ समरूपता तक --- प्रश्न का प्रयोग करें
        3 और तथ्य यह है कि $\operatorname{Aut}(\Z/q\Z) \cong
        (\Z/q\Z)^\times $ क्रम का चक्रीय है $ q - 1$, यहां भर्ती कराया गया
        और में साबित हुआ \cref{ch:b3:galois} (की चक्रीयता
        $\mathbb F_q^\times $). आदेश के लिए समाप्त करें $15$.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Order $8$.} होने देना $ G $ के नॉनबेलियन बनें
आदेश $8$.
\begin{enumerate}[resume]
  \item बताते हैं कि $ G $ एक तत्व है $ r $ आदेश की $4$ (उपयोग करें
        प्रश्न 1) और वह $ N = \langle r\rangle $ है \omterm{def:b3:groups:normal}{सामान्य}.
  \item होने देना $ s \notin N $. बताते हैं कि $ s^2 \in N $
        (\cref{exo:b3:groups:1}(बी)), वह $ srs^{-1} = r^{-1}$
        (संभावित छवियों की जांच करें $ r $ संयुग्मन के अंतर्गत,
        जिसमें क्रम 4 होना चाहिए, और बहिष्कृत होना चाहिए $ srs^{-1} = r $), ओर वो $ s^2 \in \{e, r^2\}$ \emph{(क्या होता है जब $ s^2 = r $
        या $ r^3$? और क्यों होना चाहिए $ s^2$ के साथ आवागमन करें $ s $?)}.
  \item यदि $ s^2 = e $, दिखाओ $ G \cong D_4$.
  \item यदि $ s^2 = r^2$, दिखाओ कि गुणन
        तालिका पूरी तरह से निर्धारित है; परिणामी समूह है
        \emph{चतुर्भुज समूह}\index{चतुर्भुज समूह} $ Q_8 =
        \{\pm 1, \pm \mathrm i, \pm \mathrm j, \pm \mathrm k\}$,
        $\mathrm i^2 = \mathrm j^2 = \mathrm k^2 = \mathrm i\,
        \mathrm j\,\mathrm k = -1$ (तय करना $ r = \mathrm i $, $ s =
        \mathrm j $). सत्यापित करें कि $ Q_8$ मौजूद है, उदा.\ अंदर
        $ GL_2(\C)$ के जरिए
        \[
        \mathrm i \mapsto \begin{pmatrix} \iu & 0\\ 0 & -\iu
        \end{pmatrix},
        \qquad
        \mathrm j \mapsto \begin{pmatrix} 0 & 1\\ -1 & 0
        \end{pmatrix}.
        \]
  \item दिखाएँ कि प्रत्येक गैर-तुच्छ उपसमूह $ Q_8$ रोकना
        $-1$; यह निष्कर्ष निकालें कि प्रत्येक उपसमूह $ Q_8$ है \omterm{def:b3:groups:normal}{सामान्य}, वह
        $ D_4 \not\cong Q_8$ (क्रम के तत्वों को गिनें $2$), ओर वो $ Q_8$ नहीं है कोई \omterm{def:b3:groups:semidirect}{अर्धप्रत्यक्ष उत्पाद} दो उचित का
        उपसमूह
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Order $12$.} होने देना $\abs G = 12$, $ P_3
\in \mathrm{Syl}_3(G)$, $ P_2 \in \mathrm{Syl}_2(G)$.
\begin{enumerate}[resume]
  \item दिखाओ $ n_3 \in \{1, 4\}$, $ n_2 \in \{1, 3\}$, ओर वो $ n_3
        = 4$ बल $ n_2 = 1$ \emph{(क्रम के तत्वों को गिनें
        $3$)}.
  \item कल्पना करना $ n_3 = 4$. संयुग्मन \omterm{def:b3:groups:action}{कार्रवाई} पर
        $\mathrm{Syl}_3$ देता है $\rho \colon G \to S_4$. बताते हैं कि
        $\ker \rho $, प्रत्येक में समाहित है $ N_G(P_3)$ और इसलिए का
        विभाजन का क्रम $3$, तुच्छ है \emph{(ऐसा क्यों नहीं हो सकता
        आदेश $3$?)}; वह छवि, क्रम का एक उपसमूह $12$ का
        $ S_4$, आवश्यक है $ A_4$ \emph{(सूचकांक 2 उपसमूह हैं
        \omterm{def:b3:groups:normal}{सामान्य} और इसमें सभी वर्ग शामिल हैं ---
        \cref{exo:b3:groups:1}; इसमें वर्ग गिनें $ S_4$)}; और
        निष्कर्ष $ G \cong A_4$.
  \item कल्पना करना $ n_3 = 1$, इसलिए $ G \cong \Z/3\Z \rtimes_\varphi P_2$
        साथ $\varphi \colon P_2 \to \operatorname{Aut}(\Z/3\Z)
        \cong \Z/2\Z $. मामलों की गणना करें: $\varphi $ तुच्छ
        पैदावार $\Z/12\Z $ और $\Z/6\Z \times \Z/2\Z $; $ P_2 =
        \Z/4\Z $ साथ $\varphi $ विशेषण उपज देता है
        \emph{द्विचक्रीय समूह} $\mathrm{Dic}_3 = \Z/3\Z \rtimes
        \Z/4\Z $; $ P_2 = (\Z/2\Z)^2$ साथ $\varphi $ विशेषण
        प्रश्न 3 की समतुल्यता तक, एकल परिणाम प्राप्त होता है
        समूह---दिखाओ यह है $ D_6$, उदा.\ किसी तत्व को प्रदर्शित करके
        क्रम का $6$ और एक प्रतिबिंब जैसा समावेश।
  \item बताते हैं कि $\Z/12\Z $, $\Z/6\Z \times \Z/2\Z $, $ D_6$, $ A_4$,
        $\mathrm{Dic}_3$ जोड़ीवार गैर-आइसोमोर्फिक हैं \emph{(गिनो
        व्यवस्था के तत्व $2$, या उपयोग करें $ n_3$)}. इससे व्यवस्था ठीक हो जाती है
        $12$.
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Synthesis.}
\begin{enumerate}[resume]
  \item वर्गीकरण तालिका को इकट्ठा करें: प्रत्येक आदेश के लिए $ n \leq
        15$, समरूपता तक के समूहों की पूरी सूची, के साथ
        मायने रखता है $1, 1, 1, 2, 1, 2, 1, 5, 2, 2, 1, 5, 1, 2, 1$.
\end{enumerate}

\medskip
\noindent\textbf{Part VI --- Beyond: the groups of order $ p^3$,
$ p $ odd.} आदेश-$8$ भाग III का विश्लेषण सुन्दर है
ऑड-प्राइम एनालॉग, वास्तव में एक नई घटना के साथ। होने देना $ p $ हो
एक अजीब प्रधानमंत्री और $ G $ ऑर्डर का नॉनबेलियन $ p^3$.
\begin{enumerate}[resume]
  \item बताते हैं कि $\abs{Z(G)} = p $, वह $ G/Z(G) \cong
        (\Z/p\Z)^2$ \emph{(द्वारा एक चक्रीय भागफल \omterm{ex:b3:groups:actions}{केंद्र} बल
        एबेलियनिटी: \cref{exo:b3:groups:2})}, ओर वो $ D(G) =
        Z(G)$ \emph{(के लिए $ D(G) \subseteq Z(G)$, उसका उपयोग करें
        $ G/Z(G)$ एबेलियन है; समानता के लिए, $ G $ नॉनबेलियन है और
        $ D(G) \neq \{e\}$)}. यह निष्कर्ष निकालें कि प्रत्येक \omterm{def:b3:groups:derived}{कम्यूटेटर}
        $[x, y] = xyx^{-1}y^{-1}$ केन्द्रीय और व्यवस्थित है
        बाँटना $ p $.
  \item (मुख्य पहचान) चलो $ x, y \in G $ और $ z = [y, x]$,
        \omterm{ex:b3:groups:actions}{केंद्र}ीय. प्रेरण द्वारा सिद्ध करें $ k $:
        \[
        (xy)^k = x^k y^k z^{k(k-1)/2} .
        \]
        \emph{(प्रत्येक को स्थानांतरित करें $ y $ प्रत्येक से आगे $ x $; हर क्रॉसिंग की लागत
        एक \omterm{ex:b3:groups:actions}{केंद्र}ीय कारक $ z $.)}
  \item उसके लिए निष्कर्ष निकालें $ p $ नक्शा अजीब $\theta\colon x \mapsto
        x^p $ से एक समूह रूपवाद है $ G $ को $ Z(G)$ \emph{(क्यों
        $ x^p $ \omterm{ex:b3:groups:actions}{केंद्र}ीय? क्यों करता है $ z^{p(p-1)/2} = e $ ज़रूरत $ p $
        विषम?)}, और निष्कर्ष निकालो $ G $ प्रतिपादक है $ p $ या $ p^2$,
        दोनों मामलों को इस बात से अलग किया जा रहा है कि क्या $\theta $ है
        तुच्छ.
  \item (प्रतिपादक $ p $) मान लीजिए प्रत्येक तत्व संतुष्ट करता है $ x^p = e $.
        चुनना $ x, y $ जिनकी कक्षाएं उत्पन्न होती हैं $ G/Z(G)$ और सेट करें $ z =
        [y, x]$. बताते हैं कि $ z \neq e $, वह प्रत्येक तत्व $ G $
        अद्वितीय है $ x^ay^bz^c $ ($0 \leq a, b, c < p $), और वह
        गुणन पूरी तरह से निर्धारित होता है
        रिश्ते $ x^p = y^p = z^p = e $, $ z $ \omterm{ex:b3:groups:actions}{केंद्र}ीय, $[y, x] =
        z $. सत्यापित करें कि \emph{हाइजेनबर्ग समूह}
        \[
        H_p = \left\{
        \begin{pmatrix} 1 & a & c\\ 0 & 1 & b\\ 0 & 0 & 1
        \end{pmatrix} : a, b, c \in \mathbb F_p \right\}
        \subseteq GL_3(\mathbb F_p)
        \]
        इन संबंधों का एहसास करता है और प्रतिपादक है $ p $ (गणना करें
        $(I + N)^p $ साथ $ N $ सख्ती से ऊपरी त्रिकोणीय, का उपयोग कर
        $ N^3 = 0$ और $ p \geq 3$): प्रत्येक प्रतिपादक-$ p $ नॉनबेलियन
        आदेश का समूह $ p^3$ के लिए समरूपी है $ H_p $.
  \item (प्रतिपादक $ p^2$) मान लीजिए कुछ $ r \in G $ आदेश है $ p^2$,
        और सेट करें $ N = \langle r\rangle $, \omterm{def:b3:groups:normal}{सामान्य} (अनुक्रमणिका $ p $:
        \cref{exo:b3:groups:10}). दिखाओ वहाँ है $ s \notin N $ साथ
        $ s^p = e $ \emph{(कोई भी ले जाओ $ t \notin N $; प्रश्न का उपयोग करना
        20, इसे सही करें: $\theta(t) = t^p \in Z(G) \subseteq N $
        --- औचित्य $ Z(G) = \langle r^p\rangle $ --- और चुनें $ a $
        साथ $ s = tr^{a}$ संतुष्टि देने वाला $ s^p = e $; कहाँ है $ p $ अजीब
        उपयोग किया गया?)}. दिखाओ $ srs^{-1} = r^{1+p}$ बदलने तक $ s $ द्वारा
        एक शक्ति, और निष्कर्ष निकालें: वास्तव में वहाँ है \emph{एक}
        ऑर्डर का नॉनबेलियन समूह $ p^3$ और प्रतिपादक $ p^2$,
        अर्थात् $\Z/p^2\Z \rtimes_\varphi \Z/p\Z $ साथ
        $\varphi(1)\colon r \mapsto r^{1+p}$ \emph{(प्रश्न का प्रयोग करें
        3; $\operatorname{Aut}(\Z/p^2\Z)$ क्रम का चक्रीय है
        $ p(p-1)$, यहां स्वीकार किया गया है, इसलिए इसका एक अद्वितीय उपसमूह है
        आदेश $ p $)}.
  \item गिनती समाप्त करें: विषम के लिए $ p $ बिलकुल हैं $5$
        आदेश के समूह $ p^3$ (तीन एबेलियन, दो नॉनबेलियन),
        बस के लिए के रूप में $ p = 2$ --- लेकिन दो नॉनबेलियन हैं
        अब नहीं $ D_4$ और $ Q_8$. सटीक रूप से इंगित करें कि कहां
        अजीब-$ p $ तर्क के लिए टूट जाता है $ p = 2$: की पहचान में
        प्रश्न 19, $ z^{k(k-1)/2}$ के लिए $ k = p = 2$ है $ z^1 \neq
        e $, इसलिए वर्ग बनाना कोई रूपवाद नहीं है --- और वास्तव में $ Q_8$
        व्यवस्था का एक अनूठा तत्व है $2$ जबकि प्रतिपादक-$4$
        $ D_4$ पाँच हैं.
\end{enumerate}

\medskip
\noindent\textbf{Part VII --- Complements.}
\begin{enumerate}[resume]
  \item के लिए $ p $ विषम, क्रम के तत्वों को गिनें $ p $ प्रत्येक में
        क्रम के दो नॉनबेलियन समूह $ p^3$: बताते हैं कि
        $ H_p $ बिलकुल है $ p^3 - 1$ उनमें से, जबकि $ M_p =
        \Z/p^2\Z \rtimes \Z/p\Z $ बिलकुल है $ p^2 - 1$
        \emph{(रूपवाद का प्रयोग करें $\theta $ प्रश्न 20 का:
        इसकी छवि पहचानें, फिर इसके कर्नेल का क्रम)}.
        के लिए संख्यात्मक रूप से सत्यापित करें $ p = 3$: $26$ ख़िलाफ़ $8$.
        स्पष्ट करें कि इस प्रकार का कोई वर्ग-रूपवाद तर्क क्यों नहीं
        अलग कर सकते हैं $ D_4$ से $ Q_8$, और कौन सी गिनती करता है
        उन्हें अलग करो.
  \item किसी पूर्णांक को कॉल करें $ n \geq 1$ \emph{चक्रीय} यदि प्रत्येक समूह
        क्रम का $ n $ चक्रीय है. दिखाओ अगर $ p^2 \mid n $ के लिए
        कुछ प्रधान $ p $, या यदि $ n $ अभाज्य भाजक हैं $ p < q $
        साथ $ p \mid q - 1$, तब $ n $ चक्रीय नहीं है \emph{(में
        प्रत्येक मामला क्रम के एक गैरचक्रीय समूह को प्रदर्शित करता है $ n $, का उपयोग करना
        दूसरे के लिए भाग II)}. उसका निष्कर्ष निकालो $ n $ चक्रीय बल
        $\gcd(n, \varphi(n)) = 1$, कहाँ $\varphi $ यूलर का है
        टोटिएंट, और प्रश्न 17 की तालिका के सामने जाँच करें:
        बीच में $ n \leq 15$, एकल समूह को ले जाने वाले आदेश
        बिलकुल हैं $ n \in \{1, 2, 3, 5, 7, 11, 13, 15\}$,
        बिल्कुल उन लोगों के साथ $\gcd(n, \varphi(n)) = 1$.
\end{enumerate}
\end{problem}
