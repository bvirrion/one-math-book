\chapter{लेब्सेग इंटीग्रल}\label{ch:b3:lebesgue}

रीमैन का अभिन्न अंग स्लाइस करता है \emph{कार्यक्षेत्र} छोटे में
अंतराल; Lebesgue के टुकड़े \emph{श्रेणी}: संघटित करना
$f $, \omterm{def:b3:measure:measure}{उपाय} सेट $\{f > t\}$. परिवर्तन निर्दोष दिखता है
और क्रांतिकारी है. सीमाएँ और अभिन्न, सदैव झगड़ते रहते हैं
रीमैन सिद्धांत में (समान अभिसरण आवश्यक है!), हैं
तीन अभिसरण प्रमेयों द्वारा सुलझाया गया --- मोनोटोन
अभिसरण, फतौ, प्रभुत्व अभिसरण --- किसकी परिकल्पना
लगभग शर्मनाक रूप से कमज़ोर हैं। यह अध्याय निर्माण करता है
एक मनमाना पर अभिन्न स्थान मापें $(X, \mathcal A,
\mu)$, तीन प्रमेयों को सिद्ध करता है, सटीक समाधान करता है
रीमैन के इंटीग्रल के साथ संबंध (एक परिबद्ध फ़ंक्शन है
रीमैन-अभिन्न यदि यह है \omterm{def:b3:topology:continuity}{निरंतर} \omterm{def:b3:lebesgue:l1}{लगभग हर जगह}), और
पैरामीटर-आश्रित के विभेदन का औद्योगीकरण करता है
इंटीग्रल्स --- वह तकनीक जिसका सप्ताहांत समस्या उपयोग करती है
गणना करें $\int_0^\infty\frac{\sin x}x\,\dd x $ और
$\int_\R \eu^{-x^2}\dd x $.

\section{मापने योग्य कार्य}

\begin{definition}\label{def:b3:lebesgue:measurable}
होने देना $(X, \mathcal A)$, $(Y, \mathcal B)$ मापने योग्य स्थान बनें।
$ f \colon X \to Y $ है \emph{औसत दर्जे का}\index{औसत दर्जे का} अगर $ f^{-1}(B) \in \mathcal A $ हरएक के लिए $ B \in
\mathcal B $. वास्तव में (या $[-\infty,+\infty]$-मूल्यांकित)
कार्य, $ Y = \R $ अपना बोरेल रखता है $\sigma $-बीजगणित, और यह
जाँच करने के लिए पर्याप्त है $ f^{-1}(\intoo t{+\infty}) = \{f > t\} \in
\mathcal A $ सभी के लिए $ t \in \R $: अच्छे सेट $\{B :
f^{-1}(B) \in \mathcal A\}$ फॉर्म ए $\sigma $-बीजगणित
(प्रीइमेज सेट ऑपरेशंस के साथ कम्यूट होती हैं) जिसमें शामिल हैं
किरणें उत्पन्न करना (\cref{def:b3:measure:borel},
\cref{met:b3:measure:goodsets}).
\end{definition}

\begin{proposition}\label{prop:b3:lebesgue:stability}
(ए) की रचनाएँ \omterm{def:b3:lebesgue:measurable}{औसत दर्जे का} मानचित्र हैं \omterm{def:b3:lebesgue:measurable}{औसत दर्जे का}; \omterm{def:b3:topology:continuity}{निरंतर}
मानचित्र हैं बोरेल-मापने योग्य.
(बी) यदि $ f, g \colon X \to \R $ हैं \omterm{def:b3:lebesgue:measurable}{औसत दर्जे का}, तो हैं $ f + g $,
$ fg $, $\max(f,g)$, $\abs f $, $\lambda f $.
(सी) यदि $(f_n)$ हैं \omterm{def:b3:lebesgue:measurable}{औसत दर्जे का} मूल्यों के साथ
$[-\infty, +\infty]$, तब $\sup_nf_n $, $\inf_nf_n $,
$\limsup f_n $, $\liminf f_n $ हैं \omterm{def:b3:lebesgue:measurable}{औसत दर्जे का}; अगर $ f_n \to f $
बिंदुवार, $ f $ है \omterm{def:b3:lebesgue:measurable}{औसत दर्जे का}.
\end{proposition}

\begin{proof}
(ए) $(g\circ f)^{-1}(B) = f^{-1}(g^{-1}(B))$; निरंतरता देता है
मापन योग्यता जनरेटिंग के माध्यम से खुले सेट
(\cref{pb:b3:measure:1}, प्रश्न 10, \omterm{def:b3:groups:normal}{सामान्य} रूप में)।
(बी) $(f, g) \colon X \to \R^2$ है \omterm{def:b3:lebesgue:measurable}{औसत दर्जे का} बोरेल के लिए
$\sigma $-बीजगणित का $\R^2$ --- खुले बक्सों की जाँच करें, जो
उत्पन्न करें ($\R^2$ के खुलेपन तर्कसंगत के गणनीय संघ हैं
बक्से): $(f,g)^{-1}(U\times V) = f^{-1}(U)\cap g^{-1}(V)$ ---
और $+, \times, \max $ हैं \omterm{def:b3:topology:continuity}{निरंतर} $\R^2 \to \R $: लिखें.
(सी) $\{\sup f_n > t\} = \bigcup_n\{f_n > t\}$; $\inf = -\sup(-)$;
$\limsup = \inf_N\sup_{n \geq N}$; एक बिंदुवार सीमा अपनी है
$\limsup $.
\end{proof}

\begin{definition}\label{def:b3:lebesgue:simple}
ए \emph{सरल कार्य}\index{सरल कार्य} एक है मापने योग्य
समारोह परिमित रूप से अनेक मानों के साथ: $ s = \sum_{i=1}^n
c_i\,\mathbf 1_{A_i}$, $ A_i \in \mathcal A $ विच्छेद, $ c_i \geq
0$ (गैर-नकारात्मक सिद्धांत के लिए)। इसका अभिन्न अंग है
\[
\int s\,\dd\mu = \sum_i c_i\,\mu(A_i) \in [0, +\infty]
\]
(सम्मेलन $0\cdot\infty = 0$); मूल्य निर्भर नहीं करता
द \omterm{def:b3:representations:rep}{प्रतिनिधित्व} (दो विभाजनों को परिष्कृत करें)।
\end{definition}

\begin{theorem}[सरल कार्यों द्वारा अनुमान]
\label{thm:b3:lebesgue:approximation}
प्रत्येक \omterm{def:b3:lebesgue:measurable}{औसत दर्जे का} $ f \colon X \to [0, +\infty]$ बिंदुवार है
एक की सीमा \emph{की बढ़ती} का क्रम \omterm{def:b3:lebesgue:simple}{सरल कार्य}:
\[
s_n = \sum_{k=1}^{n2^n} \frac{k-1}{2^n}\,
\mathbf 1_{\{\frac{k-1}{2^n} \leq f < \frac k{2^n}\}}
+ n\,\mathbf 1_{\{f \geq n\}} \nearrow f .
\]
\end{theorem}

\begin{proof}
प्रत्येक $ s_n $ \omterm{def:b3:groups:simple}{सरल} है (सेट बोरेल सेट की पूर्वछवियाँ हैं)।
एकरसता: से गुजरना $ n $ को $ n+1$ प्रत्येक डायडिक को विभाजित करता है
स्तर दो में है और निर्दिष्ट मान (एक बिंदु) कभी कम नहीं होता है
के साथ $\frac{k-1}{2^n} \leq f(x) < \frac k{2^n}$ या तो मिलता है
$\frac{2k-2}{2^{n+1}}$ या $\frac{2k-1}{2^{n+1}}$, दोनों $\geq
\frac{k-1}{2^n}$; द कैप $ n $ भी बढ़ता है)। अभिसरण: यदि
$ f(x) < \infty $, के लिए $ n > f(x)$ हमारे पास है $ f(x) - s_n(x) \leq
2^{-n}$; अगर $ f(x) = \infty $, $ s_n(x) = n \to \infty $.
\end{proof}

\section{अभिन्न और अभिसरण प्रमेय}

\begin{definition}\label{def:b3:lebesgue:integral}
के लिए \omterm{def:b3:lebesgue:measurable}{औसत दर्जे का} $ f \geq 0$:
\[
\int f \,\dd\mu = \sup\Bigl\{\int s\,\dd\mu : s \text{ simple},
\ 0 \leq s \leq f\Bigr\} \in [0, +\infty].
\]
इसमें एकरसता है $ f $ निर्माण द्वारा, और \omterm{def:b3:groups:simple}{सरल} का विस्तार करता है
मामला (\omterm{def:b3:groups:simple}{सरल} के लिए $ f $, पर समर्थन प्राप्त होता है $ f $: की तुलना
\omterm{def:b3:groups:normal}{सामान्य} परिशोधन के माध्यम से \omterm{def:b3:groups:simple}{सरल} इंटीग्रल)।
\end{definition}

\begin{theorem}[मोनोटोन अभिसरण, बेप्पो लेवी]
\label{thm:b3:lebesgue:mct}
अगर $0 \leq f_n \nearrow f $ बिंदुवार (\omterm{def:b3:lebesgue:measurable}{औसत दर्जे का}), तब
\[
\int f_n\,\dd\mu \nearrow \int f\,\dd\mu .
\]
\index{मोनोटोन अभिसरण प्रमेय}
\end{theorem}

\begin{proof}
$ f $ है \omterm{def:b3:lebesgue:measurable}{औसत दर्जे का} (\cref{prop:b3:lebesgue:stability}(सी)) और
$\int f_n $ कुछ तक बढ़ जाता है $ L \leq \int f $ (एकरसता)।
इसके विपरीत, एक \omterm{def:b3:groups:simple}{सरल} ठीक करें $ s = \sum c_i\mathbf 1_{A_i} \leq f $
और $\theta \in (0,1)$; सेट $ E_n = \{f_n \geq \theta s\}$
हैं \omterm{def:b3:lebesgue:measurable}{औसत दर्जे का} और बढ़ाएँ $ X $ (कहाँ $ s(x) > 0$: $ f(x)
\geq s(x) > \theta s(x)$, तो अंततः $ f_n(x) \geq \theta
s(x)$; कहाँ $ s(x) = 0$: तुच्छ रूप से)। तब
\[
\int f_n \geq \int_{E_n}\theta s\,\dd\mu
= \theta\sum_i c_i\,\mu(A_i \cap E_n)
\xrightarrow[n\to\infty]{} \theta\sum_ic_i\,\mu(A_i)
= \theta\int s
\]
द्वारा निरंतरता नीचे की ओर से (\cref{prop:b3:measure:basics}(सी))।
इसलिए $ L \geq \theta\int s $ सभी के लिए $\theta < 1$ और सब कुछ \omterm{def:b3:groups:simple}{सरल} $ s
\leq f $: $ L \geq \int f $.
\end{proof}

\begin{corollary}\label{cor:b3:lebesgue:additivity}
के लिए \omterm{def:b3:lebesgue:measurable}{औसत दर्जे का} $ f, g \geq 0$ और $ c \geq 0$: $\int(f + g) =
\int f + \int g $ और $\int cf = c\int f $; की एक श्रृंखला के लिए
गैरनकारात्मक मापने योग्य कार्य, $\int\sum_nf_n =
\sum_n\int f_n $.
\end{corollary}

\begin{proof}
के लिए \omterm{def:b3:lebesgue:simple}{सरल कार्य}, additivity एक \omterm{def:b3:groups:normal}{सामान्य} पर एक गणना है
परिशोधन. \omterm{def:b3:groups:normal}{सामान्य} तौर पर लीजिए $ s_n \nearrow f $, $ t_n \nearrow g $
(\cref{thm:b3:lebesgue:approximation}): $ s_n + t_n \nearrow f +
g $, और एमसीटी सीमा तक संवेदनशीलता को पार करता है। शृंखला
कथन आंशिक रकम पर एमसीटी लागू होता है।
\end{proof}

\begin{theorem}[फ़तौ की लेम्मा]\label{thm:b3:lebesgue:fatou}
के लिए \omterm{def:b3:lebesgue:measurable}{औसत दर्जे का} $ f_n \geq 0$:
\[
\int \liminf_n f_n \,\dd\mu \;\leq\; \liminf_n \int
f_n\,\dd\mu .
\]
\index{फ़तौ की लेम्मा}
\end{theorem}

\begin{proof}
होने देना $ g_N = \inf_{n\geq N}f_n $: \omterm{def:b3:lebesgue:measurable}{औसत दर्जे का}, $0 \leq g_N \nearrow
\liminf f_n $, और $ g_N \leq f_n $ हरएक के लिए $ n \geq N $, इसलिए
$\int g_N \leq \inf_{n \geq N}\int f_n $. बाईं ओर एमसीटी लगाएं
पक्ष: $\int\liminf f_n = \lim_N\int g_N \leq
\lim_N\inf_{n\geq N}\int f_n = \liminf\int f_n $.
\end{proof}

\begin{definition}\label{def:b3:lebesgue:l1}
ए \omterm{def:b3:lebesgue:measurable}{औसत दर्जे का} $ f \colon X \to \R $ (या $\C $) है
\emph{समाकलनीय}\index{समाकलनीय} अगर $\int\abs
f\,\dd\mu < \infty $; तब $\int f = \int f^+ - \int f^-$
(सकारात्मक और नकारात्मक भाग; वास्तविक और काल्पनिक भाग
जटिल मामला)। समाकलनीय फलनों पर समाकलन रैखिक है
(सकारात्मक भागों को विघटित और पुनः संयोजित करें; जटिल मामला
वास्तविक को कम कर देता है) और संतुष्ट करता है $\abs{\int f} \leq
\int\abs f $ (वास्तविक मामला: $\pm\int f = \int(\pm f) \leq
\int\abs f $; जटिल मामला: एक यूनिमॉड्यूलर स्थिरांक से गुणा करें
अभिन्न को वास्तविक बनाने के लिए)। एक संपत्ति रखती है \emph{लगभग
हर जगह} (ए.ई.)\index{लगभग हर जगह} यदि यह केवल विफल रहता है
ए $\mu $-शून्य सेट; बदलाव $ f $ शून्य सेट पर परिवर्तन संख्या
अभिन्न (अंतर का प्रभुत्व है $\infty\cdot\mathbf
1_N $, अभिन्न का $0$).
\end{definition}

\begin{theorem}[प्रभुत्व अभिसरण]\label{thm:b3:lebesgue:dct}
होने देना $ f_n \to f $ ए.ई., साथ $\abs{f_n} \leq g $ ए.ई.\ एक निश्चित के लिए
\emph{\omterm{def:b3:lebesgue:l1}{समाकलनीय}} $ g $. तब $ f $ है \omterm{def:b3:lebesgue:l1}{समाकलनीय} और
\[
\int f_n\,\dd\mu \longrightarrow \int f\,\dd\mu,
\qquad\text{indeed}\quad \int\abs{f_n - f}\,\dd\mu \to 0 .
\]
\index{प्रभुत्व अभिसरण प्रमेय}
\end{theorem}

\begin{proof}
परिकल्पनाओं को बिंदुवार बनाने के लिए एक शून्य सेट को हटा दें। $\abs f
\leq g $: $ f $ है \omterm{def:b3:lebesgue:l1}{समाकलनीय}. कार्य $ h_n = 2g - \abs{f_n
- f} \geq 0$ संतुष्ट $\liminf h_n = 2g $; फतौ देता है
\[
\int 2g \leq \liminf\int\bigl(2g - \abs{f_n - f}\bigr)
= \int 2g - \limsup\int\abs{f_n - f},
\]
इसलिए $\limsup\int\abs{f_n - f} \leq 0$ (घटाव कानूनी है:
$\int 2g < \infty $). अंत में $\abs{\int f_n - \int f} \leq
\int\abs{f_n - f} \to 0$.
\end{proof}

\begin{method}\label{met:b3:lebesgue:convergence}
का सामना करना पड़ा $\lim_n\int f_n $: प्रयास करें, क्रम में --- (1) है
अनुक्रम मोनोटोन (या गैर-नकारात्मक शब्दों की एक श्रृंखला)? एमसीटी, नहीं
अभिन्नता आवश्यकता है। (2) क्या कोई एकल है \omterm{def:b3:lebesgue:l1}{समाकलनीय}
प्रभुत्व रखने वाला $ g \geq \abs{f_n}$, अपरिष्कृत सीमाओं द्वारा पाया गया
(``$\sup_n $''अनुमान)? डीसीटी. (3) कोई वर्चस्व नहीं, नहीं
एकरसता? फ़तौ अभी भी एक पक्ष को सीमित करता है, और समानता हो सकती है
वास्तव में असफल: बच निकलने वाली टक्कर $ f_n = n\mathbf
1_{\intoo0{1/n}}$ है $\int f_n = 1$ लेकिन $ f_n \to 0$ ए.ई.
वर्चस्व बिल्कुल वही है जो बड़े पैमाने पर भागने से रोकता है
अनंत, लंबवत या क्षैतिज रूप से।
\end{method}

\section{रीमैन बनाम लेब्सग्यू}

\begin{theorem}[लेबेस्ग्यू की कसौटी]
\label{thm:b3:lebesgue:riemann}
होने देना $ f \colon \intcc ab \to \R $ होना \emph{घिरे}. तब $ f $ है
रीमैन-अभिन्न आईएफएफ $ f $ है \omterm{def:b3:topology:continuity}{निरंतर} $\lambda $-लगभग
हर जगह; उस मामले में $ f $ है लेब्सग्यू-अभिन्न और दो
अभिन्न अंग मेल खाते हैं।\index{रीमैन बनाम लेबेसेग इंटीग्रल}
\end{theorem}

\begin{proof}
एक उपविभाग के लिए $\sigma = (a = x_0 < \dots < x_N = b)$, होने देना
$ L_\sigma $ और $ U_\sigma $ प्रत्येक पर चरण कार्य समान हों
$\intoo{x_{i-1}}{x_i}$, को $ m_i = \inf_{[x_{i-1}, x_i]}f $ और
$ M_i = \sup $; डार्बौक्स योग उनके अभिन्न अंग हैं (रीमैन और
लेबेस्ग्यू कदम कार्यों पर सहमत हैं, दोनों दे रहे हैं $\sum
m_i\Delta x_i $). उपविभागों का एक क्रम लीजिए $\sigma_n $,
प्रत्येक जाल का अंतिम शोधन $\to 0$, डार्बौक्स रकम के साथ
के निचले और ऊपरी डार्बौक्स इंटीग्रल्स में परिवर्तित हो रहा है $ f $.
परिशोधन करते हैं $ L_{\sigma_n}$ न घटने वाला और
$ U_{\sigma_n}$ गणनीय सेट से बिंदुवार गैर-वृद्धि
$ D $ सभी विभाजन बिंदुओं का; सीमा बुलाओ $\ell $ और $ u $
(\omterm{def:b3:lebesgue:measurable}{औसत दर्जे का}, \cref{prop:b3:lebesgue:stability}). के लिए $ x \notin
D $, लिखना $ I_n(x)$ खुले के लिए $\sigma_n $-अंतराल
युक्त $ x $: $\ell(x) = \sup_n\inf_{I_n(x)}f $ और $ u(x) =
\inf_n\sup_{I_n(x)}f $; चूँकि जालियां सिकुड़ जाती हैं $0$, ये
निचले और ऊपरी हैं \emph{लिफाफे} का $ f $ पर $ x $
--- $ u(x) - \ell(x)$ का दोलन है $ f $ पर $ x $ --- ताकि \emph{$\ell(x) = u(x)$ आईएफएफ $ f $ है \omterm{def:b3:topology:continuity}{निरंतर} पर $ x $}. एमसीटी/डीसीटी द्वारा (परिमित, परिमित अंतराल):
\[
\int_{\intcc ab}\ell\,\dd\lambda = \lim_n\int L_{\sigma_n}
= \underline{\int}f,
\qquad
\int_{\intcc ab}u\,\dd\lambda = \overline{\int}f .
\]
$ f $ रीमैन-अभिन्न $\iff $ $\underline\int f =
\overline\int f $ $\iff $ $\int(u - \ell) = 0$ $\iff $ $ u = \ell $
ए.ई.\ ($ u - \ell \geq 0$; \cref{exo:b3:lebesgue:5}) $\iff $ $ f $
\omterm{def:b3:topology:continuity}{निरंतर} ए.ई. उस मामले में $\ell \leq f \leq u $ साथ $\ell =
u $ ए.ई.: $ f $ के बराबर है \omterm{def:b3:lebesgue:measurable}{औसत दर्जे का} $\ell $ ए.ई., इसलिए है
लेब्सग्यू-मापन योग्य (पूर्णता का $\lambda $) साथ
$\int f\,\dd\lambda = \int\ell\,\dd\lambda = \underline\int f =
\int_a^bf $.
\end{proof}

\begin{example}\label{ex:b3:lebesgue:riemannexamples}
$\mathbf 1_\Q $ कहीं नहीं है \omterm{def:b3:topology:continuity}{निरंतर}: नहीं रीमैन-अभिन्न
--- लेकिन लेब्सग्यू-तुच्छ: $\int\mathbf 1_\Q\,\dd\lambda =
\lambda(\Q) = 0$. थॉमे का कार्य (\,$\frac1q $ तर्कसंगत पर
$\frac pq $, $0$ अन्यत्र) है \omterm{def:b3:topology:continuity}{निरंतर} बिल्कुल पर
तर्कहीन: रीमैन-अभिन्न अभिन्न के साथ $0$. और
\emph{अनुचित} रीमैन इंटीग्रल्स एक अलग धारणा है:
$\int_0^\infty\frac{\sin x}x\,\dd x $ की सीमा के रूप में अभिसरण होता है
$\int_0^A $ (सप्ताहांत समस्या इसकी गणना करती है $= \frac\pi2$),
लेकिन $\frac{\sin x}x \notin L^1(\intoo0{+\infty})$: पूर्ण
हार्मोनिक श्रृंखला की तरह अभिन्न विचलन
(\cref{exo:b3:lebesgue:6}). लेबेस्ग्यू का सिद्धांत व्यापार करता है
मजबूत सीमा प्रमेयों के लिए सशर्त अभिसरण।
\end{example}

\section{पैरामीटर के साथ इंटीग्रल}

लगातार, $(X, \mathcal A, \mu)$ एक है स्थान मापें, $ T $ ए
मीट्रिक स्थान (पैरामीटर), और $ f \colon T \times X \to \C $
साथ $ f(t, \cdot)$ \omterm{def:b3:lebesgue:l1}{समाकलनीय} प्रत्येक के लिए $ t $; तय करना $ F(t) = \int_X
f(t, x)\,\dd\mu(x)$.

\begin{theorem}[निरंतरता]\label{thm:b3:lebesgue:paramcont}
कल्पना करना: $ t \mapsto f(t,x)$ है \omterm{def:b3:topology:continuity}{निरंतर} पर $ t_0$ ए.ई. के लिए\
$ x $, और वहाँ एक है \omterm{def:b3:lebesgue:l1}{समाकलनीय} $ g $ साथ $\abs{f(t,x)} \leq
g(x)$ सभी के लिए $ t $ में एक \omterm{def:b3:topology:topology}{अड़ोस-पड़ोस} का $ t_0$ और ए.ई.\ $ x $.
तब $ F $ है \omterm{def:b3:topology:continuity}{निरंतर} पर $ t_0$.\index{पैरामीटर इंटीग्रल}
\end{theorem}

\begin{proof}
किसी भी क्रम के लिए $ t_n \to t_0$: $ f(t_n, \cdot) \to f(t_0,
\cdot)$ एई, का प्रभुत्व $ g $: डीसीटी देता है $ F(t_n) \to F(t_0)$;
अनुक्रमिक निरंतरता मीट्रिक रिक्त स्थान में पर्याप्त है
(\cref{rem:b3:topology:sequences}).
\end{proof}

\begin{theorem}[अभिन्न के अंतर्गत भेद]
\label{thm:b3:lebesgue:paramdiff}
होने देना $ T $ का एक खुला अंतराल हो $\R $. मान लीजिए: ए.ई. के लिए\ $ x $,
$ t \mapsto f(t,x)$ पर भिन्न है $ T $, साथ
\[
\Bigl|\frac{\partial f}{\partial t}(t, x)\Bigr| \leq g(x)
\quad \text{for all } t \in T,\ \text{a.e. } x,
\]
$ g $ \omterm{def:b3:lebesgue:l1}{समाकलनीय}. तब $ F $ पर भिन्न है $ T $ साथ $ F'(t)
= \int_X \frac{\partial f}{\partial t}(t, x)\,\dd\mu(x)$.
\end{theorem}

\begin{proof}
हल करना $ t $ और $ h_n \to 0$: अंतर भागफल
\[
\varphi_n(x) = \frac{f(t + h_n, x) - f(t, x)}{h_n}
\longrightarrow \frac{\partial f}{\partial t}(t,x)
\quad\text{a.e.},
\]
और औसत मूल्य असमानता सीमा $\abs{\varphi_n(x)} \leq
\sup_{s}\abs{\partial_tf(s,x)} \leq g(x)$: डीसीटी लागू होता है, और
$\frac{F(t + h_n) - F(t)}{h_n} = \int\varphi_n \to
\int\partial_t f(t, \cdot)$.
\end{proof}

\begin{example}[गामा फ़ंक्शन]\label{ex:b3:lebesgue:gamma}
के लिए $ t > 0$ होने देना\index{गामा फ़ंक्शन}
\[
\Gamma(t) = \int_0^{+\infty} x^{t-1}\eu^{-x}\,\dd x .
\]
अभिन्न अभिसरण: निकट $0$, $ x^{t-1}$ है \omterm{def:b3:lebesgue:l1}{समाकलनीय} ($ t >
0$); अनंत पर, $ x^{t-1}\eu^{-x} \leq C\eu^{-x/2}$.
भागों द्वारा एकीकरण (चालू) $[\varepsilon, A]$, फिर सीमा के माध्यम से
एमसीटी) कार्यात्मक समीकरण देता है $\Gamma(t + 1) =
t\,\Gamma(t)$, कहाँ से $\Gamma(n+1) = n!$: भाज्य
अंतर्वेशित सभी पर $\intcc ab \subseteq \intoo0{+\infty}$,
$\partial_t\bigl(x^{t-1}\eu^{-x}\bigr) = \ln
x\cdot x^{t-1}\eu^{-x}$ का बोलबाला है $\abs{\ln x}(x^{a-1} +
x^{b-1})\eu^{-x}$, \omterm{def:b3:lebesgue:l1}{समाकलनीय}: $\Gamma $ है $\mathcal C^1$, और
प्रेरण द्वारा $\mathcal C^\infty $, साथ $\Gamma^{(k)}(t) =
\int_0^\infty(\ln x)^kx^{t-1}\eu^{-x}\dd x $. मूल्य
$\Gamma(\frac12) = \sqrt\pi $ में गाऊसी अभिन्न अंग है
भेष बदलना (\cref{pb:b3:lebesgue:1}).
\end{example}

\begin{exercise}[$\star\star $]\label{exo:b3:lebesgue:5}

\begin{exercise}[$\star\star $]\label{exo:b3:lebesgue:7}

\begin{exercise}[$\star\star $]\label{exo:b3:lebesgue:9}
होने देना $ f \geq 0$ होना \omterm{def:b3:lebesgue:measurable}{औसत दर्जे का} पर $(X, \mathcal A, \mu)$. बताते हैं कि $\nu(A) = \int_A f\,\dd\mu $ ए को परिभाषित करता है \omterm{def:b3:measure:measure}{उपाय}

\begin{exercise}[$\star\star\star $]\label{exo:b3:lebesgue:10}

(ए) वह दिखाओ $ f $ अच्छी तरह से परिभाषित है और \omterm{def:b3:topology:continuity}{निरंतर} पर $\R $, और
$\mathcal C^1$ साथ $ f'(t) = \int_0^\infty\frac{\cos(tx)}{1 +

\emph{दोबारा} अभिन्न के तहत अवैध है.
(बी) स्वीकार करना $ f'(t) = \frac\pi2\eu^{-t}$ के लिए $ t > 0$ (साबित

$\int_0^\infty\frac{x\sin(tx)}{1+x^2}\dd x $ के लिए $ t > 0$, और

\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:lebesgue:11}
(शेफ़\'ई की लेम्मा) चलो $ f_n, f \geq 0$ होना \omterm{def:b3:lebesgue:l1}{समाकलनीय} साथ
$ f_n \to f $ ए.ई.\ और $\int f_n \to \int f $.
(ए) वह दिखाओ $\int\abs{f_n - f} \to 0$. \emph{(लागू करें
अभिसरण पर प्रभुत्व $ g_n = (f - f_n)^+ \leq f $, और लिखा $\int\abs{f_n - f} = 2\int g_n - \int(f - f_n)$.)}
(बी) उदाहरण द्वारा दिखाएँ कि परिकल्पना $\int f_n \to \int
f $ गिराया नहीं जा सकता (एक फिसलने या ध्यान \omterm{ex:b3:groups:actions}{केंद्र}ित करने वाला उभार), और
कि निष्कर्ष हस्ताक्षरित होने में विफल रहता है $ f_n $ बिना
पूर्ण-मूल्य नियंत्रण: $ f_n = n\mathbf 1_{\intoc0{1/n}} -
n\mathbf 1_{\intoc{-1/n}0}$ है $ f_n \to 0$ ए.ई., $\int f_n
= 0 \to 0$, अभी तक $\int\abs{f_n} = 2$.
(सी) अनुप्रयोग (\omterm{ex:b3:lebesgue:gamma}{घनत्व}): यदि संभाव्यता \omterm{ex:b3:lebesgue:gamma}{घनत्व} $ p_n
\to p $ ए.ई., तो स्वचालित रूप से $\int\abs{p_n - p} \to 0$:
\omterm{ex:b3:lebesgue:gamma}{घनत्व}ों का बिंदुवार अभिसरण है $ L^1$ अभिसरण ---
निःशुल्क एक अभिसरण उन्नयन।
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:lebesgue:12}
एमसीटी/डीसीटी के माध्यम से पूर्ण औचित्य के साथ शास्त्रीय सीमाएँ:
\[
\text{(a)}\ \lim_{n\to\infty}\int_0^n\Bigl(1 -
\frac xn\Bigr)^n\eu^{x/2}\,\dd x, \qquad
\text{(b)}\ \lim_{n\to\infty}\int_0^1\frac{n\,x^{n-1}}{1 +
x}\,\dd x,
\]
\[
\text{(c)}\ \lim_{n\to\infty}\int_0^\infty
\frac{\dd x}{(1 + x/n)^n\,x^{1/n}} .
\]
\emph{(एक के लिए): $(1 - x/n)^n \nearrow \eu^{-x}$ निश्चित के लिए
$ x $ ---के माध्यम से एकरसता सिद्ध करें $\log $; (बी) के लिए,
भागों या स्थानापन्न द्वारा एकीकृत करें $ x = u^{1/n}$ और पहचानो
एक सीमा एकाग्रता; (सी) के लिए, एक खोजें \omterm{def:b3:lebesgue:l1}{समाकलनीय}
डोमिनेटर सभी के लिए मान्य है $ n \geq 2$ पर विभाजित करके $ x =
1$.)}
\end{exercise}

\section{समस्या: दो प्रसिद्ध अभिन्न अंग}

\begin{problem}[{सप्ताहांत समस्या --- गॉसियन इंटीग्रल और
डिरिचलेट का अभिन्न अंग, अकेले मापदंडों द्वारा}]
\label{pb:b3:lebesgue:1}
दो अभिन्न नियम लागू विश्लेषण:
\[
G = \int_{-\infty}^{+\infty}\eu^{-x^2}\dd x = \sqrt\pi,
\qquad
D = \int_0^{+\infty}\frac{\sin x}{x}\,\dd x = \frac\pi2
\]
(दूसरा एक अनुचित अभिन्न अंग के रूप में,
\cref{ex:b3:lebesgue:riemannexamples}). हम दोनों का प्रयोग करके सिद्ध करते हैं
केवल इस अध्याय के उपकरण.

\medskip
\noindent\textbf{Part I --- The Gaussian.} के लिए $ t \geq 0$ तय करना
\[
A(t) = \Bigl(\int_0^t\eu^{-x^2}\dd x\Bigr)^{2},
\qquad
B(t) = \int_0^1\frac{\eu^{-t^2(1 + x^2)}}{1 + x^2}\,\dd x .
\]
\begin{enumerate}
  \item उसे उचित ठहराओ $ A $ और $ B $ हैं $\mathcal C^1$ पर
        $\intoo0{+\infty}$ और गणना करें $ A'$ और $ B'$; दिखाओ
        $ A'(t) + B'(t) = 0$. \emph{(में $ B'$, विकल्प $ u =
        tx $.)}
  \item गणना करना $ A(0) + B(0)$ और $\lim_{t\to+\infty}(A + B)(t)$
        --- इंटीग्रल इन के अंतर्गत सीमा का औचित्य सिद्ध करें $ B $.
  \item निष्कर्ष निकालना $\int_0^\infty \eu^{-x^2}\dd x =
        \frac{\sqrt\pi}2$, इस तरह $ G = \sqrt\pi $, और निष्कर्ष निकालें
        $\Gamma(\tfrac12) = \sqrt\pi $ \emph{(विकल्प $ x =
        u^2$ में $\Gamma(\frac12)$)}.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Dirichlet's integral.} के लिए $ t
\geq 0$ तय करना
\[
F(t) = \int_0^{+\infty}\eu^{-tx}\,\frac{\sin x}{x}\,\dd x .
\]
\begin{enumerate}[resume]
  \item दिखाएँ कि अभिन्न परिभाषा $ F(t)$ के लिए एकत्रित होता है
        प्रत्येक $ t > 0$ लेबेस्ग इंटीग्रल के रूप में, और के लिए $ t = 0$
        एक अनुचित अभिन्न अंग के रूप में; बताते हैं कि $ D =
        \lim_{A\to\infty}\int_0^A\frac{\sin x}x\dd x $ मौजूद है
        \emph{(पर भागों द्वारा एकीकृत करें $[\pi, A]$)}.
  \item बताते हैं कि $ F $ है $\mathcal C^1$ पर
        $\intoo0{+\infty}$ साथ
        \[
        F'(t) = -\int_0^{+\infty}\eu^{-tx}\sin x\,\dd x
        = -\frac{1}{1 + t^2}
        \]
        \emph{(वर्चस्व पर $[t_0, \infty)$ प्रत्येक के लिए $ t_0 >
        0$; भागों द्वारा दो एकीकरणों द्वारा अंतिम अभिन्न अंग या
        जटिल घातांक)}.
  \item दिखाओ $ F(t) \to 0$ जैसा $ t \to +\infty $, और निष्कर्ष निकालें $ F(t)
        = \frac\pi2 - \arctan t $ पर $\intoo0{+\infty}$.
  \item नाजुक बिंदु: $ D = \lim_{t\to0^+}F(t)$. इसे साबित करो
        पूँछ के एकसमान नियंत्रण द्वारा: के लिए $0 \leq t \leq 1$
        और $ A \geq \pi $, दिखाने के लिए भागों द्वारा एकीकृत करें
        \[
        \Bigl|\int_A^{+\infty}\eu^{-tx}\frac{\sin
        x}x\,\dd x\Bigr| \leq \frac{C}{A}
        \]
        साथ $ C $ से स्वतंत्र $ t $ \emph{(अंतर करें
        $\frac{\eu^{-tx}}x $ और बाध्य $\abs{\cos}$ द्वारा $1$;
        टिप्पणी $ t\eu^{-tx} \leq 1/x\cdot(tx\eu^{-tx})$ साथ
        $\sup_{u\geq0}u\eu^{-u} < 1$)}; फिर विभाजित $ F(t) - D $
        में $[0, A]$ (जहां डीसीटी लागू होता है $ t \to 0$) और
        $[A, \infty)$.
  \item निष्कर्ष: $ D = \frac\pi2$.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Dividends.}
\begin{enumerate}[resume]
  \item गणना करना $\int_0^{+\infty}\frac{\sin^2x}{x^2}\,\dd x $
        \emph{(भागों द्वारा एकीकृत करें और कम करें $ D $ के जरिए $\sin
        2x = 2\sin x\cos x $)}.
  \item गणना करना $\int_0^{+\infty}\frac{1 -
        \cos x}{x^2}\,\dd x $, और इसकी स्थिरता की जांच करें
        दो परिणाम.
  \item के लिए $ a > 0$, गणना करें
        $\int_0^{+\infty}\frac{\sin(ax)}x\dd x $ और
        $\int_{-\infty}^{+\infty}\eu^{-ax^2}\dd x $, और रिकॉर्ड करें
        स्केलिंग नियम (वे काम के घोड़े होंगे
        \cref{ch:b3:fouriertransform}).
  \item सटीक कारण बताएं $ D $ इलाज नहीं हो सका
        सीधे डीसीटी द्वारा $ t = 0$ (नहीं \omterm{def:b3:lebesgue:l1}{समाकलनीय} पर प्रभुत्व
        $[0,1]\times[0,\infty)$), और पूंछ-विभाजन क्यों
        प्रश्न 7 ईमानदार विकल्प है --- यह पैटर्न
        (``वर्दी अभिन्नता पूँछों का'') पूरे समय दोहराया जाता है
        विश्लेषण.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The \omterm{ex:b3:lebesgue:gamma}{गामा फ़ंक्शन} according to
Bohr and Mollerup.} समारोह $\Gamma $
(\cref{ex:b3:lebesgue:gamma}) संतुष्ट करता है $\Gamma(1) = 1$ और
$\Gamma(x+1) = x\Gamma(x)$ --- लेकिन ऐसा अपरिमित रूप से कई अन्य लोग भी करते हैं
फ़ंक्शंस (किसी से गुणा करें)। $1$-आवधिक डगमगाहट)। एक उत्तलता
हालत पिन $\Gamma $ नीचे विशिष्ट रूप से, और यह अधिक गहरा है
फिर पहचान यंत्रवत् ख़त्म हो जाती है। एक सकारात्मक कार्य $ f $
एक अंतराल पर है \emph{लॉग-उत्तल} अगर $\log f $ उत्तल है.
\begin{enumerate}[resume]
  \item दिखाएँ कि लॉग-उत्तल का तात्पर्य उत्तल से है, जिसका गुणनफल है
        लॉग-उत्तल फ़ंक्शन और उनकी रचनाएँ एफ़िन के साथ
        मानचित्र लॉग-उत्तल हैं, और --- दो-फ़ंक्शन के माध्यम से
        एच\"पुरानी असमानता $\int\abs{uv} \leq
        \bigl(\int\abs u^p\bigr)^{1/p}\bigl(\int\abs
        v^q\bigr)^{1/q}$, सीधे यंग से साबित हुआ
        असमानता --- वह $\Gamma $ लॉग-उत्तल चालू है
        $\intoo0\infty $.
  \item (ढलान लेम्मा) चलो $ g $ पर उत्तल हो $\intoo0\infty $
        साथ $ g(n+1) - g(n) = \log n $ प्रत्येक पूर्णांक के लिए $ n
        \geq 1$. के लिए $ x \in \intoc01$ और $ n \geq 2$, तुलना करें
        की ढलानें $ g $ ऊपर $[n-1, n]$, $[n, n+x]$ और
        $[n, n+1]$, और निष्कर्ष निकालें
        \[
        x\log(n-1) \;\leq\; g(n + x) - g(n) \;\leq\; x\log n .
        \]
  \item (बोह्र--मोलेरुप) चलो $ f > 0$ संतुष्ट $ f(1) = 1$,
        $ f(x+1) = xf(x)$, और $\log f $ उत्तल. खोलना
        में पुनरावृत्ति $ f(n + x) = x(x+1)\cdots(x + n -
        1)\,f(x)$ और $ f(n) = (n-1)!$, प्रश्न 14 से निष्कर्ष निकालें
        उस के लिए $ x \in \intoc01$
        \[
        f(x) = \lim_{n\to\infty}
        \frac{n!\,n^x}{x(x+1)\cdots(x+n)} :
        \]
        $ f $ अद्वितीय है, इसलिए $ f = \Gamma $, और \emph{गॉस का
        सीमा सूत्र} धारण करता है (सभी तक फैलाओ $ x > 0$ द्वारा
        प्रत्यावर्तन)।
  \item को परिभाषित करो \emph{बीटा फ़ंक्शन} $ B(x, y) =
        \int_0^1t^{x-1}(1-t)^{y-1}\,\dd t $ ($ x, y > 0$). साबित करो
        अभिसरण, प्रत्यावर्तन $ B(x+1, y) =
        \frac{x}{x+y}\,B(x, y)$ (भागों द्वारा एकीकृत), और
        $ B(1, y) = \frac1y $.
  \item बताते हैं कि $ x \mapsto B(x, y)$ लॉग-उत्तल है (H\"पुराना
        फिर से), और बोह्र--मोलेरुप को लागू करें
        \[
        f(x) = \frac{B(x, y)\,\Gamma(x + y)}{\Gamma(y)}
        \]
        समाप्त करने के लिए \emph{यूलर का सूत्र}: $ B(x, y) =
        \dfrac{\Gamma(x)\Gamma(y)}{\Gamma(x+y)}$ --- कोई दोहरा नहीं
        कहीं भी अभिन्न.
  \item गणना करना $ B(\frac12, \frac12)$ सीधे (विकल्प) $ t
        = \sin^2\theta $) और निष्कर्ष निकालें $\Gamma(\frac12) =
        \sqrt\pi $: भाग I का गाऊसी अभिन्न अंग, पुनःप्राप्त
        शुद्ध उत्तलता द्वारा. दोनों प्रमाणों की एक में तुलना करें
        प्रत्येक वाक्य.
  \item (लीजेंड्रे डुप्लिकेशन) वह दिखाओ
        \[
        g(x) = \frac{2^{x-1}}{\sqrt\pi}\,
        \Gamma\Bigl(\frac x2\Bigr)
        \Gamma\Bigl(\frac{x+1}2\Bigr)
        \]
        तीन बोह्र-मोलेरुप परिकल्पनाओं को संतुष्ट करता है, और
        निष्कर्ष $ g = \Gamma $, यानी\ $\Gamma(2z) =
        \frac{2^{2z-1}}{\sqrt\pi}\,\Gamma(z)\,\Gamma(z +
        \tfrac12)$ सभी के लिए $ z > 0$.
  \item \omterm{def:b3:topology:interior}{बंद} प्रपत्र का निष्कर्ष निकालें $\Gamma\bigl(n + \tfrac12\bigr)
        = \dfrac{(2n)!}{4^n\,n!}\sqrt\pi $, और साबित करें, द्वारा
        स्लोप लेम्मा को लागू किया गया $\log\Gamma $ चारों ओर बड़ा
        पूर्णांक, स्पर्शोन्मुख
        \[
        \frac{\Gamma(n + \frac12)}{\Gamma(n)\,\sqrt n}
        \longrightarrow 1 .
        \]
  \item अंतिम दो प्रश्नों को \omterm{ex:b3:groups:actions}{केंद्र}ीय में संयोजित करें
        द्विपद स्पर्शोन्मुखता
        \[
        \binom{2n}{n} \sim \frac{4^n}{\sqrt{\pi n}},
        \]
        और संख्यात्मक रूप से सत्यापित करें $ n = 10$ ($\binom{20}{10} =
        184756$, ख़िलाफ़ $4^{10}/\sqrt{10\pi} \approx
        187079$: अनुपात $\approx 0.988$).
  \item (संश्लेषण) स्थिरांक $\sqrt\pi $ अब के रूप में सामने आया है
        गॉसियन इंटीग्रल (भाग I), जैसे $ B(\frac12,
        \frac12)$ (प्रश्न 18), और अंदर दोहराव
        (प्रश्न 19); \omterm{ex:b3:groups:actions}{केंद्र}ीय द्विपद अनुमान
        दोनों स्टर्लिंग का अनुमान है (\cref{ch:b3:product}का
        सप्ताहांत समस्या) और डी मोइवर--लाप्लास। मैप करें
        कनेक्शन: कौन से कथन किसके समतुल्य हैं,
        और प्रत्येक तकनीक --- भेदभाव के अंतर्गत क्या करती है
        अभिन्न बनाम उत्तलता --- योगदान है कि
        अन्य नहीं कर सकते?
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Three more dividends.}
\begin{enumerate}[resume]
  \item (वालिस, बीटा द्वारा) के लिए $ p > -1$, विकल्प $ t =
        \sin^2\theta $ जाहिर करना
        \[
        W_p = \int_0^{\pi/2}\sin^p\theta\,\dd\theta
        = \frac12\,B\Bigl(\frac{p+1}2, \frac12\Bigr),
        \]
        और बीटा रिकर्सन (प्रश्न 16) से निष्कर्ष निकालें
        $ W_{n+2} = \frac{n+1}{n+2}\,W_n $ पूर्णांकों के लिए $ n \geq
        0$. गणना करना $ W_{2n}$ और $ W_{2n+1}$ \omterm{def:b3:topology:interior}{बंद} रूप में,
        दिखाओ $ W_{2n+1}/W_{2n} \to 1$ निचोड़कर, और निष्कर्ष निकालें
        के साथ \emph{वालिस का उत्पाद}
        \[
        \frac\pi2 = \lim_{n\to\infty}
        \prod_{k=1}^{n}\frac{4k^2}{4k^2 - 1} .
        \]
  \item (गॉसियन एक आवृत्ति से मिलता है) के लिए $ b \in \R $ तय करना
        \[
        \Phi(b) = \int_{-\infty}^{+\infty}
        \eu^{-x^2}\cos(2bx)\,\dd x .
        \]
        बताते हैं कि $\Phi $ है $\mathcal C^1$ पर $\R $, वह एक
        भागों द्वारा एकीकरण से विभेदक समीकरण प्राप्त होता है
        $\Phi'(b) = -2b\,\Phi(b)$, और निष्कर्ष निकालें
        \[
        \Phi(b) = \sqrt\pi\,\eu^{-b^2} :
        \]
        गॉसियन इस परिवर्तन के तहत स्वयं को पुन: उत्पन्न करता है ---
        जिस पर एकल पहचान
        \cref{ch:b3:fouriertransform} दौड़ेगा।
  \item (फ्रुल्लानी का अभिन्न अंग) के लिए $0 < a < b $, बताते हैं कि
        \[
        \int_0^{+\infty}
        \frac{\eu^{-ax} - \eu^{-bx}}{x}\,\dd x
        = \log\frac ba ,
        \]
        पैरामीटर में अंतर करके $ a $ (उचित ठहराएं
        हर एक पर आधिपत्य $\intco{a_0}{+\infty}$, $ a_0 > 0$,
        और देकर स्थिरांक की पहचान करें $ a \to b $).
        वास्तव में इंटीग्रैंड को इसके हटाने योग्य की आवश्यकता कहां है
        विलक्षणता पर $ x = 0$?
\end{enumerate}
\end{problem}
