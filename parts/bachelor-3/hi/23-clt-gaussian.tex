\chapter{विशेषता कार्य और केंद्रीय सीमा
प्रमेय}\label{ch:b3:clt}

बड़ी संख्याओं का नियम कहता है कि औसत अभिसरण होते हैं; \omterm{ex:b3:groups:actions}{केंद्र}ीय
सीमा प्रमेय कहता है \emph{उनमें कैसे उतार-चढ़ाव होता है}: त्रुटि,
द्वारा बढ़ाया गया $\sqrt n $, स्पर्शोन्मुख है \omterm{def:b3:clt:gaussianvector}{गाऊसी} ---
चाहे किसी ने भी किसी भी \omterm{def:b3:probability:space}{कानून} से शुरुआत की हो। यह सार्वभौमिकता है
प्राथमिक संभाव्यता का सबसे गहरा तथ्य, और इसका प्राकृतिक प्रमाण
फूरियर-विश्लेषणात्मक है: \emph{\omterm{def:b3:clt:cf}{विशेषता कार्य}}
(एक \omterm{def:b3:probability:space}{कानून} का फूरियर रूपांतरण) परिवर्तित करता है \omterm{def:b3:probability:independence}{स्वतंत्र} रकम
उत्पादों में, और \cref{ch:b3:fouriertransform}की मशीनरी
--- इंजेक्शन, \omterm{def:b3:clt:gaussianvector}{गाऊसी} निश्चित बिंदु --- बिंदुवार रूपांतरित करता है
इन उत्पादों का \omterm{def:b3:probability:space}{कानून}ों के अभिसरण में अभिसरण
(एल\'ईवी का प्रमेय, पूर्ण रूप से सिद्ध)। अध्याय समाप्त होता है
\omterm{def:b3:clt:gaussianvector}{गाऊसी} सदिश और आत्मविश्वास की ईमानदार व्युत्पत्ति
सांख्यिकी में हर जगह उपयोग किए जाने वाले अंतराल; सप्ताहांत की समस्या
एक स्पष्टता के साथ लिंडेबर्ग को सीएलटी का दूसरा प्रमाण देता है
त्रुटि दर.

\section{चारित्रिक कार्य}

\begin{definition}\label{def:b3:clt:cf}
 \emph{विशेषता कार्य}\index{विशेषता कार्य} एक असली का \omterm{def:b3:probability:space}{अनियमित परिवर्तनशील वस्तु} $ X $ है
\[
\varphi_X(\xi) = \E\bigl[\eu^{\iu\xi X}\bigr]
= \int_\R \eu^{\iu\xi x}\,\dd\P_X(x)
\qquad (\xi \in \R)
\]
(स्थानांतरण प्रमेय इसकी गणना \omterm{def:b3:probability:space}{कानून} से करता है; a के लिए \omterm{ex:b3:lebesgue:gamma}{घनत्व}
$ f $, $\varphi_X(\xi) = \hat f(-\xi)$ में
\cref{ch:b3:fouriertransform}का सम्मेलन)।
\end{definition}

\begin{proposition}\label{prop:b3:clt:cfbasics}
(ए) $\varphi_X(0) = 1$, $\abs{\varphi_X} \leq 1$, और
$\varphi_X $ समान रूप से है \omterm{def:b3:topology:continuity}{निरंतर};
$\varphi_{aX + b}(\xi) = \eu^{\iu b\xi}\varphi_X(a\xi)$.
(बी) यदि $ X, Y $ हैं \emph{\omterm{def:b3:probability:independence}{स्वतंत्र}}:
$\varphi_{X+Y} = \varphi_X\,\varphi_Y $.
(सी) यदि $\E\abs X^k < \infty $, तब $\varphi_X \in \mathcal
C^k $ साथ $\varphi_X^{(j)}(0) = \iu^j\,\E[X^j]$ के लिए $ j \leq
k $; विशेष रूप से, केन्द्रित के लिए $ X \in L^2$ विचरण के साथ
$\sigma^2$:
\[
\varphi_X(\xi) = 1 - \frac{\sigma^2\xi^2}{2} +
o(\xi^2) \qquad (\xi \to 0).
\]
(डी) \omterm{def:b3:clt:gaussianvector}{गाऊसी}: $ X \sim \mathcal N(m, \sigma^2)$ है
$\varphi_X(\xi) = \eu^{\iu m\xi - \sigma^2\xi^2/2}$.
\end{proposition}

\begin{proof}
(ए) सीमाएं तत्काल हैं; निरंतरता: $\abs{\varphi(\xi + h)
- \varphi(\xi)} \leq \E\abs{\eu^{\iu hX} - 1} \to 0$ जैसा $ h
\to 0$ प्रभुत्व अभिसरण द्वारा, समान रूप से में $\xi $. द
एफ़िन नियम एक प्रतिस्थापन है। (बी) $\eu^{\iu\xi(X+Y)} =
\eu^{\iu\xi X}\eu^{\iu\xi Y}$, और अपेक्षाएं उत्पादों का
का \omterm{def:b3:probability:independence}{स्वतंत्र} चर कारक
(\cref{thm:b3:probability:independence}, वास्तविक और पर लागू किया गया
काल्पनिक भाग)। (सी) के तहत भेदभाव \omterm{def:b3:probability:space}{अपेक्षा},
का प्रभुत्व है $\E\abs X^j $
(\cref{thm:b3:lebesgue:paramdiff}); टेलर विस्तार पर
$0$ तो फिर टेलर--यंग के लिए है $\mathcal C^2$ समारोह
$\varphi $. (डी) के लिए $\mathcal N(0,1)$: द \omterm{def:b3:clt:gaussianvector}{गाऊसी} परिवर्तन
(\cref{ex:b3:fouriertransform:gaussian} साथ $ a = \frac12$)
देता है $\int\eu^{\iu\xi x}\frac{\eu^{-x^2/2}}{\sqrt{2\pi}}\dd
x = \eu^{-\xi^2/2}$; एफ़िन नियम द्वारा \omterm{def:b3:groups:normal}{सामान्य} मामला।
\end{proof}

\begin{theorem}[इंजेक्शन]\label{thm:b3:clt:injectivity}
अगर $\varphi_X = \varphi_Y $, तब $ X $ और $ Y $ समान है
\omterm{def:b3:probability:space}{कानून}. अधिक सटीक रूप से, के लिए $ N \sim \mathcal N(0,1)$ \omterm{def:b3:probability:independence}{स्वतंत्र}
का $ X $ और $\varepsilon > 0$, चिकना चर $ X +
\varepsilon N $ के पास है \omterm{ex:b3:lebesgue:gamma}{घनत्व}
\[
p_\varepsilon(x) = \frac1{2\pi}\int_\R
\varphi_X(-\xi)\,\eu^{-\varepsilon^2\xi^2/2}\,
\eu^{\iu\xi x}\,\dd\xi ,
\]
द्वारा निर्धारित $\varphi_X $ अकेला; दे $\varepsilon \to 0$
का \omterm{def:b3:probability:space}{कानून} पुनः प्राप्त करता है $ X $.\index{विशेषता कार्य,
इंजेक्शन}
\end{theorem}

\begin{proof}
$ X + \varepsilon N $ के पास है \omterm{ex:b3:lebesgue:gamma}{घनत्व} $ p_\varepsilon(x) =
\E\bigl[g_\varepsilon(x - X)\bigr]$, कहाँ $ g_\varepsilon $ है
द $\mathcal N(0, \varepsilon^2)$ \omterm{ex:b3:lebesgue:gamma}{घनत्व}: वास्तव में बोरेल के लिए
$ B $, स्वतंत्रता और टोनेली देते हैं $\P(X + \varepsilon N \in
B) = \int\!\!\int\mathbf 1_B(x + \varepsilon
n)g_1(n)\,\dd n\,\dd\P_X(x) = \int_B\E[g_\varepsilon(t -
X)]\dd t $ (स्थानापन्न, फिर टोनेली फिर से)। लिखना
$ g_\varepsilon $ फूरियर द्वारा इसके परिवर्तन का व्युत्क्रमण
(\cref{exo:b3:fouriertransform:4}, पुनर्मूल्यांकित):
$ g_\varepsilon(u) = \frac1{2\pi}\int
\eu^{-\varepsilon^2\xi^2/2}\eu^{\iu\xi u}\dd\xi $, और फ़ुबिनी
(हर चीज़ पर प्रभुत्व है \omterm{def:b3:clt:gaussianvector}{गाऊसी} कारक):
\[
p_\varepsilon(x) = \frac1{2\pi}\int_\R
\varphi_X(-\xi)\,\eu^{-\varepsilon^2\xi^2/2}\,
\eu^{\iu\xi x}\,\dd\xi ,
\]
का एक कार्यात्मक $\varphi_X $ अकेला। अगर $\varphi_X =
\varphi_Y $: $ X + \varepsilon N $ और $ Y + \varepsilon N $ है
सबके लिए समान \omterm{def:b3:probability:space}{कानून} $\varepsilon $; बाध्य के लिए \omterm{def:b3:topology:continuity}{निरंतर}
$ f $, $\E f(X + \varepsilon N) \to \E f(X)$ जैसा $\varepsilon
\to 0$ (प्रभुत्व अभिसरण, $ X + \varepsilon N \to X $
उत्पाद स्थान पर बिंदुवार), इसलिए $\E f(X) = \E f(Y)$ के लिए
ऐसे सभी $ f $ --- और यह \omterm{def:b3:probability:space}{कानून} निर्धारित करता है: प्रत्येक के लिए $ t $,
निचोड़ $\mathbf 1_{\intoc{-\infty}t}$ सीमाबद्ध के बीच
\omterm{def:b3:topology:continuity}{निरंतर} रैंप $ f_k^\pm $ (के बराबर $1$ पर
$\intoc{-\infty}{t \mp \frac1k}$, को $0$ आगे $ t \pm
\frac1k $, बीच में संबंध); में सीमा तक जाना $\E
f_k^-(X) \leq F_X(t) \leq \E f_k^+(X)$ देता है $ F_X(t) =
F_Y(t)$ प्रत्येक पर $ t $ दोनों कहां हैं \omterm{def:b3:topology:continuity}{निरंतर}, इसलिए
हर जगह से सही-निरंतरता और \omterm{ex:b3:lebesgue:gamma}{घनत्व} आम का
निरंतरता अंक (दोनों $ F $की अनगिनत छलाँगें हैं);
समान वितरण कार्य समान \omterm{def:b3:probability:space}{कानून}ों को लागू करते हैं
(\cref{exo:b3:measure:3}, पर निर्भर है
\cref{thm:b3:measure:uniqueness}).
\end{proof}

\section{वितरण में अभिसरण}

\begin{definition}\label{def:b3:clt:cid}
$ X_n $ \emph{वितरण में अभिसरण होता है} (या \omterm{def:b3:probability:space}{कानून} में) को $ X $,
लिखा हुआ $ X_n \Rightarrow X $, अगर\index{में अभिसरण
वितरण}
\[
\E\bigl[f(X_n)\bigr] \longrightarrow \E\bigl[f(X)\bigr]
\qquad\text{for every bounded continuous } f\colon\R\to\R .
\]
समतुल्य (\cref{exo:b3:clt:4}): $ F_{X_n}(t) \to F_X(t)$
प्रत्येक पर निरंतरता बिंदु $ t $ का $ F_X $. $ X_n $ जरूरत नहीं
एक आम पर रहते हैं \omterm{def:b3:probability:space}{संभाव्यता स्थान}: केवल \omterm{def:b3:probability:space}{कानून} मायने रखते हैं।
\end{definition}

\begin{theorem}[हेली का चयन प्रमेय]
\label{thm:b3:clt:helly}
हर क्रम $(F_n)$ वितरण कार्यों का एक है
परिणाम बिंदुवार अभिसरण, प्रत्येक पर निरंतरता बिंदु
सीमा का, न घटने तक सही-निरंतर $ G \colon
\R \to \intcc01$ --- संभवतः साथ $ G(+\infty) - G(-\infty) <
1$ (द्रव्यमान अनंत तक बढ़ सकता है)।\index{हेली का प्रमेय}
\end{theorem}

\begin{proof}
विकर्ण निष्कर्षण देता है $ F_{n_k}(q) \to \ell(q)$ प्रत्येक के लिए
तर्कसंगत $ q $ (में मान कॉम्पैक्ट $\intcc01$). परिभाषित करना
$ G(t) = \inf\{\ell(q) : q \in \Q, q > t\}$: न घटनेवाला;
सही-निरंतर (एक असीम से अधिक सिकुड़ती हुई तर्कसंगतता
पड़ोस दाएं से)। ए पर निरंतरता बिंदु $ t $ का
$ G $:तर्कसंगत के लिए $ q_1 < t < q_2$,
\[
\ell(q_1) \leq \liminf F_{n_k}(t) \leq \limsup F_{n_k}(t)
\leq \ell(q_2),
\]
प्रत्येक की एकरसता से $ F_{n_k}$. की परिभाषा से
$ G $ की अनंत और एकरसता के रूप में $\ell $ पर
तर्कसंगत: $ G(s) \leq \ell(q) \leq G(q)$ जब कभी भी $ s < q $.
लेना $ s < q_1 < t $ देता है $\ell(q_1) \geq G(s)$, और
$\ell(q_2) \leq G(q_2)$; दे $ s \uparrow t $ और $ q_2
\downarrow t $, निरंतरता का $ G $ पर $ t $ दोनों को निचोड़ता है
$\liminf $ और यह $\limsup $ को $ G(t)$.
\end{proof}

\begin{lemma}[चारित्रिक कार्य से जकड़न]
\label{lem:b3:clt:tightness}
किसी के लिए \omterm{def:b3:probability:space}{अनियमित परिवर्तनशील वस्तु} $ X $ और $ u > 0$:
\[
\P\Bigl(\abs X \geq \frac2u\Bigr) \;\leq\;
\frac1u\int_{-u}^{u}\bigl(1 -
\operatorname{Re}\varphi_X(\xi)\bigr)\,\dd\xi .
\]
\end{lemma}

\begin{proof}
टोनेली द्वारा - फ़ुबिनी (अभिन्न सीमाबद्ध, क्षेत्र परिमित
$\xi $):
\[
\frac1u\int_{-u}^u\bigl(1 -
\operatorname{Re}\varphi_X(\xi)\bigr)\dd\xi
= \E\Bigl[\frac1u\int_{-u}^u(1 - \cos(\xi X))\,\dd\xi\Bigr]
= 2\,\E\Bigl[1 - \frac{\sin(uX)}{uX}\Bigr]
\]
(ब्रैकेट को उसकी सीमा के रूप में समझें $0$ पर $ X = 0$). द
इंटीग्रैंड नॉननेगेटिव है ($\abs{\sin t} \leq \abs t $), और के लिए $\abs{uX} \geq 2$: $1 - \frac{\sin(uX)}{uX} \geq 1 -
\frac1{\abs{uX}} \geq \frac12$. केवल घटना को ध्यान में रखते हुए
$\{\abs{uX} \geq 2\}$ के अंदर \omterm{def:b3:probability:space}{अपेक्षा} इसलिए
कम से कम छोड़ देता है $2 \cdot \frac12\,\P(\abs X \geq \frac2u)$,
जो दावा है.
\end{proof}

\begin{theorem}[एल\'ईवी की निरंतरता प्रमेय]
\label{thm:b3:clt:levy}
होने देना $(X_n)$ होना यादृच्छिक चर किसका \omterm{def:b3:clt:cf}{विशेषता
कार्य} बिंदुवार अभिसरण करें: $\varphi_{X_n}(\xi) \to
\varphi(\xi)$ हरएक के लिए $\xi $, कहाँ $\varphi =
\varphi_X $ है \omterm{def:b3:clt:cf}{विशेषता कार्य} कुछ का \omterm{def:b3:probability:space}{अनियमित परिवर्तनशील वस्तु} $ X $. तब $ X_n \Rightarrow X $.\index{Lévy's
continuity theorem}
\end{theorem}

\begin{proof}
\emph{जकड़न.} हल करना $\varepsilon > 0$. तब से $\varphi $ है
\omterm{def:b3:topology:continuity}{निरंतर} पर $0$ साथ $\varphi(0) = 1$, चुनना $ u > 0$ साथ
$\frac1u\int_{-u}^u(1 - \operatorname{Re}\varphi) <
\varepsilon $; प्रभुत्व अभिसरण द्वारा (अभिन्न से घिरा हुआ)।
$2$ निश्चित पर $[-u,u]$), के लिए समान अभिन्न अंग
$\varphi_{X_n}$ है $< 2\varepsilon $ के लिए $ n $ बड़ा:
\cref{lem:b3:clt:tightness} देता है $\P(\abs{X_n} \geq \frac2u)
\leq 2\varepsilon $ बड़े के लिए $ n $, और \omterm{def:b3:ode:stability}{स्थिर}ांक को बड़ा करना
बहुत से अन्य को संभालता है: \omterm{def:b3:probability:space}{कानून} हैं \emph{कसा हुआ}
--- कोई सामूहिक पलायन नहीं.

\emph{परिणाम।} होने देना $(F_{n_k})$ कोई भी परिणाम हो; द्वारा
हेली (\cref{thm:b3:clt:helly}) निकालना $ F_{n_{k_j}} \to G $
पर निरंतरता अंक. जकड़न बल $ G(-\infty) = 0$,
$ G(+\infty) = 1$ ($ G(\frac2u) - G(-\frac2u) \geq 1 -
2\varepsilon $ पर निरंतरता अंक): $ G $ एक असली है
कुछ का वितरण कार्य \omterm{def:b3:probability:space}{अनियमित परिवर्तनशील वस्तु} $ Y $. तब
$ X_{n_{k_j}} \Rightarrow Y $ (\cref{exo:b3:clt:4},
से वितरणात्मक अभिसरण $ F $'s), तो
$\varphi_{X_{n_{k_j}}} \to \varphi_Y $ \emph{बिंदुवार} ($ x
\mapsto \eu^{\iu\xi x}$ घिरा है \omterm{def:b3:topology:continuity}{निरंतर}, असली और
काल्पनिक भाग अलग से); परिकल्पना के साथ तुलना:
$\varphi_Y = \varphi = \varphi_X $, और इंजेक्शन
(\cref{thm:b3:clt:injectivity}) देता है $ Y \sim X $, यानी\ $ G =
F_X $.

\emph{निष्कर्ष।} के हर बाद $(F_n)$ एक है
उप-अनुवर्ती में अभिसरण \emph{वही} $ F_X $ (पर इसके
निरंतरता अंक); इस तरह $ F_n(t) \to F_X(t)$ प्रत्येक पर
निरंतरता बिंदु $ t $ (सभी का एक वास्तविक अनुक्रम
अनुवर्ती में समान सीमा वाले अनुवर्ती होते हैं
अभिसरण): $ X_n \Rightarrow X $.
\end{proof}

\section{केंद्रीय सीमा प्रमेय}

\begin{theorem}[केंद्रीय सीमा प्रमेय]\label{thm:b3:clt:clt}
होने देना $(X_n)$ आई.आई.डी. हो\ साथ $\E X_1 = m $ और $\V(X_1) =
\sigma^2 \in \intoo0\infty $. तब\index{केंद्रीय सीमा प्रमेय}
\[
\frac{S_n - nm}{\sigma\sqrt n} \;\Longrightarrow\; \mathcal
N(0, 1) :
\qquad
\P\Bigl(a \leq \frac{S_n - nm}{\sigma\sqrt n} \leq
b\Bigr) \longrightarrow
\frac{1}{\sqrt{2\pi}}\int_a^b\eu^{-x^2/2}\,\dd x
\]
सभी के लिए $ a < b $.
\end{theorem}

\begin{proof}
\omterm{ex:b3:groups:actions}{केंद्र} और \omterm{def:b3:groups:normal}{सामान्य}ीकृत करें: $ Z_i = \frac{X_i - m}{\sigma}$
(i.i.d., मतलब $0$, विचरण $1$) और $ T_n =
\frac1{\sqrt n}\sum_{i\leq n}Z_i $. द्वारा स्वतंत्रता और
एफ़िन नियम (\cref{prop:b3:clt:cfbasics}):
\[
\varphi_{T_n}(\xi) =
\varphi_{Z}\Bigl(\frac{\xi}{\sqrt n}\Bigr)^{n},
\qquad
\varphi_Z(\eta) = 1 - \frac{\eta^2}2 + \eta^2\rho(\eta),\quad
\rho(\eta)\to0 .
\]
हल करना $\xi $ और चलो $ a_n = \varphi_Z(\xi/\sqrt n)$, $ b_n = 1 -
\frac{\xi^2}{2n}$: दोनों में मापांक है $\leq 1$ के लिए $ n $ बड़ा
($\abs{b_n} \leq 1$ एक बार $\xi^2 \leq 4n $; $\abs{a_n} \leq 1$
हमेशा)। प्राथमिक असमानता $\abs{a^n - b^n} \leq
n\abs{a - b}$ के लिए $\abs a, \abs b \leq 1$ (दूरबीन
$ a^n - b^n = \sum a^k(a - b)b^{n-1-k}$) देता है
\[
\Bigl|\varphi_{T_n}(\xi) - \Bigl(1 -
\frac{\xi^2}{2n}\Bigr)^{n}\Bigr|
\leq n\,\Bigl|\varphi_Z\Bigl(\frac\xi{\sqrt n}\Bigr) - 1 +
\frac{\xi^2}{2n}\Bigr|
= \xi^2\,\Bigl|\rho\Bigl(\frac{\xi}{\sqrt n}\Bigr)\Bigr|
\longrightarrow 0,
\]
जबकि $\bigl(1 - \frac{\xi^2}{2n}\bigr)^n \to
\eu^{-\xi^2/2}$ (वास्तविक लघुगणक). इसलिए $\varphi_{T_n}(\xi) \to
\eu^{-\xi^2/2} = \varphi_{\mathcal N(0,1)}(\xi)$
(\cref{prop:b3:clt:cfbasics}(डी)) प्रत्येक के लिए $\xi $: एल\'ईवी
(\cref{thm:b3:clt:levy}) समाप्त होता है $ T_n \Rightarrow \mathcal
N(0,1)$. अंतराल की संभावनाएँ तब से अनुसरण करती हैं $ F_{\mathcal
N}$ है \omterm{def:b3:topology:continuity}{निरंतर} हर जगह.
\end{proof}

\begin{example}[आत्मविश्वास अंतराल, ईमानदारी से व्युत्पन्न]
\label{ex:b3:clt:confidence}
मतदान $ n $ \omterm{def:b3:probability:independence}{स्वतंत्र} मतदाता; $\hat p_n = S_n/n $ का अनुमान है
सच $ p $, साथ $\sigma^2 = p(1-p) \leq \frac14$. सीएलटी
देता है, बड़े के लिए $ n $,
\[
\P\Bigl(\abs{\hat p_n - p} \leq
\frac{z}{2\sqrt n}\Bigr)
\;\geq\; \P\Bigl(\Bigl|\frac{S_n - np}{\sigma\sqrt n}\Bigr|
\leq z\Bigr)
\longrightarrow \Phi(z) - \Phi(-z),
\]
कहाँ $\Phi $ मानक है \omterm{def:b3:clt:gaussianvector}{गाऊसी} वितरण समारोह।
साथ $ z = 1.96$: स्पर्शोन्मुख आत्मविश्वास $95\%$, और मार्जिन
$\frac{1.96}{2\sqrt n} \leq 3\%$ आवश्यक है $ n \geq
\bigl(\frac{1.96}{0.06}\bigr)^2 \approx 1068$ --- संख्या
हर `` के पीछे $\pm3$ अंक, $95\%$'' एक पढ़ता है; तुलना करें
चेबीशेव का $5556$ (\cref{exo:b3:probability:7}).
$\sqrt n $ सार्वभौमिक है: त्रुटि को आधा करना, चार गुना करना
नमूना --- वही \omterm{def:b3:probability:space}{कानून} जो ठीक करता है मोंटे कार्लोकी लागत
(\cref{exo:b3:clt:7}).
\end{example}

\section{गाऊसी सदिश}

\begin{definition}\label{def:b3:clt:gaussianvector}
एक यादृच्छिक वेक्टर $ X = (X_1, \dots, X_d)$ है
\emph{गाऊसी}\index{गाऊसी} यदि प्रत्येक रैखिक
संयोजन $\langle t, X\rangle = \sum t_iX_i $ एक है
(संभवतः पतित) वास्तविक गाऊसी चर। इसका \omterm{def:b3:probability:space}{कानून} है
माध्य वेक्टर द्वारा निर्धारित $ m = (\E X_i)$ और यह
\emph{सहप्रसरण मैट्रिक्स} $\Sigma = \bigl(\operatorname{Cov}
(X_i, X_j)\bigr)$: वास्तव में \omterm{def:b3:clt:cf}{विशेषता कार्य} का
वेक्टर, $\varphi_X(t) = \E\eu^{\iu\langle t, X\rangle}$, है
पर मूल्य $1$ के सीएफ का $\langle t, X\rangle $:
\[
\varphi_X(t) = \exp\Bigl(\iu\langle t, m\rangle -
\tfrac12\,t^{\mathsf T}\Sigma\,t\Bigr),
\]
और $ d $आयामी विशिष्ट कार्य इंजेक्शन हैं
(समान स्मूथिंग प्रमाण के रूप में \cref{thm:b3:clt:injectivity},
निर्देशांकवार गाऊसी)।
\end{definition}

\begin{theorem}\label{thm:b3:clt:gaussianvector}
होने देना $ X $ एक हो \omterm{def:b3:clt:gaussianvector}{गाऊसी} वेक्टर.
\begin{enumerate}
  \item प्रत्येक एफ़िन छवि $ AX + b $ एक है \omterm{def:b3:clt:gaussianvector}{गाऊसी} वेक्टर.
  \item घटक $ X_i $ हैं \emph{\omterm{def:b3:probability:independence}{स्वतंत्र}} अगर और केवल अगर $\Sigma $ विकर्ण है: संयुक्त रूप से \omterm{def:b3:clt:gaussianvector}{गाऊसी}
        परिवर्तनशील, असंबद्ध $=$ \omterm{def:b3:probability:independence}{स्वतंत्र}.
  \item अगर $\Sigma $ उलटा है, $ X $ के पास है \omterm{ex:b3:lebesgue:gamma}{घनत्व}
        $\frac{1}{(2\pi)^{d/2}\sqrt{\det\Sigma}}
        \exp\bigl(-\frac12(x - m)^{\mathsf T}\Sigma^{-1}(x -
        m)\bigr)$.
\end{enumerate}
\end{theorem}

\begin{proof}
(1) के घटकों का रैखिक संयोजन $ AX + b $ एफ़िन हैं
के रैखिक संयोजनों के कार्य $ X $: \omterm{def:b3:clt:gaussianvector}{गाऊसी} (एक एफ़िन
एक की छवि \omterm{def:b3:clt:gaussianvector}{गाऊसी} परिवर्तनशील है \omterm{def:b3:clt:gaussianvector}{गाऊसी}).
(2) यदि $\Sigma $ विकर्ण है, \omterm{def:b3:clt:cf}{विशेषता कार्य}
कारक: $\varphi_X(t) = \prod_i\exp(\iu t_im_i -
\frac12\Sigma_{ii}t_i^2) = \prod\varphi_{X_i}(t_i)$, वह कौन सा है \omterm{def:b3:clt:cf}{विशेषता कार्य} उत्पाद \omterm{def:b3:probability:space}{कानून} का
(\cref{thm:b3:probability:independence} के माध्यम से पढ़ा
$ d $-आयामी इंजेक्टिविटी): घटक हैं \omterm{def:b3:probability:independence}{स्वतंत्र}.
व्युत्क्रम सहप्रसरणों का लुप्त हो जाना है \omterm{def:b3:probability:independence}{स्वतंत्र}
$ L^2$ चर.
(3) विकर्ण करना $\Sigma = P D P^{\mathsf T}$ ($ P $ ओर्थोगोनल,
$ D > 0$ विकर्ण --- \cref{exo:b3:submanifolds:8}); द
वेक्टर $ Y = P^{\mathsf T}(X - m)$ है \omterm{def:b3:clt:gaussianvector}{गाऊसी} के साथ
सहप्रसरण $ D $: (2) द्वारा इसके घटक हैं \omterm{def:b3:probability:independence}{स्वतंत्र}
$\mathcal N(0, d_i)$, इसलिए $ Y $ उत्पाद है \omterm{ex:b3:lebesgue:gamma}{घनत्व}; धक्का
वॉल्यूम-संरक्षण द्वारा आगे $ x = m + PY $
(\cref{thm:b3:product:linearchange}, $\abs{\det P} = 1$) और
घातांक को अपरिवर्तनीय रूप से पुनः लिखें।
\end{proof}

\begin{theorem}[बहुआयामी सीएलटी]
\label{thm:b3:clt:multiclt}
होने देना $(X_n)$ आई.आई.डी. हो\ वर्ग-अभिन्न यादृच्छिक
\emph{वैक्टर} का $\R^d $ मतलब के साथ $ m $ और सहप्रसरण मैट्रिक्स
$\Sigma $. तब $\frac{S_n - nm}{\sqrt n}$ में एकत्रित हो जाता है
को वितरण \omterm{def:b3:clt:gaussianvector}{गाऊसी} वेक्टर $\mathcal N(0,
\Sigma)$.\index{केंद्रीय सीमा प्रमेय! बहुआयामी}
\admitted
\end{theorem}

\begin{remark}\label{rem:b3:clt:multiclt}
लगभग सब कुछ पहले से ही हमारे हाथ में है। प्रत्येक के लिए
दिशा $ t \in \R^d $, वास्तविक चर $\langle t,
\frac{S_n - nm}{\sqrt n}\rangle $ का एक \omterm{def:b3:groups:normal}{सामान्य}ीकृत योग है
आई.आई.डी.\ विचरण के वास्तविक चर $ t^{\mathsf T}\Sigma t $,
इसलिए \cref{thm:b3:clt:clt}की गणना बिन्दुवार देती है
का अभिसरण $ d $आयामी विशिष्ट कार्य
को $\eu^{-t^{\mathsf T}\Sigma t/2}$, द विशेषता
समारोह का $\mathcal N(0, \Sigma)$
(\cref{def:b3:clt:gaussianvector}). हमारे पास क्या नहीं है
पुनः सिद्ध एल है\'ईवी का निरंतरता प्रमेय \emph{में $\R^d $}:
हेली का चयन और जकड़न का अनुमान \omterm{def:b3:groups:normal}{सामान्य}ीकृत है
नियमित रूप से (समन्वयानुसार), और यह रटना\'एर--वोल्ड
किसी भी स्नातक में कमी ईमानदारी से की जाती है
संभाव्यता पाठ्यक्रम; इस अध्याय की विधियों से परे कुछ भी नहीं है
जरुरत.
\end{remark}

\begin{method}\label{met:b3:clt:method}
एक सीमा \omterm{def:b3:probability:space}{कानून} की पहचान करने के लिए: गणना करें विशिष्ट कार्य,
बिंदुवार सीमा लें, इसे पहचानें (\omterm{def:b3:clt:gaussianvector}{गाऊसी}
$\eu^{-\sigma^2\xi^2/2}$, पॉइसन
$\eu^{\lambda(\eu^{\iu\xi}-1)}$, घातीय $\frac{\lambda}
{\lambda - \iu\xi}$, \dots) और एल का आह्वान करें\'ईवी. तीन कदम
अनुष्ठान (स्वतंत्रता $\to $ उत्पाद; टेलर पर $0$ $\to $
घातीय सीमा; एल\'ईवी $\to $ \omterm{def:b3:probability:space}{कानून} में अभिसरण) सिद्ध होता है
सीएलटी, दुर्लभ घटनाओं का पॉइसन \omterm{def:b3:probability:space}{कानून}
(\cref{exo:b3:clt:5}), और प्रत्येक शास्त्रीय सीमा प्रमेय
यह कोर्स. के लिए के रूप में।\ कथन, पर वापस जाएँ
\cref{ch:b3:probability}का टूलकिट: दो अध्याय उत्तर देते हैं
उसी के बारे में अलग-अलग प्रश्न $ S_n $.
\end{method}

(\omterm{ex:b3:probability:sllnapps}{मोंटे कार्लो} त्रुटि) की सेटिंग में

\begin{exercise}[$\star\star\star $]\label{exo:b3:clt:9}
होने देना $ X \sim \mathcal N(0,1)$ और $\varepsilon $ \omterm{def:b3:probability:independence}{स्वतंत्र}
साथ $\P(\varepsilon = \pm1) = \frac12$; तय करना $ Y =
\varepsilon X $.
(एक शो $ Y \sim \mathcal N(0,1)$ और
$\operatorname{Cov}(X, Y) = 0$.
(बी) दिखाओ $ X $ और $ Y $ हैं \emph{नहीं} \omterm{def:b3:probability:independence}{स्वतंत्र}, ओर वो
$(X, Y)$ नहीं है कोई \omterm{def:b3:clt:gaussianvector}{गाऊसी} वेक्टर \emph{(गणना करें $\P(X + Y =
0)$)}.
(सी) नैतिक: \cref{thm:b3:clt:gaussianvector}(2) आवश्यकता है
संयुक्त गाऊशियनिटी --- ``असंबंधित \omterm{def:b3:clt:gaussianvector}{गाऊसी}'' अकेले
कुछ भी साबित नहीं होता.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:clt:10}
कॉची \omterm{def:b3:probability:space}{कानून} है \omterm{ex:b3:lebesgue:gamma}{घनत्व} $\frac1{\pi(1 + x^2)}$.
(ए) इसे दिखाओ \omterm{def:b3:clt:cf}{विशेषता कार्य} है $\eu^{-\abs\xi}$
(\cref{exo:b3:fouriertransform:1} और उलटा)।
(बी) दिखाओ अगर $ X_1, \dots, X_n $ क्या आई.आई.डी.\ कॉची, फिर
$\frac{S_n}n $ फिर से कॉची है --- \emph{वही} क़ानून: द
औसत कभी ध्यान \omterm{ex:b3:groups:actions}{केंद्र}ित नहीं करता.
(सी) बड़ी संख्या के \omterm{def:b3:probability:space}{कानून}ों और सीएलटी के साथ सामंजस्य स्थापित करें:
कौन सी परिकल्पनाएँ विफल हो जाती हैं? (गणना करें $\E\abs{X_1}$.)
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:clt:11}
(भ्रूण में \omterm{def:b3:ode:stability}{स्थिर} नियम) चलो $(X_n)$ आई.आई.डी. हो\ मानक
कॉची (\cref{exo:b3:clt:10}).
(ए) इसे किसी के लिए दिखाएं $ a, b > 0$, $ aX_1 + bX_2$ \omterm{def:b3:probability:space}{कानून} है
का $(a + b)X_1$: कॉची परिवार है \emph{सख्ती से
\omterm{def:b3:ode:stability}{स्थिर}} सूचकांक का $1$.
(बी) दिखाएँ कि \omterm{def:b3:clt:gaussianvector}{गाऊसी} परिवार सख्ती से \omterm{def:b3:ode:stability}{स्थिर} है
सूचकांक $2$: $ aX_1 + bX_2 \sim \sqrt{a^2 + b^2}\,X_1$ के लिए
$ X_i $ आई.आई.डी.\ $\mathcal N(0,1)$.
(सी) के माध्यम से समझाएं विशिष्ट कार्य रूप का
$\eu^{-c\abs\xi^\alpha}$, क्यों सूचकांक-$\alpha $ स्थिरता
\omterm{def:b3:groups:normal}{सामान्य}ीकरण को बाध्य करता है $ n^{1/\alpha}$ रकम के लिए, और क्या
यह सीएलटी के आकर्षण के \omterm{def:b3:topology:basis}{आधार}ों के बारे में कहता है: जो
आई.आई.डी.\ एफ़िन \omterm{def:b3:groups:normal}{सामान्य}ीकरण के बाद, रकम एक में परिवर्तित हो सकती है
एक के बजाय कौची \omterm{def:b3:probability:space}{कानून} \omterm{def:b3:clt:gaussianvector}{गाऊसी}?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:clt:12}
(अनुभवजन्य वितरण समारोह) चलो $(X_n)$ आई.आई.डी. हो\
वितरण समारोह के साथ $ F $, और $ F_n(t) =
\frac1n\#\{k \leq n : X_k \leq t\}$.
(ए) ठीक करें $ t $. बताते हैं कि $ n F_n(t) \sim \mathcal B(n, F(t))$,
वह $ F_n(t) \to F(t)$ जैसा।\ (\cref{thm:b3:probability:slln}),
ओर वो
\[
\sqrt n\,\bigl(F_n(t) - F(t)\bigr) \Longrightarrow
\mathcal N\bigl(0,\ F(t)(1 - F(t))\bigr) .
\]
(बी) जिस पर $ t $ क्या स्पर्शोन्मुख विचरण अधिकतम है?
व्याख्या करें: माध्यिका वह है जहां अनुभवजन्य वितरण होता है
पिन करना सबसे कठिन।
(सी) के लिए $ F $ \omterm{def:b3:topology:continuity}{निरंतर}, दिखाओ कि का \omterm{def:b3:probability:space}{कानून}
$\sup_t\abs{F_n(t) - F(t)}$ पर निर्भर नहीं है $ F $
\emph{(के माध्यम से एकसमान चर को कम करें
\cref{exo:b3:probability:1})} --- वि\omterm{def:b3:modules:free}{तरण-मुक्त}
कोलमोगोरोव-स्मिरनोव परीक्षण के पीछे का चमत्कार; कोई गणना नहीं
उस \omterm{def:b3:probability:space}{कानून} के बारे में पूछा जाता है.
\end{exercise}

\section{समस्या: लिंडेबर्ग का सीएलटी का प्रमाण, एक के साथ
दर}

\begin{problem}[{सप्ताहांत समस्या --- प्रतिस्थापन विधि}]
\label{pb:b3:clt:1}
लिंडेबर्ग (1922) ने एक विचार द्वारा \omterm{ex:b3:groups:actions}{केंद्र}ीय सीमा प्रमेय को सिद्ध किया
निःशस्त्र करने वाली सरलता का: \emph{सारांश को एक पर स्वैप करें
के लिए समय \omterm{def:b3:clt:gaussianvector}{गाऊसी}} और प्रत्येक स्वैप को टेलर द्वारा नियंत्रित करें
विस्तार. विधि को फूरियर विश्लेषण की आवश्यकता नहीं है, एक उत्पादन करता है
स्पष्ट त्रुटि दर, और आज सार्वभौमिकता प्रमाण को शक्ति प्रदान करता है
संभाव्यता सिद्धांत के पार। होने देना $(X_i)$ आई.आई.डी. होना, केन्द्रित होना,
$\V(X_1) = 1$, साथ $\beta = \E\abs{X_1}^3 < \infty $; होने देना
$(N_i)$ आई.आई.डी. हो\ $\mathcal N(0,1)$, \omterm{def:b3:probability:independence}{स्वतंत्र} की
$ X_i $ (अस्तित्व: \cref{thm:b3:probability:existence}). तय करना
\[
T_n = \frac{X_1 + \dots + X_n}{\sqrt n},
\qquad
G_n = \frac{N_1 + \dots + N_n}{\sqrt n} \sim \mathcal N(0,1).
\]

\medskip
\noindent\textbf{Part I --- The swapping identity.} हल करना $ f
\in \mathcal C^3_b(\R)$ (तीन सीमाबद्ध \omterm{def:b3:topology:continuity}{निरंतर}
व्युत्पन्न; $ M_3 = \sup\abs{f'''}$). के लिए $0 \leq i \leq n $
संकर योगों को परिभाषित करें
\[
H_i = \frac{X_1 + \dots + X_i + N_{i+1} + \dots +
N_n}{\sqrt n},
\]
इसलिए $ H_n = T_n $ और $ H_0 = G_n $.
\begin{enumerate}
  \item लिखना $ H_i = W_i + \frac{X_i}{\sqrt n}$ और $ H_{i-1}
        = W_i + \frac{N_i}{\sqrt n}$ साथ $ W_i =
        \frac{1}{\sqrt n}\bigl(\sum_{j<i}X_j +
        \sum_{j>i}N_j\bigr)$, और उस पर ध्यान दें $ W_i $ है
        \omterm{def:b3:probability:independence}{स्वतंत्र} जोड़ी का $(X_i, N_i)$. औचित्य।
  \item अभिन्न या लैग्रेंज शेष के साथ टेलर: किसी के लिए
        असली $ w, h $:
        \[
        \Bigl|f(w + h) - f(w) - f'(w)h -
        \tfrac12f''(w)h^2\Bigr| \leq
        \frac{M_3\,\abs h^3}{6} .
        \]
  \item प्रश्न 2 को दो बार लागू करें ($ h = \frac{X_i}{\sqrt n}$ और
        $ h = \frac{N_i}{\sqrt n}$ पर $ w = W_i $), लेना
        अपेक्षाएं, और उपयोग करें स्वतंत्रता प्लस मिलान
        के पहले दो क्षणों में से $ X_i $ और $ N_i $ जाहिर करना
        \[
        \bigl|\E f(H_i) - \E f(H_{i-1})\bigr|
        \leq \frac{M_3}{6}\cdot
        \frac{\beta + \gamma}{n^{3/2}},
        \qquad \gamma = \E\abs{N_1}^3 =
        \frac{2\sqrt2}{\sqrt\pi} .
        \]
  \item टेलीस्कोप ख़त्म $ i $ और निष्कर्ष निकालो \emph{लिंडेबर्ग
        बाध्य}:
        \[
        \bigl|\E f(T_n) - \E f(G_n)\bigr| \leq
        \frac{M_3\,(\beta + \gamma)}{6\,\sqrt n} .
        \]
\end{enumerate}

\medskip
\noindent\textbf{Part II --- From smooth $ f $ to the CLT.}
\begin{enumerate}[resume]
  \item बताते हैं कि $\E f(T_n) \to \E f(N)$ हरएक के लिए $ f \in
        \mathcal C_b^3$, और सभी बाउंडेड में अपग्रेड करें
        \omterm{def:b3:topology:continuity}{निरंतर} $ f $: ऐसा दिया गया $ f $ और $\varepsilon $,
        निर्माण $ f_\varepsilon \in \mathcal C^3_b $ साथ
        $\norm{f - f_\varepsilon}_\infty \leq \varepsilon $
        बड़े अंतराल पर --- जैसे\ संघटित $ f $ के साथ
        $\mathcal C^\infty $ उभार
        (\cref{thm:b3:lp:regularization}) --- और संभालो
        मजबूती से पूंछ ($\V(T_n) = 1$ और चेबीशेव)।
        निष्कर्ष निकालना $ T_n \Rightarrow \mathcal N(0, 1)$: द
        \omterm{ex:b3:groups:actions}{केंद्र}ीय सीमा प्रमेय, पुनः सिद्ध।
  \item सबूत का उपयोग कहां किया गया कि $ X_i $ हैं
        \emph{हूबहू} वितरित? दिखाओ कि यह बमुश्किल है
        किया: संस्करण बताएं और सिद्ध करें \omterm{def:b3:probability:independence}{स्वतंत्र},
        केन्द्रित, गैर-समान $ X_i $ साथ
        $\sum_i\V(X_i) = s_n^2$ और तीसरा क्षण, प्राप्त करना
        त्रुटि $\frac{M_3}{6s_n^3}\sum_i\bigl(\E\abs{X_i}^3
        + \V(X_i)^{3/2}\gamma\bigr)$ --- लिंडेबर्ग सच है
        प्रमेय अपने ल्यपुनोव रूप में।
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Quantitative dividends.}
\begin{enumerate}[resume]
  \item (वितरण कार्य) चलो $ t \in \R $ और
        अनुमानित $\mathbf 1_{\intoc{-\infty}t}$ ऊपर और
        नीचे द्वारा $\mathcal C^3_b $ चौड़ाई के रैंप $\delta $
        (उनका निर्माण करें, साथ $ M_3 = O(\delta^{-3})$).
        भाग I के साथ संयोजन करके, दो-अवधि की बाध्यता प्राप्त करें
        \[
        \sup_{t\in\R}\,\bigl|\P(T_n \leq t) -
        \Phi(t)\bigr| \;\leq\;
        \frac{C_1(\beta + \gamma)}{\delta^3\sqrt n} +
        C_2\,\delta
        \qquad (\text{every } \delta > 0),
        \]
        स्पष्ट \omterm{def:b3:ode:stability}{स्थिर}ांक के साथ ( $ C_2\delta $ शब्द का उपयोग
        वह $\Phi $ है \omterm{ex:b3:lebesgue:gamma}{घनत्व} से घिरा
        $\frac1{\sqrt{2\pi}}$), और अनुकूलन करें $\delta \sim
        n^{-1/8}$ ऑर्डर की एक समान दर प्राप्त करने के लिए
        $ n^{-1/8}$.
        (इष्टतम $ n^{-1/2}$ --- बेरी--एसेन ---
        बेहतर उपकरणों की जरूरत है; मुद्दा एक है \emph{मुखर}
        प्राथमिक स्वैपिंग से दर.)
  \item (डी मोइवर--लाप्लास, परिमाणित) में विशेषज्ञता
        $ X_i = 2B_i - 1$ (उचित सिक्कों के लक्षण): तुलना करें
        के स्थानीय अनुमान के साथ निष्कर्ष
        \cref{pb:b3:product:1}, प्रश्न 7 --- क्या करता है
        प्रत्येक विधि वह देती है जो दूसरी नहीं देती?
  \item (सार्वभौमिकता) एक पैराग्राफ में बताएं कि क्यों
        प्रतिस्थापन विधि सीएलटी से अधिक दिखाती है: कोई भी
        प्रपत्र का आँकड़ा $\E f(\text{sum})$ साथ $ f $
        सुचारू क्रम में असंवेदनशील है $ n^{-1/2}$, तक
        \emph{संपूर्ण \omterm{def:b3:probability:space}{कानून}} इसके प्रथम से परे सारांशों का
        दो क्षण --- ``अपरिवर्तनीय सिद्धांत''
        आधुनिक सार्वभौमिकता परिणाम (यादृच्छिक) को रेखांकित करता है
        मैट्रिक्स, यादृच्छिक बहुपद), जिनमें से सीएलटी है
        पहला उदाहरण.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Smoothing, pushed: better
rates.} से हानि $ n^{-1/2}$ (चिकना $ f $) को
$ n^{-1/8}$ (वितरण कार्य) चार्जिंग से आया
$ f'''$ सुपर \omterm{def:b3:rings:ideal}{आदर्श} में. संकर इसके कुछ हिस्से की मरम्मत कर सकते हैं: वे
सम्मिलित \omterm{def:b3:clt:gaussianvector}{गाऊसी} सारांश, और \omterm{def:b3:clt:gaussianvector}{गाऊसी} \emph{चिकना}.
\begin{enumerate}[resume]
  \item (एक छिपा हुआ \omterm{def:b3:clt:gaussianvector}{गाऊसी}) के लिए $1 \leq i \leq n - 1$, $ h =
        \frac{X_i}{\sqrt n}$ या $\frac{N_i}{\sqrt n}$, और
        $\theta \in \intcc01$, लिखना $ W_i + \theta h = A +
        Z $ साथ $ Z = \frac{N_{i+1} + \dots + N_n}{\sqrt
        n}$. बताते हैं कि $ Z \sim \mathcal N\bigl(0,
        \frac{n-i}n\bigr)$ है \omterm{def:b3:probability:independence}{स्वतंत्र} जोड़ी का $(A,
        h)$, और प्रत्येक के लिए निष्कर्ष निकालें \omterm{def:b3:topology:continuity}{निरंतर} $ g \in
        L^1(\R)$,
        \[
        \E\bigl[\abs h^3\,\abs{g(W_i + \theta h)}\bigr]
        \;\leq\; \sqrt{\frac{n}{2\pi(n - i)}}\;
        \norm{g}_{L^1}\;\E\abs h^3 .
        \]
  \item प्रश्न 10 को पूर्णांक रूप के साथ संयोजित करें
        टेलर शेष,
        \[
        f(w + h) = f(w) + f'(w)h + \tfrac12f''(w)h^2 +
        \int_0^1\frac{(1 - \theta)^2}2\,f'''(w + \theta
        h)\,h^3\,\dd\theta,
        \]
        प्रश्न 3--4 को दोबारा करने के लिए: के लिए $ f \in \mathcal C^3_b $
        इसके अलावा के साथ $ f''' \in L^1(\R)$,
        \[
        \bigl|\E f(T_n) - \E f(G_n)\bigr| \leq
        \frac{\beta + \gamma}{3\sqrt{2\pi}}\cdot
        \frac{\norm{f'''}_{L^1}}{\sqrt n} +
        \frac{M_3(\beta + \gamma)}{6\,n^{3/2}}
        \]
        \emph{(प्रश्न 10 स्वैप को संभालता है $ i \leq n - 1$
        --- उपयोग $\sum_{m=1}^{n-1}m^{-1/2} \leq 2\sqrt n $
        --- और प्रश्न 3 का क्रूड बाउंड अंतिम को संभालता है
        एक)}. जांचें कि प्रश्न 7 के रैंप संतुष्ट हैं
        $\norm{\psi_\delta'''}_{L^1} = K_1\delta^{-2}$
        जबकि $ M_3 = K\delta^{-3}$, उन्हें खिलाओ, और
        अनुकूलित करें $\delta $: वर्दी
        वितरण-कार्य दर में सुधार होता है
        $ O(n^{-1/6})$.
  \item (एक क्षण और मिलाते हुए) इसके अतिरिक्त मान लीजिए $\E
        X_1^3 = 0$ और $\beta_4 = \E X_1^4 < \infty $.
        गणना करना $\E N_1^3$ और $\E N_1^4$, चौथे तक विस्तार करें
        आदेश दें, और उसी तर्ज पर सिद्ध करें कि
        वितरण-कार्य दर बन जाती है $ O(n^{-1/4})$
        \emph{(अब $\norm{\psi_\delta^{(4)}}_{L^1} =
        K_2\delta^{-3}$ और $ M_4 = K'\delta^{-4}$; चुनना
        $\delta = n^{-1/4}$)}.
  \item (बाधा) मान लीजिए पहले $ k $ के क्षण
        $ X_1$ से सहमत हूँ \omterm{def:b3:clt:gaussianvector}{गाऊसी} वाले ($ k = 2$ हमेशा;
        $ k = 3$ बिल्कुल कब $\E X_1^3 = 0$; $ k \geq 4$
        अनिवार्य रूप से कभी नहीं, जैसे $\E N_1^4 = 3$). उसे सत्यापित करें
        प्रश्न 10-12 की योजना प्रदान करती है
        वितरण-कार्य दर
        $ n^{-(k-1)/(2k+2)}$, संतुलन बनाकर
        $\delta^{-k}n^{-(k-1)/2}$ ख़िलाफ़ $\delta $, और
        निरीक्षण करें कि घातांक निकट आता है
        बेरी--एसेन मूल्य $\frac12$ केवल के रूप में $ k \to
        \infty $. कुछ वाक्यों में स्पष्ट करें कि ऐसा क्यों है
        स्वैपिंग विधि संतृप्त होती है: प्रत्येक स्वैप में शुल्क लिया जाता है
        पूर्ण मूल्य, जबकि फूरियर मार्ग
        (एस्सेन की असमानता को दूर करना) शोषण करता है
        विशेषता-कार्य का दोलन
        अंतर और पहुंच $ C\beta n^{-1/2}$ तीन के साथ
        केवल क्षण.
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Two dimensions: the
multidimensional CLT, by swapping.} अब चलो $ X_i $ हो
आई.आई.डी.\ यादृच्छिक केन्द्रित \emph{वैक्टर} का $\R^2$ के साथ
सहप्रसरण मैट्रिक्स $\Sigma $ और $\beta' = \E\norm{X_1}^3 <
\infty $ (यूक्लिडियन मानदंड)।
\begin{enumerate}[resume]
  \item (\omterm{def:b3:clt:gaussianvector}{गाऊसी} सदिश, ऑर्डर करने के लिए) विकर्णित करें $\Sigma =
        PDP^{\mathsf T}$ (\cref{exo:b3:submanifolds:8}) और सेट करें $ C = P\sqrt DP^{\mathsf T}$. के लिए $ Z = (Z^1,
        Z^2)$ का एक जोड़ा \omterm{def:b3:probability:independence}{स्वतंत्र} मानक \omterm{def:b3:clt:gaussianvector}{गाऊसी}
        (\cref{thm:b3:probability:existence}), बताते हैं कि $ N
        = CZ $ एक है \omterm{def:b3:clt:gaussianvector}{गाऊसी} वेक्टर
        (\cref{def:b3:clt:gaussianvector}) मतलब का $0$,
        सहप्रसरण $\Sigma $, साथ $\gamma' = \E\norm N^3 <
        \infty $; ओर वो $ G_n = \frac{N_1 + \dots +
        N_n}{\sqrt n}$ \omterm{def:b3:probability:space}{कानून} है $\mathcal N(0, \Sigma)$
        \emph{बिल्कुल} आई.आई.डी. के लिए\ प्रतियां $ N_i $.
  \item (टेलर दो चर में) के लिए $ f \colon \R^2 \to
        \R $ \omterm{def:b3:groups:action}{कक्षा} का $\mathcal C^3$ साथ $ M_3 =
        \max_{\abs\alpha = 3}\sup\abs{\partial^\alpha f} <
        \infty $, सिद्ध करना
        \[
        \Bigl|f(w + h) - f(w) - \langle\nabla f(w),
        h\rangle - \tfrac12\langle h, D^2f(w)\,h\rangle
        \Bigr| \leq \frac{M_3}6\,\bigl(\abs{h_1} +
        \abs{h_2}\bigr)^3 \leq \frac{\sqrt2\,M_3}3\,
        \norm h^3
        \]
        \emph{(अध्ययन $ t \mapsto f(w + th)$ पर
        $\intcc01$)}.
  \item (सीएलटी में $\R^2$) प्रतिस्थापन योजना चालू करें
        वेक्टर संकर $ H_i $: दिखाओ कि पहले- और
        दूसरे क्रम के पद रद्द (साधन और सहप्रसरण)।
        मैच), टेलीस्कोप, और अपग्रेड जैसा कि प्रश्न 5 में है
        (जकड़न से $\E\norm{T_n}^2 =
        \operatorname{tr}\Sigma $; मोलिफिकेशन अब अंदर है
        $\R^2$, \cref{thm:b3:lp:regularization}) को
        निष्कर्ष निकालें: प्रत्येक सीमा के लिए \omterm{def:b3:topology:continuity}{निरंतर} $ f \colon
        \R^2 \to \R $,
        \[
        \E\,f\Bigl(\frac{X_1 + \dots + X_n}{\sqrt n}\Bigr)
        \longrightarrow \E\,f(N), \qquad N \sim \mathcal
        N(0, \Sigma) :
        \]
        \cref{thm:b3:clt:multiclt} आयाम में $2$, एक के साथ
        चिकनी के लिए दर $ f $ और कोई फूरियर विश्लेषण नहीं।
  \item (रत्ता मार\'एर--वोल्ड, और एक संयुक्त उतार-चढ़ाव) घटाना
        वह $\langle t, \frac{S_n}{\sqrt n}\rangle
        \Rightarrow \mathcal N(0, t^{\mathsf T}\Sigma t)$
        प्रत्येक निश्चित के लिए $ t \in \R^2$. आवेदन: के लिए
        आई.आई.डी.\ असली $(\xi_i)$, \omterm{ex:b3:groups:actions}{केंद्र}ित, $\E\xi_1^2 = 1$,
        $\E\xi_1^6 < \infty $ (ताकि भाग V लागू हो
        $ V_i = (\xi_i, \xi_i^2 - 1)$), दिखाओ
        \[
        \frac1{\sqrt n}\Bigl(\sum_{i\leq n}\xi_i,\
        \sum_{i\leq n}(\xi_i^2 - 1)\Bigr) \Longrightarrow
        \mathcal N\Bigl(0, \begin{pmatrix} 1 & \E\xi_1^3\\
        \E\xi_1^3 & \E\xi_1^4 - 1\end{pmatrix}\Bigr) :
        \]
        अनुभवजन्य माध्य और अनुभवजन्य दूसरा क्षण
        संयुक्त रूप से गॉसियन रूप से उतार-चढ़ाव --- \omterm{def:b3:probability:independence}{स्वतंत्र} रूप से
        सीमा यदि और केवल यदि $\E\xi_1^3 = 0$
        (\cref{thm:b3:clt:gaussianvector}).
\end{enumerate}

\medskip
\noindent\textbf{Part VI --- The delta method.}
\begin{enumerate}[resume]
  \item होने देना $(\hat\theta_n)$ होना यादृच्छिक चर साथ
        $\sqrt n(\hat\theta_n - \theta) \Rightarrow
        \mathcal N(0, \sigma^2)$ एक वास्तविक पैरामीटर के लिए
        $\theta $, और चलो $ g $ पर भिन्न होना
        $\theta $. सिद्ध करो \emph{डेल्टा विधि}:
        \[
        \sqrt n\bigl(g(\hat\theta_n) - g(\theta)\bigr)
        \Longrightarrow \mathcal N\bigl(0,
        g'(\theta)^2\sigma^2\bigr)
        \]
        \emph{(लिखना $ g(x) - g(\theta) = (g'(\theta) +
        \eta(x))(x - \theta)$ साथ $\eta \to 0$ पर
        $\theta $; दिखाओ $\hat\theta_n \to \theta $, तब
        $\eta(\hat\theta_n) \to 0$, संभाव्यता में; ख़त्म करो
        स्लटस्की के साथ, \cref{exo:b3:clt:8}, और
        \cref{exo:b3:clt:4}(बी))}.
  \item अनुप्रयोग। (ए) आई.आई.डी. के लिए\ असली $(\xi_i)$ के साथ
        मतलब $\mu $ और विचरण $\sigma^2$, और $\bar X_n =
        \frac1n\sum_{i\leq n}\xi_i $: दिखाओ $\sqrt n(\bar
        X_n^2 - \mu^2) \Rightarrow \mathcal N(0,
        4\mu^2\sigma^2)$ कब $\mu \neq 0$, और वह के लिए
        $\mu = 0$ सही कथन दूसरे स्थान पर रहता है
        पैमाना: $ n\bar X_n^2 \Rightarrow \sigma^2N^2$ साथ
        $ N \sim \mathcal N(0,1)$ (सीमा की पहचान करें
        वितरण समारोह)। (बी) (विचरण)
        \omterm{def:b3:ode:stability}{स्थिर}ीकरण) के लिए $\hat p_n $ सफलता की आवृत्ति
        एक का $\mathcal B(1, p)$ नमूना, $ p \in \intoo01$:
        बताते हैं कि $ g(p) = \arcsin\sqrt p $ संतुष्ट
        \[
        \sqrt n\,\bigl(g(\hat p_n) - g(p)\bigr)
        \Longrightarrow \mathcal N\Bigl(0, \frac14\Bigr)
        \]
        \emph{जो कुछ भी} $ p $ --- एक स्पर्शोन्मुख त्रुटि पट्टी
        अज्ञात पैरामीटर से \omterm{def:b3:modules:free}{मुक्त}; साथ तुलना करें
        \cref{ex:b3:clt:confidence}.
\end{enumerate}

\medskip
\noindent\textbf{Part VII --- Poisson, by the same method:
Le Cam's theorem.} प्रतिस्थापन एक दूसरी सार्वभौमिकता जानता है
वर्ग: अनेक का योग \omterm{def:b3:probability:independence}{स्वतंत्र} \emph{दुर्लभ} घटनाएँ. के लिए
पर \omterm{def:b3:probability:space}{कानून} $\N $ सही दूरी है \emph{कुल भिन्नता},
\[
d_{\mathrm{TV}}(\mu, \nu) = \sup_{A\subseteq\N}\,
\abs{\mu(A) - \nu(A)} .
\]
\begin{enumerate}[resume]
  \item बताते हैं कि $ d_{\mathrm{TV}}(\mu, \nu) =
        \frac12\sum_{k\geq0}\abs{\mu(\{k\}) -
        \nu(\{k\})}$, और युग्मन को बाध्य साबित करें: के लिए
        \emph{कोई} जोड़ा $(X, Y)$ का यादृच्छिक चर के साथ
        \omterm{def:b3:probability:space}{कानून} $\mu $ और $\nu $ उसी जगह पर,
        $ d_{\mathrm{TV}}(\mu, \nu) \leq \P(X \neq Y)$.
  \item सटीक रूप से गणना करें, के लिए $ p \in \intoo01$:
        \[
        d_{\mathrm{TV}}\bigl(\mathcal B(1, p), \mathcal
        P(p)\bigr) = p\bigl(1 - \eu^{-p}\bigr) \leq p^2 .
        \]
  \item (ले कैम, अदला-बदली करके) चलो $ X_i \sim \mathcal B(1,
        p_i)$ और $ Y_i \sim \mathcal P(p_i)$, द $2n $
        चर \omterm{def:b3:probability:independence}{स्वतंत्र}; $ S = X_1 + \dots + X_n $, और
        स्मरण करो $ Y_1 + \dots + Y_n \sim \mathcal P(\lambda)$
        साथ $\lambda = \sum_ip_i $ (\cref{exo:b3:clt:1}).
        एक समय में एक निर्देशांक को पूर्णांक में स्वैप करें
        संकर $ H_i = Y_1 + \dots + Y_i + X_{i+1} + \dots
        + X_n $: दिखाएँ, प्रत्येक के लिए $ A \subseteq \N $,
        \[
        \abs{\P(H_{i-1} \in A) - \P(H_i \in A)} \leq
        d_{\mathrm{TV}}\bigl(\mathcal B(1, p_i), \mathcal
        P(p_i)\bigr),
        \]
        और निष्कर्ष निकालो \emph{ले कैम की असमानता}:
        \[
        d_{\mathrm{TV}}\bigl(\text{law of } S,\ \mathcal
        P(\lambda)\bigr) \leq \sum_{i=1}^np_i^2 .
        \]
  \item लाभांश. (ए) के लिए $ p_i = \frac\lambda n $: द
        बाध्य है $\frac{\lambda^2}n $ --- दुर्लभ का नियम
        घटनाएँ (\cref{exo:b3:clt:5}) को एक में अपग्रेड किया गया
        स्पष्ट दर, सभी घटनाओं पर एकसमान और वैध
        असमान के लिए $ p_i $ भी। (बी) $500$ पत्र हैं
        वितरित, प्रत्येक \omterm{def:b3:probability:independence}{स्वतंत्र} रूप से भटक रहा है
        संभाव्यता $\frac1{500}$: की त्रुटि को सीमित करें
        पैरामीटर का पॉइसन मॉडल $1$, और अनुमान लगाएं
        संभावना है कि कोई भी पत्र भटक न जाए। (सी) \omterm{def:b3:topology:interior}{बंद} करें
        समस्या: दो सार्वभौमिकता वर्गों की तुलना करें
        यहाँ मिले --- \omterm{def:b3:clt:gaussianvector}{गाऊसी} (कई छोटे फैले हुए
        योगदान; दो क्षण मेल खाते हैं; टेलर) और
        पॉइसन (कई दुर्लभ योगदान; एक मतलब
        मिलान किया हुआ; एक सटीक कुल-भिन्नता युग्मन) --- और
        दोनों के पीछे एकल प्रतिस्थापन विधि।

  \item (सापेक्ष त्रुटि और लॉग परिवर्तन) चलो $(X_n)$
        आई.आई.डी. हो, सकारात्मक, मतलबी $\mu > 0$, विचरण
        $\sigma^2$, और $\bar X_n $ अनुभवजन्य माध्य. दिखाओ
        डेल्टा विधि द्वारा
        \[
        \sqrt n\,\bigl(\ln\bar X_n - \ln\mu\bigr)
        \Longrightarrow
        \mathcal N\Bigl(0,\ \frac{\sigma^2}{\mu^2}\Bigr) :
        \]
        का स्पर्शोन्मुख पैरामीटर $\ln\bar X_n $ है
        \emph{गुणांक का परिवर्तन} $\sigma/\mu $ ---
        सापेक्ष, स्के\omterm{def:b3:modules:free}{ल-मुक्त} त्रुटि। एक निष्कर्ष निकालें $95\%$
        के लिए विश्वास अंतराल $\mu $ गुणक का
        प्रपत्र $\bar X_n\cdot\eu^{\pm1.96\,\sigma/(\mu\sqrt
        n)}$, और समझाएं कि यह कब बेहतर है
        योगात्मक एक.
  \item (तीसरा क्षण त्रुटि को दूर करता है) के लिए $ X \sim $
        बरनौली($ p $) केन्द्रित, गणना करें $\E\bigl[(X -
        p)^3\bigr] = p(1-p)(1-2p)$. भाग IV के विश्लेषण का उपयोग करना
        (स्वैप त्रुटि तीसरे क्षण से प्रेरित होती है), समझाएं
        का \omterm{def:b3:groups:normal}{सामान्य} सन्निकटन क्यों $\mathcal B(n, p)$
        के लिए असममित है $ p \neq \frac12$ --- ओवरशूटिंग
        एक तरफ, दूसरी तरफ अंडरशूटिंग --- और क्यों
        $ p = \frac12$ तेज़ मिलान-क्षण दर का आनंद लेता है।
        संख्यात्मक रूप से तिरछा चिह्न सत्यापित करें $\mathcal
        B(20, 0.1)$ ख़िलाफ़ $\mathcal N(2, 1.8)$: तुलना करना
        $\P(S = 0) = 0.9^{20}$ साथ \omterm{def:b3:clt:gaussianvector}{गाऊसी} का समूह
        $\intoo{-\infty}{0.5}$.
\end{enumerate}
\end{problem}
