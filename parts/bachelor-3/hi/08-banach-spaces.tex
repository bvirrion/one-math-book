\chapter{बानाच स्पेस और मौलिक प्रमेय}
\label{ch:b3:banach}

कार्यात्मक विश्लेषण अनंत-आयामी मानक स्थानों का अध्ययन करता है
उन पर रहने वाले ऑपरेटरों और कार्यात्मकताओं के माध्यम से। यह
संस्थापक खोज वह है \emph{पूर्णता}, बेयर के माध्यम से
प्रमेय, मजबूत एकरूपता को बल देता है: बिंदुवार बंधे हुए परिवार
ऑपरेटर मानक-बद्ध हैं (बानाच--स्टीनहॉस), \omterm{def:b3:topology:continuity}{निरंतर}
आपत्तियां हैं \omterm{def:b3:topology:continuity}{निरंतर} व्युत्क्रम (खुला मानचित्रण), और ग्राफ़
पता लगाएं निरंतरता (\omterm{def:b3:topology:interior}{बंद} ग्राफ़). दूसरा स्तंभ,
\emph{हाहन--बनाच}, जरूरत नहीं पूर्णता बिलकुल --- केवल
ज़ोर्न की लेम्मा --- और इसकी गारंटी देती है दोहरे स्थान काफी अमीर हैं
प्रत्येक वेक्टर को देखने के लिए. यह अध्याय सभी चार प्रमेयों को सिद्ध करता है और
शास्त्रीय अनुक्रम स्थानों पर उनका परीक्षण करता है $\ell^p $, पर
ठोस \omterm{def:b3:banach:operator}{दोहरी} गणनाएँ, और वास्तव में आश्चर्यजनक
आवेदन: वहाँ मौजूद हैं \omterm{def:b3:topology:continuity}{निरंतर} $2\pi $-आवधिक कार्य
जिसकी फूरियर श्रृंखला \emph{विचलन} एक बिंदु पर --- समाधान हो रहा है
नकारात्मक एक प्रश्न वर्ष 2 खुला छोड़ दिया गया।

लगातार, $ E, F $ ऊपर मानक स्थान हैं $ K = \R $ या $\C $;
\emph{बनच स्थान}\index{बनच स्थान} मतलब \omterm{def:b3:complete:complete}{पूरा} मानकीकृत
अंतरिक्ष.

\section{बंधे हुए ऑपरेटर; अनुक्रम स्थान}

\begin{definition}\label{def:b3:banach:operator}
$\mathcal L(E, F)$ के स्थान को दर्शाता है \emph{घिरे}
(= \omterm{def:b3:topology:continuity}{निरंतर}, वर्ष 2) के साथ रेखीय मानचित्र \emph{ऑपरेटर
आदर्श}\index{ऑपरेटर
आदर्श} $\vertiii T = \sup_{\norm x \leq
1}\norm{Tx}$; यह उपगुणक है: $\vertiii{ST} \leq
\vertiii S\,\vertiii T $. \emph{दोहरी}\index{दोहरी} है
$ E' = \mathcal L(E, K)$.
\end{definition}

\begin{proposition}\label{prop:b3:banach:llcomplete}
अगर $ F $ एक बानाच स्थान है, ऐसा ही है $\mathcal L(E, F)$; विशेष रूप से $ E'$ हमेशा एक बनच स्थान होता है।
\end{proposition}

\begin{proof}
होने देना $(T_n)$ के लिए कॉची बनें $\vertiii\cdot $. प्रत्येक के लिए $ x $,
$\norm{T_nx - T_mx} \leq \vertiii{T_n - T_m}\norm x $: $(T_nx)$
कॉची में है $ F $, अभिसरण; सीमा को कॉल करें $ Tx $. $ T $ है
रैखिक (रैखिक पहचान की सीमा); में सीमा तक जाना
$\norm{T_nx - T_mx} \leq \varepsilon\norm x $ ($ n, m \geq N $)
देता है $\norm{T_nx - Tx} \leq \varepsilon\norm x $: $ T_n \to T $
में ऑपरेटर मानदंड, और $\vertiii T \leq \vertiii{T_N} +
\varepsilon < \infty $.
\end{proof}

\begin{definition}\label{def:b3:banach:lp}
शास्त्रीय अनुक्रम स्थान\index{sequence spaces $\ell^p $}
(ऊपर $ K $, द्वारा अनुक्रमित $\N $):
\[
\ell^p = \Bigl\{x = (x_n) : \norm x_p = \Bigl(\sum_n
\abs{x_n}^p\Bigr)^{1/p} < \infty\Bigr\}\quad (1 \leq p <
\infty),
\]
\[
\ell^\infty = \{x : \norm x_\infty = \sup\abs{x_n} < \infty\},
\]
और $ c_0 = \{x : x_n \to 0\}$ साथ $\norm\cdot_\infty $. वह
$\norm\cdot_p $ यह मिन्कोव्स्की असमानता का अनुसरण करने वाला एक मानदंड है,
असतत मामले में बिल्कुल वैसा ही साबित हुआ \cref{ch:b3:lp} (या
वर्ष 2 की परिमित-आयामी असमानता को जोड़कर)। सब
बनच स्थान हैं, और $ c_0$ का एक \omterm{def:b3:topology:interior}{बंद} उपस्थान है
$\ell^\infty $ (\cref{exo:b3:banach:3}).
\end{definition}

\begin{proposition}[न्यूमैन श्रृंखला]\label{prop:b3:banach:neumann}
होने देना $ E $ बानाच हो और $ T \in \mathcal L(E) = \mathcal L(E,E)$
साथ $\vertiii T < 1$. तब $ I - T $ में उलटा है $\mathcal
L(E)$, साथ $(I - T)^{-1} = \sum_{n\geq0}T^n $ (में अभिसरण)
ऑपरेटर मानदंड). फलस्वरूप व्युत्क्रमणीय संचालकों का समुच्चय है
खुला है, और उलटा है \omterm{def:b3:topology:continuity}{निरंतर} उस पर.\index{न्यूमैन श्रृंखला}
\end{proposition}

\begin{proof}
श्रृंखला बिल्कुल एकाग्र होती है ($\vertiii{T^n} \leq \vertiii
T^n $, ज्यामितीय) बनच में $\mathcal L(E)$
(\cref{prop:b3:banach:llcomplete};
\cref{exo:b3:complete:1}(बी))। टेलीस्कोपिंग, $(I - T)\sum_{n \leq
N}T^n = I - T^{N+1} \to I $, और इसी तरह दूसरी तरफ भी।
खुलेपन के लिए: यदि $ S $ उलटा है और $\vertiii{H} <
1/\vertiii{S^{-1}}$, तब $ S + H = S(I + S^{-1}H)$ साथ
$\vertiii{S^{-1}H} < 1$: उलटा. निरंतरता उलटा का:
श्रृंखला अभिव्यक्ति देती है $\vertiii{(S+H)^{-1} - S^{-1}} =
O(\vertiii H)$ स्थानीय स्तर पर.
\end{proof}

\begin{example}[न्यूमैन द्वारा हल किया गया वोल्टेरा समीकरण]
\label{ex:b3:banach:volterraexample}
पर $ E = \mathcal C(\intcc01)$, अभिन्न समीकरण पर विचार करें
\[
u(x) = 1 + \lambda\int_0^xu(t)\,\dd t,
\qquad\text{i.e.}\qquad u = \mathbf 1 + \lambda Tu,
\quad (Tu)(x) = \int_0^xu .
\]
यहाँ $\vertiii T \leq 1$, अभीतक के लिए तो $\abs\lambda < 1$  \omterm{prop:b3:banach:neumann}{न्यूमैन}
शृंखला सीधे लागू होता है: $ u = (I - \lambda T)^{-1}\mathbf 1 =
\sum_n\lambda^nT^n\mathbf 1$. कंप्यूटिंग, $ T^n\mathbf 1 =
\frac{x^n}{n!}$, इसलिए
\[
u(x) = \sum_{n\geq0}\frac{(\lambda x)^n}{n!} =
\eu^{\lambda x} ,
\]
जैसा कि विभेदन पुष्टि करता है। बेहतर: $\vertiii{T^n} \leq
\frac1{n!}$ (पुनरावृत्त कर्नेल तथ्यात्मक रूप से सिकुड़ता है), इसलिए
$\sum\lambda^nT^n $ के लिए एकत्रित होता है \emph{प्रत्येक} $\lambda $ ---
संचालिका $ I - \lambda T $ सभी के लिए उलटा है $\lambda
\in \C $, चाहे $\vertiii{\lambda T} \geq 1$ अंततः:
जो मायने रखता है वह शक्तियों का वर्णक्रमीय क्षय है, न कि शक्तियों का
पहला मानदंड. वोल्टेरा ऑपरेटरों के पास यह तथ्यात्मक क्षय है
में निर्मित (\cref{exo:b3:complete:4}(ए) बिल्कुल शोषण किया
यह), यही कारण है कि आरंभिक मूल्य संबंधी समस्याएं कभी प्रभावित नहीं होतीं
सीमा मूल्य समस्याओं की अनुनाद घटनाएँ
(\cref{ch:b3:spectral}).
\end{example}

\section{हाहन--बनाच}

\begin{theorem}[हन--बनाच, विश्लेषणात्मक रूप]
\label{thm:b3:banach:hahnbanach}
होने देना $ E $ एक हो \emph{असली} सदिश स्थल, $ p \colon E \to \R $
\emph{अधरेखीय} ($ p(x + y) \leq p(x) + p(y)$ और $ p(tx) =
tp(x)$ के लिए $ t \geq 0$), $ F \subseteq E $ एक उपस्थान और $ f
\colon F \to \R $ के साथ रैखिक $ f \leq p $ पर $ F $. तब $ f $ विस्तार करता है
एक रेखीय के लिए $\tilde f \colon E \to \R $ साथ $\tilde f \leq p $
पर $ E $.\index{हैन--बानाच प्रमेय}
\end{theorem}

\begin{proof}
\emph{एक-चरणीय विस्तार.} होने देना $ x_0 \notin F $; हम विस्तार करते हैं $ f $ को
$ F \oplus \R x_0$ चुनकर $\alpha = \tilde f(x_0)$
सही ढंग से: हमें सभी के लिए चाहिए $ y \in F $, $ t > 0$,
\[
f(y) + t\alpha \leq p(y + tx_0)
\quad\text{and}\quad
f(y) - t\alpha \leq p(y - tx_0),
\]
जिसे विभाजित करने के बाद $ t $ (sublinearity) को कम करें
\[
\sup_{v \in F}\ \bigl[f(v) - p(v - x_0)\bigr]
\;\leq\; \alpha \;\leq\;
\inf_{u \in F}\ \bigl[p(u + x_0) - f(u)\bigr].
\]
ऐसा $\alpha $ अस्तित्व में है यदि प्रत्येक बायां सदस्य मौजूद है $\leq $ प्रत्येक
सही सदस्य: वास्तव में $ f(v) + f(u) = f(u + v) \leq p(u + v) \leq
p(u + x_0) + p(v - x_0)$, यानी\ $ f(v) - p(v - x_0) \leq p(u +
x_0) - f(u)$.

\emph{ज़ोर्न.} के एक्सटेंशन ऑर्डर करें $ f $ का प्रभुत्व है $ p $
(जोड़े: उपस्थान, कार्यात्मक) विस्तार से; एक श्रृंखला है
ऊपरी सीमा के रूप में संघ; सभी पर एक अधिकतम तत्व परिभाषित किया जाना चाहिए
का $ E $, अन्यथा एक-चरणीय विस्तार अधिकतमता का खंडन करता है।
\end{proof}

\begin{corollary}\label{cor:b3:banach:hbnormed}
होने देना $ E $ एक मानक स्थान बनें ($ K = \R $ या $\C $).
\begin{enumerate}
  \item प्रत्येक $ f \in F'$ ($ F $ एक उपस्थान) तक फैला हुआ है $\tilde f
        \in E'$ साथ $\norm{\tilde f}_{E'} = \norm f_{F'}$.
  \item हरएक के लिए $ x \neq 0$ वहाँ है $ f \in E'$ साथ $\norm f
        = 1$ और $ f(x) = \norm x $. विशेष रूप से $ E'$ अलग करता है
        के अंक $ E $, और $\norm x = \sup_{\norm f \leq
        1}\abs{f(x)}$.
  \item एक \omterm{def:b3:topology:interior}{बंद} उपस्थान के लिए $ F $ और $ x \notin F $, वहाँ है $ f
        \in E'$ पर गायब हो रहा है $ F $ साथ $ f(x) = d(x, F)$ और
        $\norm f \leq 1$.
\end{enumerate}
\end{corollary}

\begin{proof}
(1) वास्तविक मामला: लागू करें \cref{thm:b3:banach:hahnbanach} साथ $ p(x)
= \norm f_{F'}\,\norm x $ (सब्लिनियर); विस्तार संतुष्ट करता है
$\pm\tilde f(x) = \tilde f(\pm x) \leq p(x)$, इसलिए
$\norm{\tilde f} \leq \norm f $, और $\geq $ प्रतिबंध है.
जटिल मामला: चलो $ u = \operatorname{Re}f $, एक वास्तविक कार्यात्मक
के साथ $\abs u \leq \norm f\norm\cdot $; टिप्पणी $ f(x) = u(x) -
\iu\,u(\iu x)$ (वास्तविक और काल्पनिक भागों की जाँच करें:
$\operatorname{Im}f(x) = -\operatorname{Re}f(\iu x)$). बढ़ाना
$ u $ वास्तविक-रैखिक रूप से समान सीमा और सेट के साथ $\tilde f(x) =
\tilde u(x) - \iu\tilde u(\iu x)$: $\C $-रैखिक (प्रत्यक्ष जांच पर
से गुणा $\iu $), फैलता है $ f $; \omterm{def:b3:rings:ideal}{आदर्श}: दिए गए के लिए $ x $
लिखना $\tilde f(x) = r\eu^{\iu\theta}$, तब $\abs{\tilde f(x)}
= \tilde f(\eu^{-\iu\theta}x) = \tilde u(\eu^{-\iu\theta}x)
\leq \norm f\,\norm x $.

(2) चालू $ F = Kx $ परिभाषित करना $ f(tx) = t\norm x $: \omterm{def:b3:rings:ideal}{आदर्श} $1$ पर $ F $;
(1) से बढ़ाएँ। द्वैत सूत्र: $\leq $ स्पष्ट है, $\geq $ इस के द्वारा $ f $.

(3) चालू $ F \oplus Kx $ परिभाषित करना $ f(y + tx) = t\,d(x, F)$; फिर के लिए
$ t \neq 0$, $\norm{y + tx} = \abs t\,\norm{x + y/t} \geq \abs
t\,d(x, F) = \abs{f(y + tx)}$: $\norm f \leq 1$ पर
उपस्थान; (1) से बढ़ाएँ।
\end{proof}

\begin{remark}\label{rem:b3:banach:bidual}
(2) द्वारा, विहित मानचित्र $ J \colon E \to E''$, $ J(x)(f) =
f(x)$, एक आइसोमेट्री है (\cref{exo:b3:banach:10}): हर मानक
अंतरिक्ष अपने bidual के अंदर बैठता है। रिक्त स्थान के साथ $ J $ विशेषण हैं
बुलाया \emph{कर्मकर्त्ता}\index{कर्मकर्त्ता}; सप्ताहांत
समस्या दिखाता है $\ell^p $ ($1 < p < \infty $) है \omterm{rem:b3:banach:bidual}{कर्मकर्त्ता} जबकि
$\ell^1$ क्या नहीं है।
\end{remark}

\section{बेयर त्रयी}

\begin{theorem}[बानाच--स्टीनहॉस, एकसमान सीमाबद्धता]
\label{thm:b3:banach:banachsteinhaus}
होने देना $ E $ एक हो \emph{बनच} अंतरिक्ष, $ F $ मानकीकृत, और $(T_i)_{i\in
I} \subseteq \mathcal L(E, F)$ एक परिवार के साथ $\sup_i
\norm{T_ix} < \infty $ हरएक के लिए $ x \in E $. तब $\sup_i
\vertiii{T_i} < \infty $.\index{बानाच--स्टीनहॉस प्रमेय}
\end{theorem}

\begin{proof}
सेट $ F_n = \{x : \sup_i\norm{T_ix} \leq n\}$ \omterm{def:b3:topology:interior}{बंद} हैं
(\omterm{def:b3:topology:interior}{बंद} गेंदों की पूर्वछवियों का प्रतिच्छेदन) और कवर $ E $.
बेयर (\cref{thm:b3:complete:baire}) देता है $ n_0$ और एक गेंद
$ B(x_0, r) \subseteq F_{n_0}$. के लिए $\norm z < r $: $\norm{T_iz}
\leq \norm{T_i(x_0 + z)} + \norm{T_ix_0} \leq 2n_0$, इसलिए
$\vertiii{T_i} \leq 2n_0/r $ हरएक के लिए $ i $.
\end{proof}

\begin{corollary}\label{cor:b3:banach:bslimits}
अगर $ E $ बनच है और $ T_n \in \mathcal L(E,F)$ एकाग्र
\emph{बिंदुवार} ($ T_nx \to Tx $ प्रत्येक के लिए $ x $), तब $\sup_n
\vertiii{T_n} < \infty $, $ T \in \mathcal L(E, F)$, और
$\vertiii T \leq \liminf \vertiii{T_n}$.
\end{corollary}

\begin{proof}
अभिसारी अनुक्रम परिबद्ध हैं: बिंदुवार परिबद्धता;
बानाच--स्टीनहॉस कुछ लोगों द्वारा मानदंडों को बांधता है $ M $; तब $\norm{Tx}
= \lim\norm{T_nx} \leq M\norm x $ ($ T $ बिंदुवार के रूप में रैखिक है
सीमा), और तेज से गुजरते हुए बाध्य $\liminf $ में
$\norm{T_nx} \leq \vertiii{T_n}\norm x $.
\end{proof}

\begin{theorem}[अपसारी फूरियर श्रृंखला]
\label{thm:b3:banach:fourierdiverge}
वहां है \omterm{def:b3:topology:continuity}{निरंतर} $2\pi $-आवधिक कार्य $ f $ किसका
फूरियर श्रृंखला में विचलन होता है $0$: $\sup_N\abs{S_N(f)(0)} =
\infty $. सचमुच ऐसा $ f $ का एक सघन उपसमुच्चय बनाएँ $\mathcal
C(S^1)$.\index{फूरियर श्रृंखला, विचलन}
\end{theorem}

\begin{proof}
में काम $ E = (\mathcal C(S^1), \norm\cdot_\infty)$, एक बानाच
अंतरिक्ष, कार्यात्मकताओं के साथ $\Lambda_N(f) = S_N(f)(0) =
\frac1{2\pi}\int_{-\pi}^{\pi}f(t)\,D_N(t)\,\dd t $ (डिरिचलेट
कर्नेल, वर्ष 2). प्रत्येक $\Lambda_N $ है \omterm{def:b3:topology:continuity}{निरंतर} साथ
\[
\norm{\Lambda_N} = \frac1{2\pi}\int_{-\pi}^\pi\abs{D_N(t)}\,\dd
t \;=\; L_N .
\]
($\leq $ स्पष्ट है; $\geq $: लेना $ f $ \omterm{def:b3:topology:continuity}{निरंतर}, $\norm f_\infty
\leq 1$, अनुमान लगाना $\operatorname{sign}D_N $ --- संकेत
इसमें बहुत सी छलाँगें हैं; के अंतराल पर प्रत्येक छलांग को सुचारू करना
लंबाई $\varepsilon $ द्वारा अभिन्न को बदलता है $ O(N\varepsilon)$.)
द \emph{लेबेस्ग स्थिरांक} $ L_N $ अनंत की ओर प्रवृत्त:
\[
L_N = \frac1{2\pi}\int_{-\pi}^{\pi}
\frac{\abs{\sin\bigl((N{+}\tfrac12)t\bigr)}}{\abs{\sin(t/2)}}
\,\dd t
\geq \frac{2}{\pi}\int_0^{\pi}
\frac{\abs{\sin\bigl((N{+}\tfrac12)t\bigr)}}{t}\,\dd t
= \frac{2}{\pi}\int_0^{(N+\frac12)\pi}\frac{\abs{\sin
u}}{u}\,\dd u
\]
(का उपयोग करते हुए $\abs{\sin(t/2)} \leq t/2$ पर $[0, \pi]$, फिर
प्रतिस्थापित करना $ u = (N + \tfrac12)t $). मेहराबों में काटना:
\[
\int_{(k-1)\pi}^{k\pi}\frac{\abs{\sin u}}u\,\dd u
\geq \frac1{k\pi}\int_{(k-1)\pi}^{k\pi}\abs{\sin u}\,\dd u =
\frac{2}{k\pi},
\qquad\text{so}\qquad
L_N \geq \frac{4}{\pi^2}\sum_{k=1}^N\frac1k
\xrightarrow[N\to\infty]{} \infty .
\]
यदि प्रत्येक \omterm{def:b3:topology:continuity}{निरंतर} $ f $ था $\sup_N
\abs{\Lambda_N(f)} < \infty $, बानाच--स्टीनहॉस मजबूर करेगा
$\sup_N\norm{\Lambda_N} < \infty $: विरोधाभास। इतना कुछ $ f $
--- वास्तव में एक तुच्छ, \omterm{def:b3:topology:interior}{घना} सेट का $ f $ (का पूरक
$\bigcup_M\{f: \sup_N\abs{\Lambda_Nf}\leq M\}$, एक गणनीय
\omterm{def:b3:topology:interior}{बंद} सेटों का संघ, जिसमें कोई नहीं है आंतरिक भाग उपरोक्त द्वारा
किसी भी गेंद में लगाया जाता है \omterm{rem:b3:complete:meagre}{अल्प}) --- है $\sup_N\abs{S_Nf(0)} =
\infty $.
\end{proof}

\begin{theorem}[मैपिंग खोलें]\label{thm:b3:banach:openmapping}
होने देना $ E, F $ बनच रिक्त स्थान बनें और $ T \in \mathcal L(E, F)$
\emph{विशेषण}. तब $ T $ खुला है: $ T(B_E(0,1)) \supseteq
B_F(0, c)$ कुछ के लिए $ c > 0$. फलस्वरूप एक विशेषण परिबद्ध है
बानाच रिक्त स्थान के बीच ऑपरेटर की एक सीमा होती है
उलटा.\index{ओपन मैपिंग प्रमेय}
\end{theorem}

\begin{proof}
लिखना $ B = B_E(0,1)$. प्रत्यक्षता देता है $ F = \bigcup_n
\overline{T(nB)} = \bigcup_n n\,\overline{T(B)}$; बैरे
(\cref{thm:b3:complete:baire}) देता है आंतरिक भाग को
$\overline{T(B)}$: कुछ $ B_F(y_0, 4c) \subseteq
\overline{T(B)}$. पुन: \omterm{ex:b3:groups:actions}{केंद्र} पर $0$: के लिए $\norm y < 4c $, दोनों
$ y_0 + y $ और $ y_0$ छवियों की सीमाएँ हैं $ Tu_k $, $ Tv_k $ साथ
$ u_k, v_k \in B $, इसलिए $ y = \lim T(u_k - v_k)$ साथ $ u_k - v_k
\in 2B $: $ B_F(0, 4c) \subseteq \overline{T(2B)}$, अर्थात\
$ B_F(0, 2c) \subseteq \overline{T(B)}$.

\emph{को हटा रहा है \omterm{def:b3:topology:interior}{बंद}} (यहाँ पूर्णता का $ E $ प्रवेश करता है):
चलो $\norm y < c $. चुनना $ x_1 \in \frac12 B $ साथ $\norm{y -
Tx_1} < c/2$ ($ B_F(0,2c) \subseteq \overline{T(B)}$ द्वारा स्केल किया गया
$\frac12$); उपपादन $ x_k \in 2^{-k}B $ साथ $\norm{y -
T(x_1 + \dots + x_k)} < c\,2^{-k}$. शृंखला $\sum x_k $
बनच में बिल्कुल अभिसरण होता है $ E $, को $ x \in B $ (\omterm{def:b3:rings:ideal}{आदर्श} $<
\sum 2^{-k} = 1$), और $ Tx = y $ द्वारा निरंतरता: $ B_F(0, c)
\subseteq T(B)$. का खुलापन $ T $ मनमाने ढंग से खुलने पर अनुसरण करता है
अनुवाद और स्केलिंग; परिणाम के लिए, खुलापन $ T $
मतलब $ T^{-1}$ है \omterm{def:b3:topology:continuity}{निरंतर}.
\end{proof}

\begin{corollary}[समतुल्य मानदंड]\label{cor:b3:banach:equivnorms}
यदि एक सदिश समष्टि है \omterm{def:b3:complete:complete}{पूरा} दो तुलनीय मानदंडों के लिए
($\norm\cdot_a \leq C\norm\cdot_b $), मानदंड समतुल्य हैं।
\end{corollary}

\begin{proof}
पहचान $(E, \norm\cdot_b) \to (E, \norm\cdot_a)$ है
बानाच रिक्त स्थान के बीच परिबद्ध और विशेषण: इसका व्युत्क्रम है
घिरा हुआ.
\end{proof}

\begin{theorem}[बंद ग्राफ़]\label{thm:b3:banach:closedgraph}
होने देना $ E, F $ बानाच हो और $ T \colon E \to F $ रैखिक. यदि
ग्राफ $\Gamma = \{(x, Tx)\}$ में \omterm{def:b3:topology:interior}{बंद} है $ E \times F $ (अर्थात्\
$ x_n \to x $ और $ Tx_n \to y $ संकेत करना $ y = Tx $), तब $ T $ घिरा है।\index{बंद ग्राफ प्रमेय}
\end{theorem}

\begin{proof}
$ E \times F $ साथ $\norm{(x,y)} = \norm x + \norm y $ बनच है;
$\Gamma $, एक \omterm{def:b3:topology:interior}{बंद} उपस्थान, बानाच है। प्रक्षेपण
$\pi_E\colon \Gamma \to E $ परिबद्ध और विशेषणात्मक है, इसलिए यह है
उलटा $ x \mapsto (x, Tx)$ घिरा है
(\cref{thm:b3:banach:openmapping}): $\norm{Tx} \leq \norm{(x,
Tx)} \leq C\norm x $.
\end{proof}

\begin{method}\label{met:b3:banach:trilogy}
किस प्रमेय तक कब पहुंचना है. \emph{हाहन--बनाच}: को
निर्धारित व्यवहार के साथ एक कार्यात्मकता उत्पन्न करें (मानदंड a
वेक्टर, एक उप-स्थान पर लुप्त हो रहा है, एक उप-स्थान से विस्तारित हो रहा है) ---
नहीं पूर्णता आवश्यकता है। \emph{बानाच--स्टीनहॉस}: परिवर्तित करना
बिंदुवार जानकारी को समान सीमा में --- आम तौर पर दिखाने के लिए
एक सीमा संचालन है \omterm{def:b3:topology:continuity}{निरंतर}, या (contrapositive) साबित करने के लिए
\emph{विचलन} कुछ तत्वों के लिए, जैसे फूरियर श्रृंखला के लिए।
\emph{ओपन मैपिंग/\omterm{def:b3:topology:interior}{बंद} ग्राफ़}: पाने के निरंतरता निःशुल्क
से \omterm{def:b3:galois:algebraic}{बीजगणितीय} वस्तुनिष्ठता या ए से \omterm{def:b3:topology:interior}{बंद} की संपत्ति
ग्राफ --- विशिष्ट उपयोग: दो की तुलना करना \omterm{def:b3:complete:complete}{पूरा} मानदंड, या साबित करना
स्वचालित निरंतरता. सभी तीन बेयर प्रमेयों की आवश्यकता है
पूर्णता \emph{स्रोत का}; अन्यथा प्रति उदाहरण
(\cref{exo:b3:banach:7}).
\end{method}

\section{दोहरी जगहें, ठोस रूप से}

\begin{theorem}\label{thm:b3:banach:duals}
सममितीय रूप से: $(c_0)' \cong \ell^1$ और $(\ell^1)' \cong
\ell^\infty $, युग्मन के माध्यम से $\langle x, y\rangle = \sum_n
x_ny_n $.\index{अनुक्रम स्थानों के दोहरे}
\end{theorem}

\begin{proof}
हम साबित करते हैं $(c_0)' \cong \ell^1$; दूसरी पहचान है
\cref{exo:b3:banach:5}. को $ y \in \ell^1$ सं\omterm{def:b3:topology:interior}{बंद}्ध करना $\Lambda_y(x)
= \sum x_ny_n $ ($ x \in c_0$): बिल्कुल अभिसरण, के साथ
$\abs{\Lambda_y(x)} \leq \norm x_\infty\norm y_1$, इसलिए
$\norm{\Lambda_y} \leq \norm y_1$. इसके विपरीत चलो $\Lambda \in
(c_0)'$; तय करना $ y_n = \Lambda(e_n)$ ($ e_n $ इकाई अनुक्रम)।
किसी के लिए $ N $, परीक्षा $ x^{(N)} = \sum_{n \leq N}
\operatorname{sign}(\overline{y_n})\,e_n \in c_0$ (\omterm{def:b3:rings:ideal}{आदर्श} $\leq
1$; जटिल मामले में यूनिमॉड्यूलर कारकों का उपयोग करें $\bar y_n/\abs
{y_n}$): $\Lambda(x^{(N)}) = \sum_{n\leq N}\abs{y_n} \leq
\norm\Lambda $. इसलिए $ y \in \ell^1$ साथ $\norm y_1 \leq
\norm\Lambda $. अंत में $\Lambda = \Lambda_y $: दोनों इस पर सहमत हैं
$ e_n $, इसलिए परिमित अनुक्रमों पर, सघन में $ c_0$ (काट-छाँट:
$\norm{x - \sum_{n \leq N}x_ne_n}_\infty = \sup_{n > N}\abs{x_n}
\to 0$ ठीक इसलिए क्योंकि $ x_n \to 0$); \omterm{def:b3:topology:continuity}{निरंतर} कार्यात्मकता
ए पर सहमति \omterm{def:b3:topology:interior}{घना} सेट बराबर हैं. पत्राचार रैखिक है,
विशेषण, और सममितीय ($\norm{\Lambda_y} = \norm y_1$ से
दो असमानताएँ)।
\end{proof}

\begin{exercise}[$\star\star\star $]\label{exo:b3:banach:8}
(हेलिंगर--टोएप्लिट्ज़) चलो $ T \colon \ell^2 \to \ell^2$ रैखिक हो
(हर जगह परिभाषित) और \emph{सममित}: $\langle Tx,
y\rangle = \langle x, Ty\rangle $ सभी के लिए $ x, y $, कहाँ $\langle
x, y \rangle = \sum x_n\bar y_n $. बताते हैं कि $ T $ घिरा है।
\emph{(\omterm{def:b3:topology:interior}{बंद} ग्राफ़: यदि $ x_k \to x $ और $ Tx_k \to z $, परीक्षण
मनमानी के ख़िलाफ़ $ y $.)} नैतिक: असीमित सममित संचालक
--- क्वांटम यांत्रिकी के हैमिल्टनियन --- कभी नहीं हो सकते
संपूर्ण स्थान पर परिभाषित।
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b3:banach:9}
(चतुर्भुज पर पोल्या का प्रमेय) प्रत्येक के लिए $ n $, होने देना $\Lambda_n(f)
= \sum_{i=0}^{n} w_{i,n}f(x_{i,n})$ पर एक चतुर्भुज नियम बनें
$\mathcal C(\intcc01)$ ($ x_{i,n} \in \intcc01$, $ w_{i,n} \in
\R $). बताते हैं कि $\Lambda_n(f) \to \int_0^1f $ के लिए \emph{प्रत्येक}
\omterm{def:b3:topology:continuity}{निरंतर} $ f $ यदि और केवल यदि: (i) $\Lambda_n(P) \to \int_0^1P $
प्रत्येक बहुपद के लिए $ P $, और (ii) $\sup_n\sum_i\abs{w_{i,n}} <
\infty $. \emph{(गणना करें $\norm{\Lambda_n}$; उपयोग करें
बानाच--स्टीनहॉस और वीयरस्ट्रैस।)} उस नियम की जाँच करें
\emph{सकारात्मक} स्थिरांकों पर सटीक भार संतुष्ट करते हैं (ii)
स्वचालित रूप से.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:banach:10}
बताते हैं कि $ J \colon E \to E''$, $ J(x)(f) = f(x)$, एक रैखिक है
आइसोमेट्री (उपयोग) \cref{cor:b3:banach:hbnormed}(2)), और यह है
विशेषण कब $\dim E < \infty $. वो भी दिखाओ अगर $ E'$ है
\omterm{def:b3:galois:separable}{वियोज्य} तो वैसा ही है $ E $. \emph{(चुनना $ x_n $ लगभग \omterm{def:b3:groups:normal}{सामान्य}ीकरण ए
का सघन क्रम $ E'$ और दिखाओ कि उनका \omterm{def:b3:topology:interior}{बंद} विस्तार है $ E $, के जरिए
\cref{cor:b3:banach:hbnormed}(3).)}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:banach:11}
(भागफल रिक्त स्थान) चलो $ E $ एक बानाच स्थान बनें और $ F \subseteq
E $ ए \emph{\omterm{def:b3:topology:interior}{बंद} किया हुआ} उपस्थान. पर $ E/F $ परिभाषित करना
\[
\norm{\bar x} \;=\; d(x, F) = \inf_{y\in F}\norm{x - y} .
\]
(ए) दिखाएँ कि यह एक अच्छी तरह से परिभाषित है \emph{\omterm{def:b3:rings:ideal}{आदर्श}} पर $ E/F $ (कहां
का बंदपन करता है $ F $ दर्ज करें?), और वह प्रक्षेपण $\pi
\colon E \to E/F $ है $\vertiii\pi \leq 1$ और खुले मानचित्र बनाता है
यूनिट बॉल को ओपन यूनिट बॉल पर डालें।
(बी) वह दिखाओ $ E/F $ है \omterm{def:b3:complete:complete}{पूरा}. \emph{(श्रृंखला का प्रयोग करें
की कसौटी \cref{exo:b3:complete:1}(बी): दी गई कक्षाएं
$\bar x_k $ साथ $\sum\norm{\bar x_k} < \infty $, प्रत्येक को ऊपर उठाएं
$ x_k \in E $ साथ $\norm{x_k} \leq \norm{\bar x_k} + 2^{-k}$
और योग करो $ E $.)}
(सी) गणना: के लिए $ E = c $ (अभिसरण अनुक्रम) और $ F =
c_0$, दिखाओ $ c/c_0 \cong K $ सममितीय रूप से के माध्यम से $\bar x \mapsto
\lim_nx_n $.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:banach:12}
(बद्ध अनुमान और पूरक उपस्थान) आइए $ E $ एक हो
बानाच स्पेस और $ P \colon E \to E $ के साथ रैखिक $ P^2 = P $
(एक \omterm{def:b3:galois:algebraic}{बीजगणितीय} प्रक्षेपण), $ V = \operatorname{im}P $, $ W =
\ker P $.
(ए) मान लीजिए $ P $ घिरा है। बताते हैं कि $ V $ और $ W $ \omterm{def:b3:topology:interior}{बंद} हैं
और $ E = V \oplus W $ विघटन के साथ $ x = Px + (x -
Px)$.
(बी) इसके विपरीत, मान लीजिए $ E = V \oplus W $ साथ \emph{दोनों} $ V,
W $ \omterm{def:b3:topology:interior}{बंद} करो, और चलो $ P $ पर प्रक्षेपण हो $ V $ साथ में $ W $.
बताते हैं कि $ P $ घिरा है। \emph{(\omterm{def:b3:topology:interior}{बंद} ग्राफ़: यदि $ x_n \to x $
और $ Px_n \to z $, तब $ z \in V $, $ x_n - Px_n \to x - z \in
W $, और अपघटन की विशिष्टता की पहचान करता है $ z =
Px $.)}
(सी) निष्कर्ष निकालें \emph{समानक}: एक उपस्थान $ V $ स्वीकार करता है ए
परिबद्ध प्रक्षेपण यदि यह \omterm{def:b3:topology:interior}{बंद} है और \omterm{def:b3:topology:interior}{बंद} है
\omterm{def:b3:galois:algebraic}{बीजगणितीय} पूरक --- और ध्यान दें (बिना प्रमाण के)।
इस संपत्ति के बिना \omterm{def:b3:topology:interior}{बंद} उप-स्थान मौजूद हैं ($ c_0$ अंदर
$\ell^\infty $ शास्त्रीय उदाहरण है): हिल्बर्ट रिक्त स्थान,
कहाँ $ V^\perp $ हमेशा काम करता है
(\cref{ch:b3:hilbert}), अपवाद हैं, नियम नहीं।
\end{exercise}

\section{समस्या: का द्वैत \texorpdfstring{$\ell^p $}%
{एलपी} रिक्त स्थान}

\begin{problem}[{सप्ताहांत समस्या --- $(\ell^p)' = \ell^q $,
संवेदनशीलता, और की विचित्रता $\ell^\infty $}]
\label{pb:b3:banach:1}
हल करना $1 < p < \infty $ और चलो $ q $ संयुग्मी प्रतिपादक बनें,
$\frac1p + \frac1q = 1$. जोड़ी भर है $\langle x,
y\rangle = \sum_n x_ny_n $.

\medskip
\noindent\textbf{Part I --- Hölder and Minkowski for
sequences.}
\begin{enumerate}
  \item (यंग की असमानता) के लिए $ a, b \geq 0$ दिखाओ $ ab \leq
        \frac{a^p}p + \frac{b^q}q $, की अवतलता का उपयोग करते हुए $\log $ या
        पढ़ाई करके $ t \mapsto \frac{t^p}p + \frac1q - t $.
  \item (एच\"पुराना) निष्कर्ष निकालें: $\abs{\langle x, y\rangle} \leq
        \norm x_p\norm y_q $ के लिए $ x \in \ell^p $, $ y \in \ell^q $;
        समानता के मामले की पहचान करें.
  \item (मिन्कोव्स्की) के लिए त्रिभुज असमानता को घटाएँ
        $\norm\cdot_p $. \emph{(लिखना $\abs{x_n + y_n}^p \leq
        \abs{x_n}\,\abs{x_n{+}y_n}^{p-1} +
        \abs{y_n}\,\abs{x_n{+}y_n}^{p-1}$ और एच लागू करें\"से अधिक उम्र का
        प्रत्येक पद।)}
  \item साबित करें कि $\ell^p $ है \omterm{def:b3:complete:complete}{पूरा} और वह परिमित है
        इसमें अनुक्रम सघन हैं।
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The duality $(\ell^p)' =
\ell^q $.}
\begin{enumerate}[resume]
  \item के लिए $ y \in \ell^q $, बताते हैं कि $\Lambda_y(x) = \langle
        x, y\rangle $ को परिभाषित करता है $\Lambda_y \in (\ell^p)'$ साथ
        $\norm{\Lambda_y} \leq \norm y_q $, और, परीक्षण जारी है
        $ x_n = \abs{y_n}^{q-1}\operatorname{sign}(y_n)$
        (उपयुक्त रूप से छोटा और \omterm{def:b3:groups:normal}{सामान्य}ीकृत), वह
        $\norm{\Lambda_y} = \norm y_q $.
  \item इसके विपरीत, दिया गया $\Lambda \in (\ell^p)'$, तय करना $ y_n =
        \Lambda(e_n)$; दिखाओ $ y \in \ell^q $ साथ $\norm y_q \leq
        \norm\Lambda $ \emph{(प्रश्न के अनुसार काट-छाँट पर परीक्षण करें
        5 और ट्रंकेशन की लंबाई बढ़ने दें)}, और निष्कर्ष निकालें
        $\Lambda = \Lambda_y $: वो नक्शा $ y \mapsto \Lambda_y $ है
        एक सममितीय समरूपता $\ell^q \to (\ell^p)'$.
  \item उसका निष्कर्ष निकालो $\ell^p $ है \emph{\omterm{rem:b3:banach:bidual}{कर्मकर्त्ता}} के लिए $1 < p <
        \infty $: दो द्वंद्वों की रचना, प्रत्येक तत्व
        $(\ell^p)''$ से आता है $\ell^p $; इसे ध्यान से सत्यापित करें
        सम्मिश्रण विहित है $ J $.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- $\ell^1$ and $\ell^\infty $ are
different animals.}
\begin{enumerate}[resume]
  \item बताते हैं कि $\ell^p $ ($1 \leq p < \infty $) और $ c_0$ हैं
        \omterm{def:b3:galois:separable}{वियोज्य}, लेकिन $\ell^\infty $ क्या नहीं है। \emph{(द
        के उपसमुच्चय के अनगिनत सूचक अनुक्रम $\N $
        जोड़ीवार दूरी पर हैं $1$.)}
  \item से निष्कर्ष निकालें \cref{exo:b3:banach:10} वह $(\ell^1)'
        \cong \ell^\infty $ लेकिन $(\ell^\infty)' \not\cong
        \ell^1$: $\ell^1$ है \emph{नहीं} \omterm{rem:b3:banach:bidual}{कर्मकर्त्ता}.
        \emph{(अगर $(\ell^\infty)'$ थे $\ell^1$, यह होगा
        अलग करने योग्य, मजबूर करने वाला $\ell^\infty $ अलग करने योग्य।)}
  \item (एक बानाच सीमा, स्पष्ट रूप से) चालू $\ell^\infty_\R $, होने देना
        $ p(x) = \limsup_n \frac{x_1 + \dots + x_n}{n}$. बताते हैं कि $ p $ सबलीनियर है, और वह सबस्पेस पर है $ c $ का
        अभिसरण अनुक्रम, $\mathrm{LIM}(x) = \lim x $
        संतुष्ट $\mathrm{LIM} \leq p $. हैन--बानाच द्वारा विस्तारित
        को $\mathrm{LIM} \colon \ell^\infty_\R \to \R $ और दिखाओ:
        $\mathrm{LIM}$ सकारात्मक है ($ x \geq 0 \Rightarrow
        \mathrm{LIM}(x) \geq 0$), शिफ्ट-अपरिवर्तनीय
        ($\mathrm{LIM}(x_2, x_3, \dots) = \mathrm{LIM}(x)$),
        सीमा बढ़ाता है, और संतुष्ट करता है $\liminf x \leq
        \mathrm{LIM}(x)\leq \limsup x $.
  \item ऐसा दिखाओ $\mathrm{LIM}$, में देखा गया
        $(\ell^\infty)'$, है \emph{नहीं} रूप का $\Lambda_y $
        किसी के लिए $ y \in \ell^1$; फिर से निष्कर्ष निकालें $(\ell^\infty)'
        \neq \ell^1$. \emph{(इकाई अनुक्रमों पर मूल्यांकन करें
        $ e_n $, फिर \omterm{def:b3:topology:continuity}{निरंतर} क्रम पर $1$.)}
  \item मूल्यांकन करना $\mathrm{LIM}$ पर $(0,1,0,1,\dots)$, और दिखाओ
        वह कोई परिवर्तन-अपरिवर्तनीय नहीं है \emph{गुणक} विस्तार
        की सीमा मौजूद हो सकती है \emph{(विचार करना $ x =
        (0,1,0,1,\dots)$ और $ x\cdot Sx $ कहाँ $ S $ है
        शिफ्ट)}.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Epilogue: why reflexivity
matters.}
\begin{enumerate}[resume]
  \item का उपयोग करते हुए \cref{cor:b3:banach:bslimits} और प्रश्न 6,
        दिखाएँ कि प्रत्येक बंधा हुआ क्रम $\ell^p $ ($1 < p <
        \infty $) का एक परिणाम है $(x^{(k)})$ वह अभिसरण करता है
        \emph{दुर्बलता से}: $\Lambda(x^{(k)})$ प्रत्येक के लिए अभिसरण होता है
        $\Lambda \in (\ell^p)'$. \emph{(विकर्ण निष्कर्षण पर
        असंख्य निर्देशांक; कमजोर सीमा की पहचान करें
        में $\ell^p $ मानदंडों की एक समान सीमा का उपयोग करना और
        एच\"पुराना।)} उदाहरण द्वारा दिखाएँ ($ e_n $ में $\ell^1$, विरुद्ध
        के अच्छी तरह से चुने गए तत्व $\ell^\infty $) कि यह विफल हो जाता है
        में $\ell^1$: कमजोर सघनता का विशेषाधिकार है
        प्रतिवर्ती स्थान.
\end{enumerate}

\medskip
\noindent\textbf{Part V --- The weak \omterm{def:b3:topology:topology}{टोपोलॉजी} at work, and
Schur's surprise.} लिखना $ x^{(k)} \rightharpoonup x $ एक में
मानक स्थान $ E $ (\emph{कमजोर अभिसरण}) कब
$\Lambda(x^{(k)}) \to \Lambda(x)$ हरएक के लिए $\Lambda \in E'$.
\begin{enumerate}[resume]
  \item \omterm{def:b3:complete:complete}{पूरा} जनगणना: दिखाओ $(c_0)' \cong \ell^1$
        सममितीय रूप से, प्रश्न 5-6 की योजना के अनुसार (क्या
        परीक्षण अनुक्रमों को प्रतिस्थापित करता है?) चेन को इकट्ठा करो
        $ c_0 \to \ell^1 \to \ell^\infty \to \dots $ का
        क्रमिक दोहरे और चिह्नित करें कि रिफ्लेक्सिविटी कहां विफल होती है।
  \item बानाच के प्रत्येक कमजोर रूप से अभिसरण अनुक्रम को दिखाएं
        अंतरिक्ष घिरा हुआ है: देखें $ x^{(k)}$ के माध्यम से
        विहित एम्बेडिंग $ J $ कार्यात्मकता के रूप में $ E'$ और
        बानाच--स्टीनहॉस लागू करें
        (\cref{thm:b3:banach:banachsteinhaus}) --- जिस पर
        बनच स्थान, और क्यों है पूर्णता वहाँ उपलब्ध है?
  \item उसे दिखाओ $\ell^p $, $1 < p < \infty $: $ x^{(k)}
        \rightharpoonup x $ आईएफएफ $\sup_k\norm{x^{(k)}}_p <
        \infty $ और $ x^{(k)}_n \to x_n $ प्रत्येक समन्वय के लिए
        $ n $ \emph{(एक दिशा बानाच--स्टीनहॉस का उपयोग करती है
        विहित एम्बेडिंग; दूसरे के लिए, अनुमानित $ y
        \in \ell^q $ परिमित अनुक्रमों द्वारा)}. परिणाम निकालना $ e_k
        \rightharpoonup 0$ में $\ell^2$ जबकि $\norm{e_k}_2 =
        1$: कमजोर सीमाएं द्रव्यमान खो सकती हैं।
  \item दिखाएँ कि मानदंड कमजोर रूप से कम अर्धविराम है:
        $ x^{(k)} \rightharpoonup x $ तात्पर्य $\norm x \leq
        \liminf_k\,\norm{x^{(k)}}$ \emph{(एक \omterm{def:b3:rings:ideal}{आदर्श} चुनें
        के लिए कार्यात्मक $ x $, \cref{cor:b3:banach:hbnormed})}.
  \item (रेडॉन--रिस्ज़ इन $\ell^2$) उसे दिखाओ $\ell^2$, कमजोर
        मानदंडों के अभिसरण के साथ अभिसरण का तात्पर्य है
        \omterm{def:b3:rings:ideal}{आदर्श} अभिसरण \emph{(बढ़ाना $\norm{x^{(k)} -
        x}_2^2$)}. उसी कथन का प्रतिउदाहरण दीजिए
        मानक परिकल्पना के बिना.
  \item (शूर, चरण 1) चलो $ x^{(k)} \rightharpoonup 0$ में
        $\ell^1$ और मान लीजिए, विरोधाभास के लिए,
        $\norm{x^{(k)}}_1 \geq \delta > 0$ साथ में ए
        परिणाम। पहले वो दिखाओ $ x^{(k)}_n \to 0$ प्रत्येक के लिए $ n $ (कौन से कार्य?), फिर निर्माण करें
        पुनरावर्ती सूचकांक $ k_1 < k_2 < \cdots $ और पूर्णांक
        $0 = N_0 < N_1 < N_2 < \cdots $ ऐसा कि द्रव्यमान
        $ x^{(k_j)}$ ब्लॉक पर ध्यान \omterm{ex:b3:groups:actions}{केंद्र}ित करता है $ B_j =
        \intoc{N_{j-1}}{N_j}$:
        \[
        \sum_{n \in B_j}\bigl|x^{(k_j)}_n\bigr| \geq
        \norm{x^{(k_j)}}_1 - \frac\delta{10} .
        \]
  \item (शूर, चरण 2) परिभाषित करें $ y \in \ell^\infty $ द्वारा $ y_n =
        \operatorname{sign}\bigl(x^{(k_j)}_n\bigr)$ के लिए $ n \in
        B_j $. दिखाओ $\langle x^{(k_j)}, y\rangle \geq
        \norm{x^{(k_j)}}_1 - \frac{2\delta}{10} \geq
        \frac{8\delta}{10}$ और इसके साथ विरोधाभास उत्पन्न करें
        $ x^{(k)} \rightharpoonup 0$. निष्कर्ष निकालना \emph{शूर का
        प्रमेय}: में $\ell^1$, कमजोर रूप से अभिसरण अनुक्रम
        \omterm{def:b3:rings:ideal}{आदर्श} में अभिसरण.
  \item उसका निष्कर्ष निकालो $(e_k)$ कोई कमजोर रूप से अभिसरण नहीं है
        बाद में $\ell^1$ (इसकी एकमात्र उम्मीदवार सीमा है
        $0$, समन्वय के अनुसार --- फिर शूर लागू करें), पुनर्प्राप्त करना
        प्रश्न 13 की कमजोर सघनता की विफलता; और संकल्प करें
        स्पष्ट विरोधाभास: में $\ell^1$ कमजोर और \omterm{def:b3:rings:ideal}{आदर्श}
        का अभिसरण \emph{दृश्यों} संयोग, फिर भी
        कमजोर और \omterm{def:b3:rings:ideal}{आदर्श} \emph{टोपोलोजी} भिन्न और परिबद्ध
        सेट अभी भी कमजोर रूप से क्रमिक रूप से विफल हो रहे हैं कॉम्पैक्ट ---
        कोई विरोधाभास नहीं, केवल संवेदनशीलता की विफलता।
  \item (संश्लेषण तालिका) के लिए $ E \in \{c_0,\ \ell^1,\ \ell^p\
        (1{<}p{<}\infty),\ \ell^\infty\}$, सारणीबद्ध करें: द \omterm{def:b3:banach:operator}{दोहरी};
        पृथक्करणीयता; संवेदनशीलता; चाहे बंधे अनुक्रम
        कमजोर रूप से अभिसरण परिणामों को स्वीकार करें; और एक
        प्रत्येक स्थान की हस्ताक्षर संपत्ति, एक में उचित
        इस समस्या से लाइन.
\end{enumerate}

\medskip
\noindent\textbf{Part VI --- Complements: nearest points,
averaged convergence, the value of a Banach limit.}
\begin{enumerate}[resume]
  \item (निकटतम बिंदु: रिफ्लेक्सिविटी का लाभांश) चलो $ F $ हो
        का एक \omterm{def:b3:topology:interior}{बंद} उपस्थान $\ell^p $ ($1 < p < \infty $) और
        $ x \in \ell^p $. बताते हैं कि $ d = \operatorname{dist}(x,
        F)$ है \emph{प्राप्त}: लघुकरण से निकालें
        अनुक्रम एक कमजोर रूप से अभिसरण अनुवर्ती (प्रश्न
        13), कमजोर सीमा को अंदर रखें $ F $ निर्माण द्वारा, के माध्यम से
        हैन--बनाच, एक क्रियात्मक रूप से लुप्त हो जाना $ F $ लेकिन पर नहीं
        इसके बाहर एक बिंदु, और प्रश्न 17 के साथ समाप्त करें।
        फिर दिखाएँ कि विशेषाधिकार सार्वभौमिक नहीं है: में $ c_0$,
        के लिए $\Lambda(x) = \sum_n2^{-n}x_n $, सिद्ध करना
        $\norm\Lambda = 1$ यूनिट बॉल पर प्राप्त नहीं होता है,
        दूरी सूत्र स्थापित करें
        $\operatorname{dist}(x, \ker\Lambda) =
        \abs{\Lambda(x)}$, और यह निष्कर्ष निकालें कि नहीं $ x \notin
        \ker\Lambda $ \omterm{def:b3:topology:interior}{बंद} में निकटतम बिंदु है
        हाइपरप्लेन $\ker\Lambda $.
  \item (बानाच--सैक्स इन $\ell^2$) होने देना $ x^{(k)}
        \rightharpoonup 0$ में $\ell^2_{\R}$ साथ
        $\norm{x^{(k)}}_2 \leq C $. एक अनुवर्ती का निर्माण करें
        $(y_j)$ साथ $\abs{\langle y_i, y_j\rangle} \leq
        \frac1j $ सभी के लिए $ i < j $, और निष्कर्ष निकालें
        \[
        \Bigl\lVert\frac{y_1 + \dots + y_m}m\Bigr\rVert_2^2
        \leq \frac{C^2 + 2}m \longrightarrow 0 :
        \]
        निष्कर्षण के बाद, Ces\`एआरओ का अर्थ है एकाग्र होना
        \emph{\omterm{def:b3:rings:ideal}{आदर्श}}. पर जाँचा $(e_k)$, जिनके साधन \omterm{def:b3:rings:ideal}{आदर्श} हैं
        $\frac1{\sqrt m}$: कमजोर अभिसरण, के लिए बेकार
        अनुक्रम ही (प्रश्न 16), \omterm{def:b3:rings:ideal}{आदर्श} बन जाता है
        औसत के लिए अभिसरण.
  \item (बैनाच सीमा का मान) चलो $ L $ कोई भी बनच हो
        सीमा (प्रश्न 10) और $ A_mx = \frac1m(x + Sx + \dots
        + S^{m-1}x)$. दिखाओ $ L(A_mx) = L(x)$ और $\liminf A_mx
        \leq L(x) \leq \limsup A_mx $; उसका निष्कर्ष निकालो \emph{सभी}
        बनच सीमाएँ मूल्य के साथ, आवधिक अनुक्रमों पर सहमत होती हैं
        एक अवधि में माध्य --- $\frac13$ पर $(1, 0, 0, 1,
        0, 0, \dots)$, प्रश्न 12 के अनुरूप
        $\frac12$. फिर दिखाएँ समझौता \omterm{def:b3:groups:normal}{सामान्य} रूप से विफल रहता है: के लिए
        ब्लॉक अनुक्रम $ x $ के बराबर $1$ पर
        $\intoc{3^{j-1}}{3^j}$ एक जैसे के लिए $ j $ और $0$ अन्यत्र,
        दिखाएँ कि Ces\`एआरओ का अर्थ है बीच में दोलन करना $\leq
        \frac13$ और $\geq \frac23$, और दो बनच का निर्माण करें
        सीमाएं $ L_\pm $ साथ $ L_-(x) \leq \frac13 < \frac23
        \leq L_+(x)$ \emph{(से बढ़ाएँ $ c \oplus \R x $ के साथ
        अत्यधिक स्वीकार्य मूल्य $\pm $: जाँच करें कि
        $\Lambda(y + tx) = \lim y + t\,p(x)$ का बोलबाला है
        सबलाइनर $ p$ प्रश्न 10 का)}.
\end{enumerate}
