\chapter{संभाव्यता: आधार और बड़े का नियम
संख्याएँ}\label{ch:b3:probability}

वर्ष 2 ने गणनीय स्थानों पर संभाव्यता निर्मित की; \omterm{def:b3:measure:measure}{उपाय} सिद्धांत
अब हर प्रतिबंध हटाता है. ए \omterm{def:b3:probability:space}{संभाव्यता स्थान} एक है
स्थान मापें कुल द्रव्यमान का $1$, यादृच्छिक चर हैं
\omterm{def:b3:lebesgue:measurable}{औसत दर्जे का} मानचित्र, \omterm{def:b3:probability:space}{अपेक्षा} लेब्सग इंटीग्रल है --- और
एक ही बार में संपूर्ण विश्लेषणात्मक शस्त्रागार
(\cref{ch:b3:measure,ch:b3:lebesgue,ch:b3:product}) पर लागू होता है
मौका. यह अध्याय शब्दकोश स्थापित करता है, निर्माण करता है
के अनंत क्रम \omterm{def:b3:probability:independence}{स्वतंत्र} यादृच्छिक चर (पर
$\intcc01$, बाइनरी अंकों से: यादृच्छिकता अंदर छिपी हुई है
\omterm{def:b3:measure:lebesgueouter}{लेब्सेग माप}), बोरेल--कैंटेली लेम्मास और साबित करता है
कोलमोगोरोव का \omterm{thm:b3:probability:zeroone}{शून्य--एक कानून}, के तरीकों को सुलझाता है
अभिसरण, और सिद्ध करता है \omterm{def:b3:probability:space}{कानून} बड़ी संख्या में ---
प्रमेय जो आवृत्तियों को संभावनाओं में परिवर्तित करता है और
आँकड़े संभव. सप्ताहांत की समस्या एटेमादी का प्रमाण देती है
ताकतवर का \omterm{def:b3:probability:space}{कानून} अपने निश्चित रूप में $ L^1$ रूप।

\section{शब्दकोष}

\begin{definition}\label{def:b3:probability:space}
ए \emph{संभाव्यता स्थान}\index{संभाव्यता स्थान} एक है
स्थान मापें $(\Omega, \mathcal A, \P)$ साथ $\P(\Omega) =
1$; घटक $\mathcal A $ हैं \emph{घटनाएँ}, और एक संपत्ति
धारण करता है \emph{लगभग निश्चित रूप से} (ए.एस.) यदि इसकी घटना है
संभाव्यता $1$. ए \emph{अनियमित परिवर्तनशील वस्तु}\index{अनियमित परिवर्तनशील वस्तु} एक है \omterm{def:b3:lebesgue:measurable}{औसत दर्जे का} नक्शा $ X \colon \Omega \to \R $ (या
$\R^d $: एक यादृच्छिक वेक्टर); इसका \emph{कानून}\index{कानून} आगे बढ़ने की संभावना है \omterm{def:b3:measure:measure}{उपाय} $\P_X =
X_*\P $ पर $\R $ (\cref{exo:b3:product:9}), द्वारा निर्धारित
\emph{वितरण समारोह} $ F_X(t) = \P(X \leq t)$
(\cref{exo:b3:measure:3}). $ X $ है \emph{\omterm{ex:b3:lebesgue:gamma}{घनत्व}} $ f $ अगर
$\P_X = f\,\dd\lambda $; यह है \emph{अलग} अगर $\P_X $ एक है
डायराक द्रव्यमान का गणनीय संयोजन।
\emph{अपेक्षा}\index{अपेक्षा} है
\[
\E[X] = \int_\Omega X\,\dd\P
\qquad (X \geq 0 \text{ or } X \in L^1(\P)),
\]
और यह \emph{स्थानांतरण प्रमेय} (\cref{exo:b3:product:9})
कानून में इसकी गणना करता है: $\E[g(X)] = \int_\R g\,\dd\P_X $ ---
$= \sum g(x_k)p_k $ असतत मामले में, $= \int
g(x)f(x)\dd x $ में \omterm{ex:b3:lebesgue:gamma}{घनत्व} मामला: वर्ष 2 के सूत्र, अभी
एक सिद्धांत के प्रमेय. \emph{झगड़ा} है $\V(X) =
\E[(X - \E X)^2] = \E[X^2] - (\E X)^2$ के लिए $ X \in L^2$.
\end{definition}

\begin{example}\label{ex:b3:probability:laws}
मानक \omterm{def:b3:probability:space}{कानून} और नोट के उनके रूपांतरण: बर्नौली
$\mathcal B(p)$, द्विपद $\mathcal B(n, p)$, ज्यामितीय,
पॉइसन $\mathcal P(\lambda)$ (अलग: वर्ष 2 की तालिकाएँ
वैध रहें); वर्दी पर $\intcc01$ (\omterm{def:b3:measure:lebesgueouter}{लेब्सेग माप}
स्वयं); घातीय $\mathcal E(\lambda)$ (\omterm{ex:b3:lebesgue:gamma}{घनत्व}
$\lambda\eu^{-\lambda x}\mathbf 1_{x>0}$);
\emph{गाऊसी}\index{गाऊसी} $\mathcal N(m, \sigma^2)$
साथ \omterm{ex:b3:lebesgue:gamma}{घनत्व} $\frac1{\sigma\sqrt{2\pi}}\exp\bigl(-\frac{(x -
m)^2}{2\sigma^2}\bigr)$ --- एक संभावना \omterm{ex:b3:lebesgue:gamma}{घनत्व} द्वारा
\cref{pb:b3:lebesgue:1}, मतलब के साथ $ m $ और विचरण
$\sigma^2$ (गाऊसी क्षण, \cref{exo:b3:product:10}).
\end{example}

\begin{proposition}[मार्कोव और चेबीशेव]
\label{prop:b3:probability:markov}
के लिए $ X \geq 0$ और $ a > 0$: $\P(X \geq a) \leq \frac{\E
X}{a}$; के लिए $ X \in L^2$: $\P\bigl(\abs{X - \E X} \geq
a\bigr) \leq \frac{\V(X)}{a^2}$.\index{मार्कोव का
असमानता}\index{चेबीशेव की असमानता}
\end{proposition}

\begin{proof}
\cref{exo:b3:lebesgue:5}(ए); चेबीशेव मार्कोव पर लागू होता है
$(X - \E X)^2$.
\end{proof}

\section{आज़ादी}

\begin{definition}\label{def:b3:probability:independence}
उप-$\sigma $-बीजगणित $\mathcal A_1, \dots, \mathcal A_n
\subseteq \mathcal A $ हैं \emph{स्वतंत्र}\index{स्वतंत्र}
अगर $\P(A_1\cap\dots\cap A_n) = \prod\P(A_i)$ सभी के लिए $ A_i \in
\mathcal A_i $; घटनाएँ स्वतंत्र हैं यदि $\sigma $-बीजगणित
$\{\varnothing, A_i, A_i^c, \Omega\}$ हैं; यादृच्छिक चर
$ X_1, \dots, X_n $ यदि $\sigma $-बीजगणित $\sigma(X_i) =
X_i^{-1}(\mathcal B(\R))$ हैं। अनंत परिवार है
यदि प्रत्येक परिमित उपपरिवार स्वतंत्र है।
\end{definition}

\begin{theorem}\label{thm:b3:probability:independence}
$ X_1, \dots, X_n $ हैं \omterm{def:b3:probability:independence}{स्वतंत्र} यदि \omterm{def:b3:probability:space}{कानून} वेक्टर का
$(X_1, \dots, X_n)$ है \omterm{thm:b3:product:existence}{उत्पाद माप}
$\P_{X_1}\otimes\cdots\otimes\P_{X_n}$. उस स्थिति में, के लिए
$ g_i \geq 0$ (या ऐसे उत्पाद हैं \omterm{def:b3:lebesgue:l1}{समाकलनीय}):
\[
\E\Bigl[\prod_ig_i(X_i)\Bigr] = \prod_i\E[g_i(X_i)],
\]
विशेष रूप से $\E[XY] = \E X\,\E Y $ और $\V(X_1 + \dots +
X_n) = \sum\V(X_i)$ के लिए \omterm{def:b3:probability:independence}{स्वतंत्र} $ L^2$ चर.
\end{theorem}

\begin{proof}
यदि $ X_i $ हैं \omterm{def:b3:probability:independence}{स्वतंत्र}, दो संभावनाएँ पैमाने
$\P_{(X_1,\dots,X_n)}$ और $\bigotimes\P_{X_i}$ सब पर सहमति
उत्पाद $ B_1\times\dots\times B_n $ बोरेल सेट का --- ए
$\pi $-सिस्टम जनरेटिंग $\mathcal B(\R^n)$
(\cref{prop:b3:product:sections}(बी)) --- इसलिए हर जगह
(\cref{thm:b3:measure:uniqueness}). इसके विपरीत, एक उत्पाद \omterm{def:b3:probability:space}{कानून}
सभी घटनाओं का कारकीकरण करता है $\bigcap_iX_i^{-1}(B_i)$: स्वतंत्रता. \omterm{def:b3:probability:space}{अपेक्षा} सूत्र तो है
टोनेली/फुबिनी (\cref{thm:b3:product:tonelli}) के माध्यम से
स्थानांतरण प्रमेय; $\E[XY] = \E X\E Y $ मामला यह है $ g_i =
\mathrm{id}$, और वर्ग का विस्तार करने से योगात्मकता मिलती है
भिन्नताओं का (विपरीत पद)। $\E[(X_i - \E X_i)(X_j - \E X_j)]
= 0$).
\end{proof}

\begin{theorem}[स्वतंत्र अनुक्रमों का अस्तित्व]
\label{thm:b3:probability:existence}
पर $\bigl(\intcc01, \mathcal L, \lambda\bigr)$ वहाँ एक मौजूद है
क्रम $(U_n)_{n\geq1}$ का \omterm{def:b3:probability:independence}{स्वतंत्र} यादृच्छिक चर,
प्रत्येक वर्दी पर $\intcc01$. नतीजतन, किसी भी निर्धारित के लिए
\omterm{def:b3:probability:space}{कानून} $(\mu_n)$ पर $\R $ वहां है \omterm{def:b3:probability:independence}{स्वतंत्र} $(X_n)$ साथ
$\P_{X_n} = \mu_n $.\index{स्वतंत्र अनुक्रम, अस्तित्व}
\end{theorem}

\begin{proof}
\emph{अंक.} के लिए $\omega \in \intcc01$, होने देना $(b_k(\omega))$
इसके द्वि\omterm{def:b3:topology:basis}{आधार}ी अंक हों ($\omega = \sum b_k2^{-k}$; चुनें
विस्तार बिल्कुल ख़त्म नहीं हो रहा है $1$ की --- अस्पष्टता केवल चिंता का विषय है
एक गणनीय, इसलिए शून्य, सेट)। प्रत्येक $b_k $ एक है \omterm{def:b3:probability:space}{अनियमित परिवर्तनशील वस्तु} ($\{b_k = 1\}$ डायडिक का एक सीमित संघ है
अंतराल) और वेक्टर $(b_1, \dots, b_m)$ प्रत्येक लेता है
में मूल्य $\{0,1\}^m $ लंबाई के एक डायडिक अंतराल पर $2^{-m}$:
द $ b_k $ हैं \omterm{def:b3:probability:independence}{स्वतंत्र} Bernoulli$(\frac12)$.

\emph{पुनः समूह बनाना।} विभाजित करना $\N^*$ अनन्त रूप से अनेक असंगठितों में
अनंत सेट $(I_n)$ (उदा.\ मुख्य शक्तियों या विकर्णों द्वारा);
होने देना $(k^n_j)_j $ एक एक करके बताना $ I_n $ और सेट करें
\[
U_n = \sum_{j\geq1} b_{k^n_j}\,2^{-j} .
\]
प्रत्येक $ U_n $ एकसमान है: इसके द्वि\omterm{def:b3:topology:basis}{आधार}ी अंक हैं \omterm{def:b3:probability:independence}{स्वतंत्र} निष्पक्ष
बिट्स, तो $\P(U_n \in [l2^{-m}, (l+1)2^{-m})) = 2^{-m}$ के लिए
प्रत्येक डायडिक अंतराल, और डायडिक अंतराल निर्धारित करते हैं \omterm{def:b3:probability:space}{कानून}
(\cref{thm:b3:measure:uniqueness}). $ U_n $ हैं \omterm{def:b3:probability:independence}{स्वतंत्र}:
वे के असंयुक्त ब्लॉकों के कार्य हैं \omterm{def:b3:probability:independence}{स्वतंत्र}
परिवार $(b_k)$ --- औपचारिक रूप से, घटनाएँ $\{U_n \in D_n\}$ के लिए
डायडिक $ D_n $ असंयुक्त से परिमित अनेक अंकों पर निर्भर
सेट, और गुणनखंडन; $\pi $-सिस्टम तर्क को अपग्रेड किया गया
सभी बोरेल सेट।

\emph{मनमाना \omterm{def:b3:probability:space}{कानून}.} होने देना $ G_n(u) = \inf\{t : F_{\mu_n}(t)
\geq u\}$ (द \emph{मात्रात्मक कार्य} वितरण का
समारोह $ F_{\mu_n}$); मुख्य तुल्यता $ G_n(u) \leq t
\iff u \leq F_{\mu_n}(t)$ (सही-निरंतरता का $ F $,
एकरसता) दर्शाता है $ X_n = G_n(U_n)$ है \omterm{def:b3:lebesgue:measurable}{औसत दर्जे का} साथ
$\P(X_n \leq t) = \P(U_n \leq F_{\mu_n}(t)) =
F_{\mu_n}(t)$: \omterm{def:b3:probability:space}{कानून} $\mu_n $; स्वतंत्रता विरासत में मिला है
(के कार्य \omterm{def:b3:probability:independence}{स्वतंत्र} चर,
\cref{exo:b3:probability:3}).
\end{proof}

\begin{example}[जन्मदिन की समस्या, ईमानदारी से]
\label{ex:b3:probability:birthday}
के बीच $ n $ के साथ लोग \omterm{def:b3:probability:independence}{स्वतंत्र}, वर्दी जन्मदिन खत्म
$ N = 365$ दिन, संभावना है कि सभी जन्मदिन अलग-अलग हों
\[
p_n = \prod_{k=1}^{n-1}\Bigl(1 - \frac kN\Bigr),
\]
पुनरावृत्त कंडीशनिंग द्वारा (या सीधे: अनुकूल
$ N(N-1)\cdots(N - n + 1)$ कुल से अधिक $ N^n $, एक गिनती
के उत्पाद सूत्र का तर्क दें स्वतंत्रता बनाता है
कठोर)। लघुगणक लेना और प्रयोग करना $-\ln(1 - x) = x +
O(x^2)$:
\[
\ln p_n = -\frac{n(n-1)}{2N} +
O\Bigl(\frac{n^3}{N^2}\Bigr),
\qquad\text{so}\qquad
p_n \approx \eu^{-n^2/2N} .
\]
निर्णायक बिंदु $ p_n = \frac12$ बैठ बैठ गया $ n \approx
\sqrt{2N\ln2} \approx 1.18\sqrt N $: के लिए $ N = 365$, $ n = 23$
($ p_{23} = 0.4927$). दो नैतिकता. सबसे पहले, आपस में टकराव $ n $
में आइटम $ N $ पैमाने पर बक्से दिखाई देते हैं $ n \sim \sqrt N $, नहीं
$ n \sim N $ --- द \emph{जन्मदिन स्केलिंग} जो हैश को नियंत्रित करता है
टकराव और $\sqrt N $ जन्मदिन पर हमलों की लागत
क्रिप्टोग्राफी. दूसरा, गणना एक टेम्पलेट है:
$\binom n2$ जोड़ी-टकराव की घटनाएँ नहीं हैं \omterm{def:b3:probability:independence}{स्वतंत्र}, अभी तक
उत्तर वैसा ही व्यवहार करता है जैसे वे थे
($\eu^{-\binom n2/N}$ बिल्कुल स्वतंत्र-जोड़े हैं
अनुमानी) --- पॉइसन का पहला उदाहरण
सन्निकटन को कठोर बनाया गया \cref{ch:b3:clt}का सप्ताहांत
समस्या (ले कैम की असमानता)।
\end{example}

\section{बोरेल--कैंटेली और शून्य--एक नियम}

\begin{theorem}[बोरेल--कैंटेली]
\label{thm:b3:probability:borelcantelli}
होने देना $(A_n)$ घटनाएँ हों और $\limsup A_n = \bigcap_N
\bigcup_{n\geq N}A_n $ (``$ A_n $ अनंत रूप से होता है
अक्सर'')।\index{बोरेल--कैंटेली लेम्मास}
\begin{enumerate}
  \item अगर $\sum\P(A_n) < \infty $, तब $\P(\limsup A_n) =
        0$.
  \item अगर $\sum\P(A_n) = \infty $ \emph{और यह $ A_n $ हैं
        \omterm{def:b3:probability:independence}{स्वतंत्र}}, तब $\P(\limsup A_n) = 1$.
\end{enumerate}
\end{theorem}

\begin{proof}
(1) है \cref{exo:b3:measure:4}. (2): के लिए $ N \leq M $,
स्वतंत्रता पूरकों की (\cref{exo:b3:probability:3})
देता है
\[
\P\Bigl(\bigcap_{n=N}^{M}A_n^c\Bigr) = \prod_{n=N}^M\bigl(1
- \P(A_n)\bigr) \leq
\exp\Bigl(-\sum_{n=N}^M\P(A_n)\Bigr) \xrightarrow[M \to
\infty]{} 0
\]
($1 - x \leq \eu^{-x}$; श्रृंखला अलग हो जाती है)। इसलिए
$\P\bigl(\bigcup_{n\geq N}A_n\bigr) = 1$ हरएक के लिए $ N $, और
घटता हुआ चौराहा खत्म $ N $ अभी भी संभावना है
$1$ (निरंतरता उपर से, \cref{prop:b3:measure:basics}).
\end{proof}

\begin{theorem}[कोलमोगोरोव का शून्य--एक नियम]
\label{thm:b3:probability:zeroone}
होने देना $(X_n)$ होना \omterm{def:b3:probability:independence}{स्वतंत्र} और $\mathcal T =
\bigcap_N\sigma(X_N, X_{N+1}, \dots)$  \emph{पूँछ
$\sigma $-बीजगणित}\index{पूँछ
$\sigma $-बीजगणित} (घटनाएँ
की किसी भी सीमित संख्या के प्रति असंवेदनशील $ X_n $: का अभिसरण
$\sum X_n $, का $\frac{S_n}n $, के मान $\limsup $'एस, \dots).
फिर हर $ T \in \mathcal T $ है $\P(T) \in \{0,
1\}$.\index{शून्य--एक कानून}
\end{theorem}

\begin{proof}
हल करना $ N $. $\sigma $-बीजगणित $\sigma(X_1, \dots, X_N)$ और
$\sigma(X_{N+1}, \dots)$ हैं \omterm{def:b3:probability:independence}{स्वतंत्र}: घटनाएँ निर्भर करती हैं
असंयुक्त ब्लॉक उत्पादन पर कारक बनाते हैं $\pi $-सिस्टम
(सिलेंडर $\bigcap_{i\leq N}\{X_i \in B_i\}$, सम्मान।\ परिमित
बाद के चरों पर स्थितियाँ), और डिनकिन
(\cref{thm:b3:measure:dynkin}, दो बार लागू किया गया, एक तरफ एक
समय) गुणनखंडन का विस्तार करता है। एक पूंछ घटना $ T $ में निहित है
$\sigma(X_{N+1}, \dots)$ हरएक के लिए $ N $: $ T $ है \omterm{def:b3:probability:independence}{स्वतंत्र} हरेक का $\sigma(X_1, \dots, X_N)$, इसलिए का $\sigma $-बीजगणित
वे उत्पन्न करते हैं, $\sigma(X_1, X_2, \dots)$ (डाइनकिन एक बार फिर:
का संघ $\sigma(X_1,\dots,X_N)$ एक है $\pi $-प्रणाली
इसे उत्पन्न करना)। लेकिन $ T \in \sigma(X_1, X_2, \dots)$ बहुत: $ T $
है \omterm{def:b3:probability:independence}{स्वतंत्र} \emph{खुद का}, $\P(T) = \P(T\cap T) =
\P(T)^2$: $\P(T) \in \{0, 1\}$.
\end{proof}

\section{अभिसरण के तरीके}

\begin{definition}\label{def:b3:probability:modes}
$ X_n \to X $ \emph{लगभग निश्चित रूप से} अगर $\P(X_n \to X) = 1$;
\emph{संभावना में} अगर $\P(\abs{X_n - X} \geq \varepsilon)
\to 0$ हरएक के लिए $\varepsilon > 0$; \emph{में $ L^p $} अगर
$\E\abs{X_n - X}^p \to 0$.\index{यादृच्छिक का अभिसरण
चर}
\end{definition}

\begin{proposition}\label{prop:b3:probability:modes}
(ए) ए.एस.\ अभिसरण का तात्पर्य संभाव्यता में अभिसरण से है;
(बी) $ L^p $ अभिसरण का तात्पर्य संभाव्यता में अभिसरण से है;
(सी) संभाव्यता में अभिसरण का अर्थ है ए.एस.\ अभिसरण
\emph{एक क्रम के साथ};
(डी) \omterm{def:b3:groups:normal}{सामान्य} तौर पर कोई अन्य निहितार्थ नहीं है।
\end{proposition}

\begin{proof}
(ए) $\P(\abs{X_n - X} \geq \varepsilon) \leq
\P\bigl(\sup_{m\geq n}\abs{X_m - X} \geq \varepsilon\bigr)
\downarrow \P\bigl(\limsup\{\abs{X_m - X} \geq
\varepsilon\}\bigr) = 0$ ए.एस. के तहत\ अभिसरण (निरंतरता
उपर से; लिम्सअप इवेंट में अभिसरण शामिल नहीं है)। (बी)
मार्कोव: $\P(\abs{X_n - X} \geq \varepsilon) \leq
\varepsilon^{-p}\,\E\abs{X_n - X}^p $. (सी) चुनें $ n_k $ साथ
$\P(\abs{X_{n_k} - X} \geq 2^{-k}) \leq 2^{-k}$;
बोरेल--कैंटेली (1) बनाता है $\abs{X_{n_k} - X} < 2^{-k}$
अंततः, ए.एस. (डी) टाइपराइटर (\cref{exo:b3:lp:3}) पर
$(\intcc01, \lambda)$ में एकत्रित हो जाता है $ L^1$ और संभाव्यता में
लेकिन बिंदुवार कहीं नहीं; $ n\mathbf 1_{\intoo0{1/n}} \to 0$
जैसा।\ लेकिन अंदर नहीं $ L^1$; विवरण और शेष
में प्रति उदाहरण \cref{exo:b3:probability:6}.
\end{proof}

\section{बड़ी संख्या का नियम}

लगातार, $(X_n)$ हैं \omterm{def:b3:probability:independence}{स्वतंत्र} उसी के साथ \omterm{def:b3:probability:space}{कानून}
(\emph{आई.आई.डी.}), $ S_n = X_1 + \dots + X_n $.

\begin{theorem}[बड़ी संख्या का कमजोर नियम]
\label{thm:b3:probability:wlln}
अगर $ X_1 \in L^2$, साथ $ m = \E X_1$:
\[
\P\Bigl(\Bigl|\frac{S_n}{n} - m\Bigr| \geq
\varepsilon\Bigr) \leq
\frac{\V(X_1)}{n\,\varepsilon^2}
\xrightarrow[n\to\infty]{} 0 :
\]
$\frac{S_n}n \to m $ संभाव्यता में (और में $ L^2$).\index{कानून
बड़ी संख्या में (कमजोर)}
\end{theorem}

\begin{proof}
$\E\frac{S_n}n = m $ और $\V\bigl(\frac{S_n}n\bigr) =
\frac{n\V(X_1)}{n^2}$
(\cref{thm:b3:probability:independence}); चेबीशेव।
\end{proof}

\begin{theorem}[बड़ी संख्या का मजबूत कानून]
\label{thm:b3:probability:slln}
अगर $ X_1 \in L^1$, तब\index{बड़ी संख्या का नियम (मजबूत)}
\[
\frac{S_n}{n} \xrightarrow[n\to\infty]{\text{a.s.}} \E[X_1].
\]
हम इसे यहां मजबूत परिकल्पना के तहत साबित करते हैं $ X_1 \in
L^4$; \omterm{def:b3:groups:normal}{सामान्य} मामला ($ L^1$: एटेमाडी का प्रमाण) है
सप्ताहांत की समस्या.
\end{theorem}

\begin{proof}[सबूत नीचे $\E X_1^4 < \infty $]
केन्द्रित करना ($ X_i \mapsto X_i - m $), मान लीजिए $ m = 0$. बढ़ाना:
\[
\E[S_n^4] = \sum_{i,j,k,l}\E[X_iX_jX_kX_l]
= n\,\E[X_1^4] + 3n(n-1)\,\bigl(\E[X_1^2]\bigr)^2 \leq
C\,n^2 ,
\]
तब से स्वतंत्रता और \omterm{ex:b3:groups:actions}{केंद्र}ीकरण युक्त प्रत्येक शब्द को मारता है
एक पृथक कारक ($\E[X_iX_jX_kX_l] =
\E[X_i]\E[\cdots] = 0$ जब तक कि सूचकांक युग्मित न हो जाएं: एकमात्र
बचे हुए लोग हैं $ n $ शर्तें $ i=j=k=l $ और यह $3n(n-1)$
दो अलग-अलग युग्मों वाले पद)। मार्कोव:
\[
\P\Bigl(\Bigl|\frac{S_n}n\Bigr| \geq \varepsilon\Bigr)
= \P\bigl(S_n^4 \geq n^4\varepsilon^4\bigr)
\leq \frac{Cn^2}{n^4\varepsilon^4} =
\frac{C}{\varepsilon^4n^2},
\]
सारांश: बोरेल--कैंटेली (1) प्रत्येक तर्कसंगत के लिए देता है
$\varepsilon $, वह $\abs{S_n/n} < \varepsilon $ अंततः,
ए.एस.; ऊपर से काटना $\varepsilon \in \Q_+^*$ (गिनती करके)
अनेक संभावनाएँ-$1$ घटनाएँ): $ S_n/n \to 0$ जैसा।
\end{proof}

\begin{example}[मजबूत कानून क्या खरीदता है]
\label{ex:b3:probability:sllnapps}
(ए) \emph{आवृत्तियों}: आई.आई.डी. के लिए\ सिक्का उछाल, देखा गया
शीर्षों की आवृत्ति अभिसरण के रूप में होती है\ को $ p $ --- अनुभवजन्य
संभाव्यता का औचित्य ही.
(बी) \emph{मोंटे कार्लो}\index{मोंटे कार्लो}: के लिए $ g \in
L^1(\intcc01)$ और $(U_n)$ आई.आई.डी.\ वर्दी
(\cref{thm:b3:probability:existence}),
$\frac1n\sum_{k\leq n}g(U_k) \to \int_0^1g $ ए.एस.: इंटीग्रल
नमूनाकरण द्वारा, किसी भी आयाम में, पर आया\omterm{def:b3:probability:independence}{म-स्वतंत्र}
दर $\sim n^{-1/2}$ में सटीक बनाया गया \cref{ch:b3:clt}.
(सी) \emph{\omterm{def:b3:groups:normal}{सामान्य} संख्या}: लगभग हर वास्तविक संख्या में, में है
इसका द्वि\omterm{def:b3:topology:basis}{आधार}ी विस्तार, स्पर्शोन्मुख आवृत्ति $\frac12$ वाले
(मजबूत लागू करें \omterm{def:b3:probability:space}{कानून} के अंक चर के लिए
\cref{thm:b3:probability:existence}) --- बोरेल प्रमेय, ए
के बारे में बयान \emph{प्रत्येक}दिन संख्या से साबित हुआ \omterm{def:b3:measure:measure}{उपाय}:
\cref{pb:b3:probability:1} \omterm{def:b3:complete:complete}{पूरा} करता है यह सभी \omterm{def:b3:topology:basis}{आधार}ों पर है।
\end{example}

\begin{method}\label{met:b3:probability:toolkit}
यादृच्छिक के बारे में स्पर्शोन्मुख कथनों के लिए कार्य क्रम
अनुक्रम: (1) \emph{क्या इवेंट एक टेल इवेंट है?} फिर यह
संभावना है $0$ या $1$
(\cref{thm:b3:probability:zeroone}) और एक को ही करना होगा
तय करें कि कौन सा. (2) \emph{ए.एस. सिद्ध करने के लिए\ कथन}:
बोरेल--कैंटेली --- के लिए योगयोग्य संभावनाएँ
मार्कोव/चेबीशेव-प्रकार की सीमाओं के माध्यम से ``खराब'' घटनाएँ
जो भी क्षण मौजूद हैं; स्वतंत्रता केवल के लिए आवश्यक है
विपरीत दिशा. (3) \emph{अनुवर्ती + सैंडविच}: साबित करो
एक प्रबंधनीय परिणाम के साथ अभिसरण, नियंत्रण
एकरसता या अधिकतम द्वारा बीच में दोलन
असमानताएँ --- एटेमाडी के प्रमाण का कंकाल। (4) के लिए
वितरणात्मक सीमाएँ, प्रतीक्षा करें \cref{ch:b3:clt}.
\end{method}

\begin{exercise}[$\star\star\star $]\label{exo:b3:probability:8}
(बर्नस्टीन) के लिए $ f \in \mathcal C(\intcc01)$ परिभाषित करें
बर्नस्टीन बहुपद $ B_nf(x) =
\sum_{k=0}^n\binom nkx^k(1-x)^{n-k}f\bigl(\frac kn\bigr)$.
(ए) पहचानो $ B_nf(x) = \E\bigl[f\bigl(\frac
{S_n}n\bigr)\bigr]$ के लिए $ S_n $ द्विपद $\mathcal B(n, x)$.
(बी) साबित करें $ B_nf \to f $ \emph{समान रूप से} पर $\intcc01$: विभाजित
पर $\{\abs{\frac{S_n}n - x} \leq \delta\}$ और उसके
पूरक, वर्दी का उपयोग कर निरंतरता और चेबीशेव के साथ
वर्दी बाध्य $\V(\frac{S_n}n) \leq \frac1{4n}$.
(सी) निष्कर्ष: इसका दूसरा, संभाव्य प्रमाण
वीयरस्ट्रैस सन्निकटन प्रमेय
(\cref{cor:b3:complete:weierstrass}), स्पष्ट दर के साथ
$\norm{B_nf - f}_\infty \leq \frac32\,\omega_f(n^{-1/2})$
के मापांक के लिए निरंतरता $\omega_f $ --- सिद्ध तो करो
द $ O(\omega_f(n^{-1/2}))$ रूप।
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b3:probability:9}
(कूपन संग्राहक) के कार्ड $ n $ प्रकार समान रूप से खींचे जाते हैं
प्रतिस्थापन के साथ; होने देना $ T_n $ सभी तक ड्रा की संख्या हो
प्रकार देखने को मिलते हैं।
(ए) लिखें $ T_n = \sum_{k=1}^{n}\tau_k $ साथ $\tau_k $
पैरामीटर का ज्यामितीय $\frac{n - k + 1}n $, द $\tau_k $
\omterm{def:b3:probability:independence}{स्वतंत्र}, और निष्कर्ष निकालें $\E T_n = n\,H_n \sim n\ln n $
($ H_n $ हार्मोनिक संख्या) और $\V(T_n) \leq
\frac{\pi^2}6n^2$.
(बी) चेबीशेव: $\frac{T_n}{n\ln n} \to 1$ संभावना में.
(सी) बोरेल--कैंटेली के साथ परिष्कृत करें: सीधे दिखाएं $\P(T_n >
\beta n\ln n) \leq n^{1 - \beta}$ के लिए $\beta > 1$
\emph{(संघ इस घटना पर बाध्य है कि कुछ प्रकार छूट गया है
के बाद $\beta n\ln n $ उपयोग करके खींचता है $1 - x \leq \eu^{-x}$)},
और उसका निष्कर्ष निकालो $ n = 2^m $, जैसा।\ $ T_n \leq \beta n\ln
n $ अंततः, प्रत्येक के लिए $\beta > 2$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:probability:10}
अंक निर्माण का उपयोग करना
(\cref{thm:b3:probability:existence}):
(ए) प्रत्यक्ष गणना द्वारा सत्यापित करें कि $ U = \sum b_{2k}2^{-k}$
(वर्दी के सम-अनुक्रमित अंक $\omega $) एकसमान है और
\omterm{def:b3:probability:independence}{स्वतंत्र} का $ V = \sum b_{2k-1}2^{-k}$;
(बी) निष्कर्ष निकालें \omterm{def:b3:lebesgue:measurable}{औसत दर्जे का} आक्षेप-अप-टू-शून्य-सेट के बीच
$\intcc01$ और $\intcc01^2$ संरक्षण \omterm{def:b3:measure:measure}{उपाय}, और टिप्पणी करें:
एक समान यादृच्छिक संख्या में दो (और गणनीय रूप से कई) होते हैं
\omterm{def:b3:probability:independence}{स्वतंत्र} वाले --- पीनो वक्र से तुलना करें
(\cref{pb:b3:topology:1}), जिसने प्रक्षेपात्मकता हासिल की लेकिन
माप-संरक्षण या इंजेक्शन नहीं।
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:probability:11}
(अभिलेख) चलो $(X_n)_{n\geq1}$ आई.आई.डी. हो\ साथ \omterm{def:b3:topology:continuity}{निरंतर}
वितरण फ़ंक्शन, और कहें ए \emph{अभिलेख} पर होता है
समय $ n $ अगर $ X_n > \max(X_1, \dots, X_{n-1})$ (समय $1$ है
एक रिकार्ड). होने देना $ R_n $ रिकॉर्ड सूचक बनें.
(एक शो $\P(R_n = 1) = \frac1n $ \emph{(समरूपता द्वारा, प्रत्येक
द $ n!$ के आदेश $ X_1, \dots, X_n $ समान रूप से संभावित है
और संबंधों की संभावना है $0$)}.
(बी) दिखाएँ कि $ R_n $ हैं \emph{\omterm{def:b3:probability:independence}{स्वतंत्र}} \emph{(गिनो
निर्धारित रिकॉर्ड स्थितियों के साथ संगत आदेश, या
तर्क है कि के सापेक्ष आदेश $ X_1, \dots, X_{n-1}$ है
\omterm{def:b3:probability:independence}{स्वतंत्र} के पद का $ X_n $ उनमें से)}.
(सी) बोरेल--कैंटेली से निष्कर्ष निकालें
(\cref{thm:b3:probability:borelcantelli}, दोनों आधे) वह
अपरिमित रूप से कई रिकॉर्ड होते हैं, लेकिन रिकॉर्ड पर
लगातार बार $ n, n+1$ के साथ अनंत बार घटित होता है
संभाव्यता --- तय करें कौन सा! --- और गणना करें
$\sum_n\P(R_n = 1, R_{n+1} = 1)$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:probability:12}
(सबसे लंबा हेड रन) एक निष्पक्ष सिक्के को अनंत बार पलटें, और
चलो $ L_n $ लगातार सबसे लंबे समय तक चलने की लंबाई हो
पहले के भीतर सिर $ n $ फ़्लिप.
(ए) इसे प्रत्येक के लिए दिखाएं $\varepsilon > 0$, जैसा।\ $ L_n \leq
(1 + \varepsilon)\log_2n $ अंततः \emph{(संभावना
वह लंबाई का कुछ भाग $\ell $ सबसे पहले शुरू होता है $ n $
फ्लिप्स अधिकतम है $ n2^{-\ell}$; बोरेल--कैंटेली साथ में $ n =
2^k $)}.
(बी) दिखाएँ कि ए.एस.\ $ L_n \geq (1 - \varepsilon)\log_2n $
अंततः \emph{(पहले काट लें $ n $ में पलट जाता है
$\lfloor n/\ell\rfloor $ लंबाई के असंयुक्त ब्लॉक $\ell =
\lceil(1 - \varepsilon)\log_2n\rceil $; ब्लॉक हैं
\omterm{def:b3:probability:independence}{स्वतंत्र}, प्रत्येक सर्व-प्रमुख संभाव्यता के साथ $2^{-\ell}$,
और संभावना यह है कि कोई भी सर्व-प्रमुख नहीं है
$\exp(-n2^{-\ell}/\ell)$; साथ में योग $ n = 2^k $ दोबारा)}.
(सी) समाप्त करें $\frac{L_n}{\log_2n} \to 1$ जैसे: दस लाख में
निष्पक्ष फ़्लिप के बारे में किसी को भी दौड़ने की उम्मीद करनी चाहिए $20$ सिर ---
और इसके बिना एक डेटासेट संभवतः गढ़ा हुआ है।
\end{exercise}

\section{समस्या: एटेमाडी का मजबूत कानून का प्रमाण}

\begin{problem}[{सप्ताहांत समस्या---बड़े का मजबूत कानून
आई.आई.डी. के लिए नंबर\ पूर्णांकीय चर}]
\label{pb:b3:probability:1}
कोलमोगोरोव मजबूत है \omterm{def:b3:probability:space}{कानून} --- $\frac{S_n}n \to \E X_1$ ए.एस.\
आई.आई.डी. के लिए\ $ X_n \in L^1$ --- लंबे समय तक केवल जटिल सबूत थे;
1981 में एन.~एटेमाडी ने आश्चर्यजनक अर्थव्यवस्था में से एक का उपयोग करते हुए पाया
इस अध्याय से परे कुछ भी नहीं (और कमजोर भी नहीं)। स्वतंत्रता
जोड़ीवार करने के लिए स्वतंत्रता). हम इसका पालन करते हैं. होने देना $(X_n)$ हो
जोड़ीदार \omterm{def:b3:probability:independence}{स्वतंत्र}, समान रूप से वितरित, \omterm{def:b3:lebesgue:l1}{समाकलनीय};
$ m = \E X_1$, $ S_n = X_1 + \dots + X_n $.

\medskip
\noindent\textbf{Part I --- Reductions.}
\begin{enumerate}
  \item दिखाएँ कि यह इलाज के लिए पर्याप्त है $ X_n \geq 0$
        \emph{(विभाजित करना $ X_n = X_n^+ - X_n^-$: दोनों की जांच करें
        आधे भाग फिर से जोड़े में हैं \omterm{def:b3:probability:independence}{स्वतंत्र} आई.आई.डी.\
        \omterm{def:b3:lebesgue:l1}{समाकलनीय})}. अब से मान लीजिए $ X_n \geq 0$.
  \item (काट-छांट) चलो $ Y_n = X_n\,\mathbf 1_{X_n \leq n}$
        और $ S_n^* = Y_1 + \dots + Y_n $. दिखाओ
        \[
        \sum_{n\geq1}\P(X_n \neq Y_n) =
        \sum_{n\geq1}\P(X_1 > n) \leq \E[X_1] < \infty
        \]
        (\cref{exo:b3:product:3}), और इसके माध्यम से निष्कर्ष निकालें
        बोरेल--कैंटेली वह $\frac{S_n - S_n^*}{n} \to 0$
        ए.एस.: यह साबित करने के लिए पर्याप्त है $\frac{S^*_n}n \to m $
        जैसा।
  \item दिखाओ $\E Y_n = \E\bigl[X_1\mathbf 1_{X_1\leq
        n}\bigr] \to m $ (मोनोटोन अभिसरण), इसलिए
        $\frac1n\sum_{k\leq n}\E Y_k \to m $ (सेस\`एआरओ): यह
        साबित करने के लिए पर्याप्त है $\frac{S_n^* - \E S_n^*}{n} \to 0$
        जैसा।
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The variance estimate.}
\begin{enumerate}[resume]
  \item दिखाओ
        \[
        \V(Y_n) \leq \E[Y_n^2] = \E\bigl[X_1^2\,\mathbf
        1_{X_1 \leq n}\bigr]
        \]
        और, लेयर केक का उपयोग करना
        (\cref{prop:b3:product:layercake}), कुंजी बंधी हुई
        \[
        \sum_{n\geq1}\frac{\V(Y_n)}{n^2}
        \leq \sum_{n\geq1}\frac1{n^2}\,
        \E\bigl[X_1^2\mathbf 1_{X_1\leq n}\bigr]
        \leq C\,\E[X_1] < \infty
        \]
        \emph{(योग और का आदान-प्रदान करें \omterm{def:b3:probability:space}{अपेक्षा} ---
        श्रृंखला के लिए टोनेली --- और बाध्य $\sum_{n \geq
        x}\frac1{n^2} \leq \frac2{\max(x,1)}$ भीतर के लिए
        अनुमान $ x^2\sum_{n\geq x}n^{-2} \leq 2x $)}.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Convergence along geometric
subsequences.} हल करना $\alpha > 1$ और चलो $ k_j =
\lfloor\alpha^j\rfloor $.
\begin{enumerate}[resume]
  \item जोड़ीवार प्रयोग करना स्वतंत्रता (भिन्नताएँ जोड़ें,
        \cref{thm:b3:probability:independence} --- उसे जांचें
        भिन्नताओं की संयोजकता केवल जोड़ीवार ही चाहिए
        स्वतंत्रता) और चेबीशेव, प्रत्येक के लिए दिखाएँ
        $\varepsilon > 0$:
        \[
        \sum_{j\geq1}\P\Bigl(\Bigl|
        \frac{S^*_{k_j} - \E S^*_{k_j}}{k_j}\Bigr| \geq
        \varepsilon\Bigr)
        \leq
        \frac1{\varepsilon^2}\sum_{j\geq1}\frac1{k_j^2}
        \sum_{n\leq k_j}\V(Y_n)
        = \frac1{\varepsilon^2}\sum_{n\geq1}\V(Y_n)
        \sum_{j\,:\,k_j\geq n}\frac1{k_j^2} .
        \]
  \item दिखाओ $\sum_{j : k_j \geq n}k_j^{-2} \leq
        \frac{C_\alpha}{n^2}$ \emph{(ज्यामितीय श्रृंखला;
        फर्श से सावधान रहें: $ k_j \geq \frac{\alpha^j}2$ के लिए
        $\alpha^j \geq 2$-प्रकार की देखभाल)}, और इसके साथ समाप्त करें
        प्रश्न 4 और बोरेल--कैंटेली:
        \[
        \frac{S^*_{k_j} - \E S^*_{k_j}}{k_j}
        \xrightarrow[j\to\infty]{\text{a.s.}} 0,
        \qquad\text{hence}\qquad
        \frac{S^*_{k_j}}{k_j} \to m \ \text{a.s.}
        \]
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Sandwich and conclusion.}
\begin{enumerate}[resume]
  \item के लिए $ k_j \leq n \leq k_{j+1}$, एकरसता का प्रयोग करें
        का $ S^*_n $ (गैर-नकारात्मक सारांश!) दिखाने के लिए
        \[
        \frac{k_j}{k_{j+1}}\,\frac{S^*_{k_j}}{k_j}
        \;\leq\; \frac{S^*_n}{n} \;\leq\;
        \frac{k_{j+1}}{k_j}\,\frac{S^*_{k_{j+1}}}{k_{j+1}},
        \]
        और निष्कर्ष निकालें, जैसे:
        \[
        \frac m\alpha \leq \liminf\frac{S^*_n}n \leq
        \limsup\frac{S^*_n}n \leq \alpha\,m .
        \]
  \item होने देना $\alpha \downarrow 1$ एक क्रम के साथ और
        निष्कर्ष $\frac{S_n^*}n \to m $ ए.एस., इसलिए (भाग I)
        द \emph{मज़बूत \omterm{def:b3:probability:space}{कानून} बड़ी संख्या का}:
        \[
        \boxed{\ \frac{S_n}{n}
        \xrightarrow[n\to\infty]{\text{a.s.}} \E[X_1].\ }
        \]
  \item जहां वास्तव में जोड़ीदार था स्वतंत्रता (बल्कि
        पूर्ण स्वतंत्रता) पर्याप्त? तीन स्थानों की सूची बनाइये
        जहां स्वतंत्रता-प्रकार की परिकल्पनाओं का आह्वान किया गया था।
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Dividends.}
\begin{enumerate}[resume]
  \item (बोरेल की \omterm{def:b3:groups:normal}{सामान्य} संख्या) वह दिखाओ
        $\lambda $-लगभग हर $ x \in \intcc01$ है
        \emph{हर \omterm{def:b3:topology:basis}{आधार} पर \omterm{def:b3:groups:normal}{सामान्य}} $ b \geq 2$: प्रत्येक अंक
        $0, \dots, b-1$ स्पर्शोन्मुख आवृत्ति के साथ प्रकट होता है
        $\frac1b $ \emph{(हल करना $ b $ और एक अंक, लागू करें
        मजबूत \omterm{def:b3:probability:space}{कानून} संकेतक चर के लिए --- औचित्य सिद्ध करें
        वह आधार-$ b $ एक समान चर के अंक हैं
        आई.आई.डी.\ वर्दी पर $\{0,\dots,b-1\}$ के रूप में
        \cref{thm:b3:probability:existence} --- फिर
        अनगिनत संभावनाओं को प्रतिच्छेद करें-एक
        घटनाएँ)}. एक स्पष्ट गै\omterm{def:b3:groups:normal}{र-सामान्य} संख्या प्रदर्शित करें,
        और प्रतिबिंबित करें: प्रमेय लगभग सामान्यता का दावा करता है
        सभी संख्याएँ, फिर भी सामान्यता सिद्ध कर रही हैं $\sqrt2$ या
        $\pi $ खुला रहता है.
  \item (\omterm{ex:b3:probability:sllnapps}{मोंटे कार्लो}, गारंटीकृत) पूरी तरह से उचित ठहराएं
        की विधि \cref{ex:b3:probability:sllnapps}(बी) के लिए
        $ g \in L^1(\intcc01^d)$: आई.आई.डी. का निर्माण करें\
        एकसमान नमूना चालू $\intcc01^d $ से
        \cref{thm:b3:probability:existence} और
        \cref{exo:b3:probability:10}, और बताएं कि क्या है
        मजबूत \omterm{def:b3:probability:space}{कानून} पहुंचाता है.
\end{enumerate}

\medskip
\noindent\textbf{Part VI --- What full स्वतंत्रता buys:
maximal inequalities and random series.} एटेमादी खर्च करता है
केवल जोड़ीवार स्वतंत्रता; शेष भाग इसका शोषण करते हैं
पूर्ण (पारस्परिक) संस्करण। होने देना $(Z_n)$ होना \omterm{def:b3:probability:independence}{स्वतंत्र} \omterm{ex:b3:groups:actions}{केंद्र}ित
के चर $ L^2$ और $ S_k = Z_1 + \dots + Z_k $ (एक ताजा
अंकन, से असंबंधित $ X_n $ ऊपर)।
\begin{enumerate}[resume]
  \item (कोलमोगोरोव की अधिकतम असमानता) के लिए $\varepsilon
        > 0$ सिद्ध करना
        \[
        \P\Bigl(\max_{1\leq k\leq n}\abs{S_k} \geq
        \varepsilon\Bigr) \;\leq\;
        \frac1{\varepsilon^2}\sum_{k=1}^n\V(Z_k) :
        \]
        चेबीशेव की कीमत सबसे ज्यादा खरीदी जाती है \emph{(विभाजन
        प्रथम सूचकांक के अनुसार घटना $ k $ साथ
        $\abs{S_k} \geq \varepsilon $; उस टुकड़े पर लिखो
        $ S_n^2 \geq S_k^2 + 2S_k(S_n - S_k)$ और उपयोग करें
        स्वतंत्रता गठबंधन का $(Z_1, \dots, Z_k)$
        और $(Z_{k+1}, \dots, Z_n)$,
        \cref{thm:b3:probability:independence})}. इंगित करें
        चरण जहां जोड़ीदार है स्वतंत्रता नहीं होगा
        अधिक समय पर्याप्त है.
  \item (खिनचिन-कोलमोगोरोव एक-श्रृंखला प्रमेय) निष्कर्ष निकालें:
        यदि $\sum_n\V(Z_n) < \infty $, तब $\sum_nZ_n $
        लगभग निश्चित रूप से अभिसरण होता है \emph{(दिखाएँ कि ए.एस.\ द
        आंशिक योग एक कॉची अनुक्रम बनाते हैं: चलो $ m \to
        \infty $ लागू अधिकतम असमानता में
        $ Z_{N+1}, \dots, Z_{N+m}$, तो करने दें $ N \to
        \infty $)}.
  \item (रेडमेकर श्रृंखला) चलो $(\varepsilon_n)$ हो
        आई.आई.डी.\ संकेत, $\P(\varepsilon_n = \pm1) =
        \frac12$ (\cref{thm:b3:probability:existence}), और
        चलो $(x_n)$ वास्तविक संख्याएँ बनें। बताते हैं कि
        $\sum_nx_n\varepsilon_n $ के रूप में अभिसरण करता है\ जैसे ही
        $\sum_nx_n^2 < \infty $; वह भी दिखाओ, जो भी हो
        $(x_n)$, संभावना है कि
        $\sum_nx_n\varepsilon_n $ अभिसरण है $0$ या $1$
        (\cref{thm:b3:probability:zeroone}).
  \item बातचीत, प्राथमिक रूप से. तय करना $ T_n = \sum_{k\leq
        n}x_k\varepsilon_k $ और $ s_n^2 = \sum_{k\leq
        n}x_k^2$, और मान लीजिए $ s_n \to \infty $.
        (ए) सिद्ध करें \emph{पैली--ज़िगमंड असमानता}: के लिए
        $ Z \geq 0$ साथ $\E Z^2 < \infty $ और $0 < \theta <
        1$,
        \[
        \P\bigl(Z > \theta\,\E Z\bigr) \;\geq\; (1 -
        \theta)^2\,\frac{(\E Z)^2}{\E Z^2}
        \]
        \emph{(विभाजित करना $\E Z $ स्तर पर $\theta\E Z $ और
        ऊपरी हिस्से पर कॉची-श्वार्ज़ लगाएं)}.
        (बी) दिखाओ $\E T_n^4 \leq 3s_n^4$.
        (सी) निष्कर्ष निकालना $\P\bigl(\abs{T_n} > \frac{s_n}2\bigr)
        \geq \frac3{16}$ और निष्कर्ष निकालो
        $\sum_nx_n\varepsilon_n $ विचलन के रूप में; इसलिए
        द्विभाजन
        \[
        \sum_nx_n\varepsilon_n\ \text{converges a.s.}
        \iff \sum_nx_n^2 < \infty .
        \]
  \item (यादृच्छिक हार्मोनिक श्रृंखला) इसे समाप्त करें
        $\sum_n\frac{\varepsilon_n}{n^s}$ के रूप में अभिसरित होता है\
        यदि और केवल यदि $ s > \frac12$. के लिए $\frac12 < s \leq
        1$ श्रृंखला एकाकार होती है जैसे\ जबकि
        $\sum_nn^{-s} = \infty $: यादृच्छिक संकेत उत्पन्न होते हैं
        वर्ग-मूल-शक्ति रद्दीकरण --- के साथ तुलना करें
        वैकल्पिक श्रृंखला $\sum_n\frac{(-1)^n}{n^s}$,
        जिसके लिए अभिसरण होता है \emph{प्रत्येक} $ s > 0$.
\end{enumerate}

\medskip
\noindent\textbf{Part VII --- Concentration: Hoeffding's
inequality.} मजबूत \omterm{def:b3:probability:space}{कानून} कहते हैं $\frac{S_n}n \to m $;
एकाग्रता असमानताएँ बताती हैं कि विचलन कितना असंभावित है
\emph{प्रत्येक पर निश्चित $ n $}.
\begin{enumerate}[resume]
  \item (हॉफडिंग की लेम्मा)
        (ए) दिखाओ $\cosh\lambda \leq \eu^{\lambda^2/2}$ सभी के लिए $\lambda \in \R $, दो श्रृंखलाओं की तुलना करके
        अवधिवार.
        (बी) चलो $ Z $ के साथ केन्द्रित होना $ a \leq Z \leq b $,
        $ a < b $. दिखाओ
        \[
        \E\,\eu^{\lambda Z} \leq
        \exp\Bigl(\frac{\lambda^2(b - a)^2}8\Bigr)
        \]
        \emph{(अवश्यंभावी $\eu^{\lambda z}$ पर $\intcc ab $ द्वारा
        इसका राग, ले लो अपेक्षाएं, और अध्ययन करें $\varphi(t)
        = -pt + \log(1 - p + p\eu^t)$ साथ $ p =
        \frac{-a}{b-a}$ और $ t = \lambda(b - a)$: दिखाओ
        $\varphi(0) = \varphi'(0) = 0$ और $\varphi''
        \leq \frac14$)}.
  \item (हॉफडिंग की असमानता) चलो $ X_1, \dots, X_n $ होना
        \omterm{def:b3:probability:independence}{स्वतंत्र} साथ $ a_i \leq X_i \leq b_i $ और $ S_n =
        X_1 + \dots + X_n $. साबित करो, के लिए $ t > 0$,
        \[
        \P\bigl(S_n - \E S_n \geq t\bigr) \leq
        \exp\Bigl(\frac{-2t^2}{\sum_{i=1}^n(b_i -
        a_i)^2}\Bigr),
        \]
        और निचली पूँछ के लिए भी यही बंधन है
        \emph{(घातांकीय चेबीशेव: बाध्य
        $\E\,\eu^{\lambda(S_n - \E S_n)}$ का उपयोग करते हुए
        स्वतंत्रता और प्रश्न 17, फिर अनुकूलन करें
        $\lambda > 0$)}.
  \item (मजबूत \omterm{def:b3:probability:space}{कानून}, परिबद्ध मामला, एक दर के साथ) चलो
        $ X_i $ आई.आई.डी. हो\ मूल्यों के साथ $\intcc ab $ और $ m
        = \E X_1$. दिखाओ
        \[
        \P\Bigl(\Bigl|\frac{S_n}n - m\Bigr| \geq
        \varepsilon\Bigr) \leq
        2\exp\Bigl(\frac{-2n\varepsilon^2}{(b - a)^2}\Bigr)
        \]
        और ठीक हो जाओ $\frac{S_n}n \to m $ जैसा।\ द्वारा
        बोरेल--कैंटेली: मजबूत का दूसरा प्रमाण \omterm{def:b3:probability:space}{कानून}
        परिबद्ध चरों के लिए --- कोई काट-छाँट नहीं, और
        प्रत्येक परिमित पर घातांकीय दर $ n $, लेकिन घिरा हुआ
        सारांश और पूर्ण स्वतंत्रता. तुलना करें
        एटेमाडी के साथ परिकल्पनाएँ।
  \item (\omterm{ex:b3:probability:sllnapps}{मोंटे कार्लो}, निश्चित पर गारंटी $ n $) होने देना $ g
        \colon \intcc01^d \to \intcc01$ होना \omterm{def:b3:lebesgue:measurable}{औसत दर्जे का} और
        $(U_k)$ आई.आई.डी.\ प्रश्न 11 का एकसमान नमूना.
        दिया गया $\varepsilon, \delta > 0$, दिखाओ
        \[
        n \geq \frac{\log(2/\delta)}{2\varepsilon^2}
        \implies
        \P\Bigl(\Bigl|\frac1n\sum_{k=1}^ng(U_k) -
        \int g\,\dd\lambda_d\Bigr| \geq \varepsilon\Bigr)
        \leq \delta,
        \]
        और इसके लिए सीमा का मूल्यांकन करें $\varepsilon =
        \delta = 10^{-2}$. सीमा में शामिल नहीं है $ d $:
        प्रश्न 11 और नियतिवाद से तुलना करें
        ग्रिड.
\end{enumerate}

\medskip
\noindent\textbf{Part VIII --- How big is a random walk?
Toward the iterated logarithm.} होने देना $ S_n = \varepsilon_1 +
\dots + \varepsilon_n $ से निर्मित \omterm{def:b3:groups:simple}{सरल} यादृच्छिक वॉक बनें
आई.आई.डी.\ उचित संकेत.
\begin{enumerate}[resume]
  \item (उप-गाऊसी पूँछ) दिखाएँ $\E\,\eu^{\lambda S_n} =
        (\cosh\lambda)^n \leq \eu^{n\lambda^2/2}$ और
        निष्कर्ष निकालना, के लिए $ x > 0$,
        \[
        \P(S_n \geq x) \leq \eu^{-x^2/(2n)},
        \qquad
        \P(\abs{S_n} \geq x) \leq 2\,\eu^{-x^2/(2n)} .
        \]
  \item बोरेल--कैंटेली के माध्यम से निष्कर्ष निकालें,
        \[
        \limsup_{n\to\infty}\frac{\abs{S_n}}
        {\sqrt{2n\log n}} \leq 1 \quad\text{a.s.}
        \]
        \emph{(के लिए $\eta > 0$, पूंछ सीमा का योग करें $ x =
        (1 + \eta)\sqrt{2n\log n}$, फिर खत्म हो जाओ
        $\eta = \frac1p $)}. विशेष रूप से चलना जारी रहता है
        सीएलटी स्केल $\sqrt n $ एक लघुगणकीय कारक तक
        --- क्रूड सीमा से काफी नीचे $\abs{S_n} \leq n $.
  \item दोहरीकरण के बाद $ n_j = 2^j $, दिखाओ
        \[
        \limsup_{j\to\infty}\frac{S_{n_j}}
        {\sqrt{2n_j\log\log n_j}} \leq 1 \quad\text{a.s.},
        \]
        और प्रतिबिंबित करें: द \emph{\omterm{def:b3:probability:space}{कानून} पुनरावृत्त का
        लघुगणक} (खिनचिन; हार्टमैन - \omterm{def:b3:groups:normal}{सामान्य} तौर पर विंटनर
        \omterm{ex:b3:groups:actions}{केंद्र}ित $ L^2$ सारांश) यह बताता है
        \[
        \limsup_{n\to\infty}\frac{S_n}
        {\sqrt{2n\log\log n}} = 1 \quad\text{a.s.}
        \]
        सटीक रूप से बताएं कि अनुवर्ती को क्या अलग करता है
        अनुमान अभी इसके ऊपरी आधे भाग से सिद्ध हुआ है
        कथन (किसी को नियंत्रण करना चाहिए $\max_{n_j \leq n \leq
        n_{j+1}}S_n $ प्रत्येक ब्लॉक के अंदर, जिसके लिए एक की आवश्यकता होती है
        अधिकतम असमानता \emph{घातीय} पैमाना)
        और उस प्रश्न 12 को मात्रात्मक रूप से जांचें
        उस उद्देश्य के लिए असमानता बहुत कमजोर है। निचला
        आधा हिस्सा दूसरे बोरेल--कैंटेली लेम्मा पर टिका हुआ है
        पर लागू किया गया \omterm{def:b3:probability:independence}{स्वतंत्र} ब्लॉक; दोनों हिस्से हैं
        एक समर्पित संभावना के लिए ईमानदार वर्ष 3 \omterm{def:b3:rings:content}{सामग्री}
        पाठ्यक्रम.

  \item (एक परिमित वर्ग पर एकसमान विचलन) चलो $ A_1,
        \dots, A_N $ दोहराए जाने योग्य प्रयोग में घटनाएँ बनें,
        और प्रत्येक संभाव्यता का अनुमान उसके अनुभवजन्य \omterm{def:b3:topology:basis}{आधार} पर लगाएं
        आवृत्ति $\hat p_i $ ऊपर $ n $ आई.आई.डी.\ पुनरावृत्ति.
        होफ़डिंग की असमानता को संघबद्धता के साथ जोड़कर,
        दिखाओ
        \[
        \P\Bigl(\max_{i\leq N}\,\abs{\hat p_i - \P(A_i)} >
        \varepsilon\Bigr) \;\leq\; 2N\,\eu^{-2n\varepsilon^2},
        \]
        और नमूना-आकार नियम निकालें: $ n \geq
        \frac{\ln(2N/\delta)}{2\varepsilon^2}$ गारंटी देता है
        सब $ N $ एक साथ अनुमान
        $\varepsilon $-संभावना के साथ सटीक $\geq 1 -
        \delta $. गणना करना $ n $ के लिए $ N = 10^6$, $\varepsilon =
        0.01$, $\delta = 0.05$: का लघुगणकीय मूल्य
        एकरूपता.
  \item (यादृच्छिक हार्मोनिक विंडो) दोनों का संयोजन
        यादृच्छिक श्रृंखला सिद्धांत के आधे भाग, यह दर्शाते हैं
        आई.आई.डी.\ लक्षण $(\varepsilon_n)$ शृंखला
        $\sum_n\frac{\varepsilon_n}{n^\alpha}$ जुटता है
        ए.एस.\ अगर $\alpha > \frac12$ और विचलन a.s.\ अगर
        $\alpha \leq \frac12$; निरपेक्ष के साथ विरोधाभास
        अभिसरण (जिसकी आवश्यकता है $\alpha > 1$): पर
        खिड़की $\alpha \in \intoc{\frac12}1$, अभिसरण
        वास्तव में एक संभाव्य घटना है ---
        रद्दीकरण, आकार नहीं.
\end{enumerate}
\end{problem}
