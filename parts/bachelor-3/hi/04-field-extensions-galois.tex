\chapter{फ़ील्ड एक्सटेंशन और गैलोज़ सिद्धांत}\label{ch:b3:galois}

क्या प्रत्येक समीकरण को हल किया जा सकता है? कण, द्विघात सूत्र के रूप में
और कार्डानो के घन सूत्र सुझाव देते हैं? क्या कोई किसी कोण को समद्विभाजित कर सकता है?
रूलर और कम्पास के साथ? दोनों प्रश्न, सदियों से खुले हैं
उत्तर --- नकारात्मक --- के एक विचार से \'इवरिस \omterm{def:b3:galois:galois}{गैलोइस}:
प्रत्येक बहुपद के साथ एक परिमित संलग्न करें \emph{समूह} की समरूपता का
इसकी जड़ें, और समूह से उत्तर पढ़ें। यह अध्याय बनाता है
शब्दकोश: फ़ील्ड एक्सटेंशन और \omterm{def:b3:galois:extension}{डिग्री}, खेतों को विभाजित करना और
\omterm{def:b3:galois:closure}{बीजीय समापन}, \omterm{thm:b3:galois:finitefields}{परिमित क्षेत्र} (एक संपूर्ण सिद्धांत --- और
की चक्रीयता का वादा किया $\mathbb F_q^\times $), पृथक्करण, फिर
द \omterm{thm:b3:galois:fundamental}{गैलोइस पत्राचार} खुद, पूरे सबूतों के साथ. हम कटाई करते हैं:
शास्त्रीय निर्माणों की असंभवता, की संरचना
साइक्लोटोमिक क्षेत्र, और क्विंटिक की अघुलनशीलता
कण --- \cref{ch:b3:groups}की सादगी $ A_5$ हड़ताली
इसका लक्ष्य.

\section{एक्सटेंशन, डिग्री, बीजगणितीयता}

\begin{definition}\label{def:b3:galois:extension}
ए \emph{फ़ील्ड विस्तार}\index{फ़ील्ड विस्तार} $ L/K $ एक मैदान है
$ L $ युक्त $ K $ एक उपक्षेत्र के रूप में; $ L $ तो फिर एक है $ K $-वेक्टर स्पेस,
और \emph{डिग्री}\index{डिग्री} $[L:K]$ है
इसका आयाम. विस्तार है \emph{परिमित} अगर $[L:K] <
\infty $. \emph{विशेषता} एक क्षेत्र का जनरेटर है
$\geq 0$ के कर्नेल का $\Z \to K $, $ n \mapsto n\cdot 1$: यह है
$0$ या एक प्रधान $ p $; तदनुसार $ K $ एक सबसे छोटा शामिल है
उपक्षेत्र (\emph{प्रधान क्षेत्र}) समरूपी से $\Q $ या करने के लिए $\mathbb
F_p = \Z/p\Z $.
\end{definition}

\begin{theorem}[टावर कानून]\label{thm:b3:galois:tower}
अगर $ K \subseteq L \subseteq M $, तब $[M:K] = [M:L]\,[L:K]$: अगर
$(e_i)$ का आधार है $ L $ ऊपर $ K $ और $(f_j)$ का एक आधार $ M $
ऊपर $ L $, तब $(e_if_j)$ का आधार है $ M $ ऊपर
$ K $.\index{टावर कानून}
\end{theorem}

\begin{proof}
उत्पन्न करना: $ x \in M $ लिखते हैं $ x = \sum_j \lambda_jf_j $
($\lambda_j \in L $), प्रत्येक $\lambda_j = \sum_i \mu_{ij}e_i $
($\mu_{ij} \in K $): $ x = \sum_{i,j}\mu_{ij}e_if_j $. स्वतंत्र:
$\sum_{i,j}\mu_{ij}e_if_j = 0$ पुनर्लेखन $\sum_j (\sum_i
\mu_{ij}e_i)f_j = 0$; आंतरिक राशियाँ अंदर हैं $ L $, तो गायब हो जाओ
($ f_j $ स्वतंत्र पर $ L $); फिर सब $\mu_{ij} = 0$ ($ e_i $
स्वतंत्र पर $ K $).
\end{proof}

\begin{definition}\label{def:b3:galois:algebraic}
होने देना $ L/K $ और $\alpha \in L $. यदि कुछ अशून्य $ P \in K[X]$ है
$ P(\alpha) = 0$, $\alpha $ है \emph{बीजगणितीय}\index{बीजगणितीय} ऊपर $ K $; मोनिक जनरेटर $\pi_\alpha $ की \omterm{def:b3:rings:ideal}{आदर्श}
$\{P : P(\alpha) = 0\}$ का $ K[X]$ इसका है \emph{न्यूनतम
बहुपद}\index{न्यूनतम
बहुपद}, एक
अलघुकरणीय बहुपद ($\pi = QR $ साथ $ Q(\alpha) = 0$ बल $ R $
की न्यूनतमता द्वारा स्थिर \omterm{def:b3:galois:extension}{डिग्री}). अन्यथा $\alpha $ है
\emph{ट्रान्सेंडैंटल}. हम लिखते हैं $ K(\alpha)$ सबसे छोटे के लिए
का उपक्षेत्र $ L $ युक्त $ K $ और $\alpha $, और $ K[\alpha]$ के लिए
सबसे छोटा सबरिंग.
\end{definition}

\begin{theorem}\label{thm:b3:galois:simple}
अगर $\alpha $ है \omterm{def:b3:galois:algebraic}{बीजगणितीय} ऊपर $ K $ साथ $ d = \deg\pi_\alpha $,
तब
\[
K(\alpha) = K[\alpha] \cong K[X]/(\pi_\alpha),
\qquad [K(\alpha) : K] = d,
\]
आधार के साथ $1, \alpha, \dots, \alpha^{d-1}$. इसके विपरीत, यदि
$[L:K] < \infty $, प्रत्येक $\alpha \in L $ है \omterm{def:b3:galois:algebraic}{बीजगणितीय} का \omterm{def:b3:galois:extension}{डिग्री}
$\deg \pi_\alpha $ डिवाइडिंग $[L:K]$.
\end{theorem}

\begin{proof}
मूल्यांकन $ K[X] \to L $, $ P \mapsto P(\alpha)$, छवि है
$ K[\alpha]$ और कर्नेल $(\pi_\alpha)$; तब से $\pi_\alpha $ है
अलघुकरणीय, $ K[X]/(\pi_\alpha)$ एक है \emph{मैदान}
(\cref{prop:b3:rings:primemaximal}: में \omterm{def:b3:rings:pidufd}{पीआईडी} $ K[X]$,
अलघुकरणीय एक उत्पन्न करता है अधिकतम \omterm{def:b3:rings:ideal}{आदर्श}), इसलिए $ K[\alpha]$ एक मैदान है
युक्त $ K $ और $\alpha $: यह बराबर है $ K(\alpha)$. कक्षाएं
का $1, X, \dots, X^{d-1}$ भागफल का आधार बनाएं (यूक्लिडियन)।
विभाजन), जहां से आधार और \omterm{def:b3:galois:extension}{डिग्री}. इसके विपरीत यदि $[L:K]
= n < \infty $: $1, \alpha, \dots, \alpha^n $ आश्रित हैं, दे रहे हैं
एक विनाशकारी बहुपद; तब $[K(\alpha):K] =
\deg\pi_\alpha $ विभाजित $ n $ से \omterm{thm:b3:galois:tower}{टावर कानून}.
\end{proof}

\begin{corollary}\label{cor:b3:galois:algclosed}
अगर $\alpha, \beta $ हैं \omterm{def:b3:galois:algebraic}{बीजगणितीय} ऊपर $ K $, तो हैं $\alpha \pm
\beta $, $\alpha\beta $, $\alpha/\beta $ ($\beta \ne 0$): द
के तत्व $ L $ \omterm{def:b3:galois:algebraic}{बीजगणितीय} ऊपर $ K $ का एक उपक्षेत्र बनाएं $ L $.
इसके अलावा बीजगणितीयता सकर्मक है: \omterm{def:b3:galois:algebraic}{बीजगणितीय} ऊपर \omterm{def:b3:galois:algebraic}{बीजगणितीय} है
\omterm{def:b3:galois:algebraic}{बीजगणितीय}.
\end{corollary}

\begin{proof}
$ K(\alpha, \beta) = (K(\alpha))(\beta)$ परिमित है
$ K(\alpha)$ ($\beta $ \omterm{def:b3:galois:algebraic}{बीजगणितीय} ऊपर $ K \subseteq K(\alpha)$) और
$ K(\alpha)/K $ परिमित है: द्वारा \omterm{thm:b3:galois:tower}{टावर कानून} $[K(\alpha,\beta):K] <
\infty $, और प्रत्येक तत्व $ K(\alpha, \beta)$ ---सहित
चार सूचीबद्ध --- है \omterm{def:b3:galois:algebraic}{बीजगणितीय} (\cref{thm:b3:galois:simple}).
परिवर्तनशीलता: यदि $\beta $ है \omterm{def:b3:galois:algebraic}{बीजगणितीय} ऊपर $ L $ और $ L/K $ है
\omterm{def:b3:galois:algebraic}{बीजगणितीय}, गुणांक $ c_0, \dots, c_{m-1}$ का
$\pi_{\beta/L}$ एक परिमित विस्तार उत्पन्न करें $ F = K(c_0, \dots,
c_{m-1})$ का $ K $ (दोहराया गया \omterm{thm:b3:galois:tower}{टावर कानून}), और $ F(\beta)/F $ है
परिमित: $[F(\beta):K] < \infty $, इसलिए $\beta $ है \omterm{def:b3:galois:algebraic}{बीजगणितीय} ऊपर
$ K $.
\end{proof}

\begin{example}\label{ex:b3:galois:degrees}
$[\Q(\sqrt2):\Q] = 2$, $[\Q(\sqrt[3]2):\Q] = 3$ ($ X^3 - 2$ है
अलघुकरणीय: \omterm{thm:b3:rings:criteria}{Eisenstein}), $[\Q(\zeta_p):\Q] = p - 1$ के लिए
$\zeta_p = \eu^{2\iu\pi/p}$ ($\Phi_p $ है अलघुकरणीय,
\cref{ex:b3:rings:eisenstein}). \omterm{thm:b3:galois:tower}{टावर कानून} पहले से ही एक है
हथियार: $\sqrt[3]2 \notin \Q(\sqrt2)$, तब से $3 \nmid 2$.
\end{example}

\section{बँटवारा क्षेत्र; बीजगणितीय समापन}

\begin{theorem}[खेतों का बंटवारा]\label{thm:b3:galois:splitting}
होने देना $ P \in K[X]$ अस्थिर रहो. वहाँ एक मौजूद है \emph{बंटवारा
फ़ील्ड}\index{बंटवारा
फ़ील्ड} का $ P $ ऊपर $ K $: विस्तार $ L =
K(\alpha_1, \dots, \alpha_n)$ की जड़ों से उत्पन्न होता है $ P $ जिसमें
$ P $ रैखिक कारकों में विभाजित होता है। तक अद्वितीय है
$ K $-समरूपता, और $[L:K] \leq (\deg P)!$.
\end{theorem}

\begin{proof}
\emph{अस्तित्व}, प्रेरण द्वारा $\deg P $: एक चुनें अलघुकरणीय
कारक $ Q $ का $ P $; फील्ड $ K_1 = K[X]/(Q)$ जड़ शामिल है
$\alpha_1 = \bar X $ का $ Q $, इसलिए का $ P $; लिखना $ P = (X -
\alpha_1)P_1$ ऊपर $ K_1$ और प्रेरण लागू करें $ P_1$ ऊपर $ K_1$;
\omterm{def:b3:galois:extension}{डिग्री} अधिकतम से गुणा करें $ n(n-1)\cdots = n!$.

\emph{विशिष्टता} बलवान से अनुसरण करता है \emph{समरूपता
विस्तार लेम्मा}: होने देना $\sigma \colon K \to K'$ एक समरूपता हो,
$ P \in K[X]$, $ P^\sigma $ मैप किए गए गुणांकों वाला बहुपद,
$ L, L'$ खेतों को विभाजित करना का $ P, P^\sigma $; तब $\sigma $ विस्तार करता है
एक समरूपता के लिए $ L \to L'$. प्रेरण चालू $[L:K]$: अगर $ P $ विभाजन
में $ K $, तब $ L = K $, और $ L' = K'$ ($ P^\sigma $ में विभाजित हो जाता है $ K'$,
और $ L'$ इसकी जड़ों से उत्पन्न होता है)। अन्यथा एक रूट चुनें
$\alpha \in L \setminus K $ की एक अलघुकरणीय कारक $ Q $ का $ P $
साथ $\deg Q \geq 2$; $ Q^\sigma $ एक अलघुकरणीय का कारक
$ P^\sigma $, एक जड़ के साथ $\beta \in L'$; तब
\[
K(\alpha) \cong K[X]/(Q) \xrightarrow{\ \sigma\ }
K'[X]/(Q^\sigma) \cong K'(\beta)
\]
का विस्तार $\sigma $ साथ $\alpha \mapsto \beta $. अब $ L $ एक है
विभाजन क्षेत्र का $ P $ ऊपर $ K(\alpha)$, और $ L'$ का $ P^\sigma $
ऊपर $ K'(\beta)$, साथ $[L : K(\alpha)] < [L:K]$: प्रेरण
आगे तक फैला हुआ है $ L \to L'$.
\end{proof}

\begin{definition}\label{def:b3:galois:closure}
एक मैदान $\Omega $ है \emph{बीजगणितीय रूप से बंद}\index{बीजगणितीय रूप से बंद} यदि प्रत्येक अचर बहुपद $\Omega[X]$ है
में एक जड़ $\Omega $ (इसलिए विभाजन)। एक \emph{बीजीय
समापन}\index{बीजीय
समापन} का $ K $ एक \omterm{def:b3:galois:algebraic}{बीजगणितीय}
विस्तार $\bar K/K $ साथ $\bar K $ बीजगणित बंद किया हुआ।
\end{definition}

\begin{theorem}[स्टीनिट्ज़]\label{thm:b3:galois:steinitz}
हर क्षेत्र $ K $ एक है \omterm{def:b3:galois:algebraic}{बीजगणितीय} समापन, तक अद्वितीय
$ K $-समरूपता.
\end{theorem}

\begin{proof}
\emph{अस्तित्व (आर्टिन का निर्माण)।} होने देना $ R = K[(X_f)_f]$ हो
एक चर वाला बहुपद वलय $ X_f $ प्रति अस्थिर
राक्षसी $ f \in K[X]$, और $ I $  \omterm{def:b3:rings:ideal}{आदर्श} सभी द्वारा उत्पन्न
$ f(X_f)$. $ I $ उचित है: एक संबंध $1 = \sum_{i=1}^r g_i\,
f_i(X_{f_i})$ इसमें परिमित रूप से अनेक बहुपद शामिल हैं; एक आम में
विभाजन क्षेत्र $ E $ का $ f_1\cdots f_r $ जड़ें उठाओ $\alpha_i $ का
$ f_i $ और मूल्यांकन करें $ X_{f_i} \mapsto \alpha_i $ (अन्य चर
$\mapsto 0$): $1 = 0$, बेतुका। होने देना $\mathfrak m \supseteq I $ हो
अधिकतम (\cref{thm:b3:rings:krull}; ज़ोर्न) और $ K_1 = R/\mathfrak
m $: ए \omterm{def:b3:galois:extension}{फ़ील्ड विस्तार} का $ K $ जिसमें प्रत्येक अचर $ f \in
K[X]$ अर्थात् एक जड़ है $\bar X_f $, और जो है \omterm{def:b3:galois:algebraic}{बीजगणितीय} ऊपर
$ K $ (यह द्वारा उत्पन्न होता है $\bar X_f $, प्रत्येक \omterm{def:b3:galois:algebraic}{बीजगणितीय}).
पुनरावृति: $ K \subseteq K_1 \subseteq K_2 \subseteq \cdots $, कहाँ
$ K_{n+1}$ करता है $ K_n $ क्या $ K_1$ किया $ K $, और चलो $\Omega =
\bigcup_n K_n $, एक मैदान. कोई भी अचर $ g \in \Omega[X]$ है
कुछ में इसके बहुत सारे गुणांक हैं $ K_n $; एक अलघुकरणीय
का कारक $ g $ ऊपर $ K_n $ में एक जड़ है $ K_{n+1} \subseteq
\Omega $: $\Omega $ है \omterm{def:b3:galois:closure}{बीजगणितीय रूप से बंद}, और \omterm{def:b3:galois:algebraic}{बीजगणितीय} ऊपर
$ K $ (प्रत्येक $ K_n $ परिवर्तनशीलता द्वारा है,
\cref{cor:b3:galois:algclosed}): $\Omega $ एक \omterm{def:b3:galois:closure}{बीजीय
समापन}.

\emph{विशिष्टता.} होने देना $\Omega, \Omega'$ दो हो \omterm{def:b3:galois:closure}{बीजीय
समापन}. जोड़ियों के सेट पर विचार करें $(E, \tau)$ कहाँ $ K
\subseteq E \subseteq \Omega $ और $\tau\colon E \to \Omega'$ एक है
$ K $-एम्बेडिंग, विस्तार द्वारा आदेशित; यह गैर-रिक्त है ($(K,
\mathrm{id})$) और आगमनात्मक (एक श्रृंखला का मिलन), इसलिए ज़ोर्न एक देता है
अधिकतम $(E_0, \tau_0)$. अगर $ E_0 \neq \Omega $, चुनना $\alpha \in
\Omega\setminus E_0$: $\pi_{\alpha/E_0}$ एक बहुपद के लिए मानचित्र
खत्म $\tau_0(E_0)$ एक जड़ होना $\beta $ में \omterm{def:b3:galois:closure}{बीजगणितीय रूप से
बंद} $\Omega'$, और $\tau_0$ तक फैला हुआ है $ E_0(\alpha) \to
\Omega'$ ($\alpha \mapsto \beta $), अधिकतमता का खंडन। तो
वहाँ एक है $ K $-एम्बेडिंग $\tau \colon \Omega \to \Omega'$; इसका
छवि, समरूपी से $\Omega $, है \omterm{def:b3:galois:closure}{बीजगणितीय रूप से बंद}, और
$\Omega'$ है \omterm{def:b3:galois:algebraic}{बीजगणितीय} इसके ऊपर: के लिए $ x \in \Omega'$,
$\pi_{x/\tau(\Omega)}$ विभाजित हो जाता है $\tau(\Omega)$, इसलिए $ x \in
\tau(\Omega)$. इस प्रकार $\tau $ पर है: एक समरूपता।
\end{proof}

\begin{remark}\label{rem:b3:galois:closureexamples}
के लिए $ K = \Q $ कोई अनंत मशीनरी से बच सकता है:
\omterm{def:b3:galois:algebraic}{बीजगणितीय} नंबर $\bar\Q = \{z \in \C : z \text{ algebraic over
} \Q\}$ आदमी के लिए \omterm{def:b3:galois:algebraic}{बीजगणितीय} समापन --- का एक उपक्षेत्र $\C $ द्वारा
\cref{cor:b3:galois:algclosed}, \omterm{def:b3:galois:closure}{बीजगणितीय रूप से बंद} क्योंकि $\C $
(डी'अलेम्बर्ट--गॉस, जटिल विश्लेषण द्वारा सिद्ध किया गया है
\cref{ch:b3:holomorphic}) और बहुपदों की जड़ें खत्म हो गईं $\bar\Q $
हैं \omterm{def:b3:galois:algebraic}{बीजगणितीय} ऊपर $\Q $ परिवर्तनशीलता द्वारा.
\end{remark}

\section{परिमित क्षेत्र}

\begin{theorem}\label{thm:b3:galois:finitefields}
होने देना $ p $ प्रधान बनो, $ n \geq 1$, $ q = p^n $.
\begin{enumerate}
  \item ए \omterm{thm:b3:galois:finitefields}{परिमित क्षेत्र} कार्डिनैलिटी में एक प्रमुख शक्ति है, और इसके लिए
        प्रत्येक $ q $ वहाँ बिल्कुल एक फ़ील्ड है $\mathbb F_q $ साथ
        $ q $ समरूपता तक के तत्व: विभाजन क्षेत्र का
        $ X^q - X $ ऊपर $\mathbb F_p $.\index{परिमित क्षेत्र}
  \item  \emph{फ्रोबेनियस}\index{फ्रोबेनियस} $ F
        \colon x \mapsto x^p $ का एक स्वप्रतिरूपण है $\mathbb
        F_q $, और ऑटोमोर्फिज्म समूह $\mathbb F_q $ है
        क्रम का चक्रीय $ n $, द्वारा उत्पन्न $ F $.
  \item $\mathbb F_{p^m}$ में एम्बेड करता है $\mathbb F_{p^n}$ आईएफएफ $ m \mid
        n $.
\end{enumerate}
\end{theorem}

\begin{proof}
(1) ए \omterm{thm:b3:galois:finitefields}{परिमित क्षेत्र} $ E $ विशेषता है $ p > 0$ और एक है
परिमित-आयामी $\mathbb F_p $-सदिश स्थल: $\abs E = p^n $.
इसके गुणक समूह में क्रम होता है $ q - 1$, तो हर $ x \in E $
संतुष्ट $ x^q = x $: $ E $ के होते हैं $ q $ की जड़ें $ X^q - X $,
इसलिए एक है विभाजन क्षेत्र यह खत्म हो गया $\mathbb F_p $ ---
निर्धारण $ E $ समरूपता तक
(\cref{thm:b3:galois:splitting}). इसके विपरीत, ए में \omterm{thm:b3:galois:splitting}{बंटवारा
फ़ील्ड} $ L $ का $ X^q - X $, सेट $ E $ इसकी जड़ों में से एक है
\emph{उप क्षेत्र}: $(x + y)^q = x^q + y^q $ को पुनरावृत्त करके
नवसिखुआ का सपना $(a+b)^p = a^p + b^p $ ($ p \mid \binom pk $), और
$(xy)^q = x^qy^q $, $(x^{-1})^q = (x^q)^{-1}$; यह बिल्कुल वैसा ही है $ q $
तब से तत्व $ X^q - X $ है \omterm{def:b3:galois:separable}{वियोज्य}: इसका व्युत्पन्न है
$ qX^{q-1} - 1 = -1$ (जैसा $ p \mid q $), इसके साथ सहअभाज्य, इसलिए नहीं
बार-बार जड़ें. इस प्रकार $ L = E $ है $ q $ तत्व.

(2) $ F $ एक क्षेत्र रूपवाद (नए व्यक्ति का सपना), इंजेक्शन है
(फ़ील्ड), इसलिए परिमित पर विशेषण $\mathbb F_q $. $ F^n =
\mathrm{id}$ ($ x^q = x $), और कोई छोटी शक्ति पहचान नहीं है:
$ F^m = \mathrm{id}$ मतलब सब $ q $ तत्वों की जड़ें हैं
$ X^{p^m} - X $, मजबूर करना $ p^m \geq q $. इसलिए $\langle F\rangle $ है
क्रम का चक्रीय $ n $; और कोई अन्य ऑटोमोर्फिज्म नहीं हैं
बाध्य $\abs{\operatorname{Aut}} \leq [\,\mathbb F_q :
\mathbb F_p\,] = n $ नीचे साबित हुआ
(\cref{prop:b3:galois:embeddings} साथ $ L = \mathbb F_q $, $ K =
\mathbb F_p $: ऑटोमोर्फिज्म प्राइम फ़ील्ड को ठीक करता है)।

(3) यदि $\mathbb F_{p^m} \subseteq \mathbb F_{p^n}$, द \omterm{thm:b3:galois:tower}{टावर कानून}
देता है $ p^n = (p^m)^d $: $ m \mid n $. इसके विपरीत यदि $ m \mid n $, तब
$ p^m - 1 \mid p^n - 1$ (ज्यामितीय योग), तो $ X^{p^m} - X $ विभाजित
$ X^{p^n} - X $ (घातांक पर वही तर्क: $ X^{a} - 1 \mid X^{b}
- 1$ कब $ a \mid b $), और पूर्व की जड़ें अंदर
$\mathbb F_{p^n}$ कार्डिनैलिटी का आवश्यक उपक्षेत्र बनाएं
$ p^m $ (पृथक्करण (1) के अनुसार)।
\end{proof}

\begin{theorem}[चक्रीयता]\label{thm:b3:galois:cyclic}
किसी क्षेत्र के गुणक समूह का प्रत्येक परिमित उपसमूह है
चक्रीय. विशेष रूप से $\mathbb F_q^\times \cong
\Z/(q-1)\Z $.\index{एक परिमित क्षेत्र का गुणक समूह}
\end{theorem}

\begin{proof}
होने देना $ G \leq K^\times $ परिमित हो. संरचना प्रमेय द्वारा
(\cref{cor:b3:modules:abelian}), $ G \cong \Z/d_1 \times \dots
\times \Z/d_s $ साथ $ d_1 \mid \dots \mid d_s $. प्रत्येक $ x \in G $
फिर संतुष्ट करता है $ x^{d_s} = 1$; लेकिन $ X^{d_s} - 1$ अधिकतम है
$ d_s $ क्षेत्र में जड़ें $ K $: $\abs G = d_1\cdots d_s \leq d_s $,
जबरदस्ती $ s = 1$: $ G $ चक्रीय है.
\end{proof}

\begin{example}\label{ex:b3:galois:f8}
$\mathbb F_8 = \mathbb F_2[X]/(X^3 + X + 1)$: घन में कोई नहीं है
में जड़ $\mathbb F_2$, इसलिए है अलघुकरणीय. लिखना $\omega =
\bar X $: $\mathbb F_8^\times $ क्रम का चक्रीय है $7$, इसलिए
\emph{प्रत्येक} तत्व $\neq 0, 1$ उत्पन्न करता है. के उपक्षेत्र
$\mathbb F_{p^{12}}$ का विभाजक जालक बनाएं $12$:
$\mathbb F_p, \mathbb F_{p^2}, \mathbb F_{p^3}, \mathbb F_{p^4},
\mathbb F_{p^6}, \mathbb F_{p^{12}}$ --- प्रथम, पूर्ण
का उदाहरण \omterm{thm:b3:galois:fundamental}{गैलोइस पत्राचार}.
\end{example}

\section{पृथक्करण और एम्बेडिंग}

\begin{definition}\label{def:b3:galois:separable}
एक बहुपद $ P \in K[X]$ है \emph{वियोज्य}\index{वियोज्य} यदि इसमें कोई दोहराई गई जड़ नहीं है विभाजन क्षेत्र ---
समकक्ष $\gcd(P, P') = 1$ (दोहराई गई जड़ एक \omterm{def:b3:groups:normal}{सामान्य} बात है
जड़; इसके विपरीत, के ऊपर विभाजन क्षेत्र, एक \omterm{def:b3:groups:normal}{सामान्य} जड़ है
दोहराया गया; और gcd के अंतर्गत परिवर्तन नहीं होता है \omterm{def:b3:galois:extension}{फ़ील्ड विस्तार},
\cref{cor:b3:modules:descent}का तर्क) एक \omterm{def:b3:galois:algebraic}{बीजगणितीय} तत्व
वियोज्य है यदि ऐसा है \omterm{def:b3:galois:algebraic}{न्यूनतम बहुपद} है; विस्तार $ L/K $ है
वियोज्य यदि इसके सभी तत्व हैं।
\end{definition}

\begin{proposition}\label{prop:b3:galois:perfect}
एक \emph{अलघुकरणीय} $ P \in K[X]$ है \omterm{def:b3:galois:separable}{वियोज्य} जब तक $ P' = 0$,
जो मजबूर करता है $\operatorname{char} K = p > 0$ और $ P \in K[X^p]$.
फलस्वरूप प्रत्येक \omterm{def:b3:galois:algebraic}{बीजगणितीय} के एक क्षेत्र का विस्तार
विशेषता $0$, और एक का \omterm{thm:b3:galois:finitefields}{परिमित क्षेत्र}, है \omterm{def:b3:galois:separable}{वियोज्य} (ऐसे
फ़ील्ड कहलाते हैं \emph{उत्तम}\index{उत्तम}).
\end{proposition}

\begin{proof}
$\gcd(P, P')$ विभाजित $ P $; यदि यह नहीं है $1$, अघुलनशीलता
बल $\gcd = P $ (एक स्थिरांक तक), इसलिए $ P \mid P'$ साथ $\deg
P' < \deg P $: $ P' = 0$. लिखना $ P = \sum a_kX^k $: $ ka_k = 0$
सभी के लिए $ k $, इसलिए विशेषता में $0$, $ P $ स्थिर है (बहिष्कृत);
विशेषता में $ p $, $ a_k = 0$ जब तक $ p \mid k $: $ P =
Q(X^p)$. एक से अधिक \omterm{thm:b3:galois:finitefields}{परिमित क्षेत्र}, प्रत्येक तत्व एक है $ p $-थ शक्ति
(\omterm{thm:b3:galois:finitefields}{फ्रोबेनियस} चालू है), इसलिए $ Q(X^p) = \sum b_k^p X^{pk} = (\sum
b_kX^k)^p $ क्या नहीं है अलघुकरणीय: $ P' = 0$ के लिए नहीं हो सकता
अलघुकरणीय $ P $ वहाँ या तो.
\end{proof}

\begin{proposition}[एम्बेडिंग की गिनती]\label{prop:b3:galois:embeddings}
होने देना $ L = K(\alpha_1, \dots, \alpha_r)$ परिमित हो $ K $, और
$\sigma \colon K \to \Omega $ एक में एम्बेडिंग \omterm{def:b3:galois:closure}{बीजगणितीय रूप से
बंद} मैदान. फिर के एक्सटेंशन की संख्या $\sigma $ को $ L $
अधिकतम है $[L:K]$, समानता के साथ यदि $ L/K $ है \omterm{def:b3:galois:separable}{वियोज्य}. विशेष रूप से $\abs{\operatorname{Aut}_K(L)} \leq [L:K]$.
\end{proposition}

\begin{proof}
प्रेरण चालू $[L:K]$ \omterm{def:b3:groups:simple}{सरल} चरणों के माध्यम से. के लिए $ L = K(\alpha)$: विस्तार $\tau $ इसके द्वारा निर्धारित किया जाता है $\tau(\alpha)$, जो एक होना चाहिए
में जड़ $\Omega $ का $\pi_\alpha^\sigma $; इसके विपरीत प्रत्येक ऐसे
रूट एक एक्सटेंशन देता है ($ K(\alpha) \cong K[X]/(\pi_\alpha)$).
एक्सटेंशन की संख्या की संख्या है \emph{विशिष्ट} जड़ें
का $\pi_\alpha^\sigma $ में $\Omega $: \omterm{def:b3:rings:primemaximal}{अधिक से अधिक} $\deg\pi_\alpha =
[K(\alpha):K]$, समानता के साथ यदि $\pi_\alpha $ है \omterm{def:b3:galois:separable}{वियोज्य}
(पृथक्करण) $\pi^\sigma $ और $\pi $ सहमत: जीसीडी के साथ
व्युत्पन्न को संरक्षित किया जाता है $\sigma $). \omterm{def:b3:groups:normal}{सामान्य} तौर पर, कारक $ L =
K(\alpha_1)(\alpha_2, \dots)$: का विस्तार $\sigma $ को
$ K(\alpha_1)$ संख्या $\leq [K(\alpha_1):K]$, और प्रत्येक का विस्तार होता है
$\leq [L : K(\alpha_1)]$ प्रेरण द्वारा तरीके; गुणा करें (\omterm{thm:b3:galois:tower}{टावर
कानून}). में \omterm{def:b3:galois:separable}{वियोज्य} मामले में दोनों गिनती समानताएं हैं: \omterm{def:b3:galois:algebraic}{न्यूनतम
बहुपद} बड़े मैदान के ऊपर $ K(\alpha_1)$ उनको बाँट दो
खत्म $ K $, इसलिए बने रहें \omterm{def:b3:galois:separable}{वियोज्य}.
\end{proof}

\begin{theorem}[आदिम तत्व]\label{thm:b3:galois:primitive}
हर परिमित \emph{\omterm{def:b3:galois:separable}{वियोज्य}} विस्तार \omterm{def:b3:groups:simple}{सरल} है: $ L =
K(\gamma)$ कुछ के लिए $\gamma $.\index{आदिम तत्व प्रमेय}
\end{theorem}

\begin{proof}
अगर $ K $ परिमित है, अतः है $ L $, और एक जनरेटर $\gamma $ का
चक्रीय समूह $ L^\times $ (\cref{thm:b3:galois:cyclic}) क्या यह।
होने देना $ K $ अनंत हो; प्रेरण द्वारा यह उपचार के लिए पर्याप्त है $ L =
K(\alpha, \beta)$. होने देना $ n = [L:K]$; द्वारा
\cref{prop:b3:galois:embeddings} वहाँ हैं $ n $ विशिष्ट
$ K $-एम्बेडिंग $\sigma_1, \dots, \sigma_n \colon L \to \Omega $
($\Omega $ एक \omterm{def:b3:galois:algebraic}{बीजगणितीय} समापन). बहुपद
\[
D(T)=\prod_{i<j}\bigl[\bigl(\sigma_i(\alpha)-\sigma_j(\alpha)
\bigr) + T\bigl(\sigma_i(\beta) - \sigma_j(\beta)\bigr)\bigr]
\]
समान रूप से शून्य नहीं है: एक कारक केवल तभी गायब हो जाता है जब
$\sigma_i, \sigma_j $ दोनों पर सहमति $\alpha $ और $\beta $, इसलिए आगे
$ L $ --- के लिए बहिष्कृत $ i \neq j $. जैसा $ K $ अनंत है, उठाओ $ c \in
K $ साथ $ D(c) \ne 0$: फिर $ n $ तत्वों $\sigma_i(\alpha +
c\beta)$ जोड़ीवार भिन्न हैं, इसलिए $\gamma = \alpha + c\beta $
कम से कम है $ n $ में अलग-अलग संयुग्मन होते हैं $\Omega $, अर्थात\
$\deg \pi_\gamma \geq n $: $[K(\gamma):K] \geq n = [L:K]$ बल
$ L = K(\gamma)$.
\end{proof}

\section{गैलोइस पत्राचार}

\begin{definition}\label{def:b3:galois:galois}
एक परिमित विस्तार $ L/K $ है \emph{गैलोइस}\index{गैलोइस}
यदि यह है विभाजन क्षेत्र एक का \emph{\omterm{def:b3:galois:separable}{वियोज्य}} बहुपद
खत्म $ K $. इसका \emph{गैलोइस समूह}\index{गैलोइस समूह} है
$\operatorname{Gal}(L/K) = \operatorname{Aut}_K(L)$, का समूह
फ़ील्ड ऑटोमोर्फिज्म $ L $ फिक्सिंग $ K $ बिंदुवार.
\end{definition}

\begin{proposition}\label{prop:b3:galois:galoisorder}
अगर $ L/K $ है \omterm{def:b3:galois:galois}{गैलोइस}, तब $\abs{\operatorname{Gal}(L/K)} =
[L:K]$; इसके अतिरिक्त $ L/F $ है \omterm{def:b3:galois:galois}{गैलोइस} प्रत्येक मध्यवर्ती क्षेत्र के लिए $ K
\subseteq F \subseteq L $, और हर $ F $-एम्बेडिंग $ L \to \Omega
\supseteq L $ छवि है $ L $ (\emph{साधारण अवस्था}).
\end{proposition}

\begin{proof}
होने देना $ L $ विभाजित करें \omterm{def:b3:galois:separable}{वियोज्य} $ P $ ऊपर $ K $, और एक ठीक करें \omterm{def:b3:galois:closure}{बीजीय
समापन} $\Omega \supseteq L $. $ L/K $ है \omterm{def:b3:galois:separable}{वियोज्य}: यह उत्पन्न होता है
की जड़ों द्वारा $ P $; प्रत्येक तत्व की पृथक्करणता निम्न से होती है
समानता का मामला नीचे है, लेकिन आइए हम सीधे बहस करें ---
\cref{prop:b3:galois:embeddings} जेनरेटर (जड़ों) पर लागू किया गया
का \omterm{def:b3:galois:separable}{वियोज्य} $ P $, किसका \omterm{def:b3:galois:algebraic}{न्यूनतम बहुपद} विभाजित करना $ P $)
बिल्कुल उपज देता है $[L:K]$ का विस्तार $ K \hookrightarrow \Omega $
(आगमनात्मक चरण में, \omterm{def:b3:galois:algebraic}{न्यूनतम बहुपद} की एक जड़ का $ P $
एक मध्यवर्ती क्षेत्र पर अभी भी विभाजन होता है $ P $, इसलिए है
\omterm{def:b3:galois:separable}{वियोज्य}). प्रत्येक ऐसे एम्बेडिंग $\tau \colon L \to \Omega $
की जड़ों को क्रमपरिवर्तित करता है $ P $ ($\tau $ गुणांक को ठीक करता है), और
$ L $ उनके द्वारा उत्पन्न होता है: $\tau(L) = L $. इसलिए एम्बेडिंग $=$
स्वप्रतिरूपण: $\abs{\operatorname{Gal}(L/K)} = [L:K]$. के लिए
मध्यवर्ती $ F $: $ L $ भी है विभाजन क्षेत्र का $ P $ ऊपर
$ F $, और $ P $ अवशेष \omterm{def:b3:galois:separable}{वियोज्य}: $ L/F $ है \omterm{def:b3:galois:galois}{गैलोइस}; वही
तर्क सामान्यता प्रदान करता है $ F $.
\end{proof}

\begin{lemma}[आर्टिन]\label{lem:b3:galois:artin}
होने देना $ G $ किसी क्षेत्र की ऑटोमोर्फिज्म का एक सीमित समूह बनें $ L $ और
$ K = L^G = \{x : \sigma(x) = x\ \forall\sigma \in G\}$ यह तय है
फ़ील्ड. तब $[L : L^G] \leq \abs G $.
\end{lemma}

\begin{proof}
होने देना $ n = \abs G $, $ G = \{\sigma_1, \dots, \sigma_n\}$, और
मान लीजिये $ x_1, \dots, x_{n+1} \in L $ पर रैखिक रूप से स्वतंत्र हैं
$ K $. की सजातीय रैखिक प्रणाली $ n $ में समीकरण $ n+1$
अननोंस $(c_j)$ ऊपर $ L $,
\[
\sum_{j=1}^{n+1} c_j\,\sigma_i(x_j) = 0
\qquad (i = 1, \dots, n),
\]
एक शून्येतर समाधान है; के साथ एक चुनें \emph{सबसे कम} शून्येतर
प्रविष्टियाँ, कहें $ c_1, \dots, c_r \neq 0$ (पुनः क्रमांकन), $ r \geq 2$
(एक भी $ c_j\sigma_i(x_j) = 0$ असंभव है), \omterm{def:b3:groups:normal}{सामान्य}ीकृत
$ c_r = 1$. सभी नहीं $ c_j $ रिहायश $ K $: के लिए समीकरण $\sigma_i =
\mathrm{id}$ स्वतंत्रता का खंडन करेगा; कहना $ c_1 \notin K $,
इसलिए $\tau(c_1) \ne c_1$ कुछ के लिए $\tau \in G $. आवेदन करना $\tau $ सभी को
समीकरण: चूँकि $\tau\sigma_i $ पर चलाता है $ G $, वेक्टर
$(\tau(c_j))_j $ एक और समाधान है; घटाना, $(c_j -
\tau(c_j))_j $ कम गैर-शून्य प्रविष्टियों वाला एक समाधान है (
$ r $-वीं प्रविष्टि $1 - 1 = 0$ गायब हो जाता है, पहला नहीं) और नहीं
शून्य: विरोधाभास. तो कोई भी $ n+1$ तत्व निर्भर हैं: $[L:K]
\leq n $.
\end{proof}

\begin{theorem}[गैलोज़ सिद्धांत का मौलिक प्रमेय]
\label{thm:b3:galois:fundamental}
होने देना $ L/K $ एक हो गैलोज़ विस्तार समूह के साथ $ G =
\operatorname{Gal}(L/K)$.\index{गैलोइस पत्राचार}
\begin{enumerate}
  \item $ L^G = K $.
  \item मानचित्र $ H \mapsto L^H $ और $ F \mapsto
        \operatorname{Gal}(L/F)$ परस्पर विपरीत हैं,
        के उपसमूहों के बीच समावेशन-विपरीत आक्षेप $ G $
        और मध्यवर्ती क्षेत्र $ K \subseteq F \subseteq L $;
        इसके अतिरिक्त $[L : L^H] = \abs H $ और $[L^H : K] = [G : H]$.
  \item $ H \trianglelefteq G $ आईएफएफ $ L^H/K $ है \omterm{def:b3:galois:galois}{गैलोइस}, और फिर
        प्रतिबंध प्रेरित करता है $\operatorname{Gal}(L^H/K) \cong
        G/H $.
\end{enumerate}
\end{theorem}

\begin{proof}
(1) स्पष्ट रूप से $ K \subseteq L^G $. इसके विपरीत चलो $\alpha \in L
\setminus K $; हम प्रदर्शन करते हैं $\sigma \in G $ साथ $\sigma(\alpha) \ne
\alpha $. \omterm{def:b3:galois:algebraic}{न्यूनतम बहुपद} $\pi_\alpha $ ऊपर $ K $ है \omterm{def:b3:galois:extension}{डिग्री}
$\geq 2$ और है \omterm{def:b3:galois:separable}{वियोज्य} ($ L/K $ \omterm{def:b3:galois:separable}{वियोज्य},
\cref{prop:b3:galois:galoisorder}), इसलिए इसकी एक और जड़ है
$\beta \neq \alpha $ एक में \omterm{def:b3:galois:algebraic}{बीजगणितीय} समापन $\Omega \supseteq L $.
का विस्तार करें $ K $-एम्बेडिंग $ K(\alpha) \to \Omega $, $\alpha \mapsto
\beta $, एक एम्बेडिंग के लिए $\tau\colon L \to \Omega $
(\cref{prop:b3:galois:embeddings}); सामान्यता से
(\cref{prop:b3:galois:galoisorder}) $\tau(L) = L $, इसलिए $\tau \in
G $, $\beta = \tau(\alpha) \in L $, और $\tau(\alpha) \neq
\alpha $.

(2) एक उपसमूह के लिए $ H $: $ L/L^H $ है \omterm{def:b3:galois:galois}{गैलोइस}
(\cref{prop:b3:galois:galoisorder}), और
$\operatorname{Gal}(L/L^H) \supseteq H $ तुच्छ रूप से, इसलिए $[L:L^H] =
\abs{\operatorname{Gal}(L/L^H)} \geq \abs H $; आर्टिन की लेम्मा
देता है $[L:L^H] \leq \abs H $: समानता, और
$\operatorname{Gal}(L/L^H) = H $. एक मध्यवर्ती क्षेत्र के लिए $ F $:
$ L/F $ \omterm{def:b3:galois:galois}{गैलोइस} देता है $ L^{\operatorname{Gal}(L/F)} = F $ द्वारा (1)
पर लागू किया गया $ L/F $. दोनों मानचित्र परस्पर विपरीत हैं; वे
स्पष्ट रूप से उलटा समावेशन। \omterm{def:b3:galois:extension}{डिग्री}: $[L:L^H] = \abs H $ बस
साबित हुआ, और $[L^H:K] = [L:K]/[L:L^H] = \abs G/\abs H $.

(3) के लिए $\sigma \in G $ और $ H \leq G $: $\sigma(L^H) =
L^{\sigma H\sigma^{-1}}$ (प्रत्यक्ष जाँच)। आपत्ति से,
$\sigma(L^H) = L^H $ सभी के लिए $\sigma $ आईएफएफ $ H \trianglelefteq G $.
अब यदि $ H \trianglelefteq G $, तय करना $ F = L^H $: प्रत्येक $\sigma \in
G $ की ऑटोमोर्फिज्म तक सीमित है $ F $, एक रूपवाद दे रहा है $\rho
\colon G \to \operatorname{Aut}_K(F)$ कर्नेल के साथ $\{\sigma :
\sigma\restriction_F = \mathrm{id}\} = \operatorname{Gal}(L/F) =
H $. इसलिए $ G/H $ में एम्बेड करता है $\operatorname{Aut}_K(F)$, कहाँ से
$\abs{\operatorname{Aut}_K(F)} \geq [G:H] = [F:K]$; उलटा
असमानता हमेशा बनी रहती है (\cref{prop:b3:galois:embeddings}):
$\abs{\operatorname{Aut}_K(F)} = [F:K]$ और $\rho $ चालू है. यह
देखना बाकी है $ F/K $ है \omterm{def:b3:galois:galois}{गैलोइस}: $ F $ है \omterm{def:b3:galois:separable}{वियोज्य} ऊपर $ K $
(अंदर \omterm{def:b3:galois:separable}{वियोज्य} $ L/K $), और $ F = K(\gamma)$
(\cref{thm:b3:galois:primitive}); बहुपद
$\prod_{\sigma \in G/H}\bigl(X - \sigma(\gamma)\bigr)$ (उत्पाद
विशिष्ट छवियों के ऊपर, जो निहित हैं $ F $: $\sigma(F) =
L^{\sigma H\sigma^{-1}} = L^H = F $ की सामान्यता से $ H $) है
गुणांक
द्वारा तय किया गया $ G $, इसलिए में $ K $ (1) द्वारा: यह एक है \omterm{def:b3:galois:separable}{वियोज्य} बहुपद
का $ K[X]$ द्वारा विभाजित $ F $, और इसकी जड़ें उत्पन्न होती हैं $ F $: $ F/K $ है
\omterm{def:b3:galois:galois}{गैलोइस}. इसके विपरीत, यदि $ F = L^H $ साथ $ F/K $ \omterm{def:b3:galois:galois}{गैलोइस}, की सामान्यता
$ F $ (\cref{prop:b3:galois:galoisorder}, एम्बेडिंग पर लागू किया गया
$ F \to \Omega $ के तत्वों से प्रतिबंधित है $ G $) देता है
$\sigma(F) = F $ सभी के लिए $\sigma \in G $, यानी\ $ H
\trianglelefteq G $.
\end{proof}

\section{साइक्लोटॉमिक एक्सटेंशन}

\begin{definition}\label{def:b3:galois:cyclotomic}
होने देना $ n \geq 1$ और $\zeta_n = \eu^{2\iu\pi/n}$. $ n $-वां
\emph{साइक्लोटोमिक बहुपद}\index{साइक्लोटोमिक बहुपद} है
$\Phi_n = \prod_{\gcd(k,n)=1,\ 1 \le k \le n} \bigl(X -
\zeta_n^k\bigr)$, का \omterm{def:b3:galois:extension}{डिग्री} $\varphi(n)$; की जड़ों को समूहीकृत करना
$ X^n - 1$ सटीक क्रम से, $ X^n - 1 = \prod_{d \mid n}\Phi_d $,
जो कि आगमनात्मक रूप से दर्शाता है $\Phi_n \in \Z[X]$ (यूक्लिडियन
राक्षसी पूर्णांक बहुपदों का विभाजन)।
\end{definition}

\begin{theorem}\label{thm:b3:galois:cyclotomicirred}
$\Phi_n $ है अलघुकरणीय ऊपर $\Q $; इस तरह $[\Q(\zeta_n) : \Q] =
\varphi(n)$ और
\[
\operatorname{Gal}\bigl(\Q(\zeta_n)/\Q\bigr) \;\cong\;
(\Z/n\Z)^\times,
\qquad \sigma_a(\zeta_n) = \zeta_n^a .
\]
विस्तृति $\Q(\zeta_n)/\Q $ इस प्रकार है \omterm{def:b3:galois:galois}{गैलोइस} साथ \emph{एबेलियन}
समूह।
\end{theorem}

\begin{proof}
होने देना $ f = \pi_{\zeta_n}$, इसलिए $\Phi_n = fg $ साथ $ f, g \in \Z[X]$
मोनिक (गॉस की लेम्मा \cref{lem:b3:rings:gauss}: \omterm{def:b3:rings:content}{सामग्री}
गुणा करें, सभी बहुपद मोनिक)। \emph{दावा: यदि $\zeta $ एक है
की जड़ $ f $ और $ p \nmid n $ तो फिर, प्रमुख है $\zeta^p $ एक जड़ है
का $ f $.} अन्यथा $\zeta^p $ की जड़ है $ g $ (यह एक आदिम है
$ n $- एकता की जड़), तो $\zeta $ की जड़ है $ g(X^p)$, और $ f
\mid g(X^p)$ में $\Z[X]$ (\omterm{def:b3:galois:algebraic}{न्यूनतम बहुपद}, फिर गॉस फिर से)।
मॉड कम करें $ p $: $\bar g(X^p) = \bar g(X)^p $ (\omterm{thm:b3:galois:finitefields}{फ्रोबेनियस} पर
$\mathbb F_p[X]$: गुणांकवार $ a^p = a $, और नवसिखुआ का
सपना), तो $\bar f \mid \bar g^{\,p}$: $\bar f $ और $\bar g $
एक साझा करें अलघुकरणीय कारक, और $\bar\Phi_n = \bar f\bar g $ है
एक दोहराया हुआ कारक. फिर वैसा ही होता है $ X^n - \bar 1$; लेकिन यह
व्युत्पन्न $\bar nX^{n-1}$ इसका सहअभाज्य है ($ p \nmid n $, और
$0$ जड़ नहीं है): विरोधाभास.

प्रत्येक आदिम जड़ $\zeta_n^k $ ($\gcd(k, n) = 1$) प्राप्त होता है
से $\zeta_n $ लगातार प्रमुख शक्तियों द्वारा विभाजित नहीं किया जा रहा है $ n $
(कारक $ k $): दावा प्रचारित होता है, इसलिए प्रत्येक आदिम जड़ एक है
की जड़ $ f $: $ f = \Phi_n $, अलघुकरणीय. फलस्वरूप
$[\Q(\zeta_n):\Q] = \varphi(n)$, और $\Q(\zeta_n)$ है
विभाजन क्षेत्र की \omterm{def:b3:galois:separable}{वियोज्य} $ X^n - 1$ (सभी जड़ें शक्तियां हैं
का $\zeta_n $): \omterm{def:b3:galois:galois}{गैलोइस}. एक ऑटोमोर्फिज्म $\sigma $ भेजता है $\zeta_n $
किसी अन्य आदिम जड़ की ओर $\zeta_n^{a(\sigma)}$, और $\sigma
\mapsto a(\sigma)$ में एक इंजेक्शन रूपवाद है
$(\Z/n\Z)^\times $; दोनों समूहों में व्यवस्था है $\varphi(n)$:
समरूपता.
\end{proof}

\section{शासक और दिशा सूचक यंत्र}

\begin{definition}\label{def:b3:galois:constructible}
से विमान की पहचान करें $\C $; से शुरु करें $\{0, 1\}$. एक बिंदु है
\emph{रचनात्मक}\index{रचनात्मक} यदि यह है
दो से होकर गुजरने वाली रेखाओं के बहुत से प्रतिच्छेदनों द्वारा प्राप्त किया जा सकता है
पहले से निर्मित बिंदुओं और वृत्तों पर केन्द्रित एक निर्माण
त्रिज्या के साथ दो निर्मित बिंदुओं की दूरी वाला बिंदु।
\end{definition}

\begin{theorem}[वांटज़ेल]\label{thm:b3:galois:wantzel}
$ z \in \C $ है \omterm{def:b3:galois:constructible}{रचनात्मक} अगर कोई टावर है $\Q = F_0
\subseteq F_1 \subseteq \dots \subseteq F_r $ साथ $[F_{i+1} :
F_i] = 2$ और $ z \in F_r $. विशेष रूप से, ए \omterm{def:b3:galois:constructible}{रचनात्मक} संख्या
है \omterm{def:b3:galois:algebraic}{बीजगणितीय} का \omterm{def:b3:galois:extension}{डिग्री} की एक शक्ति $2$ ऊपर
$\Q $.\index{वांटज़ेल का प्रमेय}
\end{theorem}

\begin{proof}
($\Rightarrow $) दो रेखाओं के प्रतिच्छेदन के निर्देशांक
एक उपक्षेत्र में निर्देशांक वाले बिंदुओं के माध्यम से $ F \subseteq \R $
एक रैखिक प्रणाली को हल करें $ F $: वे अंदर रहते हैं $ F $. रेखा--वृत्त
और सर्कल--सर्कल चौराहों का नेतृत्व, को खत्म करने के बाद
रैखिक भाग (दो वृत्त समीकरणों को घटाने पर एक रेखा प्राप्त होती है),
एक द्विघात समीकरण पर $ F $: नए निर्देशांक निहित हैं $ F $
या में $ F(\sqrt d)$ कुछ के लिए $ d \in F $, $ d > 0$. प्रेरण द्वारा,
प्रत्येक निर्मित बिंदु में द्विघात के टॉवर में निर्देशांक होते हैं
का विस्तार $\Q $; और $ z = x + \iu y $ एक द्विघात मीनार में है
भी (बगल $\iu $: एक और द्विघात कदम)। \omterm{def:b3:galois:extension}{डिग्री}
परिणाम: $[\Q(z):\Q]$ विभाजित $[F_r : \Q] = 2^r $ (\omterm{thm:b3:galois:tower}{टावर
कानून}).

($\Leftarrow $) द \omterm{def:b3:galois:constructible}{रचनात्मक} संख्याएँ एक फ़ील्ड बनाएं: रकम और
समांतर चतुर्भुज द्वारा अंतर (समानांतर हैं \omterm{def:b3:galois:constructible}{रचनात्मक}:
लंबों को दो बार गिराना और उठाना --- शास्त्रीय
एक बिंदु के माध्यम से लंबवत एक वृत्त और दो चाप का उपयोग करता है);
थेल्स के अवरोधन विन्यास द्वारा उत्पाद और भागफल
(लंबाई दी गई है $ a, b $ CONSTRUCT $ ab $ और $ a/b $ समान के साथ
दो किरणों पर त्रिकोण)। और मैदान चौक के नीचे बंद है
जड़ें: के लिए $ a > 0$, व्यास का वृत्त $1 + a $ और
जंक्शन बिंदु पर लंबवत ऊंचाई पर मिलते हैं $\sqrt a $
(एक समकोण त्रिभुज में ऊंचाई-ज्यामितीय-माध्य संबंध); एक के लिए
जटिल $ w = \rho\eu^{\iu\theta}$, निर्माण $\sqrt\rho $ और
समद्विभाजित $\theta $ (कोण द्विभाजन एक कम्पास निर्माण है)।
एक द्विघात मीनार के सदस्यों के वास्तविक और काल्पनिक भाग होते हैं
इसलिए \omterm{def:b3:galois:constructible}{रचनात्मक} टावर पर प्रेरण द्वारा: प्रत्येक चरण
एक द्विघात की जड़ों से जुड़ता है, जो क्षेत्र संचालन द्वारा व्यक्त किया जा सकता है
और पहले से निर्मित संख्या का एक वर्गमूल (द
द्विघात सूत्र; विशेषता में $0$).
\end{proof}

\begin{corollary}\label{cor:b3:galois:impossible}
तीन शास्त्रीय समस्याएं शासक द्वारा हल नहीं की जा सकतीं और
दिशा सूचक यंत्र:\index{शासक और कम्पास}
\begin{enumerate}
  \item \emph{घन का दोहराव}: $\sqrt[3]2$ है \omterm{def:b3:galois:extension}{डिग्री}
        $3$, की शक्ति नहीं $2$.
  \item \emph{कोण का त्रिखंड}: त्रिविभाजन $60^\circ $
        आवश्यक है $\cos 20^\circ $, की एक जड़ अलघुकरणीय $8X^3
        - 6X - 1$: \omterm{def:b3:galois:extension}{डिग्री} $3$.
  \item \emph{वृत्त का वर्ग करना}: $\sqrt\pi $ पारलौकिक है
        ($\pi $ है --- लिंडमैन का प्रमेय, यहां स्वीकार किया गया: यह
        प्रमाण पारगमन सिद्धांत के एक पाठ्यक्रम से संबंधित है)।
\end{enumerate}
साथ ही, नियमित $ n $-गोन है \omterm{def:b3:galois:constructible}{रचनात्मक} आईएफएफ
$\varphi(n)$ की एक शक्ति है $2$ (गॉस--वैंटज़ेल; ``अगर'' का उपयोग करता है
सप्ताहांत समस्या की विधि, ``केवल यदि'' है
\cref{thm:b3:galois:wantzel} के लिए आवेदन किया $\zeta_n $, का \omterm{def:b3:galois:extension}{डिग्री}
$\varphi(n)$). के लिए $ n = 7$: $\varphi(7) = 6$: नियमित
सप्तकोण असंभव है; के लिए $ n = 17$: $\varphi(17) = 16 = 2^4$:
\omterm{def:b3:galois:constructible}{रचनात्मक} --- सप्ताहांत की समस्या इसका निर्माण करती है।
\end{corollary}

\begin{proof}
(1) $ X^3 - 2$ है अलघुकरणीय (\omterm{thm:b3:rings:criteria}{Eisenstein}). (2) से $\cos
3\theta = 4\cos^3\theta - 3\cos\theta $ साथ $3\theta =
60^\circ $: $8c^3 - 6c = 1$ के लिए $ c = \cos 20^\circ $; घन
$8X^3 - 6X - 1$ कोई तर्कसंगत जड़ नहीं है (उम्मीदवार)। $\pm1, \pm\frac
12, \pm\frac14, \pm\frac18$ विफल), इसलिए है अलघुकरणीय: \omterm{def:b3:galois:extension}{डिग्री}
$3$. एक जनरल $60^\circ $ कोण है \omterm{def:b3:galois:constructible}{रचनात्मक}, तो एक त्रिभाजक
निर्माण करेगा $ c $. (3) यदि $\sqrt\pi $ थे \omterm{def:b3:galois:constructible}{रचनात्मक} यह
होगा \omterm{def:b3:galois:algebraic}{बीजगणितीय}, इसलिए भी $\pi $. $ n $-गॉन स्टेटमेंट: द
\omterm{def:b3:galois:extension}{डिग्री} का $\zeta_n $ है $\varphi(n)$
(\cref{thm:b3:galois:cyclotomicirred}); आवश्यकता इस प्रकार है
वांटज़ेल; पर्याप्तता के लिए, \omterm{def:b3:galois:galois}{गैलोइस समूह}, आदेश का एबेलियन
$2^m $, सूचकांक की एक श्रृंखला को स्वीकार करता है-$2$ उपसमूह (एक परिमित) $2$-समूह
करता है: \cref{exo:b3:groups:10}), जिनके निश्चित फ़ील्ड एक बनाते हैं
द्विघात टावर पर समाप्त होता है $\Q(\zeta_n)$
(\cref{thm:b3:galois:fundamental}); द्वारा समाप्त करें
\cref{thm:b3:galois:wantzel}.
\end{proof}

\section{कट्टरपंथियों द्वारा समाधानशीलता}

\begin{definition}\label{def:b3:galois:radical}
विस्तार $ L/K $ (विशेषता $0$ इस पूरे खंड में)
है \emph{मौलिक} अगर कोई टावर है $ K = F_0 \subseteq \dots
\subseteq F_r = L $ साथ $ F_{i+1} = F_i(\alpha_i)$,
$\alpha_i^{n_i} \in F_i $: प्रत्येक चरण एक से जुड़ता है $ n_i $-वाँ जड़. ए
बहुपद $ P \in K[X]$ है \emph{द्वारा हल करने योग्य
कट्टरपंथी}\index{द्वारा हल करने योग्य
कट्टरपंथी} यदि यह है विभाजन क्षेत्र है
के कुछ मौलिक विस्तार में निहित है $ K $.
\end{definition}

\begin{lemma}\label{lem:b3:galois:cyclicsteps}
होने देना $ K $ एक आदिम शामिल है $ n $- एकता की जड़ $\zeta $, अर्थात\
$\zeta $ आदेश की $ n $ में $ K^\times $, और $ a \in K^\times $. तब
$ K(\sqrt[n]a)/K $ है \omterm{def:b3:galois:galois}{गैलोइस} साथ \emph{चक्रीय} समूह। इसके विपरीत
--- नीचे की आवश्यकता नहीं है --- प्रत्येक चक्रीय विस्तार \omterm{def:b3:galois:extension}{डिग्री} $ n $ है
इस रूप का. इसके अतिरिक्त $ K(\zeta_n)/K $ है \omterm{def:b3:galois:galois}{गैलोइस} साथ
\emph{एबेलियन} समूह, किसी के लिए $ K $ विशेषता का $0$.
\end{lemma}

\begin{proof}
$ X^n - a $ है \omterm{def:b3:galois:separable}{वियोज्य} ($\gcd $ साथ $ nX^{n-1}$: $ a \neq 0$) और
में विभाजित हो जाता है $ K(\alpha)$, $\alpha^n = a $: इसकी जड़ें हैं
$\zeta^k\alpha \in K(\alpha)$. इसलिए $ K(\alpha)/K $ है \omterm{def:b3:galois:galois}{गैलोइस}; वो नक्शा $\sigma \mapsto \sigma(\alpha)/\alpha \in \mu_n = \langle
\zeta\rangle $ एक इंजेक्शन रूपवाद है ($\sigma\tau(\alpha) =
\sigma(\tau(\alpha)/\alpha \cdot \alpha) =
\tau(\alpha)/\alpha\cdot\sigma(\alpha)$, भागफल अंदर है
$ K $), एक चक्रीय समूह में: $\operatorname{Gal}$ चक्रीय है. द
बातचीत कुमेर सिद्धांत है, जिसकी हमें आवश्यकता नहीं होगी (देखें)।
नीचे टिप्पणी करें)। के लिए $ K(\zeta_n)$: यह विभाजित करता है \omterm{def:b3:galois:separable}{वियोज्य} $ X^n -
1$, और $\sigma \mapsto a(\sigma)$ साथ $\sigma(\zeta_n) =
\zeta_n^{a(\sigma)}$ समूह को एबेलियन में एम्बेड करता है
$(\Z/n\Z)^\times $ के रूप में \cref{thm:b3:galois:cyclotomicirred}
(इंजेक्शन की केवल आवश्यकता है $\zeta_n $ की जड़ें उत्पन्न करने के लिए
एकता शामिल है)।
\end{proof}

\begin{theorem}[गैलोइस]\label{thm:b3:galois:solvable}
होने देना $ K $ विशेषतायुक्त हो $0$ और $ P \in K[X]$ साथ \omterm{thm:b3:galois:splitting}{बंटवारा
फ़ील्ड} $ L $. अगर $ P $ है कट्टरपंथियों द्वारा हल करने योग्य, तब
$\operatorname{Gal}(L/K)$ एक है हल करने योग्य समूह. (विपरीत यह है
सच भी है; हमें इसकी आवश्यकता नहीं होगी।)\index{गैलोइस की सॉल्वेबिलिटी
प्रमेय}
\end{theorem}

\begin{proof}
स्टेप 1: \emph{को बड़ा करें मौलिक टावर से ए \omterm{def:b3:galois:galois}{गैलोइस} एक।} होने देना $ L
\subseteq M $ साथ $ M/K $ मौलिक, मूल घातांक के साथ $ n_1, \dots,
n_r $ और $ n = n_1\cdots n_r $. पहले जोड़ो $\zeta_n $: टावर
$ K \subseteq K(\zeta_n) \subseteq M(\zeta_n)$ अब भी है मौलिक
($\zeta_n $ एकता की जड़ है: ए मौलिक कदम, $\zeta_n^n = 1$),
और पहले से आगे इसके चरण युक्त क्षेत्रों पर होते हैं
एकता की जड़ों की जरूरत है. अगला, बदलें $ M(\zeta_n)$ द्वारा
मिश्रित $ N $ के सभी $\sigma(M(\zeta_n))$, $\sigma $ से लेकर
(अंततः अनेक) $ K $-के एम्बेडिंग $ M(\zeta_n)$ एक निश्चित में
\omterm{def:b3:galois:algebraic}{बीजगणितीय} समापन: $ N $ है विभाजन क्षेत्र के उत्पाद का
\omterm{def:b3:galois:algebraic}{न्यूनतम बहुपद} एक जनरेटिंग सेट का (विशेषता) $0$:
परिमित और \omterm{def:b3:galois:separable}{वियोज्य}), इस तरह $ N/K $ है \omterm{def:b3:galois:galois}{गैलोइस}; और $ N $ है
मौलिक ऊपर $ K $: प्रत्येक $\sigma(M(\zeta_n))$ है मौलिक ऊपर $ K $
(आवेदन करना $\sigma $ एक को मौलिक टावर), और का एक सम्मिश्रण मौलिक
एक्सटेंशन है मौलिक (टावरों को संयोजित करें: यदि $ F'/K $ है
मौलिक टावर से सटे हुए $\beta_j $, तब $ F''(\beta_j)$-प्रकार
चरण बाकी हैं मौलिक किसी भी बड़े आधार पर)।

चरण दो: \emph{पढ़ना गलने की योग्यता से परे \omterm{def:b3:galois:galois}{गैलोइस} टावर.} तो मान लीजिये
$ L \subseteq N $, $ N/K $ \omterm{def:b3:galois:galois}{गैलोइस} और मौलिक टावर के साथ $ K
\subseteq K(\zeta_n) = E_0 \subseteq E_1 \subseteq \dots
\subseteq E_s = N $, प्रत्येक $ E_{i+1} = E_i(\sqrt[n_i]{a_i})$ साथ
$\zeta_{n_i} \in E_0 \subseteq E_i $. होने देना $ G =
\operatorname{Gal}(N/K)$ और $ G_i = \operatorname{Gal}(N/E_i)$:
घटती हुई शृंखला $ G \supseteq G_0 \supseteq G_1 \supseteq
\dots \supseteq G_s = \{e\}$. प्रत्येक $ E_{i+1}/E_i $ है \omterm{def:b3:galois:galois}{गैलोइस} के साथ
चक्रीय समूह (\cref{lem:b3:galois:cyclicsteps}), तो द्वारा
मौलिक प्रमेय को लागू किया गया गैलोज़ विस्तार $ N/E_i $
(\cref{thm:b3:galois:fundamental}(3), परिवेश समूह के साथ $ G_i $):
$ G_{i+1} \trianglelefteq G_i $ साथ $ G_i/G_{i+1} \cong
\operatorname{Gal}(E_{i+1}/E_i)$ चक्रीय. उसी प्रकार $ E_0/K $ है
\omterm{def:b3:galois:galois}{गैलोइस} एबेलियन समूह के साथ $ G/G_0$
(\cref{lem:b3:galois:cyclicsteps}). श्रृंखला प्रदर्शित करती है $ G $ जैसा
\omterm{def:b3:groups:derived}{व्याख्या करने योग्य} (\cref{prop:b3:groups:derived}). अंत में
$\operatorname{Gal}(L/K)$ एक है \emph{भागफल} का $ G $: $ L/K $ है
\omterm{def:b3:galois:galois}{गैलोइस} ($ P $ \omterm{def:b3:galois:separable}{वियोज्य} विशेषता में $0$) और प्रतिबंध $ G
\to \operatorname{Gal}(L/K)$ चालू है
(\cref{thm:b3:galois:fundamental}(3) साथ $ H =
\operatorname{Gal}(N/L)$); के भागफल हल करने योग्य समूह हैं
\omterm{def:b3:groups:derived}{व्याख्या करने योग्य}.
\end{proof}

\begin{corollary}[क्विंटिक की अघुलनशीलता]
\label{cor:b3:galois:quintic}
के बहुपद हैं \omterm{def:b3:galois:extension}{डिग्री} $5$ ऊपर $\Q $ वह नहीं है
कट्टरपंथियों द्वारा हल करने योग्य: उदाहरण के लिए $ X^5 - 4X + 2$, किसका \omterm{def:b3:galois:galois}{गैलोइस
समूह} है $ S_5$ (\cref{exo:b3:galois:11}), ए न सुलझ सकने वाला समूह
(\cref{cor:b3:groups:snnotsolvable}). में कोई \omterm{def:b3:groups:normal}{सामान्य} फार्मूला नहीं
कण के लिए मौजूद हो सकता है \omterm{def:b3:galois:extension}{डिग्री} $\geq 5$.\index{क्विंटिक,
अघुलनशीलता}
\end{corollary}

\begin{remark}\label{rem:b3:galois:converse}
का व्युत्क्रम \cref{thm:b3:galois:solvable} --- ए \omterm{def:b3:groups:derived}{व्याख्या करने योग्य}
\omterm{def:b3:galois:galois}{गैलोइस समूह} तात्पर्य गलने की योग्यता द्वारा कण ---से सिद्ध होता है
उतरते हुए व्युत्पन्न श्रृंखला और प्रत्येक चक्रीय विस्तार दिखा रहा है
(एकता की पर्याप्त जड़ों के साथ) है मौलिक, के जरिए \emph{लैग्रेंज
विलायक}; यह बताता है क्यों \omterm{def:b3:galois:extension}{डिग्री} $2, 3, 4$ सूत्र हैं:
$ S_2, S_3, S_4$ हैं \omterm{def:b3:groups:derived}{व्याख्या करने योग्य}
(\cref{ex:b3:groups:solvableexamples}). हम इसे स्वीकार करना छोड़ देते हैं
यह स्तर; एक पूर्ण उपचार एक मास्टर कोर्स से संबंधित है, लेकिन
\cref{exo:b3:galois:8} इसे घन के लिए ठोस बनाता है।
\end{remark}

\section{अभ्यास}

\begin{exercise}[$\star $]\label{exo:b3:galois:1}
दिखाओ $[\Q(\sqrt2, \sqrt3):\Q] = 4$, वह $\Q(\sqrt2 + \sqrt3) =
\Q(\sqrt2, \sqrt3)$, और गणना करें \omterm{def:b3:galois:algebraic}{न्यूनतम बहुपद} का
$\sqrt2 + \sqrt3$ ऊपर $\Q $.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b3:galois:2}
होने देना $\alpha = \sqrt[3]2$. बताते हैं कि $\Q(\alpha)/\Q $ नहीं है
\omterm{def:b3:groups:normal}{सामान्य} (एक एम्बेडिंग प्रदर्शित करें $\Q(\alpha) \to \C $ जिसकी छवि है
नहीं $\Q(\alpha)$), निर्धारित करें विभाजन क्षेत्र $ L $ का $ X^3 -
2$ और $[L:\Q]$, और जांचें $\operatorname{Aut}_\Q(\Q(\alpha)) =
\{\mathrm{id}\}$: के लिए गै\omterm{def:b3:galois:galois}{र-गैलोइस} एक्सटेंशन, ऑटोमोर्फिज्म
समूह की तुलना में बहुत छोटा हो सकता है \omterm{def:b3:galois:extension}{डिग्री}.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b3:galois:3}
निर्माण $\mathbb F_9$ जैसा $\mathbb F_3[X]/(X^2+1)$ और एक खोजें
का जनरेटर $\mathbb F_9^\times $. मोनिक की सूची बनाएं अलघुकरणीय
के बहुपद \omterm{def:b3:galois:extension}{डिग्री} $1, 2, 3$ ऊपर $\mathbb F_2$, और सत्यापित करें
$ X^8 - X = X(X+1)(X^3+X+1)(X^3+X^2+1)$ ऊपर $\mathbb F_2$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:galois:4}
(ए) \omterm{def:b3:modules:module}{मॉड्यूल}ो की सभी आदिम जड़ें खोजें $7$ और \omterm{def:b3:modules:module}{मॉड्यूल}ो $11$ (अर्थात्\
के जनरेटर $\mathbb F_7^\times $, $\mathbb F_{11}^\times $).
(बी) उसके लिए दिखाओ $ p $ विषम, $ x \in \mathbb F_p^\times $ एक है
वर्ग यदि $ x^{(p-1)/2} = 1$ (यूलर का मानदंड), और पुनर्प्राप्त करें
के लिए मानदंड $-1$ का \cref{pb:b3:rings:1}.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:galois:5}
बताते हैं कि $\mathbb F_{p^m} \cap \mathbb F_{p^n} = \mathbb
F_{p^{\gcd(m,n)}}$ और $\mathbb F_{p^m}\mathbb F_{p^n} = \mathbb
F_{p^{\operatorname{lcm}(m,n)}}$ एक निश्चित के अंदर \omterm{def:b3:galois:closure}{बीजीय
समापन} $\bar{\mathbb F}_p $, और वर्णन करें
$\operatorname{Gal}(\mathbb F_{p^n}/\mathbb F_{p^m})$ के लिए $ m
\mid n $.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:galois:6}
होने देना $ I_d(q)$ मोनिक की संख्या हो अलघुकरणीय के बहुपद
\omterm{def:b3:galois:extension}{डिग्री} $ d $ ऊपर $\mathbb F_q $. सिद्ध करना
\[
X^{q^n} - X \;=\; \prod_{d \mid n}\ \prod_{P \text{ irred.
monic, } \deg P = d} P ,
\qquad\text{hence}\qquad
q^n = \sum_{d \mid n} d\, I_d(q).
\]
परिणाम निकालना $ I_1, I_2, I_3, I_4$ स्पष्ट रूप से, और $ I_d(q) \geq 1$ हरएक के लिए $ d $ (इसलिए एक्सटेंशन $\mathbb F_{q^d}/\mathbb F_q $ के रूप में मौजूद हैं
भागफल $\mathbb F_q[X]/(P)$ सभी के लिए $ d $).
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:galois:7}
ठानना $\operatorname{Gal}(\Q(\sqrt2,\sqrt3)/\Q)$ और
मध्यवर्ती क्षेत्रों की पूरी जाली। के लिए वही प्रश्न
विभाजन क्षेत्र का $(X^2-2)(X^2-3)(X^2-6)$ --- आपको क्या लगा?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:galois:8}
(घन, उसके समूह द्वारा हल किया गया) चलो $ P = X^3 + pX + q \in
\Q[X]$ होना अलघुकरणीय जड़ों के साथ $ x_1, x_2, x_3$ और \omterm{thm:b3:galois:splitting}{बंटवारा
फ़ील्ड} $ L $. होने देना $\delta = (x_1 - x_2)(x_1 - x_3)(x_2 - x_3)$ और
$\Delta = \delta^2 = -4p^3 - 27q^2$ (इस शास्त्रीय को स्वीकार करें
सममित कार्यों का विस्तार करके इसे पहचानें या सत्यापित करें)।
(एक शो $\operatorname{Gal}(L/\Q) \cong A_3$ या $ S_3$, के अनुसार
चाहे $\Delta $ में एक वर्ग है या नहीं है $\Q $.
(बी) के साथ $ j = \zeta_3$, लैग्रेंज रिज़ॉल्वेंट्स को परिभाषित करें $ u = x_1
+ jx_2 + j^2x_3$ और $ v = x_1 + j^2x_2 + jx_3$. दिखाओ $ u^3 + v^3
= -27q $ और $ uv = -3p $, और के लिए हल करें $ u^3, v^3$: कार्डानो का
सूत्र छूट जाते हैं. जहाँ किया गलने की योग्यता का $ S_3$ इस्तेमाल किया जाना?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:galois:9}
में $\Q(\zeta_5)$:दिखाएँ कि अद्वितीय द्विघात उपक्षेत्र है
$\Q(\sqrt5)$, गॉस रकम के माध्यम से $\eta_0 = \zeta_5 + \zeta_5^4$,
$\eta_1 = \zeta_5^2 + \zeta_5^3$: गणना करें $\eta_0 + \eta_1$ और
$\eta_0\eta_1$, और निष्कर्ष निकालें $\cos\frac{2\pi}5 = \frac{\sqrt5 -
1}4$. निष्कर्ष निकालिए कि नियमित पंचकोण है \omterm{def:b3:galois:constructible}{रचनात्मक}.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b3:galois:10}
होने देना $ K = \mathbb F_p(S, T)$ (दो में तर्कसंगत कार्य
अनिश्चित) और $ L = K(S^{1/p}, T^{1/p})$.
(एक शो $[L:K] = p^2$ ओर वो $\alpha^p \in K $ हरएक के लिए
$\alpha \in L $.
(बी) उसका निष्कर्ष निकालें $ L/K $ है \emph{नहीं} \omterm{def:b3:groups:simple}{सरल}: कोई आदिम तत्व नहीं
मौजूद है--- अविभाज्यता के लिए घातक है
\cref{thm:b3:galois:primitive}.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b3:galois:11}
होने देना $ P = X^5 - 4X + 2$ और $ G $ इसका \omterm{def:b3:galois:galois}{गैलोइस समूह} ऊपर $\Q $,
पर अभिनय $5$ जड़ें.
(एक शो $ P $ है अलघुकरणीय, और निष्कर्ष निकालें $5 \mid \abs G $; समाप्त करें कि $ G $ एक शामिल है $5$-साइकिल (कॉची,
\cref{thm:b3:groups:cauchy}).
(बी) की विविधताओं का अध्ययन करके दिखाएँ $ x \mapsto x^5 - 4x +
2$, वह $ P $ बिलकुल है $3$ असली जड़ें; उस कॉम्प्लेक्स का अनुमान लगाएं
संयुग्मन एक स्थानान्तरण तक सीमित है $ G $.
(सी) दिखाएँ कि एक उपसमूह $ S_5$ एक ट्रांसपोज़िशन युक्त और
ए $5$-चक्र है $ S_5$ \emph{(स्थानांतरण को संयुग्मित करें
चक्र की शक्तियां)}. निष्कर्ष निकालना $ G \cong S_5$ और, साथ
\cref{thm:b3:galois:solvable}, वह $ P $ क्या नहीं है \omterm{def:b3:galois:radical}{द्वारा हल करने योग्य
कट्टरपंथी}.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b3:galois:12}
(डायहेड्रल चतुर्थक) आइए $\alpha = \sqrt[4]2$ और $ L =
\Q(\alpha, \iu)$, द विभाजन क्षेत्र का $ X^4 - 2$ ऊपर $\Q $.
(एक शो $[L : \Q] = 8$ ओर वो $ G =
\operatorname{Gal}(L/\Q)$ द्वारा उत्पन्न होता है $\sigma\colon
\alpha \mapsto \iu\alpha,\ \iu \mapsto \iu $ और जटिल
संयुग्मन $\tau $, साथ $\sigma^4 = \tau^2 = e $ और
$\tau\sigma\tau = \sigma^{-1}$: $ G \cong D_4$.
(बी) के उपसमूह जाली की सूची बनाएं $ D_4$ (दस उपसमूह) और
प्रत्येक को उसके निश्चित क्षेत्र से मिलाएँ; इसे विशेष रूप से सत्यापित करें
$\Q(\sqrt2)$, $\Q(\iu)$, $\Q(\iu\sqrt2)$ तीन हैं
द्विघात उपक्षेत्र, और पता लगाएं $\Q(\alpha)$,
$\Q(\iu\alpha)$, $\Q(\sqrt2, \iu)$.
(सी) कौन से मध्यवर्ती क्षेत्र हैं \omterm{def:b3:galois:galois}{गैलोइस} ऊपर $\Q $? मिलान
के विरुद्ध आपका उत्तर \omterm{def:b3:groups:normal}{सामान्य} उपसमूह का $ D_4$, और
समझाओ क्यों $\Q(\alpha)/\Q $ जबकि विफल रहता है $\Q(\sqrt2)/\Q $
सफल होता है.
\end{exercise}

\section{समस्या: गॉस और नियमित 17-गॉन}

\begin{problem}[{सप्ताहांत समस्या --- की रचनाशीलता
17-गॉन}]\label{pb:b3:galois:1}
30 मार्च 1796 को उन्नीस वर्षीय गॉस ने दिखाया कि
नियमित $17$-गोन है \omterm{def:b3:galois:constructible}{रचनात्मक} --- पर पहली प्रगति
प्राचीन काल से प्रश्न. हम उसकी गणना का पुनर्निर्माण करते हैं
इस अध्याय के उपकरण. तय करना $\zeta = \eu^{2\iu\pi/17}$, $ L =
\Q(\zeta)$, $ G = \operatorname{Gal}(L/\Q)$.

\medskip
\noindent\textbf{Part I --- The group and its filtration.}
\begin{enumerate}
  \item औचित्य: $[L:\Q] = 16$, $ G \cong (\Z/17\Z)^\times $,
        क्रम का चक्रीय $16$. सत्यापित करें कि $3$ का जनरेटर है
        $(\Z/17\Z)^\times $ \emph{(की शक्तियों की गणना करें $3$
        सापेक्ष $17$: $3, 9, 10, 13, 5, 15, 11, 16, \dots $)}.
  \item होने देना $\sigma \in G $ साथ $\sigma(\zeta) = \zeta^3$, और
        $ H_k = \langle \sigma^{2^k}\rangle $ के लिए $ k = 0, \dots,
        4$. बताते हैं कि $ G = H_0 \supset H_1 \supset H_2 \supset
        H_3 \supset H_4 = \{e\}$ प्रत्येक सूचकांक के साथ $[H_k :
        H_{k+1}] = 2$, और वह निश्चित फ़ील्ड $\Q = L_0
        \subset L_1 \subset L_2 \subset L_3 \subset L_4 = L $
        द्विघात विस्तारों का एक टावर बनाएं।
  \item निष्कर्ष निकालना \emph{संभवतः}, का उपयोग करना
        \cref{thm:b3:galois:wantzel}, वह $\zeta $ --- इसलिए
        $17$-गॉन --- है \omterm{def:b3:galois:constructible}{रचनात्मक}. बाकी समस्या
        टावर को स्पष्ट बनाता है।
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The periods of length 8.}
को परिभाषित करो \emph{गॉस काल}
\[
\eta_0 = \sum_{k \text{ even}} \zeta^{3^k \bmod 17}
= \zeta^{1} + \zeta^{9} + \zeta^{13} + \zeta^{15} + \zeta^{16} +
\zeta^{8} + \zeta^{4} + \zeta^{2},
\qquad
\eta_1 = \sum_{k \text{ odd}} \zeta^{3^k \bmod 17}.
\]
\begin{enumerate}[resume]
  \item बताते हैं कि $\eta_0, \eta_1$ द्वारा तय किये गये हैं $ H_1$ और अदला-बदली की गई
        द्वारा $\sigma $; परिणाम निकालना $\eta_0, \eta_1 \in L_1$ और वह
        वे एक द्विघात ओवर की दो जड़ें हैं $\Q $.
  \item गणना करना $\eta_0 + \eta_1 = -1$. दिखाओ $\eta_0\eta_1 = -4$
        \emph{(प्रत्येक उत्पाद $\zeta^a\zeta^b $ कुछ है $\zeta^c $,
        $ c \neq 0$; प्रत्येक को कितनी बार गिनें $ c $ होता है, या
        तर्क दें कि उत्पाद एक परिमेय पूर्णांक है जिसे निर्धारित किया गया है
        $ G $, कुल योग के बराबर $64$ उत्पाद, और उपयोग
        प्रत्येक अशून्य अवशेष समान रूप से बार-बार प्रकट होता है)}.
  \item परिणाम निकालना $\eta_0 = \frac{-1 + \sqrt{17}}2$, $\eta_1 =
        \frac{-1-\sqrt{17}}2$ (पहचानें कि कौन सा है
        संख्यात्मक रूप से: $\eta_0 \approx 1.56$), और $ L_1 =
        \Q(\sqrt{17})$.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Periods of length 4 and 2.}
परिभाषित करना
\[
\beta_0 = \zeta + \zeta^{13} + \zeta^{16} + \zeta^{4},\quad
\beta_1 = \zeta^3 + \zeta^5 + \zeta^{14} + \zeta^{12},\quad
\beta_2 = \zeta^9 + \zeta^{15} + \zeta^{8} + \zeta^{2},\quad
\beta_3 = \zeta^{10} + \zeta^{11} + \zeta^{7} + \zeta^{6}.
\]
\begin{enumerate}[resume]
  \item दिखाओ $\beta_0 + \beta_2 = \eta_0$, $\beta_1 + \beta_3 =
        \eta_1$, ओर वो $\beta_0, \beta_2$ द्वारा तय किये गये हैं $ H_2$,
        द्वारा अदला-बदली की गई $\sigma^2$.
  \item गणना करना $\beta_0\beta_2 = -1$ और $\beta_1\beta_3 = -1$
        \emph{(विस्तृत करें: सोलह प्रतिपादकों ने कवर प्राप्त किया
        $1, \dots, 16$ बिल्कुल एक बार)}.
  \item परिणाम निकालना $\beta_0 = \frac{\eta_0 + \sqrt{\eta_0^2 + 4}}2$
        (चिह्न को संख्यात्मक रूप से जांचें: $\beta_0 \approx 2.05$) और
        के लिए अनुरूप सूत्र $\beta_1$; इस तरह $ L_2 =
        \Q(\beta_0)$, द्विघात ओवर $ L_1$.
  \item होने देना $\gamma_0 = \zeta + \zeta^{16} =
        2\cos\frac{2\pi}{17}$ और $\gamma_1 = \zeta^{13} +
        \zeta^4$. दिखाओ $\gamma_0 + \gamma_1 = \beta_0$ और
        $\gamma_0\gamma_1 = \beta_1$, ताकि $\gamma_0 =
        \frac{\beta_0 + \sqrt{\beta_0^2 - 4\beta_1}}2$.
  \item व्यक्त करने वाले सूत्रों की श्रृंखला को इकट्ठा करें
        $\cos\frac{2\pi}{17}$ नेस्टेड वर्गमूलों द्वारा, और a दें
        दशमलव जाँच ($\cos\frac{2\pi}{17} \approx 0.93247$).
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Epilogue.}
\begin{enumerate}[resume]
  \item वास्तव में तर्क ने उसका उपयोग कहां किया $17$ एक है
        \emph{फ़र्मेट प्राइम} ($17 = 2^{2^2} + 1$)? उसके लिए दिखाओ
        एक प्रधान $ p $, नियमित $ p $-गोन है \omterm{def:b3:galois:constructible}{रचनात्मक} आईएफएफ $ p
        = 2^{2^t} + 1$ कुछ के लिए $ t $ \emph{(अगर $ p - 1 = 2^m $,
        दिखाओ $ m $ की स्वयं एक शक्ति होनी चाहिए $2$)}.
  \item की पूरी सूची निकालें \omterm{def:b3:galois:constructible}{रचनात्मक} नियमित
        $ n $-गॉन्स फॉर $ n \leq 20$, गॉस--वांटज़ेल का उपयोग करते हुए
        की कसौटी \cref{cor:b3:galois:impossible}.
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Gauss sums and quadratic
reciprocity.} भाग II की अवधियों में एक खजाना छिपा है। एक के लिए
अजीब प्रधान $ p $, द \emph{पौराणिक प्रतीक} $\bigl(\frac
ap\bigr)$ है $+1$ अगर $ a $ एक शून्येतर वर्ग मॉड है $ p $, $-1$ यदि यह नहीं है, $0$ अगर $ p \mid a $; \cref{exo:b3:galois:4}(बी)
(यूलर की कसौटी) देता है $\bigl(\frac ap\bigr) \equiv
a^{(p-1)/2} \pmod p $, जहां से बहुलता. लिखना $\zeta =
\eu^{2\iu\pi/p}$, $ p^* = (-1)^{(p-1)/2}p $, और परिभाषित करें
\emph{गॉस राशि}
\[
g \;=\; \sum_{a=1}^{p-1}\Bigl(\frac ap\Bigr)\zeta^a .
\]
\begin{enumerate}[resume]
  \item दिखाओ $\sum_{a=1}^{p-1}\bigl(\frac ap\bigr) = 0$
        (जितने वर्ग उतने गैर वर्ग), और सिद्ध करें
        वैकल्पिक रूप $ g = \sum_{a=0}^{p-1}\zeta^{a^2}$
        \emph{(प्रत्येक गैर-शून्य वर्ग पर दो बार प्रहार किया जाता है, और
        $\sum_{a}\zeta^a = 0$)}. के लिए $ p = 17$: संबंधित $ g $ को
        भाग II की अवधियाँ ---दिखाएँ $ g = \eta_0 - \eta_1$
        \emph{(वर्ग मॉड $17$ बिलकुल सम हैं
        जनरेटर की शक्तियाँ $3$)}.
  \item सिद्ध करना $ g^2 = p^*$: बढ़ाना
        \[
        g^2 = \sum_{a,b\neq0}\Bigl(\frac{ab}p\Bigr)
        \zeta^{a+b}
        = \sum_{c}\ \sum_{a \neq 0}\Bigl(\frac{a(c -
        a)}p\Bigr)\zeta^{c}
        \]
        (तय करना $ b = c - a $), विकल्प $ c - a = at $ मूल्यांकन करना
        आंतरिक योग के रूप में $\bigl(\frac{-1}p\bigr)(p - 1)$ के लिए
        $ c = 0$ और $-\bigl(\frac{-1}p\bigr)$ अन्यथा, और
        प्रश्न 14 के साथ समाप्त करें। संख्यात्मक रूप से जांचें: के लिए $ p =
        17$, $(\eta_0 - \eta_1)^2 = 17$ (भाग II).
  \item परिणाम निकालना $\sqrt{p^*} \in \Q(\zeta_p)$, और निष्कर्ष निकालें
        वह \emph{अद्वितीय} का द्विघात उपक्षेत्र
        $\Q(\zeta_p)$ है $\Q(\sqrt{p^*})$ --- अद्वितीय क्योंकि
        $\operatorname{Gal}(\Q(\zeta_p)/\Q)$ चक्रीय है
        (\cref{thm:b3:galois:cyclotomicirred}) और एक चक्रीय
        समूह में सूचकांक का बिल्कुल एक उपसमूह होता है $2$. (हर
        कुछ साइक्लोटोमिक क्षेत्र में द्विघात क्षेत्र एम्बेड होता है ---
        क्रोनकर-वेबर का यह पहला मामला है
        प्रमेय, जिसका \omterm{def:b3:groups:normal}{सामान्य} रूप बहुत आगे है।)
  \item अब चलो $ q \neq p $ एक और अजीब अभाज्य बनें। में काम कर रहा हूँ
        अंगूठी $\Z[\zeta]$ सापेक्ष $ q $, सिद्ध करना
        \[
        g^q \equiv \Bigl(\frac qp\Bigr)\,g \pmod{q\Z[\zeta]}
        \]
        \emph{(नए व्यक्ति का सपना: $(x + y)^q \equiv x^q + y^q $
        आधुनिक $ q $ किसी भी क्रमविनिमेय वलय में; तब $ g^q \equiv
        \sum_a\bigl(\frac ap\bigr)^q\zeta^{aq}$, पुनः अनुक्रमणिका $ b =
        aq $ और बाहर खींचो $\bigl(\frac{q^{-1}}p\bigr) =
        \bigl(\frac qp\bigr)$)}.
  \item वहीं दूसरी ओर, $ g^q = g\,(g^2)^{(q-1)/2} =
        g\,(p^*)^{(q-1)/2}$; यूलर के मानदंड मॉड का उपयोग करना $ q $,
        निष्कर्ष निकालना $ g^q \equiv \bigl(\frac{p^*}q\bigr)g
        \pmod{q\Z[\zeta]}$, फिर --- दोनों को गुणा करना
        के लिए अभिव्यक्तियाँ $ g^q $ द्वारा $ g $ और उपयोग कर रहे हैं $ g^2 = p^*$,
        उलटा मॉड $ q $ --- निष्कर्ष निकालें
        \[
        \Bigl(\frac qp\Bigr) = \Bigl(\frac{p^*}q\Bigr) .
        \]
        \emph{(पूर्णांकों के बीच सर्वांगसमता क्यों होती है?
        $\pm p^*$ सापेक्ष $ q\Z[\zeta]$ उनकी समानता का संकेत?
        के साथ प्रतिच्छेद करें $\Z $.)}
  \item उधेड़ना $\bigl(\frac{p^*}q\bigr) =
        \bigl(\frac{-1}q\bigr)^{(p-1)/2}\bigl(\frac pq\bigr)$
        और $\bigl(\frac{-1}q\bigr) = (-1)^{(q-1)/2}$ को
        प्राप्त करें \emph{द्विघात पारस्परिकता का नियम}:
        \[
        \Bigl(\frac pq\Bigr)\Bigl(\frac qp\Bigr)
        = (-1)^{\frac{p-1}2\cdot\frac{q-1}2} .
        \]
        इसे सत्यापित करें $(p, q) = (17, 3)$ वर्गों को सूचीबद्ध करके
        मॉड $17$ और मॉड $3$, और तीन में निर्णय लेने के लिए इसका उपयोग करें
        पंक्तियाँ चाहे $ x^2 \equiv 219 \pmod{383}$ है \omterm{def:b3:groups:derived}{व्याख्या करने योग्य}
        ($383$ प्रधान है, $219 = 3\cdot73$).
\end{enumerate}

\medskip
\noindent\textbf{Part VI --- Counting अलघुकरणीय polynomials:
the prime number theorem of $\mathbb F_q[X]$.} एक प्राइम पावर ठीक करें
$ q $ और चलो $ N_q(n)$ की संख्या हो \emph{राक्षसी अलघुकरणीय}
के बहुपद \omterm{def:b3:galois:extension}{डिग्री} $ n $ ऊपर $\mathbb F_q $; से याद करें
\cref{exo:b3:galois:6} का गुणनखंडन $ X^{q^n} - X $ और
पहचान $ q^n = \sum_{d\mid n}d\,N_q(d)$, जो अब हम हैं
पलटना, पुनर्व्याख्या करना और शोषण करना।
\begin{enumerate}[resume]
  \item (शब्द) किसी शब्द को बुलाओ $ w \in \mathbb F_q^n $
        \emph{प्राचीन} यदि यह कोई शक्ति नहीं है $ u^{n/d} =
        u\cdots u $ बिल्कुल छोटे शब्द का $ u $, और चलो
        $ A(d)$ लंबाई के आदिम शब्दों की संख्या हो $ d $.
        दिखाएँ कि प्रत्येक शब्द की लंबाई कितनी है $ n $ विशिष्ट रूप से एक शक्ति है
        कुछ लंबाई के एक आदिम शब्द का $ d \mid n $, ताकि
        $ q^n = \sum_{d \mid n}A(d)$; के साथ तुलना करना
        \cref{exo:b3:galois:6}, निष्कर्ष निकालें $ A(d) = d\,N_q(d)$
        हरएक के लिए $ d $, और इस संयोग को एक द्वारा समझाएं
        स्पष्ट आक्षेप: एक तत्व $\alpha \in
        \mathbb F_{q^d}$ का \omterm{def:b3:galois:extension}{डिग्री} $ d $ है \omterm{thm:b3:galois:finitefields}{फ्रोबेनियस} \omterm{def:b3:groups:action}{कक्षा}
        $(\alpha, \alpha^q, \dots, \alpha^{q^{d-1}})$ का
        बिलकुल $ d $ विशिष्ट तत्व, और के तत्व \omterm{def:b3:galois:extension}{डिग्री}
        $ d $ अनुरूप $ d $-से-एक से इरेड्यूसिबल्स का \omterm{def:b3:galois:extension}{डिग्री}
        $ d $.
  \item सिद्ध करो \emph{एम\"ओबियस व्युत्क्रम सूत्र}: अगर $ f(n) =
        \sum_{d\mid n}g(d)$ सभी के लिए $ n $, तब $ g(n) =
        \sum_{d\mid n}\mu(d)\,f(n/d)$, कहाँ $\mu $ है
        एम\"ओबियस फ़ंक्शन ($\mu(m) = (-1)^{\#\text{prime
        factors}}$ अगर $ m $ वर्गमुक्त है, $0$ अन्यथा)
        \emph{(कुंजी लेम्मा: $\sum_{d \mid m}\mu(d) = 0$ के लिए $ m >
        1$ --- भाजक को एक निश्चित अभाज्य के साथ और उसके बिना युग्मित करें
        कारक)}. परिणाम निकालना
        \[
        N_q(n) = \frac1n\sum_{d \mid n}\mu(d)\,q^{n/d} .
        \]
  \item दिखाओ $ N_q(n) \geq \frac1n\bigl(q^n - 2q^{n/2}\bigr) > 0$
        हरएक के लिए $ n \geq 1$: इसका एक नया प्रमाण $\mathbb F_{q^n}$
        सभी के लिए मौजूद है $ n $. प्रमुख शब्द की व्याख्या करें: एक यादृच्छिक
        का मोनिक बहुपद \omterm{def:b3:galois:extension}{डिग्री} $ n $ है अलघुकरणीय के साथ
        संभाव्यता $\sim \frac1n $ --- द \omterm{prop:b3:galois:perfect}{उत्तम} का एनालॉग
        अभाज्य संख्या प्रमेय, के साथ $\log x $ के लिए व्यापार किया गया $ n $;
        के लिए संख्यात्मक रूप से सत्यापित करें $ q = 2$, $ n \leq 4$
        (\cref{exo:b3:galois:6} गिनती सूचीबद्ध करता है)।
  \item प्रश्न 21 का गुणक सहचर सिद्ध करें:
        \[
        \prod_{\substack{\pi \text{ monic irred.}\\ \deg\pi =
        n}}\pi \;=\; \prod_{d \mid n}
        \bigl(X^{q^d} - X\bigr)^{\mu(n/d)}
        \]
        \emph{(एम\"एबेलियन समूह में ओबियस उलटा
        शून्येतर तर्कसंगत कार्य)}; इसके लिए इसे हाथ से सत्यापित करें $ q
        = 2$, $ n = 2$: $(X^4 - X)/(X^2 - X) = X^2 + X + 1$.
\end{enumerate}

\medskip
\noindent\textbf{Part VII --- Two codas.}
\begin{enumerate}[resume]
  \item (दूसरा पूरक) भाग V की विधि भी गणना करती है
        $\bigl(\frac2q\bigr)$. होने देना $\omega = \eu^{2\iu\pi/8}$
        और $ g = \omega + \omega^{-1}$. दिखाओ $ g^2 = 2$
        \emph{($\omega^2 = \iu $)}; फिर, एक विषम अभाज्य के लिए $ q $,
        में साबित करो $\Z[\omega]$ सापेक्ष $ q $ वह
        \[
        g^q \equiv \omega^q + \omega^{-q} \pmod{q\Z[\omega]},
        \]
        और दाहिना भाग बराबर होता है $ g $ अगर $ q \equiv \pm1
        \pmod 8$ और $-g $ अगर $ q \equiv \pm3 \pmod 8$. के साथ तुलना करना $ g^q = g\,(g^2)^{(q-1)/2} \equiv
        \bigl(\frac2q\bigr)g $ प्रश्न 18 के अनुसार, निष्कर्ष निकालें
        \[
        \Bigl(\frac2q\Bigr) = (-1)^{(q^2-1)/8},
        \]
        उसकी जाँच कर रहा हूँ $(q^2 - 1)/8$ ठीक तब भी है जब $ q
        \equiv \pm1 \pmod 8$. सत्यापित करें: $2$ एक वर्गाकार मॉड है $7$
        और मॉड $17$ ($3^2$ और $6^2$), मॉड नहीं $3$ न ही मॉड
        $5$.
  \item (जीटा फ़ंक्शन का $\mathbb F_q[X]$) सिद्ध करें
        औपचारिक शक्ति श्रृंखला की पहचान $ t $:
        \[
        \prod_{n \geq 1}\bigl(1 - t^n\bigr)^{-N_q(n)}
        = \frac1{1 - qt}
        \]
        \emph{(मोनिक में अद्वितीय गुणनखंडन इरेड्यूसिबल्स:
        प्रत्येक कारक को एक ज्यामितीय श्रृंखला के रूप में विस्तारित करें और गिनें
        के मोनिक बहुपद \omterm{def:b3:galois:extension}{डिग्री} $ n $)}. पहचान पुनः प्राप्त करें
        $ q^m = \sum_{d \mid m}d\,N_q(d)$ का
        \cref{exo:b3:galois:6} लघुगणक लेकर. की जाँच करें
        का गुणांक $ t^2$ के लिए हाथ से $ q = 2$, और उपयोग करें
        प्रश्न 21 की गणना करने का सूत्र $ N_2(6) = 9$,
        सत्यापन कर रहा है $2^6 = 1\cdot2 + 2\cdot1 + 3\cdot2 +
        6\cdot9$.
\end{enumerate}
\end{problem}
