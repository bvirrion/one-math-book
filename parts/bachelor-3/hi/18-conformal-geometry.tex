\chapter{अनुरूप मानचित्र और रीमैन मानचित्रण प्रमेय}
\label{ch:b3:conformal}

ए \omterm{def:b3:holomorphic:holo}{होलोमार्फिक} दो डोमेन के बीच आक्षेप सभी को स्थानांतरित करता है
एक से दूसरे का जटिल विश्लेषण: ऐसे मानचित्र ---
\emph{कोन्फोर्मल}, क्योंकि वे कोणों को सुरक्षित रखते हैं --- हैं
की समरूपता \omterm{def:b3:holomorphic:holo}{होलोमार्फिक} दुनिया। यह अध्याय वर्गीकृत करता है
उन्हें जहां वर्गीकरण संभव है (डिस्क, विमान:
श्वार्ज़ की लेम्मा कुंजी है, आश्चर्यजनक की असमानता
शक्ति), के सघनता सिद्धांत का निर्माण करती है \omterm{def:b3:holomorphic:holo}{होलोमार्फिक} परिवार
(मॉन्टेल), और सबसे गहरे अस्तित्व प्रमेय को सिद्ध करता है
विषय: \emph{प्रत्येक} \omterm{def:b3:conformal:simplyconnected}{बस जुड़ा हुआ} है का उचित उपडोमेन
$\C $, चाहे इसकी सीमा कितनी भी टेढ़ी-मेढ़ी क्यों न हो अनुरूप रूप से समतुल्य
यूनिट डिस्क के लिए. हम साथ \omterm{def:b3:topology:interior}{बंद} करते हैं हार्मोनिक कार्य और यह
\omterm{thm:b3:conformal:poisson}{पॉइसन कर्नेल}, डिस्क पर डिरिचलेट समस्या का समाधान ---
अनुरूप ज्यामिति का विश्लेषणात्मक लाभ। लगातार,
$\mathbb D = D(0,1)$ और $\mathbb H = \{\operatorname{Im}z >
0\}$.

\section{अनुरूप मानचित्र; एम\"ओबियस परिवर्तन}

\begin{definition}\label{def:b3:conformal:conformal}
ए \emph{अनुरूप मानचित्र}\index{अनुरूप मानचित्र} (या
बिहोलोमोर्फिज्म) के बीच खुले सेट एक है \omterm{def:b3:holomorphic:holo}{होलोमार्फिक} आक्षेप;
इसका व्युत्क्रम स्वतः होता है \omterm{def:b3:holomorphic:holo}{होलोमार्फिक}
(\cref{cor:b3:residues:openmapping}). दो डोमेन हैं
\emph{अनुरूप रूप से समतुल्य} यदि ऐसा कोई नक्शा मौजूद है;
$\operatorname{Aut}(\Omega)$ अनुरूप के समूह को दर्शाता है
स्व-मानचित्र। कहाँ $ f' \neq 0$ --- हर जगह, इंजेक्शन के लिए $ f $
(\cref{cor:b3:residues:openmapping}का प्रमाण) --- द
अंतर से गुणा है $ f'(z) \neq 0$: ए
समानता, इसलिए अनुरूप मानचित्र वक्रों के बीच कोणों को संरक्षित करते हैं,
अभिविन्यास सहित.
\end{definition}

\begin{example}[एम\"ओबियस परिवर्तन]
\label{ex:b3:conformal:mobius}
के लिए $\bigl(\begin{smallmatrix}a & b\\ c &
d\end{smallmatrix}\bigr) \in GL_2(\C)$, द \emph{एम\"ओबियस
परिवर्तन}\index{एम\"ओबियस
परिवर्तन} $ z \mapsto
\frac{az + b}{cz + d}$ से अनुरूप है
$\C\setminus\{-d/c\}$ पर $\C\setminus\{a/c\}$ (इसके विपरीत
व्युत्क्रम मैट्रिक्स से एक ही प्रकार; रचना
मैट्रिक्स उत्पाद से मेल खाता है)। \emph{केली
नक्शा}\index{केली
नक्शा}
\[
\varphi(z) = \frac{z - \iu}{z + \iu}
\]
एमएपीएस $\mathbb H $ अनुरूप रूप से पर $\mathbb D $: वास्तव में $\abs{z
- \iu} < \abs{z + \iu}$ बिल्कुल कब $ z $ के करीब है $\iu $
की तुलना में $-\iu $, यानी\ $\operatorname{Im}z > 0$; उलटा है
$ w \mapsto \iu\frac{1 + w}{1 - w}$. एम\"ओबियस मानचित्र भेजते हैं
पारिवारिक मंडलियाँ-और-पंक्तियाँ अपने आप में (\cref{exo:b3:conformal:1}).
\end{example}

\begin{example}[जौकोव्स्की मानचित्र]\label{ex:b3:conformal:joukowski}
एम से परे\"ओबियस, सबसे उपयोगी \omterm{def:b3:conformal:conformal}{अनुरूप मानचित्र} शास्त्रीय का
अनुप्रयुक्त गणित है
\[
J(z) = \frac12\Bigl(z + \frac1z\Bigr) .
\]
बाहरी भाग पर $\Omega = \{\abs z > 1\}$ यूनिट डिस्क का,
$ J $ इंजेक्शन है: $ J(z) = J(w)$ देता है $(z - w)(1 -
\frac1{zw}) = 0$ और $\abs{zw} > 1$. इसका व्युत्पन्न $ J'(z) =
\frac12(1 - z^{-2})$ पर ही लुप्त हो जाता है $ z = \pm1$, पर
सीमा: $ J $ से एक अनुरूप तुल्यता है $\Omega $ पर
इसकी छवि, जो है $\C\setminus\intcc{-1}1$ --- इकाई
सर्कल स्वयं खंड पर दो-से-एक मुड़ा हुआ है
($ J(\eu^{\iu\theta}) = \cos\theta $). इस प्रकार ए का बाहरी भाग
\emph{खंड}, एक स्लिट प्लेन है जिसकी कोई चिकनी सीमा नहीं है
अनुरूप रूप से डिस्क का बाहरी भाग: कोने कोई बाधा नहीं हैं
अनुरूप तुल्यता के लिए, केवल सीमा चिकनाई के लिए। छवियाँ
वृत्तों के माध्यम से $\pm1$ लेकिन \omterm{ex:b3:groups:actions}{केंद्र} से बाहर एयरफ़ोइल के आकार के हैं
वक्र, और रचना $ J $ एम के साथ\"ओबियस मानचित्रों का परिवहन किया गया
एक सिलेंडर के पार प्रवाह --- हाथ से गणना योग्य --- प्रवाह तक
पिछले एक विंग: बीसवीं सदी के पहले भाग के लिए,
यह उदाहरण \emph{था} वायुगतिकी। यह का द्वार भी है
चेबीशेव: $ J $ संयुग्मित करता है $ z \mapsto z^n $ चेबीशेव को
बहुपद $ T_n $ (\cref{pb:b3:hilbert:1}, भाग V), तब से
$ J(z^n) = \cos(n\theta)$ कब $ z = \eu^{\iu\theta}$.
\end{example}

\section{श्वार्ज़ की लेम्मा और ऑटोमोर्फिज्म समूह}

\begin{theorem}[श्वार्ज़ लेम्मा]\label{thm:b3:conformal:schwarz}
होने देना $ f \colon \mathbb D \to \mathbb D $ होना \omterm{def:b3:holomorphic:holo}{होलोमार्फिक} साथ
$ f(0) = 0$. तब\index{श्वार्ज़ लेम्मा}
\[
\abs{f(z)} \leq \abs z \ \ (z \in \mathbb D)
\qquad\text{and}\qquad \abs{f'(0)} \leq 1 ,
\]
और यदि $\abs{f(z_0)} = \abs{z_0}$ एक के लिए $ z_0 \neq 0$, या
$\abs{f'(0)} = 1$, तब $ f(z) = \eu^{\iu\theta}z $ एक है
घूर्णन.
\end{theorem}

\begin{proof}
$ g(z) = f(z)/z $ का विस्तार समग्र रूप से को $\mathbb D $ (द
विलक्षणता पर $0$ हटाने योग्य है: $ g $ के पास घिरा हुआ है $0$,
\cref{thm:b3:residues:riemanncw}; इसका मूल्य पर $0$ है
$ f'(0)$). पर $\abs z = r < 1$: $\abs g \leq \frac1r $, तो द्वारा
अधिकतम सिद्धांत (\cref{thm:b3:holomorphic:maximum})
$\abs g \leq \frac1r $ पर $\bar D(0,r)$; होने देना $ r \to 1$: $\abs
g \leq 1$ पर $\mathbb D $, जो दोनों असमानताएं हैं। समानता
एक पर आंतरिक भाग बिंदु बनाता है $\abs g $ एक प्राप्त करें आंतरिक भाग
अधिकतम: $ g $ मापांक का स्थिरांक $1$.
\end{proof}

\begin{theorem}[डिस्क की ऑटोमोर्फिज्म]
\label{thm:b3:conformal:autdisc}
के लिए $ a \in \mathbb D $, द \emph{ब्लाश्के
कारक}\index{ब्लाश्के
कारक}
\[
\varphi_a(z) = \frac{z - a}{1 - \bar a z}
\]
का एक स्वप्रतिरूपण है $\mathbb D $ का आदान प्रदान $ a $ और $0$, साथ
$\varphi_a^{-1} = \varphi_{-a}$. की प्रत्येक ऑटोमोर्फिज्म
$\mathbb D $ है $\eu^{\iu\theta}\varphi_a $ अद्वितीय के लिए $\theta
\in \R/2\pi\Z $, $ a \in \mathbb D $.
\end{theorem}

\begin{proof}
पर $\abs z = 1$: $\abs{1 - \bar az} = \abs{\bar z}\abs{1 -
\bar az} = \abs{\bar z - \bar a\abs z^2} = \abs{\bar z - \bar
a} = \abs{z - a}$, इसलिए $\abs{\varphi_a} = 1$ वहाँ; द्वारा
अधिकतम सिद्धांत $\varphi_a(\mathbb D) \subseteq \bar{\mathbb
D}$, और खुलापन छवि को अंदर रखता है $\mathbb D $. \omterm{def:b3:galois:algebraic}{बीजगणितीय}
पहचान $\varphi_{-a}\circ\varphi_a = \mathrm{id}$ (प्रत्यक्ष
संगणना) वस्तुनिष्ठता दर्शाती है। अब चलो $ f \in
\operatorname{Aut}(\mathbb D)$ और $ a = f^{-1}(0)$: $ g =
f\circ\varphi_{-a}$ एक ऑटोमोर्फिज्म फिक्सिंग है $0$. श्वार्ज़
पर लागू किया गया $ g $ और को $ g^{-1}$: $\abs{g(z)} \leq \abs z $ और
$\abs{g^{-1}(w)} \leq \abs w $, इसलिए $\abs{g(z)} = \abs z $:
घूर्णन, $ g = \eu^{\iu\theta}\,\mathrm{id}$, यानी\ $ f =
\eu^{\iu\theta}\varphi_a $. विशिष्टता: $ a = f^{-1}(0)$ और
$\theta $ से $ f'$-प्रकार मूल्यांकन (या से $ f(0) =
-\eu^{\iu\theta}a $ और एक और मूल्य)।
\end{proof}

\begin{theorem}[विमान की ऑटोमोर्फिज्म]
\label{thm:b3:conformal:autplane}
$\operatorname{Aut}(\C) = \{z \mapsto az + b : a \in \C^*,\ b
\in \C\}$. फलस्वरूप $\C $ और $\mathbb D $ नहीं हैं
अनुरूप रूप से समतुल्य.
\end{theorem}

\begin{proof}
होने देना $ f \in \operatorname{Aut}(\C)$ और विचार करें $ g(z) =
f(1/z)$ पर $\C^*$: ए होलोमोर्फिक फ़ंक्शन एक साथ \omterm{def:b3:residues:singularities}{पृथक
विलक्षणता} पर $0$. यदि यह आवश्यक होता,
कैसोराती--वीयरस्ट्रैस (\cref{thm:b3:residues:riemanncw}) करना होगा $ g\bigl(D(0,\varepsilon)\setminus\{0\}\bigr)$ \omterm{def:b3:topology:interior}{घना},
जबकि $ f(D(0, 1))$ खुला है और इससे अलग है ($ f $
इंजेक्शन: दो सेट असंयुक्त सेट की छवियाँ हैं) ---
एक के लिए असंभव \omterm{def:b3:topology:interior}{घना} सेट और एक गैररिक्त \omterm{def:b3:topology:topology}{खुला सेट}. इसलिए $0$ है
के लिए एक पोल या हटाने योग्य $ g $, यानी\ $\abs{f(z)}$ अधिकतम है
बहुपद वृद्धि: $ f $ एक बहुपद है
(\cref{exo:b3:holomorphic:4}(ए))। इंजेक्शन बल की \omterm{def:b3:galois:extension}{डिग्री}
$1$: एक उच्\omterm{def:b3:galois:extension}{च-डिग्री} बहुपद का या तो एक से अधिक मूल होता है
$ f - c $ कहीं ($ f'$ गायब हो जाता है) या कई अलग
प्रीइमेज (डी'अलेम्बर्ट--गॉस, \cref{pb:b3:holomorphic:1});
किसी भी तरह से इंजेक्शन विफल हो जाता है। अंत में, एक अनुरूप $\C \to
\mathbb D $ एक परिबद्ध संपूर्ण फ़ंक्शन होगा: स्थिरांक
(लिउविले) --- कोई तुल्यता नहीं।
\end{proof}

\section{मोंटेल का प्रमेय}

\begin{theorem}[मोंटेल]\label{thm:b3:conformal:montel}
होने देना $\mathcal F \subseteq \mathcal H(\Omega)$ होना \emph{स्थानीय स्तर पर
घिरा हुआ}: हर बिंदु में एक है \omterm{def:b3:topology:topology}{अड़ोस-पड़ोस} जिस पर $\sup_{f\in
\mathcal F}\sup\abs f < \infty $. फिर हर क्रम
$\mathcal F $ इसका परिणाम सभी पर समान रूप से अभिसरित होता है
कॉम्पैक्ट के उपसमुच्चय $\Omega $ (ए को \omterm{def:b3:holomorphic:holo}{होलोमार्फिक}
सीमा).\index{मोंटेल का प्रमेय}
\end{theorem}

\begin{proof}
स्थानीय समनिरंतरता: अगर $\abs f \leq M $ पर $ D(a, 2r)
\subseteq \Omega $ सभी के लिए $ f \in \mathcal F $, कॉची
सूत्र देता है, के लिए $ z, z' \in D(a, r)$,
\[
\abs{f(z) - f(z')} = \frac{\abs{z - z'}}{2\pi}
\Bigl|\int_{C_{2r}}\frac{f(w)\,\dd w}{(w-z)(w-z')}\Bigr|
\leq \frac{\abs{z - z'}\,2\pi\cdot2r\,M}{2\pi\,r^2}
= \frac{2M}{r}\,\abs{z - z'} :
\]
एक समान लिप्सचिट्ज़ बाध्य। निकास $\Omega $ द्वारा कॉम्पैक्ट कैमरों $ K_m $;
प्रत्येक $ K_m $ ऐसी अनेक डिस्कों द्वारा कवर किया गया है, इसलिए
$\mathcal F $ समान रूप से घिरा हुआ है और \omterm{def:b3:complete:equicontinuous}{समनिरंतर} पर
$ K_m $: अर्ज़ेल\`ए--एस्कोली (\cref{thm:b3:complete:ascoli})
समान रूप से अभिसरण करते हुए एक परिणाम निकालता है $ K_m $;
तिरछा करना $ m $. हद है \omterm{def:b3:holomorphic:holo}{होलोमार्फिक} द्वारा
\cref{thm:b3:holomorphic:weierstrassconv}.
\end{proof}

\section{रीमैन मानचित्रण प्रमेय}

\begin{definition}\label{def:b3:conformal:simplyconnected}
एक खुला \omterm{def:b3:topology:connected}{जुड़े हुए} $\Omega \subseteq \C $ है \emph{बस
जुड़ा हुआ}\index{बस
जुड़ा हुआ} (होमोलॉजिकल में
अर्थ, हमारे सभी उद्देश्यों के लिए पर्याप्त) यदि
$\operatorname{Ind}_\gamma(w) = 0$ हर चक्र के लिए $\gamma $ में
$\Omega $ और हर $ w \notin \Omega $ --- ``कोई चक्र नहीं
$\Omega $ एक छेद को घेर लेता है''। वैश्विक कॉची प्रमेय द्वारा
(\cref{thm:b3:residues:globalcauchy}) और
\cref{prop:b3:holomorphic:primitive}, ऐसे पर $\Omega $
\emph{प्रत्येक होलोमोर्फिक फ़ंक्शन एक आदिम है}; इसलिए हर
शून्\omterm{def:b3:modules:free}{य-मुक्त} $ f \in \mathcal H(\Omega)$ एक \omterm{def:b3:holomorphic:holo}{होलोमार्फिक}
लघुगणक ($\exp\circ $(आदिम $ f'/f $), ए द्वारा समायोजित
स्थिर, जैसे $(f\eu^{-L})' = 0$) और \omterm{def:b3:holomorphic:holo}{होलोमार्फिक} $ n $-वें जड़ें
$\eu^{L/n}$.\index{होलोमोर्फिक लघुगणक}
\end{definition}

\begin{theorem}[रीमैन मानचित्रण प्रमेय]
\label{thm:b3:conformal:rmt}
प्रत्येक \omterm{def:b3:conformal:simplyconnected}{बस जुड़ा हुआ} है खुला $\Omega \subsetneq \C $, $\Omega
\neq \varnothing $, है अनुरूप रूप से समतुल्य को $\mathbb D $;
दिया गया $ z_0 \in \Omega $, एक अनोखा अनुरूप है $ f \colon
\Omega \to \mathbb D $ साथ $ f(z_0) = 0$ और $ f'(z_0) >
0$.\index{रीमैन मानचित्रण प्रमेय}
\end{theorem}

\begin{proof}
\emph{चरण 0: परिवार खाली नहीं है।} चुनना $ b \notin \Omega $:
$ z - b $ पर शून्\omterm{def:b3:modules:free}{य-मुक्त} है $\Omega $, तो इसमें एक है \omterm{def:b3:holomorphic:holo}{होलोमार्फिक}
वर्गमूल $ h $ ($ h^2 = z - b $). $ h $ इंजेक्शन है ($ h(z) =
h(z')$ वर्गों को $ z = z'$), और यदि $ w \in h(\Omega)$ तब $-w
\notin h(\Omega)$ ($ h(z) = -h(z')$ को भी वर्ग $ z = z'$,
देना $ w = -w = 0$, असंभव जैसा $ h $ शून्\omterm{def:b3:modules:free}{य-मुक्त} है)। तब से
$ h(\Omega)$ खुला है, इसमें एक डिस्क है $ D(h(z_0), \rho)$;
तब $ D(-h(z_0), \rho) \cap h(\Omega) = \varnothing $, अर्थात\
$\abs{h(z) + h(z_0)} \geq \rho $ हरएक के लिए $ z \in \Omega $.
इस तरह
\[
g(z) = \frac{\rho}{2\,\bigl(h(z) + h(z_0)\bigr)}
\]
है \omterm{def:b3:holomorphic:holo}{होलोमार्फिक}, इंजेक्शन (एक एम\"ओबियस मानचित्र की रचना की गई
इंजेक्शन $ h $), साथ $\abs g \leq \frac12 < 1$. ए के साथ रचना \omterm{thm:b3:conformal:autdisc}{ब्लाश्के कारक}
(\cref{thm:b3:conformal:autdisc}) हिलना डुलना $ g(z_0)$ को $0$, द
परिवार
\[
\mathcal F = \{f \colon \Omega \to \mathbb D \text{
holomorphic, injective, } f(z_0) = 0\}
\]
गैर-रिक्त है.

\emph{चरण 1: एक चरम तत्व.} होने देना $ s =
\sup_{\mathcal F}\abs{f'(z_0)} \in \intoc0{+\infty}$ ($> 0$:
सदस्य इंजेक्शन हैं, इसलिए $ f'(z_0) \neq 0$). लेना $ f_n \in
\mathcal F $ साथ $\abs{f_n'(z_0)} \to s $: परिवार है
से घिरा हुआ $1$, तो मोंटेल (\cref{thm:b3:conformal:montel})
अर्क $ f_n \to f $ समान रूप से चालू कॉम्पैक्ट कैमरों; $ f $ है
\omterm{def:b3:holomorphic:holo}{होलोमार्फिक}, $ f(z_0) = 0$, $\abs{f'(z_0)} = s $
(\cref{thm:b3:holomorphic:weierstrassconv} के लिए
डेरिवेटिव), विशेष रूप से $ f $ अचर है; $ f $ है
हर्विट्ज़ द्वारा इंजेक्शन (\cref{exo:b3:residues:8}(बैंड
$ f(\Omega) \subseteq \bar{\mathbb D}$, इस तरह $\subseteq
\mathbb D $ (मैपिंग खोलें)। इसलिए $ f \in \mathcal F $ को प्राप्त करता है
सर्वोच्च: $ s < \infty $.

\emph{चरण 2: चरम मानचित्र चालू है।} कल्पना करना $ a \in
\mathbb D\setminus f(\Omega)$. ब्लाश्के परिवहन
$\varphi_a\circ f $ पर शून्\omterm{def:b3:modules:free}{य-मुक्त} है \omterm{def:b3:conformal:simplyconnected}{बस जुड़ा हुआ} है
$\Omega $: यह है एक \omterm{def:b3:holomorphic:holo}{होलोमार्फिक} वर्गमूल $ F $ (साथ
$ F(\Omega) \subseteq \mathbb D $, जैसा $\abs F^2 =
\abs{\varphi_a\circ f} < 1$), इंजेक्शन (वर्ग)।
अंतर करना)। \omterm{def:b3:groups:normal}{सामान्य}ीकृत करें: $ G = \varphi_{F(z_0)}\circ F \in
\mathcal F $. पूर्ववत करना: $ f = \varphi_{-a}\circ s_2 \circ
\varphi_{-F(z_0)}\circ G $ कहाँ $ s_2(w) = w^2$; वो नक्शा $\Psi
= \varphi_{-a}\circ s_2\circ\varphi_{-F(z_0)} \colon \mathbb D
\to \mathbb D $ है \omterm{def:b3:holomorphic:holo}{होलोमार्फिक} साथ $\Psi(0) = f(z_0) = 0$ और है \emph{नहीं} एक घूर्णन (यह इंजेक्शन नहीं है: $ s_2$ क्या नहीं है)।
श्वार्ज़ की लेम्मा (सख्त मामला): $\abs{\Psi'(0)} < 1$, और
शृंखला नियम $ f = \Psi\circ G $ देता है $\abs{f'(z_0)} =
\abs{\Psi'(0)}\,\abs{G'(z_0)} < \abs{G'(z_0)}$ ---
अधिकतमता का खंडन (नोट) $ G \in \mathcal F $). इस तरह $ f $
चालू है: एक अनुरूप तुल्यता।

\emph{चरण 3: \omterm{def:b3:groups:normal}{सामान्य}ीकरण और विशिष्टता।} गुणा $ f $ द्वारा
$\eu^{-\iu\arg f'(z_0)}$ बनाने के लिए $ f'(z_0) > 0$ (यह अंदर रहता है
$\mathcal F $). अगर $ f_1, f_2$ दोनों काम करते हैं, $\psi =
f_2\circ f_1^{-1} \in \operatorname{Aut}(\mathbb D)$ ठीक करता है
$0$ साथ $\psi'(0) = f_2'(z_0)/f_1'(z_0) > 0$; द्वारा
\cref{thm:b3:conformal:autdisc} $\psi $ एक चक्र है
$\eu^{\iu\theta}$ साथ $\eu^{\iu\theta} > 0$: $\psi =
\mathrm{id}$.
\end{proof}

\begin{remark}\label{rem:b3:conformal:rmtscope}
प्रमेय आश्चर्यजनक का शुद्ध अस्तित्व कथन है
दायरा: एक वर्ग, एक आधा समतल, एक झिरी का पूरक, द
दो स्पर्शरेखा वृत्तों के बीच का क्षेत्र, एक भग्न-सीमा डोमेन
--- सभी अनुरूप रूप से के समान $\mathbb D $. यह क्या नहीं करता
दें: कोई भी सूत्र (स्पष्ट मानचित्र अपवाद हैं:

\end{remark}

\section{हार्मोनिक फ़ंक्शंस और पॉइसन कर्नेल}

\begin{proposition}\label{prop:b3:conformal:harmonicholo}
होने देना $\Omega $ होना \omterm{def:b3:conformal:simplyconnected}{बस जुड़ा हुआ} है और $ u \colon \Omega \to \R $
लयबद्ध\index{हार्मोनिक फ़ंक्शन} ($\mathcal C^2$ साथ
$\Delta u = u_{xx} + u_{yy} = 0$). तब $ u =
\operatorname{Re}F $ एक के लिए \omterm{def:b3:holomorphic:holo}{होलोमार्फिक} $ F $, एक तक अद्वितीय
काल्पनिक स्थिरांक. फलस्वरूप $ u $ है $\mathcal C^\infty $,
औसत मूल्य संपत्ति को संतुष्ट करता है, और अधिकतम का पालन करता है
सिद्धांत (कोई सख्त नहीं) आंतरिक भाग चरम जब तक कि स्थिर न हो)।
\end{proposition}

\begin{proof}
$ g = u_x - \iu u_y $ को संतुष्ट करता है \omterm{prop:b3:holomorphic:cauchyriemann}{कॉची--रीमैन समीकरण}
($ P = u_x $, $ Q = -u_y $: $ P_x = u_{xx} = -u_{yy} = Q_y $ और
$ P_y = u_{xy} = u_{yx} = -Q_x $) साथ \omterm{def:b3:topology:continuity}{निरंतर} आंशिक: $ g $
है \omterm{def:b3:holomorphic:holo}{होलोमार्फिक} (\cref{prop:b3:holomorphic:cauchyriemann};
$\R $-विभेदीकरण से अनुसरण होता है $\mathcal C^1$). होने देना $ F_0$
एक आदिम बनें (\omterm{def:b3:groups:simple}{सरल} कनेक्टिविटी,
\cref{def:b3:conformal:simplyconnected}); तब
$\operatorname{Re}F_0$ ढाल है $(u_x, u_y)$
($ F_0' = g $ कॉची--रीमैन के माध्यम से ठीक उसी तरह अनपैक करता है
$ F_0$), इसलिए $ u - \operatorname{Re}F_0$ स्थिर है ($\Omega $
\omterm{def:b3:topology:connected}{जुड़े हुए}): समायोजित करना $ F = F_0 + c $. संपत्तियों का स्थानांतरण
से \cref{thm:b3:holomorphic:maximum} और
\cref{exo:b3:holomorphic:10} (अधिकतम सिद्धांत के लिए
पर लागू किया गया $ u $ स्वयं, उपयोग करें $\eu^{F}$ जैसा कि)।
\end{proof}

\begin{theorem}[पॉइसन सूत्र; डिरिचलेट समस्या पर
डिस्क]\label{thm:b3:conformal:poisson}
के लिए $0 \leq r < 1$ को परिभाषित करो \emph{पॉइसन
गिरी}\index{पॉइसन
गिरी}
\[
P_r(\theta) = \sum_{n\in\Z}r^{\abs n}\eu^{\iu n\theta}
= \frac{1 - r^2}{1 - 2r\cos\theta + r^2} \;>\; 0 .
\]
होने देना $ g \colon \partial\mathbb D \to \R $ होना \omterm{def:b3:topology:continuity}{निरंतर}, और
सेट, के लिए $ z = r\eu^{\iu\varphi} \in \mathbb D $,
\[
u(z) = \frac1{2\pi}\int_0^{2\pi}
P_r(\varphi - t)\,g(\eu^{\iu t})\,\dd t .
\]
तब $ u $ हार्मोनिक चालू है $\mathbb D $ और फैलता है लगातार
को $\bar{\mathbb D}$ सीमा मूल्यों के साथ $ g $: अनोखा ऐसा
\omterm{prop:b3:conformal:harmonicholo}{हार्मोनिक फ़ंक्शन}.\index{डिरिचलेट समस्या}
\end{theorem}

\begin{proof}
\emph{कर्नेल पहचान}: दो ज्यामितीय श्रृंखलाओं का योग,
\[
\sum_{n\in\Z}r^{\abs n}\eu^{\iu n\theta}
= \operatorname{Re}\frac{1 + r\eu^{\iu\theta}}{1 -
r\eu^{\iu\theta}}
= \frac{1 - r^2}{\abs{1 - r\eu^{\iu\theta}}^2},
\]
जो प्रदर्शित भागफल है; सकारात्मकता स्पष्ट है, और
$\frac1{2\pi}\int_0^{2\pi}P_r = 1$ (केवल $ n = 0$ जीवित रहता है)।

\emph{सामंजस्य}: साथ $ z = r\eu^{\iu\varphi}$,
\[
u(z) = \operatorname{Re}\biggl[\frac1{2\pi}\int_0^{2\pi}
\frac{\eu^{\iu t} + z}{\eu^{\iu t} - z}\,
g(\eu^{\iu t})\,\dd t\biggr],
\]
(ब्रैकेटेड कर्नेल में वास्तविक भाग होता है $ P_r(\varphi - t)$:
गणना करें), और ब्रैकेट है \omterm{def:b3:holomorphic:holo}{होलोमार्फिक} में $ z $ पर $\mathbb D $
(\cref{exo:b3:holomorphic:7}): $ u $ का असली हिस्सा है
होलोमोर्फिक फ़ंक्शन, इसलिए हार्मोनिक।

\emph{सीमा मान}: $ P_r(\cdot)$ एक अनुमानित है
के रूप में पहचान $ r \to 1^-$: द्रव्यमान $1$, और के लिए $\delta \leq
\abs\theta \leq \pi $, $ P_r(\theta) \leq \frac{1 - r^2}{1 -
2r\cos\delta + r^2} \to 0$ समान रूप से। मानक विभाजन
(निरंतरता का $ g $ पास में $\eu^{\iu\varphi_0}$, सीमाबद्धता
अन्यत्र) देता है $ u(r\eu^{\iu\varphi}) \to
g(\eu^{\iu\varphi_0})$ जैसा $ r\eu^{\iu\varphi} \to
\eu^{\iu\varphi_0}$, समान रूप से सीमा बिंदु में:
विस्तार है \omterm{def:b3:topology:continuity}{निरंतर}. \emph{विशिष्टता}: का अंतर
दो समाधान हार्मोनिक चालू हैं $\mathbb D $, \omterm{def:b3:topology:continuity}{निरंतर} पर
\omterm{def:b3:topology:interior}{बंद}, सीमा पर शून्य: अधिकतम सिद्धांत द्वारा
(पर लागू किया गया $\pm $ अंतर), यह गायब हो जाता है।
\end{proof}

\begin{exercise}[$\star\star\star $]\label{exo:b3:conformal:7}
इसका प्रमाण कहां है \cref{thm:b3:conformal:rmt} प्रत्येक का उपयोग करें
परिकल्पना? ट्रेस: (i) \omterm{def:b3:groups:simple}{सरल} कनेक्टिविटी (दो बार); (ii)
$\Omega \neq \C $; (iii) संयुक्तता. फिर दिखाओ कि
प्रमेय \emph{विफल रहता है} के लिए $\Omega = \C $ और वलय के लिए,
यह इंगित करना कि प्रत्येक मामले में प्रमाण का कौन सा चरण टूटता है।
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:conformal:8}
डिरिचलेट समस्या को हल करें $\mathbb D $ सीमा के लिए
डेटा: (ए) $ g(\eu^{\iu t}) = \cos t $; (बी) $ g(\eu^{\iu t}) =
\cos^2 t $; (सी) $ g = \mathbf 1_{\text{upper semicircle}}$ ---
(सी) गणना के लिए $ u(0)$ और माध्य मान के माध्यम से व्याख्या करें
संपत्ति. \emph{(बढ़ाना $ g $ फूरियर श्रृंखला और उपयोग में
$ P_r $की श्रृंखला: $ u(r\eu^{\iu\varphi}) = \sum_n c_n(g)
r^{\abs n}\eu^{\iu n\varphi}$.)}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b3:conformal:9}
(हार्नैक) चलो $ u \geq 0$ पर हार्मोनिक हो $\mathbb D $. साबित करो,
के लिए $\abs z = r < 1$:
\[
\frac{1 - r}{1 + r}\,u(0) \;\leq\; u(z) \;\leq\;
\frac{1 + r}{1 - r}\,u(0)
\]
\emph{(बाउंड द \omterm{thm:b3:conformal:poisson}{पॉइसन कर्नेल} बीच में $\frac{1-r}{1+r}$ और
$\frac{1+r}{1-r}$; लागू करें \omterm{def:b3:representations:rep}{प्रतिनिधित्व} थोड़ा पर
छोटी डिस्क और सीमा तक पास)}. निष्कर्ष: ए हार्मोनिक
समारोह पर $\C $ नीचे परिबद्ध स्थिर है।
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:conformal:10}
हार्मोनिकिटी के अनुरूप अपरिवर्तनीयता का उपयोग करना ($ u\circ f $ है
हार्मोनिक जब $ u $ हार्मोनिक है और $ f $ \omterm{def:b3:holomorphic:holo}{होलोमार्फिक} --- सिद्ध करो
इसके माध्यम से \cref{prop:b3:conformal:harmonicholo} स्थानीय रूप से), हल करें
सीमा के साथ ऊपरी आधे तल पर डिरिचलेट समस्या
डेटा $\mathbf 1_{\intoo{-\infty}0}$: बताते हैं कि

u(x + \iu y) = \frac1\pi\,\arg(x + \iu y)
\qquad (\arg \in \intoo0\pi \text{ on } \mathbb H)
\]
हार्मोनिक चालू है $\mathbb H $ (ए का काल्पनिक भाग \omterm{def:b3:conformal:simplyconnected}{होलोमोर्फिक

\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:conformal:11}

D \to \mathbb D $ होना \omterm{def:b3:holomorphic:holo}{होलोमार्फिक}.
(ए) दिखाओ अगर $ f $ है \emph{दो} विशिष्ट निश्चित बिंदु,
फिर $ f = \mathrm{id}$ \emph{(एक को यहां ले जाएं $0$ एक द्वारा

(सी) के साथ चित्रण करें $ f(z) = \frac{z^2 + z}2$: निश्चित बिंदु,
और के अभिसरण की दर \omterm{def:b3:groups:action}{कक्षा} का $ z_0 =
\frac12$.
\end{exercise}

\section{समस्या: क्षेत्रफल प्रमेय और कोएबे का चतुर्थांश
प्रमेय}

\begin{problem}[{सप्ताहांत समस्या---कितना चाहिए
असमान मानचित्र कवर?}]\label{pb:b3:conformal:1}
ए \emph{एकसमान} फ़ंक्शन एक है \omterm{def:b3:holomorphic:holo}{होलोमार्फिक} इंजेक्शन. द
डिस्क पर \omterm{def:b3:groups:normal}{सामान्य}ीकृत असमान कार्य,
\[
\mathcal S = \bigl\{f \in \mathcal H(\mathbb D) \text{
injective},\ f(z) = z + a_2z^2 + a_3z^3 + \cdots\bigr\},
\]
कठोरता से बाध्य हैं: हम बीबरबैक की असमानता को साबित करते हैं
$\abs{a_2} \leq 2$ और निष्कर्ष निकालें \emph{कोबे क्वार्टर
प्रमेय}: किसी की छवि $ f \in \mathcal S $ शामिल है
डिस्क $ D(0, \frac14)$ --- का तीव्र सार्वभौमिक स्थिरांक
अनुरूप ज्यामिति.\index{कोएबे तिमाही प्रमेय}\index{क्षेत्र
प्रमेय}

\medskip
\noindent\textbf{Part I --- The area theorem.} होने देना $ g(w) = w
+ b_0 + \frac{b_1}w + \frac{b_2}{w^2} + \cdots $ होना \omterm{def:b3:holomorphic:holo}{होलोमार्फिक}
और \emph{इंजेक्शन} पर $\{\abs w > 1\}$.
\begin{enumerate}
  \item के लिए $\rho > 1$, होने देना $ A_\rho $ क्षेत्र हो (लेब्सग्यू
        \omterm{def:b3:measure:measure}{उपाय}) की कॉम्पैक्ट तय करना $ K_\rho = \C\setminus
        g(\{\abs w > \rho\})$, से घिरा क्षेत्र
        चिकनी जॉर्डन वक्र $ g(C_\rho)$. क्षेत्र का उपयोग करना
        वर्ष 2 खंड के ग्रीन-रीमैन प्रमेय का सूत्र
        --- संलग्न क्षेत्र है $\frac1{2\iu}
        \oint\bar\zeta\,\dd\zeta $ सकारात्मक रूप से साथ
        उन्मुखी सीमा---दिखाओ
        \[
        A_\rho = \frac{1}{2\iu}\int_{C_\rho}
        \overline{g(w)}\,g'(w)\,\dd w
        = \pi\Bigl(\rho^2 -
        \sum_{n\geq1}n\,\abs{b_n}^2\rho^{-2n}\Bigr) :
        \]
        स्थानापन्न करें \omterm{thm:b3:residues:laurent}{लॉरेंट श्रृंखला} का $\bar g $ और $ g'$ पर
        $ C_\rho $ और पद दर पद एकीकृत करें (\omterm{def:b3:groups:normal}{सामान्य})।
        अभिसरण; केवल आवृत्ति-शून्य उत्पाद
        जीवित रहें)।
  \item होने देना $\rho \to 1^+$ और निष्कर्ष निकालो \emph{क्षेत्र
        प्रमेय}:
        \[
        \sum_{n\geq1}n\,\abs{b_n}^2 \;\leq\; 1 .
        \]
        विशेष रूप से $\abs{b_1} \leq 1$. कब है $\abs{b_1}
        = 1$?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Bieberbach's $\abs{a_2} \leq
2$.} होने देना $ f = z + a_2z^2 + \cdots \in \mathcal S $.
\begin{enumerate}[resume]
  \item बताते हैं कि $ f(z^2)/z^2$ है \omterm{def:b3:holomorphic:holo}{होलोमार्फिक} और शून्\omterm{def:b3:modules:free}{य-मुक्त}
        पर $\mathbb D $, और स्वीकार करता है ए \omterm{def:b3:holomorphic:holo}{होलोमार्फिक} वर्गमूल
        $\varphi $ साथ $\varphi(0) = 1$; तय करना $ h(z) =
        z\varphi(z^2)$, ताकि $ h(z)^2 = f(z^2)$. बताते हैं कि
        $ h $ एक \emph{विषम} एकसमान कार्य चालू $\mathbb
        D $ विस्तार के साथ $ h(z) = z + \frac{a_2}2z^3 +
        \cdots $. \emph{(इंजेक्शन: $ h(z)^2 = h(z')^2$
        बल $ z^2 = z'^2$; समाप्त करने के लिए विषमता का उपयोग करें।)}
  \item क्षेत्र प्रमेय को लागू करें $ g(w) = 1/h(1/w) = w -
        \frac{a_2}{2w} + \cdots $ पर $\{\abs w > 1\}$
        (असमानता और विस्तार को सत्यापित करें), और निष्कर्ष निकालें
        $\abs{a_2} \leq 2$.
  \item दिखाओ कि \emph{कोएबे समारोह}
        \[
        k(z) = \frac{z}{(1 - z)^2} = \sum_{n\geq1}n\,z^n
        \]
        से संबंधित $\mathcal S $, है $ a_2 = 2$, और मानचित्र
        $\mathbb D $ पर $\C\setminus\intoc{-\infty}{-\frac14}$
        \emph{(लिखना $ k = \frac14\bigl[\bigl(\frac{1+z}{1 -
        z}\bigr)^2 - 1\bigr]$ और छवियों को ट्रैक करें)}: सभी
        आने वाली असमानताएँ तीव्र हैं।
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The quarter theorem.}
\begin{enumerate}[resume]
  \item होने देना $ f \in \mathcal S $ और $ c \notin f(\mathbb D)$.
        बताते हैं कि
        \[
        F(z) = \frac{c\,f(z)}{c - f(z)}
        \]
        से संबंधित $\mathcal S $, और इसके दूसरे की गणना करें
        गुणांक: $ A_2 = a_2 + \frac1c $.
  \item दोनों पर बीबरबैक लागू करें $ f $ और $ F $: निष्कर्ष
        $\abs{\frac1c} \leq \abs{A_2} + \abs{a_2} \leq 4$,
        यानी\ $\abs c \geq \frac14$. \emph{प्रत्येक छोड़ा गया मान
        मापांक है $\geq \frac14$}: $ f(\mathbb D) \supseteq
        D(0, \frac14)$ --- कोएबे का चतुर्थांश प्रमेय. जांचें
        कोएबे समारोह पर तीक्ष्णता.
  \item एक मात्रात्मक रीमैन-मानचित्र अनुमान निकालें: यदि
        $\varphi \colon \Omega \to \mathbb D $ रीमैन है
        का नक्शा \cref{thm:b3:conformal:rmt} पर $ z_0$, तब
        \[
        \frac{d\bigl(z_0, \partial\Omega\bigr)}{4}
        \;\leq\; \frac1{\varphi'(z_0)} \;\leq\;
        4\,d\bigl(z_0, \partial\Omega\bigr)
        \]
        --- कम से कम बची हुई असमानता को लागू करके सिद्ध करें
        कोएबे को $\varphi^{-1}$ उपयुक्त रूप से \omterm{def:b3:groups:normal}{सामान्य}ीकृत, और
        श्वार्ज़ द्वारा सही पर लागू किया गया $\varphi $ डिस्क पर
        $ D(z_0, d) \subseteq \Omega $.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Perspective.}
\begin{enumerate}[resume]
  \item बीबरबैक का अनुमान (1916) $\abs{a_n} \leq n $ सभी के लिए $ n $, केवल घूर्णन के लिए समानता के साथ
        कोएबे समारोह; डी ब्रैंजेस ने 1985 में इसे साबित किया। सत्यापित करें
        कोएबे फ़ंक्शन और उसके लिए हाथ से अनुमान
        घूर्णन $\eu^{-\iu\theta}k(\eu^{\iu\theta}z)$. फिर
        प्रश्न 4 के विस्तार को एक शब्द आगे बढ़ाएं: लिखना
        $ h(z) = z + \frac{a_2}2z^3 + c_5z^5 + \cdots $, दिखाओ
        $ c_5 = \frac{a_3}2 - \frac{a_2^2}8$ और
        \[
        g(w) = w - \frac{a_2}{2}\,w^{-1} +
        \Bigl(\frac{3a_2^2}8 - \frac{a_3}2\Bigr)w^{-3} +
        \cdots ,
        \]
        इसलिए क्षेत्र प्रमेय परिष्कृत असमानता उत्पन्न करता है
        $\bigl|\frac{a_2}2\bigr|^2 + 3\bigl|\frac{3a_2^2}8 -
        \frac{a_3}2\bigr|^2 \leq 1$. इसे कोएबे पर जांचें
        फ़ंक्शन ($ a_2 = 2$, $ a_3 = 3$).
\end{enumerate}

\medskip
\noindent\textbf{Part V --- The distortion theorem.}
बीबरबैक की असमानता, डिस्क के चारों ओर पहुंचाई गई
ऑटोमोर्फिज्म, नियंत्रण $ f'$ हर जगह. हल करना $ f \in \mathcal
S $.
\begin{enumerate}[resume]
  \item (कोएबे ट्रांसफॉर्म) के लिए $ z_0 \in \mathbb D $ होने देना
        $\varphi(z) = \frac{z + z_0}{1 + \bar z_0z}$, एक डिस्क
        ऑटोमोर्फिज्म (\cref{thm:b3:conformal:autdisc}) साथ
        $\varphi(0) = z_0$. बताते हैं कि
        \[
        F(z) = \frac{f(\varphi(z)) - f(z_0)}
        {f'(z_0)\,\bigl(1 - \abs{z_0}^2\bigr)}
        \]
        से संबंधित $\mathcal S $ \emph{(असमानता है
        विरासत में मिला; गणना करें $\varphi'(0) = 1 - \abs{z_0}^2$
        और \omterm{def:b3:groups:normal}{सामान्य}ीकरण की जाँच करें; याद करना $ f' \neq 0$ के लिए
        इंजेक्शन $ f $,
        \cref{def:b3:conformal:conformal})}.
  \item दूसरे गुणांक की गणना करें $ A_2 = \frac12F''(0)$
        का $ F $:
        \[
        A_2 = \frac12\Bigl[\bigl(1 - \abs{z_0}^2\bigr)
        \frac{f''(z_0)}{f'(z_0)} - 2\bar z_0\Bigr] ,
        \]
        और बीबरबैक (प्रश्न 4) से निष्कर्ष निकालें $ z =
        r\eu^{\iu\theta}$, द \emph{मौलिक
        असमानता}:
        \[
        \Bigl|\,z\,\frac{f''(z)}{f'(z)} -
        \frac{2r^2}{1 - r^2}\Bigr|
        \;\leq\; \frac{4r}{1 - r^2} .
        \]
  \item वास्तविक भाग निकालें:
        \[
        \frac{2r^2 - 4r}{1 - r^2} \;\leq\;
        \operatorname{Re}\Bigl(z\,\frac{f''(z)}{f'(z)}\Bigr)
        \;\leq\; \frac{2r^2 + 4r}{1 - r^2} .
        \]
  \item बताते हैं कि $\frac{\dd}{\dd t}\log\bigl|
        f'(t\eu^{\iu\theta})\bigr| = \frac1t
        \operatorname{Re}\bigl(z\frac{f''(z)}{f'(z)}\bigr)$
        पर $ z = t\eu^{\iu\theta}$ \emph{(एक गैर-लुप्त होने के लिए
        $\mathcal C^1$ समारोह $ g $ एक वास्तविक चर का,
        $\frac{\dd}{\dd t}\log\abs g =
        \operatorname{Re}(g'/g)$)}, और प्रश्न को एकीकृत करें
        प्राप्त करने के लिए किरण के अनुदिश 12 की सीमाएँ
        \emph{विरूपण प्रमेय}:
        \[
        \frac{1 - r}{(1 + r)^3} \;\leq\; \abs{f'(z)}
        \;\leq\; \frac{1 + r}{(1 - r)^3},
        \qquad \abs z = r .
        \]
  \item निष्कर्ष निकालें \emph{विकास प्रमेय}:
        \[
        \frac{r}{(1 + r)^2} \;\leq\; \abs{f(z)} \;\leq\;
        \frac{r}{(1 - r)^2},
        \qquad \abs z = r
        \]
        \emph{(ऊपरी सीमा: एकीकृत करें $ f'$ खंड पर
        $[0, z]$; निचली सीमा: यदि $\abs{f(z)} < \frac14$,
        खंड $[0, f(z)]$ में निहित है $ f(\mathbb D)$ द्वारा
        प्रश्न 7; इसे वापस खींचो $ f^{-1}$ ---
        \omterm{def:b3:holomorphic:holo}{होलोमार्फिक} द्वारा
        \cref{cor:b3:residues:openmapping} --- और बाध्य
        $\abs{f(z)} = \int_\gamma\abs{f'(\zeta)}
        \,\abs{\dd\zeta} \geq \int_0^r\frac{1 - t}{(1 +
        t)^3}\,\dd t $, का उपयोग करना $\abs{\dd\zeta} \geq
        \dd\abs\zeta $)}.
  \item सत्यापित करें कि कोबे फ़ंक्शन समानता प्राप्त करता है
        सभी चार सीमाएँ, पर $ z = r $ ऊपरी लोगों के लिए और
        $ z = -r $ निचले वालों के लिए: $ k $ एक साथ है
        सबसे अधिक विस्तार करने वाला और, एंटीपोड पर, सबसे अधिक
        के अनुबंधित सदस्य $\mathcal S $.
\end{enumerate}

\medskip
\noindent\textbf{Part VI --- Extremal rigidity.} हर में
असमानता अब तक, समानता कोएबे फ़ंक्शन की पहचान करती है
घूमने के लिए. हम इसे साबित करते हैं, फिर फसल काटते हैं।
\begin{enumerate}[resume]
  \item कल्पना करना $ f \in \mathcal S $ है $\abs{a_2} = 2$.
        प्रश्न 2-4 के माध्यम से समानता का पीछा करें: क्षेत्र
        प्रमेय बल $ g(w) = w + b_0 + \eu^{\iu\alpha}/w $;
        की विचित्रता $ h $ बनाता है $ g $ अजीब, तो $ b_0 = 0$; उलटना
        खोजने के लिए $ h $, तब $ f $, और निष्कर्ष निकालो
        \[
        f(z) = \eu^{-\iu\theta}k\bigl(\eu^{\iu\theta}z\bigr)
        \quad\text{with } \eu^{\iu\theta} =
        -\eu^{\iu\alpha} :
        \]
        कोबे फ़ंक्शन के घूर्णन ही एकमात्र हैं
        के सदस्य $\mathcal S $ साथ $\abs{a_2} = 2$.
  \item दिखाओ अगर $ f \in \mathcal S $ एक मान छोड़ देता है $ c $
        साथ $\abs c = \frac14$ बिल्कुल, तो फिर $ f $ एक है
        कोएबे फ़ंक्शन का घूर्णन, और पहचान $ c =
        -\eu^{-\iu\theta}/4$ \emph{(के माध्यम से समानता का पता लगाएं
        प्रश्न 7 की शृंखला $4 = \abs{1/c} = \abs{A_2 - a_2}
        \leq \abs{A_2} + \abs{a_2} \leq 4$)}: तिमाही
        प्रमेय का स्थिरांक केवल चरम द्वारा ही प्राप्त होता है
        परिवार.
  \item (सस्ते पर गुणांक) वृद्धि को संयोजित करें
        प्रमेय के साथ \omterm{thm:b3:holomorphic:analytic}{कॉची अनुमान}
        (\cref{thm:b3:holomorphic:analytic}) वृत्त पर
        $\abs z = 1 - \frac1n $ साबित करना
        \[
        \abs{a_n} \;\leq\; \eu\,n^2 \qquad (n \geq 2) .
        \]
        (डी ब्रैंजेस, 1985: $\abs{a_n} \leq n $; कारक
        $\eu n $ प्राथमिक उपकरणों की कीमत है।)
  \item (उपडिस्कों को कवर करते हुए) इसे प्रत्येक के लिए दिखाएं $0 < r <
        1$,
        \[
        f\bigl(D(0, r)\bigr) \supseteq
        D\Bigl(0, \frac{r}{(1 + r)^2}\Bigr),
        \]
        कोएबे फ़ंक्शन के लिए तेज़, और पुनर्प्राप्त करें
        तिमाही प्रमेय के रूप में $ r \to 1^-$. \emph{(सीमा
        खुली छवि के बिंदु $ f(D(0,r))$ पर झूठ
        $ f(\partial D(0,r))$, इसलिए मापांक है $\geq
        \frac{r}{(1+r)^2}$ प्रश्न 14 द्वारा; से एक खंड
        $0$ छोटे मापांक का एक छूटा हुआ बिंदु होगा
        उस सीमा को पार करने के लिए.)}
  \item (कोएबे हर बिंदु पर) चलो $ f $ पर एकसमान रहें
        $\mathbb D $, जरूरी नहीं कि \omterm{def:b3:groups:normal}{सामान्य}ीकृत हो, और $ z_0
        \in \mathbb D $. सिद्ध करना
        \[
        \tfrac14\bigl(1 - \abs{z_0}^2\bigr)\abs{f'(z_0)}
        \;\leq\;
        d\bigl(f(z_0), \partial f(\mathbb D)\bigr)
        \;\leq\;
        \bigl(1 - \abs{z_0}^2\bigr)\abs{f'(z_0)}
        \]
        \emph{(बाएं: तिमाही प्रमेय कोएबे पर लागू किया गया
        प्रश्न 10 का परिवर्तन; दाएँ: श्वार्ज
        (\cref{thm:b3:conformal:schwarz}) के लिए आवेदन किया
        $\psi^{-1}\circ\hat g $, कहाँ $\hat g(w) =
        f^{-1}\bigl(f(z_0) + dw\bigr)$, $ d $ दूरी,
        और $\psi $ एक डिस्क ऑटोमोर्फिज्म भेजना $0$ को
        $ z_0$)}. क्यों $\partial f(\mathbb D)$ गैररिक्त?
  \item प्रश्न 20 पर जाँच करें $ f = k $ पर $ z_0 = r \in
        \intoo01$: गणना करें $ d\bigl(k(r), \partial k(\mathbb
        D)\bigr) = \frac{(1+r)^2}{4(1-r)^2}$ और उसे सत्यापित करें
        वाम असमानता एक है \emph{समानता}: कोबे
        फ़ंक्शन प्रत्येक बिंदु पर अपने स्वयं के प्रमेय को संतृप्त करता है
        $\intoo01$.
  \item (नैतिक) दस पंक्तियों में: क्या एकल सिद्धांत
        क्षेत्र प्रमेय को रेखांकित करता है, और बीबरबैक कैसे करते हैं,
        तिमाही प्रमेय, विकृति, वृद्धि, और
        इससे सभी प्रवाह को कवर किया जा रहा है? से तुलना करें
        श्वार्ज़--की दुनिया चुनें
        \cref{thm:b3:conformal:schwarz}: दोनों में, एक
        आंतरिक भाग असमानता पूरी ज्यामिति को कठोर बना देती है,
        और चरम घूर्णन तक अद्वितीय हैं।

\end{enumerate}
\medskip
\noindent\textbf{Part VII --- Compactness, inverses, and a
reality check.}
\begin{enumerate}[resume]
  \item दिखाएँ कि \omterm{def:b3:groups:action}{कक्षा} $\mathcal S $ है \emph{कॉम्पैक्ट}
        स्थानीय रूप से समान अभिसरण के लिए: यह स्थानीय रूप से है
        विकास प्रमेय से घिरा हुआ, इसलिए \omterm{def:b3:groups:normal}{सामान्य} है
        (\cref{thm:b3:conformal:montel}); और एक स्थानीय स्तर पर
        के सदस्यों की एक समान सीमा $\mathcal S $ फिर से अंदर है
        $\mathcal S $ \emph{(\omterm{def:b3:groups:normal}{सामान्य}ीकरण पास हो जाता है
        डेरिवेटिव के वीयरस्ट्रैस अभिसरण द्वारा सीमा;
        हर्विट्ज़ द्वारा इंजेक्शन जीवित रहता है,
        \cref{exo:b3:residues:8}, सीमा है
        अस्थिर)}. अतिवादियों के लिए यह बात क्यों मायने रखती है?
        बीबरबैक जैसी समस्याएं?
  \item के लिए $ f \in \mathcal S $, होने देना $ g = f^{-1}$, परिभाषित
        निकट $0$. दिखाओ $ g(w) = w - a_2w^2 + O(w^3)$, तो
        व्युत्क्रम का दूसरा गुणांक समान तीव्र का पालन करता है
        बाध्य $\abs{A_2} = \abs{a_2} \leq 2$, समानता के साथ
        बिल्कुल घुमाए गए कोएबे फ़ंक्शंस के लिए।
  \item किसके लिए निर्धारित करें $ a \in \C $ बहुपद $ f(z)
        = z + az^2$ से संबंधित $\mathcal S $: दिखाओ $ f $ है
        इंजेक्शन चालू $\mathbb D $ आईएफएफ $\abs a \leq \frac12$
        \emph{(कारक $ f(z_1) - f(z_2)$)}. निष्कर्ष: के लिए
        घात-दो बहुपदों का वास्तविक गुणांक परिबद्ध है
        $\frac12$, बीबरबैक से चार गुना छोटा $2$
        --- की चरम सीमा $\mathcal S$ वास्तव में हैं
        पारलौकिक वस्तुएँ, और कोई बहुपद नहीं आता
        \omterm{def:b3:topology:interior}{बंद} करें.
\end{enumerate}
\end{problem}
