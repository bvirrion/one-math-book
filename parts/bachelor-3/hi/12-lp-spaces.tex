\chapter{\texorpdfstring{$L^p$}{Lp} स्थान}\label{ch:b3:lp}

लेबेस्ग इंटीग्रल विश्लेषण के लिए बनाया गया था; $ L^p $ रिक्त स्थान
यहीं वह विश्लेषण रहता है। वे बानाच स्थान हैं
(रिस्ज़--फिशर) --- द पूर्तियों वह
\cref{exo:b3:complete:1} दिखाया \omterm{def:b3:topology:continuity}{निरंतर} कार्य
कमी है --- और वे एक स्मूथिंग तकनीक का समर्थन करते हैं,
\emph{के विरुद्ध दृढ़ संकल्प मोलिफ़ायर}, यह किसी भी का अनुमान लगाता है
$ L^p $ द्वारा कार्य करें $\mathcal C^\infty $ वाले. यह अध्याय सिद्ध करता है
एच के अभिन्न संस्करण\"वृद्ध और मिन्कोव्स्की, पूर्णता,
द \omterm{ex:b3:lebesgue:gamma}{घनत्व} प्रमेय, और नियमितीकरण मशीन, समाप्त
के समावेशन और प्रक्षेप भूगोल के साथ $ L^p $
पैमाना। लगातार, $(X, \mathcal A, \mu)$ एक है स्थान मापें
और फ़ंक्शन जटिल-मूल्यवान हैं; पर $\R^d $, द \omterm{def:b3:measure:measure}{उपाय} है
$\lambda_d $.

\section{परिभाषा; एच\"वृद्ध और मिन्कोव्स्की}

\begin{definition}\label{def:b3:lp:space}
के लिए $1 \leq p < \infty $, $\mathcal L^p(\mu)$ का सेट है
\omterm{def:b3:lebesgue:measurable}{औसत दर्जे का} $ f $ साथ $\norm f_p = \bigl(\int\abs
f^p\dd\mu\bigr)^{1/p} < \infty $, और $\mathcal L^\infty(\mu)$
के समुच्चय $ f $ एक अशक्त सेट के बाहर घिरा हुआ, साथ $\norm
f_\infty $  \emph{आवश्यक समर्थन} --- कम से कम $ M $ साथ
$\abs f \leq M $ ए.ई.\ (जानकारी प्राप्त होती है: शून्य को प्रतिच्छेद करें
के लिए सेट करता है $ M + \frac1n $). तब से $\norm f_p = 0$ केवल बल $ f
= 0$ \emph{ए.ई.} (\cref{exo:b3:lebesgue:5}), हम परिभाषित करते हैं
\[
L^p(\mu) = \mathcal L^p(\mu)/\{f = 0 \text{ a.e.}\} :
\]
तत्व फ़ंक्शंस \omterm{def:b3:modules:module}{मॉड्यूल}ो नल सेट के वर्ग हैं, और
$\norm\cdot_p $ पर एक वास्तविक \omterm{def:b3:rings:ideal}{आदर्श} है
$ L^p $.\index{एलपी स्पेस@$ L^p $ स्पेस}
\end{definition}

\begin{theorem}[एच\"वृद्धों की असमानता]\label{thm:b3:lp:holder}
होने देना $1 \leq p, q \leq \infty $ साथ $\frac1p + \frac1q = 1$
(संयुग्म घातांक). के लिए \omterm{def:b3:lebesgue:measurable}{औसत दर्जे का} $ f, g $:\index{Hölder's
inequality}
\[
\norm{fg}_1 \leq \norm f_p\,\norm g_q ,
\]
समानता के साथ (के लिए) $1 < p < \infty $, परिमित मानदंड, $ f,g\ne0$)
यदि $\abs f^p $ और $\abs g^q $ आनुपातिक हैं ae
\end{theorem}

\begin{proof}
मामले $\{p, q\} = \{1, \infty\}$ प्रत्यक्ष हैं ($\abs{fg} \leq
\norm g_\infty\abs f $ ए.ई.) होने देना $1 < p < \infty $; \omterm{def:b3:groups:normal}{सामान्य}
$\norm f_p = \norm g_q = 1$ (एकरूपता; शून्य या अनंत
मानदंड तुच्छ हैं)। यंग की असमानता $ ab \leq \frac{a^p}p +
\frac{b^q}q $ ($ a, b \geq 0$; की अवतलता $\ln $, के रूप में
\cref{pb:b3:banach:1}) बिंदुवार देता है $\abs{f g} \leq
\frac{\abs f^p}p + \frac{\abs g^q}q $; एकीकृत करें: $\norm{fg}_1
\leq \frac1p + \frac1q = 1$. समानता बल ए.ई.\ में समानता
युवा, यानी\ $\abs f^p = \abs g^q $ ए.ई.\ (के बाद
\omterm{def:b3:groups:normal}{सामान्य}ीकरण; इसे पूर्ववत करना, आनुपातिकता)।
\end{proof}

\begin{theorem}[मिन्कोवस्की की असमानता]\label{thm:b3:lp:minkowski}
के लिए $1 \leq p \leq \infty $: $\norm{f + g}_p \leq \norm f_p +
\norm g_p $.\index{मिन्कोवस्की की असमानता}
\end{theorem}

\begin{proof}
$ p = 1, \infty $: बिंदुवार/ए.ई.\ त्रिकोण असमानता. के लिए $1 <
p < \infty $, मान लीजिए $\norm{f+g}_p < \infty $ (अन्यथा उपयोग करें
$\abs{f+g}^p \leq 2^{p-1}(\abs f^p + \abs g^p)$, उत्तलता से
का $ t^p $, यह देखने के लिए कि बायां पक्ष सीमित है जबकि दायां भाग सीमित है)।
तब
\[
\norm{f{+}g}_p^p \leq \int\abs f\,\abs{f{+}g}^{p-1} +
\int\abs g\,\abs{f{+}g}^{p-1}
\leq \bigl(\norm f_p + \norm g_p\bigr)\,
\bigl\|\abs{f{+}g}^{p-1}\bigr\|_q
\]
एच द्वारा\"पुराने, और $\norm{\abs{f+g}^{p-1}}_q =
\norm{f+g}_p^{p/q}$ तब से $(p-1)q = p $; द्वारा विभाजित करें
$\norm{f+g}_p^{p/q}$ (यदि शून्येतर; अन्यथा तुच्छ) और उपयोग करें $ p -
\frac pq = 1$.
\end{proof}

\section{पूर्णता और उसके साथी}

\begin{theorem}[रिज़्ज़--फिशर]\label{thm:b3:lp:complete}
के लिए $1 \leq p \leq \infty $, $ L^p(\mu)$ एक बानाच स्थान है।
इसके अलावा, प्रत्येक अनुक्रम में अभिसरण होता है $ L^p $ एक अनुवर्ती है
एकत्रित होना \emph{\omterm{def:b3:lebesgue:l1}{लगभग हर जगह}} (एक साथ $ L^p $ प्रभुत्व रखने वाला
मामले में $ p < \infty $).\index{रिज़्ज़--फिशर प्रमेय}
\end{theorem}

\begin{proof}
$ p = \infty $: ए $\norm\cdot_\infty $-कौची अनुक्रम है, \omterm{def:b3:topology:interior}{बंद} ए
एकल अशक्त समुच्चय (अनेक का संघ), समान रूप से कॉची:
यह इससे समान रूप से एकत्रित होता है; हो गया। होने देना $ p < \infty $. द्वारा
\cref{exo:b3:complete:1}(बी) यह बिल्कुल संक्षेप में बताने के लिए पर्याप्त है
अभिसरण श्रृंखला: चलो $\sum\norm{f_k}_p = M < \infty $. तय करना
$ G_n = \sum_{k \leq n}\abs{f_k}$ और $ G = \sum_k\abs{f_k}$
(बिंदुवार में $[0,\infty]$): मिन्कोव्स्की द्वारा $\norm{G_n}_p \leq
M $, और एमसीटी ($ G_n^p \nearrow G^p $) देता है $\int G^p \leq M^p $:
$ G < \infty $ ए.ई., तो श्रृंखला $\sum f_k(x)$ जुटता है
बिल्कुल ए.ई. के लिए\ $ x $; योग को बुलाओ $ S(x)$ (किसी भी मूल्य पर
शून्य सेट)। तब $\abs{S - \sum_{k\leq n}f_k}^p \leq (2G)^p
\in L^1$, और DCT देता है $\norm{S - \sum_{k\leq n}f_k}_p \to 0$:
श्रृंखला में अभिसरण होता है $ L^p $.

अनुवर्ती कथन: यदि $ f_n \to f $ में $ L^p $, चुनना $ n_k $
साथ $\norm{f_{n_{k+1}} - f_{n_k}}_p \leq 2^{-k}$; शृंखला
$\sum(f_{n_{k+1}} - f_{n_k})$ पूर्व के अंतर्गत आता है
तर्क: यह पूरी तरह से अभिसरण करता है, ए पर हावी है $ G \in
L^p $, इसलिए $ f_{n_k} \to f_{n_1} + \sum(\cdots) $ ए.ई., और यह
a.e.-सीमा (का एक प्रतिनिधि) होनी चाहिए $ f $ (दोनों हैं $ L^p $
सीमाएँ)। प्रभुत्वकर्ता: $\abs{f_{n_k}} \leq \abs{f_{n_1}} + G $.
\end{proof}

\begin{remark}\label{rem:b3:lp:modes}
$ L^p $ अभिसरण का अर्थ ए.ई. नहीं है\ अभिसरण (
\emph{टाइपराइटर} अनुक्रम, \cref{exo:b3:lp:3}), न ही
इसके विपरीत (धक्कों से बचना): दो मोड केवल \omterm{def:b3:topology:connected}{जुड़े हुए} हैं
परिणाम और प्रभुत्व के माध्यम से. प्रति उदाहरण रखते हुए
का \cref{exo:b3:lp:3} दिमाग में सबसे अच्छा टीका है.
\end{remark}

\section{घनत्व प्रमेय}

\begin{theorem}\label{thm:b3:lp:density}
होने देना $1 \leq p < \infty $.
\begin{enumerate}
  \item \omterm{def:b3:lebesgue:simple}{सरल कार्य} (साथ परिमित-माप समर्थन) हैं
        में \omterm{def:b3:topology:interior}{घना} $ L^p(\mu)$.
  \item में $ L^p(\R^d)$, द \omterm{def:b3:topology:continuity}{निरंतर} सघन रूप से समर्थित
        कार्य $\mathcal C_c(\R^d)$ सघन हैं.
  \item अनुवाद है \omterm{def:b3:topology:continuity}{निरंतर} पर $ L^p(\R^d)$: लिखना
        $\tau_hf = f(\cdot - h)$, $\norm{\tau_hf - f}_p \to 0$
        जैसा $ h \to 0$.
\end{enumerate}
तीनों में से कोई भी टिकता नहीं है $ p = \infty $.
\end{theorem}

\begin{proof}
(1) के लिए $ f \geq 0$: डायडिक $ s_n \nearrow f $ का
\cref{thm:b3:lebesgue:approximation} संतुष्ट $\abs{f - s_n}^p
\leq f^p \in L^1$: डीसीटी. (प्रत्येक $ s_n \leq f $ में निहित है $ L^p $, और
इसके स्तर सेट सीमित हैं \omterm{def:b3:measure:measure}{उपाय} मूल्य कहां है
सकारात्मक: $\mu(s_n \geq c) \leq c^{-p}\int f^p $.) विभाजन
\omterm{def:b3:groups:normal}{सामान्य} $ f $ चार गैर-नकारात्मक भागों में।

(2) (1) से यह अनुमान लगाना पर्याप्त है $\mathbf 1_A $, $ A $ बोरेल
के साथ $\lambda_d(A) < \infty $. नियमितता (प्रमाण के रूप में)
\cref{thm:b3:measure:regularity}) देता है कॉम्पैक्ट $ K \subseteq A
\subseteq U $ के साथ खोलें $\lambda_d(U\setminus K) <
\varepsilon $; उरीसोहन का कार्य
\[
\varphi(x) = \frac{d(x, \R^d\setminus U)}{d(x, \R^d\setminus U)
+ d(x, K)}
\]
है \omterm{def:b3:topology:continuity}{निरंतर}, $1$ पर $ K $, $0$ बाहर $ U $, और लिया जा सकता है
सघन रूप से समर्थित (सिकुड़ें)। $ U $ पहले एक बंधे हुए खुले में)। तब
$\norm{\mathbf 1_A - \varphi}_p^p \leq \lambda_d(U\setminus K)
< \varepsilon $.

(3) के लिए $ g \in \mathcal C_c $: वर्दी निरंतरता देता है
$\norm{\tau_hg - g}_\infty \to 0$, एक निश्चित में समर्थन के साथ
कॉम्पैक्ट के लिए $\abs h \leq 1$: $\norm{\tau_hg - g}_p \to 0$. के लिए
\omterm{def:b3:groups:normal}{सामान्य} $ f $: चुनना $ g \in \mathcal C_c $ साथ $\norm{f - g}_p <
\varepsilon $; तब $\norm{\tau_hf - f}_p \leq 2\norm{f - g}_p +
\norm{\tau_hg - g}_p $ (मानदंड का अनुवाद अपरिवर्तनीयता)।

के लिए $ p = \infty $: का एकसमान सन्निकटन $\mathbf
1_{\intoo0\infty}$ द्वारा \omterm{def:b3:topology:continuity}{निरंतर} कार्य असंभव है
(छलांग), और $\norm{\tau_h\mathbf 1_{\intoo0\infty} - \mathbf
1_{\intoo0\infty}}_\infty = 1$ के लिए $ h \neq 0$.
\end{proof}

\section{कनवल्शन और नियमितीकरण}

\begin{theorem}[यंग की असमानता]\label{thm:b3:lp:young}
होने देना $1 \leq p \leq \infty $, $ f \in L^1(\R^d)$, $ g \in
L^p(\R^d)$. तब $ f * g $ परिभाषित किया गया है अर्थात्, से संबंधित है $ L^p $,
और\index{यंग की असमानता}
\[
\norm{f * g}_p \leq \norm f_1\,\norm g_p .
\]
\end{theorem}

\begin{proof}
$ p = \infty $: प्रत्यक्ष बाध्य. $ p = 1$:
\cref{thm:b3:product:convolution}. होने देना $1 < p < \infty $, $ q $
संयुग्मित. विभाजित करना $\abs{f(y)} = \abs{f(y)}^{1/q}\cdot
\abs{f(y)}^{1/p}$ और एच लागू करें\"पुराना:
\[
\int\abs{f(y)}\,\abs{g(x{-}y)}\,\dd y
\leq \Bigl(\int\abs f\Bigr)^{1/q}
\Bigl(\int\abs{f(y)}\,\abs{g(x - y)}^p\,\dd y\Bigr)^{1/p} .
\]
तक बढ़ाएँ $ p $-th शक्ति और में एकीकृत करें $ x $; टोनेली पर
दूसरा कारक देता है $\norm f_1^{p/q}\cdot\norm f_1\norm g_p^p $,
यानी\ $\norm{f*g}_p^p \leq \norm f_1^{1 + p/q}\norm g_p^p =
(\norm f_1\norm g_p)^p $ --- और टोनेली की सीमितता
इंटीग्रल एई को उचित ठहराता है\ पूर्ण अभिसरण जैसा कि
\cref{thm:b3:product:convolution}.
\end{proof}

\begin{definition}[मोलिफ़ायर]\label{def:b3:lp:mollifier}
समारोह\index{शांत करनेवाला}\index{टक्कर समारोह}
\[
\rho(x) = \begin{cases}
c\,\exp\Bigl(-\dfrac{1}{1 - \norm x^2}\Bigr) & \norm x < 1,\\
0 & \norm x \geq 1,
\end{cases}
\]
साथ $ c $ \omterm{def:b3:groups:normal}{सामान्य} $\int\rho = 1$, है $\mathcal C^\infty $ पर
$\R^d $: बात यह है $ t \mapsto \eu^{-1/t}\mathbf
1_{t>0}$ है $\mathcal C^\infty $ पर $\R $, इसके सभी डेरिवेटिव पर
$0^+$ प्राणी $0$ (प्रत्येक व्युत्पन्न है $ P(1/t)\eu^{-1/t}$ एक के लिए
बहुपद $ P $, जिसकी प्रवृत्ति होती है $0$; प्रेरण)। के लिए
$\varepsilon > 0$ तय करना $\rho_\varepsilon(x) =
\varepsilon^{-d}\rho(x/\varepsilon)$: में समर्थित $\bar B(0,
\varepsilon)$, अभी भी अभिन्न का $1$.
\end{definition}

\begin{theorem}[नियमितीकरण]\label{thm:b3:lp:regularization}
होने देना $1 \leq p < \infty $ और $ f \in L^p(\R^d)$. तब:
\begin{enumerate}
  \item $ f * \rho_\varepsilon \in \mathcal C^\infty(\R^d)$,
        साथ $\partial^\alpha(f * \rho_\varepsilon) = f *
        \partial^\alpha\rho_\varepsilon $;
  \item $\norm{f * \rho_\varepsilon - f}_p \to 0$ जैसा
        $\varepsilon \to 0$;
  \item फलस्वरूप $\mathcal C^\infty_c(\R^d)$ में सघन है
        $ L^p(\R^d)$.
\end{enumerate}
\end{theorem}

\begin{proof}
(1)अभिन्न के अंतर्गत भेद
(\cref{thm:b3:lebesgue:paramdiff}) में $ x $: के लिए $ x $ एक गेंद में
$ B $, $\abs{\partial_{x_i}\rho_\varepsilon(x - y)} \leq
C_\varepsilon\,\mathbf 1_{K}(y)$ साथ $ K $ कॉम्पैक्ट ($ y $ अंदर
$\varepsilon $ का $ B $), और $\abs f\,\mathbf 1_K \in L^1$
(एच\"पुराने खिलाफ $\mathbf 1_K $): प्रमेय लागू होता है;
उच्च डेरिवेटिव के लिए पुनरावृत्त करें।

(2) चूँकि $\int\rho_\varepsilon = 1$:
\[
(f * \rho_\varepsilon)(x) - f(x)
= \int \bigl(f(x - y) - f(x)\bigr)\rho_\varepsilon(y)\,\dd y ,
\]
और मिन्कोव्स्की का \emph{अभिन्न} असमानता --- या सीधे:
एच\"संभावना के साथ पुराने/जेन्सेन \omterm{def:b3:measure:measure}{उपाय}
$\rho_\varepsilon\dd y $ और टोनेली ---
\[
\norm{f*\rho_\varepsilon - f}_p^p
\leq \int\Bigl(\int\abs{f(x-y) -
f(x)}^p\dd x\Bigr)\rho_\varepsilon(y)\,\dd y
= \int \norm{\tau_yf - f}_p^p\;\rho_\varepsilon(y)\,\dd y
\]
(मध्य चरण: जेन्सेन की असमानता लागू करें,
\cref{exo:b3:lp:10}, भीतर तक $ y $-इंटीग्रल, फिर टोनेली)।
इंटीग्रैंड का समर्थन किया जाता है $\norm y \leq \varepsilon $ और
की ओर प्रवृत्त होता है $0$ वहाँ समान रूप से के रूप में $\varepsilon \to 0$
(\cref{thm:b3:lp:density}(3)): संपूर्ण अभिव्यक्ति प्रवृत्त होती है
$0$.

(3) अनुमानित $ f $ द्वारा $ g \in \mathcal C_c $
(\cref{thm:b3:lp:density}(2)), फिर $ g $ द्वारा $ g *
\rho_\varepsilon \in \mathcal C_c^\infty $ (कॉम्पैक्ट समर्थन:
समर्थन का योग)।
\end{proof}

\begin{example}[शांत करनेवाला $\abs x $, दरों के साथ]
\label{ex:b3:lp:mollifyexample}
लेना $ f(x) = \abs x $ पर $\R $ (स्थानीय तौर पर $ L^1$; प्रमेय
प्रत्येक बाउंड विंडो पर लागू होता है) और एक सममित \omterm{def:b3:lp:mollifier}{शांत करनेवाला}
$\rho_\varepsilon $. तब
\[
f_\varepsilon(x) = (f * \rho_\varepsilon)(x)
= \int\abs{x - y}\,\rho_\varepsilon(y)\,\dd y
\]
है $\mathcal C^\infty $; किंक से दूर, कुछ नहीं होता:
के लिए $\abs x \geq \varepsilon $, $\abs{x - y}$ में रैखिक है $ x $
के समर्थन पर $\rho_\varepsilon $, इसलिए $ f_\varepsilon(x) =
\abs x $ \emph{बिल्कुल} (समरूपता सुधार को समाप्त कर देती है)।
पास में $0$, चौरसाई लागत सटीक रूप से
\[
0 \leq f_\varepsilon(0) = \int\abs
y\,\rho_\varepsilon(y)\,\dd y \leq \varepsilon,
\qquad
\norm{f_\varepsilon - f}_\infty \leq \varepsilon :
\]
सन्निकटन त्रुटि यहीं तक सीमित है
$\varepsilon $-एकवचन का पड़ोस और उसका है
आकार. इस दौरान $ f_\varepsilon'' \geq 0$ हर जगह
($ f $ उत्तल है, और कनवल्शन विरुद्ध है $\rho_\varepsilon
\geq 0$ उत्तलता को बरकरार रखता है), साथ में $\int f_\varepsilon'' =
f_\varepsilon'(\infty) - f_\varepsilon'(-\infty) = 2$: द
दूसरा व्युत्पन्न द्रव्यमान का एक उभार है $2$ चौड़ाई में निचोड़ा हुआ
$ O(\varepsilon)$, इसलिए $\norm{f_\varepsilon''}_\infty \gtrsim
\varepsilon^{-1}$. स्मूथिंग एक व्यापार है: एकसमान त्रुटि
$ O(\varepsilon)$ व्युत्पन्न ब्लो-अप के विरुद्ध
$ O(\varepsilon^{-1})$ --- सटीक विनिमय दर
मात्रात्मक विश्लेषण (प्रक्षेप असमानताएं,
\cref{pb:b3:lp:1}विचारों का चक्र) औपचारिक हो जाता है।
\end{example}

\begin{corollary}[के कलन की मौलिक प्रमेयिका
विविधताएँ]\label{cor:b3:lp:fundlemma}
होने देना $ f \in L^1_{\mathrm{loc}}(\R^d)$ (\omterm{def:b3:lebesgue:l1}{समाकलनीय} पर कॉम्पैक्ट कैमरों)
साथ $\int f\varphi = 0$ हरएक के लिए $\varphi \in \mathcal
C^\infty_c(\R^d)$. तब $ f = 0$ ए.ई.\index{की मौलिक प्रमेयिका
विविधताओं की गणना}
\end{corollary}

\begin{proof}
एक गेंद ठीक करो $ B = B(0, R)$ और चलो $ g = f\mathbf 1_{B(0, R+1)}
\in L^1$. के लिए $ x \in B $ और $\varepsilon < 1$: $(g *
\rho_\varepsilon)(x) = \int f(y)\rho_\varepsilon(x - y)\dd y =
0$, परीक्षण कार्य किया जा रहा है $ y \mapsto \rho_\varepsilon(x-y)
\in \mathcal C_c^\infty $. लेकिन $ g * \rho_\varepsilon \to g $ में
$ L^1$ (\cref{thm:b3:lp:regularization}): $ g = 0$ ए.ई.\ पर $ B $;
निकास $\R^d $.
\end{proof}

\section{ \texorpdfstring{$ L^p $}{Lp} भूगोल}

\begin{proposition}\label{prop:b3:lp:inclusions}
(ए) यदि $\mu(X) < \infty $ और $1 \leq p \leq q \leq \infty $,
तब $ L^q \subseteq L^p $ साथ $\norm f_p \leq
\mu(X)^{\frac1p - \frac1q}\,\norm f_q $.
(बी) चालू $\R^d $ (अनंत \omterm{def:b3:measure:measure}{उपाय}) कोई समावेशन नहीं है:
के लिए $ p \neq q $ में कार्य हैं $ L^p\setminus L^q $.
(सी) (इंटरपोलेशन) यदि $ p < r < q $ और $\alpha \in \intoo01$
द्वारा परिभाषित किया गया है $\frac1r = \frac\alpha p + \frac{1 - \alpha}q $,
तब
\[
\norm f_r \leq \norm f_p^{\alpha}\,\norm f_q^{1 - \alpha} ;
\]
विशेष रूप से $ L^p \cap L^q \subseteq L^r $.
\end{proposition}

\begin{proof}
(ए) एच\"घातांक के साथ पुराने $\frac qp $ और इसका संयुग्म:
$\int\abs f^p\cdot 1 \leq \norm{\abs f^p}_{q/p}\,\norm
1_{(q/p)'} = \norm f_q^p\,\mu(X)^{1 - p/q}$ ($ q = \infty $
सीधे)। (बी) निकट $0$ और पास $\infty $, शक्तियाँ $ x^{-\alpha}$
अंशांकन: \cref{exo:b3:lp:2}. (सी) लिखें $\abs f^r = \abs
f^{r\alpha}\abs f^{r(1-\alpha)}$ और एच लागू करें\"के साथ पुराने
संयुग्मी युग्म $\frac p{r\alpha}$, $\frac q{r(1-\alpha)}$
(की परिभाषा के अनुसार सटीक रूप से संयुग्मित होता है $\alpha $):
$\int\abs f^r \leq \norm f_p^{r\alpha}\norm f_q^{r(1 -
\alpha)}$.
\end{proof}

\begin{method}\label{met:b3:lp:toolkit}
 $ L^p $ टूलकिट, जैसा कि नीचे हर जगह उपयोग किया गया है: साबित करने के लिए
सभी के लिए पहचान या असमानता $ f \in L^p $ --- इसे a पर सिद्ध करें
सघन वर्ग ($\mathcal C_c^\infty $ के जरिए
\cref{thm:b3:lp:regularization}) और द्वारा विस्तार करें निरंतरता
(\cref{thm:b3:complete:extension}, दोनों पक्ष हैं
$ L^p $-निरंतर); साबित करना $ f = 0$, के विरुद्ध परीक्षण करें $\mathcal
C_c^\infty $ (\cref{cor:b3:lp:fundlemma}); चिकनाहट पाने के लिए,
उलझाना; व्यापार घातांकों के लिए, एच\"पुराना और प्रक्षेप। द
फूरियर सिद्धांत \cref{ch:b3:fouriertransform} एक लंबा है
इस विधि का अनुप्रयोग.
\end{method}

\section{समस्या: हार्डी की असमानता}

\begin{problem}[{सप्ताहांत समस्या---हार्डी की असमानता और
यह तीव्र स्थिरांक है}]\label{pb:b3:lp:1}
के लिए $ f \in L^p(\intoo0{+\infty})$, $1 < p < \infty $, को परिभाषित करो \emph{हार्डी ऑपरेटर}
\[
(Hf)(x) = \frac1x\int_0^x f(t)\,\dd t .
\]
हार्डी की असमानता (1920) का दावा है\index{हार्डी की असमानता}
\[
\norm{Hf}_p \;\leq\; \frac{p}{p-1}\,\norm f_p ,
\]
और स्थिरांक $\frac p{p-1}$ इष्टतम है और प्राप्त नहीं हुआ है।
यह समस्या सब कुछ साबित करती है, फिर श्रृंखला तक फैल जाती है।

\medskip
\noindent\textbf{Part I --- The inequality.} पहले मान लीजिये $ f
\geq 0$ \omterm{def:b3:topology:continuity}{निरंतर} साथ कॉम्पैक्ट समर्थन करें
$\intoo0{+\infty}$, और चलो $ F(x) = \int_0^xf $.
\begin{enumerate}
  \item बताते हैं कि $ Hf \in L^p $: पास में $0$, $ F $ पर गायब हो जाता है
        \omterm{def:b3:topology:topology}{अड़ोस-पड़ोस} का $0$; पास में $\infty $, $ F $ घिरा हुआ है, इसलिए
        $(Hf)(x) = O(1/x)$, और $ x \mapsto \frac1x $ से संबंधित
        $ L^p(\intoo1{+\infty})$ के लिए $ p > 1$.
  \item दिखाने के लिए भागों द्वारा एकीकृत करें
        \[
        \int_0^\infty \Bigl(\frac Fx\Bigr)^p\dd x
        = \frac{p}{p-1}\int_0^\infty\Bigl(\frac
        Fx\Bigr)^{p-1}f(x)\,\dd x .
        \]
        \emph{(अंतर करें $ x^{1-p}F^p $; सीमा शर्तें
        लुप्त --- दोनों सिरों को उचित ठहराएँ।)}
  \item एच लागू करें\"दाहिनी ओर पुराना और निष्कर्ष निकालें
        $\norm{Hf}_p \leq \frac p{p-1}\norm f_p $ ऐसे के लिए $ f $.
  \item सभी तक विस्तार करें $ L^p $: के लिए $ f \geq 0$, निर्माण
        $ f_n $ \omterm{def:b3:topology:continuity}{निरंतर} साथ कॉम्पैक्ट समर्थन करें
        $\intoo0{+\infty}$, $0 \leq f_n \nearrow f $ ए.ई.\
        \emph{(काट-छांट करें, फिर नीरस रूप से अनुमानित करें ---
        निर्माण का औचित्य सिद्ध करें)}; तब $ Hf_n \nearrow Hf $
        बिंदुवार (औसत के अंदर एमसीटी) और एमसीटी पास हो जाता है
        असमानता की सीमा. हस्ताक्षरित या जटिल के लिए $ f $,
        के साथ समाप्त करें $\abs{Hf} \leq H\abs f $.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Optimality.}
\begin{enumerate}[resume]
  \item के लिए $ A > 1$ होने देना $ f_A(t) = t^{-1/p}\,\mathbf
        1_{\intcc1A}(t)$. गणना करना $\norm{f_A}_p^p = \ln A $
        और के लिए $1 \leq x \leq A $,
        \[
        (Hf_A)(x) = \frac p{p-1}\;x^{-1/p}\,
        \bigl(1 - x^{-(1 - 1/p)}\bigr).
        \]
  \item परिणाम निकालना $\liminf_{A\to\infty} \norm{Hf_A}_p/
        \norm{f_A}_p \geq \frac{p}{p-1}$, और निष्कर्ष निकालो
        स्थिरांक इष्टतम है.
  \item वह समानता दिखाओ $\norm{Hf}_p = \frac
        p{p-1}\norm f_p $ साथ $ f \neq 0$ असंभव है.
        \emph{(एच के समानता मामले को ट्रैक करें\"प्रश्न में वृद्ध
        3: यह मजबूर करेगा $ f = cx^{-1/p}$-प्रकार का व्यवहार, जो
        में नहीं है $ L^p $.)}
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The discrete inequality.}
\begin{enumerate}[resume]
  \item एक गैर वृद्धि के लिए $ g \geq 0$ पर $(0,\infty)$ और
        $ a_n = g(n)$, तुलना करना $\sum a_n^p $ और $\int g^p $,
        और $ H $-तदनुसार औसत, भाग I से निकालना
        \emph{हार्डी की असतत असमानता}: के लिए $ a_n \geq 0$,
        \[
        \sum_{n\geq1}\Bigl(\frac{a_1 + \dots +
        a_n}{n}\Bigr)^{p}
        \;\leq\;
        \Bigl(\frac{p}{p-1}\Bigr)^{p}\,\sum_{n\geq1}a_n^p
        \]
        ---पहले इसे न बढ़ाने के लिए सिद्ध करें $(a_n)$ के माध्यम से
        उपरोक्त तुलना करें, फिर \omterm{def:b3:groups:normal}{सामान्य} मामले को कम करें
        पुनर्व्यवस्था द्वारा एक को न बढ़ाना \emph{(स्वीकार करें, साथ
        एक पंक्ति का औचित्य, वह छँटाई $(a_n)$ में
        घटते क्रम में केवल बायीं ओर ही वृद्धि हो सकती है
        अधिकार ठीक करना)}.
  \item निष्कर्ष निकालें: यदि $\sum a_n^p < \infty $ फिर सेस\`एआरओ
        का मतलब है $(a_n)$ फिर से हैं $\ell^p $ --- और एक दे दो
        उदाहरण ($ p = 2$) कहाँ $(a_n) \in \ell^2$ लेकिन $ a_n $
        सारांशित नहीं है, फिर भी हार्डी अभी भी साधनों को नियंत्रित करता है।
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Epilogue.}
\begin{enumerate}[resume]
  \item दिखाएँ कि हार्डी की असमानता विफल रहती है $ p = 1$: साथ
        $ f = \mathbf 1_{\intcc01}$, गणना करें $ Hf $ और निरीक्षण करें
        $ Hf \notin L^1$. सबूत कहां टूटता है?
\end{enumerate}

\medskip
\noindent\textbf{Part V --- The maximal function, and
Lebesgue's differentiation theorem.} हार्डी का औसत
उत्पत्ति; हार्डी - लिटिलवुड औसत \emph{प्रत्येक बिंदु के आसपास}.
के लिए $ f \in L^1(\R)$ परिभाषित करना
\[
Mf(x) = \sup_{r>0}\ \frac1{2r}\int_{x-r}^{x+r}\abs
f\,\dd\lambda .
\]
\begin{enumerate}[resume]
  \item (विटाली, परिमित संस्करण) चलो $ B_1, \dots, B_N $ हो
        खुले अंतराल. दिखाएँ कि एक असंयुक्त उपपरिवार है
        $ B_{i_1}, \dots, B_{i_k}$ साथ $\bigcup_jB_j
        \subseteq \bigcup_l3B_{i_l}$, कहाँ $3B $ को दर्शाता है
        उसी के साथ अंतराल \omterm{ex:b3:groups:actions}{केंद्र} और तिगुनी लंबाई
        \emph{(लालची: बार-बार सबसे लंबा अंतराल चुनें
        पहले से चुने गए लोगों से अलग)}.
  \item (कमजोर प्रकार $(1,1)$) इसे प्रत्येक के लिए दिखाएँ $ t > 0$,
        \[
        \lambda\bigl(\{Mf > t\}\bigr) \;\leq\;
        \frac3t\,\norm f_1 :
        \]
        प्रत्येक $ x $ साथ $ Mf(x) > t $ एक केन्द्रित अंतराल का स्वामी है
        $ B_x $ साथ $\int_{B_x}\abs f > t\,\lambda(B_x)$; ले लो
        ए कॉम्पैक्ट $ K \subseteq \{Mf > t\}$ (आंतरिक नियमितता),
        इसे बहुत से लोगों द्वारा कवर करें $ B_x $, प्रश्न 11 लागू करें,
        और निकास.
  \item गणना करना $ M\mathbf 1_{\intcc01}(x)$ के लिए $ x > 1$ और
        उसका निष्कर्ष निकालो $ Mf \notin L^1$ हरएक के लिए $ f \neq 0$
        ($ Mf(x) \geq \frac c{\abs x}$ अनंत पर): पर $ p =
        1$प्रश्न 12 की कमजोर असमानता सर्वोत्तम है
        संभावित कथन.
  \item (के लिए मजबूत प्रकार $ p > 1$) के लिए $ f \in L^p $: विभाजित करना $ f =
        f\,\mathbf 1_{\abs f > t/2} + f\,\mathbf 1_{\abs f
        \leq t/2}$, निरीक्षण $ Mf \leq M\bigl(f\mathbf 1_{\abs
        f > t/2}\bigr) + \frac t2$, और प्रश्न 12 को संयोजित करें
        लेयर-केक फ़ॉर्मूले के साथ
        (\cref{prop:b3:product:layercake}) और टोनेली को
        साबित करो
        \[
        \norm{Mf}_p^p \;\leq\;
        \frac{6p\,2^{p-1}}{p-1}\,\norm f_p^p .
        \]
        (ब्लो-अप के रूप में $ p \downarrow 1$ प्रश्न 13 का है
        विफलता, परिमाणित.)
  \item (लेब्सेग विभेदीकरण प्रमेय) सिद्ध करें: के लिए $ f \in
        L^1(\R)$,
        \[
        \frac1{2r}\int_{x-r}^{x+r}f\,\dd\lambda
        \;\longrightarrow\; f(x)
        \qquad (r \to 0)\quad\text{for a.e.\ }x .
        \]
        \emph{(के लिए साफ़ करें \omterm{def:b3:topology:continuity}{निरंतर} $ f $. \omterm{def:b3:groups:normal}{सामान्य} तौर पर लिखें $ f
        = g + h $, $ g $ \omterm{def:b3:topology:continuity}{निरंतर} साथ कॉम्पैक्ट सहायता,
        $\norm h_1 < \varepsilon $
        (\cref{thm:b3:lp:density}); सेट कहाँ है
        $\limsup_{r\to0}$ औसत दोलन से अधिक है
        $\delta $ अंदर बैठता है $\{Mh > \delta/2\} \cup \{\abs h
        > \delta/2\}$, का \omterm{def:b3:measure:measure}{उपाय} $ O(\varepsilon/\delta)$;
        होने देना $\varepsilon \to 0$, तब $\delta \to 0$ साथ में ए
        क्रम.)}
  \item निष्कर्ष निकालें: (ए) लगभग हर बिंदु एक है \emph{लेब्सग्यू
        बिंदु} का $ f $; (बी) के लिए $ f \in L^1$, आदिम
        $ F(x) = \int_0^xf $ अवकलनीय है ए.ई.\ साथ $ F' =
        f $ ए.ई.\ --- मौलिक का अभिन्न अंग
        लेबेस्ग विश्व में कैलकुलस का प्रमेय, \omterm{thm:b3:complete:completion}{समापन}
        सीढ़ी द्वारा खोला गया घेरा
        (\cref{pb:b3:measure:1}), जिसने उलटा दिखाया
        आधा असफल हो सकता है.
  \item (\omterm{ex:b3:lebesgue:gamma}{घनत्व} अंक) के लिए \omterm{def:b3:lebesgue:measurable}{औसत दर्जे का} $ A \subseteq \R $,
        दिखाओ कि लगभग हर $ x \in A $ संतुष्ट
        $\frac{\lambda(A\cap\intcc{x-r}{x+r})}{2r} \to 1$:
        \omterm{def:b3:lebesgue:measurable}{औसत दर्जे का} स्थानीय स्तर पर लगभग सभी सेट भरे हुए हैं
        उनके अंक. इसका उत्पादन कैसे होता है, इसे दो पंक्तियों में रेखांकित करें
        स्टीनहॉस प्रमेय का एक और प्रमाण
        (\cref{exo:b3:measure:8}).
\end{enumerate}

\medskip
\noindent\textbf{Part VI --- Variations on the theme of
averaging.}
\begin{enumerate}[resume]
  \item (वेटेड हार्डी) के लिए $\alpha < p - 1$ और $ f \geq 0$,
        दिखाओ
        \[
        \int_0^\infty\Bigl(\frac{F(x)}x\Bigr)^{p}
        x^{\alpha}\,\dd x \;\leq\;
        \Bigl(\frac{p}{p - 1 - \alpha}\Bigr)^{p}
        \int_0^\infty f(x)^p\,x^{\alpha}\,\dd x
        \]
        भागों द्वारा समान एकीकरण द्वारा, और जाँचें कि
        सीमा रेखा $\alpha = p - 1$ वास्तव में वर्जित है
        (प्रश्न 10 के प्रतिउदाहरण को अनुकूलित करें)।
  \item (संयुक्त) चलो $ H^*f(x) =
        \int_x^{\infty}\frac{f(t)}t\,\dd t $. दिखाओ $\langle
        Hf, g\rangle = \langle f, H^*g\rangle $ के लिए
        गैरनकारात्मक $ f, g $ (टोनेली), और साबित करो
        $\norm{H^*f}_p \leq p\,\norm f_p $ \emph{(सीधे द्वारा
        भागों, या हार्डी से संयुग्म घातांक पर
        द्वैत --- मन जो प्रतिपादक उठाता है जो
        स्थिर)}.
  \item (एक हिल्बर्ट-प्रकार की असमानता) इसके लिए निष्कर्ष निकालें
        गैरनकारात्मक $ f \in L^p $, $ g \in L^q $:
        \[
        \int_0^\infty\!\!\int_0^\infty
        \frac{f(x)\,g(y)}{\max(x,y)}\,\dd x\,\dd y
        \;\leq\; (p + q)\,\norm f_p\,\norm g_q
        \]
        \emph{(साथ में विभाजित करें $ y < x $ / $ y \geq x $: प्रत्येक आधा है
        हार्डी ट्रांसफॉर्म के विरुद्ध एक फ़ंक्शन की जोड़ी
        दूसरे का)}.
  \item (इष्टतमता, असतत) दिखाएँ कि स्थिरांक
        $\bigl(\frac p{p-1}\bigr)^p $ प्रश्न 8 का भी है
        इष्टतम: पर परीक्षण करें $ a_n = n^{-1/p}\,\mathbf 1_{n \leq
        N}$, दोनों पक्षों की तुलना इंटीग्रल से करें, और जाने दें $ N
        \to \infty $ (भाग II का असतत दर्पण)।
  \item (संश्लेषण) तीन औसत ऑपरेटर सामने आए
        यह समस्या: हार्डी की $ H $, असतत सेस\`एआरओ
        माध्य, और अधिकतम संचालिका $ M $. एक में बताएं
        प्रत्येक को पंक्तिबद्ध करें कि उसकी सीमा क्या कहती है, उस सबका निरीक्षण करें
        तीन बिल्कुल विफल हो जाते हैं $ p = 1$, और समझाएं कि ऐसा क्यों है
        एक ही विफलता तीन बार (हार्मोनिक पूंछ
        $\frac1x $).
\end{enumerate}

\medskip
\noindent\textbf{Part VII --- Carleman's inequality, and how
sharp is sharp.}
\begin{enumerate}[resume]
  \item (कार्लमैन की असमानता) चलो $ a_n \geq 0$ साथ $\sum
        a_n < \infty $. की असतत हार्डी असमानता को लागू करें
        प्रश्न 8 से $ b_n = a_n^{1/p}$, का उपयोग करें
        अंकगणित--ज्यामितीय माध्य असमानता, और चलो $ p \to
        \infty $ (बताते हैं कि $ p \mapsto
        \bigl(\frac p{p-1}\bigr)^p $ तक कम हो जाता है $\eu $) प्राप्त करने के लिए
        \[
        \sum_{n\geq1}\bigl(a_1a_2\cdots a_n\bigr)^{1/n}
        \;\leq\; \eu\,\sum_{n\geq1}a_n :
        \]
        संक्षेपणीय अनुक्रम के ज्यामितीय साधन हैं
        सारांश योग्य, अधिकतम लागत पर $\eu $.
  \item (\omterm{def:b3:topology:continuity}{निरंतर} $\eu $ इष्टतम है) परीक्षण $ a_n =
        \frac1n\,\mathbf 1_{n\leq N}$: स्टर्लिंग का उपयोग करना
        की ब्रैकेटिंग \cref{pb:b3:product:1}, दिखाओ
        $(n!)^{-1/n} = \frac\eu n\bigl(1 +
        O\bigl(\frac{\ln n}n\bigr)\bigr)$, यह निष्कर्ष निकालें कि दोनों
        कार्लेमैन के किनारे जैसे बढ़ते हैं $\eu\ln N $, और निष्कर्ष निकालें
        इससे छोटा कोई स्थिरांक नहीं $\eu $ काम कर सकते हैं. (निरीक्षण करें
        पैटर्न: हार्डी और कार्लमैन के अनुकूलक
        दोनों हार्मोनिक-प्रकार के अनुक्रम हैं जो विफल हो जाते हैं
        अंतरिक्ष में रहो.)
  \item (``शार्प'' कितनी धीमी गति से आता है?) लीजिए $ p = 2$.
        के लिए $ f = \mathbf 1_{\intcc01}$, गणना करें
        $\norm{Hf}_2/\norm f_2 = \sqrt2$, सीमा के विरुद्ध
        $2$. निकट-अनुकूलकों के लिए $ f_A $ प्रश्न 5 का,
        सटीक पहचान सिद्ध करें
        \[
        \norm{Hf_A}_2^2 = 4\ln A - 8 + \frac{8}{\sqrt A},
        \qquad\text{so}\qquad
        \frac{\norm{Hf_A}_2^2}{\norm{f_A}_2^2}
        = 4 - \frac{8 - 8A^{-1/2}}{\ln A} .
        \]
        पर मूल्यांकन करें $ A = \eu^{10}$ (अनुपात $\approx 1.790$)
        और टिप्पणी: सर्वोच्च $2$ गति से संपर्क किया जाता है
        $1/\ln A$ केवल --- एक इष्टतम स्थिरांक सभी हो सकता है लेकिन
        संख्यात्मक रूप से अदृश्य.
\end{enumerate}
\end{problem}
