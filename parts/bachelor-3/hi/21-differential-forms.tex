\chapter{विभेदक रूप और स्टोक्स प्रमेय}\label{ch:b3:forms}

विश्लेषण के एक प्रमेय ने, दो शताब्दियों से अधिक समय में, सभी को समाहित कर लिया है
अपनी तरह के अन्य: कैलकुलस का मौलिक प्रमेय,
हरा--रीमैन (वर्ष~2 खंड में सिद्ध), विचलन
गॉस का प्रमेय, केल्विन का कर्ल प्रमेय--स्टोक्स --- प्रत्येक
कहता है कि किसी क्षेत्र पर कुछ व्युत्पन्न का अभिन्न अंग बराबर होता है
मूल वस्तु का अभिन्न अंग सीमा. द
की भाषा \emph{\omterm{def:b3:forms:diffform}{विभेदक रूप}} उन्हें एक बयान देता है,
$\int_M\dd\omega = \int_{\partial M}\omega $, और वह बनाता है
एक झटके में साबित हो सकता है बयान यह अध्याय बनाता है
भाषा ईमानदारी से --- वैकल्पिक बहुरेखीय बीजगणित,
\omterm{def:b3:forms:d}{बाहरी व्युत्पन्न}, पुलबैक, \omterm{def:b3:forms:orientation}{अभिविन्यास}, पर एकीकरण
के उपमानव \cref{ch:b3:submanifolds} --- स्टोक्स साबित करता है'
प्रमेय, और पहले चेक को भुनाता है: शास्त्रीय अभिन्न अंग
प्रमेय, \omterm{def:b3:forms:winding}{घुमावदार संख्या} जो चोरी छुपे भागा
\cref{ch:b3:residues}, और, सप्ताहांत की समस्या में, ब्रौवर की
निश्चित-बिंदु प्रमेय. लगातार, \emph{चिकना} मतलब $\mathcal
C^\infty $; जब तक कहा न जाए, हर नक्शा और फॉर्म सुचारू है
अन्यथा. इस स्तर पर कोई सामान्यता रखने लायक नहीं है
और हाथों को \omterm{def:b3:modules:free}{मुक्त} कर देता है.

\section{वैकल्पिक बहुरेखीय बीजगणित}

\begin{definition}\label{def:b3:forms:alternating}
होने देना $ E $ आयाम का एक वास्तविक सदिश समष्टि बनें $ n $. ए
\emph{$ k $-रैखिक प्रत्यावर्ती रूप}\index{$ k $-राखिक प्रत्ययविद्या रूप} पर
$ E $ एक नक्शा है $\alpha\colon E^k \to \R $, प्रत्येक में रैखिक
परिवर्तनशील, साथ $\alpha(v_1, \dots, v_k) = 0$ जब भी दो
तर्क बराबर हैं. उनका स्थान लिखा हुआ है $\Lambda^k E^*$;
रिवाज के सन्दर्भ मे $\Lambda^0E^* = \R $. प्रत्यावर्तन बल
एंटीसिममेट्री: दो तर्कों के आदान-प्रदान से संकेत बदल जाता है
(विस्तार करें $\alpha(\dots, v + w, \dots, v + w, \dots) = 0$), और
अधिक सामान्यतः $\alpha(v_{\sigma(1)}, \dots, v_{\sigma(k)}) =
\varepsilon(\sigma)\,\alpha(v_1, \dots, v_k)$ प्रत्येक के लिए
क्रमपरिवर्तन $\sigma $.
\end{definition}

\begin{example}
पर $ E = \R^n $: $\Lambda^1E^* = E^*$ है \omterm{def:b3:banach:operator}{दोहरी} जगह; द
विहित \omterm{def:b3:topology:basis}{आधार} पर निर्धारक एक है $ n $-रैखिक
\omterm{def:b3:forms:alternating}{वैकल्पिक रूप}, और \cref{prop:b3:forms:basis} इसे दिखाऊंगा
फैला हुआ $\Lambda^nE^*$ --- गहन कारण निर्धारक है
पैमाने तक अद्वितीय. के लिए $ k > n $, $\Lambda^kE^* = \{0\}$:
$ k $ वैक्टर आश्रित हैं, और एक के साथ दूसरे का विस्तार कर रहे हैं
मारता है $\alpha $ प्रत्यावर्तन द्वारा.
\end{example}

\begin{definition}\label{def:b3:forms:wedge}
के लिए $\ell_1, \dots, \ell_k \in E^*$, उनका \emph{बाहरी
उत्पाद}\index{बाहरी
उत्पाद} है $ k $-रैखिक बारी-बारी से
प्रपत्र
\[
(\ell_1 \wedge \dots \wedge \ell_k)(v_1, \dots, v_k)
= \det\bigl(\ell_i(v_j)\bigr)_{1 \leq i, j \leq k} .
\]
प्रत्यावर्तन और बहुरेखीयता निर्धारक हैं
इसके कॉलम.
\end{definition}

\begin{proposition}[का आधार $\Lambda^k $]\label{prop:b3:forms:basis}
होने देना $(e_1, \dots, e_n)$ का \omterm{def:b3:topology:basis}{आधार} बनें $ E $ साथ \omterm{def:b3:banach:operator}{दोहरी} \omterm{def:b3:topology:basis}{आधार}
$(e_1^*, \dots, e_n^*)$. रूपों
\[
e_I^* = e_{i_1}^* \wedge \dots \wedge e_{i_k}^*,
\qquad I = \{i_1 < \dots < i_k\} \subseteq \{1, \dots, n\},
\]
का एक \omterm{def:b3:topology:basis}{आधार} बनाएं $\Lambda^kE^*$; इस तरह $\dim\Lambda^kE^* =
\binom nk $. स्पष्ट रूप से, $\alpha =
\sum_{\abs I = k}\alpha(e_{i_1}, \dots, e_{i_k})\,e_I^*$.
\end{proposition}

\begin{proof}
\emph{उत्पन्न करना।} होने देना $\alpha \in \Lambda^kE^*$ और $\beta =
\sum_I\alpha(e_I)\,e_I^*$, कहाँ $\alpha(e_I)$ संक्षिप्तीकरण
$\alpha(e_{i_1}, \dots, e_{i_k})$. दोनों पक्ष हैं $ k $-रैखिक
और बारी-बारी से, इसलिए वे सभी पर सहमत होते ही सहमत हो जाते हैं
$ k $-ट्यूपल्स $(e_{j_1}, \dots, e_{j_k})$ साथ $ j_1 < \dots <
j_k $ (बहुरेखीयता \omterm{def:b3:topology:basis}{आधार} वैक्टर के टुपल्स तक कम हो जाती है,
कड़ाई से बढ़ते लोगों का विकल्प)। और $ e_I^*(e_{j_1},
\dots, e_{j_k}) = \det(e_{i_r}^*(e_{j_s})) = \delta_{IJ}$:
के लिए $ I = J $ मैट्रिक्स पहचान है; के लिए $ I \neq J $ कुछ
पंक्ति शून्य है. इसलिए $\beta(e_J) = \alpha(e_J)$ सभी के लिए $ J $:
$\beta = \alpha $. \emph{उन्मुक्तता.} अगर $\sum_I c_Ie_I^* = 0$,
पर मूल्यांकन $(e_{j_1}, \dots, e_{j_k})$ देता है $ c_J = 0$.
\end{proof}

\begin{definition}\label{def:b3:forms:wedgegeneral}
 \omterm{def:b3:forms:wedge}{बाहरी उत्पाद} एक द्विरेखीय मानचित्र तक विस्तारित है
$\Lambda^kE^* \times \Lambda^\ell E^* \to
\Lambda^{k+\ell}E^*$, द्विरेखीयता द्वारा निर्धारित और
$(e_I^*) \wedge (e_J^*) = e_I^* \wedge e_J^*$ (संक्षिप्त करें
और पुन: व्यवस्थित करें; उत्पाद है $0$ अगर $ I \cap J \neq
\varnothing $). यह सहयोगी है, और
\emph{ग्रेडेड-एंटीकम्यूटेटिव}:
\[
\beta \wedge \alpha = (-1)^{k\ell}\,\alpha \wedge \beta
\qquad (\alpha \in \Lambda^k, \ \beta \in \Lambda^\ell).
\]
\end{definition}

\begin{proof}[\omnameProof]
दोनों गुणों को \omterm{def:b3:topology:basis}{आधार} तत्वों के \omterm{def:b3:topology:basis}{आधार} पर जांचा जाता है और बढ़ाया जाता है
द्विरेखीयता साहचर्यता: के दोनों कोष्ठक
$ e_I^* \wedge e_J^* \wedge e_K^*$ की कील के बराबर
का संघटित परिवार $1$-रूप, निर्धारक सूत्र द्वारा
का \cref{def:b3:forms:wedge} (ब्लॉक द्वारा लाप्लास विस्तार)।
संकेत नियम: प्रत्येक को हिलाना $\ell $ के कारक $\beta $
अतीत $ k $ के कारक $\alpha $ प्रति आसन्न एक चिह्न खर्च होता है
ट्रांसपोज़िशन (निर्धारक की दो पंक्तियों की अदला-बदली), इसलिए
$(-1)^{k\ell}$ कुल मिलाकर।
\end{proof}

\begin{proposition}[पुलबैक, रैखिक मामला]
\label{prop:b3:forms:linearpullback}
एक रेखीय मानचित्र $ u\colon E \to F $ प्रत्येक के लिए प्रेरित करता है $ k $, द
रेखीय मानचित्र $ u^*\colon \Lambda^kF^* \to \Lambda^kE^*$,
$(u^*\alpha)(v_1, \dots, v_k) = \alpha(u(v_1), \dots,
u(v_k))$. यह संतुष्ट करता है $ u^*(\alpha \wedge \beta) = u^*\alpha
\wedge u^*\beta $ और $(u \circ w)^* = w^* \circ u^*$.
इसके अतिरिक्त:
\begin{enumerate}
  \item अगर $\dim E = n $ और $ u\colon E \to E $, फिर पर
        लाइन $\Lambda^nE^*$: $ u^*\alpha = (\det u)\,\alpha $.
  \item अगर $\operatorname{rk}u < k $, तब $ u^* = 0$ पर
        $\Lambda^kF^*$.
\end{enumerate}
\end{proposition}

\begin{proof}
कार्यात्मक पहचान परिभाषाओं से तत्काल हैं
(उत्पाद नियम के लिए, वेजेज की जांच करें $1$-के साथ प्रपत्र
निर्धारक सूत्र --- $\det(\ell_i(uv_j)) =
\det((u^*\ell_i)(v_j))$ --- फिर द्विरेखीय रूप से विस्तार करें)।
(1) $ u^*$ एक-आयामी मानचित्र बनाता है $\Lambda^nE^*$
(\cref{prop:b3:forms:basis}) अपने आप को, इसलिए $ u^*\alpha =
c\,\alpha $ साथ $ c $ से स्वतंत्र $\alpha \neq 0$; परीक्षण
पर $\alpha = e_1^* \wedge \dots \wedge e_n^*$ और $(v_j) =
(e_j)$ देता है $ c = \det(e_i^*(ue_j)) = \det u $.
(2) के लिए $ v_1, \dots, v_k \in E $, वैक्टर $ u(v_1), \dots,
u(v_k)$ की छवि में झूठ बोलते हैं $ u $, आयाम का $< k $: वे हैं
रैखिक रूप से निर्भर, और एक \omterm{def:b3:forms:alternating}{वैकल्पिक रूप} पर गायब हो जाता है
आश्रित परिवार (आश्रित वेक्टर का विस्तार करें
अन्य)।
\end{proof}

\section{विभेदक रूप और बाह्य व्युत्पन्न}

\begin{definition}\label{def:b3:forms:diffform}
होने देना $ U \subseteq \R^n $ खुले रहो. ए \emph{अंतर
$ k $-रूप}\index{अंतर
$ क $-रूप} पर $ U $ एक सहज मानचित्र है
$\omega\colon U \to \Lambda^k(\R^n)^*$; के \omterm{def:b3:topology:basis}{आधार} पर
\cref{prop:b3:forms:basis} (साथ $\dd x_i $ के लिए लिखा गया है
$ e_i^*$),
\[
\omega = \sum_{\abs I = k} a_I\,\dd x_I,
\qquad
\dd x_I = \dd x_{i_1} \wedge \dots \wedge \dd x_{i_k},
\]
सहज गुणांक के साथ $ a_I \in \mathcal C^\infty(U)$.
उनका स्थान है $\Omega^k(U)$; $\Omega^0(U) = \mathcal
C^\infty(U)$. ए $0$-फॉर्म एक फ़ंक्शन है; ए $1$-फॉर्म एक फ़ील्ड है
रैखिक रूपों का (उदा.\ अंतर $\dd f $ एक का
फ़ंक्शन); एक $ n $-रूप है $ a\,\dd x_1\wedge\dots\wedge\dd
x_n $, का प्राकृतिक एकीकरण \cref{ch:b3:product}.
\end{definition}

\begin{definition}[बाह्य व्युत्पन्न]\label{def:b3:forms:d}
 \emph{बाहरी व्युत्पन्न}\index{बाहरी व्युत्पन्न} है
रेखीय मानचित्र $\dd\colon \Omega^k(U) \to \Omega^{k+1}(U)$
द्वारा परिभाषित
\[
\dd\Bigl(\sum_I a_I\,\dd x_I\Bigr)
= \sum_I \dd a_I \wedge \dd x_I
= \sum_I\sum_{j=1}^n \frac{\partial a_I}{\partial
x_j}\,\dd x_j \wedge \dd x_I .
\]
पर $0$-रूप यह \omterm{def:b3:groups:normal}{सामान्य} अंतर है।
\end{definition}

\begin{theorem}\label{thm:b3:forms:dsquare}
(ए) $\dd(\omega \wedge \eta) = \dd\omega \wedge \eta +
(-1)^k\,\omega \wedge \dd\eta $ के लिए $\omega \in \Omega^k $
(द \emph{श्रेणीबद्ध लीबनिज नियम}).
(बी) $\dd \circ \dd = 0$.\index{d squared zero@$\dd^2 = 0$}
\end{theorem}

\begin{proof}
(ए) द्विरेखीयता से उपचार करना पर्याप्त है $\omega = a\,\dd
x_I $, $\eta = b\,\dd x_J $. तब $\omega \wedge \eta =
ab\,\dd x_I \wedge \dd x_J $ और
\[
\dd(\omega\wedge\eta) = (b\,\dd a + a\,\dd b) \wedge \dd x_I
\wedge \dd x_J
= (\dd a \wedge \dd x_I) \wedge (b\,\dd x_J) +
a\,\dd b \wedge \dd x_I \wedge \dd x_J,
\]
और आगे बढ़ रहा है $1$-रूप $\dd b $ अतीत $ k $ के कारक $\dd
x_I $ लागत $(-1)^k $ (\cref{def:b3:forms:wedgegeneral}): द
दूसरा पद है $(-1)^k\,\omega \wedge \dd\eta $.
(बी) एक समारोह के लिए: $\dd(\dd a) = \sum_{i,j}
\frac{\partial^2a}{\partial x_j\partial x_i}\,\dd x_j \wedge
\dd x_i $. गुणांक सममित है $(i,j)$ द्वारा
मिश्रित आंशिक भाग पर श्वार्ज़ का प्रमेय (वर्ष~2 में सिद्ध हुआ)।
आयतन; $ a $ है $\mathcal C^\infty $), जबकि $\dd x_j \wedge
\dd x_i $ एंटीसिमेट्रिक है: शब्दों को जोड़ना $(i,j)$ और
$(j,i)$, सब कुछ रद्द हो जाता है। एक जनरल के लिए $\omega = \sum
a_I\dd x_I $: $\dd\dd\omega = \sum\dd(\dd a_I \wedge \dd x_I)
= \sum(\dd\dd a_I\wedge\dd x_I - \dd a_I\wedge\dd(\dd x_I))$
(ए) द्वारा, और दोनों पद गायब हो जाते हैं ($\dd(\dd x_I) = 0$ के बाद से
गुणांक \omterm{def:b3:ode:stability}{स्थिर} है)।
\end{proof}

\begin{definition}[ठहराना]\label{def:b3:forms:pullback}
होने देना $\varphi\colon U \to V $ चिकना हो ($ U \subseteq \R^m $,
$ V \subseteq \R^n $ खुला)। \emph{ठहराना}\index{ठहराना}
$\varphi^*\colon \Omega^k(V) \to \Omega^k(U)$ परिभाषित किया गया है
अंतर के साथ रैखिक पुलबैक द्वारा बिंदुवार:
$(\varphi^*\omega)_x = (D\varphi(x))^*\,\omega_{\varphi(x)}$.
ठोस रूप से, $\varphi^*$ स्थानापन्न: $\varphi^*f = f \circ
\varphi $ कार्यों पर, $\varphi^*(\dd y_i) =
\dd\varphi_i = \sum_j\frac{\partial\varphi_i}{\partial
x_j}\dd x_j $, और $\varphi^*(a\,\dd y_{i_1}\wedge\dots\wedge
\dd y_{i_k}) =
(a\circ\varphi)\,\dd\varphi_{i_1}\wedge\dots\wedge
\dd\varphi_{i_k}$.
\end{definition}

\begin{theorem}\label{thm:b3:forms:pullback}
(ए) $\varphi^*(\omega \wedge \eta) = \varphi^*\omega \wedge
\varphi^*\eta $ और $(\psi \circ \varphi)^* = \varphi^* \circ
\psi^*$.
(बी) $\varphi^*(\dd\omega) = \dd(\varphi^*\omega)$: द
\omterm{def:b3:forms:d}{बाहरी व्युत्पन्न} हर सहज प्रतिस्थापन के साथ आवागमन करता है
--- पहचान जो इसे बनाती है \emph{} का व्युत्पन्न
सिद्धांत.
(सी) यदि $\varphi\colon U \to V $ के खुलने के बीच चिकना है
$\R^n $ और $\omega = a\,\dd y_1 \wedge \dots \wedge \dd
y_n $, तब $\varphi^*\omega = (a \circ
\varphi)\,\det\bigl(D\varphi\bigr)\,\dd x_1 \wedge \dots
\wedge \dd x_n $.
\end{theorem}

\begin{proof}
(ए) रैखिक के बारे में बिंदुवार कथन पुलबैक
(\cref{prop:b3:forms:linearpullback}), साथ ही श्रृंखला नियम
$ D(\psi\circ\varphi)(x) = D\psi(\varphi(x))\,D\varphi(x)$.
(बी) के लिए $0$-रूप $ f $: $\varphi^*(\dd f) = \dd f \circ
D\varphi = \dd(f \circ \varphi)$ श्रृंखला नियम है. के लिए
$\omega = a\,\dd y_I $: (ए) का उपयोग करते हुए, $\varphi^*\omega =
(a\circ\varphi)\,\dd\varphi_{i_1}\wedge\dots\wedge
\dd\varphi_{i_k}$, तो लीबनिज़ नियम द्वारा
(\cref{thm:b3:forms:dsquare}(ए)) और $\dd\dd\varphi_{i_r} =
0$:
\[
\dd(\varphi^*\omega) = \dd(a\circ\varphi) \wedge
\dd\varphi_{i_1}\wedge\dots\wedge\dd\varphi_{i_k}
= \varphi^*(\dd a) \wedge \varphi^*(\dd y_I)
= \varphi^*(\dd a \wedge \dd y_I) =
\varphi^*(\dd\omega).
\]
(सी) बिंदुवार यह बिल्कुल सही है $ u^*\alpha = (\det u)\alpha $
शीर्\omterm{def:b3:galois:extension}{ष-डिग्री} प्रपत्रों पर (\cref{prop:b3:forms:linearpullback}(1)
के साथ $ u = D\varphi(x)$).
\end{proof}

\begin{example}[ध्रुवीय निर्देशांक]\label{ex:b3:forms:polar}
के लिए $\varphi(r, \theta) = (r\cos\theta, r\sin\theta)$:
$\varphi^*\dd x = \cos\theta\,\dd r -
r\sin\theta\,\dd\theta $, $\varphi^*\dd y = \sin\theta\,\dd r
+ r\cos\theta\,\dd\theta $, इसलिए
\[
\varphi^*(\dd x \wedge \dd y) =
(\cos\theta\,\dd r - r\sin\theta\,\dd\theta) \wedge
(\sin\theta\,\dd r + r\cos\theta\,\dd\theta)
= r\,\dd r \wedge \dd\theta,
\]
जैकोबियन का \cref{ex:b3:product:polar} शुद्ध रूप से प्रकट होना
बीजगणित --- नहीं \omterm{def:b3:measure:measure}{उपाय} लिखित। \cref{exo:b3:forms:9} बदल जाता है
एक बयान में यह टिप्पणी: उन्मुख अभिन्न के लिए,
परिवर्तन-परिवर्तन सूत्र \emph{है}  \omterm{def:b3:forms:pullback}{ठहराना} सूत्र.
\end{example}

\section{बंद और सटीक रूप; पोइंकार\'ई लेम्मा}

\begin{definition}\label{def:b3:forms:closedexact}
$\omega \in \Omega^k(U)$ है \emph{बंद किया हुआ}\index{बंद किया हुआ}
अगर $\dd\omega = 0$, \emph{एकदम सही}\index{एकदम सही} अगर
$\omega = \dd\eta $ कुछ के लिए $\eta \in \Omega^{k-1}(U)$ (ए
\emph{प्राचीन} का $\omega $). एकदम सही $\Rightarrow $ द्वारा \omterm{def:b3:topology:interior}{बंद} कर दिया गया
$\dd^2 = 0$; व्युत्क्रम के बारे में एक प्रश्न है
\emph{आकार} का $ U $.
\end{definition}

\begin{example}[कोणीय रूप]\label{ex:b3:forms:angular}
पर $ U = \R^2 \setminus \{0\}$,
\[
\omega_\theta = \frac{x\,\dd y - y\,\dd x}{x^2 + y^2}
\]
है \omterm{def:b3:forms:closedexact}{बंद किया हुआ} (प्रत्यक्ष गणना:
\cref{exo:b3:forms:4}) लेकिन नहीं \omterm{def:b3:forms:closedexact}{एकदम सही}: यह साथ में अभिन्न है
यूनिट सर्कल है $2\pi \neq 0$, जबकि अभिन्न अंग सटीक रूप
साथ में \omterm{def:b3:forms:closedexact}{बंद किया हुआ} वक्र गायब हो जाते हैं (\cref{prop:b3:forms:exactloop}).
स्थानीय स्तर पर, $\omega_\theta = \dd\theta $ किसी भी चिकनी के लिए
दृढ़ संकल्प $\theta $ ध्रुवीय कोण का --- जहां से
नाम और बाधा: ऐसा कोई निर्धारण मौजूद नहीं है
सभी $ U $. यह एकल प्रपत्र चलता है \omterm{def:b3:forms:winding}{घुमावदार संख्या}
(\cref{sec:b3:forms:winding}) और, इसके माध्यम से, \omterm{def:b3:residues:singularities}{अवशेष}
का प्रमेय \cref{ch:b3:residues}.
\end{example}

\begin{theorem}[पोइंकार\'ई लेम्मा]\label{thm:b3:forms:poincare}
होने देना $ U \subseteq \R^n $ खुले रहें और \emph{तारे के आकार का} इसके संबंध में $0$. प्रत्येक \omterm{def:b3:forms:closedexact}{बंद किया हुआ} $ k $-पर प्रपत्र $ U $ ($ k \geq 1$) है
\omterm{def:b3:forms:closedexact}{एकदम सही}.\index{Poincaré lemma}
\end{theorem}

\begin{proof}
हम एक रैखिक निर्माण करते हैं \emph{होमोटॉपी ऑपरेटर} $ h\colon
\Omega^k(U) \to \Omega^{k-1}(U)$ साथ
\begin{equation}\label{eq:b3:forms:homotopy}
\dd(h\omega) + h(\dd\omega) = \omega
\qquad (k \geq 1);
\end{equation}
अगर $\dd\omega = 0$, तब $\omega = \dd(h\omega)$ और हम हैं
किया. के लिए $\omega = \sum_I a_I\,\dd x_I $ तय करना
\[
h\omega = \sum_{\abs I = k}\ \sum_{r=1}^{k}(-1)^{r-1}
\Bigl(\int_0^1 t^{k-1}a_I(tx)\,\dd t\Bigr)\,
x_{i_r}\,\dd x_{i_1}\wedge\dots\wedge
\widehat{\dd x_{i_r}}\wedge\dots\wedge\dd x_{i_k}
\]
(हैट एक कारक को हटा देता है; इंटीग्रल सुचारू रूप से चलते हैं $ x $
अभिन्न के अंतर्गत विभेदन द्वारा,
\cref{thm:b3:lebesgue:paramdiff}, सभी व्युत्पन्न हैं
पर हावी हो गया कॉम्पैक्ट कैमरों). चेकिंग
\eqref{eq:b3:forms:homotopy} एक गणना एक बार की जाती है
जीवनकाल, इसलिए हम इसे पूर्ण रूप से करते हैं। हल करना $ I $ और ले लो $\omega =
a\,\dd x_I $ (रैखिकता). पहला,
\[
\dd(h\omega) = k\Bigl(\int_0^1t^{k-1}a(tx)\dd t\Bigr)\dd x_I
+ \sum_{r=1}^k(-1)^{r-1}\sum_{j=1}^n
\Bigl(\int_0^1t^{k}\,\partial_ja(tx)\,\dd t\Bigr)
x_{i_r}\,\dd x_j\wedge\dd x_{I\setminus i_r} :
\]
पहला समूह उन शर्तों को एकत्रित करता है जिनमें $\dd $ हिट करता है
कारक $ x_{i_r}$ --- कील $\dd x_{i_r}\wedge\dd
x_{I\setminus i_r}$ पुनः संयोजन करता है $\dd x_I $ एक संकेत के साथ
$(-1)^{r-1}$ जो प्रीफैक्टर को रद्द कर देता है, और $ k $ मूल्य
का $ r $ कारक दीजिए $ k $ --- जबकि दूसरा समूह
जहां शर्तों को एकत्रित करता है $\dd $ इंटीग्रल (श्रृंखला) को हिट करता है
नियम सामने लाता है $ t\,\partial_ja(tx)$). अगला, $\dd\omega =
\sum_j\partial_ja\,\dd x_j\wedge\dd x_I $, और लागू करना
की परिभाषा $ h $ \omterm{def:b3:galois:extension}{डिग्री} में $ k+1$, सूचकांक $ j $ कब्ज़ा
पहला स्लॉट:
\[
h(\dd\omega) = \sum_{j=1}^n\Bigl(\int_0^1t^{k}
\partial_ja(tx)\dd t\Bigr)x_j\,\dd x_I
- \sum_{j=1}^n\sum_{r=1}^k(-1)^{r-1}
\Bigl(\int_0^1t^{k}\partial_ja(tx)\dd t\Bigr)
x_{i_r}\,\dd x_j\wedge\dd x_{I\setminus i_r} .
\]
\omterm{def:b3:banach:operator}{दोहरी} रकम रद्द हो जाती है $\dd(h\omega) + h(\dd\omega)$,
जो इसलिए बराबर है
\[
\Bigl(\int_0^1\bigl(k\,t^{k-1}a(tx) +
t^k{\textstyle\sum_j}x_j\,\partial_ja(tx)\bigr)\dd t\Bigr)
\dd x_I
= \Bigl(\int_0^1\frac{\dd}{\dd t}\bigl(t^ka(tx)\bigr)\dd
t\Bigr)\dd x_I
= a(x)\,\dd x_I = \omega,
\]
कैलकुलस के मौलिक प्रमेय द्वारा. तारे के आकार का होना
जहां उसे जाना था वहां प्रवेश किया: $ tx \in U $ के लिए $ t \in \intcc01$,
ताकि $ a(tx)$ समझ में आता है।
\end{proof}

\begin{remark}
के लिए $ k = 1$ और $\omega = \sum_j a_j\dd x_j $, आदिम
है $ f(x) = \int_0^1\sum_ja_j(tx)\,x_j\,\dd t $ --- लाइन
का अभिन्न अंग $\omega $ खंड के साथ $[0, x]$: प्रमेय
कई-चर ``सममित जैकोबियन वाला एक क्षेत्र है
वर्ष~2 का एक ग्रेडिएंट है, अब हर \omterm{def:b3:galois:extension}{डिग्री} में। कोणीय
फॉर्म (\cref{ex:b3:forms:angular}) पर परिकल्पना दर्शाता है
$ U $ सजावटी नहीं है: $\R^2\setminus\{0\}$ नहीं है
तारे के आकार का, और वहां \omterm{def:b3:topology:interior}{बंद} होने का मतलब सटीकता नहीं है।
\omterm{def:b3:groups:normal}{सामान्य} पर क्या टिकता है \omterm{def:b3:topology:topology}{खुला सेट} द्वारा मापा जाता है
\emph{डी राम कोहोमोलॉजी} $ H^k(U) =
\ker\dd/\operatorname{im}\dd $ --- देखना
\cref{exo:b3:forms:12} पहली गैरतुच्छ गणना के लिए.
\end{remark}

\section{सबमेनफोल्ड्स पर अभिविन्यास और एकीकरण}

एकीकृत करना ए $ k $-फॉर्म की आवश्यकता है $ k $-आयामी उन्मुख
क्षेत्र. से स्मरण करें \cref{ch:b3:submanifolds}
(\cref{thm:b3:submanifolds:characterizations}) कि एक
$ k $-सबमैनिफोल्ड $ M \subseteq \R^n $ स्थानीय रूप से एक की छवि है
नियमित पैरामीट्रिजेशन $\gamma\colon V \to M \cap W $ ($ V
\subseteq \R^k $ खुला, $\gamma $ ए \omterm{def:b3:topology:continuity}{होमियोमोर्फिज्म} इसकी छवि पर
इंजेक्शन अंतर के साथ)।

\begin{definition}\label{def:b3:forms:orientation}
एक \emph{अभिविन्यास}\index{अभिविन्यास} का $ M $ एक विकल्प है,
प्रत्येक के लिए $ p \in M $, के दो अभिविन्यास वर्गों में से एक
के \omterm{def:b3:topology:basis}{आधार} \omterm{def:b3:submanifolds:tangent}{स्पर्शरेखा स्थान} $ T_pM $, जो है \emph{स्थानीय स्तर पर
सुसंगत}: प्रत्येक बिंदु के चारों ओर एक पैरामीट्रिजेशन है
$\gamma $ जिसका समन्वय फ्रेम $(\partial_1\gamma, \dots,
\partial_k\gamma)$ के हर बिंदु पर सकारात्मक रूप से उन्मुख है
इसका डोमेन. ऐसे पैरामीट्रिजेशन कहलाते हैं \emph{प्रत्यक्ष}.
$ M $ है \emph{उन्मुख} यदि कोई अभिविन्यास मौजूद है; उन्हें\"ओबियस बैंड दिखाता है कि यह विफल हो सकता है। इसमें सभी उपमान हैं
अध्याय उन्मुख हैं.
\end{definition}

\begin{definition}[एक रूप का अभिन्न अंग]\label{def:b3:forms:integral}
होने देना $ M $ एक उन्मुख हो $ k $-सबमैनिफोल्ड और $\omega $ ए
$ k $-फॉर्म ए पर परिभाषित किया गया है \omterm{def:b3:topology:topology}{अड़ोस-पड़ोस} का $ M $, साथ
$\operatorname{supp}\omega \cap M $ कॉम्पैक्ट.
(ए) यदि $\operatorname{supp}\omega \cap M \subseteq
\gamma(V)$ एकल प्रत्यक्ष पैरामीट्रिजेशन के लिए, सेट करें
\[
\int_M\omega = \int_V\gamma^*\omega
\]
--- दाहिनी ओर लेब्सगेग इंटीग्रल ओवर है $ V $
(\cref{ch:b3:product}) गुणांक का $ g $ का
$\gamma^*\omega = g\,\dd u_1\wedge\dots\wedge\dd u_k $, जो
है \omterm{def:b3:topology:continuity}{निरंतर} साथ कॉम्पैक्ट सहायता।
(बी) \omterm{def:b3:groups:normal}{सामान्य} तौर पर, बहुत से प्रत्यक्ष पैरामीट्रिज़ेशन चुनें
$\gamma_i(V_i)$ को कवर करना कॉम्पैक्ट
$\operatorname{supp}\omega \cap M $ और एक अधीनस्थ
\omterm{lem:b3:forms:partition}{एकता का विभाजन} $(\chi_i)$
(\cref{lem:b3:forms:partition}), और सेट करें $\int_M\omega =
\sum_i\int_M\chi_i\,\omega $, प्रत्येक पद की गणना (ए) द्वारा की जाती है।
एक वक्र के लिए ($ k = 1$) द्वारा पैरामीट्रिज्ड
$\gamma\colon\intcc ab\to\R^n $ हम लिखते हैं
$\int_\gamma\omega = \int_a^b\gamma^*\omega $, कोई इंजेक्शन नहीं
आवश्यक.
\end{definition}

\begin{lemma}[स्थिरता]\label{lem:b3:forms:consistency}
परिभाषा (ए) प्रत्यक्ष पर निर्भर नहीं करती
पैरामीट्रिजेशन, और परिभाषा (बी) किसी पर भी निर्भर नहीं करती है
न ही कवर करें और न ही पर \omterm{lem:b3:forms:partition}{एकता का विभाजन}. इसके अलावा, यदि
$\Phi $ की एक भिन्नता है पड़ोस दो उन्मुख का
सबमैनिफोल्ड्स के साथ $\Phi(M) = M'$, का सीधा फ्रेम ले जाना
$ M $ के एक सीधे फ्रेम के लिए $ M'$ प्रत्येक के किसी बिंदु पर
का घटक $ M $, तब $\int_{M'}\omega =
\int_M\Phi^*\omega $.
\end{lemma}

\begin{proof}
(ए) चलो $\gamma\colon V \to M $, $\delta\colon V' \to M $ हो
प्रत्यक्ष पैरामीट्रिज़ेशन जिनकी छवियां शामिल हैं
$\operatorname{supp}\omega\cap M $. संक्रमण $\tau =
\delta^{-1}\circ\gamma $ के बीच एक भिन्नता है
के प्रासंगिक खुले उपसमुच्चय $ V, V'$ (संक्रमण की सहजता:
\cref{thm:b3:submanifolds:characterizations}, स्थानीय के माध्यम से
ग्राफ़ विवरण), और $\gamma = \delta\circ\tau $ इसलिए
$\gamma^*\omega = \tau^*(\delta^*\omega)$
(\cref{thm:b3:forms:pullback}(ए))। लिखना $\delta^*\omega =
g\,\dd u_1\wedge\dots\wedge\dd u_k $; तब
(\cref{thm:b3:forms:pullback}(सी)) $\tau^*(\delta^*\omega) =
(g\circ\tau)\,\det(D\tau)\,\dd u_1\wedge\dots\wedge\dd u_k $.
दोनों फ्रेम सीधे हैं, $ D\tau $ के लिए एक सकारात्मक \omterm{def:b3:topology:basis}{आधार} प्रस्तुत करता है
सकारात्मक \omterm{def:b3:topology:basis}{आधार}: $\det D\tau > 0$, इसलिए $\det D\tau =
\abs{\det D\tau}$ और चर-परिवर्तन प्रमेय
(\cref{thm:b3:product:changeofvar}) देता है
$\int(g\circ\tau)\abs{\det D\tau} = \int g $: दो
अभिन्न सहमत हैं. \emph{ये है पूरी वजह \omterm{def:b3:forms:orientation}{अभिविन्यास}
मौजूद है}: संकेत नियंत्रण के बिना, जैकोबियन और उसके
निरपेक्ष मान भिन्न होता है और अभिन्न अंग अपरिभाषित होता है।
(बी) यदि $(\chi_i)$ और $(\tilde\chi_j)$ दो स्वीकार्य हैं
विभाजन (कवर शामिल), फिर (ए) और परिमित द्वारा
संवेदनशीलता, $\sum_i\int\chi_i\omega =
\sum_{i,j}\int\chi_i\tilde\chi_j\omega =
\sum_j\int\tilde\chi_j\omega $, प्रत्येक दोहरा पद गणना योग्य है
किसी भी चार्ट में. अंतिम कथन: यदि $\gamma $ भाग जाता है
का प्रत्यक्ष पैरामीट्रिजेशन $ M $, तब $\Phi\circ\gamma $ चलता है
के प्रत्यक्ष पैरामीट्रिज़ेशन पर $ M'$ (द \omterm{def:b3:forms:orientation}{अभिविन्यास}
तुलना स्थानीय रूप से \omterm{def:b3:ode:stability}{स्थिर} है, और एक बिंदु पर निश्चित है
घटक), और $(\Phi\circ\gamma)^*\omega =
\gamma^*(\Phi^*\omega)$.
\end{proof}

\begin{lemma}[एकता का विभाजन, सघन मामला]
\label{lem:b3:forms:partition}
होने देना $ K \subseteq \R^n $ होना कॉम्पैक्ट और $ W_1, \dots, W_m $ \omterm{def:b3:topology:topology}{खुला
सेट} कवर $ K $. वहां है $\chi_1, \dots, \chi_m \in
\mathcal C^\infty(\R^n)$ साथ $0 \leq \chi_i \leq 1$,
$\operatorname{supp}\chi_i \subseteq W_i $ कॉम्पैक्ट, और
$\sum\chi_i = 1$ एक पर \omterm{def:b3:topology:topology}{अड़ोस-पड़ोस} का
$ K $.\index{एकता का विभाजन}
\end{lemma}

\begin{proof}
प्रत्येक $ x \in K $ कुछ में झूठ है $ W_{i(x)}$ के साथ \omterm{def:b3:forms:closedexact}{बंद किया हुआ} गेंद
$\bar B(x, 2r_x) \subseteq W_{i(x)}$; सघनता अर्क
$ x_1, \dots, x_N $ गेंदों के साथ $ B(x_s, r_{x_s})$ कवर
$ K $. प्रत्येक के लिए $ s $ एक टक्कर ले लो $\theta_s \in \mathcal
C^\infty $, $0 \leq \theta_s \leq 1$, $\theta_s = 1$ पर
$\bar B(x_s, r_{x_s})$, $\operatorname{supp}\theta_s
\subseteq B(x_s, 2r_{x_s})$ (के सूचक को शांत करें
त्रिज्या की गेंद $\frac32r_{x_s}$,
\cref{thm:b3:lp:regularization}). प्रत्येक को असाइन करें $ s $ एक को
सूचकांक $ i(s)$ साथ $ B(x_s, 2r_{x_s}) \subseteq W_{i(s)}$ और सेट करें $\Theta_i = \sum_{i(s) = i}\theta_s $. पर \omterm{def:b3:topology:topology}{खुला सेट}
$\Omega_0 = \{\sum_j\Theta_j > \tfrac12\} \supseteq K $, द
कार्य $\Theta_i/\sum_j\Theta_j $ नौकरी करो लेकिन हैं
केवल वहीं परिभाषित; वैश्वीकरण करने के लिए, चलो $\rho \in \mathcal
C^\infty(\R^n)$ संतुष्ट $\rho = 0$ कहाँ $\sum_j\Theta_j
\geq 1$ और $\rho > 0$ कहाँ $\sum_j\Theta_j \leq \tfrac12$
(एक उपयुक्त कट-ऑफ को कम करें $1 - \sum_j\Theta_j $), और सेट करें
\[
\chi_i = \frac{\Theta_i}{\rho + \sum_j\Theta_j} .
\]
विभाजक हर जगह है $> 0$ और बराबर है
$\sum_j\Theta_j $ पर $\{\sum_j\Theta_j \geq 1\}$, एक खुला
\omterm{def:b3:topology:topology}{अड़ोस-पड़ोस} का $ K $ (प्रत्येक बिंदु $ K $ किसी गेंद में पड़ा है
$ B(x_s, r_{x_s})$ कहाँ $\theta_s = 1$); वहाँ
$\sum_i\chi_i = 1$. समर्थन और सीमाएँ स्पष्ट हैं।
\end{proof}

\begin{definition}\label{def:b3:forms:boundary}
ए \emph{$ k $- सीमा के साथ सबमैनिफोल्ड}\index{$ k $- सीमा के साथ सबमैनिफोल्ड} $ M \subseteq \R^n $ नियमित द्वारा कवर किया गया एक सेट है
दो प्रकार के पैरामीट्रिजेशन: \emph{आंतरिक भाग चार्ट}
$\gamma\colon V \to M \cap W $ साथ $ V \subseteq \R^k $ खुला,
और \emph{सीमा चार्ट} $\gamma\colon V \cap H^k \to M
\cap W $, कहाँ $ H^k = \{u \in \R^k : u_k \geq 0\}$ और
$\gamma $ खुले में सुचारू रूप से और नियमित रूप से फैलता है $ V $.
 \emph{सीमा} $\partial M $ बिंदुओं का समूह है
पर पहुंच गया $ u_k = 0$; यह एक है $(k-1)$-सबमैनिफोल्ड बिना
सीमा, मानचित्रों द्वारा पैरामीट्रिज्ड $ u' \mapsto \gamma(u',
0)$. एक \omterm{def:b3:forms:orientation}{अभिविन्यास} का $ M $ किसी को प्रेरित करता है $\partial M $ से \emph{बाह्य-सामान्य-प्रथम} नियम: पर $ p \in \partial M $,
एक \omterm{def:b3:topology:basis}{आधार} $(w_1, \dots, w_{k-1})$ का $ T_p\partial M $ है
सकारात्मक यदि $(\nu, w_1, \dots, w_{k-1})$ एक सकारात्मक है
का \omterm{def:b3:topology:basis}{आधार} $ T_pM $, कहाँ $\nu \in T_pM \setminus T_p\partial M $
की ओर इंगित करता है $ M $ (एक सीमा चार्ट में: $\nu =
-\partial_k\gamma $, स्पर्शरेखीय घटकों को जोड़ने तक ---
द \omterm{def:b3:forms:orientation}{अभिविन्यास} \omterm{def:b3:groups:action}{कक्षा} उन्हें नहीं देखती)।
\end{definition}

\begin{lemma}[हाफ स्पेस पर स्टोक्स]
\label{lem:b3:forms:halfspace}
होने देना $\eta $ एक चिकनी हो $(k-1)$-पर प्रपत्र $\R^k $ साथ कॉम्पैक्ट
सहायता। तब
\[
\int_{H^k}\dd\eta = \int_{\partial H^k}\eta,
\]
कहाँ $ H^k $ मानक वहन करता है \omterm{def:b3:forms:orientation}{अभिविन्यास} का $\R^k $ और
$\partial H^k = \{u_k = 0\} \cong \R^{k-1}$ प्रेरित एक,
जो है $(-1)^k $ मानक से गुना \omterm{def:b3:forms:orientation}{अभिविन्यास} का
$\R^{k-1}$.
\end{lemma}

\begin{proof}
सबसे पहले \omterm{def:b3:forms:orientation}{अभिविन्यास} बहीखाता: बाहरी \omterm{def:b3:groups:normal}{सामान्य} साथ
$\partial H^k $ है $-e_k $, और
\[
\det(-e_k, e_1, \dots, e_{k-1}) =
-\det(e_k, e_1, \dots, e_{k-1}) = -(-1)^{k-1}\det(e_1,
\dots, e_k) = (-1)^k :
\]
फ़्रेम $(e_1, \dots, e_{k-1})$ का $\partial H^k $ है
प्रेरित के लिए सकारात्मक \omterm{def:b3:forms:orientation}{अभिविन्यास} बिल्कुल कब $ k $ है
यहां तक कि, बताई गई तुलना कहां से आई है। रैखिकता से ले लो $\eta
= f\,\dd u_1\wedge\dots\wedge\widehat{\dd
u_i}\wedge\dots\wedge\dd u_k $, $ f \in \mathcal
C^\infty_c(\R^k)$; तब $\dd\eta =
(-1)^{i-1}\,\partial_if\,\dd u_1\wedge\dots\wedge\dd u_k $
(चल रहा है $\dd u_i $ इसकी स्लॉट लागत में $ i - 1$ अदला-बदली)। दो
मामले, दोनों टोनेली-फ़ुबिनी द्वारा
(\cref{thm:b3:product:fubini}) और एक-चर
कैलकुलस का मौलिक प्रमेय.

\emph{मामला $ i < k $.} सबसे पहले एकीकरण करना $ u_i $ ऊपर $\R $:
$\int_\R\partial_if\,\dd u_i = 0$ (कॉम्पैक्ट समर्थन), तो
$\int_{H^k}\dd\eta = 0$. और का प्रतिबंध $\eta $ को
$\{u_k = 0\}$ कारक शामिल है $\dd u_k $, जो प्रतिबंधित करता है
को $0$ ($ u_k $ वहां \omterm{def:b3:ode:stability}{स्थिर} है): $\int_{\partial H^k}\eta =
0$ बहुत।

\emph{मामला $ i = k $.} सबसे पहले एकीकरण करना $ u_k $ ऊपर
$\intco0\infty $:
\begin{align*}
\int_{H^k}\dd\eta &= (-1)^{k-1}\int_{\R^{k-1}}
\Bigl(\int_0^\infty\partial_kf\,\dd u_k\Bigr)\dd u' \\
&= (-1)^{k-1}\int_{\R^{k-1}}\bigl(0 - f(u', 0)\bigr)\dd u'
= (-1)^k\int_{\R^{k-1}}f(u',0)\,\dd u' .
\end{align*}
पर सीमा ओर, $\eta $ तक सीमित है $ f(u',
0)\,\dd u_1\wedge\dots\wedge\dd u_{k-1}$, और प्रेरित
\omterm{def:b3:forms:orientation}{अभिविन्यास} प्राणी $(-1)^k $ मानक एक से गुना,
$\int_{\partial H^k}\eta =
(-1)^k\int_{\R^{k-1}}f(u',0)\,\dd u'$. दोनों पक्ष सहमत हैं.
\end{proof}

\begin{theorem}[स्टोक्स]\label{thm:b3:forms:stokes}
होने देना $ M \subseteq \R^n $ एक हो कॉम्पैक्ट उन्मुखी $ k $-सबमैनिफोल्ड
के साथ सीमा, $\partial M $ प्रेरित ले जाना
\omterm{def:b3:forms:orientation}{अभिविन्यास}, और चलो $\omega $ एक चिकनी हो $(k-1)$-ए पर फार्म
\omterm{def:b3:topology:topology}{अड़ोस-पड़ोस} का $ M $. तब\index{स्टोक्स प्रमेय}
\[
\int_M\dd\omega = \int_{\partial M}\omega .
\]
विशेषकर, यदि $\partial M = \varnothing $:
$\int_M\dd\omega = 0$.
\end{theorem}

\begin{proof}
आवरण कॉम्पैक्ट $ M $ प्रत्यक्ष की बहुत सारी छवियों द्वारा
चार्ट (आंतरिक भाग या सीमा), और एक ले लो \omterm{lem:b3:forms:partition}{एकता का विभाजन}
$(\chi_i)$ संबंधित के अधीन खुले सेट $ W_i $
का $\R^n $ (\cref{lem:b3:forms:partition}), साथ $\sum\chi_i
= 1$ एक पर \omterm{def:b3:topology:topology}{अड़ोस-पड़ोस} का $ M $. उस पर \omterm{def:b3:topology:topology}{अड़ोस-पड़ोस}
$\dd(\sum\chi_i) = 0$, इसलिए
\[
\int_M\dd\omega = \sum_i\int_M\dd(\chi_i\omega),
\qquad
\int_{\partial M}\omega =
\sum_i\int_{\partial M}\chi_i\omega :
\]
दोनों पक्ष योगात्मक हैं, और यह साबित करने के लिए पर्याप्त है
एकल चार्ट छवि में समर्थित प्रपत्र के लिए प्रमेय।

\emph{आंतरिक भाग चार्ट।} अगर $\operatorname{supp}\omega \cap M
\subseteq \gamma(V)$ साथ $ V $ में खोलें $\R^k $: बढ़ाना $\eta =
\gamma^*\omega $ शून्य से $\R^k $ (चिकना, कॉम्पैक्ट समर्थन करें $ V $) और लागू करें \cref{lem:b3:forms:halfspace} के साथ
समर्थन से दूर $\partial H^k $ (अनुवाद $ V $ में
ऊपरी आधा स्थान खोलें --- या बस मामले को दोहराएं $ i < k $
सब पर गणना $\R^k $): $\int_M\dd\omega =
\int_{\R^k}\dd\eta = 0$, और $\int_{\partial M}\omega = 0$
तब से $\omega $ के पास गायब हो जाता है $\partial M $.

\emph{\omterm{def:b3:forms:boundary}{सीमा चार्ट}।} अगर $\operatorname{supp}\omega\cap M
\subseteq \gamma(V \cap H^k)$: साथ $\eta = \gamma^*\omega $
शून्य से बढ़ाया गया,
$\gamma^*(\dd\omega) = \dd\eta $
(\cref{thm:b3:forms:pullback}(बी)), तो द्वारा
\cref{def:b3:forms:integral}:
\[
\int_M\dd\omega = \int_{H^k}\dd\eta
\overset{\text{\cref{lem:b3:forms:halfspace}}}{=}
\int_{\partial H^k}\eta .
\]
अभी दाहिनी ओर की पहचान करना बाकी है $\int_{\partial
M}\omega $. सीमा द्वारा पैरामीट्रिज्ड किया गया है $\beta(u') =
\gamma(u', 0)$, और $\beta^*\omega $ का प्रतिबंध है
$\eta $ को $\{u_k = 0\}$ (\omterm{def:b3:forms:pullback}{ठहराना} समावेशन के अंतर्गत $ u'
\mapsto (u', 0)$ के साथ रचित $\gamma $). \omterm{def:b3:forms:orientation}{अभिविन्यास}
तुलना समान है $(-1)^k $ दोनों तरफ: फ्रेम
$(\partial_1\beta, \dots, \partial_{k-1}\beta)$ में बैठता है
प्रेरित \omterm{def:b3:forms:orientation}{अभिविन्यास} का $\partial M $ संकेत के साथ
$\det(-e_k, e_1, \dots, e_{k-1}) = (-1)^k $ में गणना की गई
चार्ट (बाहरी वेक्टर वापस खींचता है $-e_k $), जो है
बिल्कुल संबंधित संकेत $\partial H^k $ प्रेरित किया है
\omterm{def:b3:forms:orientation}{अभिविन्यास} मानक के अनुसार $\R^{k-1}$
(\cref{lem:b3:forms:halfspace}). दो हस्ताक्षर सम्मेलन
रद्द करें:
$\int_{\partial H^k}\eta = \int_{\partial M}\omega $.
\end{proof}

\begin{example}[शास्त्रीय प्रमेय]
\label{ex:b3:forms:classical}
होने देना $ D \subseteq \R^2$ एक हो कॉम्पैक्ट चिकनी के साथ डोमेन
सीमा, मानक रूप से उन्मुख। के लिए $\omega = P\,\dd x +
Q\,\dd y $: $\dd\omega = \bigl(\partial_xQ -
\partial_yP\bigr)\dd x\wedge\dd y $, और स्टोक्स पढ़ता है
\[
\int_D\Bigl(\frac{\partial Q}{\partial x} -
\frac{\partial P}{\partial y}\Bigr)\dd x\,\dd y
= \oint_{\partial D}P\,\dd x + Q\,\dd y :
\]
\emph{हरा--रीमैन}, में प्राथमिक डोमेन के लिए सिद्ध हुआ
वर्ष~2 मात्रा और अब प्राकृतिक व्यापकता में। में $\R^3$,
स्टोक्स ने फ्लक्स पर आवेदन किया $2$-ए पर एक वेक्टर फ़ील्ड का रूप
कॉम्पैक्ट डोमेन देता है \emph{विचलन प्रमेय}
$\int_\Omega\operatorname{div}F =
\int_{\partial\Omega}\langle F, \nu\rangle\,\dd S $, और
ए पर लागू किया गया $1$-एक सतह पर-सीमा के साथ फार्म,
शास्त्रीय \emph{केल्विन-स्टोक्स} कर्ल प्रमेय;
\cref{exo:b3:forms:7} दोनों शब्दकोशों का उच्चारण करता है।
\end{example}

\section{घुमावदार संख्या}\label{sec:b3:forms:winding}

\begin{definition}\label{def:b3:forms:winding}
होने देना $\gamma\colon\intcc01\to\R^2\setminus\{a\}$ एक चिकनी हो
\omterm{def:b3:forms:closedexact}{बंद किया हुआ} वक्र. इसका \emph{घुमावदार संख्या}\index{घुमावदार संख्या} आस-पास $ a $ है
\[
\operatorname{Ind}_\gamma(a) =
\frac1{2\pi}\int_\gamma\omega_\theta^a,
\qquad
\omega_\theta^a = \frac{(x - a_1)\,\dd y - (y - a_2)\,\dd
x}{(x - a_1)^2 + (y - a_2)^2} .
\]
\end{definition}

\begin{proposition}\label{prop:b3:forms:windinginteger}
$\operatorname{Ind}_\gamma(a) \in \Z $; के एक समारोह के रूप में $ a $
यह प्रत्येक पर \omterm{def:b3:ode:stability}{स्थिर} है \omterm{def:b3:topology:components}{जुड़ा हुआ घटक} का
$\R^2\setminus\gamma(\intcc01)$ और असीमित पर शून्य
घटक. के लिए $\gamma(t) = a + r(\cos2\pi t, \sin2\pi t)$:
$\operatorname{Ind}_\gamma(a) = 1$.
\end{proposition}

\begin{proof}
लेना $ a = 0$ और लिखा $\gamma = (x, y)$, $\rho =
\norm\gamma > 0$, $\vartheta(t) =
\int_0^t\gamma^*\omega_\theta $, ताकि $\vartheta' =
\frac{xy' - yx'}{x^2 + y^2}$. जटिल संकेतन में चलो $ u(t)
= \gamma(t)\,\rho(t)^{-1}\eu^{-\iu\vartheta(t)}$; तब
$\abs u = 1$ और एक सीधी गणना देती है
\[
\frac{u'}{u} = \frac{\gamma'}{\gamma} - \frac{\rho'}{\rho}
- \iu\vartheta'
= \frac{\bar\gamma\gamma'}{\abs\gamma^2} -
\frac{\rho'}{\rho} - \iu\,\frac{xy' -
yx'}{\rho^2}
= \frac{(xx' + yy')}{\rho^2} - \frac{\rho'}{\rho} = 0,
\]
तब से $\bar\gamma\gamma' = (xx' + yy') + \iu(xy' - yx')$ और
$\rho\rho' = xx' + yy'$. इसलिए $ u $ \omterm{def:b3:ode:stability}{स्थिर} है: $\gamma(t) =
\rho(t)\,c\,\eu^{\iu\vartheta(t)}$ साथ $\abs c = 1$, और
$\gamma(1) = \gamma(0)$ साथ $\rho(1) = \rho(0)$ बल
$\eu^{\iu\vartheta(1)} = \eu^{\iu\vartheta(0)} = 1$:
$\vartheta(1) \in 2\pi\Z $, अर्थात\
$\operatorname{Ind}_\gamma(0) \in \Z $. के एक समारोह के रूप में $ a $
के खुले पूरक पर कॉम्पैक्ट वक्र, परिभाषित
अभिन्न है \omterm{def:b3:topology:continuity}{निरंतर} (\cref{thm:b3:lebesgue:paramcont},
एक पर प्रभुत्व \omterm{def:b3:topology:topology}{अड़ोस-पड़ोस} प्रत्येक का $ a $); ए \omterm{def:b3:topology:continuity}{निरंतर}
पूर्णांक-मूल्यवान फ़ंक्शन स्थानीय रूप से \omterm{def:b3:ode:stability}{स्थिर} है, इसलिए \omterm{def:b3:ode:stability}{स्थिर} है
घटकों पर. के लिए $\norm a $ बड़ा एकीकरण है
$ O(1/\norm a)$ समान रूप से में $ t $, इसलिए सूचकांक की प्रवृत्ति होती है $0$
और असीमित घटक पर लुप्त हो जाता है। वृत्त के लिए:
$\gamma^*\omega_\theta^a = 2\pi\,\dd t $ सीधे.
\end{proof}

\begin{remark}
अंतर्गत $\C \cong \R^2$, $\frac{\dd z}{z} = \frac{x\,\dd x +
y\,\dd y}{x^2 + y^2} + \iu\,\omega_\theta $, इसलिए
$\operatorname{Ind}_\gamma(a) =
\frac1{2\iu\pi}\oint_\gamma\frac{\dd z}{z - a}$: यह है
का सूचकांक \cref{ch:b3:residues}, और
\cref{prop:b3:forms:windinginteger} इसे पुनः सिद्ध करता है
वास्तविक-परिवर्तनीय साधनों द्वारा अभिन्नता और स्थानीय स्थिरता ---
द संस्थानिक का आधा \omterm{def:b3:residues:singularities}{अवशेष} प्रमेय, अब पर खड़ा है
स्टोक्स.
\end{remark}

\begin{proposition}\label{prop:b3:forms:exactloop}
अगर $\omega = \dd f $ है \omterm{def:b3:forms:closedexact}{एकदम सही} खुले में $ U $ और
$\gamma\colon\intcc01\to U $ एक है \omterm{def:b3:forms:closedexact}{बंद किया हुआ} वक्र, तो
$\int_\gamma\omega = 0$.
\end{proposition}

\begin{proof}
$\int_\gamma\dd f = \int_0^1(f\circ\gamma)'(t)\,\dd t =
f(\gamma(1)) - f(\gamma(0)) = 0$ ---श्रृंखला नियम
पहचानता है $\gamma^*(\dd f)$ साथ $(f\circ\gamma)'\,\dd t $.
\end{proof}

\begin{method}\label{met:b3:forms:method}
प्रपत्रों के साथ कंप्यूटिंग: (1) मशीनीकरण --- वेजेज को फिर से व्यवस्थित करें
संकेत, $\dd $ गुणांकों को ताज़ा में विभेदित करता है $\dd
x_j $'एस, पुलबैक विकल्प; बीजगणित पर भरोसा करें, यह एन्कोड करता है
हर जैकोबियन. (2) किसी फॉर्म को सबमैनिफोल्ड पर एकीकृत करने के लिए:
सीधे पैरामीट्रिज़ करें, पीछे खींचें, गुणांक को एकीकृत करें;
\omterm{def:b3:forms:orientation}{अभिविन्यास} एकमात्र जाल है --- एक फ्रेम की जाँच करें। (3)को
एक अभिन्न पहचान साबित करें, स्टोक्स आकार की तलाश करें: है
एकीकृत \omterm{def:b3:forms:closedexact}{एकदम सही}? डोमेन ए है सीमा? (4) तुलना करना
दो ``समानांतर'' उपमानों पर इंटीग्रल, स्टोक्स लागू करें
उनके बीच के क्षेत्र में (विरूपण तर्क,
\cref{exo:b3:forms:10}). (5) ए का एक शून्येतर अभिन्न अंग \omterm{def:b3:topology:interior}{बंद}
प्रपत्र प्रमाणित करता है ए संस्थानिक रुकावट --- कोई आदिम नहीं,
कोई प्रत्यावर्तन नहीं, कोई शून्\omterm{def:b3:modules:free}{य-मुक्त} विस्तार नहीं: इस प्रकार
सप्ताहांत की समस्या गेंद को पीछे हटने से रोकती है।
\end{method}

\section{अभ्यास}

\begin{exercise}[$\star $]\label{exo:b3:forms:1}
पर $\R^3$, होने देना $\omega = x\,\dd y - z\,\dd x $ और $\eta =
\dd x + y\,\dd z $. गणना करना $\omega\wedge\eta $, $\dd\omega $,
$\dd\eta $, और $\dd(\omega\wedge\eta)$, और सत्यापित करें
इस उदाहरण पर वर्गीकृत लीबनिज़ नियम।
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b3:forms:2}
पहचानें, पर $\R^3$, के तीन अवतार $\dd $: के लिए
$ f \in \Omega^0$, $\dd f \leftrightarrow \nabla f $; के लिए
कार्य रूप $\omega_F = F_1\dd x + F_2\dd y + F_3\dd z $,
$\dd\omega_F \leftrightarrow \operatorname{curl}F $; के लिए
\omterm{ex:b3:ode:planeclassification}{प्रवाह} रूप $\sigma_F = F_1\,\dd y\wedge\dd z + F_2\,\dd
z\wedge\dd x + F_3\,\dd x\wedge\dd y $, $\dd\sigma_F
\leftrightarrow \operatorname{div}F $. से निष्कर्ष निकालें $\dd^2 =
0$ पहचान $\operatorname{curl}\nabla f = 0$ और
$\operatorname{div}\operatorname{curl}F = 0$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:forms:3}
तय करें कि क्या प्रत्येक $1$-रूप है \omterm{def:b3:forms:closedexact}{बंद किया हुआ}, \omterm{def:b3:forms:closedexact}{एकदम सही} इसके डोमेन पर,
और जब कोई मौजूद हो तो एक आदिम की गणना करें:
(ए) $(2xy + z^2)\,\dd x + x^2\,\dd y + 2xz\,\dd z $ पर
$\R^3$;
(बी) $\dfrac{x\,\dd x + y\,\dd y}{x^2 + y^2}$ पर
$\R^2\setminus\{0\}$;
(सी) $\dfrac{-y\,\dd x + x\,\dd y}{x^2 + y^2}$ पर
आधा समतल $\{x > 0\}$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:forms:4}
(कोणीय रूप) उसे सत्यापित करें $\omega_\theta $
(\cref{ex:b3:forms:angular}) है \omterm{def:b3:forms:closedexact}{बंद किया हुआ}; गणना
$\int_\gamma\omega_\theta $ के लिए $\gamma $ त्रिज्या का चक्र
$ r $ आस-पास $0$; समाप्त करें कि $\omega_\theta $ क्या नहीं है \omterm{def:b3:forms:closedexact}{एकदम सही}
पर $\R^2\setminus\{0\}$, ओर वो $\R^2\setminus\{0\}$ है
इसके किसी भी बिंदु के संबंध में तारे के आकार का (दो मार्ग:
के माध्यम से \cref{thm:b3:forms:poincare}, और सीधे से
ज्यामिति)।
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:forms:5}
(हाइपरसर्फेस का क्षेत्र रूप) चलो $ M \subseteq \R^n $ एक हो
कॉम्पैक्ट ओरिएंटेड हाइपरसर्फेस जिसका \omterm{def:b3:forms:orientation}{अभिविन्यास} द्वारा दिया गया है
एक इकाई \omterm{def:b3:groups:normal}{सामान्य} फ़ील्ड $\nu $ ($(w_1, \dots, w_{n-1})$ सकारात्मक
यदि $(\nu, w_1, \dots, w_{n-1})$ में सकारात्मक $\R^n $). दिखाओ
वह $(n-1)$-रूप $\sigma_\nu =
\sum_i(-1)^{i-1}\nu_i\,\dd x_1\wedge\dots\wedge\widehat{\dd
x_i}\wedge\dots\wedge\dd x_n $ पर प्रतिबंध लगाता है $ M $ तक
\emph{क्षेत्र प्रपत्र}: प्रत्यक्ष पैरामीट्रिजेशन के लिए $\gamma $,
$\gamma^*\sigma_\nu = \sqrt{\det G}\,\dd u_1\wedge\dots
\wedge\dd u_{n-1}$, कहाँ $ G = ({}^t D\gamma)(D\gamma)$ है
ग्राम मैट्रिक्स. \emph{(ध्यान दें कि
$(\gamma^*\sigma_\nu)(e_1, \dots, e_{n-1}) = \det(\nu,
\partial_1\gamma, \dots, \partial_{n-1}\gamma)$ और चौकोर
यह निर्धारक.)} गणना करना $\int_{S^2}\sigma_\nu = 4\pi $.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:forms:6}
गणना करना $\int_{S^2}\omega $ के लिए $\omega = x\,\dd y\wedge\dd z
+ y\,\dd z\wedge\dd x + z\,\dd x\wedge\dd y $: (ए) सीधे
गोलाकार निर्देशांक में; (बी) यूनिट बॉल पर स्टोक्स के माध्यम से।
परिणाम निकालना $\operatorname{vol}(B^3) = \frac{4\pi}3$ से
का क्षेत्रफल $ S^2$, और \omterm{def:b3:groups:normal}{सामान्य}ीकरण करें:
$ n\operatorname{vol}(B^n) = \operatorname{area}(S^{n-1})$,
के साथ संगत \cref{thm:b3:product:ballvolume}.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:forms:7}
(शब्दकोशों से) ध्यानपूर्वक निकालें
\cref{thm:b3:forms:stokes}: (ए) विचलन प्रमेय
$\R^3$ (मिलाना \cref{exo:b3:forms:2} और
\cref{exo:b3:forms:5}); (बी) केल्विन-स्टोक्स प्रमेय
$\int_S\langle\operatorname{curl}F, \nu\rangle\,\dd S =
\oint_{\partial S}\langle F, \tau\rangle\,\dd\ell $ एक के लिए
कॉम्पैक्ट के साथ उन्मुख सतह सीमा में $\R^3$. जाँचें
\omterm{def:b3:forms:orientation}{अभिविन्यास} सम्मेलन ऊपरी आधे क्षेत्र पर सहमत हैं
भूमध्य रेखा से घिरा हुआ.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:forms:8}
(ग्रीन की पहचान) के लिए $ u, v $ एक पर चिकनी \omterm{def:b3:topology:topology}{अड़ोस-पड़ोस} का
ए कॉम्पैक्ट कार्यक्षेत्र $\Omega \subseteq \R^n $ चिकनी के साथ
सीमा, सिद्ध करना
\[
\int_\Omega\bigl(u\,\Delta v + \langle\nabla u, \nabla
v\rangle\bigr) = \int_{\partial\Omega}u\,\partial_\nu
v\,\dd S,
\qquad
\int_\Omega\bigl(u\,\Delta v - v\,\Delta u\bigr) =
\int_{\partial\Omega}\bigl(u\,\partial_\nu v -
v\,\partial_\nu u\bigr)\dd S .
\]
निष्कर्ष: ए \omterm{prop:b3:conformal:harmonicholo}{हार्मोनिक फ़ंक्शन} पर $\Omega $ पर गायब हो रहा है
$\partial\Omega $ समान रूप से गायब हो जाता है, और दो हार्मोनिक
कार्य उसी के साथ सीमा मान मेल खाते हैं ---
डिरिचलेट समस्या में विशिष्टता
\cref{ch:b3:conformal}.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:forms:9}
(चर का परिवर्तन, उन्मुख रूप) चलो $\varphi\colon U
\to V $ के खुलने की एक भिन्नता हो $\R^n $ साथ $\det
D\varphi > 0$, और $ f $ \omterm{def:b3:topology:continuity}{निरंतर} साथ कॉम्पैक्ट समर्थन करें
$ V $. दिखाओ कि \omterm{def:b3:forms:pullback}{ठहराना} पहचान
$\int_U\varphi^*(f\,\dd x_1\wedge\dots\wedge\dd x_n) =
\int_Vf\,\dd x_1\wedge\dots\wedge\dd x_n $ के बराबर है
चर-परिवर्तन प्रमेय
(\cref{thm:b3:product:changeofvar}) ऐसे के लिए $\varphi $, और
स्पष्ट करें कि जैकोबियन पर निरपेक्ष मान कहाँ है
गया.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b3:forms:10}
(विरूपण) चलो $\omega $ एक हो \emph{\omterm{def:b3:forms:closedexact}{बंद किया हुआ}} $2$-पर प्रपत्र
$\R^3\setminus\{0\}$ और $ S_r $ त्रिज्या का क्षेत्र $ r $
पर केन्द्रित $0$. बताते हैं कि $\int_{S_r}\omega $ नहीं करता
पर निर्भर है $ r > 0$ \emph{(बीच में शेल पर स्टोक्स लगाएं
दो त्रिज्याएँ; दोनों का ध्यान रखें सीमा झुकाव)}. के लिए आवेदन करें
द \emph{ठोस-कोण रूप}
\[
\omega = \frac{x\,\dd y\wedge\dd z + y\,\dd z\wedge\dd x +
z\,\dd x\wedge\dd y}{(x^2 + y^2 + z^2)^{3/2}} :
\]
जांचें कि यह है \omterm{def:b3:forms:closedexact}{बंद किया हुआ}, गणना करें $\int_{S_r}\omega = 4\pi $, और
निष्कर्ष यह है \omterm{def:b3:forms:closedexact}{बंद किया हुआ} लेकिन नहीं \omterm{def:b3:forms:closedexact}{एकदम सही} पर
$\R^3\setminus\{0\}$ --- का द्वि-आयामी सहोदर
$\omega_\theta $, और गॉस के नियम की ज्यामितीय \omterm{def:b3:rings:content}{सामग्री}
इलेक्ट्रोस्टैटिक्स में.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:forms:11}
होने देना $\gamma $ एक चिकनी हो \omterm{def:b3:forms:closedexact}{बंद किया हुआ} में वक्र
$\R^2\setminus\{0\}$ साथ $ n = \operatorname{Ind}_\gamma(0)$.
दिखाओ $\int_\gamma\omega = n\int_{S^1}\omega $ के लिए
\emph{प्रत्येक} \omterm{def:b3:forms:closedexact}{बंद किया हुआ} $1$-रूप $\omega $ पर
$\R^2\setminus\{0\}$ \emph{(लिखना $\omega = c\,\omega_\theta
+ \dd f $ द्वारा \cref{exo:b3:forms:12})}. व्याख्या: पर
पंचर विमान, \omterm{def:b3:forms:winding}{घुमावदार संख्या} एकमात्र बाधा है
पीरियड्स के लुप्त होने तक.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b3:forms:12}
(प्रथम डी राम गणना) दिखाएँ कि प्रत्येक \omterm{def:b3:forms:closedexact}{बंद किया हुआ} $1$-रूप
$\omega $ पर $ U = \R^2\setminus\{0\}$ अद्वितीय है
\[
\omega = c\,\omega_\theta + \dd f,
\qquad c = \frac1{2\pi}\int_{S^1}\omega,\quad f \in
\mathcal C^\infty(U) :
\]
परिभाषित करना $ f(p)$ एकीकृत करके $\omega - c\,\omega_\theta $
साथ में ए पथ से $(1,0)$ को $ p $ (रेडियल टुकड़ा फिर गोलाकार
आर्क), दिखाएँ कि परिणाम विकल्पों से स्वतंत्र है
ठीक है क्योंकि $ S^1$-अवधि गायब हो जाती है, और जाँच करें $\dd
f = \omega - c\,\omega_\theta $. निष्कर्ष:
$ H^1(\R^2\setminus\{0\}) \cong \R $, द्वारा उत्पन्न
कोणीय रूप.
\end{exercise}

\section{समस्या: ब्रौवर का निश्चित-बिंदु प्रमेय}

\begin{problem}[{सप्ताहांत समस्या --- कोई वापसी नहीं, नहीं
पलायन}]\label{pb:b3:forms:1}
ब्रौवर का प्रमेय यह बताता है \emph{प्रत्येक सतत मानचित्र की \omterm{def:b3:forms:closedexact}{बंद किया हुआ} यूनिट बॉल $\bar B = \bar B^n \subseteq \R^n $ में
स्वयं का एक निश्चित बिंदु है} --- गणित के महानतम में से एक
गेम थ्योरी के परिणामों के साथ प्रमेय (नैश)।
संतुलन) मैट्रिक्स विश्लेषण के लिए। भेद-रूप प्रमाण
ज्ञात सबसे साफ़ है: स्टोक्स दिखाता है कि गोला नहीं है
गेंद को वापस लेना, और सब कुछ इस प्रकार हो जाता है। लगातार,
$ S = S^{n-1} = \partial\bar B $, $ n \geq 2$, और
\[
\sigma = \sum_{i=1}^n(-1)^{i-1}x_i\,
\dd x_1\wedge\dots\wedge\widehat{\dd x_i}\wedge\dots\wedge
\dd x_n
\]
(टोपी कारक को हटा देती है)।

\medskip
\noindent\textbf{Part I --- The measuring instrument.}
\begin{enumerate}
  \item गणना करना $\dd\sigma $, और स्टोक्स से निष्कर्ष निकालें
        (\cref{thm:b3:forms:stokes}) वह $\int_S\sigma =
        n\operatorname{vol}(\bar B) > 0$, कहाँ $ S $ ले जाता है
        द सीमा \omterm{def:b3:forms:orientation}{अभिविन्यास} गेंद का.
  \item के लिए $ n = 2$ और $ n = 3$, के प्रतिबंध की पहचान करें
        $\sigma $ को $ S $ चाप लंबाई और क्षेत्र रूपों के साथ
        (\cref{exo:b3:forms:5} साथ $\nu(x) = x $) और
        पुनर्गणना $\int_S\sigma $ सीधे.
  \item होने देना $ W \subseteq \R^N $ खुले रहें और $\varphi\colon W
        \to \R^n $ के साथ चिकना $\norm{\varphi(x)} = 1$ सभी के लिए $ x \in W $. बताते हैं कि
        $\varphi^*(\dd\sigma) = 0$. \emph{(अंतर करें
        $\norm\varphi^2 = 1$: की छवि $ D\varphi(x)$
        हाइपरप्लेन में स्थित है $\varphi(x)^\perp $, का
        आयाम $ n - 1$; फिर आवेदन करें
        \cref{prop:b3:forms:linearpullback}(2).)}
  \item निम्नलिखित ``प्रमाण'' में दोष कहाँ है?
        $\int_S\sigma = 0$: ``$ S $ है कॉम्पैक्ट बिना
        सीमा, और $\sigma $ तक सीमित $ S $ एक शीर्ष है
        इस पर फार्म, इसलिए \omterm{def:b3:forms:closedexact}{बंद किया हुआ}, इस तरह $\int_S\sigma =
        \int_S\dd(\text{something}) = 0$ स्टोक्स द्वारा''?
        (गलत शब्द का पता लगाएं।)
  \item भाग II की रणनीति को एक पैराग्राफ में समझाएं:
        क्या एकीकृत किया जाएगा, किस पर, और कहाँ
        विरोधाभास आएगा.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- No smooth retraction.} मान लीजिए,
विरोधाभास के लिए, वह $ r $ एक है \emph{चिकनी वापसी} का
गेंद अपने गोले पर: $ r $ एक पर चिकनी है \omterm{def:b3:topology:topology}{अड़ोस-पड़ोस} का
$\bar B $, $ r(\bar B) \subseteq S $, और $ r(x) = x $ सभी के लिए
$ x \in S $.
\begin{enumerate}[resume]
  \item औचित्य $\int_Sr^*\sigma = \int_S\sigma $
        \emph{(पर $ S $, $ r $ पहचान तक सीमित है: यदि
        $\gamma $ के एक टुकड़े का प्रत्यक्ष पैरामीट्रिजेशन है
        $ S $, तब $ r\circ\gamma = \gamma $)}.
  \item स्टोक्स का उपयोग करना $\bar B $, दिखाओ $\int_Sr^*\sigma =
        \int_{\bar B}\dd(r^*\sigma)$.
  \item दिखाओ $\dd(r^*\sigma) = r^*(\dd\sigma) = 0$ (प्रश्न
        3 पर लागू किया गया $\varphi = r $), और निष्कर्ष निकालें:
        \emph{कोई सहज वापसी नहीं है $\bar B \to S $.}
  \item बहिष्कृत मामले का निपटारा करें $ n = 1$ हाथ से: दिखाओ
        सीधे तौर पर वह नहीं सतत मानचित्र $\intcc{-1}1 \to
        \{-1, 1\}$ दोनों \omterm{thm:b3:complete:completion}{समापन} बिंदुओं को ठीक करता है, और नाम देता है
        आपके द्वारा उपयोग किया गया प्रमेय.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Smooth Brouwer.} होने देना $ g $ हो
एक पर चिकनी \omterm{def:b3:topology:topology}{अड़ोस-पड़ोस} का $\bar B $ साथ $ g(\bar B)
\subseteq \bar B $ और \emph{नहीं} में निश्चित बिंदु $\bar B $.
\begin{enumerate}[resume]
  \item दिखाओ $\delta = \min_{x\in\bar B}\norm{g(x) - x} >
        0$.
  \item के लिए $ x \in \bar B $ होने देना $ u(x) = \frac{x -
        g(x)}{\norm{x - g(x)}}$ और चलो $ r(x) = x +
        t(x)\,u(x)$ किरण का प्रतिच्छेदन हो
        $\{x + tu(x) : t \geq 0\}$ साथ $ S $. हल करें
        द्विघात $\norm{x + tu}^2 = 1$ और प्राप्त करें
        \[
        t(x) = -\langle x, u(x)\rangle +
        \sqrt{1 - \norm x^2 + \langle x, u(x)\rangle^2}
        \;\geq\; 0 .
        \]
  \item दिखाएँ कि रेडिकैंड है \emph{कठोरता से} सकारात्मक
        पर $\bar B $: अगर $1 - \norm x^2 + \langle x,
        u(x)\rangle^2 = 0$ तब $\norm x = 1$ और $\langle
        x, u(x)\rangle = 0$, यानी\ $\langle x, x -
        g(x)\rangle = 0$, यानी\ $\langle x, g(x)\rangle =
        1$; कॉची-श्वार्ज़ द्वारा $\norm x = 1$,
        $\norm{g(x)} \leq 1$, यह बल देता है $ g(x) = x $ ---
        छोड़ा गया। उसका निष्कर्ष निकालो $ r $ एक पर चिकनी है
        \omterm{def:b3:topology:topology}{अड़ोस-पड़ोस} का $\bar B $.
  \item दिखाओ $ r(\bar B) \subseteq S $ और $ r(x) = x $ के लिए $ x
        \in S $ \emph{(के लिए $\norm x = 1$, जाँच करना $ t(x) = 0$
        का उपयोग करते हुए $\langle x, u(x)\rangle \geq 0$, जो स्वयं
        से अनुसरण करता है $\langle x, x - g(x)\rangle = 1 -
        \langle x, g(x)\rangle \geq 0$)}. के साथ \omterm{thm:b3:complete:completion}{समापन} करें
        भाग II: \emph{का हर सहज स्व-मानचित्र $\bar B $ है
        एक निश्चित बिंदु.}
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- \omterm{def:b3:topology:continuity}{निरंतर} Brouwer.} होने देना
$ f\colon\bar B\to\bar B $ होना \omterm{def:b3:topology:continuity}{निरंतर} बिना किसी निश्चित बिंदु के.
\begin{enumerate}[resume]
  \item दिखाओ $\varepsilon = \min_{\bar B}\norm{f - x} > 0$,
        और एक उत्पादन करें \emph{बहुपद} नक्शा $ p\colon\R^n\to
        \R^n $ साथ $\sup_{\bar B}\norm{p - f} <
        \varepsilon/2$ (स्टोन--वीयरस्ट्रैस,
        \cref{thm:b3:complete:stoneweierstrass}, समन्वय करें
        निर्देशांक द्वारा --- अदिश से मार्ग का औचित्य सिद्ध करें
        वेक्टर सन्निकटन)।
  \item वो नक्शा $ p $ गेंद छोड़ सकते हैं; तय करना $ g = \frac{p}{1
        + \varepsilon/2}$. दिखाओ $ g(\bar B) \subseteq \bar B $
        और $\sup_{\bar B}\norm{g - f} < \varepsilon $.
  \item भाग III से विरोधाभास निकालें और निष्कर्ष निकालें:
        \emph{प्रत्येक सतत मानचित्र $\bar B^n \to \bar B^n $
        एक निश्चित बिंदु है.}
  \item उदाहरण द्वारा दिखाएँ कि प्रमेय इस पर विफल रहता है: खुला
        गेंद; गोला $ S $; ए \omterm{def:b3:forms:closedexact}{बंद किया हुआ} वलय. जो
        की संपत्ति $\bar B $ क्या प्रत्येक प्रति-उदाहरण खो जाता है?
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Dividends.}
\begin{enumerate}[resume]
  \item (पेरोन--फ्रोबेनियस, अस्तित्व) चलो $ A $ एक हो
        $ n\times n $ सभी प्रविष्टियों के साथ मैट्रिक्स $> 0$, और
        $\Delta = \{x \in \R^n : x_i \geq 0,\ \sum x_i =
        1\}$. नक्शा दिखाओ $ x \mapsto Ax/\norm{Ax}_1$ है
        अच्छी तरह से परिभाषित और \omterm{def:b3:topology:continuity}{निरंतर} पर $\Delta $, वह
        $\Delta $ है होम्योमॉर्फिक एक को \omterm{def:b3:forms:closedexact}{बंद किया हुआ} की गेंद
        $\R^{n-1}$ (रेडियल \omterm{def:b3:topology:continuity}{होमियोमोर्फिज्म} उत्तल से
        कॉम्पैक्ट गैर-रिक्त के साथ आंतरिक भाग इसके एफ़िन अवधि में),
        और निष्कर्ष निकालो $ A $ के साथ एक eigenvector है
        सख्ती से सकारात्मक प्रविष्टियाँ और eigenvalue $> 0$.
  \item निष्कर्ष निकालें कि प्रत्येक स्टोकेस्टिक मैट्रिक्स सकारात्मक है
        प्रविष्टियाँ (स्तंभों का योग $1$) के पास एक स्टेशनरी है
        संभाव्यता वेक्टर $\pi = A\pi $ --- द
        पेजरैंक-प्रकार वेक्टर. (विशिष्टता भी कायम है लेकिन
        अन्य उपकरणों की आवश्यकता है।)
  \item (बालों वाली गेंद, सेटअप) चलो $ v $ एक पर सहज रहें
        \omterm{def:b3:topology:topology}{अड़ोस-पड़ोस} का $ S = S^{n-1}$ साथ $\langle v(x),
        x\rangle = 0$ और $\norm{v(x)} = 1$ के लिए $ x \in S $
        (ए \emph{इकाई स्पर्शरेखा क्षेत्र}). के लिए $ t \in \R $ तय करना
        $ F_t(x) = x + t\,v(x)$. दिखाओ
        $\norm{F_t(x)} = \sqrt{1 + t^2}$ पर $ S $: $ F_t $ एमएपीएस
        $ S $ क्षेत्र में $\sqrt{1+t^2}\,S $.
  \item बताते हैं कि $ P(t) = \int_SF_t^*\sigma $ एक है
        \emph{बहुपद} में $ t $ \emph{(प्रत्येक गुणांक
        का कार्य $ F_t^*\sigma $ एक चार्ट में बहुपद है
        में $ t $, चार्ट में गुणांक सुचारू के साथ
        परिवर्तनशील; एकीकरण रैखिक है)}.
  \item उसके लिए दिखाओ $\abs t $ छोटा, $ F_t $ एक है
        भिन्नतारूपता से $ S $ पर $\sqrt{1+t^2}\,S $:
        के लिए इंजेक्शन $ t\operatorname{Lip}(v) < 1$; स्थानीय
        व्युत्क्रम फलन प्रमेय द्वारा भिन्नता
        (\cref{thm:b3:submanifolds:ift} चार्ट में); छवि
        खुला और \omterm{def:b3:forms:closedexact}{बंद किया हुआ} में \omterm{def:b3:topology:connected}{जुड़े हुए} लक्ष्य क्षेत्र.
        प्रयोग करके निष्कर्ष निकालें \cref{lem:b3:forms:consistency} और
        स्केलिंग $\sigma_{\lambda x} =
        \lambda^{n}\,\sigma_x $ अंतर्गत $ x \mapsto \lambda x $
        (इसे जांचें), वह
        \[
        P(t) = \pm(1 + t^2)^{n/2}\int_S\sigma,
        \qquad\text{with the sign } + \text{ for small } t
        \]
        (\omterm{def:b3:forms:orientation}{अभिविन्यास} द्वारा संरक्षित निरंतरता से $ t = 0$).
  \item निष्कर्ष (मिल्नोर): यदि $ n $ है \emph{विषम}, $(1 +
        t^2)^{n/2}$ में एक बहुपद नहीं है $ t $, फिर भी यह
        बहुपद से सहमत है $ P(t)/\int_S\sigma $ पास में
        $0$ --- विरोधाभास। इस तरह \emph{द
        सम-आयामी गोले $ S^{n-1}$ ($ n $ विषम) ले जाना
        कोई इकाई स्पर्शरेखा क्षेत्र नहीं}, और, \omterm{def:b3:groups:normal}{सामान्य}ीकरण द्वारा और
        चौरसाई (घटकवार और परियोजना ---
        दोनों चरणों का औचित्य सिद्ध करें), नहीं \omterm{def:b3:topology:continuity}{निरंतर} कहीं नहीं-गायब
        बिल्कुल स्पर्शरेखा क्षेत्र: पृथ्वी पर प्रत्येक हवा एक छोड़ती है
        शांत बिंदु.

  \item (विषम गोले स्वतंत्र रूप से कंघी करते हैं) प्रदर्शन जारी रखें $ S^{2m-1}
        \subseteq \R^{2m} \cong \C^m $ एक स्पष्ट चिकनी
        इकाई स्पर्शरेखा क्षेत्र: $ v(x) = \iu x $ जटिल में
        अंकन, यानी
        \[
        v(x_1, y_1, \dots, x_m, y_m) =
        (-y_1, x_1, \dots, -y_m, x_m) .
        \]
        स्पर्शरेखा और इकाई लंबाई सत्यापित करें, और निष्कर्ष निकालें
        प्रश्न 23 का समता द्वंद्व तीव्र है: a
        गोला ठीक उसी स्थिति में दहनशील होता है जब उसका आयाम होता है
        अजीब. प्रश्न का बहुपद तर्क कहाँ है?
        22 सम के लिए ब्रेक $ n $?
  \item (शून्य से सीमा व्यवहार) चलो $ f \colon \bar
        B^n \to \R^n $ होना \omterm{def:b3:topology:continuity}{निरंतर} साथ $\langle f(x),
        x\rangle \geq 0$ हरएक के लिए $ x \in S^{n-1}$. बताते हैं कि $ f $ कहीं गायब हो जाता है $\bar B^n $.
        \emph{(यदि नहीं, $ g(x) = -f(x)/\norm{f(x)}$ एमएपीएस
        $\bar B $ लगातार में $ S \subseteq \bar B $;
        ब्रौवर को लागू करें $ g $ और इसका खंडन करें सीमा
        परिकल्पना।)} निष्कर्ष निकालें \emph{प्रक्षेपात्मकता
        कसौटी}: ए \omterm{def:b3:topology:continuity}{निरंतर} $ F\colon\R^n\to\R^n $ साथ
        $\frac{\langle F(x), x\rangle}{\norm x} \to
        +\infty $ जैसा $\norm x \to \infty$ विशेषण है ---
        ज़बरदस्ती का परिमित-आयामी पूर्वज
        अरेखीय विश्लेषण के तर्क.
\end{enumerate}
\end{problem}
