\chapter{माप सिद्धांत}\label{ch:b3:measure}

का एक उपसमुच्चय कितने समय का होता है $\R $? भोला जवाब --- को सौंपें
प्रत्येक ने एक अनुवाद-अपरिवर्तनीय लंबाई का विस्तार किया
अंतराल --- है \emph{असंभव}: विटाली का निर्माण, पर
इस अध्याय का अंत, बिना किसी संगति के एक सेट तैयार करता है
लंबाई. \omterm{def:b3:measure:measure}{उपाय} सिद्धांत अनुशासित वापसी है: प्रतिबंधित
के एक अमीर वर्ग पर ध्यान \emph{औसत दर्जे का} सेट, जिस पर ए
गणनीय रूप से योगात्मक लंबाई मौजूद है और अद्वितीय है। पुरस्कार हैं
अपार --- लेब्सग्यू का अभिन्न अंग (\cref{ch:b3:lebesgue}), द
$ L^p $ कार्यात्मक विश्लेषण के स्थान, और संपूर्ण आधुनिक
संभाव्यता (\cref{ch:b3:probability}) तीनों पर बने हैं
प्रमेय यहाँ सिद्ध हुए: डिनकिन की विशिष्टता प्रमेयिका,
कैरथ\'ईओडोरी का विस्तार प्रमेय, और इसका अस्तित्व
\omterm{def:b3:measure:lebesgueouter}{लेब्सेग माप}.

\section{$\sigma $-बीजगणित}

\begin{definition}\label{def:b3:measure:sigmaalgebra}
ए \emph{$\sigma $-बीजगणित}\index{$\sigma $-बीजगणित}
एक सेट पर $ X $ एक परिवार है $\mathcal A $ युक्त उपसमुच्चय का
$\varnothing $, पूरक के तहत और नीचे स्थिर
\emph{गणनीय} यूनियन (इसलिए गणनीय चौराहे, सेट
मतभेद, और इसमें शामिल है $ X $). जोड़ी $(X, \mathcal A)$
एक है \emph{मापने योग्य स्थान}; के सदस्य $\mathcal A $ हैं
\emph{मापने योग्य सेट}. किसी भी परिवार के लिए $\mathcal E $ उपसमुच्चय का,
$\sigma(\mathcal E)$ सबसे छोटे को दर्शाता है $\sigma $-बीजगणित
युक्त $\mathcal E $ (उन सभी का प्रतिच्छेदन ---
का एक चौराहा $\sigma $-बीजगणित एक है)।
\end{definition}

\begin{definition}\label{def:b3:measure:borel}
 \emph{बोरेल $\sigma $-बीजगणित}\index{बोरेल $\sigma $-बीजगणित} एक का संस्थानिक अंतरिक्ष है
$\mathcal B(X) = \sigma(\{\text{open sets}\})$. पर $\R $:
$\mathcal B(\R)$ खुले अंतरालों द्वारा भी उत्पन्न होता है
\omterm{def:b3:topology:interior}{बंद} अंतराल, किरणों द्वारा $(-\infty, a]$, और किरणों द्वारा
तर्कसंगत \omterm{thm:b3:complete:completion}{समापन} बिंदु के साथ (\cref{exo:b3:measure:1}) --- प्रत्येक
परिवार गणनीय परिचालनों द्वारा ओपन उत्पन्न करता है, उदाहरण के लिए\
प्रत्येक \omterm{def:b3:topology:topology}{खुला सेट} का $\R $ खुले अंतरालों का एक गणनीय संघ है
तर्कसंगत डेटा के साथ.
\end{definition}

\begin{definition}\label{def:b3:measure:dynkin}
ए \emph{$\pi $-प्रणाली}\index{$\pi $-प्रणाली} एक परिवार है
परिमित चौराहों के अंतर्गत स्थिर। ए
\emph{$\lambda $-प्रणाली}\index{$\lambda $-प्रणाली}
(डाइनकिन वर्ग) एक परिवार है $\mathcal D $ साथ: $ X \in \mathcal
D $; $ A, B \in \mathcal D $, $ A \subseteq B $ $\Rightarrow $ $ B
\setminus A \in \mathcal D $; और $ A_n \uparrow A $, $ A_n \in
\mathcal D $ $\Rightarrow $ $ A \in \mathcal D $.
\end{definition}

\begin{theorem}[डिन्किन का $\pi $--$\lambda $ लेम्मा]
\label{thm:b3:measure:dynkin}
यदि एक $\lambda $-प्रणाली $\mathcal D $ एक शामिल है $\pi $-प्रणाली
$\mathcal P $, तब $\mathcal D \supseteq
\sigma(\mathcal P)$.\index{डिनकिन की लेम्मा}
\end{theorem}

\begin{proof}
होने देना $\mathcal D_0$ सबसे छोटा हो $\lambda $-प्रणाली युक्त
$\mathcal P $ (ऐसे सभी का प्रतिच्छेदन); यह दिखाने के लिए पर्याप्त है
$\mathcal D_0$ एक है $\sigma $-बीजगणित, तब से
$\sigma(\mathcal P) \subseteq \mathcal D_0 \subseteq \mathcal
D $. ए $\lambda $-परिमित चौराहों के तहत सिस्टम स्थिर
\emph{है} ए $\sigma $-बीजगणित: पूरक ($ X \setminus A = X
\setminus A $ साथ $ A \subseteq X $), परिमित संघ ($ A \cup B =
X \setminus ((X\setminus A)\cap(X\setminus B))$), और गणनीय
यूनियनों के माध्यम से $\bigcup_{k \leq n}A_k \uparrow \bigcup_kA_k $. तो हम
साबित करो $\mathcal D_0$ एक है $\pi $-सिस्टम, दो चरणों में। होने देना
\[
\mathcal D_1 = \{A \in \mathcal D_0 : A \cap P \in \mathcal
D_0 \ \forall P \in \mathcal P\}.
\]
$\mathcal D_1$ एक है $\lambda $-प्रणाली (सभी तीन स्वयंसिद्ध हैं
के साथ प्रतिच्छेद करके सत्यापित किया गया $ P $: जैसे\ $(B\setminus A)\cap P
= (B \cap P)\setminus(A \cap P)$, भीतर एक उचित अंतर
$\mathcal D_0$) और शामिल है $\mathcal P $ ($\pi $-प्रणाली):
$\mathcal D_1 = \mathcal D_0$. अब चलो
\[
\mathcal D_2 = \{A \in \mathcal D_0 : A \cap D \in \mathcal
D_0\ \forall D \in \mathcal D_0\}.
\]
पिछले चरण से, $\mathcal D_2 \supseteq \mathcal P $; और
$\mathcal D_2$ एक है $\lambda $-समान सत्यापन द्वारा प्रणाली:
$\mathcal D_2 = \mathcal D_0$, जो बिल्कुल यही कहता है
$\mathcal D_0$ चौराहों के नीचे स्थिर है।
\end{proof}

\section{पैमाने}

\begin{definition}\label{def:b3:measure:measure}
ए \emph{उपाय}\index{उपाय} पर $(X, \mathcal A)$ एक नक्शा है
$\mu \colon \mathcal A \to [0, +\infty]$ साथ $\mu(\varnothing)
= 0$ वह है \emph{$\sigma $-योजक}: जोड़ीवार असंयुक्त के लिए
$(A_n)_{n\in\N}$,
\[
\mu\Bigl(\bigsqcup_n A_n\Bigr) = \sum_n \mu(A_n).
\]
$(X, \mathcal A, \mu)$ एक है \emph{स्थान मापें}; $\mu $ है
\emph{परिमित} अगर $\mu(X) < \infty $, ए \emph{संभाव्यता
उपाय} अगर $\mu(X) = 1$, \emph{$\sigma $-परिमित} अगर $ X $ एक है
परिमित माप के सेटों का गणनीय संघ। उदाहरण: गिनती
पर मापें $(\N, \mathcal P(\N))$; डिराक मास $\delta_a(A)
= \mathbf 1_{a \in A}$; और, इस अध्याय का उद्देश्य,
\omterm{def:b3:measure:lebesgueouter}{लेब्सेग माप}.
\end{definition}

\begin{proposition}\label{prop:b3:measure:basics}
होने देना $\mu $ एक हो \omterm{def:b3:measure:measure}{उपाय}. (ए) एकरसता: $ A \subseteq B
\Rightarrow \mu(A) \leq \mu(B)$. (बी) गणनीय उप-अयोग्यता:
$\mu(\bigcup A_n) \leq \sum\mu(A_n)$. (सी) निरंतरता नीचे की ओर से: $ A_n \uparrow A \Rightarrow \mu(A_n) \to \mu(A)$. (डी)
निरंतरता उपर से: $ A_n \downarrow A $ \emph{साथ
$\mu(A_1) < \infty $} $\Rightarrow \mu(A_n) \to \mu(A)$.
\end{proposition}

\begin{proof}
(ए) $ B = A \sqcup (B\setminus A)$. (बी) विच्छेदन: $ B_n = A_n
\setminus \bigcup_{k<n}A_k $ एक ही संघ से विच्छेदित हैं,
और $\mu(B_n) \leq \mu(A_n)$. (सी) $ A = \bigsqcup_n (A_n
\setminus A_{n-1})$ ($ A_0 = \varnothing $): का आंशिक योग
$\sum\mu(A_n\setminus A_{n-1})$ हैं $\mu(A_n)$. (डी) लागू करें (सी)
को $ A_1 \setminus A_n \uparrow A_1 \setminus A $ और घटाओ
से $\mu(A_1)$ --- परिमितता घटाव बनाती है
वैध. इसके बिना प्रति उदाहरण: $ A_n = [n, \infty)$ के लिए
\omterm{def:b3:measure:lebesgueouter}{लेब्सेग माप}: $ A_n \downarrow \varnothing $ लेकिन $\mu(A_n) =
\infty $.
\end{proof}

\begin{theorem}[विशिष्टता]\label{thm:b3:measure:uniqueness}
होने देना $\mu, \nu $ होना पैमाने पर $\sigma(\mathcal P)$, $\mathcal
P $ ए $\pi $-प्रणाली, साथ $\mu = \nu $ पर $\mathcal P $. अगर वहाँ
सेट हैं $ P_k \in \mathcal P $ साथ $ P_k \uparrow X $ और
$\mu(P_k) < \infty $, तब $\mu = \nu $ सब पर
$\sigma(\mathcal P)$.
\end{theorem}

\begin{proof}
हल करना $ k $ और परिमित पर विचार करें पैमाने $\mu_k(A) = \mu(A \cap
P_k)$ और $\nu_k(A) = \nu(A \cap P_k)$ पर $\sigma(\mathcal
P)$: वे इस पर सहमत हैं $\mathcal P $, तब से $ P \cap P_k \in \mathcal
P $ ($\pi $-सिस्टम), और वे असाइन करते हैं $ X $ वही परिमित मान
$\mu(P_k)$. \omterm{def:b3:groups:action}{कक्षा} $\mathcal D = \{A : \mu_k(A) =
\nu_k(A)\}$ एक है $\lambda $-प्रणाली: $ X \in \mathcal D $; उचित
घटाव द्वारा अंतर (सीमित मान); बढ़ती सीमा
द्वारा निरंतरता नीचे की ओर से (\cref{prop:b3:measure:basics}(सी))।
इसमें शामिल है $\pi $-प्रणाली $\mathcal P $, तो डिनकिन
(\cref{thm:b3:measure:dynkin}) देता है $\mathcal D \supseteq
\sigma(\mathcal P)$: $\mu_k = \nu_k $ हर जगह. अंत में, के लिए
कोई भी $ A \in \sigma(\mathcal P)$, निरंतरता नीचे से साथ में $ A
\cap P_k \uparrow A $ देता है $\mu(A) = \lim_k\mu_k(A) =
\lim_k\nu_k(A) = \nu(A)$.
\end{proof}

\section{बाहरी उपाय और कैरथ\'ईओडोरी का प्रमेय}

\begin{definition}\label{def:b3:measure:outer}
एक \emph{बाहरी माप}\index{बाहरी माप} पर $ X $ एक नक्शा है
$\mu^* \colon \mathcal P(X) \to [0, \infty]$ साथ
$\mu^*(\varnothing) = 0$, मोनोटोन, और गणनीय रूप से उपयोज्य।
एक सेट $ A $ है \emph{$\mu^*$-मापने योग्य} (कैराथ\'ईओडोरी) यदि यह
प्रत्येक सेट को योगात्मक रूप से विभाजित करता है:
\[
\mu^*(E) = \mu^*(E \cap A) + \mu^*(E \setminus A)
\qquad \text{for every } E \subseteq X
\]
($\leq $ सदैव उपादेयता द्वारा धारण करता है; \omterm{def:b3:rings:content}{सामग्री} है $\geq $).
\end{definition}

\begin{theorem}[कैरथ\'ईओडोरी]\label{thm:b3:measure:caratheodory}
 $\mu^*$-मापने योग्य सेट फॉर्म ए $\sigma $-बीजगणित $\mathcal
M $, और $\mu^*\restriction_{\mathcal M}$ एक है \omterm{def:b3:measure:measure}{उपाय}.
इसके अलावा हर सेट के साथ $\mu^*(N) = 0$ से संबंधित $\mathcal M $
(द \omterm{def:b3:measure:measure}{उपाय} है \emph{\omterm{def:b3:complete:complete}{पूरा}}).\index{Carathéodory's
theorem}
\end{theorem}

\begin{proof}
$\mathcal M $ रोकना $\varnothing $ और नीचे स्थिर है
पूरक (परिभाषित स्थिति सममित है $ A $,
$ X\setminus A $). \emph{परिमित संघ}: होने देना $ A, B \in \mathcal
M $ और $ E $ मनमाना; बंटवारे $ E $ द्वारा $ A $, फिर प्रत्येक टुकड़ा
$ B $:
\[
\mu^*(E) = \mu^*(E\cap A\cap B) + \mu^*(E\cap A\setminus B) +
\mu^*(E\cap B\setminus A) + \mu^*(E\setminus(A\cup B)).
\]
पहले तीन टुकड़े कवर करते हैं $ E \cap (A \cup B)$, तो
उपादेयता देता है $\mu^*(E) \geq \mu^*(E\cap(A\cup B)) +
\mu^*(E\setminus(A\cup B))$: $ A \cup B \in \mathcal M $. द्वारा
प्रेरण, परिमित संघ; पूरकों के साथ, परिमित असम्बद्धता के साथ
जोड़-तोड़ उपलब्ध हैं.

\emph{एडिटिविटी चालू $\mathcal M $}: विच्छेदन के लिए $ A, B \in
\mathcal M $ और कोई भी $ E $: $\mu^*(E\cap(A\sqcup B)) = \mu^*(E\cap
A) + \mu^*(E \cap B)$ (द्वारा विभाजित $ A $); प्रेरण द्वारा,
\[
\mu^*\Bigl(E \cap \bigsqcup_{k\leq n}A_k\Bigr)
= \sum_{k\leq n}\mu^*(E\cap A_k).
\tag{$*$}
\]

\emph{गणनीय संघ}: होने देना $(A_k) \subseteq \mathcal M $ हो
असंयुक्त (पर्याप्त, बीजगणित के भीतर असंयुक्तीकरण द्वारा
$\mathcal M $), $ A = \bigsqcup A_k $, $ E $ मनमाना। का उपयोग करते हुए
$\bigsqcup_{k\leq n}A_k \in \mathcal M $ और एकरसता:
\[
\mu^*(E) = \mu^*\Bigl(E\cap\bigsqcup_{k\leq n}A_k\Bigr) +
\mu^*\Bigl(E\setminus\bigsqcup_{k\leq n}A_k\Bigr)
\geq \sum_{k \leq n}\mu^*(E\cap A_k) + \mu^*(E\setminus A)
\]
द्वारा ($*$). होने देना $ n \to \infty $ और गणनीय उप-अयोग्यता का उपयोग करें
पीछे की ओर:
\[
\mu^*(E) \geq \sum_{k}\mu^*(E\cap A_k) + \mu^*(E\setminus A)
\geq \mu^*(E \cap A) + \mu^*(E\setminus A) \geq \mu^*(E):
\]
सभी असमानताएँ समानताएँ हैं। यह दोनों को सिद्ध करता है $ A \in
\mathcal M $ और, ले रहा हूँ $ E = A $, की गणनीय योगात्मकता
$\mu^*$ पर $\mathcal M $.

\emph{शून्य सेट}: अगर $\mu^*(N) = 0$, फिर किसी के लिए $ E $:
$\mu^*(E \cap N) + \mu^*(E\setminus N) \leq 0 + \mu^*(E)$:
$ N \in \mathcal M $.
\end{proof}

\section{लेब्सग्यू उपाय चालू \texorpdfstring{$\R $}{R}}

\begin{definition}\label{def:b3:measure:lebesgueouter}
 \emph{लेबेस्गे बाहरी माप}\index{लेबेस्गे बाहरी माप} का $ A
\subseteq \R $ है
\[
\lambda^*(A) = \inf\Bigl\{\sum_{n} (b_n - a_n) :
A \subseteq \bigcup_n \intoo{a_n}{b_n}\Bigr\}
\]
(खुले अंतरालों द्वारा गणनीय कवर)।
\end{definition}

\begin{lemma}\label{lem:b3:measure:outerbasics}
$\lambda^*$ एक \omterm{def:b3:measure:outer}{बाहरी माप}, अनुवाद के अंतर्गत अपरिवर्तनीय,
और $\lambda^*(I) = \ell(I)$ (लंबाई) प्रत्येक अंतराल के लिए
$ I $.
\end{lemma}

\begin{proof}
\omterm{def:b3:measure:outer}{बाहरी माप}: $\varnothing $ मनमाने ढंग से छोटे द्वारा कवर किया गया है
अंतराल; एकरसता स्पष्ट है; सबएडिटिविटी: दिए गए कवर
प्रत्येक का $ A_n $ अंदर $\varepsilon 2^{-n}$ अनंत का,
उनका संघ कवर करता है $\bigcup A_n $ कुल लंबाई के साथ $\leq \sum
\lambda^*(A_n) + \varepsilon $. अनुवाद अपरिवर्तनीयता:
कवर का अनुवाद करें.

लंबाई: यह उपचार के लिए पर्याप्त है $ I = \intcc ab $ (अन्य प्रकार
\omterm{thm:b3:complete:completion}{समापन} बिंदुओं से भिन्न होता है, जो है \omterm{def:b3:measure:outer}{बाहरी माप} $0$: द्वारा कवर करें
छोटे अंतराल; फिर निचोड़ें $\intcc{a+\varepsilon}{b -
\varepsilon} \subseteq \intoo ab $-प्रकार की तुलना)।
$\lambda^*(\intcc ab) \leq b - a $: द्वारा कवर
$\intoo{a-\varepsilon}{b+\varepsilon}$. इसके विपरीत चलो $\intcc
ab \subseteq \bigcup_n\intoo{a_n}{b_n}$: द्वारा
\emph{सघनता} (बोरेल--लेब्सग्यू,
\cref{thm:b3:topology:metriccompact}), परिमित रूप से अनेक अंतराल
पर्याप्त, कहो $ I_1, \dots, I_N $. हम दिखाते हैं $\sum_{k\leq N}(b_k -
a_k) \geq b - a $ पर प्रेरण द्वारा $ N $: चुनना $ I_{k_1} \ni a $; अगर
$ b_{k_1} > b $ हो गया ($ b_{k_1} - a_{k_1} > b - a $); अन्यथा
खंड $\intcc{b_{k_1}}b $ शेष द्वारा कवर किया गया है $ N - 1$
अंतराल, और प्रेरण देता है $\sum_{k \neq k_1}(b_k - a_k)
\geq b - b_{k_1}$, जबकि $ b_{k_1} - a_{k_1} > b_{k_1} - a $:
अंदाज़ करना।
\end{proof}

\begin{theorem}[लेब्सेग माप]\label{thm:b3:measure:lebesgue}
प्रत्येक बोरेल सेट $\R $ है $\lambda^*$-मापने योग्य. द
प्रतिबंध $\lambda $ का $\lambda^*$ तक $\sigma $-बीजगणित
$\mathcal L = \mathcal M_{\lambda^*} \supseteq \mathcal B(\R)$
(द \emph{लेब्सग्यू $\sigma $-बीजगणित}) अद्वितीय है \omterm{def:b3:measure:measure}{उपाय} पर
$\mathcal B(\R)$ प्रत्येक अंतराल को उसकी लंबाई निर्दिष्ट करना; यह है
अनुवाद-अपरिवर्तनीय और $\sigma $-परिमित.
\end{theorem}

\begin{proof}
द्वारा \cref{thm:b3:measure:caratheodory} यह प्रत्येक को दिखाने के लिए पर्याप्त है
किरण $ A = \intoo{-\infty}c $ है $\lambda^*$-मापने योग्य
(किरणें उत्पन्न होती हैं $\mathcal B $,
\cref{def:b3:measure:borel}). होने देना $ E \subseteq \R $ साथ
$\lambda^*(E) < \infty $ और $\bigcup I_n \supseteq E $ एक आवरण
के साथ $\sum\ell(I_n) \leq \lambda^*(E) + \varepsilon $. प्रत्येक
$ I_n $ दो अंतरालों में विभाजित हो जाता है $ I_n' = I_n \cap A $ और
$ I_n'' = I_n\setminus A $ (एक अंतराल शून्य से एक किरण है
अंतराल) के साथ $\ell(I_n') + \ell(I_n'') = \ell(I_n)$;
$ I_n'$ ढकना $ E \cap A $ और यह $ I_n''$ ढकना $ E \setminus A $
(प्रत्येक को लंबाई के एक खुले अंतराल तक बड़ा करें $\ell +
\varepsilon2^{-n}$ परिभाषा के भीतर रहने के लिए), इसलिए
\[
\lambda^*(E\cap A) + \lambda^*(E\setminus A)
\leq \sum_n\bigl(\ell(I_n') + \ell(I_n'')\bigr) + 2\varepsilon
\leq \lambda^*(E) + 3\varepsilon .
\]
विशिष्टता: दो पैमाने की लंबाई से सहमत हूं
$\pi $-अंतराल की प्रणाली $\intoc ab $ (उन पर परिमित) सहमत हूं
$\sigma(\text{intervals}) = \mathcal B $ द्वारा
\cref{thm:b3:measure:uniqueness} साथ $ P_k = \intoc{-k}k $.
$\sigma $-अंतता: $\R = \bigcup(-k, k]$.
\end{proof}

\begin{theorem}[नियमितता]\label{thm:b3:measure:regularity}
हरएक के लिए $ A \in \mathcal L $:\index{लेबेस्गुए की नियमितता
उपाय}
\[
\lambda(A) = \inf\{\lambda(U) : U \supseteq A \text{ open}\}
= \sup\{\lambda(K) : K \subseteq A \text{ compact}\}.
\]
\end{theorem}

\begin{proof}
\emph{आउटर}: एक कवर $\bigcup I_n $ साथ $\sum\ell(I_n) \leq
\lambda(A) + \varepsilon $ एक \omterm{def:b3:topology:topology}{खुला सेट} $ U \supseteq A $ साथ
$\lambda(U) \leq \lambda(A) + \varepsilon $ (अवसादात्मकता); अगर
$\lambda(A) = \infty $ बयान तुच्छ है. \emph{भीतरी}:
पहले चलो $ A $ सीमाबद्ध होना, $ A \subseteq [-M, M]$. एक खुला चुनें
$ U \supseteq ([-M,M]\setminus A)$ साथ $\lambda(U) \leq
\lambda([-M,M]\setminus A) + \varepsilon $; तब $ K = [-M,
M]\setminus U $ है कॉम्पैक्ट, $ K \subseteq A $, और
\[
\lambda(K) = \lambda([-M,M]) - \lambda([-M,M]\cap U)
\geq \lambda([-M,M]) - \bigl(\lambda([-M,M]) - \lambda(A) +
\varepsilon\bigr) = \lambda(A) - \varepsilon .
\]
\omterm{def:b3:groups:normal}{सामान्य} के लिए $ A $: $\lambda(A) = \lim_M\lambda(A\cap[-M,M])$
(निरंतरता नीचे से) और बाउंडेड केस को अंदर लगाएं।
\end{proof}

\begin{example}\label{ex:b3:measure:cantor}
कैंटर सेट (\cref{exo:b3:topology:10}) है $\lambda(C) = 0$:
$ C \subseteq C_n $, का एक संघ $2^n $ लंबाई का अंतराल
$3^{-n}$, इसलिए $\lambda(C) \leq (2/3)^n \to 0$. एक बेशुमार
शून्य सेट --- कार्डिनैलिटी नहीं दिखती \omterm{def:b3:measure:measure}{उपाय}. इसके विपरीत,
\emph{मोटा कैंटर सेट} (\cref{exo:b3:measure:5}) कहीं नहीं हैं
सकारात्मकता से सघन \omterm{def:b3:measure:measure}{उपाय}: \omterm{def:b3:topology:topology}{टोपोलॉजी} नहीं देखता \omterm{def:b3:measure:measure}{उपाय}
दोनों में से एक। सप्ताहांत की समस्या इस परस्पर क्रिया को आगे बढ़ाती है
आश्चर्यजनक निष्कर्ष: लेब्सेग-मापने योग्य सेट हैं
बोरेल नहीं हैं.
\end{example}

\begin{theorem}[विटाली]\label{thm:b3:measure:vitali}
कोई नहीं है \omterm{def:b3:measure:measure}{उपाय} पर \emph{सभी} के उपसमुच्चय $\R $ वह है
अनुवाद-अपरिवर्तनीय और प्रत्येक अंतराल को उसकी लंबाई निर्दिष्ट करता है।
विशेष रूप से $\mathcal L \neq \mathcal P(\R)$: गैर मापने योग्य
सेट मौजूद हैं.\index{विटाली का गैर-मापने योग्य सेट}
\end{theorem}

\begin{proof}
कल्पना करना $\mu $ हम एक हैं। पर $\intcc01$, समतुल्यता पर विचार करें
$ x \sim y \iff x - y \in \Q $; से \emph{पसंद का सिद्धांत}
एक प्रतिनिधि चुनें $\intcc01$ प्रति वर्ग: एक सेट $ V $. के लिए
$ q \in \Q\cap\intcc{-1}1$, अनुवाद $ V + q $ जोड़ीवार हैं
असंयुक्त (दो बिंदु) $ V $ एक तर्कसंगत द्वारा भिन्न होगा
समकक्ष फिर भी विशिष्ट प्रतिनिधि) और
\[
\intcc01 \subseteq \bigsqcup_{q \in \Q\cap\intcc{-1}1}(V + q)
\subseteq \intcc{-1}2 :
\]
प्रत्येक के बाद पहला समावेश $ x \in \intcc01$ से भिन्न है
इसका प्रतिनिधि $ v $ एक तर्कसंगत द्वारा $ q = x - v \in
\intcc{-1}1$. एकरसता और $\sigma $-एडिटिविटी देना
\[
1 \leq \sum_{q}\mu(V + q) \leq 3,
\qquad\text{with } \mu(V + q) = \mu(V) \text{ for all } q .
\]
स्थिरांक का अनंत योग $\mu(V)$ है $0$ या $\infty $:
दोनों सीमाएँ कायम नहीं रह सकतीं। तो ऐसा नहीं $\mu $ मौजूद है --- और $ V
\notin \mathcal L $, तब से $\lambda $ पर $\mathcal L $ सब कुछ है
प्रयुक्त गुण.
\end{proof}

\begin{remark}\label{rem:b3:measure:banachtarski}
में $\R^3$ विफलता अधिक नाटकीय है: बानाच--टार्स्की
विरोधाभास एक गेंद को पाँच टुकड़ों में विघटित करता है जो पुनः एकत्रित हो जाते हैं
घूर्णन और अनुवाद, में \emph{दो} उसी की गेंदें
त्रिज्या --- अत: एक परिमित योगात्मक घूर्णन-अपरिवर्तनीय भी नहीं
के सभी उपसमूहों पर वॉल्यूम $\R^3$ मौजूद है. टुकड़े हैं, के
बेशक, गैर-मापने योग्य। मापनीयता नौकरशाही नहीं है
सावधानी; यह सुसंगति की सीमा है।
\end{remark}

\begin{method}\label{met:b3:measure:goodsets}
 \emph{अच्छे सेट सिद्धांत}: यह साबित करने के लिए कि सभी सेट
$\sigma(\mathcal E)$ एक संपत्ति है, दिखाओ कि अच्छे सेट
फॉर्म ए $\sigma $-बीजगणित (या ए $\lambda $-सिस्टम, यदि संपत्ति
माप-सैद्धांतिक है और $\mathcal E $ एक है $\pi $-प्रणाली ---
फिर डिनकिन) युक्त $\mathcal E $. का लगभग हर प्रमाण
यह अध्याय और अगला एक उदाहरण है। दो को सिद्ध करना पैमाने
बराबर: उन्हें जनरेटिंग पर जांचें $\pi $-सिस्टम प्लस
$\sigma $-अंतता (\cref{thm:b3:measure:uniqueness}). To
एक निर्माण करें \omterm{def:b3:measure:measure}{उपाय}: एक निर्माण करें \omterm{def:b3:measure:outer}{बाहरी माप} कवर और उद्धरण द्वारा
कैरथ\'ईओडोरी.
\end{method}

\begin{exercise}[$\star $]\label{exo:b3:measure:2}

(बी) इसे दर्शाने वाला एक उदाहरण दीजिए निरंतरता उपर से

\begin{exercise}[$\star\star $]\label{exo:b3:measure:3}
होने देना $\mu, \nu $ संभावना हो पैमाने पर $\mathcal B(\R)$

\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:measure:4}
(बोरेल--कैंटेली, \omterm{def:b3:measure:measure}{उपाय} संस्करण) चलो $(A_n)$ मापने योग्य हो
के साथ $\sum_n\mu(A_n) < \infty $, और $\limsup A_n =
\bigcap_N\bigcup_{n\geq N}A_n $ (अंक संबंधित हैं
असीम रूप से अनेक $ A_n $). दिखाओ $\mu(\limsup A_n) = 0$.
आवेदन: लगभग हर के लिए $ x \in \intcc01$, केवल अंतिम रूप से
अनेक $ n $ संतुष्ट $\abs{x - p/q_n} \leq 4^{-n}$ के लिए
$ n $-वें तर्कसंगत $ p/q_n $ की एक गणना का
$\Q\cap\intcc01$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:measure:5}
(फैट कैंटर सेट) कैंटर निर्माण को दोहराएं $\intcc01$,
लेकिन कदम पर $ n $ प्रत्येक से हटाएँ $2^{n-1}$ अंतराल ए
\emph{\omterm{ex:b3:groups:actions}{केंद्र}ित} लंबाई का खुला अंतराल $4^{-n}$ केवल। दिखाओ
कि परिणाम $ K = \bigcap K_n $ है कॉम्पैक्ट, खाली है
आंतरिक भाग (कोई अंतराल नहीं बचता), और
\[
\lambda(K) = 1 - \sum_{n\geq1}2^{n-1}4^{-n} = \tfrac12 :
\]
कहीं नहीं \omterm{def:b3:topology:interior}{घना} सेट का \omterm{def:b3:measure:measure}{उपाय} $\frac12$. एक निष्कर्ष निकालें
\emph{\omterm{rem:b3:complete:meagre}{अल्प}} का उपसमुच्चय $\intcc01$ पूर्ण का \omterm{def:b3:measure:measure}{उपाय} $1$, और एक
का खुला सघन उपसमुच्चय \omterm{def:b3:measure:measure}{उपाय} $< \varepsilon $.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:measure:6}
होने देना $\mu $ एक हो \omterm{def:b3:measure:measure}{उपाय} पर $\mathcal B(\R)$, के अंतर्गत अपरिवर्तनीय
अनुवाद, के साथ $ c = \mu(\intoc01) < \infty $. दिखाओ $\mu =
c\,\lambda $ पर $\mathcal B(\R)$. \emph{(गणना करें $\mu $ विभाजित करके डायडिक अंतराल पर
$\intoc01$ में $2^n $ अनुवाद करें, फिर आह्वान करें
\cref{thm:b3:measure:uniqueness}.)}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:measure:7}
(अनुमान) चलो $ A \in \mathcal L $ साथ $\lambda(A) <
\infty $ और $\varepsilon > 0$. दिखाओ वहाँ एक है \emph{परिमित}
अंतरालों का मिलन $ B $ साथ $\lambda(A\,\triangle\,B) <
\varepsilon $ ($\triangle $ = सममित अंतर).
\emph{(नियमितता: निचोड़ें $ K \subseteq A \subseteq U $ और उपयोग करें
खुले की संरचना $ U $ के गणनीय संघ के रूप में
अंतराल, साथ ही की सघनता $ K $.)}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b3:measure:8}
(स्टीनहौस) चलो $ A \in \mathcal L $ साथ $\lambda(A) > 0$. बताते हैं कि $ A - A = \{x - y : x, y \in A\}$ एक अंतराल शामिल है
चारों ओर $0$.
\emph{(कम करें $\lambda(A) < \infty $; द्वारा

साथ $\lambda(A \cap I) > \frac34\ell(I)$; फिर के लिए $\abs t <
\frac12\ell(I)$, सेट $ A\cap I $ और $(A\cap I) + t $ दोनों
लंबाई के अंतराल में बैठें $\frac32\ell(I)$ और कुल है
\omterm{def:b3:measure:measure}{उपाय} $> \frac32\ell(I)$: उन्हें प्रतिच्छेद करना होगा।)}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b3:measure:9}
वह हर दिखाओ $ A \in \mathcal L $ साथ $\lambda(A) > 0$
इसमें एक गैर-मापने योग्य उपसमुच्चय शामिल है। \emph{(प्रतिच्छेद करें $ A $ के साथ
अनुवाद करता है $ V + q $ विटाली के सेट का: यदि सभी $ A \cap (V+q)$ थे

\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:measure:10}
बताते हैं कि $ A \subseteq \R $ साथ $\lambda^*(A) < \infty $ है
प्रत्येक के लिए लेबेस्ग्यू-मापने योग्य आईएफएफ $\varepsilon > 0$ वहाँ एक है

\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:measure:11}
(निरंतरता एकस्वर सीमा के साथ, और इसकी तीक्ष्णता)
(ए) दिखाएं कि मापने योग्य सेट के लिए, $\mu(\liminf A_n) \leq
\liminf\mu(A_n)$ (सेट के लिए फ़तौ), और वह यदि
$\mu\bigl(\bigcup A_n\bigr) < \infty $, भी
$\limsup\mu(A_n) \leq \mu(\limsup A_n)$.
(बी) प्रदर्शन, के लिए \omterm{def:b3:measure:lebesgueouter}{लेब्सेग माप} पर $\R $, के साथ एक क्रम
$\mu(A_n) = 1$ सभी के लिए $ n $ अभी तक $\mu(\limsup A_n) = 0$: द
दूसरी असमानता में परिमितता परिकल्पना नहीं है
सजावटी.
(सी) निष्कर्ष निकालें: यदि $\sum\mu(A_n) < \infty $ तब
$\mu(\limsup A_n) = 0$ (बोरेल--कैंटेली फिर से), और यदि
$ A_n $ वृद्धि या कमी (के साथ) $\mu(A_1) < \infty $ में
घटते मामले), $\mu(\lim A_n) = \lim\mu(A_n)$.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b3:measure:12}
(ईगोरोव का प्रमेय) चलो $\mu(X) < \infty $ और $ f_n \to f $
बिंदुवार, सभी मापने योग्य (वास्तविक-मूल्यवान)। के लिए $ k, N \geq 1$
तय करना
\[
E_{k,N} = \bigcap_{n \geq N}\Bigl\{x : \abs{f_n(x) - f(x)}
\leq \tfrac1k\Bigr\} .
\]
(ए) उसे निश्चित रूप से दिखाएँ $ k $, $ E_{k,N} \nearrow X $ जैसा $ N \to
\infty $, और निष्कर्ष निकालें $ N_k $ साथ $\mu(X \setminus E_{k,N_k})
\leq \varepsilon2^{-k}$.
(बी) समाप्त करें \emph{ईगोरोव का प्रमेय}: हरएक के लिए $\varepsilon
> 0$ वहाँ एक मापने योग्य है $ A $ साथ $\mu(X\setminus A) \leq
\varepsilon $ ऐसा है कि $ f_n \to f $ \emph{समान रूप से चालू $ A $}
---बिंदुवार अभिसरण एक समान अभिसरण है
मनमाने ढंग से छोटा सेट.
(सी) दिखाएँ कि प्रमेय विफल रहता है $(\R, \lambda)$: गतिशील
धक्कों $ f_n = \mathbf 1_{\intcc n{n+1}}$ बिंदुवार अभिसरण करें
को $0$ लेकिन समान रूप से किसी के पूरक पर नहीं परिमित-माप
तय करना। (ए) का उपयोग कहां किया गया $\mu(X) < \infty $?
\end{exercise}

\begin{problem}[{सप्ताहांत समस्या---शैतान की सीढ़ी,
और $\mathcal B(\R) \subsetneq \mathcal L $}]
\label{pb:b3:measure:1}
हम निर्माण करते हैं \emph{कैंटोर--विटाली फ़ंक्शन} (शैतान का
सीढ़ी), परिवहन के लिए इसका उपयोग करें \omterm{def:b3:measure:measure}{उपाय} पैथोलॉजिकल रूप से, और
एक प्रमेय के साथ निष्कर्ष निकालें जो कोई नरम तर्क नहीं देता: वहाँ
लेब्सेग-मापने योग्य सेट मौजूद हैं जो बोरेल नहीं हैं। संकेतन:
$ C $ कैंटर सेट है, $ C_n $ इसका $ n $-वां चरण ($2^n $ अंतराल
लम्बाई का $3^{-n}$), और हर $ x \in C $ त्रिगुट अंक हैं
$ x = \sum 2b_n3^{-n}$, $ b_n \in \{0,1\}$
(\cref{exo:b3:topology:10}).

\medskip
\noindent\textbf{Part I --- The staircase.} परिभाषित करना $ c_0(x) = x $
और $ c_{n+1}$ से $ c_n $ द्वारा
\[
c_{n+1}(x) = \begin{cases}
\tfrac12\,c_n(3x) & 0 \leq x \leq \tfrac13,\\[2pt]
\tfrac12 & \tfrac13 \leq x \leq \tfrac23,\\[2pt]
\tfrac12 + \tfrac12\,c_n(3x - 2) & \tfrac23 \leq x \leq 1.
\end{cases}
\]
\begin{enumerate}
  \item प्रत्येक को दिखाओ $ c_n $ है \omterm{def:b3:topology:continuity}{निरंतर}, न घटने वाला, साथ
        $ c_n(0) = 0$, $ c_n(1) = 1$, ओर वो
        $\norm{c_{n+1} - c_n}_\infty \leq
        \tfrac12\norm{c_n - c_{n-1}}_\infty $.
  \item उसका निष्कर्ष निकालो $(c_n)$ एक में समान रूप से परिवर्तित हो जाता है \omterm{def:b3:topology:continuity}{निरंतर}
        न घटनेवाला $ c $ साथ $ c(0) = 0$, $ c(1) = 1$
        (\emph{} कैंटोर--विटाली फ़ंक्शन), जो संतुष्ट करता है
        के समान स्व-समान संबंध $ c_{n+1}$ ऊपर।
  \item बताते हैं कि $ c $ प्रत्येक पर स्थिर है \omterm{def:b3:topology:components}{जुड़ा हुआ घटक}
        का $\intcc01\setminus C $, और वह के लिए $ x = \sum_n
        2b_n3^{-n} \in C $, $ c(x) = \sum_n b_n2^{-n}$: द
        सीढ़ी कैंटर अंकों को बाइनरी में पढ़ती है
        (\cref{pb:b3:topology:1}का कार्य $ g $, एकरस बना दिया
        और वैश्विक)।
  \item उसका निष्कर्ष निकालो $ c $ भिन्न है, के साथ $ c' = 0$, पर
        के हर बिंदु $\intcc01\setminus C $: $ c' = 0$
        \emph{$\lambda $-लगभग हर जगह}
        (\cref{ex:b3:measure:cantor}). निष्कर्ष निकालें कि
        कैलकुलस का मौलिक प्रमेय विफल रहता है $ c $:
        \[
        c(1) - c(0) = 1 \neq 0 = \int_0^1 c'(t)\,\dd t
        \]
        (अभिन्न से ऊपर जा रहा है पूर्ण-माप कहां सेट करें
        $ c' = 0$; आशंका \cref{ch:b3:lebesgue}, शून्य सेट
        अभिन्नों को प्रभावित न करें)। की कौन सी परिकल्पना
        $\mathcal C^1$ मौलिक प्रमेय का उल्लंघन हुआ है?
  \item बताते हैं कि $ c(C) = \intcc01$: शून्य सेट $ C $ मैप किया गया है
        \emph{पर} \omterm{def:b3:complete:complete}{पूरा} का एक सेट \omterm{def:b3:measure:measure}{उपाय}.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The crooked \omterm{def:b3:topology:continuity}{होमियोमोर्फिज्म}.} होने देना
$ h(x) = \frac{c(x) + x}{2}$.
\begin{enumerate}[resume]
  \item बताते हैं कि $ h \colon \intcc01 \to \intcc01$ एक है
        \omterm{def:b3:topology:continuity}{होमियोमोर्फिज्म} (सख्ती से बढ़ रहा है, \omterm{def:b3:topology:continuity}{निरंतर},
        विशेषण).
  \item बताते हैं कि $\lambda\bigl(h(\intcc01\setminus C)\bigr) =
        \tfrac12$: लंबाई के प्रत्येक अंतराल पर $\ell $, $ h $ के रूप में कार्य करता है
        ढलान का एफाइन मानचित्र $\tfrac12$, और अंतराल कुल हैं
        लंबाई $1$.
  \item परिणाम निकालना $\lambda\bigl(h(C)\bigr) = \tfrac12$: द
        होम्योमॉर्फिक एक शून्य सेट की छवि सकारात्मक हो सकती है
        \omterm{def:b3:measure:measure}{उपाय}. (यह कहां भोले अंतर्ज्ञान का खंडन करता है
        ``आकार'' के बारे में?)
\end{enumerate}

\medskip
\noindent\textbf{Part III --- A measurable set that is not
Borel.}
\begin{enumerate}[resume]
  \item द्वारा \cref{exo:b3:measure:9}, एक गैर-मापने योग्य चुनें $ W
        \subseteq h(C)$. बताते हैं कि $ Z = h^{-1}(W) \subseteq C $
        लेब्सग्यू-मापन योग्य है। \emph{(यह शून्य का उपसमुच्चय है
        सेट; पूर्णता, \cref{thm:b3:measure:caratheodory}.)}
  \item दिखाएँ कि बोरेल की पूर्व छवि a के अंतर्गत सेट की गई है
        सतत मानचित्र बोरेल है. \emph{(अच्छे सेट सिद्धांत:
        $\{B : h^{-1}(B) \in \mathcal B\}$ एक है
        $\sigma $-बीजगणित युक्त खुले सेट --- ध्यान रखें
        मानचित्र की दिशा.)}
  \item समाप्त करें कि $ Z $ है \emph{नहीं} बोरेल: यदि ऐसा होता,
        $ W = (h^{-1})^{-1}(Z)$ बोरेल होगा (प्रश्न लागू करें
        10 से \omterm{def:b3:topology:continuity}{निरंतर} $ h^{-1}$), इसलिए मापने योग्य ---
        विरोधाभास. इसलिए
        \[
        \boxed{\ \mathcal B(\R) \subsetneq \mathcal L\ }
        \]
        और पूर्णता वास्तव में बोरेल दुनिया का विस्तार करता है।
  \item एक लेब्सग्यू-मापने योग्य फ़ंक्शन प्रदर्शित करें $ g $ और ए
        \omterm{def:b3:topology:continuity}{निरंतर} समारोह $\varphi $ ऐसा है कि $ g \circ
        \varphi $ लेब्सग्यू-मापन योग्य नहीं है: मापनीयता,
        विपरीत निरंतरता, रचना नहीं करता. \emph{(लेना $ g =
        \mathbf 1_Z $ और $\varphi = h^{-1}$, का अनुमान
        मापने योग्य कार्यों की परिभाषा
        \cref{ch:b3:lebesgue}: बोरेल सेट की पूर्वछवियाँ हैं
        लेब्सेग सेट. कहां किसको लेकर सावधान रहना होगा
        $\sigma $-बीजगणित का प्रयोग लक्ष्य पर किया जाता है?)}
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Epilogue.}
\begin{enumerate}[resume]
  \item निम्नलिखित वर्गों को सख्त समावेशन के \omterm{def:b3:topology:basis}{आधार} पर क्रमबद्ध करें और
        इसमें से एक उदाहरण के साथ प्रत्येक सख्ती को उचित ठहराएँ
        अध्याय और उसकी समस्या: गणनीय समुच्चय; बोरेल शून्य
        सेट; लेब्सेग नल सेट; बोरेल सेट; लेब्सेग सेट;
        मनमाना सेट.
\end{enumerate}

\medskip
\noindent\textbf{Part V --- The Cantor \omterm{def:b3:measure:measure}{उपाय}: mass on a null
set.} सीढ़ी एक का वितरण कार्य है
उल्लेखनीय \omterm{def:b3:measure:measure}{उपाय}, जिसे अब हम इस अध्याय के साथ बना रहे हैं
उपकरण.
\begin{enumerate}[resume]
  \item (लेब्सग्यू--स्टिल्टजेस, अस्तित्व) आइए $ F\colon\R\to\R $
        न घटनेवाला हो, \omterm{def:b3:topology:continuity}{निरंतर}, घिरा हुआ। आधे खुले पर
        अंतराल परिभाषित करते हैं $\rho\bigl(\intoc ab\bigr) = F(b) -
        F(a)$ और के लिए $ A \subseteq \R $,
        \[
        \mu_F^*(A) = \inf\Bigl\{\sum_k\bigl(F(b_k) -
        F(a_k)\bigr) : A \subseteq
        \bigcup_k\intoc{a_k}{b_k}\Bigr\} .
        \]
        बताते हैं कि $\mu_F^*$ एक \omterm{def:b3:measure:outer}{बाहरी माप} ओर वो
        $\mu_F^*\bigl(\intoc ab\bigr) = F(b) - F(a)$
        \emph{(कॉम्पैक्टनेस तर्क का अनुकरण करें
        \cref{thm:b3:measure:lebesgue}, प्रत्येक को बड़ा करना
        $\intoc{a_k}{b_k}$ एक खुले अंतराल पर
        $ F $-लागत $\leq \varepsilon2^{-k}$ --- कहाँ है
        निरंतरता का $ F $ इस्तेमाल किया गया?)}.
  \item दिखाएँ कि प्रत्येक बोरेल सेट है
        $\mu_F^*$-कैराथ में मापने योग्य\'दुर्गंधयुक्त अनुभूति
        \emph{(जैसा कि लेबेस्ग मामले में, यह परीक्षण करने के लिए पर्याप्त है
        आधी पंक्तियाँ; के प्रमाण का पालन करें
        \cref{thm:b3:measure:caratheodory}का आवेदन)}, ताकि $\mu_F = \mu_F^*$ तक सीमित $\mathcal B(\R)$
        एक है \omterm{def:b3:measure:measure}{उपाय} साथ $\mu_F(\intoc ab) = F(b) - F(a)$:
        द \emph{लेबेस्ग्यू--स्टिल्टजेस \omterm{def:b3:measure:measure}{उपाय}} का $ F $.
  \item इसे सीढ़ी पर लगाएं ($ F = c $ द्वारा बढ़ाया गया $0$
        पर $\R_-$ और $1$ पर $\intco1\infty $): द
        \emph{कैंटर \omterm{def:b3:measure:measure}{उपाय}} $\mu $. दिखाओ $\mu(\R) = 1$, वह
        कैंटर सेट का हर गैप है $\mu $-व्यर्थ ($ c $ है
        वहां स्थिर), और निष्कर्ष निकालें
        \[
        \mu(C) = 1, \qquad \lambda(C) = 0 :
        \]
        $\mu $ और $\lambda $ असंयुक्त वाहकों पर रहते हैं ($ C $ और
        इसका पूरक)। दो पैमाने इस पद पर हैं
        बुलाया \emph{परस्पर एकवचन}, लिखा हुआ $\mu \perp
        \lambda $.
  \item बताते हैं कि $\mu $ कोई परमाणु नहीं है: $\mu(\{x\}) = 0$ हरएक के लिए $ x $ \emph{(निरंतरता का $ c $)}. परमाणुहीन
        संभाव्यता \omterm{def:b3:measure:measure}{उपाय} लेब्सेग-नल द्वारा ले जाया गया कॉम्पैक्ट:
        केवल से तुलना करें पैमाने अब तक देखा.
  \item (छिपाकर सिक्का उछालते हुए) एक शब्द के लिए $(\varepsilon_1,
        \dots, \varepsilon_m) \in \{0,1\}^m $, होने देना
        $ C_{\varepsilon}$ का सेट हो $ x \in C $ जिसका टर्नरी
        अंक संतुष्ट करते हैं $ b_i(x) = \varepsilon_i $ के लिए $ i \leq m $
        (निम्न में से एक $2^m $ गहराई के कैंटर टुकड़े $ m $). दिखाओ
        $\mu(C_\varepsilon) = 2^{-m}$ \emph{(सीढ़ी
        चढ़ता है $2^{-m}$ उस टुकड़े के पार: भाग I का उपयोग करें,
        प्रश्न 3)}. कैंटर \omterm{def:b3:measure:measure}{उपाय} का कानून है
        टर्नरी में पढ़े गए सिक्कों के निष्पक्ष उछाल का अनंत क्रम
        --- \cref{ch:b3:probability} इसे सटीक बना देगा.
  \item आत्म-समानता सिद्ध करें: प्रत्येक बोरेल के लिए $ A $,
        \[
        \mu(A) = \tfrac12\,\mu(3A) +
        \tfrac12\,\mu(3A - 2) ,
        \]
        कहाँ $3A - 2 = \{3x - 2 : x \in A\}$ \emph{(इसे जांचें
        उत्पन्न करने वाले अंतराल पर $\intoc ab $ के माध्यम से
        के स्व-समान संबंध $ c $, फिर विशिष्टता का आह्वान करें,
        \cref{thm:b3:measure:uniqueness})}.
  \item वह प्रतिबिम्ब दिखाओ $ s(x) = 1 - x $ बरकरार रखता है
        $\mu $: $\mu(s(A)) = \mu(A)$ \emph{(के जरिए $ c(1 - x) = 1 -
        c(x)$, जो की समरूपता से अनुसरण करता है
        निर्माण --- इसे सिद्ध करें)}.
  \item के पहले दो क्षणों की गणना करें $\mu $, यानी\ एक का
        यादृच्छिक बिंदु $ X $ कानून के साथ $\mu $ (अभिन्न हो सकते हैं
        गहराई से अधिक रकम की सीमा के रूप में संभाला गया-$ m $ टुकड़े,
        प्रत्याशित \cref{ch:b3:lebesgue}): समरूपता देता है
        $\int x\,\dd\mu = \frac12$, और आत्म-समानता देता है
        \[
        \int x^2\,\dd\mu = \frac38,
        \qquad\text{hence}\qquad
        \operatorname{Var}(X) = \frac18 .
        \]
        पर समान कानून से तुलना करें $\intcc01$ (विचरण
        $\frac1{12}$): कैंटर द्रव्यमान, किनारों पर धकेल दिया गया,
        फैलता है \emph{अधिक}.
  \item दिखाएँ कि (संस्थानिक) \emph{सहायता} का $\mu $
        --- पूर्ण का सबसे छोटा \omterm{def:b3:topology:interior}{बंद} सेट \omterm{def:b3:measure:measure}{उपाय} --- है
        बिलकुल $ C $.
  \item (संश्लेषण) सीढ़ी $ c $ है \omterm{def:b3:topology:continuity}{निरंतर} और
        नॉनडिक्रीजिंग अभी भी मौलिक प्रमेय को विफल करता है
        कैलकुलस (भाग I); \omterm{def:b3:measure:measure}{उपाय} $\mu_c $ एक है
        संभाव्यता, परमाणु रहित, एकवचन के संबंध में
        $\lambda $. एक संक्षिप्त पैराग्राफ में बताएं कि ये कैसे हैं
        एक घटना के दो पहलू, और \omterm{def:b3:groups:normal}{सामान्य} बताएं
        नैतिक: गैर-घटते कार्य इसके अनुरूप हैं पैमाने
        ($ F \leftrightarrow \mu_F $), भिन्नता ए.ई.\
        ``बिल्कुल'' से मेल खाता है \omterm{def:b3:topology:continuity}{निरंतर} भाग'', और
        $ c $ मानक गवाह है कि ए \omterm{def:b3:topology:continuity}{निरंतर} $ F $ कर सकते हैं
        ले जाना \emph{नहीं} बिल्कुल \omterm{def:b3:topology:continuity}{निरंतर} बिल्कुल हिस्सा.
  \item (तीव्र मापांक निरंतरता) होने देना $ s =
        \frac{\ln 2}{\ln 3}$. बताते हैं कि $ c $ एच है\"पुराने
        \omterm{def:b3:topology:continuity}{निरंतर} प्रतिपादक का $ s $:
        \[
        \abs{c(x) - c(y)} \leq 4\,\abs{x - y}^{s}
        \qquad (x, y \in \intcc01),
        \]
        और वह कोई प्रतिपादक नहीं है $ t > s $ स्थानीय स्तर पर भी काम कर सकते हैं।
        माप-सैद्धांतिक रूप निकालें: प्रत्येक के लिए $ x $ और
        प्रत्येक $ r \in \intoc01$,
        \[
        \mu\bigl(\intcc{x - r}{x + r}\bigr) \leq 8\,r^{s} .
        \]
        \emph{(गहराई की तुलना करें-$ m $ पैमाने के साथ त्रियादिक ग्रिड
        का $\abs{x - y}$; प्रश्न 18 पार चढ़ना बताता है
        प्रत्येक टुकड़ा. प्रतिपादक $ s $ है \omterm{def:b3:topology:hausdorff}{हॉसडॉर्फ़} आयाम
        का $ C $, जैसा कि बाद के पाठ्यक्रम कहेंगे।)}
  \item (स्वयं-समानता की विशेषता है $\mu $) सिद्ध करें
        प्रश्न 19 का व्युत्क्रम: यदि $\nu $ एक संभावना है
        \omterm{def:b3:measure:measure}{उपाय} पर $\mathcal B(\R)$ द्वारा किया गया $\intcc01$ और
        संतोषजनक
        \[
        \nu(A) = \tfrac12\,\nu(3A) + \tfrac12\,\nu(3A - 2)
        \qquad (A \in \mathcal B(\R)),
        \]
        तब $\nu = \mu $. \emph{(संबंध को पुनरावृत्त करें $ m $
        फैलने का समय $\nu $ ऊपर $2^m $ गहराई-$ m $ कैंटर
        टुकड़े, अनुमान $\nu(\intoc ab)$ की गिनती के विरुद्ध
        अंदर के टुकड़े $\intoc ab $, और चलो $ m \to \infty$;
        के साथ खत्म करें \cref{thm:b3:measure:uniqueness}.)}
\end{enumerate}
\end{problem}
