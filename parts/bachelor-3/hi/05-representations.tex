\chapter{परिमित समूहों का प्रतिनिधित्व}\label{ch:b3:representations}

किसी अमूर्त समूह को समझने के लिए, उसे एक सदिश समष्टि पर कार्य करने को कहें
और रैखिक बीजगणित का उपयोग करें --- eigenvalues, निशान, आंतरिक उत्पाद ---
\omterm{def:b3:groups:action}{कार्रवाई} पर. यह कार्यक्रम, \emph{\omterm{def:b3:representations:rep}{प्रतिनिधित्व} लिखित}, है
परिमित समूहों के लिए आश्चर्यजनक रूप से प्रभावी $\C $: प्रत्येक
\omterm{def:b3:representations:rep}{प्रतिनिधित्व} में विभाजित हो जाता है \omterm{def:b3:representations:rep}{अलघुकरणीय} वाले (मास्चके), द
\omterm{def:b3:representations:rep}{अलघुकरणीय} लोगों को उनके द्वारा पिन किया गया है \emph{अक्षर}
(निशान), और अक्षर ऑर्थोगोनैलिटी संबंधों को संतुष्ट करें
गणनाओं को यांत्रिक बनाएं। अध्याय इस गणना का निर्माण करता है
और इसकी पहली उत्कृष्ट कृतियाँ --- \omterm{def:b3:representations:character}{चरित्र} सारणी छोटे का
समूह --- और सप्ताहांत की समस्या एक प्रमेय से कहीं आगे निकल जाती है
नंगे समूह सिद्धांत की पहुंच: बर्नसाइड $ p^aq^b $ प्रमेय.
लगातार, $ G $ एक परिमित समूह है और सभी सदिश स्थान हैं
परिमित-आयामी ओवर $\C $.

\section{अभ्यावेदन, माश्के, शूर}

\begin{definition}\label{def:b3:representations:rep}
ए \emph{प्रतिनिधित्व}\index{प्रतिनिधित्व} का $ G $ एक है
रूपवाद $\rho \colon G \to GL(V)$ एक के लिए $\C $-सदिश स्थल $ V $;
$\dim V $ इसका है \emph{\omterm{def:b3:galois:extension}{डिग्री}}. एक उपस्थान $ W \subseteq V $ है
\emph{अचल} अगर $\rho(g)W \subseteq W $ सभी के लिए $ g $;
प्रतिबंध बनाता है $ W $ ए \emph{उपप्रतिनिधित्व}. $\rho $ है
\emph{अलघुकरणीय}\index{अलघुकरणीय} अगर $ V \neq
0$ और इसके एकमात्र अपरिवर्तनीय उपस्थान हैं $0$ और $ V $. ए
\emph{आकारिता} बीच में $(\rho, V)$ और $(\sigma, W)$ एक रैखिक है
$ f\colon V \to W $ साथ $ f\rho(g) = \sigma(g)f $ सभी के लिए $ g $
(समतुल्यता); द्विभाजित $ f $ हैं \emph{समरूपताएँ}.
\end{definition}

\begin{example}\label{ex:b3:representations:examples}
(एक श्रेणी $1$: रूपवाद $ G \to \C^\times $.
(बी) द \emph{नियमित प्रतिनिधित्व}\index{नियमित प्रतिनिधित्व}: $ V = \C^G $ आधार के साथ $(e_h)_{h \in G}$,
$\rho(g)e_h = e_{gh}$; \omterm{def:b3:galois:extension}{डिग्री} $\abs G $.
(सी) की एक क्रमपरिवर्तन \omterm{def:b3:groups:action}{कार्रवाई} $ G $ एक सीमित सेट पर $ X $ देता है
\emph{परिवर्तन \omterm{def:b3:representations:rep}{प्रतिनिधित्व}} पर $\C^X $: $\rho(g)e_x =
e_{g\cdot x}$.
(डी) $ S_n $ पर कार्य करता है $\C^n $ निर्देशांकों को क्रमपरिवर्तित करके; हाइपरप्लेन
$\{\sum x_i = 0\}$ अपरिवर्तनीय है: \emph{मानक
\omterm{def:b3:representations:rep}{प्रतिनिधित्व}}, \omterm{def:b3:galois:extension}{डिग्री} का $ n - 1$.
\end{example}

\begin{theorem}[माश्के]\label{thm:b3:representations:maschke}
प्रत्येक अपरिवर्तनीय उपस्थान $ W $ एक का \omterm{def:b3:representations:rep}{प्रतिनिधित्व} $(V, \rho)$
एक अपरिवर्तनीय पूरक को स्वीकार करता है। फलस्वरूप प्रत्येक \omterm{def:b3:representations:rep}{प्रतिनिधित्व}
का सीधा योग है \omterm{def:b3:representations:rep}{अलघुकरणीय} वाले
(\emph{अर्धसरलता}).\index{माश्के का प्रमेय}
\end{theorem}

\begin{proof}
होने देना $ p \colon V \to V $ होना \emph{कोई} छवि के साथ प्रक्षेपण $ W $
(कोई भी पूरक चुनें)। इसे समूह पर औसत करें:
\[
\tilde p = \frac1{\abs G}\sum_{g \in G} \rho(g)\,p\,\rho(g)^{-1}.
\]
प्रत्येक पद मानचित्र $ V $ में $ W $ ($ W $ अपरिवर्तनीय), और ठीक करता है $ W $
बिंदुवार: के लिए $ w \in W $, $\rho(g)^{-1}w \in W $, $ p $ इसे ठीक करता है,
और $\rho(g)$ इसे वापस लाता है --- तो $\tilde p $ फिर से एक है
पर प्रक्षेपण $ W $. यह समतुल्य है: के लिए $ h \in G $,
$\rho(h)\tilde p\rho(h)^{-1}$ उसी राशि को पुनः अनुक्रमित करता है। इस तरह $\ker
\tilde p $ का एक अपरिवर्तनीय पूरक है $ W $. पर पुनरावृति
सारांश (परिमित आयाम) विघटित हो जाता है $ V $ में इरेड्यूसिबल्स.
\end{proof}

\begin{theorem}[शूर की लेम्मा]\label{thm:b3:representations:schur}
होने देना $ f \colon V \to W $ का एक रूपवाद हो \emph{\omterm{def:b3:representations:rep}{अलघुकरणीय}}
अभ्यावेदन. तब $ f = 0$ या $ f $ एक समरूपता है; और यदि
$(V,\rho) = (W,\sigma)$, तब $ f = \lambda\,\mathrm{id}$ कुछ के लिए
$\lambda \in \C $. इस तरह
$\dim\operatorname{Hom}_G(V, W)$ है $1$ अगर $ V \cong W $, $0$
अन्यथा।\index{शूर की लेम्मा}
\end{theorem}

\begin{proof}
$\ker f $ और $\operatorname{im} f $ अपरिवर्तनीय (समतुल्य) हैं,
तो प्रत्येक है $0$ या सब कुछ: या तो $ f = 0$, या $ f $ इंजेक्शन है
पूरी छवि के साथ. अगर $ V = W $: $ f $ एक eigenvalue है $\lambda $ ($\C $
\omterm{def:b3:galois:closure}{बीजगणितीय रूप से बंद}); $ f - \lambda\,\mathrm{id}$ एक है
गैर इंजेक्शन आकारिकी $ V \to V $, इस तरह $0$. आयाम के लिए
गिनें कब $ V \cong W $: एक समरूपता को ठीक करना $ u $, कोई रूपवाद
$ f $ एंडोमोर्फिज्म देता है $ u^{-1}f = \lambda\,\mathrm{id}$: $ f =
\lambda u $.
\end{proof}

\section{वर्ण और रूढ़िवादिता}

\begin{definition}\label{def:b3:representations:character}
 \emph{चरित्र}\index{चरित्र} का $(V, \rho)$ है
$\chi_\rho(g) = \operatorname{tr}\rho(g)$. यह संतुष्ट करता है
$\chi_\rho(e) = \dim V $, $\chi_\rho(hgh^{-1}) = \chi_\rho(g)$
(निशान संयुग्मन-अपरिवर्तनीय हैं): वर्ण हैं \emph{कक्षा
कार्य}\index{कक्षा
कार्य} --- अंतरिक्ष के तत्व
$\mathcal{CF}(G)$ कार्यों का निरंतर चालू रहना \omterm{ex:b3:groups:actions}{संयुग्मता वर्ग},
हर्मिटियन आंतरिक उत्पाद से सुसज्जित
\[
\langle \varphi, \psi\rangle = \frac1{\abs G}\sum_{g \in G}
\overline{\varphi(g)}\,\psi(g).
\]
\end{definition}

\begin{proposition}\label{prop:b3:representations:charbasics}
$\rho(g)$ एकता के eigenvalues जड़ों के साथ विकर्णीय है;
$\chi_\rho(g^{-1}) = \overline{\chi_\rho(g)}$, और
$\abs{\chi_\rho(g)} \leq \chi_\rho(e)$ समानता के साथ यदि
$\rho(g)$ एक अदिश राशि है. अक्षर प्रत्यक्ष रकम जोड़ें: $\chi_{V
\oplus W} = \chi_V + \chi_W $.
\end{proposition}

\begin{proof}
$\rho(g)^N = \mathrm{id}$ के लिए $ N = \abs G $ (लैग्रेंज): $\rho(g)$
सत्यानाश कर देता है $ X^N - 1$, \omterm{def:b3:groups:simple}{सरल} जड़ों से विभाजित, तो यह है
eigenvalues के साथ विकर्ण $\lambda_i \in \mu_N $. तब
$\chi(g^{-1}) = \sum \lambda_i^{-1} = \sum\bar\lambda_i =
\overline{\chi(g)}$, और $\abs{\chi(g)} = \abs{\sum\lambda_i}
\leq \dim V $, त्रिभुज असमानता में समानता के साथ यदि सभी
$\lambda_i $ बराबर हैं, यानी\ $\rho(g) = \lambda\,\mathrm{id}$.
प्रत्यक्ष योगों पर योगात्मकता: निशानों को अवरुद्ध करें।
\end{proof}

\begin{lemma}\label{lem:b3:representations:homtrace}
होने देना $(V, \rho)$, $(W, \sigma)$ होना अभ्यावेदन. औसत
ऑपरेटर चालू $\operatorname{Hom}(W, V)$,
\[
c(A) = \frac1{\abs G}\sum_{g}\rho(g)\,A\,\sigma(g)^{-1},
\]
पर एक प्रक्षेपण है $\operatorname{Hom}_G(W, V)$, और के लिए
नक्शा $\Phi_g \colon A \mapsto \rho(g)A\sigma(g)^{-1}$ एक के पास है
$\operatorname{tr}\Phi_g =
\chi_\rho(g)\,\overline{\chi_\sigma(g)}$.
\end{lemma}

\begin{proof}
$ c(A)$ समतुल्य है (मास्चके के अनुसार योग को पुनः अनुक्रमित करें), और $ c $
समतुल्य मानचित्रों को ठीक करता है (प्रत्येक पद बराबर होता है $ A $): $ c $ एक है
छवि के साथ प्रक्षेपण $\operatorname{Hom}_G(W, V)$. ट्रेस: में
आधार, $\Phi_g(A) = BAC $ साथ $ B = \rho(g)$, $ C = \sigma(g)^{-1}$;
आधार पर $(E_{kl})$ मैट्रिक्स का, $ BE_{kl}C = \sum_{m,n}
b_{mk}c_{ln}E_{mn}$, तो का गुणांक $ E_{kl}$ में
$\Phi_g(E_{kl})$ है $ b_{kk}c_{ll}$: $\operatorname{tr}\Phi_g =
\sum_{k,l}b_{kk}c_{ll} = \operatorname{tr}(B)
\operatorname{tr}(C) = \chi_\rho(g)\chi_\sigma(g^{-1})$, और
$\chi_\sigma(g^{-1}) = \overline{\chi_\sigma(g)}$.
\end{proof}

\begin{theorem}[पहला रूढ़िवादिता संबंध]
\label{thm:b3:representations:orthogonality}
होने देना $\rho, \sigma $ होना \emph{\omterm{def:b3:representations:rep}{अलघुकरणीय}}. तब\index{रूढ़िवादिता
पात्रों का}
\[
\langle\chi_\sigma, \chi_\rho\rangle =
\begin{cases}
1 & \text{if } \rho \cong \sigma,\\
0 & \text{otherwise:}
\end{cases}
\]
 \omterm{def:b3:representations:rep}{अलघुकरणीय} अक्षर में एक असामान्य परिवार बनाएं
$\mathcal{CF}(G)$.
\end{theorem}

\begin{proof}
एक प्रक्षेपण का निशान उसकी छवि का आयाम है:
\[
\dim\operatorname{Hom}_G(W, V) = \operatorname{tr} c
= \frac1{\abs G}\sum_g \operatorname{tr}\Phi_g
= \frac1{\abs G}\sum_g \chi_\rho(g)\overline{\chi_\sigma(g)}
= \langle \chi_\sigma, \chi_\rho\rangle,
\]
और शूर की लेम्मा बायीं ओर का मूल्यांकन करती है $\delta_{\rho \cong
\sigma}$.
\end{proof}

\begin{corollary}\label{cor:b3:representations:multiplicity}
विघटित $ V \cong \bigoplus_i V_i^{\oplus m_i}$ अलग में
इरेड्यूसिबल्स ($ V_i \not\cong V_j $). तब $ m_i = \langle
\chi_{V_i}, \chi_V\rangle $: बहुलताएँ --- इसलिए
\omterm{def:b3:representations:rep}{प्रतिनिधित्व} समरूपता तक --- द्वारा निर्धारित होते हैं
\omterm{def:b3:representations:character}{चरित्र}. इसके अतिरिक्त $\langle \chi_V, \chi_V\rangle = \sum_i
m_i^2$; विशेष रूप से $ V $ है \omterm{def:b3:representations:rep}{अलघुकरणीय} आईएफएफ $\langle\chi_V,
\chi_V\rangle = 1$.
\end{corollary}

\begin{proof}
$\chi_V = \sum m_i\chi_{V_i}$
(\cref{prop:b3:representations:charbasics}); आंतरिक उत्पाद लें
प्रत्येक के साथ $\chi_{V_i}$ और ऑर्थोनॉर्मलिटी का उपयोग करें। दो
अभ्यावेदन बराबर के साथ अक्षर समान बहुलता है,
इसलिए समरूपी हैं।
\end{proof}

\begin{theorem}[नियमित प्रतिनिधित्व]
\label{thm:b3:representations:regular}
होने देना $\chi_1, \dots, \chi_r $ विशिष्ट बनें \omterm{def:b3:representations:rep}{अलघुकरणीय}
अक्षर, डिग्रियों का $ n_i = \chi_i(e)$. \omterm{ex:b3:representations:examples}{नियमित
प्रतिनिधित्व} बहुलता के साथ विघटित होता है $ m_i = n_i $;
फलस्वरूप
\[
\sum_{i=1}^{r} n_i^2 = \abs G,
\qquad
\sum_i n_i\chi_i(g) = 0 \quad (g \neq e).
\]
\end{theorem}

\begin{proof}
नियमित \omterm{def:b3:representations:character}{चरित्र}: $\chi_{\mathrm{reg}}(g) =
\#\{h : gh = h\}$, जो है $\abs G $ के लिए $ g = e $ और $0$
अन्यथा। इसलिए $ m_i = \langle \chi_i,\chi_{\mathrm{reg}}\rangle =
\frac1{\abs G}\overline{\chi_i(e)}\,\abs G = n_i $. का मूल्यांकन
$\chi_{\mathrm{reg}} = \sum n_i\chi_i $ पर $ e $ और कम से $ g \neq e $
दो प्रदर्शित पहचान देता है।
\end{proof}

\begin{theorem}\label{thm:b3:representations:numberirr}
 \omterm{def:b3:representations:rep}{अलघुकरणीय} अक्षर एक ऑर्थोनॉर्मल बनाएं \emph{आधार} का
$\mathcal{CF}(G)$: की संख्या अघुलनशील अभ्यावेदन
की संख्या के बराबर है \omterm{ex:b3:groups:actions}{संयुग्मता वर्ग} का $ G $.
\end{theorem}

\begin{proof}
केवल पूर्णता शेष है: चलो $ f \in \mathcal{CF}(G)$ हो
प्रत्येक के लिए ओर्थोगोनल $\chi_i $; हम दिखाते हैं $ f = 0$. एक के लिए
\omterm{def:b3:representations:rep}{प्रतिनिधित्व} $(V,\rho)$, तय करना $ T_{f,\rho} = \frac{1}{\abs G}
\sum_g \overline{f(g)}\,\rho(g)$. यह समतुल्य है: के लिए $ h \in
G $,
\[
\rho(h)T_{f,\rho}\rho(h)^{-1} = \frac1{\abs G}\sum_g
\overline{f(g)}\rho(hgh^{-1})
= \frac1{\abs G}\sum_{g'}\overline{f(h^{-1}g'h)}\rho(g') =
T_{f,\rho}
\]
($ f $ एक है वर्ग समारोह). अगर $\rho $ है \omterm{def:b3:representations:rep}{अलघुकरणीय} \omterm{def:b3:galois:extension}{डिग्री} का
$ n $, शूर देता है $ T_{f,\rho} = \lambda\,\mathrm{id}$ साथ
\[
\lambda = \frac{\operatorname{tr}T_{f,\rho}}{n}
= \frac{1}{n\abs G}\sum_g \overline{f(g)}\,\chi_\rho(g)
= \frac1n\,\langle f, \chi_\rho\rangle = 0 .
\]
इसलिए $ T_{f,\rho} = 0$ सभी पर \omterm{def:b3:representations:rep}{अलघुकरणीय}, इसलिए (प्रत्यक्ष योग) पर
\emph{प्रत्येक} \omterm{def:b3:representations:rep}{प्रतिनिधित्व} --- विशेष रूप से नियमित पर।
इसे आधार वेक्टर पर लागू करें $ e_e $: $0 = T_{f,\mathrm{reg}}e_e =
\frac1{\abs G}\sum_g\overline{f(g)}e_g $, हर किसी को मजबूर कर रहा है
$\overline{f(g)} = 0$. इस प्रकार ऑर्थोनॉर्मल परिवार
$(\chi_i)$ तक फैला $\mathcal{CF}(G)$, जिसका आयाम है
की संख्या \omterm{ex:b3:groups:actions}{संयुग्मता वर्ग}.
\end{proof}

\begin{corollary}[स्तम्भ रूढ़िवादिता]
\label{cor:b3:representations:column}
के लिए $ g, h \in G $:
\[
\sum_{i=1}^{r}\overline{\chi_i(g)}\,\chi_i(h) =
\begin{cases}
\abs{Z_G(g)} & \text{if } g, h \text{ are conjugate},\\
0 & \text{otherwise.}
\end{cases}
\]
\end{corollary}

\begin{proof}
होने देना $ g_1, \dots, g_r $ वर्गों का \omterm{def:b3:representations:rep}{प्रतिनिधित्व} करें, $ c_j $ \omterm{def:b3:groups:action}{कक्षा}
आकार. $ r \times r $ मैट्रिक्स $ U_{ij} =
\sqrt{c_j/\abs G}\;\chi_i(g_j)$ लम्बवत पंक्तियाँ हैं
(\cref{thm:b3:representations:orthogonality} कक्षावार लिखा गया:
$\sum_j \frac{c_j}{\abs G}\chi_i(g_j)\overline{\chi_{i'}(g_j)} =
\delta_{ii'}$), यानी\ $ UU^* = I $; के साथ एक वर्ग मैट्रिक्स $ UU^* =
I $ भी है $ U^*U = I $: कॉलम ऑर्थोनॉर्मल हैं, जो अनपैक होते हैं
प्रदर्शित पहचान ($\abs G/c_j = \abs{Z_G(g_j)}$, द
कक्षा--संयुग्मन के लिए स्थिरीकरण संबंध)।
\end{proof}

\begin{proposition}[एक आयामी अक्षर; उठाना]
\label{prop:b3:representations:onedim}
(ए) $ G $ एबेलियन है यदि यह सब है अघुलनशील अभ्यावेदन है
\omterm{def:b3:galois:extension}{डिग्री} $1$; \omterm{def:b3:galois:extension}{डिग्री} की संख्या-$1$ अक्षर किसी का $ G $ है
$[G : D(G)]$ (वे हैं अक्षर एबेलियनाइजेशन का)।
(बी) यदि $ N \trianglelefteq G $, द अघुलनशील अभ्यावेदन का
$ G/N $ लिफ्ट (साथ लिखें) $ G \to G/N $) बिल्कुल उन्हीं के लिए
अघुलनशील अभ्यावेदन का $ G $ जिसके कर्नेल में शामिल है $ N $.
\end{proposition}

\begin{proof}
(ए) यदि $ G $ एबेलियन है, प्रत्येक वर्ग एक सिंगलटन है: $ r = \abs G $,
और $\sum n_i^2 = \abs G $ सभी को मजबूर करता है $ n_i = 1$; इसके विपरीत यदि
सब $ n_i = 1$, द \omterm{ex:b3:representations:examples}{नियमित प्रतिनिधित्व} का योग है
एक-आयामी वाले, इसलिए $\rho_{\mathrm{reg}}(G)$ है
एक साथ विकर्णीय, इसलिए क्रमविनिमेय, और
$\rho_{\mathrm{reg}}$ वफादार है: $ G $ एबेलियन। डिग्री-$1$
अभ्यावेदन रूपवाद हैं $ G \to \C^\times $ एबेलियन के साथ
लक्ष्य: वे कारक बनते हैं $ G^{\mathrm{ab}} = G/D(G)$
(\cref{exo:b3:groups:9}), और विशिष्ट अक्षर एबेलियन का
$ G^{\mathrm{ab}}$ संख्या $\abs{G^{\mathrm{ab}}}$ (इसमें वह है
कई कक्षाएँ, सभी डिग्रियाँ $1$). (बी) के साथ रचना
प्रक्षेपण अपरिवर्तनीयता (अपरिवर्तनीय उप-स्थान) को संरक्षित करता है
अनुरूप), और ए \omterm{def:b3:representations:rep}{प्रतिनिधित्व} पर तुच्छ $ N $ के माध्यम से कारक
भागफल (\cref{thm:b3:groups:firstiso}).
\end{proof}

\section{चरित्र सारणी}

\begin{definition}\label{def:b3:representations:table}
 \emph{चरित्र तालिका}\index{चरित्र तालिका} का $ G $ है
$ r \times r $ मैट्रिक्स $\bigl(\chi_i(g_j)\bigr)$: द्वारा अनुक्रमित पंक्तियाँ
\omterm{def:b3:representations:rep}{अलघुकरणीय} अक्षर, कॉलम द्वारा \omterm{ex:b3:groups:actions}{संयुग्मता वर्ग} (उनके साथ
आकार प्रदर्शित)। भारित उत्पाद के लिए पंक्तियाँ लम्बवत् हैं,
कॉलम ऑर्थोगोनल (\cref{cor:b3:representations:column}): द
तालिका गंभीर रूप से अतिनिर्धारित है, जो इसे बनाती है
गणना योग्य.
\end{definition}

\begin{example}[की मेज $ S_3$]\label{ex:b3:representations:s3}
कक्षाएं: $ e $ (आकार 1), स्थानान्तरण (3), $3$-चक्र (2); इसलिए $ r
= 3$ इरेड्यूसिबल्स, डिग्रियों का $ n_i $ साथ $\sum n_i^2 = 6$: $1, 1,
2$. \omterm{def:b3:galois:extension}{डिग्री} $1$: मामूली $\mathbf 1$ और हस्ताक्षर $\varepsilon $.
अंतिम पंक्ति स्तंभ ऑर्थोगोनैलिटी (या से) का अनुसरण करती है
$\chi_{\mathrm{std}} = \chi_{\mathrm{perm}} - \mathbf 1$):
\[
\begin{array}{c|ccc}
S_3 & e & (1\,2)\ [3] & (1\,2\,3)\ [2]\\
\hline
\mathbf 1 & 1 & 1 & 1\\
\varepsilon & 1 & -1 & 1\\
\chi_{\mathrm{std}} & 2 & 0 & -1
\end{array}
\]
जाँच करना: $\langle\chi_{\mathrm{std}},\chi_{\mathrm{std}}\rangle =
\frac{1}{6}(4 + 0 + 2) = 1$: \omterm{def:b3:representations:rep}{अलघुकरणीय}.
\end{example}

\begin{example}[की मेज $ S_4$]\label{ex:b3:representations:s4}
कक्षाएं: $ e $ [1], ट्रांसपोज़िशन [6], डबल ट्रांसपोज़िशन [3],
$3$-चक्र [8], $4$-चक्र [6]: पाँच इरेड्यूसिबल्स, $\sum n_i^2 =
24$ दो \omterm{def:b3:galois:extension}{डिग्री} के साथ-$1$'एस ($\mathbf 1, \varepsilon $; $[S_4 :
D(S_4)] = [S_4 : A_4] = 2$): \omterm{def:b3:galois:extension}{डिग्री} $1, 1, 2, 3, 3$. श्रेणी-$2$ से उठाता है $ S_4/V \cong S_3$
(\cref{prop:b3:representations:onedim}(बी), $ V $ क्लेन समूह);
\omterm{def:b3:galois:extension}{डिग्री} $3$: मानक \omterm{def:b3:representations:rep}{प्रतिनिधित्व} और इसके द्वारा मोड़
$\varepsilon $:
\[
\begin{array}{c|ccccc}
S_4 & e\,[1] & (1\,2)\,[6] & (1\,2)(3\,4)\,[3] &
(1\,2\,3)\,[8] & (1\,2\,3\,4)\,[6]\\
\hline
\mathbf 1 & 1 & 1 & 1 & 1 & 1\\
\varepsilon & 1 & -1 & 1 & 1 & -1\\
\chi_2 & 2 & 0 & 2 & -1 & 0\\
\chi_{\mathrm{std}} & 3 & 1 & -1 & 0 & -1\\
\varepsilon\chi_{\mathrm{std}} & 3 & -1 & -1 & 0 & 1
\end{array}
\]
($\chi_{\mathrm{std}}(g) = \operatorname{fix}(g) - 1$;
$\chi_2$ का मूल्यांकन करता है $ S_3$-प्रत्येक \omterm{def:b3:groups:action}{कक्षा} की छवि पर तालिका
मॉड $ V $.) सभी पंक्ति और स्तंभ ऑर्थोगोनैलिटी जांच पास हो जाती हैं ---
उनमें से दो को चलाना है \cref{exo:b3:representations:3}वार्मअप.
\end{example}

\begin{method}\label{met:b3:representations:table}
एक निर्माण करने के लिए \omterm{def:b3:representations:table}{चरित्र तालिका}: (1) सूची \omterm{ex:b3:groups:actions}{संयुग्मता वर्ग} और आकार;
(2) गिनें डिग्री-$1$ अक्षर के जरिए $ G/D(G)$ और उन्हें लिखें; (3)
से शेष डिग्रियाँ ज्ञात कीजिए $\sum n_i^2 = \abs G $ (छोटा
पूर्णांक कॉम्बिनेटरिक्स); (4) सस्ता प्राप्त करें इरेड्यूसिबल्स: से उठाओ
भागफल, घटाना $\mathbf 1$ क्रमपरिवर्तन से अक्षर
(जाँच करना $\langle\chi,\chi\rangle = 1$), गुणा ज्ञात अक्षर
\omterm{def:b3:galois:extension}{डिग्री} से-$1$ वाले; (5) अज्ञात पंक्तियों को कॉलम द्वारा समाप्त करें
ऑर्थोगोनैलिटी --- प्रत्येक कॉलम कॉलम के लिए ऑर्थोगोनल है
पहले से ही पूरा, और column $ e $ डिग्रियाँ लेकर चलता है। सत्यापित करें
पूर्ण ओर्थोगोनैलिटी स्वीप के साथ सब कुछ।
\end{method}

अपरिवर्तनीयता), और दो \omterm{def:b3:galois:extension}{डिग्री} पूरी करें-$3$ पंक्ति दर स्तंभ

पर $5$-साइकिल कक्षाएं।
(सी) तैयार तालिका से सत्यापित करें कि $ A_5$ साधारण है
(\cref{exo:b3:representations:6}(सी))।
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:representations:10}
होने देना $\rho $ एक हो अघुलनशील \omterm{def:b3:representations:rep}{प्रतिनिधित्व} \omterm{def:b3:galois:extension}{डिग्री} का $ n $ और $ z
\in Z(G)$. दिखाओ $\rho(z) = \lambda_z\,\mathrm{id}$ साथ
$\lambda \colon Z(G) \to \C^\times $ एक रूपवाद (
\emph{\omterm{ex:b3:groups:actions}{केंद्र}ीय \omterm{def:b3:representations:character}{चरित्र}}), और निष्कर्ष निकालें $\abs{\chi(z)} = n $ के लिए
\omterm{ex:b3:groups:actions}{केंद्र}ीय $ z $. आवेदन: यदि $ G $ एक वफादार है अपरिवर्तनीय
\omterm{def:b3:representations:rep}{प्रतिनिधित्व}, तब $ Z(G)$ चक्रीय है.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:representations:11}
(आइसोटाइपिक अनुमान) चलो $(V, \rho)$ एक हो \omterm{def:b3:representations:rep}{प्रतिनिधित्व} का
$ G $ और $\chi_i $ एक \omterm{def:b3:representations:rep}{अलघुकरणीय} \omterm{def:b3:representations:character}{चरित्र} \omterm{def:b3:galois:extension}{डिग्री} का $ n_i $.
परिभाषित करना
\[
p_i = \frac{n_i}{\abs G}\sum_{g\in G}
\overline{\chi_i(g)}\,\rho(g) \;\in\; \mathcal L(V).
\]
(ए) वह दिखाओ $ p_i $ है $ G $-समतुल्य, और इसकी गणना करें
एक के लिए प्रतिबंध \omterm{def:b3:representations:rep}{अलघुकरणीय} उपप्रतिनिधित्व $ W \subseteq
V $ का \omterm{def:b3:representations:character}{चरित्र} $\chi_j $: यह है $\delta_{ij}\,
\mathrm{id}_W $ \emph{(शूर; पहचानने के लिए निशान लें
अदिश)}.
(बी) उसका निष्कर्ष निकालें $ p_i $ योग पर एक प्रक्षेपण है $ V_i $ के सभी \omterm{def:b3:representations:rep}{अलघुकरणीय} का उपप्रतिनिधित्व \omterm{def:b3:representations:character}{चरित्र} $\chi_i $ (द
\emph{आइसोटाइपिक घटक}), वह $\sum_ip_i =
\mathrm{id}_V $, और वह विघटन $ V =
\bigoplus_iV_i $ विहित है --- महीन विभाजन के विपरीत
प्रत्येक का $ V_i $ में इरेड्यूसिबल्स.
(सी) के लिए \omterm{ex:b3:representations:examples}{नियमित प्रतिनिधित्व} का $ S_3$ और संकेत
\omterm{def:b3:representations:character}{चरित्र} $\varepsilon $, लिखें $ p_\varepsilon $ स्पष्ट रूप से
समूह बीजगणित और जाँच के एक तत्व के रूप में $ p_\varepsilon^2
= p_\varepsilon $ हाथ से.
\end{exercise}

\section{समस्या: बर्नसाइड का \texorpdfstring{$ p^aq^b $}{paqb}
प्रमेय}

\begin{problem}[{सप्ताहांत समस्या---समूहों की समाधानशीलता
आदेश $ p^aq^b $}]\label{pb:b3:representations:1}
बर्नसाइड ने 1904 में साबित किया कि प्रत्येक समूह जिसका ऑर्डर अधिकतम होता है
दो प्रमुख कारक हैं \omterm{def:b3:groups:derived}{व्याख्या करने योग्य} --- अमूर्त के बारे में एक बयान
ऐसे समूह जिनके बारे में आधी शताब्दी तक केवल ज्ञात साक्ष्य ही मिले
\omterm{def:b3:representations:character}{चरित्र} लिखित। यह समस्या पूर्णतः प्रमाण बनाती है,
संयोजन \cref{ch:b3:groups} (गलने की योग्यता),
\cref{ch:b3:modules} (\omterm{def:b3:modules:free}{अंतिम रूप से उत्पन्न} $\Z $-मॉड्यूल) और यह
अध्याय. लगातार, $\chi_1, \dots, \chi_r $ हैं \omterm{def:b3:representations:rep}{अलघुकरणीय}
अक्षर का $ G $, $ n_i = \chi_i(e)$.

\medskip
\noindent\textbf{Part I --- \omterm{def:b3:galois:algebraic}{बीजगणितीय} integers.} एक
\emph{\omterm{def:b3:galois:algebraic}{बीजगणितीय} पूर्णांक} की एक जड़ है \emph{राक्षसी} बहुपद
का $\Z[X]$.
\begin{enumerate}
  \item बताते हैं कि $\alpha $ एक \omterm{def:b3:galois:algebraic}{बीजगणितीय} पूर्णांक यदि अंगूठी
        $\Z[\alpha]$ एक है \omterm{def:b3:modules:free}{अंतिम रूप से उत्पन्न} $\Z $-मॉड्यूल.
  \item यह निष्कर्ष निकालें कि \omterm{def:b3:galois:algebraic}{बीजगणितीय} पूर्णांक एक उप-रिंग बनाते हैं
        $\C $. \emph{(अगर $\Z[\alpha], \Z[\beta]$ हैं परिमित रूप से
        उत्पन्न, ऐसा ही है $\Z[\alpha, \beta]$, और के सबमॉड्यूल
        \omterm{def:b3:modules:free}{अंतिम रूप से उत्पन्न} $\Z $-मॉड्यूल हैं \omterm{def:b3:modules:free}{अंतिम रूप से उत्पन्न},
        द्वारा \cref{thm:b3:modules:submodule} और एक प्रस्तुति
        तर्क --- या सीधे: का एक सबमॉड्यूल $\Z^n $ मुफ़्त है
        रैंक का $\leq n $.)}
  \item दिखाओ कि ए \emph{तर्कसंगत} \omterm{def:b3:galois:algebraic}{बीजगणितीय} पूर्णांक एक है
        पूर्णांक. \emph{(तर्कसंगत मूल प्रमेय।)}
  \item वह हर दिखाओ \omterm{def:b3:representations:character}{चरित्र} कीमत $\chi(g)$ एक
        \omterm{def:b3:galois:algebraic}{बीजगणितीय} पूर्णांक.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The class-sum relations.} एक ठीक करें
\omterm{def:b3:representations:rep}{अलघुकरणीय} $(V, \rho)$ \omterm{def:b3:galois:extension}{डिग्री} का $ n $ और \omterm{def:b3:representations:character}{चरित्र} $\chi $. एक के लिए
संयुग्मन वर्ग $ C $, होने देना $ S_C = \sum_{g \in C}\rho(g) \in
\mathcal L(V)$.
\begin{enumerate}[resume]
  \item बताते हैं कि $ S_C $ इसलिए, समतुल्य है $ S_C =
        \omega_C\,\mathrm{id}$ साथ
        \[
        \omega_C = \frac{\abs C\,\chi(g_C)}{n}
        \qquad (g_C \in C \text{ any representative}).
        \]
  \item बताते हैं कि $ S_CS_{C'} = \sum_{C''}
        a_{CC'C''}\,S_{C''}$ कहाँ $ a_{CC'C''} \in \N $ मायने रखता है,
        एक निश्चित के लिए $ z \in C''$, जोड़े $(x, y) \in C \times
        C'$ साथ $ xy = z $. यह निष्कर्ष निकालें कि $\omega_C $ संतुष्ट
        $\omega_C\,\omega_{C'} = \sum_{C''}
        a_{CC'C''}\,\omega_{C''}$.
  \item प्रत्येक का निष्कर्ष निकालें $\omega_C $ एक \omterm{def:b3:galois:algebraic}{बीजगणितीय} पूर्णांक.
        \emph{(द $\Z $-मॉड्यूल द्वारा उत्पन्न $1$ और यह
        $\omega_C $ एक है \omterm{def:b3:modules:free}{अंतिम रूप से उत्पन्न} अँगूठी; प्रश्न लागू करें
        1 की कसौटी---अधिक सटीक रूप से दिखायें $ M = \Z\text{-span}
        (1, (\omega_C)_C)$ संतुष्ट $\omega_{C_0} M \subseteq
        M $ और एक निर्धारक/केली--हैमिल्टन चाल का उपयोग करें, या
        प्रश्न 2 का सबरिंग तर्क।)}
  \item परिणाम निकालना \emph{\omterm{thm:b3:galois:finitefields}{फ्रोबेनियस} भाजकत्व}: $ n_i $ विभाजित
        $\abs G $ हरएक के लिए \omterm{def:b3:representations:rep}{अलघुकरणीय} \omterm{def:b3:galois:extension}{डिग्री}।
        \emph{(गणना करें $\frac{\abs G}{n} = \frac{\abs G}{n}
        \langle\chi,\chi\rangle = \sum_C \omega_C\,
        \overline{\chi(g_C)}$: एक \omterm{def:b3:galois:algebraic}{बीजगणितीय} पूर्णांक अर्थात
        तर्कसंगत.)}
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Burnside's simplicity criterion.}
\begin{enumerate}[resume]
  \item होने देना $\chi $ होना \omterm{def:b3:representations:rep}{अलघुकरणीय} \omterm{def:b3:galois:extension}{डिग्री} का $ n $ और $ C $ एक वर्ग
        के साथ $\gcd(\abs C, n) = 1$. बी का उपयोग करनाézout और प्रश्न
        4--7, वह दिखाओ $\frac{\chi(g_C)}{n}$ एक \omterm{def:b3:galois:algebraic}{बीजगणितीय}
        पूर्णांक.
  \item इसके अलावा मान लीजिए $0 < \abs{\chi(g_C)} < n $. दिखाओ यह है
        असंभव: \omterm{def:b3:galois:algebraic}{बीजगणितीय} पूर्णांक $\alpha =
        \chi(g_C)/n $ उसके पास सब कुछ है \emph{संयुग्मित करता है} --- संख्या $\alpha_\sigma = \frac1n\sigma(\chi(g_C))$ के लिए
        $\sigma \in \operatorname{Gal}(\Q(\zeta_{\abs G})/\Q)$,
        प्रत्येक का औसत $ n $ एकता की जड़ें --- मापांक की
        $\leq 1$,
        तो उत्पाद $ N = \prod_\sigma\alpha_\sigma $ एक है
        तर्कसंगत \omterm{def:b3:galois:algebraic}{बीजगणितीय} पूर्णांक के साथ $0 < \abs N < 1$ ---
        उद्धृत करते हुए प्रत्येक दावे को उचित ठहराएँ
        \cref{thm:b3:galois:cyclotomicirred} के लिए \omterm{def:b3:galois:galois}{गैलोइस
        समूह} और तथ्य यह है कि $\sigma $ की जड़ों को क्रमपरिवर्तित करता है
        एकता. निष्कर्ष: या तो $\chi(g_C) = 0$ या $\rho(g_C)$
        अदिश है (\cref{prop:b3:representations:charbasics}).
  \item (बर्नसाइड की कसौटी) चलो $ C \neq \{e\}$ एक हो
        संयुग्मन वर्ग का \emph{प्रधान शक्ति} आकार $ p^k > 1$,
        और मान लीजिए $ G $ \omterm{def:b3:groups:simple}{सरल} नॉनबेलियन है। स्तम्भ
        के स्तंभ पर रूढ़िवादिता $ C $ के कॉलम के विरुद्ध
        $ e $ देता है $1 + \sum_{i \geq 2} n_i\chi_i(g_C) = 0$. दिखाओ
        वह कुछ गैरतुच्छ $\chi_i $ साथ $ p \nmid n_i $ है
        $\chi_i(g_C) \neq 0$ \emph{(अन्यथा $\frac1p $ होगा
        एक \omterm{def:b3:galois:algebraic}{बीजगणितीय} पूर्णांक)}; प्रश्न 10 से, $\rho_i(g_C)$
        अदिश है; सरलता से विरोधाभास निकालें
        \emph{(के समुच्चय $ g $ साथ $\rho_i(g)$ अदिश राशि है एक
        \omterm{def:b3:groups:normal}{सामान्य} उपसमूह; से वफ़ादारी का प्रयोग करें
        \cref{exo:b3:representations:6})}. निष्कर्ष: \emph{नहीं
        \omterm{def:b3:groups:simple}{सरल} नॉनबेलियन समूह में एक है संयुग्मन वर्ग प्राइम का
        शक्ति का आकार $> 1$.}
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The theorem.}
\begin{enumerate}[resume]
  \item होने देना $\abs G = p^aq^b $ साथ $ a + b \geq 1$. अगर $ G $ है
        \omterm{def:b3:groups:simple}{सरल}, दिखाओ यह एबेलियन है: उठाओ $ z \neq e $ में
        \omterm{ex:b3:groups:actions}{केंद्र} एक \omterm{def:b3:groups:sylow}{सिलो} का $ q $-उपसमूह
        (\cref{thm:b3:groups:pfixed}) और के आकार पर विचार करें
        इसका संयुग्मन वर्ग $[G : Z_G(z)]$, की एक शक्ति $ p $
        (क्यों?); प्रश्न 11 लागू करें.
  \item पर प्रेरण द्वारा समापन करें $\abs G $: \emph{का हर समूह
        आदेश $ p^aq^b $ है \omterm{def:b3:groups:derived}{व्याख्या करने योग्य}} (बर्नसाइड)। क्यों करता है
        तीन अभाज्य संख्याओं के लिए तर्क विराम --- और यह अवश्य दिया जाना चाहिए
        $\abs{A_5} = 2^2\cdot3\cdot5$?
\end{enumerate}

\medskip
\noindent\textbf{Part V --- The \omterm{def:b3:representations:table}{चरित्र तालिका} of $ A_5$.} द
बर्नसाइड का प्रमेय सबसे छोटे समूह को नहीं छू सकता, जो इसके पूर्ण होने का हकदार है
चित्र; नीचे दी गई हर चीज़ केवल इस अध्याय प्लस का उपयोग करती है
\cref{exo:b3:groups:11}.
\begin{enumerate}[resume]
  \item से स्मरण करें \cref{exo:b3:groups:11} पांच संयुग्मन
        कक्षाएं का $ A_5$: $\{e\}$, द $15$ दोगुना
        स्थानान्तरण, $20$ तीन-चक्र, और दो वर्ग
        का $12$ पाँच-चक्रों में से प्रत्येक का \omterm{def:b3:representations:rep}{प्रतिनिधित्व} किया जाता है $ c =
        (1\,2\,3\,4\,5)$ और $ c^2$. स्पष्ट करें कि पाँच-चक्र क्यों हैं
        दो हिस्सों में बंट गया $ A_5$-वर्ग हालांकि वे एक ही बनाते हैं
        $ S_5$-कक्षा।
  \item दिखाओ कि \omterm{def:b3:representations:rep}{अलघुकरणीय} की \omterm{def:b3:galois:extension}{डिग्री} $ A_5$ बिलकुल हैं
        $1, 3, 3, 4, 5$: उपयोग $\sum_in_i^2 = 60$ साथ $ r = 5$
        कक्षाएं, और तथ्य यह है कि $ A_5$ है \omterm{prop:b3:galois:perfect}{उत्तम} ($ D(A_5) =
        A_5$), तो तुच्छ है \omterm{def:b3:representations:character}{चरित्र} क्या यह एकमात्र रैखिक है;
        फिर हर दूसरे मल्टीसेट को हटा दें \emph{(लिखना $59$
        पूर्णांकों के चार वर्गों के योग के रूप में $\geq 2$: इसे जांचें
        सभी सारांश प्रशंसनीय होने पर केवल एक ही तरीके से होता है
        \omterm{def:b3:galois:extension}{डिग्री})}.
  \item होने देना $\pi $ क्रमपरिवर्तन हो \omterm{def:b3:representations:character}{चरित्र} का $ A_5$ पर
        $\{1, \dots, 5\}$: $\pi(g) = \#\operatorname{Fix}(g)$,
        मूल्यों के साथ $5, 1, 2, 0, 0$ पाँच वर्गों पर. गणना करना
        $\langle\pi, \mathbf 1\rangle $ और $\langle\pi,
        \pi\rangle $, और उसका निष्कर्ष निकालो $\chi_4 = \pi - \mathbf 1$
        है \omterm{def:b3:representations:rep}{अलघुकरणीय} \omterm{def:b3:galois:extension}{डिग्री} का $4$, मूल्यों के साथ $4, 0, 1, -1,
        -1$.
  \item पर वही खेल $10$ अव्यवस्थित जोड़े $\{i, j\}$: द
        निश्चित-बिंदु गणनाएँ हैं $10, 2, 1, 0, 0$. गणना करना
        $\langle\pi_{10}, \pi_{10}\rangle $, $\langle\pi_{10},
        \mathbf 1\rangle $ और $\langle\pi_{10}, \chi_4\rangle $,
        अपघटन निकालें $\pi_{10} = \mathbf 1 + \chi_4
        + \chi_5$, और प्राप्त करें \omterm{def:b3:representations:rep}{अलघुकरणीय} $\chi_5$ \omterm{def:b3:galois:extension}{डिग्री} का
        $5$ मूल्यों के साथ $5, 1, -1, 0, 0$.
  \item दो बचे हैं इरेड्यूसिबल्स $\chi_2, \chi_3$ है
        \omterm{def:b3:galois:extension}{डिग्री} $3$. कॉलम ऑर्थोगोनैलिटी (प्रत्येक गैर-पहचान
        पहचान कॉलम के सामने कॉलम, और प्रत्येक कॉलम के साथ
        स्वयं) पाँच-चक्रों से उनके मान निर्धारित करता है:
        दिखाओ $\chi_2(g) = \chi_3(g) = -1$ डबल पर
        स्थानान्तरण और $0$ तीन चक्रों पर.
  \item पाँच-चक्र कक्षाओं पर, सेट करें $ x = \chi_2(c)$ और $ y =
        \chi_2(c^2)$; समरूपता किसी को लेने देती है $\chi_3(c) = y $,
        $\chi_3(c^2) = x $. के कॉलम से $ c $ के साथ जोड़ा गया
        पहचान कॉलम और के कॉलम के साथ $ c^2$, व्युत्पन्न
        $ x + y = 1$ और $ xy = -1$ (और मूल्य की जाँच करें $ x^2 +
        y^2 = 3$ के कॉलम द्वारा दिया गया है $ c $ स्वयं के साथ), इसलिए
        \[
        \{x, y\} = \Bigl\{\frac{1 + \sqrt5}2,\ \frac{1 -
        \sqrt5}2\Bigr\} :
        \]
        स्वर्णिम अनुपात और उसका संयुग्म। इकट्ठा करो
        पूर्ण \omterm{def:b3:representations:table}{चरित्र तालिका} का $ A_5$.
  \item जाँच चलाएँ: पंक्ति मानदंड $\chi_2$ है $1$ (उपयोग
        $\varphi^2 + \bar\varphi^2 = 3$), $\langle\chi_2,
        \chi_3\rangle = 0$, और \omterm{thm:b3:galois:finitefields}{फ्रोबेनियस} विभाज्यता (प्रश्न)
        8) सभी पाँच \omterm{def:b3:galois:extension}{डिग्री} के लिए। आप तालिका में कहां हैं
        \emph{देखना} के साथ एक अंतर $ S_5$, किसका सब
        \omterm{def:b3:representations:character}{चरित्र} मान तर्कसंगत पूर्णांक हैं?
  \item अकेले तालिका से यह निष्कर्ष निकालें $ A_5$ \omterm{def:b3:groups:simple}{सरल} है: ए
        \omterm{def:b3:groups:normal}{सामान्य} उपसमूह का एक संघ है \omterm{ex:b3:groups:actions}{संयुग्मता वर्ग}
        युक्त $ e $ जिसकी प्रमुखता विभाजित होती है $60$ --- जाँच करें
        इसका कोई उचित उप-योग नहीं है $1 + 15 + 20 + 12 + 12$
        शब्द युक्त $1$ विभाजित $60$. के साथ क्रॉस-चेक करें
        प्रश्न 11 की कसौटी: सत्यापित करें कि कोई वर्ग नहीं है $ A_5$
        प्राइम-पावर आकार है $> 1$.
  \item (आइकोसाहेड्रल कोडा) $ A_5$ का घूर्णन समूह है
        इकोसाहेड्रोन, और डिग्री-$3$ अभ्यावेदन हैं
        दो ज्यामितीय क्रियाएँ चालू $\R^3$. ट्रेस सत्यापित करें
        पहचान: कोण द्वारा घूर्णन $\theta $ निशान है $1 +
        2\cos\theta $, और $1 + 2\cos\frac{2\pi}5 =
        \frac{1+\sqrt5}2 = \varphi $. बिना कुछ बताए समझाओ
        गणना दूसरी \omterm{def:b3:galois:extension}{डिग्री} क्यों-$3$ \omterm{def:b3:representations:character}{चरित्र} अवश्य
        संयुग्म मान ले जाएं: द \omterm{def:b3:galois:galois}{गैलोइस समूह} का
        $\Q(\sqrt5)/\Q $ समग्र रूप से कार्य करता है \omterm{def:b3:representations:table}{चरित्र तालिका}
        (प्रविष्टि अनुसार), क्रमपरिवर्तित करना \omterm{def:b3:representations:rep}{अलघुकरणीय} अक्षर.
\end{enumerate}

\medskip
\noindent\textbf{Part VI --- Complements: a central bound and
the tensor square.}
\begin{enumerate}[resume]
  \item (विभाज्यता से भी तीव्र) चलो $\chi $ होना \omterm{def:b3:representations:rep}{अलघुकरणीय}
        \omterm{def:b3:galois:extension}{डिग्री} का $ n $. बताते हैं कि $\abs{\chi(z)} = n $ हरएक के लिए
        $ z \in Z(G)$ \emph{(शूर की लेम्मा: $\rho(z)$ एक है
        अदिश, परिमित क्रम का)}, और से निष्कर्ष निकालें
        $\langle\chi, \chi\rangle = 1$ बाध्य
        \[
        n^2 \leq [G : Z(G)] .
        \]
        दिखाएँ कि आदेश के नॉनबेलियन समूह $8$ पास होना
        \omterm{def:b3:representations:rep}{अलघुकरणीय} \omterm{def:b3:galois:extension}{डिग्री} $1, 1, 1, 1, 2$ \emph{(पाँच
        \omterm{ex:b3:groups:actions}{संयुग्मता वर्ग}; लिखना $8$ पाँच वर्गों के योग के रूप में)}
        और समानता प्राप्त करें $4 = [G : Z(G)]$; सीमा की जाँच करें
        पर $ A_5$, किसका \omterm{ex:b3:groups:actions}{केंद्र} तुच्छ है.
  \item (टेन्सर वर्ग का $\chi_2$) के लिए $ g $ परिमित क्रम का,
        $\rho(g)$ एकता की जड़ के साथ विकर्णीय है
        eigenvalues; का निष्कर्ष निकालें \omterm{def:b3:representations:character}{चरित्र} सूत्रों
        \[
        \chi_{\operatorname{Sym}^2 V}(g)
        = \frac{\chi(g)^2 + \chi(g^2)}2,
        \qquad
        \chi_{\Lambda^2 V}(g)
        = \frac{\chi(g)^2 - \chi(g^2)}2 .
        \]
        उन्हें लागू करें $\chi_2$ का $ A_5$ \emph{(टिप्पणी $ g^2$ चलता है
        की \omterm{def:b3:groups:action}{कक्षा} के माध्यम से $ c^2$ कब $ g $ उसके माध्यम से चलता है
        का $ c $, और इसके विपरीत)}: दिखाओ $\Lambda^2\chi_2 =
        \chi_2$ और $\operatorname{Sym}^2\chi_2 = \mathbf 1 +
        \chi_5$, इस तरह
        \[
        \chi_2\otimes\chi_2 = \mathbf 1 + \chi_2 + \chi_5 .
        \]
        व्याख्या $\Lambda^2\chi_2 = \chi_2$ ज्यामितीय रूप से के माध्यम से
        क्रॉस उत्पाद पर $\R^3$.
  \item (तालिका का अंतिम ऑडिट) संख्यात्मक रूप से सत्यापित करें: कॉलम
        दो पाँच-चक्र स्तंभों के बीच ऑर्थोगोनैलिटी
        ($\varphi\bar\varphi = -1$), मूल्य $5 =
        \abs{Z_{A_5}(c)}$ के कॉलम के लिए $ c $ खुद के खिलाफ,
        और नियमित का लुप्त होना \omterm{def:b3:representations:character}{चरित्र} $\sum_i
        n_i\chi_i(g) = 0$ चार गैर-पहचान में से प्रत्येक पर
        तालिका के स्तंभ.
\end{enumerate}
\end{problem}
