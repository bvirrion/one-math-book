\chapter{उत्पाद माप, फ़ुबिनी, चर का परिवर्तन}
\label{ch:b3:product}

एक-आयामी लेबेस्ग सिद्धांत बहु-आयामी बन जाता है
दो प्रमेयों के माध्यम से गणना। \emph{टोनेली - फ़ुबिनी} ऐसा कहते हैं
उत्पादों पर इंटीग्रल पुनरावृत्त इंटीग्रल हैं --- स्लाइसिंग है
वैध, किसी भी क्रम में, परिकल्पना के तहत वास्तव में कोई भी कर सकता है
जांचें. \emph{चर सूत्र का परिवर्तन} परिवहन
साथ में अभिन्न $\mathcal C^1$ भिन्नता, के साथ
मात्रा के लिए विनिमय दर के रूप में जैकोबियन निर्धारक; हम साबित करते हैं
यह पूरी तरह से, रैखिक मामले से शुरू होता है जहां यह समझाता है
निर्धारक क्या है \emph{है}. अनुप्रयोग कैस्केड: परत
केक फार्मूला, कनवल्शन, \omterm{ex:b3:product:polar}{ध्रुवीय निर्देशांक}, की मात्रा
$ n $-बॉल --- और, सप्ताहांत समस्या में, स्टर्लिंग का सूत्र
एक ईमानदार त्रुटि विश्लेषण के साथ.

\section{उत्पाद $\sigma $-बीजगणित और उत्पाद उपाय}

\begin{definition}\label{def:b3:product:sigma}
के लिए \omterm{def:b3:lebesgue:measurable}{औसत दर्जे का} खाली स्थान $(X, \mathcal A)$, $(Y, \mathcal B)$, द
\emph{उत्पाद $\sigma $-बीजगणित}\index{उत्पाद $\sigma $-बीजगणित} $\mathcal A \otimes \mathcal B $ पर $ X \times
Y $ द्वारा उत्पन्न होता है \emph{आयत} $ A \times B $ ($ A \in
\mathcal A $, $ B \in \mathcal B $) --- ए $\pi $-प्रणाली। के लिए $ E
\subseteq X\times Y $ और $ x \in X $, द \emph{अनुभाग} है $ E_x
= \{y : (x,y) \in E\}$; एक समारोह के लिए $ f $ उत्पाद पर,
$ f_x = f(x, \cdot)$.
\end{definition}

\begin{proposition}\label{prop:b3:product:sections}
(ए) यदि $ E \in \mathcal A\otimes\mathcal B $, हर अनुभाग $ E_x
\in \mathcal B $ (और सममित रूप से); अगर $ f $ है $\mathcal
A\otimes\mathcal B $-मापने योग्य, प्रत्येक $ f_x $ है $\mathcal
B $-मापने योग्य.
(बी) $\mathcal B(\R^m)\otimes\mathcal B(\R^n) = \mathcal
B(\R^{m+n})$.
\end{proposition}

\begin{proof}
(ए) अच्छे सेट: $\{E : E_x \in \mathcal B\ \forall x\}$ एक है
$\sigma $-बीजगणित (अनुभाग पूरकों के साथ चलते हैं और
गणनीय संघ) जिसमें आयतें हों। के लिए $ f $:
$(f_x)^{-1}(B) = (f^{-1}(B))_x $.
(बी) ($\subseteq $) बोरेल सेट के आयत: यह पर्याप्त है
खुला$\times $ खुले बक्सों में बोरेल हैं $\R^{m+n}$ (वे खुले हैं)
और वह \omterm{def:b3:groups:normal}{सामान्य} बोरेल आयतें सीमाएँ हैं --- अच्छे सेट
पुनः: $\{A : A\times\R^n \in \mathcal B(\R^{m+n})\}$ एक है
$\sigma $-बीजगणित जिसमें खुलता है; ऐसे दो को प्रतिच्छेद करें।
($\supseteq $) प्रत्येक \omterm{def:b3:topology:topology}{खुला सेट} का $\R^{m+n}$ एक गणनीय संघ है
तर्कसंगत खुले बक्सों की $ U\times V $: उत्पाद में निहित है
$\sigma $-बीजगणित.
\end{proof}

\begin{theorem}[उत्पाद माप]\label{thm:b3:product:existence}
होने देना $(X, \mathcal A, \mu)$ और $(Y, \mathcal B, \nu)$ होना
\emph{$\sigma $-परिमित}. हरएक के लिए $ E \in \mathcal
A\otimes\mathcal B $, समारोह $ x \mapsto \nu(E_x)$ है
\omterm{def:b3:lebesgue:measurable}{औसत दर्जे का}, और\index{उत्पाद माप}
\[
(\mu\otimes\nu)(E) = \int_X \nu(E_x)\,\dd\mu(x)
\]
अद्वितीय को परिभाषित करता है \omterm{def:b3:measure:measure}{उपाय} पर $\mathcal A\otimes\mathcal B $
साथ $(\mu\otimes\nu)(A\times B) = \mu(A)\nu(B)$. यह है
$\sigma $-परिमित, और सममित: वही \omterm{def:b3:measure:measure}{उपाय} द्वारा प्राप्त किया जाता है
एकीकरण $ x $-विरुद्ध धाराएँ $\nu $.
\end{theorem}

\begin{proof}
\emph{मापन योग्यता का $ x \mapsto \nu(E_x)$.} पहले चलो $\nu $
परिमित हो. \omterm{def:b3:groups:action}{कक्षा} $\mathcal D $ का $ E $ जिसके लिए नक्शा है
\omterm{def:b3:lebesgue:measurable}{औसत दर्जे का} आयतें शामिल हैं ($\nu((A\times B)_x) =
\nu(B)\mathbf 1_A(x)$) और एक है $\lambda $-प्रणाली: के लिए $ E
\subseteq F $ में $\mathcal D $, $\nu((F\setminus E)_x) =
\nu(F_x) - \nu(E_x)$ (अनंतता); के लिए $ E_n \uparrow E $,
$\nu((E_n)_x) \uparrow \nu(E_x)$ (निरंतरता नीचे से), और
की एकस्वर सीमा मापने योग्य कार्य हैं \omterm{def:b3:lebesgue:measurable}{औसत दर्जे का}.
आयतों से एक बनता है $\pi $-प्रणाली: डिनकिन
(\cref{thm:b3:measure:dynkin}) देता है $\mathcal D = \mathcal
A\otimes\mathcal B $. अगर $\nu $ है $\sigma $-परिमित, लिखो $ Y =
\bigcup Y_k $, $ Y_k \uparrow $, $\nu(Y_k) < \infty $: $\nu(E_x) =
\lim_k\nu_k(E_x)$ साथ $\nu_k = \nu(\cdot\cap Y_k)$ परिमित.

\emph{\omterm{def:b3:measure:measure}{उपाय}.} $\sigma $-की अतिरिक्तता $ E \mapsto
\int\nu(E_x)\dd\mu $ से अनुसरण करता है
\cref{cor:b3:lebesgue:additivity} (असंयुक्त सेटों के अनुभाग
असंयुक्त हैं)। आयतों पर यह देता है $\mu(A)\nu(B)$.
\emph{विशिष्टता}: दो उम्मीदवार इस पर सहमत हैं $\pi $-की व्यवस्था
आयत; $\sigma $-परिमितता आयत प्रदान करती है $ X_k\times
Y_k \uparrow X\times Y $ परिमित का \omterm{def:b3:measure:measure}{उपाय}:
\cref{thm:b3:measure:uniqueness}. समरूपता:
अन्य क्रम का निर्माण भी एक है \omterm{def:b3:measure:measure}{उपाय} पर सहमति
आयत --- अद्वितीय, इसलिए समान।
\end{proof}

\begin{definition}\label{def:b3:product:lebesgue}
\emph{लेब्सग्यू उपाय चालू $\R^d $}\index{लेब्सग्यू उपाय चालू $\R^d $} है $\lambda_d = \lambda\otimes\cdots\otimes\lambda $
($ d $ कारक; निर्माण की संबद्धता की जाँच की जाती है
बक्से और विशिष्टता द्वारा प्रचारित)। यह अनोखा बोरेल है
\omterm{def:b3:measure:measure}{उपाय} प्रत्येक डिब्बा दे रहे हैं $\prod\intoc{a_i}{b_i}$ इसकी मात्रा
$\prod(b_i - a_i)$; यह अनुवाद-अपरिवर्तनीय (अनुवाद) है
बक्सों पर सहमत), $\sigma $-परिमित, और \omterm{def:b3:complete:complete}{पूरा} के बाद
कैरथ\'ईओडोरी \omterm{thm:b3:complete:completion}{समापन} --- हम लिखते हैं $\lambda_d $ के लिए
\omterm{def:b3:complete:complete}{पूरा} हुआ \omterm{def:b3:measure:measure}{उपाय} और तदनुसार एकीकृत करें।
\end{definition}

\section{टोनेली और फ़ुबिनी}

\begin{theorem}[टोनेली]\label{thm:b3:product:tonelli}
$\mu, \nu $ $\sigma $-परिमित, $ f \colon X\times Y \to [0,
+\infty]$ \omterm{def:b3:lebesgue:measurable}{औसत दर्जे का}. तब $ x \mapsto \int_Y f_x\,\dd\nu $ है
\omterm{def:b3:lebesgue:measurable}{औसत दर्जे का} और\index{टोनेली का प्रमेय}
\[
\int_{X\times Y}f\,\dd(\mu\otimes\nu)
= \int_X\Bigl(\int_Y f(x,y)\,\dd\nu(y)\Bigr)\dd\mu(x)
= \int_Y\Bigl(\int_X f(x,y)\,\dd\mu(x)\Bigr)\dd\nu(y).
\]
\end{theorem}

\begin{proof}
मानक मशीन. के लिए $ f = \mathbf 1_E $ यह है
\cref{thm:b3:product:existence} (और इसका सममित रूप)। द्वारा
रैखिकता यह \omterm{def:b3:groups:simple}{सरल} के लिए रखती है $ f \geq 0$. \omterm{def:b3:groups:normal}{सामान्य} के लिए $ f \geq
0$: \omterm{def:b3:groups:simple}{सरल} ले लो $ s_n \nearrow f $
(\cref{thm:b3:lebesgue:approximation}); तब $\int_Y(s_n)_x
\dd\nu \nearrow \int_Y f_x\dd\nu $ प्रत्येक के लिए $ x $ (एमसीटी में $ Y $),
इसलिए वामपंथी सदस्य एमसीटी द्वारा एकजुट हुए $ X $, जबकि
$\int s_n\,\dd(\mu\otimes\nu) \nearrow \int f $ एमसीटी द्वारा
उत्पाद.
\end{proof}

\begin{theorem}[फ़ुबिनी]\label{thm:b3:product:fubini}
$\mu, \nu $ $\sigma $-परिमित, $ f \in L^1(\mu\otimes\nu)$. फिर
के लिए $\mu $-ए.ई.\ $ x $ अनुभाग $ f_x $ है $\nu $-अभिन्न, द
एई-परिभाषित फ़ंक्शन $ x \mapsto \int f_x\dd\nu $ है
\omterm{def:b3:lebesgue:l1}{समाकलनीय}, और दो पुनरावृत्त अभिन्न दोनों बराबर हैं
$\int f\,\dd(\mu\otimes\nu)$.\index{फ़ुबिनी का प्रमेय}
\end{theorem}

\begin{proof}
टोनेली ने आवेदन किया $\abs f $ दिखाता है $\varphi(x) = \int\abs{f_x}
\dd\nu $ परिमित अभिन्न है, इसलिए परिमित है। $ f_x \in
L^1(\nu)$ ए.ई. के लिए\ $ x $. विभाजित करना $ f = f^+ - f^-$ (वास्तविक मामला;
घटकों द्वारा जटिल): टोनेली प्रत्येक पुनरावृत्त की गणना करता है
का अभिन्न अंग $ f^\pm $ जैसा $\int f^\pm\dd(\mu\otimes\nu) <
\infty $, और एई-परिभाषित अंतर एकीकृत होता है
अंतर. दूसरे क्रम के लिए सममित रूप से.
\end{proof}

\begin{method}\label{met:b3:product:fubini}
दो अभिन्नों (या एक अभिन्न और एक योग, या दो) को आपस में बदलने के लिए
रकम): यदि इंटीग्रैंड है \emph{गैर नकारात्मक}, अदला-बदली
स्वतंत्र रूप से (टोनेली)। अन्यथा, पहले टोनेली लगाएं $\abs f $
जिस भी क्रम में अनुमान लगाना आसान हो; यदि परिणाम है
परिमित, फ़ुबिनी इंटरचेंज को वैध बनाता है। इसे कभी न छोड़ें
$\abs f $ जाँच करना: \cref{exo:b3:product:4}के इंटीग्रैंड में दो हैं
के साथ पुनरावृत्त अभिन्न अंग \emph{विभिन्न मूल्य}.
\end{method}

\begin{proposition}[परतों वाला केक]\label{prop:b3:product:layercake}
के लिए $ f \geq 0$ \omterm{def:b3:lebesgue:measurable}{औसत दर्जे का} पर $(X, \mathcal A, \mu)$
$\sigma $-परिमित:\index{परत केक फार्मूला}
\[
\int_X f\,\dd\mu = \int_0^{+\infty}\mu(\{f > t\})\,\dd t,
\qquad
\int_X f^p\,\dd\mu = p\int_0^{+\infty}t^{p-1}\mu(\{f >
t\})\,\dd t \quad (p \geq 1).
\]
\end{proposition}

\begin{proof}
टोनेली लगाएं $\mathbf 1_{\{(x,t) : 0 < t < f(x)\}}$ पर $ X
\times \intoo0{+\infty}$ (\omterm{def:b3:lebesgue:measurable}{औसत दर्जे का}: यह है $\{(x,t): f(x) - t
> 0\}\cap\{t > 0\}$, का एक बोरेल-प्रकार संयोजन
\omterm{def:b3:lebesgue:measurable}{औसत दर्जे का} $(x,t)\mapsto f(x) - t $): में एकीकृत करना $ t $ प्रथम
देता है $\int f\,\dd\mu $; में $ x $ पहला, $\int_0^\infty\mu(f >
t)\dd t $. के लिए $ f^p $: विकल्प $ t = s^p $ में $\int\mu(f^p > t)
\dd t $, यानी\ पहले सूत्र को लागू करें $ f^p $ और बदलो
एक आयामी अभिन्न में चर ($\{f^p > s^p\} =
\{f > s\}$).
\end{proof}

\begin{theorem}[कनवल्शन चालू $ L^1$]\label{thm:b3:product:convolution}
के लिए $ f, g \in L^1(\R^d, \lambda_d)$, अभिन्न
\[
(f * g)(x) = \int_{\R^d} f(x - y)\,g(y)\,\dd y
\]
ए.ई. के लिए बिल्कुल अभिसरण करता है\ $ x $, परिभाषित करता है $ f * g \in
L^1(\R^d)$ साथ $\norm{f*g}_1 \leq \norm f_1\norm g_1$, और
$*$ क्रमविनिमेय और साहचर्य है।\index{कनवल्शन}
\end{theorem}

\begin{proof}
$(x, y) \mapsto f(x-y)g(y)$ है \omterm{def:b3:lebesgue:measurable}{औसत दर्जे का} ($ (x,y)\mapsto x -
y $ है \omterm{def:b3:topology:continuity}{निरंतर}; लिखें और गुणा करें)। टोनेली:
\[
\int\!\!\int \abs{f(x-y)}\abs{g(y)}\,\dd y\,\dd x
= \int\abs{g(y)}\Bigl(\int\abs{f(x - y)}\dd x\Bigr)\dd y
= \norm f_1\norm g_1 < \infty
\]
(अनुवाद अपरिवर्तनीयता का $\lambda_d $ आंतरिक अभिन्न में)।
अतः दोहरा समाकलन परिमित है; फ़ुबिनी देता है ए.ई.\ निरपेक्ष
अभिसरण और \omterm{def:b3:rings:ideal}{आदर्श} बाध्य $\norm{f*g}_1 \leq \norm
f_1\norm g_1$. क्रमविनिमेयता: स्थानापन्न $ y \mapsto x - y $
(अनुवाद और प्रतिबिंब अपरिवर्तन --- प्रतिबिंब
बक्सों पर अपरिवर्तनशीलता कायम रहती है, इसलिए हर जगह विशिष्टता होती है)।
सहयोगीता: टोनेली - ट्रिपल इंटीग्रल पर फ़ुबिनी।
\end{proof}

\section{चरों का परिवर्तन}

\begin{theorem}[चरों का रैखिक परिवर्तन]
\label{thm:b3:product:linearchange}
के लिए $ T \in GL_d(\R)$ और $ A \in \mathcal B(\R^d)$:
$\lambda_d(T(A)) = \abs{\det T}\,\lambda_d(A)$; फलस्वरूप
$\int f(y)\dd y = \abs{\det T}\int f(Tx)\,\dd x $ के लिए $ f \geq 0$
या \omterm{def:b3:lebesgue:l1}{समाकलनीय}.\index{चरों का रैखिक परिवर्तन}
\end{theorem}

\begin{proof}
 \omterm{def:b3:measure:measure}{उपाय} $\mu_T(A) = \lambda_d(T(A))$ एक बोरेल है \omterm{def:b3:measure:measure}{उपाय}
(\omterm{def:b3:topology:continuity}{होमियोमोर्फिज्म} बोरेल सेट को सुरक्षित रखें,
\cref{pb:b3:measure:1}), अनुवाद-अपरिवर्तनीय
($ T(A + x) = T(A) + Tx $), यूनिट बॉक्स पर परिमित: द्वारा
का लक्षण वर्णन \omterm{def:b3:measure:lebesgueouter}{लेब्सेग माप}
(\cref{exo:b3:measure:6}, जिसका प्रमाण शब्दशः काम करता है
$\R^d $ डायडिक क्यूब्स के साथ), $\mu_T = c(T)\lambda_d $ साथ $ c(T)
= \lambda_d(T(\intco01^d))$. वो नक्शा $ T \mapsto c(T)$ है
गुणक ($ c(ST) = c(S)c(T)$, रचना करके), तो यह
गणना करने के लिए पर्याप्त है $ c $ के जनरेटर पर $ GL_d $: प्राथमिक
matrices. विकर्ण $\operatorname{diag}(a, 1, \dots, 1)$:
इकाई घन को आयतन के एक बॉक्स में मैप करता है $\abs a $: $ c = \abs a =
\abs\det $. निर्देशांक का स्थानान्तरण: घन को क्रमपरिवर्तित करता है: $ c
= 1 = \abs\det $. ट्रांसवेक्शन $ T(x) = x + \alpha
x_2e_1$: इकाई घन की छवि एक कतरनी प्रिज्म है; द्वारा
टोनेली यह \omterm{def:b3:measure:measure}{उपाय} है $\int\lambda_1(\text{section})\dd x_2
\cdots \dd x_d = 1$, प्रत्येक $ x_1$-खंड का अंतराल है
लंबाई $1$: $ c = 1 = \abs{\det}$. प्रत्येक व्युत्क्रमणीय मैट्रिक्स एक है
इनका उत्पाद (गाऊसी उन्मूलन), और दोनों $ c $ और
$\abs\det $ गुणक हैं: $ c(T) = \abs{\det T}$. द
अभिन्न सूत्र मानक मशीन द्वारा अनुसरण किया जाता है (संकेतक,
\omterm{def:b3:groups:simple}{सरल}, एमसीटी)।
\end{proof}

\begin{theorem}[चरों का परिवर्तन]\label{thm:b3:product:changeofvar}
होने देना $ U, V \subseteq \R^d $ खुले रहें और $\Phi \colon U \to V $ ए
$\mathcal C^1$ भिन्नता. हरएक के लिए \omterm{def:b3:lebesgue:measurable}{औसत दर्जे का} $ f \colon V
\to [0, +\infty]$ (या $ f \in L^1(V)$):\index{का परिवर्तन
चर सूत्र}
\[
\int_V f(y)\,\dd y = \int_U
f\bigl(\Phi(x)\bigr)\,\abs{\det D\Phi(x)}\,\dd x .
\]
\end{theorem}

\begin{proof}
लिखना $ J(x) = \abs{\det D\Phi(x)}$. प्रमाण का हृदय है
असमानता
\[
\lambda_d\bigl(\Phi(A)\bigr) \leq \int_A J\,\dd\lambda_d
\qquad\text{for every Borel } A \subseteq U;
\tag{$*$}
\]
अपग्रेड के नीचे चरण 4 $(*)$ --- दोनों पर लागू $\Phi $ और
$\Phi^{-1}$ --- प्रमेय की समानता के लिए. ध्यान दें कि
$\Phi^{-1}$ स्वयं एक है $\mathcal C^1$ भिन्नतावाद के साथ
जैकोबियन $\abs{\det D\Phi^{-1}(y)} = J(\Phi^{-1}y)^{-1}$
(श्रृंखला नियम चालू $\Phi\circ\Phi^{-1} = \mathrm{id}$).

\emph{स्टेप 1: $(*)$ विरूपण कारक वाले घनों के लिए।} ठीक करें ए
\omterm{def:b3:topology:interior}{बंद} घन $ Q \subseteq U $ का \omterm{ex:b3:groups:actions}{केंद्र} $ x_0$ और ओर $2r $
(सुपर-नॉर्म बॉल)। दावा: प्रत्येक के लिए $\varepsilon > 0$, अगर $\Phi $
पर भिन्न है $ Q $ साथ
$\norm{D\Phi(x) - D\Phi(x_0)} \leq \varepsilon $ पर $ Q $
(ऑपरेटर मानदंड सुपर-नॉर्म के लिए), फिर
\[
\Phi(Q) \subseteq \Phi(x_0) + D\Phi(x_0)\Bigl(\,\bigl(1 +
\varepsilon\norm{D\Phi(x_0)^{-1}}\bigr)\,(Q - x_0)\Bigr),
\]
क्योंकि के लिए $ x \in Q $, माध्य मूल्य असमानता पर लागू होता है
$\Phi(x) - \Phi(x_0) - D\Phi(x_0)(x - x_0)$ देता है
$\norm{\Phi(x) - \Phi(x_0) - D\Phi(x_0)(x-x_0)}_\infty \leq
\varepsilon\norm{x - x_0}_\infty \leq \varepsilon r $, और
$ D\Phi(x_0)^{-1}$ इस दोष को एक में खींचता है
$\varepsilon\norm{D\Phi(x_0)^{-1}}\,r $-घन का विस्तार.
द्वारा \cref{thm:b3:product:linearchange},
\[
\lambda_d(\Phi(Q)) \leq \abs{\det D\Phi(x_0)}\,
\bigl(1 + \varepsilon\,C\bigr)^{d}\,\lambda_d(Q),
\qquad C = \sup_{Q}\norm{D\Phi(\cdot)^{-1}} .
\]

\emph{चरण दो: $(*)$ के लिए कॉम्पैक्ट क्यूब्स, उपविभाजन द्वारा।} होने देना $ Q
\subseteq U $ एक हो कॉम्पैक्ट घन और $\varepsilon > 0$. पर $ Q $,
$ D\Phi $ समान रूप से है \omterm{def:b3:topology:continuity}{निरंतर} और $\norm{D\Phi^{-1}}$
घिरा हुआ (कॉम्पैक्टनेस); प्रतिभाग $ Q $ में $2^{kd}$ उपकोश
$ Q_i $ इतना छोटा कि दोलन $ D\Phi $ प्रत्येक पर है
$\leq \varepsilon $. प्रत्येक उपघन पर चरण 1 (\omterm{ex:b3:groups:actions}{केंद्र} $ x_i $):
\[
\lambda_d(\Phi(Q)) \leq \sum_i\lambda_d(\Phi(Q_i))
\leq (1 + C\varepsilon)^d \sum_i \abs{\det
D\Phi(x_i)}\,\lambda_d(Q_i)
\leq (1 + C\varepsilon)^d\Bigl(\int_Q J + \varepsilon'\Bigr),
\]
अंतिम चरण क्योंकि $\sum_i\abs{\det D\Phi(x_i)}\mathbf
1_{Q_i} \to J $ समान रूप से चालू $ Q $ (निरंतरता का $\det D\Phi $)
--- रीमैन-सम तुलना। होने देना $\varepsilon \to 0$: $(*)$
के लिए रखता है कॉम्पैक्ट क्यूब्स.

\emph{चरण 3: $(*)$ सभी बोरेल के लिए $ A $.} सेट फ़ंक्शन $ A
\mapsto \lambda_d(\Phi(A))$, के बोरेल उपसमुच्चय पर $ U $, एक है
\omterm{def:b3:measure:measure}{उपाय} ($\Phi $ पर आपत्ति है $ V $ बोरेल सेट का संरक्षण
और गणनीय असम्बद्धता), और ऐसा ही है $ A \mapsto \int_AJ $.
का प्रत्येक खुला उपसमुच्चय $ U $ का एक गणनीय संघ है
लगभग-असंबद्ध डायडिक कॉम्पैक्ट क्यूब्स (मानक डायडिक
अपघटन: इसमें निहित अधिकतम डायडिक क्यूब्स लें \omterm{def:b3:topology:topology}{खुला
सेट}), और दोनों पैमाने उनके (सीमाओं) पार योगात्मक हैं
घन के हैं $\lambda_d $-शून्य, और उनका $\Phi $-छवियां हैं
चेहरे के पतले घन-आवरण पर लागू चरण 2 द्वारा शून्य):
$(*)$ क्यूब्स से गुजरता है खुले सेट. जनरल बोरेल $ A $:
निकास $ U $ द्वारा कॉम्पैक्ट कैमरों $ K_m \uparrow U $ साथ $ K_m \subseteq
\mathring K_{m+1}$, और ठीक करें $ m $; पर $\mathring K_{m+1}$, $ J $
कुछ से घिरा हुआ है $ M_m $. की बाहरी नियमितता से $\lambda_d $
(जैसा कि प्रमाण है \cref{thm:b3:measure:regularity}, बक्सों के साथ),
चुनें खुले सेट $ O_n $ साथ $ A \cap K_m \subseteq O_n
\subseteq \mathring K_{m+1}$ और $\lambda_d\bigl(O_n \setminus
(A\cap K_m)\bigr) \to 0$. तब
\[
\lambda_d\bigl(\Phi(A\cap K_m)\bigr) \leq
\lambda_d\bigl(\Phi(O_n)\bigr) \leq \int_{O_n}J
\leq \int_{A\cap K_m}J + M_m\,\lambda_d\bigl(O_n\setminus(A\cap
K_m)\bigr) \xrightarrow[n\to\infty]{} \int_{A\cap K_m}J .
\]
होने देना $ m \to \infty $: निरंतरता नीचे से बायीं ओर, एमसीटी चालू
सही. यह स्थापित करता है $(*)$.

\emph{चरण 4: समानता और अभिन्न सूत्र।} पहला
विस्तार $(*)$ सेट से इंटीग्रल तक: प्रत्येक के लिए \omterm{def:b3:lebesgue:measurable}{औसत दर्जे का} $ g
\geq 0$ पर $ V $,
\[
\int_V g(y)\,\dd y \leq \int_U g(\Phi(x))\,J(x)\,\dd x .
\tag{$**$}
\]
वास्तव में, के लिए $ g = \mathbf 1_B $ यह है $(*)$ साथ $ A =
\Phi^{-1}(B)$; रैखिकता इसे \omterm{def:b3:groups:simple}{सरल} तक विस्तारित करती है $ g $, और एमसीटी को
सब $ g \geq 0$ (मानक मशीन). अब आवेदन करें $(**)$ दो बार:
सबसे पहले $ g $, फिर --- भिन्नता के लिए $\Phi^{-1}$ ---
समारोह के लिए $ x \mapsto g(\Phi(x))J(x)$:
\[
\int_Vg \leq \int_U g(\Phi(x))J(x)\dd x
\leq \int_V g(y)\,J(\Phi^{-1}y)\,\abs{\det
D\Phi^{-1}(y)}\,\dd y = \int_V g ,
\]
तब से $ J(\Phi^{-1}y)\abs{\det D\Phi^{-1}(y)} = \abs{\det\bigl(
D\Phi(\Phi^{-1}y)\,D\Phi^{-1}(y)\bigr)} = 1$ (श्रृंखला नियम चालू
$\Phi\circ\Phi^{-1} = \mathrm{id}$). सारी असमानताएं हैं
समानताएँ: सूत्र लागू होता है $ g \geq 0$, और के लिए $ L^1$
विघटन द्वारा कार्य करता है।
\end{proof}

\begin{example}[ध्रुवीय निर्देशांक; गाऊसी फिर से]
\label{ex:b3:product:polar}
$\Phi(r, \theta) = (r\cos\theta, r\sin\theta)$ एक है $\mathcal
C^1$ भिन्नतारूपता से $\intoo0{+\infty}\times\intoo0{2\pi}$
पर $\R^2$ शून्य से एक आधी पंक्ति (शून्य सेट), साथ $\det D\Phi =
r $:\index{ध्रुवीय निर्देशांक}
\[
\int_{\R^2}f(x, y)\,\dd x\,\dd y
= \int_0^{2\pi}\!\!\int_0^{+\infty}
f(r\cos\theta, r\sin\theta)\,r\,\dd r\,\dd\theta .
\]
के लिए $ f = \eu^{-x^2-y^2}$, टोनेली और यह सूत्र देते हैं
\[
G^2 = \Bigl(\int_\R \eu^{-x^2}\dd x\Bigr)^2
= \int_{\R^2}\eu^{-x^2-y^2}
= 2\pi\int_0^\infty r\eu^{-r^2}\dd r = \pi:
\]
का शास्त्रीय दो-पंक्ति प्रमाण $ G = \sqrt\pi $, अब पूरी तरह से
उचित (पैरामीटर प्रमाण की तुलना करें
\cref{pb:b3:lebesgue:1}).
\end{example}

\begin{theorem}[यूनिट बॉल का आयतन]
\label{thm:b3:product:ballvolume}
होने देना $ v_d = \lambda_d(B(0,1))$ में $\R^d $. तब\index{की मात्रा
यूनिट बॉल}
\[
v_d = \frac{\pi^{d/2}}{\Gamma\bigl(\frac d2 + 1\bigr)} :
\qquad
v_1 = 2,\quad v_2 = \pi,\quad v_3 = \tfrac{4\pi}3,\quad
v_4 = \tfrac{\pi^2}2,\ \dots
\]
\end{theorem}

\begin{proof}
गणना करना $ I = \int_{\R^d}\eu^{-\norm x_2^2}\dd\lambda_d $ दो बार।
टोनेली द्वारा यह कारक हैं: $ I = G^d = \pi^{d/2}$. परत केक द्वारा
सूत्र (\cref{prop:b3:product:layercake}) साथ $ f = \eu^{-
\norm x^2}$, जिनके स्तर सेट गेंदें हैं: $\{f > t\} =
B\bigl(0, \sqrt{-\ln t}\bigr)$ के लिए $0 < t < 1$, का \omterm{def:b3:measure:measure}{उपाय}
$ v_d(-\ln t)^{d/2}$ (फैलाव द्वारा $\rho $ तराजू $\lambda_d $ द्वारा
$\rho^d $: \cref{thm:b3:product:linearchange}), इसलिए
\[
I = \int_0^1 v_d\,(-\ln t)^{d/2}\,\dd t
\overset{t = \eu^{-s}}{=}
v_d\int_0^\infty s^{d/2}\eu^{-s}\,\dd s
= v_d\,\Gamma\Bigl(\frac d2 + 1\Bigr).
\]
बराबरी करो. (मान: $\Gamma(\frac32) = \frac{\sqrt\pi}2$,
$\Gamma(2) = 1$, आदि) नोट $ v_d \to 0$ जैसा $ d \to \infty $ ---
सप्ताहांत समस्या स्टर्लिंग के माध्यम से कितनी तेजी से मापती है।
\end{proof}

\begin{exercise}[$\star $]\label{exo:b3:product:2}
इंटरचेंज को उचित ठहराएं और डिरिचलेट के इंटीग्रल को पुनः प्राप्त करें: के लिए
$ A > 0$,
\[
\int_0^A\frac{\sin x}x\,\dd x
= \int_0^A\!\!\int_0^{+\infty}\eu^{-xy}\sin x\,\dd y\,\dd x
= \int_0^{+\infty}\!\!\int_0^A \eu^{-xy}\sin x\,\dd x\,\dd y,
\]
\omterm{def:b3:topology:interior}{बंद} रूप में आंतरिक अभिन्न अंग की गणना करें, और जाने दें $ A \to
+\infty $ (हावी करें $ y $-अभिन्न) प्राप्त करना
$\int_0^\infty\frac{\sin x}x\dd x = \frac\pi2$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:product:3}
(ए) इसे साबित करें $ f \geq 0$ \omterm{def:b3:lebesgue:measurable}{औसत दर्जे का} और $\mu $ परिमित:

(बी) उसका निष्कर्ष निकालें $ f \in L^1(\mu)$ ($\mu $ परिमित) यदि

\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:product:4}

\int_0^1\!\!\int_0^1 f\,\dd y\,\dd x = \frac\pi4,
\qquad
\int_0^1\!\!\int_0^1 f\,\dd x\,\dd y = -\frac\pi4

\emph{(टिप्पणी $ f = \partial_y\bigl(\frac{y}{x^2+y^2}\bigr)$)},

अभिन्नता परिकल्पना सजावटी नहीं है.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:product:5}

$ f * g $ है \omterm{def:b3:topology:continuity}{निरंतर}. \emph{(डीसीटी के माध्यम से) निरंतरता का

\begin{exercise}[$\star\star $]\label{exo:b3:product:7}
जिसके लिए $ s > 0$ क्या निम्नलिखित परिमित हैं? के साथ औचित्य सिद्ध करें \omterm{ex:b3:product:polar}{ध्रुवीय

\int_{B(0,1)\subseteq\R^2}\frac{\dd x\,\dd y}{(x^2 +

\begin{exercise}[$\star\star\star $]\label{exo:b3:product:8}
(बीटा--गामा) के लिए $ p, q > 0$, होने देना $ B(p, q) = \int_0^1t^{p-1}(1
- t)^{q-1}\dd t $. से शुरू $\Gamma(p)\Gamma(q)$ एक के रूप में
दोहरा अभिन्न, स्थानापन्न $(x, y) = (uv,\, u(1 - v))$ (ए
खुले चतुर्थांश की भिन्नता
$\intoo0\infty\times\intoo01$; इसके जैकोबियन की गणना करें $= u $) और
निष्कर्ष\index{बीटा फ़ंक्शन}
\[
B(p, q) = \frac{\Gamma(p)\,\Gamma(q)}{\Gamma(p + q)} .
\]
परिणाम निकालना $\int_0^{\pi/2}\sin^{2p-1}\theta\cos^{2q-1}\theta\,
\dd\theta = \frac12B(p,q)$ और वालिस का मूल्य
अभिन्न $ W_n = \int_0^{\pi/2}\sin^n $.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:product:9}
(स्थानांतरण सूत्र) चलो $ T \colon (X, \mathcal A, \mu) \to (Y,
\mathcal B)$ होना \omterm{def:b3:lebesgue:measurable}{औसत दर्जे का} और $ T_*\mu(B) = \mu(T^{-1}(B))$
 \emph{आगे की ओर धकेलने का उपाय}\index{आगे की ओर धकेलने का उपाय}.
इसे हर एक के लिए दिखाएँ \omterm{def:b3:lebesgue:measurable}{औसत दर्जे का} $ g \geq 0$ पर $ Y $:
\[
\int_Y g\,\dd(T_*\mu) = \int_X g\circ T\,\dd\mu
\]
(मानक मशीन). फिर तुलना करें
\cref{thm:b3:product:linearchange}: अतिरिक्त जानकारी क्या करती है
चर सूत्र का परिवर्तन वह सार ले जाता है
स्थानांतरण सूत्र नहीं है? \emph{(ट्रांसफर फॉर्मूला कभी नहीं
पहचानता है $ T_*\mu $; चर प्रमेय के परिवर्तन की गणना करता है
$ T_*\lambda_d $ स्पष्ट रूप से एक के रूप में \omterm{ex:b3:lebesgue:gamma}{घनत्व} \omterm{def:b3:measure:measure}{उपाय}.)}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b3:product:10}
(गाऊसी क्षण) का उपयोग करना \omterm{ex:b3:product:polar}{ध्रुवीय निर्देशांक} और फ़ुबिनी, गणना करें
मानक गाऊसी भार के लिए $\R^d $:
\[
\int_{\R^d}\norm x_2^2\;\eu^{-\norm x_2^2}\,\dd x
\qquad\text{and}\qquad
\int_{\R^d}x_1^2\,\eu^{-\norm x^2_2}\,\dd x,
\]
स्थिरता की जाँच करें ($\norm x^2 = \sum x_i^2$), और निष्कर्ष निकालें
का दूसरा क्षण \omterm{def:b3:measure:measure}{उपाय}
$\pi^{-d/2}\eu^{-\norm x^2}\dd x $.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:product:11}
(ग्राफ और हाइपोग्राफ) आइए $ f \colon \R^d \to \intco0\infty $
होना \omterm{def:b3:lebesgue:measurable}{औसत दर्जे का}.
(ए) दिखाएँ कि \emph{हाइपोग्राफ} $ H = \{(x, y) \in
\R^d\times\R : 0 < y < f(x)\}$ है \omterm{def:b3:lebesgue:measurable}{औसत दर्जे का} में
$\R^{d+1}$ साथ
\[
\lambda_{d+1}(H) = \int_{\R^d}f\,\dd\lambda_d :
\]
``अभिन्न ग्राफ़ के अंतर्गत आने वाला क्षेत्र है'', अंत में
प्रमेय. \emph{(अनुभाग; टोनेली।)}
(बी) वह ग्राफ दिखाएँ $\{(x, f(x)) : x \in \R^d\}$ एक है
का शून्य सेट $\R^{d+1}$.
(सी) गोला होने का दो-पंक्ति प्रमाण निकालें $ S^{d-1}$ है
लेब्सग्यू-नल इन $\R^d $.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:product:12}
(एक प्रसिद्ध दोहरा समाकलन) ज्यामितीय श्रृंखला का उपयोग करना और
टोनेली चालू $\intoo01^2$, सिद्ध करना
\[
\int_0^1\!\!\int_0^1\frac{\dd x\,\dd y}{1 - xy}
= \sum_{n\geq1}\frac1{n^2} = \zeta(2),
\qquad
\int_0^1\!\!\int_0^1\frac{\dd x\,\dd y}{1 + xy}
= \sum_{n\geq1}\frac{(-1)^{n-1}}{n^2} = \frac{\zeta(2)}2 .
\]
(दूसरी श्रृंखला की पहचान: विभाजित सम और विषम सूचकांक।)
साथ $\zeta(2) = \frac{\pi^2}6$ (\cref{ch:b3:spectral}), दो
मासूम दिखने वाले अभिन्न अंग का मूल्यांकन करते हैं $\frac{\pi^2}6$ और
$\frac{\pi^2}{12}$; वास्तव में टोनेली की सकारात्मकता कहां है
परिकल्पना अपना कार्य करती है?
\end{exercise}

\section{समस्या: स्टर्लिंग का सूत्र}

\begin{problem}[{सप्ताहांत समस्या --- $ n! \sim
\sqrt{2\pi n}\,(n/\eu)^n $, प्रभुत्व अभिसरण द्वारा}]
\label{pb:b3:product:1}
स्टर्लिंग का सूत्र इस पुस्तक में प्रत्येक स्पर्शोन्मुख गणना को नियंत्रित करता है
--- गेंद की मात्रा, द्विपद गुणांक, \omterm{ex:b3:groups:actions}{केंद्र}ीय सीमा
प्रमेय का स्थानीय रूप. हम इसे से साबित करते हैं $\Gamma $ अभिन्न
(\cref{ex:b3:lebesgue:gamma}) साथ \emph{लाप्लास विधि},
इसके सबसे स्वच्छ प्रभुत्व-अभिसरण रूप में, फिर एकत्र करें
लाभांश\index{स्टर्लिंग का सूत्र}

\medskip
\noindent\textbf{Part I --- The formula.} के लिए $ t > 0$,
$\Gamma(t + 1) = \int_0^\infty x^{t}\eu^{-x}\dd x $.
\begin{enumerate}
  \item विकल्प $ x = t + \sqrt t\,u $ और दिखाओ
        \[
        \frac{\Gamma(t+1)}{t^{t}\eu^{-t}\sqrt t}
        = \int_{-\sqrt t}^{+\infty}
        \exp\Bigl(t\ln\Bigl(1 + \frac u{\sqrt
        t}\Bigr) - \sqrt t\,u\Bigr)\,\dd u
        \;=\;\int_\R g_t(u)\,\dd u,
        \]
        कहाँ $ g_t(u) = \exp\bigl(t\ln(1 + u/\sqrt t) -
        \sqrt t\,u\bigr)\mathbf 1_{u > -\sqrt t}$.
  \item बिंदुवार सीमा दिखाएँ: प्रत्येक निश्चित के लिए $ u $, $ g_t(u)
        \to \eu^{-u^2/2}$ जैसा $ t \to +\infty $ \emph{(बढ़ाना
        $\ln(1 + h)$ दूसरे क्रम तक)}.
  \item प्रभुत्व. होने देना $\varphi(h) = \ln(1 + h) - h $, ताकि
        $ g_t(u) = \exp\bigl(t\,\varphi(u/\sqrt t)\bigr)$ के लिए
        $ u > -\sqrt t $. दो सीमाएं सिद्ध करें
        \[
        \varphi(h) \leq -\frac{h^2}4 \quad (-1 < h \leq 1),
        \qquad
        \varphi(h) \leq -c\,h \quad (h \geq 1),\ \ c = 1 -
        \ln 2 > 0
        \]
        \emph{(अध्ययन $\varphi(h) + \frac{h^2}4$ और $\varphi(h)
        + ch $: डेरिवेटिव की गणना करें और साइन ऑन की जांच करें
        प्रत्येक रेंज)}. निष्कर्ष निकालें, के लिए $ t \geq 1$:
        \[
        g_t(u) \leq \eu^{-u^2/4}\ \ (\abs u \leq \sqrt t),
        \qquad
        g_t(u) \leq \eu^{-cu}\ \ (u \geq \sqrt t),
        \]
        ताकि $ g_t(u) \leq \eu^{-u^2/4} +
        \eu^{-cu}\,\mathbf 1_{u > 0}$: एक \omterm{def:b3:lebesgue:l1}{समाकलनीय} प्रभुत्व रखने वाला
        से स्वतंत्र $ t \geq 1$.
  \item प्रभुत्व अभिसरण प्रमेय और के साथ निष्कर्ष निकालें
        गाऊसी इंटीग्रल (\cref{ex:b3:product:polar}):
        \[
        \Gamma(t + 1) \;\sim\;
        \sqrt{2\pi t}\;\Bigl(\frac t\eu\Bigr)^{t}
        \qquad (t \to +\infty),
        \]
        खास तरीके से $ n! \sim \sqrt{2\pi
        n}\,(n/\eu)^n $.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Dividends.}
\begin{enumerate}[resume]
  \item (वालिस) से \cref{exo:b3:product:8}, $ W_{2n} =
        \frac\pi2\binom{2n}n4^{-n}$-प्रकार के सूत्र: व्युत्पन्न
        $\binom{2n}{n} \sim \frac{4^n}{\sqrt{\pi n}}$ से
        स्टर्लिंग, और पुनरावृत्ति के विरुद्ध इसकी जाँच करें $ W_{n} =
        \frac{n-1}nW_{n-2}$.
  \item (बॉल वॉल्यूम ढह गया) दिखाएँ
        \[
        v_d = \frac{\pi^{d/2}}{\Gamma(\frac d2 + 1)}
        \;\sim\; \frac{1}{\sqrt{\pi d}}
        \Bigl(\frac{2\pi\eu}{d}\Bigr)^{d/2},
        \]
        इसलिए $ v_d \to 0$ किसी भी ज्यामितीय अनुक्रम से तेज़;
        अधिकतम आयाम ज्ञात कीजिए $ v_d $ (संख्यात्मक रूप से: $ d =
        5$).
  \item (द्विपद की सघनता---का पूर्वावलोकन
        \cref{ch:b3:clt}) स्टर्लिंग का उपयोग करके स्थानीय दिखाएँ
        अनुमान, के लिए $ k = n/2 + s\sqrt n/2$ साथ $ s $ निश्चित
        और $ n $ यहां तक की:
        \[
        2^{-n}\binom{n}{k} \;\sim\;
        \sqrt{\frac{2}{\pi n}}\;\eu^{-s^2/2},
        \]
        असतत गॉसियन प्रोफ़ाइल: डी मोइवरे--लाप्लास इन
        भ्रूण.
  \item भाग I के प्रमाण का वास्तव में कहां उपयोग किया गया: (i) एमसीटी या
        डीसीटी; (ii) गॉसियन इंटीग्रल; (iii) अपरिवर्तनशीलता
        के गुण \omterm{def:b3:measure:lebesgueouter}{लेब्सेग माप}? एक-एक वाक्य.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The error term: Stirling with
bars.} तय करना $ d_n = \ln n! - \bigl(n + \tfrac12\bigr)\ln n + n -
\ln\sqrt{2\pi}$, तो वह भाग I कहता है $ d_n \to 0$.
\begin{enumerate}[resume]
  \item दिखाओ $ d_n - d_{n+1} = \bigl(n +
        \tfrac12\bigr)\ln\bigl(1 + \tfrac1n\bigr) - 1$.
  \item साथ $ t = \frac1{2n+1}$, सत्यापित करें $\frac{n+1}n =
        \frac{1+t}{1-t}$ और विस्तार करें:
        \[
        d_n - d_{n+1} = \frac{t^2}3 + \frac{t^4}5 +
        \frac{t^6}7 + \cdots,
        \]
        और दोतरफा सीमाएँ निकालें
        \[
        \frac1{3(2n+1)^2} \;<\; d_n - d_{n+1} \;<\;
        \frac1{12n} - \frac1{12(n+1)} .
        \]
  \item टेलीस्कोप (का उपयोग करके) $ d_m \to 0$) और सुखद की जाँच करें
        \omterm{def:b3:galois:algebraic}{बीजगणितीय} पहचान $\frac1{3(2m+1)^2} >
        \frac1{12m+1} - \frac1{12(m+1)+1}$ के लिए $ m \geq 1$, को
        शास्त्रीय ब्रैकेटिंग प्राप्त करें
        \[
        \sqrt{2\pi n}\Bigl(\frac n\eu\Bigr)^n
        \eu^{1/(12n+1)} \;<\; n! \;<\;
        \sqrt{2\pi n}\Bigl(\frac n\eu\Bigr)^n\eu^{1/(12n)} .
        \]
  \item दो परिणाम: (ए) स्टर्लिंग की सापेक्ष त्रुटि
        सूत्र है $< 10^{-6}$ जैसे ही $ n \geq 83\,334$;
        (बी) अनुमान $100!$ हाथ से चार महत्वपूर्ण अंक
        ब्रैकेटिंग से ($100! \approx 9.3326\cdot
        10^{157}$), और एक की सटीकता पर संक्षेप में आश्चर्य हुआ
        एक बहुत ही सीमित पर स्पर्शोन्मुख सूत्र $ n $.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The Wallis route: Stirling
without the Gaussian.} ऐतिहासिक रूप से स्थिरांक $\sqrt{2\pi}$
वालिस से आया, गॉस से नहीं; यह भाग स्टर्लिंग को पुनः सिद्ध करता है
भाग I--II से स्वतंत्र, और इस प्रकार पुनः सिद्ध होता है
गाऊसी अभिन्न. होने देना $ W_n = \int_0^{\pi/2}\sin^n\theta\,
\dd\theta $.
\begin{enumerate}[resume]
  \item स्थापित करना $ W_n = \frac{n-1}nW_{n-2}$ (द्वारा एकीकृत करें
        भाग), \omterm{def:b3:topology:interior}{बंद} रूप
        \[
        W_{2n} = \frac\pi2\binom{2n}n4^{-n}, \qquad
        W_{2n+1} = \frac{4^n}{(2n+1)\binom{2n}n},
        \]
        और पहचान $ W_nW_{n-1} = \frac\pi{2n}$.
  \item की एकरसता से $(W_n)$ परिणाम निकालना
        $ W_{2n}/W_{2n+1} \to 1$, तब
        \[
        W_{2n} \sim \frac12\sqrt{\frac\pi n}
        \qquad\text{and}\qquad
        \binom{2n}n4^{-n}\sqrt n \longrightarrow
        \frac1{\sqrt\pi} :
        \]
        वालिस का प्रमेय, स्टर्लिंग के बिना प्राप्त किया गया।
  \item अकेले भाग III की दूरबीन द्वारा दिखाएँ (कोई मूल्य नहीं)।
        \omterm{def:b3:topology:continuity}{निरंतर} आवश्यकता की), वह $ e_n = \ln n! - (n +
        \frac12)\ln n + n $ किसी सीमा तक एकत्रित हो जाता है $\ell $;
        इसके तुल्य $ n! \sim K\,n^{n+1/2}\eu^{-n}$ साथ $ K =
        \eu^\ell > 0$ अभी तक पहचान नहीं हुई है.
  \item इस एसिम्प्टोटिक को इसमें डालें $\binom{2n}n4^{-n}\sqrt n $
        और प्रश्न 14 का उपयोग करके, एकमात्र संभव की पहचान करें
        मूल्य: $ K = \sqrt{2\pi}$. तर्क को इकट्ठा करें: भाग
        III--IV एक साथ दें \omterm{def:b3:complete:complete}{पूरा} का दूसरा प्रमाण
        स्टर्लिंग --- और इसलिए, भाग I का प्रतिस्थापन चला रहा हूँ
        पीछे की ओर, का एक स्वतंत्र मूल्यांकन
        $\int_\R\eu^{-u^2/2}\dd u = \sqrt{2\pi}$. दो स्तंभ,
        इनमें से कोई एक दूसरे का समर्थन करता है।
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Last dividends.}
\begin{enumerate}[resume]
  \item (पूर्ण स्थानीय प्रोफ़ाइल) पूर्णांकों के लिए $\abs j \leq
        K\sqrt n $ ($ K $ निश्चित), दिखाएँ
        \[
        \frac{\binom{2n}{n+j}}{\binom{2n}{n}} =
        \prod_{i=1}^{\abs j}\frac{n - i + 1}{n + i}
        = \exp\Bigl(-\frac{j^2}n +
        O\Bigl(\frac1{\sqrt n}\Bigr)\Bigr),
        \]
        समान रूप से में $ j $ \emph{(लघुगणक लें और उपयोग करें $\ln
        \frac{1-x}{1+y} = -(x + y) + O(x^2 + y^2)$)}. ये है
        प्रश्न 7 का दोतरफा संस्करण और सटीक
        में उद्धृत अनुमान \cref{ch:b3:clt}सप्ताहांत की समस्या.
  \item (एक पॉइसन पूर्वावलोकन) स्टर्लिंग के साथ दिखाएँ
        $\eu^{-n}\dfrac{n^n}{n!} \sim \dfrac1{\sqrt{2\pi n}}$:
        बड़े माध्य के पॉइसन नियम का तरीका $ n $ ले जाता है
        द्रव्यमान $\approx (2\pi n)^{-1/2}$, बिल्कुल \omterm{ex:b3:groups:actions}{केंद्र}ीय के रूप में
        सीमा प्रमेय भविष्यवाणी करेगा.
  \item (गामा अनुपात) के लिए $ a \in \intoo01$, सिद्ध करना
        $\dfrac{\Gamma(n + a)}{\Gamma(n)\,n^a} \to 1$ का उपयोग कर रहे हैं
        लॉग-उत्तलता ढलान सीमा
        \cref{pb:b3:lebesgue:1} (वहां प्रश्न 14), और
        प्रत्येक वास्तविक तक विस्तार करें $ a > 0$ कार्यात्मक द्वारा
        समीकरण. (यह वही है ``$\Gamma(t+1) \sim $
        स्टर्लिंग'' का अर्थ पूर्णांकों के बीच है।)
  \item (गेंदें, दोहराना) से $ v_d = \pi^{d/2}/\Gamma(\frac d2
        + 1)$: सारणीबद्ध करें $ v_1, \dots, v_7$ बिल्कुल, सत्यापित करें
        एकरूपता के माध्यम से $\frac{v_d}{v_{d-2}} = \frac{2\pi}d $
        (बढ़ते समय $ d < 2\pi $, के बाद घट रही है), और
        हड़ताली उत्पन्न करने वाली पहचान साबित करें
        \[
        \sum_{k\geq0}v_{2k}\,x^{2k} = \eu^{\pi x^2} :
        \]
        सभी सम-आयामी यूनिट-बॉल वॉल्यूम को एक में पैक किया गया
        घातीय.
  \item (एंट्रॉपी एसिम्प्टोटिक्स) निश्चित के लिए $\alpha \in \intoo01$
        साथ $\alpha n \in \N $, स्टर्लिंग से निष्कर्ष निकालें
        \[
        \binom{n}{\alpha n} \;\sim\;
        \frac{\eu^{n\,H(\alpha)}}
        {\sqrt{2\pi\,\alpha(1-\alpha)\,n}},
        \qquad
        H(\alpha) = -\alpha\ln\alpha -
        (1-\alpha)\ln(1-\alpha) :
        \]
        द्विपद गुणांकों की घातीय वृद्धि दर
        है \emph{एन्ट्रापी} $ H $ --- जाँच करें कि $\alpha =
        \frac12$ प्रश्न 5 को पुनर्प्राप्त करता है, और वह $ H(\alpha) <
        \ln2$ के लिए $\alpha \neq \frac12$ (इसलिए \omterm{ex:b3:groups:actions}{केंद्र} से बाहर
        द्विपद तेजी से नगण्य हैं $2^n $).
  \item (सतह क्षेत्र) इकाई क्षेत्र का क्षेत्रफल $ S^{d-1}$
        है $ s_{d-1} = d\,v_d $ (जैसा सिद्ध हुआ
        \cref{exo:b3:forms:6} विभेदक-रूपों में
        अध्याय; यहां, इसे परिभाषा के रूप में लें)। समतल
        $ s_0, \dots, s_6$, अधिकतम एक का पता लगाएं ($ d - 1 =
        6$, $ s_6 = \frac{16\pi^3}{15} \approx 33.07$), और दिखाओ $ s_{d-1} \to 0$ सुपर-ज्यामितीय रूप से भी ---
        उच्च-आयामी गोले, प्रत्येक यूक्लिडियन द्वारा होते हैं
        पैमाना, लुप्त हो रहा छोटा।
  \item (पहला सुधार पद) से निष्कर्ष निकालें
        प्रश्न 11 का कोष्ठक $ d_n = \frac1{12n} +
        O\bigl(\frac1{n^2}\bigr)$, इस तरह
        \[
        n! = \sqrt{2\pi n}\,\Bigl(\frac
        n\eu\Bigr)^{n}\Bigl(1 + \frac1{12n} +
        O\Bigl(\frac1{n^2}\Bigr)\Bigr).
        \]
        पर सत्यापित करें $ n = 10$: नंगे सूत्र देता है
        $3\,598\,696$ (सापेक्ष त्रुटि $8.3\cdot10^{-3}$),
        सही किया हुआ $3\,628\,685$ ख़िलाफ़ $10! =
        3\,628\,800$ (सापेक्ष त्रुटि $3.2\cdot10^{-5}$) ---
        श्रृंखला का एक पद ढाई अंक खरीदता है।
  \item (माध्यिका) $\Gamma $) बताते हैं कि
        \[
        \frac{1}{\Gamma(t+1)}
        \int_0^{t} x^{t}\eu^{-x}\,\dd x
        \;\longrightarrow\; \frac12
        \qquad (t \to +\infty) :
        \]
        स्पर्शोन्मुख रूप से, के द्रव्यमान का बिल्कुल आधा
        $\Gamma $ इंटीग्रैंड इसके मोड के नीचे बैठता है $ x = t $.
        \emph{(भाग I के प्रतिस्थापन को काटे गए भाग पर चलाएँ
        अभिन्न; प्रश्न 3 का प्रभुत्वकर्ता पहले से ही अंदर है
        जगह.)}
  \item (एन्ट्रॉपी, गैर-स्पर्शोन्मुख रूप से) के लिए $\alpha \in
        \intoc0{\frac12}$ बाध्य साबित करें, के लिए वैध
        \emph{प्रत्येक} $ n \geq 1$:
        \[
        \sum_{k=0}^{\lfloor\alpha n\rfloor}\binom nk
        \;\leq\; \eu^{n\,H(\alpha)} ,
        \]
        योग की तुलना करके $\sum_k\binom
        nk\lambda^{k-\alpha n}$ झुकाव के लिए $\lambda =
        \frac{\alpha}{1-\alpha} \leq 1$. जाँच करें कि यह
        का चुनाव $\lambda $ इष्टतम है, और इसके साथ सामंजस्य स्थापित करें
        प्रश्न 21: घातीय दर $ H(\alpha)$ का
        स्पर्शोन्मुख कथन एक-पंक्ति द्वारा प्राप्त किया जाता है
        बिल्कुल भी स्पर्शोन्मुखता के साथ असमानता।
\end{enumerate}
\end{problem}
