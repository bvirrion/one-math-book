\chapter{एक प्रिंसिपल आइडियल डोमेन पर मॉड्यूल}\label{ch:b3:modules}

ए पर रैखिक बीजगणित \emph{अँगूठी} फ़ील्ड के बजाय: यह छोटा
परिकल्पना का परिवर्तन बीजगणित के महान एकीकरण में से एक का निर्माण करता है
प्रमेय ए \omterm{def:b3:modules:module}{मॉड्यूल} ऊपर $\Z $ एक एबेलियन समूह है; ए \omterm{def:b3:modules:module}{मॉड्यूल} ऊपर
$ K[X]$ एंडोमोर्फिज्म से सुसज्जित एक सदिश स्थान है। द
के लिए संरचना प्रमेय \omterm{def:b3:modules:free}{अंतिम रूप से उत्पन्न} \omterm{def:b3:modules:module}{मॉड्यूल} एक से अधिक \omterm{def:b3:rings:pidufd}{पीआईडी}
इसलिए, एक ही झटके में, सभी को वर्गीकृत कर देता है \omterm{def:b3:modules:free}{अंतिम रूप से उत्पन्न}
एबेलियन समूह \emph{और} समानता तक सभी एंडोमोर्फिज्म ---
जॉर्डन की कमी, जो वर्ष 2 नाजुक प्रेरणों द्वारा प्राप्त की गई,
अपने सूक्ष्म सहोदर, द के साथ, एक परिणाम के रूप में सामने आता है
\emph{तर्कसंगत} विहित रूप, हर क्षेत्र पर मान्य। द
कम्प्यूटेशनल इंजन है \omterm{thm:b3:modules:smith}{स्मिथ सामान्य रूप में}, का एक अंकगणित
यूक्लिड के योग्य मैट्रिक्स।

लगातार, $ A $ एक क्रमविनिमेय वलय है, जल्द ही ए \omterm{def:b3:rings:pidufd}{पीआईडी}; ``\omterm{def:b3:modules:module}{मॉड्यूल}''
मतलब $ A $-मॉड्यूल.

\section{मॉड्यूल, मुफ़्त मॉड्यूल}

\begin{definition}\label{def:b3:modules:module}
एक \emph{$ A $-मॉड्यूल}\index{$ A $-मॉड्यूल} एक एबेलियन समूह है $(M, +)$
अदिश गुणन के साथ $ A \times M \to M $ को संतुष्ट करना
वेक्टर-स्पेस स्वयंसिद्ध: $ a(x + y) = ax + ay $, $(a + b)x = ax + bx $,
$(ab)x = a(bx)$, $1x = x $. सबमॉड्यूल, भागफल $ M/N $, रूपवाद
($ A $-रैखिक मानचित्र), प्रत्यक्ष योग $\bigoplus_i M_i $, और
समरूपता प्रमेय को शब्द दर शब्द परिभाषित और सिद्ध किया जाता है
वेक्टर रिक्त स्थान और एबेलियन समूह; विशेष रूप से $ M/\ker f \cong
\operatorname{im} f $ एक रूपवाद के लिए $ f $.
\end{definition}

\begin{example}\label{ex:b3:modules:examples}
तीन प्रेरक मामले.
\begin{enumerate}
  \item $ A = K $ एक फ़ील्ड: \omterm{def:b3:modules:module}{मॉड्यूल} सदिश स्थान हैं.
  \item $ A = \Z $: \omterm{def:b3:modules:module}{मॉड्यूल} बिल्कुल एबेलियन समूह हैं ($ nx $ है
        होने को मजबूर $ x + \dots + x $), सबमॉड्यूल उपसमूह हैं।
  \item $ A = K[X]$: ए \omterm{def:b3:modules:module}{मॉड्यूल} एक है $ K $-सदिश स्थल $ V $ इसके साथ $ K $-रेखीय मानचित्र $ u\colon x \mapsto X\cdot x $ ---
        इसके विपरीत, प्रत्येक जोड़ी $(V, u)$ साथ $ u \in \mathcal L(V)$
        एक बन जाता है $ K[X]$-मॉड्यूल द्वारा $ P \cdot x = P(u)(x)$. द
        सबमॉड्यूल बिल्कुल वही हैं $ u $-स्थिर उपस्थान.
\end{enumerate}
एक \omterm{def:b3:rings:ideal}{आदर्श} का $ A $ बिल्कुल एक सबमॉड्यूल है $ A $; ए \omterm{def:b3:rings:ideal}{भागफल अंगूठी}
$ A/I $ एक $ A $-मॉड्यूल. वेक्टर रिक्त स्थान के विपरीत, \omterm{def:b3:modules:module}{मॉड्यूल} हो सकता है
\emph{\omterm{def:b3:modules:torsion}{टोशन}}: में $\Z/6\Z $, तत्व $\bar 3 \ne 0$ मारा जाता है
द्वारा $2 \neq 0$.
\end{example}

\begin{definition}\label{def:b3:modules:free}
$ M $ है \emph{अंतिम रूप से उत्पन्न}\index{अंतिम रूप से उत्पन्न}
अगर $ M = Ax_1 + \dots + Ax_n $ कुछ के लिए $ x_i $. $ M $ है
\emph{मुक्त}\index{मुक्त} रैंक का $ n $ अगर $ M \cong A^n $,
यानी\ अगर यह है \emph{आधार} (एक उत्पादक परिवार जो है
$ A $-रैखिक रूप से स्वतंत्र)। प्रत्येक सूक्ष्मता से उत्पन्न $ M $ एक है
एक मुक्त का भागफल \omterm{def:b3:modules:module}{मॉड्यूल}: $(a_1, \dots, a_n) \mapsto \sum a_ix_i $
एमएपीएस $ A^n $ पर $ M $.
\end{definition}

\begin{proposition}[पद का अपरिवर्तनीय होना]\label{prop:b3:modules:rank}
अगर $ A \neq 0$ और $ A^m \cong A^n $, तब $ m = n $.
\end{proposition}

\begin{proof}
एक चुनें अधिकतम \omterm{def:b3:rings:ideal}{आदर्श} $\mathfrak m $ का $ A $ (\cref{thm:b3:rings:krull})
और सेट करें $ k = A/\mathfrak m $, एक मैदान. एक समरूपता $ f \colon A^m
\to A^n $ एमएपीएस $\mathfrak m A^m $ में $\mathfrak m A^n $
(रैखिकता), इसलिए भागफल की समरूपता उत्पन्न करता है
\[
A^m/\mathfrak m A^m \;\cong\; A^n/\mathfrak m A^n,
\qquad\text{i.e.}\qquad k^m \cong k^n
\]
जैसा $ k $-वेक्टर रिक्त स्थान (भागफल) $ A^m/\mathfrak m A^m $ है
द्वारा मारा गया $\mathfrak m $, इतना $ A $-कार्रवाई कारकों के माध्यम से $ k $;
मानक आधार की छवियाँ एक बनाती हैं $ k $-आधार)। आयाम
क्षेत्र पर सिद्धांत $ k $ देता है $ m = n $.
\end{proof}

\begin{theorem}[निःशुल्क मॉड्यूल के सबमॉड्यूल]\label{thm:b3:modules:submodule}
होने देना $ A $ एक हो \omterm{def:b3:rings:pidufd}{पीआईडी} और $ M \subseteq A^n $ एक सबमॉड्यूल. तब $ M $ है
\omterm{def:b3:modules:free}{मुक्त} रैंक का $\leq n $.
\end{theorem}

\begin{proof}
प्रेरण चालू $ n $. के लिए $ n = 1$: $ M $ एक \omterm{def:b3:rings:ideal}{आदर्श}, इसलिए $ M = (0)$
(\omterm{def:b3:modules:free}{मुक्त} रैंक का $0$) या $ M = dA \cong A $ ($ x \mapsto dx $ है
इंजेक्शन: डोमेन). के लिए $ n > 1$: होने देना $\pi \colon A^n \to A $ हो
अंतिम समन्वय. तब $\pi(M)$ एक \omterm{def:b3:rings:ideal}{आदर्श}, $(0)$ या $ dA $. अगर
$(0)$: $ M \subseteq A^{n-1} \times \{0\}$ और प्रेरण लागू होता है.
अन्यथा चुनें $ x_0 \in M $ साथ $\pi(x_0) = d $. प्रत्येक $ x \in M $
अनोखा लिखता है
\[
x = \underbrace{\frac{\pi(x)}{d}}_{\in A}\, x_0 +
\Bigl(x - \tfrac{\pi(x)}d x_0\Bigr),
\qquad x - \tfrac{\pi(x)}d x_0 \in M \cap \ker\pi
\]
($\pi(x) \in dA $, तो गुणांक में है $ A $). इस प्रकार $ M = Ax_0
\oplus (M \cap \ker \pi)$: योग तब से प्रत्यक्ष है $\pi(ax_0) =
ad = 0$ बल $ a = 0$. प्रेरण द्वारा $ M \cap \ker\pi \subseteq
\ker \pi \cong A^{n-1}$ है \omterm{def:b3:modules:free}{मुक्त} रैंक का $\leq n - 1$; साथ लगा हुआ
$ x_0$ (से स्वतंत्र $\ker\pi $ जैसा कि अभी देखा गया) का आधार देता है
$ M $ प्रमुखता का $\leq n $.
\end{proof}

\begin{remark}\label{rem:b3:modules:presentation}
नतीजतन, एक से अधिक \omterm{def:b3:rings:pidufd}{पीआईडी} प्रत्येक \omterm{def:b3:modules:free}{अंतिम रूप से उत्पन्न} \omterm{def:b3:modules:module}{मॉड्यूल} $ M $ एक
\emph{परिमित प्रस्तुति}: एक अनुमान $\varphi\colon A^n \to M $
है \omterm{def:b3:modules:free}{मुक्त} आधार के साथ कर्नेल $ c_1, \dots, c_k $ ($ k \leq n $), और $ M
\cong A^n / \operatorname{im}(C)$ कहाँ $ C \in M_{n,k}(A)$ है
मैट्रिक्स जिसके कॉलम हैं $ c_j $. समझ $ M $ मतलब
एक को समझना \emph{मैट्रिक्स खत्म $ A $} में आधार परिवर्तन तक
स्रोत और लक्ष्य --- अगले भाग का विषय।
\end{remark}

\section{स्मिथ सामान्य रूप में}

\begin{definition}\label{def:b3:modules:equivalence}
दो आव्यूह $ B, C \in M_{n,k}(A)$ हैं \emph{समकक्ष} अगर $ C =
QBP $ साथ $ Q \in GL_n(A)$, $ P \in GL_k(A)$ (उलटा
\emph{ऊपर $ A $}: में निर्धारक $ A^\times $). समतुल्य
प्रेजेंटेशन मैट्रिसेस आइसोमोर्फिक को परिभाषित करते हैं \omterm{def:b3:modules:module}{मॉड्यूल} $ A^n/
\operatorname{im}$ (आधार बदलें $ A^n $ और $ A^k $).
\end{definition}

\begin{theorem}[स्मिथ सामान्य रूप में]\label{thm:b3:modules:smith}
होने देना $ A $ एक हो \omterm{def:b3:rings:pidufd}{पीआईडी} और $ B \in M_{n,k}(A)$. तब $ B $ के बराबर है
एक विकर्ण मैट्रिक्स\index{स्मिथ सामान्य रूप में}
\[
\operatorname{diag}(d_1, d_2, \dots, d_r, 0, \dots, 0),
\qquad d_1 \mid d_2 \mid \cdots \mid d_r \neq 0 ,
\]
और यह $ d_i $ तक अद्वितीय हैं \omterm{def:b3:rings:divisibility}{एसोसिएट्स}: $ d_1 \cdots d_i $ एक है
की जी.सी.डी $ i \times i $ के नाबालिग $ B $ (विशेष रूप से वह जीसीडी है
तुल्यता का एक अपरिवर्तनीय)। $ d_i $ हैं \emph{अपरिवर्तनीय
कारक}\index{अपरिवर्तनीय
कारक} का $ B $.
\end{theorem}

\begin{proof}
\emph{अस्तित्व।} अगर $ B = 0$, हो गया। अन्यथा, के सेट पर विचार करें
\omterm{def:b3:rings:ideal}{आदर्श}ों $(b)$ के समतुल्य मैट्रिक्स की प्रविष्टियों द्वारा उत्पन्न $ B $;
तब से $ A $ है \omterm{def:b3:rings:noetherian}{नोथेरियन}, एक मैट्रिक्स चुनें $ B'$ के बराबर $ B $
और एक प्रविष्टि $ d $ का $ B'$ साथ $(d)$ \emph{\omterm{def:b3:rings:primemaximal}{अधिक से अधिक}} इस सेट में.
कदम $ d $ स्थिति के लिए $(1,1)$ पंक्ति और स्तंभ स्वैप द्वारा।

\emph{दावा करना: $ d $ की प्रत्येक प्रविष्टि को विभाजित करता है $ B'$.} सबसे पहले, कॉलम 1: यदि
$ b_{i1}$ का गुणज नहीं है $ d $, होने देना $ e = \gcd(d, b_{i1}) = ud +
vb_{i1}$ (बीézout), तो $(e) \supsetneq (d)$. $2 \times 2$
मैट्रिक्स ट्रिक: पंक्तियों पर कार्य करना $1$ और $ i $ द्वारा
\[
\begin{pmatrix} u & v \\ -\dfrac{b_{i1}}e & \dfrac de
\end{pmatrix}
\in GL_2(A)
\qquad
\Bigl(\det = \tfrac{ud + v b_{i1}}e = 1\Bigr)
\]
प्रविष्टि के साथ एक समतुल्य मैट्रिक्स उत्पन्न करता है $ e $ स्थिति में $(1,1)$:
यह की अधिकतमता का खंडन करता है $(d)$. इसलिए $ d $ स्तंभ को विभाजित करता है $1$,
और सममित रूप से पंक्ति $1$. पंक्ति 1 और के गुणजों को घटाना
कॉलम 1 उन्हें साफ़ करता है: $ B'$ के बराबर है $\begin{pmatrix} d &
0\\ 0 & B''\end{pmatrix}$. अगला, $ d $ प्रत्येक प्रविष्टि को विभाजित करता है $ b $ का
$ B''$: की पंक्ति जोड़ें $ b $ पंक्ति 1 से (एक प्रारंभिक ऑपरेशन; द
नई पहली पंक्ति में शामिल है $ d $ और $ b $-प्रविष्टियाँ), और दोहराएँ
स्तंभ-समाशोधन तर्क: एक गैर-एकाधिक फिर से सुधार करेगा
$(d)$. अब आकार पर शामिल करें: $ B''$, जिनकी सभी प्रविष्टियाँ हैं
से विभाज्य $ d $, का स्मिथ रूप है $\operatorname{diag}(d_2,
\dots)$ जिनकी प्रविष्टियाँ विभाज्य रहती हैं $ d $ (किसी की भी प्रत्येक प्रविष्टि
$ QB''P $ एक $ A $-की प्रविष्टियों का संयोजन $ B''$); तय करना $ d_1 =
d $.

\emph{विशिष्टता.} होने देना $ D_i(B)$ सभी की एक जीसीडी निरूपित करें $ i \times i $
नाबालिग. पंक्ति और स्तंभ संचालन, और अधिक सामान्यतः
से गुणा \emph{कोई} मैट्रिक्स, gcd को छोटा नहीं कर सकता:
$ i \times i $ के नाबालिग $ QB $ हैं $ A $-उनका संयोजन $ B $
(कॉची--बिनेट विस्तार; या सीधे: की प्रत्येक पंक्ति $ QB $ एक है
की पंक्तियों का संयोजन $ B $, और माइनर पंक्तियों में बहुरेखीय हैं)।
इसलिए $ D_i(QBP)$ और $ D_i(B)$ एक दूसरे को बांटें: $ D_i $ एक है
तुल्यता अपरिवर्तनीय. विकर्ण रूप पर, शून्येतर $ i
\times i $ अवयस्क के उत्पाद हैं $ i $ की $ d_j $'s, और
विभाज्यता $ d_1 \mid \dots \mid d_r $ बनाता है $ d_1 \cdots d_i $ द
जी.सी.डी. इस तरह $ d_1 \cdots d_i = D_i(B)$ इकाइयों तक, और $ d_i =
D_i/D_{i-1}$ निर्धारित है.
\end{proof}

\begin{method}\label{met:b3:modules:smith}
एक से अधिक यूक्लिडियन डोमेन ($\Z $, $ K[X]$), स्मिथ रिडक्शन एक है
एल्गोरिदम --- अधिकतमता तर्क की आवश्यकता नहीं: की प्रविष्टि लाएं
स्थिति के अनुसार सबसे छोटा यूक्लिडियन आकार $(1,1)$; यदि यह विभाजित करने में विफल रहता है
इसकी पंक्ति या स्तंभ की कुछ प्रविष्टि, एक यूक्लिडियन डिवीजन छोड़ती है
वहां सख्ती से छोटा शेष --- इसे स्वैप करें और पुनरारंभ करें
(समाप्ति: आकार में कमी); जब यह अपनी पूरी पंक्ति को विभाजित कर देता है और
स्तंभ, उन्हें साफ़ करें; यदि यह किसी आंतरिक प्रविष्टि को विभाजित करने में विफल रहता है, तो उसे जोड़ें
पंक्ति से पंक्ति $1$ और पुनः आरंभ करें; आंतरिक ब्लॉक पर पुनरावृत्ति करें। में
पूर्णांक आव्यूहों पर अभ्यास करें: गणना करें $ D_1 = \gcd $ प्रविष्टियों की,
$ D_2$, \dots\ छोटे आकारों के लिए माइनर्स के माध्यम से, या एल्गोरिथम चलाएँ।
\end{method}

\begin{example}[स्मिथ की कमी, पूरी तरह से]
\label{ex:b3:modules:smithworked}
कम करना $ M = \begin{pmatrix} 1 & 2 & 3\\ 4 & 5 & 6\\ 7 & 8 & 9
\end{pmatrix}$ ऊपर $\Z $. कोना $1$ सब कुछ बांट देता है:
इसकी पंक्ति और स्तंभ साफ़ करें ($ L_2 \leftarrow L_2 - 4L_1$, $ L_3
\leftarrow L_3 - 7L_1$, तब $ C_2 \leftarrow C_2 - 2C_1$, $ C_3
\leftarrow C_3 - 3C_1$):
\[
M \sim \begin{pmatrix} 1 & 0 & 0\\ 0 & -3 & -6\\ 0 & -6 & -12
\end{pmatrix} .
\]
भीतरी खंड में, कोना $-3$ सभी प्रविष्टियों को विभाजित करता है: $ L_3
\leftarrow L_3 - 2L_2$ और $ C_3 \leftarrow C_3 - 2C_2$ इसे साफ़ करें
को $\operatorname{diag}(-3, 0)$. संकेतों को समायोजित करना (ए गुणा करें)।
पंक्ति द्वारा $-1$, एक कानूनी \omterm{def:b3:groups:action}{कार्रवाई}):
\[
M \sim \operatorname{diag}(1, 3, 0),
\qquad
\Z^3/M\Z^3 \cong \Z/3\Z \times \Z .
\]
निर्धारक विभाजकों द्वारा क्रॉस-चेक: $ D_1 =
\gcd(\text{entries}) = 1$; प्रत्येक $2\times2$ का नाबालिग $ M $ एक है
का एकाधिक $3$ (उदा.\ $\det\bigl(\begin{smallmatrix}1 & 2\\ 4
& 5\end{smallmatrix}\bigr) = -3$) और एक बराबर है $-3$: $ D_2 =
3$; $ D_3 = \det M = 0$. इस तरह $ d_1 = 1$, $ d_2 = 3$, $ d_3 = 0$:
वही उत्तर. दो पाठ: एक शून्य \omterm{thm:b3:modules:smith}{अपरिवर्तनीय कारक} रिकॉर्ड करता है
रैंक ड्रॉप (कोकर्नेल उठाता है)। \omterm{def:b3:modules:free}{मुक्त} $\Z $ सारांश), और
विभाज्यता श्रृंखला $1 \mid 3 \mid 0$ स्मिथ प्रमाणपत्र है
--- एक विकर्ण कमी जो श्रृंखला का उल्लंघन करती है (मान लीजिए
$\operatorname{diag}(2, 3)$, जो लापरवाह पैदा कर सकता है
से $\bigl(\begin{smallmatrix}2 & 0\\ 0 &
3\end{smallmatrix}\bigr)$ बहुत जल्दी रुककर: सही स्मिथ
प्रपत्र $\operatorname{diag}(1, 6)$, जैसा $ D_1 = 1$ यहाँ!) नहीं है
समाप्त.
\end{example}

\section{संरचना प्रमेय}

\begin{definition}\label{def:b3:modules:torsion}
होने देना $ A $ एक डोमेन बनें और $ M $ एक $ A $-मॉड्यूल.
\emph{टोशन}\index{टोशन} सबमॉड्यूल है
\[
T(M) = \{x \in M : ax = 0 \text{ for some } a \neq 0\}
\]
(एक सबमॉड्यूल: यदि $ ax = by = 0$ तब $ ab(x + y) = 0$, $ ab \ne 0$).
$ M $ है \emph{मरोड़ \omterm{def:b3:modules:free}{मुक्त}} अगर $ T(M) = 0$, ए \emph{टोशन \omterm{def:b3:modules:module}{मॉड्यूल}}
अगर $ T(M) = M $.
\end{definition}

\begin{theorem}[एक से अधिक परिमित रूप से उत्पन्न मॉड्यूल की संरचना
पीआईडी]\label{thm:b3:modules:structure}
होने देना $ A $ एक हो \omterm{def:b3:rings:pidufd}{पीआईडी} और $ M $ ए \omterm{def:b3:modules:free}{अंतिम रूप से उत्पन्न} $ A $-मॉड्यूल. वहाँ
एक अद्वितीय अस्तित्व $ r \in \N $ और शून्येतर गैरइकाइयाँ $ d_1 \mid d_2 \mid
\cdots \mid d_s $, तक अद्वितीय \omterm{def:b3:rings:divisibility}{एसोसिएट्स}, साथ\index{संरचना
मॉड्यूल के लिए प्रमेय}
\[
M \;\cong\; A^r \,\oplus\, A/(d_1) \oplus \cdots \oplus A/(d_s).
\]
इसके अतिरिक्त $ T(M) \cong A/(d_1)\oplus\dots\oplus A/(d_s)$ और $ M/T(M)
\cong A^r $: ए \omterm{def:b3:modules:free}{अंतिम रूप से उत्पन्न} \emph{मरोड़ \omterm{def:b3:modules:free}{मुक्त}} \omterm{def:b3:modules:module}{मॉड्यूल} एक से अधिक
\omterm{def:b3:rings:pidufd}{पीआईडी} है \omterm{def:b3:modules:free}{मुक्त}.
\end{theorem}

\begin{proof}
\emph{अस्तित्व।} उपस्थित $ M \cong A^n/\operatorname{im}(C)$
(\cref{rem:b3:modules:presentation}) और डाल दिया $ C $ स्मिथ फॉर्म में:
दो आधार परिवर्तन के बाद, $ M \cong A^n / (d_1A \times \dots
\times d_rA \times 0 \times \dots \times 0) \cong A/(d_1) \oplus
\dots \oplus A/(d_r) \oplus A^{\,n - r}$. कारकों को त्यागें
कहाँ $ d_i $ एक इकाई है ($ A/(d_i) = 0$); विभाज्यता श्रृंखला
जीवित रहता है.

\emph{ \omterm{def:b3:modules:torsion}{टोशन} पहचान.} विघटन में, $ A^r $ है
मरोड़ \omterm{def:b3:modules:free}{मुक्त} (एक डोमेन में कोई शून्य विभाजक नहीं है) और प्रत्येक $ A/(d_i)$
है \omterm{def:b3:modules:torsion}{टोशन} (द्वारा मारा गया $ d_i \ne 0$); एक सीधा योग विभाजित हो जाता है \omterm{def:b3:modules:torsion}{टोशन}
इसलिए: $ T(M) = \bigoplus_i A/(d_i)$ और $ M/T(M) \cong A^r $.

\emph{की विशिष्टता $ r $}: $ M/T(M) \cong A^r $ पर ही निर्भर करता है $ M $,
और \cref{prop:b3:modules:rank} पिंस $ r $ नीचे।

\emph{की विशिष्टता $ d_i $}: यह इलाज के लिए पर्याप्त है मरोड़
\omterm{def:b3:modules:module}{मॉड्यूल} $ T = T(M)$. प्रत्येक को विघटित करें $ d_i $ अभाज्य संख्याओं में और विभाजित करके
चीनी शेषफल प्रमेय (\cref{thm:b3:rings:crt}; विशिष्ट
अभाज्य संख्याएँ उत्पन्न होती हैं कॉमएक्सिमल \omterm{def:b3:rings:ideal}{आदर्श}):
\[
A/(d) \cong \bigoplus_{p \mid d} A/\bigl(p^{v_p(d)}\bigr):
\qquad
T \cong \bigoplus_{p} \bigoplus_{j} A/\bigl(p^{k_{p,j}}\bigr),
\]
 \emph{प्राथमिक विभाजक}\index{प्राथमिक विभाजक}
$ p^{k_{p,j}}$. इसके विपरीत $ d_i $ से पुनः निर्मित किये गये हैं
का मल्टीसेट \omterm{thm:b3:modules:structure}{प्राथमिक विभाजक} ($ d_s $ = उच्चतम का गुणनफल
प्रत्येक अभाज्य की शक्ति, आदि), इसलिए यह साबित करने के लिए पर्याप्त है
मल्टीसेट $\{k_{p,j}\}_j $ इसके द्वारा निर्धारित किया जाता है $ T $, प्रत्येक अभाज्य के लिए
$ p $. हल करना $ p $; के लिए $ j \geq 1$ पर विचार करें $ A/(p)$-वेक्टर रिक्त स्थान
$ p^{j-1}T/p^jT $. चक्रीय कारक पर $ A/(p^k)$:
\[
p^{j-1}\bigl(A/(p^k)\bigr)\big/p^{j}\bigl(A/(p^k)\bigr)
\cong
\begin{cases}
A/(p) & \text{if } j \leq k,\\
0 & \text{if } j > k,
\end{cases}
\]
और एक कारक पर $ A/(q^k)$, $ q \neq p $: से गुणा $ p $ है
वहाँ विशेषण ($ p $ उलटा मॉड $ q^k $: बीézout), तो
भागफल है $0$. प्रत्यक्ष रकम गुजरती है: $\dim_{A/(p)}
p^{j-1}T/p^jT = \#\{i : k_{p,i} \geq j\}$. ये आंतरिक
आयाम घातांकों के बहुसमूह को निर्धारित करते हैं।
\end{proof}

\begin{corollary}[अंतिम रूप से उत्पन्न एबेलियन समूह]
\label{cor:b3:modules:abelian}
प्रत्येक \omterm{def:b3:modules:free}{अंतिम रूप से उत्पन्न} एबेलियन समूह है $\Z^r \times
\Z/d_1\Z\times\dots\times\Z/d_s\Z $ साथ $ d_1 \mid \dots \mid
d_s $, विशिष्ट रूप से। प्रत्येक \emph{परिमित} एबेलियन समूह का एक उत्पाद है
प्राइम-पावर क्रम के चक्रीय समूह, एक के रूप में अद्वितीय
मल्टीसेट\index{एबेलियन समूह, संरचना}
\end{corollary}

\begin{example}\label{ex:b3:modules:abelian}
आदेश के एबेलियन समूह $ p^n $ के अनुरूप \emph{विभाजन}
का $ n $: के लिए $ p^4$: $\Z/p^4$, $\Z/p^3\times\Z/p $,
$(\Z/p^2)^2$, $\Z/p^2 \times (\Z/p)^2$, $(\Z/p)^4$ --- पाँच
समूह, जैसे $4$ पाँच विभाजन हैं। मिश्रित आदेश गुणा करते हैं
अभाज्य दर अभाज्य (सीआरटी) की गिनती होती है: हैं $5 \times 2$ एबेलियन
आदेश के समूह $2^4 \cdot 3^2 = 144$.
\end{example}

\section{अनुप्रयोग: एंडोमोर्फिज्म के विहित रूप}

होने देना $ K $ एक मैदान बनो, $ V $ ए $ K $-परिमित आयाम का वेक्टर स्थान
$ n $, और $ u \in \mathcal L(V)$; बनाना $ V $ ए $ K[X]$-मॉड्यूल के माध्यम से $ P
\cdot x = P(u)(x)$ (\cref{ex:b3:modules:examples}). यह \omterm{def:b3:modules:module}{मॉड्यूल}
है \omterm{def:b3:modules:free}{अंतिम रूप से उत्पन्न} (ए $ K $-आधार उत्पन्न होता है) और \omterm{def:b3:modules:torsion}{टोशन}: प्रत्येक के लिए $ x $, द $ n+1$ वैक्टर $ x, u(x), \dots, u^n(x)$ हैं
$ K $-निर्भर, एक गैर-शून्य विनाशकारी बहुपद प्रदान करता है।

\begin{definition}\label{def:b3:modules:companion}
के लिए $ P = X^m + a_{m-1}X^{m-1} + \dots + a_0$ मोनिक, द
\emph{साथी मैट्रिक्स}\index{साथी मैट्रिक्स} है
\[
C_P = \begin{pmatrix}
0 & & & -a_0\\
1 & \ddots & & -a_1\\
& \ddots & 0 & \vdots\\
& & 1 & -a_{m-1}
\end{pmatrix} :
\]
``गुणा का मैट्रिक्स $ X $'' पर $ K[X]/(P)$ में
आधार $1, \bar X, \dots, \bar X^{m-1}$.
\end{definition}

\begin{theorem}[फ्रोबेनियस: तर्कसंगत विहित रूप]
\label{thm:b3:modules:frobenius}
राक्षसी अचर बहुपदों का एक अनोखा क्रम है $ P_1
\mid P_2 \mid \dots \mid P_s $ (द \emph{समानता
अपरिवर्तनीय}\index{समानता
अपरिवर्तनीय} का $ u $) ऐसा कि, जैसे
$ K[X]$-मॉड्यूल,
\[
V \cong K[X]/(P_1) \oplus \cdots \oplus K[X]/(P_s):
\]
उपयुक्त आधार पर, $ u $ ब्लॉक-विकर्ण मैट्रिक्स है
$\operatorname{diag}(C_{P_1}, \dots, C_{P_s})$. इसके अतिरिक्त:
\begin{enumerate}
  \item $ P_s = \mu_u $ (न्यूनतम बहुपद) और $ P_1\cdots P_s =
        \chi_u $ (विशेषता बहुपद); विशेष रूप से $\mu_u
        \mid \chi_u $ \emph{(केली--हैमिल्टन ने पुनः सिद्ध किया)} और
        $\chi_u \mid \mu_u^{\,s}$, इसलिए $\chi_u $ और $\mu_u $ एक ही लें अलघुकरणीय कारक.
  \item दो एंडोमोर्फिज्म (या वर्ग मैट्रिक्स) समान हैं यदि
        उनके पास समान है \omterm{thm:b3:modules:frobenius}{समानता अपरिवर्तनीय}.
\end{enumerate}
\end{theorem}

\begin{proof}
संरचना प्रमेय (\cref{thm:b3:modules:structure}) लागू किया गया
को \omterm{def:b3:rings:pidufd}{पीआईडी} $ K[X]$: द मरोड़ \omterm{def:b3:modules:module}{मॉड्यूल} $ V $ के साथ विघटित हो जाता है
\omterm{thm:b3:modules:smith}{अपरिवर्तनीय कारक} $ P_i $, \omterm{def:b3:groups:normal}{सामान्य}ीकृत मोनिक (इकाइयाँ)। $ K[X]$ हैं
$ K^\times $); नहीं \omterm{def:b3:modules:free}{मुक्त} भाग होता है ($ V $ है \omterm{def:b3:modules:torsion}{टोशन}). प्रत्येक चक्रीय पर
कारक $ K[X]/(P_i)$, से गुणा $ X $ मैट्रिक्स है $ C_{P_i}$
की शक्तियों के आधार पर $\bar X $: आधारों को जोड़ने से मिलता है
ब्लॉक फॉर्म.

(1) विनाशक $ V = \bigoplus K[X]/(P_i)$ है $(P_1)\cap
\dots\cap(P_s) = (P_s)$ (विभाज्यता श्रृंखला: $ P_s $ एक आम बात है
एकाधिक, और का वर्ग $1$ अंतिम कारक मारा जाता है
बिल्कुल द्वारा $(P_s)$): $\mu_u = P_s $. के लिए $\chi_u $: चक्रीय पर
कारक, $\chi_{C_P} = P $, प्रेरण द्वारा $ m = \deg P $. विस्तार
$\det(XI_m - C_P)$ पहली पंक्ति के साथ (जिसकी प्रविष्टियाँ हैं $ X $, फिर
फिर शून्य $ a_0$ अंतिम कॉलम में):
\[
\det(XI_m - C_P) = X\,\det\bigl(XI_{m-1} - C_{\tilde P}\bigr)
+ (-1)^{1+m}\,a_0\,\det L ,
\]
कहाँ $\tilde P = X^{m-1} + a_{m-1}X^{m-2} + \dots + a_1$ (वही
आकार, एक आकार नीचे) और $ L $ विकर्ण के साथ त्रिकोणीय है
$(-1, \dots, -1)$, इसलिए $\det L = (-1)^{m-1}$. प्रेरण द्वारा
पहला पद है $ X\tilde P $, और दूसरा है $ a_0$: कुल है
$ X\tilde P + a_0 = P $ (बेस केस $ m=1$: $\det(X + a_0) = P $).
निर्धारक ब्लॉकों से गुणा होते हैं:
$\chi_u = \prod P_i $. केली-हैमिल्टन: $\chi_u \in (\mu_u)$
तब से $ P_s \mid $ प्रत्येक\dots\ इसके विपरीत प्रत्येक $ P_i \mid P_s $, इसलिए
$\chi_u = \prod P_i $ विभाजित $ P_s^{\,s} = \mu_u^s $; और $\mu_u =
P_s $ विभाजित $\chi_u $ इसके कारकों में से एक के रूप में।

(2) समान एंडोमोर्फिज्म संयुग्मी होते हैं \omterm{def:b3:modules:module}{मॉड्यूल} संरचनाएं, इसलिए
समान अपरिवर्तनीयता (विशिष्टता) है
\cref{thm:b3:modules:structure}); इसके विपरीत समान अपरिवर्तनीय
आइसोमोर्फिक दें $ K[X]$-मॉड्यूल, और ए \omterm{def:b3:modules:module}{मॉड्यूल} समरूपता है
बिल्कुल एक रैखिक आक्षेप जो दो एंडोमोर्फिज्म को आपस में जोड़ता है: ए
समानता.
\end{proof}

\begin{corollary}[समानता क्षेत्र विस्तार के प्रति असंवेदनशील है]
\label{cor:b3:modules:descent}
होने देना $ K \subseteq L $ फ़ील्ड बनें और $ M, N \in M_n(K)$. अगर $ M $ और
$ N $ ऊपर समान हैं $ L $, वे समान हैं $ K $.
\end{corollary}

\begin{proof}
 \omterm{thm:b3:modules:frobenius}{समानता अपरिवर्तनीय} का $ M $ स्मिथ के माइनर द्वारा गणना की जाती है
सूत्र (\cref{thm:b3:modules:smith}) प्रेजेंटेशन पर लागू किया गया
मैट्रिक्स $ XI_n - M $ ऊपर $ K[X]$ --- वास्तव में $ K[X]$-मॉड्यूल
$ V_M = K^n $ प्रेजेंटेशन है $ XI_n - M $: वो नक्शा $ K[X]^n \to V_M $,
$(Q_i) \mapsto \sum Q_i(M)e_i $, द्वारा उत्पन्न कर्नेल के साथ चालू है
के कॉलम $ XI_n - M $ (एक प्रत्यक्ष सत्यापन: \omterm{def:b3:modules:module}{मॉड्यूल}ो वो
कॉलम, प्रत्येक तत्व $ K[X]^n $ एक स्थिर सदिश में बदल जाता है,
और स्थिर सदिश वस्तुनिष्ठ रूप से मानचित्रित होते हैं; सप्ताहांत समस्या मंत्र
यह बाहर). बहुपदों की Gcd फ़ील्ड के अंतर्गत नहीं बदलती हैं
विस्तार: यदि $ d $ में मोनिक जीसीडी है $ K[X]$ एक परिवार का $(f_j)$,
बीézout देता है $ d = \sum u_jf_j $ साथ $ u_j \in K[X]$, तो हर
का \omterm{def:b3:groups:normal}{सामान्य} भाजक $ f_j $ \emph{में $ L[X]$} विभाजित $ d $; जैसा $ d $
यह स्वयं एक \omterm{def:b3:groups:normal}{सामान्य} भाजक है, यह जीसीडी है $ L[X]$ बहुत। इसलिए
द \omterm{thm:b3:modules:smith}{अपरिवर्तनीय कारक} का $ XI_n - M $, क्रमिक के भागफल
छोटी-मोटी जीसीडी, वही हैं $ K $ और खत्म होता है $ L $: $ M, N $ एक ही लें \omterm{thm:b3:modules:frobenius}{समानता अपरिवर्तनीय} ऊपर $ L $ अगर खत्म हो गया $ K $; द्वारा समाप्त करें
\cref{thm:b3:modules:frobenius}(2).
\end{proof}

\begin{theorem}[जॉर्डन फॉर्म, पुनः व्युत्पन्न]\label{thm:b3:modules:jordan}
कल्पना करना $\chi_u $ विभाजित हो जाता है $ K $ (उदा.\ $ K = \C $). के लिए आवेदन कर रहा हूँ
$ V $ प्रारंभिक-विभाजक अपघटन (का प्रमाण)
\cref{thm:b3:modules:structure}) के बजाय \omterm{thm:b3:modules:smith}{अपरिवर्तनीय कारक}:
\[
V \cong \bigoplus_{\lambda, j} K[X]\big/\bigl((X -
\lambda)^{k_{\lambda,j}}\bigr),
\]
और आधार में $\bigl(\overline{(X-\lambda)^{k-1}}, \dots,
\overline{(X - \lambda)}, \bar 1\bigr)$ प्रत्येक कारक का, $ u $ कार्य करता है
जॉर्डन ब्लॉक के रूप में $ J_k(\lambda)$: विभाजन के साथ प्रत्येक एंडोमोर्फिज्म
विशिष्ट बहुपद में जॉर्डन आधार और बहुसमुच्चय होता है
ब्लॉक $(\lambda, k)$ निराला है।\index{जॉर्डन फॉर्म}
\end{theorem}

\begin{proof}
 \omterm{thm:b3:modules:structure}{प्राथमिक विभाजक} की मरोड़ \omterm{def:b3:modules:module}{मॉड्यूल} $ V $ हैं
$(X - \lambda)^k $ साथ $ X - \lambda $ से लेकर अलघुकरणीय
के कारक $\mu_u $ (जो विभाजित हो जाता है, तब से $\chi_u $ करता है और दोनों
समान है अलघुकरणीय कारक,
\cref{thm:b3:modules:frobenius}). में $ W = K[X]/((X-\lambda)^k)$,
रखना $ f_j = \overline{(X - \lambda)^{k-j}}$ के लिए $ j = 1, \dots, k $:
तब $(X - \lambda)f_j = f_{j-1}$ (साथ $ f_0 = 0$), यानी\ $ u(f_j)
= \lambda f_j + f_{j-1}$: का मैट्रिक्स $ u $ पर $(f_1, \dots,
f_k)$ बिलकुल है $ J_k(\lambda)$ (विकर्ण के ऊपर वाले)।
के मल्टीसेट की विशिष्टता \omterm{thm:b3:modules:structure}{प्राथमिक विभाजक} है
\cref{thm:b3:modules:structure}.
\end{proof}

\begin{remark}\label{rem:b3:modules:overview}
विहित रूपों का पदानुक्रम अब पारदर्शी है:
\emph{तर्कसंगत} फॉर्म हर फ़ील्ड पर मौजूद है और पता लगाता है
बिल्कुल समानता (\cref{cor:b3:modules:descent});
\emph{जॉर्डन} रूप उसका परिष्कार है जब $\chi_u $ विभाजन
वर्ष 2 के आयाम-गणना जॉर्डन के प्रमेय के प्रमाण हैं
सम्मिलित: सभी कॉम्बिनेटरिक्स का अंकगणित था \omterm{def:b3:rings:pidufd}{पीआईडी}
$ K[X]$.
\end{remark}

\begin{exercise}[$\star $]\label{exo:b3:modules:2}
क्रम के एबेलियन समूहों की सूची बनाएं $360$ दोनों में समरूपता तक
प्रारंभिक-विभाजक और अपरिवर्तनीय-कारक रूप। कितने एबेलियन
आदेश के समूह $ p^5$ वहाँ हैं?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b3:modules:3}
की गणना करें \omterm{thm:b3:modules:smith}{स्मिथ सामान्य रूप में} ऊपर $\Z $ का
\[
B = \begin{pmatrix} 2 & 4\\ 6 & 8 \end{pmatrix},
\qquad
C = \begin{pmatrix} 2 & 0 & 0\\ 0 & 3 & 0\\ 0 & 0 & 12
\end{pmatrix},
\]
और एबेलियन समूहों की पहचान करें $\Z^2/B\Z^2$ और $\Z^3/C\Z^3$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:modules:4}
होने देना $ L \subseteq \Z^n $ रैंक का एक उपसमूह बनें $ n $ आधार के साथ
के कॉलम $ B \in M_n(\Z)$, $\det B \neq 0$. बताते हैं कि
$\Z^n/L $ प्रमुखता की परिमित है $\abs{\det B}$, ओर वो $\Z^n/L
\cong \prod_i \Z/d_i\Z $ के लिए \omterm{thm:b3:modules:smith}{अपरिवर्तनीय कारक} $ d_i $ का $ B $.
के साथ चित्रण करें $ L = \Z(2,0) + \Z(1,3)$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:modules:5}
होने देना $ A $ एक डोमेन हो. (ए) उसे सत्यापित करें $ T(M)$ एक सबमॉड्यूल है और
वह $ M/T(M)$ है मरोड़ \omterm{def:b3:modules:free}{मुक्त}. (बी) दिखाएँ कि \omterm{def:b3:rings:ideal}{आदर्श} $(X, Y)$
का $ K[X,Y]$, के तौर पर $ K[X,Y]$-मॉड्यूल, है मरोड़ \omterm{def:b3:modules:free}{मुक्त} लेकिन नहीं \omterm{def:b3:modules:free}{मुक्त}:
संरचना प्रमेय को वास्तव में इसकी आवश्यकता है \omterm{def:b3:rings:pidufd}{पीआईडी} परिकल्पना।
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:modules:6}
(ए) वह दिखाओ $2\Z $ में कोई प्रत्यक्ष पूरक नहीं है $\Z $-मॉड्यूल
$\Z $: के सबमॉड्यूल मुफ़्त \omterm{def:b3:modules:module}{मॉड्यूल} हैं \omterm{def:b3:modules:free}{मुक्त}
(\cref{thm:b3:modules:submodule}), लेकिन उन्हें सीधे सारांश की आवश्यकता है
नहीं होना. (बी) दिखाओ अगर $ M \subseteq A^n $ ($ A $ ए \omterm{def:b3:rings:pidufd}{पीआईडी}) संतुष्ट करता है:
$ A^n/M $ है मरोड़ \omterm{def:b3:modules:free}{मुक्त}, तब $ M $ \emph{है} एक सीधा सारांश.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:modules:7}
में हल करें $\Z^2$ प्रणाली
\[
\begin{cases}
2x + 4y \equiv b_1 \pmod{20},\\
6x + 8y \equiv b_2 \pmod{20},
\end{cases}
\]
किस जोड़े के लिए $(b_1, b_2)$ स्मिथ का उपयोग करके समाधान मौजूद हैं
का रूप \cref{exo:b3:modules:3} (चरों के व्युत्क्रमणीय परिवर्तन
दोनों तरफ)।
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:modules:8}
(ए) सभी का निर्धारण करें \omterm{thm:b3:modules:frobenius}{समानता अपरिवर्तनीय} और संभव है जॉर्डन बनता है
एक शून्यशक्तिशाली का $4 \times 4$ मैट्रिक्स, के विभाजन द्वारा क्रमबद्ध
$4$ उन्होंने अनुभव किया।
(बी) प्रदर्शन दो $4\times4$ उसी के साथ जटिल मैट्रिक्स
विशेषता \emph{और} न्यूनतम बहुपद जो नहीं हैं
समान, और इसके लिए सिद्ध करें $ n \leq 3$ यह तो हो न सकता।
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b3:modules:9}
होने देना $ u \in \mathcal L(V)$, $\dim V = n $. निम्नलिखित दिखाएँ
समतुल्य हैं: (i) $ V $ एक चक्रीय है $ K[X]$-मॉड्यूल (वहां है $ x $
साथ $ V = K[u]x $, ए \emph{चक्रीय सदिश}); (ii) $\mu_u = \chi_u $;
(iii) $ s = 1$ में \cref{thm:b3:modules:frobenius}. निष्कर्ष निकालें कि ए
\omterm{def:b3:modules:companion}{साथी मैट्रिक्स} एक चक्रीय वेक्टर है, और निर्धारित करें कि कब
विकर्ण मैट्रिक्स में एक है।
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b3:modules:10}
के लिए $ M \in M_n(\Z)$ की एंडोमोर्फिज्म के रूप में देखा जाता है $\Z^n $, साबित करो
सूचकांक सूत्र: यदि $\det M \ne 0$, तब
$[\Z^n : M\Z^n] = \abs{\det M}$, और उसका निष्कर्ष निकालो $ M \in GL_n(\Z)$
आईएफएफ $\det M = \pm 1$. आवेदन: समूह $\Z^2$ बिलकुल है
$\sigma_1(m) = \sum_{d \mid m} d $ सूचकांक के उपसमूह $ m $.
\emph{(आव्यूहों को हर्माइट रूप में गिनें $\bigl(\begin{smallmatrix}
a & b\\ 0 & d\end{smallmatrix}\bigr)$, $ ad = m $, $0 \leq b <
d $.)}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b3:modules:11}
(समीकरण $ x^k = e $ एबेलियन समूहों में) चलो $ G $ एक परिमित हो
एबेलियन समूह के साथ \omterm{thm:b3:modules:smith}{अपरिवर्तनीय कारक} $ d_1 \mid d_2 \mid \dots
\mid d_s $.
(ए) इसे प्रत्येक के लिए दिखाएं $ k \geq 1$,
\[
\#\{x \in G : x^k = e\} \;=\; \prod_{i=1}^{s}\gcd(k, d_i) .
\]
(बी) निष्कर्ष निकालें: एक परिमित एबेलियन समूह चक्रीय है यदि और केवल यदि
प्रत्येक के लिए $ k $, समीकरण $ x^k = e $ अधिकतम है $ k $
समाधान.
(सी) के परिमित उपसमूहों की चक्रीयता पुनर्प्राप्त करें $ K^\times $
($ K $ एक मैदान, \cref{ch:b3:galois}): बहुपद क्यों होता है
$ X^k - 1$ (बी) की कसौटी की गारंटी?
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b3:modules:12}
(प्राथमिक आव्यूह उत्पन्न करते हैं) (ए) वह दिखाओ $ M \in M_n(\Z)$
में उलटा है $ M_n(\Z)$ आईएफएफ $\det M = \pm1$.
(बी) वह दिखाओ $ SL_2(\Z)$ दो प्राथमिक द्वारा उत्पन्न होता है
matrices
$ E = \bigl(\begin{smallmatrix}1 & 1\\ 0 &
1\end{smallmatrix}\bigr)$ और
$ F = \bigl(\begin{smallmatrix}1 & 0\\ 1 &
1\end{smallmatrix}\bigr)$.
\emph{(के पहले कॉलम पर यूक्लिडियन एल्गोरिदम चलाएँ $ M
\in SL_2(\Z)$ की घातों द्वारा बाएँ गुणन द्वारा $ E, F $,
पहुँचना $\pm\bigl(\begin{smallmatrix}1 & *\\ 0 &
1\end{smallmatrix}\bigr)$; हाथ से समाप्त करें --- ध्यान दें
$-I = (EF^{-1}E)^2$.)}
(सी) स्मिथ रिडक्शन के साथ संबंध स्पष्ट करें: ओवर $\Z $,
निर्धारक की पंक्ति और स्तंभ संचालन $1$ के लिए पर्याप्त है
संकेतों तक विकर्ण करें।
\end{exercise}

\section{समस्या: कम्यूटेंट और डबल कम्यूटेंट}

\begin{problem}[{सप्ताहांत समस्या --- तर्कसंगत रूप, कम्यूटेंट,
द्वि-उत्परिवर्ती}]\label{pb:b3:modules:1}
होने देना $ K $ एक मैदान बनो, $ V $ ए $ K $-आयाम का वेक्टर स्थान $ n \geq
1$, और $ u \in \mathcal L(V)$. हम अध्ययन करते हैं \emph{कम्यूटेंट}
\[
\mathcal C(u) = \{v \in \mathcal L(V) : uv = vu\},
\]
का एक उपबीजगणित $\mathcal L(V)$ युक्त $ K[u] = \{P(u) : P \in
K[X]\}$, और हम फ्रोबेनियस के आयाम सूत्र और दोहरे को सिद्ध करते हैं
क्रमपरिवर्ती प्रमेय: $\mathcal C(\mathcal C(u)) = K[u]$. लगातार,
$ V $ है $ K[X]$-मॉड्यूल द्वारा परिभाषित $ u $, साथ \omterm{thm:b3:modules:smith}{अपरिवर्तनीय कारक}
$ P_1 \mid \cdots \mid P_s $ और चक्रीय अपघटन $ V =
\bigoplus_{i=1}^s V_i $, $ V_i = K[u]\,x_i \cong K[X]/(P_i)$, $ n_i
= \deg P_i $ (\cref{thm:b3:modules:frobenius}).

\medskip
\noindent\textbf{Part I --- The presentation matrix $ XI - M $, and
warm-ups.}
\begin{enumerate}
  \item होने देना $ M \in M_n(K)$ और चलो $\varphi \colon K[X]^n \to V_M
        = K^n $ भेजना $(Q_1, \dots, Q_n)$ को $\sum_i Q_i(M)e_i $.
        बताते हैं कि $\varphi $ का एक विशेषण रूपवाद है
        $ K[X]$-मॉड्यूल और वह प्रत्येक कॉलम $ XI_n - M $ में निहित है $\ker\varphi $.
  \item दिखाएँ कि, के कॉलम modulo $ XI_n - M $, हर
        का तत्व $ K[X]^n $ एक स्थिर सदिश के सर्वांगसम है
        \emph{(का उपयोग करके डिग्री कम करें $ Xe_i \equiv \sum_j m_{ji}e_j $)},
        और निष्कर्ष निकालें $\ker\varphi = (XI_n - M)\,K[X]^n $: द \omterm{def:b3:modules:module}{मॉड्यूल}
        $ V_M $ प्रेजेंटेशन मैट्रिक्स है $ XI_n - M $. वापस पाना
        \cref{cor:b3:modules:descent}का प्रारंभिक बिंदु:
        \omterm{thm:b3:modules:frobenius}{समानता अपरिवर्तनीय} का $ M $ गैर इकाई हैं \omterm{thm:b3:modules:smith}{अपरिवर्तनीय
        कारक} का $ XI_n - M $.
  \item की गणना करें \omterm{thm:b3:modules:frobenius}{समानता अपरिवर्तनीय} का: एक अदिश मैट्रिक्स
        $\lambda I_n $; भिन्न विकर्ण वाला एक विकर्ण मैट्रिक्स
        प्रविष्टियाँ; $ n \times n $ जॉर्डन ब्लॉक $ J_n(0)$;
        $\operatorname{diag}(J_2(0), J_1(0))$ के लिए $ n = 3$.
  \item बताते हैं कि $\dim K[u] = \deg \mu_u = n_s $.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Morphisms between cyclic \omterm{def:b3:modules:module}{मॉड्यूल}.}
\begin{enumerate}[resume]
  \item होने देना $ P, Q $ राक्षसी अस्थिर रहो. दिखाओ कि ए
        $ K[X]$-रूपवाद $ f \colon K[X]/(P) \to K[X]/(Q)$ इसके द्वारा निर्धारित किया जाता है $ f(\bar 1)$, ओर वो $\bar c \in K[X]/(Q)$
        के रूप में सेवा कर सकते हैं $ f(\bar 1)$ आईएफएफ $ P\bar c = 0$ में $ K[X]/(Q)$.
  \item परिणाम निकालना
        \[
        \operatorname{Hom}_{K[X]}\bigl(K[X]/(P),\, K[X]/(Q)\bigr)
        \;\cong\; K[X]\big/\bigl(\gcd(P, Q)\bigr),
        \]
        आयाम का $\deg \gcd(P, Q)$ ऊपर $ K $. \emph{(वह दिखाओ
        समाधान $\bar c $ का $ P\bar c = 0$ में $ K[X]/(Q)$
        द्वारा उत्पन्न चक्रीय सबमॉड्यूल का निर्माण करें
        $\overline{Q/\gcd(P,Q)}$.)}
  \item सिद्ध करना \emph{फ्रोबेनियस का सूत्र}:
        \[
        \dim_K \mathcal C(u) = \sum_{i,j=1}^{s} \deg\gcd(P_i,
        P_j) = \sum_{i=1}^{s} (2s - 2i + 1)\, n_i .
        \]
        \emph{(एक आवागमन $ v $ बिल्कुल एक है $ K[X]$-एंडोमोर्फिज्म का
        $ V $; घुलना $\operatorname{End}(\bigoplus_i V_i)$ जैसे
        आकारिकी के आव्यूह $ V_j \to V_i $ और उपयोग करें
        विभाज्यता श्रृंखला।)}
  \item परिणाम निकालना $\dim \mathcal C(u) \geq n $, समानता के साथ यदि $ u $
        चक्रीय है ($ s = 1$), और गणना करें $\dim\mathcal C(u)$ के लिए
        $ u = \lambda\,\mathrm{id}$: सूत्र के दोनों चरम.
  \item फ्रोबेनियस के सूत्र को सीधे सत्यापित करें
        $\operatorname{diag}(J_2(0), J_1(0))$ की गणना करके
        कम्यूटेंट स्पष्ट रूप से के रूप में $3\times3$ matrices.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The double commutant theorem.}
होने देना $ w \in \mathcal C(\mathcal C(u))$; हम साबित करते हैं $ w \in K[u]$.
\begin{enumerate}[resume]
  \item दिखाओ $ K[u] \subseteq \mathcal C(\mathcal C(u))$, और वह
        प्रत्येक $ w \in \mathcal C(\mathcal C(u))$ के साथ आवागमन करता है $ u $
        ---तो समावेशन सिद्ध हो,
        $\mathcal C(\mathcal C(u)) \subseteq K[u]$, एक वास्तविक है
        को तेज़ करना $ w \in \mathcal C(u)$.
  \item पहले मान लीजिए कि $ u $ है \emph{चक्रीय}, $ V = K[u]x $.
        उसे सीधे दिखाओ $\mathcal C(u) = K[u]$
        \emph{(यात्रा का मूल्यांकन करें $ v $ पर $ x $: $ v(x) = P(u)x $
        कुछ के लिए $ P $, और तुलना करें $ v $ साथ $ P(u)$ आधार पर
        $ u^k x $)}, और इस मामले में प्रमेय को समाप्त करें।
  \item \omterm{def:b3:groups:normal}{सामान्य} मामले पर वापस जाएँ। प्रत्येक के लिए $ i $, होने देना $\pi_i\colon
        V \to V_i $ अन्य सारांशों के साथ प्रक्षेपण बनें।
        दिखाओ $\pi_i \in \mathcal C(u)$, और उसका निष्कर्ष निकालो $ w $
        प्रत्येक को सुरक्षित रखता है $ V_i $ और साथ आवागमन करता है $ u_i =
        u\restriction_{V_i}$; लागू प्रश्न 11 के माध्यम से निष्कर्ष निकालें
        चक्रीय $ u_i $: बहुपद हैं $ Q_i $ साथ
        $ w\restriction_{V_i} = Q_i(u)\restriction_{V_i}$.
  \item इसे चिपकाना बाकी है $ Q_i $ एक बहुपद में. के लिए $ i
        \leq j $ (इसलिए $ P_i \mid P_j $), बताते हैं कि $\eta_{ij} \colon
        V_j \to V_i $, $ R(u)x_j \mapsto R(u)x_i $, एक है
        अच्छी तरह से परिभाषित $ K[X]$-रूपवाद \emph{(क्या जांचना चाहिए
        वह है $ R(u)x_j = 0$ तात्पर्य $ R(u)x_i = 0$)}, ओर वो
        $\tilde\eta_{ij} = \eta_{ij}\circ\pi_j $, द्वारा बढ़ाया गया $0$
        अन्य सारांशों पर, निहित है $\mathcal C(u)$.
  \item का उपयोग करते हुए $ w\tilde\eta_{ij} = \tilde\eta_{ij}w $, दिखाओ $ Q_i
        \equiv Q_j \pmod{P_i}$ के लिए $ i \leq j $. उसका निष्कर्ष निकालो $ Q =
        Q_s $
        संतुष्ट $ Q \equiv Q_i \pmod {P_i}$ सभी के लिए $ i $, इस तरह
        $ w = Q(u)$ सभी पर $ V_i $, इसलिए आगे $ V $:
        \[
        \boxed{\ \mathcal C(\mathcal C(u)) = K[u].\ }
        \]
  \item (कोडा) प्रमेय से निष्कर्ष निकालें: यदि $ v $ के साथ आवागमन करता है
        \emph{प्रत्येक} मैट्रिक्स के साथ आवागमन $ u $, और $ u $ है
        चक्रीय, तो $ v $ में एक बहुपद है $ u $; और एक दे दो
        उदाहरण दिखा रहा हूँ $\mathcal C(u) = K[u]$ के लिए विफल रहता है $ u =
        \mathrm{id}$, $ n \geq 2$ --- वास्तव में कहां होता है
        चक्रीयता दर्ज करें?
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Dividends of the \omterm{thm:b3:modules:frobenius}{समानता
अपरिवर्तनीय}.} तर्कसंगत विहित रूप एक मशीन है; यहाँ हैं
इसके पांच शास्त्रीय आउटपुट।
\begin{enumerate}[resume]
  \item (ट्रांसपोज़) दिखाएँ कि प्रत्येक $ M \in M_n(K)$ के समान है
        इसका स्थानांतरण ${}^tM $. \emph{(ऑपरेशन जो लाते हैं
        $ XI - M $ स्मिथ को रूप देना, स्थानांतरित करना, लाना $ XI - {}^tM $ को
        वही स्मिथ फॉर्म: बराबर \omterm{thm:b3:modules:frobenius}{समानता अपरिवर्तनीय}.)}
  \item (समानता का अवतरण) चलो $ K \subseteq L $ एक क्षेत्र हो
        विस्तार और $ M, N \in M_n(K)$. दिखाओ अगर $ M $ और $ N $
        ऊपर समान हैं $ L $, वे समान हैं $ K $.
        \emph{(स्मिथ का रूप $ XI - M $ में गणना की गई $ K[X]$ है
        अभी भी स्मिथ फॉर्म में हैं $ L[X]$ --- ऐसा क्यों करें \omterm{thm:b3:modules:smith}{अपरिवर्तनीय
        कारक} परिवर्तन नहीं?)} याद रखने लायक परिणाम: दो
        वास्तविक आव्यूह संयुग्मित होते हैं $ GL_n(\C)$ में संयुग्मित हैं
        $ GL_n(\R)$.
  \item (निलपोटेंट वर्गीकरण) चलो $ u $ शून्यशक्तिशाली बनो. दिखाओ
        कि आकार के ब्लॉकों की संख्या $\geq k $ इसके में
        निलपोटेंट जॉर्डन ब्लॉकों में अपघटन बराबर होता है
        $\operatorname{rk}u^{k-1} - \operatorname{rk}u^k $, और
        निष्कर्ष निकालें: निलपोटेंट वर्ग $ M_n(K)$, किसी भी क्षेत्र के लिए $ K $,
        के विभाजन पर आपत्ति जता रहे हैं $ n $. कितने
        निलपोटेंट क्लासेस में $ M_5(K)$?
  \item (एक ठोस जोड़ी) निर्धारित करें \omterm{thm:b3:modules:frobenius}{समानता अपरिवर्तनीय} का
        व्युत्पत्ति $ D\colon P \mapsto P'$ अंतरिक्ष पर अभिनय
        $ K_{n-1}[X]$ डिग्री के बहुपदों का $< n $: (ए) के लिए $ K =
        \Q $; (बी) के लिए $ K = \mathbb F_p $ साथ $ p < n $
        \emph{(विशेषता में $ p $, $(X^p)' = 0$: गणना करें
        $\ker D^k $ और प्रश्न 18 का उपयोग करें)}.
  \item (\omterm{ex:b3:groups:actions}{संयुग्मता वर्ग} का $ GL_2(\mathbb F_q)$) प्रयोग करना
        \omterm{thm:b3:modules:smith}{अपरिवर्तनीय कारक}, दिखाएँ कि हर वर्ग
        $ GL_2(\mathbb F_q)$ बिल्कुल चार प्रकारों में से एक है:
        \omterm{ex:b3:groups:actions}{केंद्र}ीय $ aI $; दो भिन्न के साथ विकर्णीय
        eigenvalues $ a \neq b $ में $\mathbb F_q^\times $;
        न्यूनतम बहुपद के साथ गैर-अर्धसरल $(X - a)^2$;
        चक्रीय साथ अलघुकरणीय विशेषता बहुपद.
  \item प्रत्येक प्रकार की कक्षाएं गिनें और निष्कर्ष निकालें:
        $ GL_2(\mathbb F_q)$ बिलकुल है $ q^2 - 1$ संयुग्मन
        कक्षाएं. \emph{(मोनिक गिनें अलघुकरणीय द्विघात खत्म
        $\mathbb F_q $; अव्यवस्थित जोड़े $\{a, b\}$; याद रखें
        व्युत्क्रमणीयता स्थिरांकों को बाधित करती है।)}
  \item (चक्रीय \omterm{def:b3:groups:normal}{सामान्य} है) वह दिखाओ $ M \in M_2(\mathbb F_q)$
        यदि चक्रीय होने में विफल रहता है $ M $ अदिश है, और यह निष्कर्ष निकालें कि a
        समान रूप से यादृच्छिक $2\times2$ मैट्रिक्स खत्म $\mathbb F_q $ है
        संभावना के साथ चक्रीय $1 - q^{-3}$. समरूप बताइये
        के लिए अनुमानी $ M_n $ और बड़ा $ q $ (कोई प्रमाण आवश्यक नहीं):
        गैर-चक्रीय मैट्रिक्स दुर्लभ हैं --- यही कारण है
        \cref{pb:b3:modules:1}के भाग III को केवल वास्तविक कार्य की आवश्यकता थी
        \omterm{def:b3:groups:normal}{सामान्य} मामले से परे.
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Complements.}
\begin{enumerate}[resume]
  \item (\omterm{ex:b3:groups:actions}{केंद्र} कम्यूटेंट का) दिखाएँ कि \omterm{ex:b3:groups:actions}{केंद्र} का
        बीजगणित $\mathcal C(u)$ बिलकुल है $ K[u]$ \emph{(गठबंधन
        भाग III के दो समावेशन)}. उसका निष्कर्ष निकालो
        $\mathcal C(u)$ क्रमविनिमेय है iff $ u $ चक्रीय है ---
        प्रश्न 8 के समानता मामले को विशुद्ध रूप से पुनर्प्राप्त करना
        संरचनात्मक मार्ग.
  \item (कौन से आयाम घटित होते हैं?) फ्रोबेनियस से निष्कर्ष निकालें
        वह सूत्र $\dim\mathcal C(u) \equiv n \pmod 2$ हरएक के लिए $ u $. फिर लिए गए मानों का सटीक सेट निर्धारित करें
        द्वारा $\dim \mathcal C(u)$ जैसा $ u $ तक फैली हुई है
        $\mathcal L(V)$ साथ $\dim V = 4$: दिखाओ यह है $\{4, 6,
        8, 10, 16\}$ \emph{(डिग्री अनुक्रमों की गणना करें $ n_1
        \leq \dots \leq n_s $ संक्षेप में $4$ और प्रत्येक को एहसास करें
        एक निलपोटेंट)}. विशेष रूप से $12$ और $14$, यद्यपि
        सही समता, प्राप्त नहीं होती: समता
        प्रतिबंध आवश्यक है लेकिन पर्याप्त नहीं है।
  \item (\omterm{cor:b3:groups:classeq}{वर्ग समीकरण} का $ GL_2(\mathbb F_3)$) के लिए $ q = 3$,
        प्रत्येक के आकार की गणना करें संयुग्मन वर्ग प्रश्न 20 का
        \omterm{def:b3:groups:action}{कक्षा} के माध्यम से--स्टेबलाइज़र: ए का सेंट्रलाइज़र
        \emph{चक्रीय} $ M $ में $ GL_2(\mathbb F_q)$ इकाई है
        का समूह $ K[M]$ (प्रश्न 11). पहचान करना $ K[M]$ में
        तीन गै\omterm{ex:b3:groups:actions}{र-केंद्र}ीय प्रकार, तीन मोनिक की सूची बनाएं अलघुकरणीय
        द्विघात खत्म $\mathbb F_3$, और सत्यापित करें \omterm{def:b3:groups:action}{कक्षा}
        समीकरण
        \[
        48 = \abs{GL_2(\mathbb F_3)}
        = 2\cdot1 + 1\cdot12 + 2\cdot8 + 3\cdot6,
        \]
        साथ $2 + 1 + 2 + 3 = 8 = q^2 - 1$ कक्षाएं, जैसे
        प्रश्न 21 द्वारा भविष्यवाणी की गई।
\end{enumerate}
\end{problem}
