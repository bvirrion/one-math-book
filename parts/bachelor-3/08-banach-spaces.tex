\chapter{Banach Spaces and the Fundamental Theorems}
\label{ch:b3:banach}

Functional analysis studies infinite-dimensional normed spaces
through the operators and functionals living on them. Its
founding discovery is that \emph{\omterm{def:b3:complete:complete}{completeness}}, via Baire's
theorem, forces strong uniformity: pointwise bounded families of
operators are norm-bounded (Banach--Steinhaus), \omterm{def:b3:topology:continuity}{continuous}
bijections have \omterm{def:b3:topology:continuity}{continuous} inverses (open mapping), and graphs
detect \omterm{def:b3:topology:continuity}{continuity} (closed graph). The other pillar,
\emph{Hahn--Banach}, needs no \omterm{def:b3:complete:complete}{completeness} at all --- only
Zorn's lemma --- and guarantees that \omterm{def:b3:banach:operator}{dual spaces} are rich enough
to see every vector. This chapter proves all four theorems and
tests them on the classical sequence spaces $\ell^p$, on
concrete \omterm{def:b3:banach:operator}{dual} computations, and on a genuinely surprising
application: there exist \omterm{def:b3:topology:continuity}{continuous} $2\pi$-periodic functions
whose Fourier series \emph{diverges} at a point --- resolving in
the negative a question Year 2 left open.

Throughout, $E, F$ are normed spaces over $K = \R$ or $\C$;
\emph{Banach space}\index{Banach space} means \omterm{def:b3:complete:complete}{complete} normed
space.

\section{Bounded operators; sequence spaces}

\begin{definition}\label{def:b3:banach:operator}
$\mathcal L(E, F)$ denotes the space of \emph{bounded}
(= \omterm{def:b3:topology:continuity}{continuous}, Year 2) linear maps with the \emph{operator
norm}\index{operator norm} $\vertiii T = \sup_{\norm x \leq
1}\norm{Tx}$; it is submultiplicative: $\vertiii{ST} \leq
\vertiii S\,\vertiii T$. The \emph{dual}\index{dual space} is
$E' = \mathcal L(E, K)$.
\end{definition}

\begin{proposition}\label{prop:b3:banach:llcomplete}
If $F$ is a Banach space, so is $\mathcal L(E, F)$; in
particular $E'$ is always a Banach space.
\end{proposition}

\begin{proof}
Let $(T_n)$ be Cauchy for $\vertiii\cdot$. For each $x$,
$\norm{T_nx - T_mx} \leq \vertiii{T_n - T_m}\norm x$: $(T_nx)$
is Cauchy in $F$, convergent; call the limit $Tx$. $T$ is
linear (limits of linear identities); passing to the limit in
$\norm{T_nx - T_mx} \leq \varepsilon\norm x$ ($n, m \geq N$)
gives $\norm{T_nx - Tx} \leq \varepsilon\norm x$: $T_n \to T$
in \omterm{def:b3:banach:operator}{operator norm}, and $\vertiii T \leq \vertiii{T_N} +
\varepsilon < \infty$.
\end{proof}

\begin{definition}\label{def:b3:banach:lp}
The classical sequence spaces\index{sequence spaces $\ell^p$}
(over $K$, indexed by $\N$):
\[
\ell^p = \Bigl\{x = (x_n) : \norm x_p = \Bigl(\sum_n
\abs{x_n}^p\Bigr)^{1/p} < \infty\Bigr\}\quad (1 \leq p <
\infty),
\]
\[
\ell^\infty = \{x : \norm x_\infty = \sup\abs{x_n} < \infty\},
\]
and $c_0 = \{x : x_n \to 0\}$ with $\norm\cdot_\infty$. That
$\norm\cdot_p$ is a norm follows from the Minkowski inequality,
proved in the discrete case exactly as in \cref{ch:b3:lp} (or
by summing the finite-dimensional inequality of Year 2). All
are Banach spaces, and $c_0$ is a closed subspace of
$\ell^\infty$ (\cref{exo:b3:banach:3}).
\end{definition}

\begin{proposition}[Neumann series]\label{prop:b3:banach:neumann}
Let $E$ be Banach and $T \in \mathcal L(E) = \mathcal L(E,E)$
with $\vertiii T < 1$. Then $I - T$ is invertible in $\mathcal
L(E)$, with $(I - T)^{-1} = \sum_{n\geq0}T^n$ (convergent in
\omterm{def:b3:banach:operator}{operator norm}). Consequently the set of invertible operators is
open, and inversion is \omterm{def:b3:topology:continuity}{continuous} on it.\index{Neumann series}
\end{proposition}

\begin{proof}
The series converges absolutely ($\vertiii{T^n} \leq \vertiii
T^n$, geometric) in the Banach $\mathcal L(E)$
(\cref{prop:b3:banach:llcomplete};
\cref{exo:b3:complete:1}(b)). Telescoping, $(I - T)\sum_{n \leq
N}T^n = I - T^{N+1} \to I$, and similarly on the other side.
For openness: if $S$ is invertible and $\vertiii{H} <
1/\vertiii{S^{-1}}$, then $S + H = S(I + S^{-1}H)$ with
$\vertiii{S^{-1}H} < 1$: invertible. \omterm{def:b3:topology:continuity}{Continuity} of inversion:
the series expression gives $\vertiii{(S+H)^{-1} - S^{-1}} =
O(\vertiii H)$ locally.
\end{proof}

\begin{example}[A Volterra equation, solved by Neumann]
\label{ex:b3:banach:volterraexample}
On $E = \mathcal C(\intcc01)$, consider the integral equation
\[
u(x) = 1 + \lambda\int_0^xu(t)\,\dd t,
\qquad\text{i.e.}\qquad u = \mathbf 1 + \lambda Tu,
\quad (Tu)(x) = \int_0^xu .
\]
Here $\vertiii T \leq 1$, so for $\abs\lambda < 1$ the \omterm{prop:b3:banach:neumann}{Neumann
series} applies directly: $u = (I - \lambda T)^{-1}\mathbf 1 =
\sum_n\lambda^nT^n\mathbf 1$. Computing, $T^n\mathbf 1 =
\frac{x^n}{n!}$, so
\[
u(x) = \sum_{n\geq0}\frac{(\lambda x)^n}{n!} =
\eu^{\lambda x} ,
\]
as differentiation confirms. Better: $\vertiii{T^n} \leq
\frac1{n!}$ (the iterated kernel shrinks factorially), so
$\sum\lambda^nT^n$ converges for \emph{every} $\lambda$ ---
the operator $I - \lambda T$ is invertible for all $\lambda
\in \C$, even though $\vertiii{\lambda T} \geq 1$ eventually:
what matters is the spectral decay of the powers, not the
first norm. Volterra operators have this factorial decay
built in (\cref{exo:b3:complete:4}(a) exploited exactly
this), which is why initial value problems never suffer the
resonance phenomena of boundary value problems
(\cref{ch:b3:spectral}).
\end{example}

\section{Hahn--Banach}

\begin{theorem}[Hahn--Banach, analytic form]
\label{thm:b3:banach:hahnbanach}
Let $E$ be a \emph{real} vector space, $p \colon E \to \R$
\emph{sublinear} ($p(x + y) \leq p(x) + p(y)$ and $p(tx) =
tp(x)$ for $t \geq 0$), $F \subseteq E$ a subspace and $f
\colon F \to \R$ linear with $f \leq p$ on $F$. Then $f$ extends
to a linear $\tilde f \colon E \to \R$ with $\tilde f \leq p$
on $E$.\index{Hahn--Banach theorem}
\end{theorem}

\begin{proof}
\emph{One-step extension.} Let $x_0 \notin F$; we extend $f$ to
$F \oplus \R x_0$ by choosing $\alpha = \tilde f(x_0)$
correctly: we need, for all $y \in F$, $t > 0$,
\[
f(y) + t\alpha \leq p(y + tx_0)
\quad\text{and}\quad
f(y) - t\alpha \leq p(y - tx_0),
\]
which after dividing by $t$ (sublinearity) reduce to
\[
\sup_{v \in F}\ \bigl[f(v) - p(v - x_0)\bigr]
\;\leq\; \alpha \;\leq\;
\inf_{u \in F}\ \bigl[p(u + x_0) - f(u)\bigr].
\]
Such $\alpha$ exists iff every left member is $\leq$ every
right member: indeed $f(v) + f(u) = f(u + v) \leq p(u + v) \leq
p(u + x_0) + p(v - x_0)$, i.e.\ $f(v) - p(v - x_0) \leq p(u +
x_0) - f(u)$.

\emph{Zorn.} Order the extensions of $f$ dominated by $p$
(pairs: subspace, functional) by extension; a chain has the
union as upper bound; a maximal element must be defined on all
of $E$, else the one-step extension contradicts maximality.
\end{proof}

\begin{corollary}\label{cor:b3:banach:hbnormed}
Let $E$ be a normed space ($K = \R$ or $\C$).
\begin{enumerate}
  \item Every $f \in F'$ ($F$ a subspace) extends to $\tilde f
        \in E'$ with $\norm{\tilde f}_{E'} = \norm f_{F'}$.
  \item For every $x \neq 0$ there is $f \in E'$ with $\norm f
        = 1$ and $f(x) = \norm x$. In particular $E'$ separates
        the points of $E$, and $\norm x = \sup_{\norm f \leq
        1}\abs{f(x)}$.
  \item For a closed subspace $F$ and $x \notin F$, there is $f
        \in E'$ vanishing on $F$ with $f(x) = d(x, F)$ and
        $\norm f \leq 1$.
\end{enumerate}
\end{corollary}

\begin{proof}
(1) Real case: apply \cref{thm:b3:banach:hahnbanach} with $p(x)
= \norm f_{F'}\,\norm x$ (sublinear); the extension satisfies
$\pm\tilde f(x) = \tilde f(\pm x) \leq p(x)$, so
$\norm{\tilde f} \leq \norm f$, and $\geq$ is restriction.
Complex case: let $u = \operatorname{Re}f$, a real functional
with $\abs u \leq \norm f\norm\cdot$; note $f(x) = u(x) -
\iu\,u(\iu x)$ (check on real and imaginary parts:
$\operatorname{Im}f(x) = -\operatorname{Re}f(\iu x)$). Extend
$u$ real-linearly with the same bound, and set $\tilde f(x) =
\tilde u(x) - \iu\tilde u(\iu x)$: $\C$-linear (direct check on
multiplication by $\iu$), extends $f$; norm: for given $x$
write $\tilde f(x) = r\eu^{\iu\theta}$, then $\abs{\tilde f(x)}
= \tilde f(\eu^{-\iu\theta}x) = \tilde u(\eu^{-\iu\theta}x)
\leq \norm f\,\norm x$.

(2) On $F = Kx$ define $f(tx) = t\norm x$: norm $1$ on $F$;
extend by (1). The duality formula: $\leq$ is clear, $\geq$ by
this $f$.

(3) On $F \oplus Kx$ define $f(y + tx) = t\,d(x, F)$; then for
$t \neq 0$, $\norm{y + tx} = \abs t\,\norm{x + y/t} \geq \abs
t\,d(x, F) = \abs{f(y + tx)}$: $\norm f \leq 1$ on the
subspace; extend by (1).
\end{proof}

\begin{remark}\label{rem:b3:banach:bidual}
By (2), the canonical map $J \colon E \to E''$, $J(x)(f) =
f(x)$, is an isometry (\cref{exo:b3:banach:10}): every normed
space sits inside its bidual. Spaces with $J$ surjective are
called \emph{reflexive}\index{reflexive space}; the weekend
problem shows $\ell^p$ ($1 < p < \infty$) is \omterm{rem:b3:banach:bidual}{reflexive} while
$\ell^1$ is not.
\end{remark}

\section{The Baire trilogy}

\begin{theorem}[Banach--Steinhaus, uniform boundedness]
\label{thm:b3:banach:banachsteinhaus}
Let $E$ be a \emph{Banach} space, $F$ normed, and $(T_i)_{i\in
I} \subseteq \mathcal L(E, F)$ a family with $\sup_i
\norm{T_ix} < \infty$ for every $x \in E$. Then $\sup_i
\vertiii{T_i} < \infty$.\index{Banach--Steinhaus theorem}
\end{theorem}

\begin{proof}
The sets $F_n = \{x : \sup_i\norm{T_ix} \leq n\}$ are closed
(intersections of preimages of closed balls) and cover $E$.
Baire (\cref{thm:b3:complete:baire}) gives $n_0$ and a ball
$B(x_0, r) \subseteq F_{n_0}$. For $\norm z < r$: $\norm{T_iz}
\leq \norm{T_i(x_0 + z)} + \norm{T_ix_0} \leq 2n_0$, so
$\vertiii{T_i} \leq 2n_0/r$ for every $i$.
\end{proof}

\begin{corollary}\label{cor:b3:banach:bslimits}
If $E$ is Banach and $T_n \in \mathcal L(E,F)$ converge
\emph{pointwise} ($T_nx \to Tx$ for each $x$), then $\sup_n
\vertiii{T_n} < \infty$, $T \in \mathcal L(E, F)$, and
$\vertiii T \leq \liminf \vertiii{T_n}$.
\end{corollary}

\begin{proof}
Convergent sequences are bounded: pointwise boundedness;
Banach--Steinhaus bounds the norms by some $M$; then $\norm{Tx}
= \lim\norm{T_nx} \leq M\norm x$ ($T$ is linear as a pointwise
limit), and the sharper bound by passing to $\liminf$ in
$\norm{T_nx} \leq \vertiii{T_n}\norm x$.
\end{proof}

\begin{theorem}[Divergent Fourier series]
\label{thm:b3:banach:fourierdiverge}
There exist \omterm{def:b3:topology:continuity}{continuous} $2\pi$-periodic functions $f$ whose
Fourier series diverges at $0$: $\sup_N\abs{S_N(f)(0)} =
\infty$. Indeed such $f$ form a dense subset of $\mathcal
C(S^1)$.\index{Fourier series, divergence}
\end{theorem}

\begin{proof}
Work in $E = (\mathcal C(S^1), \norm\cdot_\infty)$, a Banach
space, with the functionals $\Lambda_N(f) = S_N(f)(0) =
\frac1{2\pi}\int_{-\pi}^{\pi}f(t)\,D_N(t)\,\dd t$ (Dirichlet
kernel, Year 2). Each $\Lambda_N$ is \omterm{def:b3:topology:continuity}{continuous} with
\[
\norm{\Lambda_N} = \frac1{2\pi}\int_{-\pi}^\pi\abs{D_N(t)}\,\dd
t \;=\; L_N .
\]
($\leq$ is clear; $\geq$: take $f$ \omterm{def:b3:topology:continuity}{continuous}, $\norm f_\infty
\leq 1$, approximating $\operatorname{sign}D_N$ --- the sign
has finitely many jumps; smoothing each jump on an interval of
length $\varepsilon$ changes the integral by $O(N\varepsilon)$.)
The \emph{Lebesgue constants} $L_N$ tend to infinity:
\[
L_N = \frac1{2\pi}\int_{-\pi}^{\pi}
\frac{\abs{\sin\bigl((N{+}\tfrac12)t\bigr)}}{\abs{\sin(t/2)}}
\,\dd t
\geq \frac{2}{\pi}\int_0^{\pi}
\frac{\abs{\sin\bigl((N{+}\tfrac12)t\bigr)}}{t}\,\dd t
= \frac{2}{\pi}\int_0^{(N+\frac12)\pi}\frac{\abs{\sin
u}}{u}\,\dd u
\]
(using $\abs{\sin(t/2)} \leq t/2$ on $[0, \pi]$, then
substituting $u = (N + \tfrac12)t$). Cutting into arches:
\[
\int_{(k-1)\pi}^{k\pi}\frac{\abs{\sin u}}u\,\dd u
\geq \frac1{k\pi}\int_{(k-1)\pi}^{k\pi}\abs{\sin u}\,\dd u =
\frac{2}{k\pi},
\qquad\text{so}\qquad
L_N \geq \frac{4}{\pi^2}\sum_{k=1}^N\frac1k
\xrightarrow[N\to\infty]{} \infty .
\]
If every \omterm{def:b3:topology:continuity}{continuous} $f$ had $\sup_N
\abs{\Lambda_N(f)} < \infty$, Banach--Steinhaus would force
$\sup_N\norm{\Lambda_N} < \infty$: contradiction. So some $f$
--- in fact a nonmeagre, \omterm{def:b3:topology:interior}{dense set} of $f$ (the complement of
$\bigcup_M\{f: \sup_N\abs{\Lambda_Nf}\leq M\}$, a countable
union of closed sets which, having no \omterm{def:b3:topology:interior}{interior} by the above
applied in any ball, is \omterm{rem:b3:complete:meagre}{meagre}) --- has $\sup_N\abs{S_Nf(0)} =
\infty$.
\end{proof}

\begin{theorem}[Open mapping]\label{thm:b3:banach:openmapping}
Let $E, F$ be Banach spaces and $T \in \mathcal L(E, F)$
\emph{surjective}. Then $T$ is open: $T(B_E(0,1)) \supseteq
B_F(0, c)$ for some $c > 0$. Consequently a bijective bounded
operator between Banach spaces has a bounded
inverse.\index{open mapping theorem}
\end{theorem}

\begin{proof}
Write $B = B_E(0,1)$. Surjectivity gives $F = \bigcup_n
\overline{T(nB)} = \bigcup_n n\,\overline{T(B)}$; Baire
(\cref{thm:b3:complete:baire}) gives \omterm{def:b3:topology:interior}{interior} to
$\overline{T(B)}$: some $B_F(y_0, 4c) \subseteq
\overline{T(B)}$. \omterm{ex:b3:groups:actions}{Re-center} at $0$: for $\norm y < 4c$, both
$y_0 + y$ and $y_0$ are limits of images $Tu_k$, $Tv_k$ with
$u_k, v_k \in B$, so $y = \lim T(u_k - v_k)$ with $u_k - v_k
\in 2B$: $B_F(0, 4c) \subseteq \overline{T(2B)}$, i.e.\
$B_F(0, 2c) \subseteq \overline{T(B)}$.

\emph{Removing the \omterm{def:b3:topology:interior}{closure}} (here \omterm{def:b3:complete:complete}{completeness} of $E$ enters):
let $\norm y < c$. Pick $x_1 \in \frac12 B$ with $\norm{y -
Tx_1} < c/2$ ($B_F(0,2c) \subseteq \overline{T(B)}$ scaled by
$\frac12$); inductively $x_k \in 2^{-k}B$ with $\norm{y -
T(x_1 + \dots + x_k)} < c\,2^{-k}$. The series $\sum x_k$
converges absolutely in the Banach $E$, to $x \in B$ (norm $<
\sum 2^{-k} = 1$), and $Tx = y$ by \omterm{def:b3:topology:continuity}{continuity}: $B_F(0, c)
\subseteq T(B)$. Openness of $T$ on arbitrary opens follows by
translation and scaling; for the corollary, openness of $T$
means $T^{-1}$ is \omterm{def:b3:topology:continuity}{continuous}.
\end{proof}

\begin{corollary}[Equivalent norms]\label{cor:b3:banach:equivnorms}
If a vector space is \omterm{def:b3:complete:complete}{complete} for two comparable norms
($\norm\cdot_a \leq C\norm\cdot_b$), the norms are equivalent.
\end{corollary}

\begin{proof}
The identity $(E, \norm\cdot_b) \to (E, \norm\cdot_a)$ is
bounded and bijective between Banach spaces: its inverse is
bounded.
\end{proof}

\begin{theorem}[Closed graph]\label{thm:b3:banach:closedgraph}
Let $E, F$ be Banach and $T \colon E \to F$ linear. If the
graph $\Gamma = \{(x, Tx)\}$ is closed in $E \times F$ (i.e.\
$x_n \to x$ and $Tx_n \to y$ imply $y = Tx$), then $T$ is
bounded.\index{closed graph theorem}
\end{theorem}

\begin{proof}
$E \times F$ with $\norm{(x,y)} = \norm x + \norm y$ is Banach;
$\Gamma$, a closed subspace, is Banach. The projection
$\pi_E\colon \Gamma \to E$ is bounded and bijective, so its
inverse $x \mapsto (x, Tx)$ is bounded
(\cref{thm:b3:banach:openmapping}): $\norm{Tx} \leq \norm{(x,
Tx)} \leq C\norm x$.
\end{proof}

\begin{method}\label{met:b3:banach:trilogy}
When to reach for which theorem. \emph{Hahn--Banach}: to
produce a functional with prescribed behavior (norming a
vector, vanishing on a subspace, extending from a subspace) ---
no \omterm{def:b3:complete:complete}{completeness} needed. \emph{Banach--Steinhaus}: to convert
pointwise information into uniform bounds --- typically to show
a limit operation is \omterm{def:b3:topology:continuity}{continuous}, or (contrapositive) to prove
\emph{divergence} for some element, as for Fourier series.
\emph{Open mapping / closed graph}: to get \omterm{def:b3:topology:continuity}{continuity} for free
from \omterm{def:b3:galois:algebraic}{algebraic} bijectivity or from a \omterm{def:b3:topology:interior}{closure} property of the
graph --- typical use: comparing two \omterm{def:b3:complete:complete}{complete} norms, or proving
automatic \omterm{def:b3:topology:continuity}{continuity}. All three Baire theorems require
\omterm{def:b3:complete:complete}{completeness} \emph{of the source}; counterexamples otherwise
(\cref{exo:b3:banach:7}).
\end{method}

\section{Dual spaces, concretely}

\begin{theorem}\label{thm:b3:banach:duals}
Isometrically: $(c_0)' \cong \ell^1$ and $(\ell^1)' \cong
\ell^\infty$, via the pairing $\langle x, y\rangle = \sum_n
x_ny_n$.\index{duals of sequence spaces}
\end{theorem}

\begin{proof}
We prove $(c_0)' \cong \ell^1$; the second identification is
\cref{exo:b3:banach:5}. To $y \in \ell^1$ associate $\Lambda_y(x)
= \sum x_ny_n$ ($x \in c_0$): absolutely convergent, with
$\abs{\Lambda_y(x)} \leq \norm x_\infty\norm y_1$, so
$\norm{\Lambda_y} \leq \norm y_1$. Conversely let $\Lambda \in
(c_0)'$; set $y_n = \Lambda(e_n)$ ($e_n$ the unit sequences).
For any $N$, test $x^{(N)} = \sum_{n \leq N}
\operatorname{sign}(\overline{y_n})\,e_n \in c_0$ (norm $\leq
1$; in the complex case use unimodular factors $\bar y_n/\abs
{y_n}$): $\Lambda(x^{(N)}) = \sum_{n\leq N}\abs{y_n} \leq
\norm\Lambda$. So $y \in \ell^1$ with $\norm y_1 \leq
\norm\Lambda$. Finally $\Lambda = \Lambda_y$: both agree on the
$e_n$, hence on finite sequences, dense in $c_0$ (truncation:
$\norm{x - \sum_{n \leq N}x_ne_n}_\infty = \sup_{n > N}\abs{x_n}
\to 0$ precisely because $x_n \to 0$); \omterm{def:b3:topology:continuity}{continuous} functionals
agreeing on a \omterm{def:b3:topology:interior}{dense set} are equal. The correspondence is linear,
bijective, and isometric ($\norm{\Lambda_y} = \norm y_1$ from
the two inequalities).
\end{proof}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:b3:banach:1}
Compute the \omterm{def:b3:banach:operator}{operator norms}:
(a) the shifts $S(x_1, x_2, \dots) = (0, x_1, x_2, \dots)$ and
$S^*(x_1, x_2, \dots) = (x_2, x_3, \dots)$ on $\ell^2$;
(b) the multiplication operator $M_a x = (a_nx_n)$ on $\ell^2$,
for $a \in \ell^\infty$;
(c) the functional $\Lambda(f) = \int_0^{1/2}f -
\int_{1/2}^1f$ on $\mathcal C(\intcc01)$ --- show
$\norm\Lambda = 1$ and that the norm is \emph{not attained}.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b3:banach:2}
Let $E$ be Banach, $T \in \mathcal L(E)$ invertible, and $S$
with $\vertiii{S - T} < 1/\vertiii{T^{-1}}$. Show that $S$ is
invertible and estimate $\vertiii{S^{-1} - T^{-1}}$. Application:
if a linear system $Tx = b$ is \omterm{def:b3:groups:derived}{solvable} with $T$ invertible, a
sufficiently small perturbation of $T$ keeps it uniquely
\omterm{def:b3:groups:derived}{solvable}, with a quantitative bound on the change of solution.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:banach:3}
(a) Prove that $\ell^1$, $\ell^\infty$ and $c_0$ are Banach
spaces, and that $c_0$ is the \omterm{def:b3:topology:interior}{closure} in $\ell^\infty$ of the
space of finite sequences.
(b) Show $\ell^p \subseteq \ell^q$ with $\norm\cdot_q \leq
\norm\cdot_p$ for $1 \leq p \leq q \leq \infty$, and that the
inclusion is strict.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:banach:4}
Let $F \subseteq E$ be a closed subspace and $x \notin F$.
Using \cref{cor:b3:banach:hbnormed}, prove the duality formula
\[
d(x, F) = \max\bigl\{\abs{f(x)} : f \in E',\ \norm f \leq 1,\
f\restriction_F = 0\bigr\}
\]
(note: a \emph{maximum}). Deduce that $F = \bigcap\{\ker f : f
\in E',\ f\restriction_F = 0\}$: closed subspaces are exactly
the intersections of kernels of functionals.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:banach:5}
Prove $(\ell^1)' \cong \ell^\infty$ isometrically, following the
scheme of \cref{thm:b3:banach:duals} (finite sequences are dense
in $\ell^1$).
Where does the argument break for $(\ell^\infty)'$?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:banach:6}
Let $E, F, G$ be normed with $E$ Banach, and $B \colon E \times
F \to G$ bilinear, \omterm{def:b3:topology:continuity}{continuous} in each variable separately. Show
that $B$ is (jointly) \omterm{def:b3:topology:continuity}{continuous}: $\norm{B(x,y)} \leq
C\norm x\norm y$. \emph{(Apply Banach--Steinhaus to the family
$(B(\cdot, y))_{\norm y \leq 1}$.)}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:banach:7}
(a) On $E = \mathcal C(\intcc01)$, compare $\norm\cdot_\infty$
and $\norm\cdot_1$: the identity $(E, \norm\cdot_\infty) \to
(E, \norm\cdot_1)$ is bounded and bijective but its inverse is
unbounded. Which hypothesis of
\cref{cor:b3:banach:equivnorms} fails?
(b) Exhibit a discontinuous linear map from a dense subspace of
$\ell^2$ to $K$ (e.g.\ on finite sequences), and explain why
this does not contradict the closed graph theorem.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b3:banach:8}
(Hellinger--Toeplitz) Let $T \colon \ell^2 \to \ell^2$ be linear
(everywhere defined) and \emph{symmetric}: $\langle Tx,
y\rangle = \langle x, Ty\rangle$ for all $x, y$, where $\langle
x, y \rangle = \sum x_n\bar y_n$. Show that $T$ is bounded.
\emph{(Closed graph: if $x_k \to x$ and $Tx_k \to z$, test
against arbitrary $y$.)} Moral: unbounded symmetric operators
--- the Hamiltonians of quantum mechanics --- can never be
defined on the whole space.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b3:banach:9}
(Polya's theorem on quadrature) For each $n$, let $\Lambda_n(f)
= \sum_{i=0}^{n} w_{i,n}f(x_{i,n})$ be a quadrature rule on
$\mathcal C(\intcc01)$ ($x_{i,n} \in \intcc01$, $w_{i,n} \in
\R$). Show that $\Lambda_n(f) \to \int_0^1f$ for \emph{every}
\omterm{def:b3:topology:continuity}{continuous} $f$ if and only if: (i) $\Lambda_n(P) \to \int_0^1P$
for every polynomial $P$, and (ii) $\sup_n\sum_i\abs{w_{i,n}} <
\infty$. \emph{(Compute $\norm{\Lambda_n}$; use
Banach--Steinhaus and Weierstrass.)} Check that rules with
\emph{positive} weights exact on constants satisfy (ii)
automatically.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:banach:10}
Show that $J \colon E \to E''$, $J(x)(f) = f(x)$, is a linear
isometry (use \cref{cor:b3:banach:hbnormed}(2)), and that it is
surjective when $\dim E < \infty$. Show also that if $E'$ is
separable then so is $E$. \emph{(Pick $x_n$ nearly norming a
dense sequence of $E'$ and show their closed span is $E$, via
\cref{cor:b3:banach:hbnormed}(3).)}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:banach:11}
(Quotient spaces) Let $E$ be a Banach space and $F \subseteq
E$ a \emph{closed} subspace. On $E/F$ define
\[
\norm{\bar x} \;=\; d(x, F) = \inf_{y\in F}\norm{x - y} .
\]
(a) Show this is a well-defined \emph{norm} on $E/F$ (where
does closedness of $F$ enter?), and that the projection $\pi
\colon E \to E/F$ has $\vertiii\pi \leq 1$ and maps the open
unit ball onto the open unit ball.
(b) Show that $E/F$ is \omterm{def:b3:complete:complete}{complete}. \emph{(Use the series
criterion of \cref{exo:b3:complete:1}(b): given classes
$\bar x_k$ with $\sum\norm{\bar x_k} < \infty$, lift each to
$x_k \in E$ with $\norm{x_k} \leq \norm{\bar x_k} + 2^{-k}$
and sum in $E$.)}
(c) Compute: for $E = c$ (convergent sequences) and $F =
c_0$, show $c/c_0 \cong K$ isometrically via $\bar x \mapsto
\lim_nx_n$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:banach:12}
(Bounded projections and complemented subspaces) Let $E$ be a
Banach space and $P \colon E \to E$ linear with $P^2 = P$
(an \omterm{def:b3:galois:algebraic}{algebraic} projection), $V = \operatorname{im}P$, $W =
\ker P$.
(a) Suppose $P$ is bounded. Show that $V$ and $W$ are closed
and $E = V \oplus W$ with the decomposition $x = Px + (x -
Px)$.
(b) Conversely, suppose $E = V \oplus W$ with \emph{both} $V,
W$ closed, and let $P$ be the projection onto $V$ along $W$.
Show that $P$ is bounded. \emph{(Closed graph: if $x_n \to x$
and $Px_n \to z$, then $z \in V$, $x_n - Px_n \to x - z \in
W$, and uniqueness of the decomposition identifies $z =
Px$.)}
(c) Deduce the \emph{equivalence}: a subspace $V$ admits a
bounded projection iff it is closed and has a closed
\omterm{def:b3:galois:algebraic}{algebraic} complement --- and note (without proof) that
closed subspaces without this property exist ($c_0$ inside
$\ell^\infty$ is the classical example): Hilbert spaces,
where $V^\perp$ always works
(\cref{ch:b3:hilbert}), are the exception, not the rule.
\end{exercise}

\section{Problem: the duality of the \texorpdfstring{$\ell^p$}%
{lp} spaces}

\begin{problem}[{Weekend problem --- $(\ell^p)' = \ell^q$,
reflexivity, and the strangeness of $\ell^\infty$}]
\label{pb:b3:banach:1}
Fix $1 < p < \infty$ and let $q$ be the conjugate exponent,
$\frac1p + \frac1q = 1$. The pairing throughout is $\langle x,
y\rangle = \sum_n x_ny_n$.

\medskip
\noindent\textbf{Part I --- H\"older and Minkowski for
sequences.}
\begin{enumerate}
  \item (Young's inequality) For $a, b \geq 0$ show $ab \leq
        \frac{a^p}p + \frac{b^q}q$, using concavity of $\log$ or
        by studying $t \mapsto \frac{t^p}p + \frac1q - t$.
  \item (H\"older) Deduce: $\abs{\langle x, y\rangle} \leq
        \norm x_p\norm y_q$ for $x \in \ell^p$, $y \in \ell^q$;
        identify the equality case.
  \item (Minkowski) Deduce the triangle inequality for
        $\norm\cdot_p$. \emph{(Write $\abs{x_n + y_n}^p \leq
        \abs{x_n}\,\abs{x_n{+}y_n}^{p-1} +
        \abs{y_n}\,\abs{x_n{+}y_n}^{p-1}$ and apply H\"older to
        each term.)}
  \item Prove that $\ell^p$ is complete and that the finite
        sequences are dense in it.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The duality $(\ell^p)' =
\ell^q$.}
\begin{enumerate}[resume]
  \item For $y \in \ell^q$, show that $\Lambda_y(x) = \langle
        x, y\rangle$ defines $\Lambda_y \in (\ell^p)'$ with
        $\norm{\Lambda_y} \leq \norm y_q$, and, testing on
        $x_n = \abs{y_n}^{q-1}\operatorname{sign}(y_n)$
        (suitably truncated and normalized), that
        $\norm{\Lambda_y} = \norm y_q$.
  \item Conversely, given $\Lambda \in (\ell^p)'$, set $y_n =
        \Lambda(e_n)$; show $y \in \ell^q$ with $\norm y_q \leq
        \norm\Lambda$ \emph{(test on truncations as in question
        5 and let the truncation length grow)}, and conclude
        $\Lambda = \Lambda_y$: the map $y \mapsto \Lambda_y$ is
        an isometric isomorphism $\ell^q \to (\ell^p)'$.
  \item Deduce that $\ell^p$ is \emph{reflexive} for $1 < p <
        \infty$: composing the two dualities, every element of
        $(\ell^p)''$ comes from $\ell^p$; verify carefully that
        the composite is the canonical $J$.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- $\ell^1$ and $\ell^\infty$ are
different animals.}
\begin{enumerate}[resume]
  \item Show that $\ell^p$ ($1 \leq p < \infty$) and $c_0$ are
        separable, but $\ell^\infty$ is not. \emph{(The
        uncountably many indicator sequences of subsets of $\N$
        are pairwise at distance $1$.)}
  \item Deduce from \cref{exo:b3:banach:10} that $(\ell^1)'
        \cong \ell^\infty$ but $(\ell^\infty)' \not\cong
        \ell^1$: $\ell^1$ is \emph{not} reflexive.
        \emph{(If $(\ell^\infty)'$ were $\ell^1$, it would be
        separable, forcing $\ell^\infty$ separable.)}
  \item (A Banach limit, explicitly) On $\ell^\infty_\R$, let
        $p(x) = \limsup_n \frac{x_1 + \dots + x_n}{n}$. Show
        that $p$ is sublinear, and that on the subspace $c$ of
        convergent sequences, $\mathrm{LIM}(x) = \lim x$
        satisfies $\mathrm{LIM} \leq p$. Extend by Hahn--Banach
        to $\mathrm{LIM} \colon \ell^\infty_\R \to \R$ and show:
        $\mathrm{LIM}$ is positive ($x \geq 0 \Rightarrow
        \mathrm{LIM}(x) \geq 0$), shift-invariant
        ($\mathrm{LIM}(x_2, x_3, \dots) = \mathrm{LIM}(x)$),
        extends the limit, and satisfies $\liminf x \leq
        \mathrm{LIM}(x)\leq \limsup x$.
  \item Show that such a $\mathrm{LIM}$, viewed in
        $(\ell^\infty)'$, is \emph{not} of the form $\Lambda_y$
        for any $y \in \ell^1$; conclude again $(\ell^\infty)'
        \neq \ell^1$. \emph{(Evaluate on the unit sequences
        $e_n$, then on the constant sequence $1$.)}
  \item Evaluate $\mathrm{LIM}$ on $(0,1,0,1,\dots)$, and show
        that no shift-invariant \emph{multiplicative} extension
        of the limit can exist \emph{(consider $x =
        (0,1,0,1,\dots)$ and $x\cdot Sx$ where $S$ is the
        shift)}.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Epilogue: why reflexivity
matters.}
\begin{enumerate}[resume]
  \item Using \cref{cor:b3:banach:bslimits} and question 6,
        show that every bounded sequence of $\ell^p$ ($1 < p <
        \infty$) has a subsequence $(x^{(k)})$ that converges
        \emph{weakly}: $\Lambda(x^{(k)})$ converges for every
        $\Lambda \in (\ell^p)'$. \emph{(Diagonal extraction on
        the countably many coordinates; identify the weak limit
        in $\ell^p$ using uniform boundedness of norms and
        H\"older.)} Show by example ($e_n$ in $\ell^1$, against
        well-chosen elements of $\ell^\infty$) that this fails
        in $\ell^1$: weak compactness is a privilege of
        reflexive spaces.
\end{enumerate}

\medskip
\noindent\textbf{Part V --- The weak topology at work, and
Schur's surprise.} Write $x^{(k)} \rightharpoonup x$ in a
normed space $E$ (\emph{weak convergence}) when
$\Lambda(x^{(k)}) \to \Lambda(x)$ for every $\Lambda \in E'$.
\begin{enumerate}[resume]
  \item Complete the census: show $(c_0)' \cong \ell^1$
        isometrically, by the scheme of questions 5--6 (what
        replaces the test sequences?). Assemble the chain
        $c_0 \to \ell^1 \to \ell^\infty \to \dots$ of
        successive duals and mark where reflexivity fails.
  \item Show that every weakly convergent sequence of a Banach
        space is bounded: view the $x^{(k)}$ through the
        canonical embedding $J$ as functionals on $E'$ and
        apply Banach--Steinhaus
        (\cref{thm:b3:banach:banachsteinhaus}) --- on which
        Banach space, and why is completeness available there?
  \item Show that in $\ell^p$, $1 < p < \infty$: $x^{(k)}
        \rightharpoonup x$ iff $\sup_k\norm{x^{(k)}}_p <
        \infty$ and $x^{(k)}_n \to x_n$ for every coordinate
        $n$ \emph{(one direction uses Banach--Steinhaus through
        the canonical embedding; for the other, approximate $y
        \in \ell^q$ by finite sequences)}. Deduce $e_k
        \rightharpoonup 0$ in $\ell^2$ while $\norm{e_k}_2 =
        1$: weak limits can lose mass.
  \item Show that the norm is weakly lower semicontinuous:
        $x^{(k)} \rightharpoonup x$ implies $\norm x \leq
        \liminf_k\,\norm{x^{(k)}}$ \emph{(pick a norming
        functional for $x$, \cref{cor:b3:banach:hbnormed})}.
  \item (Radon--Riesz in $\ell^2$) Show that in $\ell^2$, weak
        convergence together with convergence of norms implies
        norm convergence \emph{(expand $\norm{x^{(k)} -
        x}_2^2$)}. Give a counterexample to the same statement
        without the norm hypothesis.
  \item (Schur, step 1) Let $x^{(k)} \rightharpoonup 0$ in
        $\ell^1$ and suppose, for contradiction,
        $\norm{x^{(k)}}_1 \geq \delta > 0$ along a
        subsequence. Show first that $x^{(k)}_n \to 0$ for
        each $n$ (which functionals?), then construct
        recursively indices $k_1 < k_2 < \cdots$ and integers
        $0 = N_0 < N_1 < N_2 < \cdots$ such that the mass of
        $x^{(k_j)}$ concentrates on the block $B_j =
        \intoc{N_{j-1}}{N_j}$:
        \[
        \sum_{n \in B_j}\bigl|x^{(k_j)}_n\bigr| \geq
        \norm{x^{(k_j)}}_1 - \frac\delta{10} .
        \]
  \item (Schur, step 2) Define $y \in \ell^\infty$ by $y_n =
        \operatorname{sign}\bigl(x^{(k_j)}_n\bigr)$ for $n \in
        B_j$. Show $\langle x^{(k_j)}, y\rangle \geq
        \norm{x^{(k_j)}}_1 - \frac{2\delta}{10} \geq
        \frac{8\delta}{10}$ and derive a contradiction with
        $x^{(k)} \rightharpoonup 0$. Conclude \emph{Schur's
        theorem}: in $\ell^1$, weakly convergent sequences
        converge in norm.
  \item Deduce that $(e_k)$ has no weakly convergent
        subsequence in $\ell^1$ (its only candidate limit is
        $0$, coordinatewise --- then apply Schur), recovering
        question 13's failure of weak compactness; and resolve
        the apparent paradox: in $\ell^1$ weak and norm
        convergence of \emph{sequences} coincide, yet the
        weak and norm \emph{topologies} differ and bounded
        sets still fail to be weakly sequentially compact ---
        no contradiction, only the failure of reflexivity.
  \item (Synthesis table) For $E \in \{c_0,\ \ell^1,\ \ell^p\
        (1{<}p{<}\infty),\ \ell^\infty\}$, tabulate: the dual;
        separability; reflexivity; whether bounded sequences
        admit weakly convergent subsequences; and one
        signature property of each space, justified in one
        line from this problem.
\end{enumerate}

\medskip
\noindent\textbf{Part VI --- Complements: nearest points,
averaged convergence, the value of a Banach limit.}
\begin{enumerate}[resume]
  \item (Nearest points: a dividend of reflexivity) Let $F$ be
        a closed subspace of $\ell^p$ ($1 < p < \infty$) and
        $x \in \ell^p$. Show that $d = \operatorname{dist}(x,
        F)$ is \emph{attained}: extract from a minimizing
        sequence a weakly convergent subsequence (question
        13), keep the weak limit inside $F$ by building, via
        Hahn--Banach, a functional vanishing on $F$ but not at
        a point outside it, and conclude with question 17.
        Then show the privilege is not universal: in $c_0$,
        for $\Lambda(x) = \sum_n2^{-n}x_n$, prove
        $\norm\Lambda = 1$ is not attained on the unit ball,
        establish the distance formula
        $\operatorname{dist}(x, \ker\Lambda) =
        \abs{\Lambda(x)}$, and deduce that no $x \notin
        \ker\Lambda$ has a nearest point in the closed
        hyperplane $\ker\Lambda$.
  \item (Banach--Saks in $\ell^2$) Let $x^{(k)}
        \rightharpoonup 0$ in $\ell^2_{\R}$ with
        $\norm{x^{(k)}}_2 \leq C$. Construct a subsequence
        $(y_j)$ with $\abs{\langle y_i, y_j\rangle} \leq
        \frac1j$ for all $i < j$, and deduce
        \[
        \Bigl\lVert\frac{y_1 + \dots + y_m}m\Bigr\rVert_2^2
        \leq \frac{C^2 + 2}m \longrightarrow 0 :
        \]
        after extraction, the Ces\`aro means converge in
        \emph{norm}. Check on $(e_k)$, whose means have norm
        $\frac1{\sqrt m}$: weak convergence, useless for the
        sequence itself (question 16), becomes norm
        convergence for averages.
  \item (The value of a Banach limit) Let $L$ be any Banach
        limit (question 10) and $A_mx = \frac1m(x + Sx + \dots
        + S^{m-1}x)$. Show $L(A_mx) = L(x)$ and $\liminf A_mx
        \leq L(x) \leq \limsup A_mx$; deduce that \emph{all}
        Banach limits agree on periodic sequences, with value
        the mean over a period --- $\frac13$ on $(1, 0, 0, 1,
        0, 0, \dots)$, consistent with question 12's
        $\frac12$. Then show agreement fails in general: for
        the block sequence $x$ equal to $1$ on
        $\intoc{3^{j-1}}{3^j}$ for even $j$ and $0$ elsewhere,
        show that the Ces\`aro means oscillate between $\leq
        \frac13$ and $\geq \frac23$, and build two Banach
        limits $L_\pm$ with $L_-(x) \leq \frac13 < \frac23
        \leq L_+(x)$ \emph{(extend from $c \oplus \R x$ with
        the extreme admissible values $\pm$: check that
        $\Lambda(y + tx) = \lim y + t\,p(x)$ is dominated by
        the sublinear $p$ of question 10)}.
\end{enumerate}
\end{problem}
