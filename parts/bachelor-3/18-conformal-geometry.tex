\chapter{Conformal Maps and the Riemann Mapping Theorem}
\label{ch:b3:conformal}

A \omterm{def:b3:holomorphic:holo}{holomorphic} bijection between two domains transports all of
complex analysis from one to the other: such maps ---
\emph{conformal}, because they preserve angles --- are the
isomorphisms of the \omterm{def:b3:holomorphic:holo}{holomorphic} world. This chapter classifies
them where classification is possible (the disc, the plane:
Schwarz's lemma is the key, an inequality of astonishing
power), builds the compactness theory of \omterm{def:b3:holomorphic:holo}{holomorphic} families
(Montel), and proves the deepest existence theorem of the
subject: \emph{every} \omterm{def:b3:conformal:simplyconnected}{simply connected} proper subdomain of
$\C$, however jagged its boundary, is \omterm{def:b3:conformal:conformal}{conformally equivalent}
to the unit disc. We close with \omterm{prop:b3:conformal:harmonicholo}{harmonic functions} and the
\omterm{thm:b3:conformal:poisson}{Poisson kernel}, solving the Dirichlet problem on the disc ---
the analytic payoff of conformal geometry. Throughout,
$\mathbb D = D(0,1)$ and $\mathbb H = \{\operatorname{Im}z >
0\}$.

\section{Conformal maps; M\"obius transformations}

\begin{definition}\label{def:b3:conformal:conformal}
A \emph{conformal map}\index{conformal map} (or
biholomorphism) between \omterm{def:b3:topology:topology}{open sets} is a \omterm{def:b3:holomorphic:holo}{holomorphic} bijection;
its inverse is automatically \omterm{def:b3:holomorphic:holo}{holomorphic}
(\cref{cor:b3:residues:openmapping}). Two domains are
\emph{conformally equivalent} if such a map exists;
$\operatorname{Aut}(\Omega)$ denotes the group of conformal
self-maps. Where $f' \neq 0$ --- everywhere, for injective $f$
(\cref{cor:b3:residues:openmapping}'s proof) --- the
differential is multiplication by $f'(z) \neq 0$: a
similarity, so conformal maps preserve angles between curves,
including orientation.
\end{definition}

\begin{example}[M\"obius transformations]
\label{ex:b3:conformal:mobius}
For $\bigl(\begin{smallmatrix}a & b\\ c &
d\end{smallmatrix}\bigr) \in GL_2(\C)$, the \emph{M\"obius
transformation}\index{M\"obius transformation} $z \mapsto
\frac{az + b}{cz + d}$ is conformal from
$\C\setminus\{-d/c\}$ onto $\C\setminus\{a/c\}$ (inverse of
the same type, from the inverse matrix; composition
corresponds to matrix product). The \emph{Cayley
map}\index{Cayley transform}
\[
\varphi(z) = \frac{z - \iu}{z + \iu}
\]
maps $\mathbb H$ \omterm{def:b3:conformal:conformal}{conformally} onto $\mathbb D$: indeed $\abs{z
- \iu} < \abs{z + \iu}$ exactly when $z$ is closer to $\iu$
than to $-\iu$, i.e.\ $\operatorname{Im}z > 0$; the inverse is
$w \mapsto \iu\frac{1 + w}{1 - w}$. M\"obius maps send the
family circles-and-lines to itself (\cref{exo:b3:conformal:1}).
\end{example}

\begin{example}[The Joukowski map]\label{ex:b3:conformal:joukowski}
Beyond M\"obius, the most useful \omterm{def:b3:conformal:conformal}{conformal map} of classical
applied mathematics is
\[
J(z) = \frac12\Bigl(z + \frac1z\Bigr) .
\]
On the exterior $\Omega = \{\abs z > 1\}$ of the unit disc,
$J$ is injective: $J(z) = J(w)$ gives $(z - w)(1 -
\frac1{zw}) = 0$ and $\abs{zw} > 1$. Its derivative $J'(z) =
\frac12(1 - z^{-2})$ vanishes only at $z = \pm1$, on the
boundary: $J$ is a conformal equivalence from $\Omega$ onto
its image, which is $\C\setminus\intcc{-1}1$ --- the unit
circle itself is folded two-to-one onto the segment
($J(\eu^{\iu\theta}) = \cos\theta$). Thus the exterior of a
\emph{segment}, a slit plane with no smooth boundary, is
\omterm{def:b3:conformal:conformal}{conformally} the exterior of a disc: corners are no obstacle
to conformal equivalence, only to boundary smoothness. Images
of circles through $\pm1$ but \omterm{ex:b3:groups:actions}{off-center} are airfoil-shaped
curves, and composing $J$ with M\"obius maps transported the
flow past a cylinder --- computable by hand --- to the flow
past a wing: for the first half of the twentieth century,
this example \emph{was} aerodynamics. It is also the door to
Chebyshev: $J$ conjugates $z \mapsto z^n$ to the Chebyshev
polynomial $T_n$ (\cref{pb:b3:hilbert:1}, Part V), since
$J(z^n) = \cos(n\theta)$ when $z = \eu^{\iu\theta}$.
\end{example}

\section{Schwarz's lemma and automorphism groups}

\begin{theorem}[Schwarz lemma]\label{thm:b3:conformal:schwarz}
Let $f \colon \mathbb D \to \mathbb D$ be \omterm{def:b3:holomorphic:holo}{holomorphic} with
$f(0) = 0$. Then\index{Schwarz lemma}
\[
\abs{f(z)} \leq \abs z \ \ (z \in \mathbb D)
\qquad\text{and}\qquad \abs{f'(0)} \leq 1 ,
\]
and if $\abs{f(z_0)} = \abs{z_0}$ for one $z_0 \neq 0$, or
$\abs{f'(0)} = 1$, then $f(z) = \eu^{\iu\theta}z$ is a
rotation.
\end{theorem}

\begin{proof}
$g(z) = f(z)/z$ extends \omterm{def:b3:holomorphic:holo}{holomorphically} to $\mathbb D$ (the
singularity at $0$ is removable: $g$ is bounded near $0$,
\cref{thm:b3:residues:riemanncw}; its value at $0$ is
$f'(0)$). On $\abs z = r < 1$: $\abs g \leq \frac1r$, so by
the maximum principle (\cref{thm:b3:holomorphic:maximum})
$\abs g \leq \frac1r$ on $\bar D(0,r)$; let $r \to 1$: $\abs
g \leq 1$ on $\mathbb D$, which is both inequalities. Equality
at an \omterm{def:b3:topology:interior}{interior} point makes $\abs g$ attain an \omterm{def:b3:topology:interior}{interior}
maximum: $g$ constant of modulus $1$.
\end{proof}

\begin{theorem}[Automorphisms of the disc]
\label{thm:b3:conformal:autdisc}
For $a \in \mathbb D$, the \emph{Blaschke
factor}\index{Blaschke factor}
\[
\varphi_a(z) = \frac{z - a}{1 - \bar a z}
\]
is an automorphism of $\mathbb D$ exchanging $a$ and $0$, with
$\varphi_a^{-1} = \varphi_{-a}$. Every automorphism of
$\mathbb D$ is $\eu^{\iu\theta}\varphi_a$ for unique $\theta
\in \R/2\pi\Z$, $a \in \mathbb D$.
\end{theorem}

\begin{proof}
On $\abs z = 1$: $\abs{1 - \bar az} = \abs{\bar z}\abs{1 -
\bar az} = \abs{\bar z - \bar a\abs z^2} = \abs{\bar z - \bar
a} = \abs{z - a}$, so $\abs{\varphi_a} = 1$ there; by the
maximum principle $\varphi_a(\mathbb D) \subseteq \bar{\mathbb
D}$, and openness puts the image in $\mathbb D$. The \omterm{def:b3:galois:algebraic}{algebraic}
identity $\varphi_{-a}\circ\varphi_a = \mathrm{id}$ (direct
computation) shows bijectivity. Now let $f \in
\operatorname{Aut}(\mathbb D)$ and $a = f^{-1}(0)$: $g =
f\circ\varphi_{-a}$ is an automorphism fixing $0$. Schwarz
applied to $g$ and to $g^{-1}$: $\abs{g(z)} \leq \abs z$ and
$\abs{g^{-1}(w)} \leq \abs w$, so $\abs{g(z)} = \abs z$:
rotation, $g = \eu^{\iu\theta}\,\mathrm{id}$, i.e.\ $f =
\eu^{\iu\theta}\varphi_a$. Uniqueness: $a = f^{-1}(0)$ and
$\theta$ from $f'$-type evaluation (or from $f(0) =
-\eu^{\iu\theta}a$ and one more value).
\end{proof}

\begin{theorem}[Automorphisms of the plane]
\label{thm:b3:conformal:autplane}
$\operatorname{Aut}(\C) = \{z \mapsto az + b : a \in \C^*,\ b
\in \C\}$. Consequently $\C$ and $\mathbb D$ are not
\omterm{def:b3:conformal:conformal}{conformally equivalent}.
\end{theorem}

\begin{proof}
Let $f \in \operatorname{Aut}(\C)$ and consider $g(z) =
f(1/z)$ on $\C^*$: a \omterm{def:b3:holomorphic:holo}{holomorphic function} with an \omterm{def:b3:residues:singularities}{isolated
singularity} at $0$. If it were essential,
Casorati--Weierstrass (\cref{thm:b3:residues:riemanncw}) would
make $g\bigl(D(0,\varepsilon)\setminus\{0\}\bigr)$ dense,
while $f(D(0, 1))$ is open and disjoint from it ($f$
injective: the two sets are images of disjoint sets) ---
impossible for a \omterm{def:b3:topology:interior}{dense set} and a nonempty \omterm{def:b3:topology:topology}{open set}. So $0$ is
a pole or removable for $g$, i.e.\ $\abs{f(z)}$ has at most
polynomial growth: $f$ is a polynomial
(\cref{exo:b3:holomorphic:4}(a)). Injectivity forces degree
$1$: a higher-degree polynomial has either a multiple root of
$f - c$ somewhere ($f'$ vanishes) or several distinct
preimages (d'Alembert--Gauss, \cref{pb:b3:holomorphic:1});
either way injectivity fails. Finally, a conformal $\C \to
\mathbb D$ would be a bounded entire function: constant
(Liouville) --- no equivalence.
\end{proof}

\section{Montel's theorem}

\begin{theorem}[Montel]\label{thm:b3:conformal:montel}
Let $\mathcal F \subseteq \mathcal H(\Omega)$ be \emph{locally
bounded}: every point has a \omterm{def:b3:topology:topology}{neighborhood} on which $\sup_{f\in
\mathcal F}\sup\abs f < \infty$. Then every sequence of
$\mathcal F$ has a subsequence converging uniformly on all
\omterm{def:b3:topology:compact}{compact} subsets of $\Omega$ (to a \omterm{def:b3:holomorphic:holo}{holomorphic}
limit).\index{Montel's theorem}
\end{theorem}

\begin{proof}
Local \omterm{def:b3:complete:equicontinuous}{equicontinuity}: if $\abs f \leq M$ on $D(a, 2r)
\subseteq \Omega$ for all $f \in \mathcal F$, the Cauchy
formula gives, for $z, z' \in D(a, r)$,
\[
\abs{f(z) - f(z')} = \frac{\abs{z - z'}}{2\pi}
\Bigl|\int_{C_{2r}}\frac{f(w)\,\dd w}{(w-z)(w-z')}\Bigr|
\leq \frac{\abs{z - z'}\,2\pi\cdot2r\,M}{2\pi\,r^2}
= \frac{2M}{r}\,\abs{z - z'} :
\]
a uniform Lipschitz bound. Exhaust $\Omega$ by \omterm{def:b3:topology:compact}{compacts} $K_m$;
each $K_m$ is covered by finitely many such discs, so
$\mathcal F$ is uniformly bounded and \omterm{def:b3:complete:equicontinuous}{equicontinuous} on
$K_m$: Arzel\`a--Ascoli (\cref{thm:b3:complete:ascoli})
extracts a subsequence converging uniformly on $K_m$;
diagonalize over $m$. The limit is \omterm{def:b3:holomorphic:holo}{holomorphic} by
\cref{thm:b3:holomorphic:weierstrassconv}.
\end{proof}

\section{The Riemann mapping theorem}

\begin{definition}\label{def:b3:conformal:simplyconnected}
An open \omterm{def:b3:topology:connected}{connected} $\Omega \subseteq \C$ is \emph{simply
connected}\index{simply connected domain} (in the homological
sense, sufficient for all our purposes) if
$\operatorname{Ind}_\gamma(w) = 0$ for every cycle $\gamma$ in
$\Omega$ and every $w \notin \Omega$ --- ``no cycle of
$\Omega$ surrounds a hole''. By the global Cauchy theorem
(\cref{thm:b3:residues:globalcauchy}) and
\cref{prop:b3:holomorphic:primitive}, on such $\Omega$
\emph{every \omterm{def:b3:holomorphic:holo}{holomorphic function} has a primitive}; hence every
zero-free $f \in \mathcal H(\Omega)$ has a \omterm{def:b3:holomorphic:holo}{holomorphic}
logarithm ($\exp\circ$(primitive of $f'/f$), adjusted by a
constant, as $(f\eu^{-L})' = 0$) and \omterm{def:b3:holomorphic:holo}{holomorphic} $n$-th roots
$\eu^{L/n}$.\index{holomorphic logarithm}
\end{definition}

\begin{theorem}[Riemann mapping theorem]
\label{thm:b3:conformal:rmt}
Every \omterm{def:b3:conformal:simplyconnected}{simply connected} open $\Omega \subsetneq \C$, $\Omega
\neq \varnothing$, is \omterm{def:b3:conformal:conformal}{conformally equivalent} to $\mathbb D$;
given $z_0 \in \Omega$, there is a unique conformal $f \colon
\Omega \to \mathbb D$ with $f(z_0) = 0$ and $f'(z_0) >
0$.\index{Riemann mapping theorem}
\end{theorem}

\begin{proof}
\emph{Step 0: the family is nonempty.} Pick $b \notin \Omega$:
$z - b$ is zero-free on $\Omega$, so it has a \omterm{def:b3:holomorphic:holo}{holomorphic}
square root $h$ ($h^2 = z - b$). $h$ is injective ($h(z) =
h(z')$ squares to $z = z'$), and if $w \in h(\Omega)$ then $-w
\notin h(\Omega)$ ($h(z) = -h(z')$ also squares to $z = z'$,
giving $w = -w = 0$, impossible as $h$ is zero-free). Since
$h(\Omega)$ is open, it contains a disc $D(h(z_0), \rho)$;
then $D(-h(z_0), \rho) \cap h(\Omega) = \varnothing$, i.e.\
$\abs{h(z) + h(z_0)} \geq \rho$ for every $z \in \Omega$.
Hence
\[
g(z) = \frac{\rho}{2\,\bigl(h(z) + h(z_0)\bigr)}
\]
is \omterm{def:b3:holomorphic:holo}{holomorphic}, injective (a M\"obius map composed with the
injective $h$), with $\abs g \leq \frac12 < 1$. Composing with a \omterm{thm:b3:conformal:autdisc}{Blaschke factor}
(\cref{thm:b3:conformal:autdisc}) to move $g(z_0)$ to $0$, the
family
\[
\mathcal F = \{f \colon \Omega \to \mathbb D \text{
holomorphic, injective, } f(z_0) = 0\}
\]
is nonempty.

\emph{Step 1: an extremal element.} Let $s =
\sup_{\mathcal F}\abs{f'(z_0)} \in \intoc0{+\infty}$ ($> 0$:
members are injective, so $f'(z_0) \neq 0$). Take $f_n \in
\mathcal F$ with $\abs{f_n'(z_0)} \to s$: the family is
bounded by $1$, so Montel (\cref{thm:b3:conformal:montel})
extracts $f_n \to f$ uniformly on \omterm{def:b3:topology:compact}{compacts}; $f$ is
\omterm{def:b3:holomorphic:holo}{holomorphic}, $f(z_0) = 0$, $\abs{f'(z_0)} = s$
(\cref{thm:b3:holomorphic:weierstrassconv} for the
derivatives), in particular $f$ is nonconstant; $f$ is
injective by Hurwitz (\cref{exo:b3:residues:8}(b)), and
$f(\Omega) \subseteq \bar{\mathbb D}$, hence $\subseteq
\mathbb D$ (open mapping). So $f \in \mathcal F$ attains the
supremum: $s < \infty$.

\emph{Step 2: the extremal map is onto.} Suppose $a \in
\mathbb D\setminus f(\Omega)$. The Blaschke transport
$\varphi_a\circ f$ is zero-free on the \omterm{def:b3:conformal:simplyconnected}{simply connected}
$\Omega$: it has a \omterm{def:b3:holomorphic:holo}{holomorphic} square root $F$ (with
$F(\Omega) \subseteq \mathbb D$, as $\abs F^2 =
\abs{\varphi_a\circ f} < 1$), injective (squares
distinguish). Normalize: $G = \varphi_{F(z_0)}\circ F \in
\mathcal F$. Undoing: $f = \varphi_{-a}\circ s_2 \circ
\varphi_{-F(z_0)}\circ G$ where $s_2(w) = w^2$; the map $\Psi
= \varphi_{-a}\circ s_2\circ\varphi_{-F(z_0)} \colon \mathbb D
\to \mathbb D$ is \omterm{def:b3:holomorphic:holo}{holomorphic} with $\Psi(0) = f(z_0) = 0$ and
is \emph{not} a rotation (it is not injective: $s_2$ is not).
Schwarz's lemma (strict case): $\abs{\Psi'(0)} < 1$, and the
chain rule $f = \Psi\circ G$ gives $\abs{f'(z_0)} =
\abs{\Psi'(0)}\,\abs{G'(z_0)} < \abs{G'(z_0)}$ ---
contradicting maximality (note $G \in \mathcal F$). Hence $f$
is onto: a conformal equivalence.

\emph{Step 3: normalization and uniqueness.} Multiply $f$ by
$\eu^{-\iu\arg f'(z_0)}$ to make $f'(z_0) > 0$ (this stays in
$\mathcal F$). If $f_1, f_2$ both work, $\psi =
f_2\circ f_1^{-1} \in \operatorname{Aut}(\mathbb D)$ fixes
$0$ with $\psi'(0) = f_2'(z_0)/f_1'(z_0) > 0$; by
\cref{thm:b3:conformal:autdisc} $\psi$ is a rotation
$\eu^{\iu\theta}$ with $\eu^{\iu\theta} > 0$: $\psi =
\mathrm{id}$.
\end{proof}

\begin{remark}\label{rem:b3:conformal:rmtscope}
The theorem is a pure existence statement of astonishing
scope: a square, a half-plane, the complement of a slit, the
region between two tangent circles, a fractal-boundary domain
--- all \omterm{def:b3:conformal:conformal}{conformally} identical to $\mathbb D$. What it does not
give: any formula (explicit maps are the exception:
\cref{exo:b3:conformal:5}), boundary behavior (a deeper theory
--- Carath\'eodory's theorem --- handles it), or uniqueness
of extension to $\C$ or multiply \omterm{def:b3:topology:connected}{connected} domains: the
annulus $\{1 < \abs z < 2\}$ is \emph{not} \omterm{def:b3:conformal:conformal}{conformally} a
punctured disc, and annuli of different radius ratios are
inequivalent (a genuinely harder fact).
\end{remark}

\section{Harmonic functions and the Poisson kernel}

\begin{proposition}\label{prop:b3:conformal:harmonicholo}
Let $\Omega$ be \omterm{def:b3:conformal:simplyconnected}{simply connected} and $u \colon \Omega \to \R$
harmonic\index{harmonic function} ($\mathcal C^2$ with
$\Delta u = u_{xx} + u_{yy} = 0$). Then $u =
\operatorname{Re}F$ for a \omterm{def:b3:holomorphic:holo}{holomorphic} $F$, unique up to an
imaginary constant. Consequently $u$ is $\mathcal C^\infty$,
satisfies the mean value property, and obeys the maximum
principle (no strict \omterm{def:b3:topology:interior}{interior} extremum unless constant).
\end{proposition}

\begin{proof}
$g = u_x - \iu u_y$ satisfies the \omterm{prop:b3:holomorphic:cauchyriemann}{Cauchy--Riemann equations}
($P = u_x$, $Q = -u_y$: $P_x = u_{xx} = -u_{yy} = Q_y$ and
$P_y = u_{xy} = u_{yx} = -Q_x$) with \omterm{def:b3:topology:continuity}{continuous} partials: $g$
is \omterm{def:b3:holomorphic:holo}{holomorphic} (\cref{prop:b3:holomorphic:cauchyriemann}; the
$\R$-differentiability follows from $\mathcal C^1$). Let $F_0$
be a primitive (simple connectivity,
\cref{def:b3:conformal:simplyconnected}); then
$\operatorname{Re}F_0$ has gradient $(u_x, u_y)$
($F_0' = g$ unpacks to exactly that via Cauchy--Riemann for
$F_0$), so $u - \operatorname{Re}F_0$ is constant ($\Omega$
\omterm{def:b3:topology:connected}{connected}): adjust $F = F_0 + c$. The properties transfer
from \cref{thm:b3:holomorphic:maximum} and
\cref{exo:b3:holomorphic:10} (for the maximum principle
applied to $u$ itself, use $\eu^{F}$ as there).
\end{proof}

\begin{theorem}[Poisson formula; Dirichlet problem on the
disc]\label{thm:b3:conformal:poisson}
For $0 \leq r < 1$ define the \emph{Poisson
kernel}\index{Poisson kernel}
\[
P_r(\theta) = \sum_{n\in\Z}r^{\abs n}\eu^{\iu n\theta}
= \frac{1 - r^2}{1 - 2r\cos\theta + r^2} \;>\; 0 .
\]
Let $g \colon \partial\mathbb D \to \R$ be \omterm{def:b3:topology:continuity}{continuous}, and
set, for $z = r\eu^{\iu\varphi} \in \mathbb D$,
\[
u(z) = \frac1{2\pi}\int_0^{2\pi}
P_r(\varphi - t)\,g(\eu^{\iu t})\,\dd t .
\]
Then $u$ is harmonic on $\mathbb D$ and extends \omterm{def:b3:topology:continuity}{continuously}
to $\bar{\mathbb D}$ with boundary values $g$: the unique such
\omterm{prop:b3:conformal:harmonicholo}{harmonic function}.\index{Dirichlet problem}
\end{theorem}

\begin{proof}
\emph{Kernel identities}: summing two geometric series,
\[
\sum_{n\in\Z}r^{\abs n}\eu^{\iu n\theta}
= \operatorname{Re}\frac{1 + r\eu^{\iu\theta}}{1 -
r\eu^{\iu\theta}}
= \frac{1 - r^2}{\abs{1 - r\eu^{\iu\theta}}^2},
\]
which is the displayed quotient; positivity is clear, and
$\frac1{2\pi}\int_0^{2\pi}P_r = 1$ (only $n = 0$ survives).

\emph{Harmonicity}: with $z = r\eu^{\iu\varphi}$,
\[
u(z) = \operatorname{Re}\biggl[\frac1{2\pi}\int_0^{2\pi}
\frac{\eu^{\iu t} + z}{\eu^{\iu t} - z}\,
g(\eu^{\iu t})\,\dd t\biggr],
\]
(the bracketed kernel has real part $P_r(\varphi - t)$:
compute), and the bracket is \omterm{def:b3:holomorphic:holo}{holomorphic} in $z$ on $\mathbb D$
(\cref{exo:b3:holomorphic:7}): $u$ is the real part of a
\omterm{def:b3:holomorphic:holo}{holomorphic function}, hence harmonic.

\emph{Boundary values}: $P_r(\cdot)$ is an approximate
identity as $r \to 1^-$: mass $1$, and for $\delta \leq
\abs\theta \leq \pi$, $P_r(\theta) \leq \frac{1 - r^2}{1 -
2r\cos\delta + r^2} \to 0$ uniformly. The standard split
(\omterm{def:b3:topology:continuity}{continuity} of $g$ near $\eu^{\iu\varphi_0}$, boundedness
elsewhere) gives $u(r\eu^{\iu\varphi}) \to
g(\eu^{\iu\varphi_0})$ as $r\eu^{\iu\varphi} \to
\eu^{\iu\varphi_0}$, uniformly in the boundary point: the
extension is \omterm{def:b3:topology:continuity}{continuous}. \emph{Uniqueness}: the difference of
two solutions is harmonic on $\mathbb D$, \omterm{def:b3:topology:continuity}{continuous} on the
\omterm{def:b3:topology:interior}{closure}, zero on the boundary: by the maximum principle
(applied to $\pm$ the difference), it vanishes.
\end{proof}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:b3:conformal:1}
(a) Verify that the \omterm{ex:b3:conformal:mobius}{Cayley map} $\varphi(z) = \frac{z -
\iu}{z + \iu}$ is a bijection $\mathbb H \to \mathbb D$ with
the stated inverse, and compute the images of $\iu$, $0$,
$1$, $\infty$ (limit).
(b) Show that $z \mapsto 1/z$ maps circles and lines to
circles and lines. \emph{(Write their common equation
$\alpha\abs z^2 + \bar\beta z + \beta\bar z + \gamma = 0$,
$\alpha, \gamma \in \R$.)} Deduce the same for all M\"obius
maps.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b3:conformal:2}
Let $f \colon \mathbb D \to \mathbb D$ be \omterm{def:b3:holomorphic:holo}{holomorphic}.
(a) If $f(0) = 0$ and $f(a) = a$ for some $a \neq 0$, show $f
= \mathrm{id}$.
(b) If $f$ is an automorphism with two distinct fixed points
in $\mathbb D$, show $f = \mathrm{id}$
\emph{(conjugate by a \omterm{thm:b3:conformal:autdisc}{Blaschke factor} to reduce to (a))}.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:conformal:3}
(Schwarz--Pick) For \omterm{def:b3:holomorphic:holo}{holomorphic} $f\colon \mathbb D \to
\mathbb D$, prove
\[
\frac{\abs{f'(z)}}{1 - \abs{f(z)}^2} \;\leq\;
\frac{1}{1 - \abs z^2}
\qquad (z \in \mathbb D),
\]
with equality (at one point, hence everywhere) iff $f \in
\operatorname{Aut}(\mathbb D)$.
\emph{(Apply Schwarz to $\varphi_{f(z)}\circ
f\circ\varphi_{-z}$.)} Interpretation: \omterm{def:b3:holomorphic:holo}{holomorphic} self-maps
contract the hyperbolic metric.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:conformal:4}
Find explicit conformal equivalences:
(a) the strip $\{0 < \operatorname{Im}z < \pi\} \to \mathbb H$;
(b) the quadrant $\{\operatorname{Re}z > 0,
\operatorname{Im}z > 0\} \to \mathbb H$;
(c) the half-disc $\mathbb D\cap\mathbb H \to$ a quadrant,
then $\to \mathbb H$;
(d) $\mathbb D \to \mathbb D$ sending $\frac12$ to $0$ with
positive derivative there.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:conformal:5}
(a) Show that no \omterm{def:b3:conformal:conformal}{conformal map} $\C \to \mathbb D$ or $\C \to
\mathbb H$ exists, and none $\mathbb D \to \C$.
(b) Which of the following are \omterm{def:b3:conformal:conformal}{conformally equivalent} to
$\mathbb D$? Justify via \cref{thm:b3:conformal:rmt} or an
obstruction: a square; $\C\setminus\intoc{-\infty}0$;
$\mathbb D\setminus\{0\}$; $\{1 < \abs z < 2\}$.
\emph{(For the last two: a conformal image of the punctured
disc would extend over the puncture by
\cref{thm:b3:residues:riemanncw}(1) --- develop this.)}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:conformal:6}
Let $\mathcal F = \{f \in \mathcal H(\mathbb D) : f(0) = 1,\
\operatorname{Re}f > 0\}$.
(a) Show that $\mathcal F$ is locally bounded.
\emph{(Compose with the Cayley-type map $w \mapsto \frac{w -
1}{w + 1}$ sending the right half-plane to $\mathbb D$, and
apply Schwarz.)}
(b) Deduce the \emph{Herglotz bound}: $\abs{f(z)} \leq
\frac{1 + \abs z}{1 - \abs z}$ for $f \in \mathcal F$, with
equality possibilities.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b3:conformal:7}
Where does the proof of \cref{thm:b3:conformal:rmt} use each
hypothesis? Trace: (i) simple connectivity (twice); (ii)
$\Omega \neq \C$; (iii) \omterm{def:b3:topology:connected}{connectedness}. Then show that the
theorem \emph{fails} for $\Omega = \C$ and for the annulus,
pinpointing which step of the proof breaks in each case.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:conformal:8}
Solve the Dirichlet problem on $\mathbb D$ for the boundary
data: (a) $g(\eu^{\iu t}) = \cos t$; (b) $g(\eu^{\iu t}) =
\cos^2 t$; (c) $g = \mathbf 1_{\text{upper semicircle}}$ ---
for (c) compute $u(0)$ and interpret via the mean value
property. \emph{(Expand $g$ in Fourier series and use
$P_r$'s series: $u(r\eu^{\iu\varphi}) = \sum_n c_n(g)
r^{\abs n}\eu^{\iu n\varphi}$.)}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b3:conformal:9}
(Harnack) Let $u \geq 0$ be harmonic on $\mathbb D$. Prove,
for $\abs z = r < 1$:
\[
\frac{1 - r}{1 + r}\,u(0) \;\leq\; u(z) \;\leq\;
\frac{1 + r}{1 - r}\,u(0)
\]
\emph{(bound the \omterm{thm:b3:conformal:poisson}{Poisson kernel} between $\frac{1-r}{1+r}$ and
$\frac{1+r}{1-r}$; apply the \omterm{def:b3:representations:rep}{representation} on slightly
smaller discs and pass to the limit)}. Deduce: a \omterm{prop:b3:conformal:harmonicholo}{harmonic
function} on $\C$ bounded below is constant.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:conformal:10}
Using conformal invariance of harmonicity ($u\circ f$ is
harmonic when $u$ is harmonic and $f$ \omterm{def:b3:holomorphic:holo}{holomorphic} --- prove
it via \cref{prop:b3:conformal:harmonicholo} locally), solve
the Dirichlet problem on the upper half-plane with boundary
data $\mathbf 1_{\intoo{-\infty}0}$: show that
\[
u(x + \iu y) = \frac1\pi\,\arg(x + \iu y)
\qquad (\arg \in \intoo0\pi \text{ on } \mathbb H)
\]
is harmonic on $\mathbb H$ (imaginary part of a \omterm{def:b3:conformal:simplyconnected}{holomorphic
logarithm}) with the required boundary limits at every $x \neq
0$, and transport it to the disc by Cayley to re-derive
\cref{exo:b3:conformal:8}(c).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:conformal:11}
(Fixed points and iteration in the disc) Let $f\colon\mathbb
D \to \mathbb D$ be \omterm{def:b3:holomorphic:holo}{holomorphic}.
(a) Show that if $f$ has \emph{two} distinct fixed points,
then $f = \mathrm{id}$ \emph{(move one to $0$ by an
automorphism and apply the equality case of Schwarz)}.
(b) Suppose $f(0) = 0$ and $f$ is not a rotation. Show that
the iterates $f^{\circ n} \to 0$ uniformly on every \omterm{def:b3:topology:compact}{compact}
$\bar D(0, r)$, $r < 1$ \emph{(Schwarz gives $\abs{f(z)}
\leq c_r\abs z$ on $\bar D(0,r)$ with $c_r < 1$ --- justify
this strict constant via the maximum principle applied to
$f(z)/z$)}.
(c) Illustrate with $f(z) = \frac{z^2 + z}2$: fixed points,
and the rate of convergence of the \omterm{def:b3:groups:action}{orbit} of $z_0 =
\frac12$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:conformal:12}
(Harmonic conjugates, concretely) Let $u(x, y) = x^3 - 3xy^2
+ 2y$.
(a) Check that $u$ is harmonic on $\R^2$, and find all
harmonic conjugates $v$ (i.e.\ $u + \iu v$ \omterm{def:b3:holomorphic:holo}{holomorphic}) by
integrating the \omterm{prop:b3:holomorphic:cauchyriemann}{Cauchy--Riemann equations}; identify $f(z) =
u + \iu v$ as a polynomial in $z$.
(b) Show that on a \emph{star-shaped} \omterm{def:b3:topology:topology}{open set}, every
\omterm{prop:b3:conformal:harmonicholo}{harmonic function} admits a harmonic conjugate, unique up to
an additive constant \emph{(the $1$-form $-u_y\,\dd x +
u_x\,\dd y$ is closed; \cref{thm:b3:holomorphic:cauchy}'s
primitive machinery, or \cref{ch:b3:forms}'s Poincar\'e
lemma)}.
(c) Give the standard counterexample on
$\C^*$: $u = \ln\abs z$ has no global conjugate --- relate
to the angular form and the winding number
(\cref{ch:b3:forms}).
\end{exercise}

\section{Problem: the area theorem and Koebe's quarter
theorem}

\begin{problem}[{Weekend problem --- how much must a
univalent map cover?}]\label{pb:b3:conformal:1}
A \emph{univalent} function is a \omterm{def:b3:holomorphic:holo}{holomorphic} injection. The
normalized univalent functions on the disc,
\[
\mathcal S = \bigl\{f \in \mathcal H(\mathbb D) \text{
injective},\ f(z) = z + a_2z^2 + a_3z^3 + \cdots\bigr\},
\]
are rigidly constrained: we prove Bieberbach's inequality
$\abs{a_2} \leq 2$ and deduce the \emph{Koebe quarter
theorem}: the image of any $f \in \mathcal S$ contains the
disc $D(0, \frac14)$ --- the sharp universal constant of
conformal geometry.\index{Koebe quarter theorem}\index{area
theorem}

\medskip
\noindent\textbf{Part I --- The area theorem.} Let $g(w) = w
+ b_0 + \frac{b_1}w + \frac{b_2}{w^2} + \cdots$ be \omterm{def:b3:holomorphic:holo}{holomorphic}
and \emph{injective} on $\{\abs w > 1\}$.
\begin{enumerate}
  \item For $\rho > 1$, let $A_\rho$ be the area (\omterm{def:b3:measure:lebesgueouter}{Lebesgue
        measure}) of the \omterm{def:b3:topology:compact}{compact} set $K_\rho = \C\setminus
        g(\{\abs w > \rho\})$, the region enclosed by the
        smooth Jordan curve $g(C_\rho)$. Using the area
        formula of the Year 2 volume's Green--Riemann theorem
        --- the enclosed area is $\frac1{2\iu}
        \oint\bar\zeta\,\dd\zeta$ along the positively
        oriented boundary --- show that
        \[
        A_\rho = \frac{1}{2\iu}\int_{C_\rho}
        \overline{g(w)}\,g'(w)\,\dd w
        = \pi\Bigl(\rho^2 -
        \sum_{n\geq1}n\,\abs{b_n}^2\rho^{-2n}\Bigr) :
        \]
        substitute the \omterm{thm:b3:residues:laurent}{Laurent series} of $\bar g$ and $g'$ on
        $C_\rho$ and integrate term by term (normal
        convergence; only the frequency-zero products
        survive).
  \item Let $\rho \to 1^+$ and conclude the \emph{area
        theorem}:
        \[
        \sum_{n\geq1}n\,\abs{b_n}^2 \;\leq\; 1 .
        \]
        In particular $\abs{b_1} \leq 1$. When is $\abs{b_1}
        = 1$?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Bieberbach's $\abs{a_2} \leq
2$.} Let $f = z + a_2z^2 + \cdots \in \mathcal S$.
\begin{enumerate}[resume]
  \item Show that $f(z^2)/z^2$ is \omterm{def:b3:holomorphic:holo}{holomorphic} and zero-free
        on $\mathbb D$, and admits a \omterm{def:b3:holomorphic:holo}{holomorphic} square root
        $\varphi$ with $\varphi(0) = 1$; set $h(z) =
        z\varphi(z^2)$, so that $h(z)^2 = f(z^2)$. Show that
        $h$ is an \emph{odd} univalent function on $\mathbb
        D$ with expansion $h(z) = z + \frac{a_2}2z^3 +
        \cdots$. \emph{(Injectivity: $h(z)^2 = h(z')^2$
        forces $z^2 = z'^2$; use oddness to finish.)}
  \item Apply the area theorem to $g(w) = 1/h(1/w) = w -
        \frac{a_2}{2w} + \cdots$ on $\{\abs w > 1\}$
        (verify univalence and the expansion), and conclude
        $\abs{a_2} \leq 2$.
  \item Show that the \emph{Koebe function}
        \[
        k(z) = \frac{z}{(1 - z)^2} = \sum_{n\geq1}n\,z^n
        \]
        belongs to $\mathcal S$, has $a_2 = 2$, and maps
        $\mathbb D$ onto $\C\setminus\intoc{-\infty}{-\frac14}$
        \emph{(write $k = \frac14\bigl[\bigl(\frac{1+z}{1 -
        z}\bigr)^2 - 1\bigr]$ and track the images)}: all
        inequalities to come are sharp.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The quarter theorem.}
\begin{enumerate}[resume]
  \item Let $f \in \mathcal S$ and $c \notin f(\mathbb D)$.
        Show that
        \[
        F(z) = \frac{c\,f(z)}{c - f(z)}
        \]
        belongs to $\mathcal S$, and compute its second
        coefficient: $A_2 = a_2 + \frac1c$.
  \item Apply Bieberbach to both $f$ and $F$: conclude
        $\abs{\frac1c} \leq \abs{A_2} + \abs{a_2} \leq 4$,
        i.e.\ $\abs c \geq \frac14$. \emph{Every omitted value
        has modulus $\geq \frac14$}: $f(\mathbb D) \supseteq
        D(0, \frac14)$ --- Koebe's quarter theorem. Check
        sharpness on the Koebe function.
  \item Deduce a quantitative Riemann-map estimate: if
        $\varphi \colon \Omega \to \mathbb D$ is the Riemann
        map of \cref{thm:b3:conformal:rmt} at $z_0$, then
        \[
        \frac{d\bigl(z_0, \partial\Omega\bigr)}{4}
        \;\leq\; \frac1{\varphi'(z_0)} \;\leq\;
        4\,d\bigl(z_0, \partial\Omega\bigr)
        \]
        --- prove at least the left inequality by applying
        Koebe to $\varphi^{-1}$ suitably normalized, and the
        right by Schwarz applied to $\varphi$ on the disc
        $D(z_0, d) \subseteq \Omega$.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Perspective.}
\begin{enumerate}[resume]
  \item Bieberbach conjectured (1916) $\abs{a_n} \leq n$ for
        all $n$, with equality only for rotations of the
        Koebe function; de Branges proved it in 1985. Verify
        the conjecture by hand for the Koebe function and its
        rotations $\eu^{-\iu\theta}k(\eu^{\iu\theta}z)$. Then
        push question 4's expansion one term further: writing
        $h(z) = z + \frac{a_2}2z^3 + c_5z^5 + \cdots$, show
        $c_5 = \frac{a_3}2 - \frac{a_2^2}8$ and
        \[
        g(w) = w - \frac{a_2}{2}\,w^{-1} +
        \Bigl(\frac{3a_2^2}8 - \frac{a_3}2\Bigr)w^{-3} +
        \cdots ,
        \]
        so the area theorem yields the refined inequality
        $\bigl|\frac{a_2}2\bigr|^2 + 3\bigl|\frac{3a_2^2}8 -
        \frac{a_3}2\bigr|^2 \leq 1$. Check it on the Koebe
        function ($a_2 = 2$, $a_3 = 3$).
\end{enumerate}

\medskip
\noindent\textbf{Part V --- The distortion theorem.}
Bieberbach's inequality, transported around the disc by
automorphisms, controls $f'$ everywhere. Fix $f \in \mathcal
S$.
\begin{enumerate}[resume]
  \item (Koebe transform) For $z_0 \in \mathbb D$ let
        $\varphi(z) = \frac{z + z_0}{1 + \bar z_0z}$, a disc
        automorphism (\cref{thm:b3:conformal:autdisc}) with
        $\varphi(0) = z_0$. Show that
        \[
        F(z) = \frac{f(\varphi(z)) - f(z_0)}
        {f'(z_0)\,\bigl(1 - \abs{z_0}^2\bigr)}
        \]
        belongs to $\mathcal S$ \emph{(univalence is
        inherited; compute $\varphi'(0) = 1 - \abs{z_0}^2$
        and check the normalization; recall $f' \neq 0$ for
        injective $f$,
        \cref{def:b3:conformal:conformal})}.
  \item Compute the second coefficient $A_2 = \frac12F''(0)$
        of $F$:
        \[
        A_2 = \frac12\Bigl[\bigl(1 - \abs{z_0}^2\bigr)
        \frac{f''(z_0)}{f'(z_0)} - 2\bar z_0\Bigr] ,
        \]
        and deduce from Bieberbach (question 4), for $z =
        r\eu^{\iu\theta}$, the \emph{fundamental
        inequality}:
        \[
        \Bigl|\,z\,\frac{f''(z)}{f'(z)} -
        \frac{2r^2}{1 - r^2}\Bigr|
        \;\leq\; \frac{4r}{1 - r^2} .
        \]
  \item Extract the real part:
        \[
        \frac{2r^2 - 4r}{1 - r^2} \;\leq\;
        \operatorname{Re}\Bigl(z\,\frac{f''(z)}{f'(z)}\Bigr)
        \;\leq\; \frac{2r^2 + 4r}{1 - r^2} .
        \]
  \item Show that $\frac{\dd}{\dd t}\log\bigl|
        f'(t\eu^{\iu\theta})\bigr| = \frac1t
        \operatorname{Re}\bigl(z\frac{f''(z)}{f'(z)}\bigr)$
        at $z = t\eu^{\iu\theta}$ \emph{(for a nonvanishing
        $\mathcal C^1$ function $g$ of a real variable,
        $\frac{\dd}{\dd t}\log\abs g =
        \operatorname{Re}(g'/g)$)}, and integrate question
        12's bounds along the ray to obtain the
        \emph{distortion theorem}:
        \[
        \frac{1 - r}{(1 + r)^3} \;\leq\; \abs{f'(z)}
        \;\leq\; \frac{1 + r}{(1 - r)^3},
        \qquad \abs z = r .
        \]
  \item Deduce the \emph{growth theorem}:
        \[
        \frac{r}{(1 + r)^2} \;\leq\; \abs{f(z)} \;\leq\;
        \frac{r}{(1 - r)^2},
        \qquad \abs z = r
        \]
        \emph{(upper bound: integrate $f'$ on the segment
        $[0, z]$; lower bound: if $\abs{f(z)} < \frac14$,
        the segment $[0, f(z)]$ lies in $f(\mathbb D)$ by
        question 7; pull it back by $f^{-1}$ ---
        \omterm{def:b3:holomorphic:holo}{holomorphic} by
        \cref{cor:b3:residues:openmapping} --- and bound
        $\abs{f(z)} = \int_\gamma\abs{f'(\zeta)}
        \,\abs{\dd\zeta} \geq \int_0^r\frac{1 - t}{(1 +
        t)^3}\,\dd t$, using $\abs{\dd\zeta} \geq
        \dd\abs\zeta$)}.
  \item Verify that the Koebe function achieves equality in
        all four bounds, at $z = r$ for the upper ones and
        $z = -r$ for the lower ones: $k$ is simultaneously
        the most expanding and, at the antipode, the most
        contracting member of $\mathcal S$.
\end{enumerate}

\medskip
\noindent\textbf{Part VI --- Extremal rigidity.} In every
inequality so far, equality identifies the Koebe function up
to rotation. We prove it, then harvest.
\begin{enumerate}[resume]
  \item Suppose $f \in \mathcal S$ has $\abs{a_2} = 2$.
        Chase equality through questions 2--4: the area
        theorem forces $g(w) = w + b_0 + \eu^{\iu\alpha}/w$;
        oddness of $h$ makes $g$ odd, so $b_0 = 0$; invert
        to find $h$, then $f$, and conclude that
        \[
        f(z) = \eu^{-\iu\theta}k\bigl(\eu^{\iu\theta}z\bigr)
        \quad\text{with } \eu^{\iu\theta} =
        -\eu^{\iu\alpha} :
        \]
        the rotations of the Koebe function are the only
        members of $\mathcal S$ with $\abs{a_2} = 2$.
  \item Show that if $f \in \mathcal S$ omits a value $c$
        with $\abs c = \frac14$ exactly, then $f$ is a
        rotation of the Koebe function, and identify $c =
        -\eu^{-\iu\theta}/4$ \emph{(trace equality through
        question 7's chain $4 = \abs{1/c} = \abs{A_2 - a_2}
        \leq \abs{A_2} + \abs{a_2} \leq 4$)}: the quarter
        theorem's constant is attained only by the extremal
        family.
  \item (Coefficients on the cheap) Combine the growth
        theorem with the \omterm{thm:b3:holomorphic:analytic}{Cauchy estimates}
        (\cref{thm:b3:holomorphic:analytic}) on the circle
        $\abs z = 1 - \frac1n$ to prove
        \[
        \abs{a_n} \;\leq\; \eu\,n^2 \qquad (n \geq 2) .
        \]
        (De Branges, 1985: $\abs{a_n} \leq n$; the factor
        $\eu n$ is the price of elementary tools.)
  \item (Covering of subdiscs) Show that for every $0 < r <
        1$,
        \[
        f\bigl(D(0, r)\bigr) \supseteq
        D\Bigl(0, \frac{r}{(1 + r)^2}\Bigr),
        \]
        sharp for the Koebe function, and recover the
        quarter theorem as $r \to 1^-$. \emph{(Boundary
        points of the open image $f(D(0,r))$ lie on
        $f(\partial D(0,r))$, hence have modulus $\geq
        \frac{r}{(1+r)^2}$ by question 14; a segment from
        $0$ to a missed point of smaller modulus would have
        to cross that boundary.)}
  \item (Koebe at every point) Let $f$ be univalent on
        $\mathbb D$, not necessarily normalized, and $z_0
        \in \mathbb D$. Prove
        \[
        \tfrac14\bigl(1 - \abs{z_0}^2\bigr)\abs{f'(z_0)}
        \;\leq\;
        d\bigl(f(z_0), \partial f(\mathbb D)\bigr)
        \;\leq\;
        \bigl(1 - \abs{z_0}^2\bigr)\abs{f'(z_0)}
        \]
        \emph{(left: quarter theorem applied to the Koebe
        transform of question 10; right: Schwarz
        (\cref{thm:b3:conformal:schwarz}) applied to
        $\psi^{-1}\circ\hat g$, where $\hat g(w) =
        f^{-1}\bigl(f(z_0) + dw\bigr)$, $d$ the distance,
        and $\psi$ a disc automorphism sending $0$ to
        $z_0$)}. Why is $\partial f(\mathbb D)$ nonempty?
  \item Check question 20 on $f = k$ at $z_0 = r \in
        \intoo01$: compute $d\bigl(k(r), \partial k(\mathbb
        D)\bigr) = \frac{(1+r)^2}{4(1-r)^2}$ and verify that
        the left inequality is an \emph{equality}: the Koebe
        function saturates its own theorem at every point of
        $\intoo01$.
  \item (The moral) In ten lines: what single principle
        underlies the area theorem, and how do Bieberbach,
        the quarter theorem, distortion, growth, and
        covering all flow from it? Compare with the
        Schwarz--Pick world of
        \cref{thm:b3:conformal:schwarz}: in both, one
        \omterm{def:b3:topology:interior}{interior} inequality rigidifies the whole geometry,
        and the extremals are unique up to rotation.

\end{enumerate}
\medskip
\noindent\textbf{Part VII --- Compactness, inverses, and a
reality check.}
\begin{enumerate}[resume]
  \item Show that the class $\mathcal S$ is \emph{\omterm{def:b3:topology:compact}{compact}}
        for locally uniform convergence: it is locally
        bounded by the growth theorem, hence normal
        (\cref{thm:b3:conformal:montel}); and a locally
        uniform limit of members of $\mathcal S$ is again in
        $\mathcal S$ \emph{(the normalizations pass to the
        limit by Weierstrass convergence of derivatives;
        injectivity survives by Hurwitz,
        \cref{exo:b3:residues:8}, the limit being
        nonconstant)}. Why does this matter for extremal
        problems like Bieberbach's?
  \item For $f \in \mathcal S$, let $g = f^{-1}$, defined
        near $0$. Show $g(w) = w - a_2w^2 + O(w^3)$, so the
        inverse's second coefficient obeys the same sharp
        bound $\abs{A_2} = \abs{a_2} \leq 2$, with equality
        exactly for the rotated Koebe functions.
  \item Determine for which $a \in \C$ the polynomial $f(z)
        = z + az^2$ belongs to $\mathcal S$: show $f$ is
        injective on $\mathbb D$ iff $\abs a \leq \frac12$
        \emph{(factor $f(z_1) - f(z_2)$)}. Conclude: for
        degree-two polynomials the true coefficient bound is
        $\frac12$, four times smaller than Bieberbach's $2$
        --- the extremals of $\mathcal S$ are genuinely
        transcendental objects, and no polynomial comes
        close.
\end{enumerate}
\end{problem}
