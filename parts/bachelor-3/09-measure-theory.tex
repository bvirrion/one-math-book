\chapter{Measure Theory}\label{ch:b3:measure}

How long is a subset of $\R$? The naive answer --- assign to
every set a translation-invariant length extending that of
intervals --- is \emph{impossible}: Vitali's construction, at
the end of this chapter, produces a set with no consistent
length. Measure theory is the disciplined retreat: restrict
attention to a rich class of \emph{measurable} sets, on which a
countably additive length exists and is unique. The rewards are
immense --- Lebesgue's integral (\cref{ch:b3:lebesgue}), the
$L^p$ spaces of functional analysis, and the whole of modern
probability (\cref{ch:b3:probability}) are built on the three
theorems proved here: Dynkin's uniqueness lemma,
Carath\'eodory's extension theorem, and the existence of
Lebesgue measure.

\section{$\sigma$-algebras}

\begin{definition}\label{def:b3:measure:sigmaalgebra}
A \emph{$\sigma$-algebra}\index{sigma-algebra@$\sigma$-algebra}
on a set $X$ is a family $\mathcal A$ of subsets containing
$\varnothing$, stable under complement and under
\emph{countable} unions (hence countable intersections, set
differences, and it contains $X$). The pair $(X, \mathcal A)$
is a \emph{measurable space}; members of $\mathcal A$ are
\emph{measurable sets}. For any family $\mathcal E$ of subsets,
$\sigma(\mathcal E)$ denotes the smallest $\sigma$-algebra
containing $\mathcal E$ (the intersection of all of them ---
an intersection of $\sigma$-algebras is one).
\end{definition}

\begin{definition}\label{def:b3:measure:borel}
The \emph{Borel $\sigma$-algebra}\index{Borel sigma-algebra@%
Borel $\sigma$-algebra} of a topological space is
$\mathcal B(X) = \sigma(\{\text{open sets}\})$. On $\R$:
$\mathcal B(\R)$ is also generated by the open intervals, by
the closed intervals, by the rays $(-\infty, a]$, and by rays
with rational endpoints (\cref{exo:b3:measure:1}) --- each
family generates the opens by countable operations, e.g.\
every open set of $\R$ is a countable union of open intervals
with rational data.
\end{definition}

\begin{definition}\label{def:b3:measure:dynkin}
A \emph{$\pi$-system}\index{pi-system@$\pi$-system} is a family
stable under finite intersections. A
\emph{$\lambda$-system}\index{lambda-system@$\lambda$-system}
(Dynkin class) is a family $\mathcal D$ with: $X \in \mathcal
D$; $A, B \in \mathcal D$, $A \subseteq B$ $\Rightarrow$ $B
\setminus A \in \mathcal D$; and $A_n \uparrow A$, $A_n \in
\mathcal D$ $\Rightarrow$ $A \in \mathcal D$.
\end{definition}

\begin{theorem}[Dynkin's $\pi$--$\lambda$ lemma]
\label{thm:b3:measure:dynkin}
If a $\lambda$-system $\mathcal D$ contains a $\pi$-system
$\mathcal P$, then $\mathcal D \supseteq
\sigma(\mathcal P)$.\index{Dynkin's lemma}
\end{theorem}

\begin{proof}
Let $\mathcal D_0$ be the smallest $\lambda$-system containing
$\mathcal P$ (intersection of all such); it suffices to show
$\mathcal D_0$ is a $\sigma$-algebra, since then
$\sigma(\mathcal P) \subseteq \mathcal D_0 \subseteq \mathcal
D$. A $\lambda$-system stable under finite intersections
\emph{is} a $\sigma$-algebra: complements ($X \setminus A = X
\setminus A$ with $A \subseteq X$), finite unions ($A \cup B =
X \setminus ((X\setminus A)\cap(X\setminus B))$), and countable
unions via $\bigcup_{k \leq n}A_k \uparrow \bigcup_kA_k$. So we
prove $\mathcal D_0$ is a $\pi$-system, in two steps. Let
\[
\mathcal D_1 = \{A \in \mathcal D_0 : A \cap P \in \mathcal
D_0 \ \forall P \in \mathcal P\}.
\]
$\mathcal D_1$ is a $\lambda$-system (all three axioms are
verified by intersecting with $P$: e.g.\ $(B\setminus A)\cap P
= (B \cap P)\setminus(A \cap P)$, a proper difference within
$\mathcal D_0$) and contains $\mathcal P$ ($\pi$-system):
$\mathcal D_1 = \mathcal D_0$. Now let
\[
\mathcal D_2 = \{A \in \mathcal D_0 : A \cap D \in \mathcal
D_0\ \forall D \in \mathcal D_0\}.
\]
By the previous step, $\mathcal D_2 \supseteq \mathcal P$; and
$\mathcal D_2$ is a $\lambda$-system by the same verification:
$\mathcal D_2 = \mathcal D_0$, which says exactly that
$\mathcal D_0$ is stable under intersections.
\end{proof}

\section{Measures}

\begin{definition}\label{def:b3:measure:measure}
A \emph{measure}\index{measure} on $(X, \mathcal A)$ is a map
$\mu \colon \mathcal A \to [0, +\infty]$ with $\mu(\varnothing)
= 0$ that is \emph{$\sigma$-additive}: for pairwise disjoint
$(A_n)_{n\in\N}$,
\[
\mu\Bigl(\bigsqcup_n A_n\Bigr) = \sum_n \mu(A_n).
\]
$(X, \mathcal A, \mu)$ is a \emph{measure space}; $\mu$ is
\emph{finite} if $\mu(X) < \infty$, a \emph{probability
measure} if $\mu(X) = 1$, \emph{$\sigma$-finite} if $X$ is a
countable union of sets of finite measure. Examples: counting
measure on $(\N, \mathcal P(\N))$; the Dirac mass $\delta_a(A)
= \mathbf 1_{a \in A}$; and, the object of this chapter,
Lebesgue measure.
\end{definition}

\begin{proposition}\label{prop:b3:measure:basics}
Let $\mu$ be a measure. (a) Monotonicity: $A \subseteq B
\Rightarrow \mu(A) \leq \mu(B)$. (b) Countable subadditivity:
$\mu(\bigcup A_n) \leq \sum\mu(A_n)$. (c) Continuity from
below: $A_n \uparrow A \Rightarrow \mu(A_n) \to \mu(A)$. (d)
Continuity from above: $A_n \downarrow A$ \emph{with
$\mu(A_1) < \infty$} $\Rightarrow \mu(A_n) \to \mu(A)$.
\end{proposition}

\begin{proof}
(a) $B = A \sqcup (B\setminus A)$. (b) Disjointify: $B_n = A_n
\setminus \bigcup_{k<n}A_k$ are disjoint with the same union,
and $\mu(B_n) \leq \mu(A_n)$. (c) $A = \bigsqcup_n (A_n
\setminus A_{n-1})$ ($A_0 = \varnothing$): the partial sums of
$\sum\mu(A_n\setminus A_{n-1})$ are $\mu(A_n)$. (d) Apply (c)
to $A_1 \setminus A_n \uparrow A_1 \setminus A$ and subtract
from $\mu(A_1)$ --- finiteness makes the subtraction
legitimate. Counterexample without it: $A_n = [n, \infty)$ for
Lebesgue measure: $A_n \downarrow \varnothing$ but $\mu(A_n) =
\infty$.
\end{proof}

\begin{theorem}[Uniqueness]\label{thm:b3:measure:uniqueness}
Let $\mu, \nu$ be measures on $\sigma(\mathcal P)$, $\mathcal
P$ a $\pi$-system, with $\mu = \nu$ on $\mathcal P$. If there
are sets $P_k \in \mathcal P$ with $P_k \uparrow X$ and
$\mu(P_k) < \infty$, then $\mu = \nu$ on all of
$\sigma(\mathcal P)$.
\end{theorem}

\begin{proof}
Fix $k$ and consider the finite measures $\mu_k(A) = \mu(A \cap
P_k)$ and $\nu_k(A) = \nu(A \cap P_k)$ on $\sigma(\mathcal
P)$: they agree on $\mathcal P$, since $P \cap P_k \in \mathcal
P$ ($\pi$-system), and they assign $X$ the same finite value
$\mu(P_k)$. The class $\mathcal D = \{A : \mu_k(A) =
\nu_k(A)\}$ is a $\lambda$-system: $X \in \mathcal D$; proper
differences by subtraction (finite values); increasing limits
by continuity from below (\cref{prop:b3:measure:basics}(c)).
It contains the $\pi$-system $\mathcal P$, so Dynkin
(\cref{thm:b3:measure:dynkin}) gives $\mathcal D \supseteq
\sigma(\mathcal P)$: $\mu_k = \nu_k$ everywhere. Finally, for
any $A \in \sigma(\mathcal P)$, continuity from below along $A
\cap P_k \uparrow A$ gives $\mu(A) = \lim_k\mu_k(A) =
\lim_k\nu_k(A) = \nu(A)$.
\end{proof}

\section{Outer measures and Carath\'eodory's theorem}

\begin{definition}\label{def:b3:measure:outer}
An \emph{outer measure}\index{outer measure} on $X$ is a map
$\mu^* \colon \mathcal P(X) \to [0, \infty]$ with
$\mu^*(\varnothing) = 0$, monotone, and countably subadditive.
A set $A$ is \emph{$\mu^*$-measurable} (Carath\'eodory) if it
splits every set additively:
\[
\mu^*(E) = \mu^*(E \cap A) + \mu^*(E \setminus A)
\qquad \text{for every } E \subseteq X
\]
($\leq$ always holds by subadditivity; the content is $\geq$).
\end{definition}

\begin{theorem}[Carath\'eodory]\label{thm:b3:measure:caratheodory}
The $\mu^*$-measurable sets form a $\sigma$-algebra $\mathcal
M$, and $\mu^*\restriction_{\mathcal M}$ is a measure.
Moreover every set with $\mu^*(N) = 0$ belongs to $\mathcal M$
(the measure is \emph{complete}).\index{Carath\'eodory's
theorem}
\end{theorem}

\begin{proof}
$\mathcal M$ contains $\varnothing$ and is stable under
complement (the defining condition is symmetric in $A$,
$X\setminus A$). \emph{Finite unions}: let $A, B \in \mathcal
M$ and $E$ arbitrary; splitting $E$ by $A$, then each piece by
$B$:
\[
\mu^*(E) = \mu^*(E\cap A\cap B) + \mu^*(E\cap A\setminus B) +
\mu^*(E\cap B\setminus A) + \mu^*(E\setminus(A\cup B)).
\]
The first three pieces cover $E \cap (A \cup B)$, so
subadditivity gives $\mu^*(E) \geq \mu^*(E\cap(A\cup B)) +
\mu^*(E\setminus(A\cup B))$: $A \cup B \in \mathcal M$. By
induction, finite unions; with complements, finite disjointness
manipulations are available.

\emph{Additivity on $\mathcal M$}: for disjoint $A, B \in
\mathcal M$ and any $E$: $\mu^*(E\cap(A\sqcup B)) = \mu^*(E\cap
A) + \mu^*(E \cap B)$ (split by $A$); by induction,
\[
\mu^*\Bigl(E \cap \bigsqcup_{k\leq n}A_k\Bigr)
= \sum_{k\leq n}\mu^*(E\cap A_k).
\tag{$*$}
\]

\emph{Countable unions}: let $(A_k) \subseteq \mathcal M$ be
disjoint (suffices, by disjointification within the algebra
$\mathcal M$), $A = \bigsqcup A_k$, $E$ arbitrary. Using
$\bigsqcup_{k\leq n}A_k \in \mathcal M$ and monotonicity:
\[
\mu^*(E) = \mu^*\Bigl(E\cap\bigsqcup_{k\leq n}A_k\Bigr) +
\mu^*\Bigl(E\setminus\bigsqcup_{k\leq n}A_k\Bigr)
\geq \sum_{k \leq n}\mu^*(E\cap A_k) + \mu^*(E\setminus A)
\]
by ($*$). Let $n \to \infty$ and use countable subadditivity
backwards:
\[
\mu^*(E) \geq \sum_{k}\mu^*(E\cap A_k) + \mu^*(E\setminus A)
\geq \mu^*(E \cap A) + \mu^*(E\setminus A) \geq \mu^*(E):
\]
all inequalities are equalities. This proves both $A \in
\mathcal M$ and, taking $E = A$, countable additivity of
$\mu^*$ on $\mathcal M$.

\emph{Null sets}: if $\mu^*(N) = 0$, then for any $E$:
$\mu^*(E \cap N) + \mu^*(E\setminus N) \leq 0 + \mu^*(E)$:
$N \in \mathcal M$.
\end{proof}

\section{Lebesgue measure on \texorpdfstring{$\R$}{R}}

\begin{definition}\label{def:b3:measure:lebesgueouter}
The \emph{Lebesgue outer measure}\index{Lebesgue measure} of $A
\subseteq \R$ is
\[
\lambda^*(A) = \inf\Bigl\{\sum_{n} (b_n - a_n) :
A \subseteq \bigcup_n \intoo{a_n}{b_n}\Bigr\}
\]
(countable covers by open intervals).
\end{definition}

\begin{lemma}\label{lem:b3:measure:outerbasics}
$\lambda^*$ is an outer measure, invariant under translations,
and $\lambda^*(I) = \ell(I)$ (the length) for every interval
$I$.
\end{lemma}

\begin{proof}
Outer measure: $\varnothing$ is covered by arbitrarily small
intervals; monotonicity is clear; subadditivity: given covers
of each $A_n$ within $\varepsilon 2^{-n}$ of the infimum,
their union covers $\bigcup A_n$ with total length $\leq \sum
\lambda^*(A_n) + \varepsilon$. Translation invariance:
translate the covers.

Length: it suffices to treat $I = \intcc ab$ (other types
differ by endpoints, which have outer measure $0$: cover by
tiny intervals; then squeeze $\intcc{a+\varepsilon}{b -
\varepsilon} \subseteq \intoo ab$-type comparisons).
$\lambda^*(\intcc ab) \leq b - a$: cover by
$\intoo{a-\varepsilon}{b+\varepsilon}$. Conversely let $\intcc
ab \subseteq \bigcup_n\intoo{a_n}{b_n}$: by
\emph{compactness} (Borel--Lebesgue,
\cref{thm:b3:topology:metriccompact}), finitely many intervals
suffice, say $I_1, \dots, I_N$. We show $\sum_{k\leq N}(b_k -
a_k) \geq b - a$ by induction on $N$: choose $I_{k_1} \ni a$; if
$b_{k_1} > b$ done ($b_{k_1} - a_{k_1} > b - a$); else the
segment $\intcc{b_{k_1}}b$ is covered by the remaining $N - 1$
intervals, and induction gives $\sum_{k \neq k_1}(b_k - a_k)
\geq b - b_{k_1}$, while $b_{k_1} - a_{k_1} > b_{k_1} - a$:
sum up.
\end{proof}

\begin{theorem}[Lebesgue measure]\label{thm:b3:measure:lebesgue}
Every Borel set of $\R$ is $\lambda^*$-measurable. The
restriction $\lambda$ of $\lambda^*$ to the $\sigma$-algebra
$\mathcal L = \mathcal M_{\lambda^*} \supseteq \mathcal B(\R)$
(the \emph{Lebesgue $\sigma$-algebra}) is the unique measure on
$\mathcal B(\R)$ assigning to each interval its length; it is
translation-invariant and $\sigma$-finite.
\end{theorem}

\begin{proof}
By \cref{thm:b3:measure:caratheodory} it suffices to show each
ray $A = \intoo{-\infty}c$ is $\lambda^*$-measurable
(rays generate $\mathcal B$,
\cref{def:b3:measure:borel}). Let $E \subseteq \R$ with
$\lambda^*(E) < \infty$ and $\bigcup I_n \supseteq E$ a cover
with $\sum\ell(I_n) \leq \lambda^*(E) + \varepsilon$. Each
$I_n$ splits into the two intervals $I_n' = I_n \cap A$ and
$I_n'' = I_n\setminus A$ (an interval minus a ray is an
interval) with $\ell(I_n') + \ell(I_n'') = \ell(I_n)$; the
$I_n'$ cover $E \cap A$ and the $I_n''$ cover $E \setminus A$
(enlarge each to an open interval of length $\ell +
\varepsilon2^{-n}$ to stay within the definition), so
\[
\lambda^*(E\cap A) + \lambda^*(E\setminus A)
\leq \sum_n\bigl(\ell(I_n') + \ell(I_n'')\bigr) + 2\varepsilon
\leq \lambda^*(E) + 3\varepsilon .
\]
Uniqueness: two measures agreeing with length on the
$\pi$-system of intervals $\intoc ab$ (finite on them) agree on
$\sigma(\text{intervals}) = \mathcal B$ by
\cref{thm:b3:measure:uniqueness} with $P_k = \intoc{-k}k$.
$\sigma$-finiteness: $\R = \bigcup(-k, k]$.
\end{proof}

\begin{theorem}[Regularity]\label{thm:b3:measure:regularity}
For every $A \in \mathcal L$:\index{regularity of Lebesgue
measure}
\[
\lambda(A) = \inf\{\lambda(U) : U \supseteq A \text{ open}\}
= \sup\{\lambda(K) : K \subseteq A \text{ compact}\}.
\]
\end{theorem}

\begin{proof}
\emph{Outer}: a cover $\bigcup I_n$ with $\sum\ell(I_n) \leq
\lambda(A) + \varepsilon$ is an open set $U \supseteq A$ with
$\lambda(U) \leq \lambda(A) + \varepsilon$ (subadditivity); if
$\lambda(A) = \infty$ the statement is trivial. \emph{Inner}:
first let $A$ be bounded, $A \subseteq [-M, M]$. Choose an open
$U \supseteq ([-M,M]\setminus A)$ with $\lambda(U) \leq
\lambda([-M,M]\setminus A) + \varepsilon$; then $K = [-M,
M]\setminus U$ is compact, $K \subseteq A$, and
\[
\lambda(K) = \lambda([-M,M]) - \lambda([-M,M]\cap U)
\geq \lambda([-M,M]) - \bigl(\lambda([-M,M]) - \lambda(A) +
\varepsilon\bigr) = \lambda(A) - \varepsilon .
\]
For general $A$: $\lambda(A) = \lim_M\lambda(A\cap[-M,M])$
(continuity from below) and apply the bounded case inside.
\end{proof}

\begin{example}\label{ex:b3:measure:cantor}
The Cantor set (\cref{exo:b3:topology:10}) has $\lambda(C) = 0$:
$C \subseteq C_n$, a union of $2^n$ intervals of length
$3^{-n}$, so $\lambda(C) \leq (2/3)^n \to 0$. An uncountable
null set --- cardinality does not see measure. Conversely,
\emph{fat Cantor sets} (\cref{exo:b3:measure:5}) are nowhere
dense with positive measure: topology does not see measure
either. The weekend problem pushes this interplay to its
striking conclusion: there are Lebesgue-measurable sets that
are not Borel.
\end{example}

\begin{theorem}[Vitali]\label{thm:b3:measure:vitali}
There is no measure on \emph{all} subsets of $\R$ that is
translation-invariant and assigns to each interval its length.
In particular $\mathcal L \neq \mathcal P(\R)$: non-measurable
sets exist.\index{Vitali's non-measurable set}
\end{theorem}

\begin{proof}
Suppose $\mu$ were one. On $\intcc01$, consider the equivalence
$x \sim y \iff x - y \in \Q$; by the \emph{axiom of choice}
pick one representative in $\intcc01$ per class: a set $V$. For
$q \in \Q\cap\intcc{-1}1$, the translates $V + q$ are pairwise
disjoint (two points of $V$ differing by a rational would be
equivalent yet distinct representatives) and
\[
\intcc01 \subseteq \bigsqcup_{q \in \Q\cap\intcc{-1}1}(V + q)
\subseteq \intcc{-1}2 :
\]
the first inclusion since every $x \in \intcc01$ differs from
its representative $v$ by a rational $q = x - v \in
\intcc{-1}1$. Monotonicity and $\sigma$-additivity give
\[
1 \leq \sum_{q}\mu(V + q) \leq 3,
\qquad\text{with } \mu(V + q) = \mu(V) \text{ for all } q .
\]
An infinite sum of the constant $\mu(V)$ is $0$ or $\infty$:
both bounds cannot hold. So no such $\mu$ exists --- and $V
\notin \mathcal L$, since $\lambda$ on $\mathcal L$ has all the
properties used.
\end{proof}

\begin{remark}\label{rem:b3:measure:banachtarski}
In $\R^3$ the failure is more dramatic: the Banach--Tarski
paradox decomposes a ball into five pieces that reassemble, by
rotations and translations, into \emph{two} balls of the same
radius --- so not even a finitely additive rotation-invariant
volume on all subsets of $\R^3$ exists. The pieces are, of
course, non-measurable. Measurability is not bureaucratic
caution; it is the boundary of coherence.
\end{remark}

\begin{method}\label{met:b3:measure:goodsets}
The \emph{good sets principle}: to prove that all sets of
$\sigma(\mathcal E)$ have a property, show that the good sets
form a $\sigma$-algebra (or a $\lambda$-system, if the property
is measure-theoretic and $\mathcal E$ is a $\pi$-system ---
then Dynkin) containing $\mathcal E$. Almost every proof of
this chapter and the next is an instance. To prove two measures
equal: check them on a generating $\pi$-system plus
$\sigma$-finiteness (\cref{thm:b3:measure:uniqueness}). To
build a measure: build an outer measure by covers and quote
Carath\'eodory.
\end{method}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:b3:measure:1}
(a) Show that $\{A \subseteq X : A$ or $X\setminus A$ is
countable$\}$ is a $\sigma$-algebra: the one generated by
singletons.
(b) Show that $\mathcal B(\R)$ is generated by each of: open
intervals; closed intervals; rays $\intoc{-\infty}a$; rays
with $a \in \Q$.
(c) Is the family of \emph{finite disjoint unions of intervals}
$\intoc ab$ a $\sigma$-algebra? An algebra (stable under
complement and finite unions)?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b3:measure:2}
(a) Prove inclusion--exclusion for a finite measure: $\mu(A\cup
B) = \mu(A) + \mu(B) - \mu(A\cap B)$, and the three-set
version.
(b) Give an example showing that continuity from above
(\cref{prop:b3:measure:basics}(d)) fails without the
finiteness assumption.
(c) Show that a countable set has Lebesgue measure zero.
Deduce $\lambda(\Q) = 0$ and $\lambda(\intcc01\setminus\Q) =
1$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:measure:3}
Let $\mu, \nu$ be probability measures on $\mathcal B(\R)$
with $\mu(\intoc{-\infty}t) = \nu(\intoc{-\infty}t)$ for all
$t \in \R$. Show $\mu = \nu$. (This makes the
\emph{distribution function} $F(t) = \mu(\intoc{-\infty}t)$ a
complete invariant --- the foundation of
\cref{ch:b3:probability}.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:measure:4}
(Borel--Cantelli, measure version) Let $(A_n)$ be measurable
with $\sum_n\mu(A_n) < \infty$, and $\limsup A_n =
\bigcap_N\bigcup_{n\geq N}A_n$ (the points belonging to
infinitely many $A_n$). Show $\mu(\limsup A_n) = 0$.
Application: for almost every $x \in \intcc01$, only finitely
many $n$ satisfy $\abs{x - p/q_n} \leq 4^{-n}$ for the
$n$-th rational $p/q_n$ of an enumeration of
$\Q\cap\intcc01$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:measure:5}
(Fat Cantor set) Repeat the Cantor construction on $\intcc01$,
but at step $n$ remove from each of the $2^{n-1}$ intervals a
\emph{centered} open interval of length $4^{-n}$ only. Show
that the resulting $K = \bigcap K_n$ is compact, has empty
interior (no interval survives), and
\[
\lambda(K) = 1 - \sum_{n\geq1}2^{n-1}4^{-n} = \tfrac12 :
\]
a nowhere dense set of measure $\frac12$. Deduce a
\emph{meagre} subset of $\intcc01$ of full measure $1$, and an
open dense subset of measure $< \varepsilon$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:measure:6}
Let $\mu$ be a measure on $\mathcal B(\R)$, invariant under
translations, with $c = \mu(\intoc01) < \infty$. Show $\mu =
c\,\lambda$ on $\mathcal B(\R)$. \emph{(Compute $\mu$ on dyadic intervals by dividing
$\intoc01$ into $2^n$ translates, then invoke
\cref{thm:b3:measure:uniqueness}.)}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:measure:7}
(Approximation) Let $A \in \mathcal L$ with $\lambda(A) <
\infty$ and $\varepsilon > 0$. Show there is a \emph{finite}
union of intervals $B$ with $\lambda(A\,\triangle\,B) <
\varepsilon$ ($\triangle$ = symmetric difference).
\emph{(Regularity: squeeze $K \subseteq A \subseteq U$ and use
the structure of the open $U$ as a countable union of
intervals, plus compactness of $K$.)}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b3:measure:8}
(Steinhaus) Let $A \in \mathcal L$ with $\lambda(A) > 0$. Show
that $A - A = \{x - y : x, y \in A\}$ contains an interval
around $0$.
\emph{(Reduce to $\lambda(A) < \infty$; by
\cref{exo:b3:measure:7}-style regularity, find an interval $I$
with $\lambda(A \cap I) > \frac34\ell(I)$; then for $\abs t <
\frac12\ell(I)$, the sets $A\cap I$ and $(A\cap I) + t$ both
sit in an interval of length $\frac32\ell(I)$ and have total
measure $> \frac32\ell(I)$: they must intersect.)}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b3:measure:9}
Show that every $A \in \mathcal L$ with $\lambda(A) > 0$
contains a non-measurable subset. \emph{(Intersect $A$ with the
translates $V + q$ of Vitali's set: if all $A \cap (V+q)$ were
measurable, each would be null by the argument of
\cref{thm:b3:measure:vitali} --- Steinhaus
(\cref{exo:b3:measure:8}) helps: a measurable set of positive
measure inside $V + q$ would give $(V+q) - (V+q) \supseteq$
an interval, contradicting that this difference set meets $\Q$
only at $0$; conclude with subadditivity.)}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:measure:10}
Show that $A \subseteq \R$ with $\lambda^*(A) < \infty$ is
Lebesgue-measurable iff for every $\varepsilon > 0$ there is an
open $U \supseteq A$ with $\lambda^*(U \setminus A) <
\varepsilon$, iff there is a $G_\delta$ set $G \supseteq A$
with $\lambda^*(G\setminus A) = 0$. (So Lebesgue sets are
Borel sets modulo null sets.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:measure:11}
(Continuity along monotone limits, and its sharpness)
(a) Show that for measurable sets, $\mu(\liminf A_n) \leq
\liminf\mu(A_n)$ (Fatou for sets), and that if
$\mu\bigl(\bigcup A_n\bigr) < \infty$, also
$\limsup\mu(A_n) \leq \mu(\limsup A_n)$.
(b) Exhibit, for Lebesgue measure on $\R$, a sequence with
$\mu(A_n) = 1$ for all $n$ yet $\mu(\limsup A_n) = 0$: the
finiteness hypothesis in the second inequality is not
decorative.
(c) Deduce: if $\sum\mu(A_n) < \infty$ then
$\mu(\limsup A_n) = 0$ (Borel--Cantelli again), and if the
$A_n$ increase or decrease (with $\mu(A_1) < \infty$ in the
decreasing case), $\mu(\lim A_n) = \lim\mu(A_n)$.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b3:measure:12}
(Egorov's theorem) Let $\mu(X) < \infty$ and $f_n \to f$
pointwise, all measurable (real-valued). For $k, N \geq 1$
set
\[
E_{k,N} = \bigcap_{n \geq N}\Bigl\{x : \abs{f_n(x) - f(x)}
\leq \tfrac1k\Bigr\} .
\]
(a) Show that for fixed $k$, $E_{k,N} \nearrow X$ as $N \to
\infty$, and deduce $N_k$ with $\mu(X \setminus E_{k,N_k})
\leq \varepsilon2^{-k}$.
(b) Conclude \emph{Egorov's theorem}: for every $\varepsilon
> 0$ there is a measurable $A$ with $\mu(X\setminus A) \leq
\varepsilon$ such that $f_n \to f$ \emph{uniformly on $A$}
--- pointwise convergence is uniform convergence off an
arbitrarily small set.
(c) Show the theorem fails on $(\R, \lambda)$: the moving
bumps $f_n = \mathbf 1_{\intcc n{n+1}}$ converge pointwise
to $0$ but uniformly on no complement of a finite-measure
set. Where did (a) use $\mu(X) < \infty$?
\end{exercise}

\begin{problem}[{Weekend problem --- the devil's staircase,
and $\mathcal B(\R) \subsetneq \mathcal L$}]
\label{pb:b3:measure:1}
We construct the \emph{Cantor--Vitali function} (devil's
staircase), use it to transport measure pathologically, and
conclude with a theorem that no soft argument gives: there
exist Lebesgue-measurable sets that are not Borel. Notation:
$C$ is the Cantor set, $C_n$ its $n$-th stage ($2^n$ intervals
of length $3^{-n}$), and every $x \in C$ has ternary digits
$x = \sum 2b_n3^{-n}$, $b_n \in \{0,1\}$
(\cref{exo:b3:topology:10}).

\medskip
\noindent\textbf{Part I --- The staircase.} Define $c_0(x) = x$
and $c_{n+1}$ from $c_n$ by
\[
c_{n+1}(x) = \begin{cases}
\tfrac12\,c_n(3x) & 0 \leq x \leq \tfrac13,\\[2pt]
\tfrac12 & \tfrac13 \leq x \leq \tfrac23,\\[2pt]
\tfrac12 + \tfrac12\,c_n(3x - 2) & \tfrac23 \leq x \leq 1.
\end{cases}
\]
\begin{enumerate}
  \item Show that each $c_n$ is continuous, nondecreasing, with
        $c_n(0) = 0$, $c_n(1) = 1$, and that
        $\norm{c_{n+1} - c_n}_\infty \leq
        \tfrac12\norm{c_n - c_{n-1}}_\infty$.
  \item Deduce that $(c_n)$ converges uniformly to a continuous
        nondecreasing $c$ with $c(0) = 0$, $c(1) = 1$
        (\emph{the} Cantor--Vitali function), which satisfies
        the same self-similar relations as the $c_{n+1}$ above.
  \item Show that $c$ is constant on each connected component
        of $\intcc01\setminus C$, and that for $x = \sum_n
        2b_n3^{-n} \in C$, $c(x) = \sum_n b_n2^{-n}$: the
        staircase reads Cantor digits in binary
        (\cref{pb:b3:topology:1}'s function $g$, made monotone
        and global).
  \item Deduce that $c$ is differentiable, with $c' = 0$, at
        every point of $\intcc01\setminus C$: $c' = 0$
        \emph{$\lambda$-almost everywhere}
        (\cref{ex:b3:measure:cantor}). Conclude that the
        fundamental theorem of calculus fails for $c$:
        \[
        c(1) - c(0) = 1 \neq 0 = \int_0^1 c'(t)\,\dd t
        \]
        (the integral being over the full-measure set where
        $c' = 0$; anticipating \cref{ch:b3:lebesgue}, null sets
        do not affect integrals). Which hypothesis of the
        $\mathcal C^1$ fundamental theorem is violated?
  \item Show that $c(C) = \intcc01$: the null set $C$ is mapped
        \emph{onto} a set of full measure.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The crooked homeomorphism.} Let
$h(x) = \frac{c(x) + x}{2}$.
\begin{enumerate}[resume]
  \item Show that $h \colon \intcc01 \to \intcc01$ is a
        homeomorphism (strictly increasing, continuous,
        surjective).
  \item Show that $\lambda\bigl(h(\intcc01\setminus C)\bigr) =
        \tfrac12$: on each gap of length $\ell$, $h$ acts as an
        affine map of slope $\tfrac12$, and the gaps have total
        length $1$.
  \item Deduce $\lambda\bigl(h(C)\bigr) = \tfrac12$: the
        homeomorphic image of a null set can have positive
        measure. (Where does this contradict naive intuition
        about ``size''?)
\end{enumerate}

\medskip
\noindent\textbf{Part III --- A measurable set that is not
Borel.}
\begin{enumerate}[resume]
  \item By \cref{exo:b3:measure:9}, choose a non-measurable $W
        \subseteq h(C)$. Show that $Z = h^{-1}(W) \subseteq C$
        is Lebesgue-measurable. \emph{(It is a subset of a null
        set; completeness, \cref{thm:b3:measure:caratheodory}.)}
  \item Show that the preimage of a Borel set under a
        continuous map is Borel. \emph{(Good sets principle:
        $\{B : h^{-1}(B) \in \mathcal B\}$ is a
        $\sigma$-algebra containing the open sets --- mind the
        direction of the map.)}
  \item Conclude that $Z$ is \emph{not} Borel: if it were,
        $W = (h^{-1})^{-1}(Z)$ would be Borel (apply question
        10 to the continuous $h^{-1}$), hence measurable ---
        contradiction. Therefore
        \[
        \boxed{\ \mathcal B(\R) \subsetneq \mathcal L\ }
        \]
        and completeness genuinely enlarges the Borel world.
  \item Exhibit a Lebesgue-measurable function $g$ and a
        continuous function $\varphi$ such that $g \circ
        \varphi$ is not Lebesgue-measurable: measurability,
        unlike continuity, does not compose. \emph{(Take $g =
        \mathbf 1_Z$ and $\varphi = h^{-1}$, anticipating the
        definition of measurable functions from
        \cref{ch:b3:lebesgue}: preimages of Borel sets are
        Lebesgue sets. Where must one be careful about which
        $\sigma$-algebra is used on the target?)}
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Epilogue.}
\begin{enumerate}[resume]
  \item Sort the following classes by strict inclusion and
        justify each strictness with an example from this
        chapter and its problem: countable sets; Borel null
        sets; Lebesgue null sets; Borel sets; Lebesgue sets;
        arbitrary sets.
\end{enumerate}

\medskip
\noindent\textbf{Part V --- The Cantor measure: mass on a null
set.} The staircase is the distribution function of a
remarkable measure, which we now build with this chapter's own
tools.
\begin{enumerate}[resume]
  \item (Lebesgue--Stieltjes, existence) Let $F\colon\R\to\R$
        be nondecreasing, continuous, bounded. On half-open
        intervals define $\rho\bigl(\intoc ab\bigr) = F(b) -
        F(a)$ and, for $A \subseteq \R$,
        \[
        \mu_F^*(A) = \inf\Bigl\{\sum_k\bigl(F(b_k) -
        F(a_k)\bigr) : A \subseteq
        \bigcup_k\intoc{a_k}{b_k}\Bigr\} .
        \]
        Show that $\mu_F^*$ is an outer measure and that
        $\mu_F^*\bigl(\intoc ab\bigr) = F(b) - F(a)$
        \emph{(imitate the compactness argument of
        \cref{thm:b3:measure:lebesgue}, enlarging each
        $\intoc{a_k}{b_k}$ to an open interval at
        $F$-cost $\leq \varepsilon2^{-k}$ --- where is
        continuity of $F$ used?)}.
  \item Show that every Borel set is
        $\mu_F^*$-measurable in the Carath\'eodory sense
        \emph{(as in the Lebesgue case, it suffices to test
        half-lines; follow the proof of
        \cref{thm:b3:measure:caratheodory}'s application)}, so
        that $\mu_F = \mu_F^*$ restricted to $\mathcal B(\R)$
        is a measure with $\mu_F(\intoc ab) = F(b) - F(a)$:
        the \emph{Lebesgue--Stieltjes measure} of $F$.
  \item Apply this to the staircase ($F = c$ extended by $0$
        on $\R_-$ and $1$ on $\intco1\infty$): the
        \emph{Cantor measure} $\mu$. Show $\mu(\R) = 1$, that
        every gap of the Cantor set is $\mu$-null ($c$ is
        constant there), and conclude
        \[
        \mu(C) = 1, \qquad \lambda(C) = 0 :
        \]
        $\mu$ and $\lambda$ live on disjoint carriers ($C$ and
        its complement). Two measures in this position are
        called \emph{mutually singular}, written $\mu \perp
        \lambda$.
  \item Show that $\mu$ has no atoms: $\mu(\{x\}) = 0$ for
        every $x$ \emph{(continuity of $c$)}. An atomless
        probability measure carried by a Lebesgue-null compact:
        compare with the only measures seen so far.
  \item (Coin tossing in disguise) For a word $(\varepsilon_1,
        \dots, \varepsilon_m) \in \{0,1\}^m$, let
        $C_{\varepsilon}$ be the set of $x \in C$ whose ternary
        digits satisfy $b_i(x) = \varepsilon_i$ for $i \leq m$
        (one of the $2^m$ Cantor pieces of depth $m$). Show
        $\mu(C_\varepsilon) = 2^{-m}$ \emph{(the staircase
        climbs $2^{-m}$ across that piece: use Part I,
        question 3)}. The Cantor measure is the law of an
        infinite sequence of fair coin flips read in ternary
        --- \cref{ch:b3:probability} will make this exact.
  \item Prove the self-similarity: for every Borel $A$,
        \[
        \mu(A) = \tfrac12\,\mu(3A) +
        \tfrac12\,\mu(3A - 2) ,
        \]
        where $3A - 2 = \{3x - 2 : x \in A\}$ \emph{(check it
        on the generating intervals $\intoc ab$ via the
        self-similar relations of $c$, then invoke uniqueness,
        \cref{thm:b3:measure:uniqueness})}.
  \item Show that the reflection $s(x) = 1 - x$ preserves
        $\mu$: $\mu(s(A)) = \mu(A)$ \emph{(via $c(1 - x) = 1 -
        c(x)$, which follows from the symmetry of the
        construction --- prove it)}.
  \item Compute the first two moments of $\mu$, i.e.\ of a
        random point $X$ with law $\mu$ (the integrals may be
        handled as limits of sums over the depth-$m$ pieces,
        anticipating \cref{ch:b3:lebesgue}): symmetry gives
        $\int x\,\dd\mu = \frac12$, and self-similarity gives
        \[
        \int x^2\,\dd\mu = \frac38,
        \qquad\text{hence}\qquad
        \operatorname{Var}(X) = \frac18 .
        \]
        Compare with the uniform law on $\intcc01$ (variance
        $\frac1{12}$): the Cantor mass, pushed to the edges,
        spreads \emph{more}.
  \item Show that the (topological) \emph{support} of $\mu$
        --- the smallest closed set of full measure --- is
        exactly $C$.
  \item (Synthesis) The staircase $c$ is continuous and
        nondecreasing yet fails the fundamental theorem of
        calculus (Part I); the measure $\mu_c$ is a
        probability, atomless, singular with respect to
        $\lambda$. Explain in a short paragraph how these are
        two faces of one phenomenon, and state the general
        moral: nondecreasing functions correspond to measures
        ($F \leftrightarrow \mu_F$), differentiability a.e.\
        corresponds to the ``absolutely continuous part'', and
        $c$ is the standard witness that a continuous $F$ can
        carry \emph{no} absolutely continuous part at all.
  \item (The sharp modulus of continuity) Let $s =
        \frac{\ln 2}{\ln 3}$. Show that $c$ is H\"older
        continuous of exponent $s$:
        \[
        \abs{c(x) - c(y)} \leq 4\,\abs{x - y}^{s}
        \qquad (x, y \in \intcc01),
        \]
        and that no exponent $t > s$ can work, even locally.
        Deduce the measure-theoretic form: for every $x$ and
        every $r \in \intoc01$,
        \[
        \mu\bigl(\intcc{x - r}{x + r}\bigr) \leq 8\,r^{s} .
        \]
        \emph{(Compare a depth-$m$ triadic grid with the scale
        of $\abs{x - y}$; question 18 gives the climb across
        each piece. The exponent $s$ is the Hausdorff dimension
        of $C$, as later courses will say.)}
  \item (Self-similarity characterizes $\mu$) Prove the
        converse of question 19: if $\nu$ is a probability
        measure on $\mathcal B(\R)$ carried by $\intcc01$ and
        satisfying
        \[
        \nu(A) = \tfrac12\,\nu(3A) + \tfrac12\,\nu(3A - 2)
        \qquad (A \in \mathcal B(\R)),
        \]
        then $\nu = \mu$. \emph{(Iterate the relation $m$
        times to spread $\nu$ over the $2^m$ depth-$m$ Cantor
        pieces, estimate $\nu(\intoc ab)$ against the count of
        pieces inside $\intoc ab$, and let $m \to \infty$;
        finish with \cref{thm:b3:measure:uniqueness}.)}
\end{enumerate}
\end{problem}
