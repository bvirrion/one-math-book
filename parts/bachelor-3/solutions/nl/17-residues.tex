\section*{Hoofdstuk \ref{ch:b3:residues} --- Laurentreeksen en
de residuenstelling}

\begin{solution}{exo:b3:residues:1}
$\frac{\sin z}z = 1 - \frac{z^2}6 + \cdots$: ophefbaar,
\omterm{def:b3:residues:singularities}{residu} $0$. $\frac{\eu^z - 1}{z^2} = \frac1z + \frac12 +
\frac z6 + \cdots$: eenvoudige pool, \omterm{def:b3:residues:singularities}{residu} $1$.
$\frac1{z(z-1)^2}$: eenvoudige pool in $0$ met \omterm{def:b3:residues:singularities}{residu}
$\frac1{(0-1)^2} = 1$; dubbele pool in $1$ met \omterm{def:b3:residues:singularities}{residu}
$\frac{\dd}{\dd z}\bigl(\frac1z\bigr)\big|_{z=1} = -1$.
$\frac{\cos z}{z^3} = \frac1{z^3} - \frac1{2z} + \cdots$: pool
van orde $3$, \omterm{def:b3:residues:singularities}{residu} $-\frac12$. $\eu^{1/z} = \sum_{n\geq0}
\frac{z^{-n}}{n!}$: essentieel, \omterm{def:b3:residues:singularities}{residu} $1$. $\frac1{\sin z}$:
eenvoudige polen in $n\pi$; \omterm{def:b3:residues:singularities}{residu} $\frac1{\cos 0} = 1$ in $0$,
$\frac1{\cos\pi} = -1$ in $\pi$
(\cref{met:b3:residues:compute}, $g/h'$).
\end{solution}

\begin{solution}{exo:b3:residues:2}
Met $z = \eu^{\iu t}$, $\cos t = \frac{z + z^{-1}}2$, $\dd t
= \frac{\dd z}{\iu z}$:
\[
\int_0^{2\pi}\frac{\dd t}{a + \cos t}
= \oint_{\abs z = 1}\frac{2\,\dd z}{\iu\,(z^2 + 2az + 1)} .
\]
De wortels $z_\pm = -a \pm \sqrt{a^2 - 1}$ voldoen aan
$z_+z_- = 1$ met $\abs{z_+} < 1 < \abs{z_-}$; het \omterm{def:b3:residues:singularities}{residu} in
$z_+$ is $\frac{1}{z_+ - z_-} = \frac1{2\sqrt{a^2-1}}$, dus
de integraal is $\frac2\iu\cdot2\iu\pi\cdot\frac1{2\sqrt{a^2-1}}
= \frac{2\pi}{\sqrt{a^2-1}}$. Als $a \to 1^+$ explodeert zij
(de integrand piekt in $t = \pi$); als $a \to \infty$
gedraagt zij zich als $\frac{2\pi}a$, passend bij
$\int\frac{\dd t}a$.
\end{solution}

\begin{solution}{exo:b3:residues:3}
Eerste integraal: bovenpolen van $\frac{z^2}{1+z^6}$ in $p =
\eu^{\iu\pi/6}, \iu, \eu^{5\iu\pi/6}$; in elk,
$\operatorname{Res} = \frac{p^2}{6p^5} = \frac{p^3}{6p^6} =
-\frac{p^3}6$, en $p^3$ neemt de waarden $\iu, -\iu, \iu$:
som van \omterm{def:b3:residues:singularities}{residuen} $-\frac{\iu}{6}$. De boog is
$O(R^{-4})\cdot O(R) \to 0$:
\[
\int_\R\frac{x^2\,\dd x}{1 + x^6} =
2\iu\pi\Bigl(-\frac \iu6\Bigr) = \frac\pi3 .
\]
Tweede: dubbele pool in $\iu$ van $\frac1{(1+z^2)^2} =
\frac1{(z-\iu)^2(z+\iu)^2}$:
\[
\operatorname{Res} = \frac{\dd}{\dd z}\,(z +
\iu)^{-2}\Big|_{z=\iu} = \frac{-2}{(2\iu)^3} =
\frac{-2}{-8\iu} = -\frac\iu4,
\qquad
\int_\R\frac{\dd x}{(1+x^2)^2} =
2\iu\pi\cdot\Bigl(-\frac\iu4\Bigr) = \frac\pi2 ,
\]
consistent met \cref{exo:b3:fouriertransform:3}(b).
\end{solution}

\begin{solution}{exo:b3:residues:4}
Sluit $\frac{\eu^{\iu tz}}{z^2 + a^2}$ in het bovenhalfvlak
($t \geq 0$): daar $\abs{\eu^{\iu tz}} =
\eu^{-t\operatorname{Im}z} \leq 1$, dus de boog draagt
$O(R^{-2})\cdot O(R) \to 0$ bij. De enige omsloten pool
$\iu a$ is \omterm{def:b3:groups:simple}{eenvoudig} met \omterm{def:b3:residues:singularities}{residu} $\frac{\eu^{-at}}{2\iu a}$:
\[
\int_\R\frac{\eu^{\iu tx}}{x^2 + a^2}\dd x =
\frac{\pi}{a}\,\eu^{-at},
\qquad\text{dus}\qquad
\int_\R\frac{\cos(tx)}{x^2+a^2}\dd x = \frac\pi
a\,\eu^{-a\abs t}
\]
(reëel deel; even in $t$). Dit is de inversietegenhanger
van $\widehat{\eu^{-a\abs x}} = \frac{2a}{a^2+\xi^2}$
(\cref{exo:b3:fouriertransform:1}): de twee berekeningen
bevestigen elkaar via
\cref{thm:b3:fouriertransform:inversion}.
\end{solution}

\begin{solution}{exo:b3:residues:5}
Partiële breuken: $f = \frac1{z-2} - \frac1{z-1}$.
Op $\abs z < 1$ (Taylor):
$f = \sum_{n\geq0}\bigl(1 - 2^{-n-1}\bigr)z^n$.
Op $1 < \abs z < 2$: $\frac1{z-2} =
-\sum_{n\geq0}\frac{z^n}{2^{n+1}}$ en $-\frac1{z-1} =
-\sum_{n\geq1}z^{-n}$: een echte tweezijdige reeks.
Op $\abs z > 2$: $f = \sum_{n\geq1}\bigl(2^{n-1} -
1\bigr)z^{-n}$. De drie verschillen omdat
Laurentontwikkelingen vastzitten aan \emph{annuli}, niet
punten: elke regio heeft haar eigen meetkundige
ontwikkelingen. De $z^{-1}$-coëfficiënten ($-1$ en $0$
respectievelijk) zijn integralen over cirkels die \emph{de
singulariteiten binnenin} omcirkelen, geen \omterm{def:b3:residues:singularities}{residuen} in $0$
($f$ is \omterm{def:b3:holomorphic:holo}{holomorf} in $0$): voor $1 < \abs z < 2$ is de
coëfficiënt $-1$ $\operatorname{Res}(f, 1)$; voor $\abs z >
2$ is de coëfficiënt $0$ $\operatorname{Res}(f,1) +
\operatorname{Res}(f,2) = -1 + 1$. De \omterm{def:b3:residues:singularities}{residuen} van $f$: $-1$
in $1$ en $+1$ in $2$.
\end{solution}

\begin{solution}{exo:b3:residues:6}
(a) De \omterm{thm:b3:residues:laurent}{Laurentreeks} $\sum_nz^{-n}/n!$ heeft oneindig veel
negatieve termen: essentieel. Oplossen $\eu^{1/z} = w$ ($w
\neq 0$): $\frac1z = \log\abs w + \iu\arg w + 2\iu\pi k$,
dus
\[
z_k = \frac1{\log\abs w + \iu\arg w + 2\iu\pi k}
\xrightarrow[k\to\infty]{} 0 :
\]
elke niet-nul waarde wordt oneindig vaak bereikt bij $0$ ---
sterker dan \omterm{ex:b3:lebesgue:gamma}{dichtheid}.
(b) Langs $z = 1/x$, $x \to +\infty$: $\eu^x \to \infty$;
langs $z = \iu/y$: modulus $1$. Een pool eist $\abs{f(z)}
\to \infty$ \emph{langs elke nadering}: hier bestaat de
limiet eenvoudigweg niet, zelfs niet in $[0, +\infty]$.
\end{solution}

\begin{solution}{exo:b3:residues:7}
(a) Op $\abs z = 1$: $\abs{z^7 + z - 1} \leq 3 < 4 =
\abs{-4z^3}$. Rouché met $f = -4z^3$, $g = z^7 + z - 1$:
drie nullen in de schijf.
(b) Op $\abs z = 1$: $\abs{z^4 + 1} \leq 2 < 5 = \abs{5z}$:
één nul in $\abs z < 1$. Op $\abs z = 2$: $\abs{5z + 1} \leq
11 < 16 = \abs{z^4}$: vier nullen in $\abs z < 2$. Dus drie
nullen in de annulus.
(c) Voor $P = z^n + a_{n-1}z^{n-1} + \dots$: op $\abs z = R >
1 + \sum\abs{a_k}$, $\abs{P - z^n} \leq
\bigl(\sum\abs{a_k}\bigr)R^{n-1} < R^n = \abs{z^n}$: $P$
heeft exact $n$ nullen in $D(0, R)$ --- d'Alembert--Gauss
met multipliciteit, door puur tellen.
\end{solution}

\begin{solution}{exo:b3:residues:8}
(a) Stel $f(a) = 0$, $f \not\equiv 0$: kies $r$ met $f$
nultvrij op de cirkel $C = \partial D(a, r)$ (\omterm{thm:b3:holomorphic:identity}{geïsoleerde
nullen}) en $m = \min_C\abs f > 0$. Door
\cref{thm:b3:holomorphic:weierstrassconv}, $f_n \to f$ en
$f_n' \to f'$ uniform op $C$; voor grote $n$, $\abs{f_n}
\geq m/2$ op $C$, dus
\[
\frac1{2\iu\pi}\int_C\frac{f_n'}{f_n}
\longrightarrow \frac1{2\iu\pi}\int_C\frac{f'}{f} \geq 1
\]
(de limiet telt de nul $a$; convergentie omdat tellers
uniform convergeren en noemers uniform van onderen
begrensd zijn). De linkerkant is een geheel getal dat nullen
van $f_n$ in de schijf telt: zij moet uiteindelijk $\geq 1$
zijn --- in tegenspraak met nultvrijheid. Dus $f$ is
nultvrij.

(b) Fixeer $w \in \Omega$ en pas (a) toe op de samenhangende
\omterm{def:b3:topology:topology}{open verzameling} $\Omega\setminus\{w\}$ (een punt verwijderen
uit een open samenhangende deelverzameling van $\C$ bewaart
samenhang) op $g_n(z) = f_n(z) - f_n(w)$, nultvrij daar door
injectiviteit, convergerend naar $g = f - f(w)$. Als $f$
niet-constant is, $g \not\equiv 0$ op $\Omega\setminus\{w\}$,
dus $g$ is nultvrij daar: $f(z) \neq f(w)$ voor alle $z \neq
w$. Omdat $w$ willekeurig was, is $f$ injectief.
\end{solution}

\begin{solution}{exo:b3:residues:9}
De sectorrand bestaat uit $[0, R]$, de boog $A_R$, en de
straal $\eu^{2\iu\pi/n}[0, R]$ omgekeerd. Binnenin ligt de
enige pool $p = \eu^{\iu\pi/n}$ van $\frac1{1+z^n}$, met
\omterm{def:b3:residues:singularities}{residu} $\frac1{np^{n-1}} = \frac{p}{np^n} = -\frac pn$. Op
de terugkeerstraal geeft $z = \eu^{2\iu\pi/n}x$ $z^n = x^n$
en $\dd z = \eu^{2\iu\pi/n}\dd x$; de boog is
$O(R^{-n})\cdot O(R) \to 0$. Dus
\[
\bigl(1 - \eu^{2\iu\pi/n}\bigr)
\int_0^\infty\frac{\dd x}{1 + x^n}
= 2\iu\pi\Bigl(-\frac{\eu^{\iu\pi/n}}n\Bigr),
\quad\text{dus}\quad
\int_0^\infty\frac{\dd x}{1+x^n}
= \frac{2\iu\pi}{n\,\bigl(\eu^{\iu\pi/n} -
\eu^{-\iu\pi/n}\bigr)} = \frac{\pi}{n\sin(\pi/n)} .
\]
$n = 2$: $\frac\pi{2\sin(\pi/2)} = \frac\pi2 =
[\arctan]_0^\infty$. Als $n \to \infty$: de waarde neigt naar
$1$, en inderdaad neigt de integrand naar $\mathbf
1_{\intco01}$ (met DCT-dominatie $\min(1, x^{-2})$ voor $n
\geq 2$).
\end{solution}

\begin{solution}{exo:b3:residues:10}
Op $\abs z = R$ groot, $\abs f \leq C R^{-2}$: $\abs{\oint_{C_R}
f} \leq 2\pi R\cdot CR^{-2} \to 0$. Maar voor $R$ voorbij alle
polen geeft de residuenstelling $\oint_{C_R}f =
2\iu\pi\sum_{\text{alle }p}\operatorname{Res}(f, p)$: de
totaalsom verdwijnt. Voor $f = \frac1{z(z-1)(z-2)}$:
\omterm{def:b3:residues:singularities}{residuen} $\frac1{(-1)(-2)} = \frac12$ in $0$,
$\frac1{1\cdot(-1)} = -1$ in $1$, $\frac1{2\cdot1} =
\frac12$ in $2$ --- sommerend tot $0$ zoals voorspeld, en
\[
\frac1{z(z-1)(z-2)} = \frac{1/2}{z} - \frac1{z - 1} +
\frac{1/2}{z-2} :
\]
de \omterm{def:b3:residues:singularities}{residuen} \emph{zijn} de partiële-breukcoëfficiënten, en
de nulsom-identiteit levert een gratis
consistentiecontrole (of bepaalt de laatste coëfficiënt uit
de andere).
\end{solution}

\begin{solution}{exo:b3:residues:11}
Op het sleutelgat, met de gekozen bepaling: net boven de
snede, $\log z = \ln x$; net eronder, $\log z = \ln x +
2\iu\pi$. De vier stukken geven
\[
\Bigl(1 - \eu^{2\iu\pi(a-1)}\Bigr)\int_\varepsilon^R
\frac{x^{a-1}}{1+x}\dd x + \int_{C_R} + \int_{C_\varepsilon}
= 2\iu\pi\operatorname{Res}(f, -1) .
\]
Bogen: $\abs{f} \leq \frac{R^{a-1}}{R - 1}$ op $C_R$, lengte
$2\pi R$: bijdrage $O(R^{a-1}) \to 0$ ($a < 1$);
$\abs f \leq \frac{\varepsilon^{a-1}}{1 - \varepsilon}$ op
$C_\varepsilon$, lengte $2\pi\varepsilon$: $O(\varepsilon^a)
\to 0$ ($a > 0$). \omterm{def:b3:residues:singularities}{Residu}: in $z = -1 = \eu^{\iu\pi}$,
$\operatorname{Res} = \eu^{(a-1)\iu\pi}$. Dus
\[
I(a) = \frac{2\iu\pi\,\eu^{\iu\pi(a-1)}}{1 -
\eu^{2\iu\pi(a-1)}}
= \frac{2\iu\pi}{\eu^{-\iu\pi(a-1)} - \eu^{\iu\pi(a-1)}}
= \frac{\pi}{-\sin(\pi(a-1))} = \frac{\pi}{\sin\pi a} .
\]
Gamma-reflectie: $B(a, 1-a) =
\int_0^1t^{a-1}(1-t)^{-a}\dd t$; de substitutie $t =
\frac x{1+x}$, $1 - t = \frac1{1+x}$, $\dd t =
\frac{\dd x}{(1+x)^2}$ maakt haar tot $\int_0^\infty
\frac{x^{a-1}}{1+x}\dd x = I(a)$, en Eulers formule
$B(a, 1-a) = \Gamma(a)\Gamma(1-a)/\Gamma(1) $
(\cref{pb:b3:lebesgue:1}) geeft
$\Gamma(a)\Gamma(1-a) = \frac\pi{\sin\pi a}$ --- in
het bijzonder $\Gamma(\tfrac12) = \sqrt\pi$ nogmaals.

\end{solution}

\begin{solution}{exo:b3:residues:12}
(a) Op $\abs z = 1$: $\abs{z^4} = 1 < 7 \leq \abs{8z + 1}$
($\abs{8z} - 1 = 7$): $P$ heeft evenveel nullen in $\mathbb
D$ als $8z + 1$, namelijk \emph{één} (in $-\frac18$).

(b) Op $\abs z = 3$: $\abs{8z + 1} \leq 25 < 81 =
\abs{z^4}$: Rouché tegen $z^4$ geeft alle vier nullen in
$\abs z < 3$, dus $4 - 1 = 3$ nullen in de annulus $1 <
\abs z < 3$. Verscherping: een nul met $\abs z = r \geq 2.1$
zou voldoen aan $r^4 = \abs{8z + 1} \leq 8r + 1$, maar $r^4
- 8r - 1$ is toenemend voor $r \geq 2$ en is gelijk aan
$19.45 - 16.8 - 1 = 1.65 > 0$ in $r = 2.1$: onmogelijk. Dus
de drie buitenste nullen liggen in $1 < \abs z < 2.1$.
(Numeriek: een reële nul bij $\approx -1.95$ en een
geconjugeerd paar bij $\approx 1.04 \pm 1.73\iu$, van
modulus $2.02$ --- waarom een Rouché-poging bij straal
exact $2$ moet falen: de stelling eist strikte
dominantie, en de nullen zitten net erbuiten.)

(c) Geen nullen op $\iu\R$: $\operatorname{Re}P(\iu t) = t^4 +
1 \geq 1$. Nullen in het rechterhalfvlak: gebruik het
argumentprincipe op de rand van de halfschijf $\{\abs z \leq
R,\ \operatorname{Re}z \geq 0\}$. Op de grote boog, $\arg P
\approx \arg z^4$ draait met $4\cdot\pi = 2\pi\cdot2$ (de
boog overspant hoek $\pi$). Langs de imaginaire as van
$\iu R$ naar $-\iu R$: $P(\iu t) = (t^4 + 1) + 8\iu t$
blijft in het rechterhalfvlak ($\operatorname{Re} > 0$),
dus $\arg P$ varieert binnen $\intoo{-\pi/2}{\pi/2}$ en
keert terug met netto verandering $\to 0$ als $R \to
\infty$ (eindpunten beide $\approx \arg t^4 = 0$). Totale
winding: $\frac{4\pi + 0}{2\pi} = 2$: \emph{twee} nullen in
het rechterhalfvlak --- consistent met de numeriek: het
geconjugeerde paar $\approx 1.04 \pm 1.73\iu$ heeft
positief reëel deel, de reële nullen $\approx -0.125$ en
$\approx -1.96$ negatief.
\end{solution}

\begin{solution}{pb:b3:residues:1}
\textbf{1.} $\sin(\pi z)$ heeft eenvoudige nullen precies in
$\Z$ ($\sin\pi z = 0$ desda $z \in \Z$, en $(\sin\pi z)' =
\pi\cos\pi z \neq 0$ daar), en $\cos(\pi n) \neq 0$:
$\pi\cot(\pi z) = \pi\cos(\pi z)/\sin(\pi z)$ heeft
eenvoudige polen in $\Z$ met
\[
\operatorname{Res}(\pi\cot\pi z,\ n)
= \frac{\pi\cos(\pi n)}{\pi\cos(\pi n)} = 1
\]
($g/h'$-regel, \cref{met:b3:residues:compute}).

\textbf{2.} $\frac{\sin(\pi z)}{\pi z} = 1 - \frac{(\pi z)^2}6
+ \frac{(\pi z)^4}{120} - \cdots$ is \omterm{def:b3:holomorphic:holo}{holomorf} en niet-nul
bij $0$: haar reciproke is \omterm{def:b3:holomorphic:holo}{holomorf}
(\cref{def:b3:holomorphic:holo}: quotiënt), met reeks $1 +
\frac{(\pi z)^2}{6} + \frac{7(\pi z)^4}{360} + \cdots$
(identificeer: $(1 - u)^{-1}$-stijlcoëfficiënten uit $u =
\frac{(\pi z)^2}6 - \frac{(\pi z)^4}{120}$: de $z^4$-coëfficiënt
is $\frac1{36} - \frac1{120} = \frac{7}{360}$).
Vermenigvuldig met $\cos(\pi z) = 1 - \frac{(\pi z)^2}2 +
\frac{(\pi z)^4}{24} - \cdots$ en deel door $z$:
\[
\pi\cot(\pi z) = \frac1z\Bigl[1 + \pi^2z^2\Bigl(\frac16 -
\frac12\Bigr) + \pi^4z^4\Bigl(\frac7{360} - \frac1{12} +
\frac1{24}\Bigr)\Bigr] + \cdots
= \frac1z - \frac{\pi^2}3\,z - \frac{\pi^4}{45}\,z^3 -
\cdots
\]
($\frac7{360} - \frac{30}{360} + \frac{15}{360} =
-\frac8{360} = -\frac1{45}$).

\textbf{3.} Verticale zijden $z = \pm(N + \frac12) + \iu y$:
door $\pi$-periodiciteit van $\cot$, $\cot(\pi z) =
\cot(\pm\frac\pi2 + \iu\pi y) = -\tan(\iu\pi y) =
-\iu\tanh(\pi y)$, van modulus $\leq 1$. Horizontale zijden
$z = x \pm \iu(N + \frac12)$: uit $\abs{\cot(a + \iu b)}^2 =
\frac{\cos^2a + \sinh^2b}{\sin^2a + \sinh^2b} \leq \frac{1
+ \sinh^2b}{\sinh^2b} = \coth^2 b$,
\[
\abs{\cot(\pi z)} \leq \coth\bigl(\pi(N + \tfrac12)\bigr)
\leq \coth(\pi/2) < 1.1 .
\]
Beide grenzen zijn $\leq 2$.

\textbf{4.} Pas \cref{thm:b3:residues:residue} toe op $F(z) =
\pi\cot(\pi z)f(z)$ op $C_N$ ($N$ voorbij alle polen van
$f$):
\[
\frac1{2\iu\pi}\oint_{C_N}F = \sum_{n=-N}^{N}f(n) +
\sum_p\operatorname{Res}(F, p),
\]
de geheeltallige polen bijdragend $f(n)$ (vraag~1; $f$
\omterm{def:b3:holomorphic:holo}{holomorf} daar). Op $C_N$: $\abs F \leq 2\pi\cdot
C\abs z^{-2} \leq 2\pi C N^{-2}$, en de omtrek is $8(N +
\frac12)$: de integraal is $O(1/N) \to 0$. Laat $N \to
\infty$: de weergegeven sommatieformule.

\textbf{5.} De afname $\abs f = O(\abs z^{-2})$ doodde de
\omterm{def:b3:holomorphic:contour}{contourintegraal} \emph{en} maakte $\sum\abs{f(n)}$
convergent. Voor $f(z) = \frac1{z + \frac12}$: de
symmetrische sommen $\sum_{-N}^N\frac1{n + \frac12}$
telescoperen tot $0$ (de termen $n$ en $-n - 1$ heffen
elkaar op), en de rechterkant is
$-\operatorname{Res}\bigl(\frac{\pi\cot\pi z}{z + \frac12},
-\frac12\bigr) = -\pi\cot(-\frac\pi2) = 0$: consistent ---
maar $\sum\abs{f(n)}$ divergeert; de methode berekent
slechts de symmetrische (hoofdwaarde-)limiet.

\textbf{6.} $g(z) = \frac{\pi\cot(\pi z)}{z^2}$: polen in de
niet-nulle gehele getallen met \omterm{def:b3:residues:singularities}{residuen} $\frac1{n^2}$, en in
$0$ waar, door vraag~2,
\[
g(z) = \frac1{z^3} - \frac{\pi^2}{3z} - \frac{\pi^4}{45}z -
\cdots :
\qquad \operatorname{Res}(g, 0) = -\frac{\pi^2}3 .
\]
De \omterm{def:b3:holomorphic:contour}{contourintegraal} over $C_N$ neigt naar $0$ als in vraag~4
($\abs{g} = O(N^{-2})$ op $C_N$). Dus $0 =
\sum_{n\neq0}\frac1{n^2} - \frac{\pi^2}3$: $2\zeta(2) =
\frac{\pi^2}3$, $\zeta(2) = \frac{\pi^2}6$.

\textbf{7.} $g(z) = \frac{\pi\cot(\pi z)}{z^4} = \frac1{z^5}
- \frac{\pi^2}{3z^3} - \frac{\pi^4}{45z} - \cdots$: \omterm{def:b3:residues:singularities}{residu}
in $0$ gelijk aan $-\frac{\pi^4}{45}$, en $0 =
2\zeta(4) - \frac{\pi^4}{45}$: $\zeta(4) =
\frac{\pi^4}{90}$.

\textbf{8.} Met $g = \pi\cot(\pi z)/z^{2k}$: het \omterm{def:b3:residues:singularities}{residu} in
$0$ is de coëfficiënt $a_{2k-1}$ van $z^{2k-1}$ in de
ontwikkeling van $\pi\cot(\pi z)$, en de verdwijnende
contour geeft $2\zeta(2k) + a_{2k-1} = 0$: $\zeta(2k) =
-a_{2k-1}/2$, een rationaal veelvoud van $\pi^{2k}$ omdat
de cotangenscoëfficiënten dat zijn. Eén delingsstap meer
levert $a_5 = -\frac{2\pi^6}{945}$, vandaar $\zeta(6) =
\frac{\pi^6}{945}$. Voor $\zeta(3)$: de natuurlijke kern $g
= \pi\cot(\pi z)/z^3$ produceert $\sum_{n\neq0}\frac1{n^3}
= 0$ door onevenheid --- de methode bewijst $0 = 0$ en is
structureel blind voor oneven zetawaarden (geen gesloten
vorm voor $\zeta(3)$ is bekend; haar irrationaliteit,
Apéry 1978, vroeg geheel andere ideeën).

\textbf{9.} $F(z) = \frac{\pi\cot(\pi z)}{(z - w)(z + w)}$
heeft polen in de gehele getallen (\omterm{def:b3:residues:singularities}{residuen} $\frac1{n^2 -
w^2}$, let op het teken: $f(n) = \frac1{(n-w)(n+w)} =
\frac1{n^2 - w^2}$) en eenvoudige polen in $\pm w$ met
\omterm{def:b3:residues:singularities}{residuen} $\frac{\pi\cot(\pm\pi w)}{\pm2w} =
\frac{\pi\cot(\pi w)}{2w}$ elk ($\cot$ is oneven). Het
argument van vraag~4 ($\abs f = O(\abs z^{-2})$) geeft
\[
\sum_{n\in\Z}\frac1{n^2 - w^2} + \frac{\pi\cot(\pi w)}{w} =
0,
\qquad\text{d.w.z.}\qquad
\pi\cot(\pi w) = \frac1w + \sum_{n\geq1}\frac{2w}{w^2 -
n^2},
\]
(de $n = 0$-term is $-\frac1{w^2}$; hergroepeer $\pm n$).
Normale convergentie op compacta van $\C\setminus\Z$: voor
$\abs w \leq R$ en $n \geq 2R$, $\abs{\frac{2w}{w^2 - n^2}}
\leq \frac{2R}{n^2 - R^2} \leq \frac{8R}{3n^2}$.

\textbf{10.} Voor $\abs w \leq r < 1$: $\frac{2w}{w^2 - n^2}
= -\frac{2w}{n^2}\cdot\frac1{1 - w^2/n^2} =
-2\sum_{k\geq0}\frac{w^{2k+1}}{n^{2k+2}}$, met
$\abs{\text{termen}} \leq 2r^{2k+1}/n^{2k+2}$, sommeerbaar
over $(n, k)$: Fubini voor reeksen herrangschikt
\[
\pi\cot(\pi w) = \frac1w -
2\sum_{k\geq0}\zeta(2k+2)\,w^{2k+1} .
\]
Matchen met vraag~2: $-2\zeta(2) = -\frac{\pi^2}3$ en
$-2\zeta(4) = -\frac{\pi^4}{45}$ --- dezelfde waarden. Drie
wegen naar $\frac{\pi^2}6$: \omterm{thm:b3:hilbert:parseval}{Parseval} (Fourierreeksen), het
spoor van de Green-operator van de snaar, en de twee
ontwikkelingen van de cotangens; dat een som over
frequenties, een operatorspoor, en een \omterm{def:b3:holomorphic:contour}{contourintegraal}
overeenstemmen is geen toeval --- elk is een gezicht van
dezelfde spectrale identiteit.

\textbf{11.} Fixeer $R \geq 1$ en laat $n_0$ het kleinste
geheel $\geq 2R$ zijn. Voor $\abs z \leq R$ en $n \geq n_0$:
$\abs{z^2/n^2} \leq \frac14$, dus $1 - z^2/n^2 \in
\bar D(1, \frac14)$, waar de principale logaritme \omterm{def:b3:holomorphic:holo}{holomorf}
is, en
\[
\Bigl|\log\Bigl(1 - \frac{z^2}{n^2}\Bigr)\Bigr|
\leq \sum_{k\geq1}\frac1k\,\Bigl|\frac{z^2}{n^2}\Bigr|^k
\leq \frac{\abs{z^2/n^2}}{1 - \abs{z^2/n^2}}
\leq \frac{2R^2}{n^2} :
\]
de som $S(z) = \sum_{n\geq n_0}\log(1 - z^2/n^2)$ convergeert
normaal op $\bar D(0, R)$, met holomorfe deelsommen $S_N$ en
$\abs{S_N} \leq 2R^2\zeta(2)$ uniform. Omdat $\abs{\eu^a -
\eu^b} \leq \eu^{\max(\abs a,\abs b)}\abs{a - b}$
(middelwaardegrens op het segment), convergeren de
staartproducten $\prod_{n_0\leq n\leq N} = \eu^{S_N}$
uniform op $\bar D(0, R)$ naar de nultvrije $\eu^S$.
Vermenigvuldigen met het vaste polynoom
$z\prod_{n<n_0}(1 - z^2/n^2)$: $P_N \to P$ uniform op
$\bar D(0, R)$, en
\cref{thm:b3:holomorphic:weierstrassconv} maakt $P$
\omterm{def:b3:holomorphic:holo}{holomorf} daar; $R$ willekeurig, $P$ is geheel. Op
$\bar D(0, R)$ zijn de nullen van $P$ die van de
polynomiale prefactor --- de gehele getallen van modulus
$\leq R$, elk \omterm{def:b3:groups:simple}{eenvoudig} ($(1 - z/n)(1 + z/n)$ heeft
verschillende eenvoudige nullen, $\eu^S$ geen): de
nulverzameling van $P$ is $\Z$, alle nullen \omterm{def:b3:groups:simple}{eenvoudig}.

\textbf{12.} Logaritmische differentiatie van het eindige
product, weg van haar nullen:
\[
\frac{P_N'(z)}{P_N(z)} = \frac1z +
\sum_{n=1}^{N}\frac{-2z/n^2}{1 - z^2/n^2}
= \frac1z + \sum_{n=1}^{N}\frac{2z}{z^2 - n^2} .
\]
Op een compactum $K \subseteq \C\setminus\Z$: $P_N \to P$ en
$P_N' \to P'$ uniform
(\cref{thm:b3:holomorphic:weierstrassconv}), en $\min_K
\abs P > 0$ ($P$ verdwijnt slechts op $\Z$), dus uiteindelijk
$\abs{P_N} \geq \frac12\min_K\abs P$ en $P_N'/P_N \to
P'/P$ uniform op $K$. Het middellid convergeert naar
$\frac1z + \sum_{n\geq1}\frac{2z}{z^2-n^2} = \pi\cot(\pi z)$
door vraag~9: dus $P'/P = \pi\cot(\pi z)$ op
$\C\setminus\Z$.

\textbf{13.} $\sin(\pi z)$ en $P$ zijn geheel met dezelfde
nulverzameling $\Z$, alle nullen \omterm{def:b3:groups:simple}{eenvoudig} (vragen~1 en~11).
Bij $m \in \Z$ schrijf $\sin(\pi z) = (z - m)\,\sigma(z)$ en
$P(z) = (z - m)\,\psi(z)$ met $\sigma, \psi$ \omterm{def:b3:holomorphic:holo}{holomorf} en
niet-verdwijnend in $m$ (factoriseer de machtreeks): $Q =
\sin(\pi z)/P = \sigma/\psi$ zet \omterm{def:b3:holomorphic:holo}{holomorf} en nultvrij voort
over elk geheel getal, en is nultvrij op $\C\setminus\Z$ als
quotiënt van nultvrije functies. Daar,
\[
\frac{Q'}{Q} = \frac{(\sin\pi z)'}{\sin\pi z} -
\frac{P'}{P} = \pi\cot(\pi z) - \pi\cot(\pi z) = 0 ,
\]
dus de gehele functie $Q'$ verdwijnt op $\C\setminus\Z$,
dus overal door continuïteit: $Q$ is constant. Als $z \to
0$: $\sin(\pi z)/z \to \pi$ en $P(z)/z \to 1$, dus $Q =
\pi$:
\[
\sin(\pi z) = \pi z\prod_{n\geq1}
\Bigl(1 - \frac{z^2}{n^2}\Bigr) .
\]

\textbf{14.} In $z = \frac12$: $1 = \sin\frac\pi2 =
\frac\pi2\prod_{n\geq1}\bigl(1 - \frac1{4n^2}\bigr) =
\frac\pi2\prod\frac{4n^2-1}{4n^2}$, dus
\[
\frac\pi2 = \prod_{n\geq1}\frac{4n^2}{4n^2 - 1}
= \lim_{N\to\infty}\prod_{n=1}^N
\frac{(2n)(2n)}{(2n-1)(2n+1)}
= \lim_{N\to\infty}
\frac{2\cdot2\cdot4\cdot4\cdots(2N)(2N)}
{1\cdot3\cdot3\cdot5\cdots(2N-1)(2N+1)} :
\]
Wallis' product, een one-liner-gevolg van Eulers
factorisatie.

\textbf{15.} Voor $\abs z \leq r < 1$ ligt elke factor in
$D(1, r^2) \subseteq D(1, 1)$, dus $P(z)/z = \exp\bigl(
\sum_{n\geq1}\log(1 - z^2/n^2)\bigr)$: elk deelproduct is
de exponentiële van een deelsom, en beide kanten gaan tot de
limiet door continuïteit van $\exp$. De dubbele reeks
\[
\sum_{n\geq1}\log\Bigl(1 - \frac{z^2}{n^2}\Bigr)
= -\sum_{n\geq1}\sum_{k\geq1}\frac{z^{2k}}{k\,n^{2k}}
= -\sum_{k\geq1}\frac{\zeta(2k)}k\,z^{2k}
= -\zeta(2)\,z^2 + O(z^4)
\]
herrangschikt door Fubini voor reeksen: $\sum_{n,k}
\frac{r^{2k}}{kn^{2k}} \leq \sum_k\zeta(2k)r^{2k} \leq
\zeta(2)\frac{r^2}{1-r^2} < \infty$. Dus
\[
P(z) = z\,\exp\bigl(-\zeta(2)z^2 + O(z^4)\bigr)
= z - \zeta(2)\,z^3 + O(z^5) ,
\]
en vraag~13 vergelijkt dit met $\sin(\pi z) = \pi z -
\frac{\pi^3}6z^3 + O(z^5)$: $\pi\zeta(2) = \frac{\pi^3}6$,
d.w.z.\ $\zeta(2) = \frac{\pi^2}6$. Het additieve gezicht
(vraag~10) en het multiplicatieve gezicht berekenen hetzelfde
getal.

\textbf{16.} Op een compactum $K \subseteq \C\setminus\Z$
convergeren de deelsommen $S_N = \frac1z + \sum_{n\leq N}\bigl(
\frac1{z-n} + \frac1{z+n}\bigr)$ (vraag~9, termen
hergroepeerd als $\frac{2z}{z^2-n^2} = \frac1{z-n} +
\frac1{z+n}$) uniform naar $\pi\cot(\pi z)$, dus
\cref{thm:b3:holomorphic:weierstrassconv} geeft $S_N' \to
(\pi\cot\pi z)' = -\pi^2/\sin^2(\pi z)$ uniform op $K$.
Omdat $S_N' = -\sum_{\abs n\leq N}(z - n)^{-2}$:
\[
\frac{\pi^2}{\sin^2(\pi z)} =
\sum_{n\in\Z}\frac1{(z - n)^2} ,
\]
de convergentie normaal op compacta van $\C\setminus\Z$
(termen $O(n^{-2})$).

\textbf{17.} In $z = \frac12$ is de linkerkant $\pi^2$;
rechts, $(\frac12 - n)^2 = \frac{(2n-1)^2}4$ met $2n - 1$
lopend over alle oneven gehele getallen precies éénmaal als
$n$ over $\Z$ loopt:
\[
\pi^2 = \sum_{n\in\Z}\frac{4}{(2n-1)^2}
= 8\sum_{m\geq0}\frac1{(2m+1)^2},
\qquad
\sum_{m\geq0}\frac1{(2m+1)^2} = \frac{\pi^2}8 .
\]
Pariteitssplitsing: $\zeta(2) = \frac{\pi^2}8 +
\sum_{n\geq1}\frac1{(2n)^2} = \frac{\pi^2}8 +
\frac{\zeta(2)}4$, dus $\frac34\zeta(2) = \frac{\pi^2}8$ en
$\zeta(2) = \frac{\pi^2}6$ nogmaals.

\textbf{18.} Met $c = \cot\theta$ en $\cot(2\theta) =
\frac{c^2-1}{2c}$: $\cot\theta - 2\cot(2\theta) = c -
\frac{c^2-1}c = \frac1c = \tan\theta$. Dus $\pi\tan(\pi z)
= \pi\cot(\pi z) - 2\pi\cot(2\pi z)$, en vraag~9 in $z$
en in $2z$ geeft
\[
\pi\cot(\pi z) = \frac1z +
\sum_{n\geq1}\frac{2z}{z^2 - n^2},
\qquad
2\pi\cot(2\pi z) = \frac1z +
\sum_{n\geq1}\frac{8z}{4z^2 - n^2} .
\]
Beide reeksen convergeren absoluut in elk vast $z$ buiten de
polen, dus het verschil mag willekeurig hergroepeerd: in de
tweede reeks geven de even termen $n = 2m$ $\frac{8z}{4z^2 -
4m^2} = \frac{2z}{z^2 - m^2}$ en heffen de eerste reeks
geheel op, latend
\[
\pi\tan(\pi z) = -\sum_{m\geq0}\frac{8z}{4z^2 - (2m+1)^2}
= \sum_{m\geq0}\frac{8z}{(2m+1)^2 - 4z^2} ,
\]
normaal op compacta die $\frac12 + \Z$ vermijden (termen
$O(m^{-2})$).

\textbf{19.} Voor $\abs z \leq r < \frac12$:
\[
\frac{8z}{(2m+1)^2 - 4z^2} = \frac{8z}{(2m+1)^2}
\sum_{k\geq0}\Bigl(\frac{4z^2}{(2m+1)^2}\Bigr)^{k},
\]
met $\sum_{m,k}8r\,(4r^2)^k(2m+1)^{-2k-2} < \infty$ omdat
$4r^2 < 1$: Fubini herrangschikt de dubbele som tot
\[
\pi\tan(\pi z) =
\sum_{k\geq0}8\cdot4^k\,\lambda(2k+2)\,z^{2k+1} .
\]
Tegen $\pi\tan(\pi z) = \pi^2z + \frac{\pi^4}3z^3 +
O(z^5)$: de coëfficiënt van $z$ geeft $8\lambda(2) = \pi^2$
--- vraag~17 opnieuw --- en die van $z^3$ geeft
$32\lambda(4) = \frac{\pi^4}3$, d.w.z.\ $\lambda(4) =
\frac{\pi^4}{96}$. De even noemers verwijderend,
$\lambda(4) = \zeta(4) - 2^{-4}\zeta(4) =
\frac{15}{16}\zeta(4)$: $\zeta(4) = \frac{16}{15}\cdot
\frac{\pi^4}{96} = \frac{\pi^4}{90}$, matchend vraag~7.

\textbf{20.} $\cot\frac\theta2 - \cot\theta =
\frac{\cos\frac\theta2\,\sin\theta -
\cos\theta\,\sin\frac\theta2}{\sin\frac\theta2\,\sin\theta}
= \frac{\sin\frac\theta2}{\sin\frac\theta2\,\sin\theta} =
\frac1{\sin\theta}$, de teller zijnde $\sin(\theta -
\frac\theta2)$. Met $\theta = \pi z$ leest vraag~9 in
$\frac z2$ als $\pi\cot\frac{\pi z}2 = \frac2z +
\sum_{n\geq1}\frac{4z}{z^2 - 4n^2}$, dus
\[
\frac{\pi}{\sin(\pi z)}
= \pi\cot\frac{\pi z}2 - \pi\cot(\pi z)
= \frac1z + \sum_{n\geq1}\frac{4z}{z^2 - 4n^2}
- \sum_{n\geq1}\frac{2z}{z^2 - n^2} .
\]
Absolute convergentie staat de pariteitshergroepering toe:
even $n = 2m$ in de afgetrokken reeks draagt
$\frac{2z}{z^2-4m^2}$ bij, latend $+\frac{2z}{z^2-4m^2}$ uit
de eerste som, terwijl oneven $n$ overleeft met teken $-$:
\[
\frac{\pi}{\sin(\pi z)} = \frac1z +
\sum_{n\geq1}(-1)^n\,\frac{2z}{z^2 - n^2} .
\]
Tekens controleren: $\operatorname{Res}\bigl(\pi/\sin(\pi z),
n\bigr) = \pi/(\pi\cos\pi n) = (-1)^n$
(\cref{met:b3:residues:compute}), en $(-1)^n\frac{2z}{z^2 -
n^2} = \frac{(-1)^n}{z-n} + \frac{(-1)^n}{z+n}$ draagt
exact dat \omterm{def:b3:residues:singularities}{residu} in $\pm n$.

\textbf{21.} Voor $\abs z \leq r < 1$, elke term
ontwikkelend als in vraag~19 ($(-1)^n\frac{2z}{z^2-n^2} =
(-1)^{n-1}\frac{2z}{n^2}\sum_k\frac{z^{2k}}{n^{2k}}$) en
Fubini toepassend:
\[
\frac{\pi}{\sin(\pi z)} = \frac1z +
2\sum_{k\geq0}\eta(2k+2)\,z^{2k+1},
\qquad
\eta(s) = \sum_{n\geq1}\frac{(-1)^{n-1}}{n^s} .
\]
Taylor-kant: $\sin(\pi z) = \pi z(1 - \frac{(\pi z)^2}6 +
O(z^4))$ geeft $\frac\pi{\sin\pi z} = \frac1z +
\frac{\pi^2}6z + O(z^3)$: $2\eta(2) = \frac{\pi^2}6$, dus
$\eta(2) = \frac{\pi^2}{12}$. Consistentie: $\eta(2) =
\zeta(2) - 2\sum_n(2n)^{-2} = (1 - 2^{1-2})\zeta(2) =
\frac{\zeta(2)}2 = \frac{\pi^2}{12}$.

\textbf{22.} In $z = \frac12$ geeft vraag~20
\[
\pi = 2 + \sum_{n\geq1}(-1)^n\frac{1}{\frac14 - n^2}
= 2 + 4\sum_{n\geq1}\frac{(-1)^{n-1}}{(2n-1)(2n+1)} .
\]
Met $\frac1{(2n-1)(2n+1)} = \frac12\bigl(\frac1{2n-1} -
\frac1{2n+1}\bigr)$, en beide alternerende reeksen
convergent (Leibniz-criterium), splitst de som als
$\frac12\bigl[L - (1 - L)\bigr] = L - \frac12$, waar $L =
1 - \frac13 + \frac15 - \cdots$ en $\frac13 - \frac15 +
\frac17 - \cdots = 1 - L$. Dus $\pi = 2 + 4(L - \frac12) =
4L$:
\[
\frac\pi4 = 1 - \frac13 + \frac15 - \frac17 + \cdots
\]
De moraal: elke kern is \omterm{def:b3:residues:singularities}{meromorf} met polen op een
rekenkundige progressie en voorgeschreven \omterm{def:b3:residues:singularities}{residuen}. De
cotangens zet \omterm{def:b3:residues:singularities}{residu} $1$ in elk geheel getal en sommeert
$f(n)$; haar afgeleide kwadrateert de polen en prijst
$\lambda(2)$; de tangens verplaatst de polen naar $\frac12
+ \Z$ en prijst de oneven noemers; $\pi/\sin$ houdt de
geheeltallige polen maar alternatieert de \omterm{def:b3:residues:singularities}{residuen}
$(-1)^n$, vandaar de alternerende reeksen. De
eerste-orde-kernen zijn alle oneven functies: $n$ paren met
$-n$ verdubbelt de even-machtscoëfficiënten en vernietigt de
oneven, dus $\zeta(2k)$ stroomt mechanisch uit terwijl
$\zeta(3)$ nooit verschijnt. De blindheid is pariteit, geen
gebrek aan techniek.

\textbf{23.} Met $w = 2\iu\pi z$:
\[
\iu\pi z + \frac{2\iu\pi z}{\eu^{2\iu\pi z} - 1}
= \iu\pi z\,\frac{\eu^{2\iu\pi z} + 1}{\eu^{2\iu\pi z} - 1}
= \iu\pi z\,\frac{\eu^{\iu\pi z} +
\eu^{-\iu\pi z}}{\eu^{\iu\pi z} - \eu^{-\iu\pi z}}
= \pi z\,\frac{\cos\pi z}{\sin\pi z} = \pi z\cot(\pi z) .
\]
Dus, met de genererende functie op $w = 2\iu\pi z$ (en
$B_1 = -\frac12$ die de $\iu\pi z$-term opheft, oneven
$B$'s verdwijnend voorbij):
\[
\pi z\cot(\pi z) = \sum_{k\geq0}\frac{B_{2k}}{(2k)!}
(2\iu\pi z)^{2k}
= 1 + \sum_{k\geq1}(-1)^k\frac{(2\pi)^{2k}B_{2k}}{(2k)!}
z^{2k} .
\]
Anderzijds geeft de partiële-breukontwikkeling (Deel~IV)
$\pi z\cot(\pi z) = 1 - 2\sum_{k\geq1}\zeta(2k)z^{2k}$
(ontwikkel elke $\frac{2z^2}{z^2 - n^2} =
-2\sum_k\frac{z^{2k}}{n^{2k}}$ en sommeer over $n$,
normale convergentie rechtvaardigend de verwisseling voor
$\abs z < 1$). Coëfficiënten vergelijken: $-2\zeta(2k) =
(-1)^k\frac{(2\pi)^{2k}B_{2k}}{(2k)!}$, de gestelde formule.

\textbf{24.} $k = 1$: $\frac{(2\pi)^2}{2\cdot2}\cdot\frac16
= \frac{\pi^2}6$. $k = 2$: $-\frac{(2\pi)^4}{2\cdot24}
\cdot\bigl(-\frac1{30}\bigr) = \frac{16\pi^4}{48\cdot30} =
\frac{\pi^4}{90}$. $k = 3$: $\frac{(2\pi)^6}{2\cdot720}
\cdot\frac1{42} = \frac{64\pi^6}{1440\cdot42} =
\frac{\pi^6}{945}$.

\textbf{25.} Schrijf $C(z) = \pi z\cot(\pi z) = 1 -
2\sum_{k\geq1}\zeta(2k)z^{2k}$ (vraag~23). Directe
differentiatie van $C = \pi z\cot(\pi z)$, met $(\cot u)'
= -1 - \cot^2u$:
\[
zC'(z) = \pi z\cot(\pi z) - \pi^2z^2\bigl(1 +
\cot^2(\pi z)\bigr) = C - \pi^2z^2 - C^2 .
\]
Ontwikkel nu beide kanten in machten van $z^2$. Linkerkant:
$\sum_k(-4k)\,\zeta(2k)\,z^{2k}$. Rechterkant: kwadrateren
van de reeks,
\[
C - C^2 = 2\sum_{k\geq1}\zeta(2k)z^{2k}
- 4\sum_{k\geq2}\Bigl(\sum_{j=1}^{k-1}\zeta(2j)
\zeta(2k-2j)\Bigr)z^{2k},
\]
en de term $-\pi^2z^2 = -6\zeta(2)z^2$ past slechts $k =
1$ aan. Coëfficiënten van $z^{2k}$ vergelijken voor $k \geq
2$:
\[
-4k\,\zeta(2k) = 2\,\zeta(2k) -
4\sum_{j=1}^{k-1}\zeta(2j)\,\zeta(2k-2j),
\]
d.w.z.\ $\bigl(k + \frac12\bigr)\zeta(2k) =
\sum_{j=1}^{k-1}\zeta(2j)\zeta(2k-2j)$. (Bij $k = 1$ leest
de identiteit $-4\zeta(2) = 2\zeta(2) - 6\zeta(2)$: een
consistentiecontrole, geen nieuwe informatie.) Toepassingen:
$k = 2$: $\frac52\zeta(4) = \zeta(2)^2 = \frac{\pi^4}{36}$,
dus $\zeta(4) = \frac{\pi^4}{90}$; $k = 3$:
$\frac72\zeta(6) = 2\zeta(2)\zeta(4) = \frac{\pi^6}{270}$,
dus $\zeta(6) = \frac{2}{7}\cdot\frac{\pi^6}{270} =
\frac{\pi^6}{945}$. Eén transcendent zaad ($\zeta(2) =
\frac{\pi^2}6$), en pure algebra genereert elke even
zetawaarde.
\end{solution}
