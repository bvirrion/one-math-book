\section*{Hoofdstuk \ref{ch:b3:clt} --- Karakteristieke
functies en de centrale limietstelling}

\begin{solution}{exo:b3:clt:1}
Uniform op $\intcc{-1}1$: $\varphi(\xi) =
\frac12\int_{-1}^1\eu^{\iu\xi x}\dd x =
\frac{\sin\xi}{\xi}$ (gelijk aan $1$ in $\xi = 0$).
Exponentieel $\mathcal E(\lambda)$: $\varphi(\xi) =
\lambda\int_0^\infty\eu^{(\iu\xi - \lambda)x}\dd x =
\frac{\lambda}{\lambda - \iu\xi}$ (de primitieve verdwijnt
in $+\infty$ want $\operatorname{Re}(\iu\xi - \lambda) <
0$). Poisson $\mathcal P(\lambda)$: door de
overdrachtsstelling voor discrete wetten,
\[
\varphi(\xi) = \sum_{k\geq0}\eu^{\iu\xi
k}\,\eu^{-\lambda}\frac{\lambda^k}{k!}
= \eu^{-\lambda}\exp\bigl(\lambda\eu^{\iu\xi}\bigr)
= \exp\bigl(\lambda(\eu^{\iu\xi} - 1)\bigr).
\]
Binomiaal $\mathcal B(n, p)$: een som van $n$
onafhankelijke Bernoullivariabelen, elk met kf $1 - p +
p\eu^{\iu\xi}$, dus $\varphi(\xi) = \bigl(1 - p +
p\eu^{\iu\xi}\bigr)^n$ (\cref{prop:b3:clt:cfbasics}(b)).
Poisson-optelbaarheid: als $X \sim \mathcal P(\lambda)$, $Y
\sim \mathcal P(\mu)$ \omterm{def:b3:probability:independence}{onafhankelijk} zijn,
\[
\varphi_{X+Y}(\xi) = \eu^{\lambda(\eu^{\iu\xi}-1)}
\eu^{\mu(\eu^{\iu\xi}-1)} =
\eu^{(\lambda+\mu)(\eu^{\iu\xi}-1)},
\]
de kf van $\mathcal P(\lambda + \mu)$; injectiviteit
(\cref{thm:b3:clt:injectivity}) identificeert de wet.
\end{solution}

\begin{solution}{exo:b3:clt:2}
(a) $\overline{\varphi_X(\xi)} = \E\eu^{-\iu\xi X} =
\varphi_{-X}(\xi)$. Dus $\varphi_X$ is reëel dan en slechts
dan als $\varphi_X = \varphi_{-X}$, dan en slechts dan als
(injectiviteit, \cref{thm:b3:clt:injectivity}) $X$ en $-X$
dezelfde wet hebben.
(b) Schrijf $\varphi_X(\xi_0) = \eu^{\iu\theta}$. Dan
\[
\E\bigl[1 - \cos(\xi_0X - \theta)\bigr]
= 1 - \operatorname{Re}\bigl(\eu^{-\iu\theta}
\varphi_X(\xi_0)\bigr) = 1 - 1 = 0 .
\]
De integrand is niet-negatief, dus $\cos(\xi_0X - \theta)
= 1$ bijna zeker (een niet-negatieve variabele met
\omterm{def:b3:probability:space}{nul-verwachting} verdwijnt b.z.), d.w.z.\ $\xi_0X - \theta
\in 2\pi\Z$ b.z.: $X$ neemt haar waarden in de rekenkundige
rij $\frac{\theta}{\xi_0} + \frac{2\pi}{\xi_0}\Z$ bijna
zeker. Als $X$ een \omterm{ex:b3:lebesgue:gamma}{dichtheid} heeft, is deze aftelbare
verzameling Lebesgue-null, dus draagt kans $0$ ---
contradictie; daarom $\abs{\varphi_X(\xi)} < 1$ voor elke
$\xi \neq 0$.
\end{solution}

\begin{solution}{exo:b3:clt:3}
Door \omterm{def:b3:probability:independence}{onafhankelijkheid} en \cref{prop:b3:clt:cfbasics}:
\[
\varphi_{X+Y}(\xi) = \eu^{\iu m_1\xi - \sigma_1^2\xi^2/2}\,
\eu^{\iu m_2\xi - \sigma_2^2\xi^2/2}
= \eu^{\iu(m_1+m_2)\xi - (\sigma_1^2+\sigma_2^2)\xi^2/2},
\]
de kf van $\mathcal N(m_1 + m_2, \sigma_1^2 + \sigma_2^2)$;
injectiviteit concludeert. Stabiliteit onder affiene
afbeeldingen is de affiene regel ($aX + b \sim \mathcal
N(am_1 + b, a^2\sigma_1^2)$, met ontaard geval $a = 0$
toegestaan), en stabiliteit onder onafhankelijke sommen
volgt door inductie op de berekening hierboven. Voor
\emph{afhankelijke} \omterm{def:b3:clt:gaussianvector}{Gaussischen} hoeft de som niet \omterm{def:b3:clt:gaussianvector}{Gaussisch}
te zijn: in \cref{exo:b3:clt:9} zijn $X$ en $Y =
\varepsilon X$ elk \omterm{def:b3:clt:gaussianvector}{standaard-Gaussisch} maar $X + Y$
verdwijnt met kans $\frac12$ zonder b.z.\ nul te zijn, dus
is niet \omterm{def:b3:clt:gaussianvector}{Gaussisch}.
\end{solution}

\begin{solution}{exo:b3:clt:4}
(a) \emph{Directe implicatie.} Zij $t$ een
continuïteitspunt van $F_X$ en $\delta > 0$. Neem de
continue hellingen $f^-$ ($= 1$ op $\intoc{-\infty}{t-
\delta}$, $0$ vanaf $t$, affien ertussen) en $f^+$ ($= 1$
op $\intoc{-\infty}t$, $0$ vanaf $t + \delta$, affien
ertussen); dan $f^- \leq \mathbf 1_{\intoc{-\infty}t} \leq
f^+$, dus
\[
\E f^-(X_n) \leq F_{X_n}(t) \leq \E f^+(X_n),
\]
en de buitenste termen convergeren naar $\E f^\pm(X)$, zelf
geklemd tussen $F_X(t - \delta)$ en $F_X(t + \delta)$.
Laten $n \to \infty$ dan $\delta \to 0$ en gebruik
continuïteit van $F_X$ in $t$: $F_{X_n}(t) \to F_X(t)$.

\emph{Omgekeerde.} Zij $f$ begrensd continu, $M =
\sup\abs f$, $\varepsilon > 0$. De continuïteitspunten van
$F_X$ zijn \omterm{def:b3:topology:interior}{dicht} ($F_X$ heeft hoogstens aftelbaar veel
sprongen), dus kies continuïteitspunten $a < b$ met $F_X(a)
< \varepsilon$ en $1 - F_X(b) < \varepsilon$. Op de
compacte $\intcc ab$ is $f$ uniform continu: kies
continuïteitspunten $a = t_0 < t_1 < \dots < t_m = b$ van
$F_X$ met oscillatie van $f$ hoogstens $\varepsilon$ op
elke $\intoc{t_{j-1}}{t_j}$, en zet $g = \sum_j
f(t_j)\,\mathbf 1_{\intoc{t_{j-1}}{t_j}}$. Dan $\abs{f -
g} \leq \varepsilon$ op $\intoc ab$, $\abs g \leq M$, en
voor $T = X_n$ of $X$:
\[
\bigl|\E f(T) - \E g(T)\bigr| \leq \varepsilon +
2M\bigl(F_T(a) + 1 - F_T(b)\bigr).
\]
Bovendien $\E g(X_n) = \sum_j f(t_j)\bigl(F_{X_n}(t_j) -
F_{X_n}(t_{j-1})\bigr) \to \E g(X)$ (eindige som van
convergerende termen, alle $t_j$ continuïteitspunten), en
$F_{X_n}(a) \to F_X(a) < \varepsilon$, $1 - F_{X_n}(b) \to
1 - F_X(b) < \varepsilon$. Samenstellen:
$\limsup_n\abs{\E f(X_n) - \E f(X)} \leq 2\varepsilon +
8M\varepsilon$; laat $\varepsilon \to 0$.

(b) De verdelingsfunctie van de constante $c$ is $\mathbf
1_{\intco c\infty}$, continu behalve in $c$. Voor
$\varepsilon > 0$ zijn de punten $c - \varepsilon$ en $c +
\frac\varepsilon2$ continuïteitspunten, dus
\[
\P(\abs{X_n - c} > \varepsilon) \leq F_{X_n}(c -
\varepsilon) + 1 - F_{X_n}\Bigl(c + \frac\varepsilon2\Bigr)
\longrightarrow 0 + 1 - 1 = 0 .
\]
\end{solution}

\begin{solution}{exo:b3:clt:5}
Zet $z_n = p_n(\eu^{\iu\xi} - 1)$, dus $\varphi_{X_n}(\xi)
= (1 + z_n)^n$ (\cref{exo:b3:clt:1}) en $\abs{z_n} \leq
2p_n \to 0$ (merk $p_n = \frac{np_n}n \to 0$). Zowel $1 +
z_n$ als $\eu^{z_n}$ hebben modulus hoogstens $1$: $\abs{1
+ z_n} = \abs{(1 - p_n) + p_n\eu^{\iu\xi}} \leq 1$ door de
driehoeksongelijkheid, en $\abs{\eu^{z_n}} =
\eu^{p_n(\cos\xi - 1)} \leq 1$. De telescoperende
ongelijkheid $\abs{a^n - b^n} \leq n\abs{a - b}$ (bewijs
van \cref{thm:b3:clt:clt}) en de machtreeksgrens
$\abs{\eu^z - 1 - z} \leq \abs z^2\eu^{\abs z}$ geven
\[
\bigl|(1 + z_n)^n - \eu^{nz_n}\bigr| \leq n\bigl|1 + z_n -
\eu^{z_n}\bigr| \leq n\,\abs{z_n}^2\,\eu^{\abs{z_n}} \leq
4\eu^2\,np_n^2 = 4\eu^2\,(np_n)\,p_n \longrightarrow 0 .
\]
Omdat $nz_n = np_n(\eu^{\iu\xi} - 1) \to
\lambda(\eu^{\iu\xi} - 1)$, concluderen we
$\varphi_{X_n}(\xi) \to \exp\bigl(\lambda(\eu^{\iu\xi} -
1)\bigr)$ voor elke $\xi$: de kf van $\mathcal P(\lambda)$,
en Lévy (\cref{thm:b3:clt:levy}) geeft $X_n \Rightarrow
\mathcal P(\lambda)$. Numeriek: $\P\bigl(\mathcal B(100,
0{,}02) = 0\bigr) = 0{,}98^{100} = \eu^{100\ln 0{,}98}
\approx \eu^{-2{,}020} \approx 0{,}1326$, terwijl
$\P\bigl(\mathcal P(2) = 0\bigr) = \eu^{-2} \approx
0{,}1353$: twee procent uit elkaar al bij deze grove $n$.
\end{solution}

\begin{solution}{exo:b3:clt:6}
(a) Eén worp heeft gemiddelde $\frac72$ en variantie
$\frac{35}{12}$, dus $S$ heeft gemiddelde $3500$, variantie
$\frac{35000}{12} \approx 2916{,}7$ en standaardafwijking
$\approx 54{,}0$. Door de CLT,
\[
\P(S > 3600) = \P\Bigl(\frac{S - 3500}{54{,}0} > 1{,}85\Bigr)
\approx 1 - \Phi(1{,}85) \approx 0{,}032 :
\]
ongeveer $3\%$ kans.
(b) $S \sim \mathcal B(100, \frac12)$: gemiddelde $50$,
standaardafwijking $5$. Met de continuïteitscorrectie,
\[
\P(45 \leq S \leq 55) \approx
\Phi\Bigl(\frac{55{,}5 - 50}{5}\Bigr) -
\Phi\Bigl(\frac{44{,}5 - 50}{5}\Bigr) = 2\Phi(1{,}1) - 1
\approx 0{,}729,
\]
tegen de exacte waarde $0{,}7287$; zonder correctie,
$2\Phi(1) - 1 \approx 0{,}683$, bijna vijf punten ernaast.
De correctie telt omdat $S$ een rooster-variabele is: het
atoom $\P(S = k)$ wordt goed benaderd door de Gaussische
massa van $\intcc{k - \frac12}{k + \frac12}$, en knippen
van het interval bij de gehele $45$ en $55$ gooit een
half atoom aan elk eind weg.
\end{solution}

\begin{solution}{exo:b3:clt:7}
De variabelen $g(U_k)$ zijn i.i.d.\ (meetbare beelden van
i.i.d.\ variabelen), kwadratisch \omterm{def:b3:lebesgue:l1}{integreerbaar}, met
gemiddelde $I$ (overdrachtsstelling,
\cref{exo:b3:product:9}) en variantie $\sigma^2$. Als
$\sigma > 0$, is \cref{thm:b3:clt:clt} toegepast op hen
precies de gestelde convergentie
\[
\sqrt n\,\Bigl(\frac1n\sum_{k\leq n}g(U_k) - I\Bigr) =
\frac{\sum_{k\leq n}\bigl(g(U_k) - I\bigr)}{\sqrt n}
\Longrightarrow \mathcal N(0, \sigma^2)
\]
(als $\sigma = 0$, is $g$ b.z.\ constant en verdwijnt de
linkerkant identiek). Dus
$\P\bigl(\abs{\frac1n\sum g(U_k) - I} \leq
1{,}96\,\sigma/\sqrt n\bigr) \to 0{,}95$: de foutbalk $\pm
1{,}96\,\sigma/\sqrt n$ ziet de dimensie $d$ alleen via de
constante $\sigma$, nooit via de ratio in $n$. De
middelpuntsregel met $n$ knopen in dimensie $d$ heeft mesh
$n^{-1/d}$ en fout van orde $n^{-2/d}$ voor $\mathcal
C^2$-integranden. \omterm{ex:b3:probability:sllnapps}{Monte Carlo}'s $n^{-1/2}$ daalt sneller
dan $n^{-2/d}$ precies als $\frac12 > \frac2d$, d.w.z.\ $d
> 4$: vanaf dimensie $5$ wint stochastische steekproeftrekking
asymptotisch van het rooster --- de vloek van de dimensie
spaart probabilistische methoden, vandaar de heerschappij
van \omterm{ex:b3:probability:sllnapps}{Monte Carlo} in hoogdimensionale integratie.
\end{solution}

\begin{solution}{exo:b3:clt:8}
\emph{Som.} Voor vaste $\xi$:
\[
\bigl|\E\eu^{\iu\xi(X_n+Y_n)} -
\eu^{\iu\xi c}\,\E\eu^{\iu\xi X_n}\bigr|
= \bigl|\E\bigl[\eu^{\iu\xi X_n}\bigl(\eu^{\iu\xi Y_n} -
\eu^{\iu\xi c}\bigr)\bigr]\bigr|
\leq \E\bigl|\eu^{\iu\xi(Y_n - c)} - 1\bigr| .
\]
Splits op de gebeurtenis $\{\abs{Y_n - c} \leq \delta\}$:
daar $\abs{\eu^{\iu\xi(Y_n-c)} - 1} \leq \abs\xi\,\delta$
(de koorde is korter dan de boog); het complement draagt
hoogstens $2\,\P(\abs{Y_n - c} > \delta) \to 0$ bij. Dus de
$\limsup$ is $\leq \abs\xi\,\delta$ voor elke $\delta >
0$: het verschil gaat naar $0$. Omdat $\E\eu^{\iu\xi X_n}
\to \varphi_X(\xi)$, krijgen we $\varphi_{X_n+Y_n}(\xi)
\to \eu^{\iu\xi c}\varphi_X(\xi) = \varphi_{X+c}(\xi)$, en
Lévy (\cref{thm:b3:clt:levy}) levert $X_n + Y_n
\Rightarrow X + c$.

\emph{Product.} Eerst, $cX_n \Rightarrow cX$:
$\varphi_{cX_n}(\xi) = \varphi_{X_n}(c\xi) \to
\varphi_X(c\xi) = \varphi_{cX}(\xi)$. Vervolgens, $(Y_n -
c)X_n \to 0$ in kans: de wetten van de $X_n$ zijn gespannen
(hun kf's convergeren naar een kf; zie de spanningsstap van
\cref{thm:b3:clt:levy}), dus gegeven $\varepsilon > 0$ kies
$M$ met $\P(\abs{X_n} > M) \leq \varepsilon$ voor alle $n$;
dan
\[
\P\bigl(\abs{(Y_n - c)X_n} > \varepsilon\bigr) \leq
\P(\abs{X_n} > M) + \P\Bigl(\abs{Y_n - c} >
\frac{\varepsilon}{M}\Bigr) \leq \varepsilon + o(1) .
\]
Schrijven $Y_nX_n = cX_n + (Y_n - c)X_n$ en toepassen van
het somdeel (waarvan het bewijs alleen gebruikte dat $Y_n'
:= (Y_n - c)X_n \to 0$ in kans, met constante $0$):
$Y_nX_n \Rightarrow cX$.

\emph{Toepassing.} Door de sterke wet van de grote aantallen
(\cref{thm:b3:probability:slln}), $\hat p_n \to p$ b.z., dus
door continuïteit $\hat\sigma_n = \sqrt{\hat p_n(1 - \hat
p_n)} \to \sigma = \sqrt{p(1 - p)} > 0$ b.z., vandaar
$\frac{\sigma}{\hat\sigma_n} \to 1$ in kans. De
productregel van Slutsky upgrade $\frac{S_n -
np}{\sigma\sqrt n} \Rightarrow \mathcal N(0,1)$ tot
$\frac{S_n - np}{\hat\sigma_n\sqrt n} =
\frac{\sigma}{\hat\sigma_n}\cdot \frac{S_n -
np}{\sigma\sqrt n} \Rightarrow \mathcal N(0,1)$: het
\emph{bruikbare} betrouwbaarheidsinterval $\hat p_n \pm
1{,}96\,\hat\sigma_n/\sqrt n$, gebouwd uit de data alleen,
houdt haar asymptotisch $95\%$-niveau.
\end{solution}

\begin{solution}{exo:b3:clt:9}
(a) Splitsen van de \omterm{def:b3:probability:space}{verwachting} over de twee waarden van
$\varepsilon$ (\omterm{def:b3:probability:independence}{onafhankelijkheid}): voor Borel $B$, $\P(Y
\in B) = \frac12\P(X \in B) + \frac12\P(-X \in B) = \P(X
\in B)$, want $-X \sim X$ ($\mathcal N(0,1)$ is
symmetrisch): $Y \sim \mathcal N(0,1)$. En
$\operatorname{Cov}(X, Y) = \E[\varepsilon X^2] =
\E[\varepsilon]\,\E[X^2] = 0 \cdot 1 = 0$.
(b) $\abs Y = \abs X$, dus $\P(\abs X \leq 1,\ \abs Y \geq
2) = 0$ terwijl $\P(\abs X \leq 1)\,\P(\abs Y \geq 2) >
0$: niet \omterm{def:b3:probability:independence}{onafhankelijk}. Als $(X, Y)$ een \omterm{def:b3:clt:gaussianvector}{Gaussische vector}
ware, zou $X + Y = (1 + \varepsilon)X$ een reële Gaussische
variabele zijn (\cref{def:b3:clt:gaussianvector} met $t =
(1,1)$); maar $\P(X + Y = 0) = \P(\varepsilon = -1) =
\frac12$, terwijl een Gaussische variabele een atoom alleen
heeft als zij b.z.\ constant is --- en $X + Y$ is gelijk
aan $2X \neq 0$ b.z.\ op $\{\varepsilon = 1\}$. Contradictie:
$(X, Y)$ is niet \omterm{def:b3:clt:gaussianvector}{Gaussisch}.
(c) Elke marginale is \omterm{def:b3:clt:gaussianvector}{Gaussisch} en de covariantie verdwijnt,
toch faalt \omterm{def:b3:probability:independence}{onafhankelijkheid} --- omdat het \emph{paar} niet
gezamenlijk \omterm{def:b3:clt:gaussianvector}{Gaussisch} is.
\cref{thm:b3:clt:gaussianvector}(2) kan niet verzwakt
worden tot ``Gaussische marginalen''.
\end{solution}

\begin{solution}{exo:b3:clt:10}
(a) \cref{exo:b3:fouriertransform:1} berekent
$\widehat{\eu^{-\abs\cdot}}(\xi) = \frac{2}{1 + \xi^2}$;
beide zijden \omterm{def:b3:lebesgue:l1}{integreerbaar}, Fourier-inversie
(\cref{thm:b3:fouriertransform:inversion}) keert dit om:
\[
\int_\R\eu^{\iu\xi x}\,\frac{\dd x}{\pi(1 + x^2)} =
\eu^{-\abs\xi},
\]
wat precies $\varphi_X(\xi)$ is voor een Cauchy-variabele
$X$.
(b) Door \omterm{def:b3:probability:independence}{onafhankelijkheid}, $\varphi_{S_n}(\xi) =
\bigl(\eu^{-\abs\xi}\bigr)^n = \eu^{-n\abs\xi}$, dus
$\varphi_{S_n/n}(\xi) = \varphi_{S_n}(\xi/n) =
\eu^{-\abs\xi}$: het empirisch gemiddelde $\frac{S_n}n$ is
opnieuw standaard-Cauchy voor elke $n$ (injectiviteit). Het
gemiddelde concentreert nooit: haar fluctuaties op tijd
$10^6$ zijn die van een enkele observatie.
(c) $\E\abs{X_1} = \frac2\pi\int_0^\infty\frac{x\,\dd x}{1
+ x^2} = +\infty$: de Cauchywet is niet \omterm{def:b3:lebesgue:l1}{integreerbaar}, dus
past de sterke wet van de grote aantallen
(\cref{thm:b3:probability:slln}) niet toe, en de CLT (die
eindige variantie eist) nog minder. Hier falen hun
conclusies echt, niet slechts hun bewijzen.
Consistentiecheck: $\varphi(\xi) = \eu^{-\abs\xi}$ is niet
differentieerbaar in $0$, zoals
\cref{prop:b3:clt:cfbasics}(c) contrapositief voorspelt
voor een niet-integreerbare variabele.
\end{solution}

\begin{solution}{exo:b3:clt:11}
(a) $\varphi_{aX_1 + bX_2}(\xi) =
\eu^{-a\abs\xi}\eu^{-b\abs\xi} = \eu^{-(a+b)\abs\xi} =
\varphi_{(a+b)X_1}(\xi)$ (\omterm{def:b3:probability:independence}{onafhankelijkheid} en
\cref{exo:b3:clt:10}); injectiviteit identificeert de
wetten.

(b) $\varphi_{aX_1+bX_2}(\xi) = \eu^{-a^2\xi^2/2}
\eu^{-b^2\xi^2/2} = \eu^{-(a^2+b^2)\xi^2/2}$: de wet van
$\sqrt{a^2+b^2}\,X_1$.

(c) Als $\varphi_X(\xi) = \eu^{-c\abs\xi^\alpha}$, dan
heeft $S_n = X_1 + \dots + X_n$ $\varphi_{S_n} =
\eu^{-cn\abs\xi^\alpha}$, en $S_n/n^{1/\alpha}$ heeft
$\varphi(\xi) = \eu^{-c\abs\xi^\alpha}$ opnieuw: exacte
zelfreproductie onder de $n^{1/\alpha}$-schaling ---
$\sqrt n$ voor de Gaussische ($\alpha = 2$), $n$ zelf voor
Cauchy ($\alpha = 1$, \cref{exo:b3:clt:10}(b)). Een som van
i.i.d.\ variabelen kan alleen (na affiene normalisatie)
convergeren naar een wet die \omterm{def:b3:ode:stability}{stabiel} is onder zulke
convoluties; de CLT zegt dat eindige variantie het
Gaussische bekken dwingt, en het Cauchybekken is
gereserveerd voor wetten met staarten zo zwaar dat $\E X^2
= \infty$ en zelfs $\E\abs X = \infty$ --- bv.\ sommen van
Cauchyvariabelen zelf. Universaliteit heeft meerdere
eilanden, geïndexeerd door de staartexponent $\alpha \in
\intoc02$.
\end{solution}

\begin{solution}{exo:b3:clt:12}
(a) De indicatoren $\mathbf 1_{X_k \leq t}$ zijn i.i.d.\
Bernoulli van parameter $p = F(t)$: hun som $nF_n(t)$ is
binomiaal $\mathcal B(n, p)$; de sterke wet geeft $F_n(t)
\to p$ b.z., en de CLT (\cref{thm:b3:clt:clt}) toegepast op
dezelfde indicatoren (variantie $p(1-p)$) geeft de
gestelde Gaussische limiet.

(b) $p(1 - p)$ is \omterm{def:b3:rings:primemaximal}{maximaal} in $p = \frac12$, d.w.z.\ waar
$F(t) = \frac12$: bij de \emph{mediaan}. Staartkansen
schatten is asymptotisch makkelijk (variantie $\to 0$ als
$p \to 0, 1$); de mediaanregio draagt de grootste
statistische ruis --- de empirische kromme wiebelt het
meest in haar midden.

(c) Voor continue $F$ zijn de variabelen $U_k = F(X_k)$
i.i.d.\ uniform op $\intoo01$
(\cref{exo:b3:probability:1}), en monotonie van $F$ geeft,
met $G_n$ de empirische verdelingsfunctie van de $U_k$:
\[
\sup_{t\in\R}\,\abs{F_n(t) - F(t)}
= \sup_{u \in \operatorname{im}F}\,\abs{G_n(u) - u}
= \sup_{u\in\intcc01}\abs{G_n(u) - u} :
\]
de eerste gelijkheid omdat $\{X_k \leq t\} = \{U_k \leq
F(t)\}$ op null-gebeurtenissen na (monotonie; strikte
ongelijkheid kan alleen falen op de platte delen van $F$,
waar beide zijden onveranderd blijven), en de tweede omdat
een continue $F$, lopend van $0$ naar $1$, elke waarde van
$\intoo01$ bereikt (tussenwaardestelling), en de eindpunten
niets toevoegen ($G_n(0) - 0 = 0$ en $G_n(1) - 1 = 0$). De
rechterkant betrekt alleen uniformen: één wet voor alle $F$
--- dus één tabel van kritische waarden (die van de
Kolmogorovverdeling) toetst \emph{elk} continu model tegen
data.
\end{solution}

\begin{solution}{pb:b3:clt:1}
\textbf{1.} De familie $(X_1, \dots, X_n, N_1, \dots, N_n)$
is \omterm{def:b3:probability:independence}{onafhankelijk}: de twee blokken zijn onderling
\omterm{def:b3:probability:independence}{onafhankelijk} door constructie en elk blok is i.i.d. $W_i$
is een \omterm{def:b3:lebesgue:measurable}{meetbare functie} van de variabelen $(X_j)_{j<i}$ en
$(N_j)_{j>i}$ alleen, alle distinct van $X_i$ en $N_i$:
door het coalitiebeginsel
(\cref{thm:b3:probability:independence}) is $W_i$
\omterm{def:b3:probability:independence}{onafhankelijk} van het paar $(X_i, N_i)$. De decomposities
$H_i = W_i + \frac{X_i}{\sqrt n}$ en $H_{i-1} = W_i +
\frac{N_i}{\sqrt n}$ volgen onmiddellijk uit de
definities: van $H_i$ naar $H_{i-1}$ wisselt de enige term
$X_i$ voor $N_i$.

\textbf{2.} Taylor--Lagrange van orde $3$: er is $c$ tussen
$w$ en $w + h$ met $f(w + h) = f(w) + f'(w)h +
\frac12f''(w)h^2 + \frac16f'''(c)h^3$, en $\abs{f'''(c)}
\leq M_3$ geeft de grens.

\textbf{3.} Aftrekken van de twee ontwikkelingen in het
gemeenschappelijke basispunt $w = W_i$:
\[
f(H_i) - f(H_{i-1}) = f'(W_i)\,\frac{X_i - N_i}{\sqrt n} +
\frac{f''(W_i)}{2}\,\frac{X_i^2 - N_i^2}{n} + R_i,
\qquad
\abs{R_i} \leq \frac{M_3}{6}\cdot
\frac{\abs{X_i}^3 + \abs{N_i}^3}{n^{3/2}} .
\]
Neem \omterm{def:b3:probability:space}{verwachtingen}. Door vraag 1 zijn $f'(W_i)$ en
$f''(W_i)$ onafhankelijk van $(X_i, N_i)$, dus de gemengde
\omterm{def:b3:probability:space}{verwachtingen} factoriseren:
\begin{align*}
\E\Bigl[f'(W_i)\,\frac{X_i - N_i}{\sqrt n}\Bigr] &=
\E\bigl[f'(W_i)\bigr]\,\frac{\E X_i - \E N_i}{\sqrt n} = 0,
\\
\E\Bigl[f''(W_i)\,\frac{X_i^2 - N_i^2}{n}\Bigr] &=
\E\bigl[f''(W_i)\bigr]\,\frac{1 - 1}{n} = 0 :
\end{align*}
de eerste twee momenten van $X_i$ en $N_i$ \emph{matchen},
en alleen de rest overleeft:
\[
\bigl|\E f(H_i) - \E f(H_{i-1})\bigr| \leq \E\abs{R_i} \leq
\frac{M_3}{6}\cdot\frac{\beta + \gamma}{n^{3/2}} .
\]
Het Gaussische derde moment: $\gamma = \E\abs{N_1}^3 =
2\int_0^\infty x^3\,\frac{\eu^{-x^2/2}}{\sqrt{2\pi}}\,\dd x
= \frac{2}{\sqrt{2\pi}}\int_0^\infty 2u\,\eu^{-u}\dd u =
\frac{4}{\sqrt{2\pi}} = \frac{2\sqrt2}{\sqrt\pi}$
(substitutie $u = x^2/2$, dan $\Gamma(2) = 1$).

\textbf{4.} Telescoperen $\E f(T_n) - \E f(G_n) =
\sum_{i=1}^n\bigl(\E f(H_i) - \E f(H_{i-1})\bigr)$ en
vraag 3 toepassen op elk van de $n$ termen:
\[
\bigl|\E f(T_n) - \E f(G_n)\bigr| \leq
n \cdot \frac{M_3(\beta + \gamma)}{6\,n^{3/2}} =
\frac{M_3\,(\beta + \gamma)}{6\,\sqrt n} .
\]

\textbf{5.} $G_n$ is $\mathcal N(0,1)$ \emph{exact} voor
elke $n$ (een genormaliseerde som van onafhankelijke
\omterm{def:b3:clt:gaussianvector}{standaard-Gaussischen}, \cref{exo:b3:clt:3}), dus $\E
f(G_n) = \E f(N)$ en vraag 4 leest $\abs{\E f(T_n) - \E
f(N)} \leq \frac{M_3(\beta+\gamma)}{6\sqrt n} \to 0$ voor
$f \in \mathcal C^3_b$. \emph{Upgrade.} Zij $f$ begrensd
continu, $M = \sup\abs f$, $\varepsilon > 0$. Kies $A \geq
1$ met $\frac1{A^2} \leq \varepsilon$: Chebyshev met
$\V(T_n) = 1$ geeft $\P(\abs{T_n} > A) \leq \varepsilon$
voor alle $n$, en evenzo $\P(\abs N > A) \leq
\varepsilon$. Zij $\chi$ $\mathcal C^\infty$ met $\mathbf
1_{\intcc{-A}A} \leq \chi \leq \mathbf
1_{\intcc{-A-1}{A+1}}$ (een glad plateau, gebouwd door
mollificatie van $\mathbf 1_{\intcc{-A-\frac12}{A+\frac12}}$,
\cref{thm:b3:lp:regularization}); $g = f\chi$ is continu
met compacte drager, dus uniform continu, zodat haar
mollificatie $g_\eta = g * \rho_\eta$ $\mathcal C^\infty$
is met begrensde afgeleiden van alle ordes en $\norm{g -
g_\eta}_\infty \leq \varepsilon$ voor $\eta$ klein genoeg.
Voor $T = T_n$ of $N$, omdat $f = g$ op $\intcc{-A}A$ en
$\abs{f - g} \leq 2M$ overal:
\[
\bigl|\E f(T) - \E g_\eta(T)\bigr| \leq
\E\abs{(f - g)(T)} + \norm{g - g_\eta}_\infty
\leq 2M\,\P(\abs T > A) + \varepsilon \leq
(2M + 1)\,\varepsilon .
\]
Combinerend met $\E g_\eta(T_n) \to \E g_\eta(N)$ (vraag 4
past toe: $g_\eta \in \mathcal C^3_b$):
\[
\limsup_n\;\bigl|\E f(T_n) - \E f(N)\bigr| \leq
2(2M + 1)\,\varepsilon ,
\]
en $\varepsilon$ was willekeurig: $\E f(T_n) \to \E f(N)$
voor elke begrensde continue $f$, d.w.z.\ $T_n \Rightarrow
\mathcal N(0,1)$.

\textbf{6.} Identieke verdeling trad alleen op via één zin:
``$X_i$ en $N_i$ hebben dezelfde eerste twee momenten''.
Laat dus $X_1, \dots, X_n$ \omterm{def:b3:probability:independence}{onafhankelijk}, gecentreerd,
met varianties $\sigma_i^2$ en eindige derde momenten,
$s_n^2 = \sum_i\sigma_i^2 > 0$, en neem $N_i \sim
\mathcal N(0, \sigma_i^2)$ \omterm{def:b3:probability:independence}{onafhankelijk} van alles.
Definieer de hybriden met normalisatie $s_n$: $H_i =
\frac1{s_n}(\sum_{j\leq i}X_j + \sum_{j>i}N_j)$. In de
$i$-de wissel doden $\E X_i = \E N_i = 0$ en $\E X_i^2 =
\E N_i^2 = \sigma_i^2$ opnieuw de $f'$- en $f''$-termen, en
de rest geeft (met $\E\abs{N_i}^3 = \sigma_i^3\gamma$ door
schaling):
\[
\bigl|\E f(H_i) - \E f(H_{i-1})\bigr| \leq
\frac{M_3}{6\,s_n^3}\bigl(\E\abs{X_i}^3 +
\sigma_i^3\gamma\bigr) .
\]
Telescoperen:
\[
\Bigl|\E f\Bigl(\frac{X_1 + \dots + X_n}{s_n}\Bigr) -
\E f(N)\Bigr| \leq \frac{M_3}{6\,s_n^3}\sum_{i=1}^n
\Bigl(\E\abs{X_i}^3 + \V(X_i)^{3/2}\,\gamma\Bigr) .
\]
Omdat $\sigma_i^3 = (\E X_i^2)^{3/2} \leq \E\abs{X_i}^3$
(de power--mean-ongelijkheid, d.w.z.\ Jensen voor $t
\mapsto t^{3/2}$ toegepast op $X_i^2$), is de rechterkant
hoogstens $\frac{M_3(1 + \gamma)}{6}\cdot
\frac{\sum_i\E\abs{X_i}^3}{s_n^3}$: onder de
\emph{Lyapunovvoorwaarde}
$\frac1{s_n^3}\sum_i\E\abs{X_i}^3 \to 0$ convergeren de
genormaliseerde sommen in wet naar $\mathcal N(0,1)$ ---
de CLT zonder identieke verdeling.

\textbf{7.} Zij $\rho \in \mathcal C^\infty_c(\intoo01)$
met $\int\rho = 1$ en zet $\psi(x) = \int_x^1\rho(s)\dd
s$: $\psi$ is $\mathcal C^\infty$, niet-stijgend, $\psi =
1$ op $\R_-$, $\psi = 0$ op $\intco1\infty$; zij $K =
\norm{\psi'''}_\infty$. Voor $t \in \R$ en $\delta > 0$
definieer $\psi_\delta(x) = \psi\bigl(\frac{x -
t}\delta\bigr)$ en $\tilde\psi_\delta(x) =
\psi\bigl(\frac{x - t}\delta + 1\bigr)$: deze zijn
$\mathcal C^3_b$ met derde afgeleide begrensd door
$K/\delta^3$, en
\[
\mathbf 1_{\intoc{-\infty}{t-\delta}} \leq
\tilde\psi_\delta \leq \mathbf 1_{\intoc{-\infty}t} \leq
\psi_\delta \leq \mathbf 1_{\intoc{-\infty}{t+\delta}} .
\]
Bovengrens: door vraag 4 toegepast op $\psi_\delta$ (met
$M_3 = K/\delta^3$),
\[
\P(T_n \leq t) \leq \E\psi_\delta(T_n) \leq
\E\psi_\delta(N) + \frac{K(\beta +
\gamma)}{6\,\delta^3\sqrt n}
\leq \Phi(t + \delta) + \frac{K(\beta +
\gamma)}{6\,\delta^3\sqrt n}
\leq \Phi(t) + \frac{\delta}{\sqrt{2\pi}} +
\frac{K(\beta + \gamma)}{6\,\delta^3\sqrt n},
\]
omdat $\Phi$ Lipschitz is met constante
$\frac1{\sqrt{2\pi}}$ (haar \omterm{ex:b3:lebesgue:gamma}{dichtheid} is begrensd door
$\frac1{\sqrt{2\pi}}$). De symmetrische ondergrens via
$\tilde\psi_\delta$ geeft de tweetermenschatting
\[
\sup_{t\in\R}\,\bigl|\P(T_n \leq t) - \Phi(t)\bigr| \leq
\frac{K(\beta + \gamma)}{6}\cdot\frac{1}{\delta^3\sqrt n} +
\frac{\delta}{\sqrt{2\pi}}
\qquad(\delta > 0\ \text{willekeurig}).
\]
De twee termen balanceren als $\delta^{-3}n^{-1/2} \asymp
\delta$, d.w.z.\ $\delta = n^{-1/8}$: beide zijn dan
$O(n^{-1/8})$, een expliciete uniforme ratio geldig voor
elke $n$. (De optimale Berry--Esseen-ratio $C\beta/\sqrt n$
eist de Fourier-gladmakingsongelijkheid; wisselen ruilt
scherpte voor volledige elementariteit.)

\textbf{8.} Voor $X_i = 2B_i - 1$ (eerlijke tekens):
gecentreerd, variantie $1$, en $\abs{X_i} = 1$ dus $\beta
= 1$. Vraag 7 begrenst dan
$\sup_t\abs{\P(\frac{S_n}{\sqrt n} \leq t) - \Phi(t)}$
expliciet en uniform voor \emph{elke} eindige $n$ --- een
globale, niet-asymptotische uitspraak over de
verdelingsfunctie. De lokale schatting van
\cref{pb:b3:product:1}, vraag 7, geeft in plaats daarvan
de exacte asymptotiek van een individueel atoom, $\P(S_{2n}
= 2k) \sim \frac{\eu^{-k^2/n}}{\sqrt{\pi n}}$: zij lost
kansen van grootte $n^{-1/2}$ op, ver onder de
$n^{-1/8}$-resolutie van vraag 7, maar is puntsgewijs,
asymptotisch (geen expliciete fout bij vaste $n$) en
gebonden aan deze specifieke roosterwet. Lokale precisie
versus globale uniformiteit: de twee methoden zijn
complementair, en sommeren van de lokale schatting over $k
\in \intint{a\sqrt n}{b\sqrt n}$ herwint De
Moivre--Laplace op intervallen --- met een scherpere ratio,
maar alleen voor deze wet.

\textbf{9.} Het wisselargument gebruikte niets over de wet
van de $X_i$ buiten $\E X_i = 0$, $\E X_i^2 = 1$ en de
eindigheid van $\E\abs{X_i}^3$: hadden we de \omterm{def:b3:clt:gaussianvector}{Gaussischen}
$N_i$ vervangen door een andere i.i.d.\ familie met
dezelfde eerste twee momenten en eindig derde moment, zou
dezelfde telescopering $\abs{\E f(\text{som}_X) - \E
f(\text{som}_Y)}$ begrensd hebben door $O(n^{-1/2})$ voor
elke gladde $f$. Gladde statistieken van grote
onafhankelijke sommen zijn daarom \emph{universeel}: tot op
een gekwantificeerde fout hangen zij van de wet van de
termen alleen af via twee getallen. Dit is het
invariantiebeginsel: bewijs een limietstelling voor de
meest berekenbare wet (de Gaussische, waar alles exact is),
dan transfer haar naar alle wetten door wisselen. Hetzelfde
schema --- met sommen vervangen door meer uitgewerkte
functionalen --- drijft de halfcirkelwet van Wigner voor
toevalsmatrices, de universaliteit van wortels van
toevalspolynomen, en veel van de moderne kansrekening; de
centrale limietstelling is haar eerste en eenvoudigste
instantie.

\textbf{10.} $Z$ is een Borelfunctie van $(N_{i+1}, \dots,
N_n)$ alleen, terwijl $A = W_i + \theta h - Z$ en $h$
functies zijn van de overige variabelen van de
onafhankelijke familie $(X_1, \dots, X_n, N_1, \dots,
N_n)$: door het coalitiebeginsel
(\cref{thm:b3:probability:independence}) is $Z$
onafhankelijk van $(A, h)$. Als som van de onafhankelijke
$N_j/\sqrt n \sim \mathcal N(0, \frac1n)$ is $Z \sim
\mathcal N(0, s^2)$ met $s^2 = \frac{n-i}n$
(\cref{exo:b3:clt:3}), met \omterm{ex:b3:lebesgue:gamma}{dichtheid} begrensd door
$\frac1{s\sqrt{2\pi}}$. De wet van $((A, h), Z)$ is het
product van de twee marginalen, dus Tonelli (overdracht)
bevriest het eerste blok: met $G(a) = \E\abs{g(a + Z)} =
\int\abs{g(a + z)}\,\varphi_s(z)\,\dd z \leq
\frac{\norm g_{L^1}}{s\sqrt{2\pi}}$ voor elke $a$,
\[
\E\bigl[\abs h^3\abs{g(A + Z)}\bigr] =
\E\bigl[\abs h^3\,G(A)\bigr] \leq
\frac{\norm g_{L^1}}{s\sqrt{2\pi}}\,\E\abs h^3
= \sqrt{\frac{n}{2\pi(n-i)}}\,\norm g_{L^1}\,\E\abs h^3 .
\]

\textbf{11.} De integraalvorm van de formule van Taylor
volgt door tweemaal partieel integreren in $\theta$ van
$f(w + h) - f(w) = h\int_0^1f'(w + \theta h)\,\dd\theta$.
\omterm{def:b3:probability:space}{Verwachtingen} nemen in de $i$-de wissel: ordes $0, 1, 2$
vallen exact uit zoals in vraag 3, en de twee resten (voor
$h = X_i/\sqrt n$ en $N_i/\sqrt n$) zijn begrensd, voor $i
\leq n - 1$, door vraag 10 met $g = f'''$:
\[
\bigl|\E f(H_i) - \E f(H_{i-1})\bigr| \leq
\int_0^1\frac{(1-\theta)^2}2\,\dd\theta\;
\sqrt{\frac{n}{2\pi(n-i)}}\,\norm{f'''}_{L^1}
\frac{\beta + \gamma}{n^{3/2}}
= \frac{\beta + \gamma}{6\,n^{3/2}}
\sqrt{\frac{n}{2\pi(n-i)}}\,\norm{f'''}_{L^1} .
\]
Sommeren, met $\sum_{i=1}^{n-1}\sqrt{\frac n{n-i}} =
\sqrt n\sum_{m=1}^{n-1}m^{-1/2} \leq 2n$, en de grens van
vraag 3 voor de laatste wissel ($i = n$, geen \omterm{def:b3:clt:gaussianvector}{Gaussisch}
meer):
\[
\bigl|\E f(T_n) - \E f(G_n)\bigr| \leq
\frac{\beta + \gamma}{3\sqrt{2\pi}}\cdot
\frac{\norm{f'''}_{L^1}}{\sqrt n} +
\frac{M_3(\beta + \gamma)}{6\,n^{3/2}} .
\]
Hellingen: $\psi_\delta'''(x) =
\delta^{-3}\psi'''\bigl(\frac{x - t}\delta\bigr)$, dus
$M_3 = K\delta^{-3}$ met $K = \norm{\psi'''}_\infty$ en
$\norm{\psi_\delta'''}_{L^1} = \delta^{-2}
\norm{\psi'''}_{L^1} = K_1\delta^{-2}$ (substitutie). De
sandwich van vraag 7 geeft dan
\[
\sup_t\,\bigl|\P(T_n \leq t) - \Phi(t)\bigr| \leq
\frac{K_1(\beta + \gamma)}{3\sqrt{2\pi}}\cdot
\frac1{\delta^2\sqrt n} +
\frac{K(\beta + \gamma)}{6}\cdot
\frac1{\delta^3n^{3/2}} + \frac\delta{\sqrt{2\pi}} .
\]
In $\delta = n^{-1/6}$ zijn de eerste en derde termen
$O(n^{-1/6})$ en de middelste $O(n^{-1})$: een uniforme
ratio $O(n^{-1/6})$, strikt beter dan de $n^{-1/8}$ van
vraag 7 --- de Gaussische helft van de hybride deed de
extra gladmaking.

\textbf{12.} $\E N_1^3 = 0$ (oneven integrand), en partieel
integreren geeft $\E N_1^4 = 3\,\E N_1^2 = 3$ ($\int
x^3\cdot x\varphi(x)\dd x = 3\int x^2\varphi$). Voor $f$
van klasse $\mathcal C^4$ met begrensde afgeleiden,
ontwikkel elke wissel tot orde vier: de derde-ordetermen
dragen de factor $\E X_i^3 - \E N_i^3 = 0$
(\omterm{def:b3:probability:independence}{onafhankelijkheid} factoriseert ze zoals in vraag 3), dus
alleen de vierde-orde-rest
$\int_0^1\frac{(1-\theta)^3}6f^{(4)}(w + \theta
h)h^4\dd\theta$ overleeft, met $\int_0^1
\frac{(1-\theta)^3}6\dd\theta = \frac1{24}$ en $\E h^4 =
\beta_4n^{-2}$ of $3n^{-2}$. Vraag 10 (met $g = f^{(4)}$)
begrenst de wissels $i \leq n - 1$, en sommeren zoals in
vraag 11:
\[
\bigl|\E f(T_n) - \E f(G_n)\bigr| \leq
\frac{\beta_4 + 3}{12\sqrt{2\pi}}\cdot
\frac{\norm{f^{(4)}}_{L^1}}{n} +
\frac{M_4(\beta_4 + 3)}{24\,n^2} .
\]
Met $\norm{\psi_\delta^{(4)}}_{L^1} = K_2\delta^{-3}$ en
$M_4 = K'\delta^{-4}$ wordt de verdelingsfunctiegrens
$C\delta^{-3}n^{-1} + C'\delta^{-4}n^{-2} +
\frac\delta{\sqrt{2\pi}}$; in $\delta = n^{-1/4}$ zijn de
buitenste termen $O(n^{-1/4})$ en de middelste $O(n^{-1})$:
ratio $O(n^{-1/4})$.

\textbf{13.} Met $k$ gematchte momenten is de overlevende
rest per wissel van orde $\E\abs h^{k+1} \asymp
n^{-(k+1)/2}$; de verborgen-Gaussische grens belast
$\norm{f^{(k+1)}}_{L^1}$ en de som over wissels draagt de
factor $2n$ bij, wat $\asymp\norm{f^{(k+1)}}_{L^1}\,
n^{-(k-1)/2}$ geeft voor gladde $f$. Hellingen kosten
$\norm{\psi_\delta^{(k+1)}}_{L^1} \asymp \delta^{-k}$,
dus de verdelingsfunctiefout is $\asymp
\delta^{-k}n^{-(k-1)/2} + \delta$, gebalanceerd in $\delta
= n^{-(k-1)/(2k+2)}$: ratio $n^{-(k-1)/(2k+2)}$, wat
$n^{-1/6}$ is voor $k = 2$, $n^{-1/4}$ voor $k = 3$, en
slechts naar $n^{-1/2}$ neigt als $k \to \infty$ --- maar
$k \geq 4$ zou $\E X_1^4 = 3$ en verder forceren, d.w.z.\
een wet die de Gaussische al imiteert. De verzadiging is
structureel: wisselen telt $n$ wisselfouten \emph{in
absolute waarde} op, en doet afstand van alle cancellatie
tussen wissels. Het Fourierbewijs vergelijkt
\omterm{def:b3:clt:cf}{karakteristieke functies}, waar de fouten met hun
oscillerende fasen verschijnen; de gladmakingsongelijkheid
van Esseen zet $\abs{\varphi_{T_n} - \varphi_N}$,
geïntegreerd tegen $\frac{\dd\xi}{\abs\xi}$, om in een
verdelingsfunctiegrens tegen slechts logaritmische kost, en
levert Berry--Esseen's $C\beta n^{-1/2}$ uit drie momenten.
Vervanging ruilt optimaliteit voor robuustheid --- en, zoals
Deel VII toont, voor overdraagbaarheid.

\textbf{14.} $\Sigma$ is symmetrisch positief semidefiniet;
met $\Sigma = PDP^{\mathsf T}$ ($P$ orthogonaal, $D \geq 0$
diagonaal, \cref{exo:b3:submanifolds:8}) voldoet de
symmetrische $C = P\sqrt DP^{\mathsf T}$ aan $C^2 =
\Sigma$. Voor elke $t \in \R^2$ is $\langle t, CZ\rangle =
\langle Ct, Z\rangle = (Ct)_1Z^1 + (Ct)_2Z^2$ een lineaire
combinatie van onafhankelijke \omterm{def:b3:clt:gaussianvector}{Gaussischen}, dus \omterm{def:b3:clt:gaussianvector}{Gaussisch}
(\cref{exo:b3:clt:3}): $N = CZ$ is een \omterm{def:b3:clt:gaussianvector}{Gaussische vector};
haar gemiddelde is $0$ en haar covariantie $\E[NN^{\mathsf
T}] = C\,\E[ZZ^{\mathsf T}]\,C^{\mathsf T} = CC^{\mathsf
T} = \Sigma$. Momenten: $\norm N^3 \leq (\abs{N_1} +
\abs{N_2})^3 \leq 4(\abs{N_1}^3 + \abs{N_2}^3)$
(convexiteit van $x^3$ op $\R_+$), en elke coördinaat is
een reële Gaussische met momenten van alle ordes
(\cref{exo:b3:product:10}): $\gamma' < \infty$. Ten slotte
is elke $\langle t, G_n\rangle = \frac1{\sqrt
n}\sum_i\langle t, N_i\rangle$ een genormaliseerde som van
i.i.d.\ $\mathcal N(0, t^{\mathsf T}\Sigma t)$, dus exact
$\mathcal N(0, t^{\mathsf T}\Sigma t)$: $G_n$ is een
\omterm{def:b3:clt:gaussianvector}{Gaussische vector} met gemiddelde $0$ en covariantie
$\Sigma$, en haar wet is $\mathcal N(0, \Sigma)$
(\cref{def:b3:clt:gaussianvector}: de wet wordt door deze
data bepaald).

\textbf{15.} Zij $\phi(t) = f(w + th)$, $t \in \intcc01$:
$\phi$ is $\mathcal C^3$ met
\[
\phi'''(t) = \sum_{j,k,l\in\{1,2\}}\partial_{jkl}f(w +
th)\,h_jh_kh_l, \qquad \abs{\phi'''(t)} \leq
M_3\Bigl(\sum_j\abs{h_j}\Bigr)^3 = M_3(\abs{h_1} +
\abs{h_2})^3 .
\]
Taylor--Lagrange van orde $3$ voor $\phi$ tussen $0$ en $1$
geeft de eerste ongelijkheid; Cauchy--Schwarz geeft
$\abs{h_1} + \abs{h_2} \leq \sqrt2\norm h$, vandaar de
constante $\frac{2\sqrt2M_3}6 = \frac{\sqrt2M_3}3$.

\textbf{16.} Definieer $H_i$ en $W_i$ zoals in vraag 1, nu
in $\R^2$; het coalitieargument is onveranderd. In de
$i$-de wissel geven de eerste-ordetermen
$\sum_j\E[\partial_jf(W_i)]\,(\E X_{i,j} - \E N_{i,j})/
\sqrt n = 0$ en de tweede-ordetermen
$\frac1{2n}\sum_{j,k}\E[\partial_{jk}f(W_i)]\,(\Sigma_{jk}
- \Sigma_{jk}) = 0$: gemiddelden en covarianties matchen.
Vraag 15 begrenst de twee resten:
\[
\bigl|\E f(H_i) - \E f(H_{i-1})\bigr| \leq
\frac{\sqrt2M_3}3\cdot\frac{\E\norm{X_i}^3 +
\E\norm{N_i}^3}{n^{3/2}}
= \frac{\sqrt2M_3(\beta' + \gamma')}{3\,n^{3/2}},
\]
en telescoperen over de $n$ wissels:
\[
\Bigl|\E f\Bigl(\frac{S_n}{\sqrt n}\Bigr) - \E
f(G_n)\Bigr| \leq \frac{\sqrt2\,M_3(\beta' +
\gamma')}{3\sqrt n},
\qquad G_n \sim \mathcal N(0, \Sigma)\ \text{exact} .
\]
Upgrade: $\E\norm{T_n}^2 = \E\norm{X_1}^2 =
\operatorname{tr}\Sigma$ (kruistermen verdwijnen door
\omterm{def:b3:probability:independence}{onafhankelijkheid} en centrering), dus $\P(\norm{T_n} > A)
\leq \operatorname{tr}\Sigma/A^2$, en evenzo voor $N$:
spanning. Gegeven een begrensde continue $f$ en
$\varepsilon > 0$, vermenigvuldig met een glad plateau
$\chi$ gelijk aan $1$ op de bal van straal $A$ en
ondersteund in straal $A + 1$ (mollificeer een indicator in
$\R^2$, \cref{thm:b3:lp:regularization}); $g = f\chi$ is
uniform continu met compacte drager, dus haar
tweedimensionale mollificatie $g_\eta$ is $\mathcal
C^\infty$ met begrensde afgeleiden van alle ordes en
$\norm{g - g_\eta}_\infty \leq \varepsilon$ voor kleine
$\eta$. De drie-$\varepsilon$-keten van vraag 5
transferreert letterlijk: $\E f(T_n) \to \E f(N)$ voor
elke begrensde continue $f \colon \R^2 \to \R$. Dit is
\cref{thm:b3:clt:multiclt} voor $d = 2$, nu bewezen ---
wisselen omzeilt de tweedimensionale stelling van Lévy die
het hoofdstuk had toegegeven.

\textbf{17.} Voor begrensde continue $g \colon \R \to \R$
is de afbeelding $x \mapsto g(\langle t, x\rangle)$
begrensd continu op $\R^2$, dus vraag 16 geeft $\E
g(\langle t, T_n\rangle) \to \E g(\langle t, N\rangle)$:
elke projectie convergeert in verdeling, en $\langle t,
N\rangle \sim \mathcal N(0, t^{\mathsf T}\Sigma t)$. (Dit
is de makkelijke richting van Cramér--Wold: gezamenlijke
convergentie impliceert convergentie van alle lineaire
beelden.) Toepassing: $V_i = (\xi_i, \xi_i^2 - 1)$ zijn
i.i.d.\ gecentreerde vectoren ($\E\xi_1^2 = 1$), met
covariantie-ingangen $\V(\xi_1) = 1$,
$\operatorname{Cov}(\xi_1, \xi_1^2 - 1) = \E\xi_1^3$ en
$\V(\xi_1^2 - 1) = \E\xi_1^4 - 1$; het derde moment
$\E\norm{V_1}^3 \leq 4\bigl(\E\abs{\xi_1}^3 +
\E\abs{\xi_1^2 - 1}^3\bigr)$ is eindig als $\xi_1 \in
L^6$. Vraag 16 levert de getoonde gezamenlijke Gaussische
limiet, en \cref{thm:b3:clt:gaussianvector}(2): de twee
limietcoördinaten zijn \omterm{def:b3:probability:independence}{onafhankelijk} precies als de
covariantie $\E\xi_1^3$ verdwijnt --- voor symmetrische
wetten ontkoppelen empirisch gemiddelde en empirische
variantie asymptotisch.

\textbf{18.} Schrijf $g(x) - g(\theta) = (g'(\theta) +
\eta(x))(x - \theta)$ waar $\eta(x) = \frac{g(x) -
g(\theta)}{x - \theta} - g'(\theta)$ voor $x \neq \theta$
en $\eta(\theta) = 0$: differentieerbaarheid in $\theta$
betekent precies $\eta(x) \to 0$ als $x \to \theta$.
\emph{Stap 1:} $\hat\theta_n \to \theta$ in kans: voor
$\varepsilon > 0$ en elke $A > 0$, uiteindelijk
$\varepsilon\sqrt n \geq A$, dus $\P(\abs{\hat\theta_n -
\theta} > \varepsilon) \leq \P(\abs{\sqrt
n(\hat\theta_n - \theta)} > A) \to \P(\sigma\abs N > A)$
(verdelingsfuncties convergeren in de
continuïteitspunten $\pm A$), en de rechterkant gaat naar
$0$ als $A \to \infty$. \emph{Stap 2:}
$\eta(\hat\theta_n) \to 0$ in kans: gegeven $\varepsilon'
> 0$, kies $\delta$ met $\abs\eta \leq \varepsilon'$ op
$\abs{x - \theta} \leq \delta$; dan
$\P(\abs{\eta(\hat\theta_n)} > \varepsilon') \leq
\P(\abs{\hat\theta_n - \theta} > \delta) \to 0$.
\emph{Stap 3:}
\[
\sqrt n\bigl(g(\hat\theta_n) - g(\theta)\bigr) =
g'(\theta)\,\sqrt n(\hat\theta_n - \theta) +
\eta(\hat\theta_n)\cdot\sqrt n(\hat\theta_n - \theta) .
\]
Door de productregel van Slutsky (\cref{exo:b3:clt:8}, met
de rij $\eta(\hat\theta_n) \to 0$ in kans en de
in-wet-convergente $\sqrt n(\hat\theta_n - \theta)$)
convergeert de tweede term in wet naar $0\cdot\mathcal
N(0, \sigma^2) = 0$, dus naar $0$ in kans
(\cref{exo:b3:clt:4}(b)); de eerste convergeert in wet
naar $g'(\theta)\mathcal N(0, \sigma^2)$ (Slutsky opnieuw,
of de affiene regel voor \omterm{def:b3:clt:cf}{karakteristieke functies}); de
somregel van Slutsky assembleert ze: de limiet is
$\mathcal N(0, g'(\theta)^2\sigma^2)$.

\textbf{19.} (a) De CLT geeft $\sqrt n(\bar X_n - \mu)
\Rightarrow \mathcal N(0, \sigma^2)$; de deltamethode met
$g(x) = x^2$, $g'(\mu) = 2\mu$, geeft $\sqrt n(\bar
X_n^2 - \mu^2) \Rightarrow \mathcal N(0, 4\mu^2\sigma^2)$
--- ontaard (limiet $0$) als $\mu = 0$. In dat geval leeft
de fluctuatie één schaal hoger: $n\bar X_n^2 = (\sqrt
n\,\bar X_n)^2$, en voor $t > 0$
\[
\P\bigl(n\bar X_n^2 \leq t\bigr) = \P\bigl(-\sqrt t \leq
\sqrt n\,\bar X_n \leq \sqrt t\bigr) \longrightarrow
\Phi\Bigl(\frac{\sqrt t}\sigma\Bigr) -
\Phi\Bigl(-\frac{\sqrt t}\sigma\Bigr) = \P(\sigma^2N^2
\leq t) :
\]
$n\bar X_n^2 \Rightarrow \sigma^2N^2$, het kwadraat van
een Gaussische (een ``chi-kwadraat''-wet) --- als de
eerste afgeleide sterft, dicteert de tweede-ordeterm van
Taylor een niet-Gaussische limiet. (b) Hier $\sqrt n(\hat
p_n - p) \Rightarrow \mathcal N(0, p(1 - p))$ en $g(p) =
\arcsin\sqrt p$ heeft $g'(p) = \frac1{2\sqrt{p(1 - p)}}$,
dus $g'(p)^2\,p(1 - p) = \frac14$: de limiet is $\mathcal
N(0, \frac14)$ voor elke $p \in \intoo01$. Op de
$\arcsin$-schaal is de asymptotische $95\%$-foutbalk $\pm
\frac{0{,}98}{\sqrt n}$, vooraf bekend --- terwijl in
\cref{ex:b3:clt:confidence} de breedte de onbekende
$\sigma = \sqrt{p(1-p)}$ betrof, te worst-casen door
$\frac12$ of te schatten: de transformatie
\emph{stabiliseert} de variantie.

\textbf{20.} Zij $A^* = \{k : \mu(\{k\}) > \nu(\{k\})\}$
en $\Delta_k = \mu(\{k\}) - \nu(\{k\})$, dus
$\sum_k\Delta_k = 0$. Voor elke $A \subseteq \N$: $\mu(A)
- \nu(A) = \sum_{k\in A}\Delta_k \leq \sum_{k\in
A^*}\Delta_k$, met gelijkheid in $A = A^*$; en omdat de
positieve en negatieve delen van $(\Delta_k)$ gelijke
totale massa hebben, $\sum_{A^*}\Delta_k =
\frac12\sum_k\abs{\Delta_k}$. $\mu, \nu$ verwisselen
behandelt het teken: $d_{\mathrm{TV}}(\mu, \nu) =
\frac12\sum_k\abs{\Delta_k}$. Koppeling: voor elke $A$,
\[
\mu(A) - \nu(A) = \E\bigl[\mathbf 1_A(X) - \mathbf
1_A(Y)\bigr] = \E\bigl[(\mathbf 1_A(X) - \mathbf
1_A(Y))\,\mathbf 1_{X\neq Y}\bigr] \leq \P(X \neq Y),
\]
en neem het supremum over $A$.

\textbf{21.} De twee wetten belasten: $k = 0$: $1 - p$
versus $\eu^{-p}$, met $\eu^{-p} > 1 - p$; $k = 1$: $p$
versus $p\,\eu^{-p} < p$; $k \geq 2$: $0$ versus de
Poissonrest $1 - \eu^{-p} - p\eu^{-p} \geq 0$. Dus
\[
d_{\mathrm{TV}} = \tfrac12\bigl[(\eu^{-p} - 1 + p) + (p -
p\eu^{-p}) + (1 - \eu^{-p} - p\eu^{-p})\bigr] =
\tfrac12\bigl(2p - 2p\eu^{-p}\bigr) = p(1 - \eu^{-p}),
\]
en $1 - \eu^{-p} \leq p$ geeft de grens $p^2$.

\textbf{22.} Schrijf $H_{i-1} = W_i + X_i$ en $H_i = W_i +
Y_i$ met $W_i = \sum_{j<i}Y_j + \sum_{j>i}X_j$,
\omterm{def:b3:probability:independence}{onafhankelijk} van het paar $(X_i, Y_i)$ (coalities). Voor
$A \subseteq \N$, conditioneren op de aftelbaar veel
waarden door \omterm{def:b3:probability:independence}{onafhankelijkheid},
\[
\P(H_{i-1} \in A) = \sum_{k\geq0}\P(X_i = k)\,\P(W_i + k
\in A),
\]
en evenzo voor $H_i$ met $Y_i$. Aftrekken, met $c_k =
\P(W_i + k \in A) \in \intcc01$ en $\Delta_k = \P(X_i =
k) - \P(Y_i = k)$ van nulsom:
\[
\abs{\P(H_{i-1} \in A) - \P(H_i \in A)} =
\Bigl|\sum_k\Delta_k\bigl(c_k - \tfrac12\bigr)\Bigr| \leq
\tfrac12\sum_k\abs{\Delta_k} =
d_{\mathrm{TV}}\bigl(\mathcal B(1, p_i), \mathcal
P(p_i)\bigr) .
\]
Telescoperen van $H_0 = S$ naar $H_n = \sum_iY_i \sim
\mathcal P(\lambda)$ (\cref{exo:b3:clt:1}, geïtereerd) en
vraag 21 gebruiken:
\[
\abs{\P(S \in A) - \P(\mathcal P(\lambda) \in A)} \leq
\sum_{i=1}^np_i\bigl(1 - \eu^{-p_i}\bigr) \leq
\sum_{i=1}^np_i^2
\qquad\text{voor elke } A :
\]
de ongelijkheid van Le Cam. (De koppelingsgrens van vraag
20 geeft een alternatieve route: koppel elk paar op één
uniforme variabele zodat $\P(X_i \neq Y_i) \leq p_i^2$ en
begrens $\P(S \neq \sum Y_i)$; wisselen heeft helemaal
geen constructie nodig.)

\textbf{23.} (a) Met $p_i = \frac\lambda n$:
$d_{\mathrm{TV}}(\text{wet van }S, \mathcal P(\lambda))
\leq \frac{\lambda^2}n$. Dit verscherpt
\cref{exo:b3:clt:5} driemaal: een expliciete fout bij elke
eindige $n$, uniformiteit over alle gebeurtenissen $A$
tegelijk (niet één interval tegelijk), en geen noodzaak
voor gelijke $p_i$ --- alleen $\sum_ip_i^2$ klein, bv.\
$\sum p_i^2 \leq \lambda\max_ip_i$: \emph{veel zeldzame
gebeurtenissen, geen dominante}. (b) Hier $n = 500$, $p_i
= \frac1{500}$, $\lambda = 1$: het Poissonmodel faalt met
hoogstens $500\cdot\frac1{500^2} = 0{,}002$ op elke
gebeurtenis; in het bijzonder, met $A = \{0\}$,
\[
\P(\text{geen brief verdwaald}) = \Bigl(1 -
\frac1{500}\Bigr)^{500},
\qquad
\Bigl|\P(\text{geen brief verdwaald}) - \eu^{-1}\Bigr| \leq
0{,}002,
\]
dus het antwoord is $\eu^{-1} \approx 0{,}368$ tot op een
gegarandeerde $0{,}002$ (de ware discrepantie is ongeveer
$4\cdot10^{-4}$). (c) Het probleem sluit op één methode
met twee regimes. Als $n$ vergelijkbare bijdragen elk
variantie $\frac1n$ dragen, maakt matchen van \emph{twee}
momenten tegen de Gaussische de wisselfouten $o(\frac1n)$
elk: sommen worden \omterm{def:b3:clt:gaussianvector}{Gaussisch} --- met Taylor als lokaal
vergelijkingsinstrument. Als $n$ bijdragen indicatoren van
kans $p_i$ zijn, maakt matchen van het \emph{gemiddelde}
tegen een Poissonatoom elke wissel $p_i^2$ kosten: tellingen
van zeldzame gebeurtenissen worden Poisson --- met totale
variatie als exact lokaal vergelijkingsinstrument. Dezelfde
hybriden, dezelfde telescopering, andere lokale schatting:
vervanging is een strategie, geen stelling, en de
Gaussische en Poissonlimieten zijn haar twee oudste
dividenden.

\textbf{24.} De CLT geeft $\sqrt n(\bar X_n - \mu)
\Rightarrow \mathcal N(0, \sigma^2)$, en $g(x) = \ln x$ is
differentieerbaar in $\mu > 0$ met $g'(\mu) = \frac1\mu$:
de deltamethode (Deel VI) levert $\sqrt n(\ln\bar X_n -
\ln\mu) \Rightarrow \mathcal N(0, \sigma^2/\mu^2)$. Het
interval $\abs{\ln\bar X_n - \ln\mu} \leq
\frac{1{,}96\,\sigma}{\mu\sqrt n}$ uitrollen door
exponentiëren:
\[
\mu \in \bar X_n\cdot
\eu^{\pm1{,}96\,\sigma/(\mu\sqrt n)}
\qquad\text{met asymptotische kans } 95\%
\]
(in de praktijk wordt $\sigma/\mu$ door haar empirische
versie vervangen, Slutsky zoals in \cref{exo:b3:clt:8}).
Het multiplicatieve interval is het natuurlijke als de data
positief zijn met fouten evenredig aan hun grootte ---
inkomens, concentraties, halfwaardetijden: grootheden die
op een logschaal leven, waar symmetrische additieve
intervallen zelfs nul kunnen kruisen.

\textbf{25.} $\E[(X - p)^3] = (1-p)^3p + (-p)^3(1 - p) =
p(1-p)\bigl[(1-p)^2 - p^2\bigr] = p(1-p)(1 - 2p)$. In de
wisselanalyse (Deel IV) draagt de leidende foutterm na
matchen van twee momenten het \emph{getekende} derde
moment: voor $p < \frac12$ is het positief (de wet helt
naar rechts: zeldzame grote uitstapjes boven het
gemiddelde), en de normale benadering plaatst systematisch
massa verkeerd --- onderschat de korte linkerstaart en
overschat de rechter --- met een fout van orde $n^{-1/2}$;
in $p = \frac12$ verdwijnt het derde moment, de Bernoulli
matcht de Gaussische tot orde drie, en de ratio verbetert
(gematchte-momentenvraag van Deel IV). Numeriek: $\P(S =
0) = 0{,}9^{20} = 0{,}1216$, terwijl de Gaussische
$\mathcal N(2, 1{,}8)$ $\Phi\bigl(\frac{0{,}5 -
2}{\sqrt{1{,}8}}\bigr) = \Phi(-1{,}118) \approx 0{,}132$
geeft: de normale kromme, onwetend van de muur bij $0$ en
van de rechtse scheefheid, legt te veel massa onderaan ---
het voorspelde teken van de fout, zichtbaar bij $n = 20$.
