\section*{Hoofdstuk \ref{ch:b3:fouriertransform} --- De Fourier-
transformatie}

\begin{solution}{exo:b3:fouriertransform:1}
$\widehat{\mathbf 1_{\intcc{-a}a}}(\xi) =
\int_{-a}^a\eu^{-\iu\xi x}\dd x = \frac{2\sin(a\xi)}{\xi}$
(waarde $2a$ in $0$).
$\widehat{\eu^{-a\abs x}}(\xi) = \int_0^\infty\eu^{-(a +
\iu\xi)x} + \eu^{-(a - \iu\xi)x}\,\dd x = \frac1{a + \iu\xi} +
\frac1{a - \iu\xi} = \frac{2a}{a^2 + \xi^2}$.
Tent: $\max(0, 1 - \abs x) = \mathbf 1_{\intcc{-1/2}{1/2}} *
\mathbf 1_{\intcc{-1/2}{1/2}}$, dus haar transform is
$\bigl(\frac{2\sin(\xi/2)}\xi\bigr)^2 =
\bigl(\frac{\sin(\xi/2)}{\xi/2}\bigr)^2$.
Laatste: $\frac{2a}{a^2+\xi^2} \in L^1$, dus inversie
(\cref{thm:b3:fouriertransform:inversion}(2)) toegepast op
$\eu^{-a\abs x}$ geeft, na hernoemen van variabelen,
\[
\widehat{\Bigl(\frac1{x^2 + a^2}\Bigr)}(\xi) =
\frac{\pi}{a}\,\eu^{-a\abs\xi} .
\]
\end{solution}

\begin{solution}{exo:b3:fouriertransform:2}
Uit \cref{prop:b3:fouriertransform:basic}:
$\widehat{f(\cdot - a)} = \eu^{-\iu a\xi}\hat f(\xi)$;
$\widehat{f\cos(b\cdot)} = \frac12\bigl(\hat f(\xi - b) + \hat
f(\xi + b)\bigr)$;
$\widehat{f(a\cdot + b)}(\xi) = \frac1a\,\eu^{\iu
b\xi/a}\,\hat f(\xi/a)$ ($a > 0$);
$\widehat{\overline{f(-\cdot)}} = \overline{\hat f}$;
$\widehat{f * f} = \hat f^2$. Op de Gauss
($\widehat{\eu^{-x^2}} = \sqrt\pi\eu^{-\xi^2/4}$) is elke
regel een eenregelige controle --- b.v.\ $\eu^{-(x-a)^2}$
heeft transform $\sqrt\pi\,\eu^{-\iu a\xi}\eu^{-\xi^2/4}$,
die de directe berekening (maak het kwadraat \omterm{def:b3:complete:complete}{volledig})
bevestigt.
\end{solution}

\begin{solution}{exo:b3:fouriertransform:3}
(a) $h = \mathbf 1_{\intcc{-1}1}*\mathbf 1_{\intcc{-1}1}$
heeft $\hat h = \bigl(\frac{2\sin\xi}\xi\bigr)^2 \in L^1$;
inversie in $x = 0$, waar $h(0) = \lambda(\intcc{-1}1\cap\intcc{-1}1)
= 2$:
\[
2 = \frac1{2\pi}\int_\R\Bigl(\frac{2\sin\xi}\xi\Bigr)^2
\dd\xi
\ \Longrightarrow\
\int_\R\Bigl(\frac{\sin\xi}\xi\Bigr)^2\dd\xi = \pi ,
\]
consistent met $\int_0^\infty\frac{\sin^2}{\xi^2} =
\frac\pi2$ (\cref{pb:b3:lebesgue:1}).

(b) Plancherel voor $f = \eu^{-\abs x}$: $\int\abs{\hat f}^2 =
2\pi\int\abs f^2$ leest $\int\frac{4\,\dd\xi}{(1 + \xi^2)^2} =
2\pi\int\eu^{-2\abs x}\dd x = 2\pi$:
$\int_\R\frac{\dd\xi}{(1+\xi^2)^2} = \frac\pi2$.
\end{solution}

\begin{solution}{exo:b3:fouriertransform:4}
(a) $g_t(x) = \frac1{2\sqrt{\pi t}}\eu^{-x^2/4t}$: via
\cref{ex:b3:fouriertransform:gaussian} met $a = \frac1{4t}$,
$\hat g_t(\xi) = \frac1{2\sqrt{\pi t}}\sqrt{4\pi
t}\,\eu^{-t\xi^2} = \eu^{-t\xi^2}$.
(b) $\widehat{g_t * g_s} = \eu^{-t\xi^2}\eu^{-s\xi^2} =
\widehat{g_{t+s}}$, en de transform is injectief op $L^1$
(\cref{thm:b3:fouriertransform:inversion}(3)): $g_t * g_s =
g_{t+s}$.
(c) $\norm{g_t}_1 = 1$ (Gauss-integraal);
$\norm{g_t}_2^2 = \frac1{4\pi t}\int\eu^{-x^2/2t}\dd x =
\frac{\sqrt{2\pi t}}{4\pi t} = \frac1{\sqrt{8\pi t}}$.
\end{solution}

\begin{solution}{exo:b3:fouriertransform:5}
Voor reële even $f$: $\hat f(\xi) = \int f\cos(\xi x)\dd x$
(het sinusdeel cancelleert): reëel en even. Oneven: $\hat
f(\xi) = -\iu\int f\sin(\xi x)$: oneven, zuiver imaginair.
$\hat f(0) = \int f$: de totale massa. Als $f \geq 0$:
$\abs{\hat f(\xi)} \leq \int\abs f = \int f = \hat f(0)$,
dus het sup wordt bereikt in $0$. Voor een kansdichtheid is
$\hat f(-\xi)$ de karakteristieke functie van
\cref{ch:b3:clt}: modulus $\leq 1$ overal, $= 1$ in de
oorsprong.
\end{solution}

\begin{solution}{exo:b3:fouriertransform:6}
(i) Injectiviteit is
\cref{thm:b3:fouriertransform:inversion}(3); continuïteit is
$\norm{\hat f}_\infty \leq \norm f_1$ (met waarden in
$\mathcal C_0$ via Riemann--Lebesgue); $\mathcal C_0$ is
gesloten in de sup-norm (uniforme limieten van
verdwijnende-in-oneindig-functies verdwijnen in oneindig):
Banach.

(ii) Een continue bijectie tussen Banachruimten heeft
continue inverse (\cref{thm:b3:banach:openmapping}): er zou
$C$ zijn met $\norm f_1 \leq C\norm{\hat f}_\infty$.

(iii) Zij $f_n(x) = \frac{\sin x}x\cdot\frac{\sin(x/n)}{x/n}$:
een product van twee $L^2$-functies, en $O(x^{-2})$ in
oneindig, dus $f_n \in L^1\cap L^2$. Omdat
$\bigl(\frac{\sin(ax)}{ax}\bigr)$ $L^2$-transform $\frac\pi
a\mathbf 1_{\intcc{-a}a}$ heeft, geeft de productformule
$\widehat{gh} = \frac1{2\pi}\hat g * \hat h$ (geldig voor
$g, h \in L^2$ met $gh \in L^1$; controleer op
Schwartz-functies via Fubini en strek uit via
$L^2$-continuïteit van beide zijden via Plancherel)
\[
\hat f_n = \frac1{2\pi}\,\bigl(\pi\mathbf
1_{\intcc{-1}1}\bigr) * \bigl(\pi n\,\mathbf
1_{\intcc{-1/n}{1/n}}\bigr):
\]
een trapezium van hoogte $\frac{\pi n}2\cdot\frac2n = \pi$:
$\norm{\hat f_n}_\infty = \pi$ voor elke $n$. Maar op $[1,
n]$, $\frac{\sin(x/n)}{x/n} \geq \sin 1 > 0$, dus
\[
\norm{f_n}_1 \geq \sin 1\int_1^n\frac{\abs{\sin x}}x\dd x
\geq c\ln n
\]
(bogentelling, zoals in \cref{thm:b3:banach:fourierdiverge}).
De grens $\norm{f_n}_1 \leq C\pi$ faalt voor grote $n$: niet
surjectief. (Het beeld is een dichte --- via
Stone--Weierstrass-type argumenten --- maar echte deelruimte
van $\mathcal C_0$.)
\end{solution}

\begin{solution}{exo:b3:fouriertransform:7}
Als $\hat f(\xi) = O(\abs\xi^{-k-1-\delta})$: dan $\xi^j\hat f
\in L^1$ voor $0 \leq j \leq k$ (\omterm{def:b3:lebesgue:l1}{integreerbaar} in oneindig
via het verval, lokaal via continuïteit van $\hat f$).
Inversie (\cref{thm:b3:fouriertransform:inversion}(2))
representeert $f$ a.e.\ door $x \mapsto \frac1{2\pi}\int\hat
f(\xi)\eu^{\iu x\xi}\dd\xi$, en differentiatie onder de
integraal (dominators $\abs{\xi^j\hat f}$) maakt deze
representant $\mathcal C^k$. Omgekeerd voor $f \in \mathcal
C_c^k$: itereren van \cref{prop:b3:fouriertransform:basic}(4),
$(\iu\xi)^k\hat f = \widehat{f^{(k)}}$, dus $\abs{\hat f}
\leq \norm{f^{(k)}}_1\abs\xi^{-k}$. Tentfunctie: continu met
compacte drager ($k = 0$: transform begrensd), en haar
transform $\sim \xi^{-2} = O(\abs\xi^{-0-1-1})$ geeft, via de
eerste richting, een $\mathcal C^0$-representant --- beide
scherp: de tent is niet $\mathcal C^1$, en haar transform
vervalt niet sneller dan $\xi^{-2}$.
\end{solution}

\begin{solution}{exo:b3:fouriertransform:8}
Partieel integreren ($f \in \mathcal S$; randtermen
verdwijnen):
\[
1 = \int f^2 = \bigl[xf^2\bigr]_{-\infty}^{\infty} - \int
x\,(f^2)' = -2\int xff' \leq 2\,\norm{xf}_2\,\norm{f'}_2 .
\]
Plancherel en $\widehat{f'} = \iu\xi\hat f$:
$\norm{f'}_2^2 = \frac1{2\pi}\int\xi^2\abs{\hat f}^2$.
Kwadrateer de weergave:
\[
\frac14 \leq \norm{xf}_2^2\cdot\frac1{2\pi}
\int\xi^2\abs{\hat f}^2\dd\xi .
\]
Gelijkheid vereist gelijkheid in Cauchy--Schwarz: $f' =
\lambda xf$ met $\lambda < 0$ (integreerbaarheid), d.w.z.\
$f(x) = c\,\eu^{\lambda x^2/2}$: Gaussen. Een signaal
geconcentreerd in $x$ (kleine $\norm{xf}_2$) moet
uitgespreid spectrum hebben, en omgekeerd: het
onzekerheidsprincipe.
\end{solution}

\begin{solution}{exo:b3:fouriertransform:9}
De Gauss $f(x) = \eu^{-\pi tx^2}$ is Schwartz, dus
\cref{thm:b3:fouriertransform:poisson} geldt, en
$\hat f(\xi) = t^{-1/2}\eu^{-\xi^2/4\pi t}$; in $\xi = 2\pi
k$ wordt de rechterkant $t^{-1/2}\eu^{-\pi k^2/t}$: de
theta-identiteit. Bij $t = 1$:
\[
\sum_{n\in\Z}\eu^{-\pi n^2} = 1 + 2\eu^{-\pi} + 2\eu^{-4\pi} +
\cdots \approx 1 + 0.0864278 + 0.0000070 = 1.0864348,
\]
nauwkeurig tot $6$ decimalen met drie termen ($\eu^{-9\pi}
\approx 5\cdot10^{-13}$). Bij $t = 10^{-2}$: de
definiërende reeks heeft $\eu^{-\pi n^2/100} < 10^{-7}$
nodig, d.w.z.\ $n \gtrsim 23$ --- ongeveer $47$ termen ---
terwijl de getransformeerde reeks $10\sum_k\eu^{-100\pi
k^2}$ is, waar al de $k = 1$-term $\sim 10^{-136}$ is: één
term volstaat.
\end{solution}

\begin{solution}{exo:b3:fouriertransform:10}
$\mathcal Ff \in L^2(\intcc{-\pi}\pi) \subseteq
L^1(\intcc{-\pi}\pi)$ (eindige \omterm{def:b3:measure:measure}{maat}), dus inversie geeft de
continue representant $f(x) =
\frac1{2\pi}\int_{-\pi}^\pi\mathcal Ff(\xi)\eu^{\iu
x\xi}\dd\xi$, met
\[
f(n) = \frac1{2\pi}\int_{-\pi}^{\pi}\mathcal
Ff(\xi)\,\eu^{\iu n\xi}\dd\xi = \langle e_{-n}, \mathcal
Ff\rangle
\]
in de notatie van \cref{thm:b3:hilbert:fourier}. Ontwikkelen
in die \omterm{def:b3:hilbert:onb}{Hilbert-basis}: $\mathcal Ff = \sum_nf(n)\,\eu^{-\iu
n\xi}$ in $L^2(\intcc{-\pi}\pi)$. Pas de $L^2$-continue
$\mathcal F^{-1}$ term-voor-term toe:
\[
\mathcal F^{-1}\bigl(\mathbf
1_{\intcc{-\pi}\pi}\eu^{-\iu n\xi}\bigr)(x)
= \frac1{2\pi}\int_{-\pi}^{\pi}\eu^{\iu\xi(x - n)}\dd\xi
= \frac{\sin\bigl(\pi(x-n)\bigr)}{\pi(x - n)} ,
\]
gevend $f = \sum_nf(n)\operatorname{sinc}(\cdot - n)$ in
$L^2$: een bandbegrensd signaal is bepaald door haar
geheeltallige samples --- Shannons samplingstelling.
\end{solution}

\begin{solution}{exo:b3:fouriertransform:11}
(a) Inversie op $\mathcal S$: $f(x) =
\frac1{2\pi}\int\hat f(\xi)\eu^{\iu x\xi}\dd\xi =
\frac1{2\pi}(\mathcal F\hat f)(-x)$, d.w.z.\ $(\mathcal
F^2f)(x) = 2\pi f(-x)$. Tweemaal toepassen: $\mathcal F^4f =
2\pi\,\mathcal F^2f(-\cdot) = (2\pi)^2f$.

(b) Als $\mathcal Ff = \lambda f$ met $f \neq 0$:
$(2\pi)^2f = \mathcal F^4f = \lambda^4f$, dus $\lambda^4 =
(2\pi)^2$: $\lambda \in \{\pm\sqrt{2\pi},
\pm\iu\sqrt{2\pi}\}$. De Gauss $g(x) = \eu^{-x^2/2}$ heeft
$\hat g = \sqrt{2\pi}\,g$
(\cref{ex:b3:fouriertransform:gaussian} bij $a = \frac12$):
eigenfunctie voor $+\sqrt{2\pi}$.

(c) $\mathcal F^2f = 2\pi f(-\cdot)$ is gelijk aan $\pm2\pi
f$ volgens pariteit. Voor $h(x) = x\eu^{-x^2/2}$:
differentiëren van $\hat g(\xi) = \sqrt{2\pi}\eu^{-\xi^2/2}$
met de regel $\widehat{xf} = \iu\frac{\dd}{\dd\xi}\hat f$:
\[
\hat h(\xi) = \iu\,\frac{\dd}{\dd\xi}\bigl(\sqrt{2\pi}
\eu^{-\xi^2/2}\bigr) = -\iu\sqrt{2\pi}\,\xi\eu^{-\xi^2/2}
= -\iu\sqrt{2\pi}\,h(\xi) :
\]
een eigenfunctie voor $-\iu\sqrt{2\pi}$. (De
Hermite-functies zetten het patroon voort, cyclerend door
de vier eigenwaarden --- de discrete Fourier-klok.)
\end{solution}

\begin{solution}{exo:b3:fouriertransform:12}
(a) $\tilde f \in L^2$ met $\norm{\tilde f}_2 = \norm f_2$;
\cref{exo:b3:lp:6} (geconjugeerde exponenten $p = q = 2$)
maakt $A_f = f * \tilde f$ begrensd en uniform continu, met
\[
A_f(x) = \int f(y)\,\overline{f(y - x)}\,\dd y,
\qquad A_f(0) = \norm f_2^2,
\qquad \abs{A_f(x)} \leq \norm f_2\,\norm{f(\cdot -
x)}_2 = A_f(0)
\]
via Cauchy--Schwarz.

(b) Voor $f \in L^1\cap L^2$: $\tilde f \in L^1$ ook, en de
convolutiestelling geeft $\widehat{A_f} = \hat f\,
\widehat{\tilde f}$; berekenend, $\widehat{\tilde f}(\xi) =
\int\overline{f(-x)}\eu^{-\iu\xi x}\dd x =
\overline{\int f(u)\eu^{-\iu\xi u}\dd u} =
\overline{\hat f(\xi)}$: $\widehat{A_f} = \abs{\hat f}^2
\geq 0$.

(c) Wanneer $\abs{\hat f}^2 \in L^1$, geldt inversie voor de
continue $A_f$ (haar transform is \omterm{def:b3:lebesgue:l1}{integreerbaar};
\cref{thm:b3:fouriertransform:inversion}):
\[
A_f(x) = \frac1{2\pi}\int\abs{\hat f(\xi)}^2
\eu^{\iu x\xi}\,\dd\xi .
\]
Voor $f = \mathbf 1_{\intcc{-1/2}{1/2}}$: $\hat f(\xi) =
\frac{2\sin(\xi/2)}\xi = \frac{\sin(\xi/2)}{\xi/2}$, en in
$x = 0$:
\[
1 = \norm f_2^2 = \frac1{2\pi}\int_\R
\Bigl(\frac{\sin(\xi/2)}{\xi/2}\Bigr)^2\dd\xi
= \frac1{2\pi}\cdot2\int_\R\Bigl(\frac{\sin u}u\Bigr)^2\dd u
\]
($\xi = 2u$), d.w.z.\ $\int_\R\bigl(\frac{\sin u}u\bigr)^2\dd
u = \pi$ --- Plancherels favoriete integraal, herwonnen via
autocorrelatie.
\end{solution}

\begin{solution}{pb:b3:fouriertransform:1}
\textbf{1.} Transformeren van de vergelijking in $x$
(formeel): $\partial_t\hat u(t,\xi) =
\widehat{\partial^2_{xx}u} = (\iu\xi)^2\hat u = -\xi^2\hat
u$, een ODE in $t$ voor elke frequentie: $\hat u(t,\xi) =
\eu^{-t\xi^2}\hat f(\xi)$. Omdat $\eu^{-t\xi^2} = \hat
g_t$ (\cref{exo:b3:fouriertransform:4}), is het product de
transform van $g_t * f$.

\textbf{2.} $\abs{u(t,x)} \leq \norm f_\infty\int g_t =
\norm f_\infty$: welgedefinieerd. Op $[t_0, T]\times[-A,
A]$: elke gemengde afgeleide $\partial^m_t\partial^n_xg_t(x
- y)$ is een polynoom in $(x - y)$ en $t^{-1}$ maal
$\eu^{-(x-y)^2/4t}$, begrensd voor $\abs y \geq 2A$ door
$C\,(1 + y^2)^N\eu^{-(\abs y - A)^2/4T}$, een integreerbare
dominator onafhankelijk van $(t, x)$ in het venster (en
begrensd voor $\abs y \leq 2A$): herhaalde differentiatie
onder de integraal (\cref{thm:b3:lebesgue:paramdiff}) geldt:
$u \in \mathcal C^\infty(\intoo0\infty\times\R)$.

\textbf{3.} Met $g_t(x) = \frac1{2\sqrt{\pi
t}}\eu^{-x^2/4t}$:
\[
\partial_tg_t = g_t\Bigl(\frac{x^2}{4t^2} - \frac1{2t}\Bigr)
= \partial^2_{xx}g_t
\]
(differentiëer tweemaal in $x$: $\partial_xg_t = -\frac
x{2t}g_t$, $\partial^2_{xx}g_t = \bigl(\frac{x^2}{4t^2} -
\frac1{2t}\bigr)g_t$). Via vraag 2 gaan de afgeleiden onder
de integraal: $\partial_tu = \partial^2_{xx}u$.

\textbf{4.} $u(t,x) - f(x) = \int g_t(y)\bigl(f(x - y) -
f(x)\bigr)\dd y$. Gegeven een compact $K$ en $\varepsilon$:
uniforme continuïteit van $f$ op een \omterm{def:b3:topology:topology}{omgeving} van $K$ geeft
$\delta$ met $\abs{f(x-y) - f(x)} < \varepsilon$ voor $x \in
K$, $\abs y \leq \delta$; de staart draagt $\leq 2\norm
f_\infty\int_{\abs y > \delta}g_t(y)\dd y = 2\norm
f_\infty\,\P$-massa voorbij $\delta$, wat
$\frac2{\sqrt\pi}\int_{\delta/2\sqrt t}^\infty\eu^{-z^2}\dd z
\to 0$ is als $t \to 0$. Voor $f \in L^p$: $\norm{u(t) -
f}_p \leq \int g_t(y)\norm{\tau_yf - f}_p\dd y$
(Minkowski/Jensen zoals in
\cref{thm:b3:lp:regularization}), splits op dezelfde manier
met \cref{thm:b3:lp:density}(3).

\textbf{5.} Instant gladmaking is vraag 2 ($u(t)$ is
$\mathcal C^\infty$ voor $t > 0$ zonder gladheid van $f$ te
gebruiken). Voor $f = \mathbf 1_{\intoo0\infty}$:
\[
u(t, x) = \int_0^\infty g_t(x - y)\dd y
= \frac1{\sqrt\pi}\int_{-x/2\sqrt t}^{\infty}\eu^{-z^2}\dd z
= \frac12\Bigl(1 +
\operatorname{erf}\Bigl(\frac{x}{2\sqrt t}\Bigr)\Bigr),
\qquad \operatorname{erf}(s) =
\frac2{\sqrt\pi}\int_0^s\eu^{-z^2}\dd z :
\]
een gladgemaakte stap waarvan de overgangszone als $\sqrt t$
verbreedt (profielen bij $t_1 < t_2 < t_3$: steeds
vlakkere hellingen door $(0, \frac12)$).

\textbf{6.} De integrand $g_t(x-y)f(y)$ is $\geq 0$ en de
kern is strikt positief: $u(t,x) = 0$ zou $f = 0$ a.e.\
forceren. Monotonie in $f$ is monotonie van de integraal.
Een plek waar $f = 0$ op een interval heeft nog steeds $u(t,
\cdot) > 0$ daar voor elke $t > 0$: warmte propageert met
oneindige snelheid (elke positiviteit ergens wordt overal
onmiddellijk gevoeld).

\textbf{7.} Tonelli ($g_t(x-y)\abs{f(y)}$ is \omterm{def:b3:lebesgue:l1}{integreerbaar} op
$\R^2$): $\int u(t,x)\dd x = \int f(y)\bigl(\int g_t(x -
y)\dd x\bigr)\dd y = \int f$.

\textbf{8.} Plancherel: $2\pi\norm{u(t)}_2^2 =
\int\eu^{-2t\xi^2}\abs{\hat f(\xi)}^2\dd\xi$, niet-stijgend
in $t$ (puntsgewijs), strikt tenzij $\hat f = 0$ a.e.\ ($=
f = 0$), met limiet $0$ als $t\to\infty$ via DCT. En
$\norm{u(t)}_\infty \leq \norm{g_t}_\infty\norm f_1 =
\frac{\norm f_1}{2\sqrt{\pi t}} \to 0$.

\textbf{9.} Voor $\varphi \in \mathcal C_c^\infty$, $t
\mapsto \langle\varphi, \hat u(t)\rangle$ is $\mathcal C^1$
met afgeleide $\langle\varphi, \partial_t\hat u\rangle =
\langle\varphi, \widehat{\partial_{xx}u}\rangle =
\langle\xi^2\varphi\dots\rangle$ --- precies,
$\widehat{\partial^2_{xx}u} = -\xi^2\hat u$ transporteert
de vergelijking. Dan voor a.e.\ $\xi$ heeft de absoluut
continue functie $t \mapsto \eu^{t\xi^2}\hat u(t,\xi)$
afgeleide $\eu^{t\xi^2}(\xi^2\hat u + \partial_t\hat u) =
0$ in de geïntegreerde zin: ze is constant, en $t \to 0$
latend ($\hat u(t) \to \hat f$ in $L^2$, a.e.\ langs een
deelrij): $\hat u(t, \xi) = \eu^{-t\xi^2}\hat f(\xi)$ a.e.
Twee oplossingen in de klasse hebben dezelfde transform:
ze zijn gelijk.

\textbf{10.} $f = g_s * h$ met $h \in L^2$ forceert $\hat f
= \eu^{-s\xi^2}\hat h$, d.w.z.\ $\hat h = \eu^{s\xi^2}\hat
f \in L^2$. Neem $\hat f(\xi) = \eu^{-\abs\xi}$: dan $f(x)
= \frac1\pi\cdot\frac1{1 + x^2}$
(\cref{exo:b3:fouriertransform:1}, inversie), een \omterm{prop:b3:galois:perfect}{perfect}
gladde $L^2$-functie; maar $\eu^{2s\xi^2 - 2\abs\xi} \to
\infty$: $\eu^{s\xi^2}\hat f \notin L^2$ voor elke $s >
0$. Het Cauchy-profiel is \emph{nooit} het resultaat van
eerdere diffusie.

\textbf{11.} De warmtesemigroep vermenigvuldigt transforms
met $\eu^{-t\xi^2}$, die nergens verdwijnt: injectief ---
formeel wordt geen informatie vernietigd. Maar haar bereik
bestaat uit functies wier transforms als $\eu^{-t\xi^2}$
vervallen: een piepkleine, dichte-maar-echte deelruimte van
$L^2$ (vraag 10 toont dat zelfs uitstekende functies
erbuiten liggen). Inverteren zou de frequentie $\xi$
versterken met $\eu^{t\xi^2}$: onbegrensd, dus instabiel
tegen elke verstoring. Diffusie is onomkeerbaar niet omdat
de afbeelding vergeet, maar omdat haar inverse niet
continu kan zijn --- een pijl van de tijd gemaakt van
functionaalanalyse.

\textbf{12.} $\hat f \in L^2(\intcc{-\Omega}\Omega)
\subseteq L^1$ (Cauchy--Schwarz op een begrensd interval),
dus $F(x) = \frac1{2\pi}\int_{-\Omega}^\Omega\hat
f(\xi)\eu^{\iu x\xi}\dd\xi$ is overal gedefinieerd, en
differentiatie onder de integraal (gedomineerd door
$\Omega^k\abs{\hat f} \in L^1$ op de band) maakt ze
$\mathcal C^\infty$ met $\abs{F^{(k)}} \leq
\frac{\Omega^k}{2\pi}\norm{\hat f}_{L^1}$ overal. En $F =
f$ a.e.: beide zijden hebben dezelfde transform, en de
transform is injectief op $L^2$
(\cref{thm:b3:fouriertransform:plancherel} en haar
$L^2$-uitbreiding).

\textbf{13.} De exponentialen $\xi \mapsto \eu^{-\iu
n\pi\xi/\Omega}$, $n \in \Z$, vormen een \omterm{def:b3:hilbert:onb}{Hilbert-basis} van
$L^2(\intcc{-\Omega}\Omega)$
(\cref{thm:b3:hilbert:fourier}, herschaald). De
coëfficiënt van $\hat f$ langs de $n$-de is
\[
\frac1{2\Omega}\int_{-\Omega}^\Omega\hat f(\xi)\,
\eu^{\iu n\pi\xi/\Omega}\dd\xi
= \frac{2\pi}{2\Omega}\cdot
\frac1{2\pi}\int_{-\Omega}^{\Omega}\hat f(\xi)\,
\eu^{\iu(n\pi/\Omega)\xi}\dd\xi
= \frac\pi\Omega\,f\Bigl(\frac{n\pi}\Omega\Bigr),
\]
via de formule van vraag 12 in $x = \frac{n\pi}\Omega$: de
gestelde ontwikkeling geldt in $L^2$ van de band.

\textbf{14.} Injecteer de ontwikkeling in de
inversieformule van vraag 12; de verwisseling van som en
integraal is de continuïteit van de $L^2$-paring tegen
$\frac1{2\pi} \eu^{\iu x\xi}\mathbf 1_{\abs\xi\leq\Omega}$
(van $L^2$-norm $\frac{\sqrt{2\Omega}}{2\pi}$,
onafhankelijk van $x$ --- vandaar uniformiteit):
\[
f(x) = \sum_nf\Bigl(\frac{n\pi}\Omega\Bigr)\cdot
\frac1{2\Omega}\int_{-\Omega}^\Omega
\eu^{\iu(x - n\pi/\Omega)\xi}\dd\xi
= \sum_nf\Bigl(\frac{n\pi}\Omega\Bigr)
\operatorname{sinc}(\Omega x - n\pi),
\]
omdat $\frac1{2\Omega}\int_{-\Omega}^\Omega\eu^{\iu
u\xi}\dd\xi = \frac{\sin(\Omega u)}{\Omega u}$.

\textbf{15.} De berekening van vraag 14 achterstevoren
lezend, is de transform van $s_n =
\operatorname{sinc}(\Omega\cdot - n\pi)$ $\hat s_n =
\frac\pi\Omega\,\eu^{-\iu n\pi\xi/\Omega}\,\mathbf
1_{\intcc{-\Omega}\Omega}$. Plancherel:
\[
\langle s_n, s_m\rangle = \frac1{2\pi}
\Bigl(\frac\pi\Omega\Bigr)^2\int_{-\Omega}^\Omega
\eu^{\iu(n-m)\pi\xi/\Omega}\dd\xi =
\frac\pi\Omega\,\delta_{nm} :
\]
een orthogonale familie van constante norm
$\sqrt{\pi/\Omega}$. Normen nemen in de ontwikkeling van
vraag 14: $\norm f_2^2 =
\frac\pi\Omega\sum_n\abs{f(n\pi/\Omega)}^2$.

\textbf{16.} $g(x) = \sin(\Omega x)\operatorname{sinc}
(\Omega x) = \frac{\sin^2(\Omega x)}{\Omega x}$ verdwijnt
in elk roosterpunt $\frac{n\pi}\Omega$ (inclusief $0$, via
de limiet) en is niet identiek nul. Haar band: schrijf $g =
\frac1{2\iu}\bigl(\eu^{\iu\Omega x} - \eu^{-\iu\Omega
x}\bigr)\operatorname{sinc}(\Omega x)$; modulatie door
$\eu^{\pm\iu\Omega x}$ verschuift de transform met
$\mp\Omega$, dus $\hat g$ is gedragen in
$\intcc{-2\Omega}{2\Omega}$ (inderdaad in de unie van twee
verschoven banden): $g \in PW_{2\Omega}$, onzichtbaar voor
$\Omega$-snelheidssampling --- aliasing in levenden lijve.

\textbf{17.} Via vraag 15 dragen de samples de energie
democratisch: $\norm f^2 = \frac\pi\Omega\sum
\abs{f(n\pi/\Omega)}^2$. Als de energie van het signaal
buiten het tijdsvenster $\intcc{-T/2}{T/2}$ $\leq
\varepsilon^2\norm f^2$ is, voldoen de samples buiten het
venster (op randtermen na gecontroleerd door de uniforme
grens van vraag 12) $\frac\pi\Omega\sum_{\abs{n\pi/\Omega}
> T/2} \abs{f(n\pi/\Omega)}^2 \approx \norm{f\,\mathbf
1_{\abs x > T/2}}^2 \leq \varepsilon^2\norm f^2$: de
samplingreeks inkorten tot de $\approx \frac{\Omega T}\pi$
in-venster indices reconstrueert $f$ tot relatieve fout
$\approx\varepsilon$. Vandaar telt het tijd--bandbreedte-
product $\frac{\Omega T}{\pi}$ de effectieve reële
vrijheidsgraden van het signaal --- de regel achter elk
audioformaat.

\textbf{18.} (a) $\operatorname{sinc}(\Omega x)$ heeft
samples $f(n\pi/\Omega) = \operatorname{sinc}(n\pi) =
\delta_{n0}$: de reeks reduceert tot haar $n = 0$-term,
$\operatorname{sinc}(\Omega x)$ --- de stelling
reproduceert haar eigen kern. (b) Als $\hat f$ gedragen is
in $\intcc{-\Omega'}{\Omega'} \subseteq
\intcc{-\Omega}\Omega$, loopt elke stap van vragen 13--14
woordelijk met de grotere band $\Omega$ (de ontwikkeling
van $\hat f$ op het grotere interval is nog steeds
legitiem): sneller samplen dan je eigen Nyquist-snelheid
verandert niets in de reconstructie --- oversampling is
onschuldig, en in de praktijk nuttig
(sneller-vervallende reconstructiekernen kunnen dan
gebruikt worden).

\textbf{19.} Ontwikkel $\eu^{-\iu\xi x} =
\sum_k\frac{(-\iu\xi x)^k}{k!}$ binnen de integraal; op
$\intcc{-A}A$ convergeert de reeks normaal
($\sum_k\frac{\abs{\xi}^kA^k}{k!}\abs f \in L^1$), dus
term-voor-term-integratie is legitiem:
\[
\hat f(\xi) = \sum_{k\geq0}\frac{(-\iu\xi)^k}{k!}m_k,
\qquad \abs{m_k} \leq A^k\norm f_1 .
\]
De grens maakt de reeks convergent voor elke complexe
$\xi$; rond elk punt $\xi_0$ geeft hergroeperen (absolute
convergentie) een machtreeks in $\xi - \xi_0$: $\hat f$ is
reëel-analytisch met oneindige straal overal.

\textbf{20.} Zij $g$ reëel-analytisch op $\R$
(Taylorreeks convergerend naar $g$ nabij elk punt) en $Z =
\{\xi : g^{(k)}(\xi) = 0\ \forall k\}$. $Z$ is gesloten
(doorsnede van gesloten verzamelingen); ze is open, omdat
in $\xi_0 \in Z$ de lokale Taylorontwikkeling van $g$ de
nulreeks is, dus $g$ verdwijnt identiek nabij $\xi_0$,
samen met alle afgeleiden. Als $g$ verdwijnt op een
interval, $Z \neq \varnothing$; via samenhang van $\R$, $Z
= \R$: $g \equiv 0$. Nu als $f \neq 0$ compacte drager
samen met $\hat f$ had: vraag 19 maakt $\hat f$
reëel-analytisch, verdwijnend buiten een compact, dus op
intervallen: $\hat f \equiv 0$, dus $f = 0$ a.e.\ via
injectiviteit --- tegenspraak. Evenzo kan een niet-nul $f
\in PW_\Omega$ niet compact gedragen zijn (verwissel de
rollen van $f$ en $\hat f$ via inversie): bandbegrensde
signalen sterven nooit, tijdsbegrensde signalen bezetten
onbegrensd spectrum.

\textbf{21.} Voor $f = \eu^{-ax^2}$: $\norm f_2^2 =
\sqrt{\frac\pi{2a}}$ en $\int x^2\abs f^2 =
\frac1{4a}\sqrt{\frac{\pi}{2a}}$ (Gauss-tweede moment);
$\hat f = \sqrt{\frac\pi a}\,\eu^{-\xi^2/4a}$
(\cref{ex:b3:fouriertransform:gaussian}) en
\[
\frac1{2\pi}\int\xi^2\abs{\hat f}^2\dd\xi =
\frac1{2\pi}\cdot\frac\pi a\int\xi^2
\eu^{-\xi^2/2a}\dd\xi = \frac1{2a}\cdot a\sqrt{2\pi a}
= \frac{\sqrt{2\pi a}}2 .
\]
Genormaliseerd product: $\frac1{4a}\sqrt{\frac\pi{2a}}\cdot
\frac{\sqrt{2\pi a}}2\big/\frac{\pi}{2a} = \frac14$,
onafhankelijk van $a$. Schaling verklaart de constantheid:
$f$ vervangen door $f(\lambda\cdot)$ vermenigvuldigt $\int
x^2\abs f^2/\norm f^2$ met $\lambda^{-2}$ en
$\frac1{2\pi}\int\xi^2\abs{\hat f}^2/\norm f^2$ met
$\lambda^{2}$: het product is dilatie-invariant, en de
Gaussen vormen één dilatiebaan.

\textbf{22.} Met $\abs f^2$ de positiesdichtheid en
$\frac1{2\pi}\abs{\hat f}^2$ de impulsdichtheid van een
kwantumtoestand (fysische eenheden voegen $\hbar$ in), leest
\cref{exo:b3:fouriertransform:8} $\sigma_x\sigma_p \geq
\frac\hbar2$: geen toestand is scherp in beide
observabelen. Over het hoofdstuk draagt één wet vijf
pakken: warmte gladmaakt onmiddellijk omdat
$\eu^{-t\xi^2}$ hoge frequenties vernietigt (Deel II); de
stroom kan niet achteruit lopen omdat ze herstellen
onbegrensd is (Deel IV); een bandbegrensd signaal is
star genoeg om op een aftelbaar rooster te leven (Deel V);
geen functie slaat Heisenbergs vloer; en geen functie is
compact gedragen aan beide zijden van de transform (vragen
19--20). Wat $\hat f$ in oneindig doet, regelt wat $f$
overal mag doen.

\textbf{23.} Zowel $g_t$ als $g_s$ liggen in $L^1$ met
$\widehat{g_t}(\xi) = \eu^{-t\xi^2}$ (berekening van vraag
1), dus de convolutiestelling geeft $\widehat{g_t * g_s} =
\eu^{-t\xi^2}\eu^{-s\xi^2} = \eu^{-(t+s)\xi^2} =
\widehat{g_{t+s}}$; twee $L^1$-functies met dezelfde
transform stemmen a.e.\ overeen (injectiviteit, via de
inversiestelling --- hier zijn beide zijden continu, dus
ze stemmen overal overeen): $g_t * g_s = g_{t+s}$.
Bijgevolg $u(t + s) = g_{t+s} * f = g_s * (g_t * f) = g_s *
u(t)$ (associativiteit van convolutie, Tonelli).
Log-convexiteit: zij $N(t) = \norm{u(t)}_2^2 =
\frac1{2\pi}\int \eu^{-2t\xi^2}\abs{\hat f}^2\dd\xi$
(Plancherel, vraag 8). Voor $t = \frac{t_1 + t_2}2$,
schrijf
\[
\eu^{-2t\xi^2}\abs{\hat f}^2
= \Bigl(\eu^{-2t_1\xi^2}\abs{\hat f}^2\Bigr)^{1/2}
\Bigl(\eu^{-2t_2\xi^2}\abs{\hat f}^2\Bigr)^{1/2},
\]
en Cauchy--Schwarz geeft $N\bigl(\frac{t_1+t_2}2\bigr) \leq
\sqrt{N(t_1)\,N(t_2)}$: $\ln N$ is middenpuntconvex, en
continu zijnde (gedomineerde convergentie in $t$), convex;
zo is $\ln\norm{u(t)}_2 = \frac12\ln N(t)$. Verval met een
convexe logaritme: de warmtestroom kan energie niet in een
uitbarsting verliezen en dan stagneren.

\textbf{24.} De momenten van de kern: $\int g_t = 1$ (vraag
7 met $f = g_s$, of direct de Gauss-integraal), $\int
x\,g_t(x)\dd x = 0$ (oneven integrand), en, substituerend
$x = 2\sqrt t\,v$,
\[
\int_\R x^2g_t(x)\,\dd x
= \frac{4t}{\sqrt\pi}\int_\R v^2\eu^{-v^2}\dd v = 2t .
\]
Substituerend $x = z + y$ in de convolutie en noteren
$\iint(\abs z + \abs y)^2g_t(z)f(y)\,\dd z\,\dd y < \infty$
(elk van $\int\abs z^kg_t$, $\int\abs y^kf$ is eindig voor
$k \leq 2$, met $\abs y \leq \frac{1 + y^2}2$), Fubini en
Tonelli gelden voor de momentintegralen hieronder:
\[
\int x\,u(t,x)\dd x = \iint (z + y)\,g_t(z)f(y)\,\dd z\,\dd
y = 0\cdot\!\int\! f + 1\cdot\!\int\! yf(y)\dd y,
\]
wat de eerste claim is; en
\[
\iint (z+y)^2g_t(z)f(y)\,\dd z\,\dd y
= 2t\int f + 2\cdot0\cdot\!\int\! yf + \int y^2f(y)\dd y ,
\]
de tweede. Gemiddelden tellen op, varianties tellen op, en
de kern draagt gemiddelde $0$ en variantie $2t$ bij: na
tijd $t$ heeft de warmte zich over een breedte van orde
$\sqrt{2t}$ verspreid --- afstand groeit als de
vierkantswortel van de tijd, de handtekening van diffusie
(en van de Brownse paden van \cref{ch:b3:probability}).

\textbf{25.} Transformzijde: $\hat f(\xi) =
\sqrt\pi\,\eu^{-\xi^2/4}$, dus $\hat u(t,\xi) =
\sqrt\pi\,\eu^{-(t + \frac14)\xi^2}$, wat de transform is
van $(1 + 4t)^{-1/2}\exp\bigl(-x^2/(1+4t)\bigr)$ (het
Gauss-woordenboek $\eu^{-ax^2} \mapsto
\sqrt{\pi/a}\,\eu^{-\xi^2/4a}$ met $a = \frac1{1+4t}$): de
gesloten vorm. Directe controle, met $\sigma = 1 + 4t$:
\[
\partial_tu = \sigma^{-1/2}\eu^{-x^2/\sigma}
\Bigl(-\frac2\sigma + \frac{4x^2}{\sigma^2}\Bigr)
= \partial^2_{xx}u ,
\]
beide zijden berekend uit $\partial_xu =
-\frac{2x}\sigma\,u$. Behoud: $\int u(t) =
\sigma^{-1/2}\sqrt{\pi\sigma} = \sqrt\pi$ voor alle $t$.
Dissipatie:
\[
\norm{u(t)}_2^2 = \frac1\sigma\int\eu^{-2x^2/\sigma}\dd x
= \frac1\sigma\sqrt{\frac{\pi\sigma}2}
= \sqrt{\frac\pi2}\,(1+4t)^{-1/2},
\]
dus $\norm{u(t)}_2 = (\pi/2)^{1/4}(1+4t)^{-1/4}$,
niet-stijgend met convexe logaritme (vraag 23); de
$L^2$-norm vervalt als $t^{-1/4}$, precies de helft van de
$t^{-1/2}$-exponent van $\norm{u(t)}_\infty$ --- consistent
met $\norm u_2^2 \leq \norm u_\infty\norm u_1$ en behoud
van $\norm u_1$. Variantie: $\int x^2u(t) =
\frac1\sigma\cdot\frac{\sigma^{3/2}\sqrt\pi}2 =
\frac{\sqrt\pi}2\,(1 + 4t) = \int x^2f + 2t\sqrt\pi$, zoals
vraag 24 voorspelt ($\int x^2f = \frac{\sqrt\pi}2$, $\int f
= \sqrt\pi$). Bij $t = 6$: $\sigma = 25$, piekhoogte $u(6,
0) = \frac15$ versus $u(0,0) = 1$, breedteschaal
$\sqrt\sigma = 5$ maal de beginwaarde, en $\int u =
\sqrt\pi \approx 1.7725$ doorheen: de vlek is vijf keer
lager, vijf keer breder, en er ontbreekt geen calorie.
\end{solution}
