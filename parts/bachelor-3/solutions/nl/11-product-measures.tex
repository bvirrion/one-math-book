\section*{Hoofdstuk \ref{ch:b3:product} --- Productmaten, Fubini,
Variabelensubstitutie}

\begin{solution}{exo:b3:product:1}
$\Delta = \bigcap_n\bigcup_{k=0}^{2^n-1}
\intcc{k2^{-n}}{(k+1)2^{-n}}^2$ is een $G_\delta$, dus
Borel/\omterm{def:b3:lebesgue:measurable}{meetbaar}. Voor vaste $x$: $\Delta_x = \{x\}$, dus
$\lambda(\Delta_x) = 0$ en $\int\lambda(\Delta_x)\,\dd\mu =
0$. Voor vaste $y$: $\Delta^y = \{y\}$, dus $\mu(\Delta^y) =
1$ (telmaat) en $\int\mu(\Delta^y)\,\dd\lambda = \lambda(
\intcc01) = 1$. De geïtereerde integralen verschillen: $0
\neq 1$. De $\sigma$-eindigheid van $\mu$ faalt (elke
verzameling van eindige telmaat is eindig, en $\intcc01$ is
geen aftelbare unie van eindige verzamelingen in de
Borel-$\sigma$-algebra~--- eigenlijk: telmaat op
$\intcc01$ is niet $\sigma$-eindig omdat elke verzameling van
eindige \omterm{def:b3:measure:measure}{maat} eindig is, en een aftelbare unie van eindige
verzamelingen aftelbaar is, terwijl $\intcc01$ overaftelbaar
is).
\end{solution}

\begin{solution}{exo:b3:product:2}
$\eu^{-xy}\abs{\sin x} \geq 0$: Tonelli op $[0,A]\times
\intoo0\infty$ rechtvaardigt de verwisseling. De inwendige
$x$-integraal: $\int_0^A\eu^{-xy}\sin x\,\dd x =
\operatorname{Im}\int_0^A\eu^{-(y-\iu)x}\dd x =
\operatorname{Im}\frac{1-\eu^{-(y-\iu)A}}{y-\iu}$. Als $A
\to \infty$, $\eu^{-yA}\to 0$ voor $y > 0$, dus de limiet is
$\operatorname{Im}\frac{1}{y-\iu} = \frac{1}{1+y^2}$. De
integrand in $y$ is begrensd door $\frac{1}{1+y^2}$ (of
gebruik $|\int_0^A\eu^{-xy}\sin x| \leq
\int_0^A\eu^{-xy}\dd x \leq 1/y$ en knijp nabij $0$): DCT of
monotone limiet geeft $\int_0^\infty\frac{\dd y}{1+y^2} =
\frac\pi2$.
\end{solution}

\begin{solution}{exo:b3:product:3}
(a) $\sum_{n\geq1}\mathbf 1_{\{f\geq n\}} \leq f \leq 1 +
\sum_{n\geq1}\mathbf 1_{\{f\geq n\}}$ puntsgewijs (want als
$k \leq f < k+1$, is de som $k$ en $k \leq f < k+1 \leq 1 +
k$). Integreer en gebruik \cref{cor:b3:lebesgue:additivity}.

(b) $f \in L^1$ d.e.s.d.a.\ $\int\abs f < \infty$ d.e.s.d.a.\
(door (a) op $\abs f$) $\sum\mu(\abs f \geq n) < \infty$
(want $\mu(X) < \infty$ maakt de middelste en rechter
uitdrukkingen equivalent tot op een eindige additieve
constante).
\end{solution}

\begin{solution}{exo:b3:product:4}
$f = \partial_y\bigl(\frac{y}{x^2+y^2}\bigr)$: voor vaste $x
\in \intoo01$,
\[
\int_0^1 f(x,y)\,\dd y = \Bigl[\frac{y}{x^2+y^2}\Bigr]_0^1
= \frac{1}{x^2+1},
\]
dus $\int_0^1\int_0^1 f\,\dd y\,\dd x = \int_0^1\frac{\dd
x}{1+x^2} = \frac\pi4$. Evenzo $\int_0^1 f(x,y)\,\dd x =
-\frac{1}{1+y^2}$, geïtereerd $-\frac\pi4$. Absolute
integreerbaarheid: nabij $(0,0)$ is $\abs f \sim
\frac{\abs{x^2-y^2}}{(x^2+y^2)^2}$; in \omterm{ex:b3:product:polar}{poolcoördinaten} $\sim
\frac{\abs{\cos 2\theta}}{r^2}$, en $\int_0^\varepsilon
\frac{\dd r}{r}$ divergeert. Dus $\iint\abs f = \infty$:
Fubini is niet van toepassing.
\end{solution}

\begin{solution}{exo:b3:product:5}
(a) $(\mathbf 1_{\intcc01}*\mathbf 1_{\intcc01})(x) =
\lambda\bigl(\intcc01 \cap (x-\intcc01)\bigr) = \lambda(
\intcc01\cap\intcc{x-1}x)$. Voor $x \in [0,1]$: lengte $x$;
voor $x \in [1,2]$: lengte $2-x$; elders $0$. Tentfunctie.
Drievoudige convolutie is stuksgewijs kwadratisch (spline van
graad $2$) ondersteund op $[0,3]$.

(b) Als $x \notin \overline{\operatorname{supp}f +
\operatorname{supp}g}$, bestaat een buurt van $x$ disjunct
van $\operatorname{supp}f + \operatorname{supp}g$, dus voor
elke $y$, ofwel $y \notin \operatorname{supp}g$ ofwel $x-y
\notin \operatorname{supp}f$: de integrand verdwijnt, $f*g(x)
= 0$.

(c) $\abs{(f*g)(x+h) - (f*g)(x)} \leq \int\abs
f(y)\,\abs{g(x+h-y)-g(x-y)}\,\dd y$. DCT: $\abs{g(\cdot+h) -
g}$ $\to 0$ puntsgewijs (continuïteit), begrensd door
$2\norm g_\infty$, en $f \in L^1$: de integraal $\to 0$.
\end{solution}

\begin{solution}{exo:b3:product:6}
(a) Voor $d = 1$: $\lambda_1(\intco01) = 1 = 1/1!$. Inductie:
door Fubini, $\lambda_d(\Delta_d) = \int_0^1
\lambda_{d-1}(\Delta_{d-1}\cdot(1-x_d))\,\dd x_d =
\int_0^1\frac{(1-x_d)^{d-1}}{(d-1)!}\dd x_d = \frac1{d!}$.

(b) $v_2 = \pi/\Gamma(2) = \pi$; $v_3 =
\pi^{3/2}/\Gamma(5/2) = \pi^{3/2}/(\frac34\sqrt\pi) =
\frac{4\pi}3$. Ellipsoïde $= T(B(0,1))$ met $T =
\operatorname{diag}(a_i)$: $\lambda_d = \abs{\det T}v_d =
v_d\prod a_i$ (\cref{thm:b3:product:linearchange}).
\end{solution}

\begin{solution}{exo:b3:product:7}
Pool: $\int_0^{2\pi}\int_0^1 r^{-2s}r\,\dd r\,\dd\theta =
2\pi\int_0^1 r^{1-2s}\dd r$, eindig d.e.s.d.a.\ $1-2s > -1$,
d.w.z.\ $s < 1$. Buiten: $2\pi\int_1^\infty r^{1-2s}\dd r$,
eindig d.e.s.d.a.\ $s > 1$. In $\R^d$: $s < d/2$ (nabij $0$)
en $s > d/2$ (op oneindig), via $\int r^{d-1-2s}\dd r$.
\end{solution}

\begin{solution}{exo:b3:product:8}
$\Gamma(p)\Gamma(q) = \int_0^\infty\int_0^\infty
x^{p-1}y^{q-1}\eu^{-x-y}\dd x\,\dd y$. Substitueer $x = uv$,
$y = u(1-v)$: Jacobiaan $\abs{\det
\frac{\partial(x,y)}{\partial(u,v)}} = u$ (want $\partial(x,y)/
\partial(u,v) = \begin{pmatrix}v&u\\1-v&-u\end{pmatrix}$, det
$= -u$). Domein: $u > 0$, $v \in \intoo01$. Dus
\begin{align*}
\Gamma(p)\Gamma(q)
&= \int_0^1\int_0^\infty
(uv)^{p-1}(u(1-v))^{q-1}\eu^{-u}\,u\,\dd u\,\dd v \\
&= \Bigl(\int_0^\infty u^{p+q-1}\eu^{-u}\dd u\Bigr)
\Bigl(\int_0^1 v^{p-1}(1-v)^{q-1}\dd v\Bigr) \\
&= \Gamma(p+q)B(p,q).
\end{align*}
Met $t = \sin^2\theta$: $\int_0^{\pi/2}\sin^{2p-1}\theta
\cos^{2q-1}\theta\,\dd\theta = \frac12 B(p,q)$. Wallis: $W_n
= \int_0^{\pi/2}\sin^n = \frac12 B(\frac{n+1}2, \frac12)$.
\end{solution}

\begin{solution}{exo:b3:product:9}
Standaardmachine: voor $g = \mathbf 1_B$, $\int g\,\dd(T_*\mu)
= T_*\mu(B) = \mu(T^{-1}B) = \int\mathbf 1_B\circ T\,\dd\mu$;
lineariteit, MCT. De transferformule zegt $\int g\,\dd(T_*\mu)
= \int g\circ T$ zonder $T_*\mu$ te identificeren; de
veranderingsstelling berekent $T_*\lambda_d =
\abs{\det DT^{-1}}\,\lambda_d$ (of equivalent
$\abs{\det D\Phi}\lambda_d$ na pullback) expliciet als
dichtheidsmaat.
\end{solution}

\begin{solution}{exo:b3:product:10}
Door rotatie-invariantie en Tonelli: $\int\norm x^2
\eu^{-\norm x^2}\dd x = d\int x_1^2\eu^{-\norm x^2}\dd x$.
Pool/Fubini: $\int_{\R^d}\eu^{-\norm x^2}\dd x = \pi^{d/2}$,
en $\int x_1^2\eu^{-\norm x^2} = \int_\R x^2\eu^{-x^2}\dd x
\cdot \pi^{(d-1)/2} = \frac12\sqrt\pi\cdot\pi^{(d-1)/2} =
\frac12\pi^{d/2}$. Dus $\int\norm x^2\eu^{-\norm x^2} =
\frac d2\pi^{d/2}$. Voor de genormaliseerde \omterm{def:b3:measure:measure}{maat}
$\pi^{-d/2}\eu^{-\norm x^2}\dd x$: tweede moment van $\norm
X^2$ is $d/2$, en van $X_1^2$ is $1/2$.
\end{solution}

\begin{solution}{exo:b3:product:11}
(a) Voor vaste $x$: $H_x = \intoo0{f(x)}$ als $f(x) > 0$,
anders leeg: $\lambda_1(H_x) = f(x)$. Meetbaarheid van $H$:
$\{(x,y): 0 < y < f(x)\} = \{(x,y): f(x) - y > 0\}\cap\{y >
0\}$, en $(x,y)\mapsto f(x)-y$ is \omterm{def:b3:lebesgue:measurable}{meetbaar}. Tonelli:
$\lambda_{d+1}(H) = \int f\,\dd\lambda_d$.

(b) De grafiek is $\bigcap_n\{(x,y): \abs{y-f(x)} <
\frac1n\}$ doorsneden met~--- of: doorsnede in $x$ is een
punt, \omterm{def:b3:measure:measure}{maat} $0$; Tonelli geeft $\lambda_{d+1}(\mathrm{grafiek})
= 0$.

(c) $S^{d-1}$ is de grafiek van $\pm\sqrt{1-\norm x'^2}$ over
de eenheidsbal in $\R^{d-1}$ (twee hemisfieren), of de
rand van de bal: nul door (b) in elke kaart, of als
$\partial B$ met $\lambda_d(B_r\setminus B_{r-\varepsilon})
\to 0$.
\end{solution}

\begin{solution}{exo:b3:product:12}
Op $\intoo01^2$, $\frac1{1-xy} = \sum_{n\geq0}(xy)^n$
(meetkundig, $0 \leq xy < 1$): niet-negatieve termen, Tonelli
geeft $\iint\frac{\dd x\,\dd y}{1-xy} =
\sum_{n\geq0}\int_0^1\int_0^1 x^n y^n =
\sum_{n\geq0}\frac1{(n+1)^2} = \zeta(2)$. Evenzo
$\frac1{1+xy} = \sum_{n\geq0}(-1)^n(xy)^n$; absolute
convergentie $\sum\int\int(xy)^n = \zeta(2) < \infty$
rechtvaardigt verwisseling (of: positieve en negatieve delen
apart via Tonelli op de even/oneven splitsing), en men krijgt
$\sum_{n\geq0}(-1)^n\frac1{(n+1)^2} =
\sum_{k\geq1}(-1)^{k-1}/k^2 = \frac{\zeta(2)}2$
(even-oneven splitsing van $\zeta(2)$). Positiviteit doet haar
werk bij de eerste integraal (Tonelli zonder absolute
convergentie-check); bij de tweede is absolute convergentie
nodig voor de alternerende reeks, of men splitst even/oneven
tot positieve reeksen.
\end{solution}

\begin{solution}{pb:b3:product:1}
\textbf{1.} Met $x = t + \sqrt t\,u$ ($\dd x = \sqrt
t\,\dd u$; $x$ loopt over $\intoo0\infty$ als $u$ over
$\intoo{-\sqrt t}\infty$ loopt):
\[
\Gamma(t{+}1) = \int_0^\infty x^t\eu^{-x}\dd x
= t^t\eu^{-t}\sqrt t\int_{-\sqrt t}^{\infty}
\exp\Bigl(t\ln\Bigl(1 + \frac u{\sqrt t}\Bigr) - \sqrt
t\,u\Bigr)\dd u,
\]
want $x^t = t^t\exp\bigl(t\ln(1 + u/\sqrt t)\bigr)$ en
$\eu^{-x} = \eu^{-t}\eu^{-\sqrt tu}$.

\textbf{2.} Voor vaste $u$ en $t \to \infty$: $t\ln(1 +
u/\sqrt t) - \sqrt tu = t\bigl(\frac u{\sqrt t} -
\frac{u^2}{2t} + o(\frac1t)\bigr) - \sqrt tu = -\frac{u^2}2 +
o(1)$: $g_t(u) \to \eu^{-u^2/2}$.

\textbf{3.} Stel $\psi_1(h) = \varphi(h) + \frac{h^2}4$ op
$\intoc{-1}1$: $\psi_1(0) = 0$ en $\psi_1'(h) = \frac1{1+h} -
1 + \frac h2 = \frac{h(h-1)}{2(1+h)}$, wat $\geq 0$ is op
$\intoc{-1}0$ en $\leq 0$ op $\intcc01$: $\psi_1 \leq 0$,
d.w.z.\ $\varphi(h) \leq -h^2/4$ daar. Stel $\psi_2(h) =
\varphi(h) + ch$ op $\intco1\infty$, $c = 1 - \ln2$: $\psi_2(1)
= \ln2 - 1 + c = 0$ en $\psi_2'(h) = c - \frac h{1+h} \leq c -
\frac12 < 0$: $\varphi(h) \leq -ch$ voor $h \geq 1$. Nu voor $t
\geq 1$: als $\abs u \leq \sqrt t$, $g_t(u) =
\eu^{t\varphi(u/\sqrt t)} \leq \eu^{-t(u/\sqrt t)^2/4} =
\eu^{-u^2/4}$; als $u \geq \sqrt t$, dan $t\,\varphi(u/\sqrt t)
\leq -ct\cdot\frac u{\sqrt t} = -c\sqrt t\,u \leq -cu$ (want
$t \geq 1$), dus $g_t(u) \leq \eu^{-cu}$. Derhalve $g_t \leq
\eu^{-u^2/4} + \eu^{-cu}\mathbf 1_{u>0}$, \omterm{def:b3:lebesgue:l1}{integreerbaar},
onafhankelijk van $t \geq 1$.

\textbf{4.} DCT: $\int_\R g_t(u)\dd u \to
\int_\R\eu^{-u^2/2}\dd u = \sqrt{2\pi}$
(\cref{ex:b3:product:polar} plus de schaling $u\mapsto\sqrt2\,
u$). Met vraag 1:
\[
\Gamma(t + 1) \sim \sqrt{2\pi t}\;\Bigl(\frac
t\eu\Bigr)^t,\qquad
n! \sim \sqrt{2\pi n}\,\Bigl(\frac n\eu\Bigr)^n .
\]

\textbf{5.} Uit \cref{exo:b3:product:8}, $W_{2n} =
\frac\pi2\,\frac{(2n)!}{4^n(n!)^2} =
\frac\pi2\,4^{-n}\binom{2n}n$. Stirling:
\[
\binom{2n}{n} = \frac{(2n)!}{(n!)^2}
\sim \frac{\sqrt{4\pi n}\,(2n/\eu)^{2n}}
{2\pi n\,(n/\eu)^{2n}} = \frac{4^n}{\sqrt{\pi n}} .
\]
Dan $W_{2n} \sim \frac12\sqrt{\pi/n}$, consistent met de
recursie $W_n = \frac{n-1}nW_{n-2}$ (die $W_n \sim W_{n-2}$
dwingt, en met $W_nW_{n-1}\cdot n = \frac\pi2$ --- de
klassieke Wallis-relatie --- $W_n \sim \sqrt{\pi/(2n)}$
geeft; de twee asymptotieken stemmen overeen).

\textbf{6.} $\Gamma(\frac d2 + 1) \sim \sqrt{2\pi\frac
d2}\,(\frac d{2\eu})^{d/2}$, dus
\[
v_d = \frac{\pi^{d/2}}{\Gamma(\frac d2 + 1)}
\sim \frac{1}{\sqrt{\pi d}}\Bigl(\frac{2\pi\eu}
d\Bigr)^{d/2} \longrightarrow 0
\]
super-meetkundig (voor $d > 2\pi\eu \approx 17$ is elke factor
$< 1$ en krimpend). Numeriek $v_1 = 2$, $v_2 \approx
3.14$, $v_3 \approx 4.19$, $v_4 \approx 4.93$, $v_5 \approx
5.26$, $v_6 \approx 5.17$: het maximum ligt bij $d = 5$.

\textbf{7.} Met $k = \frac n2 + \frac{s\sqrt n}2$ (geheel,
$n$ even, $s$ vast): neem logaritmen in $2^{-n}\binom nk =
2^{-n}\frac{n!}{k!(n-k)!}$ en pas Stirling toe op de drie
factorialen. Schrijvend $k = \frac n2(1 + \varepsilon)$, $n -
k = \frac n2(1 - \varepsilon)$ met $\varepsilon = s/\sqrt n$:
\[
\ln\Bigl(2^{-n}\binom nk\Bigr)
= -\frac n2\bigl[(1{+}\varepsilon)\ln(1{+}\varepsilon) +
(1{-}\varepsilon)\ln(1{-}\varepsilon)\bigr]
+ \frac12\ln\frac{2}{\pi n(1 - \varepsilon^2)} + o(1),
\]
en de haak is $\varepsilon^2 + O(\varepsilon^4) =
\frac{s^2}n + O(n^{-2})$: de weergave streeft naar
$-\frac{s^2}2 + \frac12\ln\frac2{\pi n}$ tot op $o(1)$, d.w.z.
\[
2^{-n}\binom nk \sim \sqrt{\frac{2}{\pi
n}}\;\eu^{-s^2/2} :
\]
het Gaussische profiel van muntworp, gekwantificeerd --- de
lokale vorm van de Moivre--Laplace, te globaliseren in
\cref{ch:b3:clt}.

\textbf{8.} (i) DCT zet de puntsgewijze limiet van vraag 2 om
in convergentie van de integralen, met de dominant van vraag
3. (ii) De Gaussische integraal evalueert de limiet
$\int\eu^{-u^2/2} = \sqrt{2\pi}$ --- Stirlings constante
$\sqrt{2\pi}$ \emph{is} de Gaussische integraal. (iii) De
substitutie $x = t + \sqrt tu$ is een affiene
variabelensubstitutie: translatie-invariantie en de
schaalregel van de \omterm{def:b3:measure:lebesgueouter}{Lebesgue-maat}
(\cref{thm:b3:product:linearchange} in dimensie $1$).

\textbf{9.} Ontwikkel beide termen:
\[
d_n - d_{n+1} = \ln\frac{n!}{(n+1)!} + \Bigl(n +
\frac32\Bigr)\ln(n+1) - \Bigl(n + \frac12\Bigr)\ln n - 1
= \Bigl(n + \frac12\Bigr)\ln\frac{n+1}n - 1,
\]
de termen $-\ln(n+1)$ en $(n + \frac32)\ln(n+1)$ combineren
tot $(n + \frac12)\ln(n+1)$.

\textbf{10.} Voor $t = \frac1{2n+1}$: $\frac{1+t}{1-t} =
\frac{2n+2}{2n} = \frac{n+1}n$, en $n + \frac12 =
\frac1{2t}$; de oneven reeks $\ln\frac{1+t}{1-t} =
2\sum_{k\geq0}\frac{t^{2k+1}}{2k+1}$ geeft
\[
\Bigl(n + \frac12\Bigr)\ln\frac{n+1}n
= \sum_{k\geq0}\frac{t^{2k}}{2k+1}
= 1 + \frac{t^2}3 + \frac{t^4}5 + \cdots
\]
Trek $1$ af. Ondergrens: de eerste term alleen,
$\frac{t^2}3 = \frac1{3(2n+1)^2}$. Bovengrens: verlaag alle
noemers tot $3$ en som de meetkundige reeks:
$\frac{t^2}{3(1 - t^2)} = \frac1{3((2n+1)^2 - 1)} =
\frac1{12n(n+1)} = \frac1{12n} - \frac1{12(n+1)}$.

\textbf{11.} Sommeren van de bovengrens van $n$ tot $\infty$
(met $d_m \to 0$): $d_n < \frac1{12n}$. Voor de ondergrens:
$\frac1{12m+1} - \frac1{12(m+1)+1} =
\frac{12}{(12m+1)(12m+13)}$, en
\begin{align*}
\frac1{3(2m+1)^2} > \frac{12}{(12m+1)(12m+13)}
&\iff (12m+1)(12m+13) > 36(2m+1)^2 \\
&\iff 168m + 13 > 144m + 36,
\end{align*}
waar voor $m \geq 1$. Sommeren van deze telescoperende
minorant: $d_n > \frac1{12n+1}$. Exponentiëren geeft de
klassieke inkadering van $n!$.

\textbf{12.} (a) Relatieve fout $= \eu^{d_n} - 1 <
\eu^{1/(12n)} - 1 < \frac{1.1}{12n}$ voor grote $n$; $<
10^{-6}$ zodra $12n \geq 1.1\cdot10^6$, en de gestelde $n
\geq 83\,334$ volstaat ($\frac1{12n} \leq 10^{-6}$
impliceert het al). (b) $\log_{10}(100!) =
\frac12\log_{10}(200\pi) + 200 - 100\log_{10}\eu +
d_{100}\log_{10}\eu = 1.39906 + 200 - 43.42945 + 0.00036
\approx 157.96997$, dus $100! \approx 10^{0.96997} \cdot
10^{157} = 9.333\cdot10^{157}$; het gegarandeerde venster
$(\eu^{1/1201}, \eu^{1/1200})$ heeft breedte onder $10^{-6}$
in relatieve termen --- een ``asymptotische'' formule die bij
$n = 100$ een precisie-instrument is.

\textbf{13.} Schrijf $\sin^n = \sin^{n-2} - \sin^{n-2}\cos^2$,
en integreer de tweede term partieel ($u = \cos\theta$,
$\dd v = \sin^{n-2}\cos\theta\,\dd\theta$, $v =
\frac{\sin^{n-1}}{n-1}$):
\[
\int_0^{\pi/2}\sin^{n-2}\cos^2 =
\Bigl[\cos\theta\,\frac{\sin^{n-1}\theta}{n-1}\Bigr]_0^{\pi/2}
+ \frac1{n-1}\int_0^{\pi/2}\sin^n = \frac{W_n}{n-1} .
\]
Dus $W_n = W_{n-2} - \frac{W_n}{n-1}$, d.w.z.\ $W_n =
\frac{n-1}nW_{n-2}$. Uit $W_0 = \frac\pi2$, $W_1 = 1$:
\[
W_{2n} = \frac{(2n-1)!!}{(2n)!!}\cdot\frac\pi2
= \frac\pi2\binom{2n}n4^{-n},
\qquad
W_{2n+1} = \frac{(2n)!!}{(2n+1)!!}
= \frac{4^n}{(2n+1)\binom{2n}n},
\]
dubbele factorialen converterend via $(2n)!! = 2^nn!$ en
$(2n-1)!! = \frac{(2n)!}{2^nn!}$. Tenslotte $nW_nW_{n-1} =
(n-1)W_{n-1}W_{n-2}$ door de recursie: constant, gelijk aan
$1\cdot W_1W_0 = \frac\pi2$.

\textbf{14.} $W_{2n+1} \leq W_{2n} \leq W_{2n-1}$ (puntsgewijze
monotonie van $\sin^n$) en $\frac{W_{2n-1}}{W_{2n+1}} =
\frac{2n+1}{2n} \to 1$ knijpen $\frac{W_{2n}}{W_{2n+1}} \to
1$. Gecombineerd met $W_{2n}W_{2n+1} = \frac{\pi}{2(2n+1)}$
(vraag 13): $W_{2n}^2 \sim \frac\pi{4n}$, dus $W_{2n} \sim
\frac12\sqrt{\frac\pi n}$ en $\binom{2n}n4^{-n} =
\frac2\pi W_{2n} \sim \frac1{\sqrt{\pi n}}$.

\textbf{15.} Vragen 9--10 gebruikten nooit de waarde van de
constante: met $e_n = \ln n! - (n+\frac12)\ln n + n$ liggen
de verschillen $e_n - e_{n+1}$ in $\bigl(0, \frac1{12n} -
\frac1{12(n+1)}\bigr)$, dus daalt $(e_n)$ terwijl $(e_n -
\frac1{12n})$ stijgt: aangrenzende rijen, convergerend naar
een gemeenschappelijke $\ell$. Dus $n! \sim K
n^{n+1/2}\eu^{-n}$, $K = \eu^\ell$.

\textbf{16.} Substitueren van de onbekende-constante Stirling
in de centrale binomiaal:
\[
\binom{2n}n4^{-n}\sqrt n \sim
\frac{K\,(2n)^{2n+1/2}\eu^{-2n}}{\bigl(K\,n^{n+1/2}
\eu^{-n}\bigr)^2}\,4^{-n}\sqrt n
= \frac{2^{2n}\sqrt{2}\,K\,n^{2n+1/2}}{K^2\,n^{2n+1}}
\,4^{-n}\sqrt n = \frac{\sqrt2}{K},
\]
en vraag 14 dwingt $\frac{\sqrt2}K = \frac1{\sqrt\pi}$:
$K = \sqrt{2\pi}$. Deel III--IV herbewijzen zo Stirling van
nul af; invoeren in Deel I's identiteit evalueert
$\int_\R\eu^{-u^2/2}\dd u = \sqrt{2\pi}$ zonder
\omterm{ex:b3:product:polar}{poolcoördinaten}: Wallis en Gauss stutten elkaar.

\textbf{17.} $\frac{\binom{2n}{n+j}}{\binom{2n}n} =
\frac{(n!)^2}{(n+j)!\,(n-j)!} =
\prod_{i=1}^{j}\frac{n-i+1}{n+i}$ voor $j \geq 0$ (en door
symmetrie voor $j < 0$). Logaritmen nemend, met $1 \leq i
\leq j \leq K\sqrt n$:
\[
\ln\frac{n-i+1}{n+i} = \ln\Bigl(1 - \frac{i-1}n\Bigr) -
\ln\Bigl(1 + \frac in\Bigr) = -\frac{2i-1}{n} +
O\Bigl(\frac{i^2}{n^2}\Bigr),
\]
en $\sum_{i\leq j}(2i - 1) = j^2$, terwijl de fout sommeert
tot $O(j^3/n^2) = O(n^{-1/2})$: uniform,
$\exp\bigl(-\frac{j^2}n + O(n^{-1/2})\bigr)$.

\textbf{18.} $\eu^{-n}\frac{n^n}{n!} \sim \eu^{-n}
\frac{n^n}{\sqrt{2\pi n}\,n^n\eu^{-n}} =
\frac1{\sqrt{2\pi n}}$. Een Poissonvariabele van mean $n$
heeft standaardafwijking $\sqrt n$, en $\frac1{\sqrt{2\pi n}}$
is precies de Gaussische piekhoogte $\frac1{\sigma\sqrt{2\pi}}$:
de lokale CLT, voorproef bij de modus.

\textbf{19.} Voor $a \in \intoo01$ geeft het hellinglemma van
\cref{pb:b3:lebesgue:1} (vraag 14 daar), toegepast op de
convexe $\log\Gamma$ rond $n$, $(n-1)^a \leq
\frac{\Gamma(n+a)}{\Gamma(n)} \leq n^a$: de ratio tot $n^a$
wordt geknepen door $(1 - \frac1n)^a \to 1$. Voor $a = m +
a'$ ($m \in \N$, $a' \in \intco01$): $\Gamma(n+a) = (n + a -
1) \cdots(n + a')\Gamma(n + a')$, en elk van de $m$ factoren
is $n(1 + O(\frac1n))$: vermenigvuldig de schattingen.

\textbf{20.} De recursie $v_d = \frac{2\pi}dv_{d-2}$
(uit $\Gamma(\frac d2 + 1) = \frac d2\Gamma(\frac d2)$)
geeft, vanaf $v_1 = 2$, $v_2 = \pi$:
\[
v_3 = \frac{4\pi}3,\quad v_4 = \frac{\pi^2}2,\quad
v_5 = \frac{8\pi^2}{15},\quad v_6 = \frac{\pi^3}6,\quad
v_7 = \frac{16\pi^3}{105}.
\]
De ratio $\frac{2\pi}d$ overschrijdt $1$ precies voor $d \leq
6$, dus stijgt elke pariteit dan daalt; numeriek $v_4
\approx 4.93$, $v_5 \approx 5.26$, $v_6 \approx 5.17$: het
globale maximum is $d = 5$. Genererende functie: $v_{2k} =
\frac{\pi^k}{k!}$, dus $\sum_kv_{2k}x^{2k} =
\eu^{\pi x^2}$ --- alle even-dimensionale balvolumes gerold
in één exponentieel, en een onmiddellijke super-meetkundige
afname-schatting voor $v_d$.

\textbf{21.} Stirling in teller en noemer, met $k =
\alpha n$:
\[
\binom n{\alpha n} \sim
\frac{\sqrt{2\pi n}\,n^n}
{\sqrt{2\pi\alpha n}\,(\alpha n)^{\alpha n}\,
\sqrt{2\pi(1-\alpha)n}\,((1-\alpha)n)^{(1-\alpha)n}}
= \frac{\eu^{nH(\alpha)}}{\sqrt{2\pi\alpha(1-\alpha)n}},
\]
want $n^n/(\alpha n)^{\alpha n}((1-\alpha)n)^{(1-\alpha)n} =
\alpha^{-\alpha n}(1-\alpha)^{-(1-\alpha)n} =
\eu^{nH(\alpha)}$ (de machten van $n$ cancelen: $\alpha n +
(1-\alpha)n = n$), en de $\eu^{-n}$'s cancelen evenzo. Bij
$\alpha = \frac12$: $H = \ln2$ en de prefactor is
$\sqrt{2/(\pi n)}$ --- vraag 5 opnieuw. Strikte concaafheid
van $H$ (tweede afgeleide $-\frac1{\alpha(1-\alpha)} < 0$)
plaatst haar maximum $\ln 2$ alleen bij $\alpha = \frac12$:
voor $\alpha \neq \frac12$ daalt $\binom n{\alpha n}2^{-n}
\approx \eu^{-n(\ln2 - H(\alpha))}$ exponentieel --- de
combinatorische motor achter elke concentratiestatement over
muntworpen.

\textbf{22.} Uit $s_{d-1} = dv_d$ en vraag 20:
\[
s_0 = 2,\ \ s_1 = 2\pi,\ \ s_2 = 4\pi,\ \ s_3 = 2\pi^2,\ \
s_4 = \frac{8\pi^2}3,\ \ s_5 = \pi^3,\ \ s_6 =
\frac{16\pi^3}{15},
\]
numeriek $2,\ 6.28,\ 12.57,\ 19.74,\ 26.32,\ 31.01,\
33.07$; en $s_7 = \frac{\pi^4}3 \approx 32.47 < s_6$: het
maximum is de $6$-sfeer. De recursie $s_{d+1} =
(d+2)\,v_{d+2} = (d+2)\,\frac{2\pi}{d+2}\,v_d = 2\pi v_d =
\frac{2\pi}d\,s_{d-1}$ toont dezelfde $\frac{2\pi}d$-gedreven
stijging en super-meetkundige val als voor volumes: voorbij
dimensie zeven krimpen sferen sneller weg dan elke meetkundige
rij.

\textbf{23.} Vraag 11 zegt precies $\frac1{12n+1} < d_n <
\frac1{12n}$, en
\[
\frac1{12n} - \frac1{12n+1} = \frac1{12n(12n+1)} =
O\Bigl(\frac1{n^2}\Bigr),
\]
dus $d_n = \frac1{12n} + O(\frac1{n^2})$ en $\eu^{d_n} = 1 +
\frac1{12n} + O(\frac1{n^2})$; vermenigvuldigen met
$\sqrt{2\pi n}(n/\eu)^n$ geeft de gecorrigeerde formule. Bij
$n = 10$: $\sqrt{20\pi}\,(10/\eu)^{10} = 7.92665 \times
453999.3 \approx 3\,598\,696$, laag met $30\,104$ (relatieve
fout $8.3\cdot10^{-3}$); vermenigvuldigen met $1 +
\frac1{120}$ geeft $3\,628\,685$, laag met $115$ (relatieve
fout $3.2\cdot10^{-5}$). De inkadering zelf pinnt $10!$ tussen
$3\,598\,696\,\eu^{1/121} \approx 3\,628\,559$ en
$3\,598\,696\,\eu^{1/120} \approx 3\,628\,808$ --- de
bovengrens mist met acht eenheden in zeven cijfers.

\textbf{24.} Deel I's substitutie $x = t + \sqrt t\,u$,
toegepast op de afgeknotte integraal, geeft
\[
\int_0^{t} x^{t}\eu^{-x}\,\dd x
= t^{t}\eu^{-t}\sqrt t\int_{-\sqrt t}^{0} g_t(u)\,\dd u ,
\]
het bereik $0 \leq x \leq t$ wordt $-\sqrt t \leq u \leq 0$.
De dominant van vraag 3 dekt $g_t\mathbf 1_{u < 0}$ ook, dus
gedomineerde convergentie levert
\[
\int_{-\sqrt t}^{0}g_t(u)\,\dd u \longrightarrow
\int_{-\infty}^{0}\eu^{-u^2/2}\dd u = \frac{\sqrt{2\pi}}2 ,
\]
terwijl vraag 4 $\Gamma(t+1) \sim
t^t\eu^{-t}\sqrt t\,\sqrt{2\pi}$ geeft. De ratio streeft naar
$\frac12$. Probabilistisch: een Gamma-toevalsvariabele van
grote vorm plaatst asymptotisch de helft van haar massa aan
elke kant van haar modus --- de symmetrie van de centrale
limietstelling, afgelezen uit één substitutie.

\textbf{25.} Zij $\lambda = \frac{\alpha}{1-\alpha} \in
\intoc01$. Voor $k \leq \lfloor\alpha n\rfloor \leq \alpha n$
geldt $\lambda^{k - \alpha n} \geq 1$, dus
\[
\sum_{k=0}^{\lfloor\alpha n\rfloor}\binom nk
\leq \lambda^{-\alpha n}\sum_{k=0}^{n}\binom nk\lambda^{k}
= \Bigl(\lambda^{-\alpha}(1 + \lambda)\Bigr)^{n},
\]
en met $\lambda = \frac\alpha{1-\alpha}$:
\[
\lambda^{-\alpha}(1+\lambda)
= \alpha^{-\alpha}(1-\alpha)^{\alpha}\cdot\frac1{1-\alpha}
= \alpha^{-\alpha}(1-\alpha)^{-(1-\alpha)}
= \eu^{H(\alpha)} .
\]
Optimaliteit: minimaliseren van $f(\lambda) = -\alpha\ln\lambda
+ \ln(1+\lambda)$ over $\lambda > 0$, de vergelijking
$f'(\lambda) = -\frac\alpha\lambda + \frac1{1+\lambda} = 0$
heeft de unieke oplossing $\lambda = \frac\alpha{1-\alpha}$,
een minimum want $f'' > 0$ --- de exponentiële-tilt-
(Chernoff-)keuze. Verzoening: door vraag 21 is de enkele term
$k = \lfloor\alpha n\rfloor$ al van orde
$\eu^{nH(\alpha)}/\sqrt{2\pi\alpha(1-\alpha)n}$, dus
\[
\frac{\eu^{nH(\alpha)}}{C\sqrt n} \leq
\sum_{k\leq\alpha n}\binom nk \leq \eu^{nH(\alpha)} :
\]
de snelheid $H(\alpha)$ is exact, de hele som kost ten
hoogste een factor $\sqrt n$ over haar grootste term. Gedeeld
door $2^n$ is dit de eerlijke-munt-staartgrens $\P(S_n \leq
\alpha n) \leq \eu^{-n(\ln2 - H(\alpha))}$ --- concentratie
van \omterm{def:b3:measure:measure}{maat} in één regel.
\end{solution}
