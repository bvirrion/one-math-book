\section*{Capítulo \ref{ch:b3:lp} --- O $L^p $ Espaços}

\begin{solution}{exo:b3:lp:1}
(um) $ p = q = 2$ em \cref{thm:b3:lp:holder}: $\abs{\int f\bar
g\,\dd\mu} \leq \int\abs{fg} \leq \norm f_2\norm g_2$ ---
Cauchy--Schwarz, com igualdade se $\abs f, \abs g $
proporcionais e as fases alinhadas.

(b) Em um espaço de probabilidade, para $ p \leq q $: aplicar Jensen
(\cref{exo:b3:lp:10}) com o convexo $\Phi(t) =
\abs t^{q/p}$ para a função $\abs f^p $:
$\bigl(\int\abs f^p\bigr)^{q/p} \leq \int\abs f^q $, ou seja\
$\norm f_p \leq \norm f_q $.

(c) $\int\abs{fg} = \norm f_1\norm g_\infty $ se $\abs{g} =
\norm g_\infty $ a.e.\ em $\{f \neq 0\}$ (a desigualdade
$\abs{fg} \leq \abs f\norm g_\infty $ deve ser um a.e.\
igualdade).
\end{solution}

\begin{solution}{exo:b3:lp:2}
$\int_0^1 x^{-p/2}\dd x < \infty $ se $ p < 2$: o primeiro está em
$ L^p $ para $ p \in \intco12$. $\int_1^\infty x^{-p/2}\dd x <
\infty $ se $ p > 2$: o segundo para $ p \in \intoo2\infty $ (e
$ p = \infty $: é limitado --- inclua-o). Terceiro: para $ p <
2$, dominado por $ x^{-p/2}$: \omterm{def:b3:lebesgue:l1}{integrável}; para $ p = 2$,
substituto $ u = -\ln x $: $\int_0^1\frac{\dd x}{x(1 + \abs{\ln
x})^2} = \int_0^\infty\frac{\dd u}{(1 + u)^2} < \infty $; para
$ p > 2$ o poder domina: divergente. Então $ p \in \intcc12$.
Quarto: $\int_\R\frac{\dd x}{(1 + \abs x)^p} < \infty $ se $ p >
1$; limitado, então também $ L^\infty $: $ p \in
\intoc1\infty $. Moral: integrabilidade no $0$ gosta de pequeno $ p $, no
$\infty $ grande $ p $; combinando ambas as obstruções, sem inclusão
entre $ L^p(\R)$ espaços.
\end{solution}

\begin{solution}{exo:b3:lp:3}
(um) $\norm{f_n}_p^p = \lambda(I_n) \to 0$ (no nível diádico $ k $
o comprimento é $2^{-k}$). Mas cada $ x $ está em um intervalo de
cada nível diádico: $ f_n(x) = 1$ infinitamente e frequentemente $= 0$
infinitamente frequente (intervalos do mesmo nível não contendo
$ x $): sem convergência em nenhum ponto.

(b) $ f_{n_k} = \mathbf 1_{\intcc0{2^{-k}}}$ (o primeiro intervalo
de cada nível) converge para $0$ a cada $ x > 0$: a.e.

(c) a.e.\ mas não $ L^1$: $ n\mathbf 1_{\intoo0{1/n}} \to 0$
ou seja, integral $1$. Em $ L^1$ mas em não $ L^p $ ($ p > 1$): $ g_n =
\eu^n\,\mathbf 1_{(0,\ \eu^{-n}/n)}$: $\norm{g_n}_1 = \frac1n
\to 0$, enquanto $\norm{g_n}_p^p = \eu^{(p-1)n}/n \to \infty $ para
cada $ p > 1$.
\end{solution}

\begin{solution}{exo:b3:lp:4}
Superior: $\norm f_p \leq \mu(X)^{1/p}\norm f_\infty $
(\cref{prop:b3:lp:inclusions}(a) com $ q = \infty $), e
$\mu(X)^{1/p} \to 1$. Inferior: para $\varepsilon > 0$, $ A =
\{\abs f > \norm f_\infty - \varepsilon\}$ tem $\mu(A) > 0$
(definição do sup essencial), e
\[
\norm f_p \geq \Bigl(\int_A \abs f^p\Bigr)^{1/p}
\geq (\norm f_\infty - \varepsilon)\,\mu(A)^{1/p}
\xrightarrow[p\to\infty]{} \norm f_\infty - \varepsilon .
\]
\end{solution}

\begin{solution}{exo:b3:lp:5}
(a) Duas vezes: a conversão $\norm{\tau_hg - g}_p \leq
C^{1/p}\norm{\tau_hg - g}_\infty $ (medida finita suporte)
degenera para $ p = \infty $ só nisso o \emph{\omterm{ex:b3:lebesgue:gamma}{densidade} de
$\mathcal C_c $} falha aí --- essa é a verdadeira lacuna: etapa (2)
de \cref{thm:b3:lp:density} não tem $ L^\infty $ análogo.

(b) Se $ f $ tem uma uniformidade \omterm{def:b3:topology:continuity}{contínuo} representante $ g $:
$\norm{\tau_hf - f}_\infty = \sup_x\abs{g(x - h) - g(x)} \to
0$. Por outro lado, suponha $\norm{\tau_hf - f}_\infty \to 0$. O
apaziguamento $ f_\varepsilon = f * \rho_\varepsilon $ são
\omterm{def:b3:topology:continuity}{contínuo}, e
\[
\norm{f_\varepsilon - f}_\infty
\leq \sup_{\norm y \leq \varepsilon}\norm{\tau_yf -
f}_\infty \longrightarrow 0
\]
(a convolução é uma média de traduções). Cada
$ f_\varepsilon $ é uniformemente \omterm{def:b3:topology:continuity}{contínuo}
($\norm{\tau_hf_\varepsilon - f_\varepsilon}_\infty \leq
\norm{\tau_hf - f}_\infty $, pela média) e um limite uniforme
de uniformemente \omterm{def:b3:topology:continuity}{contínuo} funções é uma: $ f $ concorda a.e.\
com uma uniformidade \omterm{def:b3:topology:continuity}{contínuo} função.
\end{solution}

\begin{solution}{exo:b3:lp:6}
Por Hölder, para cada $ x $ o integrando $ y \mapsto f(x -
y)g(y)$ está em $ L^1$ com $\abs{(f*g)(x)} \leq \norm f_p\norm
g_q $: em todos os lugares definidos e limitados. Uniforme continuidade ($ p <
\infty $):
\[
\abs{(f*g)(x + h) - (f*g)(x)}
= \Bigl|\int\bigl(\tau_{-h}f - f\bigr)(x - y)\,g(y)\dd y\Bigr|
\leq \norm{\tau_{-h}f - f}_p\,\norm g_q
\xrightarrow[h\to0]{} 0,
\]
uniformemente em $ x $ (\cref{thm:b3:lp:density}(3)). Se $ p =
\infty $, então $ q = 1$: escrever $ f * g = g * f $ e execute o mesmo
vinculado à tradução agindo sobre $ g \in L^1$.
\end{solution}

\begin{solution}{exo:b3:lp:7}
Consertar $\chi \in \mathcal C_c^\infty(\intoo ab)$ com $\int\chi =
1$ e definir $ c = \int f\chi $. Deixar $\varphi \in \mathcal
C_c^\infty $ ser arbitrário e $\psi = \varphi -
\bigl(\int\varphi\bigr)\chi $: então $\int\psi = 0$, então $\Phi(x)
= \int_a^x\psi $ define $\Phi \in \mathcal C_c^\infty(\intoo
ab)$ (desaparece perto de ambas as extremidades: perto $ a $ trivialmente, perto $ b $
porque a integral total é $0$) com $\Phi' = \psi $. O
hipótese dá $\int f\psi = \int f\Phi' = 0$, por isso
\[
\int f\varphi = \Bigl(\int\varphi\Bigr)\int f\chi = \int
c\,\varphi
\quad\text{for every } \varphi:
\qquad \int(f - c)\varphi = 0 .
\]
Por \cref{cor:b3:lp:fundlemma} (localizado em $\intoo ab $), $ f =
c $ a.e.
\end{solution}

\begin{solution}{exo:b3:lp:8}
Deixar $3\delta < d(K, \R^d\setminus U)$ (positivo:
\cref{exo:b3:topology:6}(b); se $ U = \R^d $ qualquer $\delta $
funciona), $ K_\delta = \{x : d(x, K) \leq \delta\}$, e
\[
\varphi = \mathbf 1_{K_\delta} * \rho_{\delta/2} .
\]
Então $\varphi \in \mathcal C^\infty $
(\cref{thm:b3:lp:regularization}(1); o indicador é $ L^1$),
$0 \leq \varphi \leq 1$ ($\int\rho = 1$), $\varphi = 1$ sobre $ K $
(para $ x \in K $, $\bar B(x, \delta/2) \subseteq K_\delta $, então
a convolução integra $\rho $ totalmente) e
$\operatorname{supp}\varphi \subseteq K_{3\delta/2} \subseteq
U $: suportado de forma compacta ($ K_\delta $ é limitado). Partição de
unidade: dada $ K \subseteq U_1\cup\dots\cup U_m $, escolha (por
compacidade) compactos $ K_i \subseteq U_i $ com $ K \subseteq
\bigcup \mathring K_i $, pegar $\varphi_i $ como acima para $(K_i,
U_i)$ e definir $\psi_i = \varphi_i\prod_{j <i}(1 -
\varphi_j)$: cada $\psi_i \in \mathcal C_c^\infty(U_i)$, e
$\sum_i\psi_i = 1 - \prod_i(1 - \varphi_i) = 1$ sobre $ K $.
\end{solution}

\begin{solution}{exo:b3:lp:9}
(a) O expoente de interpolação para $(p_0, q_0) = (1, \infty)$
no $ r = p $ é $\alpha = \frac1p $:
\cref{prop:b3:lp:inclusions}(c) dá $\norm f_p \leq \norm
f_1^{1/p}\norm f_\infty^{1 - 1/p}$.

(b) Tomando logaritmos em
\cref{prop:b3:lp:inclusions}(c): $\ln\norm f_r \leq
\alpha\ln\norm f_p + (1-\alpha)\ln\norm f_q $ onde $\frac1r $
é a mesma combinação convexa de $\frac1p, \frac1q $: $\frac1p
\mapsto \ln\norm f_p $ é convexo. Exemplo com $ L^p $-filiação
exatamente em $\intoo{p_0}{p_1}$:
\[
f(x) = x^{-1/p_1}\,\mathbf 1_{\intoo01}(x) +
x^{-1/p_0}\,\mathbf 1_{\intco1\infty}(x):
\]
o primeiro termo é $ L^p $ se $ p < p_1$, o segundo se $ p >
p_0$.
\end{solution}

\begin{solution}{exo:b3:lp:10}
Deixar $ m = \int f\,\dd\mu \in \R $. A convexidade fornece um suporte
linha em $ m $: há $ s $ com $\Phi(t) \geq \Phi(m) + s(t -
m)$ para todos $ t $ (pegar $ s $ entre as derivadas unilaterais,
que existem para funções convexas). Substituir $ t = f(x)$ e
integrar contra a probabilidade $\mu $:
\[
\int\Phi\circ f\,\dd\mu \geq \Phi(m) + s\Bigl(\int f - m\Bigr)
= \Phi\Bigl(\int f\,\dd\mu\Bigr)
\]
(mensurabilidade: $\Phi $ é \omterm{def:b3:topology:continuity}{contínuo}; integrabilidade do
parte negativa de $\Phi\circ f $ é garantido pelo apoio
linha). AM--GM: em um conjunto finito com pesos $ w_i $, pegar
$\Phi = \exp $ e $ f = \sum(\ln a_i)\mathbf 1_i $:
$\exp\bigl(\sum w_i\ln a_i\bigr) \leq \sum w_ia_i $, ou seja\
$\prod a_i^{w_i} \leq \sum w_ia_i $. A norma monotonicidade é
\cref{exo:b3:lp:1}(b).
\end{solution}

\begin{solution}{exo:b3:lp:11}
(a) Suponha primeiro $ p, q, r < \infty $ e $ f, g \geq 0$
(substituir por valores absolutos). Os três expoentes $ r $,
$\alpha = \frac{pr}{r-p}$, $\beta = \frac{qr}{r-q}$
satisfazer $\frac1r + \frac1\alpha + \frac1\beta = \frac1r +
\frac1p - \frac1r + \frac1q - \frac1r = 1$ (a escala
relação). Dividido, para fixo $ x $,
\[
f(y)g(x-y) = \bigl[f(y)^pg(x-y)^q\bigr]^{1/r}\cdot
f(y)^{1 - p/r}\cdot g(x-y)^{1 - q/r},
\]
e Hölder com os três expoentes dá
\[
(f*g)(x) \leq \Bigl(\int f^pg(x-\cdot)^q\Bigr)^{1/r}
\norm f_p^{\,p/\alpha\cdot\alpha/p}\cdots
\]
mais precisamente: o segundo fator é
$\bigl(\int f^{(1-p/r)\alpha}\bigr)^{1/\alpha} =
\norm f_p^{p(1/p - 1/r)}$ desde $(1 - \frac pr)\alpha = p $,
e da mesma forma o terceiro é $\norm g_q^{q(1/q - 1/r)}$. Aumentar
para o $ r $-th poder e integrar em $ x $ (Tonelli no
primeiro fator):
\[
\norm{f*g}_r^r \leq \norm f_p^p\,\norm g_q^q\cdot
\norm f_p^{\,rp(1/p - 1/r)}\,\norm g_q^{\,rq(1/q - 1/r)}
= \norm f_p^r\,\norm g_q^r .
\]
Os casos de endpoint ($ r = \infty $ ou um expoente igual a
seu limite) são \omterm{def:b3:groups:simple}{simples} Hëstimativas mais antigas ou diretas.

(b) $ r = \infty $ forças $ q = p'$: $\abs{f*g(x)} \leq \norm
f_p\norm g_{p'}$ ---Hölder após tradução-reflexão.
$ q = 1$ dá $ r = p $: $\norm{f*g}_p \leq \norm
g_1\norm f_p $, o burro de carga da apaziguamento
(\cref{thm:b3:lp:regularization}motor). $ p = q = 1$
dá $ r = 1$: a álgebra de convolução
(\cref{thm:b3:product:convolution}).

(c) Substituir $ f, g $ por $ f_\lambda = f(\lambda\cdot)$,
$ g_\lambda = g(\lambda\cdot)$: então $ f_\lambda*g_\lambda =
\lambda^{-d}(f*g)(\lambda\cdot)$ e comparando normas,
\[
\text{LHS} \sim \lambda^{-d - d/r}, \qquad
\text{RHS} \sim \lambda^{-d/p - d/q} :
\]
uma desigualdade válida para todos $ f, g $ força as duas escalas
expoentes para corresponder, ou seja,\ $1 + \frac1r = \frac1p + \frac1q $
exatamente. Qualquer outra combinação morre em $\lambda \to 0$ ou
$\infty $: A relação de Young não é uma conveniência, mas uma
lei de escala.
\end{solution}

\begin{solution}{exo:b3:lp:12}
(a) Normalizar $\norm f_p = \norm g_q = 1$. A prova de
Hölder integra a desigualdade de Young $\abs{fg} \leq
\frac{\abs f^p}p + \frac{\abs g^q}q $; igualdade do
integrais forçam a igualdade a.e.\ em Young, que (estrito
convexidade de $\exp $; igualdade se $ a^p = b^q $) significa $\abs
f^p = \abs g^q $ a.e. Desfazendo a normalização:
$\abs f^p\norm g_q^q = \abs g^q\norm f_p^p $ a.e.\ ---
proporcionalidade.

(b) Minkowski é dois Hïdosos se candidataram $\abs{f + g}^{p-1}
\abs f $ e $\abs{f+g}^{p-1}\abs g $; forças de igualdade (a)'s
proporcionalidade em ambos: $\abs f^p $ e $\abs g^p $ cada
proporcional a $\abs{f+g}^{(p-1)q} = \abs{f+g}^p $, então
$\abs g = t\abs f $ a.e.\ para uma constante $ t \geq 0$; e o
desigualdade triangular pontual inicial $\abs{f + g} \leq
\abs f + \abs g $ também deve ser uma igualdade a.e.\, que para
valores complexos significa $ f $ e $ g $ tenho um.e.\ o mesmo
argumento onde ambos são diferentes de zero. Combinando: $ g = tf $ a.e.,
$ t > 0$ (ambos diferentes de zero).

(c) $ p = 1$: igualdade em $\int\abs{f + g} = \int\abs f +
\int\abs g $ mantém sempre $ f, g $ tem o mesmo sinal
padrão (mesmo argumento a.e.) --- nenhuma proporcionalidade necessária:
$ f = \mathbf 1_{\intcc01}$ e $ g = \mathbf 1_{\intcc02}$
trabalhar. $ p = \infty $: $\norm{f+g}_\infty = \norm f_\infty +
\norm g_\infty $ assim que as duas funções atingem o pico
compatível em um ponto comum (ou ao longo de uma sequência comum):
$ f = g\,$ perto de um ponto é suficiente, independentemente do comportamento
em outro lugar. A convexidade estrita do $ L^p $ bolas para $1 <
p < \infty $ --- e sua falha nos pontos finais --- é
exactamente o que estes casos de igualdade testemunham.
\end{solution}

\begin{solution}{pb:b3:lp:1}
\textbf{1.} $ f $ tem suporte em alguns $[\alpha, \beta]
\subseteq \intoo0\infty $, então $ F = 0$ sobre $[0, \alpha]$ e $ F
\equiv F(\beta)$ sobre $[\beta, \infty)$: $ Hf $ desaparece perto $0$
e é $ O(1/x)$ no infinito; $\int_1^\infty x^{-p}\dd x <
\infty $ para $ p > 1$, e $ Hf $ é \omterm{def:b3:topology:continuity}{contínuo}: $ Hf \in L^p $.

\textbf{2.} $\bigl(x^{1-p}F(x)^p\bigr)' = (1-p)x^{-p}F^p +
p\,x^{1-p}F^{p-1}f $. Ambos os valores limite desaparecem: em $0$
porque $ F = 0$ aproximar $0$; no $\infty $ porque $ x^{1-p}F^p \leq
F(\beta)^p x^{1-p} \to 0$ ($ p > 1$). Integrando a identidade
acabou $\intoo0\infty $:
\[
0 = (1 - p)\int_0^\infty\Bigl(\frac Fx\Bigr)^p\dd x
+ p\int_0^\infty\Bigl(\frac Fx\Bigr)^{p-1}f(x)\,\dd x,
\]
qual é a relação exibida.

\textbf{3.} Hölder com expoentes $ q = \frac p{p-1}$ e $ p $:
\[
\int\Bigl(\frac Fx\Bigr)^{p-1}f
\leq \Bigl(\int\Bigl(\frac
Fx\Bigr)^{p}\Bigr)^{1 - 1/p}\,\norm f_p ,
\]
então $\norm{Hf}_p^p \leq \frac p{p-1}\norm{Hf}_p^{p-1}\norm
f_p $; dividir (finito pela questão 1, e se $0$ existe
nada a provar).

\textbf{4.} Deixar $ f \in L^p $, $ f \geq 0$. Escolher $\varphi_k \in
\mathcal C_c(\intoo0\infty)$ com $\varphi_k \to f $ em $ L^p $
(\cref{thm:b3:lp:density}(2), intersectado com o aberto
meia linha --- aproximado $ f\mathbf 1_{[1/k, k]}$ e
diagonalizar) e substituir $\varphi_k $ por $\abs{\varphi_k}$
(ainda \omterm{def:b3:topology:continuity}{contínuo}, e mais perto de $ f \geq 0$). Para cada fixo
$ x > 0$, Hölder em $\intoo0x $ dá
\[
\abs{H\varphi_k(x) - Hf(x)} \leq
\frac1x\,x^{1 - 1/p}\,\norm{\varphi_k - f}_p \to 0 :
\]
$ H\varphi_k \to Hf $ pontualmente. Fatou e pergunta 3:
\[
\int (Hf)^p \leq \liminf_k\int(H\varphi_k)^p
\leq \Bigl(\frac p{p-1}\Bigr)^p\liminf_k\norm{\varphi_k}_p^p
= \Bigl(\frac p{p-1}\Bigr)^p\norm f_p^p .
\]
Para assinado ou complexo $ f $: $\abs{Hf} \leq H\abs f $ pontualmente,
e o caso não negativo se aplica a $\abs f $.

\textbf{5.} $\norm{f_A}_p^p = \int_1^A t^{-1}\dd t = \ln A $.
Para $1 \leq x \leq A $:
\[
(Hf_A)(x) = \frac1x\int_1^x t^{-1/p}\dd t
= \frac{x^{1 - 1/p} - 1}{(1 - \tfrac1p)\,x}
= \frac p{p-1}\,x^{-1/p}\bigl(1 - x^{-(1 - 1/p)}\bigr).
\]

\textbf{6.} Consertar $\varepsilon > 0$ e $ X_0$ com $(1 -
x^{-(1-1/p)})^p \geq 1 - \varepsilon $ para $ x \geq X_0$. Então
\[
\norm{Hf_A}_p^p \geq \int_{X_0}^A\Bigl(\frac
p{p-1}\Bigr)^p\frac{1 - \varepsilon}{x}\,\dd x
= \Bigl(\frac p{p-1}\Bigr)^p(1 - \varepsilon)\,(\ln A - \ln
X_0),
\]
então $\dfrac{\norm{Hf_A}_p^p}{\norm{f_A}_p^p} \geq \bigl(\frac
p{p-1}\bigr)^p(1 - \varepsilon)\bigl(1 - \frac{\ln X_0}{\ln
A}\bigr) \to \bigl(\frac p{p-1}\bigr)^p(1 - \varepsilon)$ como
$ A \to \infty $: a constante não pode ser melhorada.

\textbf{7.} A igualdade na questão 3 força a igualdade em
Hölder: $ f^p $ proporcional a $\bigl(\frac Fx\bigr)^{(p-1)q}
= \bigl(\frac Fx\bigr)^p $ a.e., ou seja\ $ f = \gamma\,\frac Fx $
a.e.\ para alguns $\gamma \geq 0$. Desde $ F(x) = \int_0^xf $ é
absolutamente \omterm{def:b3:topology:continuity}{contínuo} com $ F' = f $ a.e., $ F $ resolve $ F' =
\gamma F/x $: em qualquer intervalo onde $ F > 0$, $(\ln F)' =
\gamma/x $, então $ F = c\,x^{\gamma}$ e $ f = c\gamma
x^{\gamma - 1}$ lá. Mas nenhuma potência diferente de zero $ x^{\gamma-1}$
pertence a $ L^p(\intoo0\infty)$ ($ p(\gamma - 1) < -1$ necessário
em $\infty $ e $> -1$ no $0$: incompatível) e $ F $ não pode
desaparecem de forma idêntica, a menos que $ f = 0$. Portanto, a igualdade exige $ f =
0$.

\textbf{8.} Dado $(a_n)$ não crescente $\geq 0$, defina o
função de passo $ g(t) = a_{\lceil t\rceil}$ sobre
$\intoo0{+\infty}$: não crescente, com $\int_0^\infty g^p =
\sum_na_n^p $ e $\int_0^ng = a_1 + \dots + a_n $, então
$(Hg)(n) = \frac{a_1 + \dots + a_n}n $. A média de um
função não crescente é não crescente, então $ Hg $ é, e
\[
\sum_{n\geq1}\Bigl(\frac{a_1{+}\dots{+}a_n}n\Bigr)^p
= \sum_{n\geq1}(Hg)(n)^p
\leq \sum_{n\geq1}\int_{n-1}^n (Hg)(t)^p\,\dd t
= \norm{Hg}_p^p
\leq \Bigl(\frac p{p-1}\Bigr)^p\sum_n a_n^p
\]
pela Parte I. Para uma sequência geral não negativa, deixe $(a_n^*)$
ser o seu rearranjo não crescente (possível quando $ a_n \to 0$,
que podemos assumir --- caso contrário ambos os lados são infinitos):
o lado direito permanece inalterado, e cada soma parcial $ a_1 + \dots
+ a_n $ é no \omterm{def:b3:rings:primemaximal}{máximo} $ a_1^* + \dots + a_n^*$ (o $ n $ maior
termos): o lado esquerdo só cresce. Daí a desigualdade para todos
$(a_n)$.

\textbf{9.} Se $(a_n) \in \ell^p $, a sequência de Cesàro
significa está em $\ell^p $ com norma $\leq \frac p{p-1}\norm
a_p $. Exemplo ($ p = 2$): $ a_n = \frac1{\sqrt n\,\ln n}$ ($ n
\geq 2$): $\sum a_n^2 = \sum\frac1{n\ln^2n} < \infty $, ainda
$\sum a_n = \infty $ (teste integral); Hardy ainda garante
$\sum_n\bigl(\frac{a_1 + \dots + a_n}n\bigr)^2 < \infty $.

\textbf{10.} Para $ f = \mathbf 1_{\intcc01}$: $ Hf(x) = 1$ sobre
$\intoc01$ e $= \frac1x $ para $ x \geq 1$: $\int_0^\infty Hf =
1 + \int_1^\infty\frac{\dd x}x = \infty $, enquanto $\norm f_1 =
1$. A prova entra em colapso em dois pontos: a constante
$\frac p{p-1}$ explode como $ p \to 1$, e o termo limite
$ x^{1-p}F^p $ não desaparece mais no infinito por $ p = 1$.
A desigualdade de Hardy é honesta $ p > 1$ fenómeno.

\textbf{11.} Escolha o intervalo mais longo $ B_{i_1}$; descartar
cada intervalo que o atende; escolha o sobrevivente mais longo
$ B_{i_2}$; iterar (finitos intervalos). Os escolhidos
são disjuntos por construção, e cada descartado $ B $ conheci um
intervalo escolhido pelo menos tão longo: um intervalo que atende a
mais longo ou igual está contido em seu triplo, $ B \subseteq
3B_{i_l}$.

\textbf{12.} $\{Mf > t\}$ está aberto: cada média $ x \mapsto
\frac1{2r}\int_{x-r}^{x+r}\abs f $ é \omterm{def:b3:topology:continuity}{contínuo} (dominado
convergência em $ x $) e um supremo de \omterm{def:b3:topology:continuity}{contínuo} funções
é inferior semicontínuo. Cada $ x $ nele possui $ B_x =
\intoo{x-r_x}{x+r_x}$ com $\int_{B_x}\abs f >
t\,\lambda(B_x)$. Para compactar $ K \subseteq \{Mf > t\}$:
finitos $ B_x $ cobrir $ K $, Vitali (questão 11) extrai
disjunto $ B_1', \dots, B_k'$ com $ K \subseteq
\bigcup_l3B_l'$, então
\[
\lambda(K) \leq 3\sum_l\lambda(B_l') <
\frac3t\sum_l\int_{B_l'}\abs f \leq \frac3t\,\norm f_1
\]
por desarticulação; regularidade interna
(\cref{thm:b3:measure:regularity}) conclui.

\textbf{13.} Para $ x > 1$: com $ r \in \intcc{x-1}x $ o
média é $\frac{r - x + 1}{2r}$, aumentando em $ r $; para $ r
\geq x $ isso é $\frac1{2r}$, decrescente: o supremo é
$\frac1{2x}$, alcançado em $ r = x $. Então $ M\mathbf 1_{\intcc01}
\notin L^1$. Em geral, se $\int_I\abs f = c > 0$ em um
intervalo limitado $ I \subseteq \intcc{-C}C $, então $ Mf(x) \geq
\frac{c}{2(\abs x + C)}$ para todos $ x $: nunca \omterm{def:b3:lebesgue:l1}{integrável}
a menos que $ f = 0$ a.e.

\textbf{14.} Com $ f_t = f\,\mathbf 1_{\abs f > t/2}$:
$ M(f - f_t) \leq \frac t2$, então $\{Mf > t\} \subseteq \{Mf_t
> \frac t2\}$ e a questão 12 dá $\lambda(Mf > t) \leq
\frac6t\int_{\abs f > t/2}\abs f $. Bolo de camadas
(\cref{prop:b3:product:layercake}) e Tonelli:
\[
\norm{Mf}_p^p = p\int_0^\infty t^{p-1}\lambda(Mf > t)\dd t
\leq 6p\int\abs f\int_0^{2\abs f}t^{p-2}\,\dd t\,\dd\lambda
= \frac{6p\,2^{p-1}}{p-1}\,\norm f_p^p .
\]

\textbf{15.} Escrever $ A_rf(x) = \frac1{2r}\int_{x-r}^{x+r}f $.
Para \omterm{def:b3:topology:continuity}{contínuo} $ g $: $ A_rg(x) \to g(x)$ em todos os lugares. Dado
$\varepsilon > 0$, dividir $ f = g + h $ com $ g $ \omterm{def:b3:topology:continuity}{contínuo} de
compactar suporte e $\norm h_1 < \varepsilon $
(\cref{thm:b3:lp:density}); então
\[
\limsup_{r\to0}\,\abs{A_rf(x) - f(x)} \leq Mh(x) +
\abs{h(x)},
\]
então $\Omega_\delta = \{\limsup_r\abs{A_rf - f} > \delta\}
\subseteq \{Mh > \tfrac\delta2\} \cup \{\abs h >
\tfrac\delta2\}$ tem \omterm{def:b3:measure:measure}{medir} $\leq \frac{6\varepsilon}\delta
+ \frac{2\varepsilon}\delta $ (questão 12; Markov).
$\varepsilon $ arbitrário: $\lambda(\Omega_\delta) = 0$; união
acabou $\delta = \frac1k $: $ A_rf \to f $ a.e.

\textbf{16.} (a) Para cada $ q \in \Q $, questão 15 aplicada a
$\abs{f - q}$ dá $ A_r\abs{f - q}(x) \to \abs{f(x) - q}$
e.e.; na intersecção destes medida completa conjuntos, escolha
$ q $ com $\abs{f(x) - q} < \eta $:
$\limsup_rA_r\abs{f - f(x)}(x) \leq 2\eta $ para cada $\eta $:
quase todos $ x $ é um ponto de Lebesgue. (b) Em um Lebesgue
ponto,
\[
\Bigl|\frac{F(x + h) - F(x)}h - f(x)\Bigr|
= \Bigl|\frac1h\int_x^{x+h}\bigl(f - f(x)\bigr)\Bigr|
\leq 2\,A_{\abs h}\abs{f - f(x)}(x) \to 0 :
\]
$ F' = f $ a.e.\ --- primitivas de $ L^1$ funções fazem
diferenciar de volta; a escada (\cref{pb:b3:measure:1})
é o contra-exemplo para o \emph{conversar} somente direção.

\textbf{17.} Aplique a questão 15 a $\mathbf 1_{A\cap[-n,n]}$
e deixe $ n $ crescer: para a.e.\ $ x \in A $ o \omterm{ex:b3:lebesgue:gamma}{densidade}
$\frac{\lambda(A\cap\intcc{x-r}{x+r})}{2r} \to 1$. Steinhaus:
em torno de um \omterm{ex:b3:lebesgue:gamma}{densidade} ponto de tomada $ r $ com \omterm{ex:b3:lebesgue:gamma}{densidade} $> \frac34$;
para $\abs t < \frac r2$, $ A $ e $ A + t $ cada um preenche mais de
$\frac32r $ de um intervalo de comprimento $\leq \frac52r $, portanto
cruzar: $\intoo{-r/2}{r/2} \subseteq A - A $.

\textbf{18.} Deixar $ G(x) = x^{\alpha+1-p}F(x)^p $: $ G $ desaparece
em $0$ ($ F $ desaparece perto $0$) e em $\infty $ ($ F $ limitado,
$\alpha + 1 - p < 0$), então $\int_0^\infty G' = 0$ com
\[
G' = (\alpha + 1 - p)\,x^{\alpha-p}F^p +
p\,x^{\alpha+1-p}F^{p-1}f
= (\alpha + 1 - p)\Bigl(\frac Fx\Bigr)^px^\alpha +
p\Bigl(\frac Fx\Bigr)^{p-1}f\,x^\alpha .
\]
Por isso $\int(F/x)^px^\alpha = \frac{p}{p-1-\alpha}
\int(F/x)^{p-1}f\,x^\alpha $; Hölder para o \omterm{def:b3:measure:measure}{medir}
$ x^\alpha\dd x $ (expoentes $\frac p{p-1}$ e $ p $) termina
como na questão 3. Borderline $\alpha = p - 1$: com $ f(t) =
\frac1t\mathbf 1_{\intcc1A}$, o lado direito é $\ln A $
enquanto a esquerda contém $\int_1^A\frac{(\ln x)^p}x\dd x =
\frac{(\ln A)^{p+1}}{p+1}$: nenhuma constante sobrevive $ A \to
\infty $.

\textbf{19.} Tonelli em $\{0 < t < x\}$:
\[
\langle Hf, g\rangle =
\int_0^\infty\frac{g(x)}x\int_0^xf(t)\,\dd t\,\dd x
= \int_0^\infty f(t)\int_t^\infty\frac{g(x)}x\,\dd x\,\dd t
= \langle f, H^*g\rangle .
\]
Dualidade: $\norm{H^*f}_p = \sup\{\langle f, Hg\rangle : g
\geq 0, \norm g_q \leq 1\} \leq \norm f_p\cdot
\sup\norm{Hg}_q \leq \frac q{q-1}\norm f_p = p\,\norm f_p $,
Hardy sendo invocado em $ L^q $, cuja constante $\frac q{q-1}$
é igual $ p $.

\textbf{20.} Divida ao longo da diagonal (nula). Sobre $\{y \leq
x\}$:
\[
\iint_{y\leq x}\frac{f(x)g(y)}{x}\,\dd y\,\dd x
= \int f(x)\,(Hg)(x)\,\dd x \leq \norm f_p\,\norm{Hg}_q
\leq p\,\norm f_p\norm g_q,
\]
desde Hardy em $ L^q $ carrega a constante $\frac q{q-1} =
p $. Simetricamente, $\iint_{y>x} = \int g\,(Hf) \leq
\norm g_q\norm{Hf}_p \leq q\,\norm f_p\norm g_q $ ($\frac
p{p-1} = q $). Total: $(p + q)\,\norm f_p\norm g_q $.

\textbf{21.} Para $ a_n = n^{-1/p}$, $ n \leq N $: a direita
lado é $\bigl(\frac p{p-1}\bigr)^p\sum_{n\leq N}\frac1n =
\bigl(\frac p{p-1}\bigr)^p\ln N + O(1)$. À esquerda, por $ n
\leq N $: $ a_1 + \dots + a_n \geq \int_1^{n+1}t^{-1/p}\dd t =
\frac{p}{p-1}\bigl((n+1)^{1-1/p} - 1\bigr)$, então o $ n $-º
Cesàro significa é $\geq \frac p{p-1}n^{-1/p}(1 -
o(1))$ uniformemente para $ n $ em qualquer faixa $ n \geq n_0(\eta)$;
elevando para o $ p $ e somando, o lado esquerdo é $\geq
\bigl(\frac p{p-1}\bigr)^p(1 - \eta)\ln N + O_\eta(1)$.
Dividindo e deixando $ N \to \infty $, então $\eta \to 0$: não
constante menor que $\bigl(\frac p{p-1}\bigr)^p $ pode funcionar.

\textbf{22.} $ H $ delimitado $ L^p $: médias cumulativas fazem
não inflar $ p $-normas (constante $\frac p{p-1}$); Cesàro ligado
$\ell^p $: o mesmo, discretizado; $ M $ delimitado $ L^p $: mesmo
a melhor média local permanece sob controle (constante
$ O(\frac1{p-1})$). Todos os três falham em $ p = 1$, e por um
razão: calcular a média de uma unidade concentrada de massa produz um
$\frac1x $ cauda (questões 10 e 13), e $\frac1x $ pertence
para cada $ L^p $ perto do infinito, exceto $ L^1$. Suavização de spreads
massa exatamente na fronteira harmônica de integrabilidade.

\textbf{23.} Definir $ b_n = a_n^{1/p}$, então $\sum b_n^p = \sum
a_n $. A questão 8 dá
\[
\sum_{n\geq1}\Bigl(\frac{b_1 + \dots + b_n}n\Bigr)^{p}
\leq \Bigl(\frac p{p-1}\Bigr)^{p}\sum_{n\geq1}a_n,
\]
e AM--GM limita cada soma abaixo:
\[
\Bigl(\frac{b_1 + \dots + b_n}n\Bigr)^{p}
\geq \bigl(b_1\cdots b_n\bigr)^{p/n}
= \bigl(a_1\cdots a_n\bigr)^{1/n}.
\]
Por isso $\sum(a_1\cdots a_n)^{1/n} \leq \bigl(\frac
p{p-1}\bigr)^p\sum a_n $ para \emph{todo} $ p > 1$. Com $ m =
p - 1$,
\[
\Bigl(\frac p{p-1}\Bigr)^{p} =
\Bigl(1 + \frac1m\Bigr)^{m+1},
\]
que diminui para $\eu $ como $ m \to \infty $ (o clássico
sequência superior monótona para $\eu $). Assumindo o ínfimo
$ p $ dá a desigualdade de Carleman com constante $\eu $.

\textbf{24.} Para $ a_n = \frac1n $, $(a_1\cdots a_n)^{1/n} =
(n!)^{-1/n}$. O escalonamento de \cref{pb:b3:product:1} dá
$\sqrt{2\pi n}\,(n/\eu)^n \leq n! \leq \sqrt{2\pi
n}\,(n/\eu)^n\eu^{1/(12n)}$, então
\[
(n!)^{1/n} = \frac n\eu\,(2\pi n)^{1/(2n)}
\eu^{O(1/n^2)}
= \frac n\eu\Bigl(1 +
O\Bigl(\frac{\ln n}{n}\Bigr)\Bigr),
\]
desde $(2\pi n)^{1/(2n)} = \exp\bigl(\frac{\ln(2\pi
n)}{2n}\bigr)$. Invertendo, $(n!)^{-1/n} = \frac\eu n(1 +
O(\frac{\ln n}n))$, e resumindo $ n \leq N $:
\[
\sum_{n\leq N}(n!)^{-1/n}
= \eu\ln N + O(1),
\qquad
\eu\sum_{n\leq N}\frac1n = \eu\ln N + O(1)
\]
(a série de erros $\sum\frac{\ln n}{n^2}$ converge). Um
Desigualdade de Carleman com constante $ c $ forçaria $\eu\ln N
+ O(1) \leq c\,(\ln N + O(1))$, por isso $ c \geq \eu $ sobre
dividindo por $\ln N $. As sequências de otimização se alinham:
A constante de Hardy é aproximada por $ n^{-1/p}$ (questão 21),
Carleman por $ n^{-1}$ --- em cada caso o tipo harmônico
sequência situada fora do espaço que está sendo calculado.

\textbf{25.} Para $ f = \mathbf 1_{\intcc01}$: $ Hf(x) = 1$ sobre
$\intoc01$ e $ Hf(x) = \frac1x $ para $ x > 1$, então
$\norm{Hf}_2^2 = 1 + \int_1^\infty x^{-2}\dd x = 2$ e o
proporção é $\sqrt2 \approx 1.414$, sobre $71\%$ do afiado
vinculado. Para $ f_A $ ($ p = 2$): $ F(x) = 2(\sqrt x - 1)$ sobre
$\intcc1A $, então nesse intervalo $ Hf_A = 2x^{-1/2} - 2x^{-1}$
e, por $ x > A $, $ Hf_A(x) = 2(\sqrt A - 1)/x $. Quadratura
e integrando,
\begin{align*}
\int_1^A\bigl(2x^{-1/2} - 2x^{-1}\bigr)^2\dd x
&= 4\ln A - 12 + \frac{16}{\sqrt A} - \frac4A,\\
\int_A^{\infty}\frac{4(\sqrt A - 1)^2}{x^2}\,\dd x
&= 4 - \frac{8}{\sqrt A} + \frac4A,
\end{align*}
de onde $\norm{Hf_A}_2^2 = 4\ln A - 8 + 8A^{-1/2}$; dividindo
por $\norm{f_A}_2^2 = \ln A $ fornece a identidade declarada. No $ A
= \eu^{10}$: $4 - \frac{8(1 - \eu^{-5})}{10} = 3.2054$, então
a proporção é $\sqrt{3.2054} \approx 1.790 < 2$. O defeito
$4 - \norm{Hf_A}_2^2/\norm{f_A}_2^2 \sim 8/\ln A $ decai
apenas logaritmicamente: para alcançar a proporção $1.99$ alguém precisaria
$\ln A \approx 200$, ou seja\ $ A \approx 10^{87}$. Afiado
constantes são teoremas, não experimentos.
\end{solution}
