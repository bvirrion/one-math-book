\section*{Capítulo \ref{ch:b3:residues} --- Série Laurent e
o Teorema do Resíduo}

\begin{solution}{exo:b3:residues:1}
$\frac{\sin z}z = 1 - \frac{z^2}6 + \cdots $: removível,
resíduo $0$. $\frac{\eu^z - 1}{z^2} = \frac1z + \frac12 +
\frac z6 + \cdots $: poste \omterm{def:b3:groups:simple}{simples}, resíduo $1$.
$\frac1{z(z-1)^2}$: poste \omterm{def:b3:groups:simple}{simples} em $0$ com resíduo
$\frac1{(0-1)^2} = 1$; pólo duplo em $1$ com resíduo
$\frac{\dd}{\dd z}\bigl(\frac1z\bigr)\big|_{z=1} = -1$.
$\frac{\cos z}{z^3} = \frac1{z^3} - \frac1{2z} + \cdots $: pólo
de ordem $3$, resíduo $-\frac12$. $\eu^{1/z} = \sum_{n\geq0}
\frac{z^{-n}}{n!}$: essencial, resíduo $1$. $\frac1{\sin z}$:
postes \omterm{def:b3:groups:simple}{simples} em $ n\pi $; resíduo $\frac1{\cos 0} = 1$ no $0$,
$\frac1{\cos\pi} = -1$ no $\pi $
(\cref{met:b3:residues:compute}, $ g/h'$).
\end{solution}

\begin{solution}{exo:b3:residues:2}
Com $ z = \eu^{\iu t}$, $\cos t = \frac{z + z^{-1}}2$, $\dd t
= \frac{\dd z}{\iu z}$:
\[
\int_0^{2\pi}\frac{\dd t}{a + \cos t}
= \oint_{\abs z = 1}\frac{2\,\dd z}{\iu\,(z^2 + 2az + 1)} .
\]
As raízes $ z_\pm = -a \pm \sqrt{a^2 - 1}$ satisfazer $ z_+z_- =
1$ com $\abs{z_+} < 1 < \abs{z_-}$; o resíduo no $ z_+$ é
$\frac{1}{z_+ - z_-} = \frac1{2\sqrt{a^2-1}}$, então a integral
é $\frac2\iu\cdot2\iu\pi\cdot\frac1{2\sqrt{a^2-1}} =
\frac{2\pi}{\sqrt{a^2-1}}$. Como $ a \to 1^+$ explode (o
O integrando atinge o pico em $ t = \pi $); como $ a \to \infty $ ele se comporta
gosto $\frac{2\pi}a $, correspondência $\int\frac{\dd t}a $.
\end{solution}

\begin{solution}{exo:b3:residues:3}
Primeira integral: pólos superiores de $\frac{z^2}{1+z^6}$ no $ p =
\eu^{\iu\pi/6}, \iu, \eu^{5\iu\pi/6}$; em cada um,
$\operatorname{Res} = \frac{p^2}{6p^5} = \frac{p^3}{6p^6} =
-\frac{p^3}6$, e $ p^3$ leva os valores $\iu, -\iu, \iu $:
soma de resíduos $-\frac{\iu}{6}$. O arco é
$ O(R^{-4})\cdot O(R) \to 0$:
\[
\int_\R\frac{x^2\,\dd x}{1 + x^6} =
2\iu\pi\Bigl(-\frac \iu6\Bigr) = \frac\pi3 .
\]
Segundo: pólo duplo em $\iu $ de $\frac1{(1+z^2)^2} =
\frac1{(z-\iu)^2(z+\iu)^2}$:
\[
\operatorname{Res} = \frac{\dd}{\dd z}\,(z +
\iu)^{-2}\Big|_{z=\iu} = \frac{-2}{(2\iu)^3} =
\frac{-2}{-8\iu} = -\frac\iu4,
\qquad
\int_\R\frac{\dd x}{(1+x^2)^2} =
2\iu\pi\cdot\Bigl(-\frac\iu4\Bigr) = \frac\pi2 ,
\]
consistente com \cref{exo:b3:fouriertransform:3}(b).
\end{solution}

\begin{solution}{exo:b3:residues:4}
Fechar $\frac{\eu^{\iu tz}}{z^2 + a^2}$ no semiplano superior
($ t \geq 0$): lá $\abs{\eu^{\iu tz}} =
\eu^{-t\operatorname{Im}z} \leq 1$, então o arco contribui
$ O(R^{-2})\cdot O(R) \to 0$. O único poste fechado $\iu a $
é \omterm{def:b3:groups:simple}{simples} com resíduo $\frac{\eu^{-at}}{2\iu a}$:
\[
\int_\R\frac{\eu^{\iu tx}}{x^2 + a^2}\dd x =
\frac{\pi}{a}\,\eu^{-at},
\qquad\text{hence}\qquad
\int_\R\frac{\cos(tx)}{x^2+a^2}\dd x = \frac\pi
a\,\eu^{-a\abs t}
\]
(parte real; mesmo em $ t $). Esta é a contrapartida da inversão
de $\widehat{\eu^{-a\abs x}} = \frac{2a}{a^2+\xi^2}$
(\cref{exo:b3:fouriertransform:1}): os dois cálculos
confirmar um ao outro através
\cref{thm:b3:fouriertransform:inversion}.
\end{solution}

\begin{solution}{exo:b3:residues:5}
Frações parciais: $ f = \frac1{z-2} - \frac1{z-1}$.
Sobre $\abs z < 1$ (Taylor):
$ f = \sum_{n\geq0}\bigl(1 - 2^{-n-1}\bigr)z^n $.
Sobre $1 < \abs z < 2$: $\frac1{z-2} =
-\sum_{n\geq0}\frac{z^n}{2^{n+1}}$ e $-\frac1{z-1} =
-\sum_{n\geq1}z^{-n}$: uma série genuína de dois lados.
Sobre $\abs z > 2$: $ f = \sum_{n\geq1}\bigl(2^{n-1} -
1\bigr)z^{-n}$. Os três diferem porque as expansões de Laurent
estão anexados a \emph{anéis}, não pontos: cada região tem seu
próprias expansões geométricas. O $ z^{-1}$-coeficientes ($-1$ e
$0$ respectivamente) são integrais sobre círculos que circundam
\emph{as singularidades dentro}, não resíduos no $0$ ($ f $ é
\omterm{def:b3:holomorphic:holo}{holomórfico} no $0$): para $1 < \abs z < 2$ o coeficiente
$-1$ é $\operatorname{Res}(f, 1)$; para $\abs z > 2$ o
coeficiente $0$ é $\operatorname{Res}(f,1) +
\operatorname{Res}(f,2) = -1 + 1$. O resíduos de $ f $: $-1$
no $1$ e $+1$ no $2$.
\end{solution}

\begin{solution}{exo:b3:residues:6}
(a) O \omterm{thm:b3:residues:laurent}{Série Laurent} $\sum_nz^{-n}/n!$ tem infinitos
termos negativos: essencial. Resolvendo $\eu^{1/z} = w $ ($ w \neq
0$): $\frac1z = \log\abs w + \iu\arg w + 2\iu\pi k $, então
\[
z_k = \frac1{\log\abs w + \iu\arg w + 2\iu\pi k}
\xrightarrow[k\to\infty]{} 0 :
\]
todo valor diferente de zero é alcançado infinitamente perto $0$ ---
mais forte que \omterm{ex:b3:lebesgue:gamma}{densidade}.
(b) Junto $ z = 1/x $, $ x \to +\infty $: $\eu^x \to \infty $;
junto $ z = \iu/y $: módulo $1$. Um poste requer
$\abs{f(z)} \to \infty $ \emph{ao longo de cada abordagem}: aqui o
limite simplesmente não existe, mesmo em $[0, +\infty]$.
\end{solution}

\begin{solution}{exo:b3:residues:7}
(a) Ligado $\abs z = 1$: $\abs{z^7 + z - 1} \leq 3 < 4 =
\abs{-4z^3}$. Rouché com $ f = -4z^3$, $ g = z^7 + z - 1$:
três zeros no disco.
(b) Ligado $\abs z = 1$: $\abs{z^4 + 1} \leq 2 < 5 = \abs{5z}$:
um zero em $\abs z < 1$. Sobre $\abs z = 2$: $\abs{5z + 1} \leq
11 < 16 = \abs{z^4}$: quatro zeros em $\abs z < 2$. Daí três
zeros no anel.
(c) Para $ P = z^n + a_{n-1}z^{n-1} + \dots $: sobre $\abs z = R >
1 + \sum\abs{a_k}$, $\abs{P - z^n} \leq
\bigl(\sum\abs{a_k}\bigr)R^{n-1} < R^n = \abs{z^n}$: $ P $ tem
exatamente $ n $ zeros em $ D(0, R)$ --- d'Alembert--Gauss com
multiplicidade, por contagem pura.
\end{solution}

\begin{solution}{exo:b3:residues:8}
(a) Suponha $ f(a) = 0$, $ f \not\equiv 0$: escolher $ r $ com $ f $
\omterm{def:b3:modules:free}{livre} de zero no círculo $ C = \partial D(a, r)$ (isolado
zeros) e $ m = \min_C\abs f > 0$. Por
\cref{thm:b3:holomorphic:weierstrassconv}, $ f_n \to f $ e
$ f_n' \to f'$ uniformemente ligado $ C $; para grande $ n $, $\abs{f_n}
\geq m/2$ sobre $ C $, então
\[
\frac1{2\iu\pi}\int_C\frac{f_n'}{f_n}
\longrightarrow \frac1{2\iu\pi}\int_C\frac{f'}{f} \geq 1
\]
(o limite conta o zero $ a $; convergência porque
os numeradores convergem uniformemente e os denominadores são uniformemente
delimitado abaixo). O lado esquerdo é um número inteiro contando zeros de
$ f_n $ no disco: deve ser $\geq 1$ eventualmente ---
contradizendo a liberdade zero. Então $ f $ é \omterm{def:b3:modules:free}{livre} de zero.

(b) Correção $ w \in \Omega $ e aplicar (a) no \omterm{def:b3:topology:connected}{conectado} conjunto aberto
$\Omega\setminus\{w\}$ (removendo um ponto de abertura \omterm{def:b3:topology:connected}{conectado}
subconjunto de $\C $ conservas conectividade) para $ g_n(z) = f_n(z) -
f_n(w)$, \omterm{def:b3:modules:free}{livre} de zero por injetividade, convergindo para $ g = f
- f(w)$. Se $ f $ é inconstante, $ g \not\equiv 0$ sobre
$\Omega\setminus\{w\}$, então $ g $ é \omterm{def:b3:modules:free}{livre} de zero lá: $ f(z) \neq
f(w)$ para todos $ z \neq w $. Como $ w $ foi arbitrário, $ f $ é
injetivo.
\end{solution}

\begin{solution}{exo:b3:residues:9}
A fronteira do setor consiste em $[0, R]$, o arco $ A_R $ e
o raio $\eu^{2\iu\pi/n}[0, R]$ invertido. Dentro está o
pólo único $ p = \eu^{\iu\pi/n}$ de $\frac1{1+z^n}$, com
resíduo $\frac1{np^{n-1}} = \frac{p}{np^n} = -\frac pn $. Ligado
o raio de retorno, $ z = \eu^{2\iu\pi/n}x $ dá $ z^n = x^n $ e
$\dd z = \eu^{2\iu\pi/n}\dd x $; o arco é
$ O(R^{-n})\cdot O(R) \to 0$. Por isso
\[
\bigl(1 - \eu^{2\iu\pi/n}\bigr)
\int_0^\infty\frac{\dd x}{1 + x^n}
= 2\iu\pi\Bigl(-\frac{\eu^{\iu\pi/n}}n\Bigr),
\quad\text{so}\quad
\int_0^\infty\frac{\dd x}{1+x^n}
= \frac{2\iu\pi}{n\,\bigl(\eu^{\iu\pi/n} -
\eu^{-\iu\pi/n}\bigr)} = \frac{\pi}{n\sin(\pi/n)} .
\]
$ n = 2$: $\frac\pi{2\sin(\pi/2)} = \frac\pi2 =
[\arctan]_0^\infty $. Como $ n \to \infty $: o valor tende a
$1$, e de fato o integrando tende a $\mathbf 1_{\intco01}$
(com dominação DCT $\min(1, x^{-2})$ para $ n \geq 2$).
\end{solution}

\begin{solution}{exo:b3:residues:10}
Sobre $\abs z = R $ grande, $\abs f \leq C R^{-2}$: $\abs{\oint_{C_R}
f} \leq 2\pi R\cdot CR^{-2} \to 0$. Mas para $ R $ além de tudo
pólos, o resíduo teorema dá $\oint_{C_R}f =
2\iu\pi\sum_{\text{all }p}\operatorname{Res}(f, p)$: o total
soma desaparece. Para $ f = \frac1{z(z-1)(z-2)}$: resíduos
$\frac1{(-1)(-2)} = \frac12$ no $0$, $\frac1{1\cdot(-1)} = -1$
no $1$, $\frac1{2\cdot1} = \frac12$ no $2$ --- somando para $0$
como previsto, e
\[
\frac1{z(z-1)(z-2)} = \frac{1/2}{z} - \frac1{z - 1} +
\frac{1/2}{z-2} :
\]
o resíduos \emph{são} os coeficientes de fração parcial, e
a identidade de soma zero fornece uma verificação de consistência gratuita (ou
determina o último coeficiente dos demais).
\end{solution}

\begin{solution}{exo:b3:residues:11}
No buraco da fechadura, com a determinação escolhida:
logo acima do corte, $\log z = \ln x $; logo abaixo, $\log z =
\ln x + 2\iu\pi $. As quatro peças dão
\[
\Bigl(1 - \eu^{2\iu\pi(a-1)}\Bigr)\int_\varepsilon^R
\frac{x^{a-1}}{1+x}\dd x + \int_{C_R} + \int_{C_\varepsilon}
= 2\iu\pi\operatorname{Res}(f, -1) .
\]
Arcos: $\abs{f} \leq \frac{R^{a-1}}{R - 1}$ sobre $ C_R $, comprimento
$2\pi R $: contribuição $ O(R^{a-1}) \to 0$ ($ a < 1$);
$\abs f \leq \frac{\varepsilon^{a-1}}{1 - \varepsilon}$ sobre
$ C_\varepsilon $, comprimento $2\pi\varepsilon $: $ O(\varepsilon^a)
\to 0$ ($ a > 0$). Resíduo: no $ z = -1 = \eu^{\iu\pi}$,
$\operatorname{Res} = \eu^{(a-1)\iu\pi}$. Por isso
\[
I(a) = \frac{2\iu\pi\,\eu^{\iu\pi(a-1)}}{1 -
\eu^{2\iu\pi(a-1)}}
= \frac{2\iu\pi}{\eu^{-\iu\pi(a-1)} - \eu^{\iu\pi(a-1)}}
= \frac{\pi}{-\sin(\pi(a-1))} = \frac{\pi}{\sin\pi a} .
\]
Reflexão gama: $ B(a, 1-a) =
\int_0^1t^{a-1}(1-t)^{-a}\dd t $; a substituição $ t =
\frac x{1+x}$, $1 - t = \frac1{1+x}$, $\dd t =
\frac{\dd x}{(1+x)^2}$ transforma isso em $\int_0^\infty
\frac{x^{a-1}}{1+x}\dd x = I(a)$, e a fórmula de Euler
$ B(a, 1-a) = \Gamma(a)\Gamma(1-a)/\Gamma(1) $
(\cref{pb:b3:lebesgue:1}) dá
$\Gamma(a)\Gamma(1-a) = \frac\pi{\sin\pi a}$ --- em particular $\Gamma(\tfrac12) = \sqrt\pi $ mais uma vez.

\end{solution}

\begin{solution}{exo:b3:residues:12}
(a) Ligado $\abs z = 1$: $\abs{z^4} = 1 < 7 \leq \abs{8z + 1}$
($\abs{8z} - 1 = 7$): $ P $ tem tantos zeros em $\mathbb D $
como $8z + 1$, nomeadamente \emph{um} (no $-\frac18$).

(b) Ligado $\abs z = 3$: $\abs{8z + 1} \leq 25 < 81 =
\abs{z^4}$: Rouché contra $ z^4$ dá todos os quatro zeros em
$\abs z < 3$, por isso $4 - 1 = 3$ zeros no anel $1 <
\abs z < 3$. Afiação: um zero com $\abs z = r \geq 2.1$
satisfaria $ r^4 = \abs{8z + 1} \leq 8r + 1$, mas $ r^4 -
8r - 1$ está aumentando para $ r \geq 2$ e é igual $19.45 -
16.8 - 1 = 1.65 > 0$ no $ r = 2.1$: impossível. Então os três
zeros externos estão em $1 < \abs z < 2.1$. (Numericamente: um verdadeiro
zero próximo $-1.95$ e um par conjugado próximo $1.04 \pm
1.73\iu $, de módulo $2.02$ --- é por isso que um Rouché
tente o raio exatamente $2$ deve falhar: o teorema exige
dominação estrita, e os zeros ficam do lado de fora.)

(c) Sem zeros $\iu\R $: $\operatorname{Re}P(\iu t) = t^4 +
1 \geq 1$. Zeros no semiplano direito: use o argumento
princípio na fronteira do meio disco $\{\abs z \leq
R,\ \operatorname{Re}z \geq 0\}$. No grande arco, $\arg P
\approx \arg z^4$ passa $4\cdot\pi = 2\pi\cdot2$ (o
arco abrange ângulo $\pi $). Ao longo do eixo imaginário de
$\iu R $ até $-\iu R $: $ P(\iu t) = (t^4 + 1) + 8\iu t $
permanece no semiplano direito ($\operatorname{Re} > 0$), então
$\arg P $ varia dentro $\intoo{-\pi/2}{\pi/2}$ e retorna
com variação líquida $\to 0$ como $ R \to \infty $ (pontos finais ambos
$\approx \arg t^4 = 0$). Enrolamento total: $\frac{4\pi + 0}
{2\pi} = 2$: \emph{dois} zeros no semiplano direito ---
consistente com os números: o par conjugado $\approx
1.04 \pm 1.73\iu $ tem parte real positiva, os zeros reais
$\approx -0.125$ e $\approx -1.96$ negativo.
\end{solution}

\begin{solution}{pb:b3:residues:1}
\textbf{1.} $\sin(\pi z)$ tem zeros \omterm{def:b3:groups:simple}{simples} exatamente em $\Z $
($\sin\pi z = 0$ se $ z \in \Z $, e $(\sin\pi z)' = \pi\cos\pi
z \neq 0$ lá) e $\cos(\pi n) \neq 0$: $\pi\cot(\pi z) =
\pi\cos(\pi z)/\sin(\pi z)$ tem pólos \omterm{def:b3:groups:simple}{simples} em $\Z $ com
\[
\operatorname{Res}(\pi\cot\pi z,\ n)
= \frac{\pi\cos(\pi n)}{\pi\cos(\pi n)} = 1
\]
($ g/h'$ regra, \cref{met:b3:residues:compute}).

\textbf{2.} $\frac{\sin(\pi z)}{\pi z} = 1 - \frac{(\pi z)^2}6
+ \frac{(\pi z)^4}{120} - \cdots $ é \omterm{def:b3:holomorphic:holo}{holomórfico} e diferente de zero
perto $0$: seu recíproco é \omterm{def:b3:holomorphic:holo}{holomórfico}
(\cref{def:b3:holomorphic:holo}: quociente), com série $1 +
\frac{(\pi z)^2}{6} + \frac{7(\pi z)^4}{360} + \cdots $
(identificar: $(1 - u)^{-1}$ coeficientes de estilo de $ u =
\frac{(\pi z)^2}6 - \frac{(\pi z)^4}{120}$: o $ z^4$
coeficiente é $\frac1{36} - \frac1{120} = \frac{7}{360}$).
Multiplicar por $\cos(\pi z) = 1 - \frac{(\pi z)^2}2 + \frac{(\pi
z)^4}{24} - \cdots $ e dividir por $ z $:
\[
\pi\cot(\pi z) = \frac1z\Bigl[1 + \pi^2z^2\Bigl(\frac16 -
\frac12\Bigr) + \pi^4z^4\Bigl(\frac7{360} - \frac1{12} +
\frac1{24}\Bigr)\Bigr] + \cdots
= \frac1z - \frac{\pi^2}3\,z - \frac{\pi^4}{45}\,z^3 -
\cdots
\]
($\frac7{360} - \frac{30}{360} + \frac{15}{360} =
-\frac8{360} = -\frac1{45}$).

\textbf{3.} Lados verticais $ z = \pm(N + \frac12) + \iu y $:
por $\pi $- periodicidade de $\cot $, $\cot(\pi z) = \cot(\pm\frac\pi2
+ \iu\pi y) = -\tan(\iu\pi y) = -\iu\tanh(\pi y)$, de módulo
$\leq 1$. Lados horizontais $ z = x \pm \iu(N + \frac12)$: de
$\abs{\cot(a + \iu b)}^2 = \frac{\cos^2a +
\sinh^2b}{\sin^2a + \sinh^2b} \leq \frac{1 +
\sinh^2b}{\sinh^2b} = \coth^2 b $,
\[
\abs{\cot(\pi z)} \leq \coth\bigl(\pi(N + \tfrac12)\bigr)
\leq \coth(\pi/2) < 1.1 .
\]
Ambos os limites são $\leq 2$.

\textbf{4.} Aplicar \cref{thm:b3:residues:residue} para $ F(z) =
\pi\cot(\pi z)f(z)$ sobre $ C_N $ ($ N $ além de todos os pólos $ f $):
\[
\frac1{2\iu\pi}\oint_{C_N}F = \sum_{n=-N}^{N}f(n) +
\sum_p\operatorname{Res}(F, p),
\]
os pólos inteiros contribuindo $ f(n)$ (questão 1; $ f $
\omterm{def:b3:holomorphic:holo}{holomórfico} lá). Sobre $ C_N $: $\abs F \leq 2\pi\cdot
C\abs z^{-2} \leq 2\pi C N^{-2}$, e o perímetro é $8(N +
\frac12)$: a integral é $ O(1/N) \to 0$. Deixar $ N \to
\infty $: a fórmula de soma exibida.

\textbf{5.} A decadência $\abs f = O(\abs z^{-2})$ matou o
\omterm{def:b3:holomorphic:contour}{integral de contorno} \emph{e} feito $\sum\abs{f(n)}$ convergem.
Para $ f(z) = \frac1{z + \frac12}$: as somas simétricas
$\sum_{-N}^N\frac1{n + \frac12}$ telescópio para $0$ (os termos
$ n $ e $-n - 1$ cancelar), e o lado direito é
$-\operatorname{Res}\bigl(\frac{\pi\cot\pi z}{z + \frac12},
-\frac12\bigr) = -\pi\cot(-\frac\pi2) = 0$: consistente ---
mas $\sum\abs{f(n)}$ diverge; o método calcula o
apenas limite simétrico (valor principal).

\textbf{6.} $ g(z) = \frac{\pi\cot(\pi z)}{z^2}$: pólos no
inteiros diferentes de zero com resíduos $\frac1{n^2}$, e em $0$
onde, pela questão 2,
\[
g(z) = \frac1{z^3} - \frac{\pi^2}{3z} - \frac{\pi^4}{45}z -
\cdots :
\qquad \operatorname{Res}(g, 0) = -\frac{\pi^2}3 .
\]
O \omterm{def:b3:holomorphic:contour}{integral de contorno} sobre $ C_N $ tende a $0$ como na questão 4
($\abs{g} = O(N^{-2})$ sobre $ C_N $). Por isso $0 =
\sum_{n\neq0}\frac1{n^2} - \frac{\pi^2}3$: $2\zeta(2) =
\frac{\pi^2}3$, $\zeta(2) = \frac{\pi^2}6$.

\textbf{7.} $ g(z) = \frac{\pi\cot(\pi z)}{z^4} = \frac1{z^5}
- \frac{\pi^2}{3z^3} - \frac{\pi^4}{45z} - \cdots $: resíduo
no $0$ igual a $-\frac{\pi^4}{45}$, e $0 =
2\zeta(4) - \frac{\pi^4}{45}$: $\zeta(4) =
\frac{\pi^4}{90}$.

\textbf{8.} Com $ g = \pi\cot(\pi z)/z^{2k}$: o resíduo no
$0$ é o coeficiente $ a_{2k-1}$ de $ z^{2k-1}$ no
expansão de $\pi\cot(\pi z)$, e o contorno que desaparece dá
$2\zeta(2k) + a_{2k-1} = 0$: $\zeta(2k) = -a_{2k-1}/2$, um
múltiplo racional de $\pi^{2k}$ já que a cotangente
coeficientes são. Mais uma etapa de divisão produz $ a_5 =
-\frac{2\pi^6}{945}$, de onde $\zeta(6) = \frac{\pi^6}{945}$.
Para $\zeta(3)$: o núcleo natural $ g = \pi\cot(\pi z)/z^3$
produz $\sum_{n\neq0}\frac1{n^3} = 0$ por estranheza --- o
método prova $0 = 0$ e é estruturalmente cego para zeta estranho
valores (sem formulário fechado para $\zeta(3)$ é conhecido; é
irracionalidade, Apéry 1978, precisava de algo totalmente diferente
ideias).

\textbf{9.} $ F(z) = \frac{\pi\cot(\pi z)}{(z - w)(z + w)}$
tem pólos nos inteiros (resíduos $\frac1{n^2 - w^2}$,
observando o sinal: $ f(n) = \frac1{(n-w)(n+w)} = \frac1{n^2 -
w^2}$) e postes \omterm{def:b3:groups:simple}{simples} em $\pm w $ com resíduos
$\frac{\pi\cot(\pm\pi w)}{\pm2w} = \frac{\pi\cot(\pi
w)}{2w}$ cada ($\cot $ é estranho). Argumento da questão 4
($\abs f = O(\abs z^{-2})$) dá
\[
\sum_{n\in\Z}\frac1{n^2 - w^2} + \frac{\pi\cot(\pi w)}{w} =
0,
\qquad\text{i.e.}\qquad
\pi\cot(\pi w) = \frac1w + \sum_{n\geq1}\frac{2w}{w^2 -
n^2},
\]
(o $ n = 0$ termo é $-\frac1{w^2}$; reagrupar $\pm n $). Normais
convergência em compactos de $\C\setminus\Z $: para $\abs w \leq
R $ e $ n \geq 2R $, $\abs{\frac{2w}{w^2 - n^2}} \leq
\frac{2R}{n^2 - R^2} \leq \frac{8R}{3n^2}$.

\textbf{10.} Para $\abs w \leq r < 1$: $\frac{2w}{w^2 - n^2} =
-\frac{2w}{n^2}\cdot\frac1{1 - w^2/n^2} =
-2\sum_{k\geq0}\frac{w^{2k+1}}{n^{2k+2}}$, com
$\abs{\text{terms}} \leq 2r^{2k+1}/n^{2k+2}$, resumido em
$(n, k)$: Fubini para reorganizações de série
\[
\pi\cot(\pi w) = \frac1w -
2\sum_{k\geq0}\zeta(2k+2)\,w^{2k+1} .
\]
Correspondendo à pergunta 2: $-2\zeta(2) = -\frac{\pi^2}3$ e
$-2\zeta(4) = -\frac{\pi^4}{45}$ --- os mesmos valores. Três
estradas para $\frac{\pi^2}6$: \omterm{thm:b3:hilbert:parseval}{Parseval} (série de Fourier), o
traço do operador Green da string e os dois da cotangente
expansões; que uma soma de frequências, um traço de operador,
e um \omterm{def:b3:holomorphic:contour}{integral de contorno} concordo não é por acaso --- cada um é um
face da mesma identidade espectral.

\textbf{11.} Consertar $ R \geq 1$ e deixe $ n_0$ seja o menor
inteiro $\geq 2R $. Para $\abs z \leq R $ e $ n \geq n_0$:
$\abs{z^2/n^2} \leq \frac14$, então $1 - z^2/n^2 \in
\bar D(1, \frac14)$, onde o logaritmo principal é
\omterm{def:b3:holomorphic:holo}{holomórfico}, e
\[
\Bigl|\log\Bigl(1 - \frac{z^2}{n^2}\Bigr)\Bigr|
\leq \sum_{k\geq1}\frac1k\,\Bigl|\frac{z^2}{n^2}\Bigr|^k
\leq \frac{\abs{z^2/n^2}}{1 - \abs{z^2/n^2}}
\leq \frac{2R^2}{n^2} :
\]
a soma $ S(z) = \sum_{n\geq n_0}\log(1 - z^2/n^2)$ converge
normalmente ligado $\bar D(0, R)$, com \omterm{def:b3:holomorphic:holo}{holomórfico} somas parciais
$ S_N $ e $\abs{S_N} \leq 2R^2\zeta(2)$ uniformemente. Desde
$\abs{\eu^a - \eu^b} \leq
\eu^{\max(\abs a,\abs b)}\abs{a - b}$ (valor médio vinculado a
o segmento), os produtos finais $\prod_{n_0\leq n\leq N} =
\eu^{S_N}$ convergem uniformemente em $\bar D(0, R)$ para o
\omterm{def:b3:modules:free}{livre} de zero $\eu^S $. Multiplicando pelo polinômio fixo
$ z\prod_{n<n_0}(1 - z^2/n^2)$: $ P_N \to P $ uniformemente ligado
$\bar D(0, R)$, e
\cref{thm:b3:holomorphic:weierstrassconv} faz $ P $
\omterm{def:b3:holomorphic:holo}{holomórfico} lá; $ R $ sendo arbitrário, $ P $ é inteiro. Sobre
$\bar D(0, R)$ os zeros de $ P $ são aqueles do polinômio
pré-fator --- os inteiros do módulo $\leq R $, cada \omterm{def:b3:groups:simple}{simples}
($(1 - z/n)(1 + z/n)$ tem zeros \omterm{def:b3:groups:simple}{simples} distintos, $\eu^S $
nenhum): o conjunto zero de $ P $ é $\Z $, todos os zeros \omterm{def:b3:groups:simple}{simples}.

\textbf{12.} Diferenciação logarítmica do finito
produto, longe de seus zeros:
\[
\frac{P_N'(z)}{P_N(z)} = \frac1z +
\sum_{n=1}^{N}\frac{-2z/n^2}{1 - z^2/n^2}
= \frac1z + \sum_{n=1}^{N}\frac{2z}{z^2 - n^2} .
\]
Em um compactar $ K \subseteq \C\setminus\Z $: $ P_N \to P $ e
$ P_N' \to P'$ uniformemente
(\cref{thm:b3:holomorphic:weierstrassconv}), e $\min_K
\abs P > 0$ ($ P $ desaparece apenas em $\Z $), então eventualmente
$\abs{P_N} \geq \frac12\min_K\abs P $ e $ P_N'/P_N \to
P'/P $ uniformemente ligado $ K $. O membro do meio converge para
$\frac1z + \sum_{n\geq1}\frac{2z}{z^2-n^2} = \pi\cot(\pi z)$
pela pergunta 9: daí $ P'/P = \pi\cot(\pi z)$ sobre
$\C\setminus\Z $.

\textbf{13.} $\sin(\pi z)$ e $ P $ são inteiros com o mesmo
conjunto zero $\Z $, todos zeros \omterm{def:b3:groups:simple}{simples} (questões 1 e 11). Aproximar
$ m \in \Z $ escrever $\sin(\pi z) = (z - m)\,\sigma(z)$ e
$ P(z) = (z - m)\,\psi(z)$ com $\sigma, \psi $ \omterm{def:b3:holomorphic:holo}{holomórfico}
e não desaparecendo em $ m $ (fatorar a série de potências): $ Q =
\sin(\pi z)/P = \sigma/\psi $ estende holomorficamente e
\omterm{def:b3:modules:free}{livre} de zero em cada inteiro e é \omterm{def:b3:modules:free}{livre} de zero em
$\C\setminus\Z $ como um quociente de funções \omterm{def:b3:modules:free}{livres} de zero. Lá,
\[
\frac{Q'}{Q} = \frac{(\sin\pi z)'}{\sin\pi z} -
\frac{P'}{P} = \pi\cot(\pi z) - \pi\cot(\pi z) = 0 ,
\]
então toda a função $ Q'$ desaparece em $\C\setminus\Z $,
portanto, em todos os lugares por continuidade: $ Q $ é constante. Como $ z \to
0$: $\sin(\pi z)/z \to \pi $ e $ P(z)/z \to 1$, então $ Q =
\pi $:
\[
\sin(\pi z) = \pi z\prod_{n\geq1}
\Bigl(1 - \frac{z^2}{n^2}\Bigr) .
\]

\textbf{14.} No $ z = \frac12$: $1 = \sin\frac\pi2 =
\frac\pi2\prod_{n\geq1}\bigl(1 - \frac1{4n^2}\bigr) =
\frac\pi2\prod\frac{4n^2-1}{4n^2}$, então
\[
\frac\pi2 = \prod_{n\geq1}\frac{4n^2}{4n^2 - 1}
= \lim_{N\to\infty}\prod_{n=1}^N
\frac{(2n)(2n)}{(2n-1)(2n+1)}
= \lim_{N\to\infty}
\frac{2\cdot2\cdot4\cdot4\cdots(2N)(2N)}
{1\cdot3\cdot3\cdot5\cdots(2N-1)(2N+1)} :
\]
O produto de Wallis, um corolário de uma linha do produto de Euler
fatoração.

\textbf{15.} Para $\abs z \leq r < 1$ cada fator está em
$ D(1, r^2) \subseteq D(1, 1)$, então $ P(z)/z = \exp\bigl(
\sum_{n\geq1}\log(1 - z^2/n^2)\bigr)$: cada produto parcial
é o exponencial de uma soma parcial, e ambos os lados passam para
o limite por continuidade de $\exp $. A série dupla
\[
\sum_{n\geq1}\log\Bigl(1 - \frac{z^2}{n^2}\Bigr)
= -\sum_{n\geq1}\sum_{k\geq1}\frac{z^{2k}}{k\,n^{2k}}
= -\sum_{k\geq1}\frac{\zeta(2k)}k\,z^{2k}
= -\zeta(2)\,z^2 + O(z^4)
\]
reorganiza por Fubini para série: $\sum_{n,k}
\frac{r^{2k}}{kn^{2k}} \leq \sum_k\zeta(2k)r^{2k} \leq
\zeta(2)\frac{r^2}{1-r^2} < \infty $. Por isso
\[
P(z) = z\,\exp\bigl(-\zeta(2)z^2 + O(z^4)\bigr)
= z - \zeta(2)\,z^3 + O(z^5) ,
\]
e a questão 13 compara isso com $\sin(\pi z) = \pi z -
\frac{\pi^3}6z^3 + O(z^5)$: $\pi\zeta(2) = \frac{\pi^3}6$,
ou seja\ $\zeta(2) = \frac{\pi^2}6$. A cara aditiva
(questão 10) e a face multiplicativa calculam o mesmo
número.

\textbf{16.} Em um compactar $ K \subseteq \C\setminus\Z $ o
somas parciais $ S_N = \frac1z + \sum_{n\leq N}\bigl(
\frac1{z-n} + \frac1{z+n}\bigr)$ (pergunta 9, termos
reagrupados como $\frac{2z}{z^2-n^2} = \frac1{z-n} +
\frac1{z+n}$) convergem uniformemente para $\pi\cot(\pi z)$, então
\cref{thm:b3:holomorphic:weierstrassconv} dá $ S_N' \to
(\pi\cot\pi z)' = -\pi^2/\sin^2(\pi z)$ uniformemente ligado $ K $.
Desde $ S_N' = -\sum_{\abs n\leq N}(z - n)^{-2}$:
\[
\frac{\pi^2}{\sin^2(\pi z)} =
\sum_{n\in\Z}\frac1{(z - n)^2} ,
\]
a convergência \omterm{def:b3:groups:normal}{normal} em compactos de $\C\setminus\Z $
(termos $ O(n^{-2})$).

\textbf{17.} No $ z = \frac12$ o lado esquerdo é $\pi^2$; à direita, $(\frac12 - n)^2 = \frac{(2n-1)^2}4$ com $2n -
1$ percorrendo todos os números inteiros ímpares exatamente uma vez como $ n $ corre
acabou $\Z $:
\[
\pi^2 = \sum_{n\in\Z}\frac{4}{(2n-1)^2}
= 8\sum_{m\geq0}\frac1{(2m+1)^2},
\qquad
\sum_{m\geq0}\frac1{(2m+1)^2} = \frac{\pi^2}8 .
\]
Divisão de paridade: $\zeta(2) = \frac{\pi^2}8 +
\sum_{n\geq1}\frac1{(2n)^2} = \frac{\pi^2}8 +
\frac{\zeta(2)}4$, então $\frac34\zeta(2) = \frac{\pi^2}8$ e
$\zeta(2) = \frac{\pi^2}6$ mais uma vez.

\textbf{18.} Com $ c = \cot\theta $ e $\cot(2\theta) =
\frac{c^2-1}{2c}$: $\cot\theta - 2\cot(2\theta) = c -
\frac{c^2-1}c = \frac1c = \tan\theta $. Por isso $\pi\tan(\pi z)
= \pi\cot(\pi z) - 2\pi\cot(2\pi z)$ e a questão 9 em $ z $
e em $2z $ dá
\[
\pi\cot(\pi z) = \frac1z +
\sum_{n\geq1}\frac{2z}{z^2 - n^2},
\qquad
2\pi\cot(2\pi z) = \frac1z +
\sum_{n\geq1}\frac{8z}{4z^2 - n^2} .
\]
Ambas as séries convergem absolutamente em cada ponto fixo $ z $ fora do
pólos, então a diferença pode ser reagrupada à vontade: no
segunda série os termos pares $ n = 2m $ dar $\frac{8z}{4z^2 -
4m^2} = \frac{2z}{z^2 - m^2}$ e cancelar a primeira série
inteiramente, deixando
\[
\pi\tan(\pi z) = -\sum_{m\geq0}\frac{8z}{4z^2 - (2m+1)^2}
= \sum_{m\geq0}\frac{8z}{(2m+1)^2 - 4z^2} ,
\]
normalmente ligado compactos evitando $\frac12 + \Z $ (termos
$ O(m^{-2})$).

\textbf{19.} Para $\abs z \leq r < \frac12$:
\[
\frac{8z}{(2m+1)^2 - 4z^2} = \frac{8z}{(2m+1)^2}
\sum_{k\geq0}\Bigl(\frac{4z^2}{(2m+1)^2}\Bigr)^{k},
\]
com $\sum_{m,k}8r\,(4r^2)^k(2m+1)^{-2k-2} < \infty $ desde
$4r^2 < 1$: Fubini reorganiza a soma dupla em
\[
\pi\tan(\pi z) =
\sum_{k\geq0}8\cdot4^k\,\lambda(2k+2)\,z^{2k+1} .
\]
Contra $\pi\tan(\pi z) = \pi^2z + \frac{\pi^4}3z^3 +
O(z^5)$: o coeficiente de $ z $ dá $8\lambda(2) = \pi^2$
--- pergunta 17 novamente --- e a de $ z^3$ dá
$32\lambda(4) = \frac{\pi^4}3$, ou seja\ $\lambda(4) =
\frac{\pi^4}{96}$. Removendo os denominadores pares,
$\lambda(4) = \zeta(4) - 2^{-4}\zeta(4) =
\frac{15}{16}\zeta(4)$: $\zeta(4) = \frac{16}{15}\cdot
\frac{\pi^4}{96} = \frac{\pi^4}{90}$, correspondendo à pergunta 7.

\textbf{20.} $\cot\frac\theta2 - \cot\theta =
\frac{\cos\frac\theta2\,\sin\theta -
\cos\theta\,\sin\frac\theta2}{\sin\frac\theta2\,\sin\theta}
= \frac{\sin\frac\theta2}{\sin\frac\theta2\,\sin\theta} =
\frac1{\sin\theta}$, sendo o numerador $\sin(\theta -
\frac\theta2)$. Com $\theta = \pi z $, pergunta 9 em
$\frac z2$ lê $\pi\cot\frac{\pi z}2 = \frac2z +
\sum_{n\geq1}\frac{4z}{z^2 - 4n^2}$, então
\[
\frac{\pi}{\sin(\pi z)}
= \pi\cot\frac{\pi z}2 - \pi\cot(\pi z)
= \frac1z + \sum_{n\geq1}\frac{4z}{z^2 - 4n^2}
- \sum_{n\geq1}\frac{2z}{z^2 - n^2} .
\]
A convergência absoluta permite o reagrupamento por paridade: mesmo $ n
= 2m $ na série subtraída contribui
$\frac{2z}{z^2-4m^2}$, saindo $+\frac{2z}{z^2-4m^2}$ de
a primeira soma, embora ímpar $ n $ sobrevive com sinal $-$:
\[
\frac{\pi}{\sin(\pi z)} = \frac1z +
\sum_{n\geq1}(-1)^n\,\frac{2z}{z^2 - n^2} .
\]
Verificação de sinais: $\operatorname{Res}\bigl(\pi/\sin(\pi z),
n\bigr) = \pi/(\pi\cos\pi n) = (-1)^n $
(\cref{met:b3:residues:compute}), e $(-1)^n\frac{2z}{z^2 -
n^2} = \frac{(-1)^n}{z-n} + \frac{(-1)^n}{z+n}$ carrega
exatamente isso resíduo no $\pm n $.

\textbf{21.} Para $\abs z \leq r < 1$, expandindo cada termo como
na questão 19 ($(-1)^n\frac{2z}{z^2-n^2} =
(-1)^{n-1}\frac{2z}{n^2}\sum_k\frac{z^{2k}}{n^{2k}}$) e
aplicando Fubini:
\[
\frac{\pi}{\sin(\pi z)} = \frac1z +
2\sum_{k\geq0}\eta(2k+2)\,z^{2k+1},
\qquad
\eta(s) = \sum_{n\geq1}\frac{(-1)^{n-1}}{n^s} .
\]
Lado Taylor: $\sin(\pi z) = \pi z(1 - \frac{(\pi z)^2}6 +
O(z^4))$ dá $\frac\pi{\sin\pi z} = \frac1z +
\frac{\pi^2}6z + O(z^3)$: $2\eta(2) = \frac{\pi^2}6$, então
$\eta(2) = \frac{\pi^2}{12}$. Consistência: $\eta(2) =
\zeta(2) - 2\sum_n(2n)^{-2} = (1 - 2^{1-2})\zeta(2) =
\frac{\zeta(2)}2 = \frac{\pi^2}{12}$.

\textbf{22.} No $ z = \frac12$, a questão 20 dá
\[
\pi = 2 + \sum_{n\geq1}(-1)^n\frac{1}{\frac14 - n^2}
= 2 + 4\sum_{n\geq1}\frac{(-1)^{n-1}}{(2n-1)(2n+1)} .
\]
Com $\frac1{(2n-1)(2n+1)} = \frac12\bigl(\frac1{2n-1} -
\frac1{2n+1}\bigr)$, e ambas as séries alternadas convergentes
(critério de Leibniz), a soma se divide como $\frac12\bigl[L -
(1 - L)\bigr] = L - \frac12$, onde $ L = 1 - \frac13 +
\frac15 - \cdots $ e $\frac13 - \frac15 + \frac17 - \cdots
= 1 - L $. Por isso $\pi = 2 + 4(L - \frac12) = 4L $:
\[
\frac\pi4 = 1 - \frac13 + \frac15 - \frac17 + \cdots
\]
Moral da história: cada kernel é \omterm{def:b3:residues:singularities}{meromorfo} com postes em um
progressão aritmética e prescrita resíduos. O
opções de venda cotangentes resíduo $1$ em cada inteiro e soma
$ f(n)$; sua derivada quadratura dos pólos e dos preços
$\lambda(2)$; a tangente move os pólos para $\frac12 + \Z $
e precifica os denominadores ímpares; $\pi/\sin $ mantém o
pólos inteiros, mas alterna o resíduos $(-1)^n $, portanto
a série alternada. Os kernels de primeira ordem são todos
funções ímpares: emparelhamento $ n $ com $-n $ duplica a potência par
coeficientes e aniquila os ímpares, então $\zeta(2k)$
derrama mecanicamente enquanto $\zeta(3)$ nunca aparece. O
cegueira é paridade, não falta de técnica.

\textbf{23.} Com $ w = 2\iu\pi z $:
\[
\iu\pi z + \frac{2\iu\pi z}{\eu^{2\iu\pi z} - 1}
= \iu\pi z\,\frac{\eu^{2\iu\pi z} + 1}{\eu^{2\iu\pi z} - 1}
= \iu\pi z\,\frac{\eu^{\iu\pi z} +
\eu^{-\iu\pi z}}{\eu^{\iu\pi z} - \eu^{-\iu\pi z}}
= \pi z\,\frac{\cos\pi z}{\sin\pi z} = \pi z\cot(\pi z) .
\]
Portanto, usando a função geradora em $ w = 2\iu\pi z $ (e
$ B_1 = -\frac12$ cancelando o $\iu\pi z $ termo, estranho $ B $ de
desaparecendo além):
\[
\pi z\cot(\pi z) = \sum_{k\geq0}\frac{B_{2k}}{(2k)!}
(2\iu\pi z)^{2k}
= 1 + \sum_{k\geq1}(-1)^k\frac{(2\pi)^{2k}B_{2k}}{(2k)!}
z^{2k} .
\]
Por outro lado, a expansão em frações parciais (Parte IV)
dá $\pi z\cot(\pi z) = 1 - 2\sum_{k\geq1}\zeta(2k)z^{2k}$
(expandir cada $\frac{2z^2}{z^2 - n^2} =
-2\sum_k\frac{z^{2k}}{n^{2k}}$ e soma mais $ n $, normalmente
convergência que justifica o intercâmbio para $\abs z < 1$).
Comparando coeficientes: $-2\zeta(2k) =
(-1)^k\frac{(2\pi)^{2k}B_{2k}}{(2k)!}$, a fórmula declarada.

\textbf{24.} $ k = 1$: $\frac{(2\pi)^2}{2\cdot2}\cdot\frac16
= \frac{\pi^2}6$. $ k = 2$: $-\frac{(2\pi)^4}{2\cdot24}
\cdot\bigl(-\frac1{30}\bigr) = \frac{16\pi^4}{48\cdot30} =
\frac{\pi^4}{90}$. $ k = 3$: $\frac{(2\pi)^6}{2\cdot720}
\cdot\frac1{42} = \frac{64\pi^6}{1440\cdot42} =
\frac{\pi^6}{945}$.

\textbf{25.} Escrever $ C(z) = \pi z\cot(\pi z) = 1 -
2\sum_{k\geq1}\zeta(2k)z^{2k}$ (questão 23). Direto
diferenciação de $ C = \pi z\cot(\pi z)$, usando $(\cot u)'
= -1 - \cot^2u $:
\[
zC'(z) = \pi z\cot(\pi z) - \pi^2z^2\bigl(1 +
\cot^2(\pi z)\bigr) = C - \pi^2z^2 - C^2 .
\]
Agora expanda ambos os lados em poderes de $ z^2$. Lado esquerdo:
$\sum_k(-4k)\,\zeta(2k)\,z^{2k}$. Lado direito: quadratura do
série,
\[
C - C^2 = 2\sum_{k\geq1}\zeta(2k)z^{2k}
- 4\sum_{k\geq2}\Bigl(\sum_{j=1}^{k-1}\zeta(2j)
\zeta(2k-2j)\Bigr)z^{2k},
\]
e o termo $-\pi^2z^2 = -6\zeta(2)z^2$ apenas ajusta $ k =
1$. Comparando os coeficientes de $ z^{2k}$ para $ k \geq 2$:
\[
-4k\,\zeta(2k) = 2\,\zeta(2k) -
4\sum_{j=1}^{k-1}\zeta(2j)\,\zeta(2k-2j),
\]
ou seja\ $\bigl(k + \frac12\bigr)\zeta(2k) =
\sum_{j=1}^{k-1}\zeta(2j)\zeta(2k-2j)$. (No $ k = 1$ o
leituras de identidade $-4\zeta(2) = 2\zeta(2) - 6\zeta(2)$: um
verificação de consistência, não novas informações.) Aplicações: $ k =
2$: $\frac52\zeta(4) = \zeta(2)^2 = \frac{\pi^4}{36}$, então
$\zeta(4) = \frac{\pi^4}{90}$; $ k = 3$: $\frac72\zeta(6) =
2\zeta(2)\zeta(4) = \frac{\pi^6}{270}$, então $\zeta(6) =
\frac{2}{7}\cdot\frac{\pi^6}{270} = \frac{\pi^6}{945}$. Um
semente transcendental ($\zeta(2) = \frac{\pi^2}6$) e puro
a álgebra gera todos os valores zeta pares.
\end{solution}
