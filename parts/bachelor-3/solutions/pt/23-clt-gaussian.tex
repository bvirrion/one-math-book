\begin{solution}{exo:b3:clt:4}

\begin{solution}{pb:b3:clt:1}
\textbf{1.} A famûÙlia $(X_1, \dots, X_n, N_1, \dots, N_n)$
ûˋ \omterm{def:b3:probability:independence}{independente}: os dois blocos sûÈo \omterm{def:b3:probability:independence}{independente} de cada
outro por construûÏûÈo e cada bloco ûˋ i.i.d. $ W_i $ ûˋ um

$(N_j)_{j>i}$ apenas, todos distintos de $ X_i $ e $ N_i $: por
o princûÙpio da coalizûÈo
(\cref{thm:b3:probability:independence}), $ W_i $ ûˋ
\omterm{def:b3:probability:independence}{independente} do par $(X_i, N_i)$. As decomposiûÏûçes
$ H_i = W_i + \frac{X_i}{\sqrt n}$ e $ H_{i-1} = W_i +
\frac{N_i}{\sqrt n}$ sûÈo imediatos das definiûÏûçes:
passando de $ H_i $ para $ H_{i-1}$ troca o pedido û¤nico
$ X_i $ para $ N_i $.

\textbf{2.} Taylor-Lagrange sob encomenda $3$: hûÀ $ c $
entre $ w $ e $ w + h $ com $ f(w + h) = f(w) + f'(w)h +
\frac12f''(w)h^2 + \frac16f'''(c)h^3$, e $\abs{f'''(c)}
\leq M_3$ dûÀ o limite.

\textbf{3.} Subtraindo as duas expansûçes no comum
ponto \omterm{def:b3:topology:basis}{base} $ w = W_i $:
\[
f(H_i) - f(H_{i-1}) = f'(W_i)\,\frac{X_i - N_i}{\sqrt n} +
\frac{f''(W_i)}{2}\,\frac{X_i^2 - N_i^2}{n} + R_i,
\qquad
\abs{R_i} \leq \frac{M_3}{6}\cdot
\frac{\abs{X_i}^3 + \abs{N_i}^3}{n^{3/2}} .
\]
Pegar \omterm{def:b3:probability:space}{expectativas}. Pela pergunta 1, $ f'(W_i)$ e $ f''(W_i)$
sûÈo \omterm{def:b3:probability:independence}{independente} de $(X_i, N_i)$, entûÈo o misto \omterm{def:b3:probability:space}{expectativas}
fator:
\begin{align*}
\E\Bigl[f'(W_i)\,\frac{X_i - N_i}{\sqrt n}\Bigr] &=
\E\bigl[f'(W_i)\bigr]\,\frac{\E X_i - \E N_i}{\sqrt n} = 0,
\\
\E\Bigl[f''(W_i)\,\frac{X_i^2 - N_i^2}{n}\Bigr] &=
\E\bigl[f''(W_i)\bigr]\,\frac{1 - 1}{n} = 0 :
\end{align*}
os dois primeiros momentos $ X_i $ e $ N_i $ \emph{corresponder}e
apenas o restante sobrevive:
\[
\bigl|\E f(H_i) - \E f(H_{i-1})\bigr| \leq \E\abs{R_i} \leq
\frac{M_3}{6}\cdot\frac{\beta + \gamma}{n^{3/2}} .
\]
O \omterm{def:b3:clt:gaussianvector}{Gaussiano} terceiro momento: $\gamma = \E\abs{N_1}^3 =
2\int_0^\infty x^3\,\frac{\eu^{-x^2/2}}{\sqrt{2\pi}}\,\dd x
= \frac{2}{\sqrt{2\pi}}\int_0^\infty 2u\,\eu^{-u}\dd u =
\frac{4}{\sqrt{2\pi}} = \frac{2\sqrt2}{\sqrt\pi}$
(substituiûÏûÈo $ u = x^2/2$, entûÈo $\Gamma(2) = 1$).

\textbf{4.} Telescû°pico $\E f(T_n) - \E f(G_n) =
\sum_{i=1}^n\bigl(\E f(H_i) - \E f(H_{i-1})\bigr)$ e
aplicando a questûÈo 3 a cada um dos $ n $ termos:
\[
\bigl|\E f(T_n) - \E f(G_n)\bigr| \leq
n \cdot \frac{M_3(\beta + \gamma)}{6\,n^{3/2}} =
\frac{M_3\,(\beta + \gamma)}{6\,\sqrt n} .
\]

\textbf{5.} $ G_n $ ûˋ $\mathcal N(0,1)$ \emph{exatamente} para
cada $ n $ (uma soma normalizada de \omterm{def:b3:probability:independence}{independente} padrûÈo
\omterm{def:b3:clt:gaussianvector}{Gaussianos}, \cref{exo:b3:clt:3}), entûÈo $\E f(G_n) = \E f(N)$
e a pergunta 4 ûˋ lida $\abs{\E f(T_n) - \E f(N)} \leq
\frac{M_3(\beta+\gamma)}{6\sqrt n} \to 0$ para $ f \in
\mathcal C^3_b $. \emph{Atualizar.} Deixar $ f $ ser limitado
\omterm{def:b3:topology:continuity}{contûÙnuo}, $ M = \sup\abs f $, $\varepsilon > 0$. Escolher $ A
\geq 1$ com $\frac1{A^2} \leq \varepsilon $: Chebyshev com
$\V(T_n) = 1$ dûÀ $\P(\abs{T_n} > A) \leq \varepsilon $
para todos $ n $, e da mesma forma $\P(\abs N > A) \leq \varepsilon $.
Deixar $\chi $ ser $\mathcal C^\infty $ com
$\mathbf 1_{\intcc{-A}A} \leq \chi \leq
\mathbf 1_{\intcc{-A-1}{A+1}}$ (um planalto suave, construûÙdo por
apaziguador $\mathbf 1_{\intcc{-A-\frac12}{A+\frac12}}$,
\cref{thm:b3:lp:regularization}); $ g = f\chi $ ûˋ \omterm{def:b3:topology:continuity}{contûÙnuo}
com compactar suporte, portanto uniformemente \omterm{def:b3:topology:continuity}{contûÙnuo}, entûÈo ûˋ
apaziguamento $ g_\eta = g * \rho_\eta $ ûˋ $\mathcal
C^\infty $ com derivadas limitadas de todas as ordens e
$\norm{g - g_\eta}_\infty \leq \varepsilon $ para $\eta $
pequeno o suficiente. Para $ T = T_n $ ou $ N $, desde $ f = g $ sobre
$\intcc{-A}A $ e $\abs{f - g} \leq 2M $ em todos os lugares:
\[
\bigl|\E f(T) - \E g_\eta(T)\bigr| \leq
\E\abs{(f - g)(T)} + \norm{g - g_\eta}_\infty
\leq 2M\,\P(\abs T > A) + \varepsilon \leq
(2M + 1)\,\varepsilon .
\]
Combinando com $\E g_\eta(T_n) \to \E g_\eta(N)$ (pergunta
4 se aplica: $ g_\eta \in \mathcal C^3_b $):
\[
\limsup_n\;\bigl|\E f(T_n) - \E f(N)\bigr| \leq

\]
e $\varepsilon $ foi arbitrûÀrio: $\E f(T_n) \to \E f(N)$ para
cada limite \omterm{def:b3:topology:continuity}{contûÙnuo} $ f $, ou seja\ $ T_n \Rightarrow
\mathcal N(0,1)$.

\textbf{6.} DistribuiûÏûÈo idûˆntica inserida apenas atravûˋs de um
