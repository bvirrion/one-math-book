\begin{solution}{exo:b3:forms:1}

\[

\end{solution}

\begin{solution}{pb:b3:forms:1}
\textbf{1.} $\dd\sigma = \sum_i(-1)^{i-1}\,\dd x_i\wedge\dd

$\dd\sigma = n\,\dd x_1\wedge\dots\wedge\dd x_n $. Acelera

n\operatorname{vol}(\bar B) > 0$.

\textbf{2.} $ n = 2$: $\sigma = x\,\dd y - y\,\dd x $; sobre
$\gamma(t) = (\cos t, \sin t)$, $\gamma^*\sigma = (\cos^2t +
\sin^2t)\,\dd t = \dd t $, a forma do comprimento de arco:
$\int_S\sigma = 2\pi = 2\operatorname{vol}(\bar B^2)$.
$ n = 3$: $\sigma\vert_S $ Ă© a forma de ĂĄrea
(\cref{exo:b3:forms:5} com $\nu(x) = x $):
$\int_S\sigma = 4\pi = 3\cdot\tfrac{4\pi}3$. Ambos combinam
pergunta 1.

\textbf{3.} Diferenciando $\norm\varphi^2 = 1$: $2\langle
D\varphi(x)h, \varphi(x)\rangle = 0$ para cada $ h $, entĂŁo
$\operatorname{im}D\varphi(x) \subseteq \varphi(x)^\perp $, um
hiperplano: $\operatorname{rk}D\varphi(x) \leq n - 1$. Desde
$\dd\sigma $ Ă© um $ n $-forma (pergunta 1), pontual
$\varphi^*(\dd\sigma)_x =
(D\varphi(x))^*(\dd\sigma)_{\varphi(x)} = 0$ por
\cref{prop:b3:forms:linearpullback}(2): um mapa do
esfera nĂŁo tem espaĂ§o para recuar um volume.

\textbf{4.} A falha Ă© o segundo daĂ­: no
$(n-1)$-dimensional $ S $, todo $(n-1)$-form Ă© trivialmente
fechado (nĂŁo hĂĄ nada diferente de zero $ n $-formas em um
$(n-1)$-variedade), mas \emph{fechado nĂŁo significa \omterm{def:b3:forms:closedexact}{exato}},
e $\int_S\dd\eta = 0$ requer um primitivo real $\eta $
definido em $ S $. A restriĂ§ĂŁo de $\sigma $ precisamente nĂŁo Ă©
\omterm{def:b3:forms:closedexact}{exato} --- sua integral Ă© $ n\operatorname{vol}(\bar B) \neq
0$ --- e essa imprecisĂŁo alimenta todo o problema.

\textbf{5.} Vamos integrar $ r^*\sigma $ sobre a esfera
e conte de duas maneiras. Porque $ r $ correĂ§Ă”es $ S $ pontualmente, o
integral igual $\int_S\sigma = n\operatorname{vol}(\bar B)
\neq 0$. Porque $ r $ Ă© definido na bola, Stokes converte
a mesma integral em $\int_{\bar B}\dd(r^*\sigma) =
\int_{\bar B}r^*(\dd\sigma)$; e porque $ r $ leva valores

nĂșmero, dois valores: a retraĂ§ĂŁo nĂŁo pode existir.

\textbf{6.} Ambas as integrais sĂŁo calculadas por meio direto

(\cref{def:b3:forms:integral}); desde $ r\circ\gamma =
\gamma $ (a parametrizaĂ§ĂŁo chega em $ S $, onde $ r $ Ă© o
identidade), $\gamma^*(r^*\sigma) = (r\circ\gamma)^*\sigma =
\gamma^*\sigma $ (\cref{thm:b3:forms:pullback}(a)): o local
integrandos coincidem, e qualquer \omterm{lem:b3:forms:partition}{partiĂ§ĂŁo da unidade} dĂĄ
$\int_Sr^*\sigma = \int_S\sigma $.

\textbf{7.} $ r^*\sigma $ Ă© um suave $(n-1)$-formulĂĄrio em um
\omterm{def:b3:topology:topology}{vizinhanĂ§a} do compactar orientado $\bar B $, cujo
limite com o induzido \omterm{def:b3:forms:orientation}{orientaĂ§ĂŁo} Ă© $ S $: Stokes
(\cref{thm:b3:forms:stokes}) dĂĄ exatamente

\[

\]
