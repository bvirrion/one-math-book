\section*{Chapitre \ref{ch:b3:clt} --- Fonctions caractéristiques
et théorème central limite}

\begin{solution}{exo:b3:clt:1}
Uniforme sur $\intcc{-1}1$~: $\varphi(\xi) =
\frac12\int_{-1}^1\eu^{\iu\xi x}\dd x =
\frac{\sin\xi}{\xi}$ (égale à $1$ en $\xi = 0$).
Exponentielle $\mathcal E(\lambda)$~: $\varphi(\xi) =
\lambda\int_0^\infty\eu^{(\iu\xi - \lambda)x}\dd x =
\frac{\lambda}{\lambda - \iu\xi}$ (la primitive s'annule en
$+\infty$ car $\operatorname{Re}(\iu\xi - \lambda) < 0$).
Poisson $\mathcal P(\lambda)$~: par le théorème de transfert
pour les \omterm{def:b3:probability:space}{lois} discrètes,
\[
\varphi(\xi) = \sum_{k\geq0}\eu^{\iu\xi
k}\,\eu^{-\lambda}\frac{\lambda^k}{k!}
= \eu^{-\lambda}\exp\bigl(\lambda\eu^{\iu\xi}\bigr)
= \exp\bigl(\lambda(\eu^{\iu\xi} - 1)\bigr).
\]
Binomiale $\mathcal B(n, p)$~: somme de $n$ Bernoulli
\omterm{def:b3:probability:independence}{indépendantes}, chacune de f.c.\ $1 - p + p\eu^{\iu\xi}$,
donc $\varphi(\xi) = \bigl(1 - p + p\eu^{\iu\xi}\bigr)^n$
(\cref{prop:b3:clt:cfbasics}(b)). Additivité de Poisson~: si
$X \sim \mathcal P(\lambda)$, $Y \sim \mathcal P(\mu)$
\omterm{def:b3:probability:independence}{indépendantes},
\[
\varphi_{X+Y}(\xi) = \eu^{\lambda(\eu^{\iu\xi}-1)}
\eu^{\mu(\eu^{\iu\xi}-1)} =
\eu{(\lambda+\mu)(\eu^{\iu\xi}-1)},
\]
la f.c.\ de $\mathcal P(\lambda + \mu)$~; l'injectivité
(\cref{thm:b3:clt:injectivity}) identifie la \omterm{def:b3:probability:space}{loi}.
\end{solution}

\begin{solution}{exo:b3:clt:2}
(a) $\overline{\varphi_X(\xi)} = \E\eu^{-\iu\xi X} =
\varphi_{-X}(\xi)$. Donc $\varphi_X$ est réelle si et
seulement si $\varphi_X = \varphi_{-X}$, si et seulement si
(injectivité, \cref{thm:b3:clt:injectivity}) $X$ et $-X$ ont
la même \omterm{def:b3:probability:space}{loi}.
(b) Écrire $\varphi_X(\xi_0) = \eu^{\iu\theta}$. Alors
\[
\E\bigl[1 - \cos(\xi_0X - \theta)\bigr]
= 1 - \operatorname{Re}\bigl(\eu^{-\iu\theta}
\varphi_X(\xi_0)\bigr) = 1 - 1 = 0 .
\]
L'intégrande est positive, donc $\cos(\xi_0X - \theta) = 1$
presque sûrement (une variable positive d'\omterm{def:b3:probability:space}{espérance} nulle
s'annule p.s.), i.e.\ $\xi_0X - \theta \in 2\pi\Z$ p.s.~: $X$
prend ses valeurs dans la progression arithmétique
$\frac{\theta}{\xi_0} + \frac{2\pi}{\xi_0}\Z$ presque
sûrement. Si $X$ a une \omterm{ex:b3:lebesgue:gamma}{densité}, cet ensemble dénombrable est
négligeable pour Lebesgue, donc porte une probabilité $0$ ---
contradiction~; par conséquent $\abs{\varphi_X(\xi)} < 1$ pour
tout $\xi \neq 0$.
\end{solution}

\begin{solution}{exo:b3:clt:3}
Par \omterm{def:b3:probability:independence}{indépendance} et \cref{prop:b3:clt:cfbasics}~:
\[
\varphi_{X+Y}(\xi) = \eu^{\iu m_1\xi - \sigma_1^2\xi^2/2}\,
\eu^{\iu m_2\xi - \sigma_2^2\xi^2/2}
= \eu^{\iu(m_1+m_2)\xi - (\sigma_1^2+\sigma_2^2)\xi^2/2},
\]
la f.c.\ de $\mathcal N(m_1 + m_2, \sigma_1^2 + \sigma_2^2)$~;
l'injectivité conclut. La stabilité sous les applications
affines est la règle affine ($aX + b \sim \mathcal N(am_1 + b,
a^2\sigma_1^2)$, en autorisant le cas dégénéré $a = 0$), et
la stabilité sous les sommes \omterm{def:b3:probability:independence}{indépendantes} suit par
récurrence sur le calcul ci-dessus. Pour des gaussiennes
\emph{dépendantes}, la somme n'est pas forcément
gaussienne~: dans \cref{exo:b3:clt:9}, $X$ et $Y =
\varepsilon X$ sont chacune standard gaussienne mais $X + Y$
s'annule avec probabilité $\frac12$ sans être p.s.\ nulle,
donc n'est pas gaussienne.
\end{solution}

\begin{solution}{exo:b3:clt:4}
(a) \emph{Implication directe.} Soit $t$ un point de continuité
de $F_X$ et $\delta > 0$. Prendre les rampes \omterm{def:b3:topology:continuity}{continues} $f^-$
($= 1$ sur $\intoc{-\infty}{t-\delta}$, $0$ à partir de $t$,
affine entre) et $f^+$ ($= 1$ sur $\intoc{-\infty}t$, $0$ à
partir de $t + \delta$, affine entre)~; alors $f^- \leq
\mathbf 1_{\intoc{-\infty}t} \leq f^+$, donc
\[
\E f^-(X_n) \leq F_{X_n}(t) \leq \E f^+(X_n),
\]
et les termes extrêmes convergent vers $\E f^\pm(X)$,
eux-mêmes encadrés entre $F_X(t - \delta)$ et $F_X(t +
\delta)$. En laissant $n \to \infty$ puis $\delta \to 0$ et en
utilisant la continuité de $F_X$ en $t$~: $F_{X_n}(t) \to
F_X(t)$.

\emph{Réciproque.} Soit $f$ \omterm{def:b3:topology:continuity}{continue} bornée, $M =
\sup\abs f$, $\varepsilon > 0$. Les points de continuité de
$F_X$ sont \omterm{def:b3:topology:interior}{denses} ($F_X$ a au plus dénombrablement beaucoup
de sauts), donc choisir des points de continuité $a < b$ avec
$F_X(a) < \varepsilon$ et $1 - F_X(b) < \varepsilon$. Sur le
compact $\intcc ab$ la fonction $f$ est uniformément \omterm{def:b3:topology:continuity}{continue}~:
choisir des points de continuité $a = t_0 < t_1 < \dots < t_m
= b$ de $F_X$ avec oscillation de $f$ au plus $\varepsilon$
sur chaque $\intoc{t_{j-1}}{t_j}$, et poser $g = \sum_j
f(t_j)\,\mathbf 1_{\intoc{t_{j-1}}{t_j}}$. Alors $\abs{f - g}
\leq \varepsilon$ sur $\intoc ab$, $\abs g \leq M$, et pour
$T = X_n$ ou $X$~:
\[
\bigl|\E f(T) - \E g(T)\bigr| \leq \varepsilon +
2M\bigl(F_T(a) + 1 - F_T(b)\bigr).
\]
De plus $\E g(X_n) = \sum_j f(t_j)\bigl(F_{X_n}(t_j) -
F_{X_n}(t_{j-1})\bigr) \to \E g(X)$ (somme finie de termes
convergents, tous les $t_j$ étant des points de continuité),
et $F_{X_n}(a) \to F_X(a) < \varepsilon$, $1 - F_{X_n}(b) \to
1 - F_X(b) < \varepsilon$. En assemblant~:
$\limsup_n\abs{\E f(X_n) - \E f(X)} \leq 2\varepsilon +
8M\varepsilon$~; laisser $\varepsilon \to 0$.

(b) La fonction de répartition de la constante $c$ est
$\mathbf 1_{\intco c\infty}$, \omterm{def:b3:topology:continuity}{continue} sauf en $c$. Pour
$\varepsilon > 0$, les points $c - \varepsilon$ et $c +
\frac\varepsilon2$ sont des points de continuité, donc
\[
\P(\abs{X_n - c} > \varepsilon) \leq F_{X_n}(c -
\varepsilon) + 1 - F_{X_n}\bigl(c +
\tfrac\varepsilon2\bigr) \to 0 + 0 = 0.
\]
\end{solution}

\begin{solution}{exo:b3:clt:5}
Poser $z_n = p_n(\eu^{\iu\xi} - 1)$, donc $\varphi_{X_n}(\xi) =
(1 + z_n)^n$ (\cref{exo:b3:clt:1}) et $\abs{z_n} \leq 2p_n
\to 0$ (noter $p_n = \frac{np_n}n \to 0$). Les deux $1 + z_n$
et $\eu^{z_n}$ ont \omterm{def:b3:modules:module}{module} au plus $1$~: $\abs{1 + z_n} =
\abs{(1 - p_n) + p_n\eu^{\iu\xi}} \leq 1$ par l'inégalité
triangulaire, et $\abs{\eu^{z_n}} = \eu^{p_n(\cos\xi - 1)}
\leq 1$. L'inégalité de télescopage $\abs{a^n - b^n} \leq
n\abs{a - b}$ (preuve de \cref{thm:b3:clt:clt}) et la borne
de série entière $\abs{\eu^z - 1 - z} \leq
\abs z^2\eu^{\abs z}$ donnent
\[
\bigl|(1 + z_n)^n - \eu^{nz_n}\bigr| \leq n\bigl|1 + z_n -
\eu^{z_n}\bigr| \leq n\,\abs{z_n}^2\,\eu^{\abs{z_n}} \leq
4\eu^2\,np_n^2 = 4\eu^2\,(np_n)\,p_n \longrightarrow 0 .
\]
Comme $nz_n = np_n(\eu^{\iu\xi} - 1) \to
\lambda(\eu^{\iu\xi} - 1)$, on conclut $\varphi_{X_n}(\xi)
\to \exp\bigl(\lambda(\eu^{\iu\xi} - 1)\bigr)$ pour tout
$\xi$~: la f.c.\ de $\mathcal P(\lambda)$, et Lévy
(\cref{thm:b3:clt:levy}) donne $X_n \Rightarrow \mathcal
P(\lambda)$. Numériquement~: $\P\bigl(\mathcal B(100, 0.02) =
0\bigr) = 0{,}98^{100} = \eu^{100\ln 0{,}98} \approx
\eu^{-2{,}020} \approx 0{,}1326$, tandis que $\P\bigl(\mathcal
P(2) = 0\bigr) = \eu^{-2} \approx 0{,}1353$~: deux pour cent
d'écart déjà à ce $n$ grossier.
\end{solution}

\begin{solution}{exo:b3:clt:6}
(a) Un lancer a moyenne $\frac72$ et variance $\frac{35}{12}$,
donc $S$ a moyenne $3500$, variance $\frac{35000}{12} \approx
2916{,}7$ et écart-type $\approx 54{,}0$. Par le TCL,
\[
\P(S > 3600) = \P\Bigl(\frac{S - 3500}{54{,}0} > 1{,}85\Bigr)
\approx 1 - \Phi(1{,}85) \approx 0{,}032 :
\]
environ $3\%$ de chance.
(b) $S \sim \mathcal B(100, \frac12)$~: moyenne $50$,
écart-type $5$. Avec la correction de continuité,
\[
\P(45 \leq S \leq 55) \approx
\Phi\Bigl(\frac{55{,}5 - 50}{5}\Bigr) -
\Phi\Bigl(\frac{44{,}5 - 50}{5}\Bigr) = 2\Phi(1{,}1) - 1
\approx 0{,}729,
\]
contre la valeur \omterm{def:b3:forms:closedexact}{exacte} $0{,}7287$~; sans la correction,
$2\Phi(1) - 1 \approx 0{,}683$, à presque cinq points. La
correction compte parce que $S$ est une variable de réseau~:
l'atome $\P(S = k)$ est bien approché par la masse
gaussienne de $\intcc{k - \frac12}{k + \frac12}$, et
tronquer l'intervalle aux entiers $45$ et $55$ jette un
demi-atome à chaque extrémité.
\end{solution}

\begin{solution}{exo:b3:clt:7}
Les variables $g(U_k)$ sont i.i.d.\ (images \omterm{def:b3:lebesgue:measurable}{mesurables} de
variables i.i.d.), de carré \omterm{def:b3:lebesgue:l1}{intégrable}, de moyenne $I$
(théorème de transfert, \cref{exo:b3:product:9}) et de
variance $\sigma^2$. Si $\sigma > 0$,
le~\cref{thm:b3:clt:clt} appliqué à elles est exactement la
convergence indiquée
\[
\sqrt n\,\Bigl(\frac1n\sum_{k\leq n}g(U_k) - I\Bigr) =
\frac{\sum_{k\leq n}\bigl(g(U_k) - I\bigr)}{\sqrt n}
\Longrightarrow \mathcal N(0, \sigma^2)
\]
(si $\sigma = 0$, $g$ est p.s.\ constante et le premier
membre s'annule identiquement). D'où
$\P\bigl(\abs{\frac1n\sum g(U_k) - I} \leq
1{,}96\,\sigma/\sqrt n\bigr) \to 0{,}95$~: la barre d'erreur
$\pm 1{,}96\,\sigma/\sqrt n$ ne voit la dimension $d$ qu'à
travers la constante $\sigma$, jamais à travers le taux en
$n$. La règle du point milieu à $n$ nœuds en dimension $d$ a
pour maille $n^{-1/d}$ et une erreur d'ordre $n^{-2/d}$ pour
des intégrandes $\mathcal C^2$. Le $n^{-1/2}$ de \omterm{ex:b3:probability:sllnapps}{Monte-Carlo}
décroît plus vite que $n^{-2/d}$ exactement quand $\frac12 >
\frac2d$, i.e.\ $d > 4$~: à partir de la dimension $5$,
l'échantillonnage aléatoire bat asymptotiquement la grille
--- la malédiction de la dimensionnalité épargne les
méthodes probabilistes, ce qui explique pourquoi \omterm{ex:b3:probability:sllnapps}{Monte-Carlo}
domine l'intégration en grande dimension.
\end{solution}

\begin{solution}{exo:b3:clt:8}
\emph{Somme.} Pour $\xi$ fixé~:
\[
\bigl|\E\eu^{\iu\xi(X_n+Y_n)} -
\eu^{\iu\xi c}\,\E\eu^{\iu\xi X_n}\bigr|
= \bigl|\E\bigl[\eu^{\iu\xi X_n}\bigl(\eu^{\iu\xi Y_n} -
\eu^{\iu\xi c}\bigr)\bigr]\bigr|
\leq \E\bigl|\eu^{\iu\xi(Y_n - c)} - 1\bigr| .
\]
Scinder sur l'événement $\{\abs{Y_n - c} \leq \delta\}$~: là,
$\abs{\eu^{\iu\xi(Y_n-c)} - 1} \leq \abs\xi\,\delta$ (la corde
est plus courte que l'arc)~; le complémentaire contribue au
plus $2\,\P(\abs{Y_n - c} > \delta) \to 0$. Donc le
$\limsup$ est $\leq \abs\xi\,\delta$ pour tout $\delta >
0$~: la différence tend vers $0$. Comme $\E\eu^{\iu\xi X_n}
\to \varphi_X(\xi)$, on obtient $\varphi_{X_n+Y_n}(\xi) \to
\eu^{\iu\xi c}\varphi_X(\xi) = \varphi_{X+c}(\xi)$, et Lévy
(\cref{thm:b3:clt:levy}) donne $X_n + Y_n \Rightarrow X +
c$.

\emph{Produit.} D'abord, $cX_n \Rightarrow cX$~:
$\varphi_{cX_n}(\xi) = \varphi_{X_n}(c\xi) \to
\varphi_X(c\xi) = \varphi_{cX}(\xi)$. Ensuite, $(Y_n -
c)X_n \to 0$ en probabilité~: les \omterm{def:b3:probability:space}{lois} des $X_n$ sont
tendues (leurs f.c.\ convergent vers une f.c.~; voir l'étape
de tension de \cref{thm:b3:clt:levy}), donc étant donné
$\varepsilon > 0$ choisir $M$ avec $\P(\abs{X_n} > M) \leq
\varepsilon$ pour tout $n$~; alors
\[
\P\bigl(\abs{(Y_n - c)X_n} > \varepsilon\bigr) \leq
\P(\abs{X_n} > M) + \P\Bigl(\abs{Y_n - c} >
\frac{\varepsilon}{M}\Bigr) \leq \varepsilon + o(1) .
\]
Écrire $Y_nX_n = cX_n + (Y_n - c)X_n$ et appliquer la partie
somme (dont la preuve n'utilisait que $Y_n' := (Y_n -
c)X_n \to 0$ en probabilité, avec constante $0$)~: $Y_nX_n
\Rightarrow cX$.

\emph{Application.} Par la \omterm{def:b3:probability:space}{loi} forte des grands nombres
(\cref{thm:b3:probability:slln}), $\hat p_n \to p$ p.s.,
donc par continuité $\hat\sigma_n = \sqrt{\hat p_n(1 -
\hat p_n)} \to \sigma = \sqrt{p(1 - p)} > 0$ p.s., d'où
$\frac{\sigma}{\hat\sigma_n} \to 1$ en probabilité. La règle
produit de Slutsky promeut $\frac{S_n - np}{\sigma\sqrt n}
\Rightarrow \mathcal N(0,1)$ en $\frac{S_n -
np}{\hat\sigma_n\sqrt n} = \frac{\sigma}{\hat\sigma_n}\cdot
\frac{S_n - np}{\sigma\sqrt n} \Rightarrow \mathcal N(0,1)$~:
l'intervalle de confiance \emph{utilisable} $\hat p_n \pm
1.96\,\hat\sigma_n/\sqrt n$, construit à partir des
données seules, conserve son niveau asymptotique $95\%$.
\end{solution}

\begin{solution}{exo:b3:clt:9}
(a) En scindant l'\omterm{def:b3:probability:space}{espérance} sur les deux valeurs de
$\varepsilon$ (\omterm{def:b3:probability:independence}{indépendance})~: pour $B$ borélien, $\P(Y \in
B) = \frac12\P(X \in B) + \frac12\P(-X \in B) = \P(X \in
B)$, car $-X \sim X$ ($\mathcal N(0,1)$ est symétrique)~: $Y
\sim \mathcal N(0,1)$. Et $\operatorname{Cov}(X, Y) =
\E[\varepsilon X^2] = \E[\varepsilon]\,\E[X^2] = 0 \cdot 1 =
0$.
(b) $\abs Y = \abs X$, donc $\P(\abs X \leq 1,\ \abs Y \geq
2) = 0$ tandis que $\P(\abs X \leq 1)\,\P(\abs Y \geq 2) >
0$~: pas \omterm{def:b3:probability:independence}{indépendantes}. Si $(X, Y)$ était un \omterm{def:b3:clt:gaussianvector}{vecteur
gaussien}, $X + Y = (1 + \varepsilon)X$ serait une variable
gaussienne réelle (\cref{def:b3:clt:gaussianvector} avec $t =
(1,1)$)~; mais $\P(X + Y = 0) = \P(\varepsilon = -1) =
\frac12$, alors qu'une variable gaussienne n'a d'atome que
si elle est p.s.\ constante --- et $X + Y$ égale $2X \neq 0$
p.s.\ sur $\{\varepsilon = 1\}$. Contradiction~: $(X, Y)$
n'est pas \omterm{def:b3:clt:gaussianvector}{gaussien}.
(c) Chaque marginale est gaussienne et la covariance
s'annule, pourtant l'\omterm{def:b3:probability:independence}{indépendance} échoue --- parce que la
\emph{paire} n'est pas conjointement gaussienne.
Le~\cref{thm:b3:clt:gaussianvector}(2) ne peut pas être
affaibli en «~marginales gaussiennes~».
\end{solution}

\begin{solution}{exo:b3:clt:10}
(a) \cref{exo:b3:fouriertransform:1} calcule
$\widehat{\eu^{-\abs\cdot}}(\xi) = \frac{2}{1 + \xi^2}$~;
les deux membres étant \omterm{def:b3:lebesgue:l1}{intégrables}, l'inversion de Fourier
(\cref{thm:b3:fouriertransform:inversion}) retourne ce
résultat~:
\[
\int_\R\eu^{\iu\xi x}\,\frac{\dd x}{\pi(1 + x^2)} =
\eu^{-\abs\xi},
\]
qui est exactement $\varphi_X(\xi)$ pour une variable de
Cauchy $X$.
(b) Par \omterm{def:b3:probability:independence}{indépendance}, $\varphi_{S_n}(\xi) =
\bigl(\eu^{-\abs\xi}\bigr)^n = \eu^{-n\abs\xi}$, donc
$\varphi_{S_n/n}(\xi) = \varphi_{S_n}(\xi/n) =
\eu^{-\abs\xi}$~: la moyenne empirique $\frac{S_n}n$ est
encore Cauchy standard pour tout $n$ (injectivité). La
moyenne ne se concentre jamais~: ses fluctuations au temps
$10^6$ sont celles d'une seule observation.
(c) $\E\abs{X_1} = \frac2\pi\int_0^\infty\frac{x\,\dd x}{1 +
x^2} = +\infty$~: la \omterm{def:b3:probability:space}{loi} de Cauchy n'est pas \omterm{def:b3:lebesgue:l1}{intégrable},
donc la \omterm{def:b3:probability:space}{loi} forte des grands nombres
(\cref{thm:b3:probability:slln}) ne s'applique pas, et le
TCL (qui demande une variance finie) encore moins. Ici leurs
conclusions échouent vraiment, pas seulement leurs preuves.
Contrôle de cohérence~: $\varphi(\xi) = \eu^{-\abs\xi}$
n'est pas différentiable en $0$, comme
\cref{prop:b3:clt:cfbasics}(c) le prédit par contraposée
pour une variable non \omterm{def:b3:lebesgue:l1}{intégrable}.
\end{solution}

\begin{solution}{exo:b3:clt:11}
(a) $\varphi_{aX_1 + bX_2}(\xi) =
\eu^{-a\abs\xi}\eu^{-b\abs\xi} = \eu^{-(a+b)\abs\xi} =
\varphi_{(a+b)X_1}(\xi)$ (\omterm{def:b3:probability:independence}{indépendance} et
\cref{exo:b3:clt:10})~; l'injectivité identifie les \omterm{def:b3:probability:space}{lois}.

(b) $\varphi_{aX_1+bX_2}(\xi) = \eu^{-a^2\xi^2/2}
\eu^{-b^2\xi^2/2} = \eu^{-(a^2+b^2)\xi^2/2}$~: la \omterm{def:b3:probability:space}{loi} de
$\sqrt{a^2+b^2}\,X_1$.

(c) Si $\varphi_X(\xi) = \eu^{-c\abs\xi^\alpha}$, alors $S_n
= X_1 + \dots + X_n$ a $\varphi_{S_n} =
\eu^{-cn\abs\xi^\alpha}$, et $S_n/n^{1/\alpha}$ a
$\varphi(\xi) = \eu^{-c\abs\xi^\alpha}$ de nouveau~:
auto-reproduction \omterm{def:b3:forms:closedexact}{exacte} sous l'échelle $n^{1/\alpha}$ ---
$\sqrt n$ pour la gaussienne ($\alpha = 2$), $n$ lui-même
pour Cauchy ($\alpha = 1$, \cref{exo:b3:clt:10}(b)). Une
somme de variables i.i.d.\ ne peut converger (après
normalisation affine) que vers une \omterm{def:b3:probability:space}{loi} \omterm{def:b3:ode:stability}{stable} sous de telles
convolutions~; le TCL dit qu'une variance finie force le
bassin \omterm{def:b3:clt:gaussianvector}{gaussien}, et le bassin de Cauchy est réservé aux \omterm{def:b3:probability:space}{lois}
à queues si lourdes que $\E X^2 = \infty$ et même $\E\abs X
= \infty$ --- p.ex.\ des sommes de variables de Cauchy
elles-mêmes. L'universalité a plusieurs îles, indexées par
l'exposant de queue $\alpha \in \intoc02$.
\end{solution}

\begin{solution}{exo:b3:clt:12}
(a) Les indicateurs $\mathbf 1_{X_k \leq t}$ sont i.i.d.\
Bernoulli de paramètre $p = F(t)$~: leur somme $nF_n(t)$ est
binomiale $\mathcal B(n, p)$~; la \omterm{def:b3:probability:space}{loi} forte donne $F_n(t)
\to p$ p.s., et le TCL (\cref{thm:b3:clt:clt}) appliqué aux
mêmes indicateurs (variance $p(1-p)$) donne la limite
gaussienne indiquée.

(b) $p(1 - p)$ est \omterm{def:b3:rings:primemaximal}{maximal} en $p = \frac12$, i.e.\ où $F(t)
= \frac12$~: à la \emph{médiane}. Estimer des probabilités
de queue est asymptotiquement facile (variance $\to 0$
quand $p \to 0, 1$)~; la région médiane porte le plus grand
bruit statistique --- la courbe empirique oscille le plus au
milieu.

(c) Pour $F$ \omterm{def:b3:topology:continuity}{continue}, les variables $U_k = F(X_k)$ sont
i.i.d.\ uniformes sur $\intoo01$
(\cref{exo:b3:probability:1}), et la monotonie de $F$
donne, en notant $G_n$ la fonction de répartition empirique
des $U_k$~:
\[
\sup_{t\in\R}\,\abs{F_n(t) - F(t)}
= \sup_{u \in \operatorname{im}F}\,\abs{G_n(u) - u}
= \sup_{u\in\intcc01}\abs{G_n(u) - u} :
\]
la première égalité car $\{X_k \leq t\} = \{U_k \leq F(t)\}$
à des événements négligeables près (monotonie~; l'inégalité
stricte ne peut échouer que sur les parties plates de $F$,
où les deux membres sont inchangés), et la seconde car une
$F$ \omterm{def:b3:topology:continuity}{continue}, allant de $0$ à $1$, atteint toute valeur de
$\intoo01$ (théorème des valeurs intermédiaires), et les
extrémités n'ajoutent rien ($G_n(0) - 0 = 0$ et $G_n(1) - 1
= 0$). Le second membre n'implique que des uniformes~: une
seule \omterm{def:b3:probability:space}{loi} pour toutes les $F$ \omterm{def:b3:topology:continuity}{continues}.
\end{solution}

\begin{solution}{pb:b3:clt:1}
\textbf{1.} La famille $(X_1, \dots, X_n, N_1, \dots, N_n)$
est indépendante~: les deux blocs sont indépendants l'un de
l'autre par construction et chaque bloc est i.i.d. $W_i$ est
une \omterm{def:b3:lebesgue:measurable}{fonction mesurable} des seules variables $(X_j)_{j<i}$ et
$(N_j)_{j>i}$, toutes distinctes de $X_i$ et $N_i$~: par le
principe de coalition
(\cref{thm:b3:probability:independence}), $W_i$ est
indépendante de la paire $(X_i, N_i)$. Les décompositions
$H_i = W_i + \frac{X_i}{\sqrt n}$ et $H_{i-1} = W_i +
\frac{N_i}{\sqrt n}$ sont immédiates d'après les définitions~:
passer de $H_i$ à $H_{i-1}$ échange le seul terme $X_i$ pour
$N_i$.

\textbf{2.} Taylor--Lagrange à l'ordre~$3$~: il existe $c$
entre $w$ et $w + h$ tel que
\[
f(w + h) = f(w) + f'(w)h + \tfrac12 f''(w)h^2 + \tfrac16 f'''(c)h^3,
\]
et $\abs{f'''(c)} \leq M_3$ donne la borne.

\textbf{3.} En soustrayant les deux développements au point de
\omterm{def:b3:topology:basis}{base} commun $w = W_i$~:
\[
f(H_i) - f(H_{i-1}) = f'(W_i)\,\frac{X_i - N_i}{\sqrt n} +
\frac{f''(W_i)}{2}\,\frac{X_i^2 - N_i^2}{n} + R_i,
\qquad
\abs{R_i} \leq \frac{M_3}{6}\cdot
\frac{\abs{X_i}^3 + \abs{N_i}^3}{n^{3/2}} .
\]
Passer aux \omterm{def:b3:probability:space}{espérances}. Par la question~1, $f'(W_i)$ et
$f''(W_i)$ sont \omterm{def:b3:probability:independence}{indépendantes} de $(X_i, N_i)$, donc les
\omterm{def:b3:probability:space}{espérances} mixtes se factorisent~:
\begin{align*}
\E\Bigl[f'(W_i)\,\frac{X_i - N_i}{\sqrt n}\Bigr] &=
\E\bigl[f'(W_i)\bigr]\,\frac{\E X_i - \E N_i}{\sqrt n} = 0,
\\
\E\Bigl[f''(W_i)\,\frac{X_i^2 - N_i^2}{n}\Bigr] &=
\E\bigl[f''(W_i)\bigr]\,\frac{1 - 1}{n} = 0 :
\end{align*}
les deux premiers moments de $X_i$ et $N_i$ \emph{coïncident},
et seul le reste survit~:
\[
\bigl|\E f(H_i) - \E f(H_{i-1})\bigr| \leq \E\abs{R_i} \leq
\frac{M_3}{6}\cdot\frac{\beta + \gamma}{n^{3/2}} .
\]
Le moment d'ordre trois \omterm{def:b3:clt:gaussianvector}{gaussien}~: $\gamma = \E\abs{N_1}^3 =
2\int_0^\infty x^3\,\frac{\eu^{-x^2/2}}{\sqrt{2\pi}}\,\dd x
= \frac{2}{\sqrt{2\pi}}\int_0^\infty 2u\,\eu^{-u}\dd u =
\frac{4}{\sqrt{2\pi}} = \frac{2\sqrt2}{\sqrt\pi}$
(substitution $u = x^2/2$, puis $\Gamma(2) = 1$).

\textbf{4.} Télescopage $\E f(T_n) - \E f(G_n) =
\sum_{i=1}^n\bigl(\E f(H_i) - \E f(H_{i-1})\bigr)$ et
application de la question~3 à chacun des $n$ termes~:
\[
\bigl|\E f(T_n) - \E f(G_n)\bigr| \leq
n \cdot \frac{M_3(\beta + \gamma)}{6\,n^{3/2}} =
\frac{M_3\,(\beta + \gamma)}{6\,\sqrt n} .
\]

\textbf{5.} $G_n$ est $\mathcal N(0,1)$ \emph{exactement}
pour tout $n$ (somme normalisée de gaussiennes standards
\omterm{def:b3:probability:independence}{indépendantes}, \cref{exo:b3:clt:3}), donc $\E f(G_n) =
\E f(N)$ et la question~4 se lit $\abs{\E f(T_n) - \E f(N)}
\leq \frac{M_3(\beta+\gamma)}{6\sqrt n} \to 0$ pour $f \in
\mathcal C^3_b$. \emph{Passage au cas général.} Soit $f$
\omterm{def:b3:topology:continuity}{continue} bornée, $M = \sup\abs f$, $\varepsilon > 0$. Choisir
$A \geq 1$ avec $\frac1{A^2} \leq \varepsilon$~: Tchebychev
avec $\V(T_n) = 1$ donne $\P(\abs{T_n} > A) \leq
\varepsilon$ pour tout $n$, et de même $\P(\abs N > A) \leq
\varepsilon$. Soit $\chi$ de classe $\mathcal C^\infty$ avec
$\mathbf 1_{\intcc{-A}A} \leq \chi \leq
\mathbf 1_{\intcc{-A-1}{A+1}}$ (plateau lisse, obtenu en
régularisant $\mathbf 1_{\intcc{-A-\frac12}{A+\frac12}}$,
\cref{thm:b3:lp:regularization})~; $g = f\chi$ est \omterm{def:b3:topology:continuity}{continue}
à support compact, donc uniformément \omterm{def:b3:topology:continuity}{continue}, et sa
régularisée $g_\eta = g * \rho_\eta$ est $\mathcal C^\infty$
à dérivées de tous ordres bornées avec $\norm{g -
g_\eta}_\infty \leq \varepsilon$ pour $\eta$ assez petit.
Pour $T = T_n$ ou $N$, comme $f = g$ sur $\intcc{-A}A$ et
$\abs{f - g} \leq 2M$ partout~:
\[
\bigl|\E f(T) - \E g_\eta(T)\bigr| \leq
\E\abs{(f - g)(T)} + \norm{g - g_\eta}_\infty
\leq 2M\,\P(\abs T > A) + \varepsilon \leq
(2M + 1)\,\varepsilon .
\]
En combinant avec $\E g_\eta(T_n) \to \E g_\eta(N)$ (la
question~4 s'applique~: $g_\eta \in \mathcal C^3_b$)~:
\[
\limsup_n\;\bigl|\E f(T_n) - \E f(N)\bigr| \leq
2(2M + 1)\,\varepsilon ,
\]
et $\varepsilon$ était arbitraire~: $\E f(T_n) \to \E f(N)$
pour toute $f$ \omterm{def:b3:topology:continuity}{continue} bornée, i.e.\ $T_n \Rightarrow
\mathcal N(0,1)$.

\textbf{6.} L'équidistribution n'entrait que par une phrase~:
«~$X_i$ et $N_i$ ont les mêmes deux premiers moments~».
Soient donc $X_1, \dots, X_n$ \omterm{def:b3:probability:independence}{indépendantes}, centrées, de
variances $\sigma_i^2$ et moments d'ordre trois finis, $s_n^2
= \sum_i\sigma_i^2 > 0$, et $N_i \sim \mathcal N(0,
\sigma_i^2)$ \omterm{def:b3:probability:independence}{indépendantes} de tout le reste. Définir les
hybrides avec la normalisation $s_n$~: $H_i =
\frac1{s_n}(\sum_{j\leq i}X_j + \sum_{j>i}N_j)$. Au $i$-ième
échange, $\E X_i = \E N_i = 0$ et $\E X_i^2 = \E N_i^2 =
\sigma_i^2$ tuent encore les termes en $f'$ et $f''$, et le
reste donne (en utilisant $\E\abs{N_i}^3 = \sigma_i^3\gamma$
par homothétie)~:
\[
\bigl|\E f(H_i) - \E f(H_{i-1})\bigr| \leq
\frac{M_3}{6\,s_n^3}\bigl(\E\abs{X_i}^3 +
\sigma_i^3\gamma\bigr) .
\]
Télescopage~:
\[
\Bigl|\E f\Bigl(\frac{X_1 + \dots + X_n}{s_n}\Bigr) -
\E f(N)\Bigr| \leq \frac{M_3}{6\,s_n^3}\sum_{i=1}^n
\Bigl(\E\abs{X_i}^3 + \V(X_i)^{3/2}\,\gamma\Bigr) .
\]
Comme $\sigma_i^3 = (\E X_i^2)^{3/2} \leq \E\abs{X_i}^3$
(inégalité des moyennes de puissance, i.e.\ Jensen pour $t
\mapsto t^{3/2}$ appliqué à $X_i^2$), le second membre est
au plus $\frac{M_3(1 + \gamma)}{6}\cdot
\frac{\sum_i\E\abs{X_i}^3}{s_n^3}$~: sous la
\emph{condition de Lyapunov} $\frac1{s_n^3}\sum_i\E\abs{X_i}^3
\to 0$, les sommes normalisées convergent en \omterm{def:b3:probability:space}{loi} vers
$\mathcal N(0,1)$ --- le TCL sans équidistribution.

\textbf{7.} Soit $\rho \in \mathcal C^\infty_c(\intoo01)$
avec $\int\rho = 1$ et posons $\psi(x) =
\int_x^1\rho(s)\dd s$~: $\psi$ est $\mathcal C^\infty$,
décroissante, $\psi = 1$ sur $\R_-$, $\psi = 0$ sur
$\intco1\infty$~; soit $K = \norm{\psi'''}_\infty$. Pour
$t \in \R$ et $\delta > 0$ définir $\psi_\delta(x) =
\psi\bigl(\frac{x - t}\delta\bigr)$ et
$\tilde\psi_\delta(x) = \psi\bigl(\frac{x - t}\delta +
1\bigr)$~: ce sont des fonctions $\mathcal C^3_b$ de dérivée
tierce bornée par $K/\delta^3$, et
\[
\mathbf 1_{\intoc{-\infty}{t-\delta}} \leq
\tilde\psi_\delta \leq \mathbf 1_{\intoc{-\infty}t} \leq
\psi_\delta \leq \mathbf 1_{\intoc{-\infty}{t+\delta}} .
\]
Borne supérieure~: par la question~4 appliquée à
$\psi_\delta$ (avec $M_3 = K/\delta^3$),
\[
\P(T_n \leq t) \leq \E\psi_\delta(T_n) \leq
\E\psi_\delta(N) + \frac{K(\beta +
\gamma)}{6\,\delta^3\sqrt n}
\leq \Phi(t + \delta) + \frac{K(\beta +
\gamma)}{6\,\delta^3\sqrt n}
\leq \Phi(t) + \frac{\delta}{\sqrt{2\pi}} +
\frac{K(\beta + \gamma)}{6\,\delta^3\sqrt n},
\]
car $\Phi$ est lipschitzienne de constante
$\frac1{\sqrt{2\pi}}$ (sa \omterm{ex:b3:lebesgue:gamma}{densité} est bornée par
$\frac1{\sqrt{2\pi}}$). La borne inférieure symétrique via
$\tilde\psi_\delta$ donne l'estimation à deux termes
\[
\sup_{t\in\R}\,\bigl|\P(T_n \leq t) - \Phi(t)\bigr| \leq
\frac{K(\beta + \gamma)}{6}\cdot\frac{1}{\delta^3\sqrt n} +
\frac{\delta}{\sqrt{2\pi}}
\qquad(\delta > 0\ \text{arbitraire}).
\]
Les deux termes s'équilibrent lorsque $\delta^{-3}n^{-1/2}
\asymp \delta$, i.e.\ $\delta = n^{-1/8}$~: tous deux sont
alors $O(n^{-1/8})$, un taux uniforme explicite valable pour
tout $n$. (Le taux optimal de Berry--Esseen $C\beta/\sqrt n$
demande l'inégalité de lissage de Fourier~; l'échange troque
la netteté contre une élémentaire complète.)

\textbf{8.} Pour $X_i = 2B_i - 1$ (signes équitables)~:
centrées, variance~$1$, et $\abs{X_i} = 1$ donc $\beta = 1$.
La question~7 borne alors $\sup_t\abs{\P(\frac{S_n}{\sqrt n}
\leq t) - \Phi(t)}$ explicitement et uniformément pour
\emph{tout} $n$ fini --- un énoncé global, non asymptotique,
sur la fonction de répartition. L'estimation locale de
\cref{pb:b3:product:1}, question~7, donne au contraire
l'asymptotique \omterm{def:b3:forms:closedexact}{exacte} d'un atome individuel, $\P(S_{2n} =
2k) \sim \frac{\eu^{-k^2/n}}{\sqrt{\pi n}}$~: elle résout
des probabilités de taille $n^{-1/2}$, bien en dessous de la
résolution $n^{-1/8}$ de la question~7, mais elle est
ponctuelle, asymptotique (pas d'erreur explicite à $n$ fixe)
et liée à cette \omterm{def:b3:probability:space}{loi} de réseau particulière. Précision locale
contre uniformité globale~: les deux méthodes sont
complémentaires, et sommer l'estimation locale sur $k \in
\intint{a\sqrt n}{b\sqrt n}$ retrouve de Moivre--Laplace sur
des intervalles --- avec un taux plus fin, mais seulement
pour cette \omterm{def:b3:probability:space}{loi}.

\textbf{9.} L'argument d'échange n'utilisait rien sur la \omterm{def:b3:probability:space}{loi}
des $X_i$ au-delà de $\E X_i = 0$, $\E X_i^2 = 1$ et la
finitude de $\E\abs{X_i}^3$~: si l'on remplaçait les
gaussiennes $N_i$ par toute autre famille i.i.d.\ de mêmes
deux premiers moments et de moment d'ordre trois fini, le
même télescopage bornerait $\abs{\E f(\mathrm{somme}_X) - \E
f(\mathrm{somme}_Y)}$ par $O(n^{-1/2})$ pour toute $f$ lisse.
Les statistiques lisses de grandes sommes \omterm{def:b3:probability:independence}{indépendantes} sont
donc \emph{universelles}~: à une erreur quantifiée près,
elles ne dépendent de la \omterm{def:b3:probability:space}{loi} des termes de la somme que par deux
nombres. C'est le principe d'invariance~: prouver un théorème
limite pour la \omterm{def:b3:probability:space}{loi} la plus calculable (la gaussienne, où tout
est exact), puis le transférer à toutes les \omterm{def:b3:probability:space}{lois} par échange.
Le même schéma --- avec des sommes remplacées par des
fonctionnelles plus élaborées --- guide la \omterm{def:b3:probability:space}{loi} du
demi-cercle de Wigner pour les matrices aléatoires,
l'universalité des racines de polynômes aléatoires, et une
grande part de la probabilité moderne~; le théorème central
limite en est la première et la plus \omterm{def:b3:groups:simple}{simple} instance.

\textbf{10.} $Z$ est une fonction borélienne des seules
$(N_{i+1}, \dots, N_n)$, tandis que $A = W_i + \theta h - Z$
et $h$ sont des fonctions des variables restantes de la
famille indépendante $(X_1, \dots, X_n, N_1, \dots, N_n)$~:
par le principe de coalition
(\cref{thm:b3:probability:independence}), $Z$ est
indépendante de $(A, h)$. Comme somme des $N_j/\sqrt n \sim
\mathcal N(0, \frac1n)$ \omterm{def:b3:probability:independence}{indépendantes}, $Z \sim \mathcal
N(0, s^2)$ avec $s^2 = \frac{n-i}n$ (\cref{exo:b3:clt:3}),
de \omterm{ex:b3:lebesgue:gamma}{densité} bornée par $\frac1{s\sqrt{2\pi}}$. La \omterm{def:b3:probability:space}{loi} de
$((A, h), Z)$ est le produit des deux \omterm{def:b3:probability:space}{lois} marginales, donc
Tonelli (transfert) fige le premier bloc~: avec $G(a) =
\E\abs{g(a + Z)} = \int\abs{g(a + z)}\,\varphi_s(z)\,\dd z
\leq \frac{\norm g_{L^1}}{s\sqrt{2\pi}}$ pour tout $a$,
\[
\E\bigl[\abs h^3\abs{g(A + Z)}\bigr] =
\E\bigl[\abs h^3\,G(A)\bigr] \leq
\frac{\norm g_{L^1}}{s\sqrt{2\pi}}\,\E\abs h^3
= \sqrt{\frac{n}{2\pi(n-i)}}\,\norm g_{L^1}\,\E\abs h^3 .
\]

\textbf{11.} La forme intégrale de Taylor s'obtient en
intégrant $f(w + h) - f(w) = h\int_0^1f'(w + \theta
h)\,\dd\theta$ par parties deux fois en $\theta$. En prenant
les \omterm{def:b3:probability:space}{espérances} dans le $i$-ième échange, les ordres $0$, $1$,
$2$ s'annulent exactement comme à la question~3, et les deux
restes (pour $h = X_i/\sqrt n$ et $N_i/\sqrt n$) sont bornés,
pour $i \leq n - 1$, par la question~10 avec $g = f'''$~:
\[
\bigl|\E f(H_i) - \E f(H_{i-1})\bigr| \leq
\int_0^1\frac{(1-\theta)^2}2\,\dd\theta\;
\sqrt{\frac{n}{2\pi(n-i)}}\,\norm{f'''}_{L^1}
\frac{\beta + \gamma}{n^{3/2}}
= \frac{\beta + \gamma}{6\,n^{3/2}}
\sqrt{\frac{n}{2\pi(n-i)}}\,\norm{f'''}_{L^1} .
\]
En sommant, avec $\sum_{i=1}^{n-1}\sqrt{\frac n{n-i}} =
\sqrt n\sum_{m=1}^{n-1}m^{-1/2} \leq 2n$, et en ajoutant la
borne de la question~3 pour le dernier échange ($i = n$, plus
aucune gaussienne)~:
\[
\bigl|\E f(T_n) - \E f(G_n)\bigr| \leq
\frac{\beta + \gamma}{3\sqrt{2\pi}}\cdot
\frac{\norm{f'''}_{L^1}}{\sqrt n} +
\frac{M_3(\beta + \gamma)}{6\,n^{3/2}} .
\]
Rampes~: $\psi_\delta'''(x) =
\delta^{-3}\psi'''\bigl(\frac{x - t}\delta\bigr)$, donc
$M_3 = K\delta^{-3}$ avec $K = \norm{\psi'''}_\infty$ et
$\norm{\psi_\delta'''}_{L^1} = \delta^{-2}
\norm{\psi'''}_{L^1} = K_1\delta^{-2}$ (substitution). Le
sandwich de la question~7 donne alors
\[
\sup_t\,\bigl|\P(T_n \leq t) - \Phi(t)\bigr| \leq
\frac{K_1(\beta + \gamma)}{3\sqrt{2\pi}}\cdot
\frac1{\delta^2\sqrt n} +
\frac{K(\beta + \gamma)}{6}\cdot
\frac1{\delta^3n^{3/2}} + \frac\delta{\sqrt{2\pi}} .
\]
En $\delta = n^{-1/6}$ les premier et troisième termes sont
$O(n^{-1/6})$ et le médian $O(n^{-1})$~: un taux uniforme
$O(n^{-1/6})$, strictement meilleur que le $n^{-1/8}$ de la
question~7 --- la moitié gaussienne de l'hybride a fait le
lissage supplémentaire.

\textbf{12.} $\E N_1^3 = 0$ (intégrande impaire), et une
intégration par parties donne $\E N_1^4 = 3\,\E N_1^2 = 3$
($\int x^3\cdot x\varphi(x)\dd x = 3\int x^2\varphi$). Pour
$f$ de classe $\mathcal C^4$ à dérivées bornées, développer
chaque échange à l'ordre quatre~: les termes d'ordre trois
portent le facteur $\E X_i^3 - \E N_i^3 = 0$ (l'\omterm{def:b3:probability:independence}{indépendance}
les factorise comme à la question~3), donc seul le reste
d'ordre quatre $\int_0^1\frac{(1-\theta)^3}6f^{(4)}(w +
\theta h)h^4\dd\theta$ survit, avec $\int_0^1
\frac{(1-\theta)^3}6\dd\theta = \frac1{24}$ et $\E h^4 =
\beta_4n^{-2}$ ou $3n^{-2}$. La question~10 (avec $g =
f^{(4)}$) borne les échanges $i \leq n - 1$, et en sommant
comme à la question~11~:
\[
\bigl|\E f(T_n) - \E f(G_n)\bigr| \leq
\frac{\beta_4 + 3}{12\sqrt{2\pi}}\cdot
\frac{\norm{f^{(4)}}_{L^1}}{n} +
\frac{M_4(\beta_4 + 3)}{24\,n^2} .
\]
Avec $\norm{\psi_\delta^{(4)}}_{L^1} = K_2\delta^{-3}$ et
$M_4 = K'\delta^{-4}$, la borne de fonction de répartition
devient $C\delta^{-3}n^{-1} + C'\delta^{-4}n^{-2} +
\frac\delta{\sqrt{2\pi}}$~; en $\delta = n^{-1/4}$ les
termes extrêmes sont $O(n^{-1/4})$ et le médian $O(n^{-1})$~:
taux $O(n^{-1/4})$.

\textbf{13.} Avec $k$ moments appariés le reste survivant
par échange est d'ordre $\E\abs h^{k+1} \asymp
n^{-(k+1)/2}$~; la borne gaussienne cachée charge
$\norm{f^{(k+1)}}_{L^1}$ et la somme sur les échanges
contribue le facteur $2n$, donnant $\asymp
\norm{f^{(k+1)}}_{L^1}\, n^{-(k-1)/2}$ pour $f$ lisse. Les
rampes coûtent $\norm{\psi_\delta^{(k+1)}}_{L^1} \asymp
\delta^{-k}$, donc l'erreur de fonction de répartition est
$\asymp \delta^{-k}n^{-(k-1)/2} + \delta$, équilibrée en
$\delta = n^{-(k-1)/(2k+2)}$~: taux $n^{-(k-1)/(2k+2)}$,
qui vaut $n^{-1/6}$ pour $k = 2$, $n^{-1/4}$ pour $k = 3$,
et ne tend vers $n^{-1/2}$ que lorsque $k \to \infty$ ---
mais $k \geq 4$ forcerait $\E X_1^4 = 3$ et au-delà, i.e.\
une \omterm{def:b3:probability:space}{loi} qui imite déjà la gaussienne. La saturation est
structurelle~: l'échange ajoute $n$ erreurs d'échange
\emph{en valeur absolue}, renonçant à toute annulation entre
échanges. La preuve de Fourier compare les \omterm{def:b3:clt:cf}{fonctions
caractéristiques}, où les erreurs apparaissent avec leurs
phases oscillantes~; l'inégalité de lissage d'Esseen convertit
$\abs{\varphi_{T_n} - \varphi_N}$, intégrée contre
$\frac{\dd\xi}{\abs\xi}$, en une borne de fonction de
répartition au seul coût logarithmique, et livre le
$C\beta n^{-1/2}$ de Berry--Esseen à partir de trois moments.
Le remplacement troque l'optimalité contre la robustesse ---
et, comme le montre la Partie~VII, contre la portabilité.

\textbf{14.} $\Sigma$ est symétrique semi-définie positive~;
avec $\Sigma = PDP^{\mathsf T}$ ($P$ orthogonale, $D \geq 0$
diagonale, \cref{exo:b3:submanifolds:8}), la symétrique $C =
P\sqrt DP^{\mathsf T}$ vérifie $C^2 = \Sigma$. Pour tout $t
\in \R^2$, $\langle t, CZ\rangle = \langle Ct, Z\rangle =
(Ct)_1Z^1 + (Ct)_2Z^2$ est une combinaison linéaire de
gaussiennes \omterm{def:b3:probability:independence}{indépendantes}, donc gaussienne
(\cref{exo:b3:clt:3})~: $N = CZ$ est un \omterm{def:b3:clt:gaussianvector}{vecteur gaussien}~;
sa moyenne est $0$ et sa covariance $\E[NN^{\mathsf T}] =
C\,\E[ZZ^{\mathsf T}]\,C^{\mathsf T} = CC^{\mathsf T} =
\Sigma$. Moments~: $\norm N^3 \leq (\abs{N_1} +
\abs{N_2})^3 \leq 4(\abs{N_1}^3 + \abs{N_2}^3)$ (convexité
de $x^3$ sur $\R_+$), et chaque coordonnée est une
gaussienne réelle à moments de tous ordres
(\cref{exo:b3:product:10})~: $\gamma' < \infty$. Enfin
chaque $\langle t, G_n\rangle = \frac1{\sqrt
n}\sum_i\langle t, N_i\rangle$ est une somme normalisée de
$\mathcal N(0, t^{\mathsf T}\Sigma t)$ i.i.d., donc
exactement $\mathcal N(0, t^{\mathsf T}\Sigma t)$~: $G_n$
est un \omterm{def:b3:clt:gaussianvector}{vecteur gaussien} de moyenne $0$ et covariance
$\Sigma$, et sa \omterm{def:b3:probability:space}{loi} est $\mathcal N(0, \Sigma)$
(\cref{def:b3:clt:gaussianvector}~: la \omterm{def:b3:probability:space}{loi} est déterminée
par ces données).

\textbf{15.} Posons $\phi(t) = f(w + th)$, $t \in \intcc01$~:
$\phi$ est $\mathcal C^3$ avec
\[
\phi'''(t) = \sum_{j,k,l\in\{1,2\}}\partial_{jkl}f(w +
th)\,h_jh_kh_l, \qquad \abs{\phi'''(t)} \leq
M_3\Bigl(\sum_j\abs{h_j}\Bigr)^3 = M_3(\abs{h_1} +
\abs{h_2})^3 .
\]
Taylor--Lagrange à l'ordre~$3$ pour $\phi$ entre $0$ et $1$
donne la première inégalité~; Cauchy--Schwarz donne
$\abs{h_1} + \abs{h_2} \leq \sqrt2\norm h$, d'où la
constante $\frac{2\sqrt2M_3}6 = \frac{\sqrt2M_3}3$.

\textbf{16.} Définir $H_i$ et $W_i$ comme à la question~1,
maintenant dans $\R^2$~; l'argument de coalition est
inchangé. Au $i$-ième échange, les termes d'ordre un donnent
$\sum_j\E[\partial_jf(W_i)]\,(\E X_{i,j} - \E N_{i,j})/
\sqrt n = 0$ et les termes d'ordre deux donnent
$\frac1{2n}\sum_{j,k}\E[\partial_{jk}f(W_i)]\,(\Sigma_{jk}
- \Sigma_{jk}) = 0$~: moyennes et covariances coïncident. La
question~15 borne les deux restes~:
\[
\bigl|\E f(H_i) - \E f(H_{i-1})\bigr| \leq
\frac{\sqrt2M_3}3\cdot\frac{\E\norm{X_i}^3 +
\E\norm{N_i}^3}{n^{3/2}}
= \frac{\sqrt2M_3(\beta' + \gamma')}{3\,n^{3/2}},
\]
et le télescopage sur les $n$ échanges~:
\[
\Bigl|\E f\Bigl(\frac{S_n}{\sqrt n}\Bigr) - \E
f(G_n)\Bigr| \leq \frac{\sqrt2\,M_3(\beta' +
\gamma')}{3\sqrt n},
\qquad G_n \sim \mathcal N(0, \Sigma)\ \text{exactement} .
\]
Passage au cas général~: $\E\norm{T_n}^2 = \E\norm{X_1}^2 =
\operatorname{tr}\Sigma$ (les termes croisés s'annulent par
\omterm{def:b3:probability:independence}{indépendance} et centrage), donc $\P(\norm{T_n} > A) \leq
\operatorname{tr}\Sigma/A^2$, et de même pour $N$~:
tension. Étant donnée $f$ \omterm{def:b3:topology:continuity}{continue} bornée et $\varepsilon
> 0$, multiplier par un plateau lisse $\chi$ égal à $1$ sur
la boule de rayon $A$ et supporté dans le rayon $A + 1$
(régulariser un indicateur dans $\R^2$,
\cref{thm:b3:lp:regularization})~; $g = f\chi$ est
uniformément \omterm{def:b3:topology:continuity}{continue} à support compact, donc sa
régularisée bidimensionnelle $g_\eta$ est $\mathcal
C^\infty$ à dérivées de tous ordres bornées avec $\norm{g -
g_\eta}_\infty \leq \varepsilon$ pour $\eta$ petit. La
chaîne à trois~$\varepsilon$ de la question~5 se transporte
mot pour mot~: $\E f(T_n) \to \E f(N)$ pour toute $f$
\omterm{def:b3:topology:continuity}{continue} bornée $\R^2 \to \R$. C'est
\cref{thm:b3:clt:multiclt} pour $d = 2$, maintenant
prouvé --- l'échange contourne le théorème de Lévy
bidimensionnel que le chapitre avait laissé admis.

\textbf{17.} Pour $g \colon \R \to \R$ \omterm{def:b3:topology:continuity}{continue} bornée, la
carte $x \mapsto g(\langle t, x\rangle)$ est \omterm{def:b3:topology:continuity}{continue}
bornée sur $\R^2$, donc la question~16 donne $\E g(\langle
t, T_n\rangle) \to \E g(\langle t, N\rangle)$~: toute
projection converge en \omterm{def:b3:probability:space}{loi}, et $\langle t, N\rangle \sim
\mathcal N(0, t^{\mathsf T}\Sigma t)$. (C'est le sens
facile de Cramér--Wold~: la convergence jointe implique la
convergence de toutes les images linéaires.) Application~:
$V_i = (\xi_i, \xi_i^2 - 1)$ sont des vecteurs i.i.d.\
centrés ($\E\xi_1^2 = 1$), avec entrées de covariance
$\V(\xi_1) = 1$, $\operatorname{Cov}(\xi_1, \xi_1^2 - 1) =
\E\xi_1^3$ et $\V(\xi_1^2 - 1) = \E\xi_1^4 - 1$~; le moment
d'ordre trois $\E\norm{V_1}^3 \leq 4\bigl(\E\abs{\xi_1}^3 +
\E\abs{\xi_1^2 - 1}^3\bigr)$ est fini dès que $\xi_1 \in
L^6$. La question~16 livre la limite gaussienne jointe
affichée, et \cref{thm:b3:clt:gaussianvector}(2)~: les deux
coordonnées limites sont \omterm{def:b3:probability:independence}{indépendantes} exactement lorsque la
covariance $\E\xi_1^3$ s'annule --- pour les \omterm{def:b3:probability:space}{lois}
symétriques, moyenne empirique et variance empirique se
découplent asymptotiquement.

\textbf{18.} Écrire $g(x) - g(\theta) = (g'(\theta) +
\eta(x))(x - \theta)$ où $\eta(x) = \frac{g(x) -
g(\theta)}{x - \theta} - g'(\theta)$ pour $x \neq \theta$
et $\eta(\theta) = 0$~: la différentiabilité en $\theta$
signifie précisément $\eta(x) \to 0$ lorsque $x \to \theta$.
\emph{Étape~1~:} $\hat\theta_n \to \theta$ en probabilité~:
pour $\varepsilon > 0$ et tout $A > 0$, éventuellement
$\varepsilon\sqrt n \geq A$, donc $\P(\abs{\hat\theta_n -
\theta} > \varepsilon) \leq \P(\abs{\sqrt
n(\hat\theta_n - \theta)} > A) \to \P(\sigma\abs N > A)$
(les fonctions de répartition convergent aux points de
continuité $\pm A$), et le second membre tend vers $0$
lorsque $A \to \infty$. \emph{Étape~2~:} $\eta(\hat\theta_n)
\to 0$ en probabilité~: étant donné $\varepsilon' > 0$,
choisir $\delta$ avec $\abs\eta \leq \varepsilon'$ sur
$\abs{x - \theta} \leq \delta$~; alors
$\P(\abs{\eta(\hat\theta_n)} > \varepsilon') \leq
\P(\abs{\hat\theta_n - \theta} > \delta) \to 0$.
\emph{Étape~3~:}
\[
\sqrt n\bigl(g(\hat\theta_n) - g(\theta)\bigr) =
g'(\theta)\,\sqrt n(\hat\theta_n - \theta) +
\eta(\hat\theta_n)\cdot\sqrt n(\hat\theta_n - \theta) .
\]
Par la règle produit de Slutsky (\cref{exo:b3:clt:8}, avec
la suite $\eta(\hat\theta_n) \to 0$ en probabilité et
$\sqrt n(\hat\theta_n - \theta)$ convergente en \omterm{def:b3:probability:space}{loi}), le
second terme converge en \omterm{def:b3:probability:space}{loi} vers $0\cdot\mathcal N(0,
\sigma^2) = 0$, donc vers $0$ en probabilité
(\cref{exo:b3:clt:4}(b))~; le premier converge en \omterm{def:b3:probability:space}{loi} vers
$g'(\theta)\mathcal N(0, \sigma^2)$ (Slutsky encore, ou la
règle affine pour les \omterm{def:b3:clt:cf}{fonctions caractéristiques})~; la règle
somme de Slutsky les assemble~: la limite est $\mathcal
N(0, g'(\theta)^2\sigma^2)$.

\textbf{19.} (a) Le TCL donne $\sqrt n(\bar X_n - \mu)
\Rightarrow \mathcal N(0, \sigma^2)$~; la méthode delta avec
$g(x) = x^2$, $g'(\mu) = 2\mu$, donne $\sqrt n(\bar X_n^2 -
\mu^2) \Rightarrow \mathcal N(0, 4\mu^2\sigma^2)$ ---
dégénérée (limite~$0$) lorsque $\mu = 0$. Dans ce cas la
fluctuation vit une échelle au-dessus~: $n\bar X_n^2 =
(\sqrt n\,\bar X_n)^2$, et pour $t > 0$
\[
\P\bigl(n\bar X_n^2 \leq t\bigr) = \P\bigl(-\sqrt t \leq
\sqrt n\,\bar X_n \leq \sqrt t\bigr) \longrightarrow
\Phi\Bigl(\frac{\sqrt t}\sigma\Bigr) -
\Phi\Bigl(-\frac{\sqrt t}\sigma\Bigr) = \P(\sigma^2N^2
\leq t) :
\]
$n\bar X_n^2 \Rightarrow \sigma^2N^2$, le carré d'une
gaussienne (\omterm{def:b3:probability:space}{loi} du «~khi-deux~») --- lorsque la dérivée
première meurt, le terme d'ordre deux de Taylor dicte une
limite non gaussienne. (b) Ici $\sqrt n(\hat p_n - p)
\Rightarrow \mathcal N(0, p(1 - p))$ et $g(p) =
\arcsin\sqrt p$ a $g'(p) = \frac1{2\sqrt{p(1 - p)}}$, donc
$g'(p)^2\,p(1 - p) = \frac14$~: la limite est $\mathcal
N(0, \frac14)$ pour tout $p \in \intoo01$. Sur l'échelle
$\arcsin$ la barre d'erreur asymptotique à $95\%$ est $\pm
\frac{0{,}98}{\sqrt n}$, connue à l'avance --- alors que
dans \cref{ex:b3:clt:confidence} la largeur impliquait le
$\sigma = \sqrt{p(1-p)}$ inconnu, à majorer par $\frac12$ ou
à estimer~: la transformation \emph{stabilise} la variance.

\textbf{20.} Soit $A^* = \{k : \mu(\{k\}) > \nu(\{k\})\}$ et
$\Delta_k = \mu(\{k\}) - \nu(\{k\})$, de sorte que
$\sum_k\Delta_k = 0$. Pour tout $A \subseteq \N$~: $\mu(A) -
\nu(A) = \sum_{k\in A}\Delta_k \leq \sum_{k\in
A^*}\Delta_k$, avec égalité en $A = A^*$~; et comme les
parties positive et négative de $(\Delta_k)$ ont même masse
totale, $\sum_{A^*}\Delta_k = \frac12\sum_k\abs{\Delta_k}$.
Échanger $\mu, \nu$ traite le signe~: $d_{\mathrm{TV}}(\mu,
\nu) = \frac12\sum_k\abs{\Delta_k}$. Couplage~: pour tout
$A$,
\[
\mu(A) - \nu(A) = \E\bigl[\mathbf 1_A(X) - \mathbf
1_A(Y)\bigr] = \E\bigl[(\mathbf 1_A(X) - \mathbf
1_A(Y))\,\mathbf 1_{X\neq Y}\bigr] \leq \P(X \neq Y),
\]
et prendre le supremum sur $A$.

\textbf{21.} Les deux \omterm{def:b3:probability:space}{lois} chargent~: $k = 0$~: $1 - p$ contre
$\eu^{-p}$, avec $\eu^{-p} > 1 - p$~; $k = 1$~: $p$ contre
$p\,\eu^{-p} < p$~; $k \geq 2$~: $0$ contre le reste Poisson
$1 - \eu^{-p} - p\eu^{-p} \geq 0$. D'où
\[
d_{\mathrm{TV}} = \tfrac12\bigl[(\eu^{-p} - 1 + p) + (p -
p\eu^{-p}) + (1 - \eu^{-p} - p\eu^{-p})\bigr] =
\tfrac12\bigl(2p - 2p\eu^{-p}\bigr) = p(1 - \eu^{-p}),
\]
et $1 - \eu^{-p} \leq p$ donne la borne $p^2$.

\textbf{22.} Écrire $H_{i-1} = W_i + X_i$ et $H_i = W_i +
Y_i$ avec $W_i = \sum_{j<i}Y_j + \sum_{j>i}X_j$,
indépendante de la paire $(X_i, Y_i)$ (coalitions). Pour $A
\subseteq \N$, en conditionnant sur les valeurs dénombrables
par \omterm{def:b3:probability:independence}{indépendance},
\[
\P(H_{i-1} \in A) = \sum_{k\geq0}\P(X_i = k)\,\P(W_i + k
\in A),
\]
et de même pour $H_i$ avec $Y_i$. En soustrayant, avec $c_k
= \P(W_i + k \in A) \in \intcc01$ et $\Delta_k = \P(X_i =
k) - \P(Y_i = k)$ de somme nulle~:
\[
\abs{\P(H_{i-1} \in A) - \P(H_i \in A)} =
\Bigl|\sum_k\Delta_k\bigl(c_k - \tfrac12\bigr)\Bigr| \leq
\tfrac12\sum_k\abs{\Delta_k} =
d_{\mathrm{TV}}\bigl(\mathcal B(1, p_i), \mathcal
P(p_i)\bigr) .
\]
Télescopage de $H_0 = S$ à $H_n = \sum_iY_i \sim \mathcal
P(\lambda)$ (\cref{exo:b3:clt:1}, itéré) et question~21~:
\[
\abs{\P(S \in A) - \P(\mathcal P(\lambda) \in A)} \leq
\sum_{i=1}^np_i\bigl(1 - \eu^{-p_i}\bigr) \leq
\sum_{i=1}^np_i^2
\qquad\text{pour tout } A :
\]
l'inégalité de Le Cam. (La borne de couplage de la
question~20 offre une voie alternative~: coupler chaque paire
sur une seule uniforme de sorte que $\P(X_i \neq Y_i) \leq
p_i^2$ et borner $\P(S \neq \sum Y_i)$~; l'échange n'exige
aucune construction.)

\textbf{23.} (a) Avec $p_i = \frac\lambda n$~:
$d_{\mathrm{TV}}(\text{loi de }S, \mathcal P(\lambda))
\leq \frac{\lambda^2}n$. Cela affine
\cref{exo:b3:clt:5} trois fois~: une erreur explicite à
chaque $n$ fini, l'uniformité sur tous les événements $A$ à
la fois (pas un intervalle à la fois), et pas besoin de
$p_i$ égaux --- seulement $\sum_ip_i^2$ petit, p.ex.\ $\sum
p_i^2 \leq \lambda\max_ip_i$~: \emph{beaucoup d'événements
rares, aucun dominant}. (b) Ici $n = 500$, $p_i =
\frac1{500}$, $\lambda = 1$~: le modèle de Poisson se trompe
d'au plus $500\cdot\frac1{500^2} = 0{,}002$ sur tout
événement~; en particulier, en prenant $A = \{0\}$,
\[
\P(\text{aucune lettre égarée}) = \Bigl(1 -
\frac1{500}\Bigr)^{500},
\qquad
\Bigl|\P(\text{aucune lettre égarée}) - \eu^{-1}\Bigr| \leq
0{,}002,
\]
donc la réponse est $\eu^{-1} \approx 0{,}368$ à $0{,}002$
garanti près (le vrai écart est d'environ $4\cdot10^{-4}$).
(c) Le problème se referme sur une méthode à deux régimes.
Lorsque $n$ contributions comparables portent chacune une
variance $\frac1n$, apparier \emph{deux} moments contre la
gaussienne rend les erreurs d'échange $o(\frac1n)$
chacune~: les sommes deviennent gaussiennes --- avec Taylor
comme outil de comparaison locale. Lorsque $n$ contributions
sont des indicateurs de probabilité $p_i$, apparier la
\emph{moyenne} contre un atome de Poisson fait coûter
$p_i^2$ chaque échange~: les comptages d'événements rares
deviennent Poisson --- avec la variation totale comme
comparaison locale \omterm{def:b3:forms:closedexact}{exacte}. Mêmes hybrides, même télescope,
estimation locale différente~: le remplacement est une
stratégie, pas un théorème, et les limites gaussienne et
Poisson en sont les deux plus anciens dividendes.

\textbf{24.} Le TCL donne $\sqrt n(\bar X_n - \mu)
\Rightarrow \mathcal N(0, \sigma^2)$, et $g(x) = \ln x$ est
différentiable en $\mu > 0$ avec $g'(\mu) = \frac1\mu$~: la
méthode delta (Partie~VI) livre $\sqrt n(\ln\bar X_n -
\ln\mu) \Rightarrow \mathcal N(0, \sigma^2/\mu^2)$. En
dépliants l'intervalle $\abs{\ln\bar X_n - \ln\mu} \leq
\frac{1{,}96\,\sigma}{\mu\sqrt n}$ par exponentiation~:
\[
\mu \in \bar X_n\cdot
\eu^{\pm1{,}96\,\sigma/(\mu\sqrt n)}
\qquad\text{avec probabilité asymptotique } 95\%
\]
(en pratique $\sigma/\mu$ est remplacé par sa version
empirique, Slutsky comme dans \cref{exo:b3:clt:8}).
L'intervalle multiplicatif est le naturel lorsque les
données sont positives avec des erreurs proportionnelles à
leur taille --- revenus, concentrations, demi-vies~:
quantités qui vivent sur une échelle logarithmique, où des
intervalles additifs symétriques pourraient même traverser
zéro.

\textbf{25.} $\E[(X - p)^3] = (1-p)^3p + (-p)^3(1 - p) =
p(1-p)\bigl[(1-p)^2 - p^2\bigr] = p(1-p)(1 - 2p)$. Dans
l'analyse d'échange (Partie~IV), le terme d'erreur principal
après appariement de deux moments porte le moment d'ordre
trois \emph{signé}~: pour $p < \frac12$ il est positif (la
\omterm{def:b3:probability:space}{loi} penche à droite~: rares grandes excursions au-dessus de
la moyenne), et l'approximation normale déplace
systématiquement la masse --- sous-estimant la queue gauche
courte et surestimant la droite --- avec une erreur d'ordre
$n^{-1/2}$~; en $p = \frac12$ le moment d'ordre trois
s'annule, la Bernoulli imite la gaussienne à l'ordre trois,
et le taux s'améliore (question des moments appariés de la
Partie~IV). Numériquement~: $\P(S = 0) = 0{,}9^{20} =
0{,}1216$, tandis que la gaussienne $\mathcal N(2, 1{,}8)$
donne $\Phi\bigl(\frac{0{,}5 - 2}{\sqrt{1{,}8}}\bigr) =
\Phi(-1{,}118) \approx 0{,}132$~: la courbe normale,
ignorante du mur en $0$ et de l'asymétrie vers la droite,
met trop de masse en bas --- le signe prédit de l'erreur,
visible à $n = 20$.
