\section*{Chapitre \ref{ch:b3:ode} --- Équations différentielles
ordinaires}

\begin{solution}{exo:b3:ode:1}
(a) Séparation des variables~: $x(t) = \frac1{1 - t}$ sur
$\intoo{-\infty}1$~: explosion en $T_+ = 1$, avec $x(t) \to
+\infty$ --- exactement le~\cref{thm:b3:ode:maximal}(2).
(b) $x(t) = \tan t$ sur $\intoo{-\pi/2}{\pi/2}$~: explosion
aux deux extrémités.
(c) $x(t) = \frac{1}{1 + \eu^{-t}}$, globale~: la solution
reste dans $\intoo01$, ensemble borné, donc le graphe ne
peut quitter tout compact de $\R\times\R$ en temps fini ---
$T_\pm = \pm\infty$. Noter que (a), (b) ne contredisent pas
le~\cref{cor:b3:ode:globallinear}~: $x^2$ et $1 + x^2$ ont
une croissance superlinéaire.
\end{solution}

\begin{solution}{exo:b3:ode:2}
(a) C'est l'estimée du~\cref{cor:b3:ode:uniqueness} avec
$t_0 = 0$. Optimalité~: pour $x' = Lx$ (globalement
$L$-lipschitzienne), deux solutions diffèrent exactement de
$(x_0 - y_0)\eu^{Lt}$.
(b) L'estimée se lit~: l'application de \omterm{ex:b3:ode:planeclassification}{flot} au temps $t$
est $\eu^{L\abs t}$-lipschitzienne en la condition initiale
--- continuité, uniformément pour $t$ dans les compacts~; sur
les ensembles bornés pour un $f$ non globalement
lipschitzien, on rejoue le même argument dans un tube compact
autour des trajectoires avec la constante locale.
\end{solution}

\begin{solution}{exo:b3:ode:3}
(a) $\abs{\sin(tx)} \leq 1$~: borné, c'est-à-dire croissance
linéaire avec $\alpha = 0$, $\beta = 1$~:
le~\cref{cor:b3:ode:globallinear} sur $U = \R\times\R$.
(b) $\abs{tx/(1 + x^2)} \leq \abs t\cdot\frac12$~: encore
sous-linéaire (en fait bornée sur les compacts de temps)~:
globale.
(c) $X = (x, x')$~: $X' = \bigl(\begin{smallmatrix}0 & 1\\ -q(t)
& 0\end{smallmatrix}\bigr)X$~: linéaire à coefficients
continus~: globale (cadre du~\cref{thm:b3:ode:linearstructure}).
(d) Globale~; Gronwall comme dans
le~\cref{cor:b3:ode:globallinear}~: $\norm{x(t)} \leq
\norm{x(t_0)}\,\eu^{M\abs{t - t_0}}$ avec $M =
\sup\vertiii{A}$.
\end{solution}

\begin{solution}{exo:b3:ode:4}
(a) $A^2 = -I$ pour la première~: $\eu^{tA} = \cos t\,I +
\sin t\,A = \bigl(\begin{smallmatrix}\cos t & -\sin t\\ \sin
t & \cos t\end{smallmatrix}\bigr)$. Bloc de Jordan~: $\lambda
I$ et $N$ commutent~: $\eu^{tA} =
\eu^{\lambda t}\bigl(\begin{smallmatrix}1 & t\\ 0 &
1\end{smallmatrix}\bigr)$. Troisième~: $A^2 = I$~: $\eu^{tA}
= \cosh t\,I + \sinh t\,A$.

(b) Système $X' = \bigl(\begin{smallmatrix}0&1\\-1&0
\end{smallmatrix}\bigr)X + \bigl(\begin{smallmatrix}0\\
\cos\omega t\end{smallmatrix}\bigr)$~; Duhamel avec la
\omterm{thm:b3:ode:linearstructure}{résolvante} de rotation donne les solutions particulières~:
pour $\omega \neq 1$, $x_p(t) = \frac{\cos(\omega t)}{1 -
\omega^2}$ (vérifier directement)~; pour $\omega = 1$
l'intégrale $\int_0^t\sin(t - s)\cos s\,\dd s = \frac
t2\sin t$ produit la croissance \emph{séculaire} $x_p =
\frac t2\sin t$~: résonance --- le forçage pompe de l'énergie
à la fréquence naturelle et l'amplitude croît linéairement.
\end{solution}

\begin{solution}{exo:b3:ode:5}
(a) $w' = x_1x_2'' - x_1''x_2 = x_1(-px_2' - qx_2) - (-px_1'
- qx_1)x_2 = -p\,w$~: $w(t) = w(t_0)\eu^{-\int_{t_0}^tp}$,
jamais nulle ou identiquement nulle. Si $w \neq 0$, les
vecteurs $(x_i, x_i')(t_0)$ sont indépendants dans $\R^2$, et
comme l'espace des solutions est de dimension~$2$
(\cref{thm:b3:ode:linearstructure} pour le système), $(x_1,
x_2)$ est une \omterm{def:b3:topology:basis}{base}.

(b) Avec $u = \int\frac{\eu^{-\int p}}{x_1^2}$~: $x_2 =
x_1u$, $x_2' = x_1'u + \frac{\eu^{-\int p}}{x_1}$, et
\[
x_2'' + px_2' + qx_2
= u\,(x_1'' + px_1' + qx_1) +
\Bigl(-\,p\frac{\eu^{-\int p}}{x_1} +
p\frac{\eu^{-\int p}}{x_1}\Bigr) = 0
\]
(les termes croisés s'annulent exactement~; développer avec
soin). Pour $t^2x'' - 2x = 0$, c'est-à-dire $x'' -
\frac2{t^2}x = 0$ ($p = 0$) avec $x_1 = t^2$~: $u = \int
t^{-4} = -\frac1{3t^3}$, donc $x_2 = -\frac1{3t}$~: solution
générale $at^2 + \frac bt$.
\end{solution}

\begin{solution}{exo:b3:ode:6}
Équilibres $0, 1$~; $f'(x) = 1 - 2x$~: $f'(0) = 1 > 0$
(instable --- les solutions voisines s'éloignent, comme le
montre la forme explicite), $f'(1) = -1 < 0$~:
asymptotiquement \omterm{def:b3:ode:stability}{stable} (\cref{thm:b3:ode:linearization} en
dimension~$1$). Pour $x(0) \in \intoo01$~: $f > 0$ là, donc
tant que la solution reste dans $\intoo01$ elle croît~; elle
ne peut jamais atteindre $0$ ni $1$ (unicité~: ce sont des
trajectoires), donc elle y reste, est bornée --- d'où
globale --- et croît vers une limite $L \in
\intoc{x(0)}1$. Si $f(L) \neq 0$, alors $x' \geq c > 0$ près
de la limite, forçant $x$ au-delà de $L$~: donc $f(L) = 0$,
$L = 1$~; symétriquement $x \to 0$ en $-\infty$. Explicitement
$x(t) = \frac1{1 + C\eu^{-t}}$ confirme tout. En général~:
si une solution autonome scalaire avait $x'(t_0) = 0$, alors
$x(t_0)$ est un équilibre et l'unicité force $x$ constante~;
sinon $f(x(t))$ garde un signe fixe (elle ne s'annule jamais,
et $t \mapsto f(x(t))$ est \omterm{def:b3:topology:continuity}{continue})~: $x$ est strictement
monotone --- donc une solution périodique non constante est
impossible.
\end{solution}

\begin{solution}{exo:b3:ode:7}
(a) $\frac{\dd}{\dd t}H(x,y) = H_xx' + H_yy' = H_xH_y +
H_y(-H_x) = 0$.
(b) Les ensembles de niveau de $H = \frac{y^2}2 +
\frac{x^4}4$ sont compacts ($H$ coercive), donc les solutions
sont piégées dans des compacts~: globales et bornées
(\cref{thm:b3:ode:maximal}). Stabilité de $(0,0)$~: $V = H$
est définie positive ($H = 0$ seulement à l'origine) avec
$\dot V = 0$~: \cref{thm:b3:ode:lyapunov}. La linéarisation
$x' = y$, $y' = 0$ a la matrice nilpotente non
diagonalisable de valeur propre $0$~:
le~\cref{thm:b3:ode:linearization} est muet (son hypothèse
$\operatorname{Re}\lambda < 0$ échoue), et de fait le système
linéarisé est instable ($y_0 \ne 0$ dérive) tandis que le non
linéaire est \omterm{def:b3:ode:stability}{stable}~: la linéarisation en un équilibre non
hyperbolique ne prouve rien.
\end{solution}

\begin{solution}{exo:b3:ode:8}
(a) $\dot V = y\,y' + \sin x\cdot x' = y(-\sin x - cy) +
y\sin x = -cy^2 \leq 0$, et $V = \frac{y^2}2 + (1 - \cos x)$
est définie positive sur $\{\abs x < 2\pi\}$ autour de
l'origine~: \omterm{def:b3:ode:stability}{stable} (\cref{thm:b3:ode:lyapunov}).
(b) La matrice linéarisée en $(0,0)$ est
$\bigl(\begin{smallmatrix}0 & 1\\ -1 &
-c\end{smallmatrix}\bigr)$, de polynôme caractéristique
$\lambda^2 + c\lambda + 1$~: racines $\frac{-c \pm \sqrt{c^2
- 4}}2$ --- toutes deux réelles négatives si $c \geq 2$,
complexes de partie réelle $-\frac c2 < 0$ si $0 < c < 2$.
Dans tous les cas $\operatorname{Re}\lambda < 0$~:
le~\cref{thm:b3:ode:linearization} donne la stabilité
asymptotique (malgré le $\dot V$ dégénéré).
(c) En $(\pi, 0)$~: $\sin(\pi + u) = -\sin u$, linéarisation
$\bigl(\begin{smallmatrix}0&1\\1&-c\end{smallmatrix}\bigr)$,
caractéristique $\lambda^2 + c\lambda - 1$~: racines de signes
opposés ($\lambda_+\lambda_- = -1$). Le long du vecteur
propre instable, le système linéaire a la solution
explicitement s'échappant $\eu^{\lambda_+t}v_+$ avec
$\lambda_+ > 0$~: le pendule inversé est instable pour tout
amortissement.
\end{solution}

\begin{solution}{exo:b3:ode:9}
Pour $\alpha \in \intoo01$~: outre $x \equiv 0$, chaque
\[
x_c(t) = \begin{cases}0 & t \leq c,\\
\bigl((1-\alpha)(t - c)\bigr)^{1/(1-\alpha)} & t \geq c,
\end{cases}
\]
est $\mathcal C^1$ et résout l'équation (l'exposant
$\frac1{1-\alpha} > 1$ fait s'annuler la dérivée en $c$)~: un
continuum de solutions passant par $(0,0)$. Pour $\alpha =
1$~: $x \mapsto \abs x$ est globalement $1$-lipschitzienne
($\abs{\abs a - \abs b} \leq \abs{a - b}$), donc
le~\cref{thm:b3:ode:cauchylipschitz} s'applique et la seule
solution passant par $0$ est $x \equiv 0$. La frontière est
exactement la condition de Lipschitz locale en $0$~:
$\abs x^\alpha$ a des quotients de différences non bornés là
pour $\alpha < 1$.
\end{solution}

\begin{solution}{exo:b3:ode:10}
(a) Supposons $\norm{x(t_2)} > R$ pour un $t_2 > 0$ avec
$\norm{x(0)} \leq R$, et soit $t_1 = \sup\{t \leq t_2 :
\norm{x(t)} \leq R\}$~: alors $\norm{x(t_1)} = R$ et
$\norm{x(t)} > R$ sur $\intoc{t_1}{t_2}$. Sur cet intervalle
$g(t) = \norm{x(t)}^2$ a $g'(t) = 2\langle x(t),
F(x(t))\rangle \leq 0$ (l'hypothèse s'applique~:
$\norm{x(t)} \geq R$), donc $g(t_2) \leq g(t_1) = R^2$~:
contradiction. La boule est positivement invariante.
(b) Une solution piégée dans la boule compacte ne peut avoir
$T_+ < \infty$ (\cref{thm:b3:ode:maximal}(2))~: globale vers
l'avant. Système de gradient~: $\frac{\dd}{\dd t}G(x(t)) =
\langle\nabla G, -\nabla G\rangle = -\norm{\nabla G(x(t))}^2
\leq 0$~: $G$ décroît, donc la solution reste dans $\{G \leq
G(x_0)\}$, qui est borné (coercivité~: hors d'une grande
boule, $G > G(x_0)$) et fermé~: compact. L'échappement est
impossible~: toute solution d'un système de gradient coercif
est globale vers l'avant, glissant indéfiniment vers le bas.
\end{solution}

\begin{solution}{exo:b3:ode:11}
(a) Lemme de comparaison~: soit $w = x - y$ sur l'intervalle
commun~; $w(0) \geq 0$ et $w' = x' - y' \geq F(x) - F(y) =
c(t)w$ avec $c(t) = \frac{F(x) - F(y)}{x - y}$ bornée sur les
intervalles de temps compacts ($F$ \omterm{def:b3:ode:lipschitz}{localement
lipschitzienne})~; alors $(w\eu^{-\int c})' \geq 0$, donc $w
\geq 0$ partout. Avec $F(x) = x^2$~: $y(t) = \frac1{1 - t}$
explose en $1$, et $x \geq y$ tant que $x$ vit~; si $T_+ >
1$, alors $x$ serait finie en $t = 1$ tout en dominant $y \to
\infty$~: absurde. $T_+ \leq 1$.

(b) La comparaison inversée (même lemme, rôles échangés)~:
sur $\intcc01\cap\intco0{T_+}$, $t^2 \leq 1$ donne $x' \leq
x^2 + 1$, tandis que $z(t) = \tan(t + \frac\pi4)$ vérifie
$z' = z^2 + 1$, $z(0) = 1 = x(0)$~: d'où $x \leq z$ tant que
les deux sont définies. Comme $z$ est finie sur
$\intco0{\frac\pi4}$, $x$ ne peut exploser avant
$\frac\pi4$~: $T_+ \geq \frac\pi4$.

(c) Ensemble~: $\frac\pi4 \leq T_+ \leq 1$ (numériquement
$T_+ \approx 0.96$). Morale~: pour $x' = F(t, x)$ avec $F$
superlinéaire en $x$, les solutions explosent en temps fini
dès que $\int^\infty\frac{\dd s}{F(s)} < \infty$ (la solution
de comparaison atteint l'infini en ce temps fini)~; la
croissance linéaire, où l'intégrale diverge, force
l'existence globale (\cref{exo:b3:ode:3}). C'est la même
intégrale d'Osgood que dans \cref{pb:b3:complete:1}, qui
gouverne maintenant l'échappement à l'infini plutôt que
l'échappement depuis zéro.
\end{solution}

\begin{solution}{exo:b3:ode:12}
(a) $W' = uv'' - u''v = u(-q_2v) - (-q_1u)v = (q_1 -
q_2)\,uv$.

(b) Soient $a < b$ zéros consécutifs de $u$~; normalisons $u >
0$ sur $\intoo ab$ (donc $u'(a) > 0$, $u'(b) < 0$ --- non
nuls par unicité, car $u(a) = u'(a) = 0$ forcerait $u \equiv
0$). Supposons que $v$ n'a pas de zéro dans $\intoo ab$~;
normalisons $v > 0$ là (d'où $v(a), v(b) \geq 0$ par
continuité). Intégrons (a)~:
\[
W(b) - W(a) = \int_a^b(q_1 - q_2)\,uv \;\leq\; 0 .
\]
Mais $W(a) = u(a)v'(a) - u'(a)v(a) = -u'(a)v(a) \leq 0$ et
$W(b) = -u'(b)v(b) \geq 0$~: donc $W(b) - W(a) \geq 0$.
Égalité partout~: $\int(q_1 - q_2)uv = 0$ avec $uv > 0$ sur
l'intervalle \omterm{def:b3:topology:topology}{ouvert} force $q_1 = q_2$ là~; et $W(a) = W(b) =
0$ force $v(a) = v(b) = 0$~; alors $W \equiv 0$ sur
$\intcc ab$ (sa dérivée s'annule), c'est-à-dire
$(v/u)' = -W/u^2 = 0$ sur $\intoo ab$~: $v \propto u$.

(c) Prendre $q_1 = m$ et $u = \sin(\sqrt m(t - t_0))$, dont
les zéros consécutifs sont distants de $\pi/\sqrt m$, et $q_2
= q \geq m$~: par (b), toute solution $v$ de $v'' + qv = 0$
s'annule dans chaque intervalle \omterm{def:b3:topology:topology}{ouvert} de longueur
$\pi/\sqrt m$ (dans l'alternative dégénérée $q \equiv m$, $v
\propto u$ s'annule aussi). Si au contraire $q \leq 0$~:
appliquer (b) avec $q_1 = q$, $u = v$, et $q_2 = 0$ avec la
solution sans zéro $\mathbf 1$ de $v'' = 0$. Si $v$ avait
deux zéros consécutifs, (b) forcerait soit un zéro de
$\mathbf 1$ entre eux, soit le cas dégénéré $\mathbf 1
\propto v$ --- tous deux absurdes~: $v$ s'annule au plus une
fois. Tests~: pour $q = 1$, $\sin t$ s'annule tous les $\pi =
\pi/\sqrt1$, comme prédit~; pour $q = -1$, $\sinh t$
s'annule exactement une fois et $\eu^t$ jamais --- au plus un
zéro, comme prédit.
\end{solution}

\begin{solution}{pb:b3:ode:1}
\textbf{1.} $\dot E = yy' + \sin x\cdot x' = -y\sin x +
y\sin x = 0$~: l'énergie est une intégrale première. Sur une
\omterm{thm:b3:ode:maximal}{solution maximale}, $y^2 = 2(E_0 + \cos x) \leq 2(E_0 + 1)$~:
$y$ est bornée~; alors $\abs{x(t)} \leq \abs{x(0)} +
t\sup\abs y$ croît au plus linéairement~: sur tout intervalle
de temps fini la trajectoire reste dans un compact de
$\R^2$, donc le~\cref{thm:b3:ode:maximal}(2) force $T_\pm =
\pm\infty$. Équilibres $(k\pi, 0)$~; linéarisation
$\bigl(\begin{smallmatrix}0&1\\
\mp1&0\end{smallmatrix}\bigr)$ avec $-\cos(k\pi) = \mp1$~:
valeurs propres $\pm\iu$ pour $k$ pair (type \omterm{ex:b3:groups:actions}{centre}, non
concluant en soi) et $\pm1$ pour $k$ impair (col).

\textbf{2.} $E$ constante le long des solutions confine
chaque trajectoire à un ensemble de niveau $\{y^2 = 2(E_0 +
\cos x)\}$. Pour $E_0 = -1$~: seulement les points $(2k\pi,
0)$. Pour $-1 < E_0 < 1$~: en écrivant $E_0 = -\cos a$,
l'ensemble est une union disjointe de courbes fermées $y =
\pm\sqrt{2(\cos x - \cos a)}$ sur $x \in [2k\pi - a, 2k\pi +
a]$, une autour de chaque équilibre \omterm{def:b3:ode:stability}{stable}. Pour $E_0 = 1$~:
les courbes $y = \pm2\cos\frac x2$ joignant les cols
consécutifs --- la séparatrice --- avec les cols
eux-mêmes. Pour $E_0 > 1$~: deux graphes $y =
\pm\sqrt{2(E_0 + \cos x)}$, définis pour tout $x$, ne
touchant jamais $y = 0$.

\textbf{3.} $V = E + 1 = \frac{y^2}2 + (1 - \cos x)$
s'annule en $(0,0)$, est positive sur un \omterm{def:b3:topology:topology}{voisinage} épointé
($\abs x < 2\pi$), et $\dot V = 0 \leq 0$~:
le~\cref{thm:b3:ode:lyapunov} donne la stabilité. Pas
asymptotique~: la solution passant par $(a, 0)$ ($0 < a$
petit) reste sur la courbe de niveau $E = -\cos a$, dont la
distance à l'origine est positive (la courbe ne rencontre
l'axe des $x$ qu'en $\pm a$)~: $\varphi_t(a, 0) \not\to
(0,0)$.

\textbf{4.} Sur la courbe de niveau $C_a$~: pas d'équilibre
($y = 0$ force $x = \pm a$ avec $\sin(\pm a) \neq 0$ pour $0
< a < \pi$), donc la vitesse $\norm{(y, -\sin x)}$ a un
minimum strictement positif $m$ sur le compact $C_a$.
Suivons la solution depuis $(a, 0)$~: dans le demi-plan
inférieur $x' = y < 0$, donc $x$ décroît de $a$ à $-a$ en
temps fini (les intégrales de quart/demi-période
convergent~: analyse de la question~5), atteignant $(-a,
0)$~; par la symétrie $(x, y) \mapsto (x, -y)$, $t \mapsto
-t$ de l'équation, le demi-plan supérieur est parcouru en
retour en le même temps $\frac T2$~: la solution revient en
$(a, 0)$ au temps $T$. L'unicité
(\cref{cor:b3:ode:uniqueness}) propage alors~: $x(t + T) =
x(t)$ pour tout $t$~: périodique.

\textbf{5.} Sur la branche où $y > 0$~: $y = \sqrt{2(\cos x
- \cos a)}$ et $\dd t = \frac{\dd x}{y}$~; intégrer $x$ de
$-a$ à $a$ donne la demi-période, et la symétrie $x \mapsto
-x$ divise encore l'intégrale par deux~:
\[
T(a) = 4\int_0^a\frac{\dd x}{\sqrt{2(\cos x - \cos a)}} .
\]
Convergence en $x = a^-$~: $\cos x - \cos a = \sin(a)(a - x)
+ O((a-x)^2)$ avec $\sin a > 0$~: l'intégrande se comporte
comme $\bigl(2\sin a\,(a - x)\bigr)^{-1/2}$, \omterm{def:b3:lebesgue:l1}{intégrable}.

\textbf{6.} Avec $\sin\frac x2 = k\sin\varphi$, $k =
\sin\frac a2$~: $\cos x - \cos a = 2(k^2 - \sin^2\frac x2)
= 2k^2\cos^2\varphi$ et $\frac12\cos\frac x2\,\dd x =
k\cos\varphi\,\dd\varphi$, donc
\[
T(a) = 4\int_0^{\pi/2}
\frac{\dd\varphi}{\sqrt{1 - k^2\sin^2\varphi}} .
\]
Quand $a \to 0^+$, $k \to 0$~: pour $k \leq k_0 < 1$
l'intégrande est dominée par $(1 -
k_0^2\sin^2\varphi)^{-1/2}$, \omterm{def:b3:topology:continuity}{continue} sur $\intcc0{\pi/2}$~:
la convergence dominée donne $T \to 4\cdot\frac\pi2 =
2\pi$. Développement de $(1 - u)^{-1/2} = 1 + \frac u2 +
\frac{3u^2}8 + \cdots$ avec $u = k^2\sin^2\varphi$
(convergence normale pour $k < 1$) et
$\int_0^{\pi/2}\sin^2 = \frac\pi4$~:
\[
T(a) = 2\pi\Bigl(1 + \frac{k^2}{4} + O(k^4)\Bigr) :
\]
l'isochronisme ne tient qu'au premier ordre~; la période croît
avec l'amplitude.

\textbf{7.} Quand $k \uparrow 1$ les intégrandes croissent
vers $(1 - \sin^2\varphi)^{-1/2} = \frac1{\cos\varphi}$,
dont l'intégrale diverge~: par convergence monotone, $T(a)
\to +\infty$ quand $a \to \pi^-$.

\textbf{8.} Sur la branche $y = 2\cos\frac x2$ ($\abs x <
\pi$)~: $\frac{\dd x}{2\cos(x/2)} = \dd t$~; avec $u =
\frac x2$, $\int\frac{\dd u}{\cos u} =
\ln\tan\bigl(\frac u2 + \frac\pi4\bigr)$, donc $t =
\ln\tan\bigl(\frac x4 + \frac\pi4\bigr)$, c'est-à-dire
\[
x(t) = 4\arctan(\eu^t) - \pi,
\qquad y(t) = x'(t) = \frac{4\eu^t}{1 + \eu^{2t}} =
\frac{2}{\cosh t} .
\]
Vérification via l'énergie~: avec $\theta = \arctan\eu^t$,
$\sin2\theta = \frac1{\cosh t}$, donc $\cos x = -\cos4\theta
= -1 + \frac{2}{\cosh^2t}$ et $E = \frac{y^2}2 - \cos x =
\frac2{\cosh^2t} + 1 - \frac2{\cosh^2t} = 1$~: la
trajectoire est sur la séparatrice, et dériver $y^2 = 2(1 +
\cos x)$ où $y > 0$ reproduit $y' = -\sin x$. Limites~: $x
\to \pm\pi$ et $y \to 0$ quand $t \to \pm\infty$.

\textbf{9.} L'\omterm{def:b3:groups:action}{orbite} tend vers le col $(\pi, 0)$ vers
l'avant et $(-\pi, 0)$ vers l'arrière mais n'arrive jamais~:
si elle atteignait $(\pi, 0)$ en un temps fini $t^*$, deux
\omterm{thm:b3:ode:maximal}{solutions maximales} distinctes --- la solution séparatrice et
la solution constante au col --- passeraient par le même
point $(t^*, (\pi, 0))$, contredisant
le~\cref{cor:b3:ode:uniqueness}. Les cols ne sont approchés
qu'asymptotiquement.

\textbf{10.} Pour $E_0 > 1$~: $y^2 = 2(E_0 + \cos x) \geq
2(E_0 - 1) > 0$~: $y$ garde son signe, et $\abs{x'} = \abs y
\geq \sqrt{2(E_0 - 1)}$~: $x$ est strictement monotone,
globale (question~1), avec $x(t) \to \pm\infty$. Comme $y(t)
= \pm\sqrt{2(E_0 + \cos x(t))}$ et $\cos$ est
$2\pi$-périodique, $y$ reprend sa valeur chaque fois que $x$
avance de $2\pi$~; le temps nécessaire est
\[
\tau(E_0) = \int_{x_0}^{x_0 + 2\pi}\frac{\dd
x}{\sqrt{2(E_0 + \cos x)}} =
\int_{-\pi}^{\pi}\frac{\dd x}{\sqrt{2(E_0 + \cos x)}}
\]
(substitution~; périodicité)~: $y$ est $\tau$-périodique ---
le pendule tourne avec un taux de rotation asymptotiquement
constant $2\pi/\tau \approx \sqrt{2E_0}$ pour les grandes
énergies.

\textbf{11.} La méthode, dans l'ordre~: l'\emph{énergie}
($\dot E = 0$) réduit le \omterm{ex:b3:ode:planeclassification}{flot} bidimensionnel à des courbes
de niveau unidimensionnelles~; la \emph{bornitude} de $y$
sur chaque niveau plus l'\omterm{thm:b3:ode:maximal}{échappement des compacts} donne
l'existence globale~; la \emph{compacité} des niveaux fermés
donne des bornes de vitesse et donc la périodicité~;
l'\emph{unicité} convertit le premier retour en périodicité
exacte, interdit l'arrivée en temps fini aux cols, et sépare
les types d'\omterm{def:b3:groups:action}{orbites}~; la \emph{linéarisation et Lyapunov}
classent les équilibres~; l'\emph{intégrale de période} est
analysée avec les théorèmes de convergence du
\cref{ch:b3:lebesgue}. À aucun moment n'avons-nous possédé
--- ni eu besoin --- d'une solution générale en forme
fermée~: la théorie qualitative a extrait chaque trait du
mouvement de l'équation elle-même.

\textbf{12.} Parties~: $W_n = \int_0^{\pi/2}\sin^{2n-1}\varphi
\cdot\sin\varphi\,\dd\varphi = (2n-1)\int_0^{\pi/2}
\sin^{2n-2}\varphi\cos^2\varphi\,\dd\varphi = (2n-1)(W_{n-1}
- W_n)$, donc $W_n = \frac{2n-1}{2n}W_{n-1}$~; avec $W_0 =
\frac\pi2$ et $\prod_{j=1}^n\frac{2j-1}{2j} =
\frac{(2n)!}{4^n(n!)^2}$ (scinder $(2n)! =
2^nn!\prod(2j-1)$), la récurrence donne la valeur affichée.
Série du binôme~: $(1 - u)^{-1/2} = \sum_{n\geq0}c_nu^n$ avec
$c_n = \frac{(2n)!}{4^n(n!)^2} \in \intoc01$, rayon $1$. Pour
$k \leq k_0 < 1$ et $u = k^2\sin^2\varphi$, la série
$\sum c_nk^{2n}\sin^{2n}\varphi$ converge normalement en
$\varphi$ ($c_nk^{2n} \leq k_0^{2n}$), donc l'intégration
terme à terme dans la formule de la question~6 est
légitime~:
\[
T(a) = 4\sum_{n\geq0}c_nk^{2n}W_n
= 4\cdot\frac\pi2\sum_{n\geq0}c_n^2\,k^{2n}
= 2\pi\sum_{n\geq0}
\Bigl(\frac{(2n)!}{4^n(n!)^2}\Bigr)^{\!2}k^{2n} .
\]
Avec $c_0 = 1$, $c_1 = \frac12$, $c_2 = \frac38$~: $T(a) =
2\pi\bigl(1 + \frac{k^2}4 + \frac{9k^4}{64} + O(k^6)\bigr)$,
le reste uniforme pour $k \leq k_0$ (queue dominée par une
série géométrique).

\textbf{13.} $k = \sin\frac a2 = \frac a2 - \frac{a^3}{48} +
O(a^5)$, donc
\[
k^2 = \frac{a^2}4 - \frac{a^4}{48} + O(a^6),
\qquad
k^4 = \frac{a^4}{16} + O(a^6) ,
\]
et
\[
\frac{T(a)}{2\pi} = 1 + \frac14\Bigl(\frac{a^2}4 -
\frac{a^4}{48}\Bigr) + \frac9{64}\cdot\frac{a^4}{16} +
O(a^6)
= 1 + \frac{a^2}{16} + \frac{11\,a^4}{3072} + O(a^6) ,
\]
puisque $-\frac1{192} + \frac9{1024} = \frac{-16 + 27}{3072}
= \frac{11}{3072}$.

\textbf{14.} Dans la forme elliptique de la question~6, $a
\mapsto k = \sin\frac a2$ est une bijection \omterm{def:b3:topology:continuity}{continue}
strictement croissante de $\intoo0\pi$ sur $\intoo01$, et
pour chaque $\varphi \in \intoc0{\pi/2}$ l'intégrande $(1 -
k^2\sin^2\varphi)^{-1/2}$ est strictement croissante en
$k$~: $T$ est strictement croissante. Continuité~: sur $k
\leq k_0 < 1$ l'intégrande est dominée par la \omterm{def:b3:topology:continuity}{continue} $(1 -
k_0^2\sin^2\varphi)^{-1/2}$, donc la convergence dominée
s'applique le long de $k' \to k$. Avec les limites $T \to
2\pi$ quand $a \to 0^+$ (question~6) et $T \to +\infty$ quand
$a \to \pi^-$ (question~7), la stricte monotonie et le
théorème des valeurs intermédiaires font de $T$ une bijection
de $\intoo0\pi$ sur $\intoo{2\pi}{+\infty}$.

\textbf{15.} Une horloge compte les oscillations~; réglée à
amplitude nulle, elle comptabilise la période harmonique
$2\pi$ par oscillation (dans l'unité de temps du pendule).
Lancée à l'amplitude $a$, la période vraie est $T(a) =
2\pi\bigl(1 + \frac{a^2}{16} + O(a^4)\bigr)$~: l'horloge
comptabilise $2\pi$ tandis que $T(a)$ s'écoule réellement,
donc elle retarde de la fraction
\[
\frac{T(a) - 2\pi}{T(a)} = \frac{a^2}{16} + O(a^4) .
\]
Pour $a = 0.2$~rad (environ $11.5$ \omterm{def:b3:galois:extension}{degrés})~: $\frac{a^2}{16}
= \frac{0.04}{16} = \frac1{400}$, et un jour a $86400$~s~:
l'horloge perd $86400/400 = 216$ secondes --- quelque trois
minutes et demie --- par jour. D'où les deux remèdes
historiques~: imposer une amplitude constante minuscule
(l'échappement), ou cintrer la contrainte pour que la période
soit exactement indépendante de l'amplitude (les joues
cycloïdales de Huygens, 1657).

\textbf{16.} Dans $\tau(E_0) = \int_{-\pi}^{\pi}\frac{\dd
x}{\sqrt{2(E_0 + \cos x)}}$ l'intégrande est, pour chaque
$x$ fixé, strictement décroissante en $E_0$~: $\tau$ est
strictement décroissante. Quand $E_0 \downarrow 1$ les
intégrandes croissent point par point vers $\bigl(2(1 + \cos
x)\bigr)^{-1/2} = \frac1{2\abs{\cos\frac x2}}$, dont
l'intégrale sur $\intoo{-\pi}\pi$ diverge ($\cos\frac x2$
s'annule à l'ordre un en $\pm\pi$)~: la convergence monotone
donne $\tau(E_0) \to +\infty$. Quand $E_0 \to +\infty$~:
\[
\sqrt{2E_0}\,\tau(E_0) = \int_{-\pi}^{\pi}
\frac{\dd x}{\sqrt{1 + \cos x/E_0}} \longrightarrow 2\pi
\]
par convergence dominée (pour $E_0 \geq 2$ l'intégrande est
au plus $\sqrt2$). Donc $\tau \approx 2\pi/\sqrt{2E_0}$~: un
tour prend le temps d'une rotation \omterm{def:b3:modules:free}{libre} à la vitesse
$\sqrt{2E_0}$, le potentiel réduit à une ride --- en accord
avec le taux de rotation de la question~10.

\textbf{17.} Les axes portent les solutions explicites $t
\mapsto (x_0\eu^{\alpha t}, 0)$ et $t \mapsto (0,
y_0\eu^{-\gamma t})$, avec l'équilibre $(0, 0)$~: ce sont
des unions d'\omterm{def:b3:groups:action}{orbites}. Le champ est $\mathcal C^1$, donc
localement lipschitzien~; une solution partant de $Q$ qui
toucherait un axe passerait par un point de l'une de ces
\omterm{def:b3:groups:action}{orbites} et, par le~\cref{cor:b3:ode:uniqueness}, coïnciderait
avec elle --- impossible, l'une vivant sur l'axe et l'autre
non. Donc $Q$ est invariant dans les deux sens du temps.
Équilibres dans $Q$~: $x > 0$ force $\alpha - \beta y = 0$ et
$y > 0$ force $\delta x - \gamma = 0$~: le point unique
$(x_*, y_*) = \bigl(\frac\gamma\delta,
\frac\alpha\beta\bigr)$.

\textbf{18.} Le long d'une solution,
\[
\dot H = \Bigl(\delta - \frac\gamma x\Bigr)x' +
\Bigl(\beta - \frac\alpha y\Bigr)y'
= (\delta x - \gamma)(\alpha - \beta y) +
(\beta y - \alpha)(\delta x - \gamma) = 0 .
\]
$f(x) = \delta x - \gamma\ln x$ a $f'' = \gamma/x^2 > 0$,
$f'$ s'annulant seulement en $x_*$, et $f \to +\infty$ en
$0^+$ et en $+\infty$~: strictement convexe et propre,
minimum $f(x_*)$~; de même $g(y) = \beta y - \alpha\ln y$,
minimum $g(y_*)$. Donc $H \geq h_*$ avec égalité seulement en
$(x_*, y_*)$, et chaque sous-niveau $\{H \leq h\}\cap Q$ est
compact~: $f(x) \leq h - g(y_*)$ confine $x$ à un intervalle
compact de $\intoo0{+\infty}$ par propreté, de même $y$, et
l'ensemble est fermé dans $\R^2$ car $H \to +\infty$ au bord
de $Q$. Une \omterm{thm:b3:ode:maximal}{solution maximale} reste sur son ensemble de
niveau compact, donc ne peut quitter tout compact en temps
fini~: le~\cref{thm:b3:ode:maximal} la rend globale.

\textbf{19.} Fixons $h > h_*$ et posons $c = h - g(y_*) >
f(x_*)$. Comme $f$ décroît strictement de $+\infty$ à
$f(x_*)$ sur $\intoc0{x_*}$ et croît strictement de nouveau
vers $+\infty$ sur $\intco{x_*}{+\infty}$, l'équation $f(x)
= c$ a exactement deux racines $x_- < x_* < x_+$, et $\{f
\leq c\} = \intcc{x_-}{x_+}$. Pour $x \in \intoo{x_-}{x_+}$~:
$g(y) = h - f(x) > g(y_*)$ a exactement deux racines $y_-(x)
< y_* < y_+(x)$, \omterm{def:b3:topology:continuity}{continues} en $x$ (inverses des restrictions
\omterm{def:b3:topology:continuity}{continues} strictement monotones de $g$ de chaque côté de
$y_*$), avec $y_\pm(x) \to y_*$ quand $x \to x_\pm$~; en $x
= x_\pm$ l'unique solution est $y = y_*$. Donc $\{H =
h\}\cap Q$ est l'union des graphes de $y_+$ et $y_-$ sur
$\intcc{x_-}{x_+}$, recollés en $(x_\pm, y_*)$~: une courbe
fermée autour de $(x_*, y_*)$ --- l'analogue des ovales du
pendule.

\textbf{20.} Soit $C_h = \{H = h\}\cap Q$ avec $h > h_*$~:
le seul équilibre de $Q$ est hors de $C_h$, donc le champ
ne s'annule jamais dessus. Signes~: $x' = \beta x(y_* - y)$,
$y' = \delta y(x - x_*)$~: le mouvement va à droite sous la
droite $y = y_*$, vers le haut à droite de $x = x_*$, à
gauche au-dessus, vers le bas à gauche --- circulation
antihoraire. Suivons la solution depuis un point $(x_0,
y_-(x_0))$ de la branche inférieure ouverte, où $x' > 0$~: le
temps pour atteindre le coin droit $B = (x_+, y_*)$ est
\[
\int_{x_0}^{x_+}\frac{\dd x}{\beta x\,\bigl(y_* -
y_-(x)\bigr)} .
\]
Près de $x_+$, choisissons $x' \in \intoo{x_*}{x_+}$~; pour
$x \in \intcc{x'}{x_+}$, $f(x_+) - f(x) \geq f'(x')\,(x_+ -
x)$ ($f'$ croissante et positive au-delà de $x_*$), tandis
que la relation de niveau et l'inégalité de Taylor donnent
$g(y_-(x)) - g(y_*) \leq \frac12\,\bigl(\max g''\bigr)\,(y_*
- y_-(x))^2$ sur la plage compacte en $y$ de $C_h$~: d'où
$y_* - y_-(x) \geq c\,\sqrt{x_+ - x}$ avec $c > 0$, et
l'intégrande est $O\bigl((x_+ - x)^{-1/2}\bigr)$~: \omterm{def:b3:lebesgue:l1}{intégrable}
--- la convergence de la question~5, transposée. Ailleurs
sur la branche l'intégrande est \omterm{def:b3:topology:continuity}{continue}. Donc $B$ est
atteint en temps fini~; là $y' = \delta y_*(x_+ - x_*) > 0$,
l'\omterm{def:b3:groups:action}{orbite} entre dans la région $x > x_*$, $y > y_*$, grimpe
au coin supérieur $(x_*, y_+)$ par l'estimée symétrique
(rôles de $f$ et $g$ échangés), et ainsi de suite autour des
quatre arcs~: après un temps fini $T > 0$ la solution
revient à son point de départ. Par
le~\cref{cor:b3:ode:uniqueness} elle est $T$-périodique ---
l'argument de la question~4, mot pour mot.

\textbf{21.} Sur une \omterm{def:b3:groups:action}{orbite} $T$-périodique dans $Q$, $t
\mapsto \ln x(t)$ est $\mathcal C^1$ et $T$-périodique, donc
\[
0 = \int_0^T(\ln x)'\,\dd t
= \int_0^T(\alpha - \beta y)\,\dd t
= \alpha T - \beta\int_0^Ty\,\dd t ,
\]
donnant $\frac1T\int_0^Ty = \frac\alpha\beta$~; de même $0 =
\int_0^T(\ln y)' = \delta\int_0^Tx\,\dd t - \gamma T$ donne
$\frac1T\int_0^Tx = \frac\gamma\delta$. Les moyennes
temporelles sont les valeurs d'équilibre, pour toute \omterm{def:b3:groups:action}{orbite}
quelle que soit l'amplitude --- une loi de conservation que
personne n'a introduite à la main.

\textbf{22.} Avec la récolte le système est encore de forme
Lotka--Volterra, avec paramètres $\alpha - \varepsilon$,
$\beta$, $\gamma + \varepsilon$, $\delta$ (l'équilibre
intérieur persiste car $\varepsilon < \alpha$). La
question~21 appliquée au nouveau système~:
\[
\bar x = \frac{\gamma + \varepsilon}{\delta}
\ \ (\text{moyenne des proies croît}),
\qquad
\bar y = \frac{\alpha - \varepsilon}{\beta}
\ \ (\text{moyenne des prédateurs décroît}) :
\]
une récolte indifférenciée déplace l'équilibre vers les
proies. Les données de d'Ancona se lisent à l'envers~: la
guerre a réduit la pêche, $\varepsilon$ a chuté, donc la
moyenne des prédateurs $(\alpha - \varepsilon)/\beta$ a
monté et celle des proies a baissé --- une plus grande
fraction de requins dans les prises, exactement ce que les
marchés adriatiques ont enregistré. C'est le principe de
Volterra, le même mécanisme derrière les paradoxes des
pesticides~: abattre les deux niveaux trophiques profite au
niveau mangé.

\textbf{23.} La recette, les deux fois~: (i) une intégrale
première --- $E$ pour le pendule, $H$ ici, trouvée en
séparant $\dd y/\dd x$ --- effondre le plan sur des
courbes~; (ii) propreté et compacité des ensembles de niveau
donnent l'existence globale via
le~\cref{thm:b3:ode:maximal}~; (iii) la géométrie des
niveaux --- ovales, de la forme de $\cos$ là-bas et de la
stricte convexité de $f$ et $g$ ici --- se lit sur
l'intégrale, non sur le \omterm{ex:b3:ode:planeclassification}{flot}~; (iv) un champ non nul sur un
ovale compact plus des singularités de coin \omterm{def:b3:lebesgue:l1}{intégrables}
force un temps de retour fini~; (v) l'unicité
(\cref{cor:b3:ode:uniqueness}) convertit le retour en
périodicité et interdit l'arrivée en temps fini aux
équilibres~; (vi) les dividendes --- développements de la
période, lois des moyennes --- viennent des théorèmes de
convergence appliqués aux intégrales résultantes. Ni $x'' =
-\sin x$ ni Lotka--Volterra n'admettent de solution
élémentaire en forme fermée (intégrales elliptiques dans un
cas, courbes de niveau transcendantes dans l'autre), et à
aucun moment on n'en a eu besoin~: l'équation elle-même,
interrogée qualitativement, a livré le mouvement tout
entier.

\textbf{24.} Sur un niveau d'énergie $E_0 > 1$, $y^2 =
2(E_0 + \cos x) > 0$~: extrêmes de $y^2$ en $\cos x =
\pm1$, donnant les bornes énoncées, atteintes en $x \equiv
0, \pi$. Moyenne temporelle sur une période $\tau =
\tau(E_0)$~: $x$ avance de exactement $2\pi$, donc
\[
\frac1\tau\int_0^\tau y\,\dd t = \frac{x(\tau) -
x(0)}{\tau} = \frac{2\pi}\tau .
\]
Le rapport des vitesses extrêmes est $\sqrt{\frac{E_0 +
1}{E_0 - 1}} = 1 + O(E_0^{-1}) \to 1$~: à haute énergie la
ride $\pm1$ du potentiel est négligeable face à $E_0$, et le
pendule tourne presque uniformément --- la planche à laver
s'aplatit.

\textbf{25.} Sur $E_0 \geq 1 + \delta$, l'intégrande de
$\tau(E_0) = \int_{-\pi}^{\pi}\frac{\dd x}{\sqrt{2(E_0 +
\cos x)}}$ est dominée par $(2\delta)^{-1/2}$ et sa dérivée
en $E_0$
\[
\partial_{E_0}\frac1{\sqrt{2(E_0 + \cos x)}}
= -\frac{1}{\bigl(2(E_0 + \cos x)\bigr)^{3/2}}
\]
par $(2\delta)^{-3/2}$, toutes deux \omterm{def:b3:lebesgue:l1}{intégrables} sur le
compact $\intcc{-\pi}\pi$~: la dérivation sous le signe
d'intégration (\cref{thm:b3:lebesgue:paramdiff}) s'applique
et donne $\tau'(E_0) < 0$ (l'intégrande est strictement
négative)~: strictement décroissante, $\mathcal C^1$.
Limites~: quand $E_0 \to \infty$, $\tau \leq
\frac{2\pi}{\sqrt{2(E_0 - 1)}} \to 0$~; quand $E_0 \to 1^+$~:
écrire $E_0 + \cos x = (E_0 - 1) + 2\cos^2\frac x2$~;
l'intégrande croît quand $E_0$ décroît, donc par
convergence monotone
\[
\tau(E_0) \;\nearrow\;
\int_{-\pi}^{\pi}\frac{\dd x}{2\,\abs{\cos\frac x2}}
= +\infty,
\]
l'intégrale limite divergeant en $x = \pm\pi$ (là
$\abs{\cos\frac x2} \sim \frac{\abs{x \mp \pi}}2$, un
$\frac1{\abs\cdot}$ non \omterm{def:b3:lebesgue:l1}{intégrable})~: la période explose en
approchant la séparatrice, en accord avec $T(a) \to \infty$
de la Partie~II du côté libration. L'axe d'énergie se
lit~: repos en $E_0 = -1$~; librations, $2\pi \nearrow
\infty$, sur $\intoo{-1}1$~; la séparatrice infiniment lente
en $E_0 = 1$~; rotations, $\infty \searrow 0$, au-delà. Une
intégrale, toute la vie du pendule.
\end{solution}
