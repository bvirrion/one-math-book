\section*{Chapter \ref{ch:b3:forms} --- Differential Forms and
Stokes' Theorem}

\begin{solution}{exo:b3:forms:1}
Expanding and killing repeated factors:
\[
\omega\wedge\eta = (x\,\dd y - z\,\dd x)\wedge(\dd x +
y\,\dd z)
= -x\,\dd x\wedge\dd y + xy\,\dd y\wedge\dd z +
yz\,\dd z\wedge\dd x
\]
(using $\dd y\wedge\dd x = -\dd x\wedge\dd y$ and $-zy\,\dd
x\wedge\dd z = yz\,\dd z\wedge\dd x$). Next $\dd\omega =
\dd x\wedge\dd y - \dd z\wedge\dd x = \dd x\wedge\dd y +
\dd x\wedge\dd z$ and $\dd\eta = \dd y\wedge\dd z$. Finally
\[
\dd(\omega\wedge\eta) = y\,\dd x\wedge\dd y\wedge\dd z +
z\,\dd y\wedge\dd z\wedge\dd x = (y + z)\,\dd x\wedge\dd
y\wedge\dd z
\]
(the first term of $\omega\wedge\eta$ contributes $\dd(-x)
\wedge\dd x\wedge\dd y = 0$; cyclic permutations of three
factors are even). Leibniz check: $\dd\omega\wedge\eta =
(\dd x\wedge\dd y + \dd x\wedge\dd z)\wedge(\dd x + y\,\dd
z) = y\,\dd x\wedge\dd y\wedge\dd z$, and
$(-1)^1\omega\wedge\dd\eta = -(x\,\dd y - z\,\dd
x)\wedge\dd y\wedge\dd z = z\,\dd x\wedge\dd y\wedge\dd z$;
the sum matches.
\end{solution}

\begin{solution}{exo:b3:forms:2}
$\dd f = \sum_i\partial_if\,\dd x_i$ has the coefficients of
$\nabla f$. For the work form,
\[
\dd\omega_F = (\partial_yF_3 - \partial_zF_2)\,\dd y\wedge\dd
z + (\partial_zF_1 - \partial_xF_3)\,\dd z\wedge\dd x +
(\partial_xF_2 - \partial_yF_1)\,\dd x\wedge\dd y =
\sigma_{\operatorname{curl}F},
\]
the flux form of the curl (collect the six terms of
$\sum_i\dd F_i\wedge\dd x_i$). For the flux form,
$\dd\sigma_F = (\partial_xF_1 + \partial_yF_2 +
\partial_zF_3)\,\dd x\wedge\dd y\wedge\dd z$: the
divergence. Then $\dd^2f = 0$ reads
$\sigma_{\operatorname{curl}\nabla f} = 0$, i.e.\
$\operatorname{curl}\nabla f = 0$, and $\dd^2\omega_F = 0$
reads $(\operatorname{div}\operatorname{curl}F)\,\dd
x\wedge\dd y\wedge\dd z = 0$: the two vector identities are
one identity, $\dd^2 = 0$, read in two degrees.
\end{solution}

\begin{solution}{exo:b3:forms:3}
(a) Closedness is the symmetry of the cross-partials:
$\partial_y(2xy + z^2) = 2x = \partial_x(x^2)$,
$\partial_z(2xy + z^2) = 2z = \partial_x(2xz)$,
$\partial_z(x^2) = 0 = \partial_y(2xz)$. The domain $\R^3$
is star-shaped: \omterm{def:b3:forms:closedexact}{exact} (\cref{thm:b3:forms:poincare}), with
primitive $f = x^2y + xz^2$ (check $\dd f$).
(b) $\dfrac{x\,\dd x + y\,\dd y}{x^2 + y^2} =
\tfrac12\,\dd\log(x^2 + y^2)$: \omterm{def:b3:forms:closedexact}{exact} on all of
$\R^2\setminus\{0\}$ (hence \omterm{def:b3:forms:closedexact}{closed}) --- the radial cousin of
the angular form is harmless.
(c) On $\{x > 0\}$ the form is $\omega_\theta$, \omterm{def:b3:forms:closedexact}{closed}
(\cref{exo:b3:forms:4}); the half-plane is convex, so it is
\omterm{def:b3:forms:closedexact}{exact} there, and indeed $f = \arctan(y/x)$ satisfies $\dd f
= \frac{-y\,\dd x + x\,\dd y}{x^2 + y^2}$. \omterm{def:b3:forms:closedexact}{Exact} on the
half-plane, \omterm{def:b3:forms:closedexact}{non-exact} on the punctured plane: the
obstruction lives in the hole, not in the formula.
\end{solution}

\begin{solution}{exo:b3:forms:4}
Closedness: with $\rho^2 = x^2 + y^2$,
\[
\partial_x\Bigl(\frac{x}{\rho^2}\Bigr) = \frac{\rho^2 -
2x^2}{\rho^4} = \frac{y^2 - x^2}{\rho^4} =
\partial_y\Bigl(\frac{-y}{\rho^2}\Bigr),
\]
so $\dd\omega_\theta = \bigl(\partial_x(x/\rho^2) -
\partial_y(-y/\rho^2)\bigr)\,\dd x\wedge\dd y = 0$. On
$\gamma(t) = (r\cos2\pi t, r\sin2\pi t)$:
\[
\gamma^*\omega_\theta = \frac{2\pi r^2(\cos^22\pi t +
\sin^22\pi t)}{r^2}\,\dd t = 2\pi\,\dd t,
\qquad\text{so}\qquad
\int_\gamma\omega_\theta = 2\pi .
\] If $\omega_\theta$ were
\omterm{def:b3:forms:closedexact}{exact} this integral would vanish
(\cref{prop:b3:forms:exactloop}): it is not \omterm{def:b3:forms:closedexact}{exact}. Were
$\R^2\setminus\{0\}$ star-shaped with respect to some $p$,
the Poincar\'e lemma (translated to $p$) would make every
\omterm{def:b3:forms:closedexact}{closed form} \omterm{def:b3:forms:closedexact}{exact} --- contradiction. Directly: for any $p
\neq 0$, the segment from $p$ to the point $-p$ of the
domain passes through $0$: star-shapedness fails at every
point.
\end{solution}

\begin{solution}{exo:b3:forms:5}
For vectors $v_1, \dots, v_{n-1}$, expanding the determinant
along its first column gives
\[
\det(\nu, v_1, \dots, v_{n-1}) =
\sum_{i=1}^n(-1)^{i-1}\nu_i\,
\det\bigl(\text{rows} \neq i \text{ of } (v_1, \dots,
v_{n-1})\bigr)
= \sigma_\nu(v_1, \dots, v_{n-1}),
\]
since $(\dd x_1\wedge\dots\wedge\widehat{\dd
x_i}\wedge\dots\wedge\dd x_n)(v_1, \dots, v_{n-1})$ is
exactly the $i$-th deleted minor
(\cref{def:b3:forms:wedge}). Applying this to $v_j =
\partial_j\gamma$: $(\gamma^*\sigma_\nu)(e_1, \dots,
e_{n-1}) = \det A$ with $A = (\nu\ \partial_1\gamma\ \cdots\
\partial_{n-1}\gamma)$. Now ${}^t\!AA$ is block diagonal:
$\langle\nu, \nu\rangle = 1$ and $\langle\nu,
\partial_j\gamma\rangle = 0$ ($\nu$ normal, the
$\partial_j\gamma$ tangent), so $(\det A)^2 =
\det({}^t\!AA) = \det G$; and $\det A > 0$ for a direct
parametrization (that is what the $\nu$-orientation means):
$\gamma^*\sigma_\nu = \sqrt{\det G}\,\dd
u_1\wedge\dots\wedge\dd u_{n-1}$, the Gram area element. For
$S^2$, $\nu(x) = x$ and the spherical parametrization
$\gamma(\theta, \varphi) = (\sin\varphi\cos\theta,
\sin\varphi\sin\theta, \cos\varphi)$ on
$(0,2\pi)\times(0,\pi)$ (direct; it misses one meridian, a
set that carries no area): $\det G = \sin^2\varphi$, so
$\int_{S^2}\sigma_\nu =
\int_0^{2\pi}\!\!\int_0^\pi\sin\varphi\,\dd\varphi\,
\dd\theta = 4\pi$.
\end{solution}

\begin{solution}{exo:b3:forms:6}
The given $\omega$ is $\sigma_\nu$ for $\nu(x) = x$ on
$S^2$, so (a) is the computation just done:
$\int_{S^2}\omega = 4\pi$. (b) $\dd\omega = 3\,\dd
x\wedge\dd y\wedge\dd z$, and Stokes on the unit ball gives
$\int_{S^2}\omega = 3\operatorname{vol}(B^3)$: hence
$\operatorname{vol}(B^3) = \frac{4\pi}3$. In general, the
form $\sigma = \sum_i(-1)^{i-1}x_i\,\dd
x_1\wedge\dots\wedge\widehat{\dd x_i}\wedge\dots\wedge\dd
x_n$ restricts on $S^{n-1}$ to the area form ($\nu = x$ in
\cref{exo:b3:forms:5}), $\dd\sigma = n\,\dd
x_1\wedge\dots\wedge\dd x_n$, and Stokes yields
$\operatorname{area}(S^{n-1}) = n\operatorname{vol}(B^n)$
--- consistent with the Gamma-function formulas of
\cref{thm:b3:product:ballvolume}.
\end{solution}

\begin{solution}{exo:b3:forms:7}
(a) Stokes applied to the flux form $\sigma_F$ on the
\omterm{def:b3:topology:compact}{compact} domain $\Omega$:
$\int_\Omega\operatorname{div}F\,\dd x =
\int_{\partial\Omega}\sigma_F$ (\cref{exo:b3:forms:2} for
the \omterm{def:b3:topology:interior}{interior} side). Identify the \omterm{def:b3:forms:boundary}{boundary} integrand: for
tangent $v_1, v_2$ at a point of $\partial\Omega$, the
first-column expansion of \cref{exo:b3:forms:5} gives
$\sigma_F(v_1, v_2) = \det(F, v_1, v_2)$; writing $F =
\langle F, \nu\rangle\nu + T$ with $T$ tangent, the column
$T$ is a combination of $v_1, v_2$'s plane, so $\det(T,
v_1, v_2) = 0$ and
$\sigma_F\vert_{\partial\Omega} = \langle F,
\nu\rangle\,\sigma_\nu = \langle F, \nu\rangle\,\dd S$: the
divergence theorem, with $\nu$ the outward normal
(outward-normal-first is exactly the induced \omterm{def:b3:forms:orientation}{orientation}).
(b) Stokes applied to $\omega_F$ on the
surface-with-boundary $S$: $\dd\omega_F =
\sigma_{\operatorname{curl}F}$ restricts to
$\langle\operatorname{curl}F, \nu\rangle\,\dd S$ by the same
identification, while on the \omterm{def:b3:forms:boundary}{boundary} curve
$\gamma^*\omega_F = \langle F\circ\gamma,
\gamma'\rangle\,\dd t$, i.e.\ $\oint\langle F,
\tau\rangle\,\dd\ell$. Upper half-sphere with outward
(radial) $\nu$: at the equator point $p = (1,0,0)$ the
outward-within-the-surface vector is $-e_3$; completing it
to positive frames shows the equator is traversed
counterclockwise seen from above ($+e_3$): the right-hand
rule, same convention on both sides of the identity.
\end{solution}

\begin{solution}{exo:b3:forms:8}
Apply the divergence theorem (\cref{exo:b3:forms:7}(a),
whose proof is dimension-free) to $F = u\nabla v$:
$\operatorname{div}(u\nabla v) = u\,\Delta v + \langle\nabla
u, \nabla v\rangle$ and $\langle F, \nu\rangle =
u\,\partial_\nu v$: the first identity. Swapping $u, v$ and
subtracting cancels the symmetric term: the second. If
$\Delta u = 0$ on $\Omega$ and $u = 0$ on $\partial\Omega$:
the first identity with $v = u$ gives
$\int_\Omega\norm{\nabla u}^2 = 0$, so $\nabla u \equiv 0$
and $u$ is constant on each component; every component's
\omterm{def:b3:topology:interior}{closure} meets $\partial\Omega$ (boundedness), where $u = 0$:
$u \equiv 0$. Two \omterm{prop:b3:conformal:harmonicholo}{harmonic functions} with equal \omterm{def:b3:forms:boundary}{boundary}
values differ by such a $u$: they coincide --- uniqueness
for the Dirichlet problem, complementing the existence
theory on the disc of \cref{ch:b3:conformal}.
\end{solution}

\begin{solution}{exo:b3:forms:9}
By \cref{thm:b3:forms:pullback}(c),
$\varphi^*(f\,\dd y_1\wedge\dots\wedge\dd y_n) =
(f\circ\varphi)\,\det(D\varphi)\,\dd
x_1\wedge\dots\wedge\dd x_n$, so the \omterm{def:b3:forms:pullback}{pullback} identity reads
\[
\int_U(f\circ\varphi)\,\det(D\varphi)\,\dd x = \int_Vf\,\dd
y .
\]
Since $\det D\varphi > 0$ everywhere, $\det D\varphi =
\abs{\det D\varphi}$, and this is verbatim the
change-of-variables formula
(\cref{thm:b3:product:changeofvar}) for \omterm{def:b3:topology:continuity}{continuous}
compactly supported integrands: each statement is the
other. The absolute value went into the \emph{hypothesis}:
\omterm{def:b3:forms:orientation}{orientation}. For orientation-reversing $\varphi$ the form
identity acquires a global minus sign (forms feel
\omterm{def:b3:forms:orientation}{orientation}), while the \omterm{def:b3:measure:measure}{measure} formula keeps $\abs{\det}$
(\omterm{def:b3:measure:measure}{measures} do not): two bookkeepings of one Jacobian.
\end{solution}

\begin{solution}{exo:b3:forms:10}
The shell $A = \{r_1 \leq \norm x \leq r_2\}$ is a \omterm{def:b3:topology:compact}{compact}
$3$-submanifold with \omterm{def:b3:forms:boundary}{boundary} $S_{r_2}\cup S_{r_1}$; the
induced \omterm{def:b3:forms:orientation}{orientations} are the usual sphere \omterm{def:b3:forms:orientation}{orientation} on
$S_{r_2}$ (outward from $A$ = away from $0$) and the
\emph{opposite} on $S_{r_1}$ (outward from $A$ = toward
$0$). Stokes with $\dd\omega = 0$:
\[
0 = \int_A\dd\omega = \int_{S_{r_2}}\omega -
\int_{S_{r_1}}\omega .
\]
Solid-angle form: with $\rho = \norm x$ and $\sigma =
x\,\dd y\wedge\dd z + y\,\dd z\wedge\dd x + z\,\dd
x\wedge\dd y$, $\omega = \rho^{-3}\sigma$ and
\[
\dd\omega = \sum_i\partial_i\bigl(x_i\rho^{-3}\bigr)\,
\dd x\wedge\dd y\wedge\dd z
= \bigl(3\rho^{-3} - 3\rho^{-5}\cdot\rho^2\bigr)\dd
x\wedge\dd y\wedge\dd z = 0 .
\]
On $S_r$, for tangent $v_1, v_2$: $\omega(v_1, v_2) =
r^{-3}\det(x, v_1, v_2) = r^{-3}\,r\det(\nu, v_1, v_2) =
r^{-2}\,\dd S(v_1, v_2)$, so $\int_{S_r}\omega =
r^{-2}\cdot4\pi r^2 = 4\pi$: constant in $r$, as deformation
predicts, and nonzero --- so $\omega$ is \omterm{def:b3:forms:closedexact}{closed} but not
\omterm{def:b3:forms:closedexact}{exact} on $\R^3\setminus\{0\}$ (an \omterm{def:b3:forms:closedexact}{exact form} integrates to
$0$ over the boundaryless $S_r$ by Stokes). This is Gauss's
law: the flux of the field of a unit charge through any
enclosing sphere is $4\pi$, whatever the radius.
\end{solution}

\begin{solution}{exo:b3:forms:11}
Write $\omega = c\,\omega_\theta + \dd f$
(\cref{exo:b3:forms:12}) with $c =
\frac1{2\pi}\int_{S^1}\omega$. Then
\[
\int_\gamma\omega = c\int_\gamma\omega_\theta +
\int_\gamma\dd f
= c\cdot2\pi\operatorname{Ind}_\gamma(0) + 0
= n\int_{S^1}\omega,
\]
by \cref{prop:b3:forms:exactloop} and the definition of the
index. The single integer $n$ controls every period on the
punctured plane: \omterm{def:b3:forms:closedexact}{closed} $1$-forms cannot distinguish two
loops with the same \omterm{def:b3:forms:winding}{winding number}.
\end{solution}

\begin{solution}{exo:b3:forms:12}
Let $\alpha = \omega - c\,\omega_\theta$: \omterm{def:b3:forms:closedexact}{closed}, and
$\int_{S^1}\alpha = 0$ by the choice of $c$
($\int_{S^1}\omega_\theta = 2\pi$). Pull back by the polar
map $\Phi(\rho, \theta) = (\rho\cos\theta, \rho\sin\theta)$,
a surjective local diffeomorphism $(0, \infty)\times\R \to
U$: $\Phi^*\alpha$ is \omterm{def:b3:forms:closedexact}{closed}
(\cref{thm:b3:forms:pullback}(b)) on the \emph{convex} open
$(0,\infty)\times\R$, hence \omterm{def:b3:forms:closedexact}{exact}
(\cref{thm:b3:forms:poincare}): $\Phi^*\alpha = \dd g$.
For fixed $\rho$: $g(\rho, \theta + 2\pi) - g(\rho, \theta)
= \int_\theta^{\theta + 2\pi}\partial_\theta g\,\dd s$ is
the integral of $\alpha$ around the circle of radius
$\rho$, which equals $\int_{S^1}\alpha = 0$ (the annulus
between the two circles is a \omterm{def:b3:topology:compact}{compact} surface with \omterm{def:b3:forms:boundary}{boundary};
Stokes as in \cref{exo:b3:forms:10}, one dimension down).
So $g$ is $2\pi$-periodic in $\theta$ and descends to a
well-defined function $f$ on $U$ with $f\circ\Phi = g$;
$f$ is smooth ($\Phi$ is a local diffeomorphism) and
$\Phi^*(\dd f) = \dd g = \Phi^*\alpha$ forces $\dd f =
\alpha$. Hence $\omega = c\,\omega_\theta + \dd f$.
Uniqueness: integrating over $S^1$ fixes $c$, since \omterm{def:b3:forms:closedexact}{exact
forms} have zero period; and $f$ is unique up to an additive
constant. The map $[\omega] \mapsto
\frac1{2\pi}\int_{S^1}\omega$ is therefore a linear
isomorphism $H^1(\R^2\setminus\{0\}) \to \R$, and the class
of $\omega_\theta$ generates: the hole is exactly
one-dimensional, cohomologically speaking.
\end{solution}

\begin{solution}{pb:b3:forms:1}
\textbf{1.} $\dd\sigma = \sum_i(-1)^{i-1}\,\dd x_i\wedge\dd
x_1\wedge\dots\wedge\widehat{\dd x_i}\wedge\dots\wedge\dd
x_n$; carrying $\dd x_i$ across the $i-1$ preceding factors
costs $(-1)^{i-1}$, which cancels the prefactor: each of the
$n$ terms equals $\dd x_1\wedge\dots\wedge\dd x_n$, so
$\dd\sigma = n\,\dd x_1\wedge\dots\wedge\dd x_n$. Stokes on
the ball: $\int_S\sigma = \int_{\bar B}\dd\sigma =
n\operatorname{vol}(\bar B) > 0$.

\textbf{2.} $n = 2$: $\sigma = x\,\dd y - y\,\dd x$; on
$\gamma(t) = (\cos t, \sin t)$, $\gamma^*\sigma = (\cos^2t +
\sin^2t)\,\dd t = \dd t$, the arclength form:
$\int_S\sigma = 2\pi = 2\operatorname{vol}(\bar B^2)$.
$n = 3$: $\sigma\vert_S$ is the area form
(\cref{exo:b3:forms:5} with $\nu(x) = x$):
$\int_S\sigma = 4\pi = 3\cdot\tfrac{4\pi}3$. Both match
question 1.

\textbf{3.} Differentiating $\norm\varphi^2 = 1$: $2\langle
D\varphi(x)h, \varphi(x)\rangle = 0$ for every $h$, so
$\operatorname{im}D\varphi(x) \subseteq \varphi(x)^\perp$, a
hyperplane: $\operatorname{rk}D\varphi(x) \leq n - 1$. Since
$\dd\sigma$ is an $n$-form (question 1), pointwise
$\varphi^*(\dd\sigma)_x =
(D\varphi(x))^*(\dd\sigma)_{\varphi(x)} = 0$ by
\cref{prop:b3:forms:linearpullback}(2): a map into the
sphere has no room to pull back a volume.

\textbf{4.} The flaw is the second ``hence'': on the
$(n-1)$-dimensional $S$, every $(n-1)$-form is trivially
\omterm{def:b3:forms:closedexact}{closed} (there are no nonzero $n$-forms on an
$(n-1)$-manifold), but \emph{\omterm{def:b3:forms:closedexact}{closed} does not mean \omterm{def:b3:forms:closedexact}{exact}},
and $\int_S\dd\eta = 0$ requires an actual primitive $\eta$
defined on $S$. The restriction of $\sigma$ is precisely not
\omterm{def:b3:forms:closedexact}{exact} --- its integral is $n\operatorname{vol}(\bar B) \neq
0$ --- and this non-exactness powers the entire problem.

\textbf{5.} We shall integrate $r^*\sigma$ over the sphere
and count two ways. Because $r$ fixes $S$ pointwise, the
integral equals $\int_S\sigma = n\operatorname{vol}(\bar B)
\neq 0$. Because $r$ is defined on the ball, Stokes converts
the same integral into $\int_{\bar B}\dd(r^*\sigma) =
\int_{\bar B}r^*(\dd\sigma)$; and because $r$ takes values
in the sphere, question 3 makes that integrand vanish. One
number, two values: the retraction cannot exist.

\textbf{6.} Both integrals are computed through direct
parametrizations $\gamma$ of pieces of $S$
(\cref{def:b3:forms:integral}); since $r\circ\gamma =
\gamma$ (the parametrization lands in $S$, where $r$ is the
identity), $\gamma^*(r^*\sigma) = (r\circ\gamma)^*\sigma =
\gamma^*\sigma$ (\cref{thm:b3:forms:pullback}(a)): the local
integrands coincide, and any \omterm{lem:b3:forms:partition}{partition of unity} gives
$\int_Sr^*\sigma = \int_S\sigma$.

\textbf{7.} $r^*\sigma$ is a smooth $(n-1)$-form on a
\omterm{def:b3:topology:topology}{neighborhood} of the \omterm{def:b3:topology:compact}{compact} oriented $\bar B$, whose
\omterm{def:b3:forms:boundary}{boundary} with the induced \omterm{def:b3:forms:orientation}{orientation} is $S$: Stokes
(\cref{thm:b3:forms:stokes}) gives exactly
$\int_Sr^*\sigma = \int_{\bar B}\dd(r^*\sigma)$.

\textbf{8.} $\dd(r^*\sigma) = r^*(\dd\sigma)$
(\cref{thm:b3:forms:pullback}(b)), which vanishes by
question 3 applied to $\varphi = r$. Chaining questions
6--8:
\[
0 < n\operatorname{vol}(\bar B) = \int_S\sigma =
\int_Sr^*\sigma = \int_{\bar B}\dd(r^*\sigma) = 0 :
\]
absurd. \emph{There is no smooth retraction of $\bar B^n$
onto $S^{n-1}$} ($n \geq 2$).

\textbf{9.} A \omterm{def:b3:topology:continuity}{continuous} $r\colon\intcc{-1}1\to\{-1,1\}$
with $r(\pm1) = \pm1$ would map a \omterm{def:b3:topology:connected}{connected} set onto the
disconnected $\{-1, 1\}$, impossible: \omterm{def:b3:topology:continuity}{continuous} images of
\omterm{def:b3:topology:connected}{connected} sets are \omterm{def:b3:topology:connected}{connected} --- equivalently, the
intermediate value theorem would force $r$ to take the value
$0$. The same statement in every dimension is exactly Part
II; \omterm{def:b3:topology:connected}{connectedness} is the $1$-dimensional shadow of the
cohomological obstruction $\int_S\sigma \neq 0$.

\textbf{10.} $x \mapsto \norm{g(x) - x}$ is \omterm{def:b3:topology:continuity}{continuous} and
everywhere $> 0$ on the \omterm{def:b3:topology:compact}{compact} $\bar B$: its minimum
$\delta$ is attained, hence $> 0$.

\textbf{11.} $\norm{x + tu}^2 = 1$ reads $t^2 + 2t\langle x,
u\rangle + \norm x^2 - 1 = 0$, whose roots are
\[
t_\pm = -\langle x, u\rangle \pm \sqrt{\langle x,
u\rangle^2 + 1 - \norm x^2} .
\]
Their product is $\norm x^2 - 1 \leq 0$: the roots straddle
$0$ (or one vanishes), so the ray parameter --- the
nonnegative root --- is $t(x) = t_+$.

\textbf{12.} If the radicand vanished at $x \in \bar B$:
both its terms being nonnegative, $\norm x = 1$ and
$\langle x, u(x)\rangle = 0$, i.e.\ $\langle x, x -
g(x)\rangle = 0$, i.e.\ $\langle x, g(x)\rangle = 1$. By
Cauchy--Schwarz, $1 = \langle x, g(x)\rangle \leq
\norm x\,\norm{g(x)} \leq 1$: equality throughout, which
forces $g(x)$ collinear with $x$, of norm $1$, positively:
$g(x) = x$ --- excluded. So the radicand is \omterm{def:b3:topology:continuity}{continuous} and
$> 0$ on $\bar B$, hence bounded below by some $c > 0$
there and on a \omterm{def:b3:topology:topology}{neighborhood} (uniform \omterm{def:b3:topology:continuity}{continuity}). On that
\omterm{def:b3:topology:topology}{neighborhood}, $u$ is smooth ($\norm{x - g(x)} \geq
\delta/2$ shrinking if needed), the radicand stays $\geq
c/2$, and the square root is smooth on $\intoo0\infty$: $r$
is smooth near $\bar B$.

\textbf{13.} $\norm{r(x)} = 1$ by construction of $t(x)$:
$r(\bar B) \subseteq S$. For $x \in S$: $\langle x,
u(x)\rangle = \frac{1 - \langle x, g(x)\rangle}{\norm{x -
g(x)}} \geq 0$ (Cauchy--Schwarz once more), and $\norm x =
1$ reduces the radicand to $\langle x, u\rangle^2$, whose
square root is $\langle x, u\rangle$ itself (it is $\geq
0$): $t(x) = 0$ and $r(x) = x$. So $r$ is a smooth
retraction of the ball onto the sphere --- contradicting
Part II. \emph{Every smooth self-map of $\bar B$ has a
fixed point.}

\textbf{14.} $\varepsilon > 0$ exactly as in question 10.
The polynomials form a subalgebra of $\mathcal C(\bar B,
\R)$ containing the constants and separating points ($x
\mapsto x_i$ do), so Stone--Weierstrass
(\cref{thm:b3:complete:stoneweierstrass}) approximates each
coordinate: pick polynomials $p_i$ with $\sup_{\bar
B}\abs{p_i - f_i} < \frac{\varepsilon}{2\sqrt n}$; the
vector map $p = (p_1, \dots, p_n)$ then satisfies
$\sup_{\bar B}\norm{p - f} \leq
\bigl(\sum_i\sup_{\bar B}\abs{p_i - f_i}^2\bigr)^{1/2} <
\varepsilon/2$.

\textbf{15.} On $\bar B$: $\norm p \leq \norm f +
\frac\varepsilon2 \leq 1 + \frac\varepsilon2$, so
$\norm{g} = \frac{\norm p}{1 + \varepsilon/2} \leq 1$:
$g(\bar B) \subseteq \bar B$. Moreover $\norm{g - p} =
\frac{\varepsilon/2}{1 + \varepsilon/2}\norm p \leq
\frac\varepsilon2$, hence $\norm{g - f} \leq \norm{g - p} +
\norm{p - f} < \varepsilon$ on $\bar B$.

\textbf{16.} $g$ is polynomial, hence smooth, and maps
$\bar B$ into itself: Part III provides $x_0 = g(x_0)$.
Then $\norm{f(x_0) - x_0} = \norm{f(x_0) - g(x_0)} <
\varepsilon = \min_{\bar B}\norm{f - \operatorname{id}}$:
contradiction. \emph{Every \omterm{def:b3:topology:continuity}{continuous map} $\bar B^n \to
\bar B^n$ has a fixed point.}

\textbf{17.} \emph{Open ball:} $f(x) = \frac{x + e_1}2$ maps
$B$ into $B$ ($\norm{f(x)} < 1$ strictly) and its only fixed
point $e_1$ lies on the sphere: compactness lost.
\emph{Sphere:} the antipodal map $x \mapsto -x$ is
fixed-point free; $S$ is \omterm{def:b3:topology:compact}{compact} but has the ``wrong''
\omterm{def:b3:topology:topology}{topology} --- it is exactly the non-retract of Part II.
\emph{Annulus:} a rotation by any angle $\not\equiv 0$ fixes
nothing; the hole shelters the rotation --- convexity (more
precisely, the ball-like \omterm{def:b3:topology:topology}{topology}) lost. Brouwer's theorem
is really about \emph{\omterm{def:b3:topology:compact}{compact} convex} sets, as question 18
exploits.

\textbf{18.} For $x \in \Delta$: some $x_j > 0$, so $(Ax)_i
\geq A_{ij}x_j > 0$ for every $i$; hence $\norm{Ax}_1 > 0$
and $T(x) = Ax/\norm{Ax}_1$ is well defined, \omterm{def:b3:topology:continuity}{continuous}, and
lands in $\Delta$ (positive entries summing to $1$).
$\Delta$ is convex, \omterm{def:b3:topology:compact}{compact}, with nonempty \omterm{def:b3:topology:interior}{interior} in the
affine hyperplane $\{\sum x_i = 1\} \cong \R^{n-1}$; the
radial map from its barycenter --- each ray from the
barycenter meets $\partial\Delta$ in exactly one point, by
convexity and compactness, and the corresponding gauge
function is \omterm{def:b3:topology:continuity}{continuous} --- is a \omterm{def:b3:topology:continuity}{homeomorphism} $\Delta \to
\bar B^{n-1}$. Transporting Brouwer through it: $T$ has a
fixed point $x^*$, i.e.\ $Ax^* = \lambda x^*$ with $\lambda
= \norm{Ax^*}_1 > 0$; and $x^* = Ax^*/\lambda$ has strictly
positive entries by the opening computation. \emph{A
positive matrix has a positive eigenvector.}

\textbf{19.} Question 18 gives $A\pi = \lambda\pi$, $\pi \in
\Delta$, $\pi > 0$. Sum the coordinates: $\sum_i(A\pi)_i =
\sum_j\pi_j\sum_iA_{ij} = \sum_j\pi_j = 1$ (columns sum to
$1$), while $\sum_i\lambda\pi_i = \lambda$. So $\lambda = 1$
and $A\pi = \pi$: a stationary probability vector --- the
equilibrium of the Markov chain, PageRank's mathematical
core.

\textbf{20.} On $S$: $\norm{F_t(x)}^2 = \norm x^2 +
2t\langle x, v(x)\rangle + t^2\norm{v(x)}^2 = 1 + 0 + t^2$
by tangency and $\norm v = 1$: $F_t(S) \subseteq
\sqrt{1+t^2}\,S$.

\textbf{21.} Fix a finite atlas of direct parametrizations
$\gamma$ and a \omterm{lem:b3:forms:partition}{partition of unity}, independent of $t$. In a
chart, $F_t\circ\gamma = \gamma + t(v\circ\gamma)$, so each
coefficient of $(F_t\circ\gamma)^*\sigma$ is a sum of
products of one factor $(x_i + tv_i)\circ\gamma$ (affine in
$t$) and an $(n-1)\times(n-1)$ determinant with entries
affine in $t$: a polynomial in $t$ of degree $\leq n$ with
coefficients smooth in the chart variable. Multiplying by
the $t$-independent partition functions and integrating
termwise: $P(t) = \sum_{k=0}^n c_kt^k$, a polynomial.

\textbf{22.} \emph{Injectivity:} $v$ is Lipschitz on $S$
(smooth on a \omterm{def:b3:topology:compact}{compact}), say with constant $L$; for $\abs t <
1/L$, $\norm{F_t(x) - F_t(y)} \geq (1 - \abs
tL)\norm{x - y} > 0$. \emph{Diffeomorphism:} $G_t =
F_t/\sqrt{1 + t^2}$ maps $S$ to $S$; in charts its Jacobians
converge uniformly to those of $G_0 = \operatorname{id}$ as
$t \to 0$, so for small $t$ they are invertible and $G_t$ is
a local diffeomorphism (\cref{thm:b3:submanifolds:ift} in
charts), injective, with image open (local diffeo) and
\omterm{def:b3:topology:compact}{compact} in the \emph{\omterm{def:b3:topology:connected}{connected}} $S^{n-1}$ ($n \geq 2$):
image $= S$, so $G_t$ is a diffeomorphism of $S$.
\emph{Scaling:} under $s_\lambda(x) = \lambda x$, each
coefficient $x_i$ gains $\lambda$ and each of the $n - 1$
differentials gains $\lambda$: $s_\lambda^*\sigma =
\lambda^n\sigma$. Since $F_t = s_{\sqrt{1+t^2}}\circ G_t$:
\[
P(t) = \int_SG_t^*\bigl(s_{\sqrt{1+t^2}}^{\,*}\sigma\bigr)
= (1 + t^2)^{n/2}\int_SG_t^*\sigma
= (1 + t^2)^{n/2}\int_S\sigma
\quad\text{for small } t,
\]
the last equality by \cref{lem:b3:forms:consistency} ($G_t$
a diffeomorphism of $S$, orientation-preserving for small
$t$: its chart Jacobian determinants vary \omterm{def:b3:topology:continuity}{continuously},
never vanish, and are positive at $t = 0$).

\textbf{23.} If $n$ is odd and a smooth unit tangent field
exists, questions 21--22 make the polynomial
$P(t)/\int_S\sigma$ (legitimate: $\int_S\sigma \neq 0$ by
question 1) agree with $(1 + t^2)^{n/2}$ near $0$; two
smooth functions agreeing near $0$ with one a polynomial
force $(1 + t^2)^{n/2}$ to be that polynomial on all of
$\R$. But if $Q(t)^2 = (1 + t^2)^n$ with $Q \in \R[t]$,
unique factorization in $\R[t]$
(\cref{ch:b3:rings}) gives the \omterm{def:b3:rings:divisibility}{irreducible} $t^2 + 1$ an even
multiplicity in $Q^2$ and the odd multiplicity $n$ in
$(1 + t^2)^n$: impossible. So no smooth unit tangent field
exists on $S^{n-1}$ for $n$ odd --- the even-dimensional
spheres. Finally, a merely \omterm{def:b3:topology:continuity}{continuous} nowhere-vanishing
tangent field $w$ would produce one: extend it to a
\omterm{def:b3:topology:topology}{neighborhood} by $\tilde w(x) = w(x/\norm x)$, mollify
componentwise (\cref{thm:b3:lp:regularization}) to a smooth
$v_0$ with $\sup_S\norm{v_0 - \tilde w} <
\frac12\min_S\norm w$, project tangentially $v_1(x) = v_0(x)
- \langle v_0(x), x\rangle x$ --- on $S$ this changes $v_0$
by at most its normal component, itself at most $\norm{v_0 -
\tilde w}$ since $\tilde w$ is tangent, so $\norm{v_1 - \tilde
w} \leq 2\norm{v_0 - \tilde w} < \min\norm w$ and $v_1$
never vanishes on $S$ --- and normalize: $v =
v_1/\norm{v_1}$ is a smooth unit tangent field. Hence on
every even-dimensional sphere, \emph{every \omterm{def:b3:topology:continuity}{continuous}
tangent field has a zero}: every wind on Earth leaves a calm
point.

\textbf{24.} $\langle v(x), x\rangle = \sum_j(-y_jx_j +
x_jy_j) = 0$: tangent; $\norm{v(x)} = \norm x = 1$: unit.
Smoothness is clear (linear map). So every odd-dimensional
sphere carries a smooth unit tangent field --- multiplication
by $\iu$ along the complex lines --- and question 23's
obstruction is exactly the parity of the dimension. In the
polynomial argument, for even $n$ the function $(1 +
t^2)^{n/2}$ \emph{is} a polynomial, and no contradiction
arises: the proof does not merely fail to apply, its
conclusion is genuinely false, as $v$ witnesses.

\textbf{25.} Suppose $f$ never vanishes on $\bar B$. Then
$g(x) = -f(x)/\norm{f(x)}$ is \omterm{def:b3:topology:continuity}{continuous} $\bar B \to S
\subseteq \bar B$, and Brouwer (question 16) provides $x^* =
g(x^*)$. Since $g$ takes values in $S$, $x^* \in S$, and
\[
\langle f(x^*), x^*\rangle = \bigl\langle f(x^*),\
-\tfrac{f(x^*)}{\norm{f(x^*)}}\bigr\rangle
= -\norm{f(x^*)} < 0,
\]
contradicting the \omterm{def:b3:forms:boundary}{boundary} hypothesis. So $f$ has a zero.
Surjectivity: given $y \in \R^n$, apply the above to $f(x) =
F(x) - y$ on a ball $\bar B(0, R)$ with $R$ so large that
$\langle F(x), x\rangle \geq \norm y\,\norm x$ on the sphere
of radius $R$ (coercivity); then $\langle f(x), x\rangle =
\langle F(x), x\rangle - \langle y, x\rangle \geq 0$ there
(Cauchy--Schwarz), and the rescaled statement gives a zero
of $f$: $F(x) = y$. Every coercive \omterm{def:b3:topology:continuity}{continuous} field is onto
--- the degree-free shadow of the variational existence
theorems, delivered by pure \omterm{def:b3:topology:topology}{topology}.
\end{solution}
