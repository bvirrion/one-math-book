\begin{solution}{exo:b3:submanifolds:8}
Inducción en $n$; $n = 1$ trivial. La función de Rayleigh
$f(x) = \langle Ax, x\rangle$ alcanza su máximo $\lambda_1$
en el \omterm{def:b3:topology:compact}{compacto} $S^{n-1}$ en algún $v_1$; Lagrange
(\cref{thm:b3:submanifolds:lagrange}, esfera como conjunto de niveles)
da $2Av_1 = 2\lambda v_1$ y $\lambda = \langle Av_1,
v_1\rangle = \lambda_1$. El hiperplano$v_1^\perp$ es
$A$-invariante (simetría: $\langle Av, v_1\rangle = \langle
v, Av_1\rangle = 0$); la restricción es simétrica y
la inducción produce una base propia ortonormal $v_2, \dots, v_n$
de $v_1^\perp$ con valores propios $\lambda_2 \geq \dots \geq
\lambda_n$, cada uno el máximo de$f$ en la esfera de la
ortocomplemento restante. Courant--Fischer sigue exactamente
como en \cref{exo:b3:spectral:8}: expandir $x = \sum c_iv_i$;
en un espacio de prueba de dimensiones $k$ se cruza con
$\operatorname{Vect}(v_k, \dots, v_n)$ (recuento de dimensiones en
$\R^n$) para obtener $\min \leq \lambda_k$, y
$\operatorname{Vect}(v_1, \dots, v_k)$ logra $\min =
\lambda_k$.
\end{solution}

\begin{solution}{exo:b3:submanifolds:11}

(b) $O_n$ es el conjunto de niveles $F = 0$ estudiado en el capítulo:

\]

(c) $\exp(tK)^{\mathsf T}\exp(tK) = \exp(tK^{\mathsf T})

\begin{solution}{pb:b3:submanifolds:1}
\textbf{1.} Mapa $q = t + x\mathrm i + y\mathrm j + z\mathrm k
\mapsto \bigl(\begin{smallmatrix}\alpha & \beta\\ -\bar\beta
& \bar\alpha\end{smallmatrix}\bigr)$con$\alpha = t + \iu
x$,$\beta = y + \iu z$: se comprueba que$1, \mathrm i,
\mathrm j, \mathrm k$vaya a$I$,$\bigl(\begin{smallmatrix}
\iu & 0\\ 0 & -\iu\end{smallmatrix}\bigr)$,
$\bigl(\begin{smallmatrix}0 & 1\\ -1 &
0\end{smallmatrix}\bigr)$,$\bigl(\begin{smallmatrix}0 &
\iu\\ \iu & 0\end{smallmatrix}\bigr)$, cuyos productos
reproducir la tabla de cuaterniones: la aplicación es una inyectiva
morfismo de álgebra, por lo que $\mathbb H$ hereda la asociatividad;
$N(q) = \abs\alpha^2 + \abs\beta^2 = \det$ es multiplicativo,
y la conjugación corresponde a la transpuesta adjugada,
dando $\overline{pq} = \bar q\bar p$. \omterm{ex:b3:groups:actions}{Centro}: desplazamientos
con fuerzas $\mathrm i$ $y = z = 0$, con fuerzas $\mathrm j$
$x = 0$: $\R$.

\textbf{2.} $q\bar q = N(q)$: para $q \neq 0$, $q^{-1} = \bar
q/N(q)$: un álgebra de división. En$S^3$:$N(pq) = 1$ y
$N(q^{-1}) = 1$: un grupo; y $S^3 \subseteq \R^4$ es el
esfera unitaria: una subcolectora \omterm{def:b3:topology:compact}{compacto} $3$.

\textbf{3.} $v$ es puro si $\bar v = -v$; entonces
$\overline{qv\bar q} = q\bar v\bar q = -qv\bar q$: $\rho_q$
conserva $P$. La linealidad es clara; $N(qv\bar q) =
N(q)N(v)N(q) = N(v)$: una isometría de$(P, N) \cong (\R^3,
\norm\cdot^2)$:$\rho_q \in O(3)$. Y$\rho_{pq}(v) =
pqv\overline{pq} = p(qv\bar q)\bar p =
\rho_p(\rho_q(v))$: un morfismo.

\textbf{4.} $\rho_q = \mathrm{id}$ y si $qv = vq$ para todos puros
$v$, si $q$ conmuta con $\mathrm i, \mathrm j, \mathrm k$,
si $q$ es central (pregunta 1): $q \in \R\cap S^3 =
\{\pm1\}$.

\textbf{5.} Escribe $q = t + p$ ($t \in \R$, $p$ puro): $1 =
N(q) = t^2 + N(p)$, entonces$t = \cos\frac\theta2$y$p =
\sin\frac\theta2\,u$con$N(u) = 1$para algunos$\theta$
(si $p = 0$, $q = \pm1$ actúa de forma trivial). Desde $u^2 = -N(u)
= -1$,$q$y$u$conmutan, y$\rho_q(u) = qu\bar q =
uq\bar q = u$: el eje. Para unidades puras$w \perp u$: el
regla de producto $vw = -\langle v, w\rangle + v\wedge w$
(expandir en coordenadas de la tabla) da $uw = u\wedge w
= -wu$. entonces
\[
\rho_q(w) = \bigl(\cos\tfrac\theta2 +
\sin\tfrac\theta2u\bigr)\,w\,\bigl(\cos\tfrac\theta2 -
\sin\tfrac\theta2u\bigr)
= \cos\theta\,w + \sin\theta\,(u\wedge w),
\]
usando $uwu = -u^2w = w$ y las fórmulas de doble ángulo: el
rotación del ángulo $\theta$ en el plano orientado $(w,
u\wedge w)$.

\textbf{6.} Cada $\rho_q$ es una rotación sobre $u$ por
$\theta$: en base ortonormal $(u, w, u\wedge w)$ es
la matriz tiene determinante $+1$: $\operatorname{im}\rho \subseteq
SO(3)$. Suryectividad: una matriz$R \in SO(3)$tiene$1$ como
valor propio, ya que
\[
\det(R - I) = \det R\,\det(I - R^{\mathsf T}) = \det(I - R)
= (-1)^3\det(R - I),
\]
entonces $\det(R - I) = 0$. Tome un vector propio unitario $u$; $R$
conserva $u^\perp$ y lo restringe a una rotación de
algún ángulo $\theta$ (plano ortogonal, determinante $1$):
$R = \rho_q$ para $q = \cos\frac\theta2 +
\sin\frac\theta2\,u$. Con la pregunta 4 y la primera
teorema de isomorfismo (\cref{thm:b3:groups:firstiso}): $SO(3)
\cong S^3/\{\pm1\}$.

\textbf{7.} $O_3$ es un subcolector dimensional \omterm{def:b3:topology:compact}{compacto} $3$
(\cref{exo:b3:submanifolds:5}); $\det$ es \omterm{def:b3:topology:continuity}{continuo} en él
con valores en $\{\pm1\}$, entonces $SO(3) = O_3\cap\{\det = 1\}$
está abierto y cerrado en $O_3$: una unión de \omterm{def:b3:topology:components}{componentes conectados},
de ahí una subvariedad \omterm{def:b3:topology:compact}{compacto} $3$, con el mismo \omterm{def:b3:submanifolds:tangent}{tangente
espacio} en $I$: las matrices antisimétricas.

\textbf{8.} $A_u^2v = u\wedge(u\wedge v) = \langle u,
v\rangle u - v$(unidad$u$), entonces$A_u^3v = u\wedge(\langle
u,v\rangle u - v) = -u\wedge v$:$A_u^3 = -A_u$. Dividir
la serie exponencial de \omterm{def:b3:residues:singularities}{residuos} de potencias mod la relación
$A^3 = -A$:
\[
\eu^{\theta A_u} = I + \Bigl(\theta - \frac{\theta^3}{3!} +
\cdots\Bigr)A_u + \Bigl(\frac{\theta^2}{2!} -
\frac{\theta^4}{4!} + \cdots\Bigr)A_u^2
= I + \sin\theta\,A_u + (1 - \cos\theta)\,A_u^2 .

En $u$: $A_uu = 0$: arreglado. En $w \perp u$: $\eu^{\theta
A_u}w = w + \sin\theta\,u\wedge w + (1 - \cos\theta)(-w) =

$u$, ángulo $\theta$ --- Rodrigues. Cada rotación tiene esto
formulario (pregunta 6): $\exp$ está en $SO(3)$ desde el
