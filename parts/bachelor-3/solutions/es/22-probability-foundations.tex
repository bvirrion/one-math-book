\begin{solution}{pb:b3:probability:1}

X_1^-$. Si el teorema se cumple para variables no negativas,
aplícalo a ambas mitades y resta:
$\frac{S_n}n = \frac{S_n^+}n - \frac{S_n^-}n \to \E X_1^+ -
\E X_1^- = m$ a.s.

\textbf{2.}$\P(X_n \neq Y_n) = \P(X_n > n) = \P(X_1 > n)$
(idéntico \omterm{def:b3:probability:space}{leyes}) y $\sum_n\P(X_1 > n) \leq \sum_n\P(X_1
\geq n) \leq \E X_1 < \infty$
(\cref{exo:b3:product:3}(a)). Borel--Cantelli (1): a.s.\
$X_n = Y_n$ para todos los $n$ grandes, por lo que $S_n - S_n^*$ es
eventualmente constante en $n$: $\frac{S_n - S_n^*}n \to 0$
a.s., y las dos sumas normalizadas comparten su asintótica
comportamiento.

\textbf{3.} $X_1\mathbf 1_{X_1 \leq n} \nearrow X_1$: MCT
da $\E Y_n \to m$; Cesàro significa un convergente
La secuencia converge al mismo límite: $\frac{\E S_n^*}n =
\frac1n\sum_{k\leq n}\E Y_k \to m$. Por lo tanto basta con
probar $\frac{S^*_n - \E S^*_n}{n} \to 0$ a.s.

\textbf{4.} $\V(Y_n) \leq \E Y_n^2 = \E[X_1^2\mathbf
1_{X_1\leq n}]$. Por Tonelli para serie,
\[
\sum_n\frac{\E[X_1^2\mathbf 1_{X_1\leq n}]}{n^2}
= \E\Bigl[X_1^2\!\!\sum_{n \geq \max(X_1, 1)}\!\frac1{n^2}
\Bigr]
\leq \E\Bigl[X_1^2\cdot\frac{4}{\max(X_1,1)}\Bigr]
\leq 4\,\E[X_1] < \infty,
\]
usando $\sum_{n\geq x}n^{-2} \leq \frac4x$ para $x \geq 1$
(para $x \geq 2$: $\leq \frac1{x-1} \leq \frac2x$; para $1
\leq x < 2$:$\leq \frac{\pi^2}6 \leq \frac4x$ desde
$\frac4x > 2$) y $X_1^2/\max(X_1, 1) \leq X_1$ en ambos
casos $X_1 \gtrless 1$.

\textbf{5.} Por pares \omterm{def:b3:probability:independence}{independencia} da $\E[(Y_i - \E
Y_i)(Y_j - \E Y_j)] = 0$por$i \neq j$ (el producto
fórmula para dos variables), por lo que las variaciones suman: $\V(S^*_k) =
\sum_{n\leq k}\V(Y_n)$. Chebyshev en cada$k_j$ y
sumando:
\[
\sum_j\P\Bigl(\abs{S^*_{k_j} - \E S^*_{k_j}} \geq
\varepsilon k_j\Bigr)
\leq \frac1{\varepsilon^2}\sum_j\frac1{k_j^2}\sum_{n\leq
k_j}\V(Y_n)
= \frac1{\varepsilon^2}\sum_n\V(Y_n)\!\!\sum_{j : k_j\geq
n}\!\frac1{k_j^2}
\]
(Tonelli para la serie doble no negativa).

\textbf{6.} $k_j = \lfloor\alpha^j\rfloor \geq
\frac{\alpha^j}2$(válido una vez$\alpha^j \geq 1$, es decir todos
$j \geq 0$: $\lfloor x\rfloor \geq \frac x2$ para $x \geq
1$). Por lo tanto
\[
\sum_{j : k_j \geq n}\frac1{k_j^2}
\leq 4\sum_{j : \alpha^j \geq n}\alpha^{-2j}
\leq \frac{4}{1 - \alpha^{-2}}\cdot\frac1{n^2}
= \frac{C_\alpha}{n^2},
\]
(serie geométrica del primer $j$ con $\alpha^j \geq
n$). Combinando con las preguntas 4 a 5, la suma doble es
finito; Borel--Cantelli (1), aplicado para cada racional

$\frac{S^*_{k_j} - \E S^*_{k_j}}{k_j} \to 0$ a.s., y con
pregunta 3: $\frac{S^*_{k_j}}{k_j} \to m$ a.s.

\textbf{7.} $Y_n \geq 0$ hace $n \mapsto S^*_n$
no decreciente: para $k_j \leq n \leq k_{j+1}$,

\]
