\begin{solution}{exo:b3:residues:1}
$\frac{\sin z}z = 1 - \frac{z^2}6 + \cdots$: extraíble,
\omterm{def:b3:residues:singularities}{residuo} $0$. $\frac{\eu^z - 1}{z^2} = \frac1z + \frac12 +
\frac z6 + \cdots$: poste simple, \omterm{def:b3:residues:singularities}{residuo}$1$.
$\frac1{z(z-1)^2}$: polo simple en $0$ con \omterm{def:b3:residues:singularities}{residuo}
$\frac1{(0-1)^2} = 1$; bipolar en $1$ con \omterm{def:b3:residues:singularities}{residuo}
$\frac{\dd}{\dd z}\bigl(\frac1z\bigr)\big|_{z=1} = -1$.
$\frac{\cos z}{z^3} = \frac1{z^3} - \frac1{2z} + \cdots$: poste
de orden $3$, \omterm{def:b3:residues:singularities}{residuo} $-\frac12$. $\eu^{1/z} = \sum_{n\geq0}
\frac{z^{-n}}{n!}$: imprescindible, \omterm{def:b3:residues:singularities}{residuo}$1$.$\frac1{\sin z}$:
postes simples en $n\pi$; \omterm{def:b3:residues:singularities}{residuo} $\frac1{\cos 0} = 1$ en $0$,
$\frac1{\cos\pi} = -1$ en $\pi$
(\cref{met:b3:residues:compute}, $g/h'$).
\end{solution}

\begin{solution}{exo:b3:residues:2}
Con $z = \eu^{\iu t}$, $\cos t = \frac{z + z^{-1}}2$, $\dd t
= \frac{\dd z}{\iu z}$:
\[
\int_0^{2\pi}\frac{\dd t}{a + \cos t}
= \oint_{\abs z = 1}\frac{2\,\dd z}{\iu\,(z^2 + 2az + 1)} .
\]
Las raíces $z_\pm = -a \pm \sqrt{a^2 - 1}$ satisfacen $z_+z_- =
1$con$\abs{z_+} < 1 < \abs{z_-}$; el \omterm{def:b3:residues:singularities}{residuo} en$z_+$ es
$\frac{1}{z_+ - z_-} = \frac1{2\sqrt{a^2-1}}$, entonces la integral
es $\frac2\iu\cdot2\iu\pi\cdot\frac1{2\sqrt{a^2-1}} =
\frac{2\pi}{\sqrt{a^2-1}}$. Como$a \to 1^+$ explota (el
el integrando alcanza su punto máximo en $t = \pi$); como se comporta $a \to \infty$
como $\frac{2\pi}a$, que coincide con $\int\frac{\dd t}a$.
\end{solution}

\begin{solution}{exo:b3:residues:3}
Primera integral: polos superiores de $\frac{z^2}{1+z^6}$ en $p =
\eu^{\iu\pi/6}, \iu, \eu^{5\iu\pi/6}$; en cada uno,
$\operatorname{Res} = \frac{p^2}{6p^5} = \frac{p^3}{6p^6} =
-\frac{p^3}6$y$p^3$toman los valores$\iu, -\iu, \iu$:
suma de \omterm{def:b3:residues:singularities}{residuos} $-\frac{\iu}{6}$. El arco es
$O(R^{-4})\cdot O(R) \to 0$:
\[
\int_\R\frac{x^2\,\dd x}{1 + x^6} =
2\iu\pi\Bigl(-\frac \iu6\Bigr) = \frac\pi3 .
\]
Segundo: doble polo en $\iu$ de $\frac1{(1+z^2)^2} =
\frac1{(z-\iu)^2(z+\iu)^2}$:
\[
\operatorname{Res} = \frac{\dd}{\dd z}\,(z +
\iu)^{-2}\Big|_{z=\iu} = \frac{-2}{(2\iu)^3} =
\frac{-2}{-8\iu} = -\frac\iu4,
\qquad
\int_\R\frac{\dd x}{(1+x^2)^2} =
2\iu\pi\cdot\Bigl(-\frac\iu4\Bigr) = \frac\pi2 ,
\]
compatible con \cref{exo:b3:fouriertransform:3}(b).
\end{solution}

\begin{solution}{exo:b3:residues:4}
Cerrar $\frac{\eu^{\iu tz}}{z^2 + a^2}$ en el semiplano superior
($t \geq 0$): ahí $\abs{\eu^{\iu tz}} =
\eu^{-t\operatorname{Im}z} \leq 1$, entonces el arco contribuye
$O(R^{-2})\cdot O(R) \to 0$. El poste cerrado único $\iu a$
es simple con \omterm{def:b3:residues:singularities}{residuo} $\frac{\eu^{-at}}{2\iu a}$:
\[
\int_\R\frac{\eu^{\iu tx}}{x^2 + a^2}\dd x =
\frac{\pi}{a}\,\eu^{-at},
\qquad\text{hence}\qquad
\int_\R\frac{\cos(tx)}{x^2+a^2}\dd x = \frac\pi
a\,\eu^{-a\abs t}
\]
(parte real; incluso en $t$). Esta es la contraparte de la inversión.
de $\widehat{\eu^{-a\abs x}} = \frac{2a}{a^2+\xi^2}$
(\cref{exo:b3:fouriertransform:1}): los dos cálculos
confirmarse mutuamente a través de
\cref{thm:b3:fouriertransform:inversion}.
\end{solution}

\begin{solution}{exo:b3:residues:5}
Fracciones parciales: $f = \frac1{z-2} - \frac1{z-1}$.
En $\abs z < 1$ (Taylor):
$f = \sum_{n\geq0}\bigl(1 - 2^{-n-1}\bigr)z^n$.
En $1 < \abs z < 2$: $\frac1{z-2} =
-\sum_{n\geq0}\frac{z^n}{2^{n+1}}$y$-\frac1{z-1} =
-\sum_{n\geq1}z^{-n}$: una auténtica serie a dos caras.
En $\abs z > 2$: $f = \sum_{n\geq1}\bigl(2^{n-1} -
1\bigr)z^{-n}$. Los tres se diferencian porque las expansiones de Laurent
se adjuntan a \emph{anillos}, no puntos: cada región tiene su
propias expansiones geométricas. Los coeficientes $z^{-1}$ ($-1$ y
$0$ respectivamente) son integrales sobre círculos que rodean
\emph{las singularidades dentro}, no \omterm{def:b3:residues:singularities}{residuos} en $0$ ($f$ es
\omterm{def:b3:holomorphic:holo}{holomorfo} en $0$): para $1 < \abs z < 2$ el coeficiente
$-1$ es $\operatorname{Res}(f, 1)$; para $\abs z > 2$ el
El coeficiente $0$ es $\operatorname{Res}(f,1) +
\operatorname{Res}(f,2) = -1 + 1$. El \omterm{def:b3:residues:singularities}{residuos} de$f$:$-1$
en $1$ y $+1$ en $2$.
\end{solution}

\begin{solution}{exo:b3:residues:6}
(a) El \omterm{thm:b3:residues:laurent}{serie laurent} $\sum_nz^{-n}/n!$ tiene infinitos
términos negativos: esencial. Resolviendo $\eu^{1/z} = w$ ($w \neq
0$):$\frac1z = \log\abs w + \iu\arg w + 2\iu\pi k$, entonces
\[
z_k = \frac1{\log\abs w + \iu\arg w + 2\iu\pi k}
\xrightarrow[k\to\infty]{} 0 :
\]
cada valor distinto de cero se alcanza infinitamente cerca de $0$ ---
más fuerte que \omterm{ex:b3:lebesgue:gamma}{densidad}.
(b) A lo largo de $z = 1/x$, $x \to +\infty$: $\eu^x \to \infty$;
a lo largo de $z = \iu/y$: módulo $1$. Un poste requiere
$\abs{f(z)} \to \infty$ \emph{a lo largo de cada enfoque}: aquí el
El límite simplemente no existe, incluso en $[0, +\infty]$.
\end{solution}

\begin{solution}{exo:b3:residues:7}
(a) El $\abs z = 1$: $\abs{z^7 + z - 1} \leq 3 < 4 =
\abs{-4z^3}$. Rouché con$f = -4z^3$,$g = z^7 + z - 1$:
tres ceros en el disco.
(b) El $\abs z = 1$: $\abs{z^4 + 1} \leq 2 < 5 = \abs{5z}$:
un cero en $\abs z < 1$. En $\abs z = 2$: $\abs{5z + 1} \leq
11 < 16 = \abs{z^4}$: cuatro ceros en$\abs z < 2$. Por lo tanto tres

(c) Para $P = z^n + a_{n-1}z^{n-1} + \dots$: en $\abs z = R >

\bigl(\sum\abs{a_k}\bigr)R^{n-1} < R^n = \abs{z^n}$:$P$ tiene
exactamente $n$ ceros en $D(0, R)$ --- d'Alembert--Gauss con
multiplicidad, por puro conteo.
\end{solution}

\begin{solution}{exo:b3:residues:8}
(a) Supongamos $f(a) = 0$, $f \not\equiv 0$: elija $r$ con $f$
cero libre en el círculo $C = \partial D(a, r)$ (\omterm{thm:b3:holomorphic:identity}{aislado
ceros}) y $m = \min_C\abs f > 0$. Por
\cref{thm:b3:holomorphic:weierstrassconv}, $f_n \to f$ y
$f_n' \to f'$ uniformemente en $C$; para $n$ grande, $\abs{f_n}
\geq m/2$en$C$, entonces
\[
\frac1{2\iu\pi}\int_C\frac{f_n'}{f_n}
\longrightarrow \frac1{2\iu\pi}\int_C\frac{f'}{f} \geq 1
\]
(el límite cuenta el cero $a$; convergencia porque
los numeradores convergen uniformemente y los denominadores son uniformemente
delimitado a continuación). El lado izquierdo es un número entero que cuenta ceros de
$f_n$ en el disco: eventualmente debe ser $\geq 1$ ---
contradiciendo la libertad cero. Entonces $f$ no tiene cero.

(b) Reparar $w \in \Omega$ y aplicar (a) en \omterm{def:b3:topology:connected}{conectado} \omterm{def:b3:topology:topology}{conjunto abierto}
$\Omega\setminus\{w\}$ (eliminando un punto de un \omterm{def:b3:topology:connected}{conectado} abierto
El subconjunto de $\C$ conserva \omterm{def:b3:topology:connected}{conectividad}) a $g_n(z) = f_n(z) -
f_n(w)$, libre de ceros allí por inyectividad, convergiendo a$g = f
- f(w)$. Si$f$no es constante,$g \not\equiv 0$ en
$\Omega\setminus\{w\}$, por lo que $g$ es cero libre allí: $f(z) \neq
f(w)$para todos los$z \neq w$. Como$w$era arbitrario,$f$ es
inyectivo.
\end{solution}

\begin{solution}{exo:b3:residues:9}
El límite del sector consta de $[0, R]$, el arco $A_R$ y
el rayo $\eu^{2\iu\pi/n}[0, R]$ invertido. En el interior se encuentra el
unipolar $p = \eu^{\iu\pi/n}$ de $\frac1{1+z^n}$, con
\omterm{def:b3:residues:singularities}{residuo}$\frac1{np^{n-1}} = \frac{p}{np^n} = -\frac pn$. encendido
el rayo de retorno, $z = \eu^{2\iu\pi/n}x$ da $z^n = x^n$ y
$\dd z = \eu^{2\iu\pi/n}\dd x$; el arco es
$O(R^{-n})\cdot O(R) \to 0$. Por lo tanto
\[
\bigl(1 - \eu^{2\iu\pi/n}\bigr)
\int_0^\infty\frac{\dd x}{1 + x^n}
= 2\iu\pi\Bigl(-\frac{\eu^{\iu\pi/n}}n\Bigr),
\quad\text{so}\quad
\int_0^\infty\frac{\dd x}{1+x^n}
= \frac{2\iu\pi}{n\,\bigl(\eu^{\iu\pi/n} -
\eu^{-\iu\pi/n}\bigr)} = \frac{\pi}{n\sin(\pi/n)} .
\]
$n = 2$: $\frac\pi{2\sin(\pi/2)} = \frac\pi2 =
[\arctan]_0^\infty$. Como$n \to \infty$: el valor tiende a
$1$, y de hecho el integrando tiende a $\mathbf 1_{\intco01}$
(con dominación DCT $\min(1, x^{-2})$ para $n \geq 2$).
\end{solution}

\begin{solution}{exo:b3:residues:10}
En $\abs z = R$ grande, $\abs f \leq C R^{-2}$: $\abs{\oint_{C_R}
f} \leq 2\pi R\cdot CR^{-2} \to 0$. Pero para$R$ más allá de todo
polos, el teorema \omterm{def:b3:residues:singularities}{residuo} da $\oint_{C_R}f =
2\iu\pi\sum_{\text{all }p}\operatorname{Res}(f, p)$: el total
la suma desaparece. Para $f = \frac1{z(z-1)(z-2)}$: \omterm{def:b3:residues:singularities}{residuos}
$\frac1{(-1)(-2)} = \frac12$ en $0$, $\frac1{1\cdot(-1)} = -1$
en $1$, $\frac1{2\cdot1} = \frac12$ en $2$ --- sumando a $0$
como se predijo, y
\[
\frac1{z(z-1)(z-2)} = \frac{1/2}{z} - \frac1{z - 1} +
\frac{1/2}{z-2} :
\]
el \omterm{def:b3:residues:singularities}{residuos} \emph{son} los coeficientes de fracción parcial, y
la identidad de suma cero proporciona una verificación de coherencia gratuita (o
determina el último coeficiente de los demás).
\end{solution}

\begin{solution}{exo:b3:residues:11}
En el ojo de la cerradura, con la determinación elegida:
justo encima del corte, $\log z = \ln x$; justo debajo, $\log z =
\ln x + 2\iu\pi$. Las cuatro piezas dan
\[
\Bigl(1 - \eu^{2\iu\pi(a-1)}\Bigr)\int_\varepsilon^R
\frac{x^{a-1}}{1+x}\dd x + \int_{C_R} + \int_{C_\varepsilon}
= 2\iu\pi\operatorname{Res}(f, -1) .
\]
Arcos: $\abs{f} \leq \frac{R^{a-1}}{R - 1}$ en $C_R$, longitud
$2\pi R$: contribución $O(R^{a-1}) \to 0$ ($a < 1$);
$\abs f \leq \frac{\varepsilon^{a-1}}{1 - \varepsilon}$ en
$C_\varepsilon$, longitud $2\pi\varepsilon$: $O(\varepsilon^a)
\to 0$($a > 0$). \omterm{def:b3:residues:singularities}{Residuo}: en$z = -1 = \eu^{\iu\pi}$,
$\operatorname{Res} = \eu^{(a-1)\iu\pi}$. Por lo tanto
\[
I(a) = \frac{2\iu\pi\,\eu^{\iu\pi(a-1)}}{1 -
\eu^{2\iu\pi(a-1)}}
= \frac{2\iu\pi}{\eu^{-\iu\pi(a-1)} - \eu^{\iu\pi(a-1)}}
= \frac{\pi}{-\sin(\pi(a-1))} = \frac{\pi}{\sin\pi a} .
\]
Reflexión gamma: $B(a, 1-a) =
\int_0^1t^{a-1}(1-t)^{-a}\dd t$; la sustitución$t =
\frac x{1+x}$,$1 - t = \frac1{1+x}$,$\dd t =
\frac{\dd x}{(1+x)^2}$lo convierte en$\int_0^\infty
\frac{x^{a-1}}{1+x}\dd x = I(a)$, y la fórmula de Euler
$B(a, 1-a) = \Gamma(a)\Gamma(1-a)/\Gamma(1)$
(\cref{pb:b3:lebesgue:1}) da
$\Gamma(a)\Gamma(1-a) = \frac\pi{\sin\pi a}$ --- en
en particular $\Gamma(\tfrac12) = \sqrt\pi$ una vez más.

\end{solution}

\begin{solution}{exo:b3:residues:12}

($\abs{8z} - 1 = 7$): $P$ tiene tantos ceros como $\mathbb D$

Los ceros exteriores se encuentran en $1 < \abs z < 2.1$. (Numéricamente: un real

\end{solution}

\begin{solution}{pb:b3:residues:1}
\textbf{1.} $\sin(\pi z)$ tiene ceros simples exactamente en $\Z$
($\sin\pi z = 0$ si $z \in \Z$, y $(\sin\pi z)' = \pi\cos\pi
z \neq 0$allí), y$\cos(\pi n) \neq 0$:$\pi\cot(\pi z) =
\pi\cos(\pi z)/\sin(\pi z)$tiene polos simples en$\Z$ con
\[
\operatorname{Res}(\pi\cot\pi z,\ n)
= \frac{\pi\cos(\pi n)}{\pi\cos(\pi n)} = 1
\]
(regla $g/h'$, \cref{met:b3:residues:compute}).

\textbf{2.} $\frac{\sin(\pi z)}{\pi z} = 1 - \frac{(\pi z)^2}6
+ \frac{(\pi z)^4}{120} - \cdots$ es \omterm{def:b3:holomorphic:holo}{holomorfo} y distinto de cero
cerca de $0$: su recíproco es \omterm{def:b3:holomorphic:holo}{holomorfo}
(\cref{def:b3:holomorphic:holo}: cociente), con serie $1 +
\frac{(\pi z)^2}{6} + \frac{7(\pi z)^4}{360} + \cdots$
(identifique: coeficientes de estilo $(1 - u)^{-1}$ de $u =
\frac{(\pi z)^2}6 - \frac{(\pi z)^4}{120}$: el$z^4$
el coeficiente es $\frac1{36} - \frac1{120} = \frac{7}{360}$).
Multiplica por $\cos(\pi z) = 1 - \frac{(\pi z)^2}2 + \frac{(\pi
z)^4}{24} - \cdots$y divide por$z$:
\[
\pi\cot(\pi z) = \frac1z\Bigl[1 + \pi^2z^2\Bigl(\frac16 -
\frac12\Bigr) + \pi^4z^4\Bigl(\frac7{360} - \frac1{12} +
\frac1{24}\Bigr)\Bigr] + \cdots
= \frac1z - \frac{\pi^2}3\,z - \frac{\pi^4}{45}\,z^3 -
\cdots
\]
($\frac7{360} - \frac{30}{360} + \frac{15}{360} =
-\frac8{360} = -\frac1{45}$).

\textbf{3.} Lados verticales $z = \pm(N + \frac12) + \iu y$:
por $\pi$-periodicidad de $\cot$, $\cot(\pi z) = \cot(\pm\frac\pi2
+ \iu\pi y) = -\tan(\iu\pi y) = -\iu\tanh(\pi y)$, de módulo
$\leq 1$. Lados horizontales $z = x \pm \iu(N + \frac12)$: desde
$\abs{\cot(a + \iu b)}^2 = \frac{\cos^2a +
\sinh^2b}{\sin^2a + \sinh^2b} \leq \frac{1 +
\sinh^2b}{\sinh^2b} = \coth^2 b$,
\[
\abs{\cot(\pi z)} \leq \coth\bigl(\pi(N + \tfrac12)\bigr)
\leq \coth(\pi/2) < 1.1 .
\]
Ambos límites son $\leq 2$.

\textbf{4.} Aplique \cref{thm:b3:residues:residue} a $F(z) =
\pi\cot(\pi z)f(z)$en$C_N$($N$más allá de todos los polos de$f$):
\[
\frac1{2\iu\pi}\oint_{C_N}F = \sum_{n=-N}^{N}f(n) +
\sum_p\operatorname{Res}(F, p),
\]
los polos enteros contribuyen a $f(n)$ (pregunta 1; $f$
\omterm{def:b3:holomorphic:holo}{holomorfo} allí). En $C_N$: $\abs F \leq 2\pi\cdot
C\abs z^{-2} \leq 2\pi C N^{-2}$, y el perímetro es$8(N +
\frac12)$: la integral es$O(1/N) \to 0$. Sea$N \to
\infty$: la fórmula de suma mostrada.

\textbf{5.} La descomposición $\abs f = O(\abs z^{-2})$ mató al
\omterm{def:b3:holomorphic:contour}{integral de contorno} \emph{y} hizo que $\sum\abs{f(n)}$ convergiera.
Para $f(z) = \frac1{z + \frac12}$: las sumas simétricas
Telescopio $\sum_{-N}^N\frac1{n + \frac12}$ a $0$ (los términos
$n$ y $-n - 1$ cancelan), y el lado derecho es
$-\operatorname{Res}\bigl(\frac{\pi\cot\pi z}{z + \frac12},
-\frac12\bigr) = -\pi\cot(-\frac\pi2) = 0$: consistente ---
pero $\sum\abs{f(n)}$ diverge; el método calcula el
límite simétrico (valor principal) únicamente.

\textbf{6.} $g(z) = \frac{\pi\cot(\pi z)}{z^2}$: polos en el
enteros distintos de cero con \omterm{def:b3:residues:singularities}{residuos} $\frac1{n^2}$ y en $0$
donde, por la pregunta 2,
\[
g(z) = \frac1{z^3} - \frac{\pi^2}{3z} - \frac{\pi^4}{45}z -
\cdots :
\qquad \operatorname{Res}(g, 0) = -\frac{\pi^2}3 .
\]
El \omterm{def:b3:holomorphic:contour}{integral de contorno} sobre $C_N$ tiende a $0$ como en la pregunta 4
($\abs{g} = O(N^{-2})$ en $C_N$). Por lo tanto $0 =
\sum_{n\neq0}\frac1{n^2} - \frac{\pi^2}3$:$2\zeta(2) =
\frac{\pi^2}3$,$\zeta(2) = \frac{\pi^2}6$.

\textbf{7.} $g(z) = \frac{\pi\cot(\pi z)}{z^4} = \frac1{z^5}
- \frac{\pi^2}{3z^3} - \frac{\pi^4}{45z} - \cdots$: \omterm{def:b3:residues:singularities}{residuo}
en $0$ igual a $-\frac{\pi^4}{45}$, y $0 =
2\zeta(4) - \frac{\pi^4}{45}$:$\zeta(4) =
\frac{\pi^4}{90}$.

\textbf{8.} Con $g = \pi\cot(\pi z)/z^{2k}$: el \omterm{def:b3:residues:singularities}{residuo} en
$0$ es el coeficiente $a_{2k-1}$ de $z^{2k-1}$ en el
expansión de $\pi\cot(\pi z)$, y el contorno que desaparece da
$2\zeta(2k) + a_{2k-1} = 0$: $\zeta(2k) = -a_{2k-1}/2$, un
múltiplo racional de $\pi^{2k}$ desde la cotangente
los coeficientes son. Un paso de división más produce $a_5 =
-\frac{2\pi^6}{945}$, de donde$\zeta(6) = \frac{\pi^6}{945}$.
Para $\zeta(3)$: el núcleo natural $g = \pi\cot(\pi z)/z^3$
produce $\sum_{n\neq0}\frac1{n^3} = 0$ por rareza --- el
El método prueba $0 = 0$ y es estructuralmente ciego a zeta impar.
valores (no se conoce ninguna forma cerrada para $\zeta(3)$; su
irracionalidad, Apéry 1978, necesitaba algo completamente diferente.
ideas).

\textbf{9.}$F(z) = \frac{\pi\cot(\pi z)}{(z - w)(z + w)}$
tiene polos en los números enteros (\omterm{def:b3:residues:singularities}{residuos} $\frac1{n^2 - w^2}$,
observando el letrero: $f(n) = \frac1{(n-w)(n+w)} = \frac1{n^2 -
w^2}$) y postes simples en$\pm w$ con \omterm{def:b3:residues:singularities}{residuos}
$\frac{\pi\cot(\pm\pi w)}{\pm2w} = \frac{\pi\cot(\pi
w)}{2w}$cada uno ($\cot$ es impar). Argumento de la pregunta 4
($\abs f = O(\abs z^{-2})$) da
\[
\sum_{n\in\Z}\frac1{n^2 - w^2} + \frac{\pi\cot(\pi w)}{w} =
0,
\qquad\text{i.e.}\qquad
\pi\cot(\pi w) = \frac1w + \sum_{n\geq1}\frac{2w}{w^2 -
n^2},
\]
(el término $n = 0$ es $-\frac1{w^2}$; reagrupe $\pm n$). normales
convergencia en \omterm{def:b3:topology:compact}{compactos} de $\C\setminus\Z$: para $\abs w \leq
R$y$n \geq 2R$,$\abs{\frac{2w}{w^2 - n^2}} \leq
\frac{2R}{n^2 - R^2} \leq \frac{8R}{3n^2}$.

\textbf{10.} Para $\abs w \leq r < 1$: $\frac{2w}{w^2 - n^2} =
-\frac{2w}{n^2}\cdot\frac1{1 - w^2/n^2} =
-2\sum_{k\geq0}\frac{w^{2k+1}}{n^{2k+2}}$, con
$\abs{\text{terms}} \leq 2r^{2k+1}/n^{2k+2}$, sumable terminado
$(n, k)$: Fubini para reordenamientos de serie
\[
\pi\cot(\pi w) = \frac1w -
2\sum_{k\geq0}\zeta(2k+2)\,w^{2k+1} .
\]
Coincidencia con la pregunta 2: $-2\zeta(2) = -\frac{\pi^2}3$ y
$-2\zeta(4) = -\frac{\pi^4}{45}$ --- los mismos valores. tres
caminos a $\frac{\pi^2}6$: \omterm{thm:b3:hilbert:parseval}{Parseval} (serie Fourier), el
rastro del operador verde de la cadena y los dos de la cotangente
expansiones; que una suma de frecuencias, una traza de operador,
y un acuerdo \omterm{def:b3:holomorphic:contour}{integral de contorno} no es un accidente --- cada uno es un
cara de la misma identidad espectral.

\textbf{11.} Arregla $R \geq 1$ y deja que $n_0$ sea el más pequeño.
entero $\geq 2R$. Para $\abs z \leq R$ y $n \geq n_0$:
$\abs{z^2/n^2} \leq \frac14$, entonces $1 - z^2/n^2 \in
\bar D(1, \frac14)$, donde el logaritmo principal es
\omterm{def:b3:holomorphic:holo}{holomorfo}, y
\[
\Bigl|\log\Bigl(1 - \frac{z^2}{n^2}\Bigr)\Bigr|
\leq \sum_{k\geq1}\frac1k\,\Bigl|\frac{z^2}{n^2}\Bigr|^k
\leq \frac{\abs{z^2/n^2}}{1 - \abs{z^2/n^2}}
\leq \frac{2R^2}{n^2} :
\]
la suma $S(z) = \sum_{n\geq n_0}\log(1 - z^2/n^2)$ converge
normalmente en $\bar D(0, R)$, con sumas parciales \omterm{def:b3:holomorphic:holo}{holomorfo}
$S_N$ y $\abs{S_N} \leq 2R^2\zeta(2)$ de manera uniforme. desde
$\abs{\eu^a - \eu^b} \leq
\eu^{\max(\abs a,\abs b)}\abs{a - b}$ (valor medio ligado a
el segmento), los productos de cola $\prod_{n_0\leq n\leq N} =
\eu^{S_N}$convergen uniformemente en$\bar D(0, R)$ al
$\eu^S$ libre de cero. Multiplicando por el polinomio fijo
$z\prod_{n<n_0}(1 - z^2/n^2)$: $P_N \to P$ uniformemente encendido
$\bar D(0, R)$, y
\cref{thm:b3:holomorphic:weierstrassconv} hace $P$
\omterm{def:b3:holomorphic:holo}{holomorfo} allí; Siendo $R$ arbitrario, $P$ está completo. encendido
$\bar D(0, R)$ los ceros de $P$ son los del polinomio
prefactor --- los números enteros del módulo $\leq R$, cada uno simple
($(1 - z/n)(1 + z/n)$ tiene ceros simples distintos, $\eu^S$
none): el conjunto de ceros de $P$ es $\Z$, todos ceros simples.

\textbf{12.} Diferenciación logarítmica de lo finito
producto, lejos de sus ceros:
\[
\frac{P_N'(z)}{P_N(z)} = \frac1z +
\sum_{n=1}^{N}\frac{-2z/n^2}{1 - z^2/n^2}
= \frac1z + \sum_{n=1}^{N}\frac{2z}{z^2 - n^2} .
\]
En un \omterm{def:b3:topology:compact}{compacto} $K \subseteq \C\setminus\Z$: $P_N \to P$ y
$P_N' \to P'$ uniformemente
(\cref{thm:b3:holomorphic:weierstrassconv}) y $\min_K
\abs P > 0$($P$desaparece sólo en$\Z$), por lo que eventualmente
$\abs{P_N} \geq \frac12\min_K\abs P$ y $P_N'/P_N \to
P'/P$uniformemente en$K$. El miembro del medio converge a
$\frac1z + \sum_{n\geq1}\frac{2z}{z^2-n^2} = \pi\cot(\pi z)$
por pregunta 9: de ahí $P'/P = \pi\cot(\pi z)$ en adelante
$\C\setminus\Z$.

\textbf{13.} $\sin(\pi z)$ y $P$ están completos con el mismo
puesta a cero $\Z$, todos ceros simples (preguntas 1 y 11). cerca
$m \in \Z$ escribe $\sin(\pi z) = (z - m)\,\sigma(z)$ y
$P(z) = (z - m)\,\psi(z)$ con $\sigma, \psi$ \omterm{def:b3:holomorphic:holo}{holomorfo}
y no desaparece en $m$ (factorice la serie de potencias): $Q =
\sin(\pi z)/P = \sigma/\psi$ extiende \omterm{def:b3:holomorphic:holo}{holomorficamente} y
libre de ceros en cada número entero y libre de ceros en
$\C\setminus\Z$ como cociente de funciones libres de cero. Allí,
\[
\frac{Q'}{Q} = \frac{(\sin\pi z)'}{\sin\pi z} -
\frac{P'}{P} = \pi\cot(\pi z) - \pi\cot(\pi z) = 0 ,
\]
entonces toda la función $Q'$ desaparece en $\C\setminus\Z$,
por lo tanto, en todas partes por \omterm{def:b3:topology:continuity}{continuidad}: $Q$ es constante. Como $z \to
0$:$\sin(\pi z)/z \to \pi$y$P(z)/z \to 1$, entonces$Q =
\pi$:
\[
\sin(\pi z) = \pi z\prod_{n\geq1}
\Bigl(1 - \frac{z^2}{n^2}\Bigr) .
\]

\textbf{14.} En $z = \frac12$: $1 = \sin\frac\pi2 =
\frac\pi2\prod_{n\geq1}\bigl(1 - \frac1{4n^2}\bigr) =
\frac\pi2\prod\frac{4n^2-1}{4n^2}$, entonces
\[
\frac\pi2 = \prod_{n\geq1}\frac{4n^2}{4n^2 - 1}
= \lim_{N\to\infty}\prod_{n=1}^N
\frac{(2n)(2n)}{(2n-1)(2n+1)}
= \lim_{N\to\infty}
\frac{2\cdot2\cdot4\cdot4\cdots(2N)(2N)}
{1\cdot3\cdot3\cdot5\cdots(2N-1)(2N+1)} :
\]
El producto de Wallis, un corolario unifilar del de Euler
factorización.

\textbf{15.} Para $\abs z \leq r < 1$ cada factor reside en
$D(1, r^2) \subseteq D(1, 1)$, entonces $P(z)/z = \exp\bigl(
\sum_{n\geq1}\log(1 - z^2/n^2)\bigr)$: cada producto parcial
es el exponencial de una suma parcial, y ambos lados pasan a
el límite por \omterm{def:b3:topology:continuity}{continuidad} de $\exp$. La doble serie
\[
\sum_{n\geq1}\log\Bigl(1 - \frac{z^2}{n^2}\Bigr)
= -\sum_{n\geq1}\sum_{k\geq1}\frac{z^{2k}}{k\,n^{2k}}
= -\sum_{k\geq1}\frac{\zeta(2k)}k\,z^{2k}
= -\zeta(2)\,z^2 + O(z^4)
\]
reorganiza por Fubini para la serie: $\sum_{n,k}
\frac{r^{2k}}{kn^{2k}} \leq \sum_k\zeta(2k)r^{2k} \leq
\zeta(2)\frac{r^2}{1-r^2} < \infty$. Por lo tanto
\[
P(z) = z\,\exp\bigl(-\zeta(2)z^2 + O(z^4)\bigr)
= z - \zeta(2)\,z^3 + O(z^5) ,
\]
y la pregunta 13 compara esto con $\sin(\pi z) = \pi z -
\frac{\pi^3}6z^3 + O(z^5)$:$\pi\zeta(2) = \frac{\pi^3}6$,
es decir,\ $\zeta(2) = \frac{\pi^2}6$. La cara aditiva
(pregunta 10) y la cara multiplicativa calculan lo mismo
número.

\textbf{16.} En un \omterm{def:b3:topology:compact}{compacto} $K \subseteq \C\setminus\Z$ el
sumas parciales $S_N = \frac1z + \sum_{n\leq N}\bigl(
\frac1{z-n} + \frac1{z+n}\bigr)$ (pregunta 9, términos
reagrupados como $\frac{2z}{z^2-n^2} = \frac1{z-n} +
\frac1{z+n}$) convergen uniformemente a$\pi\cot(\pi z)$, por lo que
\cref{thm:b3:holomorphic:weierstrassconv} da $S_N' \to
(\pi\cot\pi z)' = -\pi^2/\sin^2(\pi z)$uniformemente en$K$.
Desde $S_N' = -\sum_{\abs n\leq N}(z - n)^{-2}$:
\[
\frac{\pi^2}{\sin^2(\pi z)} =
\sum_{n\in\Z}\frac1{(z - n)^2} ,
\]
la convergencia normal en \omterm{def:b3:topology:compact}{compactos} de $\C\setminus\Z$
(términos $O(n^{-2})$).

\textbf{17.} En $z = \frac12$ el lado izquierdo es $\pi^2$; en
a la derecha, $(\frac12 - n)^2 = \frac{(2n-1)^2}4$ con $2n -
1$ejecutándose sobre todos los enteros impares exactamente una vez mientras se ejecuta$n$
sobre $\Z$:
\[
\pi^2 = \sum_{n\in\Z}\frac{4}{(2n-1)^2}
= 8\sum_{m\geq0}\frac1{(2m+1)^2},
\qquad
\sum_{m\geq0}\frac1{(2m+1)^2} = \frac{\pi^2}8 .
\]
División de paridad: $\zeta(2) = \frac{\pi^2}8 +
\sum_{n\geq1}\frac1{(2n)^2} = \frac{\pi^2}8 +
\frac{\zeta(2)}4$, por lo que$\frac34\zeta(2) = \frac{\pi^2}8$ y
$\zeta(2) = \frac{\pi^2}6$ una vez más.

\textbf{18.} Con $c = \cot\theta$ y $\cot(2\theta) =
\frac{c^2-1}{2c}$:$\cot\theta - 2\cot(2\theta) = c -
\frac{c^2-1}c = \frac1c = \tan\theta$. De ahí$\pi\tan(\pi z)
= \pi\cot(\pi z) - 2\pi\cot(2\pi z)$y la pregunta 9 en$z$.
y en $2z$ da
\[
\pi\cot(\pi z) = \frac1z +
\sum_{n\geq1}\frac{2z}{z^2 - n^2},
\qquad
2\pi\cot(2\pi z) = \frac1z +
\sum_{n\geq1}\frac{8z}{4z^2 - n^2} .
\]
Ambas series convergen absolutamente en cada $z$ fijo fuera del
polos, por lo que la diferencia puede reagruparse a voluntad: en el
segunda serie los términos pares $n = 2m$ dan $\frac{8z}{4z^2 -
4m^2} = \frac{2z}{z^2 - m^2}$ y cancelan la primera serie
enteramente, dejando
\[
\pi\tan(\pi z) = -\sum_{m\geq0}\frac{8z}{4z^2 - (2m+1)^2}
= \sum_{m\geq0}\frac{8z}{(2m+1)^2 - 4z^2} ,
\]
normalmente en \omterm{def:b3:topology:compact}{compactos} evitando $\frac12 + \Z$ (términos
$O(m^{-2})$).

\textbf{19.} Para $\abs z \leq r < \frac12$:
\[
\frac{8z}{(2m+1)^2 - 4z^2} = \frac{8z}{(2m+1)^2}
\sum_{k\geq0}\Bigl(\frac{4z^2}{(2m+1)^2}\Bigr)^{k},
\]
con $\sum_{m,k}8r\,(4r^2)^k(2m+1)^{-2k-2} < \infty$ desde
$4r^2 < 1$: Fubini reordena la suma doble en
\[
\pi\tan(\pi z) =
\sum_{k\geq0}8\cdot4^k\,\lambda(2k+2)\,z^{2k+1} .
\]
Contra $\pi\tan(\pi z) = \pi^2z + \frac{\pi^4}3z^3 +
O(z^5)$: el coeficiente de$z$da$8\lambda(2) = \pi^2$
--- pregunta 17 nuevamente --- y la de $z^3$ da
$32\lambda(4) = \frac{\pi^4}3$, es decir $\lambda(4) =
\frac{\pi^4}{96}$. Quitando los denominadores pares,
$\lambda(4) = \zeta(4) - 2^{-4}\zeta(4) =
\frac{15}{16}\zeta(4)$:$\zeta(4) = \frac{16}{15}\cdot
\frac{\pi^4}{96} = \frac{\pi^4}{90}$, pregunta coincidente 7.

\textbf{20.} $\cot\frac\theta2 - \cot\theta =
\frac{\cos\frac\theta2\,\sin\theta -
\cos\theta\,\sin\frac\theta2}{\sin\frac\theta2\,\sin\theta}
= \frac{\sin\frac\theta2}{\sin\frac\theta2\,\sin\theta} =
\frac1{\sin\theta}$, siendo el numerador$\sin(\theta -
\frac\theta2)$. Con$\theta = \pi z$, pregunta 9 en
$\frac z2$ lee $\pi\cot\frac{\pi z}2 = \frac2z +
\sum_{n\geq1}\frac{4z}{z^2 - 4n^2}$, entonces
\[
\frac{\pi}{\sin(\pi z)}
= \pi\cot\frac{\pi z}2 - \pi\cot(\pi z)
= \frac1z + \sum_{n\geq1}\frac{4z}{z^2 - 4n^2}
- \sum_{n\geq1}\frac{2z}{z^2 - n^2} .
\]
La convergencia absoluta permite la reagrupación de paridad: incluso $n
= 2m$ en la serie restada contribuye
$\frac{2z}{z^2-4m^2}$, dejando $+\frac{2z}{z^2-4m^2}$ de
la primera suma, mientras que el impar $n$ sobrevive con el signo $-$:
\[
\frac{\pi}{\sin(\pi z)} = \frac1z +
\sum_{n\geq1}(-1)^n\,\frac{2z}{z^2 - n^2} .
\]
Comprobación de señales: $\operatorname{Res}\bigl(\pi/\sin(\pi z),
n\bigr) = \pi/(\pi\cos\pi n) = (-1)^n$
(\cref{met:b3:residues:compute}) y $(-1)^n\frac{2z}{z^2 -
n^2} = \frac{(-1)^n}{z-n} + \frac{(-1)^n}{z+n}$ lleva
exactamente ese \omterm{def:b3:residues:singularities}{residuo} en $\pm n$.

\textbf{21.} Para $\abs z \leq r < 1$, expandiendo cada término como
en la pregunta 19 ($(-1)^n\frac{2z}{z^2-n^2} =
(-1)^{n-1}\frac{2z}{n^2}\sum_k\frac{z^{2k}}{n^{2k}}$) y
aplicando Fubini:
\[
\frac{\pi}{\sin(\pi z)} = \frac1z +
2\sum_{k\geq0}\eta(2k+2)\,z^{2k+1},
\qquad
\eta(s) = \sum_{n\geq1}\frac{(-1)^{n-1}}{n^s} .
\]
Lado de Taylor: $\sin(\pi z) = \pi z(1 - \frac{(\pi z)^2}6 +
O(z^4))$da$\frac\pi{\sin\pi z} = \frac1z +
\frac{\pi^2}6z + O(z^3)$:$2\eta(2) = \frac{\pi^2}6$, entonces
$\eta(2) = \frac{\pi^2}{12}$. Consistencia: $\eta(2) =
\zeta(2) - 2\sum_n(2n)^{-2} = (1 - 2^{1-2})\zeta(2) =
\frac{\zeta(2)}2 = \frac{\pi^2}{12}$.

\textbf{22.} En $z = \frac12$, la pregunta 20 da
\[
\pi = 2 + \sum_{n\geq1}(-1)^n\frac{1}{\frac14 - n^2}
= 2 + 4\sum_{n\geq1}\frac{(-1)^{n-1}}{(2n-1)(2n+1)} .
\]
Con $\frac1{(2n-1)(2n+1)} = \frac12\bigl(\frac1{2n-1} -
\frac1{2n+1}\bigr)$, y ambas series alternas convergentes
(criterio de Leibniz), la suma se divide en $\frac12\bigl[L -
(1 - L)\bigr] = L - \frac12$, donde$L = 1 - \frac13 +
\frac15 - \cdots$y$\frac13 - \frac15 + \frac17 - \cdots
= 1 - L$. Por lo tanto$\pi = 2 + 4(L - \frac12) = 4L$:
\[
\frac\pi4 = 1 - \frac13 + \frac15 - \frac17 + \cdots
\]
La moraleja: cada núcleo es \omterm{def:b3:residues:singularities}{meromórfico} con polos en un
progresión aritmética y prescrito \omterm{def:b3:residues:singularities}{residuos}. el
cotangente pone \omterm{def:b3:residues:singularities}{residuo} $1$ en cada número entero y suma
$f(n)$; su derivada cuadra los polos y los precios
$\lambda(2)$; la tangente mueve los polos a $\frac12 + \Z$
y valora los denominadores impares; $\pi/\sin$ mantiene el
polos enteros pero alterna el \omterm{def:b3:residues:singularities}{residuos} $(-1)^n$, por lo tanto
la serie alterna. Los núcleos de primer orden son todos
funciones impares: emparejar $n$ con $-n$ duplica la potencia par
coeficientes y aniquila los impares, por lo $\zeta(2k)$
se derrama mecánicamente mientras que $\zeta(3)$ nunca aparece. el
la ceguera es paridad, no falta de técnica.

\textbf{23.} Con $w = 2\iu\pi z$:
\[
\iu\pi z + \frac{2\iu\pi z}{\eu^{2\iu\pi z} - 1}
= \iu\pi z\,\frac{\eu^{2\iu\pi z} + 1}{\eu^{2\iu\pi z} - 1}
= \iu\pi z\,\frac{\eu^{\iu\pi z} +
\eu^{-\iu\pi z}}{\eu^{\iu\pi z} - \eu^{-\iu\pi z}}
= \pi z\,\frac{\cos\pi z}{\sin\pi z} = \pi z\cot(\pi z) .
\]
Por lo tanto, utilizando la función generadora en $w = 2\iu\pi z$ (y
$B_1 = -\frac12$ cancelando el término $\iu\pi z$, $B$ impares
desapareciendo más allá):
\[
\pi z\cot(\pi z) = \sum_{k\geq0}\frac{B_{2k}}{(2k)!}
(2\iu\pi z)^{2k}
= 1 + \sum_{k\geq1}(-1)^k\frac{(2\pi)^{2k}B_{2k}}{(2k)!}
z^{2k} .
\]
Por otro lado la expansión en fracción parcial (Parte IV)
da $\pi z\cot(\pi z) = 1 - 2\sum_{k\geq1}\zeta(2k)z^{2k}$
(expandir cada $\frac{2z^2}{z^2 - n^2} =
-2\sum_k\frac{z^{2k}}{n^{2k}}$y sumar sobre$n$, normal
convergencia que justifica el intercambio por $\abs z < 1$).
Comparando coeficientes: $-2\zeta(2k) =
(-1)^k\frac{(2\pi)^{2k}B_{2k}}{(2k)!}$, la fórmula indicada.

\textbf{25.} Escribe $C(z) = \pi z\cot(\pi z) = 1 -
2\sum_{k\geq1}\zeta(2k)z^{2k}$ (pregunta 23). directo
diferenciación de $C = \pi z\cot(\pi z)$, usando $(\cot u)'
= -1 - \cot^2u$:
\[
zC'(z) = \pi z\cot(\pi z) - \pi^2z^2\bigl(1 +
\cot^2(\pi z)\bigr) = C - \pi^2z^2 - C^2 .
\]
Ahora expanda ambos lados en potencias de $z^2$. Lado izquierdo:
$\sum_k(-4k)\,\zeta(2k)\,z^{2k}$. Lado derecho: elevando al cuadrado
serie,
\[
C - C^2 = 2\sum_{k\geq1}\zeta(2k)z^{2k}
- 4\sum_{k\geq2}\Bigl(\sum_{j=1}^{k-1}\zeta(2j)
\zeta(2k-2j)\Bigr)z^{2k},
\]
y el término $-\pi^2z^2 = -6\zeta(2)z^2$ solo ajusta $k =
1$. Comparando los coeficientes de$z^{2k}$para$k \geq 2$:
\[
-4k\,\zeta(2k) = 2\,\zeta(2k) -
4\sum_{j=1}^{k-1}\zeta(2j)\,\zeta(2k-2j),
\]
es decir,\ $\bigl(k + \frac12\bigr)\zeta(2k) =
\sum_{j=1}^{k-1}\zeta(2j)\zeta(2k-2j)$. (En$k = 1$ el
la identidad dice $-4\zeta(2) = 2\zeta(2) - 6\zeta(2)$: a
verificación de coherencia, no información nueva). Aplicaciones: $k =
2$:$\frac52\zeta(4) = \zeta(2)^2 = \frac{\pi^4}{36}$, por lo que
$\zeta(4) = \frac{\pi^4}{90}$; $k = 3$: $\frac72\zeta(6) =
2\zeta(2)\zeta(4) = \frac{\pi^6}{270}$, entonces$\zeta(6) =
\frac{2}{7}\cdot\frac{\pi^6}{270} = \frac{\pi^6}{945}$. uno
semilla trascendental ($\zeta(2) = \frac{\pi^2}6$), y pura
El álgebra genera cada valor zeta par.
