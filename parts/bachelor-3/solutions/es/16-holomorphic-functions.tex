\begin{solution}{exo:b3:holomorphic:1}

\begin{solution}{exo:b3:holomorphic:8}
Escribe $P = c\prod_k(X - a_k)^{m_k}$
(\cref{pb:b3:holomorphic:1}). Deje $P'(z) = 0$. Si $P(z) = 0$,
entonces $z$ es uno de los $a_k$: en el casco. De lo contrario, el
derivada logarítmica da
\[
0 = \frac{P'(z)}{P(z)} = \sum_k\frac{m_k}{z - a_k}
= \sum_k m_k\,\frac{\bar z - \bar a_k}{\abs{z - a_k}^2} ;
\]
conjugando, $\sum_kw_k(z - a_k) = 0$ con $w_k = m_k/\abs{z -
a_k}^2 > 0$:$z = \sum_k\frac{w_k}{\sum w}\,a_k$, un convexo
combinación de las raíces. Para $P = z^3 - 1$: enraíza el cubo
raíces de la unidad, $P' = 3z^2$ con doble raíz $0$ --- el
centroide del triángulo equilátero.
\end{solution}

\begin{solution}{exo:b3:holomorphic:9}
El cuadrado unitario cerrado $K = \{x + \iu y : 0 \leq x, y \leq
1\}$es \omterm{def:b3:topology:compact}{compacto}:$M = \sup_K\abs f < \infty$. Cada$z \in \C$
difiere de un punto de $K$ por un elemento de $\Z + \iu\Z$
(resta partes enteras), y $f$ es invariante bajo esos

funciones de la teoría clásica --- deben tener polos.
\end{solution}

\begin{solution}{pb:b3:holomorphic:1}
\textbf{1.} Para $\abs z \geq 1$:
\[
\abs{P(z)} \geq \abs z^n\Bigl(1 -
\frac{\abs{a_{n-1}}}{\abs z} - \dots -
\frac{\abs{a_0}}{\abs z^n}\Bigr)
\geq \abs z^n\Bigl(1 - \frac{A}{\abs z}\Bigr),
\qquad A = \sum_k\abs{a_k} :
\]
para $\abs z \geq R_0 = \max(1, 2A)$, $\abs{P(z)} \geq
\frac12\abs z^n \to \infty$.

\textbf{2.} Elija $R \geq R_0$ con $\frac12R^n \geq
\abs{P(0)}$. En el \omterm{def:b3:topology:compact}{compacto}$\bar D(0, R)$ el \omterm{def:b3:topology:continuity}{continuo}
$\abs P$ alcanza un mínimo, en algún momento $z_0$; afuera, $\abs P
\geq \frac12R^n \geq \abs{P(0)} \geq \abs{P(z_0)}$: el
el mínimo es global.

\textbf{3.} $Q(h) = P(z_0 + h)/P(z_0)$ es un polinomio en $h$
con $Q(0) = 1$; no es constante ($P$ lo es), por lo que algunos
el coeficiente más allá de la constante es distinto de cero: $Q(h) = 1 +
c_kh^k + h^{k+1}S(h)$con$k \geq 1$mínimo,$c_k \neq 0$,
$S \in \C[X]$.

\textbf{4.} Raíces: cualquier $w = \rho\eu^{\iu\varphi} \neq 0$ tiene
la $k$-ésima raíz $\rho^{1/k}\eu^{\iu\varphi/k}$, donde
$\rho^{1/k}$ existe según el teorema del valor intermedio aplicado
a $t \mapsto t^k$ en $\intco0\infty$ --- sin circularidad.
Elija $\omega$ con $\omega^k = -1/c_k$. entonces
\[
Q(t\omega) = 1 - t^k + t^{k+1}\,\omega^{k+1}S(t\omega),
\qquad
\abs{Q(t\omega)} \leq 1 - t^k + C\,t^{k+1}
\quad (0 < t \leq 1),
\]
con $C = \abs\omega^{k+1}\sup_{\abs h \leq
\abs\omega}\abs{S(h)}$(nota$1 - t^k \geq 0$en$\intcc01$).

\textbf{5.} Para $0 < t < \min(1, 1/C)$: $\abs{Q(t\omega)}
\leq 1 - t^k(1 - Ct) < 1$, es decir $\abs{P(z_0 + t\omega)} <
\abs{P(z_0)}$ --- contradice la minimalidad global. entonces
$P(z_0) = 0$: d'Alembert--la prueba de Argand es \omterm{def:b3:complete:complete}{completo}.

\textbf{6.} Versión Liouville: si $P$ nunca desaparece, $1/P$
está entero; por pregunta 1, $\abs{1/P} \leq 2R_0^{-n}$ afuera
$\bar D(0, R_0)$ y $1/P$ es \omterm{def:b3:topology:continuity}{continuo} en ese \omterm{def:b3:topology:compact}{compacto}
disco, por lo tanto limitado allí también: limitado entero, tan constante
(\cref{cor:b3:holomorphic:liouville}), haciendo que $P$ sea constante:
absurdo. Ingredientes: d'Alembert utiliza compacidad (existencia
del mínimo) y la existencia en forma polar de raíces $k$-ésima;
Liouville utiliza todo el aparato de Cauchy (Goursat ---
en sí mismo un argumento \omterm{def:b3:topology:compact}{compactos anidados} --- y el \omterm{thm:b3:holomorphic:analytic}{cauchy
estimaciones}) más la misma coercitividad. Compacidad de cerrado.
Los discos son el núcleo común \omterm{def:b3:rings:divisibility}{irreducible}.

\textbf{7.} Si es $\deg P \geq 1$, elija una raíz $\alpha$
(preguntas 5); dividir: $P = (X - \alpha)Q + P(\alpha) = (X -
\alpha)Q$, con$\deg Q = n - 1$; instalar. Agrupación igual
factores: $P = c\prod_i(X - \alpha_i)^{m_i}$, $\sum m_i = n$.

\textbf{8.} De verdad $P$: $P(\bar\alpha) =
\overline{P(\alpha)} = 0$, y las multiplicidades coinciden
(conjugar la factorización): las raíces no reales vienen en pares,
contribuyendo $(X - \alpha)(X - \bar\alpha) = X^2 -
2\operatorname{Re}(\alpha)X + \abs\alpha^2$, una cuadrática real
con discriminante $< 0$. De ahí la lista indicada de
\omterm{def:b3:rings:divisibility}{irreductibles}, y un polinomio real de grado impar, que tiene un
número par de raíces no reales, debe tener una real. directo
prueba: $P(x) \to \pm\infty$ como $x \to \pm\infty$ (grado impar,
coeficiente principal positivo, digamos), por lo que $P$ cambia de signo, y
se aplica el teorema del valor intermedio. Para $X^3 - X - 1$:
ambos argumentos dan la única raíz real $\approx 1.3247$
(y un par conjugado).

matriz, $\chi$ siempre dividiéndose. (ii) Sea $P \in \bar\Q[X]$
