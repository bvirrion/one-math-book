\section*{अध्याय \ref{ch:b3:fouriertransform} --- फूरियर
परिवर्तन}

\begin{solution}{exo:b3:fouriertransform:1}
$\widehat{\mathbf 1_{\intcc{-a}a}}(\xi) =
\int_{-a}^a\eu^{-\iu\xi x}\dd x = \frac{2\sin(a\xi)}{\xi}$
(कीमत $2a $ पर $0$).
$\widehat{\eu^{-a\abs x}}(\xi) = \int_0^\infty\eu^{-(a +
\iu\xi)x} + \eu^{-(a - \iu\xi)x}\,\dd x = \frac1{a + \iu\xi} +
\frac1{a - \iu\xi} = \frac{2a}{a^2 + \xi^2}$.
तंबू: $\max(0, 1 - \abs x) = \mathbf 1_{\intcc{-1/2}{1/2}} *
\mathbf 1_{\intcc{-1/2}{1/2}}$, तो इसका परिवर्तन है
$\bigl(\frac{2\sin(\xi/2)}\xi\bigr)^2 =
\bigl(\frac{\sin(\xi/2)}{\xi/2}\bigr)^2$.
अंतिम: $\frac{2a}{a^2+\xi^2} \in L^1$, तो उलटा
(\cref{thm:b3:fouriertransform:inversion}(2)) पर लागू किया गया
$\eu^{-a\abs x}$ वेरिएबल्स का नाम बदलने के बाद देता है,
\[
\widehat{\Bigl(\frac1{x^2 + a^2}\Bigr)}(\xi) =
\frac{\pi}{a}\,\eu^{-a\abs\xi} .
\]
\end{solution}

\begin{solution}{exo:b3:fouriertransform:2}
से \cref{prop:b3:fouriertransform:basic}:
$\widehat{f(\cdot - a)} = \eu^{-\iu a\xi}\hat f(\xi)$;
$\widehat{f\cos(b\cdot)} = \frac12\bigl(\hat f(\xi - b) + \hat
f(\xi + b)\bigr)$;
$\widehat{f(a\cdot + b)}(\xi) = \frac1a\,\eu^{\iu
b\xi/a}\,\hat f(\xi/a)$ ($ a > 0$);
$\widehat{\overline{f(-\cdot)}} = \overline{\hat f}$;
$\widehat{f * f} = \hat f^2$. गॉसियन पर
($\widehat{\eu^{-x^2}} = \sqrt\pi\eu^{-\xi^2/4}$) प्रत्येक नियम है
एक-पंक्ति जांच --- उदा.\ $\eu^{-(x-a)^2}$ परिवर्तन हुआ है
$\sqrt\pi\,\eu^{-\iu a\xi}\eu^{-\xi^2/4}$, जो प्रत्यक्ष है
गणना (\omterm{def:b3:complete:complete}{पूरा} वर्ग) पुष्टि करता है।
\end{solution}

\begin{solution}{exo:b3:fouriertransform:3}
(ए) $ h = \mathbf 1_{\intcc{-1}1}*\mathbf 1_{\intcc{-1}1}$ है
$\hat h = \bigl(\frac{2\sin\xi}\xi\bigr)^2 \in L^1$; उलटा
पर $ x = 0$, कहाँ $ h(0) = \lambda(\intcc{-1}1\cap\intcc{-1}1)
= 2$:
\[
2 = \frac1{2\pi}\int_\R\Bigl(\frac{2\sin\xi}\xi\Bigr)^2
\dd\xi
\ \Longrightarrow\
\int_\R\Bigl(\frac{\sin\xi}\xi\Bigr)^2\dd\xi = \pi ,
\]
के साथ संगत $\int_0^\infty\frac{\sin^2}{\xi^2} =
\frac\pi2$ (\cref{pb:b3:lebesgue:1}).

(बी) प्लैंचरेल के लिए $ f = \eu^{-\abs x}$: $\int\abs{\hat f}^2 =
2\pi\int\abs f^2$ पढ़ता $\int\frac{4\,\dd\xi}{(1 + \xi^2)^2} =
2\pi\int\eu^{-2\abs x}\dd x = 2\pi $:
$\int_\R\frac{\dd\xi}{(1+\xi^2)^2} = \frac\pi2$.
\end{solution}

\begin{solution}{exo:b3:fouriertransform:4}
(ए) $ g_t(x) = \frac1{2\sqrt{\pi t}}\eu^{-x^2/4t}$: द्वारा
\cref{ex:b3:fouriertransform:gaussian} साथ $ a = \frac1{4t}$,
$\hat g_t(\xi) = \frac1{2\sqrt{\pi t}}\sqrt{4\pi
t}\,\eu^{-t\xi^2} = \eu^{-t\xi^2}$.
(बी) $\widehat{g_t * g_s} = \eu^{-t\xi^2}\eu^{-s\xi^2} =
\widehat{g_{t+s}}$, और परिवर्तन इंजेक्शन पर है $ L^1$
(\cref{thm:b3:fouriertransform:inversion}(3)): $ g_t * g_s =
g_{t+s}$.
(सी) $\norm{g_t}_1 = 1$ (गाऊसी इंटीग्रल);
$\norm{g_t}_2^2 = \frac1{4\pi t}\int\eu^{-x^2/2t}\dd x =
\frac{\sqrt{2\pi t}}{4\pi t} = \frac1{\sqrt{8\pi t}}$.
\end{solution}

\begin{solution}{exo:b3:fouriertransform:5}
वास्तव में भी $ f $: $\hat f(\xi) = \int f\cos(\xi x)\dd x $ (द
साइन पार्ट कैंसिल): वास्तविक और सम। विषम: $\hat f(\xi) =
-\iu\int f\sin(\xi x)$: अजीब, पूर्णतः काल्पनिक. $\hat f(0) =

का विशिष्ट कार्य \cref{ch:b3:clt}: मापांक $\leq 1$
हर जगह, $= 1$ मूल पर.
\end{solution}

\begin{solution}{exo:b3:fouriertransform:6}
(i) इंजेक्टिविटी है
\cref{thm:b3:fouriertransform:inversion}(3); निरंतरता है
$\norm{\hat f}_\infty \leq \norm f_1$ (मानों के साथ
$\mathcal C_0$ रीमैन द्वारा--लेब्सग्यू); $\mathcal C_0$ \omterm{def:b3:topology:interior}{बंद} है
सुपर नॉर्म में (अनंत-पर-लुप्त होने की एकसमान सीमाएं)।
फ़ंक्शन अनंत पर लुप्त हो जाते हैं): बनच।

(ii) ए \omterm{def:b3:topology:continuity}{निरंतर} बानाच स्थानों के बीच आपत्ति है
\omterm{def:b3:topology:continuity}{निरंतर} श्लोक में (\cref{thm:b3:banach:openmapping}): वहां होंगे $ C $ साथ $\norm f_1 \leq C\norm{\hat f}_\infty $.

(iii) चलो $ f_n(x) = \frac{\sin x}x\cdot\frac{\sin(x/n)}{x/n}$:
दो का एक उत्पाद $ L^2$ कार्य, और $ O(x^{-2})$ पर
अनंत, तो $ f_n \in L^1\cap L^2$. तब से
$\bigl(\frac{\sin(ax)}{ax}\bigr)$ है $ L^2$-परिवर्तन
$\frac\pi a\mathbf 1_{\intcc{-a}a}$, उत्पाद सूत्र
$\widehat{gh} = \frac1{2\pi}\hat g * \hat h $ (के लिए मान्य $ g, h
\in L^2$ साथ $ gh \in L^1$; इसे श्वार्ट्ज फ़ंक्शंस पर जांचें
फ़ुबिनी और द्वारा विस्तार $ L^2$-दोनों पक्षों की निरंतरता के माध्यम से
प्लैंचरेल) देता है
\[
\hat f_n = \frac1{2\pi}\,\bigl(\pi\mathbf
1_{\intcc{-1}1}\bigr) * \bigl(\pi n\,\mathbf
1_{\intcc{-1/n}{1/n}}\bigr):
\]
ऊँचाई का एक समलम्ब चतुर्भुज $\frac{\pi n}2\cdot\frac2n = \pi $:
$\norm{\hat f_n}_\infty = \pi $ हरएक के लिए $ n $. लेकिन पर $[1,
n]$, $\frac{\sin(x/n)}{x/n} \geq \sin 1 > 0$, इसलिए
\[
\norm{f_n}_1 \geq \sin 1\int_1^n\frac{\abs{\sin x}}x\dd x
\geq c\ln n
\]
(आर्क-गिनती, जैसा कि \cref{thm:b3:banach:fourierdiverge}).
बाध्य $\norm{f_n}_1 \leq C\pi $ बड़े पैमाने पर विफल रहता है $ n $: नहीं
विशेषण. (छवि घनी है --- द्वारा
स्टोन--वीयरस्ट्रैस-प्रकार के तर्क---लेकिन उचित उप-स्थान
$\mathcal C_0$.)
\end{solution}

\begin{solution}{exo:b3:fouriertransform:7}
अगर $\hat f(\xi) = O(\abs\xi^{-k-1-\delta})$: तब $\xi^j\hat f
\in L^1$ के लिए $0 \leq j \leq k $ (\omterm{def:b3:lebesgue:l1}{समाकलनीय} द्वारा अनंत पर
क्षय, स्थानीय स्तर पर द्वारा निरंतरता का $\hat f $). उलटा
(\cref{thm:b3:fouriertransform:inversion}(2)) दर्शाता है $ f $
ए.ई.\ द्वारा $ x \mapsto \frac1{2\pi}\int\hat f(\xi)\eu^{\iu
x\xi}\dd\xi $, और अभिन्न के तहत भेदभाव
(प्रभुत्वधारी $\abs{\xi^j\hat f}$) यह प्रतिनिधि बनाता है
$\mathcal C^k $. इसके विपरीत $ f \in \mathcal C_c^k $:
पुनरावर्ती \cref{prop:b3:fouriertransform:basic}(4),
$(\iu\xi)^k\hat f = \widehat{f^{(k)}}$, इसलिए $\abs{\hat f} \leq
\norm{f^{(k)}}_1\abs\xi^{-k}$. तम्बू समारोह: \omterm{def:b3:topology:continuity}{निरंतर} साथ
कॉम्पैक्ट सहायता ($ k = 0$:परिवर्तन परिबद्ध), और इसका
परिवर्तन $\sim \xi^{-2} = O(\abs\xi^{-0-1-1})$ रिटर्न, द्वारा
पहली दिशा, ए $\mathcal C^0$ प्रतिनिधि --- दोनों
तेज: तम्बू नहीं है $\mathcal C^1$, और इसका परिवर्तन
से अधिक तेजी से क्षय नहीं होता $\xi^{-2}$.
\end{solution}

\begin{solution}{exo:b3:fouriertransform:8}
भागों द्वारा एकीकरण ($ f \in \mathcal S $; सीमा शर्तें
गायब हो जाना):
\[
1 = \int f^2 = \bigl[xf^2\bigr]_{-\infty}^{\infty} - \int
x\,(f^2)' = -2\int xff' \leq 2\,\norm{xf}_2\,\norm{f'}_2 .
\]
प्लांचरेल और $\widehat{f'} = \iu\xi\hat f $:
$\norm{f'}_2^2 = \frac1{2\pi}\int\xi^2\abs{\hat f}^2$. चुकता करना
प्रदर्शन:
\[
\frac14 \leq \norm{xf}_2^2\cdot\frac1{2\pi}
\int\xi^2\abs{\hat f}^2\dd\xi .
\]
कॉची-श्वार्ज़ में समानता के लिए समानता की आवश्यकता होती है: $ f' =
\lambda xf $ साथ $\lambda < 0$ (अभिन्नता), यानी\ $ f(x) =
c\,\eu^{\lambda x^2/2}$: गॉसियन। एक संकेत \omterm{ex:b3:groups:actions}{केंद्र}ित है
$ x $ (छोटा $\norm{xf}_2$) फैला हुआ स्पेक्ट्रम होना चाहिए, और
इसके विपरीत: अनिश्चितता सिद्धांत.
\end{solution}

\begin{solution}{exo:b3:fouriertransform:10}
$\mathcal Ff \in L^2(\intcc{-\pi}\pi) \subseteq
L^1(\intcc{-\pi}\pi)$ (परिमित \omterm{def:b3:measure:measure}{उपाय}), इसलिए उलटा देता है
\omterm{def:b3:topology:continuity}{निरंतर} प्रतिनिधि $ f(x) =
\frac1{2\pi}\int_{-\pi}^\pi\mathcal Ff(\xi)\eu^{\iu
x\xi}\dd\xi $, साथ
\[
f(n) = \frac1{2\pi}\int_{-\pi}^{\pi}\mathcal
Ff(\xi)\,\eu^{\iu n\xi}\dd\xi = \langle e_{-n}, \mathcal
Ff\rangle
\]
के अंकन में \cref{thm:b3:hilbert:fourier}. में विस्तार हो रहा है
वह \omterm{def:b3:hilbert:onb}{हिल्बर्ट आधार}: $\mathcal Ff = \sum_nf(n)\,\eu^{-\iu
n\xi}$ में $ L^2(\intcc{-\pi}\pi)$. लागू करें
$ L^2$-निरंतर $\mathcal F^{-1}$ पद दर पद:
\[
\mathcal F^{-1}\bigl(\mathbf
1_{\intcc{-\pi}\pi}\eu^{-\iu n\xi}\bigr)(x)
= \frac1{2\pi}\int_{-\pi}^{\pi}\eu^{\iu\xi(x - n)}\dd\xi
= \frac{\sin\bigl(\pi(x-n)\bigr)}{\pi(x - n)} ,
\]
दे रही है $ f = \sum_nf(n)\operatorname{sinc}(\cdot - n)$ में
$ L^2$: एक बैंड-सीमित सिग्नल उसके पूर्णांक द्वारा निर्धारित होता है
नमूने --- शैनन का नमूनाकरण प्रमेय।
\end{solution}

\begin{solution}{exo:b3:fouriertransform:11}
(ए) उलटा चालू $\mathcal S $: $ f(x) =
\frac1{2\pi}\int\hat f(\xi)\eu^{\iu x\xi}\dd\xi =
\frac1{2\pi}(\mathcal F\hat f)(-x)$, यानी\ $(\mathcal
F^2f)(x) = 2\pi f(-x)$. दो बार आवेदन करना: $\mathcal F^4f =
2\pi\,\mathcal F^2f(-\cdot) = (2\pi)^2f $.

(बी) यदि $\mathcal Ff = \lambda f $ साथ $ f \neq 0$:
$(2\pi)^2f = \mathcal F^4f = \lambda^4f $, इसलिए $\lambda^4 =
(2\pi)^2$: $\lambda \in \{\pm\sqrt{2\pi},
\pm\iu\sqrt{2\pi}\}$. गाऊसी $ g(x) = \eu^{-x^2/2}$ है
$\hat g = \sqrt{2\pi}\,g $
(\cref{ex:b3:fouriertransform:gaussian} पर $ a = \frac12$):
eigenfunction के लिए $+\sqrt{2\pi}$.

(सी) $\mathcal F^2f = 2\pi f(-\cdot)$ के बराबर होती है $\pm2\pi f $
समता के अनुसार. के लिए $ h(x) = x\eu^{-x^2/2}$:
विभेद करना $\hat g(\xi) = \sqrt{2\pi}\eu^{-\xi^2/2}$
नियम के साथ $\widehat{xf} = \iu\frac{\dd}{\dd\xi}\hat f $:
\[
\hat h(\xi) = \iu\,\frac{\dd}{\dd\xi}\bigl(\sqrt{2\pi}
\eu^{-\xi^2/2}\bigr) = -\iu\sqrt{2\pi}\,\xi\eu^{-\xi^2/2}
= -\iu\sqrt{2\pi}\,h(\xi) :
\]
के लिए एक eigenfunction $-\iu\sqrt{2\pi}$. (द हर्मिट
फ़ंक्शन पैटर्न को जारी रखते हैं, चारों के माध्यम से साइकिल चलाते हैं
eigenvalues --- असतत फूरियर घड़ी।)

\end{solution}

\begin{solution}{exo:b3:fouriertransform:12}
(ए) $\tilde f \in L^2$ साथ $\norm{\tilde f}_2 = \norm f_2$;
\cref{exo:b3:lp:6} (संयुग्म घातांक $ p = q = 2$) बनाता है
$ A_f = f * \tilde f $ बंधा हुआ और समान रूप से \omterm{def:b3:topology:continuity}{निरंतर}, साथ
\[
A_f(x) = \int f(y)\,\overline{f(y - x)}\,\dd y,
\qquad A_f(0) = \norm f_2^2,
\qquad \abs{A_f(x)} \leq \norm f_2\,\norm{f(\cdot -
x)}_2 = A_f(0)
\]
कॉची-श्वार्ज़ द्वारा।

(बी) के लिए $ f \in L^1\cap L^2$: $\tilde f \in L^1$ भी, और
कनवल्शन प्रमेय देता है $\widehat{A_f} = \hat f\,
\widehat{\tilde f}$; कंप्यूटिंग, $\widehat{\tilde f}(\xi) =
\int\overline{f(-x)}\eu^{-\iu\xi x}\dd x =
\overline{\int f(u)\eu^{-\iu\xi u}\dd u} =
\overline{\hat f(\xi)}$: $\widehat{A_f} = \abs{\hat f}^2
\geq 0$.

(सी) कब $\abs{\hat f}^2 \in L^1$, व्युत्क्रमण लागू होता है
\omterm{def:b3:topology:continuity}{निरंतर} $ A_f $ (इसका रूपांतरण है \omterm{def:b3:lebesgue:l1}{समाकलनीय};
\cref{thm:b3:fouriertransform:inversion}):
\[
A_f(x) = \frac1{2\pi}\int\abs{\hat f(\xi)}^2
\eu^{\iu x\xi}\,\dd\xi .
\]
के लिए $ f = \mathbf 1_{\intcc{-1/2}{1/2}}$: $\hat f(\xi) =
\frac{2\sin(\xi/2)}\xi = \frac{\sin(\xi/2)}{\xi/2}$, और कम से
$ x = 0$:
\[
1 = \norm f_2^2 = \frac1{2\pi}\int_\R
\Bigl(\frac{\sin(\xi/2)}{\xi/2}\Bigr)^2\dd\xi
= \frac1{2\pi}\cdot2\int_\R\Bigl(\frac{\sin u}u\Bigr)^2\dd u
\]
($\xi = 2u $), यानी\ $\int_\R\bigl(\frac{\sin u}u\bigr)^2\dd
u = \pi $ --- प्लांचरेल का पसंदीदा इंटीग्रल, द्वारा पुनर्प्राप्त
स्वसहसंबंध.
\end{solution}

\begin{solution}{pb:b3:fouriertransform:1}
\textbf{1.} में समीकरण बदलना $ x $ (औपचारिक रूप से):
$\partial_t\hat u(t,\xi) = \widehat{\partial^2_{xx}u} =
(\iu\xi)^2\hat u = -\xi^2\hat u $, में एक ODE $ t $ प्रत्येक के लिए
आवृत्ति: $\hat u(t,\xi) = \eu^{-t\xi^2}\hat f(\xi)$. तब से
$\eu^{-t\xi^2} = \hat g_t $
(\cref{exo:b3:fouriertransform:4}), उत्पाद है
का परिवर्तन $ g_t * f $.

\textbf{2.} $\abs{u(t,x)} \leq \norm f_\infty\int g_t = \norm
f_\infty $: अच्छी तरह से परिभाषित. पर $[t_0, T]\times[-A, A]$: प्रत्येक
मिश्रित व्युत्पन्न $\partial^m_t\partial^n_xg_t(x - y)$ एक है
बहुपद में $(x - y)$ और $ t^{-1}$ टाइम्स
$\eu^{-(x-y)^2/4t}$, के लिए बाध्य $\abs y \geq 2A $ द्वारा
$ C\,(1 + y^2)^N\eu^{-(\abs y - A)^2/4T}$, एक \omterm{def:b3:lebesgue:l1}{समाकलनीय}
प्रभुत्व से स्वतंत्र $(t, x)$ खिड़की में (और घिरा हुआ
के लिए $\abs y \leq 2A $): के अंतर्गत बार-बार भेदभाव
अभिन्न (\cref{thm:b3:lebesgue:paramdiff}) लागू होता है: $ u \in
\mathcal C^\infty(\intoo0\infty\times\R)$.

\textbf{3.} साथ $ g_t(x) = \frac1{2\sqrt{\pi
t}}\eu^{-x^2/4t}$:
\[
\partial_tg_t = g_t\Bigl(\frac{x^2}{4t^2} - \frac1{2t}\Bigr)
= \partial^2_{xx}g_t
\]
(में दो बार अंतर करें $ x $: $\partial_xg_t = -\frac
x{2t}g_t $, $\partial^2_{xx}g_t = \bigl(\frac{x^2}{4t^2} -
\frac1{2t}\bigr)g_t $). प्रश्न 2 से व्युत्पन्न पास हो जाते हैं
अभिन्न के अंतर्गत: $\partial_tu = \partial^2_{xx}u $.

\textbf{4.} $ u(t,x) - f(x) = \int g_t(y)\bigl(f(x - y) -
f(x)\bigr)\dd y $. दिया गया ए कॉम्पैक्ट $ K $ और $\varepsilon $:
वर्दी निरंतरता का $ f $ एक पर \omterm{def:b3:topology:topology}{अड़ोस-पड़ोस} का $ K $ देता है
$\delta $ साथ $\abs{f(x-y) - f(x)} < \varepsilon $ के लिए $ x \in
K $, $\abs y \leq \delta $; पूँछ योगदान देती है $\leq 2\norm
f_\infty\int_{\abs y > \delta}g_t(y)\dd y = 2\norm
f_\infty\,\P $-मास परे $\delta $, जो है
$\frac2{\sqrt\pi}\int_{\delta/2\sqrt t}^\infty\eu^{-z^2}\dd z
\to 0$ जैसा $ t \to 0$. के लिए $ f \in L^p $: $\norm{u(t) - f}_p \leq
\int g_t(y)\norm{\tau_yf - f}_p\dd y $ (मिन्कोव्स्की/जेन्सेन जैसा कि
\cref{thm:b3:lp:regularization}), का उपयोग करके उसी तरह विभाजित करें
\cref{thm:b3:lp:density}(3).

\textbf{5.} त्वरित स्मूथिंग प्रश्न 2 है ($ u(t)$ है
$\mathcal C^\infty $ के लिए $ t > 0$ बिना किसी चिकनाई के $ f $
इस्तेमाल किया गया)। के लिए $ f = \mathbf 1_{\intoo0\infty}$:
\[
u(t, x) = \int_0^\infty g_t(x - y)\dd y
= \frac1{\sqrt\pi}\int_{-x/2\sqrt t}^{\infty}\eu^{-z^2}\dd z
= \frac12\Bigl(1 +
\operatorname{erf}\Bigl(\frac{x}{2\sqrt t}\Bigr)\Bigr),
\qquad \operatorname{erf}(s) =
\frac2{\sqrt\pi}\int_0^s\eu^{-z^2}\dd z :
\]
एक सुचारु कदम जिसका संक्रमण क्षेत्र चौड़ा होता जाता है $\sqrt t $
(प्रोफाइल पर $ t_1 < t_2 < t_3$: कभी चापलूसी रैंप के माध्यम से
$(0, \frac12)$).

\textbf{6.} एकात्म $ g_t(x-y)f(y)$ है $\geq 0$ और
कर्नेल सख्ती से सकारात्मक है: $ u(t,x) = 0$ जबरदस्ती करेगा $ f = 0$
ए.ई. में एकरसता $ f $ अभिन्न की एकरसता है. ए
स्थान कहाँ $ f = 0$ एक अंतराल पर अभी भी है $ u(t, \cdot) > 0$
वहाँ हर एक के लिए $ t > 0$: ऊष्मा अनंत गति से फैलती है
(कहीं भी कोई भी सकारात्मकता तुरंत हर जगह महसूस होती है)।

\textbf{7.} टोनेली ($ g_t(x-y)\abs{f(y)}$ है \omterm{def:b3:lebesgue:l1}{समाकलनीय} पर
$\R^2$): $\int u(t,x)\dd x = \int f(y)\bigl(\int g_t(x -
y)\dd x\bigr)\dd y = \int f $.

\textbf{8.} प्लैंचरेल: $2\pi\norm{u(t)}_2^2 =
\int\eu^{-2t\xi^2}\abs{\hat f(\xi)}^2\dd\xi $, में वृद्धि न होना
$ t $ (बिंदुवार), कड़ाई से जब तक $\hat f = 0$ ए.ई.\ ($= f =
0$), सीमा के साथ $0$ जैसा $ t\to\infty $ डीसीटी द्वारा. और
$\norm{u(t)}_\infty \leq \norm{g_t}_\infty\norm f_1 =
\frac{\norm f_1}{2\sqrt{\pi t}} \to 0$.

\textbf{9.} के लिए $\varphi \in \mathcal C_c^\infty $, $ t \mapsto
\langle\varphi, \hat u(t)\rangle $ है $\mathcal C^1$ के साथ
व्युत्पन्न $\langle\varphi, \partial_t\hat u\rangle =
\langle\varphi, \widehat{\partial_{xx}u}\rangle =
\langle\xi^2\varphi\dots\rangle $ --- एकदम सही,
$\widehat{\partial^2_{xx}u} = -\xi^2\hat u $ स्थानांतरित करता है
समीकरण. फिर ए.ई. के लिए\ $\xi $, बिलकुल \omterm{def:b3:topology:continuity}{निरंतर}
समारोह $ t \mapsto \eu^{t\xi^2}\hat u(t,\xi)$ व्युत्पन्न है
$\eu^{t\xi^2}(\xi^2\hat u + \partial_t\hat u) = 0$ में
एकीकृत भाव: यह स्थिर है, और दे रहा है $ t \to 0$
($\hat u(t) \to \hat f $ में $ L^2$, ए.ई.\ एक अनुवर्ती के साथ):
$\hat u(t, \xi) = \eu^{-t\xi^2}\hat f(\xi)$ ए.ई. दो
\omterm{def:b3:groups:action}{कक्षा} में समाधानों में समान परिवर्तन होता है: वे हैं
बराबर.

\textbf{10.} $ f = g_s * h $ साथ $ h \in L^2$ बल $\hat f =
\eu^{-s\xi^2}\hat h $, यानी\ $\hat h = \eu^{s\xi^2}\hat f \in
L^2$. लेना $\hat f(\xi) = \eu^{-\abs\xi}$: तब $ f(x) =
\frac1\pi\cdot\frac1{1 + x^2}$
(\cref{exo:b3:fouriertransform:1}, उलटा), एक पूर्णतः
चिकना $ L^2$ समारोह; लेकिन $\eu^{2s\xi^2 - 2\abs\xi} \to
\infty $: $\eu^{s\xi^2}\hat f \notin L^2$ हरएक के लिए $ s > 0$.
कॉची प्रोफ़ाइल है \emph{कभी नहीं} पूर्व का परिणाम
प्रसार.

\textbf{11.} ताप अर्धसमूह गुणनफल द्वारा रूपांतरित होता है
$\eu^{-t\xi^2}$, जो कहीं गायब नहीं होता: इंजेक्शन---
औपचारिक रूप से, कोई भी जानकारी नष्ट नहीं की जाती है। लेकिन इसकी सीमा शामिल है
ऐसे फ़ंक्शंस जिनके परिवर्तन क्षय की तरह होते हैं $\eu^{-t\xi^2}$: ए
का छोटा, घना-लेकिन-उचित उपस्थान $ L^2$ (प्रश्न 10 दिखाता है
यहाँ तक कि उत्कृष्ट कार्य भी बाहर पड़े रहते हैं)। उलटा होगा
आवृत्ति बढ़ाएँ $\xi $ द्वारा $\eu^{t\xi^2}$: असीमित,
इसलिए किसी भी गड़बड़ी के प्रति अस्थिर है। प्रसार है
अपरिवर्तनीय इसलिए नहीं कि नक्शा भूल जाता है, बल्कि इसलिए कि वह भूल जाता है
उलटा नहीं हो सकता \omterm{def:b3:topology:continuity}{निरंतर} ---समय का एक तीर बना हुआ
कार्यात्मक विश्लेषण.

\textbf{12.} $\hat f \in L^2(\intcc{-\Omega}\Omega)
\subseteq L^1$ (कॉची-श्वार्ज़ एक सीमित अंतराल पर), इसलिए
$ F(x) = \frac1{2\pi}\int_{-\Omega}^\Omega\hat
f(\xi)\eu^{\iu x\xi}\dd\xi $ हर जगह परिभाषित किया गया है, और
अभिन्न के तहत भेदभाव (द्वारा प्रभुत्व)
$\Omega^k\abs{\hat f} \in L^1$ बैंड पर) इसे बनाता है
$\mathcal C^\infty $ साथ $\abs{F^{(k)}} \leq
\frac{\Omega^k}{2\pi}\norm{\hat f}_{L^1}$ हर जगह. और $ F
= f $ ए.ई.: दोनों पक्षों में समान परिवर्तन होता है, और
परिवर्तन इंजेक्शन पर है $ L^2$
(\cref{thm:b3:fouriertransform:plancherel} और इसके
$ L^2$ विस्तार)।

\textbf{13.} घातांक $\xi \mapsto \eu^{-\iu
n\pi\xi/\Omega}$, $ n \in \Z $, फॉर्म ए \omterm{def:b3:hilbert:onb}{हिल्बर्ट आधार} का

= \frac{2\pi}{2\Omega}\cdot
\frac1{2\pi}\int_{-\Omega}^{\Omega}\hat f(\xi)\,
\eu^{\iu(n\pi/\Omega)\xi}\dd\xi
= \frac\pi\Omega\,f\Bigl(\frac{n\pi}\Omega\Bigr),

प्रश्न 12 के सूत्र द्वारा $ x = \frac{n\pi}\Omega $: द

\textbf{14.} व्युत्क्रम सूत्र में विस्तार सम्मिलित करें

\textbf{19.} बढ़ाना $\eu^{-\iu\xi x} =
\sum_k\frac{(-\iu\xi x)^k}{k!}$ अभिन्न के अंदर; पर
$\intcc{-A}A $ श्रृंखला \omterm{def:b3:groups:normal}{सामान्य} रूप से अभिसरण करती है
($\sum_k\frac{\abs{\xi}^kA^k}{k!}\abs f \in L^1$), तो
अवधि दर अवधि एकीकरण वैध है:
\[
\hat f(\xi) = \sum_{k\geq0}\frac{(-\iu\xi)^k}{k!}m_k,
\qquad \abs{m_k} \leq A^k\norm f_1 .
\]
बाउंड प्रत्येक कॉम्प्लेक्स के लिए श्रृंखला को अभिसरण बनाता है $\xi $;
किसी भी बिंदु के आसपास $\xi_0$, पुनर्समूहन (पूर्ण अभिसरण)
में एक शक्ति श्रृंखला देता है $\xi - \xi_0$: $\hat f $ है
हर जगह अनंत त्रिज्या के साथ वास्तविक-विश्लेषणात्मक।

\textbf{20.} होने देना $ g $ पर वास्तविक-विश्लेषणात्मक बनें $\R $ (टेलर श्रृंखला
में एकत्रित होना $ g $ प्रत्येक बिंदु के पास) और $ Z = \{\xi :
g^{(k)}(\xi) = 0\ \forall k\}$. $ Z $ \omterm{def:b3:topology:interior}{बंद} है (चौराहा
\omterm{def:b3:topology:interior}{बंद} सेटों का); यह तब से खुला है $\xi_0 \in Z $ स्थानीय
टेलर का विस्तार $ g $ शून्य श्रृंखला है, इसलिए $ g $ गायब हो जाता है
समान रूप से निकट $\xi_0$, सभी डेरिवेटिव के साथ। अगर
$ g $ एक अंतराल पर गायब हो जाता है, $ Z \neq \varnothing $; द्वारा
संयुक्तता का $\R $, $ Z = \R $: $ g \equiv 0$. अब यदि $ f
\neq 0$ था कॉम्पैक्ट साथ मिलकर समर्थन करें $\hat f $: प्रश्न
19 बनाता है $\hat f $ वास्तविक-विश्लेषणात्मक, बाहर लुप्त हो जाना कॉम्पैक्ट,
इसलिए अंतराल पर: $\hat f \equiv 0$, इसलिए $ f = 0$ ए.ई.\ द्वारा
इंजेक्शनिविटी--विरोधाभास। इसी प्रकार एक अशून्य $ f \in
PW_\Omega $ सघन रूप से समर्थित नहीं किया जा सकता (भूमिकाओं की अदला-बदली करें)।
$ f $ और $\hat f $ व्युत्क्रम के माध्यम से): बैंड-सीमित सिग्नल कभी नहीं
मरो, समय-सीमित सिग्नल असीमित स्पेक्ट्रम पर कब्जा कर लेते हैं।

\textbf{21.} के लिए $ f = \eu^{-ax^2}$: $\norm f_2^2 =
\sqrt{\frac\pi{2a}}$ और $\int x^2\abs f^2 =
\frac1{4a}\sqrt{\frac{\pi}{2a}}$ (गाऊसी दूसरा क्षण);
$\hat f = \sqrt{\frac\pi a}\,\eu^{-\xi^2/4a}$
(\cref{ex:b3:fouriertransform:gaussian}) और
\[
\frac1{2\pi}\int\xi^2\abs{\hat f}^2\dd\xi =
\frac1{2\pi}\cdot\frac\pi a\int\xi^2
\eu^{-\xi^2/2a}\dd\xi = \frac1{2a}\cdot a\sqrt{2\pi a}
= \frac{\sqrt{2\pi a}}2 .
\]
\omterm{def:b3:groups:normal}{सामान्य}ीकृत उत्पाद: $\frac1{4a}\sqrt{\frac\pi{2a}}\cdot
\frac{\sqrt{2\pi a}}2\big/\frac{\pi}{2a} = \frac14$,
से स्वतंत्र $ a $. स्केलिंग स्थिरता की व्याख्या करती है: प्रतिस्थापित करना
$ f $ द्वारा $ f(\lambda\cdot)$ गुणा $\int x^2\abs f^2/\norm
f^2$ द्वारा $\lambda^{-2}$ और $\frac1{2\pi}\int\xi^2\abs{\hat
f}^2/\norm f^2$ द्वारा $\lambda^{2}$: उत्पाद एक फैलाव है
अपरिवर्तनीय, और गॉसियन एक फैलाव बनाते हैं \omterm{def:b3:groups:action}{कक्षा}.

\textbf{22.} साथ $\abs f^2$ स्थिति \omterm{ex:b3:lebesgue:gamma}{घनत्व} और
$\frac1{2\pi}\abs{\hat f}^2$ गति \omterm{ex:b3:lebesgue:gamma}{घनत्व} एक का
क्वांटम अवस्था (भौतिक इकाइयाँ सम्मिलित करें $\hbar $),
\cref{exo:b3:fouriertransform:8} पढ़ता $\sigma_x\sigma_p
\geq \frac\hbar2$: दोनों अवलोकनों में कोई भी स्थिति तीव्र नहीं है।
पूरे अध्याय में, एक कानून पांच सूट पहनता है: तुरंत गर्मी
चिकना क्योंकि $\eu^{-t\xi^2}$ उच्च आवृत्तियों को नष्ट कर देता है
(भाग II); पुनर्स्थापना के कारण प्रवाह पीछे की ओर नहीं चल सकता
वे असीमित हैं (भाग IV); एक बैंड-सीमित सिग्नल कठोर होता है
गणनीय ग्रिड पर रहने के लिए पर्याप्त (भाग V); कोई कार्य नहीं
हाइजेनबर्ग के फर्श को हराया; और कोई भी कार्य सघन रूप से नहीं होता है
परिवर्तन के दोनों पक्षों द्वारा समर्थित (प्रश्न 19--20)।
क्या $\hat f $ अनन्त पर क्या नियंत्रित करता है $ f $ कर सकते हैं
कहीं भी.

\textbf{23.} दोनों $ g_t $ और $ g_s $ में हैं $ L^1$ साथ
$\widehat{g_t}(\xi) = \eu^{-t\xi^2}$ (प्रश्न 1
संगणना), इसलिए कनवल्शन प्रमेय देता है
$\widehat{g_t * g_s} = \eu^{-t\xi^2}\eu^{-s\xi^2} =
\eu^{-(t+s)\xi^2} = \widehat{g_{t+s}}$; दो $ L^1$ कार्य
उसी परिवर्तन से सहमत ए.ई.\ (इंजेक्शन, के माध्यम से
व्युत्क्रम प्रमेय --- यहाँ दोनों पक्ष हैं \omterm{def:b3:topology:continuity}{निरंतर}, तो
वे हर जगह सहमत हैं): $ g_t * g_s = g_{t+s}$. फलस्वरूप
$ u(t + s) = g_{t+s} * f = g_s * (g_t * f) = g_s * u(t)$
(कन्वोल्यूशन की संबद्धता, टोनेली)। लॉग-उत्तलता: चलो
$ N(t) = \norm{u(t)}_2^2 = \frac1{2\pi}\int
\eu^{-2t\xi^2}\abs{\hat f}^2\dd\xi $ (प्लान्चेरेल, प्रश्न
8). के लिए $ t = \frac{t_1 + t_2}2$, लिखना
\[
\eu^{-2t\xi^2}\abs{\hat f}^2
= \Bigl(\eu^{-2t_1\xi^2}\abs{\hat f}^2\Bigr)^{1/2}
\Bigl(\eu^{-2t_2\xi^2}\abs{\hat f}^2\Bigr)^{1/2},
\]
और कॉची-श्वार्ज़ देता है $ N\bigl(\frac{t_1+t_2}2\bigr) \leq
\sqrt{N(t_1)\,N(t_2)}$: $\ln N $ मध्यबिंदु-उत्तल है, और
किया जा रहा है \omterm{def:b3:topology:continuity}{निरंतर} (में अभिसरण हावी है $ t $), उत्तल; तो
है $\ln\norm{u(t)}_2 = \frac12\ln N(t)$. उत्तल के साथ क्षय
लघुगणक: ऊष्मा प्रवाह विस्फोट में ऊर्जा नहीं खो सकता है
फिर रुको.

\textbf{24.} कर्नेल के क्षण: $\int g_t = 1$ (प्रश्न
7 के साथ $ f = g_s $, या सीधे गाऊसी इंटीग्रल), $\int
x\,g_t(x)\dd x = 0$ (विषम इंटीग्रैंड), और, प्रतिस्थापित करना $ x =
2\sqrt t\,v $,
\[
\int_\R x^2g_t(x)\,\dd x
= \frac{4t}{\sqrt\pi}\int_\R v^2\eu^{-v^2}\dd v = 2t .
\]
स्थानापन्न $ x = z + y $ कनवल्शन और नोटिंग में
$\iint(\abs z + \abs y)^2g_t(z)f(y)\,\dd z\,\dd y < \infty $
(की प्रत्येक $\int\abs z^kg_t $, $\int\abs y^kf $ के लिए सीमित है $ k
\leq 2$, का उपयोग करना $\abs y \leq \frac{1 + y^2}2$), फ़ुबिनी और
टोनेली नीचे दिए गए क्षणिक इंटीग्रल पर लागू होता है:
\[
\int x\,u(t,x)\dd x = \iint (z + y)\,g_t(z)f(y)\,\dd z\,\dd
y = 0\cdot\!\int\! f + 1\cdot\!\int\! yf(y)\dd y,
\]
जो पहला दावा है; और
\[
\iint (z+y)^2g_t(z)f(y)\,\dd z\,\dd y
= 2t\int f + 2\cdot0\cdot\!\int\! yf + \int y^2f(y)\dd y ,
\]
दूसरा। इसका मतलब है जोड़ना, भिन्नता जोड़ना, और कर्नेल
मतलब योगदान देता है $0$ और विचरण $2t $: समय हो चुका $ t $ द
गर्मी पूरे क्षेत्र में फैल गई है $\sqrt{2t}$ ---
दूरी समय के वर्गमूल, हस्ताक्षर की तरह बढ़ती है
प्रसार का (और ब्राउनियन का)। के रास्ते का
\cref{ch:b3:probability}).

\textbf{25.} परिवर्तन पक्ष: $\hat f(\xi) =
\sqrt\pi\,\eu^{-\xi^2/4}$, इसलिए $\hat u(t,\xi) =
\sqrt\pi\,\eu^{-(t + \frac14)\xi^2}$, जो परिवर्तन है
का $(1 + 4t)^{-1/2}\exp\bigl(-x^2/(1+4t)\bigr)$ (द
गाऊसी शब्दकोश $\eu^{-ax^2} \mapsto
\sqrt{\pi/a}\,\eu^{-\xi^2/4a}$ साथ $ a = \frac1{1+4t}$):
\omterm{def:b3:topology:interior}{बंद} प्रपत्र. सीधी जांच, साथ $\sigma = 1 + 4t $:
\[
\partial_tu = \sigma^{-1/2}\eu^{-x^2/\sigma}
\Bigl(-\frac2\sigma + \frac{4x^2}{\sigma^2}\Bigr)
= \partial^2_{xx}u ,
\]
दोनों पक्षों से गणना की गई $\partial_xu =
-\frac{2x}\sigma\,u $. संरक्षण: $\int u(t) =
\sigma^{-1/2}\sqrt{\pi\sigma} = \sqrt\pi $ सभी के लिए $ t $.
अपव्यय:
\[
\norm{u(t)}_2^2 = \frac1\sigma\int\eu^{-2x^2/\sigma}\dd x
= \frac1\sigma\sqrt{\frac{\pi\sigma}2}
= \sqrt{\frac\pi2}\,(1+4t)^{-1/2},
\]
इसलिए $\norm{u(t)}_2 = (\pi/2)^{1/4}(1+4t)^{-1/4}$,
उत्तल लघुगणक के साथ गैर-वृद्धि (प्रश्न 23);
$ L^2$ \omterm{def:b3:rings:ideal}{आदर्श} की तरह क्षय होता है $ t^{-1/4}$, बिलकुल आधा
$ t^{-1/2}$ के प्रतिपादक $\norm{u(t)}_\infty $ --- के साथ संगत $\norm u_2^2 \leq \norm u_\infty\norm u_1$ और
का संरक्षण $\norm u_1$. विचरण: $\int x^2u(t) =
\frac1\sigma\cdot\frac{\sigma^{3/2}\sqrt\pi}2 =
\frac{\sqrt\pi}2\,(1 + 4t) = \int x^2f + 2t\sqrt\pi $, जैसे
प्रश्न 24 भविष्यवाणी करता है ($\int x^2f = \frac{\sqrt\pi}2$, $\int
f = \sqrt\pi $). पर $ t = 6$: $\sigma = 25$, शिखर ऊंचाई
$ u(6, 0) = \frac15$ बनाम $ u(0,0) = 1$, चौड़ाई पैमाना
$\sqrt\sigma = 5$ आरंभिक समय का गुना, और $\int u =
\sqrt\pi \approx 1.7725$ संपूर्ण: स्थान पाँच गुना है
निचला, पाँच गुना चौड़ा, और एक भी कैलोरी गायब नहीं है।
\end{solution}
