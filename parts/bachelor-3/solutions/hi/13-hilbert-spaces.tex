\section*{अध्याय \ref{ch:b3:hilbert} --- हिल्बर्ट स्पेस}

\begin{solution}{exo:b3:hilbert:1}
(ए) वास्तविक: विस्तार करें $\norm{x \pm y}^2 = \norm x^2 \pm 2\langle
x,y\rangle + \norm y^2$ और घटाओ. जटिल (\omterm{def:b3:hilbert:inner}{अंदरूनी प्रोडक्ट}
दूसरे स्लॉट में रैखिक): ऊपर बताए अनुसार विस्तार करना,
\[
\langle x, y\rangle = \frac14\sum_{k=0}^{3}
\iu^k\,\bigl\|\iu^kx + y\bigr\|^2,
\]
प्रत्येक पद योगदान दे रहा है $\iu^k\cdot2\operatorname{Re}\bigl(
(-\iu)^k\langle x,y\rangle\bigr)$, जिसका योग है $4\langle
x,y\rangle $ (के चार मानों की जाँच करें $ k $; $\sum\iu^k
(\norm x^2 + \norm y^2) = 0$).

(बी) $ L^1$: $ f = \mathbf 1_{\intcc0{1/2}}$, $ g = \mathbf
1_{\intcc{1/2}1}$: $\norm{f\pm g}_1^2 = 1$ प्रत्येक, योग $2$;
$2\norm f_1^2 + 2\norm g_1^2 = 1 \neq 2$. सुपर मानदंड: $ f =
\mathbf 1$, $ g(t) = t $ पर $\intcc01$: $\norm{f + g}_\infty^2 +
\norm{f-g}_\infty^2 = 4 + 1 = 5 \neq 4 = 2 + 2$. असफल होना
समांतर चतुर्भुज कानून, ये मानदंड संख्या से प्रेरित हैं \omterm{def:b3:hilbert:inner}{अंदरूनी प्रोडक्ट}
(जो इसे प्रत्यक्ष विस्तार द्वारा बाध्य करेगा)।
\end{solution}

(सी) $ d^2 = \bigl\|x - \tfrac12\bigr\|_2^2 = \int_0^1(x -

\begin{solution}{exo:b3:hilbert:3}
(ए) तुल्यता है \cref{thm:b3:hilbert:decomposition}
($\bar F = (F^\perp)^\perp $, और $\bar F = H \iff F^\perp =
\{0\}$). उदाहरण: अंतरिक्ष $ F $ परिमित अनुक्रम सघन है
में $\ell^2$ (छंटनी) और उचित: $ F^\perp = \{0\}$ अभी तक $ F
\neq \ell^2$ --- एक गै\omterm{def:b3:topology:interior}{र-बंद} उपस्थान के लिए, $ H = F \oplus
F^\perp $ स्पष्ट रूप से विफल रहता है ($ F \oplus \{0\} \neq H $).

(बी) $\abs{\langle x_n, y_n\rangle - \langle x, y\rangle} \leq
\abs{\langle x_n - x, y_n\rangle} + \abs{\langle x, y_n -
y\rangle} \leq \norm{x_n - x}\sup_n\norm{y_n} + \norm
x\,\norm{y_n - y} \to 0$ (अभिसरण अनुक्रम बंधे हुए हैं)।
प्रयुक्त: में \cref{thm:b3:hilbert:parseval}(1) देखना $ x - \lim
S_N \perp e_k $, और में \cref{thm:b3:hilbert:decomposition}
देखने के लिए $ F^\perp = \bar F^{\,\perp}$.
\end{solution}

\begin{solution}{exo:b3:hilbert:4}
$ e_0 = \frac1{\sqrt2}$. अगला, $ x \perp \mathbf 1$ पहले से
($\int_{-1}^1x = 0$), और $\int_{-1}^1x^2 = \frac23$: $ e_1 =
\sqrt{\tfrac32}\,x $. तब $ x^2 - \langle e_0, x^2\rangle e_0 =
x^2 - \frac13$ (और $\perp e_1$ समता द्वारा), साथ
\[
\int_{-1}^1\Bigl(x^2 - \frac13\Bigr)^2\dd x = \frac25 -
\frac49 + \frac29 = \frac{8}{45}:
\qquad e_2 = \sqrt{\tfrac{45}8}\,\Bigl(x^2 - \frac13\Bigr).
\]
तुलना: $ P_0 = 1$, $ P_1 = x $, $ P_2 = \frac{3x^2 - 1}2$, और
$\sqrt{n + \tfrac12}\,P_n $ देता है $\frac1{\sqrt2}$,
$\sqrt{\frac32}x $, $\sqrt{\frac52}\,\frac{3x^2-1}2 =
\sqrt{\frac{45}8}\bigl(x^2 - \frac13\bigr)$: बिल्कुल $ e_0, e_1,

\end{solution}

\begin{solution}{exo:b3:hilbert:5}
के लिए $ f(t) = t $: $ c_0 = 0$ और, भागों द्वारा एकीकृत करके, $ c_n =
\frac{\iu(-1)^n}{n}$ के लिए $ n \neq 0$: $\abs{c_n}^2 =
\frac1{n^2}$. \omterm{thm:b3:hilbert:parseval}{पारसेवल}:
\[
\frac1{2\pi}\int_{-\pi}^\pi t^2\dd t = \frac{\pi^2}3
= \sum_{n\neq0}\frac1{n^2} = 2\sum_{n\geq1}\frac1{n^2}
\ \Longrightarrow\ \sum_{n\geq1}\frac1{n^2} = \frac{\pi^2}6 .
\]
के लिए $ f(t) = t^2$: $ c_0 = \frac{\pi^2}3$, $ c_n =
\frac{2(-1)^n}{n^2}$ ($ n \ne 0$). \omterm{thm:b3:hilbert:parseval}{पारसेवल}:
\[
\frac1{2\pi}\int_{-\pi}^{\pi}t^4\dd t = \frac{\pi^4}5
= \frac{\pi^4}9 + \sum_{n\neq0}\frac4{n^4}
\ \Longrightarrow\
\sum_{n\geq1}\frac1{n^4} = \frac18\Bigl(\frac{\pi^4}5 -
\frac{\pi^4}9\Bigr) = \frac{\pi^4}{90} .
\]
कोई टुकड़ा-टुकड़ा नहीं-$\mathcal C^1$ चेतावनियों की आवश्यकता है:
\cref{thm:b3:hilbert:fourier} प्रत्येक को कवर करता है $ L^2$ समारोह।
\end{solution}

\begin{solution}{exo:b3:hilbert:7}
(ए) चलो $ M = \sup_k\norm{x_k}$. अदिश अनुक्रम
$(\langle e_n, x_k\rangle)_k $ से बंधे हैं $ M $: एक विकर्ण
निष्कर्षण उपज $ x_{k_j}$ साथ $\langle e_n, x_{k_j}\rangle
\to \gamma_n $ हरएक के लिए $ n $. प्रत्येक के लिए $ N $: $\sum_{n\leq
N}\abs{\gamma_n}^2 = \lim_j\sum_{n\leq N}\abs{\langle e_n,
x_{k_j}\rangle}^2 \leq M^2$ (बेसेल), तो $(\gamma_n) \in
\ell^2$ और $ x = \sum_n\gamma_ne_n \in H $
(\cref{thm:b3:hilbert:parseval}(3)). के लिए $ y \in H $:
\[
\abs{\langle y, x_{k_j} - x\rangle}
\leq \Bigl|\sum_{n\leq N}\overline{c_n(y)}\bigl(\langle e_n,
x_{k_j}\rangle - \gamma_n\bigr)\Bigr|
+ 2M\Bigl(\sum_{n>N}\abs{c_n(y)}^2\Bigr)^{1/2},
\]
विस्तार का उपयोग करना $\langle y, z\rangle =
\sum\overline{c_n(y)}c_n(z)$ और कॉची--पूंछ पर श्वार्ज़;
चुनना $ N $ तब $ j $: कमजोर अभिसरण $ x $.

(बी) $\langle y, e_n\rangle = c_n(y) \to 0$ हरएक के लिए $ y $
($\ell^2$ पूँछ): $ e_n \rightharpoonup 0$, अभी तक $\norm{e_n} =
1$: \omterm{def:b3:rings:ideal}{आदर्श} कमजोर नहीं है \omterm{def:b3:topology:continuity}{निरंतर}. में एक): $\norm x^2 =
\sum\abs{\gamma_n}^2 \leq \liminf_j\norm{x_{k_j}}^2$ (परिमित
अनुभाग और बेसेल फिर से): कमजोर सीमाएँ केवल \omterm{def:b3:rings:ideal}{आदर्श} खो सकती हैं।
\end{solution}

\begin{solution}{exo:b3:hilbert:8}
तय के लिए $ y $, $ x \mapsto \langle y, Tx\rangle $ एक है
\omterm{def:b3:topology:continuity}{निरंतर} रैखिक कार्यात्मक; रिज़्ज़ एक अनोखा देता है $ T^*y $ साथ
$\langle y, Tx\rangle = \langle T^*y, x\rangle $ सभी के लिए $ x $
---संयुग्मन, $\langle Tx, y\rangle = \langle x,
T^*y\rangle $. विशिष्टता बनाती है $ T^*$ रैखिक;
\[
\norm{T^*y} = \sup_{\norm x = 1}\abs{\langle T^*y, x\rangle}
= \sup_{\norm x=1}\abs{\langle y, Tx\rangle}
\leq \vertiii T\,\norm y,
\]
इसलिए $\vertiii{T^*} \leq \vertiii T $, और $ T^{**} = T $ देता है
समानता. बदलाव: $\langle Sx, y\rangle = \sum_{n\geq1}
x_n\bar y_{n+1} = \langle x, S^*y\rangle $ साथ $(S^*y)_n =
y_{n+1}$:पिछड़ा बदलाव. कर्नेल--छवि: $ T^*y = 0$ आईएफएफ
$\langle x, T^*y\rangle = 0$ सभी के लिए $ x $ आईएफएफ $\langle Tx,
y\rangle = 0$ सभी के लिए $ x $ आईएफएफ $ y \perp \operatorname{im}T $:
$\ker T^* = (\operatorname{im}T)^\perp $; ले रहा हूँ $\perp $ और
का उपयोग कर रहे हैं \cref{thm:b3:hilbert:decomposition},
$\overline{\operatorname{im}T} = (\ker T^*)^\perp $.
\end{solution}

\begin{solution}{exo:b3:hilbert:9}
अगर $ a(u, \cdot) = \varphi $: किसी के लिए $ w $,
\[
J(u + w) - J(u) = a(u, w) - \varphi(w) + \tfrac12a(w,w)
= \tfrac12a(w,w) \geq \tfrac\alpha2\norm w^2,
\]
के लिए सख्ती से सकारात्मक $ w \neq 0$: $ u $ अद्वितीय मिनिमाइज़र है.
इसके विपरीत, एक मिनिमाइज़र पर फ़ंक्शन $ t \mapsto J(u + tw)$
(एक द्विघात बहुपद में $ t $) का व्युत्पन्न लुप्त हो रहा है
$0$: $ a(u, w) - \varphi(w) = 0$ हरएक के लिए $ w $. प्रक्षेपण
पुनः व्युत्पन्न: एक \omterm{def:b3:topology:interior}{बंद} उपस्थान के लिए $ F $, लैक्स-मिलग्राम लगाएं
द \omterm{def:b3:hilbert:inner}{हिल्बर्ट स्थान} $ F $ साथ $ a(u,v) = \langle u, v\rangle $
($ M = \alpha = 1$) और $\varphi(v) = \langle x, v\rangle $: ए
अद्वितीय $ p \in F $ साथ $\langle p, v\rangle = \langle x,
v\rangle $ सभी के लिए $ v \in F $, यानी\ $ x - p \perp F $ --- और द्वारा
सममित मामला, $ p $ कम करता है $\frac12\norm v^2 - \langle
x, v\rangle = \frac12\norm{v - x}^2 - \frac12\norm x^2$ ऊपर
$ F $: प्रक्षेपण.
\end{solution}

\begin{solution}{exo:b3:hilbert:10}

ऑर्थोगोनैलिटी: दो अलग-अलग Haar फ़ंक्शंस हैं
विसंयुक्त (आंतरिक सज्जा का) समर्थन करता है (उत्पाद शून्य ए.ई.), या

वह \omterm{def:b3:topology:continuity}{निरंतर} समय $\int h_{\text{finer}} = 0$; ख़िलाफ़ $ h_0
= \mathbf 1$, दोबारा $\int h_n = 0$. समग्रता: का विस्तार
$\{h_0, \dots, h_{2^J-1}\}$ पर चरणबद्ध कार्य शामिल हैं
चरण का डायडिक ग्रिड $2^{-J}$; दोनों स्थानों का आयाम है
$2^J $ और हार फ़ंक्शन स्वतंत्र (ऑर्थोनॉर्मल) हैं:
अवधि है \emph{सभी} ऐसे चरण कार्य. डायडिक चरण
कार्य सघन हैं $ L^2(\intcc01)$: \omterm{def:b3:lebesgue:simple}{सरल कार्य} हैं
\omterm{def:b3:topology:interior}{घना} (\cref{thm:b3:lp:density}(1)), \omterm{def:b3:lebesgue:measurable}{औसत दर्जे का} सेट हैं
अंतरालों के परिमित संघों द्वारा अनुमानित
(\cref{exo:b3:measure:7}), और डायडिक वाले द्वारा अंतराल
(अंतबिंदु आगे बढ़ते हैं $\leq 2^{-J}$). द्वारा
\cref{thm:b3:hilbert:parseval}, हार प्रणाली एक है \omterm{def:b3:hilbert:onb}{हिल्बर्ट
आधार}.
\end{solution}

\begin{solution}{exo:b3:hilbert:11}
(मैं) $\Rightarrow $ (ii): ऑर्थोगोनल प्रक्षेपण के लिए,
$\langle Px, y\rangle = \langle Px, Py\rangle = \langle x,
Py\rangle $ (विघटन सम्मिलित करें $ x = Px + (x - Px)$
वगैरह।\ और क्रॉस टर्म्स को मारें)। (ii) $\Rightarrow $ (iii):
$\norm{Px}^2 = \langle P^2x, x\rangle = \langle Px, x\rangle
\leq \norm{Px}\norm x $, इसलिए $\vertiii P \leq 1$, और $ Pu = u $
शून्येतर छवि पर: $= 1$. (iii) $\Rightarrow $ (मैं): $ H =
\operatorname{im}P \oplus \ker P $ (बीजगणित, से $ P^2
= P $); लेना $ u = Pu \in \operatorname{im}P $, $ v \in \ker P $,
$ t \in \R $: $\norm{P(u + tv)}^2 = \norm u^2$ होना ही चाहिए $\leq
\norm{u + tv}^2 = \norm u^2 + 2t\operatorname{Re}\langle u,
v\rangle + t^2\norm v^2$ हरएक के लिए $ t $, मजबूर करना
$\operatorname{Re}\langle u, v\rangle = 0$ (तुलना करें
रैखिक पद के रूप में $ t \to 0^\pm $); की जगह $ v $ द्वारा $\iu v $
काल्पनिक भाग को भी ख़त्म कर देता है: $\operatorname{im}P \perp
\ker P $, जो बिल्कुल प्रक्षेपण की रूढ़िवादिता है।
उदाहरण: $ P(x, y) = (x + y, 0)$ पर $\R^2$: $ P^2 = P $,
छवि $ x $-एक्सिस, कर्नेल लाइन $ y = -x $, और
$\vertiii P = \sup\frac{\abs{x+y}}{\norm{(x,y)}} = \sqrt2$
(पर प्राप्त किया गया $(1,1)/\sqrt2$): एक तिरछे प्रक्षेपण का \omterm{def:b3:rings:ideal}{आदर्श} होता है
$> 1$. (रिकॉर्ड के लिए, (ii) सीधे (i) भी देता है:
$\ker P = \ker P^* = (\operatorname{im}P)^\perp $ द्वारा
\cref{exo:b3:hilbert:8}.)
\end{solution}

\begin{solution}{exo:b3:hilbert:12}
(ए) एकात्मक के लिए $ U $: $\norm{Ux - x}^2 = 2\norm x^2 -
2\operatorname{Re}\langle Ux, x\rangle $ और $\norm{U^*x -
x}^2 = 2\norm x^2 - 2\operatorname{Re}\langle x, Ux\rangle $:
दोनों एक साथ गायब हो जाते हैं, इसलिए $\ker(U - I) = \ker(U^* - I)$.
फिर, का उपयोग कर $\ker T^* = (\operatorname{im}T)^\perp $
(\cref{exo:b3:hilbert:8}) साथ $ T = U - I $ और $ T^* = U^* -
I $:
\[
\overline{\operatorname{im}(U - I)} = \bigl(\ker(U^* -
I)\bigr)^\perp = F^\perp .
\]

(बी) चालू $ F $: $ U^kx = x $, इसलिए $ A_nx = x $. के लिए $ x = (U - I)y $:
$ A_nx = \frac1n(U^ny - y)$, \omterm{def:b3:rings:ideal}{आदर्श} का $\leq \frac2n\norm y \to
0$. के लिए $ x $ में \omterm{def:b3:topology:interior}{बंद}: दिया गया $\varepsilon $, चुनना $ x' =
(U - I)y $ साथ $\norm{x - x'} < \varepsilon $; तब से
$\vertiii{A_n} \leq \frac1n\sum\vertiii{U^k} = 1$,
$\norm{A_nx} \leq \norm{A_n(x - x')} + \norm{A_nx'} \leq
\varepsilon + o(1)$.

(सी) विघटित करें $ x = Px + (x - Px)$ साथ $ Px \in F $ और $ x -
Px \in F^\perp = \overline{\operatorname{im}(U - I)}$ (भाग
(ए)): $ A_nx = Px + A_n(x - Px) \to Px + 0$.

(डी) फूरियर \omterm{def:b3:topology:basis}{आधार} पर $ e_m(x) = \eu^{2\iu\pi mx}$: $ Ue_m
= \eu^{2\iu\pi m\alpha}e_m $, इसलिए $ Ue_m = e_m $ आईएफएफ $ m\alpha
\in \Z $ आईएफएफ $ m = 0$ ($\alpha $ तर्कहीन): $ F = \C\mathbf 1$
और $ Pf = \langle\mathbf 1, f\rangle\mathbf 1 = \int_0^1f $.
प्रमेय पढ़ता है $\frac1n\sum_{k<n}f(\cdot + k\alpha) \to
\int_0^1f $ में $ L^2(\R/\Z)$: द \omterm{def:b3:groups:action}{कक्षा} एक का औसत
अपरिमेय घूर्णन समवितरण --- $ L^2$ की छाया
वेइल का समवितरण प्रमेय, शुद्ध हिल्बर्ट द्वारा प्राप्त किया गया
ज्यामिति.
\end{solution}

\begin{solution}{pb:b3:hilbert:1}
\textbf{1.} ग्राम--श्मिट गारंटी देता है
$\operatorname{Vect}(p_0, \dots, p_n) =
\operatorname{Vect}(1, \dots, t^n) = \R_n[t]$ और $ p_n \perp
p_k $ ($ k < n $), इस तरह $ p_n \perp \R_{n-1}[t]$. $ p_k $, का
सख्ती से बढ़ती \omterm{def:b3:galois:extension}{डिग्री}, स्वतंत्र हैं: एक \omterm{def:b3:topology:basis}{आधार}।

\textbf{2.} $ t\,p_n $ \omterm{def:b3:galois:extension}{डिग्री} का राक्षसी है $ n + 1$: बढ़ाना
$ t\,p_n = p_{n+1} + \sum_{k\leq n}c_kp_k $ साथ $ c_k =
\langle p_k, tp_n\rangle/\norm{p_k}^2$. के लिए $ k \leq n - 2$:
$\langle p_k, tp_n\rangle = \langle tp_k, p_n\rangle = 0$
(\omterm{def:b3:galois:extension}{डिग्री} $ k + 1 < n $). इसलिए $ tp_n = p_{n+1} + a_np_n +
b_np_{n-1}$, बताई गई पुनरावृत्ति, के साथ
\[
b_n = \frac{\langle p_{n-1}, tp_n\rangle}{\norm{p_{n-1}}^2}
= \frac{\langle tp_{n-1}, p_n\rangle}{\norm{p_{n-1}}^2}
= \frac{\langle p_n + (\text{lower}),\
p_n\rangle}{\norm{p_{n-1}}^2}
= \frac{\norm{p_n}^2}{\norm{p_{n-1}}^2} > 0 .
\]

\textbf{3.} होने देना $ t_1 < \dots < t_m $ अंक हो आंतरिक भाग को

t_i)$ (साथ $ q = 1$ अगर $ m = 0$). तब $ p_nq $ \omterm{def:b3:topology:continuity}{निरंतर} संकेत है
पर $ I $ और ए.ई. नहीं है\ शून्य: $\int_Ip_nq\,w \neq 0$. अगर $ m <
n $, यह विरोधाभासी है $ p_n \perp \R_{n-1}[t]$. इसलिए $ m = n $:
$ p_n $ है $ n $ विशिष्ट आंतरिक भाग जड़ें (अधिकतम होती हैं)। $ n $
सभी में जड़ें)।

\textbf{4.} $(t^2 - 1)^n $ की \omterm{def:b3:galois:extension}{डिग्री} है $2n $; $ n $ व्युत्पन्न
\omterm{def:b3:galois:extension}{डिग्री} छोड़ो $ n $, अग्रणी गुणांक के साथ
$\frac{(2n)(2n-1)\cdots(n+1)}{2^nn!} =

$< n $ का $(t^2-1)^n $, जो गायब हो जाता है $\pm1$ (आदेश की जड़
$ n $); बाद $ n $ कदम इंटीग्रैंड वहन करता है $ Q^{(n)} = 0$.

\textbf{5.} साथ $ u = (t^2 - 1)^n $:

\textbf{7.} से $\frac{\dd^{n+1}}{\dd t^{n+1}}\eu^{-t^2} =

\textbf{8.} होने देना $ f \in L^2(\R, \eu^{-t^2}\dd t)$ ऑर्थोगोनल बनें
प्रत्येक बहुपद के लिए, और $ g = f\eu^{-t^2}$. तब $ g \in
L^1$: $\int\abs f\eu^{-t^2} \leq \bigl(\int\abs
f^2\eu^{-t^2}\bigr)^{1/2}\bigl(\int\eu^{-t^2}\bigr)^{1/2}$
(कॉची-श्वार्ज़)। के लिए $\xi \in \R $, बढ़ाना $\eu^{-\iu\xi t}$:
आंशिक रकम तब से हावी है
\[
\sum_k\frac{\abs\xi^k}{k!}\int\abs f\,\abs t^k\eu^{-t^2}\dd t
\leq \Bigl(\int \abs f^2\eu^{-t^2}\Bigr)^{1/2}
\sum_k\frac{\abs\xi^k}{k!}\Bigl(\int
t^{2k}\eu^{-t^2}\Bigr)^{1/2} < \infty
\]
(अंतिम श्रृंखला अभिसरण करती है: $\int t^{2k}\eu^{-t^2} =
\Gamma(k+\frac12) \leq k!\,\sqrt\pi $, तो शर्तें हैं
$ O(\abs\xi^k/\sqrt{k!})$). अवधि-दर-अवधि एकीकरण
(\cref{cor:b3:lebesgue:additivity} निरपेक्ष पर लागू किया गया
श्रृंखला, फिर श्रृंखला के लिए फ़ुबिनी) देता है
\[
\int_\R g(t)\,\eu^{-\iu\xi t}\dd t
= \sum_k\frac{(-\iu\xi)^k}{k!}\int f(t)\,t^k\,\eu^{-t^2}\dd t
= 0 ,
\]
प्रत्येक अभिन्न प्राणी $\langle t^k, f\rangle_w $-प्रकार $= 0$. द्वारा
फूरियर रूपांतरण की स्वीकृत इंजेक्टिविटी
(\cref{ch:b3:fouriertransform}), $ g = 0$ ए.ई., तो $ f = 0$
ए.ई.: हरमाइट परिवार (जिसके विस्तार बहुपद हैं) है
कुल.

\]
विकल्प $ x\,p_n(x) = p_{n+1}(x) + a_np_n(x) +
b_np_{n-1}(x)$ और $ y\,p_n(y) = p_{n+1}(y) + a_np_n(y) +
b_np_{n-1}(y)$ दूसरे कार्यकाल में: $ a_n $ योगदान
रद्द करें, और $ b_n = \frac{h_n}{h_{n-1}}$ योगदान
प्रेरण अवधि रद्द करें; जो बचता है वह है
$\frac{p_{n+1}(x)p_n(y) - p_n(x)p_{n+1}(y)}{h_n}$. द
संगम रूप देने से अनुसरण करता है $ y \to x $ (दोनों पक्ष हैं
बहुपद में $ y $).

\textbf{20.} संगम स्वरूप देता है $ p_{n+1}'p_n -
p_n'p_{n+1} = h_n\sum_{k\leq n}\frac{p_k^2}{h_k} \geq
\frac{h_n}{h_0} > 0$ हर जगह. एक जड़ पर $ x_0$ का
$ p_{n+1}$: $ p_{n+1}'(x_0)\,p_n(x_0) > 0$, इसलिए $ p_n(x_0) \neq
0$ (कोई \omterm{def:b3:groups:normal}{सामान्य} जड़ नहीं)। लगातार जड़ों के बीच $ x_0 < x_1$
का $ p_{n+1}$ (सभी \omterm{def:b3:groups:simple}{सरल}, भाग I), $ p_{n+1}'$ विपरीत है
संकेत, इसलिए ऐसा होता है $ p_n $: की एक जड़ $ p_n $ प्रत्येक में निहित है
द $ n $ अंतराल --- और वह समाप्त हो जाता है $ n $ जड़ें:
गूंथना।

\textbf{21.} विस्तार $ D_n(t) = \det(tI_n - J_n)$ साथ में
अंतिम पंक्ति: $ D_n = (t - a_{n-1})D_{n-1} - b_{n-1}D_{n-2}$,
साथ $ D_0 = 1$, $ D_1 = t - a_0$: की पुनरावृत्ति और बीज
राक्षसी $ p_n $, इसलिए $ D_n = p_n $. की जड़ें $ p_n $ =
सममिति के eigenvalues $ J_n $: वास्तविक, और \omterm{def:b3:groups:simple}{सरल} द्वारा
प्रश्न 19--- गॉस चतुर्भुज एक का वर्णक्रमीय सिद्धांत है
भेस में त्रिविकर्ण मैट्रिक्स, परिमित-आयामी
की छाया \cref{ch:b3:spectral}.

\textbf{23.} अस्तित्व: रेखीय मानचित्र $\R_{2n-1}[t] \to
\R^{2n}$, $ P \mapsto (P(t_1), P'(t_1), \dots, P(t_n),
P'(t_n))$, इंजेक्शन है (ए $ P $ कर्नेल में है $ n $ दोगुना
जड़ें और \omterm{def:b3:galois:extension}{डिग्री} $\leq 2n - 1$, इसलिए $ P = 0$) के रिक्त स्थान के बीच
समान आयाम $2n $: विशेषण. बिंदुवार त्रुटि: ठीक करें $ t $
एक नोड नहीं और चुनें $ K $ ताकि $ g(s) = f(s) - Hf(s) -
K\,p_n(s)^2$ पर गायब हो जाता है $ s = t $. तब $ g $ पर गायब हो जाता है
$ n + 1$ विशिष्ट बिंदु $ t, t_1, \dots, t_n $, और $ g'$
प्रत्येक पर गायब हो जाता है $ t_i $ भी (दोनों $ f - Hf $ और $ p_n^2$ है
वहां डबल शून्य)। रोले देता है $ n $ का शून्य $ g'$ सख्ती से
के लगातार शून्य के बीच $ g $ --- नोड्स से अलग
--- तो $ g'$ है $2n $ विशिष्ट शून्य; रोले लगाना $2n - 1$
अधिक बार उत्पादन करता है $\xi_t $ साथ $ g^{(2n)}(\xi_t) = 0$.
तब से $\deg Hf \leq 2n - 1$ और $ p_n^2$ \omterm{def:b3:galois:extension}{डिग्री} का राक्षसी है
$2n $, $ g^{(2n)} = f^{(2n)} - K\,(2n)!$, कहाँ से $ K =
f^{(2n)}(\xi_t)/(2n)!$ --- और पहचान तुच्छ है
नोड्स. एकीकरण: $ Q_n(f) = Q_n(Hf)$ ($ Hf $ माचिस $ f $
नोड्स पर) और $ Q_n(Hf) = \int Hf\,w $ तक सटीकता से
\omterm{def:b3:galois:extension}{डिग्री} $2n - 1$ (प्रश्न 9), अतः चतुर्भुज त्रुटि है
$\int(f - Hf)\,w $. साथ $ m, M $ की चरम सीमा $ f^{(2n)}$ पर
$ I $, बिंदुवार पहचान निचोड़ती है
\[
\frac{m\,h_n}{(2n)!} \;\leq\; \int_I(f - Hf)\,w
\;\leq\; \frac{M\,h_n}{(2n)!} ,
\]
और मध्यवर्ती मूल्य प्रमेय पर लागू होता है \omterm{def:b3:topology:continuity}{निरंतर}
$ f^{(2n)}$ उद्धार $\xi $. (लेजेन्ड्रे के लिए $ n = 2$: $ h_2
= \int_{-1}^1(t^2 - \frac13)^2\dd t = \frac8{45}$, तो
त्रुटि है $ f^{(4)}(\xi)/135$.)

\textbf{24.} कर्नेल पुनरुत्पादन करता है $\R_{n-1}[t]$: विस्तार
$ q = \sum_k\frac{\langle p_k, q\rangle}{h_k}p_k $ देता है
$\int_I K_n(t_i, t)\,q(t)\,w(t)\dd t = q(t_i)$ हरएक के लिए
$ q $ \omterm{def:b3:galois:extension}{डिग्री} का $\leq n - 1$. लेना $ q = \ell_i $: बाईं ओर
बराबर $\ell_i(t_i) = 1$. लेकिन $ t \mapsto K_n(t_i,
t)\,\ell_i(t)$ \omterm{def:b3:galois:extension}{डिग्री} का एक बहुपद है $\leq (n - 1) + (n
- 1) = 2n - 2$, जिस पर $ Q_n $ सटीक है (प्रश्न 9), और यह
प्रत्येक नोड पर गायब हो जाता है $ t_j \neq t_i $ (कारक $\ell_i $), इसलिए
\[
1 = \int_I K_n(t_i, t)\,\ell_i(t)\,w(t)\dd t
= w_i\,K_n(t_i, t_i)
= w_i\sum_{k=0}^{n-1}\frac{p_k(t_i)^2}{h_k} .
\]
योग है $> 0$ (इसका $ k = 0$ शब्द है $1/h_0 > 0$): द
बताया गया सूत्र, और फिर से सकारात्मकता। जाँच करना ($ n = 2$,
लीजेंड्रे): $ p_0 = 1$, $ h_0 = 2$, $ p_1 = t $, $ h_1 = \frac23$;
पर $ t_i = \pm\frac1{\sqrt3}$,
\[
K_2(t_i, t_i) = \frac12 + \frac{1/3}{2/3} = 1,
\qquad w_i = 1,
\]
जैसा कि प्रश्न 11 में पाया गया है।

\textbf{25.} स्थानापन्न $ t = \cos\theta $, अभिन्न है
$\int_0^\pi\cos^6\theta\,\dd\theta =
\pi\,\frac{5\cdot3\cdot1}{6\cdot4\cdot2} = \frac{5\pi}{16}$
(वालिस, \cref{exo:b3:product:8}). $ n = 3$
चेबीशेव--गॉस नियम (प्रश्न 17) में नोड्स हैं
$\cos\frac\pi6 = \frac{\sqrt3}2$, $\cos\frac\pi2 = 0$,
$\cos\frac{5\pi}6 = -\frac{\sqrt3}2$ और बराबर वजन
$\frac\pi3$:
\[
Q_3(t^6) = \frac\pi3\Bigl(2\cdot\Bigl(\frac{\sqrt3}2
\Bigr)^{6}\Bigr) = \frac\pi3\cdot\frac{54}{64}
= \frac{9\pi}{32},
\qquad
\frac{5\pi}{16} - \frac{9\pi}{32} = \frac\pi{32} .
\]
भविष्यवाणी: राक्षसी डिग्री-$3$ \omterm{pb:b3:hilbert:1}{ऑर्थोगोनल बहुपद} है
$2^{-2}T_3 = t^3 - \frac34t $, साथ $ h_3 =
\frac1{16}\norm{T_3}_w^2 = \frac1{16}\cdot\frac\pi2 =
\frac\pi{32}$; और $ f = t^6$ स्थिरांक है $ f^{(6)} = 720 =
6!$, इसलिए प्रश्न 23 त्रुटि देता है $\frac{6!}{6!}\,h_3 =
\frac\pi{32}$ --- बिना किसी निर्भरता के $\xi$ बाएँ,
सूत्र को सटीक होने के लिए मजबूर किया जाता है, और यह है।
