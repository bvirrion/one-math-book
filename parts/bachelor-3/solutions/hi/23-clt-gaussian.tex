\section*{अध्याय \ref{ch:b3:clt} --- विशेषता
कार्य और केंद्रीय सीमा प्रमेय}

\begin{solution}{exo:b3:clt:1}
वर्दी पर $\intcc{-1}1$: $\varphi(\xi) =
\frac12\int_{-1}^1\eu^{\iu\xi x}\dd x =
\frac{\sin\xi}{\xi}$ (के बराबर $1$ पर $\xi = 0$).
घातीय $\mathcal E(\lambda)$: $\varphi(\xi) =
\lambda\int_0^\infty\eu^{(\iu\xi - \lambda)x}\dd x =
\frac{\lambda}{\lambda - \iu\xi}$ (प्रतिव्युत्पन्न
पर गायब हो जाता है $+\infty $ तब से $\operatorname{Re}(\iu\xi -
\lambda) < 0$). प्वासों $\mathcal P(\lambda)$: द्वारा
असतत \omterm{def:b3:probability:space}{कानून}ों के लिए स्थानांतरण प्रमेय,
\[
\varphi(\xi) = \sum_{k\geq0}\eu^{\iu\xi
k}\,\eu^{-\lambda}\frac{\lambda^k}{k!}
= \eu^{-\lambda}\exp\bigl(\lambda\eu^{\iu\xi}\bigr)
= \exp\bigl(\lambda(\eu^{\iu\xi} - 1)\bigr).
\]
द्विपद $\mathcal B(n, p)$: का योग $ n $ \omterm{def:b3:probability:independence}{स्वतंत्र}
बर्नौली चर, प्रत्येक सीएफ के साथ $1 - p + p\eu^{\iu\xi}$,
इसलिए $\varphi(\xi) = \bigl(1 - p + p\eu^{\iu\xi}\bigr)^n $
(\cref{prop:b3:clt:cfbasics}(बी))। पॉइसन एडिटिविटी: यदि $ X
\sim \mathcal P(\lambda)$, $ Y \sim \mathcal P(\mu)$ हैं
\omterm{def:b3:probability:independence}{स्वतंत्र},
\[
\varphi_{X+Y}(\xi) = \eu^{\lambda(\eu^{\iu\xi}-1)}
\eu^{\mu(\eu^{\iu\xi}-1)} =
\eu^{(\lambda+\mu)(\eu^{\iu\xi}-1)},
\]
का सीएफ $\mathcal P(\lambda + \mu)$; इंजेक्शन
(\cref{thm:b3:clt:injectivity}) \omterm{def:b3:probability:space}{कानून} की पहचान करता है।
\end{solution}

\begin{solution}{exo:b3:clt:2}
(ए) $\overline{\varphi_X(\xi)} = \E\eu^{-\iu\xi X} =
\varphi_{-X}(\xi)$. इसलिए $\varphi_X $ वास्तविक है यदि और केवल यदि
$\varphi_X = \varphi_{-X}$, यदि और केवल यदि (इंजेक्शन,
\cref{thm:b3:clt:injectivity}) $ X $ और $-X $ समान है
\omterm{def:b3:probability:space}{कानून}.
(बी) लिखें $\varphi_X(\xi_0) = \eu^{\iu\theta}$. तब
\[
\E\bigl[1 - \cos(\xi_0X - \theta)\bigr]
= 1 - \operatorname{Re}\bigl(\eu^{-\iu\theta}
\varphi_X(\xi_0)\bigr) = 1 - 1 = 0 .
\]
इंटीग्रैंड गैर-नकारात्मक है, इसलिए $\cos(\xi_0X - \theta) = 1$
लगभग निश्चित रूप से (शून्य के साथ एक गैर-नकारात्मक चर)। \omterm{def:b3:probability:space}{अपेक्षा}
गायब हो जाता है जैसे), यानी।\ $\xi_0X - \theta \in 2\pi\Z $ जैसा।: $ X $
अपने मानों को अंकगणितीय प्रगति में लेता है
$\frac{\theta}{\xi_0} + \frac{2\pi}{\xi_0}\Z $ लगभग निश्चित रूप से.
अगर $ X $ एक \omterm{ex:b3:lebesgue:gamma}{घनत्व}, यह गणनीय समुच्चय लेब्सगे-नल है,
इसलिए यह संभावना रखता है $0$ --- विरोधाभास; इसलिए
$\abs{\varphi_X(\xi)} < 1$ हरएक के लिए $\xi \neq 0$.
\end{solution}

\begin{solution}{exo:b3:clt:3}
द्वारा स्वतंत्रता और \cref{prop:b3:clt:cfbasics}:
\[
\varphi_{X+Y}(\xi) = \eu^{\iu m_1\xi - \sigma_1^2\xi^2/2}\,
\eu^{\iu m_2\xi - \sigma_2^2\xi^2/2}
= \eu^{\iu(m_1+m_2)\xi - (\sigma_1^2+\sigma_2^2)\xi^2/2},
\]
का सीएफ $\mathcal N(m_1 + m_2, \sigma_1^2 + \sigma_2^2)$;
इंजेक्शन समाप्त होता है। एफ़िन मानचित्रों के अंतर्गत स्थिरता है
एफ़िन नियम ($ aX + b \sim \mathcal N(am_1 + b,
a^2\sigma_1^2)$, पतित मामले की अनुमति $ a = 0$), और
के अंतर्गत स्थिरता \omterm{def:b3:probability:independence}{स्वतंत्र} योग प्रेरण द्वारा अनुसरण करता है
उपरोक्त गणना. के लिए \emph{आश्रित} \omterm{def:b3:clt:gaussianvector}{गाऊसी} योग
होने की आवश्यकता नहीं है \omterm{def:b3:clt:gaussianvector}{गाऊसी}: में \cref{exo:b3:clt:9}, $ X $ और $ Y =
\varepsilon X $ प्रत्येक मानक हैं \omterm{def:b3:clt:gaussianvector}{गाऊसी} लेकिन $ X + Y $
संभावना के साथ गायब हो जाता है $\frac12$ बिना ए.एस. हुए\
शून्य, तो यह नहीं है \omterm{def:b3:clt:gaussianvector}{गाऊसी}.
\end{solution}

\begin{solution}{exo:b3:clt:4}
(ए) \emph{सीधा निहितार्थ.} होने देना $ t $ एक हो निरंतरता का बिंदु $ F_X $ और $\delta > 0$. ले लो \omterm{def:b3:topology:continuity}{निरंतर} रैंप
$ f^-$ ($= 1$ पर $\intoc{-\infty}{t-\delta}$, $0$ से $ t $
पर, बीच में जुड़ें) और $ f^+$ ($= 1$ पर $\intoc{-\infty}t $,
$0$ से $ t + \delta $ पर, बीच में संबंध); तब $ f^- \leq
\mathbf 1_{\intoc{-\infty}t} \leq f^+$, इसलिए
\[
\E f^-(X_n) \leq F_{X_n}(t) \leq \E f^+(X_n),
\]
और बाहरी पद आपस में मिलते हैं $\E f^\pm(X)$, स्वयं
बीच में निचोड़ा हुआ $ F_X(t - \delta)$ और $ F_X(t + \delta)$.
दे $ n \to \infty $ तब $\delta \to 0$ और उपयोग कर रहे हैं
निरंतरता का $ F_X $ पर $ t $: $ F_{X_n}(t) \to F_X(t)$.

\emph{बातचीत.} होने देना $ f $ बाध्य होना \omterm{def:b3:topology:continuity}{निरंतर}, $ M =
\sup\abs f $, $\varepsilon > 0$. निरंतरता के अंक
$ F_X $ घने हैं ($ F_X $ \omterm{def:b3:rings:primemaximal}{अधिक से अधिक} अनगिनत छलाँगें हैं), इसलिए
चुनें निरंतरता अंक $ a < b $ साथ $ F_X(a) < \varepsilon $
और $1 - F_X(b) < \varepsilon $. पर कॉम्पैक्ट $\intcc ab $
समारोह $ f $ समान रूप से है \omterm{def:b3:topology:continuity}{निरंतर}: चुनना निरंतरता
अंक $ a = t_0 < t_1 < \dots < t_m = b $ का $ F_X $ के साथ
का दोलन $ f $ \omterm{def:b3:rings:primemaximal}{अधिक से अधिक} $\varepsilon $ सभी के ऊपर
$\intoc{t_{j-1}}{t_j}$, और सेट करें $ g = \sum_j
f(t_j)\,\mathbf 1_{\intoc{t_{j-1}}{t_j}}$. तब $\abs{f - g}
\leq \varepsilon $ पर $\intoc ab $, $\abs g \leq M $, और के लिए
$ T = X_n $ या $ X $:
\[
\bigl|\E f(T) - \E g(T)\bigr| \leq \varepsilon +
2M\bigl(F_T(a) + 1 - F_T(b)\bigr).
\]
इसके अतिरिक्त $\E g(X_n) = \sum_j f(t_j)\bigl(F_{X_n}(t_j) -
F_{X_n}(t_{j-1})\bigr) \to \E g(X)$ (का एक सीमित योग
अभिसारी शर्तें, सभी $ t_j $ प्राणी निरंतरता अंक), और
$ F_{X_n}(a) \to F_X(a) < \varepsilon $, $1 - F_{X_n}(b) \to 1
- F_X(b) < \varepsilon $. संयोजन:
$\limsup_n\abs{\E f(X_n) - \E f(X)} \leq 2\varepsilon +
8M\varepsilon $; होने देना $\varepsilon \to 0$.

(बी) \omterm{def:b3:ode:stability}{स्थिर}ांक का वितरण कार्य $ c $ है
$\mathbf 1_{\intco c\infty}$, \omterm{def:b3:topology:continuity}{निरंतर} को छोड़कर $ c $. के लिए
$\varepsilon > 0$, बिन्दु $ c - \varepsilon $ और $ c +
\frac\varepsilon2$ हैं निरंतरता अंक, तो
\[
\P(\abs{X_n - c} > \varepsilon) \leq F_{X_n}(c -
\varepsilon) + 1 - F_{X_n}\Bigl(c + \frac\varepsilon2\Bigr)
\longrightarrow 0 + 1 - 1 = 0 .
\]
\end{solution}

\begin{solution}{exo:b3:clt:5}
होने देना $ z_n = p_n(\eu^{\iu\xi} - 1)$, इसलिए $\varphi_{X_n}(\xi) =
(1 + z_n)^n $ (\cref{exo:b3:clt:1}) और $\abs{z_n} \leq 2p_n
\to 0$ (टिप्पणी $ p_n = \frac{np_n}n \to 0$). दोनों $1 + z_n $ और
$\eu^{z_n}$ अधिकतम मापांक है $1$: $\abs{1 + z_n} =
\abs{(1 - p_n) + p_n\eu^{\iu\xi}} \leq 1$ त्रिकोण द्वारा
असमानता, और $\abs{\eu^{z_n}} = \eu^{p_n(\cos\xi - 1)}
\leq 1$. दूरबीन असमानता $\abs{a^n - b^n} \leq
n\abs{a - b}$ (का सबूत \cref{thm:b3:clt:clt}) और शक्ति
श्रृंखलाबद्ध $\abs{\eu^z - 1 - z} \leq
\abs z^2\eu^{\abs z}$ देना
\[
\bigl|(1 + z_n)^n - \eu^{nz_n}\bigr| \leq n\bigl|1 + z_n -
\eu^{z_n}\bigr| \leq n\,\abs{z_n}^2\,\eu^{\abs{z_n}} \leq
4\eu^2\,np_n^2 = 4\eu^2\,(np_n)\,p_n \longrightarrow 0 .
\]
तब से $ nz_n = np_n(\eu^{\iu\xi} - 1) \to
\lambda(\eu^{\iu\xi} - 1)$, हम निष्कर्ष निकालते हैं $\varphi_{X_n}(\xi)
\to \exp\bigl(\lambda(\eu^{\iu\xi} - 1)\bigr)$ हरएक के लिए
$\xi $: का सीएफ $\mathcal P(\lambda)$, और एल\'ईवी
(\cref{thm:b3:clt:levy}) देता है $ X_n \Rightarrow \mathcal
P(\lambda)$. संख्यात्मक रूप से: $\P\bigl(\mathcal B(100, 0.02) =
0\bigr) = 0.98^{100} = \eu^{100\ln 0.98} \approx
\eu^{-2.020} \approx 0.1326$, जबकि $\P\bigl(\mathcal P(2) =
0\bigr) = \eu^{-2} \approx 0.1353$: दो प्रतिशत अलग
पहले से ही इस मोटे स्तर पर $ n $.
\end{solution}

\begin{solution}{exo:b3:clt:7}
चर $ g(U_k)$ क्या आई.आई.डी.\ (\omterm{def:b3:lebesgue:measurable}{औसत दर्जे का} की छवियां
आई.आई.डी.\ चर), वर्ग-अभिन्न, मतलब के साथ $ I $
(स्थानांतरण प्रमेय, \cref{exo:b3:product:9}) और विचरण
$\sigma^2$. अगर $\sigma > 0$, \cref{thm:b3:clt:clt} लागू किया गया
उनके लिए बिल्कुल कथित अभिसरण है
\[
\sqrt n\,\Bigl(\frac1n\sum_{k\leq n}g(U_k) - I\Bigr) =
\frac{\sum_{k\leq n}\bigl(g(U_k) - I\bigr)}{\sqrt n}
\Longrightarrow \mathcal N(0, \sigma^2)
\]
(अगर $\sigma = 0$, $ g $ ए.एस. है\ \omterm{def:b3:ode:stability}{स्थिर}ांक और बायां हाथ
पक्ष समान रूप से गायब हो जाता है)। इस तरह
$\P\bigl(\abs{\frac1n\sum g(U_k) - I} \leq
1.96\,\sigma/\sqrt n\bigr) \to 0.95$: त्रुटि पट्टी $\pm
1.96\,\sigma/\sqrt n $ आयाम देखता है $ d $ केवल के माध्यम से
\omterm{def:b3:ode:stability}{स्थिर}ांक $\sigma $, कभी भी दर के माध्यम से नहीं $ n $. द
मध्यबिंदु नियम के साथ $ n $ आयाम में नोड्स $ d $ जाली है
$ n^{-1/d}$ और आदेश की त्रुटि $ n^{-2/d}$ के लिए $\mathcal C^2$
integrands. \omterm{ex:b3:probability:sllnapps}{मोंटे कार्लो}'एस $ n^{-1/2}$ से अधिक तेजी से क्षय होता है
$ n^{-2/d}$ बिल्कुल कब $\frac12 > \frac2d $, यानी\ $ d > 4$:
आयाम से $5$ पर, यादृच्छिक नमूना स्पर्शोन्मुख रूप से धड़कता है
ग्रिड --- आयामीता का अभिशाप स्पेयर करता है
संभाव्य विधियाँ, यही कारण है \omterm{ex:b3:probability:sllnapps}{मोंटे कार्लो} नियम
उच्च आयामी एकीकरण.
\end{solution}

\begin{solution}{exo:b3:clt:8}
\emph{जोड़।} तय के लिए $\xi $:
\[
\bigl|\E\eu^{\iu\xi(X_n+Y_n)} -
\eu^{\iu\xi c}\,\E\eu^{\iu\xi X_n}\bigr|
= \bigl|\E\bigl[\eu^{\iu\xi X_n}\bigl(\eu^{\iu\xi Y_n} -
\eu^{\iu\xi c}\bigr)\bigr]\bigr|
\leq \E\bigl|\eu^{\iu\xi(Y_n - c)} - 1\bigr| .
\]
घटना पर फूट $\{\abs{Y_n - c} \leq \delta\}$: वहाँ,
$\abs{\eu^{\iu\xi(Y_n-c)} - 1} \leq \abs\xi\,\delta $ (द
जीवा चाप से छोटी होती है); पूरक योगदान देता है
\omterm{def:b3:rings:primemaximal}{अधिक से अधिक} $2\,\P(\abs{Y_n - c} > \delta) \to 0$. इसलिए
$\limsup $ है $\leq \abs\xi\,\delta $ हरएक के लिए $\delta > 0$:
अंतर हो जाता है $0$. तब से $\E\eu^{\iu\xi X_n} \to
\varphi_X(\xi)$, हम पाते हैं $\varphi_{X_n+Y_n}(\xi) \to
\eu^{\iu\xi c}\varphi_X(\xi) = \varphi_{X+c}(\xi)$, और
एल\'ईवी (\cref{thm:b3:clt:levy}) पैदावार $ X_n + Y_n
\Rightarrow X + c $.

\emph{उत्पाद।} पहला, $ cX_n \Rightarrow cX $:
$\varphi_{cX_n}(\xi) = \varphi_{X_n}(c\xi) \to
\varphi_X(c\xi) = \varphi_{cX}(\xi)$. अगला, $(Y_n - c)X_n
\to 0$ संभाव्यता में: के \omterm{def:b3:probability:space}{कानून} $ X_n $ तंग हैं (उनके
सीएफएस एक सीएफ में परिवर्तित हो जाते हैं; का कसाव कदम देखें
\cref{thm:b3:clt:levy}), तो दिया गया $\varepsilon > 0$ चुनना $ M $
साथ $\P(\abs{X_n} > M) \leq \varepsilon $ सभी के लिए $ n $; तब
\[
\P\bigl(\abs{(Y_n - c)X_n} > \varepsilon\bigr) \leq
\P(\abs{X_n} > M) + \P\Bigl(\abs{Y_n - c} >
\frac{\varepsilon}{M}\Bigr) \leq \varepsilon + o(1) .
\]
लिखना $ Y_nX_n = cX_n + (Y_n - c)X_n $ और योग लागू करना
भाग (जिसका प्रमाण केवल उपयोग किया गया है $ Y_n' := (Y_n - c)X_n \to 0$ में
संभाव्यता, \omterm{def:b3:ode:stability}{स्थिर}ांक के साथ $0$): $ Y_nX_n \Rightarrow cX $.

\emph{आवेदन पत्र।} बड़ी संख्या के मजबूत \omterm{def:b3:probability:space}{कानून} द्वारा
(\cref{thm:b3:probability:slln}), $\hat p_n \to p $ ए.एस., तो
द्वारा निरंतरता $\hat\sigma_n = \sqrt{\hat p_n(1 - \hat p_n)}
\to \sigma = \sqrt{p(1 - p)} > 0$ ए.एस., इसलिए
$\frac{\sigma}{\hat\sigma_n} \to 1$ संभावना में. स्लटस्की का
उत्पाद नियम उन्नयन $\frac{S_n - np}{\sigma\sqrt n}
\Rightarrow \mathcal N(0,1)$ को $\frac{S_n -
np}{\hat\sigma_n\sqrt n} = \frac{\sigma}{\hat\sigma_n}\cdot
\frac{S_n - np}{\sigma\sqrt n} \Rightarrow \mathcal N(0,1)$:
द \emph{प्रयोग करने योग्य} विश्वास अंतराल $\hat p_n \pm
1.96\,\hat\sigma_n/\sqrt n $, अकेले डेटा से निर्मित,
अपने स्पर्शोन्मुख रखता है $95\%$ स्तर।
\end{solution}

\begin{solution}{exo:b3:clt:9}
(ए)विभाजन \omterm{def:b3:probability:space}{अपेक्षा} के दो मूल्यों पर
$\varepsilon $ (स्वतंत्रता): बोरेल के लिए $ B $, $\P(Y \in B) =
\frac12\P(X \in B) + \frac12\P(-X \in B) = \P(X \in B)$,
तब से $-X \sim X $ ($\mathcal N(0,1)$ सममित है): $ Y \sim
\mathcal N(0,1)$. और $\operatorname{Cov}(X, Y) =
\E[\varepsilon X^2] = \E[\varepsilon]\,\E[X^2] = 0 \cdot 1 =
0$.
(बी) $\abs Y = \abs X $, इसलिए $\P(\abs X \leq 1,\ \abs Y \geq 2)

(1 + \varepsilon)X $ वास्तविक होगा \omterm{def:b3:clt:gaussianvector}{गाऊसी} चर
(\cref{def:b3:clt:gaussianvector} साथ $ t = (1,1)$); लेकिन
$\P(X + Y = 0) = \P(\varepsilon = -1) = \frac12$, जबकि ए
\omterm{def:b3:clt:gaussianvector}{गाऊसी} वेरिएबल में एक परमाणु तभी होता है जब वह a.s. हो।\ \omterm{def:b3:ode:stability}{स्थिर}

$\{\varepsilon = 1\}$. विरोधाभास: $(X, Y)$ क्या नहीं है
\omterm{def:b3:clt:gaussianvector}{गाऊसी}.
(सी) प्रत्येक सीमांत है \omterm{def:b3:clt:gaussianvector}{गाऊसी} और सहप्रसरण लुप्त हो जाता है,

\end{solution}

\begin{solution}{pb:b3:clt:1}
\textbf{1.} परिवार $(X_1, \dots, X_n, N_1, \dots, N_n)$
है \omterm{def:b3:probability:independence}{स्वतंत्र}: दो ब्लॉक हैं \omterm{def:b3:probability:independence}{स्वतंत्र} प्रत्येक का
अन्य निर्माण द्वारा और प्रत्येक ब्लॉक i.i.d. है $ W_i $ एक है
मापने योग्य कार्य चरों का $(X_j)_{j<i}$ और
$(N_j)_{j>i}$ केवल, सभी से भिन्न $ X_i $ और $ N_i $: द्वारा
गठबंधन सिद्धांत
(\cref{thm:b3:probability:independence}), $ W_i $ है
\omterm{def:b3:probability:independence}{स्वतंत्र} जोड़ी का $(X_i, N_i)$. विघटन
$ H_i = W_i + \frac{X_i}{\sqrt n}$ और $ H_{i-1} = W_i +
\frac{N_i}{\sqrt n}$ परिभाषाओं से तत्काल हैं:
से गुजर रहा है $ H_i $ को $ H_{i-1}$ एकल सारांश स्वैप करता है
$ X_i $ के लिए $ N_i $.

\textbf{2.} टेलर--लैग्रेंज क्रम में $3$: वहाँ है $ c $
बीच में $ w $ और $ w + h $ साथ $ f(w + h) = f(w) + f'(w)h +
\frac12f''(w)h^2 + \frac16f'''(c)h^3$, और $\abs{f'''(c)}
\leq M_3$ बाउंड देता है.

\textbf{3.} दोनों विस्तारों को उभयनिष्ठ पर घटाना
\omterm{def:b3:topology:basis}{आधार} बिंदु $ w = W_i $:
\[
f(H_i) - f(H_{i-1}) = f'(W_i)\,\frac{X_i - N_i}{\sqrt n} +
\frac{f''(W_i)}{2}\,\frac{X_i^2 - N_i^2}{n} + R_i,
\qquad
\abs{R_i} \leq \frac{M_3}{6}\cdot
\frac{\abs{X_i}^3 + \abs{N_i}^3}{n^{3/2}} .
\]
लेना अपेक्षाएं. प्रश्न 1 से, $ f'(W_i)$ और $ f''(W_i)$
हैं \omterm{def:b3:probability:independence}{स्वतंत्र} का $(X_i, N_i)$, तो मिश्रित अपेक्षाएं
कारक:
\begin{align*}
\E\Bigl[f'(W_i)\,\frac{X_i - N_i}{\sqrt n}\Bigr] &=
\E\bigl[f'(W_i)\bigr]\,\frac{\E X_i - \E N_i}{\sqrt n} = 0,
\\
\E\Bigl[f''(W_i)\,\frac{X_i^2 - N_i^2}{n}\Bigr] &=
\E\bigl[f''(W_i)\bigr]\,\frac{1 - 1}{n} = 0 :
\end{align*}
के पहले दो क्षण $ X_i $ और $ N_i $ \emph{मिलान}, और
केवल शेष बचता है:
\[
\bigl|\E f(H_i) - \E f(H_{i-1})\bigr| \leq \E\abs{R_i} \leq
\frac{M_3}{6}\cdot\frac{\beta + \gamma}{n^{3/2}} .
\]
 \omterm{def:b3:clt:gaussianvector}{गाऊसी} तीसरा क्षण: $\gamma = \E\abs{N_1}^3 =
2\int_0^\infty x^3\,\frac{\eu^{-x^2/2}}{\sqrt{2\pi}}\,\dd x
= \frac{2}{\sqrt{2\pi}}\int_0^\infty 2u\,\eu^{-u}\dd u =
\frac{4}{\sqrt{2\pi}} = \frac{2\sqrt2}{\sqrt\pi}$
(प्रतिस्थापन $ u = x^2/2$, तब $\Gamma(2) = 1$).

\textbf{4.} अंतःसर्पण $\E f(T_n) - \E f(G_n) =
\sum_{i=1}^n\bigl(\E f(H_i) - \E f(H_{i-1})\bigr)$ और
प्रत्येक में प्रश्न 3 लागू करना $ n $ शर्तें:
\[
\bigl|\E f(T_n) - \E f(G_n)\bigr| \leq
n \cdot \frac{M_3(\beta + \gamma)}{6\,n^{3/2}} =
\frac{M_3\,(\beta + \gamma)}{6\,\sqrt n} .
\]

\textbf{5.} $ G_n $ है $\mathcal N(0,1)$ \emph{बिल्कुल} हरएक के लिए $ n $ (\omterm{def:b3:groups:normal}{सामान्य}ीकृत योग) \omterm{def:b3:probability:independence}{स्वतंत्र} मानक
\omterm{def:b3:clt:gaussianvector}{गाऊसी}, \cref{exo:b3:clt:3}), इसलिए $\E f(G_n) = \E f(N)$
और प्रश्न 4 पढ़ता है $\abs{\E f(T_n) - \E f(N)} \leq
\frac{M_3(\beta+\gamma)}{6\sqrt n} \to 0$ के लिए $ f \in
\mathcal C^3_b $. \emph{उन्नत करना।} होने देना $ f $ बाध्य होना
\omterm{def:b3:topology:continuity}{निरंतर}, $ M = \sup\abs f $, $\varepsilon > 0$. चुनना $ A
\geq 1$ साथ $\frac1{A^2} \leq \varepsilon $: चेबीशेव के साथ
$\V(T_n) = 1$ देता है $\P(\abs{T_n} > A) \leq \varepsilon $
सभी के लिए $ n $, और इसी तरह $\P(\abs N > A) \leq \varepsilon $.
होने देना $\chi $ होना $\mathcal C^\infty $ साथ
$\mathbf 1_{\intcc{-A}A} \leq \chi \leq
\mathbf 1_{\intcc{-A-1}{A+1}}$ (एक चिकना पठार, द्वारा निर्मित
\omterm{def:b3:lp:mollifier}{शांत करनेवाला} $\mathbf 1_{\intcc{-A-\frac12}{A+\frac12}}$,
\cref{thm:b3:lp:regularization}); $ g = f\chi $ है \omterm{def:b3:topology:continuity}{निरंतर}
साथ कॉम्पैक्ट समर्थन, इसलिए समान रूप से \omterm{def:b3:topology:continuity}{निरंतर}, तो यह है
मोलीकरण $ g_\eta = g * \rho_\eta $ है $\mathcal
C^\infty $ सभी आदेशों के बंधे हुए डेरिवेटिव के साथ और
$\norm{g - g_\eta}_\infty \leq \varepsilon $ के लिए $\eta $
काफी छोटा। के लिए $ T = T_n $ या $ N $, तब से $ f = g $ पर
$\intcc{-A}A $ और $\abs{f - g} \leq 2M $ हर जगह:
\[
\bigl|\E f(T) - \E g_\eta(T)\bigr| \leq
\E\abs{(f - g)(T)} + \norm{g - g_\eta}_\infty
\leq 2M\,\P(\abs T > A) + \varepsilon \leq
(2M + 1)\,\varepsilon .
\]
के साथ संयोजन $\E g_\eta(T_n) \to \E g_\eta(N)$ (प्रश्न
4 लागू होता है: $ g_\eta \in \mathcal C^3_b $):
\[
\limsup_n\;\bigl|\E f(T_n) - \E f(N)\bigr| \leq
2(2M + 1)\,\varepsilon ,
\]
और $\varepsilon $ मनमाना था: $\E f(T_n) \to \E f(N)$ के लिए
हर सीमाबद्ध \omterm{def:b3:topology:continuity}{निरंतर} $ f $, यानी\ $ T_n \Rightarrow
\mathcal N(0,1)$.

\textbf{6.} समान वितरण केवल एक के माध्यम से दर्ज किया गया
वाक्य: ``$ X_i $ और $ N_i $ पहले दो समान हैं
क्षण'' तो चलिए $ X_1, \dots, X_n $ होना \omterm{def:b3:probability:independence}{स्वतंत्र},
केन्द्रित, भिन्नताओं के साथ $\sigma_i^2$ और परिमित तीसरा
क्षण, $ s_n^2 = \sum_i\sigma_i^2 > 0$, और ले लो $ N_i \sim
\mathcal N(0, \sigma_i^2)$ \omterm{def:b3:probability:independence}{स्वतंत्र} हर चीज की।
\omterm{def:b3:groups:normal}{सामान्य}ीकरण के साथ संकरों को परिभाषित करें $ s_n $: $ H_i =
\frac1{s_n}(\sum_{j\leq i}X_j + \sum_{j>i}N_j)$. में
$ i $-वें स्वैप, $\E X_i = \E N_i = 0$ और $\E X_i^2 = \E N_i^2
= \sigma_i^2$ फिर से मार डालो $ f'$ और $ f''$ शर्तें, और
शेषफल देता है (उपयोग करके)। $\E\abs{N_i}^3 = \sigma_i^3\gamma $ द्वारा
स्केलिंग):
\[
\bigl|\E f(H_i) - \E f(H_{i-1})\bigr| \leq
\frac{M_3}{6\,s_n^3}\bigl(\E\abs{X_i}^3 +
\sigma_i^3\gamma\bigr) .
\]
टेलीस्कोपिंग:
\[
\Bigl|\E f\Bigl(\frac{X_1 + \dots + X_n}{s_n}\Bigr) -
\E f(N)\Bigr| \leq \frac{M_3}{6\,s_n^3}\sum_{i=1}^n
\Bigl(\E\abs{X_i}^3 + \V(X_i)^{3/2}\,\gamma\Bigr) .
\]
तब से $\sigma_i^3 = (\E X_i^2)^{3/2} \leq \E\abs{X_i}^3$
(शक्ति--मतलब असमानता, यानी\ जेन्सेन के लिए $ t \mapsto
t^{3/2}$ के लिए आवेदन किया $ X_i^2$), दाहिनी ओर अधिकतम है
$\frac{M_3(1 + \gamma)}{6}\cdot
\frac{\sum_i\E\abs{X_i}^3}{s_n^3}$: अंतर्गत \emph{लायपुनोव का
शर्त} $\frac1{s_n^3}\sum_i\E\abs{X_i}^3 \to 0$, द
\omterm{def:b3:groups:normal}{सामान्य}ीकृत रकमें \omterm{def:b3:probability:space}{कानून} में परिवर्तित हो जाती हैं $\mathcal N(0,1)$ ---
समान वितरण के बिना सीएलटी।

\textbf{7.} होने देना $\rho \in \mathcal C^\infty_c(\intoo01)$
साथ $\int\rho = 1$ और सेट करें $\psi(x) = \int_x^1\rho(s)\dd
s $: $\psi $ है $\mathcal C^\infty $, गैर वृद्धिशील, $\psi = 1$
पर $\R_-$, $\psi = 0$ पर $\intco1\infty $; होने देना $ K =
\norm{\psi'''}_\infty $. के लिए $ t \in \R $ और $\delta > 0$
परिभाषित करना $\psi_\delta(x) = \psi\bigl(\frac{x - t}\delta\bigr)$
और $\tilde\psi_\delta(x) = \psi\bigl(\frac{x - t}\delta +
1\bigr)$: ये हैं $\mathcal C^3_b $ तीसरे व्युत्पन्न के साथ
से घिरा हुआ $ K/\delta^3$, और
\[
\mathbf 1_{\intoc{-\infty}{t-\delta}} \leq
\tilde\psi_\delta \leq \mathbf 1_{\intoc{-\infty}t} \leq
\psi_\delta \leq \mathbf 1_{\intoc{-\infty}{t+\delta}} .
\]
ऊपरी सीमा: प्रश्न 4 द्वारा लागू $\psi_\delta $ (साथ
$ M_3 = K/\delta^3$),
\[
\P(T_n \leq t) \leq \E\psi_\delta(T_n) \leq
\E\psi_\delta(N) + \frac{K(\beta +
\gamma)}{6\,\delta^3\sqrt n}
\leq \Phi(t + \delta) + \frac{K(\beta +
\gamma)}{6\,\delta^3\sqrt n}
\leq \Phi(t) + \frac{\delta}{\sqrt{2\pi}} +
\frac{K(\beta + \gamma)}{6\,\delta^3\sqrt n},
\]
क्योंकि $\Phi $ \omterm{def:b3:ode:stability}{स्थिर}ांक के साथ लिप्सचिट्ज़ है
$\frac1{\sqrt{2\pi}}$ (इसका \omterm{ex:b3:lebesgue:gamma}{घनत्व} से घिरा हुआ है
$\frac1{\sqrt{2\pi}}$). सममित निचली सीमा के माध्यम से
$\tilde\psi_\delta $ दो-अवधि का अनुमान देता है
\[
\sup_{t\in\R}\,\bigl|\P(T_n \leq t) - \Phi(t)\bigr| \leq
\frac{K(\beta + \gamma)}{6}\cdot\frac{1}{\delta^3\sqrt n} +
\frac{\delta}{\sqrt{2\pi}}
\qquad(\delta > 0\ \text{arbitrary}).
\]
दो पद कब संतुलित होते हैं $\delta^{-3}n^{-1/2} \asymp
\delta $, यानी\ $\delta = n^{-1/8}$: तो फिर दोनों हैं
$ O(n^{-1/8})$, प्रत्येक के लिए मान्य एक स्पष्ट समान दर
$ n $. (इष्टतम बेरी--एसेन दर $ C\beta/\sqrt n $
असमानता को दूर करने के लिए फूरियर की आवश्यकता है; व्यापारों की अदला-बदली
तीक्ष्णता के लिए \omterm{def:b3:complete:complete}{पूरा} प्राथमिकता.)

\textbf{8.} के लिए $ X_i = 2B_i - 1$ (उचित संकेत): केन्द्रित,
विचरण $1$, और $\abs{X_i} = 1$ इसलिए $\beta = 1$. प्रश्न 7
फिर सीमा $\sup_t\abs{\P(\frac{S_n}{\sqrt n} \leq t) -
\Phi(t)}$ के लिए स्पष्ट रूप से और समान रूप से \emph{प्रत्येक} परिमित
$ n $ --- के बारे में एक वैश्विक, गैर-स्पर्शोन्मुख कथन
वितरण समारोह. का स्थानीय अनुमान
\cref{pb:b3:product:1}, प्रश्न 7, इसके बजाय सटीक देता है
एक व्यक्तिगत परमाणु के स्पर्शोन्मुखता, $\P(S_{2n} = 2k) \sim
\frac{\eu^{-k^2/n}}{\sqrt{\pi n}}$: यह हल करता है
आकार की संभावनाएँ $ n^{-1/2}$, प्रश्न 7 से बहुत नीचे
$ n^{-1/8}$ संकल्प, लेकिन यह बिंदुवार, स्पर्शोन्मुख है
(कोई स्पष्ट त्रुटि निश्चित नहीं है $ n $) और इससे बंधा हुआ
विशेष जाली \omterm{def:b3:probability:space}{कानून}. स्थानीय परिशुद्धता बनाम वैश्विक
एकरूपता: दो विधियाँ पूरक और योग हैं
स्थानीय अनुमान ख़त्म $ k \in \intint{a\sqrt n}{b\sqrt n}$
डे मोइवरे--लाप्लास को अंतराल पर---एक तेज धार के साथ ठीक करता है
दर, लेकिन केवल इस \omterm{def:b3:probability:space}{कानून} के लिए.

\textbf{9.} अदला-बदली के तर्क में \omterm{def:b3:probability:space}{कानून} के बारे में कुछ भी इस्तेमाल नहीं किया गया
का $ X_i $ आगे $\E X_i = 0$, $\E X_i^2 = 1$ और
की परिमितता $\E\abs{X_i}^3$: क्या हमने प्रतिस्थापित कर दिया था \omterm{def:b3:clt:gaussianvector}{गाऊसी}
$ N_i $ किसी अन्य आई.आई.डी. द्वारा\ पहले दो समान परिवार
क्षण और परिमित तीसरा क्षण, वही दूरबीन होगी
बाध्य $\abs{\E f(\text{sum}_X) - \E f(\text{sum}_Y)}$ द्वारा
$ O(n^{-1/2})$ हर चिकनी के लिए $ f $. के चिकने आँकड़े
बड़ा \omterm{def:b3:probability:independence}{स्वतंत्र} रकम इसलिए हैं \emph{सार्वभौमिक}: तक
एक परिमाणित त्रुटि, वे केवल सारांश \omterm{def:b3:probability:space}{कानून} पर निर्भर करते हैं
दो नंबरों के माध्यम से. यह अपरिवर्तनशील सिद्धांत है:
सबसे अधिक गणना योग्य \omterm{def:b3:probability:space}{कानून} के लिए एक सीमा प्रमेय साबित करें (
\omterm{def:b3:clt:gaussianvector}{गाऊसी}, जहां सब कुछ सटीक है), फिर इसे स्थानांतरित करें
सारे \omterm{def:b3:probability:space}{कानून} अदला-बदली से। वही योजना --- बदली गई रकम के साथ
अधिक विस्तृत कार्यात्मकताओं द्वारा --- विग्नर के अर्धवृत्त को चलाता है
यादृच्छिक मैट्रिक्स के लिए \omterm{def:b3:probability:space}{कानून}, यादृच्छिक की जड़ों की सार्वभौमिकता
बहुपद, और अधिकांश आधुनिक संभाव्यता; \omterm{ex:b3:groups:actions}{केंद्र}ीय
सीमा प्रमेय इसका पहला और सरलतम उदाहरण है।

\textbf{10.} $ Z $ का एक बोरेल फ़ंक्शन है $(N_{i+1}, \dots,
N_n)$ केवल, जबकि $ A = W_i + \theta h - Z $ और $ h $ हैं
के शेष चर के कार्य \omterm{def:b3:probability:independence}{स्वतंत्र}
परिवार $(X_1, \dots, X_n, N_1, \dots, N_n)$: द्वारा
गठबंधन सिद्धांत (\cref{thm:b3:probability:independence}),
$ Z $ है \omterm{def:b3:probability:independence}{स्वतंत्र} का $(A, h)$. के योग के रूप में \omterm{def:b3:probability:independence}{स्वतंत्र}
$ N_j/\sqrt n \sim \mathcal N(0, \frac1n)$, $ Z \sim \mathcal
N(0, s^2)$ साथ $ s^2 = \frac{n-i}n $ (\cref{exo:b3:clt:3}),
साथ \omterm{ex:b3:lebesgue:gamma}{घनत्व} से घिरा $\frac1{s\sqrt{2\pi}}$. का \omterm{def:b3:probability:space}{कानून}
$((A, h), Z)$ दो सीमांत \omterm{def:b3:probability:space}{कानून}ों का उत्पाद है, इसलिए
टोनेली (स्थानांतरण) पहले ब्लॉक को फ्रीज करता है: साथ $ G(a) =
\E\abs{g(a + Z)} = \int\abs{g(a + z)}\,\varphi_s(z)\,\dd z
\leq \frac{\norm g_{L^1}}{s\sqrt{2\pi}}$ हरएक के लिए $ a $,
\[
\E\bigl[\abs h^3\abs{g(A + Z)}\bigr] =
\E\bigl[\abs h^3\,G(A)\bigr] \leq
\frac{\norm g_{L^1}}{s\sqrt{2\pi}}\,\E\abs h^3
= \sqrt{\frac{n}{2\pi(n-i)}}\,\norm g_{L^1}\,\E\abs h^3 .
\]

\textbf{11.} टेलर के सूत्र का अभिन्न रूप इस प्रकार है
एकीकृत करके $ f(w + h) - f(w) =
h\int_0^1f'(w + \theta h)\,\dd\theta $ दो बार भागों द्वारा
$\theta $. लेना अपेक्षाएं में $ i $-वें स्वैप, द
आदेश $0, 1, 2$ बिल्कुल प्रश्न 3 की तरह रद्द करें, और
दो शेष (के लिए) $ h = X_i/\sqrt n $
और $ N_i/\sqrt n $) के लिए बाध्य हैं $ i \leq n - 1$, द्वारा
प्रश्न 10 के साथ $ g = f'''$:
\[
\bigl|\E f(H_i) - \E f(H_{i-1})\bigr| \leq
\int_0^1\frac{(1-\theta)^2}2\,\dd\theta\;
\sqrt{\frac{n}{2\pi(n-i)}}\,\norm{f'''}_{L^1}
\frac{\beta + \gamma}{n^{3/2}}
= \frac{\beta + \gamma}{6\,n^{3/2}}
\sqrt{\frac{n}{2\pi(n-i)}}\,\norm{f'''}_{L^1} .
\]
संक्षेप में, साथ में $\sum_{i=1}^{n-1}\sqrt{\frac n{n-i}} =
\sqrt n\sum_{m=1}^{n-1}m^{-1/2} \leq 2n $, और जोड़ना
प्रश्न 3 अंतिम स्वैप के लिए बाध्य है ($ i = n $, नहीं \omterm{def:b3:clt:gaussianvector}{गाऊसी}
बाएं):
\[
\bigl|\E f(T_n) - \E f(G_n)\bigr| \leq
\frac{\beta + \gamma}{3\sqrt{2\pi}}\cdot
\frac{\norm{f'''}_{L^1}}{\sqrt n} +
\frac{M_3(\beta + \gamma)}{6\,n^{3/2}} .
\]
रैम्प: $\psi_\delta'''(x) =
\delta^{-3}\psi'''\bigl(\frac{x - t}\delta\bigr)$, इसलिए
$ M_3 = K\delta^{-3}$ साथ $ K = \norm{\psi'''}_\infty $ और
$\norm{\psi_\delta'''}_{L^1} = \delta^{-2}
\norm{\psi'''}_{L^1} = K_1\delta^{-2}$ (प्रतिस्थापन)। द
फिर प्रश्न 7 का सैंडविच देता है
\[
\sup_t\,\bigl|\P(T_n \leq t) - \Phi(t)\bigr| \leq
\frac{K_1(\beta + \gamma)}{3\sqrt{2\pi}}\cdot
\frac1{\delta^2\sqrt n} +
\frac{K(\beta + \gamma)}{6}\cdot
\frac1{\delta^3n^{3/2}} + \frac\delta{\sqrt{2\pi}} .
\]
पर $\delta = n^{-1/6}$ पहला और तीसरा पद हैं
$ O(n^{-1/6})$ और बीच वाला $ O(n^{-1})$: एक वर्दी
दर $ O(n^{-1/6})$, प्रश्न 7 से बिल्कुल बेहतर
$ n^{-1/8}$ --- द \omterm{def:b3:clt:gaussianvector}{गाऊसी} हाइब्रिड के आधे ने किया
अतिरिक्त चिकनाई.

\textbf{12.} $\E N_1^3 = 0$ (विषम इंटीग्रैंड), और
भागों द्वारा एकीकरण देता है $\E N_1^4 = 3\,\E N_1^2 = 3$
($\int x^3\cdot x\varphi(x)\dd x =
3\int x^2\varphi $). के लिए $ f $ \omterm{def:b3:groups:action}{कक्षा} का $\mathcal C^4$ के साथ
परिबद्ध डेरिवेटिव, प्रत्येक स्वैप को चौथे क्रम तक विस्तारित करें:
तीसरे क्रम के पद कारक को वहन करते हैं $\E X_i^3 - \E N_i^3 =
0$ (स्वतंत्रता उन्हें प्रश्न 3 के अनुसार गुणनखंडित करें), केवल इसलिए
चौथे क्रम का शेष
$\int_0^1\frac{(1-\theta)^3}6f^{(4)}(w + \theta
h)h^4\dd\theta $ जीवित रहता है, साथ $\int_0^1
\frac{(1-\theta)^3}6\dd\theta = \frac1{24}$ और $\E h^4 =
\beta_4n^{-2}$ या $3n^{-2}$. प्रश्न 10 (साथ) $ g =
f^{(4)}$) स्वैप को सीमित करता है $ i \leq n - 1$, और संक्षेप में
प्रश्न 11 में:
\[
\bigl|\E f(T_n) - \E f(G_n)\bigr| \leq
\frac{\beta_4 + 3}{12\sqrt{2\pi}}\cdot
\frac{\norm{f^{(4)}}_{L^1}}{n} +
\frac{M_4(\beta_4 + 3)}{24\,n^2} .
\]
साथ $\norm{\psi_\delta^{(4)}}_{L^1} = K_2\delta^{-3}$ और
$ M_4 = K'\delta^{-4}$, वितरण-फ़ंक्शन बाध्य
बन जाता है $ C\delta^{-3}n^{-1} + C'\delta^{-4}n^{-2} +
\frac\delta{\sqrt{2\pi}}$; पर $\delta = n^{-1/4}$ द
बाहरी शब्द हैं $ O(n^{-1/4})$ और मध्य $ O(n^{-1})$:
दर $ O(n^{-1/4})$.

\textbf{13.} साथ $ k $ बचे हुए क्षणों का मिलान हुआ
प्रति स्वैप शेष क्रम का है $\E\abs h^{k+1} \asymp
n^{-(k+1)/2}$; छिपा हु\omterm{def:b3:clt:gaussianvector}{आ-गाऊसी} बाध्य शुल्क
$\norm{f^{(k+1)}}_{L^1}$ और स्वैप पर योग का योगदान होता है
कारक $2n $, देना $\asymp\norm{f^{(k+1)}}_{L^1}\,
n^{-(k-1)/2}$ चिकनी के लिए $ f $. रैंप की लागत
$\norm{\psi_\delta^{(k+1)}}_{L^1} \asymp \delta^{-k}$, तो
वितरण-फ़ंक्शन त्रुटि है $\asymp
\delta^{-k}n^{-(k-1)/2} + \delta $, पर संतुलित $\delta =
n^{-(k-1)/(2k+2)}$: दर $ n^{-(k-1)/(2k+2)}$, जो है
$ n^{-1/6}$ के लिए $ k = 2$, $ n^{-1/4}$ के लिए $ k = 3$, और प्रवृत्त होता है
को $ n^{-1/2}$ केवल के रूप में $ k \to \infty $ --- लेकिन $ k \geq 4$
जबरदस्ती करेगा $\E X_1^4 = 3$ और परे, यानी\ एक \omterm{def:b3:probability:space}{कानून} कि
पहले से ही नकल करता है \omterm{def:b3:clt:gaussianvector}{गाऊसी}. संतृप्ति है
सं\omterm{def:b3:galois:constructible}{रचनात्मक}: स्वैपिंग जोड़ता है $ n $ स्वैप त्रुटियाँ \emph{में
निरपेक्ष मूल्य}, बीच में सभी रद्दीकरण का त्याग
अदला-बदली। फूरियर प्रमाण तुलना करता है विशिष्ट कार्य,
जहां त्रुटियां उनके दोलन चरणों के साथ प्रकट होती हैं;
एसेन की सहजता असमानता को परिवर्तित करती है $\abs{\varphi_{T_n}
- \varphi_N}$, के विरुद्ध एकीकृत $\frac{\dd\xi}{\abs\xi}$,
केवल लघुगणक पर बंधे वितरण-फ़ंक्शन में
लागत, और बेरी - एसेन की डिलीवरी $ C\beta n^{-1/2}$ से
तीन क्षण. प्रतिस्थापन इष्टतमता का व्यापार करता है
मजबूती --- और, जैसा कि भाग VII दिखाता है, पोर्टेबिलिटी के लिए।

इसका \omterm{def:b3:probability:space}{कानून} है $\mathcal N(0, \Sigma)$

इन आंकड़ों द्वारा)।

\textbf{15.} होने देना $\phi(t) = f(w + th)$, $ t \in \intcc01$:
$\phi $ है $\mathcal C^3$ साथ
\[
\phi'''(t) = \sum_{j,k,l\in\{1,2\}}\partial_{jkl}f(w +
th)\,h_jh_kh_l, \qquad \abs{\phi'''(t)} \leq
M_3\Bigl(\sum_j\abs{h_j}\Bigr)^3 = M_3(\abs{h_1} +
\abs{h_2})^3 .
\]
टेलर--लैग्रेंज क्रम में $3$ के लिए $\phi $ बीच में $0$ और
$1$ पहली असमानता देता है; कॉची-श्वार्ज़ देता है
$\abs{h_1} + \abs{h_2} \leq \sqrt2\norm h $, कहाँ से
\omterm{def:b3:ode:stability}{स्थिर} $\frac{2\sqrt2M_3}6 = \frac{\sqrt2M_3}3$.

\textbf{16.} परिभाषित करना $ H_i $ और $ W_i $ जैसा कि अब प्रश्न 1 में है
में $\R^2$; गठबंधन का तर्क अपरिवर्तित है. में
$ i $-वें स्वैप, प्रथम-क्रम शर्तें देते हैं
$\sum_j\E[\partial_jf(W_i)]\,(\E X_{i,j} - \E N_{i,j})/
\sqrt n = 0$ और दूसरे क्रम के पद देते हैं
$\frac1{2n}\sum_{j,k}\E[\partial_{jk}f(W_i)]\,(\Sigma_{jk}
- \Sigma_{jk}) = 0$: साधन और सहप्रसरण मेल खाते हैं। प्रश्न
15 दो शेषफलों को सीमाबद्ध करता है:
\[
\bigl|\E f(H_i) - \E f(H_{i-1})\bigr| \leq
\frac{\sqrt2M_3}3\cdot\frac{\E\norm{X_i}^3 +
\E\norm{N_i}^3}{n^{3/2}}
= \frac{\sqrt2M_3(\beta' + \gamma')}{3\,n^{3/2}},
\]
और दूरबीन से $ n $ स्वैप:
\[
\Bigl|\E f\Bigl(\frac{S_n}{\sqrt n}\Bigr) - \E
f(G_n)\Bigr| \leq \frac{\sqrt2\,M_3(\beta' +
\gamma')}{3\sqrt n},
\qquad G_n \sim \mathcal N(0, \Sigma)\ \text{exactly} .
\]
उन्नत करना: $\E\norm{T_n}^2 = \E\norm{X_1}^2 =
\operatorname{tr}\Sigma $ (क्रॉस टर्म गायब हो जाते हैं
स्वतंत्रता और केन्द्रित करना), तो $\P(\norm{T_n} > A) \leq
\operatorname{tr}\Sigma/A^2$, और इसी तरह के लिए $ N $:
जकड़न. एक सीमा दी गई \omterm{def:b3:topology:continuity}{निरंतर} $ f $ और $\varepsilon
> 0$, एक चिकने पठार से गुणा करें $\chi $ के बराबर $1$ पर
त्रिज्या की गेंद $ A $ और त्रिज्या में समर्थित है $ A + 1$
(एक संकेतक को नरम करें $\R^2$,
\cref{thm:b3:lp:regularization}); $ g = f\chi $ समान रूप से है
\omterm{def:b3:topology:continuity}{निरंतर} साथ कॉम्पैक्ट समर्थन, इसलिए यह द्वि-आयामी है
मोलीकरण $ g_\eta $ है $\mathcal C^\infty $ बाध्य के साथ
सभी आदेशों के व्युत्पन्न और $\norm{g - g_\eta}_\infty
\leq \varepsilon $ छोटे के लिए $\eta $. तीन-$\varepsilon $
प्रश्न 5 की श्रृंखला तब शब्दशः स्थानांतरित होती है: $\E f(T_n)
\to \E f(N)$ प्रत्येक सीमा के लिए \omterm{def:b3:topology:continuity}{निरंतर} $ f \colon \R^2
\to \R $. यह है \cref{thm:b3:clt:multiclt} के लिए $ d = 2$,
अब साबित हो गया है --- अदला-बदली द्वि-आयामी को दरकिनार कर देती है
एल\'ईवी प्रमेय जिसे अध्याय ने छोड़ दिया था उसे स्वीकार कर लिया गया।

\textbf{17.} बाध्य के लिए \omterm{def:b3:topology:continuity}{निरंतर} $ g \colon \R \to \R $,
वो नक्शा $ x \mapsto g(\langle t, x\rangle)$ घिरा है

प्रक्षेपण वितरण में परिवर्तित होता है, और $\langle t,
N\rangle \sim \mathcal N(0, t^{\mathsf T}\Sigma t)$. (यह
क्रैम की आसान दिशा है\'एर--वॉल्ड: संयुक्त अभिसरण
तात्पर्य सभी रैखिक छवियों के अभिसरण से है।) अनुप्रयोग:
$ V_i = (\xi_i, \xi_i^2 - 1)$ क्या आई.आई.डी.\ केन्द्रित सदिश
($\E\xi_1^2 = 1$), सहप्रसरण प्रविष्टियों के साथ $\V(\xi_1) =
1$, $\operatorname{Cov}(\xi_1, \xi_1^2 - 1) = \E\xi_1^3$
और $\V(\xi_1^2 - 1) = \E\xi_1^4 - 1$; तीसरा क्षण
$\E\norm{V_1}^3 \leq 4\bigl(\E\abs{\xi_1}^3 +

सीमा, और \cref{thm:b3:clt:gaussianvector}(2): दो
सीमा निर्देशांक हैं \omterm{def:b3:probability:independence}{स्वतंत्र} बिल्कुल जब
सहप्रसरण $\E\xi_1^3$ गायब हो जाता है --- सममित \omterm{def:b3:probability:space}{कानून}ों के लिए,
अनुभवजन्य माध्य और अनुभवजन्य विचरण को अलग करना
स्पर्शोन्मुख रूप से।

\textbf{18.} लिखना $ g(x) - g(\theta) = (g'(\theta) +
\eta(x))(x - \theta)$ कहाँ $\eta(x) = \frac{g(x) -
g(\theta)}{x - \theta} - g'(\theta)$ के लिए $ x \neq \theta $
और $\eta(\theta) = 0$: भिन्नता पर $\theta $
सटीक मतलब है $\eta(x) \to 0$ जैसा $ x \to \theta $.
\emph{स्टेप 1:} $\hat\theta_n \to \theta $ संभाव्यता में:
के लिए $\varepsilon > 0$ और कोई भी $ A > 0$, अंततः
$\varepsilon\sqrt n \geq A $, इसलिए $\P(\abs{\hat\theta_n -
\theta} > \varepsilon) \leq \P(\abs{\sqrt
n(\hat\theta_n - \theta)} > A) \to \P(\sigma\abs N > A)$
(वितरण कार्य एकाग्र होते हैं निरंतरता अंक
$\pm A $), और दाहिनी ओर झुक जाता है $0$ जैसा $ A \to
\infty $. \emph{चरण दो:} $\eta(\hat\theta_n) \to 0$ में
संभाव्यता: दिया गया $\varepsilon' > 0$, चुनना $\delta $ साथ
$\abs\eta \leq \varepsilon'$ पर $\abs{x - \theta} \leq
\delta $; तब $\P(\abs{\eta(\hat\theta_n)} > \varepsilon')
\leq \P(\abs{\hat\theta_n - \theta} > \delta) \to 0$.
\emph{चरण 3:}
\[
\sqrt n\bigl(g(\hat\theta_n) - g(\theta)\bigr) =
g'(\theta)\,\sqrt n(\hat\theta_n - \theta) +
\eta(\hat\theta_n)\cdot\sqrt n(\hat\theta_n - \theta) .
\]
स्लटस्की के उत्पाद नियम द्वारा (\cref{exo:b3:clt:8}, के साथ
क्रम $\eta(\hat\theta_n) \to 0$ संभाव्यता में और
अभिसरण-ससुराल $\sqrt n(\hat\theta_n - \theta)$), द
दूसरा पद \omterm{def:b3:probability:space}{कानून} में अभिसरण करता है $0\cdot\mathcal N(0,
\sigma^2) = 0$, इसलिए करने के लिए $0$ संभावना में
(\cref{exo:b3:clt:4}(बी)); पहला \omterm{def:b3:probability:space}{कानून} में अभिसरण करता है
$ g'(\theta)\mathcal N(0, \sigma^2)$ (स्लटस्की फिर से, या
एफ़िन नियम के लिए विशिष्ट कार्य); स्लटस्की का योग
नियम उन्हें जोड़ता है: सीमा है $\mathcal N(0,
g'(\theta)^2\sigma^2)$.

\textbf{19.} (ए) सीएलटी देता है $\sqrt n(\bar X_n - \mu)

\[
\P\bigl(n\bar X_n^2 \leq t\bigr) = \P\bigl(-\sqrt t \leq

$ n\bar X_n^2 \Rightarrow \sigma^2N^2$, ए का वर्ग

\textbf{20.} होने देना $ A^* = \{k : \mu(\{k\}) > \nu(\{k\})\}$
और $\Delta_k = \mu(\{k\}) - \nu(\{k\})$, इसलिए
$\sum_k\Delta_k = 0$. किसी के लिए $ A \subseteq \N $: $\mu(A) -
\nu(A) = \sum_{k\in A}\Delta_k \leq \sum_{k\in
A^*}\Delta_k $, समानता के साथ $ A = A^*$; और तब से
के सकारात्मक और नकारात्मक भाग $(\Delta_k)$ बराबर है
कुल द्रव्यमान, $\sum_{A^*}\Delta_k =
\frac12\sum_k\abs{\Delta_k}$. का आदान प्रदान $\mu, \nu $ हैंडल
संकेत: $ d_{\mathrm{TV}}(\mu, \nu) =
\frac12\sum_k\abs{\Delta_k}$. युग्मन: किसी के लिए $ A $,
\[
\mu(A) - \nu(A) = \E\bigl[\mathbf 1_A(X) - \mathbf
1_A(Y)\bigr] = \E\bigl[(\mathbf 1_A(X) - \mathbf
1_A(Y))\,\mathbf 1_{X\neq Y}\bigr] \leq \P(X \neq Y),
\]
और सर्वोच्च को अपने अधिकार में ले लो $ A $.

\textbf{21.} दो \omterm{def:b3:probability:space}{कानून} आरोप लगाते हैं: $ k = 0$: $1 - p $ बनाम
$\eu^{-p}$, साथ $\eu^{-p} > 1 - p $; $ k = 1$: $ p $ बनाम
$ p\,\eu^{-p} < p $; $ k \geq 2$: $0$ पॉइसन बनाम
शेष $1 - \eu^{-p} - p\eu^{-p} \geq 0$. इस तरह
\[
d_{\mathrm{TV}} = \tfrac12\bigl[(\eu^{-p} - 1 + p) + (p -
p\eu^{-p}) + (1 - \eu^{-p} - p\eu^{-p})\bigr] =
\tfrac12\bigl(2p - 2p\eu^{-p}\bigr) = p(1 - \eu^{-p}),
\]
और $1 - \eu^{-p} \leq p $ बाउंड देता है $ p^2$.

\textbf{22.} लिखना $ H_{i-1} = W_i + X_i $ और $ H_i = W_i +
Y_i $ साथ $ W_i = \sum_{j<i}Y_j + \sum_{j>i}X_j $,
\omterm{def:b3:probability:independence}{स्वतंत्र} जोड़ी का $(X_i, Y_i)$ (गठबंधन)। के लिए $ A
\subseteq \N $, अनगिनत मूल्यों पर कंडीशनिंग
द्वारा स्वतंत्रता,
\[
\P(H_{i-1} \in A) = \sum_{k\geq0}\P(X_i = k)\,\P(W_i + k
\in A),
\]
और इसी तरह के लिए $ H_i $ साथ $ Y_i $. घटाना, के साथ $ c_k
= \P(W_i + k \in A) \in \intcc01$ और $\Delta_k = \P(X_i =
k) - \P(Y_i = k)$ शून्य योग का:
\[
\abs{\P(H_{i-1} \in A) - \P(H_i \in A)} =
\Bigl|\sum_k\Delta_k\bigl(c_k - \tfrac12\bigr)\Bigr| \leq
\tfrac12\sum_k\abs{\Delta_k} =
d_{\mathrm{TV}}\bigl(\mathcal B(1, p_i), \mathcal
P(p_i)\bigr) .
\]
टेलीस्कोपिंग से $ H_0 = S $ को $ H_n = \sum_iY_i \sim
\mathcal P(\lambda)$ (\cref{exo:b3:clt:1}, पुनरावृत्त) और
प्रश्न 21 का उपयोग करते हुए:
\[
\abs{\P(S \in A) - \P(\mathcal P(\lambda) \in A)} \leq
\sum_{i=1}^np_i\bigl(1 - \eu^{-p_i}\bigr) \leq
\sum_{i=1}^np_i^2
\qquad\text{for every } A :
\]
ले कैम की असमानता. (प्रश्न 20 का युग्म बाउंड एक देता है
वैकल्पिक मार्ग: प्रत्येक जोड़ी को एक समान पर जोड़े
परिवर्तनीय ताकि $\P(X_i \neq Y_i) \leq p_i^2$ और बाध्य
$\P(S \neq \sum Y_i)$; स्वैपिंग के लिए किसी निर्माण की आवश्यकता नहीं है
सभी.)

p(1-p)\bigl[(1-p)^2 - p^2\bigr] = p(1-p)(1 - 2p)$. में
स्वैपिंग विश्लेषण (भाग IV), इसके बाद प्रमुख त्रुटि शब्द
दो क्षणों का मिलान वहन करता है \emph{पर हस्ताक्षर किए} तीसरा
पल: के लिए $ p < \frac12$ यह सकारात्मक है (\omterm{def:b3:probability:space}{कानून} झुकता है)।
दाएं: औसत से ऊपर दुर्लभ बड़े भ्रमण), और
\omterm{def:b3:groups:normal}{सामान्य} सन्निकटन व्यवस्थित रूप से द्रव्यमान का गलत स्थान रखता है ---
छोटी बायीं पूँछ को कम आँकना और अधिक आँकना
सही --- आदेश की त्रुटि के साथ $ n^{-1/2}$; पर $ p =
\frac12$ तीसरा क्षण गायब हो जाता है, बर्नौली मेल खाता है
द \omterm{def:b3:clt:gaussianvector}{गाऊसी} तीसरे क्रम तक, और दर में सुधार होता है (भाग)।
IV का मिलान-क्षण प्रश्न)। संख्यात्मक रूप से: $\P(S = 0) =
0.9^{20} = 0.1216$, जब \omterm{def:b3:clt:gaussianvector}{गाऊसी}
$\mathcal N(2, 1.8)$ देता है $\Phi\bigl(\frac{0.5 -
2}{\sqrt{1.8}}\bigr) = \Phi(-1.118) \approx 0.132$: द
\omterm{def:b3:groups:normal}{सामान्य} वक्र, दीवार से अनभिज्ञ $0$ और का
दाहिनी ओर तिरछा होना, तल पर बहुत अधिक द्रव्यमान डालता है ---
त्रुटि का अनुमानित संकेत, पर दिखाई दे रहा है $ n = 20$.
\end{solution}
