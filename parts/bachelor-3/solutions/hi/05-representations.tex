\section*{अध्याय \ref{ch:b3:representations} --- अभ्यावेदन
परिमित समूहों का}

\begin{solution}{exo:b3:representations:1}
(ए) $\Z/n\Z $ एबेलियन है: सभी इरेड्यूसिबल्स \omterm{def:b3:galois:extension}{डिग्री} है $1$
(\cref{prop:b3:representations:onedim}), यानी\ रूपवाद हैं
$\chi \colon \Z/n\Z \to \C^\times $, द्वारा निर्धारित $\chi(\bar 1)
= \omega $ साथ $\omega^n = 1$: द $ n $ अक्षर $\chi_k(\bar
m) = \eu^{2\iu\pi km/n}$. के लिए $ n = 4$ (कक्षाएं $= $ तत्वों
$\bar0,\bar1,\bar2,\bar3$):
\[
\begin{array}{c|cccc}
& \bar0 & \bar1 & \bar2 & \bar3\\
\hline
\chi_0 & 1 & 1 & 1 & 1\\
\chi_1 & 1 & \iu & -1 & -\iu\\
\chi_2 & 1 & -1 & 1 & -1\\
\chi_3 & 1 & -\iu & -1 & \iu
\end{array}
\]
(बी) पंक्तियाँ: $\langle\chi_k,\chi_l\rangle = \frac14\sum_m
\eu^{2\iu\pi(l-k)m/4} = \delta_{kl}$ (ज्यामितीय योग); कॉलम
इसी तरह. गणित का सवाल $(\eu^{2\iu\pi km/n})_{k,m}$ है
\emph{असतत फूरियर रूपांतरण} मैट्रिक्स --- वही
रूट्स-ऑफ़-यूनिटी फ़िल्टर का उपयोग वर्ष 2 खंड में किया गया है
सृजन-कार्य अध्याय; वर्णों की रूढ़िवादिता
डीएफटी के व्युत्क्रम सूत्र को \omterm{def:b3:groups:normal}{सामान्य}ीकृत करता है।
\end{solution}

\begin{solution}{exo:b3:representations:2}
(ए) का मैट्रिक्स $\rho(g)$ आधार में $(e_x)$ एक है
क्रमपरिवर्तन मैट्रिक्स, की संख्या का पता लगाएं $ x $ साथ $ g\cdot x =
x $. तब
\[
\langle\mathbf 1, \chi\rangle = \frac1{\abs G}\sum_g
\abs{\operatorname{Fix}(g)} = \#\{\text{orbits}\}
\]
बर्नसाइड की काउंटिंग लेम्मा द्वारा (\cref{exo:b3:groups:5}) ---
समान रूप से, यह तुच्छता की बहुलता की गणना करता है
\omterm{def:b3:representations:rep}{प्रतिनिधित्व}, जिसका समप्रतिरूपी स्थान का स्थान है
$ G $-अपरिवर्तनीय वैक्टर, आयाम की संख्या कक्षाओं (एक
सूचक प्रति \omterm{def:b3:groups:action}{कक्षा}).

(बी) चूंकि $\operatorname{Fix}_{X\times X}(g) =
\operatorname{Fix}_X(g)^2$, भाग (ए) पर लागू होता है $ X \times X $
देता है $\langle\chi,\chi\rangle = \frac1{\abs
G}\sum\abs{\operatorname{Fix}(g)}^2 = \#\{\text{orbits on }
X\times X\}$ ($\chi $ यह सचमुच का है)। लिखना $\chi = \mathbf 1 +
\psi $: $\langle\mathbf1,\chi\rangle = 1$ (परिवर्तनशीलता), इसलिए
$\langle\psi,\psi\rangle = \langle\chi,\chi\rangle - 1$. द
पर \omterm{def:b3:groups:action}{कार्रवाई} $ X\times X $ विकर्ण एक है \omterm{def:b3:groups:action}{कक्षा}; वहाँ है
बिल्कुल एक दूसरे \omterm{def:b3:groups:action}{कक्षा} आईएफएफ $ G $ अलग-अलग युग्मों पर सकर्मक है:
$\langle\psi,\psi\rangle = 1$ आईएफएफ $2$-सकर्मक
(\cref{cor:b3:representations:multiplicity}).

(सी) $ S_n $ है $2$-सकर्मक चालू $\{1,\dots,n\}$ (कोई भी भेजें
कहीं भी अलग जोड़ी): $\psi = \chi_{\mathrm{std}}$ है
\omterm{def:b3:representations:rep}{अलघुकरणीय}.
\end{solution}

\begin{solution}{exo:b3:representations:3}
$ S_3$: तीन वर्ग, $\sum n_i^2 = 6 = 1 + 1 + 4$; दो
डिग्री-$1$ अक्षर हैं $\mathbf 1, \varepsilon $ ($[S_3 :
A_3] = 2$); तीसरी पंक्ति $(2, a, b)$ कॉलम से अनुसरण करता है
के साथ रूढ़िवादिता $ e $-स्तंभ: $1 - 1 + 2a = 0$ और $1 + 1 +
2b = 0$: $ a = 0$, $ b = -1$ --- की तालिका
\cref{ex:b3:representations:s3}.

$ S_4$ जाँच (पंक्तियाँ): $\langle\chi_{\mathrm{std}},
\varepsilon\chi_{\mathrm{std}}\rangle = \frac1{24}(9 - 6 + 3 + 0
- 6) = 0$; $\langle\chi_2,\chi_2\rangle = \frac1{24}(4 + 0 + 12
+ 8 + 0) = 1$. कॉलम: $ e $ ख़िलाफ़ $(1\,2)$: $1 - 1 + 0 + 3 - 3
= 0$; $(1\,2)$ स्वयं के विरुद्ध: $1 + 1 + 0 + 1 + 1 = 4 =
\abs{Z_{S_4}((1\,2))} = 24/6$.

परिवर्तन \omterm{def:b3:representations:character}{चरित्र} पर $4$ अंक: $(4, 2, 0, 1, 0) = \mathbf
1 + \chi_{\mathrm{std}}$ (निश्चित-बिंदु गिनती; शीर्ष घटाएं
पंक्ति). के लिए $\chi_{\mathrm{std}}^2 = (9, 1, 1, 0, 1)$:
\[
\langle\mathbf1,\cdot\rangle = \tfrac{9 + 6 + 3 + 0 + 6}{24} =
1,\quad
\langle\varepsilon,\cdot\rangle = 0,\quad
\langle\chi_2,\cdot\rangle = \tfrac{18 + 6}{24} = 1,\quad
\langle\chi_{\mathrm{std}},\cdot\rangle = 1,\quad
\langle\varepsilon\chi_{\mathrm{std}},\cdot\rangle = 1:
\]
$\chi_{\mathrm{std}}^2 = \mathbf 1 + \chi_2 +
\chi_{\mathrm{std}} + \varepsilon\chi_{\mathrm{std}}$
(आयाम: $9 = 1 + 2 + 3 + 3$).
\end{solution}

\begin{solution}{exo:b3:representations:4}
दोनों समूहों में पांच कक्षाएं और \omterm{def:b3:galois:extension}{डिग्री} पैटर्न हैं $(1,1,1,1,2)$
(चार डिग्री-$1$ एबेलियनाइजेशन से है $\cong (\Z/2\Z)^2$,
तब $\sum n_i^2 = 8$). कक्षाओं का आदेश देना $ e $, $ z $ (द
\omterm{ex:b3:groups:actions}{केंद्र}ीय समावेशन: $ r^2$, सम्मान।\ $-1$), और तीन
दो-तत्व वर्ग:
\[
\begin{array}{c|ccccc}
& e & z & C_1 & C_2 & C_3\\
\hline
\chi^{(1)} & 1 & 1 & 1 & 1 & 1\\
\chi^{(2)} & 1 & 1 & 1 & -1 & -1\\
\chi^{(3)} & 1 & 1 & -1 & 1 & -1\\
\chi^{(4)} & 1 & 1 & -1 & -1 & 1\\
\chi^{(5)} & 2 & -2 & 0 & 0 & 0
\end{array}
\]
(स्तंभ ऑर्थोगोनैलिटी से अंतिम पंक्ति)। के लिए समान तालिकाएँ
$ D_4$ और $ Q_8$, जो समरूपी नहीं हैं
(\cref{pb:b3:groups:1}). टेबल \emph{करता है} कब्जा: $\abs
G $, वर्ग आकार, द \omterm{ex:b3:groups:actions}{केंद्र} ($\{g : \abs{\chi_i(g)} = n_i\
\forall i\}$: आदेश $2$ दोनों में), अबेलियनाइजेशन, संपूर्ण
की जाली \omterm{def:b3:groups:normal}{सामान्य} उपसमूह (गुठली और चौराहे,
\cref{exo:b3:representations:6}). यह तत्व को पकड़ने में विफल रहता है
आदेश: $ D_4$ पाँच अवतरण हैं, $ Q_8$ एक है --- तो
समरूपता प्रकार वास्तव में की तुलना में बेहतर है \omterm{def:b3:representations:table}{चरित्र तालिका}.
\end{solution}

\begin{solution}{exo:b3:representations:5}
(ए) में $ S_4$, का \omterm{ex:b3:groups:actions}{केंद्रीकरणकर्ता} $(1\,2\,3)$ आदेश है $24/8 =
3$: यह है $\langle(1\,2\,3)\rangle \subseteq A_4$. इसलिए
$ Z_{A_4}((1\,2\,3))$ आदेश है $3$ और यह $ A_4$-क्लास है $12/3
= 4$ तत्व: आठ $3$-साइकिलें दो हिस्सों में बंट गईं $ A_4$-कक्षाएँ
(द्वारा \omterm{def:b3:representations:rep}{प्रतिनिधित्व} किया गया $(1\,2\,3)$ और इसका उलटा)।

(बी) $ A_4/V \cong \Z/3\Z $ तीन \omterm{def:b3:galois:extension}{डिग्री} देता है-$1$ अक्षर
($\omega = \eu^{2\iu\pi/3}$; $3$-साइकिल कक्षाओं का मानचित्र
$\bar1, \bar2$); का प्रतिबंध $\chi_{\mathrm{std}}$
रहता है \omterm{def:b3:representations:rep}{अलघुकरणीय} ($\langle\chi,\chi\rangle = \frac1{12}(9 + 3
\cdot 1 + 0 + 0) = 1$):
\[
\begin{array}{c|cccc}
A_4 & e\,[1] & (1\,2)(3\,4)\,[3] & (1\,2\,3)\,[4] &
(1\,3\,2)\,[4]\\
\hline
\mathbf 1 & 1 & 1 & 1 & 1\\
\chi_\omega & 1 & 1 & \omega & \omega^2\\
\chi_{\bar\omega} & 1 & 1 & \omega^2 & \omega\\
\chi_3 & 3 & -1 & 0 & 0
\end{array}
\]
(सी) गुठली: $\ker\chi_\omega = \ker\chi_{\bar\omega} = V $;
$\ker\chi_3 = \{e\}$ (किसी अन्य प्रविष्टि में मापांक नहीं है $3$).
\omterm{def:b3:groups:normal}{सामान्य} उपसमूह गुठली के प्रतिच्छेदन हैं
(\cref{exo:b3:representations:6}): $\{e\}$, $ V $, $ A_4$ --- विशेष रूप से $ A_4$ है कोई \omterm{def:b3:groups:normal}{सामान्य} उपसमूह आदेश की $2$ या सूचकांक
$2$.
\end{solution}

\begin{solution}{exo:b3:representations:6}
(ए) यदि $\chi(g) = \chi(e) = n $: में समानता
$\abs{\chi(g)} \leq n $ बल $\rho(g) = \lambda\,\mathrm{id}$
(\cref{prop:b3:representations:charbasics}) साथ $ n\lambda = n $:
$\rho(g) = \mathrm{id}$. बातचीत स्पष्ट है.

(बी) चलो $ N \trianglelefteq G $. \omterm{ex:b3:representations:examples}{नियमित प्रतिनिधित्व} का
$ G/N $ वफादार है; इसे विघटित करें इरेड्यूसिबल्स का $ G/N $ और
उन्हें उठाओ $ G $ (\cref{prop:b3:representations:onedim}(बी)):
\omterm{def:b3:representations:rep}{अलघुकरणीय} अक्षर $\chi_{i_1}, \dots $ का $ G $ जिसकी गुठली
सम्मिलित $ N $ और किसका \emph{\omterm{def:b3:groups:normal}{सामान्य}} कर्नेल वास्तव में है
की पूर्वछवि $\{e\}$, यानी\ $ N $ (भागफल पर निष्ठा)।
इसलिए $ N = \bigcap_j \ker\chi_{i_j}$.

(सी) यदि $ G $ \omterm{def:b3:groups:simple}{सरल} है: एक गैरतुच्छ के लिए \omterm{def:b3:representations:rep}{अलघुकरणीय} $\chi $,
$\ker\chi \trianglelefteq G $ क्या नहीं है $ G $ (एक अपरिवर्तनीय
\omterm{def:b3:representations:rep}{प्रतिनिधित्व} सब पर तुच्छ $ G $ तुच्छ है \omterm{def:b3:representations:character}{चरित्र}),
इसलिए $\ker\chi = \{e\}$. इसके विपरीत, मान लीजिए कि सभी गैर-तुच्छ हैं
गुठलियाँ तुच्छ हैं, और चलो $ N \trianglelefteq G $ साथ $ N \neq
G $. (बी) की अभिव्यक्ति में $ N $ गुठली के प्रतिच्छेदन के रूप में,
कुछ \omterm{def:b3:representations:character}{चरित्र} इसमें शामिल गैर-तुच्छ है (यदि सभी तुच्छ थे, तो
चौराहा होगा $ G $), और इसका कर्नेल है $\{e\}$: $ N =
\{e\}$. तो एकमात्र \omterm{def:b3:groups:normal}{सामान्य} उपसमूह हैं $\{e\}$ और $ G $.
\end{solution}

\begin{solution}{exo:b3:representations:7}
आदेश $8$ नॉनबेलियन: \omterm{def:b3:galois:extension}{डिग्री} की संख्या-$1$ अक्षर है
$[G : D(G)]$, का एक उचित विभाजक $8$ (नॉनबेलियन: $ D(G) \neq
\{e\}$), और $\sum n_i^2 = 8$. साथ $ k $ एक और शेष
\omterm{def:b3:galois:extension}{डिग्री} $\geq 2$: $8 - k \equiv 0$ वर्गों के साथ $\geq 4$, और $ k
\mid 8$, $ k < 8$. $ k = 4$: एक \omterm{def:b3:galois:extension}{डिग्री} $2$ --- सुसंगत। $ k =
2$: $6$ बाएँ, वर्गों का योग नहीं $\geq 4$. $ k = 1$:
असंभव, तब से $ k = [G : D(G)] \geq 2$ --- $ G/D(G)$ एक है
गैरतुच्छ एबेलियन $2$-समूह, जैसे $ G $ एक है $2$-समूह के साथ
$ D(G) \neq G $ द्वारा गलने की योग्यता का $ p $-समूह
(\cref{ex:b3:groups:solvableexamples}). तो पैटर्न है
$(1,1,1,1,2)$. की वफ़ादारी $\chi_5$: चार डिग्री-$1$
अक्षर सभी शामिल हैं $ D(G)$ उनकी गुठली में; अगर $\ker\chi_5
\supseteq N \neq \{e\}$ कुछ न्यूनतम \omterm{def:b3:groups:normal}{सामान्य} के लिए $ N $ ($ N
\subseteq D(G)$ या नहीं---ले जाओ $ N \subseteq \ker\chi_5$
गैरतुच्छ), फिर $ N $ पाँचों गुठलियों में पड़ा होगा, किसका
चौराहा तुच्छ है ( \omterm{ex:b3:representations:examples}{नियमित प्रतिनिधित्व} है
वफ़ादार): विरोधाभास. आदेश $ p^3$ नॉनबेलियन: $ Z(G)$ है
आदेश $ p $ (आदेश $ p^2$ करना होगा $ G/Z $ चक्रीय, $ G $ एबेलियन),
$ G/Z(G)$ आदेश की $ p^2$ एबेलियन है, इसलिए $ D(G) \subseteq Z(G)$,
और $ D(G) \ne \{e\}$: $ D(G) = Z(G)$, देना $ p^2$ अक्षर का
\omterm{def:b3:galois:extension}{डिग्री} $1$. शेष डिग्रियाँ संतुष्ट करती हैं $\sum n_i^2 = p^3 -
p^2$ प्रत्येक के साथ $ n_i > 1$ डिवाइडिंग $\abs G $
(\cref{pb:b3:representations:1}, प्रश्न 8) इसलिए $ n_i \in
\{p\}$ ($ n_i = p^2$ से अधिक होगा: $ p^4 > p^3 - p^2$): बिल्कुल $ p
- 1$ अक्षर \omterm{def:b3:galois:extension}{डिग्री} का $ p $.
\end{solution}

\begin{solution}{exo:b3:representations:8}
परिभाषित करें, के लिए अभ्यावेदन $\rho, \sigma $ का $ G, H $ में $ V, W $,
द \omterm{def:b3:representations:rep}{प्रतिनिधित्व} $\rho \boxtimes \sigma $ का $ G \times H $ पर
$ V \otimes W $ --- ठोस रूप से, मैट्रिक्स पर: $(\rho\boxtimes
\sigma)(g,h)$ क्रोनकर उत्पाद है $\rho(g)\otimes
\sigma(h)$, जिसका निशान है
$\operatorname{tr}\rho(g)\operatorname{tr}\sigma(h) =
\chi(g)\psi(h)$ (मैट्रिसेस का क्रोनकर उत्पाद $ A \otimes B $
निशान है $\operatorname{tr}A\operatorname{tr}B $: इसका विकर्ण
है $ a_{kk}b_{ll}$). इसलिए $\chi\psi $ एक है \omterm{def:b3:representations:character}{चरित्र}, और
\[
\langle\chi\psi, \chi'\psi'\rangle_{G\times H}
= \frac{1}{\abs G\abs H}\sum_{g,h}
\overline{\chi(g)\psi(h)}\,\chi'(g)\psi'(h)
= \langle\chi,\chi'\rangle_G\,\langle\psi,\psi'\rangle_H
= \delta_{\chi\chi'}\delta_{\psi\psi'}.
\]
विशेष रूप से $\langle\chi\psi,\chi\psi\rangle = 1$: प्रत्येक
$\chi\psi $ है \omterm{def:b3:representations:rep}{अलघुकरणीय}
(\cref{cor:b3:representations:multiplicity}). ये हैं
$ r_Gr_H $ विशिष्ट \omterm{def:b3:representations:rep}{अलघुकरणीय} अक्षर; की कक्षाएं
$ G\times H $ वर्गों के उत्पाद हैं ($(g,h) \sim (g',h')$
घटकवार), तो वहाँ हैं $ r_Gr_H $ उनमें से: सूची है
पूरा (\cref{thm:b3:representations:numberirr}). के लिए
$(\Z/2\Z)^2$: चार चिह्न अक्षर
$(\pm1)\otimes(\pm1)$ --- की तालिका
\cref{exo:b3:representations:4}शीर्ष-बाएँ ब्लॉक. एक परिमित
एबेलियन समूह चक्रीय समूहों का एक उत्पाद है
(\cref{cor:b3:modules:abelian}); इसका \omterm{def:b3:representations:rep}{अलघुकरणीय} अक्षर हैं
चक्रीय वाले के उत्पाद (\cref{exo:b3:representations:1}):
सभी \omterm{def:b3:galois:extension}{डिग्री} के $1$.
\end{solution}

\begin{solution}{exo:b3:representations:9}
(ए) $ D(A_5) = A_5$ (\omterm{def:b3:groups:simple}{सरल} नॉनबेलियन), तो एकमात्र डिग्री-$1$
\omterm{def:b3:representations:character}{चरित्र} है $\mathbf 1$
(\cref{prop:b3:representations:onedim}). ज़रुरत है $ n_2^2 + n_3^2
+ n_4^2 + n_5^2 = 59$ प्रत्येक के साथ $ n_i \geq 2$; का परीक्षण
वर्ग $4, 9, 16, 25, 36, 49$: एकमात्र मल्टीसेट जो काम करता है
$\{9, 9, 16, 25\}$: सबसे बड़े वर्ग के साथ $49$, शेष $10$
तीन वर्गों का योग नहीं है $\geq 4$; साथ $36$, शेष

\begin{solution}{pb:b3:representations:1}
\textbf{1.} अगर $\alpha^n + c_{n-1}\alpha^{n-1} + \dots + c_0 =
0$ ($ c_i \in \Z $), तब $\alpha^n \in \Z\text{-span}(1, \dots,
\alpha^{n-1})$, और आगमनात्मक रूप से प्रत्येक शक्ति है: $\Z[\alpha]$ है
द्वारा उत्पन्न $1, \alpha, \dots, \alpha^{n-1}$. इसके विपरीत चलो
$\Z[\alpha] = \Z g_1 + \dots + \Z g_m $. लिखना $\alpha g_i =
\sum_j m_{ij}g_j $ साथ $ M = (m_{ij}) \in M_m(\Z)$: वेक्टर
$ g = (g_i)$ संतुष्ट $(\alpha I - M)g = 0$; से गुणा करना
सहायक मैट्रिक्स, $\det(\alpha I - M)\,g_i = 0$ हरएक के लिए $ i $,
और तबसे $1 \in \Z[\alpha]$ एक है $\Z $-का संयोजन
$ g_i $, $\det(\alpha I - M) = 0$: $\alpha $ की जड़ है
राक्षसी $\det(XI - M) \in \Z[X]$.

\textbf{2.} अगर $\Z[\alpha]$ द्वारा फैलाया गया है $\alpha $ की शक्तियाँ ऊपर
को $n-1$ और $\Z[\beta]$ द्वारा $\beta $ तक है $m - 1$, तब
$\Z[\alpha,\beta]$ द्वारा फैलाया गया है $ nm $ उत्पादों
$\alpha^i\beta^j $ (किसी भी एकपदी को कम करें)। उपछलियाँ
$\Z[\alpha + \beta]$ और $\Z[\alpha\beta]$ हैं $\Z $-सबमॉड्यूल
का \omterm{def:b3:modules:free}{अंतिम रूप से उत्पन्न} $\Z $-मॉड्यूल $\Z[\alpha, \beta]$,
इस तरह \omterm{def:b3:modules:free}{अंतिम रूप से उत्पन्न} ($\Z[\alpha,\beta]$, द्वारा उत्पन्न
$ nm $ तत्वों, की एक छवि है $\Z^{nm}$; एक सबमॉड्यूल खींचता है
के एक सबमॉड्यूल पर वापस जाएं $\Z^{nm}$, पद से \omterm{def:b3:modules:free}{मुक्त} $\leq nm $ द्वारा
\cref{thm:b3:modules:submodule}, और इसकी छवि उत्पन्न होती है)। द्वारा
प्रश्न 1, $\alpha + \beta $ और $\alpha\beta $ हैं \omterm{def:b3:galois:algebraic}{बीजगणितीय}
पूर्णांक

\textbf{3.} अगर $\frac pq $ (निम्नतम पद) एक मोनिक की जड़ है
\omterm{def:b3:galois:extension}{डिग्री} का पूर्णांक बहुपद $ n $, तर्कसंगत मूल प्रमेय
(स्पष्ट हर: $ p^n = -q(\cdots)$) देता है $ q \mid p^n $,
इसलिए $ q = \pm1$.

\textbf{4.} $\chi(g)$ एकता की जड़ों का योग है
(\cref{prop:b3:representations:charbasics}), प्रत्येक एक \omterm{def:b3:galois:algebraic}{बीजगणितीय}
पूर्णांक (का मूल) $ X^N - 1$); प्रश्न 2 से समाप्त करें।

\textbf{5.} के लिए $ h \in G $: $\rho(h)S_C\rho(h)^{-1} = \sum_{g\in
C}\rho(hgh^{-1}) = S_C $ ($ C $ एक वर्ग है)। शूर द्वारा, $ S_C =
\omega_C\,\mathrm{id}$; निशान लेना, $\abs C\,\chi(g_C) =
\omega_C\, n $.

\textbf{6.} $ S_CS_{C'} = \sum_{x \in C, y \in C'}\rho(xy) =
\sum_{z \in G} a(z)\rho(z)$ साथ $ a(z) = \#\{(x,y) \in C\times
C' : xy = z\}$. द्वारा संयुग्मन $ h $ के लिए समाधान प्रस्तुत करता है
$ z $ उन लोगों के साथ $ hzh^{-1}$: $ a $ एक है वर्ग समारोह के साथ
में मान $\N $, इसलिए $ S_CS_{C'} = \sum_{C''}a_{CC'C''}S_{C''}$.
स्थानापन्न $ S_C = \omega_C\,\mathrm{id}$ भर में और
अदिश राशि की पहचान करना:
$\omega_C\omega_{C'} = \sum_{C''}a_{CC'C''}\,\omega_{C''}$.

\textbf{7.} होने देना $ M $ हो $\Z $-मॉड्यूल द्वारा फैलाया गया $1$ और सब
उत्पाद $\omega_{C_1}\cdots\omega_{C_k}$; प्रत्येक प्रश्न 6 द्वारा
ऐसा उत्पाद कम हो जाता है $\Z $-की संयोजन $1$ और यह
$\omega_C $: $ M $ है \omterm{def:b3:modules:free}{अंतिम रूप से उत्पन्न}, और $\omega_C M
\subseteq M $ प्रत्येक के लिए $ C $. विशेष रूप से $\Z[\omega_C]
\subseteq M $ है \omterm{def:b3:modules:free}{अंतिम रूप से उत्पन्न} (सबमॉड्यूल, जैसा कि प्रश्न में है
2), और प्रश्न 1 बनाता है $\omega_C $ एक \omterm{def:b3:galois:algebraic}{बीजगणितीय} पूर्णांक.

\textbf{8.} एक के लिए \omterm{def:b3:representations:rep}{अलघुकरणीय} $\chi $ \omterm{def:b3:galois:extension}{डिग्री} का $ n $:
\[
\frac{\abs G}{n} = \frac{\abs G}{n}\langle\chi,\chi\rangle
= \frac1n \sum_{g}\chi(g)\overline{\chi(g)}
= \sum_{C}\frac{\abs C\,\chi(g_C)}{n}\,\overline{\chi(g_C)}
= \sum_C \omega_C\,\overline{\chi(g_C)} .
\]
प्रत्येक $\overline{\chi(g_C)} = \chi(g_C^{-1})$ एक \omterm{def:b3:galois:algebraic}{बीजगणितीय}
पूर्णांक (प्रश्न 4), इसलिए दाहिना पक्ष एक है (प्रश्न 2,
7); यह तर्कसंगत है, इसलिए एक पूर्णांक है (प्रश्न 3): $ n \mid \abs
G $.

\textbf{9.} बी\'बाहर: $ u\abs C + vn = 1$ साथ $ u, v \in \Z $.
तब
\[
\frac{\chi(g_C)}{n} = u\,\frac{\abs C\,\chi(g_C)}{n} +
v\,\chi(g_C) = u\,\omega_C + v\,\chi(g_C),
\]
एक \omterm{def:b3:galois:algebraic}{बीजगणितीय} पूर्णांक.

\textbf{10.} कल्पना करना $0 < \abs{\chi(g_C)} < n $ और चलो $\alpha
= \chi(g_C)/n $. सभी मूल्य $\chi(g)$ रिहायश $\Q(\zeta_N)$, $ N =
\abs G $ (का योग $ N $- एकता की जड़ें)। के लिए $\sigma \in
\operatorname{Gal}(\Q(\zeta_N)/\Q)$: $\sigma $ की जड़ों को मैप करता है
एकता की जड़ों तक एकता ($\sigma(\zeta^k) = \zeta^{ak}$,
\cref{thm:b3:galois:cyclotomicirred}), इसलिए $\sigma(\chi(g_C))$
पुनः का योग है $ n $ एकता की जड़ें: $\abs{\sigma(\alpha)}
\leq 1$; भी $\sigma(\alpha)$ एक \omterm{def:b3:galois:algebraic}{बीजगणितीय} पूर्णांक (यह है
वही \omterm{def:b3:galois:algebraic}{न्यूनतम बहुपद} जैसा $\alpha $). उत्पाद $ P =
\prod_\sigma \sigma(\alpha)$ समग्र द्वारा तय किया गया है \omterm{def:b3:galois:galois}{गैलोइस
समूह}, इसलिए तर्कसंगत (\cref{thm:b3:galois:fundamental}(1)),
और यह एक है \omterm{def:b3:galois:algebraic}{बीजगणितीय} पूर्णांक के साथ
\[
0 < \abs P \leq \abs\alpha < 1
\]
(कोई भी कारक गायब नहीं होता: $\sigma(\alpha) = 0$ जबरदस्ती करेगा $\alpha =
0$). यह प्रश्न 3 का खंडन करता है। इसलिए $\chi(g_C) = 0$ या
$\abs{\chi(g_C)} = n $, और बाद वाले मामले में $\rho(g_C)$ है
अदिश (\cref{prop:b3:representations:charbasics}).

\textbf{11.} कॉलम ऑर्थोगोनैलिटी ($ C \neq \{e\}$):
$\sum_i \chi_i(e)\overline{\chi_i(g_C)} = 0$, यानी\ $1 +
\sum_{i \geq 2} n_i\overline{\chi_i(g_C)} = 0$. यदि प्रत्येक
गैरतुच्छ $\chi_i $ साथ $ p \nmid n_i $ पर गायब हो गया $ g_C $, फिर
बाकियों को उनके कारक के आधार पर समूहित करना $ p $:
\[
-\frac1p = \sum_{i \geq 2,\ p \mid n_i}
\frac{n_i}{p}\,\overline{\chi_i(g_C)},
\]
एक \omterm{def:b3:galois:algebraic}{बीजगणितीय} पूर्णांक --- विरोधाभासी प्रश्न 3. तो कुछ
गैरतुच्छ $\chi_i $ है $ p \nmid n_i $ और $\chi_i(g_C) \neq 0$;
तब से $\abs C = p^k $, $\gcd(\abs C, n_i) = 1$, और प्रश्न 10
बनाता है $\rho_i(g_C)$ एक अदिश राशि. अब $ G $ \omterm{def:b3:groups:simple}{सरल} नॉनबेलियन:
$\chi_i $ वफादार है (\cref{exo:b3:representations:6}(सी)), और
$ Z_i = \{g : \rho_i(g) \text{ scalar}\}$ एक है \omterm{def:b3:groups:normal}{सामान्य} उपसमूह
(नीचे की प्रारंभिक छवि $\rho_i $ अदिश राशि के, जो एक बनाते हैं
\omterm{def:b3:groups:normal}{सामान्य} --- वास्तव में \omterm{ex:b3:groups:actions}{केंद्र}ीय --- छवि का उपसमूह) युक्त
$ g_C \neq e $: $ Z_i = G $. तब $\rho_i(G)$ एबेलियन है और
वफादार, बनाना $ G $ एबेलियन: विरोधाभास। \emph{कोई \omterm{def:b3:groups:simple}{सरल} नहीं
नॉनबेलियन समूह के पास एक है संयुग्मन वर्ग प्राइम-पावर आकार का
$> 1$.}

\textbf{23.} के लिए $ z \in Z(G)$, $\rho(z)$ प्रत्येक के साथ आवागमन करता है
$\rho(g)$, तो शूर की लेम्मा द्वारा $\rho(z) = \lambda\,
\mathrm{id}$; तब से $ z $ परिमित क्रम है, $\lambda $ एक जड़ है
एकता की, और $\abs{\chi(z)} = \abs\lambda\,n = n $. तब
\[
\abs G = \abs G\,\langle\chi, \chi\rangle
= \sum_{g \in G}\abs{\chi(g)}^2
\geq \sum_{z \in Z(G)}\abs{\chi(z)}^2
= \abs{Z(G)}\,n^2,
\]
यानी\ $ n^2 \leq [G : Z(G)]$. आदेश का एक नॉनबेलियन समूह $8$
($ D_4$ या $ Q_8$) में पाँच हैं \omterm{ex:b3:groups:actions}{संयुग्मता वर्ग}, तो पाँच
\omterm{def:b3:representations:rep}{अलघुकरणीय} के साथ \omterm{def:b3:galois:extension}{डिग्री} $\sum n_i^2 = 8$; का एकमात्र रास्ता
लिखो $8$ पाँच वर्गों के योग के रूप में $\geq 1$ है $1 + 1 + 1 + 1
+ 4$: \omterm{def:b3:galois:extension}{डिग्री} $1, 1, 1, 1, 2$. दोनों समूहों के पास है \omterm{ex:b3:groups:actions}{केंद्र} आदेश की $2$, और डिग्री-$2$ \omterm{def:b3:representations:character}{चरित्र} समानता प्राप्त करता है:
$2^2 = 4 = [G : Z(G)]$ --- सीमा तीव्र है. के लिए $ A_5$,
$ Z = \{e\}$ और बाउंड पढ़ता है $ n^2 \leq 60$: से संतुष्ट
$1, 3, 3, 4, 5$ अतिरिक्त जगह के साथ ($25 \leq 60$), जैसा कि होना चाहिए
चूँकि समानता बल देगी (उसी श्रृंखला द्वारा) $\chi $ को
गायब हो जाओ \omterm{ex:b3:groups:actions}{केंद्र}.

\textbf{24.} $\rho(g)^m = \mathrm{id}$ के लिए $ m $ के लिए
$ g $, इसलिए $\rho(g)$ द्वारा नष्ट कर दिया जाता है $ X^m - 1$, के साथ विभाजित
\omterm{def:b3:groups:simple}{सरल} जड़ें खत्म $\C $: विकर्णीय, eigenvalues के साथ
$\lambda_1, \dots, \lambda_n $ एकता की जड़ें, एक में
eigenbasis $(e_i)$. उत्पाद $ e_ie_j $ ($ i \leq j $) फॉर्म ए
का eigenbasis $\operatorname{Sym}^2V $ eigenvalues के साथ
$\lambda_i\lambda_j $, और $ e_i \wedge e_j $ ($ i < j $) में से एक
$\Lambda^2V $; तब से
\[
\sum_{i<j}\lambda_i\lambda_j
= \frac{(\sum_i\lambda_i)^2 - \sum_i\lambda_i^2}2
= \frac{\chi(g)^2 - \chi(g^2)}2,
\]
और सममित योग जुड़ जाता है $\sum_i\lambda_i^2$ के बजाय
इसे घटाने पर दोनों सूत्र अनुसरण करते हैं। के लिए $\chi_2 = (3, -1,
0, \varphi, \bar\varphi)$ कक्षाओं पर $(e, (2,2)\text{-},
3\text{-cycles}, c, c^2)$: स्क्वेरिंग डबल भेजता है
में स्थानान्तरण $ e $, तीन-चक्र से तीन-चक्र,
की \omterm{def:b3:groups:action}{कक्षा} $ c $ उस पर $ c^2$ और इसके विपरीत ($ c^{-1} \sim
c $ में $ A_5$, इसलिए $ c^4 \sim c $). इस तरह $\chi_2(g^2)$ पढ़ता $(3,
3, 0, \bar\varphi, \varphi)$, और, का उपयोग करना $\varphi^2 = \varphi
+ 1$, $\bar\varphi = 1 - \varphi $:
\[
\Lambda^2\chi_2 = (3, -1, 0, \varphi, \bar\varphi) = \chi_2,
\qquad
\operatorname{Sym}^2\chi_2 = (6, 2, 0, 1, 1) .
\]
वर्ग आकार के साथ, बाद वाले को विघटित करना $1, 15, 20, 12,
12$: $\langle\cdot, \mathbf 1\rangle = \frac1{60}(6 +
15\cdot2 + 0 + 12 + 12) = 1$; $\langle\cdot, \chi_5\rangle =
\frac1{60}(30 + 30) = 1$; $\langle\cdot, \chi_4\rangle =
\frac1{60}(24 - 12 - 12) = 0$; $\langle\cdot, \chi_2\rangle =
\frac1{60}(18 - 30 + 12(\varphi + \bar\varphi)) = 0$, और
इसी तरह के लिए $\chi_3$. इसलिए $\operatorname{Sym}^2\chi_2 =
\mathbf 1 + \chi_5$ (आयाम $6 = 1 + 5$) और
$\chi_2\otimes\chi_2 = \mathbf 1 + \chi_2 + \chi_5$
(आयाम $9 = 1 + 3 + 5$). ज्यामिति: समतुल्य
समरूपता $\Lambda^2\R^3 \to \R^3$, $ u \wedge v \mapsto u
\times v $, बिलकुल है $\Lambda^2\chi_2 = \chi_2$ एक के लिए
घूर्णन समूह; सारांश $\mathbf 1$ सममिति का
वर्ग अपरिवर्तनीय द्विघात रूप है $ x^2 + y^2 + z^2$, और
$\chi_5$ ट्रेसलेस के पांच-आयामी स्थान पर रहता है
सममित टेंसर (हार्मोनिक क्वाड्रैटिक्स)।

\textbf{25.} का स्तम्भ $ c $ के कॉलम के विरुद्ध $ c^2$:
\[
1\cdot1 + \varphi\bar\varphi + \bar\varphi\varphi +
(-1)(-1) + 0 = 1 - 1 - 1 + 1 + 0 = 0,
\]
चूँकि अलग-अलग वर्गों के लिए रूढ़िवादिता की माँग होती है ($\varphi
\bar\varphi = -1$). का स्तम्भ $ c $ स्वयं के विरुद्ध: $1 +
\varphi^2 + \bar\varphi^2 + 1 + 0 = 1 + 3 + 1 = 5 = 60/12 =
\abs{Z_{A_5}(c)}$. नियमित \omterm{def:b3:representations:character}{चरित्र} $\sum_in_i\chi_i $ पर
चार गैर-पहचान कॉलम:
\[
1 - 3 - 3 + 0 + 5 = 0, \qquad
1 + 0 + 0 + 4 - 5 = 0,
\]
दोहरे स्थानान्तरण और तीन-चक्रों पर, और \omterm{def:b3:groups:action}{कक्षा} पर
का $ c $ (की है कि $ c^2$ इसका है \omterm{def:b3:galois:galois}{गैलोइस} संयुग्मित):
\[
1 + 3\varphi + 3\bar\varphi - 4 + 0 = 1 + 3 - 4 = 0,
\]
का उपयोग करते हुए $\varphi + \bar\varphi = 1$. तालिका प्रत्येक से गुजरती है
ऑडिट: यह है \omterm{def:b3:representations:table}{चरित्र तालिका} का $ A_5$.
\end{solution}
