\section*{अध्याय \ref{ch:b3:galois} --- फ़ील्ड एक्सटेंशन और
गैलोइस सिद्धांत}

\begin{solution}{exo:b3:galois:1}
$\sqrt3 \notin \Q(\sqrt2)$: से $\sqrt3 = a + b\sqrt2$ ($ a, b
\in \Q $), चुकता करने से मिलता है $3 = a^2 + 2b^2 + 2ab\sqrt2$, इसलिए $ ab =
0$; $ b = 0$ बनाता है $\sqrt3$ तर्कसंगत, $ a = 0$ देता है $\sqrt6 = 2b
\in \Q $ --- दोनों झूठे (मानक अभाज्य-कारकीकरण तर्क)।
इस तरह $[\Q(\sqrt2,\sqrt3) : \Q(\sqrt2)] = 2$ और यह \omterm{thm:b3:galois:tower}{टावर कानून}
देता है \omterm{def:b3:galois:extension}{डिग्री} $4$.

होने देना $\gamma = \sqrt2 + \sqrt3$. तब $\gamma^2 = 5 + 2\sqrt6$ और
$(\gamma^2 - 5)^2 = 24$: $\gamma $ सत्यानाश कर देता है $ X^4 - 10X^2 + 1$.
इसके अतिरिक्त $\gamma^3 = 11\sqrt2 + 9\sqrt3$, इसलिए $\gamma^3 - 9\gamma
= 2\sqrt2$: $\sqrt2 \in \Q(\gamma)$, तब $\sqrt3 = \gamma -
\sqrt2 \in \Q(\gamma)$: $\Q(\gamma) = \Q(\sqrt2,\sqrt3)$, का
\omterm{def:b3:galois:extension}{डिग्री} $4$. विनाशक चतुर्थक, युक्त \omterm{def:b3:galois:extension}{डिग्री} की
\omterm{def:b3:galois:algebraic}{न्यूनतम बहुपद}, \emph{है}  \omterm{def:b3:galois:algebraic}{न्यूनतम बहुपद}: $ X^4 -
10X^2 + 1$ (विशेष रूप से यह है अलघुकरणीय ऊपर $\Q $).
\end{solution}

\begin{solution}{exo:b3:galois:2}
की तीन जड़ें $ X^3 - 2$ में $\C $ हैं $\alpha, j\alpha,
j^2\alpha $ साथ $ j = \eu^{2\iu\pi/3}$. वो नक्शा $\alpha \mapsto
j\alpha $ ए को परिभाषित करता है $\Q $-एम्बेडिंग $\Q(\alpha) \to \C $
(\cref{thm:b3:galois:simple}: दोनों \omterm{def:b3:galois:extension}{डिग्री} उत्पन्न करते हैं-$3$ एक्सटेंशन
उसी के साथ \omterm{def:b3:galois:algebraic}{न्यूनतम बहुपद}), जिसकी छवि $\Q(j\alpha)
\not\subseteq \R $ से मतभेद होना $\Q(\alpha) \subseteq \R $:
$\Q(\alpha)/\Q $ \omterm{def:b3:groups:normal}{सामान्य} नहीं है. विभाजन क्षेत्र है $ L =
\Q(\alpha, j)$, साथ $[L:\Q] = [L:\Q(\alpha)]\,[\Q(\alpha):\Q] =
2 \cdot 3 = 6$ ($ j $ संतुष्ट $ X^2 + X + 1$, अलघुकरणीय खत्म
वास्तविक क्षेत्र $\Q(\alpha)$). का एक स्वप्रतिरूपण $\Q(\alpha)$
भेजना होगा $\alpha $ की एक जड़ तक $ X^3 - 2$ \emph{अंदर}
$\Q(\alpha) \subseteq \R $: केवल $\alpha $ योग्य है, इसलिए
$\operatorname{Aut}_\Q(\Q(\alpha)) = \{\mathrm{id}\}$, आदेश की
$1 < 3$.
\end{solution}

\begin{solution}{exo:b3:galois:3}
$ X^2 + 1$ में कोई जड़ नहीं है $\mathbb F_3$ ($0, 1, 2 \mapsto 1, 2,
2$), इसलिए $\mathbb F_9 = \mathbb F_3[X]/(X^2+1)$ के साथ एक फ़ील्ड है
$9$ तत्व; लिखना $\omega = \bar X $, $\omega^2 = -1$. द
समूह $\mathbb F_9^\times $ क्रम का चक्रीय है $8$; $\omega $ है
आदेश $4$, लेकिन $1 + \omega $ काम करता है: $(1+\omega)^2 = 1 + 2\omega +
\omega^2 = 2\omega $, $(1+\omega)^4 = 4\omega^2 = \omega^2 = -1
\ne 1$: आदेश $8$.

ऊपर $\mathbb F_2$ --- \omterm{def:b3:galois:extension}{डिग्री} $1$: $ X $, $ X + 1$; \omterm{def:b3:galois:extension}{डिग्री} $2$:
$ X^2 + X + 1$ (अन्य तीन चतुर्भुजों की जड़ें हैं); \omterm{def:b3:galois:extension}{डिग्री}
$3$: $ X^3 + X + 1$ और $ X^3 + X^2 + 1$ (कोई जड़ नहीं $\mathbb
F_2$; अन्य छह घनों की जड़ें हैं)। सत्यापन:
\[
(X^3{+}X{+}1)(X^3{+}X^2{+}1) = X^6 + X^5 + X^4 + X^3 + X^2 + X
+ 1,
\]
और $ X(X{+}1)(X^6 + \dots + 1) = X(X^7 + 1) = X^8 + X = X^8 -
X $ ऊपर $\mathbb F_2$ --- बिलकुल इरेड्यूसिबल्स का \omterm{def:b3:galois:extension}{डिग्री}
डिवाइडिंग $3$, जैसा \cref{exo:b3:galois:6} भविष्यवाणी करता है (\omterm{def:b3:galois:extension}{डिग्री} $2$ है
अनुपस्थित: $2 \nmid 3$).
\end{solution}

\begin{solution}{exo:b3:galois:4}
(ए) मॉड $7$: की शक्तियां $3$ हैं $3, 2, 6, 4, 5, 1$: आदेश
$6$, एक जनरेटर; आदिम जड़ें हैं $3^k $ साथ
$\gcd(k, 6) = 1$: $3$ और $3^5 = 5$. आधुनिक $11$: की शक्तियां $2$:
$2, 4, 8, 5, 10, 9, 7, 3, 6, 1$: एक जनरेटर; आदिम जड़ें
$2^k $, $\gcd(k, 10) = 1$: $2, 2^3 = 8, 2^7 = 7, 2^9 = 6$.

(बी) लिखें $ x = g^k $ साथ $ g $ एक जनरेटर
(\cref{thm:b3:galois:cyclic}). तब $ x $ एक वर्ग है यदि $ k $ है
सम (वर्ग हैं) $ g^{2l}$, और $ g^{2l} = g^{k}$ आईएफएफ $ k
\equiv 2l \bmod p-1$, \omterm{def:b3:groups:derived}{व्याख्या करने योग्य} आईएफएफ $ k $ यहां तक की, $ p - 1$ किया जा रहा है
सम)। और $ x^{(p-1)/2} = g^{k(p-1)/2} = 1$ आईएफएफ $(p-1) \mid
k\frac{p-1}2$ आईएफएफ $ k $ सम: दोनों शर्तें सहमत हैं। के लिए $ x =
-1 = g^{(p-1)/2}$: यह एक वर्ग है यदि $\frac{p-1}2$ सम है, यदि
$ p \equiv 1 \pmod 4$ --- \cref{pb:b3:rings:1} दोबारा।
\end{solution}

\begin{solution}{exo:b3:galois:5}
अंदर $\bar{\mathbb F}_p $, $\mathbb F_{p^k} = \{x : x^{p^k} =
x\}$ का निश्चित सेट है $ F^k $. चौराहा $\mathbb F_{p^m}
\cap \mathbb F_{p^n}$ द्वारा तय किया गया है $ F^m $ और $ F^n $, इसलिए द्वारा
$ F^{\gcd(m,n)}$ ($\gcd = am + bn $: एक निश्चित तत्व पर, $ F^{am +
bn} = (F^m)^a(F^n)^b $ तुच्छ कार्य करता है --- घातांक लिया जा सकता है
आवधिकता द्वारा सकारात्मक); तो यह इसमें निहित है $\mathbb
F_{p^{\gcd(m,n)}}$, जो दोनों में विपरीत रूप से निहित है
(\cref{thm:b3:galois:finitefields}(3)). समग्र $\mathbb
F_{p^m}\mathbb F_{p^n}$: कोई भी फ़ील्ड जिसमें दोनों हों \omterm{def:b3:galois:extension}{डिग्री}
द्वारा विभाज्य $ m $ और $ n $, इसलिए द्वारा $\operatorname{lcm}(m,n)$; और
$\mathbb F_{p^{\operatorname{lcm}}}$ दोनों शामिल हैं: यह है
समग्र. के लिए $ m \mid n $: $\operatorname{Gal}(\mathbb
F_{p^n}/\mathbb F_{p^m})$ की शक्तियों से युक्त है $ F $ फिक्सिंग
$\mathbb F_{p^m}$, यानी\ का $ F^m $: क्रम का चक्रीय $ n/m $,
द्वारा उत्पन्न $ F^m \colon x \mapsto x^{p^m}$ (जैसा क्रम में है
\cref{thm:b3:galois:finitefields}(2)).
\end{solution}

\begin{solution}{exo:b3:galois:6}
$ X^{q^n} - X $ है \omterm{def:b3:galois:separable}{वियोज्य} (व्युत्पन्न $-1$) रूट सेट के साथ
$\mathbb F_{q^n}$. होने देना $ P $ साधु बनो अलघुकरणीय का \omterm{def:b3:galois:extension}{डिग्री} $ d $.
अगर $ d \mid n $: $\mathbb F_q[X]/(P) \cong \mathbb F_{q^d}
\subseteq \mathbb F_{q^n}$, इसलिए $ P $ एक जड़ है $\alpha \in
\mathbb F_{q^n}$; $\alpha^{q^n} = \alpha $, और $ P =
\pi_\alpha $ विभाजित $ X^{q^n} - X $. अगर $ P \mid X^{q^n} - X $: जड़ $\alpha \in \mathbb F_{q^n}$ उत्पन्न करता है $\mathbb F_{q^d}
\subseteq \mathbb F_{q^n}$, इसलिए $ d \mid n $
(\cref{thm:b3:galois:finitefields}(3)). विशिष्ट इरेड्यूसिबल्स
सहअभाज्य हैं और उत्पाद है \omterm{def:b3:galois:separable}{वियोज्य}: प्रत्येक $ P $ के साथ प्रकट होता है
प्रतिपादक बिल्कुल $1$, और की हर जड़ $ X^{q^n}-X $ एक जड़ है
इसके \omterm{def:b3:galois:algebraic}{न्यूनतम बहुपद}: गुणनखंडन धारण करता है। की तुलना
\omterm{def:b3:galois:extension}{डिग्री}: $ q^n = \sum_{d\mid n} d\,I_d(q)$.

फलस्वरूप $ I_1 = q $; $ q^2 = I_1 + 2I_2$ देता है $ I_2 =
\frac{q^2 - q}2$; $ q^3 = I_1 + 3I_3$ देता है $ I_3 = \frac{q^3 -
q}3$; $ q^4 = I_1 + 2I_2 + 4I_4$ देता है $ I_4 = \frac{q^4 -
q^2}4$. अस्तित्व: $ nI_n = q^n - \sum_{d \mid n,\, d < n}
dI_d \geq q^n - \sum_{d \leq n/2} q^d > q^n - q^{n/2 + 1} \geq
0$ के लिए $ n \geq 2$ (और $ I_1 = q \geq 1$): $ I_n \geq 1$ हमेशा।
\end{solution}

\begin{solution}{exo:b3:galois:7}
$ L = \Q(\sqrt2, \sqrt3)$ है विभाजन क्षेत्र का $(X^2 -
2)(X^2 - 3)$, \omterm{def:b3:galois:separable}{वियोज्य}: \omterm{def:b3:galois:galois}{गैलोइस} का \omterm{def:b3:galois:extension}{डिग्री} $4$
(\cref{exo:b3:galois:1}). एक ऑटोमोर्फिज्म भेजता है $\sqrt2 \mapsto
\pm\sqrt2$ और $\sqrt3 \mapsto \pm\sqrt3$: \omterm{def:b3:rings:primemaximal}{अधिक से अधिक} $4$ विकल्प,
और $\abs G = 4$ सब कुछ पता चलता है: $ G \cong (\Z/2\Z)^2$, के साथ
तत्व $\mathrm{id}, \sigma (\sqrt2 \mapsto -\sqrt2), \tau
(\sqrt3\mapsto-\sqrt3), \sigma\tau $. क्रम के उपसमूह $2$:
$\langle\sigma\rangle, \langle\tau\rangle,
\langle\sigma\tau\rangle $, निश्चित फ़ील्ड के साथ $\Q(\sqrt3)$,
$\Q(\sqrt2)$, $\Q(\sqrt6)$ (टिप्पणी $\sigma\tau $ ठीक करता है $\sqrt6 =
\sqrt2\sqrt3$). जाली: $\Q $ नीचे, तीन द्विघात
बीच में खेत, $ L $ शीर्ष पर --- और कुछ नहीं
(\cref{thm:b3:galois:fundamental}). के लिए $(X^2-2)(X^2-3)(X^2-6)$:
द विभाजन क्षेत्र है \emph{वही} $ L $ ($\sqrt6 =
\sqrt2\sqrt3$), इसलिए उत्तर समान है: द \omterm{thm:b3:galois:fundamental}{गैलोइस
पत्राचार} का एक अपरिवर्तनीय है \emph{विस्तार}, का नहीं
इसे प्रस्तुत करने के लिए बहुपद को चुना गया।
\end{solution}

\begin{solution}{exo:b3:galois:8}
(ए) $ G $ पर निष्ठापूर्वक और परिवर्तनीय (अघुलनशील) कार्य करता है
तीन जड़ें: $ G \hookrightarrow S_3$ साथ $3 \mid \abs G $: $ G
\cong A_3$ या $ S_3$. प्रत्येक $\sigma \in G $ को क्रमपरिवर्तित करता है $ x_i $, और
$\sigma(\delta) = \varepsilon(\sigma)\,\delta $ ($\delta $ है
जड़ों में बारी-बारी से)। अगर $\Delta $ में एक वर्ग है $\Q $:
$\delta \in \Q^\times $ (टिप्पणी $\delta \neq 0$: \omterm{def:b3:galois:separable}{वियोज्य}), इसलिए
$\varepsilon(\sigma) = 1$ सभी के लिए $\sigma $: $ G \subseteq A_3$,
इस तरह $= A_3$. यदि नहीं: $\delta \notin \Q $, इतना कुछ $\sigma $ है
$\varepsilon(\sigma) = -1$: $ G = S_3$. (दोनों ही मामलों में $\Q(\delta)
= L^{G \cap A_3}$.)

(बी) के साथ $ x_1 + x_2 + x_3 = 0$: $ u + v = 2x_1 - (x_2 + x_3) =
3x_1$. भी
\[
uv = \sum_i x_i^2 + (j + j^2)\sum_{i<k}x_ix_k
= \Bigl(\sum x_i\Bigr)^2 - 3\sum_{i<k}x_ix_k = -3p,
\]
का उपयोग करते हुए $ j + j^2 = -1$ और $\sum_{i<k}x_ix_k = p $. तब
\[
u^3 + v^3 = (u+v)^3 - 3uv(u+v) = 27x_1^3 + 9p\cdot 3x_1
= 27\,(x_1^3 + px_1) = -27q .
\]
इसलिए $ u^3, v^3$ की जड़ें हैं $ Y^2 + 27qY - 27p^3 = 0$
(उत्पाद $(uv)^3 = -27p^3$): $ u^3 = \frac{-27q +
\sqrt{729q^2 + 108p^3}}2$, और $ x_1 = \frac{u + v}3$ साथ $ v =
-3p/u $: कार्डानो. गलने की योग्यता का $ S_3$ कंकाल है:
टावर $\Q \subseteq \Q(\delta) \subseteq \Q(\delta, j, u)$
पहले एक वर्गमूल जोड़ता है ($\delta $, का निश्चित क्षेत्र $ A_3$:
कदम $ S_3 \to S_3/A_3$), फिर एक घनमूल ($ u $, तब से $ u^3
\in \Q(\delta, j)$: कदम $ A_3 \to \{e\}$) --- द व्युत्पन्न
शृंखला $ S_3 \supset A_3 \supset \{e\}$ मांस बनाया.
\end{solution}

\begin{solution}{exo:b3:galois:9}
$\eta_0 + \eta_1 = \zeta_5 + \zeta_5^2 + \zeta_5^3 + \zeta_5^4 =
-1$ (सभी का योग $5$- एकता की जड़ें हैं $0$). $\eta_0\eta_1 =
(\zeta + \zeta^4)(\zeta^2 + \zeta^3) = \zeta^3 + \zeta^4 +
\zeta^6 + \zeta^7 = \zeta^3 + \zeta^4 + \zeta + \zeta^2 = -1$
(सूचकांक मॉड $5$). इसलिए $\eta_0, \eta_1$ की जड़ें हैं $ Y^2 + Y
- 1$: $\frac{-1 \pm \sqrt5}2$. तब से $\eta_0 =
2\cos\frac{2\pi}5 > 0$: $\eta_0 = \frac{\sqrt5 - 1}2$, कहाँ से
$\cos\frac{2\pi}5 = \frac{\sqrt5 - 1}4$, और $\eta_1 =
\frac{-1-\sqrt5}2$. समूह
$\operatorname{Gal}(\Q(\zeta_5)/\Q) \cong (\Z/5\Z)^\times $ है
क्रम का चक्रीय $4$: यह है एक \emph{अद्वितीय} आदेश का उपसमूह
$2$ ($\{\pm 1\}$, यानी\ $\zeta \mapsto \zeta^{\pm1}$), इस तरह
$\Q(\zeta_5)$ एक अद्वितीय द्विघात उपक्षेत्र है
(\cref{thm:b3:galois:fundamental}), जिसमें है $\eta_0
\notin \Q $: यह है $\Q(\eta_0) = \Q(\sqrt5)$. रचनाशीलता:
$\cos\frac{2\pi}5$ द्विघात मीनार में स्थित है $\Q \subseteq
\Q(\sqrt5)$, और $\zeta_5$ ऊपर एक द्विघात चरण:
\cref{thm:b3:galois:wantzel} पंचकोण का निर्माण करता है.
\end{solution}

\begin{solution}{exo:b3:galois:10}
(ए) लिखें $ s = S^{1/p}$, $ t = T^{1/p}$ (किसी चुने हुए के तत्व
\omterm{def:b3:galois:algebraic}{बीजगणितीय} समापन साथ $ s^p = S $, $ t^p = T $). $ X^p - S $ है
अलघुकरणीय ऊपर $ K = \mathbb F_p(S, T) = (\mathbb
F_p(T))(S)$-अंश: \omterm{thm:b3:rings:criteria}{Eisenstein} पर प्रधान तत्व $ S $ की \omterm{def:b3:rings:pidufd}{यूएफडी} $\mathbb F_p(T)[S]$ (\cref{thm:b3:rings:criteria}). इसलिए
$[K(s):K] = p $; वैसे ही $ X^p - T $ है \omterm{thm:b3:rings:criteria}{Eisenstein} पर $ T $ ऊपर
$ K(s) = \mathbb F_p(s)(T)$-अंश --- $ T $ में प्रमुख रहता है
$\mathbb F_p(s)[T]$ --- देना $[L : K(s)] = p $ और $[L:K] =
p^2$. के लिए $\alpha \in L $: $ L = K[s, t]$, इसलिए $\alpha = \sum
c_{ij}s^it^j $ ($ c_{ij} \in K $), और द्वारा \omterm{thm:b3:galois:finitefields}{फ्रोबेनियस} आकारिता
$\alpha^p = \sum c_{ij}^p S^iT^j \in K $.

(बी) यदि $ L = K(\alpha)$, तब $[K(\alpha):K] = p^2$; लेकिन
$\alpha^p = a \in K $ मतलब $\alpha $ सत्यानाश कर देता है $ X^p - a $, इसलिए
$\deg\pi_\alpha \leq p < p^2$: विरोधाभास। कोई आदिम नहीं
तत्व: \cref{thm:b3:galois:primitive} वास्तव में जरूरत है
पृथक्करणीयता (यहाँ हर $\pi_\alpha $ कुछ को बाँट देता है $ X^p - a =
(X - \alpha)^p $: विशुद्ध रूप से अविभाज्य)।
\end{solution}

\begin{solution}{exo:b3:galois:11}

अलघुकरणीय. अगर $\alpha $ एक जड़ है, $[\Q(\alpha):\Q] = 5$

इसलिए $[L : \Q(\alpha)] = 2$ और $[L:\Q] = 8$. विस्तार है
\omterm{def:b3:galois:galois}{गैलोइस} (विभाजन क्षेत्र एक का \omterm{def:b3:galois:separable}{वियोज्य} बहुपद: जड़ें

$\alpha $ चार जड़ों में से एक को और $\iu $ को $\pm\iu $:
\omterm{def:b3:rings:primemaximal}{अधिक से अधिक} $8$ मानचित्र, सब साकार। कहा गया $\sigma $ (आदेश

\[
\tau\sigma\tau(\alpha) = \tau\sigma(\alpha) =
\tau(\iu\alpha) = -\iu\alpha = \sigma^{-1}(\alpha),
\qquad \tau\sigma\tau(\iu) = \iu\cdot(-1)(-1) = \iu ,
\]
और अधिक ध्यान से: $\tau\sigma\tau(\iu) = \tau\sigma(-\iu) =
\tau(-\iu) = \iu = \sigma^{-1}(\iu)$. इसलिए $\tau\sigma\tau =
\sigma^{-1}$: की प्रस्तुति $ D_4$.

(बी) के दस उपसमूह $ D_4 = \langle\sigma,
\tau\rangle $: $\{e\}$; क्रम के पाँच $2$:
$\langle\sigma^2\rangle $, $\langle\tau\rangle $,
$\langle\sigma^2\tau\rangle $, $\langle\sigma\tau\rangle $,
$\langle\sigma^3\tau\rangle $; क्रम के तीन $4$:
$\langle\sigma\rangle $, $\{e, \sigma^2, \tau,
\sigma^2\tau\}$, $\{e, \sigma^2, \sigma\tau,
\sigma^3\tau\}$; और $ D_4$. निश्चित फ़ील्ड (\omterm{def:b3:galois:extension}{डिग्री} = सूचकांक):
$\{e\} \leftrightarrow L $; आदेश देना-$2$ उपसमूहों
$\leftrightarrow $ पांच चतुर्थक क्षेत्र
\[
\langle\tau\rangle \leftrightarrow \Q(\alpha),\quad
\langle\sigma^2\tau\rangle \leftrightarrow \Q(\iu\alpha),
\quad
\langle\sigma^2\rangle \leftrightarrow \Q(\sqrt2, \iu),\quad
\langle\sigma\tau\rangle \leftrightarrow
\Q\bigl((1+\iu)\alpha\bigr),\quad
\langle\sigma^3\tau\rangle \leftrightarrow
\Q\bigl((1-\iu)\alpha\bigr) .
\]
जाँच: $\tau $ असली को ठीक करता है $\alpha $; $\sigma^2\tau $ भेजता है
$\alpha \mapsto -\alpha $ और $\iu \mapsto -\iu $, ठीक करना
$\iu\alpha $; और तबसे $\sigma\tau(\alpha) = \iu\alpha $,
$\sigma\tau(\iu) = -\iu $:
\[
\sigma\tau\bigl((1+\iu)\alpha\bigr) =
(1 - \iu)\,\iu\alpha = (1 + \iu)\alpha,
\qquad
\sigma^3\tau\bigl((1-\iu)\alpha\bigr) =
(1 + \iu)(-\iu)\alpha = (1 - \iu)\alpha :
\]
प्रत्येक प्रतिबिंब अपने जनरेटर और निश्चित क्षेत्र को ठीक करता है,

(जनरेटर की जड़ है $ X^4 + 8$, अलघुकरणीय). आदेश देना-$4$ उपसमूहों
$\leftrightarrow $ तीन द्विघात क्षेत्र:
$\langle\sigma\rangle \leftrightarrow \Q(\iu)$ ($\sigma $
ठीक करता है $\iu $); $\{e, \sigma^2, \tau, \sigma^2\tau\}
\leftrightarrow \Q(\sqrt2)$ (सभी चार ठीक करें $\alpha^2$ तक

\leftrightarrow \Q(\iu\sqrt2)$ ($\sigma\tau(\iu\alpha^2) =
(-\iu)(\iu\alpha)^2 = \iu\alpha^2$).

(सी) \omterm{def:b3:galois:galois}{गैलोइस} ऊपर $\Q $ $\leftrightarrow $ \omterm{def:b3:groups:normal}{सामान्य} उपसमूह का

\textbf{12.} निर्माण की आवश्यकता है $[L:\Q] = p - 1$ एक होना
की शक्ति $2$, ताकि सूचकांक की एक पूरी श्रृंखला-$2$ उपसमूह
मौजूद है. अगर $ p = 2^m + 1$ प्रधान है और $ m = ab $ साथ $ a $ विषम $>
1$: $ x + 1 \mid x^a + 1$ पर $ x = 2^b $ दिखाता है $2^b + 1$ ठीक से
विभाजित करता है $ p $ --- असंभव। इसलिए $ m $ की एक शक्ति है $2$: $ p =
2^{2^t} + 1$, ए \emph{फ़र्मेट प्राइम} ($3, 5, 17, 257, 65537$,
\dots). इसके विपरीत ऐसे के लिए $ p $, $\varphi(p) = 2^{2^t}$ और
प्रश्नों का तर्क 1--3 (या
\cref{cor:b3:galois:impossible}) लागू होता है: नियमित $ p $-गोन
है \omterm{def:b3:galois:constructible}{रचनात्मक} आईएफएफ $ p $ एक फ़र्मेट प्राइम है।

\textbf{13.} $\varphi(n)$ की एक शक्ति है $2$ बिल्कुल कब $ n =
2^a p_1\cdots p_r $ विशिष्ट फ़र्मेट प्राइम के साथ $ p_i $
(गुणात्मकता $\varphi $; एक अजीब प्रधान शक्ति $ p^k $, $ k
\geq 2$, कारक का योगदान देता है $ p \nmid 2^m $). के लिए $ n \leq
20$, द \omterm{def:b3:galois:constructible}{रचनात्मक} नियमित $ n $-गॉन हैं
\[
n = 3, 4, 5, 6, 8, 10, 12, 15, 16, 17, 20
\]
संबंधित मूल्यों के साथ
\[
\varphi(n) = 2,\ 2,\ 4,\ 2,\ 4,\ 4,\ 4,\ 8,\ 8,\ 16,\ 8 .
\]
असंभव हैं $ n = 7, 9, 11, 13, 14, 18, 19$, कहाँ
$\varphi(n) = 6, 6, 10, 12, 6, 6, 18$ एक अजीब अभाज्य गुणनखंड है.

\textbf{14.} वर्ग वर्ग की छवि बनाते हैं
चक्रीय पर रूपवाद $(\Z/p\Z)^\times $, सूचकांक का $2$:
$\frac{p-1}2$ वर्ग, $\frac{p-1}2$ नॉनस्क्वायर, इसलिए
प्रतीकों का योग है $0$. तब
\[
\sum_{a=0}^{p-1}\zeta^{a^2} = 1 + 2\sum_{b \text{ square}
\neq 0}\zeta^b
= 1 + \sum_{b\neq0}\Bigl(1 + \Bigl(\frac bp\Bigr)\Bigr)
\zeta^b = \sum_{b}\zeta^b + g = g,
\]
का उपयोग करते हुए $1 + \bigl(\frac bp\bigr) = \#\{a : a^2 = b\}$ और
$\sum_{b=0}^{p-1}\zeta^b = 0$. के लिए $ p = 17$: वर्ग मॉड
$17$ जनरेटर की सम शक्तियाँ हैं $3$, यानी\ द
प्रतिपादक प्रकट हो रहे हैं $\eta_0$ (भाग II), तो $ g =
\sum_{\text{even }k}\zeta^{3^k} - \sum_{\text{odd }k}
\zeta^{3^k} = \eta_0 - \eta_1$.

\textbf{15.} साथ $ b = c - a $ ($ a, b $ शून्येतर पर दौड़ें
अवशेष, $ c = a + b $ सभी अवशेषों पर):
\[
g^2 = \sum_c\zeta^c\sum_{a\neq0,\,a\neq c}
\Bigl(\frac{a(c-a)}p\Bigr) .
\]
के लिए $ c = 0$: $\bigl(\frac{-a^2}p\bigr) =
\bigl(\frac{-1}p\bigr)$, सारांशित किया गया $ p - 1$ मूल्य. के लिए $ c \neq 0$: विकल्प $ c - a = at $, यानी\ $ t = c/a -
1$; जैसा $ a $ गैर-शून्य अवशेषों पर चलता है, $ t $ चलता है
अवशेषों पर विशेष रूप से $\neq -1$ (उलटा: $ a = c/(1 +
t)$). सारांश बन जाता है $\bigl(\frac{a^2t}p\bigr) =
\bigl(\frac tp\bigr)$, और
\[
\sum_{t \neq -1}\Bigl(\frac tp\Bigr)
= -\Bigl(\frac{-1}p\Bigr)
\]
(प्रश्न 14 से पूरी राशि गायब हो जाती है)। इस तरह
\[
g^2 = \Bigl(\frac{-1}p\Bigr)\Bigl[(p-1) -
\sum_{c\neq0}\zeta^c\Bigr]
= \Bigl(\frac{-1}p\Bigr)\,p = p^* ,
\]
का उपयोग करते हुए $\sum_{c\neq0}\zeta^c = -1$ और यूलर की कसौटी
$\bigl(\frac{-1}p\bigr) = (-1)^{(p-1)/2}$. के लिए $ p = 17$:
$(\eta_0 - \eta_1)^2 = (\eta_0 + \eta_1)^2 - 4\eta_0\eta_1 =
1 + 16 = 17$, भाग II से मेल खाता हुआ।

\textbf{16.} $ g^2 = p^*$ प्रदर्श $\sqrt{p^*} = \pm g \in
\Q(\zeta_p)$, इसलिए $\Q(\sqrt{p^*})$ एक द्विघात उपक्षेत्र है.
विशिष्टता: के उपक्षेत्र \omterm{def:b3:galois:extension}{डिग्री} $2$ पत्र-व्यवहार, द्वारा \omterm{thm:b3:galois:fundamental}{गैलोइस
पत्राचार}, सूचकांक के उपसमूहों के लिए $2$ चक्रीय का
$\operatorname{Gal}(\Q(\zeta_p)/\Q) \cong (\Z/p\Z)^\times $,
और सम क्रम के एक चक्रीय समूह में बिल्कुल ऐसा ही एक होता है
उपसमूह (वर्ग)। प्रत्येक द्विघात क्षेत्र है
$\Q(\sqrt{d})$ साथ $ d $ स्क्वायरफ्री, और फ़ील्ड्स का संयोजन
$\Q(\sqrt{p^*})$, $\Q(\iu) \subseteq \Q(\zeta_4)$ और
$\Q(\sqrt2) \subseteq \Q(\zeta_8)$ एक आम के अंदर
$\Q(\zeta_N)$ प्रत्येक को पकड़ लेता है $\sqrt d $: का द्विघात मामला
क्रोनकर--वेबर।

\textbf{17.} किसी भी क्रमविनिमेय वलय में, $(x + y)^q = x^q + y^q
+ q(\cdots)$: द्विपद गुणांक $\binom qk $, $0 < k <
q $, अभाज्य से विभाज्य हैं $ q $. पर पुनरावृति $ p - 1$
की शर्तें $ g $:
\[
g^q \equiv \sum_{a}\Bigl(\frac ap\Bigr)^{q}\zeta^{aq}
= \sum_a\Bigl(\frac ap\Bigr)\zeta^{aq}
\pmod{q\Z[\zeta]},
\]
($ q $ विषम: प्रतीक अपरिवर्तित है)। अनुक्रमणः $ b = aq $: $ a =
q^{-1}b $ और $\bigl(\frac{q^{-1}b}p\bigr) =
\bigl(\frac{q}p\bigr)\bigl(\frac bp\bigr)$ (गुणात्मकता;
$\bigl(\frac{q^{-1}}p\bigr) = \bigl(\frac qp\bigr)$ के बाद से
व्युत्क्रम का प्रतीक, प्रतीक के बराबर होता है): $ g^q \equiv
\bigl(\frac qp\bigr)g $.

\textbf{18.} $ g^q = g\,(g^2)^{(q-1)/2} = g\,(p^*)^{(q-1)/2}$
बिल्कुल (प्रश्न 15), और यूलर की कसौटी $\Z $ देता है
$(p^*)^{(q-1)/2} \equiv \bigl(\frac{p^*}q\bigr) \pmod q $,
इसलिए मॉड $ q\Z[\zeta]$: $ g^q \equiv \bigl(\frac{p^*}q\bigr)g $.
प्रश्न 17 से तुलना करना और गुणा करना $ g $:
\[
\Bigl(\frac qp\Bigr)p^* \equiv \Bigl(\frac{p^*}q\Bigr)p^*
\pmod{q\Z[\zeta]} .
\]
दोनों पक्ष परिमेय पूर्णांक हैं; उनका अंतर, $0$ या
$\pm2p^*$, में निहित है $ q\Z[\zeta] \cap \Z = q\Z $ (पूर्णांक $ m
\in q\Z[\zeta]$ है $ m/q \in \Q \cap \Z[\zeta] = \Z $, द
उत्तरार्द्ध क्योंकि $1, \zeta, \dots, \zeta^{p-2}$ एक है
$\Q $-तर्कसंगत निर्देशांक के आधार पर पढ़ना
अभिन्नता)। तब से $ q \nmid 2p^*$ ($ q $ विषम, $ q \neq p $),
अंतर यह है $0$: $\bigl(\frac qp\bigr) =
\bigl(\frac{p^*}q\bigr)$.

\textbf{19.} गुणात्मकता से, $\bigl(\frac{p^*}q\bigr) =
\bigl(\frac{-1}q\bigr)^{(p-1)/2}\bigl(\frac pq\bigr) =
(-1)^{\frac{q-1}2\cdot\frac{p-1}2}\bigl(\frac pq\bigr)$,
तो प्रश्न 18 पढ़ता है $\bigl(\frac qp\bigr)\bigl(\frac
pq\bigr) = (-1)^{\frac{p-1}2\frac{q-1}2}$: पारस्परिकता.
जाँच करना $(17, 3)$: प्रतिपादक $\frac{16}2\cdot\frac22 = 8$
सम है, इसलिए दोनों प्रतीकों का सहमत होना आवश्यक है; वर्ग मॉड $3$ हैं $\{1\}$ और $17 \equiv 2$: 
$\bigl(\frac{17}3\bigr) = -1$; वर्ग मॉड $17$ हैं $\{1, 4,
9, 16, 8, 2, 15, 13\}$ और $3$ अनुपस्थित है:
$\bigl(\frac3{17}\bigr) = -1$. उत्पाद $+1$, जैसी कि भविष्यवाणी की गई थी। के लिए $ x^2
\equiv 219 \pmod{383}$: $\bigl(\frac{219}{383}\bigr) =
\bigl(\frac3{383}\bigr)\bigl(\frac{73}{383}\bigr)$. पहला:
$383 \equiv 3 \pmod4$ और $3 \equiv 3$: पारस्परिकता देता है
$\bigl(\frac3{383}\bigr) = -\bigl(\frac{383}3\bigr) =
-\bigl(\frac23\bigr) = -(-1) = +1$. दूसरा: $73 \equiv 1
\pmod 4$: $\bigl(\frac{73}{383}\bigr) =
\bigl(\frac{383}{73}\bigr) = \bigl(\frac{18}{73}\bigr) =
\bigl(\frac2{73}\bigr)$ ($18 = 2\cdot3^2$), और $73 \equiv 1
\pmod 8$ बनाता है $2$ एक चौकोर मॉड $73$ (पूरक कानून,
द्वारा सिद्ध $ g = \zeta_8 + \zeta_8^{-1} = \sqrt2$ में
$\Q(\zeta_8)$ उसी विधि से): $+1$. कुल $+1$: द
सर्वांगसमता है \omterm{def:b3:groups:derived}{व्याख्या करने योग्य}.

\textbf{20.} आदिम जड़ का अस्तित्व एवं विशिष्टता:
यदि $ w $ अवधि निर्धारित है $\{d : w = u^{n/d},\ \abs u = d\}$,
न्यूनतम ऐसा $ d_0$ हर दूसरे काल को विभाजित करता है $ d $
(अगर $ w $ दोनों एक है $ d $-शक्ति और ए $ d'$-शक्ति, यह एक है
$\gcd(d, d')$-शक्ति: सहमत सूचकांकों पर अक्षरों की तुलना करें
\omterm{def:b3:modules:module}{मॉड्यूल}ो जीसीडी, बी के माध्यम सेézout), और लंबाई-$ d_0$ ब्लॉक है
आदिम. क्रमबद्ध करना $ q^n $ शब्द उनकी लंबाई के अनुसार
आदिम जड़: $ q^n = \sum_{d \mid n}A(d)$. तब से $ A $ और
$ d\,N_q(d)$ समान मानों के साथ समान रिकर्सन को संतुष्ट करें
के लिए $ n = 1$ (दोनों एक दूसरे को आगमनात्मक रूप से निर्धारित करते हैं $ q^n
= \sum_{d\mid n}(\cdot)$), वे बराबर हैं: $ A(d) =
d\,N_q(d)$. आक्षेप: एक तत्व $\alpha $ का \omterm{def:b3:galois:extension}{डिग्री} $ d $
शब्द उत्पन्न करता है $ w_\alpha $ के गुणांकों का\dots\
बेहतर, सीधे: के तत्व \omterm{def:b3:galois:extension}{डिग्री} $ d $ में
$\overline{\mathbb F_q}$ की जड़ें हैं $ N_q(d)$
इरेड्यूसिबल्स का \omterm{def:b3:galois:extension}{डिग्री} $ d $, प्रत्येक अपना योगदान दे रहा है $ d $
विशिष्ट जड़ें (पृथक्करण): $ d\,N_q(d)$ घटक
\omterm{def:b3:galois:extension}{डिग्री} $ d $, गिनती का मिलान $ q^n = \sum_{d\mid n}\#\{
\text{elements of degree } d \text{ in } \mathbb F_{q^n}\}$
--- एक ही छलनी, एक बार शब्दों पर, एक बार फ़ील्ड तत्वों पर।

\textbf{21.} लेम्मा: $\sum_{d \mid m}\mu(d) = \mathbf
1_{m=1}$. के लिए $ m > 1$ एक प्राइम ठीक करें $ p \mid m $: वर्गमुक्त
के विभाजक $ m $ के रूप में जोड़ी बनाएं $\{d, pd\}$ साथ $ p \nmid d $,
और $\mu(pd) = -\mu(d)$: योग रद्द हो जाता है. फिर, के लिए $ f(n) =
\sum_{e\mid n}g(e)$:
\[
\sum_{d \mid n}\mu(d)\,f\bigl(\tfrac nd\bigr)
= \sum_{d \mid n}\mu(d)\!\!\sum_{e \mid n/d}\!\!g(e)
= \sum_{e \mid n}g(e)\!\!\sum_{d \mid n/e}\!\!\mu(d)
= g(n) .
\]
साथ $ f(n) = q^n $ और $ g(n) = nN_q(n)$
(\cref{exo:b3:galois:6}): $ N_q(n) =
\frac1n\sum_{d\mid n}\mu(d)\,q^{n/d}$.

\textbf{22.}  $ d = 1$ शब्द है $ q^n $; हर दूसरे शब्द में है
$\abs{\mu(d)q^{n/d}} \leq q^{n/2}$, और बेरहमी से
$\sum_{d \mid n, d > 1}q^{n/d} \leq \sum_{j \leq
n/2}q^j < 2q^{n/2}$ (ज्यामितीय, $ q \geq 2$). इसलिए $ nN_q(n) >
q^n - 2q^{n/2} \geq 0$ के लिए $ n \geq 1$: इरेड्यूसिबल्स हरेक का
\omterm{def:b3:galois:extension}{डिग्री} अस्तित्व में है, और $\mathbb F_q[X]/(\pi) = \mathbb F_{q^n}$ है
(पुनः) निर्मित --- जनगणना के साथ अस्तित्व। का अनुपात
इरेड्यूसिबल्स बिच में $ q^n $ मोनिक डिग्री-$ n $ बहुपद है
$\frac1n(1 + O(q^{-n/2}))$: का अभाज्य संख्या प्रमेय
$\mathbb F_q[X]$, साथ $ n $ खेलना $\log x $. के लिए $ q = 2$ द
मायने रखता है $2, 1, 2, 3$ का \cref{exo:b3:galois:6} का मिलान करें
सूत्र: उदा.\ $ N_2(4) = \frac14(2^4 - 2^2) = 3$.

\textbf{23.} अशून्य के गुणनात्मक एबेलियन समूह में
तर्कसंगत कार्य समाप्त $\mathbb F_q $, तय करना $ F(n) = X^{q^n} -
X $ और $ G(n) = \prod_{\deg\pi = n}\pi $;
\cref{exo:b3:galois:6} कहते हैं $ F(n) = \prod_{d\mid n}G(d)$.
उन्हें\"प्रश्न 21 का ओबियस तर्क, लिखित
गुणात्मक रूप से (घातांक बिल्कुल वैसे ही जोड़ते हैं जैसे योग जोड़ा गया था),
देता है $ G(n) = \prod_{d \mid n}F(d)^{\mu(n/d)}$. के लिए $ q = 2$,
$ n = 2$: $ G(2) = \frac{X^4 - X}{X^2 - X} = \frac{X(X^3 -
1)}{X(X - 1)} = X^2 + X + 1$, अद्वितीय अलघुकरणीय
द्विघात ओवर $\mathbb F_2$, जैसा कि यह होना चाहिए।

\textbf{24.} $\omega^2 = \iu $ और $\omega^{-2} = -\iu $, इसलिए
$ g^2 = \omega^2 + 2 + \omega^{-2} = 2$. नये आदमी का सपना
क्रमविनिमेय वलय $\Z[\omega]/q\Z[\omega]$: $ g^q =
(\omega + \omega^{-1})^q \equiv \omega^q + \omega^{-q}$. का मान है $\omega^q + \omega^{-q}$ पर ही निर्भर करता है $ q \bmod 8$:
के लिए $ q \equiv \pm1$, $\omega^q + \omega^{-q} = \omega^{\pm1} +
\omega^{\mp1} = g $; के लिए $ q \equiv \pm3$, का उपयोग करना $\omega^4 =
-1$, $\omega^{3} = -\omega^{-1}$ और $\omega^{-3} = -\omega $,
इसलिए $\omega^q + \omega^{-q} = -g $. वहीं दूसरी ओर $ g^q =
g\,(g^2)^{(q-1)/2} = g\,2^{(q-1)/2} \equiv \bigl(\frac2q\bigr)
g \pmod{q\Z[\omega]}$ यूलर के मानदंड मॉड द्वारा $ q $. तुलना करना
और से गुणा करना $ g $: $2\bigl(\frac2q\bigr) \equiv
\pm2 \pmod{q\Z[\omega]}$; यदि संकेत असहमत हों, $ q $ होगा
बाँटना $4$ में $\Z[\omega]$, इसलिए में $\Z $ ($ q\Z[\omega] \cap
\Z = q\Z $: आधार पर समन्वय करता है $1, \omega, \omega^2,
\omega^3$), के लिए असंभव $ q $ विषम। इसलिए $\bigl(\frac2q\bigr) =
+1$ आईएफएफ $ q \equiv \pm1 \pmod 8$. समता जाँच: $ q = 8k \pm 1$
देता है $(q^2 - 1)/8 = 2k(4k \pm 1)$, यहां तक की; $ q = 8k \pm 3$
देता है $(q^2 - 1)/8 = 8k^2 \pm 6k + 1$, विषम: सूत्र
$(-1)^{(q^2-1)/8}$ केस स्प्लिट को एन्कोड करता है। संख्यात्मक रूप से: $3^2
= 9 \equiv 2 \pmod 7$ ($7 \equiv -1$), $6^2 = 36 \equiv 2
\pmod{17}$ ($17 \equiv 1$); वर्ग मॉड $3$ हैं $\{0,
1\}$ और मॉड $5$ हैं $\{0, 1, 4\}$, न ही युक्त $2$
($3 \equiv 3$, $5 \equiv -3 \pmod 8$).

\textbf{25.} हर राक्षस $ f \in \mathbb F_q[X]$ कारक
विशिष्ट रूप से $\prod_\pi\pi^{e_\pi}$ मोनिक के ऊपर
इरेड्यूसिबल्स: के अनुसार क्रमबद्ध करना \omterm{def:b3:galois:extension}{डिग्री},
\[
\sum_{f \text{ monic}}t^{\deg f}
= \prod_\pi\ \sum_{e \geq 0}t^{e\deg\pi}
= \prod_\pi\bigl(1 - t^{\deg\pi}\bigr)^{-1}
= \prod_{n \geq 1}\bigl(1 - t^n\bigr)^{-N_q(n)},
\]
सभी उत्पाद टी-एडली वैध (केवल) \omterm{def:b3:galois:extension}{डिग्री} $\leq m $
के गुणांक को स्पर्श करें $ t^m $, और निश्चित रूप से बहुत सारे हैं
इरेड्यूसिबल्स प्रत्येक का \omterm{def:b3:galois:extension}{डिग्री}). बाईं ओर है $\sum_m
q^mt^m = (1 - qt)^{-1}$: पहचान. लघुगणक: $-\log(1 -
qt) = \sum_m\frac{q^m}mt^m $, जबकि $\sum_nN_q(n)\bigl(-\log(1
- t^n)\bigr) = \sum_nN_q(n)\sum_k\frac{t^{nk}}k $; द
का गुणांक $ t^m $ देता है $\frac{q^m}m = \sum_{dk = m}
\frac{N_q(d)}k = \frac1m\sum_{d \mid m}d\,N_q(d)$, अर्थात\
$ q^m = \sum_{d\mid m}d\,N_q(d)$. हाथ से जांचें, $ q = 2$,
का गुणांक $ t^2$: $ N_2(1) = 2$, $ N_2(2) = 1$, और
$(1 - t)^{-2}(1 - t^2)^{-1} = (1 + 2t + 3t^2 + \dots)(1 + t^2
+ \dots)$ है $ t^2$-गुणांक $3 + 1 = 4 = 2^2$. अंततः,
भाजक के साथ प्रश्न 21 का सूत्र $1, 2, 3, 6$:
\[
N_2(6) = \tfrac16\bigl(2^6 - 2^3 - 2^2 + 2\bigr)
= \tfrac{54}6 = 9,
\]
सचमुच $1\cdot2 + 2\cdot1 + 3\cdot2 + 6\cdot9 = 2 + 2 +
6 + 54 = 64 = 2^6$.
\end{solution}
