\begin{solution}{exo:b3:residues:1}
$\frac{\sin z}z = 1 - \frac{z^2}6 + \cdots $: हटाने योग्य,
\omterm{def:b3:residues:singularities}{अवशेष} $0$. $\frac{\eu^z - 1}{z^2} = \frac1z + \frac12 +
\frac z6 + \cdots $: \omterm{def:b3:groups:simple}{सरल} ध्रुव, \omterm{def:b3:residues:singularities}{अवशेष} $1$.
$\frac1{z(z-1)^2}$: \omterm{def:b3:groups:simple}{सरल} पोल पर $0$ साथ \omterm{def:b3:residues:singularities}{अवशेष}
$\frac1{(0-1)^2} = 1$; डबल पोल पर $1$ साथ \omterm{def:b3:residues:singularities}{अवशेष}
$\frac{\dd}{\dd z}\bigl(\frac1z\bigr)\big|_{z=1} = -1$.
$\frac{\cos z}{z^3} = \frac1{z^3} - \frac1{2z} + \cdots $: पोल
क्रम का $3$, \omterm{def:b3:residues:singularities}{अवशेष} $-\frac12$. $\eu^{1/z} = \sum_{n\geq0}
\frac{z^{-n}}{n!}$: आवश्यक, \omterm{def:b3:residues:singularities}{अवशेष} $1$. $\frac1{\sin z}$:
साधारण डंडे पर $ n\pi $; \omterm{def:b3:residues:singularities}{अवशेष} $\frac1{\cos 0} = 1$ पर $0$,
$\frac1{\cos\pi} = -1$ पर $\pi $
(\cref{met:b3:residues:compute}, $ g/h'$).
\end{solution}

\begin{solution}{exo:b3:residues:2}
साथ $ z = \eu^{\iu t}$, $\cos t = \frac{z + z^{-1}}2$, $\dd t
= \frac{\dd z}{\iu z}$:
\[
\int_0^{2\pi}\frac{\dd t}{a + \cos t}
= \oint_{\abs z = 1}\frac{2\,\dd z}{\iu\,(z^2 + 2az + 1)} .
\]
जड़ें $ z_\pm = -a \pm \sqrt{a^2 - 1}$ संतुष्ट $ z_+z_- =
1$ साथ $\abs{z_+} < 1 < \abs{z_-}$; \omterm{def:b3:residues:singularities}{अवशेष} पर $ z_+$ है
$\frac{1}{z_+ - z_-} = \frac1{2\sqrt{a^2-1}}$, तो अभिन्न
है $\frac2\iu\cdot2\iu\pi\cdot\frac1{2\sqrt{a^2-1}} =
\frac{2\pi}{\sqrt{a^2-1}}$. जैसा $ a \to 1^+$ यह फूट जाता है (द
इंटीग्रैंड शिखर पर $ t = \pi $); जैसा $ a \to \infty $ यह व्यवहार करता है
जैसे $\frac{2\pi}a $, मेल मिलाना $\int\frac{\dd t}a $.
\end{solution}

\begin{solution}{exo:b3:residues:3}
पहला अभिन्न अंग: के ऊपरी ध्रुव $\frac{z^2}{1+z^6}$ पर $ p =
\eu^{\iu\pi/6}, \iu, \eu^{5\iu\pi/6}$; हरेक पर,
$\operatorname{Res} = \frac{p^2}{6p^5} = \frac{p^3}{6p^6} =
-\frac{p^3}6$, और $ p^3$ मान लेता है $\iu, -\iu, \iu $:
की राशि \omterm{def:b3:residues:singularities}{अवशेष} $-\frac{\iu}{6}$. चाप है
$ O(R^{-4})\cdot O(R) \to 0$:
\[
\int_\R\frac{x^2\,\dd x}{1 + x^6} =
2\iu\pi\Bigl(-\frac \iu6\Bigr) = \frac\pi3 .
\]
दूसरा: डबल पोल पर $\iu $ का $\frac1{(1+z^2)^2} =
\frac1{(z-\iu)^2(z+\iu)^2}$:
\[
\operatorname{Res} = \frac{\dd}{\dd z}\,(z +
\iu)^{-2}\Big|_{z=\iu} = \frac{-2}{(2\iu)^3} =
\frac{-2}{-8\iu} = -\frac\iu4,
\qquad
\int_\R\frac{\dd x}{(1+x^2)^2} =
2\iu\pi\cdot\Bigl(-\frac\iu4\Bigr) = \frac\pi2 ,
\]
के साथ संगत \cref{exo:b3:fouriertransform:3}(बी)।
\end{solution}

\begin{solution}{exo:b3:residues:4}
\omterm{def:b3:topology:interior}{बंद} करना $\frac{\eu^{\iu tz}}{z^2 + a^2}$ ऊपरी आधे तल में
($ t \geq 0$): वहाँ $\abs{\eu^{\iu tz}} =
\eu^{-t\operatorname{Im}z} \leq 1$, इसलिए चाप योगदान देता है
$ O(R^{-2})\cdot O(R) \to 0$. एकल संलग्न खंभा $\iu a $
के साथ \omterm{def:b3:groups:simple}{सरल} है \omterm{def:b3:residues:singularities}{अवशेष} $\frac{\eu^{-at}}{2\iu a}$:
\[
\int_\R\frac{\eu^{\iu tx}}{x^2 + a^2}\dd x =
\frac{\pi}{a}\,\eu^{-at},
\qquad\text{hence}\qquad
\int_\R\frac{\cos(tx)}{x^2+a^2}\dd x = \frac\pi
a\,\eu^{-a\abs t}
\]
(वास्तविक भाग; यहाँ तक कि $ t $). यह व्युत्क्रम प्रतिरूप है
का $\widehat{\eu^{-a\abs x}} = \frac{2a}{a^2+\xi^2}$
(\cref{exo:b3:fouriertransform:1}): दो गणनाएँ
के माध्यम से एक दूसरे की पुष्टि करें
\cref{thm:b3:fouriertransform:inversion}.
\end{solution}

\begin{solution}{exo:b3:residues:5}
आंशिक अंश: $ f = \frac1{z-2} - \frac1{z-1}$.
पर $\abs z < 1$ (टेलर):
$ f = \sum_{n\geq0}\bigl(1 - 2^{-n-1}\bigr)z^n $.
पर $1 < \abs z < 2$: $\frac1{z-2} =
-\sum_{n\geq0}\frac{z^n}{2^{n+1}}$ और $-\frac1{z-1} =
-\sum_{n\geq1}z^{-n}$: एक वास्तविक दोतरफा श्रृंखला।
पर $\abs z > 2$: $ f = \sum_{n\geq1}\bigl(2^{n-1} -
1\bigr)z^{-n}$. लॉरेंट विस्तार के कारण तीनों भिन्न हैं
से \omterm{def:b3:topology:connected}{जुड़े हुए} हैं \emph{वार्षिकी}, अंक नहीं: प्रत्येक क्षेत्र का अपना होता है
स्वयं के ज्यामितीय विस्तार। $ z^{-1}$-गुणांक ($-1$ और
$0$ क्रमशः) घेरने वाले वृत्तों पर अभिन्न अंग हैं
\emph{अंदर की विलक्षणताएँ}, नहीं \omterm{def:b3:residues:singularities}{अवशेष} पर $0$ ($ f $ है
\omterm{def:b3:holomorphic:holo}{होलोमार्फिक} पर $0$): के लिए $1 < \abs z < 2$ गुणांक
$-1$ है $\operatorname{Res}(f, 1)$; के लिए $\abs z > 2$ द
गुणांक $0$ है $\operatorname{Res}(f,1) +
\operatorname{Res}(f,2) = -1 + 1$. \omterm{def:b3:residues:singularities}{अवशेष} का $ f $: $-1$
पर $1$ और $+1$ पर $2$.
\end{solution}

\begin{solution}{exo:b3:residues:6}
(ए) द \omterm{thm:b3:residues:laurent}{लॉरेंट श्रृंखला} $\sum_nz^{-n}/n!$ अपरिमित रूप से अनेक हैं
नकारात्मक शब्द: आवश्यक. हल $\eu^{1/z} = w $ ($ w \neq
0$): $\frac1z = \log\abs w + \iu\arg w + 2\iu\pi k $, इसलिए
\[
z_k = \frac1{\log\abs w + \iu\arg w + 2\iu\pi k}
\xrightarrow[k\to\infty]{} 0 :
\]
प्रत्येक अशून्य मान अनंत बार निकट प्राप्त होता है $0$ ---
की तुलना में मजबूत \omterm{ex:b3:lebesgue:gamma}{घनत्व}.
(बी) साथ में $ z = 1/x $, $ x \to +\infty $: $\eu^x \to \infty $;
साथ में $ z = \iu/y $: मापांक $1$. एक पोल की आवश्यकता है
$\abs{f(z)} \to \infty $ \emph{हर दृष्टिकोण के साथ}: यहाँ
सीमा का अस्तित्व ही नहीं है, यहाँ तक कि इसमें भी $[0, +\infty]$.
\end{solution}

\begin{solution}{exo:b3:residues:7}
(ए) चालू $\abs z = 1$: $\abs{z^7 + z - 1} \leq 3 < 4 =
\abs{-4z^3}$. रॉच\'ई के साथ $ f = -4z^3$, $ g = z^7 + z - 1$:
डिस्क में तीन शून्य.
(बी) चालू $\abs z = 1$: $\abs{z^4 + 1} \leq 2 < 5 = \abs{5z}$:
एक शून्य में $\abs z < 1$. पर $\abs z = 2$: $\abs{5z + 1} \leq
11 < 16 = \abs{z^4}$: चार शून्य में $\abs z < 2$. अत: तीन
वलय में शून्य.
(सी) के लिए $ P = z^n + a_{n-1}z^{n-1} + \dots $: पर $\abs z = R >
1 + \sum\abs{a_k}$, $\abs{P - z^n} \leq
\bigl(\sum\abs{a_k}\bigr)R^{n-1} < R^n = \abs{z^n}$: $ P $ है
बिलकुल $ n $ शून्य में $ D(0, R)$ --- डी'अलेम्बर्ट--गॉस के साथ
बहुलता, शुद्ध गणना द्वारा।
\end{solution}

\begin{solution}{exo:b3:residues:8}
(ए) मान लीजिए $ f(a) = 0$, $ f \not\equiv 0$: चुनना $ r $ साथ $ f $
वृत्त पर शून्\omterm{def:b3:modules:free}{य-मुक्त} $ C = \partial D(a, r)$ (\omterm{thm:b3:holomorphic:identity}{पृथक
शून्य}) और $ m = \min_C\abs f > 0$. द्वारा
\cref{thm:b3:holomorphic:weierstrassconv}, $ f_n \to f $ और
$ f_n' \to f'$ समान रूप से चालू $ C $; बड़े के लिए $ n $, $\abs{f_n}
\geq m/2$ पर $ C $, इसलिए
\[
\frac1{2\iu\pi}\int_C\frac{f_n'}{f_n}
\longrightarrow \frac1{2\iu\pi}\int_C\frac{f'}{f} \geq 1
\]
(सीमा शून्य को गिनती है $ a $; अभिसरण क्योंकि
अंश एकसमान रूप से अभिसरित होते हैं और हर एकसमान रूप से मिलते हैं
नीचे परिबद्ध)। बाईं ओर शून्यों की गिनती करने वाला एक पूर्णांक है
$ f_n $ डिस्क में: यह होना ही चाहिए $\geq 1$ अंततः ---
शून्\omterm{def:b3:modules:free}{य-मुक्त}ि का खंडन। इसलिए $ f $ शून्\omterm{def:b3:modules:free}{य-मुक्त} है.

(बी) ठीक करें $ w \in \Omega $ और पर (ए) लागू करें \omterm{def:b3:topology:connected}{जुड़े हुए} \omterm{def:b3:topology:topology}{खुला सेट}
$\Omega\setminus\{w\}$ (एक खुले बिंदु को हटाना \omterm{def:b3:topology:connected}{जुड़े हुए}
का उपसमुच्चय $\C $ बरकरार रखता है संयुक्तता) को $ g_n(z) = f_n(z) -
f_n(w)$, इंजेक्शन द्वारा वहां शून्\omterm{def:b3:modules:free}{य-मुक्त}, अभिसरण $ g = f
- f(w)$. अगर $ f $ अचर है, $ g \not\equiv 0$ पर
$\Omega\setminus\{w\}$, इसलिए $ g $ वहां शून्\omterm{def:b3:modules:free}{य-मुक्त} है: $ f(z) \neq
f(w)$ सभी के लिए $ z \neq w $. जैसा $ w $ मनमाना था, $ f $ है
इंजेक्शन
\end{solution}

\begin{solution}{exo:b3:residues:9}
सेक्टर की सीमा शामिल है $[0, R]$, चाप $ A_R $, और
किरण $\eu^{2\iu\pi/n}[0, R]$ उलटा। अंदर झूठ है
एकल ध्रुव $ p = \eu^{\iu\pi/n}$ का $\frac1{1+z^n}$, साथ
\omterm{def:b3:residues:singularities}{अवशेष} $\frac1{np^{n-1}} = \frac{p}{np^n} = -\frac pn $. पर
लौटती किरण, $ z = \eu^{2\iu\pi/n}x $ देता है $ z^n = x^n $ और
$\dd z = \eu^{2\iu\pi/n}\dd x $; चाप है
$ O(R^{-n})\cdot O(R) \to 0$. इस तरह
\[
\bigl(1 - \eu^{2\iu\pi/n}\bigr)
\int_0^\infty\frac{\dd x}{1 + x^n}
= 2\iu\pi\Bigl(-\frac{\eu^{\iu\pi/n}}n\Bigr),
\quad\text{so}\quad
\int_0^\infty\frac{\dd x}{1+x^n}
= \frac{2\iu\pi}{n\,\bigl(\eu^{\iu\pi/n} -
\eu^{-\iu\pi/n}\bigr)} = \frac{\pi}{n\sin(\pi/n)} .
\]
$ n = 2$: $\frac\pi{2\sin(\pi/2)} = \frac\pi2 =
[\arctan]_0^\infty $. जैसा $ n \to \infty $: मान की ओर प्रवृत्त होता है
$1$, और वास्तव में इंटीग्रैंड की प्रवृत्ति होती है $\mathbf 1_{\intco01}$
(डीसीटी प्रभुत्व के साथ $\min(1, x^{-2})$ के लिए $ n \geq 2$).
\end{solution}

\begin{solution}{exo:b3:residues:10}
पर $\abs z = R $ बड़ा, $\abs f \leq C R^{-2}$: $\abs{\oint_{C_R}
f} \leq 2\pi R\cdot CR^{-2} \to 0$. लेकिन के लिए $ R $ सबसे परे
डंडे, \omterm{def:b3:residues:singularities}{अवशेष} प्रमेय देता है $\oint_{C_R}f =
2\iu\pi\sum_{\text{all }p}\operatorname{Res}(f, p)$: कुल
राशि गायब हो जाती है. के लिए $ f = \frac1{z(z-1)(z-2)}$: \omterm{def:b3:residues:singularities}{अवशेष}
$\frac1{(-1)(-2)} = \frac12$ पर $0$, $\frac1{1\cdot(-1)} = -1$
पर $1$, $\frac1{2\cdot1} = \frac12$ पर $2$ --- संक्षेप में $0$
जैसी कि भविष्यवाणी की गई थी, और
\[
\frac1{z(z-1)(z-2)} = \frac{1/2}{z} - \frac1{z - 1} +
\frac{1/2}{z-2} :
\]
 \omterm{def:b3:residues:singularities}{अवशेष} \emph{हैं} आंशिक अंश गुणांक, और
शून्य-राशि पहचान एक निःशुल्क संगतता जांच (या) प्रदान करती है
दूसरों से अंतिम गुणांक निर्धारित करता है)।
\end{solution}

\begin{solution}{exo:b3:residues:11}
कीहोल पर, चुने गए दृढ़ संकल्प के साथ:
कट के ठीक ऊपर, $\log z = \ln x $; बस नीचे, $\log z =
\ln x + 2\iu\pi $. चार टुकड़े देते हैं
\[
\Bigl(1 - \eu^{2\iu\pi(a-1)}\Bigr)\int_\varepsilon^R
\frac{x^{a-1}}{1+x}\dd x + \int_{C_R} + \int_{C_\varepsilon}
= 2\iu\pi\operatorname{Res}(f, -1) .
\]
आर्क्स: $\abs{f} \leq \frac{R^{a-1}}{R - 1}$ पर $ C_R $, लंबाई
$2\pi R $: योगदान $ O(R^{a-1}) \to 0$ ($ a < 1$);
$\abs f \leq \frac{\varepsilon^{a-1}}{1 - \varepsilon}$ पर
$ C_\varepsilon $, लंबाई $2\pi\varepsilon $: $ O(\varepsilon^a)
\to 0$ ($ a > 0$). \omterm{def:b3:residues:singularities}{अवशेष}: पर $ z = -1 = \eu^{\iu\pi}$,
$\operatorname{Res} = \eu^{(a-1)\iu\pi}$. इस तरह
\[
I(a) = \frac{2\iu\pi\,\eu^{\iu\pi(a-1)}}{1 -
\eu^{2\iu\pi(a-1)}}
= \frac{2\iu\pi}{\eu^{-\iu\pi(a-1)} - \eu^{\iu\pi(a-1)}}
= \frac{\pi}{-\sin(\pi(a-1))} = \frac{\pi}{\sin\pi a} .
\]
गामा प्रतिबिंब: $ B(a, 1-a) =
\int_0^1t^{a-1}(1-t)^{-a}\dd t $; प्रतिस्थापन $ t =
\frac x{1+x}$, $1 - t = \frac1{1+x}$, $\dd t =
\frac{\dd x}{(1+x)^2}$ में बदल देता है $\int_0^\infty
\frac{x^{a-1}}{1+x}\dd x = I(a)$, और यूलर का सूत्र
$ B(a, 1-a) = \Gamma(a)\Gamma(1-a)/\Gamma(1) $
(\cref{pb:b3:lebesgue:1}) देता है
$\Gamma(a)\Gamma(1-a) = \frac\pi{\sin\pi a}$ --- विशेष रूप से $\Gamma(\tfrac12) = \sqrt\pi $ एक बार फिर.

\end{solution}

\begin{solution}{exo:b3:residues:12}
(ए) चालू $\abs z = 1$: $\abs{z^4} = 1 < 7 \leq \abs{8z + 1}$
($\abs{8z} - 1 = 7$): $ P $ में उतने ही शून्य हैं $\mathbb D $
जैसा $8z + 1$, अर्थात् \emph{एक} (पर $-\frac18$).

(बी) चालू $\abs z = 3$: $\abs{8z + 1} \leq 25 < 81 =
\abs{z^4}$: रूच\'ई विरुद्ध $ z^4$ सभी चार शून्य देता है
$\abs z < 3$, इस तरह $4 - 1 = 3$ वलय में शून्य $1 <
\abs z < 3$. पैनापन: एक शून्य के साथ $\abs z = r \geq 2.1$
संतुष्ट करेगा $ r^4 = \abs{8z + 1} \leq 8r + 1$, लेकिन $ r^4 -
8r - 1$ के लिए बढ़ रहा है $ r \geq 2$ और बराबर है $19.45 -
16.8 - 1 = 1.65 > 0$ पर $ r = 2.1$: असंभव। तो तीन
बाहरी शून्य अंदर पड़े हैं $1 < \abs z < 2.1$. (संख्यात्मक रूप से: एक वास्तविक
शून्य निकट $-1.95$ और पास में एक संयुग्मी जोड़ा $1.04 \pm
1.73\iu $, मापांक का $2.02$ --- यही कारण है कि एक रोच\'ई
बिल्कुल त्रिज्या पर प्रयास करें $2$ विफल होना चाहिए: प्रमेय की मांग है
सख्त प्रभुत्व, और शून्य बाहर ही बैठते हैं।)

(सी) कोई शून्य नहीं $\iu\R $: $\operatorname{Re}P(\iu t) = t^4 +
1 \geq 1$. दाएँ आधे तल में शून्य: तर्क का प्रयोग करें
अर्ध-डिस्क की सीमा पर सिद्धांत $\{\abs z \leq
R,\ \operatorname{Re}z \geq 0\}$. बड़े चाप पर, $\arg P
\approx \arg z^4$ द्वारा बदल जाता है $4\cdot\pi = 2\pi\cdot2$ (द
चाप कोण फैलाता है $\pi $). से काल्पनिक अक्ष के अनुदिश
$\iu R $ नीचे तक $-\iu R $: $ P(\iu t) = (t^4 + 1) + 8\iu t $
दाहिने आधे तल में रहता है ($\operatorname{Re} > 0$), इसलिए
$\arg P $ भीतर भिन्न होता है $\intoo{-\pi/2}{\pi/2}$ और लौट आता है
नेट परिवर्तन के साथ $\to 0$ जैसा $ R \to \infty $ (दोनों \omterm{thm:b3:complete:completion}{समापन} बिंदु
$\approx \arg t^4 = 0$). कुल वाइंडिंग: $\frac{4\pi + 0}

संख्यात्मकता के अनुरूप: संयुग्मी जोड़ी $\approx

\begin{solution}{pb:b3:residues:1}
\textbf{1.} $\sin(\pi z)$ बिल्कुल \omterm{def:b3:groups:simple}{सरल} शून्य है $\Z $
($\sin\pi z = 0$ आईएफएफ $ z \in \Z $, और $(\sin\pi z)' = \pi\cos\pi
z \neq 0$ वहाँ और $\cos(\pi n) \neq 0$: $\pi\cot(\pi z) =
\pi\cos(\pi z)/\sin(\pi z)$ पर \omterm{def:b3:groups:simple}{सरल} ध्रुव हैं $\Z $ साथ
\[
\operatorname{Res}(\pi\cot\pi z,\ n)
= \frac{\pi\cos(\pi n)}{\pi\cos(\pi n)} = 1
\]
($ g/h'$ नियम, \cref{met:b3:residues:compute}).

\textbf{2.} $\frac{\sin(\pi z)}{\pi z} = 1 - \frac{(\pi z)^2}6
+ \frac{(\pi z)^4}{120} - \cdots $ है \omterm{def:b3:holomorphic:holo}{होलोमार्फिक} और शून्येतर
निकट $0$: इसका व्युत्क्रम है \omterm{def:b3:holomorphic:holo}{होलोमार्फिक}
(\cref{def:b3:holomorphic:holo}: भागफल), श्रृंखला के साथ $1 +
\frac{(\pi z)^2}{6} + \frac{7(\pi z)^4}{360} + \cdots $
(पहचान करना: $(1 - u)^{-1}$-शैली गुणांक से $ u =
\frac{(\pi z)^2}6 - \frac{(\pi z)^4}{120}$: द $ z^4$
गुणांक है $\frac1{36} - \frac1{120} = \frac{7}{360}$).
गुणा करके $\cos(\pi z) = 1 - \frac{(\pi z)^2}2 + \frac{(\pi
z)^4}{24} - \cdots $ और से विभाजित करें $ z $:
\[
\pi\cot(\pi z) = \frac1z\Bigl[1 + \pi^2z^2\Bigl(\frac16 -
\frac12\Bigr) + \pi^4z^4\Bigl(\frac7{360} - \frac1{12} +
\frac1{24}\Bigr)\Bigr] + \cdots
= \frac1z - \frac{\pi^2}3\,z - \frac{\pi^4}{45}\,z^3 -
\cdots
\]
($\frac7{360} - \frac{30}{360} + \frac{15}{360} =
-\frac8{360} = -\frac1{45}$).

\textbf{3.} लंबवत भुजाएँ $ z = \pm(N + \frac12) + \iu y $:
द्वारा $\pi $-आवधिकता $\cot $, $\cot(\pi z) = \cot(\pm\frac\pi2
+ \iu\pi y) = -\tan(\iu\pi y) = -\iu\tanh(\pi y)$, मापांक का
$\leq 1$. क्षैतिज भुजाएँ $ z = x \pm \iu(N + \frac12)$: से
$\abs{\cot(a + \iu b)}^2 = \frac{\cos^2a +
\sinh^2b}{\sin^2a + \sinh^2b} \leq \frac{1 +
\sinh^2b}{\sinh^2b} = \coth^2 b $,
\[
\abs{\cot(\pi z)} \leq \coth\bigl(\pi(N + \tfrac12)\bigr)
\leq \coth(\pi/2) < 1.1 .
\]
दोनों सीमाएँ हैं $\leq 2$.

\textbf{4.} आवेदन करना \cref{thm:b3:residues:residue} को $ F(z) =
\pi\cot(\pi z)f(z)$ पर $ C_N $ ($ N $ के सभी ध्रुवों से परे $ f $):
\[
\frac1{2\iu\pi}\oint_{C_N}F = \sum_{n=-N}^{N}f(n) +
\sum_p\operatorname{Res}(F, p),
\]
पूर्णांक ध्रुव योगदान दे रहे हैं $ f(n)$ (प्रश्न 1; $ f $
\omterm{def:b3:holomorphic:holo}{होलोमार्फिक} वहाँ)। पर $ C_N $: $\abs F \leq 2\pi\cdot
C\abs z^{-2} \leq 2\pi C N^{-2}$, और परिधि है $8(N +
\frac12)$: अभिन्न है $ O(1/N) \to 0$. होने देना $ N \to
\infty $: प्रदर्शित योग सूत्र।

\textbf{5.} क्षय $\abs f = O(\abs z^{-2})$ को मार डाला
\omterm{def:b3:holomorphic:contour}{समोच्च अभिन्न} \emph{और} बनाया $\sum\abs{f(n)}$ एकत्रित होना
के लिए $ f(z) = \frac1{z + \frac12}$: सममित योग
$\sum_{-N}^N\frac1{n + \frac12}$ दूरबीन को $0$ (शर्तें
$ n $ और $-n - 1$ रद्द करें), और दाईं ओर है
$-\operatorname{Res}\bigl(\frac{\pi\cot\pi z}{z + \frac12},
-\frac12\bigr) = -\pi\cot(-\frac\pi2) = 0$: सुसंगत ---
लेकिन $\sum\abs{f(n)}$ विचलन; विधि गणना करती है
केवल सममित (प्रधान मूल्य) सीमा।

\textbf{6.} $ g(z) = \frac{\pi\cot(\pi z)}{z^2}$: डंडे पर
शून्येतर पूर्णांक के साथ \omterm{def:b3:residues:singularities}{अवशेष} $\frac1{n^2}$, और कम से $0$
जहां, प्रश्न 2 से,
\[
g(z) = \frac1{z^3} - \frac{\pi^2}{3z} - \frac{\pi^4}{45}z -
\cdots :
\qquad \operatorname{Res}(g, 0) = -\frac{\pi^2}3 .
\]
 \omterm{def:b3:holomorphic:contour}{समोच्च अभिन्न} ऊपर $ C_N $ आदत है $0$ जैसा कि प्रश्न 4 में है
($\abs{g} = O(N^{-2})$ पर $ C_N $). इस तरह $0 =
\sum_{n\neq0}\frac1{n^2} - \frac{\pi^2}3$: $2\zeta(2) =
\frac{\pi^2}3$, $\zeta(2) = \frac{\pi^2}6$.

\textbf{7.} $ g(z) = \frac{\pi\cot(\pi z)}{z^4} = \frac1{z^5}
- \frac{\pi^2}{3z^3} - \frac{\pi^4}{45z} - \cdots $: \omterm{def:b3:residues:singularities}{अवशेष}

2\zeta(4) - \frac{\pi^4}{45}$: $\zeta(4) =
\frac{\pi^4}{90}$.

\textbf{8.} साथ $ g = \pi\cot(\pi z)/z^{2k}$: द \omterm{def:b3:residues:singularities}{अवशेष} पर
$0$ गुणांक है $ a_{2k-1}$ का $ z^{2k-1}$ में
का विस्तार $\pi\cot(\pi z)$, और लुप्त होती रूपरेखा देती है
$2\zeta(2k) + a_{2k-1} = 0$: $\zeta(2k) = -a_{2k-1}/2$, ए
का तर्कसंगत गुणज $\pi^{2k}$ कोटाजेंट के बाद से
गुणांक हैं. एक और विभाजन कदम उपज देता है $ a_5 =
-\frac{2\pi^6}{945}$, कहाँ से $\zeta(6) = \frac{\pi^6}{945}$.
के लिए $\zeta(3)$: प्राकृतिक गिरी $ g = \pi\cot(\pi z)/z^3$
का उत्पादन $\sum_{n\neq0}\frac1{n^3} = 0$ विचित्रता से ---
विधि सिद्ध करती है $0 = 0$ और सं\omterm{def:b3:galois:constructible}{रचनात्मक} रूप से विषम जेटा के प्रति अंधा है
मान (इसके लिए कोई \omterm{def:b3:topology:interior}{बंद} प्रपत्र नहीं) $\zeta(3)$ ज्ञात है; इसका
अतार्किकता, एपी\'हर 1978 में, पूरी तरह से अलग की जरूरत थी
विचार)।

\textbf{9.} $ F(z) = \frac{\pi\cot(\pi z)}{(z - w)(z + w)}$
पूर्णांकों पर ध्रुव हैं (\omterm{def:b3:residues:singularities}{अवशेष} $\frac1{n^2 - w^2}$,
संकेत पर ध्यान देना: $ f(n) = \frac1{(n-w)(n+w)} = \frac1{n^2 -
w^2}$) और \omterm{def:b3:groups:simple}{सरल} ध्रुव पर $\pm w $ साथ \omterm{def:b3:residues:singularities}{अवशेष}
$\frac{\pi\cot(\pm\pi w)}{\pm2w} = \frac{\pi\cot(\pi
w)}{2w}$ प्रत्येक ($\cot $ अजीब है)। प्रश्न 4 का तर्क
($\abs f = O(\abs z^{-2})$) देता है
\[
\sum_{n\in\Z}\frac1{n^2 - w^2} + \frac{\pi\cot(\pi w)}{w} =
0,
\qquad\text{i.e.}\qquad
\pi\cot(\pi w) = \frac1w + \sum_{n\geq1}\frac{2w}{w^2 -
n^2},
\]
(द $ n = 0$ शब्द है $-\frac1{w^2}$; पुनः समूहित करना $\pm n $). \omterm{def:b3:groups:normal}{सामान्य}
पर अभिसरण कॉम्पैक्ट कैमरों का $\C\setminus\Z $: के लिए $\abs w \leq
R $ और $ n \geq 2R $, $\abs{\frac{2w}{w^2 - n^2}} \leq
\frac{2R}{n^2 - R^2} \leq \frac{8R}{3n^2}$.

\textbf{10.} के लिए $\abs w \leq r < 1$: $\frac{2w}{w^2 - n^2} =
-\frac{2w}{n^2}\cdot\frac1{1 - w^2/n^2} =
-2\sum_{k\geq0}\frac{w^{2k+1}}{n^{2k+2}}$, साथ
$\abs{\text{terms}} \leq 2r^{2k+1}/n^{2k+2}$, सारांशित
$(n, k)$: श्रृंखला पुनर्व्यवस्था के लिए फ़ुबिनी
\[
\pi\cot(\pi w) = \frac1w -
2\sum_{k\geq0}\zeta(2k+2)\,w^{2k+1} .
\]
प्रश्न 2 से मिलान: $-2\zeta(2) = -\frac{\pi^2}3$ और
$-2\zeta(4) = -\frac{\pi^4}{45}$ --- वही मूल्य. तीन
के लिए सड़कें $\frac{\pi^2}6$: \omterm{thm:b3:hilbert:parseval}{पारसेवल} (फूरियर श्रृंखला), द
स्ट्रिंग के ग्रीन ऑपरेटर और कोटैंजेंट के दो का पता लगाएं
विस्तार; कि आवृत्तियों पर एक योग, एक ऑपरेटर ट्रेस,
और ए \omterm{def:b3:holomorphic:contour}{समोच्च अभिन्न} सहमत होना कोई दुर्घटना नहीं है --- प्रत्येक एक है
समान वर्णक्रमीय पहचान का चेहरा।

\textbf{11.} हल करना $ R \geq 1$ और चलो $ n_0$ सबसे छोटा हो
पूर्णांक $\geq 2R $. के लिए $\abs z \leq R $ और $ n \geq n_0$:
$\abs{z^2/n^2} \leq \frac14$, इसलिए $1 - z^2/n^2 \in
\bar D(1, \frac14)$, जहां मुख्य लघुगणक है
\omterm{def:b3:holomorphic:holo}{होलोमार्फिक}, और
\[
\Bigl|\log\Bigl(1 - \frac{z^2}{n^2}\Bigr)\Bigr|
\leq \sum_{k\geq1}\frac1k\,\Bigl|\frac{z^2}{n^2}\Bigr|^k
\leq \frac{\abs{z^2/n^2}}{1 - \abs{z^2/n^2}}
\leq \frac{2R^2}{n^2} :
\]
योग $ S(z) = \sum_{n\geq n_0}\log(1 - z^2/n^2)$ जुटता है
\omterm{def:b3:groups:normal}{सामान्य} रूप से चालू $\bar D(0, R)$, साथ \omterm{def:b3:holomorphic:holo}{होलोमार्फिक} आंशिक रकम
$ S_N $ और $\abs{S_N} \leq 2R^2\zeta(2)$ समान रूप से। तब से
$\abs{\eu^a - \eu^b} \leq
\eu^{\max(\abs a,\abs b)}\abs{a - b}$ (माध्य मान पर बाध्य है
खंड), पूंछ उत्पाद $\prod_{n_0\leq n\leq N} =
\eu^{S_N}$ समान रूप से अभिसरण करें $\bar D(0, R)$ को
शून्\omterm{def:b3:modules:free}{य-मुक्त} $\eu^S $. निश्चित बहुपद से गुणा करना
$ z\prod_{n<n_0}(1 - z^2/n^2)$: $ P_N \to P $ समान रूप से चालू
$\bar D(0, R)$, और
\cref{thm:b3:holomorphic:weierstrassconv} बनाता है $ P $
\omterm{def:b3:holomorphic:holo}{होलोमार्फिक} वहाँ; $ R $ मनमाना होना, $ P $ संपूर्ण है. पर
$\bar D(0, R)$ के शून्य $ P $ बहुपद के हैं
प्रीफ़ेक्टर --- मापांक के पूर्णांक $\leq R $, प्रत्येक \omterm{def:b3:groups:simple}{सरल}
($(1 - z/n)(1 + z/n)$ अलग-अलग \omterm{def:b3:groups:simple}{सरल} शून्य हैं, $\eu^S $
कोई नहीं): का शून्य सेट $ P $ है $\Z $, सभी शून्य \omterm{def:b3:groups:simple}{सरल}।

\textbf{12.} परिमित का लघुगणकीय विभेदन
उत्पाद, अपने शून्य से दूर:
\[
\frac{P_N'(z)}{P_N(z)} = \frac1z +
\sum_{n=1}^{N}\frac{-2z/n^2}{1 - z^2/n^2}
= \frac1z + \sum_{n=1}^{N}\frac{2z}{z^2 - n^2} .
\]
एक पर कॉम्पैक्ट $ K \subseteq \C\setminus\Z $: $ P_N \to P $ और
$ P_N' \to P'$ समान रूप से
(\cref{thm:b3:holomorphic:weierstrassconv}), और $\min_K
\abs P > 0$ ($ P $ पर ही गायब हो जाता है $\Z $), तो अंततः
$\abs{P_N} \geq \frac12\min_K\abs P $ और $ P_N'/P_N \to
P'/P $ समान रूप से चालू $ K $. मध्य सदस्य जुट जाता है
$\frac1z + \sum_{n\geq1}\frac{2z}{z^2-n^2} = \pi\cot(\pi z)$
प्रश्न 9 से: इसलिए $ P'/P = \pi\cot(\pi z)$ पर
$\C\setminus\Z $.

\textbf{13.} $\sin(\pi z)$ और $ P $ पूरे उसी के साथ हैं
शून्य सेट $\Z $, सभी शून्य \omterm{def:b3:groups:simple}{सरल} (प्रश्न 1 और 11)। पास में
$ m \in \Z $ लिखना $\sin(\pi z) = (z - m)\,\sigma(z)$ और
$ P(z) = (z - m)\,\psi(z)$ साथ $\sigma, \psi $ \omterm{def:b3:holomorphic:holo}{होलोमार्फिक}
और लुप्त न होना $ m $ (शक्ति श्रृंखला का कारक): $ Q =
\sin(\pi z)/P = \sigma/\psi $ का विस्तार समग्र रूप से और
प्रत्येक पूर्णांक पर शून्\omterm{def:b3:modules:free}{य-मुक्त} है, और पर शून्\omterm{def:b3:modules:free}{य-मुक्त} है
$\C\setminus\Z $ शून्\omterm{def:b3:modules:free}{य-मुक्त} कार्यों के भागफल के रूप में। वहाँ,
\[
\frac{Q'}{Q} = \frac{(\sin\pi z)'}{\sin\pi z} -
\frac{P'}{P} = \pi\cot(\pi z) - \pi\cot(\pi z) = 0 ,
\]
तो \omterm{def:b3:complete:complete}{पूरा} समारोह $ Q'$ पर गायब हो जाता है $\C\setminus\Z $,
इसलिए हर जगह से निरंतरता: $ Q $ स्थिर है. जैसा $ z \to
0$: $\sin(\pi z)/z \to \pi $ और $ P(z)/z \to 1$, इसलिए $ Q =
\pi $:
\[
\sin(\pi z) = \pi z\prod_{n\geq1}
\Bigl(1 - \frac{z^2}{n^2}\Bigr) .
\]

\textbf{14.} पर $ z = \frac12$: $1 = \sin\frac\pi2 =
\frac\pi2\prod_{n\geq1}\bigl(1 - \frac1{4n^2}\bigr) =
\frac\pi2\prod\frac{4n^2-1}{4n^2}$, इसलिए
\[
\frac\pi2 = \prod_{n\geq1}\frac{4n^2}{4n^2 - 1}
= \lim_{N\to\infty}\prod_{n=1}^N
\frac{(2n)(2n)}{(2n-1)(2n+1)}
= \lim_{N\to\infty}
\frac{2\cdot2\cdot4\cdot4\cdots(2N)(2N)}
{1\cdot3\cdot3\cdot5\cdots(2N-1)(2N+1)} :
\]
वालिस का उत्पाद, यूलर का एक-पंक्ति परिणाम
गुणनखंडीकरण.

\textbf{15.} के लिए $\abs z \leq r < 1$ प्रत्येक कारक निहित है
$ D(1, r^2) \subseteq D(1, 1)$, इसलिए $ P(z)/z = \exp\bigl(
\sum_{n\geq1}\log(1 - z^2/n^2)\bigr)$: प्रत्येक आंशिक उत्पाद
आंशिक योग का घातांक है, और दोनों पक्ष पास होते हैं
द्वारा सीमा निरंतरता का $\exp $. \omterm{def:b3:banach:operator}{दोहरी} शृंखला
\[
\sum_{n\geq1}\log\Bigl(1 - \frac{z^2}{n^2}\Bigr)
= -\sum_{n\geq1}\sum_{k\geq1}\frac{z^{2k}}{k\,n^{2k}}
= -\sum_{k\geq1}\frac{\zeta(2k)}k\,z^{2k}
= -\zeta(2)\,z^2 + O(z^4)
\]
श्रृंखला के लिए फ़ुबिनी द्वारा पुनर्व्यवस्थित: $\sum_{n,k}
\frac{r^{2k}}{kn^{2k}} \leq \sum_k\zeta(2k)r^{2k} \leq
\zeta(2)\frac{r^2}{1-r^2} < \infty $. इस तरह
\[
P(z) = z\,\exp\bigl(-\zeta(2)z^2 + O(z^4)\bigr)
= z - \zeta(2)\,z^3 + O(z^5) ,
\]
और प्रश्न 13 इसकी तुलना इससे करता है $\sin(\pi z) = \pi z -
\frac{\pi^3}6z^3 + O(z^5)$: $\pi\zeta(2) = \frac{\pi^3}6$,
यानी\ $\zeta(2) = \frac{\pi^2}6$. योगात्मक चेहरा
(प्रश्न 10) और गुणक फलक समान गणना करते हैं
संख्या.

\textbf{16.} एक पर कॉम्पैक्ट $ K \subseteq \C\setminus\Z $ द
आंशिक रकम $ S_N = \frac1z + \sum_{n\leq N}\bigl(
\frac1{z-n} + \frac1{z+n}\bigr)$ (प्रश्न 9, शर्तें
के रूप में पुनः समूहीकृत किया गया $\frac{2z}{z^2-n^2} = \frac1{z-n} +
\frac1{z+n}$) समान रूप से अभिसरण करें $\pi\cot(\pi z)$, इसलिए
\cref{thm:b3:holomorphic:weierstrassconv} देता है $ S_N' \to
(\pi\cot\pi z)' = -\pi^2/\sin^2(\pi z)$ समान रूप से चालू $ K $.
तब से $ S_N' = -\sum_{\abs n\leq N}(z - n)^{-2}$:
\[
\frac{\pi^2}{\sin^2(\pi z)} =
\sum_{n\in\Z}\frac1{(z - n)^2} ,
\]
अभिसरण \omterm{def:b3:groups:normal}{सामान्य} पर कॉम्पैक्ट कैमरों का $\C\setminus\Z $
(शर्तें $ O(n^{-2})$).

\textbf{17.} पर $ z = \frac12$ बाईं ओर है $\pi^2$; दायीं तरफ, $(\frac12 - n)^2 = \frac{(2n-1)^2}4$ साथ $2n -
1$ सभी विषम पूर्णांकों को ठीक एक बार चलाना $ n $ पर चलाता है $\Z $:
\[
\pi^2 = \sum_{n\in\Z}\frac{4}{(2n-1)^2}
= 8\sum_{m\geq0}\frac1{(2m+1)^2},
\qquad
\sum_{m\geq0}\frac1{(2m+1)^2} = \frac{\pi^2}8 .
\]
समता विभाजन: $\zeta(2) = \frac{\pi^2}8 +
\sum_{n\geq1}\frac1{(2n)^2} = \frac{\pi^2}8 +
\frac{\zeta(2)}4$, इसलिए $\frac34\zeta(2) = \frac{\pi^2}8$ और
$\zeta(2) = \frac{\pi^2}6$ एक बार फिर.

\textbf{18.} साथ $ c = \cot\theta $ और $\cot(2\theta) =
\frac{c^2-1}{2c}$: $\cot\theta - 2\cot(2\theta) = c -
\frac{c^2-1}c = \frac1c = \tan\theta $. इस तरह $\pi\tan(\pi z)
= \pi\cot(\pi z) - 2\pi\cot(2\pi z)$, और प्रश्न 9 बजे $ z $
और कम से $2z $ देता है
\[
\pi\cot(\pi z) = \frac1z +
\sum_{n\geq1}\frac{2z}{z^2 - n^2},
\qquad
2\pi\cot(2\pi z) = \frac1z +
\sum_{n\geq1}\frac{8z}{4z^2 - n^2} .
\]
दोनों शृंखलाएं प्रत्येक निश्चित बिंदु पर बिल्कुल एकाग्र होती हैं $ z $ \omterm{def:b3:topology:interior}{बंद}
ध्रुव, इसलिए अंतर को इच्छानुसार पुनः समूहीकृत किया जा सकता है: में
दूसरी श्रृंखला सम पदों पर है $ n = 2m $ देना $\frac{8z}{4z^2 -
4m^2} = \frac{2z}{z^2 - m^2}$ और पहली श्रृंखला रद्द करें
पूरी तरह से, जा रहा हूँ
\[
\pi\tan(\pi z) = -\sum_{m\geq0}\frac{8z}{4z^2 - (2m+1)^2}
= \sum_{m\geq0}\frac{8z}{(2m+1)^2 - 4z^2} ,
\]
\omterm{def:b3:groups:normal}{सामान्य} रूप से चालू कॉम्पैक्ट कैमरों टाल $\frac12 + \Z $ (शर्तें
$ O(m^{-2})$).

\textbf{19.} के लिए $\abs z \leq r < \frac12$:
\[
\frac{8z}{(2m+1)^2 - 4z^2} = \frac{8z}{(2m+1)^2}
\sum_{k\geq0}\Bigl(\frac{4z^2}{(2m+1)^2}\Bigr)^{k},
\]
साथ $\sum_{m,k}8r\,(4r^2)^k(2m+1)^{-2k-2} < \infty $ तब से
$4r^2 < 1$: फ़ुबिनी दोहरे योग को पुनर्व्यवस्थित करती है
\[
\pi\tan(\pi z) =
\sum_{k\geq0}8\cdot4^k\,\lambda(2k+2)\,z^{2k+1} .
\]
ख़िलाफ़ $\pi\tan(\pi z) = \pi^2z + \frac{\pi^4}3z^3 +
O(z^5)$: का गुणांक $ z $ देता है $8\lambda(2) = \pi^2$
---प्रश्न 17 पुनः ---और वह का $ z^3$ देता है
$32\lambda(4) = \frac{\pi^4}3$, यानी\ $\lambda(4) =
\frac{\pi^4}{96}$. सम हरों को हटाकर,
$\lambda(4) = \zeta(4) - 2^{-4}\zeta(4) =
\frac{15}{16}\zeta(4)$: $\zeta(4) = \frac{16}{15}\cdot
\frac{\pi^4}{96} = \frac{\pi^4}{90}$, मिलान प्रश्न 7.

\textbf{20.} $\cot\frac\theta2 - \cot\theta =
\frac{\cos\frac\theta2\,\sin\theta -
\cos\theta\,\sin\frac\theta2}{\sin\frac\theta2\,\sin\theta}
= \frac{\sin\frac\theta2}{\sin\frac\theta2\,\sin\theta} =
\frac1{\sin\theta}$, अंश है $\sin(\theta -
\frac\theta2)$. साथ $\theta = \pi z $, प्रश्न 9 बजे
$\frac z2$ पढ़ता $\pi\cot\frac{\pi z}2 = \frac2z +
\sum_{n\geq1}\frac{4z}{z^2 - 4n^2}$, इसलिए
\[
\frac{\pi}{\sin(\pi z)}
= \pi\cot\frac{\pi z}2 - \pi\cot(\pi z)
= \frac1z + \sum_{n\geq1}\frac{4z}{z^2 - 4n^2}
- \sum_{n\geq1}\frac{2z}{z^2 - n^2} .
\]
पूर्ण अभिसरण समता को पुनर्समूहित करने की अनुमति देता है: सम $ n
= 2m $ घटाई गई श्रृंखला में योगदान देता है
$\frac{2z}{z^2-4m^2}$, जा रहा हूँ $+\frac{2z}{z^2-4m^2}$ से
पहला योग, जबकि विषम $ n $ चिन्ह के साथ जीवित रहता है $-$:
\[
\frac{\pi}{\sin(\pi z)} = \frac1z +
\sum_{n\geq1}(-1)^n\,\frac{2z}{z^2 - n^2} .
\]
लक्षण जांचें: $\operatorname{Res}\bigl(\pi/\sin(\pi z),
n\bigr) = \pi/(\pi\cos\pi n) = (-1)^n $
(\cref{met:b3:residues:compute}), और $(-1)^n\frac{2z}{z^2 -
n^2} = \frac{(-1)^n}{z-n} + \frac{(-1)^n}{z+n}$ ले जाता है
बिलकुल वैसा ही \omterm{def:b3:residues:singularities}{अवशेष} पर $\pm n $.

\textbf{21.} के लिए $\abs z \leq r < 1$, प्रत्येक पद का विस्तार इस प्रकार करें
प्रश्न 19 में ($(-1)^n\frac{2z}{z^2-n^2} =
(-1)^{n-1}\frac{2z}{n^2}\sum_k\frac{z^{2k}}{n^{2k}}$) और
फ़ुबिनी लगाना:
\[
\frac{\pi}{\sin(\pi z)} = \frac1z +
2\sum_{k\geq0}\eta(2k+2)\,z^{2k+1},
\qquad
\eta(s) = \sum_{n\geq1}\frac{(-1)^{n-1}}{n^s} .
\]
टेलर पक्ष: $\sin(\pi z) = \pi z(1 - \frac{(\pi z)^2}6 +
O(z^4))$ देता है $\frac\pi{\sin\pi z} = \frac1z +
\frac{\pi^2}6z + O(z^3)$: $2\eta(2) = \frac{\pi^2}6$, इसलिए
$\eta(2) = \frac{\pi^2}{12}$. स्थिरता: $\eta(2) =
\zeta(2) - 2\sum_n(2n)^{-2} = (1 - 2^{1-2})\zeta(2) =
\frac{\zeta(2)}2 = \frac{\pi^2}{12}$.

\textbf{22.} पर $ z = \frac12$, प्रश्न 20 देता है
\[
\pi = 2 + \sum_{n\geq1}(-1)^n\frac{1}{\frac14 - n^2}
= 2 + 4\sum_{n\geq1}\frac{(-1)^{n-1}}{(2n-1)(2n+1)} .
\]
साथ $\frac1{(2n-1)(2n+1)} = \frac12\bigl(\frac1{2n-1} -
\frac1{2n+1}\bigr)$, और दोनों वैकल्पिक श्रृंखला अभिसरण
(लीबनिज़ मानदंड), योग इस प्रकार विभाजित होता है $\frac12\bigl[L -
(1 - L)\bigr] = L - \frac12$, कहाँ $ L = 1 - \frac13 +
\frac15 - \cdots $ और $\frac13 - \frac15 + \frac17 - \cdots
= 1 - L $. इस तरह $\pi = 2 + 4(L - \frac12) = 4L $:
\[
\frac\pi4 = 1 - \frac13 + \frac15 - \frac17 + \cdots
\]
नैतिक: प्रत्येक कर्नेल है \omterm{def:b3:residues:singularities}{मेरोमोर्फिक} एक पर डंडे के साथ
अंकगणितीय प्रगति और निर्धारित \omterm{def:b3:residues:singularities}{अवशेष}. द
कोटैंजेंट डालता है \omterm{def:b3:residues:singularities}{अवशेष} $1$ प्रत्येक पूर्णांक और योग पर
$ f(n)$; इसका व्युत्पन्न ध्रुवों और कीमतों को वर्गित करता है
$\lambda(2)$; स्पर्शरेखा ध्रुवों को स्थानांतरित करती है $\frac12 + \Z $
और कीमतें विषम भाजक; $\pi/\sin $ रखता है
पूर्णांक ध्रुव लेकिन वैकल्पिक \omterm{def:b3:residues:singularities}{अवशेष} $(-1)^n $, इसलिए
वैकल्पिक श्रृंखला. प्रथम क्रम की गुठली सभी हैं
अजीब कार्य: जोड़ी बनाना $ n $ साथ $-n $ सम-शक्ति को दोगुना कर देता है
गुणांक और विषम को नष्ट कर देता है, इसलिए $\zeta(2k)$
जबकि यंत्रवत् बाहर निकलता है $\zeta(3)$ कभी प्रकट नहीं होता. द
अंधापन समता है, तकनीक की कमी नहीं।

\textbf{23.} साथ $ w = 2\iu\pi z $:
\[
\iu\pi z + \frac{2\iu\pi z}{\eu^{2\iu\pi z} - 1}
= \iu\pi z\,\frac{\eu^{2\iu\pi z} + 1}{\eu^{2\iu\pi z} - 1}
= \iu\pi z\,\frac{\eu^{\iu\pi z} +
\eu^{-\iu\pi z}}{\eu^{\iu\pi z} - \eu^{-\iu\pi z}}
= \pi z\,\frac{\cos\pi z}{\sin\pi z} = \pi z\cot(\pi z) .
\]
इसलिए, जनरेटिंग फ़ंक्शन का उपयोग करना $ w = 2\iu\pi z $ (और
$ B_1 = -\frac12$ को रद्द कर रहा हूँ $\iu\pi z $ पद, विषम $ B $का
परे लुप्त हो जाना):
\[
\pi z\cot(\pi z) = \sum_{k\geq0}\frac{B_{2k}}{(2k)!}
(2\iu\pi z)^{2k}
= 1 + \sum_{k\geq1}(-1)^k\frac{(2\pi)^{2k}B_{2k}}{(2k)!}
z^{2k} .
\]
दूसरी ओर आंशिक अंश विस्तार (भाग IV)
देता है $\pi z\cot(\pi z) = 1 - 2\sum_{k\geq1}\zeta(2k)z^{2k}$
(प्रत्येक का विस्तार करें $\frac{2z^2}{z^2 - n^2} =
-2\sum_k\frac{z^{2k}}{n^{2k}}$ और संक्षेप में बताएं $ n $, \omterm{def:b3:groups:normal}{सामान्य}
अभिसरण के लिए विनिमय को उचित ठहराना $\abs z < 1$).
तुलना गुणांक: $-2\zeta(2k) =
(-1)^k\frac{(2\pi)^{2k}B_{2k}}{(2k)!}$, बताया गया सूत्र।

\textbf{24.} $ k = 1$: $\frac{(2\pi)^2}{2\cdot2}\cdot\frac16
= \frac{\pi^2}6$. $ k = 2$: $-\frac{(2\pi)^4}{2\cdot24}
\cdot\bigl(-\frac1{30}\bigr) = \frac{16\pi^4}{48\cdot30} =
\frac{\pi^4}{90}$. $ k = 3$: $\frac{(2\pi)^6}{2\cdot720}
\cdot\frac1{42} = \frac{64\pi^6}{1440\cdot42} =
\frac{\pi^6}{945}$.

\textbf{25.} लिखना $ C(z) = \pi z\cot(\pi z) = 1 -
2\sum_{k\geq1}\zeta(2k)z^{2k}$ (प्रश्न 23). प्रत्यक्ष
का विभेदन $ C = \pi z\cot(\pi z)$, का उपयोग करना $(\cot u)'
= -1 - \cot^2u $:
\[
zC'(z) = \pi z\cot(\pi z) - \pi^2z^2\bigl(1 +
\cot^2(\pi z)\bigr) = C - \pi^2z^2 - C^2 .
\]
अब दोनों पक्षों की शक्तियों का विस्तार करें $ z^2$. बाईं तरफ:
$\sum_k(-4k)\,\zeta(2k)\,z^{2k}$. दाहिनी ओर: का वर्ग करना
शृंखला,
\[
C - C^2 = 2\sum_{k\geq1}\zeta(2k)z^{2k}
- 4\sum_{k\geq2}\Bigl(\sum_{j=1}^{k-1}\zeta(2j)
\zeta(2k-2j)\Bigr)z^{2k},
\]
और शब्द $-\pi^2z^2 = -6\zeta(2)z^2$ केवल समायोजित करता है $ k =
1$. के गुणांकों की तुलना करना $ z^{2k}$ के लिए $ k \geq 2$:
\[
-4k\,\zeta(2k) = 2\,\zeta(2k) -
4\sum_{j=1}^{k-1}\zeta(2j)\,\zeta(2k-2j),
\]
यानी\ $\bigl(k + \frac12\bigr)\zeta(2k) =
\sum_{j=1}^{k-1}\zeta(2j)\zeta(2k-2j)$. (पर $ k = 1$ द
पहचान पढ़ता है $-4\zeta(2) = 2\zeta(2) - 6\zeta(2)$: ए
संगति जांच, नई जानकारी नहीं।) अनुप्रयोग: $ k =
2$: $\frac52\zeta(4) = \zeta(2)^2 = \frac{\pi^4}{36}$, इसलिए
$\zeta(4) = \frac{\pi^4}{90}$; $ k = 3$: $\frac72\zeta(6) =
2\zeta(2)\zeta(4) = \frac{\pi^6}{270}$, इसलिए $\zeta(6) =
\frac{2}{7}\cdot\frac{\pi^6}{270} = \frac{\pi^6}{945}$. एक
पारलौकिक बीज ($\zeta(2) = \frac{\pi^2}6$), और शुद्ध
बीजगणित प्रत्येक सम जीटा मान उत्पन्न करता है।
