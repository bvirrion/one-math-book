\section*{अध्याय \ref{ch:b3:product} --- उत्पाद उपाय,
फ़ुबिनी, चरों का परिवर्तन}

\begin{solution}{exo:b3:product:1}
$\Delta $ में \omterm{def:b3:topology:interior}{बंद} है $\intcc01^2$, इसलिए बोरेल, और $\mathcal
B(\intcc01^2)$ उत्पाद है $\sigma $-बीजगणित
(\cref{prop:b3:product:sections}(बी))। एक तरह से पुनरावृत्ति:
\[
\int_{\intcc01}\Bigl(\int \mathbf 1_\Delta(x,y)
\,\dd\lambda(y)\Bigr)\dd\mu(x)
= \int \lambda(\{x\})\,\dd\mu(x) = 0 ;
\]
दूसरा तरीका:
\[
\int_{\intcc01}\Bigl(\int\mathbf
1_\Delta(x,y)\,\dd\mu(x)\Bigr)\dd\lambda(y)
= \int \mu(\{y\})\,\dd\lambda(y) = \int 1\,\dd\lambda = 1 .
\]
असफल परिकल्पना है $\sigma $-गिनती की परिमितता
\omterm{def:b3:measure:measure}{उपाय} $\mu $ बेशुमार पर $\intcc01$: कोई गणनीय नहीं
परिमित का परिवार-$\mu $ सेट इसे कवर करता है।
\end{solution}

\begin{solution}{exo:b3:product:2}
पर $\intcc0A\times\intoo0{+\infty}$:
$\int_0^A\int_0^\infty\eu^{-xy}\abs{\sin x}\,\dd y\,\dd x =
\int_0^A\frac{\abs{\sin x}}x\dd x \leq A < \infty $ (टोनेली के लिए
पूर्ण मूल्य): फ़ुबिनी लागू होता है, और तब से
$\int_0^\infty\eu^{-xy}\dd y = \frac1x $,
\[
\int_0^A\frac{\sin x}x\dd x
= \int_0^\infty\Bigl(\int_0^A\eu^{-xy}\sin x\,\dd
x\Bigr)\dd y
= \int_0^\infty \frac{1 - \eu^{-Ay}(\cos A + y\sin A)}{1 +
y^2}\,\dd y
\]
(आंतरिक अभिन्न: $\operatorname{Im}\int_0^A\eu^{(\iu -
y)x}\dd x $, सीधे गणना की गई)। जैसा $ A \to \infty $, द
सुधार अवधि से घिरा है $\int_0^\infty\eu^{-Ay}\frac{1 +
y}{1 + y^2}\dd y \leq \frac32\int_0^\infty\eu^{-Ay}\dd y =
\frac3{2A} \to 0$; मुख्य शब्द है
$\int_0^\infty\frac{\dd y}{1+y^2} = \frac\pi2$. इस तरह
$\int_0^\infty\frac{\sin x}x\dd x = \frac\pi2$ --- डिरिचलेट का
फ़ुबिनी द्वारा अभिन्न।
\end{solution}

\begin{solution}{exo:b3:product:3}
(ए) लेयर केक (\cref{prop:b3:product:layercake}): $\int
f\,\dd\mu = \int_0^\infty\mu(f > t)\,\dd t $, और $ t \mapsto
\mu(f > t)$ नहीं बढ़ रहा है. पर $[n-1, n]$: $\mu(f \geq n)
\leq \mu(f > t) \leq \mu(f > n - 1) \leq \mu(f \geq n - 1)$;
इकाई अंतराल पर अभिन्नों का योग:
\[
\sum_{n\geq1}\mu(f \geq n) \leq \int f\,\dd\mu \leq
\sum_{n\geq1}\mu(f \geq n - 1) = \mu(f \geq 0) +
\sum_{n\geq1}\mu(f\geq n) \leq \mu(X) + \sum_{n\geq1}\mu(f\geq
n).
\]

(बी) को (ए) लागू करें $\abs f $: अभिन्न और की परिमितता
श्रृंखला समतुल्य है (अतिरिक्त)। $\mu(X)$ परिमित है)।
\end{solution}

\begin{solution}{exo:b3:product:4}
तब से $ f(x,y) = \partial_y\bigl(\frac{y}{x^2+y^2}\bigr)$ के लिए
$ x \ne 0$:
\[
\int_0^1 f(x, y)\,\dd y = \frac{1}{x^2 + 1}
\ \Longrightarrow\
\int_0^1\!\!\int_0^1 f\,\dd y\,\dd x =
\int_0^1\frac{\dd x}{1 + x^2} = \frac\pi4 ;
\]
प्रतिसममिति द्वारा $ f(y,x) = -f(x,y)$, दूसरा आदेश देता है
$-\frac\pi4$. निरपेक्ष मान: के लिए $0 < y < x $,
\[
\int_0^x f(x,y)\,\dd y = \Bigl[\frac{y}{x^2 +
y^2}\Bigr]_0^x = \frac1{2x},
\quad\text{and } f \geq 0 \text{ there, so}\quad
\int_0^1\!\!\int_0^1\abs f \geq \int_0^1\frac{\dd x}{2x} =
+\infty .
\]
फ़ुबिनी के साथ कोई विरोधाभास नहीं: इसकी परिकल्पना $ f \in L^1$
विफल रहता है, और दो पुनरावृत्त अभिन्न अंग बस दो भिन्न हैं
नंबर.
\end{solution}

\begin{solution}{exo:b3:product:5}
(ए) $(\mathbf 1_{\intcc01}*\mathbf 1_{\intcc01})(x) =
\lambda\bigl(\intcc01\cap\intcc{x-1}x\bigr)$: $0$ के लिए $ x
\notin \intcc02$, $ x $ के लिए $0 \leq x \leq 1$, $2 - x $ के लिए $1
\leq x \leq 2$: टेंट। पुनः संघटित करने पर एक मिलता है $\mathcal
C^1$ टुकड़े-टुकड़े-चतुष्कोणीय उभार पर $\intcc03$ (द्विघात
बी-स्पलाइन): प्रत्येक कनवल्शन एक \omterm{def:b3:galois:extension}{डिग्री} की चिकनाई प्राप्त करता है ---
इसके पीछे स्मूथिंग सिद्धांत है \cref{ch:b3:lp}के मोलिफ़ायर्स.

(बी) यदि $ x \notin \overline{\operatorname{supp}f +
\operatorname{supp}g}$, चारों ओर एक गेंद है $ x $ विच्छेद
योग समुच्चय से; के लिए $ y \in \operatorname{supp}g $, $ x - y
\notin\operatorname{supp}f $, इसलिए इंटीग्रैंड गायब हो जाता है
समान रूप से: $ f * g = 0$ पास में $ x $.

(सी) के लिए $ x_n \to x $: $(f*g)(x_n) = \int
f(y)g(x_n - y)\,\dd y $; इंटीग्रैंड्स बिंदुवार अभिसरण करते हैं
(निरंतरता का $ g $) और का प्रभुत्व है $\norm
g_\infty\,\abs f \in L^1$: डीसीटी देता है $(f*g)(x_n) \to (f*g)(x)$.
\end{solution}

\begin{solution}{exo:b3:product:6}
(ए) फ़ुबिनी और इंडक्शन द्वारा, आखिरी के साथ स्लाइसिंग
समन्वय:
\[
\lambda_d(\Delta_d) = \int_0^1
\lambda_{d-1}\bigl((1 - t)\,\Delta_{d-1}\bigr)\,\dd t

\rho^{d-1}\lambda_{d-1}(A)$
(\cref{thm:b3:product:linearchange}); साथ
$\lambda_1(\Delta_1) = 1$: आयतन $\frac1{d!}$.

(बी) $ v_2 = \pi/\Gamma(2) = \pi $; $ v_3 =
\pi^{3/2}/\Gamma(\frac52) = \pi^{3/2}/(\frac32\cdot\frac12

$ T = \operatorname{diag}(a_1, \dots, a_d)$:
\cref{thm:b3:product:linearchange} वॉल्यूम देता है

\end{solution}

\begin{solution}{exo:b3:product:7}
में $\R^2$, \omterm{ex:b3:product:polar}{ध्रुवीय निर्देशांक} (\cref{ex:b3:product:polar}):
\[
\int_{B(0,1)}\frac{\dd x\dd y}{(x^2+y^2)^s}
= 2\pi\int_0^1 r^{1 - 2s}\,\dd r,
\qquad
\int_{\R^2\setminus B(0,1)} = 2\pi\int_1^\infty r^{1-2s}\dd r:
\]
परिमित यदि $1 - 2s > -1$ ($ s < 1$), सम्मान.\ $1 - 2s < -1$ ($ s >
1$). में $\R^d $, परत के साथ गोलाकार निर्देशांक से बचें
केक: $\lambda_d(\{\norm x^{-2s} > t\}\cap B(0,1)) =
\lambda_d(B(0, \min(1, t^{-1/2s}))) = v_d\min(1, t^{-d/2s})$,
और $\int_0^\infty v_d\min(1, t^{-d/(2s)})\dd t < \infty $ आईएफएफ

\begin{solution}{exo:b3:product:8}
टोनेली (सकारात्मक इंटीग्रैंड्स) और चर के परिवर्तन द्वारा
$(x, y) = \Phi(u, v) = (uv,\ u(1-v))$, ए $\mathcal C^1$
की भिन्नता $\intoo0\infty\times\intoo01$ खुले में
चतुर्थांश के साथ
\[
\det D\Phi = \det\begin{pmatrix} v & u\\ 1 - v & -u
\end{pmatrix} = -uv - u(1 - v) = -u,
\qquad \abs{\det} = u :
\]
\[
\Gamma(p)\Gamma(q) =
\iint x^{p-1}y^{q-1}\eu^{-x-y}\dd x\,\dd y
= \iint (uv)^{p-1}\bigl(u(1{-}v)\bigr)^{q-1}\eu^{-u}\,u\,
\dd u\,\dd v
= \Gamma(p + q)\,B(p, q).
\]
स्थानापन्न $ t = \sin^2\theta $ में $ B(p,q)$ देता है
$2\int_0^{\pi/2}\sin^{2p-1}\theta\cos^{2q-1}\theta\,\dd\theta =
B(p, q)$. वालिस: $ W_n =
\int_0^{\pi/2}\sin^n\theta\,\dd\theta = \frac12B\bigl(\frac{n +
1}2, \frac12\bigr) = \frac{\Gamma(\frac{n+1}2)\sqrt\pi}
{2\,\Gamma(\frac n2 + 1)}$ --- जैसे\ $ W_{2n} =
\frac\pi2\cdot\frac{(2n)!}{4^n(n!)^2}$ का उपयोग करते हुए
$\Gamma(n + \frac12) = \frac{(2n)!}{4^nn!}\sqrt\pi $.
\end{solution}

\begin{solution}{exo:b3:product:12}
पर $\intoo01^2$, $\frac1{1 - xy} = \sum_{n\geq0}(xy)^n $ के साथ
गैर-नकारात्मक शब्द: टोनेली शब्द-दर-अवधि एकीकरण की अनुमति देता है,
\[
\iint\frac{\dd x\,\dd y}{1 - xy}
= \sum_{n\geq0}\Bigl(\int_0^1x^n\dd x\Bigr)
\Bigl(\int_0^1y^n\dd y\Bigr)
= \sum_{n\geq0}\frac1{(n+1)^2} = \zeta(2) .
\]
वैकल्पिक मामले के लिए, $\frac1{1 + xy} =
\sum_n(-1)^n(xy)^n $ एक सकारात्मक शृंखला नहीं है; लेकिन
का अभिन्न अंग \emph{निरपेक्ष} श्रृंखला है $\zeta(2) <
\infty $, तो फ़ुबिनी (अभिन्नता अब स्थापित) लागू होता है:
$\iint\frac{\dd x\dd y}{1 + xy} =
\sum_n\frac{(-1)^n}{(n+1)^2}$. श्रृंखला की पहचान:
\[
\sum_{n\geq1}\frac{(-1)^{n-1}}{n^2}
= \sum_{n\geq1}\frac1{n^2} - 2\sum_{k\geq1}\frac1{(2k)^2}
= \zeta(2) - \frac{\zeta(2)}2 = \frac{\zeta(2)}2 .
\]
साथ $\zeta(2) = \frac{\pi^2}6$ (\cref{pb:b3:spectral:1}):
अभिन्न हैं $\frac{\pi^2}6$ और $\frac{\pi^2}{12}$.
टोनेली की सकारात्मकता पहले पूरे खेल में थी
गणना --- नहीं अभिन्नता पहले जांच की जरूरत है
अदला-बदली; दूसरे में, निरपेक्ष की सकारात्मकता
श्रृंखला क्या है \emph{प्रमाणित} अभिन्नता ताकि
फ़ुबिनी हस्ताक्षरित पर चल सकता है।
\end{solution}

\begin{solution}{pb:b3:product:1}
\textbf{1.} साथ $ x = t + \sqrt t\,u $ ($\dd x = \sqrt
t\,\dd u $; $ x $ तक फैली हुई है $\intoo0\infty $ जैसा $ u $ तक फैली हुई है
$\intoo{-\sqrt t}\infty $):
\[
\Gamma(t{+}1) = \int_0^\infty x^t\eu^{-x}\dd x
= t^t\eu^{-t}\sqrt t\int_{-\sqrt t}^{\infty}
\exp\Bigl(t\ln\Bigl(1 + \frac u{\sqrt t}\Bigr) - \sqrt
t\,u\Bigr)\dd u,
\]
तब से $ x^t = t^t\exp\bigl(t\ln(1 + u/\sqrt t)\bigr)$ और
$\eu^{-x} = \eu^{-t}\eu^{-\sqrt tu}$.

\textbf{2.} तय के लिए $ u $ और $ t \to \infty $: $ t\ln(1 +
u/\sqrt t) - \sqrt tu = t\bigl(\frac u{\sqrt t} -
\frac{u^2}{2t} + o(\frac1t)\bigr) - \sqrt tu = -\frac{u^2}2 +
o(1)$: $ g_t(u) \to \eu^{-u^2/2}$.

\textbf{3.} तय करना $\psi_1(h) = \varphi(h) + \frac{h^2}4$ पर
$\intoc{-1}1$: $\psi_1(0) = 0$ और $\psi_1'(h) = \frac1{1+h} -
1 + \frac h2 = \frac{h(h-1)}{2(1+h)}$, जो है $\geq 0$ पर
$\intoc{-1}0$ और $\leq 0$ पर $\intcc01$: $\psi_1 \leq 0$,
यानी\ $\varphi(h) \leq -h^2/4$ वहाँ। तय करना $\psi_2(h) =
\varphi(h) + ch $ पर $\intco1\infty $, $ c = 1 - \ln2$: $\psi_2(1)
= \ln2 - 1 + c = 0$ और $\psi_2'(h) = c - \frac h{1+h} \leq c -
\frac12 < 0$: $\varphi(h) \leq -ch $ के लिए $ h \geq 1$. अब के लिए $ t
\geq 1$: अगर $\abs u \leq \sqrt t $, $ g_t(u) =
\eu^{t\varphi(u/\sqrt t)} \leq \eu^{-t(u/\sqrt t)^2/4} =
\eu^{-u^2/4}$; अगर $ u \geq \sqrt t $, तब $ t\,\varphi(u/\sqrt t) \leq
-ct\cdot\frac u{\sqrt t} = -c\sqrt t\,u \leq -cu $ (जैसा $ t \geq
1$), इसलिए $ g_t(u) \leq \eu^{-cu}$. इस तरह $ g_t \leq \eu^{-u^2/4} +
\eu^{-cu}\mathbf 1_{u>0}$, \omterm{def:b3:lebesgue:l1}{समाकलनीय}, से स्वतंत्र $ t \geq
1$.

\textbf{4.} डीसीटी: $\int_\R g_t(u)\dd u \to
\int_\R\eu^{-u^2/2}\dd u = \sqrt{2\pi}$
(\cref{ex:b3:product:polar} प्लस स्केलिंग
$ u\mapsto\sqrt2\,u $). प्रश्न 1 के साथ:
\[
\Gamma(t + 1) \sim \sqrt{2\pi t}\;\Bigl(\frac
t\eu\Bigr)^t,\qquad
n! \sim \sqrt{2\pi n}\,\Bigl(\frac n\eu\Bigr)^n .
\]

\textbf{5.} से \cref{exo:b3:product:8}, $ W_{2n} =
\frac\pi2\,\frac{(2n)!}{4^n(n!)^2} =
\frac\pi2\,4^{-n}\binom{2n}n $. स्टर्लिंग:
\[
\binom{2n}{n} = \frac{(2n)!}{(n!)^2}
\sim \frac{\sqrt{4\pi n}\,(2n/\eu)^{2n}}
{2\pi n\,(n/\eu)^{2n}} = \frac{4^n}{\sqrt{\pi n}} .
\]
तब $ W_{2n} \sim \frac12\sqrt{\pi/n}$, के अनुरूप
प्रत्यावर्तन $ W_n = \frac{n-1}nW_{n-2}$ (जो मजबूर करता है $ W_n \sim
W_{n-2}$, और साथ में $ W_nW_{n-1}\cdot n = \frac\pi2$ --- द
शास्त्रीय वालिस संबंध --- देता है $ W_n \sim
\sqrt{\pi/(2n)}$; दोनों स्पर्शोन्मुख सहमत हैं)।

\textbf{6.} $\Gamma(\frac d2 + 1) \sim \sqrt{2\pi\frac
d2}\,(\frac d{2\eu})^{d/2}$, इसलिए
\[
v_d = \frac{\pi^{d/2}}{\Gamma(\frac d2 + 1)}
\sim \frac{1}{\sqrt{\pi d}}\Bigl(\frac{2\pi\eu}
d\Bigr)^{d/2} \longrightarrow 0
\]
सुपर-ज्यामितीय रूप से (के लिए) $ d > 2\pi\eu \approx 17$, प्रत्येक कारक
$< 1$ और सिकुड़ रहा है)। संख्यानुसार $ v_1 = 2$, $ v_2 \approx
3.14$, $ v_3 \approx 4.19$, $ v_4 \approx 4.93$, $ v_5 \approx
5.26$, $ v_6 \approx 5.17$: अधिकतम पर है $ d = 5$.

\textbf{7.} साथ $ k = \frac n2 + \frac{s\sqrt n}2$ (पूर्णांक,
$ n $ यहां तक की, $ s $ निश्चित): लघुगणक लें $2^{-n}\binom nk =
2^{-n}\frac{n!}{k!(n-k)!}$ और तीनों पर स्टर्लिंग लगाएं
भाज्य. लिखना $ k = \frac n2(1 + \varepsilon)$, $ n - k =
\frac n2(1 - \varepsilon)$ साथ $\varepsilon = s/\sqrt n $:
\[
\ln\Bigl(2^{-n}\binom nk\Bigr)
= -\frac n2\bigl[(1{+}\varepsilon)\ln(1{+}\varepsilon) +
(1{-}\varepsilon)\ln(1{-}\varepsilon)\bigr]
+ \frac12\ln\frac{2}{\pi n(1 - \varepsilon^2)} + o(1),
\]
और ब्रैकेट है $\varepsilon^2 + O(\varepsilon^4) =
\frac{s^2}n + O(n^{-2})$: प्रदर्शन की प्रवृत्ति होती है $-\frac{s^2}2 +
\frac12\ln\frac2{\pi n}$ तक $ o(1)$, यानी
\[
2^{-n}\binom nk \sim \sqrt{\frac{2}{\pi
n}}\;\eu^{-s^2/2} :
\]
सिक्का उछालने की गॉसियन प्रोफ़ाइल, परिमाणित ---
डी मोइवरे--लाप्लास का स्थानीय रूप, जिसे वैश्वीकृत किया जाना है
\cref{ch:b3:clt}.

\textbf{8.} (i) डीसीटी प्रश्न 2 की बिंदुवार सीमा को परिवर्तित करता है

प्रभुत्व रखने वाला (ii) गॉसियन इंटीग्रल सीमा का मूल्यांकन करता है
$\int\eu^{-u^2/2} = \sqrt{2\pi}$ --- स्टर्लिंग स्थिरांक
$\sqrt{2\pi}$ \emph{है} गाऊसी अभिन्न. (iii) द
प्रतिस्थापन $ x = t + \sqrt tu $ का एक एफ़िन परिवर्तन है
चर: अनुवाद अपरिवर्तनीयता और स्केलिंग नियम
\omterm{def:b3:measure:lebesgueouter}{लेब्सेग माप} (\cref{thm:b3:product:linearchange} में

\textbf{9.} दोनों शब्दों का विस्तार करें:

d_n - d_{n+1} = \ln\frac{n!}{(n+1)!} + \Bigl(n +

= \Bigl(n + \frac12\Bigr)\ln\frac{n+1}n - 1,
\]
शर्तें $-\ln(n+1)$ और $(n + \frac32)\ln(n+1)$ संयोजन
में $(n + \frac12)\ln(n+1)$.

\textbf{10.} के लिए $ t = \frac1{2n+1}$: $\frac{1+t}{1-t} =
\frac{2n+2}{2n} = \frac{n+1}n $, और $ n + \frac12 =
\frac1{2t}$; अजीब श्रृंखला $\ln\frac{1+t}{1-t} =
2\sum_{k\geq0}\frac{t^{2k+1}}{2k+1}$ देता है
\[
\Bigl(n + \frac12\Bigr)\ln\frac{n+1}n
= \sum_{k\geq0}\frac{t^{2k}}{2k+1}
= 1 + \frac{t^2}3 + \frac{t^4}5 + \cdots
\]
घटाना $1$. निचली सीमा: अकेले पहला पद,
$\frac{t^2}3 = \frac1{3(2n+1)^2}$. ऊपरी सीमा: निचला सब
हर को $3$ और ज्यामितीय श्रृंखला का योग करें:
$\frac{t^2}{3(1 - t^2)} = \frac1{3((2n+1)^2 - 1)} =
\frac1{12n(n+1)} = \frac1{12n} - \frac1{12(n+1)}$.

\textbf{11.} से ऊपरी सीमा का योग $ n $ को $\infty $
(साथ $ d_m \to 0$): $ d_n < \frac1{12n}$. निचली सीमा के लिए:
$\frac1{12m+1} - \frac1{12(m+1)+1} =
\frac{12}{(12m+1)(12m+13)}$, और
\begin{align*}
\frac1{3(2m+1)^2} > \frac{12}{(12m+1)(12m+13)}
&\iff (12m+1)(12m+13) > 36(2m+1)^2 \\
&\iff 168m + 13 > 144m + 36,
\end{align*}
के लिए सच है $ m \geq 1$. इस टेलीस्कोपिंग माइनरेंट को सारांशित करें:
$ d_n > \frac1{12n+1}$. घातांकीकरण शास्त्रीयता प्रदान करता है
की ब्रैकेटिंग $ n!$.

\textbf{13.} लिखना $\sin^n = \sin^{n-2} - \sin^{n-2}\cos^2$,
और दूसरे पद को भागों द्वारा एकीकृत करें ($ u = \cos\theta $,
$\dd v = \sin^{n-2}\cos\theta\,\dd\theta $, $ v =
\frac{\sin^{n-1}}{n-1}$):
\[
\int_0^{\pi/2}\sin^{n-2}\cos^2 =
\Bigl[\cos\theta\,\frac{\sin^{n-1}\theta}{n-1}\Bigr]_0^{\pi/2}
+ \frac1{n-1}\int_0^{\pi/2}\sin^n = \frac{W_n}{n-1} .
\]
इस तरह $ W_n = W_{n-2} - \frac{W_n}{n-1}$, यानी\ $ W_n =
\frac{n-1}nW_{n-2}$. से $ W_0 = \frac\pi2$, $ W_1 = 1$:
\[
W_{2n} = \frac{(2n-1)!!}{(2n)!!}\cdot\frac\pi2
= \frac\pi2\binom{2n}n4^{-n},
\qquad
W_{2n+1} = \frac{(2n)!!}{(2n+1)!!}
= \frac{4^n}{(2n+1)\binom{2n}n},
\]
द्वारा दोहरे फैक्टोरियल को परिवर्तित करना $(2n)!! = 2^nn!$ और
$(2n-1)!! = \frac{(2n)!}{2^nn!}$. अंत में $ nW_nW_{n-1} =
(n-1)W_{n-1}W_{n-2}$ प्रत्यावर्तन द्वारा: स्थिरांक, के बराबर
$1\cdot W_1W_0 = \frac\pi2$.

\textbf{14.} $ W_{2n+1} \leq W_{2n} \leq W_{2n-1}$ (बिंदुवार
की एकरसता $\sin^n $) और $\frac{W_{2n-1}}{W_{2n+1}} =
\frac{2n+1}{2n} \to 1$ निचोड़ $\frac{W_{2n}}{W_{2n+1}} \to
1$. के साथ संयुक्त $ W_{2n}W_{2n+1} = \frac{\pi}{2(2n+1)}$
(प्रश्न 13): $ W_{2n}^2 \sim \frac\pi{4n}$, इसलिए $ W_{2n} \sim
\frac12\sqrt{\frac\pi n}$ और $\binom{2n}n4^{-n} =
\frac2\pi W_{2n} \sim \frac1{\sqrt{\pi n}}$.

\textbf{15.} प्रश्न 9--10 में कभी भी के मान का प्रयोग नहीं किया गया

\textbf{16.} अज्ञात-स्थिर स्टर्लिंग को प्रतिस्थापित करना
\omterm{ex:b3:groups:actions}{केंद्र}ीय द्विपद:
\[
\binom{2n}n4^{-n}\sqrt n \sim
\frac{K\,(2n)^{2n+1/2}\eu^{-2n}}{\bigl(K\,n^{n+1/2}

= \frac{2^{2n}\sqrt{2}\,K\,n^{2n+1/2}}{K^2\,n^{2n+1}}
\,4^{-n}\sqrt n = \frac{\sqrt2}{K},
\]
और प्रश्न 14 बल $\frac{\sqrt2}K = \frac1{\sqrt\pi}$:

खरोंच; इसे भाग I की पहचान मूल्यांकन में फीड करना

\textbf{17.} $\frac{\binom{2n}{n+j}}{\binom{2n}n} =
\frac{(n!)^2}{(n+j)!\,(n-j)!} =
\prod_{i=1}^{j}\frac{n-i+1}{n+i}$ के लिए $ j \geq 0$ (और द्वारा
के लिए समरूपता $ j < 0$). लघुगणक लेते हुए, साथ $1 \leq i
\leq j \leq K\sqrt n $:
\[
\ln\frac{n-i+1}{n+i} = \ln\Bigl(1 - \frac{i-1}n\Bigr) -
\ln\Bigl(1 + \frac in\Bigr) = -\frac{2i-1}{n} +
O\Bigl(\frac{i^2}{n^2}\Bigr),
\]
और $\sum_{i\leq j}(2i - 1) = j^2$, जबकि त्रुटि का योग है
$ O(j^3/n^2) = O(n^{-1/2})$: समान रूप से,
$\exp\bigl(-\frac{j^2}n + O(n^{-1/2})\bigr)$.

\textbf{18.} $\eu^{-n}\frac{n^n}{n!} \sim \eu^{-n}
\frac{n^n}{\sqrt{2\pi n}\,n^n\eu^{-n}} =
\frac1{\sqrt{2\pi n}}$. माध्य का एक पॉइसन चर $ n $ है
मानक विचलन $\sqrt n $, और $\frac1{\sqrt{2\pi n}}$ है
बिल्कुल गॉसियन शिखर की ऊंचाई $\frac1{\sigma\sqrt{2\pi}}$:
स्थानीय सीएलटी, मोड पर पूर्वावलोकन किया गया।

\textbf{19.} के लिए $ a \in \intoo01$, का ढलान लेम्मा

\textbf{21.} अंश और हर में स्टर्लिंग, साथ में $ k =
\alpha n $:
\[
\binom n{\alpha n} \sim
\frac{\sqrt{2\pi n}\,n^n}
{\sqrt{2\pi\alpha n}\,(\alpha n)^{\alpha n}\,
\sqrt{2\pi(1-\alpha)n}\,((1-\alpha)n)^{(1-\alpha)n}}
= \frac{\eu^{nH(\alpha)}}{\sqrt{2\pi\alpha(1-\alpha)n}},
\]
तब से $ n^n/(\alpha n)^{\alpha n}((1-\alpha)n)^{(1-\alpha)n} =
\alpha^{-\alpha n}(1-\alpha)^{-(1-\alpha)n} =
\eu^{nH(\alpha)}$ (की शक्तियां $ n $ रद्द करना: $\alpha n +
(1-\alpha)n = n $), और यह $\eu^{-n}$ इसी तरह रद्द कर दिया गया है। पर
$\alpha = \frac12$: $ H = \ln2$ और प्रीफ़ेक्टर है
$\sqrt{2/(\pi n)}$ --- प्रश्न 5 फिर से। की सख्त अवतलता
$ H $ (इसका दूसरा व्युत्पन्न $-\frac1{\alpha(1-\alpha)} < 0$)
अपना अधिकतम रखता है $\ln 2$ पर $\alpha = \frac12$ केवल
$\alpha \neq \frac12$, $\binom n{\alpha n}2^{-n} \approx
\eu^{-n(\ln2 - H(\alpha))}$ तेजी से क्षय होता है ---
प्रत्येक एकाग्रता कथन के पीछे कॉम्बिनेटरियल इंजन
सिक्के उछालने के बारे में.

\textbf{25.} होने देना $\lambda = \frac{\alpha}{1-\alpha} \in
\intoc01$. के लिए $ k \leq \lfloor\alpha n\rfloor \leq \alpha n $
हमारे पास है $\lambda^{k - \alpha n} \geq 1$, इस तरह
\[
\sum_{k=0}^{\lfloor\alpha n\rfloor}\binom nk
\leq \lambda^{-\alpha n}\sum_{k=0}^{n}\binom nk\lambda^{k}
= \Bigl(\lambda^{-\alpha}(1 + \lambda)\Bigr)^{n},
\]
और साथ में $\lambda = \frac\alpha{1-\alpha}$:
\[
\lambda^{-\alpha}(1+\lambda)
= \alpha^{-\alpha}(1-\alpha)^{\alpha}\cdot\frac1{1-\alpha}
= \alpha^{-\alpha}(1-\alpha)^{-(1-\alpha)}
= \eu^{H(\alpha)} .
\]
इष्टतमता: न्यूनतम करना $ f(\lambda) = -\alpha\ln\lambda +
\ln(1+\lambda)$ ऊपर $\lambda > 0$, समीकरण $ f'(\lambda)
= -\frac\alpha\lambda + \frac1{1+\lambda} = 0$ अद्वितीय है
समाधान $\lambda = \frac\alpha{1-\alpha}$, न्यूनतम तब से
$ f'' > 0$ --- घातीय-झुकाव (चेर्नॉफ़) विकल्प।
समाधान: प्रश्न 21 द्वारा एकल पद $ k =
\lfloor\alpha n\rfloor $ पहले से ही ऑर्डर है
$\eu^{nH(\alpha)}/\sqrt{2\pi\alpha(1-\alpha)n}$, इसलिए
\[
\frac{\eu^{nH(\alpha)}}{C\sqrt n} \leq
\sum_{k\leq\alpha n}\binom nk \leq \eu^{nH(\alpha)} :
\]
दर $ H(\alpha)$ सटीक है, पूरी राशि की लागत अधिकतम होगी
एक कारक $\sqrt n $ अपने सबसे बड़े कार्यकाल में। द्वारा विभाजित $2^n $,
यह फेयर-कॉइन टेल बाउंड है $\P(S_n \leq \alpha n) \leq
\eu^{-n(\ln2 - H(\alpha))}$ ---की सांद्रता \omterm{def:b3:measure:measure}{उपाय} में
एक पंक्ति.
\end{solution}
