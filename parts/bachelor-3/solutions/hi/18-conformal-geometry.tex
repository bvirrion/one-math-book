\section*{अध्याय \ref{ch:b3:conformal} --- अनुरूप मानचित्र और
रीमैन मैपिंग प्रमेय}

\begin{solution}{exo:b3:conformal:1}
(ए) $\varphi $ और $\psi(w) = \iu\frac{1+w}{1-w}$ के लिए लिखें
दोनों आदेशों में पहचान (प्रत्यक्ष गणना); $\varphi $
एमएपीएस $\mathbb H $ में $\mathbb D $ और $\psi $ पीछे
(\cref{ex:b3:conformal:mobius}). मान: $\varphi(\iu) = 0$,
$\varphi(0) = -1$, $\varphi(1) = \frac{1 - \iu}{1 + \iu} =
-\iu $, और $\varphi(z) \to 1$ जैसा $ z \to \infty $.

(बी) वृत्त और रेखाएं समाधान सेट हैं
$\alpha\abs z^2 + \bar\beta z + \beta\bar z + \gamma = 0$
($\alpha, \gamma \in \R $, $\beta \in \C $, $\abs\beta^2 >
\alpha\gamma $): $\alpha \neq 0$ वृत्त, $\alpha = 0$ पंक्तियाँ.
स्थानापन्न $ z = 1/w $ और से गुणा करना $\abs w^2$:
$\gamma\abs w^2 + \beta w + \bar\beta\bar w + \alpha = 0$ ---
एक ही परिवार. एफ़िन मानचित्र स्पष्ट रूप से परिवार को संरक्षित करते हैं, और
हर एम\"ओबियस मानचित्र एफ़िन मानचित्रों और एक की एक रचना है
उलटा ($\frac{az+b}{cz+d} = \frac ac + \frac{bc -
ad}{c}\cdot\frac1{cz + d}$ के लिए $ c \neq 0$).
\end{solution}

\begin{solution}{exo:b3:conformal:2}
(ए) श्वार्ज़ देता है $\abs{f(a)} \leq \abs a $ समानता के साथ (दोनों)
पक्ष $= \abs a $): समानता का मामला बल देता है $ f(z) =
\eu^{\iu\theta}z $, और $ f(a) = a $ पिंस $\eu^{\iu\theta} = 1$.

(बी) चलो $ a \neq b $ निश्चित बिंदु हों और $ g =
\varphi_a\circ f\circ\varphi_{-a} \in
\operatorname{Aut}(\mathbb D)$ --- का उपयोग करना
\cref{thm:b3:conformal:autdisc} के लिए $\varphi_{\pm a}$. तब
$ g(0) = \varphi_a(f(a)) = 0$ और $ g(c) = c $ के लिए $ c =
\varphi_a(b) \neq 0$: (ए) द्वारा, $ g = \mathrm{id}$, इसलिए $ f =
\varphi_{-a}\circ\varphi_a = \mathrm{id}$.
\end{solution}

\begin{solution}{exo:b3:conformal:3}
हल करना $ z $ और सेट करें $ g = \varphi_{f(z)}\circ f\circ\varphi_{-z}$:
\omterm{def:b3:holomorphic:holo}{होलोमार्फिक} $\mathbb D \to \mathbb D $ साथ $ g(0) = 0$, इसलिए
$\abs{g'(0)} \leq 1$ (श्वार्ज़)। चेन नियम के साथ
$\varphi_a'(\zeta) = \frac{1 - \abs a^2}{(1 - \bar
a\zeta)^2}$:
\[
g'(0) = \varphi_{f(z)}'\bigl(f(z)\bigr)\cdot f'(z)\cdot
\varphi_{-z}'(0)
= \frac{1}{1 - \abs{f(z)}^2}\cdot f'(z)\cdot(1 - \abs z^2),
\]
श्वार्ज़ कहाँ से--असमानता चुनें। कुछ में समानता $ z $
बनाता है $ g $ एक घूर्णन, इसलिए $ f =
\varphi_{-f(z)}\circ(\text{rotation})\circ\varphi_z \in
\operatorname{Aut}(\mathbb D)$ --- और तब समानता कायम रहती है
हर जगह (गणना करें, या भूमिकाओं के साथ पुनः लागू करें)। $ f, f^{-1}$
आदान-प्रदान किया गया)। \omterm{def:b3:holomorphic:holo}{होलोमार्फिक} डिस्क के स्व-मानचित्र हैं
$1$-हाइपरबोलिक मीट्रिक के लिए लिप्सचिट्ज़ $\frac{2\abs{\dd
z}}{1 - \abs z^2}$; ऑटोमोर्फिज्म इसकी आइसोमेट्री हैं।
\end{solution}

\begin{solution}{exo:b3:conformal:4}
(ए) $ z \mapsto \eu^z $: मानचित्र $\{0 < \operatorname{Im}z <
\pi\}$ वस्तुपरक रूप से पर $\mathbb H $ ($\eu^{x+\iu y} =
\eu^x\eu^{\iu y}$: मापांक \omterm{def:b3:modules:free}{मुक्त}, तर्क $ y \in
\intoo0\pi $), \omterm{def:b3:holomorphic:holo}{होलोमार्फिक} गैर-लुप्तप्राय व्युत्पन्न के साथ और
\omterm{def:b3:holomorphic:holo}{होलोमार्फिक} उलटा (प्रिंसिपल) $\log $).
(बी) $ z \mapsto z^2$ दोहरे तर्क: खुला चतुर्थांश
$\{0 < \arg z < \frac\pi2\}$ एमएपीएस अनुरूप रूप से पर $\mathbb
H $ (उलटा: मुख्य वर्गमूल)।
(सी) $ z \mapsto \frac{1 + z}{1 - z}$ एमएपीएस $\mathbb D $ पर
दायां आधा तल और ऊपरी/निचली समरूपता को संरक्षित करता है: यह
ऊपरी अर्ध-डिस्क को पहले चतुर्थांश पर भेजता है; फिर
वर्ग, (बी) द्वारा, तक पहुँचने के लिए $\mathbb H $: $ z \mapsto \bigl(\frac{1
+ z}{1 - z}\bigr)^2$.
(डी) द \omterm{thm:b3:conformal:autdisc}{ब्लाश्के कारक} $\varphi_{1/2}(z) = \frac{z -
\frac12}{1 - \frac z2}$: $\varphi_{1/2}(\tfrac12) = 0$ और
$\varphi_{1/2}'(\tfrac12) = \frac{1 - \frac14}{(1 -
\frac14)^2} = \frac43 > 0$.
\end{solution}

\begin{solution}{exo:b3:conformal:5}
(ए) एक अनुरूप $\C \to \mathbb D $ (या $\mathbb H $, के बाद
केली के साथ रचना करना) एक संपूर्ण कार्य है:
लिउविल द्वारा स्थिरांक --- विशेषण नहीं। एक अनुरूप $\mathbb
D \to \C $ एक अनुरूप व्युत्क्रम होगा $\C \to \mathbb D $:
वही विरोधाभास.

(बी) इसलिए, वर्ग उत्तल है \omterm{def:b3:conformal:simplyconnected}{बस जुड़ा हुआ} है, और उचित:
अनुरूप रूप से $\mathbb D $ (\cref{thm:b3:conformal:rmt}). कटौती
विमान $\C\setminus\intoc{-\infty}0$ लगभग तारे के आकार का है $1$
(सेगमेंट $1$ कटौती से बचें), इसलिए \omterm{def:b3:conformal:simplyconnected}{बस जुड़ा हुआ} है,
और उचित: अनुरूप रूप से $\mathbb D $. पंचर डिस्क: यदि
$ g \colon \mathbb D\setminus\{0\} \to \mathbb D $ थे
अनुरूप, $ g $ घिरा हुआ है, इसलिए $0$ हटाने योग्य है
(\cref{thm:b3:residues:riemanncw}): $ g $ तक फैला हुआ है $\tilde g
\colon \mathbb D \to \mathbb D $, और $\tilde g(0)$, में होना
खुली छवि $ g(\mathbb D\setminus\{0\}) = \mathbb D $, ई आल्सो $ g(w)$ कुछ के लिए $ w \neq 0$; दो विसंयुक्त पड़ोस
का $0$ और $ w $ ऐसी छवियाँ हैं जो खुली हैं और मूल्य साझा करती हैं
$\tilde g(0)$, इसलिए साझा करें \emph{अन्य} मान भी (\omterm{def:b3:topology:topology}{खुला
सेट}): $ g $ दो बार कुछ मान लेता है
$\mathbb D\setminus\{0\}$ ---विरोधाभास इंजेक्शन. द
वलय $ A = \{1 < \abs z < 2\}$: कल्पना करना $ F \colon \mathbb D
\to A $ अनुरूप. $ F $ पर शून्\omterm{def:b3:modules:free}{य-मुक्त} है \omterm{def:b3:conformal:simplyconnected}{बस जुड़ा हुआ} है
$\mathbb D $, इसलिए $ F = \eu^L $ के लिए \omterm{def:b3:holomorphic:holo}{होलोमार्फिक} $ L $
(\cref{def:b3:conformal:simplyconnected}). होने देना $\sigma $ हो
घेरा $\abs z = \frac32$ में $ A $ और $\gamma =
F^{-1}\circ\sigma $, एक \omterm{def:b3:topology:interior}{बंद} पथ में $\mathbb D $; तब
\[
1 = \operatorname{Ind}_\sigma(0)
= \frac1{2\iu\pi}\int_{F\circ\gamma}\frac{\dd w}{w}
= \frac1{2\iu\pi}\int_\gamma\frac{F'}{F}
= \frac1{2\iu\pi}\int_\gamma L' = 0
\]
($ L'$ एक आदिम है): विरोधाभास। न ही पंक्चर हुआ
डिस्क और न ही वलय भेष में एक डिस्क है।
\end{solution}

\begin{solution}{exo:b3:conformal:6}
(ए) $ T(w) = \frac{w - 1}{w + 1}$ एमएपीएस $\{\operatorname{Re}w >
0\}$ अनुरूप रूप से पर $\mathbb D $ (केली ने घुमाया: $\abs{w -
1} < \abs{w + 1}$ आईएफएफ $\operatorname{Re}w > 0$), साथ $ T(1) =
0$. के लिए $ f \in \mathcal F $, $ g = T\circ f\colon \mathbb D \to
\mathbb D $ है \omterm{def:b3:holomorphic:holo}{होलोमार्फिक} साथ $ g(0) = 0$: श्वार्ज देता है
$\abs{g(z)} \leq \abs z $.

(बी) उलटा करना $ T $: $ f = \frac{1 + g}{1 - g}$, इसलिए
\[
\abs{f(z)} \leq \frac{1 + \abs{g(z)}}{1 - \abs{g(z)}}
\leq \frac{1 + \abs z}{1 - \abs z} :
\]
स्थानीय रूप से बाउंड (समान रूप से चालू)। $\abs z \leq r < 1$). समानता
पर $ z_0 \neq 0$ बल $\abs{g(z_0)} = \abs{z_0}$ और
संरेखण: $ g $ एक रोटेशन, यानी\ $ f(z) = \frac{1 +
\eu^{\iu\theta}z}{1 - \eu^{\iu\theta}z}$ --- द
\emph{हर्ग्लोट्ज़ चरमपंथी}, \omterm{def:b3:conformal:conformal}{अनुरूप मानचित्र} दाईं ओर
आधा समतल.
\end{solution}

\begin{solution}{exo:b3:conformal:7}
(i) \omterm{def:b3:groups:simple}{सरल} कनेक्टिविटी ठीक दो बार प्रवेश करती है, के माध्यम से
का अस्तित्व \omterm{def:b3:holomorphic:holo}{होलोमार्फिक} शून्\omterm{def:b3:modules:free}{य-मुक्त} कार्यों के वर्गमूल
(\cref{def:b3:conformal:simplyconnected}): चरण 0 में (रूट
का $ z - b $) और चरण 2 में (की जड़ $\varphi_a\circ f $).
(ii) $\Omega \neq \C $ बिंदु प्रदान करता है $ b $ चरण 0 का ---
इसके बिना परिवार $\mathcal F $ इंजेक्शन से खाली है
बंधे हुए नक्शे (लिउविले)। (iii) संयुक्तता जब भी प्रयोग किया जाता है
पहचान प्रमेय या हर्विट्ज़ (\cref{exo:b3:residues:8})
बोलता है: चरम सीमा ``इंजेक्शनिव या स्थिर'' है, और
स्थिरता को बाहर रखा गया है $ s > 0$; ``शून्य व्युत्पन्न में भी
''स्थिर'' का तात्पर्य है। के लिए विफलता $\C $: चरण 0 असंभव, और
निष्कर्ष गलत है (\cref{exo:b3:conformal:5}(ए))।
वलय के लिए विफलता: नहीं \omterm{def:b3:conformal:simplyconnected}{बस जुड़ा हुआ} है --- द
वर्गमूल निर्माण टूटता है (उदा.\ $ z $ स्वयं, शून्\omterm{def:b3:modules:free}{य-मुक्त}
पर $ A $, है कोई \omterm{def:b3:holomorphic:holo}{होलोमार्फिक} वर्गमूल: समान सूचकांक
गणना के रूप में \cref{exo:b3:conformal:5}(बी) के साथ
$\frac12\int_\sigma\frac{\dd z}z \notin 2\iu\pi\Z $) --- और
निष्कर्ष भी गलत है.
\end{solution}

\begin{solution}{exo:b3:conformal:8}
के फूरियर विस्तार को प्रतिस्थापित करना $ g $ पॉइसन में
अभिन्न और उपयोग
$\frac1{2\pi}\int P_r(\varphi - t)\eu^{\iu nt}\dd t =
r^{\abs n}\eu^{\iu n\varphi}$ (पढ़ लिया $ P_r $की श्रृंखला):
$ u(r\eu^{\iu\varphi}) = \sum_nc_n(g)\,r^{\abs
n}\eu^{\iu n\varphi}$, इंटरचेंज \omterm{def:b3:groups:normal}{सामान्य} द्वारा उचित है
अभिसरण ($\abs{c_n} \leq \norm g_\infty $, $ r < 1$).

(ए) $ g = \cos t $: $ c_{\pm1} = \frac12$, इसलिए $ u =
r\cos\varphi = \operatorname{Re}z = x $ --- वास्तव में हार्मोनिक
सही सीमा मानों के साथ.

(बी) $\cos^2t = \frac12 + \frac{\cos 2t}2$: $ u = \frac12 +
\frac{r^2\cos2\varphi}2 = \frac12 +
\frac{\operatorname{Re}(z^2)}2 = \frac12 + \frac{x^2 -
y^2}{2}$.

(सी) $ g = \mathbf 1_{(0,\pi)}$ (ऊपरी अर्धवृत्त): $ c_0 =
\frac12$ और $ c_n = \frac{1 - (-1)^n}{2\iu\pi n}$ के लिए $ n \neq
0$, इसलिए
\[
u(r\eu^{\iu\varphi}) = \frac12 + \frac2\pi\sum_{k\geq0}
\frac{r^{2k+1}\sin\bigl((2k+1)\varphi\bigr)}{2k + 1},
\qquad u(0) = \frac12 :
\]
 \omterm{ex:b3:groups:actions}{केंद्र} सीमा डेटा का बिल्कुल औसत देखता है ---
व्यक्तिगत रूप से संपत्ति का औसत मूल्य।
\end{solution}

\begin{solution}{exo:b3:conformal:9}
से $(1-r)^2 \leq 1 - 2r\cos\theta + r^2 \leq (1+r)^2$:
\[
\frac{1-r}{1+r} = \frac{1 - r^2}{(1+r)^2} \leq P_r(\theta)
\leq \frac{1 - r^2}{(1 - r)^2} = \frac{1+r}{1-r} .
\]
के लिए $ u \geq 0$ हार्मोनिक चालू $\mathbb D $ और $ s < 1$: $ u_s(z) =
u(sz)$ ए पर हार्मोनिक है \omterm{def:b3:topology:topology}{अड़ोस-पड़ोस} का $\bar{\mathbb D}$,
इसलिए यह इसके पॉइसन इंटीग्रल के बराबर है
(\cref{thm:b3:conformal:poisson}, विशिष्टता, उस पर लागू होती है
स्वयं की सीमा मान); कर्नेल को सैंडविच करना और उपयोग करना
माध्य मान $\frac1{2\pi}\int u_s(\eu^{\iu t})\dd t = u(0)$:
\[
\frac{1-r}{1+r}\,u(0) \leq u(s\,r\eu^{\iu\varphi}) \leq
\frac{1+r}{1-r}\,u(0).
\]
होने देना $ s \to 1^-$ निश्चित पर $ r\eu^{\iu\varphi}$ (निरंतरता का
$ u $): हार्नैक असमानताएँ। अगर $ u $ हार्मोनिक चालू है $\C $
साथ $ u \geq m $: हार्नैक को लागू करें $ u - m $ डिस्क पर $ D(0, R)$,
यानी\ को $ z \mapsto u(Rz) - m $: तय के लिए $ z $ और $ r =
\abs z/R \to 0$, दोनों सीमाएँ प्रवृत्त होती हैं $ u(0) - m $: $ u(z) =
u(0)$ --- स्थिर (एक तरफा से दो तरफा लिउविले
बाध्य)।
\end{solution}

\begin{solution}{exo:b3:conformal:10}
स्थानीय स्तर पर, $ u = \operatorname{Re}F $ साथ $ F $ \omterm{def:b3:holomorphic:holo}{होलोमार्फिक}
(\cref{prop:b3:conformal:harmonicholo}), इसलिए $ u\circ f =
\operatorname{Re}(F\circ f)$ जहाँ भी परिभाषित किया गया है वह हार्मोनिक है:
सामंजस्य है अनुरूप रूप से अपरिवर्तनीय. पर $\mathbb H $: द
प्रमुख लघुगणक देता है $\log z = \ln\abs z + \iu\arg z $
\omterm{def:b3:holomorphic:holo}{होलोमार्फिक} पर $\mathbb H $, इसलिए $ u = \frac1\pi\arg z =
\operatorname{Im}\bigl(\frac1\pi\log z\bigr)$ हार्मोनिक है,
सीमा सीमाओं के साथ: के लिए $ x > 0$, $\arg \to 0$, $ u \to 0$;
के लिए $ x < 0$, $\arg \to \pi $, $ u \to 1$: आंकड़ा $\mathbf
1_{\intoo{-\infty}0}$ प्रत्येक पर $ x \neq 0$. द्वारा परिवहन करना
द \omterm{ex:b3:conformal:mobius}{केली नक्शा} (जो भेजता है $\mathbb D \to \mathbb H $ के बाद
उलटा और ऊपरी अर्धवृत्त को नकारात्मक से मेल खाता है
अक्ष, तीन सीमाओं का पीछा करते हुए तय किए गए घूर्णन तक
अंक), $ u\circ(\text{Cayley})$ की डिस्क समस्या का समाधान करता है
\cref{exo:b3:conformal:8}(सी); पर मूल्यांकन कर रहे हैं \omterm{ex:b3:groups:actions}{केंद्र}
retrieves $ u = \frac12$ वहाँ, और \omterm{def:b3:topology:interior}{बंद} प्रपत्र
$\frac1\pi\arg $ श्रृंखला के विरुद्ध योग द्वारा जाँच की जा सकती है
$\sum\frac{r^{2k+1}\sin((2k+1)\varphi)}{2k+1} =
\frac12\arctan\frac{2r\sin\varphi}{1 - r^2}$-प्रकार की पहचान
--- उसी उत्तर का प्राथमिक मार्ग।
\end{solution}

\begin{solution}{exo:b3:conformal:11}
(ए) चलो $ a \neq b $ ठीक किया जाए. द्वारा संयुग्मित करना $\varphi_a(z) =
\frac{z - a}{1 - \bar az}$ (एक ऑटोमोर्फिज्म एक्सचेंजिंग $ a $
और $0$), $ g = \varphi_a\circ f\circ\varphi_a^{-1}$ ठीक करता है
$0$ और बात $ c = \varphi_a(b) \neq 0$. श्वार्ज़:
$\abs{g(z)} \leq \abs z $, और कम से $ z = c $ समानता रखती है
($ g(c) = c $): समानता का मामला बल देता है $ g(z) = \lambda z $
साथ $\abs\lambda = 1$, और $\lambda c = c $ देता है $\lambda =
1$: $ g = \mathrm{id}$, इस तरह $ f = \mathrm{id}$.

(बी) $ h(z) = f(z)/z $ (\omterm{thm:b3:residues:riemanncw}{हटाने योग्य विलक्षणता} पर $0$) है
\omterm{def:b3:holomorphic:holo}{होलोमार्फिक} पर $\mathbb D $ साथ $\abs h \leq 1$ (श्वार्ज़);
$\abs h < 1$ सर्वत्र, अन्यथा अधिकतम सिद्धांत
(आंतरिक भाग अधिकतम $\abs h $) करना होगा $ h $ एक यूनिमॉड्यूलर
स्थिर, यानी\ $ f $ एक रोटेशन---बहिष्कृत। पर कॉम्पैक्ट
$\bar D(0,r)$, $ c_r = \max\abs h < 1$, इसलिए $\abs{f(z)} \leq
c_r\abs z $ वहाँ; इसके अतिरिक्त $ f $ एमएपीएस $\bar D(0,r)$ में
स्वयं ($ c_r\abs z \leq r $), इसलिए बाउंड पुनरावृत्त होता है:
$\abs{f^{\circ n}(z)} \leq c_r^n\,r \to 0$ समान रूप से चालू
$\bar D(0, r)$.

(सी) के निश्चित बिंदु $\frac{z^2 + z}2$: $ z^2 + z = 2z $ आईएफएफ
$ z(z - 1) = 0$; केवल $ z = 0$ में निहित है $\mathbb D $ ($ z = 1$ है
सीमा पर) कोई घूर्णन नहीं ($ f'(0) = \frac12$), इसलिए
कक्षाओं प्रवृत्त $0$; मात्रात्मक $ f(z) = \frac z2(1 + z)$
देता है $\abs{f(z)} \leq \frac{3}{4}\abs z $ पर $\abs z \leq
\frac12$, और एक बार \omterm{def:b3:groups:action}{कक्षा} छोटा है, $\abs{f(z)} \approx
\frac{\abs z}2$: अनुपात के साथ स्पर्शोन्मुख रूप से ज्यामितीय
$ f'(0) = \frac12$. से $ z_0 = \frac12$: $ z_1 = \frac38$,
$ z_2 \approx 0.258$, $ z_3 \approx 0.162$ --- प्रति आधा करना
चरण, जैसा कि गुणक द्वारा भविष्यवाणी की गई थी।
\end{solution}

\begin{solution}{pb:b3:conformal:1}
\textbf{1.} $ g $ इंजेक्शन है और \omterm{def:b3:holomorphic:holo}{होलोमार्फिक}; पर $ C_\rho $
($\rho > 1$) यह चिकना है, और संलग्न क्षेत्र है
$\frac1{2\iu}\oint_{g(C_\rho)}\bar\zeta\,\dd\zeta $ (वर्ष 2
हरा--रीमैन क्षेत्र सूत्र, सकारात्मक के साथ लागू
ओरिएंटेशन)। स्थानापन्न $\zeta = g(w)$, $ w =
\rho\eu^{\iu\theta}$:
\[
A_\rho = \frac1{2\iu}\int_{C_\rho}\overline{g(w)}\,g'(w)\dd
w = \frac\rho2\int_0^{2\pi}\overline{g(\rho\eu^{\iu\theta})}
\,g'(\rho\eu^{\iu\theta})\,\eu^{\iu\theta}\,\dd\theta .
\]
डालना $\bar g = \rho\eu^{-\iu\theta} + \bar b_0 +
\sum_m\bar b_m\rho^{-m}\eu^{\iu m\theta}$ और $ g' = 1 -
\sum_nnb_n\rho^{-n-1}\eu^{-\iu(n+1)\theta}$: के बाद
से गुणा करना $\eu^{\iu\theta}$, केवल आवृत्ति-शून्य
उत्पाद जीवित रहते हैं $\theta $-एकीकरण (\omterm{def:b3:groups:normal}{सामान्य}
अभिसरण शब्द-दर-अवधि कार्य को उचित ठहराता है): जोड़ी
$(\rho\eu^{-\iu\theta})\cdot1$ योगदान $2\pi\rho $, और
प्रत्येक जोड़ी $\bar b_n\rho^{-n}\eu^{\iu
n\theta}\cdot(-nb_n\rho^{-n-1}\eu^{-\iu(n+1)\theta})$
योगदान $-2\pi n\abs{b_n}^2\rho^{-2n-1}$:
\[
A_\rho = \pi\rho^2 - \pi\sum_{n\geq1}n\abs{b_n}^2\rho^{-2n}.
\]

\textbf{2.} क्षेत्र गैर-नकारात्मक हैं: $\sum_{n\leq
N}n\abs{b_n}^2\rho^{-2n} \leq \rho^2$ हरएक के लिए $ N $; होने देना
$\rho \to 1^+$ तब $ N \to \infty $:
$\sum_{n\geq1}n\abs{b_n}^2 \leq 1$. में समानता $\abs{b_1}
\leq 1$ बाकी सभी को मजबूर करता है $ b_n = 0$: $ g(w) = w + b_0 +
\eu^{\iu\alpha}/w $, जो a के पूरक पर मैप करता है
लंबाई का खंड $4$ (जौकोव्स्की-प्रकार का नक्शा): चरमपंथी।

\textbf{3.} $ f(z)/z = 1 + a_2z + \cdots $ है \omterm{def:b3:holomorphic:holo}{होलोमार्फिक} और
शून्\omterm{def:b3:modules:free}{य-मुक्त} पर $\mathbb D $ ($ f $ पर ही लुप्त हो जाता है $0$, बस:
इंजेक्टिविटी), इसलिए ऐसा है $ f(z^2)/z^2$, जिसमें एक है
\omterm{def:b3:holomorphic:holo}{होलोमार्फिक} वर्गमूल $\varphi $ साथ $\varphi(0) = 1$
(\cref{def:b3:conformal:simplyconnected}; $\mathbb D $ है
उत्तल)। तब $ h(z) = z\varphi(z^2)$ संतुष्ट $ h^2 =
f(z^2)$, $ h(z) = z\bigl(1 + \frac{a_2}2z^2 + \cdots\bigr)$
(मूल के लिए द्विपद श्रृंखला), और $ h(z) = z\varphi(z^2)$ है
निर्माण से विषम ($\varphi(z^2)$ सम है) इंजेक्शन: $ h(z) = h(z')$ देता है $ f(z^2) =
f(z'^2)$, इसलिए $ z^2 = z'^2$, यानी\ $ z' = \pm z $; अगर $ z' = -z $
फिर विषमता देती है $ h(z) = -h(z)$, इसलिए $ h(z) = 0$, मजबूर करना $ z
= 0$ ($\varphi $ शून्\omterm{def:b3:modules:free}{य-मुक्त}): $ z = z' = 0$.

\textbf{4.} $ g(w) = 1/h(1/w)$: के लिए $\abs w > 1$, $1/w \in
\mathbb D\setminus\{0\}$ और $ h \neq 0$ वहाँ: अच्छी तरह से परिभाषित,
इंजेक्शन (इंजेक्शन की संरचना), विस्तार के साथ
\[
g(w) = \frac{1}{\frac1w\bigl(1 + \frac{a_2}{2w^2} +
\cdots\bigr)} = w\Bigl(1 - \frac{a_2}{2w^2} + \cdots\Bigr)
= w - \frac{a_2}{2}\,w^{-1} + \cdots :
\]
भाग I के साथ प्रपत्र $ b_1 = -\frac{a_2}2$. क्षेत्र प्रमेय
देता है $\abs{\frac{a_2}2} \leq 1$: $\abs{a_2} \leq 2$.

\textbf{5.} $ k(z) = \frac z{(1-z)^2} = z\sum_{m\geq0}(m +
1)z^m = \sum_{n\geq1}nz^n $: गुणांक $ a_n = n $, इसलिए $ a_2 =
2$. एकरूपता और छवि: $ k = \frac14\bigl[w^2 - 1\bigr]$
साथ $ w = \frac{1 + z}{1 - z}$, ए \omterm{def:b3:conformal:conformal}{अनुरूप मानचित्र} का $\mathbb D $
दाहिने आधे तल पर; $ w^2$ उस आधे-तल को मानचित्रित करता है
अनुरूप रूप से पर $\C\setminus\intoc{-\infty}0$; तब
$\frac{\cdot - 1}4$ देता है
$\C\setminus\intoc{-\infty}{-\frac14}$: प्रत्येक पर इंजेक्शन
मंच, छवि जैसा दावा किया गया है।

\textbf{6.} $ F = \frac{cf}{c - f}$: तब से $ c \notin f(\mathbb
D)$, हर कभी लुप्त नहीं होता: $ F $ है \omterm{def:b3:holomorphic:holo}{होलोमार्फिक}, और
इंजेक्शन ($ w \mapsto \frac{cw}{c - w}$ एम है\"ओबियस, इंजेक्शन
\omterm{def:b3:topology:interior}{बंद} $ w = c $). विस्तार: साथ $ f = z + a_2z^2 + \cdots $,
\[
F = f\cdot\frac1{1 - f/c} = \bigl(z + a_2z^2\bigr)\Bigl(1 +
\frac zc\Bigr) + O(z^3)
= z + \Bigl(a_2 + \frac1c\Bigr)z^2 + O(z^3):
\]
$ F \in \mathcal S $ साथ $ A_2 = a_2 + \frac1c $.

\textbf{7.} बीबरबैक दो बार: $\abs{a_2} \leq 2$ और
$\abs{a_2 + \frac1c} \leq 2$, इसलिए $\abs{\frac1c} \leq 4$:
$\abs c \geq \frac14$. छोड़ा गया प्रत्येक मान बाहर है
$ D(0,\frac14)$, यानी\ $ f(\mathbb D) \supseteq D(0, \frac14)$.
तीव्र: कोएबे फ़ंक्शन छोड़ दिया जाता है $-\frac14$ (प्रश्न 5).

\textbf{8.} होने देना $ d = d(z_0, \partial\Omega)$ और $\psi =
\varphi^{-1} \colon \mathbb D \to \Omega $, $\psi(0) = z_0$,
$\psi'(0) = 1/\varphi'(z_0) > 0$. \omterm{def:b3:groups:normal}{सामान्य}ीकरण $\tilde
f(z) = \frac{\psi(z) - z_0}{\psi'(0)}$ में निहित है $\mathcal S $,
तो इसकी छवि शामिल है $ D(0, \frac14)$; पीछे हटना,
$\Omega = \psi(\mathbb D) \supseteq D\bigl(z_0,
\tfrac{\psi'(0)}4\bigr)$, इसलिए $ d \geq \frac{\psi'(0)}4 =
\frac1{4\varphi'(z_0)}$: वाम असमानता. अधिकार के लिए: $\varphi\circ(z_0 + d\,\cdot)$
एमएपीएस $\mathbb D $ में $\mathbb D $ साथ $0 \mapsto 0$; श्वार्ज़
इसके व्युत्पन्न को सीमाबद्ध करता है $0$: $ d\,\varphi'(z_0) \leq 1$, अर्थात\
$\frac1{\varphi'(z_0)} \geq d $ --- एक साथ
\[
d \leq \frac1{\varphi'(z_0)} \leq 4d .
\]
(बयान में प्रदर्शित बाईं असमानता है
उसी श्रृंखला को पुनः व्यवस्थित किया गया।)

\textbf{9.} कोएबे समारोह के लिए $ a_n = n $: समानता
पूरे अनुमान के दौरान; इसके घूर्णन
$\eu^{-\iu\theta}k(\eu^{\iu\theta}z) = \sum
n\eu^{\iu(n-1)\theta}z^n $ पास होना $\abs{a_n} = n $ बहुत। धक्का देना
प्रश्न 4: $ h = z + \frac{a_2}2z^3 + c_5z^5 + \cdots $ साथ
$ h^2 = f(z^2)$; की तुलना $ z^6$-गुणांक: $2c_5 +
\frac{a_2^2}4 = a_3$, इसलिए $ c_5 = \frac{a_3}2 -
\frac{a_2^2}8$. तब
\[
g(w) = \frac1{h(1/w)} = w\Bigl(1 - \frac{a_2}{2w^2} +
\Bigl(\frac{a_2^2}4 - c_5\Bigr)\frac1{w^4} + \cdots\Bigr)
= w - \frac{a_2}2w^{-1} + \Bigl(\frac{3a_2^2}8 -
\frac{a_3}2\Bigr)w^{-3} + \cdots,
\]
और क्षेत्र प्रमेय ($\sum n\abs{b_n}^2 \leq 1$) पैदावार
\[
\Bigl|\frac{a_2}2\Bigr|^2 + 3\,\Bigl|\frac{3a_2^2}8 -
\frac{a_3}2\Bigr|^2 \leq 1 .
\]
कोबे चेक ($ a_2 = 2$, $ a_3 = 3$): $\abs{\frac{a_2}2}^2 = 1$
और $\frac{3\cdot4}8 - \frac32 = 0$: कुल बिल्कुल $1$ ---
चरम, जैसा कि यह होना चाहिए।

\textbf{10.} $\varphi $ के साथ एक डिस्क ऑटोमोर्फिज्म है
$\varphi(0) = z_0$, इसलिए $ f\circ\varphi $ पर एकसमान है
$\mathbb D $ (इंजेक्शन की संरचना), और $ f'(z_0) \neq
0$ (\cref{def:b3:conformal:conformal}): $ F $ अच्छी तरह से परिभाषित है
और एकसमान. $ F(0) = 0$. भागफल नियम:
\[
\varphi'(z) = \frac{(1 + \bar z_0z) - (z + z_0)\bar z_0}
{(1 + \bar z_0z)^2}
= \frac{1 - \abs{z_0}^2}{(1 + \bar z_0z)^2},
\qquad \varphi'(0) = 1 - \abs{z_0}^2 ,
\]
इसलिए $(f\circ\varphi)'(0) = f'(z_0)(1 - \abs{z_0}^2)$ और
$ F'(0) = 1$: $ F \in \mathcal S $.

\textbf{11.} साथ $ G = f\circ\varphi $: $ G'' =
f''(\varphi)\,\varphi'^2 + f'(\varphi)\,\varphi''$, और
$\varphi''(z) = -2\bar z_0(1 - \abs{z_0}^2)(1 + \bar
z_0z)^{-3}$ देता है $\varphi''(0) = -2\bar z_0(1 -
\abs{z_0}^2)$. इस तरह
\[
A_2 = \frac{G''(0)}{2f'(z_0)(1 - \abs{z_0}^2)}
= \frac12\Bigl[\bigl(1 - \abs{z_0}^2\bigr)
\frac{f''(z_0)}{f'(z_0)} - 2\bar z_0\Bigr] .
\]
बीबरबैक के लिए $ F $ (प्रश्न 4): $\abs{A_2} \leq 2$, अर्थात\
$\bigl|(1 - \abs{z_0}^2)\frac{f''(z_0)}{f'(z_0)} - 2\bar
z_0\bigr| \leq 4$. से गुणा करें
$ z_0/(1 - \abs{z_0}^2)$, जिसका मापांक है
$ r/(1 - r^2)$ के लिए $ z_0 = r\eu^{\iu\theta}$, और उपयोग करें $ z_0\bar
z_0 = r^2$:
\[
\Bigl|\,z_0\frac{f''(z_0)}{f'(z_0)} -
\frac{2r^2}{1 - r^2}\Bigr| \leq \frac{4r}{1 - r^2} .
\]

\textbf{12.} दूरी के भीतर एक सम्मिश्र संख्या $\rho $ का
वास्तविक बिंदु $ c $ में वास्तविक भाग है $\intcc{c - \rho}{c +
\rho}$: इसे लागू करें $ c = \frac{2r^2}{1-r^2}$, $\rho =
\frac{4r}{1-r^2}$.

\textbf{13.} न मिटने के लिए $\mathcal C^1$ $ g $:
$\log\abs g = \frac12\log(g\bar g)$, इसलिए $\frac{\dd}{\dd
t}\log\abs g = \frac{g'\bar g + g\bar g'}{2\abs g^2} =
\operatorname{Re}\frac{g'}g $. साथ $ g(t) =
f'(t\eu^{\iu\theta})$ (शून्\omterm{def:b3:modules:free}{य-मुक्त}: एकरूपता), $ g'(t) =
\eu^{\iu\theta}f''(t\eu^{\iu\theta})$, तो पर $ z =
t\eu^{\iu\theta}$:
\[
\frac{\dd}{\dd t}\log\bigl|f'(t\eu^{\iu\theta})\bigr|
= \operatorname{Re}\Bigl(\eu^{\iu\theta}
\frac{f''}{f'}\Bigr)
= \frac1t\operatorname{Re}\Bigl(z\frac{f''}{f'}\Bigr)
\in \Bigl[\frac{2t - 4}{1 - t^2},\
\frac{2t + 4}{1 - t^2}\Bigr]
\]
त्रिज्या पर प्रश्न 12 द्वारा $ t $. तब से $\frac{\dd}{\dd
t}\log\frac{1+t}{(1-t)^3} = \frac1{1+t} + \frac3{1-t} =
\frac{2t+4}{1-t^2}$ और $\frac{\dd}{\dd
t}\log\frac{1-t}{(1+t)^3} = \frac{-1}{1-t} - \frac3{1+t} =
\frac{2t-4}{1-t^2}$, से एकीकृत करना $0$ को $ r $ (तीनों
फ़ंक्शंस गायब हो जाते हैं $ t = 0$, $ f'(0) = 1$) देता है
\[
\log\frac{1-r}{(1+r)^3} \leq \log\abs{f'(z)} \leq
\log\frac{1+r}{(1-r)^3} :
\]
विरूपण प्रमेय, घातांकीकरण के बाद।

\textbf{14.} ऊपरी: खंड के साथ $[0, z]$,
\[
\abs{f(z)} = \Bigl|\int_0^rf'(t\eu^{\iu\theta})
\eu^{\iu\theta}\dd t\Bigr|
\leq \int_0^r\frac{1+t}{(1-t)^3}\dd t
= \frac{r}{(1-r)^2}
\]
($\frac{\dd}{\dd t}\frac t{(1-t)^2} = \frac{1+t}{(1-t)^3}$).
निचला: नोट $\frac r{(1+r)^2} < \frac14$ के लिए $ r < 1$ (इसे कहते हैं $(1-r)^2 > 0$), तो यदि $\abs{f(z)} \geq \frac14$ वहाँ
साबित करने के लिए कुछ भी नहीं है. अन्यथा खंड $[0, f(z)]$ में निहित है $ D(0, \frac14) \subseteq f(\mathbb D)$ (प्रश्न 7), और
$\gamma = f^{-1}\circ[0, f(z)]$ एक है $\mathcal C^1$ पथ
से $0$ को $ z $ में $\mathbb D $ ($ f^{-1}$ \omterm{def:b3:holomorphic:holo}{होलोमार्फिक},
\cref{cor:b3:residues:openmapping}). स्थानापन्न $ w =
f(\zeta)$,
\[
\abs{f(z)} = \int_{[0,f(z)]}\abs{\dd w}
= \int_\gamma\abs{f'(\zeta)}\,\abs{\dd\zeta}
\geq \int_0^1\frac{1 - \rho(s)}{(1 + \rho(s))^3}
\,\abs{\gamma'(s)}\,\dd s
\]
साथ $\rho = \abs\gamma $, निचली विरूपण सीमा का उपयोग करते हुए
त्रिज्या पर $\rho(s)$. तब से $\abs{\gamma'} \geq \rho'$
(जहाँ भी परिभाषित हो; $\rho $ लिप्सचिट्ज़ है) और इंटीग्रैंड
कारक सकारात्मक है,
\[
\abs{f(z)} \geq \int_0^1
\frac{1 - \rho(s)}{(1 + \rho(s))^3}\,\rho'(s)\,\dd s
= \int_0^{r}\frac{1 - u}{(1 + u)^3}\,\dd u
= \frac{r}{(1 + r)^2}
\]
($\rho(0) = 0$, $\rho(1) = r $; प्रतिस्थापन केवल एक का उपयोग करता है
प्रतिव्युत्पन्न, $\frac{\dd}{\dd u}\frac u{(1+u)^2} =
\frac{1-u}{(1+u)^3}$, एकरसता नहीं)।

\textbf{15.} $ k'(z) = \frac{\dd}{\dd z}\,z(1 - z)^{-2} =
\frac{1 + z}{(1 - z)^3}$. पर $ z = r $: $ k'(r) =
\frac{1+r}{(1-r)^3}$ और $ k(r) = \frac r{(1-r)^2}$ --- दोनों
ऊपरी सीमा प्राप्त हुई. पर $ z = -r $: $ k'(-r) =
\frac{1-r}{(1+r)^3}$ और $\abs{k(-r)} = \frac r{(1+r)^2}$
--- दोनों निचली सीमाएँ प्राप्त हो गईं। कोएबे फ़ंक्शन फैलता है
इसकी धनात्मक धुरी अधिकतम सुदूर सीमा की ओर है
एंटीपोडल किरण को उसके सिरे की ओर अधिकतम रूप से संपीड़ित करता है
भट्ठा

\textbf{16.} $\abs{a_2} = 2$ मतलब $\abs{b_1} = 1$ के लिए $ g(w)
= 1/h(1/w) = w - \frac{a_2}2w^{-1} + \cdots $ (प्रश्न 4);
क्षेत्र प्रमेय $\sum n\abs{b_n}^2 \leq 1$ फिर हर एक को मार देता है
अन्य गुणांक: $ g(w) = w + b_0 + \eu^{\iu\alpha}/w $ साथ
$\eu^{\iu\alpha} = -\frac{a_2}2$. तब से $ h $ अजीब है, $ g(-w) =
1/h(-1/w) = -1/h(1/w) = -g(w)$: $ g $ अजीब है, इसलिए $ b_0 = 0$.
उलटा, $ h(z) = 1/g(1/z) = \frac{z}{1 +
\eu^{\iu\alpha}z^2}$, और $ f(z^2) = h(z)^2$ देता है
\[
f(w) = \frac{w}{(1 + \eu^{\iu\alpha}w)^2}
= \frac{w}{(1 - \eu^{\iu\theta}w)^2}
= \eu^{-\iu\theta}k\bigl(\eu^{\iu\theta}w\bigr),
\qquad \eu^{\iu\theta} = -\eu^{\iu\alpha} .
\]
इसके विपरीत प्रत्येक घूर्णन में होता है $ a_2 = 2\eu^{\iu\theta}$ का
मापांक $2$: बीबरबैक के चरम बिल्कुल वही हैं
कोएबे फ़ंक्शन घुमाए गए।

\textbf{17.} अगर $ c $ के साथ छोड़ दिया गया है $\abs c = \frac14$:
प्रश्न 6 देता है $ F \in \mathcal S $ साथ $ A_2 = a_2 +
\frac1c $, इसलिए $\frac1c = A_2 - a_2$ और
\[
4 = \Bigl|\frac1c\Bigr| = \abs{A_2 - a_2}
\leq \abs{A_2} + \abs{a_2} \leq 2 + 2 = 4 :
\]
विशेषकर सर्वत्र समानता $\abs{a_2} = 2$. द्वारा
प्रश्न 16, $ f = \eu^{-\iu\theta}k(\eu^{\iu\theta}\cdot)$,
जिसका छोड़ा गया सेट है $\eu^{-\iu\theta}
\intoc{-\infty}{-\frac14}$: का अद्वितीय छोड़ा गया मान
मापांक $\frac14$ है $ c = -\eu^{-\iu\theta}/4$. (जाँच करना:
$ a_2 = 2\eu^{\iu\theta}$ और $\frac1c = -4\eu^{\iu\theta}$,
इसलिए $ A_2 = -2\eu^{\iu\theta} = -a_2$: त्रिकोण
असमानता संरेखण-विरोधी से संतृप्त है, जैसा कि होना ही चाहिए।)

\textbf{18.}  \omterm{thm:b3:holomorphic:analytic}{कॉची अनुमान} वृत्त पर $\abs z =
r $ (\cref{thm:b3:holomorphic:analytic}), के साथ संयुक्त
विकास प्रमेय, दे
\[
\abs{a_n} \leq r^{-n}\sup_{\abs z = r}\abs f
\leq r^{1-n}\,(1 - r)^{-2} .
\] चुनना $ r = 1 - \frac1n $ ($ n \geq
2$): $ r^{1-n} = \bigl(1 + \frac1{n-1}\bigr)^{n-1} < \eu $
(सीमा के साथ बढ़ता क्रम $\eu $) और $(1 - r)^{-2} =
n^2$: $\abs{a_n} < \eu\,n^2$.

\textbf{19.} $ U = f(D(0, r))$ खुला है
(\cref{cor:b3:residues:openmapping}) और शामिल है $0$.
सीमा: यदि $ p \in \partial U $, लिखना $ p = \lim f(z_k)$
साथ $ z_k \in D(0, r)$; एक अनुवर्ती देता है $ z_k \to z_\infty
\in \bar D(0, r)$, और $ f(z_\infty) = p \notin U $ बल
$ z_\infty \in \partial D(0, r)$: $\partial U \subseteq
f(\partial D(0, r))$, इसलिए प्रत्येक सीमा बिंदु $ U $ है
मापांक $\geq \frac r{(1+r)^2}$ (प्रश्न 14). अब चलो
$\abs w < \frac r{(1+r)^2}$ और मान लीजिए $ w \notin U $. द
खंड $[0, w]$ है \omterm{def:b3:topology:connected}{जुड़े हुए}, मिलता है $ U $ (पर $0$) और इसके
पूरक (पर $ w $), तो यह मिलता है $\partial U $; लेकिन यह सब है
बिंदुओं का मापांक होता है $\leq \abs w < \frac r{(1+r)^2}$:
विरोधाभास। इस तरह $ D\bigl(0, \frac r{(1+r)^2}\bigr)
\subseteq U $. कुशाग्रता: $ k(-r) = -\frac r{(1+r)^2}$ है
के एक सीमा बिंदु की छवि $ D(0,r)$, और $ k $ इंजेक्शन है,
तो $ k(-r) \notin k(D(0, r))$: त्रिज्या नहीं हो सकती
बढ़ गया. जैसा $ r \to 1^-$, $\frac r{(1+r)^2} \to \frac14$:
पूर्ण डिस्क के लिए तिमाही प्रमेय।

\textbf{20.} पहला, $\partial f(\mathbb D) \neq \emptyset $:
अन्यथा $ f(\mathbb D)$, खुला, \omterm{def:b3:topology:interior}{बंद}, गैररिक्त, होगा
सभी $\C $, और $ f^{-1} \colon \C \to \mathbb D $ एक होगा
गैर-स्थिर संपूर्ण फ़ंक्शन से घिरा हुआ, विरुद्ध
\cref{cor:b3:holomorphic:liouville}. वाम असमानता: द
कोबे परिवर्तन $ F $ प्रश्न 10 में निहित है $\mathcal S $ और
$ F(\mathbb D) = \frac{f(\mathbb D) -
f(z_0)}{f'(z_0)(1-\abs{z_0}^2)}$; प्रश्न 7 द्वारा इसमें शामिल है
$ D(0, \frac14)$, इसलिए
\[
f(\mathbb D) \supseteq f(z_0) + D\Bigl(0,\
\tfrac14\abs{f'(z_0)}\bigl(1 - \abs{z_0}^2\bigr)\Bigr) ,
\]
और हर बिंदु $\partial f(\mathbb D)$ ---से विच्छेद
द \omterm{def:b3:topology:topology}{खुला सेट} $ f(\mathbb D)$ --- दूरी पर है $\geq
\frac14(1-\abs{z_0}^2)\abs{f'(z_0)}$ से $ f(z_0)$. ठीक है
असमानता: चलो $ d = d(f(z_0), \partial f(\mathbb D)) <
\infty $. डिस्क $ D(f(z_0), d)$ में निहित है $ f(\mathbb D)$: ए
से खंड $ f(z_0)$ इसके किसी भी बिंदु से दूरी पर रहता है
$< d $ से $ f(z_0)$, तो कभी नहीं मिलते $\partial f(\mathbb D)$,
और यह संयुक्तता प्रश्न 19 का तर्क इसे बरकरार रखता है
$ f(\mathbb D)$. तब $\hat g(w) = f^{-1}(f(z_0) + dw)$ एमएपीएस
$\mathbb D $ में $\mathbb D $ साथ $\hat g(0) = z_0$, और
$\chi = \psi^{-1}\circ\hat g $, साथ $\psi(z) = \frac{z +
z_0}{1 + \bar z_0z}$, ठीक करता है $0$: श्वार्ज
(\cref{thm:b3:conformal:schwarz}) देता है $\abs{\chi'(0)}
\leq 1$. तब से $\chi'(0) = \frac{\hat g'(0)}{\psi'(0)} =
\frac{d}{f'(z_0)\,(1 - \abs{z_0}^2)}$, यह है $ d \leq (1 -
\abs{z_0}^2)\abs{f'(z_0)}$.

\textbf{21.} $ k(\mathbb D) = \C\setminus
\intoc{-\infty}{-\frac14}$, इसलिए $\partial k(\mathbb D) =
\intoc{-\infty}{-\frac14}$, और सकारात्मक वास्तविक बिंदु के लिए
$ k(r) = \frac r{(1-r)^2}$ निकटतम सीमा बिंदु है
$-\frac14$:
\[
d = k(r) + \frac14 = \frac{4r + (1-r)^2}{4(1-r)^2}
= \frac{(1+r)^2}{4(1-r)^2} .
\]
प्रश्न 20 के बाएँ सदस्य: $\frac14(1 - r^2)k'(r) =
\frac14\,\frac{(1-r^2)(1+r)}{(1-r)^3} =
\frac{(1+r)^2}{4(1-r)^2} = d $: समानता. (सही सदस्य
बराबर $4d $: पूर्ण कारक $4$ दोनों पक्षों को अलग करता है,
और कोएबे फ़ंक्शन बिल्कुल नीचे बैठता है।)

\textbf{22.} एकल सिद्धांत है \omterm{thm:b3:hilbert:parseval}{पारसेवल}: एकरूपता
ओवरलैप होने से रोकता है, इसलिए छवि के पूरक का क्षेत्र
का $\{\abs w > \rho\}$, फूरियर मोड में विस्तारित
वृत्त, गैर-नकारात्मक है --- क्षेत्र प्रमेय एक है $ L^2$
चिन्ह से पहचान. बाकी सब तो वह असमानता है
परिवहन: एक वर्गमूल (प्रश्न 3) इसे में बदल देता है
$\abs{a_2} \leq 2$; पूर्वाह्न\"ओबियस प्रतिबिंब एक छोड़े गए से
मूल्य (प्रश्न 6) बदल जाता है $\abs{a_2} \leq 2$ तिमाही में
प्रमेय; डिस्क ऑटोमोर्फिज्म (प्रश्न 10) फैल गया $\abs{a_2}
\leq 2$ विरूपण प्रमेय के रूप में संपूर्ण डिस्क पर;
रेडियल एकीकरण विकृति को विकास में परिवर्तित करता है, और
आवरण में वृद्धि. हर स्तर पर समानता का मामला
भी फैलता है, हमेशा घूमते हुए कोएबे पर उतरता है
कार्य---पूरे सिद्धांत के लिए एक चरम परिवार, बस
क्योंकि घुमाव श्वार्ज़ के अद्वितीय चरम हैं
लेम्मा. एक आंतरिक भाग असमानता, साथ ही इसकी कठोरता
समानता का मामला, संपूर्ण ज्यामिति को नियंत्रित करता है: अनुरूप
मानचित्रण ऐसी असमानताओं का फायदा उठाने की कला है।

\textbf{23.} विकास प्रमेय सीमा $\abs f \leq
\frac{r}{(1-r)^2}$ प्रत्येक पर समान रूप से $\bar D(0, r)$, $ r <
1$, सभी के लिए $ f \in \mathcal S $ एक बार में: स्थानीय रूप से बाध्य, इसलिए
$\mathcal S $ एक \omterm{def:b3:groups:normal}{सामान्य} परिवार है
(\cref{thm:b3:conformal:montel}). अगर $ f_n \in \mathcal S \to
f $ स्थानीय रूप से समान रूप से: $ f $ है \omterm{def:b3:holomorphic:holo}{होलोमार्फिक} साथ $ f_n' \to f'$
स्थानीय रूप से समान रूप से (वीयरस्ट्रैस), इसलिए $ f(0) = 0$, $ f'(0) = 1$
--- विशेष रूप से $ f $ अचर है --- और हर्विट्ज़
(\cref{exo:b3:residues:8}) इंजेक्शन मानचित्रों की सीमा बनाता है
इंजेक्शन: $ f \in \mathcal S $. ए \omterm{def:b3:topology:continuity}{निरंतर} कार्यात्मक (जैसे
जैसे $ f \mapsto \abs{a_2} = \frac{\abs{f''(0)}}2$) एक पर
कॉम्पैक्ट \omterm{def:b3:groups:action}{कक्षा} \emph{उपलब्ध हो जाता है} इसका सर्वोच्च: चरम
फ़ंक्शंस मौजूद हैं इससे पहले कि कोई यह जान सके कि वे क्या हैं ---
गुणांक पर प्रत्येक परिवर्तनशील हमले का प्रारंभिक बिंदु
समस्याएँ, बीबरबैक की शामिल हैं।

\textbf{24.} लिखना $ g(w) = w + A_2w^2 + O(w^3)$ और लिखें:
\[
z = g(f(z)) = f(z) + A_2f(z)^2 + O(z^3)
= z + (a_2 + A_2)z^2 + O(z^3),
\]
इसलिए $ A_2 = -a_2$ और $\abs{A_2} \leq 2$ (बीबरबैक), साथ में
समानता यदि $\abs{a_2} = 2$, यानी\ आईएफएफ $ f $ एक घुमाया हुआ है
कोएबे फ़ंक्शन (प्रश्न 16) --- और फिर $ g $ है
संगत व्युत्क्रम, स्लिट तल पर परिभाषित।

\textbf{25.} $ f(z_1) - f(z_2) = (z_1 - z_2)\bigl(1 + a(z_1 +
z_2)\bigr)$. अगर $\abs a \leq \frac12$: के लिए $ z_1 \neq z_2$ में
$\mathbb D $, $\abs{a(z_1 + z_2)} < 2\abs a \leq 1$ (कठोर:
$\abs{z_1 + z_2} < 2$), इसलिए दूसरा कारक गायब नहीं हो सकता:
इंजेक्शन, और $ f \in \mathcal S $ (\omterm{def:b3:groups:normal}{सामान्य}ीकरण हैं
में निर्मित)। अगर $\abs a > \frac12$: मुद्दा $ s =
-\frac1a $ है $\abs s < 2$, इसलिए $ z_{1,2} = \frac s2 \pm
\varepsilon $ रिहायश $\mathbb D $ छोटे के लिए $\varepsilon > 0$,
विशिष्ट हैं, और $ z_1 + z_2 = s $ कारक को मारता है: $ f(z_1)
= f(z_2)$, इंजेक्शन नहीं. इसलिए $\mathcal S $ रोकना $ z +
az^2$ बिल्कुल के लिए $\abs a \leq \frac12$. बीबरबैक बाध्य है
$\abs{a_2} \leq 2$ इस प्रकार बेतहाशा असंतृप्त है
\omterm{def:b3:galois:extension}{डिग्री} के बहुपद $2$ --- कोएबे फ़ंक्शन
गुणांक $ a_n = n $ एक अनंत शृंखला से आते हैं
छोड़ी गई किरण के साथ षडयंत्र, एक ऐसा व्यवहार जो बहुपद नहीं है
(जिसका संबंध है $\mathcal S$ केवल छोटे गुणांकों के साथ)
नकल कर सकते हैं.
\end{solution}
