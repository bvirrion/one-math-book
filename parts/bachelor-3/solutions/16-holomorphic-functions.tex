\section*{Chapter \ref{ch:b3:holomorphic} --- Holomorphic
Functions}

\begin{solution}{exo:b3:holomorphic:1}
(a) $\bar z$: $P = x$, $Q = -y$, so $P_x = 1 \neq -1 = Q_y$:
nowhere $\C$-differentiable. $\abs z^2$: $P = x^2 + y^2$, $Q =
0$: Cauchy--Riemann demands $2x = 0 = 2y$: differentiable at
$0$ only --- and holomorphic nowhere (no open set).
$\operatorname{Re}z$: $P_x = 1 \neq 0 = Q_y$: nowhere.

(b) $x^2 - y^2 = \operatorname{Re}(z^2)$: $f(z) = z^2$ works.
All solutions: if $\operatorname{Re}f = \operatorname{Re}g$
with $f, g$ holomorphic on the connected $\C$, then $h = f -
g$ has $\operatorname{Re}h = 0$; Cauchy--Riemann gives $h' =
P_x + \iu Q_x = 0 - \iu P_y = 0$: $h$ is an imaginary
constant. Answer: $f(z) = z^2 + \iu c$, $c \in \R$.
\end{solution}

\begin{solution}{exo:b3:holomorphic:2}
On the unit circle $\gamma(t) = \eu^{\iu t}$:
\[
\int_C z^n\,\dd z = \int_0^{2\pi}\eu^{\iu nt}\,\iu\eu^{\iu
t}\dd t = \iu\int_0^{2\pi}\eu^{\iu(n+1)t}\dd t
= \begin{cases} 2\iu\pi & n = -1,\\ 0 & n \neq -1.\end{cases}
\]
For $\bar z$: along $[0, 1+\iu]$, $\gamma(t) = t(1 + \iu)$:
$\int_0^1 t(1 - \iu)(1 + \iu)\dd t = \int_0^12t\,\dd t = 1$.
Along $0 \to 1 \to 1 + \iu$: $\int_0^1t\dd t +
\int_0^1(1 - \iu t)\,\iu\,\dd t = \frac12 + \iu + \frac12 = 1
+ \iu$. Different values between the same endpoints: by
\cref{prop:b3:holomorphic:primitive}, $\bar z$ has no
primitive on any open set containing both paths.
\end{solution}

\begin{solution}{exo:b3:holomorphic:3}
(a) The cut plane $\Omega = \C\setminus\intoc{-\infty}0$ is
star-shaped about $1$, and $\frac1z \in \mathcal H(\Omega)$:
\cref{thm:b3:holomorphic:cauchy} provides a primitive $L$
with $L(1) = 0$. Then $\bigl(z\eu^{-L(z)}\bigr)' =
\eu^{-L}(1 - z\cdot\frac1z) = 0$: $z = c\,\eu^{L(z)}$ with $c
= 1$ (at $z = 1$). Writing $L = u + \iu v$: $\abs z = \eu^u$
and $z = \abs z\eu^{\iu v}$ with $v$ continuous, $v(1) = 0$,
$v \in \intoo{-\pi}\pi$ ($v$ is a continuous argument of $z$
on the connected $\Omega$, so its image avoids the odd
multiples of $\pi$ --- no point of $\Omega$ lies on $\R_-$
--- and, containing $v(1) = 0$, stays in $\intoo{-\pi}\pi$:
$v$ is the principal argument): $L = \log$.

(b) If $g$ were a continuous logarithm on $\C^*$: $h(t) =
g(\eu^{\iu t})$ satisfies $\eu^{h(t)} = \eu^{\iu t}$, so $h(t)
- \iu t \in 2\iu\pi\Z$, and by continuity $h(t) = \iu t +
2\iu\pi k$ for a fixed integer $k$. Then $g(1) = h(0) =
2\iu\pi k$ and $g(1) = h(2\pi) = 2\iu\pi(k + 1)$:
contradiction.

(c) On $D(0,1)$: $\log(1 + z) =
\sum_{n\geq1}\frac{(-1)^{n+1}}{n}z^n$ --- both sides vanish at
$0$ and have derivative $\frac1{1+z} = \sum(-1)^nz^n$
(\cref{ex:b3:holomorphic:examples}); a primitive is unique up
to a constant on the connected disc.
\end{solution}

\begin{solution}{exo:b3:holomorphic:4}
(a) Expand at $0$ (radius $\infty$): by the Cauchy estimates
on $C_r$, $\abs{c_k} \leq C(1 + r)^n/r^k \to 0$ as $r \to
\infty$ for $k > n$: $f = \sum_{k\leq n}c_kz^k$.

(b) If $\operatorname{Re}f \leq M$: $g = \eu^f$ is entire with
$\abs g = \eu^{\operatorname{Re}f} \leq \eu^M$: constant by
Liouville. Then $g' = f'g = 0$ with $g$ nonvanishing: $f' =
0$, and $f$ is constant.

(c) If $f$ omits the half-plane $H$, an affine map $w \mapsto
\alpha w + \beta$ sends $\C\setminus H$ into
$\{\operatorname{Re} \leq M\}$; apply (b) to $\alpha f +
\beta$.
\end{solution}

\begin{solution}{exo:b3:holomorphic:5}
(a) $g(z) = f(z) - z^2$ vanishes at the points $\frac1n$,
which accumulate at $0 \in \Omega$: by the identity theorem
($\Omega$ connected), $g \equiv 0$: $f(z) = z^2$.

(b) Yes: $f(z) = z\cos(\pi/z)$ is holomorphic on $\C^*$
(composition) and $f(\frac1n) = \frac1n\cos(n\pi) =
\frac{(-1)^n}n$. No contradiction with (a): the accumulation
point $0$ of the interpolation nodes does \emph{not} belong to
$\C^*$, so the identity theorem is silent --- two distinct
functions ($z\cos(\pi/z)$ and, say, the one from another
interpolation) may share these values.

(c) $f \equiv 0$ everywhere, versus $g = 0$ on $D(0,1)$ and
$g = 1$ on $D(3,1)$: holomorphic on the disconnected union,
equal on $D(0,1)$, different. The identity theorem's
open-closed argument needs connectedness to propagate from one
component to the other --- and cannot.
\end{solution}

\begin{solution}{exo:b3:holomorphic:6}
(a) By the maximum principle applied on the bounded domain:
$\sup_{\mathbb D}\abs f = \sup_{\partial\mathbb D}\abs f = 1$.
Since $f$ has no zeros, $1/f$ is holomorphic on $\mathbb D$,
continuous on the closure, with boundary modulus $1$: likewise
$\abs{1/f} \leq 1$, i.e.\ $\abs f \geq 1$. So $\abs f \equiv
1$: the modulus attains an interior maximum, and
\cref{thm:b3:holomorphic:maximum}(2) forces $f$ constant.

(b) $f(z) = z$: boundary modulus $1$, zero at the origin,
nonconstant --- the zero is exactly what blocks the $1/f$
argument.
\end{solution}

\begin{solution}{exo:b3:holomorphic:7}
Continuity of $G$: dominated convergence with dominator $h$
(locally uniform bound). Holomorphy by Morera (established
inside \cref{thm:b3:holomorphic:weierstrassconv}): for a
closed triangle $T$ in a disc where $\abs g \leq h$,
\[
\int_{\partial T}G(z)\,\dd z =
\int_X\Bigl(\int_{\partial T}g(x, z)\,\dd z\Bigr)\dd\mu(x) =
0,
\]
the interchange by Fubini ($\int_X\int_{\partial T}\abs
g \leq \operatorname{length}(\partial T)\int h < \infty$) and
the inner vanishing by Goursat. For $\Gamma$: on the strip $a
\leq \operatorname{Re}z \leq b$ ($0 < a \leq b$),
$\abs{t^{z-1}\eu^{-t}} = t^{\operatorname{Re}z-1}\eu^{-t} \leq
(t^{a-1} + t^{b-1})\eu^{-t}$, integrable on
$\intoo0{+\infty}$: $\Gamma$ is holomorphic on
$\{\operatorname{Re} z > 0\}$ (the measure is only
$\sigma$-finite, but the argument needs only the integrable
dominator). By the identity theorem, the functional equation
$\Gamma(z + 1) = z\Gamma(z)$, proved on
$\intoo0{+\infty}$ (\cref{ex:b3:lebesgue:gamma}), holds on the
whole half-plane.
\end{solution}

\begin{solution}{exo:b3:holomorphic:8}
Write $P = c\prod_k(X - a_k)^{m_k}$
(\cref{pb:b3:holomorphic:1}). Let $P'(z) = 0$. If $P(z) = 0$,
then $z$ is one of the $a_k$: in the hull. Otherwise, the
logarithmic derivative gives
\[
0 = \frac{P'(z)}{P(z)} = \sum_k\frac{m_k}{z - a_k}
= \sum_k m_k\,\frac{\bar z - \bar a_k}{\abs{z - a_k}^2} ;
\]
conjugating, $\sum_kw_k(z - a_k) = 0$ with $w_k = m_k/\abs{z -
a_k}^2 > 0$: $z = \sum_k\frac{w_k}{\sum w}\,a_k$, a convex
combination of the roots. For $P = z^3 - 1$: roots the cube
roots of unity, $P' = 3z^2$ with double root $0$ --- the
centroid of the equilateral triangle.
\end{solution}

\begin{solution}{exo:b3:holomorphic:9}
The closed unit square $K = \{x + \iu y : 0 \leq x, y \leq
1\}$ is compact: $M = \sup_K\abs f < \infty$. Every $z \in \C$
differs from a point of $K$ by an element of $\Z + \iu\Z$
(subtract integer parts), and $f$ is invariant under those
translations (iterate the two relations): $\abs f \leq M$ on
$\C$. Liouville: $f$ is constant. Hence any nonconstant
doubly periodic meromorphic function --- the elliptic
functions of the classical theory --- must have poles.
\end{solution}

\begin{solution}{exo:b3:holomorphic:10}
(a) Take real parts in the mean value formula
(\cref{thm:b3:holomorphic:maximum}(1)). Smoothness: $f$ is
analytic, so $P, Q \in \mathcal C^\infty$; differentiating
Cauchy--Riemann: $P_{xx} = (Q_y)_x = (Q_x)_y = (-P_y)_y =
-P_{yy}$ (Schwarz symmetry of second derivatives): $\Delta P =
0$.

(b) If $\operatorname{Re}f$ attained an interior maximum on a
connected $\Omega$: $g = \eu^f$ has $\abs g =
\eu^{\operatorname{Re}f}$ attaining an interior maximum, so
$g$, hence $\operatorname{Re}f = \ln\abs g$, is constant
(\cref{thm:b3:holomorphic:maximum}(2)). On a bounded domain
with continuity up to the boundary,
$\sup_{\bar\Omega}\operatorname{Re}f =
\sup_{\partial\Omega}\operatorname{Re}f$.
\end{solution}

\begin{solution}{pb:b3:holomorphic:1}
\textbf{1.} For $\abs z \geq 1$:
\[
\abs{P(z)} \geq \abs z^n\Bigl(1 -
\frac{\abs{a_{n-1}}}{\abs z} - \dots -
\frac{\abs{a_0}}{\abs z^n}\Bigr)
\geq \abs z^n\Bigl(1 - \frac{A}{\abs z}\Bigr),
\qquad A = \sum_k\abs{a_k} :
\]
for $\abs z \geq R_0 = \max(1, 2A)$, $\abs{P(z)} \geq
\frac12\abs z^n \to \infty$.

\textbf{2.} Choose $R \geq R_0$ with $\frac12R^n \geq
\abs{P(0)}$. On the compact $\bar D(0, R)$ the continuous
$\abs P$ attains a minimum, at some $z_0$; outside, $\abs P
\geq \frac12R^n \geq \abs{P(0)} \geq \abs{P(z_0)}$: the
minimum is global.

\textbf{3.} $Q(h) = P(z_0 + h)/P(z_0)$ is a polynomial in $h$
with $Q(0) = 1$; it is nonconstant ($P$ is), so some
coefficient beyond the constant is nonzero: $Q(h) = 1 +
c_kh^k + h^{k+1}S(h)$ with $k \geq 1$ minimal, $c_k \neq 0$,
$S \in \C[X]$.

\textbf{4.} Roots: any $w = \rho\eu^{\iu\varphi} \neq 0$ has
the $k$-th root $\rho^{1/k}\eu^{\iu\varphi/k}$, where
$\rho^{1/k}$ exists by the intermediate value theorem applied
to $t \mapsto t^k$ on $\intco0\infty$ --- no circularity.
Pick $\omega$ with $\omega^k = -1/c_k$. Then
\[
Q(t\omega) = 1 - t^k + t^{k+1}\,\omega^{k+1}S(t\omega),
\qquad
\abs{Q(t\omega)} \leq 1 - t^k + C\,t^{k+1}
\quad (0 < t \leq 1),
\]
with $C = \abs\omega^{k+1}\sup_{\abs h \leq
\abs\omega}\abs{S(h)}$ (note $1 - t^k \geq 0$ on $\intcc01$).

\textbf{5.} For $0 < t < \min(1, 1/C)$: $\abs{Q(t\omega)}
\leq 1 - t^k(1 - Ct) < 1$, i.e.\ $\abs{P(z_0 + t\omega)} <
\abs{P(z_0)}$ --- contradicting global minimality. So
$P(z_0) = 0$: d'Alembert--Argand's proof is complete.

\textbf{6.} Liouville version: if $P$ never vanishes, $1/P$
is entire; by question 1, $\abs{1/P} \leq 2R_0^{-n}$ outside
$\bar D(0, R_0)$, and $1/P$ is continuous on that compact
disc, hence bounded there too: bounded entire, so constant
(\cref{cor:b3:holomorphic:liouville}), making $P$ constant:
absurd. Ingredients: d'Alembert uses compactness (existence
of the minimum) and the polar-form existence of $k$-th roots;
Liouville uses the entire Cauchy apparatus (Goursat ---
itself a nested-compacts argument --- and the Cauchy
estimates) plus the same coercivity. Compactness of closed
discs is the common, irreducible core.

\textbf{7.} If $\deg P \geq 1$, pick a root $\alpha$
(questions 5); divide: $P = (X - \alpha)Q + P(\alpha) = (X -
\alpha)Q$, with $\deg Q = n - 1$; induct. Grouping equal
factors: $P = c\prod_i(X - \alpha_i)^{m_i}$, $\sum m_i = n$.

\textbf{8.} For real $P$: $P(\bar\alpha) =
\overline{P(\alpha)} = 0$, and multiplicities agree
(conjugate the factorization): nonreal roots come in pairs,
contributing $(X - \alpha)(X - \bar\alpha) = X^2 -
2\operatorname{Re}(\alpha)X + \abs\alpha^2$, a real quadratic
with discriminant $< 0$. Hence the stated list of
irreducibles, and a real polynomial of odd degree, having an
even number of nonreal roots, must have a real one. Direct
proof: $P(x) \to \pm\infty$ as $x \to \pm\infty$ (odd degree,
positive leading coefficient say), so $P$ changes sign, and
the intermediate value theorem applies. For $X^3 - X - 1$:
both arguments give the single real root $\approx 1.3247$
(and a conjugate pair).

\textbf{9.} (i) $\chi_u \in \C[X]$ is nonconstant: it has a
root $\lambda$, and $\det(u - \lambda\,\mathrm{id}) = 0$
gives an eigenvector; the elementary-divisor machinery of
\cref{thm:b3:modules:jordan} then applies to any complex
matrix, $\chi$ always splitting. (ii) Let $P \in \bar\Q[X]$
be nonconstant. As a polynomial over $\C$ it has a root $z
\in \C$; $z$ is algebraic over $\bar\Q$, hence over $\Q$ by
transitivity (\cref{cor:b3:galois:algclosed}), so $z \in
\bar\Q$: every nonconstant polynomial over $\bar\Q$ has a
root in $\bar\Q$.

\textbf{10.} Over $\Q(\iu)$, question 4 already fails:
$k$-th roots need not exist (no $\sqrt2$), and even granting
roots, question 2 fails --- a minimizing sequence need not
converge, $\Q$ lacking completeness; in Liouville's route,
Goursat's nested compact triangles have empty intersection
over $\Q(\iu)$-points. Both proofs consume the completeness
(equivalently, via bounded monotone convergence, the
order-completeness) of $\R$; connectedness powers the
intermediate value theorem behind polar form. The statement
``$\C$ is algebraically closed'' is algebra; every known
proof of it is analysis smuggled through the definition of
$\R$.
\end{solution}
