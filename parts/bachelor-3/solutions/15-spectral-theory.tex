\section*{Chapter \ref{ch:b3:spectral} --- Compact Operators
and the Spectral Theorem}

\begin{solution}{exo:b3:spectral:1}
(a) $T(B)$ is a bounded subset of the finite-dimensional
$\operatorname{im}T$: relatively compact by Heine--Borel
(\cref{cor:b3:topology:heineborel}, transported by a linear
homeomorphism with $\R^n$).
(b) $I$ compact means the closed unit ball is compact, which
by Riesz's theorem (Year 2) happens exactly in finite
dimension. If a compact $T$ had bounded inverse $T^{-1}$, then
$I = T^{-1}T$ would be compact
(\cref{prop:b3:spectral:ideal}): impossible in infinite
dimension.
\end{solution}

\begin{solution}{exo:b3:spectral:2}
(a) $\norm{Tx}^2 = \sum\abs{\lambda_n}^2\abs{x_n}^2 \leq
\sup\abs{\lambda_n}^2\norm x^2$, with near-equality on the
$e_n$ realizing the sup.
(b) ($\Leftarrow$) The truncations $T_N$ (keep $n \leq N$,
zero beyond) have finite rank and $\vertiii{T - T_N} =
\sup_{n>N}\abs{\lambda_n} \to 0$: compact by
\cref{prop:b3:spectral:ideal}. ($\Rightarrow$) If
$\abs{\lambda_{n_k}} \geq \delta > 0$ along a subsequence:
$\norm{Te_{n_k} - Te_{n_l}}^2 = \abs{\lambda_{n_k}}^2 +
\abs{\lambda_{n_l}}^2 \geq 2\delta^2$: no convergent
subsequence of $(Te_{n_k})$.
(c) $T^* = $ diagonal with $(\bar\lambda_n)$: self-adjoint iff
all $\lambda_n \in \R$. Then the standard basis $(e_n)$ is an
orthonormal basis of eigenvectors, eigenvalues $\lambda_n \to
0$: the spectral theorem verbatim.
\end{solution}

\begin{solution}{exo:b3:spectral:3}
Let $(x_n)$ be bounded. Then $(Bx_n)$ is bounded
($\vertiii B < \infty$); compactness of $S$ extracts
$SBx_{n_k} \to y$; continuity of $A$ gives $ASBx_{n_k} \to
Ay$: $ASB$ is compact. If $ST = TS = I$ with $T$ compact and
$\dim H = \infty$: $I = ST$ would be compact, contradicting
\cref{exo:b3:spectral:1}(b) --- which is the same statement
seen from the other side.
\end{solution}

\begin{solution}{exo:b3:spectral:4}
(a) By Cauchy--Schwarz in $y$: $\abs{T_kf(x)}^2 \leq
\bigl(\int\abs{k(x,y)}^2\dd y\bigr)\norm f_2^2$; integrate in
$x$ (Tonelli): $\norm{T_kf}_2 \leq \norm k_{L^2(\square)}
\norm f_2$.

(b) The family $e_{mn}(x,y) = e_m(x)\overline{e_n(y)}$ is
orthonormal in $L^2(\intcc01^2)$ (Tonelli separates the double
integral). Total: if $h \perp$ all $e_{mn}$, then for each
$m$, the function $y \mapsto \int h(x,y)\overline{e_m(x)}\dd
x$ (in $L^2$ by Cauchy--Schwarz and Tonelli) is orthogonal to
every $\overline{e_n}$ --- and the conjugates
$(\overline{e_n})$ form a Hilbert basis whenever $(e_n)$ does
(conjugation is an isometric bijection of $L^2$ preserving
orthogonality and totality) --- so it is $0$ a.e.; then for a.e.\ $y$, $h(\cdot, y) \perp$ every
$e_m$: $h(\cdot, y) = 0$ a.e.: $h = 0$ (Tonelli). So
$(e_{mn})$ is a Hilbert basis; expand $k = \sum c_{mn}e_{mn}$.
The truncation $k_N$ (indices $\leq N$) gives $T_{k_N}$ of
finite rank (range in $\operatorname{Vect}(e_1, \dots,
e_N)$), and by (a),
\[
\vertiii{T_k - T_{k_N}} \leq \norm{k - k_N}_{L^2} \to 0 :
\]
$T_k$ is a norm-limit of finite-rank operators: compact
(\cref{prop:b3:spectral:ideal}).
\end{solution}

\begin{solution}{exo:b3:spectral:5}
(a) $\lambda\norm x^2 = \langle x, Tx\rangle \geq 0$ on an
eigenvector. The formula is
\cref{prop:b3:spectral:sanorm} with all values $\langle x,
Tx\rangle \geq 0$: the absolute value is redundant.
(b) $(x, y) \mapsto \langle x, Ty\rangle$ is a Hermitian
positive (possibly degenerate) sesquilinear form; the usual
Cauchy--Schwarz proof (expand $\langle x + t\eu^{\iu\theta}y,
T(x + t\eu^{\iu\theta}y)\rangle \geq 0$ and take the
discriminant) never uses definiteness.
\end{solution}

\begin{solution}{exo:b3:spectral:6}
(a) $\norm{Mf}_2 \leq \norm f_2$, and on $f_n = \sqrt n\,
\mathbf 1_{\intcc{1 - 1/n}1}$ (unit vectors), $\norm{Mf_n}_2
\geq 1 - \frac1n$: $\vertiii M = 1$; self-adjoint since the
multiplier is real. Eigenvalues: $xf(x) = \lambda f(x)$ a.e.\
forces $f = 0$ a.e.\ off the null set $\{x = \lambda\}$: $f =
0$ in $L^2$.
(b) With the same $f_n$: $\norm{Mf_n - f_n}_2 \leq \frac1n \to
0$, while $f_n \rightharpoonup 0$ (for fixed $g \in L^2$,
$\abs{\langle g, f_n\rangle} \leq \norm{g\,\mathbf
1_{\intcc{1-1/n}1}}_2 \to 0$ by DCT). If $Mf_{n_k} \to h$ in
norm, then $f_{n_k} \to h$, forcing $h = 0$ (weak limit) yet
$\norm h = 1$: no convergent subsequence.
(c) In \cref{lem:b3:spectral:existence}, precisely the
extraction ``$Tx_{n_k} \to y$'' uses compactness; for $M$ the
maximizing sequences concentrate near $x = 1$ and their
images converge weakly to $0$, never in norm: the eigenvector
at the top of the numerical range simply fails to exist.
\end{solution}

\begin{solution}{exo:b3:spectral:7}
(a) $V = T_k$ with $k(x, y) = \mathbf 1_{y < x} \in
L^2(\intcc01^2)$: compact (\cref{exo:b3:spectral:4}). Its
adjoint is the kernel operator with kernel $\overline{k(y,
x)} = \mathbf 1_{y > x}$: $V^*f(x) = \int_x^1f$; $V \neq V^*$
(test on $f = \mathbf 1$).
(b) If $Vf = \lambda f$, $\lambda \ne 0$: $Vf$ is continuous
on $\intcc01$ (dominated convergence in $\int_0^x f$), so $f
= \frac1\lambda Vf$ has a continuous representative; then
$Vf$ is $\mathcal C^1$ (fundamental theorem of calculus for
continuous integrands), so $f$ is $\mathcal C^1$, and
$\lambda f' = f$ with $f(0) = \frac1\lambda Vf(0) = 0$: $f =
C\eu^{x/\lambda}$ with $C = f(0) = 0$.
(c) $V$ is compact with no eigenvalue at all except possibly
$0$ ($Vf = 0$ forces $f = 0$ a.e.\ by differentiating the
integral --- so not even $0$): the spectral machinery
genuinely requires self-adjointness, not just compactness.
\end{solution}

\begin{solution}{exo:b3:spectral:8}
Write $x = \sum_ic_ie_i + z$, $z \in \ker T$, so $\langle x,
Tx\rangle = \sum_i\mu_i\abs{c_i}^2$. \emph{Lower bound}: on
the unit sphere of $V_k = \operatorname{Vect}(e_1, \dots,
e_k)$, $\langle x, Tx\rangle = \sum_{i\leq
k}\mu_i\abs{c_i}^2 \geq \mu_k$: the max over $V$ of the min
is $\geq \mu_k$. \emph{Upper bound}: let $\dim V = k$ and $W_k
= \overline{\operatorname{Vect}}(e_k, e_{k+1}, \dots) +
\ker T$, of codimension $k - 1$ (its orthogonal complement is
$V_{k-1}$); $V \cap W_k \neq \{0\}$ (a linear map $V \to
H/W_k \cong V_{k-1}$ of rank $\leq k - 1$ has nontrivial
kernel), and a unit $x \in V\cap W_k$ has $\langle x,
Tx\rangle = \sum_{i \geq k}\mu_i\abs{c_i}^2 \leq \mu_k$: the
min over $V$ is $\leq \mu_k$. Together: the first formula;
the second is proved symmetrically (test $W = W_k$; for
arbitrary $W$ of codimension $k-1$, $W \cap V_k \neq 0$ gives
a unit vector with $\langle x, Tx\rangle \geq \mu_k$).
Monotonicity: $\langle x, Tx\rangle \leq \langle x,
Sx\rangle$ pointwise transfers through $\max\min$.
\end{solution}

\begin{solution}{exo:b3:spectral:9}
Rewrite $f - \lambda Tf = g$. For $\lambda = 0$: $f = g$,
always uniquely solvable. For $\lambda \neq 0$: this is
$(T - \frac1\lambda)f = -\frac g\lambda$, and by the Fredholm
alternative (\cref{thm:b3:spectral:fredholm}) with the
eigenvalues $\lambda_n = \bigl((n + \frac12)\pi\bigr)^{-2}$ of
$T$ (\cref{ex:b3:spectral:minxy}): unique solvability for all
$g$ iff $\frac1\lambda \neq \lambda_n$ for every $n$, i.e.
\[
\lambda \neq \Bigl(n + \tfrac12\Bigr)^2\pi^2
\qquad (n = 0, 1, 2, \dots).
\]
At an exceptional $\lambda = (n+\frac12)^2\pi^2$: solutions
exist iff $g \perp \sin\bigl((n{+}\frac12)\pi x\bigr)$, and
are then unique up to adding multiples of that sine.
\end{solution}

\begin{solution}{exo:b3:spectral:10}
(a) $Sx = \lambda x$: comparing coordinates, $0 = \lambda
x_1$ and $x_n = \lambda x_{n+1}$; if $\lambda \ne 0$ then
$x_1 = 0$ and inductively $x = 0$; if $\lambda = 0$, $Sx = 0$
forces $x = 0$ ($S$ isometric). No eigenvalues. $S^*x =
\lambda x$ reads $x_{n+1} = \lambda x_n$: $x =
x_1(1, \lambda, \lambda^2, \dots)$, in $\ell^2$ exactly when
$\abs\lambda < 1$: a full open disc of eigenvalues.
(b) $\norm{Se_n - Se_m} = \norm{e_{n+1} - e_{m+1}} = \sqrt2$:
the image of the bounded $(e_n)$ has no Cauchy subsequence;
likewise $S^*e_{n+1} = e_n$. Neither is compact.
(c) For compact self-adjoint operators the eigenvalues
capture everything (\cref{thm:b3:spectral:spectral}); dropping
either hypothesis, eigenvalues may vanish entirely ($S$,
Volterra) or fill a disc ($S^*$): the robust object is the
spectrum $\{\lambda : T - \lambda I \text{ not invertible}\}$,
whose theory belongs to a later course.
\end{solution}

\begin{solution}{pb:b3:spectral:1}
\textbf{1.} Continuity: $\min$ and $\max$ are continuous;
symmetry: swapping $x, y$ swaps neither $\min(x,y)$ nor $1 -
\max(x,y)$. Bounds: $0 \leq g$, and $g(x,y) \leq
\max\cdot(1-\max)$-type bounds give $g \leq \frac14$ (for $u
= \max$: $\min \leq u$ so $g \leq u(1-u) \leq \frac14$). $g
\in L^2(\square)$: Hilbert--Schmidt, hence $G$ compact
(\cref{exo:b3:spectral:4}); the kernel is real symmetric: $G$
self-adjoint.

\textbf{2.} Splitting at $y = x$:
\[
u(x) = (1 - x)\int_0^x y\,f(y)\,\dd y +
x\int_x^1(1 - y)\,f(y)\,\dd y .
\]
For continuous $f$, differentiate (product and fundamental
theorem):
\[
u'(x) = -\int_0^xyf + \int_x^1(1-y)f
\qquad\text{(the boundary terms cancel)},
\]
and $u''(x) = -xf(x) - (1 - x)f(x) = -f(x)$; clearly $u(0) =
u(1) = 0$. Conversely if $u \in \mathcal C^2$ vanishes at both
ends, $w = u - G(-u'')$ satisfies $w'' = 0$, $w(0) = w(1) =
0$: $w$ is affine and vanishes twice, so $w = 0$.

\textbf{3.} Let $Gf = 0$, $f \in L^2$. For $\psi \in \mathcal
C_c^\infty(\intoo01)$: $\psi = G(-\psi'')$ by question 2, so
\[
\langle f, \psi\rangle = \langle f, G(-\psi'')\rangle
= \langle Gf, -\psi''\rangle = 0
\]
($G$ self-adjoint): by the fundamental lemma
(\cref{cor:b3:lp:fundlemma}), $f = 0$ a.e. Positivity: for
continuous $f$, with $u = Gf$,
\[
\langle f, Gf\rangle = \int_0^1 fu = \int_0^1(-u'')u
= \bigl[-u'u\bigr]_0^1 + \int_0^1(u')^2 = \int_0^1(u')^2
\geq 0 ;
\]
for $f \in L^2$, approximate in $L^2$ by continuous $f_n$:
both sides pass to the limit ($G$ bounded).

\textbf{4.} If $Gu = \lambda u$, $\lambda \neq 0$: $Gu$ is
continuous ($\abs{Gu(x) - Gu(x')} \leq \norm{g(x,\cdot) -
g(x',\cdot)}_2\norm u_2$, and the kernel is uniformly
continuous), so $u$ has a continuous representative; then
question 2's formulas show $Gu \in \mathcal C^2$, so $u =
\frac1\lambda Gu \in \mathcal C^2$ with $-\lambda u'' =
-(Gu)'' = u$ and $u(0) = u(1) = 0$.

\textbf{5.} $-\lambda u'' = u$, $u(0) = 0$: $u = A\sin(x/\sqrt
\lambda)$ (positive $\lambda$: by question 3, $\lambda =
\langle u, Gu\rangle/\norm u^2 > 0$ on eigenvectors). $u(1) =
0$ forces $\frac1{\sqrt\lambda} = n\pi$: $\lambda_n =
\frac1{n^2\pi^2}$, eigenfunctions $\sin(n\pi x)$, normalized
$e_n = \sqrt2\sin(n\pi x)$ ($\int_0^12\sin^2(n\pi x)\dd x =
1$). Orthogonality check: $2\sin(m\pi x)\sin(n\pi x) =
\cos((m-n)\pi x) - \cos((m+n)\pi x)$ integrates to $0$ for $m
\neq n$.

\textbf{6.} $\ker G = \{0\}$ (question 3), so
\cref{thm:b3:spectral:spectral}(1) gives $H =
\overline{\operatorname{Vect}}(e_n)$: the sines are a Hilbert
basis of $L^2(\intcc01)$. For $f(x) = x(1 - x)$:
\[
c_n = \sqrt2\int_0^1x(1-x)\sin(n\pi x)\,\dd x
= \sqrt2\;\frac{2\bigl(1 - (-1)^n\bigr)}{n^3\pi^3}
= \begin{cases}\dfrac{4\sqrt2}{n^3\pi^3} & n \text{ odd},\\
0 & n \text{ even},\end{cases}
\]
(two integrations by parts). Parseval: $\int_0^1x^2(1-x)^2\dd
x = \frac1{30} = \sum_{n \text{ odd}}\frac{32}{n^6\pi^6}$,
i.e.\ $\sum_{n\text{ odd}}n^{-6} = \frac{\pi^6}{960}$.

\textbf{7.} $\langle e_n, Ge_n\rangle = \lambda_n\norm{e_n}^2
= \lambda_n$. For fixed $x$, the coefficients of $g(x,
\cdot)$: $\langle e_n, g(x,\cdot)\rangle = (Ge_n)(x) =
\lambda_ne_n(x)$, so $g(x,\cdot) =
\sum_n\lambda_ne_n(x)\,e_n$ in $L^2$. The explicit series
$\sum_n\lambda_ne_n(x)e_n(y) = \sum_n\frac{2\sin(n\pi
x)\sin(n\pi y)}{n^2\pi^2}$ converges normally on the square
($\abs{\text{term}} \leq \frac2{n^2\pi^2}$): its sum is
continuous, and for each $x$ it has the same
$L^2(\dd y)$-coefficients as $g(x, \cdot)$: the two continuous
functions agree for every $(x, y)$. Setting $y = x$ and
integrating (normal convergence allows term-by-term
integration):
\[
\int_0^1g(x,x)\,\dd x =
\sum_n\lambda_n\int_0^1e_n(x)^2\dd x = \sum_n\lambda_n .
\]

\textbf{8.} $\int_0^1g(x,x)\dd x = \int_0^1x(1 - x)\dd x =
\frac16$, so $\sum_{n\geq1}\frac1{n^2\pi^2} = \frac16$:
\[
\zeta(2) = \sum_{n\geq1}\frac1{n^2} = \frac{\pi^2}6 .
\]

\textbf{9.} For $f = \mathbf 1$: $c_n =
\sqrt2\int_0^1\sin(n\pi x)\dd x = \sqrt2\,\frac{1 -
(-1)^n}{n\pi}$: $c_n = \frac{2\sqrt2}{n\pi}$ for odd $n$, $0$
for even. Parseval: $1 = \sum_{n\text{
odd}}\frac{8}{n^2\pi^2}$, so $\sum_{n\text{ odd}}n^{-2} =
\frac{\pi^2}8$, and $\zeta(2) = \frac{\pi^2}8\cdot\frac{1}{1
- \frac14} = \frac{\pi^2}6$ (even terms are $\frac14\zeta(2)$).
Comparison: Parseval for one $f$ sums
$\abs{\langle e_n, f\rangle}^2$; the trace formula integrates
the diagonal of the \emph{kernel}, which amounts to summing
Parseval over an entire orthonormal family at once ---
$\sum_n\langle e_n, Ge_n\rangle$ --- and is therefore blind to
any particular choice of test function.

\textbf{10.} $\abs{c_n\cos(n\pi t)} \leq \abs{c_n}$ with
$\sum\abs{c_n}^2 < \infty$: for each $t$ the series converges
in $L^2$ (orthonormal expansion,
\cref{thm:b3:hilbert:parseval}(3)); the tail bound
$\norm{u(t) - u_N(t)}_2^2 \leq \sum_{n>N}\abs{c_n}^2$ is
uniform in $t$, and each partial sum is continuous in $t$
(finitely many cosines): $t \mapsto u(t,\cdot)$ is continuous
into $L^2$. For $f = \sum_{n\leq N}c_ne_n$: each mode
$\cos(n\pi t)\sin(n\pi x)$ satisfies $\partial_t^2 =
-n^2\pi^2 = \partial_x^2$ applied to it, vanishes at $x = 0,
1$, has value $\sin(n\pi x)$ and time-derivative $0$ at $t =
0$: the finite sum solves everything. Musically: the string's
motion is a superposition of \emph{standing waves} $e_n$,
whose frequencies $n\pi$ are the fundamental and its
overtones; the spectral theorem says every initial shape
decomposes uniquely into these pure tones, the coefficients
$c_n$ being the timbre. Hearing a string is computing an
orthonormal expansion.
\end{solution}
