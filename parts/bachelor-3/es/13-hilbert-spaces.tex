\chapter{Espacios Hilbert}\label{ch:b3:hilbert}

Un \omterm{def:b3:hilbert:inner}{Espacio de Hilbert} es un espacio de Banach cuya norma proviene de un
\omterm{def:b3:hilbert:inner}{producto interno} --- y esa única estructura extra restaura, en
dimensión infinita, casi toda la geometría euclidiana:
existen proyecciones ortogonales, cada funcional \omterm{def:b3:topology:continuity}{continuo} es un
\omterm{def:b3:hilbert:inner}{producto interno} contra un vector fijo (Riesz) y ortonormal
las bases expanden cada vector en una serie convergente con
Contabilidad pitagórica (\omterm{thm:b3:hilbert:parseval}{Parseval}). El clímax del capítulo es un
deuda pagada: el sistema trigonométrico es una base ortonormal
de $L^2$, por lo que la identidad de \omterm{thm:b3:hilbert:parseval}{Parseval} es válida para \emph{cada}
Función \omterm{def:b3:lebesgue:l1}{cuadrado integrable} --- la declaración del Año 2 solo podría
probar para funciones $\mathcal C^1$ por partes. terminamos con
Lax--Milgram, el lema del caballo de batalla del enfoque variacional
a ecuaciones diferenciales.

En todo momento, $H$ es un espacio vectorial sobre $K = \R$ o $\C$.

\section{Productos internos; el teorema de la proyección}

\begin{definition}\label{def:b3:hilbert:inner}
Un \emph{producto interno}\index{producto interno} es una aplicación $\langle
\cdot,\cdot\rangle \colon H\times H \to K$, lineal en el
segunda variable, con $\langle y, x\rangle =
\overline{\langle x, y\rangle}$y$\langle x, x\rangle > 0$
para $x \neq 0$. Induce la norma $\norm x = \langle x,
x\rangle^{1/2}$, la \emph{Desigualdad de Cauchy-Schwarz}
$\abs{\langle x, y\rangle} \leq \norm x\norm y$ (Año 2
prueba --- el discriminante --- no cambia), y el
\emph{ley del paralelogramo}
\[
\norm{x + y}^2 + \norm{x - y}^2 = 2\norm x^2 + 2\norm y^2 .
\]
Un \emph{Espacio de Hilbert}\index{Espacio de Hilbert} es un
espacio de producto interno \omterm{def:b3:complete:complete}{completo} para esta norma. Ejemplos: $\ell^2$
(\cref{pb:b3:banach:1}) y, el fundamental, $L^2(\mu)$
con $\langle f, g\rangle = \int\bar fg\,\dd\mu$ ---
\omterm{def:b3:complete:complete}{completo} de Riesz--Fischer (\cref{thm:b3:lp:complete}); el
producto interno es finito por Cauchy--Schwarz ($=$ Hölder en
$p = q = 2$).
\end{definition}

\begin{theorem}[Proyección sobre un conjunto convexo cerrado]
\label{thm:b3:hilbert:projection}
Sea $C \neq \varnothing$ un subconjunto cerrado \emph{convexo} de
el \omterm{def:b3:hilbert:inner}{Espacio de Hilbert} $H$ y $x \in H$. Hay un $p_C(x)
\in C$ único con\index{teorema de proyección}
\[
\norm{x - p_C(x)} = d(x, C),
\]
caracterizado por: $\operatorname{Re}\langle x - p_C(x),\ c -
p_C(x)\rangle \leq 0$para todos los$c \in C$. La aplicación$p_C$ es
$1$-Lipschitz.
\end{theorem}

\begin{proof}
Deje $d = d(x, C)$ y $(c_n) \subseteq C$ con $\norm{x - c_n}
\to d$. Paralelogramo en$x - c_n$y$x - c_m$:
\[
\norm{c_n - c_m}^2 = 2\norm{x - c_n}^2 + 2\norm{x - c_m}^2 -
4\,\bigl\|x - \tfrac{c_n + c_m}2\bigr\|^2
\leq 2\norm{x{-}c_n}^2 + 2\norm{x{-}c_m}^2 - 4d^2
\]
(la convexidad pone el punto medio en $C$): el lado derecho tiende a
$0$, entonces $(c_n)$ es Cauchy y su límite $p \in C$ (cerrado)
alcanza $d$. Unicidad: dos minimizadores dan, por igual
identidad, $\norm{p - p'}^2 \leq 2d^2 + 2d^2 - 4d^2 = 0$.

Caracterización: para $c \in C$, $t \in \intoc01$, el vector
$p + t(c - p) \in C$, entonces
\[
d^2 \leq \norm{x - p - t(c-p)}^2
= d^2 - 2t\operatorname{Re}\langle x - p, c - p\rangle +
t^2\norm{c-p}^2 ;
\]
dividir por $t \to 0^+$: $\operatorname{Re}\langle x - p, c -
p\rangle \leq 0$. Por el contrario, esta desigualdad da$\norm{x -
c}^2 = \norm{x - p}^2 - 2\operatorname{Re}\langle x - p, c -
p\rangle + \norm{p - c}^2 \geq \norm{x-p}^2$. Lipschitz: para
$x, y$ con proyecciones $p, q$, agregue los dos variacionales
desigualdades (con $c = q$, resp.\ $c = p$):
$\operatorname{Re}\langle x - y - (p - q), p - q\rangle \geq
0$, entonces$\norm{p - q}^2 \leq \operatorname{Re}\langle x - y, p -
q\rangle \leq \norm{x - y}\norm{p - q}$.
\end{proof}

\begin{theorem}[Descomposición ortogonal]
\label{thm:b3:hilbert:decomposition}
Sea $F$ un \emph{subespacio cerrado} de $H$. Entonces $p_F$ es
lineal, $x - p_F(x) \perp F$ para todos $x$ y \index{ortogonal
complemento}
\[
H = F \oplus F^\perp,
\qquad F^\perp = \{y : \langle y, f\rangle = 0\ \forall f\in
F\},
\qquad (F^\perp)^\perp = F .
\]
Para un subespacio general, $(F^\perp)^\perp = \bar F$; en
particular $F$ es denso si $F^\perp = \{0\}$.
\end{theorem}

\begin{proof}
Para un subespacio, la caracterización variacional con $c =
p_F(x) \pm f$($f \in F$, ambos signos, y$\iu f$ en el
caso complejo) obliga a $\langle x - p_F(x), f\rangle = 0$: el
El residual es ortogonal a $F$. Descomposición $x = p_F(x) + (x
- p_F(x))$con$F \cap F^\perp = \{0\}$($\langle y, y
\rangle = 0$); la linealidad de$p_F$ se deriva de la unicidad de
tales descomposiciones (ambos lados lineales en ellas). $(F^\perp)
^\perp \supseteq F$siempre; por el contrario si$x \perp F^\perp$,
escriba $x = f + g$: $g = x - f \in F^\perp$ y $\langle g,
g\rangle = \langle x, g\rangle - \langle f, g\rangle = 0$:$x
= f \in F$. Para un subespacio general$F$:$F^\perp = \bar
F^{\,\perp}$ (\omterm{def:b3:topology:continuity}{continuidad} del \omterm{def:b3:hilbert:inner}{producto interno}), entonces
$(F^\perp)^\perp = \bar F$ por el caso cerrado; \omterm{ex:b3:lebesgue:gamma}{densidad} y si
$\bar F = H$ y si $F^\perp = 0$.
\end{proof}

\begin{theorem}[Representación Riesz]\label{thm:b3:hilbert:riesz}
Por cada \omterm{def:b3:topology:continuity}{continuo} funcional lineal $\varphi \in H'$ hay
es un $a \in H$ único con\index{Teorema de representación de Riesz}
\[
\varphi(x) = \langle a, x\rangle \quad (x \in H),
\qquad \norm\varphi_{H'} = \norm a .
\]
\end{theorem}

\begin{proof}
Si $\varphi = 0$: $a = 0$. De lo contrario $F = \ker\varphi$ es un
subespacio propio cerrado; elija $u \in F^\perp$, $\norm u = 1$
(\cref{thm:b3:hilbert:decomposition}: $F^\perp \neq 0$ desde
$F \neq H$). Para cualquier $x$, el vector $\varphi(x)u -
\varphi(u)x \in \ker\varphi$, por lo tanto$\perp u$:
\[
0 = \langle u, \varphi(x)u - \varphi(u)x\rangle
= \varphi(x) - \varphi(u)\langle u, x\rangle :
\qquad \varphi(x) = \langle
\overline{\varphi(u)}\,u,\ x\rangle .
\]
Entonces $a = \overline{\varphi(u)}u$ funciona. Unicidad: $\langle a
- a', x\rangle = 0$para todos los$x$, prueba$x = a - a'$. Normas:
$\abs{\varphi(x)} \leq \norm a\norm x$ (Cauchy--Schwarz) con
igualdad en $x = a$.
\end{proof}

\begin{example}[Una proyección, calculada hasta el final.]
\label{ex:b3:hilbert:projectionexample}
En $H = L^2(\intcc01)$, ¿cuál es la mejor aproximación de
$f(x) = x^2$ por una función afín? El subespacio $F =
\operatorname{Vect}(1, x)$ es cerrado (de dimensión finita),
y $p_F(f) = a + bx$ se caracteriza por la ortogonalidad de
el residual a $1$ y a $x$:
\[
\int_0^1(x^2 - a - bx)\,\dd x = 0,
\qquad
\int_0^1x\,(x^2 - a - bx)\,\dd x = 0,
\]
es decir,\ $\frac13 = a + \frac b2$ y $\frac14 = \frac a2 +
\frac b3$:$a = -\frac16$,$b = 1$. Entonces$p_F(x^2) = x -
\frac16$, y el error es
\[
d(f, F)^2 = \int_0^1\Bigl(x^2 - x + \frac16\Bigr)^2\dd x =
\frac1{180},
\qquad d(f, F) = \frac1{6\sqrt5} .
\]
Dos comentarios que vale la pena internalizar. Primero, el cálculo es
nada más que un sistema lineal $2\times2$ --- el \emph{normales
ecuaciones}; para la base monomial su matriz
$\bigl(\frac1{i+j+1}\bigr)$ es el notoriamente
matriz de Hilbert mal condicionada y ortogonalizando primero
(Polinomios de Legendre, \cref{pb:b3:hilbert:1}) es la cura.
En segundo lugar, la mejor aproximación \emph{uniforme} de $x^2$ por
funciones afines es diferente ($x - \frac18$, por
equioscilación): cada norma tiene su propia geometría, y sólo
el hilbertiano responde con un sistema lineal.
\end{example}

\section{Bases ortonormales}

\begin{definition}\label{def:b3:hilbert:onb}
Una familia $(e_i)_{i\in I}$ es \emph{ortonormal} si es $\langle
e_i, e_j\rangle = \delta_{ij}$, y una \emph{Hilbert
base}\index{base de hilabert} (base ortonormal) si además
sus combinaciones lineales finitas son densas en $H$ (la familia es
\emph{total}). Tratamos el caso contable $I = \N$, que por
Gram--Schmidt cubre todos los \emph{separable} $H$
(\cref{prop:b3:hilbert:gramschmidt}).
\end{definition}

\begin{theorem}[Bessel, Parseval]\label{thm:b3:hilbert:parseval}
Sea $(e_n)_{n\in\N}$ ortonormal en $H$ y $c_n(x) =
\langle e_n, x\rangle$.
\begin{enumerate}
  \item (Bessel) $\sum_n\abs{c_n(x)}^2 \leq \norm x^2$, y
        la serie $\sum_nc_n(x)e_n$ converge en $H$, con
        suma $p_F(x)$, $F = \overline{\operatorname{Vect}}(e_n)$.
  \item Los siguientes son equivalentes: (i) $(e_n)$ es un \omterm{def:b3:hilbert:onb}{Hilbert
        base}; (ii) $x = \sum_nc_n(x)e_n$ por cada $x$; (iii)
        \emph{Parseval}\index{La identidad de Parseval.}: $\norm x^2
        = \sum_n\abs{c_n(x)}^2$por cada$x$; (iv) el único
        El vector ortogonal a todos los $e_n$ es $0$.
  \item Si $(e_n)$ es \omterm{def:b3:hilbert:onb}{base de hilabert}, $x \mapsto (c_n(x))_n$
        es un isomorfismo isométrico $H \to \ell^2$
        (\emph{cada} \omterm{def:b3:hilbert:inner}{Hilbert
        espacio} separable de dimensión infinita ``es'' $\ell^2$) y $\langle x, y\rangle =
        \sum_n\overline{c_n(x)}c_n(y)$.
\end{enumerate}
\end{theorem}

\begin{proof}
(1) Para $N$ finito: $x - \sum_{n\leq N}c_ne_n \perp e_k$ ($k
\leq N$), entonces Pitágoras da$\norm x^2 = \sum_{n\leq
N}\abs{c_n}^2 + \norm{x - \sum_{n\leq N}c_ne_n}^2$: Bessel.
Las sumas parciales $S_N = \sum_{n\leq N}c_ne_n$ son Cauchy:
$\norm{S_N - S_M}^2 = \sum_{M<n\leq N}\abs{c_n}^2$, cola de a
series convergentes; el límite está en $F$ y $x - \lim S_N
\perp$cada$e_k$(\omterm{def:b3:topology:continuity}{continuidad}), por lo tanto$\perp F$: por unicidad
de la descomposición ortogonal, $\lim S_N = p_F(x)$.

(2) (i)$\Rightarrow$(ii): $F = H$, entonces $p_F = \mathrm{id}$.
(ii)$\Rightarrow$(iii): Pitágoras en el límite ($\norm{S_N}^2
= \sum_{n \leq N}\abs{c_n}^2 \to \norm x^2$).
(iii)$\Rightarrow$(iv): $x \perp$ todo $e_n$ da $\norm x^2 =
0$. (iv)$\Rightarrow$(i):$F^\perp = \{0\}$
(la ortogonalidad a todos los $e_n$ es la ortogonalidad a $F$), por lo que $F$
es denso por \cref{thm:b3:hilbert:decomposition}; pero $F$, un
\omterm{def:b3:topology:interior}{cierre}, ya está cerrado: $F = H$.

(3) La aplicación es lineal, isométrico por (iii) (por lo tanto inyectivo),
y sobreyectiva: dado $(c_n) \in \ell^2$, la serie $\sum
c_ne_n$ converge (Cauchy como en (1)) a una preimagen. La fórmula \omterm{def:b3:hilbert:inner}{interior
producto} es la polarización de (iii), o un límite directo
cálculo.
\end{proof}

\begin{proposition}[Abuela--Schmidt]\label{prop:b3:hilbert:gramschmidt}
Sea $(x_n)$ una secuencia linealmente independiente. Configuración
inductivamente $\tilde e_n = x_n - \sum_{k<n}\langle e_k,
x_n\rangle e_k$y$e_n = \tilde e_n/\norm{\tilde e_n}$
produce un $(e_n)$ ortonormal con los mismos tramos finitos:
$\operatorname{Vect}(e_1, \dots, e_n) = \operatorname{Vect}
(x_1, \dots, x_n)$. En consecuencia cada \omterm{def:b3:hilbert:inner}{Espacio de Hilbert} separable
(uno con un subconjunto denso contable) tiene un \omterm{def:b3:hilbert:onb}{Hilbert
base}.\index{Ortonormalización de Gram-Schmidt}
\end{proposition}

\begin{proof}
Inducción: $\tilde e_n \perp e_k$ ($k < n$) por construcción,
y $\tilde e_n \neq 0$ por independencia; los tramos coinciden en
cada etapa (cambio triangular de base). Para un $H$ separable:
de una secuencia densa extraer un linealmente independiente
subfamilia con tramo denso (deseche cada vector en el tramo de
sus predecesores - el lapso no cambia), ortonormalizar:
el resultado es total.
\end{proof}

\begin{theorem}[El sistema trigonométrico; Parseval por fin]
\label{thm:b3:hilbert:fourier}
En $L^2(\intcc{-\pi}\pi)$ con $\langle f, g\rangle =
\frac1{2\pi}\int_{-\pi}^\pi \bar fg$, la familia$e_n(t) =
\eu^{\iu nt}$,$n \in \Z$, es una \omterm{def:b3:hilbert:onb}{base de hilabert}. En consecuencia,
para \emph{cada} $f \in L^2$ --- en particular cada pieza
\omterm{def:b3:topology:continuity}{continuo} $2\pi$-periódico $f$ --- con $c_n(f) =
\frac1{2\pi}\int_{-\pi}^{\pi}f(t)\eu^{-\iu nt}\dd t$:
\[
f = \sum_{n\in\Z}c_n(f)\,\eu^{\iu nt} \ \ \text{in } L^2,
\qquad
\frac1{2\pi}\int_{-\pi}^{\pi}\abs f^2 =
\sum_{n\in\Z}\abs{c_n(f)}^2 .
\]
Esto prueba, con total generalidad, la identidad \omterm{thm:b3:hilbert:parseval}{Parseval} que
Admitido 2º año.\index{La identidad de Parseval (general)}
\end{theorem}

\begin{proof}
La ortonormalidad es un cálculo directo (Año 2). Totalidad: dejar
$f \in L^2$ sea $\perp$ todo $e_n$, es decir,\ todo Fourier
los coeficientes desaparecen. \omterm{def:b3:topology:continuity}{Continuo} $2\pi$-las funciones periódicas son
denso en $L^2(\intcc{-\pi}\pi)$: efectivamente $\mathcal
C_c(\intoo{-\pi}\pi)$ es denso
(\cref{thm:b3:lp:density}(2)) y dichas funciones se extienden
periódicamente y \omterm{def:b3:topology:continuity}{continuamente}. Los polinomios trigonométricos son
$\norm\cdot_\infty$-denso entre \omterm{def:b3:topology:continuity}{continuo} funciones periódicas
(Piedra--Weierstrass, \cref{cor:b3:complete:weierstrass}(c)),
y $\norm\cdot_2 \leq \norm\cdot_\infty$: trigonométrico
los polinomios son densos en $L^2$. Pero $f \perp$ cada
polinomio trigonométrico, por lo tanto $f \perp$ un subespacio denso: $f
\in (\text{dense})^\perp = \{0\}$
(\cref{thm:b3:hilbert:decomposition}). Criterio (iv) de
\cref{thm:b3:hilbert:parseval} concluye; (ii) y (iii)
desembalar en la pantalla (reindexando el contable $\Z$; el
la serie de dos extremos converge incondicionalmente --- la parcial
Las sumas de cualquier familia agotadora convergen, por la cola $\ell^2$.
argumento).
\end{proof}

\begin{theorem}[Lax--Milgram]\label{thm:b3:hilbert:laxmilgram}
Sea $H$ un bilineal real \omterm{def:b3:hilbert:inner}{Espacio de Hilbert} y $a \colon H\times H \to
\R$, \emph{\omterm{def:b3:topology:continuity}{continuo}} ($\abs{a(u,v)} \leq M\norm
u\norm v$) y \emph{coactivo} ($a(u, u) \geq \alpha\norm u^2$,
$\alpha > 0$). Entonces por cada $\varphi \in H'$ hay un
único $u \in H$ con\index{Teorema de Lax-Milgram}
\[
a(u, v) = \varphi(v) \qquad \text{for all } v \in H .
\]
\end{theorem}

\begin{proof}
Para $u$ fijo, $v \mapsto a(u, v)$ es \omterm{def:b3:topology:continuity}{continuo} lineal: Riesz
da un $Au \in H$ único con $a(u,v) = \langle Au,
v\rangle$;$A$es lineal con$\norm{Au} \leq M\norm u$
(singularidad de los representantes, luego vinculados). Coercitividad:
$\alpha\norm u^2 \leq a(u,u) = \langle Au, u\rangle \leq
\norm{Au}\norm u$, entonces$\norm{Au} \geq \alpha\norm u$:$A$ es
inyectivo con rango cerrado (una secuencia de imágenes de Cauchy $Au_n$
fuerzas $u_n$ Cauchy). El rango es denso: $w \perp
\operatorname{im}A$da$0 = \langle Aw, w\rangle \geq
\alpha\norm w^2$. Cerrado y denso:$A$ es biyectivo. dado
$\varphi$, dejemos que $f$ lo represente (Riesz) y $u = A^{-1}f$:
$a(u, v) = \langle f, v\rangle = \varphi(v)$, únicamente
($a(u - u', \cdot) = 0$ y coercitividad).
\end{proof}

\begin{remark}\label{rem:b3:hilbert:variational}
Cuando $a$ es simétrico, la solución de Lax-Milgram es la única
minimizador del \emph{energía} $J(v) = \frac12a(v,v) -
\varphi(v)$ (\cref{exo:b3:hilbert:9}): existencia de soluciones
a problemas variacionales de un solo golpe. Aplicado a adecuado
espacios funcionales (los espacios de Sobolev de un curso posterior), esto
resuelve problemas de valores límite para ecuaciones diferenciales ---
el punto de entrada moderno a las ecuaciones diferenciales parciales.
\end{remark}

(Adjuntos) Para $T \in \mathcal L(H)$, muestre que hay un único
$T^* \in \mathcal L(H)$ con $\langle Tx, y\rangle = \langle

Calcule el adjunto del turno $S$ en $\ell^2$ y demuestre

$\overline{\operatorname{im}T} = (\ker T^*)^\perp$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:hilbert:9}
Sea $a$ como en Lax--Milgram y además \emph{simétrico}.
Demuestre que $u$ resuelve $a(u, \cdot) = \varphi$ si $u$ minimiza
$J(v) = \frac12a(v, v) - \varphi(v)$, y que el mínimo es
alcanzado exactamente en un punto. \emph{(\omterm{def:b3:complete:complete}{Completo} the square:
$J(u + w) - J(u) = \frac12a(w,w) \geq

\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b3:hilbert:10}
(El sistema Haar) En $\intcc01$, deje que $h_{0} = \mathbf 1$ y
para $n = 2^j + k$ ($j \geq 0$, $0 \leq k < 2^j$):
\[
h_n = 2^{j/2}\Bigl(\mathbf 1_{[k2^{-j},\,(k +
\frac12)2^{-j})} - \mathbf 1_{[(k+\frac12)2^{-j},\,(k+1)2^{-j})}
\Bigr).
\]
Demuestre que $(h_n)_{n\geq0}$ es ortonormal en $L^2(\intcc01)$,
y total. \emph{(Orthogonality: disjoint or nested supports;
totality: finite spans contain all dyadic step functions,
which are dense --- via \cref{thm:b3:lp:density}(1) and dyadic
approximation of intervals.)} El sistema Haar es el ancestro
de ondas.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:hilbert:11}
(Proyecciones ortogonales, caracterizadas) Sea $H$ un \omterm{def:b3:hilbert:inner}{Hilbert
espacio} y $P \in \mathcal L(H)$ con $P^2 = P$, $P \neq 0$.
Muestre la equivalencia de:
(i) $P$ es la proyección ortogonal sobre
$\operatorname{im}P$;
(ii) $P = P^*$ (\cref{exo:b3:hilbert:8});
(iii) $\vertiii P = 1$.
\emph{(For (iii) $\Rightarrow$ (i): if some $x \in
(\ker P)^\perp$had$Px \neq x$, consider$x + t(Px - x)$
--- or directly: for $u \in \operatorname{im}P$ and $v \in
\ker P$, expand$\norm{P(u + tv)}^2 \leq \norm{u + tv}^2$
for all $t \in \R$ and conclude $\langle u, v\rangle = 0$.)}
Muestre una proyección no ortogonal en $\R^2$ y calcule
su norma.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b3:hilbert:12}
(Teorema ergódico de von Neumann) Sea $U \in \mathcal L(H)$
\emph{unitario} ($U^*U = UU^* = I$), $F = \ker(U - I)$ el
espacio fijo, $P$ la proyección ortogonal sobre $F$, y
$A_n = \frac1n\sum_{k=0}^{n-1}U^k$.
(a) Muestre $\ker(U - I) = \ker(U^* - I)$ \emph{(from
$\norm{Ux - x}^2 = 2\norm x^2 - 2\operatorname{Re}\langle
Ux, x\rangle$ and unitarity)} y deduzca
$\overline{\operatorname{im}(U - I)} = F^\perp$.
(b) Demuestre que $A_nx \to x$ para $x \in F$ y $A_nx \to 0$
para $x \in \operatorname{im}(U - I)$ \emph{(telescópico)},
luego para $x \in \overline{\operatorname{im}(U - I)}$
(uniforme cota $\vertiii{A_n} \leq 1$).
(c) Concluir: $A_nx \to Px$ para \emph{cada} $x \in H$ ---
los promedios de tiempo convergen a la proyección sobre las invariantes.
(d) Explíquelo para $H = L^2(\R/\Z)$ y $Uf = f(\cdot +
\alpha)$con$\alpha$irracional: identifique$F$ (use
Serie de Fourier, \cref{thm:b3:hilbert:fourier}) y deducir
que $\frac1n\sum_{k<n}f(x + k\alpha) \to \int_0^1f$ en
$L^2$: la equidistribución $L^2$ de rotaciones irracionales.
\end{exercise}

\section{Problema: polinomios ortogonales}

\begin{problem}[{Problema del fin de semana --- Legendre, Hermite y
Cuadratura de Gauss}]\label{pb:b3:hilbert:1}
Sea $I \subseteq \R$ un intervalo y $w > 0$ un \omterm{def:b3:topology:continuity}{continuo}
\emph{peso} sobre el \omterm{def:b3:topology:interior}{interior} de $I$ tal que $\int_I
\abs t^nw(t)\dd t < \infty$para todos los$n$; trabajar en$H = L^2(I,
w\,\dd\lambda)$con$\langle f, g\rangle = \int_I \bar
fg\,w$. Gram--Schmidt aplicado a$1, t, t^2, \dots$ produce
el \emph{\omterm{pb:b3:hilbert:1}{polinomios ortogonales}} $(p_n)$ para $w$ (monic
normalización: $p_n = t^n + \cdots$).\index{ortogonal
polinomios}

\medskip
\noindent\textbf{Parte I --- General theory.}
\begin{enumerate}
  \item Demuestre que $p_n$ es ortogonal a todo polinomio de
        grado $< n$, y que $(p_0, \dots, p_n)$ es una base
        de $\R_n[t]$.
  \item (Recurrencia de tres términos) Mostrar que hay reales $a_n,
        b_n$ con
        \[
        p_{n+1}(t) = (t - a_n)\,p_n(t) - b_n\,p_{n-1}(t),
        \qquad b_n = \frac{\norm{p_n}^2}{\norm{p_{n-1}}^2} >
        0 .
        \]
        \emph{(Expand $t\,p_n$ in the basis $(p_k)_{k \leq
        n+1}$ and kill coefficients by orthogonality, using
        $\langle tp_n, p_k\rangle = \langle p_n,
        tp_k\rangle$.)}
  \item (Raíces) Muestre que $p_n$ tiene raíces $n$ \emph{distinto},
        todos \omterm{def:b3:topology:interior}{interior} a $I$. \emph{(Let $t_1 < \dots < t_m$ be
        the \omterm{def:b3:topology:interior}{interior} sign changes of $p_n$; if $m < n$,
        test $p_n$ against $\prod_{i\leq m}(t - t_i)$ and
        contradict orthogonality.)}
\end{enumerate}

\medskip
\noindent\textbf{Parte II --- Legendre ($I = \intcc{-1}1$, $w =
1$).} Defina$P_n(t) = \frac{1}{2^nn!}\,\frac{\dd^n}{\dd
t^n}\bigl[(t^2 - 1)^n\bigr]$ (Rodrigues).
\begin{enumerate}[resume]
  \item Mostrar $\deg P_n = n$ con coeficiente principal
        $\frac{(2n)!}{2^n(n!)^2}$, y, integrando por partes
        $n$ veces, que $\langle P_n, Q\rangle = 0$ por cada
        polinomio $Q$ de grado $< n$: los $P_n$ son (hasta
        normalización) el \omterm{pb:b3:hilbert:1}{polinomios ortogonales} para $w = 1$.
  \item Calcular $\norm{P_n}_2^2 = \frac{2}{2n+1}$
        \emph{(integrate by parts $n$ times against itself and
        reduce to a Beta/Wallis integral,
        \cref{exo:b3:product:8})}.
  \item Demuestre que los polinomios de Legendre normalizados forman una
        \omterm{def:b3:hilbert:onb}{base de hilabert} de $L^2(\intcc{-1}1)$
        \emph{(Weierstrass, \cref{cor:b3:complete:weierstrass},
        plus \omterm{ex:b3:lebesgue:gamma}{densidad} of $\mathcal C$ in $L^2$)} y expandir
        $f(t) = \abs t$ hasta el grado $2$: calcular el mejor
        $L^2$ cuadrático: aproximación de $\abs t$.
\end{enumerate}

\medskip
\noindent\textbf{Parte III --- Hermite ($I = \R$, $w(t) =
\eu^{-t^2}$).} Defina$H_n(t) =
(-1)^n\eu^{t^2}\frac{\dd^n}{\dd t^n}\eu^{-t^2}$.
\begin{enumerate}[resume]
  \item Demuestre que $H_n$ es un polinomio de grado $n$ con
        coeficiente principal $2^n$, ese $H_{n+1} = 2tH_n -
        H_n'$y ese$\langle H_m, H_n\rangle_w =
        \delta_{mn}\,2^nn!\sqrt\pi$ \emph{(se separa de nuevo)}.
  \item Demuestre que la familia Hermite es total en $L^2(\R,
        \eu^{-t^2}\dd t)$, admitiendo un resultado de
        \cref{ch:b3:fouriertransform}: si $g \in L^1(\R)$ tiene
        $\int g(t)\eu^{-\iu\xi t}\dd t = 0$ para todos los $\xi$,
        luego $g = 0$ a.e. \emph{(For $f \perp$ all $H_n$,
        i.e.\ $\perp$ all polynomials: show $z \mapsto \int
        f(t)\eu^{-t^2}\eu^{-\iu zt}\dd t$ is well defined,
        expand the exponential as a series, justify the
        interchange by domination, and conclude that the
        Fourier transform of $f\eu^{-t^2}$ vanishes.)}
\end{enumerate}

\medskip
\noindent\textbf{Parte IV --- Gauss quadrature.} Reparar $n$, dejar
$t_1 < \dots < t_n$ sean las raíces de $p_n$ (Parte I) y definan
los pesos $w_i = \int_I \ell_i(t)\,w(t)\dd t$ donde
$\ell_i$ son los polinomios de la base de interpolación de Lagrange en
el $t_i$.
\begin{enumerate}[resume]
  \item Demuestre que la regla de cuadratura $Q(f) = \sum_iw_if(t_i)$
        es exacto en todos los polinomios de grado $\leq n - 1$
        (interpolación), y de hecho --- el milagro --- en
        todos los polinomios de grado $\leq 2n - 1$: escribe $P =
        qp_n + r$y usa ortogonalidad en el cociente$q$.
  \item Mostrar que los pesos son positivos.
        \emph{(apply the rule to $\ell_i^2$, of degree $2n -
        2$)}, y deducir del teorema de Polya
        (\cref{exo:b3:banach:9}) que la cuadratura de Gauss
        converge: $Q_n(f) \to \int_I fw$ por cada \omterm{def:b3:topology:continuity}{continuo}
        $f$ en un \omterm{def:b3:topology:compact}{compacto} $I$.
  \item Para $n = 2$, $I = \intcc{-1}1$, $w = 1$: calcule el
        nodos $\pm\frac1{\sqrt3}$ y pesos $1, 1$, y
        verifique la exactitud en $1, t, t^2, t^3$ a mano. Comparar
        con la regla del trapezoide en las mismas dos evaluaciones
        puntos.
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Chebyshev: the polynomials that
oscillate best.} Ahora $I = \intcc{-1}1$ y $w(t) =
\frac1{\sqrt{1 - t^2}}$.
\begin{enumerate}[resume]
  \item Demuestre que $T_n(\cos\theta) = \cos n\theta$ define un
        polinomio $T_n$ de grado $n$ (establecer $T_{n+1} =
        2t\,T_n - T_{n-1}$ a partir de una identidad trigonométrica),
        con coeficiente principal $2^{n-1}$ para $n \geq 1$;
        y que la sustitución $t = \cos\theta$ da
        \[
        \langle T_m, T_n\rangle_w =
        \int_0^\pi\cos m\theta\,\cos n\theta\,\dd\theta
        = 0 \ (m \neq n), \qquad
        \norm{T_0}_w^2 = \pi,\ \ \norm{T_n}_w^2 = \frac\pi2 :
        \]
        los $T_n$ son los \omterm{pb:b3:hilbert:1}{polinomios ortogonales} para esto
        peso y expansiones de Chebyshev \emph{son} Fourier
        Serie de cosenos disfrazada.
  \item Ubique explícitamente las raíces $n$ $t_k =
        \cos\frac{(2k-1)\pi}{2n}$y los extremos$n + 1$
        $s_j = \cos\frac{j\pi}n$ de $T_n$ en $\intcc{-1}1$,
        donde $T_n(s_j) = (-1)^j$: el gráfico
        \emph{equioscila} entre $\pm1$.
  \item (Minimax) Muestra que entre todos \emph{mónico}
        polinomios de grado $n$, el polinomio
        $2^{1-n}T_n$ tiene la norma sup más pequeña en
        $\intcc{-1}1$, es decir $2^{1-n}$ --- y es el
        minimizador único. \emph{(If a monic $P$ had
        $\sup\abs P < 2^{1-n}$, the difference $2^{1-n}T_n -
        P$, of degree$\leq n-1$, would alternate in sign at
        the $n+1$ equioscillation points.)}
  \item Aplicación a la interpolación: para los nodos $t_1 < \dots
        < t_n$en$\intcc{-1}1$, el error de Lagrange

        $\omega(t) = \prod_i(t - t_i)$. Muestra que elegir
        Las raíces de Chebyshev se minimizan como nudos.
        $\sup_{\intcc{-1}1}\abs\omega$, y dé el
        límite resultante $\norm{f -
        L_nf}_\infty \leq \frac{\norm{f^{(n)}}_\infty}
        {2^{n-1}\,n!}$ --- comparar con nodos igualmente espaciados
        (Exponga el fenómeno de Runge como advertencia).
  \item Verificar $\abs{T_n'(\pm1)} = n^2$ \emph{(differentiate
        $T_n(\cos\theta) = \cos n\theta$ and take limits
        $\theta \to 0, \pi$)}: polinomios acotados por $1$ en
        $\intcc{-1}1$ puede tener una derivada tan grande como $n^2$
        en el borde (la desigualdad de Markov dice que no es mayor ---
        declaración únicamente). ¿Dónde en el intervalo está el
        ¿Derivado consolidado sólo $O(n)$?
  \item (Chebyshev--cuadratura de Gauss) Demuestre que la curva de Gauss
        regla para el peso $w$ en las raíces de Chebyshev $n$
        tiene pesos \emph{igual} $w_i = \frac\pi n$
        \emph{(exactness on $T_0, \dots, T_{n-1}$ plus the
        trigonometric sums
        $\sum_{k=1}^n\cos\bigl(j\tfrac{(2k-1)\pi}{2n}\bigr) =
        0$for$1 \leq j \leq n - 1$)}: el más uniforme de
        todas las cuadraturas. Escríbalo para $n = 3$.
\end{enumerate}

\medskip
\noindent\textbf{Part VI --- Christoffel--Darboux,
interlacing, and the Jacobi matrix.} Volver a un peso general;
$h_k = \norm{p_k}^2$ (mónico $p_k$), $b_k = h_k/h_{k-1}$.
\begin{enumerate}[resume]
  \item (Norma mínima) Muestra que entre todos \emph{mónico}
        polinomios de grado $n$, el ortogonal $p_n$ es
        el único de norma mínima $L^2(w)$ --- identificar
        la minimización como una proyección ortogonal sobre
        $\R_{n-1}[t]$ (\cref{thm:b3:hilbert:projection} o
        la proyección de dimensión finita del Año 2). el

        con $L^\infty$ en lugar de $L^2$: mismo héroe, dos
        normas.
  \item (Christoffel--Darboux) Demostrar, por inducción en $n$
        utilizando la recurrencia de tres términos, la identidad
        \[
        \sum_{k=0}^{n}\frac{p_k(x)\,p_k(y)}{h_k}
        = \frac{p_{n+1}(x)\,p_n(y) -
        p_n(x)\,p_{n+1}(y)}{h_n\,(x - y)}
        \qquad (x \neq y),
        \]
        y su forma confluente ($y \to x$):
        $\sum_{k\leq n}\frac{p_k(x)^2}{h_k} =
        \frac{p_{n+1}'(x)p_n(x) - p_n'(x)p_{n+1}(x)}{h_n}$.
  \item Deduzca que $p_n$ y $p_{n+1}$ no tienen raíz común,
        y que en cada raíz $x_0$ de $p_{n+1}$:
        $p_n(x_0)\,p_{n+1}'(x_0) > 0$. Concluir el
        \emph{entrelazando} de raíces: entre dos consecutivas
        Las raíces de $p_{n+1}$ se encuentran exactamente en una raíz de $p_n$.
  \item (matriz de Jacobi) Sea $J_n$ el $n\times n$
        Matriz tridiagonal simétrica con diagonal $a_0,
        \dots, a_{n-1}$y entradas fuera de la diagonal$\sqrt{b_1},
        \dots, \sqrt{b_{n-1}}$. Demuestre por inducción que
        $\det(tI_n - J_n) = p_n(t)$, entonces las raíces de $p_n$
        son los valores propios de una matriz simétrica real ---
        volviendo a demostrar en una línea que son reales, y (con
        el entrelazado de arriba) atando \omterm{pb:b3:hilbert:1}{polinomios ortogonales}
        al mundo espectral de \cref{ch:b3:spectral}.
  \item (Síntesis) Armar el diccionario para los tres
        familias clásicas (Legendre, Hermite, Chebyshev):
        intervalo, peso, fórmula definitoria, tres términos
        recurrencia, norma y hábitat natural de cada uno.
        (cuadratura y aproximación en compacta; gaussiano
        análisis; métodos minimax y coseno de Fourier). uno
        frase sobre lo que la teoría general (Partes I, VI)
        dio que ningún cálculo individual podría hacerlo.
\end{enumerate}

behind the weights.} Aquí $I$ es \omterm{def:b3:topology:compact}{compacto} y $f \in \mathcal

\begin{enumerate}[resume]
  \item (fórmula del error de Gauss) Sea $Hf$ el Hermite
        interpolante de grado $\leq 2n - 1$ que coincide con $f$ y
        $f'$ en los nodos $t_1, \dots, t_n$ (probar su

        \[
        f(t) - Hf(t) =
        \frac{f^{(2n)}(\xi_t)}{(2n)!}\;p_n(t)^2
        \]
        por el argumento habitual de función auxiliar). Deducir, por
        integrando esta identidad contra $w$ y apretando
        entre los extremos de $f^{(2n)}$, que
        \[
        \int_I f\,w - Q_n(f)
        = \frac{f^{(2n)}(\xi)}{(2n)!}\,h_n
        \qquad\text{for some } \xi \in I,
        \]

  \item (Los pesos son valores de Christoffel) Usando el

        exactitud de $Q_n$ hasta el grado $2n - 2$, demostrar

        w_i \;=\;

        \]

        fórmula. Verifique que recupere $w_1 = w_2 = 1$ para $n =
        2$,$I = \intcc{-1}1$,$w = 1$.
  \item (Todo se verifica en una integral) Para Chebyshev
        peso y nodos $n = 3$, calcule ambos lados de
        \[

        \qquad

        \]

\end{enumerate}
\end{problem}
