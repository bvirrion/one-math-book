\chapter{La integral de Lebesgue}\label{ch:b3:lebesgue}

La integral de Riemann corta el \emph{dominio} en pequeños
intervalos; Lonchas de Lebesgue el \emph{rango}: para integrar
$f$, \omterm{def:b3:measure:measure}{medida} los conjuntos $\{f > t\}$. El cambio parece inocente.
y es revolucionario. Límites e integrales, siempre peleando
en la teoría de Riemann (¡se requiere convergencia uniforme!), son
conciliado por tres teoremas de convergencia --- monótono
convergencia, Fatou, dominó la convergencia --- cuyas hipótesis
son casi vergonzosamente débiles. Este capítulo construye el
integral sobre un \omterm{def:b3:measure:measure}{medir el espacio} $(X, \mathcal A,
\mu)$ arbitrario, prueba los tres teoremas, establece la solución exacta
relación con la integral de Riemann (una función acotada es
\omterm{def:b3:lebesgue:l1}{integrable en Riemann} si es \omterm{def:b3:topology:continuity}{continuo} \omterm{def:b3:lebesgue:l1}{casi en todas partes}), y
industrializa la diferenciación de parámetros dependientes
integrales --- la técnica que utiliza el problema del fin de semana para
calcular $\int_0^\infty\frac{\sin x}x\,\dd x$ y
$\int_\R \eu^{-x^2}\dd x$.

\section{Funciones medibles}

\begin{definition}\label{def:b3:lebesgue:measurable}
Sean $(X, \mathcal A)$, $(Y, \mathcal B)$ espacios medibles.
$f \colon X \to Y$ es \emph{mensurable}\index{medible
función} si $f^{-1}(B) \in \mathcal A$ por cada $B \in
\mathcal B$. De verdad (o valorado en$[-\infty,+\infty]$)
funciones, $Y = \R$ lleva su álgebra Borel $\sigma$, y
basta con comprobar $f^{-1}(\intoo t{+\infty}) = \{f > t\} \in
\mathcal A$para todos los$t \in \R$: los buenos conjuntos$\{B :
f^{-1}(B) \in \mathcal A\}$forman un álgebra$\sigma$
(las preimágenes conmutan con operaciones establecidas) que contienen el
generando rayos (\cref{def:b3:measure:borel},
\cref{met:b3:measure:goodsets}).
\end{definition}

\begin{proposition}\label{prop:b3:lebesgue:stability}
(a) Las composiciones de las aplicaciones \omterm{def:b3:lebesgue:measurable}{mensurable} son \omterm{def:b3:lebesgue:measurable}{mensurable}; \omterm{def:b3:topology:continuity}{continuo
mapas} son \omterm{def:b3:lebesgue:measurable}{Borel mensurable}.
(b) Si $f, g \colon X \to \R$ son \omterm{def:b3:lebesgue:measurable}{mensurable}, también lo son $f + g$,
$fg$, $\max(f,g)$, $\abs f$, $\lambda f$.
(c) Si $(f_n)$ son \omterm{def:b3:lebesgue:measurable}{mensurable} con valores en
$[-\infty, +\infty]$, luego $\sup_nf_n$, $\inf_nf_n$,
$\limsup f_n$, $\liminf f_n$ son \omterm{def:b3:lebesgue:measurable}{mensurable}; si $f_n \to f$
puntualmente, $f$ es \omterm{def:b3:lebesgue:measurable}{mensurable}.
\end{proposition}

\begin{proof}
(a) $(g\circ f)^{-1}(B) = f^{-1}(g^{-1}(B))$; \omterm{def:b3:topology:continuity}{continuidad} da
\omterm{def:b3:lebesgue:measurable}{mensurabilidad} a través de la generación \omterm{def:b3:topology:topology}{conjuntos abiertos}
(\cref{pb:b3:measure:1}, pregunta 10, en forma general).
(b) $(f, g) \colon X \to \R^2$ es \omterm{def:b3:lebesgue:measurable}{mensurable} para Borel
$\sigma$-álgebra de $\R^2$ --- marque las casillas abiertas, que
generar (las aperturas de $\R^2$ son uniones contables de racionales
cajas): $(f,g)^{-1}(U\times V) = f^{-1}(U)\cap g^{-1}(V)$ ---
y $+, \times, \max$ son \omterm{def:b3:topology:continuity}{continuo} $\R^2 \to \R$: redactar.
(c)$\{\sup f_n > t\} = \bigcup_n\{f_n > t\}$; $\inf = -\sup(-)$;
$\limsup = \inf_N\sup_{n \geq N}$; un límite puntual es propio
$\limsup$.
\end{proof}

\begin{definition}\label{def:b3:lebesgue:simple}
Un \emph{función sencilla}\index{función sencilla} es un \omterm{def:b3:lebesgue:measurable}{medible
función} con un número finito de valores: $s = \sum_{i=1}^n
c_i\,\mathbf 1_{A_i}$,$A_i \in \mathcal A$disjunto,$c_i \geq
0$ (para la teoría no negativa). Su integral es
\[
\int s\,\dd\mu = \sum_i c_i\,\mu(A_i) \in [0, +\infty]
\]
(convenio $0\cdot\infty = 0$); el valor no depende de
el \omterm{def:b3:representations:rep}{representación} (refinar dos particiones).
\end{definition}

\begin{theorem}[Aproximación mediante funciones simples]
\label{thm:b3:lebesgue:approximation}
Cada \omterm{def:b3:lebesgue:measurable}{mensurable} $f \colon X \to [0, +\infty]$ es puntual
límite de una secuencia \emph{creciente} de \omterm{def:b3:lebesgue:simple}{funciones simples}:
\[
s_n = \sum_{k=1}^{n2^n} \frac{k-1}{2^n}\,
\mathbf 1_{\{\frac{k-1}{2^n} \leq f < \frac k{2^n}\}}
+ n\,\mathbf 1_{\{f \geq n\}} \nearrow f .
\]
\end{theorem}

\begin{proof}
Cada $s_n$ es simple (los conjuntos son preimágenes de conjuntos de Borel).
Monotonicidad: pasar de $n$ a $n+1$ divide cada diádica
nivel en dos y nunca disminuye el valor asignado (un punto
con $\frac{k-1}{2^n} \leq f(x) < \frac k{2^n}$ obtiene cualquiera de los dos
$\frac{2k-2}{2^{n+1}}$ o $\frac{2k-1}{2^{n+1}}$, ambos $\geq
\frac{k-1}{2^n}$; el límite$n$ también sube). Convergencia: si
$f(x) < \infty$, para $n > f(x)$ tenemos $f(x) - s_n(x) \leq
2^{-n}$; si$f(x) = \infty$,$s_n(x) = n \to \infty$.
\end{proof}

\section{Los teoremas de la integral y de la convergencia}

\begin{definition}\label{def:b3:lebesgue:integral}
Para \omterm{def:b3:lebesgue:measurable}{mensurable} $f \geq 0$:
\[
\int f \,\dd\mu = \sup\Bigl\{\int s\,\dd\mu : s \text{ simple},
\ 0 \leq s \leq f\Bigr\} \in [0, +\infty].
\]
Es monótono en $f$ por construcción y extiende el simple
caso (para $f$ simple, la sup se alcanza en $f$: comparación de
integrales simples mediante refinamientos comunes).
\end{definition}

\begin{theorem}[Convergencia monótona, Beppo Levi]
\label{thm:b3:lebesgue:mct}
Si $0 \leq f_n \nearrow f$ puntual (\omterm{def:b3:lebesgue:measurable}{mensurable}), entonces
\[
\int f_n\,\dd\mu \nearrow \int f\,\dd\mu .
\]
\index{teorema de convergencia monótona}
\end{theorem}

\begin{proof}
$f$ es \omterm{def:b3:lebesgue:measurable}{mensurable} (\cref{prop:b3:lebesgue:stability}(c)) y
$\int f_n$ aumenta a algo de $L \leq \int f$ (monotonicidad).
Por el contrario, arregle un simple $s = \sum c_i\mathbf 1_{A_i} \leq f$
y $\theta \in (0,1)$; los conjuntos $E_n = \{f_n \geq \theta s\}$
son \omterm{def:b3:lebesgue:measurable}{mensurable} y aumentan a $X$ (donde $s(x) > 0$: $f(x)
\geq s(x) > \theta s(x)$, por lo que eventualmente$f_n(x) \geq \theta
s(x)$; donde$s(x) = 0$: trivialmente). entonces
\[
\int f_n \geq \int_{E_n}\theta s\,\dd\mu
= \theta\sum_i c_i\,\mu(A_i \cap E_n)
\xrightarrow[n\to\infty]{} \theta\sum_ic_i\,\mu(A_i)
= \theta\int s
\]
por \omterm{def:b3:topology:continuity}{continuidad} desde abajo (\cref{prop:b3:measure:basics}(c)).
Entonces $L \geq \theta\int s$ para todos los $\theta < 1$ y todos los $s
\leq f$simples:$L \geq \int f$.
\end{proof}

\begin{corollary}\label{cor:b3:lebesgue:additivity}
Para \omterm{def:b3:lebesgue:measurable}{mensurable} $f, g \geq 0$ y $c \geq 0$: $\int(f + g) =
\int f + \int g$y$\int cf = c\int f$; para una serie de
\omterm{def:b3:lebesgue:measurable}{funciones medibles}, $\int\sum_nf_n =
\sum_n\int f_n$ no negativos.
\end{corollary}

\begin{proof}
Para \omterm{def:b3:lebesgue:simple}{funciones simples}, la aditividad es un cálculo sobre una base común
refinamiento. En general tome $s_n \nearrow f$, $t_n \nearrow g$
(\cref{thm:b3:lebesgue:approximation}): $s_n + t_n \nearrow f +
g$ y MCT pasa la aditividad al límite. la serie
La declaración es MCT aplicada a las sumas parciales.
\end{proof}

\begin{theorem}[Lema de Fatou]\label{thm:b3:lebesgue:fatou}
Para \omterm{def:b3:lebesgue:measurable}{mensurable} $f_n \geq 0$:
\[
\int \liminf_n f_n \,\dd\mu \;\leq\; \liminf_n \int
f_n\,\dd\mu .
\]
\index{El lema de Fatou}
\end{theorem}

\begin{proof}
Sea $g_N = \inf_{n\geq N}f_n$: \omterm{def:b3:lebesgue:measurable}{mensurable}, $0 \leq g_N \nearrow
\liminf f_n$y$g_N \leq f_n$por cada$n \geq N$, entonces
$\int g_N \leq \inf_{n \geq N}\int f_n$. Aplicar MCT a la izquierda
lado: $\int\liminf f_n = \lim_N\int g_N \leq
\lim_N\inf_{n\geq N}\int f_n = \liminf\int f_n$.
\end{proof}

\begin{definition}\label{def:b3:lebesgue:l1}
Un \omterm{def:b3:lebesgue:measurable}{mensurable} $f \colon X \to \R$ (o $\C$) es
\emph{integrable}\index{función integrable} si $\int\abs
f\,\dd\mu < \infty$; entonces$\int f = \int f^+ - \int f^-$
(partes positivas y negativas; partes reales e imaginarias en el
caso complejo). La integral es lineal en funciones integrables.
(descomponer y recombinar partes positivas; el caso complejo
se reduce al real) y satisface $\abs{\int f} \leq
\int\abs f$(caso real:$\pm\int f = \int(\pm f) \leq
\int\abs f$; caso complejo: multiplicar por una constante unimodular
para hacer real la integral). Una propiedad contiene \emph{casi
en todas partes} (a.e.)\index{casi en todas partes} si falla solo en
un conjunto nulo $\mu$; modificar $f$ en un conjunto nulo cambia no
integral (la diferencia está dominada por $\infty\cdot\mathbf
1_N$, de la integral$0$).
\end{definition}

\begin{theorem}[Convergencia dominada]\label{thm:b3:lebesgue:dct}
Sea $f_n \to f$ a.e., con $\abs{f_n} \leq g$ a.e.\ para un valor fijo
\emph{\omterm{def:b3:lebesgue:l1}{integrable}}$g$. Entonces $f$ es \omterm{def:b3:lebesgue:l1}{integrable} y
\[
\int f_n\,\dd\mu \longrightarrow \int f\,\dd\mu,
\qquad\text{indeed}\quad \int\abs{f_n - f}\,\dd\mu \to 0 .
\]
\index{teorema de convergencia dominada}
\end{theorem}

\begin{proof}
Descarta un conjunto nulo para formular las hipótesis puntualmente. $\abs f
\leq g$:$f$es \omterm{def:b3:lebesgue:l1}{integrable}. Las funciones$h_n = 2g - \abs{f_n
- f} \geq 0$satisfacen$\liminf h_n = 2g$; fatou da
\[
\int 2g \leq \liminf\int\bigl(2g - \abs{f_n - f}\bigr)
= \int 2g - \limsup\int\abs{f_n - f},
\]
entonces $\limsup\int\abs{f_n - f} \leq 0$ (la resta es legal:
$\int 2g < \infty$). Finalmente $\abs{\int f_n - \int f} \leq
\int\abs{f_n - f} \to 0$.
\end{proof}

\begin{method}\label{met:b3:lebesgue:convergence}
Frente a $\lim_n\int f_n$: prueba, en orden --- (1) es el
¿Secuencia monótona (o una serie de términos no negativos)? MCT, no
\omterm{def:b3:lebesgue:l1}{integrabilidad} necesario. (2) ¿Existe un único \omterm{def:b3:lebesgue:l1}{integrable}?
dominador $g \geq \abs{f_n}$, encontrado por límites crudos
(``$\sup_n$'' las estimaciones)? DCT. (3) Sin dominación, no
¿monotonicidad? Fatou todavía limita a un lado, y la igualdad puede
realmente falla: el tope de escape $f_n = n\mathbf
1_{\intoo0{1/n}}$tiene$\int f_n = 1$pero$f_n \to 0$ a.e.
La dominación es exactamente lo que prohíbe a la masa escapar hacia
infinito, vertical u horizontalmente.
\end{method}

\section{Riemann contra Lebesgue}

\begin{theorem}[criterio de Lebesgue]
\label{thm:b3:lebesgue:riemann}
Sea $f \colon \intcc ab \to \R$ \emph{encerrado}. Entonces $f$ es
\omterm{def:b3:lebesgue:l1}{integrable en Riemann} si $f$ es \omterm{def:b3:topology:continuity}{continuo} $\lambda$-casi
en todas partes; en ese caso $f$ es \omterm{def:b3:lebesgue:l1}{integrable en Lebesgue} y los dos
las integrales coinciden.\index{Integral de Riemann vs Lebesgue}
\end{theorem}

\begin{proof}
Para una subdivisión $\sigma = (a = x_0 < \dots < x_N = b)$, sea
$L_\sigma$ y $U_\sigma$ sean las funciones escalonadas iguales, en cada
$\intoo{x_{i-1}}{x_i}$, a $m_i = \inf_{[x_{i-1}, x_i]}f$ y
$M_i = \sup$; las sumas de Darboux son sus integrales (Riemann y
Lebesgue está de acuerdo en las funciones escalonadas, ambas dan $\sum
m_i\Delta x_i$). Tome una secuencia de subdivisiones$\sigma_n$,
cada uno refinando el último, de malla $\to 0$, con sumas Darboux
convergiendo a las integrales de Darboux inferior y superior de $f$.
Los refinamientos hacen que $L_{\sigma_n}$ sea no decreciente y
$U_{\sigma_n}$ punto no creciente fuera del conjunto contable
$D$ de todos los puntos de división; llame a los límites $\ell$ y $u$
(\omterm{def:b3:lebesgue:measurable}{mensurable}, \cref{prop:b3:lebesgue:stability}). Para $x \notin
D$, escriba$I_n(x)$para el intervalo abierto$\sigma_n$
que contiene $x$: $\ell(x) = \sup_n\inf_{I_n(x)}f$ y $u(x) =
\inf_n\sup_{I_n(x)}f$; Dado que las mallas se reducen a$0$, estas
son el \emph{sobres} inferior y superior de $f$ en $x$
--- $u(x) - \ell(x)$ es la oscilación de $f$ en $x$ --- entonces
que \emph{$\ell(x) = u(x)$ iff $f$ is \omterm{def:b3:topology:continuity}{continuo} at $x$}. Por MCT/DCT (intervalo finito acotado):
\[
\int_{\intcc ab}\ell\,\dd\lambda = \lim_n\int L_{\sigma_n}
= \underline{\int}f,
\qquad
\int_{\intcc ab}u\,\dd\lambda = \overline{\int}f .
\]
$f$ \omterm{def:b3:lebesgue:l1}{integrable en Riemann} $\iff$ $\underline\int f =
\overline\int f$ $\iff$ $\int(u - \ell) = 0$ $\iff$ $u = \ell$
e.e.\ ($u - \ell \geq 0$; \cref{exo:b3:lebesgue:5}) $\iff$ $f$
\omterm{def:b3:topology:continuity}{continuo} a.e. En ese caso $\ell \leq f \leq u$ con $\ell =
u$a.e.:$f$es igual a \omterm{def:b3:lebesgue:measurable}{mensurable}$\ell$ a.e., por lo tanto es
\omterm{def:b3:lebesgue:measurable}{Lebesgue-medible} (\omterm{def:b3:complete:complete}{lo completo} de $\lambda$) con
$\int f\,\dd\lambda = \int\ell\,\dd\lambda = \underline\int f =
\int_a^bf$.
\end{proof}

\begin{example}\label{ex:b3:lebesgue:riemannexamples}
$\mathbf 1_\Q$ no está en ninguna parte \omterm{def:b3:topology:continuity}{continuo}: no \omterm{def:b3:lebesgue:l1}{integrable en Riemann}
--- pero Lebesgue-trivial: $\int\mathbf 1_\Q\,\dd\lambda =
\lambda(\Q) = 0$. Función de Thomae (\,$\frac1q$ en racionales
$\frac pq$, $0$ en otros lugares) es \omterm{def:b3:topology:continuity}{continuo} exactamente en el
irracionales: \omterm{def:b3:lebesgue:l1}{integrable en Riemann} con integral $0$. y
\emph{incorrecto} Las integrales de Riemann son una noción diferente:
$\int_0^\infty\frac{\sin x}x\,\dd x$ converge como un límite de
$\int_0^A$ (el problema del fin de semana lo calcula $= \frac\pi2$),
pero $\frac{\sin x}x \notin L^1(\intoo0{+\infty})$: lo absoluto
integral diverge como la serie armónica
(\cref{exo:b3:lebesgue:6}). La teoría de Lebesgue opera
convergencia condicional para teoremas de límites robustos.
\end{example}

\section{Integrales con parámetros}

En todo momento, $(X, \mathcal A, \mu)$ es un \omterm{def:b3:measure:measure}{medir el espacio}, $T$ es un
espacio métrico (el parámetro) y $f \colon T \times X \to \C$
con $f(t, \cdot)$ \omterm{def:b3:lebesgue:l1}{integrable} por cada $t$; establezca $F(t) = \int_X
f(t, x)\,\dd\mu(x)$.

\begin{theorem}[continuidad]\label{thm:b3:lebesgue:paramcont}
Supongamos: $t \mapsto f(t,x)$ es \omterm{def:b3:topology:continuity}{continuo} en $t_0$ para a.e.\
$x$, y hay un \omterm{def:b3:lebesgue:l1}{integrable} $g$ con $\abs{f(t,x)} \leq
g(x)$para todos los$t$en un \omterm{def:b3:topology:topology}{entorno} de$t_0$y a.e.$x$.
Entonces $F$ es \omterm{def:b3:topology:continuity}{continuo} en $t_0$.\index{integrales de parámetros}
\end{theorem}

\begin{proof}
Para cualquier secuencia $t_n \to t_0$: $f(t_n, \cdot) \to f(t_0,
\cdot)$a.e., dominada por$g$: DCT da$F(t_n) \to F(t_0)$;
\omterm{def:b3:topology:continuity}{continuidad} secuencial es suficiente en espacios métricos
(\cref{rem:b3:topology:sequences}).
\end{proof}

\begin{theorem}[Diferenciación bajo la integral]
\label{thm:b3:lebesgue:paramdiff}
Sea $T$ un intervalo abierto de $\R$. Supongamos: para a.e.\ $x$,
$t \mapsto f(t,x)$ es diferenciable en $T$, con
\[
\Bigl|\frac{\partial f}{\partial t}(t, x)\Bigr| \leq g(x)
\quad \text{for all } t \in T,\ \text{a.e. } x,
\]
$g$ \omterm{def:b3:lebesgue:l1}{integrable}. Entonces $F$ es diferenciable en $T$ con $F'(t)
= \int_X \frac{\partial f}{\partial t}(t, x)\,\dd\mu(x)$.
\end{theorem}

\begin{proof}
Arreglar $t$ y $h_n \to 0$: los cocientes de diferencia
\[
\varphi_n(x) = \frac{f(t + h_n, x) - f(t, x)}{h_n}
\longrightarrow \frac{\partial f}{\partial t}(t,x)
\quad\text{a.e.},
\]
y los límites de desigualdad del valor medio $\abs{\varphi_n(x)} \leq
\sup_{s}\abs{\partial_tf(s,x)} \leq g(x)$: se aplica DCT, y
$\frac{F(t + h_n) - F(t)}{h_n} = \int\varphi_n \to
\int\partial_t f(t, \cdot)$.
\end{proof}

\begin{example}[La función Gamma]\label{ex:b3:lebesgue:gamma}
Para $t > 0$ deje \index{función gamma}
\[
\Gamma(t) = \int_0^{+\infty} x^{t-1}\eu^{-x}\,\dd x .
\]
La integral converge: cerca de $0$, $x^{t-1}$ es \omterm{def:b3:lebesgue:l1}{integrable} ($t >
0$); en el infinito,$x^{t-1}\eu^{-x} \leq C\eu^{-x/2}$.
Integración por partes (en $[\varepsilon, A]$, luego límites vía
MCT) da la ecuación funcional $\Gamma(t + 1) =
t\,\Gamma(t)$, de donde$\Gamma(n+1) = n!$: el factorial
interpolado. En cada $\intcc ab \subseteq \intoo0{+\infty}$,
$\partial_t\bigl(x^{t-1}\eu^{-x}\bigr) = \ln
x\cdot x^{t-1}\eu^{-x}$está dominado por$\abs{\ln x}(x^{a-1} +
x^{b-1})\eu^{-x}$, \omterm{def:b3:lebesgue:l1}{integrable}:$\Gamma$es$\mathcal C^1$, y
por inducción $\mathcal C^\infty$, con $\Gamma^{(k)}(t) =
\int_0^\infty(\ln x)^kx^{t-1}\eu^{-x}\dd x$. el valor
$\Gamma(\frac12) = \sqrt\pi$ es la integral gaussiana en
disfrazar (\cref{pb:b3:lebesgue:1}).
\end{example}

(a) Demuestre que una función monótona $\R \to \R$ es

\end{exercise}

\section{Problema: dos integrales célebres}

\begin{problem}[{Problema de fin de semana --- la integral gaussiana y
Integral de Dirichlet, solo por parámetros}]
\label{pb:b3:lebesgue:1}
Análisis aplicado de dos reglas integrales:
\[
G = \int_{-\infty}^{+\infty}\eu^{-x^2}\dd x = \sqrt\pi,
\qquad
D = \int_0^{+\infty}\frac{\sin x}{x}\,\dd x = \frac\pi2
\]
(la segunda como integral impropia,
\cref{ex:b3:lebesgue:riemannexamples}). Probamos ambos usando
sólo las herramientas de este capítulo.

\medskip
\noindent\textbf{Parte I --- The Gaussian.} Para juego $t \geq 0$
\[
A(t) = \Bigl(\int_0^t\eu^{-x^2}\dd x\Bigr)^{2},
\qquad
B(t) = \int_0^1\frac{\eu^{-t^2(1 + x^2)}}{1 + x^2}\,\dd x .
\]
\begin{enumerate}
  \item Justifique que $A$ y $B$ son $\mathcal C^1$ en
        $\intoo0{+\infty}$ y calcula $A'$ y $B'$; mostrar
        $A'(t) + B'(t) = 0$. \emph{(En $B'$, sustituye $u =
        tx$.)}
  \item Calcular $A(0) + B(0)$ y $\lim_{t\to+\infty}(A + B)(t)$
        --- justifique el límite bajo la integral en $B$.
  \item Concluya $\int_0^\infty \eu^{-x^2}\dd x =
        \frac{\sqrt\pi}2$, por lo tanto$G = \sqrt\pi$, y deduzca
        $\Gamma(\tfrac12) = \sqrt\pi$\emph{(substitute $x =
        u^2$in$\Gamma(\frac12)$)}.
\end{enumerate}

\medskip
\noindent\textbf{Parte II --- Dirichlet's integral.} Para juego $t
\geq 0$
\[
F(t) = \int_0^{+\infty}\eu^{-tx}\,\frac{\sin x}{x}\,\dd x .
\]
\begin{enumerate}[resume]
  \item Demuestre que la integral que define $F(t)$ converge para
        cada $t > 0$ como una integral de Lebesgue, y para $t = 0$
        como integral impropia; mostrar que $D =
        \lim_{A\to\infty}\int_0^A\frac{\sin x}x\dd x$ existe
        \emph{(integrate by parts on $[\pi, A]$)}.
  \item Mostrar que $F$ es $\mathcal C^1$ en
        $\intoo0{+\infty}$ con
        \[
        F'(t) = -\int_0^{+\infty}\eu^{-tx}\sin x\,\dd x
        = -\frac{1}{1 + t^2}
        \]
        \emph{(domination on $[t_0, \infty)$ for each $t_0 >
        0$; the last integral by two integrations by parts or
        complex exponentials)}.
  \item Muestre $F(t) \to 0$ como $t \to +\infty$ y deduzca $F(t)
        = \frac\pi2 - \arctan t$en$\intoo0{+\infty}$.
  \item El punto delicado: $D = \lim_{t\to0^+}F(t)$. Pruébalo
        por control uniforme de la cola: para $0 \leq t \leq 1$
        y $A \geq \pi$, integrar por partes para mostrar
        \[
        \Bigl|\int_A^{+\infty}\eu^{-tx}\frac{\sin
        x}x\,\dd x\Bigr| \leq \frac{C}{A}
        \]
        con $C$ independiente de $t$ \emph{(differentiate
        $\frac{\eu^{-tx}}x$ and bound $\abs{\cos}$ by $1$;
        note $t\eu^{-tx} \leq 1/x\cdot(tx\eu^{-tx})$ with
        $\sup_{u\geq0}u\eu^{-u} < 1$)}; luego dividir $F(t) - D$
        en $[0, A]$ (donde DCT se aplica como $t \to 0$) y
        $[A, \infty)$.
  \item Concluye: $D = \frac\pi2$.
\end{enumerate}

\medskip
\noindent\textbf{Parte III --- Dividends.}
\begin{enumerate}[resume]
  \item Calcular $\int_0^{+\infty}\frac{\sin^2x}{x^2}\,\dd x$
        \emph{(integrate by parts and reduce to $D$ via $\sin
        2x = 2\sin x\cos x$)}.
  \item Calcule $\int_0^{+\infty}\frac{1 -
        \cos x}{x^2}\,\dd x$ y verifique la coherencia del
        dos resultados.
  \item Para $a > 0$, calcule
        $\int_0^{+\infty}\frac{\sin(ax)}x\dd x$ y
        $\int_{-\infty}^{+\infty}\eu^{-ax^2}\dd x$ y registrar
        las reglas de escala (serán los caballos de batalla de
        \cref{ch:b3:fouriertransform}).
  \item Explique con precisión por qué $D$ no pudo haber sido tratado.
        por DCT directamente en $t = 0$ (no hay dominador \omterm{def:b3:lebesgue:l1}{integrable} en
        $[0,1]\times[0,\infty)$), y por qué el corte de cola de
        la pregunta 7 es el sustituto honesto --- este patrón
        (``uniforme \omterm{def:b3:lebesgue:l1}{integrabilidad} de colas'') se repite a lo largo de
        análisis.
\end{enumerate}

\medskip
\noindent\textbf{Parte IV --- The \omterm{ex:b3:lebesgue:gamma}{función gamma} according to
Bohr and Mollerup.} La función $\Gamma$
(\cref{ex:b3:lebesgue:gamma}) satisface $\Gamma(1) = 1$ y
$\Gamma(x+1) = x\Gamma(x)$ --- pero también lo hacen infinitos otros
funciones (multiplicar por cualquier oscilación periódica $1$). una convexidad
La condición fija $\Gamma$ hacia abajo de manera única, y es más profunda.
las identidades luego caen mecánicamente. Una función positiva $f$
en un intervalo es \emph{log-convexo} si $\log f$ es convexo.
\begin{enumerate}[resume]
  \item Demuestre que log-convexo implica convexo, que los productos de
        funciones log-convexas y sus composiciones con afines
        las aplicaciones son log-convexos y --- a través de las dos funciones
        desigualdad de Hölder $\int\abs{uv} \leq
        \bigl(\int\abs u^p\bigr)^{1/p}\bigl(\int\abs
        v^q\bigr)^{1/q}$, probada directamente de Young
        desigualdad --- que $\Gamma$ es log-convexo en
        $\intoo0\infty$.
  \item (Lema de pendiente) Sea $g$ convexo en $\intoo0\infty$
        con $g(n+1) - g(n) = \log n$ para cada número entero $n
        \geq 1$. Para$x \in \intoc01$y$n \geq 2$, compare
        las pendientes de $g$ sobre $[n-1, n]$, $[n, n+x]$ y
        $[n, n+1]$, y deducir
        \[
        x\log(n-1) \;\leq\; g(n + x) - g(n) \;\leq\; x\log n .
        \]
  \item (Bohr--Mollerup) Deje que $f > 0$ satisfaga a $f(1) = 1$,
        $f(x+1) = xf(x)$ y $\log f$ convexos. Desenrollando el
        recursión en $f(n + x) = x(x+1)\cdots(x + n -
        1)\,f(x)$y$f(n) = (n-1)!$, deducir de la pregunta 14
        eso para $x \in \intoc01$
        \[
        f(x) = \lim_{n\to\infty}
        \frac{n!\,n^x}{x(x+1)\cdots(x+n)} :
        \]
        $f$ es único, por lo tanto $f = \Gamma$ y \emph{de gauss
        fórmula límite} se mantienen (extendidos a todos los $x > 0$ por el
        recursión).
  \item Defina \emph{función beta} $B(x, y) =
        \int_0^1t^{x-1}(1-t)^{y-1}\,\dd t$($x, y > 0$). probar
        convergencia, la recursividad $B(x+1, y) =
        \frac{x}{x+y}\,B(x, y)$ (integrar por partes), y
        $B(1, y) = \frac1y$.
  \item Demuestre que $x \mapsto B(x, y)$ es log-convexo (Hölder
        de nuevo), y aplicar Bohr-Mollerup a
        \[
        f(x) = \frac{B(x, y)\,\Gamma(x + y)}{\Gamma(y)}
        \]
        para concluir \emph{la fórmula de euler}: $B(x, y) =
        \dfrac{\Gamma(x)\Gamma(y)}{\Gamma(x+y)}$ --- sin doble
        integrales en cualquier lugar.
  \item Calcule $B(\frac12, \frac12)$ directamente (sustituya $t
        = \sin^2\theta$) y deduzca$\Gamma(\frac12) =
        \sqrt\pi$: la integral gaussiana de la Parte I, recuperada
        por pura convexidad. Compara las dos pruebas en una.
        frase cada uno.
  \item (duplicación de Legendre) Demuestre que
        \[
        g(x) = \frac{2^{x-1}}{\sqrt\pi}\,
        \Gamma\Bigl(\frac x2\Bigr)
        \Gamma\Bigl(\frac{x+1}2\Bigr)
        \]
        satisface las tres hipótesis de Bohr-Mollerup, y
        concluir $g = \Gamma$, es decir\ $\Gamma(2z) =
        \frac{2^{2z-1}}{\sqrt\pi}\,\Gamma(z)\,\Gamma(z +
        \tfrac12)$para todos los$z > 0$.
  \item Deduzca la forma cerrada $\Gamma\bigl(n + \tfrac12\bigr)
        = \dfrac{(2n)!}{4^n\,n!}\sqrt\pi$, y pruebe, por la
        Lema de pendiente aplicado a $\log\Gamma$ alrededor de grandes
        números enteros, las asintóticas
        \[
        \frac{\Gamma(n + \frac12)}{\Gamma(n)\,\sqrt n}
        \longrightarrow 1 .
        \]
  \item Combine las dos últimas preguntas en la pregunta central
        asintóticas binomiales
        \[
        \binom{2n}{n} \sim \frac{4^n}{\sqrt{\pi n}},
        \]
        y verificar numéricamente para $n = 10$ ($\binom{20}{10} =
        184756$, contra$4^{10}/\sqrt{10\pi} \approx
        187079$: relación$\approx 0.988$).
  \item (Síntesis) La constante $\sqrt\pi$ ahora ha aparecido como
        la integral gaussiana (Parte I), como $B(\frac12,
        \frac12)$ (pregunta 18), y duplicación interna
        (pregunta 19); la estimación binomial central
        anticipa tanto Stirling (\cref{ch:b3:product}'s
        problema del fin de semana) y de Moivre--Laplace. Mapear el
        Conexiones: qué declaraciones son equivalentes a cuáles,
        y ¿qué significa cada técnica --- diferenciación bajo
        la integral versus la convexidad --- contribuyen a que la
        ¿Otros no pueden?
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Three more dividends.}
\begin{enumerate}[resume]
  \item (Wallis, por Beta) Para $p > -1$, sustituya $t =
        \sin^2\theta$ para mostrar
        \[
        W_p = \int_0^{\pi/2}\sin^p\theta\,\dd\theta
        = \frac12\,B\Bigl(\frac{p+1}2, \frac12\Bigr),
        \]
        y deducir de la recursividad Beta (pregunta 16) que

        muestre $W_{2n+1}/W_{2n} \to 1$ apretando y concluya
        con \emph{El producto de Wallis.}
        \[
        \frac\pi2 = \lim_{n\to\infty}
        \prod_{k=1}^{n}\frac{4k^2}{4k^2 - 1} .
        \]
  \item (El gaussiano cumple con una frecuencia) Para $b \in \R$ configure
        \[
        \Phi(b) = \int_{-\infty}^{+\infty}
        \eu^{-x^2}\cos(2bx)\,\dd x .
        \]
        Demuestre que $\Phi$ es $\mathcal C^1$ en $\R$, que un
        la integración por partes produce la ecuación diferencial
        $\Phi'(b) = -2b\,\Phi(b)$, y concluir
        \[
        \Phi(b) = \sqrt\pi\,\eu^{-b^2} :
        \]
        el gaussiano se reproduce bajo esta transformación ---
        la identidad única en la que

  \item (integral de Frullani) Para $0 < a < b$, demuestre que
        \[
        \int_0^{+\infty}

        \]
        diferenciando en el parámetro $a$ (justificar el

        singularidad en $x = 0$?
\end{enumerate}
\end{problem}
