\chapter{Mapas conformes y el teorema de mapeo de Riemann}
\label{ch:b3:conformal}

Una biyección \omterm{def:b3:holomorphic:holo}{holomorfo} entre dos dominios transporta todos
análisis complejo de uno a otro: tales mapas ---
\emph{conforme}, porque conservan los ángulos --- son los
isomorfismos del mundo \omterm{def:b3:holomorphic:holo}{holomorfo}. Este capítulo clasifica
donde es posible la clasificación (el disco, el plano:
El lema de Schwarz es la clave, una desigualdad de proporciones sorprendentes.
potencia), construye la teoría de la compacidad de las familias \omterm{def:b3:holomorphic:holo}{holomorfo}
(Montel), y demuestra el teorema de existencia más profundo de la
Asunto: \emph{cada} \omterm{def:b3:conformal:simplyconnected}{simplemente conectado} subdominio adecuado de
$\C$, por muy irregular que sea su límite, es \omterm{def:b3:conformal:conformal}{equivalente conforme}
al disco unitario. Cerramos con \omterm{prop:b3:conformal:harmonicholo}{funciones armónicas} y el
\omterm{thm:b3:conformal:poisson}{núcleo venenoso}, resolviendo el problema de Dirichlet en el disco ---
La recompensa analítica de la geometría conforme. A lo largo,
$\mathbb D = D(0,1)$ y $\mathbb H = \{\operatorname{Im}z >
0\}$.

\section{Mapas conformes; Transformaciones de Möbius}

\begin{definition}\label{def:b3:conformal:conformal}
A \emph{aplicación conforme}\index{aplicación conforme} (o
biholomorfismo) entre \omterm{def:b3:topology:topology}{conjuntos abiertos} es una biyección \omterm{def:b3:holomorphic:holo}{holomorfo};
su inverso es automáticamente \omterm{def:b3:holomorphic:holo}{holomorfo}
(\cref{cor:b3:residues:openmapping}). Dos dominios son
\emph{equivalente conforme} si dicho mapa existe;
$\operatorname{Aut}(\Omega)$ denota el grupo de conforme
automapas. Donde $f' \neq 0$ --- en todas partes, para inyectivo $f$
(prueba de \cref{cor:b3:residues:openmapping}) --- el
el diferencial es la multiplicación por $f'(z) \neq 0$: a
similitud, por lo que las aplicaciones conformes preservan los ángulos entre curvas,
incluida la orientación.
\end{definition}

\begin{example}[transformaciones de Möbius]
\label{ex:b3:conformal:mobius}
Para $\bigl(\begin{smallmatrix}a & b\\ c &
d\end{smallmatrix}\bigr) \in GL_2(\C)$, el \emph{Möbius
transformation}\index{Möbius transformation} $z \mapsto
\frac{az + b}{cz + d}$ es conforme desde
$\C\setminus\{-d/c\}$ sobre $\C\setminus\{a/c\}$ (inverso de
del mismo tipo, de la matriz inversa; composición
corresponde al producto de la matriz). El \emph{Cayley
mapa}\index{Transformación de Cayley}
\[
\varphi(z) = \frac{z - \iu}{z + \iu}
\]
asigna $\mathbb H$ \omterm{def:b3:conformal:conformal}{conformemente} a $\mathbb D$: de hecho, $\abs{z
- \iu} < \abs{z + \iu}$exactamente cuando$z$está más cerca de$\iu$
que a $-\iu$, es decir\ $\operatorname{Im}z > 0$; lo inverso es
$w \mapsto \iu\frac{1 + w}{1 - w}$. Las transformaciones de Möbius envían el
círculos y líneas familiares consigo mismo (\cref{exo:b3:conformal:1}).
\end{example}

\begin{example}[La aplicación de Joukowski]\label{ex:b3:conformal:joukowski}
Más allá de Möbius, el \omterm{def:b3:conformal:conformal}{aplicación conforme} más útil del clásico
las matemáticas aplicadas son
\[
J(z) = \frac12\Bigl(z + \frac1z\Bigr) .
\]
En el exterior $\Omega = \{\abs z > 1\}$ del disco unitario,
$J$ es inyectivo: $J(z) = J(w)$ da $(z - w)(1 -
\frac1{zw}) = 0$y$\abs{zw} > 1$. Su derivada$J'(z) =
\frac12(1 - z^{-2})$desaparece sólo en$z = \pm1$, en el
límite: $J$ es una equivalencia conforme de $\Omega$ en
su imagen, que es $\C\setminus\intcc{-1}1$ --- la unidad
el círculo mismo se dobla dos a uno sobre el segmento
($J(\eu^{\iu\theta}) = \cos\theta$). Así, el exterior de un
\emph{segmento}, un plano de hendidura sin límite suave, es
\omterm{def:b3:conformal:conformal}{conformemente} el exterior de un disco: las esquinas no son obstáculo
a la equivalencia conforme, sólo a la suavidad de los límites. Imágenes
de círculos a través de $\pm1$ pero \omterm{ex:b3:groups:actions}{descentrado} tienen forma de perfil aerodinámico
curvas, y componiendo $J$ con mapas Möbius transportó el
flujo que pasa por un cilindro --- computable a mano --- al flujo
más allá de un ala: durante la primera mitad del siglo XX,
este ejemplo \emph{era} aerodinámica. También es la puerta a
Chebyshev: $J$ conjuga $z \mapsto z^n$ con Chebyshev
polinomio $T_n$ (\cref{pb:b3:hilbert:1}, Parte V), ya que
$J(z^n) = \cos(n\theta)$ cuando $z = \eu^{\iu\theta}$.
\end{example}

\section{Lema de Schwarz y grupos de automorfismo}

\begin{theorem}[lema negro]\label{thm:b3:conformal:schwarz}
Sea $f \colon \mathbb D \to \mathbb D$ \omterm{def:b3:holomorphic:holo}{holomorfo} con
$f(0) = 0$. Entonces\index{Lema negro}
\[
\abs{f(z)} \leq \abs z \ \ (z \in \mathbb D)
\qquad\text{and}\qquad \abs{f'(0)} \leq 1 ,
\]
y si $\abs{f(z_0)} = \abs{z_0}$ por un $z_0 \neq 0$, o
$\abs{f'(0)} = 1$, entonces $f(z) = \eu^{\iu\theta}z$ es un
rotación.
\end{theorem}

\begin{proof}
$g(z) = f(z)/z$ extiende \omterm{def:b3:holomorphic:holo}{holomorficamente} a $\mathbb D$ (el
la singularidad en $0$ es removible: $g$ está limitada cerca de $0$,
\cref{thm:b3:residues:riemanncw}; su valor en $0$ es
$f'(0)$). En $\abs z = r < 1$: $\abs g \leq \frac1r$, entonces por
el principio máximo (\cref{thm:b3:holomorphic:maximum})
$\abs g \leq \frac1r$ en $\bar D(0,r)$; sea $r \to 1$: $\abs
g \leq 1$en$\mathbb D$, que son ambas desigualdades. Igualdad
en un punto \omterm{def:b3:topology:interior}{interior} hace que $\abs g$ alcance un \omterm{def:b3:topology:interior}{interior}
máximo: $g$ constante del módulo $1$.
\end{proof}

\begin{theorem}[Automorfismos del disco]
\label{thm:b3:conformal:autdisc}
Para $a \in \mathbb D$, el \emph{Blaschke
factor}\index{factor de Blaschke}
\[
\varphi_a(z) = \frac{z - a}{1 - \bar a z}
\]
es un automorfismo de $\mathbb D$ intercambiando $a$ y $0$, con
$\varphi_a^{-1} = \varphi_{-a}$. Cada automorfismo de
$\mathbb D$ es $\eu^{\iu\theta}\varphi_a$ para $\theta
\in \R/2\pi\Z$,$a \in \mathbb D$ únicos.
\end{theorem}

\begin{proof}
En $\abs z = 1$: $\abs{1 - \bar az} = \abs{\bar z}\abs{1 -
\bar az} = \abs{\bar z - \bar a\abs z^2} = \abs{\bar z - \bar
a} = \abs{z - a}$, entonces$\abs{\varphi_a} = 1$ ahí; por el
principio máximo $\varphi_a(\mathbb D) \subseteq \bar{\mathbb
D}$, y la apertura coloca la imagen en$\mathbb D$. El \omterm{def:b3:galois:algebraic}{algebraico}
identidad $\varphi_{-a}\circ\varphi_a = \mathrm{id}$ (directa
cálculo) muestra biyectividad. Ahora vamos a $f \in
\operatorname{Aut}(\mathbb D)$y$a = f^{-1}(0)$:$g =
f\circ\varphi_{-a}$es un automorfismo que corrige$0$. negro
aplicado a $g$ y a $g^{-1}$: $\abs{g(z)} \leq \abs z$ y
$\abs{g^{-1}(w)} \leq \abs w$, entonces $\abs{g(z)} = \abs z$:
rotación, $g = \eu^{\iu\theta}\,\mathrm{id}$, es decir\ $f =
\eu^{\iu\theta}\varphi_a$. Unicidad:$a = f^{-1}(0)$ y
$\theta$ de evaluación tipo $f'$ (o de $f(0) =
-\eu^{\iu\theta}a$ y un valor más).
\end{proof}

\begin{theorem}[Automorfismos del avion]
\label{thm:b3:conformal:autplane}
$\operatorname{Aut}(\C) = \{z \mapsto az + b : a \in \C^*,\ b
\in \C\}$. En consecuencia,$\C$y$\mathbb D$ no son
\omterm{def:b3:conformal:conformal}{equivalente conforme}.
\end{theorem}

\begin{proof}
Dejemos $f \in \operatorname{Aut}(\C)$ y consideremos $g(z) =
f(1/z)$en$\C^*$: un \omterm{def:b3:holomorphic:holo}{función holomorfa} con un \omterm{def:b3:residues:singularities}{aislado
singularidad} en $0$. Si fuera imprescindible,
Casorati--Weierstrass (\cref{thm:b3:residues:riemanncw}) sería
hacer que $g\bigl(D(0,\varepsilon)\setminus\{0\}\bigr)$ sea denso,
mientras que $f(D(0, 1))$ está abierto y separado de él ($f$
inyectivo: los dos conjuntos son imágenes de conjuntos disjuntos) ---
imposible para un \omterm{def:b3:topology:interior}{conjunto denso} y un \omterm{def:b3:topology:topology}{conjunto abierto} no vacío. Entonces $0$ es
un poste o extraíble para $g$, es decir, \ $\abs{f(z)}$ tiene como máximo
crecimiento polinómico: $f$ es un polinomio
(\cref{exo:b3:holomorphic:4}(a)). Grado de fuerzas de inyectividad
$1$: un polinomio de grado superior tiene una raíz múltiple de
$f - c$ en algún lugar ($f'$ desaparece) o varios
preimágenes (d'Alembert--Gauss, \cref{pb:b3:holomorphic:1});
De cualquier manera, la inyectividad falla. Finalmente, un $\C \to
\mathbb D$ conforme sería una función completa acotada: constante
(Liouville) --- no hay equivalencia.
\end{proof}

\section{teorema de montel}

\begin{theorem}[montel]\label{thm:b3:conformal:montel}
Sea $\mathcal F \subseteq \mathcal H(\Omega)$ \emph{localmente
delimitado}: cada punto tiene un \omterm{def:b3:topology:topology}{entorno} en el que $\sup_{f\in
\mathcal F}\sup\abs f < \infty$. Entonces cada secuencia de
$\mathcal F$ tiene una subsecuencia que converge uniformemente en todos
\omterm{def:b3:topology:compact}{compacto} subconjuntos de $\Omega$ (a un \omterm{def:b3:holomorphic:holo}{holomorfo}
límite).\index{teorema de montel}
\end{theorem}

\begin{proof}
Local \omterm{def:b3:complete:equicontinuous}{equicontinuidad}: si $\abs f \leq M$ en $D(a, 2r)
\subseteq \Omega$para todos los$f \in \mathcal F$, el Cauchy
la fórmula da, para $z, z' \in D(a, r)$,
\[
\abs{f(z) - f(z')} = \frac{\abs{z - z'}}{2\pi}
\Bigl|\int_{C_{2r}}\frac{f(w)\,\dd w}{(w-z)(w-z')}\Bigr|
\leq \frac{\abs{z - z'}\,2\pi\cdot2r\,M}{2\pi\,r^2}
= \frac{2M}{r}\,\abs{z - z'} :
\]
un Lipschitz uniforme cota. Escape $\Omega$ por \omterm{def:b3:topology:compact}{compactos} $K_m$;
cada $K_m$ está cubierto por un número finito de discos de este tipo, por lo que
$\mathcal F$ está uniformemente delimitado y \omterm{def:b3:complete:equicontinuous}{equicontinuo} en
$K_m$: Arzelà--Ascoli (\cref{thm:b3:complete:ascoli})
extrae una subsecuencia que converge uniformemente en $K_m$;
diagonalizar sobre $m$. El límite es \omterm{def:b3:holomorphic:holo}{holomorfo} por
\cref{thm:b3:holomorphic:weierstrassconv}.
\end{proof}

\section{El teorema de mapeo de Riemann}

\begin{definition}\label{def:b3:conformal:simplyconnected}
Un \omterm{def:b3:topology:connected}{conectado} $\Omega \subseteq \C$ abierto es \emph{simplemente
conectado}\index{dominio simplemente conectado} (en el formato homológico).
sentido, suficiente para todos nuestros propósitos) si
$\operatorname{Ind}_\gamma(w) = 0$ para cada ciclo $\gamma$ en
$\Omega$ y cada $w \notin \Omega$ --- ``sin ciclo de
$\Omega$ rodea un agujero''. Por el teorema global de Cauchy
(\cref{thm:b3:residues:globalcauchy}) y
\cref{prop:b3:holomorphic:primitive}, en tal $\Omega$
\emph{every \omterm{def:b3:holomorphic:holo}{función holomorfa} has a primitive}; por lo tanto cada
$f \in \mathcal H(\Omega)$ libre de cero tiene un \omterm{def:b3:holomorphic:holo}{holomorfo}
logaritmo ($\exp\circ$(primitiva de $f'/f$), ajustado por un
constante, como $(f\eu^{-L})' = 0$) y \omterm{def:b3:holomorphic:holo}{holomorfo} $n$-ésimas raíces
$\eu^{L/n}$.\index{logaritmo holomorfo}
\end{definition}

\begin{theorem}[Teorema de mapeo de Riemann]
\label{thm:b3:conformal:rmt}
Cada \omterm{def:b3:conformal:simplyconnected}{simplemente conectado} abierto $\Omega \subsetneq \C$, $\Omega
\neq \varnothing$, es \omterm{def:b3:conformal:conformal}{equivalente conforme} a$\mathbb D$;
dado $z_0 \in \Omega$, hay un $f \colon
\Omega \to \mathbb D$conforme único con$f(z_0) = 0$y$f'(z_0) >
0$.\index{Teorema de mapeo de Riemann}
\end{theorem}

\begin{proof}
\emph{Paso 0: la familia no está vacía.} Elija $b \notin \Omega$:
$z - b$ no tiene cero en $\Omega$, por lo que tiene un \omterm{def:b3:holomorphic:holo}{holomorfo}
raíz cuadrada $h$ ($h^2 = z - b$). $h$ es inyectivo ($h(z) =
h(z')$cuadra a$z = z'$), y si$w \in h(\Omega)$entonces$-w
\notin h(\Omega)$($h(z) = -h(z')$también cuadra a$z = z'$,
dando $w = -w = 0$, imposible ya que $h$ no tiene cero). desde
$h(\Omega)$ está abierto, contiene un disco $D(h(z_0), \rho)$;
entonces $D(-h(z_0), \rho) \cap h(\Omega) = \varnothing$, es decir\
$\abs{h(z) + h(z_0)} \geq \rho$ por cada $z \in \Omega$.
Por lo tanto
\[
g(z) = \frac{\rho}{2\,\bigl(h(z) + h(z_0)\bigr)}
\]
es \omterm{def:b3:holomorphic:holo}{holomorfo}, inyectivo (una aplicación Möbius compuesto con el
inyectivo $h$), con $\abs g \leq \frac12 < 1$. Componiendo con un \omterm{thm:b3:conformal:autdisc}{factor de Blaschke}
(\cref{thm:b3:conformal:autdisc}) para mover $g(z_0)$ a $0$, el
familia
\[
\mathcal F = \{f \colon \Omega \to \mathbb D \text{
holomorphic, injective, } f(z_0) = 0\}
\]
no está vacío.

\emph{Paso 1: un elemento extremo.} Vamos $s =
\sup_{\mathcal F}\abs{f'(z_0)} \in \intoc0{+\infty}$($> 0$:
Los miembros son inyectivos, por lo que $f'(z_0) \neq 0$). Tome $f_n \in
\mathcal F$con$\abs{f_n'(z_0)} \to s$: la familia es
delimitado por $1$, por lo que Montel (\cref{thm:b3:conformal:montel})
extrae $f_n \to f$ uniformemente en \omterm{def:b3:topology:compact}{compactos}; $f$ es
\omterm{def:b3:holomorphic:holo}{holomorfo}, $f(z_0) = 0$, $\abs{f'(z_0)} = s$
(\cref{thm:b3:holomorphic:weierstrassconv} para el
derivados), en particular $f$ no es constante; $f$ es
inyectivo de Hurwitz (\cref{exo:b3:residues:8}(b)), y
$f(\Omega) \subseteq \bar{\mathbb D}$, por lo tanto $\subseteq
\mathbb D$(asignación abierta). Entonces$f \in \mathcal F$ alcanza el
supremo: $s < \infty$.

\emph{Paso 2: la aplicación extremo está en marcha.} Supongamos $a \in
\mathbb D\setminus f(\Omega)$. El transporte Blaschke
$\varphi_a\circ f$ está libre de cero en el \omterm{def:b3:conformal:simplyconnected}{simplemente conectado}
$\Omega$: tiene una raíz cuadrada \omterm{def:b3:holomorphic:holo}{holomorfo} $F$ (con
$F(\Omega) \subseteq \mathbb D$, como $\abs F^2 =
\abs{\varphi_a\circ f} < 1$), inyectivo (cuadrados
distinguir). Normalizar: $G = \varphi_{F(z_0)}\circ F \in
\mathcal F$. Deshacer:$f = \varphi_{-a}\circ s_2 \circ
\varphi_{-F(z_0)}\circ G$donde$s_2(w) = w^2$; la aplicación$\Psi
= \varphi_{-a}\circ s_2\circ\varphi_{-F(z_0)} \colon \mathbb D
\to \mathbb D$es \omterm{def:b3:holomorphic:holo}{holomorfo} con$\Psi(0) = f(z_0) = 0$ y
es \emph{no} una rotación (no es inyectiva: $s_2$ no lo es).
Lema de Schwarz (caso estricto): $\abs{\Psi'(0)} < 1$, y el
la regla de la cadena $f = \Psi\circ G$ da $\abs{f'(z_0)} =
\abs{\Psi'(0)}\,\abs{G'(z_0)} < \abs{G'(z_0)}$ ---
maximalidad contradictoria (nota $G \in \mathcal F$). Por lo tanto $f$
está en: una equivalencia conforme.

\emph{Paso 3: normalización y unicidad.} Multiplica $f$ por
$\eu^{-\iu\arg f'(z_0)}$ para hacer $f'(z_0) > 0$ (esto permanece en
$\mathcal F$). Si $f_1, f_2$ funcionan, $\psi =
f_2\circ f_1^{-1} \in \operatorname{Aut}(\mathbb D)$ soluciona
$0$ con $\psi'(0) = f_2'(z_0)/f_1'(z_0) > 0$; por
\cref{thm:b3:conformal:autdisc} $\psi$ es una rotación
$\eu^{\iu\theta}$ con $\eu^{\iu\theta} > 0$: $\psi =
\mathrm{id}$.
\end{proof}

\begin{remark}\label{rem:b3:conformal:rmtscope}
El teorema es una declaración de existencia pura de sorprendente
alcance: un cuadrado, un semiplano, el complemento de una rendija, el
región entre dos círculos tangentes, un dominio de límite fractal
--- todos los \omterm{def:b3:conformal:conformal}{conformemente} idénticos a $\mathbb D$. lo que no hace
dar: cualquier fórmula (las aplicaciones explícitos son la excepción:
\cref{exo:b3:conformal:5}), comportamiento límite (una teoría más profunda
--- Teorema de Carathéodory --- lo maneja), o unicidad
de extensión a $\C$ o multiplicar dominios \omterm{def:b3:topology:connected}{conectado}: el
el anillo $\{1 < \abs z < 2\}$ es \emph{no} \omterm{def:b3:conformal:conformal}{conformemente} a
disco perforado y anillos de diferentes relaciones de radio son
desigual (un hecho realmente más difícil).
\end{remark}

\section{Funciones armónicas y el núcleo de Poisson.}

\begin{proposition}\label{prop:b3:conformal:harmonicholo}
Sea $\Omega$ \omterm{def:b3:conformal:simplyconnected}{simplemente conectado} y $u \colon \Omega \to \R$
armónico\index{función armónica} ($\mathcal C^2$ con
$\Delta u = u_{xx} + u_{yy} = 0$). Luego $u =
\operatorname{Re}F$para un \omterm{def:b3:holomorphic:holo}{holomorfo}$F$, único hasta un
constante imaginaria. En consecuencia $u$ es $\mathcal C^\infty$,
satisface la propiedad del valor medio y obedece al máximo
principio (sin extremo estricto \omterm{def:b3:topology:interior}{interior} a menos que sea constante).
\end{proposition}

\begin{proof}
$g = u_x - \iu u_y$ satisface el \omterm{prop:b3:holomorphic:cauchyriemann}{Ecuaciones de Cauchy-Riemann}
($P = u_x$, $Q = -u_y$: $P_x = u_{xx} = -u_{yy} = Q_y$ y
$P_y = u_{xy} = u_{yx} = -Q_x$) con parciales \omterm{def:b3:topology:continuity}{continuo}: $g$
es \omterm{def:b3:holomorphic:holo}{holomorfo} (\cref{prop:b3:holomorphic:cauchyriemann}; el
$\R$ (la diferenciabilidad se deriva de $\mathcal C^1$). Vamos $F_0$
ser un primitivo (conectividad simple,
\cref{def:b3:conformal:simplyconnected}); entonces
$\operatorname{Re}F_0$ tiene gradiente $(u_x, u_y)$
($F_0' = g$ desempaqueta exactamente eso vía Cauchy--Riemann para
$F_0$), por lo que $u - \operatorname{Re}F_0$ es constante ($\Omega$
\omterm{def:b3:topology:connected}{conectado}): ajustar $F = F_0 + c$. la transferencia de propiedades
de \cref{thm:b3:holomorphic:maximum} y
\cref{exo:b3:holomorphic:10} (para el principio máximo
aplicado al propio $u$, use $\eu^{F}$ como allí).
\end{proof}

\begin{theorem}[fórmula de Poisson; El problema de Dirichlet en el
discoteca]\label{thm:b3:conformal:poisson}
Para $0 \leq r < 1$ defina \emph{Poison
núcleo}\index{núcleo venenoso}
\[
P_r(\theta) = \sum_{n\in\Z}r^{\abs n}\eu^{\iu n\theta}
= \frac{1 - r^2}{1 - 2r\cos\theta + r^2} \;>\; 0 .
\]
Sea $g \colon \partial\mathbb D \to \R$ \omterm{def:b3:topology:continuity}{continuo} y
configurado, para $z = r\eu^{\iu\varphi} \in \mathbb D$,
\[
u(z) = \frac1{2\pi}\int_0^{2\pi}
P_r(\varphi - t)\,g(\eu^{\iu t})\,\dd t .
\]
Entonces $u$ es armónico en $\mathbb D$ y extiende a \omterm{def:b3:topology:continuity}{continuamente}.
a $\bar{\mathbb D}$ con valores límite $g$: el único tal
\omterm{prop:b3:conformal:harmonicholo}{función armónica}.\index{problema de dirichlet}
\end{theorem}

\begin{proof}
\emph{Identidades del núcleo}: sumando dos series geométricas,
\[
\sum_{n\in\Z}r^{\abs n}\eu^{\iu n\theta}
= \operatorname{Re}\frac{1 + r\eu^{\iu\theta}}{1 -
r\eu^{\iu\theta}}
= \frac{1 - r^2}{\abs{1 - r\eu^{\iu\theta}}^2},
\]
cuál es el cociente mostrado; La positividad es clara y
$\frac1{2\pi}\int_0^{2\pi}P_r = 1$ (solo sobrevive $n = 0$).

\emph{Armonicidad}: con $z = r\eu^{\iu\varphi}$,
\[
u(z) = \operatorname{Re}\biggl[\frac1{2\pi}\int_0^{2\pi}
\frac{\eu^{\iu t} + z}{\eu^{\iu t} - z}\,
g(\eu^{\iu t})\,\dd t\biggr],
\]
(el núcleo entre corchetes tiene la parte real $P_r(\varphi - t)$:
calcular), y el soporte es \omterm{def:b3:holomorphic:holo}{holomorfo} en $z$ en $\mathbb D$
(\cref{exo:b3:holomorphic:7}): $u$ es la parte real de un
\omterm{def:b3:holomorphic:holo}{función holomorfa}, por lo tanto armónico.

\emph{Valores límite}: $P_r(\cdot)$ es un aproximado
identidad como $r \to 1^-$: masa $1$, y para $\delta \leq
\abs\theta \leq \pi$,$P_r(\theta) \leq \frac{1 - r^2}{1 -
2r\cos\delta + r^2} \to 0$ uniformemente. La división estándar
(\omterm{def:b3:topology:continuity}{continuidad} de $g$ cerca de $\eu^{\iu\varphi_0}$, acotación
en otro lugar) da $u(r\eu^{\iu\varphi}) \to
g(\eu^{\iu\varphi_0})$como$r\eu^{\iu\varphi} \to
\eu^{\iu\varphi_0}$, uniformemente en el punto límite: el
la extensión es \omterm{def:b3:topology:continuity}{continuo}. \emph{Unicidad}: la diferencia de
dos soluciones son armónicas en $\mathbb D$, \omterm{def:b3:topology:continuity}{continuo} en el
\omterm{def:b3:topology:interior}{cierre}, cero en el límite: por el principio de máximo
(aplicado a $\pm$ la diferencia), desaparece.
\end{proof}

\section{Problema: el teorema del área y el cuarto de Koebe
teorema}

\begin{problem}[{Problema de fin de semana --- ¿cuánto debe
¿Cubierta de mapa univalente?}]\label{pb:b3:conformal:1}
Una función \emph{univalente} es una inyección \omterm{def:b3:holomorphic:holo}{holomorfo}. el
funciones univalentes normalizadas en el disco,
\[
\mathcal S = \bigl\{f \in \mathcal H(\mathbb D) \text{
injective},\ f(z) = z + a_2z^2 + a_3z^3 + \cdots\bigr\},
\]
están rígidamente restringidos: demostramos la desigualdad de Bieberbach
$\abs{a_2} \leq 2$ y deducir el \emph{Barrio de Koebe
teorema}: la imagen de cualquier $f \in \mathcal S$ contiene el
disco $D(0, \frac14)$ --- la constante universal aguda de
geometría conforme.\index{Teorema del cuarto de Koebe}\index{zona
teorema}

\medskip
\noindent\textbf{Parte I --- The area theorem.} Sea $g(w) = w
+ b_0 + \frac{b_1}w + \frac{b_2}{w^2} + \cdots$ \omterm{def:b3:holomorphic:holo}{holomorfo}
y \emph{inyectivo} en $\{\abs w > 1\}$.
\begin{enumerate}
  \item Para $\rho > 1$, sea $A_\rho$ el área (\omterm{def:b3:measure:lebesgueouter}{Lebesgue
        medida}) del conjunto \omterm{def:b3:topology:compact}{compacto} $K_\rho = \C\setminus
        g(\{\abs w > \rho\})$, la región encerrada por el
        suave curva de Jordan $g(C_\rho)$. Usando el área
        Fórmula del teorema de Green-Riemann del volumen del año 2
        --- el área cerrada es $\frac1{2\iu}
        \oint\bar\zeta\,\dd\zeta$ a lo largo del lado positivo
        límite orientado --- demuestre que
        \[
        A_\rho = \frac{1}{2\iu}\int_{C_\rho}
        \overline{g(w)}\,g'(w)\,\dd w
        = \pi\Bigl(\rho^2 -
        \sum_{n\geq1}n\,\abs{b_n}^2\rho^{-2n}\Bigr) :
        \]
        sustituya el \omterm{thm:b3:residues:laurent}{serie laurent} de $\bar g$ y $g'$ en
        $C_\rho$ e integrar término por término (normal
        convergencia; sólo los productos de frecuencia cero
        sobrevivir).
  \item Dejemos $\rho \to 1^+$ y concluyamos el \emph{zona
        teorema}:
        \[
        \sum_{n\geq1}n\,\abs{b_n}^2 \;\leq\; 1 .
        \]
        En particular $\abs{b_1} \leq 1$. ¿Cuándo es $\abs{b_1}
        = 1$?
\end{enumerate}

\medskip
\noindent\textbf{Parte II --- Bieberbach's $\abs{a_2} \leq
2$.} Vamos$f = z + a_2z^2 + \cdots \in \mathcal S$.
\begin{enumerate}[resume]
  \item Demuestre que $f(z^2)/z^2$ es \omterm{def:b3:holomorphic:holo}{holomorfo} y está libre de ceros
        en $\mathbb D$, y admite una raíz cuadrada \omterm{def:b3:holomorphic:holo}{holomorfo}
        $\varphi$ con $\varphi(0) = 1$; configure $h(z) =
        z\varphi(z^2)$, de modo que$h(z)^2 = f(z^2)$. mostrar eso
        $h$ es una función univalente \emph{extraño} en $\mathbb
        D$con expansión$h(z) = z + \frac{a_2}2z^3 +
        \cdots$. \emph{(Inyectividad:$h(z)^2 = h(z')^2$
        fuerzas $z^2 = z'^2$; usa la rareza para terminar.)}
  \item Aplicar el teorema del área a $g(w) = 1/h(1/w) = w -
        \frac{a_2}{2w} + \cdots$en$\{\abs w > 1\}$
        (verifique la univalencia y la expansión) y concluya
        $\abs{a_2} \leq 2$.
  \item Demuestre que el \emph{función de koebe}
        \[
        k(z) = \frac{z}{(1 - z)^2} = \sum_{n\geq1}n\,z^n
        \]
        pertenece a $\mathcal S$, tiene $a_2 = 2$ y mapas
        $\mathbb D$ en $\C\setminus\intoc{-\infty}{-\frac14}$
        \emph{(write $k = \frac14\bigl[\bigl(\frac{1+z}{1 -
        z}\bigr)^2 - 1\bigr]$ and track the images)}: todos
        Las desigualdades futuras son marcadas.
\end{enumerate}

\medskip
\noindent\textbf{Parte III --- The quarter theorem.}
\begin{enumerate}[resume]
  \item Dejemos $f \in \mathcal S$ y $c \notin f(\mathbb D)$.
        mostrar eso
        \[
        F(z) = \frac{c\,f(z)}{c - f(z)}
        \]
        pertenece a $\mathcal S$ y calcula su segundo
        coeficiente: $A_2 = a_2 + \frac1c$.
  \item Aplicar Bieberbach tanto a $f$ como a $F$: concluir
        $\abs{\frac1c} \leq \abs{A_2} + \abs{a_2} \leq 4$,
        es decir,\ $\abs c \geq \frac14$. \emph{Every omitted value
        has modulus $\geq \frac14$}: $f(\mathbb D) \supseteq
        D(0, \frac14)$ --- Teorema del cuarto de Koebe. comprobar
        nitidez en la función de Koebe.
  \item Deduzca una estimación cuantitativa del mapa de Riemann: si
        $\varphi \colon \Omega \to \mathbb D$ es el Riemann
        mapa de \cref{thm:b3:conformal:rmt} en $z_0$, luego
        \[
        \frac{d\bigl(z_0, \partial\Omega\bigr)}{4}
        \;\leq\; \frac1{\varphi'(z_0)} \;\leq\;
        4\,d\bigl(z_0, \partial\Omega\bigr)
        \]
        --- demostrar al menos la desigualdad izquierda aplicando
        Koebe a $\varphi^{-1}$ adecuadamente normalizado, y el
        derecho de Schwarz aplicado a $\varphi$ en el disco
        $D(z_0, d) \subseteq \Omega$.
\end{enumerate}

\medskip
\noindent\textbf{Parte IV --- Perspective.}
\begin{enumerate}[resume]
  \item Bieberbach conjeturó (1916) $\abs{a_n} \leq n$ para
        todos $n$, con igualdad solo para rotaciones del
        función de Koebe; de Branges lo demostró en 1985. Verificar
        la conjetura a mano para la función de Koebe y su
        rotaciones $\eu^{-\iu\theta}k(\eu^{\iu\theta}z)$. entonces
        impulsar la expansión de la pregunta 4 un término más: escritura
        $h(z) = z + \frac{a_2}2z^3 + c_5z^5 + \cdots$, mostrar
        $c_5 = \frac{a_3}2 - \frac{a_2^2}8$ y
        \[
        g(w) = w - \frac{a_2}{2}\,w^{-1} +
        \Bigl(\frac{3a_2^2}8 - \frac{a_3}2\Bigr)w^{-3} +
        \cdots ,
        \]
        entonces el teorema del área produce la desigualdad refinada
        $\bigl|\frac{a_2}2\bigr|^2 + 3\bigl|\frac{3a_2^2}8 -
        \frac{a_3}2\bigr|^2 \leq 1$. Compruébalo en el Koebe
        función ($a_2 = 2$, $a_3 = 3$).
\end{enumerate}

\medskip
\noindent\textbf{Part V --- The distortion theorem.}
La desigualdad de Bieberbach, transportada alrededor del disco por
automorfismos, controla $f'$ en todas partes. Reparar $f \in \mathcal
S$.
\begin{enumerate}[resume]
  \item (transformada de Koebe) Para $z_0 \in \mathbb D$, deje
        $\varphi(z) = \frac{z + z_0}{1 + \bar z_0z}$, un disco
        automorfismo (\cref{thm:b3:conformal:autdisc}) con
        $\varphi(0) = z_0$. mostrar eso
        \[
        F(z) = \frac{f(\varphi(z)) - f(z_0)}
        {f'(z_0)\,\bigl(1 - \abs{z_0}^2\bigr)}
        \]
        pertenece a $\mathcal S$ \emph{(univalence is
        inherited; compute $\varphi'(0) = 1 - \abs{z_0}^2$
        and check the normalization; recall $f' \neq 0$ for
        injective $f$,
        \cref{def:b3:conformal:conformal})}.
  \item Calcular el segundo coeficiente $A_2 = \frac12F''(0)$
        de $F$:
        \[
        A_2 = \frac12\Bigl[\bigl(1 - \abs{z_0}^2\bigr)
        \frac{f''(z_0)}{f'(z_0)} - 2\bar z_0\Bigr] ,
        \]
        y deducir de Bieberbach (pregunta 4), para $z =
        r\eu^{\iu\theta}$, el \emph{fundamental
        desigualdad}:
        \[
        \Bigl|\,z\,\frac{f''(z)}{f'(z)} -
        \frac{2r^2}{1 - r^2}\Bigr|
        \;\leq\; \frac{4r}{1 - r^2} .
        \]
  \item Extrae la parte real:
        \[
        \frac{2r^2 - 4r}{1 - r^2} \;\leq\;
        \operatorname{Re}\Bigl(z\,\frac{f''(z)}{f'(z)}\Bigr)
        \;\leq\; \frac{2r^2 + 4r}{1 - r^2} .
        \]
  \item Muestra que $\frac{\dd}{\dd t}\log\bigl|
        f'(t\eu^{\iu\theta})\bigr| = \frac1t
        \operatorname{Re}\bigl(z\frac{f''(z)}{f'(z)}\bigr)$
        en $z = t\eu^{\iu\theta}$ \emph{(for a nonvanishing
        $\mathcal C^1$ function $g$ of a real variable,
        $\frac{\dd}{\dd t}\log\abs g =
        \operatorname{Re}(g'/g)$)} e integrar la pregunta
        límites de 12 a lo largo del rayo para obtener el
        \emph{teorema de distorsión}:
        \[
        \frac{1 - r}{(1 + r)^3} \;\leq\; \abs{f'(z)}
        \;\leq\; \frac{1 + r}{(1 - r)^3},
        \qquad \abs z = r .
        \]
  \item Deducir el \emph{teorema de crecimiento}:
        \[
        \frac{r}{(1 + r)^2} \;\leq\; \abs{f(z)} \;\leq\;
        \frac{r}{(1 - r)^2},
        \qquad \abs z = r
        \]
        \emph{(upper bound: integrate $f'$ on the segment
        $[0, z]$; lower bound: if $\abs{f(z)} < \frac14$,
        the segment $[0, f(z)]$ lies in $f(\mathbb D)$ by
        question 7; pull it back by $f^{-1}$ ---
        \omterm{def:b3:holomorphic:holo}{holomorfo} by
        \cref{cor:b3:residues:openmapping} --- and bound
        $\abs{f(z)} = \int_\gamma\abs{f'(\zeta)}
        \,\abs{\dd\zeta} \geq \int_0^r\frac{1 - t}{(1 +
        t)^3}\,\dd t$, using$\abs{\dd\zeta} \geq
        \dd\abs\zeta$)}.
  \item Verificar que la función de Koebe logra la igualdad en
        los cuatro límites, en $z = r$ para los superiores y
        $z = -r$ para los inferiores: $k$ es simultáneamente
        el más en expansión y, en la antípoda, el más
        miembro contratante de $\mathcal S$.
\end{enumerate}

\medskip
\noindent\textbf{Part VI --- Extremal rigidity.} En cada
desigualdad hasta ahora, la igualdad identifica la función de Koebe hasta
a la rotación. Lo probamos y luego cosechamos.
\begin{enumerate}[resume]
  \item Supongamos que $f \in \mathcal S$ tiene $\abs{a_2} = 2$.
        Persigue la igualdad a través de las preguntas 2 a 4: el área
        teorema fuerzas $g(w) = w + b_0 + \eu^{\iu\alpha}/w$;
        la rareza de $h$ hace que $g$ sea impar, por lo que $b_0 = 0$; invertir
        para encontrar $h$, luego $f$ y concluir que
        \[
        f(z) = \eu^{-\iu\theta}k\bigl(\eu^{\iu\theta}z\bigr)
        \quad\text{with } \eu^{\iu\theta} =
        -\eu^{\iu\alpha} :
        \]
        las rotaciones de la función de Koebe son las únicas
        miembros de $\mathcal S$ con $\abs{a_2} = 2$.
  \item Demuestre que si $f \in \mathcal S$ omite un valor $c$
        con $\abs c = \frac14$ exactamente, entonces $f$ es un
        rotación de la función de Koebe e identificar $c =
        -\eu^{-\iu\theta}/4$ \emph{(trace equality through
        question 7's chain $4 = \abs{1/c} = \abs{A_2 - a_2}
        \leq \abs{A_2} + \abs{a_2} \leq 4$)}: el trimestre
        La constante del teorema se alcanza sólo por el extremo.
        familia.
  \item (Coeficientes baratos) Combina el crecimiento
        teorema con el \omterm{thm:b3:holomorphic:analytic}{Estimaciones de Cauchy}
        (\cref{thm:b3:holomorphic:analytic}) en el círculo
        $\abs z = 1 - \frac1n$ para demostrar
        \[
        \abs{a_n} \;\leq\; \eu\,n^2 \qquad (n \geq 2) .
        \]
        (De Branges, 1985: $\abs{a_n} \leq n$; el factor
