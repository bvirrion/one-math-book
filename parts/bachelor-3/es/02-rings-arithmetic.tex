\section{Ideales, cocientes y el teorema del isomorfismo}

\begin{definition}\label{def:b3:rings:ideal}
Un \emph{ideal}\index{ideal} $I$ de un anillo $A$ es un aditivo
subgrupo tal que $AI \subseteq I$. El \emph{cociente
anillo}\index{anillo cociente} $A/I$ es el \omterm{thm:b3:groups:quotient}{grupo cociente} $(A, +)/I$
con la multiplicaciĂłn $(a + I)(b + I) = ab + I$: bien definido,
desde que se cambiĂł $a$ a $a + x$, $b$ a $b + y$ ($x, y \in I$)
cambia $ab$ por $ay + xb + xy \in I$. La proyecciĂłn $\pi \colon A
\to A/I$es un morfismo de anillo sobreyectivo con nĂşcleo$I$ y nĂşcleos
de los morfismos de anillo son exactamente los ideales.
\end{definition}

\begin{theorem}[Primer teorema del isomorfismo]\label{thm:b3:rings:firstiso}
Si $f \colon A \to B$ es un morfismo de anillo, entonces $\bar f\colon
A/\ker f \to \operatorname{im} f$,$a + \ker f \mapsto f(a)$, es un
isomorfismo de anillo. De manera mĂĄs general, $f$ factoriza a travĂŠs de $A/I$ para cualquier
\omterm{def:b3:rings:ideal}{ideal} $I \subseteq \ker f$. Los \omterm{def:b3:rings:ideal}{ideales} de $A/I$ son los $J/I$ para
$J \supseteq I$ y \omterm{def:b3:rings:ideal}{ideal} de $A$ (teorema de correspondencia).
\end{theorem}

\begin{proof}
En cuanto a grupos (\cref{thm:b3:groups:firstiso,%
thm:b3:groups:correspondence}), destacando que todos las aplicaciones a la vista tambiĂŠn
respetar productos: $\bar f$ estĂĄ bien definido, biyectivo en el
imagen y multiplicativo; la correspondencia $J \mapsto J/I$,
$\bar J \mapsto \pi^{-1}(\bar J)$ conserva \omterm{def:b3:rings:ideal}{ideales} en ambos
direcciones porque $\pi$ es un morfismo de anillo sobreyectivo.
\end{proof}

\begin{definition}\label{def:b3:rings:primemaximal}
Sea $I \subsetneq A$ un \omterm{def:b3:rings:ideal}{ideal} adecuado. $I$ es
\emph{principal}\index{ideal principal} si $ab \in I \Rightarrow a \in I$
o $b \in I$; $I$ es \emph{mĂĄximo}\index{ideal mĂĄximo} si no
\omterm{def:b3:rings:ideal}{ideal} se encuentra estrictamente entre $I$ y $A$.
\end{definition}

\begin{proposition}\label{prop:b3:rings:primemaximal}
$I$ es primo $\iff$ $A/I$ es un dominio integral; $I$ es \omterm{def:b3:rings:primemaximal}{mĂĄximo}
$\iff$ $A/I$ es un campo. En particular \omterm{def:b3:rings:primemaximal}{ideales maximales} son primos.
\end{proposition}

\begin{proof}
Escriba $\bar a$ para las clases en $A/I$. ``$I$ principal'' traduce
textualmente a ``$\bar a\bar b = 0 \Rightarrow \bar a = 0$ o $\bar b
= 0$'', y$A/I \neq 0$a$I \neq A$: esa es la definiciĂłn de
un dominio. Para maximalidad, utilice el teorema de correspondencia: no \omterm{def:b3:rings:ideal}{ideal}
estrictamente entre $I$ y $A$ $\iff$ $A/I$ no tiene \omterm{def:b3:rings:ideal}{ideal} mĂĄs que
$0$ y $\iff$ $A/I$ es un campo --- para el Ăşltimo paso: en un
campo el Ăşnico \omterm{def:b3:rings:ideal}{ideales} es $0$ y todo (un \omterm{def:b3:rings:ideal}{ideal} que contiene
$x \ne 0$ contiene $x^{-1}x = 1$); por el contrario, si cada $x$ distinto de cero
genera la unidad \omterm{def:b3:rings:ideal}{ideal}, luego $xy = 1$ para algunos $y$. Los campos son
dominios, por lo que \omterm{def:b3:rings:primemaximal}{ideales maximales} son primos.
\end{proof}

\begin{example}\label{ex:b3:rings:primemaximal}
En $\Z$: los \omterm{def:b3:rings:primemaximal}{ideales primordiales} son $(0)$ y los $(p)$, $p$ primos; el
Los \omterm{def:b3:rings:primemaximal}{mĂĄximo} son los $(p)$ ($\Z/p\Z = \mathbb F_p$ es un campo,
$\Z/(0) = \Z$ no lo es). En $K[X, Y]$: $(X) \subsetneq (X, Y)$ son
ambos primos ($K[X,Y]/(X) \cong K[Y]$, un dominio;
$K[X,Y]/(X,Y) \cong K$, un campo), por lo que $(X)$ es primo pero no
\omterm{def:b3:rings:primemaximal}{mĂĄximo}.
\end{example}

Para garantizar que \omterm{def:b3:rings:primemaximal}{ideales maximales} \emph{existir} en toda su generalidad,
Necesitamos un principio de teorĂ­a de conjuntos. Un conjunto parcialmente ordenado es
\emph{inductivo} si cada subconjunto totalmente ordenado (\emph{cadena})
tiene un lĂ­mite superior.

\begin{theorem}[lema de zorn]\label{thm:b3:rings:zorn}
Todo conjunto inductivo parcialmente ordenado no vacĂ­o tiene un \omterm{def:b3:rings:primemaximal}{mĂĄximo}
elemento.\index{Lema de Zorn}
\admitted
\end{theorem}

\begin{remark}\label{rem:b3:rings:zorn}
Este no es un teorema de las matemĂĄticas ordinarias sino un \emph{axioma}:
es equivalente, sobre los axiomas bĂĄsicos de Zermelo-Fraenkel de conjunto
teorĂ­a, al axioma de elecciĂłn ("todo producto de conjuntos no vacĂ­os
no estĂĄ vacĂ­o"), que aceptamos a lo largo de este libro. Marcamos cada uno
uso. El anĂĄlisis lo invocarĂĄ nuevamente (Hahn--Banach,
\cref{ch:b3:banach}).
\end{remark}

\begin{theorem}[krĂźll]\label{thm:b3:rings:krull}
Cada \omterm{def:b3:rings:ideal}{ideal} $I \subsetneq A$ adecuado estĂĄ contenido en un \omterm{def:b3:rings:primemaximal}{mĂĄximo
ideal}.\index{teorema de krul}
\end{theorem}

\begin{proof}
Ordene por inclusiĂłn el conjunto $\mathcal E$ de \omterm{def:b3:rings:ideal}{ideales} adecuado
que contiene $I$; no estĂĄ vacĂ­o ($I \in \mathcal E$). una cadena
$(J_\lambda)$ en $\mathcal E$ tiene un lĂ­mite superior $J = \bigcup
J_\lambda$: un \omterm{def:b3:rings:ideal}{ideal} (cualquier$a, b \in J$se encuentra en un$J_\lambda$ comĂşn
por totalidad), propia ($1 \notin J_\lambda$ para todos los $\lambda$),
que contiene $I$. El lema de Zorn produce un elemento \omterm{def:b3:rings:primemaximal}{mĂĄximo} de $\mathcal
E$, que es un \omterm{def:b3:rings:primemaximal}{ideal mĂĄximo} que contiene$I$ (un \omterm{def:b3:rings:ideal}{ideal} estrictamente
encima y apropiado estarĂ­a en $\mathcal E$).
\end{proof}

\begin{theorem}[Teorema del resto chino]\label{thm:b3:rings:crt}
Sea $I_1, \dots, I_n$ por pares \emph{comaximal}\index{comaximal
ideales} \omterm{def:b3:rings:ideal}{ideales} de $A$ ($I_k + I_l = A$ para $k \neq l$). entonces
\[
A\Big/\bigcap_{k=1}^n I_k \;\xrightarrow{\;\sim\;}\;
\prod_{k=1}^n A/I_k,
\qquad
a \longmapsto (a + I_1, \dots, a + I_n),
\]
y ademĂĄs $\bigcap_k I_k = I_1 I_2 \cdots I_n$ (el \omterm{def:b3:rings:ideal}{ideal}
generado por productos).\index{Teorema del resto chino}
\end{theorem}

\begin{proof}
La aplicaciĂłn $f(a) = (a + I_k)_k$ es un morfismo de anillo con kernel
$\bigcap I_k$; por \cref{thm:b3:rings:firstiso} basta con
demostrar la sobrejetividad. Reparar $k$; para cada $l \neq k$ escriba $1 = u_l +
v_l$con$u_l \in I_k$,$v_l \in I_l$ (comaximalidad). entonces
\[
e_k = \prod_{l \neq k} v_l = \prod_{l\neq k}(1 - u_l)
\equiv 1 \pmod{I_k},
\qquad e_k \in I_l \ (l \neq k),
\]

I_k)_k$, el elemento$\sum_k a_k e_k$ se asigna a ĂŠl.

Productos vs intersecciĂłn: $I_1\cdots I_n \subseteq \bigcap I_k$
siempre. Por el contrario, por inducciĂłn basta con tratar $n = 2$
(se verifica que $I_1$ y $I_2\cdots I_n$ sean \omterm{thm:b3:rings:crt}{comaximal}: multiplicar
$1 = u_l + v_l$ sobre $l \geq 2$ da $1 \in I_1 + I_2\cdots
I_n$). Para$n = 2$: escriba$1 = u + v$,$u \in I_1$,$v \in I_2$;
para $x \in I_1 \cap I_2$, $x = xu + xv \in I_2I_1 + I_1I_2 =
I_1I_2$.
\end{proof}

m_n)\Z \cong \prod \Z/m_k\Z$ --- el volumen del aĂąo 2 en chino
teorema del resto. RestricciĂłn a unidades: $(\Z/mn\Z)^\times \cong
(\Z/m\Z)^\times \times (\Z/n\Z)^\times$para$\gcd(m,n)=1$, de donde
la multiplicatividad de $\varphi$ de Euler
(\cref{exo:b3:rings:8}).
\end{example}

\section{Divisibilidad: euclidiana, principal, factorial}

\begin{definition}\label{def:b3:rings:divisibility}
Sea $A$ un dominio integral, $a, b \in A$. Nosotros decimos $a$
\emph{divide} $b$ ($a \mid b$) si $b \in (a) = aA$. Los elementos $a,
b$son \emph{asociados}\index{asociar elementos} si$a = ub$
con $u \in A^\times$ (equivalentemente $(a) = (b)$). Un distinto de cero
no unidad $p$ es:
\begin{itemize}
  \item \emph{irreducible}\index{elemento irreducible} si $p = ab$
        fuerzas $a \in A^\times$ o $b \in A^\times$;
  \item \emph{principal}\index{elemento principal} si $p \mid ab$ fuerza a $p
        \mid a$o$p \mid b$(es decir, el \omterm{def:b3:rings:ideal}{ideal}$(p)$ es primo).
\end{itemize}
\end{definition}

\begin{proposition}\label{prop:b3:rings:primeirred}
En cualquier dominio, prima $\Rightarrow$ \omterm{def:b3:rings:divisibility}{irreducible}. Lo contrario es
falso en general: en $\Z[\iu\sqrt 5] = \{a + \iu b\sqrt5 : a, b
\in \Z\}$, el elemento$2$ es \omterm{def:b3:rings:divisibility}{irreducible} pero no es primo.
\end{proposition}

\begin{proof}
Sea $p$ primo y $p = ab$. Luego $p \mid ab$, diga $p \mid a$:
$a = pc$, entonces $p = pcb$, y al cancelar $p$ (ÂĄdominio!) se obtiene $cb =
1$:$b \in A^\times$.

En $\Z[\iu\sqrt5]$ se utiliza la norma $N(x + \iu y\sqrt 5) = x^2 +
5y^2$, que es multiplicativa (es$\abs z^2$). Si$2 = ab$
con $a, b$ no unidades, luego $4 = N(a)N(b)$ con $N(a), N(b) \neq
1$(los elementos norma-$1$son$\pm1$, las unidades), por lo que$N(a) = 2$:
imposible, $x^2 + 5y^2 = 2$ no tiene una soluciĂłn entera. Entonces $2$ es
\omterm{def:b3:rings:divisibility}{irreducible}. Pero $2 \mid 6 = (1 + \iu\sqrt5)(1 - \iu\sqrt5)$
mientras que $2$ no divide ningĂşn factor ($\frac12 \pm \frac{\iu\sqrt5}2
\notin \Z[\iu\sqrt5]$): no es primo.
\end{proof}

\begin{definition}\label{def:b3:rings:pidufd}
Un dominio integral $A$ es:
\begin{itemize}
  \item \emph{euclidiano}\index{Dominio euclidiano} si hay una aplicaciĂłn
        $\nu \colon A \setminus \{0\} \to \N$ (un \emph{euclidiano
        funciĂłn}) tal que para todos los $a, b$ con $b \ne 0$ hay
        existen $q, r$ con $a = bq + r$ y ($r = 0$ o $\nu(r) <
        \nu(b)$);
  \item \emph{principal} (a \emph{PID}\index{ideal principal
        dominio}) si cada \omterm{def:b3:rings:ideal}{ideal} tiene la forma $(a)$;
  \item \emph{factorial} (a \emph{UFD}\index{factorizaciĂłn Ăşnica
        dominio}) si cada no unidad distinta de cero es un producto de
        \omterm{def:b3:rings:divisibility}{irreductibles}, Ăşnicamente bajo pedido y \omterm{def:b3:rings:divisibility}{asociados}.
\end{itemize}
\end{definition}

\begin{theorem}\label{thm:b3:rings:euclideanpid}
\omterm{def:b3:rings:pidufd}{euclidiano} $\Rightarrow$ principal.
\end{theorem}

\begin{proof}
Sea $I \neq (0)$ un \omterm{def:b3:rings:ideal}{ideal} y $b \in I \setminus\{0\}$ con
$\nu(b)$ mĂ­nimo. Para $a \in I$, divida: $a = bq + r$; entonces $r = a
- bq \in I$y$\nu(r) < \nu(b)$ contradirĂ­an la minimalidad, por lo que
$r = 0$ y $a \in (b)$: $I = (b)$.
\end{proof}

\begin{example}\label{ex:b3:rings:euclidean}
$\Z$ (con $\nu = \abs\cdot$) y $K[X]$ (con $\nu = \deg$) son
\omterm{def:b3:rings:pidufd}{euclidiano} --- el volumen del AĂąo 2 demostrĂł ambas divisiones. AsĂ­ es
$\Z[\iu]$, siendo $\nu = N$ la norma cuadrada
(\cref{exo:b3:rings:4}); la geometrĂ­a de la prueba estĂĄ en el
figura a continuaciĂłn. Existe un \omterm{def:b3:rings:pidufd}{PID} que no es \omterm{def:b3:rings:pidufd}{euclidiano} pero es delicado
para certificar (el ejemplo estĂĄndar es
$\Z\bigl[\frac{1+\iu\sqrt{19}}2\bigr]$); un \omterm{def:b3:rings:pidufd}{UFD} que no es un \omterm{def:b3:rings:pidufd}{PID}

\begin{lemma}[Cadenas ascendentes de ideales principales]
\label{lem:b3:rings:acc}
En un \omterm{def:b3:rings:pidufd}{PID}, cada secuencia creciente de \omterm{def:b3:rings:ideal}{ideales} $I_1 \subseteq I_2
\subseteq \cdots$ es finalmente constante.
\end{lemma}

\begin{proof}
$I = \bigcup_n I_n$ es \omterm{def:b3:rings:ideal}{ideal} (la uniĂłn estĂĄ aumentando), por lo que $I =
(a)$; el elemento$a$se encuentra en algĂşn$I_N$, y luego$I = (a)
\subseteq I_N \subseteq I_n \subseteq I$en$n \geq N$.
\end{proof}

\begin{lemma}[BĂŠzout; Lema de Euclides]\label{lem:b3:rings:bezout}
Sea $A$ un \omterm{def:b3:rings:pidufd}{PID} y un $a, b \in A$. Entonces $(a) + (b) = (d)$ para algunos
$d$, a \emph{mĂĄximo comĂşn divisor}\index{mayor comĂşn
divisor}: $d \mid a$, $d \mid b$, y cada divisor comĂşn de $a,
b$divide a$d$; ademĂĄs$d = au + bv$para algunos$u, v$
(BĂŠzout). En consecuencia, cada \omterm{def:b3:rings:divisibility}{elemento irreducible} de un \omterm{def:b3:rings:pidufd}{PID} es
prima.
\end{lemma}

\begin{proof}
$(a) + (b)$ es \omterm{def:b3:rings:ideal}{ideal}, por lo tanto $(d)$; $a, b \in (d)$ da $d \mid
a, b$; y$d = au + bv \in (a) + (b)$. Un divisor comĂşn$c$ de
$a, b$ divide $au + bv = d$.

Euclides: sea $p$ \omterm{def:b3:rings:divisibility}{irreducible}, $p \mid ab$, $p \nmid a$. un mcd
$d$ de $p$ y $a$ divide a $p$, por lo que $d$ es una unidad o un asociado
de $p$ (irreductibilidad); El asociado estĂĄ excluido por $p \nmid a$. entonces
$1 = pu + av$, de donde $b = pub + abv$ y $p$ dividen ambos tĂŠrminos:
$p \mid b$.
\end{proof}

\begin{theorem}\label{thm:b3:rings:pidufd}
Principal $\Rightarrow$ factorial.
\end{theorem}

\begin{proof}
\emph{Existencia.} Supongamos que alguna no unidad $a$ distinta de cero no tiene
factorizaciĂłn en \omterm{def:b3:rings:divisibility}{irreductibles}. Entonces $a$ no es \omterm{def:b3:rings:divisibility}{irreducible}: $a =
a_1b_1$ con ambos factores no unidades; al menos uno de ellos, digamos
$a_1$, nuevamente no tiene factorizaciĂłn (un producto de dos factorizables
elementos son factorizables). Iterando, obtenemos $a = a_0, a_1, a_2,
\dots$, cada uno de los cuales es un divisor adecuado del Ăşltimo sin factorizaciĂłn,
entonces $(a_0) \subsetneq (a_1) \subsetneq (a_2) \subsetneq \cdots$
--- las inclusiones son estrictas porque $a_n = a_{n+1}c$ con $c$ a
no unidad significa que $(a_n) = (a_{n+1})$ forzarĂ­a a $c \in A^\times$
(cancelar en un dominio). Esto contradice \cref{lem:b3:rings:acc}.

\emph{Unicidad.} Deja $p_1 \cdots p_r = q_1 \cdots q_s$ con todos
factores \omterm{def:b3:rings:divisibility}{irreducible}, $r \leq s$, por inducciĂłn en $r$. el mejor
(\cref{lem:b3:rings:bezout}) $p_1$ divide el lado derecho, por lo que
divide algunos $q_j$; renumerar $j = 1$. Como $q_1$ es \omterm{def:b3:rings:divisibility}{irreducible} y
$p_1$ no es una unidad, $q_1 = u p_1$ con $u \in A^\times$: $p_1,
q_1$son \omterm{def:b3:rings:divisibility}{asociados}. Cancelar$p_1$:$p_2 \cdots p_r = (u q_2)
q_3\cdots q_s$y concluir por inducciĂłn ($r = 1$fuerza a$s = 1$:
una unidad multiplicada por \omterm{def:b3:rings:divisibility}{irreductibles} no puede ser $1$).
\end{proof}

\begin{remark}\label{rem:b3:rings:ufdgcd}
En un \omterm{def:b3:rings:pidufd}{UFD}, existen mcd (tome exponentes mĂ­nimos en el
factorizaciones) y el lema de Euclides se cumple --- \omterm{def:b3:rings:divisibility}{irreducible} $=$ primo
(\cref{exo:b3:rings:2}) --- pero BĂŠzout puede fallar: en $\Z[X]$,
$\gcd(2, X) = 1$ y $1 \neq 2U + XV$ (evaluar en $X = 0$:
$1 = 2U(0)$, imposible). Las identidades BĂŠzout son exclusivas
propiedad de \omterm{def:b3:rings:pidufd}{PID}.
\end{remark}

\section{Polinomios sobre una UFD: Gauss y Eisenstein}

A lo largo de esta secciĂłn $A$ es un \omterm{def:b3:rings:pidufd}{UFD} con campo de fracciĂłn $K$
(construido como el campo de cocientes formales $a/b$, $b \neq 0$,
exactamente como $\Q$ de $\Z$; el volumen del aĂąo 2 hizo esto
construcciĂłn para $\Q$, y se traslada textualmente). Nuestro objetivo:
la factorialidad pasa de $A$ a $A[X]$, y la irreducibilidad pasa
$A$ es esencialmente irreductibilidad sobre el campo mĂĄs grande $K$.

\begin{definition}\label{def:b3:rings:content}
El \emph{contenido}\index{contenido de un polinomio} $c(P)$ de un
distinto de cero $P \in A[X]$ es un mcd de sus coeficientes (definidos hasta un
unidad); $P$ es \emph{primitivo} si $c(P) \in A^\times$. Cada $P
\in A[X]$escribe$P = c(P)\,P_1$con la primitiva$P_1$, y cada
$P \in K[X]\setminus\{0\}$ escribe $P = \lambda P_1$ con $\lambda
\in K^\times$y$P_1 \in A[X]$ primitivas (denominadores claros,
luego factorice el contenido).
\end{definition}

\begin{lemma}[Gauss]\label{lem:b3:rings:gauss}
El producto de dos polinomios primitivos de $A[X]$ es primitivo;
en consecuencia $c(PQ) = c(P)c(Q)$ hasta unidades.\index{Lema de Gauss}
\end{lemma}

\begin{proof}
Sea $P, Q$ primitivo y supongamos que algĂşn \omterm{def:b3:rings:divisibility}{irreducible} (= primo,
\omterm{def:b3:rings:pidufd}{UFD}) $p$ divide todos los coeficientes de $PQ$. Reducir mĂłdulo $p$: en
$(A/(p))[X]$, $\bar P \bar Q = 0$. Pero $A/(p)$ es un dominio
($(p)$ primo), por lo que $(A/(p))[X]$ es un dominio (coeficientes principales
multiplicar), forzando $\bar P = 0$ o $\bar Q = 0$: $p$ divide todo
coeficientes de $P$ o todo $Q$, contradiciendo la primitividad. Para
la consecuencia, escriba $P = c(P)P_1$, $Q = c(Q)Q_1$: $PQ =
c(P)c(Q) P_1Q_1$con$P_1Q_1$ primitiva.
\end{proof}

\begin{theorem}\label{thm:b3:rings:gaussufd}
Sea $A$ un \omterm{def:b3:rings:pidufd}{UFD} con campo de fracciĂłn $K$.
\begin{enumerate}
  \item Una primitiva $P \in A[X]$ de grado $\geq 1$ es \omterm{def:b3:rings:divisibility}{irreducible}
        en $A[X]$ si es \omterm{def:b3:rings:divisibility}{irreducible} en $K[X]$.
  \item $A[X]$ es un \omterm{def:b3:rings:pidufd}{UFD}; sus \omterm{def:b3:rings:divisibility}{irreductibles} son los \omterm{def:b3:rings:divisibility}{irreductibles} de
        $A$ y los polinomios primitivos \omterm{def:b3:rings:divisibility}{irreducible} sobre $K$. en
        particular $\Z[X]$, y por inducciĂłn $K[X_1, \dots, X_n]$
        y $\Z[X_1, \dots, X_n]$, son \omterm{def:b3:rings:pidufd}{UFD}.
\end{enumerate}
\end{theorem}

\begin{proof}
(1) ($\Leftarrow$) Si $P = QR$ en $A[X]$ con $Q, R$ no unidades,
entonces ninguno de los factores es constante (un factor constante de una primitiva
el polinomio es una unidad), por lo que la factorizaciĂłn es adecuada en $K[X]$.
($\Rightarrow$) Supongamos $P = QR$ con $Q, R \in K[X]$ de grados
$\geq 1$. Escriba $Q = \lambda Q_1$, $R = \mu R_1$ con la primitiva $Q_1, R_1
\in A[X]$:$P = \lambda\mu\, Q_1R_1$y$Q_1R_1$ es
primitivo de Gauss. Tomando \omterm{def:b3:rings:content}{contenido}, $\lambda\mu \in A^\times$
(ambos lados tienen la unidad \omterm{def:b3:rings:content}{contenido}; formalmente, $\lambda\mu = c(P) \in
A^\times$hasta unidades, y en particular$\lambda \mu \in A$):$P
= (\lambda\mu Q_1) R_1$es una factorizaciĂłn adecuada en$A[X]$.

(2) Existencia: dado $P \neq 0$ no unidad, factor $P = c(P)P_1$,
factorice $c(P)$ en \omterm{def:b3:rings:divisibility}{irreductibles} de $A$ y factorice $P_1$ en el
\omterm{def:b3:rings:pidufd}{UFD} $K[X]$ como $\prod Q_i$ con $Q_i \in K[X]$ \omterm{def:b3:rings:divisibility}{irreducible}; escribiendo
$Q_i = \lambda_i R_i$ con la primitiva $R_i \in A[X]$ (por lo tanto
\omterm{def:b3:rings:divisibility}{irreducible} sobre $K$, por lo tanto en $A[X]$ por (1)), el producto
$\prod \lambda_i$ es una unidad de $A$ como antes y $P_1 = u\prod
R_i$. Unicidad: compara la parte constante de una factorizaciĂłn y
parte polinĂłmica; las constantes se multiplican a $c(P)$ (Gauss), Ăşnica
por factorialidad de $A$; las partes polinĂłmicas dan dos
factorizaciones en $K[X]$ del mismo polinomio, para que coincidan
a constantes de $K^\times$ (factorialidad de $K[X]$,
\cref{thm:b3:rings:pidufd}), y polinomios primitivos coincidentes
asociados en $K[X]$ estĂĄn asociados en $A[X]$: si $R = \lambda R'$
con $R, R'$ primitiva y $\lambda \in K^\times$, luego tomando
\omterm{def:b3:rings:content}{contenido} fuerza a $\lambda \in A^\times$.
\end{proof}

$X^n - p$ es \omterm{def:b3:rings:divisibility}{irreducible} sobre $\Q$ para cada primo $p$ y $n

\end{method}

\begin{theorem}[Teorema de la base de Hilbert]\label{thm:b3:rings:hilbert}
Si $A$ es \omterm{def:b3:rings:noetherian}{noetheriano}, tambiĂŠn lo es $A[X]$. Por lo tanto tambiĂŠn lo son $A[X_1, \dots,
X_n]$, y cada cociente de ellos.\index{Teorema de la base de Hilbert}
\end{theorem}

\begin{proof}
Sea $I$ un \omterm{def:b3:rings:ideal}{ideal} de $A[X]$ y supongamos que $I$ no es finitamente
generado. Construye una secuencia: $f_1 \in I \setminus \{0\}$ de
grado mĂ­nimo, e inductivamente $f_{k+1} \in I \setminus (f_1,
\dots, f_k)$ de grado mĂ­nimo (el conjunto no estĂĄ vacĂ­o por
supuesto). Los grados $d_k = \deg f_k$ no son decrecientes (por
minimalidad de cada elecciĂłn: $f_{k+1}$ estaba disponible en el paso
$k+1$... precisamente, $f_{k+1} \notin (f_1,\dots,f_k) \supseteq
(f_1, \dots, f_{k-1})$, por lo que$f_{k+1}$compitiĂł en el paso$k$ y perdiĂł

a_2) \subseteq \cdots$se estabiliza:$a_{n+1} \in (a_1, \dots,
a_n)$para algunos$n$, digamos$a_{n+1} = \sum_{k\leq n} u_k a_k$.
considerar
\[

\]
Entonces $g \in I \setminus (f_1, \dots, f_n)$ (la suma estĂĄ en el
\omterm{def:b3:rings:ideal}{ideal}, $f_{n+1}$ no), pero el coeficiente de grado

Iterando, $A[X_1, \dots, X_n] = (A[X_1, \dots, X_{n-1}])[X_n]$

$J/I$ se eleva a \omterm{def:b3:rings:ideal}{ideales} de $A$ (correspondencia), donde hay un nĂşmero finito

\end{proof}

\section{Problema: teorema de los dos cuadrados de Fermat}

\begin{problem}[{Problema de fin de semana --- sumas de dos cuadrados, vĂ­a
$\Z[\iu]$}]\label{pb:b3:rings:1}
ÂżQuĂŠ nĂşmeros enteros son sumas de dos cuadrados? La respuesta de Fermat (1640) es
una de las joyas de la aritmĂŠtica; los enteros gaussianos convierten su prueba
en la teorĂ­a de los anillos. En todo momento, $N(x + \iu y) = x^2 + y^2$ denota
la norma, $\Z[\iu]$ es \omterm{def:b3:rings:pidufd}{euclidiano} (\cref{exo:b3:rings:4}), por lo tanto,
\omterm{def:b3:rings:pidufd}{PID} y \omterm{def:b3:rings:pidufd}{UFD}, y \emph{primo gaussiano} significa primo (=
\omterm{def:b3:rings:divisibility}{irreducible}) elemento de $\Z[\iu]$.

\medskip
\noindent\textbf{Parte I --- Norms and Gaussian primes.}
\begin{enumerate}
  \item Verifique $N(zw) = N(z)N(w)$, deduzca nuevamente $\Z[\iu]^\times =
        \{\pm1, \pm\iu\}$ y pruebe el \emph{Brahmagupta
        identidad}: un producto de dos sumas de dos cuadrados es una suma
        de dos cuadrados.
  \item Demuestre que si $N(z)$ es un nĂşmero primo, entonces $z$ es un
        Prima gaussiana.
  \item Demuestre que todo primo gaussiano $\pi$ divide exactamente a uno
        nĂşmero primo $p$ \emph{(consider $N(\pi) = \pi\bar\pi$)},
        y que entonces $N(\pi) \in \{p, p^2\}$.
  \item Deducir la dicotomĂ­a: para cada primo $p$, ya sea $p$
        permanece primo en $\Z[\iu]$ (y no hay primo gaussiano de norma
        $p$ existe), o $p = \pi\bar\pi$ con $\pi$ un gaussiano
        prima de la norma $p$ --- y luego $p = a^2 + b^2$.
\end{enumerate}

\medskip
\noindent\textbf{Parte II --- Wilson's theorem and $-1$ modulo
$p$.}
\begin{enumerate}[resume]
  \item Demuestre \emph{teorema de wilson}: para $p$ principal, $(p-1)!
        \equiv -1 \pmod p$. \emph{(Empareje cada residuo con su
        inversa; ÂżCuĂĄles son autoemparejados?)}
  \item Sea $p$ un primo impar y $m = \frac{p-1}2$. mostrar eso
        $(m!)^2 \equiv (-1)^{m+1} \pmod p$\emph{(en $(p-1)!$,
        reemplazar cada factor $k > m$ por $-(p - k)$)}.
  \item Concluye: $-1$ es un mĂłdulo cuadrado $p$ si $p = 2$ o $p
        \equiv 1 \pmod 4$. \emph{(For the ``only if'': if$x^2
        \equiv -1$, what is the order of$x$in$(\Z/p\Z)^\times$,
        and what does Lagrange say?)}
\end{enumerate}

\medskip
\noindent\textbf{Parte III --- The splitting law.}
\begin{enumerate}[resume]
  \item Deje $p \equiv 1 \pmod 4$ y $x$ con $p \mid x^2 + 1 =
        (x + \iu)(x - \iu)$. Demuestre que$p$ es \emph{no} a
        Primo gaussiano y concluir con la Parte I: $p = a^2 + b^2$.
  \item Vamos $p \equiv 3 \pmod 4$. Mostrar directamente que $p$ no es un
        suma de dos cuadrados \emph{(cuadrados mod $4$)}, y deducir
        que $p$ sigue siendo un primo gaussiano.
  \item Resuelva $p = 2$: exhiba la factorizaciĂłn $2 =
        -\iu(1+\iu)^2$y verifique que$1 + \iu$ sea un gaussiano.
        prima. ($2$ es el Ăşnico primo \emph{ramificado}: divisible
        por el cuadrado de una prima gaussiana hasta una unidad.)
  \item Armar la clasificaciĂłn de los primos gaussianos, hasta
        unidades: $1 + \iu$; los nĂşmeros enteros $p \equiv 3 \pmod 4$; el
        pares conjugados $\pi, \bar\pi$ de la norma $p \equiv 1 \pmod
        4$. VerifĂ­quelo en$5 = (2+\iu)(2-\iu)$y en$3$.
\end{enumerate}

\medskip
\noindent\textbf{Parte IV --- The two-squares theorem.}
\begin{enumerate}[resume]
  \item Demostrar la mitad directa: si en la factorizaciĂłn $n =
        \prod p_i^{\alpha_i}$todo primo$\equiv 3 \pmod 4$
        aparece con un exponente par, entonces $n$ es una suma de dos
        cuadrados. \emph{(Brahmagupta + Partes II--III.)}
  \item Demuestre lo contrario: si $n = a^2 + b^2 = N(a + \iu b)$ y
        $q \equiv 3 \pmod 4$ divide $n$, muestra que $q$, a
        Primo gaussiano, divide a $a + \iu b$ o $a - \iu b$, que
        de hecho, divide tanto $a$ como $b$, y concluye por
        inducciĂłn en $n$ de que el exponente de $q$ en $n$ es par.
  \item Enuncie el teorema final. ÂżCuĂĄl de $2025$, $2026$, $2027$?
        son sumas de dos cuadrados? \emph{($2025 = 81 \cdot 25$;
        $2026 = 2 \cdot 1013$, $1013$ prime; $2027$ prime.)}
  \item (EpĂ­logo) Demuestre que un primo $p \equiv 1 \pmod 4$ es un
        suma de dos cuadrados de una manera esencialmente Ăşnica: si $p =
        a^2 + b^2 = c^2 + d^2$(enteros positivos), entonces$\{a, b\}
        = \{c, d\}$. \emph{(Uniqueness of factorization in
        $\Z[\iu]$.)}
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Counting representations: Jacobi's
formula and Leibniz's series.} Escribe $r_2(n) = \#\{(a, b) \in
\Z^2 : a^2 + b^2 = n\}$ (pares ordenados, signos y ceros
incluido), y deje que $\chi$ sea el mod de carĂĄcter no trivial $4$:
$\chi(d) = +1$ si $d \equiv 1$, $-1$ si $d \equiv 3 \pmod4$,
$0$ si $d$ par.
\begin{enumerate}[resume]
  \item (Calentamiento, por el contrario) ÂżQuĂŠ nĂşmeros enteros son
        \emph{diferencias} de dos cuadrados? Mostrar: $n = a^2 - b^2$
        con $a, b \in \Z$ y si $n \not\equiv 2 \pmod 4$ ---
        no se necesita teorĂ­a de anillos y no hay estructura comparable a
        lo que sigue.
  \item Demuestre que $r_2(n)$ es el nĂşmero de $z \in \Z[\iu]$ con
        $N(z) = n$. Escribiendo $n = 2^{a}\prod_jp_j^{b_j}
        \prod_kq_k^{c_k}$con$p_j \equiv 1$,$q_k \equiv 3
        \pmod4$, utilice la clasificaciĂłn de la pregunta 11 y
        factorizaciĂłn Ăşnica para mostrar: tal $z$ existe si todo
        $c_k$ son pares, y en ese caso
        \[

        \]
        \emph{(Count: $z = u\,(1+\iu)^{a}\prod_j\pi_j^{s_j}

        unit and $0 \leq s_j \leq b_j$; why is this list

  \item Demuestre que $d \mapsto \chi(d)$ es completamente
        multiplicativo, deduzca que $n \mapsto
        \sum_{d \mid n}\chi(d)$ es multiplicativo y calculelo

  \item Concluye \emph{teorema de jacobi}:

        \]
        donde $d_i(n)$ cuenta los divisores $\equiv i \pmod 4$.
        Verifique en $n = 3, 5, 9, 25$ y enumere el $16$
        representaciones de $65$.
  \item (El cĂ­rculo) Demuestre que $\sum_{n \leq x}r_2(n)$ es el

        \emph{(cada punto de la red posee un cuadrado unitario; comparar
        ĂĄreas, el error de vivir en un anillo de ancho

  \item (Leibniz, leĂ­do aritmĂŠticamente) Combine las preguntas 19--20:
        \[

        xd\Bigr\rfloor = \frac{\pi x}4 + O(\sqrt x),

        \]
        La serie alterna de recĂ­procos impares \emph{is} la
        exceso promedio de divisores $\equiv 1$ sobre divisores
        $\equiv 3$: anĂĄlisis calculado por aritmĂŠtica.
  \item (ÂżQuĂŠ tan raras son las sumas de dos cuadrados?) Demuestre que no
        El nĂşmero entero $\equiv 3 \pmod 4$ es una suma de dos cuadrados (dos
        formas: cuadrados mod $4$, o el criterio de paridad de
        pregunta 17), por lo que al menos una cuarta parte de todos los nĂşmeros enteros son
        perdido; y demostrar que el promedio $\frac1x\sum_{n\leq
        x}r_2(n) \to \pi$ de la pregunta 20 es compatible con

        (tomar productos de muchos nĂşmeros primos $\equiv 1 \bmod 4$) a
        explicar cĂłmo una proporciĂłn que se desvanece todavĂ­a puede conllevar una

\end{enumerate}
