\chapter{Representaciones de grupos finitos}\label{ch:b3:representations}

Para entender un grupo abstracto, hazlo actuar en un espacio vectorial.
y usar álgebra lineal --- valores propios, trazas, productos internos ---
sobre la acción. Este programa, \emph{\omterm{def:b3:representations:rep}{representación} theory}, es
sorprendentemente eficaz para grupos finitos sobre $\C$: cada
\omterm{def:b3:representations:rep}{representación} se divide en \omterm{def:b3:representations:rep}{irreducible} (Maschke), el
\omterm{def:b3:representations:rep}{irreducible} están inmovilizados por sus \emph{\omterm{def:b3:representations:character}{personajes}}
(trazas), y el \omterm{def:b3:representations:character}{personajes} satisfacen relaciones de ortogonalidad que
hacer cálculos mecánicos. El capítulo construye este cálculo.
y sus primeras obras maestras --- \omterm{def:b3:representations:table}{tablas de caracteres} de la pequeña
grupos --- y el problema del fin de semana genera un teorema mucho más allá
El alcance de la teoría de grupos desnudos: el teorema $p^aq^b$ de Burnside.
En todo momento, $G$ es un grupo finito y todos los espacios vectoriales son
de dimensión finita sobre $\C$.

\section{Representaciones, Maschke, Schur}

\begin{definition}\label{def:b3:representations:rep}
Un \emph{representación}\index{representación} de $G$ es un
morfismo $\rho \colon G \to GL(V)$ para un espacio vectorial $\C$ $V$;
$\dim V$ es su \emph{grado}. Un subespacio $W \subseteq V$ es
\emph{invariante} si $\rho(g)W \subseteq W$ para todos los $g$;
La restricción convierte a $W$ en \emph{subrepresentación}. $\rho$ es
\emph{irreducible}\index{representación irreductible} si $V \neq
0$y sus únicos subespacios invariantes son$0$y$V$. un
\emph{morfismo} entre $(\rho, V)$ y $(\sigma, W)$ es un lineal
$f\colon V \to W$ con $f\rho(g) = \sigma(g)f$ para todos los $g$
(equivarianza); biyectivo $f$ son \emph{isomorfismos}.
\end{definition}

\begin{example}\label{ex:b3:representations:examples}
(a) Grado $1$: morfismos $G \to \C^\times$.
(b) El \emph{representación regular}\index{regular
representación}: $V = \C^G$ con base $(e_h)_{h \in G}$,
$\rho(g)e_h = e_{gh}$; grado $\abs G$.
(c) Una acción de permutación de $G$ en un conjunto finito $X$ da la
\emph{permutation \omterm{def:b3:representations:rep}{representación}} en $\C^X$: $\rho(g)e_x =
e_{g\cdot x}$.
(d) $S_n$ actúa sobre $\C^n$ permutando coordenadas; el hiperplano
$\{\sum x_i = 0\}$ es invariante: el \emph{standard
\omterm{def:b3:representations:rep}{representación}}, de grado $n - 1$.
\end{example}

\begin{theorem}[máscara]\label{thm:b3:representations:maschke}
Cada subespacio invariante $W$ de un \omterm{def:b3:representations:rep}{representación} $(V, \rho)$
admite un complemento invariante. En consecuencia cada \omterm{def:b3:representations:rep}{representación}
es una suma directa de \omterm{def:b3:representations:rep}{irreducible} unos
(\emph{semisimplicidad}).\index{teorema de maschke}
\end{theorem}

\begin{proof}
Sea $p \colon V \to V$ la proyección \emph{cualquier} con la imagen $W$
(elige cualquier complemento). Promedilo sobre el grupo:
\[
\tilde p = \frac1{\abs G}\sum_{g \in G} \rho(g)\,p\,\rho(g)^{-1}.
\]
Cada término asigna $V$ a $W$ (invariante $W$) y corrige $W$.
puntualmente: para $w \in W$, $\rho(g)^{-1}w \in W$, $p$ lo arregla,
y $\rho(g)$ lo trae de vuelta --- entonces $\tilde p$ es nuevamente un
proyección sobre $W$. Es equivariante: para $h \in G$,
$\rho(h)\tilde p\rho(h)^{-1}$ reindexa la misma suma. Por tanto, $\ker
\tilde p$es un complemento invariante de$W$. Iterando sobre el
sumandos (dimensión finita) descompone $V$ en \omterm{def:b3:representations:rep}{irreductibles}.
\end{proof}

\begin{theorem}[Lema de Schur]\label{thm:b3:representations:schur}
Sea $f \colon V \to W$ un morfismo de \emph{\omterm{def:b3:representations:rep}{irreducible}}
\omterm{def:b3:representations:rep}{representaciones}. Entonces $f = 0$ o $f$ es un isomorfismo; y si
$(V,\rho) = (W,\sigma)$, luego $f = \lambda\,\mathrm{id}$ para algunos
$\lambda \in \C$. Por lo tanto
$\dim\operatorname{Hom}_G(V, W)$ es $1$ si $V \cong W$, $0$
de lo contrario.\index{Lema de Schur}
\end{theorem}

\begin{proof}
$\ker f$ y $\operatorname{im} f$ son invariantes (equivarianza),
entonces cada uno es $0$ o todo: $f = 0$ o $f$ es inyectivo
con imagen completa. Si $V = W$: $f$ tiene un valor propio $\lambda$ ($\C$
\omterm{def:b3:galois:closure}{algebraicamente cerrado}); $f - \lambda\,\mathrm{id}$ es un
morfismo no inyectivo $V \to V$, por lo tanto $0$. Para la dimensión
contar cuando $V \cong W$: arreglando un isomorfismo $u$, cualquier morfismo
$f$ da el endomorfismo $u^{-1}f = \lambda\,\mathrm{id}$: $f =
\lambda u$.
\end{proof}

\section{Caracteres y ortogonalidad}

\begin{definition}\label{def:b3:representations:character}
El \emph{personaje}\index{personaje} de $(V, \rho)$ es
$\chi_\rho(g) = \operatorname{tr}\rho(g)$. Satisface
$\chi_\rho(e) = \dim V$, $\chi_\rho(hgh^{-1}) = \chi_\rho(g)$
(los rastros son invariantes en la conjugación): los caracteres son \emph{clase
funciones}\index{función de clase} --- elementos del espacio
$\mathcal{CF}(G)$ de funciones constantes en \omterm{ex:b3:groups:actions}{clases de conjugación},
Equipado con el producto interior hermitiano.
\[
\langle \varphi, \psi\rangle = \frac1{\abs G}\sum_{g \in G}
\overline{\varphi(g)}\,\psi(g).
\]
\end{definition}

\begin{proposition}\label{prop:b3:representations:charbasics}
$\rho(g)$ es diagonalizable con valores propios de raíces unitarias;
$\chi_\rho(g^{-1}) = \overline{\chi_\rho(g)}$, y
$\abs{\chi_\rho(g)} \leq \chi_\rho(e)$ con igualdad y sii
$\rho(g)$ es un escalar. \omterm{def:b3:representations:character}{Personajes} sumas directas: $\chi_{V
\oplus W} = \chi_V + \chi_W$.
\end{proposition}

\begin{proof}
$\rho(g)^N = \mathrm{id}$ para $N = \abs G$ (Lagrange): $\rho(g)$
aniquila $X^N - 1$, dividido con raíces simples, por lo que es
diagonalizable con valores propios $\lambda_i \in \mu_N$. entonces
$\chi(g^{-1}) = \sum \lambda_i^{-1} = \sum\bar\lambda_i =
\overline{\chi(g)}$y$\abs{\chi(g)} = \abs{\sum\lambda_i}
\leq \dim V$, con igualdad en el triángulo desigualdad si todo
$\lambda_i$ son iguales, es decir\ $\rho(g) = \lambda\,\mathrm{id}$.
Aditividad en sumas directas: trazas de bloques.
\end{proof}

\begin{lemma}\label{lem:b3:representations:homtrace}
Sea $(V, \rho)$, $(W, \sigma)$ \omterm{def:b3:representations:rep}{representaciones}. el promedio
operador en $\operatorname{Hom}(W, V)$,
\[
c(A) = \frac1{\abs G}\sum_{g}\rho(g)\,A\,\sigma(g)^{-1},
\]
es una proyección sobre $\operatorname{Hom}_G(W, V)$, y para el
mapa $\Phi_g \colon A \mapsto \rho(g)A\sigma(g)^{-1}$ uno tiene
$\operatorname{tr}\Phi_g =
\chi_\rho(g)\,\overline{\chi_\sigma(g)}$.
\end{lemma}

\begin{proof}
$c(A)$ es equivariante (reindexar la suma como en Maschke), y $c$
corrige mapas equivariantes (cada término es igual a $A$): $c$ es un
proyección con imagen $\operatorname{Hom}_G(W, V)$. Traza: en
bases, $\Phi_g(A) = BAC$ con $B = \rho(g)$, $C = \sigma(g)^{-1}$;
sobre la base $(E_{kl})$ de matrices, $BE_{kl}C = \sum_{m,n}
b_{mk}c_{ln}E_{mn}$, por lo que el coeficiente de$E_{kl}$ en
$\Phi_g(E_{kl})$ es $b_{kk}c_{ll}$: $\operatorname{tr}\Phi_g =
\sum_{k,l}b_{kk}c_{ll} = \operatorname{tr}(B)
\operatorname{tr}(C) = \chi_\rho(g)\chi_\sigma(g^{-1})$, y
$\chi_\sigma(g^{-1}) = \overline{\chi_\sigma(g)}$.
\end{proof}

\begin{theorem}[Primeras relaciones de ortogonalidad]
\label{thm:b3:representations:orthogonality}
Sea $\rho, \sigma$ \emph{\omterm{def:b3:representations:rep}{irreducible}}. Entonces\index{ortogonalidad
de personajes}
\[
\langle\chi_\sigma, \chi_\rho\rangle =
\begin{cases}
1 & \text{if } \rho \cong \sigma,\\
0 & \text{otherwise:}
\end{cases}
\]
el \omterm{def:b3:representations:rep}{irreducible} \omterm{def:b3:representations:character}{personajes} forma una familia ortonormal en
$\mathcal{CF}(G)$.
\end{theorem}

\begin{proof}
La huella de una proyección es la dimensión de su imagen:
\[
\dim\operatorname{Hom}_G(W, V) = \operatorname{tr} c
= \frac1{\abs G}\sum_g \operatorname{tr}\Phi_g
= \frac1{\abs G}\sum_g \chi_\rho(g)\overline{\chi_\sigma(g)}
= \langle \chi_\sigma, \chi_\rho\rangle,
\]
y el lema de Schur evalúa el lado izquierdo como $\delta_{\rho \cong
\sigma}$.
\end{proof}

\begin{corollary}\label{cor:b3:representations:multiplicity}
Descomponer $V \cong \bigoplus_i V_i^{\oplus m_i}$ en distintos
\omterm{def:b3:representations:rep}{irreductibles} ($V_i \not\cong V_j$). Entonces $m_i = \langle
\chi_{V_i}, \chi_V\rangle$: las multiplicidades --- de ahí el
\omterm{def:b3:representations:rep}{representación} hasta isomorfismo --- están determinados por el
\omterm{def:b3:representations:character}{personaje}. Además $\langle \chi_V, \chi_V\rangle = \sum_i
m_i^2$; en particular$V$es \omterm{def:b3:representations:rep}{irreducible} si es$\langle\chi_V,
\chi_V\rangle = 1$.
\end{corollary}

\begin{proof}
$\chi_V = \sum m_i\chi_{V_i}$
(\cref{prop:b3:representations:charbasics}); tomar productos internos
con cada $\chi_{V_i}$ y use ortonormalidad. dos
\omterm{def:b3:representations:rep}{representaciones} con \omterm{def:b3:representations:character}{personajes} iguales tienen multiplicidades iguales,
por tanto son isomórficos.
\end{proof}

\begin{theorem}[La representación regular]
\label{thm:b3:representations:regular}
Sea $\chi_1, \dots, \chi_r$ el \omterm{def:b3:representations:rep}{irreducible} distinto
\omterm{def:b3:representations:character}{personajes}, de grados $n_i = \chi_i(e)$. El \omterm{ex:b3:representations:examples}{regular
representación} se descompone con multiplicidades $m_i = n_i$;
en consecuencia
\[
\sum_{i=1}^{r} n_i^2 = \abs G,
\qquad
\sum_i n_i\chi_i(g) = 0 \quad (g \neq e).
\]
\end{theorem}

\begin{proof}
El \omterm{def:b3:representations:character}{personaje} normal: $\chi_{\mathrm{reg}}(g) =
\#\{h : gh = h\}$, que es$\abs G$para$g = e$y$0$.
de lo contrario. Entonces $m_i = \langle \chi_i,\chi_{\mathrm{reg}}\rangle =
\frac1{\abs G}\overline{\chi_i(e)}\,\abs G = n_i$. evaluando
$\chi_{\mathrm{reg}} = \sum n_i\chi_i$ en $e$ y en $g \neq e$
da las dos identidades mostradas.
\end{proof}

\begin{theorem}\label{thm:b3:representations:numberirr}
El \omterm{def:b3:representations:rep}{irreducible} \omterm{def:b3:representations:character}{personajes} forma un \emph{base} ortonormal de
$\mathcal{CF}(G)$: el número de \omterm{def:b3:representations:rep}{representaciones irreductibles}
es igual al número de \omterm{ex:b3:groups:actions}{clases de conjugación} de $G$.
\end{theorem}

\begin{proof}
Sólo queda la integridad: sea $f \in \mathcal{CF}(G)$
ortogonal a cada $\chi_i$; mostramos $f = 0$. por un
\omterm{def:b3:representations:rep}{representación} $(V,\rho)$, establezca $T_{f,\rho} = \frac{1}{\abs G}
\sum_g \overline{f(g)}\,\rho(g)$. Es equivariante: para$h \in
G$,
\[
\rho(h)T_{f,\rho}\rho(h)^{-1} = \frac1{\abs G}\sum_g
\overline{f(g)}\rho(hgh^{-1})
= \frac1{\abs G}\sum_{g'}\overline{f(h^{-1}g'h)}\rho(g') =
T_{f,\rho}
\]
($f$ es un \omterm{def:b3:representations:character}{función de clase}). Si $\rho$ es \omterm{def:b3:representations:rep}{irreducible} de grado
$n$, Schur da $T_{f,\rho} = \lambda\,\mathrm{id}$ con
\[
\lambda = \frac{\operatorname{tr}T_{f,\rho}}{n}
= \frac{1}{n\abs G}\sum_g \overline{f(g)}\,\chi_\rho(g)
= \frac1n\,\langle f, \chi_\rho\rangle = 0 .
\]
Entonces $T_{f,\rho} = 0$ en cada \omterm{def:b3:representations:rep}{irreducible}, por lo tanto (sumas directas) en
\emph{cada} \omterm{def:b3:representations:rep}{representación} --- en particular en el normal.
Aplicarlo al vector base $e_e$: $0 = T_{f,\mathrm{reg}}e_e =
\frac1{\abs G}\sum_g\overline{f(g)}e_g$, forzando cada
$\overline{f(g)} = 0$. Así, la familia ortonormal
$(\chi_i)$ abarca $\mathcal{CF}(G)$, cuya dimensión es la
número de \omterm{ex:b3:groups:actions}{clases de conjugación}.
\end{proof}

\begin{corollary}[ortogonalidad de columna]
\label{cor:b3:representations:column}
Para $g, h \in G$:
\[
\sum_{i=1}^{r}\overline{\chi_i(g)}\,\chi_i(h) =
\begin{cases}
\abs{Z_G(g)} & \text{if } g, h \text{ are conjugate},\\
0 & \text{otherwise.}
\end{cases}
\]
\end{corollary}

\begin{proof}
Dejemos que $g_1, \dots, g_r$ represente las clases, $c_j$ la clase.
tamaños. La matriz $r \times r$ $U_{ij} =
\sqrt{c_j/\abs G}\;\chi_i(g_j)$ tiene filas ortonormales
(\cref{thm:b3:representations:orthogonality} written classwise:
$\sum_j \frac{c_j}{\abs G}\chi_i(g_j)\overline{\chi_{i'}(g_j)} =
\delta_{ii'}$), i.e.$UU^* = I$; una matriz cuadrada con$UU^* =
I$también tiene$U^*U = I$: las columnas son ortonormales, lo que se descomprime en
the displayed identity ($\abs G/c_j = \abs{Z_G(g_j)}$, the
orbit--stabilizer relation for conjugation).
\end{proof}

\begin{proposition}[Características unidimensionales; levantamiento]
\label{prop:b3:representations:onedim}
(a) $G$ es abeliano si todos sus \omterm{def:b3:representations:rep}{representaciones irreductibles} tienen
grado $1$; el número de grado-$1$ \omterm{def:b3:representations:character}{personajes} de cualquier $G$ es
$[G : D(G)]$ (son los \omterm{def:b3:representations:character}{personajes} de la abelianización).
(b) Si $N \trianglelefteq G$, el \omterm{def:b3:representations:rep}{representaciones irreductibles} de
$G/N$ levante (componga con $G \to G/N$) exactamente a esos
\omterm{def:b3:representations:rep}{representaciones irreductibles} de $G$ cuyo kernel contiene $N$.
\end{proposition}

\begin{proof}
(a) Si $G$ es abeliano, cada clase es un singleton: $r = \abs G$,
y $\sum n_i^2 = \abs G$ fuerza todo $n_i = 1$; por el contrario si
todo $n_i = 1$, el \omterm{ex:b3:representations:examples}{representación regular} es una suma de
unidimensionales, por lo que $\rho_{\mathrm{reg}}(G)$ es
simultáneamente diagonalizable, por lo tanto conmutativo, y
$\rho_{\mathrm{reg}}$ es fiel: $G$ abeliano. Grado-$1$
\omterm{def:b3:representations:rep}{representaciones} son morfismos $G \to \C^\times$ con abeliano
objetivo: factorizan $G^{\mathrm{ab}} = G/D(G)$
(\cref{exo:b3:groups:9}), y distinto \omterm{def:b3:representations:character}{personajes} del abeliano
$G^{\mathrm{ab}}$ número $\abs{G^{\mathrm{ab}}}$ (tiene ese
muchas clases, todos los grados $1$). (b) Composición con el
la proyección preserva la irreductibilidad (subespacios invariantes
corresponden), y un \omterm{def:b3:representations:rep}{representación} trivial en $N$ factores a través
el cociente (\cref{thm:b3:groups:firstiso}).
\end{proof}

\section{Tablas de caracteres}

\begin{definition}\label{def:b3:representations:table}
El \emph{tabla de caracteres}\index{tabla de caracteres} de $G$ es el
$r \times r$ matriz $\bigl(\chi_i(g_j)\bigr)$: filas indexadas por
\omterm{def:b3:representations:rep}{irreducible} \omterm{def:b3:representations:character}{personajes}, columnas de \omterm{ex:b3:groups:actions}{clases de conjugación} (con sus
tamaños mostrados). Las filas son ortonormales para el producto ponderado,
columnas ortogonales (\cref{cor:b3:representations:column}): las
tabla está severamente sobredeterminada, que es lo que la hace
computable.
\end{definition}

\begin{example}[La tabla de $S_3$]\label{ex:b3:representations:s3}
Clases: $e$ (tamaño 1), transposiciones (3), $3$-ciclos (2); entonces $r
= 3$\omterm{def:b3:representations:rep}{irreductibles}, de grados$n_i$con$\sum n_i^2 = 6$:$1, 1,
2$. Grado$1$: trivial$\mathbf 1$y firma$\varepsilon$.
La última fila se deriva de la ortogonalidad de la columna (o de
$\chi_{\mathrm{std}} = \chi_{\mathrm{perm}} - \mathbf 1$):
\[
\begin{array}{c|ccc}
S_3 & e & (1\,2)\ [3] & (1\,2\,3)\ [2]\\
\hline
\mathbf 1 & 1 & 1 & 1\\
\varepsilon & 1 & -1 & 1\\
\chi_{\mathrm{std}} & 2 & 0 & -1
\end{array}
\]
Comprobar: $\langle\chi_{\mathrm{std}},\chi_{\mathrm{std}}\rangle =
\frac{1}{6}(4 + 0 + 2) = 1$: \omterm{def:b3:representations:rep}{irreducible}.
\end{example}

\begin{example}[La tabla de $S_4$]\label{ex:b3:representations:s4}
Clases: $e$ [1], transposiciones [6], transposiciones dobles [3],
$3$-ciclos [8], $4$-ciclos [6]: cinco \omterm{def:b3:representations:rep}{irreductibles}, $\sum n_i^2 =
24$con dos grados$1$($\mathbf 1, \varepsilon$;$[S_4 :
D(S_4)] = [S_4 : A_4] = 2$): grados$1, 1, 2, 3, 3$. el
grados-$2$ levanta desde $S_4/V \cong S_3$
(\cref{prop:b3:representations:onedim}(b), $V$ el grupo Klein);
grado $3$: el estándar \omterm{def:b3:representations:rep}{representación} y su giro por
$\varepsilon$:
\[
\begin{array}{c|ccccc}
S_4 & e\,[1] & (1\,2)\,[6] & (1\,2)(3\,4)\,[3] &
(1\,2\,3)\,[8] & (1\,2\,3\,4)\,[6]\\
\hline
\mathbf 1 & 1 & 1 & 1 & 1 & 1\\
\varepsilon & 1 & -1 & 1 & 1 & -1\\
\chi_2 & 2 & 0 & 2 & -1 & 0\\
\chi_{\mathrm{std}} & 3 & 1 & -1 & 0 & -1\\
\varepsilon\chi_{\mathrm{std}} & 3 & -1 & -1 & 0 & 1
\end{array}
\]
($\chi_{\mathrm{std}}(g) = \operatorname{fix}(g) - 1$;
$\chi_2$ evalúa la tabla $S_3$ en la imagen de cada clase
mod $V$.) Todas las comprobaciones de ortogonalidad de filas y columnas pasan ---
ejecutar dos de ellos es el calentamiento de \cref{exo:b3:representations:3}.
\end{example}

\begin{method}\label{met:b3:representations:table}
Para construir un \omterm{def:b3:representations:table}{tabla de caracteres}: (1) enumere \omterm{ex:b3:groups:actions}{clases de conjugación} y sus tamaños;
(2) cuente los grados-$1$ \omterm{def:b3:representations:character}{personajes} a través de $G/D(G)$ y escríbalos; (3)
encuentre los grados restantes de $\sum n_i^2 = \abs G$ (pequeño
combinatoria de números enteros); (4) obtenga \omterm{def:b3:representations:rep}{irreductibles} barato: levante desde
cocientes, restar $\mathbf 1$ de la permutación \omterm{def:b3:representations:character}{personajes}
(verifique $\langle\chi,\chi\rangle = 1$), conocido multiplicado \omterm{def:b3:representations:character}{personajes}
por grado-$1$ unos; (5) terminar filas desconocidas por columna
ortogonalidad --- cada columna es ortogonal a las columnas
ya está completo, y la columna $e$ lleva los grados. verificar
todo con un barrido de ortogonalidad completo.
\end{method}

\section{Ceremonias}

\begin{exercise}[$\star$]\label{exo:b3:representations:1}
(a) Demuestre que los \omterm{def:b3:representations:rep}{irreducible} \omterm{def:b3:representations:character}{personajes} de $\Z/n\Z$ son los
$\chi_k(\bar m) = \eu^{2\iu\pi km/n}$, $k = 0, \dots, n-1$ y
escriba el \omterm{def:b3:representations:table}{tabla de caracteres} de $\Z/4\Z$.
(b) Verifique ambas relaciones de ortogonalidad en él --- y reconozca
La matriz: ¿dónde la ha visto este libro antes?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b3:representations:2}
Sea $G$ actuar sobre un conjunto finito $X$ y $\chi$ sea el \omterm{def:b3:representations:character}{personaje} de
la permutación \omterm{def:b3:representations:rep}{representación} $\C^X$.
(a) Mostrar $\chi(g) = \abs{\operatorname{Fix}_X(g)}$ y
$\langle \mathbf 1, \chi\rangle = \#\{\text{orbits}\}$ ---
El lema de conteo de Burnside (\cref{exo:b3:groups:5}) es un \omterm{def:b3:representations:character}{personaje}
cálculo.
(b) Supongamos que la acción es transitiva, entonces $\chi = \mathbf 1 +

\begin{problem}[{Problema de fin de semana --- solucionabilidad de grupos de
orden $p^aq^b$}]\label{pb:b3:representations:1}
Burnside demostró en 1904 que todo grupo cuyo orden tenga como máximo
dos factores primos es \omterm{def:b3:groups:derived}{soluble} --- una declaración sobre resumen
grupos cuyas únicas pruebas conocidas durante medio siglo pasaron por
\omterm{def:b3:representations:character}{personaje} teoría. Este problema construye la prueba en su totalidad,
ensamblando \cref{ch:b3:groups} (\omterm{def:b3:groups:derived}{solubilidad}),
\cref{ch:b3:modules} (\omterm{def:b3:modules:free}{finitamente generado} $\Z$-módulos) y este
capítulo. En todo momento, $\chi_1, \dots, \chi_r$ son los \omterm{def:b3:representations:rep}{irreducible}.
\omterm{def:b3:representations:character}{personajes} de $G$, $n_i = \chi_i(e)$.

\medskip
\noindent\textbf{Parte I --- \omterm{def:b3:galois:algebraic}{Algebraico} integers.} Un
\emph{\omterm{def:b3:galois:algebraic}{algebraico} integer} es una raíz de un polinomio \emph{mónico}
de $\Z[X]$.
\begin{enumerate}
  \item Demuestre que $\alpha$ es un entero \omterm{def:b3:galois:algebraic}{algebraico} si el anillo
        $\Z[\alpha]$ es un módulo \omterm{def:b3:modules:free}{finitamente generado} $\Z$.
  \item Deduzca que los números enteros \omterm{def:b3:galois:algebraic}{algebraico} forman un subanillo de
        $\C$. \emph{(If $\Z[\alpha], \Z[\beta]$ are \omterm{def:b3:modules:free}{finitamente
        generado}, so is $\Z[\alpha, \beta]$, and submodules of
        \omterm{def:b3:modules:free}{finitamente generado} $\Z$-modules are \omterm{def:b3:modules:free}{finitamente generado},
        by \cref{thm:b3:modules:submodule} and a presentation
        argument --- or directly: a submodule of $\Z^n$ is free
        of rank $\leq n$.)}
  \item Demuestre que un entero \emph{racional} \omterm{def:b3:galois:algebraic}{algebraico} es un
        entero. \emph{(Teorema de la raíz racional).}
  \item Demuestre que cada valor de \omterm{def:b3:representations:character}{personaje} $\chi(g)$ es un
        \omterm{def:b3:galois:algebraic}{algebraico} entero.
\end{enumerate}

\medskip
\noindent\textbf{Parte II --- The class-sum relations.} Reparar un
\omterm{def:b3:representations:rep}{irreducible} $(V, \rho)$ de grado $n$ y \omterm{def:b3:representations:character}{personaje} $\chi$. por un
\omterm{ex:b3:groups:actions}{clase de conjugación} $C$, digamos $S_C = \sum_{g \in C}\rho(g) \in
\mathcal L(V)$.
\begin{enumerate}[resume]
  \item Demuestre que $S_C$ es equivariante, por lo tanto $S_C =
        \omega_C\,\mathrm{id}$ con
        \[
        \omega_C = \frac{\abs C\,\chi(g_C)}{n}
        \qquad (g_C \in C \text{ any representative}).
        \]
  \item Muestre que $S_CS_{C'} = \sum_{C''}
        a_{CC'C''}\,S_{C''}$donde$a_{CC'C''} \in \N$ cuenta,
        para un $z \in C''$ fijo, los pares $(x, y) \in C \times

        \emph{(The $\Z$-module generated by $1$ and the
        $\omega_C$ is a \omterm{def:b3:modules:free}{finitamente generado} ring; apply question

  \item Deducir \emph{\omterm{thm:b3:galois:finitefields}{Frobenius} divisibility}: $n_i$ divide
        $\abs G$ por cada grado \omterm{def:b3:representations:rep}{irreducible}.
        \emph{(Compute $\frac{\abs G}{n} = \frac{\abs G}{n}
        \langle\chi,\chi\rangle = \sum_C \omega_C\,

\noindent\textbf{Parte III --- Burnside's simplicity criterion.}
\begin{enumerate}[resume]
  \item Sea $\chi$ \omterm{def:b3:representations:rep}{irreducible} de grado $n$ y $C$ una clase
        con $\gcd(\abs C, n) = 1$. Uso de Bézout y preguntas
        4--7, muestra que $\frac{\chi(g_C)}{n}$ es un \omterm{def:b3:galois:algebraic}{algebraico}
        entero.
  \item Supongamos además $0 < \abs{\chi(g_C)} < n$. mostrar esto es
        imposible: el entero \omterm{def:b3:galois:algebraic}{algebraico} $\alpha =
        \chi(g_C)/n$ tiene todos sus \emph{conjugados} --- el
        números $\alpha_\sigma = \frac1n\sigma(\chi(g_C))$ para
        $\sigma \in \operatorname{Gal}(\Q(\zeta_{\abs G})/\Q)$,

        $\leq 1$,

\begin{enumerate}[resume]
  \item (Más nítido que una divisibilidad) Sea $\chi$ \omterm{def:b3:representations:rep}{irreducible}
        de grado $n$. Muestre que $\abs{\chi(z)} = n$ para cada
        $z \in Z(G)$ \emph{(Schur's lemma: $\rho(z)$ is a
        scalar, of finite order)}, y deducir de
        $\langle\chi, \chi\rangle = 1$ el atado
        \[
        n^2 \leq [G : Z(G)] .
        \]
        Demuestre que los grupos nobelianos de orden $8$ tienen
        \omterm{def:b3:representations:rep}{irreducible} grados $1, 1, 1, 1, 2$ \emph{(five
        \omterm{ex:b3:groups:actions}{clases de conjugación}; write $8$ as a sum of five squares)}
        y alcanzar la igualdad $4 = [G : Z(G)]$; comprobar el límite
        en $A_5$, cuyo \omterm{ex:b3:groups:actions}{centro} es trivial.
  \item (Tensor cuadrado de $\chi_2$) Para $g$ de orden finito,
        $\rho(g)$ es diagonalizable con raíz de unidad
        valores propios; deducir las fórmulas \omterm{def:b3:representations:character}{personaje}
        \[
        \chi_{\operatorname{Sym}^2 V}(g)
        = \frac{\chi(g)^2 + \chi(g^2)}2,
        \qquad
        \chi_{\Lambda^2 V}(g)
        = \frac{\chi(g)^2 - \chi(g^2)}2 .
        \]
        Aplicarlos a $\chi_2$ de $A_5$ \emph{(tenga en cuenta que $g^2$ se ejecuta
        a través de la clase de $c^2$ cuando $g$ pasa por eso
        de $c$, y viceversa)}: mostrar $\Lambda^2\chi_2 =
        \chi_2$y$\operatorname{Sym}^2\chi_2 = \mathbf 1 +
        \chi_5$, por lo tanto
        \[
        \chi_2\otimes\chi_2 = \mathbf 1 + \chi_2 + \chi_5 .
        \]
        Interpretar $\Lambda^2\chi_2 = \chi_2$ geométricamente mediante
        el producto cruzado en $\R^3$.
  \item (Auditoría final de la tabla) Verificar numéricamente: columna
        ortogonalidad entre las dos columnas de cinco ciclos
        ($\varphi\bar\varphi = -1$), el valor $5 =
        \abs{Z_{A_5}(c)}$para la columna de$c$ contra sí misma,
        y la desaparición del \omterm{def:b3:representations:character}{personaje} $\sum_i
        n_i\chi_i(g) = 0$ regular en cada uno de los cuatro
        columnas de la tabla.
\end{enumerate}
\end{problem}
