\section{El teorema de la función inversa}

\begin{theorem}[Teorema de la función inversa]
\label{thm:b3:submanifolds:ift}
Deje que $U \subseteq \R^n$ esté abierto, $f \colon U \to \R^n$ de
clase $\mathcal C^1$ y $a \in U$ con $Df(a)$ invertible.
Luego están \omterm{def:b3:topology:topology}{conjuntos abiertos} $V \ni a$, $W \ni f(a)$ tales que
$f \colon V \to W$ es una biyección con $\mathcal C^1$ inversa,
y\index{teorema de la función inversa}
\[
D(f^{-1})(y) = \bigl(Df(f^{-1}(y))\bigr)^{-1}
\qquad (y \in W).
\]
Si $f$ es $\mathcal C^k$, también lo es $f^{-1}$.
\end{theorem}

\begin{proof}
Normalizar: reemplazando $f$ por $x \mapsto Df(a)^{-1}\bigl(f(a +
x) - f(a)\bigr)$, podemos asumir$a = 0$,$f(0) = 0$,$Df(0) =
I$ (la declaración general sigue componiendo con el
biyecciones afines). Escriba $f(x) = x + g(x)$: $Dg(0) = 0$, y
por \omterm{def:b3:topology:continuity}{continuidad} de $Dg$ elige $r > 0$ con
$\vertiii{Dg(x)} \leq \frac12$ en $\bar B(0, r)$; la media
La desigualdad de valores da $\norm{g(x) - g(x')} \leq \frac12
\norm{x - x'}$ allí.

\emph{Bijectivity onto a \omterm{def:b3:topology:topology}{entorno}.} Para $y \in B(0,
\frac r2)$, resolver$f(x) = y$ significa encontrar un punto fijo de
$\Phi_y(x) = y - g(x)$; $\Phi_y$ asigna $\bar B(0, r)$ a
sí mismo ($\norm{\Phi_y(x)} \leq \norm y + \frac12\norm x \leq
r$) y es$\frac12$-Lipschitz: Banach
(\cref{thm:b3:complete:banach}) ofrece una solución única
$x = \varphi(y) \in \bar B(0,r)$. Además $f$ es inyectivo.
en $\bar B(0,r)$:
\[
\norm{f(x) - f(x')} \geq \norm{x - x'} - \norm{g(x) - g(x')}
\geq \tfrac12\norm{x - x'} .
\tag{$*$}
\]
Configure $W = B(0, \frac r2)$ y $V = f^{-1}(W)\cap B(0, r)$:
abierto (\omterm{def:b3:topology:continuity}{continuidad}), con $f \colon V \to W$ biyectivo.

\emph{\omterm{def:b3:topology:continuity}{Continuidad} and differentiability of the inverse.} $(*)$
dice $\varphi = f^{-1}$ es $2$-Lipschitz. Reparar $y_0 = f(x_0)
\in W$; invertibilidad de$A = Df(x_0)$(su distancia a$I$
es $\leq \frac12$: Neumann,
\cref{prop:b3:banach:neumann}) y diferenciabilidad de $f$
dar, para $y = f(x)$ cerca de $y_0$:
\[
\varphi(y) - \varphi(y_0) - A^{-1}(y - y_0)
= -A^{-1}\bigl(f(x) - f(x_0) - A(x - x_0)\bigr)
= -A^{-1}\,o(\norm{x - x_0}) = o(\norm{y - y_0}),
\]
usando $(*)$ para convertir $\norm{x - x_0} \leq 2\norm{y - y_0}$:
$\varphi$ es diferenciable en $y_0$ con la inversa
diferencial. \omterm{def:b3:topology:continuity}{Continuidad} de $y \mapsto D\varphi(y) =
Df(\varphi(y))^{-1}$: composición de \omterm{def:b3:topology:continuity}{mapas continuos}
(la inversión es \omterm{def:b3:topology:continuity}{continuo},
\cref{prop:b3:banach:neumann}): $\varphi \in \mathcal C^1$;
Al arrancar la misma fórmula se obtiene $\mathcal C^k$.
\end{proof}

\begin{theorem}[Teorema de función implícita]
\label{thm:b3:submanifolds:implicit}
Sea $F \colon U \subseteq \R^p\times\R^q \to \R^q$
$\mathcal C^1$ cerca de $(a, b)$, $F(a,b) = 0$, y supongamos que
diferencial parcial $D_yF(a,b) \in \mathcal L(\R^q)$ es
reversible. Luego están \omterm{def:b3:topology:topology}{barrios} $A \ni a$, $B \ni b$
y una aplicación $\mathcal C^1$ $\psi \colon A \to B$ con
\[
\bigl\{(x, y)\in A\times B : F(x,y) = 0\bigr\}
= \{(x, \psi(x)) : x \in A\},
\]
y $D\psi(x) = -D_yF(x, \psi(x))^{-1}\,D_xF(x,
\psi(x))$.\index{teorema de la función implícita}
\end{theorem}

\begin{proof}
Aplicar \cref{thm:b3:submanifolds:ift} a $\Theta(x, y) = (x,
F(x,y))$: su diferencial en$(a,b)$, bloque-triangular con
Los bloques diagonales invertibles $I$ y $D_yF$ son invertibles. el
el inverso local tiene la forma $\Theta^{-1}(x, z) = (x, h(x,
z))$; establecer$\psi(x) = h(x, 0)$: entonces$F(x, y) = 0$ si y así
$\Theta(x,y) = (x, 0)$ y si $y = \psi(x)$, localmente. el
fórmula: diferenciar $F(x, \psi(x)) = 0$ por la cadena
regla.
\end{proof}

\section{Subvariedades: cuatro definiciones}

\begin{theorem}[Caracterizaciones equivalentes]
\label{thm:b3:submanifolds:characterizations}
Sean $M \subseteq \R^n$, $d \in \{0, \dots, n\}$ y $k \geq
1$. Los siguientes son equivalentes, para cada punto$a \in M$
(y $M$ es \emph{$d$-dimensional submanifold of class
$\mathcal C^k$}\index{subcolector} si se mantienen en cada $a
\in M$):
\begin{enumerate}
  \item (\emph{Alisado}) Hay un $\mathcal C^k$
        difeomorfismo $\Phi$ de un $\Omega \ni a$ abierto
        en un $\Omega' \subseteq \R^n$ abierto con
        \[
        \Phi(M\cap\Omega) = \Omega' \cap
        \bigl(\R^d\times\{0\}\bigr).
        \]
  \item (\emph{conjunto de niveles}) Hay un $\mathcal C^k$
        \emph{sumersión}$F \colon \Omega \to \R^{n-d}$
        (es decir,\ $DF(x)$ sobreyectiva) en un $\Omega \ni a$ abierto
        con $M\cap\Omega = F^{-1}(0)$.
  \item (\emph{Gráfico}) Hasta permutar coordenadas, $M$ es
        localmente el gráfico de una aplicación $\mathcal C^k$ $\psi
        \colon A \subseteq \R^d \to \R^{n-d}$.
  \item (\emph{Parametrización}) Hay un $\mathcal C^k$
        \emph{inmersión} $\varphi \colon A \subseteq \R^d \to
        \R^n$(inyectiva$D\varphi(u)$) con$A$ abierto,
        $\varphi$ a \omterm{def:b3:topology:continuity}{homeomorfismo} de $A$ a $M \cap
        \Omega$para algunos$\Omega \ni a$ abiertos.
\end{enumerate}
\end{theorem}

\begin{proof}
(1)$\Rightarrow$(2): $F = (\Phi_{d+1}, \dots, \Phi_n)$ (último
coordenadas de $\Phi$): una inmersión ($D\Phi$ invertible).
(2)$\Rightarrow$(3): $DF(a)$ sobreyectiva: algunos $q \times q$
menor del jacobiano es invertible ($q = n - d$); después
permutando coordenadas, $D_yF(a)$ es invertible, y el
teorema de la función implícita (\cref{thm:b3:submanifolds:implicit})
expresa $M$ localmente como un gráfico $y = \psi(x)$.
(3)$\Rightarrow$(4): $\varphi(x) = (x, \psi(x))$: un
inmersión (inyección diferencial $\bigl(\begin{smallmatrix}I\\
D\psi\end{smallmatrix}\bigr)$), un \omterm{def:b3:topology:continuity}{homeomorfismo} en
el gráfico (inverso: la proyección, \omterm{def:b3:topology:continuity}{continuo}).
(4)$\Rightarrow$(1): sea $\varphi(u_0) = a$; \omterm{def:b3:complete:complete}{completo}
$\operatorname{im}D\varphi(u_0)$ por un suplemento $E$
($\dim E = n - d$) y defina $\Theta(u, v) = \varphi(u) + v$
en $A\times E$: $D\Theta(u_0, 0)$ es biyectivo (imagen
contiene $\operatorname{im}D\varphi(u_0)$ y $E$), por lo que
$\Theta$ es un difeomorfismo local
(\cref{thm:b3:submanifolds:ift}); su inverso $\Phi$
se endereza: cerca de $a$, los puntos de $M$ son exactamente los
$\varphi(u) = \Theta(u, 0)$ --- para esto, el \omterm{def:b3:topology:continuity}{homeomorfismo}
La hipótesis en (4) garantiza que $M\cap\Omega$, para pequeñas
$\Omega$, no contiene otras hojas ($\varphi(u') + v = m \in
M$cerca de$a$con$v \neq 0$ pequeño deben excluirse:
$m = \varphi(u'')$ para algunos $u''$ cerca de $u_0$ por el
Propiedad \omterm{def:b3:topology:continuity}{homeomorfismo} e inyectividad local de $\Theta$
fuerzas $v = 0$). Luego $\Phi(M\cap\Omega) = (A\times\{0\})
\cap\Phi(\Omega)$ hasta encogerse.
\end{proof}

\begin{example}\label{ex:b3:submanifolds:examples}
La esfera $S^{n-1} = \{\norm x^2 = 1\}$: conjunto de niveles de la
inmersión $F(x) = \norm x_2^2 - 1$ en
$\R^n\setminus\{0\}$ ($DF(x) = 2x^{\mathsf T} \ne 0$):
un subcolector $\mathcal C^\infty$ de dimensión $n - 1$. el
toro en $\R^3$: conjunto de niveles de $\bigl(\sqrt{x^2 + y^2} -
R\bigr)^2 + z^2 - r^2$($0 < r < R$). El cono$\{x^2 + y^2 =
z^2\}$es \emph{no} un subcolector en$0$
(\cref{exo:b3:submanifolds:1}). Grupos de matrices: $SL_n$ y
$O_n$ son subvariedades de $M_n(\R)$
(\cref{exo:b3:submanifolds:5,exo:b3:submanifolds:6}) --- el
Punto de partida de la teoría de la mentira.
\end{example}

\section{Espacios tangentes}

\begin{definition}\label{def:b3:submanifolds:tangent}
Sea $M$ una subcolectora $d$ y $a \in M$. El \emph{tangente
espacio}\index{espacio tangente} $T_aM$ es el conjunto de velocidad
vectores $\gamma'(0)$ de $\mathcal C^1$ curvas $\gamma \colon
\intoo{-\varepsilon}\varepsilon \to M$con$\gamma(0) = a$.
\end{definition}

\begin{proposition}\label{prop:b3:submanifolds:tangent}
$T_aM$ es un subespacio vectorial de dimensión $d$ de $\R^n$ y:
\begin{enumerate}
  \item si $M = F^{-1}(0)$ localmente con $F$ una inmersión:
        $T_aM = \ker DF(a)$;
  \item si $M$ está parametrizado por inmersión $\varphi$
        ($\varphi(u_0) = a$): $T_aM =
        \operatorname{im}D\varphi(u_0)$.
\end{enumerate}
\end{proposition}

\begin{proof}
Las curvas en $M$ satisfacen $F(\gamma(t)) = 0$; la regla de la cadena en
$0$ da $DF(a)\gamma'(0) = 0$: $T_aM \subseteq \ker DF(a)$.
Por el contrario, el enderezamiento (\cref{thm:b3:submanifolds:%
characterizations}(1)) transporta líneas de
$\R^d\times\{0\}$ a curvas de $M$: cada vector de a

\begin{theorem}[Multiplicadores de Lagrange]
\label{thm:b3:submanifolds:lagrange}
Sea $M = F^{-1}(0)$ con $F = (F_1, \dots, F_q) \colon \Omega
\to \R^q$una inmersión$\mathcal C^1$, y sea$f \colon
\Omega \to \R$ $\mathcal C^1$. Si la restricción
$f\restriction_M$ tiene un extremo local en $a \in M$, entonces
hay reales únicos $\lambda_1, \dots, \lambda_q$
(\emph{\omterm{thm:b3:submanifolds:lagrange}{Multiplicadores de Lagrange}}) con\index{Multiplicadores de Lagrange}
\[
\nabla f(a) = \lambda_1\nabla F_1(a) + \dots +
\lambda_q\nabla F_q(a) .
\]
\end{theorem}

\begin{proof}
Para cada curva $\gamma$ en $M$ a $a$: $t \mapsto
f(\gamma(t))$tiene un extremo local en$0$, por lo que$0 =
\frac{\dd}{\dd t}f(\gamma(t))\big|_0 = \langle\nabla f(a),
\gamma'(0)\rangle$:$\nabla f(a) \perp T_aM = \ker DF(a)$
(\cref{prop:b3:submanifolds:tangent}). Ahora $\ker DF(a)^\perp
= \operatorname{im}DF(a)^{\mathsf T}$ (el rango/ortogonalidad
identidad del Año 2, o \cref{exo:b3:hilbert:8} en finito
dimensión: $(\ker T)^\perp = \operatorname{im}T^*$), que es
abarcado por los gradientes $\nabla F_i(a)$ --- independiente, como
$DF(a)$ es sobreyectivo: los multiplicadores existen y son únicos.
\end{proof}

\begin{example}[El teorema espectral, geométricamente]
\label{ex:b3:submanifolds:rayleigh}
Sea $A$ una matriz $n\times n$ simétrica real y maximice
$f(x) = \langle Ax, x\rangle$ en la esfera $S^{n-1}$
(\omterm{def:b3:topology:compact}{compacto}: se alcanza el máximo, en algún momento $v_1$). Lagrange
con $F(x) = \norm x^2 - 1$: $\nabla f = 2Ax$ y $\nabla F =
2x$dan$Av_1 = \lambda_1v_1$ --- un vector propio, con
$\lambda_1 = \max_{S^{n-1}}\langle Ax, x\rangle$. Restringir
$A$ a $v_1^\perp$ (invariante: $\langle Av, v_1\rangle =
\langle v, Av_1\rangle = \lambda_1\langle v, v_1\rangle = 0$)
e iterar: una base ortonormal de vectores propios. el
Teorema espectral del año 2, reprobado mediante optimización pura.
--- y la sombra de dimensión infinita del mismo argumento
probado \cref{lem:b3:spectral:existence}.
\end{example}

\begin{method}\label{met:b3:submanifolds:method}
Para demostrar que un conjunto es una subvariedad: exhibirlo localmente como
$F^{-1}(0)$ con $DF$ sobreyectiva \emph{en el set} (la más
ruta común), o como un gráfico. Para calcular su dimensión y
\omterm{def:b3:submanifolds:tangent}{espacio tangente}: $d = n - q$ y $T_a = \ker DF(a)$. a
optimizarlo: Lagrange --- siempre verifique la compacidad (o
coercitividad) primero, por lo que existe un extremo al cual el teorema
puede aplicarse, y recuerde que la ecuación multiplicadora es sólo
\emph{necesario}: recopile todos los puntos críticos y luego compare
valores. Para grupos de matrices, diferencie las curvas en el
identidad para identificar \omterm{def:b3:submanifolds:tangent}{espacios tangentes}.
\end{method}

velocidad $AK$ \emph{(verify $\exp(tK) \in O_n$ using
$\exp(X)^{\mathsf T} = \exp(X^{\mathsf T})$ and
$\exp(-X)\exp(X) = I$)}: todo vector tangente se realiza por
una curva explícita, sin necesidad de un teorema de función implícito.
\end{exercise}

\section{Problema: \texorpdfstring{$SO(3)$}{SO(3)} y el
cuaterniones}

\begin{problem}[{Problema fin de semana --- rotaciones, el grupo
$S^3$, y la doble tapa}]\label{pb:b3:submanifolds:1}
Los cuaterniones $\mathbb H = \{t + x\mathrm i + y\mathrm j +
z\mathrm k\}$ --- el álgebra cuyo grupo unitario contiene
\cref{pb:b3:groups:1}'s $Q_8$ --- parametrizar
rotaciones tridimensionales dos veces: la aplicación ``conjugado
por un cuaternión unitario '' es un morfismo sobreyectivo $S^3 \to
SO(3)$con núcleo$\{\pm1\}$. Construimos todo.
Recordar/definir: la multiplicación es $\R$-bilineal con
$\mathrm i^2 = \mathrm j^2 = \mathrm k^2 =
\mathrm{ijk} = -1$; el conjugado de$q = t + x\mathrm i +
y\mathrm j + z\mathrm k$es$\bar q = t - x\mathrm i -
y\mathrm j - z\mathrm k$;$N(q) = q\bar q = t^2 + x^2 + y^2
+ z^2$.

\medskip
\noindent\textbf{Parte I --- The algebra $\mathbb H$ and the
group $S^3$.}
\begin{enumerate}
  \item Verifique que $\mathbb H$ sea asociativo
        $\R$-álgebra con \omterm{ex:b3:groups:actions}{centro} $\R$, ese $\overline{pq} =
        \bar q\,\bar p$y ese$N(pq) = N(p)N(q)$
        \emph{(one clean route: represent $q$ as the
        $2\times2$ complex matrix
        $\bigl(\begin{smallmatrix}\alpha & \beta\\
        -\bar\beta & \bar\alpha\end{smallmatrix}\bigr)$,$q =
        \alpha + \beta\mathrm j$, and use$\det$)}.
  \item Deduzca que todo $q \neq 0$ es invertible ($q^{-1} =
        \bar q/N(q)$):$\mathbb H$ es un (no conmutativo)
        campo, y $S^3 = \{N(q) = 1\}$ es un grupo ---
        y un subcolector \omterm{def:b3:topology:compact}{compacto} $3$ de $\R^4$
        (\cref{ex:b3:submanifolds:examples}).
\end{enumerate}

\medskip
\noindent\textbf{Parte II --- The rotation morphism.} Identificar
$\R^3$ con \emph{cuaterniones puros} $P = \{x\mathrm i +
y\mathrm j + z\mathrm k\}$, y para$q \in S^3$ definir
$\rho_q(v) = q\,v\,\bar q$.
\begin{enumerate}[resume]
  \item Muestre que $\rho_q$ asigna $P$ a $P$ \emph{(pure
        quaternions are those with $\bar v = -v$)}, es
        $\R$-lineal, conserva la norma y que $\rho
        \colon q \mapsto \rho_q$es un morfismo de grupo$S^3
        \to O(3)$.
  \item Calcular el kernel: $\rho_q = \mathrm{id}$ y si $q$
        viaja con $\mathrm i, \mathrm j, \mathrm k$ si y así
        $q \in \R\cap S^3 = \{\pm1\}$.
  \item Escribir $q = \cos\frac\theta2 + \sin\frac\theta2\,u$
        con $u \in P$, $N(u) = 1$ (¿por qué siempre sucede esto?)
        posible para $q \in S^3$?). Mostrar que $\rho_q$ corrige
        $u$ y, en el avión $u^\perp\cap P$, actúa como
        rotación del ángulo $\theta$ \emph{(compute
        $\rho_q(w)$ for $w \perp u$ using $uw = -wu$ for
        orthogonal pure units --- prove this identity from
        the multiplication table, or from $uw + wu =
        -2\langle u, w\rangle$)}.
  \item Conclusión: $\operatorname{im}\rho \subseteq SO(3)$
        (cada $\rho_q$ es una rotación con eje y ángulo como
        calculado --- determinante $+1$ por \omterm{def:b3:topology:continuity}{continuidad} de $q
        \mapsto \det\rho_q$en \omterm{def:b3:topology:connected}{conectado}$S^3$, o
        directamente), y $\rho$ es \emph{sobre} $SO(3)$: cada
        La rotación de $\R^3$ tiene un eje (probar: un verdadero
        $3\times3$ matriz ortogonal con $\det = 1$ tiene
        valor propio $1$ --- considere la característica
        polinomio) y por lo tanto es algo $\rho_q$. Resumen:
        \[
        SO(3) \;\cong\; S^3/\{\pm 1\} .
        \]
\end{enumerate}

\medskip
\noindent\textbf{Parte III --- $SO(3)$ as a submanifold;
Rodrigues.}
\begin{enumerate}[resume]
  \item Demuestre que $SO(3)$ es un \omterm{def:b3:topology:compact}{compacto} $3$-dimensional
        subcolector de $M_3(\R)$ con $T_ISO(3) =$
        matrices antisimétricas
        (\cref{exo:b3:submanifolds:5}; el determinante
        condición selecciona una unión de componentes).
  \item Para la matriz antisimétrica $A_u$ asociada a
        $u \in \R^3$ ($A_uv = u\wedge v$, la cruz
        producto), prueba \emph{La fórmula de Rodrigues}:
        \[
        \eu^{\theta A_u} = I + \sin\theta\,A_u + (1 -
        \cos\theta)\,A_u^2 \qquad (\norm u = 1)
        \]
        \emph{(de $A_u^3 = -A_u$: dividir el exponencial
        serie a lo largo de los poderes de $A_u$)} e identifíquelo como el
        rotación del eje $u$ y ángulo $\theta$. deducir eso
        $\exp$ mapea las matrices antisimétricas \emph{sobre}
        $SO(3)$.
  \item Relacione las dos parametrizaciones: muestre que $t
        \mapsto \rho_{q(t)}$con$q(t) = \cos\frac t2 +
        \sin\frac t2\,u$ es un grupo de un solo parámetro
        rotaciones cuya derivada en $t = 0$ es $A_u$
        --- el cuaterniónico y \omterm{thm:b3:ode:matrixexp}{exponenciales matriciales} dicen
        la misma historia a media y máxima velocidad respectivamente.
\end{enumerate}

\medskip
\noindent\textbf{Parte IV --- The double cover, felt.}
\begin{enumerate}[resume]
  \item Muestre que \omterm{def:b3:topology:pathconnected}{camino} $q(t) = \cos\frac t2 + \sin\frac
        t2\,\mathrm k$,$t \in \intcc0{2\pi}$, es un bucle en
        $SO(3)$ (su imagen $\rho_{q(t)}$ vuelve a la
        identidad) cuya sustentación cuaterniónica es \emph{no} a
        bucle: $q(2\pi) = -q(0)$. Continuando con $t = 4\pi$
        cierra el ascensor. Explica en un breve párrafo qué
        esto dice: una rotación $2\pi$ no es \omterm{def:b3:topology:continuity}{continuamente}
        deshacer mientras se realiza una rotación $4\pi$ (la correa
        truco), porque los bucles de $SO(3)$ se detectan en su
        doble tapa $S^3$.
  \item Deduzca también el dividendo práctico: composición de
        rotaciones = multiplicación de cuaterniones
        ($4$ datos de multiplicaciones en lugar de $9$,
        sin desviación de la ortogonalidad) --- verificar en el
        composición de dos cuartos de vuelta aproximadamente $\mathrm i$
        y $\mathrm j$: calcular el eje y el ángulo del
        producto.
\end{enumerate}

\medskip
\noindent\textbf{Part V --- The explicit matrix:
Euler--Rodrigues.} Escriba $q = a + b\mathrm i + c\mathrm j +
d\mathrm k \in S^3$, de modo que$a^2 + b^2 + c^2 + d^2 = 1$.
\begin{enumerate}[resume]
  \item Calcular $\rho_q(\mathrm i)$ en su totalidad a partir del
        tabla de multiplicar; luego obtenga $\rho_q(\mathrm
        j)$y$\rho_q(\mathrm k)$ por el cíclico
        sustitución $\mathrm i \to \mathrm j \to \mathrm k
        \to \mathrm i$,$(b, c, d) \to (c, d, b)$
        \emph{(justify it: cycling $\mathrm i, \mathrm j,
        \mathrm k$extends to an automorphism of$\mathbb
        H$, because the defining relations are cyclically
        symmetric)}. Concluya que la matriz de $\rho_q$
        en la base $(\mathrm i, \mathrm j, \mathrm k)$ es
        el \emph{Matriz de Euler-Rodrigues}
        \[
        R_q = \begin{pmatrix}
        a^2 + b^2 - c^2 - d^2 & 2(bc - ad) & 2(bd + ac)\\
        2(bc + ad) & a^2 - b^2 + c^2 - d^2 & 2(cd - ab)\\
        2(bd - ac) & 2(cd + ab) & a^2 - b^2 - c^2 + d^2
        \end{pmatrix}.
        \]
  \item (Leer una rotación al revés) Demuestre que
        \[
        \operatorname{tr}R_q = 4a^2 - 1 = 1 + 2\cos\theta,
        \qquad
        \tfrac12\bigl(R_q - R_q^{\mathsf T}\bigr) =
        \sin\theta\,A_u,
        \]
        en la notación de las preguntas 5 y 8. Deduzca una
        algoritmo que recupera $\pm q$ de una matriz de rotación
        $R$: el ángulo de la traza; el eje de la
        parte antisimétrica cuando $0 < \theta < \pi$; y,
        cuando $\theta = \pi$, acreditar y utilizar la identidad $R
        + I = 2\,uu^{\mathsf T}$.
  \item Evalúe $R_q$ para el producto $q =
        \frac12(1 + \mathrm i + \mathrm j + \mathrm k)$ de la pregunta 11: a
        Aparece la matriz de permutación. Identificar la rotación
        y conciliar con el eje y el ángulo que se encuentran en
        pregunta 11.
\end{enumerate}

\medskip
\noindent\textbf{Part VI --- Inside $S^3$: $SU(2)$,
\omterm{ex:b3:groups:actions}{clases de conjugación}, exponentials.}
\begin{enumerate}[resume]
  \item Demuestre que la matriz \omterm{def:b3:representations:rep}{representación} de la pregunta 1
        (llámelo $\Phi$) se restringe a un isomorfismo de grupo
        desde $S^3$ hasta \emph{grupo unitario especial}
        \[
        SU(2) = \bigl\{U \in M_2(\C) : U^*U = I,\ \det U =
        1\bigr\}
        \]
        \emph{(for surjectivity, write out the equations
        $U^{-1} = U^*$ and $\det U = 1$ for a general
        $2\times2$ complex matrix)}.
  \item Demuestre que la parte real es una invariante de conjugación
        en $S^3$ --- $\operatorname{Re}(pq\bar p) =
        \operatorname{Re}q$para todos$p \in S^3$ --- y,
        por el contrario, que dos cuaterniones unitarios con el mismo
        la parte real está conjugada en $S^3$ \emph{(reducir a
        moviendo un eje puro unitario sobre otro, que Parte
        II proporciona)}. Describa el \omterm{ex:b3:groups:actions}{clases de conjugación} de
        $S^3$ geométricamente; traducir a $SU(2)$ (nivel
        conjuntos de la traza); y proyecto de $\rho$: dos
        las rotaciones son conjugadas en $SO(3)$ si y solo si
        tienen el mismo ángulo $\theta \in \intcc0\pi$.
  \item Defina $\exp$ en $\mathbb H$ por el exponencial
        serie; comprobar la convergencia absoluta, utilizando
        $\abs{pq} = \abs p\,\abs q$ para $\abs q =
        \sqrt{N(q)}$. Muestre, para una unidad pura$u$y$\theta
        \in \R$,
        \[
        \exp(\theta u) = \cos\theta + \sin\theta\,u ,
        \]
        deduzca que $\exp$ asigna el hiperplano $P$ a
        $S^3$, y verifique que $\rho_{\exp(su)} =
        \eu^{2sA_u}$: el fenómeno del medio ángulo en cuestión
        9 de nuevo.
  \item Para cuaterniones puros $v, w$ probar la regla del producto
        $vw = -\langle v, w\rangle + v\wedge w$, de ahí el
        \omterm{def:b3:groups:derived}{conmutador} identidad $vw - wv = 2\,v\wedge w$; probar
        también $[A_v, A_w] = A_{v\wedge w}$ para las matrices
        de la pregunta 8. Concluya que la derivada de
        $\rho$ en $1$ a lo largo de las curvas $t \mapsto \exp(tv)$
        es el isomorfismo lineal $v \mapsto 2A_v$ de $P$
        sobre las matrices antisimétricas, y que
        transporta el cuaternión \omterm{def:b3:groups:derived}{conmutador} a la matriz
        \omterm{def:b3:groups:derived}{conmutador}.
\end{enumerate}

\noindent\textbf{Part VII --- Global structure.}
\begin{enumerate}[resume]
  \item (Sin sección \omterm{def:b3:topology:continuity}{continuo}) Supongamos que $s \colon SO(3) \to
        S^3$es \omterm{def:b3:topology:continuity}{continuo} con$\rho \circ s =
        \operatorname{id}$. Para el bucle$R(t) =
        \rho_{q(t)}$de la pregunta 10, configure$\varepsilon(t) =
        s(R(t))\,q(t)^{-1}$para$t \in \intcc0{2\pi}$. Mostrar
        que $\varepsilon$ es \omterm{def:b3:topology:continuity}{continuo} con valores en
        $\{\pm1\}$, y derivar una contradicción: no existe
        \omterm{def:b3:topology:continuity}{continuo} elección global de un cuaternión unitario
        representando cada rotación.
  \item (El modelo de bola) Sea $\bar B \subseteq \R^3$ el
        bola cerrada de radio $\pi$ y $E(v) = \eu^{A_v}$,
        con $E(0) = I$. Muestre que $E$ asigna $\bar B$ a
        $SO(3)$, es inyectiva sobre la bola abierta, y sobre la
        La esfera límite identifica exactamente las antípodas:
        $E(\pi u) = E(-\pi u) = 2uu^{\mathsf T} - I$, con
        No hay otras coincidencias. Así $SO(3)$ es la pelota.
        con puntos límite antípodas pegados --- el
        espacio proyectivo $\mathbb{RP}^3$ --- y un diámetro
        se convierte en el bucle no contráctil de la pregunta 10.
  \item Demuestre que $\rho_p\rho_q\rho_p^{-1} = \rho_{pq\bar
        p}$; que las involuciones de$SO(3)$(la$R \neq
        I$con$R^2 = I$) son exactamente las medias vueltas
        $\rho_w$ con $w$ una unidad de cuaternión puro; y eso
        el \omterm{ex:b3:groups:actions}{centro} de $SO(3)$ es trivial.
  \item Demuestre que cada rotación es producto de dos
        medias vueltas: para $q = \cos\frac\theta2 +
        \sin\frac\theta2\,u$, elegir una unidad pura$w \perp
        u$, comprobar que$w' = qw$ sea nuevamente una unidad pura
        cuaternión y verifique $\rho_q =
        \rho_{w'}\rho_w$. ¿Dónde se encuentran los dos ejes y
        ¿Qué ángulo forman?
  \item Concluir el resumen \omterm{def:b3:topology:topology}{topológico}: $SO(3)$ es
        \omterm{def:b3:topology:compact}{compacto} y \omterm{def:b3:topology:pathconnected}{conectado por camino} (da dos pruebas:

\item (Una composición, tres formas) Sea $R_1$ la rotación
        por $\frac\pi2$ sobre el eje $z$ y $R_2$ el
        rotación por $\frac\pi2$ sobre el eje $x$. calcular
        el eje y ángulo de $R_2R_1$: (i) multiplicando
        las dos matrices $3\times3$ y usando
        parte traza/antisimétrica (Parte V); (ii) por
        multiplicar los cuaterniones unitarios correspondientes
        $q_2q_1$. Compruebe que las dos respuestas concuerden: ángulo
        $\frac{2\pi}3$, eje $\frac1{\sqrt3}(1, -1, 1)$.
  \item (El \omterm{ex:b3:conformal:mobius}{Transformación de Cayley}) Para $K$ antisimétrico, muestre
        que $I + K$ es invertible y
        \[
        C(K) = (I - K)(I + K)^{-1} \in SO(n),
        \]
        con $-1$ nunca un valor propio de $C(K)$; muestra que

        matrices en $\{R \in SO(n) : -1 \notin

\end{enumerate}
