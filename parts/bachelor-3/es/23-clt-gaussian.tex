\chapter{Funciones características y límite central
teorema}\label{ch:b3:clt}

La ley de los grandes números dice que los promedios convergen; el central
teorema del límite dice \emph{cómo fluctúan}: el error,
magnificado por $\sqrt n$, es asintóticamente \omterm{def:b3:clt:gaussianvector}{gaussiano} ---
cualquiera que sea la ley por la que se parta. Esta universalidad es la
El hecho más profundo de la probabilidad elemental y su prueba natural.
es analítico de Fourier: el \emph{\omterm{def:b3:clt:cf}{función característica}}
(la transformada de Fourier de una ley) convierte sumas \omterm{def:b3:probability:independence}{independiente}
en productos, y la maquinaria de \cref{ch:b3:fouriertransform}
--- inyectividad, puntos fijos \omterm{def:b3:clt:gaussianvector}{gaussiano} --- convierte puntualmente
convergencia de estos productos en convergencia de leyes
(Teorema de Lévy, demostrado en su totalidad). El capítulo termina con
\omterm{def:b3:clt:gaussianvector}{Vectores gaussianos} y la honesta derivación de la confianza.
intervalos utilizados en todas partes en estadística; el problema del fin de semana
da la segunda prueba de Lindeberg del CLT, con una explícita
tasa de error.

\section{Funciones características}

\varphi_X(\xi) = \E\bigl[\eu^{\iu\xi X}\bigr]

\qquad (\xi \in \R)
\]
(el teorema de transferencia lo calcula a partir de la ley; para un \omterm{ex:b3:lebesgue:gamma}{densidad}
$f$, $\varphi_X(\xi) = \hat f(-\xi)$ en

\end{definition}

\begin{proposition}\label{prop:b3:clt:cfbasics}

\]

$\varphi_X(\xi) = \eu^{\iu m\xi - \sigma^2\xi^2/2}$.
\end{proposition}

\begin{theorem}[Inyectividad]\label{thm:b3:clt:injectivity}
Si $\varphi_X = \varphi_Y$, entonces $X$ y $Y$ tienen el mismo
ley. Más precisamente, para $N \sim \mathcal N(0,1)$ \omterm{def:b3:probability:independence}{independiente}
de $X$ y $\varepsilon > 0$, la variable suavizada $X +
\varepsilon N$ tiene el \omterm{ex:b3:lebesgue:gamma}{densidad}
\[
p_\varepsilon(x) = \frac1{2\pi}\int_\R
\varphi_X(-\xi)\,\eu^{-\varepsilon^2\xi^2/2}\,
\eu^{\iu\xi x}\,\dd\xi ,
\]
determinado únicamente por $\varphi_X$; dejando $\varepsilon \to 0$
recupera la ley de $X$.\index{función característica,
inyectividad}
\end{theorem}

\begin{proof}
$X + \varepsilon N$ tiene el \omterm{ex:b3:lebesgue:gamma}{densidad} $p_\varepsilon(x) =
\E\bigl[g_\varepsilon(x - X)\bigr]$, donde$g_\varepsilon$ es
el $\mathcal N(0, \varepsilon^2)$ \omterm{ex:b3:lebesgue:gamma}{densidad}: efectivamente para Borel
$B$, \omterm{def:b3:probability:independence}{independencia} y Tonelli dan $\P(X + \varepsilon N \in
B) = \int\!\!\int\mathbf 1_B(x + \varepsilon
n)g_1(n)\,\dd n\,\dd\P_X(x) = \int_B\E[g_\varepsilon(t -
X)]\dd t$ (sustituto, luego Tonelli nuevamente). escribiendo
$g_\varepsilon$ por inversión de Fourier de su transformada
(\cref{exo:b3:fouriertransform:4}, reescalado):
$g_\varepsilon(u) = \frac1{2\pi}\int
\eu^{-\varepsilon^2\xi^2/2}\eu^{\iu\xi u}\dd\xi$ y Fubini
(todo dominado por el factor \omterm{def:b3:clt:gaussianvector}{gaussiano}):
\[
p_\varepsilon(x) = \frac1{2\pi}\int_\R
\varphi_X(-\xi)\,\eu^{-\varepsilon^2\xi^2/2}\,
\eu^{\iu\xi x}\,\dd\xi ,
\]
un funcional de $\varphi_X$ solo. Si $\varphi_X =
\varphi_Y$:$X + \varepsilon N$y$Y + \varepsilon N$ tienen
leyes iguales para cada $\varepsilon$; para \omterm{def:b3:topology:continuity}{continuo} acotado
$f$, $\E f(X + \varepsilon N) \to \E f(X)$ como $\varepsilon
\to 0$(convergencia dominada,$X + \varepsilon N \to X$
puntualmente en el espacio del producto), por lo que $\E f(X) = \E f(Y)$ para
todos esos $f$ --- y esto determina la ley: para cada $t$,
aprieta $\mathbf 1_{\intoc{-\infty}t}$ entre los acotados
\omterm{def:b3:topology:continuity}{continuo} rampas $f_k^\pm$ (igual a $1$ en
$\intoc{-\infty}{t \mp \frac1k}$, a $0$ más allá de $t \pm
\frac1k$, afín entre); pasar al límite en$\E
f_k^-(X) \leq F_X(t) \leq \E f_k^+(X)$da$F_X(t) =
F_Y(t)$en cada$t$ donde ambos son \omterm{def:b3:topology:continuity}{continuo}, por lo tanto
en todas partes por \omterm{def:b3:topology:continuity}{continuidad derecha} y \omterm{ex:b3:lebesgue:gamma}{densidad} de común
puntos \omterm{def:b3:topology:continuity}{continuidad} (ambos $F$ tienen muchos saltos contables);
funciones de distribución iguales fuerzan leyes iguales
(\cref{exo:b3:measure:3}, descansando sobre
\cref{thm:b3:measure:uniqueness}).
\end{proof}

\section{Convergencia en la distribución}

\begin{definition}\label{def:b3:clt:cid}
$X_n$ \emph{converge en distribución} (o en la ley) a $X$,
escrito $X_n \Rightarrow X$, si\index{convergencia en
distribución}
\[
\E\bigl[f(X_n)\bigr] \longrightarrow \E\bigl[f(X)\bigr]
\qquad\text{for every bounded continuous } f\colon\R\to\R .
\]
Equivalentemente (\cref{exo:b3:clt:4}): $F_{X_n}(t) \to F_X(t)$
en cada punto \omterm{def:b3:topology:continuity}{continuidad} $t$ de $F_X$. El $X_n$ no necesita
vivir de un \omterm{def:b3:probability:space}{espacio de probabilidad} común: sólo importan las leyes.
\end{definition}

\begin{theorem}[Teorema de selección de Helly]
\label{thm:b3:clt:helly}
Cada secuencia $(F_n)$ de funciones de distribución tiene un
subsecuencia convergente puntualmente, en cada punto \omterm{def:b3:topology:continuity}{continuidad}
del límite, a un \omterm{def:b3:topology:continuity}{continuo a la derecha} $G \colon
\R \to \intcc01$no decreciente --- posiblemente con$G(+\infty) - G(-\infty) <
1$ (la masa puede escapar al infinito).\index{teorema de helly}
\end{theorem}

\begin{proof}
La extracción diagonal da $F_{n_k}(q) \to \ell(q)$ por cada
racional $q$ (valores en el \omterm{def:b3:topology:compact}{compacto} $\intcc01$). Definir
$G(t) = \inf\{\ell(q) : q \in \Q, q > t\}$: no decreciente;
\omterm{def:b3:topology:continuity}{continuo a la derecha} (un mínimo sobre la reducción racional
\omterm{def:b3:topology:topology}{barrios} desde la derecha). En un punto \omterm{def:b3:topology:continuity}{continuidad} $t$ de
$G$: para racionales $q_1 < t < q_2$,
\[
\ell(q_1) \leq \liminf F_{n_k}(t) \leq \limsup F_{n_k}(t)
\leq \ell(q_2),
\]
por monotonicidad de cada $F_{n_k}$. De la definición de
$G$ como mínimo y monotonicidad de $\ell$ en el
racionales: $G(s) \leq \ell(q) \leq G(q)$ siempre que $s < q$.
Al tomar $s < q_1 < t$ se obtiene $\ell(q_1) \geq G(s)$, y
$\ell(q_2) \leq G(q_2)$; dejando que $s \uparrow t$ y $q_2
\downarrow t$, \omterm{def:b3:topology:continuity}{continuidad} de$G$en$t$ expriman ambos
$\liminf$ y $\limsup$ a $G(t)$.
\end{proof}

\begin{lemma}[Estanqueidad desde la función característica]
\label{lem:b3:clt:tightness}
Para cualquier \omterm{def:b3:probability:space}{variable aleatoria} $X$ y $u > 0$:
\[
\P\Bigl(\abs X \geq \frac2u\Bigr) \;\leq\;
\frac1u\int_{-u}^{u}\bigl(1 -
\operatorname{Re}\varphi_X(\xi)\bigr)\,\dd\xi .
\]
\end{lemma}

\begin{proof}
Por Tonelli--Fubini (integrando acotado, región finita en
$\xi$):
\[
\frac1u\int_{-u}^u\bigl(1 -
\operatorname{Re}\varphi_X(\xi)\bigr)\dd\xi
= \E\Bigl[\frac1u\int_{-u}^u(1 - \cos(\xi X))\,\dd\xi\Bigr]
= 2\,\E\Bigl[1 - \frac{\sin(uX)}{uX}\Bigr]
\]
(interprete el corchete como su límite $0$ en $X = 0$). el
el integrando no es negativo ($\abs{\sin t} \leq \abs t$), y
para $\abs{uX} \geq 2$: $1 - \frac{\sin(uX)}{uX} \geq 1 -
\frac1{\abs{uX}} \geq \frac12$. Manteniendo solo el evento
$\{\abs{uX} \geq 2\}$ dentro del \omterm{def:b3:probability:space}{expectativa} por lo tanto
deja al menos $2 \cdot \frac12\,\P(\abs X \geq \frac2u)$,
cual es el reclamo.
\end{proof}

\begin{theorem}[teorema de continuidad de Lévy]
\label{thm:b3:clt:levy}
Sea $(X_n)$ \omterm{def:b3:probability:space}{variables aleatorias} cuyo \omterm{def:b3:clt:cf}{característica
funciones} converge puntualmente: $\varphi_{X_n}(\xi) \to
\varphi(\xi)$por cada$\xi$, donde$\varphi =
\varphi_X$ es el \omterm{def:b3:clt:cf}{función característica} de algún \omterm{def:b3:probability:space}{aleatorio
variable} $X$. Entonces $X_n \Rightarrow X$.\index{Lévy's
continuity theorem}
\end{theorem}

\begin{proof}
\emph{Opresión.} Reparar $\varepsilon > 0$. Dado que $\varphi$ es
\omterm{def:b3:topology:continuity}{continuo} en $0$ con $\varphi(0) = 1$, elija $u > 0$ con
$\frac1u\int_{-u}^u(1 - \operatorname{Re}\varphi) <
\varepsilon$; por convergencia dominada (integrando acotado por
$2$ en el fijo $[-u,u]$), la misma integral para
$\varphi_{X_n}$ es $< 2\varepsilon$ para $n$ grande:
\cref{lem:b3:clt:tightness} da $\P(\abs{X_n} \geq \frac2u)
\leq 2\varepsilon$para$n$ grande y amplía la constante
maneja un número finito de otros: las leyes son \emph{ajustado}
--- no hay fugas masivas.

\emph{Subsecuencias.} Sea $(F_{n_k})$ cualquier subsecuencia; por
Helly (\cref{thm:b3:clt:helly}) extracto $F_{n_{k_j}} \to G$
en puntos \omterm{def:b3:topology:continuity}{continuidad}. Fuerzas de estanqueidad $G(-\infty) = 0$,
$G(+\infty) = 1$ ($G(\frac2u) - G(-\frac2u) \geq 1 -
2\varepsilon$en puntos \omterm{def:b3:topology:continuity}{continuidad}):$G$ es un producto genuino
función de distribución, de algunos \omterm{def:b3:probability:space}{variable aleatoria} $Y$. entonces
$X_{n_{k_j}} \Rightarrow Y$ (\cref{exo:b3:clt:4},
convergencia distributiva de $F$), por lo que
$\varphi_{X_{n_{k_j}}} \to \varphi_Y$ \emph{puntualmente} ($x
\mapsto \eu^{\iu\xi x}$ está acotado \omterm{def:b3:topology:continuity}{continuo}, real y
partes imaginarias por separado); comparando con la hipótesis:
$\varphi_Y = \varphi = \varphi_X$ y inyectividad.
(\cref{thm:b3:clt:injectivity}) da $Y \sim X$, es decir,\ $G =
F_X$.

\emph{Conclusión.} Cada subsecuencia de $(F_n)$ tiene un
subsubsecuencia que converge al \emph{mismo} $F_X$ (en su
\omterm{def:b3:topology:continuity}{continuidad} puntos); por lo tanto $F_n(t) \to F_X(t)$ en cada
\omterm{def:b3:topology:continuity}{continuidad} punto $t$ (una secuencia real cuya totalidad
las subsecuencias tienen subsecuencias con el mismo límite
converge): $X_n \Rightarrow X$.
\end{proof}

\section{El teorema del límite central}

\begin{theorem}[Teorema del límite central]\label{thm:b3:clt:clt}
Sea $(X_n)$ i.i.d.\ con $\E X_1 = m$ y $\V(X_1) =
\sigma^2 \in \intoo0\infty$. Entonces\index{límite central
teorema}
\[
\frac{S_n - nm}{\sigma\sqrt n} \;\Longrightarrow\; \mathcal
N(0, 1) :
\qquad
\P\Bigl(a \leq \frac{S_n - nm}{\sigma\sqrt n} \leq
b\Bigr) \longrightarrow
\frac{1}{\sqrt{2\pi}}\int_a^b\eu^{-x^2/2}\,\dd x
\]
para todos los $a < b$.
\end{theorem}

\begin{proof}
\omterm{ex:b3:groups:actions}{Centro} y normalizar: $Z_i = \frac{X_i - m}{\sigma}$
(i.i.d., media $0$, varianza $1$) y $T_n =
\frac1{\sqrt n}\sum_{i\leq n}Z_i$. Por \omterm{def:b3:probability:independence}{independencia} y el
regla afín (\cref{prop:b3:clt:cfbasics}):
\[
\varphi_{T_n}(\xi) =
\varphi_{Z}\Bigl(\frac{\xi}{\sqrt n}\Bigr)^{n},
\qquad
\varphi_Z(\eta) = 1 - \frac{\eta^2}2 + \eta^2\rho(\eta),\quad
\rho(\eta)\to0 .
\]
Reparar $\xi$ y dejar $a_n = \varphi_Z(\xi/\sqrt n)$, $b_n = 1 -
\frac{\xi^2}{2n}$: ambos tienen módulo$\leq 1$para$n$ grande
($\abs{b_n} \leq 1$ una vez $\xi^2 \leq 4n$; $\abs{a_n} \leq 1$
siempre). La desigualdad elemental $\abs{a^n - b^n} \leq
n\abs{a - b}$para$\abs a, \abs b \leq 1$ (telescópica
$a^n - b^n = \sum a^k(a - b)b^{n-1-k}$) da
\[
\Bigl|\varphi_{T_n}(\xi) - \Bigl(1 -
\frac{\xi^2}{2n}\Bigr)^{n}\Bigr|
\leq n\,\Bigl|\varphi_Z\Bigl(\frac\xi{\sqrt n}\Bigr) - 1 +
\frac{\xi^2}{2n}\Bigr|
= \xi^2\,\Bigl|\rho\Bigl(\frac{\xi}{\sqrt n}\Bigr)\Bigr|
\longrightarrow 0,
\]
mientras que $\bigl(1 - \frac{\xi^2}{2n}\bigr)^n \to
\eu^{-\xi^2/2}$(logaritmo real). Entonces$\varphi_{T_n}(\xi) \to
\eu^{-\xi^2/2} = \varphi_{\mathcal N(0,1)}(\xi)$
(\cref{prop:b3:clt:cfbasics}(d)) por cada $\xi$: Lévy
(\cref{thm:b3:clt:levy}) concluye $T_n \Rightarrow \mathcal
N(0,1)$. Las probabilidades de intervalo siguen ya que$F_{\mathcal
N}$ es \omterm{def:b3:topology:continuity}{continuo} en todas partes.
\end{proof}

\begin{example}[Intervalos de confianza, derivados honestamente]
\label{ex:b3:clt:confidence}
Encuesta $n$ \omterm{def:b3:probability:independence}{independiente} votantes; $\hat p_n = S_n/n$ estima el
verdadero $p$, con $\sigma^2 = p(1-p) \leq \frac14$. El CLT
da, para grande $n$,
\[
\P\Bigl(\abs{\hat p_n - p} \leq
\frac{z}{2\sqrt n}\Bigr)
\;\geq\; \P\Bigl(\Bigl|\frac{S_n - np}{\sigma\sqrt n}\Bigr|
\leq z\Bigr)
\longrightarrow \Phi(z) - \Phi(-z),
\]
donde $\Phi$ es la función de distribución estándar \omterm{def:b3:clt:gaussianvector}{gaussiano}.
Con $z = 1.96$: confianza asintótica $95\%$ y margen
$\frac{1.96}{2\sqrt n} \leq 3\%$ requiere $n \geq
\bigl(\frac{1.96}{0.06}\bigr)^2 \approx 1068$ --- el número
detrás de cada punto ``$\pm3$, $95\%$'' se lee; comparar
$5556$ (\cref{exo:b3:probability:7}) de Chebyshev. el
$\sqrt n$ es universal: para reducir a la mitad el error, cuadriplica el
muestra --- la misma ley que fija el costo de \omterm{ex:b3:probability:sllnapps}{Montecarlo}
(\cref{exo:b3:clt:7}).
\end{example}

\section{Vectores gaussianos}

\begin{definition}\label{def:b3:clt:gaussianvector}
Un vector aleatorio $X = (X_1, \dots, X_d)$ es
\emph{gaussiano}\index{vector gaussiano} si cada lineal
combinación $\langle t, X\rangle = \sum t_iX_i$ es una
(posiblemente degenerada) variable gaussiana real. Su ley es
determinado por el vector medio $m = (\E X_i)$ y el
\emph{matriz de covarianza} $\Sigma = \bigl(\operatorname{Cov}
(X_i, X_j)\bigr)$: efectivamente el \omterm{def:b3:clt:cf}{función característica} del
El vector $\varphi_X(t) = \E\eu^{\iu\langle t, X\rangle}$ es
el valor en $1$ del cf de $\langle t, X\rangle$:
\[
\varphi_X(t) = \exp\Bigl(\iu\langle t, m\rangle -
\tfrac12\,t^{\mathsf T}\Sigma\,t\Bigr),
\]
y $d$-dimensional \omterm{def:b3:clt:cf}{funciones características} son inyectivos
(la misma prueba de suavizado que \cref{thm:b3:clt:injectivity},
Gaussianos coordinados).
\end{definition}

\begin{theorem}\label{thm:b3:clt:gaussianvector}
Sea $X$ un \omterm{def:b3:clt:gaussianvector}{vector gaussiano}.
\begin{enumerate}
  \item Cada imagen afín $AX + b$ es una \omterm{def:b3:clt:gaussianvector}{vector gaussiano}.
  \item Los componentes $X_i$ son \emph{\omterm{def:b3:probability:independence}{independiente}} si y
        sólo si $\Sigma$ es diagonal: para \omterm{def:b3:clt:gaussianvector}{gaussiano} en conjunto
        variables, no correlacionadas $=$ \omterm{def:b3:probability:independence}{independiente}.
  \item Si $\Sigma$ es invertible, $X$ tiene el \omterm{ex:b3:lebesgue:gamma}{densidad}
        $\frac{1}{(2\pi)^{d/2}\sqrt{\det\Sigma}}
        \exp\bigl(-\frac12(x - m)^{\mathsf T}\Sigma^{-1}(x -
        m)\bigr)$.
\end{enumerate}
\end{theorem}

\begin{proof}
(1) Las combinaciones lineales de componentes de $AX + b$ son afines
funciones de combinaciones lineales de $X$: \omterm{def:b3:clt:gaussianvector}{gaussiano} (un afín
imagen de una variable \omterm{def:b3:clt:gaussianvector}{gaussiano} es \omterm{def:b3:clt:gaussianvector}{gaussiano}).
(2) Si $\Sigma$ es diagonal, el \omterm{def:b3:clt:cf}{función característica}
factoriza: $\varphi_X(t) = \prod_i\exp(\iu t_im_i -
\frac12\Sigma_{ii}t_i^2) = \prod\varphi_{X_i}(t_i)$, que es
el \omterm{def:b3:clt:cf}{función característica} de la ley de productos
(\cref{thm:b3:probability:independence} leer detenidamente
$d$-inyectividad dimensional): los componentes son \omterm{def:b3:probability:independence}{independiente}.
Lo contrario es la desaparición de las covarianzas de \omterm{def:b3:probability:independence}{independiente}.
$L^2$ variables.
(3) Diagonalizar $\Sigma = P D P^{\mathsf T}$ ($P$ ortogonal,
$D > 0$ diagonal --- \cref{exo:b3:submanifolds:8}); el
el vector $Y = P^{\mathsf T}(X - m)$ es \omterm{def:b3:clt:gaussianvector}{gaussiano} con
covarianza $D$: por (2) sus componentes son \omterm{def:b3:probability:independence}{independiente}
$\mathcal N(0, d_i)$, por lo que $Y$ tiene el producto \omterm{ex:b3:lebesgue:gamma}{densidad}; empujar
adelante por el $x = m + PY$ que preserva el volumen
(\cref{thm:b3:product:linearchange}, $\abs{\det P} = 1$) y
reescribe el exponente invariantemente.
\end{proof}

\begin{method}\label{met:b3:clt:method}
Para identificar una ley límite: calcule \omterm{def:b3:clt:cf}{funciones características},
toma el límite puntual, reconócelo (\omterm{def:b3:clt:gaussianvector}{gaussiano}
$\eu^{-\sigma^2\xi^2/2}$, Poison
$\eu^{\lambda(\eu^{\iu\xi}-1)}$, exponencial $\frac{\lambda}
{\lambda - \iu\xi}$, \dots) e invocar Lévy. los tres pasos
ritual (\omterm{def:b3:probability:independence}{independencia} $\to$ producto; Taylor en $0$ $\to$
límite exponencial; Lévy $\to$ convergencia en el derecho) demuestra
el CLT, la ley de Poisson de eventos raros
(\cref{exo:b3:clt:5}), y cada teorema del límite clásico de
este curso. Para declaraciones a.s.\, regrese a
Conjunto de herramientas de \cref{ch:b3:probability}: respuesta de los dos capítulos

\end{method}

\section{Problema: la prueba de Lindeberg del CLT, con una
tarifa}

\begin{problem}[{Problema del fin de semana --- el método de reemplazo}]
\label{pb:b3:clt:1}
Lindeberg (1922) demostró el teorema del límite central mediante una idea
de una sencillez que desarma: \emph{swap the summands one at a
time for \omterm{def:b3:clt:gaussianvector}{gaussianos}} y controla cada swap mediante una Taylor
expansión. El método no necesita análisis de Fourier, produce una
tasa de error explícito y hoy impulsa las pruebas de universalidad
a través de la teoría de la probabilidad. Sea $(X_i)$ i.i.d., centrado,
$\V(X_1) = 1$, con $\beta = \E\abs{X_1}^3 < \infty$; dejar
$(N_i)$ sea i.i.d.\ $\mathcal N(0,1)$, \omterm{def:b3:probability:independence}{independiente} del
$X_i$ (existencia: \cref{thm:b3:probability:existence}). conjunto
\[
T_n = \frac{X_1 + \dots + X_n}{\sqrt n},
\qquad
G_n = \frac{N_1 + \dots + N_n}{\sqrt n} \sim \mathcal N(0,1).
\]

\medskip
\noindent\textbf{Parte I --- The swapping identity.} Reparar $f
\in \mathcal C^3_b(\R)$ (tres \omterm{def:b3:topology:continuity}{continuo} acotados
derivados; $M_3 = \sup\abs{f'''}$). Para $0 \leq i \leq n$
definir las sumas híbridas
\[
H_i = \frac{X_1 + \dots + X_i + N_{i+1} + \dots +
N_n}{\sqrt n},
\]
entonces $H_n = T_n$ y $H_0 = G_n$.
\begin{enumerate}
  \item Escriba $H_i = W_i + \frac{X_i}{\sqrt n}$ y $H_{i-1}
        = W_i + \frac{N_i}{\sqrt n}$con$W_i =
        \frac{1}{\sqrt n}\bigl(\sum_{j<i}X_j +
        \sum_{j>i}N_j\bigr)$y observe que$W_i$ es
        \omterm{def:b3:probability:independence}{independiente} del par $(X_i, N_i)$. Justificar.
  \item Taylor con resto integral o Lagrange: para cualquier
        reales $w, h$:
        \[
        \Bigl|f(w + h) - f(w) - f'(w)h -
        \tfrac12f''(w)h^2\Bigr| \leq
        \frac{M_3\,\abs h^3}{6} .
        \]
  \item Aplicar la pregunta 2 dos veces ($h = \frac{X_i}{\sqrt n}$ y
        $h = \frac{N_i}{\sqrt n}$ en $w = W_i$), tome
        \omterm{def:b3:probability:space}{esperanzas de heredar} y utilice \omterm{def:b3:probability:independence}{independencia} más el correspondiente
        de los dos primeros momentos de $X_i$ y $N_i$ para mostrar
        \[
        \bigl|\E f(H_i) - \E f(H_{i-1})\bigr|
        \leq \frac{M_3}{6}\cdot
        \frac{\beta + \gamma}{n^{3/2}},
        \qquad \gamma = \E\abs{N_1}^3 =
        \frac{2\sqrt2}{\sqrt\pi} .
        \]
  Telescopio \item sobre $i$ y concluir el \emph{Lindeberg
        atado}:
        \[
        \bigl|\E f(T_n) - \E f(G_n)\bigr| \leq
        \frac{M_3\,(\beta + \gamma)}{6\,\sqrt n} .
        \]
\end{enumerate}

\medskip
\noindent\textbf{Parte II --- From smooth $f$ to the CLT.}
\begin{enumerate}[resume]
  \item Muestre que $\E f(T_n) \to \E f(N)$ para cada $f \in
        \mathcal C_b^3$ y actualice a todos los límites
        \omterm{def:b3:topology:continuity}{continuo} $f$: dados tales $f$ y $\varepsilon$,
        construir $f_\varepsilon \in \mathcal C^3_b$ con
        $\norm{f - f_\varepsilon}_\infty \leq \varepsilon$
        en un intervalo grande --- por ejemplo\ convoluciona $f$ con un
        $\mathcal C^\infty$ golpe
        (\cref{thm:b3:lp:regularization}) --- y manejar el
        colas por tensión ($\V(T_n) = 1$ y Chebyshev).
        Concluye $T_n \Rightarrow \mathcal N(0, 1)$: el
        Teorema del límite central, nuevamente demostrado.
  \item ¿De dónde se utilizó la prueba de que están los $X_i$?
        \emph{idénticamente} distribuido? Demuestra que apenas
        hizo: indicar y probar la versión para \omterm{def:b3:probability:independence}{independiente},
        centrado, no idéntico $X_i$ con
        $\sum_i\V(X_i) = s_n^2$ y terceros momentos, obteniendo
        el error $\frac{M_3}{6s_n^3}\sum_i\bigl(\E\abs{X_i}^3
        + \V(X_i)^{3/2}\gamma\bigr)$ --- Lindeberg es cierto
        teorema en su forma de Lyapunov.
\end{enumerate}

\medskip
\noindent\textbf{Parte III --- Quantitative dividends.}
\begin{enumerate}[resume]
  \item (Funciones de distribución) Dejemos que $t \in \R$ y
        aproximado $\mathbf 1_{\intoc{-\infty}t}$ arriba y
        abajo por $\mathcal C^3_b$ rampas de ancho $\delta$
        (constrúyalos, con $M_3 = O(\delta^{-3})$).
        Combinando con la Parte I, obtenga la cota de dos términos
        \[
        \sup_{t\in\R}\,\bigl|\P(T_n \leq t) -
        \Phi(t)\bigr| \;\leq\;
        \frac{C_1(\beta + \gamma)}{\delta^3\sqrt n} +
        C_2\,\delta
        \qquad (\text{every } \delta > 0),
        \]
        con constantes explícitas (el término $C_2\delta$ usa
        que $\Phi$ tiene \omterm{ex:b3:lebesgue:gamma}{densidad} delimitado por
        $\frac1{\sqrt{2\pi}}$) y optimice $\delta \sim
        n^{-1/8}$ para obtener una tasa uniforme de pedidos.

\begin{enumerate}[resume]

        Z$con$Z = \frac{N_{i+1} + \dots + N_n}{\sqrt

        \;\leq\; \sqrt{\frac{n}{2\pi(n - i)}}\;

  \item (Coincidencia de un momento más) Supongamos además $\E

\end{enumerate}

\medskip
\noindent\textbf{Part V --- Two dimensions: the
multidimensional CLT, by swapping.} Ahora deja que el $X_i$ sea
i.i.d.\ \emph{vectores} aleatorio centrado de $\R^2$ con

\begin{enumerate}[resume]
  \item (\omterm{def:b3:clt:gaussianvector}{Vectores gaussianos}, bajo pedido) Diagonalizar $\Sigma =
        PDP^{\mathsf T}$ (\cref{exo:b3:submanifolds:8}) y
        establezca $C = P\sqrt DP^{\mathsf T}$. Para $Z = (Z^1,

        (\cref{thm:b3:probability:existence}), muestra que $N
        = CZ$ es un \omterm{def:b3:clt:gaussianvector}{vector gaussiano}
        (\cref{def:b3:clt:gaussianvector}) de la media $0$,

        \infty$; y que$G_n = \frac{N_1 + \dots +

        \emph{exactamente} para i.i.d.\ copias $N_i$.
  \item (Taylor en dos variables) Para $f \colon \R^2 \to
        \R$de clase$\mathcal C^3$con$M_3 =

        h\rangle - \tfrac12\langle h, D^2f(w)\,h\rangle

        \abs{h_2}\bigr)^3 \leq \frac{\sqrt2\,M_3}3\,
        \norm h^3
        \]
        \emph{(study $t \mapsto f(w + th)$ on
        $\intcc01$)}.
  \item (El CLT en $\R^2$) Ejecute el esquema de reemplazo en
        los híbridos vectoriales $H_i$: muestran que el primero y el
        términos de segundo orden se cancelan (medias y covarianzas)
        coincidencia), telescopio y actualización como en la pregunta 5
        (tensión de $\E\norm{T_n}^2 =
        \operatorname{tr}\Sigma$; apaciguamiento ahora en
        $\R^2$, \cref{thm:b3:lp:regularization}) a
        concluir: para cada \omterm{def:b3:topology:continuity}{continuo} $f \colon
        \R^2 \to \R$ acotado,
        \[
        \E\,f\Bigl(\frac{X_1 + \dots + X_n}{\sqrt n}\Bigr)
        \longrightarrow \E\,f(N), \qquad N \sim \mathcal
        N(0, \Sigma) :
        \]
        \cref{thm:b3:clt:multiclt} en dimensión $2$, con un
        tasa para $f$ suave y sin análisis de Fourier.
  \item (Cramér--Wold y una fluctuación conjunta) Deducir
        que $\langle t, \frac{S_n}{\sqrt n}\rangle
        \Rightarrow \mathcal N(0, t^{\mathsf T}\Sigma t)$
        por cada $t \in \R^2$ fijo. Aplicación: para
        i.i.d.\ real $(\xi_i)$, centrado, $\E\xi_1^2 = 1$,
        $\E\xi_1^6 < \infty$ (para que la Parte V se aplique a
        $V_i = (\xi_i, \xi_i^2 - 1)$), mostrar
        \[
        \frac1{\sqrt n}\Bigl(\sum_{i\leq n}\xi_i,\
        \sum_{i\leq n}(\xi_i^2 - 1)\Bigr) \Longrightarrow
        \mathcal N\Bigl(0, \begin{pmatrix} 1 & \E\xi_1^3\\
        \E\xi_1^3 & \E\xi_1^4 - 1\end{pmatrix}\Bigr) :
        \]
        media empírica y segundo momento empírico
        fluctúan conjuntamente de forma gaussiana --- independientemente en
        el límite si y sólo si $\E\xi_1^3 = 0$
        (\cref{thm:b3:clt:gaussianvector}).
\end{enumerate}

\medskip
\noindent\textbf{Part VI --- The delta method.}
\begin{enumerate}[resume]
  \item Sea $(\hat\theta_n)$ \omterm{def:b3:probability:space}{variables aleatorias} con
        $\sqrt n(\hat\theta_n - \theta) \Rightarrow
        \mathcal N(0, \sigma^2)$ para un parámetro real
        $\theta$, y sea $g$ diferenciable en
        $\theta$. Pruebe el \emph{método delta}:
        \[
        \sqrt n\bigl(g(\hat\theta_n) - g(\theta)\bigr)
        \Longrightarrow \mathcal N\bigl(0,
        g'(\theta)^2\sigma^2\bigr)
        \]
        \emph{(write $g(x) - g(\theta) = (g'(\theta) +
        \eta(x))(x - \theta)$with$\eta \to 0$ at
        $\theta$; show $\hat\theta_n \to \theta$, then
        $\eta(\hat\theta_n) \to 0$, in probability; finish
        with Slutsky, \cref{exo:b3:clt:8}, and
        \cref{exo:b3:clt:4}(b))}.
  \item Aplicaciones. (a) Para i.i.d.\ real $(\xi_i)$ con
        media $\mu$ y varianza $\sigma^2$, y $\bar X_n =
        \frac1n\sum_{i\leq n}\xi_i$: muestra$\sqrt n(\bar
        X_n^2 - \mu^2) \Rightarrow \mathcal N(0,
        4\mu^2\sigma^2)$cuando$\mu \neq 0$, y eso para
        $\mu = 0$ la afirmación correcta vive en otro
        escala: $n\bar X_n^2 \Rightarrow \sigma^2N^2$ con
        $N \sim \mathcal N(0,1)$ (identifique el límite
        función de distribución). (b) (Varianza
        estabilización) Para $\hat p_n$ la frecuencia de éxito
        de una muestra $\mathcal B(1, p)$, $p \in \intoo01$:
        mostrar que $g(p) = \arcsin\sqrt p$ satisface
        \[
        \sqrt n\,\bigl(g(\hat p_n) - g(p)\bigr)
        \Longrightarrow \mathcal N\Bigl(0, \frac14\Bigr)
        \]
        \emph{lo que} $p$ --- una barra de error asintótica
        libre del parámetro desconocido; comparar con
        \cref{ex:b3:clt:confidence}.
\end{enumerate}

\medskip
\noindent\textbf{Part VII --- Poisson, by the same method:
Le Cam's theorem.} El reemplazo conoce una segunda universalidad
clase: sumas de muchos eventos \omterm{def:b3:probability:independence}{independiente} \emph{extraño}. Para
leyes sobre $\N$ la distancia correcta es \emph{variación total},
\[
d_{\mathrm{TV}}(\mu, \nu) = \sup_{A\subseteq\N}\,
\abs{\mu(A) - \nu(A)} .
\]
\begin{enumerate}[resume]
  \item Demuestre que $d_{\mathrm{TV}}(\mu, \nu) =
        \frac12\sum_{k\geq0}\abs{\mu(\{k\}) -
        \nu(\{k\})}$ y pruebe el vínculo del acoplamiento: para
        \emph{cualquier} par $(X, Y)$ de \omterm{def:b3:probability:space}{variables aleatorias} con
        leyes $\mu$ y $\nu$ en el mismo espacio,
        $d_{\mathrm{TV}}(\mu, \nu) \leq \P(X \neq Y)$.
  \item Calcule exactamente, para $p \in \intoo01$:
        \[
        d_{\mathrm{TV}}\bigl(\mathcal B(1, p), \mathcal
        P(p)\bigr) = p\bigl(1 - \eu^{-p}\bigr) \leq p^2 .
        \]
  \item (Le Cam, mediante intercambio) Deja que $X_i \sim \mathcal B(1,
        p_i)$y$Y_i \sim \mathcal P(p_i)$, el$2n$
        variables\omterm{def:b3:probability:independence}{independiente}; $S = X_1 + \dots + X_n$, y
        recordar $Y_1 + \dots + Y_n \sim \mathcal P(\lambda)$
        con $\lambda = \sum_ip_i$ (\cref{exo:b3:clt:1}).
        Intercambiar una coordenada a la vez en el número entero
        híbridos $H_i = Y_1 + \dots + Y_i + X_{i+1} + \dots
        + X_n$: mostrar, por cada$A \subseteq \N$,
        \[
        \abs{\P(H_{i-1} \in A) - \P(H_i \in A)} \leq
        d_{\mathrm{TV}}\bigl(\mathcal B(1, p_i), \mathcal
        P(p_i)\bigr),
        \]
        y concluir \emph{La desigualdad de Le Cam}:
        \[
        d_{\mathrm{TV}}\bigl(\text{law of } S,\ \mathcal
        P(\lambda)\bigr) \leq \sum_{i=1}^np_i^2 .
        \]
  \item Dividendos. (a) Para $p_i = \frac\lambda n$: el
        obligado es $\frac{\lambda^2}n$ --- la ley de lo raro
        eventos (\cref{exo:b3:clt:5}) actualizados a un
        tasa explícita, uniforme en todos los eventos y válida
        para $p_i$ desigual también. (b) Las letras $500$ son
        entregados, cada uno extraviándose independientemente con
        probabilidad $\frac1{500}$: acotó el error del
        modelo de Poisson del parámetro $1$ y estimar el
        Probabilidad de que ninguna letra se extravíe. (c) Cerrar
        El problema: comparar las dos clases de universalidad.
        se reunieron aquí --- \omterm{def:b3:clt:gaussianvector}{gaussiano} (muchos pequeños dispersos
        contribuciones; dos momentos coincidentes; Taylor) y
        Poisson (muchas contribuciones raras; una significa
        emparejado; un acoplamiento exacto de variación total) --- y
        el método de reemplazo único detrás de ambos.

\item (Error relativo y transformación del registro) Let $(X_n)$
        ser i.i.d., positivo, media $\mu > 0$, varianza
        $\sigma^2$ y $\bar X_n$ la media empírica. Mostrar
        por el método delta que
        \[
        \sqrt n\,\bigl(\ln\bar X_n - \ln\mu\bigr)
        \Longrightarrow
        \mathcal N\Bigl(0,\ \frac{\sigma^2}{\mu^2}\Bigr) :
        \]
        el parámetro asintótico de $\ln\bar X_n$ es el
        \emph{coeficiente de variación} $\sigma/\mu$ ---
        error relativo, sin escala. Deducir un $95\%$
        intervalo de confianza para $\mu$ del multiplicativo
        formulario $\bar X_n\cdot\eu^{\pm1.96\,\sigma/(\mu\sqrt
        n)}$, y explique cuándo es preferible al
        uno aditivo.
  \item (El tercer momento dirige el error) Para $X \sim$
        Bernoulli($p$) centrado, calcule $\E\bigl[(X -
        p)^3\bigr] = p(1-p)(1-2p)$. Utilizando el análisis de la Parte IV
        (el error de intercambio se debe a terceros momentos), explique
        ¿Por qué la aproximación normal de $\mathcal B(n, p)$?
        es asimétrico para $p \neq \frac12$ --- sobrepasando
        por un lado, insuficiente por el otro --- y por qué
        $p = \frac12$ disfruta de la velocidad de momento coincidente más rápida.
        Verifique numéricamente el signo del sesgo en $\mathcal
        B(20, 0.1)$con$\mathcal N(2, 1.8)$: compare
        $\P(S = 0) = 0.9^{20}$ con la masa \omterm{def:b3:clt:gaussianvector}{gaussiano} de
        $\intoo{-\infty}{0.5}$.
\end{enumerate}
\end{problem}
