\chapter{Serie de Laurent y el teorema del residuo}
\label{ch:b3:residues}

¿Qué le sucede a un \omterm{def:b3:holomorphic:holo}{función holomorfa} cerca de un punto donde
¿Está definido \emph{no}? La respuesta es una tricotomía \omterm{def:b3:complete:complete}{completo} ---
punto removible, polo o singularidad esencial --- leer una
Serie de potencias de dos colas, la expansión de Laurent. Un coeficiente
de esa expansión, el \emph{\omterm{def:b3:residues:singularities}{residuo}}, controla cada \omterm{def:b3:holomorphic:contour}{contorno
integral} alrededor de la singularidad: el teorema \omterm{def:b3:residues:singularities}{residuo} convierte
integrales definidas duras en álgebra finita, cuenta ceros de
funciones (principio de argumento, Rouché), y demuestra la apertura
teorema de mapeo. Primero actualizamos el teorema de Cauchy desde
dominios en forma de estrella a su forma definitiva y libre de homología ---
El elegante argumento de Dixon --- para que todos los contornos con
el devanado número cero alrededor del complemento esté disponible.

\section{El teorema global de Cauchy}

A \emph{ciclo}\index{ciclo} $\Gamma$ es una suma formal finita de
cerrado \omterm{def:b3:topology:pathconnected}{caminos} $\gamma_1, \dots, \gamma_m$; integrales y
Los índices a lo largo de $\Gamma$ son las sumas correspondientes, y
$\operatorname{im}\Gamma = \bigcup\operatorname{im}\gamma_j$.

\begin{theorem}[Cauchy, forma global]
\label{thm:b3:residues:globalcauchy}
Deje que $\Omega \subseteq \C$ esté abierto, $f \in \mathcal H(\Omega)$,
y $\Gamma$ un ciclo en $\Omega$ tal que
\[
\operatorname{Ind}_\Gamma(w) = 0
\qquad\text{for every } w \notin \Omega .
\]
Entonces, para todos los $z \in \Omega\setminus\operatorname{im}\Gamma$,
\[
\frac1{2\iu\pi}\int_\Gamma\frac{f(w)}{w - z}\,\dd w =
\operatorname{Ind}_\Gamma(z)\,f(z),
\qquad\text{and}\qquad
\int_\Gamma f(w)\,\dd w = 0 .
\]
\index{Teorema de Cauchy (global)}
\end{theorem}

\begin{proof}[Prueba (Dixon)]
Definir $g \colon \Omega\times\Omega \to \C$ por
\[
g(z, w) = \begin{cases}
\dfrac{f(w) - f(z)}{w - z} & w \neq z,\\[4pt]
f'(z) & w = z .
\end{cases}
\]
\emph{$g$ is \omterm{def:b3:topology:continuity}{continuo}}: fuera de la diagonal, claro. Cerca de un
punto diagonal $(a, a)$, expandir $f$ en una serie de potencias en $a$
(\cref{thm:b3:holomorphic:analytic}): $f(w) - f(z) =
\sum_{n\geq1}c_n\bigl((w-a)^n - (z-a)^n\bigr)$, y dividiendo
cada término por $w - z$ (factorización de $u^n - v^n$) da, por
$z, w \in D(a, r)$,
\[
g(z, w) =
\sum_{n\geq1}c_n\sum_{j=0}^{n-1}(w-a)^{\,j}(z-a)^{\,n-1-j},
\]
válido también en la diagonal (cada suma interna se convierte en
$n(z-a)^{n-1}$, sumando $f'(z)$). Para $r$ pequeña la serie
converge uniformemente en $D(a,r)^2$
($\abs{\text{term}} \leq n\abs{c_n}r^{n-1}$, sumable dentro
el radio): la suma es \omterm{def:b3:topology:continuity}{continuo}.

Establecer $h(z) = \frac1{2\iu\pi}\int_\Gamma g(z, w)\,\dd w$ en
$\Omega$: \omterm{def:b3:topology:continuity}{continuo} (uniforme \omterm{def:b3:topology:continuity}{continuidad} de $g$ en \omterm{def:b3:topology:compact}{compactos}),
y \omterm{def:b3:holomorphic:holo}{holomorfo} --- por Morera
(criterio de \cref{thm:b3:holomorphic:weierstrassconv}): para un
triángulo $T \subseteq \Omega$, Fubini da $\int_{\partial
T}h = \frac1{2\iu\pi}\int_\Gamma\bigl(\int_{\partial T}g(z,
w)\dd z\bigr)\dd w = 0$, la integral interna desaparece porque
$z \mapsto g(z, w)$ es \omterm{def:b3:holomorphic:holo}{holomorfo} en $\Omega$ (en $z = w$ el
la singularidad es removible: $g$ es \omterm{def:b3:topology:continuity}{continuo} allí y
\omterm{def:b3:holomorphic:holo}{holomorfo} en otro lugar --- el argumento de extensión de
Prueba de \cref{thm:b3:holomorphic:formula}).

En \omterm{def:b3:topology:topology}{conjunto abierto} $\Omega' = \{z \notin \operatorname{im}\Gamma
: \operatorname{Ind}_\Gamma(z) = 0\}$, defina$h_1(z) =
\frac1{2\iu\pi}\int_\Gamma\frac{f(w)}{w - z}\dd w$:
\omterm{def:b3:holomorphic:holo}{holomorfo} en $\Omega'$ (Morera o diferenciación bajo el
integrales). Para $z \in \Omega\cap\Omega'$:
\[
h(z) = \frac1{2\iu\pi}\int_\Gamma\frac{f(w)}{w-z}\dd w
- f(z)\operatorname{Ind}_\Gamma(z) = h_1(z) .
\]
Por hipótesis $\Omega\cup\Omega' = \C$ ($w \notin \Omega
\Rightarrow \operatorname{Ind}_\Gamma(w) = 0$), entonces$h$ y
$h_1$ pega a una función completa $H$. Desde lo ilimitado
componente del complemento de $\operatorname{im}\Gamma$ se encuentra
en $\Omega'$ y $h_1(z) \to 0$ como $\abs z \to \infty$ (ML
cota), $H$ está acotado y tiende a $0$: Liouville
(\cref{cor:b3:holomorphic:liouville}) da $H \equiv 0$. Así
$h \equiv 0$ en $\Omega$, que es la fórmula integral.
Aplicándolo, para $a \in
\Omega\setminus\operatorname{im}\Gamma$fijo, a$\tilde f(w) =
(w - a)f(w)$en$z = a$:
\[
\frac1{2\iu\pi}\int_\Gamma f(w)\dd w =
\frac1{2\iu\pi}\int_\Gamma \frac{\tilde f(w)}{w - a}\dd w =
\operatorname{Ind}_\Gamma(a)\,\tilde f(a) = 0 .
\]
\end{proof}

\section{Series de Laurent y singularidades aisladas}

\begin{theorem}[Expansión Laurent]\label{thm:b3:residues:laurent}
Sea $f$ \omterm{def:b3:holomorphic:holo}{holomorfo} en el anillo $A = \{r < \abs{z - a} <
R\}$($0 \leq r < R \leq \infty$). Entonces\index{serie laurent}
\[
f(z) = \sum_{n\in\Z}c_n\,(z - a)^n
\qquad\text{on } A,
\qquad
c_n = \frac1{2\iu\pi}\int_{C_\rho}\frac{f(w)}{(w -
a)^{n+1}}\,\dd w
\]
para cualquier $r < \rho < R$ (independiente de $\rho$), los dos
medias series que convergen normalmente en los subannuli \omterm{def:b3:topology:compact}{compacto}. el
La expansión es única.
\end{theorem}

\begin{proof}
Fije $r < \rho_1 < \abs{z - a} < \rho_2 < R$ y deje que $\Gamma =
C_{\rho_2} - C_{\rho_1}$ (exterior en sentido antihorario, interior
en el sentido de las agujas del reloj): un ciclo en $A$ con
$\operatorname{Ind}_\Gamma(w) = 0$ para todos $w \notin A$
(puntos dentro del disco pequeño: $1 - 1 = 0$; fuera del disco grande
uno: $0 - 0$). Por \cref{thm:b3:residues:globalcauchy},
$\operatorname{Ind}_\Gamma(z) = 1 - 0 = 1$ da
\[
f(z) = \frac1{2\iu\pi}\int_{C_{\rho_2}}\frac{f(w)}{w - z}\dd
w - \frac1{2\iu\pi}\int_{C_{\rho_1}}\frac{f(w)}{w - z}\dd w .
\]
Expanda el primer núcleo como en
\cref{thm:b3:holomorphic:analytic} (potencias de $\frac{z -
a}{w - a}$, módulo$< 1$): la parte no negativa
$\sum_{n\geq0}c_n(z-a)^n$. En el segundo, expande el otro.
manera: $\frac{-1}{w - z} = \frac1{(z-a)(1 - \frac{w - a}{z -
a})} = \sum_{m\geq0}\frac{(w-a)^m}{(z - a)^{m+1}}$, normalmente
convergente en $C_{\rho_1}$: la parte negativa
$\sum_{n\leq-1}c_n(z-a)^n$ con los coeficientes indicados (índice
$n = -m-1$). Independencia de $\rho$: las integrales de coeficientes
más de $C_{\rho}$ y $C_{\rho'}$ se diferencian por
$\int_\Gamma$ durante un ciclo de índice nulo del \omterm{def:b3:holomorphic:holo}{holomorfo}
$\frac{f(w)}{(w-a)^{n+1}}$ en $A$: cero, por
\cref{thm:b3:residues:globalcauchy} nuevamente. Unicidad:
integrar $\sum c_n(z-a)^n$ contra $(z - a)^{-m-1}$ sobre
$C_\rho$ término por término (convergencia normal): sólo $n = m$
sobrevive.
\end{proof}

\begin{definition}\label{def:b3:residues:singularities}
Si $f$ es \omterm{def:b3:holomorphic:holo}{holomorfo} en un disco perforado $D(a, R)\setminus
\{a\}$, expanda por Laurent ($r = 0$). Tres exclusivos
casos:\index{singularidad aislada}
\begin{itemize}
  \item todos $c_n = 0$ para $n < 0$: \emph{\omterm{thm:b3:residues:riemanncw}{extraíble
        singularidad}} (la serie no negativa extiende $f$
        \omterm{def:b3:holomorphic:holo}{holomorficamente} a $a$);
  \item $c_n \neq 0$ para un número finito, al menos uno, $n < 0$:
        un \emph{polo} de orden $m = -\min\{n : c_n \neq 0\}$;
        equivalentemente $f = g/(z-a)^m$, $g$ \omterm{def:b3:holomorphic:holo}{holomorfo}, $g(a)
        \neq 0$; equivalentemente$\abs{f(z)} \to \infty$como$z\to
        a$;
  \item infinitos negativos $c_n \neq 0$:
        \emph{singularidad esencial}.
\end{itemize}
El \emph{residuo}\index{residuo} es
$\operatorname{Res}(f, a) = c_{-1}$. Una función \omterm{def:b3:holomorphic:holo}{holomorfo} en
$\Omega$ menos un conjunto de polos es
\emph{meromórfico}\index{función meromórfica} en $\Omega$.
\end{definition}

\begin{theorem}[Riemann; Casorati--Weierstrass]
\label{thm:b3:residues:riemanncw}
Sea $f$ \omterm{def:b3:holomorphic:holo}{holomorfo} en $D(a,R)\setminus\{a\}$.
\begin{enumerate}
  \item (Riemann) Si $f$ es \emph{encerrado} cerca de $a$, el
        la singularidad es removible.\index{singularidad removible}
  \item (Casorati--Weierstrass) Si $a$ es esencial, entonces
        $f\bigl(D(a,\varepsilon)\setminus\{a\}\bigr)$ es denso
        en $\C$ por cada $\varepsilon$.
        \index{Casorati--teorema de Weierstrass}
\end{enumerate}
\end{theorem}

\begin{proof}
(1) Para $n < 0$ y $\rho \to 0$: $\abs{c_n} \leq
M\rho^{-n-1}\cdot\rho\cdot\rho^{-\,n}\dots$ por ML en
$C_\rho$: $\abs{c_n} \leq \frac{1}{2\pi}\,2\pi\rho\cdot
M\rho^{-(n+1)} = M\rho^{-n} \to 0$(como$-n > 0$): todos
los coeficientes negativos desaparecen.
(2) Si no se aproximara algún valor $b$: $\abs{f - b} \geq
\delta$cerca de$a$, entonces$g = 1/(f - b)$ es \omterm{def:b3:holomorphic:holo}{holomorfo} y
delimitado cerca de $a$: extraíble (1), $g$ se extiende con el valor $c$.
Si $c \neq 0$, $f = b + 1/g$ está delimitado cerca de $a$: extraíble
--- excluido. Si $c = 0$, $g$ tiene un cero de orden finito $m$
en $a$ (\cref{thm:b3:holomorphic:identity}; $g \not\equiv 0$),
y $f = b + 1/g$ tiene un polo de orden $m$: excluido nuevamente.
\end{proof}

\section{El teorema del residuo}

\begin{theorem}[Teorema del residuo]\label{thm:b3:residues:residue}
Sea $\Omega$ abierto, $S \subseteq \Omega$ finito, $f \in
\mathcal H(\Omega\setminus S)$y$\Gamma$ un ciclo en
$\Omega\setminus S$ con $\operatorname{Ind}_\Gamma(w) = 0$
por cada $w \notin \Omega$. Entonces\index{teorema del residuo}
\[
\frac{1}{2\iu\pi}\int_\Gamma f(z)\,\dd z
= \sum_{a\in S}\operatorname{Ind}_\Gamma(a)\,
\operatorname{Res}(f, a) .
\]
\end{theorem}

\begin{proof}
Para cada $a \in S$, sea $P_a(z) = \sum_{n\leq-1}c_n^{(a)}(z -
a)^n$la parte principal de$f$en$a$: una serie
convergiendo en $\C\setminus\{a\}$ (su radio en $1/(z-a)$ es
infinito: la cola de Laurent converge para todos los pequeños $\abs{z-a}$
por lo tanto, es una serie de potencias en $(z-a)^{-1}$, en todas partes), y
\omterm{def:b3:holomorphic:holo}{holomorfo} allí. Entonces $g = f - \sum_{a\in S}P_a$ tiene
singularidades removibles en cada punto de $S$ (su Laurent
la expansión en $a$ no tiene parte negativa: los otros $P_{a'}$ son
\omterm{def:b3:holomorphic:holo}{holomorfo} en $a$), por lo que $g$ extiende \omterm{def:b3:holomorphic:holo}{holomorficamente} a
$\Omega$ y \cref{thm:b3:residues:globalcauchy} dan
$\int_\Gamma g = 0$. Queda por integrar cada $P_a$:
término por término (convergencia normal en el \omterm{def:b3:topology:compact}{compacto}
$\operatorname{im}\Gamma$, que evita $a$),
\[
\frac1{2\iu\pi}\int_\Gamma(z - a)^n\,\dd z = 0 \ (n \leq -2:
\text{primitive } \tfrac{(z-a)^{n+1}}{n+1}),
\qquad
\frac1{2\iu\pi}\int_\Gamma\frac{\dd z}{z - a} =
\operatorname{Ind}_\Gamma(a),
\]
entonces $\frac1{2\iu\pi}\int_\Gamma P_a =
c_{-1}^{(a)}\operatorname{Ind}_\Gamma(a)$. Suma sobre$a$.
\end{proof}

\begin{method}[Cálculo de residuos]\label{met:b3:residues:compute}
Polo simple: $\operatorname{Res}(f, a) = \lim_{z\to a}(z -
a)f(z)$; para$f = g/h$con$g(a) \neq 0$,$h(a) = 0$,$h'(a)
\neq 0$:$\operatorname{Res} = g(a)/h'(a)$. Polo de orden$m$:
$\operatorname{Res}(f, a) = \frac1{(m-1)!}\lim_{z\to
a}\bigl((z-a)^mf(z)\bigr)^{(m-1)}$. Singularidades esenciales:
expanda y lea $c_{-1}$ (por ejemplo,\ de series conocidas). siempre
verificar qué polos realmente rodea el contorno, y con
cual índice.
\end{method}

\begin{example}[Los cuatro tipos integrales clásicos]
\label{ex:b3:residues:integrals}
(a) \emph{Rational over $\R$}: para $\int_\R\frac{\dd x}{1 +
x^4}$, cierre con un semicírculo grande$S_R$ en la parte superior.
semiplano: el integrando es $O(R^{-4})$ allí, entonces
$\int_{S_R} \to 0$ (ML), y el teorema \omterm{def:b3:residues:singularities}{residuo} con el
polos $\eu^{\iu\pi/4}, \eu^{3\iu\pi/4}$ (simple, \omterm{def:b3:residues:singularities}{residuos}
$\frac1{4z^3} = \frac{z}{4z^4} = -\frac z4$ en un poste) da
\[
\int_\R\frac{\dd x}{1 + x^4}
= 2\iu\pi\Bigl(-\frac{\eu^{\iu\pi/4}}4 -
\frac{\eu^{3\iu\pi/4}}4\Bigr)
= \frac{\pi}{\sqrt2} .
\]
(b) \emph{tipo fourier}: para $t \geq 0$,
$\int_\R\frac{\eu^{\iu tx}}{1 + x^2}\dd x =
2\iu\pi\operatorname{Res}\bigl(\tfrac{\eu^{\iu tz}}{1+z^2},
\iu\bigr) = 2\iu\pi\frac{\eu^{-t}}{2\iu} = \pi\eu^{-t}$ ---
el semicírculo superior funciona porque $\abs{\eu^{\iu tz}} =
\eu^{-t\operatorname{Im}z} \leq 1$ ahí; tomando partes reales:
$\int_\R\frac{\cos(tx)}{1+x^2}\dd x = \pi\eu^{-\abs t}$,
resolviendo la fórmula admitida de \cref{exo:b3:lebesgue:10}.
(c) \emph{Trigonométrica durante un período}: sustituir $z =
\eu^{\iu t}$,$\cos t = \frac{z + z^{-1}}2$,$\dd t =
\frac{\dd z}{\iu z}$:$\int_0^{2\pi}\frac{\dd t}{a + \cos t}$
($a > 1$) se convierte en un recuento \omterm{def:b3:residues:singularities}{residuo} dentro del círculo unitario
(\cref{exo:b3:residues:2}).
(d) \emph{Serie}: par $f$ con $\pi\cot(\pi z)$, cuyos polos
son los números enteros con \omterm{def:b3:residues:singularities}{residuo} $1$: las sumas del problema del fin de semana
$\sum n^{-2}$ y $\sum n^{-4}$ de esta manera.
\end{example}

\section{El principio del argumento y el teorema de Rouch'e}

\begin{theorem}[Principio de argumento]
\label{thm:b3:residues:argument}
Sea $f$ \omterm{def:b3:residues:singularities}{meromórfico} en $\Omega$, con ceros $z_j$ (órdenes
$m_j$) y polos $p_k$ (pedidos $\mu_k$), y $\gamma$ un cerrado
\omterm{def:b3:topology:pathconnected}{camino} en $\Omega$ evitándolos a todos, con
$\operatorname{Ind}_\gamma = 0$ de $\Omega$. entonces
\index{principio de argumento}
\[
\frac1{2\iu\pi}\int_\gamma\frac{f'(z)}{f(z)}\,\dd z
= \sum_j m_j\operatorname{Ind}_\gamma(z_j)
- \sum_k \mu_k\operatorname{Ind}_\gamma(p_k)
\]
(un finito de términos son distintos de cero). por un simple
contorno en sentido antihorario, la integral cuenta ceros menos
polos en el interior, con multiplicidad --- e igual al devanado
número de la imagen \omterm{def:b3:topology:pathconnected}{camino} $f\circ\gamma$ alrededor de $0$.
\end{theorem}

\begin{proof}
Cerca de un cero de orden $m$: $f = (z-a)^mg$, $g(a) \neq 0$, entonces
$\frac{f'}f = \frac m{z - a} + \frac{g'}g$ con el segundo
término \omterm{def:b3:holomorphic:holo}{holomorfo} cerca de $a$: un polo simple de \omterm{def:b3:residues:singularities}{residuo} $m$. cerca
un polo de orden $\mu$: $f = (z-a)^{-\mu}g$ da \omterm{def:b3:residues:singularities}{residuo}
$-\mu$. En otros lugares $\frac{f'}f$ es \omterm{def:b3:holomorphic:holo}{holomorfo}. (Los ceros y
Los polos con índice distinto de cero se encuentran en una región \omterm{def:b3:topology:compact}{compacto} rodeada por
$\gamma$; por el teorema de la identidad son finitos en número
allí, $f \not\equiv 0$.) Aplicar
\cref{thm:b3:residues:residue}. El último comentario:
$\frac1{2\iu\pi}\int_\gamma\frac{f'}f =
\frac1{2\iu\pi}\int_{f\circ\gamma}\frac{\dd w}w =
\operatorname{Ind}_{f\circ\gamma}(0)$(sustituye$w =
f(\gamma(t))$).
\end{proof}

\begin{theorem}[Rouché]\label{thm:b3:residues:rouche}
Sea $f, g$ \omterm{def:b3:holomorphic:holo}{holomorfo} en $\Omega$ y $\gamma$ un cerrado
\omterm{def:b3:topology:pathconnected}{camino} con $\operatorname{Ind}_\gamma \in \{0,1\}$, cero desactivado
$\Omega$ (un contorno simple). Si\index{Rouché's theorem}
\[
\abs{g(z)} < \abs{f(z)} \qquad \text{on }
\operatorname{im}\gamma,
\]
entonces $f$ y $f + g$ tienen el mismo número de ceros (con
multiplicidad) en la región $\{\operatorname{Ind}_\gamma =
1\}$.
\end{theorem}

\begin{proof}
Para $t \in \intcc01$, $f_t = f + tg$ no tiene cero en
$\operatorname{im}\gamma$ ($\abs{f_t} \geq \abs f - \abs g >
0$), entonces
\[
N(t) = \frac1{2\iu\pi}\int_\gamma
\frac{f_t'(z)}{f_t(z)}\,\dd z
\]
está bien definido; cuenta los ceros en la región cerrada
(\cref{thm:b3:residues:argument}; sin polos). $N$ es
\omterm{def:b3:topology:continuity}{continuo} en $t$ (el integrando es conjuntamente \omterm{def:b3:topology:continuity}{continuo}, el
denominadores uniformemente acotados debajo --- dominados
convergencia) y de valor entero: constante. $N(0) = N(1)$.
\end{proof}

\begin{corollary}[Teorema de mapeo abierto]
\label{cor:b3:residues:openmapping}
Un \omterm{def:b3:holomorphic:holo}{función holomorfa} no constante en un \omterm{def:b3:topology:connected}{conectado} \omterm{def:b3:topology:topology}{conjunto abierto} es
una aplicación abierto. En particular (nuevamente) el principio máximo
tiene, y una biyección \omterm{def:b3:holomorphic:holo}{holomorfo} tiene \omterm{def:b3:holomorphic:holo}{holomorfo}
inverso.\index{teorema de mapeo abierto (holomórfico)}
\end{corollary}

\begin{proof}
Sea $f(a) = b$; $f - b$ tiene un cero de algún orden finito $m
\geq 1$en$a$(teorema de identidad:$f \not\equiv b$). elegir
$r$ con $f - b$ sin cero en $\bar D(a, r)\setminus\{a\}$
(\omterm{thm:b3:holomorphic:identity}{ceros aislados}) y dejar $\delta = \min_{\abs{z - a} =
r}\abs{f(z) - b} > 0$. Para$\abs{w - b} < \delta$: en el
círculo, $\abs{(b - w)} < \delta \leq \abs{f - b}$, entonces
Rouché ($f - b$ versus constante $b - w$) dice que $f - w$ tiene
exactamente $m$ ceros en $D(a, r)$: se alcanza cada uno de esos $w$
--- $f(D(a,r)) \supseteq D(b, \delta)$: abierto. Máximo
principio: un máximo \omterm{def:b3:topology:interior}{interior} de $\abs f$ es imposible para
un $f$ no constante, su imagen alrededor de $f(a)$ contiene puntos
de módulo mayor. Inversa: una biyección \omterm{def:b3:holomorphic:holo}{holomorfo} $f$ es
abierto, por lo que $f^{-1}$ es \omterm{def:b3:topology:continuity}{continuo}; el cero de $f - f(a)$ en
$a$ es simple ($m \geq 2$ le daría a $m$ preimágenes de lugares cercanos
valores --- distintos, ya que $f'$ desaparece solo en
puntos aislados, por lo que cerca de $a$ los ceros $m$ de $f - w$ son
simple y distinto para genérico pequeño $w$: contradicción
con inyectividad); entonces $f'(a) \neq 0$ y la diferencia
cociente de $f^{-1}$ converge: $\bigl(f^{-1}\bigr)'(b) =
1/f'(a)$.
\end{proof}

integrando $f(z) = \frac{z^{a-1}}{1+z} =

El polo $z = -1$ tiene \omterm{def:b3:residues:singularities}{residuo} $\eu^{\iu\pi(a - 1)}$. deducir también

\begin{exercise}[$\star\star$]\label{exo:b3:residues:12}
(Contando ceros con el principio argumental, numéricamente)
Deje $P(z) = z^4 + 8z + 1$.

(b) ¿Cuántos hay en el anillo $1 < \abs z < 3$?

\end{exercise}

\section{Problema: \texorpdfstring{$\zeta(2k)$}{zeta(2k)} por el
cotangente}

\begin{problem}[{Problema de fin de semana --- sumando $\sum n^{-2k}$
con \omterm{def:b3:residues:singularities}{residuos}}]\label{pb:b3:residues:1}
La serie de sumas de teoremas \omterm{def:b3:residues:singularities}{residuo}: emparejamiento de una función racional
con $\pi\cot(\pi z)$, cuyos polos se encuentran en los números enteros, se convierte
$\sum_{n}f(n)$ en un recuento \omterm{def:b3:residues:singularities}{residuo}. Probamos el método y
calcular $\zeta(2) = \frac{\pi^2}{6}$ y $\zeta(4) =
\frac{\pi^4}{90}$ --- los valores encontrados por la serie de Fourier en
Año 2 y por seguimientos de operador en \cref{ch:b3:spectral}, ahora por
integración del contorno.

\medskip
\noindent\textbf{Parte I --- The cotangent kernel.}
\begin{enumerate}
  \item Mostrar que $\pi\cot(\pi z)$ es \omterm{def:b3:residues:singularities}{meromórfico} en $\C$ con
        polos simples exactamente en $z = n \in \Z$, cada uno de
        \omterm{def:b3:residues:singularities}{residuo} $1$ \emph{(compute
        $\lim_{z\to n}(z-n)\pi\cot\pi z$)}.
  \item Calcule el comienzo de la expansión Laurent en $0$:
        \[
        \pi\cot(\pi z) = \frac1z - \frac{\pi^2}{3}\,z -
        \frac{\pi^4}{45}\,z^3 + O(z^5) ,
        \]
        dividiendo la serie de potencias de $\cos$ por la de
        $\sin$ (justificar la división: $\frac{\sin\pi z}{\pi z}$
        es \omterm{def:b3:holomorphic:holo}{holomorfo} y distinto de cero cerca de $0$, por lo que es
        recíproco es \omterm{def:b3:holomorphic:holo}{holomorfo}; identificar coeficientes
        mediante pedido $3$).
  \item Sea $C_N$ el límite del cuadrado con vértices
        $(\pm1\pm\iu)(N + \frac12)$. mostrar eso
        $\abs{\cot(\pi z)} \leq 2$ en $C_N$ por cada $N \geq
        1$. \emph{(On vertical sides,$\cot(\pi(\pm(N +
        \frac12) + \iu y)) = \mp\tan(\iu\pi y)$, of modulus
        $\abs{\tanh(\pi y)} \leq 1$; on horizontal sides
        $\abs y = N + \frac12$, bound
        $\abs{\cot(\pi(x\pm\iu y))} \leq \coth(\pi y) \leq
        \coth(\pi/2) < 1.1$.)}
\end{enumerate}

\medskip
\noindent\textbf{Parte II --- The summation theorem.}
\begin{enumerate}[resume]
  \item Sea $f$ racional, \omterm{def:b3:holomorphic:holo}{holomorfo} en los números enteros,
        con $\deg(\text{denom}) \geq \deg(\text{num}) + 2$.
        Usando el teorema \omterm{def:b3:residues:singularities}{residuo} en $C_N$ y el límite de
        Pregunta 3, prueba:
        \[
        \lim_{N\to\infty}\ \sum_{n = -N}^{N} f(n)
        = -\sum_{p\ \text{pole of}\ f}
        \operatorname{Res}\bigl(\pi\cot(\pi z)f(z),\,p\bigr).
        \]
  \item ¿Dónde necesita el argumento la condición de grado?
        Mostrar con ejemplo (tome $f(z) = 1/(z + \frac12)$)
        que para una desintegración más lenta el límite simétrico
        todavía puede existir mientras la serie bilateral diverge
        --- y que la fórmula luego calcula el
        \emph{valor principal}.
\end{enumerate}

\medskip
\noindent\textbf{Parte III --- The values.}
\begin{enumerate}[resume]
  \item Aplicar el método a $f(z) = 1/z^2$: aquí $f$ tiene su
        polo \emph{at} un número entero, así que ejecute el argumento
        directamente --- integrar $g(z) = \frac{\pi\cot(\pi
        z)}{z^2}$sobre$C_N$, mostrar la integral$\to 0$ y
        calcule $\operatorname{Res}(g, 0)$ a partir de la pregunta 2.
        Concluir:
        \[
        2\sum_{n\geq1}\frac1{n^2} = \frac{\pi^2}{3},
        \qquad \zeta(2) = \frac{\pi^2}6 .
        \]
  \item Lo mismo con $g(z) = \frac{\pi\cot(\pi z)}{z^4}$:
        calcular $\operatorname{Res}(g, 0)$ y deducir
        $\zeta(4) = \frac{\pi^4}{90}$.
  \item Explique el patrón general: por cada $k \geq 1$,
        $\zeta(2k)$ es $-\frac12$ veces el coeficiente de
        $z^{2k-1}$ en la expansión de Laurent de $\pi\cot(\pi
        z)$en$0$--- un múltiplo racional de$\pi^{2k}$.
        Calcule $\zeta(6)$ empujando la división de la pregunta 2
        un paso más. ¿Qué dice el método sobre
        $\zeta(3)$ --- ¿y por qué no dice nada?
\end{enumerate}

\medskip
\noindent\textbf{Parte IV --- The partial fraction expansion
of the cotangent.}
\begin{enumerate}[resume]
  \item Repare $w \in \C\setminus\Z$ y aplique el método de
        Parte II de $f(z) = \dfrac{1}{(z - w)(z + w)}$ ---
        teniendo en cuenta que $\pi\cot(\pi z)f(z)$ ahora tiene adicionales
        polos simples en $\pm w$, cuyo \omterm{def:b3:residues:singularities}{residuos} debe unirse al
        contar. Deducir la expansión de fracción parcial
        \[
        \pi\cot(\pi w) = \frac1w +
        \sum_{n\geq1}\frac{2w}{w^2 - n^2},
        \]
        la serie converge normalmente en \omterm{def:b3:topology:compact}{compacto} subconjuntos de
        $\C\setminus\Z$.
  \item Recuperarse de esta expansión, ampliando cada término
        en potencias de $w$ (justificar el intercambio), el
        \emph{mismo} Coeficientes de Laurent como en la pregunta 2 ---
        el círculo se cierra: la fórmula de Euler
        $\sum\frac1{n^2} = \frac{\pi^2}6$ es el coeficiente
        de $w$ en las dos caras de la cotangente. Comparar con el
        Prueba de series de Fourier (año 2) y prueba de trazas
        (\cref{pb:b3:spectral:1}): tres teorías, una
        número.
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Euler's product for the sine.}
El desarrollo de la pregunta 9 es la derivada logarítmica de
un producto infinito; ahora demostramos la factorización de Euler de 1734
honestamente.
\begin{enumerate}[resume]
  \item Para $N \geq 1$ establezca $P_N(z) =
        z\prod_{n=1}^{N}\bigl(1 - \frac{z^2}{n^2}\bigr)$.
        Demuestre que $P_N$ converge uniformemente en cada disco.
        $\bar D(0, R)$, a una función completa $P$ cuyos ceros
        son exactamente los números enteros, todos simples. \emph{(For $n
        \geq 2R$write the factor as$\exp\log(1 -
        z^2/n^2)$ with the principal logarithm of
        \cref{exo:b3:holomorphic:3}, bound $\abs{\log(1+u)}
        \leq 2\abs u$for$\abs u \leq \frac12$ via the
        series, and exponentiate the normally convergent sum
        of logarithms; the finitely many remaining factors
        are a polynomial. Conclude with
        \cref{thm:b3:holomorphic:weierstrassconv}.)}
  \item Muéstralo en $\C\setminus\Z$,
        \[
        \frac{P'(z)}{P(z)} = \frac1z +
        \sum_{n\geq1}\frac{2z}{z^2 - n^2} = \pi\cot(\pi z)
        \]
        \emph{(differentiate the finite products, pass to
        the limit using
        \cref{thm:b3:holomorphic:weierstrassconv} and the
        zero-freeness of $P$ off $\Z$, and quote question
        9)}.
  \item Demuestre que $Q = \sin(\pi z)/P(z)$ se extiende a un
        función completa libre de cero con $Q' = 0$, y
        concluir \emph{producto de euler}:
        \[
        \sin(\pi z) = \pi z\prod_{n\geq1}
        \Bigl(1 - \frac{z^2}{n^2}\Bigr)
        \qquad (z \in \C) .
        \]
  \item (Wallis, 1655) Evaluar en $z = \frac12$:
        \[
        \frac\pi2 = \prod_{n\geq1}\frac{4n^2}{4n^2 - 1}
        = \lim_{N\to\infty}
        \frac{2\cdot2\cdot4\cdot4\cdots(2N)(2N)}
        {1\cdot3\cdot3\cdot5\cdots(2N-1)(2N+1)} .
        \]
  \item Para $\abs z < 1$, expanda el logaritmo del
        producto como una serie doble (justificar el
        reordenamiento) y recuperar $\zeta(2) =
        \frac{\pi^2}6$haciendo coincidir el coeficiente de$z^3$
        en $\sin(\pi z) = \pi z - \frac{\pi^3}6z^3 + \cdots$
        --- la cara del producto del número de Euler.
\end{enumerate}

\medskip
\noindent\textbf{Part VI --- Sister kernels.} La cotangente
tiene hermanos; cada uno valora su propia familia de series.
\begin{enumerate}[resume]
  \item Diferenciar término por término la expansión de la pregunta 9
        (justificado por
        \cref{thm:b3:holomorphic:weierstrassconv}) a
        obtener, normalmente en \omterm{def:b3:topology:compact}{compactos} de $\C\setminus\Z$,
        \[
        \frac{\pi^2}{\sin^2(\pi z)} =
        \sum_{n\in\Z}\frac1{(z - n)^2} .
        \]
  \item Evaluar en $z = \frac12$:
        $\sum_{m\geq0}\frac1{(2m+1)^2} = \frac{\pi^2}8$;
        recupere $\zeta(2)$ una vez más dividiendo el
        números enteros por paridad.
  \item Verifique la identidad de duplicación $\tan\theta =
        \cot\theta - 2\cot(2\theta)$ y deduzca
        \[
        \pi\tan(\pi z) =
        \sum_{m\geq0}\frac{8z}{(2m+1)^2 - 4z^2} ,
        \]
        normalmente en \omterm{def:b3:topology:compact}{compactos} evitando $\frac12 + \Z$.
  \item Expandir alrededor de $0$ ($\abs z < \frac12$; Fubini
        nuevamente): con $\lambda(s) =
        \sum_{m\geq0}(2m+1)^{-s}$,
        \[
        \pi\tan(\pi z) =
        \sum_{k\geq0}8\cdot4^k\,\lambda(2k+2)\,z^{2k+1} ;
        \]
        comparar con $\tan u = u + \frac{u^3}3 + O(u^5)$ para
        recuperar $\lambda(2) = \frac{\pi^2}8$ y obtener
        $\lambda(4) = \frac{\pi^4}{96}$, luego verifique
        $\zeta(4) = \frac{\pi^4}{90}$ vía $\lambda(4) = (1 -
        2^{-4})\,\zeta(4)$.
  \item Verificar $\frac1{\sin\theta} = \cot\frac\theta2 -
        \cot\theta$ y deducir
        \[
        \frac{\pi}{\sin(\pi z)} = \frac1z +
        \sum_{n\geq1}(-1)^n\,\frac{2z}{z^2 - n^2} .
        \]
        Verifique las señales con el \omterm{def:b3:residues:singularities}{residuos} de
        $\pi/\sin(\pi z)$ en los números enteros.
  \item Lea el coeficiente de $z$: $\eta(2) =
        \sum_{n\geq1}\frac{(-1)^{n-1}}{n^2} =
        \frac{\pi^2}{12}$ y confirme la coherencia.
        $\eta(2) = (1 - 2^{1-2})\,\zeta(2)$.
  \item (Final) Evalúe la expansión de la pregunta 20 en $z =
        \frac12$ y deduzca \emph{La fórmula de Leibniz}
        \[
        \frac\pi4 = 1 - \frac13 + \frac15 - \frac17 +
        \cdots
        \]
        Cierra con un breve párrafo: un grano por
        aritmética --- qué núcleo fija el precio de qué familia de
        series, y por qué todas ellas son estructuralmente ciegas
        a $\zeta(3)$.

\end{enumerate}
\medskip
\noindent\textbf{Part VII --- The full price list: Bernoulli
numbers.}
\begin{enumerate}[resume]
  \item Combine la expansión de fracción parcial de $\pi
        z\cot(\pi z)$ con la función generadora del
        Números de Bernoulli ($\frac{w}{\eu^w - 1} =
        \sum_n\frac{B_n}{n!}w^n$,
        \cref{pb:b3:holomorphic:1}, Parte VI): de
        \[
        \pi z\cot(\pi z) = \iu\pi z +
        \frac{2\iu\pi z}{\eu^{2\iu\pi z} - 1}
        \]
        (pruebe esta identidad primero), deduzca la forma cerrada
        \[
        \zeta(2k) = (-1)^{k+1}\,
        \frac{(2\pi)^{2k}\,B_{2k}}{2\,(2k)!}
        \qquad (k \geq 1).
        \]
  \item Verifique la fórmula con $B_2 = \frac16$, $B_4 =
        -\frac1{30}$,$B_6 = \frac1{42}$: recupere$\zeta(2) =
        \frac{\pi^2}6$,$\zeta(4) = \frac{\pi^4}{90}$ y
        calcular $\zeta(6) = \frac{\pi^6}{945}$.
  \item (recursión de Euler) Expande ambos lados de
        $\bigl(z\cot z\bigr)' = \cot z - z(1 + \cot^2z)$ ---
        o elevar al cuadrado la serie cotangente directamente --- para demostrar
        \[
        \Bigl(k + \frac12\Bigr)\zeta(2k) =
        \sum_{j=1}^{k-1}\zeta(2j)\,\zeta(2k - 2j)
        \qquad (k \geq 2),
        \]
        y verifique que calcule $\zeta(4)$ desde $\zeta(2)$ y
        $\zeta(6)$ de $\zeta(2), \zeta(4)$ --- todos pares
        valores zeta de la semilla única $\frac{\pi^2}6$,
        sin nueva integración.
\end{enumerate}
\end{problem}
