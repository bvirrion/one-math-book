\chapter{Espacios de Banach y los teoremas fundamentales}
\label{ch:b3:banach}

El análisis funcional estudia espacios normados de dimensión infinita.
a través de los operadores y funcionales que viven en ellos. su
descubrimiento fundamental es que \emph{\omterm{def:b3:complete:complete}{lo completo}}, a través de Baire
teorema, fuerza una fuerte uniformidad: familias puntualmente acotadas de
los operadores están limitados por normas (Banach--Steinhaus), \omterm{def:b3:topology:continuity}{continuo}
las biyecciones tienen inversas \omterm{def:b3:topology:continuity}{continuo} (mapeo abierto) y gráficas
detectar \omterm{def:b3:topology:continuity}{continuidad} (gráfico cerrado). El otro pilar,
\emph{Hahn--Banach}, no necesita \omterm{def:b3:complete:complete}{lo completo} en absoluto --- solo
El lema de Zorn --- y garantiza que \omterm{def:b3:banach:operator}{espacios duales} son lo suficientemente ricos
para ver cada vector. Este capítulo prueba los cuatro teoremas y
los prueba en los espacios de secuencia clásicos $\ell^p$, en
cálculos concretos \omterm{def:b3:banach:operator}{dual}, y de una manera realmente sorprendente
aplicación: existen funciones periódicas \omterm{def:b3:topology:continuity}{continuo} $2\pi$
cuya serie de Fourier \emph{diverge} en un punto --- resolviendo en
lo negativo, una pregunta que el año 2 dejó abierta.

En todo momento, $E, F$ son espacios normados sobre $K = \R$ o $\C$;
\emph{espacio banach}\index{espacio banach} significa \omterm{def:b3:complete:complete}{completo} normalizado
espacio.

\section{Operadores acotados; espacios de secuencia}

\begin{definition}\label{def:b3:banach:operator}
$\mathcal L(E, F)$ denota el espacio de \emph{encerrado}
(= \omterm{def:b3:topology:continuity}{continuo}, Año 2) mapas lineales con el \emph{operador
norma}\index{norma del operador} $\vertiii T = \sup_{\norm x \leq
1}\norm{Tx}$; es submultiplicativo:$\vertiii{ST} \leq
\vertiii S\,\vertiii T$. El \emph{dual}\index{espacio dual} es
$E' = \mathcal L(E, K)$.
\end{definition}

\begin{proposition}\label{prop:b3:banach:llcomplete}
Si $F$ es un espacio de Banach, también lo es $\mathcal L(E, F)$; en
En particular, $E'$ es siempre un espacio de Banach.
\end{proposition}

\begin{proof}
Sea $(T_n)$ Cauchy para $\vertiii\cdot$. Para cada $x$,
$\norm{T_nx - T_mx} \leq \vertiii{T_n - T_m}\norm x$: $(T_nx)$
es Cauchy en $F$, convergente; Llame al límite $Tx$. $T$ es
lineal (límites de identidades lineales); pasando al limite en
$\norm{T_nx - T_mx} \leq \varepsilon\norm x$ ($n, m \geq N$)
da $\norm{T_nx - Tx} \leq \varepsilon\norm x$: $T_n \to T$
en \omterm{def:b3:banach:operator}{norma del operador} y $\vertiii T \leq \vertiii{T_N} +
\varepsilon < \infty$.
\end{proof}

\begin{definition}\label{def:b3:banach:lp}
Los espacios de secuencia clásica\index{sequence spaces $\ell^p$}
(sobre $K$, indexado por $\N$):
\[
\ell^p = \Bigl\{x = (x_n) : \norm x_p = \Bigl(\sum_n
\abs{x_n}^p\Bigr)^{1/p} < \infty\Bigr\}\quad (1 \leq p <
\infty),
\]
\[
\ell^\infty = \{x : \norm x_\infty = \sup\abs{x_n} < \infty\},
\]
y $c_0 = \{x : x_n \to 0\}$ con $\norm\cdot_\infty$. eso
$\norm\cdot_p$ es una norma que se deriva de la desigualdad de Minkowski,
demostrado en el caso discreto exactamente como en \cref{ch:b3:lp} (o
sumando la desigualdad de dimensión finita del año 2). Todos
son espacios de Banach, y $c_0$ es un subespacio cerrado de
$\ell^\infty$ (\cref{exo:b3:banach:3}).
\end{definition}

\begin{proposition}[Serie Neumann]\label{prop:b3:banach:neumann}
Sea $E$ Banach y $T \in \mathcal L(E) = \mathcal L(E,E)$
con $\vertiii T < 1$. Entonces $I - T$ es invertible en $\mathcal
L(E)$, con$(I - T)^{-1} = \sum_{n\geq0}T^n$ (convergente en
\omterm{def:b3:banach:operator}{norma del operador}). En consecuencia, el conjunto de operadores reversibles es
abierto y la inversión es \omterm{def:b3:topology:continuity}{continuo} en él.\index{serie neumann}
\end{proposition}

\begin{proof}
La serie converge absolutamente ($\vertiii{T^n} \leq \vertiii
T^n$, geométrica) en el Banach$\mathcal L(E)$
(\cref{prop:b3:banach:llcomplete};
\cref{exo:b3:complete:1}(b)). Telescópico, $(I - T)\sum_{n \leq
N}T^n = I - T^{N+1} \to I$, y similar en el otro lado.
Para apertura: si $S$ es invertible y $\vertiii{H} <
1/\vertiii{S^{-1}}$, entonces$S + H = S(I + S^{-1}H)$ con
$\vertiii{S^{-1}H} < 1$: reversible. \omterm{def:b3:topology:continuity}{Continuidad} de inversión:
la expresión de la serie da $\vertiii{(S+H)^{-1} - S^{-1}} =
O(\vertiii H)$ localmente.
\end{proof}

\begin{example}[Una ecuación de Volterra, resultado de Neumann.]
\label{ex:b3:banach:volterraexample}
En $E = \mathcal C(\intcc01)$, considere la ecuación integral
\[
u(x) = 1 + \lambda\int_0^xu(t)\,\dd t,
\qquad\text{i.e.}\qquad u = \mathbf 1 + \lambda Tu,
\quad (Tu)(x) = \int_0^xu .
\]
Aquí $\vertiii T \leq 1$, por lo que para $\abs\lambda < 1$ se aplica directamente el \omterm{prop:b3:banach:neumann}{neumann
serie}: $u = (I - \lambda T)^{-1}\mathbf 1 =
\sum_n\lambda^nT^n\mathbf 1$. Computación,$T^n\mathbf 1 =
\frac{x^n}{n!}$, entonces
\[
u(x) = \sum_{n\geq0}\frac{(\lambda x)^n}{n!} =
\eu^{\lambda x} ,
\]
como lo confirma la diferenciación. Mejor: $\vertiii{T^n} \leq
\frac1{n!}$ (el núcleo iterado se reduce factorialmente), por lo que
$\sum\lambda^nT^n$ converge para \emph{cada} $\lambda$ ---
el operador $I - \lambda T$ es invertible para todos los $\lambda
\in \C$, aunque$\vertiii{\lambda T} \geq 1$ eventualmente:
lo que importa es la decadencia espectral de los poderes, no la
primera norma. Los operadores de Volterra tienen este decaimiento factorial
incorporado (\cref{exo:b3:complete:4}(a) explotado exactamente
esto), razón por la cual los problemas de valor inicial nunca sufren la misma
fenómenos de resonancia de problemas de valores límite
(\cref{ch:b3:spectral}).
\end{example}

\section{Hahn--Banach}

\begin{theorem}[Hahn--Banach, forma analítica]
\label{thm:b3:banach:hahnbanach}
Sea $E$ un espacio vectorial \emph{real}, $p \colon E \to \R$
\emph{sublineal} ($p(x + y) \leq p(x) + p(y)$ y $p(tx) =
tp(x)$para$t \geq 0$),$F \subseteq E$un subespacio y$f
\colon F \to \R$lineal con$f \leq p$en$F$. Entonces$f$ se extiende
a un $\tilde f \colon E \to \R$ lineal con $\tilde f \leq p$
en $E$.\index{Teorema de Hahn-Banach}
\end{theorem}

\begin{proof}
\emph{Extensión en un solo paso.} Dejar $x_0 \notin F$; ampliamos $f$ a
$F \oplus \R x_0$ eligiendo $\alpha = \tilde f(x_0)$
correctamente: necesitamos, para todos $y \in F$, $t > 0$,
\[
f(y) + t\alpha \leq p(y + tx_0)
\quad\text{and}\quad
f(y) - t\alpha \leq p(y - tx_0),
\]
que después de dividir por $t$ (sublinealidad) se reduce a
\[
\sup_{v \in F}\ \bigl[f(v) - p(v - x_0)\bigr]
\;\leq\; \alpha \;\leq\;
\inf_{u \in F}\ \bigl[p(u + x_0) - f(u)\bigr].
\]
Tal $\alpha$ existe si cada miembro izquierdo es $\leq$ cada
miembro derecho: de hecho $f(v) + f(u) = f(u + v) \leq p(u + v) \leq
p(u + x_0) + p(v - x_0)$, es decir $f(v) - p(v - x_0) \leq p(u +
x_0) - f(u)$.

\emph{Zorn.} Ordene las extensiones de $f$ dominadas por $p$
(pares: subespacial, funcional) por extensión; una cadena tiene el
unión como límite superior; se debe definir un elemento máximo en todos
de $E$; de lo contrario, la extensión de un paso contradice la maximalidad.
\end{proof}

\begin{corollary}\label{cor:b3:banach:hbnormed}
Sea $E$ un espacio normado ($K = \R$ o $\C$).
\begin{enumerate}
  \item Cada $f \in F'$ ($F$ un subespacio) se extiende a $\tilde f
        \in E'$con$\norm{\tilde f}_{E'} = \norm f_{F'}$.
  \item Por cada $x \neq 0$ hay $f \in E'$ con $\norm f
        = 1$y$f(x) = \norm x$. En particular$E'$ separa
        los puntos de $E$ y $\norm x = \sup_{\norm f \leq
        1}\abs{f(x)}$.
  \item Para un subespacio cerrado $F$ y $x \notin F$, hay $f
        \in E'$que desaparece en$F$con$f(x) = d(x, F)$ y
        $\norm f \leq 1$.
\end{enumerate}
\end{corollary}

\begin{proof}
(1) Caso real: aplicar \cref{thm:b3:banach:hahnbanach} con $p(x)
= \norm f_{F'}\,\norm x$ (sublineal); la extensión satisface
$\pm\tilde f(x) = \tilde f(\pm x) \leq p(x)$, entonces
$\norm{\tilde f} \leq \norm f$ y $\geq$ son restricciones.
Caso complejo: let $u = \operatorname{Re}f$, un verdadero funcional
con $\abs u \leq \norm f\norm\cdot$; nota $f(x) = u(x) -
\iu\,u(\iu x)$ (verifique las partes reales e imaginarias:
$\operatorname{Im}f(x) = -\operatorname{Re}f(\iu x)$). Extender
$u$ linealmente real con el mismo límite y establezca $\tilde f(x) =
\tilde u(x) - \iu\tilde u(\iu x)$:$\C$-lineal (verificación directa en
multiplicación por $\iu$), extiende $f$; norma: para dado $x$
escriba $\tilde f(x) = r\eu^{\iu\theta}$, luego $\abs{\tilde f(x)}
= \tilde f(\eu^{-\iu\theta}x) = \tilde u(\eu^{-\iu\theta}x)
\leq \norm f\,\norm x$.

(2) En $F = Kx$ defina $f(tx) = t\norm x$: norma $1$ en $F$;
extender en (1). La fórmula de la dualidad: $\leq$ es clara, $\geq$ por
este $f$.

(3) En $F \oplus Kx$ defina $f(y + tx) = t\,d(x, F)$; entonces para
$t \neq 0$, $\norm{y + tx} = \abs t\,\norm{x + y/t} \geq \abs
t\,d(x, F) = \abs{f(y + tx)}$:$\norm f \leq 1$ en el
subespacio; extender en (1).
\end{proof}

\begin{remark}\label{rem:b3:banach:bidual}
Por (2), la aplicación canónico $J \colon E \to E''$, $J(x)(f) =
f(x)$, es una isometría (\cref{exo:b3:banach:10}): todo normado
El espacio se asienta dentro de su bidual. Los espacios con sobreyectiva $J$ son
llamado \emph{reflexivo}\index{espacio reflexivo}; el fin de semana
El problema muestra que $\ell^p$ ($1 < p < \infty$) es \omterm{rem:b3:banach:bidual}{reflexivo} mientras
$\ell^1$ no lo es.
\end{remark}

\section{La trilogía de Baire}

\begin{theorem}[Banach--Steinhaus, delimitación uniforme]
\label{thm:b3:banach:banachsteinhaus}
Sea $E$ un espacio \emph{Banach}, $F$ normado y $(T_i)_{i\in
I} \subseteq \mathcal L(E, F)$una familia con$\sup_i
\norm{T_ix} < \infty$por cada$x \in E$. Entonces$\sup_i
\vertiii{T_i} < \infty$.\index{Teorema de Banach-Steinhaus}
\end{theorem}

\begin{proof}
Los conjuntos $F_n = \{x : \sup_i\norm{T_ix} \leq n\}$ están cerrados.
(intersecciones de preimágenes de bolas cerradas) y portada $E$.
Baire (\cref{thm:b3:complete:baire}) le da a $n_0$ y una pelota
$B(x_0, r) \subseteq F_{n_0}$. Para $\norm z < r$: $\norm{T_iz}
\leq \norm{T_i(x_0 + z)} + \norm{T_ix_0} \leq 2n_0$, entonces
$\vertiii{T_i} \leq 2n_0/r$ por cada $i$.
\end{proof}

\begin{corollary}\label{cor:b3:banach:bslimits}
Si $E$ es Banach y $T_n \in \mathcal L(E,F)$ convergen
\emph{puntualmente} ($T_nx \to Tx$ para cada $x$), luego $\sup_n
\vertiii{T_n} < \infty$,$T \in \mathcal L(E, F)$ y
$\vertiii T \leq \liminf \vertiii{T_n}$.
\end{corollary}

\begin{proof}
Las secuencias convergentes están acotadas: acotación puntual;
Banach--Steinhaus limita las normas a algunos $M$; entonces $\norm{Tx}
= \lim\norm{T_nx} \leq M\norm x$($T$ es lineal como un punto
límite), y el límite más nítido pasando a $\liminf$ en
$\norm{T_nx} \leq \vertiii{T_n}\norm x$.
\end{proof}

\begin{theorem}[Serie de Fourier divergente]
\label{thm:b3:banach:fourierdiverge}
Existen \omterm{def:b3:topology:continuity}{continuo} $2\pi$ funciones periódicas $f$ cuyas
La serie de Fourier diverge en $0$: $\sup_N\abs{S_N(f)(0)} =
\infty$. De hecho, tales$f$forman un subconjunto denso de$\mathcal
C(S^1)$.\index{Serie de Fourier, divergencia.}.
\end{theorem}

\begin{proof}
Trabaja en $E = (\mathcal C(S^1), \norm\cdot_\infty)$, un Banach
espacio, con los funcionales $\Lambda_N(f) = S_N(f)(0) =
\frac1{2\pi}\int_{-\pi}^{\pi}f(t)\,D_N(t)\,\dd t$ (Dirichlet
núcleo, Año 2). Cada $\Lambda_N$ es \omterm{def:b3:topology:continuity}{continuo} con
\[
\norm{\Lambda_N} = \frac1{2\pi}\int_{-\pi}^\pi\abs{D_N(t)}\,\dd
t \;=\; L_N .
\]
($\leq$ está claro; $\geq$: tome $f$ \omterm{def:b3:topology:continuity}{continuo}, $\norm f_\infty
\leq 1$, aproximandose a$\operatorname{sign}D_N$ --- el signo
tiene un número finito de saltos; suavizando cada salto en un intervalo de
longitud $\varepsilon$ cambia la integral por $O(N\varepsilon)$.)
Los \emph{Constantes de Lebesgue} $L_N$ tienden al infinito:
\[
L_N = \frac1{2\pi}\int_{-\pi}^{\pi}
\frac{\abs{\sin\bigl((N{+}\tfrac12)t\bigr)}}{\abs{\sin(t/2)}}
\,\dd t
\geq \frac{2}{\pi}\int_0^{\pi}
\frac{\abs{\sin\bigl((N{+}\tfrac12)t\bigr)}}{t}\,\dd t
= \frac{2}{\pi}\int_0^{(N+\frac12)\pi}\frac{\abs{\sin
u}}{u}\,\dd u
\]
(usando $\abs{\sin(t/2)} \leq t/2$ en $[0, \pi]$, luego
sustituyendo $u = (N + \tfrac12)t$). Cortar en arcos:
\[
\int_{(k-1)\pi}^{k\pi}\frac{\abs{\sin u}}u\,\dd u
\geq \frac1{k\pi}\int_{(k-1)\pi}^{k\pi}\abs{\sin u}\,\dd u =
\frac{2}{k\pi},
\qquad\text{so}\qquad
L_N \geq \frac{4}{\pi^2}\sum_{k=1}^N\frac1k
\xrightarrow[N\to\infty]{} \infty .
\]
Si cada \omterm{def:b3:topology:continuity}{continuo} $f$ tuviera $\sup_N
\abs{\Lambda_N(f)} < \infty$, Banach--Steinhaus obligaría
$\sup_N\norm{\Lambda_N} < \infty$: contradicción. Entonces algunos $f$
--- de hecho un no escaso, \omterm{def:b3:topology:interior}{conjunto denso} de $f$ (el complemento de
$\bigcup_M\{f: \sup_N\abs{\Lambda_Nf}\leq M\}$, un contable
unión de conjuntos cerrados que, al no tener \omterm{def:b3:topology:interior}{interior} por lo anterior
aplicado en cualquier bola, es \omterm{rem:b3:complete:meagre}{pobre}) --- tiene $\sup_N\abs{S_Nf(0)} =
\infty$.
\end{proof}

\begin{theorem}[Abrir mapa]\label{thm:b3:banach:openmapping}
Sean $E, F$ espacios de Banach y $T \in \mathcal L(E, F)$
\emph{sobreyectivo}. Entonces $T$ está abierto: $T(B_E(0,1)) \supseteq
B_F(0, c)$para algunos$c > 0$. En consecuencia, un biyectivo acotado
operador entre espacios de Banach tiene un límite
inverso.\index{teorema de mapeo abierto}
\end{theorem}

\begin{proof}
Escribe $B = B_E(0,1)$. La sobreyectividad da $F = \bigcup_n
\overline{T(nB)} = \bigcup_n n\,\overline{T(B)}$; Baire
(\cref{thm:b3:complete:baire}) le da \omterm{def:b3:topology:interior}{interior} a
$\overline{T(B)}$: algunos $B_F(y_0, 4c) \subseteq
\overline{T(B)}$. \omterm{ex:b3:groups:actions}{Volver a centrar} en$0$: para$\norm y < 4c$, ambos
$y_0 + y$ y $y_0$ son límites de imágenes $Tu_k$, $Tv_k$ con
$u_k, v_k \in B$, entonces $y = \lim T(u_k - v_k)$ con $u_k - v_k
\in 2B$:$B_F(0, 4c) \subseteq \overline{T(2B)}$, es decir\
$B_F(0, 2c) \subseteq \overline{T(B)}$.

\emph{Removing the \omterm{def:b3:topology:interior}{cierre}} (aquí entra \omterm{def:b3:complete:complete}{lo completo} de $E$):
deja $\norm y < c$. Elija $x_1 \in \frac12 B$ con $\norm{y -
Tx_1} < c/2$($B_F(0,2c) \subseteq \overline{T(B)}$ escalado por
$\frac12$); inductivamente $x_k \in 2^{-k}B$ con $\norm{y -
T(x_1 + \dots + x_k)} < c\,2^{-k}$. La serie$\sum x_k$
converge absolutamente en el Banach $E$, a $x \in B$ (norma $<
\sum 2^{-k} = 1$), y$Tx = y$por \omterm{def:b3:topology:continuity}{continuidad}:$B_F(0, c)
\subseteq T(B)$. La apertura de$T$ en aperturas arbitrarias sigue por
traducción y escalado; para el corolario, apertura de $T$
significa que $T^{-1}$ es \omterm{def:b3:topology:continuity}{continuo}.
\end{proof}

\begin{corollary}[Normas equivalentes]\label{cor:b3:banach:equivnorms}
Si un espacio vectorial es \omterm{def:b3:complete:complete}{completo} para dos normas comparables
($\norm\cdot_a \leq C\norm\cdot_b$), las normas son equivalentes.
\end{corollary}

\begin{proof}
La identidad $(E, \norm\cdot_b) \to (E, \norm\cdot_a)$ es
acotado y biyectivo entre espacios de Banach: su inverso es
acotado.
\end{proof}

\begin{theorem}[gráfico cerrado]\label{thm:b3:banach:closedgraph}
Sea $E, F$ Banach y $T \colon E \to F$ lineal. si el
El gráfico $\Gamma = \{(x, Tx)\}$ está cerrado en $E \times F$ (es decir,\
$x_n \to x$ y $Tx_n \to y$ implican $y = Tx$), entonces $T$ es
acotado.\index{teorema del gráfico cerrado}
\end{theorem}

\begin{proof}
$E \times F$ con $\norm{(x,y)} = \norm x + \norm y$ es Banach;
$\Gamma$, un subespacio cerrado, es Banach. la proyección
$\pi_E\colon \Gamma \to E$ es acotado y biyectivo, por lo que es
$x \mapsto (x, Tx)$ inverso está acotado
(\cref{thm:b3:banach:openmapping}): $\norm{Tx} \leq \norm{(x,
Tx)} \leq C\norm x$.
\end{proof}

\begin{method}\label{met:b3:banach:trilogy}
Cuándo alcanzar qué teorema. \emph{Hahn--Banach}: a
producir una conducta funcional con prescrita (normalizar una
vector, que desaparece en un subespacio, se extiende desde un subespacio) ---
no se necesita \omterm{def:b3:complete:complete}{lo completo}. \emph{Banach--Steinhaus}: para convertir
información puntual en límites uniformes --- típicamente para mostrar
una operación límite es \omterm{def:b3:topology:continuity}{continuo}, o (contrapositivo) para demostrar
\emph{divergencia} para algún elemento, como para series de Fourier.
\emph{Mapeo abierto/gráfico cerrado}: para obtener \omterm{def:b3:topology:continuity}{continuidad} gratis
de la biyectividad \omterm{def:b3:galois:algebraic}{algebraico} o de una propiedad \omterm{def:b3:topology:interior}{cierre} de la
gráfico --- uso típico: comparar dos normas \omterm{def:b3:complete:complete}{completo} o demostrar
automático \omterm{def:b3:topology:continuity}{continuidad}. Los tres teoremas de Baire requieren
\omterm{def:b3:complete:complete}{lo completo}\emph{de la fuente}; contraejemplos de lo contrario
(\cref{exo:b3:banach:7}).
\end{method}

\section{Espacios duales, concretamente}

\begin{theorem}\label{thm:b3:banach:duals}
Isométricamente: $(c_0)' \cong \ell^1$ y $(\ell^1)' \cong
\ell^\infty$, mediante el emparejamiento$\langle x, y\rangle = \sum_n
x_ny_n$.\index{duales de espacios de secuencia}
\end{theorem}

\begin{proof}
Probamos $(c_0)' \cong \ell^1$; la segunda identificación es
\cref{exo:b3:banach:5}. A $y \in \ell^1$ asociar $\Lambda_y(x)
= \sum x_ny_n$($x \in c_0$): absolutamente convergente, con
$\abs{\Lambda_y(x)} \leq \norm x_\infty\norm y_1$, entonces
$\norm{\Lambda_y} \leq \norm y_1$. Por el contrario, dejemos que $\Lambda \in
(c_0)'$; configure$y_n = \Lambda(e_n)$($e_n$ las secuencias de la unidad).
Para cualquier $N$, pruebe $x^{(N)} = \sum_{n \leq N}
\operatorname{sign}(\overline{y_n})\,e_n \in c_0$(norma$\leq
1$; en el caso complejo utilice factores unimodulares$\bar y_n/\abs
{y_n}$):$\Lambda(x^{(N)}) = \sum_{n\leq N}\abs{y_n} \leq
\norm\Lambda$. Entonces$y \in \ell^1$con$\norm y_1 \leq
\norm\Lambda$. Finalmente$\Lambda = \Lambda_y$: ambos coinciden en el
$e_n$, por lo tanto en secuencias finitas, densas en $c_0$ (truncamiento:
$\norm{x - \sum_{n \leq N}x_ne_n}_\infty = \sup_{n > N}\abs{x_n}
\to 0$precisamente porque$x_n \to 0$); \omterm{def:b3:topology:continuity}{continuo} funcionales
acordar un \omterm{def:b3:topology:interior}{conjunto denso} son iguales. La correspondencia es lineal,
biyectivo e isométrico ($\norm{\Lambda_y} = \norm y_1$ de
las dos desigualdades).
\end{proof}

\section{Problema: la dualidad del \texorpdfstring{$\ell^p$}%
{lp} espacios}

\begin{problem}[{Problema de fin de semana --- $(\ell^p)' = \ell^q$,
reflexividad y la extrañeza de $\ell^\infty$}]
\label{pb:b3:banach:1}
Arregla $1 < p < \infty$ y deja que $q$ sea el exponente conjugado,
$\frac1p + \frac1q = 1$. El emparejamiento es $\langle x,
y\rangle = \sum_n x_ny_n$.

\medskip
\noindent\textbf{Parte I --- Hölder and Minkowski for
sequences.}
\begin{enumerate}
  \item (Desigualdad de Young) Para $a, b \geq 0$ muestre $ab \leq
        \frac{a^p}p + \frac{b^q}q$, usando la concavidad de$\log$ o
        estudiando $t \mapsto \frac{t^p}p + \frac1q - t$.
  \item (Hölder) Deducir: $\abs{\langle x, y\rangle} \leq
        \norm x_p\norm y_q$para$x \in \ell^p$,$y \in \ell^q$;
        Identificar el caso de igualdad.
  \item (Minkowski) Deduce la desigualdad del triángulo para
        $\norm\cdot_p$. \emph{(Write $\abs{x_n + y_n}^p \leq
        \abs{x_n}\,\abs{x_n{+}y_n}^{p-1} +
        \abs{y_n}\,\abs{x_n{+}y_n}^{p-1}$ and apply Hölder to
        each term.)}
  \item Demuestre que $\ell^p$ es \omterm{def:b3:complete:complete}{completo} y que el finito
        Las secuencias son densas en él.
\end{enumerate}

\medskip
\noindent\textbf{Parte II --- The duality $(\ell^p)' =
\ell^q$.}
\begin{enumerate}[resume]
  \item Para $y \in \ell^q$, muestre que $\Lambda_y(x) = \langle
        x, y\rangle$define$\Lambda_y \in (\ell^p)'$ con
        $\norm{\Lambda_y} \leq \norm y_q$, y probando en
        $x_n = \abs{y_n}^{q-1}\operatorname{sign}(y_n)$
        (adecuadamente truncado y normalizado), que
        $\norm{\Lambda_y} = \norm y_q$.
  \item Por el contrario, dado $\Lambda \in (\ell^p)'$, establezca $y_n =
        \Lambda(e_n)$; muestre$y \in \ell^q$con$\norm y_q \leq
        \norm\Lambda$ \emph{(prueba sobre truncamientos como en cuestión
        5 y dejar crecer la longitud del truncamiento)} y concluya
        $\Lambda = \Lambda_y$: la aplicación $y \mapsto \Lambda_y$ es
        un isomorfismo isométrico $\ell^q \to (\ell^p)'$.
  \item Deduzca que $\ell^p$ es \emph{\omterm{rem:b3:banach:bidual}{reflexivo}} para $1 < p <
        \infty$: componiendo las dos dualidades, cada elemento de
        $(\ell^p)''$ proviene de $\ell^p$; verificar cuidadosamente que
        el compuesto es el canónico $J$.
\end{enumerate}

\medskip
\noindent\textbf{Parte III --- $\ell^1$ and $\ell^\infty$ are
different animals.}
\begin{enumerate}[resume]
  \item Muestre que $\ell^p$ ($1 \leq p < \infty$) y $c_0$ son
        separable, pero $\ell^\infty$ no lo es. \emph{(The
        uncountably many indicator sequences of subsets of $\N$
        are pairwise at distance $1$.)}
  \item Deduce de \cref{exo:b3:banach:10} que $(\ell^1)'
        \cong \ell^\infty$pero$(\ell^\infty)' \not\cong
        \ell^1$:$\ell^1$ es \emph{no} \omterm{rem:b3:banach:bidual}{reflexivo}.
        \emph{(If $(\ell^\infty)'$ were $\ell^1$, it would be
        separable, forcing $\ell^\infty$ separable.)}
  \item (Un límite de Banach, explícitamente) En $\ell^\infty_\R$, deje
        $p(x) = \limsup_n \frac{x_1 + \dots + x_n}{n}$. Mostrar
        que $p$ es sublineal, y que en el subespacio $c$ de
        secuencias convergentes, $\mathrm{LIM}(x) = \lim x$
        satisface $\mathrm{LIM} \leq p$. Ampliar por Hahn--Banach
        a $\mathrm{LIM} \colon \ell^\infty_\R \to \R$ y mostrar:
        $\mathrm{LIM}$ es positivo ($x \geq 0 \Rightarrow
        \mathrm{LIM}(x) \geq 0$), invariante de cambio
        ($\mathrm{LIM}(x_2, x_3, \dots) = \mathrm{LIM}(x)$),
        extiende el límite y satisface $\liminf x \leq
        \mathrm{LIM}(x)\leq \limsup x$.
  \item Demuestre que tal $\mathrm{LIM}$, visto en
        $(\ell^\infty)'$, es \emph{no} de la forma $\Lambda_y$
        para cualquier $y \in \ell^1$; concluir nuevamente $(\ell^\infty)'
        \neq \ell^1$. \emph{(Evaluar en las secuencias unitarias
        $e_n$, luego en la secuencia constante $1$.)}
  \item Evalúe $\mathrm{LIM}$ en $(0,1,0,1,\dots)$ y muestre
        que no hay extensión \emph{multiplicativo} invariante por desplazamiento
        del límite puede existir \emph{(consider $x =
        (0,1,0,1,\dots)$and$x\cdot Sx$where$S$ is the
        shift)}.
\end{enumerate}

\medskip
\noindent\textbf{Parte IV --- Epilogue: why reflexivity
matters.}
\begin{enumerate}[resume]
  \item Usando \cref{cor:b3:banach:bslimits} y la pregunta 6,
        mostrar que cada secuencia acotada de $\ell^p$ ($1 < p <
        \infty$) tiene una subsecuencia$(x^{(k)})$ que converge
        \emph{enclenque}: $\Lambda(x^{(k)})$ converge para cada
        $\Lambda \in (\ell^p)'$. \emph{(Diagonal extraction on
        the countably many coordinates; identify the weak limit
        in $\ell^p$ using uniform boundedness of norms and
        Hölder.)} Mostrar por ejemplo ($e_n$ en $\ell^1$, contra
        elementos bien elegidos de $\ell^\infty$) que esto falla
        en $\ell^1$: la compacidad débil es un privilegio de
        \omterm{rem:b3:banach:bidual}{espacios reflexivos}.
\end{enumerate}

\medskip
\noindent\textbf{Part V --- The weak \omterm{def:b3:topology:topology}{topología} at work, and
Schur's surprise.} Escribe $x^{(k)} \rightharpoonup x$ en un
espacio normado $E$ (\emph{convergencia débil}) cuando
$\Lambda(x^{(k)}) \to \Lambda(x)$ por cada $\Lambda \in E'$.
\begin{enumerate}[resume]
  \item \omterm{def:b3:complete:complete}{Completo} el censo: mostrar $(c_0)' \cong \ell^1$
        isométricamente, según el esquema de las preguntas 5--6 (¿qué
        reemplaza las secuencias de prueba?). Montar la cadena
        $c_0 \to \ell^1 \to \ell^\infty \to \dots$ de

  \item Demuestre que toda secuencia débilmente convergente de un Banach
        el espacio está limitado: vea el $x^{(k)}$ a través del
        incrustación canónica $J$ como funcionales en $E'$ y

  \item Muestre que en $\ell^p$, $1 < p < \infty$: $x^{(k)}

  \item (Radón--Riesz en $\ell^2$) Muestre que en $\ell^2$, débil

  \item (Schur, paso 1) Deje entrar a $x^{(k)} \rightharpoonup 0$

  \item Deduzca que $(e_k)$ no tiene convergencia débil.

        hiperplano $\ker\Lambda$.

  \item (El valor de un límite de Banach) Sea $L$ cualquier Banach
