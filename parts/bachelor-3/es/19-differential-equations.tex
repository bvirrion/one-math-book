\section{Cauchy--Lipschitz}

\begin{definition}\label{def:b3:ode:lipschitz}
$f$ es \emph{localmente Lipschitz en $x$}\index{localmente Lipschitz}
si cada punto de $U$ tiene un \omterm{def:b3:topology:topology}{entorno} $V$ y una constante
$L$ con $\norm{f(t, x_1) - f(t, x_2)} \leq L\norm{x_1 -
x_2}$para$(t, x_1), (t, x_2) \in V$. Si$f$es$\mathcal
C^1$(o simplemente$\partial_xf$ existe y es \omterm{def:b3:topology:continuity}{continuo}),
es localmente Lipschitz en $x$: en un \omterm{def:b3:topology:compact}{compacto} convexo
\omterm{def:b3:topology:topology}{entorno}, la desigualdad del valor medio con $L =
\sup\vertiii{\partial_xf}$.
\end{definition}

\begin{theorem}[Cauchy--Lipschitz, local]
\label{thm:b3:ode:cauchylipschitz}
Sea $f$ \omterm{def:b3:topology:continuity}{continuo} y \omterm{def:b3:ode:lipschitz}{localmente Lipschitz} en $x$ y $(t_0,
x_0) \in U$. Hay$T > 0$ tal que el problema de Cauchy
\[
x' = f(t, x), \qquad x(t_0) = x_0
\]
tiene exactamente una soluciĂłn en $[t_0 - T, t_0 +
T]$.\index{Teorema de Cauchy-Lipschitz}
\end{theorem}

\begin{proof}
Elija $a, b > 0$ con $Q = [t_0 - a, t_0 + a]\times\bar
B(x_0, b) \subseteq U$, en el que estĂĄ$\norm f \leq M$y$f$.
$L$-Lipschitz en $x$. Una funciĂłn $\mathcal C^1$ es una soluciĂłn
si satisface la ecuaciĂłn integral
\[
x(t) = x_0 + \int_{t_0}^{t}f\bigl(s, x(s)\bigr)\,\dd s
\]
(teorema fundamental del cĂĄlculo, en ambos sentidos). Dejemos que $T =
\min\bigl(a, \frac bM, \frac1{2L}\bigr)$,$I = [t_0 - T, t_0
+ T]$ y
\[
\mathcal E = \{x \in \mathcal C(I, \R^d) : \norm{x(t) - x_0}
\leq b\ \text{on } I\},
\]
un subconjunto cerrado del espacio de Banach $(\mathcal C(I, \R^d),
\norm\cdot_\infty)$: \omterm{def:b3:complete:complete}{completo}
(\cref{def:b3:complete:complete}). Defina $\Phi(x)(t) = x_0 +
\int_{t_0}^tf(s, x(s))\dd s$: para$x \in \mathcal E$,
$\norm{\Phi(x)(t) - x_0} \leq M\abs{t - t_0} \leq MT \leq b$
--- $\Phi$ asigna $\mathcal E$ a sĂ­ mismo --- y para $x, y \in
\mathcal E$:
\[
\norm{\Phi(x)(t) - \Phi(y)(t)} \leq \Bigl|\int_{t_0}^t
L\,\norm{x(s) - y(s)}\,\dd s\Bigr| \leq LT\,\norm{x -
y}_\infty \leq \tfrac12\norm{x - y}_\infty :
\]
una contracciĂłn. El punto fijo de Banach
(\cref{thm:b3:complete:banach}) proporciona un punto fijo Ăşnico en
$\mathcal E$: existencia y unicidad entre soluciones.
permanecer en $\bar B(x_0, b)$ --- pero cualquier soluciĂłn en $I$ permanece
allĂ­ ($\norm{x(t) - x_0} \leq M\abs{t - t_0} \leq b$ siempre y cuando
como el grĂĄfico queda en $Q$, un argumento \omterm{def:b3:topology:continuity}{continuidad}):
unicidad en $I$.
\end{proof}

\begin{lemma}[Verde la pared]\label{lem:b3:ode:gronwall}
Sea $u \colon I \to \intco0\infty$ \omterm{def:b3:topology:continuity}{continuo}, $t_0 \in
I$ y supongamos \index{GrĂśnwall's lemma}.
\[
u(t) \leq \alpha + \beta\,\Bigl|\int_{t_0}^{t}u(s)\,\dd
s\Bigr| \qquad (t \in I)
\]
con $\alpha \geq 0$, $\beta > 0$. Luego $u(t) \leq
\alpha\,\eu^{\beta\abs{t - t_0}}$en$I$.
\end{lemma}

\begin{proof}
Para $t \geq t_0$: sea $v(t) = \alpha +
\beta\int_{t_0}^tu(s)\dd s$, entonces$u \leq v$,$v' = \beta u
\leq \beta v$y$(v\eu^{-\beta(t - t_0)})' \leq 0$:$v(t)
\leq v(t_0)\eu^{\beta(t-t_0)} = \alpha\eu^{\beta(t - t_0)}$.
Para $t \leq t_0$, aplique lo mismo a $\tilde u(t) = u(2t_0 -
t)$.
\end{proof}

\begin{corollary}[Singularidad y dependencia continua]
\label{cor:b3:ode:uniqueness}
Bajo las hipĂłtesis de \cref{thm:b3:ode:cauchylipschitz},
dos soluciones de $x' = f(t,x)$ que concuerdan en un punto concuerdan
en su intervalo comĂşn de definiciĂłn. Cuantitativamente, si
$x, y$ son dos soluciones con grĂĄficos en una regiĂłn donde $f$ es
$L$-Lipschitz en $x$, entonces
\[
\norm{x(t) - y(t)} \leq \norm{x(t_0) -
y(t_0)}\,\eu^{L\abs{t - t_0}} .
\]
\end{corollary}

\begin{proof}
La estimaciĂłn: $u = \norm{x - y}$ satisface $u(t) \leq u(t_0)
+ L\abs{\int_{t_0}^tu}$ (resta las ecuaciones integrales);
GrĂŤn la pared. Unicidad global: el conjunto de acuerdos $\{t : x(t)
= y(t)\}$ estĂĄ cerrado en el intervalo comĂşn, no estĂĄ vacĂ­o y
abierto --- alrededor de cualquier punto de acuerdo, cubra una pieza \omterm{def:b3:topology:compact}{compacto} de
la grĂĄfica comĂşn por un nĂşmero finito de cajas de Lipschitz y aplicar
la estimaciĂłn con $u(t_1) = 0$ en cada uno: localmente $x \equiv y$.
Un subconjunto cerrado abierto no vacĂ­o de un intervalo lo es todo.
\end{proof}

\section{Soluciones maximas}

\begin{theorem}[Soluciones mĂĄximas; escapar de los compactos]
\label{thm:b3:ode:maximal}
Supongamos $f$ \omterm{def:b3:topology:continuity}{continuo}, \omterm{def:b3:ode:lipschitz}{localmente Lipschitz} en $x$.
\begin{enumerate}
  \item Cada problema de Cauchy tiene un Ăşnico \emph{mĂĄximo}
        soluciĂłn $x \colon \intoo{T_-}{T_+} \to \R^d$: cada
        Otra soluciĂłn a travĂŠs de $(t_0, x_0)$ es su
        restricciĂłn. El intervalo estĂĄ abierto.
  \item (\emph{Escapar}) Por cada \omterm{def:b3:topology:compact}{compacto} $K \subseteq U$
        hay $\varepsilon > 0$ tal que $(t, x(t)) \notin
        K$para todos los$t \in \intoo{T_+ - \varepsilon}{T_+}$ (y
        simĂŠtricamente en $T_-$): \emph{the graph of a \omterm{thm:b3:ode:maximal}{mĂĄximo
        soluciĂłn} eventually leaves every \omterm{def:b3:topology:compact}{compacto} subset of
        $U$}. En particular,
        para $U = \R\times\R^d$ y $T_+ < +\infty$:
        $\norm{x(t)} \to +\infty$ como $t \to T_+^-$
        (\emph{explosiĂłn}).\index{soluciĂłn mĂĄxima}\index{escapar
        de compactos}\index{explosiĂłn}
\end{enumerate}
\end{theorem}

\begin{proof}
(1) Sea $\mathcal S$ el conjunto de todas las soluciones hasta
$(t_0, x_0)$; por \cref{cor:b3:ode:uniqueness} cualquiera de los dos estĂĄ de acuerdo
la intersecciĂłn de sus intervalos, por lo que se pegan: en $J =
\bigcup_{y \in \mathcal S}I_y$, defina$x(t) = y(t)$ para cualquier
$y$ definido en $t$: una soluciĂłn bien definida, evidentemente
mĂĄximo y Ăşnico. $J$ estĂĄ abierto: una soluciĂłn definida en un
punto final podrĂ­a prolongarse por
\cref{thm:b3:ode:cauchylipschitz} en ese punto final.

(2) Supongamos que el reclamo falla en $T_+$: hay $t_n \to
T_+$con$(t_n, x(t_n)) \in K$; tenga en cuenta que esto obliga a$T_+ <
\infty$o, si$T_+ = \infty$, no hay nada que probar ($K$
estĂĄ acotado en el tiempo). Entonces dejemos que $T_+ < \infty$. Compacidad: las constantes uniformes $M, L, a,
b$funcionan para todos los datos de Cauchy en un \omterm{def:b3:topology:topology}{entorno} de$K$
--- concretamente, cubra $K$ con un nĂşmero finito de casillas $Q_i$ como en
la prueba del teorema local y sea $T^* > 0$ el mĂ­nimo de
los tiempos de existencia correspondientes: cualquier dato de Cauchy en $K$
lanza una soluciĂłn que vive al menos $T^*$ mĂĄs allĂĄ de su inicial
tiempo. La aplicaciĂłn de esto en $(t_n, x(t_n))$ con $t_n > T_+ -
T^*/2$extiende$x$mĂĄs allĂĄ de$T_+$(la extensiĂłn concuerda con$x$
por unicidad, luego la prolonga): contradicciĂłn con
maximalidad. Entonces el grĂĄfico deja $K$ definitivamente antes
$T_+$. Para $U = \R\times\R^d$: si $\norm{x(t)}\not\to\infty$,
una secuencia $t_n \to T_+$ mantiene $(t_n, x(t_n))$ en el \omterm{def:b3:topology:compact}{compacto}
$[t_0, T_+]\times\bar B(0, R)$: excluido.
\end{proof}

\begin{corollary}[Existencia global bajo crecimiento lineal]
\label{cor:b3:ode:globallinear}
Si $U = I\times\R^d$ (intervalo abierto $I$) y $\norm{f(t, x)}
\leq \alpha(t)\norm x + \beta(t)$con$\alpha, \beta$
\omterm{def:b3:topology:continuity}{continuo}, entonces cada \omterm{thm:b3:ode:maximal}{soluciĂłn mĂĄxima} se define en todos
$I$.
\end{corollary}

\begin{proof}
En un \omterm{def:b3:topology:compact}{compacto} $[t_0, T] \subseteq I$: $\norm{x(t)} \leq
\norm{x_0} + \int_{t_0}^t(\alpha\norm x + \beta)$, entonces por
GrĂśnwall (con $\alpha, \beta$ delimitado por $A, B$ allĂ­)
$\norm{x(t)} \leq (\norm{x_0} + B(T - t_0))\eu^{A(T - t_0)}$:
acotado. Si $T_+ < \sup I$, el grĂĄfico queda en un \omterm{def:b3:topology:compact}{compacto} de
$I\times\R^d$ cerca de $T_+$: contradice la fuga
(\cref{thm:b3:ode:maximal}).
\end{proof}

\section{Sistemas lineales}

A lo largo de esta secciĂłn $A \colon I \to M_d(\R)$ y $b
\colon I \to \R^d$son \omterm{def:b3:topology:continuity}{continuo}; el sistema es$x' = A(t)x
+ b(t)$ --- crecimiento lineal: todos los \omterm{thm:b3:ode:maximal}{soluciones maximas} viven en todos
de $I$ (\cref{cor:b3:ode:globallinear}).

\begin{theorem}[Estructura]\label{thm:b3:ode:linearstructure}
Las soluciones del sistema homogĂŠneo $x' = A(t)x$ forman una
$d$ espacio vectorial dimensional $S_H$; para cada $t_0$, el
La evaluaciĂłn $x \mapsto x(t_0)$ es un isomorfismo $S_H \to
\R^d$. El \emph{disolvente}\index{resolutivo de un lineal
sistema} $R(t, s) \in GL_d(\R)$, definido por: $t \mapsto R(t,
s)v$es la soluciĂłn con valor$v$en$s$, satisface
\[
R(s,s) = I,\quad R(t, u)R(u, s) = R(t, s),\quad
\partial_tR(t,s) = A(t)R(t,s),
\]
y el problema no homogĂŠneo se resuelve con \emph{Duhamel's
fĂłrmula}\index{La fĂłrmula de Duhamel}:
\[
x(t) = R(t, t_0)\,x_0 + \int_{t_0}^{t}R(t, s)\,b(s)\,\dd s .
\]
Finalmente el \emph{Wronskiano}\index{Wronskiano} $w(t) = \det
R(t, s)$obedece a la fĂłrmula de Liouville$w' =
\operatorname{tr}A(t)\,w$, por lo que$w(t) =
\exp\bigl(\int_s^t\operatorname{tr}A\bigr) > 0$.
\end{theorem}

\begin{proof}
La linealidad de la ecuaciĂłn convierte las soluciones en un espacio vectorial;
la evaluaciĂłn es lineal, inyectiva (unicidad: una soluciĂłn
que desaparece en $t_0$ es $\equiv 0$) y sobreyectivo (existencia):
dimensiĂłn $d$. Las propiedades \omterm{thm:b3:ode:linearstructure}{disolvente} reafirman la unicidad
(ambos lados de cada identidad resuelven el mismo problema de Cauchy);
invertibilidad de $R(s,t)R(t,s) = I$. Duhamel: diferenciar
la fĂłrmula --- $x'(t) = A(t)R(t,t_0)x_0 + R(t,t)b(t) +
\int_{t_0}^tA(t)R(t,s)b(s)\dd s = A(t)x(t) + b(t)$
(la diferenciaciĂłn bajo la integral es legĂ­tima: la
el integrando es $\mathcal C^1$ en $t$ con derivada \omterm{def:b3:topology:continuity}{continuo}
en $(t,s)$; o verificar mediante la ecuaciĂłn integral). Liouville:
$w(t + h) = \det\bigl(R(t+h, t)\bigr)w(t)$ y $R(t + h, t) =
I + hA(t) + o(h)$(de la ecuaciĂłn integral), por lo que$\det = 1
+ h\operatorname{tr}A(t) + o(h)$(expansiĂłn de$\det$ en
$I$): $w'(t) = \operatorname{tr}A(t)\,w(t)$; integrar el
EDO lineal escalar.
\end{proof}

\begin{theorem}[matriz exponencial]\label{thm:b3:ode:matrixexp}
Para $A \in M_d(\C)$, la serie $\eu^{A} =
\sum_{n\geq0}\frac{A^n}{n!}$ converge (absolutamente, en cualquier
norma submultiplicativa), $\eu^{A+B} = \eu^A\eu^B$ siempre que
$AB = BA$, y $t \mapsto \eu^{tA}$ es el \omterm{thm:b3:ode:linearstructure}{disolvente} del
sistema constante: $R(t,s) = \eu^{(t-s)A}$; es $\mathcal
C^\infty$con$\frac{\dd}{\dd t}\eu^{tA} = A\eu^{tA}$.
AdemĂĄs: si $\operatorname{Re}\lambda < -\alpha < 0$ para
cada valor propio $\lambda$ de $A$, luego $\vertiii{\eu^{tA}}
\leq C\eu^{-\alpha t}$para$t \geq 0$.\index{matriz
exponencial}
\end{theorem}

\begin{proof}
Convergencia: $\vertiii{A^n/n!} \leq \vertiii A^n/n!$, sumable
(\cref{exo:b3:complete:1}(b) en el ĂĄlgebra de Banach
$M_d$). Para desplazarse $A, B$: el producto de Cauchy de los dos
series absolutamente convergentes se reorganizan, a travĂŠs del binomio
teorema (vĂĄlido cuando $AB = BA$), en
$\sum_n\frac{(A+B)^n}{n!}$. Diferenciabilidad, directamente:
$\frac{\eu^{(t+h)A} - \eu^{tA}}h = \eu^{tA}\frac{\eu^{hA} -
I}{h} \to \eu^{tA}A$desde$\norm{\frac{\eu^{hA} - I}h - A}
\leq \sum_{n\geq2}\frac{\abs h^{n-1}\vertiii A^n}{n!} =
O(h)$. Por lo tanto$t \mapsto \eu^{(t - s)A}v$ resuelve el problema de Cauchy.
problema al definir $R(t,s)v$. LĂ­mite espectral: por el \omterm{thm:b3:modules:jordan}{forma jordana}
(\cref{thm:b3:modules:jordan}), $A = P(D + N)P^{-1}$ con $D$
diagonal que lleva los valores propios, $N$ nilpotente y $DN =
ND$. Entonces$\eu^{tA} = P\,\eu^{tD}\eu^{tN}P^{-1}$ con
$\vertiii{\eu^{tD}} \leq \eu^{-(\alpha + \delta)t}$ para algunos
Polinomio $\delta > 0$ ($t \geq 0$) y $\eu^{tN}$ en $t$

(polinomio superado por $\eu^{-\delta t}$).
\end{proof}

\section{Flujos, equilibrios, estabilidad.}

Considere ahora la ecuaciĂłn \emph{autĂłnomo} $x' = F(x)$, $F
\colon \Omega \to \R^d$\omterm{def:b3:ode:lipschitz}{localmente Lipschitz} en el abierto$\Omega
\subseteq \R^d$. Escriba$\varphi_t(x_0) = x(t)$ para el
\omterm{thm:b3:ode:maximal}{soluciĂłn mĂĄxima} con $x(0) = x_0$ (el
\emph{fluir}\index{flujo de un campo vectorial}); la autonomĂ­a da
la propiedad del grupo $\varphi_{t+s} =
\varphi_t\circ\varphi_s$ donde se definiĂł (ambos lados resuelven el
mismo problema en el momento $s$).

\begin{definition}\label{def:b3:ode:stability}
Un \emph{equilibrio} es un punto $\bar x$ con $F(\bar x) =
0$(es decir, $\varphi_t(\bar x) = \bar x$). es
\emph{estable}\index{estabilidad (Lyapunov)} si por cada
$\varepsilon > 0$ existe $\delta > 0$ tal que
$\norm{x_0 - \bar x} < \delta$ implica que la soluciĂłn
existe para todos los $t \geq 0$ con $\norm{\varphi_t(x_0) - \bar
x} < \varepsilon$; \emph{asintĂłticamente estable} si ademĂĄs
$\varphi_t(x_0) \to \bar x$ para todos los $x_0$ cerca de $\bar x$.
\end{definition}

\begin{theorem}[Funciones de Lyapunov]\label{thm:b3:ode:lyapunov}
Sea $\bar x$ un equilibrio y $V \colon \mathcal V \to
\R$una funciĂłn$\mathcal C^1$sobre un \omterm{def:b3:topology:topology}{entorno} de$\bar x$
con:\index{funciĂłn de Lyapunov}
\[
V(\bar x) = 0,\qquad V(x) > 0 \text{ for } x \neq \bar x,
\qquad \dot V(x) = \nabla V(x)\cdot F(x) \leq 0 .
\]
Entonces $\bar x$ es \omterm{def:b3:ode:stability}{estable}. Si ademĂĄs $\dot V < 0$ estĂĄ fuera de $\bar
x$, entonces$\bar x$ es asintĂłticamente \omterm{def:b3:ode:stability}{estable}.
\end{theorem}

\begin{proof}
A lo largo de una soluciĂłn, $\frac{\dd}{\dd t}V(x(t)) = \dot V(x(t))
\leq 0$:$V$disminuye. Dado$\varepsilon$ (lo suficientemente pequeĂąo
que $\bar B(\bar x, \varepsilon) \subseteq \mathcal V$), dejemos
$m = \min\{V(x) : \norm{x - \bar x} = \varepsilon\} > 0$
(compacidad, positividad) y elija $\delta < \varepsilon$
con $V < m$ en $B(\bar x, \delta)$ (\omterm{def:b3:topology:continuity}{continuidad}). una soluciĂłn
comenzando en $B(\bar x, \delta)$ tiene $V(x(t)) < m$ para todos
tiempos posteriores, por lo que nunca podrĂĄ llegar a la esfera $\norm{x -
\bar x} = \varepsilon$(donde$V \geq m$): permanece en la
bola --- y luego existe para todos los $t \geq 0$: la soluciĂłn
permanece en el \omterm{def:b3:topology:compact}{compacto} $\bar B$, por lo que
Fuerzas \cref{thm:b3:ode:maximal}(2) (\omterm{thm:b3:ode:maximal}{escapar de los compactos})
$T_+ = +\infty$. Estabilidad.

Caso asintĂłtico: dejemos que $x(t)$ comience en $B(\bar x, \delta)$;
$V(x(t))$ disminuye a algo de $c \geq 0$. Si $c > 0$: el
trayectoria permanece en $K = \{x \in \bar B(\bar x,
\varepsilon): V(x) \geq c\}$, un conjunto \omterm{def:b3:topology:compact}{compacto} que excluye un
\omterm{def:b3:topology:topology}{entorno} de $\bar x$ ($V$ es \omterm{def:b3:topology:continuity}{continuo} con $V(\bar x)
= 0 < c$). En$K$, la funciĂłn$\dot V$ es \omterm{def:b3:topology:continuity}{continuo},
estrictamente negativo, por lo tanto $\mu = \max_K\dot V < 0$
(compacidad); luego $V(x(t)) \leq V(x(0)) + \mu t \to
-\infty$: absurdo,$V \geq 0$. entonces
$c = 0$ y $x(t) \to \bar x$ (los puntos a la distancia $\geq
\rho$de$\bar x$dentro de la pelota tienen$V \geq m_\rho >

\end{proof}

\begin{theorem}[Estabilidad por linealizaciĂłn]
\label{thm:b3:ode:linearization}
Sea $F$ $\mathcal C^1$, $F(\bar x) = 0$, $A = DF(\bar x)$.
Si cada valor propio de $A$ tiene $\operatorname{Re}\lambda <
0$, entonces$\bar x$ es asintĂłticamente \omterm{def:b3:ode:stability}{estable}.\index{estabilidad por
linealizaciĂłn}
\end{theorem}

\begin{proof}
Traduce $\bar x$ a $0$ y escribe $F(x) = Ax + g(x)$ con
$g(x) = o(\norm x)$ (diferenciabilidad $\mathcal C^1$). elegir
$\alpha > 0$ con $\vertiii{\eu^{tA}} \leq C\eu^{-\alpha t}$
($t \geq 0$; \cref{thm:b3:ode:matrixexp}) y $r > 0$ con
$\norm{g(x)} \leq \frac{\alpha}{2C}\norm x$ para $\norm x \leq
r$. Duhamel con$b(s) = g(x(s))$:
\[
x(t) = \eu^{tA}x_0 + \int_0^t\eu^{(t-s)A}g(x(s))\,\dd s,
\]
vĂĄlido hasta $\norm{x(s)} \leq r$. Entonces $u(t) =
\eu^{\alpha t}\norm{x(t)}$ satisface
\[
u(t) \leq C\norm{x_0} +
\int_0^{t}C\,\frac{\alpha}{2C}\,u(s)\,\dd s ,
\]
entonces GrĂśnwall da $u(t) \leq C\norm{x_0}\eu^{\alpha t/2}$,
es decir,\ $\norm{x(t)} \leq C\norm{x_0}\eu^{-\alpha t/2}$. si
$\norm{x_0} < r/C$, el lĂ­mite a priori mantiene $\norm{x(t)} <
r$para todos los$t$ (un argumento \omterm{def:b3:topology:continuity}{continuidad}/bootstrap: el conjunto de
horarios en los que $\norm x \leq r$ estĂĄ abierto y cerrado
$\intco0{T_+}$ dada la estimaciĂłn estricta), la soluciĂłn es
global, y converge a $0$ exponencialmente: asintĂłtico
estabilidad.
\end{proof}

\end{method}
