\chapter{Medidas del Producto, Fubini, Cambio de Variables}
\label{ch:b3:product}

La teoría unidimensional de Lebesgue se vuelve multidimensional
Cálculo a través de dos teoremas. \emph{Tonelli--Fubini} dice que
integrales sobre productos son integrales iteradas --- cortar es
legítimo, en cualquier orden, bajo hipótesis uno realmente puede
cheque. El \emph{fórmula de cambio de variables} transporta
integrales a lo largo de difeomorfismos $\mathcal C^1$, con el
Determinante jacobiano como tipo de cambio por volumen; probamos
completamente, a partir del caso lineal donde explica
cuál es el determinante \emph{is}. Cascada de aplicaciones: la capa
fórmula de pastel, convolución, \omterm{ex:b3:product:polar}{coordenadas polares}, el volumen del
$n$-ball --- y, en el problema del fin de semana, la fórmula de Stirling
con un análisis de errores honesto.

\section{Producto $\sigma$-álgebras y medidas de producto}

\begin{definition}\label{def:b3:product:sigma}
Para \omterm{def:b3:lebesgue:measurable}{mensurable} espacios $(X, \mathcal A)$, $(Y, \mathcal B)$, el
\emph{product $\sigma$-algebra}\index{product
$\sigma$-algebra} $\mathcal A \otimes \mathcal B$ en $X \times
Y$es generado por el \emph{rectángulos}$A \times B$($A \in
\mathcal A$,$B \in \mathcal B$) --- un sistema$\pi$. Para$E
\subseteq X\times Y$y$x \in X$, \emph{sección} es$E_x
= \{y : (x,y) \in E\}$; para una función$f$ en el producto,
$f_x = f(x, \cdot)$.
\end{definition}

\begin{proposition}\label{prop:b3:product:sections}
(a) Si $E \in \mathcal A\otimes\mathcal B$, cada sección $E_x
\in \mathcal B$(y simétricamente); si$f$es medible$\mathcal
A\otimes\mathcal B$, cada$f_x$es medible$\mathcal
B$.
(b) $\mathcal B(\R^m)\otimes\mathcal B(\R^n) = \mathcal
B(\R^{m+n})$.
\end{proposition}

\begin{proof}
(a) Buenos conjuntos: $\{E : E_x \in \mathcal B\ \forall x\}$ es un
$\sigma$-álgebra (las secciones conmutan con complementos y
uniones contables) que contienen los rectángulos. Para $f$:
$(f_x)^{-1}(B) = (f^{-1}(B))_x$.
(b) ($\subseteq$) Rectángulos de conjuntos de Borel: basta con que
open$\times$las cajas abiertas son Borel en $\R^{m+n}$ (están abiertas)
y que los rectángulos generales de Borel son límites --- buenos conjuntos
nuevamente: $\{A : A\times\R^n \in \mathcal B(\R^{m+n})\}$ es un
$\sigma$-álgebra que contiene las aperturas; cruzar dos de esos.
($\supseteq$) Cada \omterm{def:b3:topology:topology}{conjunto abierto} de $\R^{m+n}$ es una unión contable
de cajas abiertas racionales $U\times V$: contenidas en el producto
$\sigma$-álgebra.
\end{proof}

\begin{theorem}[Medida del producto]\label{thm:b3:product:existence}
Sean $(X, \mathcal A, \mu)$ y $(Y, \mathcal B, \nu)$
\emph{$\sigma$-finite}. Para cada $E \in \mathcal
A\otimes\mathcal B$, la función$x \mapsto \nu(E_x)$ es
\omterm{def:b3:lebesgue:measurable}{mensurable} y \index{medida del producto}
\[
(\mu\otimes\nu)(E) = \int_X \nu(E_x)\,\dd\mu(x)
\]
define el único \omterm{def:b3:measure:measure}{medida} en $\mathcal A\otimes\mathcal B$
con $(\mu\otimes\nu)(A\times B) = \mu(A)\nu(B)$. es
$\sigma$-finito y simétrico: el mismo \omterm{def:b3:measure:measure}{medida} se obtiene mediante
integrando secciones $x$ contra $\nu$.
\end{theorem}

\begin{proof}
\emph{\omterm{def:b3:lebesgue:measurable}{Mensurabilidad} of $x \mapsto \nu(E_x)$.} Primero deja $\nu$
ser finito. La clase $\mathcal D$ de $E$ para la cual la aplicación es
\omterm{def:b3:lebesgue:measurable}{mensurable} contiene los rectángulos ($\nu((A\times B)_x) =
\nu(B)\mathbf 1_A(x)$) y es un sistema$\lambda$: para$E
\subseteq F$en$\mathcal D$,$\nu((F\setminus E)_x) =
\nu(F_x) - \nu(E_x)$(finitud); para$E_n \uparrow E$,
$\nu((E_n)_x) \uparrow \nu(E_x)$ (\omterm{def:b3:topology:continuity}{continuidad} desde abajo), y
Los límites monótonos de \omterm{def:b3:lebesgue:measurable}{funciones medibles} son \omterm{def:b3:lebesgue:measurable}{mensurable}.
Los rectángulos forman un sistema $\pi$: Dynkin
(\cref{thm:b3:measure:dynkin}) da $\mathcal D = \mathcal
A\otimes\mathcal B$. Si$\nu$es$\sigma$-finito, escriba$Y =
\bigcup Y_k$,$Y_k \uparrow$,$\nu(Y_k) < \infty$:$\nu(E_x) =
\lim_k\nu_k(E_x)$con$\nu_k = \nu(\cdot\cap Y_k)$ finito.

\emph{\omterm{def:b3:measure:measure}{Medida}.} $\sigma$: la aditividad de $E \mapsto
\int\nu(E_x)\dd\mu$ se deriva de
\cref{cor:b3:lebesgue:additivity} (secciones de conjuntos disjuntos
son disjuntos). En rectángulos da $\mu(A)\nu(B)$.
\emph{Unicidad}: dos candidatos coinciden en el sistema $\pi$ de
rectángulos; $\sigma$-finiteness proporciona rectángulos $X_k\times
Y_k \uparrow X\times Y$ de \omterm{def:b3:measure:measure}{medida} finito:
\cref{thm:b3:measure:uniqueness}. Simetría: la
la construcción de otro orden también es un acuerdo \omterm{def:b3:measure:measure}{medida} sobre
rectángulos --- únicos, por lo tanto iguales.
\end{proof}

\begin{definition}\label{def:b3:product:lebesgue}
\emph{Lebesgue measure on $\R^d$}\index{Lebesgue measure on
$\R^d$} es $\lambda_d = \lambda\otimes\cdots\otimes\lambda$
(factores $d$; la asociatividad de la construcción se verifica en
cajas y propagadas por unicidad). Es el único Borel
\omterm{def:b3:measure:measure}{medida} dándole a cada caja $\prod\intoc{a_i}{b_i}$ su volumen
$\prod(b_i - a_i)$; es invariante en la traducción (traduce
están de acuerdo en las casillas), $\sigma$-finito y \omterm{def:b3:complete:complete}{completo} después
Carathéodory \omterm{thm:b3:complete:completion}{terminación} --- escribimos $\lambda_d$ para el
completó \omterm{def:b3:measure:measure}{medida} e integre en consecuencia.
\end{definition}

\section{Tonelli y Fubini}

\begin{theorem}[tonelli]\label{thm:b3:product:tonelli}
$\mu, \nu$ $\sigma$-finito, $f \colon X\times Y \to [0,
+\infty]$\omterm{def:b3:lebesgue:measurable}{mensurable}. Entonces$x \mapsto \int_Y f_x\,\dd\nu$ es
\omterm{def:b3:lebesgue:measurable}{mensurable} y\index{teorema de tonelli}
\[
\int_{X\times Y}f\,\dd(\mu\otimes\nu)
= \int_X\Bigl(\int_Y f(x,y)\,\dd\nu(y)\Bigr)\dd\mu(x)
= \int_Y\Bigl(\int_X f(x,y)\,\dd\mu(x)\Bigr)\dd\nu(y).
\]
\end{theorem}

\begin{proof}
La máquina estándar. Para $f = \mathbf 1_E$ esto es
\cref{thm:b3:product:existence} (y su forma simétrica). Por
linealidad que se cumple para el simple $f \geq 0$. Para $f \geq
0$general: tome$s_n \nearrow f$ simple
(\cref{thm:b3:lebesgue:approximation}); luego $\int_Y(s_n)_x
\dd\nu \nearrow \int_Y f_x\dd\nu$para cada$x$(MCT en$Y$),
por lo que los miembros de la izquierda convergen por MCT en $X$, mientras que
$\int s_n\,\dd(\mu\otimes\nu) \nearrow \int f$ por MCT en el
producto.
\end{proof}

\begin{theorem}[fubini]\label{thm:b3:product:fubini}
$\mu, \nu$ $\sigma$-finito, $f \in L^1(\mu\otimes\nu)$. entonces
para $\mu$-a.e.\ $x$ la sección $f_x$ es integrable en $\nu$, la
La función definida por e.e. $x \mapsto \int f_x\dd\nu$ es
\omterm{def:b3:lebesgue:l1}{integrable}, y las dos integrales iteradas son iguales
$\int f\,\dd(\mu\otimes\nu)$.\index{teorema de fubini}
\end{theorem}

\begin{proof}
Tonelli aplicado a $\abs f$ muestra que $\varphi(x) = \int\abs{f_x}
\dd\nu$tiene una integral finita, por lo tanto, es finita ae:$f_x \in
L^1(\nu)$para ae$x$. Dividir$f = f^+ - f^-$ (caso real;
complejo por componentes): Tonelli calcula cada iterado
integral de $f^\pm$ como $\int f^\pm\dd(\mu\otimes\nu) <
\infty$, y la diferencia definida por a.e. se integra a la
diferencia. Simétricamente para el otro orden.
\end{proof}

\begin{method}\label{met:b3:product:fubini}
Para intercambiar dos integrales (o una integral y una suma, o dos
sumas): si el integrando es \emph{no negativo}, intercambie
libremente (Tonelli). De lo contrario, aplique primero Tonelli a $\abs f$
en el orden que sea más fácil de estimar; si el resultado es
finito, Fubini legitima el intercambio. Nunca te saltes el
Comprobación $\abs f$: el integrando de \cref{exo:b3:product:4} tiene dos
integrales iteradas con \emph{diferentes valores}.
\end{method}

\begin{proposition}[Pastel de capas]\label{prop:b3:product:layercake}
Para $f \geq 0$ \omterm{def:b3:lebesgue:measurable}{mensurable} en $(X, \mathcal A, \mu)$
$\sigma$-finito:\index{fórmula de pastel de capas}
\[
\int_X f\,\dd\mu = \int_0^{+\infty}\mu(\{f > t\})\,\dd t,
\qquad
\int_X f^p\,\dd\mu = p\int_0^{+\infty}t^{p-1}\mu(\{f >
t\})\,\dd t \quad (p \geq 1).
\]
\end{proposition}

\begin{proof}
Aplicar Tonelli a $\mathbf 1_{\{(x,t) : 0 < t < f(x)\}}$ en $X
\times \intoo0{+\infty}$(\omterm{def:b3:lebesgue:measurable}{mensurable}: es$\{(x,t): f(x) - t
> 0\}\cap\{t > 0\}$, una combinación tipo Borel del
\omterm{def:b3:lebesgue:measurable}{mensurable} $(x,t)\mapsto f(x) - t$): integrando primero en $t$
da $\int f\,\dd\mu$; en $x$ primero, $\int_0^\infty\mu(f >
t)\dd t$. Para$f^p$: sustituya$t = s^p$en$\int\mu(f^p > t)
\dd t$, es decir, aplique la primera fórmula a$f^p$ y cambie
variables en la integral unidimensional ($\{f^p > s^p\} =
\{f > s\}$).
\end{proof}

\begin{theorem}[Convolución en $L^1$]\label{thm:b3:product:convolution}
Para $f, g \in L^1(\R^d, \lambda_d)$, la integral
\[
(f * g)(x) = \int_{\R^d} f(x - y)\,g(y)\,\dd y
\]
converge absolutamente para a.e.\ $x$, define $f * g \in
L^1(\R^d)$con$\norm{f*g}_1 \leq \norm f_1\norm g_1$, y
$*$ es conmutativo y asociativo.\index{circunvolución}
\end{theorem}

\begin{proof}
$(x, y) \mapsto f(x-y)g(y)$ es \omterm{def:b3:lebesgue:measurable}{mensurable} ($ (x,y)\mapsto x -
y$ es \omterm{def:b3:topology:continuity}{continuo}; componer y multiplicar). Tonelli:
\[
\int\!\!\int \abs{f(x-y)}\abs{g(y)}\,\dd y\,\dd x
= \int\abs{g(y)}\Bigl(\int\abs{f(x - y)}\dd x\Bigr)\dd y
= \norm f_1\norm g_1 < \infty
\]
(invarianza de traducción de $\lambda_d$ en la integral interna).
Entonces la integral doble es finita; Fubini da a.e.\ absoluta
convergencia y el límite de norma $\norm{f*g}_1 \leq \norm
f_1\norm g_1$. Conmutatividad: sustituir$y \mapsto x - y$
(invariancia de traducción y reflexión --- reflexión
la invariancia se mantiene en las cajas, por lo tanto en todas partes por unicidad).
Asociatividad: Tonelli--Fubini sobre una integral triple.
\end{proof}

\section{Cambio de variables}

\begin{theorem}[Cambio lineal de variables]
\label{thm:b3:product:linearchange}
Para $T \in GL_d(\R)$ y $A \in \mathcal B(\R^d)$:
$\lambda_d(T(A)) = \abs{\det T}\,\lambda_d(A)$; en consecuencia
$\int f(y)\dd y = \abs{\det T}\int f(Tx)\,\dd x$ para $f \geq 0$
o \omterm{def:b3:lebesgue:l1}{integrable}.\index{cambio lineal de variables}
\end{theorem}

\begin{proof}
El \omterm{def:b3:measure:measure}{medida} $\mu_T(A) = \lambda_d(T(A))$ es un Borel \omterm{def:b3:measure:measure}{medida}
(\omterm{def:b3:topology:continuity}{homeomorfismos} preservar conjuntos de Borel,
\cref{pb:b3:measure:1}), invariante de traducción
($T(A + x) = T(A) + Tx$), finito en la caja unitaria: por el
caracterización de \omterm{def:b3:measure:lebesgueouter}{medida de lebesgue}
(\cref{exo:b3:measure:6}, cuya prueba funciona textualmente en
$\R^d$ con cubos diádicos), $\mu_T = c(T)\lambda_d$ con $c(T)
= \lambda_d(T(\intco01^d))$. La aplicación$T \mapsto c(T)$ es
multiplicativo ($c(ST) = c(S)c(T)$, componiendo), por lo que
basta con calcular $c$ en generadores de $GL_d$: elemental
matrices. Diagonal $\operatorname{diag}(a, 1, \dots, 1)$:
asigna el cubo unitario a una caja de volumen $\abs a$: $c = \abs a =
\abs\det$. Transposición de coordenadas: permuta el cubo:$c
= 1 = \abs\det$. Transvección$T(x) = x + \alpha
x_2e_1$: la imagen del cubo unitario es un prisma cizallado; por
Tonelli su \omterm{def:b3:measure:measure}{medida} es $\int\lambda_1(\text{section})\dd x_2
\cdots \dd x_d = 1$, siendo cada sección$x_1$ un intervalo de
longitud $1$: $c = 1 = \abs{\det}$. Toda matriz invertible es una
producto de estos (eliminación gaussiana), y tanto $c$ como
$\abs\det$ son multiplicativos: $c(T) = \abs{\det T}$. el
La fórmula integral sigue la máquina estándar (indicadores,
simple, MCT).
\end{proof}

\begin{theorem}[Cambio de variables]\label{thm:b3:product:changeofvar}
Deje que $U, V \subseteq \R^d$ esté abierto y $\Phi \colon U \to V$ a
$\mathcal C^1$ difeomorfismo. Por cada \omterm{def:b3:lebesgue:measurable}{mensurable} $f \colon V
\to [0, +\infty]$(o$f \in L^1(V)$):\index{cambio de
fórmula de variables}
\[
\int_V f(y)\,\dd y = \int_U
f\bigl(\Phi(x)\bigr)\,\abs{\det D\Phi(x)}\,\dd x .
\]
\end{theorem}

\begin{proof}
Escribe $J(x) = \abs{\det D\Phi(x)}$. El corazón de la prueba es
la desigualdad
\[
\lambda_d\bigl(\Phi(A)\bigr) \leq \int_A J\,\dd\lambda_d
\qquad\text{for every Borel } A \subseteq U;
\tag{$*$}
\]
El paso 4 a continuación actualiza $(*)$ --- se aplica tanto a $\Phi$ como a
$\Phi^{-1}$ --- a la igualdad del teorema. Tenga en cuenta que
$\Phi^{-1}$ es en sí mismo un difeomorfismo $\mathcal C^1$ con
Jacobiano $\abs{\det D\Phi^{-1}(y)} = J(\Phi^{-1}y)^{-1}$
(regla de la cadena en $\Phi\circ\Phi^{-1} = \mathrm{id}$).

\emph{Paso 1: $(*)$ para cubos con factor de distorsión.} Arreglar un
cubo cerrado $Q \subseteq U$ de \omterm{ex:b3:groups:actions}{centro} $x_0$ y lado $2r$
(bola superior a la normal). Reclamo: por cada $\varepsilon > 0$, si $\Phi$
es diferenciable en $Q$ con
$\norm{D\Phi(x) - D\Phi(x_0)} \leq \varepsilon$ en $Q$
(\omterm{def:b3:banach:operator}{norma del operador} para la norma superior), luego
\[
\Phi(Q) \subseteq \Phi(x_0) + D\Phi(x_0)\Bigl(\,\bigl(1 +
\varepsilon\norm{D\Phi(x_0)^{-1}}\bigr)\,(Q - x_0)\Bigr),
\]
porque para $x \in Q$, la desigualdad del valor medio se aplicó a
$\Phi(x) - \Phi(x_0) - D\Phi(x_0)(x - x_0)$ da
$\norm{\Phi(x) - \Phi(x_0) - D\Phi(x_0)(x-x_0)}_\infty \leq
\varepsilon\norm{x - x_0}_\infty \leq \varepsilon r$, y
$D\Phi(x_0)^{-1}$ saca este defecto a una
$\varepsilon\norm{D\Phi(x_0)^{-1}}\,r$-ampliación del cubo.
Por \cref{thm:b3:product:linearchange},
\[
\lambda_d(\Phi(Q)) \leq \abs{\det D\Phi(x_0)}\,
\bigl(1 + \varepsilon\,C\bigr)^{d}\,\lambda_d(Q),
\qquad C = \sup_{Q}\norm{D\Phi(\cdot)^{-1}} .
\]

\emph{Step 2: $(*)$ for \omterm{def:b3:topology:compact}{compacto} cubes, by subdivision.} Sea $Q
\subseteq U$un cubo \omterm{def:b3:topology:compact}{compacto} y$\varepsilon > 0$. En$Q$,
$D\Phi$ es uniformemente \omterm{def:b3:topology:continuity}{continuo} y $\norm{D\Phi^{-1}}$
acotado (compacto); subdividir $Q$ en subcubos $2^{kd}$
$Q_i$ lo suficientemente pequeño como para que la oscilación de $D\Phi$ en cada uno sea
$\leq \varepsilon$. Paso 1 en cada subcubo (\omterm{ex:b3:groups:actions}{centro} $x_i$):
\[
\lambda_d(\Phi(Q)) \leq \sum_i\lambda_d(\Phi(Q_i))
\leq (1 + C\varepsilon)^d \sum_i \abs{\det
D\Phi(x_i)}\,\lambda_d(Q_i)
\leq (1 + C\varepsilon)^d\Bigl(\int_Q J + \varepsilon'\Bigr),
\]
el último paso porque $\sum_i\abs{\det D\Phi(x_i)}\mathbf
1_{Q_i} \to J$uniformemente en$Q$(\omterm{def:b3:topology:continuity}{continuidad} de$\det D\Phi$)
--- Comparación de suma de Riemann. Deja $\varepsilon \to 0$: $(*)$
es válido para cubos \omterm{def:b3:topology:compact}{compacto}.

\emph{Paso 3: $(*)$ para todos los Borel $A$.} La función establecida $A
\mapsto \lambda_d(\Phi(A))$, en los subconjuntos de Borel de$U$, es una
\omterm{def:b3:measure:measure}{medida} ($\Phi$ es una biyección sobre $V$ que preserva los conjuntos de Borel
y desunión contable), y también lo es $A \mapsto \int_AJ$.
Cada subconjunto abierto de $U$ es una unión contable de
cubos \omterm{def:b3:topology:compact}{compacto} diádicos casi inconexos (diádicos estándar
descomposición: tome los cubos diádicos máximos contenidos en \omterm{def:b3:topology:topology}{abierto
conjunto}), y ambos \omterm{def:b3:measure:measure}{medidas} son aditivos entre ellos (límites
de cubos son $\lambda_d$ nulos y sus imágenes $\Phi$ son
nulo en el Paso 2 aplicado a delgadas cubiertas de cubos de las caras):
$(*)$ pasa de cubos a \omterm{def:b3:topology:topology}{conjuntos abiertos}. General Borel $A$:
agotar $U$ por \omterm{def:b3:topology:compact}{compactos} $K_m \uparrow U$ con $K_m \subseteq
\mathring K_{m+1}$, y arreglar$m$; en$\mathring K_{m+1}$,$J$
está limitado por algunos $M_m$. Por regularidad exterior de $\lambda_d$
(prueba como en \cref{thm:b3:measure:regularity}, con cajas),
elija \omterm{def:b3:topology:topology}{conjuntos abiertos} $O_n$ con $A \cap K_m \subseteq O_n
\subseteq \mathring K_{m+1}$y$\lambda_d\bigl(O_n \setminus
(A\cap K_m)\bigr) \to 0$. entonces
\[
\lambda_d\bigl(\Phi(A\cap K_m)\bigr) \leq
\lambda_d\bigl(\Phi(O_n)\bigr) \leq \int_{O_n}J
\leq \int_{A\cap K_m}J + M_m\,\lambda_d\bigl(O_n\setminus(A\cap
K_m)\bigr) \xrightarrow[n\to\infty]{} \int_{A\cap K_m}J .
\]
Deje $m \to \infty$: \omterm{def:b3:topology:continuity}{continuidad} desde abajo a la izquierda, MCT encendido
la derecha. Esto establece $(*)$.

\emph{Paso 4: igualdad y fórmula integral.} Primero
extienda $(*)$ de conjuntos a integrales: por cada \omterm{def:b3:lebesgue:measurable}{mensurable} $g
\geq 0$en$V$,
\[
\int_V g(y)\,\dd y \leq \int_U g(\Phi(x))\,J(x)\,\dd x .
\tag{$**$}
\]
De hecho, para $g = \mathbf 1_B$ esto es $(*)$ con $A =
\Phi^{-1}(B)$; la linealidad lo extiende a$g$ simple y MCT a
todos $g \geq 0$ (la máquina estándar). Ahora aplique $(**)$ dos veces:
primero a $g$, luego --- para el difeomorfismo $\Phi^{-1}$ ---
a la función $x \mapsto g(\Phi(x))J(x)$:
\[
\int_Vg \leq \int_U g(\Phi(x))J(x)\dd x
\leq \int_V g(y)\,J(\Phi^{-1}y)\,\abs{\det
D\Phi^{-1}(y)}\,\dd y = \int_V g ,
\]
desde $J(\Phi^{-1}y)\abs{\det D\Phi^{-1}(y)} = \abs{\det\bigl(
D\Phi(\Phi^{-1}y)\,D\Phi^{-1}(y)\bigr)} = 1$ (regla de cadena en
$\Phi\circ\Phi^{-1} = \mathrm{id}$). Todas las desigualdades son
igualdades: la fórmula es válida para $g \geq 0$ y para $L^1$
funciones por descomposición.
\end{proof}

\begin{example}[Coordenadas polares; el gaussiano otra vez]
\label{ex:b3:product:polar}
$\Phi(r, \theta) = (r\cos\theta, r\sin\theta)$ es un difeomorfismo $\mathcal
C^1$de$\intoo0{+\infty}\times\intoo0{2\pi}$
en $\R^2$ menos media línea (conjunto nulo), con $\det D\Phi =
r$:\index{coordenadas polares}
\[
\int_{\R^2}f(x, y)\,\dd x\,\dd y
= \int_0^{2\pi}\!\!\int_0^{+\infty}
f(r\cos\theta, r\sin\theta)\,r\,\dd r\,\dd\theta .
\]
Para $f = \eu^{-x^2-y^2}$, Tonelli y esta fórmula dan
\[
G^2 = \Bigl(\int_\R \eu^{-x^2}\dd x\Bigr)^2
= \int_{\R^2}\eu^{-x^2-y^2}
= 2\pi\int_0^\infty r\eu^{-r^2}\dd r = \pi:
\]
la prueba clásica de dos líneas de $G = \sqrt\pi$, ahora completamente
justificado (compare la prueba del parámetro de
\cref{pb:b3:lebesgue:1}).
\end{example}

\begin{theorem}[Volumen de la bola unitaria]
\label{thm:b3:product:ballvolume}
Deje $v_d = \lambda_d(B(0,1))$ en $\R^d$. Entonces\index{volumen de
la bola unitaria}
\[
v_d = \frac{\pi^{d/2}}{\Gamma\bigl(\frac d2 + 1\bigr)} :
\qquad
v_1 = 2,\quad v_2 = \pi,\quad v_3 = \tfrac{4\pi}3,\quad
v_4 = \tfrac{\pi^2}2,\ \dots
\]
\end{theorem}

\begin{proof}
Calcule $I = \int_{\R^d}\eu^{-\norm x_2^2}\dd\lambda_d$ dos veces.
Por Tonelli factoriza: $I = G^d = \pi^{d/2}$. Por el pastel de capas
fórmula (\cref{prop:b3:product:layercake}) con $f = \eu^{-
\norm x^2}$, cuyos conjuntos de niveles son bolas:$\{f > t\} =
B\bigl(0, \sqrt{-\ln t}\bigr)$para$0 < t < 1$, de \omterm{def:b3:measure:measure}{medida}
$v_d(-\ln t)^{d/2}$ (dilatación por $\rho$ escalas $\lambda_d$ por
$\rho^d$: \cref{thm:b3:product:linearchange}), entonces
\[
I = \int_0^1 v_d\,(-\ln t)^{d/2}\,\dd t
\overset{t = \eu^{-s}}{=}
v_d\int_0^\infty s^{d/2}\eu^{-s}\,\dd s
= v_d\,\Gamma\Bigl(\frac d2 + 1\Bigr).
\]
Igualar. (Los valores: $\Gamma(\frac32) = \frac{\sqrt\pi}2$,
$\Gamma(2) = 1$, etc.) Nota $v_d \to 0$ como $d \to \infty$ ---
el problema del fin de semana cuantifica qué tan rápido, vía Stirling.
\end{proof}

\section{Problema: la fórmula de Stirling}

\begin{problem}[{Problema de fin de semana --- $n! \sim
\sqrt{2\pi n}\,(n/\eu)^n$, por convergencia dominada}]
\label{pb:b3:product:1}
La fórmula de Stirling gobierna todos los recuentos asintóticos de este libro.
--- volúmenes de bolas, coeficientes binomiales, el límite central
la forma local del teorema. Lo demostramos a partir de la integral $\Gamma$.
(\cref{ex:b3:lebesgue:gamma}) con el \emph{método de Laplace},
en su forma más limpia de convergencia dominada, luego recopile
dividendos.\index{La fórmula de Stirling.}

\medskip
\noindent\textbf{Parte I --- The formula.} Para $t > 0$,
$\Gamma(t + 1) = \int_0^\infty x^{t}\eu^{-x}\dd x$.
\begin{enumerate}
  \item Sustituya $x = t + \sqrt t\,u$ y muestre
        \[
        \frac{\Gamma(t+1)}{t^{t}\eu^{-t}\sqrt t}
        = \int_{-\sqrt t}^{+\infty}
        \exp\Bigl(t\ln\Bigl(1 + \frac u{\sqrt
        t}\Bigr) - \sqrt t\,u\Bigr)\,\dd u
        \;=\;\int_\R g_t(u)\,\dd u,
        \]
        donde $g_t(u) = \exp\bigl(t\ln(1 + u/\sqrt t) -
        \sqrt t\,u\bigr)\mathbf 1_{u > -\sqrt t}$.
  \item Muestra el límite puntual: por cada $u$ fijo, $g_t(u)
        \to \eu^{-u^2/2}$como$t \to +\infty$ \emph{(expand
        $\ln(1 + h)$ to second order)}.
  \item Dominación. Sea $\varphi(h) = \ln(1 + h) - h$, de modo que
        $g_t(u) = \exp\bigl(t\,\varphi(u/\sqrt t)\bigr)$ para
        $u > -\sqrt t$. Demostrar los dos límites
        \[
        \varphi(h) \leq -\frac{h^2}4 \quad (-1 < h \leq 1),
        \qquad
        \varphi(h) \leq -c\,h \quad (h \geq 1),\ \ c = 1 -
        \ln 2 > 0
        \]
        \emph{(study $\varphi(h) + \frac{h^2}4$ and $\varphi(h)
        + ch$: compute the derivatives and check the sign on
        each range)}. Deducir, para $t \geq 1$:
        \[
        g_t(u) \leq \eu^{-u^2/4}\ \ (\abs u \leq \sqrt t),
        \qquad
        g_t(u) \leq \eu^{-cu}\ \ (u \geq \sqrt t),
        \]
        de modo que $g_t(u) \leq \eu^{-u^2/4} +
        \eu^{-cu}\,\mathbf 1_{u > 0}$: un dominador \omterm{def:b3:lebesgue:l1}{integrable}
        independiente de $t \geq 1$.
  \item Concluimos con el teorema de convergencia dominada y el
        Integral gaussiana (\cref{ex:b3:product:polar}):
        \[
        \Gamma(t + 1) \;\sim\;
        \sqrt{2\pi t}\;\Bigl(\frac t\eu\Bigr)^{t}
        \qquad (t \to +\infty),
        \]
        y en particular $n! \sim \sqrt{2\pi
        n}\,(n/\eu)^n$.
\end{enumerate}

\medskip
\noindent\textbf{Parte II --- Dividends.}
\begin{enumerate}[resume]
  \item (Wallis) De fórmulas de tipo \cref{exo:b3:product:8}, $W_{2n} =
        \frac\pi2\binom{2n}n4^{-n}$: derivar
        $\binom{2n}{n} \sim \frac{4^n}{\sqrt{\pi n}}$ de
        Stirling y compárelo con la recursividad $W_{n} =
        \frac{n-1}nW_{n-2}$.
  \item (Colapso del volumen de bolas) Mostrar
        \[
        v_d = \frac{\pi^{d/2}}{\Gamma(\frac d2 + 1)}
        \;\sim\; \frac{1}{\sqrt{\pi d}}
        \Bigl(\frac{2\pi\eu}{d}\Bigr)^{d/2},
        \]
        entonces $v_d \to 0$ más rápido que cualquier secuencia geométrica;
        encuentre la dimensión que maximiza $v_d$ (numéricamente: $d =
        5$).
  \item (Concentración del binomio --- un adelanto de
        \cref{ch:b3:clt}) Usando Stirling, muestra el local
        estimación, para $k = n/2 + s\sqrt n/2$ con $s$ fijo
        y $n$ incluso:
        \[
        2^{-n}\binom{n}{k} \;\sim\;
        \sqrt{\frac{2}{\pi n}}\;\eu^{-s^2/2},
        \]
        el perfil gaussiano discreto: de Moivre--Laplace en
        embrión.
  \item ¿Dónde se utilizó exactamente la prueba de la Parte I: (i) MCT o
        DCT; (ii) la integral gaussiana; (iii) la invariancia
        propiedades de \omterm{def:b3:measure:lebesgueouter}{medida de lebesgue}? Una frase cada uno.
\end{enumerate}

\medskip
\noindent\textbf{Parte III --- The error term: Stirling with
bars.} Establezca $d_n = \ln n! - \bigl(n + \tfrac12\bigr)\ln n + n -
\ln\sqrt{2\pi}$, de modo que la Parte I diga$d_n \to 0$.
\begin{enumerate}[resume]
  \item Mostrar $d_n - d_{n+1} = \bigl(n +
        \tfrac12\bigr)\ln\bigl(1 + \tfrac1n\bigr) - 1$.
  \item Con $t = \frac1{2n+1}$, verifique $\frac{n+1}n =
        \frac{1+t}{1-t}$ y expanda:
        \[
        d_n - d_{n+1} = \frac{t^2}3 + \frac{t^4}5 +
        \frac{t^6}7 + \cdots,
        \]
        y deducir los límites de dos lados
        \[
        \frac1{3(2n+1)^2} \;<\; d_n - d_{n+1} \;<\;
        \frac1{12n} - \frac1{12(n+1)} .
        \]
  Telescopio \item (usando $d_m \to 0$) y compruebe la agradable
        \omterm{def:b3:galois:algebraic}{algebraico} identidad $\frac1{3(2m+1)^2} >
        \frac1{12m+1} - \frac1{12(m+1)+1}$para$m \geq 1$, a
        obtener el bracketing clásico
        \[
        \sqrt{2\pi n}\Bigl(\frac n\eu\Bigr)^n
        \eu^{1/(12n+1)} \;<\; n! \;<\;
        \sqrt{2\pi n}\Bigl(\frac n\eu\Bigr)^n\eu^{1/(12n)} .
        \]
  \item Dos consecuencias: (a) el error relativo de Stirling
        la fórmula es $< 10^{-6}$ tan pronto como $n \geq 83\,334$;
        (b) estime $100!$ a cuatro dígitos significativos a mano
        del corchetes ($100! \approx 9.3326\cdot
        10^{157}$), y maravíllese brevemente ante la precisión de un
        Fórmula asintótica en un $n$ muy finito.
\end{enumerate}

\medskip
\noindent\textbf{Parte IV --- The Wallis route: Stirling
without the Gaussian.} Históricamente la constante $\sqrt{2\pi}$
vino de Wallis, no de Gauss; esta parte vuelve a probar a Stirling
independientemente de las Partes I--II, y por lo tanto vuelve a probar la
Integral gaussiana. Deje $W_n = \int_0^{\pi/2}\sin^n\theta\,
\dd\theta$.
\begin{enumerate}[resume]
  \item Establecer $W_n = \frac{n-1}nW_{n-2}$ (integrar por
        partes), las formas cerradas
        \[
        W_{2n} = \frac\pi2\binom{2n}n4^{-n}, \qquad
        W_{2n+1} = \frac{4^n}{(2n+1)\binom{2n}n},
        \]
        y la identidad $W_nW_{n-1} = \frac\pi{2n}$.
  \item De la monotonía de $(W_n)$ se deduce
        $W_{2n}/W_{2n+1} \to 1$, entonces
        \[
        W_{2n} \sim \frac12\sqrt{\frac\pi n}
        \qquad\text{and}\qquad
        \binom{2n}n4^{-n}\sqrt n \longrightarrow
        \frac1{\sqrt\pi} :
        \]
        Teorema de Wallis, obtenido sin Stirling.
  \item Show, por el alcance telescópico de la Parte III únicamente (sin valor
        de la constante necesaria), que $e_n = \ln n! - (n +
        \frac12)\ln n + n$converge a algún límite$\ell$;
        equivalentemente $n! \sim K\,n^{n+1/2}\eu^{-n}$ con $K =
        \eu^\ell > 0$ aún no identificado.
  \item Inserte este asintótico en $\binom{2n}n4^{-n}\sqrt n$
        e identificar, utilizando la pregunta 14, la única posible
        valor: $K = \sqrt{2\pi}$. Armar la lógica: Piezas
        III--IV juntos dan una segunda prueba \omterm{def:b3:complete:complete}{completo} de
        Stirling --- y por lo tanto, ejecutar la sustitución de la Parte I
        al revés, una evaluación independiente de
        $\int_\R\eu^{-u^2/2}\dd u = \sqrt{2\pi}$. Dos pilares,
        cualquiera de los cuales apoya al otro.
\end{enumerate}

\begin{enumerate}[resume]
