\chapter{La transformada de Fourier}\label{ch:b3:fouriertransform}

Las series de Fourier descomponen señales periódicas en discretas
armónicos; el Fourier \emph{transformar} hace lo mismo para
señales en toda la línea, con un continuo de frecuencias. eso
convierte la diferenciación en multiplicación, la convolución en
productos y gaussianos en gaussianos --- las razones por las que
Resuelve ecuaciones diferenciales, impulsa el procesamiento de señales y
demostrará el teorema del límite central en \cref{ch:b3:clt}. esto
El capítulo desarrolla la teoría $L^1$ (Riemann--Lebesgue,
inversión, inyectividad), el \omterm{def:b3:fouriertransform:schwartz}{clase de schwartz} donde el
transform es una biyección \omterm{prop:b3:galois:perfect}{perfecto}, y la teoría $L^2$
(Plancherel: la transformada es, hasta una constante, una unidad
operador), con dos aplicaciones destacadas: la ecuación del calor,
resuelto de principio a fin el problema del fin de semana, y el Poisson
fórmula de suma. Convención:
\[
\hat f(\xi) = \int_\R f(x)\,\eu^{-\iu\xi x}\,\dd x .
\]

\section{La transformación en \texorpdfstring{$L^1$}{L1}}

\begin{proposition}\label{prop:b3:fouriertransform:basic}
Para $f \in L^1(\R)$: $\hat f$ está bien definido, acotado
($\norm{\hat f}_\infty \leq \norm f_1$), \omterm{def:b3:topology:continuity}{continuo} y:
\begin{enumerate}
  \item$\widehat{\tau_af}(\xi) = \eu^{-\iu a\xi}\hat f(\xi)$
        y $\widehat{\eu^{\iu ax}f}(\xi) = \hat f(\xi - a)$;
  \item $\widehat{f(\cdot/\lambda)}(\xi) = \lambda\hat
        f(\lambda\xi)$para$\lambda > 0$;
  \item si $xf \in L^1$, entonces $\hat f$ es $\mathcal C^1$ con
        $(\hat f)'(\xi) = \widehat{(-\iu x)f}(\xi)$;
  \item si $f \in \mathcal C^1$ con $f' \in L^1$ (y $f \to
        0$en$\pm\infty$, automático aquí), entonces
        $\widehat{f'}(\xi) = \iu\xi\hat f(\xi)$;
  \item $\widehat{f * g} = \hat f\,\hat g$ para $f, g \in L^1$.
\end{enumerate}
\end{proposition}

\begin{proof}
Limitación: $\abs{\hat f} \leq \int\abs f$. \omterm{def:b3:topology:continuity}{Continuidad}: DCT
con dominador $\abs f$ (\cref{thm:b3:lebesgue:paramcont}).
(1), (2): sustituciones
(\cref{thm:b3:product:linearchange}). (3): diferenciación
bajo la integral, dominador $\abs{xf}$
(\cref{thm:b3:lebesgue:paramdiff}). (4): primero, $f(x) = f(0)
+ \int_0^xf'$tiene un límite en$\pm\infty$($f' \in L^1$), que
debe ser $0$ ($f \in L^1$); luego integrar por partes en $[-A,
A]$y dejar$A \to \infty$. (5): Fubini, legítimo desde
$(x,y)\mapsto f(x - y)g(y)\eu^{-\iu\xi x}$ es absolutamente
\omterm{def:b3:lebesgue:l1}{integrable} (\cref{thm:b3:product:convolution}):
\[
\widehat{f*g}(\xi) = \iint f(x - y)g(y)\eu^{-\iu\xi(x - y)}
\eu^{-\iu\xi y}\dd x\,\dd y = \hat f(\xi)\,\hat g(\xi).
\]
\end{proof}

\begin{example}\label{ex:b3:fouriertransform:gaussian}
El gaussiano: para $a > 0$,\index{Transformada gaussiana y de Fourier}
\[
\widehat{\eu^{-ax^2}}(\xi) =
\sqrt{\frac\pi a}\;\eu^{-\xi^2/4a} :
\]
por \cref{exo:b3:lebesgue:7} (la ODA $F' = -\frac\xi{2}F$
truco, reescalado), o por (3): $g = \widehat{\eu^{-ax^2}}$
satisface $g'(\xi) = -\frac{\xi}{2a}g(\xi)$ (integrar por
piezas), $g(0) = \sqrt{\pi/a}$. Los gaussianos son puntos fijos de
la transformación hasta la escala --- la razón profunda por la que gobiernan el
teorema del límite central.
\end{example}

\begin{theorem}[Riemann--Lebesgue]
\label{thm:b3:fouriertransform:riemannlebesgue}
Para $f \in L^1(\R)$: $\hat f(\xi) \to 0$ como $\abs\xi \to
\infty$. Por lo tanto,$\widehat{\phantom f} \colon L^1 \to \mathcal
C_0(\R)$ (las funciones \omterm{def:b3:topology:continuity}{continuo} desaparecen en
infinito).\index{Lema de Riemann-Lebesgue}
\end{theorem}

\begin{proof}
Para un indicador de un intervalo, $\widehat{\mathbf 1_{\intcc
ab}}(\xi) = \frac{\eu^{-\iu a\xi} - \eu^{-\iu b\xi}}{\iu\xi}
\to 0$; por lo tanto, para funciones escalonadas. Las funciones escalonadas son densas en
$L^1$ (\cref{thm:b3:lp:density}(1) más aproximación de
\omterm{def:b3:measure:measure}{medida finita} conjuntos por uniones finitas de intervalos,
\cref{exo:b3:measure:7}), y la transformación es
$\norm\cdot_\infty$-$\norm\cdot_1$ \omterm{def:b3:topology:continuity}{continuo}: para $\norm{f -
s}_1 < \varepsilon$,$\limsup_{\abs\xi\to\infty}\abs{\hat
f(\xi)} \leq \varepsilon$.
\end{proof}

\section{Inversión e inyectividad}

\begin{lemma}[Fórmula de multiplicación]
\label{lem:b3:fouriertransform:multiplication}
Para $f, g \in L^1(\R)$: $\displaystyle\int \hat f\,g =
\int f\,\hat g$.
\end{lemma}

\begin{proof}
Ambos lados son iguales a $\iint f(x)g(\xi)\eu^{-\iu x\xi}\dd
x\,\dd\xi$ (Tonelli--Fubini: la integral doble de la
El valor absoluto es $\norm f_1\norm g_1$).
\end{proof}

\begin{theorem}[inversión]\label{thm:b3:fouriertransform:inversion}
Deja que $f \in L^1(\R)$.\index{Teorema de inversión de Fourier}
\begin{enumerate}
  \item (sumabilidad gaussiana) Por cada $x$,
        \[
        (f * g_\varepsilon)(x) =
        \frac1{2\pi}\int_\R \hat f(\xi)\,
        \eu^{-\varepsilon\xi^2}\,\eu^{\iu x\xi}\,\dd\xi,
        \qquad\text{where } g_\varepsilon(y) =
        \frac{1}{2\sqrt{\pi\varepsilon}}\,
        \eu^{-y^2/4\varepsilon},
        \]
        y $f * g_\varepsilon \to f$ en $L^1$ como $\varepsilon
        \to 0$.
  \item Si además $\hat f \in L^1$, entonces para casi todos
        $x$
        \[
        f(x) = \frac{1}{2\pi}\int_\R \hat
        f(\xi)\,\eu^{\iu x\xi}\,\dd\xi ,
        \]
        y $f$ tiene un representante \omterm{def:b3:topology:continuity}{continuo}.
  \item (Inyectividad) Si $\hat f = 0$ entonces $f = 0$ a.e.
\end{enumerate}
\end{theorem}

\begin{proof}
(1) Reparar $x$ y aplicar
\cref{lem:b3:fouriertransform:multiplication} a $f$ y
$g(\xi) = \frac1{2\pi}\eu^{-\varepsilon\xi^2}\eu^{\iu x\xi}$:
por \cref{ex:b3:fouriertransform:gaussian} (con la modulación
regla),
\[
\hat g(y) = \frac1{2\pi}\sqrt{\frac\pi\varepsilon}\,
\eu^{-(y - x)^2/4\varepsilon} = g_\varepsilon(x - y),
\]
entonces $\frac1{2\pi}\int\hat f(\xi)\eu^{-\varepsilon\xi^2}
\eu^{\iu x\xi}\dd\xi = \int f(y)g_\varepsilon(x - y)\dd y =
(f*g_\varepsilon)(x)$. Los$g_\varepsilon$ son un
identidad aproximada: $g_\varepsilon \geq 0$,
$\int g_\varepsilon = 1$ (integral gaussiana), concentrándose en
$0$; se aplica la prueba de \cref{thm:b3:lp:regularization}(2)
palabra por palabra (sólo $\int g_\varepsilon = 1$ y la concentración fueron
usado: para la cola, $\int_{\abs y > \delta}g_\varepsilon \to
0$):$\norm{f * g_\varepsilon - f}_1 \to 0$.

(2) Si $\hat f \in L^1$: el lado derecho de (1) converge, por
DCT (dominator $\abs{\hat f}$), a $\frac1{2\pi}\int\hat
f(\xi)\eu^{\iu x\xi}\dd\xi$para \emph{cada}$x$, y este
La función de límite es \omterm{def:b3:topology:continuity}{continuo} (DCT nuevamente). Por otro lado, $f
* g_\varepsilon \to f$en$L^1$, por lo que a lo largo de una subsecuencia a.e.\
(\cref{thm:b3:lp:complete}): los dos límites coinciden a.e.

(3) $\hat f = 0$ hace que el lado derecho de (1) desaparezca: $f *
g_\varepsilon = 0$para todos los$\varepsilon$y$f *
g_\varepsilon \to f$en$L^1$:$f = 0$ a.e.
\end{proof}

\section{La clase Schwartz}

\begin{definition}\label{def:b3:fouriertransform:schwartz}
El \emph{clase de schwartz}\index{clase de schwartz} $\mathcal
S(\R)$consta de las funciones$\mathcal C^\infty$ $f$ con
$\sup_x\abs{x^m f^{(n)}(x)} < \infty$ para todos $m, n \geq 0$
(todas las derivadas decaen más rápido que cualquier potencia). Ejemplos:
$\eu^{-ax^2}$, $\mathcal C_c^\infty$. Claramente $\mathcal S
\subseteq L^p$por cada$p$(limitado por$C(1 + x^2)^{-1}$), y
$\mathcal S$ es estable bajo derivadas, multiplicación por
polinomios y productos.
\end{definition}

\begin{theorem}\label{thm:b3:fouriertransform:schwartz}
La transformada de Fourier asigna $\mathcal S(\R)$ biyectivamente a
en sí, con la inversa $\check g(x) = \frac1{2\pi}\int
g(\xi)\eu^{\iu x\xi}\dd\xi$.
\end{theorem}

\begin{proof}
Vamos $f \in \mathcal S$. Iterando
\cref{prop:b3:fouriertransform:basic}(3), $(\hat f)^{(n)} =
\widehat{(-\iu x)^nf}$(cada$x^kf \in L^1$); iterando (4)
con $h = (-\iu x)^nf \in \mathcal S$ (todos cuyos
derivados son \omterm{def:b3:lebesgue:l1}{integrable}), $(\iu\xi)^m\hat h =
\widehat{h^{(m)}}$. combinando,
\[
\abs{\xi^m\,(\hat f)^{(n)}(\xi)}
= \bigl|\widehat{\,h^{(m)}}(\xi)\bigr|
\leq \bigl\|\bigl((-\iu x)^nf\bigr)^{(m)}\bigr\|_1 < \infty
\]
uniformemente en $\xi$: $\hat f \in \mathcal S$. Desde $\hat f \in L^1$, inversión
(\cref{thm:b3:fouriertransform:inversion}(2)) mantiene
en todas partes (ambos lados \omterm{def:b3:topology:continuity}{continuo}): $\check{\hat f} = f$, y
simétricamente $\widehat{\check g} = g$ (la transformación de verificación
es $g \mapsto \frac1{2\pi}\hat g(-\cdot)$, conservando nuevamente
$\mathcal S$): biyección.
\end{proof}

\section{Plancherel y \texorpdfstring{$L^2$}{L2}}

\begin{theorem}[Plancherel]\label{thm:b3:fouriertransform:plancherel}
Para $f \in L^1 \cap L^2(\R)$:\index{teorema de plancherel}
\[
\norm{\hat f}_2^2 = 2\pi\,\norm f_2^2 .
\]
En consecuencia, $\widehat{\phantom f}$ se extiende únicamente a un
\omterm{def:b3:topology:continuity}{continuo} aplicación lineal $\mathcal F \colon L^2(\R) \to L^2(\R)$
con $\norm{\mathcal Ff}_2 = \sqrt{2\pi}\norm f_2$; $\mathcal
F$es biyectivo, con$\mathcal F^{-1} =
\frac1{2\pi}\,\mathcal F\circ\sigma$donde$\sigma f =
f(-\cdot)$, y conserva \omterm{def:b3:hilbert:inner}{productos internos} hasta el factor
$2\pi$.
\end{theorem}

\begin{proof}
Deje $f \in L^1\cap L^2$ y $h = f * \tilde f$ con
$\tilde f(x) = \overline{f(-x)}$. Entonces $h \in L^1$
(\cref{thm:b3:product:convolution}), $h$ es \omterm{def:b3:topology:continuity}{continuo} y
acotado (\cref{exo:b3:lp:6}: $f, \tilde f \in L^2$), $h(0) =
\int f\bar f = \norm f_2^2$y$\hat h = \hat
f\,\widehat{\tilde f} = \hat f\,\overline{\hat f} =
\abs{\hat f}^2 \geq 0$(calcule$\widehat{\tilde f} =
\overline{\hat f}$). Aplicar
\cref{thm:b3:fouriertransform:inversion}(1) a $h$ en $x = 0$:
\[
(h * g_\varepsilon)(0) = \frac1{2\pi}\int \hat
h(\xi)\,\eu^{-\varepsilon\xi^2}\dd\xi .
\]
Como $\varepsilon \to 0$: el lado izquierdo tiende a $h(0)$ ($h$
\omterm{def:b3:topology:continuity}{continuo} acotado: $(h*g_\varepsilon)(0) - h(0) = \int(h(-y)
- h(0))g_\varepsilon(y)\dd y \to 0$ dividiendo pequeño/grande
$y$); el lado derecho aumenta a $\frac1{2\pi}\int\hat h$ en
MCT ($\hat h \geq 0$). Por lo tanto $\frac1{2\pi}\int\abs{\hat f}^2 =
\norm f_2^2$, finito o no \emph{a priori} --- y finito,
acreditando tanto la pertenencia como la identidad.

Extensión: $L^1\cap L^2 \supseteq \mathcal C_c$ es denso en
$L^2$ (\cref{thm:b3:lp:density}); la transformación es
$\sqrt{2\pi}$-isométrico allí, por lo que se extiende únicamente a un
isometría hasta la constante $\mathcal F$ en $L^2$
(\cref{thm:b3:complete:extension}). Inversión para $\mathcal S$
(\cref{thm:b3:fouriertransform:schwartz}) transferencias por parte del
mismo \omterm{ex:b3:lebesgue:gamma}{densidad} (ambos lados $L^2$-continuo): $\mathcal
F\bigl(\frac1{2\pi}\mathcal F(\sigma f)\bigr) = f$ en
$\mathcal S$, de ahí $L^2$: biyectividad. \omterm{def:b3:hilbert:inner}{productos internos}:
polarización desde la identidad normativa.
\end{proof}

\begin{theorem}[Suma de Poisson]
\label{thm:b3:fouriertransform:poisson}
Deje $f \in \mathcal S(\R)$ (\omterm{def:b3:topology:continuity}{continuo} $f$ con $\abs{f} +
\abs{\hat f} \leq C(1 + \abs\cdot)^{-2}$
basta). Entonces\index{Fórmula de suma de Poisson}
\[
\sum_{n\in\Z} f(n) \;=\; \sum_{k\in\Z}\hat f(2\pi k) .
\]
\end{theorem}

\begin{proof}
Sea $F(x) = \sum_{n\in\Z}f(x + n)$: la serie converge
normalmente en \omterm{def:b3:topology:compact}{compactos} (desintegración de $f$), por lo que $F$ es \omterm{def:b3:topology:continuity}{continuo}, y
es $1$-periódico. Sus coeficientes de Fourier (período $1$:
$c_k(F) = \int_0^1F(t)\eu^{-2\iu\pi kt}\dd t$):
\[
c_k(F) = \sum_n\int_0^1 f(t + n)\,\eu^{-2\iu\pi kt}\dd t
= \int_\R f(t)\,\eu^{-2\iu\pi kt}\dd t = \hat f(2\pi k)
\]
(la convergencia normal justifica el intercambio; la fase es
$1$-periódico). La serie $\sum_k\abs{c_k(F)}$ converge
(desintegración de $\hat f$), por lo que la serie de Fourier de $F$ converge
normalmente; su suma es una función \omterm{def:b3:topology:continuity}{continuo} con el mismo
coeficientes de Fourier como $F$, por lo tanto es igual a $F$ (inyectividad en
el círculo: la diferencia tiene coeficientes cero, y
\cref{thm:b3:hilbert:fourier} da cero en $L^2$, por lo tanto
en todas partes por \omterm{def:b3:topology:continuity}{continuidad}). Evaluar en $x = 0$.
\end{proof}

\begin{example}[La identidad theta]
\label{ex:b3:fouriertransform:theta}
Aplicando Poisson a $f(x) = \eu^{-\pi tx^2}$ ($t > 0$), cuyo
la transformación es $\hat f(\xi) = t^{-1/2}\eu^{-\xi^2/4\pi t}$
(\cref{ex:b3:fouriertransform:gaussian} con $a = \pi t$):
\[
\sum_{n\in\Z}\eu^{-\pi n^2t}
= \frac1{\sqrt t}\sum_{k\in\Z}\eu^{-\pi k^2/t} :
\]
la ecuación funcional de la función theta de Jacobi, clave para la
ecuación funcional de $\zeta$ de Riemann --- y una espectacular
acelerador numérico: para $t$ pequeño, el lado izquierdo converge
lentamente, el lado derecho increíblemente rápido.
\end{example}

\begin{method}\label{met:b3:fouriertransform:method}
Rangos de trabajo: $L^1$ --- transformación definida puntualmente,
la inversión necesita $\hat f \in L^1$; $\mathcal S$ --- todo
es legal, pruébelo aquí primero; $L^2$ --- transformación definida por
\omterm{ex:b3:lebesgue:gamma}{densidad} (¡no por la integral!), \omterm{prop:b3:galois:perfect}{perfecto} simetría, \omterm{thm:b3:hilbert:parseval}{Parseval}
contabilidad. Para calcular una transformación: reducir a la tabla
(indicador, exponencial, gaussiano) por las reglas de
\cref{prop:b3:fouriertransform:basic}; para demostrar una identidad:
establecerlo en $\mathcal S$ (o $\mathcal C_c^\infty$) y
extender por \omterm{ex:b3:lebesgue:gamma}{densidad} y \omterm{def:b3:topology:continuity}{continuidad}
(\cref{met:b3:lp:toolkit}); para resolver una PDE u ODE lineal con
coeficientes constantes: transformar, dividir, invertir.
\end{method}

\begin{exercise}[$\star\star\star$]\label{exo:b3:fouriertransform:6}

\section{Problema: la ecuación del calor en la recta.}

\begin{problem}[{Problema del fin de semana --- $\partial_tu =
\partial^2_{xx}u$, solucionado de principio a fin}]
\label{pb:b3:fouriertransform:1}
El calor se propaga; la ecuación $\partial_tu = \partial_{xx}^2u$
dice que su \omterm{ex:b3:lebesgue:gamma}{densidad} se difunde a una velocidad dada por el local
curvatura del perfil de temperatura. Resolvemos el Cauchy
problema en $\R$ --- dado $f$, busque $u(t, x)$ para $t > 0$ con
$u(0, \cdot) = f$ --- demuestra lo notable de la solución
propiedades y vea por qué el tiempo no se puede revertir. A lo largo,
$g_t(x) = \frac1{2\sqrt{\pi t}}\eu^{-x^2/4t}$ es el
\emph{núcleo de calor}\index{núcleo de calor} y $u(t, \cdot) = g_t *
f$.

\medskip
\noindent\textbf{Parte I --- Deriving the kernel.} Trabajo
formalmente primero: supongamos que $u(t, \cdot) \in \mathcal S$ resuelve
la ecuación, y sea $\hat u(t, \xi)$ la transformada en
$x$.
\begin{enumerate}
  \item Mostrar (formalmente) $\partial_t\hat u = -\xi^2\hat u$,
        por lo tanto $\hat u(t,\xi) = \eu^{-t\xi^2}\hat f(\xi)$, y
        reconocer $u(t) = g_t * f$
        (\cref{exo:b3:fouriertransform:4}). Esto motiva a la
        \emph{definición} de $u$; todo esta ahora probado
        directamente, para $f \in \mathcal C_b(\R)$ (limitado
        \omterm{def:b3:topology:continuity}{continuo}) o $f \in L^p$.
\end{enumerate}

\medskip
\noindent\textbf{Parte II --- Verification.}
\begin{enumerate}[resume]
  \item Demuestre que para $t > 0$, $u(t, x) = \int
        g_t(x-y)f(y)\dd y$está bien definido para$f \in \mathcal
        C_b$y que$u$es$\mathcal C^\infty$en$(t, x)$.
        en $\intoo0\infty\times\R$ \emph{(diferenciar bajo
        la integral; dominar las derivadas gaussianas localmente
        uniformemente en $(t,x)$)}.
  \item Verificar $\partial_tg_t = \partial^2_{xx}g_t$ por directo
        cálculo y deducir $\partial_tu =
        \partial^2_{xx}u$para$t > 0$.
  \item (Condición inicial) Muestre que para $f \in \mathcal
        C_b$,$u(t, x) \to f(x)$como$t \to 0^+$, uniformemente en
        \omterm{def:b3:topology:compact}{compactos}\emph{(approximate identity: split $\abs y
        \leq \delta$,$\abs y > \delta$)}; para$f \in L^p$
        ($p < \infty$), muestra $\norm{u(t) - f}_p \to 0$.
  \item (Suavizado instantáneo) Conclusión: incluso para aplicaciones simplemente limitadas
        \omterm{def:b3:topology:continuity}{continuo} $f$, la solución es $\mathcal C^\infty$
        por cada $t > 0$ --- el calor se borra instantáneamente
        rugosidad. Calcule $u(t, \cdot)$ explícitamente para $f =
        \mathbf 1_{\intoo0\infty}$ (una función de error) y
        dibuje su perfil para tres valores de $t$.
\end{enumerate}

\medskip
\noindent\textbf{Parte III --- Structural properties.}
\begin{enumerate}[resume]
  \item (Positividad y comparación) Si $f \geq 0$ entonces $u > 0$
        para todos los $t > 0$ (estrictamente, a menos que $f = 0$ a.e.); si
        $f_1 \leq f_2$ y luego $u_1 \leq u_2$. Un punto frío se calienta
        al instante: comentar.
  \item (Conservación) Para $f \in L^1$: $\int u(t, x)\dd x =
        \int f$para todos$t$ \emph{(Tonelli)} --- el calor total es
        conservado.
  \item (Disipación) Para $f \in L^1\cap L^2$, mostrar vía
        Plancherel que $t \mapsto \norm{u(t)}_2$ es
        no creciente, estrictamente a menos que sea $f = 0$, y calcule
        su límite es $t \to \infty$. Mostrar además
        $\norm{u(t)}_\infty \leq \frac{\norm
        f_1}{2\sqrt{\pi t}} \to 0$: el calor se propaga y muere.
  \item (Singularidad, clase $L^2$) Sea $u$ una solución con
        $u(t) \in L^2$ para todos los $t$, $u \in \mathcal
        C^1(\intoo0\infty, L^2)$ en el sentido natural y
        $u(t) \to f$ en $L^2$ como $t\to0$; admitiendo que el
        transform lo convierte a $\partial_t\hat u =
        -\xi^2\hat u$puntualmente ae\ en$\xi$para ae$t$

        $\hat u(t,\xi) = \eu^{-t\xi^2}\hat f(\xi)$, por lo tanto

\end{enumerate}

\noindent\textbf{Parte IV --- The arrow of time.}

        PW_\Omega$,
