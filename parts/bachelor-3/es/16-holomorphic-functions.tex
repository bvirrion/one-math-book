\chapter{Funciones holomorfas}\label{ch:b3:holomorphic}

La diferenciabilidad compleja parece una pequeña variación de la
Teoría real: un límite, un cociente. Es en cambio un
universo diferente. Una función diferenciable \emph{una vez} en un
el subconjunto abierto de $\C$ es automáticamente diferenciable infinitamente,
analítico, determinado en un dominio completo por sus valores cerca de un
punto único y limitado por principios globales rígidos
(Liouville, módulo máximo). Todo esto fluye de uno
milagro, el teorema de Cauchy: la integral de un \omterm{def:b3:holomorphic:holo}{holomorfo
función} a lo largo de un \omterm{def:b3:topology:pathconnected}{camino} cerrado en un dominio en forma de estrella desaparece.
Este capítulo prueba el milagro (el argumento de Goursat, sin
regularidad asumida más allá de la diferenciabilidad), cosecha su
consecuencias clásicas y establece un teorema que este libro tiene
utilizado a crédito desde \cref{ch:b3:galois}: cada no constante
polinomio complejo tiene raíz.

En todo momento, $\Omega \subseteq \C$ está abierto y $D(a, r)$
denota el disco abierto.

\section{Diferenciabilidad compleja}

\begin{definition}\label{def:b3:holomorphic:holo}
$f \colon \Omega \to \C$ es \emph{holomorfo}\index{holomorfo
función} en $\Omega$ si por cada $z_0 \in \Omega$
\[
f'(z_0) = \lim_{h\to0}\frac{f(z_0 + h) - f(z_0)}{h}
\]
existe ($h \in \C^*$). Sumas, productos, cocientes (que no desaparecen
denominadores), las composiciones de funciones holomorfas son
holomorfa, con las fórmulas habituales (las pruebas del año 1-2 son
textualmente: sólo utilizan operaciones de campo y límites).
$\mathcal H(\Omega)$ denota el conjunto de funciones holomorfas.
en $\Omega$.
\end{definition}

\begin{proposition}[Cauchy--Riemann]
\label{prop:b3:holomorphic:cauchyriemann}
Escribe $f(x + \iu y) = P(x,y) + \iu Q(x,y)$. Entonces $f$ es
\omterm{def:b3:holomorphic:holo}{holomorfo} en $z_0$ si $f$ es $\R$-diferenciable en $z_0$
(como una aplicación de dos variables reales) y\index{Cauchy--Riemann
ecuaciones}
\[
\frac{\partial P}{\partial x} = \frac{\partial Q}{\partial y},
\qquad
\frac{\partial P}{\partial y} = -\frac{\partial Q}{\partial x}
\qquad \text{at } z_0 ;
\]
de manera equivalente, el diferencial real es la multiplicación por el
número complejo $f'(z_0)$.
\end{proposition}

\begin{proof}
$\C$-diferenciabilidad dice $f(z_0 + h) = f(z_0) + ch +
o(\abs h)$con$c = f'(z_0)$: un diferencial lineal$\R$
que es multiplicación por $c = a + \iu b$, es decir\ tiene matriz
$\bigl(\begin{smallmatrix} a & -b\\ b & a
\end{smallmatrix}\bigr)$en la base$(1, \iu)$ --- exactamente
las relaciones mostradas para los parciales. Por el contrario, tal
El diferencial es $\C$-lineal y las definiciones $o(\abs h)$
partido.
\end{proof}

Polinomios en $z$, funciones racionales fuera de sus polos y

\section{Integrales de contorno}

\begin{definition}\label{def:b3:holomorphic:contour}
Un \emph{\omterm{def:b3:topology:pathconnected}{camino}} es una aplicación $\mathcal C^1$ por partes $\gamma \colon
\intcc ab \to \C$; es \emph{cerrado} si$\gamma(a) =
\gamma(b)$. Para \omterm{def:b3:topology:continuity}{continuo}$f$ en la imagen de
$\gamma$:\index{integral de contorno}
\[
\int_\gamma f(z)\,\dd z =
\int_a^b f(\gamma(t))\,\gamma'(t)\,\dd t,
\qquad
\Bigl|\int_\gamma f\,\dd z\Bigr| \leq
\sup_{\gamma}\abs f\cdot\operatorname{length}(\gamma)
\]
(el \emph{Desigualdad de AA}; longitud $= \int_a^b\abs{\gamma'}$).
La integral es invariante al aumentar $\mathcal C^1$.
reparametrización y cambios de signo bajo inversión de orientación.
\end{definition}

\begin{proposition}[Primitivos]\label{prop:b3:holomorphic:primitive}
Para \omterm{def:b3:topology:continuity}{continuo} $f$ en $\Omega$, lo siguiente es equivalente:
(i) $f$ tiene un \emph{primitivo} $F \in \mathcal H(\Omega)$ ($F'
= f$); (ii)$\int_\gamma f\,\dd z = 0$ por cada \omterm{def:b3:topology:pathconnected}{camino} cerrado
$\gamma$ en $\Omega$. En ese caso $\int_\gamma f\,\dd z =
F(\gamma(b)) - F(\gamma(a))$ por cada \omterm{def:b3:topology:pathconnected}{camino}.
\end{proposition}

\begin{proof}
(i)$\Rightarrow$: $\frac{\dd}{\dd t}F(\gamma(t)) =
F'(\gamma(t))\gamma'(t)$ (regla de la cadena, válida por partes), por lo que el
telescopios integrales hasta la diferencia del punto final; cerrado \omterm{def:b3:topology:pathconnected}{caminos}
dar $0$. (ii)$\Rightarrow$(i): corrija $z_*$ en un \omterm{def:b3:topology:components}{conectado
componente}, defina $F(z) = \int_{\gamma_z}f\,\dd z$ junto con cualquier
\omterm{def:b3:topology:pathconnected}{camino} de $z_*$ a $z$ (bien definido: dos \omterm{def:b3:topology:pathconnected}{caminos} se diferencian por un
cerrado \omterm{def:b3:topology:pathconnected}{camino}); para $h$ pequeño, tomando el segmento de $z$ a
$z + h$,
\[
\frac{F(z + h) - F(z)}{h} - f(z)
= \frac1h\int_{[z, z+h]}\bigl(f(w) - f(z)\bigr)\dd w
\longrightarrow 0
\]
por la desigualdad ML y \omterm{def:b3:topology:continuity}{continuidad} de $f$ en $z$.
\end{proof}

\begin{definition}[Número de bobinado]\label{def:b3:holomorphic:index}
Para un \omterm{def:b3:topology:pathconnected}{camino} $\gamma$ y $z \notin
\operatorname{im}\gamma$ cerrado, el \emph{índice}\index{bobinado
numero}\index{índice de un camino} es
\[
\operatorname{Ind}_\gamma(z) = \frac1{2\iu\pi}
\int_\gamma\frac{\dd w}{w - z} .
\]
Es un \emph{entero}: configurando $\varphi(t) =
\int_a^t\frac{\gamma'(s)}{\gamma(s) - z}\dd s$, la función
$(\gamma(t) - z)\eu^{-\varphi(t)}$ tiene derivada cero
(por partes), por lo tanto es constante; en $t = b$, $\eu^{\varphi(b)} =
\frac{\gamma(b) - z}{\gamma(a) - z} = 1$, entonces$\varphi(b) \in
2\iu\pi\Z$. En función de$z$, el índice es \omterm{def:b3:topology:continuity}{continuo} en
$\C\setminus\operatorname{im}\gamma$ (convergencia dominada),
por lo tanto, constante en cada \omterm{def:b3:topology:components}{componente conectado} y $0$ en el
componente ilimitado (ML: la integral tiende a $0$ como $z \to
\infty$). Para el círculo$\gamma(t) = a + r\eu^{\iu t}$,$t
\in \intcc0{2\pi}$:$\operatorname{Ind}_\gamma(z) = 1$para$z
\in D(a,r)$(calcular en$z = a$:$\frac1{2\iu\pi}
\int_0^{2\pi}\frac{r\iu\eu^{\iu t}}{r\eu^{\iu t}}\dd t = 1$;
la constancia hace el resto).
\end{definition}

\section{teorema de cauchy}

\begin{theorem}[goursat]\label{thm:b3:holomorphic:goursat}
Dejemos que $f \in \mathcal H(\Omega)$ y $T \subseteq \Omega$ a
triángulo sólido cerrado. Entonces $\int_{\partial T}f\,\dd z = 0$
(el límite se ejecuta una vez, cualquier orientación).\index{teorema de goursat}
\end{theorem}

\begin{proof}
Vamos $I(T) = \int_{\partial T}f\,\dd z$. Uniendo los puntos medios
de los lados divide $T$ en cuatro triángulos de tamaño medio $T^{(1)},
\dots, T^{(4)}$, y los bordes internos se cancelan en pares:$I(T) =
\sum_iI(T^{(i)})$. Elija$T_1$ entre ellos con
$\abs{I(T_1)} \geq \frac14\abs{I(T)}$, e iterar: un anidado
secuencia $T \supseteq T_1 \supseteq T_2 \supseteq\cdots$ con
\[
\abs{I(T_n)} \geq 4^{-n}\abs{I(T)},
\qquad
\operatorname{diam}T_n = 2^{-n}\operatorname{diam}T,
\quad
\operatorname{length}(\partial T_n) =
2^{-n}\operatorname{length}(\partial T).
\]
La intersección $\bigcap T_n$ es un solo punto $z_0$
(\omterm{def:b3:topology:compact}{compactos} anidado con diámetros de fuga,
\cref{thm:b3:topology:compactprops}(3)). Diferenciabilidad en
$z_0$: dado $\varepsilon$, para $n$ grande, en $T_n$,
\[
f(z) = f(z_0) + f'(z_0)(z - z_0) + R(z),
\qquad \abs{R(z)} \leq \varepsilon\abs{z - z_0} \leq
\varepsilon\operatorname{diam}T_n .
\]
La parte afín tiene una primitiva: su integral en la cerrada.
$\partial T_n$ desaparece (\cref{prop:b3:holomorphic:primitive}),
saliendo
\[
\abs{I(T_n)} = \Bigl|\int_{\partial T_n}R\Bigr|
\leq \varepsilon\operatorname{diam}(T_n)\,
\operatorname{length}(\partial T_n)
= \varepsilon\,4^{-n}\operatorname{diam}(T)
\operatorname{length}(\partial T) .
\]
Comparando con $\abs{I(T_n)} \geq 4^{-n}\abs{I(T)}$:
$\abs{I(T)} \leq \varepsilon\cdot\text{const}$ por cada
$\varepsilon$: $I(T) = 0$.
\end{proof}

\begin{theorem}[Teorema de Cauchy, versión en forma de estrella]
\label{thm:b3:holomorphic:cauchy}
Sea $\Omega$ \emph{en forma de estrella} sobre $c$ (cada segmento
$[c, z]$, $z \in \Omega$, se encuentra en $\Omega$) --- p.ej.\ convexo.
Cada $f \in \mathcal H(\Omega)$ tiene una primitiva en $\Omega$;
en consecuencia $\int_\gamma f\,\dd z = 0$ para \emph{cada}
cerrado \omterm{def:b3:topology:pathconnected}{camino} $\gamma$ en $\Omega$.\index{teorema de cauchy}
\end{theorem}

\begin{proof}
Defina $F(z) = \int_{[c,z]}f\,\dd w$. Para $z, z + h \in
\Omega$con$[z, z+h] \subseteq \Omega$(verdadero para$h$ pequeño),
el triángulo con vértices $c, z, z+h$ se encuentra en $\Omega$
(forma de estrella: cada una de sus puntas se encuentra en un segmento $[c,
w]$con$w \in [z, z+h] \subseteq \Omega$): Goursat da
\[
F(z + h) - F(z) = \int_{[z, z+h]}f\,\dd w,
\]
y el cálculo del cociente de diferencias de
\cref{prop:b3:holomorphic:primitive} produce $F' = f$. el
La desaparición de las integrales \omterm{def:b3:topology:pathconnected}{camino cerrado} se deriva del mismo
propuesta.
\end{proof}

\begin{theorem}[Fórmula integral de Cauchy]
\label{thm:b3:holomorphic:formula}
Deje que $f \in \mathcal H(\Omega)$, $\bar D(a, r) \subseteq
\Omega$y$C_r$ejecuten el círculo$\partial D(a,r)$ una vez.
en sentido antihorario. Luego por cada $z \in D(a, r)$:\index{cauchy
fórmula integral}
\[
f(z) = \frac1{2\iu\pi}\int_{C_r}\frac{f(w)}{w - z}\,\dd w .
\]
\end{theorem}

\begin{proof}
Repare $z$ y defina en $\Omega$
\[
g(w) = \begin{cases}
\dfrac{f(w) - f(z)}{w - z} & w \neq z,\\[4pt]
f'(z) & w = z :
\end{cases}
\]
$g$ es \omterm{def:b3:topology:continuity}{continuo} en $\Omega$ y \omterm{def:b3:holomorphic:holo}{holomorfo} en $z$. goursat
se cumple para $g$ en cada triángulo $T \subseteq \Omega'$, donde
$\Omega'$ es un disco ligeramente más grande que $\bar D(a,r)$ en el interior
$\Omega$: si $z \notin T$, directamente; si $z \in T$, dividir $T$
en triángulos pequeños que tienen $z$ como vértice más triángulos
evitando $z$; en un triángulo con vértice $z$ el límite ML da
$\abs{\int_{\partial T'}g} \leq \sup_{T'}\abs
g\cdot\operatorname{length} \to 0$ a medida que el triángulo se encoge,
y las piezas restantes desaparecen por Goursat --- por lo tanto
$\int_{\partial T}g = 0$ en todos los casos. la prueba de
\cref{thm:b3:holomorphic:cauchy} usó solo este triángulo
propiedad: $g$ tiene una primitiva en el $\Omega'$ convexo, por lo que
$\int_{C_r}g = 0$, es decir
\[
\frac1{2\iu\pi}\int_{C_r}\frac{f(w)}{w - z}\dd w
= f(z)\,\frac{1}{2\iu\pi}\int_{C_r}\frac{\dd w}{w - z}
= f(z)\operatorname{Ind}_{C_r}(z) = f(z) .
\]
\end{proof}

Sean $f \in \mathcal H(\Omega)$ y $D(a, R) \subseteq \Omega$.

r}\abs{f(w)}}{r^{n}} .

La serie de potencias es infinitamente diferenciable con $c_n =

Depende de $r$. Las estimaciones: acotado el coeficiente integral

\begin{corollary}[Liouville; d'Alembert--Gauss]
\label{cor:b3:holomorphic:liouville}
Una función entera acotada es constante. En consecuencia cada
polinomio no constante sobre $\C$ tiene una raíz: \emph{$\C$ is
\omterm{def:b3:galois:closure}{algebraicamente cerrado}}.\index{teorema de liouville}\index{fundamental

\end{corollary}

= \abs{f'(a)} \leq M/r \to 0$:$f' \equiv 0$y$f$ son
constante (en el \omterm{def:b3:topology:connected}{conectado} $\C$: derivada cero implica

\begin{theorem}[Los ceros están aislados; teorema de identidad]
\label{thm:b3:holomorphic:identity}
Sea $\Omega$ \emph{\omterm{def:b3:topology:connected}{conectado}} y $f \in \mathcal
H(\Omega)$,$f \not\equiv 0$. Entonces cada cero$a$de$f$ tiene
finito \emph{orden}: $f(z) = (z - a)^m\,g(z)$ con $g \in
\mathcal H(\Omega)$,$g(a) \neq 0$y los ceros de$f$ tienen
sin punto de acumulación en $\Omega$. En consecuencia, si dos
\omterm{def:b3:holomorphic:holo}{funciones holomorfas} en $\Omega$ acuerdan un conjunto con un
punto de acumulación en $\Omega$, están de acuerdo
en todas partes.\index{teorema de identidad}\index{ceros aislados}
\end{theorem}

\begin{proof}
Sea $Z$ el conjunto de puntos donde todas las derivadas de $f$
desaparecer. $Z$ es cerrado (intersección de conjuntos cerrados) y abierto:
si todo $c_n = 0$ está en $a$, la expansión de la serie de potencias hace que $f
\equiv 0$en un disco sea alrededor de$a$. \omterm{def:b3:topology:connected}{Conectividad}:$Z =
\varnothing$o$Z = \Omega$; este último está excluido por$f
\not\equiv 0$. Entonces, en un$a$ cero, algún coeficiente es distinto de cero:
dejemos que $m$ sea mínimo con $c_m \neq 0$; luego $f(z) = (z -
a)^m\sum_{n\geq m}c_n(z-a)^{n-m}$ en un disco, y la suma
define $g$ \omterm{def:b3:holomorphic:holo}{holomorfo} cerca de $a$ con $g(a) = c_m \neq 0$;
extienda $g = f/(z-a)^m$ fuera de $a$ (\omterm{def:b3:holomorphic:holo}{holomorfo} allí). desde
$g(a) \neq 0$ y $g$ es \omterm{def:b3:topology:continuity}{continuo}, $f$ no tiene otro cero en
a \omterm{def:b3:topology:topology}{entorno} de $a$: los ceros están aislados y un conjunto de
puntos aislados no tiene punto de acumulación en $\Omega$ (un
el punto de acumulación de ceros es un cero --- \omterm{def:b3:topology:continuity}{continuidad} --- y
no estaría aislado). Identidad: aplicar a la diferencia,
cuyo conjunto cero tiene un punto de acumulación, obligándolo a entrar en el
Rama $Z = \Omega$.
\end{proof}

\begin{theorem}[Teorema de convergencia de Weierstrass]
\label{thm:b3:holomorphic:weierstrassconv}
Si $f_n \in \mathcal H(\Omega)$ converge a $f$ uniformemente en
cada subconjunto \omterm{def:b3:topology:compact}{compacto} de $\Omega$, luego $f \in \mathcal
H(\Omega)$y$f_n^{(k)} \to f^{(k)}$ uniformemente en \omterm{def:b3:topology:compact}{compactos},
por cada $k$.\index{Teorema de convergencia de Weierstrass}
\end{theorem}

\begin{proof}
$f$ es \omterm{def:b3:topology:continuity}{continuo}. Para cualquier triángulo cerrado $T \subseteq
\Omega$:$\int_{\partial T}f = \lim\int_{\partial T}f_n = 0$
(convergencia uniforme en el \omterm{def:b3:topology:compact}{compacto} $\partial T$; Goursat para
$f_n$). Por el argumento de \cref{thm:b3:holomorphic:cauchy},
$f$ tiene primitivas locales (los discos son convexos; sólo el triángulo
se utilizó la propiedad), es decir,\ $f = F'$ localmente con $F$
\omterm{def:b3:holomorphic:holo}{holomorfo}; $F$ es analítico
(\cref{thm:b3:holomorphic:analytic}), por lo tanto también lo es $f = F'$:
\omterm{def:b3:holomorphic:holo}{holomorfo}. (Esto es \emph{Morera's
teorema}\index{teorema de morera}: \omterm{def:b3:topology:continuity}{continuo} con desaparición
integrales de triángulo implica \omterm{def:b3:holomorphic:holo}{holomorfo}.) Derivadas: para
$\bar D(a, 2r) \subseteq \Omega$ y $z \in \bar D(a, r)$,
Fórmula de Cauchy para derivadas (diferenciar
\cref{thm:b3:holomorphic:formula} bajo la integral, o uso
la fórmula del coeficiente) da
\[
\abs{f_n'(z) - f'(z)} =
\Bigl|\frac{1}{2\iu\pi}\int_{C_{2r}}\frac{f_n(w) - f(w)}{(w -
z)^2}\,\dd w\Bigr|
\leq \frac{2r\,\sup_{C_{2r}}\abs{f_n - f}}{r^2} \to 0
\]
uniformemente en $\bar D(a,r)$; cubrir un \omterm{def:b3:topology:compact}{compacto} por un número finito
dichos discos e iterar para obtener $k$ superior.
\end{proof}

\end{method}

\section{Problema: el teorema fundamental del álgebra, dos veces}

\begin{problem}[{Problema de fin de semana --- $\C$ es algebraicamente
cerrado: la prueba de d'Alembert, la prueba de Liouville y la
cosecha}]\label{pb:b3:holomorphic:1}
Sea $P(z) = z^n + a_{n-1}z^{n-1} + \dots + a_0$, $n \geq 1$.
Probamos dos veces que $P$ tiene una raíz, luego recopilamos qué álgebra
ha estado esperando desde \cref{ch:b3:galois}.

\medskip
\noindent\textbf{Parte I --- Coercivity and the minimum.}
\begin{enumerate}
  \item Mostrar que $\abs{P(z)} \to +\infty$ como $\abs z \to
        \infty$: precisamente,$\abs{P(z)} \geq \frac12\abs z^n$
        Para $\abs z \geq R_0$ adecuado.
  \item Deduzca que $\abs P$ alcanza un mínimo global en $\C$:
        hay $z_0$ con $\abs{P(z_0)} = \inf_\C\abs P$
        \emph{(compactness of a large closed disc,
        \cref{cor:b3:topology:heineborel})}.
\end{enumerate}

\medskip
\noindent\textbf{Parte II --- d'Alembert's descent.} Supongamos,
por contradicción, $P(z_0) \neq 0$.
\begin{enumerate}[resume]
  \item Expande $Q(h) = P(z_0 + h)/P(z_0)$ como un polinomio en
        $h$: $Q(h) = 1 + c_kh^k + h^{k+1}S(h)$ con $c_k \neq
        0$,$k \geq 1$,$S$ un polinomio.
  \item Elige la dirección de descenso: para pequeños $t > 0$,
        establezca $h = t\,\omega$ donde $\omega^k = -1/c_k$ (por qué
        ¿Existe tal $\omega$? --- \emph{probar} el
        existencia de $k$-ésima raíz de cualquier número complejo a través de
        forma polar, independientemente de que el teorema sea
        demostrado). Mostrar
        \[
        \abs{Q(t\omega)} \leq 1 - t^k + C\,t^{k+1}
        \]
        para $t$ pequeño, con una constante explícita $C$.
  \item Concluir $\abs{Q(t\omega)} < 1$ para pequeños $t$ ---
        contradiciendo la minimalidad de $\abs{P(z_0)}$. Por lo tanto
        $P(z_0) = 0$: \emph{cada complejo no constante
        polinomio tiene una raíz} (d'Alembert--Argand).
\end{enumerate}

\medskip
\noindent\textbf{Parte III --- Liouville's one-liner,
in full.}
\begin{enumerate}[resume]
  \item Escriba cuidadosamente la prueba de
        \cref{cor:b3:holomorphic:liouville}: si $P$ no tiene
        raíz, verifique que $1/P$ esté completo, acotado (cuantificar,
        usando la pregunta 1), por lo tanto constante, y concluimos.
        Compara las dos pruebas: ¿qué ingredientes tiene cada uno?
        utilizar? (La compacidad aparece en ambos --- ¿dónde?)
\end{enumerate}

\medskip

\begin{enumerate}[resume]
  \item Demuestre que cada $P \in \C[X]$ de grado $n$ se divide:
        $P = c\prod_{i}(X - \alpha_i)^{m_i}$ con $\sum m_i =
        n$\emph{(induction, Euclidean division by$(X -
        \alpha)$)}.
  \item Demuestre que los polinomios \omterm{def:b3:rings:divisibility}{irreducible} de $\R[X]$ son

        raíz, y dar una segunda prueba teórica de orden de

        comprobando que están de acuerdo con $X^3 - X - 1$.
  \item Deduzca las deudas que este libro puede pagar ahora: (i) cada

  \item (desigualdad de Landau) Para $P = \sum_{k=0}^na_kX^k$,
        demostrar el valor medio $\frac1{2\pi}\int_0^{2\pi}

        19 que

        \abs c\prod_{i}\max\bigl(1, \abs{\alpha_i}\bigr)
        \;\leq\;
        \Bigl(\sum_{k=0}^{n}\abs{a_k}^2\Bigr)^{1/2} :
        \]
        el producto de las raíces fuera del disco unitario es
        controlado por el tamaño $\ell^2$ de los coeficientes
        (maneje las raíces \emph{on} del círculo aplicando el

  \item (los números de Bernoulli crecen factorialmente) Demuestre que
        El radio de convergencia de $\sum\frac{B_n}{n!}z^n$ es

        \]

  \item (Las raíces se mueven \omterm{def:b3:topology:continuity}{continuamente}) Dejemos que $(P_j)$ sea monic de

        Los conjuntos múltiples raíz de $P_j$ convergen al de $P$.
        Demuestra finalmente que \omterm{def:b3:topology:continuity}{continuidad} es lo mejor que se puede
