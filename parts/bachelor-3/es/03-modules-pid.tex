\section{Módulos, módulos gratuitos}

\begin{definition}\label{def:b3:modules:module}
Un \emph{módulo $A$}\index{módulo} es un grupo abeliano $(M, +)$
con una multiplicación escalar $A \times M \to M$ que satisface la
axiomas del espacio vectorial: $a(x + y) = ax + ay$, $(a + b)x = ax + bx$,
$(ab)x = a(bx)$, $1x = x$. Submódulos, cocientes $M/N$, morfismos
($A$-mapas lineales), sumas directas $\bigoplus_i M_i$ y el
Los teoremas de isomorfismo se definen y demuestran palabra por palabra como para
espacios vectoriales y grupos abelianos; en particular $M/\ker f \cong
\operatorname{im} f$para un morfismo$f$.
\end{definition}

\begin{example}\label{ex:b3:modules:examples}
Los tres casos motivadores.
\begin{enumerate}
  \item $A = K$ un campo: \omterm{def:b3:modules:module}{módulos} son espacios vectoriales.
  \item $A = \Z$: \omterm{def:b3:modules:module}{módulos} son exactamente grupos abelianos ($nx$ es
        forzado a ser $x + \dots + x$), los submódulos son subgrupos.
  \item $A = K[X]$: un \omterm{def:b3:modules:module}{módulo} es un espacio vectorial $K$ $V$ juntos
        con la aplicación lineal $K$ $u\colon x \mapsto X\cdot x$ ---
        por el contrario, cada par $(V, u)$ con $u \in \mathcal L(V)$
        se convierte en un módulo $K[X]$ por $P \cdot x = P(u)(x)$. el
        Los submódulos son precisamente los subespacios estables $u$.
\end{enumerate}
Un \omterm{def:b3:rings:ideal}{ideal} de $A$ es exactamente un submódulo de $A$; a \omterm{def:b3:rings:ideal}{anillo cociente}
$A/I$ es un módulo $A$. A diferencia de los espacios vectoriales, \omterm{def:b3:modules:module}{módulos} puede tener
\emph{\omterm{def:b3:modules:torsion}{torsión}}: en $\Z/6\Z$, el elemento $\bar 3 \ne 0$ es eliminado
por $2 \neq 0$.
\end{example}

\begin{definition}\label{def:b3:modules:free}
$M$ es \emph{finitamente generado}\index{módulo generado finitamente}
si $M = Ax_1 + \dots + Ax_n$ para algunos $x_i$. $M$ es
\emph{libre}\index{módulo gratuito} de rango $n$ si $M \cong A^n$,
es decir,\ si tiene un \emph{base} (una familia generativa que es
$A$-linealmente independiente). Cada $M$ generado finitamente es un
cociente de un \omterm{def:b3:modules:module}{módulo} libre: $(a_1, \dots, a_n) \mapsto \sum a_ix_i$
asigna $A^n$ a $M$.
\end{definition}

\begin{proposition}[Invariancia de rango]\label{prop:b3:modules:rank}
Si $A \neq 0$ y $A^m \cong A^n$, entonces $m = n$.
\end{proposition}

\begin{proof}
Elija un \omterm{def:b3:rings:primemaximal}{ideal máximo} $\mathfrak m$ de $A$ (\cref{thm:b3:rings:krull})
y establezca $k = A/\mathfrak m$, un campo. Un isomorfismo $f \colon A^m
\to A^n$asigna$\mathfrak m A^m$a$\mathfrak m A^n$
(linealidad), por lo tanto induce un isomorfismo de cocientes
\[
A^m/\mathfrak m A^m \;\cong\; A^n/\mathfrak m A^n,
\qquad\text{i.e.}\qquad k^m \cong k^n
\]
como espacios vectoriales $k$ (el cociente $A^m/\mathfrak m A^m$ es
asesinado por $\mathfrak m$, por lo que la acción $A$ influye en $k$;
las imágenes de la base estándar forman una base $k$). Dimensión
La teoría sobre el campo $k$ da $m = n$.
\end{proof}

\begin{theorem}[Submódulos de módulos libres]\label{thm:b3:modules:submodule}
Sea $A$ un \omterm{def:b3:rings:pidufd}{PID} y $M \subseteq A^n$ un submódulo. Entonces $M$ es
\omterm{def:b3:modules:free}{libre} de rango $\leq n$.
\end{theorem}

\begin{proof}
Inducción en $n$. Para $n = 1$: $M$ es un \omterm{def:b3:rings:ideal}{ideal}, por lo que $M = (0)$
(\omterm{def:b3:modules:free}{libre} de rango $0$) o $M = dA \cong A$ ($x \mapsto dx$ es
inyectivo: dominio). Para $n > 1$: sea $\pi \colon A^n \to A$ el
última coordenada. Entonces $\pi(M)$ es \omterm{def:b3:rings:ideal}{ideal}, $(0)$ o $dA$. si
$(0)$: $M \subseteq A^{n-1} \times \{0\}$ y se aplica la inducción.
De lo contrario, elija $x_0 \in M$ con $\pi(x_0) = d$. Cada $x \in M$
escribe de forma única
\[
x = \underbrace{\frac{\pi(x)}{d}}_{\in A}\, x_0 +
\Bigl(x - \tfrac{\pi(x)}d x_0\Bigr),
\qquad x - \tfrac{\pi(x)}d x_0 \in M \cap \ker\pi
\]
($\pi(x) \in dA$, por lo que el coeficiente está en $A$). Así $M = Ax_0
\oplus (M \cap \ker \pi)$: la suma es directa ya que$\pi(ax_0) =
ad = 0$fuerza a$a = 0$. Por inducción$M \cap \ker\pi \subseteq
\ker \pi \cong A^{n-1}$es \omterm{def:b3:modules:free}{libre} de rango$\leq n - 1$; contiguo
$x_0$ (independiente de $\ker\pi$ como se acaba de ver) proporciona una base de
$M$ de cardinalidad $\leq n$.
\end{proof}

\begin{remark}\label{rem:b3:modules:presentation}
En consecuencia, sobre un \omterm{def:b3:rings:pidufd}{PID} cada \omterm{def:b3:modules:free}{módulo generado finitamente} $M$ tiene un
\emph{presentación finita}: una sobreyección $\varphi\colon A^n \to M$
tiene un kernel \omterm{def:b3:modules:free}{libre} con base $c_1, \dots, c_k$ ($k \leq n$) y $M
\cong A^n / \operatorname{im}(C)$donde$C \in M_{n,k}(A)$ es el
matriz cuyas columnas son las $c_j$. Comprender los medios $M$
entendiendo un \emph{matriz sobre $A$} hasta cambio de bases en
origen y destino: el tema de la siguiente sección.
\end{remark}

\section{Forma normal de Smith}

\begin{definition}\label{def:b3:modules:equivalence}
Dos matrices $B, C \in M_{n,k}(A)$ son \emph{equivalente} si $C =
QBP$con$Q \in GL_n(A)$,$P \in GL_k(A)$ (invertible
\emph{más de $A$}: determinante en $A^\times$). equivalente
Las matrices de presentación definen \omterm{def:b3:modules:module}{módulos} $A^n/
\operatorname{im}$isomórficas (cambie las bases en$A^n$y$A^k$).
\end{definition}

\begin{theorem}[Forma normal de Smith]\label{thm:b3:modules:smith}
Sea $A$ un \omterm{def:b3:rings:pidufd}{PID} y un $B \in M_{n,k}(A)$. Entonces $B$ es equivalente a
una matriz diagonal\index{Forma normal de Smith}
\[
\operatorname{diag}(d_1, d_2, \dots, d_r, 0, \dots, 0),
\qquad d_1 \mid d_2 \mid \cdots \mid d_r \neq 0 ,
\]
y el $d_i$ son únicos hasta \omterm{def:b3:rings:divisibility}{asociados}: $d_1 \cdots d_i$ es un
mcd de los $i \times i$ menores de $B$ (en particular que mcd es
una invariante de equivalencia). Los $d_i$ son los \emph{invariante
factores}\index{factores invariantes} de $B$.
\end{theorem}

\begin{proof}
\emph{Existencia.} Si $B = 0$, listo. En caso contrario, considere el conjunto de
\omterm{def:b3:rings:ideal}{ideales} $(b)$ generado por entradas de matrices equivalentes a $B$;
ya que $A$ es \omterm{def:b3:rings:noetherian}{noetheriano}, elija una matriz $B'$ equivalente a $B$
y una entrada $d$ de $B'$ con $(d)$ \emph{máximo} en este conjunto.
Mueva $d$ a la posición $(1,1)$ mediante intercambios de filas y columnas.

\emph{Reclamación: $d$ divide cada entrada de $B'$.} Primero, columna 1: si
$b_{i1}$ no es un múltiplo de $d$, dejemos que $e = \gcd(d, b_{i1}) = ud +
vb_{i1}$(Bézout), entonces$(e) \supsetneq (d)$. El$2 \times 2$
truco de matriz: actuar en las filas $1$ y $i$ por
\[
\begin{pmatrix} u & v \\ -\dfrac{b_{i1}}e & \dfrac de
\end{pmatrix}
\in GL_2(A)
\qquad
\Bigl(\det = \tfrac{ud + v b_{i1}}e = 1\Bigr)
\]
produce una matriz equivalente con la entrada $e$ en la posición $(1,1)$:
esto contradice la maximización de $(d)$. Entonces $d$ divide la columna $1$,
y simétricamente la fila $1$. Restar múltiplos de la fila 1 y
la columna 1 los borra: $B'$ es equivalente a $\begin{pmatrix} d &
0\\ 0 & B''\end{pmatrix}$. A continuación,$d$divide cada entrada$b$ de
$B''$: agregar la fila de $b$ a la fila 1 (una operación elemental; el
nueva primera fila contiene las entradas $d$ y $b$) y repita el
argumento de limpieza de columnas: un no múltiplo mejoraría nuevamente
$(d)$. Ahora induzca el tamaño: $B''$, todas cuyas entradas son
divisible por $d$, tiene una forma Smith $\operatorname{diag}(d_2,
\dots)$cuyas entradas siguen siendo divisibles por$d$ (cada entrada de cualquier
$QB''P$ es una combinación $A$ de entradas de $B''$); establezca $d_1 =
d$.

\emph{Unicidad.} Sea $D_i(B)$ un mcd de todos los $i \times i$
menores. Operaciones de filas y columnas, y más en general
multiplicación por la matriz \emph{cualquier}, no se puede reducir el mcd: el
$i \times i$ menores de $QB$ son combinaciones $A$ de las de $B$
(Cauchy--expansión de Binet; o directamente: cada fila de $QB$ es una
combinación de filas de $B$, y los menores son multilineales en filas).
Entonces $D_i(QBP)$ y $D_i(B)$ se dividen entre sí: $D_i$ es un
invariante de equivalencia. En la forma diagonal, los menores $i
\times i$distintos de cero son los productos de$i$de los$d_j$, y
la divisibilidad $d_1 \mid \dots \mid d_r$ hace que $d_1 \cdots d_i$ sea el
mcd. Por lo tanto $d_1 \cdots d_i = D_i(B)$ hasta unidades, y se determina $d_i =
D_i/D_{i-1}$.
\end{proof}

\begin{method}\label{met:b3:modules:smith}
Sobre un \omterm{def:b3:rings:pidufd}{Dominio euclidiano} ($\Z$, $K[X]$), la reducción de Smith es una
algoritmo --- no se necesita argumento de maximalidad: trae la entrada de
tamaño euclidiano más pequeño hasta la posición $(1,1)$; si no logra dividirse
alguna entrada de su fila o columna, una división euclidiana deja una
Resto estrictamente más pequeño allí --- cámbielo y reinicie
(terminación: las dimensiones disminuyen); cuando divide toda su fila y
columna, bórrelos; si no logra dividir una entrada interna, agregue eso
fila a fila $1$ y reiniciar; recursivo en el bloque interior. en
práctica con matrices enteras: calcular $D_1 = \gcd$ de entradas,
$D_2$, \dots\ a través de menores para tamaños pequeños, o ejecute el algoritmo.
\end{method}

\begin{example}[Una reducción de Smith, completa]
\label{ex:b3:modules:smithworked}
Reducir $M = \begin{pmatrix} 1 & 2 & 3\\ 4 & 5 & 6\\ 7 & 8 & 9
\end{pmatrix}$sobre$\Z$. La esquina$1$ lo divide todo:
borre su fila y columna ($L_2 \leftarrow L_2 - 4L_1$, $L_3
\leftarrow L_3 - 7L_1$, luego$C_2 \leftarrow C_2 - 2C_1$,$C_3
\leftarrow C_3 - 3C_1$):
\[
M \sim \begin{pmatrix} 1 & 0 & 0\\ 0 & -3 & -6\\ 0 & -6 & -12
\end{pmatrix} .
\]

\leftarrow L_3 - 2L_2$y$C_3 \leftarrow C_3 - 2C_2$ límpiela.
a $\operatorname{diag}(-3, 0)$. Ajustar signos (multiplicar un
fila por $-1$, una operación legal):
\[
M \sim \operatorname{diag}(1, 3, 0),
\qquad
\Z^3/M\Z^3 \cong \Z/3\Z \times \Z .
\]
Verificación cruzada por divisores determinantes: $D_1 =
\gcd(\text{entries}) = 1$; cada$2\times2$menor de$M$ es un
múltiplo de $3$ (p. ej.$\det\bigl(\begin{smallmatrix}1 & 2\\ 4
& 5\end{smallmatrix}\bigr) = -3$) y uno es igual a$-3$:$D_2 =
3$;$D_3 = \det M = 0$. Por lo tanto$d_1 = 1$,$d_2 = 3$,$d_3 = 0$:
misma respuesta. Dos lecciones: un factor invariante cero registra la

\end{example}

\section{El teorema de la estructura}

\begin{definition}\label{def:b3:modules:torsion}
Sea $A$ un dominio y $M$ un módulo $A$. el
El submódulo \emph{torsión}\index{torsión} es
\[
T(M) = \{x \in M : ax = 0 \text{ for some } a \neq 0\}
\]
(un submódulo: si $ax = by = 0$ entonces $ab(x + y) = 0$, $ab \ne 0$).
$M$ es \emph{\omterm{def:b3:modules:free}{sin torsión}} si $T(M) = 0$, un \emph{torsion \omterm{def:b3:modules:module}{módulo}}
si $T(M) = M$.
\end{definition}

\begin{theorem}[Estructura de módulos generados finitamente sobre un
PID]\label{thm:b3:modules:structure}
Sea $A$ un módulo \omterm{def:b3:rings:pidufd}{PID} y $M$ un módulo \omterm{def:b3:modules:free}{finitamente generado} $A$. allí
existe un $r \in \N$ único y unidades no nulas $d_1 \mid d_2 \mid
\cdots \mid d_s$, únicas hasta \omterm{def:b3:rings:divisibility}{asociados}, con\index{estructura
teorema para módulos}
\[
M \;\cong\; A^r \,\oplus\, A/(d_1) \oplus \cdots \oplus A/(d_s).
\]
Además $T(M) \cong A/(d_1)\oplus\dots\oplus A/(d_s)$ y $M/T(M)
\cong A^r$: a \omterm{def:b3:modules:free}{finitamente generado} \emph{\omterm{def:b3:modules:torsion}{sin torsión}} \omterm{def:b3:modules:module}{módulo} sobre un
\omterm{def:b3:rings:pidufd}{PID} es \omterm{def:b3:modules:free}{libre}.
\end{theorem}

\begin{proof}
\emph{Existencia.} Presente $M \cong A^n/\operatorname{im}(C)$
(\cref{rem:b3:modules:presentation}) y ponga $C$ en forma Smith:
después de los dos cambios de base, $M \cong A^n / (d_1A \times \dots
\times d_rA \times 0 \times \dots \times 0) \cong A/(d_1) \oplus
\dots \oplus A/(d_r) \oplus A^{\,n - r}$. Descartar los factores
donde $d_i$ es una unidad ($A/(d_i) = 0$); la cadena de divisibilidad
sobrevive.

\emph{The \omterm{def:b3:modules:torsion}{torsión} identification.} En la descomposición, $A^r$ es
\omterm{def:b3:modules:torsion}{sin torsión} (un dominio no tiene divisores de cero) y cada $A/(d_i)$
es \omterm{def:b3:modules:torsion}{torsión} (asesinado por $d_i \ne 0$); una suma directa divide \omterm{def:b3:modules:torsion}{torsión}
en consecuencia: $T(M) = \bigoplus_i A/(d_i)$ y $M/T(M) \cong A^r$.

\emph{Unicidad de $r$}: $M/T(M) \cong A^r$ depende sólo de $M$,
y \cref{prop:b3:modules:rank} fija a $r$ hacia abajo.

\emph{Unicidad del $d_i$}: basta con tratar el \omterm{def:b3:modules:torsion}{torsión
módulo} $T = T(M)$. Descomponga cada $d_i$ en números primos y divídalo por
el teorema del resto chino (\cref{thm:b3:rings:crt}; distinto
los primos generan \omterm{thm:b3:rings:crt}{ideales comaximales}):
\[
A/(d) \cong \bigoplus_{p \mid d} A/\bigl(p^{v_p(d)}\bigr):
\qquad
T \cong \bigoplus_{p} \bigoplus_{j} A/\bigl(p^{k_{p,j}}\bigr),
\]
el \emph{divisores elementales}\index{divisores elementales}
$p^{k_{p,j}}$. Por el contrario, los $d_i$ se reconstruyen a partir del
conjunto múltiple de \omterm{thm:b3:modules:structure}{divisores elementales} ($d_s$ = producto del más alto
potencia de cada primo, etc.), por lo que basta con demostrar la
El multiconjunto $\{k_{p,j}\}_j$ está determinado por $T$, para cada número primo.
$p$. Reparar $p$; para $j \geq 1$ considere los espacios vectoriales $A/(p)$
$p^{j-1}T/p^jT$. En un factor cíclico $A/(p^k)$:
\[
p^{j-1}\bigl(A/(p^k)\bigr)\big/p^{j}\bigl(A/(p^k)\bigr)
\cong
\begin{cases}
A/(p) & \text{if } j \leq k,\\
0 & \text{if } j > k,
\end{cases}
\]
y en un factor $A/(q^k)$, $q \neq p$: la multiplicación por $p$ es
biyectivo allí ($p$ mod invertible $q^k$: Bézout), por lo que el
el cociente es $0$. Pasan sumas directas: $\dim_{A/(p)}
p^{j-1}T/p^jT = \#\{i : k_{p,i} \geq j\}$. Estos intrínsecos
las dimensiones determinan el conjunto múltiple de exponentes.
\end{proof}

\begin{corollary}[Grupos abelianos generados finitamente]
\label{cor:b3:modules:abelian}
Cada grupo abeliano \omterm{def:b3:modules:free}{finitamente generado} es $\Z^r \times
\Z/d_1\Z\times\dots\times\Z/d_s\Z$con$d_1 \mid \dots \mid
d_s$, de forma única. Cada grupo abeliano \emph{finito} es producto de
grupos cíclicos de orden de potencia primaria, únicos como
conjunto múltiple.\index{grupo abeliano, estructura}
\end{corollary}

\begin{example}\label{ex:b3:modules:abelian}
Los grupos abelianos de orden $p^n$ corresponden a \emph{particiones}
de $n$: para $p^4$: $\Z/p^4$, $\Z/p^3\times\Z/p$,
$(\Z/p^2)^2$, $\Z/p^2 \times (\Z/p)^2$, $(\Z/p)^4$ --- cinco
grupos, ya que $4$ tiene cinco particiones. Los pedidos mixtos multiplican el
cuenta primo por primo (CRT): hay $5 \times 2$ abelianos
grupos de orden $2^4 \cdot 3^2 = 144$.
\end{example}

\section{Aplicación: formas canónicas de endomorfismos.}

Sea $K$ un campo, $V$ un espacio vectorial $K$ de dimensión finita
$n$ y $u \in \mathcal L(V)$; convierta $V$ en un módulo $K[X]$ a través de $P
\cdot x = P(u)(x)$ (\cref{ex:b3:modules:examples}). Este \omterm{def:b3:modules:module}{módulo}
es \omterm{def:b3:modules:free}{finitamente generado} (se genera una base $K$) y \omterm{def:b3:modules:torsion}{torsión}: para
cada $x$, los vectores $n+1$ $x, u(x), \dots, u^n(x)$ son
Depende de $K$, lo que proporciona un polinomio de aniquilación distinto de cero.

\begin{definition}\label{def:b3:modules:companion}
Para $P = X^m + a_{m-1}X^{m-1} + \dots + a_0$ monic, el
\emph{matriz acompañante}\index{matriz acompañante} es
\[
C_P = \begin{pmatrix}
0 & & & -a_0\\
1 & \ddots & & -a_1\\
& \ddots & 0 & \vdots\\
& & 1 & -a_{m-1}
\end{pmatrix} :
\]
la matriz de ``multiplicación por $X$'' en $K[X]/(P)$ en el
base $1, \bar X, \dots, \bar X^{m-1}$.
\end{definition}

\begin{theorem}[Frobenius: forma canónica racional]
\label{thm:b3:modules:frobenius}
Existe una secuencia única de polinomios mónicos no constantes $P_1
\mid P_2 \mid \dots \mid P_s$ (el \emph{similitud
invariantes}\index{invariantes de similitud} de $u$) tal que, como
$K[X]$-módulos,
\[
V \cong K[X]/(P_1) \oplus \cdots \oplus K[X]/(P_s):
\]
en una base adecuada, $u$ tiene matriz diagonal de bloques
$\operatorname{diag}(C_{P_1}, \dots, C_{P_s})$. Además:
\begin{enumerate}
  \item $P_s = \mu_u$ (polinomio mínimo) y $P_1\cdots P_s =
        \chi_u$(polinomio característico); en particular$\mu_u
        \mid \chi_u$ \emph{(Cayley--Hamilton reprobado)} y
        $\chi_u \mid \mu_u^{\,s}$, por lo que $\chi_u$ y $\mu_u$ tienen
        los mismos factores \omterm{def:b3:rings:divisibility}{irreducible}.
  \item Dos endomorfismos (o matrices cuadradas) son similares si y así
        tienen el mismo \omterm{thm:b3:modules:frobenius}{invariantes de similitud}.
\end{enumerate}
\end{theorem}

\begin{proof}
El teorema de estructura (\cref{thm:b3:modules:structure}) aplicado
al \omterm{def:b3:rings:pidufd}{PID} $K[X]$: el \omterm{def:b3:modules:torsion}{módulo de torsión} $V$ se descompone con
\omterm{thm:b3:modules:smith}{factores invariantes} $P_i$, mónico normalizado (las unidades de $K[X]$ son
$K^\times$); no ocurre ninguna pieza \omterm{def:b3:modules:free}{libre} ($V$ es \omterm{def:b3:modules:torsion}{torsión}). En cada cíclico
factor $K[X]/(P_i)$, la multiplicación por $X$ tiene matriz $C_{P_i}$
en la base de potencias de $\bar X$: la concatenación de bases da la
forma de bloque.

(1) El aniquilador de $V = \bigoplus K[X]/(P_i)$ es $(P_1)\cap
\dots\cap(P_s) = (P_s)$(cadena de divisibilidad:$P_s$ es un común
múltiple, y la clase de $1$ en el último factor se elimina
exactamente por $(P_s)$): $\mu_u = P_s$. Para $\chi_u$: en forma cíclica
factor, $\chi_{C_P} = P$, por inducción en $m = \deg P$. Expandiendo
$\det(XI_m - C_P)$ a lo largo de la primera fila (cuyas entradas son $X$, luego
ceros, luego $a_0$ en la última columna):
\[
\det(XI_m - C_P) = X\,\det\bigl(XI_{m-1} - C_{\tilde P}\bigr)
+ (-1)^{1+m}\,a_0\,\det L ,
\]
donde $\tilde P = X^{m-1} + a_{m-1}X^{m-2} + \dots + a_1$ (mismo
forma, una talla menos) y $L$ es triangular con diagonal
$(-1, \dots, -1)$, entonces $\det L = (-1)^{m-1}$. Por inducción el
el primer término es $X\tilde P$ y el segundo es $a_0$: el total es
$X\tilde P + a_0 = P$ (caso base $m=1$: $\det(X + a_0) = P$).
Los determinantes se multiplican por bloques:
$\chi_u = \prod P_i$. Cayley--Hamilton: $\chi_u \in (\mu_u)$
desde $P_s \mid$ cada\dots\ a la inversa cada $P_i \mid P_s$, entonces
$\chi_u = \prod P_i$ divide $P_s^{\,s} = \mu_u^s$; y $\mu_u =
P_s$divide$\chi_u$ como uno de sus factores.

(2) Endomorfismos similares son estructuras conjugadas \omterm{def:b3:modules:module}{módulo}, por lo tanto
tener invariantes iguales (unicidad en
\cref{thm:b3:modules:structure}); invariantes inversamente iguales
dar módulos $K[X]$ isomórficos, y un isomorfismo \omterm{def:b3:modules:module}{módulo} es
exactamente una biyección lineal que entrelaza los dos endomorfismos: a
similitud.
\end{proof}

\begin{corollary}[La similitud no depende de la extensión del campo.]
\label{cor:b3:modules:descent}
Sean $K \subseteq L$ campos y $M, N \in M_n(K)$. Si $M$ y
$N$ son similares a $L$, son similares a $K$.
\end{corollary}

\begin{proof}
Los \omterm{thm:b3:modules:frobenius}{invariantes de similitud} de $M$ se calculan mediante el menor de Smith.
fórmula (\cref{thm:b3:modules:smith}) aplicada a la presentación
matriz $XI_n - M$ sobre $K[X]$ --- de hecho, el módulo $K[X]$
$V_M = K^n$ tiene la presentación $XI_n - M$: la aplicación $K[X]^n \to V_M$,
$(Q_i) \mapsto \sum Q_i(M)e_i$, está en funcionamiento con el kernel generado por
las columnas de $XI_n - M$ (una verificación directa: módulo aquellos
columnas, cada elemento de $K[X]^n$ se reduce a un vector constante,
y los vectores constantes se asignan biyectivamente; los hechizos del problema del fin de semana
esto fuera). Los mcd de polinomios no cambian bajo el campo
extensión: si $d$ es el monic mcd en $K[X]$ de una familia $(f_j)$,
Bézout da $d = \sum u_jf_j$ con $u_j \in K[X]$, por lo que cada
el divisor común del $f_j$ \emph{in $L[X]$} divide al $d$; como $d$
es en sí mismo un divisor común, también es el mcd en $L[X]$. Por lo tanto
el \omterm{thm:b3:modules:smith}{factores invariantes} de $XI_n - M$, cocientes de sucesivos
mcds menores, son los mismos sobre $K$ y sobre $L$: $M, N$ tienen el
mismo \omterm{thm:b3:modules:frobenius}{invariantes de similitud} sobre $L$ si sobre $K$; concluir por
\cref{thm:b3:modules:frobenius}(2).
\end{proof}

\begin{theorem}[forma Jordan, rederivada]\label{thm:b3:modules:jordan}
Supongamos que $\chi_u$ se divide en $K$ (por ejemplo,\ $K = \C$). Aplicando a
$V$ la descomposición del divisor elemental (prueba de
\cref{thm:b3:modules:structure}) en lugar de \omterm{thm:b3:modules:smith}{factores invariantes}:
\[
V \cong \bigoplus_{\lambda, j} K[X]\big/\bigl((X -
\lambda)^{k_{\lambda,j}}\bigr),
\]
y en la base $\bigl(\overline{(X-\lambda)^{k-1}}, \dots,
\overline{(X - \lambda)}, \bar 1\bigr)$de cada factor actúa$u$
como el bloque Jordan $J_k(\lambda)$: cada endomorfismo con división
polinomio característico tiene una base de Jordan, y el multiconjunto de
bloques $(\lambda, k)$ es único.\index{forma jordana}
\end{theorem}

\begin{proof}
El \omterm{thm:b3:modules:structure}{divisores elementales} del \omterm{def:b3:modules:torsion}{módulo de torsión} $V$ son los
$(X - \lambda)^k$ con $X - \lambda$ que abarca el \omterm{def:b3:rings:divisibility}{irreducible}
factores de $\mu_u$ (que se divide, ya que $\chi_u$ lo hace y ambos
tener los mismos factores \omterm{def:b3:rings:divisibility}{irreducible},
\cref{thm:b3:modules:frobenius}). En $W = K[X]/((X-\lambda)^k)$,
ponga $f_j = \overline{(X - \lambda)^{k-j}}$ para $j = 1, \dots, k$:
luego $(X - \lambda)f_j = f_{j-1}$ (con $f_0 = 0$), es decir\ $u(f_j)
= \lambda f_j + f_{j-1}$: la matriz de$u$en$(f_1, \dots,
f_k)$es exactamente$J_k(\lambda)$ (unos por encima de la diagonal).
La unicidad del conjunto múltiple de \omterm{thm:b3:modules:structure}{divisores elementales} es
\cref{thm:b3:modules:structure}.
\end{proof}

\begin{remark}\label{rem:b3:modules:overview}
La jerarquía de las formas canónicas es ahora transparente: la
El formulario \emph{racional} existe en todos los campos y detecta
similitud absoluta (\cref{cor:b3:modules:descent}); el
La forma \emph{Jordán} es su refinamiento cuando $\chi_u$ se divide.
Las pruebas de conteo de dimensiones del teorema de Jordan del año 2 son
subsumido: toda la combinatoria era la aritmética del \omterm{def:b3:rings:pidufd}{PID}
$K[X]$.
\end{remark}
