\chapter{Los espacios \texorpdfstring{$L^p$}{Lp}}\label{ch:b3:lp}

La integral de Lebesgue se construyó para el análisis; los espacios $L^p$
es donde vive ese análisis. Son espacios de Banach
(Riesz--Fischer) --- el \omterm{thm:b3:complete:completion}{terminaciones} que
\cref{exo:b3:complete:1} mostró las funciones \omterm{def:b3:topology:continuity}{continuo}
faltaba --- y apoyan una tecnología de suavizado,
\emph{convolution against \omterm{def:b3:lp:mollifier}{apaciguadores}}, que se aproxima a cualquier
Función $L^p$ por $\mathcal C^\infty$. Este capítulo demuestra
las versiones integrales de Hölder y Minkowski, \omterm{def:b3:complete:complete}{lo completo},
los teoremas \omterm{ex:b3:lebesgue:gamma}{densidad} y la máquina de regularización, terminando
con la geografía de inclusión e interpolación del $L^p$
escala. En todo momento, $(X, \mathcal A, \mu)$ es un \omterm{def:b3:measure:measure}{medir el espacio}.
y las funciones tienen valores complejos; en $\R^d$, el \omterm{def:b3:measure:measure}{medida} es
$\lambda_d$.

\section{Definición; Hölder y Minkowski}

\begin{definition}\label{def:b3:lp:space}
Para $1 \leq p < \infty$, $\mathcal L^p(\mu)$ es el conjunto de
\omterm{def:b3:lebesgue:measurable}{mensurable} $f$ con $\norm f_p = \bigl(\int\abs
f^p\dd\mu\bigr)^{1/p} < \infty$y$\mathcal L^\infty(\mu)$
el conjunto de $f$ acotado fuera de un conjunto nulo, con $\norm
f_\infty$el \emph{sup esencial} --- el menos$M$ con
$\abs f \leq M$ a.e.\ (se alcanza la inf: intersecta el nulo
conjuntos para $M + \frac1n$). Dado que $\norm f_p = 0$ solo fuerza a $f
= 0$ \emph{a.e.} (\cref{exo:b3:lebesgue:5}), definimos
\[
L^p(\mu) = \mathcal L^p(\mu)/\{f = 0 \text{ a.e.}\} :
\]
los elementos son clases de funciones conjuntos módulo nulo, y
$\norm\cdot_p$ es una norma genuina en
$L^p$.\index{Espacio Lp@$L^p$ espacio}
\end{definition}

\begin{theorem}[desigualdad de Hölder]\label{thm:b3:lp:holder}
Vamos $1 \leq p, q \leq \infty$ con $\frac1p + \frac1q = 1$
(exponentes conjugados). Para \omterm{def:b3:lebesgue:measurable}{mensurable} $f, g$:\index{Hölder's
inequality}
\[
\norm{fg}_1 \leq \norm f_p\,\norm g_q ,
\]
con igualdad (para $1 < p < \infty$, normas finitas, $f,g\ne0$)
si $\abs f^p$ y $\abs g^q$ son proporcionales, e.g.
\end{theorem}

\begin{proof}
Los casos $\{p, q\} = \{1, \infty\}$ son directos ($\abs{fg} \leq
\norm g_\infty\abs f$a.e.). Sea$1 < p < \infty$; normalizar
$\norm f_p = \norm g_q = 1$ (homogeneidad; cero o infinito
las normas son triviales). Desigualdad de Young $ab \leq \frac{a^p}p +
\frac{b^q}q$($a, b \geq 0$; concavidad de$\ln$, como en
\cref{pb:b3:banach:1}) da $\abs{f g} \leq
\frac{\abs f^p}p + \frac{\abs g^q}q$puntual; integrar:$\norm{fg}_1
\leq \frac1p + \frac1q = 1$. La igualdad obliga a.e.\ igualdad en
Joven, es decir\ $\abs f^p = \abs g^q$ a.e.\ (después de la
normalización; deshacerlo, proporcionalidad).
\end{proof}

\begin{theorem}[desigualdad de Minkowski]\label{thm:b3:lp:minkowski}
Para $1 \leq p \leq \infty$: $\norm{f + g}_p \leq \norm f_p +
\norm g_p$.\index{La desigualdad de Minkowski}
\end{theorem}

\begin{proof}
$p = 1, \infty$: desigualdad puntual/a.e.\ triángulo. Para $1 <
p < \infty$, asuma$\norm{f+g}_p < \infty$ (de lo contrario utilice
$\abs{f+g}^p \leq 2^{p-1}(\abs f^p + \abs g^p)$, de convexidad
de $t^p$, para ver el lado izquierdo es finito cuando el derecho lo es).
entonces
\[
\norm{f{+}g}_p^p \leq \int\abs f\,\abs{f{+}g}^{p-1} +
\int\abs g\,\abs{f{+}g}^{p-1}
\leq \bigl(\norm f_p + \norm g_p\bigr)\,
\bigl\|\abs{f{+}g}^{p-1}\bigr\|_q
\]
por Hölder y $\norm{\abs{f+g}^{p-1}}_q =
\norm{f+g}_p^{p/q}$desde$(p-1)q = p$; dividir por
$\norm{f+g}_p^{p/q}$ (si es distinto de cero; en caso contrario, trivial) y utilice $p -
\frac pq = 1$.
\end{proof}

\section{La integridad y sus acompañantes.}

\begin{theorem}[Riesz--Fischer]\label{thm:b3:lp:complete}
Para $1 \leq p \leq \infty$, $L^p(\mu)$ es un espacio de Banach.
Además, cada secuencia que converge en $L^p$ tiene una subsecuencia
\emph{\omterm{def:b3:lebesgue:l1}{casi en todas partes}} convergente (con un dominador $L^p$
en el caso $p < \infty$).\index{Teorema de Riesz-Fischer}
\end{theorem}

\begin{proof}
$p = \infty$: una secuencia $\norm\cdot_\infty$-Cauchy es, fuera de un
conjunto nulo único (unión de muchos contables), uniformemente Cauchy:
converge uniformemente fuera de él; hecho. Deje $p < \infty$. Por
\cref{exo:b3:complete:1}(b) basta con sumar absolutamente
Serie convergente: sea $\sum\norm{f_k}_p = M < \infty$. conjunto
$G_n = \sum_{k \leq n}\abs{f_k}$ y $G = \sum_k\abs{f_k}$
(puntualmente en $[0,\infty]$): por Minkowski $\norm{G_n}_p \leq
M$, y MCT ($G_n^p \nearrow G^p$) da$\int G^p \leq M^p$:
$G < \infty$ a.e., por lo que la serie $\sum f_k(x)$ converge
absolutamente para a.e.\ $x$; llame a la suma $S(x)$ (cualquier valor en
el conjunto nulo). Luego $\abs{S - \sum_{k\leq n}f_k}^p \leq (2G)^p
\in L^1$y DCT da$\norm{S - \sum_{k\leq n}f_k}_p \to 0$:
la serie converge en $L^p$.

La declaración de subsecuencia: si $f_n \to f$ en $L^p$, elija $n_k$
con $\norm{f_{n_{k+1}} - f_{n_k}}_p \leq 2^{-k}$; la serie
$\sum(f_{n_{k+1}} - f_{n_k})$ cae bajo el anterior
argumento: converge absolutamente a.e., dominado por un $G \in
L^p$, por lo que$f_{n_k} \to f_{n_1} + \sum(\cdots) $ a.e., y esto
a.e.-limit debe ser (un representante de) $f$ (ambos son $L^p$
límites). El dominador: $\abs{f_{n_k}} \leq \abs{f_{n_1}} + G$.
\end{proof}

\begin{remark}\label{rem:b3:lp:modes}
$L^p$ convergencia no implica convergencia a.e.\ (la
secuencia \emph{máquina de escribir}, \cref{exo:b3:lp:3}), ni
por el contrario (escapar de obstáculos): los dos modos están vinculados únicamente
a través de subsecuencias y dominación. Manteniendo los contraejemplos
de \cref{exo:b3:lp:3} en mente es la mejor vacuna.
\end{remark}

\section{Teoremas de densidad}

\begin{theorem}\label{thm:b3:lp:density}
Deje $1 \leq p < \infty$.
\begin{enumerate}
  \item \omterm{def:b3:lebesgue:simple}{Funciones simples} (con soportes \omterm{def:b3:measure:measure}{medida finita}) son
        denso en $L^p(\mu)$.
  \item En $L^p(\R^d)$, el \omterm{def:b3:topology:continuity}{continuo} admite de forma compacta
        Las funciones $\mathcal C_c(\R^d)$ son densas.
  \item La traducción es \omterm{def:b3:topology:continuity}{continuo} en $L^p(\R^d)$: escritura
        $\tau_hf = f(\cdot - h)$, $\norm{\tau_hf - f}_p \to 0$
        como $h \to 0$.
\end{enumerate}
Ninguno de los tres es válido para $p = \infty$.
\end{theorem}

\begin{proof}
(1) Para $f \geq 0$: el diádico $s_n \nearrow f$ de
\cref{thm:b3:lebesgue:approximation} satisface $\abs{f - s_n}^p
\leq f^p \in L^1$: DCT. (Cada$s_n \leq f$se encuentra en$L^p$, y
sus conjuntos de niveles tienen \omterm{def:b3:measure:measure}{medida} finito donde el valor es
positivo: $\mu(s_n \geq c) \leq c^{-p}\int f^p$.) Dividir
general $f$ en cuatro partes no negativas.

(2) Por (1) basta con aproximar $\mathbf 1_A$, $A$ Borel
con $\lambda_d(A) < \infty$. Regularidad (prueba como en
\cref{thm:b3:measure:regularity}) da \omterm{def:b3:topology:compact}{compacto} $K \subseteq A
\subseteq U$abierto con$\lambda_d(U\setminus K) <
\varepsilon$; Función de Urysohn
\[
\varphi(x) = \frac{d(x, \R^d\setminus U)}{d(x, \R^d\setminus U)
+ d(x, K)}
\]
es \omterm{def:b3:topology:continuity}{continuo}, $1$ en $K$, $0$ fuera de $U$ y se puede tomar
compatible de forma compacta (reduzca $U$ a una apertura limitada primero). entonces
$\norm{\mathbf 1_A - \varphi}_p^p \leq \lambda_d(U\setminus K)
< \varepsilon$.

(3) Para $g \in \mathcal C_c$: uniforme \omterm{def:b3:topology:continuity}{continuidad} da
$\norm{\tau_hg - g}_\infty \to 0$, con soportes en posición fija.
\omterm{def:b3:topology:compact}{compacto} para $\abs h \leq 1$: $\norm{\tau_hg - g}_p \to 0$. Para
general $f$: elija $g \in \mathcal C_c$ con $\norm{f - g}_p <
\varepsilon$; luego$\norm{\tau_hf - f}_p \leq 2\norm{f - g}_p +
\norm{\tau_hg - g}_p$ (invariancia de traducción de la norma).

Para $p = \infty$: la aproximación uniforme de las funciones $\mathbf
1_{\intoo0\infty}$ por \omterm{def:b3:topology:continuity}{continuo} es imposible
(saltar) y $\norm{\tau_h\mathbf 1_{\intoo0\infty} - \mathbf
1_{\intoo0\infty}}_\infty = 1$para$h \neq 0$.
\end{proof}

\section{Convolución y regularización}

\begin{theorem}[Desigualdad de jóvenes]\label{thm:b3:lp:young}
Sea $1 \leq p \leq \infty$, $f \in L^1(\R^d)$, $g \in
L^p(\R^d)$. Entonces$f * g$se define a.e., pertenece a$L^p$,
y\index{La desigualdad de Young}
\[
\norm{f * g}_p \leq \norm f_1\,\norm g_p .
\]
\end{theorem}

\begin{proof}
$p = \infty$: encuadernación directa. $p = 1$:
\cref{thm:b3:product:convolution}. Dejar $1 < p < \infty$, $q$
conjugar. Divida $\abs{f(y)} = \abs{f(y)}^{1/q}\cdot
\abs{f(y)}^{1/p}$ y aplique Hölder:
\[
\int\abs{f(y)}\,\abs{g(x{-}y)}\,\dd y
\leq \Bigl(\int\abs f\Bigr)^{1/q}
\Bigl(\int\abs{f(y)}\,\abs{g(x - y)}^p\,\dd y\Bigr)^{1/p} .
\]
Elevar a la potencia $p$ e integrar en $x$; Tonelli en el
el segundo factor da $\norm f_1^{p/q}\cdot\norm f_1\norm g_p^p$,
es decir\ $\norm{f*g}_p^p \leq \norm f_1^{1 + p/q}\norm g_p^p =
(\norm f_1\norm g_p)^p$ --- y la finitud del Tonelli
integral justifica a.e.\ convergencia absoluta como en
\cref{thm:b3:product:convolution}.
\end{proof}

\begin{definition}[Mollificadores]\label{def:b3:lp:mollifier}
La función\index{calmante}\index{función de golpe}
\[
\rho(x) = \begin{cases}
c\,\exp\Bigl(-\dfrac{1}{1 - \norm x^2}\Bigr) & \norm x < 1,\\
0 & \norm x \geq 1,
\end{cases}
\]
con $c$ normalizando $\int\rho = 1$, está $\mathcal C^\infty$ encendido
$\R^d$: la cuestión es que $t \mapsto \eu^{-1/t}\mathbf
1_{t>0}$es$\mathcal C^\infty$sobre$\R$, todos sus derivados en
$0^+$ siendo $0$ (cada derivada es $P(1/t)\eu^{-1/t}$ para un
polinomio $P$, que tiende a $0$; inducción). Para
$\varepsilon > 0$ conjunto $\rho_\varepsilon(x) =
\varepsilon^{-d}\rho(x/\varepsilon)$: compatible con$\bar B(0,
\varepsilon)$, aún de$1$ integral.
\end{definition}

\begin{theorem}[Regularización]\label{thm:b3:lp:regularization}
Sean $1 \leq p < \infty$ y $f \in L^p(\R^d)$. Entonces:
\begin{enumerate}
  \item$f * \rho_\varepsilon \in \mathcal C^\infty(\R^d)$,
        con $\partial^\alpha(f * \rho_\varepsilon) = f *
        \partial^\alpha\rho_\varepsilon$;
  \item $\norm{f * \rho_\varepsilon - f}_p \to 0$ como
        $\varepsilon \to 0$;
  \item en consecuencia $\mathcal C^\infty_c(\R^d)$ es denso en
        $L^p(\R^d)$.
\end{enumerate}
\end{theorem}

\begin{proof}
(1) Diferenciación bajo la integral
(\cref{thm:b3:lebesgue:paramdiff}) en $x$: para $x$ en una bola
$B$, $\abs{\partial_{x_i}\rho_\varepsilon(x - y)} \leq
C_\varepsilon\,\mathbf 1_{K}(y)$con$K$\omterm{def:b3:topology:compact}{compacto} ($y$ dentro
$\varepsilon$ de $B$) y $\abs f\,\mathbf 1_K \in L^1$
(Hölder contra $\mathbf 1_K$): se aplica el teorema;
iterar para obtener derivadas superiores.

(2) Desde $\int\rho_\varepsilon = 1$:
\[
(f * \rho_\varepsilon)(x) - f(x)
= \int \bigl(f(x - y) - f(x)\bigr)\rho_\varepsilon(y)\,\dd y ,
\]
y la desigualdad \emph{integral} de Minkowski --- o directamente:
Hölder/Jensen con la probabilidad \omterm{def:b3:measure:measure}{medida}
$\rho_\varepsilon\dd y$ y Tonelli ---
\[
\norm{f*\rho_\varepsilon - f}_p^p
\leq \int\Bigl(\int\abs{f(x-y) -
f(x)}^p\dd x\Bigr)\rho_\varepsilon(y)\,\dd y
= \int \norm{\tau_yf - f}_p^p\;\rho_\varepsilon(y)\,\dd y
\]
(el paso intermedio: aplicar la desigualdad de Jensen,
\cref{exo:b3:lp:10}, a la integral $y$ interna, luego Tonelli).
El integrando es compatible con $\norm y \leq \varepsilon$ y
tiende a $0$ allí uniformemente como $\varepsilon \to 0$
(\cref{thm:b3:lp:density}(3)): toda la expresión tiende a
$0$.

(3) $f$ aproximado por $g \in \mathcal C_c$
(\cref{thm:b3:lp:density}(2)), luego $g$ por $g *
\rho_\varepsilon \in \mathcal C_c^\infty$ (\omterm{def:b3:topology:compact}{compacto} admite:
suma de apoyos).
\end{proof}

\begin{example}[Apaciguador $\abs x$, con tarifas]
\label{ex:b3:lp:mollifyexample}
Tome $f(x) = \abs x$ en $\R$ (localmente $L^1$; el teorema
se aplica en cada ventana delimitada) y un \omterm{def:b3:lp:mollifier}{calmante} simétrico
$\rho_\varepsilon$. entonces
\[
f_\varepsilon(x) = (f * \rho_\varepsilon)(x)
= \int\abs{x - y}\,\rho_\varepsilon(y)\,\dd y
\]
es $\mathcal C^\infty$; lejos del problema, no pasa nada:
para $\abs x \geq \varepsilon$, $\abs{x - y}$ es lineal en $x$
sobre el soporte de $\rho_\varepsilon$, entonces $f_\varepsilon(x) =
\abs x$ \emph{exactamente} (la simetría anula la corrección).
Cerca de $0$, suavizar los costos con precisión
\[
0 \leq f_\varepsilon(0) = \int\abs
y\,\rho_\varepsilon(y)\,\dd y \leq \varepsilon,
\qquad
\norm{f_\varepsilon - f}_\infty \leq \varepsilon :
\]
el error de aproximación se limita a la
$\varepsilon$-barrio de la singularidad y es de su
tamaño. Mientras tanto $f_\varepsilon'' \geq 0$ en todas partes
($f$ es convexo y la convolución contra $\rho_\varepsilon
\geq 0$preserva la convexidad), con$\int f_\varepsilon'' =
f_\varepsilon'(\infty) - f_\varepsilon'(-\infty) = 2$: el
La segunda derivada es un bulto de masa $2$ comprimido a lo ancho.
$O(\varepsilon)$, entonces $\norm{f_\varepsilon''}_\infty \gtrsim
\varepsilon^{-1}$. El suavizado es un oficio: error uniforme
$O(\varepsilon)$ contra la explosión derivada
$O(\varepsilon^{-1})$ --- el tipo de cambio exacto que
análisis cuantitativo (desigualdades de interpolación,
\cref{pb:b3:lp:1} círculo de ideas) se formaliza.
\end{example}

\begin{corollary}[Lema fundamental del calculo de
variaciones]\label{cor:b3:lp:fundlemma}
Dejar $f \in L^1_{\mathrm{loc}}(\R^d)$ (\omterm{def:b3:lebesgue:l1}{integrable} en \omterm{def:b3:topology:compact}{compactos})
con $\int f\varphi = 0$ por cada $\varphi \in \mathcal
C^\infty_c(\R^d)$. Entonces$f = 0$ a.e.\index{lema fundamental de
el calculo de variaciones}
\end{corollary}

\begin{proof}
Fija una bola $B = B(0, R)$ y deja que $g = f\mathbf 1_{B(0, R+1)}
\in L^1$. Para$x \in B$y$\varepsilon < 1$:$(g *
\rho_\varepsilon)(x) = \int f(y)\rho_\varepsilon(x - y)\dd y =
0$, siendo la función de prueba$y \mapsto \rho_\varepsilon(x-y)
\in \mathcal C_c^\infty$. Pero$g * \rho_\varepsilon \to g$ en
$L^1$ (\cref{thm:b3:lp:regularization}): $g = 0$ a.e.\ en $B$;
escape $\R^d$.
\end{proof}

\section{La geografía \texorpdfstring{$L^p$}{Lp}}

\begin{proposition}\label{prop:b3:lp:inclusions}
(a) Si $\mu(X) < \infty$ y $1 \leq p \leq q \leq \infty$,
luego $L^q \subseteq L^p$ con $\norm f_p \leq
\mu(X)^{\frac1p - \frac1q}\,\norm f_q$.
(b) En $\R^d$ (infinito \omterm{def:b3:measure:measure}{medida}) no hay inclusiones:
para $p \neq q$ hay funciones en $L^p\setminus L^q$.
(c) (Interpolación) Si $p < r < q$ y $\alpha \in \intoo01$
está definido por $\frac1r = \frac\alpha p + \frac{1 - \alpha}q$,
entonces
\[
\norm f_r \leq \norm f_p^{\alpha}\,\norm f_q^{1 - \alpha} ;
\]
en particular $L^p \cap L^q \subseteq L^r$.
\end{proposition}

\begin{proof}
(a) Hölder con exponentes $\frac qp$ y su conjugado:
$\int\abs f^p\cdot 1 \leq \norm{\abs f^p}_{q/p}\,\norm
1_{(q/p)'} = \norm f_q^p\,\mu(X)^{1 - p/q}$($q = \infty$
directamente). (b) Cerca de $0$ y cerca de $\infty$, potencias $x^{-\alpha}$
calibrar: \cref{exo:b3:lp:2}. (c) Escriba $\abs f^r = \abs
f^{r\alpha}\abs f^{r(1-\alpha)}$ y aplique Hölder con el
par conjugado $\frac p{r\alpha}$, $\frac q{r(1-\alpha)}$
(conjuga precisamente por la definición de $\alpha$):
$\int\abs f^r \leq \norm f_p^{r\alpha}\norm f_q^{r(1 -
\alpha)}$.
\end{proof}

\begin{method}\label{met:b3:lp:toolkit}
El kit de herramientas $L^p$, como se utiliza a continuación: para demostrar una
identidad o desigualdad para todos $f \in L^p$ --- demostrarlo en un
clase densa ($\mathcal C_c^\infty$ vía
\cref{thm:b3:lp:regularization}) y extender por \omterm{def:b3:topology:continuity}{continuidad}
(\cref{thm:b3:complete:extension}, ambos lados están
$L^p$-continuo); para probar $f = 0$, pruebe con $\mathcal
C_c^\infty$ (\cref{cor:b3:lp:fundlemma}); para ganar suavidad,
convolucionar; para intercambiar exponentes, Hölder e interpolación. el
La teoría de Fourier de \cref{ch:b3:fouriertransform} es larga.
aplicación de este método.
\end{method}

(c) ¿Cuándo Hölder es una igualdad para $p = 1$, $q = \infty$?

\begin{exercise}[$\star$]\label{exo:b3:lp:2}
¿A qué pertenece $p \in \intco1\infty$ lo siguiente?
$L^p$?
\[

x^{-1/2}\mathbf 1_{\intoo1\infty},\qquad

\qquad

\]
Concluir: en $\intoo01$ pequeño $p$ es más fácil, en
$\intoo1\infty$ grande $p$ es más fácil y ningún $L^p$ contiene
otro en $\R$.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b3:lp:11}
(Desigualdad de convolución de Young) Sea $1 \leq p, q, r \leq
\infty$con$\frac1p + \frac1q = 1 + \frac1r$y$f \in
L^p(\R^d)$,$g \in L^q(\R^d)$.

\emph{(Write, for conjugate exponents worked out from $p, q,
r$,
\[
\abs{f(y)g(x-y)} = \bigl(\abs f^p\abs g^q\bigr)^{1/r}
\cdot\abs f^{\,p(1/p - 1/r)}\cdot\abs g^{\,q(1/q - 1/r)},
\]
and apply the three-factor Hölder inequality with
exponents $r$, $\frac{pr}{r - p}$, $\frac{qr}{r - q}$; then
integrate in $x$ by Tonelli.)}
(b) Marque los tres casos especiales ya conocidos: $r =
\infty$(Hölder, \cref{exo:b3:lp:6});$q = 1$

(c) ¿Por qué no hay desigualdad con $\frac1p + \frac1q < 1 +

\end{exercise}

\section{Problema: la desigualdad de Hardy}

\begin{problem}[{Problema del fin de semana --- Desigualdad de Hardy y
su constante aguda}]\label{pb:b3:lp:1}
Para $f \in L^p(\intoo0{+\infty})$, $1 < p < \infty$, defina
el \emph{operador resistente}
\[
(Hf)(x) = \frac1x\int_0^x f(t)\,\dd t .
\]
La desigualdad de Hardy (1920) afirma\index{La desigualdad de Hardy}
\[
\norm{Hf}_p \;\leq\; \frac{p}{p-1}\,\norm f_p ,
\]
y la constante $\frac p{p-1}$ es óptima y no se alcanza.
Este problema lo prueba todo y luego se extiende a la serie.

\medskip
\noindent\textbf{Parte I --- The inequality.} Suponga primero $f
\geq 0$ \omterm{def:b3:topology:continuity}{continuo} con soporte \omterm{def:b3:topology:compact}{compacto} en
$\intoo0{+\infty}$ y deja que $F(x) = \int_0^xf$.
\begin{enumerate}
  \item Muestre que $Hf \in L^p$: cerca de $0$, $F$ desaparece en un
        \omterm{def:b3:topology:topology}{entorno} de $0$; cerca de $\infty$, $F$ está limitado, por lo que
        $(Hf)(x) = O(1/x)$ y $x \mapsto \frac1x$ pertenecen a
        $L^p(\intoo1{+\infty})$ para $p > 1$.
  \item Integrar por partes para mostrar
        \[
        \int_0^\infty \Bigl(\frac Fx\Bigr)^p\dd x
        = \frac{p}{p-1}\int_0^\infty\Bigl(\frac
        Fx\Bigr)^{p-1}f(x)\,\dd x .
        \]
        \emph{(Differentiate $x^{1-p}F^p$; boundary terms
        vanish --- justify both ends.)}
  \item Aplique Hölder al lado derecho y deduzca
        $\norm{Hf}_p \leq \frac p{p-1}\norm f_p$ para tal $f$.
  \item Extender a todo $L^p$: para $f \geq 0$, construir
        $f_n$ \omterm{def:b3:topology:continuity}{continuo} con soporte \omterm{def:b3:topology:compact}{compacto} en
        $\intoo0{+\infty}$, $0 \leq f_n \nearrow f$ a.e.\
        \emph{(truncar, luego aproximar monótonamente ---
        justificar la construcción)}; entonces $Hf_n \nearrow Hf$
        puntualmente (MCT dentro del promedio) y MCT pasa el
        desigualdad hasta el límite. Para $f$ firmado o complejo,
        concluir con $\abs{Hf} \leq H\abs f$.
\end{enumerate}

\medskip
\noindent\textbf{Parte II --- Optimality.}
\begin{enumerate}[resume]
  \item Para $A > 1$ sea $f_A(t) = t^{-1/p}\,\mathbf
        1_{\intcc1A}(t)$. Calcular$\norm{f_A}_p^p = \ln A$
        y, para $1 \leq x \leq A$,
        \[
        (Hf_A)(x) = \frac p{p-1}\;x^{-1/p}\,
        \bigl(1 - x^{-(1 - 1/p)}\bigr).
        \]
  \item Deduzca $\liminf_{A\to\infty} \norm{Hf_A}_p/
        \norm{f_A}_p \geq \frac{p}{p-1}$ y concluya que
        la constante es óptima.
  \item Demuestre que la igualdad $\norm{Hf}_p = \frac
        p{p-1}\norm f_p$con$f \neq 0$ es imposible.
        \emph{(Track the equality case of Hölder in question
        3: it would force $f = cx^{-1/p}$-type behavior, which
        is not in $L^p$.)}
\end{enumerate}

\medskip
\noindent\textbf{Parte III --- The discrete inequality.}
\begin{enumerate}[resume]
  \item Para un $g \geq 0$ no creciente en $(0,\infty)$ y
        $a_n = g(n)$, compare $\sum a_n^p$ y $\int g^p$,
        y $H$-promedios en consecuencia, a deducir de la Parte I
        \emph{La desigualdad discreta de Hardy}: para $a_n \geq 0$,
        \[
        \sum_{n\geq1}\Bigl(\frac{a_1 + \dots +
        a_n}{n}\Bigr)^{p}
        \;\leq\;
        \Bigl(\frac{p}{p-1}\Bigr)^{p}\,\sum_{n\geq1}a_n^p
        \]
        --- pruébelo primero para $(a_n)$ no creciente a través del
        comparación anterior, luego reduzca el caso general a la
        uno no creciente por reordenamiento \emph{(admite, con
        una justificación de una línea, que ordenar $(a_n)$ en
        El orden decreciente solo puede aumentar el lado izquierdo mientras
        arreglando el derecho)}.
  \item Deducir: si $\sum a_n^p < \infty$ entonces el Cesàro
        medios de $(a_n)$ son nuevamente $\ell^p$ --- y dan un
        ejemplo ($p = 2$) donde $(a_n) \in \ell^2$ pero $a_n$
        No es sumable, pero Hardy aún controla los medios.
\end{enumerate}

\medskip
\noindent\textbf{Parte IV --- Epilogue.}
\begin{enumerate}[resume]
  \item Demuestre que la desigualdad de Hardy falla para $p = 1$: con
        $f = \mathbf 1_{\intcc01}$, calcule $Hf$ y observe
        $Hf \notin L^1$. ¿Dónde se rompe la prueba?
\end{enumerate}

\medskip
\noindent\textbf{Part V --- The maximal function, and
Lebesgue's differentiation theorem.} Promedios resistentes del
origen; Hardy--promedio de Littlewood \emph{alrededor de cada punto}.
Para $f \in L^1(\R)$ defina
\[
Mf(x) = \sup_{r>0}\ \frac1{2r}\int_{x-r}^{x+r}\abs
f\,\dd\lambda .
\]
\begin{enumerate}[resume]
  \item (Vitali, versión finita) Sea $B_1, \dots, B_N$
        intervalos abiertos. Mostrar que hay una subfamilia separada
        $B_{i_1}, \dots, B_{i_k}$ con $\bigcup_jB_j
        \subseteq \bigcup_l3B_{i_l}$, donde$3B$ denota el
        intervalo con el mismo \omterm{ex:b3:groups:actions}{centro} y triple longitud
        \emph{(codicioso: elige repetidamente el intervalo más largo
        separados de los ya elegidos)}.
  \item (tipo débil $(1,1)$) Muestre que por cada $t > 0$,
        \[
        \lambda\bigl(\{Mf > t\}\bigr) \;\leq\;
        \frac3t\,\norm f_1 :
        \]
        cada $x$ con $Mf(x) > t$ posee un intervalo centrado
        $B_x$ con $\int_{B_x}\abs f > t\,\lambda(B_x)$; tomar
        a \omterm{def:b3:topology:compact}{compacto} $K \subseteq \{Mf > t\}$ (regularidad interna),
        cubrirlo por un número finito de $B_x$, aplicar la pregunta 11,
        y escape.
  \item Calcular $M\mathbf 1_{\intcc01}(x)$ para $x > 1$ y
        deducir que $Mf \notin L^1$ por cada $f \neq 0$
        ($Mf(x) \geq \frac c{\abs x}$ en infinito): en $p =
        1$, la desigualdad débil de la pregunta 12 es la mejor
        posible enunciado.
  \item (tipo fuerte para $p > 1$) Para $f \in L^p$: divida $f =
        f\,\mathbf 1_{\abs f > t/2} + f\,\mathbf 1_{\abs f
        \leq t/2}$, observe$Mf \leq M\bigl(f\mathbf 1_{\abs
        f > t/2}\bigr) + \frac t2$ y combine la pregunta 12
        con la fórmula del pastel de capas
        (\cref{prop:b3:product:layercake}) y Tonelli a
        probar
        \[
        \norm{Mf}_p^p \;\leq\;
        \frac{6p\,2^{p-1}}{p-1}\,\norm f_p^p .
        \]
        (La ampliación como $p \downarrow 1$ es la pregunta 13
        fracaso, cuantificado.)
  \item (teorema de diferenciación de Lebesgue) Demuestre: para $f \in
        L^1(\R)$,
        \[
        \frac1{2r}\int_{x-r}^{x+r}f\,\dd\lambda
        \;\longrightarrow\; f(x)
        \qquad (r \to 0)\quad\text{for a.e.\ }x .
        \]
        \emph{(Clear for \omterm{def:b3:topology:continuity}{continuo} $f$. In general write $f
        = g + h$,$g$ \omterm{def:b3:topology:continuity}{continuo} with \omterm{def:b3:topology:compact}{compacto} support,
        $\norm h_1 < \varepsilon$
        (\cref{thm:b3:lp:density}); the set where
        $\limsup_{r\to0}$ of the averaged oscillation exceeds
        $\delta$ sits inside $\{Mh > \delta/2\} \cup \{\abs h
        > \delta/2\}$, of \omterm{def:b3:measure:measure}{medida}$O(\varepsilon/\delta)$;
        let $\varepsilon \to 0$, then $\delta \to 0$ along a
        sequence.)}
  \item Deduzca: (a) casi todos los puntos son \emph{Lebesgue
        punto} de $f$; (b) para $f \in L^1$, la primitiva
        $F(x) = \int_0^xf$ es diferenciable a.e.\ con $F' =
        f$ a.e.\ --- la mitad integral de la fundamental
        teorema de cálculo en el mundo de Lebesgue, cierre
        el círculo abierto por la escalera
        (\cref{pb:b3:measure:1}), que mostró lo contrario
        la mitad puede fallar.
  \item (puntos \omterm{ex:b3:lebesgue:gamma}{Densidad}) Para \omterm{def:b3:lebesgue:measurable}{mensurable} $A \subseteq \R$,
        muestran que casi todos los $x \in A$ satisfacen
        $\frac{\lambda(A\cap\intcc{x-r}{x+r})}{2r} \to 1$:
        \omterm{def:b3:lebesgue:measurable}{mensurable} los conjuntos están localmente llenos en casi todos
        sus puntos. Dibuja, en dos líneas, cómo se obtiene esto.
        Otra prueba más del teorema de Steinhaus.
        (\cref{exo:b3:measure:8}).
\end{enumerate}

\medskip
\noindent\textbf{Part VI --- Variations on the theme of
averaging.}
\begin{enumerate}[resume]
  \item (resistente ponderado) Para $\alpha < p - 1$ y $f \geq 0$,
        mostrar
        \[
        \int_0^\infty\Bigl(\frac{F(x)}x\Bigr)^{p}
        x^{\alpha}\,\dd x \;\leq\;
        \Bigl(\frac{p}{p - 1 - \alpha}\Bigr)^{p}
        \int_0^\infty f(x)^p\,x^{\alpha}\,\dd x
        \]
        por la misma integración por partes, y comprobar que la
        límite $\alpha = p - 1$ está realmente prohibido
        (adapte el contraejemplo de la pregunta 10).
  \item (El adjunto) Deje $H^*f(x) =
        \int_x^{\infty}\frac{f(t)}t\,\dd t$. Mostrar$\langle
        Hf, g\rangle = \langle f, H^*g\rangle$ para
        $f, g$ (Tonelli) no negativo y demostrar
        $\norm{H^*f}_p \leq p\,\norm f_p$\emph{(directamente por
        partes, o de Hardy en el exponente conjugado por
        dualidad --- tenga en cuenta qué exponente recoge cuál
        constante)}.
  \item (Una desigualdad de tipo Hilbert) Deduce que para
        no negativo $f \in L^p$, $g \in L^q$:
        \[
        \int_0^\infty\!\!\int_0^\infty
        \frac{f(x)\,g(y)}{\max(x,y)}\,\dd x\,\dd y
        \;\leq\; (p + q)\,\norm f_p\,\norm g_q
        \]
        \emph{(split along $y < x$ / $y \geq x$: each half is
        a pairing of one function against a Hardy transform
        of the other)}.
  \item (Optimidad, discreta) Demuestre que la constante
        $\bigl(\frac p{p-1}\bigr)^p$ de la pregunta 8 también es
        óptimo: pruebe en $a_n = n^{-1/p}\,\mathbf 1_{n \leq
        N}$, compare ambos lados con integrales y deje$N
        \to \infty$ (el espejo discreto de la Parte II).
  \item (Síntesis) Aparecieron tres operadores promediadores en
        este problema: $H$ de Hardy, el discreto Cesàro
        media y el operador máximo $M$. estado en uno
        línea cada uno lo que dice su delimitación, observe que todos
        tres fallan exactamente en $p = 1$ y explica por qué es
        el mismo fallo tres veces (la cola armónica
        $\frac1x$).
\end{enumerate}

\medskip
\noindent\textbf{Part VII --- Carleman's inequality, and how
sharp is sharp.}
\begin{enumerate}[resume]
  \item (Desigualdad de Carleman) Sea $a_n \geq 0$ con $\sum
        a_n < \infty$. Aplicar la desigualdad discreta de Hardy de
        pregunta 8 a $b_n = a_n^{1/p}$, utilice el
        desigualdad media aritmética-geométrica, y sea $p \to
        \infty$(muestre que$p \mapsto
        \bigl(\frac p{p-1}\bigr)^p$disminuye a$\eu$) para
        obtener
        \[
        \sum_{n\geq1}\bigl(a_1a_2\cdots a_n\bigr)^{1/n}
        \;\leq\; \eu\,\sum_{n\geq1}a_n :
        \]
        las medias geométricas de una secuencia sumable son
        sumable, con un coste como máximo $\eu$.
  \item (La constante $\eu$ es óptima) Prueba $a_n =
        \frac1n\,\mathbf 1_{n\leq N}$: usando Stirling
        horquillado de \cref{pb:b3:product:1}, mostrar
        $(n!)^{-1/n} = \frac\eu n\bigl(1 +
        O\bigl(\frac{\ln n}n\bigr)\bigr)$, deduce que ambos
        lados de Carleman crecen como $\eu\ln N$, y concluyen
        que ninguna constante menor que $\eu$ puede funcionar. (Observar
        el patrón: los optimizadores de Hardy y de Carleman
        son ambas secuencias de tipo armónico que simplemente no logran
        estar en el espacio.)
  \item (¿Con qué lentitud se acerca "nítido"?) Tome $p = 2$.
        Para $f = \mathbf 1_{\intcc01}$, calcule
        $\norm{Hf}_2/\norm f_2 = \sqrt2$, contra el atado
        $2$. Para los casi optimizadores $f_A$ de la pregunta 5,
        demostrar la identidad exacta
        \[
        \norm{Hf_A}_2^2 = 4\ln A - 8 + \frac{8}{\sqrt A},
        \qquad\text{so}\qquad
        \frac{\norm{Hf_A}_2^2}{\norm{f_A}_2^2}
        = 4 - \frac{8 - 8A^{-1/2}}{\ln A} .
        \]
        Evaluar en $A = \eu^{10}$ (relación $\approx 1.790$)
        y comenta: el supremo $2$ se acerca a gran velocidad
        $1/\ln A$ solamente --- una constante óptima puede ser casi
        invisible numéricamente.
\end{enumerate}
\end{problem}
