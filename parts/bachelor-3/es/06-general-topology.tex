\section{Topologías, conjuntos abiertos, continuidad.}

\begin{definition}\label{def:b3:topology:topology}
Un \emph{topología}\index{topología} en un set $X$ es una familia
$\mathcal T$ de subconjuntos de $X$ --- llamado \emph{abierto
conjuntos}\index{conjunto abierto} --- tal que: $\varnothing, X \in
\mathcal T$; cualquier unión de conjuntos abiertos es abierta; cualquier \emph{finito}
la intersección de conjuntos abiertos es abierta. El par $(X, \mathcal T)$ es
un \emph{espacio topológico}. Los complementos de conjuntos abiertos son
\emph{cerrado}. A \emph{entorno}\index{entorno} de $x$
es un conjunto que contiene un conjunto abierto que contiene $x$.
\end{definition}

\begin{example}\label{ex:b3:topology:examples}
(a) Un espacio métrico, con "abierto" como en el Año 2 (uniones de abiertos
bolas): el \emph{metric \omterm{def:b3:topology:topology}{topología}}; diferentes métricas pueden dar
el mismo \omterm{def:b3:topology:topology}{topología} (métricas equivalentes). Un espacio cuyo \omterm{def:b3:topology:topology}{topología}
surge de alguna métrica es \emph{metrizable}.
(b) El \emph{discreto} \omterm{def:b3:topology:topology}{topología} (todos los subconjuntos) y el
\emph{indiscreto} \omterm{def:b3:topology:topology}{topología} $\{\varnothing, X\}$.
(c) El \emph{cofinito} \omterm{def:b3:topology:topology}{topología} en un conjunto infinito: abre $=$
vacío o cofinito. No metrizable, como veremos
(\cref{exo:b3:topology:3}).
(d) El $\R$, el habitual \omterm{def:b3:topology:topology}{topología}; en $\bar\R = \R \cup
\{\pm\infty\}$, el orden \omterm{def:b3:topology:topology}{topología} generado por rayos --- haciendo
``$x_n \to +\infty$'' es un ejemplo de convergencia simple.
\end{example}

\begin{definition}\label{def:b3:topology:interior}
Para $A \subseteq X$: el \emph{interior} $\mathring A$ es el
mayor \omterm{def:b3:topology:topology}{conjunto abierto} dentro de $A$ (unión de todos ellos); el
\emph{cierre}\index{cierre} $\bar A$ el conjunto cerrado más pequeño
que contiene $A$; el \emph{límite} $\partial A = \bar A
\setminus \mathring A$.$A$ es \emph{denso}\index{conjunto denso} si
$\bar A = X$. Uno tiene $x \in \bar A$ si cada \omterm{def:b3:topology:topology}{entorno} de
$x$ cumple con $A$ (si algún \omterm{def:b3:topology:topology}{entorno} no cumple con $A$, su núcleo abierto
el complemento es un conjunto cerrado más pequeño alrededor de $A$; por el contrario).
\end{definition}

\begin{definition}\label{def:b3:topology:basis}
Un \emph{base}\index{base de una topología} de $\mathcal T$ es un
familia $\mathcal B \subseteq \mathcal T$ tal que cada \omterm{def:b3:topology:topology}{abierto
conjunto} es una unión de miembros de $\mathcal B$ (por ejemplo,\ bolas abiertas en un
espacio métrico; intervalos abiertos en $\R$). Una familia $\mathcal B$ de
subconjuntos de $X$ es una base de \emph{alguno} \omterm{def:b3:topology:topology}{topología} si cubre
$X$ y, para $B_1, B_2 \in \mathcal B$ y $x \in B_1 \cap B_2$,
algún $B_3 \in \mathcal B$ tiene $x \in B_3 \subseteq B_1\cap B_2$
--- entonces ``uniones de miembros'' es un \omterm{def:b3:topology:topology}{topología}, el \omterm{def:b3:topology:topology}{topología}
\emph{generado} por $\mathcal B$.
\end{definition}

\begin{definition}\label{def:b3:topology:continuity}
$f \colon X \to Y$ es \emph{continuo}\index{aplicación continua} si
$f^{-1}(V)$ está abierto para cada $V \subseteq Y$ abierto ---
de manera equivalente, las preimágenes de conjuntos cerrados son cerradas; equivalentemente,
por cada $x$ y \omterm{def:b3:topology:topology}{entorno} $V$ de $f(x)$, $f^{-1}(V)$ es un
\omterm{def:b3:topology:topology}{entorno} de $x$ (continuidad \emph{at} cada $x$). eso
basta con comprobar un \omterm{def:b3:topology:basis}{base} de $Y$. Composiciones de continuo.
Los mapas son continuos. A \emph{homeomorfismo}\index{homeomorfismo}
es una biyección continua con inversa continua; \omterm{def:b3:topology:topology}{topología}
estudia las propiedades conservadas por los homeomorfismos.
\end{definition}

\begin{proposition}[Continuidad vs cierre; pegado]
\label{prop:b3:topology:gluing}
(a) $f$ es \omterm{def:b3:topology:continuity}{continuo} si y si $f(\bar A) \subseteq \overline{f(A)}$
para todos los $A \subseteq X$.
(b) Si $X = F_1 \cup F_2$ con $F_1, F_2$ cerrado y $f\colon X
\to Y$restringe \omterm{def:b3:topology:continuity}{continuamente} a cada$F_i$, entonces$f$ es
\omterm{def:b3:topology:continuity}{continuo}.
\end{proposition}

\begin{proof}
(a) Si $f$ \omterm{def:b3:topology:continuity}{continuo}: $f^{-1}(\overline{f(A)})$ está cerrado y
contiene $A$, por lo tanto contiene $\bar A$. Por el contrario aplicar el
criterio para $A = f^{-1}(F)$, $F$ cerrado: $f(\bar A) \subseteq
\overline{f(f^{-1}(F))} \subseteq \bar F = F$, por lo que$\bar A
\subseteq f^{-1}(F) = A$: las preimágenes de conjuntos cerrados están cerradas.
(b) Para $F \subseteq Y$ cerrado: $f^{-1}(F) =
(f\restriction_{F_1})^{-1}(F) \cup (f\restriction_{F_2})^{-1}
(F)$, una unión de dos conjuntos cerrados en$F_1$, resp. $F_2$, por lo tanto
cerrado en $X$ ($F_i$ cerrado: cerrado en cerrado está cerrado).
\end{proof}

\begin{definition}\label{def:b3:topology:hausdorff}
$X$ es \emph{Hausdorff}\index{espacio hausdorff} (o
\emph{apartado}) si dos puntos distintos tienen puntos disjuntos
\omterm{def:b3:topology:topology}{barrios}. Los espacios métricos son Hausdorff (bolas de radio
$d(x,y)/2$). Una secuencia $(x_n)$ \emph{converge} a $x$ si cada
\omterm{def:b3:topology:topology}{entorno} de $x$ contiene todos menos un número finito de $x_n$; en un
Espacio de Hausdorff, los límites son únicos (dos límites tendrían
disjuntos \omterm{def:b3:topology:topology}{barrios} cada uno con una cola). En un Hausdorff
espacio, los puntos --- por lo tanto los conjuntos finitos --- son cerrados.
\end{definition}

En espacios métricos, las secuencias detectan todo: $x \in \bar A$
si alguna secuencia de $A$ converge a $x$ (tome $x_n \in A \cap
B(x, 1/n)$), y$f$ es \omterm{def:b3:topology:continuity}{continuo} si es secuencial
\omterm{def:b3:topology:continuity}{continuo}. En espacios generales ambas equivalencias fallan; el
Los sustitutos de secuencia correctos (filtros, redes) pertenecen a una
curso avanzado. Indicaremos los resultados basados en secuencias en métricas.
espacios y resultados basados en coberturas en general --- y probarlos
equivalente donde se encuentren.
\end{remark}

\section{Subespacios, productos, cocientes.}

\begin{definition}\label{def:b3:topology:constructions}
Tres formas de crear espacios nuevos a partir de los antiguos:\index{producto
topología}\index{topología del cociente}
\begin{enumerate}
  \item \emph{Subespacio}: en $A \subseteq X$, los \omterm{def:b3:topology:topology}{conjuntos abiertos} son
        el $U \cap A$, $U$ abierto en $X$ --- el más tosco
        \omterm{def:b3:topology:topology}{topología} realizando la inclusión \omterm{def:b3:topology:continuity}{continuo}.
  \item \emph{Producto}: en $X \times Y$ (y productos finitos),
        el \omterm{def:b3:topology:topology}{topología} con \omterm{def:b3:topology:basis}{base} el \emph{cajas abiertas} $U \times
        V$; en un producto infinito$\prod_i X_i$, el \omterm{def:b3:topology:basis}{base}
        consta de cajas $\prod U_i$ con $U_i = X_i$ para
        \emph{todos menos un número finito} $i$ --- el más tosco
        \omterm{def:b3:topology:topology}{topología} realizando cada proyección \omterm{def:b3:topology:continuity}{continuo}.
  \item \emph{Cociente}: si $\sim$ es una equivalencia de $X$ y
        $\pi\colon X \to X/{\sim}$ la proyección, declarar $V
        \subseteq X/{\sim}$abierto si$\pi^{-1}(V)$ está abierto ---
        el mejor \omterm{def:b3:topology:topology}{topología} haciendo $\pi$ \omterm{def:b3:topology:continuity}{continuo}.
\end{enumerate}
\end{definition}

\begin{proposition}[propiedades universales]
\label{prop:b3:topology:universal}
(a) $f \colon Z \to X \times Y$ es \omterm{def:b3:topology:continuity}{continuo} si ambos
Los componentes $\pi_X \circ f$, $\pi_Y \circ f$ son (iguales para
productos arbitrarios).
(b) $g \colon X/{\sim} \to Z$ es \omterm{def:b3:topology:continuity}{continuo} si y si $g \circ \pi
\colon X \to Z$ lo es.
\end{proposition}

\begin{proof}
a) Necesidad: composiciones. Suficiencia: basta con comprobar
preimágenes de cajas \omterm{def:b3:topology:basis}{base}: $f^{-1}(U \times V) = (\pi_Xf)^{-1}(U)
\cap (\pi_Yf)^{-1}(V)$, abiertas (un número finito de factores$\neq X_i$
en el caso infinito). b) Necesidad: composición. Suficiencia:
para $W \subseteq Z$ abierto, $\pi^{-1}(g^{-1}(W)) = (g\pi)^{-1}(W)$
está abierto, lo que según la definición de \omterm{def:b3:topology:constructions}{topología del cociente} significa
$g^{-1}(W)$ está abierto.
\end{proof}

\begin{example}\label{ex:b3:topology:circle}
El cociente $\R/\Z$ (identificar $x$ y $x + n$) es \omterm{def:b3:topology:continuity}{homeomórfico}
al círculo $S^1 = \{z \in \C : \abs z = 1\}$: la aplicación $x
\mapsto \eu^{2\iu\pi x}$ pasa a una biyección \omterm{def:b3:topology:continuity}{continuo}
$\R/\Z \to S^1$ (\cref{prop:b3:topology:universal}(b)); es
la inversa es \omterm{def:b3:topology:continuity}{continuo} por el argumento de compacidad de
\cref{cor:b3:topology:compacthomeo} a continuación
(\cref{exo:b3:topology:5} detalla todo, incluido el motivo
$\R/\Z$ es \omterm{def:b3:topology:hausdorff}{Hausdorff} y \omterm{def:b3:topology:compact}{compacto}). Asimismo $[0,1]$ con
extremos pegados es $S^1$, el cuadrado con lados opuestos pegados
es el toroide, y pegar es finalmente un teorema, no una imagen.
\end{example}

\section{Compacidad}

\begin{definition}\label{def:b3:topology:compact}
Un \emph{tapa abierta} de $X$ es una familia $(U_i)_{i\in I}$ de \omterm{def:b3:topology:topology}{abierto
conjuntos} con $\bigcup U_i = X$. $X$ es
\emph{compacto}\index{espacio compacto} si es \omterm{def:b3:topology:hausdorff}{Hausdorff} y cada
Tapa abierta admite un \emph{subcubierta finita}. De manera equivalente (tomando
complementos): cada familia de conjuntos cerrados con \emph{finito
propiedad de intersección} (todas las subfamilias finitas tienen no vacíos
intersección) tiene una intersección total no vacía.
\end{definition}

\begin{theorem}[Primeras propiedades]\label{thm:b3:topology:compactprops}
Sea $X$ \omterm{def:b3:topology:compact}{compacto}.
\begin{enumerate}
  \item Un subconjunto cerrado de $X$ es \omterm{def:b3:topology:compact}{compacto}; un subconjunto \omterm{def:b3:topology:compact}{compacto} de un
        \omterm{def:b3:topology:hausdorff}{espacio hausdorff} está cerrado.
  \item Una imagen \omterm{def:b3:topology:continuity}{continuo} de un \omterm{def:b3:topology:compact}{espacio compacto} en un \omterm{def:b3:topology:hausdorff}{Hausdorff
        espacio} es \omterm{def:b3:topology:compact}{compacto}. En particular, un \omterm{def:b3:topology:continuity}{continuo} $f\colon X
        \to \R$ está acotado y alcanza sus límites.
  \item Una secuencia decreciente de subconjuntos cerrados no vacíos de $X$
        tiene intersección no vacía.
\end{enumerate}
\end{theorem}

\begin{proof}
(1) Deje $F \subseteq X$ cerrada y $(U_i)$ una tapa abierta de $F$
(por apertura de $X$): agregar $X \setminus F$ da una portada abierta de
$X$; una subcubierta finita, menos $X \setminus F$, cubre $F$.
El subespacio \omterm{def:b3:topology:hausdorff}{Hausdorff} está claro. Por el contrario, deje que $K \subseteq Y$
\omterm{def:b3:topology:compact}{compacto}, $Y$ \omterm{def:b3:topology:hausdorff}{Hausdorff} y $y \notin K$: para cada $x \in K$
elegir abierto disjunto $U_x \ni x$, $V_x \ni y$; finitos $U_x$
cubre $K$, y la intersección del correspondiente $V_x$ es una
\omterm{def:b3:topology:topology}{entorno} de $y$ está separado de la portada de $K$, por lo tanto de $K$:
el complemento de $K$ está abierto.

(2) Si $(V_j)$ cubre $f(X)$, entonces $(f^{-1}(V_j))$ cubre $X$; un
la subcobertura finita de abajo proviene de la subcobertura finita
arriba. La imagen es \omterm{def:b3:topology:hausdorff}{Hausdorff} como subespacio. Para $f$ reales:
$f(X)$ es \omterm{def:b3:topology:compact}{compacto} en $\R$, por lo tanto cerrado y acotado (cubierto por
$(-n, n)$ para limitación; cerrado por (1)), y un cerrado acotado
el conjunto contiene su supremo.

(3) Si $\bigcap F_n = \varnothing$, abre $X \setminus F_n$
cubierta $X$; un número finito es suficiente, por lo que algo de $F_{n_0} =
\varnothing$ (la secuencia disminuye) --- contradicción.
\end{proof}

\begin{corollary}\label{cor:b3:topology:compacthomeo}
Un \omterm{def:b3:topology:continuity}{continuo} \emph{biyección} de un \omterm{def:b3:topology:compact}{espacio compacto} a un
\omterm{def:b3:topology:hausdorff}{espacio hausdorff} es un \omterm{def:b3:topology:continuity}{homeomorfismo}.
\end{corollary}

\begin{proof}
Lo inverso es \omterm{def:b3:topology:continuity}{continuo} si las imágenes directas de conjuntos cerrados son
cerrado; un $F$ cerrado es \omterm{def:b3:topology:compact}{compacto}
(\cref{thm:b3:topology:compactprops}(1)), su imagen es \omterm{def:b3:topology:compact}{compacto}
(2), por lo tanto cerró (1) en el objetivo \omterm{def:b3:topology:hausdorff}{Hausdorff}.
\end{proof}

\begin{theorem}[Productos finitos]\label{thm:b3:topology:tychonoff}
Un producto finito de \omterm{def:b3:topology:compact}{espacios compactos} es \omterm{def:b3:topology:compact}{compacto}.\index{Tychonoff
teorema (caso finito)}
\end{theorem}

\begin{proof}
Basta con tratar $X \times Y$. \omterm{def:b3:topology:hausdorff}{Hausdorff} se hereda
(separados en una coordenada). Sea $(W_i)$ una portada abierta de $X
\times Y$; podemos asumir que$W_i$son cuadros \omterm{def:b3:topology:basis}{base}$U_i \times
V_i$(refinar: cada punto se encuentra en un cuadro dentro de algunos$W_i$; a
la subcubierta finita de cajas produce una de $W_i$). Arreglar $x \in X$:
el \emph{rebanada} $\{x\}\times Y \cong Y$ es \omterm{def:b3:topology:compact}{compacto}, así que finitamente
muchas cajas $U_1\times V_1, \dots, U_k\times V_k$ lo tapan, con
$x \in U_j$ para todos los $j$; entonces $U^x = \bigcap_{j \leq k} U_j$ es
un \omterm{def:b3:topology:topology}{entorno} abierto de $x$ con $U^x \times Y$ cubierto por
un número finito de cajas (el \emph{lema del tubo}\index{lema del tubo}:
para $(x', y) \in U^x \times Y$, $y \in V_j$
para algunos $j$ ya que las cajas cubrían $\{x\}\times Y$ en el nivel
$y$ y $x' \in U^x \subseteq U_j$, por lo que $(x', y) \in U_j
\times V_j$). Ahora un número finito de$U^x$
cubrir el \omterm{def:b3:topology:compact}{compacto} $X$; las correspondientes colecciones finitas de
tapa cajas $X \times Y$.
\end{proof}

\begin{theorem}[Compacidad en espacios métricos]
\label{thm:b3:topology:metriccompact}
Para un espacio métrico $(X, d)$, lo siguiente es
equivalente:\index{compacidad secuencial}
\begin{enumerate}
  \item $X$ es \omterm{def:b3:topology:compact}{compacto} (Borel--Lebesgue);
  \item cada secuencia en $X$ tiene una subsecuencia convergente
        (\omterm{thm:b3:topology:metriccompact}{compacidad secuencial} --- Definición del año 2);
  \item $X$ está completo y \emph{totalmente delimitado}: por cada
        $\varepsilon > 0$, un número finito de bolas de radio
        $\varepsilon$ cubierta $X$.
\end{enumerate}
\end{theorem}

\begin{proof}
(1)$\Rightarrow$(2): Dejemos que $(x_n)$ no tenga convergencia
subsecuencia. Entonces cada $x \in X$ tiene una bola abierta $B_x$
que contiene $x_n$ sólo para un número finito de $n$ (de lo contrario, un
la subsecuencia converge a $x$: tomar radios $1/k$). finitos
$B_x$ cubre $X$, por lo que solo existen un número finito de índices $n$:
absurdo.

(2)$\Rightarrow$(3): Completitud: una secuencia de Cauchy con un
la subsecuencia convergente converge (Año 2). Limitación total: si
algunos $\varepsilon$ no admiten cobertura finita, elijan inductivamente
$x_{n+1}$ fuera de $\bigcup_{k\leq n} B(x_k, \varepsilon)$: el
la secuencia tiene $d(x_m, x_n) \geq \varepsilon$ para $m \neq n$, no
Subsecuencia de Cauchy, no convergente.

(3)$\Rightarrow$(1): Primero, (3) implica (2): dado $(x_n)$,
cubrir $X$ por un número finito de bolas de radio $1$: una, $B_1$,
contiene una subsecuencia; tapa $X$ por bolas de radio $1/2$: una
contiene una subsecuencia adicional; iterar y diagonalizar: el
subsecuencia diagonal es Cauchy (dos términos más allá de la etapa $k$ se encuentran
en una bola común de radio $2^{-k}$, hasta la habitual $2\cdot$),
por tanto converge. Ahora dejemos que $(U_i)$ sea una portada abierta y supongamos que no.
subcobertura finita. \emph{Argumento del número de Lebesgue}: por cada $n$,
alguna bola $B(y_n, 2^{-n})$ no está cubierta por un número finito
$U_i$ --- de hecho, cubre $X$ con un número finito de bolas de radio
$2^{-n}$; si cada uno estuviera cubierto de forma finita, también lo estaría $X$. Por
(2), una subsecuencia $y_{n_k} \to y$; elija $i$ con $y \in U_i$
y $r > 0$ con $B(y, r) \subseteq U_i$. Para $k$ grande,
$B(y_{n_k}, 2^{-n_k}) \subseteq B(y, r) \subseteq U_i$: cubierto
por \emph{uno} $U_i$ --- contradicción.
\end{proof}

\begin{definition}\label{def:b3:topology:locallycompact}
$X$ es \emph{localmente compacto}\index{espacio localmente compacto} si
es \omterm{def:b3:topology:hausdorff}{Hausdorff} y cada punto tiene un \omterm{def:b3:topology:compact}{compacto} \omterm{def:b3:topology:topology}{entorno} ($\R^n$;

\cref{exo:b3:topology:9}). Cada \omterm{def:b3:topology:compact}{espacio compacto} localmente se incrusta en
una \omterm{def:b3:topology:compact}{compacto}: la \emph{un punto
compactación}\index{compactación en un punto} $\hat X = X
\cup \{\infty\}$, cuyas aperturas son las de$X$ junto con la
complementos (en $\hat X$) de \omterm{def:b3:topology:compact}{compacto} subconjuntos de $X$. uno comprueba
los axiomas directamente; $\hat X$ es \omterm{def:b3:topology:compact}{compacto} (una portada tiene un miembro
que contiene $\infty$, cuyo complemento es \omterm{def:b3:topology:compact}{compacto}, amparado por
un número finito de otros) y \omterm{def:b3:topology:hausdorff}{Hausdorff} (separar $x$ de $\infty$
por un \omterm{def:b3:topology:compact}{compacto} \omterm{def:b3:topology:topology}{entorno} de $x$ y su complemento). Ejemplo:
$\hat\R \cong S^1$ y $\widehat{\R^n} \cong S^n$ por
proyección estereográfica (\cref{exo:b3:topology:11}).
\end{definition}

\section{Conectividad}

\begin{definition}\label{def:b3:topology:connected}
$X$ es \emph{conectado}\index{espacio conectado} si no es el
unión de dos \omterm{def:b3:topology:topology}{conjuntos abiertos} disjuntos y no vacíos --- equivalentemente, su
sólo los subconjuntos abiertos y cerrados son $\varnothing$ y $X$;
equivalentemente, cada \omterm{def:b3:topology:continuity}{aplicación continua} $X \to \{0, 1\}$ (discreto)
es constante. Un subconjunto es conexo si es un subespacio.
\end{definition}

\begin{theorem}\label{thm:b3:topology:connectedbasics}
\begin{enumerate}
  \item Los subconjuntos \omterm{def:b3:topology:connected}{conectado} de $\R$ son exactamente los intervalos.
  \item \omterm{def:b3:topology:continuity}{Continuo} las imágenes de los conjuntos \omterm{def:b3:topology:connected}{conectado} son \omterm{def:b3:topology:connected}{conectado} (de donde
        el teorema del valor intermedio: una aplicación real \omterm{def:b3:topology:continuity}{continuo} en
        a \omterm{def:b3:topology:connected}{espacio conectado} tiene un intervalo como imagen).
  \item Si $(A_i)$ son \omterm{def:b3:topology:connected}{conectado} con un punto en común, $\bigcup
        A_i$es \omterm{def:b3:topology:connected}{conectado}. Si$A$es \omterm{def:b3:topology:connected}{conectado} y$A \subseteq
        B \subseteq \bar A$, entonces$B$ es \omterm{def:b3:topology:connected}{conectado}.
  \item Los productos finitos de \omterm{def:b3:topology:connected}{espacios conectados} son \omterm{def:b3:topology:connected}{conectado}.
\end{enumerate}
\end{theorem}

\begin{proof}
En todo momento utilizamos el criterio $\{0,1\}$: $f \colon X \to
\{0,1\}$ \omterm{def:b3:topology:continuity}{continuo} debe ser constante.

(1) Un $A$ sin intervalo pierde algunos $c$ entre $a, b \in A$:
$A = (A \cap (-\infty, c)) \sqcup (A \cap (c, +\infty))$
se desconecta. Por el contrario, sea $I$ un intervalo y $f \colon I
\to \{0,1\}$\omterm{def:b3:topology:continuity}{continuo} con$f(a) = 0$,$f(b) = 1$,$a < b$.
Sea $c = \sup\{x \in [a,b] : f(x) = 0\}$; \omterm{def:b3:topology:continuity}{continuidad} en $c$
fuerzas $f(c) = 0$ (límite de valores $0$: cada \omterm{def:b3:topology:topology}{entorno} de
$c$ se encuentra con $\{f = 0\}$; $\{f=0\}$ está cerrado) y luego $c < b$
con $f \equiv 1$ en $(c, b]$, entonces $f(c) = 1$ por el mismo
Argumento \omterm{def:b3:topology:interior}{cierre} sobre $\{f = 1\} \ni c$: contradicción.

(2) A \omterm{def:b3:topology:continuity}{continuo} $g \colon f(X) \to \{0,1\}$ produce $g \circ f$
constante, por lo que $g$ es constante en la imagen.

(3) A \omterm{def:b3:topology:continuity}{continuo} $f\colon \bigcup A_i \to \{0,1\}$ es constante
en cada $A_i$, con el mismo valor en el punto común. Para el
\omterm{def:b3:topology:interior}{cierre}: $f \colon B \to \{0,1\}$ es constante $= c$ en $A$; cualquiera
$x \in B \subseteq \bar A$ está en \omterm{def:b3:topology:interior}{cierre} de $A$, y $f^{-1}
(f(x))$es \omterm{def:b3:topology:topology}{entorno} de$x$ (preimagen de abierto), que debe
conocer $A$: $f(x) = c$.

(4) Para $X \times Y$ y $f\colon X\times Y \to \{0,1\}$: cualquiera
dos puntos $(x,y), (x',y')$ están unidos por el ``codo''
$\{x\}\times Y \cup X \times \{y'\}$, una unión de dos \omterm{def:b3:topology:connected}{conectado}
conjuntos (\omterm{def:b3:topology:continuity}{homeomórfico} a $Y$, $X$) reunidos en $(x, y')$: por (3) y
(2), $f$ está de acuerdo en los dos puntos.
\end{proof}

\begin{definition}\label{def:b3:topology:pathconnected}
$X$ es \emph{conectado por camino}\index{espacio conectado por caminos} si corresponde
dos puntos están unidos por un \emph{camino} (\omterm{def:b3:topology:continuity}{continuo} $\gamma
\colon [0,1] \to X$). \omterm{def:b3:topology:connected}{Conectado por camino} implica \omterm{def:b3:topology:connected}{conectado}: dos
Los valores de \omterm{def:b3:topology:continuity}{continuo} $f \colon X \to \{0,1\}$ en $x, y$ son
valores de la constante $f\circ\gamma$
(\cref{thm:b3:topology:connectedbasics}(1)--(2)). Subconjuntos convexos
de espacios normados son \omterm{def:b3:topology:connected}{conectado por camino} (segmentos); también lo son $\R^n
\setminus \{0\}$para$n \geq 2$ (ir alrededor del origen), y
$S^n$ para $n \geq 1$ (rutas de proyecto de $\R^{n+1}\setminus
\{0\}$).
\end{definition}

\begin{example}[La curva sinusoidal del topólogo.]
\label{ex:b3:topology:sinecurve}
Sean $\Gamma = \{(x, \sin\frac1x) : 0 < x \leq 1\}$ y $S =
\bar\Gamma = \Gamma \cup (\{0\}\times[-1,1])$(cada punto$(0,
y)$,$\abs y \leq 1$, es un límite de puntos de$\Gamma$: resolver
$\sin\frac1x = y$ cerca de $0$). Entonces $S$ es
\emph{\omterm{def:b3:topology:connected}{conectado}} --- \omterm{def:b3:topology:interior}{cierre} del \omterm{def:b3:topology:connected}{conectado} $\Gamma$, un
\omterm{def:b3:topology:continuity}{continuo} imagen de $(0, 1]$
(\cref{thm:b3:topology:connectedbasics}(3)) --- pero \emph{not
\omterm{def:b3:topology:pathconnected}{conectado por camino}}: un \omterm{def:b3:topology:pathconnected}{camino} de $(1, \sin 1)$ a $(0,0)$ habría
para recorrer las abscisas $\to 0$ mientras la ordenada oscila
entre $\pm1$; \cref{exo:b3:topology:9} hace esto riguroso.
\omterm{def:b3:topology:connected}{Conectividad} y \omterm{def:b3:topology:connected}{conexión de camino} genuinamente
difieren.\index{curva sinusoidal del topólogo}
\end{example}

\begin{definition}\label{def:b3:topology:components}
El \emph{componente conectado}\index{componente conectado} de $x
\in X$es la unión de todos los subconjuntos \omterm{def:b3:topology:connected}{conectado} que contienen$x$ ---
el más grande (\cref{thm:b3:topology:connectedbasics}(3)).
Los componentes particionan $X$ y están cerrados (\omterm{def:b3:topology:interior}{cierres} de \omterm{def:b3:topology:connected}{conectado}
los conjuntos son \omterm{def:b3:topology:connected}{conectado}). $X$ es \emph{totalmente desconectado} si todos
los componentes son singletons ($\Q$; el conjunto Cantor del fin de semana
problema).
\end{definition}

\begin{theorem}\label{thm:b3:topology:rnotr2}
$\R$ es \omterm{def:b3:topology:continuity}{homeomórfico} y no $\R^n$ con $n \geq 2$.
\end{theorem}

\begin{proof}
Supongamos que $h \colon \R \to \R^n$ es un \omterm{def:b3:topology:continuity}{homeomorfismo}. Quitar un
punto: $\R \setminus \{0\}$ es \omterm{def:b3:topology:continuity}{homeomórfico} a $\R^n \setminus
\{h(0)\}$. Pero$\R\setminus\{0\}$ está desconectado, mientras
$\R^n\setminus\{\text{point}\}$ es \omterm{def:b3:topology:pathconnected}{conectado por camino} para $n \geq
2$ (\cref{def:b3:topology:pathconnected}; traduce el punto a
$0$): \omterm{def:b3:topology:connected}{conectividad} es una invariante \omterm{def:b3:topology:continuity}{homeomorfismo}
(\cref{thm:b3:topology:connectedbasics}(2)) --- contradicción.
(Ese $\R^2 \not\cong \R^3$ requiere invariantes más finos ---
\omterm{def:b3:galois:algebraic}{algebraico}\omterm{def:b3:topology:topology}{topología}; El problema del fin de semana muestra el peligro:
\emph{\omterm{def:b3:topology:continuity}{continuo} surjections} $\R \to \R^2$ existen).
\end{proof}

\begin{method}\label{met:b3:topology:connectargument}
Para demostrar que un conjunto es \omterm{def:b3:topology:connected}{conectado}: muéstrelo como una imagen \omterm{def:b3:topology:continuity}{continuo},
una unión de conjuntos \omterm{def:b3:topology:connected}{conectado} superpuestos, un \omterm{def:b3:topology:interior}{cierre} o un producto
(\cref{thm:b3:topology:connectedbasics}); para subconjuntos de normados
espacios, pruebe \omterm{def:b3:topology:connected}{conexión de camino} con \omterm{def:b3:topology:pathconnected}{caminos} explícito (segmentos,
arcos, codos). Para demostrar dos espacios son \emph{no} \omterm{def:b3:topology:continuity}{homeomórfico}:
encuentre una invariante \omterm{def:b3:topology:topology}{topológico} que difiera --- compacidad,
\omterm{def:b3:topology:connected}{conectividad}, número de componentes, o componentes después
eliminar un conjunto finito bien elegido (el truco de
\cref{thm:b3:topology:rnotr2}: también prueba $[0,1) \not\cong
(0,1)$y$S^1 \not\cong [0,1]$).
\end{method}

\section{Problema: el conjunto de Cantor y una curva que llena el espacio}

\begin{problem}[{Problema del fin de semana --- Las curvas de Peano existen y por qué
no son homeomorfismos}]\label{pb:b3:topology:1}
En 1890 Peano sorprendió al análisis con un \emph{\omterm{def:b3:topology:continuity}{continuo}
surjection} $[0,1] \to [0,1]^2$: una curva que llena un cuadrado. Nosotros
construye uno con las manos desnudas del set Cantor $C$ de
\cref{exo:b3:topology:10} (cuyos resultados podrán ser utilizados libremente),
y luego demostrar que tal aplicación no puede ser inyectiva: los cuadrados son
no curvas. En todo momento, los elementos de $C$ se escriben $x =
\sum_{n \geq 1} \frac{2b_n(x)}{3^n}$con dígitos$b_n(x) \in
\{0, 1\}$.

\medskip
\noindent\textbf{Parte I --- Reading off digits \omterm{def:b3:topology:continuity}{continuamente}.}
\begin{enumerate}
  \item Demuestre que si $x, y \in C$ satisface $\abs{x - y} <
        3^{-m}$, entonces$b_n(x) = b_n(y)$para todos los$n \leq m$.
        \emph{(If the first differing digit is $b_k$, then
        $\abs{x - y} \geq \frac{2}{3^k} - \sum_{j>k}\frac2{3^j}
        = \frac1{3^k}$.)}
  \item Deduzca que cada función de dígito $b_n \colon C \to \{0,
        1\}$es \omterm{def:b3:topology:continuity}{continuo} y vuelva a derivar el \omterm{def:b3:topology:continuity}{homeomorfismo}$C
        \cong \{0,1\}^\N$ de \cref{exo:b3:topology:10}(c).
\end{enumerate}

\medskip
\noindent\textbf{Parte II --- A \omterm{def:b3:topology:continuity}{continuo} surjection $C \to
[0,1]^2$.}
\begin{enumerate}[resume]
  \item Defina $g \colon C \to [0,1]$ por $g(x) = \sum_{n\geq1}
        b_n(x)\,2^{-n}$ (lea los dígitos de Cantor como binarios).
        Demuestre que $g$ es \omterm{def:b3:topology:continuity}{continuo} (use la pregunta 1) y
        sobreyectivo. ¿Es inyectivo?
  \item Definir $\Phi \colon C \to [0,1]^2$ por
        \[
        \Phi(x) = \Bigl(\ \sum_{n\geq1} b_{2n-1}(x)\,2^{-n},\
        \sum_{n\geq1} b_{2n}(x)\,2^{-n}\ \Bigr)
        \]
        (Los dígitos impares dan la abscisa, los dígitos pares la
        ordenada). Demuestre que $\Phi$ es \omterm{def:b3:topology:continuity}{continuo} y
        sobreyectivo.
\end{enumerate}

\medskip
\noindent\textbf{Parte III --- Filling the gaps: the Peano
curve.}
\begin{enumerate}[resume]
  \item El complemento $[0,1] \setminus C$ es una unión contable
        de intervalos abiertos disjuntos $(u, v)$ (el medio eliminado
        tercios) cuyos puntos finales se encuentran en $C$. Defina $f \colon
        [0,1] \to [0,1]^2$por$f = \Phi$en$C$, extendido
        \emph{afinmente} en cada espacio:
        \[
        f(t) = \frac{v - t}{v - u}\,\Phi(u) +
        \frac{t - u}{v - u}\,\Phi(v),
        \qquad t \in (u, v).
        \]
        Demuestre que $f$ está bien definido y es sobreyectivo sobre
        $[0,1]^2$.
  \item Demuestre que $f$ es \omterm{def:b3:topology:continuity}{continuo} en cada punto de $[0,1]
        \setminus C$ (localmente afín), y en cada punto de
        $C$: dado $\varepsilon$, elija $m$ con $2^{-m} <
        \varepsilon/2$y use la pregunta 1 para controlar$\Phi$ en
        $C$ cerca de $x$, luego verifique que el afín
        la interpolación no puede escapar: en un hueco $(u,v) \subseteq
        (x - 3^{-m}, x + 3^{-m})$, los valores$f(t)$ se encuentran en el
        segmento $[\Phi(u), \Phi(v)]$, ambos extremos cerca de
        $\Phi(x)$. Concluir: $f$ es un \emph{\omterm{def:b3:topology:continuity}{continuo}
        surjection} $[0,1] \to [0,1]^2$.
  \item Deducir \omterm{def:b3:topology:continuity}{continuo} sobrejecciones $[0,1] \to [0,1]^n$ para
        cada $n \geq 2$ y $\R \to \R^2$.
\end{enumerate}

\medskip
\noindent\textbf{Parte IV --- But never injective.}
\begin{enumerate}[resume]
  \item Mostrar que un \omterm{def:b3:topology:continuity}{continuo} \emph{inyección} $f \colon [0,1]
        \to [0,1]^2$ sería un \omterm{def:b3:topology:continuity}{homeomorfismo} en su imagen
        (\cref{cor:b3:topology:compacthomeo}).
  \item Demuestre que $[0,1]$ y $[0,1]^2$ no son \omterm{def:b3:topology:continuity}{homeomórfico}:
        eliminar un punto bien elegido y comparar \omterm{def:b3:topology:connected}{conectividad}
        (\cref{met:b3:topology:connectargument}).
  \item Conclusión: una sobreyección \omterm{def:b3:topology:continuity}{continuo} $[0,1] \to [0,1]^2$
        nunca puede ser inyectivo --- uno inyectivo haría
        $[0,1]^2 = f([0,1])$ \omterm{def:b3:topology:continuity}{homeomórfico} a $[0,1]$ por
        pregunta 8, contradiciendo la pregunta 9. ¿Dónde exactamente
        ¿La compacidad de $[0,1]$ ingresa el argumento?
  \item (Culminación) Armar la moraleja: hay un \omterm{def:b3:topology:continuity}{continuo}
        sobreyección pero no \omterm{def:b3:topology:continuity}{continuo} biyección $[0,1] \to
        [0,1]^2$. ¿Qué dice esto acerca de la "dimensión" como
        \omterm{def:b3:topology:topology}{topológico} noción? Formule precisamente un teorema
        demostrado en este problema y una afirmación plausible de que
        permanece más allá de nuestras herramientas (invariancia de dominio).
\end{enumerate}

\noindent\textbf{Part VII --- Codas: how regular, and how
far from \omterm{prop:b3:galois:perfect}{perfecto}.}
\begin{enumerate}[resume]
  \item (regularidad de Hölder de la curva) Configure $\alpha =
        \frac{\ln 2}{2\ln 3} \approx 0.3155$ y observe la
        identidad $3^\alpha = \sqrt2$. Usando la pregunta 1, muestre
        eso
        \[
        \norm{\Phi(x) - \Phi(y)}_\infty \leq 2\,\abs{x -
        y}^\alpha \qquad (x, y \in C),
        \]
        luego propagar la estimación a través de las brechas afines:
        demuestre que la curva de Peano $f$ de la pregunta 6 satisface
        $\norm{f(t) - f(s)}_\infty \leq 6\abs{t - s}^\alpha$ en
        todo $[0,1]$ \emph{(treat a pair in the same gap by
        interpolation, then a general pair by passing through
        the extreme points of $C \cap [t, s]$)}.
  \item (El exponente $\frac12$ es un muro) Demuestre que no
        la sobreyección $F \colon [0,1] \to [0,1]^2$ puede ser
        $\alpha$-Hölder con $\alpha > \frac12$: corte $[0,1]$
        en intervalos $N$, unieron los diámetros de sus
        imágenes y contar los puntos de la cuadrícula.
        $\bigl(\frac iM, \frac jM\bigr)$, $0 \leq i, j \leq M$,
        que puede contener un conjunto de diámetro $< \frac1M$; elegir
        $M$ del pedido $N^\alpha$ y deja $N \to \infty$. localizar
        nuestra curva ($\alpha \approx 0.3155$) relativa a la
        pared, y registrar sin pruebas que la curva de Hilbert
        alcanza el exponente crítico $\frac12$.
  \item (Conjuntos derivados: medición de imperfecciones) Para $F$ cerrado
        en $\R$, sea $F'$ el conjunto de puntos límite de $F$ (un
        subconjunto cerrado) e iterar: $F^{(0)} = F$, $F^{(i+1)}
        = (F^{(i)})'$. verificar que
        \[
        K = \{0\} \cup \Bigl\{\frac1m\Bigr\}_{m \geq 1} \cup
        \Bigl\{\frac1m + \frac1n : m \geq 1,\ n > m^2\Bigr\}
        \]
        es un conjunto contable \omterm{def:b3:topology:compact}{compacto} con $K' = \{0\} \cup
        \{\frac1m\}$,$K'' = \{0\}$,$K''' = \varnothing$
        \emph{(check that the $m$-th cluster lives in the
        interval $(\frac1m, \frac1{m-1})$, so the clusters do
        not interleave)}, y que sus puntos aislados son
        \omterm{def:b3:topology:interior}{denso} en $K$, como predice la pregunta 21. Describe el
        inducción produciendo, por cada $j$, un \omterm{def:b3:topology:compact}{compacto} contable

        \varnothing$, y contrastan con los conjuntos \omterm{prop:b3:galois:perfect}{perfecto} de
        pregunta 22, para la cual la derivación nunca se mueve:
        El rango finito es exactamente lo opuesto a la perfección.
\end{enumerate}
