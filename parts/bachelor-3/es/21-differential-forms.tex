\section{Álgebra multilineal alterna}

\begin{definition}\label{def:b3:forms:alternating}
Sea $E$ un espacio vectorial real de dimensión $n$. un
\emph{$k$ -forma alterna lineal}\index{forma alterna} en
$E$ es una aplicación $\alpha\colon E^k \to \R$, lineal en cada uno
variable, con $\alpha(v_1, \dots, v_k) = 0$ siempre que dos
Los argumentos son iguales. Su espacio está escrito $\Lambda^k E^*$;
por convención $\Lambda^0E^* = \R$. Fuerzas de alternancia
antisimetría: intercambiar dos argumentos cambia el signo
(expandir $\alpha(\dots, v + w, \dots, v + w, \dots) = 0$), y
más generalmente $\alpha(v_{\sigma(1)}, \dots, v_{\sigma(k)}) =
\varepsilon(\sigma)\,\alpha(v_1, \dots, v_k)$ para cada
permutación $\sigma$.
\end{definition}

\begin{example}
En $E = \R^n$: $\Lambda^1E^* = E^*$ es el \omterm{def:b3:banach:operator}{espacio dual}; el
determinante en la base canónica es un $n$-lineal
\omterm{def:b3:forms:alternating}{forma alterna} y \cref{prop:b3:forms:basis} lo mostrarán.
abarca $\Lambda^nE^*$ --- la razón profunda por la que el determinante es
Único a escala. Para $k > n$, $\Lambda^kE^* = \{0\}$:
Los vectores $k$ son dependientes y se expanden uno junto con el otro.
mata a $\alpha$ por alternancia.
\end{example}

\begin{definition}\label{def:b3:forms:wedge}
Para $\ell_1, \dots, \ell_k \in E^*$, su \emph{exteriores
producto}\index{producto exterior} es el $k$-lineal \omterm{def:b3:forms:alternating}{alternando
forma}
\[
(\ell_1 \wedge \dots \wedge \ell_k)(v_1, \dots, v_k)
= \det\bigl(\ell_i(v_j)\bigr)_{1 \leq i, j \leq k} .
\]
La alternancia y la multilinealidad son las del determinante en
sus columnas.
\end{definition}

\begin{proposition}[Base de $\Lambda^k$]\label{prop:b3:forms:basis}
Sea $(e_1, \dots, e_n)$ una base de $E$ con una base \omterm{def:b3:banach:operator}{dual}
$(e_1^*, \dots, e_n^*)$. las formas
\[
e_I^* = e_{i_1}^* \wedge \dots \wedge e_{i_k}^*,
\qquad I = \{i_1 < \dots < i_k\} \subseteq \{1, \dots, n\},
\]
formar una base de $\Lambda^kE^*$; por lo tanto $\dim\Lambda^kE^* =
\binom nk$. Explícitamente,$\alpha =
\sum_{\abs I = k}\alpha(e_{i_1}, \dots, e_{i_k})\,e_I^*$.
\end{proposition}

\begin{proof}
\emph{Generando.} Sea $\alpha \in \Lambda^kE^*$ y $\beta =
\sum_I\alpha(e_I)\,e_I^*$, donde$\alpha(e_I)$ abrevia
$\alpha(e_{i_1}, \dots, e_{i_k})$. Ambos lados son $k$-lineales
y alternándose, por lo que se ponen de acuerdo en cuanto se ponen de acuerdo en todos
$k$-tuplas $(e_{j_1}, \dots, e_{j_k})$ con $j_1 < \dots <
j_k$ (la multilinealidad se reduce a tuplas de vectores base,
alternancia a estrictamente crecientes). Y $e_I^*(e_{j_1},
\dots, e_{j_k}) = \det(e_{i_r}^*(e_{j_s})) = \delta_{IJ}$:
para $I = J$ la matriz es la identidad; para $I \neq J$ algunos
fila es cero. Entonces $\beta(e_J) = \alpha(e_J)$ para todos los $J$:
$\beta = \alpha$. \emph{Libertad.} Si $\sum_I c_Ie_I^* = 0$,
evaluar en $(e_{j_1}, \dots, e_{j_k})$ da $c_J = 0$.
\end{proof}

\begin{definition}\label{def:b3:forms:wedgegeneral}
El \omterm{def:b3:forms:wedge}{producto exterior} se extiende a una aplicación bilineal
$\Lambda^kE^* \times \Lambda^\ell E^* \to
\Lambda^{k+\ell}E^*$, determinado por bilinealidad y
$(e_I^*) \wedge (e_J^*) = e_I^* \wedge e_J^*$ (concatenar
y reordenar; el producto es $0$ si $I \cap J \neq
\varnothing$). Es asociativo y
\emph{anticonmutativo graduado}:
\[
\beta \wedge \alpha = (-1)^{k\ell}\,\alpha \wedge \beta
\qquad (\alpha \in \Lambda^k, \ \beta \in \Lambda^\ell).
\]
\end{definition}

\begin{proof}[\omnameProof]
Ambas propiedades se verifican en elementos básicos y se amplían mediante
bilinealidad. Asociatividad: ambos paréntesis de
$e_I^* \wedge e_J^* \wedge e_K^*$ igual a la cuña del
familia concatenada de formas $1$, mediante la fórmula determinante
de \cref{def:b3:forms:wedge} (Ampliación de Laplace por bloques).
La regla de los signos: moviendo cada uno de los factores $\ell$ de $\beta$
más allá de los factores $k$ de $\alpha$ cuesta un signo por adyacente
transposición (un intercambio de dos filas del determinante), por lo tanto
$(-1)^{k\ell}$ en total.
\end{proof}

\begin{proposition}[retroceso, caja lineal]
\label{prop:b3:forms:linearpullback}
Una aplicación lineal $u\colon E \to F$ induce, para cada $k$, el
aplicación lineal $u^*\colon \Lambda^kF^* \to \Lambda^kE^*$,
$(u^*\alpha)(v_1, \dots, v_k) = \alpha(u(v_1), \dots,
u(v_k))$. Satisface$u^*(\alpha \wedge \beta) = u^*\alpha
\wedge u^*\beta$y$(u \circ w)^* = w^* \circ u^*$.
Además:
\begin{enumerate}
  \item Si $\dim E = n$ y $u\colon E \to E$, entonces en el
        línea $\Lambda^nE^*$: $u^*\alpha = (\det u)\,\alpha$.
  \item Si $\operatorname{rk}u < k$, entonces $u^* = 0$ en
        $\Lambda^kF^*$.
\end{enumerate}
\end{proposition}

\begin{proof}
Las identidades funcionales son inmediatas a partir de las definiciones.
(para la regla del producto, verifique las cuñas de formas $1$ con el
fórmula determinante --- $\det(\ell_i(uv_j)) =
\det((u^*\ell_i)(v_j))$ --- luego extender bilinealmente).
(1) $u^*$ asigna el $\Lambda^nE^*$ unidimensional
(\cref{prop:b3:forms:basis}) a sí mismo, por lo que $u^*\alpha =
c\,\alpha$con$c$es independiente de$\alpha \neq 0$; prueba
en $\alpha = e_1^* \wedge \dots \wedge e_n^*$ y $(v_j) =
(e_j)$da$c = \det(e_i^*(ue_j)) = \det u$.
(2) Para $v_1, \dots, v_k \in E$, los vectores $u(v_1), \dots,
u(v_k)$se encuentran en la imagen de$u$, de dimensión$< k$: son
linealmente dependiente, y un \omterm{def:b3:forms:alternating}{forma alterna} desaparece en un
familia dependiente (expandir el vector dependiente a lo largo del
otros).
\end{proof}

\section{Formas diferenciales y la derivada exterior.}

\begin{definition}\label{def:b3:forms:diffform}
Deje que $U \subseteq \R^n$ esté abierto. Un \emph{diferencial
$k$-forma}\index{forma diferencial} en $U$ es una aplicación fluido
$\omega\colon U \to \Lambda^k(\R^n)^*$; en la base de
\cref{prop:b3:forms:basis} (con $\dd x_i$ escrito para
$e_i^*$),
\[
\omega = \sum_{\abs I = k} a_I\,\dd x_I,
\qquad
\dd x_I = \dd x_{i_1} \wedge \dots \wedge \dd x_{i_k},
\]
con coeficientes suaves $a_I \in \mathcal C^\infty(U)$.
Su espacio es $\Omega^k(U)$; $\Omega^0(U) = \mathcal
C^\infty(U)$. Una forma$0$es una función; un formulario$1$ es un campo
de formas lineales (por ejemplo, el diferencial $\dd f$ de un
función); una forma $n$ es $a\,\dd x_1\wedge\dots\wedge\dd
x_n$, el integrando natural de \cref{ch:b3:product}.
\end{definition}

\begin{definition}[exterior derivado]\label{def:b3:forms:d}
El \emph{derivado exterior}\index{derivado exterior} es
la aplicación lineal $\dd\colon \Omega^k(U) \to \Omega^{k+1}(U)$
definido por
\[
\dd\Bigl(\sum_I a_I\,\dd x_I\Bigr)
= \sum_I \dd a_I \wedge \dd x_I
= \sum_I\sum_{j=1}^n \frac{\partial a_I}{\partial
x_j}\,\dd x_j \wedge \dd x_I .
\]
En los formularios $0$ es el diferencial habitual.
\end{definition}

\begin{theorem}\label{thm:b3:forms:dsquare}
(a) $\dd(\omega \wedge \eta) = \dd\omega \wedge \eta +
(-1)^k\,\omega \wedge \dd\eta$para$\omega \in \Omega^k$
(el \emph{regla graduada de Leibniz}).
(b)$\dd \circ \dd = 0$.\index{d squared zero@$\dd^2 = 0$}
\end{theorem}

\begin{proof}
(a) Por bilinealidad basta con tratar $\omega = a\,\dd
x_I$,$\eta = b\,\dd x_J$. Luego$\omega \wedge \eta =
ab\,\dd x_I \wedge \dd x_J$ y
\[
\dd(\omega\wedge\eta) = (b\,\dd a + a\,\dd b) \wedge \dd x_I
\wedge \dd x_J
= (\dd a \wedge \dd x_I) \wedge (b\,\dd x_J) +
a\,\dd b \wedge \dd x_I \wedge \dd x_J,
\]
y mover el formato $1$ $\dd b$ más allá de los factores $k$ de $\dd
x_I$cuesta$(-1)^k$ (\cref{def:b3:forms:wedgegeneral}): el
El segundo término es $(-1)^k\,\omega \wedge \dd\eta$.
(b) Para una función: $\dd(\dd a) = \sum_{i,j}
\frac{\partial^2a}{\partial x_j\partial x_i}\,\dd x_j \wedge
\dd x_i$. El coeficiente es simétrico en$(i,j)$ por
Teorema de Schwarz sobre parciales mixtos (probado en el Año~2
volumen; $a$ es $\mathcal C^\infty$), mientras que $\dd x_j \wedge
\dd x_i$es antisimétrico: emparejando los términos$(i,j)$ y
$(j,i)$, todo se cancela. Para un $\omega = \sum
a_I\dd x_I$general:$\dd\dd\omega = \sum\dd(\dd a_I \wedge \dd x_I)
= \sum(\dd\dd a_I\wedge\dd x_I - \dd a_I\wedge\dd(\dd x_I))$
por (a), y ambos términos desaparecen ($\dd(\dd x_I) = 0$ desde el
el coeficiente es constante).
\end{proof}

\begin{definition}[retroceder]\label{def:b3:forms:pullback}
Deje que $\varphi\colon U \to V$ sea suave ($U \subseteq \R^m$,
$V \subseteq \R^n$ abierto). El \emph{obstáculo}\index{obstáculo}
$\varphi^*\colon \Omega^k(V) \to \Omega^k(U)$ está definido
puntualmente por el retroceso lineal a lo largo del diferencial:
$(\varphi^*\omega)_x = (D\varphi(x))^*\,\omega_{\varphi(x)}$.
Concretamente, $\varphi^*$ sustituye a: $\varphi^*f = f \circ
\varphi$en funciones,$\varphi^*(\dd y_i) =
\dd\varphi_i = \sum_j\frac{\partial\varphi_i}{\partial
x_j}\dd x_j$y$\varphi^*(a\,\dd y_{i_1}\wedge\dots\wedge
\dd y_{i_k}) =
(a\circ\varphi)\,\dd\varphi_{i_1}\wedge\dots\wedge
\dd\varphi_{i_k}$.
\end{definition}

\begin{theorem}\label{thm:b3:forms:pullback}
(a) $\varphi^*(\omega \wedge \eta) = \varphi^*\omega \wedge
\varphi^*\eta$y$(\psi \circ \varphi)^* = \varphi^* \circ
\psi^*$.
(b) $\varphi^*(\dd\omega) = \dd(\varphi^*\omega)$: el
\omterm{def:b3:forms:d}{derivado exterior} viaja con cada sustitución fluida
--- la identidad que lo convierte en \emph{el} derivado del
teoría.
(c) Si $\varphi\colon U \to V$ es suave entre aperturas de
$\R^n$ y $\omega = a\,\dd y_1 \wedge \dots \wedge \dd
y_n$, luego$\varphi^*\omega = (a \circ
\varphi)\,\det\bigl(D\varphi\bigr)\,\dd x_1 \wedge \dots
\wedge \dd x_n$.
\end{theorem}

\begin{proof}
(a) Declaraciones puntuales sobre \omterm{def:b3:forms:pullback}{retrocesos} lineal
(\cref{prop:b3:forms:linearpullback}), más la regla de la cadena
$D(\psi\circ\varphi)(x) = D\psi(\varphi(x))\,D\varphi(x)$.
(b) Para un formulario $0$ $f$: $\varphi^*(\dd f) = \dd f \circ
D\varphi = \dd(f \circ \varphi)$ es la regla de la cadena. Para
$\omega = a\,\dd y_I$: usando (a), $\varphi^*\omega =
(a\circ\varphi)\,\dd\varphi_{i_1}\wedge\dots\wedge
\dd\varphi_{i_k}$, por lo que según la regla de Leibniz
(\cref{thm:b3:forms:dsquare}(a)) y $\dd\dd\varphi_{i_r} =
0$:
\[
\dd(\varphi^*\omega) = \dd(a\circ\varphi) \wedge
\dd\varphi_{i_1}\wedge\dots\wedge\dd\varphi_{i_k}
= \varphi^*(\dd a) \wedge \varphi^*(\dd y_I)
= \varphi^*(\dd a \wedge \dd y_I) =
\varphi^*(\dd\omega).
\]
(c) Puntualmente esto es exactamente $u^*\alpha = (\det u)\alpha$
en formularios de primer grado (\cref{prop:b3:forms:linearpullback}(1)
con $u = D\varphi(x)$).
\end{proof}

\begin{example}[Coordenadas polares]\label{ex:b3:forms:polar}
Para $\varphi(r, \theta) = (r\cos\theta, r\sin\theta)$:
$\varphi^*\dd x = \cos\theta\,\dd r -
r\sin\theta\,\dd\theta$,$\varphi^*\dd y = \sin\theta\,\dd r
+ r\cos\theta\,\dd\theta$, entonces
\[
\varphi^*(\dd x \wedge \dd y) =
(\cos\theta\,\dd r - r\sin\theta\,\dd\theta) \wedge
(\sin\theta\,\dd r + r\cos\theta\,\dd\theta)
= r\,\dd r \wedge \dd\theta,
\]
el jacobiano de \cref{ex:b3:product:polar} apareciendo por pura
álgebra --- no teoría \omterm{def:b3:measure:measure}{medida}. \cref{exo:b3:forms:9} vueltas
esta observación en una declaración: para integrales orientadas, la
fórmula de cambio de variables \emph{is} la fórmula \omterm{def:b3:forms:pullback}{obstáculo}.
\end{example}

\begin{definition}\label{def:b3:forms:closedexact}
$\omega \in \Omega^k(U)$ es \emph{cerrado}\index{forma cerrada}
si $\dd\omega = 0$, \emph{exacto}\index{forma exacta} si
$\omega = \dd\eta$ para algunos $\eta \in \Omega^{k-1}(U)$ (un
\emph{primitivo} de $\omega$). Exacto $\Rightarrow$ cerrado por
$\dd^2 = 0$; lo contrario es una pregunta sobre el
\emph{forma} de $U$.
\end{definition}

En $U = \R^2 \setminus \{0\}$,

\omega_\theta = \frac{x\,\dd y - y\,\dd x}{x^2 + y^2}
\]
es \omterm{def:b3:forms:closedexact}{cerrado} (cálculo directo:
\cref{exo:b3:forms:4}) pero no \omterm{def:b3:forms:closedexact}{exacto}: es integral a lo largo del
El círculo unitario es $2\pi \neq 0$, mientras que las integrales de \omterm{def:b3:forms:closedexact}{formas exactas}
a lo largo de \omterm{def:b3:forms:closedexact}{cerrado} las curvas desaparecen (\cref{prop:b3:forms:exactloop}).
Localmente, $\omega_\theta = \dd\theta$ para cualquier fluidez.
determinación $\theta$ del ángulo polar --- de donde
nombre y la obstrucción: no existe tal determinación en
todo $U$. Este formulario único ejecuta el \omterm{def:b3:forms:winding}{número de liquidación}

\end{example}

\begin{theorem}[Poincar'e lema]\label{thm:b3:forms:poincare}
Dejemos que $U \subseteq \R^n$ esté abierto y \emph{en forma de estrella} con
respecto a $0$. Cada formulario \omterm{def:b3:forms:closedexact}{cerrado} $k$ en $U$ ($k \geq 1$) es
\omterm{def:b3:forms:closedexact}{exacto}.\index{Poincaré lemma}
\end{theorem}

\begin{proof}
Construimos un \emph{operador de homotopía} $h\colon
\Omega^k(U) \to \Omega^{k-1}(U)$ lineal con
\begin{equation}\label{eq:b3:forms:homotopy}
\dd(h\omega) + h(\dd\omega) = \omega
\qquad (k \geq 1);
\end{equation}
si $\dd\omega = 0$, entonces $\omega = \dd(h\omega)$ y estamos
hecho. Para juego $\omega = \sum_I a_I\,\dd x_I$
\[
h\omega = \sum_{\abs I = k}\ \sum_{r=1}^{k}(-1)^{r-1}
\Bigl(\int_0^1 t^{k-1}a_I(tx)\,\dd t\Bigr)\,
x_{i_r}\,\dd x_{i_1}\wedge\dots\wedge
\widehat{\dd x_{i_r}}\wedge\dots\wedge\dd x_{i_k}
\]
(el sombrero elimina un factor; las integrales son suaves en $x$
por diferenciación bajo la integral,
\cref{thm:b3:lebesgue:paramdiff}, siendo todos los derivados
dominado en \omterm{def:b3:topology:compact}{compactos}). comprobando
\eqref{eq:b3:forms:homotopy} es un cálculo que se realiza una vez al
vida, por eso lo hacemos en su totalidad. Repare $I$ y tome $\omega =
a\,\dd x_I$ (linealidad). Primero,
\[
\dd(h\omega) = k\Bigl(\int_0^1t^{k-1}a(tx)\dd t\Bigr)\dd x_I
+ \sum_{r=1}^k(-1)^{r-1}\sum_{j=1}^n
\Bigl(\int_0^1t^{k}\,\partial_ja(tx)\,\dd t\Bigr)
x_{i_r}\,\dd x_j\wedge\dd x_{I\setminus i_r} :
\]
el primer grupo recoge los términos en los que $\dd$ llega al
factor $x_{i_r}$ --- la cuña $\dd x_{i_r}\wedge\dd
x_{I\setminus i_r}$vuelve a ensamblar$\dd x_I$ con un signo
$(-1)^{r-1}$ que cancela el prefactor, y los valores de $k$
de $r$ dan el factor $k$ --- mientras que el segundo grupo
recoge los términos donde $\dd$ llega a la integral (cadena
la regla muestra $t\,\partial_ja(tx)$). A continuación, $\dd\omega =
\sum_j\partial_ja\,\dd x_j\wedge\dd x_I$, y aplicando el
definición de $h$ en el grado $k+1$, ocupando el índice $j$
la primera ranura:
\[
h(\dd\omega) = \sum_{j=1}^n\Bigl(\int_0^1t^{k}
\partial_ja(tx)\dd t\Bigr)x_j\,\dd x_I
- \sum_{j=1}^n\sum_{r=1}^k(-1)^{r-1}
\Bigl(\int_0^1t^{k}\partial_ja(tx)\dd t\Bigr)
x_{i_r}\,\dd x_j\wedge\dd x_{I\setminus i_r} .
\]
Las sumas dobles se cancelan en $\dd(h\omega) + h(\dd\omega)$,
que por lo tanto es igual
\[
\Bigl(\int_0^1\bigl(k\,t^{k-1}a(tx) +
t^k{\textstyle\sum_j}x_j\,\partial_ja(tx)\bigr)\dd t\Bigr)
\dd x_I
= \Bigl(\int_0^1\frac{\dd}{\dd t}\bigl(t^ka(tx)\bigr)\dd
t\Bigr)\dd x_I
= a(x)\,\dd x_I = \omega,
\]
por el teorema fundamental del cálculo. forma de estrella
ingresó donde tenía que hacerlo: $tx \in U$ para $t \in \intcc01$,
para que $a(tx)$ tenga sentido.
\end{proof}

\section{Orientación e integración en subvariedades.}

La integración de un formulario $k$ requiere orientación tridimensional $k$
territorio. Recuperar de \cref{ch:b3:submanifolds}
(\cref{thm:b3:submanifolds:characterizations}) que un
$k$-subcolector $M \subseteq \R^n$ es localmente la imagen de un
parametrización regular $\gamma\colon V \to M \cap W$ ($V
\subseteq \R^k$abierto,$\gamma$ a \omterm{def:b3:topology:continuity}{homeomorfismo} en su imagen
con diferencial inyectivo).

\begin{definition}\label{def:b3:forms:orientation}
Un \emph{orientación}\index{orientación} de $M$ es una opción,
para cada $p \in M$, de una de las dos clases de orientación de
bases del \omterm{def:b3:submanifolds:tangent}{espacio tangente} $T_pM$, que es \emph{localmente
coherente}: alrededor de cada punto hay una parametrización
$\gamma$ cuyo marco de coordenadas $(\partial_1\gamma, \dots,
\partial_k\gamma)$ está orientado positivamente en cada punto de
su dominio. Estas parametrizaciones se denominan \emph{directo}.
$M$ es \emph{orientable} si existe una orientación; el
La banda Möbius muestra que esto puede fallar. Todas las subvariedades en este
Los capítulos están orientados.
\end{definition}

\begin{definition}[Integral de un formulario]\label{def:b3:forms:integral}
Sea $M$ una subvariedad $k$ orientada y $\omega$ una
Formulario $k$ definido en un \omterm{def:b3:topology:topology}{entorno} de $M$, con
$\operatorname{supp}\omega \cap M$ \omterm{def:b3:topology:compact}{compacto}.
(a) Si $\operatorname{supp}\omega \cap M \subseteq
\gamma(V)$ para una única parametrización directa, establezca
\[
\int_M\omega = \int_V\gamma^*\omega
\]
--- el lado derecho es la integral de Lebesgue sobre $V$
(\cref{ch:b3:product}) del coeficiente $g$ de
$\gamma^*\omega = g\,\dd u_1\wedge\dots\wedge\dd u_k$, que
es \omterm{def:b3:topology:continuity}{continuo} con soporte \omterm{def:b3:topology:compact}{compacto}.
(b) En general, elija un número finito de parametrizaciones directas
$\gamma_i(V_i)$ cubriendo el \omterm{def:b3:topology:compact}{compacto}
$\operatorname{supp}\omega \cap M$ y un subordinado
\omterm{lem:b3:forms:partition}{partición de unidad}$(\chi_i)$
(\cref{lem:b3:forms:partition}) y establezca $\int_M\omega =
\sum_i\int_M\chi_i\,\omega$, cada término calculado por (a).
Para una curva ($k = 1$) parametrizada por
$\gamma\colon\intcc ab\to\R^n$ escribimos
$\int_\gamma\omega = \int_a^b\gamma^*\omega$, sin inyectividad
requerido.
\end{definition}

\begin{lemma}[constancia]\label{lem:b3:forms:consistency}
La definición (a) no depende de la directa
parametrización, y la definición (b) no depende ni de la
cubierta ni en el \omterm{lem:b3:forms:partition}{partición de unidad}. Es más, si
$\Phi$ es un difeomorfismo de \omterm{def:b3:topology:topology}{barrios} de dos orientados
subcolectores con $\Phi(M) = M'$, que llevan un marco directo de
$M$ a una trama directa de $M'$ en algún punto de cada
componente de $M$, luego $\int_{M'}\omega =
\int_M\Phi^*\omega$.
\end{lemma}

\begin{proof}
(a) Sean $\gamma\colon V \to M$, $\delta\colon V' \to M$
parametrizaciones directas cuyas imágenes contienen
$\operatorname{supp}\omega\cap M$. La transición $\tau =
\delta^{-1}\circ\gamma$ es un difeomorfismo entre la
subconjuntos abiertos relevantes de $V, V'$ (suavidad de las transiciones:
\cref{thm:b3:submanifolds:characterizations}, vía local
descripción del gráfico), y $\gamma = \delta\circ\tau$ allí, por lo que
$\gamma^*\omega = \tau^*(\delta^*\omega)$
(\cref{thm:b3:forms:pullback}(a)). Escribe $\delta^*\omega =
g\,\dd u_1\wedge\dots\wedge\dd u_k$; entonces
(\cref{thm:b3:forms:pullback}(c)) $\tau^*(\delta^*\omega) =
(g\circ\tau)\,\det(D\tau)\,\dd u_1\wedge\dots\wedge\dd u_k$.
Al ser ambos fotogramas directos, $D\tau$ asigna una base positiva a una
base positiva: $\det D\tau > 0$, entonces $\det D\tau =
\abs{\det D\tau}$ y el teorema de cambio de variables
(\cref{thm:b3:product:changeofvar}) da
$\int(g\circ\tau)\abs{\det D\tau} = \int g$: los dos
las integrales concuerdan. \emph{Here is the whole reason \omterm{def:b3:forms:orientation}{orientación}
exists}: sin el control de signos, el jacobiano y su
los valores absolutos difieren y la integral está mal definida.
(b) Si $(\chi_i)$ y $(\tilde\chi_j)$ son dos admisibles
particiones (cubiertas incluidas), luego por (a) y finitas
aditividad, $\sum_i\int\chi_i\omega =
\sum_{i,j}\int\chi_i\tilde\chi_j\omega =
\sum_j\int\tilde\chi_j\omega$, cada término doble computable
en cualquiera de los gráficos. La última declaración: si $\gamma$ se desborda
parametrizaciones directas de $M$, luego se ejecuta $\Phi\circ\gamma$
sobre parametrizaciones directas de $M'$ (el \omterm{def:b3:forms:orientation}{orientación}
comparación es localmente constante y fija en un punto por
componente) y $(\Phi\circ\gamma)^*\omega =
\gamma^*(\Phi^*\omega)$.
\end{proof}

$K$.\index{partición de unidad}
\end{lemma}

Cada $x \in K$ se encuentra en algún $W_{i(x)}$ con una bola \omterm{def:b3:forms:closedexact}{cerrado}
$\bar B(x, 2r_x) \subseteq W_{i(x)}$; extractos de compacidad
$x_1, \dots, x_N$ con las bolas $B(x_s, r_{x_s})$ cubriendo
$K$. Para cada $s$, realice un golpe $\theta_s \in \mathcal
C^\infty$,$0 \leq \theta_s \leq 1$,$\theta_s = 1$ en
$\bar B(x_s, r_{x_s})$, $\operatorname{supp}\theta_s
\subseteq B(x_s, 2r_{x_s})$ (apaciguar el indicador del
bola de radio $\frac32r_{x_s}$,
\cref{thm:b3:lp:regularization}). Asigne cada $s$ a uno
índice $i(s)$ con $B(x_s, 2r_{x_s}) \subseteq W_{i(s)}$ y
establezca $\Theta_i = \sum_{i(s) = i}\theta_s$. En el \omterm{def:b3:topology:topology}{conjunto abierto}
$\Omega_0 = \{\sum_j\Theta_j > \tfrac12\} \supseteq K$, el
Las funciones $\Theta_i/\sum_j\Theta_j$ hacen el trabajo pero son
definido sólo allí; para globalizar, deje que $\rho \in \mathcal
C^\infty(\R^n)$satisfaga$\rho = 0$donde$\sum_j\Theta_j
\geq 1$y$\rho > 0$donde$\sum_j\Theta_j \leq \tfrac12$

\begin{definition}\label{def:b3:forms:boundary}
A \emph{$k$-subvariedad con límite}\index{subcolector con
límite} $M \subseteq \R^n$ es un conjunto cubierto por
parametrizaciones de dos tipos: \emph{\omterm{def:b3:topology:interior}{interior} charts}
$\gamma\colon V \to M \cap W$ con $V \subseteq \R^k$ abierto,
y \emph{gráficos de límites} $\gamma\colon V \cap H^k \to M
\cap W$, donde$H^k = \{u \in \R^k : u_k \geq 0\}$ y
$\gamma$ se extiende suave y regularmente al abierto $V$.
El \emph{límite} $\partial M$ es el conjunto de puntos
alcanzado en $u_k = 0$; es un subcolector $(k-1)$ sin
límite, parametrizado por las aplicaciones $u' \mapsto \gamma(u',
0)$. Un \omterm{def:b3:forms:orientation}{orientación} de$M$induce a uno en$\partial M$ por
la regla \emph{afuera-normal-primero}: en $p \in \partial M$,
una base $(w_1, \dots, w_{k-1})$ de $T_p\partial M$ es
positivo si $(\nu, w_1, \dots, w_{k-1})$ es positivo
base de $T_pM$, donde $\nu \in T_pM \setminus T_p\partial M$
señala $M$ (en un gráfico de límites: $\nu =
-\partial_k\gamma$, hasta agregar componentes tangenciales ---
la clase \omterm{def:b3:forms:orientation}{orientación} no los ve).
\end{definition}

\begin{lemma}[Aviva el medio espacio]
\label{lem:b3:forms:halfspace}
Deje que $\eta$ sea una forma $(k-1)$ fluida en $\R^k$ con \omterm{def:b3:topology:compact}{compacto}
apoyo. entonces
\[
\int_{H^k}\dd\eta = \int_{\partial H^k}\eta,
\]
donde $H^k$ lleva el estándar \omterm{def:b3:forms:orientation}{orientación} de $\R^k$ y
$\partial H^k = \{u_k = 0\} \cong \R^{k-1}$ el inducido,
que es $(-1)^k$ veces el estándar \omterm{def:b3:forms:orientation}{orientación} de
$\R^{k-1}$.
\end{lemma}

\begin{proof}
Primero la contabilidad \omterm{def:b3:forms:orientation}{orientación}: la normalidad exterior a lo largo
$\partial H^k$ es $-e_k$, y
\[
\det(-e_k, e_1, \dots, e_{k-1}) =
-\det(e_k, e_1, \dots, e_{k-1}) = -(-1)^{k-1}\det(e_1,
\dots, e_k) = (-1)^k :
\]
el marco $(e_1, \dots, e_{k-1})$ de $\partial H^k$ es
positivo para el \omterm{def:b3:forms:orientation}{orientación} inducido exactamente cuando $k$ es
Incluso, de ahí la comparación indicada. Por linealidad, tome $\eta
= f\,\dd u_1\wedge\dots\wedge\widehat{\dd
u_i}\wedge\dots\wedge\dd u_k$,$f \in \mathcal
C^\infty_c(\R^k)$; entonces$\dd\eta =
(-1)^{i-1}\,\partial_if\,\dd u_1\wedge\dots\wedge\dd u_k$
(mover $\dd u_i$ a su ranura cuesta intercambios $i - 1$). dos
casos, ambos de Tonelli--Fubini
(\cref{thm:b3:product:fubini}) y el de una variable
teorema fundamental del cálculo.

\emph{Caso $i < k$.} Integrando primero en $u_i$ sobre $\R$:
$\int_\R\partial_if\,\dd u_i = 0$ (compatible con \omterm{def:b3:topology:compact}{compacto}), por lo que
$\int_{H^k}\dd\eta = 0$. Y la restricción de $\eta$ a
$\{u_k = 0\}$ contiene el factor $\dd u_k$, que restringe
a $0$ ($u_k$ es constante allí): $\int_{\partial H^k}\eta =
0$ también.

\emph{Caso $i = k$.} Integrando primero en $u_k$ terminado
$\intco0\infty$:
\begin{align*}
\int_{H^k}\dd\eta &= (-1)^{k-1}\int_{\R^{k-1}}
\Bigl(\int_0^\infty\partial_kf\,\dd u_k\Bigr)\dd u' \\
&= (-1)^{k-1}\int_{\R^{k-1}}\bigl(0 - f(u', 0)\bigr)\dd u'
= (-1)^k\int_{\R^{k-1}}f(u',0)\,\dd u' .
\end{align*}
En el lado \omterm{def:b3:forms:boundary}{límite}, $\eta$ se restringe a $f(u',
0)\,\dd u_1\wedge\dots\wedge\dd u_{k-1}$ y el inducido
\omterm{def:b3:forms:orientation}{orientación} siendo $(-1)^k$ veces el estándar,
$\int_{\partial H^k}\eta =
(-1)^k\int_{\R^{k-1}}f(u',0)\,\dd u'$. Las dos partes están de acuerdo.
\end{proof}

\begin{theorem}[alimenta]\label{thm:b3:forms:stokes}
Sea $M \subseteq \R^n$ una subcolectora $k$ orientada a \omterm{def:b3:topology:compact}{compacto}
con \omterm{def:b3:forms:boundary}{límite}, $\partial M$ llevando el inducido
\omterm{def:b3:forms:orientation}{orientación}, y deje que $\omega$ sea una forma $(k-1)$ fluida en un
\omterm{def:b3:topology:topology}{entorno} de $M$. Entonces\index{teorema de stokes}
\[
\int_M\dd\omega = \int_{\partial M}\omega .
\]
En particular, si $\partial M = \varnothing$:
$\int_M\dd\omega = 0$.
\end{theorem}

\begin{proof}
Cubra el \omterm{def:b3:topology:compact}{compacto} $M$ con un número finito de imágenes de directo
gráficos (\omterm{def:b3:topology:interior}{interior} o \omterm{def:b3:forms:boundary}{límite}) y tome un \omterm{lem:b3:forms:partition}{partición de unidad}
$(\chi_i)$ subordinado al correspondiente \omterm{def:b3:topology:topology}{conjuntos abiertos} $W_i$
de $\R^n$ (\cref{lem:b3:forms:partition}), con $\sum\chi_i
= 1$en un \omterm{def:b3:topology:topology}{entorno} de$M$. Sobre eso \omterm{def:b3:topology:topology}{entorno}
$\dd(\sum\chi_i) = 0$, entonces
\[
\int_M\dd\omega = \sum_i\int_M\dd(\chi_i\omega),
\qquad
\int_{\partial M}\omega =
\sum_i\int_{\partial M}\chi_i\omega :
\]
ambos lados son aditivos y basta con demostrar la
Teorema para una forma compatible con una sola imagen de gráfico.

\emph{\omterm{def:b3:topology:interior}{Interior} chart.} Si $\operatorname{supp}\omega \cap M
\subseteq \gamma(V)$con$V$abierto en$\R^k$: extienda$\eta =
\gamma^*\omega$por cero a$\R^k$ (suave, soporte \omterm{def:b3:topology:compact}{compacto}
en $V$) y aplicar \cref{lem:b3:forms:halfspace} con el
soporte fuera de $\partial H^k$ (traducir $V$ al
abra el medio espacio superior --- o simplemente repita el caso $i < k$
cálculo sobre todo $\R^k$): $\int_M\dd\omega =
\int_{\R^k}\dd\eta = 0$y$\int_{\partial M}\omega = 0$
ya que $\omega$ desaparece cerca de $\partial M$.

\emph{\omterm{def:b3:forms:boundary}{Límite} chart.} Si $\operatorname{supp}\omega\cap M
\subseteq \gamma(V \cap H^k)$: con$\eta = \gamma^*\omega$
extendido por cero,
$\gamma^*(\dd\omega) = \dd\eta$
(\cref{thm:b3:forms:pullback}(b)), por lo que mediante
\cref{def:b3:forms:integral}:
\[
\int_M\dd\omega = \int_{H^k}\dd\eta
\overset{\text{\cref{lem:b3:forms:halfspace}}}{=}
\int_{\partial H^k}\eta .
\]
Falta identificar el lado derecho con $\int_{\partial
M}\omega$. El \omterm{def:b3:forms:boundary}{límite} está parametrizado por$\beta(u') =
\gamma(u', 0)$y$\beta^*\omega$ es la restricción de
$\eta$ a $\{u_k = 0\}$ (\omterm{def:b3:forms:pullback}{obstáculo} bajo la inclusión $u'
\mapsto (u', 0)$compuesta con$\gamma$). El \omterm{def:b3:forms:orientation}{orientación}
la comparación es la misma $(-1)^k$ en ambos lados: el marco
$(\partial_1\beta, \dots, \partial_{k-1}\beta)$ se sienta en el
inducido \omterm{def:b3:forms:orientation}{orientación} de $\partial M$ con el signo
$\det(-e_k, e_1, \dots, e_{k-1}) = (-1)^k$ calculado en el
gráfico (el vector de salida retrocede a $-e_k$), que es
exactamente el signo que relaciona la inducción de $\partial H^k$
\omterm{def:b3:forms:orientation}{orientación} al estándar $\R^{k-1}$
(\cref{lem:b3:forms:halfspace}). Las dos convenciones de signos.
cancelar:
$\int_{\partial H^k}\eta = \int_{\partial M}\omega$.
\end{proof}

\begin{example}[Los teoremas clásicos]
\label{ex:b3:forms:classical}
Deje que $D \subseteq \R^2$ sea un dominio \omterm{def:b3:topology:compact}{compacto} con fluidez
\omterm{def:b3:forms:boundary}{límite}, orientación estándar. Para $\omega = P\,\dd x +
Q\,\dd y$:$\dd\omega = \bigl(\partial_xQ -
\partial_yP\bigr)\dd x\wedge\dd y$ y Stokes lee
\[
\int_D\Bigl(\frac{\partial Q}{\partial x} -
\frac{\partial P}{\partial y}\Bigr)\dd x\,\dd y
= \oint_{\partial D}P\,\dd x + Q\,\dd y :
\]
\emph{Verde--Riemann}, probado para dominios elementales en el
Volumen del año ~2 y ahora en generalidad natural. En $\R^3$,
Stokes aplicados a la forma de flujo $2$ de un campo vectorial en un
El dominio \omterm{def:b3:topology:compact}{compacto} da el \emph{teorema de divergencia}
$\int_\Omega\operatorname{div}F =
\int_{\partial\Omega}\langle F, \nu\rangle\,\dd S$, y
aplicado a un formulario $1$ en una superficie con límite, el
teorema de rizo clásico \emph{Kelvin--Stokes};
\cref{exo:b3:forms:7} detalla ambos diccionarios.
\end{example}

\section{El número de liquidación}\label{sec:b3:forms:winding}

\begin{definition}\label{def:b3:forms:winding}
Deja que $\gamma\colon\intcc01\to\R^2\setminus\{a\}$ sea suave
\omterm{def:b3:forms:closedexact}{cerrado} curva. Su \emph{número de liquidación}\index{bobinado
numero} alrededor de $a$ es
\[
\operatorname{Ind}_\gamma(a) =
\frac1{2\pi}\int_\gamma\omega_\theta^a,
\qquad
\omega_\theta^a = \frac{(x - a_1)\,\dd y - (y - a_2)\,\dd
x}{(x - a_1)^2 + (y - a_2)^2} .
\]
\end{definition}

\begin{proposition}\label{prop:b3:forms:windinginteger}
$\operatorname{Ind}_\gamma(a) \in \Z$; en función de $a$
es constante en cada \omterm{def:b3:topology:components}{componente conectado} de
$\R^2\setminus\gamma(\intcc01)$ y cero en lo ilimitado
componente. Para $\gamma(t) = a + r(\cos2\pi t, \sin2\pi t)$:
$\operatorname{Ind}_\gamma(a) = 1$.
\end{proposition}

\begin{proof}
Tome $a = 0$ y escriba $\gamma = (x, y)$, $\rho =
\norm\gamma > 0$,$\vartheta(t) =
\int_0^t\gamma^*\omega_\theta$, de modo que$\vartheta' =
\frac{xy' - yx'}{x^2 + y^2}$. En notación compleja, sea$u(t)
= \gamma(t)\,\rho(t)^{-1}\eu^{-\iu\vartheta(t)}$; entonces
$\abs u = 1$ y un cálculo directo da
\[
\frac{u'}{u} = \frac{\gamma'}{\gamma} - \frac{\rho'}{\rho}
- \iu\vartheta'
= \frac{\bar\gamma\gamma'}{\abs\gamma^2} -
\frac{\rho'}{\rho} - \iu\,\frac{xy' -
yx'}{\rho^2}
= \frac{(xx' + yy')}{\rho^2} - \frac{\rho'}{\rho} = 0,
\]
desde $\bar\gamma\gamma' = (xx' + yy') + \iu(xy' - yx')$ y
$\rho\rho' = xx' + yy'$. Entonces $u$ es constante: $\gamma(t) =
\rho(t)\,c\,\eu^{\iu\vartheta(t)}$con$\abs c = 1$, y
$\gamma(1) = \gamma(0)$ con fuerzas $\rho(1) = \rho(0)$
$\eu^{\iu\vartheta(1)} = \eu^{\iu\vartheta(0)} = 1$:
$\vartheta(1) \in 2\pi\Z$, es decir\
$\operatorname{Ind}_\gamma(0) \in \Z$. En función de $a$
en el complemento abierto de la curva \omterm{def:b3:topology:compact}{compacto}, la definición
integral es \omterm{def:b3:topology:continuity}{continuo} (\cref{thm:b3:lebesgue:paramcont},
dominación sobre un \omterm{def:b3:topology:topology}{entorno} de cada $a$); un \omterm{def:b3:topology:continuity}{continuo}
la función de valor entero es localmente constante, por lo tanto constante
sobre componentes. Para $\norm a$ grande el integrando es
$O(1/\norm a)$ uniformemente en $t$, por lo que el índice tiende a $0$
y desaparece en el componente ilimitado. Para el círculo:
$\gamma^*\omega_\theta^a = 2\pi\,\dd t$ directamente.
\end{proof}

\begin{remark}
Bajo $\C \cong \R^2$, $\frac{\dd z}{z} = \frac{x\,\dd x +
y\,\dd y}{x^2 + y^2} + \iu\,\omega_\theta$, entonces
$\operatorname{Ind}_\gamma(a) =
\frac1{2\iu\pi}\oint_\gamma\frac{\dd z}{z - a}$: este es el
índice de \cref{ch:b3:residues}, y
\cref{prop:b3:forms:windinginteger} vuelve a demostrar su
integralidad y constancia local por medios de variable real ---
la mitad \omterm{def:b3:topology:topology}{topológico} del teorema \omterm{def:b3:residues:singularities}{residuo}, que ahora se encuentra en
Alimenta.
\end{remark}

\begin{proposition}\label{prop:b3:forms:exactloop}
Si $\omega = \dd f$ es \omterm{def:b3:forms:closedexact}{exacto} en el $U$ abierto y
$\gamma\colon\intcc01\to U$ es una curva \omterm{def:b3:forms:closedexact}{cerrado}, entonces
$\int_\gamma\omega = 0$.
\end{proposition}

\begin{proof}
$\int_\gamma\dd f = \int_0^1(f\circ\gamma)'(t)\,\dd t =
f(\gamma(1)) - f(\gamma(0)) = 0$ --- la regla de la cadena
identifica $\gamma^*(\dd f)$ con $(f\circ\gamma)'\,\dd t$.
\end{proof}

\begin{method}\label{met:b3:forms:method}
Computación con formas: (1) mecanizar --- cuñas reordenar con
signos, $\dd$ diferencia los coeficientes en $\dd
x_j$ nuevos, sustituto \omterm{def:b3:forms:pullback}{retrocesos}; confía en el álgebra, codifica
cada jacobiano. (2) Para integrar una forma sobre una subvariedad:
parametrizar directamente, retroceder, integrar el coeficiente;
\omterm{def:b3:forms:orientation}{orientación} es la única trampa: comprueba un fotograma. (3) Para
probar una identidad integral, buscar una forma de Stokes: ¿es la
integrando \omterm{def:b3:forms:closedexact}{exacto}? ¿El dominio es \omterm{def:b3:forms:boundary}{límite}? (4) Para comparar
integrales sobre dos subvariedades "paralelas", aplique Stokes
a la región entre ellos (el argumento de la deformación,
\cref{exo:b3:forms:10}). (5) Una integral distinta de cero de \omterm{def:b3:forms:closedexact}{cerrado
forma} certifica una obstrucción \omterm{def:b3:topology:topology}{topológico} --- sin primitiva,
Sin retracción, sin extensión libre de cero: así es como el
El problema del fin de semana acaba con las retracciones del balón.
\end{method}

\section{Problema: teorema del punto fijo de Brouwer}

\begin{problem}[{Problema de fin de semana --- sin retractación, sin
escapar}]\label{pb:b3:forms:1}
El teorema de Brouwer establece que \emph{every \omterm{def:b3:topology:continuity}{aplicación continua} of
the \omterm{def:b3:forms:closedexact}{cerrado} unit ball $\bar B = \bar B^n \subseteq \R^n$ into
itself has a fixed point} --- uno de los grandes de las matemáticas.
teoremas, con consecuencias de la teoría de juegos (Nash
equilibrios) al análisis matricial. La prueba de forma diferencial
es el más limpio conocido: Stokes muestra que la esfera no es una
retrae la pelota y todo sigue. A lo largo,
$S = S^{n-1} = \partial\bar B$, $n \geq 2$ y
\[
\sigma = \sum_{i=1}^n(-1)^{i-1}x_i\,
\dd x_1\wedge\dots\wedge\widehat{\dd x_i}\wedge\dots\wedge
\dd x_n
\]
(el sombrero borra el factor).

\medskip
\noindent\textbf{Parte I --- The measuring instrument.}
\begin{enumerate}
  \item Calcular $\dd\sigma$ y deducir de Stokes
        (\cref{thm:b3:forms:stokes}) que $\int_S\sigma =
        n\operatorname{vol}(\bar B) > 0$, donde$S$ lleva
        el \omterm{def:b3:forms:boundary}{límite} \omterm{def:b3:forms:orientation}{orientación} de la pelota.
  \item Para $n = 2$ y $n = 3$, identifique la restricción de
        $\sigma$ a $S$ con las formas de longitud de arco y área
        (\cref{exo:b3:forms:5} con $\nu(x) = x$) y
        Vuelva a calcular $\int_S\sigma$ directamente.
  \item Deje que $W \subseteq \R^N$ esté abierto y $\varphi\colon W
        \to \R^n$suave con$\norm{\varphi(x)} = 1$ para
        todos $x \in W$. mostrar eso
        $\varphi^*(\dd\sigma) = 0$. \emph{(Differentiate
        $\norm\varphi^2 = 1$: the image of $D\varphi(x)$
        lies in the hyperplane $\varphi(x)^\perp$, of
        dimension $n - 1$; then apply
        \cref{prop:b3:forms:linearpullback}(2).)}
  \item ¿Dónde está el defecto en la siguiente "prueba" de que
        $\int_S\sigma = 0$: ``$S$ es \omterm{def:b3:topology:compact}{compacto} sin
        \omterm{def:b3:forms:boundary}{límite} y $\sigma$ restringido a $S$ es un top
        formulario en él, de ahí \omterm{def:b3:forms:closedexact}{cerrado}, de ahí $\int_S\sigma =
        \int_S\dd(\text{something}) = 0$ de Stokes''?
        (Señale la palabra equivocada).
  \item Explique en un párrafo la estrategia de la Parte II:
        qué se integrará, sobre qué y dónde
        vendrá la contradicción.
\end{enumerate}

\medskip
\noindent\textbf{Parte II --- No smooth retraction.} Supongamos,
por contradicción, que $r$ es un \emph{retracción suave} de
la bola sobre su esfera: $r$ es suave en un \omterm{def:b3:topology:topology}{entorno} de
$\bar B$, $r(\bar B) \subseteq S$ y $r(x) = x$ para todos
$x \in S$.
\begin{enumerate}[resume]
  \item Justificar $\int_Sr^*\sigma = \int_S\sigma$
        \emph{(on $S$, $r$ restricts to the identity: if
        $\gamma$ is a direct parametrization of a piece of
        $S$, then $r\circ\gamma = \gamma$)}.
  \item Usando Stokes en $\bar B$, muestre $\int_Sr^*\sigma =
        \int_{\bar B}\dd(r^*\sigma)$.
  \item Mostrar $\dd(r^*\sigma) = r^*(\dd\sigma) = 0$ (pregunta
        3 aplicado a $\varphi = r$), y concluir:
        \emph{there is no smooth retraction $\bar B \to S$.}
  \item Resolver el caso excluido $n = 1$ a mano: mostrar
        directamente que ningún \omterm{def:b3:topology:continuity}{aplicación continua} $\intcc{-1}1 \to
        \{-1, 1\}$ corrige ambos puntos finales y nombra el
        teorema que usaste.
\end{enumerate}

\medskip
\noindent\textbf{Parte III --- Smooth Brouwer.} Sea $g$
suave en un \omterm{def:b3:topology:topology}{entorno} de $\bar B$ con $g(\bar B)
\subseteq \bar B$y \emph{no} punto fijo en$\bar B$.
\begin{enumerate}[resume]
  \item Mostrar $\delta = \min_{x\in\bar B}\norm{g(x) - x} >
        0$.
  \item Para $x \in \bar B$ sea $u(x) = \frac{x -
        g(x)}{\norm{x - g(x)}}$y sea$r(x) = x +
        t(x)\,u(x)$ la intersección del rayo.
        $\{x + tu(x) : t \geq 0\}$ con $S$. Resuelve el
        $\norm{x + tu}^2 = 1$ cuadrática y obtener
        \[
        t(x) = -\langle x, u(x)\rangle +
        \sqrt{1 - \norm x^2 + \langle x, u(x)\rangle^2}
        \;\geq\; 0 .
        \]
  \item Demuestre que el radicando es \emph{estrictamente} positivo
        en $\bar B$: si $1 - \norm x^2 + \langle x,
        u(x)\rangle^2 = 0$entonces$\norm x = 1$y$\langle
        x, u(x)\rangle = 0$, es decir $\langle x, x -
        g(x)\rangle = 0$, es decir $\langle x, g(x)\rangle =
        1$; por Cauchy--Schwarz con$\norm x = 1$,
        $\norm{g(x)} \leq 1$, esto obliga a $g(x) = x$ ---
        excluido. Deduzca que $r$ es suave en un
        \omterm{def:b3:topology:topology}{entorno} de $\bar B$.
  \item Mostrar $r(\bar B) \subseteq S$ y $r(x) = x$ para $x
        \in S$\emph{(for$\norm x = 1$, check$t(x) = 0$
        using $\langle x, u(x)\rangle \geq 0$, which itself
        follows from $\langle x, x - g(x)\rangle = 1 -
        \langle x, g(x)\rangle \geq 0$)}. Concluir con
        Parte II: \emph{every smooth self-map of $\bar B$ has
        a fixed point.}
\end{enumerate}

\medskip
\noindent\textbf{Parte IV --- \omterm{def:b3:topology:continuity}{Continuo} Brouwer.} Deja
$f\colon\bar B\to\bar B$ será \omterm{def:b3:topology:continuity}{continuo} sin punto fijo.
\begin{enumerate}[resume]
  \item Mostrar $\varepsilon = \min_{\bar B}\norm{f - x} > 0$,
        y producir una aplicación \emph{polinomio} $p\colon\R^n\to
        \R^n$con$\sup_{\bar B}\norm{p - f} <
        \varepsilon/2$ (Stone--Weierstrass,
        \cref{thm:b3:complete:stoneweierstrass}, coordenada
        por coordenada --- justificar el paso de escalar a
        aproximación vectorial).
  \item La aplicación $p$ puede dejar la pelota; establezca $g = \frac{p}{1
        + \varepsilon/2}$. Mostrar$g(\bar B) \subseteq \bar B$
        y $\sup_{\bar B}\norm{g - f} < \varepsilon$.
  \item Derive una contradicción con la Parte III y concluya:
        \emph{every \omterm{def:b3:topology:continuity}{aplicación continua} $\bar B^n \to \bar B^n$
        has a fixed point.}
  \item Demuestre con el ejemplo que el teorema falla: el abierto
        pelota; la esfera $S$; un anillo \omterm{def:b3:forms:closedexact}{cerrado}. cual
        propiedad de $\bar B$ ¿pierde cada contraejemplo?
\end{enumerate}

\medskip

  \item (Ceros del comportamiento \omterm{def:b3:forms:boundary}{límite}) Sea $f \colon \bar
