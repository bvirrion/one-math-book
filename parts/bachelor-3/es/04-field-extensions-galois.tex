\chapter{Extensiones de campo y teoría de Galois}\label{ch:b3:galois}

¿Se pueden resolver todas las ecuaciones con \omterm{def:b3:galois:radical}{radicales}, como fórmula cuadrática?
¿Y sugieren las fórmulas cúbicas de Cardano? ¿Se puede trisecar un ángulo?
¿Con regla y compás? Ambas cuestiones, abiertas desde hace siglos, son
respondido --- negativamente --- por una sola idea de Évariste \omterm{def:b3:galois:galois}{Galois}:
adjuntar a cada polinomio un \emph{grupo} finito de simetrías de
sus raíces y lea la respuesta del grupo. Este capítulo construye
el diccionario: \omterm{def:b3:galois:extension}{extensiones de campo} y \omterm{def:b3:galois:extension}{grados}, \omterm{thm:b3:galois:splitting}{dividir campos} y
\omterm{def:b3:galois:closure}{cierres algebraicos}, \omterm{thm:b3:galois:finitefields}{campos finitos} (una teoría completa --- y el
ciclicidad prometida de $\mathbb F_q^\times$), separabilidad, luego
el propio \omterm{thm:b3:galois:fundamental}{Correspondencia de Galois}, con pruebas completas. Cosechamos:
la imposibilidad de las construcciones clásicas, la estructura de
campos ciclotómicos y la insolubilidad de la quíntica por
\omterm{def:b3:galois:radical}{radicales} --- La simplicidad de \cref{ch:b3:groups} de $A_5$ llama la atención
su objetivo.

\section{Extensiones, grado, algebraicidad.}

\begin{definition}\label{def:b3:galois:extension}
A \emph{extensión de campo}\index{extensión de campo} $L/K$ es un campo
$L$ que contiene $K$ como subcampo; $L$ es entonces un espacio vectorial $K$,
y el \emph{grado}\index{grado de extensión} $[L:K]$ es
su dimensión. La extensión es \emph{finito} si $[L:K] <
\infty$. El \emph{característica} de un campo es el generador.
$\geq 0$ del núcleo de $\Z \to K$, $n \mapsto n\cdot 1$: es
$0$ o un primo $p$; correspondientemente $K$ contiene un más pequeño
subcampo (\emph{campo principal}) isomorfo a $\Q$ o a $\mathbb
F_p = \Z/p\Z$.
\end{definition}

\begin{theorem}[ley de la torre]\label{thm:b3:galois:tower}
Si $K \subseteq L \subseteq M$, entonces $[M:K] = [M:L]\,[L:K]$: si
$(e_i)$ es una base de $L$ sobre $K$ y $(f_j)$ es una base de $M$
sobre $L$, entonces $(e_if_j)$ es una base de $M$ sobre
$K$.\index{ley de la torre}
\end{theorem}

\begin{proof}
Generando: $x \in M$ escribe $x = \sum_j \lambda_jf_j$
($\lambda_j \in L$), cada $\lambda_j = \sum_i \mu_{ij}e_i$
($\mu_{ij} \in K$): $x = \sum_{i,j}\mu_{ij}e_if_j$. Independiente:
$\sum_{i,j}\mu_{ij}e_if_j = 0$ reescribe $\sum_j (\sum_i
\mu_{ij}e_i)f_j = 0$; las sumas internas están en$L$, por lo que desaparecen
($f_j$ independiente sobre $L$); entonces todos $\mu_{ij} = 0$ ($e_i$
independiente sobre $K$).
\end{proof}

\begin{definition}\label{def:b3:galois:algebraic}
Sean $L/K$ y $\alpha \in L$. Si algún $P \in K[X]$ distinto de cero tiene
$P(\alpha) = 0$, $\alpha$ es \emph{algebraico}\index{algebraico
elemento} sobre $K$; el generador monic $\pi_\alpha$ del \omterm{def:b3:rings:ideal}{ideal}
$\{P : P(\alpha) = 0\}$ de $K[X]$ es su \emph{mínimo
polinomio}\index{polinomio mínimo (teoría de campos)}, un
Polinomio \omterm{def:b3:rings:divisibility}{irreducible} ($\pi = QR$ con fuerzas $Q(\alpha) = 0$ $R$
constante por minimalidad del \omterm{def:b3:galois:extension}{grado}). De lo contrario $\alpha$ es
\emph{trascendental}. Escribimos $K(\alpha)$ para los más pequeños.
subcampo de $L$ que contiene $K$ y $\alpha$, y $K[\alpha]$ para
el subanillo más pequeño.
\end{definition}

\begin{theorem}\label{thm:b3:galois:simple}
Si $\alpha$ es \omterm{def:b3:galois:algebraic}{algebraico} sobre $K$ con $d = \deg\pi_\alpha$,
entonces
\[
K(\alpha) = K[\alpha] \cong K[X]/(\pi_\alpha),
\qquad [K(\alpha) : K] = d,
\]
con base $1, \alpha, \dots, \alpha^{d-1}$. Por el contrario, si
$[L:K] < \infty$, cada $\alpha \in L$ es \omterm{def:b3:galois:algebraic}{algebraico} de \omterm{def:b3:galois:extension}{grado}
$\deg \pi_\alpha$ dividiendo $[L:K]$.
\end{theorem}

\begin{proof}
Evaluación $K[X] \to L$, $P \mapsto P(\alpha)$, tiene imagen
$K[\alpha]$ y núcleo $(\pi_\alpha)$; ya que $\pi_\alpha$ es
\omterm{def:b3:rings:divisibility}{irreducible}, $K[X]/(\pi_\alpha)$ es un \emph{campo}
(\cref{prop:b3:rings:primemaximal}: en el \omterm{def:b3:rings:pidufd}{PID} $K[X]$,
\omterm{def:b3:rings:divisibility}{irreducible} genera un \omterm{def:b3:rings:primemaximal}{ideal máximo}), por lo que $K[\alpha]$ es un campo
que contiene $K$ y $\alpha$: equivale a $K(\alpha)$. las clases
de $1, X, \dots, X^{d-1}$ forman una base del cociente (Euclidean
división), de donde la base y el \omterm{def:b3:galois:extension}{grado}. Por el contrario, si $[L:K]
= n < \infty$:$1, \alpha, \dots, \alpha^n$ son dependientes, dando
un polinomio aniquilador; luego $[K(\alpha):K] =
\deg\pi_\alpha$divide$n$ por \omterm{thm:b3:galois:tower}{ley de la torre}.
\end{proof}

\begin{corollary}\label{cor:b3:galois:algclosed}
Si $\alpha, \beta$ son \omterm{def:b3:galois:algebraic}{algebraico} sobre $K$, también lo son $\alpha \pm
\beta$,$\alpha\beta$,$\alpha/\beta$($\beta \ne 0$):
Los elementos de $L$ \omterm{def:b3:galois:algebraic}{algebraico} sobre $K$ forman un subcampo de $L$.
Además la algebraicidad es transitiva: \omterm{def:b3:galois:algebraic}{algebraico} sobre \omterm{def:b3:galois:algebraic}{algebraico} es
\omterm{def:b3:galois:algebraic}{algebraico}.
\end{corollary}

\begin{proof}
$K(\alpha, \beta) = (K(\alpha))(\beta)$ es finito
$K(\alpha)$ ($\beta$ \omterm{def:b3:galois:algebraic}{algebraico} sobre $K \subseteq K(\alpha)$) y
$K(\alpha)/K$ es finito: por \omterm{thm:b3:galois:tower}{ley de la torre} $[K(\alpha,\beta):K] <
\infty$, y cada elemento de$K(\alpha, \beta)$ --- incluido
los cuatro enumerados --- es \omterm{def:b3:galois:algebraic}{algebraico} (\cref{thm:b3:galois:simple}).
Transitividad: si $\beta$ es \omterm{def:b3:galois:algebraic}{algebraico} sobre $L$ y $L/K$ es
\omterm{def:b3:galois:algebraic}{algebraico}, los coeficientes $c_0, \dots, c_{m-1}$ de
$\pi_{\beta/L}$ genera una extensión finita $F = K(c_0, \dots,
c_{m-1})$de$K$(repetida \omterm{thm:b3:galois:tower}{ley de la torre}), y$F(\beta)/F$ es
finito: $[F(\beta):K] < \infty$, por lo que $\beta$ es \omterm{def:b3:galois:algebraic}{algebraico} terminado
$K$.
\end{proof}

\begin{example}\label{ex:b3:galois:degrees}
$[\Q(\sqrt2):\Q] = 2$, $[\Q(\sqrt[3]2):\Q] = 3$ ($X^3 - 2$ es
\omterm{def:b3:rings:divisibility}{irreducible}: \omterm{thm:b3:rings:criteria}{Eisenstein}), $[\Q(\zeta_p):\Q] = p - 1$ para
$\zeta_p = \eu^{2\iu\pi/p}$ ($\Phi_p$ es \omterm{def:b3:rings:divisibility}{irreducible},
\cref{ex:b3:rings:eisenstein}). El \omterm{thm:b3:galois:tower}{ley de la torre} ya es un
arma: $\sqrt[3]2 \notin \Q(\sqrt2)$, desde $3 \nmid 2$.
\end{example}

\section{División de campos; cierre algebraico}

\begin{theorem}[división de campos]\label{thm:b3:galois:splitting}
Sea $P \in K[X]$ no constante. Existe un \emph{dividir
campo}\index{campo dividido} de $P$ sobre $K$: una extensión $L =
K(\alpha_1, \dots, \alpha_n)$generada por raíces de$P$ en la que
$P$ se divide en factores lineales. Es único hasta
$K$-isomorfismo y $[L:K] \leq (\deg P)!$.
\end{theorem}

\begin{proof}
\emph{Existencia}, por inducción en $\deg P$: elige un \omterm{def:b3:rings:divisibility}{irreducible}
factor $Q$ de $P$; el campo $K_1 = K[X]/(Q)$ contiene la raíz
$\alpha_1 = \bar X$ de $Q$, por tanto de $P$; escriba $P = (X -
\alpha_1)P_1$sobre$K_1$y aplique inducción a$P_1$sobre$K_1$;
\omterm{def:b3:galois:extension}{grados} multiplica hasta $n(n-1)\cdots = n!$ como máximo.

\emph{Unicidad} se sigue del \emph{isomorfismo
lema de extensión} más fuerte: sea $\sigma \colon K \to K'$ un isomorfismo,
$P \in K[X]$, $P^\sigma$ el polinomio con coeficientes mapeados,
$L, L'$ \omterm{thm:b3:galois:splitting}{dividir campos} de $P, P^\sigma$; entonces $\sigma$ se extiende
a un isomorfismo $L \to L'$. Inducción en $[L:K]$: si $P$ se divide
en $K$, luego $L = K$ y $L' = K'$ ($P^\sigma$ se divide en $K'$,
y $L'$ es generado por sus raíces). De lo contrario, elija una raíz
$\alpha \in L \setminus K$ de un factor \omterm{def:b3:rings:divisibility}{irreducible} $Q$ de $P$
con $\deg Q \geq 2$; $Q^\sigma$ es un factor \omterm{def:b3:rings:divisibility}{irreducible} de
$P^\sigma$, con raíz $\beta \in L'$; entonces
\[
K(\alpha) \cong K[X]/(Q) \xrightarrow{\ \sigma\ }
K'[X]/(Q^\sigma) \cong K'(\beta)
\]
extiende $\sigma$ con $\alpha \mapsto \beta$. Ahora $L$ es un
\omterm{thm:b3:galois:splitting}{campo dividido} de $P$ sobre $K(\alpha)$ y $L'$ de $P^\sigma$
sobre $K'(\beta)$, con $[L : K(\alpha)] < [L:K]$: inducción
se extiende más allá hasta $L \to L'$.
\end{proof}

\begin{definition}\label{def:b3:galois:closure}
Un campo $\Omega$ es \emph{algebraicamente cerrado}\index{algebraicamente
campo cerrado} si cada polinomio no constante de $\Omega[X]$ tiene
una raíz en $\Omega$ (por lo tanto, divisiones). Un \emph{algebraico
cierre}\index{cierre algebraico} de $K$ es un \omterm{def:b3:galois:algebraic}{algebraico}
extensión $\bar K/K$ con $\bar K$ \omterm{def:b3:galois:algebraic}{algebraicamente} cerrada.
\end{definition}

\begin{theorem}[Steinitz]\label{thm:b3:galois:steinitz}
Cada campo $K$ tiene un \omterm{def:b3:galois:closure}{cierre algebraico}, único hasta
$K$-isomorfismo.
\end{theorem}

\begin{proof}
\emph{Existencia (construcción de Artin).} Sea $R = K[(X_f)_f]$
el anillo polinomial con una variable $X_f$ por no constante
monic $f \in K[X]$, y $I$ el \omterm{def:b3:rings:ideal}{ideal} generado por todos
$f(X_f)$. $I$ es propia: una relación $1 = \sum_{i=1}^r g_i\,
f_i(X_{f_i})$ involucra un número finito de polinomios; en un común
\omterm{thm:b3:galois:splitting}{campo dividido} $E$ de $f_1\cdots f_r$ recoger raíces $\alpha_i$ de
$f_i$ y evaluar $X_{f_i} \mapsto \alpha_i$ (otras variables
$\mapsto 0$): $1 = 0$, absurdo. Sea $\mathfrak m \supseteq I$
maximal (\cref{thm:b3:rings:krull}; Zorn) y $K_1 = R/\mathfrak
m$: un \omterm{def:b3:galois:extension}{extensión de campo} de$K$en el que cada$f \in
K[X]$no constante tiene una raíz, es decir, $\bar X_f$, y que es \omterm{def:b3:galois:algebraic}{algebraico} sobre
$K$ (lo genera el $\bar X_f$, cada \omterm{def:b3:galois:algebraic}{algebraico}).
Iterar: $K \subseteq K_1 \subseteq K_2 \subseteq \cdots$, donde
$K_{n+1}$ le hace a $K_n$ lo que $K_1$ le hizo a $K$, y deja a $\Omega =
\bigcup_n K_n$, un campo. Cualquier$g \in \Omega[X]$ no constante tiene
sus finitos coeficientes en algunos $K_n$; un \omterm{def:b3:rings:divisibility}{irreducible}
El factor de $g$ sobre $K_n$ tiene una raíz en $K_{n+1} \subseteq
\Omega$:$\Omega$ es \omterm{def:b3:galois:closure}{algebraicamente cerrado} y \omterm{def:b3:galois:algebraic}{algebraico} sobre
$K$ (cada $K_n$ es, por transitividad,
\cref{cor:b3:galois:algclosed}): $\Omega$ es un \omterm{def:b3:galois:closure}{algebraico
cierre}.

\emph{Unicidad.} Sean $\Omega, \Omega'$ dos \omterm{def:b3:galois:closure}{algebraico
cierres}. Considere el conjunto de pares $(E, \tau)$ donde $K
\subseteq E \subseteq \Omega$y$\tau\colon E \to \Omega'$ es un
$K$-incrustación, ordenada por extensión; no está vacío ($(K,
\mathrm{id})$) y es inductivo (unión de una cadena), por lo que Zorn da una
máximo $(E_0, \tau_0)$. Si es $E_0 \neq \Omega$, elija $\alpha \in
\Omega\setminus E_0$:$\pi_{\alpha/E_0}$ se asigna a un polinomio.
sobre $\tau_0(E_0)$ que tiene una raíz $\beta$ en \omterm{def:b3:galois:closure}{algebraicamente
cerrado} $\Omega'$, y $\tau_0$ se extiende a $E_0(\alpha) \to
\Omega'$($\alpha \mapsto \beta$), contradiciendo la maximalidad. entonces
hay un $\tau \colon \Omega \to \Omega'$ incrustado en $K$; es
la imagen, isomorfa a $\Omega$, es \omterm{def:b3:galois:closure}{algebraicamente cerrado}, y
$\Omega'$ es \omterm{def:b3:galois:algebraic}{algebraico} encima: para $x \in \Omega'$,
$\pi_{x/\tau(\Omega)}$ se divide en $\tau(\Omega)$, por lo que $x \in
\tau(\Omega)$. Por lo tanto,$\tau$ está en: un isomorfismo.
\end{proof}

\section{campos finitos}

\begin{theorem}\label{thm:b3:galois:finitefields}
Sea $p$ primo, $n \geq 1$, $q = p^n$.
\begin{enumerate}
  \item A \omterm{thm:b3:galois:finitefields}{campo finito} tiene la cardinalidad como potencia primordial, y para
        cada $q$ hay exactamente un campo $\mathbb F_q$ con
        Elementos $q$ hasta isomorfismo: el \omterm{thm:b3:galois:splitting}{campo dividido} de
        $X^q - X$ sobre $\mathbb F_p$.\index{campo finito}
  \item El \emph{Frobenius}\index{Endomorfismo de Frobenius} $F
        \colon x \mapsto x^p$es un automorfismo de$\mathbb
        F_q$, y el grupo de automorfismos de$\mathbb F_q$ es
        cíclico de orden $n$, generado por $F$.
  \item $\mathbb F_{p^m}$ se incrusta en $\mathbb F_{p^n}$ y si $m \mid
        n$.
\end{enumerate}
\end{theorem}

\begin{proof}
(1) Un \omterm{thm:b3:galois:finitefields}{campo finito} $E$ tiene la característica $p > 0$ y es un
Espacio vectorial $\mathbb F_p$ de dimensión finita: $\abs E = p^n$.
Su grupo multiplicativo tiene orden $q - 1$, por lo que cada $x \in E$
satisface $x^q = x$: $E$ consta de raíces $q$ de $X^q - X$,
por lo tanto, hay un \omterm{thm:b3:galois:splitting}{campo dividido} sobre $\mathbb F_p$ ---
determinando $E$ hasta isomorfismo
(\cref{thm:b3:galois:splitting}). Por el contrario, en un \omterm{thm:b3:galois:splitting}{dividir
campo} $L$ de $X^q - X$, el conjunto $E$ de sus raíces es un
\emph{subcampo}: $(x + y)^q = x^q + y^q$ iterando el
el sueño del estudiante de primer año $(a+b)^p = a^p + b^p$ ($p \mid \binom pk$), y
$(xy)^q = x^qy^q$, $(x^{-1})^q = (x^q)^{-1}$; tiene exactamente $q$
elementos desde $X^q - X$ es \omterm{def:b3:galois:separable}{separable}: su derivada es
$qX^{q-1} - 1 = -1$ (como $p \mid q$), coprime, así que no
raíces repetidas. Por tanto, $L = E$ tiene elementos $q$.

(2) $F$ es un morfismo de campo (sueño de primer año), inyectivo
(campos), por lo tanto biyectivo en el finito $\mathbb F_q$. $F^n =
\mathrm{id}$($x^q = x$), y no menor poder es la identidad:
$F^m = \mathrm{id}$ significa que todos los elementos $q$ son raíces de
$X^{p^m} - X$, forzando a $p^m \geq q$. Entonces $\langle F\rangle$ es
cíclico de orden $n$; y no hay otros automorfismos, por
el $\abs{\operatorname{Aut}} \leq [\,\mathbb F_q :
\mathbb F_p\,] = n$ cota se demostró a continuación
(\cref{prop:b3:galois:embeddings} con $L = \mathbb F_q$, $K =
\mathbb F_p$: los automorfismos fijan el campo primo).

(3) Si $\mathbb F_{p^m} \subseteq \mathbb F_{p^n}$, el \omterm{thm:b3:galois:tower}{ley de la torre}
da $p^n = (p^m)^d$: $m \mid n$. Por el contrario, si $m \mid n$, entonces
$p^m - 1 \mid p^n - 1$ (suma geométrica), por lo que $X^{p^m} - X$ divide
$X^{p^n} - X$ (mismo argumento en exponentes: $X^{a} - 1 \mid X^{b}
- 1$cuando$a \mid b$), y las raíces del primero dentro
$\mathbb F_{p^n}$ del subcampo requerido, de cardinalidad
$p^m$ (separabilidad como en (1)).
\end{proof}

\begin{theorem}[Ciclicidad]\label{thm:b3:galois:cyclic}
Todo subgrupo finito del grupo multiplicativo de un campo es
cíclico. En particular $\mathbb F_q^\times \cong
\Z/(q-1)\Z$.\index{grupo multiplicativo de un cuerpo finito}
\end{theorem}

\begin{proof}
Sea $G \leq K^\times$ finito. Por el teorema de la estructura
(\cref{cor:b3:modules:abelian}), $G \cong \Z/d_1 \times \dots
\times \Z/d_s$con$d_1 \mid \dots \mid d_s$. Cada$x \in G$
luego satisface $x^{d_s} = 1$; pero $X^{d_s} - 1$ tiene como máximo
$d_s$ raíces en el campo $K$: $\abs G = d_1\cdots d_s \leq d_s$,
forzando $s = 1$: $G$ es cíclico.
\end{proof}

\begin{example}\label{ex:b3:galois:f8}
$\mathbb F_8 = \mathbb F_2[X]/(X^3 + X + 1)$: la cúbica no tiene
root en $\mathbb F_2$, por lo tanto es \omterm{def:b3:rings:divisibility}{irreducible}. Escribiendo $\omega =
\bar X$:$\mathbb F_8^\times$es cíclico de orden$7$, por lo que
\emph{cada} se genera el elemento $\neq 0, 1$. Los subcampos de
$\mathbb F_{p^{12}}$ forma la red divisoria de $12$:
$\mathbb F_p, \mathbb F_{p^2}, \mathbb F_{p^3}, \mathbb F_{p^4},
\mathbb F_{p^6}, \mathbb F_{p^{12}}$ --- un primero, completo
instancia del \omterm{thm:b3:galois:fundamental}{Correspondencia de Galois}.
\end{example}

\section{Separabilidad e incrustaciones}

\begin{definition}\label{def:b3:galois:separable}
Un polinomio $P \in K[X]$ es \emph{separable}\index{separable
polinomio} si no tiene raíz repetida en un \omterm{thm:b3:galois:splitting}{campo dividido} ---
equivalentemente $\gcd(P, P') = 1$ (una raíz repetida es una raíz común
raíz; por el contrario, sobre \omterm{thm:b3:galois:splitting}{campo dividido}, una raíz común es
repetido; y el mcd no cambia bajo \omterm{def:b3:galois:extension}{extensión de campo},
argumento de \cref{cor:b3:modules:descent}). Un \omterm{def:b3:galois:algebraic}{elemento algebraico}
es separable si su \omterm{def:b3:galois:algebraic}{polinomio mínimo} lo es; una extensión $L/K$ es
separable si todos sus elementos lo son.
\end{definition}

\begin{proposition}\label{prop:b3:galois:perfect}
Un \emph{\omterm{def:b3:rings:divisibility}{irreducible}} $P \in K[X]$ es \omterm{def:b3:galois:separable}{separable} a menos que $P' = 0$,
lo que fuerza a $\operatorname{char} K = p > 0$ y $P \in K[X^p]$.
En consecuencia, cada extensión \omterm{def:b3:galois:algebraic}{algebraico} de un campo de
característica $0$, y de un \omterm{thm:b3:galois:finitefields}{campo finito}, es \omterm{def:b3:galois:separable}{separable} (como
Los campos se llaman \emph{perfecto}\index{campo perfecto}).
\end{proposition}

\begin{proof}
$\gcd(P, P')$ divide $P$; si no es $1$, irreductibilidad
fuerza $\gcd = P$ (hasta una constante), por lo que $P \mid P'$ con $\deg
P' < \deg P$:$P' = 0$. Escritura$P = \sum a_kX^k$:$ka_k = 0$
para todos los $k$, por lo que en la característica $0$, $P$ es constante (excluido);
en la característica $p$, $a_k = 0$ a menos que $p \mid k$: $P =
Q(X^p)$. Sobre un \omterm{thm:b3:galois:finitefields}{campo finito}, cada elemento es una potencia$p$-ésima
(\omterm{thm:b3:galois:finitefields}{Frobenius} está activado), por lo que $Q(X^p) = \sum b_k^p X^{pk} = (\sum
b_kX^k)^p$no es \omterm{def:b3:rings:divisibility}{irreducible}:$P' = 0$ no puede suceder para
\omterm{def:b3:rings:divisibility}{irreducible} $P$ allí tampoco.
\end{proof}

\begin{proposition}[Contando incrustaciones]\label{prop:b3:galois:embeddings}
Sea $L = K(\alpha_1, \dots, \alpha_r)$ finito sobre $K$, y
$\sigma \colon K \to \Omega$ una incrustación en un \omterm{def:b3:galois:closure}{algebraicamente
campo cerrado}. Luego el número de extensiones de $\sigma$ a $L$
es como máximo $[L:K]$, con igualdad si $L/K$ es \omterm{def:b3:galois:separable}{separable}. en
en particular $\abs{\operatorname{Aut}_K(L)} \leq [L:K]$.
\end{proposition}

\begin{proof}
Inducción en $[L:K]$ mediante sencillos pasos. Para $L = K(\alpha)$: un
La extensión $\tau$ está determinada por $\tau(\alpha)$, que debe ser un
raíz en $\Omega$ de $\pi_\alpha^\sigma$; por el contrario cada uno de estos
root proporciona una extensión ($K(\alpha) \cong K[X]/(\pi_\alpha)$).
El número de extensiones es el número de raíces \emph{distinto}.
de $\pi_\alpha^\sigma$ en $\Omega$: como máximo $\deg\pi_\alpha =
[K(\alpha):K]$, con igualdad si$\pi_\alpha$ es \omterm{def:b3:galois:separable}{separable}
(la separabilidad de $\pi^\sigma$ y $\pi$ coincide: mcd con el
El derivado se conserva en $\sigma$). En general, factor $L =
K(\alpha_1)(\alpha_2, \dots)$: extensiones de$\sigma$ a
$K(\alpha_1)$ número $\leq [K(\alpha_1):K]$, y cada uno se extiende en
$\leq [L : K(\alpha_1)]$ vías por inducción; multiplicar (\omterm{thm:b3:galois:tower}{torre
ley}). En el caso \omterm{def:b3:galois:separable}{separable} ambos conteos son igualdades: \omterm{def:b3:galois:algebraic}{mínimo
polinomios} sobre el campo más grande $K(\alpha_1)$ divide esos
sobre $K$, por lo tanto permanece \omterm{def:b3:galois:separable}{separable}.
\end{proof}

\begin{theorem}[elemento primitivo]\label{thm:b3:galois:primitive}
Cada extensión finita \emph{\omterm{def:b3:galois:separable}{separable}} es simple: $L =
K(\gamma)$para algunos$\gamma$.\index{teorema del elemento primitivo}
\end{theorem}

\begin{proof}
Si $K$ es finito, también lo es $L$, y un generador $\gamma$ del
El grupo cíclico $L^\times$ (\cref{thm:b3:galois:cyclic}) lo hace.
Sea $K$ infinito; por inducción basta con tratar $L =
K(\alpha, \beta)$. Sea$n = [L:K]$; por
\cref{prop:b3:galois:embeddings} hay $n$ distintos
$K$-incrustaciones $\sigma_1, \dots, \sigma_n \colon L \to \Omega$
($\Omega$ y \omterm{def:b3:galois:closure}{cierre algebraico}). El polinomio
\[
D(T)=\prod_{i<j}\bigl[\bigl(\sigma_i(\alpha)-\sigma_j(\alpha)
\bigr) + T\bigl(\sigma_i(\beta) - \sigma_j(\beta)\bigr)\bigr]
\]
no es idénticamente cero: un factor desaparece idénticamente sólo si
$\sigma_i, \sigma_j$ está de acuerdo tanto en $\alpha$ como en $\beta$, de ahí en adelante
$L$ --- excluido para $i \neq j$. Como $K$ es infinito, elija $c \in
K$con$D(c) \ne 0$: entonces los elementos$n$ $\sigma_i(\alpha +
c\beta)$son distintos por pares, por lo que$\gamma = \alpha + c\beta$
tiene al menos $n$ conjugados distintos en $\Omega$, es decir\
$\deg \pi_\gamma \geq n$: $[K(\gamma):K] \geq n = [L:K]$ fuerzas
$L = K(\gamma)$.
\end{proof}

\section{La correspondencia de Galois}

\begin{definition}\label{def:b3:galois:galois}
Una extensión finita $L/K$ es \emph{Galois}\index{extensión de galois}
si es el \omterm{thm:b3:galois:splitting}{campo dividido} de un polinomio \emph{\omterm{def:b3:galois:separable}{separable}}
sobre $K$. Su \emph{grupo galois}\index{grupo galois} es
$\operatorname{Gal}(L/K) = \operatorname{Aut}_K(L)$, el grupo de
automorfismos de campo de $L$ fijando $K$ puntualmente.
\end{definition}

\begin{proposition}\label{prop:b3:galois:galoisorder}
Si $L/K$ es \omterm{def:b3:galois:galois}{Galois}, entonces $\abs{\operatorname{Gal}(L/K)} =
[L:K]$; además,$L/F$es \omterm{def:b3:galois:galois}{Galois} para cada campo intermedio$K
\subseteq F \subseteq L$, y cada$L \to \Omega
\supseteq L$incrustado en$F$tiene la imagen$L$ (\emph{normalidad}).
\end{proposition}

\begin{proof}
Deje que $L$ divida \omterm{def:b3:galois:separable}{separable} $P$ sobre $K$ y arregle un \omterm{def:b3:galois:closure}{algebraico
cierre} $\Omega \supseteq L$. $L/K$ es \omterm{def:b3:galois:separable}{separable}: se genera
por raíces de $P$; La separabilidad de cada elemento se deriva de la
caso de igualdad a continuación, pero discutamos directamente ---
\cref{prop:b3:galois:embeddings} aplicado a los generadores (raíces
del \omterm{def:b3:galois:separable}{separable} $P$, cuyo \omterm{def:b3:galois:algebraic}{polinomios mínimos} divide a $P$)
produce exactamente extensiones $[L:K]$ de $K \hookrightarrow \Omega$
(en el paso inductivo, el \omterm{def:b3:galois:algebraic}{polinomio mínimo} de una raíz de $P$
sobre un campo intermedio todavía divide $P$, por lo tanto es
\omterm{def:b3:galois:separable}{separable}). Cada una de estas incrustaciones $\tau \colon L \to \Omega$
permuta las raíces de $P$ ($\tau$ fija los coeficientes), y
$L$ es generado por ellos: $\tau(L) = L$. Por lo tanto incrustaciones $=$
automorfismos: $\abs{\operatorname{Gal}(L/K)} = [L:K]$. Para
intermedio $F$: $L$ es también el \omterm{thm:b3:galois:splitting}{campo dividido} de $P$ sobre
$F$ y $P$ siguen siendo \omterm{def:b3:galois:separable}{separable}: $L/F$ es \omterm{def:b3:galois:galois}{Galois}; lo mismo
El argumento da normalidad sobre $F$.
\end{proof}

\begin{lemma}[Artín]\label{lem:b3:galois:artin}
Sea $G$ un grupo finito de automorfismos de un campo $L$ y
$K = L^G = \{x : \sigma(x) = x\ \forall\sigma \in G\}$ está arreglado
campo. Entonces $[L : L^G] \leq \abs G$.
\end{lemma}

\begin{proof}
Dejemos que $n = \abs G$, $G = \{\sigma_1, \dots, \sigma_n\}$ y
supongamos que $x_1, \dots, x_{n+1} \in L$ son linealmente independientes sobre
$K$. El sistema lineal homogéneo de ecuaciones $n$ en $n+1$
incógnitas $(c_j)$ sobre $L$,
\[
\sum_{j=1}^{n+1} c_j\,\sigma_i(x_j) = 0
\qquad (i = 1, \dots, n),
\]
tiene una solución distinta de cero; elija uno con el \emph{menos} distinto de cero
entradas, digamos $c_1, \dots, c_r \neq 0$ (renumeración), $r \geq 2$
(un solo $c_j\sigma_i(x_j) = 0$ es imposible), normalizado
$c_r = 1$. No todos los $c_j$ se encuentran en $K$: la ecuación para $\sigma_i =
\mathrm{id}$contradeciría la independencia; diga$c_1 \notin K$,
entonces $\tau(c_1) \ne c_1$ para algunos $\tau \in G$. Aplicar $\tau$ a todos
ecuaciones: dado que $\tau\sigma_i$ pasa por $G$, el vector
$(\tau(c_j))_j$ es otra solución; restando, $(c_j -
\tau(c_j))_j$ es una solución con menos entradas distintas de cero (la
$r$-ésima entrada $1 - 1 = 0$ desaparece, la primera no) y no
cero: contradicción. Entonces, cualquier elemento $n+1$ es dependiente: $[L:K]
\leq n$.
\end{proof}

\begin{theorem}[Teorema fundamental de la teoría de Galois]
\label{thm:b3:galois:fundamental}
Sea $L/K$ un \omterm{def:b3:galois:galois}{extensión de galois} con el grupo $G =
\operatorname{Gal}(L/K)$.\index{Correspondencia de Galois}
\begin{enumerate}
  \item $L^G = K$.
  \item Los mapas $H \mapsto L^H$ y $F \mapsto
        \operatorname{Gal}(L/F)$ son mutuamente inversos,
        biyecciones de inversión de inclusión entre subgrupos de $G$
        y campos intermedios $K \subseteq F \subseteq L$;
        además $[L : L^H] = \abs H$ y $[L^H : K] = [G : H]$.
  \item $H \trianglelefteq G$ si $L^H/K$ es \omterm{def:b3:galois:galois}{Galois}, y luego
        la restricción induce $\operatorname{Gal}(L^H/K) \cong
        G/H$.
\end{enumerate}
\end{theorem}

\begin{proof}
(1) Claramente $K \subseteq L^G$. Por el contrario, dejemos que $\alpha \in L
\setminus K$; exhibimos$\sigma \in G$con$\sigma(\alpha) \ne
\alpha$. El \omterm{def:b3:galois:algebraic}{polinomio mínimo}$\pi_\alpha$sobre$K$ tiene \omterm{def:b3:galois:extension}{grado}
$\geq 2$ y es \omterm{def:b3:galois:separable}{separable} ($L/K$ \omterm{def:b3:galois:separable}{separable},
\cref{prop:b3:galois:galoisorder}), por lo que tiene otra raíz
$\beta \neq \alpha$ en un \omterm{def:b3:galois:closure}{cierre algebraico} $\Omega \supseteq L$.
Ampliar el $K$-incrustado $K(\alpha) \to \Omega$, $\alpha \mapsto
\beta$, a un$\tau\colon L \to \Omega$ incrustado
(\cref{prop:b3:galois:embeddings}); por normalidad
(\cref{prop:b3:galois:galoisorder}) $\tau(L) = L$, entonces $\tau \in
G$,$\beta = \tau(\alpha) \in L$y$\tau(\alpha) \neq
\alpha$.

(2) Para un subgrupo $H$: $L/L^H$ es \omterm{def:b3:galois:galois}{Galois}
(\cref{prop:b3:galois:galoisorder}), y
$\operatorname{Gal}(L/L^H) \supseteq H$ trivialmente, entonces $[L:L^H] =
\abs{\operatorname{Gal}(L/L^H)} \geq \abs H$; El lema de Artin
da $[L:L^H] \leq \abs H$: igualdad, y
$\operatorname{Gal}(L/L^H) = H$. Para un campo intermedio $F$:
$L/F$ \omterm{def:b3:galois:galois}{Galois} da $L^{\operatorname{Gal}(L/F)} = F$ por (1)
aplicado a $L/F$. Los dos mapas son mutuamente inversos; ellos
inclusiones inversas evidentemente. \omterm{def:b3:galois:extension}{Grados}: $[L:L^H] = \abs H$ solo
probado, y $[L^H:K] = [L:K]/[L:L^H] = \abs G/\abs H$.

(3) Para $\sigma \in G$ y $H \leq G$: $\sigma(L^H) =
L^{\sigma H\sigma^{-1}}$ (verificación directa). Por la biyección,
$\sigma(L^H) = L^H$ para todos los $\sigma$ y si $H \trianglelefteq G$.
Ahora, si es $H \trianglelefteq G$, configure $F = L^H$: cada $\sigma \in
G$se restringe a un automorfismo de$F$, dando un morfismo$\rho
\colon G \to \operatorname{Aut}_K(F)$con el núcleo$\{\sigma :
\sigma\restriction_F = \mathrm{id}\} = \operatorname{Gal}(L/F) =
H$. Entonces$G/H$se incrusta en$\operatorname{Aut}_K(F)$, de donde
$\abs{\operatorname{Aut}_K(F)} \geq [G:H] = [F:K]$; al revés
la desigualdad siempre se cumple (\cref{prop:b3:galois:embeddings}):
$\abs{\operatorname{Aut}_K(F)} = [F:K]$ y $\rho$ están activados. eso
Queda por ver $F/K$ es \omterm{def:b3:galois:galois}{Galois}: $F$ es \omterm{def:b3:galois:separable}{separable} sobre $K$
(dentro de \omterm{def:b3:galois:separable}{separable} $L/K$) y $F = K(\gamma)$
(\cref{thm:b3:galois:primitive}); el polinomio
$\prod_{\sigma \in G/H}\bigl(X - \sigma(\gamma)\bigr)$ (producto
sobre las distintas imágenes, que se encuentran en $F$: $\sigma(F) =
L^{\sigma H\sigma^{-1}} = L^H = F$por normalidad de$H$) tiene
coeficientes
arreglado por $G$, por lo tanto en $K$ por (1): es un \omterm{def:b3:galois:separable}{polinomio separable}
de $K[X]$ dividido por $F$, y sus raíces generan $F$: $F/K$ es
\omterm{def:b3:galois:galois}{Galois}. Por el contrario, si $F = L^H$ con $F/K$ \omterm{def:b3:galois:galois}{Galois}, la normalidad de
$F$ (\cref{prop:b3:galois:galoisorder}, aplicado a incrustaciones
$F \to \Omega$ restringido de elementos de $G$) proporciona
$\sigma(F) = F$ para todos los $\sigma \in G$, es decir,\ $H
\trianglelefteq G$.
\end{proof}

\section{Extensiones ciclotómicas}

\begin{definition}\label{def:b3:galois:cyclotomic}
Sean $n \geq 1$ y $\zeta_n = \eu^{2\iu\pi/n}$. El $n$-ésimo
\emph{polinomio ciclotómico}\index{polinomio ciclotómico} es
$\Phi_n = \prod_{\gcd(k,n)=1,\ 1 \le k \le n} \bigl(X -
\zeta_n^k\bigr)$, de \omterm{def:b3:galois:extension}{grado}$\varphi(n)$; agrupando las raíces de
$X^n - 1$ por orden exacto, $X^n - 1 = \prod_{d \mid n}\Phi_d$,
lo que muestra inductivamente que $\Phi_n \in \Z[X]$ (Euclidean
división de polinomios enteros mónicos).
\end{definition}

\begin{theorem}\label{thm:b3:galois:cyclotomicirred}
$\Phi_n$ es \omterm{def:b3:rings:divisibility}{irreducible} sobre $\Q$; por lo tanto $[\Q(\zeta_n) : \Q] =
\varphi(n)$ y
\[
\operatorname{Gal}\bigl(\Q(\zeta_n)/\Q\bigr) \;\cong\;
(\Z/n\Z)^\times,
\qquad \sigma_a(\zeta_n) = \zeta_n^a .
\]
La extensión $\Q(\zeta_n)/\Q$ es, por tanto, \omterm{def:b3:galois:galois}{Galois} con \emph{abeliano}.
grupo.
\end{theorem}

\begin{proof}
Sea $f = \pi_{\zeta_n}$, entonces $\Phi_n = fg$ con $f, g \in \Z[X]$
monic (lema de Gauss \cref{lem:b3:rings:gauss}: contenido
multiplicar, todos los polinomios mónicos). \emph{Claim: if $\zeta$ is a
root of $f$ and $p \nmid n$ is prime, then $\zeta^p$ is a root
of $f$.} De lo contrario $\zeta^p$ es una raíz de $g$ (es una raíz primitiva
$n$-ésima raíz de la unidad), por lo que $\zeta$ es una raíz de $g(X^p)$ y $f
\mid g(X^p)$en$\Z[X]$ (\omterm{def:b3:galois:algebraic}{polinomio mínimo}, luego Gauss nuevamente).
Reducir mod $p$: $\bar g(X^p) = \bar g(X)^p$ (\omterm{thm:b3:galois:finitefields}{Frobenius} en
$\mathbb F_p[X]$: coeficiente $a^p = a$ y de primer año
sueño), entonces $\bar f \mid \bar g^{\,p}$: $\bar f$ y $\bar g$
comparten un factor \omterm{def:b3:rings:divisibility}{irreducible} y $\bar\Phi_n = \bar f\bar g$ tiene
un factor repetido. Entonces también lo hace $X^n - \bar 1$; pero es
derivado $\bar nX^{n-1}$ es coprimo ($p \nmid n$, y
$0$ no es raíz): contradicción.

Se obtiene cada raíz primitiva $\zeta_n^k$ ($\gcd(k, n) = 1$)
de $\zeta_n$ por potencias primarias sucesivas que no dividen $n$
(factor $k$): el reclamo se propaga, por lo que cada raíz primitiva es una
raíz de $f$: $f = \Phi_n$, \omterm{def:b3:rings:divisibility}{irreducible}. En consecuencia
$[\Q(\zeta_n):\Q] = \varphi(n)$ y $\Q(\zeta_n)$ es el
\omterm{thm:b3:galois:splitting}{campo dividido} del \omterm{def:b3:galois:separable}{separable} $X^n - 1$ (todas las raíces son potencias
de $\zeta_n$): \omterm{def:b3:galois:galois}{Galois}. Un automorfismo $\sigma$ envía $\zeta_n$
a otra raíz primitiva $\zeta_n^{a(\sigma)}$, y $\sigma
\mapsto a(\sigma)$ es un morfismo inyectivo en
$(\Z/n\Z)^\times$; ambos grupos tienen orden $\varphi(n)$:
isomorfismo.
\end{proof}

\section{Regla y compás}

\begin{definition}\label{def:b3:galois:constructible}
Identificar el avión con $\C$; comenzar desde $\{0, 1\}$. un punto es
\emph{construible}\index{número construible} si es
obtenible mediante un número finito de intersecciones de líneas que pasan por dos
puntos ya construidos y círculos centrados en un punto construido
punto con radio a una distancia de dos puntos construidos.
\end{definition}

\begin{theorem}[Wantzel]\label{thm:b3:galois:wantzel}
$z \in \C$ es \omterm{def:b3:galois:constructible}{construible} si hay una torre $\Q = F_0
\subseteq F_1 \subseteq \dots \subseteq F_r$con$[F_{i+1} :
F_i] = 2$y$z \in F_r$. En particular, un \omterm{def:b3:galois:constructible}{número construible}
es \omterm{def:b3:galois:algebraic}{algebraico} de \omterm{def:b3:galois:extension}{grado} una potencia de $2$ sobre
$\Q$.\index{Teorema de Wantzel}
\end{theorem}

\begin{proof}
($\Rightarrow$) Las coordenadas de la intersección de dos líneas.
a través de puntos con coordenadas en un subcampo $F \subseteq \R$
resuelven un sistema lineal sobre $F$: se quedan en $F$. Línea-círculo
y las intersecciones círculo-círculo conducen, después de eliminar la
parte lineal (restando las dos ecuaciones circulares se obtiene una línea),
a una ecuación cuadrática sobre $F$: las nuevas coordenadas se encuentran en $F$
o en $F(\sqrt d)$ para algunos $d \in F$, $d > 0$. Por inducción,
cada punto construido tiene coordenadas en una torre de cuadrática
extensiones de $\Q$; y $z = x + \iu y$ está en una torre cuadrática
también (junto a $\iu$: un paso cuadrático más). El \omterm{def:b3:galois:extension}{grado}
consecuencia: $[\Q(z):\Q]$ divide $[F_r : \Q] = 2^r$ (\omterm{thm:b3:galois:tower}{torre
ley}).

($\Leftarrow$) El \omterm{def:b3:galois:constructible}{números construibles} forma un campo: sumas y
diferencias por paralelogramos (los paralelos son \omterm{def:b3:galois:constructible}{construible}:
bajar y subir perpendiculares dos veces --- el clásico
perpendicular que pasa por un punto utiliza un círculo y dos arcos);
productos y cocientes según las configuraciones de intersección de Thales
(longitudes dadas $a, b$ construyen $ab$ y $a/b$ con similares
triángulos sobre dos rayos). Y el campo se cierra bajo la plaza.
raíces: para $a > 0$, el círculo de diámetro $1 + a$ y el
perpendicular en el punto de unión se encuentra en la altura $\sqrt a$
(relación altitud-geométrica-media en un triángulo rectángulo); por un
complejo $w = \rho\eu^{\iu\theta}$, construir $\sqrt\rho$ y
bisecta $\theta$ (la bisección de ángulo es una construcción de compás).
Las partes reales e imaginarias de los miembros de una torre cuadrática son
por lo tanto \omterm{def:b3:galois:constructible}{construible} por inducción en la torre: cada paso
linda con raíces de una cuadrática, expresable mediante operaciones de campo
y una raíz cuadrada de un número ya construido (el
fórmula cuadrática; en la característica $0$).
\end{proof}

\begin{corollary}\label{cor:b3:galois:impossible}
Los tres problemas clásicos no tienen solución para el gobernante y
brújula:\index{regla y compás}
\begin{enumerate}
  \item \emph{Duplicación del cubo}: $\sqrt[3]2$ tiene \omterm{def:b3:galois:extension}{grado}
        $3$, no una potencia de $2$.
  \item \emph{Trisección del ángulo}: trisección $60^\circ$
        requiere $\cos 20^\circ$, una raíz del \omterm{def:b3:rings:divisibility}{irreducible} $8X^3
        - 6X - 1$: \omterm{def:b3:galois:extension}{grado}$3$.
  \item \emph{La cuadratura del circulo}: $\sqrt\pi$ es trascendental
        ($\pi$ es --- el teorema de Lindemann, admitido aquí: es
        la prueba pertenece a un curso de teoría de la trascendencia).
\end{enumerate}
Además, el $n$-gon normal es \omterm{def:b3:galois:constructible}{construible} si y así
$\varphi(n)$ es una potencia de $2$ (Gauss--Wantzel; el ``si'' usa
En el método del problema del fin de semana, el "sólo si" es
\cref{thm:b3:galois:wantzel} aplicado a $\zeta_n$, de \omterm{def:b3:galois:extension}{grado}
$\varphi(n)$). Para $n = 7$: $\varphi(7) = 6$: el normal
el heptágono es imposible; para $n = 17$: $\varphi(17) = 16 = 2^4$:
\omterm{def:b3:galois:constructible}{construible} --- el problema del fin de semana lo construye.
\end{corollary}

\begin{proof}
(1) $X^3 - 2$ es \omterm{def:b3:rings:divisibility}{irreducible} (\omterm{thm:b3:rings:criteria}{Eisenstein}). (2) De $\cos
3\theta = 4\cos^3\theta - 3\cos\theta$con$3\theta =
60^\circ$:$8c^3 - 6c = 1$para$c = \cos 20^\circ$; el cúbico
$8X^3 - 6X - 1$ no tiene raíz racional (los candidatos $\pm1, \pm\frac
12, \pm\frac14, \pm\frac18$ fallan), por lo tanto es \omterm{def:b3:rings:divisibility}{irreducible}: \omterm{def:b3:galois:extension}{grado}
$3$. Un ángulo general $60^\circ$ es \omterm{def:b3:galois:constructible}{construible}, por lo que un trisector
construiría $c$. (3) Si $\sqrt\pi$ fuera \omterm{def:b3:galois:constructible}{construible},
sería \omterm{def:b3:galois:algebraic}{algebraico}, por lo tanto también $\pi$. La declaración $n$-gon: la
\omterm{def:b3:galois:extension}{grado} de $\zeta_n$ es $\varphi(n)$
(\cref{thm:b3:galois:cyclotomicirred}); la necesidad se deriva de
Wantzel; por suficiencia, el \omterm{def:b3:galois:galois}{grupo galois}, abeliano de orden
$2^m$, admite una cadena de subgrupos índice-$2$ (un grupo finito $2$
hace: \cref{exo:b3:groups:10}), cuyos campos fijos forman un
torre cuadrática que termina en $\Q(\zeta_n)$
(\cref{thm:b3:galois:fundamental}); concluir por
\cref{thm:b3:galois:wantzel}.
\end{proof}

\section{Solubilidad por radicales}

\begin{definition}\label{def:b3:galois:radical}
Una extensión $L/K$ (característica $0$ a lo largo de esta sección)
es \emph{radical} si hay una torre $K = F_0 \subseteq \dots
\subseteq F_r = L$con$F_{i+1} = F_i(\alpha_i)$,
$\alpha_i^{n_i} \in F_i$: cada paso linda con una raíz $n_i$. un
polinomio $P \in K[X]$ es \emph{solucionable por
radicales}\index{solucionable por radicales} si su \omterm{thm:b3:galois:splitting}{campo dividido} es
contenido en alguna extensión radical de $K$.
\end{definition}

\begin{lemma}\label{lem:b3:galois:cyclicsteps}
Deje que $K$ contenga una raíz primitiva $n$ de la unidad $\zeta$, es decir,\
$\zeta$ del pedido $n$ en $K^\times$ y $a \in K^\times$. entonces
$K(\sqrt[n]a)/K$ es \omterm{def:b3:galois:galois}{Galois} con el grupo \emph{cíclico}. Por el contrario
--- no es necesario a continuación --- cada extensión cíclica de \omterm{def:b3:galois:extension}{grado} $n$ es
de esta forma. Además $K(\zeta_n)/K$ es \omterm{def:b3:galois:galois}{Galois} con
Grupo \emph{abeliano}, para cualquier $K$ de característica $0$.
\end{lemma}

\begin{proof}
$X^n - a$ es \omterm{def:b3:galois:separable}{separable} ($\gcd$ con $nX^{n-1}$: $a \neq 0$) y
se divide en $K(\alpha)$, $\alpha^n = a$: sus raíces son las
$\zeta^k\alpha \in K(\alpha)$. Entonces $K(\alpha)/K$ es \omterm{def:b3:galois:galois}{Galois}; el
map $\sigma \mapsto \sigma(\alpha)/\alpha \in \mu_n = \langle
\zeta\rangle$es un morfismo inyectivo ($\sigma\tau(\alpha) =
\sigma(\tau(\alpha)/\alpha \cdot \alpha) =
\tau(\alpha)/\alpha\cdot\sigma(\alpha)$, estando el cociente en
$K$), en un grupo cíclico: $\operatorname{Gal}$ es cíclico. el
lo contrario es la teoría de Kummer, que no necesitaremos (ver el
comentario a continuación). Para $K(\zeta_n)$: divide \omterm{def:b3:galois:separable}{separable} $X^n -
1$y$\sigma \mapsto a(\sigma)$con$\sigma(\zeta_n) =
\zeta_n^{a(\sigma)}$ incrusta el grupo en el abeliano.
$(\Z/n\Z)^\times$ como en \cref{thm:b3:galois:cyclotomicirred}
(la inyectividad solo necesita $\zeta_n$ para generar las raíces de
unidad involucrada).
\end{proof}

\begin{theorem}[Galois]\label{thm:b3:galois:solvable}
Sea $K$ de característica $0$ y $P \in K[X]$ con \omterm{thm:b3:galois:splitting}{dividir
campo} $L$. Si $P$ es \omterm{def:b3:galois:radical}{solucionable por radicales}, entonces
$\operatorname{Gal}(L/K)$ es un \omterm{def:b3:groups:derived}{grupo solucionable}. (Lo contrario es
también cierto; no lo necesitaremos.)\index{Solubilidad de Galois
teorema}
\end{theorem}

\begin{proof}
Paso 1: \emph{enlarge the \omterm{def:b3:galois:radical}{radical} tower to a \omterm{def:b3:galois:galois}{Galois} one.} Sea $L
\subseteq M$con$M/K$\omterm{def:b3:galois:radical}{radical}, con exponentes raíz$n_1, \dots,
n_r$y$n = n_1\cdots n_r$. Primer anexo$\zeta_n$: la torre
$K \subseteq K(\zeta_n) \subseteq M(\zeta_n)$ sigue siendo \omterm{def:b3:galois:radical}{radical}
($\zeta_n$ es una raíz de unidad: un paso \omterm{def:b3:galois:radical}{radical}, $\zeta_n^n = 1$),
y sus pasos más allá del primero ocurren sobre campos que contienen el
necesitaba raíces de unidad. A continuación, reemplace $M(\zeta_n)$ por el
compuesto $N$ de todos los $\sigma(M(\zeta_n))$, $\sigma$ que abarcan
las (finitas) incrustaciones $K$ de $M(\zeta_n)$ en un sistema fijo
\omterm{def:b3:galois:closure}{cierre algebraico}: $N$ es el \omterm{thm:b3:galois:splitting}{campo dividido} del producto de
\omterm{def:b3:galois:algebraic}{polinomios mínimos} de un grupo electrógeno (característica $0$:
finito y \omterm{def:b3:galois:separable}{separable}), por lo tanto $N/K$ es \omterm{def:b3:galois:galois}{Galois}; y $N$ es
\omterm{def:b3:galois:radical}{radical} sobre $K$: cada $\sigma(M(\zeta_n))$ es \omterm{def:b3:galois:radical}{radical} sobre $K$
(aplicar $\sigma$ a una torre \omterm{def:b3:galois:radical}{radical}) y un compuesto de \omterm{def:b3:galois:radical}{radical}
extensiones es \omterm{def:b3:galois:radical}{radical} (concatenar las torres: si $F'/K$ es
\omterm{def:b3:galois:radical}{radical} con torre contigua a $\beta_j$, luego tipo $F''(\beta_j)$
los pasos permanecen \omterm{def:b3:galois:radical}{radical} sobre cualquier base más grande).

Paso 2: \emph{read \omterm{def:b3:groups:derived}{solubilidad} off the \omterm{def:b3:galois:galois}{Galois} tower.} Así que supongamos
$L \subseteq N$, $N/K$ \omterm{def:b3:galois:galois}{Galois} y \omterm{def:b3:galois:radical}{radical} con torre $K
\subseteq K(\zeta_n) = E_0 \subseteq E_1 \subseteq \dots
\subseteq E_s = N$, cada$E_{i+1} = E_i(\sqrt[n_i]{a_i})$ con
$\zeta_{n_i} \in E_0 \subseteq E_i$. Sean $G =
\operatorname{Gal}(N/K)$y$G_i = \operatorname{Gal}(N/E_i)$:
una cadena decreciente $G \supseteq G_0 \supseteq G_1 \supseteq
\dots \supseteq G_s = \{e\}$. Cada$E_{i+1}/E_i$ es \omterm{def:b3:galois:galois}{Galois} con
grupo cíclico (\cref{lem:b3:galois:cyclicsteps}), por lo que por el
teorema fundamental aplicado al \omterm{def:b3:galois:galois}{extensión de galois} $N/E_i$
(\cref{thm:b3:galois:fundamental}(3), con grupo ambiental $G_i$):
$G_{i+1} \trianglelefteq G_i$ con $G_i/G_{i+1} \cong
\operatorname{Gal}(E_{i+1}/E_i)$cíclico. Del mismo modo$E_0/K$ es
\omterm{def:b3:galois:galois}{Galois} con grupo abeliano $G/G_0$
(\cref{lem:b3:galois:cyclicsteps}). La cadena exhibe $G$ como
\omterm{def:b3:groups:derived}{soluble} (\cref{prop:b3:groups:derived}). Finalmente
$\operatorname{Gal}(L/K)$ es un \emph{cociente} de $G$: $L/K$ es
\omterm{def:b3:galois:galois}{Galois} ($P$ \omterm{def:b3:galois:separable}{separable} en la característica $0$) y la restricción $G
\to \operatorname{Gal}(L/K)$ está en
(\cref{thm:b3:galois:fundamental}(3) con $H =
\operatorname{Gal}(N/L)$); los cocientes de \omterm{def:b3:groups:derived}{grupos solucionables} son
\omterm{def:b3:groups:derived}{soluble}.
\end{proof}

\begin{corollary}[Insolubilidad de la quintica]
\label{cor:b3:galois:quintic}
Hay polinomios de \omterm{def:b3:galois:extension}{grado} $5$ sobre $\Q$ que no son
\omterm{def:b3:galois:radical}{solucionable por radicales}: por ejemplo $X^5 - 4X + 2$, cuyo \omterm{def:b3:galois:galois}{Galois
grupo} es $S_5$ (\cref{exo:b3:galois:11}), un grupo \omterm{def:b3:groups:derived}{no solucionable}
(\cref{cor:b3:groups:snnotsolvable}). No hay una fórmula general en
\omterm{def:b3:galois:radical}{radicales} puede existir para \omterm{def:b3:galois:extension}{grado} $\geq 5$.\index{quintico,
irresolvebilidad}
\end{corollary}

\begin{remark}\label{rem:b3:galois:converse}
El inverso de \cref{thm:b3:galois:solvable} --- un \omterm{def:b3:groups:derived}{soluble}
\omterm{def:b3:galois:galois}{grupo galois} implica \omterm{def:b3:groups:derived}{solubilidad} por \omterm{def:b3:galois:radical}{radicales} --- se prueba por
descendiendo el \omterm{def:b3:groups:derived}{serie derivada} y mostrando cada extensión cíclica
(con suficientes raíces de unidad) es \omterm{def:b3:galois:radical}{radical}, vía \emph{Lagrange
resolutivos}; explica por qué \omterm{def:b3:galois:extension}{grados} $2, 3, 4$ tiene fórmulas:
$S_2, S_3, S_4$ son \omterm{def:b3:groups:derived}{soluble}
(\cref{ex:b3:groups:solvableexamples}). Lo dejamos admitido en
este nivel; un tratamiento completo pertenece a un curso de maestría, pero
\cref{exo:b3:galois:8} lo hace concreto para la cúbica.
\end{remark}

\section{Problema: Gauss y el 17 gon regular}

\begin{problem}[{Problema de fin de semana --- constructibilidad del
17 gon}]\label{pb:b3:galois:1}
El 30 de marzo de 1796, Gauss, de diecinueve años, demostró que el
$17$-gon regular es \omterm{def:b3:galois:constructible}{construible} --- el primer progreso en el
pregunta desde la antigüedad. Reconstruimos su cálculo con el
herramientas de este capítulo. Configure $\zeta = \eu^{2\iu\pi/17}$, $L =
\Q(\zeta)$,$G = \operatorname{Gal}(L/\Q)$.

\medskip
\noindent\textbf{Parte I --- The group and its filtration.}
\begin{enumerate}
  \item Justificar: $[L:\Q] = 16$, $G \cong (\Z/17\Z)^\times$,
        cíclico de orden $16$. Verifique que $3$ sea un generador de
        $(\Z/17\Z)^\times$\emph{(compute the powers of $3$
        modulo $17$: $3, 9, 10, 13, 5, 15, 11, 16, \dots$)}.
  \item Deje $\sigma \in G$ con $\sigma(\zeta) = \zeta^3$, y
        $H_k = \langle \sigma^{2^k}\rangle$ para $k = 0, \dots,
        4$. Muestre que$G = H_0 \supset H_1 \supset H_2 \supset
        H_3 \supset H_4 = \{e\}$con cada índice$[H_k :
        H_{k+1}] = 2$, y que los campos fijos$\Q = L_0
        \subset L_1 \subset L_2 \subset L_3 \subset L_4 = L$
        Forman una torre de extensiones cuadráticas.
  \item Concluya \emph{a priori}, utilizando
        \cref{thm:b3:galois:wantzel}, que $\zeta$ --- de ahí el
        $17$-gon --- es \omterm{def:b3:galois:constructible}{construible}. El resto del problema
        hace explícita la torre.
\end{enumerate}

\medskip
\noindent\textbf{Parte II --- The periods of length 8.}
Definir el \emph{Períodos de Gauss}
\[
\eta_0 = \sum_{k \text{ even}} \zeta^{3^k \bmod 17}
= \zeta^{1} + \zeta^{9} + \zeta^{13} + \zeta^{15} + \zeta^{16} +
\zeta^{8} + \zeta^{4} + \zeta^{2},
\qquad
\eta_1 = \sum_{k \text{ odd}} \zeta^{3^k \bmod 17}.
\]
\begin{enumerate}[resume]
  \item Muestra que $\eta_0, \eta_1$ son reparados por $H_1$ e intercambiados
        por $\sigma$; deducir $\eta_0, \eta_1 \in L_1$ y que
        son las dos raíces de una cuadrática sobre $\Q$.
  \item Calcular $\eta_0 + \eta_1 = -1$. Mostrar $\eta_0\eta_1 = -4$
        \emph{(each product $\zeta^a\zeta^b$ is some $\zeta^c$,
        $c \neq 0$; count how many times each $c$ occurs, or
        argue that the product is a rational integer fixed by
        $G$, equal to the sum over all $64$ products, and use
        that each nonzero residue appears equally often)}.
  \item Deducir $\eta_0 = \frac{-1 + \sqrt{17}}2$, $\eta_1 =
        \frac{-1-\sqrt{17}}2$ (identificar cuál es cuál
        numéricamente: $\eta_0 \approx 1.56$) y $L_1 =
        \Q(\sqrt{17})$.
\end{enumerate}

\medskip
\noindent\textbf{Parte III --- Periods of length 4 and 2.}
Definir
\[
\beta_0 = \zeta + \zeta^{13} + \zeta^{16} + \zeta^{4},\quad
\beta_1 = \zeta^3 + \zeta^5 + \zeta^{14} + \zeta^{12},\quad
\beta_2 = \zeta^9 + \zeta^{15} + \zeta^{8} + \zeta^{2},\quad
\beta_3 = \zeta^{10} + \zeta^{11} + \zeta^{7} + \zeta^{6}.
\]
\begin{enumerate}[resume]
  \item Muestra $\beta_0 + \beta_2 = \eta_0$, $\beta_1 + \beta_3 =
        \eta_1$y que$\beta_0, \beta_2$están arreglados por$H_2$,
        intercambiado por $\sigma^2$.
  \item Calcular $\beta_0\beta_2 = -1$ y $\beta_1\beta_3 = -1$
        \emph{(expand: the sixteen exponents obtained cover
        $1, \dots, 16$ exactly once)}.
  \item Deducir $\beta_0 = \frac{\eta_0 + \sqrt{\eta_0^2 + 4}}2$
        (verifique el signo numéricamente: $\beta_0 \approx 2.05$) y
        la fórmula análoga para $\beta_1$; por lo tanto $L_2 =
        \Q(\beta_0)$, cuadrático sobre$L_1$.
  \item Dejemos $\gamma_0 = \zeta + \zeta^{16} =
        2\cos\frac{2\pi}{17}$y$\gamma_1 = \zeta^{13} +
        \zeta^4$. Mostrar$\gamma_0 + \gamma_1 = \beta_0$ y
        $\gamma_0\gamma_1 = \beta_1$, de modo que $\gamma_0 =
        \frac{\beta_0 + \sqrt{\beta_0^2 - 4\beta_1}}2$.
  \item Armar la cadena de fórmulas que expresan
        $\cos\frac{2\pi}{17}$ por raíces cuadradas anidadas y dar un
        verificación decimal ($\cos\frac{2\pi}{17} \approx 0.93247$).
\end{enumerate}

\medskip
\noindent\textbf{Parte IV --- Epilogue.}
\begin{enumerate}[resume]
  \item ¿Dónde exactamente se usó el argumento de que $17$ es un
        \emph{Fermat primer} ($17 = 2^{2^2} + 1$)? Muéstralo por
        un $p$ primo, el $p$-gon normal es \omterm{def:b3:galois:constructible}{construible} iff $p
        = 2^{2^t} + 1$para algunos$t$\emph{(si$p - 1 = 2^m$,
        show $m$ debe ser en sí mismo una potencia de $2$)}.
  \item Deduzca la lista completa de \omterm{def:b3:galois:constructible}{construible} regulares
        $n$-gons para $n \leq 20$, usando Gauss--Wantzel
        criterio de \cref{cor:b3:galois:impossible}.
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Gauss sums and quadratic
reciprocity.} Los períodos de la Parte II esconden un tesoro. para un
primo impar $p$, el \emph{símbolo de leyenda} $\bigl(\frac
ap\bigr)$es$+1$si$a$es un mod cuadrado distinto de cero$p$,$-1$ si
no lo es, $0$ si $p \mid a$; \cref{exo:b3:galois:4}(b)
(El criterio de Euler) da $\bigl(\frac ap\bigr) \equiv
a^{(p-1)/2} \pmod p$, de ahí la multiplicatividad. Escriba$\zeta =
\eu^{2\iu\pi/p}$,$p^* = (-1)^{(p-1)/2}p$ y defina el
\emph{Suma de Gauss}
\[
g \;=\; \sum_{a=1}^{p-1}\Bigl(\frac ap\Bigr)\zeta^a .
\]
\begin{enumerate}[resume]
  \item Mostrar $\sum_{a=1}^{p-1}\bigl(\frac ap\bigr) = 0$
        (tantos cuadrados como no cuadrados), y demostrar la
        forma alternativa $g = \sum_{a=0}^{p-1}\zeta^{a^2}$
        \emph{(each nonzero square is hit twice, and
        $\sum_{a}\zeta^a = 0$)}. Para $p = 17$: relacione $g$ con
        los periodos de la Parte II --- show $g = \eta_0 - \eta_1$
        \emph{(los cuadrados mod $17$ son exactamente pares
        potencias del generador $3$)}.
  \item Demostrar $g^2 = p^*$: ampliar
        \[
        g^2 = \sum_{a,b\neq0}\Bigl(\frac{ab}p\Bigr)
        \zeta^{a+b}
        = \sum_{c}\ \sum_{a \neq 0}\Bigl(\frac{a(c -
        a)}p\Bigr)\zeta^{c}
        \]
        (establezca $b = c - a$), sustituya $c - a = at$ para evaluar
        la suma interna como $\bigl(\frac{-1}p\bigr)(p - 1)$ para
        $c = 0$ y $-\bigl(\frac{-1}p\bigr)$ en caso contrario, y
        Concluya con la pregunta 14. Marque numéricamente: para $p =
        17$,$(\eta_0 - \eta_1)^2 = 17$ (Parte II).
  \item Deduzca $\sqrt{p^*} \in \Q(\zeta_p)$ y concluya
        que el subcampo cuadrático \emph{único} de
        $\Q(\zeta_p)$ es $\Q(\sqrt{p^*})$ --- único porque
        $\operatorname{Gal}(\Q(\zeta_p)/\Q)$ es cíclico
        (\cref{thm:b3:galois:cyclotomicirred}) y un cíclico
        El grupo tiene exactamente un subgrupo del índice $2$. (Cada
        campo cuadrático incrustado en algún campo ciclotómico ---
        Éste es el primer caso del Kronecker--Weber.
        teorema, cuya forma general está muy por delante.)
  \item Ahora sea $q \neq p$ otro primo impar. trabajando en
        el anillo $\Z[\zeta]$ módulo $q$, probar
        \[
        g^q \equiv \Bigl(\frac qp\Bigr)\,g \pmod{q\Z[\zeta]}
        \]
        \emph{(freshman's dream: $(x + y)^q \equiv x^q + y^q$
        mod $q$ in any commutative ring; then $g^q \equiv
        \sum_a\bigl(\frac ap\bigr)^q\zeta^{aq}$, reindex$b =
        aq$and pull out$\bigl(\frac{q^{-1}}p\bigr) =
        \bigl(\frac qp\bigr)$)}.
  \item Por otro lado, $g^q = g\,(g^2)^{(q-1)/2} =
        g\,(p^*)^{(q-1)/2}$; utilizando el criterio de Euler mod$q$,
        deduzca $g^q \equiv \bigl(\frac{p^*}q\bigr)g
        \pmod{q\Z[\zeta]}$, luego --- multiplicando los dos
        expresiones para $g^q$ por $g$ y usando $g^2 = p^*$,
        mod invertible $q$ --- concluir
        \[
        \Bigl(\frac qp\Bigr) = \Bigl(\frac{p^*}q\Bigr) .
        \]
        \emph{(Why does a congruence between the integers
        $\pm p^*$ modulo $q\Z[\zeta]$ imply their equality?
        Intersect with $\Z$.)}
  \item Desplegar $\bigl(\frac{p^*}q\bigr) =
        \bigl(\frac{-1}q\bigr)^{(p-1)/2}\bigl(\frac pq\bigr)$
        y $\bigl(\frac{-1}q\bigr) = (-1)^{(q-1)/2}$ a
        obtener el \emph{ley de reciprocidad cuadrática}:
        \[
        \Bigl(\frac pq\Bigr)\Bigl(\frac qp\Bigr)
        = (-1)^{\frac{p-1}2\cdot\frac{q-1}2} .
        \]
        Verifíquelo en $(p, q) = (17, 3)$ enumerando los cuadrados
        mod $17$ y mod $3$, y úsalo para decidir en tres
        líneas si $x^2 \equiv 219 \pmod{383}$ es \omterm{def:b3:groups:derived}{soluble}
        ($383$ es primo, $219 = 3\cdot73$).
\end{enumerate}

\noindent\textbf{Part VI --- Counting \omterm{def:b3:rings:divisibility}{irreducible} polynomials:
the prime number theorem of $\mathbb F_q[X]$.} Arreglar una potencia primaria
$q$ y sea $N_q(n)$ el número de \emph{monic \omterm{def:b3:rings:divisibility}{irreducible}}
polinomios de \omterm{def:b3:galois:extension}{grado} $n$ sobre $\mathbb F_q$; recordar de
\cref{exo:b3:galois:6} la factorización de $X^{q^n} - X$ y
la identidad $q^n = \sum_{d\mid n}d\,N_q(d)$, que ahora
invertir, reinterpretar y explotar.
\begin{enumerate}[resume]
  \item (Palabras) Llamar una palabra $w \in \mathbb F_q^n$
        \emph{primitivo} si no es una potencia $u^{n/d} =
        u\cdots u$de una palabra estrictamente más corta$u$, y deja
        $A(d)$ sea el número de palabras primitivas de longitud $d$.
        Demuestre que cada palabra de longitud $n$ es únicamente una potencia.
        de una palabra primitiva de cierta longitud $d \mid n$, de modo que
        $q^n = \sum_{d \mid n}A(d)$; comparando con
        \cref{exo:b3:galois:6}, concluir $A(d) = d\,N_q(d)$
        para cada $d$, y explique esta coincidencia mediante una
        biyección explícita: un elemento $\alpha \in
        \mathbb F_{q^d}$de \omterm{def:b3:galois:extension}{grado}$d$ tiene \omterm{thm:b3:galois:finitefields}{Frobenius} \omterm{def:b3:groups:action}{órbita}
        $(\alpha, \alpha^q, \dots, \alpha^{q^{d-1}})$ de
        exactamente $d$ elementos distintos, y elementos de \omterm{def:b3:galois:extension}{grado}
        $d$ corresponde $d$ a uno a \omterm{def:b3:rings:divisibility}{irreductibles} de \omterm{def:b3:galois:extension}{grado}
        $d$.
  \item Demuestre el \emph{Möbius inversion formula}: si $f(n) =
        \sum_{d\mid n}g(d)$para todos los$n$, entonces$g(n) =
        \sum_{d\mid n}\mu(d)\,f(n/d)$, donde$\mu$ es el
        función de Möbius ($\mu(m) = (-1)^{\#\text{prime
        factors}}$si$m$no tiene cuadrados,$0$ en caso contrario)
        \emph{(key lemma: $\sum_{d \mid m}\mu(d) = 0$ for $m >
        1$ --- pair the divisors with and without a fixed prime
        factor)}. deducir
        \[
        N_q(n) = \frac1n\sum_{d \mid n}\mu(d)\,q^{n/d} .
        \]
  \item Mostrar $N_q(n) \geq \frac1n\bigl(q^n - 2q^{n/2}\bigr) > 0$
        por cada $n \geq 1$: una nueva prueba de que $\mathbb F_{q^n}$
        existe para todos los $n$. Interpretar el término principal: un azar.
        polinomio mónico de \omterm{def:b3:galois:extension}{grado} $n$ es \omterm{def:b3:rings:divisibility}{irreducible} con
        probabilidad $\sim \frac1n$ --- el análogo \omterm{prop:b3:galois:perfect}{perfecto} de
        el teorema de los números primos, con $\log x$ intercambiado por $n$;
        verificar numéricamente para $q = 2$, $n \leq 4$
        (\cref{exo:b3:galois:6} enumera los recuentos).
  \item Demuestre el compañero multiplicativo de la pregunta 21:
        \[
        \prod_{\substack{\pi \text{ monic irred.}\\ \deg\pi =
        n}}\pi \;=\; \prod_{d \mid n}
        \bigl(X^{q^d} - X\bigr)^{\mu(n/d)}
        \]
        \emph{(Möbius inversion in the abelian group of
        nonzero rational functions)}; verifíquelo a mano para $q
        = 2$,$n = 2$:$(X^4 - X)/(X^2 - X) = X^2 + X + 1$.
\end{enumerate}
