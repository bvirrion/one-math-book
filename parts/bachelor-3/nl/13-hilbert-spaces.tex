\chapter{Hilbert-ruimten}\label{ch:b3:hilbert}

Een \omterm{def:b3:hilbert:inner}{Hilbert-ruimte} is een Banachruimte waarvan de norm afkomstig
is van een \omterm{def:b3:hilbert:inner}{inwendig product} --- en die ene extra structuur herstelt,
in oneindige dimensie, vrijwel de gehele Euclidische meetkunde:
orthogonale projecties bestaan, elke continue functionaal is een
\omterm{def:b3:hilbert:inner}{inwendig product} met een vaste vector (Riesz), en orthonormale
bases ontwikkelen elke vector in een convergente reeks met
PythagoreÃ¯sche boekhouding (\omterm{thm:b3:hilbert:parseval}{Parseval}). Het hoogtepunt van het
hoofdstuk is een ingeloste schuld: het trigonometrische stelsel
is een orthonormale basis van $L^2$, zodat de identiteit van
\omterm{thm:b3:hilbert:parseval}{Parseval} geldt voor \emph{elke} kwadratisch \omterm{def:b3:lebesgue:l1}{integreerbare
functie} --- de uitspraak die Bachelor jaar~2 alleen kon bewijzen
voor stuksgewijs $\mathcal C^1$-functies. We eindigen met
Lax--Milgram, het werkpaardlemma van de variationele aanpak van
differentiaalvergelijkingen.

Doorheen is $H$ een vectorruimte over $K = \R$ of $\C$.

\section{Inwendige producten; de projectiestelling}

\begin{definition}\label{def:b3:hilbert:inner}
Een \emph{inwendig product}\index{inwendig product} is een
afbeelding $\langle \cdot,\cdot\rangle \colon H\times H \to
K$, lineair in de tweede variabele, met $\langle y, x\rangle =
\overline{\langle x, y\rangle}$ en $\langle x, x\rangle > 0$
voor $x \neq 0$. Het induceert de norm $\norm x = \langle x,
x\rangle^{1/2}$, de \emph{ongelijkheid van Cauchy--Schwarz}
$\abs{\langle x, y\rangle} \leq \norm x\norm y$ (het bewijs
van Jaar~2 --- de discriminant --- blijft ongewijzigd), en de
\emph{parallellogramwet}
\[
\norm{x + y}^2 + \norm{x - y}^2 = 2\norm x^2 + 2\norm y^2 .
\]
Een \emph{Hilbert-ruimte}\index{Hilbert-ruimte} is een
inwendig-productruimte die \omterm{def:b3:complete:complete}{volledig} is voor deze norm.
Voorbeelden: $\ell^2$ (\cref{pb:b3:banach:1}) en, het
fundamentele, $L^2(\mu)$ met $\langle f, g\rangle =
\int\bar fg\,\dd\mu$ --- \omterm{def:b3:complete:complete}{volledig} door Riesz--Fischer
(\cref{thm:b3:lp:complete}); het inwendig product is
eindig door Cauchy--Schwarz ($= $ HÃ¶lder bij $p = q = 2$).
\end{definition}

\begin{theorem}[Projectie op een gesloten convexe verzameling]
\label{thm:b3:hilbert:projection}
Zij $C \neq \varnothing$ een gesloten \emph{convexe}
deelverzameling van de \omterm{def:b3:hilbert:inner}{Hilbert-ruimte} $H$ en $x \in H$. Er is
een unieke $p_C(x) \in C$ met\index{projectiestelling}
\[
\norm{x - p_C(x)} = d(x, C),
\]
gekarakteriseerd door: $\operatorname{Re}\langle x - p_C(x),\ c -
p_C(x)\rangle \leq 0$ voor alle $c \in C$. De afbeelding $p_C$ is
$1$-Lipschitz.
\end{theorem}

\begin{proof}
Zij $d = d(x, C)$ en $(c_n) \subseteq C$ met $\norm{x - c_n}
\to d$. Parallellogram op $x - c_n$ en $x - c_m$:
\[
\norm{c_n - c_m}^2 = 2\norm{x - c_n}^2 + 2\norm{x - c_m}^2 -
4\,\bigl\|x - \tfrac{c_n + c_m}2\bigr\|^2
\leq 2\norm{x{-}c_n}^2 + 2\norm{x{-}c_m}^2 - 4d^2
\]
(convexiteit plaatst het midden in $C$): de rechterkant gaat
naar $0$, dus $(c_n)$ is Cauchy, en haar limiet $p \in C$
(gesloten) bereikt $d$. Uniciteit: twee minimizers geven, via
dezelfde identiteit, $\norm{p - p'}^2 \leq 2d^2 + 2d^2 -
4d^2 = 0$.

Karakterisering: voor $c \in C$, $t \in \intoc01$, de vector
$p + t(c - p) \in C$, dus
\[
d^2 \leq \norm{x - p - t(c-p)}^2
= d^2 - 2t\operatorname{Re}\langle x - p, c - p\rangle +
t^2\norm{c-p}^2 ;
\]
deel door $t \to 0^+$: $\operatorname{Re}\langle x - p, c -
p\rangle \leq 0$. Omgekeerd geeft deze ongelijkheid $\norm{x -
c}^2 = \norm{x - p}^2 - 2\operatorname{Re}\langle x - p, c -
p\rangle + \norm{p - c}^2 \geq \norm{x-p}^2$. Lipschitz: voor
$x, y$ met projecties $p, q$, tel de twee variationele
ongelijkheden op (met $c = q$, resp.\ $c = p$):
$\operatorname{Re}\langle x - y - (p - q), p - q\rangle \geq
0$, dus $\norm{p - q}^2 \leq \operatorname{Re}\langle x - y, p -
q\rangle \leq \norm{x - y}\norm{p - q}$.
\end{proof}

\begin{theorem}[Orthogonale ontbinding]
\label{thm:b3:hilbert:decomposition}
Zij $F$ een \emph{gesloten deelruimte} van $H$. Dan is $p_F$
lineair, $x - p_F(x) \perp F$ voor alle $x$, en\index{orthogonaal
complement}
\[
H = F \oplus F^\perp,
\qquad F^\perp = \{y : \langle y, f\rangle = 0\ \forall f\in
F\},
\qquad (F^\perp)^\perp = F .
\]
Voor een algemene deelruimte geldt $(F^\perp)^\perp = \bar F$;
in het bijzonder is $F$ \omterm{def:b3:topology:interior}{dicht} desda $F^\perp = \{0\}$.
\end{theorem}

\begin{proof}
Voor een deelruimte dwingt de variationele karakterisering met
$c = p_F(x) \pm f$ ($f \in F$, beide tekens, en $\iu f$ in het
complexe geval) $\langle x - p_F(x), f\rangle = 0$: de rest is
orthogonaal op $F$. Ontbinding $x = p_F(x) + (x - p_F(x))$ met
$F \cap F^\perp = \{0\}$ ($\langle y, y \rangle = 0$);
lineariteit van $p_F$ volgt uit uniciteit van zulke
ontbindingen (beide zijden lineair daarin). $(F^\perp)^\perp
\supseteq F$ altijd; omgekeerd, als $x \perp F^\perp$, schrijf
$x = f + g$: $g = x - f \in F^\perp$ en $\langle g, g\rangle
= \langle x, g\rangle - \langle f, g\rangle = 0$: $x = f \in
F$. Voor een algemene deelruimte $F$: $F^\perp = \bar
F^{\,\perp}$ (continuÃ¯teit van het \omterm{def:b3:hilbert:inner}{inwendig product}), dus
$(F^\perp)^\perp = \bar F$ via het gesloten geval; \omterm{ex:b3:lebesgue:gamma}{dichtheid}
desda $\bar F = H$ desda $F^\perp = 0$.
\end{proof}

\begin{theorem}[Representatiestelling van Riesz]\label{thm:b3:hilbert:riesz}
Voor elke continue lineaire functionaal $\varphi \in H'$ is er
een unieke $a \in H$ met\index{representatiestelling van Riesz}
\[
\varphi(x) = \langle a, x\rangle \quad (x \in H),
\qquad \norm\varphi_{H'} = \norm a .
\]
\end{theorem}

\begin{proof}
Als $\varphi = 0$: $a = 0$. Anders is $F = \ker\varphi$ een
gesloten echte deelruimte; kies $u \in F^\perp$, $\norm u = 1$
(\cref{thm:b3:hilbert:decomposition}: $F^\perp \neq 0$ omdat
$F \neq H$). Voor willekeurige $x$ geldt $\varphi(x)u -
\varphi(u)x \in \ker\varphi$, dus $\perp u$:
\[
0 = \langle u, \varphi(x)u - \varphi(u)x\rangle
= \varphi(x) - \varphi(u)\langle u, x\rangle :
\qquad \varphi(x) = \langle
\overline{\varphi(u)}\,u,\ x\rangle .
\]
Dus $a = \overline{\varphi(u)}u$ werkt. Uniciteit: $\langle a
- a', x\rangle = 0$ voor alle $x$, test $x = a - a'$. Normen:
$\abs{\varphi(x)} \leq \norm a\norm x$ (Cauchy--Schwarz) met
gelijkheid bij $x = a$.
\end{proof}

\begin{example}[Een projectie, tot het eind uitgerekend]
\label{ex:b3:hilbert:projectionexample}
In $H = L^2(\intcc01)$, wat is de beste benadering van
$f(x) = x^2$ door een affiene functie? De deelruimte $F =
\operatorname{Vect}(1, x)$ is gesloten (eindig-dimensionaal),
en $p_F(f) = a + bx$ wordt gekarakteriseerd door orthogonaliteit
van de rest op $1$ en op $x$:
\[
\int_0^1(x^2 - a - bx)\,\dd x = 0,
\qquad
\int_0^1x\,(x^2 - a - bx)\,\dd x = 0,
\]
d.w.z.\ $\frac13 = a + \frac b2$ en $\frac14 = \frac a2 +
\frac b3$: $a = -\frac16$, $b = 1$. Dus $p_F(x^2) = x -
\frac16$, en de fout is
\[
d(f, F)^2 = \int_0^1\Bigl(x^2 - x + \frac16\Bigr)^2\dd x =
\frac1{180},
\qquad d(f, F) = \frac1{6\sqrt5} .
\]
Twee opmerkingen om te internaliseren. Ten eerste is de
berekening niets anders dan een $2\times2$-lineair stelsel ---
de \emph{normaalvergelijkingen}; voor de monoomiale basis is
hun matrix $\bigl(\frac1{i+j+1}\bigr)$ de berucht slecht
geconditioneerde Hilbert-matrix, en eerst orthogonaliseren
(Legendre-polynomen, \cref{pb:b3:hilbert:1}) is het
medicijn. Ten tweede is de beste \emph{uniforme} benadering
van $x^2$ door affiene functies anders ($x - \frac18$, via
equioscillatie): elke norm heeft haar eigen meetkunde, en
alleen de Hilbertse beantwoordt met een lineair stelsel.
\end{example}

\section{Orthonormale bases}

\begin{definition}\label{def:b3:hilbert:onb}
Een familie $(e_i)_{i\in I}$ is \emph{orthonormaal} als
$\langle e_i, e_j\rangle = \delta_{ij}$, en een
\emph{Hilbert-basis}\index{Hilbert-basis} (orthonormale basis)
als bovendien haar eindige lineaire combinaties \omterm{def:b3:topology:interior}{dicht} liggen in
$H$ (de familie is \emph{totaal}). We behandelen het aftelbare
geval $I = \N$, dat via Gram--Schmidt elke \emph{separabele}
$H$ dekt (\cref{prop:b3:hilbert:gramschmidt}).
\end{definition}

\begin{theorem}[Bessel, Parseval]\label{thm:b3:hilbert:parseval}
Zij $(e_n)_{n\in\N}$ orthonormaal in $H$, en $c_n(x) =
\langle e_n, x\rangle$.
\begin{enumerate}
  \item (Bessel) $\sum_n\abs{c_n(x)}^2 \leq \norm x^2$, en
        de reeks $\sum_nc_n(x)e_n$ convergeert in $H$, met
        som $p_F(x)$, $F = \overline{\operatorname{Vect}}(e_n)$.
  \item Het volgende is equivalent: (i) $(e_n)$ is een
        \omterm{def:b3:hilbert:onb}{Hilbert-basis}; (ii) $x = \sum_nc_n(x)e_n$ voor elke
        $x$; (iii) \emph{Parseval}\index{identiteit van
        Parseval}: $\norm x^2 = \sum_n\abs{c_n(x)}^2$ voor
        elke $x$; (iv) de enige vector orthogonaal op alle
        $e_n$ is $0$.
  \item Als $(e_n)$ een \omterm{def:b3:hilbert:onb}{Hilbert-basis} is, is $x \mapsto
        (c_n(x))_n$ een isometrisch isomorfisme $H \to
        \ell^2$ (\emph{elke} oneindig-dimensionale separabele
        \omterm{def:b3:hilbert:inner}{Hilbert-ruimte} ``is'' $\ell^2$), en $\langle x,
        y\rangle = \sum_n\overline{c_n(x)}c_n(y)$.
\end{enumerate}
\end{theorem}

\begin{proof}
(1) Voor eindige $N$: $x - \sum_{n\leq N}c_ne_n \perp e_k$
($k \leq N$), dus Pythagoras geeft $\norm x^2 = \sum_{n\leq
N}\abs{c_n}^2 + \norm{x - \sum_{n\leq N}c_ne_n}^2$: Bessel.
De deelsommen $S_N = \sum_{n\leq N}c_ne_n$ zijn Cauchy:
$\norm{S_N - S_M}^2 = \sum_{M<n\leq N}\abs{c_n}^2$, staart
van een convergente reeks; de limiet ligt in $F$, en $x -
\lim S_N \perp$ elke $e_k$ (continuÃ¯teit), dus $\perp F$:
via uniciteit van de orthogonale ontbinding, $\lim S_N =
p_F(x)$.

(2) (i)$\Rightarrow$(ii): $F = H$, dus $p_F = \mathrm{id}$.
(ii)$\Rightarrow$(iii): Pythagoras in de limiet
($\norm{S_N}^2 = \sum_{n \leq N}\abs{c_n}^2 \to \norm
x^2$). (iii)$\Rightarrow$(iv): $x \perp$ alle $e_n$ geeft
$\norm x^2 = 0$. (iv)$\Rightarrow$(i): $F^\perp = \{0\}$
(orthogonaliteit op alle $e_n$ is orthogonaliteit op $F$),
dus $F$ is \omterm{def:b3:topology:interior}{dicht} door \cref{thm:b3:hilbert:decomposition};
maar $F$, een \omterm{def:b3:topology:interior}{afsluiting}, is al gesloten: $F = H$.

(3) De afbeelding is lineair, isometrisch via (iii) (dus
injectief), en surjectief: gegeven $(c_n) \in \ell^2$,
convergeert de reeks $\sum c_ne_n$ (Cauchy zoals in (1))
naar een originelen. De inwendig-productformule is
polarisatie uit (iii), of een directe limietberekening.
\end{proof}

\begin{proposition}[Gram--Schmidt]\label{prop:b3:hilbert:gramschmidt}
Zij $(x_n)$ een lineair onafhankelijke rij. Stel inductief
$\tilde e_n = x_n - \sum_{k<n}\langle e_k, x_n\rangle e_k$ en
$e_n = \tilde e_n/\norm{\tilde e_n}$ produceert een
orthonormale $(e_n)$ met dezelfde eindige opspanningen:
$\operatorname{Vect}(e_1, \dots, e_n) =
\operatorname{Vect}(x_1, \dots, x_n)$. Bijgevolg heeft elke
separabele \omterm{def:b3:hilbert:inner}{Hilbert-ruimte} (Ã©Ã©n met een aftelbare dichte
deelverzameling) een \omterm{def:b3:hilbert:onb}{Hilbert-basis}.\index{Gram--Schmidt-orthonormalisatie}
\end{proposition}

\begin{proof}
Inductie: $\tilde e_n \perp e_k$ ($k < n$) per constructie,
en $\tilde e_n \neq 0$ door onafhankelijkheid; de opspanningen
kloppen in elke stap (driehoekige basistransformatie). Voor
een separabele $H$: uit een dichte rij extraheer een lineair
onafhankelijke subfamilie met dichte opspanning (verwerp
elke vector in de opspanning van haar voorgangers --- de
opspanning blijft gelijk), orthonormaliseer: het resultaat is
totaal.
\end{proof}

\begin{theorem}[Het trigonometrische stelsel; Parseval eindelijk]
\label{thm:b3:hilbert:fourier}
In $L^2(\intcc{-\pi}\pi)$ met $\langle f, g\rangle =
\frac1{2\pi}\int_{-\pi}^\pi \bar fg$ is de familie $e_n(t) =
\eu^{\iu nt}$, $n \in \Z$, een \omterm{def:b3:hilbert:onb}{Hilbert-basis}. Bijgevolg,
voor \emph{elke} $f \in L^2$ --- in het bijzonder elke
stuksgewijs continue $2\pi$-periodieke $f$ --- met $c_n(f) =
\frac1{2\pi}\int_{-\pi}^{\pi}f(t)\eu^{-\iu nt}\dd t$:
\[
f = \sum_{n\in\Z}c_n(f)\,\eu^{\iu nt} \ \ \text{in } L^2,
\qquad
\frac1{2\pi}\int_{-\pi}^{\pi}\abs f^2 =
\sum_{n\in\Z}\abs{c_n(f)}^2 .
\]
Dit bewijst, in volle algemeenheid, de Parseval-identiteit die
Jaar~2 toegaf.\index{identiteit van Parseval (algemeen)}
\end{theorem}

\begin{proof}
Orthonormaliteit is een directe berekening (Jaar~2).
Totaliteit: zij $f \in L^2$ $\perp$ alle $e_n$, d.w.z.\ alle
FouriercoÃ«fficiÃ«nten verdwijnen. Continue $2\pi$-periodieke
functies zijn \omterm{def:b3:topology:interior}{dicht} in $L^2(\intcc{-\pi}\pi)$: inderdaad is
$\mathcal C_c(\intoo{-\pi}\pi)$ \omterm{def:b3:topology:interior}{dicht}
(\cref{thm:b3:lp:density}(2)) en zulke functies strekken
periodiek en continu uit. Trigonometrische veeltermen zijn
$\norm\cdot_\infty$-dicht onder continue periodieke functies
(Stone--Weierstrass, \cref{cor:b3:complete:weierstrass}(c)),
en $\norm\cdot_2 \leq \norm\cdot_\infty$: trigonometrische
veeltermen zijn \omterm{def:b3:topology:interior}{dicht} in $L^2$. Maar $f \perp$ elke
trigonometrische veelterm, dus $f \perp$ een dichte
deelruimte: $f \in (\text{dicht})^\perp = \{0\}$
(\cref{thm:b3:hilbert:decomposition}). Criterium (iv) van
\cref{thm:b3:hilbert:parseval} concludeert; (ii) en (iii)
pakken zich uit tot de weergave (herindexering van het
aftelbare $\Z$; de tweezijdige reeks convergeert
onvoorwaardelijk --- de deelsommen over elke uitputtende
familie convergeren, via het $\ell^2$-staartargument).
\end{proof}

\begin{theorem}[Lax--Milgram]\label{thm:b3:hilbert:laxmilgram}
Zij $H$ een reÃ«le \omterm{def:b3:hilbert:inner}{Hilbert-ruimte} en $a \colon H\times H \to
\R$ bilineair, \emph{continu} ($\abs{a(u,v)} \leq M\norm
u\norm v$) en \emph{coÃ«rcief} ($a(u, u) \geq \alpha\norm u^2$,
$\alpha > 0$). Dan is er voor elke $\varphi \in H'$ een
unieke $u \in H$ met\index{stelling van Lax--Milgram}
\[
a(u, v) = \varphi(v) \qquad \text{voor alle } v \in H .
\]
\end{theorem}

\begin{proof}
Voor vaste $u$ is $v \mapsto a(u, v)$ continu lineair: Riesz
geeft een unieke $Au \in H$ met $a(u,v) = \langle Au,
v\rangle$; $A$ is lineair met $\norm{Au} \leq M\norm u$
(uniciteit van representanten, daarna de grens).
CoÃ«rciviteit: $\alpha\norm u^2 \leq a(u,u) = \langle Au,
u\rangle \leq \norm{Au}\norm u$, dus $\norm{Au} \geq
\alpha\norm u$: $A$ is injectief met gesloten beeld (een
Cauchy-beeldrij $Au_n$ dwingt $u_n$ Cauchy). Het beeld is
\omterm{def:b3:topology:interior}{dicht}: $w \perp \operatorname{im}A$ geeft $0 = \langle Aw,
w\rangle \geq \alpha\norm w^2$. Gesloten en \omterm{def:b3:topology:interior}{dicht}: $A$ is
bijectief. Gegeven $\varphi$, laat $f$ haar representeren
(Riesz) en $u = A^{-1}f$: $a(u, v) = \langle f, v\rangle =
\varphi(v)$, uniek ($a(u - u', \cdot) = 0$ en
coÃ«rciviteit).
\end{proof}

\end{remark}

met $\langle y, x_{k_j}\rangle \to \langle y, x\rangle$ voor

(b) Toon $e_n \rightharpoonup 0$ maar $\norm{e_n} = 1$:

\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:hilbert:8}
(Toegevoegden) Voor $T \in \mathcal L(H)$, toon dat er een

\begin{exercise}[$\star\star$]\label{exo:b3:hilbert:9}

Toon dat $u$ $a(u, \cdot) = \varphi$ oplost desda $u$

\emph{(Maak het kwadraat \omterm{def:b3:complete:complete}{volledig}: $J(u + w) - J(u) =

\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b3:hilbert:10}
(Het Haar-stelsel) Op $\intcc01$, zij $h_{0} = \mathbf 1$, en

\[

\]

\begin{exercise}[$\star\star$]\label{exo:b3:hilbert:11}
(Orthogonale projecties, gekarakteriseerd) Zij $H$ een
\omterm{def:b3:hilbert:inner}{Hilbert-ruimte} en $P \in \mathcal L(H)$ met $P^2 = P$, $P
\neq 0$. Toon de equivalentie van:
(i) $P$ is de orthogonale projectie op $\operatorname{im}P$;
(ii) $P = P^*$ (\cref{exo:b3:hilbert:8});
(iii) $\vertiii P = 1$.
\emph{(Voor (iii) $\Rightarrow$ (i): als een $x \in
(\ker P)^\perp$ $Px \neq x$ had, beschouw $x + t(Px - x)$
--- of direct: voor $u \in \operatorname{im}P$ en $v \in
\ker P$, ontwikkel $\norm{P(u + tv)}^2 \leq \norm{u + tv}^2$
voor alle $t \in \R$ en concludeer $\langle u, v\rangle =

bereken haar norm.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b3:hilbert:12}
(De ergodische stelling van von Neumann) Zij $U \in \mathcal
L(H)$ \emph{unitair} ($U^*U = UU^* = I$), $F = \ker(U - I)$
de vaste-ruimte, $P$ de orthogonale projectie op $F$, en
$A_n = \frac1n\sum_{k=0}^{n-1}U^k$.
(a) Toon $\ker(U - I) = \ker(U^* - I)$ \emph{(uit
$\norm{Ux - x}^2 = 2\norm x^2 - 2\operatorname{Re}\langle
Ux, x\rangle$ en unitariteit)}, en concludeer
$\overline{\operatorname{im}(U - I)} = F^\perp$.
(b) Toon dat $A_nx \to x$ voor $x \in F$, en $A_nx \to 0$
voor $x \in \operatorname{im}(U - I)$ \emph{(telescoperen)},
dan voor $x \in \overline{\operatorname{im}(U - I)}$
(uniforme grens $\vertiii{A_n} \leq 1$).
(c) Concludeer: $A_nx \to Px$ voor \emph{elke} $x \in H$ ---
tijdgemiddelden convergeren naar de projectie op de
invarianten.
(d) Werk het uit voor $H = L^2(\R/\Z)$ en $Uf = f(\cdot +
\alpha)$ met $\alpha$ irrationaal: identificeer $F$ (gebruik
Fourierreeksen, \cref{thm:b3:hilbert:fourier}) en concludeer
dat $\frac1n\sum_{k<n}f(x + k\alpha) \to \int_0^1f$ in
$L^2$: de $L^2$-equidistributie van irrationale rotaties.
\end{exercise}

\section{Probleem: orthogonale polynomen}

\begin{problem}[{Weekendopgave --- Legendre, Hermite, en
Gauss-kwadratuur}]\label{pb:b3:hilbert:1}
Zij $I \subseteq \R$ een interval en $w > 0$ een continu
\emph{gewicht} op het inwendige van $I$ zodat $\int_I
\abs t^nw(t)\dd t < \infty$ voor alle $n$; werk in $H =
L^2(I, w\,\dd\lambda)$ met $\langle f, g\rangle = \int_I \bar
fg\,w$. Gram--Schmidt toegepast op $1, t, t^2, \dots$
produceert de \emph{\omterm{pb:b3:hilbert:1}{orthogonale polynomen}} $(p_n)$ voor $w$
(monische normalisatie: $p_n = t^n + \cdots$).\index{orthogonale
polynomen}

\medskip
\noindent\textbf{Deel I --- Algemene theorie.}
\begin{enumerate}
  \item Toon dat $p_n$ orthogonaal is op elke polynoom van
        graad $< n$, en dat $(p_0, \dots, p_n)$ een basis
        is van $\R_n[t]$.
  \item (Drietermenrecurrentie) Toon dat er reÃ«le $a_n,
        b_n$ bestaan met
        \[
        p_{n+1}(t) = (t - a_n)\,p_n(t) - b_n\,p_{n-1}(t),
        \qquad b_n = \frac{\norm{p_n}^2}{\norm{p_{n-1}}^2} >
        0 .
        \]
        \emph{(Ontwikkel $t\,p_n$ in de basis $(p_k)_{k \leq
        n+1}$ en dood coÃ«fficiÃ«nten door orthogonaliteit,
        met $\langle tp_n, p_k\rangle = \langle p_n,
        tp_k\rangle$.)}
  \item (Wortels) Toon dat $p_n$ $n$ \emph{onderscheiden}
        wortels heeft, alle inwendig in $I$. \emph{(Zij $t_1
        < \dots < t_m$ de inwendige tekenwisselingen van
        $p_n$; als $m < n$, test $p_n$ tegen
        $\prod_{i\leq m}(t - t_i)$ en weerspreek
        orthogonaliteit.)}
\end{enumerate}

\medskip
\noindent\textbf{Deel II --- Legendre ($I = \intcc{-1}1$, $w =
1$).} Definieer $P_n(t) = \frac{1}{2^nn!}\,\frac{\dd^n}{\dd
t^n}\bigl[(t^2 - 1)^n\bigr]$ (Rodrigues).
\begin{enumerate}[resume]
  \item Toon $\deg P_n = n$ met leidende coÃ«fficiÃ«nt
        $\frac{(2n)!}{2^n(n!)^2}$, en, $n$ keer partieel
        integrerend, dat $\langle P_n, Q\rangle = 0$ voor
        elke polynoom $Q$ van graad $< n$: de $P_n$ zijn (op
        normalisatie na) de \omterm{pb:b3:hilbert:1}{orthogonale polynomen} voor $w =
        1$.
  \item Bereken $\norm{P_n}_2^2 = \frac{2}{2n+1}$
        \emph{(integreer $n$ keer partieel tegen zichzelf en
        reduceer tot een Beta/Wallis-integraal,
        \cref{exo:b3:product:8})}.
  \item Toon dat de genormaliseerde Legendre-polynomen een
        \omterm{def:b3:hilbert:onb}{Hilbert-basis} van $L^2(\intcc{-1}1)$ vormen
        \emph{(Weierstrass, \cref{cor:b3:complete:weierstrass},
        plus \omterm{ex:b3:lebesgue:gamma}{dichtheid} van $\mathcal C$ in $L^2$)}, en
        ontwikkel $f(t) = \abs t$ tot graad $2$: bereken de
        beste kwadratische $L^2$-benadering van $\abs t$.
\end{enumerate}

\medskip
\noindent\textbf{Deel III --- Hermite ($I = \R$, $w(t) =
\eu^{-t^2}$).} Definieer $H_n(t) =
(-1)^n\eu^{t^2}\frac{\dd^n}{\dd t^n}\eu^{-t^2}$.
\begin{enumerate}[resume]
  \item Toon dat $H_n$ een polynoom van graad $n$ is met
        leidende coÃ«fficiÃ«nt $2^n$, dat $H_{n+1} = 2tH_n -
        H_n'$, en dat $\langle H_m, H_n\rangle_w =
        \delta_{mn}\,2^nn!\sqrt\pi$ \emph{(opnieuw partieel)}.
  \item Toon dat de Hermite-familie totaal is in $L^2(\R,
        \eu^{-t^2}\dd t)$, Ã©Ã©n resultaat uit
        \cref{ch:b3:fouriertransform} toelatend: als $g \in
        L^1(\R)$ $\int g(t)\eu^{-\iu\xi t}\dd t = 0$ heeft
        voor alle $\xi$, dan $g = 0$ a.e. \emph{(Voor $f
        \perp$ alle $H_n$, d.w.z.\ $\perp$ alle polynomen:
        toon dat $z \mapsto \int f(t)\eu^{-t^2}\eu^{-\iu
        zt}\dd t$ welgedefinieerd is, ontwikkel de
        exponentiaal als reeks, rechtvaardig de verwisseling
        door dominering, en concludeer dat de
        Fouriertransformatie van $f\eu^{-t^2}$ verdwijnt.)}
\end{enumerate}

\medskip
\noindent\textbf{Deel IV --- Gauss-kwadratuur.} Fixeer $n$,
laat $t_1 < \dots < t_n$ de wortels van $p_n$ zijn (Deel I),
en definieer de gewichten $w_i = \int_I \ell_i(t)\,w(t)\dd t$
waar $\ell_i$ de Lagrange-interpolatiebasispolynomen in de
$t_i$ zijn.
\begin{enumerate}[resume]
  \item Toon dat de kwadratuurregel $Q(f) = \sum_iw_if(t_i)$
        exact is op alle polynomen van graad $\leq n - 1$
        (interpolatie), en in feite --- het mirakel --- op
        alle polynomen van graad $\leq 2n - 1$: schrijf $P =
        qp_n + r$ en gebruik orthogonaliteit op het quotiÃ«nt
        $q$.
  \item Toon dat de gewichten positief zijn
        \emph{(pas de regel toe op $\ell_i^2$, van graad $2n -

        (\cref{exo:b3:banach:9}) dat Gauss-kwadratuur
        convergeert: $Q_n(f) \to \int_I fw$ voor elke
        continue $f$ op een compact $I$.
  \item Voor $n = 2$, $I = \intcc{-1}1$, $w = 1$: bereken de
        knooppunten $\pm\frac1{\sqrt3}$ en gewichten $1, 1$,
        en verifieer exactheid op $1, t, t^2, t^3$ met de

        twee evaluatiepunten.
\end{enumerate}

\medskip
\noindent\textbf{Deel V --- Chebyshev: de polynomen die het
best oscilleren.} Nu $I = \intcc{-1}1$ en $w(t) =
\frac1{\sqrt{1 - t^2}}$.
\begin{enumerate}[resume]
  \item Toon dat $T_n(\cos\theta) = \cos n\theta$ een polynoom
        $T_n$ van graad $n$ definieert (stel $T_{n+1} =
        2t\,T_n - T_{n-1}$ vast vanuit een trigonometrische
        identiteit), met leidende coÃ«fficiÃ«nt $2^{n-1}$ voor
        $n \geq 1$; en dat de substitutie $t = \cos\theta$
        geeft
        \[
        \langle T_m, T_n\rangle_w =
        \int_0^\pi\cos m\theta\,\cos n\theta\,\dd\theta
        = 0 \ (m \neq n), \qquad
        \norm{T_0}_w^2 = \pi,\ \ \norm{T_n}_w^2 = \frac\pi2 :
        \]
        de $T_n$ zijn de \omterm{pb:b3:hilbert:1}{orthogonale polynomen} voor dit
        gewicht, en Chebyshev-ontwikkelingen \emph{zijn}
        Fourier-cosinusreeksen in vermomming.
  \item Lokaliseer expliciet de $n$ wortels $t_k =
        \cos\frac{(2k-1)\pi}{2n}$ en de $n + 1$ extrema
        $s_j = \cos\frac{j\pi}n$ van $T_n$ op $\intcc{-1}1$,
        waar $T_n(s_j) = (-1)^j$: de grafiek
        \emph{equioscilleert} tussen $\pm1$.
  \item (Minimax) Toon dat onder alle \emph{monische}
        polynomen van graad $n$ de polynoom $2^{1-n}T_n$ de
        kleinste sup-norm heeft op $\intcc{-1}1$, namelijk
        $2^{1-n}$ --- en dat hij de unieke minimizer is.
        \emph{(Als een monische $P$ $\sup\abs P < 2^{1-n}$
        had, zou het verschil $2^{1-n}T_n - P$, van graad
        $\leq n-1$, van teken wisselen in de $n+1$
        equioscillatiepunten.)}
  \item Toepassing op interpolatie: voor knooppunten $t_1 <
        \dots < t_n$ in $\intcc{-1}1$ betrekt de fout van
        Lagrange-interpolatie van een $\mathcal C^n$-functie
        $\omega(t) = \prod_i(t - t_i)$. Toon dat de keuze
        van Chebyshev-wortels als knooppunten
        $\sup_{\intcc{-1}1}\abs\omega$ minimaliseert, en geef
        de resulterende grens $\norm{f - L_nf}_\infty \leq
        \frac{\norm{f^{(n)}}_\infty}{2^{n-1}\,n!}$ ---
        vergelijk met equidistante knooppunten (noem het
        fenomeen van Runge als waarschuwend verhaal).
  \item Verifieer $\abs{T_n'(\pm1)} = n^2$
        \emph{(differentiÃ«er $T_n(\cos\theta) = \cos n\theta$
        en neem limieten $\theta \to 0, \pi$)}: polynomen
        begrensd door $1$ op $\intcc{-1}1$ kunnen afgeleide
        zo groot als $n^2$ hebben aan de rand (de
        ongelijkheid van Markov zegt niet groter --- alleen
        de uitspraak). Waar in het interval is de
        afgeleidegrens slechts $O(n)$?
  \item (Chebyshev--Gauss-kwadratuur) Toon dat de
        Gauss-regel voor het gewicht $w$ in de $n$
        Chebyshev-wortels \emph{gelijke} gewichten $w_i =
        \frac\pi n$ heeft \emph{(exactheid op $T_0, \dots,
        T_{n-1}$ plus de trigonometrische sommen
        $\sum_{k=1}^n\cos\bigl(j\tfrac{(2k-1)\pi}{2n}\bigr)
        = 0$ voor $1 \leq j \leq n - 1$)}: de meest uniforme
        van alle kwadraturen. Schrijf het uit voor $n = 3$.
\end{enumerate}

\medskip
\noindent\textbf{Deel VI --- Christoffel--Darboux,
interlacering, en de Jacobi-matrix.} Terug naar een algemeen
gewicht; $h_k = \norm{p_k}^2$ (monische $p_k$), $b_k =
h_k/h_{k-1}$.
\begin{enumerate}[resume]
  \item (Kleinste norm) Toon dat onder alle \emph{monische}
        polynomen van graad $n$ de orthogonale $p_n$ de
        unieke is van minimale $L^2(w)$-norm ---
        identificeer de minimalisatie als een orthogonale
        projectie op $\R_{n-1}[t]$
        (\cref{thm:b3:hilbert:projection} of de
        eindig-dimensionale projectie van Jaar~2). De
        minimax-eigenschap van vraag 14 is dezelfde
        uitspraak met $L^\infty$ in plaats van $L^2$:
        dezelfde held, twee normen.
  \item (Christoffel--Darboux) Bewijs, door inductie op $n$
        met de drietermenrecurrentie, de identiteit
        \[
        \sum_{k=0}^{n}\frac{p_k(x)\,p_k(y)}{h_k}
        = \frac{p_{n+1}(x)\,p_n(y) -
        p_n(x)\,p_{n+1}(y)}{h_n\,(x - y)}
        \qquad (x \neq y),
        \]
        en haar confluente vorm ($y \to x$):
        $\sum_{k\leq n}\frac{p_k(x)^2}{h_k} =

        opeenvolgende wortels van $p_{n+1}$ ligt precies Ã©Ã©n
        wortel van $p_n$.

\noindent\textbf{Deel VII --- De foutterm, en de kern achter

  \item (Gauss-foutformule) Zij $Hf$ de Hermite-interpolant

        \[
