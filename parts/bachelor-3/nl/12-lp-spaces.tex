\chapter{De \texorpdfstring{$L^p$}{Lp}-ruimten}\label{ch:b3:lp}

De Lebesgue-integraal is gebouwd voor analyse; de $L^p$-ruimten
zijn waar die analyse leeft. Zij zijn Banachruimten
(Riesz--Fischer) --- de voltooiingen die
\cref{exo:b3:complete:1} toonde dat de continue functies
mistten --- en zij ondersteunen een gladmakingstechnologie,
\emph{convolutie tegen \omterm{def:b3:lp:mollifier}{mollifiers}}, die elke $L^p$-functie door
$\mathcal C^\infty$-functies benadert. Dit hoofdstuk bewijst de
integraalversies van HÃ¶lder en Minkowski, volledigheid, de
dichtheidsstellingen, en de regularisatiemachine, eindigend met
de inclusie- en interpolatiegeografie van de $L^p$-schaal.
Doorheen is $(X, \mathcal A, \mu)$ een maatruimte en zijn
functies complexwaardig; op $\R^d$ is de \omterm{def:b3:measure:measure}{maat} $\lambda_d$.

\section{Definitie; HÃ¶lder en Minkowski}

\begin{definition}\label{def:b3:lp:space}
Voor $1 \leq p < \infty$ is $\mathcal L^p(\mu)$ de verzameling
van meetbare $f$ met $\norm f_p = \bigl(\int\abs
f^p\dd\mu\bigr)^{1/p} < \infty$, en $\mathcal L^\infty(\mu)$
de verzameling van $f$ begrensd buiten een nulverzameling, met
$\norm f_\infty$ het \emph{essentiÃ«le supremum} --- de kleinste
$M$ met $\abs f \leq M$ b.o.\ (het inf wordt bereikt: snijd de
nulverzamelingen voor $M + \frac1n$). Omdat $\norm f_p = 0$
alleen $f = 0$ \emph{b.o.} dwingt (\cref{exo:b3:lebesgue:5}),
definiÃ«ren we
\[
L^p(\mu) = \mathcal L^p(\mu)/\{f = 0 \text{ b.o.}\} :
\]
elementen zijn klassen van functies modulo nulverzamelingen, en
$\norm\cdot_p$ is een echte norm op
$L^p$.\index{Lp-ruimte@$L^p$-ruimte}
\end{definition}

\begin{theorem}[Ongelijkheid van HÃ¶lder]\label{thm:b3:lp:holder}
Zijn $1 \leq p, q \leq \infty$ met $\frac1p + \frac1q = 1$
(geconjugeerde exponenten). Voor meetbare $f, g$:\index{ongelijkheid
van HÃ¶lder}
\[
\norm{fg}_1 \leq \norm f_p\,\norm g_q ,
\]
met gelijkheid (voor $1 < p < \infty$, eindige normen, $f,g
\neq0$) d.e.s.d.a.\ $\abs f^p$ en $\abs g^q$ proportioneel
b.o.\ zijn.
\end{theorem}

\begin{proof}
De gevallen $\{p, q\} = \{1, \infty\}$ zijn direct ($\abs{fg}
\leq \norm g_\infty\abs f$ b.o.). Zij $1 < p < \infty$;
normaliseer $\norm f_p = \norm g_q = 1$ (homogeniteit; nul- of
oneindige normen zijn triviaal). Youngs ongelijkheid $ab \leq
\frac{a^p}p + \frac{b^q}q$ ($a, b \geq 0$; concaafheid van
$\ln$, als in \cref{pb:b3:banach:1}) geeft puntsgewijs $\abs{f
g} \leq \frac{\abs f^p}p + \frac{\abs g^q}q$; integreer:
$\norm{fg}_1 \leq \frac1p + \frac1q = 1$. Gelijkheid dwingt
b.o.\ gelijkheid in Young, d.w.z.\ $\abs f^p = \abs g^q$ b.o.\
(na de normalisatie; die ongedaan makend, proportionaliteit).
\end{proof}

\begin{theorem}[Ongelijkheid van Minkowski]\label{thm:b3:lp:minkowski}
Voor $1 \leq p \leq \infty$: $\norm{f + g}_p \leq \norm f_p +
\norm g_p$.\index{ongelijkheid van Minkowski}
\end{theorem}

\begin{proof}
$p = 1, \infty$: puntsgewijze/b.o.\ driehoeksongelijkheid. Voor
$1 < p < \infty$, neem $\norm{f+g}_p < \infty$ aan (anders
gebruik $\abs{f+g}^p \leq 2^{p-1}(\abs f^p + \abs g^p)$, uit
convexiteit van $t^p$, om te zien dat de linkerkant eindig is
wanneer de rechter dat is). Dan
\[
\norm{f{+}g}_p^p \leq \int\abs f\,\abs{f{+}g}^{p-1} +
\int\abs g\,\abs{f{+}g}^{p-1}
\leq \bigl(\norm f_p + \norm g_p\bigr)\,
\bigl\|\abs{f{+}g}^{p-1}\bigr\|_q
\]
door HÃ¶lder, en $\norm{\abs{f+g}^{p-1}}_q =
\norm{f+g}_p^{p/q}$ want $(p-1)q = p$; deel door
$\norm{f+g}_p^{p/q}$ (als nonzero; anders triviaal) en gebruik
$p - \frac pq = 1$.
\end{proof}

\section{Volledigheid en haar metgezellen}

\begin{theorem}[Riesz--Fischer]\label{thm:b3:lp:complete}
Voor $1 \leq p \leq \infty$ is $L^p(\mu)$ een Banachruimte.
Bovendien heeft elke rij die in $L^p$ convergeert een deelrij
die \emph{\omterm{def:b3:lebesgue:l1}{bijna overal}} convergeert (met een $L^p$-dominant in
het geval $p < \infty$).\index{stelling van Riesz--Fischer}
\end{theorem}

\begin{proof}
$p = \infty$: een $\norm\cdot_\infty$-Cauchyrij is, buiten een
enkele nulverzameling (unie van aftelbaar veel), uniform
Cauchy: zij convergeert uniform daarbuiten; klaar. Zij $p <
\infty$. Door \cref{exo:b3:complete:1}(b) volstaat het
absoluut convergente reeksen te sommeren: zij
$\sum\norm{f_k}_p = M < \infty$. Stel $G_n = \sum_{k \leq
n}\abs{f_k}$ en $G = \sum_k\abs{f_k}$ (puntsgewijs in
$[0,\infty]$): door Minkowski $\norm{G_n}_p \leq M$, en MCT
($G_n^p \nearrow G^p$) geeft $\int G^p \leq M^p$: $G < \infty$
b.o., dus de reeks $\sum f_k(x)$ convergeert absoluut voor
b.o.\ $x$; noem de som $S(x)$ (willekeurige waarde op de
nulverzameling). Dan $\abs{S - \sum_{k\leq n}f_k}^p \leq
(2G)^p \in L^1$, en DCT geeft $\norm{S - \sum_{k\leq
n}f_k}_p \to 0$: de reeks convergeert in $L^p$.

De deelrij-uitspraak: als $f_n \to f$ in $L^p$, kies $n_k$
met $\norm{f_{n_{k+1}} - f_{n_k}}_p \leq 2^{-k}$; de reeks
$\sum(f_{n_{k+1}} - f_{n_k})$ valt onder het vorige argument:
zij convergeert absoluut b.o., gedomineerd door een $G \in
L^p$, dus $f_{n_k} \to f_{n_1} + \sum(\cdots)$ b.o., en deze
b.o.-limiet moet (een representant van) $f$ zijn (beide zijn
$L^p$-limieten). De dominant: $\abs{f_{n_k}} \leq
\abs{f_{n_1}} + G$.
\end{proof}

\begin{remark}\label{rem:b3:lp:modes}
$L^p$-convergentie impliceert geen b.o.-convergentie (de
\emph{typemachine}-rij, \cref{exo:b3:lp:3}), noch omgekeerd
(ontsnappende bulten): de twee modi zijn alleen via deelrijen
en dominatie verbonden. De tegenvoorbeelden van
\cref{exo:b3:lp:3} in gedachten houden is het beste vaccin.
\end{remark}

\section{Dichtheidsstellingen}

\begin{theorem}\label{thm:b3:lp:density}
Zij $1 \leq p < \infty$.
\begin{enumerate}
  \item \omterm{def:b3:lebesgue:simple}{Eenvoudige functies} (met steunen van eindige \omterm{def:b3:measure:measure}{maat}) zijn
        \omterm{def:b3:topology:interior}{dicht} in $L^p(\mu)$.
  \item In $L^p(\R^d)$ zijn de continue compact ondersteunde
        functies $\mathcal C_c(\R^d)$ \omterm{def:b3:topology:interior}{dicht}.
  \item Translatie is continu op $L^p(\R^d)$: schrijvend
        $\tau_hf = f(\cdot - h)$, $\norm{\tau_hf - f}_p \to 0$
        als $h \to 0$.
\end{enumerate}
Geen van de drie geldt voor $p = \infty$.
\end{theorem}

\begin{proof}
(1) Voor $f \geq 0$: de dyadische $s_n \nearrow f$ van
\cref{thm:b3:lebesgue:approximation} voldoen $\abs{f -
s_n}^p \leq f^p \in L^1$: DCT. (Elke $s_n \leq f$ ligt in
$L^p$, en haar niveaus hebben eindige \omterm{def:b3:measure:measure}{maat} waar de waarde
positief is: $\mu(s_n \geq c) \leq c^{-p}\int f^p$.) Split
algemene $f$ in vier niet-negatieve delen.

(2) Door (1) volstaat het $\mathbf 1_A$ te benaderen, $A$ Borel
met $\lambda_d(A) < \infty$. Regulariteit (bewijs als in
\cref{thm:b3:measure:regularity}) geeft compact $K \subseteq A
\subseteq U$ open met $\lambda_d(U\setminus K) <
\varepsilon$; Urysohns functie
\[
\varphi(x) = \frac{d(x, \R^d\setminus U)}{d(x, \R^d\setminus U)
+ d(x, K)}
\]
is continu, $1$ op $K$, $0$ buiten $U$, en kan compact
ondersteund genomen worden (krimp $U$ eerst tot een begrensde
open). Dan $\norm{\mathbf 1_A - \varphi}_p^p \leq
\lambda_d(U\setminus K) < \varepsilon$.

(3) Voor $g \in \mathcal C_c$: uniforme continuÃ¯teit geeft
$\norm{\tau_hg - g}_\infty \to 0$, met steunen in een vast
compactum voor $\abs h \leq 1$: $\norm{\tau_hg - g}_p \to 0$.
Voor algemene $f$: kies $g \in \mathcal C_c$ met $\norm{f -
g}_p < \varepsilon$; dan $\norm{\tau_hf - f}_p \leq 2\norm{f -
g}_p + \norm{\tau_hg - g}_p$ (translatie-invariantie van de
norm).

Voor $p = \infty$: uniforme benadering van $\mathbf
1_{\intoo0\infty}$ door continue functies is onmogelijk
(sprong), en $\norm{\tau_h\mathbf 1_{\intoo0\infty} - \mathbf
1_{\intoo0\infty}}_\infty = 1$ voor $h \neq 0$.
\end{proof}

\section{Convolutie en regularisatie}

\begin{theorem}[Ongelijkheid van Young]\label{thm:b3:lp:young}
Zij $1 \leq p \leq \infty$, $f \in L^1(\R^d)$, $g \in
L^p(\R^d)$. Dan is $f * g$ b.o.\ gedefinieerd, behoort tot
$L^p$, en\index{ongelijkheid van Young}
\[
\norm{f * g}_p \leq \norm f_1\,\norm g_p .
\]
\end{theorem}

\begin{proof}
$p = \infty$: directe grens. $p = 1$:
\cref{thm:b3:product:convolution}. Zij $1 < p < \infty$, $q$
geconjugeerd. Split $\abs{f(y)} = \abs{f(y)}^{1/q}\cdot
\abs{f(y)}^{1/p}$ en pas HÃ¶lder toe:
\[
\int\abs{f(y)}\,\abs{g(x{-}y)}\,\dd y
\leq \Bigl(\int\abs f\Bigr)^{1/q}
\Bigl(\int\abs{f(y)}\,\abs{g(x - y)}^p\,\dd y\Bigr)^{1/p} .
\]
Verhef tot de $p$-de macht en integreer in $x$; Tonelli op de
tweede factor geeft $\norm f_1^{p/q}\cdot\norm f_1\norm g_p^p$,
d.w.z.\ $\norm{f*g}_p^p \leq \norm f_1^{1 + p/q}\norm g_p^p =
(\norm f_1\norm g_p)^p$ --- en de eindigheid van de
Tonelli-integraal rechtvaardigt b.o.\ absolute convergentie als
in \cref{thm:b3:product:convolution}.
\end{proof}

\begin{definition}[Mollifiers]\label{def:b3:lp:mollifier}
De functie\index{mollifier}\index{bumpfunctie}
\[
\rho(x) = \begin{cases}
c\,\exp\Bigl(-\dfrac{1}{1 - \norm x^2}\Bigr) & \norm x < 1,\\
0 & \norm x \geq 1,
\end{cases}
\]
met $c$ normaliserend $\int\rho = 1$, is $\mathcal C^\infty$ op
$\R^d$: het punt is dat $t \mapsto \eu^{-1/t}\mathbf
1_{t>0}$ $\mathcal C^\infty$ is op $\R$, al haar afgeleiden in
$0^+$ zijnde $0$ (elke afgeleide is $P(1/t)\eu^{-1/t}$ voor een
polynoom $P$, die naar $0$ streeft; inductie). Voor
$\varepsilon > 0$ stel $\rho_\varepsilon(x) =
\varepsilon^{-d}\rho(x/\varepsilon)$: ondersteund in $\bar
B(0, \varepsilon)$, nog steeds van integraal $1$.
\end{definition}

\begin{theorem}[Regularisatie]\label{thm:b3:lp:regularization}
Zij $1 \leq p < \infty$ en $f \in L^p(\R^d)$. Dan:
\begin{enumerate}
  \item $f * \rho_\varepsilon \in \mathcal C^\infty(\R^d)$,
        met $\partial^\alpha(f * \rho_\varepsilon) = f *
        \partial^\alpha\rho_\varepsilon$;
  \item $\norm{f * \rho_\varepsilon - f}_p \to 0$ als
        $\varepsilon \to 0$;
  \item bijgevolg is $\mathcal C^\infty_c(\R^d)$ \omterm{def:b3:topology:interior}{dicht} in
        $L^p(\R^d)$.
\end{enumerate}
\end{theorem}

\begin{proof}
(1) Differentiatie onder de integraal
(\cref{thm:b3:lebesgue:paramdiff}) in $x$: voor $x$ in een bal
$B$, $\abs{\partial_{x_i}\rho_\varepsilon(x - y)} \leq
C_\varepsilon\,\mathbf 1_{K}(y)$ met $K$ compact ($y$ binnen
$\varepsilon$ van $B$), en $\abs f\,\mathbf 1_K \in L^1$
(HÃ¶lder tegen $\mathbf 1_K$): de stelling is van toepassing;
itereer voor hogere afgeleiden.

(2) Omdat $\int\rho_\varepsilon = 1$:
\[
(f * \rho_\varepsilon)(x) - f(x)
= \int \bigl(f(x - y) - f(x)\bigr)\rho_\varepsilon(y)\,\dd y ,
\]
en Minkowski's \emph{integraal}ongelijkheid --- of direct:
HÃ¶lder/Jensen met de kansmaat $\rho_\varepsilon\dd y$ en
Tonelli ---
\[
\norm{f*\rho_\varepsilon - f}_p^p
\leq \int\Bigl(\int\abs{f(x-y) -
f(x)}^p\dd x\Bigr)\rho_\varepsilon(y)\,\dd y
= \int \norm{\tau_yf - f}_p^p\;\rho_\varepsilon(y)\,\dd y
\]
(de middelste stap: pas Jensens ongelijkheid toe,
\cref{exo:b3:lp:10}, op de inwendige $y$-integraal, dan
Tonelli). De integrand is ondersteund in $\norm y \leq
\varepsilon$ en streeft daar uniform naar $0$ als
$\varepsilon \to 0$ (\cref{thm:b3:lp:density}(3)): de hele
uitdrukking streeft naar $0$.

(3) Benader $f$ door $g \in \mathcal C_c$
(\cref{thm:b3:lp:density}(2)), dan $g$ door $g *
\rho_\varepsilon \in \mathcal C_c^\infty$ (compacte steun: som
van steunen).
\end{proof}

\begin{example}[Mollificeren van $\abs x$, met snelheden]
\label{ex:b3:lp:mollifyexample}
Neem $f(x) = \abs x$ op $\R$ (lokaal $L^1$; de stelling is van
toepassing op elk begrensd venster) en een symmetrische
\omterm{def:b3:lp:mollifier}{mollifier} $\rho_\varepsilon$. Dan
\[
f_\varepsilon(x) = (f * \rho_\varepsilon)(x)
= \int\abs{x - y}\,\rho_\varepsilon(y)\,\dd y
\]
is $\mathcal C^\infty$; weg van de knik gebeurt niets: voor
$\abs x \geq \varepsilon$ is $\abs{x - y}$ lineair in $x$ op
de steun van $\rho_\varepsilon$, dus $f_\varepsilon(x) =
\abs x$ \emph{exact} (symmetrie doodt de correctie).
Nabij $0$ kost gladmaking precies
\[
0 \leq f_\varepsilon(0) = \int\abs
y\,\rho_\varepsilon(y)\,\dd y \leq \varepsilon,
\qquad
\norm{f_\varepsilon - f}_\infty \leq \varepsilon :
\]
de benaderingsfout is beperkt tot de
$\varepsilon$-buurt van de singulariteit en is van haar
grootte. Ondertussen $f_\varepsilon'' \geq 0$ overal ($f$ is
convex, en convolutie tegen $\rho_\varepsilon \geq 0$ bewaart
convexiteit), met $\int f_\varepsilon'' =
f_\varepsilon'(\infty) - f_\varepsilon'(-\infty) = 2$: de
tweede afgeleide is een bult van massa $2$ geperst in breedte
$O(\varepsilon)$, dus $\norm{f_\varepsilon''}_\infty \gtrsim
\varepsilon^{-1}$. Gladmaking is een ruil: uniforme fout
$O(\varepsilon)$ tegen afgeleide-ontploffing
$O(\varepsilon^{-1})$ --- de exacte wisselkoers die
kwantitatieve analyse (interpolatieongelijkheden, de
ideeÃ«ncirkel van \cref{pb:b3:lp:1}) formaliseert.
\end{example}

\begin{corollary}[Fundamenteel lemma van de variatierekening]
\label{cor:b3:lp:fundlemma}
Zij $f \in L^1_{\mathrm{loc}}(\R^d)$ (\omterm{def:b3:lebesgue:l1}{integreerbaar} op
compacta) met $\int f\varphi = 0$ voor elke $\varphi \in
\mathcal C^\infty_c(\R^d)$. Dan $f = 0$ b.o.\index{fundamenteel
lemma van de variatierekening}
\end{corollary}

\begin{proof}
Fixeer een bal $B = B(0, R)$ en zij $g = f\mathbf 1_{B(0,
R+1)} \in L^1$. Voor $x \in B$ en $\varepsilon < 1$: $(g *
\rho_\varepsilon)(x) = \int f(y)\rho_\varepsilon(x - y)\dd y =
0$, de testfunctie zijnde $y \mapsto \rho_\varepsilon(x-y)
\in \mathcal C_c^\infty$. Maar $g * \rho_\varepsilon \to g$ in
$L^1$ (\cref{thm:b3:lp:regularization}): $g = 0$ b.o.\ op $B$;
put $\R^d$ uit.
\end{proof}

\section{De \texorpdfstring{$L^p$}{Lp}-geografie}

\begin{proposition}\label{prop:b3:lp:inclusions}
(a) Als $\mu(X) < \infty$ en $1 \leq p \leq q \leq \infty$,
dan $L^q \subseteq L^p$ met $\norm f_p \leq
\mu(X)^{\frac1p - \frac1q}\,\norm f_q$.
(b) Op $\R^d$ (oneindige \omterm{def:b3:measure:measure}{maat}) zijn er geen inclusies: voor
$p \neq q$ bestaan functies in $L^p\setminus L^q$.
(c) (Interpolatie) Als $p < r < q$ en $\alpha \in \intoo01$
gedefinieerd is door $\frac1r = \frac\alpha p + \frac{1 -
\alpha}q$, dan
\[
\norm f_r \leq \norm f_p^{\alpha}\,\norm f_q^{1 - \alpha} ;
\]
in het bijzonder $L^p \cap L^q \subseteq L^r$.
\end{proposition}

\begin{proof}
(a) HÃ¶lder met exponenten $\frac qp$ en haar geconjugeerde:
$\int\abs f^p\cdot 1 \leq \norm{\abs f^p}_{q/p}\,\norm
1_{(q/p)'} = \norm f_q^p\,\mu(X)^{1 - p/q}$ ($q = \infty$
direct). (b) Nabij $0$ en nabij $\infty$ kalibreren machten
$x^{-\alpha}$: \cref{exo:b3:lp:2}. (c) Schrijf $\abs f^r =
\abs f^{r\alpha}\abs f^{r(1-\alpha)}$ en pas HÃ¶lder toe met
het geconjugeerde paar $\frac p{r\alpha}$, $\frac
q{r(1-\alpha)}$ (geconjugeerd precies door de definitie van
$\alpha$): $\int\abs f^r \leq \norm f_p^{r\alpha}\norm
f_q^{r(1 - \alpha)}$.
\end{proof}

\begin{method}\label{met:b3:lp:toolkit}
De $L^p$-gereedschapskist, zoals overal hieronder gebruikt:
om een identiteit of ongelijkheid voor alle $f \in L^p$ te
bewijzen --- bewijs haar op een dichte klasse ($\mathcal
C_c^\infty$ via \cref{thm:b3:lp:regularization}) en breid uit
door continuÃ¯teit (\cref{thm:b3:complete:extension}, beide
kanten $L^p$-continu); om $f = 0$ te bewijzen, test tegen
$\mathcal C_c^\infty$ (\cref{cor:b3:lp:fundlemma}); om
gladheid te winnen, convolueer; om exponenten te ruilen,
HÃ¶lder en interpolatie. De Fouriertheorie van
\cref{ch:b3:fouriertransform} is Ã©Ã©n lange toepassing van
deze methode.
\end{method}

\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b3:lp:8}

\begin{exercise}[$\star\star$]\label{exo:b3:lp:9}

\begin{exercise}[$\star\star\star$]\label{exo:b3:lp:11}
(Youngs convolutie-ongelijkheid) Zijn $1 \leq p, q, r \leq

\emph{(Schrijf, voor geconjugeerde exponenten uitgewerkt uit

\section{Probleem: Hardys ongelijkheid}

\begin{problem}[{Weekendprobleem --- Hardys ongelijkheid en
haar scherpe constante}]\label{pb:b3:lp:1}
Voor $f \in L^p(\intoo0{+\infty})$, $1 < p < \infty$,
definieer de \emph{Hardy-operator}
\[
(Hf)(x) = \frac1x\int_0^x f(t)\,\dd t .
\]
Hardys ongelijkheid (1920) beweert\index{ongelijkheid van Hardy}
\[
\norm{Hf}_p \;\leq\; \frac{p}{p-1}\,\norm f_p ,
\]
en de constante $\frac p{p-1}$ is optimaal en wordt niet
bereikt. Dit probleem bewijst alles, en breidt dan uit tot
reeksen.

\medskip
\noindent\textbf{Deel I --- De ongelijkheid.} Neem eerst $f
\geq 0$ continu met compacte steun in $\intoo0{+\infty}$ aan,
en zij $F(x) = \int_0^xf$.
\begin{enumerate}
  \item Toon dat $Hf \in L^p$: nabij $0$ verdwijnt $F$ op een
        buurt van $0$; nabij $\infty$ is $F$ begrensd, dus
        $(Hf)(x) = O(1/x)$, en $x \mapsto \frac1x$ behoort tot
        $L^p(\intoo1{+\infty})$ voor $p > 1$.
  \item Integreer partieel om te tonen
        \[
        \int_0^\infty \Bigl(\frac Fx\Bigr)^p\dd x
        = \frac{p}{p-1}\int_0^\infty\Bigl(\frac
        Fx\Bigr)^{p-1}f(x)\,\dd x .
        \]
        \emph{(DifferentiÃ«er $x^{1-p}F^p$; randtermen verdwijnen
        --- rechtvaardig beide einden.)}
  \item Pas HÃ¶lder toe op de rechterkant en leid af
        $\norm{Hf}_p \leq \frac p{p-1}\norm f_p$ voor zulke $f$.
  \item Breid uit tot heel $L^p$: voor $f \geq 0$, construeer
        $f_n$ continu met compacte steun in
        $\intoo0{+\infty}$, $0 \leq f_n \nearrow f$ b.o.\
        \emph{(knoop af, benader dan monotoon ---
        rechtvaardig de constructie)}; dan $Hf_n \nearrow Hf$
        puntsgewijs (MCT binnen het gemiddelde) en MCT
        transporteert de ongelijkheid naar de limiet. Voor
        getekende of complexe $f$, besluit met $\abs{Hf} \leq
        H\abs f$.
\end{enumerate}

\medskip
\noindent\textbf{Deel II --- Optimaliteit.}
\begin{enumerate}[resume]
  \item Voor $A > 1$ zij $f_A(t) = t^{-1/p}\,\mathbf
        1_{\intcc1A}(t)$. Bereken $\norm{f_A}_p^p = \ln A$
        en, voor $1 \leq x \leq A$,
        \[
        (Hf_A)(x) = \frac p{p-1}\;x^{-1/p}\,
        \bigl(1 - x^{-(1 - 1/p)}\bigr).
        \]
  \item Leid af $\liminf_{A\to\infty} \norm{Hf_A}_p/
        \norm{f_A}_p \geq \frac{p}{p-1}$, en besluit dat de
        constante optimaal is.
  \item Toon dat gelijkheid $\norm{Hf}_p = \frac
        p{p-1}\norm f_p$ met $f \neq 0$ onmogelijk is.
        \emph{(Volg het gelijkheidsgeval van HÃ¶lder in vraag
        3: het zou $f = cx^{-1/p}$-type gedrag dwingen, dat
        niet in $L^p$ zit.)}
\end{enumerate}

\medskip
\noindent\textbf{Deel III --- De discrete ongelijkheid.}
\begin{enumerate}[resume]
  \item Voor een niet-stijgende $g \geq 0$ op $(0,\infty)$ en
        $a_n = g(n)$, vergelijk $\sum a_n^p$ en $\int g^p$,
        en $H$-gemiddelden dienovereenkomstig, om uit Deel I
        \emph{Hardys discrete ongelijkheid} af te leiden:
        voor $a_n \geq 0$,
        \[
        \sum_{n\geq1}\Bigl(\frac{a_1 + \dots +
        a_n}{n}\Bigr)^{p}
        \;\leq\;
        \Bigl(\frac{p}{p-1}\Bigr)^{p}\,\sum_{n\geq1}a_n^p
        \]
        --- bewijs haar eerst voor niet-stijgende $(a_n)$ via
        de vergelijking hierboven, reduceer dan het algemene
        geval tot het niet-stijgende door herrangschikking
        \emph{(geef toe, met een Ã©Ã©n-regel-rechtvaardiging,
        dat sorteren van $(a_n)$ in dalende volgorde de
        linkerkant alleen kan verhogen terwijl de rechter vast
        blijft)}.
  \item Leid af: als $\sum a_n^p < \infty$ dan zijn de
        CesÃ ro-gemiddelden van $(a_n)$ opnieuw $\ell^p$ --- en
        geef een voorbeeld ($p = 2$) waar $(a_n) \in \ell^2$
        maar $a_n$ niet sommeerbaar is, terwijl Hardy de
        gemiddelden nog controleert.
\end{enumerate}

\medskip
\noindent\textbf{Deel IV --- Epiloog.}
\begin{enumerate}[resume]
  \item Toon dat Hardys ongelijkheid faalt voor $p = 1$: met
        $f = \mathbf 1_{\intcc01}$, bereken $Hf$ en observeer
        $Hf \notin L^1$. Waar breekt het bewijs?
\end{enumerate}

\medskip
\noindent\textbf{Deel V --- De maximale functie, en Lebesgues
differentiatiestelling.} Hardy middelen vanaf de oorsprong;
Hardy--Littlewood middelen \emph{rond elk punt}. Voor $f \in
L^1(\R)$ definieer
\[
Mf(x) = \sup_{r>0}\ \frac1{2r}\int_{x-r}^{x+r}\abs
f\,\dd\lambda .
\]
\begin{enumerate}[resume]
  \item (Vitali, eindige versie) Zijn $B_1, \dots, B_N$ open
        intervallen. Toon dat er een disjuncte deelfamilie
        $B_{i_1}, \dots, B_{i_k}$ bestaat met $\bigcup_jB_j
        \subseteq \bigcup_l3B_{i_l}$, waar $3B$ het interval
        met hetzelfde \omterm{ex:b3:groups:actions}{centrum} en drievoudige lengte aanduidt
        \emph{(gretig: kies herhaaldelijk het langste
        interval disjunct van die al gekozen)}.
  \item (Zwak type $(1,1)$) Toon dat voor elke $t > 0$,
        \[
        \lambda\bigl(\{Mf > t\}\bigr) \;\leq\;
        \frac3t\,\norm f_1 :
        \]
        elke $x$ met $Mf(x) > t$ bezit een gecentreerd
        interval $B_x$ met $\int_{B_x}\abs f > t\,\lambda(B_x)$;
        neem een compactum $K \subseteq \{Mf > t\}$
        (inwendige regulariteit), overdek het door eindig veel
        $B_x$, pas vraag 11 toe, en put uit.
  \item Bereken $M\mathbf 1_{\intcc01}(x)$ voor $x > 1$ en
        leid af dat $Mf \notin L^1$ voor elke $f \neq 0$
        ($Mf(x) \geq \frac c{\abs x}$ op oneindig): bij $p =
        1$ is de zwakke ongelijkheid van vraag 12 de best
        mogelijke uitspraak.
  \item (Sterk type voor $p > 1$) Voor $f \in L^p$: split $f =
        f\,\mathbf 1_{\abs f > t/2} + f\,\mathbf 1_{\abs f
        \leq t/2}$, observeer $Mf \leq M\bigl(f\mathbf
        1_{\abs f > t/2}\bigr) + \frac t2$, en combineer
        vraag 12 met de laagjescakeformule
        (\cref{prop:b3:product:layercake}) en Tonelli om te
        bewijzen
        \[
        \norm{Mf}_p^p \;\leq\;
        \frac{6p\,2^{p-1}}{p-1}\,\norm f_p^p .
        \]
        (De ontploffing als $p \downarrow 1$ is vraag 13's
        mislukking, gekwantificeerd.)
  \item (Differentiatiestelling van Lebesgue) Bewijs: voor $f
        \in L^1(\R)$,
        \[
        \frac1{2r}\int_{x-r}^{x+r}f\,\dd\lambda
        \;\longrightarrow\; f(x)
        \qquad (r \to 0)\quad\text{voor b.o.\ }x .
        \]
        \emph{(Duidelijk voor continue $f$. In het algemeen
        schrijf $f = g + h$, $g$ continu met compacte steun,
        $\norm h_1 < \varepsilon$
        (\cref{thm:b3:lp:density}); de verzameling waar
        $\limsup_{r\to0}$ van de gemiddelde oscillatie
        $\delta$ overschrijdt zit in $\{Mh > \delta/2\} \cup
        \{\abs h > \delta/2\}$, van \omterm{def:b3:measure:measure}{maat}
        $O(\varepsilon/\delta)$; laat $\varepsilon \to 0$,
        dan $\delta \to 0$ langs een rij.)}
  \item Leid af: (a) bijna elk punt is een \emph{Lebesgue-
        punt} van $f$; (b) voor $f \in L^1$ is de primitieve
        $F(x) = \int_0^xf$ differentieerbaar b.o.\ met $F' =
        f$ b.o.\ --- de integraalhelft van de fundamentele
        stelling van de calculus in de Lebesgue-wereld, die de
        cirkel sluit die de trap opende
        (\cref{pb:b3:measure:1}), die toonde dat de omgekeerde
        helft kan falen.
  \item (Dichtheidspunten) Voor meetbare $A \subseteq \R$,
        toon dat bijna elke $x \in A$ voldoet aan
        $\frac{\lambda(A\cap\intcc{x-r}{x+r})}{2r} \to 1$:
        meetbare verzamelingen zijn lokaal vol in bijna al
        hun punten. Schets, in twee regels, hoe dit nog een
        bewijs van Steinhaus' stelling oplevert
        (\cref{exo:b3:measure:8}).
\end{enumerate}

\medskip
\noindent\textbf{Deel VI --- Variaties op het thema van
middelen.}
\begin{enumerate}[resume]
  \item (Gewogen Hardy) Voor $\alpha < p - 1$ en $f \geq 0$,
        toon
        \[
        \int_0^\infty\Bigl(\frac{F(x)}x\Bigr)^{p}
        x^{\alpha}\,\dd x \;\leq\;
        \Bigl(\frac{p}{p - 1 - \alpha}\Bigr)^{p}
        \int_0^\infty f(x)^p\,x^{\alpha}\,\dd x
        \]
        door dezelfde partiÃ«le integratie, en controleer dat
        de grens $\alpha = p - 1$ werkelijk verboden is
        (pas vraag 10's tegenvoorbeeld aan).
  \item (De geadjungeerde) Zij $H^*f(x) =
        \int_x^{\infty}\frac{f(t)}t\,\dd t$. Toon $\langle
        Hf, g\rangle = \langle f, H^*g\rangle$ voor
        niet-negatieve $f, g$ (Tonelli), en bewijs
        $\norm{H^*f}_p \leq p\,\norm f_p$ \emph{(direct door
        partiÃ«le integratie, of uit Hardy op de geconjugeerde
        exponent door dualiteit --- let op welke exponent
        welke constante oppikt)}.
  \item (Een Hilbert-type ongelijkheid) Leid af dat voor
        niet-negatieve $f \in L^p$, $g \in L^q$:
        \[
        \int_0^\infty\!\!\int_0^\infty
        \frac{f(x)\,g(y)}{\max(x,y)}\,\dd x\,\dd y
        \;\leq\; (p + q)\,\norm f_p\,\norm g_q
        \]
        \emph{(splits langs $y < x$ / $y \geq x$: elke helft
        is een pairing van Ã©Ã©n functie tegen een
        Hardy-transformatie van de andere)}.
  \item (Optimaliteit, discreet) Toon dat de constante
        $\bigl(\frac p{p-1}\bigr)^p$ van vraag 8 ook optimaal
        is: test op $a_n = n^{-1/p}\,\mathbf 1_{n \leq N}$,
        vergelijk beide kanten met integralen, en laat $N
        \to \infty$ (de discrete spiegel van Deel II).
  \item (Synthese) Drie middelingoperatoren verschenen in dit
        probleem: Hardys $H$, het discrete CesÃ ro-gemiddelde,
        en de maximale operator $M$. Formuleer in Ã©Ã©n regel
        elk wat haar begrensdheid zegt, observeer dat alle
        drie precies bij $p = 1$ falen, en leg uit waarom het
        dezelfde mislukking driemaal is (de harmonische
        staart $\frac1x$).
\end{enumerate}

\medskip
\noindent\textbf{Deel VII --- Carlemans ongelijkheid, en hoe
scherp is scherp.}
\begin{enumerate}[resume]
  \item (Carlemans ongelijkheid) Zij $a_n \geq 0$ met $\sum
        a_n < \infty$. Pas de discrete Hardy-ongelijkheid van
        vraag 8 toe op $b_n = a_n^{1/p}$, gebruik de
        rekenkundig-meetkundige ongelijkheid, en laat $p \to
        \infty$ (toon dat $p \mapsto
        \bigl(\frac p{p-1}\bigr)^p$ daalt naar $\eu$) om te
        verkrijgen
        \[
        \sum_{n\geq1}\bigl(a_1a_2\cdots a_n\bigr)^{1/n}
        \;\leq\; \eu\,\sum_{n\geq1}a_n :
        \]
        de meetkundige gemiddelden van een sommeerbare rij
        zijn sommeerbaar, ten koste van ten hoogste $\eu$.
  \item (De constante $\eu$ is optimaal) Test $a_n =
        \frac1n\,\mathbf 1_{n\leq N}$: met de Stirling-
        inkadering van \cref{pb:b3:product:1}, toon
        $(n!)^{-1/n} = \frac\eu n\bigl(1 +
        O\bigl(\frac{\ln n}n\bigr)\bigr)$, leid af dat beide
        kanten van Carleman groeien als $\eu\ln N$, en besluit
        dat geen constante kleiner dan $\eu$ kan werken.
        (Observeer het patroon: de optimalisatoren van Hardy
        en van Carleman zijn beide de harmonische-type rijen
        die net falen in de ruimte te zitten.)
  \item (Hoe langzaam wordt ``scherp'' benaderd?) Neem $p =
        2$. Voor $f = \mathbf 1_{\intcc01}$, bereken
        $\norm{Hf}_2/\norm f_2 = \sqrt2$, tegen de grens $2$.
        Voor de bijna-optimalisatoren $f_A$ van vraag 5,
        bewijs de exacte identiteit
        \[
        \norm{Hf_A}_2^2 = 4\ln A - 8 + \frac{8}{\sqrt A},
        \qquad\text{dus}\qquad
        \frac{\norm{Hf_A}_2^2}{\norm{f_A}_2^2}
        = 4 - \frac{8 - 8A^{-1/2}}{\ln A} .
        \]
        Evalueer bij $A = \eu^{10}$ (ratio $\approx 1.790$)
        en commentarieer: het supremum $2$ wordt benaderd
        met snelheid $1/\ln A$ alleen --- een optimale
        constante kan numeriek vrijwel onzichtbaar zijn.
\end{enumerate}
