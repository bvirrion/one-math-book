\chapter{Holomorfe functies}\label{ch:b3:holomorphic}

Complexe differentieerbaarheid lijkt een kleine variatie op de
reële theorie --- één limiet, één quotiënt. Het is veeleer een
ander universum. Een functie die \emph{éénmaal} differentieerbaar
is op een open deelverzameling van $\C$ is automatisch oneindig
vaak differentieerbaar, analytisch, op een heel gebied bepaald
door haar waarden bij één enkel punt, en onderworpen aan starre
globale principes (Liouville, maximummodulus). Dit alles vloeit
voort uit één mirakel, de stelling van Cauchy: de integraal van
een \omterm{def:b3:holomorphic:holo}{holomorfe functie} langs een gesloten pad in een stervormig
gebied verdwijnt. Dit hoofdstuk bewijst het mirakel (het
argument van Goursat, zonder regulariteit voorbij
differentieerbaarheid), oogst de klassieke gevolgen, en regelt
een stelling die dit boek sinds \cref{ch:b3:galois} op krediet
gebruikt: elk niet-constant complex polynoom heeft een wortel.

Doorheen is $\Omega \subseteq \C$ open, en $D(a, r)$ duidt de
open schijf aan.

\section{Complexe differentieerbaarheid}

\begin{definition}\label{def:b3:holomorphic:holo}
$f \colon \Omega \to \C$ is \emph{holomorf}\index{holomorfe
functie} op $\Omega$ wanneer voor elke $z_0 \in \Omega$
\[
f'(z_0) = \lim_{h\to0}\frac{f(z_0 + h) - f(z_0)}{h}
\]
bestaat ($h \in \C^*$). Sommen, producten, quotiënten
(niet-verdwijnende noemers), samenstellingen van holomorfe
functies zijn holomorf, met de gebruikelijke formules (de
bewijzen van Jaar~1--2 gelden letterlijk: zij gebruiken slechts
lichaamsoperaties en limieten). $\mathcal H(\Omega)$ duidt de
verzameling holomorfe functies op $\Omega$ aan.
\end{definition}

\begin{proposition}[Cauchy--Riemann]
\label{prop:b3:holomorphic:cauchyriemann}
Schrijf $f(x + \iu y) = P(x,y) + \iu Q(x,y)$. Dan is $f$
\omterm{def:b3:holomorphic:holo}{holomorf} in $z_0$ desda $f$ $\R$-differentieerbaar is in $z_0$
(als afbeelding van twee reële variabelen) en\index{Cauchy--Riemannvergelijkingen}
\[
\frac{\partial P}{\partial x} = \frac{\partial Q}{\partial y},
\qquad
\frac{\partial P}{\partial y} = -\frac{\partial Q}{\partial x}
\qquad \text{in } z_0 ;
\]
equivalent: de reële differentiaal is vermenigvuldiging met
het complexe getal $f'(z_0)$.
\end{proposition}

\begin{proof}
$\C$-differentieerbaarheid zegt $f(z_0 + h) = f(z_0) + ch +
o(\abs h)$ met $c = f'(z_0)$: een $\R$-lineaire differentiaal
die vermenigvuldiging met $c = a + \iu b$ is, d.w.z.\ matrix
$\bigl(\begin{smallmatrix} a & -b\\ b & a
\end{smallmatrix}\bigr)$ heeft in de basis $(1, \iu)$ ---
precies de weergegeven relaties voor de partiële afgeleiden.
Omgekeerd is zo'n differentiaal $\C$-lineair, en de
$o(\abs h)$-definities komen overeen.
\end{proof}

\begin{example}\label{ex:b3:holomorphic:examples}
Polynomen in $z$, rationale functies buiten hun polen, en
--- door de term-voor-term-differentiatiestelling voor
machtreeksen uit Jaar~2, waarvan het bewijs identiek werkt
over $\C$ --- elke som van een machtreeks $\sum a_n(z - a)^n$
binnen haar convergentieschijf: \omterm{def:b3:holomorphic:holo}{holomorf}, met afgeleide $\sum
na_n(z - a)^{n-1}$ (zelfde straal). In het bijzonder is
$\exp z = \sum z^n/n!$ geheel (\omterm{def:b3:holomorphic:holo}{holomorf} op $\C$) met
$\exp' = \exp$. Anderzijds zijn $z \mapsto \bar z$, $\abs z$,
$\operatorname{Re}z$ nergens \omterm{def:b3:holomorphic:holo}{holomorf} (Cauchy--Riemann faalt
overal): holomorfie is starheid die oriëntatie en hoeken
bewaart, geen gladheid.
\end{example}

\section{Contourintegralen}

\begin{definition}\label{def:b3:holomorphic:contour}
Een \emph{pad} is een stuksgewijs $\mathcal C^1$-afbeelding
$\gamma \colon \intcc ab \to \C$; het is \emph{gesloten}
wanneer $\gamma(a) = \gamma(b)$. Voor continue $f$ op het
beeld van $\gamma$:\index{contourintegraal}
\[
\int_\gamma f(z)\,\dd z =
\int_a^b f(\gamma(t))\,\gamma'(t)\,\dd t,
\qquad
\Bigl|\int_\gamma f\,\dd z\Bigr| \leq
\sup_{\gamma}\abs f\cdot\operatorname{length}(\gamma)
\]
(de \emph{ML-ongelijkheid}; lengte $= \int_a^b\abs{\gamma'}$).
De integraal is invariant onder toenemende $\mathcal
C^1$-herparametrisering en wisselt van teken onder
oriëntatieomkering.
\end{definition}

\begin{proposition}[Primitieven]\label{prop:b3:holomorphic:primitive}
Voor continue $f$ op $\Omega$ zijn de volgende equivalent:
(i) $f$ heeft een \emph{primitieve} $F \in \mathcal H(\Omega)$
($F' = f$); (ii) $\int_\gamma f\,\dd z = 0$ voor elk gesloten
pad $\gamma$ in $\Omega$. In dat geval is $\int_\gamma
f\,\dd z = F(\gamma(b)) - F(\gamma(a))$ voor elk pad.
\end{proposition}

\begin{proof}
(i)$\Rightarrow$: $\frac{\dd}{\dd t}F(\gamma(t)) =
F'(\gamma(t))\gamma'(t)$ (kettingregel, stuksgewijs geldig),
dus de integraal telescopieert tot het eindpuntsverschil;
gesloten paden geven $0$. (ii)$\Rightarrow$(i): fixeer $z_*$
in een \omterm{def:b3:topology:components}{samenhangscomponent}, definieer $F(z) =
\int_{\gamma_z}f\,\dd z$ langs enig pad van $z_*$ naar $z$
(goed gedefinieerd: twee paden verschillen door een gesloten
pad); voor kleine $h$, neem het segment van $z$ naar $z + h$,
\[
\frac{F(z + h) - F(z)}{h} - f(z)
= \frac1h\int_{[z, z+h]}\bigl(f(w) - f(z)\bigr)\dd w
\longrightarrow 0
\]
door de ML-ongelijkheid en continuïteit van $f$ in $z$.
\end{proof}

\begin{definition}[Windinggetal]\label{def:b3:holomorphic:index}
Voor een gesloten pad $\gamma$ en $z \notin
\operatorname{im}\gamma$ is de \emph{index}\index{windinggetal}
\index{index van een pad}
\[
\operatorname{Ind}_\gamma(z) = \frac1{2\iu\pi}
\int_\gamma\frac{\dd w}{w - z} .
\]
Het is een \emph{geheel getal}: zet $\varphi(t) =
\int_a^t\frac{\gamma'(s)}{\gamma(s) - z}\dd s$, dan heeft
$(\gamma(t) - z)\eu^{-\varphi(t)}$ afgeleide nul
(stuksgewijs), is dus constant; in $t = b$ is
$\eu^{\varphi(b)} = \frac{\gamma(b) - z}{\gamma(a) - z} =
1$, dus $\varphi(b) \in 2\iu\pi\Z$. Als functie van $z$ is
de index continu op $\C\setminus\operatorname{im}\gamma$
(gedomineerde convergentie), dus constant op elke
\omterm{def:b3:topology:components}{samenhangscomponent}, en $0$ op de onbegrensde component
(ML: de integraal neigt naar $0$ als $z \to \infty$). Voor
de cirkel $\gamma(t) = a + r\eu^{\iu t}$, $t \in
\intcc0{2\pi}$: $\operatorname{Ind}_\gamma(z) = 1$ voor $z
\in D(a,r)$ (bereken in $z = a$: $\frac1{2\iu\pi}
\int_0^{2\pi}\frac{r\iu\eu^{\iu t}}{r\eu^{\iu t}}\dd t = 1$;
constantheid doet de rest).
\end{definition}

\section{De stelling van Cauchy}

\begin{theorem}[Goursat]\label{thm:b3:holomorphic:goursat}
Zij $f \in \mathcal H(\Omega)$ en $T \subseteq \Omega$ een
gesloten solide driehoek. Dan $\int_{\partial T}f\,\dd z = 0$
(rand eenmaal doorlopen, willekeurige oriëntatie).\index{stelling van Goursat}
\end{theorem}

\begin{proof}
Zet $I(T) = \int_{\partial T}f\,\dd z$. Het verbinden van de
middens van de zijden splitst $T$ in vier halfgrote
driehoeken $T^{(1)}, \dots, T^{(4)}$, en de binnenranden
heffen elkaar paarsgewijs op: $I(T) = \sum_iI(T^{(i)})$.
Kies $T_1$ daaronder met $\abs{I(T_1)} \geq
\frac14\abs{I(T)}$, en herhaal: een geneste rij $T \supseteq
T_1 \supseteq T_2 \supseteq\cdots$ met
\[
\abs{I(T_n)} \geq 4^{-n}\abs{I(T)},
\qquad
\operatorname{diam}T_n = 2^{-n}\operatorname{diam}T,
\quad
\operatorname{length}(\partial T_n) =
2^{-n}\operatorname{length}(\partial T).
\]
De doorsnede $\bigcap T_n$ is een enkel punt $z_0$ (geneste
compacta met verdwijnende diameters,
\cref{thm:b3:topology:compactprops}(3)). Differentieerbaarheid
in $z_0$: gegeven $\varepsilon$, voor grote $n$, op $T_n$,
\[
f(z) = f(z_0) + f'(z_0)(z - z_0) + R(z),
\qquad \abs{R(z)} \leq \varepsilon\abs{z - z_0} \leq
\varepsilon\operatorname{diam}T_n .
\]
Het affiene deel heeft een primitieve: zijn integraal over
de gesloten $\partial T_n$ verdwijnt
(\cref{prop:b3:holomorphic:primitive}), en er blijft
\[
\abs{I(T_n)} = \Bigl|\int_{\partial T_n}R\Bigr|
\leq \varepsilon\operatorname{diam}(T_n)\,
\operatorname{length}(\partial T_n)
= \varepsilon\,4^{-n}\operatorname{diam}(T)
\operatorname{length}(\partial T) .
\]
Vergelijken met $\abs{I(T_n)} \geq 4^{-n}\abs{I(T)}$:
$\abs{I(T)} \leq \varepsilon\cdot\text{const}$ voor elke
$\varepsilon$: $I(T) = 0$.
\end{proof}

\begin{theorem}[Stelling van Cauchy, stervormige versie]
\label{thm:b3:holomorphic:cauchy}
Zij $\Omega$ \emph{stervormig} om $c$ (elk segment $[c, z]$,
$z \in \Omega$, ligt in $\Omega$) --- b.v.\ convex. Elke $f
\in \mathcal H(\Omega)$ heeft een primitieve op $\Omega$;
bijgevolg $\int_\gamma f\,\dd z = 0$ voor \emph{elk} gesloten
pad $\gamma$ in $\Omega$.\index{stelling van Cauchy}
\end{theorem}

\begin{proof}
Definieer $F(z) = \int_{[c,z]}f\,\dd w$. Voor $z, z + h \in
\Omega$ met $[z, z+h] \subseteq \Omega$ (waar voor kleine
$h$), ligt de driehoek met hoekpunten $c, z, z+h$ in $\Omega$
(stervormigheid: elk van haar punten ligt op een segment
$[c, w]$ met $w \in [z, z+h] \subseteq \Omega$): Goursat
geeft
\[
F(z + h) - F(z) = \int_{[z, z+h]}f\,\dd w,
\]
en de differentiequotiëntberekening van
\cref{prop:b3:holomorphic:primitive} levert $F' = f$. Het
verdwijnen van integralen over gesloten paden volgt uit
dezelfde propositie.
\end{proof}

\begin{theorem}[Integrale formule van Cauchy]
\label{thm:b3:holomorphic:formula}
Zij $f \in \mathcal H(\Omega)$, $\bar D(a, r) \subseteq
\Omega$, en $C_r$ de cirkel $\partial D(a,r)$ eenmaal
tegen de klok in doorlopen. Dan voor elke $z \in D(a,
r)$:\index{integrale formule van Cauchy}
\[
f(z) = \frac1{2\iu\pi}\int_{C_r}\frac{f(w)}{w - z}\,\dd w .
\]
\end{theorem}

\begin{proof}
Fixeer $z$ en definieer op $\Omega$
\[
g(w) = \begin{cases}
\dfrac{f(w) - f(z)}{w - z} & w \neq z,\\[4pt]
f'(z) & w = z :
\end{cases}
\]
$g$ is continu op $\Omega$ en \omterm{def:b3:holomorphic:holo}{holomorf} buiten $z$. Goursat
geldt voor $g$ op elke driehoek $T \subseteq \Omega'$, waar
$\Omega'$ een schijf iets groter dan $\bar D(a,r)$ binnen
$\Omega$ is: als $z \notin T$, direct; als $z \in T$, splits
$T$ in kleine driehoeken met $z$ als hoekpunt plus
driehoeken die $z$ vermijden; op een driehoek met hoekpunt
$z$ geeft de ML-grens $\abs{\int_{\partial T'}g} \leq
\sup_{T'}\abs g\cdot\operatorname{length} \to 0$ als de
driehoek krimpt, en de overige stukken verdwijnen door
Goursat --- dus $\int_{\partial T}g = 0$ in alle gevallen.
Het bewijs van \cref{thm:b3:holomorphic:cauchy} gebruikte
slechts deze driehoekseigenschap: $g$ heeft een primitieve
op het convexe $\Omega'$, dus $\int_{C_r}g = 0$, d.w.z.
\[
\frac1{2\iu\pi}\int_{C_r}\frac{f(w)}{w - z}\dd w
= f(z)\,\frac{1}{2\iu\pi}\int_{C_r}\frac{\dd w}{w - z}
= f(z)\operatorname{Ind}_{C_r}(z) = f(z) .
\]
\end{proof}

\section{Analyticiteit en haar cascade}

\begin{theorem}[Holomorf $=$ analytisch]
\label{thm:b3:holomorphic:analytic}
Zij $f \in \mathcal H(\Omega)$ en $D(a, R) \subseteq \Omega$.
Dan\index{analyticiteit van holomorfe functies}
\[
f(z) = \sum_{n\geq0}c_n\,(z - a)^n \quad \text{op } D(a, R),
\qquad
c_n = \frac1{2\iu\pi}\int_{C_r}\frac{f(w)}{(w -
a)^{n+1}}\,\dd w \ \ (0 < r < R),
\]
waarbij de coëfficiënten onafhankelijk van $r$ zijn.
Bijgevolg is $f$ oneindig $\C$-differentieerbaar, $c_n =
f^{(n)}(a)/n!$, en gelden de \emph{\omterm{thm:b3:holomorphic:analytic}{Cauchy-schattingen}}%
\index{Cauchy-schattingen}:
\[
\abs{c_n} \;\leq\; \frac{\sup_{\abs{w - a} =
r}\abs{f(w)}}{r^{n}} .
\]
\end{theorem}

\begin{proof}
Voor $\abs{z - a} < r$: ontwikkel de Cauchy-kern in de
meetkundige reeks
\[
\frac1{w - z} = \frac1{(w - a)\bigl(1 - \frac{z - a}{w -
a}\bigr)} = \sum_{n\geq0}\frac{(z - a)^n}{(w - a)^{n+1}},
\]
normaal convergent in $w$ op $C_r$ ($\abs{\frac{z-a}{w-a}} =
\frac{\abs{z-a}}r < 1$): integreer term voor term tegen
$\frac{f(w)}{2\iu\pi}$ (uniforme convergentie rechtvaardigt
de verwisseling) en pas \cref{thm:b3:holomorphic:formula} toe.
Een machtreeks is oneindig differentieerbaar met $c_n =
f^{(n)}(a)/n!$ (Jaar~2), wat ook toont dat de $c_n$ niet van
$r$ afhangen. De schattingen: begrens de
coëfficiëntintegraal met ML.
\end{proof}

\begin{example}[Singulariteiten dicteren stralen]
\label{ex:b3:holomorphic:radius}
Waarom heeft de onschuldige reële functie $\frac1{1 + x^2}$
een Taylorreeks in $x = 3$ die slechts convergeert voor
$\abs{x - 3} < \sqrt{10}$, terwijl op de reële lijn niets
misgaat? Omdat de stelling hierboven de convergentiestraal in
$a$ gelijk maakt aan de afstand van $a$ tot het dichtstbijzijnde
punt waar holomorfie faalt. Hier is $f(z) = \frac1{1 + z^2}$
precies \omterm{def:b3:holomorphic:holo}{holomorf} op $\C\setminus\{\pm\iu\}$, dus de
ontwikkeling in $a = 3$ convergeert op de grootste schijf die
$\pm\iu$ vermijdt, met straal $\abs{3 - \iu} = \sqrt{10}$ ---
en kan op geen grotere convergeren, want de som zou $f$
\omterm{def:b3:holomorphic:holo}{holomorf} voortzetten naar een \omterm{def:b3:topology:topology}{omgeving} van $\pm\iu$, waar
$\abs f \to \infty$. De reële theorie ziet de mysterieuze
straal $\sqrt{10}$; het complexe vlak ziet twee polen.
Dit is de praktische regel: \emph{om een convergentiestraal te
vinden, lokaliseer de singulariteiten} --- b.v.\ de
Taylorreeks van $\tan$ in $0$ heeft straal $\frac\pi2$
(dichtstbijzijnde nullen van $\cos$), en de
Bernoulli-genererende functie $\frac z{\eu^z - 1}$
(\cref{pb:b3:holomorphic:1}, Deel~VI) heeft straal $2\pi$
(dichtstbijzijnde niet-nulle nullen van $\eu^z - 1$:
$\pm2\iu\pi$).
\end{example}

\begin{corollary}[Liouville; d'Alembert--Gauss]
\label{cor:b3:holomorphic:liouville}
Een begrensde gehele functie is constant. Bijgevolg heeft elk
niet-constant polynoom over $\C$ een wortel: \emph{$\C$ is
\omterm{def:b3:galois:closure}{algebraïsch gesloten}}.\index{stelling van Liouville}\index{hoofdstelling van de algebra}
\end{corollary}

\begin{proof}
Als $\abs f \leq M$ op $\C$: voor elke $a$ en $r$,
$\abs{c_1(a)} = \abs{f'(a)} \leq M/r \to 0$: $f' \equiv 0$,
en $f$ is constant (op het samenhangende $\C$: nulafgelide
impliceert lokaal constant --- integreer langs segmenten).
Als $P$ geen wortel had, zou $1/P$ geheel en begrensd zijn
($\abs{P(z)} \to \infty$ als $\abs z \to \infty$: de
hoofdterm domineert, dus $\abs{1/P}$ is klein buiten een
grote schijf en continu op de compacte schijf): constant ---
absurd voor niet-constant $P$. (Het weekendprobleem geeft een
tweede, elementair bewijs en de algebraïsche gevolgen.)
\end{proof}

\begin{theorem}[Nullen zijn geïsoleerd; identiteitsstelling]
\label{thm:b3:holomorphic:identity}
Zij $\Omega$ \emph{\omterm{def:b3:topology:connected}{samenhangend}} en $f \in \mathcal
H(\Omega)$, $f \not\equiv 0$. Dan heeft elke nul $a$ van $f$
eindige \emph{orde}: $f(z) = (z - a)^m\,g(z)$ met $g \in
\mathcal H(\Omega)$, $g(a) \neq 0$, en de nullen van $f$
hebben geen ophopingspunt in $\Omega$. Bijgevolg, als twee
\omterm{def:b3:holomorphic:holo}{holomorfe functies} op $\Omega$ overeenstemmen op een
verzameling met een ophopingspunt in $\Omega$, stemmen zij
overal overeen.\index{identiteitsstelling}\index{geïsoleerde nullen}
\end{theorem}

\begin{proof}
Zij $Z$ de verzameling punten waar alle afgeleiden van $f$
verdwijnen. $Z$ is gesloten (doorsnede van gesloten
verzamelingen) en open: als alle $c_n = 0$ in $a$, maakt de
machtreeksontwikkeling $f \equiv 0$ op een schijf om $a$.
Samenhang: $Z = \varnothing$ of $Z = \Omega$; het laatste is
uitgesloten door $f \not\equiv 0$. Dus in een nul $a$ is
enige coëfficiënt niet-nul: laat $m$ minimaal met $c_m \neq
0$; dan $f(z) = (z - a)^m\sum_{n\geq m}c_n(z-a)^{n-m}$ op een
schijf, en de som definieert $g$ \omterm{def:b3:holomorphic:holo}{holomorf} bij $a$ met $g(a) =
c_m \neq 0$; zet $g = f/(z-a)^m$ voort buiten $a$ (daar
\omterm{def:b3:holomorphic:holo}{holomorf}). Omdat $g(a) \neq 0$ en $g$ continu is, heeft $f$
geen andere nul in een \omterm{def:b3:topology:topology}{omgeving} van $a$: nullen zijn
geïsoleerd, en een verzameling geïsoleerde punten heeft geen
ophopingspunt in $\Omega$ (een ophopingspunt van nullen is
een nul --- continuïteit --- en zou niet geïsoleerd zijn).
Identiteit: pas toe op het verschil, waarvan de nulverzameling
een ophopingspunt heeft, wat haar in de tak $Z = \Omega$
dwingt.
\end{proof}

\begin{theorem}[Middelwaarde en maximummodulus]
\label{thm:b3:holomorphic:maximum}
Zij $f \in \mathcal H(\Omega)$.
\begin{enumerate}
  \item (Middelwaarde) Voor $\bar D(a, r) \subseteq \Omega$:
        $f(a) = \frac1{2\pi}\int_0^{2\pi}f(a +
        r\eu^{\iu t})\,\dd t$.
  \item (Maximumprincipe) Als $\Omega$ \omterm{def:b3:topology:connected}{samenhangend} is en
        $\abs f$ een lokaal maximum bereikt in enig punt van
        $\Omega$, dan is $f$ constant. Bijgevolg, voor
        begrensd $\Omega$ en $f$ continu op $\bar\Omega$:
        $\sup_{\bar\Omega}\abs f =
        \sup_{\partial\Omega}\abs f$.\index{maximumprincipe}
\end{enumerate}
\end{theorem}

\begin{proof}
(1) is de formule van Cauchy in het \omterm{ex:b3:groups:actions}{centrum}: parametriseer
$C_r$.
(2) Zeg $\abs f \leq \abs{f(a)}$ op $\bar D(a, \rho)
\subseteq \Omega$. Als $f(a) = 0$, is $f \equiv 0$ bij $a$.
Anders, voor $0 < r \leq \rho$, geeft de middelwaarde
\[
\abs{f(a)} \leq \frac1{2\pi}\int_0^{2\pi}\abs{f(a +
r\eu^{\iu t})}\,\dd t \leq \abs{f(a)} :
\]
de continue niet-negatieve integrand $\abs{f(a)} - \abs{f(a +
r\eu^{\iu t})}$ heeft middelwaarde nul, verdwijnt dus:
$\abs f$ is \emph{constant} $= \abs{f(a)} \ne 0$ op de
schijf. Een \omterm{def:b3:holomorphic:holo}{holomorfe functie} van constante niet-nul
modulus op een schijf is constant: differentiëren van $P^2 +
Q^2 = \text{const}$ geeft $PP_x + QQ_x = 0$ en $PP_y + QQ_y
= 0$; substitueren van de Cauchy--Riemannrelaties $P_y =
-Q_x$, $Q_y = P_x$ in de tweede vergelijking levert het
lineaire stelsel
\[
P\,P_x + Q\,Q_x = 0, \qquad -P\,Q_x + Q\,P_x = 0,
\]
met determinant $P^2 + Q^2 \neq 0$: $P_x = Q_x = 0$, dus $f'
= P_x + \iu Q_x = 0$ op de schijf: $f$ constant daar. De
identiteitsstelling verspreidt constantheid over heel
$\Omega$. De randvorm: $\abs f$ bereikt haar sup op het
compacte $\bar\Omega$; een inwendig maximum maakt $f$
constant, en het sup wordt in elk geval op de rand bereikt.
\end{proof}

\begin{theorem}[Convergentstelling van Weierstrass]
\label{thm:b3:holomorphic:weierstrassconv}
Als $f_n \in \mathcal H(\Omega)$ naar $f$ convergeren,
uniform op elke compacte deelverzameling van $\Omega$, dan $f
\in \mathcal H(\Omega)$ en $f_n^{(k)} \to f^{(k)}$ uniform
op compacta, voor elke $k$.\index{convergentstelling van Weierstrass}
\end{theorem}

\begin{proof}
$f$ is continu. Voor elke gesloten driehoek $T \subseteq
\Omega$: $\int_{\partial T}f = \lim\int_{\partial T}f_n = 0$
(uniforme convergentie op het compacte $\partial T$; Goursat
voor $f_n$). Door het argument van
\cref{thm:b3:holomorphic:cauchy} heeft $f$ lokale
primitieven (schijven zijn convex; slechts de
driehoekseigenschap werd gebruikt), d.w.z.\ $f = F'$ lokaal
met $F$ \omterm{def:b3:holomorphic:holo}{holomorf}; $F$ is analytisch
(\cref{thm:b3:holomorphic:analytic}), dus ook $f = F'$:
\omterm{def:b3:holomorphic:holo}{holomorf}. (Dit is de \emph{stelling van Morera}%
\index{stelling van Morera}: continu met verdwijnende
driehoeksintegralen impliceert \omterm{def:b3:holomorphic:holo}{holomorf}.) Afgeleiden: voor
$\bar D(a, 2r) \subseteq \Omega$ en $z \in \bar D(a, r)$,
geeft de formule van Cauchy voor afgeleiden
(differentiëren van \cref{thm:b3:holomorphic:formula}
onder de integraal, of de coëfficiëntformule) 
\[
\abs{f_n'(z) - f'(z)} =
\Bigl|\frac{1}{2\iu\pi}\int_{C_{2r}}\frac{f_n(w) - f(w)}{(w -
z)^2}\,\dd w\Bigr|
\leq \frac{2r\,\sup_{C_{2r}}\abs{f_n - f}}{r^2} \to 0
\]
uniform op $\bar D(a,r)$; bedek een compactum door eindig
veel zulke schijven, en itereer voor hogere $k$.
\end{proof}

\begin{method}\label{met:b3:holomorphic:toolkit}
De dagelijkse gereedschapskist. Om een functie \omterm{def:b3:holomorphic:holo}{holomorf} te
bewijzen: vertoon haar als machtreeks, samenstelling, lokaal
uniforme limiet (\cref{thm:b3:holomorphic:weierstrassconv}),
of integraal met holomorfe parameter
(\cref{exo:b3:holomorphic:7} --- differentieer onder $\int$
of pas Morera--Fubini toe). Om identiteiten te bewijzen:
bewijs ze op een segment of deelgebied en roep de
identiteitsstelling aan. Om te begrenzen:
\omterm{thm:b3:holomorphic:analytic}{Cauchy-schattingen} op de grootste beschikbare cirkel. Om
constantheid/niet-bestaan te bewijzen: Liouville of het
maximumprincipe. Ken altijd \emph{waar} je functie \omterm{def:b3:holomorphic:holo}{holomorf}
is en \emph{welke} schijven in $\Omega$ passen.
\end{method}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:b3:holomorphic:1}
(a) In welke punten zijn $z \mapsto \bar z$, $\abs z^2$,
$\operatorname{Re}z$ complex-differentieerbaar? \omterm{def:b3:holomorphic:holo}{Holomorf} op
een \omterm{def:b3:topology:topology}{open verzameling}?
(b) Toon dat $P(x, y) = x^2 - y^2$ het reële deel is van een
\omterm{def:b3:holomorphic:holo}{holomorfe functie} op $\C$, expliciet gevonden, en bepaal
\emph{alle} daarvan.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b3:holomorphic:2}
Bereken uit de definities: $\int_{C}z^n\,\dd z$ voor alle $n
\in \Z$, $C$ de eenheidscirkel; $\int_\gamma\bar z\,\dd z$
langs het segment $[0, 1+\iu]$ en langs het tweesegmentenpad
via $1$: concludeer dat $\bar z$ geen primitieve heeft op
enige \omterm{def:b3:topology:topology}{omgeving} van deze paden.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:holomorphic:3}
(a) Toon dat de principale logaritme $\log z = \ln\abs z +
\iu\arg z$ ($\arg \in \intoo{-\pi}\pi$) \omterm{def:b3:holomorphic:holo}{holomorf} is op
$\C\setminus\intoc{-\infty}0$ met afgeleide $\frac1z$
\emph{(primitieve van $\frac1z$ op het stervormige
snijvlak: \cref{thm:b3:holomorphic:cauchy}; fixeer de
constante)}.
(b) Toon dat er geen continue logaritme bestaat op $\C^*$
\emph{(haar afgeleide-vrije obstructie: de \omterm{def:b3:holomorphic:index}{index} van de
eenheidscirkel)}.
(c) Ontwikkel $\log(1 + z)$ in machtreeks op $D(0,1)$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:holomorphic:4}
(a) Zij $f$ geheel met $\abs{f(z)} \leq C(1 + \abs
z)^{n}$. Toon dat $f$ een polynoom is van graad $\leq n$
\emph{(\omterm{thm:b3:holomorphic:analytic}{Cauchy-schattingen} op grote cirkels)}.
(b) Zij $f$ geheel met $\operatorname{Re}f$ van boven
begrensd. Toon dat $f$ constant is \emph{(beschouw
$\eu^{f}$)}.
(c) Leid ``kleine Picard voor affiene afbeeldingen'' af: een
gehele functie die een halfvlak weglaat is constant.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:holomorphic:5}
(a) Zij $f$ \omterm{def:b3:holomorphic:holo}{holomorf} op een \omterm{def:b3:topology:connected}{samenhangend} $\Omega \ni 0$ met
$f(\frac1n) = \frac1{n^2}$ voor alle grote $n$. Bepaal $f$.
(b) Bestaat er een holomorfe $f$ op $\C^*$ met $f(\frac1n) =
\frac{(-1)^n}{n}$ voor alle $n \geq 1$? Motiveer.
(c) Vertoon twee verschillende \omterm{def:b3:holomorphic:holo}{holomorfe functies} op
$D(0,1)\sqcup D(3,1)$ die overeenstemmen op $D(0,1)$: waar
gebruikt de identiteitsstelling samenhang?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:holomorphic:6}
Zij $f$ \omterm{def:b3:holomorphic:holo}{holomorf} op de open eenheidsschijf $\mathbb D$,
continu op $\bar{\mathbb D}$, met $\abs f \equiv 1$ op de
randcirkel.
(a) Als $f$ geen nul in $\mathbb D$ heeft, toon dat $f$
constant is \emph{(pas het maximumprincipe toe op $f$ en op
$1/f$)}.
(b) Geef een voorbeeld met een nul waar $f$ niet constant is.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:holomorphic:7}
(Holomorfie onder de integraal) Zij $\mu$ een eindige \omterm{def:b3:measure:measure}{maat}
op een ruimte $X$ en $g \colon X\times\Omega \to \C$ met:
$g(x, \cdot) \in \mathcal H(\Omega)$ voor elke $x$, $g$
\omterm{def:b3:lebesgue:measurable}{meetbaar} in $x$, en $\abs g \leq h(x)$ met $h$ \omterm{def:b3:lebesgue:l1}{integreerbaar},
lokaal uniform in $z$. Toon dat $G(z) = \int_Xg(x,
z)\dd\mu(x)$ \omterm{def:b3:holomorphic:holo}{holomorf} is op $\Omega$. \emph{(Morera:
driehoeksintegralen verdwijnen door Fubini en Goursat;
continuïteit door gedomineerde convergentie. Pas daarna toe
op $\Gamma(z) = \int_0^\infty t^{z-1}\eu^{-t}\dd t$ op
$\{\operatorname{Re}z > 0\}$.)}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b3:holomorphic:8}
(Gauss--Lucas) Zij $P \in \C[X]$ niet-constant. Toon dat
elke wortel van $P'$ in de convexe omhullende van de
wortels van $P$ ligt. \emph{(Schrijf $\frac{P'}{P} =
\sum_k\frac{m_k}{z - a_k}$ in een wortel $z$ van $P'$ die
geen wortel van $P$ is, neem conjugaten, en lees een
convexe combinatie.)} Illustreer op $P = z^3 - 1$.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b3:holomorphic:9}
Zij $f$ geheel en dubbel periodiek: $f(z + 1) = f(z + \iu)
= f(z)$ voor alle $z$. Toon dat $f$ constant is.
\emph{(Begrens $f$ op het compacte fundamentele vierkant,
dan overal; Liouville.)} Moraal: niet-constante elliptische
functies moeten polen hebben --- het thema van
\cref{ch:b3:residues}.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:holomorphic:10}
(a) Toon dat $P = \operatorname{Re}f$ van een holomorfe $f$
de middelwaarde-eigenschap $P(a) =
\frac1{2\pi}\int_0^{2\pi}P(a + r\eu^{\iu t})\dd t$
vervult en \emph{harmonisch} is: $\partial^2_{xx}P +
\partial^2_{yy}P = 0$ (differentiëer Cauchy--Riemann; gebruik
\cref{thm:b3:holomorphic:analytic} voor de nodige
gladheid).
(b) Leid het maximumprincipe af voor reële delen van
\omterm{def:b3:holomorphic:holo}{holomorfe functies} op begrensde gebieden.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b3:holomorphic:11}
(Schwarz-reflectie) Zij $\Omega^+ = \{z : \abs z < 1,\
\operatorname{Im}z > 0\}$, $I = \intoo{-1}1$, en $f$
\omterm{def:b3:holomorphic:holo}{holomorf} op $\Omega^+$, continu op $\Omega^+\cup I$,
\emph{reëelwaardig op $I$}. Definieer
\[
F(z) = \begin{cases} f(z) & z \in \Omega^+\cup I,\\
\overline{f(\bar z)} & \bar z \in \Omega^+ . \end{cases}
\]
(a) Toon dat $F$ goed gedefinieerd en continu is op $\Omega
= \Omega^+\cup I\cup\Omega^-$, en \omterm{def:b3:holomorphic:holo}{holomorf} op $\Omega^\pm$
\emph{(voor $\Omega^-$: controleer Cauchy--Riemann voor
$\overline{f(\bar z)}$, of ontwikkel $f$ in lokale
machtreeksen en conjugateer coëfficiënten)}.
(b) Toon dat $F$ \omterm{def:b3:holomorphic:holo}{holomorf} is op heel $\Omega$ via het
criterium van Morera: $\int_{\partial T}F = 0$ voor elke
driehoek $T \subseteq \Omega$ \emph{(splits driehoeken bij
$I$ en duw hun horizontale zijden met $\varepsilon$ van de
as af, met uniforme continuïteit)}.
(c) Leid af: een \omterm{def:b3:holomorphic:holo}{holomorfe functie} op de schijf, reëel op
een diameter, voldoet aan $f(\bar z) = \overline{f(z)}$; en
een niet-constante \omterm{def:b3:holomorphic:holo}{holomorfe functie} kan niet reëelwaardig
zijn op enige niet-lege open deelverzameling van haar
(samenhangende) gebied.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:holomorphic:12}
(De complexe Pythagoreïsche vergelijking) Vind alle paren
\emph{gehele} functies met $f^2 + g^2 = 1$.
(a) Toon dat $h = f + \iu g$ geheel en nultvrij is, en dat
elke nultvrije gehele functie $\eu^{\varphi}$ is voor enige
gehele $\varphi$ \emph{($h'/h$ is geheel, heeft dus een
primitieve op het stervormige $\C$; pas de constante aan en
toon dat $h\eu^{-\varphi}$ constant is)}.
(b) Concludeer $f = \cos\varphi$, $g = \sin\varphi$ met
$\varphi$ geheel, en controleer het omgekeerde. Wat zijn de
gehele oplossingen van $f^2 + g^2 = 0$?
\end{exercise}

\section{Probleem: de hoofdstelling van de algebra, tweemaal}

\begin{problem}[{Weekendprobleem --- $\C$ is algebraïsch
gesloten: het bewijs van d'Alembert, het bewijs van Liouville,
en de oogst}]\label{pb:b3:holomorphic:1}
Zij $P(z) = z^n + a_{n-1}z^{n-1} + \dots + a_0$, $n \geq 1$.
We bewijzen tweemaal dat $P$ een wortel heeft, en verzamelen
dan wat de algebra sinds \cref{ch:b3:galois} heeft staan
wachten.

\medskip
\noindent\textbf{Deel I --- Coërciviteit en het minimum.}
\begin{enumerate}
  \item Toon dat $\abs{P(z)} \to +\infty$ als $\abs z \to
        \infty$: precies, $\abs{P(z)} \geq \frac12\abs z^n$
        voor $\abs z \geq R_0$ geschikt.
  \item Leid af dat $\abs P$ een globaal minimum op $\C$
        bereikt: er is $z_0$ met $\abs{P(z_0)} =
        \inf_\C\abs P$ \emph{(compactheid van een grote
        gesloten schijf, \cref{cor:b3:topology:heineborel})}.
\end{enumerate}

\medskip
\noindent\textbf{Deel II --- De afdaling van d'Alembert.}
Stel, voor contradictie, $P(z_0) \neq 0$.
\begin{enumerate}[resume]
  \item Ontwikkel $Q(h) = P(z_0 + h)/P(z_0)$ als polynoom in
        $h$: $Q(h) = 1 + c_kh^k + h^{k+1}S(h)$ met $c_k \neq
        0$, $k \geq 1$, $S$ een polynoom.
  \item Kies de afdalingsrichting: voor kleine $t > 0$, zet $h
        = t\,\omega$ waar $\omega^k = -1/c_k$ (waarom bestaat
        zo'n $\omega$? --- \emph{bewijs} het bestaan van
        $k$-de wortels van elk complex getal via de
        poolvorm, onafhankelijk van de te bewijzen stelling).
        Toon
        \[
        \abs{Q(t\omega)} \leq 1 - t^k + C\,t^{k+1}
        \]
        voor kleine $t$, met een expliciete constante $C$.
  \item Concludeer $\abs{Q(t\omega)} < 1$ voor kleine $t$ ---
        in tegenspraak met de minimaliteit van
        $\abs{P(z_0)}$. Dus $P(z_0) = 0$: \emph{elk
        niet-constant complex polynoom heeft een wortel}
        (d'Alembert--Argand).
\end{enumerate}

\medskip
\noindent\textbf{Deel III --- De one-liner van Liouville,
\omterm{def:b3:complete:complete}{volledig}.}
\begin{enumerate}[resume]
  \item Schrijf zorgvuldig het bewijs van
        \cref{cor:b3:holomorphic:liouville} uit: als $P$ geen
        wortel heeft, verifieer dat $1/P$ geheel is,
        begrensd (kwantificeer, met vraag~1), dus constant,
        en concludeer. Vergelijk de twee bewijzen: welke
        ingrediënten gebruikt elk? (Compactheid verschijnt
        in beide --- waar?)
\end{enumerate}

\medskip
\noindent\textbf{Deel IV --- De oogst.}
\begin{enumerate}[resume]
  \item Toon dat elk $P \in \C[X]$ van graad $n$ splitst: $P
        = c\prod_{i}(X - \alpha_i)^{m_i}$ met $\sum m_i = n$
        \emph{(inductie, Euclidische deling door $(X -
        \alpha)$)}.
  \item Toon dat de irreducibele polynomen van $\R[X]$ de
        lineaire zijn en de kwadratische met negatieve
        discriminant \emph{(paar geconjugeerde wortels)};
        leid af dat elk reëel polynoom van oneven graad een
        reële wortel heeft, en geef een tweede,
        orden-theoretisch bewijs van dat laatste feit
        (tussenwaardestelling) --- controleer dat zij
        overeenstemmen op $X^3 - X - 1$.
  \item Leid de schulden af die dit boek nu kan afbetalen:
        (i) elk endomorfisme van een niet-nul
        eindigdimensionale $\C$-vectorruimte heeft een
        eigenwaarde, dus elke complexe matrix heeft een
        Jordan-vorm (\cref{thm:b3:modules:jordan}); (ii) het
        lichaam $\bar\Q$ van algebraïsche getallen gebruikt
        in \cref{rem:b3:galois:closureexamples} is inderdaad
        een \omterm{def:b3:galois:closure}{algebraïsche afsluiting} van $\Q$.
  \item (Finale) Wijs aan waar elk bewijs zou breken over
        een lichaam als $\Q(\iu)$: welke stappen gebruiken
        het bestaan van $k$-de wortels (vraag~4), en welke
        compactheid of volledigheid (vragen~2 en~6)?
        Concludeer in vijf regels: de stelling is echt
        \emph{analytisch} --- elk bewijs roept ergens de
        volledigheid of samenhang van $\R$ aan --- ook al is
        haar uitspraak puur \omterm{def:b3:galois:algebraic}{algebraïsch}.
\end{enumerate}

\medskip
\noindent\textbf{Deel V --- De starheidsladder van gehele
functies.} Liouville is de eerste sport van een ladder; we
beklimmen haar.
\begin{enumerate}[resume]
  \item (\omterm{thm:b3:holomorphic:analytic}{Cauchy-schattingen}) Uit de Cauchy-formule op de
        cirkel van straal $r$ om $a$, bewijs
        \[
        \bigl|f^{(n)}(a)\bigr| \leq
        \frac{n!\,\sup_{\abs{z-a}=r}\abs f}{r^n} ,
        \]
        en herwin Liouville als het geval $n = 1$, $r \to
        \infty$.
  \item (Polynomiale groei) Toon dat een gehele $f$ met
        $\abs{f(z)} \leq A + B\abs z^m$ voor alle $z$ een
        polynoom is van graad $\leq m$ \emph{(dood de
        Taylorcoëfficiënten voorbij $m$ met vraag~11)}.
  \item (Begrensd reëel deel) Toon dat een gehele $f$ met
        $\operatorname{Re}f$ van boven begrensd constant is
        \emph{(pas Liouville toe op $\eu^{f}$)}.
  \item (Dubbele periodiciteit) Zij $f$ geheel met $f(z + 1)
        = f(z)$ en $f(z + \iu) = f(z)$ voor alle $z$. Toon
        dat $f$ constant is. Concludeer: een niet-constante
        ``elliptische'' functie moet singulariteiten hebben
        --- de historische reden waarom polen de complexe
        analyse binnentreden.
  \item (Dichte beeldruimte) Toon dat het beeld van een
        niet-constante gehele functie \omterm{def:b3:topology:interior}{dicht} is in $\C$: als
        $f(\C)$ een schijf $D(a, r)$ mist, dan is
        $\frac1{f - a}$ geheel en begrensd. (Picard bewees
        dat het beeld hoogstens \emph{één punt} mist;
        \omterm{ex:b3:lebesgue:gamma}{dichtheid} is het niveau dat onze gereedschappen
        bereiken.)
  \item (Proper $\Rightarrow$ polynoom) Stel $f$ is geheel en
        $\abs{f(z)} \to \infty$ als $\abs z \to \infty$.
        Toon: de nullen van $f$ zijn eindig in aantal ($z_1,
        \dots, z_p$, met multipliciteiten $m_i$); het
        quotiënt $g = f/\prod(z - z_i)^{m_i}$ is geheel en
        nultvrij; $1/g$ heeft polynomiale groei, is dus
        (vraag~12) een polynoom, noodzakelijk constant
        (nultvrij); concludeer dat $f$ een polynoom is. Dus
        onder gehele functies zijn polynomen precies de
        properen --- $\eu^z$ faalt properheid langs $\R_-$.
\end{enumerate}

\medskip
\noindent\textbf{Deel VI --- Harmonische schaduwen en een
middelwaarde van Gauss.}
\begin{enumerate}[resume]
  \item Zij $f = u + \iu v$ \omterm{def:b3:holomorphic:holo}{holomorf} op een \omterm{def:b3:topology:topology}{open verzameling}.
        Verifieer dat $u = \operatorname{Re}f$ de
        middelwaarde-eigenschap
        \[
        u(a) = \frac1{2\pi}\int_0^{2\pi}
        u\bigl(a + r\eu^{\iu\theta}\bigr)\,\dd\theta
        \]
        vervult (reële deel van de Cauchy-formule), en leid
        het maximumprincipe voor $u$ af op een begrensd
        gebied, met hetzelfde samenhangsbewijs als voor
        $\abs f$.
  \item (Middelwaarde van Gauss) Voor $a \in \C$ en $r > 0$
        met $\abs a \neq r$, bewijs
        \[
        \frac1{2\pi}\int_0^{2\pi}
        \log\bigl|a - r\eu^{\iu\theta}\bigr|\,\dd\theta
        = \log\max\bigl(\abs a, r\bigr)
        \]
        \emph{(als $\abs a > r$: $z \mapsto \log\abs{a - z}$
        is het reële deel van een holomorfe logaritme op een
        \omterm{def:b3:topology:topology}{omgeving} van de gesloten schijf --- waarom bestaat er
        één? --- dus vraag~17 past toe; als $\abs a < r$:
        factoriseer $\abs{a - r\eu^{\iu\theta}} = r\,\abs{1 -
        \frac ar\eu^{-\iu\theta}}$ en hergebruik het eerste
        geval)}.
  \item (\omterm{def:b3:measure:measure}{Mahler-maat}) Voor $P = c\prod_{i=1}^n(X -
        \alpha_i) \in \C[X]$, leid \emph{de formule van
        Jensen voor polynomen} af:
        \[
        \frac1{2\pi}\int_0^{2\pi}\log\bigl|P(\eu^{\iu\theta})
        \bigr|\,\dd\theta
        = \log\Bigl(\abs c\prod_{i}\max(1,
        \abs{\alpha_i})\Bigr) :
        \]
        het meetkundig gemiddelde van $\abs P$ op de
        eenheidscirkel leest de wortels buiten de schijf af.
        Verifieer op $P = X^2 - X$ en op $P = 2X - 1$.
  \item (Bernoulligetallen) Definieer de coëfficiënten $B_n$
        door $\frac{z}{\eu^z - 1} =
        \sum_{n\geq0}\frac{B_n}{n!} z^n$ bij $0$ (waarom is
        de linkerkant analytisch in $0$?). Leid de
        recurrentie $\sum_{k=0}^{n}\binom{n+1}kB_k = 0$ ($n
        \geq 1$) af uit $(\eu^z - 1)\cdot\frac z{\eu^z-1} =
        z$, bereken $B_0, \dots, B_6$, en toon $B_{2k+1} =
        0$ voor $k \geq 1$ \emph{(de functie $\frac
        z{\eu^z-1} + \frac z2$ is even)}. Deze getallen
        prijzen elke $\zeta(2k)$ in \cref{ch:b3:residues}.
  \item (Realiteit) Toon dat een gehele functie die reële
        waarden op $\R$ aanneemt overal $f(\bar z) =
        \overline{f(z)}$ vervult \emph{(vergelijk
        Taylorcoëfficiënten in $0$, of pas de
        identiteitsstelling toe op $z \mapsto
        \overline{f(\bar z)}$)}; leid opnieuw af dat
        niet-reële wortels van reële polynomen in
        geconjugeerde paren komen (de paarvorming van
        vraag~8, analytisch herbewijs).
  \item (De moraal) Monteer de starheidsladder: begrensd
        $\Rightarrow$ constant; polynomiaal begrensd
        $\Rightarrow$ polynoom; proper $\Rightarrow$
        polynoom; een schijf missen $\Rightarrow$ constant;
        dubbel periodiek $\Rightarrow$ constant. Contrasteer
        in een korte alinea met reële $\mathcal
        C^\infty$-functies (\omterm{def:b3:lp:mollifier}{bumpfuncties},
        \cref{thm:b3:lp:regularization}): waarom legt
        holomorfie, een puur lokale voorwaarde, globale wet
        en orde op?
\end{enumerate}

\medskip
\noindent\textbf{Deel VII --- Laatste oogst.}
\begin{enumerate}[resume]
  \item (Ongelijkheid van Landau) Voor $P =
        \sum_{k=0}^na_kX^k$, bewijs de middelwaarde
        $\frac1{2\pi}\int_0^{2\pi}
        \abs{P(\eu^{\iu\theta})}^2\dd\theta =
        \sum_k\abs{a_k}^2$ (orthogonaliteit van de
        $\eu^{\iu k\theta}$), en, met de puntsgewijze grens
        $\log t \leq t - 1$ om de middelwaarden van
        $\log\abs P^2$ en $\abs P^2$ te vergelijken, leid uit
        vraag~19 af dat
        \[
        \abs c\prod_{i}\max\bigl(1, \abs{\alpha_i}\bigr)
        \;\leq\;
        \Bigl(\sum_{k=0}^{n}\abs{a_k}^2\Bigr)^{1/2} :
        \]
        het product van de wortels buiten de
        eenheidsschijf wordt gecontroleerd door de
        $\ell^2$-grootte van de coëfficiënten (behandel
        wortels \emph{op} de cirkel door de ongelijkheid op
        $P(rX)$ toe te passen en $r \to 1$ te laten).
        Controleer op $X^2 - X$.
  \item (Bernoulligetallen groeien factorieel) Toon dat de
        convergentiestraal van $\sum\frac{B_n}{n!}z^n$
        \emph{exact} $2\pi$ is: minstens $2\pi$ omdat
        $z/(\eu^z - 1)$ \omterm{def:b3:holomorphic:holo}{holomorf} voortzet naar $D(0, 2\pi)$,
        hoogstens $2\pi$ omdat de som anders begrensd zou
        blijven bij $2\pi\iu$, waar $\abs{z/(\eu^z-1)} \to
        \infty$. Leid af
        \[
        \limsup_{k\to\infty}
        \Bigl(\frac{\abs{B_{2k}}}{(2k)!}\Bigr)^{1/2k}
        = \frac1{2\pi} :
        \]
        Bernoulligetallen groeien factorieel. Toegevend
        $B_{12} = -\frac{691}{2730}$ (de recurrentie van
        vraag~20, verder uitgevoerd), vergelijk
        $\abs{B_{12}} \approx 0.25311$ met de scherpere
        voorspelling $2\,(2k)!/(2\pi)^{2k} \approx 0.25305$
        bij $k = 6$ --- vier overeenstemmende cijfers van
        een asymptotische wet die \cref{ch:b3:residues}
        exact zal bewijzen, via $\zeta(2k)$.
  \item (Wortels bewegen continu) Zij $(P_j)$ monisch van
        graad $n$ met coëfficiënten die convergeren naar die
        van (monisch) $P$. Bewijs de \emph{Cauchy-grens}:
        elke wortel van een monisch $Q = X^n +
        \sum_{k<n}q_kX^k$ voldoet aan $\abs\alpha \leq 1 +
        \max_k\abs{q_k}$; leid af dat de wortels van de
        $P_j$ in een vast compactum blijven, en, door
        convergente deelrijen van wortelvectoren te
        extraheren en over te gaan tot de limiet in de
        factorisatie van vraag~7, dat de
        wortelmultisets van $P_j$ naar die van $P$
        convergeren. Toon ten slotte dat continuïteit het
        beste is wat men kan zeggen: voor $P_\varepsilon =
        X^2 - 2X + 1 + \varepsilon$ verplaatst een
        perturbatie van grootte $\varepsilon = 10^{-4}$ de
        dubbele wortel $1$ met $10^{-2}$ ---
        Hölder-exponent $\frac1m$ bij een $m$-voudige
        wortel, nooit Lipschitz: numeriek kosten meervoudige
        wortels de helft van de cijfers.
\end{enumerate}
\end{problem}
