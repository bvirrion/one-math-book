\chapter{Maattheorie}\label{ch:b3:measure}

Hoe lang is een deelverzameling van $\R$? Het naïeve antwoord ---
aan elke verzameling een translatie-invariante lengte toekennen die
die van intervallen uitbreidt --- is \emph{onmogelijk}: de constructie
van Vitali, aan het eind van dit hoofdstuk, produceert een verzameling
zonder consistente lengte. Maattheorie is de gedisciplineerde terugtocht:
beperk de aandacht tot een rijke klasse van \emph{meetbare}
verzamelingen, waarop een aftelbaar additieve lengte bestaat en uniek
is. De beloning is immens --- de integraal van Lebesgue
(\cref{ch:b3:lebesgue}), de $L^p$-ruimten van de functionaalanalyse,
en de hele moderne kansrekening (\cref{ch:b3:probability}) rusten op
de drie stellingen die hier bewezen worden: het uniciteitslemma van
Dynkin, de uitbreidingsstelling van Carathéodory, en het bestaan van
de \omterm{def:b3:measure:lebesgueouter}{Lebesgue-maat}.

\section{$\sigma$-algebra's}

\begin{definition}\label{def:b3:measure:sigmaalgebra}
Een \emph{$\sigma$-algebra}\index{sigma-algebra@$\sigma$-algebra}
op een verzameling $X$ is een familie $\mathcal A$ van
deelverzamelingen die $\varnothing$ bevat, stabiel is onder
complement en onder \emph{aftelbare} unies (vandaar aftelbare
doorsnedes, verzamelingsverschillen, en zij bevat $X$). Het paar
$(X, \mathcal A)$ is een \emph{meetbare ruimte}; elementen van
$\mathcal A$ heten \emph{meetbare verzamelingen}. Voor een
willekeurige familie $\mathcal E$ van deelverzamelingen duidt
$\sigma(\mathcal E)$ de kleinste $\sigma$-algebra aan die
$\mathcal E$ bevat (de doorsnede van alle zulke --- een doorsnede
van $\sigma$-algebra's is er een).
\end{definition}

\begin{definition}\label{def:b3:measure:borel}
De \emph{Borel-$\sigma$-algebra}\index{Borel sigma-algebra@%
Borel $\sigma$-algebra} van een topologische ruimte is
$\mathcal B(X) = \sigma(\{\text{open verzamelingen}\})$. Op $\R$:
$\mathcal B(\R)$ wordt ook voortgebracht door de open intervallen,
door de gesloten intervallen, door de stralen $(-\infty, a]$, en
door stralen met rationale eindpunten (\cref{exo:b3:measure:1}) ---
elke familie brengt de opens voort via aftelbare operaties, b.v.\
is elke \omterm{def:b3:topology:topology}{open verzameling} van $\R$ een aftelbare unie van open
intervallen met rationale data.
\end{definition}

\begin{definition}\label{def:b3:measure:dynkin}
Een \emph{$\pi$-systeem}\index{pi-system@$\pi$-systeem} is een
familie stabiel onder eindige doorsnedes. Een
\emph{$\lambda$-systeem}\index{lambda-system@$\lambda$-systeem}
(Dynkin-klasse) is een familie $\mathcal D$ met: $X \in \mathcal
D$; $A, B \in \mathcal D$, $A \subseteq B$ $\Rightarrow$ $B
\setminus A \in \mathcal D$; en $A_n \uparrow A$, $A_n \in
\mathcal D$ $\Rightarrow$ $A \in \mathcal D$.
\end{definition}

\begin{theorem}[Het $\pi$--$\lambda$-lemma van Dynkin]
\label{thm:b3:measure:dynkin}
Als een $\lambda$-systeem $\mathcal D$ een $\pi$-systeem
$\mathcal P$ bevat, dan $\mathcal D \supseteq
\sigma(\mathcal P)$.\index{lemma van Dynkin}
\end{theorem}

\begin{proof}
Zij $\mathcal D_0$ het kleinste $\lambda$-systeem dat $\mathcal
P$ bevat (doorsnede van alle zulke); het volstaat te tonen dat
$\mathcal D_0$ een $\sigma$-algebra is, want dan $\sigma(\mathcal
P) \subseteq \mathcal D_0 \subseteq \mathcal D$. Een
$\lambda$-systeem dat stabiel is onder eindige doorsnedes
\emph{is} een $\sigma$-algebra: complementen ($X \setminus A = X
\setminus A$ met $A \subseteq X$), eindige unies ($A \cup B =
X \setminus ((X\setminus A)\cap(X\setminus B))$), en aftelbare
unies via $\bigcup_{k \leq n}A_k \uparrow \bigcup_kA_k$. We
bewijzen dus dat $\mathcal D_0$ een $\pi$-systeem is, in twee
stappen. Zij
\[
\mathcal D_1 = \{A \in \mathcal D_0 : A \cap P \in \mathcal
D_0 \ \forall P \in \mathcal P\}.
\]
$\mathcal D_1$ is een $\lambda$-systeem (alle drie de axioma's
worden geverifieerd door doorsnijden met $P$: b.v.\ $(B\setminus
A)\cap P = (B \cap P)\setminus(A \cap P)$, een echt verschil
binnen $\mathcal D_0$) en bevat $\mathcal P$ ($\pi$-systeem):
$\mathcal D_1 = \mathcal D_0$. Zij nu
\[
\mathcal D_2 = \{A \in \mathcal D_0 : A \cap D \in \mathcal
D_0\ \forall D \in \mathcal D_0\}.
\]
Door de vorige stap $\mathcal D_2 \supseteq \mathcal P$; en
$\mathcal D_2$ is een $\lambda$-systeem door dezelfde
verificatie: $\mathcal D_2 = \mathcal D_0$, wat precies zegt dat
$\mathcal D_0$ stabiel is onder doorsnedes.
\end{proof}

\section{Maten}

\begin{definition}\label{def:b3:measure:measure}
Een \emph{maat}\index{maat} op $(X, \mathcal A)$ is een
afbeelding $\mu \colon \mathcal A \to [0, +\infty]$ met
$\mu(\varnothing) = 0$ die \emph{$\sigma$-additief} is: voor
paarsgewijs disjuncte $(A_n)_{n\in\N}$,
\[
\mu\Bigl(\bigsqcup_n A_n\Bigr) = \sum_n \mu(A_n).
\]
$(X, \mathcal A, \mu)$ is een \emph{maatruimte}; $\mu$ is
\emph{eindig} als $\mu(X) < \infty$, een \emph{kansmaat} als
$\mu(X) = 1$, \emph{$\sigma$-eindig} als $X$ een aftelbare unie
is van verzamelingen van eindige maat. Voorbeelden: de
telmaat op $(\N, \mathcal P(\N))$; de Dirac-massa $\delta_a(A)
= \mathbf 1_{a \in A}$; en, het object van dit hoofdstuk, de
\omterm{def:b3:measure:lebesgueouter}{Lebesgue-maat}.
\end{definition}

\begin{proposition}\label{prop:b3:measure:basics}
Zij $\mu$ een \omterm{def:b3:measure:measure}{maat}. (a) Monotonie: $A \subseteq B
\Rightarrow \mu(A) \leq \mu(B)$. (b) Aftelbare subadditiviteit:
$\mu(\bigcup A_n) \leq \sum\mu(A_n)$. (c) Continuïteit van
onderen: $A_n \uparrow A \Rightarrow \mu(A_n) \to \mu(A)$. (d)
Continuïteit van boven: $A_n \downarrow A$ \emph{met
$\mu(A_1) < \infty$} $\Rightarrow \mu(A_n) \to \mu(A)$.
\end{proposition}

\begin{proof}
(a) $B = A \sqcup (B\setminus A)$. (b) Disjunctificeer: $B_n =
A_n \setminus \bigcup_{k<n}A_k$ zijn disjunct met dezelfde unie,
en $\mu(B_n) \leq \mu(A_n)$. (c) $A = \bigsqcup_n (A_n
\setminus A_{n-1})$ ($A_0 = \varnothing$): de partiële sommen
van $\sum\mu(A_n\setminus A_{n-1})$ zijn $\mu(A_n)$. (d) Pas
(c) toe op $A_1 \setminus A_n \uparrow A_1 \setminus A$ en trek
af van $\mu(A_1)$ --- eindigheid maakt de aftrekking legitiem.
Tegenvoorbeeld zonder die aanname: $A_n = [n, \infty)$ voor de
\omterm{def:b3:measure:lebesgueouter}{Lebesgue-maat}: $A_n \downarrow \varnothing$ maar $\mu(A_n) =
\infty$.
\end{proof}

\begin{theorem}[Uniciteit]\label{thm:b3:measure:uniqueness}
Zijn $\mu, \nu$ maten op $\sigma(\mathcal P)$, $\mathcal
P$ een $\pi$-systeem, met $\mu = \nu$ op $\mathcal P$. Als er
verzamelingen $P_k \in \mathcal P$ bestaan met $P_k \uparrow X$
en $\mu(P_k) < \infty$, dan $\mu = \nu$ op heel
$\sigma(\mathcal P)$.
\end{theorem}

\begin{proof}
Fixeer $k$ en beschouw de eindige maten $\mu_k(A) = \mu(A \cap
P_k)$ en $\nu_k(A) = \nu(A \cap P_k)$ op $\sigma(\mathcal
P)$: zij stemmen overeen op $\mathcal P$, want $P \cap P_k \in
\mathcal P$ ($\pi$-systeem), en zij kennen $X$ dezelfde
eindige waarde $\mu(P_k)$ toe. De klasse $\mathcal D = \{A :
\mu_k(A) = \nu_k(A)\}$ is een $\lambda$-systeem: $X \in
\mathcal D$; echte verschillen door aftrekking (eindige
waarden); toenemende limieten door continuïteit van onderen
(\cref{prop:b3:measure:basics}(c)). Zij bevat het
$\pi$-systeem $\mathcal P$, dus Dynkin
(\cref{thm:b3:measure:dynkin}) geeft $\mathcal D \supseteq
\sigma(\mathcal P)$: $\mu_k = \nu_k$ overal. Tenslotte, voor
elke $A \in \sigma(\mathcal P)$, continuïteit van onderen langs
$A \cap P_k \uparrow A$ geeft $\mu(A) = \lim_k\mu_k(A) =
\lim_k\nu_k(A) = \nu(A)$.
\end{proof}

\section{Uitwendige maten en de stelling van Carathéodory}

\begin{definition}\label{def:b3:measure:outer}
Een \emph{uitwendige maat}\index{uitwendige maat} op $X$ is een
afbeelding $\mu^* \colon \mathcal P(X) \to [0, \infty]$ met
$\mu^*(\varnothing) = 0$, monotoon, en aftelbaar subadditief.
Een verzameling $A$ is \emph{$\mu^*$-meetbaar} (Carathéodory)
als zij elke verzameling additief splitst:
\[
\mu^*(E) = \mu^*(E \cap A) + \mu^*(E \setminus A)
\qquad \text{voor elke } E \subseteq X
\]
($\leq$ geldt altijd door subadditiviteit; de \omterm{def:b3:rings:content}{inhoud} is $\geq$).
\end{definition}

\begin{theorem}[Carathéodory]\label{thm:b3:measure:caratheodory}
De $\mu^*$-meetbare verzamelingen vormen een $\sigma$-algebra
$\mathcal M$, en $\mu^*\restriction_{\mathcal M}$ is een \omterm{def:b3:measure:measure}{maat}.
Bovendien behoort elke verzameling met $\mu^*(N) = 0$ tot
$\mathcal M$ (de \omterm{def:b3:measure:measure}{maat} is \emph{\omterm{def:b3:complete:complete}{volledig}}).\index{stelling van
Carathéodory}
\end{theorem}

\begin{proof}
$\mathcal M$ bevat $\varnothing$ en is stabiel onder complement
(de definiërende voorwaarde is symmetrisch in $A$,
$X\setminus A$). \emph{Eindige unies}: zij $A, B \in \mathcal
M$ en $E$ willekeurig; splits $E$ door $A$, dan elk stuk door
$B$:
\[
\mu^*(E) = \mu^*(E\cap A\cap B) + \mu^*(E\cap A\setminus B) +
\mu^*(E\cap B\setminus A) + \mu^*(E\setminus(A\cup B)).
\]
De eerste drie stukken overdekken $E \cap (A \cup B)$, dus
subadditiviteit geeft $\mu^*(E) \geq \mu^*(E\cap(A\cup B)) +
\mu^*(E\setminus(A\cup B))$: $A \cup B \in \mathcal M$. Door
inductie, eindige unies; met complementen zijn eindige
disjunctheidsmanipulaties beschikbaar.

\emph{Additiviteit op $\mathcal M$}: voor disjuncte $A, B \in
\mathcal M$ en willekeurige $E$: $\mu^*(E\cap(A\sqcup B)) =
\mu^*(E\cap A) + \mu^*(E \cap B)$ (splits door $A$); door
inductie,
\[
\mu^*\Bigl(E \cap \bigsqcup_{k\leq n}A_k\Bigr)
= \sum_{k\leq n}\mu^*(E\cap A_k).
\tag{$*$}
\]

\emph{Aftelbare unies}: zij $(A_k) \subseteq \mathcal M$
disjunct (voldoende, door disjunctificatie binnen de algebra
$\mathcal M$), $A = \bigsqcup A_k$, $E$ willekeurig. Met
$\bigsqcup_{k\leq n}A_k \in \mathcal M$ en monotonie:
\[
\mu^*(E) = \mu^*\Bigl(E\cap\bigsqcup_{k\leq n}A_k\Bigr) +
\mu^*\Bigl(E\setminus\bigsqcup_{k\leq n}A_k\Bigr)
\geq \sum_{k \leq n}\mu^*(E\cap A_k) + \mu^*(E\setminus A)
\]
door ($*$). Laat $n \to \infty$ en gebruik aftelbare
subadditiviteit terug:
\[
\mu^*(E) \geq \sum_{k}\mu^*(E\cap A_k) + \mu^*(E\setminus A)
\geq \mu^*(E \cap A) + \mu^*(E\setminus A) \geq \mu^*(E):
\]
alle ongelijkheden zijn gelijkheden. Dit bewijst zowel $A \in
\mathcal M$ als, met $E = A$, aftelbare additiviteit van
$\mu^*$ op $\mathcal M$.

\emph{Nulverzamelingen}: als $\mu^*(N) = 0$, dan voor elke $E$:
$\mu^*(E \cap N) + \mu^*(E\setminus N) \leq 0 + \mu^*(E)$:
$N \in \mathcal M$.
\end{proof}

\section{Lebesgue-maat op \texorpdfstring{$\R$}{R}}

\begin{definition}\label{def:b3:measure:lebesgueouter}
De \emph{uitwendige Lebesgue-maat}\index{Lebesgue-maat} van $A
\subseteq \R$ is
\[
\lambda^*(A) = \inf\Bigl\{\sum_{n} (b_n - a_n) :
A \subseteq \bigcup_n \intoo{a_n}{b_n}\Bigr\}
\]
(aftelbare overdekkingen door open intervallen).
\end{definition}

\begin{lemma}\label{lem:b3:measure:outerbasics}
$\lambda^*$ is een \omterm{def:b3:measure:outer}{uitwendige maat}, invariant onder
translaties, en $\lambda^*(I) = \ell(I)$ (de lengte) voor elk
interval $I$.
\end{lemma}

\begin{proof}
\omterm{def:b3:measure:outer}{Uitwendige maat}: $\varnothing$ wordt overdekt door willekeurig
kleine intervallen; monotonie is duidelijk; subadditiviteit:
gegeven overdekkingen van elke $A_n$ binnen $\varepsilon
2^{-n}$ van het infimum, overdekt hun unie $\bigcup A_n$ met
totale lengte $\leq \sum \lambda^*(A_n) + \varepsilon$.
Translatie-invariantie: translateer de overdekkingen.

Lengte: het volstaat $I = \intcc ab$ te behandelen (andere
types verschillen door eindpunten, die \omterm{def:b3:measure:outer}{uitwendige maat} $0$
hebben: overdek door piepkleine intervallen; dan knijp
$\intcc{a+\varepsilon}{b - \varepsilon} \subseteq
\intoo ab$-type vergelijkingen). $\lambda^*(\intcc ab) \leq b
- a$: overdek door $\intoo{a-\varepsilon}{b+\varepsilon}$.
Omgekeerd zij $\intcc ab \subseteq \bigcup_n\intoo{a_n}{b_n}$:
door \emph{compactheid} (Borel--Lebesgue,
\cref{thm:b3:topology:metriccompact}) volstaan eindig veel
intervallen, zeg $I_1, \dots, I_N$. We tonen $\sum_{k\leq
N}(b_k - a_k) \geq b - a$ door inductie op $N$: kies $I_{k_1}
\ni a$; als $b_{k_1} > b$ klaar ($b_{k_1} - a_{k_1} > b - a$);
anders wordt het segment $\intcc{b_{k_1}}b$ overdekt door de
resterende $N - 1$ intervallen, en inductie geeft $\sum_{k
\neq k_1}(b_k - a_k) \geq b - b_{k_1}$, terwijl $b_{k_1} -
a_{k_1} > b_{k_1} - a$: optellen.
\end{proof}

\begin{theorem}[Lebesgue-maat]\label{thm:b3:measure:lebesgue}
Elke Borelverzameling van $\R$ is $\lambda^*$-meetbaar. De
restrictie $\lambda$ van $\lambda^*$ tot de $\sigma$-algebra
$\mathcal L = \mathcal M_{\lambda^*} \supseteq \mathcal B(\R)$
(de \emph{Lebesgue-$\sigma$-algebra}) is de unieke \omterm{def:b3:measure:measure}{maat} op
$\mathcal B(\R)$ die aan elk interval zijn lengte toekent; zij
is translatie-invariant en $\sigma$-eindig.
\end{theorem}

\begin{proof}
Door \cref{thm:b3:measure:caratheodory} volstaat het te tonen
dat elke straal $A = \intoo{-\infty}c$ $\lambda^*$-meetbaar is
(stralen brengen $\mathcal B$ voort,
\cref{def:b3:measure:borel}). Zij $E \subseteq \R$ met
$\lambda^*(E) < \infty$ en $\bigcup I_n \supseteq E$ een
overdekking met $\sum\ell(I_n) \leq \lambda^*(E) +
\varepsilon$. Elk $I_n$ splitst in de twee intervallen $I_n' =
I_n \cap A$ en $I_n'' = I_n\setminus A$ (een interval min een
straal is een interval) met $\ell(I_n') + \ell(I_n'') =
\ell(I_n)$; de $I_n'$ overdekken $E \cap A$ en de $I_n''$
overdekken $E \setminus A$ (vergroot elk tot een open interval
van lengte $\ell + \varepsilon2^{-n}$ om binnen de definitie te
blijven), dus
\[
\lambda^*(E\cap A) + \lambda^*(E\setminus A)
\leq \sum_n\bigl(\ell(I_n') + \ell(I_n'')\bigr) + 2\varepsilon
\leq \lambda^*(E) + 3\varepsilon .
\]
Uniciteit: twee maten die met lengte overeenstemmen op het
$\pi$-systeem van intervallen $\intoc ab$ (eindig daarop)
stemmen overeen op $\sigma(\text{intervallen}) = \mathcal B$
door \cref{thm:b3:measure:uniqueness} met $P_k =
\intoc{-k}k$. $\sigma$-eindigheid: $\R = \bigcup(-k, k]$.
\end{proof}

\begin{theorem}[Regulariteit]\label{thm:b3:measure:regularity}
Voor elke $A \in \mathcal L$:\index{regulariteit van de
Lebesgue-maat}
\[
\lambda(A) = \inf\{\lambda(U) : U \supseteq A \text{ open}\}
= \sup\{\lambda(K) : K \subseteq A \text{ compact}\}.
\]
\end{theorem}

\begin{proof}
\emph{Uitwendig}: een overdekking $\bigcup I_n$ met
$\sum\ell(I_n) \leq \lambda(A) + \varepsilon$ is een \omterm{def:b3:topology:topology}{open
verzameling} $U \supseteq A$ met $\lambda(U) \leq \lambda(A) +
\varepsilon$ (subadditiviteit); als $\lambda(A) = \infty$ is de
uitspraak triviaal. \emph{Inwendig}: zij eerst $A$ begrensd, $A
\subseteq [-M, M]$. Kies een open $U \supseteq ([-M,M]\setminus
A)$ met $\lambda(U) \leq \lambda([-M,M]\setminus A) +
\varepsilon$; dan is $K = [-M, M]\setminus U$ compact, $K
\subseteq A$, en
\[
\lambda(K) = \lambda([-M,M]) - \lambda([-M,M]\cap U)
\geq \lambda([-M,M]) - \bigl(\lambda([-M,M]) - \lambda(A) +
\varepsilon\bigr) = \lambda(A) - \varepsilon .
\]
Voor algemene $A$: $\lambda(A) = \lim_M\lambda(A\cap[-M,M])$
(continuïteit van onderen) en pas het begrensde geval toe.
\end{proof}

\begin{example}\label{ex:b3:measure:cantor}
De Cantorverzameling (\cref{exo:b3:topology:10}) heeft
$\lambda(C) = 0$: $C \subseteq C_n$, een unie van $2^n$
intervallen van lengte $3^{-n}$, dus $\lambda(C) \leq
(2/3)^n \to 0$. Een overaftelbare nulverzameling ---
kardinaliteit ziet \omterm{def:b3:measure:measure}{maat} niet. Omgekeerd zijn \emph{vette
Cantorverzamelingen} (\cref{exo:b3:measure:5}) nergens \omterm{def:b3:topology:interior}{dicht}
met positieve \omterm{def:b3:measure:measure}{maat}: \omterm{def:b3:topology:topology}{topologie} ziet \omterm{def:b3:measure:measure}{maat} evenmin. Het
weekendprobleem duwt dit samenspel tot zijn treffende
conclusie: er bestaan Lebesgue-meetbare verzamelingen die niet
Borel zijn.
\end{example}

\begin{theorem}[Vitali]\label{thm:b3:measure:vitali}
Er bestaat geen \omterm{def:b3:measure:measure}{maat} op \emph{alle} deelverzamelingen van $\R$
die translatie-invariant is en aan elk interval zijn lengte
toekent. In het bijzonder $\mathcal L \neq \mathcal P(\R)$:
niet-meetbare verzamelingen bestaan.\index{niet-meetbare
verzameling van Vitali}
\end{theorem}

\begin{proof}
Stel $\mu$ is er een. Op $\intcc01$, beschouw de
equivalentierelatie $x \sim y \iff x - y \in \Q$; kies via het
\emph{keuzeaxioma} één representant in $\intcc01$ per klasse:
een verzameling $V$. Voor $q \in \Q\cap\intcc{-1}1$ zijn de
translaten $V + q$ paarsgewijs disjunct (twee punten van $V$ die
een rationaal verschillen zouden equivalente maar verschillende
representanten zijn) en
\[
\intcc01 \subseteq \bigsqcup_{q \in \Q\cap\intcc{-1}1}(V + q)
\subseteq \intcc{-1}2 :
\]
de eerste inclusie omdat elke $x \in \intcc01$ van haar
representant $v$ een rationaal $q = x - v \in \intcc{-1}1$
verschilt. Monotonie en $\sigma$-additiviteit geven
\[
1 \leq \sum_{q}\mu(V + q) \leq 3,
\qquad\text{met } \mu(V + q) = \mu(V) \text{ voor alle } q .
\]
Een oneindige som van de constante $\mu(V)$ is $0$ of $\infty$:
beide grenzen kunnen niet gelden. Dus bestaat zulk een $\mu$
niet --- en $V \notin \mathcal L$, want $\lambda$ op $\mathcal
L$ heeft alle gebruikte eigenschappen.
\end{proof}

\begin{remark}\label{rem:b3:measure:banachtarski}
In $\R^3$ is de mislukking dramatischer: de paradox van
Banach--Tarski ontleedt een bol in vijf stukken die zich, door
rotaties en translaties, herassembleren tot \emph{twee} bollen
van dezelfde straal --- zodat zelfs geen eindig additief
rotatie-invariant volume op alle deelverzamelingen van $\R^3$
bestaat. De stukken zijn natuurlijk niet-meetbaar. Meetbaarheid
is geen bureaucratische voorzorg; zij is de grens van coherentie.
\end{remark}

\begin{method}\label{met:b3:measure:goodsets}
Het \emph{principe van de goede verzamelingen}: om te bewijzen
dat alle verzamelingen van $\sigma(\mathcal E)$ een eigenschap
hebben, toon dat de goede verzamelingen een $\sigma$-algebra
vormen (of een $\lambda$-systeem, als de eigenschap
maattheoretisch is en $\mathcal E$ een $\pi$-systeem --- dan
Dynkin) die $\mathcal E$ bevat. Bijna elk bewijs in dit
hoofdstuk en het volgende is een instantie. Om twee maten
gelijk te bewijzen: controleer ze op een voortbrengend
$\pi$-systeem plus $\sigma$-eindigheid
(\cref{thm:b3:measure:uniqueness}). Om een \omterm{def:b3:measure:measure}{maat} te bouwen: bouw
een \omterm{def:b3:measure:outer}{uitwendige maat} door overdekkingen en citeer Carathéodory.
\end{method}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:b3:measure:1}
(a) Toon dat $\{A \subseteq X : A$ of $X\setminus A$ is
aftelbaar$\}$ een $\sigma$-algebra is: die voortgebracht door
de singletonen.
(b) Toon dat $\mathcal B(\R)$ wordt voortgebracht door elk van:
open intervallen; gesloten intervallen; stralen
$\intoc{-\infty}a$; stralen met $a \in \Q$.
(c) Is de familie van \emph{eindige disjuncte unies van
intervallen} $\intoc ab$ een $\sigma$-algebra? Een algebra
(stabiel onder complement en eindige unies)?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b3:measure:2}
(a) Bewijs inclusie-exclusie voor een eindige \omterm{def:b3:measure:measure}{maat}: $\mu(A\cup
B) = \mu(A) + \mu(B) - \mu(A\cap B)$, en de drieverzamelingen-
versie.
(b) Geef een voorbeeld dat toont dat continuïteit van boven
(\cref{prop:b3:measure:basics}(d)) faalt zonder de
eindigheidsassumptie.
(c) Toon dat een aftelbare verzameling \omterm{def:b3:measure:lebesgueouter}{Lebesgue-maat} nul heeft.
Leid af $\lambda(\Q) = 0$ en $\lambda(\intcc01\setminus\Q) =
1$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:measure:3}
Zijn $\mu, \nu$ kansmaten op $\mathcal B(\R)$
met $\mu(\intoc{-\infty}t) = \nu(\intoc{-\infty}t)$ voor alle
$t \in \R$. Toon $\mu = \nu$. (Dit maakt de
\emph{verdelingsfunctie} $F(t) = \mu(\intoc{-\infty}t)$ een
\omterm{def:b3:complete:complete}{volledig} invariant --- de grondslag van
\cref{ch:b3:probability}.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:measure:4}
(Borel--Cantelli, maatversie) Zij $(A_n)$ meetbaar
met $\sum_n\mu(A_n) < \infty$, en $\limsup A_n =
\bigcap_N\bigcup_{n\geq N}A_n$ (de punten die tot oneindig veel
$A_n$ behoren). Toon $\mu(\limsup A_n) = 0$.
Toepassing: voor bijna elke $x \in \intcc01$ voldoen slechts
eindig veel $n$ aan $\abs{x - p/q_n} \leq 4^{-n}$ voor de
$n$-de rationale $p/q_n$ van een enumeratie van
$\Q\cap\intcc01$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:measure:5}
(Vette Cantorverzameling) Herhaal de Cantorconstructie op
$\intcc01$, maar verwijder bij stap $n$ uit elk van de
$2^{n-1}$ intervallen een \emph{gecentreerd} open interval van
lengte $4^{-n}$ alleen. Toon dat de resulterende $K =
\bigcap K_n$ compact is, lege inwendige heeft (geen interval
overleeft), en
\[
\lambda(K) = 1 - \sum_{n\geq1}2^{n-1}4^{-n} = \tfrac12 :
\]
een nergens \omterm{def:b3:topology:interior}{dichte verzameling} van \omterm{def:b3:measure:measure}{maat} $\frac12$. Leid een
\emph{magere} deelverzameling van $\intcc01$ van volle \omterm{def:b3:measure:measure}{maat} $1$
af, en een open dichte deelverzameling van \omterm{def:b3:measure:measure}{maat} $< \varepsilon$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:measure:6}
Zij $\mu$ een \omterm{def:b3:measure:measure}{maat} op $\mathcal B(\R)$, invariant onder
translaties, met $c = \mu(\intoc01) < \infty$. Toon $\mu =
c\,\lambda$ op $\mathcal B(\R)$. \emph{(Bereken $\mu$ op
dyadische intervallen door $\intoc01$ in $2^n$ translaten te
splitsen, en roep dan \cref{thm:b3:measure:uniqueness} aan.)}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:measure:7}
(Approximatie) Zij $A \in \mathcal L$ met $\lambda(A) <
\infty$ en $\varepsilon > 0$. Toon dat er een \emph{eindige}
unie van intervallen $B$ bestaat met $\lambda(A\,\triangle\,B)
< \varepsilon$ ($\triangle$ = symmetrisch verschil).
\emph{(Regulariteit: knijp $K \subseteq A \subseteq U$ en
gebruik de structuur van de open $U$ als aftelbare unie van
intervallen, plus compactheid van $K$.)}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b3:measure:8}
(Steinhaus) Zij $A \in \mathcal L$ met $\lambda(A) > 0$. Toon
dat $A - A = \{x - y : x, y \in A\}$ een interval rond $0$
bevat.
\emph{(Reduceer tot $\lambda(A) < \infty$; door
regulariteit à la \cref{exo:b3:measure:7}, vind een interval $I$
met $\lambda(A \cap I) > \frac34\ell(I)$; dan voor $\abs t <
\frac12\ell(I)$ liggen de verzamelingen $A\cap I$ en $(A\cap I)
+ t$ beide in een interval van lengte $\frac32\ell(I)$ en hebben
totale \omterm{def:b3:measure:measure}{maat} $> \frac32\ell(I)$: zij moeten snijden.)}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b3:measure:9}
Toon dat elke $A \in \mathcal L$ met $\lambda(A) > 0$
een niet-meetbare deelverzameling bevat. \emph{(Snijd $A$ met de
translaten $V + q$ van de verzameling van Vitali: als alle $A
\cap (V+q)$ meetbaar waren, zou elk nul zijn door het argument
van \cref{thm:b3:measure:vitali} --- Steinhaus
(\cref{exo:b3:measure:8}) helpt: een meetbare verzameling van
positieve \omterm{def:b3:measure:measure}{maat} binnen $V + q$ zou $(V+q) - (V+q) \supseteq$ een
interval geven, in tegenspraak met dat dit verschil $\Q$ alleen
in $0$ snijdt; besluit met subadditiviteit.)}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:measure:10}
Toon dat $A \subseteq \R$ met $\lambda^*(A) < \infty$
Lebesgue-meetbaar is d.e.s.d.a.\ voor elke $\varepsilon > 0$ er
een open $U \supseteq A$ bestaat met $\lambda^*(U \setminus A)
< \varepsilon$, d.e.s.d.a.\ er een $G_\delta$-verzameling $G
\supseteq A$ bestaat met $\lambda^*(G\setminus A) = 0$. (Dus
Lebesgue-verzamelingen zijn Borelverzamelingen modulo
nulverzamelingen.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:measure:11}
(Continuïteit langs monotone limieten, en haar scherpte)
(a) Toon dat voor meetbare verzamelingen $\mu(\liminf A_n)
\leq \liminf\mu(A_n)$ (Fatou voor verzamelingen), en dat als
$\mu\bigl(\bigcup A_n\bigr) < \infty$, ook
$\limsup\mu(A_n) \leq \mu(\limsup A_n)$.
(b) Vertoon, voor de \omterm{def:b3:measure:lebesgueouter}{Lebesgue-maat} op $\R$, een rij met
$\mu(A_n) = 1$ voor alle $n$ maar $\mu(\limsup A_n) = 0$: de
eindigheidshypothese in de tweede ongelijkheid is geen
decoratie.
(c) Leid af: als $\sum\mu(A_n) < \infty$ dan
$\mu(\limsup A_n) = 0$ (Borel--Cantelli opnieuw), en als de
$A_n$ toenemen of afnemen (met $\mu(A_1) < \infty$ in het
afnemende geval), $\mu(\lim A_n) = \lim\mu(A_n)$.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b3:measure:12}
(Stelling van Egorov) Zij $\mu(X) < \infty$ en $f_n \to f$
puntsgewijs, alle meetbaar (reëelwaardig). Voor $k, N \geq 1$
stel
\[
E_{k,N} = \bigcap_{n \geq N}\Bigl\{x : \abs{f_n(x) - f(x)}
\leq \tfrac1k\Bigr\} .
\]
(a) Toon dat voor vaste $k$, $E_{k,N} \nearrow X$ als $N \to
\infty$, en leid $N_k$ af met $\mu(X \setminus E_{k,N_k})
\leq \varepsilon2^{-k}$.
(b) Besluit de \emph{stelling van Egorov}: voor elke
$\varepsilon > 0$ bestaat een meetbare $A$ met $\mu(X\setminus
A) \leq \varepsilon$ zodat $f_n \to f$ \emph{uniform op $A$}
--- puntsgewijze convergentie is uniforme convergentie buiten
een willekeurig kleine verzameling.
(c) Toon dat de stelling faalt op $(\R, \lambda)$: de
bewegende bulten $f_n = \mathbf 1_{\intcc n{n+1}}$ convergeren
puntsgewijs naar $0$ maar uniform op geen complement van een
verzameling van eindige \omterm{def:b3:measure:measure}{maat}. Waar gebruikte (a) $\mu(X) <
\infty$?
\end{exercise}

\section{Probleem: de trap van Cantor--Vitali en een
meetbare verzameling die niet Borel is}

{\small De trap van Cantor--Vitali: constant op elke spleet van

\begin{problem}[{Weekendprobleem --- de duivelstrap,
en $\mathcal B(\R) \subsetneq \mathcal L$}]
\label{pb:b3:measure:1}
We construeren de \emph{Cantor--Vitali-functie} (duivelstrap),
gebruiken haar om \omterm{def:b3:measure:measure}{maat} pathologisch te transporteren, en
besluiten met een stelling die geen zacht argument levert: er
bestaan Lebesgue-meetbare verzamelingen die niet Borel zijn.
Notatie: $C$ is de Cantorverzameling, $C_n$ haar $n$-de stadium
($2^n$ intervallen van lengte $3^{-n}$), en elke $x \in C$
heeft ternaire cijfers $x = \sum 2b_n3^{-n}$, $b_n \in \{0,1\}$
(\cref{exo:b3:topology:10}).

\medskip
\noindent\textbf{Deel I --- De trap.} Definieer $c_0(x) = x$
en $c_{n+1}$ uit $c_n$ door
\[
c_{n+1}(x) = \begin{cases}
\tfrac12\,c_n(3x) & 0 \leq x \leq \tfrac13,\\[2pt]
\tfrac12 & \tfrac13 \leq x \leq \tfrac23,\\[2pt]
\tfrac12 + \tfrac12\,c_n(3x - 2) & \tfrac23 \leq x \leq 1.
\end{cases}
\]
\begin{enumerate}
  \item Toon dat elke $c_n$ continu is, niet-dalend, met
        $c_n(0) = 0$, $c_n(1) = 1$, en dat
        $\norm{c_{n+1} - c_n}_\infty \leq
        \tfrac12\norm{c_n - c_{n-1}}_\infty$.
  \item Leid af dat $(c_n)$ uniform convergeert naar een
        continue niet-dalende $c$ met $c(0) = 0$, $c(1) = 1$
        (\emph{de} Cantor--Vitali-functie), die dezelfde
        zelfgelijke relaties vervult als de $c_{n+1}$ hierboven.
  \item Toon dat $c$ constant is op elke \omterm{def:b3:topology:components}{samenhangscomponent}
        van $\intcc01\setminus C$, en dat voor $x = \sum_n
        2b_n3^{-n} \in C$, $c(x) = \sum_n b_n2^{-n}$: de trap
        leest Cantor-cijfers in binair
        (\cref{pb:b3:topology:1}'s functie $g$, monotoon en
        globaal gemaakt).
  \item Leid af dat $c$ differentieerbaar is, met $c' = 0$, in
        elk punt van $\intcc01\setminus C$: $c' = 0$
        \emph{$\lambda$-bijna overal}
        (\cref{ex:b3:measure:cantor}). Besluit dat de
        fundamentele stelling van de calculus faalt voor $c$:
        \[
        c(1) - c(0) = 1 \neq 0 = \int_0^1 c'(t)\,\dd t
        \]
        (de integraal over de verzameling van volle \omterm{def:b3:measure:measure}{maat} waar
        $c' = 0$; anticiperend op \cref{ch:b3:lebesgue}
        beïnvloeden nulverzamelingen integralen niet). Welke
        hypothese van de $\mathcal C^1$ fundamentele stelling
        is geschonden?
  \item Toon dat $c(C) = \intcc01$: de nulverzameling $C$ wordt
        \emph{op} een verzameling van volle \omterm{def:b3:measure:measure}{maat} afgebeeld.
\end{enumerate}

\medskip
\noindent\textbf{Deel II --- Het scheve \omterm{def:b3:topology:continuity}{homeomorfisme}.} Zij
$h(x) = \frac{c(x) + x}{2}$.
\begin{enumerate}[resume]
  \item Toon dat $h \colon \intcc01 \to \intcc01$ een
        \omterm{def:b3:topology:continuity}{homeomorfisme} is (strikt stijgend, continu,
        surjectief).
  \item Toon dat $\lambda\bigl(h(\intcc01\setminus C)\bigr) =
        \tfrac12$: op elke spleet van lengte $\ell$ werkt $h$
        als een affiene afbeelding van helling $\tfrac12$, en
        de spleten hebben totale lengte $1$.
  \item Leid af $\lambda\bigl(h(C)\bigr) = \tfrac12$: het
        homeomorfe beeld van een nulverzameling kan positieve
        \omterm{def:b3:measure:measure}{maat} hebben. (Waar spreekt dit naïeve intuïtie over
        ``grootte'' tegen?)
\end{enumerate}

\medskip
\noindent\textbf{Deel III --- Een meetbare verzameling die niet
Borel is.}
\begin{enumerate}[resume]
  \item Kies via \cref{exo:b3:measure:9} een niet-meetbare $W
        \subseteq h(C)$. Toon dat $Z = h^{-1}(W) \subseteq C$
        Lebesgue-meetbaar is. \emph{(Zij is een deelverzameling
        van een nulverzameling; volledigheid,
        \cref{thm:b3:measure:caratheodory}.)}
  \item Toon dat het originieel van een Borelverzameling onder
        een \omterm{def:b3:topology:continuity}{continue afbeelding} Borel is. \emph{(Principe van
        de goede verzamelingen: $\{B : h^{-1}(B) \in \mathcal
        B\}$ is een $\sigma$-algebra die de \omterm{def:b3:topology:topology}{open verzamelingen}
        bevat --- let op de richting van de afbeelding.)}
  \item Besluit dat $Z$ \emph{niet} Borel is: als zij het was,
        zou $W = (h^{-1})^{-1}(Z)$ Borel zijn (pas vraag 10 toe
        op de continue $h^{-1}$), dus meetbaar ---
        tegenspraak. Derhalve
        \[
        \boxed{\ \mathcal B(\R) \subsetneq \mathcal L\ }
        \]
        en volledigheid vergroot werkelijk de Borelwereld.
  \item Vertoon een Lebesgue-meetbare functie $g$ en een
        continue functie $\varphi$ zodat $g \circ
        \varphi$ niet Lebesgue-meetbaar is: meetbaarheid,
        anders dan continuïteit, composeert niet.
        \emph{(Neem $g = \mathbf 1_Z$ en $\varphi = h^{-1}$,
        anticiperend op de definitie van meetbare functies uit
        \cref{ch:b3:lebesgue}: originelen van Borelverzamelingen
        zijn Lebesgue-verzamelingen. Waar moet men voorzichtig
        zijn over welke $\sigma$-algebra op het doel gebruikt
        wordt?)}
\end{enumerate}

\medskip
\noindent\textbf{Deel IV --- Epiloog.}
\begin{enumerate}[resume]
  \item Sorteer de volgende klassen op strikte inclusie en
        rechtvaardig elke strengheid met een voorbeeld uit dit
        hoofdstuk en zijn probleem: aftelbare verzamelingen;
        Borel-nulverzamelingen; Lebesgue-nulverzamelingen;
        Borelverzamelingen; Lebesgue-verzamelingen;
        willekeurige verzamelingen.
\end{enumerate}

\medskip
\noindent\textbf{Deel V --- De \omterm{def:b3:measure:measure}{Cantor-maat}: massa op een
nulverzameling.} De trap is de verdelingsfunctie van een
opmerkelijke \omterm{def:b3:measure:measure}{maat}, die we nu bouwen met de gereedschappen van
dit hoofdstuk.
\begin{enumerate}[resume]
  \item (Lebesgue--Stieltjes, bestaan) Zij $F\colon\R\to\R$
        niet-dalend, continu, begrensd. Op halfopen intervallen
        definieer $\rho\bigl(\intoc ab\bigr) = F(b) - F(a)$ en,
        voor $A \subseteq \R$,
        \[
        \mu_F^*(A) = \inf\Bigl\{\sum_k\bigl(F(b_k) -
        F(a_k)\bigr) : A \subseteq
        \bigcup_k\intoc{a_k}{b_k}\Bigr\} .
        \]
        Toon dat $\mu_F^*$ een \omterm{def:b3:measure:outer}{uitwendige maat} is en dat
        $\mu_F^*\bigl(\intoc ab\bigr) = F(b) - F(a)$
        \emph{(imiteer het compactheidsargument van
        \cref{thm:b3:measure:lebesgue}, door elk
        $\intoc{a_k}{b_k}$ te vergroten tot een open interval
        tegen $F$-kost $\leq \varepsilon2^{-k}$ --- waar wordt
        continuïteit van $F$ gebruikt?)}.
  \item Toon dat elke Borelverzameling
        $\mu_F^*$-meetbaar is in de zin van Carathéodory
        \emph{(zoals in het Lebesgue-geval volstaat het
        halfrechten te testen; volg het bewijs van de
        toepassing van \cref{thm:b3:measure:caratheodory})},
        zodat $\mu_F = \mu_F^*$ gerestricteerd tot $\mathcal
        B(\R)$ een \omterm{def:b3:measure:measure}{maat} is met $\mu_F(\intoc ab) = F(b) -
        F(a)$: de \emph{Lebesgue--Stieltjes-maat} van $F$.
  \item Pas dit toe op de trap ($F = c$ uitgebreid met $0$ op
        $\R_-$ en $1$ op $\intco1\infty$): de
        \emph{\omterm{def:b3:measure:measure}{Cantor-maat}} $\mu$. Toon $\mu(\R) = 1$, dat
        elke spleet van de Cantorverzameling $\mu$-nul is ($c$
        is daar constant), en besluit
        \[
        \mu(C) = 1, \qquad \lambda(C) = 0 :
        \]
        $\mu$ en $\lambda$ leven op disjuncte dragers ($C$ en
        haar complement). Twee maten in deze positie heten
        \emph{onderling singulier}, genoteerd $\mu \perp
        \lambda$.
  \item Toon dat $\mu$ geen atomen heeft: $\mu(\{x\}) = 0$ voor
        elke $x$ \emph{(continuïteit van $c$)}. Een
        atoomloze kansmaat gedragen door een Lebesgue-nul
        compactum: vergelijk met de enige maten tot nu toe
        gezien.
  \item (Muntworp in vermomming) Voor een woord $(\varepsilon_1,
        \dots, \varepsilon_m) \in \{0,1\}^m$, zij
        $C_{\varepsilon}$ de verzameling van $x \in C$ waarvan
        de ternaire cijfers $b_i(x) = \varepsilon_i$ voor $i
        \leq m$ vervullen (een van de $2^m$ Cantorstukken van
        diepte $m$). Toon $\mu(C_\varepsilon) = 2^{-m}$
        \emph{(de trap klimt $2^{-m}$ over dat stuk: gebruik
        Deel I, vraag 3)}. De \omterm{def:b3:measure:measure}{Cantor-maat} is de wet van een
        oneindige rij eerlijke muntworpen gelezen in ternair
        --- \cref{ch:b3:probability} zal dit exact maken.
  \item Bewijs de zelfgelijkheid: voor elke Borel $A$,
        \[
        \mu(A) = \tfrac12\,\mu(3A) +
        \tfrac12\,\mu(3A - 2) ,
        \]
        waar $3A - 2 = \{3x - 2 : x \in A\}$ \emph{(controleer
        op de voortbrengende intervallen $\intoc ab$ via de
        zelfgelijke relaties van $c$, en roep dan uniciteit
        aan, \cref{thm:b3:measure:uniqueness})}.
  \item Toon dat de reflectie $s(x) = 1 - x$ $\mu$ bewaart:
        $\mu(s(A)) = \mu(A)$ \emph{(via $c(1 - x) = 1 -
        c(x)$, wat volgt uit de symmetrie van de constructie
        --- bewijs het)}.
  \item Bereken de eerste twee momenten van $\mu$, d.w.z.\ van
        een willekeurig punt $X$ met wet $\mu$ (de integralen
        mogen als limieten van sommen over de diepte-$m$-stukken
        behandeld worden, anticiperend op
        \cref{ch:b3:lebesgue}): symmetrie geeft $\int
        x\,\dd\mu = \frac12$, en zelfgelijkheid geeft
        \[
        \int x^2\,\dd\mu = \frac38,
        \qquad\text{dus}\qquad
        \operatorname{Var}(X) = \frac18 .
        \]
        Vergelijk met de uniforme wet op $\intcc01$
        (variantie $\frac1{12}$): de Cantor-massa, naar de
        randen geduwd, spreidt \emph{meer}.
  \item Toon dat de (topologische) \emph{drager} van $\mu$ ---
        de kleinste gesloten verzameling van volle \omterm{def:b3:measure:measure}{maat} ---
        precies $C$ is.
  \item (Synthese) De trap $c$ is continu en niet-dalend maar
        faalt de fundamentele stelling van de calculus (Deel
        I); de \omterm{def:b3:measure:measure}{maat} $\mu_c$ is een kans, atoomloos, singulier
        t.o.v.\ $\lambda$. Leg in een korte alinea uit hoe dit
        twee gezichten van één fenomeen zijn, en formuleer de
        algemene moraal: niet-dalende functies corresponderen
        met maten ($F \leftrightarrow \mu_F$),
        differentieerbaarheid b.o.\ correspondeert met het
        ``absoluut continue deel'', en $c$ is de standaardgetuige
        dat een continue $F$ \emph{geen} absoluut continu deel
        kan dragen.
  \item (De scherpe continuïteitsmodulus) Zij $s =
        \frac{\ln 2}{\ln 3}$. Toon dat $c$ Hölder-continu is
        van exponent $s$:
        \[
        \abs{c(x) - c(y)} \leq 4\,\abs{x - y}^{s}
        \qquad (x, y \in \intcc01),
        \]
        en dat geen exponent $t > s$ werkt, zelfs lokaal.
        Leid de maattheoretische vorm af: voor elke $x$ en
        elke $r \in \intoc01$,
        \[
        \mu\bigl(\intcc{x - r}{x + r}\bigr) \leq 8\,r^{s} .
        \]
        \emph{(Vergelijk een diepte-$m$ triadisch rooster met
        de schaal van $\abs{x - y}$; vraag 18 geeft de klim
        over elk stuk. De exponent $s$ is de Hausdorff-dimensie
        van $C$, zoals latere cursussen zullen zeggen.)}
  \item (Zelfgelijkheid karakteriseert $\mu$) Bewijs het
        omgekeerde van vraag 19: als $\nu$ een kansmaat is op
        $\mathcal B(\R)$ gedragen door $\intcc01$ en
        voldoet aan
        \[
        \nu(A) = \tfrac12\,\nu(3A) + \tfrac12\,\nu(3A - 2)
        \qquad (A \in \mathcal B(\R)),
        \]
        dan $\nu = \mu$. \emph{(Itereer de relatie $m$ maal om
        $\nu$ over de $2^m$ diepte-$m$ Cantorstukken te
        spreiden, schat $\nu(\intoc ab)$ af tegen het aantal
        stukken binnen $\intoc ab$, en laat $m \to \infty$;
        eindig met \cref{thm:b3:measure:uniqueness}.)}
\end{enumerate}
\end{problem}
