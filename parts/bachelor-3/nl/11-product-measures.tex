\chapter{Productmaten, Fubini, Variabelensubstitutie}
\label{ch:b3:product}

Eéndimensionale Lebesgue-theorie wordt meerdimensionale
calculus via twee stellingen. \emph{Tonelli--Fubini} zegt dat
integralen over producten geïtereerde integralen zijn ---
snijden is legitiem, in elke volgorde, onder hypotheses die men
werkelijk kan controleren. De \emph{variabelensubstitutiformule}
transporteert integralen langs $\mathcal C^1$-diffeomorfismen,
met de Jacobiaandeterminant als wisselkoers voor volume; we
bewijzen haar \omterm{def:b3:complete:complete}{volledig}, startend van het lineaire geval waar
zij uitlegt wat de determinant \emph{is}. Toepassingen
cascaderen: de laagjescakeformule, convolutie, \omterm{ex:b3:product:polar}{poolcoördinaten},
het volume van de $n$-bal --- en, in het weekendprobleem,
Stirlings formule met een eerlijke foutanalyse.

\section{Product-$\sigma$-algebra's en productmaten}

\begin{definition}\label{def:b3:product:sigma}
Voor meetbare ruimten $(X, \mathcal A)$, $(Y, \mathcal B)$ is
de \emph{product-$\sigma$-algebra}\index{product-$\sigma$-algebra}
$\mathcal A \otimes \mathcal B$ op $X \times Y$ voortgebracht
door de \emph{rechthoeken} $A \times B$ ($A \in \mathcal A$,
$B \in \mathcal B$) --- een $\pi$-systeem. Voor $E \subseteq
X\times Y$ en $x \in X$ is de \emph{doorsnede} $E_x = \{y :
(x,y) \in E\}$; voor een functie $f$ op het product, $f_x =
f(x, \cdot)$.
\end{definition}

\begin{proposition}\label{prop:b3:product:sections}
(a) Als $E \in \mathcal A\otimes\mathcal B$, dan $E_x \in
\mathcal B$ voor elke $x$ (en symmetrisch); als $f$ $\mathcal
A\otimes\mathcal B$-meetbaar is, is elke $f_x$ $\mathcal
B$-meetbaar.
(b) $\mathcal B(\R^m)\otimes\mathcal B(\R^n) = \mathcal
B(\R^{m+n})$.
\end{proposition}

\begin{proof}
(a) Goede verzamelingen: $\{E : E_x \in \mathcal B\ \forall
x\}$ is een $\sigma$-algebra (doorsnedes commuteren met
complementen en aftelbare unies) die de rechthoeken bevat. Voor
$f$: $(f_x)^{-1}(B) = (f^{-1}(B))_x$.
(b) ($\subseteq$) Rechthoeken van Borelverzamelingen: het
volstaat dat open$\times$open dozen Borel in $\R^{m+n}$ zijn
(zij zijn open) en dat algemene Borel-rechthoeken limieten zijn
--- goede verzamelingen opnieuw: $\{A : A\times\R^n \in
\mathcal B(\R^{m+n})\}$ is een $\sigma$-algebra die de opens
bevat; snijd twee zulke. ($\supseteq$) Elke \omterm{def:b3:topology:topology}{open verzameling}
van $\R^{m+n}$ is een aftelbare unie van rationale open dozen
$U\times V$: bevat in de product-$\sigma$-algebra.
\end{proof}

\begin{theorem}[Productmaat]\label{thm:b3:product:existence}
Zijn $(X, \mathcal A, \mu)$ en $(Y, \mathcal B, \nu)$
\emph{$\sigma$-eindig}. Voor elke $E \in \mathcal
A\otimes\mathcal B$ is de functie $x \mapsto \nu(E_x)$
\omterm{def:b3:lebesgue:measurable}{meetbaar}, en\index{productmaat}
\[
(\mu\otimes\nu)(E) = \int_X \nu(E_x)\,\dd\mu(x)
\]
definieert de unieke \omterm{def:b3:measure:measure}{maat} op $\mathcal A\otimes\mathcal B$
met $(\mu\otimes\nu)(A\times B) = \mu(A)\nu(B)$. Zij is
$\sigma$-eindig, en symmetrisch: dezelfde \omterm{def:b3:measure:measure}{maat} wordt verkregen
door $x$-doorsnedes tegen $\nu$ te integreren.
\end{theorem}

\begin{proof}
\emph{Meetbaarheid van $x \mapsto \nu(E_x)$.} Zij eerst $\nu$
eindig. De klasse $\mathcal D$ van $E$ waarvoor de afbeelding
\omterm{def:b3:lebesgue:measurable}{meetbaar} is bevat de rechthoeken ($\nu((A\times B)_x) =
\nu(B)\mathbf 1_A(x)$) en is een $\lambda$-systeem: voor $E
\subseteq F$ in $\mathcal D$, $\nu((F\setminus E)_x) =
\nu(F_x) - \nu(E_x)$ (eindigheid); voor $E_n \uparrow E$,
$\nu((E_n)_x) \uparrow \nu(E_x)$ (continuïteit van onderen),
en monotone limieten van \omterm{def:b3:lebesgue:measurable}{meetbare functies} zijn \omterm{def:b3:lebesgue:measurable}{meetbaar}.
Rechthoeken vormen een $\pi$-systeem: Dynkin
(\cref{thm:b3:measure:dynkin}) geeft $\mathcal D = \mathcal
A\otimes\mathcal B$. Als $\nu$ $\sigma$-eindig is, schrijf $Y
= \bigcup Y_k$, $Y_k \uparrow$, $\nu(Y_k) < \infty$:
$\nu(E_x) = \lim_k\nu_k(E_x)$ met $\nu_k = \nu(\cdot\cap
Y_k)$ eindig.

\emph{\omterm{def:b3:measure:measure}{Maat}.} $\sigma$-additiviteit van $E \mapsto
\int\nu(E_x)\dd\mu$ volgt uit
\cref{cor:b3:lebesgue:additivity} (doorsnedes van disjuncte
verzamelingen zijn disjunct). Op rechthoeken geeft zij
$\mu(A)\nu(B)$. \emph{Uniciteit}: twee kandidaten stemmen
overeen op het $\pi$-systeem van rechthoeken;
$\sigma$-eindigheid levert rechthoeken $X_k\times Y_k \uparrow
X\times Y$ van eindige \omterm{def:b3:measure:measure}{maat}:
\cref{thm:b3:measure:uniqueness}. Symmetrie: de
andere-volgorde-constructie is ook een \omterm{def:b3:measure:measure}{maat} die op rechthoeken
overeenstemt --- uniek, dus dezelfde.
\end{proof}

\begin{definition}\label{def:b3:product:lebesgue}
\emph{Lebesgue-maat op $\R^d$}\index{Lebesgue-maat op $\R^d$}
is $\lambda_d = \lambda\otimes\cdots\otimes\lambda$ ($d$
factoren; associativiteit van de constructie wordt op dozen
gecontroleerd en door uniciteit voortgeplant). Zij is de unieke
\omterm{def:b3:measure:measure}{Borel-maat} die aan elke doos $\prod\intoc{a_i}{b_i}$ haar
volume $\prod(b_i - a_i)$ toekent; zij is
translatie-invariant (translaten stemmen overeen op dozen),
$\sigma$-eindig, en \omterm{def:b3:complete:complete}{volledig} na Carathéodory-voltooiing --- we
schrijven $\lambda_d$ voor de voltooide \omterm{def:b3:measure:measure}{maat} en integreren
dienovereenkomstig.
\end{definition}

\section{Tonelli en Fubini}

\begin{theorem}[Tonelli]\label{thm:b3:product:tonelli}
$\mu, \nu$ $\sigma$-eindig, $f \colon X\times Y \to [0,
+\infty]$ \omterm{def:b3:lebesgue:measurable}{meetbaar}. Dan is $x \mapsto \int_Y f_x\,\dd\nu$
\omterm{def:b3:lebesgue:measurable}{meetbaar} en\index{stelling van Tonelli}
\[
\int_{X\times Y}f\,\dd(\mu\otimes\nu)
= \int_X\Bigl(\int_Y f(x,y)\,\dd\nu(y)\Bigr)\dd\mu(x)
= \int_Y\Bigl(\int_X f(x,y)\,\dd\mu(x)\Bigr)\dd\nu(y).
\]
\end{theorem}

\begin{proof}
De standaardmachine. Voor $f = \mathbf 1_E$ is dit
\cref{thm:b3:product:existence} (en haar symmetrische vorm).
Door lineariteit geldt het voor eenvoudige $f \geq 0$. Voor
algemene $f \geq 0$: neem eenvoudige $s_n \nearrow f$
(\cref{thm:b3:lebesgue:approximation}); dan $\int_Y(s_n)_x
\dd\nu \nearrow \int_Y f_x\dd\nu$ voor elke $x$ (MCT in $Y$),
dus de linkerleden convergeren door MCT in $X$, terwijl $\int
s_n\,\dd(\mu\otimes\nu) \nearrow \int f$ door MCT op het
product.
\end{proof}

\begin{theorem}[Fubini]\label{thm:b3:product:fubini}
$\mu, \nu$ $\sigma$-eindig, $f \in L^1(\mu\otimes\nu)$. Dan
is voor $\mu$-b.o.\ $x$ de doorsnede $f_x$ $\nu$-integreerbaar,
de b.o.-gedefinieerde functie $x \mapsto \int f_x\dd\nu$ is
\omterm{def:b3:lebesgue:l1}{integreerbaar}, en de twee geïtereerde integralen zijn beide
gelijk aan $\int f\,\dd(\mu\otimes\nu)$.\index{stelling van
Fubini}
\end{theorem}

\begin{proof}
Tonelli toegepast op $\abs f$ toont dat $\varphi(x) =
\int\abs{f_x} \dd\nu$ eindige integraal heeft, dus eindig
b.o.: $f_x \in L^1(\nu)$ voor b.o.\ $x$. Split $f = f^+ -
f^-$ (reëel geval; complex per componenten): Tonelli berekent
elke geïtereerde integraal van $f^\pm$ als $\int
f^\pm\dd(\mu\otimes\nu) < \infty$, en het b.o.-gedefinieerde
verschil integreert tot het verschil. Symmetrisch voor de
andere volgorde.
\end{proof}

\begin{method}\label{met:b3:product:fubini}
Om twee integralen te verwisselen (of een integraal en een
som, of twee sommen): als de integrand \emph{niet-negatief}
is, verwissel vrij (Tonelli). Anders, pas eerst Tonelli toe op
$\abs f$ in de volgorde die makkelijker te schatten is; als het
resultaat eindig is, legitimeert Fubini de verwisseling. Sla
nooit de $\abs f$-controle over: de integrand van
\cref{exo:b3:product:4} heeft twee geïtereerde integralen met
\emph{verschillende waarden}.
\end{method}

\begin{proposition}[Laagjescake]\label{prop:b3:product:layercake}
Voor $f \geq 0$ \omterm{def:b3:lebesgue:measurable}{meetbaar} op $(X, \mathcal A, \mu)$
$\sigma$-eindig:\index{laagjescakeformule}
\[
\int_X f\,\dd\mu = \int_0^{+\infty}\mu(\{f > t\})\,\dd t,
\qquad
\int_X f^p\,\dd\mu = p\int_0^{+\infty}t^{p-1}\mu(\{f >
t\})\,\dd t \quad (p \geq 1).
\]
\end{proposition}

\begin{proof}
Pas Tonelli toe op $\mathbf 1_{\{(x,t) : 0 < t < f(x)\}}$ op $X
\times \intoo0{+\infty}$ (\omterm{def:b3:lebesgue:measurable}{meetbaar}: het is $\{(x,t): f(x) - t
> 0\}\cap\{t > 0\}$, een Borel-type combinatie van de meetbare
$(x,t)\mapsto f(x) - t$): integreren in $t$ eerst geeft $\int
f\,\dd\mu$; in $x$ eerst, $\int_0^\infty\mu(f > t)\dd t$. Voor
$f^p$: substitueer $t = s^p$ in $\int\mu(f^p > t) \dd t$, d.w.z.\
pas de eerste formule toe op $f^p$ en verander variabelen in
de ééndimensionale integraal ($\{f^p > s^p\} = \{f > s\}$).
\end{proof}

\begin{theorem}[Convolutie op $L^1$]\label{thm:b3:product:convolution}
Voor $f, g \in L^1(\R^d, \lambda_d)$ convergeert de integraal
\[
(f * g)(x) = \int_{\R^d} f(x - y)\,g(y)\,\dd y
\]
absoluut voor b.o.\ $x$, definieert $f * g \in L^1(\R^d)$ met
$\norm{f*g}_1 \leq \norm f_1\norm g_1$, en $*$ is commutatief
en associatief.\index{convolutie}
\end{theorem}

\begin{proof}
$(x, y) \mapsto f(x-y)g(y)$ is \omterm{def:b3:lebesgue:measurable}{meetbaar} ($(x,y)\mapsto x - y$
is continu; composeer en vermenigvuldig). Tonelli:
\[
\int\!\!\int \abs{f(x-y)}\abs{g(y)}\,\dd y\,\dd x
= \int\abs{g(y)}\Bigl(\int\abs{f(x - y)}\dd x\Bigr)\dd y
= \norm f_1\norm g_1 < \infty
\]
(translatie-invariantie van $\lambda_d$ in de inwendige
integraal). Dus is de dubbelintegraal eindig; Fubini geeft
b.o.\ absolute convergentie en de normgrens $\norm{f*g}_1 \leq
\norm f_1\norm g_1$. Commutativiteit: substitueer $y \mapsto
x - y$ (translatie- en reflectie-invariantie ---
reflectie-invariantie geldt op dozen, dus overal door
uniciteit). Associativiteit: Tonelli--Fubini op een
drievoudige integraal.
\end{proof}

\section{Variabelensubstitutie}

\begin{theorem}[Lineaire variabelensubstitutie]
\label{thm:b3:product:linearchange}
Voor $T \in GL_d(\R)$ en $A \in \mathcal B(\R^d)$:
$\lambda_d(T(A)) = \abs{\det T}\,\lambda_d(A)$; bijgevolg
$\int f(y)\dd y = \abs{\det T}\int f(Tx)\,\dd x$ voor $f \geq
0$ of \omterm{def:b3:lebesgue:l1}{integreerbaar}.\index{lineaire variabelensubstitutie}
\end{theorem}

\begin{proof}
De \omterm{def:b3:measure:measure}{maat} $\mu_T(A) = \lambda_d(T(A))$ is een \omterm{def:b3:measure:measure}{Borel-maat}
(\omterm{def:b3:topology:continuity}{homeomorfismen} bewaren Borelverzamelingen,
\cref{pb:b3:measure:1}), translatie-invariant ($T(A + x) =
T(A) + Tx$), eindig op de eenheidsdoos: door de
karakterisering van de \omterm{def:b3:measure:lebesgueouter}{Lebesgue-maat}
(\cref{exo:b3:measure:6}, waarvan het bewijs letterlijk in
$\R^d$ met dyadische kubussen werkt), $\mu_T = c(T)\lambda_d$
met $c(T) = \lambda_d(T(\intco01^d))$. De afbeelding $T
\mapsto c(T)$ is multiplicatief ($c(ST) = c(S)c(T)$, door
compositie), dus volstaat het $c$ te berekenen op voortbrengers
van $GL_d$: elementaire matrices. Diagonaal
$\operatorname{diag}(a, 1, \dots, 1)$: beeldt de eenheidskubus
af op een doos van volume $\abs a$: $c = \abs a = \abs\det$.
Transpositie van coördinaten: permuteert de kubus: $c = 1 =
\abs\det$. Transvectie $T(x) = x + \alpha x_2e_1$: het beeld
van de eenheidskubus is een geschoren prisma; door Tonelli is
haar \omterm{def:b3:measure:measure}{maat} $\int\lambda_1(\text{doorsnede})\dd x_2 \cdots \dd
x_d = 1$, elke $x_1$-doorsnede een interval van lengte $1$: $c
= 1 = \abs{\det}$. Elke inverteerbare matrix is een product van
deze (Gauss-eliminatie), en zowel $c$ als $\abs\det$ zijn
multiplicatief: $c(T) = \abs{\det T}$. De integraalformule
volgt door de standaardmachine (indicatoren, \omterm{def:b3:groups:simple}{eenvoudig}, MCT).
\end{proof}

\begin{theorem}[Variabelensubstitutie]\label{thm:b3:product:changeofvar}
Zijn $U, V \subseteq \R^d$ open en $\Phi \colon U \to V$ een
$\mathcal C^1$-diffeomorfisme. Voor elke meetbare $f \colon V
\to [0, +\infty]$ (of $f \in L^1(V)$):\index{variabelensubstitutiformule}
\[
\int_V f(y)\,\dd y = \int_U
f\bigl(\Phi(x)\bigr)\,\abs{\det D\Phi(x)}\,\dd x .
\]
\end{theorem}

\begin{proof}
Schrijf $J(x) = \abs{\det D\Phi(x)}$. Het hart van het bewijs
is de ongelijkheid
\[
\lambda_d\bigl(\Phi(A)\bigr) \leq \int_A J\,\dd\lambda_d
\qquad\text{voor elke Borel } A \subseteq U;
\tag{$*$}
\]
Stap 4 hieronder upgradeert $(*)$ --- toegepast op zowel
$\Phi$ als $\Phi^{-1}$ --- tot de gelijkheid van de stelling.
Noteer dat $\Phi^{-1}$ zelf een $\mathcal C^1$-diffeomorfisme
is met Jacobiaan $\abs{\det D\Phi^{-1}(y)} =
J(\Phi^{-1}y)^{-1}$ (kettingregel op $\Phi\circ\Phi^{-1} =
\mathrm{id}$).

\emph{Stap 1: $(*)$ voor kubussen met een
vervormingsfactor.} Fixeer een gesloten kubus $Q \subseteq U$
van \omterm{ex:b3:groups:actions}{centrum} $x_0$ en zijde $2r$ (sup-norm-bal). Claim: voor
elke $\varepsilon > 0$, als $\Phi$ differentieerbaar is op $Q$
met $\norm{D\Phi(x) - D\Phi(x_0)} \leq \varepsilon$ op $Q$
(\omterm{def:b3:banach:operator}{operatornorm} voor de sup-norm), dan
\[
\Phi(Q) \subseteq \Phi(x_0) + D\Phi(x_0)\Bigl(\,\bigl(1 +
\varepsilon\norm{D\Phi(x_0)^{-1}}\bigr)\,(Q - x_0)\Bigr),
\]
want voor $x \in Q$ geeft de middelwaarde-ongelijkheid
toegepast op $\Phi(x) - \Phi(x_0) - D\Phi(x_0)(x - x_0)$ dat
$\norm{\Phi(x) - \Phi(x_0) - D\Phi(x_0)(x-x_0)}_\infty \leq
\varepsilon\norm{x - x_0}_\infty \leq \varepsilon r$, en
$D\Phi(x_0)^{-1}$ trekt dit defect in een
$\varepsilon\norm{D\Phi(x_0)^{-1}}\,r$-vergroting van de
kubus. Door \cref{thm:b3:product:linearchange},
\[
\lambda_d(\Phi(Q)) \leq \abs{\det D\Phi(x_0)}\,
\bigl(1 + \varepsilon\,C\bigr)^{d}\,\lambda_d(Q),
\qquad C = \sup_{Q}\norm{D\Phi(\cdot)^{-1}} .
\]

\emph{Stap 2: $(*)$ voor compacte kubussen, door
onderverdeling.} Zij $Q \subseteq U$ een compacte kubus en
$\varepsilon > 0$. Op $Q$ is $D\Phi$ uniform continu en
$\norm{D\Phi^{-1}}$ begrensd (compactheid); onderverdeel $Q$
in $2^{kd}$ subkubussen $Q_i$ klein genoeg dat de oscillatie
van $D\Phi$ op elk $\leq \varepsilon$ is. Stap 1 op elke
subkubus (\omterm{ex:b3:groups:actions}{centrum} $x_i$):
\[
\lambda_d(\Phi(Q)) \leq \sum_i\lambda_d(\Phi(Q_i))
\leq (1 + C\varepsilon)^d \sum_i \abs{\det
D\Phi(x_i)}\,\lambda_d(Q_i)
\leq (1 + C\varepsilon)^d\Bigl(\int_Q J + \varepsilon'\Bigr),
\]
de laatste stap omdat $\sum_i\abs{\det D\Phi(x_i)}\mathbf
1_{Q_i} \to J$ uniform op $Q$ (continuïteit van $\det D\Phi$)
--- Riemann-somvergelijking. Laat $\varepsilon \to 0$: $(*)$
geldt voor compacte kubussen.

\emph{Stap 3: $(*)$ voor alle Borel $A$.} De
verzamelingsfunctie $A \mapsto \lambda_d(\Phi(A))$, op
Boreldeelverzamelingen van $U$, is een \omterm{def:b3:measure:measure}{maat} ($\Phi$ is een
bijectie op $V$ die Borelverzamelingen en aftelbare
disjunctheid bewaart), en zo ook $A \mapsto \int_AJ$. Elke
open deelverzameling van $U$ is een aftelbare unie van bijna-
disjuncte dyadische compacte kubussen (standaard dyadische
ontleding: neem maximale dyadische kubussen bevat in de \omterm{def:b3:topology:topology}{open
verzameling}), en beide maten zijn additief over hen (randen
van kubussen zijn $\lambda_d$-nul, en hun $\Phi$-beelden zijn
nul door Stap 2 toegepast op dunne kubusoverdekkingen van de
vlakken): $(*)$ gaat van kubussen naar \omterm{def:b3:topology:topology}{open verzamelingen}.
Algemene Borel $A$: put $U$ uit door compacta $K_m \uparrow U$
met $K_m \subseteq \mathring K_{m+1}$, en fixeer $m$; op
$\mathring K_{m+1}$ is $J$ begrensd door enige $M_m$. Door
uitwendige regulariteit van $\lambda_d$ (bewijs als in
\cref{thm:b3:measure:regularity}, met dozen), kies \omterm{def:b3:topology:topology}{open
verzamelingen} $O_n$ met $A \cap K_m \subseteq O_n \subseteq
\mathring K_{m+1}$ en $\lambda_d\bigl(O_n \setminus (A\cap
K_m)\bigr) \to 0$. Dan
\[
\lambda_d\bigl(\Phi(A\cap K_m)\bigr) \leq
\lambda_d\bigl(\Phi(O_n)\bigr) \leq \int_{O_n}J
\leq \int_{A\cap K_m}J + M_m\,\lambda_d\bigl(O_n\setminus(A\cap
K_m)\bigr) \xrightarrow[n\to\infty]{} \int_{A\cap K_m}J .
\]
Laat $m \to \infty$: continuïteit van onderen links, MCT
rechts. Dit vestigt $(*)$.

\emph{Stap 4: gelijkheid en de integraalformule.} Breid eerst
$(*)$ uit van verzamelingen naar integralen: voor elke meetbare
$g \geq 0$ op $V$,
\[
\int_V g(y)\,\dd y \leq \int_U g(\Phi(x))\,J(x)\,\dd x .
\tag{$**$}
\]
Inderdaad, voor $g = \mathbf 1_B$ is dit $(*)$ met $A =
\Phi^{-1}(B)$; lineariteit breidt het uit tot eenvoudige $g$,
en MCT tot alle $g \geq 0$ (de standaardmachine). Pas nu
$(**)$ tweemaal toe: eerst op $g$, dan --- voor het
diffeomorfisme $\Phi^{-1}$ --- op de functie $x \mapsto
g(\Phi(x))J(x)$:
\[
\int_Vg \leq \int_U g(\Phi(x))J(x)\dd x
\leq \int_V g(y)\,J(\Phi^{-1}y)\,\abs{\det
D\Phi^{-1}(y)}\,\dd y = \int_V g ,
\]
want $J(\Phi^{-1}y)\abs{\det D\Phi^{-1}(y)} = \abs{\det\bigl(
D\Phi(\Phi^{-1}y)\,D\Phi^{-1}(y)\bigr)} = 1$ (kettingregel op
$\Phi\circ\Phi^{-1} = \mathrm{id}$). Alle ongelijkheden zijn
gelijkheden: de formule geldt voor $g \geq 0$, en voor
$L^1$-functies door ontleding.
\end{proof}

\begin{example}[Poolcoördinaten; de Gaussiaan opnieuw]
\label{ex:b3:product:polar}
$\Phi(r, \theta) = (r\cos\theta, r\sin\theta)$ is een $\mathcal
C^1$-diffeomorfisme van $\intoo0{+\infty}\times\intoo0{2\pi}$
op $\R^2$ minus een halfrechte (nulverzameling), met $\det
D\Phi = r$:\index{poolcoördinaten}
\[
\int_{\R^2}f(x, y)\,\dd x\,\dd y
= \int_0^{2\pi}\!\!\int_0^{+\infty}
f(r\cos\theta, r\sin\theta)\,r\,\dd r\,\dd\theta .
\]
Voor $f = \eu^{-x^2-y^2}$ geven Tonelli en deze formule
\[
G^2 = \Bigl(\int_\R \eu^{-x^2}\dd x\Bigr)^2
= \int_{\R^2}\eu^{-x^2-y^2}
= 2\pi\int_0^\infty r\eu^{-r^2}\dd r = \pi:
\]
het klassieke twee-regels-bewijs van $G = \sqrt\pi$, nu
\omterm{def:b3:complete:complete}{volledig} gerechtvaardigd (vergelijk het parameterbewijs van
\cref{pb:b3:lebesgue:1}).
\end{example}

\begin{theorem}[Volume van de eenheidsbal]
\label{thm:b3:product:ballvolume}
Zij $v_d = \lambda_d(B(0,1))$ in $\R^d$. Dan\index{volume van
de eenheidsbal}
\[
v_d = \frac{\pi^{d/2}}{\Gamma\bigl(\frac d2 + 1\bigr)} :
\qquad
v_1 = 2,\quad v_2 = \pi,\quad v_3 = \tfrac{4\pi}3,\quad
v_4 = \tfrac{\pi^2}2,\ \dots
\]
\end{theorem}

\begin{proof}
Bereken $I = \int_{\R^d}\eu^{-\norm x_2^2}\dd\lambda_d$
tweemaal. Door Tonelli factoriseert zij: $I = G^d =
\pi^{d/2}$. Door de laagjescakeformule
(\cref{prop:b3:product:layercake}) met $f = \eu^{-
\norm x^2}$, waarvan de niveaus ballen zijn: $\{f > t\} =
B\bigl(0, \sqrt{-\ln t}\bigr)$ voor $0 < t < 1$, van \omterm{def:b3:measure:measure}{maat}
$v_d(-\ln t)^{d/2}$ (dilatie met $\rho$ schaalt $\lambda_d$
met $\rho^d$: \cref{thm:b3:product:linearchange}), dus
\[
I = \int_0^1 v_d\,(-\ln t)^{d/2}\,\dd t
\overset{t = \eu^{-s}}{=}
v_d\int_0^\infty s^{d/2}\eu^{-s}\,\dd s
= v_d\,\Gamma\Bigl(\frac d2 + 1\Bigr).
\]
Stel gelijk. (De waarden: $\Gamma(\frac32) =
\frac{\sqrt\pi}2$, $\Gamma(2) = 1$, enz.) Noteer $v_d \to 0$
als $d \to \infty$ --- het weekendprobleem kwantificeert hoe
snel, via Stirling.
\end{proof}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:b3:product:1}
Zij $\mu$ de telmaat op $(\intcc01, \mathcal
B(\intcc01))$ (niet $\sigma$-eindig) en $\lambda$ de
\omterm{def:b3:measure:lebesgueouter}{Lebesgue-maat}, en zij $\Delta = \{(x,x)\}$ de diagonaal in
$\intcc01^2$. Toon dat $\Delta$ \omterm{def:b3:lebesgue:measurable}{meetbaar} is, en bereken de
twee geïtereerde integralen van $\mathbf 1_\Delta$ tegen
$\lambda$ en $\mu$: zij verschillen. Welke hypothese van
\cref{thm:b3:product:tonelli} faalt?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b3:product:2}
Rechtvaardig de verwisseling en herleid Dirichlets integraal:
voor $A > 0$,
\[
\int_0^A\frac{\sin x}x\,\dd x
= \int_0^A\!\!\int_0^{+\infty}\eu^{-xy}\sin x\,\dd y\,\dd x
= \int_0^{+\infty}\!\!\int_0^A \eu^{-xy}\sin x\,\dd x\,\dd y,
\]
bereken de inwendige integraal in gesloten vorm, en laat $A
\to +\infty$ (domineer de $y$-integraal) om
$\int_0^\infty\frac{\sin x}x\dd x = \frac\pi2$ te krijgen.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:product:3}
(a) Bewijs dat voor $f \geq 0$ \omterm{def:b3:lebesgue:measurable}{meetbaar} en $\mu$ eindig:
$\sum_{n\geq1}\mu(\{f \geq n\}) \leq \int f\,\dd\mu \leq
\mu(X) + \sum_{n\geq1}\mu(\{f\geq n\})$: integreerbaarheid is
sommeerbaarheid van de staartmaten.
(b) Leid af dat $f \in L^1(\mu)$ ($\mu$ eindig) d.e.s.d.a.\
$\sum_n\mu(\abs f \geq n) < \infty$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:product:4}
Voor $f(x, y) = \dfrac{x^2 - y^2}{(x^2 + y^2)^2}$ op
$\intoo01^2$, toon
\[
\int_0^1\!\!\int_0^1 f\,\dd y\,\dd x = \frac\pi4,
\qquad
\int_0^1\!\!\int_0^1 f\,\dd x\,\dd y = -\frac\pi4
\]
\emph{(noteer $f = \partial_y\bigl(\frac{y}{x^2+y^2}\bigr)$)},
en verifieer direct dat $\int\!\int\abs f = +\infty$: Fubini's
integreerbaarheidshypothese is geen decoratie.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:product:5}
(a) Bereken $\mathbf 1_{\intcc01} * \mathbf 1_{\intcc01}$
expliciet (een tentfunctie), en de algemene vorm van $(\mathbf
1 * \mathbf 1 * \mathbf 1)$.
(b) Toon $\operatorname{supp}(f * g) \subseteq
\overline{\operatorname{supp}f + \operatorname{supp}g}$.
(c) Toon dat als $f \in L^1$ en $g$ begrensd en continu is, $f
* g$ continu is. \emph{(DCT via continuïteit van translatie op
de begrensde $g$.)}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:product:6}
(a) Toon dat de simplex $\Delta_d = \{x \in \intco0\infty^d
: x_1 + \dots + x_d \leq 1\}$ volume $\frac1{d!}$ heeft
\emph{(inductie en Fubini)}.
(b) Herwin $v_2 = \pi$, $v_3 = \frac{4\pi}3$ uit
\cref{thm:b3:product:ballvolume}, en toon
$\lambda_d(\text{ellipsoïde met halfassen } a_i) =
v_d\prod a_i$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:product:7}
Voor welke $s > 0$ zijn de volgende eindig? Rechtvaardig met
\omterm{ex:b3:product:polar}{poolcoördinaten}:
\[
\int_{B(0,1)\subseteq\R^2}\frac{\dd x\,\dd y}{(x^2 +
y^2)^{s}},
\qquad
\int_{\R^2\setminus B(0,1)}\frac{\dd x\,\dd y}{(x^2 +
y^2)^{s}} .
\]
Generaliseer tot $\R^d$ (de drempels $s < d/2$ en $s > d/2$).
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b3:product:8}
(Beta--Gamma) Voor $p, q > 0$, zij $B(p, q) = \int_0^1t^{p-1}(1
- t)^{q-1}\dd t$. Startend van $\Gamma(p)\Gamma(q)$ als
dubbelintegraal, substitueer $(x, y) = (uv,\, u(1 - v))$ (een
diffeomorfisme van het open kwadrant op
$\intoo0\infty\times\intoo01$; bereken haar Jacobiaan $= u$) en
besluit\index{Beta-functie}
\[
B(p, q) = \frac{\Gamma(p)\,\Gamma(q)}{\Gamma(p + q)} .
\]
Leid af $\int_0^{\pi/2}\sin^{2p-1}\theta\cos^{2q-1}\theta\,
\dd\theta = \frac12B(p,q)$ en de waarde van de Wallis-
integralen $W_n = \int_0^{\pi/2}\sin^n$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:product:9}
(Transferformule) Zij $T \colon (X, \mathcal A, \mu) \to (Y,
\mathcal B)$ \omterm{def:b3:lebesgue:measurable}{meetbaar} en $T_*\mu(B) = \mu(T^{-1}(B))$ de
\emph{pushforward-maat}\index{pushforward-maat}.
Toon dat voor elke meetbare $g \geq 0$ op $Y$:
\[
\int_Y g\,\dd(T_*\mu) = \int_X g\circ T\,\dd\mu
\]
(standaardmachine). Vergelijk dan met
\cref{thm:b3:product:linearchange}: welke extra informatie
draagt de variabelensubstitutiformule die de abstracte
transferformule niet heeft? \emph{(De transferformule
identificeert $T_*\mu$ nooit; de
variabelensubstitutiestelling berekent $T_*\lambda_d$
expliciet als een dichtheidsmaat.)}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b3:product:10}
(Gaussische momenten) Met \omterm{ex:b3:product:polar}{poolcoördinaten} en Fubini, bereken
voor het standaard Gaussische gewicht op $\R^d$:
\[
\int_{\R^d}\norm x_2^2\;\eu^{-\norm x_2^2}\,\dd x
\qquad\text{en}\qquad
\int_{\R^d}x_1^2\,\eu^{-\norm x^2_2}\,\dd x,
\]
controleer de consistentie ($\norm x^2 = \sum x_i^2$), en leid
het tweede moment van de \omterm{def:b3:measure:measure}{maat}
$\pi^{-d/2}\eu^{-\norm x^2}\dd x$ af.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:product:11}
(Grafiek en hypograaf) Zij $f \colon \R^d \to \intco0\infty$
\omterm{def:b3:lebesgue:measurable}{meetbaar}.
(a) Toon dat de \emph{hypograaf} $H = \{(x, y) \in
\R^d\times\R : 0 < y < f(x)\}$ \omterm{def:b3:lebesgue:measurable}{meetbaar} is in $\R^{d+1}$ met
\[
\lambda_{d+1}(H) = \int_{\R^d}f\,\dd\lambda_d :
\]
``de integraal is de oppervlakte onder de grafiek'', eindelijk
een stelling. \emph{(Doorsnedes; Tonelli.)}
(b) Toon dat de grafiek $\{(x, f(x)) : x \in \R^d\}$ een
nulverzameling van $\R^{d+1}$ is.
(c) Leid een twee-regels-bewijs af dat de sfeer $S^{d-1}$
Lebesgue-nul is in $\R^d$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:product:12}
(Een beroemde dubbelintegraal) Met de meetkundige reeks en
Tonelli op $\intoo01^2$, bewijs
\[
\int_0^1\!\!\int_0^1\frac{\dd x\,\dd y}{1 - xy}
= \sum_{n\geq1}\frac1{n^2} = \zeta(2),
\qquad
\int_0^1\!\!\int_0^1\frac{\dd x\,\dd y}{1 + xy}
= \sum_{n\geq1}\frac{(-1)^{n-1}}{n^2} = \frac{\zeta(2)}2 .
\]
(De tweede reeksidentiteit: splits even en oneven indices.)
Met $\zeta(2) = \frac{\pi^2}6$ (\cref{ch:b3:spectral})
evalueren twee onschuldig ogende integralen tot
$\frac{\pi^2}6$ en $\frac{\pi^2}{12}$; waar precies doet
Tonelli's positiviteitshypothese haar werk?
\end{exercise}

\section{Probleem: Stirlings formule}

\begin{problem}[{Weekendprobleem --- $n! \sim
\sqrt{2\pi n}\,(n/\eu)^n$, via gedomineerde convergentie}]
\label{pb:b3:product:1}
Stirlings formule regelt elke asymptotische telling in dit boek
--- balvolumes, binomiaalcoëfficiënten, de lokale vorm van de
centrale limietstelling. We bewijzen haar uit de
$\Gamma$-integraal (\cref{ex:b3:lebesgue:gamma}) met de
\emph{Laplacemethode}, in haar schoonste
gedomineerde-convergentievorm, en verzamelen dan
dividenden.\index{formule van Stirling}

\medskip
\noindent\textbf{Deel I --- De formule.} Voor $t > 0$,
$\Gamma(t + 1) = \int_0^\infty x^{t}\eu^{-x}\dd x$.
\begin{enumerate}
  \item Substitueer $x = t + \sqrt t\,u$ en toon
        \[
        \frac{\Gamma(t+1)}{t^{t}\eu^{-t}\sqrt t}
        = \int_{-\sqrt t}^{+\infty}
        \exp\Bigl(t\ln\Bigl(1 + \frac u{\sqrt
        t}\Bigr) - \sqrt t\,u\Bigr)\,\dd u
        \;=\;\int_\R g_t(u)\,\dd u,
        \]
        waar $g_t(u) = \exp\bigl(t\ln(1 + u/\sqrt t) -
        \sqrt t\,u\bigr)\mathbf 1_{u > -\sqrt t}$.
  \item Toon de puntsgewijze limiet: voor elke vaste $u$,
        $g_t(u) \to \eu^{-u^2/2}$ als $t \to +\infty$
        \emph{(ontwikkel $\ln(1 + h)$ tot tweede orde)}.
  \item Dominatie. Zij $\varphi(h) = \ln(1 + h) - h$, zodat
        $g_t(u) = \exp\bigl(t\,\varphi(u/\sqrt t)\bigr)$ voor
        $u > -\sqrt t$. Bewijs de twee grenzen
        \[
        \varphi(h) \leq -\frac{h^2}4 \quad (-1 < h \leq 1),
        \qquad
        \varphi(h) \leq -c\,h \quad (h \geq 1),\ \ c = 1 -
        \ln 2 > 0
        \]
        \emph{(bestudeer $\varphi(h) + \frac{h^2}4$ en
        $\varphi(h) + ch$: bereken de afgeleiden en
        controleer het teken op elk bereik)}. Leid af, voor $t
        \geq 1$:
        \[
        g_t(u) \leq \eu^{-u^2/4}\ \ (\abs u \leq \sqrt t),
        \qquad
        g_t(u) \leq \eu^{-cu}\ \ (u \geq \sqrt t),
        \]
        zodat $g_t(u) \leq \eu^{-u^2/4} +
        \eu^{-cu}\,\mathbf 1_{u > 0}$: een integreerbare
        dominant onafhankelijk van $t \geq 1$.
  \item Besluit met de gedomineerde convergentiestelling en de
        Gaussische integraal (\cref{ex:b3:product:polar}):
        \[
        \Gamma(t + 1) \;\sim\;
        \sqrt{2\pi t}\;\Bigl(\frac t\eu\Bigr)^{t}
        \qquad (t \to +\infty),
        \]
        en in het bijzonder $n! \sim \sqrt{2\pi
        n}\,(n/\eu)^n$.
\end{enumerate}

\medskip
\noindent\textbf{Deel II --- Dividenden.}
\begin{enumerate}[resume]
  \item (Wallis) Uit \cref{exo:b3:product:8}, $W_{2n} =
        \frac\pi2\binom{2n}n4^{-n}$-type formules: leid
        $\binom{2n}{n} \sim \frac{4^n}{\sqrt{\pi n}}$ af uit
        Stirling, en controleer tegen de recursie $W_{n} =
        \frac{n-1}nW_{n-2}$.
  \item (Balvolumes storten in) Toon
        \[
        v_d = \frac{\pi^{d/2}}{\Gamma(\frac d2 + 1)}
        \;\sim\; \frac{1}{\sqrt{\pi d}}
        \Bigl(\frac{2\pi\eu}{d}\Bigr)^{d/2},
        \]
        zodat $v_d \to 0$ sneller dan elke meetkundige rij;
        vind de dimensie die $v_d$ maximaliseert (numeriek: $d
        = 5$).
  \item (Concentratie van de binomiaal --- een voorproef van
        \cref{ch:b3:clt}) Met Stirling, toon de lokale
        schatting, voor $k = n/2 + s\sqrt n/2$ met $s$ vast en
        $n$ even:
        \[
        2^{-n}\binom{n}{k} \;\sim\;
        \sqrt{\frac{2}{\pi n}}\;\eu^{-s^2/2},
        \]
        het discrete Gaussische profiel: de Moivre--Laplace in
        embryonale vorm.
  \item Waar precies gebruikte het bewijs van Deel I: (i) MCT
        of DCT; (ii) de Gaussische integraal; (iii) de
        invariantie-eigenschappen van de \omterm{def:b3:measure:lebesgueouter}{Lebesgue-maat}? Eén
        zin elk.
\end{enumerate}

\medskip
\noindent\textbf{Deel III --- De foutterm: Stirling met
staven.} Stel $d_n = \ln n! - \bigl(n + \tfrac12\bigr)\ln n +
n - \ln\sqrt{2\pi}$, zodat Deel I zegt $d_n \to 0$.
\begin{enumerate}[resume]
  \item Toon $d_n - d_{n+1} = \bigl(n +
        \tfrac12\bigr)\ln\bigl(1 + \tfrac1n\bigr) - 1$.
  \item Met $t = \frac1{2n+1}$, verifieer $\frac{n+1}n =
        \frac{1+t}{1-t}$ en ontwikkel:
        \[
        d_n - d_{n+1} = \frac{t^2}3 + \frac{t^4}5 +
        \frac{t^6}7 + \cdots,
        \]
        en leid de tweezijdige grenzen af
        \[
        \frac1{3(2n+1)^2} \;<\; d_n - d_{n+1} \;<\;
        \frac1{12n} - \frac1{12(n+1)} .
        \]
  \item Telescopeer (met $d_m \to 0$) en controleer de
        prettige algebraïsche identiteit
        $\frac1{3(2m+1)^2} > \frac1{12m+1} -
        \frac1{12(m+1)+1}$ voor $m \geq 1$, om de klassieke
        inkadering te verkrijgen
        \[
        \sqrt{2\pi n}\Bigl(\frac n\eu\Bigr)^n
        \eu^{1/(12n+1)} \;<\; n! \;<\;
        \sqrt{2\pi n}\Bigl(\frac n\eu\Bigr)^n\eu^{1/(12n)} .
        \]
  \item Twee gevolgen: (a) de relatieve fout van Stirlings
        formule is $< 10^{-6}$ zodra $n \geq 83\,334$; (b)
        schat $100!$ tot vier significante cijfers met de hand
        uit de inkadering ($100! \approx 9.3326\cdot
        10^{157}$), en verwonder je kort over de precisie van
        een asymptotische formule bij een zeer eindige $n$.
\end{enumerate}

\medskip
\noindent\textbf{Deel IV --- De Wallis-route: Stirling zonder
de Gaussiaan.} Historisch kwam de constante $\sqrt{2\pi}$ van
Wallis, niet van Gauss; dit deel herbewijst Stirling
onafhankelijk van Deel I--II, en herbewijst daarmee de
Gaussische integraal. Zij $W_n = \int_0^{\pi/2}\sin^n\theta\,
\dd\theta$.
\begin{enumerate}[resume]
  \item Vestig $W_n = \frac{n-1}nW_{n-2}$ (partieel
        integreren), de gesloten vormen
        \[
        W_{2n} = \frac\pi2\binom{2n}n4^{-n}, \qquad
        W_{2n+1} = \frac{4^n}{(2n+1)\binom{2n}n},
        \]
        en de identiteit $W_nW_{n-1} = \frac\pi{2n}$.
  \item Uit de monotonie van $(W_n)$ leid af
        $W_{2n}/W_{2n+1} \to 1$, dan
        \[
        W_{2n} \sim \frac12\sqrt{\frac\pi n}
        \qquad\text{en}\qquad
        \binom{2n}n4^{-n}\sqrt n \longrightarrow
        \frac1{\sqrt\pi} :
        \]
        Wallis' stelling, verkregen zonder Stirling.
  \item Toon, door alleen het telescoperen van Deel III (geen
        waarde van de constante nodig), dat $e_n = \ln n! -
        (n + \frac12)\ln n + n$ convergeert naar enige limiet
        $\ell$; equivalent $n! \sim K\,n^{n+1/2}\eu^{-n}$ met
        $K = \eu^\ell > 0$ nog niet geïdentificeerd.
  \item Steek deze asymptotiek in
        $\binom{2n}n4^{-n}\sqrt n$ en identificeer, met vraag
        14, de enige mogelijke waarde: $K = \sqrt{2\pi}$.
        Assembleer de logica: Deel III--IV samen geven een
        \omterm{def:b3:complete:complete}{volledig} tweede bewijs van Stirling --- en dus, Deel
        I's substitutie achterwaarts lopend, een onafhankelijke
        evaluatie van $\int_\R\eu^{-u^2/2}\dd u =
        \sqrt{2\pi}$. Twee pijlers, die elkaar elk ondersteunen.
\end{enumerate}

\medskip
\noindent\textbf{Deel V --- Laatste dividenden.}
\begin{enumerate}[resume]
  \item (Het volledige lokale profiel) Voor gehele $\abs j
        \leq K\sqrt n$ ($K$ vast), toon
        \[
        \frac{\binom{2n}{n+j}}{\binom{2n}{n}} =
        \prod_{i=1}^{\abs j}\frac{n - i + 1}{n + i}
        = \exp\Bigl(-\frac{j^2}n +
        O\Bigl(\frac1{\sqrt n}\Bigr)\Bigr),
        \]
        uniform in $j$ \emph{(neem logaritmen en gebruik $\ln
        \frac{1-x}{1+y} = -(x + y) + O(x^2 + y^2)$)}. Dit is
        de tweezijdige versie van vraag 7 en de exacte
        schatting geciteerd in \cref{ch:b3:clt}'s
        weekendprobleem.
  \item (Een Poisson-voorproef) Toon met Stirling dat
        $\eu^{-n}\dfrac{n^n}{n!} \sim \dfrac1{\sqrt{2\pi n}}$:
        de modus van een Poissonwet van grote mean $n$ draagt
        massa $\approx (2\pi n)^{-1/2}$, precies zoals de
        centrale limietstelling zal voorspellen.
  \item (Gamma-ratio's) Voor $a \in \intoo01$, bewijs
        $\dfrac{\Gamma(n + a)}{\Gamma(n)\,n^a} \to 1$ met de
        log-convexiteit-hellinggrenzen van
        \cref{pb:b3:lebesgue:1} (vraag 14 daar), en breid uit
        tot elke reële $a > 0$ door de
        functionaalvergelijking. (Dit is wat ``$\Gamma(t+1)
        \sim$ Stirling'' tussen de gehele getallen betekent.)
  \item (Ballen, bis) Uit $v_d = \pi^{d/2}/\Gamma(\frac d2
        + 1)$: tabuleer $v_1, \dots, v_7$ exact, verifieer
        unimodaliteit via $\frac{v_d}{v_{d-2}} = \frac{2\pi}d$
        (stijgend zolang $d < 2\pi$, dalend daarna), en bewijs
        de treffende genererende identiteit
        \[
        \sum_{k\geq0}v_{2k}\,x^{2k} = \eu^{\pi x^2} :
        \]
        alle even-dimensionale eenheidsbalvolumes verpakt in
        één exponentieel.
  \item (Entropie-asymptotiek) Voor vaste $\alpha \in
        \intoo01$ met $\alpha n \in \N$, leid uit Stirling af
        \[
        \binom{n}{\alpha n} \;\sim\;
        \frac{\eu^{n\,H(\alpha)}}
        {\sqrt{2\pi\,\alpha(1-\alpha)\,n}},
        \qquad
        H(\alpha) = -\alpha\ln\alpha -
        (1-\alpha)\ln(1-\alpha) :
        \]
        de exponentiële groeisnelheid van
        binomiaalcoëfficiënten is de \emph{entropie} $H$ ---
        controleer dat $\alpha = \frac12$ vraag 5 herwint, en
        dat $H(\alpha) < \ln2$ voor $\alpha \neq \frac12$ (dus
        off-center binomialen zijn exponentieel verwaarloosbaar
        in $2^n$).
  \item (Oppervlakten) De oppervlakte van de eenheidssfeer
        $S^{d-1}$ is $s_{d-1} = d\,v_d$ (bewwezen als
        \cref{exo:b3:forms:6} in het differentiaalvormen-
        hoofdstuk; hier, neem het als definitie). Tabuleer
        $s_0, \dots, s_6$, lokaliseer de maximale ($d - 1 =
        6$, $s_6 = \frac{16\pi^3}{15} \approx 33.07$), en toon
        $s_{d-1} \to 0$ super-meetkundig eveneens ---
        hoogdimensionale sferen zijn, naar elke Euclidische
        maatstaf, verdwijnend klein.
  \item (De eerste correctieterm) Leid uit de inkadering van
        vraag 11 af dat $d_n = \frac1{12n} +
        O\bigl(\frac1{n^2}\bigr)$, dus
        \[
        n! = \sqrt{2\pi n}\,\Bigl(\frac
        n\eu\Bigr)^{n}\Bigl(1 + \frac1{12n} +
        O\Bigl(\frac1{n^2}\Bigr)\Bigr).
        \]
        Verifieer bij $n = 10$: de bare formule geeft
        $3\,598\,696$ (relatieve fout $8.3\cdot10^{-3}$), de
        gecorrigeerde $3\,628\,685$ tegen $10! =
        3\,628\,800$ (relatieve fout $3.2\cdot10^{-5}$) ---
        één term van de reeks koopt tweeënhalve cijfers.
  \item (De mediaan van $\Gamma$) Toon dat
        \[
        \frac{1}{\Gamma(t+1)}
        \int_0^{t} x^{t}\eu^{-x}\,\dd x
        \;\longrightarrow\; \frac12
        \qquad (t \to +\infty) :
        \]
        asymptotisch zit precies de helft van de massa van de
        $\Gamma$-integrand onder haar modus $x = t$.
        \emph{(Voer Deel I's substitutie uit op de afgeknotte
        integraal; de dominant van vraag 3 staat al klaar.)}
  \item (Entropie, niet-asymptotisch) Voor $\alpha \in
        \intoc0{\frac12}$ bewijs de grens, geldig voor
        \emph{elke} $n \geq 1$:
        \[
        \sum_{k=0}^{\lfloor\alpha n\rfloor}\binom nk
        \;\leq\; \eu^{n\,H(\alpha)} ,
        \]
        door de som te vergelijken met $\sum_k\binom
        nk\lambda^{k-\alpha n}$ voor de tilt $\lambda =
        \frac{\alpha}{1-\alpha} \leq 1$. Controleer dat deze
        keuze van $\lambda$ optimaal is, en verzoen met vraag
        21: de exponentiële snelheid $H(\alpha)$ van de
        asymptotische uitspraak wordt bereikt door een
        één-regel-ongelijkheid zonder enige asymptotiek.
\end{enumerate}
\end{problem}
