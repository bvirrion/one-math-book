\chapter{Laurentreeksen en de residuenstelling}
\label{ch:b3:residues}

Wat gebeurt er met een \omterm{def:b3:holomorphic:holo}{holomorfe functie} bij een punt waar zij
\emph{niet} gedefinieerd is? Het antwoord is een complete
trichotomie --- ophefbaar punt, pool, of essentiële
singulariteit --- afgelezen uit een tweezijdige machtreeks, de
Laurentontwikkeling. Eén coëfficiënt van die ontwikkeling, het
\emph{\omterm{def:b3:residues:singularities}{residu}}, controleert elke \omterm{def:b3:holomorphic:contour}{contourintegraal} om de
singulariteit: de residuenstelling zet harde bepaalde
integralen om in eindige algebra, telt nullen van functies
(argumentprincipe, Rouché), en bewijst de open-afbeeldingsstelling.
We upgraden eerst de stelling van Cauchy van stervormige
gebieden naar haar definitieve, homologie-vrije vorm --- het
elegante argument van Dixon --- zodat alle contouren met
\omterm{def:b3:holomorphic:index}{windinggetal} nul om het complement beschikbaar worden.

\section{De globale stelling van Cauchy}

Een \emph{cykel}\index{cykel} $\Gamma$ is een eindige formele
som van gesloten paden $\gamma_1, \dots, \gamma_m$;
integralen en indices langs $\Gamma$ zijn de bijbehorende
sommen, en $\operatorname{im}\Gamma =
\bigcup\operatorname{im}\gamma_j$.

\begin{theorem}[Cauchy, globale vorm]
\label{thm:b3:residues:globalcauchy}
Zij $\Omega \subseteq \C$ open, $f \in \mathcal H(\Omega)$,
en $\Gamma$ een cykel in $\Omega$ zodanig dat
\[
\operatorname{Ind}_\Gamma(w) = 0
\qquad\text{voor elke } w \notin \Omega .
\]
Dan, voor alle $z \in \Omega\setminus\operatorname{im}\Gamma$,
\[
\frac1{2\iu\pi}\int_\Gamma\frac{f(w)}{w - z}\,\dd w =
\operatorname{Ind}_\Gamma(z)\,f(z),
\qquad\text{en}\qquad
\int_\Gamma f(w)\,\dd w = 0 .
\]
\index{stelling van Cauchy (globaal)}
\end{theorem}

\begin{proof}[Bewijs (Dixon)]
Definieer $g \colon \Omega\times\Omega \to \C$ door
\[
g(z, w) = \begin{cases}
\dfrac{f(w) - f(z)}{w - z} & w \neq z,\\[4pt]
f'(z) & w = z .
\end{cases}
\]
\emph{$g$ is continu}: buiten de diagonaal, duidelijk. Bij
een diagonaalpunt $(a, a)$, ontwikkel $f$ in een machtreeks
in $a$ (\cref{thm:b3:holomorphic:analytic}): $f(w) - f(z) =
\sum_{n\geq1}c_n\bigl((w-a)^n - (z-a)^n\bigr)$, en delen van
elke term door $w - z$ (factorisatie van $u^n - v^n$) geeft,
voor $z, w \in D(a, r)$,
\[
g(z, w) =
\sum_{n\geq1}c_n\sum_{j=0}^{n-1}(w-a)^{\,j}(z-a)^{\,n-1-j},
\]
ook geldig op de diagonaal (elke innersom wordt
$n(z-a)^{n-1}$, sommerend tot $f'(z)$). Voor kleine $r$
convergeert de reeks uniform op $D(a,r)^2$
($\abs{\text{term}} \leq n\abs{c_n}r^{n-1}$, sommeerbaar
binnen de straal): de som is continu.

Zet $h(z) = \frac1{2\iu\pi}\int_\Gamma g(z, w)\,\dd w$ op
$\Omega$: continu (uniforme continuïteit van $g$ op
compacta), en \omterm{def:b3:holomorphic:holo}{holomorf} --- door Morera
(\cref{thm:b3:holomorphic:weierstrassconv}'s criterium): voor
een driehoek $T \subseteq \Omega$ geeft Fubini
$\int_{\partial T}h = \frac1{2\iu\pi}\int_\Gamma\bigl(
\int_{\partial T}g(z, w)\dd z\bigr)\dd w = 0$, de innersom
verdwijnend omdat $z \mapsto g(z, w)$ \omterm{def:b3:holomorphic:holo}{holomorf} is op $\Omega$
(in $z = w$ is de singulariteit ophefbaar: $g$ is daar
continu en elders \omterm{def:b3:holomorphic:holo}{holomorf} --- het voortzettingsargument van
het bewijs van \cref{thm:b3:holomorphic:formula}).

Op de \omterm{def:b3:topology:topology}{open verzameling} $\Omega' = \{z \notin
\operatorname{im}\Gamma : \operatorname{Ind}_\Gamma(z) =
0\}$, definieer $h_1(z) =
\frac1{2\iu\pi}\int_\Gamma\frac{f(w)}{w - z}\dd w$:
\omterm{def:b3:holomorphic:holo}{holomorf} op $\Omega'$ (Morera of differentiatie onder de
integraal). Voor $z \in \Omega\cap\Omega'$:
\[
h(z) = \frac1{2\iu\pi}\int_\Gamma\frac{f(w)}{w-z}\dd w
- f(z)\operatorname{Ind}_\Gamma(z) = h_1(z) .
\]
Per hypothese $\Omega\cup\Omega' = \C$ ($w \notin \Omega
\Rightarrow \operatorname{Ind}_\Gamma(w) = 0$), dus $h$ en
$h_1$ plakken tot een gehele functie $H$. Omdat de
onbegrensde component van het complement van
$\operatorname{im}\Gamma$ in $\Omega'$ ligt en $h_1(z) \to
0$ als $\abs z \to \infty$ (ML-grens), is $H$ begrensd en
neigt naar $0$: Liouville
(\cref{cor:b3:holomorphic:liouville}) geeft $H \equiv 0$.
Dus $h \equiv 0$ op $\Omega$, wat de integrale formule is.
Toepassen, voor vast $a \in
\Omega\setminus\operatorname{im}\Gamma$, op $\tilde f(w) =
(w - a)f(w)$ in $z = a$:
\[
\frac1{2\iu\pi}\int_\Gamma f(w)\dd w =
\frac1{2\iu\pi}\int_\Gamma \frac{\tilde f(w)}{w - a}\dd w =
\operatorname{Ind}_\Gamma(a)\,\tilde f(a) = 0 .
\]
\end{proof}

\section{Laurentreeksen en geïsoleerde singulariteiten}

\begin{theorem}[Laurentontwikkeling]\label{thm:b3:residues:laurent}
Zij $f$ \omterm{def:b3:holomorphic:holo}{holomorf} op de annulus $A = \{r < \abs{z - a} <
R\}$ ($0 \leq r < R \leq \infty$). Dan\index{Laurentreeks}
\[
f(z) = \sum_{n\in\Z}c_n\,(z - a)^n
\qquad\text{op } A,
\qquad
c_n = \frac1{2\iu\pi}\int_{C_\rho}\frac{f(w)}{(w -
a)^{n+1}}\,\dd w
\]
voor elke $r < \rho < R$ (onafhankelijk van $\rho$), de twee
halfreeksen normaal convergent op compacte subannuli. De
ontwikkeling is uniek.
\end{theorem}

\begin{proof}
Fixeer $r < \rho_1 < \abs{z - a} < \rho_2 < R$ en laat
$\Gamma = C_{\rho_2} - C_{\rho_1}$ (buiten tegen de klok in,
binnen met de klok mee): een cykel in $A$ met
$\operatorname{Ind}_\Gamma(w) = 0$ voor alle $w \notin A$
(punten binnen de kleine schijf: $1 - 1 = 0$; buiten de
grote: $0 - 0$). Door
\cref{thm:b3:residues:globalcauchy},
$\operatorname{Ind}_\Gamma(z) = 1 - 0 = 1$ geeft
\[
f(z) = \frac1{2\iu\pi}\int_{C_{\rho_2}}\frac{f(w)}{w - z}\dd
w - \frac1{2\iu\pi}\int_{C_{\rho_1}}\frac{f(w)}{w - z}\dd w .
\]
Ontwikkel de eerste kern als in
\cref{thm:b3:holomorphic:analytic} (machten van $\frac{z -
a}{w - a}$, modulus $< 1$): het niet-negatieve deel
$\sum_{n\geq0}c_n(z-a)^n$. In de tweede, ontwikkel de andere
kant op: $\frac{-1}{w - z} = \frac1{(z-a)(1 - \frac{w - a}{z -
a})} = \sum_{m\geq0}\frac{(w-a)^m}{(z - a)^{m+1}}$, normaal
convergent op $C_{\rho_1}$: het negatieve deel
$\sum_{n\leq-1}c_n(z-a)^n$ met de gestelde coëfficiënten
(\omterm{def:b3:holomorphic:index}{index} $n = -m-1$). Onafhankelijkheid van $\rho$: de
coëfficiëntintegralen over $C_{\rho}$ en $C_{\rho'}$
verschillen door $\int_\Gamma$ over een nul-index-cykel van
de holomorfe $\frac{f(w)}{(w-a)^{n+1}}$ in $A$: nul, door
\cref{thm:b3:residues:globalcauchy} opnieuw. Uniciteit:
integreer $\sum c_n(z-a)^n$ tegen $(z - a)^{-m-1}$ over
$C_\rho$ term voor term (normale convergentie): slechts $n =
m$ overleeft.
\end{proof}

\begin{definition}\label{def:b3:residues:singularities}
Als $f$ \omterm{def:b3:holomorphic:holo}{holomorf} is op een gepuncteerde schijf
$D(a, R)\setminus \{a\}$, ontwikkel via Laurent ($r = 0$).
Drie exclusieve gevallen:\index{geïsoleerde singulariteit}
\begin{itemize}
  \item alle $c_n = 0$ voor $n < 0$: \emph{\omterm{thm:b3:residues:riemanncw}{ophefbare
        singulariteit}} (de niet-negatieve reeks zet $f$
        \omterm{def:b3:holomorphic:holo}{holomorf} voort naar $a$);
  \item $c_n \neq 0$ voor eindig veel, minstens één, $n < 0$:
        een \emph{pool} van orde $m = -\min\{n : c_n \neq
        0\}$; equivalent $f = g/(z-a)^m$, $g$ \omterm{def:b3:holomorphic:holo}{holomorf},
        $g(a) \neq 0$; equivalent $\abs{f(z)} \to \infty$
        als $z\to a$;
  \item oneindig veel negatieve $c_n \neq 0$:
        \emph{essentiële singulariteit}.
\end{itemize}
Het \emph{residu}\index{residu} is
$\operatorname{Res}(f, a) = c_{-1}$. Een functie \omterm{def:b3:holomorphic:holo}{holomorf} op
$\Omega$ minus een verzameling polen is
\emph{meromorf}\index{meromorfe functie} op $\Omega$.
\end{definition}

\begin{theorem}[Riemann; Casorati--Weierstrass]
\label{thm:b3:residues:riemanncw}
Zij $f$ \omterm{def:b3:holomorphic:holo}{holomorf} op $D(a,R)\setminus\{a\}$.
\begin{enumerate}
  \item (Riemann) Als $f$ \emph{begrensd} is bij $a$, is de
        singulariteit ophefbaar.\index{ophefbare singulariteit}
  \item (Casorati--Weierstrass) Als $a$ essentieel is, dan is
        $f\bigl(D(a,\varepsilon)\setminus\{a\}\bigr)$ \omterm{def:b3:topology:interior}{dicht}
        in $\C$ voor elke $\varepsilon$.
        \index{stelling van Casorati--Weierstrass}
\end{enumerate}
\end{theorem}

\begin{proof}
(1) Voor $n < 0$ en $\rho \to 0$: $\abs{c_n} \leq
M\rho^{-n-1}\cdot\rho\cdot\rho^{-\,n}\dots$ door ML op
$C_\rho$: $\abs{c_n} \leq \frac{1}{2\pi}\,2\pi\rho\cdot
M\rho^{-(n+1)} = M\rho^{-n} \to 0$ (omdat $-n > 0$): alle
negatieve coëfficiënten verdwijnen.
(2) Als enige waarde $b$ niet benaderd werd: $\abs{f - b}
\geq \delta$ bij $a$, dus $g = 1/(f - b)$ is \omterm{def:b3:holomorphic:holo}{holomorf} en
begrensd bij $a$: ophefbaar (1), $g$ zet voort met waarde
$c$. Als $c \neq 0$, is $f = b + 1/g$ begrensd bij $a$:
ophefbaar --- uitgesloten. Als $c = 0$, heeft $g$ een nul van
eindige orde $m$ in $a$ (\cref{thm:b3:holomorphic:identity};
$g \not\equiv 0$), en $f = b + 1/g$ heeft een pool van orde
$m$: opnieuw uitgesloten.
\end{proof}

\section{De residuenstelling}

\begin{theorem}[Residuenstelling]\label{thm:b3:residues:residue}
Zij $\Omega$ open, $S \subseteq \Omega$ eindig, $f \in
\mathcal H(\Omega\setminus S)$, en $\Gamma$ een cykel in
$\Omega\setminus S$ met $\operatorname{Ind}_\Gamma(w) = 0$
voor elke $w \notin \Omega$. Dan\index{residuenstelling}
\[
\frac{1}{2\iu\pi}\int_\Gamma f(z)\,\dd z
= \sum_{a\in S}\operatorname{Ind}_\Gamma(a)\,
\operatorname{Res}(f, a) .
\]
\end{theorem}

\begin{proof}
Voor elke $a \in S$, zij $P_a(z) = \sum_{n\leq-1}c_n^{(a)}(z -
a)^n$ het principale deel van $f$ in $a$: een reeks
convergent op $\C\setminus\{a\}$ (haar straal in $1/(z-a)$ is
oneindig: de Laurentstaart convergeert voor alle kleine
$\abs{z-a}$ dus, als machtreeks in $(z-a)^{-1}$, overal), en
\omterm{def:b3:holomorphic:holo}{holomorf} daar. Dan heeft $g = f - \sum_{a\in S}P_a$
\omterm{thm:b3:residues:riemanncw}{ophefbare singulariteiten} in elk punt van $S$ (haar
Laurentontwikkeling in $a$ heeft geen negatief deel: de
andere $P_{a'}$ zijn \omterm{def:b3:holomorphic:holo}{holomorf} in $a$), dus $g$ zet \omterm{def:b3:holomorphic:holo}{holomorf}
voort naar $\Omega$, en
\cref{thm:b3:residues:globalcauchy} geeft $\int_\Gamma g =
0$. Rest te integreren van elke $P_a$: term-voor-term
(normale convergentie op het compacte
$\operatorname{im}\Gamma$, dat $a$ vermijdt),
\[
\frac1{2\iu\pi}\int_\Gamma(z - a)^n\,\dd z = 0 \ (n \leq -2:
\text{primitieve } \tfrac{(z-a)^{n+1}}{n+1}),
\qquad
\frac1{2\iu\pi}\int_\Gamma\frac{\dd z}{z - a} =
\operatorname{Ind}_\Gamma(a),
\]
dus $\frac1{2\iu\pi}\int_\Gamma P_a =
c_{-1}^{(a)}\operatorname{Ind}_\Gamma(a)$. Sommeer over $a$.
\end{proof}

\begin{method}[Residuen berekenen]\label{met:b3:residues:compute}
Eenvoudige pool: $\operatorname{Res}(f, a) = \lim_{z\to a}(z -
a)f(z)$; voor $f = g/h$ met $g(a) \neq 0$, $h(a) = 0$, $h'(a)
\neq 0$: $\operatorname{Res} = g(a)/h'(a)$. Pool van orde $m$:
$\operatorname{Res}(f, a) = \frac1{(m-1)!}\lim_{z\to
a}\bigl((z-a)^mf(z)\bigr)^{(m-1)}$. Essentiële
singulariteiten: ontwikkel en lees $c_{-1}$ (b.v.\ uit
bekende reeksen). Verifieer altijd welke polen de contour
werkelijk omcirkelt, en met welke \omterm{def:b3:holomorphic:index}{index}.
\end{method}

\begin{example}[De vier klassieke integraaltypen]
\label{ex:b3:residues:integrals}
(a) \emph{Rationaal over $\R$}: voor $\int_\R\frac{\dd x}{1 +
x^4}$, sluit met een grote halve cirkel $S_R$ in het bovenhalfvlak:
de integrand is $O(R^{-4})$ daar, dus $\int_{S_R} \to 0$
(ML), en de residuenstelling met de polen
$\eu^{\iu\pi/4}, \eu^{3\iu\pi/4}$ (\omterm{def:b3:groups:simple}{eenvoudig}, \omterm{def:b3:residues:singularities}{residuen}
$\frac1{4z^3} = \frac{z}{4z^4} = -\frac z4$ in een pool)
geeft
\[
\int_\R\frac{\dd x}{1 + x^4}
= 2\iu\pi\Bigl(-\frac{\eu^{\iu\pi/4}}4 -
\frac{\eu^{3\iu\pi/4}}4\Bigr)
= \frac{\pi}{\sqrt2} .
\]
(b) \emph{Fourier-type}: voor $t \geq 0$,
$\int_\R\frac{\eu^{\iu tx}}{1 + x^2}\dd x =
2\iu\pi\operatorname{Res}\bigl(\tfrac{\eu^{\iu tz}}{1+z^2},
\iu\bigr) = 2\iu\pi\frac{\eu^{-t}}{2\iu} = \pi\eu^{-t}$ ---
de bovenhalve cirkel werkt omdat $\abs{\eu^{\iu tz}} =
\eu^{-t\operatorname{Im}z} \leq 1$ daar; reële delen nemen:
$\int_\R\frac{\cos(tx)}{1+x^2}\dd x = \pi\eu^{-\abs t}$,
regelend de toegegeven formule van
\cref{exo:b3:lebesgue:10}.
(c) \emph{Trigonometrisch over een periode}: substitueer $z =
\eu^{\iu t}$, $\cos t = \frac{z + z^{-1}}2$, $\dd t =
\frac{\dd z}{\iu z}$: $\int_0^{2\pi}\frac{\dd t}{a + \cos t}$
($a > 1$) wordt een residuentelling binnen de
eenheidscirkel (\cref{exo:b3:residues:2}).
(d) \emph{Reeksen}: paar $f$ met $\pi\cot(\pi z)$, waarvan
de polen de gehele getallen zijn met \omterm{def:b3:residues:singularities}{residu} $1$: het
weekendprobleem sommeert $\sum n^{-2}$ en $\sum n^{-4}$ zo.
\end{example}

\section{Het argumentprincipe en de stelling van Rouché}

\begin{theorem}[Argumentprincipe]
\label{thm:b3:residues:argument}
Zij $f$ \omterm{def:b3:residues:singularities}{meromorf} op $\Omega$, met nullen $z_j$ (ordes $m_j$)
en polen $p_k$ (ordes $\mu_k$), en $\gamma$ een gesloten pad
in $\Omega$ dat ze alle vermijdt, met
$\operatorname{Ind}_\gamma = 0$ buiten $\Omega$. Dan
\index{argumentprincipe}
\[
\frac1{2\iu\pi}\int_\gamma\frac{f'(z)}{f(z)}\,\dd z
= \sum_j m_j\operatorname{Ind}_\gamma(z_j)
- \sum_k \mu_k\operatorname{Ind}_\gamma(p_k)
\]
(eindig veel termen zijn niet-nul). Voor een eenvoudige
tegen-de-klok-in-contour telt de integraal nullen minus polen
binnenin, met multipliciteit --- en is gelijk aan het
\omterm{def:b3:holomorphic:index}{windinggetal} van het beeldpad $f\circ\gamma$ om $0$.
\end{theorem}

\begin{proof}
Bij een nul van orde $m$: $f = (z-a)^mg$, $g(a) \neq 0$, dus
$\frac{f'}f = \frac m{z - a} + \frac{g'}g$ met de tweede
term \omterm{def:b3:holomorphic:holo}{holomorf} bij $a$: een eenvoudige pool van \omterm{def:b3:residues:singularities}{residu} $m$.
Bij een pool van orde $\mu$: $f = (z-a)^{-\mu}g$ geeft
\omterm{def:b3:residues:singularities}{residu} $-\mu$. Elders is $\frac{f'}f$ \omterm{def:b3:holomorphic:holo}{holomorf}. (De nullen en
polen met niet-nul \omterm{def:b3:holomorphic:index}{index} liggen in een compacte regio
omcirkeld door $\gamma$; door de identiteitsstelling zijn
zij daar eindig in aantal, $f \not\equiv 0$.) Pas
\cref{thm:b3:residues:residue} toe. De laatste opmerking:
$\frac1{2\iu\pi}\int_\gamma\frac{f'}f =
\frac1{2\iu\pi}\int_{f\circ\gamma}\frac{\dd w}w =
\operatorname{Ind}_{f\circ\gamma}(0)$ (substitueer $w =
f(\gamma(t))$).
\end{proof}

\begin{theorem}[Rouché]\label{thm:b3:residues:rouche}
Zij $f, g$ \omterm{def:b3:holomorphic:holo}{holomorf} op $\Omega$, en $\gamma$ een gesloten
pad met $\operatorname{Ind}_\gamma \in \{0,1\}$, nul buiten
$\Omega$ (een eenvoudige contour). Als\index{stelling van Rouché}
\[
\abs{g(z)} < \abs{f(z)} \qquad \text{op }
\operatorname{im}\gamma,
\]
dan hebben $f$ en $f + g$ hetzelfde aantal nullen (met
multipliciteit) in de regio $\{\operatorname{Ind}_\gamma =
1\}$.
\end{theorem}

\begin{proof}
Voor $t \in \intcc01$ heeft $f_t = f + tg$ geen nul op
$\operatorname{im}\gamma$ ($\abs{f_t} \geq \abs f - \abs g >
0$), dus
\[
N(t) = \frac1{2\iu\pi}\int_\gamma
\frac{f_t'(z)}{f_t(z)}\,\dd z
\]
is goed gedefinieerd; zij telt de nullen in de omsloten regio
(\cref{thm:b3:residues:argument}; geen polen). $N$ is
continu in $t$ (de integrand is gezamenlijk continu, de
noemers uniform van onderen begrensd --- gedomineerde
convergentie) en geheelwaardig: constant. $N(0) = N(1)$.
\end{proof}

\begin{corollary}[Open-afbeeldingsstelling]
\label{cor:b3:residues:openmapping}
Een niet-constante \omterm{def:b3:holomorphic:holo}{holomorfe functie} op een samenhangende
\omterm{def:b3:topology:topology}{open verzameling} is een open afbeelding. In het bijzonder
(opnieuw) geldt het maximumprincipe, en een holomorfe
bijectie heeft holomorfe inverse.\index{open-afbeeldingsstelling (holomorf)}
\end{corollary}

\begin{proof}
Zij $f(a) = b$; $f - b$ heeft een nul van enige eindige orde
$m \geq 1$ in $a$ (identiteitsstelling: $f \not\equiv b$).
Kies $r$ met $f - b$ nultvrij op $\bar D(a, r)\setminus\{a\}$
(\omterm{thm:b3:holomorphic:identity}{geïsoleerde nullen}) en laat $\delta = \min_{\abs{z - a} =
r}\abs{f(z) - b} > 0$. Voor $\abs{w - b} < \delta$: op de
cirkel, $\abs{(b - w)} < \delta \leq \abs{f - b}$, dus
Rouché ($f - b$ versus constante $b - w$) zegt dat $f - w$
exact $m$ nullen in $D(a, r)$ heeft: elke zulke $w$ wordt
bereikt --- $f(D(a,r)) \supseteq D(b, \delta)$: open.
Maximumprincipe: een inwendig maximum van $\abs f$ is
onmogelijk voor niet-constante $f$, haar beeld om $f(a)$
punten van grotere modulus bevattend. Inverse: een holomorfe
bijectie $f$ is open, dus $f^{-1}$ is continu; de nul van $f
- f(a)$ in $a$ is \omterm{def:b3:groups:simple}{eenvoudig} ($m \geq 2$ zou $m$ originelen
van nabije waarden geven --- verschillende, omdat $f'$
slechts in geïsoleerde punten verdwijnt, dus bij $a$ zijn de
$m$ nullen van $f - w$ \omterm{def:b3:groups:simple}{eenvoudig} en verschillend voor
generieke kleine $w$: tegenspraak met injectiviteit); dan
$f'(a) \neq 0$ en het differentiequotiënt van $f^{-1}$
convergeert: $\bigl(f^{-1}\bigr)'(b) = 1/f'(a)$.
\end{proof}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:b3:residues:1}
Classificeer de singulariteit in $0$ en bereken het \omterm{def:b3:residues:singularities}{residu}:
\[
\frac{\sin z}{z},\qquad
\frac{\eu^z - 1}{z^2},\qquad
\frac{1}{z(z-1)^2},\qquad
\frac{\cos z}{z^3},\qquad
\eu^{1/z},\qquad
\frac1{\sin z} .
\]
Geef ook het \omterm{def:b3:residues:singularities}{residu} van de derde in $z = 1$ en van de
laatste in $z = \pi$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b3:residues:2}
Voor $a > 1$ bereken, via $z = \eu^{\iu t}$:
\[
\int_0^{2\pi}\frac{\dd t}{a + \cos t}
= \frac{2\pi}{\sqrt{a^2 - 1}} .
\]
Controleer de limietgedragingen $a \to 1^+$ en $a \to
\infty$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:residues:3}
Bereken met halfcirkelcontouren, de boogschattingen
rechtvaardigend:
\[
\int_\R\frac{x^2}{1 + x^6}\,\dd x = \frac\pi3,
\qquad
\int_\R\frac{\dd x}{(1 + x^2)^{2}} = \frac\pi2
\quad\text{(herleidend \cref{exo:b3:fouriertransform:3})}.
\]
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:residues:4}
Bewijs, voor $t \geq 0$ en $a > 0$:
\[
\int_\R\frac{\cos(tx)}{x^2 + a^2}\,\dd x =
\frac{\pi}{a}\,\eu^{-at},
\]
en leid de Fouriertransformatie van $x \mapsto \eu^{-a\abs
x}$ af door inversie --- vergelijkend met
\cref{exo:b3:fouriertransform:1}.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:residues:5}
Ontwikkel $f(z) = \dfrac1{(z-1)(z-2)}$ in Laurentreeksen in
elk van de drie regio's $\abs z < 1$, $1 < \abs z < 2$,
$\abs z > 2$. Waarom verschillen de drie ontwikkelingen?
Verklaar waarom de coëfficiënt van $z^{-1}$ in de tweede en
derde ontwikkeling \emph{geen} \omterm{def:b3:residues:singularities}{residu} van $f$ in $0$ is
($f$ heeft daar geen singulariteit), en bereken de
werkelijke \omterm{def:b3:residues:singularities}{residuen} van $f$, in $1$ en in $2$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:residues:6}
(a) Toon dat $\eu^{1/z}$ een essentiële singulariteit in $0$
heeft en verifieer Casorati--Weierstrass met de hand: los
$\eu^{1/z} = w$ expliciet op voor elke $w \neq 0$,
oplossingen willekeurig \omterm{def:b3:topology:interior}{dicht} bij $0$ vertonend.
(b) Toon dat $\abs{\eu^{1/z}}$ onbegrensd is op elke
gepuncteerde \omterm{def:b3:topology:topology}{omgeving} van $0$ terwijl $\eu^{1/z}$ geen pool
heeft: welke limiet faalt?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:residues:7}
Tel met Rouché:
(a) de nullen van $z^7 - 4z^3 + z - 1$ in $\abs z < 1$;
(b) de nullen van $z^4 + 5z + 1$ in $\abs z < 1$ en in $1 <
\abs z < 2$;
(c) herbewijs d'Alembert--Gauss: een monisch polynoom van
graad $n$ heeft $n$ nullen in enige grote schijf
\emph{(vergelijk met $z^n$)}.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b3:residues:8}
(Hurwitz) Zij $f_n \to f$ uniform op compacta, $f_n \in
\mathcal H(\Omega)$, $\Omega$ \omterm{def:b3:topology:connected}{samenhangend}, $f \not\equiv 0$.
(a) Toon dat als alle $f_n$ nultvrij zijn, $f$ dat ook is.
\emph{(Als $f(a) = 0$: argumentprincipe op een kleine
cirkel om $a$, en \cref{thm:b3:holomorphic:weierstrassconv}
om tot de limiet over te gaan in $\int f_n'/f_n$.)}
(b) Toon dat als alle $f_n$ injectief zijn, $f$ injectief of
constant is. \emph{(Pas (a) toe op $z \mapsto f_n(z) -
f_n(w)$ op $\Omega\setminus\{w\}$.)}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b3:residues:9}
Voor $n \geq 2$, integreer $\frac1{1 + z^n}$ over de rand
van de sector $\{0 \leq \arg z \leq \frac{2\pi}n,\ \abs z
\leq R\}$ en leid af
\[
\int_0^{+\infty}\frac{\dd x}{1 + x^n} =
\frac{\pi}{n\,\sin(\pi/n)} .
\]
Verifieer $n = 2$ tegen $\arctan$, en de limiet $n \to
\infty$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:residues:10}
Zij $f$ een rationale functie met $\deg(\text{noemer})
\geq \deg(\text{teller}) + 2$. Toon dat de som van
\emph{alle} \omterm{def:b3:residues:singularities}{residuen} van $f$ nul is \emph{(integreer over
steeds grotere cirkels)}. Gebruik dit om de
partiële-breukontbinding van $\frac1{z(z-1)(z-2)}$ te
herberekenen zonder lineaire algebra.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b3:residues:11}
(Het sleutelgat: Eulers reflectie-integraal) Voor $0 < a <
1$, bereken
\[
I(a) = \int_0^{\infty}\frac{x^{a-1}}{1 + x}\,\dd x =
\frac{\pi}{\sin(\pi a)}
\]
door $f(z) = \frac{z^{a-1}}{1+z} =
\frac{\eu^{(a-1)\log z}}{1 + z}$ te integreren (logaritme
gesneden langs $\R_+$, $\arg z \in \intoo0{2\pi}$) over de
sleutelgatcontour: uit langs de top van de snede van
$\varepsilon$ naar $R$, om $C_R$, terug onder de snede, om
$C_\varepsilon$. Rechtvaardig: de twee rechte stukken
verschillen door de factor $\eu^{2\iu\pi(a-1)}$, de
cirkelbijdragen verdwijnen ($R^{a-1}\cdot R \to 0$ en
$\varepsilon^{a-1}\cdot\varepsilon \to 0$), en de unieke
pool $z = -1$ heeft \omterm{def:b3:residues:singularities}{residu} $\eu^{\iu\pi(a - 1)}$. Leid ook
af $\Gamma(a)\Gamma(1 - a) = \frac\pi{\sin\pi a}$
\emph{(schrijf $\Gamma(a)\Gamma(1-a) = B(a, 1-a)$ door
\cref{pb:b3:lebesgue:1} en substitueer $t =
\frac{x}{1+x}$)}.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:residues:12}
(Nullen tellen met het argumentprincipe, numeriek)
Zij $P(z) = z^4 + 8z + 1$.
(a) Hoeveel nullen in de eenheidsschijf? \emph{(Rouché
tegen $8z + 1$.)}
(b) Hoeveel in de annulus $1 < \abs z < 3$?
\emph{(Rouché tegen $z^4$ op $\abs z = 3$.)} Verscherp:
toon dat elke nul modulus $< 2.1$ heeft.
(c) Hoeveel in het rechterhalfvlak? \emph{(Tel eerst op
$\abs z = 2$; volg dan het beeld van de imaginaire as:
$P(\iu t) = t^4 + 1 + 8\iu t$ heeft positief reëel deel
doorheen, dus geen nullen op de as, en de
argumentvariatie erlangs is berekenbaar --- concludeer met
een grote halfschijf.)}
\end{exercise}

\section{Probleem: \texorpdfstring{$\zeta(2k)$}{zeta(2k)} via de
cotangens}

\begin{problem}[{Weekendprobleem --- $\sum n^{-2k}$ sommeren
met \omterm{def:b3:residues:singularities}{residuen}}]\label{pb:b3:residues:1}
De residuenstelling sommeert reeksen: het paren van een
rationale functie met $\pi\cot(\pi z)$, waarvan de polen in
de gehele getallen zitten, zet $\sum_{n}f(n)$ om in een
residuentelling. We bewijzen de methode en berekenen
$\zeta(2) = \frac{\pi^2}{6}$ en $\zeta(4) =
\frac{\pi^4}{90}$ --- de waarden gevonden door Fourierreeksen
in Jaar~2 en door operator-sporen in \cref{ch:b3:spectral},
nu door contourintegratie.

\medskip
\noindent\textbf{Deel I --- De cotangenskern.}
\begin{enumerate}
  \item Toon dat $\pi\cot(\pi z)$ \omterm{def:b3:residues:singularities}{meromorf} is op $\C$ met
        eenvoudige polen precies in $z = n \in \Z$, elk van
        \omterm{def:b3:residues:singularities}{residu} $1$ \emph{(bereken
        $\lim_{z\to n}(z-n)\pi\cot\pi z$)}.
  \item Bereken het begin van de Laurentontwikkeling in $0$:
        \[
        \pi\cot(\pi z) = \frac1z - \frac{\pi^2}{3}\,z -
        \frac{\pi^4}{45}\,z^3 + O(z^5) ,
        \]
        door de machtreeks van $\cos$ te delen door die van
        $\sin$ (rechtvaardig de deling: $\frac{\sin\pi z}{\pi
        z}$ is \omterm{def:b3:holomorphic:holo}{holomorf} en niet-nul bij $0$, dus haar
        reciproke is \omterm{def:b3:holomorphic:holo}{holomorf}; identificeer coëfficiënten
        tot orde $3$).
  \item Zij $C_N$ de rand van het vierkant met hoekpunten
        $(\pm1\pm\iu)(N + \frac12)$. Toon dat
        $\abs{\cot(\pi z)} \leq 2$ op $C_N$ voor elke $N \geq
        1$. \emph{(Op verticale zijden, $\cot(\pi(\pm(N +
        \frac12) + \iu y)) = \mp\tan(\iu\pi y)$, van modulus
        $\abs{\tanh(\pi y)} \leq 1$; op horizontale zijden
        $\abs y = N + \frac12$, begrens
        $\abs{\cot(\pi(x\pm\iu y))} \leq \coth(\pi y) \leq
        \coth(\pi/2) < 1.1$.)}
\end{enumerate}

\medskip
\noindent\textbf{Deel II --- De sommatiestelling.}
\begin{enumerate}[resume]
  \item Zij $f$ rationaal, \omterm{def:b3:holomorphic:holo}{holomorf} in de gehele getallen,
        met $\deg(\text{noemer}) \geq \deg(\text{teller}) +
        2$. Met de residuenstelling op $C_N$ en de grens van
        vraag~3, bewijs:
        \[
        \lim_{N\to\infty}\ \sum_{n = -N}^{N} f(n)
        = -\sum_{p\ \text{pool van}\ f}
        \operatorname{Res}\bigl(\pi\cot(\pi z)f(z),\,p\bigr).
        \]
  \item Waar heeft het argument de graadvoorwaarde nodig?
        Toon met een voorbeeld (neem $f(z) = 1/(z +
        \frac12)$) dat voor tragere afname de symmetrische
        limiet nog kan bestaan terwijl de tweezijdige reeks
        divergeert --- en dat de formule dan de
        \emph{hoofdwaarde} berekent.
\end{enumerate}

\medskip
\noindent\textbf{Deel III --- De waarden.}
\begin{enumerate}[resume]
  \item Pas de methode toe op $f(z) = 1/z^2$: hier heeft $f$
        haar pool \emph{in} een geheel getal, dus voer het
        argument direct uit --- integreer $g(z) =
        \frac{\pi\cot(\pi z)}{z^2}$ over $C_N$, toon dat de
        integraal $\to 0$, en bereken
        $\operatorname{Res}(g, 0)$ uit vraag~2. Concludeer:
        \[
        2\sum_{n\geq1}\frac1{n^2} = \frac{\pi^2}{3},
        \qquad \zeta(2) = \frac{\pi^2}6 .
        \]
  \item Hetzelfde met $g(z) = \frac{\pi\cot(\pi z)}{z^4}$:
        bereken $\operatorname{Res}(g, 0)$ en leid
        $\zeta(4) = \frac{\pi^4}{90}$ af.
  \item Verklaar het algemene patroon: voor elke $k \geq 1$
        is $\zeta(2k)$ $-\frac12$ maal de coëfficiënt van
        $z^{2k-1}$ in de Laurentontwikkeling van
        $\pi\cot(\pi z)$ in $0$ --- een rationaal
        veelvoud van $\pi^{2k}$. Bereken $\zeta(6)$ door de
        deling van vraag~2 één stap verder te duwen. Wat
        zegt de methode over $\zeta(3)$ --- en waarom zegt
        zij niets?
\end{enumerate}

\medskip
\noindent\textbf{Deel IV --- De partiële-breukontwikkeling
van de cotangens.}
\begin{enumerate}[resume]
  \item Fixeer $w \in \C\setminus\Z$ en pas de methode van
        Deel~II toe op $f(z) = \dfrac{1}{(z - w)(z + w)}$ ---
        merkend dat $\pi\cot(\pi z)f(z)$ nu bijkomende
        eenvoudige polen in $\pm w$ heeft, waarvan de
        \omterm{def:b3:residues:singularities}{residuen} bij de telling moeten. Leid de
        partiële-breukontwikkeling af
        \[
        \pi\cot(\pi w) = \frac1w +
        \sum_{n\geq1}\frac{2w}{w^2 - n^2},
        \]
        de reeks normaal convergent op compacte
        deelverzamelingen van $\C\setminus\Z$.
  \item Herwin uit deze ontwikkeling, door elke term in
        machten van $w$ te ontwikkelen (rechtvaardig de
        verwisseling), dezelfde Laurentcoëfficiënten als in
        vraag~2 --- de cirkel sluit: Eulers formule
        $\sum\frac1{n^2} = \frac{\pi^2}6$ is de
        coëfficiënt van $w$ in de twee gezichten van de
        cotangens. Vergelijk met het Fourierreeksbewijs
        (Jaar~2) en het spoorbewijs
        (\cref{pb:b3:spectral:1}): drie theorieën, één
        getal.
\end{enumerate}

\medskip
\noindent\textbf{Deel V --- Eulers product voor de sinus.}
De ontwikkeling van vraag~9 is de logaritmische afgeleide
van een oneindig product; we bewijzen nu Eulers
factorisatie van 1734 eerlijk.
\begin{enumerate}[resume]
  \item Voor $N \geq 1$ zet $P_N(z) =
        z\prod_{n=1}^{N}\bigl(1 - \frac{z^2}{n^2}\bigr)$.
        Toon dat $P_N$ convergeert, uniform op elke schijf
        $\bar D(0, R)$, naar een gehele functie $P$ waarvan
        de nullen precies de gehele getallen zijn, alle
        \omterm{def:b3:groups:simple}{eenvoudig}. \emph{(Voor $n \geq 2R$ schrijf de
        factor als $\exp\log(1 - z^2/n^2)$ met de
        principale logaritme van
        \cref{exo:b3:holomorphic:3}, begrens
        $\abs{\log(1+u)} \leq 2\abs u$ voor $\abs u \leq
        \frac12$ via de reeks, en exponentieer de normaal
        convergente som van logaritmen; de eindig veel
        resterende factoren zijn een polynoom. Concludeer
        met \cref{thm:b3:holomorphic:weierstrassconv}.)}
  \item Toon dat op $\C\setminus\Z$,
        \[
        \frac{P'(z)}{P(z)} = \frac1z +
        \sum_{n\geq1}\frac{2z}{z^2 - n^2} = \pi\cot(\pi z)
        \]
        \emph{(differentiëer de eindige producten, ga tot de
        limiet met \cref{thm:b3:holomorphic:weierstrassconv}
        en de nultvrijheid van $P$ buiten $\Z$, en citeer
        vraag~9)}.
  \item Toon dat $Q = \sin(\pi z)/P(z)$ voortzet tot een
        nultvrije gehele functie met $Q' = 0$, en concludeer
        \emph{Eulers product}:
        \[
        \sin(\pi z) = \pi z\prod_{n\geq1}
        \Bigl(1 - \frac{z^2}{n^2}\Bigr)
        \qquad (z \in \C) .
        \]
  \item (Wallis, 1655) Evalueer in $z = \frac12$:
        \[
        \frac\pi2 = \prod_{n\geq1}\frac{4n^2}{4n^2 - 1}
        = \lim_{N\to\infty}
        \frac{2\cdot2\cdot4\cdot4\cdots(2N)(2N)}
        {1\cdot3\cdot3\cdot5\cdots(2N-1)(2N+1)} .
        \]
  \item Voor $\abs z < 1$, ontwikkel de logaritme van het
        product als dubbele reeks (rechtvaardig de
        herrangschikking) en herwin $\zeta(2) =
        \frac{\pi^2}6$ door de coëfficiënt van $z^3$ te
        matchen in $\sin(\pi z) = \pi z -
        \frac{\pi^3}6z^3 + \cdots$ --- het productgezicht
        van Eulers getal.
\end{enumerate}

\medskip
\noindent\textbf{Deel VI --- Zuskerner.} De cotangens heeft
zusters; elk prijst haar eigen familie van reeksen.
\begin{enumerate}[resume]
  \item Differentiëer de ontwikkeling van vraag~9 term voor
        term (gerechtvaardigd door
        \cref{thm:b3:holomorphic:weierstrassconv}) om,
        normaal op compacta van $\C\setminus\Z$, te
        verkrijgen
        \[
        \frac{\pi^2}{\sin^2(\pi z)} =
        \sum_{n\in\Z}\frac1{(z - n)^2} .
        \]
  \item Evalueer in $z = \frac12$:
        $\sum_{m\geq0}\frac1{(2m+1)^2} = \frac{\pi^2}8$;
        herwin $\zeta(2)$ nogmaals door de gehele getallen
        te splitsen naar pariteit.
  \item Verifieer de verdubbelingsidentiteit $\tan\theta =
        \cot\theta - 2\cot(2\theta)$ en leid af
        \[
        \pi\tan(\pi z) =
        \sum_{m\geq0}\frac{8z}{(2m+1)^2 - 4z^2} ,
        \]
        normaal op compacta die $\frac12 + \Z$ vermijden.
  \item Ontwikkel om $0$ ($\abs z < \frac12$; Fubini
        opnieuw): met $\lambda(s) =
        \sum_{m\geq0}(2m+1)^{-s}$,
        \[
        \pi\tan(\pi z) =
        \sum_{k\geq0}8\cdot4^k\,\lambda(2k+2)\,z^{2k+1} ;
        \]
        vergelijk met $\tan u = u + \frac{u^3}3 + O(u^5)$ om
        $\lambda(2) = \frac{\pi^2}8$ te herwinnen en
        $\lambda(4) = \frac{\pi^4}{96}$ te krijgen, en
        kruiscontroleer $\zeta(4) = \frac{\pi^4}{90}$ via
        $\lambda(4) = (1 - 2^{-4})\,\zeta(4)$.
  \item Verifieer $\frac1{\sin\theta} = \cot\frac\theta2 -
        \cot\theta$ en leid af
        \[
        \frac{\pi}{\sin(\pi z)} = \frac1z +
        \sum_{n\geq1}(-1)^n\,\frac{2z}{z^2 - n^2} .
        \]
        Controleer de tekens tegen de \omterm{def:b3:residues:singularities}{residuen} van
        $\pi/\sin(\pi z)$ in de gehele getallen.
  \item Lees de coëfficiënt van $z$ af: $\eta(2) =
        \sum_{n\geq1}\frac{(-1)^{n-1}}{n^2} =
        \frac{\pi^2}{12}$, en bevestig de consistentie
        $\eta(2) = (1 - 2^{1-2})\,\zeta(2)$.
  \item (Finale) Evalueer de ontwikkeling van vraag~20 in
        $z = \frac12$ en leid \emph{Leibniz' formule} af
        \[
        \frac\pi4 = 1 - \frac13 + \frac15 - \frac17 +
        \cdots
        \]
        Sluit met een korte alinea: één kern per aritmetiek
        --- welke kern prijst welke familie van reeksen, en
        waarom zijn zij structureel blind voor $\zeta(3)$.

\end{enumerate}
\medskip
\noindent\textbf{Deel VII --- De volledige prijslijst:
Bernoulligetallen.}
\begin{enumerate}[resume]
  \item Combineer de partiële-breukontwikkeling van $\pi
        z\cot(\pi z)$ met de genererende functie van de
        Bernoulligetallen ($\frac{w}{\eu^w - 1} =
        \sum_n\frac{B_n}{n!}w^n$,
        \cref{pb:b3:holomorphic:1}, Deel~VI): uit
        \[
        \pi z\cot(\pi z) = \iu\pi z +
        \frac{2\iu\pi z}{\eu^{2\iu\pi z} - 1}
        \]
        (bewijs deze identiteit eerst), leid de gesloten
        vorm af
        \[
        \zeta(2k) = (-1)^{k+1}\,
        \frac{(2\pi)^{2k}\,B_{2k}}{2\,(2k)!}
        \qquad (k \geq 1).
        \]
  \item Verifieer de formule tegen $B_2 = \frac16$, $B_4 =
        -\frac1{30}$, $B_6 = \frac1{42}$: herwin $\zeta(2)
        = \frac{\pi^2}6$, $\zeta(4) = \frac{\pi^4}{90}$, en
        bereken $\zeta(6) = \frac{\pi^6}{945}$.
  \item (Eulers recursie) Ontwikkel beide kanten van
        $\bigl(z\cot z\bigr)' = \cot z - z(1 + \cot^2z)$ ---
        of kwadrateer de cotangensreeks direct --- om te
        bewijzen
        \[
        \Bigl(k + \frac12\Bigr)\zeta(2k) =
        \sum_{j=1}^{k-1}\zeta(2j)\,\zeta(2k - 2j)
        \qquad (k \geq 2),
        \]
        en controleer dat zij $\zeta(4)$ berekent uit
        $\zeta(2)$ en $\zeta(6)$ uit $\zeta(2), \zeta(4)$
        --- alle even zetawaarden uit het enkele zaad
        $\frac{\pi^2}6$, zonder nieuwe integratie.
\end{enumerate}
\end{problem}
