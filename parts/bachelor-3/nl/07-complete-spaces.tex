\chapter{Volledige ruimten: Baire, Ascoli, Stone--Weierstrass}
\label{ch:b3:complete}

Volledigheid --- elke Cauchyrij convergeert --- is de
eigenschap die de analyse objecten laat \emph{produceren}:
vaste punten van contracties, sommen van reeksen, oplossingen
van vergelijkingen verkregen als limieten. Dit hoofdstuk
verzamelt de drie grote bestaansmachines van de metrische
theorie. De \emph{stelling van Baire} toont dat een volledige
ruimte geen aftelbare unie van verwaarloosbare stukken kan zijn,
en roept objecten (continue \omterm{thm:b3:complete:nowherediff}{nergens differentieerbare
functies}!) op uit pure kardinaliteitsredenering.
\emph{Arzelà--Ascoli} identificeert de compacte
deelverzamelingen van $\mathcal C(K)$ en is het
compactheidswerkpaard van de analyse --- het weekendprobleem
gebruikt haar om de bestaansstelling van Peano voor
differentiaalvergelijkingen te bewijzen. \emph{Stone--Weierstrass}
toont dat polynomen, en veel meer, \omterm{def:b3:topology:interior}{dicht} liggen in $\mathcal
C(K)$: benadering wordt een algebraïsche verificatie. Onderweg
construeren we \omterm{thm:b3:complete:completion}{vervolledigingen} en bewijzen de
uitbreidingsstelling voor uniform \omterm{def:b3:topology:continuity}{continue afbeeldingen}, het
dagelijks brood van
\cref{ch:b3:lp,ch:b3:hilbert,ch:b3:fouriertransform}.

\section{Volledige ruimten, vervolledigingen, uitbreidingen}

\begin{definition}\label{def:b3:complete:complete}
Een metrische ruimte is \emph{volledig}\index{volledige metrische ruimte}
als elke Cauchyrij convergeert (Jaar~2: $\R^n$ is volledig;
$\mathcal C(\intcc01)$ met $\norm\cdot_\infty$ is volledig). Een
gesloten deelverzameling van een volledige ruimte is volledig;
een volledige deelverzameling van elke metrische ruimte is
gesloten.
\end{definition}

\begin{proof}
Voor de twee uitspraken: een Cauchyrij van de gesloten $F$
convergeert in $X$, en haar limiet, adherent aan $F$, ligt in
$F$; een in-$X$-convergente rij van een volledige $A$ is Cauchy,
convergeert dus in $A$, en limieten zijn uniek.
\end{proof}

\begin{theorem}[Uitbreiding van uniform continue afbeeldingen]
\label{thm:b3:complete:extension}
Zij $D \subseteq X$ \omterm{def:b3:topology:interior}{dicht}, $Y$ \omterm{def:b3:complete:complete}{volledig}, en $f \colon D \to Y$
uniform continu. Dan breidt $f$ \emph{uniek} uit tot een
continue $\bar f \colon X \to Y$, en $\bar f$ is uniform
continu.\index{uitbreidingsstelling (uniforme continuïteit)}
\end{theorem}

\begin{proof}
Uniciteit: twee continue uitbreidingen stemmen overeen op de
dichte $D$, dus overal (de overeenkomstverzameling $\{g = h\}$
is gesloten: origineel van de gesloten diagonaal onder $x\mapsto
(g(x), h(x))$). Bestaan: voor $x \in X$ kies $d_n \to x$, $d_n
\in D$. De rij $(f(d_n))$ is Cauchy: gegeven $\varepsilon$,
uniforme continuïteit levert $\delta$ met $d(u,v) < \delta
\Rightarrow d(f(u), f(v)) < \varepsilon$, en $(d_n)$ is Cauchy.
Definieer $\bar f(x) = \lim f(d_n)$; de limiet hangt niet af van
de gekozen rij (verweef er twee). $\bar f$ breidt $f$ uit
(constante rijen) en erft de continuïteitsmodulus: als $d(x,
x') < \delta$, benader beide door punten van $D$ op afstand $<
\frac{\delta - d(x,x')}2$ en krijg $d(\bar f(x), \bar f(x'))
\leq \varepsilon$ in de limiet --- $\bar f$ is uniform
continu.
\end{proof}

\begin{theorem}[Vervollediging]\label{thm:b3:complete:completion}
Elke metrische ruimte $X$ embedt isometrisch als dichte
deelverzameling van een \omterm{def:b3:complete:complete}{volledige metrische ruimte} $\hat X$,
uniek op isometrie die $X$ puntgewijs fixeert na: haar
\emph{vervollediging}\index{vervollediging van een metrische ruimte}.
\end{theorem}

\begin{proof}
\emph{Bestaan.} Zij $\mathcal X$ de verzameling Cauchyrijen van
$X$, met de pseudo-afstand
\[
D\bigl((x_n), (y_n)\bigr) = \lim_n d(x_n, y_n),
\]
de limiet bestaand omdat $\abs{d(x_n, y_n) - d(x_m, y_m)}
\leq d(x_n, x_m) + d(y_n, y_m)$ de reële rij Cauchy maakt. Zij
$\hat X = \mathcal X/{\sim}$, identificerend rijen op
$D$-afstand $0$; $D$ daalt af tot een afstand. Embed $X$ via
constante rijen: een isometrie, met \omterm{def:b3:topology:interior}{dicht} beeld (een Cauchyrij
wordt $D$-benaderd door de constanten gebouwd op haar eigen
termen: $D\bigl((x_n), (x_k)_{\rm const}\bigr) = \lim_n d(x_n,
x_k) \to 0$ als $k \to \infty$ door Cauchy). Volledigheid van
$\hat X$: zij $(\xi^k)$ Cauchy in $\hat X$; door dichtheid kies
$x_k \in X$ met $D(\xi^k, x_k) \leq 2^{-k}$; dan is $(x_k)$
Cauchy in $X$ (driehoeksongelijkheid via de $\xi$'s), definieert
een punt $\xi \in \hat X$, en $D(\xi^k, \xi) \leq 2^{-k} +
D(x_k, \xi) \to 0$ (de afstand van de constante $x_k$ tot de
klasse van $(x_j)_j$ is $\lim_j d(x_k, x_j)$, klein voor grote
$k$).

\emph{Uniciteit}: twee \omterm{thm:b3:complete:completion}{vervolledigingen} $\hat X_1, \hat X_2$
bevatten $X$ \omterm{def:b3:topology:interior}{dicht}; de identiteit van $X$, een isometrie, is
uniform continu, breidt dus uit tot $\hat X_1 \to \hat X_2$
(\cref{thm:b3:complete:extension}), nog isometrie op een \omterm{def:b3:topology:interior}{dichte
verzameling} dus overal; symmetrisch in de andere richting, en de
samenstellingen fixeren de dichte $X$: ze zijn de identiteiten.
\end{proof}

\begin{theorem}[Vastepuntsstelling van Banach]\label{thm:b3:complete:banach}
Zij $X$ \omterm{def:b3:complete:complete}{volledig}, niet-leeg, en $f \colon X \to X$ een
\emph{contractie}: $d(f(x), f(y)) \leq k\,d(x,y)$ met $k < 1$.
Dan heeft $f$ een uniek vast punt $x^*$, en elke \omterm{def:b3:groups:action}{baan} convergeert
ernaar, met de expliciete snelheid $d(x_n, x^*) \leq
\frac{k^n}{1-k}\,d(x_1, x_0)$.\index{vastepuntsstelling van Banach}
\end{theorem}

\begin{proof}
(Jaar~2 bewees dit; we hernoteren het tweeregelige argument voor
zelfstandigheid.) De \omterm{def:b3:groups:action}{baan} $x_{n+1} = f(x_n)$ heeft $d(x_{n+1},
x_n) \leq k^nd(x_1, x_0)$, is dus Cauchy (meetkundige reeks);
haar limiet $x^*$ is vast (continuïteit van $f$), uniek omdat
twee vaste punten $d \leq k\,d$ voldoen. De snelheid: sommeer de
meetkundige staart.
\end{proof}

\begin{example}[De identiteit verstoren]
\label{ex:b3:complete:perturbident}
Zij $g \colon \R^d \to \R^d$ $k$-Lipschitz met $k < 1$. Dan is
$\varphi = \mathrm{id} + g$ een \emph{\omterm{def:b3:topology:continuity}{homeomorfisme} van $\R^d$
op $\R^d$}. Injectiviteit, met kwantitatieve modulus:
\[
\norm{\varphi(x) - \varphi(y)} \geq \norm{x - y} -
\norm{g(x) - g(y)} \geq (1 - k)\norm{x - y} .
\]
Surjectiviteit is de vastepuntsstelling: $\varphi(x) = y$
oplossen betekent $x = y - g(x)$, en $x \mapsto y - g(x)$ is een
$k$-contractie van de volledige $\R^d$ --- een unieke oplossing
$x = \psi(y)$ bestaat voor elke $y$. De getoonde ongelijkheid
maakt de inverse $\psi$ Lipschitz met constante $\frac1{1-k}$:
een \omterm{def:b3:topology:continuity}{homeomorfisme}, met expliciete grenzen op beide moduli. Deze
onschuldig ogende uitspraak is de motor in de inversefunctiestelling
(\cref{ch:b3:submanifolds}): nabij een punt waar $Df$
inverteerbaar is, \emph{is} $f$ een inverteerbare lineaire
afbeelding plus een kleine Lipschitz-perturbatie, en het
huidige voorbeeld doet de rest. Zij kwantificeert ook numerieke
robustheid: een systeem verstoord met minder dan de marge van de
inverse blijft \omterm{def:b3:groups:derived}{oplosbaar}, met de oplossing die beweegt met
hoogstens $\frac{1}{1-k}$ maal de perturbatie.
\end{example}

\section{De stelling van Baire}

\begin{theorem}[Baire]\label{thm:b3:complete:baire}
In een \omterm{def:b3:complete:complete}{volledige metrische ruimte} is een aftelbare doorsnede van
dichte \omterm{def:b3:topology:topology}{open verzamelingen} \omterm{def:b3:topology:interior}{dicht}. Equivalent: als $X =
\bigcup_{n}F_n$ met elke $F_n$ gesloten, dan heeft enige $F_n$
niet-leeg inwendige.\index{stelling van Baire}
\end{theorem}

\begin{proof}
Zij $(U_n)$ dichte opens en $B_0 = B(x_0, r_0)$ een willekeurige
open bol; we vinden een punt van $\bigcap U_n$ in $B_0$.
Inductief: $U_{n}$ \omterm{def:b3:topology:interior}{dicht} en open, snijdt de open bol $B_{n-1}$
in een \omterm{def:b3:topology:topology}{open verzameling}, die een gesloten bol $\bar B(x_n, r_n)$
bevat met $0 < r_n \leq r_{n-1}/2$ en $\bar B(x_n, r_n)
\subseteq B_{n-1}\cap U_n$. De centra vormen een Cauchyrij
($x_m \in B_n$ voor $m \geq n$, stralen $\to 0$); de limiet $x$
ligt in elke $\bar B(x_n, r_n)$ (geslotenheid), dus in elke $U_n$
en in $B_0$. Voor de tweede vorm: als geen $F_n$ inwendige
heeft, zijn de $U_n = X \setminus F_n$ open en \omterm{def:b3:topology:interior}{dicht}, en een
punt van $\bigcap U_n$ ontsnapt $\bigcup F_n = X$: absurd.
\end{proof}

\begin{remark}\label{rem:b3:complete:meagre}
Vocabulaire: een verzameling is \emph{nergens \omterm{def:b3:topology:interior}{dicht}} als haar
\omterm{def:b3:topology:interior}{afsluiting} leeg inwendige heeft, \emph{mager}\index{magere verzameling}
(eerste categorie) als zij een aftelbare unie van nergens \omterm{def:b3:topology:interior}{dichte
verzamelingen} is. Baire: \emph{een \omterm{def:b3:complete:complete}{volledige metrische ruimte} is
niet \omterm{rem:b3:complete:meagre}{mager} in zichzelf}, en het complement van een \omterm{rem:b3:complete:meagre}{magere
verzameling} is \omterm{def:b3:topology:interior}{dicht}. ``\omterm{rem:b3:complete:meagre}{Mager}'' is een kleinheidsnotie orthogonaal
aan maat (\cref{ch:b3:measure} produceert \omterm{rem:b3:complete:meagre}{magere verzamelingen}
van volle maat), en Baire-argumenten bewijzen \emph{bestaan door
overvloed}: om één object zonder eigenschap P te vertonen, toon
dat de P-objecten een \omterm{rem:b3:complete:meagre}{magere verzameling} vormen.
\end{remark}

\begin{corollary}\label{cor:b3:complete:bairecors}
(a) $\R$ is overaftelbaar. (b) $\Q$ is geen aftelbare doorsnede
van open deelverzamelingen van $\R$, en een niet-lege \omterm{def:b3:complete:complete}{volledige
metrische ruimte} zonder geïsoleerde punten is overaftelbaar.
\end{corollary}

\begin{proof}
(a) $\R = \bigcup_{x}\{x\}$ over een aftelbare verzameling zou
enige singleton inwendige geven. (b) Als $\Q = \bigcap_n V_n$ met
$V_n$ open (noodzakelijk \omterm{def:b3:topology:interior}{dicht}, want $\supseteq \Q$), dan vormen
de $V_n$ en de complementen $\R\setminus\{q\}$, $q \in \Q$, een
aftelbare familie dichte opens met lege doorsnede --- in
tegenspraak met Baire. Als $X$ \omterm{def:b3:complete:complete}{volledig} is zonder geïsoleerde
punten en aftelbaar, vertoont $X = \bigcup_{x \in X}\{x\}$ haar
als aftelbare unie van gesloten verzamelingen met leeg inwendige
(geen geïsoleerde punten): opnieuw Baire.
\end{proof}

\begin{theorem}[De monsters van Weierstrass bestaan]
\label{thm:b3:complete:nowherediff}
Er bestaan continue functies op $\intcc01$ differentieerbaar in
geen enkel punt. Sterker, de verzameling van $f \in \mathcal
C(\intcc01)$ met een (eindige) afgeleide in zelfs één punt is
\omterm{rem:b3:complete:meagre}{mager} in $(\mathcal C(\intcc01), \norm\cdot_\infty)$.\index{nergens differentieerbare functie}
\end{theorem}

\begin{proof}
Voor $n \geq 1$ zij
\[
F_n = \Bigl\{f : \exists x \in \intcc01,\
\forall h \neq 0 \text{ met } x + h \in \intcc01,\
\abs{f(x+h) - f(x)} \leq n\abs h \Bigr\}.
\]
Als $f$ differentieerbaar is in $x$, dan $f \in F_n$ voor enige
$n$: het quotiënt $\abs{f(x+h)-f(x)}/\abs h$ is begrensd voor
$\abs h \leq \delta$ (differentieerbaarheid: het neigt naar
$\abs{f'(x)}$) en begrensd door $2\norm f_\infty/\delta$ voor
$\abs h \geq \delta$. Dus $\bigcup F_n$ bevat alle ergens-
differentieerbare functies, en het volstaat te tonen dat elke
$F_n$ gesloten is met leeg inwendige.

\emph{Gesloten}: zij $f_k \to f$ uniform, $f_k \in F_n$ met
getuigen $x_k \to x$ (compactheid, na extractie). Voor $h$ met
$x + h \in \intcc01$: kies $h_k \to h$ met $x_k + h_k \in
\intcc01$ (bijv.\ $h_k = h + x - x_k$ geknipt); dan
$\abs{f(x + h) - f(x)} = \lim \abs{f_k(x_k + h_k) - f_k(x_k)}
\leq \lim n\abs{h_k} = n\abs h$, gebruikmakend van uniforme
convergentie en continuïteit van $f$ in de relevante punten: $f
\in F_n$.

\emph{Leeg inwendige}: gegeven $f \in F_n$ en $\varepsilon > 0$,
vinden we $g$ met $\norm{g - f}_\infty \leq \varepsilon$ en $g
\notin F_n$. Benader eerst $f$ binnen $\varepsilon/2$ door een
stuksgewijs affiene $\varphi$ (uniforme continuïteit:
interpoleer op een fijn rooster), van hellingen begrensd door
enige $M$. Voeg een kleine zaagtand toe: $g = \varphi +
\frac\varepsilon2\,s_N$, waarbij $s_N(x)$ de $\frac1N$-periodieke
zigzag is van amplitude $1$ en helling $\pm 2N$. In elk $x$ is
er aan één kant $h$ willekeurig klein met de zaagtand die
helling $\pm 2N\cdot\frac\varepsilon2 = \pm\varepsilon N$
bijdraagt over $[x, x+h]$: het differentiequotiënt van $g$
overschrijdt $\varepsilon N - M > n$ voor $N$ groot. Dus $g
\notin F_n$, op uniforme afstand $\leq \varepsilon$ van $f$.
Conclusie: $\bigcup F_n$ is \omterm{rem:b3:complete:meagre}{mager}; door Baire is haar complement
--- gemaakt van \omterm{thm:b3:complete:nowherediff}{nergens differentieerbare functies} --- \omterm{def:b3:topology:interior}{dicht} in
$\mathcal C(\intcc01)$: zulke functies bestaan \emph{in
overvloed}.
\end{proof}

\section{Arzelà--Ascoli}

Doorheen is $K$ een \emph{compacte} metrische ruimte en $\mathcal
C(K) = \mathcal C(K, \R^d)$, met $\norm f_\infty = \sup_K
\norm{f(x)}$: een volledige ruimte (uniforme limieten van
continue functies zijn continu --- Jaar~2).

\begin{definition}\label{def:b3:complete:equicontinuous}
Een familie $\mathcal F \subseteq \mathcal C(K)$ is
\emph{equicontinu}\index{equicontinuïteit} als voor elke
$\varepsilon > 0$ er $\delta > 0$ is zodat
\[
d(x, y) < \delta \implies \norm{f(x) - f(y)} < \varepsilon
\quad\text{voor \emph{alle} } f \in \mathcal F
\]
(één $\delta$ voor de hele familie --- bijv.\ elke familie met
een gemeenschappelijke Lipschitzconstante, of een
gemeenschappelijke Hölder-modulus), en \emph{puntsgewijs
begrensd} als $\sup_{f}\norm{f(x)} < \infty$ voor elke $x$.
\end{definition}

\begin{theorem}[Arzelà--Ascoli]\label{thm:b3:complete:ascoli}
Een deelverzameling $\mathcal F \subseteq \mathcal C(K)$ is
relatief compact (heeft compacte \omterm{def:b3:topology:interior}{afsluiting}) desda \omterm{def:b3:complete:equicontinuous}{equicontinu}
en puntsgewijs begrensd. In het bijzonder heeft elke
equicontinue, puntsgewijs begrensde \emph{rij} een uniform
convergente deelrij.\index{stelling van Arzelà--Ascoli}
\end{theorem}

\begin{proof}
($\Leftarrow$) Zij $(f_k)$ een rij van $\mathcal F$. Compacte
metrische $K$ is \emph{\omterm{def:b3:galois:separable}{separabel}}: voor elke $n$ overdekken
eindig veel bollen van straal $\frac1n$ $K$
(\cref{thm:b3:topology:metriccompact}); hun centra vormen een
aftelbare \omterm{def:b3:topology:interior}{dichte verzameling} $D = \{x_1, x_2, \dots\}$. Door
puntsgewijze begrensdheid en Bolzano--Weierstrass, extraheer
successievelijk deelrijen convergerend in $x_1$, dan ook in
$x_2$, enz., en neem de \emph{diagonaal}deelrij $(g_j)$: zij
convergeert in elk punt van $D$. \omterm{def:b3:complete:equicontinuous}{Equicontinuïteit} tilt dit op
tot uniform Cauchy: gegeven $\varepsilon$, neem $\delta$ zoals
in de definitie, overdek $K$ door eindig veel bollen $B(x_i,
\delta)$ met $x_i \in D$ ($i \leq m$), en kies $J$ zo groot dat
$\norm{g_j(x_i) - g_{j'}(x_i)} < \varepsilon$ voor $j, j' \geq
J$, $i \leq m$. Voor willekeurige $x \in B(x_i, \delta)$:
\[
\norm{g_j(x) - g_{j'}(x)}
\leq \norm{g_j(x) - g_j(x_i)} + \norm{g_j(x_i) - g_{j'}(x_i)}
+ \norm{g_{j'}(x_i) - g_{j'}(x)} < 3\varepsilon .
\]
Dus is $(g_j)$ uniform Cauchy, en convergeert in de volledige
$\mathcal C(K)$. Dus heeft elke rij van $\mathcal F$ een
convergente deelrij: $\bar{\mathcal F}$ is (sequentieel, dus
door \cref{thm:b3:topology:metriccompact}) compact.

($\Rightarrow$) Als $\bar{\mathcal F}$ compact is: puntsgewijze
begrensdheid is duidelijk (evaluatie is continu). Voor
\omterm{def:b3:complete:equicontinuous}{equicontinuïteit}, overdek $\mathcal F$ door eindig veel bollen
$B(f_i, \varepsilon)$ van $\mathcal C(K)$; elke $f_i$ is
uniform continu (Heine,
\cref{cor:b3:topology:heineborel}), wat een gemeenschappelijke
$\delta$ geeft voor $i \leq m$; dan voor $f \in B(f_i,
\varepsilon)$ en $d(x,y) < \delta$: $\norm{f(x)-f(y)} \leq
2\varepsilon + \norm{f_i(x)-f_i(y)} < 3\varepsilon$.
\end{proof}

\begin{example}\label{ex:b3:complete:ascoliexamples}
De gesloten eenheidsbol van $\mathcal C(\intcc01)$ is
\emph{niet} compact ($f_n(x) = x^n$ heeft geen uniform
convergente deelrij: de puntsgewijze limiet is discontinu), en
inderdaad is $(x^n)$ niet \omterm{def:b3:complete:equicontinuous}{equicontinu} in $1$. Daarentegen is
$\{f : \norm f_\infty \leq 1,\ \operatorname{Lip}(f) \leq 1\}$
compact: begrensd en $1$-Lipschitz-equicontinu, en gesloten.
Ascoli verklaart \emph{waarom} compactheid faalt in oneindige
dimensie (Riesz, Jaar~2) en wat toe te voegen om haar te
herstellen: een uniforme continuïteitsmodulus.
\end{example}

\section{Stone--Weierstrass}

\begin{lemma}[Dini]\label{lem:b3:complete:dini}
Zij $K$ compact en $(f_n)$ een \emph{monotone} rij continue
reële functies die \emph{puntsgewijs} convergeert naar een
continue $f$. Dan is de convergentie
uniform.\index{stelling van Dini}
\end{lemma}

\begin{proof}
Zeg $f_n \uparrow f$; zij $g_n = f - f_n \downarrow 0$,
continu. Gegeven $\varepsilon$, de \omterm{def:b3:topology:topology}{open verzamelingen} $U_n =
\{g_n < \varepsilon\}$ stijgen en overdekken $K$ (puntsgewijze
convergentie); extraheer een eindige deeloerdekking: $K =
U_{n_0}$ voor enige $n_0$ (stijgende familie), d.w.z.\ $0 \leq
g_n < \varepsilon$ overal voor $n \geq n_0$.
\end{proof}

\begin{lemma}\label{lem:b3:complete:sqrt}
Er is een rij \emph{polynomen} $u_n$ met $u_n(t) \to \sqrt t$
uniform op $\intcc01$.
\end{lemma}

\begin{proof}
Definieer $u_0 = 0$, $u_{n+1}(t) = u_n(t) + \frac12\bigl(t -
u_n(t)^2\bigr)$: polynomen. Door inductie $0 \leq u_n(t) \leq
\sqrt t$ op $\intcc01$: aannemend voor $n$,
\[
\sqrt t - u_{n+1}(t) = \bigl(\sqrt t - u_n(t)\bigr)
\Bigl(1 - \tfrac12\bigl(\sqrt t + u_n(t)\bigr)\Bigr) \geq 0,
\]
omdat $\sqrt t + u_n \leq 2$; en $u_{n+1} \geq u_n \geq 0$.
Dus is $(u_n(t))$ niet-dalend, begrensd door $\sqrt t$: zij
convergeert puntsgewijs, en de limiet $\ell(t)$ voldoet $\ell =
\ell + \frac12(t - \ell^2)$: $\ell(t) = \sqrt t$, continu.
Dini (\cref{lem:b3:complete:dini}) tilt op tot uniform.
\end{proof}

\begin{theorem}[Stone--Weierstrass, reële versie]
\label{thm:b3:complete:stoneweierstrass}
Zij $K$ een compacte (\omterm{def:b3:topology:hausdorff}{Hausdorff}) ruimte en $\mathcal A \subseteq
\mathcal C(K, \R)$ een \emph{subalgebra} (stabiel onder sommen,
producten, scalaire veelvouden) die de \emph{constanten bevat}
en \emph{punten scheidt} (voor $x \neq y$ heeft enige $f \in
\mathcal A$ $f(x) \neq f(y)$). Dan is $\mathcal A$ \omterm{def:b3:topology:interior}{dicht} in
$(\mathcal C(K,\R), \norm\cdot_\infty)$.\index{stelling van Stone--Weierstrass}
\end{theorem}

\begin{proof}
Zij $\bar{\mathcal A}$ de \omterm{def:b3:topology:interior}{afsluiting}, opnieuw een algebra
(producten van uniforme limieten op begrensde verzamelingen
convergeren).

\emph{Stap 1: $\bar{\mathcal A}$ is een tralie}, d.w.z.\ stabiel
onder $\max$ en $\min$. Omdat $\max(f,g) = \frac{f + g +
\abs{f-g}}2$ en $\min$ evenzo, volstaat dat $f \in
\bar{\mathcal A} \Rightarrow \abs f \in \bar{\mathcal A}$: met
$M = \norm f_\infty > 0$, $\abs f = M\sqrt{(f/M)^2}$, en
\cref{lem:b3:complete:sqrt} geeft polynomen $u_n$ met
$u_n\bigl((f/M)^2\bigr) \to \abs f/M$ uniform; polynomen in
leden van de algebra (met constanteterm: constanten zijn er)
blijven in $\bar{\mathcal A}$.

\emph{Stap 2: tweepuntsinterpolatie.} Voor $x \neq y$ en $a, b
\in \R$ heeft enige $g \in \mathcal A$ $g(x) = a$, $g(y) = b$:
neem $h$ die $x, y$ scheidt en zet $g = a + (b -
a)\frac{h - h(x)}{h(y) - h(x)}$.

\emph{Stap 3.} Zij $f \in \mathcal C(K)$, $\varepsilon > 0$.
Voor elk paar $x, y$ kies $g_{x,y} \in \mathcal A$ met
$g_{x,y}(x) = f(x)$, $g_{x,y}(y) = f(y)$ (Stap 2; voor $x = y$
neem de constante functie $g_{x,x} = f(x)$).
Fixeer $x$: voor elke $y$ bevat de open $V_y = \{g_{x,y} < f +
\varepsilon\}$ $y$; compactheid extraheert $y_1, \dots, y_m$ met
$K = \bigcup V_{y_j}$, en $h_x = \min_j g_{x, y_j} \in
\bar{\mathcal A}$ (Stap 1) voldoet $h_x < f + \varepsilon$
overal, $h_x(x) = f(x)$. Varieer nu $x$: $W_x = \{h_x > f -
\varepsilon\}$ is open, bevat $x$; extraheer $x_1, \dots, x_l$
die $K$ overdekken, en $h = \max_i h_{x_i} \in \bar{\mathcal
A}$ voldoet $f - \varepsilon < h < f + \varepsilon$:
$\norm{f - h}_\infty \leq \varepsilon$. Dus $f \in
\bar{\mathcal A}$.
\end{proof}

\begin{corollary}\label{cor:b3:complete:weierstrass}
(a) (\emph{Weierstrass}) Polynomen liggen \omterm{def:b3:topology:interior}{dicht} in $\mathcal
C(\intcc ab, \R)$; polynomen in $n$ variabelen liggen \omterm{def:b3:topology:interior}{dicht} in
$\mathcal C(K, \R)$ voor compacte $K \subseteq \R^n$.
(b) (\emph{Complexe versie}) Als $\mathcal A \subseteq \mathcal
C(K, \C)$ een subalgebra is die constanten bevat, punten
scheidt, en \emph{stabiel is onder conjugatie}, is zij \omterm{def:b3:topology:interior}{dicht}.
(c) (\emph{Trigonometrische versie}) Trigonometrische polynomen
$\sum_{\abs n \leq N}c_n\eu^{\iu n t}$ liggen \omterm{def:b3:topology:interior}{dicht} in de
ruimte van continue $2\pi$-periodieke functies met
$\norm\cdot_\infty$.\index{approximatiestelling van Weierstrass}
\end{corollary}

\begin{proof}
(a) Polynomen vormen een algebra met constanten; de
coördinaatfuncties scheiden punten van $\R^n$.
(b) De reële en imaginaire delen $\frac{f + \bar f}2$,
$\frac{f - \bar f}{2\iu}$ van leden van $\mathcal A$ vormen
een reële algebra $\mathcal A_\R \subseteq \mathcal C(K, \R)$
met constanten; zij scheidt punten ($f(x) \ne f(y)$ dwingt
$\operatorname{Re}f$ of $\operatorname{Im}f$ te scheiden). Pas
de reële stelling toe en combineer opnieuw.
(c) Bekijk $2\pi$-periodieke functies als $\mathcal C(S^1,
\C)$ ($S^1 = \R/2\pi\Z$, compact:
\cref{exo:b3:topology:5}); de algebra voortgebracht door
$\eu^{\iu t}, \eu^{-\iu t}$ en constanten is stabiel onder
conjugatie en scheidt punten van de cirkel ($\eu^{\iu t}$ is
injectief erop). Pas (b) toe.
\end{proof}

\begin{remark}\label{rem:b3:complete:swremark}
De trigonometrische versie herstelt, en generaliseert sterk, de
kloof gelaten in het Fourier-hoofdstuk van Jaar~2: dichtheid van
trigonometrische polynomen in $(\mathcal C(S^1),
\norm\cdot_2)$ volgt a fortiori ($\norm\cdot_2 \leq
\norm\cdot_\infty$ tot op de normaliserende constante), wat het
Fouriersysteem een orthonormale \emph{basis} maakt in
\cref{ch:b3:hilbert}, en Parseval eindelijk in volle
algemeenheid bewijst.
\end{remark}

\section{Probleem: de bestaansstelling van Peano}

\begin{problem}[{Weekendprobleem --- bestaan van oplossingen van
$x' = f(t, x)$ zonder Lipschitz}]\label{pb:b3:complete:1}
Cauchy--Lipschitz (Jaar~2; herbewijs in \cref{ch:b3:ode}) eist
$f$ Lipschitz in $x$. Peano (1890): \emph{continuïteit van $f$
levert al bestaan} --- zij het geen uniciteit. We bewijzen het
met Euler-veelhoeken en Ascoli. Setting: $f \colon R \to \R^d$
continu op de rechthoek $R = \intcc{t_0 - a}{t_0 + a}\times
\bar B(x_0, b) \subseteq \R\times\R^d$, $M = \sup_R\norm f$, en
\[
T = \min\bigl(a,\ b/M\bigr)
\qquad (\text{als } M = 0 \text{ is het probleem triviaal}).
\]

\medskip
\noindent\textbf{Deel I --- Euler-veelhoeken.} Voor $n \geq 1$,
onderverdeel $[t_0, t_0 + T]$ door $t_k = t_0 + kT/n$ ($0 \leq
k \leq n$) en definieer $\varphi_n$ stuksgewijs affien:
$\varphi_n(t_0) = x_0$ en, op $[t_k, t_{k+1}]$,
\[
\varphi_n(t) = \varphi_n(t_k) + (t -
t_k)\,f\bigl(t_k, \varphi_n(t_k)\bigr).
\]
\begin{enumerate}
  \item Toon door inductie dat $\varphi_n$ welgedefinieerd is,
        met $\norm{\varphi_n(t) - x_0} \leq M(t - t_0) \leq b$
        op $[t_0, t_0 + T]$ --- zodat de evaluatiepunten in $R$
        blijven. \emph{(Hier komt $T \leq b/M$ binnen.)}
  \item Toon dat elke $\varphi_n$ $M$-Lipschitz is.
  \item Leid uit Arzelà--Ascoli
        (\cref{thm:b3:complete:ascoli}) af dat een deelrij
        $(\varphi_{n_j})$ uniform op $[t_0, t_0 + T]$
        convergeert naar enige $\varphi$, zelf $M$-Lipschitz met
        $\varphi(t_0) = x_0$.
\end{enumerate}

\medskip
\noindent\textbf{Deel II --- De limiet lost de vergelijking op.}
Definieer het \emph{defect} $\Delta_n(t) = \varphi_n'(t) -
f\bigl(t, \varphi_n(t)\bigr)$ in niet-roosterpunten.
\begin{enumerate}[resume]
  \item Toon dat $f$ uniform continu is op $R$, en leid af: voor
        elke $\varepsilon > 0$ is er $n_0$ zodat voor $n \geq
        n_0$ en elke niet-rooster $t$, $\norm{\Delta_n(t)} \leq
        \varepsilon$. \emph{(Op $(t_k, t_{k+1})$ is
        $\varphi_n'(t) = f(t_k, \varphi_n(t_k))$, en $(t,
        \varphi_n(t))$ is binnen afstand $(1 + M)\,T/n$ van
        $(t_k, \varphi_n(t_k))$.)}
  \item Stel de integraalvorm vast: voor alle $t$,
        \[
        \varphi_n(t) = x_0 + \int_{t_0}^{t}
        f\bigl(s, \varphi_n(s)\bigr)\dd s +
        \int_{t_0}^t \Delta_n(s)\,\dd s,
        \]
        de middelste integrand stuksgewijs continu.
  \item Ga naar de limiet langs $(n_j)$: toon $f(s,
        \varphi_{n_j}(s)) \to f(s, \varphi(s))$ \emph{uniform}
        (opnieuw uniforme continuïteit van $f$), en concludeer
        \[
        \varphi(t) = x_0 + \int_{t_0}^t
        f\bigl(s, \varphi(s)\bigr)\dd s .
        \]
  \item Leid af dat $\varphi$ $\mathcal C^1$ is op $[t_0, t_0 +
        T]$ en $x' = f(t, x)$, $x(t_0) = x_0$ oplost; breid de
        constructie uit tot $[t_0 - T, t_0]$ (tijdsomkering).
        \emph{Dit is de stelling van Peano.}
\end{enumerate}

\medskip
\noindent\textbf{Deel III --- Uniciteit faalt werkelijk.}
Beschouw $x' = 2\sqrt{\abs x}$, $x(0) = 0$, op $\R$.
\begin{enumerate}[resume]
  \item Controleer dat $f(x) = 2\sqrt{\abs x}$ continu is maar
        niet Lipschitz op enige \omterm{def:b3:topology:topology}{omgeving} van $0$.
  \item Verifieer dat $x \equiv 0$ en, voor elke $c \geq 0$,
        \[
        x_c(t) = \begin{cases} 0 & t \leq c,\\ (t - c)^2 & t
        \geq c,\end{cases}
        \]
        alle oplossingen door $(0,0)$ zijn: een continuum van
        verschillende oplossingen.
  \item Waar breekt het Picard-iteratieargument (vaste punt van
        Banach) voor deze $f$?
\end{enumerate}

\medskip
\noindent\textbf{Deel IV --- Grenzen van de methode.}
\begin{enumerate}[resume]
  \item Toon dat de stelling van Peano faalt in oneindige
        dimensie: we geven toe (of u mag op geloof aannemen) het
        klassieke voorbeeld van Dieudonné in de ruimte $c_0$ van
        nulrijen; bewijs in plaats daarvan het
        eindigdimensionale ingrediënt dat daar faalt: de
        gesloten eenheidsbol van $c_0$ (sup-norm) is niet
        compact --- vertoon een begrensde rij zonder convergente
        deelrij, en leg uit welke stap van Deel I breekt.
  \item Vat samen: welke hypothesen geven bestaan? bestaan en
        uniciteit? Formuleer precies de twee stellingen (Peano;
        Cauchy--Lipschitz) naast elkaar.
\end{enumerate}

\medskip
\noindent\textbf{Deel V --- Osgood: uniciteit voorbij
Lipschitz.} Zij $\omega\colon\intoo0\infty\to\intoo0\infty$
continu, niet-dalend, met
\[
\int_0^1\frac{\dd r}{\omega(r)} = +\infty
\qquad(\text{divergentie in } 0),
\]
en veronderstel dat $f$ voldoet aan $\norm{f(t, x) - f(t, y)}
\leq \omega\bigl(\norm{x - y}\bigr)$ op $R$.
\begin{enumerate}[resume]
  \item Controleer dat $\omega(r) = Lr$ kwalificeert
        (Lipschitz), dat $\omega(r) = r\log\frac1r$ (uitgebreid
        door continuïteit, voor kleine $r$) kwalificeert hoewel
        zij \emph{niet} $O(r)$ is, en dat $\omega(r) =
        2\sqrt r$ dat niet doet. Bereken de integraal in elk
        geval.
  \item Zij $x_1, x_2$ de vergelijking oplossen op $[t_0, t_0 +
        T]$ met $x_1(t_0) = x_2(t_0)$, en $\delta(t) =
        \norm{x_1(t) - x_2(t)}$. Toon, alleen uit de
        integraalvormen, dat voor $t_0 \leq s \leq t$:
        \[
        \delta(t) \leq \delta(s) +
        \int_s^t\omega\bigl(\delta(v)\bigr)\dd v .
        \]
  \item (Stelling van Osgood) Veronderstel $\delta(t_1) > 0$
        voor enige $t_1$, en zij $\tau = \sup\{t \leq t_1 :
        \delta(t) = 0\}$. Voor $s \in \intoo\tau{t_1}$, zet
        $u(t) = \delta(s) + \int_s^t\omega(\delta(v))\,\dd v$.
        Toon $\delta \leq u$, $u' = \omega(\delta) \leq
        \omega(u)$, en leid af
        \[
        \int_{\delta(s)}^{u(t_1)}\frac{\dd r}{\omega(r)}
        \leq t_1 - s \leq t_1 - \tau .
        \]
        Laat $s \downarrow \tau$ en leid een tegenspraak af met
        de divergentie van de integraal. Concludeer:
        \emph{oplossingen door een gemeenschappelijke
        beginvoorwaarde vallen samen} --- uniciteit onder de
        voorwaarde van Osgood.
  \item Trek de gevolgen: Cauchy--Lipschitz-uniciteit is het
        geval $\omega(r) = Lr$; de vergelijking $x' =
        x\log\frac1{\abs x}$ (uitgebreid met $0$ in $0$) heeft
        unieke oplossingen hoewel haar rechterlid niet Lipschitz
        is in $0$; en voor $x' = 2\sqrt{\abs x}$ is de
        convergentie van $\int_0\frac{\dd r}{2\sqrt r}$ precies
        wat een oplossing $0$ in eindige tijd laat verlaten ---
        match de waarde van de integraal $\int_0^{h^2}\frac{\dd
        r}{2\sqrt r} = h$ met het ontsnappingsgedrag van $x_c$.
\end{enumerate}

\medskip
\noindent\textbf{Deel VI --- Snelheden, schema's, trechters.}
\begin{enumerate}[resume]
  \item (Integraal-Grönwall-lemma) Zij $e, \eta \geq 0$ continu
        op $[t_0, t_0+T]$ met $e(t) \leq \int_{t_0}
        ^t\bigl(L\,e(s) + \eta(s)\bigr)\dd s$ voor alle $t$.
        Toon
        \[
        e(t) \leq \frac{\sup\eta}{L}\bigl(\eu^{L(t - t_0)} -
        1\bigr)
        \]
        \emph{(zet $G(t) = \int_{t_0}^t(Le + \eta)$, noteer $G'
        \leq LG + \sup\eta$, en differentieer
        $\eu^{-Lt}G(t)$)}.
  \item (Euler convergeert met een snelheid) Veronderstel nu
        dat $f$ $L$-Lipschitz is in $x$ en $L'$-Lipschitz in $t$
        op $R$. Combinerend de defectgrens van vraag 4
        (kwantitatief gemaakt: $\norm{\Delta_n} \leq (L' +
        LM)T/n$) met vraag 17, bewijs
        \[
        \sup_{[t_0, t_0+T]}\norm{\varphi_n - \varphi}
        \;\leq\; \frac{(L' + LM)\,T}{n}\cdot
        \frac{\eu^{LT} - 1}{L} = O\Bigl(\frac1n\Bigr),
        \]
        waar $\varphi$ \emph{de} oplossing is: met Lipschitz-
        data convergeert de hele rij, met een expliciete
        snelheid --- geen deelrijen nodig. Waarom tilt
        uniciteit deelsequentiele convergentie op tot volle
        convergentie zelfs zonder deze berekening?
  \item (Het schema kiest) Voor $x' = 2\sqrt{\abs x}$, $x(0) =
        0$: toon dat elke Euler-veelhoek identiek nul is, zodat
        het schema convergeert naar de oplossing $x \equiv 0$;
        maar gestart in $x(0) = \varepsilon > 0$ convergeert zij
        (als $n \to \infty$, dan $\varepsilon \to 0$) naar $t
        \mapsto t^2$, een \emph{andere} oplossing door de
        oorsprong. Non-uniciteit herrijst als gevoeligheid van
        het numerieke schema voor perturbaties.
  \item Toon dat de verzameling $\mathcal S$ van \emph{alle}
        oplossingen van $x' = f(t,x)$, $x(t_0) = x_0$ op
        $[t_0, t_0+T]$ (waarden in $\bar B(x_0, b)$) niet-leeg
        is (Deel II), uniform $M$-Lipschitz, en gesloten in
        $\bigl(\mathcal C([t_0, t_0+T], \R^d),
        \norm\cdot_\infty\bigr)$; concludeer met Ascoli dat
        $\mathcal S$ \emph{compact} is. (De stelling van Kneser
        voegt toe dat $\mathcal S$ \omterm{def:b3:topology:connected}{samenhangend} is; we bewijzen
        dat niet.)
  \item Verifieer het fenomeen van Kneser op het voorbeeld: voor
        $x' = 2\sqrt{\abs x}$, $x(0) = 0$, op $[0, 1]$, toon
        $\mathcal S = \{x_c : c \in \intcc0\infty\}$ met
        $x_\infty \equiv 0$ \emph{(voor elke oplossing, zij $c
        = \sup\{t : x(t) = 0\}$ en integreer $(\sqrt x)' = 1$
        op $\{x > 0\}$)}, dat $c \mapsto x_c$ continu is van
        $\intcc0\infty$ (\omterm{def:b3:topology:locallycompact}{éénpuntscompactificatie}, d.w.z.\ $c =
        \infty$ verlijmd als limiet) naar $\mathcal C([0,1])$,
        en concludeer dat $\mathcal S$ inderdaad compact en
        \omterm{def:b3:topology:connected}{samenhangend} is --- een segmentvormige trechter.
  \item (Bereikbare verzamelingen) Leid uit vraag 20 af dat voor
        elke vaste $t$ de \emph{bereikbare verzameling}
        $\mathcal S(t) = \{x(t) : x \in \mathcal S\}$ compact
        is; bereken haar voor het voorbeeld van vraag 21 en
        controleer dat zij ook \omterm{def:b3:topology:connected}{samenhangend} is: $\mathcal S(t)
        = \intcc0{t^2}$ --- elke intermediaire toestand wordt
        door enige oplossing bereikt.
\end{enumerate}

\medskip
\noindent\textbf{Deel VII --- Complementen: afhankelijkheid,
optimaliteit, en een schema met de hand berekend.}
\begin{enumerate}[resume]
  \item (Continue afhankelijkheid) Veronderstel dat $f$
        $L$-Lipschitz is in $x$ op $R$, en zij $x, y$ twee
        oplossingen met beginwaarden $x_0, y_0$ in $t_0$.
        Aanpassend het bewijs van vraag 17 aan de ongelijkheid
        $e(t) \leq \norm{x_0 - y_0} + \int_{t_0}^tL\,e(s)\dd s$,
        toon
        \[
        \norm{x(t) - y(t)} \leq \norm{x_0 - y_0}\,
        \eu^{L(t - t_0)},
        \]
        en controleer op $x' = Lx$ dat de grens bereikt wordt:
        Grönwall is scherp. Leid opnieuw uniciteit af ($x_0 =
        y_0$), en dat de stromingsafbeelding $x_0 \mapsto x(t;
        x_0)$ Lipschitz is, met constante $\eu^{LT}$, waar zij
        gedefinieerd is.
  \item (Osgood is optimaal) Omgekeerd, zij $\omega$ continu,
        niet-dalend, positief op $\intoo0\infty$, met
        $\omega(r) \to 0$ als $r \to 0^+$ maar
        \[
        \int_0^1\frac{\dd r}{\omega(r)} < +\infty,
        \]
        en breid $f(x) = \omega(\abs x)$ uit door $f(0) = 0$.
        Toon dat $\Omega(x) = \int_0^x\frac{\dd r}{\omega(r)}$
        een stijgende bijectie is van $\intoc0{1}$ op
        $\intoc0{\Omega(1)}$, dat haar inverse $g$ $g' =
        \omega(g)$ oplost met $g(0^+) = 0$, en dat $g$,
        uitgebreid met $0$ voor $t \leq 0$, een $\mathcal
        C^1$-oplossing is van $x' = f(x)$ door $(0, 0)$
        verschillend van $x \equiv 0$ \emph{(voor $g'(0) = 0$,
        begrens $\frac{g(t)}t$ door $\omega(g(t))$)}.
        Concludeer: de divergentiehypothese van vraag 15 is geen
        gemak maar de exacte grens van uniciteit; herwin Deel
        III uit $\omega(r) = 2\sqrt r$, $\Omega(x) = \sqrt x$.
  \item (Euler met de hand berekend) Voor $x' = x$, $x(0) = 1$
        op $[0, 1]$: toon dat de Euler-veelhoek voldoet aan
        $\varphi_n(1) = \bigl(1 + \frac1n\bigr)^n$. Bewijs de
        ontwikkeling
        \[
        \Bigl(1 + \frac1n\Bigr)^{n}
        = \eu\Bigl(1 - \frac1{2n} + O\Bigl(\frac1{n^2}\Bigr)
        \Bigr),
        \]
        zodat de fout in $t = 1$ $\frac{\eu}{2n} + O(n^{-2})$
        is: de $O(\frac1n)$-snelheid van vraag 18, met de exacte
        constante. Controleer numeriek voor $n = 10$: $1.1^{10}
        = 2.59374$ tegen $\eu \approx 2.71828$, een fout
        $0.12454$ te vergelijken met $\frac{\eu}{20} \approx
        0.13591$.
\end{enumerate}
\end{problem}
