\chapter{Groepentheorie}\label{ch:b3:groups}

Het Bachelor jaar~2-volume leerde groepen als
boekhoudkundige hulpmiddelen: de stelling van Lagrange,
cyclische groepen, de symmetrische groep en haar signatuur.
Dit hoofdstuk maakt van de groepentheorie een \emph{methode}.
De motor is het begrip van een groep die op een verzameling
\emph{werkt}: het tellen van banen en vaste punten levert de
\omterm{cor:b3:groups:classeq}{klassevergelijking}, de stelling van Cauchy en de drie
Sylowstellingen --- het fundamentele lokaal-naar-globaal-principe
van de eindige groepentheorie. Daarna leren we groepen samen
te stellen (directe en \omterm{def:b3:groups:semidirect}{semidirecte producten}) en te ontleden
(compositiereeksen, \omterm{def:b3:groups:derived}{oplosbare groepen}), en bewijzen we de
stelling die, in \cref{ch:b3:galois}, een drie eeuwen oude
vraag over veeltermvergelijkingen zal afsluiten: de
\omterm{thm:b3:groups:ansimple}{alternerende groep} $A_n$ is \omterm{def:b3:groups:simple}{eenvoudig} voor $n \geq 5$.

\section{Quotiëntgroepen en de isomorfiestellingen}

Overal is $G$ een multiplicatief genoteerde groep, $e$ haar
eenheidselement. Herinner uit het Bachelor jaar~2-volume:
deelgroepen, nevenklassen $gH$, de stelling van Lagrange
($\abs G = [G:H]\,\abs H$ voor eindige $G$), de orde van een
element, cyclische groepen, en de symmetrische groep $S_n$ met
haar signatuurmorfisme $\varepsilon \colon S_n \to \{\pm1\}$.

\begin{definition}\label{def:b3:groups:normal}
Een deelgroep $N$ van $G$ is \emph{normaal}\index{normale
deelgroep} (notatie $N \trianglelefteq G$) wanneer $gNg^{-1} =
N$ voor elke $g \in G$ --- equivalent, wanneer linker- en
rechternevenklassen samenvallen: $gN = Ng$ voor alle $g$.
\end{definition}

\begin{theorem}[Quotiëntgroep]\label{thm:b3:groups:quotient}
Zij $N \trianglelefteq G$. De verzameling $G/N$ van
nevenklassen, met de vermenigvuldiging $(gN)(hN) = ghN$, is
een welgedefinieerde groep, de
\emph{quotiëntgroep}\index{quotiëntgroep}, en de
\emph{canonieke projectie} $\pi \colon G \to G/N$, $g \mapsto
gN$, is een surjectief morfisme met kern $N$. Omgekeerd is
elke kern van een groepsmorfisme \omterm{def:b3:groups:normal}{normaal}: \omterm{def:b3:groups:normal}{normale deelgroepen}
zijn precies de kernen.
\end{theorem}

\begin{proof}
Welgedefinieerdheid is de hele kwestie. Als $gN = g'N$ en $hN
= h'N$, schrijf $g' = gn$, $h' = hm$ met $n, m \in N$. Dan $g'h'
= gnhm = gh\,(h^{-1}nh)\,m \in ghN$ omdat $h^{-1}nh \in N$ door
normaliteit: het product van nevenklassen hangt niet van de
representanten af. Associativiteit, eenheid $eN = N$ en
inversen $(gN)^{-1} = g^{-1}N$ erft men van $G$. Duidelijk is
$\pi$ een surjectief morfisme en $\pi(g) = N \iff g \in N$.

Als $f \colon G \to H$ een morfisme is en $k \in \ker f$, dan
$f(gkg^{-1}) = f(g)f(k)f(g)^{-1} = e$: kernen zijn \omterm{def:b3:groups:normal}{normaal}.
\end{proof}

\begin{theorem}[Universele eigenschap; eerste isomorfiestelling]
\label{thm:b3:groups:firstiso}
Zij $f \colon G \to H$ een morfisme en $N \trianglelefteq G$
met $N \subseteq \ker f$. Er is een uniek morfisme $\bar f
\colon G/N \to H$ met $f = \bar f \circ \pi$. In het bijzonder,
met $N = \ker f$:\index{isomorfiestellingen}
\[
G/\ker f \;\xrightarrow{\;\sim\;}\; \operatorname{im} f,
\qquad gN \mapsto f(g).
\]
\end{theorem}

\begin{proof}
Uniciteit: $\bar f(gN)$ moet $f(g)$ zijn. Bestaan: als $gN =
g'N$ dan $g^{-1}g' \in N \subseteq \ker f$, dus $f(g) = f(g')$
en $\bar f(gN) = f(g)$ is welgedefinieerd; het is een morfisme
omdat $f$ dat is. Voor $N = \ker f$: $\bar f$ is injectief,
want $\bar f(gN) = e$ betekent $g \in \ker f$, d.w.z.\ $gN =
N$; haar beeld is dat van $f$.
\end{proof}

\begin{theorem}[Tweede en derde isomorfiestelling]
\label{thm:b3:groups:secondiso}
Zij $H \leq G$ en $N \trianglelefteq G$.
\begin{enumerate}
  \item $HN = \{hn : h \in H,\, n \in N\}$ is een deelgroep, $N
        \trianglelefteq HN$, $H \cap N \trianglelefteq H$, en
        \[
        H/(H \cap N) \;\cong\; HN/N .
        \]
  \item Als bovendien $N \subseteq K \trianglelefteq G$, dan
        $K/N \trianglelefteq G/N$ en $(G/N)\big/(K/N) \cong G/K$.
\end{enumerate}
\end{theorem}

\begin{proof}
(1) $HN$ is een deelgroep: $(hn)(h'n') = hh'\,(h'^{-1}nh')n' \in
HN$ en $(hn)^{-1} = h^{-1}(hn^{-1}h^{-1}) \in HN$, met
gebruik van normaliteit van $N$. Stel samen $H \hookrightarrow
HN \xrightarrow{\pi} HN/N$: dit morfisme is surjectief ($hnN =
hN$) met kern $\{h \in H : h \in N\} = H \cap N$; pas
\cref{thm:b3:groups:firstiso} toe.

(2) De projectie $G/N \to G/K$, $gN \mapsto gK$, is
welgedefinieerd ($N \subseteq K$), surjectief, met kern $K/N$;
pas opnieuw \cref{thm:b3:groups:firstiso} toe.
\end{proof}

\begin{theorem}[Correspondentietheorema]\label{thm:b3:groups:correspondence}
Zij $N \trianglelefteq G$. De afbeelding $H \mapsto H/N$ is een
bijectie tussen de deelgroepen van $G$ die $N$ bevatten en de
deelgroepen van $G/N$, die inclusies, indices en normaliteit
bewaart (in beide richtingen).
\end{theorem}

\begin{proof}
Haar inverse is $\bar H \mapsto \pi^{-1}(\bar H)$. Beide
afbeeldingen sturen deelgroepen naar deelgroepen, en zijn
elkaars inverse: $\pi^{-1}(H/N) = HN = H$ omdat $N \subseteq
H$, en $\pi(\pi^{-1}(\bar H)) = \bar H$ door surjectiviteit
van $\pi$. Inclusies worden duidelijk bewaard; $[G:H] =
[G/N : H/N]$ omdat $gH \mapsto (gN)(H/N)$ een welgedefinieerde
bijectie tussen nevenklasseruimten is; en $gHg^{-1} = H$ voor
alle $g$ desda $(gN)(H/N)(gN)^{-1} = H/N$ voor alle $gN$,
opnieuw door surjectiviteit van $\pi$.
\end{proof}

\begin{example}\label{ex:b3:groups:firstiso}
$\varepsilon \colon S_n \to \{\pm 1\}$ geeft $S_n/A_n \cong
\{\pm1\}$; $\det \colon GL_n(K) \to K^\times$ geeft
$GL_n(K)/SL_n(K) \cong K^\times$; $t \mapsto \eu^{2\iu\pi t}$
geeft $\R/\Z \cong \mathbb U$, de cirkelgroep. De eerste
isomorfiestelling is hoe quotiënten in de praktijk
\emph{berekend} worden: vind een surjectie met de juiste kern.
\end{example}

\begin{method}\label{met:b3:groups:normal}
Om $N \trianglelefteq G$ te bewijzen, in afnemende elegantie:
vertoon $N$ als kern van een morfisme gedefinieerd op $G$;
controleer $gNg^{-1} \subseteq N$ voor alle $g$ (dit volstaat:
toepassen op $g^{-1}$ en conjugeren geeft de omgekeerde
inclusie); verifieer dat $N$ een unie van \omterm{ex:b3:groups:actions}{conjugatieklassen} is;
of noteer dat $[G:N] = 2$ (dan is $gN = Ng$ gedwongen ---
\cref{exo:b3:groups:1}).
\end{method}

\section{Groepswerkingen}

\begin{definition}\label{def:b3:groups:action}
Een \emph{werking}\index{groepswerking} van $G$ op een
verzameling $X$ is een morfisme $\varphi \colon G \to
\mathfrak{S}(X)$ naar de groep van bijecties van $X$; men
schrijft $g \cdot x$ voor $\varphi(g)(x)$. Equivalent: een
afbeelding $G \times X \to X$ met $e \cdot x = x$ en $g \cdot
(h \cdot x) = (gh) \cdot x$. De \emph{baan}\index{baan} van
$x$ is $\mathcal O_x = \{g \cdot x : g \in G\}$, haar
\emph{stabilisator}\index{stabilisator} is de deelgroep $G_x =
\{g : g\cdot x = x\}$, en $X^G = \{x : \forall g,\ g \cdot x
= x\}$ is de verzameling van \emph{vaste punten}. De werking
is \emph{transitief} als er precies één baan is, \emph{trouw}
als $\varphi$ injectief is, \emph{vrij} als alle
stabilisatoren triviaal zijn.
\end{definition}

\begin{example}\label{ex:b3:groups:actions}
Vijf \omterm{def:b3:groups:action}{werkingen} dragen de hele eindige groepentheorie:
\begin{enumerate}
  \item $G$ op zichzelf door \emph{linkertranslatie} $g \cdot x
        = gx$: vrij en transitief.
  \item $G$ op zichzelf door \emph{conjugatie} $g \cdot x =
        gxg^{-1}$: banen zijn de
        \emph{conjugatieklassen}\index{conjugatieklasse},
        \omterm{def:b3:groups:action}{stabilisatoren} de
        \emph{centralisatoren}\index{centralisator} $Z_G(x) =
        \{g : gx = xg\}$, vaste punten het
        \emph{centrum}\index{centrum} $Z(G)$.
  \item $G$ op de nevenklasseruimte $G/H$ door $g \cdot xH =
        gxH$: transitief, met \omterm{def:b3:groups:action}{stabilisator} van de nevenklasse
        $H$ gelijk aan $H$. Elke transitieve \omterm{def:b3:groups:action}{werking} is van
        deze vorm (\cref{exo:b3:groups:8}).
  \item $G$ op haar verzameling van deelgroepen door conjugatie:
        de \omterm{def:b3:groups:action}{stabilisator} van $H$ is de
        \emph{normalisator}\index{normalisator} $N_G(H) = \{g
        : gHg^{-1} = H\}$, de grootste deelgroep van $G$ waarin
        $H$ \omterm{def:b3:groups:normal}{normaal} is.
  \item $S_n$ op $\intint{1}{n}$: de moeder van alle voorbeelden.
\end{enumerate}
\end{example}

\begin{theorem}[Baan--stabilisator]\label{thm:b3:groups:orbitstab}
De afbeelding $gG_x \mapsto g \cdot x$ is een welgedefinieerde
bijectie $G/G_x \to \mathcal O_x$. In het bijzonder, voor
eindige $G$,\index{baan--stabilisatorstelling}
\[
\abs{\mathcal O_x} = [G : G_x] \quad\text{deelt } \abs G,
\]
en, omdat de banen $X$ partitioneren (ze zijn de klassen van de
equivalentie $x \sim y \iff y \in \mathcal O_x$),
\[
\abs X = \sum_{i} \,[G : G_{x_i}]
\qquad (x_i\colon \text{één punt per baan}).
\]
\end{theorem}

\begin{proof}
Welgedefinieerd en injectief: $gG_x = hG_x \iff h^{-1}g \in
G_x \iff h^{-1}g \cdot x = x \iff g \cdot x = h \cdot x$; lees
de keten in beide richtingen. Surjectiviteit is de definitie
van de \omterm{def:b3:groups:action}{baan}. De teluitspraken volgen uit de stelling van
Lagrange en de partitie van $X$ in banen.
\end{proof}

\begin{corollary}[Klassevergelijking]\label{cor:b3:groups:classeq}
Voor een eindige groep $G$, met één representant $x_i$ in elke
conjugatieklasse met meer dan één element:\index{klassevergelijking}
\[
\abs G = \abs{Z(G)} + \sum_i \,[G : Z_G(x_i)],
\qquad\text{elke } [G : Z_G(x_i)] > 1 \text{ deelt } \abs G.
\]
\end{corollary}

\begin{proof}
Pas \cref{thm:b3:groups:orbitstab} toe op de conjugatiewerking:
singletonbanen zijn precies de elementen van $Z(G)$.
\end{proof}

\begin{theorem}[Vaste punten van $p$-groepen]\label{thm:b3:groups:pfixed}
Zij $p$ een priemgetal. Een \emph{$p$-groep}\index{p-groep@$p$-groep}
is een eindige groep waarvan de orde een macht van $p$ is. Als
een $p$-groep $G$ werkt op een eindige verzameling $X$, dan
\[
\abs{X^G} \equiv \abs X \pmod p .
\]
Gevolgen: een niet-triviale $p$-groep heeft niet-triviaal
\omterm{ex:b3:groups:actions}{centrum}, en elke groep van orde $p^2$ is abels.
\end{theorem}

\begin{proof}
Elke \omterm{def:b3:groups:action}{baan} heeft kardinaliteit $[G:G_x]$, een macht van $p$;
deze macht is $1$ precies op vaste punten en anders deelbaar
door $p$. Sommeren over banen geeft de congruentie. Voor het
\omterm{ex:b3:groups:actions}{centrum}: de conjugatiewerking van $G$ op zichzelf heeft $X^G =
Z(G)$, dus $\abs{Z(G)} \equiv \abs G \equiv 0 \pmod p$, en
$Z(G) \ni e$ dwingt $\abs{Z(G)} \geq p$. Orde $p^2$: als $Z(G)
\neq G$ dan $\abs{Z(G)} = p$ en $G/Z(G)$ is cyclisch van orde
$p$, wat $G$ abels dwingt (\cref{exo:b3:groups:2}) ---
tegenspraak.
\end{proof}

\begin{theorem}[Cauchy]\label{thm:b3:groups:cauchy}
Als een priemgetal $p$ $\abs G$ deelt, dan bevat $G$ een element
van orde $p$.\index{stelling van Cauchy}
\end{theorem}

\begin{proof}[Bewijs (McKay)]
Zij $X = \{(g_1, \dots, g_p) \in G^p : g_1 g_2 \cdots g_p =
e\}$. Vrij kiezen van $g_1, \dots, g_{p-1}$ bepaalt $g_p$:
$\abs X = \abs G^{p-1}$, deelbaar door $p$. De cyclische groep
$\Z/p\Z$ werkt op $X$ door cyclische verschuiving $(g_1, \dots,
g_p) \mapsto (g_2, \dots, g_p, g_1)$ --- dit bewaart $X$, want
$g_2 \cdots g_p g_1 = g_1^{-1}(g_1 \cdots g_p)g_1 = e$. Door
\cref{thm:b3:groups:pfixed}, $\abs{X^{\Z/p\Z}} \equiv \abs X
\equiv 0 \pmod p$. Vaste punten zijn de constante tupels $(g,
\dots, g)$ met $g^p = e$; het tupel $(e, \dots, e)$ is er één
van, dus er zijn er minstens $p$, en daarmee minstens één $g
\neq e$ met $g^p = e$: haar orde is precies $p$.
\end{proof}

\begin{theorem}[Cayley]\label{thm:b3:groups:cayley}
Elke groep van orde $n$ injecteert in $S_n$.\index{stelling van
Cayley}
\end{theorem}

\begin{proof}
Linkertranslatie $\varphi \colon G \to \mathfrak S(G) \cong
S_n$ is een morfisme; $\varphi(g) = \mathrm{id}$ dwingt $g =
ge = e$: het is trouw.
\end{proof}

\begin{method}\label{met:b3:groups:counting}
Het tellen van vaste punten is de universele openingszet van de
eindige groepentheorie. Om te bewijzen dat iets \emph{bestaat}
(een centraal element, een element van orde $p$, een \omterm{def:b3:groups:normal}{normale
deelgroep}, een vast punt), laat een goed gekozen groep werken
op een goed gekozen eindige verzameling, en vergelijk
$\abs{X^G}$ met $\abs X$ modulo $p$, of laat baangroottes de
groeporde delen. De bewijzen van
\cref{thm:b3:groups:pfixed,thm:b3:groups:cauchy} en van alle
drie de Sylowstellingen hieronder zijn vijf variaties op dit
ene idee.
\end{method}

\section{De Sylowstellingen}

De stelling van Lagrange zegt dat de orde van een deelgroep
$\abs G$ deelt; het omgekeerde faalt ($A_4$, van orde $12$,
heeft geen deelgroep van orde $6$ --- \cref{exo:b3:groups:1}).
De Sylowstellingen redden het omgekeerde voor priemmachten, en
hun telclausule is het scherpste algemene gereedschap dat we
hebben om \omterm{def:b3:groups:normal}{normale deelgroepen} te produceren.

\begin{definition}\label{def:b3:groups:sylow}
Schrijf $\abs G = p^a m$ met $p \nmid m$. Een \emph{Sylow
$p$-deelgroep}\index{Sylowdeelgroep} van $G$ is een deelgroep
van orde $p^a$ --- een $p$-deelgroep van de grootst denkbare
orde. Het aantal Sylow $p$-deelgroepen van $G$ noteert men
$n_p$.
\end{definition}

\begin{lemma}\label{lem:b3:groups:binom}
Als $\abs G = p^a m$ met $p \nmid m$, dan $\dbinom{p^a m}{p^a}
\equiv m \pmod p$.
\end{lemma}

\begin{proof}
In $\mathbb F_p[X]$ itereert de beginnelingsdroom $(1+X)^p = 1
+ X^p$ (de coëfficiënten $\binom pk$, $0<k<p$, zijn deelbaar
door $p$: $p$ deelt de teller van $\frac{p!}{k!(p-k)!}$ maar
niet de noemer) tot $(1+X)^{p^a} = 1 + X^{p^a}$, vandaar
\[
(1+X)^{p^a m} = \bigl(1 + X^{p^a}\bigr)^m
= \sum_{k=0}^{m} \binom mk X^{k p^a} \quad\text{in } \mathbb
F_p[X].
\]
Identificeer de coëfficiënt van $X^{p^a}$: links $\binom{p^a
m}{p^a} \bmod p$, rechts $\binom m1 = m$.
\end{proof}

\begin{theorem}[Sylow I: bestaan]\label{thm:b3:groups:sylow1}
Voor elk priemgetal $p$ bestaan Sylow $p$-deelgroepen van $G$.
\end{theorem}

\begin{proof}[Bewijs (Wielandt)]
Zij $\Omega$ de verzameling van \emph{deelverzamelingen} van
$G$ van kardinaliteit $p^a$; $G$ werkt op $\Omega$ door
linkertranslatie $g \cdot S = gS$. Door
\cref{lem:b3:groups:binom}, $\abs\Omega = \binom{p^a m}{p^a}
\equiv m \not\equiv 0 \pmod p$, dus heeft enige \omterm{def:b3:groups:action}{baan} $\mathcal
O_S$ grootte copriem met $p$ (als $p$ elke baangrootte deelde,
zou ze $\abs\Omega$ delen). Zij $H = G_S$ de \omterm{def:b3:groups:action}{stabilisator} van
zo'n $S$. Omdat $[G : H] = \abs{\mathcal O_S}$ copriem is met
$p$ en $p^a \mid \abs G = [G:H]\,\abs H$, krijgt men $p^a \mid
\abs H$. Omgekeerd, fixeer $s \in S$: de afbeelding $H \to S$,
$h \mapsto hs$, is injectief en landt in $S$ omdat $hS = S$;
dus $\abs H \leq \abs S = p^a$. Dus $\abs H = p^a$.
\end{proof}

\begin{theorem}[Sylow II: dominantie en conjugatie]\label{thm:b3:groups:sylow2}
Zij $P$ een Sylow $p$-deelgroep en $Q$ een willekeurige
$p$-deelgroep van $G$. Dan $Q \subseteq gPg^{-1}$ voor enige $g
\in G$. In het bijzonder zijn alle Sylow $p$-deelgroepen
geconjugeerd, en $P \trianglelefteq G \iff n_p = 1$.
\end{theorem}

\begin{proof}
Laat $Q$ werken op de nevenklasseruimte $X = G/P$, van
kardinaliteit $m \not\equiv 0 \pmod p$. Door
\cref{thm:b3:groups:pfixed} toegepast op de $p$-groep $Q$,
$\abs{X^Q} \equiv m \not\equiv 0 \pmod p$: er is een vaste
nevenklasse $gP$, d.w.z.\ $QgP = gP$, d.w.z.\ $g^{-1}Qg
\subseteq P$. Als $Q$ zelf een \omterm{def:b3:groups:sylow}{Sylowdeelgroep} is, maakt
gelijkheid van ordes van $Q \subseteq gPg^{-1}$ een
gelijkheid. Ten slotte $P \trianglelefteq G$ desda haar
conjugaten $\{gPg^{-1}\}$ --- die door het voorgaande
\emph{alle} Sylow $p$-deelgroepen zijn --- reduceren tot
$\{P\}$.
\end{proof}

\begin{theorem}[Sylow III: telling]\label{thm:b3:groups:sylow3}
$n_p \equiv 1 \pmod p$, en $n_p = [G : N_G(P)]$, wat $m$ deelt.
\end{theorem}

\begin{proof}
Zij $\mathrm{Syl}_p$ de verzameling van Sylow $p$-deelgroepen;
$G$ werkt er transitief op door conjugatie
(\cref{thm:b3:groups:sylow2}), met \omterm{def:b3:groups:action}{stabilisator} van $P$ de
\omterm{ex:b3:groups:actions}{normalisator} $N_G(P) \supseteq P$: $n_p = [G : N_G(P)]$, en $m
= [G:P] = [G:N_G(P)]\,[N_G(P):P]$ toont $n_p \mid m$.

Beperk nu de \omterm{def:b3:groups:action}{werking} tot $P$ en tel vaste punten. Als $Q \in
\mathrm{Syl}_p$ vast is onder $P$, dan $P \subseteq N_G(Q)$;
zowel $P$ als $Q$ zijn Sylow $p$-deelgroepen van de groep
$N_G(Q)$, en dus daarin geconjugeerd
(\cref{thm:b3:groups:sylow2} toegepast op $N_G(Q)$); maar $Q
\trianglelefteq N_G(Q)$, dus is $Q$ daar haar enige conjugaat:
$P = Q$. Het enige vaste punt is dus $P$ zelf, en
\cref{thm:b3:groups:pfixed} geeft $n_p = \abs{\mathrm{Syl}_p}
\equiv \abs{\mathrm{Syl}_p^P} = 1 \pmod p$.
\end{proof}

\begin{method}\label{met:b3:groups:sylowmethod}
Om een groep van gegeven orde $n = p^a m$ te analyseren: som de
delers van $m$ op die congruent $1$ mod $p$ zijn --- dit zijn de
kandidaten voor $n_p$. Als de enige kandidaat $1$ is, is de
Sylow $p$-deelgroep \omterm{def:b3:groups:normal}{normaal}. Als $n_p > 1$ gedwongen klein is,
werk door conjugatie op $\mathrm{Syl}_p$ om een morfisme $G \to
S_{n_p}$ met kleine kern te krijgen. En tel elementen:
onderscheiden Sylow $p$-deelgroepen van \emph{priemorde} $p$
snijden triviaal, dus dragen ze $n_p(p-1)$ elementen van orde
precies $p$; overlappende tellingen voor verschillende
priemgetallen dwingen vaak een tegenspraak
(\cref{exo:b3:groups:7}).
\end{method}

\begin{example}\label{ex:b3:groups:pq}
Zij $\abs G = pq$ met $p < q$ priem en $p \nmid q - 1$. Dan
dwingen $n_q \mid p$ en $n_q \equiv 1 \bmod q$ dat $n_q = 1$
(omdat $p < q$); $n_p \mid q$ en $n_p \equiv 1 \bmod p$
dwingen $n_p = 1$ (omdat $q \not\equiv 1 \bmod p$). Zij $P, Q$
de twee normale Sylows: $P \cap Q = \{e\}$ (coprieme ordes),
dus $\abs{PQ} = pq$ (\cref{exo:b3:groups:4}) en $G \cong P
\times Q \cong \Z/p\Z \times \Z/q\Z \cong \Z/pq\Z$ door
\cref{prop:b3:groups:direct} hieronder. Elke groep van orde
$15$, $33$, $35$, \dots\ is cyclisch. Het uitgesloten geval $p
\mid q - 1$ produceert precies één extra groep, niet-abels ---
zie het weekendprobleem (\cref{pb:b3:groups:1}).
\end{example}

\begin{example}[Een volledige Sylowcensus: $S_4$]
\label{ex:b3:groups:s4sylow}
Laten we de methode op $G = S_4$ toepassen, $\abs G = 24 =
2^3\cdot3$. \emph{Sylow $3$:} $n_3 \mid 8$, $n_3 \equiv 1
\bmod 3$, dus $n_3 \in \{1, 4\}$; omdat $\langle(123)\rangle$
en $\langle(124)\rangle$ verschillend zijn, $n_3 = 4$ --- de
vier deelgroepen $\langle(abc)\rangle$, één voor elke
$3$-elementendeelverzameling $\{a, b, c\}$, die de $8$
driecykels verklaren. Door Sylow~II zijn ze geconjugeerd, en
het conjugatiemorfisme $S_4 \to S_{\mathrm{Syl}_3} \cong S_4$
is hier een isomorfisme (haar kern zit in $N =
N_G(\langle(123)\rangle)$ van orde $24/4 = 6$, en een \omterm{def:b3:groups:normal}{normale
deelgroep} van $S_4$ binnen $S_3$-achtige $N$ moet triviaal
zijn: ze zou uit even permutaties bestaan die de vier Sylows
fixeren, en alleen $e$ doet dat). \emph{Sylow $2$:} $n_2 \mid
3$, $n_2 \equiv 1 \bmod 2$: $n_2 \in \{1, 3\}$. De deelgroep $D
= \langle(1234), (13)\rangle$ heeft orde $8$ (een diëder $D_4$:
de symmetrieën van het vierkant met hoekpunten $1, 2, 3, 4$),
is niet \omterm{def:b3:groups:normal}{normaal} ($(12)(1234)(12) = (2134)$ brengt een andere
$4$-cykeldeelgroep voort), dus $n_2 = 3$: de drie kopieën van
$D_4$ corresponderen met de drie manieren om $4$ punten tot
een ``vierkant'' te paren. Let op de moraal van de census:
$\abs{S_4} = 24$ laat ruimte voor het falen van normaliteit
van beide Sylows, en beide falen --- vergelijk orde $12$, waar
de telling één van beide \omterm{def:b3:groups:normal}{normaal} dwingt (Deel~IV van
\cref{pb:b3:groups:1}).
\end{example}

\section{Producten, direct en semidirect}

\begin{proposition}[Een direct product herkennen]\label{prop:b3:groups:direct}
Zij $H, K \trianglelefteq G$ met $H \cap K = \{e\}$ en $HK =
G$. Dan is $(h,k) \mapsto hk$ een isomorfisme $H \times K \to
G$.\index{direct product}
\end{proposition}

\begin{proof}
Voor $h \in H$, $k \in K$ ligt de \omterm{def:b3:groups:derived}{commutator} $hkh^{-1}k^{-1}$
in $K$ (lees hem als $(hkh^{-1})k^{-1}$, met normaliteit van
$K$) en in $H$ (lees hem als $h(kh^{-1}k^{-1})$): hij is $e$,
dus $H$ en $K$ commuteren elementsgewijs en de afbeelding is
een morfisme. Ze is surjectief omdat $HK = G$, en injectief
omdat $hk = e$ geeft $h = k^{-1} \in H \cap K = \{e\}$.
\end{proof}

Normaliteit van \emph{beide} factoren is wat het vaakst faalt:
in $S_3 = \langle (1\,2\,3)\rangle \,\langle(1\,2)\rangle$
snijden beide factoren triviaal en brengen voort, maar $S_3
\not\cong \Z/3\Z \times \Z/2\Z$. Het juiste begrip wanneer
slechts één factor \omterm{def:b3:groups:normal}{normaal} is:

\begin{definition}\label{def:b3:groups:semidirect}
Zij $H$, $K$ groepen en $\varphi \colon K \to
\operatorname{Aut}(H)$ een morfisme. Het \emph{semidirecte
product}\index{semidirect product} $H \rtimes_\varphi K$ is de
verzameling $H \times K$ uitgerust met
\[
(h, k)\,(h', k') = \bigl(h\,\varphi(k)(h'),\; kk'\bigr).
\]
\end{definition}

\begin{proposition}\label{prop:b3:groups:semidirect}
$H \rtimes_\varphi K$ is een groep; $H \times \{e\}$ is een
\omterm{def:b3:groups:normal}{normale deelgroep} isomorf met $H$, $\{e\} \times K$ een
deelgroep isomorf met $K$; ze snijden triviaal en brengen
voort. Omgekeerd, als $G = NK$ met $N \trianglelefteq G$, $K
\leq G$ en $N \cap K = \{e\}$, dan $G \cong N \rtimes_\varphi
K$ voor $\varphi(k) = (n \mapsto knk^{-1})$.
\end{proposition}

\begin{proof}
Directe verificatie: associativiteit reduceert tot
$\varphi(kk') = \varphi(k)\circ\varphi(k')$ en elk
$\varphi(k)$ is een morfisme; de eenheid is $(e,e)$ en
$(h,k)^{-1} = \bigl(\varphi(k^{-1})(h^{-1}), k^{-1}\bigr)$. De
projectie $(h,k) \mapsto k$ is een morfisme op $K$ met kern $H
\times \{e\}$, die dus \omterm{def:b3:groups:normal}{normaal} is. Voor het omgekeerde:
elk $g \in G$ schrijft zich \emph{uniek} als $nk$ met $n \in
N$, $k \in K$ (bestaan: $G = NK$; uniciteit: $nk = n'k'$ geeft
$n'^{-1}n = k'k^{-1} \in N \cap K$), en
\[
(nk)(n'k') = n\,(kn'k^{-1})\;kk'
\]
toont dat $nk \mapsto (n, k)$ de wet van $G$ transporteert
naar die van $N \rtimes_\varphi K$.
\end{proof}

\begin{example}\label{ex:b3:groups:semidirectexamples}
(a) De \emph{diëdergroep}\index{diëdergroep} $D_n$ ($n \geq 3$)
van de $2n$ symmetrieën van een regelmatige $n$-hoek: de
rotaties vormen een \omterm{def:b3:groups:normal}{normale deelgroep} van index $2$, elke
spiegeling brengt een complement voort, en conjugeren van een
rotatie door een spiegeling keert haar om: $D_n \cong \Z/n\Z
\rtimes_\varphi \Z/2\Z$ met $\varphi(1) = (x \mapsto -x)$.
(b) De affiene groep van een lijn, $\{x \mapsto ax + b : a \in
K^\times,\, b \in K\} \cong K \rtimes K^\times$: translaties
\omterm{def:b3:groups:normal}{normaal}, homothetieën een complement.
(c) $S_n \cong A_n \rtimes \Z/2\Z$ (complement: elke
transpositie).
(d) De \omterm{pb:b3:groups:1}{quaternionengroep} $Q_8$ is \emph{geen} \omterm{def:b3:groups:semidirect}{semidirect
product} van eigenlijke deelgroepen: elke niet-triviale
deelgroep bevat $-1$ (\cref{pb:b3:groups:1}), dus snijden geen
twee eigenlijke deelgroepen triviaal.
\end{example}

\section{Oplosbare groepen; eenvoud van \texorpdfstring{$A_n$}{An}}

\begin{definition}\label{def:b3:groups:derived}
De \emph{commutator}\index{commutator} van $x, y \in G$ is
$[x,y] = xyx^{-1}y^{-1}$; de \emph{afgeleide
deelgroep}\index{afgeleide deelgroep} $D(G)$ is de deelgroep
voortgebracht door alle commutatoren. De \emph{afgeleide
reeks} is $D^0(G) = G$, $D^{i+1}(G) = D(D^i(G))$, en $G$ is
\emph{oplosbaar}\index{oplosbare groep} als $D^n(G) = \{e\}$
voor enige $n$.
\end{definition}

\begin{proposition}\label{prop:b3:groups:derived}
$D(G)$ is \omterm{def:b3:groups:normal}{normaal} (zelfs stabiel onder elke automorfisme),
$G/D(G)$ is abels, en voor $N \trianglelefteq G$: $G/N$ abels
$\iff D(G) \subseteq N$. Bovendien is $G$ \omterm{def:b3:groups:derived}{oplosbaar} desda er
een keten $G = G_0 \trianglerighteq G_1 \trianglerighteq
\dots \trianglerighteq G_n = \{e\}$ is met elke $G_{i+1}
\trianglelefteq G_i$ en elk quotiënt $G_i/G_{i+1}$ abels.
Deelgroepen en quotiënten van \omterm{def:b3:groups:derived}{oplosbare groepen} zijn
\omterm{def:b3:groups:derived}{oplosbaar}; omgekeerd, als $N$ en $G/N$ \omterm{def:b3:groups:derived}{oplosbaar} zijn, is $G$
dat ook.
\end{proposition}

\begin{proof}
Een automorfisme $\alpha$ stuurt $[x,y]$ naar $[\alpha x,
\alpha y]$: het permuteert de \omterm{def:b3:groups:derived}{commutatoren}, dus bewaart de
deelgroep die ze voortbrengen; conjugaties zijn
automorfismen, vandaar normaliteit. In $G/D(G)$ is $\bar x\bar
y \bar x^{-1}\bar y^{-1} = \overline{[x,y]} = \bar e$: het
quotiënt is abels. Als $G/N$ abels is, ligt elke $[x,y] \in N$,
dus $D(G) \subseteq N$; omgekeerd, als $D(G) \subseteq N$ dan
is $G/N$, een quotiënt van de abelse $G/D(G)$ door de derde
isomorfiestelling, abels.

Als $G$ \omterm{def:b3:groups:derived}{oplosbaar} is, is de afgeleide reeks zo'n keten.
Omgekeerd, gegeven een keten, $D^i(G) \subseteq G_i$ door
inductie: $G_i/G_{i+1}$ abels geeft $D(G_i) \subseteq
G_{i+1}$, dus $D^{i+1}(G) = D(D^i G) \subseteq D(G_i)
\subseteq G_{i+1}$; vandaar $D^n(G) = \{e\}$.

Overerving: $D^i(H) \subseteq D^i(G)$ voor $H \leq G$
(inductie), en $D^i(G/N) = \pi(D^i(G))$ omdat $\pi$
\omterm{def:b3:groups:derived}{commutatoren} op \omterm{def:b3:groups:derived}{commutatoren} afbeeldt; dit geeft de
uitspraken voor deelgroepen en quotiënten. Uitbreiding: als
$D^m(G/N) = \{e\}$ dan $D^m(G) \subseteq N$, en $D^n(N) =
\{e\}$ geeft $D^{m+n}(G) = D^n(D^m(G)) \subseteq D^n(N) =
\{e\}$.
\end{proof}

\begin{example}\label{ex:b3:groups:solvableexamples}
Abelse groepen zijn \omterm{def:b3:groups:derived}{oplosbaar}. $p$-groepen zijn \omterm{def:b3:groups:derived}{oplosbaar},
door inductie op de orde: $Z(G) \neq \{e\}$ en $G/Z(G)$ is een
kleinere $p$-groep. $S_3$ en $S_4$ zijn \omterm{def:b3:groups:derived}{oplosbaar}: $S_4
\trianglerighteq A_4 \trianglerighteq V \trianglerighteq
\{e\}$, waar $V = \{e, (1\,2)(3\,4), (1\,3)(2\,4),
(1\,4)(2\,3)\}$ de Kleingroep van dubbele transposities is
(\omterm{def:b3:groups:normal}{normaal} in $S_4$: een unie van \omterm{ex:b3:groups:actions}{conjugatieklassen}), met abelse
quotiënten $\Z/2\Z$, $\Z/3\Z$, $V$. In \cref{ch:b3:galois}
zal ``de algemene vergelijking van graad $n$ is \omterm{def:b3:groups:derived}{oplosbaar} door
radicalen'' letterlijk \emph{betekenen} ``$S_n$ is een
\omterm{def:b3:groups:derived}{oplosbare groep}''. Vandaar het belang van de volgende
definitie.
\end{example}

\begin{definition}\label{def:b3:groups:simple}
Een groep $G \neq \{e\}$ is \emph{eenvoudig}\index{eenvoudige
groep} als haar enige \omterm{def:b3:groups:normal}{normale deelgroepen} $\{e\}$ en $G$ zijn.
Een niet-abelse eenvoudige groep is niet \omterm{def:b3:groups:derived}{oplosbaar}: $D(G)
\trianglelefteq G$ is niet $\{e\}$ (anders $G$ abels), dus
$D(G) = G$ en de afgeleide reeks is constant. De abelse
eenvoudige groepen zijn precies de $\Z/p\Z$, $p$ priem (een
abelse groep is eenvoudig desda ze geen eigenlijke
niet-triviale deelgroep heeft, desda ze cyclisch van priemorde
is door Lagrange).
\end{definition}

\begin{lemma}\label{lem:b3:groups:threecycles}
Voor $n \geq 3$ wordt $A_n$ voortgebracht door $3$-cykels; voor
$n \geq 5$ zijn alle $3$-cykels geconjugeerd \emph{in $A_n$}.
\end{lemma}

\begin{proof}
Een element van $A_n$ is een product van een even aantal
transposities; paar ze en gebruik (samenstelling van rechts
naar links)
\[
(a\,b)(c\,d) = (a\,c\,b)(a\,c\,d), \qquad
(a\,b)(b\,c) = (a\,b\,c), \qquad
(a\,b)(a\,b) = e
\]
voor disjuncte, overlappende en gelijke paren respectievelijk:
elk paar transposities is een product van $3$-cykels.

Conjugatie: $\sigma(a\,b\,c)\sigma^{-1} = (\sigma a\; \sigma
b\; \sigma c)$, dus zijn twee $3$-cykels geconjugeerd door
enige $\sigma \in S_n$. Als $\sigma$ oneven is, vervang door
$\sigma' = \sigma (d\,e)$ waar $d, e$ twee punten buiten $\{a,
b, c\}$ zijn --- ze bestaan omdat $n \geq 5$; dan is $\sigma'$
even en $\sigma'(a\,b\,c) \sigma'^{-1} =
\sigma(a\,b\,c)\sigma^{-1}$, omdat $(d\,e)$ commuteert met
$(a\,b\,c)$.
\end{proof}

\begin{theorem}[Eenvoud van de alternerende groep]\label{thm:b3:groups:ansimple}
$A_n$ is \omterm{def:b3:groups:simple}{eenvoudig} voor $n \geq 5$.\index{alternerende groep}
\end{theorem}

\begin{proof}
Zij $N \trianglelefteq A_n$, $N \neq \{e\}$. Door
\cref{lem:b3:groups:threecycles} volstaat te tonen dat $N$
\emph{één} $3$-cykel bevat: normaliteit en conjugatie van
$3$-cykels in $A_n$ stoppen dan alle $3$-cykels in $N$, dus $N
= A_n$.

Voor $\rho \in S_n$ zij $F(\rho) = \{x : \rho(x) \neq x\}$
haar \emph{steun} en $f(\rho) = \abs{F(\rho)}$. Kies $\sigma
\in N \setminus \{e\}$ met $f(\sigma)$ \emph{minimaal}. Merk
op dat een niet-triviale even permutatie $f \geq 3$ heeft, en
dat $f(\sigma) = 4$ onmogelijk is voor $\sigma \in A_n$ tenzij
$\sigma$ een dubbele transpositie is (een $4$-cykel is
oneven). We tonen dat $\sigma$ een $3$-cykel is.

\emph{Geval A: $\sigma$ is een product van disjuncte
transposities}, zeg $\sigma = (a\,b)(c\,d)\cdots$ met
$f(\sigma) \geq 4$. Kies $e' \notin \{a, b, c, d\}$ (mogelijk:
$n \geq 5$), stel $\tau = (c\,d\,e')$ en
\[
\sigma' = \tau\sigma\tau^{-1}\,\sigma^{-1} \in N
\qquad (\tau\sigma\tau^{-1} \in N \text{ door normaliteit}).
\]
Omdat $\sigma\tau^{-1}\sigma^{-1} = (\sigma c\;\sigma
e'\;\sigma d) = (d\;\sigma e'\;c)$ (met $\sigma c = d$,
$\sigma d = c$), krijgt men $\sigma' = (c\,d\,e')(d\;\sigma
e'\;c)$.

Als $\sigma e' = e'$ (wat in het bijzonder geldt wanneer
$f(\sigma) = 4$, d.w.z.\ $\sigma = (a\,b)(c\,d)$): dan
$(d\,e'\,c) = (c\,d\,e')$ en $\sigma' = (c\,d\,e')^2 =
(c\,e'\,d)$, een $3$-cykel in $N$, met $f(\sigma') = 3 < 4
\leq f(\sigma)$ --- in strijd met minimaliteit.

Als $\sigma e' \neq e'$: dan $\sigma e' \notin \{a, b, c, d,
e'\}$ ($\sigma$ wisselt $a,b$ en $c,d$, en $e' \notin
\{a,b,c,d\}$ met $\sigma$ injectief), dus beweegt $\sigma$ de
zes punten $a, b, c, d, e', \sigma e'$: $f(\sigma) \geq 6$.
Anderzijds voldoet $\sigma'$, een product van twee $3$-cykels
met steunen in $\{c, d, e', \sigma e'\}$, aan $f(\sigma')
\leq 4$; en $\sigma' \neq e$, omdat $\sigma'(d) =
\tau\sigma\tau^{-1}(c) = \tau\sigma(e') = \sigma e' \neq d$
($\tau$ fixeert $\sigma e' \notin \{c,d,e'\}$). Dus $\sigma'
\in N \setminus\{e\}$ met $f(\sigma') \leq 4 < f(\sigma)$:
minimaliteit is tegengesproken.

\emph{Geval B: enige cykel van $\sigma$ heeft lengte $\geq
3$}, zeg $\sigma(a) = b$, $\sigma(b) = c$ met $a, b, c$
onderscheiden. Als $\sigma$ precies deze $3$-cykel is, zijn we
klaar. Anders $f(\sigma) \geq 5$ (het geval $f(\sigma) = 4$
met een $\geq3$-cykel is de oneven $4$-cykel, uitgesloten),
dus mogen we $d, e' \in F(\sigma) \setminus \{a, b, c\}$
kiezen. Stel $\tau = (c\,d\,e')$ en $\sigma' =
\tau\sigma\tau^{-1}\sigma^{-1} \in N$. Zoals eerder beweegt
$\sigma' = (c\,d\,e')\,(\sigma c\;\sigma e'\;\sigma d)$
alleen punten van
\[
M = \{c, d, e'\} \cup \{\sigma c, \sigma d, \sigma e'\}
\subseteq F(\sigma)
\]
(beelden van bewogen punten zijn bewogen: $\sigma(x) \ne x$
impliceert $\sigma(\sigma x) \neq \sigma x$, $\sigma$ is
injectief). Ook $b \notin M$: de vijf punten $a, b, c, d, e'$
zijn onderscheiden, dus $b \notin \{c, d, e'\}$; en $b \in
\{\sigma c, \sigma d, \sigma e'\}$ zou $a \in \{c, d, e'\}$
dwingen (pas $\sigma^{-1}$ toe, met $\sigma a = b$), wat vals
is. Dus fixeert $\sigma'$ het punt $b$, terwijl $\sigma$ $b$
beweegt; en $F(\sigma') \subseteq F(\sigma)$. Ten slotte
$\sigma' \neq e$: $\sigma^{-1}(c) = b$, $\tau^{-1}(b) = b$,
$\sigma(b) = c$, $\tau(c) = d$, dus $\sigma'(c) = d \neq c$.
Dus $\sigma' \in N \setminus \{e\}$ met $f(\sigma') \leq
f(\sigma) - 1$, in strijd met minimaliteit.

Beide gevallen onmogelijk zijnde, is $\sigma$ een $3$-cykel.
\end{proof}

\begin{corollary}\label{cor:b3:groups:snnotsolvable}
Voor $n \geq 5$: $A_n$ en $S_n$ zijn niet \omterm{def:b3:groups:derived}{oplosbaar}, en de
enige \omterm{def:b3:groups:normal}{normale deelgroepen} van $S_n$ zijn $\{e\}$, $A_n$ en
$S_n$.
\end{corollary}

\begin{proof}
$A_n$ is niet-abels \omterm{def:b3:groups:simple}{eenvoudig}, dus niet \omterm{def:b3:groups:derived}{oplosbaar}
(\cref{def:b3:groups:simple}); een groep die een niet-oplosbare
deelgroep bevat is niet \omterm{def:b3:groups:derived}{oplosbaar}
(\cref{prop:b3:groups:derived}). Zij $N \trianglelefteq S_n$:
dan is $N \cap A_n \trianglelefteq A_n$ gelijk aan $\{e\}$ of
$A_n$. Als $N \cap A_n = A_n$, dan $A_n \subseteq N$ en $N \in
\{A_n, S_n\}$ door de index. Als $N \cap A_n = \{e\}$, is de
beperking tot $N$ van de projectie $S_n \to S_n/A_n \cong
\Z/2\Z$ injectief, dus $\abs N \leq 2$; als $N = \{e,
\sigma\}$, maakt normaliteit de conjugatieklasse van $\sigma$
gelijk aan $\{\sigma\}$, d.w.z.\ $\sigma \in Z(S_n)$. Maar
$Z(S_n) = \{e\}$ voor $n \geq 3$: als $\sigma \ne e$ $a$ naar
$b \neq a$ stuurt, kies $c \notin \{a, b\}$; dan stuurt
$(b\,c)\sigma(b\,c)^{-1}$ $a$ naar $c \ne b$, dus verschilt
van $\sigma$. Vandaar $N = \{e\}$.
\end{proof}

\begin{theorem}[Jordan--Hölder]\label{thm:b3:groups:jordanholder}
Elke eindige groep $G \neq \{e\}$ laat een
\emph{compositiereeks}\index{compositiereeks} toe
\[
\{e\} = G_0 \trianglelefteq G_1 \trianglelefteq \cdots
\trianglelefteq G_r = G,
\qquad G_{i}/G_{i-1} \text{ eenvoudig},
\]
en de multiset van \emph{compositiefactoren} $G_i/G_{i-1}$,
tot op isomorfie, hangt niet van de gekozen reeks af. Een
eindige groep is \omterm{def:b3:groups:derived}{oplosbaar} desda al haar compositiefactoren
cyclisch van priemorde zijn.\index{stelling van Jordan--Hölder}
\end{theorem}

\begin{proof}
\emph{Bestaan}: inductie op $\abs G$. Als $G$ \omterm{def:b3:groups:simple}{eenvoudig} is,
neem $\{e\} \trianglelefteq G$. Anders kies een maximale
eigenlijke \omterm{def:b3:groups:normal}{normale deelgroep} $N$ (er zijn eindig veel
deelgroepen); $G/N$ is \omterm{def:b3:groups:simple}{eenvoudig} door het
correspondentietheorema (een eigenlijke niet-triviale \omterm{def:b3:groups:normal}{normale
deelgroep} van $G/N$ zou tillen tot een \omterm{def:b3:groups:normal}{normale deelgroep} van
$G$ strikt tussen $N$ en $G$). Voeg $N \trianglelefteq G$ toe
aan een \omterm{thm:b3:groups:jordanholder}{compositiereeks} van $N$.

\emph{Uniciteit}: inductie op $\abs G$, het geval $G$
\omterm{def:b3:groups:simple}{eenvoudig} is duidelijk. Neem twee compositiereeksen, met
voorlaatste termen $M \trianglelefteq G$ en $N
\trianglelefteq G$ (dus $G/M$, $G/N$ zijn \omterm{def:b3:groups:simple}{eenvoudig}). Als $M =
N$, besluit door inductie toegepast op $M$. Anders is $MN$,
\omterm{def:b3:groups:normal}{normaal} in $G$ en strikt groter dan $M$, gelijk aan $G$ ($M$ is
maximaal \omterm{def:b3:groups:normal}{normaal}: elke normale $M \subsetneq L \subsetneq G$
zou afbeelden op een eigenlijke niet-triviale \omterm{def:b3:groups:normal}{normale
deelgroep} van de eenvoudige $G/M$). De tweede
isomorfiestelling geeft
\[
G/M = MN/M \cong N/(M \cap N),
\qquad
G/N = MN/N \cong M/(M \cap N).
\]
Stel $K = M \cap N$ ($\trianglelefteq G$) en fixeer een
\omterm{thm:b3:groups:jordanholder}{compositiereeks} van $K$. Dan draagt $M$ twee
compositiereeksen: haar oorspronkelijke, en de reeks van $K$
gevolgd door $K \trianglelefteq M$ (het quotiënt $M/K \cong
G/N$ is \omterm{def:b3:groups:simple}{eenvoudig}). Door inductie (toegepast op $M$) zijn de
factoren van de oorspronkelijke reeks van $M$
$\{\text{factoren van } K\} \cup \{G/N\}$; evenzo voor $N$.
Dus hebben beide reeksen van $G$ factoren
\[
\{\text{factoren van } K\} \;\cup\; \{\,G/N,\; G/M\,\},
\]
dezelfde multiset.

Oplosbaarheid: als alle factoren $\Z/p_i\Z$ zijn, is de reeks
een keten met abelse quotiënten, dus is $G$ \omterm{def:b3:groups:derived}{oplosbaar}
(\cref{prop:b3:groups:derived}). Omgekeerd is een
compositiefactor van een \omterm{def:b3:groups:derived}{oplosbare groep} \omterm{def:b3:groups:derived}{oplosbaar} (een
quotiënt van een deelgroep) en \omterm{def:b3:groups:simple}{eenvoudig}; een oplosbare
\omterm{def:b3:groups:simple}{eenvoudige groep} is abels ($D(G) \ne G$ dwingt $D(G) =
\{e\}$), dus enige $\Z/p\Z$.
\end{proof}

\begin{remark}\label{rem:b3:groups:jhmeaning}
Jordan--Hölder zegt dat elke eindige groep is opgebouwd uit
\omterm{def:b3:groups:simple}{eenvoudige groepen}, met een welgedefinieerde onderdelenlijst
--- een rekenkunde van groepen waarin de \omterm{def:b3:groups:simple}{eenvoudige groepen} de
priemgetallen zijn, en waarin \emph{hoe de onderdelen gelijmd
worden} (uitbreidingsdata, zoals in het \omterm{def:b3:groups:semidirect}{semidirecte product})
pure vermenigvuldiging vervangt. De classificatie van de
eindige \omterm{def:b3:groups:simple}{eenvoudige groepen} --- de cyclische $\Z/p\Z$, de
alternerende $A_{n \geq 5}$, zestien families van Lie-type, en
$26$ sporadische groepen --- is een van de monumenten van de
twintigste-eeuwse wiskunde; haar bewijs, verspreid over zo'n
tienduizend tijdschriftpagina's, ligt ver buiten deze cursus.
\end{remark}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:b3:groups:1}
(a) Toon dat elke deelgroep van index $2$ \omterm{def:b3:groups:normal}{normaal} is.
(b) Toon dat als $[G : H] = 2$, dan $x^2 \in H$ voor elke $x
\in G$.
(c) Leid af dat $A_4$ geen deelgroep van orde $6$ heeft: het
omgekeerde van Lagrange faalt. \emph{(Tel de kwadraten van
$3$-cykels.)}
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b3:groups:2}
Toon dat als $G/Z(G)$ cyclisch is, $G$ abels is. Leid opnieuw
af dat elke groep van orde $p^2$ abels is, en vertoon, voor
elk priemgetal $p$, een niet-abelse groep van orde $p^3$.
\emph{(Denk aan boven-driehoekige matrices met
eenheidsdiagonaal over $\mathbb F_p$.)}
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b3:groups:3}
(a) Toon dat $\operatorname{Aut}(\Z/n\Z) \cong (\Z/n\Z)^\times$.
(b) Toon dat de inwendige automorfismen $\iota_g \colon x
\mapsto gxg^{-1}$ een \omterm{def:b3:groups:normal}{normale deelgroep}
$\operatorname{Inn}(G) \trianglelefteq \operatorname{Aut}(G)$
vormen, met $\operatorname{Inn}(G) \cong G/Z(G)$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:groups:4}
Zij $H, K$ deelgroepen van een eindige groep $G$.
(a) Bewijs de \emph{productformule} $\abs{HK}\,\abs{H\cap K} =
\abs H\, \abs K$, door de vezels van de afbeelding $H \times K
\to HK$, $(h,k) \mapsto hk$ te tellen.
(b) Toon dat $HK$ een deelgroep is desda $HK = KH$ (automatisch
wanneer één van de twee \omterm{def:b3:groups:normal}{normaal} is).
(c) Als $H, K \trianglelefteq G$ en $H \cap K = \{e\}$, toon
dat $hk = kh$ voor alle $h \in H$, $k \in K$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:groups:5}
(Het tellemma van Burnside) Een eindige groep $G$ werkt op
een eindige verzameling $X$. Toon dat het aantal banen het
\emph{gemiddeld aantal vaste punten} is:
\[
\#\{\text{banen}\} = \frac{1}{\abs G}\sum_{g \in G}
\abs{\operatorname{Fix}(g)},
\qquad \operatorname{Fix}(g) = \{x \in X : g \cdot x = x\},
\]
door de verzameling $\{(g,x) : g\cdot x = x\}$ op twee manieren
te tellen. Toepassing: $\Z/p\Z$ ($p$ priem) werkt door rotatie
op halskettingen van $p$ kralen met $a$ beschikbare kleuren;
leid de kleine stelling van Fermat $a^p \equiv a \pmod p$ af.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b3:groups:6}
Toon met de Sylowstellingen dat elke groep van orde $15$
cyclisch is en dat elke groep van orde $45$ abels is.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:groups:7}
Toon dat geen groep van orde $30$, en geen van orde $56$,
\omterm{def:b3:groups:simple}{eenvoudig} is. \emph{(Voor $30$: als $n_3 \neq 1$ en $n_5 \neq
1$, tel de elementen van ordes $3$ en $5$. Voor $56$: tel de
elementen van orde $7$.)}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:groups:8}
(a) Zij $H \leq G$ van index $n$. Toon dat de \omterm{def:b3:groups:action}{werking} van $G$
op $G/H$ een morfisme $G \to S_n$ oplevert waarvan de kern
$\bigcap_{g \in G} gHg^{-1}$ de grootste \omterm{def:b3:groups:normal}{normale deelgroep} van
$G$ is die in $H$ zit.
(b) Leid af: als $G$ eindig is en $p$ de \emph{kleinste}
priemdeler van $\abs G$ is, is elke deelgroep van index $p$
\omterm{def:b3:groups:normal}{normaal}.
(c) Toon dat elke transitieve \omterm{def:b3:groups:action}{werking} van $G$ op een
verzameling $X$ isomorf is met de \omterm{def:b3:groups:action}{werking} op een
nevenklasseruimte: er is een bijectie $X \to G/G_x$ die
commuteert met de \omterm{def:b3:groups:action}{werkingen}.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:groups:9}
(a) Toon dat $D(G)$ de kleinste \omterm{def:b3:groups:normal}{normale deelgroep} van $G$ met
abels quotiënt is, en dat elk morfisme van $G$ naar een abelse
groep uniek factoreert door de
\emph{abelianisering} $G^{\mathrm{ab}} = G/D(G)$.
(b) Bereken $D(S_n)$ en $S_n^{\mathrm{ab}}$ voor $n \geq 2$,
en $D(Q_8)$ en $Q_8^{\mathrm{ab}}$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:groups:10}
Zij $G$ een $p$-groep en $H \subsetneq G$ een eigenlijke
deelgroep. Toon dat $H \subsetneq N_G(H)$ (``\omterm{ex:b3:groups:actions}{normalisatoren}
groeien''), en leid af dat elke maximale deelgroep van een
$p$-groep \omterm{def:b3:groups:normal}{normaal} is van index $p$. \emph{(Inductie op $\abs
G$, met $Z(G) \neq \{e\}$: behandel apart $Z(G) \subseteq H$
en $Z(G) \not\subseteq H$.)}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b3:groups:11}
(Eenvoud van $A_5$, met de hand)
(a) Toon dat de \omterm{ex:b3:groups:actions}{conjugatieklassen} van $A_5$ kardinaliteiten
$1$, $15$, $20$, $12$, $12$ hebben. Let op de splitsing van de
$S_5$-klasse van $5$-cykels: voor een $5$-cykel $\sigma$,
vergelijk de \omterm{ex:b3:groups:actions}{centralisatoren} van $\sigma$ in $S_5$ en in
$A_5$.
(b) Leid af dat $A_5$ \omterm{def:b3:groups:simple}{eenvoudig} is: een \omterm{def:b3:groups:normal}{normale deelgroep} is
een unie van \omterm{ex:b3:groups:actions}{conjugatieklassen}, bevat $e$, en heeft
kardinaliteit die $60$ deelt.
(c) Toon dat een \omterm{def:b3:groups:simple}{eenvoudige groep} van orde $60$ noodzakelijk
$n_5 = 6$ heeft.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:groups:12}
(\omterm{ex:b3:groups:actions}{Normalisatoren} van \omterm{def:b3:groups:sylow}{Sylowdeelgroepen} zijn zelf-normaliserend)
Zij $P$ een Sylow $p$-deelgroep van een eindige groep $G$ en
$H = N_G(P)$.
(a) Toon dat $P$ de \emph{unieke} Sylow $p$-deelgroep van $H$
is.
(b) Leid af $N_G(H) = H$. \emph{(Voor $g \in N_G(H)$: $gPg^{-1}$
is een Sylow $p$-deelgroep van $H$, dus $gPg^{-1} = P$.)}
(c) Concludeer dat geen Sylownormalisator in een eigenlijke
\omterm{def:b3:groups:normal}{normale deelgroep} van $G$ zit, en dat een maximale deelgroep
die $N_G(P)$ bevat zelf-normaliserend is.
\end{exercise}

\section{Probleem: de groepen van orde ten hoogste 15}

\begin{problem}[{Weekendprobleem --- classificatie van kleine
groepen}]\label{pb:b3:groups:1}
Het doel is een volledige classificatie, met complete
bewijzen, van de groepen van orde $\leq 15$ tot op isomorfie.
Ordes $1, 2, 3, 5, 7, 11, 13$ zijn afgedaan door Lagrange
(cyclisch), en ordes $4$ en $9$ door
\cref{thm:b3:groups:pfixed} plus de analyse hieronder van
$p^2$: over blijven $6, 8, 10, 12, 14, 15$.

\medskip
\noindent\textbf{Deel I --- Gereedschap.}
\begin{enumerate}
  \item Toon dat een groep waarin elk element $x^2 = e$
        vervult abels is; leid af dat zo'n eindige groep orde
        $2^k$ heeft en isomorf is met $(\Z/2\Z)^k$.
        \emph{(Bekijk haar als vectorruimte over $\mathbb
        F_2$.)}
  \item Toon dat een groep van orde $p^2$ isomorf is met
        $\Z/p^2\Z$ of $(\Z/p\Z)^2$. Som de abelse groepen van
        orde $8$ tot op isomorfie op: $\Z/8\Z$, $\Z/4\Z \times
        \Z/2\Z$, $(\Z/2\Z)^3$ --- bewijs dat de lijst volledig
        en irredundant is \emph{zonder} de structuurstelling
        van \cref{ch:b3:modules} (bespreek via de maximale
        orde van een element).
  \item Zij $\varphi, \varphi' \colon K \to
        \operatorname{Aut}(H)$ twee \omterm{def:b3:groups:action}{werkingen}. Toon dat als
        $\varphi' = \varphi \circ \alpha$ met $\alpha \in
        \operatorname{Aut}(K)$, dan $H \rtimes_{\varphi} K
        \cong H \rtimes_{\varphi'} K$.
  \item Bepaal $\operatorname{Aut}(\Z/n\Z)$ voor $n = 3, 4, 5,
        7$ expliciet, en toon $\operatorname{Aut}\bigl((\Z/2\Z)^2
        \bigr) \cong S_3$.
\end{enumerate}

\medskip
\noindent\textbf{Deel II --- Ordes $2p$ ($6$, $10$, $14$) en
$pq$.}
\begin{enumerate}[resume]
  \item Zij $\abs G = 2p$ met $p$ een oneven priemgetal. Toon
        dat $G$ een \omterm{def:b3:groups:normal}{normale deelgroep} $N = \langle r \rangle$
        van orde $p$ heeft en een element $s$ van orde $2$
        buiten $N$.
  \item Leid af $G \cong \Z/p\Z \rtimes_\varphi \Z/2\Z$, waar
        $\varphi(1) \in \operatorname{Aut}(\Z/p\Z)$ een
        involutie is, en concludeer: $G \cong \Z/2p\Z$ of $G
        \cong D_p$; controleer dat deze twee niet isomorf
        zijn. Dit regelt ordes $6$, $10$, $14$.
  \item Algemener, zij $\abs G = pq$ met $p < q$ priem. Toon
        dat als $p \nmid q-1$ dan $G$ cyclisch is
        (\cref{ex:b3:groups:pq}), en dat als $p \mid q - 1$ er
        naast $\Z/pq\Z$ \emph{precies één} niet-abelse groep
        $\Z/q\Z \rtimes \Z/p\Z$ is tot op isomorfie ---
        gebruik vraag 3 en het feit dat
        $\operatorname{Aut}(\Z/q\Z) \cong (\Z/q\Z)^\times$
        cyclisch van orde $q - 1$ is, hier toegelaten en
        bewezen in \cref{ch:b3:galois} (cycliciteit van
        $\mathbb F_q^\times$). Concludeer voor orde $15$.
\end{enumerate}

\medskip
\noindent\textbf{Deel III --- Orde $8$.} Zij $G$ niet-abels
van orde $8$.
\begin{enumerate}[resume]
  \item Toon dat $G$ een element $r$ van orde $4$ heeft
        (gebruik vraag 1) en dat $N = \langle r\rangle$
        \omterm{def:b3:groups:normal}{normaal} is.
  \item Zij $s \notin N$. Toon dat $s^2 \in N$
        (\cref{exo:b3:groups:1}(b)), dat $srs^{-1} = r^{-1}$
        (onderzoek de mogelijke beelden van $r$ onder
        conjugatie, die orde 4 moeten hebben, en sluit
        $srs^{-1} = r$ uit), en dat $s^2 \in \{e, r^2\}$
        \emph{(wat gebeurt als $s^2 = r$ of $r^3$? en waarom
        moet $s^2$ met $s$ commuteren?)}.
  \item In het geval $s^2 = e$, toon $G \cong D_4$.
  \item In het geval $s^2 = r^2$, toon dat de
        vermenigvuldigingstabel volledig bepaald is; de
        resulterende groep is de
        \emph{quaternionengroep}\index{quaternionengroep} $Q_8
        = \{\pm 1, \pm \mathrm i, \pm \mathrm j, \pm \mathrm
        k\}$, $\mathrm i^2 = \mathrm j^2 = \mathrm k^2 =
        \mathrm i\, \mathrm j\,\mathrm k = -1$ (stel $r =
        \mathrm i$, $s = \mathrm j$). Verifieer dat $Q_8$
        bestaat, bijv.\ binnen $GL_2(\C)$ via
        \[
        \mathrm i \mapsto \begin{pmatrix} \iu & 0\\ 0 & -\iu
        \end{pmatrix},
        \qquad
        \mathrm j \mapsto \begin{pmatrix} 0 & 1\\ -1 & 0
        \end{pmatrix}.
        \]
  \item Toon dat elke niet-triviale deelgroep van $Q_8$ $-1$
        bevat; leid af dat elke deelgroep van $Q_8$ \omterm{def:b3:groups:normal}{normaal}
        is, dat $D_4 \not\cong Q_8$ (tel elementen van orde
        $2$), en dat $Q_8$ geen \omterm{def:b3:groups:semidirect}{semidirect product} is van twee
        eigenlijke deelgroepen.
\end{enumerate}

\medskip
\noindent\textbf{Deel IV --- Orde $12$.} Zij $\abs G = 12$,
$P_3 \in \mathrm{Syl}_3(G)$, $P_2 \in \mathrm{Syl}_2(G)$.
\begin{enumerate}[resume]
  \item Toon $n_3 \in \{1, 4\}$, $n_2 \in \{1, 3\}$, en dat
        $n_3 = 4$ dwingt $n_2 = 1$ \emph{(tel elementen van
        orde $3$)}.
  \item Stel $n_3 = 4$. De conjugatiewerking op
        $\mathrm{Syl}_3$ geeft $\rho \colon G \to S_4$. Toon
        dat $\ker \rho$, bevat in elke $N_G(P_3)$ en dus van
        orde die $3$ deelt, triviaal is \emph{(waarom kan ze
        geen orde $3$ hebben?)}; dat het beeld, een deelgroep
        van orde $12$ van $S_4$, noodzakelijk $A_4$ is
        \emph{(deelgroepen van index 2 zijn \omterm{def:b3:groups:normal}{normaal} en
        bevatten alle kwadraten --- \cref{exo:b3:groups:1};
        tel de kwadraten in $S_4$)}; en concludeer $G \cong
        A_4$.
  \item Stel $n_3 = 1$, dus $G \cong \Z/3\Z \rtimes_\varphi
        P_2$ met $\varphi \colon P_2 \to
        \operatorname{Aut}(\Z/3\Z) \cong \Z/2\Z$. Som de
        gevallen op: $\varphi$ triviaal levert $\Z/12\Z$ en
        $\Z/6\Z \times \Z/2\Z$; $P_2 = \Z/4\Z$ met $\varphi$
        surjectief levert de \emph{dicyclische groep}
        $\mathrm{Dic}_3 = \Z/3\Z \rtimes \Z/4\Z$; $P_2 =
        (\Z/2\Z)^2$ met $\varphi$ surjectief levert, tot op de
        equivalentie van vraag 3, een unieke groep --- toon
        dat ze $D_6$ is, bijv.\ door een element van orde $6$
        en een spiegelingsachtige involutie te vertonen.
  \item Toon dat $\Z/12\Z$, $\Z/6\Z \times \Z/2\Z$, $D_6$,
        $A_4$, $\mathrm{Dic}_3$ paarsgewijs niet-isomorf zijn
        \emph{(tel elementen van orde $2$, of gebruik
        $n_3$)}. Dit regelt orde $12$.
\end{enumerate}

\medskip
\noindent\textbf{Deel V --- Synthese.}
\begin{enumerate}[resume]
  \item Stel de classificatietabel samen: voor elke orde $n
        \leq 15$, de volledige lijst van groepen tot op
        isomorfie, met de aantallen $1, 1, 1, 2, 1, 2, 1, 5,
        2, 2, 1, 5, 1, 2, 1$.
\end{enumerate}

\medskip
\noindent\textbf{Deel VI --- Verder: de groepen van orde
$p^3$, $p$ oneven.} De orde-$8$-analyse van Deel III heeft
een fraai oneven-priem-analoog, met één echt nieuw fenomeen.
Zij $p$ een oneven priemgetal en $G$ niet-abels van orde
$p^3$.
\begin{enumerate}[resume]
  \item Toon dat $\abs{Z(G)} = p$, dat $G/Z(G) \cong
        (\Z/p\Z)^2$ \emph{(een cyclisch quotiënt door het
        \omterm{ex:b3:groups:actions}{centrum} dwingt abelheid: \cref{exo:b3:groups:2})}, en
        dat $D(G) = Z(G)$ \emph{(voor $D(G) \subseteq Z(G)$,
        gebruik dat $G/Z(G)$ abels is; voor gelijkheid, $G$ is
        niet-abels en $D(G) \neq \{e\}$)}. Leid af dat elke
        \omterm{def:b3:groups:derived}{commutator} $[x, y] = xyx^{-1}y^{-1}$ centraal is en
        van orde die $p$ deelt.
  \item (De sleutelidentiteit) Zij $x, y \in G$ en $z = [y,
        x]$, centraal. Bewijs door inductie op $k$:
        \[
        (xy)^k = x^k y^k z^{k(k-1)/2} .
        \]
        \emph{(Verplaats elke $y$ langs elke $x$; elke
        oversteek kost één centrale factor $z$.)}
  \item Leid af dat voor $p$ oneven de afbeelding
        $\theta\colon x \mapsto x^p$ een groepsmorfisme van
        $G$ naar $Z(G)$ is \emph{(waarom is $x^p$ centraal?
        waarom heeft $z^{p(p-1)/2} = e$ $p$ oneven nodig?)},
        en concludeer dat $G$ exponent $p$ of $p^2$ heeft, de
        twee gevallen onderscheiden door of $\theta$ triviaal
        is.
  \item (Exponent $p$) Stel dat elk element $x^p = e$
        vervult. Kies $x, y$ waarvan de klassen $G/Z(G)$
        voortbrengen en stel $z = [y, x]$. Toon dat $z \neq
        e$, dat elk element van $G$ uniek $x^ay^bz^c$ is ($0
        \leq a, b, c < p$), en dat de vermenigvuldiging
        volledig bepaald is door de relaties $x^p = y^p =
        z^p = e$, $z$ centraal, $[y, x] = z$. Verifieer dat
        de \emph{Heisenberggroep}
        \[
        H_p = \left\{
        \begin{pmatrix} 1 & a & c\\ 0 & 1 & b\\ 0 & 0 & 1
        \end{pmatrix} : a, b, c \in \mathbb F_p \right\}
        \subseteq GL_3(\mathbb F_p)
        \]
        deze relaties realiseert en exponent $p$ heeft
        (bereken $(I + N)^p$ met $N$ strikt bovendriehoekig,
        met $N^3 = 0$ en $p \geq 3$): elke niet-abelse groep
        van orde $p^3$ en exponent $p$ is isomorf met $H_p$.
  \item (Exponent $p^2$) Stel dat enige $r \in G$ orde $p^2$
        heeft, en stel $N = \langle r\rangle$, \omterm{def:b3:groups:normal}{normaal} (index
        $p$: \cref{exo:b3:groups:10}). Toon dat er $s \notin
        N$ is met $s^p = e$ \emph{(neem enige $t \notin N$;
        corrigeer met vraag 20: $\theta(t) = t^p \in Z(G)
        \subseteq N$ --- rechtvaardig $Z(G) =
        \langle r^p\rangle$ --- en kies $a$ met $s = tr^{a}$
        die $s^p = e$ vervult; waar wordt $p$ oneven
        gebruikt?)}. Toon $srs^{-1} = r^{1+p}$ tot op
        vervanging van $s$ door een macht, en concludeer: er
        is precies \emph{één} niet-abelse groep van orde $p^3$
        en exponent $p^2$, namelijk $\Z/p^2\Z \rtimes_\varphi
        \Z/p\Z$ met $\varphi(1)\colon r \mapsto r^{1+p}$
        \emph{(gebruik vraag 3;
        $\operatorname{Aut}(\Z/p^2\Z)$ is cyclisch van orde
        $p(p-1)$, hier toegelaten, dus heeft ze een unieke
        deelgroep van orde $p$)}.
  \item Concludeer de telling: voor oneven $p$ zijn er precies
        $5$ groepen van orde $p^3$ (drie abels, twee
        niet-abels), net als voor $p = 2$ --- maar de twee
        niet-abelse zijn niet langer $D_4$ en $Q_8$. Wijs
        precies aan waar het oneven-$p$-argument breekt voor
        $p = 2$: in de identiteit van vraag 19 is
        $z^{k(k-1)/2}$ voor $k = p = 2$ gelijk aan $z^1 \neq
        e$, dus is kwadrateren geen morfisme --- en inderdaad
        heeft $Q_8$ een uniek element van orde $2$ terwijl
        exponent-$4$ $D_4$ er vijf heeft.
\end{enumerate}

\medskip
\noindent\textbf{Deel VII --- Complementen.}
\begin{enumerate}[resume]
  \item Voor $p$ oneven, tel de elementen van orde $p$ in elk
        van de twee niet-abelse groepen van orde $p^3$: toon
        dat $H_p$ er precies $p^3 - 1$ heeft, terwijl $M_p =
        \Z/p^2\Z \rtimes \Z/p\Z$ er precies $p^2 - 1$ heeft
        \emph{(gebruik het morfisme $\theta$ van vraag 20:
        identificeer haar beeld, dan de orde van haar kern)}.
        Verifieer numeriek voor $p = 3$: $26$ tegen $8$.
        Verklaar waarom geen kwadrateringsmorfisme-argument
        van dit type $D_4$ van $Q_8$ kan scheiden, en welke
        telling dat wél doet.
  \item Noem een geheel getal $n \geq 1$ \emph{cyclisch} als
        elke groep van orde $n$ cyclisch is. Toon dat als $p^2
        \mid n$ voor enig priemgetal $p$, of als $n$
        priemdelers $p < q$ heeft met $p \mid q - 1$, dan $n$
        niet cyclisch is \emph{(vertoon in elk geval een
        niet-cyclische groep van orde $n$, met Deel II voor
        de tweede)}. Leid af dat $n$ cyclisch dwingt
        $\gcd(n, \varphi(n)) = 1$, waar $\varphi$ de totient
        van Euler is, en controleer tegen de tabel van vraag
        17: onder $n \leq 15$ zijn de ordes die één groep
        dragen precies $n \in \{1, 2, 3, 5, 7, 11, 13, 15\}$,
        precies die met $\gcd(n, \varphi(n)) = 1$.
\end{enumerate}
\end{problem}
