\chapter{Differentiaalvormen en de stelling van Stokes}\label{ch:b3:forms}

Eén stelling van de analyse heeft, over twee eeuwen, alle
andere van haar soort geabsorbeerd: de fundamentele stelling
van de calculus, Green--Riemann (bewezen in het volume
Jaar~2), de divergentiestelling van Gauss, de
rotorstelling van Kelvin--Stokes --- elk zegt dat de
integraal van enige afgeleide over een gebied gelijk is aan
de integraal van het oorspronkelijke object over de
\emph{\omterm{def:b3:forms:boundary}{rand}}. De taal van \emph{\omterm{def:b3:forms:diffform}{differentiaalvormen}} maakt
ze tot één uitspraak, $\int_M\dd\omega = \int_{\partial
M}\omega$, en maakt die uitspraak in één slag bewijsbaar.
Dit hoofdstuk bouwt de taal eerlijk op --- alternerende
multilineaire algebra, de \emph{\omterm{def:b3:forms:d}{uitwendige afgeleide}},
\emph{\omterm{def:b3:forms:pullback}{pullbacks}}, \emph{\omterm{def:b3:forms:orientation}{oriëntatie}}, integratie op de
deelvariëteiten van \cref{ch:b3:submanifolds} --- bewijst de
stelling van Stokes, en cashte de eerste cheques: de
klassieke integraalstellingen, het \emph{\omterm{def:b3:forms:winding}{omwentelingsgetal}}
dat stiekem \cref{ch:b3:residues} aandreef, en, in het
weekendprobleem, de vastpuntstelling van Brouwer. Overal
betekent \emph{glad} $\mathcal C^\infty$; elke afbeelding en
vorm is glad tenzij anders gesteld. Dit kost geen
algemeenheid die op dit niveau de moeite waard is en bevrijdt
de handen.

\section{Alternerende multilineaire algebra}

\begin{definition}\label{def:b3:forms:alternating}
Zij $E$ een reële vectorruimte van dimensie $n$. Een
\emph{$k$-lineaire alternerende vorm}\index{alternerende
vorm} op $E$ is een afbeelding $\alpha\colon E^k \to \R$,
lineair in elke variabele, met $\alpha(v_1, \dots, v_k) = 0$
wanneer twee argumenten gelijk zijn. Hun ruimte wordt
geschreven $\Lambda^k E^*$; bij conventie $\Lambda^0E^* =
\R$. Alternatie dwingt antisymmetrie: twee argumenten
verwisselen verandert het teken (ontwikkel $\alpha(\dots, v
+ w, \dots, v + w, \dots) = 0$), en algemener
$\alpha(v_{\sigma(1)}, \dots, v_{\sigma(k)}) =
\varepsilon(\sigma)\,\alpha(v_1, \dots, v_k)$ voor elke
permutatie $\sigma$.
\end{definition}

\begin{example}
Op $E = \R^n$: $\Lambda^1E^* = E^*$ is de \omterm{def:b3:banach:operator}{duale ruimte}; de
determinant in de canonieke basis is een $n$-lineaire
\omterm{def:b3:forms:alternating}{alternerende vorm}, en \cref{prop:b3:forms:basis} toont dat
zij $\Lambda^nE^*$ opspant --- de diepe reden waarom de
determinant uniek is tot op schaling. Voor $k > n$ is
$\Lambda^kE^* = \{0\}$: $k$ vectoren zijn afhankelijk, en
één langs de anderen ontwikkelen doodt $\alpha$ door
alternatie.
\end{example}

\begin{definition}\label{def:b3:forms:wedge}
Voor $\ell_1, \dots, \ell_k \in E^*$ is hun \emph{uitwendig
product}\index{uitwendig product} de $k$-lineaire
\omterm{def:b3:forms:alternating}{alternerende vorm}
\[
(\ell_1 \wedge \dots \wedge \ell_k)(v_1, \dots, v_k)
= \det\bigl(\ell_i(v_j)\bigr)_{1 \leq i, j \leq k} .
\]
Alternatie en multilineaireit zijn die van de determinant in
haar kolommen.
\end{definition}

\begin{proposition}[Basis van $\Lambda^k$]\label{prop:b3:forms:basis}
Zij $(e_1, \dots, e_n)$ een basis van $E$ met \omterm{def:b3:banach:operator}{duale} basis
$(e_1^*, \dots, e_n^*)$. De vormen
\[
e_I^* = e_{i_1}^* \wedge \dots \wedge e_{i_k}^*,
\qquad I = \{i_1 < \dots < i_k\} \subseteq \{1, \dots, n\},
\]
vormen een basis van $\Lambda^kE^*$; dus $\dim\Lambda^kE^* =
\binom nk$. Expliciet, $\alpha =
\sum_{\abs I = k}\alpha(e_{i_1}, \dots, e_{i_k})\,e_I^*$.
\end{proposition}

\begin{proof}
\emph{Genereren.} Zij $\alpha \in \Lambda^kE^*$ en $\beta =
\sum_I\alpha(e_I)\,e_I^*$, waar $\alpha(e_I)$
$\alpha(e_{i_1}, \dots, e_{i_k})$ afkort. Beide kanten zijn
$k$-lineair en alternerend, dus komen zij overeen zodra zij
overeenkomen op alle $k$-tupels $(e_{j_1}, \dots, e_{j_k})$
met $j_1 < \dots < j_k$ (multilineaireit reduceert tot
tupels van basisvectoren, alternatie tot strikt stijgende).
En $e_I^*(e_{j_1}, \dots, e_{j_k}) = \det(e_{i_r}^*(e_{j_s}))
= \delta_{IJ}$: voor $I = J$ is de matrix de identiteit;
voor $I \neq J$ is enige rij nul. Dus $\beta(e_J) =
\alpha(e_J)$ voor alle $J$: $\beta = \alpha$.
\emph{Vrijheid.} Als $\sum_I c_Ie_I^* = 0$, geeft evalueren
op $(e_{j_1}, \dots, e_{j_k})$ $c_J = 0$.
\end{proof}

\begin{definition}\label{def:b3:forms:wedgegeneral}
Het \omterm{def:b3:forms:wedge}{uitwendige product} strekt zich uit tot een bilineaire
afbeelding $\Lambda^kE^* \times \Lambda^\ell E^* \to
\Lambda^{k+\ell}E^*$, bepaald door bilineaireit en
$(e_I^*) \wedge (e_J^*) = e_I^* \wedge e_J^*$ (concateneer
en herorden; het product is $0$ als $I \cap J \neq
\varnothing$). Het is associatief, en
\emph{gegradeerd-anticommutatief}:
\[
\beta \wedge \alpha = (-1)^{k\ell}\,\alpha \wedge \beta
\qquad (\alpha \in \Lambda^k, \ \beta \in \Lambda^\ell).
\]
\end{definition}

\begin{proof}[\omnameProof]
Beide eigenschappen worden gecontroleerd op basiselementen
en uitgebreid door bilineaireit. Associativiteit: beide
haakjes van $e_I^* \wedge e_J^* \wedge e_K^*$ zijn gelijk
aan de wedge van de geconcateneerde familie van $1$-vormen,
via de determinantformule van \cref{def:b3:forms:wedge}
(Laplace-ontwikkeling in blokken). De tekenregel: elk van de
$\ell$ factoren van $\beta$ voorbij de $k$ factoren van
$\alpha$ bewegen kost een teken per aangrenzende
transpositie (een verwisseling van twee rijen van de
determinant), dus in totaal $(-1)^{k\ell}$.
\end{proof}

\begin{proposition}[Pullback, lineair geval]
\label{prop:b3:forms:linearpullback}
Een lineaire afbeelding $u\colon E \to F$ induceert, voor
elke $k$, de lineaire afbeelding $u^*\colon \Lambda^kF^* \to
\Lambda^kE^*$, $(u^*\alpha)(v_1, \dots, v_k) =
\alpha(u(v_1), \dots, u(v_k))$. Zij voldoet aan
$u^*(\alpha \wedge \beta) = u^*\alpha \wedge u^*\beta$ en
$(u \circ w)^* = w^* \circ u^*$. Bovendien:
\begin{enumerate}
  \item Als $\dim E = n$ en $u\colon E \to E$, dan op de
        lijn $\Lambda^nE^*$: $u^*\alpha = (\det u)\,\alpha$.
  \item Als $\operatorname{rk}u < k$, dan $u^* = 0$ op
        $\Lambda^kF^*$.
\end{enumerate}
\end{proposition}

\begin{proof}
De functoriële identiteiten volgen onmiddellijk uit de
definities (voor de productregel, controleer op wedges van
$1$-vormen met de determinantformule --- $\det(\ell_i(uv_j))
= \det((u^*\ell_i)(v_j))$ --- en breid bilineair uit).
(1) $u^*$ beeldt de eendimensionale $\Lambda^nE^*$
(\cref{prop:b3:forms:basis}) op zichzelf af, dus $u^*\alpha
= c\,\alpha$ met $c$ onafhankelijk van $\alpha \neq 0$;
testen op $\alpha = e_1^* \wedge \dots \wedge e_n^*$ en
$(v_j) = (e_j)$ geeft $c = \det(e_i^*(ue_j)) = \det u$.
(2) Voor $v_1, \dots, v_k \in E$ liggen de vectoren
$u(v_1), \dots, u(v_k)$ in het beeld van $u$, van dimensie
$< k$: zij zijn lineair afhankelijk, en een \omterm{def:b3:forms:alternating}{alternerende
vorm} verdwijnt op een afhankelijke familie (ontwikkel de
afhankelijke vector langs de anderen).
\end{proof}

\section{Differentiaalvormen en de uitwendige afgeleide}

\begin{definition}\label{def:b3:forms:diffform}
Zij $U \subseteq \R^n$ open. Een \emph{differentiële
$k$-vorm}\index{differentiaalvorm} op $U$ is een gladde
afbeelding $\omega\colon U \to \Lambda^k(\R^n)^*$; in de
basis van \cref{prop:b3:forms:basis} (met $\dd x_i$
geschreven voor $e_i^*$),
\[
\omega = \sum_{\abs I = k} a_I\,\dd x_I,
\qquad
\dd x_I = \dd x_{i_1} \wedge \dots \wedge \dd x_{i_k},
\]
met gladde coëfficiënten $a_I \in \mathcal C^\infty(U)$.
Hun ruimte is $\Omega^k(U)$; $\Omega^0(U) = \mathcal
C^\infty(U)$. Een $0$-vorm is een functie; een $1$-vorm is
een veld van lineaire vormen (b.v.\ de differentiaal $\dd f$
van een functie); een $n$-vorm is $a\,\dd
x_1\wedge\dots\wedge\dd x_n$, de natuurlijke integrand van
\cref{ch:b3:product}.
\end{definition}

\begin{definition}[Uitwendige afgeleide]\label{def:b3:forms:d}
De \emph{uitwendige afgeleide}\index{uitwendige afgeleide}
is de lineaire afbeelding $\dd\colon \Omega^k(U) \to
\Omega^{k+1}(U)$ gedefinieerd door
\[
\dd\Bigl(\sum_I a_I\,\dd x_I\Bigr)
= \sum_I \dd a_I \wedge \dd x_I
= \sum_I\sum_{j=1}^n \frac{\partial a_I}{\partial
x_j}\,\dd x_j \wedge \dd x_I .
\]
Op $0$-vormen is zij de gewone differentiaal.
\end{definition}

\begin{theorem}\label{thm:b3:forms:dsquare}
(a) $\dd(\omega \wedge \eta) = \dd\omega \wedge \eta +
(-1)^k\,\omega \wedge \dd\eta$ voor $\omega \in \Omega^k$
(de \emph{gegradeerde Leibniz-regel}).
(b) $\dd \circ \dd = 0$.\index{d squared zero@$\dd^2 = 0$}
\end{theorem}

\begin{proof}
(a) Door bilineaireit volstaat $\omega = a\,\dd x_I$, $\eta
= b\,\dd x_J$. Dan $\omega \wedge \eta = ab\,\dd x_I \wedge
\dd x_J$ en
\[
\dd(\omega\wedge\eta) = (b\,\dd a + a\,\dd b) \wedge \dd x_I
\wedge \dd x_J
= (\dd a \wedge \dd x_I) \wedge (b\,\dd x_J) +
a\,\dd b \wedge \dd x_I \wedge \dd x_J,
\]
en de $1$-vorm $\dd b$ voorbij de $k$ factoren van $\dd
x_I$ bewegen kost $(-1)^k$
(\cref{def:b3:forms:wedgegeneral}): de tweede term is
$(-1)^k\,\omega \wedge \dd\eta$.
(b) Voor een functie: $\dd(\dd a) = \sum_{i,j}
\frac{\partial^2a}{\partial x_j\partial x_i}\,\dd x_j \wedge
\dd x_i$. De coëfficiënt is symmetrisch in $(i,j)$ door de
stelling van Schwarz over gemengde partiëlen (bewezen in
het volume Jaar~2; $a$ is $\mathcal C^\infty$), terwijl $\dd
x_j \wedge \dd x_i$ antisymmetrisch is: de termen $(i,j)$
en $(j,i)$ paren, alles heft op. Voor een algemene $\omega
= \sum a_I\dd x_I$: $\dd\dd\omega = \sum\dd(\dd a_I \wedge
\dd x_I) = \sum(\dd\dd a_I\wedge\dd x_I - \dd
a_I\wedge\dd(\dd x_I))$ via (a), en beide termen verdwijnen
($\dd(\dd x_I) = 0$ omdat de coëfficiënt constant is).
\end{proof}

\begin{definition}[Pullback]\label{def:b3:forms:pullback}
Zij $\varphi\colon U \to V$ glad ($U \subseteq \R^m$, $V
\subseteq \R^n$ open). De \emph{pullback}\index{pullback}
$\varphi^*\colon \Omega^k(V) \to \Omega^k(U)$ is puntsgewijs
gedefinieerd door de lineaire pullback langs de
differentiaal: $(\varphi^*\omega)_x =
(D\varphi(x))^*\,\omega_{\varphi(x)}$. Concreet
substitueert $\varphi^*$: $\varphi^*f = f \circ \varphi$
op functies, $\varphi^*(\dd y_i) = \dd\varphi_i =
\sum_j\frac{\partial\varphi_i}{\partial x_j}\dd x_j$, en
$\varphi^*(a\,\dd y_{i_1}\wedge\dots\wedge \dd y_{i_k}) =
(a\circ\varphi)\,\dd\varphi_{i_1}\wedge\dots\wedge
\dd\varphi_{i_k}$.
\end{definition}

\begin{theorem}\label{thm:b3:forms:pullback}
(a) $\varphi^*(\omega \wedge \eta) = \varphi^*\omega \wedge
\varphi^*\eta$ en $(\psi \circ \varphi)^* = \varphi^* \circ
\psi^*$.
(b) $\varphi^*(\dd\omega) = \dd(\varphi^*\omega)$: de
\omterm{def:b3:forms:d}{uitwendige afgeleide} commuteert met elke gladde substitutie
--- de identiteit die haar tot \emph{de} afgeleide van de
theorie maakt.
(c) Als $\varphi\colon U \to V$ glad is tussen openen van
$\R^n$ en $\omega = a\,\dd y_1 \wedge \dots \wedge \dd
y_n$, dan $\varphi^*\omega = (a \circ
\varphi)\,\det\bigl(D\varphi\bigr)\,\dd x_1 \wedge \dots
\wedge \dd x_n$.
\end{theorem}

\begin{proof}
(a) Puntsgewijze uitspraken over lineaire \omterm{def:b3:forms:pullback}{pullbacks}
(\cref{prop:b3:forms:linearpullback}), plus de
kettingregel $D(\psi\circ\varphi)(x) =
D\psi(\varphi(x))\,D\varphi(x)$.
(b) Voor een $0$-vorm $f$: $\varphi^*(\dd f) = \dd f \circ
D\varphi = \dd(f \circ \varphi)$ is de kettingregel. Voor
$\omega = a\,\dd y_I$: met (a), $\varphi^*\omega =
(a\circ\varphi)\,\dd\varphi_{i_1}\wedge\dots\wedge
\dd\varphi_{i_k}$, dus door de Leibniz-regel
(\cref{thm:b3:forms:dsquare}(a)) en $\dd\dd\varphi_{i_r} =
0$:
\[
\dd(\varphi^*\omega) = \dd(a\circ\varphi) \wedge
\dd\varphi_{i_1}\wedge\dots\wedge\dd\varphi_{i_k}
= \varphi^*(\dd a) \wedge \varphi^*(\dd y_I)
= \varphi^*(\dd a \wedge \dd y_I) =
\varphi^*(\dd\omega).
\]
(c) Puntsgewijs is dit precies $u^*\alpha = (\det u)\alpha$
op topgraadvormen (\cref{prop:b3:forms:linearpullback}(1)
met $u = D\varphi(x)$).
\end{proof}

\begin{example}[Poolcoördinaten]\label{ex:b3:forms:polar}
Voor $\varphi(r, \theta) = (r\cos\theta, r\sin\theta)$:
$\varphi^*\dd x = \cos\theta\,\dd r -
r\sin\theta\,\dd\theta$, $\varphi^*\dd y = \sin\theta\,\dd r
+ r\cos\theta\,\dd\theta$, dus
\[
\varphi^*(\dd x \wedge \dd y) =
(\cos\theta\,\dd r - r\sin\theta\,\dd\theta) \wedge
(\sin\theta\,\dd r + r\cos\theta\,\dd\theta)
= r\,\dd r \wedge \dd\theta,
\]
de Jacobiaan van \cref{ex:b3:product:polar} die door pure
algebra verschijnt --- geen maattheorie.
\cref{exo:b3:forms:9} maakt van deze opmerking een
uitspraak: voor georiënteerde integralen \emph{is} de
variabelensubstitutieformule de pullback-formule.
\end{example}

\section{Gesloten en exacte vormen; het lemma van Poincaré}

\begin{definition}\label{def:b3:forms:closedexact}
$\omega \in \Omega^k(U)$ is \emph{gesloten}\index{gesloten
vorm} als $\dd\omega = 0$, \emph{exact}\index{exacte vorm}
als $\omega = \dd\eta$ voor enige $\eta \in
\Omega^{k-1}(U)$ (een \emph{primitieve} van $\omega$). Exact
$\Rightarrow$ gesloten door $\dd^2 = 0$; het omgekeerde is
een vraag over de \emph{vorm} van $U$.
\end{definition}

\begin{example}[De hoekvorm]\label{ex:b3:forms:angular}
Op $U = \R^2 \setminus \{0\}$,
\[
\omega_\theta = \frac{x\,\dd y - y\,\dd x}{x^2 + y^2}
\]
is \omterm{def:b3:forms:closedexact}{gesloten} (rechtstreekse berekening:
\cref{exo:b3:forms:4}) maar niet \omterm{def:b3:forms:closedexact}{exact}: haar integraal langs
de eenheidscirkel is $2\pi \neq 0$, terwijl integralen van
\omterm{def:b3:forms:closedexact}{exacte vormen} langs \omterm{def:b3:forms:closedexact}{gesloten} krommen verdwijnen
(\cref{prop:b3:forms:exactloop}). Lokaal is $\omega_\theta
= \dd\theta$ voor elke gladde bepaling $\theta$ van de
poolhoek --- vandaar de naam en de obstructie: geen zo'n
bepaling bestaat op heel $U$. Deze ene vorm drijft het
\omterm{def:b3:forms:winding}{omwentelingsgetal} (\cref{sec:b3:forms:winding}) en, daardoor,
de residustelling van \cref{ch:b3:residues}.
\end{example}

\begin{theorem}[Lemma van Poincaré]\label{thm:b3:forms:poincare}
Zij $U \subseteq \R^n$ open en \emph{stervormig} ten
opzichte van $0$. Elke \omterm{def:b3:forms:closedexact}{gesloten} $k$-vorm op $U$ ($k \geq 1$)
is \omterm{def:b3:forms:closedexact}{exact}.\index{lemma van Poincare}
\end{theorem}

\begin{proof}
We bouwen een lineaire \emph{homotopie-operator} $h\colon
\Omega^k(U) \to \Omega^{k-1}(U)$ met
\begin{equation}\label{eq:b3:forms:homotopy}
\dd(h\omega) + h(\dd\omega) = \omega
\qquad (k \geq 1);
\end{equation}
als $\dd\omega = 0$, dan $\omega = \dd(h\omega)$ en we zijn
klaar. Voor $\omega = \sum_I a_I\,\dd x_I$ stel
\[
h\omega = \sum_{\abs I = k}\ \sum_{r=1}^{k}(-1)^{r-1}
\Bigl(\int_0^1 t^{k-1}a_I(tx)\,\dd t\Bigr)\,
x_{i_r}\,\dd x_{i_1}\wedge\dots\wedge
\widehat{\dd x_{i_r}}\wedge\dots\wedge\dd x_{i_k}
\]
(de hat verwijdert een factor; de integralen zijn glad in
$x$ door differentiatie onder de integraal,
\cref{thm:b3:lebesgue:paramdiff}, alle afgeleiden
gedomineerd op compacta). Controleren van
\eqref{eq:b3:forms:homotopy} is een berekening die men één
maal in een leven doet, dus doen we haar \omterm{def:b3:complete:complete}{volledig}. Fixeer
$I$ en neem $\omega = a\,\dd x_I$ (lineariteit). Eerst,
\[
\dd(h\omega) = k\Bigl(\int_0^1t^{k-1}a(tx)\dd t\Bigr)\dd x_I
+ \sum_{r=1}^k(-1)^{r-1}\sum_{j=1}^n
\Bigl(\int_0^1t^{k}\,\partial_ja(tx)\,\dd t\Bigr)
x_{i_r}\,\dd x_j\wedge\dd x_{I\setminus i_r} :
\]
de eerste groep verzamelt de termen waarin $\dd$ de factor
$x_{i_r}$ raakt --- de wedge $\dd x_{i_r}\wedge\dd
x_{I\setminus i_r}$ herstelt $\dd x_I$ met een teken
$(-1)^{r-1}$ dat de prefactor opheft, en de $k$ waarden van
$r$ geven de factor $k$ --- terwijl de tweede groep de
termen verzamelt waar $\dd$ de integraal raakt
(kettingregel brengt $t\,\partial_ja(tx)$). Vervolgens,
$\dd\omega = \sum_j\partial_ja\,\dd x_j\wedge\dd x_I$, en de
definitie van $h$ in graad $k+1$ toepassend, de \omterm{def:b3:holomorphic:index}{index} $j$ in
het eerste slot:
\[
h(\dd\omega) = \sum_{j=1}^n\Bigl(\int_0^1t^{k}
\partial_ja(tx)\dd t\Bigr)x_j\,\dd x_I
- \sum_{j=1}^n\sum_{r=1}^k(-1)^{r-1}
\Bigl(\int_0^1t^{k}\partial_ja(tx)\dd t\Bigr)
x_{i_r}\,\dd x_j\wedge\dd x_{I\setminus i_r} .
\]
De dubbele sommen heffen op in $\dd(h\omega) +
h(\dd\omega)$, dat dus gelijk is aan
\[
\Bigl(\int_0^1\bigl(k\,t^{k-1}a(tx) +
t^k{\textstyle\sum_j}x_j\,\partial_ja(tx)\bigr)\dd t\Bigr)
\dd x_I
= \Bigl(\int_0^1\frac{\dd}{\dd t}\bigl(t^ka(tx)\bigr)\dd
t\Bigr)\dd x_I
= a(x)\,\dd x_I = \omega,
\]
via de fundamentele stelling van de analyse. Stervormigheid
kwam binnen waar zij moest: $tx \in U$ voor $t \in
\intcc01$, zodat $a(tx)$ zin heeft.
\end{proof}

\begin{remark}
Voor $k = 1$ en $\omega = \sum_j a_j\dd x_j$ is de
primitieve $f(x) = \int_0^1\sum_ja_j(tx)\,x_j\,\dd t$ ---
de lijnintegraal van $\omega$ langs het segment $[0, x]$: de
stelling is de meerdere-variabelen-``een veld met
symmetrische Jacobiaan is een gradiënt'' van Jaar~2, nu in
elke graad. De hoekvorm (\cref{ex:b3:forms:angular}) toont
dat de hypothese op $U$ niet decoratief is:
$\R^2\setminus\{0\}$ is niet stervormig, en daar impliceert
geslotenheid geen exactheid. Wat overleeft op een algemene
\omterm{def:b3:topology:topology}{open verzameling} wordt gemeten door de \emph{de
Rham-cohomologie} $H^k(U) = \ker\dd/\operatorname{im}\dd$
--- zie \cref{exo:b3:forms:12} voor de eerste niet-triviale
berekening.
\end{remark}

\section{Oriëntatie en integratie op deelvariëteiten}

Integreren van een $k$-vorm vereist $k$-dimensionaal
georiënteerd territorium. Herinner uit
\cref{ch:b3:submanifolds}
(\cref{thm:b3:submanifolds:characterizations}) dat een
$k$-deelvariëteit $M \subseteq \R^n$ lokaal het beeld is van
een reguliere parametrisering $\gamma\colon V \to M \cap W$
($V \subseteq \R^k$ open, $\gamma$ een \omterm{def:b3:topology:continuity}{homeomorfisme} op haar
beeld met injectieve differentiaal).

\begin{definition}\label{def:b3:forms:orientation}
Een \emph{oriëntatie}\index{oriëntatie} van $M$ is een keuze,
voor elke $p \in M$, van één van de twee oriëntatieklassen
van basissen van de \omterm{def:b3:submanifolds:tangent}{raakruimte} $T_pM$, die \emph{lokaal
coherent} is: rond elk punt bestaat een parametrisering
$\gamma$ waarvan het coördinaatframe $(\partial_1\gamma,
\dots, \partial_k\gamma)$ in elk punt van haar domein
positief georiënteerd is. Zulke parametriseringen heten
\emph{direct}. $M$ is \emph{oriënteerbaar} als een
oriëntatie bestaat; de Möbiusband toont dat dit kan falen.
Alle deelvariëteiten in dit hoofdstuk zijn georiënteerd.
\end{definition}

\begin{definition}[Integraal van een vorm]\label{def:b3:forms:integral}
Zij $M$ een georiënteerde $k$-deelvariëteit en $\omega$ een
$k$-vorm gedefinieerd op een \omterm{def:b3:topology:topology}{omgeving} van $M$, met
$\operatorname{supp}\omega \cap M$ compact.
(a) Als $\operatorname{supp}\omega \cap M \subseteq
\gamma(V)$ voor één enkele directe parametrisering, stel
\[
\int_M\omega = \int_V\gamma^*\omega
\]
--- de rechterkant is de Lebesgue-integraal over $V$
(\cref{ch:b3:product}) van de coëfficiënt $g$ van
$\gamma^*\omega = g\,\dd u_1\wedge\dots\wedge\dd u_k$, die
continu is met compacte support.
(b) In het algemeen, kies eindig veel directe
parametriseringen $\gamma_i(V_i)$ die de compacte
$\operatorname{supp}\omega \cap M$ overdekken en een
ondergeschikte \omterm{lem:b3:forms:partition}{partitie van de eenheid} $(\chi_i)$
(\cref{lem:b3:forms:partition}), en stel $\int_M\omega =
\sum_i\int_M\chi_i\,\omega$, elke term berekend via (a).
Voor een kromme ($k = 1$) geparametriseerd door
$\gamma\colon\intcc ab\to\R^n$ schrijven we
$\int_\gamma\omega = \int_a^b\gamma^*\omega$, geen
injectiviteit vereist.
\end{definition}

\begin{lemma}[Consistentie]\label{lem:b3:forms:consistency}
Definitie (a) hangt niet af van de directe parametrisering,
en definitie (b) hangt noch van de overdekking noch van de
\omterm{lem:b3:forms:partition}{partitie van de eenheid} af. Bovendien, als $\Phi$ een
diffeomorfisme is van \omterm{def:b3:topology:topology}{omgevingen} van twee georiënteerde
deelvariëteiten met $\Phi(M) = M'$, dat een direct frame van
$M$ naar een direct frame van $M'$ draagt in enig punt van
elke component van $M$, dan $\int_{M'}\omega =
\int_M\Phi^*\omega$.
\end{lemma}

\begin{proof}
(a) Zij $\gamma\colon V \to M$, $\delta\colon V' \to M$
directe parametriseringen waarvan de beelden
$\operatorname{supp}\omega\cap M$ bevatten. De overgang
$\tau = \delta^{-1}\circ\gamma$ is een diffeomorfisme tussen
de relevante open deelverzamelingen van $V, V'$
(gladheid van overgangen:
\cref{thm:b3:submanifolds:characterizations}, via de lokale
grafiekbeschrijving), en $\gamma = \delta\circ\tau$ daar,
dus $\gamma^*\omega = \tau^*(\delta^*\omega)$
(\cref{thm:b3:forms:pullback}(a)). Schrijf $\delta^*\omega =
g\,\dd u_1\wedge\dots\wedge\dd u_k$; dan
(\cref{thm:b3:forms:pullback}(c)) $\tau^*(\delta^*\omega) =
(g\circ\tau)\,\det(D\tau)\,\dd u_1\wedge\dots\wedge\dd u_k$.
Beide frames direct, beeldt $D\tau$ een positieve basis af
op een positieve basis: $\det D\tau > 0$, dus $\det D\tau =
\abs{\det D\tau}$ en de variabelensubstitutiestelling
(\cref{thm:b3:product:changeofvar}) geeft
$\int(g\circ\tau)\abs{\det D\tau} = \int g$: de twee
integralen komen overeen. \emph{Hier is de hele reden
waarom \omterm{def:b3:forms:orientation}{oriëntatie} bestaat}: zonder tekencontrole verschillen
de Jacobiaan en haar absolute waarde en is de integraal
slecht gedefinieerd.
(b) Als $(\chi_i)$ en $(\tilde\chi_j)$ twee toelaatbare
partities zijn (overdekkingen inbegrepen), dan door (a) en
eindige additiviteit $\sum_i\int\chi_i\omega =
\sum_{i,j}\int\chi_i\tilde\chi_j\omega =
\sum_j\int\tilde\chi_j\omega$, elke dubbele term berekenbaar
in elke kaart. De laatste uitspraak: als $\gamma$ over
directe parametriseringen van $M$ loopt, loopt
$\Phi\circ\gamma$ over directe parametriseringen van $M'$
(de oriëntatievergelijking is lokaal constant, en vast in
één punt per component), en $(\Phi\circ\gamma)^*\omega =
\gamma^*(\Phi^*\omega)$.
\end{proof}

\begin{lemma}[Partities van de eenheid, compact geval]
\label{lem:b3:forms:partition}
Zij $K \subseteq \R^n$ compact en $W_1, \dots, W_m$ \omterm{def:b3:topology:topology}{open
verzamelingen} die $K$ overdekken. Er bestaan $\chi_1,
\dots, \chi_m \in \mathcal C^\infty(\R^n)$ met $0 \leq
\chi_i \leq 1$, $\operatorname{supp}\chi_i \subseteq W_i$
compact, en $\sum\chi_i = 1$ op een \omterm{def:b3:topology:topology}{omgeving} van
$K$.\index{partitie van de eenheid}
\end{lemma}

\begin{proof}
Elke $x \in K$ ligt in enige $W_{i(x)}$ met een \omterm{def:b3:forms:closedexact}{gesloten}
bal $\bar B(x, 2r_x) \subseteq W_{i(x)}$; compactheid haalt
$x_1, \dots, x_N$ eruit met de ballen $B(x_s, r_{x_s})$ die
$K$ overdekken. Voor elke $s$ neem een bump $\theta_s \in
\mathcal C^\infty$, $0 \leq \theta_s \leq 1$, $\theta_s =
1$ op $\bar B(x_s, r_{x_s})$, $\operatorname{supp}\theta_s
\subseteq B(x_s, 2r_{x_s})$ (mollificeer de indicator van de
bal van straal $\frac32r_{x_s}$,
\cref{thm:b3:lp:regularization}). Wijs elke $s$ toe aan één
\omterm{def:b3:holomorphic:index}{index} $i(s)$ met $B(x_s, 2r_{x_s}) \subseteq W_{i(s)}$ en
stel $\Theta_i = \sum_{i(s) = i}\theta_s$. Op de \omterm{def:b3:topology:topology}{open
verzameling} $\Omega_0 = \{\sum_j\Theta_j > \tfrac12\}
\supseteq K$ doen de functies $\Theta_i/\sum_j\Theta_j$ het
werk maar zijn alleen daar gedefinieerd; om te globaliseren,
laat $\rho \in \mathcal C^\infty(\R^n)$ voldoen aan $\rho =
0$ waar $\sum_j\Theta_j \geq 1$ en $\rho > 0$ waar
$\sum_j\Theta_j \leq \tfrac12$ (mollificeer een geschikte
afsnijding van $1 - \sum_j\Theta_j$), en stel
\[
\chi_i = \frac{\Theta_i}{\rho + \sum_j\Theta_j} .
\]
De noemer is overal $> 0$ en gelijk aan $\sum_j\Theta_j$ op
$\{\sum_j\Theta_j \geq 1\}$, een open \omterm{def:b3:topology:topology}{omgeving} van $K$ (elk
punt van $K$ ligt in enige bal $B(x_s, r_{x_s})$ waar
$\theta_s = 1$); daar $\sum_i\chi_i = 1$. Supports en
grenzen zijn duidelijk.
\end{proof}

\section{De stelling van Stokes}

\begin{definition}\label{def:b3:forms:boundary}
Een \emph{$k$-deelvariëteit met rand}\index{deelvariëteit met
rand} $M \subseteq \R^n$ is een verzameling overdekt door
reguliere parametriseringen van twee soorten:
\emph{inwendige kaarten} $\gamma\colon V \to M \cap W$ met
$V \subseteq \R^k$ open, en \emph{randkaarten} $\gamma\colon
V \cap H^k \to M \cap W$, waar $H^k = \{u \in \R^k : u_k
\geq 0\}$ en $\gamma$ zich glad en regulier voortzet tot de
open $V$. De \emph{rand} $\partial M$ is de verzameling
punten bereikt in $u_k = 0$; zij is een
$(k-1)$-deelvariëteit zonder rand, geparametriseerd door de
afbeeldingen $u' \mapsto \gamma(u', 0)$. Een \omterm{def:b3:forms:orientation}{oriëntatie} van
$M$ induceert er één op $\partial M$ door de regel
\emph{uitwaartse-normaal-eerst}: in $p \in \partial M$ is
een basis $(w_1, \dots, w_{k-1})$ van $T_p\partial M$
positief desda $(\nu, w_1, \dots, w_{k-1})$ een positieve
basis is van $T_pM$, waar $\nu \in T_pM \setminus
T_p\partial M$ uit $M$ wijst (in een randkaart: $\nu =
-\partial_k\gamma$, op toevoegen van tangentiële componenten
na --- de oriëntatieklasse ziet ze niet).
\end{definition}

\begin{lemma}[Stokes op de halfruimte]
\label{lem:b3:forms:halfspace}
Zij $\eta$ een gladde $(k-1)$-vorm op $\R^k$ met compacte
support. Dan
\[
\int_{H^k}\dd\eta = \int_{\partial H^k}\eta,
\]
waar $H^k$ de standaardoriëntatie van $\R^k$ draagt en
$\partial H^k = \{u_k = 0\} \cong \R^{k-1}$ de geïnduceerde,
die $(-1)^k$ maal de standaardoriëntatie van $\R^{k-1}$ is.
\end{lemma}

\begin{proof}
Eerst de oriëntatieboekhouding: de uitwaartse normaal langs
$\partial H^k$ is $-e_k$, en
\[
\det(-e_k, e_1, \dots, e_{k-1}) =
-\det(e_k, e_1, \dots, e_{k-1}) = -(-1)^{k-1}\det(e_1,
\dots, e_k) = (-1)^k :
\]
het frame $(e_1, \dots, e_{k-1})$ van $\partial H^k$ is
positief voor de geïnduceerde \omterm{def:b3:forms:orientation}{oriëntatie} precies wanneer $k$
even is, vandaar de gestelde vergelijking. Door lineariteit
neem $\eta = f\,\dd u_1\wedge\dots\wedge\widehat{\dd
u_i}\wedge\dots\wedge\dd u_k$, $f \in \mathcal
C^\infty_c(\R^k)$; dan $\dd\eta =
(-1)^{i-1}\,\partial_if\,\dd u_1\wedge\dots\wedge\dd u_k$
($\dd u_i$ naar haar slot bewegen kost $i - 1$
verwisselingen). Twee gevallen, beide via Tonelli--Fubini
(\cref{thm:b3:product:fubini}) en de éénvariabele
fundamentele stelling van de analyse.

\emph{Geval $i < k$.} Eerst integreren in $u_i$ over $\R$:
$\int_\R\partial_if\,\dd u_i = 0$ (compacte support), dus
$\int_{H^k}\dd\eta = 0$. En de restrictie van $\eta$ tot
$\{u_k = 0\}$ bevat de factor $\dd u_k$, die naar $0$
restrictieert ($u_k$ is daar constant): $\int_{\partial
H^k}\eta = 0$ ook.

\emph{Geval $i = k$.} Eerst integreren in $u_k$ over
$\intco0\infty$:
\begin{align*}
\int_{H^k}\dd\eta &= (-1)^{k-1}\int_{\R^{k-1}}
\Bigl(\int_0^\infty\partial_kf\,\dd u_k\Bigr)\dd u' \\
&= (-1)^{k-1}\int_{\R^{k-1}}\bigl(0 - f(u', 0)\bigr)\dd u'
= (-1)^k\int_{\R^{k-1}}f(u',0)\,\dd u' .
\end{align*}
Aan de randkant restrictieert $\eta$ tot $f(u',
0)\,\dd u_1\wedge\dots\wedge\dd u_{k-1}$, en de
geïnduceerde \omterm{def:b3:forms:orientation}{oriëntatie} is $(-1)^k$ maal de standaard,
$\int_{\partial H^k}\eta =
(-1)^k\int_{\R^{k-1}}f(u',0)\,\dd u'$. De twee kanten komen
overeen.
\end{proof}

\begin{theorem}[Stokes]\label{thm:b3:forms:stokes}
Zij $M \subseteq \R^n$ een compacte georiënteerde
$k$-deelvariëteit met \omterm{def:b3:forms:boundary}{rand}, $\partial M$ de geïnduceerde
\omterm{def:b3:forms:orientation}{oriëntatie} dragend, en zij $\omega$ een gladde
$(k-1)$-vorm op een \omterm{def:b3:topology:topology}{omgeving} van $M$.
Dan\index{stelling van Stokes}
\[
\int_M\dd\omega = \int_{\partial M}\omega .
\]
In het bijzonder, als $\partial M = \varnothing$:
$\int_M\dd\omega = 0$.
\end{theorem}

\begin{proof}
Overdek de compacte $M$ met eindig veel beelden van directe
kaarten (inwendig of \omterm{def:b3:forms:boundary}{rand}), en neem een \omterm{lem:b3:forms:partition}{partitie van de
eenheid} $(\chi_i)$ ondergeschikt aan de bijbehorende \omterm{def:b3:topology:topology}{open
verzamelingen} $W_i$ van $\R^n$
(\cref{lem:b3:forms:partition}), met $\sum\chi_i = 1$ op een
\omterm{def:b3:topology:topology}{omgeving} van $M$. Op die \omterm{def:b3:topology:topology}{omgeving} $\dd(\sum\chi_i) = 0$, dus
\[
\int_M\dd\omega = \sum_i\int_M\dd(\chi_i\omega),
\qquad
\int_{\partial M}\omega =
\sum_i\int_{\partial M}\chi_i\omega :
\]
beide kanten zijn additief, en het volstaat de stelling te
bewijzen voor een vorm met support in één kaartbeeld.

\emph{Inwendige kaart.} Als $\operatorname{supp}\omega \cap
M \subseteq \gamma(V)$ met $V$ open in $\R^k$: zet $\eta =
\gamma^*\omega$ met nul voort naar $\R^k$ (glad, compacte
support in $V$) en pas \cref{lem:b3:forms:halfspace} toe met
de support weg van $\partial H^k$ (transleer $V$ in de open
bovenhalfruimte --- of herhaal gewoon de berekening van
geval $i < k$ over heel $\R^k$): $\int_M\dd\omega =
\int_{\R^k}\dd\eta = 0$, en $\int_{\partial M}\omega = 0$
want $\omega$ verdwijnt nabij $\partial M$.

\emph{Randkaart.} Als $\operatorname{supp}\omega\cap M
\subseteq \gamma(V \cap H^k)$: met $\eta = \gamma^*\omega$
met nul voortgezet, $\gamma^*(\dd\omega) = \dd\eta$
(\cref{thm:b3:forms:pullback}(b)), dus via
\cref{def:b3:forms:integral}:
\[
\int_M\dd\omega = \int_{H^k}\dd\eta
\overset{\text{\cref{lem:b3:forms:halfspace}}}{=}
\int_{\partial H^k}\eta .
\]
Het rest te identificeren van de rechterkant met
$\int_{\partial M}\omega$. De \omterm{def:b3:forms:boundary}{rand} is geparametriseerd door
$\beta(u') = \gamma(u', 0)$, en $\beta^*\omega$ is de
restrictie van $\eta$ tot $\{u_k = 0\}$ (\omterm{def:b3:forms:pullback}{pullback} onder de
inclusie $u' \mapsto (u', 0)$ samengesteld met $\gamma$).
De oriëntatievergelijking is dezelfde $(-1)^k$ aan beide
kanten: het frame $(\partial_1\beta, \dots,
\partial_{k-1}\beta)$ zit in de geïnduceerde \omterm{def:b3:forms:orientation}{oriëntatie} van
$\partial M$ met het teken $\det(-e_k, e_1, \dots, e_{k-1})
= (-1)^k$ berekend in de kaart (de uitwaartse vector trekt
terug naar $-e_k$), precies het teken dat de geïnduceerde
\omterm{def:b3:forms:orientation}{oriëntatie} van $\partial H^k$ relateert aan de standaard
$\R^{k-1}$ (\cref{lem:b3:forms:halfspace}). De twee
tekenconventies heffen op: $\int_{\partial H^k}\eta =
\int_{\partial M}\omega$.
\end{proof}

\begin{example}[De klassieke stellingen]
\label{ex:b3:forms:classical}
Zij $D \subseteq \R^2$ een compact domein met gladde \omterm{def:b3:forms:boundary}{rand},
standaard georiënteerd. Voor $\omega = P\,\dd x + Q\,\dd y$:
$\dd\omega = \bigl(\partial_xQ - \partial_yP\bigr)\dd
x\wedge\dd y$, en Stokes leest
\[
\int_D\Bigl(\frac{\partial Q}{\partial x} -
\frac{\partial P}{\partial y}\Bigr)\dd x\,\dd y
= \oint_{\partial D}P\,\dd x + Q\,\dd y :
\]
\emph{Green--Riemann}, bewezen voor elementaire domeinen in
het volume Jaar~2 en nu in natuurlijke algemeenheid. In
$\R^3$ geeft Stokes toegepast op de flux-$2$-vorm van een
vectorveld op een compact domein de
\emph{divergentiestelling} $\int_\Omega\operatorname{div}F =
\int_{\partial\Omega}\langle F, \nu\rangle\,\dd S$, en
toegepast op een $1$-vorm op een oppervlak-met-rand, de
klassieke \emph{Kelvin--Stokes}-rotorstelling;
\cref{exo:b3:forms:7} spelt beide woordenboeken uit.
\end{example}

\section{Het omwentelingsgetal}\label{sec:b3:forms:winding}

\begin{definition}\label{def:b3:forms:winding}
Zij $\gamma\colon\intcc01\to\R^2\setminus\{a\}$ een gladde
\omterm{def:b3:forms:closedexact}{gesloten} kromme. Haar \emph{omwentelingsgetal}\index{omwentelingsgetal}
rond $a$ is
\[
\operatorname{Ind}_\gamma(a) =
\frac1{2\pi}\int_\gamma\omega_\theta^a,
\qquad
\omega_\theta^a = \frac{(x - a_1)\,\dd y - (y - a_2)\,\dd
x}{(x - a_1)^2 + (y - a_2)^2} .
\]
\end{definition}

\begin{proposition}\label{prop:b3:forms:windinginteger}
$\operatorname{Ind}_\gamma(a) \in \Z$; als functie van $a$
is zij constant op elke \omterm{def:b3:topology:components}{samenhangscomponent} van
$\R^2\setminus\gamma(\intcc01)$ en nul op de onbegrensde
component. Voor $\gamma(t) = a + r(\cos2\pi t, \sin2\pi t)$:
$\operatorname{Ind}_\gamma(a) = 1$.
\end{proposition}

\begin{proof}
Neem $a = 0$ en schrijf $\gamma = (x, y)$, $\rho =
\norm\gamma > 0$, $\vartheta(t) =
\int_0^t\gamma^*\omega_\theta$, zodat $\vartheta' =
\frac{xy' - yx'}{x^2 + y^2}$. In complexe notatie laat $u(t)
= \gamma(t)\,\rho(t)^{-1}\eu^{-\iu\vartheta(t)}$; dan $\abs u
= 1$ en een rechtstreekse berekening geeft
\[
\frac{u'}{u} = \frac{\gamma'}{\gamma} - \frac{\rho'}{\rho}
- \iu\vartheta'
= \frac{\bar\gamma\gamma'}{\abs\gamma^2} -
\frac{\rho'}{\rho} - \iu\,\frac{xy' -
yx'}{\rho^2}
= \frac{(xx' + yy')}{\rho^2} - \frac{\rho'}{\rho} = 0,
\]
want $\bar\gamma\gamma' = (xx' + yy') + \iu(xy' - yx')$ en
$\rho\rho' = xx' + yy'$. Dus is $u$ constant: $\gamma(t) =
\rho(t)\,c\,\eu^{\iu\vartheta(t)}$ met $\abs c = 1$, en
$\gamma(1) = \gamma(0)$ met $\rho(1) = \rho(0)$ dwingt
$\eu^{\iu\vartheta(1)} = \eu^{\iu\vartheta(0)} = 1$:
$\vartheta(1) \in 2\pi\Z$, d.w.z.\
$\operatorname{Ind}_\gamma(0) \in \Z$. Als functie van $a$
op het open complement van de compacte kromme is de
definiërende integraal continu
(\cref{thm:b3:lebesgue:paramcont}, dominantie op een
\omterm{def:b3:topology:topology}{omgeving} van elke $a$); een continue geheelwaardige functie
is lokaal constant, dus constant op componenten. Voor
$\norm a$ groot is de integrand $O(1/\norm a)$ uniform in
$t$, dus neigt de \omterm{def:b3:holomorphic:index}{index} naar $0$ en verdwijnt op de
onbegrensde component. Voor de cirkel: $\gamma^*\omega_\theta^a
= 2\pi\,\dd t$ rechtstreeks.
\end{proof}

\begin{remark}
Onder $\C \cong \R^2$ is $\frac{\dd z}{z} = \frac{x\,\dd x +
y\,\dd y}{x^2 + y^2} + \iu\,\omega_\theta$, dus
$\operatorname{Ind}_\gamma(a) =
\frac1{2\iu\pi}\oint_\gamma\frac{\dd z}{z - a}$: dit is de
\omterm{def:b3:holomorphic:index}{index} van \cref{ch:b3:residues}, en
\cref{prop:b3:forms:windinginteger} herbewezen haar
geheelheid en lokale constantheid met reële-variabele
middelen --- de topologische helft van de residustelling,
nu steunend op Stokes.
\end{remark}

\begin{proposition}\label{prop:b3:forms:exactloop}
Als $\omega = \dd f$ \omterm{def:b3:forms:closedexact}{exact} is op de open $U$ en
$\gamma\colon\intcc01\to U$ een \omterm{def:b3:forms:closedexact}{gesloten} kromme is, dan
$\int_\gamma\omega = 0$.
\end{proposition}

\begin{proof}
$\int_\gamma\dd f = \int_0^1(f\circ\gamma)'(t)\,\dd t =
f(\gamma(1)) - f(\gamma(0)) = 0$ --- de kettingregel
identificeert $\gamma^*(\dd f)$ met $(f\circ\gamma)'\,\dd t$.
\end{proof}

\begin{method}\label{met:b3:forms:method}
Rekenen met vormen: (1) mechaniseer --- wedges herorden met
tekens, $\dd$ differentieert coëfficiënten tot verse $\dd
x_j$'s, \omterm{def:b3:forms:pullback}{pullbacks} substitueren; vertrouw de algebra, zij
codeert elke Jacobiaan. (2) Om een vorm over een
deelvariëteit te integreren: parametriseer direct, trek
terug, integreer de coëfficiënt; \omterm{def:b3:forms:orientation}{oriëntatie} is de enige val
--- controleer één frame. (3) Om een
integraalidentiteit te bewijzen, zoek een Stokes-vorm: is de
integrand \omterm{def:b3:forms:closedexact}{exact}? is het domein een \omterm{def:b3:forms:boundary}{rand}? (4) Om integralen
over twee ``parallelle'' deelvariëteiten te vergelijken, pas
Stokes toe op het gebied ertussen (het
deformatie-argument, \cref{exo:b3:forms:10}). (5) Een
niet-nul-integraal van een gesloten vorm certificeert een
topologische obstructie --- geen primitieve, geen retractie,
geen nulpuntvrije voortzetting: zo doodt het
weekendprobleem retracties van de bal.
\end{method}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:b3:forms:1}
Op $\R^3$, zij $\omega = x\,\dd y - z\,\dd x$ en $\eta =
\dd x + y\,\dd z$. Bereken $\omega\wedge\eta$, $\dd\omega$,
$\dd\eta$, en $\dd(\omega\wedge\eta)$, en verifieer de
gegradeerde Leibniz-regel op dit voorbeeld.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b3:forms:2}
Identificeer, op $\R^3$, de drie incarnaties van $\dd$:
voor $f \in \Omega^0$, $\dd f \leftrightarrow \nabla f$;
voor de arbeid-vorm $\omega_F = F_1\dd x + F_2\dd y +
F_3\dd z$, $\dd\omega_F \leftrightarrow
\operatorname{curl}F$; voor de flux-vorm $\sigma_F =
F_1\,\dd y\wedge\dd z + F_2\,\dd z\wedge\dd x + F_3\,\dd
x\wedge\dd y$, $\dd\sigma_F \leftrightarrow
\operatorname{div}F$. Leid uit $\dd^2 = 0$ de identiteiten
$\operatorname{curl}\nabla f = 0$ en
$\operatorname{div}\operatorname{curl}F = 0$ af.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:forms:3}
Beslis of elke $1$-vorm \omterm{def:b3:forms:closedexact}{gesloten} is, \omterm{def:b3:forms:closedexact}{exact} op haar domein,
en bereken een primitieve wanneer er één bestaat:
(a) $(2xy + z^2)\,\dd x + x^2\,\dd y + 2xz\,\dd z$ op
$\R^3$;
(b) $\dfrac{x\,\dd x + y\,\dd y}{x^2 + y^2}$ op
$\R^2\setminus\{0\}$;
(c) $\dfrac{-y\,\dd x + x\,\dd y}{x^2 + y^2}$ op het
halfvlak $\{x > 0\}$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:forms:4}
(De hoekvorm) Verifieer dat $\omega_\theta$
(\cref{ex:b3:forms:angular}) \omterm{def:b3:forms:closedexact}{gesloten} is; bereken
$\int_\gamma\omega_\theta$ voor $\gamma$ de cirkel van
straal $r$ rond $0$; concludeer dat $\omega_\theta$ niet
\omterm{def:b3:forms:closedexact}{exact} is op $\R^2\setminus\{0\}$, en dat
$\R^2\setminus\{0\}$ stervormig is ten opzichte van geen
van haar punten (twee routes: via
\cref{thm:b3:forms:poincare}, en rechtstreeks uit de
meetkunde).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:forms:5}
(Oppervlaktevorm van een hyperoppervlak) Zij $M \subseteq
\R^n$ een compact georiënteerd hyperoppervlak waarvan de
\omterm{def:b3:forms:orientation}{oriëntatie} wordt gegeven door een eenheidsnormaalveld $\nu$
($(w_1, \dots, w_{n-1})$ positief desda $(\nu, w_1, \dots,
w_{n-1})$ positief in $\R^n$). Toon aan dat de
$(n-1)$-vorm $\sigma_\nu =
\sum_i(-1)^{i-1}\nu_i\,\dd x_1\wedge\dots\wedge\widehat{\dd
x_i}\wedge\dots\wedge\dd x_n$ op $M$ restrictieert tot de
\emph{oppervlaktevorm}: voor een directe parametrisering
$\gamma$, $\gamma^*\sigma_\nu = \sqrt{\det G}\,\dd
u_1\wedge\dots \wedge\dd u_{n-1}$, waar $G = ({}^t
D\gamma)(D\gamma)$ de Gram-matrix is. \emph{(Merk op dat
$(\gamma^*\sigma_\nu)(e_1, \dots, e_{n-1}) = \det(\nu,
\partial_1\gamma, \dots, \partial_{n-1}\gamma)$ en
kwadrateer deze determinant.)} Bereken $\int_{S^2}\sigma_\nu
= 4\pi$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:forms:6}
Bereken $\int_{S^2}\omega$ voor $\omega = x\,\dd y\wedge\dd z
+ y\,\dd z\wedge\dd x + z\,\dd x\wedge\dd y$: (a)
rechtstreeks in bolcoördinaten; (b) via Stokes op de
eenheidsbal. Leid $\operatorname{vol}(B^3) = \frac{4\pi}3$
af uit de oppervlakte van $S^2$, en generaliseer:
$n\operatorname{vol}(B^n) = \operatorname{area}(S^{n-1})$,
consistent met \cref{thm:b3:product:ballvolume}.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:forms:7}
(De woordenboeken) Leid zorgvuldig af uit
\cref{thm:b3:forms:stokes}: (a) de divergentiestelling in
$\R^3$ (combineer \cref{exo:b3:forms:2} en
\cref{exo:b3:forms:5}); (b) de Kelvin--Stokes-stelling
$\int_S\langle\operatorname{curl}F, \nu\rangle\,\dd S =
\oint_{\partial S}\langle F, \tau\rangle\,\dd\ell$ voor een
compact georiënteerd oppervlak met \omterm{def:b3:forms:boundary}{rand} in $\R^3$.
Controleer dat de oriëntatieconventies overeenkomen op de
bovenste halfsfeer begrensd door de evenaar.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:forms:8}
(Identiteiten van Green) Voor $u, v$ glad op een \omterm{def:b3:topology:topology}{omgeving} van
een compact domein $\Omega \subseteq \R^n$ met gladde \omterm{def:b3:forms:boundary}{rand},
bewijs
\[
\int_\Omega\bigl(u\,\Delta v + \langle\nabla u, \nabla
v\rangle\bigr) = \int_{\partial\Omega}u\,\partial_\nu
v\,\dd S,
\qquad
\int_\Omega\bigl(u\,\Delta v - v\,\Delta u\bigr) =
\int_{\partial\Omega}\bigl(u\,\partial_\nu v -
v\,\partial_\nu u\bigr)\dd S .
\]
Leid af: een \omterm{prop:b3:conformal:harmonicholo}{harmonische functie} op $\Omega$ die verdwijnt
op $\partial\Omega$ verdwijnt identiek, en twee \omterm{prop:b3:conformal:harmonicholo}{harmonische
functies} met dezelfde randwaarden vallen samen ---
eenduidigheid in het Dirichlet-probleem van
\cref{ch:b3:conformal}.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:forms:9}
(Variabelensubstitutie, georiënteerde vorm) Zij
$\varphi\colon U \to V$ een diffeomorfisme van openen van
$\R^n$ met $\det D\varphi > 0$, en $f$ continu met compacte
support in $V$. Toon aan dat de pullback-identiteit
$\int_U\varphi^*(f\,\dd x_1\wedge\dots\wedge\dd x_n) =
\int_Vf\,\dd x_1\wedge\dots\wedge\dd x_n$ equivalent is met
de variabelensubstitutiestelling
(\cref{thm:b3:product:changeofvar}) voor zulke $\varphi$, en
leg precies uit waar de absolute waarde op de Jacobiaan
heen ging.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b3:forms:10}
(Deformatie) Zij $\omega$ een \emph{\omterm{def:b3:forms:closedexact}{gesloten}} $2$-vorm op
$\R^3\setminus\{0\}$ en $S_r$ de sfeer van straal $r$ met
middelpunt $0$. Toon aan dat $\int_{S_r}\omega$ niet van $r
> 0$ afhangt \emph{(pas Stokes toe op de schil tussen twee
stralen; let op de twee randoriëntaties)}. Pas toe op de
\emph{ruimtehoekvorm}
\[
\omega = \frac{x\,\dd y\wedge\dd z + y\,\dd z\wedge\dd x +
z\,\dd x\wedge\dd y}{(x^2 + y^2 + z^2)^{3/2}} :
\]
controleer dat zij \omterm{def:b3:forms:closedexact}{gesloten} is, bereken $\int_{S_r}\omega =
4\pi$, en concludeer dat zij \omterm{def:b3:forms:closedexact}{gesloten} maar niet \omterm{def:b3:forms:closedexact}{exact} is op
$\R^3\setminus\{0\}$ --- de tweedimensionale zuster van
$\omega_\theta$, en de meetkundige \omterm{def:b3:rings:content}{inhoud} van de wet van
Gauss in de elektrostatica.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:forms:11}
Zij $\gamma$ een gladde \omterm{def:b3:forms:closedexact}{gesloten} kromme in
$\R^2\setminus\{0\}$ met $n = \operatorname{Ind}_\gamma(0)$.
Toon aan $\int_\gamma\omega = n\int_{S^1}\omega$ voor
\emph{elke} \omterm{def:b3:forms:closedexact}{gesloten} $1$-vorm $\omega$ op
$\R^2\setminus\{0\}$ \emph{(schrijf $\omega =
c\,\omega_\theta + \dd f$ via \cref{exo:b3:forms:12})}.
Interpretatie: op het doorboorde vlak is het
\omterm{def:b3:forms:winding}{omwentelingsgetal} de enige obstructie tot het verdwijnen van
perioden.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b3:forms:12}
(Eerste de Rham-berekening) Toon aan dat elke \omterm{def:b3:forms:closedexact}{gesloten}
$1$-vorm $\omega$ op $U = \R^2\setminus\{0\}$ uniek is
\[
\omega = c\,\omega_\theta + \dd f,
\qquad c = \frac1{2\pi}\int_{S^1}\omega,\quad f \in
\mathcal C^\infty(U) :
\]
definieer $f(p)$ door $\omega - c\,\omega_\theta$ te
integreren langs een pad van $(1,0)$ naar $p$ (radiaal stuk
dan cirkelboog), toon aan dat het resultaat onafhankelijk is
van de keuzes precies omdat de $S^1$-periode verdwijnt, en
controleer $\dd f = \omega - c\,\omega_\theta$. Concludeer:
$H^1(\R^2\setminus\{0\}) \cong \R$, voortgebracht door de
hoekvorm.
\end{exercise}

\section{Probleem: de vastpuntstelling van Brouwer}

\begin{problem}[{Weekendprobleem --- geen retractie, geen
ontsnapping}]\label{pb:b3:forms:1}
De stelling van Brouwer zegt dat \emph{elke \omterm{def:b3:topology:continuity}{continue
afbeelding} van de \omterm{def:b3:forms:closedexact}{gesloten} eenheidsbal $\bar B = \bar B^n
\subseteq \R^n$ in zichzelf een vast punt heeft} --- één van
de grote stellingen van de wiskunde, met gevolgen van
speltheorie (Nash-evenwichten) tot matrixanalyse. Het
differentiaalvorm-bewijs is het schoonst bekende: Stokes
toont dat de sfeer geen retractie van de bal is, en al het
overige volgt. Overal is $S = S^{n-1} = \partial\bar B$, $n
\geq 2$, en
\[
\sigma = \sum_{i=1}^n(-1)^{i-1}x_i\,
\dd x_1\wedge\dots\wedge\widehat{\dd x_i}\wedge\dots\wedge
\dd x_n
\]
(de hat verwijdert de factor).

\medskip
\noindent\textbf{Deel I --- Het meetinstrument.}
\begin{enumerate}
  \item Bereken $\dd\sigma$, en leid uit Stokes
        (\cref{thm:b3:forms:stokes}) af dat $\int_S\sigma =
        n\operatorname{vol}(\bar B) > 0$, waar $S$ de
        randoriëntatie van de bal draagt.
  \item Voor $n = 2$ en $n = 3$, identificeer de restrictie
        van $\sigma$ tot $S$ met de booglengte- en
        oppervlaktevormen (\cref{exo:b3:forms:5} met
        $\nu(x) = x$) en herbereken $\int_S\sigma$
        rechtstreeks.
  \item Zij $W \subseteq \R^N$ open en $\varphi\colon W \to
        \R^n$ glad met $\norm{\varphi(x)} = 1$ voor alle $x
        \in W$. Toon aan dat $\varphi^*(\dd\sigma) = 0$.
        \emph{(Differentieer $\norm\varphi^2 = 1$: het beeld
        van $D\varphi(x)$ ligt in het hypervlak
        $\varphi(x)^\perp$, van dimensie $n - 1$; pas dan
        \cref{prop:b3:forms:linearpullback}(2) toe.)}
  \item Waar zit de fout in het volgende ``bewijs'' dat
        $\int_S\sigma = 0$: ``$S$ is compact zonder \omterm{def:b3:forms:boundary}{rand}, en
        $\sigma$ gerestricteerd tot $S$ is een topvorm
        erop, dus \omterm{def:b3:forms:closedexact}{gesloten}, dus $\int_S\sigma =
        \int_S\dd(\text{iets}) = 0$ door Stokes''?
        (Wijs het foute woord aan.)
  \item Leg in één alinea de strategie van Deel~II uit: wat
        zal worden geïntegreerd, over wat, en waar de
        tegenspraak vandaan komt.
\end{enumerate}

\medskip
\noindent\textbf{Deel II --- Geen gladde retractie.}
Veronderstel, voor tegenspraak, dat $r$ een \emph{gladde
retractie} is van de bal op haar sfeer: $r$ is glad op een
\omterm{def:b3:topology:topology}{omgeving} van $\bar B$, $r(\bar B) \subseteq S$, en $r(x) =
x$ voor alle $x \in S$.
\begin{enumerate}[resume]
  \item Rechtvaardig $\int_Sr^*\sigma = \int_S\sigma$
        \emph{(op $S$ restrictieert $r$ tot de identiteit:
        als $\gamma$ een directe parametrisering is van een
        stuk van $S$, dan $r\circ\gamma = \gamma$)}.
  \item Met Stokes op $\bar B$, toon aan $\int_Sr^*\sigma =
        \int_{\bar B}\dd(r^*\sigma)$.
  \item Toon aan $\dd(r^*\sigma) = r^*(\dd\sigma) = 0$
        (vraag~3 toegepast op $\varphi = r$), en
        concludeer: \emph{er is geen gladde retractie $\bar
        B \to S$.}
  \item Regel het uitgesloten geval $n = 1$ met de hand:
        toon rechtstreeks dat geen \omterm{def:b3:topology:continuity}{continue afbeelding}
        $\intcc{-1}1 \to \{-1, 1\}$ beide eindpunten
        fixeert, en noem de stelling die u gebruikte.
\end{enumerate}

\medskip
\noindent\textbf{Deel III --- Gladde Brouwer.} Zij $g$ glad
op een \omterm{def:b3:topology:topology}{omgeving} van $\bar B$ met $g(\bar B) \subseteq \bar
B$ en \emph{geen} vast punt in $\bar B$.
\begin{enumerate}[resume]
  \item Toon aan $\delta = \min_{x\in\bar B}\norm{g(x) - x}
        > 0$.
  \item Voor $x \in \bar B$ stel $u(x) = \frac{x -
        g(x)}{\norm{x - g(x)}}$ en laat $r(x) = x +
        t(x)\,u(x)$ de snijding zijn van de straal $\{x +
        tu(x) : t \geq 0\}$ met $S$. Los de kwadratische
        $\norm{x + tu}^2 = 1$ op en verkrijg
        \[
        t(x) = -\langle x, u(x)\rangle +
        \sqrt{1 - \norm x^2 + \langle x, u(x)\rangle^2}
        \;\geq\; 0 .
        \]
  \item Toon aan dat de radicand \emph{strikt} positief is
        op $\bar B$: als $1 - \norm x^2 + \langle x,
        u(x)\rangle^2 = 0$ dan $\norm x = 1$ en $\langle x,
        u(x)\rangle = 0$, d.w.z.\ $\langle x, x - g(x)
        \rangle = 0$, d.w.z.\ $\langle x, g(x)\rangle =
        1$; door Cauchy--Schwarz met $\norm x = 1$,
        $\norm{g(x)} \leq 1$, dwingt dit $g(x) = x$ ---
        uitgesloten. Leid af dat $r$ glad is op een \omterm{def:b3:topology:topology}{omgeving}
        van $\bar B$.
  \item Toon aan $r(\bar B) \subseteq S$ en $r(x) = x$ voor
        $x \in S$ \emph{(voor $\norm x = 1$, controleer
        $t(x) = 0$ met $\langle x, u(x)\rangle \geq 0$, wat
        zelf volgt uit $\langle x, x - g(x)\rangle = 1 -
        \langle x, g(x)\rangle \geq 0$)}. Concludeer met
        Deel~II: \emph{elke gladde zelfafbeelding van $\bar
        B$ heeft een vast punt.}
\end{enumerate}

\medskip
\noindent\textbf{Deel IV --- Continue Brouwer.} Zij
$f\colon\bar B\to\bar B$ continu zonder vast punt.
\begin{enumerate}[resume]
  \item Toon aan $\varepsilon = \min_{\bar B}\norm{f - x} >
        0$, en produceer een \emph{polynomiale} afbeelding
        $p\colon\R^n\to \R^n$ met $\sup_{\bar B}\norm{p -
        f} < \varepsilon/2$ (Stone--Weierstrass,
        \cref{thm:b3:complete:stoneweierstrass},
        coördinaat voor coördinaat --- rechtvaardig de
        overgang van scalaire naar vectorbenadering).
  \item De afbeelding $p$ kan de bal verlaten; stel $g =
        \frac{p}{1 + \varepsilon/2}$. Toon aan $g(\bar B)
        \subseteq \bar B$ en $\sup_{\bar B}\norm{g - f} <
        \varepsilon$.
  \item Leid een tegenspraak af met Deel~III en concludeer:
        \emph{elke \omterm{def:b3:topology:continuity}{continue afbeelding} $\bar B^n \to \bar
        B^n$ heeft een vast punt.}
  \item Toon met een voorbeeld dat de stelling faalt op: de
        open bal; de sfeer $S$; een \omterm{def:b3:forms:closedexact}{gesloten} annulus. Welke
        eigenschap van $\bar B$ verliest elk
        tegenvoorbeeld?
\end{enumerate}

\medskip
\noindent\textbf{Deel V --- Dividenden.}
\begin{enumerate}[resume]
  \item (Perron--Frobenius, bestaan) Zij $A$ een
        $n\times n$-matrix met alle entries $> 0$, en
        $\Delta = \{x \in \R^n : x_i \geq 0,\ \sum x_i =
        1\}$. Toon aan dat de afbeelding $x \mapsto
        Ax/\norm{Ax}_1$ welgedefinieerd en continu is op
        $\Delta$, dat $\Delta$ homeomorf is met een
        \omterm{def:b3:forms:closedexact}{gesloten} bal van $\R^{n-1}$ (radiaal \omterm{def:b3:topology:continuity}{homeomorfisme}
        vanaf een convex compact met niet-leeg inwendige in
        haar affiene span), en concludeer dat $A$ een
        eigenvector heeft met strikt positieve entries en
        eigenwaarde $> 0$.
  \item Leid af dat elke stochastische matrix met positieve
        entries (kolommen sommerend tot $1$) een stationaire
        kansvector $\pi = A\pi$ heeft --- de
        PageRank-achtige vector. (Eenduidigheid geldt ook
        maar vraagt andere gereedschappen.)
  \item (Harige bal, opzet) Zij $v$ glad op een \omterm{def:b3:topology:topology}{omgeving} van
        $S = S^{n-1}$ met $\langle v(x), x\rangle = 0$ en
        $\norm{v(x)} = 1$ voor $x \in S$ (een
        \emph{eenheidstangentveld}). Voor $t \in \R$ stel
        $F_t(x) = x + t\,v(x)$. Toon aan $\norm{F_t(x)} =
        \sqrt{1 + t^2}$ op $S$: $F_t$ beeldt $S$ af in de
        sfeer $\sqrt{1+t^2}\,S$.
  \item Toon aan dat $P(t) = \int_SF_t^*\sigma$ een
        \emph{polynoom} is in $t$ \emph{(elke
        coëfficiëntfunctie van $F_t^*\sigma$ in een kaart is
        polynomiaal in $t$, met coëfficiënten glad in de
        kaartvariabele; integratie is lineair)}.
  \item Toon aan dat voor $\abs t$ klein $F_t$ een
        diffeomorfisme is van $S$ op $\sqrt{1+t^2}\,S$:
        injectiviteit voor $t\operatorname{Lip}(v) < 1$;
        lokaal diffeomorfisme door de stelling van de
        inverse functie (\cref{thm:b3:submanifolds:ift} in
        kaarten); beeld open en \omterm{def:b3:forms:closedexact}{gesloten} in de
        samenhangende doelsfeer. Leid af, met
        \cref{lem:b3:forms:consistency} en de schaling
        $\sigma_{\lambda x} = \lambda^{n}\,\sigma_x$ onder
        $x \mapsto \lambda x$ (controleer het), dat
        \[
        P(t) = \pm(1 + t^2)^{n/2}\int_S\sigma,
        \qquad\text{met het teken } + \text{ voor kleine } t
        \]
        (\omterm{def:b3:forms:orientation}{oriëntatie} bewaard door continuïteit vanaf $t =
        0$).
  \item Concludeer (Milnor): als $n$ \emph{oneven} is, is
        $(1 + t^2)^{n/2}$ geen polynoom in $t$, maar komt
        wel overeen met het polynoom $P(t)/\int_S\sigma$
        nabij $0$ --- tegenspraak. Dus \emph{dragen de
        even-dimensionale bollen $S^{n-1}$ ($n$ oneven)
        geen eenheidstangentveld}, en, door normaliseren en
        gladmaken (convolueer componentgewijs en projecteer
        --- rechtvaardig beide stappen), helemaal geen
        continu nergens-verdwijnend tangentveld: elke wind
        op aarde laat een stil punt over.

  \item (Oneven bollen kammen vrij) Exhibeer op $S^{2m-1}
        \subseteq \R^{2m} \cong \C^m$ een expliciet glad
        eenheidstangentveld: $v(x) = \iu x$ in complexe
        notatie, d.w.z.
        \[
        v(x_1, y_1, \dots, x_m, y_m) =
        (-y_1, x_1, \dots, -y_m, x_m) .
        \]
        Verifieer raakheid en eenheidslengte, en concludeer
        dat de pariteitsdichotomie van vraag~23 scherp is:
        een sfeer is kambaar precies wanneer haar dimensie
        oneven is. Waar breekt het polynoomargument van
        vraag~22 voor even $n$?
  \item (Nulpunten uit randgedrag) Zij $f \colon \bar B^n
        \to \R^n$ continu met $\langle f(x), x\rangle \geq
        0$ voor elke $x \in S^{n-1}$. Toon aan dat $f$
        ergens in $\bar B^n$ verdwijnt. \emph{(Als niet,
        beeldt $g(x) = -f(x)/\norm{f(x)}$ $\bar B$ continu
        af in $S \subseteq \bar B$; pas Brouwer toe op $g$
        en spreek de randhypothese tegen.)} Leid het
        \emph{surjectiviteitscriterium} af: een continue
        $F\colon\R^n\to\R^n$ met $\frac{\langle F(x),
        x\rangle}{\norm x} \to +\infty$ als $\norm x \to
        \infty$ is surjectief --- de eindigdimensionale
        voorouder van de coërciviteitsargumenten van de
        niet-lineaire analyse.
\end{enumerate}
\end{problem}
