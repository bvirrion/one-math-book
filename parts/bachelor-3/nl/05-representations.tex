\chapter{Representaties van eindige groepen}\label{ch:b3:representations}

Om een abstracte groep te begrijpen, laat haar werken op een
vectorruimte en gebruik lineaire algebra --- eigenwaarden,
sporen, inwendige producten --- op de \omterm{def:b3:groups:action}{werking}. Dit programma,
\emph{representatietheorie}, is verbazingwekkend effectief voor
eindige groepen over $\C$: elke \omterm{def:b3:representations:rep}{representatie} splitst in
irreducibele (Maschke), de irreducibele worden vastgelegd door
hun \emph{\omterm{def:b3:representations:character}{karakters}} (sporen), en de \omterm{def:b3:representations:character}{karakters} voldoen aan
\omterm{thm:b3:representations:orthogonality}{orthogonaliteitsrelaties} die berekeningen mechanisch maken. Dit
hoofdstuk bouwt deze calculus en haar eerste meesterwerken ---
karaktertabellen van de kleine groepen --- en het
weekendprobleem oogst een stelling ver buiten het bereik van
pure groepentheorie: de $p^aq^b$-stelling van Burnside, die
beslist dat elke groep van orde $p^aq^b$ \omterm{def:b3:groups:derived}{oplosbaar} is. Overal is
$G$ een eindige groep en zijn alle vectorruimten
eindigdimensionaal over $\C$.

\section{Representaties, Maschke, Schur}

\begin{definition}\label{def:b3:representations:rep}
Een \emph{representatie}\index{representatie} van $G$ is een
morfisme $\rho \colon G \to GL(V)$ voor een $\C$-vectorruimte
$V$; $\dim V$ is haar \emph{graad}. Een deelruimte $W \subseteq
V$ is \emph{invariant} als $\rho(g)W \subseteq W$ voor alle $g$;
beperking maakt $W$ tot een \emph{subrepresentatie}. $\rho$ is
\emph{irreducibel}\index{irreducibele representatie} als $V \neq
0$ en haar enige invariante deelruimten $0$ en $V$ zijn. Een
\emph{morfisme} tussen $(\rho, V)$ en $(\sigma, W)$ is een
lineaire $f\colon V \to W$ met $f\rho(g) = \sigma(g)f$ voor alle
$g$ (equivariantie); bijectieve $f$ zijn \emph{isomorfismen}.
\end{definition}

\begin{example}\label{ex:b3:representations:examples}
(a) Graad $1$: morfismen $G \to \C^\times$.
(b) De \emph{reguliere representatie}\index{reguliere
representatie}: $V = \C^G$ met basis $(e_h)_{h \in G}$,
$\rho(g)e_h = e_{gh}$; graad $\abs G$.
(c) Een permutatiewerking van $G$ op een eindige verzameling $X$
geeft de \emph{permutatierepresentatie} op $\C^X$: $\rho(g)e_x
= e_{g\cdot x}$.
(d) $S_n$ werkt op $\C^n$ door coördinaten te permuteren; het
hypervlak $\{\sum x_i = 0\}$ is invariant: de
\emph{standaardrepresentatie}, van graad $n - 1$.
\end{example}

\begin{theorem}[Maschke]\label{thm:b3:representations:maschke}
Elke invariante deelruimte $W$ van een \omterm{def:b3:representations:rep}{representatie} $(V, \rho)$
laat een invariant complement toe. Bijgevolg is elke
\omterm{def:b3:representations:rep}{representatie} een directe som van irreducibele
(\emph{semisimpliciteit}).\index{stelling van Maschke} Zonder Maschke
zou de rest van de theorie --- \omterm{def:b3:representations:character}{karakters}, tabellen, Burnside ---
op drijfzand rusten.
\end{theorem}

\begin{proof}
Zij $p \colon V \to V$ \emph{enige} projectie met beeld $W$
(kies enig complement). Middel over de groep:
\[
\tilde p = \frac1{\abs G}\sum_{g \in G} \rho(g)\,p\,\rho(g)^{-1}.
\]
Elke term beeldt $V$ af in $W$ ($W$ invariant), en fixeert $W$
puntsgewijs: voor $w \in W$, $\rho(g)^{-1}w \in W$, $p$ fixeert
het, en $\rho(g)$ brengt het terug --- dus is $\tilde p$ opnieuw
een projectie op $W$. Ze is equivariant: voor $h \in G$
herindexeert $\rho(h)\tilde p\rho(h)^{-1}$ dezelfde som. Dus is
$\ker \tilde p$ een invariant complement van $W$. Itereren op de
sommanden (eindige dimensie) ontbindt $V$ in \omterm{def:b3:representations:rep}{irreducibelen}.
\end{proof}

\begin{theorem}[Lemma van Schur]\label{thm:b3:representations:schur}
Zij $f \colon V \to W$ een morfisme van \emph{irreducibele}
\omterm{def:b3:representations:rep}{representaties}. Dan is $f = 0$ of $f$ een isomorfisme; en als
$(V,\rho) = (W,\sigma)$, dan $f = \lambda\,\mathrm{id}$ voor
enige $\lambda \in \C$. Dus is $\dim\operatorname{Hom}_G(V, W)$
gelijk aan $1$ als $V \cong W$, $0$ anders.\index{lemma van
Schur}
\end{theorem}

\begin{proof}
$\ker f$ en $\operatorname{im} f$ zijn invariant
(equivariantie), dus is elk $0$ of alles: ofwel $f = 0$, ofwel
$f$ is injectief met vol beeld. Als $V = W$: $f$ heeft een
eigenwaarde $\lambda$ ($\C$ \omterm{def:b3:galois:closure}{algebraïsch gesloten}); $f -
\lambda\,\mathrm{id}$ is een niet-injectief morfisme $V \to V$,
dus $0$. Voor de dimensietelling wanneer $V \cong W$: met één
isomorfisme $u$ vast, geeft elk morfisme $f$ het endomorfisme
$u^{-1}f = \lambda\,\mathrm{id}$: $f = \lambda u$. De ruimte
$\operatorname{Hom}_G(V,V)$ is dus eendimensionaal --- de
schaal van de identiteit.
\end{proof}

\section{Karakters en orthogonaliteit}

\begin{definition}\label{def:b3:representations:character}
Het \emph{karakter}\index{karakter} van $(V, \rho)$ is
$\chi_\rho(g) = \operatorname{tr}\rho(g)$. Het vervult
$\chi_\rho(e) = \dim V$, $\chi_\rho(hgh^{-1}) = \chi_\rho(g)$
(sporen zijn conjugatie-invariant): karakters zijn
\emph{klassefuncties}\index{klassefunctie} --- elementen van de
ruimte $\mathcal{CF}(G)$ van functies constant op
\omterm{ex:b3:groups:actions}{conjugatieklassen}, uitgerust met het Hermitische inwendige
product
\[
\langle \varphi, \psi\rangle = \frac1{\abs G}\sum_{g \in G}
\overline{\varphi(g)}\,\psi(g).
\]
\end{definition}

\begin{proposition}\label{prop:b3:representations:charbasics}
$\rho(g)$ is diagonaliseerbaar met eigenwaarden
eenheidswortels; $\chi_\rho(g^{-1}) = \overline{\chi_\rho(g)}$,
en $\abs{\chi_\rho(g)} \leq \chi_\rho(e)$ met gelijkheid desda
$\rho(g)$ een scalair is. \omterm{def:b3:representations:character}{Karakters} tellen op bij directe
sommen: $\chi_{V \oplus W} = \chi_V + \chi_W$.
\end{proposition}

\begin{proof}
$\rho(g)^N = \mathrm{id}$ voor $N = \abs G$ (Lagrange):
$\rho(g)$ annuleert $X^N - 1$, gesplitst met eenvoudige
wortels, dus is ze diagonaliseerbaar met eigenwaarden $\lambda_i
\in \mu_N$. Dan $\chi(g^{-1}) = \sum \lambda_i^{-1} = \sum
\bar\lambda_i = \overline{\chi(g)}$, en $\abs{\chi(g)} =
\abs{\sum\lambda_i} \leq \dim V$, met gelijkheid in de
driehoeksongelijkheid desda alle $\lambda_i$ gelijk zijn,
d.w.z.\ $\rho(g) = \lambda\,\mathrm{id}$. Additiviteit op
directe sommen: bloksporen. Karaktertheorie begint hier: de
spoorfunctie ziet de ontbinding.
\end{proof}

\begin{lemma}\label{lem:b3:representations:homtrace}
Zij $(V, \rho)$, $(W, \sigma)$ \omterm{def:b3:representations:rep}{representaties}. De
middelingsoperator op $\operatorname{Hom}(W, V)$,
\[
c(A) = \frac1{\abs G}\sum_{g}\rho(g)\,A\,\sigma(g)^{-1},
\]
is een projectie op $\operatorname{Hom}_G(W, V)$, en voor de
afbeelding $\Phi_g \colon A \mapsto \rho(g)A\sigma(g)^{-1}$
heeft men $\operatorname{tr}\Phi_g =
\chi_\rho(g)\,\overline{\chi_\sigma(g)}$.
\end{lemma}

\begin{proof}
$c(A)$ is equivariant (herindexeer de som zoals bij Maschke), en
$c$ fixeert equivariante afbeeldingen (elke term is $A$): $c$ is
een projectie met beeld $\operatorname{Hom}_G(W, V)$. Spoor: in
basissen, $\Phi_g(A) = BAC$ met $B = \rho(g)$, $C =
\sigma(g)^{-1}$; op de basis $(E_{kl})$ van matrices,
$BE_{kl}C = \sum_{m,n} b_{mk}c_{ln}E_{mn}$, dus is de
coëfficiënt van $E_{kl}$ in $\Phi_g(E_{kl})$ gelijk aan
$b_{kk}c_{ll}$: $\operatorname{tr}\Phi_g =
\sum_{k,l}b_{kk}c_{ll} = \operatorname{tr}(B)
\operatorname{tr}(C) = \chi_\rho(g)\chi_\sigma(g^{-1})$, en
$\chi_\sigma(g^{-1}) = \overline{\chi_\sigma(g)}$.
\end{proof}

\begin{theorem}[Eerste orthogonaliteitsrelaties]
\label{thm:b3:representations:orthogonality}
Zij $\rho, \sigma$ \emph{\omterm{def:b3:representations:rep}{irreducibel}}. Dan\index{orthogonaliteit
van karakters}
\[
\langle\chi_\sigma, \chi_\rho\rangle =
\begin{cases}
1 & \text{als } \rho \cong \sigma,\\
0 & \text{anders:}
\end{cases}
\]
de irreducibele \omterm{def:b3:representations:character}{karakters} vormen een orthonormale familie in
$\mathcal{CF}(G)$.
\end{theorem}

\begin{proof}
Het spoor van een projectie is de dimensie van haar beeld:
\[
\dim\operatorname{Hom}_G(W, V) = \operatorname{tr} c
= \frac1{\abs G}\sum_g \operatorname{tr}\Phi_g
= \frac1{\abs G}\sum_g \chi_\rho(g)\overline{\chi_\sigma(g)}
= \langle \chi_\sigma, \chi_\rho\rangle,
\]
en het lemma van Schur evalueert de linkerkant tot $\delta_{\rho
\cong \sigma}$.
\end{proof}

\begin{corollary}\label{cor:b3:representations:multiplicity}
Ontbind $V \cong \bigoplus_i V_i^{\oplus m_i}$ in onderscheiden
\omterm{def:b3:representations:rep}{irreducibelen} ($V_i \not\cong V_j$). Dan $m_i = \langle
\chi_{V_i}, \chi_V\rangle$: de multipliciteiten --- dus de
\omterm{def:b3:representations:rep}{representatie} tot op isomorfie --- worden bepaald door het
\omterm{def:b3:representations:character}{karakter}. Bovendien $\langle \chi_V, \chi_V\rangle = \sum_i
m_i^2$; in het bijzonder is $V$ \omterm{def:b3:representations:rep}{irreducibel} desda
$\langle\chi_V, \chi_V\rangle = 1$.
\end{corollary}

\begin{proof}
$\chi_V = \sum m_i\chi_{V_i}$
(\cref{prop:b3:representations:charbasics}); neem inwendige
producten met elke $\chi_{V_i}$ en gebruik orthonormaliteit.
Twee \omterm{def:b3:representations:rep}{representaties} met gelijke \omterm{def:b3:representations:character}{karakters} hebben gelijke
multipliciteiten, zijn dus isomorf.
\end{proof}

\begin{theorem}[De reguliere representatie]
\label{thm:b3:representations:regular}
Zij $\chi_1, \dots, \chi_r$ de onderscheiden irreducibele
\omterm{def:b3:representations:character}{karakters}, van graden $n_i = \chi_i(e)$. De \omterm{ex:b3:representations:examples}{reguliere
representatie} ontbindt met multipliciteiten $m_i = n_i$;
bijgevolg
\[
\sum_{i=1}^{r} n_i^2 = \abs G,
\qquad
\sum_i n_i\chi_i(g) = 0 \quad (g \neq e).
\]
\end{theorem}

\begin{proof}
Het reguliere \omterm{def:b3:representations:character}{karakter}: $\chi_{\mathrm{reg}}(g) = \#\{h : gh =
h\}$, wat $\abs G$ is voor $g = e$ en $0$ anders. Dus $m_i =
\langle \chi_i,\chi_{\mathrm{reg}}\rangle =
\frac1{\abs G}\overline{\chi_i(e)}\,\abs G = n_i$. Evalueren
van $\chi_{\mathrm{reg}} = \sum n_i\chi_i$ in $e$ en in $g \neq
e$ geeft de twee weergegeven identiteiten.
\end{proof}

\begin{theorem}\label{thm:b3:representations:numberirr}
De irreducibele \omterm{def:b3:representations:character}{karakters} vormen een orthonormale \emph{basis}
van $\mathcal{CF}(G)$: het aantal \omterm{def:b3:representations:rep}{irreducibele representaties} is
gelijk aan het aantal \omterm{ex:b3:groups:actions}{conjugatieklassen} van $G$.
\end{theorem}

\begin{proof}
Alleen volledigheid rest: zij $f \in \mathcal{CF}(G)$ orthogonaal
op elke $\chi_i$; we tonen $f = 0$. Voor een \omterm{def:b3:representations:rep}{representatie}
$(V,\rho)$, stel $T_{f,\rho} = \frac{1}{\abs G} \sum_g
\overline{f(g)}\,\rho(g)$. Ze is equivariant: voor $h \in G$,
\[
\rho(h)T_{f,\rho}\rho(h)^{-1} = \frac1{\abs G}\sum_g
\overline{f(g)}\rho(hgh^{-1})
= \frac1{\abs G}\sum_{g'}\overline{f(h^{-1}g'h)}\rho(g') =
T_{f,\rho}
\]
($f$ is een \omterm{def:b3:representations:character}{klassefunctie}). Als $\rho$ \omterm{def:b3:representations:rep}{irreducibel} is van graad
$n$, geeft Schur $T_{f,\rho} = \lambda\,\mathrm{id}$ met
\[
\lambda = \frac{\operatorname{tr}T_{f,\rho}}{n}
= \frac{1}{n\abs G}\sum_g \overline{f(g)}\,\chi_\rho(g)
= \frac1n\,\langle f, \chi_\rho\rangle = 0 .
\]
Dus $T_{f,\rho} = 0$ op elke irreducibele, dus (directe sommen)
op \emph{elke} \omterm{def:b3:representations:rep}{representatie} --- in het bijzonder op de
reguliere. Pas toe op de basisvector $e_e$: $0 =
T_{f,\mathrm{reg}}e_e = \frac1{\abs G}\sum_g\overline{f(g)}e_g$,
wat elke $\overline{f(g)} = 0$ dwingt. Dus spant de orthonormale
familie $(\chi_i)$ $\mathcal{CF}(G)$ op, waarvan de dimensie het
aantal \omterm{ex:b3:groups:actions}{conjugatieklassen} is.
\end{proof}

\begin{corollary}[Kolomorthogonaliteit]
\label{cor:b3:representations:column}
Voor $g, h \in G$:
\[
\sum_{i=1}^{r}\overline{\chi_i(g)}\,\chi_i(h) =
\begin{cases}
\abs{Z_G(g)} & \text{als } g, h \text{ geconjugeerd zijn},\\
0 & \text{anders.}
\end{cases}
\]
\end{corollary}

\begin{proof}
Zij $g_1, \dots, g_r$ representanten van de klassen, $c_j$ de
klassegroottes. De $r \times r$-matrix $U_{ij} =
\sqrt{c_j/\abs G}\;\chi_i(g_j)$ heeft orthonormale rijen
(\cref{thm:b3:representations:orthogonality} klassegewijs
geschreven: $\sum_j \frac{c_j}{\abs G}\chi_i(g_j)
\overline{\chi_{i'}(g_j)} = \delta_{ii'}$), d.w.z.\ $UU^* = I$;
een vierkante matrix met $UU^* = I$ heeft ook $U^*U = I$:
kolommen zijn orthonormaal, wat uitpakt tot de weergegeven
identiteit ($\abs G/c_j = \abs{Z_G(g_j)}$, de
baan--stabilisatorrelatie voor conjugatie).
\end{proof}

\begin{proposition}[Eendimensionale karakters; tillen]
\label{prop:b3:representations:onedim}
(a) $G$ is abels desda al haar \omterm{def:b3:representations:rep}{irreducibele representaties} graad
$1$ hebben; het aantal graad-$1$-karakters van enige $G$ is
$[G : D(G)]$ (ze zijn de \omterm{def:b3:representations:character}{karakters} van de abelianisering).
(b) Als $N \trianglelefteq G$, tillen de \omterm{def:b3:representations:rep}{irreducibele
representaties} van $G/N$ (samenstellen met $G \to G/N$) precies
tot die \omterm{def:b3:representations:rep}{irreducibele representaties} van $G$ waarvan de kern $N$
bevat.
\end{proposition}

\begin{proof}
(a) Als $G$ abels is, is elke klasse een singleton: $r = \abs G$,
en $\sum n_i^2 = \abs G$ dwingt alle $n_i = 1$; omgekeerd, als
alle $n_i = 1$, is de \omterm{ex:b3:representations:examples}{reguliere representatie} een som van
eendimensionale, dus is $\rho_{\mathrm{reg}}(G)$ gelijktijdig
diagonaliseerbaar, dus commutatief, en $\rho_{\mathrm{reg}}$ is
trouw: $G$ abels. Graad-$1$-representaties zijn morfismen $G \to
\C^\times$ met abels doel: ze factoreren door $G^{\mathrm{ab}} =
G/D(G)$ (\cref{exo:b3:groups:9}), en onderscheiden \omterm{def:b3:representations:character}{karakters} van
de abelse $G^{\mathrm{ab}}$ tellen $\abs{G^{\mathrm{ab}}}$ (ze
heeft zoveel klassen, alle graden $1$). (b) Samenstelling met de
projectie bewaart irreducibiliteit (invariante deelruimten
corresponderen), en een \omterm{def:b3:representations:rep}{representatie} triviaal op $N$ factoreert
door het quotiënt (\cref{thm:b3:groups:firstiso}).
\end{proof}

\section{Karaktertabellen}

\begin{definition}\label{def:b3:representations:table}
De \emph{karaktertabel}\index{karaktertabel} van $G$ is de $r
\times r$-matrix $\bigl(\chi_i(g_j)\bigr)$: rijen geïndexeerd
door irreducibele \omterm{def:b3:representations:character}{karakters}, kolommen door \omterm{ex:b3:groups:actions}{conjugatieklassen}
(met hun groottes weergegeven). Rijen zijn orthonormaal voor het
gewogen product, kolommen orthogonaal
(\cref{cor:b3:representations:column}): de tabel is zwaar
overbepaald, wat haar berekenbaar maakt.
\end{definition}

\begin{example}[De tabel van $S_3$]\label{ex:b3:representations:s3}
Klassen: $e$ (grootte 1), transposities (3), $3$-cykels (2); dus
$r = 3$ \omterm{def:b3:representations:rep}{irreducibelen}, van graden $n_i$ met $\sum n_i^2 = 6$:
$1, 1, 2$. Graad $1$: triviale $\mathbf 1$ en signatuur
$\varepsilon$. De laatste rij volgt uit \omterm{thm:b3:representations:orthogonality}{kolomorthogonaliteit} (of
uit $\chi_{\mathrm{std}} = \chi_{\mathrm{perm}} - \mathbf 1$):
\[
\begin{array}{c|ccc}
S_3 & e & (1\,2)\ [3] & (1\,2\,3)\ [2]\\
\hline
\mathbf 1 & 1 & 1 & 1\\
\varepsilon & 1 & -1 & 1\\
\chi_{\mathrm{std}} & 2 & 0 & -1
\end{array}
\]
Controle: $\langle\chi_{\mathrm{std}},\chi_{\mathrm{std}}\rangle
= \frac{1}{6}(4 + 0 + 2) = 1$: \omterm{def:b3:representations:rep}{irreducibel}. \omterm{thm:b3:representations:orthogonality}{Kolomorthogonaliteit}
op de transpositiekolom: $1 + (-1) + 0 = 0$, en op de
$3$-cykelkolom: $1 + 1 + (-1) = 1 = \abs{Z_{S_3}((123))}/\cdots$
--- consistent met $\abs{Z} = 3$.
\end{example}

\begin{example}[De tabel van $S_4$]\label{ex:b3:representations:s4}
Klassen: $e$ [1], transposities [6], dubbele transposities [3],
$3$-cykels [8], $4$-cykels [6]: vijf \omterm{def:b3:representations:rep}{irreducibelen}, $\sum n_i^2 =
24$ met twee graad-$1$'s ($\mathbf 1, \varepsilon$; $[S_4 :
D(S_4)] = [S_4 : A_4] = 2$): graden $1, 1, 2, 3, 3$. De
graad-$2$ tilt van $S_4/V \cong S_3$
(\cref{prop:b3:representations:onedim}(b), $V$ de Kleingroep);
graad $3$: de standaardrepresentatie en haar twist door
$\varepsilon$:
\[
\begin{array}{c|ccccc}
S_4 & e\,[1] & (1\,2)\,[6] & (1\,2)(3\,4)\,[3] &
(1\,2\,3)\,[8] & (1\,2\,3\,4)\,[6]\\
\hline
\mathbf 1 & 1 & 1 & 1 & 1 & 1\\
\varepsilon & 1 & -1 & 1 & 1 & -1\\
\chi_2 & 2 & 0 & 2 & -1 & 0\\
\chi_{\mathrm{std}} & 3 & 1 & -1 & 0 & -1\\
\varepsilon\chi_{\mathrm{std}} & 3 & -1 & -1 & 0 & 1
\end{array}
\]
($\chi_{\mathrm{std}}(g) = \operatorname{fix}(g) - 1$;
$\chi_2$ evalueert de $S_3$-tabel op het beeld van elke klasse
mod $V$.) Alle rij- en kolomorthogonaliteitscontroles slagen ---
twee ervan draaien is de opwarming van
\cref{exo:b3:representations:3}.
\end{example}

\begin{method}\label{met:b3:representations:table}
Om een \omterm{def:b3:representations:table}{karaktertabel} te bouwen: (1) som \omterm{ex:b3:groups:actions}{conjugatieklassen} en
groottes op; (2) tel graad-$1$-karakters via $G/D(G)$ en
schrijf ze; (3) vind de overige graden uit $\sum n_i^2 = \abs
G$ (kleine gehele combinatoriek); (4) verkrijg goedkope
\omterm{def:b3:representations:rep}{irreducibelen}: til van quotiënten, trek $\mathbf 1$ af van
permutatiekarakters (controleer $\langle\chi,\chi\rangle = 1$),
vermenigvuldig bekende \omterm{def:b3:representations:character}{karakters} met graad-$1$-en; (5) maak
onbekende rijen af met \omterm{thm:b3:representations:orthogonality}{kolomorthogonaliteit} --- elke kolom is
orthogonaal op de reeds complete kolommen, en kolom $e$ draagt
de graden. Verifieer alles met een volledige
orthogonaliteitsveeg. De overbepaaldheid van de tabel is haar
kracht: elke fout in een entry wordt door de relaties
opgemerkt. In de praktijk vult men eerst de goedkope rijen in
en laat orthogonaliteit de rest dwingen.
\end{method}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:b3:representations:1}
(a) Toon dat de irreducibele \omterm{def:b3:representations:character}{karakters} van $\Z/n\Z$ de
$\chi_k(\bar m) = \eu^{2\iu\pi km/n}$ zijn, $k = 0, \dots, n-1$,
en schrijf de \omterm{def:b3:representations:table}{karaktertabel} van $\Z/4\Z$.
(b) Verifieer beide \omterm{thm:b3:representations:orthogonality}{orthogonaliteitsrelaties} erop --- en herken
de matrix: waar heeft dit boek haar eerder gezien?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b3:representations:2}
Zij $G$ werkend op een eindige verzameling $X$ en $\chi$ het

onderscheiden punten). \emph{(Bereken $\langle\chi,\chi\rangle$
als het aantal banen op $X \times X$.)}
(c) Concludeer dat de standaardrepresentatie van $S_n$ ($n \geq
2$) \omterm{def:b3:representations:rep}{irreducibel} is.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:representations:10}
Zij $\rho$ een \omterm{def:b3:representations:rep}{irreducibele representatie} van graad $n$ en $z

\begin{exercise}[$\star\star$]\label{exo:b3:representations:11}
(Isotypische projecties) Zij $(V, \rho)$ een \omterm{def:b3:representations:rep}{representatie} van
$G$ en $\chi_i$ een \omterm{def:b3:representations:rep}{irreducibel} \omterm{def:b3:representations:character}{karakter} van graad $n_i$.
Definieer
\[
p_i = \frac{n_i}{\abs G}\sum_{g\in G}
\overline{\chi_i(g)}\,\rho(g) \;\in\; \mathcal L(V).
\]
(a) Toon dat $p_i$ $G$-equivariant is, en bereken haar
beperking tot een irreducibele subrepresentatie $W \subseteq V$
van \omterm{def:b3:representations:character}{karakter} $\chi_j$: ze is $\delta_{ij}\, \mathrm{id}_W$
\emph{(Schur; neem sporen om de scalair te identificeren)}.
(b) Leid af dat $p_i$ een projectie is op de som $V_i$ van alle
irreducibele subrepresentaties van \omterm{def:b3:representations:character}{karakter} $\chi_i$ (de
\emph{isotypische component}), dat $\sum_ip_i =
\mathrm{id}_V$, en dat de ontbinding $V = \bigoplus_iV_i$
canoniek is --- in tegenstelling tot de fijnere splitsing van
elke $V_i$ in \omterm{def:b3:representations:rep}{irreducibelen}.
(c) Voor de \omterm{ex:b3:representations:examples}{reguliere representatie} van $S_3$ en het
signatuurkarakter $\varepsilon$, schrijf $p_\varepsilon$
expliciet uit als element van de groepsalgebra en controleer
$p_\varepsilon^2 = p_\varepsilon$ met de hand.
\end{exercise}

\section{Probleem: de $p^aq^b$-stelling van Burnside}

\begin{problem}[{Weekendprobleem --- oplosbaarheid van groepen
van orde $p^aq^b$}]\label{pb:b3:representations:1}
Burnside bewees in 1904 dat elke groep waarvan de orde hoogstens
twee priemfactoren heeft \omterm{def:b3:groups:derived}{oplosbaar} is --- een uitspraak over
abstracte groepen waarvan de enige bekende bewijzen een halve
eeuw door karaktertheorie gingen. Pas veel later verschenen
bewijzen die de \omterm{def:b3:representations:character}{karakters} vermijden; het karakterbewijs blijft
het kortste en het meest instructieve. Dit probleem bouwt het bewijs
volledig, met \cref{ch:b3:groups} (oplosbaarheid),
\cref{ch:b3:modules} (eindig voortgebrachte $\Z$-modulen) en dit
hoofdstuk. Overal zijn $\chi_1, \dots, \chi_r$ de irreducibele
\omterm{def:b3:representations:character}{karakters} van $G$, $n_i = \chi_i(e)$.

\medskip
\noindent\textbf{Deel I --- Algebraïsche gehelen.} Een
\emph{\omterm{def:b3:galois:algebraic}{algebraïsch} geheel} is een wortel van een \emph{monische}
veelterm van $\Z[X]$.
\begin{enumerate}
  \item Toon dat $\alpha$ een \omterm{def:b3:galois:algebraic}{algebraïsch} geheel is desda de
        ring $\Z[\alpha]$ een eindig voortgebrachte
        $\Z$-module is.
  \item Leid af dat de algebraïsche gehelen een deelring van
        $\C$ vormen. \emph{(Als $\Z[\alpha], \Z[\beta]$ \omterm{def:b3:modules:free}{eindig
        voortgebracht} zijn, is $\Z[\alpha, \beta]$ dat ook, en
        submodulen van eindig voortgebrachte $\Z$-modulen zijn
        \omterm{def:b3:modules:free}{eindig voortgebracht}, door
        \cref{thm:b3:modules:submodule}.)}
  \item Toon dat een \emph{rationaal} \omterm{def:b3:galois:algebraic}{algebraïsch} geheel een
        geheel getal is. \emph{(Rationale-wortelstelling.)}
  \item Toon dat elke karakterwaarde $\chi(g)$ een \omterm{def:b3:galois:algebraic}{algebraïsch}
        geheel is.
\end{enumerate}

\medskip
\noindent\textbf{Deel II --- De klassesomrelaties.} Fixeer een
irreducibele $(V, \rho)$ van graad $n$ en \omterm{def:b3:representations:character}{karakter} $\chi$. Voor
een conjugatieklasse $C$, zij $S_C = \sum_{g \in C}\rho(g) \in
\mathcal L(V)$.
\begin{enumerate}[resume]
  \item Toon dat $S_C$ equivariant is, dus $S_C =
        \omega_C\,\mathrm{id}$ met
        \[
        \omega_C = \frac{\abs C\,\chi(g_C)}{n}
        \qquad (g_C \in C \text{ enige representant}).
        \]
  \item Toon dat $S_CS_{C'} = \sum_{C''}
        a_{CC'C''}\,S_{C''}$ waar $a_{CC'C''} \in \N$ voor
        vast $z \in C''$ de paren $(x, y) \in C \times C'$ met
        $xy = z$ telt. Leid af dat de $\omega_C$ voldoen aan
        $\omega_C\,\omega_{C'} = \sum_{C''}
        a_{CC'C''}\,\omega_{C''}$.
  \item Concludeer dat elke $\omega_C$ een \omterm{def:b3:galois:algebraic}{algebraïsch} geheel
        is.
  \item Leid \emph{Frobenius-deelbaarheid} af: $n_i$ deelt
        $\abs G$ voor elke irreducibele graad.
        \emph{(Bereken $\frac{\abs G}{n} = \frac{\abs G}{n}
        \langle\chi,\chi\rangle = \sum_C \omega_C\,
        \overline{\chi(g_C)}$: een \omterm{def:b3:galois:algebraic}{algebraïsch} geheel dat
        rationaal is.)}
\end{enumerate}

\medskip
\noindent\textbf{Deel III --- Burnsides eenvoudcriterium.}
\begin{enumerate}[resume]
  \item Zij $\chi$ \omterm{def:b3:representations:rep}{irreducibel} van graad $n$ en $C$ een klasse
        met $\gcd(\abs C, n) = 1$. Toon met Bézout en vragen
        4--7 dat $\frac{\chi(g_C)}{n}$ een \omterm{def:b3:galois:algebraic}{algebraïsch} geheel
        is.
  \item Stel bovendien $0 < \abs{\chi(g_C)} < n$. Toon dat dit
        onmogelijk is: het algebraïsche geheel $\alpha =
        \chi(g_C)/n$ heeft al haar \emph{conjugaten} van modulus
        $\leq 1$, dus is het product $N = \prod_\sigma\alpha_\sigma$
        een rationaal \omterm{def:b3:galois:algebraic}{algebraïsch} geheel met $0 < \abs N < 1$.
        Concludeer: ofwel $\chi(g_C) = 0$ ofwel $\rho(g_C)$ is
        scalair.
  \item (Criterium van Burnside) Zij $C \neq \{e\}$ een
        conjugatieklasse van \emph{priemmachtorde} $p^k > 1$, en
        stel dat $G$ \omterm{def:b3:groups:simple}{eenvoudig} niet-abels is. \omterm{thm:b3:representations:orthogonality}{Kolomorthogonaliteit}
        op de kolom van $C$ tegen die van $e$ geeft $1 +
        \sum_{i \geq 2} n_i\chi_i(g_C) = 0$. Toon dat enig
        niet-triviaal $\chi_i$ met $p \nmid n_i$ $\chi_i(g_C)
        \neq 0$ heeft; door vraag 10 is $\rho_i(g_C)$ scalair;
        leid een tegenspraak met eenvoud af. Concludeer:
        \emph{geen eenvoudige niet-abelse groep heeft een
        conjugatieklasse van priemmachtorde $> 1$.}
\end{enumerate}

\medskip
\noindent\textbf{Deel IV --- De stelling.}
\begin{enumerate}[resume]
  \item Zij $\abs G = p^aq^b$ met $a + b \geq 1$. Als $G$
        \omterm{def:b3:groups:simple}{eenvoudig} is, toon dat ze abels is: kies $z \neq e$ in
        het \omterm{ex:b3:groups:actions}{centrum} van een Sylow $q$-deelgroep
        (\cref{thm:b3:groups:pfixed}) en beschouw de grootte van
        haar conjugatieklasse $[G : Z_G(z)]$, een macht van $p$;
        pas vraag 11 toe.
  \item Concludeer door inductie op $\abs G$: \emph{elke groep
        van orde $p^aq^b$ is \omterm{def:b3:groups:derived}{oplosbaar}} (Burnside). Waarom breekt
        het argument voor drie priemen --- en moet het, gegeven
        $\abs{A_5} = 2^2\cdot3\cdot5$?
\end{enumerate}

\medskip
\noindent\textbf{Deel V --- De \omterm{def:b3:representations:table}{karaktertabel} van $A_5$.} De
kleinste groep die Burnsides stelling niet raakt verdient haar
volledige portret; alles hieronder gebruikt alleen dit hoofdstuk
plus \cref{exo:b3:groups:11}.
\begin{enumerate}[resume]
  \item Herinner uit \cref{exo:b3:groups:11} de vijf
        \omterm{ex:b3:groups:actions}{conjugatieklassen} van $A_5$: $\{e\}$, de $15$ dubbele
        transposities, de $20$ driecykels, en twee klassen van
        $12$ vijfcykels elk. Verklaar waarom de vijfcykels in
        twee $A_5$-klassen splitsen.
  \item Toon dat de irreducibele graden van $A_5$ precies $1, 3,
        3, 4, 5$ zijn.
  \item Zij $\pi$ het permutatiekarakter van $A_5$ op $\{1,
        \dots, 5\}$. Bereken $\langle\pi, \mathbf 1\rangle$ en
        $\langle\pi, \pi\rangle$, en leid af dat $\chi_4 = \pi
        - \mathbf 1$ \omterm{def:b3:representations:rep}{irreducibel} is van graad $4$.
  \item Hetzelfde spel op de $10$ ongeordende paren: verkrijg
        $\chi_5$ van graad $5$.
  \item De twee overblijvende \omterm{def:b3:representations:rep}{irreducibelen} $\chi_2, \chi_3$
        hebben graad $3$. \omterm{thm:b3:representations:orthogonality}{Kolomorthogonaliteit} bepaalt hun
        waarden buiten de vijfcykels.
  \item Op de vijfcykelklassen: de gulden snede en haar
        geconjugeerde. Stel de volledige \omterm{def:b3:representations:table}{karaktertabel} van $A_5$
        samen.
  \item Draai de controles: rijnormen, Frobenius-deelbaarheid.
  \item Leid uit de tabel alleen af dat $A_5$ \omterm{def:b3:groups:simple}{eenvoudig} is.
  \item (Icosaëdrische coda) $A_5$ is de rotatiegroep van de
        icosaëder; de graad-$3$-representaties zijn de twee
        meetkundige \omterm{def:b3:groups:action}{werkingen} op $\R^3$. Verifieer $1 +
        2\cos\frac{2\pi}5 = \frac{1+\sqrt5}2$. Verklaar
        zonder berekening waarom het andere graad-$3$-karakter
        de geconjugeerde waarde moet dragen: de \omterm{def:b3:galois:galois}{Galoisgroep} van
        $\Q(\sqrt5)/\Q$ werkt op de hele \omterm{def:b3:representations:table}{karaktertabel}.
\end{enumerate}

\medskip
\noindent\textbf{Deel VI --- Complementen: een centrale grens en
het tensorkwadraat.}
\begin{enumerate}[resume]
  \item Zij $\chi$ \omterm{def:b3:representations:rep}{irreducibel} van graad $n$. Toon
        $\abs{\chi(z)} = n$ voor elke $z \in Z(G)$, en leid af
        $n^2 \leq [G : Z(G)]$.
  \item (Tensorkwadraat van $\chi_2$) Leid de karakterformules
        voor $\operatorname{Sym}^2$ en $\Lambda^2$ af, pas toe
        op $\chi_2$ van $A_5$: toon $\chi_2\otimes\chi_2 =
        \mathbf 1 + \chi_2 + \chi_5$.
  \item (Eindcontrole van de tabel) Verifieer
        \omterm{thm:b3:representations:orthogonality}{kolomorthogonaliteit} tussen de twee
        vijfcykelkolommen en de regulariteitsidentiteit.
\end{enumerate}
\end{problem}
