\chapter{De Lebesgue-integraal}\label{ch:b3:lebesgue}

De Riemann-integraal snijdt het \emph{domein} in kleine
intervallen; die van Lebesgue snijdt het \emph{beeld}: om $f$
te integreren, meet de verzamelingen $\{f > t\}$. De
verandering lijkt onschuldig en is revolutionair. Limieten en
integralen, eeuwig ruziënd in de Riemanntheorie (uniforme
convergentie vereist!), worden verzoend door drie
convergenstiestellingen --- monotone convergentie, Fatou,
gedomineerde convergentie --- waarvan de hypotheses bijna
beschamend zwak zijn. Dit hoofdstuk construeert de integraal
over een willekeurige maatruimte $(X, \mathcal A, \mu)$,
bewijst de drie stellingen, regelt de exacte relatie met de
Riemann-integraal (een begrensde functie is
\omterm{def:b3:lebesgue:l1}{Riemann-integreerbaar} d.e.s.d.a.\ zij \omterm{def:b3:lebesgue:l1}{bijna overal} continu is),
en industrialiseert de differentiatie van
parameterafhankelijke integralen --- de techniek die het
weekendprobleem gebruikt om $\int_0^\infty\frac{\sin x}x\,\dd
x$ en $\int_\R \eu^{-x^2}\dd x$ te berekenen.

\section{Meetbare functies}

\begin{definition}\label{def:b3:lebesgue:measurable}
Zijn $(X, \mathcal A)$, $(Y, \mathcal B)$ meetbare ruimten.
$f \colon X \to Y$ is \emph{meetbaar}\index{meetbare functie}
als $f^{-1}(B) \in \mathcal A$ voor elke $B \in \mathcal B$.
Voor reële (of $[-\infty,+\infty]$-waardige) functies draagt $Y
= \R$ haar Borel-$\sigma$-algebra, en het volstaat te
controleren $f^{-1}(\intoo t{+\infty}) = \{f > t\} \in
\mathcal A$ voor alle $t \in \R$: de goede verzamelingen $\{B
: f^{-1}(B) \in \mathcal A\}$ vormen een $\sigma$-algebra
(originelen commuteren met verzamelingsoperaties) die de
voortbrengende stralen bevat (\cref{def:b3:measure:borel},
\cref{met:b3:measure:goodsets}).
\end{definition}

\begin{proposition}\label{prop:b3:lebesgue:stability}
(a) Composities van meetbare afbeeldingen zijn \omterm{def:b3:lebesgue:measurable}{meetbaar};
\omterm{def:b3:topology:continuity}{continue afbeeldingen} zijn \omterm{def:b3:lebesgue:measurable}{Borel-meetbaar}.
(b) Als $f, g \colon X \to \R$ \omterm{def:b3:lebesgue:measurable}{meetbaar} zijn, dan ook $f + g$,
$fg$, $\max(f,g)$, $\abs f$, $\lambda f$.
(c) Als $(f_n)$ \omterm{def:b3:lebesgue:measurable}{meetbaar} zijn met waarden in
$[-\infty, +\infty]$, dan zijn $\sup_nf_n$, $\inf_nf_n$,
$\limsup f_n$, $\liminf f_n$ \omterm{def:b3:lebesgue:measurable}{meetbaar}; als $f_n \to f$
puntsgewijs, is $f$ \omterm{def:b3:lebesgue:measurable}{meetbaar}.
\end{proposition}

\begin{proof}
(a) $(g\circ f)^{-1}(B) = f^{-1}(g^{-1}(B))$; continuïteit
geeft meetbaarheid via de voortbrengende \omterm{def:b3:topology:topology}{open verzamelingen}
(\cref{pb:b3:measure:1}, vraag 10, in algemene vorm).
(b) $(f, g) \colon X \to \R^2$ is \omterm{def:b3:lebesgue:measurable}{meetbaar} voor de
Borel-$\sigma$-algebra van $\R^2$ --- controleer op open
dozen, die voortbrengen (opens van $\R^2$ zijn aftelbare unies
van rationale dozen): $(f,g)^{-1}(U\times V) =
f^{-1}(U)\cap g^{-1}(V)$ --- en $+, \times, \max$ zijn
continu $\R^2 \to \R$: composeer.
(c) $\{\sup f_n > t\} = \bigcup_n\{f_n > t\}$; $\inf =
-\sup(-)$; $\limsup = \inf_N\sup_{n \geq N}$; een puntsgewijze
limiet is haar eigen $\limsup$.
\end{proof}

\begin{definition}\label{def:b3:lebesgue:simple}
Een \emph{eenvoudige functie}\index{eenvoudige functie} is een
\omterm{def:b3:lebesgue:measurable}{meetbare functie} met eindig veel waarden: $s = \sum_{i=1}^n
c_i\,\mathbf 1_{A_i}$, $A_i \in \mathcal A$ disjunct, $c_i
\geq 0$ (voor de niet-negatieve theorie). Haar integraal is
\[
\int s\,\dd\mu = \sum_i c_i\,\mu(A_i) \in [0, +\infty]
\]
(conventie $0\cdot\infty = 0$); de waarde hangt niet af van de
\omterm{def:b3:representations:rep}{representatie} (verfijn twee partities).
\end{definition}

\begin{theorem}[Approximatie door eenvoudige functies]
\label{thm:b3:lebesgue:approximation}
Elke meetbare $f \colon X \to [0, +\infty]$ is de puntsgewijze
limiet van een \emph{toenemende} rij \omterm{def:b3:lebesgue:simple}{eenvoudige functies}:
\[
s_n = \sum_{k=1}^{n2^n} \frac{k-1}{2^n}\,
\mathbf 1_{\{\frac{k-1}{2^n} \leq f < \frac k{2^n}\}}
+ n\,\mathbf 1_{\{f \geq n\}} \nearrow f .
\]
\end{theorem}

\begin{proof}
Elke $s_n$ is \omterm{def:b3:groups:simple}{eenvoudig} (de verzamelingen zijn originelen van
Borelverzamelingen). Monotonie: van $n$ naar $n+1$ splitst elk
dyadisch niveau in twee en verlaagt de toegekende waarde nooit
(een punt met $\frac{k-1}{2^n} \leq f(x) < \frac k{2^n}$ krijgt
ofwel $\frac{2k-2}{2^{n+1}}$ ofwel $\frac{2k-1}{2^{n+1}}$,
beide $\geq \frac{k-1}{2^n}$; de kap $n$ stijgt ook).
Convergentie: als $f(x) < \infty$, dan voor $n > f(x)$ geldt
$f(x) - s_n(x) \leq 2^{-n}$; als $f(x) = \infty$, $s_n(x) = n
\to \infty$.
\end{proof}

\section{De integraal en de convergenstiestellingen}

\begin{definition}\label{def:b3:lebesgue:integral}
Voor meetbare $f \geq 0$:
\[
\int f \,\dd\mu = \sup\Bigl\{\int s\,\dd\mu : s \text{
eenvoudig},
\ 0 \leq s \leq f\Bigr\} \in [0, +\infty].
\]
Zij is monotoon in $f$ per constructie, en breidt het
eenvoudige geval uit (voor eenvoudige $f$ wordt het sup
bereikt bij $f$: vergelijking van eenvoudige integralen via
gemeenschappelijke verfijningen).
\end{definition}

\begin{theorem}[Monotone convergentie, Beppo Levi]
\label{thm:b3:lebesgue:mct}
Als $0 \leq f_n \nearrow f$ puntsgewijs (\omterm{def:b3:lebesgue:measurable}{meetbaar}), dan
\[
\int f_n\,\dd\mu \nearrow \int f\,\dd\mu .
\]
\index{monotone convergentiestelling}
\end{theorem}

\begin{proof}
$f$ is \omterm{def:b3:lebesgue:measurable}{meetbaar} (\cref{prop:b3:lebesgue:stability}(c)) en
$\int f_n$ stijgt naar enige $L \leq \int f$ (monotonie).
Omgekeerd, fixeer een eenvoudige $s = \sum c_i\mathbf 1_{A_i}
\leq f$ en $\theta \in (0,1)$; de verzamelingen $E_n = \{f_n
\geq \theta s\}$ zijn \omterm{def:b3:lebesgue:measurable}{meetbaar} en stijgen naar $X$ (waar $s(x)
> 0$: $f(x) \geq s(x) > \theta s(x)$, dus uiteindelijk $f_n(x)
\geq \theta s(x)$; waar $s(x) = 0$: triviaal). Dan
\[
\int f_n \geq \int_{E_n}\theta s\,\dd\mu
= \theta\sum_i c_i\,\mu(A_i \cap E_n)
\xrightarrow[n\to\infty]{} \theta\sum_ic_i\,\mu(A_i)
= \theta\int s
\]
door continuïteit van onderen (\cref{prop:b3:measure:basics}(c)).
Dus $L \geq \theta\int s$ voor alle $\theta < 1$ en alle
eenvoudige $s \leq f$: $L \geq \int f$.
\end{proof}

\begin{corollary}\label{cor:b3:lebesgue:additivity}
Voor meetbare $f, g \geq 0$ en $c \geq 0$: $\int(f + g) =
\int f + \int g$ en $\int cf = c\int f$; voor een reeks van
niet-negatieve \omterm{def:b3:lebesgue:measurable}{meetbare functies}, $\int\sum_nf_n =
\sum_n\int f_n$.
\end{corollary}

\begin{proof}
Voor \omterm{def:b3:lebesgue:simple}{eenvoudige functies} is additiviteit een berekening op een
gemeenschappelijke verfijning. In het algemeen neem $s_n
\nearrow f$, $t_n \nearrow g$
(\cref{thm:b3:lebesgue:approximation}): $s_n + t_n \nearrow f
+ g$, en MCT transporteert additiviteit naar de limiet. De
reeksuitspraak is MCT toegepast op de partiële sommen.
\end{proof}

\begin{theorem}[Lemma van Fatou]\label{thm:b3:lebesgue:fatou}
Voor meetbare $f_n \geq 0$:
\[
\int \liminf_n f_n \,\dd\mu \;\leq\; \liminf_n \int
f_n\,\dd\mu .
\]
\index{lemma van Fatou}
\end{theorem}

\begin{proof}
Zij $g_N = \inf_{n\geq N}f_n$: \omterm{def:b3:lebesgue:measurable}{meetbaar}, $0 \leq g_N \nearrow
\liminf f_n$, en $g_N \leq f_n$ voor elke $n \geq N$, dus
$\int g_N \leq \inf_{n \geq N}\int f_n$. Pas MCT toe op de
linkerkant: $\int\liminf f_n = \lim_N\int g_N \leq
\lim_N\inf_{n\geq N}\int f_n = \liminf\int f_n$.
\end{proof}

\begin{definition}\label{def:b3:lebesgue:l1}
Een meetbare $f \colon X \to \R$ (of $\C$) is
\emph{integreerbaar}\index{integreerbare functie} als
$\int\abs f\,\dd\mu < \infty$; dan $\int f = \int f^+ - \int
f^-$ (positieve en negatieve delen; reële en imaginaire delen
in het complexe geval). De integraal is lineair op
integreerbare functies (ontleed en combineer positieve delen;
het complexe geval reduceert tot het reële) en voldoet aan
$\abs{\int f} \leq \int\abs f$ (reëel geval: $\pm\int f =
\int(\pm f) \leq \int\abs f$; complex geval: vermenigvuldig met
een unimodulaire constante om de integraal reëel te maken). Een
eigenschap geldt \emph{bijna overal} (b.o.)\index{bijna overal}
als zij alleen faalt op een $\mu$-nulverzameling; $f$ wijzigen
op een nulverzameling verandert geen integraal (het verschil
wordt gedomineerd door $\infty\cdot\mathbf 1_N$, van integraal
$0$).
\end{definition}

\begin{theorem}[Gedomineerde convergentie]\label{thm:b3:lebesgue:dct}
Zij $f_n \to f$ b.o., met $\abs{f_n} \leq g$ b.o.\ voor een
vaste \emph{integreerbare} $g$. Dan is $f$ \omterm{def:b3:lebesgue:l1}{integreerbaar} en
\[
\int f_n\,\dd\mu \longrightarrow \int f\,\dd\mu,
\qquad\text{inderdaad}\quad \int\abs{f_n - f}\,\dd\mu \to 0 .
\]
\index{gedomineerde convergentiestelling}
\end{theorem}

\begin{proof}
Gooi een nulverzameling weg om de hypotheses puntsgewijs te
maken. $\abs f \leq g$: $f$ is \omterm{def:b3:lebesgue:l1}{integreerbaar}. De functies $h_n
= 2g - \abs{f_n - f} \geq 0$ voldoen $\liminf h_n = 2g$; Fatou
geeft
\[
\int 2g \leq \liminf\int\bigl(2g - \abs{f_n - f}\bigr)
= \int 2g - \limsup\int\abs{f_n - f},
\]
dus $\limsup\int\abs{f_n - f} \leq 0$ (de aftrekking is
legaal: $\int 2g < \infty$). Tenslotte $\abs{\int f_n - \int f}
\leq \int\abs{f_n - f} \to 0$.
\end{proof}

\begin{method}\label{met:b3:lebesgue:convergence}
Geconfronteerd met $\lim_n\int f_n$: probeer, in volgorde ---
(1) is de rij monotoon (of een reeks niet-negatieve termen)?
MCT, geen integreerbaarheid nodig. (2) Is er een enkele
integreerbare dominant $g \geq \abs{f_n}$, gevonden door grove
grenzen (``$\sup_n$'' de schattingen)? DCT. (3) Geen
dominatie, geen monotonie? Fatou begrensd nog één kant, en
gelijkheid kan werkelijk falen: de ontsnappende bult $f_n =
n\mathbf 1_{\intoo0{1/n}}$ heeft $\int f_n = 1$ maar $f_n \to
0$ b.o. Dominatie is precies wat massa verbiedt te ontsnappen
naar oneindig, verticaal of horizontaal.
\end{method}

\section{Riemann versus Lebesgue}

\begin{theorem}[Criterium van Lebesgue]
\label{thm:b3:lebesgue:riemann}
Zij $f \colon \intcc ab \to \R$ \emph{begrensd}. Dan is $f$
\omterm{def:b3:lebesgue:l1}{Riemann-integreerbaar} d.e.s.d.a.\ $f$ $\lambda$-bijna overal
continu is; in dat geval is $f$ \omterm{def:b3:lebesgue:l1}{Lebesgue-integreerbaar} en
vallen de twee integralen samen.\index{Riemann versus
Lebesgue-integraal}
\end{theorem}

\begin{proof}
Voor een onderverdeling $\sigma = (a = x_0 < \dots < x_N =
b)$, zij $L_\sigma$ en $U_\sigma$ de trapfuncties die op elk
$\intoo{x_{i-1}}{x_i}$ gelijk zijn aan $m_i =
\inf_{[x_{i-1}, x_i]}f$ en $M_i = \sup$; de Darboux-sommen zijn
hun integralen (Riemann en Lebesgue stemmen overeen op
trapfuncties, beide geven $\sum m_i\Delta x_i$). Neem een rij
onderverdelingen $\sigma_n$, elk de vorige verfijnend, van mesh
$\to 0$, met Darboux-sommen convergerend naar de onderste en
bovenste Darboux-integralen van $f$. De verfijningen maken
$L_{\sigma_n}$ niet-dalend en $U_{\sigma_n}$ niet-stijgend
puntsgewijs buiten de aftelbare verzameling $D$ van alle
deelpunten; noem de limieten $\ell$ en $u$ (\omterm{def:b3:lebesgue:measurable}{meetbaar},
\cref{prop:b3:lebesgue:stability}). Voor $x \notin D$, met
$I_n(x)$ het open $\sigma_n$-interval dat $x$ bevat: $\ell(x)
= \sup_n\inf_{I_n(x)}f$ en $u(x) = \inf_n\sup_{I_n(x)}f$;
aangezien de meshes naar $0$ krimpen, zijn dit de onderste en
bovenste \emph{enveloppes} van $f$ in $x$ --- $u(x) - \ell(x)$
is de oscillatie van $f$ in $x$ --- zodat \emph{$\ell(x) =
u(x)$ d.e.s.d.a.\ $f$ continu is in $x$}. Door MCT/DCT
(begrensd, eindig interval):
\[
\int_{\intcc ab}\ell\,\dd\lambda = \lim_n\int L_{\sigma_n}
= \underline{\int}f,
\qquad
\int_{\intcc ab}u\,\dd\lambda = \overline{\int}f .
\]
$f$ \omterm{def:b3:lebesgue:l1}{Riemann-integreerbaar} $\iff$ $\underline\int f =
\overline\int f$ $\iff$ $\int(u - \ell) = 0$ $\iff$ $u =
\ell$ b.o.\ ($u - \ell \geq 0$; \cref{exo:b3:lebesgue:5})
$\iff$ $f$ continu b.o. In dat geval $\ell \leq f \leq u$ met
$\ell = u$ b.o.: $f$ is b.o.\ gelijk aan de meetbare $\ell$,
dus \omterm{def:b3:lebesgue:measurable}{Lebesgue-meetbaar} (volledigheid van $\lambda$) met $\int
f\,\dd\lambda = \int\ell\,\dd\lambda = \underline\int f =
\int_a^bf$.
\end{proof}

\begin{example}\label{ex:b3:lebesgue:riemannexamples}
$\mathbf 1_\Q$ is nergens continu: niet \omterm{def:b3:lebesgue:l1}{Riemann-integreerbaar}
--- maar Lebesgue-triviaal: $\int\mathbf 1_\Q\,\dd\lambda =
\lambda(\Q) = 0$. De functie van Thomae ($\,\frac1q$ bij
rationalen $\frac pq$, $0$ elders) is precies continu in de
irrationalen: \omterm{def:b3:lebesgue:l1}{Riemann-integreerbaar} met integraal $0$. En
\emph{oneigenlijke} Riemann-integralen zijn een ander begrip:
$\int_0^\infty\frac{\sin x}x\,\dd x$ convergeert als limiet van
$\int_0^A$ (het weekendprobleem berekent haar $= \frac\pi2$),
maar $\frac{\sin x}x \notin L^1(\intoo0{+\infty})$: de absolute
integraal divergeert als de harmonische reeks
(\cref{exo:b3:lebesgue:6}). De theorie van Lebesgue ruilt
voorwaardelijke convergentie voor robuuste limietstellingen.
\end{example}

\section{Integralen met parameters}

Doorheen is $(X, \mathcal A, \mu)$ een maatruimte, $T$ een
metrische ruimte (de parameter), en $f \colon T \times X \to
\C$ met $f(t, \cdot)$ \omterm{def:b3:lebesgue:l1}{integreerbaar} voor elke $t$; stel $F(t)
= \int_X f(t, x)\,\dd\mu(x)$.

\begin{theorem}[Continuïteit]\label{thm:b3:lebesgue:paramcont}
Stel: $t \mapsto f(t,x)$ is continu in $t_0$ voor b.o.\ $x$, en
er is een integreerbare $g$ met $\abs{f(t,x)} \leq g(x)$ voor
alle $t$ in een buurt van $t_0$ en b.o.\ $x$. Dan is $F$
continu in $t_0$.\index{parameterintegralen}
\end{theorem}

\begin{proof}
Voor elke rij $t_n \to t_0$: $f(t_n, \cdot) \to f(t_0,
\cdot)$ b.o., gedomineerd door $g$: DCT geeft $F(t_n) \to
F(t_0)$; sequentiële continuïteit volstaat in metrische
ruimten (\cref{rem:b3:topology:sequences}).
\end{proof}

\begin{theorem}[Differentiatie onder het integraalteken]
\label{thm:b3:lebesgue:paramdiff}
Zij $T$ een open interval van $\R$. Stel: voor b.o.\ $x$ is $t
\mapsto f(t,x)$ differentieerbaar op $T$, met
\[
\Bigl|\frac{\partial f}{\partial t}(t, x)\Bigr| \leq g(x)
\quad \text{voor alle } t \in T,\ \text{b.o.\ } x,
\]
$g$ \omterm{def:b3:lebesgue:l1}{integreerbaar}. Dan is $F$ differentieerbaar op $T$ met
$F'(t) = \int_X \frac{\partial f}{\partial t}(t,
x)\,\dd\mu(x)$.
\end{theorem}

\begin{proof}
Fixeer $t$ en $h_n \to 0$: de differentiequotiënten
\[
\varphi_n(x) = \frac{f(t + h_n, x) - f(t, x)}{h_n}
\longrightarrow \frac{\partial f}{\partial t}(t,x)
\quad\text{b.o.},
\]
en de middelwaarde-ongelijkheid begrensd $\abs{\varphi_n(x)}
\leq \sup_{s}\abs{\partial_tf(s,x)} \leq g(x)$: DCT is van
toepassing, en $\frac{F(t + h_n) - F(t)}{h_n} = \int\varphi_n
\to \int\partial_t f(t, \cdot)$.
\end{proof}

\begin{example}[De Gamma-functie]\label{ex:b3:lebesgue:gamma}
Voor $t > 0$ zij\index{Gamma-functie}
\[
\Gamma(t) = \int_0^{+\infty} x^{t-1}\eu^{-x}\,\dd x .
\]
De integraal convergeert: nabij $0$ is $x^{t-1}$ \omterm{def:b3:lebesgue:l1}{integreerbaar}
($t > 0$); op oneindig, $x^{t-1}\eu^{-x} \leq C\eu^{-x/2}$.
Partiële integratie (op $[\varepsilon, A]$, dan limieten via
MCT) geeft de functionaalvergelijking $\Gamma(t + 1) =
t\,\Gamma(t)$, vandaar $\Gamma(n+1) = n!$: de faculteit
geïnterpoleerd. Op elk $\intcc ab \subseteq \intoo0{+\infty}$
wordt $\partial_t\bigl(x^{t-1}\eu^{-x}\bigr) = \ln
x\cdot x^{t-1}\eu^{-x}$ gedomineerd door $\abs{\ln x}(x^{a-1}
+ x^{b-1})\eu^{-x}$, \omterm{def:b3:lebesgue:l1}{integreerbaar}: $\Gamma$ is $\mathcal C^1$,
en door inductie $\mathcal C^\infty$, met $\Gamma^{(k)}(t) =
\int_0^\infty(\ln x)^kx^{t-1}\eu^{-x}\dd x$. De waarde
$\Gamma(\frac12) = \sqrt\pi$ is de Gaussische integraal in
vermomming (\cref{pb:b3:lebesgue:1}).
\end{example}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:b3:lebesgue:1}
(a) Toon dat een monotone functie $\R \to \R$
\omterm{def:b3:lebesgue:measurable}{Borel-meetbaar} is, en dat een afgeleide (van een overal
differentieerbare functie) \omterm{def:b3:lebesgue:measurable}{Borel-meetbaar} is.
(b) Toon dat $f \colon X \to \R$ \omterm{def:b3:lebesgue:measurable}{meetbaar} is d.e.s.d.a.\ $\{f
> q\} \in \mathcal A$ voor elke \emph{rationale} $q$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b3:lebesgue:2}
Bereken, met volledige rechtvaardiging:
\[
\lim_{n\to\infty}\int_0^{+\infty}
\frac{\cos x}{(1 + x/n)^{n}}\,\dd x,
\qquad
\lim_{n\to\infty}\int_0^1 \frac{n\,x^{n-1}}{1 + x}\,\dd x .
\]
\emph{(Voor de tweede: substitueer $u = x^n$ alvorens te
domineren.)}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:lebesgue:3}
(a) Vertoon strikte ongelijkheid in het lemma van Fatou.
(b) Vertoon $f_n \to 0$ puntsgewijs met $\int f_n = 1$ op drie
manieren: ontsnapping in hoogte, in breedte, naar oneindig.
Welke enkele hypothese van DCT schendt elk?
(c) Toon dat men in het lemma van Fatou $\liminf$ niet door
$\limsup$ kan vervangen aan weerszijden.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:lebesgue:4}
(a) Toon $\displaystyle\int_0^{+\infty}\frac{x}{\eu^x -
1}\,\dd x = \sum_{n\geq1}\frac1{n^2} = \frac{\pi^2}6$
\emph{(ontwikkel $\frac1{\eu^x - 1}$ in een meetkundige reeks en
integreer term-voor-term --- welke stelling staat dat toe?)}.
(b) (Sophomore's dream) Toon
$\displaystyle\int_0^1 x^{-x}\,\dd x = \sum_{n\geq1}n^{-n}$.
\emph{(Schrijf $x^{-x} = \eu^{-x\ln x} = \sum_k\frac{(-x\ln
x)^k}{k!}$ en bereken $\int_0^1(-x\ln x)^k\dd x$ door te
substitueren $x = \eu^{-u/(k+1)}$, herkennend $\Gamma$.)}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:lebesgue:5}
(a) Toon dat $f \geq 0$ \omterm{def:b3:lebesgue:measurable}{meetbaar} met $\int f\,\dd\mu = 0$
voldoet aan $f = 0$ b.o. \emph{(Beschouw $\{f \geq 1/n\}$ en
de ongelijkheid van Markov: $\mu(\{f \geq a\}) \leq
\frac1a\int f$ --- bewijs haar.)}
(b) Toon dat een integreerbare $f$ eindig b.o.\ is.
(c) Toon dat als $\int_A f\,\dd\mu = 0$ voor \emph{elke}
meetbare $A$, dan $f = 0$ b.o.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:lebesgue:6}
(a) Pas \cref{thm:b3:lebesgue:riemann} toe om
Riemann-integreerbaarheid te beslissen van: $\mathbf 1_\Q$; de
functie van Thomae; $\mathbf 1_K$ voor $K$ een vette
Cantorverzameling (\cref{exo:b3:measure:5}).
(b) Toon dat $\int_1^{+\infty}\abs{\frac{\sin x}x}\,\dd x =
+\infty$, terwijl $\lim_{A\to\infty}\int_1^A\frac{\sin
x}x\,\dd x$ bestaat \emph{(partieel integreren)}: oneigenlijke
convergentie zonder integreerbaarheid.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:lebesgue:7}
Rechtvaardig dat $F(t) = \int_0^{+\infty}\eu^{-x^2}\cos(tx)\,
\dd x$ $\mathcal C^1$ is op $\R$ en voldoet aan $F'(t) =
-\frac t2F(t)$ \emph{(partieel integreren)}; leid af $F(t) =
F(0)\,\eu^{-t^2/4}$. (Met $F(0) = \frac{\sqrt\pi}2$ uit het
weekendprobleem: de Gaussiaan is in wezen haar eigen
Fourier-transformatie --- \cref{ch:b3:fouriertransform} zal dit
systematiseren.)
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b3:lebesgue:8}
(Frullani) Zij $0 < a < b$. Toon
\[
\int_0^{+\infty}\frac{\eu^{-ax} - \eu^{-bx}}{x}\,\dd x =
\ln\frac ba,
\]
door de integrand te schrijven als $\int_a^b \eu^{-xt}\,\dd t$
en de verwisseling te rechtvaardigen via de niet-negatieve
theorie (\cref{cor:b3:lebesgue:additivity} in continue vorm ---
anticipeer Tonelli, of snijd $[a,b]$ in $n$ gelijke delen en
ga naar de limiet).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:lebesgue:9}
Zij $f \geq 0$ \omterm{def:b3:lebesgue:measurable}{meetbaar} op $(X, \mathcal A, \mu)$. Toon dat
$\nu(A) = \int_A f\,\dd\mu$ een \omterm{def:b3:measure:measure}{maat} definieert
(\emph{dichtheid}\index{dichtheid van een maat} $f$ t.o.v.\
$\mu$), en dat $\int g\,\dd\nu = \int gf\,\dd\mu$ voor alle
meetbare $g \geq 0$ \emph{(bewijs het voor indicatoren, dan
\omterm{def:b3:lebesgue:simple}{eenvoudige functies}, dan MCT --- de standaardmachine)}.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b3:lebesgue:10}
(Een Weierstrass-stijl mislukking) Definieer $f(t) =
\int_0^{+\infty}\frac{\sin(tx)}{x(1 + x^2)}\,\dd x$.
(a) Toon dat $f$ welgedefinieerd en continu is op $\R$, en
$\mathcal C^1$ met $f'(t) = \int_0^\infty\frac{\cos(tx)}{1 +
x^2}\dd x$ voor elke $t$ --- maar dat \emph{nogmaals}
differentiëren onder de integraal onwettig is.
(b) Aangenomen $f'(t) = \frac\pi2\eu^{-t}$ voor $t > 0$
(bewwezen in \cref{ch:b3:residues}), wat is
$\int_0^\infty\frac{x\sin(tx)}{1+x^2}\dd x$ voor $t > 0$, en
waarom bevestigt haar formule de mislukking in (a)?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:lebesgue:11}
(Lemma van Scheffé) Zijn $f_n, f \geq 0$ \omterm{def:b3:lebesgue:l1}{integreerbaar} met
$f_n \to f$ b.o.\ en $\int f_n \to \int f$.
(a) Toon dat $\int\abs{f_n - f} \to 0$. \emph{(Pas
gedomineerde convergentie toe op $g_n = (f - f_n)^+ \leq f$, en
schrijf $\int\abs{f_n - f} = 2\int g_n - \int(f - f_n)$.)}
(b) Toon met een voorbeeld dat de hypothese $\int f_n \to \int
f$ niet mag worden weggelaten (een schuivende of concentrerende
bult), en dat de conclusie faalt voor getekende $f_n$ zonder
absolute-waarde-controle: $f_n = n\mathbf 1_{\intoc0{1/n}} -
n\mathbf 1_{\intoc{-1/n}0}$ heeft $f_n \to 0$ b.o., $\int f_n
= 0 \to 0$, maar $\int\abs{f_n} = 2$.
(c) Toepassing (dichtheden): als kansdichtheden $p_n \to p$
b.o., dan automatisch $\int\abs{p_n - p} \to 0$: puntsgewijze
convergentie van dichtheden is $L^1$-convergentie --- een
convergentie-upgrade gratis.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:lebesgue:12}
Klassieke limieten, met volledige rechtvaardiging via MCT/DCT:
\[
\text{(a)}\ \lim_{n\to\infty}\int_0^n\Bigl(1 -
\frac xn\Bigr)^n\eu^{x/2}\,\dd x, \qquad
\text{(b)}\ \lim_{n\to\infty}\int_0^1\frac{n\,x^{n-1}}{1 +
x}\,\dd x,
\]
\[
\text{(c)}\ \lim_{n\to\infty}\int_0^\infty
\frac{\dd x}{(1 + x/n)^n\,x^{1/n}} .
\]
\emph{(Voor (a): $(1 - x/n)^n \nearrow \eu^{-x}$ voor vaste
$x$ --- bewijs de monotonie via $\log$; voor (b), partieel
integreren of substitueer $x = u^{1/n}$ en identificeer een
randconcentratie; voor (c), vind een integreerbare dominant
geldig voor alle $n \geq 2$ door te splitsen bij $x = 1$.)}
\end{exercise}

\section{Probleem: twee gevierde integralen}

\begin{problem}[{Weekendprobleem --- de Gaussische integraal en
Dirichlets integraal, met alleen parameters}]
\label{pb:b3:lebesgue:1}
Twee integralen regeren de toegepaste analyse:
\[
G = \int_{-\infty}^{+\infty}\eu^{-x^2}\dd x = \sqrt\pi,
\qquad
D = \int_0^{+\infty}\frac{\sin x}{x}\,\dd x = \frac\pi2
\]
(de tweede als oneigenlijke integraal,
\cref{ex:b3:lebesgue:riemannexamples}). We bewijzen beide met
alleen de gereedschappen van dit hoofdstuk.

\medskip
\noindent\textbf{Deel I --- De Gaussiaan.} Voor $t \geq 0$ stel
\[
A(t) = \Bigl(\int_0^t\eu^{-x^2}\dd x\Bigr)^{2},
\qquad
B(t) = \int_0^1\frac{\eu^{-t^2(1 + x^2)}}{1 + x^2}\,\dd x .
\]
\begin{enumerate}
  \item Rechtvaardig dat $A$ en $B$ $\mathcal C^1$ zijn op
        $\intoo0{+\infty}$ en bereken $A'$ en $B'$; toon
        $A'(t) + B'(t) = 0$. \emph{(In $B'$, substitueer $u =
        tx$.)}
  \item Bereken $A(0) + B(0)$ en $\lim_{t\to+\infty}(A + B)(t)$
        --- rechtvaardig de limiet onder de integraal in $B$.
  \item Besluit $\int_0^\infty \eu^{-x^2}\dd x =
        \frac{\sqrt\pi}2$, dus $G = \sqrt\pi$, en leid af
        $\Gamma(\tfrac12) = \sqrt\pi$ \emph{(substitueer $x =
        u^2$ in $\Gamma(\frac12)$)}.
\end{enumerate}

\medskip
\noindent\textbf{Deel II --- Dirichlets integraal.} Voor $t
\geq 0$ stel
\[
F(t) = \int_0^{+\infty}\eu^{-tx}\,\frac{\sin x}{x}\,\dd x .
\]
\begin{enumerate}[resume]
  \item Toon dat de integraal die $F(t)$ definieert voor elke
        $t > 0$ convergeert als Lebesgue-integraal, en voor $t
        = 0$ als oneigenlijke integraal; toon dat $D =
        \lim_{A\to\infty}\int_0^A\frac{\sin x}x\dd x$ bestaat
        \emph{(partieel integreren op $[\pi, A]$)}.
  \item Toon dat $F$ $\mathcal C^1$ is op
        $\intoo0{+\infty}$ met
        \[
        F'(t) = -\int_0^{+\infty}\eu^{-tx}\sin x\,\dd x
        = -\frac{1}{1 + t^2}
        \]
        \emph{(dominatie op $[t_0, \infty)$ voor elke $t_0 >
        0$; de laatste integraal door twee partiële
        integraties of complexe exponentialen)}.
  \item Toon $F(t) \to 0$ als $t \to +\infty$, en leid af
        $F(t) = \frac\pi2 - \arctan t$ op $\intoo0{+\infty}$.
  \item Het delicate punt: $D = \lim_{t\to0^+}F(t)$. Bewijs
        het door uniforme controle van de staart: voor $0 \leq
        t \leq 1$ en $A \geq \pi$, partieel integreren om te
        tonen
        \[
        \Bigl|\int_A^{+\infty}\eu^{-tx}\frac{\sin
        x}x\,\dd x\Bigr| \leq \frac{C}{A}
        \]
        met $C$ onafhankelijk van $t$ \emph{(differentiëer
        $\frac{\eu^{-tx}}x$ en begrens $\abs{\cos}$ door $1$;
        noteer $t\eu^{-tx} \leq 1/x\cdot(tx\eu^{-tx})$ met
        $\sup_{u\geq0}u\eu^{-u} < 1$)}; splits dan $F(t) - D$
        in $[0, A]$ (waar DCT van toepassing is als $t \to 0$)
        en $[A, \infty)$.
  \item Besluit: $D = \frac\pi2$.
\end{enumerate}

\medskip
\noindent\textbf{Deel III --- Dividenden.}
\begin{enumerate}[resume]
  \item Bereken $\int_0^{+\infty}\frac{\sin^2x}{x^2}\,\dd x$
        \emph{(partieel integreren en reduceer tot $D$ via
        $\sin 2x = 2\sin x\cos x$)}.
  \item Bereken $\int_0^{+\infty}\frac{1 -
        \cos x}{x^2}\,\dd x$, en controleer de consistentie van
        de twee resultaten.
  \item Voor $a > 0$, bereken
        $\int_0^{+\infty}\frac{\sin(ax)}x\dd x$ en
        $\int_{-\infty}^{+\infty}\eu^{-ax^2}\dd x$, en noteer
        de schaalregels (zij zullen de werkpaarden van
        \cref{ch:b3:fouriertransform} zijn).
  \item Leg precies uit waarom $D$ niet direct door DCT bij $t
        = 0$ behandeld kon worden (geen integreerbare dominant
        op $[0,1]\times[0,\infty)$), en waarom de
        staartsplitsing van vraag 7 de eerlijke substitutie is
        --- dit patroon (``uniforme integreerbaarheid van
        staarten'') keert doorheen de analyse terug.
\end{enumerate}

\medskip
\noindent\textbf{Deel IV --- De Gamma-functie volgens Bohr en
Mollerup.} De functie $\Gamma$
(\cref{ex:b3:lebesgue:gamma}) voldoet $\Gamma(1) = 1$ en
$\Gamma(x+1) = x\Gamma(x)$ --- maar dat doen oneindig veel
andere functies ook (vermenigvuldig met elke $1$-periodieke
wobble). Eén convexiteitsvoorwaarde pinnt $\Gamma$ uniek vast,
en haar diepere identiteiten vallen dan mechanisch uit. Een
positieve functie $f$ op een interval is \emph{log-convex} als
$\log f$ convex is.
\begin{enumerate}[resume]
  \item Toon dat log-convexiteit convexiteit impliceert, dat
        producten van log-convexe functies en hun composities
        met affiene afbeeldingen log-convex zijn, en --- via
        de twee-functies Hölder-ongelijkheid $\int\abs{uv}
        \leq \bigl(\int\abs u^p\bigr)^{1/p}\bigl(\int\abs
        v^q\bigr)^{1/q}$, direct bewezen uit de
        Young-ongelijkheid --- dat $\Gamma$ log-convex is op
        $\intoo0\infty$.
  \item (Hellinglemma) Zij $g$ convex op $\intoo0\infty$
        met $g(n+1) - g(n) = \log n$ voor elk geheel $n
        \geq 1$. Voor $x \in \intoc01$ en $n \geq 2$,
        vergelijk de hellingen van $g$ over $[n-1, n]$, $[n,
        n+x]$ en $[n, n+1]$, en leid af
        \[
        x\log(n-1) \;\leq\; g(n + x) - g(n) \;\leq\; x\log n .
        \]
  \item (Bohr--Mollerup) Zij $f > 0$ met $f(1) = 1$,
        $f(x+1) = xf(x)$, en $\log f$ convex. De recursie
        afwikkelen tot $f(n + x) = x(x+1)\cdots(x + n -
        1)\,f(x)$ en $f(n) = (n-1)!$, leid uit vraag 14 af dat
        voor $x \in \intoc01$
        \[
        f(x) = \lim_{n\to\infty}
        \frac{n!\,n^x}{x(x+1)\cdots(x+n)} :
        \]
        $f$ is uniek, dus $f = \Gamma$, en \emph{Gauss' limietformule}
        geldt (breid uit tot alle $x > 0$ door de recursie).
  \item Definieer de \emph{Beta-functie} $B(x, y) =
        \int_0^1t^{x-1}(1-t)^{y-1}\,\dd t$ ($x, y > 0$).
        Bewijs convergentie, de recursie $B(x+1, y) =
        \frac{x}{x+y}\,B(x, y)$ (partieel integreren), en
        $B(1, y) = \frac1y$.
  \item Toon dat $x \mapsto B(x, y)$ log-convex is (Hölder
        opnieuw), en pas Bohr--Mollerup toe op
        \[
        f(x) = \frac{B(x, y)\,\Gamma(x + y)}{\Gamma(y)}
        \]
        om \emph{Eulers formule} te besluiten: $B(x, y) =
        \dfrac{\Gamma(x)\Gamma(y)}{\Gamma(x+y)}$ --- nergens
        dubbelintegralen.
  \item Bereken $B(\frac12, \frac12)$ direct (substitueer $t
        = \sin^2\theta$) en leid af $\Gamma(\frac12) =
        \sqrt\pi$: de Gaussische integraal van Deel I,
        herwonnen door pure convexiteit. Vergelijk de twee
        bewijzen in één zin elk.
  \item (Duplicatie van Legendre) Toon dat
        \[
        g(x) = \frac{2^{x-1}}{\sqrt\pi}\,
        \Gamma\Bigl(\frac x2\Bigr)
        \Gamma\Bigl(\frac{x+1}2\Bigr)
        \]
        de drie Bohr--Mollerup-hypotheses vervult, en besluit
        $g = \Gamma$, d.w.z.\ $\Gamma(2z) =
        \frac{2^{2z-1}}{\sqrt\pi}\,\Gamma(z)\,\Gamma(z +
        \tfrac12)$ voor alle $z > 0$.
  \item Leid de gesloten vorm $\Gamma\bigl(n + \tfrac12\bigr)
        = \dfrac{(2n)!}{4^n\,n!}\sqrt\pi$ af, en bewijs, door
        het hellinglemma toegepast op $\log\Gamma$ rond grote
        gehele getallen, de asymptotiek
        \[
        \frac{\Gamma(n + \frac12)}{\Gamma(n)\,\sqrt n}
        \longrightarrow 1 .
        \]
  \item Combineer de laatste twee vragen tot de centrale
        binomiale asymptotiek
        \[
        \binom{2n}{n} \sim \frac{4^n}{\sqrt{\pi n}},
        \]
        en verifieer numeriek voor $n = 10$ ($\binom{20}{10} =
        184756$, tegen $4^{10}/\sqrt{10\pi} \approx
        187079$: ratio $\approx 0.988$).
  \item (Synthese) De constante $\sqrt\pi$ is nu verschenen als
        de Gaussische integraal (Deel I), als $B(\frac12,
        \frac12)$ (vraag 18), en binnen duplicatie (vraag 19);
        de centrale binomiale schatting anticipeert zowel
        Stirling (\cref{ch:b3:product}'s weekendprobleem) als
        de Moivre--Laplace. Breng de verbanden in kaart: welke
        uitspraken zijn equivalent met welke, en wat draagt
        elke techniek --- differentiatie onder de integraal
        versus convexiteit --- bij dat de andere niet kan?
\end{enumerate}

\medskip
\noindent\textbf{Deel V --- Drie extra dividenden.}
\begin{enumerate}[resume]
  \item (Wallis, via Beta) Voor $p > -1$, substitueer $t =
        \sin^2\theta$ om te tonen
        \[
        W_p = \int_0^{\pi/2}\sin^p\theta\,\dd\theta
        = \frac12\,B\Bigl(\frac{p+1}2, \frac12\Bigr),
        \]
        en leid uit de Beta-recursie (vraag 16) af dat $W_{n+2}
        = \frac{n+1}{n+2}\,W_n$ voor gehele $n \geq 0$.
        Bereken $W_{2n}$ en $W_{2n+1}$ in gesloten vorm, toon
        $W_{2n+1}/W_{2n} \to 1$ door knijpen, en besluit met
        \emph{Wallis' product}
        \[
        \frac\pi2 = \lim_{n\to\infty}
        \prod_{k=1}^{n}\frac{4k^2}{4k^2 - 1} .
        \]
  \item (De Gaussiaan ontmoet een frequentie) Voor $b \in \R$
        stel
        \[
        \Phi(b) = \int_{-\infty}^{+\infty}
        \eu^{-x^2}\cos(2bx)\,\dd x .
        \]
        Toon dat $\Phi$ $\mathcal C^1$ is op $\R$, dat
        partiële integratie de differentiaalvergelijking
        $\Phi'(b) = -2b\,\Phi(b)$ oplevert, en besluit
        \[
        \Phi(b) = \sqrt\pi\,\eu^{-b^2} :
        \]
        de Gaussiaan reproduceert zichzelf onder deze
        transformatie --- de enkele identiteit waarop
        \cref{ch:b3:fouriertransform} zal lopen.
  \item (Frullani's integraal) Voor $0 < a < b$, toon dat
        \[
        \int_0^{+\infty}
        \frac{\eu^{-ax} - \eu^{-bx}}{x}\,\dd x
        = \log\frac ba ,
        \]
        door te differentiëren in de parameter $a$
        (rechtvaardig de dominatie op elk $\intco{a_0}{+\infty}$,
        $a_0 > 0$, en identificeer de constante door $a \to b$
        te laten). Waar precies heeft de integrand haar
        ophefbare singulariteit in $x = 0$ nodig?
\end{enumerate}
\end{problem}
