\chapter{Conforme afbeeldingen en de afbeeldingsstelling van
Riemann}
\label{ch:b3:conformal}

Een holomorfe bijectie tussen twee gebieden transporteert alle
complexe analyse van het ene naar het andere: zulke afbeeldingen
--- \emph{conform}, omdat zij hoeken bewaren --- zijn de
isomorfismen van de holomorfe wereld. Dit hoofdstuk
classificeert ze waar classificatie mogelijk is (de schijf, het
vlak: het lemma van Schwarz is de sleutel, een ongelijkheid van
verbluffende kracht), bouwt de compactheidstheorie van holomorfe
families (Montel), en bewijst de diepste bestaansstelling van
het vak: \emph{elk} \omterm{def:b3:conformal:simplyconnected}{enkelvoudig samenhangend} proper deelgebied
van $\C$, hoe gekarteld ook de rand, is conform equivalent met
de eenheidsschijf. We sluiten met \omterm{prop:b3:conformal:harmonicholo}{harmonische functies} en de
\omterm{thm:b3:conformal:poisson}{Poissonkern}, het Dirichlet-probleem op de schijf oplossend ---
de analytische opbrengst van conforme meetkunde. Doorheen,
$\mathbb D = D(0,1)$ en $\mathbb H = \{\operatorname{Im}z >
0\}$.

\section{Conforme afbeeldingen; Möbiustransformaties}

\begin{definition}\label{def:b3:conformal:conformal}
Een \emph{conforme afbeelding}\index{conforme afbeelding} (of
biholomorfisme) tussen \omterm{def:b3:topology:topology}{open verzamelingen} is een holomorfe
bijectie; haar inverse is automatisch \omterm{def:b3:holomorphic:holo}{holomorf}
(\cref{cor:b3:residues:openmapping}). Twee gebieden zijn
\emph{conform equivalent} als zo'n afbeelding bestaat;
$\operatorname{Aut}(\Omega)$ duidt de groep van conforme
zelfafbeeldingen aan. Waar $f' \neq 0$ --- overal, voor
injectieve $f$ (bewijs van
\cref{cor:b3:residues:openmapping}) --- is de differentiaal
vermenigvuldiging met $f'(z) \neq 0$: een gelijkvormigheid, dus
conforme afbeeldingen bewaren hoeken tussen krommen, inclusief
oriëntatie.
\end{definition}

\begin{example}[Möbiustransformaties]
\label{ex:b3:conformal:mobius}
Voor $\bigl(\begin{smallmatrix}a & b\\ c &
d\end{smallmatrix}\bigr) \in GL_2(\C)$ is de
\emph{Möbiustransformatie}\index{Möbiustransformatie} $z
\mapsto \frac{az + b}{cz + d}$ conform van
$\C\setminus\{-d/c\}$ op $\C\setminus\{a/c\}$ (inverse van
hetzelfde type, uit de inversematrix; samenstelling
correspondeert met matrixproduct). De
\emph{Cayley-afbeelding}\index{Cayley-transformatie}
\[
\varphi(z) = \frac{z - \iu}{z + \iu}
\]
beeldt $\mathbb H$ conform af op $\mathbb D$: immers
$\abs{z - \iu} < \abs{z + \iu}$ precies wanneer $z$ dichter
bij $\iu$ is dan bij $-\iu$, d.w.z.\ $\operatorname{Im}z >
0$; de inverse is $w \mapsto \iu\frac{1 + w}{1 - w}$.
Möbiusafbeeldingen sturen de familie cirkels-en-lijnen naar
zichzelf (\cref{exo:b3:conformal:1}).
\end{example}

\begin{example}[De Joukowski-afbeelding]\label{ex:b3:conformal:joukowski}
Voorbij Möbius is de nuttigste \omterm{def:b3:conformal:conformal}{conforme afbeelding} van de
klassieke toegepaste wiskunde
\[
J(z) = \frac12\Bigl(z + \frac1z\Bigr) .
\]
Op het exterieur $\Omega = \{\abs z > 1\}$ van de
eenheidsschijf is $J$ injectief: $J(z) = J(w)$ geeft $(z -
w)(1 - \frac1{zw}) = 0$ en $\abs{zw} > 1$. Haar afgeleide
$J'(z) = \frac12(1 - z^{-2})$ verdwijnt slechts in $z =
\pm1$, op de rand: $J$ is een conforme equivalentie van
$\Omega$ op haar beeld, dat is $\C\setminus\intcc{-1}1$ ---
de eenheidscirkel zelf wordt twee-op-één op het segment
gevouwen ($J(\eu^{\iu\theta}) = \cos\theta$). Dus het
exterieur van een \emph{segment}, een snijvlak zonder gladde
rand, is conform het exterieur van een schijf: hoeken zijn
geen obstakel voor conforme equivalentie, slechts voor
randgladheid. Beelden van cirkels door $\pm1$ maar
\omterm{ex:b3:groups:actions}{uit-centrum} zijn vleugelvormige krommen, en samenstellen van
$J$ met Möbiusafbeeldingen transporteerde de stroming om een
cilinder --- met de hand berekenbaar --- naar de stroming om
een vleugel: voor de eerste helft van de twintigste eeuw
\emph{was} dit voorbeeld aerodynamica. Het is ook de deur
naar Chebyshev: $J$ conjugueert $z \mapsto z^n$ naar het
Chebyshevpolynoom $T_n$ (\cref{pb:b3:hilbert:1}, Deel~V),
omdat $J(z^n) = \cos(n\theta)$ wanneer $z =
\eu^{\iu\theta}$.
\end{example}

\section{Het lemma van Schwarz en automorfismegroepen}

\begin{theorem}[Lemma van Schwarz]\label{thm:b3:conformal:schwarz}
Zij $f \colon \mathbb D \to \mathbb D$ \omterm{def:b3:holomorphic:holo}{holomorf} met $f(0) =
0$. Dan\index{lemma van Schwarz}
\[
\abs{f(z)} \leq \abs z \ \ (z \in \mathbb D)
\qquad\text{en}\qquad \abs{f'(0)} \leq 1 ,
\]
en als $\abs{f(z_0)} = \abs{z_0}$ voor één $z_0 \neq 0$, of
$\abs{f'(0)} = 1$, dan is $f(z) = \eu^{\iu\theta}z$ een
rotatie.
\end{theorem}

\begin{proof}
$g(z) = f(z)/z$ zet \omterm{def:b3:holomorphic:holo}{holomorf} voort naar $\mathbb D$ (de
singulariteit in $0$ is ophefbaar: $g$ is begrensd bij $0$,
\cref{thm:b3:residues:riemanncw}; haar waarde in $0$ is
$f'(0)$). Op $\abs z = r < 1$: $\abs g \leq \frac1r$, dus
door het maximumprincipe (\cref{thm:b3:holomorphic:maximum})
$\abs g \leq \frac1r$ op $\bar D(0,r)$; laat $r \to 1$:
$\abs g \leq 1$ op $\mathbb D$, wat beide ongelijkheden is.
Gelijkheid in een inwendig punt maakt dat $\abs g$ een
inwendig maximum bereikt: $g$ constant van modulus $1$.
\end{proof}

\begin{theorem}[Automorfismen van de schijf]
\label{thm:b3:conformal:autdisc}
Voor $a \in \mathbb D$ is de \emph{\omterm{thm:b3:conformal:autdisc}{Blaschke-factor}}%
\index{Blaschke-factor}
\[
\varphi_a(z) = \frac{z - a}{1 - \bar a z}
\]
een automorfisme van $\mathbb D$ dat $a$ en $0$ verwisselt,
met $\varphi_a^{-1} = \varphi_{-a}$. Elk automorfisme van
$\mathbb D$ is $\eu^{\iu\theta}\varphi_a$ voor unieke
$\theta \in \R/2\pi\Z$, $a \in \mathbb D$.
\end{theorem}

\begin{proof}
Op $\abs z = 1$: $\abs{1 - \bar az} = \abs{\bar z}\abs{1 -
\bar az} = \abs{\bar z - \bar a\abs z^2} = \abs{\bar z -
\bar a} = \abs{z - a}$, dus $\abs{\varphi_a} = 1$ daar; door
het maximumprincipe $\varphi_a(\mathbb D) \subseteq
\bar{\mathbb D}$, en openheid plaatst het beeld in $\mathbb
D$. De algebraïsche identiteit
$\varphi_{-a}\circ\varphi_a = \mathrm{id}$ (directe
berekening) toont bijectiviteit. Nu zij $f \in
\operatorname{Aut}(\mathbb D)$ en $a = f^{-1}(0)$: $g =
f\circ\varphi_{-a}$ is een automorfisme dat $0$ fixeert.
Schwarz toegepast op $g$ en op $g^{-1}$: $\abs{g(z)} \leq
\abs z$ en $\abs{g^{-1}(w)} \leq \abs w$, dus $\abs{g(z)} =
\abs z$: rotatie, $g = \eu^{\iu\theta}\,\mathrm{id}$,
d.w.z.\ $f = \eu^{\iu\theta}\varphi_a$. Uniciteit: $a =
f^{-1}(0)$ en $\theta$ uit $f'$-type evaluatie (of uit
$f(0) = -\eu^{\iu\theta}a$ en één extra waarde).
\end{proof}

\begin{theorem}[Automorfismen van het vlak]
\label{thm:b3:conformal:autplane}
$\operatorname{Aut}(\C) = \{z \mapsto az + b : a \in \C^*,\
b \in \C\}$. Bijgevolg zijn $\C$ en $\mathbb D$ niet conform
equivalent.
\end{theorem}

\begin{proof}
Zij $f \in \operatorname{Aut}(\C)$ en beschouw $g(z) =
f(1/z)$ op $\C^*$: een \omterm{def:b3:holomorphic:holo}{holomorfe functie} met een \omterm{def:b3:residues:singularities}{geïsoleerde
singulariteit} in $0$. Als zij essentieel was, zou
Casorati--Weierstrass (\cref{thm:b3:residues:riemanncw})
$g\bigl(D(0,\varepsilon)\setminus\{0\}\bigr)$ \omterm{def:b3:topology:interior}{dicht} maken,
terwijl $f(D(0, 1))$ open is en er disjunct van ($f$
injectief: de twee verzamelingen zijn beelden van disjuncte
verzamelingen) --- onmogelijk voor een \omterm{def:b3:topology:interior}{dichte verzameling} en
een niet-lege \omterm{def:b3:topology:topology}{open verzameling}. Dus $0$ is een pool of
ophefbaar voor $g$, d.w.z.\ $\abs{f(z)}$ heeft hoogstens
polynomiale groei: $f$ is een polynoom
(\cref{exo:b3:holomorphic:4}(a)). Injectiviteit forceert
graad $1$: een polynoom van hogere graad heeft ofwel een
meervoudige wortel van $f - c$ ergens ($f'$ verdwijnt) of
meerdere verschillende originelen (d'Alembert--Gauss,
\cref{pb:b3:holomorphic:1}); in beide gevallen faalt
injectiviteit. Ten slotte zou een conforme $\C \to \mathbb
D$ een begrensde gehele functie zijn: constant (Liouville)
--- geen equivalentie.
\end{proof}

\section{De stelling van Montel}

\begin{theorem}[Montel]\label{thm:b3:conformal:montel}
Zij $\mathcal F \subseteq \mathcal H(\Omega)$ \emph{lokaal
begrensd}: elk punt heeft een \omterm{def:b3:topology:topology}{omgeving} waarop
$\sup_{f\in\mathcal F}\sup\abs f < \infty$. Dan heeft elke
rij van $\mathcal F$ een deelrij die uniform convergeert op
alle compacte deelverzamelingen van $\Omega$ (naar een
holomorfe limiet).\index{stelling van Montel}
\end{theorem}

\begin{proof}
Lokale \omterm{def:b3:complete:equicontinuous}{equicontinuïteit}: als $\abs f \leq M$ op $D(a, 2r)
\subseteq \Omega$ voor alle $f \in \mathcal F$, geeft de
Cauchy-formule, voor $z, z' \in D(a, r)$,
\[
\abs{f(z) - f(z')} = \frac{\abs{z - z'}}{2\pi}
\Bigl|\int_{C_{2r}}\frac{f(w)\,\dd w}{(w-z)(w-z')}\Bigr|
\leq \frac{\abs{z - z'}\,2\pi\cdot2r\,M}{2\pi\,r^2}
= \frac{2M}{r}\,\abs{z - z'} :
\]
een uniforme Lipschitzgrens. Exhausteer $\Omega$ door
compacta $K_m$; elk $K_m$ wordt bedekt door eindig veel
zulke schijven, dus $\mathcal F$ is uniform begrensd en
\omterm{def:b3:complete:equicontinuous}{equicontinu} op $K_m$: Arzelà--Ascoli
(\cref{thm:b3:complete:ascoli}) extraheert een deelrij die
uniform convergeert op $K_m$; diagonaliseer over $m$. De
limiet is \omterm{def:b3:holomorphic:holo}{holomorf} door
\cref{thm:b3:holomorphic:weierstrassconv}.
\end{proof}

\section{De afbeeldingsstelling van Riemann}

\begin{definition}\label{def:b3:conformal:simplyconnected}
Een open \omterm{def:b3:topology:connected}{samenhangend} $\Omega \subseteq \C$ is
\emph{enkelvoudig samenhangend}\index{enkelvoudig samenhangend
gebied} (in de homologische zin, voldoende voor al onze
doelen) als $\operatorname{Ind}_\gamma(w) = 0$ voor elke
cykel $\gamma$ in $\Omega$ en elke $w \notin \Omega$ ---
``geen cykel van $\Omega$ omringt een gat''. Door de globale
stelling van Cauchy (\cref{thm:b3:residues:globalcauchy}) en
\cref{prop:b3:holomorphic:primitive} heeft op zulk $\Omega$
\emph{elke \omterm{def:b3:holomorphic:holo}{holomorfe functie} een primitieve}; dus heeft elke
nultvrije $f \in \mathcal H(\Omega)$ een holomorfe
logaritme ($\exp\circ$(primitieve van $f'/f$), aangepast met
een constante, omdat $(f\eu^{-L})' = 0$) en holomorfe
$n$-de wortels $\eu^{L/n}$.\index{holomorfe logaritme}
\end{definition}

\begin{theorem}[Afbeeldingsstelling van Riemann]
\label{thm:b3:conformal:rmt}
Elk \omterm{def:b3:conformal:simplyconnected}{enkelvoudig samenhangend} open $\Omega \subsetneq \C$,
$\Omega \neq \varnothing$, is conform equivalent met
$\mathbb D$; gegeven $z_0 \in \Omega$, is er een unieke
conforme $f \colon \Omega \to \mathbb D$ met $f(z_0) = 0$
en $f'(z_0) > 0$.\index{afbeeldingsstelling van Riemann}
\end{theorem}

\begin{proof}
\emph{Stap 0: de familie is niet-leeg.} Kies $b \notin
\Omega$: $z - b$ is nultvrij op $\Omega$, dus heeft een
holomorfe vierkantswortel $h$ ($h^2 = z - b$). $h$ is
injectief ($h(z) = h(z')$ kwadrateert tot $z = z'$), en als
$w \in h(\Omega)$ dan $-w \notin h(\Omega)$ ($h(z) =
-h(z')$ kwadrateert ook tot $z = z'$, gevend $w = -w = 0$,
onmogelijk omdat $h$ nultvrij is). Omdat $h(\Omega)$ open
is, bevat het een schijf $D(h(z_0), \rho)$; dan
$D(-h(z_0), \rho) \cap h(\Omega) = \varnothing$, d.w.z.\
$\abs{h(z) + h(z_0)} \geq \rho$ voor elke $z \in \Omega$.
Dus
\[
g(z) = \frac{\rho}{2\,\bigl(h(z) + h(z_0)\bigr)}
\]
is \omterm{def:b3:holomorphic:holo}{holomorf}, injectief (een Möbiusafbeelding samengesteld
met de injectieve $h$), met $\abs g \leq \frac12 < 1$.
Samenstellen met een \omterm{thm:b3:conformal:autdisc}{Blaschke-factor}
(\cref{thm:b3:conformal:autdisc}) om $g(z_0)$ naar $0$ te
verplaatsen, is de familie
\[
\mathcal F = \{f \colon \Omega \to \mathbb D \text{
holomorf, injectief, } f(z_0) = 0\}
\]
niet-leeg.

\emph{Stap 1: een extremale element.} Zij $s =
\sup_{\mathcal F}\abs{f'(z_0)} \in \intoc0{+\infty}$ ($>
0$: leden zijn injectief, dus $f'(z_0) \neq 0$). Neem $f_n
\in \mathcal F$ met $\abs{f_n'(z_0)} \to s$: de familie is
begrensd door $1$, dus Montel (\cref{thm:b3:conformal:montel})
extraheert $f_n \to f$ uniform op compacta; $f$ is
\omterm{def:b3:holomorphic:holo}{holomorf}, $f(z_0) = 0$, $\abs{f'(z_0)} = s$
(\cref{thm:b3:holomorphic:weierstrassconv} voor de
afgeleiden), in het bijzonder is $f$ niet-constant; $f$ is
injectief door Hurwitz (\cref{exo:b3:residues:8}(b)), en
$f(\Omega) \subseteq \bar{\mathbb D}$, dus $\subseteq
\mathbb D$ (open afbeelding). Dus $f \in \mathcal F$
bereikt het supremum: $s < \infty$.

\emph{Stap 2: de extremale afbeelding is surjectief.} Stel
$a \in \mathbb D\setminus f(\Omega)$. Het
Blaschke-transport $\varphi_a\circ f$ is nultvrij op het
enkelvoudig samenhangende $\Omega$: het heeft een holomorfe
vierkantswortel $F$ (met $F(\Omega) \subseteq \mathbb D$,
omdat $\abs F^2 = \abs{\varphi_a\circ f} < 1$), injectief
(kwadraten onderscheiden). Normaliseer: $G =
\varphi_{F(z_0)}\circ F \in \mathcal F$. Ontbinden: $f =
\varphi_{-a}\circ s_2 \circ \varphi_{-F(z_0)}\circ G$ waar
$s_2(w) = w^2$; de afbeelding $\Psi =
\varphi_{-a}\circ s_2\circ\varphi_{-F(z_0)} \colon \mathbb D
\to \mathbb D$ is \omterm{def:b3:holomorphic:holo}{holomorf} met $\Psi(0) = f(z_0) = 0$ en is
\emph{geen} rotatie (zij is niet injectief: $s_2$ is dat
niet). Het lemma van Schwarz (strikt geval):
$\abs{\Psi'(0)} < 1$, en de kettingregel $f = \Psi\circ G$
geeft $\abs{f'(z_0)} = \abs{\Psi'(0)}\,\abs{G'(z_0)} <
\abs{G'(z_0)}$ --- in tegenspraak met maximaliteit (merk op
$G \in \mathcal F$). Dus $f$ is surjectief: een conforme
equivalentie.

\emph{Stap 3: normalisatie en uniciteit.} Vermenigvuldig $f$
met $\eu^{-\iu\arg f'(z_0)}$ om $f'(z_0) > 0$ te maken (dit
blijft in $\mathcal F$). Als $f_1, f_2$ beide werken, $\psi
= f_2\circ f_1^{-1} \in \operatorname{Aut}(\mathbb D)$
fixeert $0$ met $\psi'(0) = f_2'(z_0)/f_1'(z_0) > 0$; door
\cref{thm:b3:conformal:autdisc} is $\psi$ een rotatie
$\eu^{\iu\theta}$ met $\eu^{\iu\theta} > 0$: $\psi =
\mathrm{id}$.
\end{proof}

\begin{remark}\label{rem:b3:conformal:rmtscope}
De stelling is een pure bestaansuitspraak van verbluffende
reikwijdte: een vierkant, een halfvlak, het complement van
een snede, de regio tussen twee rakende cirkels, een
fractaal-randgebied --- alle conform identiek aan $\mathbb
D$. Wat zij niet geeft: enige formule (expliciete
afbeeldingen zijn de uitzondering:
\cref{exo:b3:conformal:5}), randgedrag (een diepere theorie
--- de stelling van Carathéodory --- behandelt het), of
uniciteit van voortzetting naar $\C$ of meervoudig
samenhangende gebieden: de annulus $\{1 < \abs z < 2\}$ is
\emph{niet} conform een gepuncteerde schijf, en annuli van
verschillende straalverhoudingen zijn inequivalent (een
echt moeilijker feit).
\end{remark}

\section{Harmonische functies en de Poissonkern}

\begin{proposition}\label{prop:b3:conformal:harmonicholo}
Zij $\Omega$ \omterm{def:b3:conformal:simplyconnected}{enkelvoudig samenhangend} en $u \colon \Omega
\to \R$ harmonisch\index{harmonische functie} ($\mathcal
C^2$ met $\Delta u = u_{xx} + u_{yy} = 0$). Dan $u =
\operatorname{Re}F$ voor een holomorfe $F$, uniek op een
imaginaire constante na. Bijgevolg is $u$ $\mathcal
C^\infty$, vervult de middelwaarde-eigenschap, en gehoorzaamt
het maximumprincipe (geen strikt inwendig extremum tenzij
constant).
\end{proposition}

\begin{proof}
$g = u_x - \iu u_y$ voldoet aan de
\omterm{prop:b3:holomorphic:cauchyriemann}{Cauchy--Riemannvergelijkingen} ($P = u_x$, $Q = -u_y$: $P_x
= u_{xx} = -u_{yy} = Q_y$ en $P_y = u_{xy} = u_{yx} =
-Q_x$) met continue partiële afgeleiden: $g$ is \omterm{def:b3:holomorphic:holo}{holomorf}
(\cref{prop:b3:holomorphic:cauchyriemann}; de
$\R$-differentieerbaarheid volgt uit $\mathcal C^1$). Zij
$F_0$ een primitieve (enkelvoudige samenhang,
\cref{def:b3:conformal:simplyconnected}); dan heeft
$\operatorname{Re}F_0$ gradiënt $(u_x, u_y)$ ($F_0' = g$
pakt precies dat uit via Cauchy--Riemann voor $F_0$), dus
$u - \operatorname{Re}F_0$ is constant ($\Omega$
\omterm{def:b3:topology:connected}{samenhangend}): pas $F = F_0 + c$ aan. De eigenschappen
dragen over van \cref{thm:b3:holomorphic:maximum} en
\cref{exo:b3:holomorphic:10} (voor het maximumprincipe
toegepast op $u$ zelf, gebruik $\eu^{F}$ zoals daar).
\end{proof}

\begin{theorem}[Poissonformule; Dirichlet-probleem op de
schijf]\label{thm:b3:conformal:poisson}
Voor $0 \leq r < 1$ definieer de \emph{\omterm{thm:b3:conformal:poisson}{Poissonkern}}%
\index{Poissonkern}
\[
P_r(\theta) = \sum_{n\in\Z}r^{\abs n}\eu^{\iu n\theta}
= \frac{1 - r^2}{1 - 2r\cos\theta + r^2} \;>\; 0 .
\]
Zij $g \colon \partial\mathbb D \to \R$ continu, en zet,
voor $z = r\eu^{\iu\varphi} \in \mathbb D$,
\[
u(z) = \frac1{2\pi}\int_0^{2\pi}
P_r(\varphi - t)\,g(\eu^{\iu t})\,\dd t .
\]
Dan is $u$ harmonisch op $\mathbb D$ en zet continu voort
naar $\bar{\mathbb D}$ met randwaarden $g$: de unieke zulke
\omterm{prop:b3:conformal:harmonicholo}{harmonische functie}.\index{Dirichlet-probleem}
\end{theorem}

\begin{proof}
\emph{Kernidentiteiten}: sommeren van twee meetkundige
reeksen,
\[
\sum_{n\in\Z}r^{\abs n}\eu^{\iu n\theta}
= \operatorname{Re}\frac{1 + r\eu^{\iu\theta}}{1 -
r\eu^{\iu\theta}}
= \frac{1 - r^2}{\abs{1 - r\eu^{\iu\theta}}^2},
\]
wat het weergegeven quotiënt is; positiviteit is duidelijk,
en $\frac1{2\pi}\int_0^{2\pi}P_r = 1$ (slechts $n = 0$
overleeft).

\emph{Harmoniciteit}: met $z = r\eu^{\iu\varphi}$,
\[
u(z) = \operatorname{Re}\biggl[\frac1{2\pi}\int_0^{2\pi}
\frac{\eu^{\iu t} + z}{\eu^{\iu t} - z}\,
g(\eu^{\iu t})\,\dd t\biggr],
\]
(de geklemde kern heeft reëel deel $P_r(\varphi - t)$:
bereken), en de klem is \omterm{def:b3:holomorphic:holo}{holomorf} in $z$ op $\mathbb D$
(\cref{exo:b3:holomorphic:7}): $u$ is het reële deel van
een \omterm{def:b3:holomorphic:holo}{holomorfe functie}, dus harmonisch.

\emph{Randwaarden}: $P_r(\cdot)$ is een benaderde
identiteit als $r \to 1^-$: massa $1$, en voor $\delta \leq
\abs\theta \leq \pi$, $P_r(\theta) \leq \frac{1 - r^2}{1 -
2r\cos\delta + r^2} \to 0$ uniform. De standaardsplitsing
(continuïteit van $g$ bij $\eu^{\iu\varphi_0}$, begrensdheid
elders) geeft $u(r\eu^{\iu\varphi}) \to
g(\eu^{\iu\varphi_0})$ als $r\eu^{\iu\varphi} \to
\eu^{\iu\varphi_0}$, uniform in het randpunt: de
voortzetting is continu. \emph{Uniciteit}: het verschil van
twee oplossingen is harmonisch op $\mathbb D$, continu op de
\omterm{def:b3:topology:interior}{afsluiting}, nul op de rand: door het maximumprincipe
(toegepast op $\pm$ het verschil) verdwijnt het.
\end{proof}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:b3:conformal:1}
(a) Verifieer dat de \omterm{ex:b3:conformal:mobius}{Cayley-afbeelding} $\varphi(z) =
\frac{z - \iu}{z + \iu}$ een bijectie $\mathbb H \to
\mathbb D$ is met de gestelde inverse, en bereken de
beelden van $\iu$, $0$, $1$, $\infty$ (limiet).
(b) Toon dat $z \mapsto 1/z$ cirkels en lijnen naar cirkels
en lijnen afbeeldt. \emph{(Schrijf hun gemeenschappelijke
vergelijking $\alpha\abs z^2 + \bar\beta z + \beta\bar z +
\gamma = 0$, $\alpha, \gamma \in \R$.)} Leid hetzelfde af
voor alle Möbiusafbeeldingen.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b3:conformal:2}
Zij $f \colon \mathbb D \to \mathbb D$ \omterm{def:b3:holomorphic:holo}{holomorf}.
(a) Als $f(0) = 0$ en $f(a) = a$ voor enige $a \neq 0$, toon
$f = \mathrm{id}$.
(b) Als $f$ een automorfisme is met twee verschillende
vaste punten in $\mathbb D$, toon $f = \mathrm{id}$
\emph{(conjugueer met een \omterm{thm:b3:conformal:autdisc}{Blaschke-factor} om tot (a) te
reduceren)}.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:conformal:3}
(Schwarz--Pick) Voor holomorfe $f\colon \mathbb D \to
\mathbb D$, bewijs
\[
\frac{\abs{f'(z)}}{1 - \abs{f(z)}^2} \;\leq\;
\frac{1}{1 - \abs z^2}
\qquad (z \in \mathbb D),
\]
met gelijkheid (in één punt, dus overal) desda $f \in
\operatorname{Aut}(\mathbb D)$.
\emph{(Pas Schwarz toe op $\varphi_{f(z)}\circ
f\circ\varphi_{-z}$.)} Interpretatie: holomorfe
zelfafbeeldingen contracteren de hyperbolische metriek.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:conformal:4}
Vind expliciete conforme equivalenties:
(a) de strook $\{0 < \operatorname{Im}z < \pi\} \to \mathbb
H$;
(b) het kwadrant $\{\operatorname{Re}z > 0,
\operatorname{Im}z > 0\} \to \mathbb H$;
(c) de halfschijf $\mathbb D\cap\mathbb H \to$ een
kwadrant, dan $\to \mathbb H$;
(d) $\mathbb D \to \mathbb D$ dat $\frac12$ naar $0$ stuurt
met positieve afgeleide daar.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:conformal:5}
(a) Toon dat er geen \omterm{def:b3:conformal:conformal}{conforme afbeelding} $\C \to \mathbb D$
of $\C \to \mathbb H$ bestaat, en geen $\mathbb D \to \C$.
(b) Welke van de volgende zijn conform equivalent met
$\mathbb D$? Motiveer via \cref{thm:b3:conformal:rmt} of
een obstructie: een vierkant; $\C\setminus\intoc{-\infty}0$;
$\mathbb D\setminus\{0\}$; $\{1 < \abs z < 2\}$.
\emph{(Voor de laatste twee: een conform beeld van de
gepuncteerde schijf zou over de punctie voortzetten door
\cref{thm:b3:residues:riemanncw}(1) --- ontwikkel dit.)}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:conformal:6}
Zij $\mathcal F = \{f \in \mathcal H(\mathbb D) : f(0) =
1,\ \operatorname{Re}f > 0\}$.
(a) Toon dat $\mathcal F$ lokaal begrensd is.
\emph{(Stel samen met de Cayley-type afbeelding $w \mapsto
\frac{w - 1}{w + 1}$ die het rechterhalfvlak naar $\mathbb
D$ stuurt, en pas Schwarz toe.)}
(b) Leid de \emph{Herglotz-grens} af: $\abs{f(z)} \leq
\frac{1 + \abs z}{1 - \abs z}$ voor $f \in \mathcal F$, met
gelijkheidsmogelijkheden.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b3:conformal:7}
Waar gebruikt het bewijs van \cref{thm:b3:conformal:rmt}
elke hypothese? Traceer: (i) enkelvoudige samenhang
(tweemaal); (ii) $\Omega \neq \C$; (iii) samenhang. Toon
dan dat de stelling \emph{faalt} voor $\Omega = \C$ en voor
de annulus, en wijs aan welke stap van het bewijs in elk
geval breekt.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:conformal:8}
Los het Dirichlet-probleem op $\mathbb D$ op voor de
randdata: (a) $g(\eu^{\iu t}) = \cos t$; (b) $g(\eu^{\iu t})
= \cos^2 t$; (c) $g = \mathbf 1_{\text{bovenste
halvecirkel}}$ --- voor (c) bereken $u(0)$ en interpreteer
via de middelwaarde-eigenschap. \emph{(Ontwikkel $g$ in
Fourierreeksen en gebruik de reeks van $P_r$:
$u(r\eu^{\iu\varphi}) = \sum_n c_n(g)
r^{\abs n}\eu^{\iu n\varphi}$.)}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b3:conformal:9}
(Harnack) Zij $u \geq 0$ harmonisch op $\mathbb D$. Bewijs,
voor $\abs z = r < 1$:
\[
\frac{1 - r}{1 + r}\,u(0) \;\leq\; u(z) \;\leq\;
\frac{1 + r}{1 - r}\,u(0)
\]
\emph{(begrens de \omterm{thm:b3:conformal:poisson}{Poissonkern} tussen $\frac{1-r}{1+r}$ en
$\frac{1+r}{1-r}$; pas de \omterm{def:b3:representations:rep}{representatie} toe op iets
kleinere schijven en ga tot de limiet)}. Leid af: een
\omterm{prop:b3:conformal:harmonicholo}{harmonische functie} op $\C$ van onder begrensd is constant.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:conformal:10}
Met conforme invariantie van harmoniciteit ($u\circ f$ is
harmonisch wanneer $u$ harmonisch is en $f$ \omterm{def:b3:holomorphic:holo}{holomorf} ---
bewijs het via \cref{prop:b3:conformal:harmonicholo}
lokaal), los het Dirichlet-probleem op het
bovenhalfvlak op met randdata $\mathbf
1_{\intoo{-\infty}0}$: toon dat
\[
u(x + \iu y) = \frac1\pi\,\arg(x + \iu y)
\qquad (\arg \in \intoo0\pi \text{ op } \mathbb H)
\]
harmonisch is op $\mathbb H$ (imaginair deel van een
\omterm{def:b3:conformal:simplyconnected}{holomorfe logaritme}) met de vereiste randlimieten in elke
$x \neq 0$, en transporteer haar naar de schijf via Cayley
om \cref{exo:b3:conformal:8}(c) te herleiden.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:conformal:11}
(Vaste punten en iteratie in de schijf) Zij
$f\colon\mathbb D \to \mathbb D$ \omterm{def:b3:holomorphic:holo}{holomorf}.
(a) Toon dat als $f$ \emph{twee} verschillende vaste punten
heeft, dan $f = \mathrm{id}$ \emph{(verplaats er één naar
$0$ met een automorfisme en pas het gelijkheidsgeval van
Schwarz toe)}.
(b) Stel $f(0) = 0$ en $f$ is geen rotatie. Toon dat de
iteraten $f^{\circ n} \to 0$ uniform op elk compactum
$\bar D(0, r)$, $r < 1$ \emph{(Schwarz geeft $\abs{f(z)}
\leq c_r\abs z$ op $\bar D(0,r)$ met $c_r < 1$ ---
rechtvaardig deze strikte constante via het
maximumprincipe toegepast op $f(z)/z$)}.
(c) Illustreer met $f(z) = \frac{z^2 + z}2$: vaste punten,
en de convergentiesnelheid van de \omterm{def:b3:groups:action}{baan} van $z_0 =
\frac12$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:conformal:12}
(Harmonische toegevoegden, concreet) Zij $u(x, y) = x^3 -
3xy^2 + 2y$.
(a) Controleer dat $u$ harmonisch is op $\R^2$, en vind
alle harmonische toegevoegden $v$ (d.w.z.\ $u + \iu v$
\omterm{def:b3:holomorphic:holo}{holomorf}) door de \omterm{prop:b3:holomorphic:cauchyriemann}{Cauchy--Riemannvergelijkingen} te
integreren; identificeer $f(z) = u + \iu v$ als polynoom in
$z$.
(b) Toon dat op een \emph{stervormige} \omterm{def:b3:topology:topology}{open verzameling}
elke \omterm{prop:b3:conformal:harmonicholo}{harmonische functie} een harmonische toegevoegde
toelaat, uniek op een additieve constante na \emph{(de
$1$-vorm $-u_y\,\dd x + u_x\,\dd y$ is gesloten; de
primitieve-machine van \cref{thm:b3:holomorphic:cauchy}, of
het lemma van Poincaré van \cref{ch:b3:forms})}.
(c) Geef het standaardtegenvoorbeeld op $\C^*$: $u =
\ln\abs z$ heeft geen globale toegevoegde --- relateer aan
de hoekvorm en het \omterm{def:b3:holomorphic:index}{windinggetal} (\cref{ch:b3:forms}).
\end{exercise}

\section{Probleem: de oppervlaktestelling en Koebes
kwartstelling}

\begin{problem}[{Weekendprobleem --- hoeveel moet een
univalente afbeelding bedekken?}]\label{pb:b3:conformal:1}
Een \emph{univalente} functie is een holomorfe injectie. De
genormaliseerde univalente functies op de schijf,
\[
\mathcal S = \bigl\{f \in \mathcal H(\mathbb D) \text{
injectief},\ f(z) = z + a_2z^2 + a_3z^3 + \cdots\bigr\},
\]
zijn star begrensd: we bewijzen Bieberbachs ongelijkheid
$\abs{a_2} \leq 2$ en leiden de \emph{Koebe-kwartstelling}
af: het beeld van elke $f \in \mathcal S$ bevat de schijf
$D(0, \frac14)$ --- de scherpe universele constante van
conforme meetkunde.\index{kwartstelling van Koebe}\index{oppervlaktestelling}

\medskip
\noindent\textbf{Deel I --- De oppervlaktestelling.} Zij
$g(w) = w + b_0 + \frac{b_1}w + \frac{b_2}{w^2} + \cdots$
\omterm{def:b3:holomorphic:holo}{holomorf} en \emph{injectief} op $\{\abs w > 1\}$.
\begin{enumerate}
  \item Voor $\rho > 1$, zij $A_\rho$ de oppervlakte
        (\omterm{def:b3:measure:lebesgueouter}{Lebesgue-maat}) van het compacte $K_\rho =
        \C\setminus g(\{\abs w > \rho\})$, de regio omsloten
        door de gladde Jordankromme $g(C_\rho)$. Met de
        oppervlakteformule van de Green--Riemannstelling uit
        het Jaar~2-boekdeel --- de omsloten oppervlakte is
        $\frac1{2\iu}\oint\bar\zeta\,\dd\zeta$ langs de
        positief georiënteerde rand --- toon dat
        \[
        A_\rho = \frac{1}{2\iu}\int_{C_\rho}
        \overline{g(w)}\,g'(w)\,\dd w
        = \pi\Bigl(\rho^2 -
        \sum_{n\geq1}n\,\abs{b_n}^2\rho^{-2n}\Bigr) :
        \]
        substitueer de \omterm{thm:b3:residues:laurent}{Laurentreeks} van $\bar g$ en $g'$ op
        $C_\rho$ en integreer term voor term (normale
        convergentie; slechts de frequentie-nulproducten
        overleven).
  \item Laat $\rho \to 1^+$ en concludeer de
        \emph{oppervlaktestelling}:
        \[
        \sum_{n\geq1}n\,\abs{b_n}^2 \;\leq\; 1 .
        \]
        In het bijzonder $\abs{b_1} \leq 1$. Wanneer is
        $\abs{b_1} = 1$?
\end{enumerate}

\medskip
\noindent\textbf{Deel II --- Bieberbachs $\abs{a_2} \leq
2$.} Zij $f = z + a_2z^2 + \cdots \in \mathcal S$.
\begin{enumerate}[resume]
  \item Toon dat $f(z^2)/z^2$ \omterm{def:b3:holomorphic:holo}{holomorf} en nultvrij is op
        $\mathbb D$, en een holomorfe vierkantswortel
        $\varphi$ toelaat met $\varphi(0) = 1$; zet $h(z) =
        z\varphi(z^2)$, zodat $h(z)^2 = f(z^2)$. Toon dat
        $h$ een \emph{oneven} univalente functie is op
        $\mathbb D$ met ontwikkeling $h(z) = z +
        \frac{a_2}2z^3 + \cdots$. \emph{(Injectiviteit:
        $h(z)^2 = h(z')^2$ forceert $z^2 = z'^2$; gebruik
        onevenheid om te voltooien.)}
  \item Pas de oppervlaktestelling toe op $g(w) =
        1/h(1/w) = w - \frac{a_2}{2w} + \cdots$ op $\{\abs w
        > 1\}$ (verifieer univalentie en de ontwikkeling),
        en concludeer $\abs{a_2} \leq 2$.
  \item Toon dat de \emph{Koebefunctie}
        \[
        k(z) = \frac{z}{(1 - z)^2} = \sum_{n\geq1}n\,z^n
        \]
        tot $\mathcal S$ behoort, $a_2 = 2$ heeft, en
        $\mathbb D$ afbeeldt op
        $\C\setminus\intoc{-\infty}{-\frac14}$
        \emph{(schrijf $k = \frac14\bigl[\bigl(\frac{1+z}{1
        - z}\bigr)^2 - 1\bigr]$ en volg de beelden)}: alle
        komende ongelijkheden zijn scherp.
\end{enumerate}

\medskip
\noindent\textbf{Deel III --- De kwartstelling.}
\begin{enumerate}[resume]
  \item Zij $f \in \mathcal S$ en $c \notin f(\mathbb D)$.
        Toon dat
        \[
        F(z) = \frac{c\,f(z)}{c - f(z)}
        \]
        tot $\mathcal S$ behoort, en bereken haar tweede
        coëfficiënt: $A_2 = a_2 + \frac1c$.
  \item Pas Bieberbach toe op zowel $f$ als $F$: concludeer
        $\abs{\frac1c} \leq \abs{A_2} + \abs{a_2} \leq 4$,
        d.w.z.\ $\abs c \geq \frac14$. \emph{Elke
        weggelaten waarde heeft modulus $\geq \frac14$}:
        $f(\mathbb D) \supseteq D(0, \frac14)$ --- Koebes
        kwartstelling. Controleer scherpte op de
        Koebefunctie.
  \item Leid een kwantitatieve Riemann-afbeeldingschatting
        af: als $\varphi \colon \Omega \to \mathbb D$ de
        Riemann-afbeelding van \cref{thm:b3:conformal:rmt}
        in $z_0$ is, dan
        \[
        \frac{d\bigl(z_0, \partial\Omega\bigr)}{4}
        \;\leq\; \frac1{\varphi'(z_0)} \;\leq\;
        4\,d\bigl(z_0, \partial\Omega\bigr)
        \]
        --- bewijs minstens de linker ongelijkheid door
        Koebe toe te passen op $\varphi^{-1}$ geschikt
        genormaliseerd, en de rechter door Schwarz
        toegepast op $\varphi$ op de schijf $D(z_0, d)
        \subseteq \Omega$.
\end{enumerate}

\medskip
\noindent\textbf{Deel IV --- Perspectief.}
\begin{enumerate}[resume]
  \item Bieberbach vermoedde (1916) $\abs{a_n} \leq n$ voor
        alle $n$, met gelijkheid slechts voor rotaties van
        de Koebefunctie; de Branges bewees het in 1985.
        Verifieer het vermoeden met de hand voor de
        Koebefunctie en haar rotaties
        $\eu^{-\iu\theta}k(\eu^{\iu\theta}z)$. Duw dan de
        ontwikkeling van vraag~4 één term verder:
        schrijvend $h(z) = z + \frac{a_2}2z^3 + c_5z^5 +
        \cdots$, toon $c_5 = \frac{a_3}2 - \frac{a_2^2}8$
        en
        \[
        g(w) = w - \frac{a_2}{2}\,w^{-1} +
        \Bigl(\frac{3a_2^2}8 - \frac{a_3}2\Bigr)w^{-3} +
        \cdots ,
        \]
        dus de oppervlaktestelling levert de verfijnde
        ongelijkheid $\bigl|\frac{a_2}2\bigr|^2 +
        3\bigl|\frac{3a_2^2}8 - \frac{a_3}2\bigr|^2 \leq
        1$. Controleer op de Koebefunctie ($a_2 = 2$, $a_3
        = 3$).
\end{enumerate}

\medskip
\noindent\textbf{Deel V --- De vervormingsstelling.}
Bieberbachs ongelijkheid, rond de schijf getransporteerd
door automorfismen, controleert $f'$ overal. Fixeer $f \in
\mathcal S$.
\begin{enumerate}[resume]
  \item (Koebetransform) Voor $z_0 \in \mathbb D$ zij
        $\varphi(z) = \frac{z + z_0}{1 + \bar z_0z}$, een
        schijfautomorfisme (\cref{thm:b3:conformal:autdisc})
        met $\varphi(0) = z_0$. Toon dat
        \[
        F(z) = \frac{f(\varphi(z)) - f(z_0)}
        {f'(z_0)\,\bigl(1 - \abs{z_0}^2\bigr)}
        \]
        tot $\mathcal S$ behoort \emph{(univalentie wordt
        geërfd; bereken $\varphi'(0) = 1 - \abs{z_0}^2$ en
        controleer de normalisatie; herinner $f' \neq 0$
        voor injectieve $f$,
        \cref{def:b3:conformal:conformal})}.
  \item Bereken de tweede coëfficiënt $A_2 =
        \frac12F''(0)$ van $F$:
        \[
        A_2 = \frac12\Bigl[\bigl(1 - \abs{z_0}^2\bigr)
        \frac{f''(z_0)}{f'(z_0)} - 2\bar z_0\Bigr] ,
        \]
        en leid uit Bieberbach (vraag~4), voor $z =
        r\eu^{\iu\theta}$, de \emph{fundamentele
        ongelijkheid} af:
        \[
        \Bigl|\,z\,\frac{f''(z)}{f'(z)} -
        \frac{2r^2}{1 - r^2}\Bigr|
        \;\leq\; \frac{4r}{1 - r^2} .
        \]
  \item Extraheer het reële deel:
        \[
        \frac{2r^2 - 4r}{1 - r^2} \;\leq\;
        \operatorname{Re}\Bigl(z\,\frac{f''(z)}{f'(z)}\Bigr)
        \;\leq\; \frac{2r^2 + 4r}{1 - r^2} .
        \]
  \item Toon dat $\frac{\dd}{\dd t}\log\bigl|
        f'(t\eu^{\iu\theta})\bigr| = \frac1t
        \operatorname{Re}\bigl(z\frac{f''(z)}{f'(z)}\bigr)$
        in $z = t\eu^{\iu\theta}$ \emph{(voor een
        niet-verdwijnende $\mathcal C^1$-functie $g$ van
        een reële variabele, $\frac{\dd}{\dd t}\log\abs g =
        \operatorname{Re}(g'/g)$)}, en integreer de grenzen
        van vraag~12 langs de straal om de
        \emph{vervormingsstelling} te verkrijgen:
        \[
        \frac{1 - r}{(1 + r)^3} \;\leq\; \abs{f'(z)}
        \;\leq\; \frac{1 + r}{(1 - r)^3},
        \qquad \abs z = r .
        \]
  \item Leid de \emph{groeistelling} af:
        \[
        \frac{r}{(1 + r)^2} \;\leq\; \abs{f(z)} \;\leq\;
        \frac{r}{(1 - r)^2},
        \qquad \abs z = r
        \]
        \emph{(bovengrens: integreer $f'$ op het segment
        $[0, z]$; ondergrens: als $\abs{f(z)} < \frac14$,
        ligt het segment $[0, f(z)]$ in $f(\mathbb D)$ door
        vraag~7; trek het terug met $f^{-1}$ --- \omterm{def:b3:holomorphic:holo}{holomorf}
        door \cref{cor:b3:residues:openmapping} --- en
        begrens $\abs{f(z)} = \int_\gamma\abs{f'(\zeta)}
        \,\abs{\dd\zeta} \geq \int_0^r\frac{1 - t}{(1 +
        t)^3}\,\dd t$, met $\abs{\dd\zeta} \geq
        \dd\abs\zeta$)}.
  \item Verifieer dat de Koebefunctie gelijkheid bereikt in
        alle vier grenzen, in $z = r$ voor de bovenste en
        $z = -r$ voor de onderste: $k$ is tegelijk de meest
        expanderende en, in de antipode, de meest
        contracterende van $\mathcal S$.
\end{enumerate}

\medskip
\noindent\textbf{Deel VI --- Extremale starheid.} In elke
ongelijkheid tot nu toe identificeert gelijkheid de
Koebefunctie op rotatie na. We bewijzen het, dan oogsten we.
\begin{enumerate}[resume]
  \item Stel $f \in \mathcal S$ heeft $\abs{a_2} = 2$.
        Volg gelijkheid door vragen~2--4: de
        oppervlaktestelling forceert $g(w) = w + b_0 +
        \eu^{\iu\alpha}/w$; onevenheid van $h$ maakt $g$
        oneven, dus $b_0 = 0$; inverseer om $h$ te vinden,
        dan $f$, en concludeer dat
        \[
        f(z) = \eu^{-\iu\theta}k\bigl(\eu^{\iu\theta}z\bigr)
        \quad\text{met } \eu^{\iu\theta} =
        -\eu^{\iu\alpha} :
        \]
        de rotaties van de Koebefunctie zijn de enige
        leden van $\mathcal S$ met $\abs{a_2} = 2$.
  \item Toon dat als $f \in \mathcal S$ een waarde $c$
        weglaat met $\abs c = \frac14$ exact, dan is $f$
        een rotatie van de Koebefunctie, en identificeer $c
        = -\eu^{-\iu\theta}/4$ \emph{(volg gelijkheid door
        de keten van vraag~7 $4 = \abs{1/c} = \abs{A_2 -
        a_2} \leq \abs{A_2} + \abs{a_2} \leq 4$)}: de
        constante van de kwartstelling wordt slechts door
        de extremale familie bereikt.
  \item (Coëfficiënten op de goedkope manier) Combineer de
        groeistelling met de \omterm{thm:b3:holomorphic:analytic}{Cauchy-schattingen}
        (\cref{thm:b3:holomorphic:analytic}) op de cirkel
        $\abs z = 1 - \frac1n$ om te bewijzen
        \[
        \abs{a_n} \;\leq\; \eu\,n^2 \qquad (n \geq 2) .
        \]
        (De Branges, 1985: $\abs{a_n} \leq n$; de factor
        $\eu n$ is de prijs van elementaire gereedschappen.)
  \item (Bedekking van subschijven) Toon dat voor elke $0 <
        r < 1$,
        \[
        f\bigl(D(0, r)\bigr) \supseteq
        D\Bigl(0, \frac{r}{(1 + r)^2}\Bigr),
        \]
        scherp voor de Koebefunctie, en herwin de
        kwartstelling als $r \to 1^-$. \emph{(Randpunten
        van het open beeld $f(D(0,r))$ liggen op
        $f(\partial D(0,r))$, hebben dus modulus $\geq
        \frac{r}{(1+r)^2}$ door vraag~14; een segment van
        $0$ naar een gemist punt van kleinere modulus zou
        die rand moeten kruisen.)}
  \item (Koebe in elk punt) Zij $f$ univalent op $\mathbb
        D$, niet noodzakelijk genormaliseerd, en $z_0 \in
        \mathbb D$. Bewijs
        \[
        \tfrac14\bigl(1 - \abs{z_0}^2\bigr)\abs{f'(z_0)}
        \;\leq\;
        d\bigl(f(z_0), \partial f(\mathbb D)\bigr)
        \;\leq\;
        \bigl(1 - \abs{z_0}^2\bigr)\abs{f'(z_0)}
        \]
        \emph{(links: kwartstelling toegepast op de
        Koebetransform van vraag~10; rechts: Schwarz
        (\cref{thm:b3:conformal:schwarz}) toegepast op
        $\psi^{-1}\circ\hat g$, waar $\hat g(w) =
        f^{-1}\bigl(f(z_0) + dw\bigr)$, $d$ de afstand, en
        $\psi$ een schijfautomorfisme dat $0$ naar $z_0$
        stuurt)}. Waarom is $\partial f(\mathbb D)$
        niet-leeg?
  \item Controleer vraag~20 op $f = k$ in $z_0 = r \in
        \intoo01$: bereken $d\bigl(k(r), \partial k(\mathbb
        D)\bigr) = \frac{(1+r)^2}{4(1-r)^2}$ en verifieer
        dat de linker ongelijkheid een \emph{gelijkheid}
        is: de Koebefunctie verzadigt haar eigen stelling
        in elk punt van $\intoo01$.
  \item (De moraal) In tien regels: welk enkel principe
        ligt ten grondslag aan de oppervlaktestelling, en
        hoe vloeien Bieberbach, de kwartstelling,
        vervorming, groei en bedekking eruit voort?
        Vergelijk met de Schwarz--Pick-wereld van
        \cref{thm:b3:conformal:schwarz}: in beide
        starreert één inwendige ongelijkheid de hele
        meetkunde, en de extremalen zijn uniek op rotatie
        na.

\end{enumerate}
\medskip
\noindent\textbf{Deel VII --- Compactheid, inversen, en een
realiteitscontrole.}
\begin{enumerate}[resume]
  \item Toon dat de klasse $\mathcal S$ \emph{compact} is
        voor lokaal uniforme convergentie: zij is lokaal
        begrensd door de groeistelling, dus normaal
        (\cref{thm:b3:conformal:montel}); en een lokaal
        uniforme limiet van leden van $\mathcal S$ is
        opnieuw in $\mathcal S$ \emph{(de normalisaties
        gaan over tot de limiet door Weierstrass-convergentie
        van afgeleiden; injectiviteit overleeft door
        Hurwitz, \cref{exo:b3:residues:8}, de limiet
        niet-constant zijnde)}. Waarom doet dit ertoe voor
        extremale problemen zoals die van Bieberbach?
  \item Voor $f \in \mathcal S$, zij $g = f^{-1}$,
        gedefinieerd bij $0$. Toon $g(w) = w - a_2w^2 +
        O(w^3)$, dus de tweede coëfficiënt van de inverse
        gehoorzaamt dezelfde scherpe grens $\abs{A_2} =
        \abs{a_2} \leq 2$, met gelijkheid exact voor de
        geroteerde Koebefuncties.
  \item Bepaal voor welke $a \in \C$ het polynoom $f(z) =
        z + az^2$ tot $\mathcal S$ behoort: toon dat $f$
        injectief is op $\mathbb D$ desda $\abs a \leq
        \frac12$ \emph{(factoriseer $f(z_1) - f(z_2)$)}.
        Concludeer: voor graad-twee-polynomen is de ware
        coëfficiëntgrens $\frac12$, vier maal kleiner dan
        Bieberbachs $2$ --- de extremalen van $\mathcal S$
        zijn echt transcendentale objecten, en geen
        polynoom komt in de buurt.
\end{enumerate}
\end{problem}
