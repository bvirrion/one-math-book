\chapter{Lichaamsuitbreidingen en Galoistheorie}\label{ch:b3:galois}

Kan elke vergelijking door radicalen opgelost worden, zoals de
kwadratische formule en de kubische formules van Cardano
suggereren? Kan men een hoek met liniaal en passer in drieën
delen? Beide vragen, eeuwen open, worden --- negatief ---
beantwoord door één idee van Évariste Galois: hecht aan elke
veelterm een eindige \emph{groep} van symmetrieën van haar
wortels, en lees het antwoord af van de groep. Dit hoofdstuk
bouwt het woordenboek: \omterm{def:b3:galois:extension}{lichaamsuitbreidingen} en graden,
splitsingslichamen en \omterm{def:b3:galois:closure}{algebraïsche afsluitingen}, eindige lichamen
(een complete theorie --- en de beloofde cycliciteit van $\mathbb
F_q^\times$), separabiliteit, dan de \omterm{thm:b3:galois:fundamental}{Galoïsche correspondentie}
zelf, met volledige bewijzen. We oogsten: de onmogelijkheid van
de klassieke constructies, de structuur van cyclotomische
lichamen, en de onoplosbaarheid van de vijfdegraadsvergelijking
door radicalen --- de eenvoud van $A_5$ uit
\cref{ch:b3:groups} die haar doel treft.

\section{Uitbreidingen, graad, algebraïciteit}

\begin{definition}\label{def:b3:galois:extension}
Een \emph{lichaamsuitbreiding}\index{lichaamsuitbreiding} $L/K$
is een lichaam $L$ dat $K$ als deellichaam bevat; $L$ is dan een
$K$-vectorruimte, en de \emph{graad}\index{graad van een
uitbreiding} $[L:K]$ is haar dimensie. De uitbreiding is
\emph{eindig} als $[L:K] < \infty$. De \emph{karakteristiek}
van een lichaam is de voortbrenger $\geq 0$ van de kern van $\Z
\to K$, $n \mapsto n\cdot 1$: ze is $0$ of een priemgetal $p$;
dienovereenkomstig bevat $K$ een kleinste deellichaam
(\emph{priemlichaam}) isomorf met $\Q$ of met $\mathbb F_p =
\Z/p\Z$.
\end{definition}

\begin{theorem}[Torenwet]\label{thm:b3:galois:tower}
Als $K \subseteq L \subseteq M$, dan $[M:K] = [M:L]\,[L:K]$:
als $(e_i)$ een basis van $L$ over $K$ is en $(f_j)$ een basis
van $M$ over $L$, dan is $(e_if_j)$ een basis van $M$ over
$K$.\index{torenwet}
\end{theorem}

\begin{proof}
Voortbrengend: $x \in M$ schrijft $x = \sum_j \lambda_jf_j$
($\lambda_j \in L$), elke $\lambda_j = \sum_i \mu_{ij}e_i$
($\mu_{ij} \in K$): $x = \sum_{i,j}\mu_{ij}e_if_j$.
Onafhankelijk: $\sum_{i,j}\mu_{ij}e_if_j = 0$ herschrijft
$\sum_j (\sum_i \mu_{ij}e_i)f_j = 0$; de binnensommen liggen in
$L$, dus verdwijnen ($f_j$ onafhankelijk over $L$); dan alle
$\mu_{ij} = 0$ ($e_i$ onafhankelijk over $K$).
\end{proof}

\begin{definition}\label{def:b3:galois:algebraic}
Zij $L/K$ en $\alpha \in L$. Als enige niet-nulle $P \in K[X]$
$P(\alpha) = 0$ heeft, is $\alpha$ \emph{algebraïsch}
\index{algebraïsch element} over $K$; de monische voortbrenger
$\pi_\alpha$ van het \omterm{def:b3:rings:ideal}{ideaal} $\{P : P(\alpha) = 0\}$ van $K[X]$
is haar \emph{minimale veelterm}\index{minimale veelterm
(lichaamstheorie)}, een irreducibele veelterm ($\pi = QR$ met
$Q(\alpha) = 0$ dwingt $R$ constant door minimaliteit van de
\omterm{def:b3:galois:extension}{graad}). Anders is $\alpha$ \emph{transcendent}. We schrijven
$K(\alpha)$ voor het kleinste deellichaam van $L$ dat $K$ en
$\alpha$ bevat, en $K[\alpha]$ voor de kleinste deelring.
\end{definition}

\begin{theorem}\label{thm:b3:galois:simple}
Als $\alpha$ \omterm{def:b3:galois:algebraic}{algebraïsch} is over $K$ met $d = \deg\pi_\alpha$,
dan
\[
K(\alpha) = K[\alpha] \cong K[X]/(\pi_\alpha),
\qquad [K(\alpha) : K] = d,
\]
met basis $1, \alpha, \dots, \alpha^{d-1}$. Omgekeerd, als
$[L:K] < \infty$, is elke $\alpha \in L$ \omterm{def:b3:galois:algebraic}{algebraïsch} van \omterm{def:b3:galois:extension}{graad}
$\deg \pi_\alpha$ die $[L:K]$ deelt.
\end{theorem}

\begin{proof}
Evaluatie $K[X] \to L$, $P \mapsto P(\alpha)$, heeft beeld
$K[\alpha]$ en kern $(\pi_\alpha)$; omdat $\pi_\alpha$
irreducibel is, is $K[X]/(\pi_\alpha)$ een \emph{lichaam}
(\cref{prop:b3:rings:primemaximal}: in de \omterm{def:b3:rings:pidufd}{PID} $K[X]$ brengt
irreducibel een \omterm{def:b3:rings:primemaximal}{maximaal ideaal} voort), dus is $K[\alpha]$ een
lichaam dat $K$ en $\alpha$ bevat: ze is gelijk aan $K(\alpha)$.
De klassen van $1, X, \dots, X^{d-1}$ vormen een basis van het
quotiënt (Euclidische deling), vandaar de basis en de \omterm{def:b3:galois:extension}{graad}.
Omgekeerd als $[L:K] = n < \infty$: $1, \alpha, \dots,
\alpha^n$ zijn afhankelijk, wat een annulerende veelterm geeft;
dan deelt $[K(\alpha):K] = \deg\pi_\alpha$ $n$ door de \omterm{thm:b3:galois:tower}{torenwet}.
\end{proof}

\begin{corollary}\label{cor:b3:galois:algclosed}
Als $\alpha, \beta$ \omterm{def:b3:galois:algebraic}{algebraïsch} over $K$ zijn, zijn $\alpha \pm
\beta$, $\alpha\beta$, $\alpha/\beta$ ($\beta \ne 0$) dat ook:
de elementen van $L$ \omterm{def:b3:galois:algebraic}{algebraïsch} over $K$ vormen een deellichaam
van $L$. Bovendien is algebraïciteit transitief: \omterm{def:b3:galois:algebraic}{algebraïsch} over
\omterm{def:b3:galois:algebraic}{algebraïsch} is \omterm{def:b3:galois:algebraic}{algebraïsch}.
\end{corollary}

\begin{proof}
$K(\alpha, \beta) = (K(\alpha))(\beta)$ is eindig over
$K(\alpha)$ ($\beta$ \omterm{def:b3:galois:algebraic}{algebraïsch} over $K \subseteq K(\alpha)$)
en $K(\alpha)/K$ is eindig: door de \omterm{thm:b3:galois:tower}{torenwet}
$[K(\alpha,\beta):K] < \infty$, en elk element van $K(\alpha,
\beta)$ --- inclusief de vier genoemde --- is \omterm{def:b3:galois:algebraic}{algebraïsch}
(\cref{thm:b3:galois:simple}). Transitiviteit: als $\beta$
\omterm{def:b3:galois:algebraic}{algebraïsch} over $L$ is en $L/K$ \omterm{def:b3:galois:algebraic}{algebraïsch}, brengen de
coëfficiënten $c_0, \dots, c_{m-1}$ van $\pi_{\beta/L}$ een
eindige uitbreiding $F = K(c_0, \dots, c_{m-1})$ van $K$ voort
(herhaalde \omterm{thm:b3:galois:tower}{torenwet}), en $F(\beta)/F$ is eindig: $[F(\beta):K] <
\infty$, dus is $\beta$ \omterm{def:b3:galois:algebraic}{algebraïsch} over $K$.
\end{proof}

\begin{example}\label{ex:b3:galois:degrees}
$[\Q(\sqrt2):\Q] = 2$, $[\Q(\sqrt[3]2):\Q] = 3$ ($X^3 - 2$ is
irreducibel: \omterm{thm:b3:rings:criteria}{Eisenstein}), $[\Q(\zeta_p):\Q] = p - 1$ voor
$\zeta_p = \eu^{2\iu\pi/p}$ ($\Phi_p$ is irreducibel,
\cref{ex:b3:rings:eisenstein}). De \omterm{thm:b3:galois:tower}{torenwet} is al een wapen:
$\sqrt[3]2 \notin \Q(\sqrt2)$, omdat $3 \nmid 2$.
\end{example}

\section{Splitsingslichamen; algebraïsche afsluiting}

\begin{theorem}[Splitsingslichamen]\label{thm:b3:galois:splitting}
Zij $P \in K[X]$ niet-constant. Er bestaat een
\emph{splitsingslichaam}\index{splitsingslichaam} van $P$ over
$K$: een uitbreiding $L = K(\alpha_1, \dots, \alpha_n)$
voortgebracht door wortels van $P$ waarin $P$ in lineaire
factoren splitst. Ze is uniek tot op $K$-isomorfie, en $[L:K]
\leq (\deg P)!$.
\end{theorem}

\begin{proof}
\emph{Bestaan}, door inductie op $\deg P$: kies een irreducibele
factor $Q$ van $P$; het lichaam $K_1 = K[X]/(Q)$ bevat de wortel
$\alpha_1 = \bar X$ van $Q$, dus van $P$; schrijf $P = (X -
\alpha_1)P_1$ over $K_1$ en pas inductie toe op $P_1$ over $K_1$;
graden vermenigvuldigen tot hoogstens $n(n-1)\cdots = n!$.

\emph{Uniciteit} volgt uit het sterkere
\emph{isomorfisme-uitbreidingslemma}: zij $\sigma \colon K \to
K'$ een isomorfisme, $P \in K[X]$, $P^\sigma$ de veelterm met
afgebeelde coëfficiënten, $L, L'$ splitsingslichamen van $P,
P^\sigma$; dan breidt $\sigma$ uit tot een isomorfisme $L \to
L'$. Inductie op $[L:K]$: als $P$ in $K$ splitst, dan $L = K$,
en $L' = K'$ ($P^\sigma$ splitst in $K'$, en $L'$ wordt
voortgebracht door haar wortels). Anders kies een wortel $\alpha
\in L \setminus K$ van een irreducibele factor $Q$ van $P$ met
$\deg Q \geq 2$; $Q^\sigma$ is een irreducibele factor van
$P^\sigma$, met een wortel $\beta \in L'$; dan
\[
K(\alpha) \cong K[X]/(Q) \xrightarrow{\ \sigma\ }
K'[X]/(Q^\sigma) \cong K'(\beta)
\]
breidt $\sigma$ uit met $\alpha \mapsto \beta$. Nu is $L$ een
\omterm{thm:b3:galois:splitting}{splitsingslichaam} van $P$ over $K(\alpha)$, en $L'$ van
$P^\sigma$ over $K'(\beta)$, met $[L : K(\alpha)] < [L:K]$:
inductie breidt verder uit tot $L \to L'$.
\end{proof}

\begin{definition}\label{def:b3:galois:closure}
Een lichaam $\Omega$ is \emph{algebraïsch gesloten}
\index{algebraïsch gesloten lichaam} als elke niet-constante
veelterm van $\Omega[X]$ een wortel in $\Omega$ heeft (dus
splitst). Een \emph{algebraïsche afsluiting}\index{algebraïsche
afsluiting} van $K$ is een algebraïsche uitbreiding $\bar K/K$
met $\bar K$ \omterm{def:b3:galois:algebraic}{algebraïsch} gesloten.
\end{definition}

\begin{theorem}[Steinitz]\label{thm:b3:galois:steinitz}
Elk lichaam $K$ heeft een \omterm{def:b3:galois:closure}{algebraïsche afsluiting}, uniek tot op
$K$-isomorfie.
\end{theorem}

\begin{proof}
\emph{Bestaan (constructie van Artin).} Zij $R = K[(X_f)_f]$ de
veeltermring met één variabele $X_f$ per niet-constante monische
$f \in K[X]$, en $I$ het \omterm{def:b3:rings:ideal}{ideaal} voortgebracht door alle
$f(X_f)$. $I$ is eigenlijk: een relatie $1 = \sum_{i=1}^r g_i\,
f_i(X_{f_i})$ betrekt eindig veel veeltermen; in een
gemeenschappelijk \omterm{thm:b3:galois:splitting}{splitsingslichaam} $E$ van $f_1\cdots f_r$ kies
wortels $\alpha_i$ van $f_i$ en evalueer $X_{f_i} \mapsto
\alpha_i$ (andere variabelen $\mapsto 0$): $1 = 0$, absurd. Zij
$\mathfrak m \supseteq I$ \omterm{def:b3:rings:primemaximal}{maximaal} (\cref{thm:b3:rings:krull};
Zorn) en $K_1 = R/\mathfrak m$: een \omterm{def:b3:galois:extension}{lichaamsuitbreiding} van $K$
waarin elke niet-constante $f \in K[X]$ een wortel heeft, namelijk
$\bar X_f$, en die \omterm{def:b3:galois:algebraic}{algebraïsch} over $K$ is (ze wordt
voortgebracht door de $\bar X_f$, elk \omterm{def:b3:galois:algebraic}{algebraïsch}). Itereer: $K
\subseteq K_1 \subseteq K_2 \subseteq \cdots$, waar $K_{n+1}$
met $K_n$ doet wat $K_1$ met $K$ deed, en zij $\Omega = \bigcup_n
K_n$, een lichaam. Elke niet-constante $g \in \Omega[X]$ heeft
haar eindig veel coëfficiënten in enige $K_n$; een irreducibele
factor van $g$ over $K_n$ heeft een wortel in $K_{n+1} \subseteq
\Omega$: $\Omega$ is \omterm{def:b3:galois:closure}{algebraïsch gesloten}, en \omterm{def:b3:galois:algebraic}{algebraïsch} over
$K$ (elke $K_n$ is dat, door transitiviteit,
\cref{cor:b3:galois:algclosed}): $\Omega$ is een \omterm{def:b3:galois:closure}{algebraïsche
afsluiting}.

\emph{Uniciteit.} Zij $\Omega, \Omega'$ twee \omterm{def:b3:galois:closure}{algebraïsche
afsluitingen}. Beschouw de verzameling van paren $(E, \tau)$ waar
$K \subseteq E \subseteq \Omega$ en $\tau\colon E \to \Omega'$
een $K$-inbedding is, geordend door uitbreiding; ze is niet-leeg
($(K, \mathrm{id})$) en inductief (unie van een keten), dus
levert Zorn een maximale $(E_0, \tau_0)$. Als $E_0 \neq \Omega$,
kies $\alpha \in \Omega\setminus E_0$: $\pi_{\alpha/E_0}$ beeldt
af op een veelterm over $\tau_0(E_0)$ die een wortel $\beta$ in
het \omterm{def:b3:galois:closure}{algebraïsch gesloten} $\Omega'$ heeft, en $\tau_0$ breidt uit
tot $E_0(\alpha) \to \Omega'$ ($\alpha \mapsto \beta$), in strijd
met maximaliteit. Dus is er een $K$-inbedding $\tau \colon
\Omega \to \Omega'$; haar beeld, isomorf met $\Omega$, is
\omterm{def:b3:galois:closure}{algebraïsch gesloten}, en $\Omega'$ is \omterm{def:b3:galois:algebraic}{algebraïsch} erover: voor $x
\in \Omega'$ splitst $\pi_{x/\tau(\Omega)}$ over $\tau(\Omega)$,
dus $x \in \tau(\Omega)$. Dus is $\tau$ surjectief: een
isomorfisme.
\end{proof}

\begin{remark}\label{rem:b3:galois:closureexamples}
Voor $K = \Q$ kan men de transfiniëte machine vermijden: de
algebraïsche getallen $\bar\Q = \{z \in \C : z \text{ algebraïsch
over } \Q\}$ vormen een \omterm{def:b3:galois:closure}{algebraïsche afsluiting} --- een
deellichaam van $\C$ door \cref{cor:b3:galois:algclosed},
\omterm{def:b3:galois:closure}{algebraïsch gesloten} omdat $\C$ dat is (d'Alembert--Gauss,
bewijs door complexe analyse in \cref{ch:b3:holomorphic}) en
wortels van veeltermen over $\bar\Q$ \omterm{def:b3:galois:algebraic}{algebraïsch} over $\Q$ zijn
door transitiviteit.
\end{remark}

\section{Eindige lichamen}

\begin{theorem}\label{thm:b3:galois:finitefields}
Zij $p$ priem, $n \geq 1$, $q = p^n$.
\begin{enumerate}
  \item Een \omterm{thm:b3:galois:finitefields}{eindig lichaam} heeft kardinaliteit een priemmacht, en
        voor elke $q$ is er precies één lichaam $\mathbb F_q$ met
        $q$ elementen tot op isomorfie: het \omterm{thm:b3:galois:splitting}{splitsingslichaam} van
        $X^q - X$ over $\mathbb F_p$.\index{eindig lichaam}
  \item De \emph{Frobenius}\index{Frobeniusendomorfisme} $F
        \colon x \mapsto x^p$ is een automorfisme van $\mathbb
        F_q$, en de automorfismengroep van $\mathbb F_q$ is
        cyclisch van orde $n$, voortgebracht door $F$.
  \item $\mathbb F_{p^m}$ injecteert in $\mathbb F_{p^n}$ desda
        $m \mid n$.
\end{enumerate}
\end{theorem}

\begin{proof}
(1) Een \omterm{thm:b3:galois:finitefields}{eindig lichaam} $E$ heeft karakteristiek $p > 0$ en is
een eindigdimensionale $\mathbb F_p$-vectorruimte: $\abs E =
p^n$. Haar multiplicatieve groep heeft orde $q - 1$, dus vervult
elk $x \in E$ $x^q = x$: $E$ bestaat uit $q$ wortels van $X^q -
X$, is dus een \omterm{thm:b3:galois:splitting}{splitsingslichaam} ervan over $\mathbb F_p$ ---
wat $E$ tot op isomorfie bepaalt
(\cref{thm:b3:galois:splitting}). Omgekeerd, in een
\omterm{thm:b3:galois:splitting}{splitsingslichaam} $L$ van $X^q - X$ is de verzameling $E$ van
haar wortels een \emph{deellichaam}: $(x + y)^q = x^q + y^q$
door de beginnelingsdroom $(a+b)^p = a^p + b^p$ te itereren ($p
\mid \binom pk$), en $(xy)^q = x^qy^q$, $(x^{-1})^q =
(x^q)^{-1}$; ze heeft precies $q$ elementen omdat $X^q - X$
\omterm{def:b3:galois:separable}{separabel} is: haar afgeleide is $qX^{q-1} - 1 = -1$ (omdat $p
\mid q$), copriem ermee, dus geen herhaalde wortels. Dus $L = E$
heeft $q$ elementen.

(2) $F$ is een lichaamsmorfisme (beginnelingsdroom), injectief
(lichamen), dus bijectief op de eindige $\mathbb F_q$. $F^n =
\mathrm{id}$ ($x^q = x$), en geen kleinere macht is de
identiteit: $F^m = \mathrm{id}$ betekent dat alle $q$ elementen
wortels van $X^{p^m} - X$ zijn, wat $p^m \geq q$ dwingt. Dus is
$\langle F\rangle$ cyclisch van orde $n$; en er zijn geen andere
automorfismen, door de grens $\abs{\operatorname{Aut}} \leq
[\,\mathbb F_q : \mathbb F_p\,] = n$ hieronder bewezen
(\cref{prop:b3:galois:embeddings} met $L = \mathbb F_q$, $K =
\mathbb F_p$: automorfismen fixeren het priemlichaam).

(3) Als $\mathbb F_{p^m} \subseteq \mathbb F_{p^n}$, geeft de
\omterm{thm:b3:galois:tower}{torenwet} $p^n = (p^m)^d$: $m \mid n$. Omgekeerd, als $m \mid n$,
dan $p^m - 1 \mid p^n - 1$ (geometrische som), dus deelt
$X^{p^m} - X$ $X^{p^n} - X$ (zelfde argument op exponenten:
$X^{a} - 1 \mid X^{b} - 1$ wanneer $a \mid b$), en de wortels
van de eerste binnen $\mathbb F_{p^n}$ vormen het vereiste
deellichaam, van kardinaliteit $p^m$ (separabiliteit als in
(1)).
\end{proof}

\begin{theorem}[Cycliciteit]\label{thm:b3:galois:cyclic}
Elke eindige deelgroep van de multiplicatieve groep van een
lichaam is cyclisch. In het bijzonder $\mathbb F_q^\times \cong
\Z/(q-1)\Z$.\index{multiplicatieve groep van een eindig lichaam}
\end{theorem}

\begin{proof}
Zij $G \leq K^\times$ eindig. Door de structuurstelling
(\cref{cor:b3:modules:abelian}), $G \cong \Z/d_1 \times \dots
\times \Z/d_s$ met $d_1 \mid \dots \mid d_s$. Elk $x \in G$
vervult dan $x^{d_s} = 1$; maar $X^{d_s} - 1$ heeft hoogstens
$d_s$ wortels in het lichaam $K$: $\abs G = d_1\cdots d_s \leq
d_s$, wat $s = 1$ dwingt: $G$ is cyclisch.
\end{proof}

\begin{example}\label{ex:b3:galois:f8}
$\mathbb F_8 = \mathbb F_2[X]/(X^3 + X + 1)$: de kubische heeft
geen wortel in $\mathbb F_2$, is dus irreducibel. Schrijvend
$\omega = \bar X$: $\mathbb F_8^\times$ is cyclisch van orde $7$,
dus brengt \emph{elk} element $\neq 0, 1$ voort. De deellichamen
van $\mathbb F_{p^{12}}$ vormen het delerrooster van $12$:
$\mathbb F_p, \mathbb F_{p^2}, \mathbb F_{p^3}, \mathbb F_{p^4},
\mathbb F_{p^6}, \mathbb F_{p^{12}}$ --- een eerste, complete
instantie van de \omterm{thm:b3:galois:fundamental}{Galoïsche correspondentie}.
\end{example}

\section{Separabiliteit en inbeddingen}

\begin{definition}\label{def:b3:galois:separable}
Een veelterm $P \in K[X]$ is \emph{separabel}\index{separabele
veelterm} als ze geen herhaalde wortel in een \omterm{thm:b3:galois:splitting}{splitsingslichaam}
heeft --- equivalent $\gcd(P, P') = 1$ (een herhaalde wortel is
een gemeenschappelijke wortel; omgekeerd, over het
\omterm{thm:b3:galois:splitting}{splitsingslichaam} is een gemeenschappelijke wortel herhaald; en
de ggd verandert niet onder \omterm{def:b3:galois:extension}{lichaamsuitbreiding}, argument van
\cref{cor:b3:modules:descent}). Een \omterm{def:b3:galois:algebraic}{algebraïsch element} is
separabel als haar \omterm{def:b3:galois:algebraic}{minimale veelterm} dat is; een uitbreiding
$L/K$ is separabel als al haar elementen dat zijn.
\end{definition}

\begin{proposition}\label{prop:b3:galois:perfect}
Een \emph{irreducibele} $P \in K[X]$ is \omterm{def:b3:galois:separable}{separabel} tenzij $P' =
0$, wat $\operatorname{char} K = p > 0$ en $P \in K[X^p]$ dwingt.
Bijgevolg is elke algebraïsche uitbreiding van een lichaam van
karakteristiek $0$, en van een \omterm{thm:b3:galois:finitefields}{eindig lichaam}, \omterm{def:b3:galois:separable}{separabel} (zulke
lichamen heten \emph{perfect}\index{perfect lichaam}).
\end{proposition}

\begin{proof}
$\gcd(P, P')$ deelt $P$; als ze niet $1$ is, dwingt
irreducibiliteit $\gcd = P$ (tot op een constante), dus $P \mid
P'$ met $\deg P' < \deg P$: $P' = 0$. Schrijvend $P = \sum
a_kX^k$: $ka_k = 0$ voor alle $k$, dus in karakteristiek $0$ is
$P$ constant (uitgesloten); in karakteristiek $p$, $a_k = 0$
tenzij $p \mid k$: $P = Q(X^p)$. Over een \omterm{thm:b3:galois:finitefields}{eindig lichaam} is elk
element een $p$-de macht (\omterm{thm:b3:galois:finitefields}{Frobenius} is surjectief), dus is
$Q(X^p) = \sum b_k^p X^{pk} = (\sum b_kX^k)^p$ niet
irreducibel: $P' = 0$ kan daar ook niet voor irreducibele $P$.
\end{proof}

\begin{proposition}[Inbeddingen tellen]\label{prop:b3:galois:embeddings}
Zij $L = K(\alpha_1, \dots, \alpha_r)$ eindig over $K$, en
$\sigma \colon K \to \Omega$ een inbedding in een \omterm{def:b3:galois:closure}{algebraïsch
gesloten lichaam}. Dan is het aantal uitbreidingen van $\sigma$
tot $L$ hoogstens $[L:K]$, met gelijkheid als $L/K$ \omterm{def:b3:galois:separable}{separabel}
is. In het bijzonder $\abs{\operatorname{Aut}_K(L)} \leq
[L:K]$.
\end{proposition}

\begin{proof}
Inductie op $[L:K]$ via eenvoudige stappen. Voor $L =
K(\alpha)$: een uitbreiding $\tau$ is bepaald door $\tau(\alpha)$,
die een wortel in $\Omega$ van $\pi_\alpha^\sigma$ moet zijn;
omgekeerd geeft elke zo'n wortel één uitbreiding ($K(\alpha)
\cong K[X]/(\pi_\alpha)$). Het aantal uitbreidingen is het
aantal \emph{onderscheiden} wortels van $\pi_\alpha^\sigma$ in
$\Omega$: hoogstens $\deg\pi_\alpha = [K(\alpha):K]$, met
gelijkheid desda $\pi_\alpha$ \omterm{def:b3:galois:separable}{separabel} is (separabiliteit van
$\pi^\sigma$ en $\pi$ stemmen overeen: ggd met de afgeleide wordt
door $\sigma$ bewaard). In het algemeen, factoriseer $L =
K(\alpha_1)(\alpha_2, \dots)$: uitbreidingen van $\sigma$ tot
$K(\alpha_1)$ tellen $\leq [K(\alpha_1):K]$, en elk breidt in
$\leq [L : K(\alpha_1)]$ manieren uit door inductie;
vermenigvuldig (\omterm{thm:b3:galois:tower}{torenwet}). In het separabele geval zijn beide
tellingen gelijkheden: \omterm{def:b3:galois:algebraic}{minimale veeltermen} over het grotere
lichaam $K(\alpha_1)$ delen die over $K$, blijven dus \omterm{def:b3:galois:separable}{separabel}.
\end{proof}

\begin{theorem}[Primitief element]\label{thm:b3:galois:primitive}
Elke eindige \emph{separabele} uitbreiding is \omterm{def:b3:groups:simple}{eenvoudig}: $L =
K(\gamma)$ voor enige $\gamma$.\index{stelling van het
primitieve element}
\end{theorem}

\begin{proof}
Als $K$ eindig is, is $L$ dat ook, en een voortbrenger $\gamma$
van de cyclische groep $L^\times$
(\cref{thm:b3:galois:cyclic}) doet het. Zij $K$ oneindig; door
inductie volstaat $L = K(\alpha, \beta)$ te behandelen. Zij $n =
[L:K]$; door \cref{prop:b3:galois:embeddings} zijn er $n$
onderscheiden $K$-inbeddingen $\sigma_1, \dots, \sigma_n \colon
L \to \Omega$ ($\Omega$ een \omterm{def:b3:galois:closure}{algebraïsche afsluiting}). De
veelterm
\[
D(T)=\prod_{i<j}\bigl[\bigl(\sigma_i(\alpha)-\sigma_j(\alpha)
\bigr) + T\bigl(\sigma_i(\beta) - \sigma_j(\beta)\bigr)\bigr]
\]
is niet identiek nul: een factor verdwijnt identiek alleen als
$\sigma_i, \sigma_j$ op zowel $\alpha$ als $\beta$ overeenstemmen,
dus op $L$ --- uitgesloten voor $i \neq j$. Omdat $K$ oneindig is,
kies $c \in K$ met $D(c) \ne 0$: dan zijn de $n$ elementen
$\sigma_i(\alpha + c\beta)$ paarsgewijs onderscheiden, dus heeft
$\gamma = \alpha + c\beta$ minstens $n$ onderscheiden conjugaten
in $\Omega$, d.w.z.\ $\deg \pi_\gamma \geq n$: $[K(\gamma):K]
\geq n = [L:K]$ dwingt $L = K(\gamma)$.
\end{proof}

\section{De Galoïsche correspondentie}

\begin{definition}\label{def:b3:galois:galois}
Een eindige uitbreiding $L/K$ is \emph{Galoïsch}\index{Galoïsche
uitbreiding} als ze het \omterm{thm:b3:galois:splitting}{splitsingslichaam} is van een
\emph{separabele} veelterm over $K$. Haar \emph{Galoisgroep}
\index{Galoisgroep} is $\operatorname{Gal}(L/K) =
\operatorname{Aut}_K(L)$, de groep van
lichaamsautomorfismen van $L$ die $K$ puntsgewijs fixeren.
\end{definition}

\begin{proposition}\label{prop:b3:galois:galoisorder}
Als $L/K$ \omterm{def:b3:galois:galois}{Galoïsch} is, dan $\abs{\operatorname{Gal}(L/K)} =
[L:K]$; bovendien is $L/F$ \omterm{def:b3:galois:galois}{Galoïsch} voor elk tussenlichaam $K
\subseteq F \subseteq L$, en heeft elke $F$-inbedding $L \to
\Omega \supseteq L$ beeld $L$ (\emph{normaliteit}).
\end{proposition}

\begin{proof}
Zij $L$ de splitsing van de separabele $P$ over $K$, en fixeer
een \omterm{def:b3:galois:closure}{algebraïsche afsluiting} $\Omega \supseteq L$. $L/K$ is
\omterm{def:b3:galois:separable}{separabel}: ze wordt voortgebracht door wortels van $P$;
separabiliteit van elk element volgt uit het
gelijkheidsgeval hieronder, maar argumenteer direct ---
\cref{prop:b3:galois:embeddings} toegepast op de voortbrengers
(wortels van de separabele $P$, waarvan de \omterm{def:b3:galois:algebraic}{minimale veeltermen}
$P$ delen) levert precies $[L:K]$ uitbreidingen van $K
\hookrightarrow \Omega$ (in de inductiestap deelt de \omterm{def:b3:galois:algebraic}{minimale
veelterm} van een wortel van $P$ over een tussenlichaam nog
steeds $P$, is dus \omterm{def:b3:galois:separable}{separabel}). Elke zo'n inbedding $\tau
\colon L \to \Omega$ permuteert de wortels van $P$ ($\tau$
fixeert de coëfficiënten), en $L$ wordt erdoor voortgebracht:
$\tau(L) = L$. Dus inbeddingen $=$ automorfismen:
$\abs{\operatorname{Gal}(L/K)} = [L:K]$. Voor tussenliggende
$F$: $L$ is ook het \omterm{thm:b3:galois:splitting}{splitsingslichaam} van $P$ over $F$, en $P$
blijft \omterm{def:b3:galois:separable}{separabel}: $L/F$ is \omterm{def:b3:galois:galois}{Galoïsch}; hetzelfde argument geeft
normaliteit over $F$.
\end{proof}

\begin{lemma}[Artin]\label{lem:b3:galois:artin}
Zij $G$ een eindige groep van automorfismen van een lichaam $L$
en $K = L^G = \{x : \sigma(x) = x\ \forall\sigma \in G\}$ haar
vaste lichaam. Dan $[L : L^G] \leq \abs G$.
\end{lemma}

\begin{proof}
Zij $n = \abs G$, $G = \{\sigma_1, \dots, \sigma_n\}$, en stel
dat $x_1, \dots, x_{n+1} \in L$ lineair onafhankelijk over $K$
zijn. Het homogene lineaire stelsel van $n$ vergelijkingen in
$n+1$ onbekenden $(c_j)$ over $L$,
\[
\sum_{j=1}^{n+1} c_j\,\sigma_i(x_j) = 0
\qquad (i = 1, \dots, n),
\]
heeft een niet-nulle oplossing; kies er één met de \emph{minste}
niet-nulle entries, zeg $c_1, \dots, c_r \neq 0$
(hernummering), $r \geq 2$ (een enkele $c_j\sigma_i(x_j) = 0$ is
onmogelijk), genormaliseerd $c_r = 1$. Niet alle $c_j$ liggen in
$K$: de vergelijking voor $\sigma_i = \mathrm{id}$ zou
onafhankelijkheid tegenspreken; zeg $c_1 \notin K$, dus
$\tau(c_1) \ne c_1$ voor enige $\tau \in G$. Pas $\tau$ toe op
alle vergelijkingen: omdat $\tau\sigma_i$ over $G$ loopt, is de
vector $(\tau(c_j))_j$ een andere oplossing; aftrekken,
$(c_j - \tau(c_j))_j$ is een oplossing met minder niet-nulle
entries (de $r$-de entry $1 - 1 = 0$ verdwijnt, de eerste niet)
en niet nul: tegenspraak. Dus zijn $n+1$ elementen afhankelijk:
$[L:K] \leq n$.
\end{proof}

\begin{theorem}[Hoofdstelling van de Galoistheorie]
\label{thm:b3:galois:fundamental}
Zij $L/K$ een \omterm{def:b3:galois:galois}{Galoïsche uitbreiding} met groep $G =
\operatorname{Gal}(L/K)$.\index{Galoïsche correspondentie}
\begin{enumerate}
  \item $L^G = K$.
  \item De afbeeldingen $H \mapsto L^H$ en $F \mapsto
        \operatorname{Gal}(L/F)$ zijn elkaars inverse,
        inclusie-omkerende bijecties tussen deelgroepen van $G$
        en tussenlichamen $K \subseteq F \subseteq L$;
        bovendien $[L : L^H] = \abs H$ en $[L^H : K] = [G : H]$.
  \item $H \trianglelefteq G$ desda $L^H/K$ \omterm{def:b3:galois:galois}{Galoïsch} is, en dan
        induceert beperking $\operatorname{Gal}(L^H/K) \cong
        G/H$.
\end{enumerate}
\end{theorem}

\begin{proof}
(1) Duidelijk $K \subseteq L^G$. Omgekeerd zij $\alpha \in L
\setminus K$; we vertonen $\sigma \in G$ met $\sigma(\alpha) \ne
\alpha$. De \omterm{def:b3:galois:algebraic}{minimale veelterm} $\pi_\alpha$ over $K$ heeft \omterm{def:b3:galois:extension}{graad}
$\geq 2$ en is \omterm{def:b3:galois:separable}{separabel} ($L/K$ \omterm{def:b3:galois:separable}{separabel},
\cref{prop:b3:galois:galoisorder}), dus heeft ze een andere
wortel $\beta \neq \alpha$ in een \omterm{def:b3:galois:closure}{algebraïsche afsluiting}
$\Omega \supseteq L$. Breid de $K$-inbedding $K(\alpha) \to
\Omega$, $\alpha \mapsto \beta$, uit tot een inbedding
$\tau\colon L \to \Omega$
(\cref{prop:b3:galois:embeddings}); door normaliteit
(\cref{prop:b3:galois:galoisorder}) $\tau(L) = L$, dus $\tau \in
G$, $\beta = \tau(\alpha) \in L$, en $\tau(\alpha) \neq
\alpha$.

(2) Voor een deelgroep $H$: $L/L^H$ is \omterm{def:b3:galois:galois}{Galoïsch}
(\cref{prop:b3:galois:galoisorder}), en
$\operatorname{Gal}(L/L^H) \supseteq H$ triviaal, dus $[L:L^H]
= \abs{\operatorname{Gal}(L/L^H)} \geq \abs H$; het lemma van
Artin geeft $[L:L^H] \leq \abs H$: gelijkheid, en
$\operatorname{Gal}(L/L^H) = H$. Voor een tussenlichaam $F$:
$L/F$ \omterm{def:b3:galois:galois}{Galoïsch} geeft $L^{\operatorname{Gal}(L/F)} = F$ door (1)
toegepast op $L/F$. De twee afbeeldingen zijn elkaars inverse;
ze keren inclusies evident om. Graden: $[L:L^H] = \abs H$ zojuist
bewijs, en $[L^H:K] = [L:K]/[L:L^H] = \abs G/\abs H$.

(3) Voor $\sigma \in G$ en $H \leq G$: $\sigma(L^H) =
L^{\sigma H\sigma^{-1}}$ (directe controle). Door de bijectie,
$\sigma(L^H) = L^H$ voor alle $\sigma$ desda $H \trianglelefteq
G$. Nu als $H \trianglelefteq G$, stel $F = L^H$: elke $\sigma
\in G$ beperkt tot een automorfisme van $F$, wat een morfisme
$\rho \colon G \to \operatorname{Aut}_K(F)$ geeft met kern
$\{\sigma : \sigma\restriction_F = \mathrm{id}\} =
\operatorname{Gal}(L/F) = H$. Dus injecteert $G/H$ in
$\operatorname{Aut}_K(F)$, vandaar $\abs{\operatorname{Aut}_K(F)}
\geq [G:H] = [F:K]$; de omgekeerde ongelijkheid geldt altijd
(\cref{prop:b3:galois:embeddings}):
$\abs{\operatorname{Aut}_K(F)} = [F:K]$ en $\rho$ is surjectief.
Het blijft te zien dat $F/K$ \omterm{def:b3:galois:galois}{Galoïsch} is: $F$ is \omterm{def:b3:galois:separable}{separabel} over
$K$ (binnen de separabele $L/K$), en $F = K(\gamma)$
(\cref{thm:b3:galois:primitive}); de veelterm
$\prod_{\sigma \in G/H}\bigl(X - \sigma(\gamma)\bigr)$ (product
over de onderscheiden beelden, die in $F$ liggen: $\sigma(F) =
L^{\sigma H\sigma^{-1}} = L^H = F$ door normaliteit van $H$)
heeft coëfficiënten vast onder $G$, dus in $K$ door (1): ze is
een \omterm{def:b3:galois:separable}{separabele veelterm} van $K[X]$ gesplitst door $F$, en haar
wortels brengen $F$ voort: $F/K$ is \omterm{def:b3:galois:galois}{Galoïsch}. Omgekeerd, als $F
= L^H$ met $F/K$ \omterm{def:b3:galois:galois}{Galoïsch}, geeft normaliteit van $F$
(\cref{prop:b3:galois:galoisorder}, toegepast op inbeddingen $F
\to \Omega$ beperkt van elementen van $G$) $\sigma(F) = F$ voor
alle $\sigma \in G$, d.w.z.\ $H \trianglelefteq G$.
\end{proof}

\section{Cyclotomische uitbreidingen}

\begin{definition}\label{def:b3:galois:cyclotomic}
Zij $n \geq 1$ en $\zeta_n = \eu^{2\iu\pi/n}$. De $n$-de
\emph{cyclotomische veelterm}\index{cyclotomische veelterm} is
$\Phi_n = \prod_{\gcd(k,n)=1,\ 1 \le k \le n} \bigl(X -
\zeta_n^k\bigr)$, van \omterm{def:b3:galois:extension}{graad} $\varphi(n)$; de wortels van $X^n -
1$ groeperend naar exacte orde, $X^n - 1 = \prod_{d \mid
n}\Phi_d$, wat inductief toont dat $\Phi_n \in \Z[X]$
(Euclidische deling van monische gehele veeltermen).
\end{definition}

\begin{theorem}\label{thm:b3:galois:cyclotomicirred}
$\Phi_n$ is irreducibel over $\Q$; dus $[\Q(\zeta_n) : \Q] =
\varphi(n)$ en
\[
\operatorname{Gal}\bigl(\Q(\zeta_n)/\Q\bigr) \;\cong\;
(\Z/n\Z)^\times,
\qquad \sigma_a(\zeta_n) = \zeta_n^a .
\]
De uitbreiding $\Q(\zeta_n)/\Q$ is dus \omterm{def:b3:galois:galois}{Galoïsch} met
\emph{abelse} groep.
\end{theorem}

\begin{proof}
Zij $f = \pi_{\zeta_n}$, dus $\Phi_n = fg$ met $f, g \in \Z[X]$
monisch (lemma van Gauss \cref{lem:b3:rings:gauss}: \omterm{def:b3:rings:content}{inhouden}
vermenigvuldigen, alle veeltermen monisch). \emph{Bewering: als
$\zeta$ een wortel van $f$ is en $p \nmid n$ priem, dan is
$\zeta^p$ een wortel van $f$.} Anders is $\zeta^p$ een wortel
van $g$ (ze is een primitieve $n$-de eenheidswortel), dus is
$\zeta$ een wortel van $g(X^p)$, en $f \mid g(X^p)$ in $\Z[X]$
(\omterm{def:b3:galois:algebraic}{minimale veelterm}, dan opnieuw Gauss). Reduceer mod $p$: $\bar
g(X^p) = \bar g(X)^p$ (\omterm{thm:b3:galois:finitefields}{Frobenius} op $\mathbb F_p[X]$:
coëfficiëntgewijs $a^p = a$, en beginnelingsdroom), dus $\bar f
\mid \bar g^{\,p}$: $\bar f$ en $\bar g$ delen een irreducibele
factor, en $\bar\Phi_n = \bar f\bar g$ heeft een herhaalde
factor. Dan heeft $X^n - \bar 1$ dat ook; maar haar afgeleide
$\bar nX^{n-1}$ is copriem ermee ($p \nmid n$, en $0$ is geen
wortel): tegenspraak.

Elke primitieve wortel $\zeta_n^k$ ($\gcd(k, n) = 1$) wordt uit
$\zeta_n$ verkregen door opeenvolgende priemmachten die $n$ niet
delen (factoriseer $k$): de bewering propageert, dus is elke
primitieve wortel een wortel van $f$: $f = \Phi_n$,
irreducibel. Bijgevolg $[\Q(\zeta_n):\Q] = \varphi(n)$, en
$\Q(\zeta_n)$ is het \omterm{thm:b3:galois:splitting}{splitsingslichaam} van de separabele $X^n -
1$ (alle wortels zijn machten van $\zeta_n$): \omterm{def:b3:galois:galois}{Galoïsch}. Een
automorfisme $\sigma$ stuurt $\zeta_n$ naar een andere
primitieve wortel $\zeta_n^{a(\sigma)}$, en $\sigma \mapsto
a(\sigma)$ is een injectief morfisme naar $(\Z/n\Z)^\times$;
beide groepen hebben orde $\varphi(n)$: isomorfisme.
\end{proof}

\section{Liniaal en passer}

\begin{definition}\label{def:b3:galois:constructible}
Identificeer het vlak met $\C$; start van $\{0, 1\}$. Een punt is
\emph{construeerbaar}\index{construeerbaar getal} als het door
eindig veel doorsnijdingen van lijnen door twee reeds
geconstrueerde punten en cirkels gecentreerd in een
geconstrueerd punt met straal een afstand tussen twee
geconstrueerde punten verkregen kan worden.
\end{definition}

\begin{theorem}[Wantzel]\label{thm:b3:galois:wantzel}
$z \in \C$ is \omterm{def:b3:galois:constructible}{construeerbaar} desda er een toren $\Q = F_0
\subseteq F_1 \subseteq \dots \subseteq F_r$ is met $[F_{i+1} :
F_i] = 2$ en $z \in F_r$. In het bijzonder is een \omterm{def:b3:galois:constructible}{construeerbaar
getal} \omterm{def:b3:galois:algebraic}{algebraïsch} van \omterm{def:b3:galois:extension}{graad} een macht van $2$ over
$\Q$.\index{stelling van Wantzel}
\end{theorem}

\begin{proof}
($\Rightarrow$) De coördinaten van de doorsnijding van twee
lijnen door punten met coördinaten in een deellichaam $F
\subseteq \R$ lossen een lineair stelsel over $F$ op: ze blijven
in $F$. Lijn--cirkel- en cirkel--cirkeldoorsnijdingen leiden, na
eliminatie van het lineaire deel (aftrekken van de twee
cirkelvergelijkingen geeft een lijn), tot een kwadratische
vergelijking over $F$: de nieuwe coördinaten liggen in $F$ of in
$F(\sqrt d)$ voor enige $d \in F$, $d > 0$. Door inductie heeft
elk geconstrueerd punt coördinaten in een toren van kwadratische
uitbreidingen van $\Q$; en $z = x + \iu y$ zit ook in een
kwadratische toren (adjointeer $\iu$: één extra kwadratische
stap). Het graadgevolg: $[\Q(z):\Q]$ deelt $[F_r : \Q] = 2^r$
(\omterm{thm:b3:galois:tower}{torenwet}).

($\Leftarrow$) De construeerbare getallen vormen een lichaam:
sommen en verschillen door parallellogrammen (parallellen zijn
\omterm{def:b3:galois:constructible}{construeerbaar}: laat loodlijnen vallen en oprichten --- de
klassieke loodlijn door een punt gebruikt één cirkel en twee
bogen); producten en quotiënten door Thales-configuraties
(gegeven lengtes $a, b$ construeer $ab$ en $a/b$ met gelijkvormige
driehoeken op twee stralen). En het lichaam is gesloten onder
vierkantswortels: voor $a > 0$ snijden de cirkel van diameter $1
+ a$ en de loodlijn in het verbindingspunt op hoogte $\sqrt a$
(hoogte-meetkundig-gemiddelde-relatie in een rechthoekige
driehoek); voor complex $w = \rho\eu^{\iu\theta}$, construeer
$\sqrt\rho$ en deel $\theta$ in tweeën (hoekdeling is een
passerconstructie). Reële en imaginaire delen van leden van een
kwadratische toren zijn dus \omterm{def:b3:galois:constructible}{construeerbaar} door inductie op de
toren: elke stap adjointeert wortels van een kwadratische,
uitdrukbaar door lichaamsoperaties en één vierkantswortel van een
reeds geconstrueerd getal (de kwadratische formule; in
karakteristiek $0$).
\end{proof}

\begin{corollary}\label{cor:b3:galois:impossible}
De drie klassieke problemen zijn onoplosbaar met liniaal en
passer:\index{liniaal en passer}
\begin{enumerate}
  \item \emph{Kubusverdubbeling}: $\sqrt[3]2$ heeft \omterm{def:b3:galois:extension}{graad} $3$,
        geen macht van $2$.
  \item \emph{Hoekdeling in drieën}: deling van $60^\circ$ in
        drieën eist $\cos 20^\circ$, een wortel van de
        irreducibele $8X^3 - 6X - 1$: \omterm{def:b3:galois:extension}{graad} $3$.
  \item \emph{Kwadratuur van de cirkel}: $\sqrt\pi$ is
        transcendent ($\pi$ is dat --- stelling van Lindemann,
        hier toegelaten: haar bewijs hoort bij een cursus
        transcendentietheorie).
\end{enumerate}
Ook is de regelmatige $n$-hoek \omterm{def:b3:galois:constructible}{construeerbaar} desda $\varphi(n)$
een macht van $2$ is (Gauss--Wantzel; de ``als'' gebruikt de
methode van het weekendprobleem, de ``alleen als'' is
\cref{thm:b3:galois:wantzel} toegepast op $\zeta_n$, van \omterm{def:b3:galois:extension}{graad}
$\varphi(n)$). Voor $n = 7$: $\varphi(7) = 6$: de regelmatige
zevenhoek is onmogelijk; voor $n = 17$: $\varphi(17) = 16 =
2^4$: \omterm{def:b3:galois:constructible}{construeerbaar} --- het weekendprobleem construeert haar.
\end{corollary}

\begin{proof}
(1) $X^3 - 2$ is irreducibel (\omterm{thm:b3:rings:criteria}{Eisenstein}). (2) Uit $\cos 3\theta
= 4\cos^3\theta - 3\cos\theta$ met $3\theta = 60^\circ$: $8c^3 -
6c = 1$ voor $c = \cos 20^\circ$; de kubische $8X^3 - 6X - 1$
heeft geen rationale wortel (kandidaten $\pm1, \pm\frac12,
\pm\frac14, \pm\frac18$ falen), is dus irreducibel: \omterm{def:b3:galois:extension}{graad} $3$.
Een algemene $60^\circ$-hoek is \omterm{def:b3:galois:constructible}{construeerbaar}, dus zou een
trisection $c$ construeren. (3) Als $\sqrt\pi$ \omterm{def:b3:galois:constructible}{construeerbaar} was,
zou ze \omterm{def:b3:galois:algebraic}{algebraïsch} zijn, dus ook $\pi$. De $n$-hoek-uitspraak: de
\omterm{def:b3:galois:extension}{graad} van $\zeta_n$ is $\varphi(n)$
(\cref{thm:b3:galois:cyclotomicirred}); noodzakelijkheid volgt
uit Wantzel; voor voldoendeheid laat de \omterm{def:b3:galois:galois}{Galoisgroep}, abels van
orde $2^m$, een keten van index-$2$-deelgroepen toe (een eindige
$2$-groep doet dat: \cref{exo:b3:groups:10}), waarvan de vaste
lichamen een kwadratische toren vormen die eindigt in
$\Q(\zeta_n)$ (\cref{thm:b3:galois:fundamental}); besluit door
\cref{thm:b3:galois:wantzel}.
\end{proof}

\section{Oplosbaarheid door radicalen}

\begin{definition}\label{def:b3:galois:radical}
Een uitbreiding $L/K$ (karakteristiek $0$ overal in deze sectie)
is \emph{radicaal}\index{radicale uitbreiding} als er een toren
$K = F_0 \subseteq \dots \subseteq F_r = L$ is met $F_{i+1} =
F_i(\alpha_i)$, $\alpha_i^{n_i} \in F_i$: elke stap adjointeert
een $n_i$-de wortel. Een veelterm $P \in K[X]$ is
\emph{oplosbaar door radicalen}\index{oplosbaar door radicalen}
als haar \omterm{thm:b3:galois:splitting}{splitsingslichaam} in enige radicale uitbreiding van $K$
zit.
\end{definition}

\begin{lemma}\label{lem:b3:galois:cyclicsteps}
Zij $K$ een primitieve $n$-de eenheidswortel $\zeta$ bevat,
d.w.z.\ $\zeta$ van orde $n$ in $K^\times$, en $a \in K^\times$.
Dan is $K(\sqrt[n]a)/K$ \omterm{def:b3:galois:galois}{Galoïsch} met \emph{cyclische} groep.
Omgekeerd --- hieronder niet nodig --- is elke cyclische
uitbreiding van \omterm{def:b3:galois:extension}{graad} $n$ van deze vorm. Bovendien is
$K(\zeta_n)/K$ \omterm{def:b3:galois:galois}{Galoïsch} met \emph{abelse} groep, voor elke $K$
van karakteristiek $0$.
\end{lemma}

\begin{proof}
$X^n - a$ is \omterm{def:b3:galois:separable}{separabel} ($\gcd$ met $nX^{n-1}$: $a \neq 0$) en
splitst in $K(\alpha)$, $\alpha^n = a$: haar wortels zijn de
$\zeta^k\alpha \in K(\alpha)$. Dus is $K(\alpha)/K$ \omterm{def:b3:galois:galois}{Galoïsch}; de
afbeelding $\sigma \mapsto \sigma(\alpha)/\alpha \in \mu_n =
\langle \zeta\rangle$ is een injectief morfisme
($\sigma\tau(\alpha) = \sigma(\tau(\alpha)/\alpha \cdot \alpha)
= \tau(\alpha)/\alpha\cdot\sigma(\alpha)$, het quotiënt ligt in
$K$), naar een cyclische groep: $\operatorname{Gal}$ is
cyclisch. Het omgekeerde is Kummertheorie, die we niet nodig
hebben. Voor $K(\zeta_n)$: ze splitst de separabele $X^n - 1$,
en $\sigma \mapsto a(\sigma)$ met $\sigma(\zeta_n) =
\zeta_n^{a(\sigma)}$ injecteert de groep in de abelse
$(\Z/n\Z)^\times$ zoals in \cref{thm:b3:galois:cyclotomicirred}
(injectiviteit eist alleen dat $\zeta_n$ de betrokken
eenheidswortels voortbrengt).
\end{proof}

\begin{theorem}[Galois]\label{thm:b3:galois:solvable}
Zij $K$ van karakteristiek $0$ en $P \in K[X]$ met
\omterm{thm:b3:galois:splitting}{splitsingslichaam} $L$. Als $P$ \omterm{def:b3:groups:derived}{oplosbaar} is door radicalen, dan
is $\operatorname{Gal}(L/K)$ een \omterm{def:b3:groups:derived}{oplosbare groep}. (Het
omgekeerde is ook waar; we zullen het niet nodig hebben.)
\index{oplosbaarheidsstelling van Galois}
\end{theorem}

\begin{proof}
Stap 1: \emph{vergroot de radicalentoren tot een Galoïsche.} Zij
$L \subseteq M$ met $M/K$ \omterm{def:b3:galois:radical}{radicaal}, met wortelexponenten $n_1,
\dots, n_r$ en $n = n_1\cdots n_r$. Adjointeer eerst $\zeta_n$:
de toren $K \subseteq K(\zeta_n) \subseteq M(\zeta_n)$ is nog
\omterm{def:b3:galois:radical}{radicaal} ($\zeta_n$ is een eenheidswortel: een radicale stap,
$\zeta_n^n = 1$), en haar stappen voorbij de eerste gebeuren over
lichamen die de nodige eenheidswortels bevatten. Vervang
vervolgens $M(\zeta_n)$ door het composiet $N$ van alle
$\sigma(M(\zeta_n))$, $\sigma$ lopend over de (eindig veel)
$K$-inbeddingen van $M(\zeta_n)$ in een vaste \omterm{def:b3:galois:closure}{algebraïsche
afsluiting}: $N$ is het \omterm{thm:b3:galois:splitting}{splitsingslichaam} van het product van
\omterm{def:b3:galois:algebraic}{minimale veeltermen} van een voortbrengende verzameling
(karakteristiek $0$: eindig en \omterm{def:b3:galois:separable}{separabel}), dus is $N/K$
\omterm{def:b3:galois:galois}{Galoïsch}; en $N$ is \omterm{def:b3:galois:radical}{radicaal} over $K$: elk $\sigma(M(\zeta_n))$
is \omterm{def:b3:galois:radical}{radicaal} over $K$ (pas $\sigma$ toe op een radicalentoren),
en een composiet van \omterm{def:b3:galois:radical}{radicale uitbreidingen} is \omterm{def:b3:galois:radical}{radicaal}
(schakel de torens aaneen).

Stap 2: \emph{lees oplosbaarheid af van de Galoïsche toren.}
Neem dus aan $L \subseteq N$, $N/K$ \omterm{def:b3:galois:galois}{Galoïsch} en \omterm{def:b3:galois:radical}{radicaal} met
toren $K \subseteq K(\zeta_n) = E_0 \subseteq E_1 \subseteq
\dots \subseteq E_s = N$, elke $E_{i+1} =
E_i(\sqrt[n_i]{a_i})$ met $\zeta_{n_i} \in E_0 \subseteq E_i$.
Zij $G = \operatorname{Gal}(N/K)$ en $G_i =
\operatorname{Gal}(N/E_i)$: een dalende keten $G \supseteq G_0
\supseteq G_1 \supseteq \dots \supseteq G_s = \{e\}$. Elke
$E_{i+1}/E_i$ is \omterm{def:b3:galois:galois}{Galoïsch} met cyclische groep
(\cref{lem:b3:galois:cyclicsteps}), dus door de hoofdstelling
toegepast op de \omterm{def:b3:galois:galois}{Galoïsche uitbreiding} $N/E_i$
(\cref{thm:b3:galois:fundamental}(3), met omgevende groep
$G_i$): $G_{i+1} \trianglelefteq G_i$ met $G_i/G_{i+1} \cong
\operatorname{Gal}(E_{i+1}/E_i)$ cyclisch. Evenzo is $E_0/K$
\omterm{def:b3:galois:galois}{Galoïsch} met abelse groep $G/G_0$
(\cref{lem:b3:galois:cyclicsteps}). De keten vertoont $G$ als
\omterm{def:b3:groups:derived}{oplosbaar} (\cref{prop:b3:groups:derived}). Ten slotte is
$\operatorname{Gal}(L/K)$ een \emph{quotiënt} van $G$: $L/K$ is
\omterm{def:b3:galois:galois}{Galoïsch} ($P$ \omterm{def:b3:galois:separable}{separabel} in karakteristiek $0$) en beperking $G
\to \operatorname{Gal}(L/K)$ is surjectief
(\cref{thm:b3:galois:fundamental}(3) met $H =
\operatorname{Gal}(N/L)$); quotiënten van \omterm{def:b3:groups:derived}{oplosbare groepen} zijn
\omterm{def:b3:groups:derived}{oplosbaar}.
\end{proof}

\begin{corollary}[Onoplosbaarheid van de vijfdegraads]
\label{cor:b3:galois:quintic}
Er zijn veeltermen van \omterm{def:b3:galois:extension}{graad} $5$ over $\Q$ die niet \omterm{def:b3:groups:derived}{oplosbaar}
zijn door radicalen: bijvoorbeeld $X^5 - 4X + 2$, waarvan de
\omterm{def:b3:galois:galois}{Galoisgroep} $S_5$ is (\cref{exo:b3:galois:11}), een niet-oplosbare
groep (\cref{cor:b3:groups:snnotsolvable}). Geen algemene formule
in radicalen kan bestaan voor \omterm{def:b3:galois:extension}{graad} $\geq 5$.\index{vijfdegraads,
onoplosbaarheid}
\end{corollary}

\begin{remark}\label{rem:b3:galois:converse}
Het omgekeerde van \cref{thm:b3:galois:solvable} --- een
oplosbare \omterm{def:b3:galois:galois}{Galoisgroep} impliceert oplosbaarheid door radicalen ---
wordt bewezen door de afgeleide reeks af te dalen en te tonen dat
elke cyclische uitbreiding (met genoeg eenheidswortels) \omterm{def:b3:galois:radical}{radicaal}
is, via \emph{Lagrange-resolventen}; het verklaart waarom graden
$2, 3, 4$ formules hebben: $S_2, S_3, S_4$ zijn \omterm{def:b3:groups:derived}{oplosbaar}
(\cref{ex:b3:groups:solvableexamples}). We laten het op dit
niveau toe; een volledige behandeling hoort bij een
mastercursus, maar \cref{exo:b3:galois:8} maakt het concreet voor
de kubische.
\end{remark}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:b3:galois:1}
Toon $[\Q(\sqrt2, \sqrt3):\Q] = 4$, dat $\Q(\sqrt2 + \sqrt3) =
\Q(\sqrt2, \sqrt3)$, en bereken de \omterm{def:b3:galois:algebraic}{minimale veelterm} van
$\sqrt2 + \sqrt3$ over $\Q$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b3:galois:2}
Zij $\alpha = \sqrt[3]2$. Toon dat $\Q(\alpha)/\Q$ niet normaal
is (vertoon een inbedding $\Q(\alpha) \to \C$ waarvan het beeld
niet $\Q(\alpha)$ is), bepaal het \omterm{thm:b3:galois:splitting}{splitsingslichaam} $L$ van $X^3
- 2$ en $[L:\Q]$, en controleer $\operatorname{Aut}_\Q(\Q(\alpha))
= \{\mathrm{id}\}$: voor niet-Galoïsche uitbreidingen kan de
automorfismengroep veel kleiner zijn dan de \omterm{def:b3:galois:extension}{graad}.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b3:galois:3}
Construeer $\mathbb F_9$ als $\mathbb F_3[X]/(X^2+1)$ en vind een
voortbrenger van $\mathbb F_9^\times$. Som de monische
irreducibele veeltermen van graden $1, 2, 3$ over $\mathbb F_2$
op, en verifieer $X^8 - X = X(X+1)(X^3+X+1)(X^3+X^2+1)$ over
$\mathbb F_2$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:galois:4}
(a) Vind alle primitieve wortels modulo $7$ en modulo $11$
(d.w.z.\ voortbrengers van $\mathbb F_7^\times$, $\mathbb
F_{11}^\times$).
(b) Toon dat voor $p$ oneven, $x \in \mathbb F_p^\times$ een
kwadraat is desda $x^{(p-1)/2} = 1$ (criterium van Euler), en
herwin het criterium voor $-1$ van \cref{pb:b3:rings:1}.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:galois:5}
Toon dat $\mathbb F_{p^m} \cap \mathbb F_{p^n} = \mathbb
F_{p^{\gcd(m,n)}}$ en $\mathbb F_{p^m}\mathbb F_{p^n} = \mathbb
F_{p^{\operatorname{lcm}(m,n)}}$ binnen een vaste \omterm{def:b3:galois:closure}{algebraïsche
afsluiting} $\bar{\mathbb F}_p$, en beschrijf
$\operatorname{Gal}(\mathbb F_{p^n}/\mathbb F_{p^m})$ voor $m
\mid n$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:galois:6}
Zij $I_d(q)$ het aantal monische irreducibele veeltermen van
\omterm{def:b3:galois:extension}{graad} $d$ over $\mathbb F_q$. Bewijs
\[
X^{q^n} - X \;=\; \prod_{d \mid n}\ \prod_{P \text{ irred.
monisch, } \deg P = d} P ,
\qquad\text{dus}\qquad
q^n = \sum_{d \mid n} d\, I_d(q).
\]
Leid $I_1, I_2, I_3, I_4$ expliciet af, en $I_d(q) \geq 1$ voor
elke $d$ (dus bestaan uitbreidingen $\mathbb F_{q^d}/\mathbb F_q$
als quotiënten $\mathbb F_q[X]/(P)$ voor alle $d$).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:galois:7}
Bepaal $\operatorname{Gal}(\Q(\sqrt2,\sqrt3)/\Q)$ en het
volledige rooster van tussenlichamen. Zelfde vraag voor het
\omterm{thm:b3:galois:splitting}{splitsingslichaam} van $(X^2-2)(X^2-3)(X^2-6)$ --- wat merk je
op?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:galois:8}
(De kubische, opgelost door haar groep) Zij $P = X^3 + pX + q
\in \Q[X]$ irreducibel met wortels $x_1, x_2, x_3$ en
\omterm{thm:b3:galois:splitting}{splitsingslichaam} $L$. Zij $\delta = (x_1 - x_2)(x_1 -
x_3)(x_2 - x_3)$ en $\Delta = \delta^2 = -4p^3 - 27q^2$ (laat
deze klassieke identiteit toe of verifieer door symmetrische
functies te ontwikkelen).
(a) Toon $\operatorname{Gal}(L/\Q) \cong A_3$ of $S_3$, naargelang
$\Delta$ wel of niet een kwadraat in $\Q$ is.
(b) Met $j = \zeta_3$, definieer de Lagrange-resolventen $u =
x_1 + jx_2 + j^2x_3$ en $v = x_1 + j^2x_2 + jx_3$. Toon $u^3 +
v^3 = -27q$ en $uv = -3p$, en los op voor $u^3, v^3$: Cardano's
formules vallen eruit. Waar werd oplosbaarheid van $S_3$
gebruikt?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:galois:9}
In $\Q(\zeta_5)$: toon dat het unieke kwadratische deellichaam
$\Q(\sqrt5)$ is, via de Gauss-sommen $\eta_0 = \zeta_5 +
\zeta_5^4$, $\eta_1 = \zeta_5^2 + \zeta_5^3$: bereken $\eta_0 +
\eta_1$ en $\eta_0\eta_1$, en leid af $\cos\frac{2\pi}5 =
\frac{\sqrt5 - 1}4$. Concludeer dat de regelmatige vijfhoek
\omterm{def:b3:galois:constructible}{construeerbaar} is.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b3:galois:10}
Zij $K = \mathbb F_p(S, T)$ (rationale functies in twee
onbepaalden) en $L = K(S^{1/p}, T^{1/p})$.
(a) Toon $[L:K] = p^2$ en dat $\alpha^p \in K$ voor elke
$\alpha \in L$.
(b) Leid af dat $L/K$ \emph{niet} \omterm{def:b3:groups:simple}{eenvoudig} is: er bestaat geen
primitief element --- inseparabiliteit is fataal voor
\cref{thm:b3:galois:primitive}.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b3:galois:11}
Zij $P = X^5 - 4X + 2$ en $G$ haar \omterm{def:b3:galois:galois}{Galoisgroep} over $\Q$,
werkend op de $5$ wortels.
(a) Toon dat $P$ irreducibel is, en leid af $5 \mid \abs G$;
concludeer dat $G$ een $5$-cykel bevat (Cauchy,
\cref{thm:b3:groups:cauchy}).
(b) Toon, door de variaties van $x \mapsto x^5 - 4x + 2$ te
bestuderen, dat $P$ precies $3$ reële wortels heeft; leid af dat
complex conjugeren beperkt tot een transpositie in $G$.
(c) Toon dat een deelgroep van $S_5$ die een transpositie en een
$5$-cykel bevat $S_5$ is \emph{(conjugeer de transpositie door
machten van de cykel)}. Concludeer $G \cong S_5$ en, met
\cref{thm:b3:galois:solvable}, dat $P$ niet \omterm{def:b3:groups:derived}{oplosbaar} is door
radicalen.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b3:galois:12}
(De diëder-kwartische) Zij $\alpha = \sqrt[4]2$ en $L =
\Q(\alpha, \iu)$, het \omterm{thm:b3:galois:splitting}{splitsingslichaam} van $X^4 - 2$ over $\Q$.
(a) Toon $[L : \Q] = 8$ en dat $G = \operatorname{Gal}(L/\Q)$
wordt voortgebracht door $\sigma\colon \alpha \mapsto
\iu\alpha,\ \iu \mapsto \iu$ en complexe conjugatie $\tau$, met
$\sigma^4 = \tau^2 = e$ en $\tau\sigma\tau = \sigma^{-1}$: $G
\cong D_4$.
(b) Som het deelgroepenrooster van $D_4$ op (tien deelgroepen) en
match elk met haar vaste lichaam; verifieer in het bijzonder dat
$\Q(\sqrt2)$, $\Q(\iu)$, $\Q(\iu\sqrt2)$ de drie kwadratische
deellichamen zijn, en lokaliseer $\Q(\alpha)$, $\Q(\iu\alpha)$,
$\Q(\sqrt2, \iu)$.
(c) Welke tussenlichamen zijn \omterm{def:b3:galois:galois}{Galoïsch} over $\Q$? Match je
antwoord tegen de \omterm{def:b3:groups:normal}{normale deelgroepen} van $D_4$, en verklaar
waarom $\Q(\alpha)/\Q$ faalt terwijl $\Q(\sqrt2)/\Q$ slaagt.
\end{exercise}

\section{Probleem: Gauss en de regelmatige 17-hoek}

\begin{problem}[{Weekendprobleem --- construeerbaarheid van de
17-hoek}]\label{pb:b3:galois:1}
Op 30 maart 1796 toonde de negentienjarige Gauss dat de
regelmatige $17$-hoek \omterm{def:b3:galois:constructible}{construeerbaar} is --- de eerste
vooruitgang op de vraag sinds de oudheid. We reconstrueren zijn
berekening met de gereedschappen van dit hoofdstuk. Stel $\zeta
= \eu^{2\iu\pi/17}$, $L = \Q(\zeta)$, $G =
\operatorname{Gal}(L/\Q)$.

\medskip
\noindent\textbf{Deel I --- De groep en haar filtratie.}
\begin{enumerate}
  \item Rechtvaardig: $[L:\Q] = 16$, $G \cong (\Z/17\Z)^\times$,
        cyclisch van orde $16$. Verifieer dat $3$ een
        voortbrenger van $(\Z/17\Z)^\times$ is \emph{(bereken de
        machten van $3$ modulo $17$: $3, 9, 10, 13, 5, 15, 11,
        16, \dots$)}.
  \item Zij $\sigma \in G$ met $\sigma(\zeta) = \zeta^3$, en
        $H_k = \langle \sigma^{2^k}\rangle$ voor $k = 0, \dots,
        4$. Toon dat $G = H_0 \supset H_1 \supset H_2 \supset
        H_3 \supset H_4 = \{e\}$ met elke index $[H_k :
        H_{k+1}] = 2$, en dat de vaste lichamen $\Q = L_0
        \subset L_1 \subset L_2 \subset L_3 \subset L_4 = L$ een
        toren van kwadratische uitbreidingen vormen.
  \item Concludeer \emph{a priori}, met
        \cref{thm:b3:galois:wantzel}, dat $\zeta$ --- dus de
        $17$-hoek --- \omterm{def:b3:galois:constructible}{construeerbaar} is. De rest van het
        probleem maakt de toren expliciet.
\end{enumerate}

\medskip
\noindent\textbf{Deel II --- De periodes van lengte 8.}
Definieer de \emph{Gaussische periodes}
\[
\eta_0 = \sum_{k \text{ even}} \zeta^{3^k \bmod 17}
= \zeta^{1} + \zeta^{9} + \zeta^{13} + \zeta^{15} + \zeta^{16} +
\zeta^{8} + \zeta^{4} + \zeta^{2},
\qquad
\eta_1 = \sum_{k \text{ odd}} \zeta^{3^k \bmod 17}.
\]
\begin{enumerate}[resume]
  \item Toon dat $\eta_0, \eta_1$ vast zijn onder $H_1$ en
        verwisseld door $\sigma$; leid af $\eta_0, \eta_1 \in
        L_1$ en dat ze de twee wortels van een kwadratische over
        $\Q$ zijn.
  \item Bereken $\eta_0 + \eta_1 = -1$. Toon $\eta_0\eta_1 = -4$
        \emph{(elk product $\zeta^a\zeta^b$ is enige $\zeta^c$,
        $c \neq 0$; tel hoe vaak elke $c$ voorkomt, of
        argumenteer dat het product een rationaal geheel is vast
        onder $G$)}.
  \item Leid af $\eta_0 = \frac{-1 + \sqrt{17}}2$, $\eta_1 =
        \frac{-1-\sqrt{17}}2$ (identificeer welke welke is
        numeriek: $\eta_0 \approx 1{,}56$), en $L_1 =
        \Q(\sqrt{17})$.
\end{enumerate}

\medskip
\noindent\textbf{Deel III --- Periodes van lengte 4 en 2.}
Definieer
\[
\beta_0 = \zeta + \zeta^{13} + \zeta^{16} + \zeta^{4},\quad
\beta_1 = \zeta^3 + \zeta^5 + \zeta^{14} + \zeta^{12},\quad
\beta_2 = \zeta^9 + \zeta^{15} + \zeta^{8} + \zeta^{2},\quad
\beta_3 = \zeta^{10} + \zeta^{11} + \zeta^{7} + \zeta^{6}.
\]
\begin{enumerate}[resume]
  \item Toon $\beta_0 + \beta_2 = \eta_0$, $\beta_1 + \beta_3 =
        \eta_1$, en dat $\beta_0, \beta_2$ vast zijn onder $H_2$,
        verwisseld door $\sigma^2$.
  \item Bereken $\beta_0\beta_2 = -1$ en $\beta_1\beta_3 = -1$
        \emph{(ontwikkel: de zestien verkregen exponenten dekken
        $1, \dots, 16$ precies één keer)}.
  \item Leid af $\beta_0 = \frac{\eta_0 + \sqrt{\eta_0^2 + 4}}2$
        (controleer het teken numeriek: $\beta_0 \approx 2{,}05$)
        en de analoge formule voor $\beta_1$; dus $L_2 =
        \Q(\beta_0)$, kwadratisch over $L_1$.
  \item Zij $\gamma_0 = \zeta + \zeta^{16} =
        2\cos\frac{2\pi}{17}$ en $\gamma_1 = \zeta^{13} +
        \zeta^4$. Toon $\gamma_0 + \gamma_1 = \beta_0$ en
        $\gamma_0\gamma_1 = \beta_1$, zodat $\gamma_0 =
        \frac{\beta_0 + \sqrt{\beta_0^2 - 4\beta_1}}2$.
  \item Stel de keten van formules samen die
        $\cos\frac{2\pi}{17}$ door geneste vierkantswortels
        uitdrukt, en geef een decimale controle
        ($\cos\frac{2\pi}{17} \approx 0{,}93247$).
\end{enumerate}

\medskip
\noindent\textbf{Deel IV --- Epiloog.}
\begin{enumerate}[resume]
  \item Waar precies gebruikte het argument dat $17$ een
        \emph{Fermat-priem} is ($17 = 2^{2^2} + 1$)? Toon dat
        voor een priem $p$ de regelmatige $p$-hoek \omterm{def:b3:galois:constructible}{construeerbaar}
        is desda $p = 2^{2^t} + 1$ voor enige $t$ \emph{(als $p
        - 1 = 2^m$, toon dat $m$ zelf een macht van $2$ moet
        zijn)}.
  \item Leid de volledige lijst van construeerbare regelmatige
        $n$-hoeken voor $n \leq 20$ af, met het
        Gauss--Wantzel-criterium van
        \cref{cor:b3:galois:impossible}.
\end{enumerate}

\noindent\textbf{Deel V --- Gauss-sommen en kwadratische
reciprociteit.} De periodes van Deel II verbergen een schat.
Voor een oneven priem $p$ is het \emph{Legendre-symbool}
$\bigl(\frac ap\bigr)$ gelijk aan $+1$ als $a$ een niet-nul
kwadraat mod $p$ is, $-1$ als dat niet zo is, $0$ als $p \mid a$;
\cref{exo:b3:galois:4}(b) (criterium van Euler) geeft
$\bigl(\frac ap\bigr) \equiv a^{(p-1)/2} \pmod p$, vandaar
multiplicativiteit. Schrijf $\zeta = \eu^{2\iu\pi/p}$, $p^* =
(-1)^{(p-1)/2}p$, en definieer de \emph{Gauss-som}
\[
g \;=\; \sum_{a=1}^{p-1}\Bigl(\frac ap\Bigr)\zeta^a .
\]
\begin{enumerate}[resume]
  \item Toon $\sum_{a=1}^{p-1}\bigl(\frac ap\bigr) = 0$ (evenveel
        kwadraten als niet-kwadraten), en bewijs de alternatieve
        vorm $g = \sum_{a=0}^{p-1}\zeta^{a^2}$. Voor $p = 17$:
        relateer $g$ aan de periodes van Deel II --- toon $g =
        \eta_0 - \eta_1$.
  \item Bewijs $g^2 = p^*$: ontwikkel $g^2$, herindexeer, en
        besluit met vraag 14. Controleer numeriek: voor $p =
        17$, $(\eta_0 - \eta_1)^2 = 17$ (Deel II).
  \item Leid af $\sqrt{p^*} \in \Q(\zeta_p)$, en concludeer dat
        het \emph{unieke} kwadratische deellichaam van
        $\Q(\zeta_p)$ $\Q(\sqrt{p^*})$ is.
  \item Zij nu $q \neq p$ een andere oneven priem. Werkend in
        $\Z[\zeta]$ modulo $q$, bewijs $g^q \equiv
        \bigl(\frac qp\bigr)\,g \pmod{q\Z[\zeta]}$.
  \item Anderzijds $g^q = g\,(g^2)^{(q-1)/2} =
        g\,(p^*)^{(q-1)/2}$; met Euler mod $q$, leid af $g^q
        \equiv \bigl(\frac{p^*}q\bigr)g \pmod{q\Z[\zeta]}$, en
        concludeer $\bigl(\frac qp\bigr) =
        \bigl(\frac{p^*}q\bigr)$.
  \item Ontvouw tot de \emph{wet van kwadratische reciprociteit}:
        \[
        \Bigl(\frac pq\Bigr)\Bigl(\frac qp\Bigr)
        = (-1)^{\frac{p-1}2\cdot\frac{q-1}2} .
        \]
        Verifieer op $(p, q) = (17, 3)$, en gebruik haar om te
        beslissen of $x^2 \equiv 219 \pmod{383}$ \omterm{def:b3:groups:derived}{oplosbaar} is.
\end{enumerate}

\noindent\textbf{Deel VI --- Irreducibele veeltermen tellen: de
priemgetalstelling van $\mathbb F_q[X]$.} Fixeer een priemmacht
$q$ en zij $N_q(n)$ het aantal \emph{monische irreducibele}
veeltermen van \omterm{def:b3:galois:extension}{graad} $n$ over $\mathbb F_q$; herinner uit
\cref{exo:b3:galois:6} de factorisatie van $X^{q^n} - X$ en de
identiteit $q^n = \sum_{d\mid n}d\,N_q(d)$.
\begin{enumerate}[resume]
  \item (Woorden) Noem een woord $w \in \mathbb F_q^n$
        \emph{primitief} als het geen macht $u^{n/d}$ is van een
        strikt korter woord $u$, en zij $A(d)$ het aantal
        primitieve woorden van lengte $d$. Toon $q^n =
        \sum_{d \mid n}A(d)$; concludeer $A(d) = d\,N_q(d)$.
  \item Bewijs de \emph{Möbius-inversieformule}: als $f(n) =
        \sum_{d\mid n}g(d)$ voor alle $n$, dan $g(n) =
        \sum_{d\mid n}\mu(d)\,f(n/d)$. Leid af
        \[
        N_q(n) = \frac1n\sum_{d \mid n}\mu(d)\,q^{n/d} .
        \]
  \item Toon $N_q(n) \geq \frac1n\bigl(q^n - 2q^{n/2}\bigr) > 0$
        voor elke $n \geq 1$: een nieuw bewijs dat $\mathbb
        F_{q^n}$ voor alle $n$ bestaat. Interpreteer de
        leidende term: een willekeurige monische veelterm van
        \omterm{def:b3:galois:extension}{graad} $n$ is irreducibel met kans $\sim \frac1n$.
  \item Bewijs de multiplicatieve metgezel:
        \[
        \prod_{\substack{\pi \text{ monisch irred.}\\ \deg\pi =
        n}}\pi \;=\; \prod_{d \mid n}
        \bigl(X^{q^d} - X\bigr)^{\mu(n/d)}.
        \]
\end{enumerate}

\noindent\textbf{Deel VII --- Twee coda's.}
\begin{enumerate}[resume]
  \item (Het tweede supplement) De methode van Deel V berekent
        ook $\bigl(\frac2q\bigr)$. Zij $\omega =
        \eu^{2\iu\pi/8}$ en $g = \omega + \omega^{-1}$. Toon $g^2
        = 2$; bewijs voor oneven priem $q$ dat
        \[
        \Bigl(\frac2q\Bigr) = (-1)^{(q^2-1)/8}.
        \]
  \item (De zetafunctie van $\mathbb F_q[X]$) Bewijs de
        identiteit van formele machtreeksen in $t$:
        \[
        \prod_{n \geq 1}\bigl(1 - t^n\bigr)^{-N_q(n)}
        = \frac1{1 - qt},
        \]
        en herwin $q^m = \sum_{d \mid m}d\,N_q(d)$ door
        logaritmen te nemen. Bereken $N_2(6) = 9$.
\end{enumerate}
