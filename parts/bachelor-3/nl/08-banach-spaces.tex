\chapter{Banachruimten en de fundamentele stellingen}
\label{ch:b3:banach}

De functionaalanalyse bestudeert oneindigdimensionale
genormeerde ruimten via de operatoren en functionalen die erop
leven. Haar stichtende ontdekking is dat \emph{volledigheid}, via
de stelling van Baire, sterke uniformiteit afdwingt:
puntsgewijs begrensde families van operatoren zijn
normbegrensd (Banach--Steinhaus), continue bijecties hebben
continue inversen (open afbeelding), en grafieken detecteren
continuïteit (gesloten grafiek). De andere pijler,
\emph{Hahn--Banach}, heeft helemaal geen volledigheid nodig ---
alleen het lemma van Zorn --- en garandeert dat \omterm{def:b3:banach:operator}{duale ruimten}
rijk genoeg zijn om elke vector te zien. Dit hoofdstuk bewijst
alle vier stellingen en test ze op de klassieke rijenruimten
$\ell^p$, op concrete \omterm{def:b3:banach:operator}{duale} berekeningen, en op een werkelijk
verrassende toepassing: er bestaan continue $2\pi$-periodieke
functies waarvan de Fourierreeks in een punt \emph{divergeert}
--- een negatieve oplossing van een vraag die Jaar~2 openliet.

Doorheen zijn $E, F$ genormeerde ruimten over $K = \R$ of $\C$;
\emph{Banachruimte}\index{Banachruimte} betekent volledige
genormeerde ruimte.

\section{Begrensde operatoren; rijenruimten}

\begin{definition}\label{def:b3:banach:operator}
$\mathcal L(E, F)$ duidt de ruimte van \emph{begrensde}
(= continue, Jaar~2) lineaire afbeeldingen aan met de
\emph{operatornorm}\index{operatornorm} $\vertiii T =
\sup_{\norm x \leq 1}\norm{Tx}$; zij is submultiplicatief:
$\vertiii{ST} \leq \vertiii S\,\vertiii T$. De
\emph{duale}\index{duale ruimte} is $E' = \mathcal L(E, K)$.
\end{definition}

\begin{proposition}\label{prop:b3:banach:llcomplete}
Als $F$ een Banachruimte is, is $\mathcal L(E, F)$ dat ook; in
het bijzonder is $E'$ altijd een Banachruimte.
\end{proposition}

\begin{proof}
Zij $(T_n)$ Cauchy voor $\vertiii\cdot$. Voor elke $x$ is
$\norm{T_nx - T_mx} \leq \vertiii{T_n - T_m}\norm x$: $(T_nx)$
is Cauchy in $F$, convergent; noem de limiet $Tx$. $T$ is
lineair (limieten van lineaire identiteiten); overgang naar de
limiet in $\norm{T_nx - T_mx} \leq \varepsilon\norm x$ ($n, m
\geq N$) geeft $\norm{T_nx - Tx} \leq \varepsilon\norm x$: $T_n
\to T$ in \omterm{def:b3:banach:operator}{operatornorm}, en $\vertiii T \leq \vertiii{T_N} +
\varepsilon < \infty$.
\end{proof}

\begin{definition}\label{def:b3:banach:lp}
De klassieke rijenruimten\index{rijenruimten $\ell^p$} (over $K$,
geïndexeerd door $\N$):
\[
\ell^p = \Bigl\{x = (x_n) : \norm x_p = \Bigl(\sum_n
\abs{x_n}^p\Bigr)^{1/p} < \infty\Bigr\}\quad (1 \leq p <
\infty),
\]
\[
\ell^\infty = \{x : \norm x_\infty = \sup\abs{x_n} < \infty\},
\]
en $c_0 = \{x : x_n \to 0\}$ met $\norm\cdot_\infty$. Dat
$\norm\cdot_p$ een norm is volgt uit de ongelijkheid van
Minkowski, bewezen in het discrete geval precies zoals in
\cref{ch:b3:lp} (of door de eindigdimensionale ongelijkheid van
Jaar~2 te sommeren). Alle zijn Banachruimten, en $c_0$ is een
gesloten deelruimte van $\ell^\infty$ (\cref{exo:b3:banach:3}).
\end{definition}

\begin{proposition}[Neumannreeks]\label{prop:b3:banach:neumann}
Zij $E$ Banach en $T \in \mathcal L(E) = \mathcal L(E,E)$ met
$\vertiii T < 1$. Dan is $I - T$ inverteerbaar in $\mathcal
L(E)$, met $(I - T)^{-1} = \sum_{n\geq0}T^n$ (convergent in
\omterm{def:b3:banach:operator}{operatornorm}). Bijgevolg is de verzameling inverteerbare
operatoren open, en is inversie continu erop.\index{Neumannreeks}
\end{proposition}

\begin{proof}
De reeks convergeert absoluut ($\vertiii{T^n} \leq \vertiii
T^n$, meetkundig) in de Banach $\mathcal L(E)$
(\cref{prop:b3:banach:llcomplete};
\cref{exo:b3:complete:1}(b)). Telescoperend, $(I -
T)\sum_{n \leq N}T^n = I - T^{N+1} \to I$, en evenzo aan de
andere kant. Voor openheid: als $S$ inverteerbaar is en
$\vertiii{H} < 1/\vertiii{S^{-1}}$, dan $S + H = S(I +
S^{-1}H)$ met $\vertiii{S^{-1}H} < 1$: inverteerbaar. Continuïteit
van inversie: de reeksuitdrukking geeft $\vertiii{(S+H)^{-1} -
S^{-1}} = O(\vertiii H)$ lokaal.
\end{proof}

\begin{example}[Een Volterra-vergelijking, opgelost door Neumann]
\label{ex:b3:banach:volterraexample}
Op $E = \mathcal C(\intcc01)$, beschouw de integraalvergelijking
\[
u(x) = 1 + \lambda\int_0^xu(t)\,\dd t,
\qquad\text{d.w.z.}\qquad u = \mathbf 1 + \lambda Tu,
\quad (Tu)(x) = \int_0^xu .
\]
Hier $\vertiii T \leq 1$, dus voor $\abs\lambda < 1$ past de
\omterm{prop:b3:banach:neumann}{Neumannreeks} rechtstreeks: $u = (I - \lambda T)^{-1}\mathbf 1 =
\sum_n\lambda^nT^n\mathbf 1$. Berekenend, $T^n\mathbf 1 =
\frac{x^n}{n!}$, dus
\[
u(x) = \sum_{n\geq0}\frac{(\lambda x)^n}{n!} =
\eu^{\lambda x} ,
\]
zoals differentiatie bevestigt. Beter: $\vertiii{T^n} \leq
\frac1{n!}$ (de geïtereerde kern krimpt factorieel), dus
$\sum\lambda^nT^n$ convergeert voor \emph{elke} $\lambda$ ---
de operator $I - \lambda T$ is inverteerbaar voor alle $\lambda
\in \C$, ook al $\vertiii{\lambda T} \geq 1$ uiteindelijk: wat
telt is het spectraal verval van de machten, niet de eerste
norm. Volterra-operatoren hebben dit factorieel verval
ingebouwd (\cref{exo:b3:complete:4}(a) exploiteerde precies
dit), daarom lijden beginwaardeproblemen nooit onder de
resonantiefenomenen van randwaardeproblemen
(\cref{ch:b3:spectral}).
\end{example}

\section{Hahn--Banach}

\begin{theorem}[Hahn--Banach, analytische vorm]
\label{thm:b3:banach:hahnbanach}
Zij $E$ een \emph{reële} vectorruimte, $p \colon E \to \R$
\emph{sublineair} ($p(x + y) \leq p(x) + p(y)$ en $p(tx) =
tp(x)$ voor $t \geq 0$), $F \subseteq E$ een deelruimte en $f
\colon F \to \R$ lineair met $f \leq p$ op $F$. Dan breidt $f$
uit tot een lineaire $\tilde f \colon E \to \R$ met $\tilde f
\leq p$ op $E$.\index{stelling van Hahn--Banach}
\end{theorem}

\begin{proof}
\emph{Eénstapsuitbreiding.} Zij $x_0 \notin F$; we breiden $f$
uit tot $F \oplus \R x_0$ door $\alpha = \tilde f(x_0)$ correct
te kiezen: we hebben, voor alle $y \in F$, $t > 0$, nodig
\[
f(y) + t\alpha \leq p(y + tx_0)
\quad\text{en}\quad
f(y) - t\alpha \leq p(y - tx_0),
\]
wat na deling door $t$ (sublineariteit) reduceert tot
\[
\sup_{v \in F}\ \bigl[f(v) - p(v - x_0)\bigr]
\;\leq\; \alpha \;\leq\;
\inf_{u \in F}\ \bigl[p(u + x_0) - f(u)\bigr].
\]
Zulk $\alpha$ bestaat desda elk linkerlid $\leq$ elk
rechterlid is: inderdaad $f(v) + f(u) = f(u + v) \leq p(u + v)
\leq p(u + x_0) + p(v - x_0)$, d.w.z.\ $f(v) - p(v - x_0) \leq
p(u + x_0) - f(u)$.

\emph{Zorn.} Orden de uitbreidingen van $f$ gedomineerd door $p$
(paren: deelruimte, functionaal) door uitbreiding; een keten
heeft de unie als bovengrens; een \omterm{def:b3:rings:primemaximal}{maximaal} element moet op heel
$E$ gedefinieerd zijn, anders spreekt de éénstapsuitbreiding de
maximaliteit tegen.
\end{proof}

\begin{corollary}\label{cor:b3:banach:hbnormed}
Zij $E$ een genormeerde ruimte ($K = \R$ of $\C$).
\begin{enumerate}
  \item Elke $f \in F'$ ($F$ een deelruimte) breidt uit tot
        $\tilde f \in E'$ met $\norm{\tilde f}_{E'} = \norm
        f_{F'}$.
  \item Voor elke $x \neq 0$ is er $f \in E'$ met $\norm f = 1$
        en $f(x) = \norm x$. In het bijzonder scheidt $E'$ de
        punten van $E$, en $\norm x = \sup_{\norm f \leq
        1}\abs{f(x)}$.
  \item Voor een gesloten deelruimte $F$ en $x \notin F$ is er
        $f \in E'$ die verdwijnt op $F$ met $f(x) = d(x, F)$ en
        $\norm f \leq 1$.
\end{enumerate}
\end{corollary}

\begin{proof}
(1) Reëel geval: pas \cref{thm:b3:banach:hahnbanach} toe met
$p(x) = \norm f_{F'}\,\norm x$ (sublineair); de uitbreiding
voldoet $\pm\tilde f(x) = \tilde f(\pm x) \leq p(x)$, dus
$\norm{\tilde f} \leq \norm f$, en $\geq$ is restrictie.
Complex geval: zij $u = \operatorname{Re}f$, een reële
functionaal met $\abs u \leq \norm f\norm\cdot$; noteer $f(x) =
u(x) - \iu\,u(\iu x)$ (controleer op reële en imaginaire
delen: $\operatorname{Im}f(x) = -\operatorname{Re}f(\iu x)$).
Breid $u$ reëel-lineair uit met dezelfde grens, en zet $\tilde
f(x) = \tilde u(x) - \iu\tilde u(\iu x)$: $\C$-lineair
(rechtstreekse controle op vermenigvuldiging met $\iu$), breidt
$f$ uit; norm: voor gegeven $x$ schrijf $\tilde f(x) =
r\eu^{\iu\theta}$, dan $\abs{\tilde f(x)} =
\tilde f(\eu^{-\iu\theta}x) = \tilde u(\eu^{-\iu\theta}x)
\leq \norm f\,\norm x$.

(2) Op $F = Kx$ definieer $f(tx) = t\norm x$: norm $1$ op $F$;
breid uit met (1). De dualiteitsformule: $\leq$ is duidelijk,
$\geq$ door deze $f$.

(3) Op $F \oplus Kx$ definieer $f(y + tx) = t\,d(x, F)$; dan
voor $t \neq 0$, $\norm{y + tx} = \abs t\,\norm{x + y/t} \geq
\abs t\,d(x, F) = \abs{f(y + tx)}$: $\norm f \leq 1$ op de
deelruimte; breid uit met (1).
\end{proof}

\begin{remark}\label{rem:b3:banach:bidual}
Door (2) is de canonieke afbeelding $J \colon E \to E''$,
$J(x)(f) = f(x)$, een isometrie (\cref{exo:b3:banach:10}): elke
genormeerde ruimte zit in haar biduale. Ruimten met $J$
surjectief heten \emph{reflexief}\index{reflexieve ruimte}; het
weekendprobleem toont dat $\ell^p$ ($1 < p < \infty$) \omterm{rem:b3:banach:bidual}{reflexief}
is terwijl $\ell^1$ dat niet is.
\end{remark}

\section{De Baire-trilogie}

\begin{theorem}[Banach--Steinhaus, uniforme begrensdheid]
\label{thm:b3:banach:banachsteinhaus}
Zij $E$ een \emph{Banach}ruimte, $F$ genormeerd, en
$(T_i)_{i\in I} \subseteq \mathcal L(E, F)$ een familie met
$\sup_i \norm{T_ix} < \infty$ voor elke $x \in E$. Dan
$\sup_i \vertiii{T_i} < \infty$.\index{stelling van Banach--Steinhaus}
\end{theorem}

\begin{proof}
De verzamelingen $F_n = \{x : \sup_i\norm{T_ix} \leq n\}$ zijn
gesloten (doorsneden van originelen van gesloten bollen) en
overdekken $E$. Baire (\cref{thm:b3:complete:baire}) geeft $n_0$
en een bol $B(x_0, r) \subseteq F_{n_0}$. Voor $\norm z < r$:
$\norm{T_iz} \leq \norm{T_i(x_0 + z)} + \norm{T_ix_0} \leq
2n_0$, dus $\vertiii{T_i} \leq 2n_0/r$ voor elke $i$.
\end{proof}

\begin{corollary}\label{cor:b3:banach:bslimits}
Als $E$ Banach is en $T_n \in \mathcal L(E,F)$
\emph{puntsgewijs} convergeren ($T_nx \to Tx$ voor elke $x$),
dan $\sup_n \vertiii{T_n} < \infty$, $T \in \mathcal L(E, F)$,
en $\vertiii T \leq \liminf \vertiii{T_n}$.
\end{corollary}

\begin{proof}
Convergente rijen zijn begrensd: puntsgewijze begrensdheid;
Banach--Steinhaus begrenst de normen door enige $M$; dan
$\norm{Tx} = \lim\norm{T_nx} \leq M\norm x$ ($T$ is lineair als
puntsgewijze limiet), en de scherpere grens door overgang naar
$\liminf$ in $\norm{T_nx} \leq \vertiii{T_n}\norm x$.
\end{proof}

\begin{theorem}[Divergerende Fourierreeksen]
\label{thm:b3:banach:fourierdiverge}
Er bestaan continue $2\pi$-periodieke functies $f$ waarvan de
Fourierreeks in $0$ divergeert: $\sup_N\abs{S_N(f)(0)} =
\infty$. Sterker, zulke $f$ vormen een dichte deelverzameling
van $\mathcal C(S^1)$.\index{Fourierreeks, divergentie}
\end{theorem}

\begin{proof}
Werk in $E = (\mathcal C(S^1), \norm\cdot_\infty)$, een
Banachruimte, met de functionalen $\Lambda_N(f) = S_N(f)(0) =
\frac1{2\pi}\int_{-\pi}^{\pi}f(t)\,D_N(t)\,\dd t$
(Dirichletkern, Jaar~2). Elke $\Lambda_N$ is continu met
\[
\norm{\Lambda_N} = \frac1{2\pi}\int_{-\pi}^\pi\abs{D_N(t)}\,\dd
t \;=\; L_N .
\]
($\leq$ is duidelijk; $\geq$: neem $f$ continu, $\norm f_\infty
\leq 1$, benaderend $\operatorname{sign}D_N$ --- het teken
heeft eindig veel sprongen; gladstrijken van elke sprong op een
interval van lengte $\varepsilon$ verandert de integraal met
$O(N\varepsilon)$.) De \emph{Lebesgueconstanten} $L_N$ neigen
naar oneindig:
\[
L_N = \frac1{2\pi}\int_{-\pi}^{\pi}
\frac{\abs{\sin\bigl((N{+}\tfrac12)t\bigr)}}{\abs{\sin(t/2)}}
\,\dd t
\geq \frac{2}{\pi}\int_0^{\pi}
\frac{\abs{\sin\bigl((N{+}\tfrac12)t\bigr)}}{t}\,\dd t
= \frac{2}{\pi}\int_0^{(N+\frac12)\pi}\frac{\abs{\sin
u}}{u}\,\dd u
\]
(gebruik $\abs{\sin(t/2)} \leq t/2$ op $[0, \pi]$, dan
substitutie $u = (N + \tfrac12)t$). Snijdend in bogen:
\[
\int_{(k-1)\pi}^{k\pi}\frac{\abs{\sin u}}u\,\dd u
\geq \frac1{k\pi}\int_{(k-1)\pi}^{k\pi}\abs{\sin u}\,\dd u =
\frac{2}{k\pi},
\qquad\text{dus}\qquad
L_N \geq \frac{4}{\pi^2}\sum_{k=1}^N\frac1k
\xrightarrow[N\to\infty]{} \infty .
\]
Als elke continue $f$ $\sup_N \abs{\Lambda_N(f)} < \infty$ had,
zou Banach--Steinhaus $\sup_N\norm{\Lambda_N} < \infty$ forceren:
tegenspraak. Dus enige $f$ --- in feite een niet-magere, \omterm{def:b3:topology:interior}{dichte
verzameling} van $f$ (het complement van $\bigcup_M\{f:
\sup_N\abs{\Lambda_Nf}\leq M\}$, een aftelbare unie van
gesloten verzamelingen die, zonder inwendige door het bovenstaande
toegepast in elke bol, \omterm{rem:b3:complete:meagre}{mager} is) --- heeft
$\sup_N\abs{S_Nf(0)} = \infty$.
\end{proof}

\begin{theorem}[Open afbeelding]\label{thm:b3:banach:openmapping}
Zij $E, F$ Banachruimten en $T \in \mathcal L(E, F)$
\emph{surjectief}. Dan is $T$ open: $T(B_E(0,1)) \supseteq
B_F(0, c)$ voor enige $c > 0$. Bijgevolg heeft een bijectieve
begrensde operator tussen Banachruimten een begrensde
inverse.\index{open-afbeeldingsstelling}
\end{theorem}

\begin{proof}
Schrijf $B = B_E(0,1)$. Surjectiviteit geeft $F = \bigcup_n
\overline{T(nB)} = \bigcup_n n\,\overline{T(B)}$; Baire
(\cref{thm:b3:complete:baire}) geeft inwendige aan
$\overline{T(B)}$: enige $B_F(y_0, 4c) \subseteq
\overline{T(B)}$. Hercentreer in $0$: voor $\norm y < 4c$ zijn
zowel $y_0 + y$ als $y_0$ limieten van beelden $Tu_k$, $Tv_k$
met $u_k, v_k \in B$, dus $y = \lim T(u_k - v_k)$ met $u_k -
v_k \in 2B$: $B_F(0, 4c) \subseteq \overline{T(2B)}$, d.w.z.\
$B_F(0, 2c) \subseteq \overline{T(B)}$.

\emph{De \omterm{def:b3:topology:interior}{afsluiting} verwijderen} (hier komt volledigheid van $E$
binnen): zij $\norm y < c$. Kies $x_1 \in \frac12 B$ met
$\norm{y - Tx_1} < c/2$ ($B_F(0,2c) \subseteq \overline{T(B)}$
geschaald met $\frac12$); inductief $x_k \in 2^{-k}B$ met
$\norm{y - T(x_1 + \dots + x_k)} < c\,2^{-k}$. De reeks $\sum
x_k$ convergeert absoluut in de Banach $E$, naar $x \in B$ (norm
$< \sum 2^{-k} = 1$), en $Tx = y$ door continuïteit: $B_F(0, c)
\subseteq T(B)$. Openheid van $T$ op willekeurige opens volgt
door translatie en schaling; voor het corollarium betekent
openheid van $T$ dat $T^{-1}$ continu is.
\end{proof}

\begin{corollary}[Equivalente normen]\label{cor:b3:banach:equivnorms}
Als een vectorruimte \omterm{def:b3:complete:complete}{volledig} is voor twee vergelijkbare normen
($\norm\cdot_a \leq C\norm\cdot_b$), zijn de normen equivalent.
\end{corollary}

\begin{proof}
De identiteit $(E, \norm\cdot_b) \to (E, \norm\cdot_a)$ is
begrensd en bijectief tussen Banachruimten: haar inverse is
begrensd.
\end{proof}

\begin{theorem}[Gesloten grafiek]\label{thm:b3:banach:closedgraph}
Zij $E, F$ Banach en $T \colon E \to F$ lineair. Als de grafiek
$\Gamma = \{(x, Tx)\}$ gesloten is in $E \times F$ (d.w.z.\
$x_n \to x$ en $Tx_n \to y$ impliceert $y = Tx$), dan is $T$
begrensd.\index{gesloten-grafiekstelling}
\end{theorem}

\begin{proof}
$E \times F$ met $\norm{(x,y)} = \norm x + \norm y$ is Banach;
$\Gamma$, een gesloten deelruimte, is Banach. De projectie
$\pi_E\colon \Gamma \to E$ is begrensd en bijectief, dus haar
inverse $x \mapsto (x, Tx)$ is begrensd
(\cref{thm:b3:banach:openmapping}): $\norm{Tx} \leq \norm{(x,
Tx)} \leq C\norm x$.
\end{proof}

\begin{method}\label{met:b3:banach:trilogy}
Wanneer welke stelling te grijpen. \emph{Hahn--Banach}: om een
functionaal met voorgeschreven gedrag te produceren (een vector
normeren, verdwijnen op een deelruimte, uitbreiden van een
deelruimte) --- geen volledigheid nodig. \emph{Banach--Steinhaus}:
om puntsgewijze informatie om te zetten in uniforme grenzen ---
typisch om te tonen dat een limietoperatie continu is, of
(contrapositief) om \emph{divergentie} voor enig element te
bewijzen, zoals voor Fourierreeksen. \emph{Open afbeelding /
gesloten grafiek}: om continuïteit gratis te krijgen uit
algebraïsche bijectiviteit of uit een afsluitingseigenschap van
de grafiek --- typisch gebruik: twee volledige normen
vergelijken, of automatische continuïteit bewijzen. Alle drie
Baire-stellingen eisen volledigheid \emph{van de bron};
tegenvoorbeelden anders (\cref{exo:b3:banach:7}).
\end{method}

\section{Duale ruimten, concreet}

\begin{theorem}\label{thm:b3:banach:duals}
Isometrisch: $(c_0)' \cong \ell^1$ en $(\ell^1)' \cong
\ell^\infty$, via de paring $\langle x, y\rangle = \sum_n
x_ny_n$.\index{dualen van rijenruimten}
\end{theorem}

\begin{proof}
We bewijzen $(c_0)' \cong \ell^1$; de tweede identificatie is
\cref{exo:b3:banach:5}. Aan $y \in \ell^1$ associeer
$\Lambda_y(x) = \sum x_ny_n$ ($x \in c_0$): absoluut
convergent, met $\abs{\Lambda_y(x)} \leq \norm x_\infty\norm
y_1$, dus $\norm{\Lambda_y} \leq \norm y_1$. Omgekeerd zij
$\Lambda \in (c_0)'$; zet $y_n = \Lambda(e_n)$ ($e_n$ de
eenheidsrijen). Voor elke $N$, test $x^{(N)} = \sum_{n \leq N}
\operatorname{sign}(\overline{y_n})\,e_n \in c_0$ (norm $\leq
1$; in het complexe geval unimodulaire factoren $\bar
y_n/\abs{y_n}$): $\Lambda(x^{(N)}) = \sum_{n\leq N}\abs{y_n}
\leq \norm\Lambda$. Dus $y \in \ell^1$ met $\norm y_1 \leq
\norm\Lambda$. Tenslotte $\Lambda = \Lambda_y$: beide stemmen
overeen op de $e_n$, dus op eindige rijen, \omterm{def:b3:topology:interior}{dicht} in $c_0$
(afkapping: $\norm{x - \sum_{n \leq N}x_ne_n}_\infty =
\sup_{n > N}\abs{x_n} \to 0$ precies omdat $x_n \to 0$);
continue functionalen die op een \omterm{def:b3:topology:interior}{dichte verzameling}
overeenstemmen zijn gelijk. De correspondentie is lineair,
bijectief, en isometrisch ($\norm{\Lambda_y} = \norm y_1$ uit
de twee ongelijkheden).
\end{proof}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:b3:banach:1}
Bereken de \omterm{def:b3:banach:operator}{operatornormen}:
(a) de shifts $S(x_1, x_2, \dots) = (0, x_1, x_2, \dots)$ en
$S^*(x_1, x_2, \dots) = (x_2, x_3, \dots)$ op $\ell^2$;
(b) de vermenigvuldigingsoperator $M_a x = (a_nx_n)$ op
$\ell^2$, voor $a \in \ell^\infty$;
(c) de functionaal $\Lambda(f) = \int_0^{1/2}f -
\int_{1/2}^1f$ op $\mathcal C(\intcc01)$ --- toon $\norm\Lambda
= 1$ en dat de norm \emph{niet bereikt} wordt.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b3:banach:2}
Zij $E$ Banach, $T \in \mathcal L(E)$ inverteerbaar, en $S$ met
$\vertiii{S - T} < 1/\vertiii{T^{-1}}$. Toon dat $S$
inverteerbaar is en schat $\vertiii{S^{-1} - T^{-1}}$.
Toepassing: als een lineair stelsel $Tx = b$ \omterm{def:b3:groups:derived}{oplosbaar} is met
$T$ inverteerbaar, houdt een voldoende kleine perturbatie van
$T$ het uniek \omterm{def:b3:groups:derived}{oplosbaar}, met een kwantitatieve grens op de
verandering van de oplossing.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:banach:3}
(a) Bewijs dat $\ell^1$, $\ell^\infty$ en $c_0$ Banachruimten
zijn, en dat $c_0$ de \omterm{def:b3:topology:interior}{afsluiting} in $\ell^\infty$ is van de
ruimte van eindige rijen.
(b) Toon $\ell^p \subseteq \ell^q$ met $\norm\cdot_q \leq
\norm\cdot_p$ voor $1 \leq p \leq q \leq \infty$, en dat de
inclusie strikt is.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:banach:4}
Zij $F \subseteq E$ een gesloten deelruimte en $x \notin F$.
Met \cref{cor:b3:banach:hbnormed}, bewijs de dualiteitsformule
\[
d(x, F) = \max\bigl\{\abs{f(x)} : f \in E',\ \norm f \leq 1,\
f\restriction_F = 0\bigr\}
\]
(noteer: een \emph{maximum}). Leid af dat $F =
\bigcap\{\ker f : f \in E',\ f\restriction_F = 0\}$: gesloten
deelruimten zijn precies de doorsneden van kernen van
functionalen.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:banach:5}
Bewijs $(\ell^1)' \cong \ell^\infty$ isometrisch, volgens het
schema van \cref{thm:b3:banach:duals} (eindige rijen zijn \omterm{def:b3:topology:interior}{dicht}
in $\ell^1$).
Waar breekt het argument voor $(\ell^\infty)'$?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:banach:6}
Zij $E, F, G$ genormeerd met $E$ Banach, en $B \colon E \times
F \to G$ bilineair, continu in elke variabele afzonderlijk. Toon
dat $B$ (gezamenlijk) continu is: $\norm{B(x,y)} \leq C\norm
x\norm y$. \emph{(Pas Banach--Steinhaus toe op de familie
$(B(\cdot, y))_{\norm y \leq 1}$.)}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:banach:7}
(a) Op $E = \mathcal C(\intcc01)$, vergelijk
$\norm\cdot_\infty$ en $\norm\cdot_1$: de identiteit $(E,
\norm\cdot_\infty) \to (E, \norm\cdot_1)$ is begrensd en
bijectief maar haar inverse is onbegrensd. Welke hypothese van
\cref{cor:b3:banach:equivnorms} faalt?
(b) Vertoon een discontinue lineaire afbeelding van een dichte
deelruimte van $\ell^2$ naar $K$ (bijv.\ op eindige rijen), en
leg uit waarom dit de gesloten-grafiekstelling niet tegenspreekt.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b3:banach:8}
(Hellinger--Toeplitz) Zij $T \colon \ell^2 \to \ell^2$ lineair
(overal gedefinieerd) en \emph{symmetrisch}: $\langle Tx,
y\rangle = \langle x, Ty\rangle$ voor alle $x, y$, waarbij
$\langle x, y \rangle = \sum x_n\bar y_n$. Toon dat $T$
begrensd is.
\emph{(Gesloten grafiek: als $x_k \to x$ en $Tx_k \to z$, test
tegen willekeurige $y$.)} Moraal: onbegrensde symmetrische
operatoren --- de Hamiltonianen van de kwantummechanica ---
kunnen nooit op de hele ruimte gedefinieerd zijn.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b3:banach:9}
(Stelling van Pólya over kwadratuur) Voor elke $n$, zij
$\Lambda_n(f) = \sum_{i=0}^{n} w_{i,n}f(x_{i,n})$ een
kwadratuurregel op $\mathcal C(\intcc01)$ ($x_{i,n} \in
\intcc01$, $w_{i,n} \in \R$). Toon dat $\Lambda_n(f) \to
\int_0^1f$ voor \emph{elke} continue $f$ desda: (i)
$\Lambda_n(P) \to \int_0^1P$ voor elk polynoom $P$, en (ii)
$\sup_n\sum_i\abs{w_{i,n}} < \infty$. \emph{(Bereken
$\norm{\Lambda_n}$; gebruik Banach--Steinhaus en Weierstrass.)}
Controleer dat regels met \emph{positieve} gewichten exact op
constanten (ii) automatisch vervullen.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:banach:10}
Toon dat $J \colon E \to E''$, $J(x)(f) = f(x)$, een lineaire
isometrie is (gebruik \cref{cor:b3:banach:hbnormed}(2)), en dat
zij surjectief is wanneer $\dim E < \infty$. Toon ook dat als
$E'$ \omterm{def:b3:galois:separable}{separabel} is, $E$ dat ook is. \emph{(Kies $x_n$ die een
dichte rij van $E'$ bijna normeren en toon dat hun gesloten
opsanning $E$ is, via \cref{cor:b3:banach:hbnormed}(3).)}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:banach:11}
(Quotiëntruimten) Zij $E$ een Banachruimte en $F \subseteq E$
een \emph{gesloten} deelruimte. Op $E/F$ definieer
\[
\norm{\bar x} \;=\; d(x, F) = \inf_{y\in F}\norm{x - y} .
\]
(a) Toon dat dit een welgedefinieerde \emph{norm} is op $E/F$
(waar komt geslotenheid van $F$ binnen?), en dat de projectie
$\pi \colon E \to E/F$ $\vertiii\pi \leq 1$ heeft en de open
eenheidsbol op de open eenheidsbol afbeeldt.
(b) Toon dat $E/F$ \omterm{def:b3:complete:complete}{volledig} is. \emph{(Gebruik het
reeks-criterium van \cref{exo:b3:complete:1}(b): gegeven klassen
$\bar x_k$ met $\sum\norm{\bar x_k} < \infty$, til elke op naar
$x_k \in E$ met $\norm{x_k} \leq \norm{\bar x_k} + 2^{-k}$ en
sommeer in $E$.)}
(c) Bereken: voor $E = c$ (convergente rijen) en $F = c_0$,
toon $c/c_0 \cong K$ isometrisch via $\bar x \mapsto
\lim_nx_n$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:banach:12}
(Begrensde projecties en gecomplementeerde deelruimten) Zij $E$
een Banachruimte en $P \colon E \to E$ lineair met $P^2 = P$
(een algebraïsche projectie), $V = \operatorname{im}P$, $W =
\ker P$.
(a) Veronderstel $P$ begrensd. Toon dat $V$ en $W$ gesloten zijn
en $E = V \oplus W$ met de ontbinding $x = Px + (x - Px)$.
(b) Omgekeerd, veronderstel $E = V \oplus W$ met \emph{beide}
$V, W$ gesloten, en zij $P$ de projectie op $V$ langs $W$. Toon
dat $P$ begrensd is. \emph{(Gesloten grafiek: als $x_n \to x$
en $Px_n \to z$, dan $z \in V$, $x_n - Px_n \to x - z \in W$,
en uniciteit van de ontbinding identificeert $z = Px$.)}
(c) Leid de \emph{equivalentie} af: een deelruimte $V$ laat een
begrensde projectie toe desda zij gesloten is en een gesloten
\omterm{def:b3:galois:algebraic}{algebraïsch} complement heeft --- en noteer (zonder bewijs) dat
gesloten deelruimten zonder deze eigenschap bestaan ($c_0$
binnen $\ell^\infty$ is het klassieke voorbeeld):
Hilbertruimten, waar $V^\perp$ altijd werkt
(\cref{ch:b3:hilbert}), zijn de uitzondering, niet de regel.
\end{exercise}

\section{Probleem: de dualiteit van de
\texorpdfstring{$\ell^p$}{lp}-ruimten}

\begin{problem}[{Weekendprobleem --- $(\ell^p)' = \ell^q$,
reflexiviteit, en de vreemdheid van $\ell^\infty$}]
\label{pb:b3:banach:1}
Fixeer $1 < p < \infty$ en zij $q$ de geconjugeerde exponent,
$\frac1p + \frac1q = 1$. De paring doorheen is $\langle x,
y\rangle = \sum_n x_ny_n$.

\medskip
\noindent\textbf{Deel I --- Hölder en Minkowski voor rijen.}
\begin{enumerate}
  \item (Ongelijkheid van Young) Voor $a, b \geq 0$ toon $ab
        \leq \frac{a^p}p + \frac{b^q}q$, met concaafheid van
        $\log$ of door $t \mapsto \frac{t^p}p + \frac1q - t$ te
        bestuderen.
  \item (Hölder) Leid af: $\abs{\langle x, y\rangle} \leq
        \norm x_p\norm y_q$ voor $x \in \ell^p$, $y \in
        \ell^q$; identificeer het gelijkheidsgeval.
  \item (Minkowski) Leid de driehoeksongelijkheid voor
        $\norm\cdot_p$ af. \emph{(Schrijf $\abs{x_n + y_n}^p
        \leq \abs{x_n}\,\abs{x_n{+}y_n}^{p-1} +
        \abs{y_n}\,\abs{x_n{+}y_n}^{p-1}$ en pas Hölder toe op
        elke term.)}
  \item Bewijs dat $\ell^p$ \omterm{def:b3:complete:complete}{volledig} is en dat de eindige rijen
        er \omterm{def:b3:topology:interior}{dicht} in liggen.
\end{enumerate}

\medskip
\noindent\textbf{Deel II --- De dualiteit $(\ell^p)' =
\ell^q$.}
\begin{enumerate}[resume]
  \item Voor $y \in \ell^q$, toon dat $\Lambda_y(x) = \langle
        x, y\rangle$ $\Lambda_y \in (\ell^p)'$ definieert met
        $\norm{\Lambda_y} \leq \norm y_q$, en, testend op $x_n
        = \abs{y_n}^{q-1}\operatorname{sign}(y_n)$ (passend
        afgekapt en genormaliseerd), dat $\norm{\Lambda_y} =
        \norm y_q$.
  \item Omgekeerd, gegeven $\Lambda \in (\ell^p)'$, zet $y_n =
        \Lambda(e_n)$; toon $y \in \ell^q$ met $\norm y_q \leq
        \norm\Lambda$ \emph{(test op afkappingen zoals in vraag
        5 en laat de afkappingslengte groeien)}, en concludeer
        $\Lambda = \Lambda_y$: de afbeelding $y \mapsto
        \Lambda_y$ is een isometrisch isomorfisme $\ell^q \to
        (\ell^p)'$.
  \item Leid af dat $\ell^p$ \emph{\omterm{rem:b3:banach:bidual}{reflexief}} is voor $1 < p <
        \infty$: de twee dualiteiten samenstellend, komt elk
        element van $(\ell^p)''$ van $\ell^p$; verifieer zorgvuldig
        dat de samenstelling de canonieke $J$ is.
\end{enumerate}

\medskip
\noindent\textbf{Deel III --- $\ell^1$ en $\ell^\infty$ zijn
andere dieren.}
\begin{enumerate}[resume]
  \item Toon dat $\ell^p$ ($1 \leq p < \infty$) en $c_0$
        \omterm{def:b3:galois:separable}{separabel} zijn, maar $\ell^\infty$ niet.
        \emph{(De overaftelbaar vele indicatorrijen van
        deelverzamelingen van $\N$ liggen paarsgewijs op
        afstand $1$.)}
  \item Leid uit \cref{exo:b3:banach:10} af dat $(\ell^1)'
        \cong \ell^\infty$ maar $(\ell^\infty)' \not\cong
        \ell^1$: $\ell^1$ is \emph{niet} \omterm{rem:b3:banach:bidual}{reflexief}.
        \emph{(Als $(\ell^\infty)'$ $\ell^1$ was, zou zij
        \omterm{def:b3:galois:separable}{separabel} zijn, wat $\ell^\infty$ \omterm{def:b3:galois:separable}{separabel} forceert.)}
  \item (Een Banachlimiet, expliciet) Op $\ell^\infty_\R$, zij
        $p(x) = \limsup_n \frac{x_1 + \dots + x_n}{n}$. Toon
        dat $p$ sublineair is, en dat op de deelruimte $c$ van
        convergente rijen, $\mathrm{LIM}(x) = \lim x$ voldoet
        aan $\mathrm{LIM} \leq p$. Breid uit met Hahn--Banach
        tot $\mathrm{LIM} \colon \ell^\infty_\R \to \R$ en
        toon: $\mathrm{LIM}$ is positief ($x \geq 0 \Rightarrow
        \mathrm{LIM}(x) \geq 0$), shift-invariant
        ($\mathrm{LIM}(x_2, x_3, \dots) = \mathrm{LIM}(x)$),
        breidt de limiet uit, en voldoet $\liminf x \leq
        \mathrm{LIM}(x)\leq \limsup x$.
  \item Toon dat zulk $\mathrm{LIM}$, bekeken in
        $(\ell^\infty)'$, \emph{niet} van de vorm $\Lambda_y$
        is voor enige $y \in \ell^1$; concludeer opnieuw
        $(\ell^\infty)' \neq \ell^1$. \emph{(Evalueer op de
        eenheidsrijen $e_n$, dan op de constante rij $1$.)}
  \item Evalueer $\mathrm{LIM}$ op $(0,1,0,1,\dots)$, en toon
        dat geen shift-invariante \emph{multiplicatieve}
        uitbreiding van de limiet kan bestaan \emph{(beschouw
        $x = (0,1,0,1,\dots)$ en $x\cdot Sx$ waarbij $S$ de
        shift is)}.
\end{enumerate}

\medskip
\noindent\textbf{Deel IV --- Epiloog: waarom reflexiviteit
telt.}
\begin{enumerate}[resume]
  \item Met \cref{cor:b3:banach:bslimits} en vraag 6, toon dat
        elke begrensde rij van $\ell^p$ ($1 < p < \infty$) een
        deelrij $(x^{(k)})$ heeft die \emph{zwak} convergeert:
        $\Lambda(x^{(k)})$ convergeert voor elke $\Lambda \in
        (\ell^p)'$. \emph{(Diagonaalextractie op de aftelbaar
        vele coördinaten; identificeer de zwakke limiet in
        $\ell^p$ met uniforme begrensdheid van normen en
        Hölder.)} Toon met voorbeeld ($e_n$ in $\ell^1$,
        tegen welgekozen elementen van $\ell^\infty$) dat dit
        faalt in $\ell^1$: zwakke compactheid is een privilege
        van \omterm{rem:b3:banach:bidual}{reflexieve ruimten}.
\end{enumerate}

\medskip
\noindent\textbf{Deel V --- De zwakke \omterm{def:b3:topology:topology}{topologie} aan het werk, en
de verrassing van Schur.} Schrijf $x^{(k)} \rightharpoonup x$
in een genormeerde ruimte $E$ (\emph{zwakke convergentie})
wanneer $\Lambda(x^{(k)}) \to \Lambda(x)$ voor elke $\Lambda
\in E'$.
\begin{enumerate}[resume]
  \item Voltooi de census: toon $(c_0)' \cong \ell^1$
        isometrisch, volgens het schema van vragen 5--6 (wat
        vervangt de testrijen?). Verzamel de keten $c_0 \to
        \ell^1 \to \ell^\infty \to \dots$ van opeenvolgende
        \omterm{def:b3:banach:operator}{dualen} en markeer waar reflexiviteit faalt.
  \item Toon dat elke zwak convergente rij van een Banachruimte
        begrensd is: bekijk de $x^{(k)}$ via de canonieke
        embedding $J$ als functionalen op $E'$ en pas
        Banach--Steinhaus toe
        (\cref{thm:b3:banach:banachsteinhaus}) --- op welke
        Banachruimte, en waarom is volledigheid daar
        beschikbaar?
  \item Toon dat in $\ell^p$, $1 < p < \infty$: $x^{(k)}
        \rightharpoonup x$ desda $\sup_k\norm{x^{(k)}}_p <
        \infty$ en $x^{(k)}_n \to x_n$ voor elke coördinaat $n$
        \emph{(één richting gebruikt Banach--Steinhaus via de
        canonieke embedding; voor de andere, benader $y \in
        \ell^q$ door eindige rijen)}. Leid af $e_k
        \rightharpoonup 0$ in $\ell^2$ terwijl $\norm{e_k}_2 =
        1$: zwakke limieten kunnen massa verliezen.
  \item Toon dat de norm zwak onderhalfcontinu is: $x^{(k)}
        \rightharpoonup x$ impliceert $\norm x \leq
        \liminf_k\,\norm{x^{(k)}}$ \emph{(kies een
        normerende functionaal voor $x$,
        \cref{cor:b3:banach:hbnormed})}.
  \item (Radon--Riesz in $\ell^2$) Toon dat in $\ell^2$ zwakke
        convergentie samen met convergentie van normen
        normconvergentie impliceert \emph{(ontwikkel
        $\norm{x^{(k)} - x}_2^2$)}. Geef een tegenvoorbeeld
        voor dezelfde uitspraak zonder de normhypothese.
  \item (Schur, stap 1) Zij $x^{(k)} \rightharpoonup 0$ in
        $\ell^1$ en veronderstel, voor tegenspraak,
        $\norm{x^{(k)}}_1 \geq \delta > 0$ langs een deelrij.
        Toon eerst dat $x^{(k)}_n \to 0$ voor elke $n$ (welke
        functionalen?), en construeer recursief indices $k_1 <
        k_2 < \cdots$ en gehele $0 = N_0 < N_1 < N_2 < \cdots$
        zodat de massa van $x^{(k_j)}$ zich concentreert op het
        blok $B_j = \intoc{N_{j-1}}{N_j}$:
        \[
        \sum_{n \in B_j}\bigl|x^{(k_j)}_n\bigr| \geq
        \norm{x^{(k_j)}}_1 - \frac\delta{10} .
        \]
  \item (Schur, stap 2) Definieer $y \in \ell^\infty$ door
        $y_n = \operatorname{sign}\bigl(x^{(k_j)}_n\bigr)$
        voor $n \in B_j$. Toon $\langle x^{(k_j)}, y\rangle
        \geq \norm{x^{(k_j)}}_1 - \frac{2\delta}{10} \geq
        \frac{8\delta}{10}$ en leid een tegenspraak af met
        $x^{(k)} \rightharpoonup 0$. Concludeer de
        \emph{stelling van Schur}: in $\ell^1$ convergeren
        zwak convergente rijen in norm.
  \item Leid af dat $(e_k)$ geen zwak convergente deelrij in
        $\ell^1$ heeft (haar enige kandidaatlimiet is $0$,
        coördinaatsgewijs --- pas dan Schur toe), en herwin de
        mislukking van zwakke compactheid van vraag 13; en
        los de schijnbare paradox op: in $\ell^1$ vallen zwakke
        en normconvergentie van \emph{rijen} samen, toch
        verschillen de zwakke en norm-\emph{topologieën} en
        falen begrensde verzamelingen nog steeds om zwak
        sequentieel compact te zijn --- geen tegenspraak, alleen
        het falen van reflexiviteit.
  \item (Synthesetabel) Voor $E \in \{c_0,\ \ell^1,\ \ell^p\
        (1{<}p{<}\infty),\ \ell^\infty\}$, tabuleer: de \omterm{def:b3:banach:operator}{duale};
        separabiliteit; reflexiviteit; of begrensde rijen zwak
        convergente deelrijen toelaten; en één
        handtekeningseigenschap van elke ruimte, gerechtvaardigd
        in één regel uit dit probleem.
\end{enumerate}

\medskip
\noindent\textbf{Deel VI --- Complementen: dichtstbijzijnde
punten, gemiddelde convergentie, de waarde van een Banachlimiet.}
\begin{enumerate}[resume]
  \item (Dichtstbijzijnde punten: een dividend van
        reflexiviteit) Zij $F$ een gesloten deelruimte van
        $\ell^p$ ($1 < p < \infty$) en $x \in \ell^p$. Toon
        dat $d = \operatorname{dist}(x, F)$ \emph{bereikt}
        wordt: extraheer uit een minimaliserende rij een zwak
        convergente deelrij (vraag 13), houd de zwakke limiet
        binnen $F$ door, via Hahn--Banach, een functionaal te
        bouwen die op $F$ verdwijnt maar niet in een punt
        erbuiten, en concludeer met vraag 17. Toon dan dat het
        privilege niet universeel is: in $c_0$, voor $\Lambda(x)
        = \sum_n2^{-n}x_n$, bewijs dat $\norm\Lambda = 1$ niet
        bereikt wordt op de eenheidsbol, stel de
        afstandsformule $\operatorname{dist}(x, \ker\Lambda) =
        \abs{\Lambda(x)}$ vast, en leid af dat geen $x \notin
        \ker\Lambda$ een dichtstbijzijnd punt in het gesloten
        hypervlak $\ker\Lambda$ heeft.
  \item (Banach--Saks in $\ell^2$) Zij $x^{(k)}
        \rightharpoonup 0$ in $\ell^2_{\R}$ met
        $\norm{x^{(k)}}_2 \leq C$. Construeer een deelrij
        $(y_j)$ met $\abs{\langle y_i, y_j\rangle} \leq
        \frac1j$ voor alle $i < j$, en leid af
        \[
        \Bigl\lVert\frac{y_1 + \dots + y_m}m\Bigr\rVert_2^2
        \leq \frac{C^2 + 2}m \longrightarrow 0 :
        \]
        na extractie convergeren de Cesàro-middelen in
        \emph{norm}. Controleer op $(e_k)$, wiens middelen
        norm $\frac1{\sqrt m}$ hebben: zwakke convergentie,
        nutteloos voor de rij zelf (vraag 16), wordt
        normconvergentie voor gemiddelden.
  \item (De waarde van een Banachlimiet) Zij $L$ enige
        Banachlimiet (vraag 10) en $A_mx = \frac1m(x + Sx +
        \dots + S^{m-1}x)$. Toon $L(A_mx) = L(x)$ en $\liminf
        A_mx \leq L(x) \leq \limsup A_mx$; leid af dat
        \emph{alle} Banachlimieten overeenstemmen op periodieke
        rijen, met waarde het gemiddelde over een periode ---
        $\frac13$ op $(1, 0, 0, 1, 0, 0, \dots)$, consistent
        met $\frac12$ van vraag 12. Toon dan dat
        overeenstemming in het algemeen faalt: voor de
        blokrij $x$ gelijk aan $1$ op $\intoc{3^{j-1}}{3^j}$
        voor even $j$ en $0$ elders, toon dat de Cesàro-
        middelen oscilleren tussen $\leq \frac13$ en $\geq
        \frac23$, en bouw twee Banachlimieten $L_\pm$ met
        $L_-(x) \leq \frac13 < \frac23 \leq L_+(x)$
        \emph{(breid uit van $c \oplus \R x$ met de extreme
        toelaatbare waarden $\pm$: controleer dat $\Lambda(y +
        tx) = \lim y + t\,p(x)$ gedomineerd wordt door de
        sublineaire $p$ van vraag 10)}.
\end{enumerate}
\end{problem}
