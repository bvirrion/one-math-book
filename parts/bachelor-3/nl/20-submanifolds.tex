\chapter{DeelvariĂ«teiten van \texorpdfstring{$\R^n$}{Rn}}
\label{ch:b3:submanifolds}

Bollen, tori, rotatiegroepen: de natuurlijke habitats van
meetkunde en mechanica zijn geen vectorruimten maar
\emph{gekromde verzamelingen die van dichtbij vlak lijken}. Dit
hoofdstuk geeft die zin een precieze betekenis ---
deelvariĂ«teiten van $\R^n$ --- en de calculus om erop te
werken. De grondslag is de stelling van de inverse functie,
hier bewezen via het Banach-vastpunt; al het overige is
coĂ¶rdinatenwisseling: de vier equivalente beschrijvingen van
een deelvariĂ«teit (lokale rechttrekking, niveaucurven,
grafieken, parametriseringen), \omterm{def:b3:submanifolds:tangent}{raakruimten}, en geoptimaliseerd
onder nevenvoorwaarden via \omterm{thm:b3:submanifolds:lagrange}{multiplicatoren van Lagrange} ---
die, als afscheidende demonstratie, de spectraalstelling voor
symmetrische matrices in drie regels meetkunde herbewezen. Het
weekendprobleem bouwt de rotatiegroep $SO(3)$ en haar
quaternionische dubbeldekking: algebra ($Q_8$ van
\cref{ch:b3:groups}, volwassen) ontmoet meetkunde.

\section{De stelling van de inverse functie}

\begin{theorem}[Stelling van de inverse functie]
\label{thm:b3:submanifolds:ift}
Zij $U \subseteq \R^n$ open, $f \colon U \to \R^n$ van klasse
$\mathcal C^1$, en $a \in U$ met $Df(a)$ inverteerbaar. Dan
bestaan \omterm{def:b3:topology:topology}{open verzamelingen} $V \ni a$, $W \ni f(a)$ zodat $f
\colon V \to W$ een bijectie is met $\mathcal C^1$-inverse,
en\index{stelling van de inverse functie}
\[
D(f^{-1})(y) = \bigl(Df(f^{-1}(y))\bigr)^{-1}
\qquad (y \in W).
\]
Als $f$ van klasse $\mathcal C^k$ is, zo ook $f^{-1}$.
\end{theorem}

\begin{proof}
Normaliseer: vervang $f$ door $x \mapsto
Df(a)^{-1}\bigl(f(a + x) - f(a)\bigr)$, zodat we $a = 0$,
$f(0) = 0$, $Df(0) = I$ mogen aannemen (de algemene uitspraak
volgt door samenstellen met de affiene bijecties). Schrijf
$f(x) = x + g(x)$: $Dg(0) = 0$, en kies door continuĂŻteit van
$Dg$ een $r > 0$ met $\vertiii{Dg(x)} \leq \frac12$ op $\bar
B(0, r)$; de middelwaardeongelijkheid geeft $\norm{g(x) -
g(x')} \leq \frac12 \norm{x - x'}$ daar.

\emph{Bijectiviteit op een \omterm{def:b3:topology:topology}{omgeving}.} Voor $y \in B(0,
\frac r2)$ betekent $f(x) = y$ oplossen het vinden van een
vast punt van $\Phi_y(x) = y - g(x)$; $\Phi_y$ beeldt $\bar
B(0, r)$ in zichzelf af ($\norm{\Phi_y(x)} \leq \norm y +
\frac12\norm x \leq r$) en is $\frac12$-Lipschitz: Banach
(\cref{thm:b3:complete:banach}) levert een unieke oplossing
$x = \varphi(y) \in \bar B(0,r)$. Bovendien is $f$ injectief
op $\bar B(0,r)$:
\[
\norm{f(x) - f(x')} \geq \norm{x - x'} - \norm{g(x) - g(x')}
\geq \tfrac12\norm{x - x'} .
\tag{$*$}
\]
Stel $W = B(0, \frac r2)$ en $V = f^{-1}(W)\cap B(0, r)$:
open (continuĂŻteit), met $f \colon V \to W$ bijectief.

\emph{ContinuĂŻteit en differentieerbaarheid van de inverse.}
$(*)$ zegt dat $\varphi = f^{-1}$ $2$-Lipschitz is. Fixeer
$y_0 = f(x_0) \in W$; inverteerbaarheid van $A = Df(x_0)$
(haar afstand tot $I$ is $\leq \frac12$: Neumann,
\cref{prop:b3:banach:neumann}) en differentieerbaarheid van
$f$ geven, voor $y = f(x)$ nabij $y_0$:
\[
\varphi(y) - \varphi(y_0) - A^{-1}(y - y_0)
= -A^{-1}\bigl(f(x) - f(x_0) - A(x - x_0)\bigr)
= -A^{-1}\,o(\norm{x - x_0}) = o(\norm{y - y_0}),
\]
met $(*)$ om $\norm{x - x_0} \leq 2\norm{y - y_0}$ om te
zetten: $\varphi$ is differentieerbaar in $y_0$ met de
inverse differentiaal. ContinuĂŻteit van $y \mapsto
D\varphi(y) = Df(\varphi(y))^{-1}$: samenstelling van
\omterm{def:b3:topology:continuity}{continue afbeeldingen} (inversie is continu,
\cref{prop:b3:banach:neumann}): $\varphi \in \mathcal C^1$;
bootstrappen van dezelfde formule geeft $\mathcal C^k$.
\end{proof}

\begin{theorem}[Stelling van de impliciete functie]
\label{thm:b3:submanifolds:implicit}
Zij $F \colon U \subseteq \R^p\times\R^q \to \R^q$ van
klasse $\mathcal C^1$ nabij $(a, b)$, $F(a,b) = 0$, en
veronderstel dat de partiĂ«le differentiaal $D_yF(a,b) \in
\mathcal L(\R^q)$ inverteerbaar is. Dan bestaan \omterm{def:b3:topology:topology}{omgevingen}
$A \ni a$, $B \ni b$ en een $\mathcal C^1$-afbeelding $\psi
\colon A \to B$ met
\[
\bigl\{(x, y)\in A\times B : F(x,y) = 0\bigr\}
= \{(x, \psi(x)) : x \in A\},
\]
en $D\psi(x) = -D_yF(x, \psi(x))^{-1}\,D_xF(x,
\psi(x))$.\index{stelling van de impliciete functie}
\end{theorem}

\begin{proof}
Pas \cref{thm:b3:submanifolds:ift} toe op $\Theta(x, y) =
(x, F(x,y))$: haar differentiaal in $(a,b)$,
blok-driehoekig met inverteerbare diagonale blokken $I$ en
$D_yF$, is inverteerbaar. De lokale inverse heeft de vorm
$\Theta^{-1}(x, z) = (x, h(x, z))$; stel $\psi(x) = h(x,
0)$: dan $F(x, y) = 0$ desda $\Theta(x,y) = (x, 0)$ desda $y
= \psi(x)$, lokaal. De formule: differentieer $F(x,
\psi(x)) = 0$ met de kettingregel.
\end{proof}

\section{DeelvariĂ«teiten: vier definities}

\begin{theorem}[Equivalente karakteriseringen]
\label{thm:b3:submanifolds:characterizations}
Zij $M \subseteq \R^n$, $d \in \{0, \dots, n\}$, en $k \geq
1$. De volgende zijn equivalent, voor elk punt $a \in M$
(en $M$ is een \emph{$d$-dimensionale deelvariĂ«teit van
klasse $\mathcal C^k$}\index{deelvariĂ«teit} als zij in elke
$a \in M$ gelden):
\begin{enumerate}
  \item (\emph{Rechttrekking}) Er bestaat een $\mathcal
        C^k$-diffeomorfisme $\Phi$ van een open $\Omega \ni
        a$ op een open $\Omega' \subseteq \R^n$ met
        \[
        \Phi(M\cap\Omega) = \Omega' \cap
        \bigl(\R^d\times\{0\}\bigr).
        \]
  \item (\emph{Niveaucurve}) Er bestaat een $\mathcal
        C^k$-\emph{submersie} $F \colon \Omega \to
        \R^{n-d}$ (d.w.z.\ $DF(x)$ surjectief) op een open
        $\Omega \ni a$ met $M\cap\Omega = F^{-1}(0)$.
  \item (\emph{Grafiek}) Op permutatie van coĂ¶rdinaten na
        is $M$ lokaal de grafiek van een $\mathcal
        C^k$-afbeelding $\psi \colon A \subseteq \R^d \to
        \R^{n-d}$.
  \item (\emph{Parametrisering}) Er bestaat een $\mathcal
        C^k$-\emph{immersie} $\varphi \colon A \subseteq
        \R^d \to \R^n$ ($D\varphi(u)$ injectief) met $A$
        open, $\varphi$ een \omterm{def:b3:topology:continuity}{homeomorfisme} van $A$ op $M
        \cap \Omega$ voor enige open $\Omega \ni a$.
\end{enumerate}
\end{theorem}

\begin{proof}
(1)$\Rightarrow$(2): $F = (\Phi_{d+1}, \dots, \Phi_n)$
(laatste coĂ¶rdinaten van $\Phi$): een submersie ($D\Phi$
inverteerbaar).
(2)$\Rightarrow$(3): $DF(a)$ surjectief: enige $q \times
q$-minor van de Jacobiaan is inverteerbaar ($q = n - d$);
na permutatie van coĂ¶rdinaten is $D_yF(a)$ inverteerbaar, en
de stelling van de impliciete functie
(\cref{thm:b3:submanifolds:implicit}) drukt $M$ lokaal uit
als grafiek $y = \psi(x)$.
(3)$\Rightarrow$(4): $\varphi(x) = (x, \psi(x))$: een
immersie (differentiaal
$\bigl(\begin{smallmatrix}I\\ D\psi\end{smallmatrix}\bigr)$
injectief), een \omterm{def:b3:topology:continuity}{homeomorfisme} op de grafiek (inverse: de
projectie, continu).
(4)$\Rightarrow$(1): zij $\varphi(u_0) = a$; vul
$\operatorname{im}D\varphi(u_0)$ aan met een supplement $E$
($\dim E = n - d$) en definieer $\Theta(u, v) = \varphi(u) +
v$ op $A\times E$: $D\Theta(u_0, 0)$ is bijectief (beeld
bevat $\operatorname{im}D\varphi(u_0)$ en $E$), dus is
$\Theta$ een lokaal diffeomorfisme
(\cref{thm:b3:submanifolds:ift}); haar inverse $\Phi$
trekt recht: nabij $a$ zijn punten van $M$ precies de
$\varphi(u) = \Theta(u, 0)$ --- hiervoor garandeert de
homeomorfisme-hypothese in (4) dat $M\cap\Omega$, voor kleine
$\Omega$, geen andere bladen bevat ($\varphi(u') + v = m \in
M$ \omterm{def:b3:topology:interior}{dicht} bij $a$ met kleine $v \neq 0$ moet worden
uitgesloten: $m = \varphi(u'')$ voor enige $u''$ nabij $u_0$
door de homeomorfisme-eigenschap, en lokale injectiviteit van
$\Theta$ dwingt $v = 0$). Dan $\Phi(M\cap\Omega) =
(A\times\{0\}) \cap\Phi(\Omega)$ na inkrimpen.
\end{proof}

\begin{example}\label{ex:b3:submanifolds:examples}
De sfeer $S^{n-1} = \{\norm x^2 = 1\}$: niveaucurve van de
submersie $F(x) = \norm x_2^2 - 1$ op
$\R^n\setminus\{0\}$ ($DF(x) = 2x^{\mathsf T} \ne 0$): een
$\mathcal C^\infty$-deelvariĂ«teit van dimensie $n - 1$. De
torus in $\R^3$: niveaucurve van $\bigl(\sqrt{x^2 + y^2} -
R\bigr)^2 + z^2 - r^2$ ($0 < r < R$). De kegel $\{x^2 + y^2
= z^2\}$ is \emph{geen} deelvariĂ«teit in $0$
(\cref{exo:b3:submanifolds:1}). Matrixgroepen: $SL_n$ en
$O_n$ zijn deelvariĂ«teiten van $M_n(\R)$
(\cref{exo:b3:submanifolds:5,exo:b3:submanifolds:6}) --- het
startpunt van de Lie-theorie.
\end{example}

\section{Raakruimten}

\begin{definition}\label{def:b3:submanifolds:tangent}
Zij $M$ een $d$-deelvariĂ«teit en $a \in M$. De
\emph{raakruimte}\index{raakruimte} $T_aM$ is de verzameling
snelheidsvectoren $\gamma'(0)$ van $\mathcal C^1$-krommen
$\gamma \colon \intoo{-\varepsilon}\varepsilon \to M$ met
$\gamma(0) = a$.
\end{definition}

\begin{proposition}\label{prop:b3:submanifolds:tangent}
$T_aM$ is een $d$-dimensionale vectordeelruimte van $\R^n$,
en:
\begin{enumerate}
  \item als $M = F^{-1}(0)$ lokaal met $F$ een submersie:
        $T_aM = \ker DF(a)$;
  \item als $M$ geparametriseerd wordt door de immersie
        $\varphi$ ($\varphi(u_0) = a$): $T_aM =
        \operatorname{im}D\varphi(u_0)$.
\end{enumerate}
\end{proposition}

\begin{proof}
Krommen in $M$ voldoen aan $F(\gamma(t)) = 0$; de
kettingregel in $0$ geeft $DF(a)\gamma'(0) = 0$: $T_aM
\subseteq \ker DF(a)$. Omgekeerd transporteert rechttrekking
(\cref{thm:b3:submanifolds:characterizations}(1)) lijnen van
$\R^d\times\{0\}$ naar krommen van $M$: elke vector van een
$d$-dimensionale deelruimte wordt gerealiseerd; dimensies
vergelijken ($\dim\ker DF(a) = n - (n - d) = d$) dwingt
gelijkheid in (1), en hetzelfde transportargument geeft (2)
($D\varphi(u_0)$ toegepast op rechte lijnen in $A$;
dimensies opnieuw).
\end{proof}

\begin{theorem}[Multiplicatoren van Lagrange]
\label{thm:b3:submanifolds:lagrange}
Zij $M = F^{-1}(0)$ met $F = (F_1, \dots, F_q) \colon
\Omega \to \R^q$ een $\mathcal C^1$-submersie, en zij $f
\colon \Omega \to \R$ van klasse $\mathcal C^1$. Als de
restrictie $f\restriction_M$ een lokaal extremum heeft in $a
\in M$, dan bestaan unieke reĂ«le $\lambda_1, \dots,
\lambda_q$ (\emph{\omterm{thm:b3:submanifolds:lagrange}{multiplicatoren van Lagrange}})
met\index{multiplicatoren van Lagrange}
\[
\nabla f(a) = \lambda_1\nabla F_1(a) + \dots +
\lambda_q\nabla F_q(a) .
\]
\end{theorem}

\begin{proof}
Voor elke kromme $\gamma$ in $M$ door $a$: $t \mapsto
f(\gamma(t))$ heeft een lokaal extremum in $0$, dus $0 =
\frac{\dd}{\dd t}f(\gamma(t))\big|_0 = \langle\nabla f(a),
\gamma'(0)\rangle$: $\nabla f(a) \perp T_aM = \ker DF(a)$
(\cref{prop:b3:submanifolds:tangent}). Nu is $\ker
DF(a)^\perp = \operatorname{im}DF(a)^{\mathsf T}$ (de
rang/orthogonaliteitsidentiteit van Jaar~2, of
\cref{exo:b3:hilbert:8} in eindige dimensie: $(\ker T)^\perp
= \operatorname{im}T^*$), die wordt opgespannen door de
gradiĂ«nten $\nabla F_i(a)$ --- onafhankelijk, omdat $DF(a)$
surjectief is: de multiplicatoren bestaan en zijn uniek.
\end{proof}

\begin{example}[De spectraalstelling, meetkundig]
\label{ex:b3:submanifolds:rayleigh}
Zij $A$ een reĂ«le symmetrische $n\times n$-matrix en
maximaliseer $f(x) = \langle Ax, x\rangle$ op de sfeer
$S^{n-1}$ (compact: het maximum wordt bereikt, in enige
$v_1$). Lagrange met $F(x) = \norm x^2 - 1$: $\nabla f =
2Ax$ en $\nabla F = 2x$ geven $Av_1 = \lambda_1v_1$ --- een
eigenvector, met $\lambda_1 = \max_{S^{n-1}}\langle Ax,
x\rangle$. Restricteer $A$ tot $v_1^\perp$ (invariant:
$\langle Av, v_1\rangle = \langle v, Av_1\rangle =
\lambda_1\langle v, v_1\rangle = 0$) en itereer: een
orthonormal basis van eigenvectoren. De spectraalstelling
van Jaar~2, herbewezen door pure optimalisatie --- en de
oneindigdimensionale schaduw van hetzelfde argument bewees
\cref{lem:b3:spectral:existence}.
\end{example}

\begin{method}\label{met:b3:submanifolds:method}
Om te bewijzen dat een verzameling een deelvariĂ«teit is:
toon haar lokaal als $F^{-1}(0)$ met $DF$ surjectief \emph{op
de verzameling} (de meest voorkomende route), of als
grafiek. Om dimensie en \omterm{def:b3:submanifolds:tangent}{raakruimte} te berekenen: $d = n - q$
en $T_a = \ker DF(a)$. Om erop te optimaliseren: Lagrange ---
controleer altijd eerst compactheid (of coĂ«rciviteit), zodat
een extremum bestaat waarop de stelling kan worden
toegepast, en onthoud dat de multiplicatorvergelijking
slechts \emph{noodzakelijk} is: verzamel alle kritieke
punten, vergelijk dan waarden. Voor matrixgroepen,
differentieer krommen in de identiteit om \omterm{def:b3:submanifolds:tangent}{raakruimten} te
identificeren.
\end{method}

\begin{exercise}[$\star\star\star$]\label{exo:b3:submanifolds:9}
(Ongelijkheid van Hadamard) Voor $M \in GL_n(\R)$ met
kolommen $c_1, \dots, c_n$:
\[
\abs{\det M} \;\leq\; \prod_{i=1}^n\norm{c_i}_2 ,
\]
met gelijkheid desda de kolommen orthogonaal zijn.
\emph{(Reduceer tot kolommen van norm $1$ door schaling;
maximaliseer $\det$ op het compacte product van bollen
$(S^{n-1})^n$; in een maximizer geeft Lagrange in elke
kolom apart $\nabla_{c_i}\det = \lambda_ic_i$, en
$\nabla_{c_i}\det$ is de $i$-de kolom van
$\operatorname{com}(M)$: leid af dat $M^{\mathsf T}M$
diagonaal is, dus $= I$, dus $\det = \pm1$.)} Meetkundige
lezing: het volume van een parallellepipedum is hoogstens
het product van zijn randlengten.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:submanifolds:11}
(De orthogonale groep als deelvariĂ«teit, kwantitatief)
(a) Toon aan dat $O_n = \{M : M^{\mathsf T}M = I\}$ compact
is: begrensd (elke kolom is een eenheidsvector, dus $\norm M
\leq \sqrt n$ voor de Euclidische matrixnorm) en gesloten.
(b) Toon aan dat haar \omterm{def:b3:submanifolds:tangent}{raakruimte} in $I$ de ruimte van
antisymmetrische matrices is, van dimensie
$\frac{n(n-1)}2$, en in een algemene $A \in O_n$: $T_AO_n =
\{AK : K^{\mathsf T} = -K\}$.
(c) Leid af dat de afbeelding $t \mapsto A\exp(tK)$ voor
elke antisymmetrische $K$ een kromme is in $O_n$ door $A$
met snelheid $AK$ \emph{(verifieer $\exp(tK) \in O_n$ met
$\exp(X)^{\mathsf T} = \exp(X^{\mathsf T})$ en
$\exp(-X)\exp(X) = I$)}: elke raakvector wordt gerealiseerd
door een expliciete kromme, zonder de stelling van de
impliciete functie.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:submanifolds:12}
(Kritieke punten van de afstand) Zij $M \subseteq \R^n$ een
deelvariĂ«teit en $p \notin M$. Toon aan dat als $x_0 \in M$
de afstand tot $p$ minimaliseert (zo'n punt bestaat wanneer
$M$ gesloten en niet-leeg is --- waarom?), dan
\[
p - x_0 \;\perp\; T_{x_0}M
\]
\emph{(differentieer $t \mapsto \norm{\gamma(t) - p}^2$
langs krommen in $M$)}. Leid af: het dichtstbijzijnde punt
op een sfeer ligt op de straal door het middelpunt; en
gebruik de voorwaarde om de afstand van $p = (2, 0)$ tot de
parabool $y = x^2$ te berekenen (reduceer tot een derdegraads
en los die numeriek op tot drie cijfers).
\end{exercise}

\section{Probleem: \texorpdfstring{$SO(3)$}{SO(3)} en de
quaternionen}

\begin{problem}[{Weekendprobleem --- rotaties, de groep
$S^3$, en de dubbeldekking}]\label{pb:b3:submanifolds:1}
De quaternionen $\mathbb H = \{t + x\mathrm i + y\mathrm j +
z\mathrm k\}$ --- de algebra waarvan de eenheidsgroep $Q_8$
van \cref{pb:b3:groups:1} bevat --- parametriseren
driedimensionale rotaties tweevoudig: de afbeelding
``conjugeren met een eenheidsquaternion'' is een surjectief
morfisme $S^3 \to SO(3)$ met kern $\{\pm1\}$. We bouwen
alles. Herinner/definieer: vermenigvuldiging is
$\R$-bilineair met $\mathrm i^2 = \mathrm j^2 = \mathrm k^2
= \mathrm{ijk} = -1$; de geconjugeerde van $q = t +
x\mathrm i + y\mathrm j + z\mathrm k$ is $\bar q = t -
x\mathrm i - y\mathrm j - z\mathrm k$; $N(q) = q\bar q =
t^2 + x^2 + y^2 + z^2$.

\medskip
\noindent\textbf{Deel I --- De algebra $\mathbb H$ en de
groep $S^3$.}
\begin{enumerate}
  \item Verifieer dat $\mathbb H$ een associatieve
        $\R$-algebra is met \omterm{ex:b3:groups:actions}{centrum} $\R$, dat
        $\overline{pq} = \bar q\,\bar p$, en dat $N(pq) =
        N(p)N(q)$ \emph{(Ă©Ă©n schone route: representeer $q$
        als de $2\times2$ complexe matrix
        $\bigl(\begin{smallmatrix}\alpha & \beta\\
        -\bar\beta & \bar\alpha\end{smallmatrix}\bigr)$, $q
        = \alpha + \beta\mathrm j$, en gebruik $\det$)}.
  \item Leid af dat elke $q \neq 0$ inverteerbaar is
        ($q^{-1} = \bar q/N(q)$): $\mathbb H$ is een
        (niet-commutatief) lichaam, en $S^3 = \{N(q) = 1\}$
        is een groep --- en een compacte
        $3$-deelvariĂ«teit van $\R^4$
        (\cref{ex:b3:submanifolds:examples}).
\end{enumerate}

\medskip
\noindent\textbf{Deel II --- Het rotatiemorfisme.}
Identificeer $\R^3$ met de \emph{pure quaternionen} $P =
\{x\mathrm i + y\mathrm j + z\mathrm k\}$, en voor $q \in
S^3$ definieer $\rho_q(v) = q\,v\,\bar q$.
\begin{enumerate}[resume]
  \item Toon aan dat $\rho_q$ $P$ naar $P$ stuurt
        \emph{(pure quaternionen zijn die met $\bar v =
        -v$)}, $\R$-lineair is, de norm bewaart, en dat
        $\rho \colon q \mapsto \rho_q$ een groeps-morfisme
        $S^3 \to O(3)$ is.
  \item Bereken de kern: $\rho_q = \mathrm{id}$ desda $q$
        commuteert met $\mathrm i, \mathrm j, \mathrm k$
        desda $q \in \R\cap S^3 = \{\pm1\}$.
  \item Schrijf $q = \cos\frac\theta2 + \sin\frac\theta2\,u$
        met $u \in P$, $N(u) = 1$ (waarom is dit altijd
        mogelijk voor $q \in S^3$?). Toon aan dat $\rho_q$
        $u$ vastzet en, op het vlak $u^\perp\cap P$, werkt
        als de rotatie over hoek $\theta$
        \emph{(bereken $\rho_q(w)$ voor $w \perp u$ met $uw
        = -wu$ voor orthogonale pure eenheden --- bewijs
        deze identiteit uit de vermenigvuldigingstafel, of
        uit $uw + wu = -2\langle u, w\rangle$)}.
  \item Concludeer: $\operatorname{im}\rho \subseteq SO(3)$
        (elke $\rho_q$ is een rotatie met as en hoek zoals
        berekend --- determinant $+1$ door continuĂŻteit van
        $q \mapsto \det\rho_q$ op de samenhangende $S^3$, of
        rechtstreeks), en $\rho$ is \emph{op} $SO(3)$: elke
        rotatie van $\R^3$ heeft een as (bewijs: een reĂ«le
        $3\times3$ orthogonale matrix met $\det = 1$ heeft
        eigenwaarde $1$ --- beschouw het karakteristieke
        polynoom) en is dus enige $\rho_q$. Samenvatting:
        \[
        SO(3) \;\cong\; S^3/\{\pm 1\} .
        \]
\end{enumerate}

\medskip
\noindent\textbf{Deel III --- $SO(3)$ als deelvariĂ«teit;
Rodrigues.}
\begin{enumerate}[resume]
  \item Toon aan dat $SO(3)$ een compacte
        $3$-dimensionale deelvariĂ«teit is van $M_3(\R)$ met
        $T_ISO(3) =$ antisymmetrische matrices
        (\cref{exo:b3:submanifolds:5}; de
        determinantvoorwaarde selecteert een unie van
        componenten).
  \item Voor de antisymmetrische matrix $A_u$ \omterm{def:b3:rings:divisibility}{geassocieerd}
        met $u \in \R^3$ ($A_uv = u\wedge v$, het
        kruisproduct), bewijs de \emph{formule van
        Rodrigues}:
        \[
        \eu^{\theta A_u} = I + \sin\theta\,A_u + (1 -
        \cos\theta)\,A_u^2 \qquad (\norm u = 1)
        \]
        \emph{(uit $A_u^3 = -A_u$: splits de
        exponentiaalreeks langs de machten van $A_u$)}, en
        identificeer haar als de rotatie om as $u$ over
        hoek $\theta$. Leid af dat $\exp$ de
        antisymmetrische matrices \emph{op} $SO(3)$
        afbeeldt.
  \item Relateer de twee parametriseringen: toon aan dat $t
        \mapsto \rho_{q(t)}$ met $q(t) = \cos\frac t2 +
        \sin\frac t2\,u$ een Ă©Ă©nparameter-groep van
        rotaties is waarvan de afgeleide in $t = 0$ $A_u$ is
        --- de quaternionische en matrix-exponentialen
        vertellen hetzelfde verhaal op half en vol tempo
        respectievelijk.
\end{enumerate}

\medskip
\noindent\textbf{Deel IV --- De dubbeldekking, gevoeld.}
\begin{enumerate}[resume]
  \item Toon aan dat het pad $q(t) = \cos\frac t2 +
        \sin\frac t2\,\mathrm k$, $t \in \intcc0{2\pi}$,
        een lus is in $SO(3)$ (haar beeld $\rho_{q(t)}$
        keert terug naar de identiteit) waarvan de
        quaternionische lift \emph{geen} lus is: $q(2\pi) =
        -q(0)$. Doorgaan tot $t = 4\pi$ sluit de lift. Leg
        in een korte alinea uit wat dit zegt: een
        $2\pi$-rotatie is niet continu ongedaan te maken
        terwijl een $4\pi$-rotatie dat wel is (de
        riemtruc), omdat lussen van $SO(3)$ worden
        gedetecteerd in haar dubbeldekking $S^3$.
  \item Leid ook het praktische dividend af: samenstelling
        van rotaties $=$ vermenigvuldiging van quaternionen
        ($4$ vermenigvuldigingen aan data in plaats van $9$,
        geen drift van orthogonaliteit) --- verifieer op de
        samenstelling van twee kwartdraaien om $\mathrm i$
        en $\mathrm j$: bereken as en hoek van het product.
\end{enumerate}

\medskip
\noindent\textbf{Deel V --- De expliciete matrix:
Euler--Rodrigues.} Schrijf $q = a + b\mathrm i + c\mathrm j
+ d\mathrm k \in S^3$, zodat $a^2 + b^2 + c^2 + d^2 = 1$.
\begin{enumerate}[resume]
  \item Bereken $\rho_q(\mathrm i)$ \omterm{def:b3:complete:complete}{volledig} uit de
        vermenigvuldigingstafel; verkrijg dan
        $\rho_q(\mathrm j)$ en $\rho_q(\mathrm k)$ door de
        cyclische substitutie $\mathrm i \to \mathrm j \to
        \mathrm k \to \mathrm i$, $(b, c, d) \to (c, d,
        b)$ \emph{(rechtvaardig: cycleren van $\mathrm i,
        \mathrm j, \mathrm k$ zet zich voort tot een
        automorfisme van $\mathbb H$, omdat de
        definiĂ«rende relaties cyclisch symmetrisch zijn)}.
        Concludeer dat de matrix van $\rho_q$ in de basis
        $(\mathrm i, \mathrm j, \mathrm k)$ de
        \emph{Euler--Rodrigues-matrix} is
        \[
        R_q = \begin{pmatrix}
        a^2 + b^2 - c^2 - d^2 & 2(bc - ad) & 2(bd + ac)\\
        2(bc + ad) & a^2 - b^2 + c^2 - d^2 & 2(cd - ab)\\
        2(bd - ac) & 2(cd + ab) & a^2 - b^2 - c^2 + d^2
        \end{pmatrix}.
        \]
  \item (Een rotatie achterstevoren lezen) Toon aan dat
        \[
        \operatorname{tr}R_q = 4a^2 - 1 = 1 + 2\cos\theta,
        \qquad
        \tfrac12\bigl(R_q - R_q^{\mathsf T}\bigr) =
        \sin\theta\,A_u,
        \]
        in de notatie van vragen~5 en~8. Leid een algoritme
        af dat $\pm q$ uit een rotatiematrix $R$
        terugwint: de hoek uit het spoor; de as uit het
        antisymmetrische deel wanneer $0 < \theta < \pi$;
        en, wanneer $\theta = \pi$, bewijs en gebruik de
        identiteit $R + I = 2\,uu^{\mathsf T}$.
  \item Evalueer $R_q$ voor het product $q =
        \frac12(1 + \mathrm i + \mathrm j + \mathrm k)$ van
        vraag~11: een permutatiematrix verschijnt.
        Identificeer de rotatie en verenig met de as en
        hoek gevonden in vraag~11.
\end{enumerate}

\medskip
\noindent\textbf{Deel VI --- Binnen $S^3$: $SU(2)$,
\omterm{ex:b3:groups:actions}{conjugatieklassen}, exponentialen.}
\begin{enumerate}[resume]
  \item Toon aan dat de matrixrepresentatie van vraag~1
        (noem haar $\Phi$) restrictie tot een
        groepsisomorfisme van $S^3$ op de \emph{speciale
        unitaire groep}
        \[
        SU(2) = \bigl\{U \in M_2(\C) : U^*U = I,\ \det U =
        1\bigr\}
        \]
        \emph{(voor surjectiviteit, schrijf de
        vergelijkingen $U^{-1} = U^*$ en $\det U = 1$ uit
        voor een algemene $2\times2$ complexe matrix)}.
  \item Toon aan dat het reĂ«le deel conjugatie-invariant is
        op $S^3$ --- $\operatorname{Re}(pq\bar p) =
        \operatorname{Re}q$ voor alle $p \in S^3$ --- en,
        omgekeerd, dat twee eenheidsquaternionen met
        hetzelfde reĂ«le deel geconjugeerd zijn in $S^3$
        \emph{(reduceer tot het verplaatsen van Ă©Ă©n pure
        eenheidsas op een andere, wat Deel~II levert)}.
        Beschrijf de \omterm{ex:b3:groups:actions}{conjugatieklassen} van $S^3$
        meetkundig; vertaal naar $SU(2)$ (niveaucurven van
        het spoor); en projecteer door $\rho$: twee
        rotaties zijn geconjugeerd in $SO(3)$ dan en
        slechts dan als zij dezelfde hoek $\theta \in
        \intcc0\pi$ hebben.
  \item Definieer $\exp$ op $\mathbb H$ via de
        exponentiaalreeks; controleer absolute
        convergentie, met $\abs{pq} = \abs p\,\abs q$ voor
        $\abs q = \sqrt{N(q)}$. Toon, voor een pure
        eenheid $u$ en $\theta \in \R$,
        \[
        \exp(\theta u) = \cos\theta + \sin\theta\,u ,
        \]
        leid af dat $\exp$ het hypervlak $P$ op $S^3$
        afbeeldt, en controleer dat $\rho_{\exp(su)} =
        \eu^{2sA_u}$: het halfhoek-fenomeen van vraag~9
        opnieuw.
  \item Voor pure quaternionen $v, w$ bewijs de
        productregel $vw = -\langle v, w\rangle +
        v\wedge w$, dus de commutatoridentiteit $vw - wv =
        2\,v\wedge w$; bewijs ook $[A_v, A_w] =
        A_{v\wedge w}$ voor de matrices van vraag~8.
        Concludeer dat de afgeleide van $\rho$ in $1$ langs
        de krommen $t \mapsto \exp(tv)$ het lineaire
        isomorfisme $v \mapsto 2A_v$ is van $P$ op de
        antisymmetrische matrices, en dat het de
        quaternioncommutator naar de matrixcommutator
        transporteert.
\end{enumerate}

\medskip
\noindent\textbf{Deel VII --- Globale structuur.}
\begin{enumerate}[resume]
  \item (Geen continue sectie) Stel $s \colon SO(3) \to
        S^3$ is continu met $\rho \circ s =
        \operatorname{id}$. Voor de lus $R(t) =
        \rho_{q(t)}$ van vraag~10, stel $\varepsilon(t) =
        s(R(t))\,q(t)^{-1}$ voor $t \in \intcc0{2\pi}$.
        Toon aan dat $\varepsilon$ continu is met waarden
        in $\{\pm1\}$, en leid een tegenspraak af: er is
        geen continue globale keuze van een
        eenheidsquaternion die elke rotatie voorstelt.
  \item (Het balmodel) Zij $\bar B \subseteq \R^3$ de
        gesloten bal van straal $\pi$ en $E(v) =
        \eu^{A_v}$, met $E(0) = I$. Toon aan dat $E$ $\bar
        B$ op $SO(3)$ afbeeldt, injectief is op de open
        bal, en op de randbol precies antipoden
        identificeert: $E(\pi u) = E(-\pi u) =
        2uu^{\mathsf T} - I$, zonder andere
        samenvallen. Dus is $SO(3)$ de bal met antipodale
        randpunten gelijmd --- de projectieve ruimte
        $\mathbb{RP}^3$ --- en een diameter wordt de
        niet-contractiele lus van vraag~10.
  \item Toon aan dat $\rho_p\rho_q\rho_p^{-1} =
        \rho_{pq\bar p}$; dat de involuties van $SO(3)$
        (de $R \neq I$ met $R^2 = I$) precies de
        halve-draaien $\rho_w$ zijn met $w$ een pure
        eenheidsquaternion; en dat het \omterm{ex:b3:groups:actions}{centrum} van $SO(3)$
        triviaal is.
  \item Toon aan dat elke rotatie een product is van twee
        halfdraaien: voor $q = \cos\frac\theta2 +
        \sin\frac\theta2\,u$, kies een pure eenheid $w
        \perp u$, controleer dat $w' = qw$ opnieuw een
        pure eenheidsquaternion is, en verifieer $\rho_q =
        \rho_{w'}\rho_w$. Waar liggen de twee assen, en
        welke hoek maken zij?
  \item Concludeer de topologische samenvatting: $SO(3)$ is
        compact en padsamenhangend (geef twee bewijzen:
        continu beeld van $S^3$ onder $\rho$; beeld van
        $\exp$), terwijl $O(3)$ precies twee
        \omterm{def:b3:topology:components}{samenhangscomponenten} heeft, elk homeomorf met
        $SO(3)$.

  \item (Een samenstelling, drie manieren) Zij $R_1$ de
        rotatie over $\frac\pi2$ om de $z$-as en $R_2$ de
        rotatie over $\frac\pi2$ om de $x$-as. Bereken as
        en hoek van $R_2R_1$: (i) door de twee
        $3\times3$-matrices te vermenigvuldigen en
        spoor/antisymmetrisch deel te gebruiken
        (Deel~V); (ii) door de bijbehorende
        eenheidsquaternionen $q_2q_1$ te
        vermenigvuldigen. Controleer dat de twee
        antwoorden overeenkomen: hoek $\frac{2\pi}3$, as
        $\frac1{\sqrt3}(1, -1, 1)$.
  \item (De Cayley-transformatie) Voor $K$ antisymmetrisch,
        toon aan dat $I + K$ inverteerbaar is en
        \[
        C(K) = (I - K)(I + K)^{-1} \in SO(n),
        \]
        met $-1$ nooit een eigenwaarde van $C(K)$; toon
        aan dat $K \mapsto C(K)$ een bijectie is van
        antisymmetrische matrices op $\{R \in SO(n) : -1
        \notin \operatorname{Sp}R\}$, met inverse $R
        \mapsto (I - R)(I + R)^{-1}$. (Een rationale kaart
        van $SO(n)$, metgezel van de transcendentale
        $\exp$ van \cref{exo:b3:submanifolds:11}.)
\end{enumerate}
\end{problem}
