\chapter{Modules sur un anneau principal}\label{ch:b3:modules}

L'algèbre linéaire sur un \emph{anneau} au lieu d'un corps~: ce
petit changement d'hypothèse produit l'un des grands théorèmes
d'unification de l'algèbre. Un \omterm{def:b3:modules:module}{module} sur $\Z$ est un groupe
abélien~; un \omterm{def:b3:modules:module}{module} sur $K[X]$ est un espace vectoriel muni d'un
endomorphisme. Le théorème de structure des \omterm{def:b3:modules:free}{modules de type fini}
sur un \omterm{def:b3:rings:pidufd}{PID} classifie donc, d'un seul coup, tous les groupes
abéliens \omterm{def:b3:modules:free}{de type fini} \emph{et} tous les endomorphismes à
similitude près --- la réduction de Jordan, que le volume de L2
obtenait par d'indélicates récurrences, tombe en corollaire, avec
son frère plus subtil, la forme canonique \emph{rationnelle},
valable sur tout corps. Le moteur calculatoire est la \omterm{thm:b3:modules:smith}{forme
normale de Smith}, une arithmétique des matrices digne d'Euclide.

Tout au long, $A$ est un anneau commutatif, bientôt un \omterm{def:b3:rings:pidufd}{PID}~;
«~\omterm{def:b3:modules:module}{module}~» signifie $A$-module.

\section{Modules, modules libres}

\begin{definition}\label{def:b3:modules:module}
Un \emph{$A$-module}\index{module} est un groupe abélien $(M, +)$
muni d'une multiplication scalaire $A \times M \to M$ vérifiant
les axiomes d'espace vectoriel~: $a(x + y) = ax + ay$, $(a + b)x
= ax + bx$, $(ab)x = a(bx)$, $1x = x$. Sous-modules, quotients
$M/N$, morphismes (applications $A$-linéaires), sommes directes
$\bigoplus_i M_i$, et théorèmes d'isomorphisme se définissent et
se démontrent mot pour mot comme pour les espaces vectoriels et
les groupes abéliens~; en particulier $M/\ker f \cong
\operatorname{im} f$ pour un morphisme $f$.
\end{definition}

\begin{example}\label{ex:b3:modules:examples}
Les trois cas motivants.
\begin{enumerate}
  \item $A = K$ un corps~: les \omterm{def:b3:modules:module}{modules} sont les espaces
        vectoriels.
  \item $A = \Z$~: les \omterm{def:b3:modules:module}{modules} sont exactement les groupes
        abéliens ($nx$ est forcé à valoir $x + \dots + x$), les
        \omterm{def:b3:modules:module}{sous-modules} sont les sous-groupes.
  \item $A = K[X]$~: un \omterm{def:b3:modules:module}{module} est un $K$-espace vectoriel $V$
        muni de l'application $K$-linéaire $u\colon x \mapsto
        X\cdot x$ --- réciproquement, toute paire $(V, u)$ avec
        $u \in \mathcal L(V)$ devient un $K[X]$-module par $P
        \cdot x = P(u)(x)$. Les \omterm{def:b3:modules:module}{sous-modules} sont précisément
        les sous-espaces $u$-stables.
\end{enumerate}
Un \omterm{def:b3:rings:ideal}{idéal} de $A$ est exactement un \omterm{def:b3:modules:module}{sous-module} de $A$~; un \omterm{def:b3:rings:ideal}{anneau
quotient} $A/I$ est un $A$-module. Contrairement aux espaces
vectoriels, les \omterm{def:b3:modules:module}{modules} peuvent avoir de la \emph{\omterm{def:b3:modules:torsion}{torsion}}~:
dans $\Z/6\Z$, l'élément $\bar 3 \ne 0$ est tué par $2 \neq 0$.
\end{example}

\begin{definition}\label{def:b3:modules:free}
$M$ est \emph{de type fini}\index{module de type fini} si $M =
Ax_1 + \dots + Ax_n$ pour certains $x_i$. $M$ est
\emph{libre}\index{module libre} de rang $n$ si $M \cong A^n$,
i.e.\ s'il a une \emph{base} (une famille génératrice
$A$-linéairement indépendante). Tout $M$ de type fini est un
quotient d'un \omterm{def:b3:modules:module}{module} libre~: $(a_1, \dots, a_n) \mapsto \sum
a_ix_i$ envoie $A^n$ sur $M$.
\end{definition}

\begin{proposition}[Invariance du rang]\label{prop:b3:modules:rank}
Si $A \neq 0$ et $A^m \cong A^n$, alors $m = n$.
\end{proposition}

\begin{proof}
Choisir un \omterm{def:b3:rings:primemaximal}{idéal maximal} $\mathfrak m$ de $A$
(\cref{thm:b3:rings:krull}) et poser $k = A/\mathfrak m$, un
corps. Un isomorphisme $f \colon A^m \to A^n$ envoie $\mathfrak
m A^m$ dans $\mathfrak m A^n$ (linéarité), donc induit un
isomorphisme de quotients
\[
A^m/\mathfrak m A^m \;\cong\; A^n/\mathfrak m A^n,
\qquad\text{i.e.}\qquad k^m \cong k^n
\]
comme $k$-espaces vectoriels (le quotient $A^m/\mathfrak m A^m$
est tué par $\mathfrak m$, donc l'\omterm{def:b3:groups:action}{action} de $A$ se factorise à
travers $k$~; les images de la base standard forment une
$k$-base). La théorie de la dimension sur le corps $k$ donne $m
= n$.
\end{proof}

\begin{theorem}[Sous-modules des modules libres]\label{thm:b3:modules:submodule}
Soit $A$ un \omterm{def:b3:rings:pidufd}{PID} et $M \subseteq A^n$ un \omterm{def:b3:modules:module}{sous-module}. Alors $M$
est \omterm{def:b3:modules:free}{libre} de rang $\leq n$.
\end{theorem}

\begin{proof}
Récurrence sur $n$. Pour $n = 1$~: $M$ est un \omterm{def:b3:rings:ideal}{idéal}, donc $M =
(0)$ (\omterm{def:b3:modules:free}{libre} de rang $0$) ou $M = dA \cong A$ ($x \mapsto dx$
est injectif~: anneau intègre). Pour $n > 1$~: soit $\pi \colon
A^n \to A$ la dernière coordonnée. Alors $\pi(M)$ est un \omterm{def:b3:rings:ideal}{idéal},
$(0)$ ou $dA$. Si $(0)$~: $M \subseteq A^{n-1} \times \{0\}$ et
la récurrence s'applique. Sinon choisir $x_0 \in M$ avec
$\pi(x_0) = d$. Tout $x \in M$ s'écrit de façon unique
\[
x = \underbrace{\frac{\pi(x)}{d}}_{\in A}\, x_0 +
\Bigl(x - \tfrac{\pi(x)}d x_0\Bigr),
\qquad x - \tfrac{\pi(x)}d x_0 \in M \cap \ker\pi
\]
($\pi(x) \in dA$, donc le coefficient est dans $A$). Ainsi $M =
Ax_0 \oplus (M \cap \ker \pi)$~: la somme est directe car
$\pi(ax_0) = ad = 0$ force $a = 0$. Par récurrence $M \cap
\ker\pi \subseteq \ker \pi \cong A^{n-1}$ est \omterm{def:b3:modules:free}{libre} de rang
$\leq n - 1$~; adjoindre $x_0$ (indépendant de $\ker\pi$ comme
on vient de le voir) donne une base de $M$ de cardinal $\leq n$.
\end{proof}

\begin{remark}\label{rem:b3:modules:presentation}
Par conséquent, sur un \omterm{def:b3:rings:pidufd}{PID} tout \omterm{def:b3:modules:module}{module} $M$ \omterm{def:b3:modules:free}{de type fini} a une
\emph{présentation finie}~: une surjection $\varphi\colon A^n
\to M$ a un noyau \omterm{def:b3:modules:free}{libre} de base $c_1, \dots, c_k$ ($k \leq n$),
et $M \cong A^n / \operatorname{im}(C)$ où $C \in M_{n,k}(A)$
est la matrice dont les colonnes sont les $c_j$. Comprendre $M$
signifie comprendre une \emph{matrice sur $A$} à changement de
bases près en source et but --- le sujet de la section suivante.
\end{remark}

\section{Forme normale de Smith}

\begin{definition}\label{def:b3:modules:equivalence}
Deux matrices $B, C \in M_{n,k}(A)$ sont \emph{équivalentes} si
$C = QBP$ avec $Q \in GL_n(A)$, $P \in GL_k(A)$ (inversibles
\emph{sur $A$}~: déterminant dans $A^\times$). Des matrices de
présentation équivalentes définissent des \omterm{def:b3:modules:module}{modules} isomorphes
$A^n/\operatorname{im}$ (changer de bases dans $A^n$ et $A^k$).
\end{definition}

\begin{theorem}[Forme normale de Smith]\label{thm:b3:modules:smith}
Soit $A$ un \omterm{def:b3:rings:pidufd}{PID} et $B \in M_{n,k}(A)$. Alors $B$ est équivalente
à une matrice diagonale\index{forme normale de Smith}
\[
\operatorname{diag}(d_1, d_2, \dots, d_r, 0, \dots, 0),
\qquad d_1 \mid d_2 \mid \cdots \mid d_r \neq 0 ,
\]
et les $d_i$ sont uniques à \omterm{def:b3:rings:divisibility}{associés} près~: $d_1 \cdots d_i$ est
un pgcd des mineurs $i \times i$ de $B$ (en particulier ce pgcd
est un invariant d'équivalence). Les $d_i$ sont les
\emph{facteurs invariants}\index{facteurs invariants} de $B$.
\end{theorem}

\begin{proof}
\emph{Existence.} Si $B = 0$, terminé. Sinon, considérer
l'ensemble des idéaux $(b)$ engendrés par les entrées de
matrices équivalentes à $B$~; comme $A$ est \omterm{def:b3:rings:noetherian}{noethérien}, choisir
une matrice $B'$ équivalente à $B$ et une entrée $d$ de $B'$
avec $(d)$ \emph{\omterm{def:b3:rings:primemaximal}{maximal}} dans cet ensemble. Amener $d$ en
position $(1,1)$ par échanges de lignes et de colonnes.

\emph{Affirmation~: $d$ divise toute entrée de $B'$.} D'abord,
colonne 1~: si $b_{i1}$ n'est pas multiple de $d$, soit $e =
\gcd(d, b_{i1}) = ud + vb_{i1}$ (Bézout), donc $(e) \supsetneq
(d)$. Le tour de la matrice $2 \times 2$~: en agissant sur les
lignes $1$ et $i$ par
\[
\begin{pmatrix} u & v \\ -\dfrac{b_{i1}}e & \dfrac de
\end{pmatrix}
\in GL_2(A)
\qquad
\Bigl(\det = \tfrac{ud + v b_{i1}}e = 1\Bigr)
\]
on produit une matrice équivalente avec entrée $e$ en position
$(1,1)$~: cela contredit la maximalité de $(d)$. Donc $d$ divise
la colonne 1, et symétriquement la ligne 1. En soustrayant des
multiples de la ligne 1 et de la colonne 1 on les nettoie~: $B'$
est équivalente à $\begin{pmatrix} d & 0\\ 0 &
B''\end{pmatrix}$. Ensuite, $d$ divise toute entrée $b$ de
$B''$~: ajouter la ligne de $b$ à la ligne 1 (opération
élémentaire~; la nouvelle première ligne contient $d$ et des
entrées en $b$), et répéter l'argument de nettoyage de colonne~:
un non-multiple améliorerait encore $(d)$. Maintenant récurrence
sur la taille~: $B''$, dont toutes les entrées sont divisibles
par $d$, a une forme de Smith $\operatorname{diag}(d_2, \dots)$
dont les entrées restent divisibles par $d$ (toute entrée d'un
$QB''P$ est une $A$-combinaison d'entrées de $B''$)~; poser $d_1
= d$.

\emph{Unicité.} Soit $D_i(B)$ un pgcd de tous les mineurs $i
\times i$. Les opérations sur les lignes et colonnes, et plus
généralement la multiplication par \emph{toute} matrice, ne
peuvent pas rétrécir le pgcd~: les mineurs $i \times i$ de $QB$
sont des $A$-combinaisons de ceux de $B$ (développement de
Cauchy--Binet~; ou directement~: chaque ligne de $QB$ est une
combinaison de lignes de $B$, et les mineurs sont multilinéaires
en les lignes). Donc $D_i(QBP)$ et $D_i(B)$ se divisent l'un
l'autre~: $D_i$ est un invariant d'équivalence. Sur la forme
diagonale, les mineurs $i \times i$ non nuls sont les produits
de $i$ des $d_j$, et la divisibilité $d_1 \mid \dots \mid d_r$
fait de $d_1 \cdots d_i$ le pgcd. D'où $d_1 \cdots d_i =
D_i(B)$ à unités près, et $d_i = D_i/D_{i-1}$ est déterminé.
\end{proof}

\begin{method}\label{met:b3:modules:smith}
Sur un \omterm{def:b3:rings:pidufd}{anneau euclidien} ($\Z$, $K[X]$), la réduction de Smith
est un algorithme --- aucun argument de maximalité n'est
nécessaire~: amener l'entrée de plus petite taille euclidienne
en position $(1,1)$~; si elle ne divise pas une entrée de sa
ligne ou colonne, une division euclidienne laisse un reste
strictement plus petit là --- l'y échanger et recommencer
(terminaison~: les tailles décroissent)~; quand elle divise toute
sa ligne et colonne, les nettoyer~; si elle ne divise pas une
entrée intérieure, ajouter cette ligne à la ligne $1$ et
recommencer~; récurrence sur le bloc intérieur. En pratique sur
des matrices entières~: calculer $D_1 = \gcd$ des entrées,
$D_2$, \dots\ via les mineurs pour les petites tailles, ou
lancer l'algorithme.
\end{method}

\begin{example}[Une réduction de Smith, en entier]
\label{ex:b3:modules:smithworked}
Réduire $M = \begin{pmatrix} 1 & 2 & 3\\ 4 & 5 & 6\\ 7 & 8 & 9
\end{pmatrix}$ sur $\Z$. Le coin $1$ divise tout~: nettoyer sa
ligne et sa colonne ($L_2 \leftarrow L_2 - 4L_1$, $L_3
\leftarrow L_3 - 7L_1$, puis $C_2 \leftarrow C_2 - 2C_1$, $C_3
\leftarrow C_3 - 3C_1$)~:
\[
M \sim \begin{pmatrix} 1 & 0 & 0\\ 0 & -3 & -6\\ 0 & -6 & -12
\end{pmatrix} .
\]
Dans le bloc intérieur, le coin $-3$ divise toutes les
entrées~: $L_3 \leftarrow L_3 - 2L_2$ et $C_3 \leftarrow C_3 -
2C_2$ le nettoient en $\operatorname{diag}(-3, 0)$. En ajustant
les signes (multiplier une ligne par $-1$, opération légale)~:
\[
M \sim \operatorname{diag}(1, 3, 0),
\qquad
\Z^3/M\Z^3 \cong \Z/3\Z \times \Z .
\]
Contrôle croisé par diviseurs déterminantaux~: $D_1 =
\gcd(\text{entrées}) = 1$~; tout mineur $2\times2$ de $M$ est
multiple de $3$ (p.ex.\ $\det\bigl(\begin{smallmatrix}1 & 2\\ 4
& 5\end{smallmatrix}\bigr) = -3$) et l'un vaut $-3$~: $D_2 =
3$~; $D_3 = \det M = 0$. D'où $d_1 = 1$, $d_2 = 3$, $d_3 = 0$~:
même réponse. Deux leçons~: un facteur invariant nul enregistre
la chute de rang (le conoyau ramasse un facteur \omterm{def:b3:modules:free}{libre} $\Z$), et
la chaîne de divisibilité $1 \mid 3 \mid 0$ est le certificat
de Smith --- une réduction diagonale qui viole la chaîne (disons
$\operatorname{diag}(2, 3)$, que l'imprudent peut produire de
$\bigl(\begin{smallmatrix}2 & 0\\ 0 &
3\end{smallmatrix}\bigr)$ en s'arrêtant trop tôt~: forme de
Smith correcte $\operatorname{diag}(1, 6)$, car $D_1 = 1$
ici~!) n'est pas terminée.
\end{example}

\section{Le théorème de structure}

\begin{definition}\label{def:b3:modules:torsion}
Soit $A$ un anneau intègre et $M$ un $A$-module. Le \omterm{def:b3:modules:module}{sous-module}
de \emph{torsion}\index{torsion} est
\[
T(M) = \{x \in M : ax = 0 \text{ pour un certain } a \neq 0\}
\]
(un \omterm{def:b3:modules:module}{sous-module}~: si $ax = by = 0$ alors $ab(x + y) = 0$, $ab
\ne 0$). $M$ est \emph{sans torsion} si $T(M) = 0$, un
\emph{\omterm{def:b3:modules:module}{module} de torsion} si $T(M) = M$.
\end{definition}

\begin{theorem}[Structure des modules de type fini sur un
PID]\label{thm:b3:modules:structure}
Soit $A$ un \omterm{def:b3:rings:pidufd}{PID} et $M$ un $A$-module \omterm{def:b3:modules:free}{de type fini}. Il existe un
unique $r \in \N$ et des non nuls non inversibles $d_1 \mid d_2
\mid \cdots \mid d_s$, uniques à \omterm{def:b3:rings:divisibility}{associés} près, avec\index{théorème de structure des modules}
\[
M \;\cong\; A^r \,\oplus\, A/(d_1) \oplus \cdots \oplus A/(d_s).
\]
De plus $T(M) \cong A/(d_1)\oplus\dots\oplus A/(d_s)$ et $M/T(M)
\cong A^r$~: un \omterm{def:b3:modules:free}{module de type fini} \emph{sans \omterm{def:b3:modules:torsion}{torsion}} sur un
\omterm{def:b3:rings:pidufd}{PID} est \omterm{def:b3:modules:free}{libre}.
\end{theorem}

\begin{proof}
\emph{Existence.} Présenter $M \cong A^n/\operatorname{im}(C)$
(\cref{rem:b3:modules:presentation}) et mettre $C$ sous forme de
Smith~: après les deux changements de base, $M \cong A^n /
(d_1A \times \dots \times d_rA \times 0 \times \dots \times 0)
\cong A/(d_1) \oplus \dots \oplus A/(d_r) \oplus A^{\,n - r}$.
Écarter les facteurs où $d_i$ est une unité ($A/(d_i) = 0$)~; la
chaîne de divisibilité survit.

\emph{L'identification de la \omterm{def:b3:modules:torsion}{torsion}.} Dans la décomposition,
$A^r$ est sans \omterm{def:b3:modules:torsion}{torsion} (un anneau intègre n'a pas de diviseurs
de zéro) et chaque $A/(d_i)$ est de \omterm{def:b3:modules:torsion}{torsion} (tué par $d_i \ne
0$)~; une somme directe scinde la \omterm{def:b3:modules:torsion}{torsion} en conséquence~:
$T(M) = \bigoplus_i A/(d_i)$ et $M/T(M) \cong A^r$.

\emph{Unicité de $r$}~: $M/T(M) \cong A^r$ ne dépend que de $M$,
et la~\cref{prop:b3:modules:rank} fixe $r$.

\emph{Unicité des $d_i$}~: il suffit de traiter le \omterm{def:b3:modules:module}{module} de
\omterm{def:b3:modules:torsion}{torsion} $T = T(M)$. Décomposer chaque $d_i$ en premiers et
scinder par le théorème des restes chinois
(\cref{thm:b3:rings:crt}~; des premiers distincts engendrent des
\omterm{thm:b3:rings:crt}{idéaux comaximaux})~:
\[
A/(d) \cong \bigoplus_{p \mid d} A/\bigl(p^{v_p(d)}\bigr):
\qquad
T \cong \bigoplus_{p} \bigoplus_{j} A/\bigl(p^{k_{p,j}}\bigr),
\]
les \emph{diviseurs élémentaires}\index{diviseurs élémentaires}
$p^{k_{p,j}}$. Réciproquement les $d_i$ se reconstruisent à
partir du multiensemble des \omterm{thm:b3:modules:structure}{diviseurs élémentaires} ($d_s =$
produit de la plus haute puissance de chaque premier, etc.),
donc il suffit de prouver que le multiensemble $\{k_{p,j}\}_j$
est déterminé par $T$, pour chaque premier $p$. Fixer $p$~; pour
$j \geq 1$ considérer les $A/(p)$-espaces vectoriels
$p^{j-1}T/p^jT$. Sur un facteur cyclique $A/(p^k)$~:
\[
p^{j-1}\bigl(A/(p^k)\bigr)\big/p^{j}\bigl(A/(p^k)\bigr)
\cong
\begin{cases}
A/(p) & \text{si } j \leq k,\\
0 & \text{si } j > k,
\end{cases}
\]
et sur un facteur $A/(q^k)$, $q \neq p$~: la multiplication par
$p$ y est bijective ($p$ inversible mod $q^k$~: Bézout), donc le
quotient est $0$. Les sommes directes passent~:
$\dim_{A/(p)} p^{j-1}T/p^jT = \#\{i : k_{p,i} \geq j\}$. Ces
dimensions intrinsèques déterminent le multiensemble des
exposants.
\end{proof}

\begin{corollary}[Groupes abéliens de type fini]
\label{cor:b3:modules:abelian}
Tout groupe abélien \omterm{def:b3:modules:free}{de type fini} est $\Z^r \times
\Z/d_1\Z\times\dots\times\Z/d_s\Z$ avec $d_1 \mid \dots \mid
d_s$, de façon unique. Tout groupe abélien \emph{fini} est un
produit de groupes cycliques d'ordre puissance de premier, unique
comme multiensemble.\index{groupe abélien, structure}
\end{corollary}

\begin{example}\label{ex:b3:modules:abelian}
Les groupes abéliens d'ordre $p^n$ correspondent aux
\emph{partitions} de $n$~: pour $p^4$~: $\Z/p^4$,
$\Z/p^3\times\Z/p$, $(\Z/p^2)^2$, $\Z/p^2 \times (\Z/p)^2$,
$(\Z/p)^4$ --- cinq groupes, car $4$ a cinq partitions. Les
ordres mixtes multiplient les comptes premier par premier
(CRT)~: il y a $5 \times 2$ groupes abéliens d'ordre $2^4 \cdot
3^2 = 144$.
\end{example}

\section{Application~: formes canoniques des endomorphismes}

Soit $K$ un corps, $V$ un $K$-espace vectoriel de dimension
finie $n$, et $u \in \mathcal L(V)$~; faire de $V$ un
$K[X]$-module via $P \cdot x = P(u)(x)$
(\cref{ex:b3:modules:examples}). Ce \omterm{def:b3:modules:module}{module} est \omterm{def:b3:modules:free}{de type fini} (une
$K$-base engendre) et de \omterm{def:b3:modules:torsion}{torsion}~: pour chaque $x$, les $n+1$
vecteurs $x, u(x), \dots, u^n(x)$ sont $K$-liés, fournissant un
polynôme annulateur non nul.

\begin{definition}\label{def:b3:modules:companion}
Pour $P = X^m + a_{m-1}X^{m-1} + \dots + a_0$ unitaire, la
\emph{matrice compagnon}\index{matrice compagnon} est
\[
C_P = \begin{pmatrix}
0 & & & -a_0\\
1 & \ddots & & -a_1\\
& \ddots & 0 & \vdots\\
& & 1 & -a_{m-1}
\end{pmatrix} :
\]
la matrice de la «~multiplication par $X$~» sur $K[X]/(P)$ dans
la base $1, \bar X, \dots, \bar X^{m-1}$.
\end{definition}

\begin{theorem}[Frobenius~: forme canonique rationnelle]
\label{thm:b3:modules:frobenius}
Il existe une unique suite de polynômes unitaires non constants
$P_1 \mid P_2 \mid \dots \mid P_s$ (les \emph{invariants de
similitude}\index{invariants de similitude} de $u$) telle que,
comme $K[X]$-modules,
\[
V \cong K[X]/(P_1) \oplus \cdots \oplus K[X]/(P_s):
\]
dans une base convenable, $u$ a pour matrice diagonale par
blocs $\operatorname{diag}(C_{P_1}, \dots, C_{P_s})$. De plus~:
\begin{enumerate}
  \item $P_s = \mu_u$ (polynôme minimal) et $P_1\cdots P_s =
        \chi_u$ (polynôme caractéristique)~; en particulier
        $\mu_u \mid \chi_u$ \emph{(Cayley--Hamilton
        redémontré)} et $\chi_u \mid \mu_u^{\,s}$, donc $\chi_u$
        et $\mu_u$ ont les mêmes facteurs irréductibles.
  \item Deux endomorphismes (ou matrices carrées) sont
        semblables ssi ils ont les mêmes \omterm{thm:b3:modules:frobenius}{invariants de
        similitude}.
\end{enumerate}
\end{theorem}

\begin{proof}
Le théorème de structure (\cref{thm:b3:modules:structure})
appliqué au \omterm{def:b3:rings:pidufd}{PID} $K[X]$~: le \omterm{def:b3:modules:module}{module} de \omterm{def:b3:modules:torsion}{torsion} $V$ se décompose
avec \omterm{thm:b3:modules:smith}{facteurs invariants} $P_i$, normalisés unitaires (les unités
de $K[X]$ sont $K^\times$)~; aucune partie \omterm{def:b3:modules:free}{libre} n'apparaît ($V$
est de \omterm{def:b3:modules:torsion}{torsion}). Sur chaque facteur cyclique $K[X]/(P_i)$, la
multiplication par $X$ a pour matrice $C_{P_i}$ dans la base des
puissances de $\bar X$~: en concaténant les bases on obtient la
forme par blocs.

(1) L'annulateur de $V = \bigoplus K[X]/(P_i)$ est $(P_1)\cap
\dots\cap(P_s) = (P_s)$ (chaîne de divisibilité~: $P_s$ est un
multiple commun, et la classe de $1$ dans le dernier facteur est
tuée exactement par $(P_s)$)~: $\mu_u = P_s$. Pour $\chi_u$~: sur
un facteur cyclique, $\chi_{C_P} = P$, par récurrence sur $m =
\deg P$. En développant $\det(XI_m - C_P)$ suivant la première
ligne (dont les entrées sont $X$, puis des zéros, puis $a_0$ en
dernière colonne)~:
\[
\det(XI_m - C_P) = X\,\det\bigl(XI_{m-1} - C_{\tilde P}\bigr)
+ (-1)^{1+m}\,a_0\,\det L ,
\]
où $\tilde P = X^{m-1} + a_{m-1}X^{m-2} + \dots + a_1$ (même
forme, une taille en moins) et $L$ est triangulaire de diagonale
$(-1, \dots, -1)$, donc $\det L = (-1)^{m-1}$. Par récurrence le
premier terme est $X\tilde P$, et le second est $a_0$~: le total
est $X\tilde P + a_0 = P$ (cas de base $m=1$~: $\det(X + a_0) =
P$). Les déterminants multiplient sur les blocs~: $\chi_u =
\prod P_i$. Cayley--Hamilton~: $\chi_u \in (\mu_u)$ car $P_s
\mid$ chaque\dots\ réciproquement chaque $P_i \mid P_s$, donc
$\chi_u = \prod P_i$ divise $P_s^{\,s} = \mu_u^s$~; et $\mu_u =
P_s$ divise $\chi_u$ comme l'un de ses facteurs.

(2) Des endomorphismes semblables sont des structures de \omterm{def:b3:modules:module}{modules}
conjuguées, donc ont des invariants égaux (unicité dans
le~\cref{thm:b3:modules:structure})~; réciproquement des
invariants égaux donnent des $K[X]$-modules isomorphes, et un
isomorphisme de \omterm{def:b3:modules:module}{modules} est exactement une bijection linéaire
entrelaçant les deux endomorphismes~: une similitude.
\end{proof}

\begin{corollary}[La similitude est insensible à l'extension de
corps]\label{cor:b3:modules:descent}
Soient $K \subseteq L$ des corps et $M, N \in M_n(K)$. Si $M$ et
$N$ sont semblables sur $L$, elles le sont sur $K$.
\end{corollary}

\begin{proof}
Les \omterm{thm:b3:modules:frobenius}{invariants de similitude} de $M$ se calculent par la formule
des mineurs de Smith (\cref{thm:b3:modules:smith}) appliquée à la
matrice de présentation $XI_n - M$ sur $K[X]$ --- en effet le
$K[X]$-module $V_M = K^n$ a pour présentation $XI_n - M$~:
l'application $K[X]^n \to V_M$, $(Q_i) \mapsto \sum Q_i(M)e_i$,
est surjective de noyau engendré par les colonnes de $XI_n - M$
(vérification directe~: modulo ces colonnes, tout élément de
$K[X]^n$ se réduit à un vecteur constant, et les vecteurs
constants s'envoient bijectivement~; le problème du week-end le
détaille). Les pgcd de polynômes ne changent pas sous extension
de corps~: si $d$ est le pgcd unitaire dans $K[X]$ d'une famille
$(f_j)$, Bézout donne $d = \sum u_jf_j$ avec $u_j \in K[X]$,
donc tout diviseur commun des $f_j$ \emph{dans $L[X]$} divise
$d$~; comme $d$ est lui-même un diviseur commun, c'est le pgcd
dans $L[X]$ aussi. D'où les \omterm{thm:b3:modules:smith}{facteurs invariants} de $XI_n - M$,
quotients de pgcd de mineurs successifs, sont les mêmes sur $K$
et sur $L$~: $M, N$ ont les mêmes \omterm{thm:b3:modules:frobenius}{invariants de similitude} sur
$L$ ssi sur $K$~; conclure par
le~\cref{thm:b3:modules:frobenius}(2).
\end{proof}

\begin{theorem}[Forme de Jordan, redérivée]\label{thm:b3:modules:jordan}
Supposer que $\chi_u$ se scinde sur $K$ (p.ex.\ $K = \C$). En
appliquant à $V$ la décomposition en \omterm{thm:b3:modules:structure}{diviseurs élémentaires}
(démonstration du~\cref{thm:b3:modules:structure}) au lieu des
\omterm{thm:b3:modules:smith}{facteurs invariants}~:
\[
V \cong \bigoplus_{\lambda, j} K[X]\big/\bigl((X -
\lambda)^{k_{\lambda,j}}\bigr),
\]
et dans la base $\bigl(\overline{(X-\lambda)^{k-1}}, \dots,
\overline{(X - \lambda)}, \bar 1\bigr)$ de chaque facteur, $u$
agit comme le bloc de Jordan $J_k(\lambda)$~: tout endomorphisme
à polynôme caractéristique scindé a une base de Jordan, et le
multiensemble de blocs $(\lambda, k)$ est unique.\index{forme de Jordan}
\end{theorem}

\begin{proof}
Les \omterm{thm:b3:modules:structure}{diviseurs élémentaires} du \omterm{def:b3:modules:module}{module} de \omterm{def:b3:modules:torsion}{torsion} $V$ sont les
$(X - \lambda)^k$ avec $X - \lambda$ parcourant les facteurs
irréductibles de $\mu_u$ (qui se scinde, car $\chi_u$ le fait et
les deux ont les mêmes facteurs irréductibles,
\cref{thm:b3:modules:frobenius}). Dans $W =
K[X]/((X-\lambda)^k)$, poser $f_j = \overline{(X -
\lambda)^{k-j}}$ pour $j = 1, \dots, k$~: alors $(X - \lambda)f_j
= f_{j-1}$ (avec $f_0 = 0$), i.e.\ $u(f_j) = \lambda f_j +
f_{j-1}$~: la matrice de $u$ sur $(f_1, \dots, f_k)$ est
exactement $J_k(\lambda)$ (des uns au-dessus de la diagonale).
L'unicité du multiensemble des \omterm{thm:b3:modules:structure}{diviseurs élémentaires} est
le~\cref{thm:b3:modules:structure}.
\end{proof}

\begin{remark}\label{rem:b3:modules:overview}
La hiérarchie des formes canoniques est désormais transparente~:
la forme \emph{rationnelle} existe sur tout corps et détecte la
similitude de façon absolue (\cref{cor:b3:modules:descent})~; la
forme de \emph{Jordan} en est le raffinement quand $\chi_u$ se
scinde. Les démonstrations par dénombrement de dimensions du
théorème de Jordan du volume de L2 sont absorbées~: toute la
combinatoire était l'arithmétique du \omterm{def:b3:rings:pidufd}{PID} $K[X]$.
\end{remark}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:b3:modules:1}
(a) Montrer que $\Q$ n'est pas \omterm{def:b3:modules:free}{de type fini} comme $\Z$-module.
(b) Montrer que $\Q$ est sans \omterm{def:b3:modules:torsion}{torsion} mais non \omterm{def:b3:modules:free}{libre}.
(c) Pourquoi aucun des deux énoncés ne contredit-il
le~\cref{thm:b3:modules:structure}~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b3:modules:2}
Lister les groupes abéliens d'ordre $360$ à isomorphisme près,
sous les deux formes \omterm{thm:b3:modules:structure}{diviseurs élémentaires} et \omterm{thm:b3:modules:smith}{facteurs
invariants}. Combien y a-t-il de groupes abéliens d'ordre
$p^5$~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b3:modules:3}
Calculer la \omterm{thm:b3:modules:smith}{forme normale de Smith} sur $\Z$ de
\[
B = \begin{pmatrix} 2 & 4\\ 6 & 8 \end{pmatrix},
\qquad
C = \begin{pmatrix} 2 & 0 & 0\\ 0 & 3 & 0\\ 0 & 0 & 12
\end{pmatrix},
\]
et identifier les groupes abéliens $\Z^2/B\Z^2$ et $\Z^3/C\Z^3$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:modules:4}
Soit $L \subseteq \Z^n$ un sous-groupe de rang $n$ de base les
colonnes de $B \in M_n(\Z)$, $\det B \neq 0$. Montrer que
$\Z^n/L$ est fini de cardinal $\abs{\det B}$, et que $\Z^n/L
\cong \prod_i \Z/d_i\Z$ pour les \omterm{thm:b3:modules:smith}{facteurs invariants} $d_i$ de
$B$. Illustrer avec $L = \Z(2,0) + \Z(1,3)$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:modules:5}
Soit $A$ un anneau intègre. (a) Vérifier que $T(M)$ est un
\omterm{def:b3:modules:module}{sous-module} et que $M/T(M)$ est sans \omterm{def:b3:modules:torsion}{torsion}. (b) Montrer que
l'\omterm{def:b3:rings:ideal}{idéal} $(X, Y)$ de $K[X,Y]$, comme $K[X,Y]$-module, est sans
\omterm{def:b3:modules:torsion}{torsion} mais non \omterm{def:b3:modules:free}{libre}~: le théorème de structure a vraiment
besoin de l'hypothèse \omterm{def:b3:rings:pidufd}{PID}.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:modules:6}
(a) Montrer que $2\Z$ n'a pas de supplémentaire direct dans le
$\Z$-module $\Z$~: les \omterm{def:b3:modules:module}{sous-modules} des \omterm{def:b3:modules:free}{modules libres} sont
\omterm{def:b3:modules:free}{libres} (\cref{thm:b3:modules:submodule}), mais ils n'ont pas
besoin d'être des facteurs directs. (b) Montrer que si $M
\subseteq A^n$ ($A$ un \omterm{def:b3:rings:pidufd}{PID}) vérifie~: $A^n/M$ est sans \omterm{def:b3:modules:torsion}{torsion},
alors $M$ \emph{est} un facteur direct.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:modules:7}
Résoudre dans $\Z^2$ le système
\[
\begin{cases}
2x + 4y \equiv b_1 \pmod{20},\\
6x + 8y \equiv b_2 \pmod{20},
\end{cases}
\]
pour quelles paires $(b_1, b_2)$ des solutions existent, en
utilisant la forme de Smith de \cref{exo:b3:modules:3}
(changements de variables inversibles des deux côtés).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:modules:8}
(a) Déterminer tous les \omterm{thm:b3:modules:frobenius}{invariants de similitude} et \omterm{thm:b3:modules:jordan}{formes de
Jordan} possibles d'une matrice nilpotente $4 \times 4$, triés
par la partition de $4$ qu'ils réalisent.
(b) Exhiber deux matrices $4\times4$ complexes de mêmes
polynômes caractéristique \emph{et} minimal qui ne sont pas
semblables, et prouver que pour $n \leq 3$ cela ne peut
arriver.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b3:modules:9}
Soit $u \in \mathcal L(V)$, $\dim V = n$. Montrer que les
énoncés suivants sont équivalents~: (i) $V$ est un
$K[X]$-module cyclique (il existe $x$ avec $V = K[u]x$, un
\emph{vecteur cyclique})~; (ii) $\mu_u = \chi_u$~; (iii) $s =
1$ dans le~\cref{thm:b3:modules:frobenius}. En déduire qu'une
\omterm{def:b3:modules:companion}{matrice compagnon} a un vecteur cyclique, et déterminer quand une
matrice diagonale en a un.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b3:modules:10}
Pour $M \in M_n(\Z)$ vue comme endomorphisme de $\Z^n$,
démontrer la formule d'indice~: si $\det M \ne 0$, alors
$[\Z^n : M\Z^n] = \abs{\det M}$, et en déduire que $M \in
GL_n(\Z)$ ssi $\det M = \pm 1$. Application~: le groupe $\Z^2$ a
exactement $\sigma_1(m) = \sum_{d \mid m} d$ sous-groupes
d'indice $m$. \emph{(Compter les matrices sous forme de Hermite
$\bigl(\begin{smallmatrix} a & b\\ 0 & d\end{smallmatrix}\bigr)$,
$ad = m$, $0 \leq b < d$.)}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:modules:11}
(Équations $x^k = e$ dans les groupes abéliens) Soit $G$ un
groupe abélien fini de \omterm{thm:b3:modules:smith}{facteurs invariants} $d_1 \mid d_2 \mid
\dots \mid d_s$.
(a) Montrer que pour tout $k \geq 1$,
\[
\#\{x \in G : x^k = e\} \;=\; \prod_{i=1}^{s}\gcd(k, d_i) .
\]
(b) En déduire~: un groupe abélien fini est cyclique si et
seulement si pour tout $k$, l'équation $x^k = e$ a au plus $k$
solutions.
(c) Retrouver la cyclicité des sous-groupes finis de $K^\times$
($K$ un corps, \cref{ch:b3:galois})~: pourquoi le polynôme $X^k
- 1$ garantit-il le critère de (b)~?
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b3:modules:12}
(Les matrices élémentaires engendrent) (a) Montrer que $M \in
M_n(\Z)$ est inversible dans $M_n(\Z)$ ssi $\det M = \pm1$.
(b) Montrer que $SL_2(\Z)$ est engendré par les deux matrices
élémentaires $E = \bigl(\begin{smallmatrix}1 & 1\\ 0 &
1\end{smallmatrix}\bigr)$ et $F =
\bigl(\begin{smallmatrix}1 & 0\\ 1 & 1\end{smallmatrix}\bigr)$.
\emph{(Lancer l'algorithme d'Euclide sur la première colonne de
$M \in SL_2(\Z)$ par multiplications à gauche par des puissances
de $E, F$, en atteignant $\pm\bigl(\begin{smallmatrix}1 & *\\ 0
& 1\end{smallmatrix}\bigr)$~; finir à la main --- noter $-I =
(EF^{-1}E)^2$.)}
(c) Expliquer le lien avec la réduction de Smith~: sur $\Z$,
des opérations sur les lignes et colonnes de déterminant $1$
suffisent à diagonaliser, aux signes près.
\end{exercise}

\section{Problème~: le commutant et le bicommutant}

\begin{problem}[{Problème du week-end --- forme rationnelle,
commutant, bicommutant}]\label{pb:b3:modules:1}
Soit $K$ un corps, $V$ un $K$-espace vectoriel de dimension $n
\geq 1$, et $u \in \mathcal L(V)$. On étudie le \emph{commutant}
\[
\mathcal C(u) = \{v \in \mathcal L(V) : uv = vu\},
\]
une sous-algèbre de $\mathcal L(V)$ contenant $K[u] = \{P(u) : P
\in K[X]\}$, et l'on démontre la formule de dimension de
Frobenius et le théorème du bicommutant~: $\mathcal C(\mathcal
C(u)) = K[u]$. Tout au long, $V$ est le $K[X]$-module défini par
$u$, de \omterm{thm:b3:modules:smith}{facteurs invariants} $P_1 \mid \cdots \mid P_s$ et de
décomposition cyclique $V = \bigoplus_{i=1}^s V_i$, $V_i =
K[u]\,x_i \cong K[X]/(P_i)$, $n_i = \deg P_i$
(\cref{thm:b3:modules:frobenius}).

\medskip
\noindent\textbf{Partie I --- La matrice de présentation $XI -
M$, et échauffements.}
\begin{enumerate}
  \item Soit $M \in M_n(K)$ et soit $\varphi \colon K[X]^n \to
        V_M = K^n$ l'application envoyant $(Q_1, \dots, Q_n)$
        sur $\sum_i Q_i(M)e_i$. Montrer que $\varphi$ est un
        morphisme surjectif de $K[X]$-modules et que toute
        colonne de $XI_n - M$ vit dans $\ker\varphi$.
  \item Montrer que, modulo les colonnes de $XI_n - M$, tout
        élément de $K[X]^n$ est congruent à un vecteur constant
        \emph{(réduire les degrés en utilisant $Xe_i \equiv
        \sum_j m_{ji}e_j$)}, et en déduire $\ker\varphi =
        (XI_n - M)\,K[X]^n$~: le \omterm{def:b3:modules:module}{module} $V_M$ a pour matrice de
        présentation $XI_n - M$. Retrouver le point de départ
        du~\cref{cor:b3:modules:descent}~: les \omterm{thm:b3:modules:frobenius}{invariants de
        similitude} de $M$ sont les \omterm{thm:b3:modules:smith}{facteurs invariants} non
        unités de $XI_n - M$.
  \item Calculer les \omterm{thm:b3:modules:frobenius}{invariants de similitude} de~: une matrice
        scalaire $\lambda I_n$~; une matrice diagonale à
        entrées diagonales distinctes~; le bloc de Jordan $n
        \times n$ $J_n(0)$~; $\operatorname{diag}(J_2(0),
        J_1(0))$ pour $n = 3$.
  \item Montrer que $\dim K[u] = \deg \mu_u = n_s$.
\end{enumerate}

\medskip
\noindent\textbf{Partie II --- Morphismes entre \omterm{def:b3:modules:module}{modules}
cycliques.}
\begin{enumerate}[resume]
  \item Soient $P, Q$ unitaires non constants. Montrer qu'un
        $K[X]$-morphisme $f \colon K[X]/(P) \to K[X]/(Q)$ est
        déterminé par $f(\bar 1)$, et que $\bar c \in K[X]/(Q)$
        peut servir de $f(\bar 1)$ ssi $P\bar c = 0$ dans
        $K[X]/(Q)$.
  \item En déduire
        \[
        \operatorname{Hom}_{K[X]}\bigl(K[X]/(P),\, K[X]/(Q)\bigr)
        \;\cong\; K[X]\big/\bigl(\gcd(P, Q)\bigr),
        \]
        de dimension $\deg \gcd(P, Q)$ sur $K$. \emph{(Montrer
        que les solutions $\bar c$ de $P\bar c = 0$ dans
        $K[X]/(Q)$ forment le \omterm{def:b3:modules:module}{sous-module} cyclique engendré par
        $\overline{Q/\gcd(P,Q)}$.)}
  \item Démontrer la \emph{formule de Frobenius}~:
        \[
        \dim_K \mathcal C(u) = \sum_{i,j=1}^{s} \deg\gcd(P_i,
        P_j) = \sum_{i=1}^{s} (2s - 2i + 1)\, n_i .
        \]
        \emph{(Un $v$ qui commute est exactement un
        $K[X]$-endomorphisme de $V$~; décomposer
        $\operatorname{End}(\bigoplus_i V_i)$ en matrices de
        morphismes $V_j \to V_i$ et utiliser la chaîne de
        divisibilité.)}
  \item En déduire $\dim \mathcal C(u) \geq n$, avec égalité ssi
        $u$ est cyclique ($s = 1$), et calculer $\dim\mathcal
        C(u)$ pour $u = \lambda\,\mathrm{id}$~: les deux
        extrêmes de la formule.
  \item Vérifier la formule de Frobenius directement pour
        $\operatorname{diag}(J_2(0), J_1(0))$ en calculant le
        commutant explicitement comme matrices $3\times3$.
\end{enumerate}

\medskip
\noindent\textbf{Partie III --- Le théorème du bicommutant.}
Soit $w \in \mathcal C(\mathcal C(u))$~; on prouve $w \in
K[u]$.
\begin{enumerate}[resume]
  \item Montrer $K[u] \subseteq \mathcal C(\mathcal C(u))$, et
        que tout $w \in \mathcal C(\mathcal C(u))$ commute avec
        $u$ --- donc l'inclusion à prouver, $\mathcal
        C(\mathcal C(u)) \subseteq K[u]$, est un véritable
        affûtage de $w \in \mathcal C(u)$.
  \item Supposer d'abord que $u$ est \emph{cyclique}, $V =
        K[u]x$. Montrer directement que $\mathcal C(u) = K[u]$
        \emph{(évaluer un $v$ qui commute sur $x$~: $v(x) =
        P(u)x$ pour un certain $P$, et comparer $v$ avec $P(u)$
        sur la base $u^k x$)}, et conclure le théorème dans ce
        cas.
  \item Retour au cas général. Pour chaque $i$, soit $\pi_i
        \colon V \to V_i$ la projection le long des autres
        facteurs. Montrer $\pi_i \in \mathcal C(u)$, et en
        déduire que $w$ préserve chaque $V_i$ et commute avec
        $u_i = u\restriction_{V_i}$~; conclure via la question
        11 appliquée au $u_i$ cyclique~: il existe des polynômes
        $Q_i$ avec $w\restriction_{V_i} =
        Q_i(u)\restriction_{V_i}$.
  \item Il reste à coller les $Q_i$ en un seul polynôme. Pour $i
        \leq j$ (donc $P_i \mid P_j$), montrer que $\eta_{ij}
        \colon V_j \to V_i$, $R(u)x_j \mapsto R(u)x_i$, est un
        $K[X]$-morphisme bien défini \emph{(ce qu'il faut
        vérifier est que $R(u)x_j = 0$ implique $R(u)x_i =
        0$)}, et que $\tilde\eta_{ij} = \eta_{ij}\circ\pi_j$,
        étendu par $0$ sur les autres facteurs, vit dans
        $\mathcal C(u)$.
  \item En utilisant $w\tilde\eta_{ij} = \tilde\eta_{ij}w$,
        montrer $Q_i \equiv Q_j \pmod{P_i}$ pour $i \leq j$. En
        déduire que $Q = Q_s$ vérifie $Q \equiv Q_i \pmod {P_i}$
        pour tout $i$, donc $w = Q(u)$ sur chaque $V_i$, donc
        sur $V$~:
        \[
        \boxed{\ \mathcal C(\mathcal C(u)) = K[u].\ }
        \]
  \item (Coda) En déduire du théorème~: si $v$ commute avec
        \emph{toute} matrice qui commute avec $u$, et $u$ est
        cyclique, alors $v$ est un polynôme en $u$~; et donner
        un exemple montrant que $\mathcal C(u) = K[u]$ échoue
        pour $u = \mathrm{id}$, $n \geq 2$ --- où exactement
        la cyclicité entre-t-elle~?
\end{enumerate}

\medskip
\noindent\textbf{Partie IV --- Dividendes des \omterm{thm:b3:modules:frobenius}{invariants de
similitude}.} La forme canonique rationnelle est une machine~;
voici cinq de ses sorties classiques.
\begin{enumerate}[resume]
  \item (Transposée) Montrer que toute $M \in M_n(K)$ est
        semblable à sa transposée ${}^tM$. \emph{(Les
        opérations qui amènent $XI - M$ sous forme de Smith,
        transposées, amènent $XI - {}^tM$ à la même forme de
        Smith~: \omterm{thm:b3:modules:frobenius}{invariants de similitude} égaux.)}
  \item (Descente de la similitude) Soient $K \subseteq L$ une
        extension de corps et $M, N \in M_n(K)$. Montrer que si
        $M$ et $N$ sont semblables sur $L$, elles le sont sur
        $K$. \emph{(La forme de Smith de $XI - M$ calculée dans
        $K[X]$ est encore une forme de Smith dans $L[X]$ ---
        pourquoi les \omterm{thm:b3:modules:smith}{facteurs invariants} ne changent-ils
        pas~?)} Conséquence digne d'être mémorisée~: deux
        matrices réelles conjuguées dans $GL_n(\C)$ le sont
        dans $GL_n(\R)$.
  \item (Classification des nilpotents) Soit $u$ nilpotent.
        Montrer que le nombre de blocs de taille $\geq k$ dans
        sa décomposition en blocs de Jordan nilpotents égale
        $\operatorname{rk}u^{k-1} - \operatorname{rk}u^k$, et
        en déduire~: les classes nilpotentes de $M_n(K)$, pour
        tout corps $K$, sont en bijection avec les partitions
        de $n$. Combien de classes nilpotentes dans
        $M_5(K)$~?
  \item (Une paire concrète) Déterminer les \omterm{thm:b3:modules:frobenius}{invariants de
        similitude} de la dérivation $D\colon P \mapsto P'$
        agissant sur l'espace $K_{n-1}[X]$ des polynômes de
        degré $< n$~: (a) pour $K = \Q$~; (b) pour $K =
        \mathbb F_p$ avec $p < n$ \emph{(en caractéristique $p$,
        $(X^p)' = 0$~: calculer $\ker D^k$ et utiliser la
        question 18)}.
  \item (\omterm{ex:b3:groups:actions}{Classes de conjugaison} de $GL_2(\mathbb F_q)$) En
        utilisant les \omterm{thm:b3:modules:smith}{facteurs invariants}, montrer que toute
        classe de $GL_2(\mathbb F_q)$ est d'exactement l'un des
        quatre types~: centrale $aI$~; diagonalisable à deux
        valeurs propres distinctes $a \neq b$ dans $\mathbb
        F_q^\times$~; non \omterm{def:b3:groups:simple}{semi-simple} de polynôme minimal $(X -
        a)^2$~; cyclique de polynôme caractéristique
        irréductible.
  \item Compter les classes de chaque type et conclure~:
        $GL_2(\mathbb F_q)$ a exactement $q^2 - 1$ \omterm{ex:b3:groups:actions}{classes de
        conjugaison}. \emph{(Compter les quadratiques unitaires
        irréductibles sur $\mathbb F_q$~; paires non ordonnées
        $\{a, b\}$~; se souvenir que l'inversibilité contraint
        les termes constants.)}
  \item (Cyclique est générique) Montrer que $M \in M_2(\mathbb
        F_q)$ échoue à être cyclique ssi $M$ est scalaire, et en
        déduire qu'une matrice $2\times2$ uniformément
        aléatoire sur $\mathbb F_q$ est cyclique avec
        probabilité $1 - q^{-3}$. Énoncer l'heuristique
        analogue pour $M_n$ et $q$ grand (aucune démonstration
        requise)~: les matrices non cycliques sont rares ---
        c'est pourquoi la partie III du~\cref{pb:b3:modules:1}
        n'a nécessité un vrai travail qu'au-delà du cas
        générique.
\end{enumerate}

\medskip
\noindent\textbf{Partie V --- Compléments.}
\begin{enumerate}[resume]
  \item (\omterm{ex:b3:groups:actions}{Centre} du commutant) Montrer que le \omterm{ex:b3:groups:actions}{centre} de
        l'algèbre $\mathcal C(u)$ est exactement $K[u]$
        \emph{(combiner les deux inclusions de la partie
        III)}. En déduire que $\mathcal C(u)$ est commutative
        ssi $u$ est cyclique --- retrouvant le cas d'égalité de
        la question 8 par une voie purement structurelle.
  \item (Quelles dimensions apparaissent~?) Déduire de la
        formule de Frobenius que $\dim\mathcal C(u) \equiv n
        \pmod 2$ pour tout $u$. Puis déterminer l'ensemble exact
        des valeurs prises par $\dim \mathcal C(u)$ quand $u$
        parcourt $\mathcal L(V)$ avec $\dim V = 4$~: montrer
        que c'est $\{4, 6, 8, 10, 16\}$ \emph{(énumérer les
        suites de degrés $n_1 \leq \dots \leq n_s$ sommant à
        $4$ et réaliser chacune par un nilpotent)}. En
        particulier $12$ et $14$, bien que de la bonne parité,
        ne sont pas atteints~: la contrainte de parité est
        nécessaire mais non suffisante.
  \item (\omterm{cor:b3:groups:classeq}{Équation aux classes} de $GL_2(\mathbb F_3)$) Pour $q =
        3$, calculer la taille de chaque \omterm{ex:b3:groups:actions}{classe de conjugaison}
        de la question 20 via orbite--stabilisateur~: le
        centralisateur d'une $M$ \emph{cyclique} dans
        $GL_2(\mathbb F_q)$ est le groupe des unités de $K[M]$
        (question 11). Identifier $K[M]$ dans les trois types
        non centraux, lister les trois quadratiques unitaires
        irréductibles sur $\mathbb F_3$, et vérifier
        l'\omterm{cor:b3:groups:classeq}{équation aux classes}
        \[
        48 = \abs{GL_2(\mathbb F_3)}
        = 2\cdot1 + 1\cdot12 + 2\cdot8 + 3\cdot6,
        \]
        avec $2 + 1 + 2 + 3 = 8 = q^2 - 1$ classes, comme prédit
        par la question 21.
\end{enumerate}
\end{problem}
