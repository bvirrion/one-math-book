\chapter{Théorie des groupes}\label{ch:b3:groups}

Le volume de L2 a rencontré les groupes comme dispositifs de
comptabilité~: théorème de Lagrange, groupes cycliques, groupe
symétrique et sa signature. Ce chapitre fait de la théorie des
groupes une \emph{méthode}. Le moteur est la notion de groupe
\emph{agissant} sur un ensemble~: le dénombrement des \omterm{def:b3:groups:action}{orbites} et
des points fixes produit l'\omterm{cor:b3:groups:classeq}{équation aux classes}, le théorème de
Cauchy et les trois théorèmes de Sylow --- le principe
local-global fondamental de la théorie des groupes finis. On
apprend ensuite à assembler les groupes (\omterm{prop:b3:groups:direct}{produits directs} et
semi-directs) et à les désassembler (\omterm{thm:b3:groups:jordanholder}{suites de composition},
\omterm{def:b3:groups:derived}{groupes résolubles}), et l'on démontre le théorème qui, dans
le~\cref{ch:b3:galois}, clôturera une question tricentenaire sur
les équations polynomiales~: le \omterm{thm:b3:groups:ansimple}{groupe alterné} $A_n$ est \omterm{def:b3:groups:simple}{simple}
pour $n \geq 5$.

\section{Groupes quotients et théorèmes d'isomorphisme}

Tout au long, $G$ est un groupe noté multiplicativement, $e$ son
élément neutre. On rappelle du volume de L2~: sous-groupes, classes
$gH$, théorème de Lagrange ($\abs G = [G:H]\,\abs H$ pour $G$
fini), ordre d'un élément, groupes cycliques, et le groupe
symétrique $S_n$ avec son morphisme de signature $\varepsilon
\colon S_n \to \{\pm1\}$.

\begin{definition}\label{def:b3:groups:normal}
Un sous-groupe $N$ de $G$ est \emph{normal}\index{sous-groupe normal}
(on note $N \trianglelefteq G$) lorsque $gNg^{-1} = N$ pour tout $g
\in G$ --- de façon équivalente, lorsque les classes à gauche et à
droite coïncident~: $gN = Ng$ pour tout $g$.
\end{definition}

\begin{theorem}[Groupe quotient]\label{thm:b3:groups:quotient}
Soit $N \trianglelefteq G$. L'ensemble $G/N$ des classes, muni de
la multiplication $(gN)(hN) = ghN$, est un groupe bien défini, le
\emph{groupe quotient}\index{groupe quotient}, et la
\emph{projection canonique} $\pi \colon G \to G/N$, $g \mapsto gN$,
est un morphisme surjectif de noyau $N$. Réciproquement, tout noyau
d'un morphisme de groupes est \omterm{def:b3:groups:normal}{normal}~: les sous-groupes normaux
sont exactement les noyaux.
\end{theorem}

\begin{proof}
La bonne définition est tout l'enjeu. Si $gN = g'N$ et $hN = h'N$,
écrivons $g' = gn$, $h' = hm$ avec $n, m \in N$. Alors $g'h' =
gnhm = gh\,(h^{-1}nh)\,m \in ghN$ car $h^{-1}nh \in N$ par
normalité~: le produit des classes ne dépend pas des
représentants. Associativité, élément neutre $eN = N$ et inverses
$(gN)^{-1} = g^{-1}N$ s'héritent de $G$. Clairement $\pi$ est un
morphisme surjectif et $\pi(g) = N \iff g \in N$.

Si $f \colon G \to H$ est un morphisme et $k \in \ker f$, alors
$f(gkg^{-1}) = f(g)f(k)f(g)^{-1} = e$~: les noyaux sont normaux.
\end{proof}

\begin{theorem}[Propriété universelle~; premier théorème d'isomorphisme]
\label{thm:b3:groups:firstiso}
Soit $f \colon G \to H$ un morphisme et $N \trianglelefteq G$ avec
$N \subseteq \ker f$. Il existe un unique morphisme $\bar f \colon
G/N \to H$ tel que $f = \bar f \circ \pi$. En particulier, en
prenant $N = \ker f$~\index{théorèmes d'isomorphisme}~:
\[
G/\ker f \;\xrightarrow{\;\sim\;}\; \operatorname{im} f,
\qquad gN \mapsto f(g).
\]
\end{theorem}

\begin{proof}
Unicité~: $\bar f(gN)$ doit valoir $f(g)$. Existence~: si $gN =
g'N$ alors $g^{-1}g' \in N \subseteq \ker f$, donc $f(g) = f(g')$
et $\bar f(gN) = f(g)$ est bien défini~; c'est un morphisme parce
que $f$ l'est. Pour $N = \ker f$~: $\bar f$ est injectif, car
$\bar f(gN) = e$ signifie $g \in \ker f$, i.e.\ $gN = N$~; son
image est celle de $f$.
\end{proof}

\begin{theorem}[Deuxième et troisième théorèmes d'isomorphisme]
\label{thm:b3:groups:secondiso}
Soient $H \leq G$ et $N \trianglelefteq G$.
\begin{enumerate}
  \item $HN = \{hn : h \in H,\, n \in N\}$ est un sous-groupe, $N
        \trianglelefteq HN$, $H \cap N \trianglelefteq H$, et
        \[
        H/(H \cap N) \;\cong\; HN/N .
        \]
  \item Si de plus $N \subseteq K \trianglelefteq G$, alors $K/N
        \trianglelefteq G/N$ et $(G/N)\big/(K/N) \cong G/K$.
\end{enumerate}
\end{theorem}

\begin{proof}
(1) $HN$ est un sous-groupe~: $(hn)(h'n') = hh'\,(h'^{-1}nh')n'
\in HN$ et $(hn)^{-1} = h^{-1}(hn^{-1}h^{-1}) \in HN$, en utilisant
la normalité de $N$. Composer $H \hookrightarrow HN
\xrightarrow{\pi} HN/N$~: ce morphisme est surjectif ($hnN = hN$)
de noyau $\{h \in H : h \in N\} = H \cap N$~; appliquer
le~\cref{thm:b3:groups:firstiso}.

(2) La projection $G/N \to G/K$, $gN \mapsto gK$, est bien définie
($N \subseteq K$), surjective, de noyau $K/N$~; appliquer à nouveau
le~\cref{thm:b3:groups:firstiso}.
\end{proof}

\begin{theorem}[Théorème de correspondance]\label{thm:b3:groups:correspondence}
Soit $N \trianglelefteq G$. L'application $H \mapsto H/N$ est une
bijection entre les sous-groupes de $G$ contenant $N$ et les
sous-groupes de $G/N$, préservant inclusions, indices et normalité
(dans les deux sens).
\end{theorem}

\begin{proof}
Son inverse est $\bar H \mapsto \pi^{-1}(\bar H)$. Les deux
applications envoient sous-groupes sur sous-groupes, et sont
mutuellement inverses~: $\pi^{-1}(H/N) = HN = H$ puisque $N
\subseteq H$, et $\pi(\pi^{-1}(\bar H)) = \bar H$ par surjectivité
de $\pi$. Les inclusions sont clairement préservées~; $[G:H] =
[G/N : H/N]$ car $gH \mapsto (gN)(H/N)$ est une bijection bien
définie entre espaces de classes~; et $gHg^{-1} = H$ pour tout $g$
ssi $(gN)(H/N)(gN)^{-1} = H/N$ pour tout $gN$, encore par
surjectivité de $\pi$.
\end{proof}

\begin{example}\label{ex:b3:groups:firstiso}
$\varepsilon \colon S_n \to \{\pm 1\}$ donne $S_n/A_n \cong
\{\pm1\}$~; $\det \colon GL_n(K) \to K^\times$ donne
$GL_n(K)/SL_n(K) \cong K^\times$~; $t \mapsto \eu^{2\iu\pi t}$
donne $\R/\Z \cong \mathbb U$, le groupe du cercle. Le premier
théorème d'isomorphisme est la façon dont on \emph{calcule} les
quotients en pratique~: trouver une surjection au bon noyau.
\end{example}

\begin{method}\label{met:b3:groups:normal}
Pour prouver $N \trianglelefteq G$, par ordre décroissant
d'élégance~: exhiber $N$ comme noyau d'un morphisme défini sur
$G$~; vérifier $gNg^{-1} \subseteq N$ pour tout $g$ (cela
suffit~: en l'appliquant à $g^{-1}$ et en conjuguant on obtient
l'inclusion réciproque)~; vérifier que $N$ est une réunion de
\omterm{ex:b3:groups:actions}{classes de conjugaison}~; ou noter que $[G:N] = 2$ (alors $gN =
Ng$ est forcé --- \cref{exo:b3:groups:1}).
\end{method}

\section{Actions de groupes}

\begin{definition}\label{def:b3:groups:action}
Une \emph{action}\index{action de groupe} de $G$ sur un ensemble
$X$ est un morphisme $\varphi \colon G \to \mathfrak{S}(X)$ vers
le groupe des bijections de $X$~; on note $g \cdot x$ pour
$\varphi(g)(x)$. De façon équivalente~: une application $G \times
X \to X$ avec $e \cdot x = x$ et $g \cdot (h \cdot x) = (gh)
\cdot x$. L'\emph{orbite}\index{orbite} de $x$ est $\mathcal O_x
= \{g \cdot x : g \in G\}$, son \emph{stabilisateur}\index{stabilisateur}
est le sous-groupe $G_x = \{g : g\cdot x = x\}$, et $X^G = \{x :
\forall g,\ g \cdot x = x\}$ est l'ensemble des \emph{points
fixes}. L'action est \emph{transitive} s'il y a exactement une
orbite, \emph{fidèle} si $\varphi$ est injectif, \emph{libre} si
tous les stabilisateurs sont triviaux.
\end{definition}

\begin{example}\label{ex:b3:groups:actions}
Cinq \omterm{def:b3:groups:action}{actions} animent toute la théorie des groupes finis~:
\begin{enumerate}
  \item $G$ sur lui-même par \emph{translation à gauche} $g \cdot
        x = gx$~: libre et transitive.
  \item $G$ sur lui-même par \emph{conjugaison} $g \cdot x =
        gxg^{-1}$~: les \omterm{def:b3:groups:action}{orbites} sont les \emph{classes de
        conjugaison}\index{classe de conjugaison}, les
        \omterm{def:b3:groups:action}{stabilisateurs} les \emph{centralisateurs}\index{centralisateur}
        $Z_G(x) = \{g : gx = xg\}$, les points fixes le
        \emph{centre}\index{centre} $Z(G)$.
  \item $G$ sur l'espace des classes $G/H$ par $g \cdot xH =
        gxH$~: transitive, avec \omterm{def:b3:groups:action}{stabilisateur} de la classe $H$
        égal à $H$. Toute \omterm{def:b3:groups:action}{action} transitive est de cette forme
        (\cref{exo:b3:groups:8}).
  \item $G$ sur son ensemble de sous-groupes par conjugaison~: le
        \omterm{def:b3:groups:action}{stabilisateur} de $H$ est le
        \emph{normalisateur}\index{normalisateur} $N_G(H) = \{g :
        gHg^{-1} = H\}$, le plus grand sous-groupe de $G$ dans
        lequel $H$ est \omterm{def:b3:groups:normal}{normal}.
  \item $S_n$ sur $\intint{1}{n}$~: la mère de tous les exemples.
\end{enumerate}
\end{example}

\begin{theorem}[Orbite--stabilisateur]\label{thm:b3:groups:orbitstab}
L'application $gG_x \mapsto g \cdot x$ est une bijection bien
définie $G/G_x \to \mathcal O_x$. En particulier, pour $G$
fini,\index{théorème orbite--stabilisateur}
\[
\abs{\mathcal O_x} = [G : G_x] \quad\text{divise } \abs G,
\]
et, les \omterm{def:b3:groups:action}{orbites} partitionnant $X$ (ce sont les classes de
l'équivalence $x \sim y \iff y \in \mathcal O_x$),
\[
\abs X = \sum_{i} \,[G : G_{x_i}]
\qquad (x_i\colon \text{un point par orbite}).
\]
\end{theorem}

\begin{proof}
Bien définie et injective~: $gG_x = hG_x \iff h^{-1}g \in G_x
\iff h^{-1}g \cdot x = x \iff g \cdot x = h \cdot x$~; lire la
chaîne dans les deux sens. La surjectivité est la définition de
l'\omterm{def:b3:groups:action}{orbite}. Les énoncés de dénombrement suivent du théorème de
Lagrange et de la partition de $X$ en \omterm{def:b3:groups:action}{orbites}.
\end{proof}

\begin{corollary}[Équation aux classes]\label{cor:b3:groups:classeq}
Pour un groupe fini $G$, en choisissant un représentant $x_i$ dans
chaque \omterm{ex:b3:groups:actions}{classe de conjugaison} à plus d'un élément~\index{équation aux classes}~:
\[
\abs G = \abs{Z(G)} + \sum_i \,[G : Z_G(x_i)],
\qquad\text{chaque } [G : Z_G(x_i)] > 1 \text{ divisant } \abs G.
\]
\end{corollary}

\begin{proof}
Appliquer le~\cref{thm:b3:groups:orbitstab} à l'\omterm{def:b3:groups:action}{action} par
conjugaison~: les \omterm{def:b3:groups:action}{orbites} singletons sont exactement les éléments
de $Z(G)$.
\end{proof}

\begin{theorem}[Points fixes des $p$-groupes]\label{thm:b3:groups:pfixed}
Soit $p$ un nombre premier. Un \emph{$p$-groupe}\index{p-groupe@$p$-groupe}
est un groupe fini dont l'ordre est une puissance de $p$. Si un
$p$-groupe $G$ agit sur un ensemble fini $X$, alors
\[
\abs{X^G} \equiv \abs X \pmod p .
\]
Conséquences~: un $p$-groupe non trivial a un \omterm{ex:b3:groups:actions}{centre} non trivial,
et tout groupe d'ordre $p^2$ est abélien.
\end{theorem}

\begin{proof}
Chaque \omterm{def:b3:groups:action}{orbite} a pour cardinal $[G:G_x]$, une puissance de $p$~;
cette puissance vaut $1$ exactement sur les points fixes et sinon
est divisible par $p$. En sommant sur les \omterm{def:b3:groups:action}{orbites} on obtient la
congruence. Pour le \omterm{ex:b3:groups:actions}{centre}~: l'\omterm{def:b3:groups:action}{action} de $G$ sur lui-même par
conjugaison a $X^G = Z(G)$, donc $\abs{Z(G)} \equiv \abs G
\equiv 0 \pmod p$, et $Z(G) \ni e$ force $\abs{Z(G)} \geq p$.
Ordre $p^2$~: si $Z(G) \neq G$ alors $\abs{Z(G)} = p$ et $G/Z(G)$
est cyclique d'ordre $p$, ce qui force $G$ abélien
(\cref{exo:b3:groups:2}) --- contradiction.
\end{proof}

\begin{theorem}[Cauchy]\label{thm:b3:groups:cauchy}
Si un nombre premier $p$ divise $\abs G$, alors $G$ contient un
élément d'ordre $p$.\index{théorème de Cauchy}
\end{theorem}

\begin{proof}[Démonstration (McKay)]
Soit $X = \{(g_1, \dots, g_p) \in G^p : g_1 g_2 \cdots g_p = e\}$.
Choisir $g_1, \dots, g_{p-1}$ librement détermine $g_p$~: $\abs X
= \abs G^{p-1}$, divisible par $p$. Le groupe cyclique $\Z/p\Z$
agit sur $X$ par permutation circulaire $(g_1, \dots, g_p)
\mapsto (g_2, \dots, g_p, g_1)$ --- cela préserve $X$, car $g_2
\cdots g_p g_1 = g_1^{-1}(g_1 \cdots g_p)g_1 = e$. Par
le~\cref{thm:b3:groups:pfixed}, $\abs{X^{\Z/p\Z}} \equiv \abs X
\equiv 0 \pmod p$. Les points fixes sont les $p$-uplets constants
$(g, \dots, g)$ avec $g^p = e$~; le $p$-uplet $(e, \dots, e)$ en
est un, donc il y en a au moins $p$, d'où au moins un $g \neq e$
avec $g^p = e$~: son ordre est exactement $p$.
\end{proof}

\begin{theorem}[Cayley]\label{thm:b3:groups:cayley}
Tout groupe d'ordre $n$ se plonge dans $S_n$.\index{théorème de Cayley}
\end{theorem}

\begin{proof}
La translation à gauche $\varphi \colon G \to \mathfrak S(G)
\cong S_n$ est un morphisme~; $\varphi(g) = \mathrm{id}$ force $g
= ge = e$~: elle est fidèle.
\end{proof}

\begin{method}\label{met:b3:groups:counting}
Le dénombrement des points fixes est le coup d'ouverture
universel de la théorie des groupes finis. Pour prouver qu'un
objet \emph{existe} (un élément central, un élément d'ordre $p$,
un \omterm{def:b3:groups:normal}{sous-groupe normal}, un point fixe), faire agir un groupe bien
choisi sur un ensemble fini bien choisi, puis comparer
$\abs{X^G}$ et $\abs X$ modulo $p$, ou laisser les tailles
d'\omterm{def:b3:groups:action}{orbites} diviser l'ordre du groupe. Les démonstrations des
\cref{thm:b3:groups:pfixed,thm:b3:groups:cauchy} et des trois
théorèmes de Sylow ci-dessous sont cinq variations sur cette
seule idée.
\end{method}

\section{Les théorèmes de Sylow}

Le théorème de Lagrange dit que l'ordre d'un sous-groupe divise
$\abs G$~; la réciproque échoue ($A_4$, d'ordre $12$, n'a pas de
sous-groupe d'ordre $6$ --- \cref{exo:b3:groups:1}). Les théorèmes
de Sylow sauvent la réciproque pour les puissances de premiers, et
leur clause de dénombrement est l'outil général le plus affûté
dont on dispose pour produire des sous-groupes normaux.

\begin{definition}\label{def:b3:groups:sylow}
Écrivons $\abs G = p^a m$ avec $p \nmid m$. Un \emph{$p$-sous-groupe
de Sylow}\index{sous-groupe de Sylow} de $G$ est un sous-groupe
d'ordre $p^a$ --- un $p$-sous-groupe de l'ordre le plus grand
concevable. Le nombre de $p$-sous-groupes de Sylow de $G$ est
noté $n_p$.
\end{definition}

\begin{lemma}\label{lem:b3:groups:binom}
Si $\abs G = p^a m$ avec $p \nmid m$, alors $\dbinom{p^a m}{p^a}
\equiv m \pmod p$.
\end{lemma}

\begin{proof}
Dans $\mathbb F_p[X]$, le rêve du collégien $(1+X)^p = 1 + X^p$
(les coefficients $\binom pk$, $0<k<p$, sont divisibles par $p$~:
$p$ divise le numérateur de $\frac{p!}{k!(p-k)!}$ mais pas le
dénominateur) itère en $(1+X)^{p^a} = 1 + X^{p^a}$, d'où
\[
(1+X)^{p^a m} = \bigl(1 + X^{p^a}\bigr)^m
= \sum_{k=0}^{m} \binom mk X^{k p^a} \quad\text{dans } \mathbb
F_p[X].
\]
Identifier le coefficient de $X^{p^a}$~: à gauche $\binom{p^a
m}{p^a} \bmod p$, à droite $\binom m1 = m$.
\end{proof}

\begin{theorem}[Sylow~I~: existence]\label{thm:b3:groups:sylow1}
Pour tout premier $p$, des $p$-sous-groupes de Sylow de $G$
existent.
\end{theorem}

\begin{proof}[Démonstration (Wielandt)]
Soit $\Omega$ l'ensemble des \emph{parties} de $G$ de cardinal
$p^a$~; $G$ agit sur $\Omega$ par translation à gauche $g \cdot S
= gS$. Par le~\cref{lem:b3:groups:binom}, $\abs\Omega =
\binom{p^a m}{p^a} \equiv m \not\equiv 0 \pmod p$, donc une
certaine \omterm{def:b3:groups:action}{orbite} $\mathcal O_S$ a une taille première à $p$ (si $p$
divisait toute taille d'\omterm{def:b3:groups:action}{orbite}, il diviserait $\abs\Omega$). Soit
$H = G_S$ le \omterm{def:b3:groups:action}{stabilisateur} d'un tel $S$. Comme $[G : H] =
\abs{\mathcal O_S}$ est premier à $p$ et $p^a \mid \abs G =
[G:H]\,\abs H$, on a $p^a \mid \abs H$. Réciproquement, fixer $s
\in S$~: l'application $H \to S$, $h \mapsto hs$, est injective et
arrive dans $S$ car $hS = S$~; d'où $\abs H \leq \abs S = p^a$.
Donc $\abs H = p^a$.
\end{proof}

\begin{theorem}[Sylow~II~: domination et conjugaison]\label{thm:b3:groups:sylow2}
Soit $P$ un $p$-sous-groupe de Sylow et $Q$ un $p$-sous-groupe
quelconque de $G$. Alors $Q \subseteq gPg^{-1}$ pour un certain $g
\in G$. En particulier tous les $p$-sous-groupes de Sylow sont
conjugués, et $P \trianglelefteq G \iff n_p = 1$.
\end{theorem}

\begin{proof}
Faisons agir $Q$ sur l'espace des classes $X = G/P$, de cardinal
$m \not\equiv 0 \pmod p$. Par le~\cref{thm:b3:groups:pfixed}
appliqué au $p$-groupe $Q$, $\abs{X^Q} \equiv m \not\equiv 0
\pmod p$~: il existe une classe fixe $gP$, i.e.\ $QgP = gP$,
i.e.\ $g^{-1}Qg \subseteq P$. Si $Q$ est lui-même un \omterm{def:b3:groups:sylow}{sous-groupe
de Sylow}, l'égalité des ordres transforme $Q \subseteq gPg^{-1}$
en une égalité. Enfin $P \trianglelefteq G$ ssi ses conjugués
$\{gPg^{-1}\}$ --- qui d'après ce qui précède sont \emph{tous}
les $p$-sous-groupes de Sylow --- se réduisent à $\{P\}$.
\end{proof}

\begin{theorem}[Sylow~III~: dénombrement]\label{thm:b3:groups:sylow3}
$n_p \equiv 1 \pmod p$, et $n_p = [G : N_G(P)]$, qui divise $m$.
\end{theorem}

\begin{proof}
Soit $\mathrm{Syl}_p$ l'ensemble des $p$-sous-groupes de Sylow~;
$G$ agit dessus transitivement par conjugaison
(le~\cref{thm:b3:groups:sylow2}), avec \omterm{def:b3:groups:action}{stabilisateur} de $P$ le
\omterm{ex:b3:groups:actions}{normalisateur} $N_G(P) \supseteq P$~: $n_p = [G : N_G(P)]$, et $m
= [G:P] = [G:N_G(P)]\,[N_G(P):P]$ montre $n_p \mid m$.

Restreignons maintenant l'\omterm{def:b3:groups:action}{action} à $P$ et comptons les points
fixes. Si $Q \in \mathrm{Syl}_p$ est fixé par $P$, alors $P
\subseteq N_G(Q)$~; $P$ et $Q$ sont tous deux des $p$-sous-groupes
de Sylow du groupe $N_G(Q)$, donc conjugués dans celui-ci
(le~\cref{thm:b3:groups:sylow2} appliqué à $N_G(Q)$)~; mais $Q
\trianglelefteq N_G(Q)$, donc $Q$ y est son unique conjugué~: $P
= Q$. Ainsi le seul point fixe est $P$ lui-même, et
le~\cref{thm:b3:groups:pfixed} donne $n_p =
\abs{\mathrm{Syl}_p} \equiv \abs{\mathrm{Syl}_p^P} = 1 \pmod p$.
\end{proof}

\begin{method}\label{met:b3:groups:sylowmethod}
Pour analyser un groupe d'ordre donné $n = p^a m$~: lister les
diviseurs de $m$ congrus à $1$ modulo $p$ --- ce sont les
candidats pour $n_p$. Si le seul candidat est $1$, le
$p$-sous-groupe de Sylow est \omterm{def:b3:groups:normal}{normal}. Si $n_p > 1$ est forcé d'être
petit, faire agir par conjugaison sur $\mathrm{Syl}_p$ pour
obtenir un morphisme $G \to S_{n_p}$ de petit noyau. Et compter
les éléments~: des $p$-sous-groupes de Sylow distincts d'ordre
\emph{premier} $p$ s'intersectent trivialement, donc ils portent
$n_p(p-1)$ éléments d'ordre exactement $p$~; des comptes qui se
chevauchent pour des premiers différents forcent souvent une
contradiction (\cref{exo:b3:groups:7}).
\end{method}

\begin{example}\label{ex:b3:groups:pq}
Soit $\abs G = pq$ avec $p < q$ premiers et $p \nmid q - 1$. Alors
$n_q \mid p$ et $n_q \equiv 1 \bmod q$ forcent $n_q = 1$ (car $p <
q$)~; $n_p \mid q$ et $n_p \equiv 1 \bmod p$ forcent $n_p = 1$
(car $q \not\equiv 1 \bmod p$). Soient $P, Q$ les deux Sylow
normaux~: $P \cap Q = \{e\}$ (ordres premiers entre eux), donc
$\abs{PQ} = pq$ (\cref{exo:b3:groups:4}) et $G \cong P \times Q
\cong \Z/p\Z \times \Z/q\Z \cong \Z/pq\Z$ par
la~\cref{prop:b3:groups:direct} ci-dessous. Tout groupe d'ordre
$15$, $33$, $35$, \dots\ est cyclique. Le cas exclu $p \mid q -
1$ produit exactement un groupe de plus, non abélien --- voir le
problème du week-end (\cref{pb:b3:groups:1}).
\end{example}

\begin{example}[Un recensement de Sylow complet~: $S_4$]
\label{ex:b3:groups:s4sylow}
Appliquons la méthode à $G = S_4$, $\abs G = 24 = 2^3\cdot3$.
\emph{Sylow $3$~:} $n_3 \mid 8$, $n_3 \equiv 1 \bmod 3$, donc $n_3
\in \{1, 4\}$~; comme $\langle(123)\rangle$ et
$\langle(124)\rangle$ sont distincts, $n_3 = 4$ --- les quatre
sous-groupes $\langle(abc)\rangle$, un pour chaque sous-ensemble
à $3$ éléments $\{a, b, c\}$, comptant pour les $8$
$3$-cycles. Par Sylow~II ils sont conjugués, et le morphisme de
conjugaison $S_4 \to S_{\mathrm{Syl}_3} \cong S_4$ est un
isomorphisme ici (son noyau est contenu dans $N =
N_G(\langle(123)\rangle)$ d'ordre $24/4 = 6$, et un \omterm{def:b3:groups:normal}{sous-groupe
normal} de $S_4$ à l'intérieur d'un $N$ de type $S_3$ doit être
trivial~: il serait formé de permutations paires fixant les quatre
Sylow, et seul $e$ le fait). \emph{Sylow $2$~:} $n_2 \mid 3$, $n_2
\equiv 1 \bmod 2$~: $n_2 \in \{1, 3\}$. Le sous-groupe $D =
\langle(1234), (13)\rangle$ a ordre $8$ (un diédral $D_4$~: les
symétries du carré de sommets $1, 2, 3, 4$), n'est pas \omterm{def:b3:groups:normal}{normal}
($(12)(1234)(12) = (2134)$ engendre un sous-groupe de $4$-cycles
différent), donc $n_2 = 3$~: les trois copies de $D_4$
correspondent aux trois façons d'apparier $4$ points en un
«~carré~». Noter la morale du recensement~: $\abs{S_4} = 24$
laisse de la place pour que l'un ou l'autre Sylow échoue à la
normalité, et les deux échouent --- comparer à l'ordre $12$, où
le compte force l'un d'eux \omterm{def:b3:groups:normal}{normal} (partie~IV
du~\cref{pb:b3:groups:1}).
\end{example}

\section{Produits, directs et semi-directs}

\begin{proposition}[Reconnaître un produit direct]\label{prop:b3:groups:direct}
Soient $H, K \trianglelefteq G$ avec $H \cap K = \{e\}$ et $HK =
G$. Alors $(h,k) \mapsto hk$ est un isomorphisme $H \times K \to
G$.\index{produit direct}
\end{proposition}

\begin{proof}
Pour $h \in H$, $k \in K$, le \omterm{def:b3:groups:derived}{commutateur} $hkh^{-1}k^{-1}$
appartient à $K$ (le lire comme $(hkh^{-1})k^{-1}$, en utilisant
la normalité de $K$) et à $H$ (le lire comme $h(kh^{-1}k^{-1})$)~:
il vaut $e$, donc $H$ et $K$ commutent élément par élément et
l'application est un morphisme. Elle est surjective car $HK = G$,
et injective car $hk = e$ donne $h = k^{-1} \in H \cap K =
\{e\}$.
\end{proof}

La normalité des \emph{deux} facteurs est ce qui échoue le plus
souvent~: dans $S_3 = \langle (1\,2\,3)\rangle \,\langle(1\,2)
\rangle$ les deux facteurs s'intersectent trivialement et
engendrent, pourtant $S_3 \not\cong \Z/3\Z \times \Z/2\Z$. La
bonne notion quand un seul facteur est \omterm{def:b3:groups:normal}{normal}~:

\begin{definition}\label{def:b3:groups:semidirect}
Soient $H$, $K$ des groupes et $\varphi \colon K \to
\operatorname{Aut}(H)$ un morphisme. Le \emph{produit
semi-direct}\index{produit semi-direct} $H \rtimes_\varphi K$ est
l'ensemble $H \times K$ muni de
\[
(h, k)\,(h', k') = \bigl(h\,\varphi(k)(h'),\; kk'\bigr).
\]
\end{definition}

\begin{proposition}\label{prop:b3:groups:semidirect}
$H \rtimes_\varphi K$ est un groupe~; $H \times \{e\}$ est un
\omterm{def:b3:groups:normal}{sous-groupe normal} isomorphe à $H$, $\{e\} \times K$ un
sous-groupe isomorphe à $K$~; ils s'intersectent trivialement et
engendrent. Réciproquement, si $G = NK$ avec $N \trianglelefteq
G$, $K \leq G$ et $N \cap K = \{e\}$, alors $G \cong N
\rtimes_\varphi K$ pour $\varphi(k) = (n \mapsto knk^{-1})$.
\end{proposition}

\begin{proof}
Vérification directe~: l'associativité se réduit à $\varphi(kk')
= \varphi(k)\circ\varphi(k')$ et à chaque $\varphi(k)$ étant un
morphisme~; l'élément neutre est $(e,e)$ et $(h,k)^{-1} =
\bigl(\varphi(k^{-1})(h^{-1}), k^{-1}\bigr)$. La projection
$(h,k) \mapsto k$ est un morphisme sur $K$ de noyau $H \times
\{e\}$, qui est donc \omterm{def:b3:groups:normal}{normal}. Pour la réciproque~: tout $g \in G$
s'écrit \emph{de façon unique} $nk$ avec $n \in N$, $k \in K$
(existence~: $G = NK$~; unicité~: $nk = n'k'$ donne $n'^{-1}n =
k'k^{-1} \in N \cap K$), et
\[
(nk)(n'k') = n\,(kn'k^{-1})\;kk'
\]
montre que $nk \mapsto (n, k)$ transporte la loi de $G$ vers
celle de $N \rtimes_\varphi K$.
\end{proof}

\begin{example}\label{ex:b3:groups:semidirectexamples}
(a) Le \emph{groupe diédral}\index{groupe diédral} $D_n$ ($n \geq
3$) des $2n$ symétries d'un $n$-gone régulier~: les rotations
forment un \omterm{def:b3:groups:normal}{sous-groupe normal} d'indice $2$, toute réflexion
engendre un complément, et conjuguer une rotation par une
réflexion l'inverse~: $D_n \cong \Z/n\Z \rtimes_\varphi \Z/2\Z$
avec $\varphi(1) = (x \mapsto -x)$.
(b) Le groupe affine d'une droite, $\{x \mapsto ax + b : a \in
K^\times,\, b \in K\} \cong K \rtimes K^\times$~: translations
\omterm{def:b3:groups:normal}{normales}, homothéties un complément.
(c) $S_n \cong A_n \rtimes \Z/2\Z$ (complément~: toute
transposition).
(d) Le \omterm{pb:b3:groups:1}{groupe des quaternions} $Q_8$ n'est \emph{pas} un \omterm{def:b3:groups:semidirect}{produit
semi-direct} de sous-groupes propres~: tout sous-groupe non
trivial contient $-1$ (\cref{pb:b3:groups:1}), donc aucun couple
de sous-groupes propres ne s'intersecte trivialement.
\end{example}

\section{Groupes résolubles~; simplicité de \texorpdfstring{$A_n$}{An}}

\begin{definition}\label{def:b3:groups:derived}
Le \emph{commutateur}\index{commutateur} de $x, y \in G$ est
$[x,y] = xyx^{-1}y^{-1}$~; le \emph{sous-groupe
dérivé}\index{sous-groupe dérivé} $D(G)$ est le sous-groupe
engendré par tous les commutateurs. La \emph{suite dérivée} est
$D^0(G) = G$, $D^{i+1}(G) = D(D^i(G))$, et $G$ est
\emph{résoluble}\index{groupe résoluble} si $D^n(G) = \{e\}$ pour
un certain $n$.
\end{definition}

\begin{proposition}\label{prop:b3:groups:derived}
$D(G)$ est \omterm{def:b3:groups:normal}{normal} (en fait stable sous tout automorphisme),
$G/D(G)$ est abélien, et pour $N \trianglelefteq G$~: $G/N$ abélien
$\iff D(G) \subseteq N$. De plus $G$ est \omterm{def:b3:groups:derived}{résoluble} ssi il existe
une chaîne $G = G_0 \trianglerighteq G_1 \trianglerighteq \dots
\trianglerighteq G_n = \{e\}$ avec chaque $G_{i+1}
\trianglelefteq G_i$ et chaque quotient $G_i/G_{i+1}$ abélien. Les
sous-groupes et quotients de \omterm{def:b3:groups:derived}{groupes résolubles} sont \omterm{def:b3:groups:derived}{résolubles}~;
réciproquement, si $N$ et $G/N$ sont \omterm{def:b3:groups:derived}{résolubles}, alors $G$ l'est.
\end{proposition}

\begin{proof}
Un automorphisme $\alpha$ envoie $[x,y]$ sur $[\alpha x, \alpha
y]$~: il permute les \omterm{def:b3:groups:derived}{commutateurs}, donc préserve le sous-groupe
qu'ils engendrent~; les conjugaisons sont des automorphismes,
d'où la normalité. Dans $G/D(G)$, $\bar x\bar y \bar x^{-1}\bar
y^{-1} = \overline{[x,y]} = \bar e$~: le quotient est abélien. Si
$G/N$ est abélien alors tout $[x,y] \in N$, donc $D(G) \subseteq
N$~; réciproquement si $D(G) \subseteq N$ alors $G/N$, quotient de
l'abélien $G/D(G)$ par le troisième théorème d'isomorphisme, est
abélien.

Si $G$ est \omterm{def:b3:groups:derived}{résoluble}, la suite dérivée est une telle chaîne.
Réciproquement, étant donnée une chaîne, $D^i(G) \subseteq G_i$
par récurrence~: $G_i/G_{i+1}$ abélien donne $D(G_i) \subseteq
G_{i+1}$, donc $D^{i+1}(G) = D(D^i G) \subseteq D(G_i) \subseteq
G_{i+1}$~; d'où $D^n(G) = \{e\}$.

Hérédité~: $D^i(H) \subseteq D^i(G)$ pour $H \leq G$
(récurrence), et $D^i(G/N) = \pi(D^i(G))$ car $\pi$ envoie
\omterm{def:b3:groups:derived}{commutateurs} sur \omterm{def:b3:groups:derived}{commutateurs}~; cela donne les énoncés pour
sous-groupes et quotients. Extension~: si $D^m(G/N) = \{e\}$
alors $D^m(G) \subseteq N$, et $D^n(N) = \{e\}$ donne
$D^{m+n}(G) = D^n(D^m(G)) \subseteq D^n(N) = \{e\}$.
\end{proof}

\begin{example}\label{ex:b3:groups:solvableexamples}
Les groupes abéliens sont \omterm{def:b3:groups:derived}{résolubles}. Les $p$-groupes sont
\omterm{def:b3:groups:derived}{résolubles}, par récurrence sur l'ordre~: $Z(G) \neq \{e\}$ et
$G/Z(G)$ est un $p$-groupe plus petit. $S_3$ et $S_4$ sont
\omterm{def:b3:groups:derived}{résolubles}~: $S_4 \trianglerighteq A_4 \trianglerighteq V
\trianglerighteq \{e\}$, où $V = \{e, (1\,2)(3\,4), (1\,3)(2\,4),
(1\,4)(2\,3)\}$ est le groupe de Klein des doubles transpositions
(\omterm{def:b3:groups:normal}{normal} dans $S_4$~: une réunion de \omterm{ex:b3:groups:actions}{classes de conjugaison}), avec
quotients abéliens $\Z/2\Z$, $\Z/3\Z$, $V$. Dans
le~\cref{ch:b3:galois}, «~l'équation générale de degré $n$ est
\omterm{def:b3:groups:derived}{résoluble} par radicaux~» voudra littéralement \emph{dire}
«~$S_n$ est un \omterm{def:b3:groups:derived}{groupe résoluble}~». D'où l'importance de la
définition suivante.
\end{example}

\begin{definition}\label{def:b3:groups:simple}
Un groupe $G \neq \{e\}$ est \emph{simple}\index{groupe simple} si
ses seuls sous-groupes normaux sont $\{e\}$ et $G$. Un groupe
simple non abélien n'est pas \omterm{def:b3:groups:derived}{résoluble}~: $D(G) \trianglelefteq G$
n'est pas $\{e\}$ (sinon $G$ abélien), donc $D(G) = G$ et la suite
dérivée est constante. Les groupes simples abéliens sont
exactement les $\Z/p\Z$, $p$ premier (un groupe abélien est simple
ssi il n'a pas de sous-groupe propre non trivial, ssi il est
cyclique d'ordre premier par Lagrange).
\end{definition}

\begin{lemma}\label{lem:b3:groups:threecycles}
Pour $n \geq 3$, $A_n$ est engendré par les $3$-cycles~; pour $n
\geq 5$, tous les $3$-cycles sont conjugués \emph{dans $A_n$}.
\end{lemma}

\begin{proof}
Un élément de $A_n$ est un produit d'un nombre pair de
transpositions~; les apparier et utiliser (composition de droite
à gauche)
\[
(a\,b)(c\,d) = (a\,c\,b)(a\,c\,d), \qquad
(a\,b)(b\,c) = (a\,b\,c), \qquad
(a\,b)(a\,b) = e
\]
pour les paires disjointes, chevauchantes et égales
respectivement~: chaque paire de transpositions est un produit de
$3$-cycles.

Conjugaison~: $\sigma(a\,b\,c)\sigma^{-1} = (\sigma a\; \sigma b\;
\sigma c)$, donc deux $3$-cycles quelconques sont conjugués par un
certain $\sigma \in S_n$. Si $\sigma$ est impair, le remplacer par
$\sigma' = \sigma (d\,e)$ où $d, e$ sont deux points hors de
$\{a, b, c\}$ --- ils existent car $n \geq 5$~; alors $\sigma'$
est pair et $\sigma'(a\,b\,c)\sigma'^{-1} =
\sigma(a\,b\,c)\sigma^{-1}$, car $(d\,e)$ commute avec $(a\,b\,c)$.
\end{proof}

\begin{theorem}[Simplicité du groupe alterné]\label{thm:b3:groups:ansimple}
$A_n$ est \omterm{def:b3:groups:simple}{simple} pour $n \geq 5$.\index{groupe alterné}
\end{theorem}

\begin{proof}
Soit $N \trianglelefteq A_n$, $N \neq \{e\}$. Par
le~\cref{lem:b3:groups:threecycles} il suffit de montrer que $N$
contient \emph{un} $3$-cycle~: la normalité et la conjugaison des
$3$-cycles dans $A_n$ mettent alors tous les $3$-cycles dans $N$,
donc $N = A_n$.

Pour $\rho \in S_n$ soit $F(\rho) = \{x : \rho(x) \neq x\}$ son
\emph{support} et $f(\rho) = \abs{F(\rho)}$. Choisir $\sigma \in N
\setminus \{e\}$ avec $f(\sigma)$ \emph{minimal}. Noter qu'une
permutation paire non triviale a $f \geq 3$, et que $f(\sigma) =
4$ est impossible pour $\sigma \in A_n$ sauf si $\sigma$ est une
double transposition (un $4$-cycle est impair). On montre que
$\sigma$ est un $3$-cycle.

\emph{Cas A~: $\sigma$ est un produit de transpositions
disjointes}, disons $\sigma = (a\,b)(c\,d)\cdots$ avec $f(\sigma)
\geq 4$. Choisir $e' \notin \{a, b, c, d\}$ (possible~: $n \geq
5$), poser $\tau = (c\,d\,e')$ et
\[
\sigma' = \tau\sigma\tau^{-1}\,\sigma^{-1} \in N
\qquad (\tau\sigma\tau^{-1} \in N \text{ par normalité}).
\]
Comme $\sigma\tau^{-1}\sigma^{-1} = (\sigma c\;\sigma e'\;\sigma
d) = (d\;\sigma e'\;c)$ (en utilisant $\sigma c = d$, $\sigma d =
c$), on obtient $\sigma' = (c\,d\,e')(d\;\sigma e'\;c)$.

Si $\sigma e' = e'$ (ce qui vaut en particulier quand $f(\sigma) =
4$, i.e.\ $\sigma = (a\,b)(c\,d)$)~: alors $(d\,e'\,c) =
(c\,d\,e')$ et $\sigma' = (c\,d\,e')^2 = (c\,e'\,d)$, un
$3$-cycle dans $N$, avec $f(\sigma') = 3 < 4 \leq f(\sigma)$ ---
contredisant la minimalité.

Si $\sigma e' \neq e'$~: alors $\sigma e' \notin \{a, b, c, d,
e'\}$ ($\sigma$ échange $a,b$ et $c,d$, et $e' \notin \{a,b,c,d\}$
avec $\sigma$ injectif), donc $\sigma$ déplace les six points $a,
b, c, d, e', \sigma e'$~: $f(\sigma) \geq 6$. D'autre part
$\sigma'$, produit de deux $3$-cycles de supports dans $\{c, d,
e', \sigma e'\}$, vérifie $f(\sigma') \leq 4$~; et $\sigma' \neq
e$, car $\sigma'(d) = \tau\sigma\tau^{-1}(c) = \tau\sigma(e') =
\sigma e' \neq d$ ($\tau$ fixe $\sigma e' \notin \{c,d,e'\}$).
Donc $\sigma' \in N \setminus\{e\}$ avec $f(\sigma') \leq 4 <
f(\sigma)$~: la minimalité est contredite.

\emph{Cas B~: un cycle de $\sigma$ a longueur $\geq 3$}, disons
$\sigma(a) = b$, $\sigma(b) = c$ avec $a, b, c$ distincts. Si
$\sigma$ est exactement ce $3$-cycle, on a terminé. Sinon
$f(\sigma) \geq 5$ (le cas $f(\sigma) = 4$ avec un cycle de
longueur $\geq 3$ est le $4$-cycle impair, exclu), donc on peut
choisir $d, e' \in F(\sigma) \setminus \{a, b, c\}$. Poser $\tau
= (c\,d\,e')$ et $\sigma' = \tau\sigma\tau^{-1}\sigma^{-1} \in N$.
Comme précédemment $\sigma' = (c\,d\,e')\,(\sigma c\;\sigma
e'\;\sigma d)$ ne déplace que des points de
\[
M = \{c, d, e'\} \cup \{\sigma c, \sigma d, \sigma e'\}
\subseteq F(\sigma)
\]
(les images de points déplacés sont déplacées~: $\sigma(x) \ne x$
implique $\sigma(\sigma x) \neq \sigma x$, $\sigma$ étant
injectif). De plus $b \notin M$~: les cinq points $a, b, c, d,
e'$ sont distincts, donc $b \notin \{c, d, e'\}$~; et $b \in
\{\sigma c, \sigma d, \sigma e'\}$ forcerait $a \in \{c, d, e'\}$
(appliquer $\sigma^{-1}$, en utilisant $\sigma a = b$), ce qui
est faux. Donc $\sigma'$ fixe $b$, tandis que $\sigma$ déplace
$b$~; et $F(\sigma') \subseteq F(\sigma)$. Enfin $\sigma' \neq
e$~: $\sigma^{-1}(c) = b$, $\tau^{-1}(b) = b$, $\sigma(b) = c$,
$\tau(c) = d$, donc $\sigma'(c) = d \neq c$. Ainsi $\sigma' \in N
\setminus \{e\}$ avec $f(\sigma') \leq f(\sigma) - 1$,
contredisant la minimalité.

Les deux cas étant impossibles, $\sigma$ est un $3$-cycle.
\end{proof}

\begin{corollary}\label{cor:b3:groups:snnotsolvable}
Pour $n \geq 5$~: $A_n$ et $S_n$ ne sont pas \omterm{def:b3:groups:derived}{résolubles}, et les
seuls sous-groupes normaux de $S_n$ sont $\{e\}$, $A_n$ et $S_n$.
\end{corollary}

\begin{proof}
$A_n$ est \omterm{def:b3:groups:simple}{simple} non abélien, donc non \omterm{def:b3:groups:derived}{résoluble}
(la~\cref{def:b3:groups:simple})~; un groupe contenant un
sous-groupe non \omterm{def:b3:groups:derived}{résoluble} n'est pas \omterm{def:b3:groups:derived}{résoluble}
(la~\cref{prop:b3:groups:derived}). Soit $N \trianglelefteq
S_n$~: alors $N \cap A_n \trianglelefteq A_n$ vaut $\{e\}$ ou
$A_n$. Si $N \cap A_n = A_n$, alors $A_n \subseteq N$ et $N \in
\{A_n, S_n\}$ par l'indice. Si $N \cap A_n = \{e\}$, la
restriction à $N$ de la projection $S_n \to S_n/A_n \cong
\Z/2\Z$ est injective, donc $\abs N \leq 2$~; si $N = \{e,
\sigma\}$, la normalité rend la \omterm{ex:b3:groups:actions}{classe de conjugaison} de $\sigma$
égale à $\{\sigma\}$, i.e.\ $\sigma \in Z(S_n)$. Mais $Z(S_n) =
\{e\}$ pour $n \geq 3$~: si $\sigma \ne e$ envoie $a$ sur $b \neq
a$, choisir $c \notin \{a, b\}$~; alors
$(b\,c)\sigma(b\,c)^{-1}$ envoie $a$ sur $c \ne b$, donc diffère
de $\sigma$. D'où $N = \{e\}$.
\end{proof}

\begin{theorem}[Jordan--Hölder]\label{thm:b3:groups:jordanholder}
Tout groupe fini $G \neq \{e\}$ admet une \emph{suite de
composition}\index{suite de composition}
\[
\{e\} = G_0 \trianglelefteq G_1 \trianglelefteq \cdots
\trianglelefteq G_r = G,
\qquad G_{i}/G_{i-1} \text{ simple},
\]
et le multiensemble des \emph{facteurs de composition}
$G_i/G_{i-1}$, à isomorphisme près, ne dépend pas de la suite
choisie. Un groupe fini est \omterm{def:b3:groups:derived}{résoluble} ssi tous ses facteurs de
composition sont cycliques d'ordre premier.\index{Jordan--Hölder, théorème de}
\end{theorem}

\begin{proof}
\emph{Existence}~: récurrence sur $\abs G$. Si $G$ est \omterm{def:b3:groups:simple}{simple},
prendre $\{e\} \trianglelefteq G$. Sinon choisir un \omterm{def:b3:groups:normal}{sous-groupe
normal} propre maximal $N$ (il y a un nombre fini de sous-groupes)~;
$G/N$ est \omterm{def:b3:groups:simple}{simple} par le théorème de correspondance (un \omterm{def:b3:groups:normal}{sous-groupe
normal} propre non trivial de $G/N$ se relèverait en un \omterm{def:b3:groups:normal}{sous-groupe
normal} de $G$ strictement entre $N$ et $G$). Ajouter $N
\trianglelefteq G$ à une \omterm{thm:b3:groups:jordanholder}{suite de composition} de $N$.

\emph{Unicité}~: récurrence sur $\abs G$, le cas $G$ \omterm{def:b3:groups:simple}{simple} étant
clair. Prendre deux \omterm{thm:b3:groups:jordanholder}{suites de composition}, avec termes
avant-derniers $M \trianglelefteq G$ et $N \trianglelefteq G$
(donc $G/M$, $G/N$ sont \omterm{def:b3:groups:simple}{simples}). Si $M = N$, conclure par
récurrence appliquée à $M$. Sinon $MN$, \omterm{def:b3:groups:normal}{normal} dans $G$ et
contenant strictement $M$, égale $G$ ($M$ est maximal \omterm{def:b3:groups:normal}{normal}~: tout
\omterm{def:b3:groups:normal}{normal} $M \subsetneq L \subsetneq G$ se projetterait en un
\omterm{def:b3:groups:normal}{sous-groupe normal} propre non trivial du \omterm{def:b3:groups:simple}{simple} $G/M$). Le
deuxième théorème d'isomorphisme donne
\[
G/M = MN/M \cong N/(M \cap N),
\qquad
G/N = MN/N \cong M/(M \cap N).
\]
Poser $K = M \cap N$ ($\trianglelefteq G$) et fixer une \omterm{thm:b3:groups:jordanholder}{suite de
composition} de $K$. Alors $M$ porte deux \omterm{thm:b3:groups:jordanholder}{suites de composition}~:
son originale, et la suite de $K$ suivie de $K \trianglelefteq M$
(le quotient $M/K \cong G/N$ est \omterm{def:b3:groups:simple}{simple}). Par récurrence
(appliquée à $M$), les facteurs de la suite originale de $M$ sont
$\{\text{facteurs de } K\} \cup \{G/N\}$~; de même pour $N$.
D'où les deux suites de $G$ ont pour facteurs
\[
\{\text{facteurs de } K\} \;\cup\; \{\,G/N,\; G/M\,\},
\]
le même multiensemble.

Résolubilité~: si tous les facteurs sont $\Z/p_i\Z$, la suite est
une chaîne à quotients abéliens, donc $G$ est \omterm{def:b3:groups:derived}{résoluble}
(la~\cref{prop:b3:groups:derived}). Réciproquement, un facteur de
composition d'un \omterm{def:b3:groups:derived}{groupe résoluble} est \omterm{def:b3:groups:derived}{résoluble} (quotient d'un
sous-groupe) et \omterm{def:b3:groups:simple}{simple}~; un \omterm{def:b3:groups:simple}{groupe simple} \omterm{def:b3:groups:derived}{résoluble} est abélien
($D(G) \ne G$ force $D(G) = \{e\}$), donc un certain $\Z/p\Z$.
\end{proof}

\begin{remark}\label{rem:b3:groups:jhmeaning}
Jordan--Hölder dit que tout groupe fini est construit à partir de
\omterm{def:b3:groups:simple}{groupes simples}, avec une liste de pièces bien définie --- une
arithmétique des groupes dans laquelle les \omterm{def:b3:groups:simple}{groupes simples} sont
les nombres premiers, et où \emph{la façon dont les pièces sont
collées} (données d'extension, comme dans le \omterm{def:b3:groups:semidirect}{produit
semi-direct}) remplace la \omterm{def:b3:groups:simple}{simple} multiplication. La classification
des groupes finis \omterm{def:b3:groups:simple}{simples} --- les cycliques $\Z/p\Z$, les
alternés $A_{n \geq 5}$, seize familles de type de Lie, et $26$
groupes sporadiques --- est l'un des monuments des mathématiques
du vingtième siècle~; sa démonstration, étalée sur quelque dix
mille pages de revues, dépasse de loin ce cours.
\end{remark}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:b3:groups:1}
(a) Montrer que tout sous-groupe d'indice $2$ est \omterm{def:b3:groups:normal}{normal}.
(b) Montrer que si $[G : H] = 2$, alors $x^2 \in H$ pour tout $x
\in G$.
(c) En déduire que $A_4$ n'a pas de sous-groupe d'ordre $6$~: la
réciproque de Lagrange échoue. \emph{(Compter les carrés des
$3$-cycles.)}
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b3:groups:2}
Montrer que si $G/Z(G)$ est cyclique alors $G$ est abélien. En
déduire à nouveau que tout groupe d'ordre $p^2$ est abélien, et
exhiber, pour chaque premier $p$, un groupe non abélien d'ordre
$p^3$. \emph{(Penser aux matrices triangulaires supérieures à
diagonale unité sur $\mathbb F_p$.)}
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b3:groups:3}
(a) Montrer que $\operatorname{Aut}(\Z/n\Z) \cong (\Z/n\Z)^\times$.
(b) Montrer que les automorphismes intérieurs $\iota_g \colon x
\mapsto gxg^{-1}$ forment un \omterm{def:b3:groups:normal}{sous-groupe normal}
$\operatorname{Inn}(G) \trianglelefteq \operatorname{Aut}(G)$,
avec $\operatorname{Inn}(G) \cong G/Z(G)$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:groups:4}
Soient $H, K$ des sous-groupes d'un groupe fini $G$.
(a) Démontrer la \emph{formule du produit} $\abs{HK}\,\abs{H\cap
K} = \abs H\, \abs K$, en comptant les fibres de l'application $H
\times K \to HK$, $(h,k) \mapsto hk$.
(b) Montrer que $HK$ est un sous-groupe ssi $HK = KH$
(automatique quand l'un des deux est \omterm{def:b3:groups:normal}{normal}).
(c) Si $H, K \trianglelefteq G$ et $H \cap K = \{e\}$, montrer que
$hk = kh$ pour tous $h \in H$, $k \in K$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:groups:5}
(Lemme de dénombrement de Burnside) Un groupe fini $G$ agit sur
un ensemble fini $X$. Montrer que le nombre d'\omterm{def:b3:groups:action}{orbites} est le
\emph{nombre moyen de points fixes}~:
\[
\#\{\text{orbites}\} = \frac{1}{\abs G}\sum_{g \in G}
\abs{\operatorname{Fix}(g)},
\qquad \operatorname{Fix}(g) = \{x \in X : g \cdot x = x\},
\]
en comptant l'ensemble $\{(g,x) : g\cdot x = x\}$ de deux façons.
Application~: $\Z/p\Z$ ($p$ premier) agit par rotation sur des
colliers de $p$ perles avec $a$ couleurs disponibles~; en déduire
le petit théorème de Fermat $a^p \equiv a \pmod p$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b3:groups:6}
En utilisant les théorèmes de Sylow, montrer que tout groupe
d'ordre $15$ est cyclique et que tout groupe d'ordre $45$ est
abélien.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:groups:7}
Montrer qu'aucun groupe d'ordre $30$, et aucun d'ordre $56$,
n'est \omterm{def:b3:groups:simple}{simple}. \emph{(Pour $30$~: si $n_3 \neq 1$ et $n_5 \neq 1$,
compter les éléments d'ordres $3$ et $5$. Pour $56$~: compter les
éléments d'ordre $7$.)}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:groups:8}
(a) Soit $H \leq G$ d'indice $n$. Montrer que l'\omterm{def:b3:groups:action}{action} de $G$ sur
$G/H$ produit un morphisme $G \to S_n$ dont le noyau $\bigcap_{g
\in G} gHg^{-1}$ est le plus grand \omterm{def:b3:groups:normal}{sous-groupe normal} de $G$
contenu dans $H$.
(b) En déduire~: si $G$ est fini et $p$ est le \emph{plus petit}
diviseur premier de $\abs G$, tout sous-groupe d'indice $p$ est
\omterm{def:b3:groups:normal}{normal}.
(c) Montrer que toute \omterm{def:b3:groups:action}{action} transitive de $G$ sur un ensemble $X$
est isomorphe à l'\omterm{def:b3:groups:action}{action} sur un espace de classes~: il existe une
bijection $X \to G/G_x$ commutant avec les \omterm{def:b3:groups:action}{actions}.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:groups:9}
(a) Montrer que $D(G)$ est le plus petit \omterm{def:b3:groups:normal}{sous-groupe normal} de $G$
à quotient abélien, et que tout morphisme de $G$ vers un groupe
abélien se factorise de façon unique à travers
l'\emph{abélianisé} $G^{\mathrm{ab}} = G/D(G)$.
(b) Calculer $D(S_n)$ et $S_n^{\mathrm{ab}}$ pour $n \geq 2$, et
$D(Q_8)$ et $Q_8^{\mathrm{ab}}$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:groups:10}
Soit $G$ un $p$-groupe et $H \subsetneq G$ un sous-groupe propre.
Montrer que $H \subsetneq N_G(H)$ («~les \omterm{ex:b3:groups:actions}{normalisateurs}
grossissent~»), et en déduire que tout sous-groupe maximal d'un
$p$-groupe est \omterm{def:b3:groups:normal}{normal} d'indice $p$. \emph{(Récurrence sur $\abs
G$, en utilisant $Z(G) \neq \{e\}$~: traiter séparément $Z(G)
\subseteq H$ et $Z(G) \not\subseteq H$.)}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b3:groups:11}
(Simplicité de $A_5$, en direct)
(a) Montrer que les \omterm{ex:b3:groups:actions}{classes de conjugaison} de $A_5$ ont pour
cardinaux $1$, $15$, $20$, $12$, $12$. Attention au scindage de
la classe de $S_5$ des $5$-cycles~: pour un $5$-cycle $\sigma$,
comparer les \omterm{ex:b3:groups:actions}{centralisateurs} de $\sigma$ dans $S_5$ et dans
$A_5$.
(b) En déduire que $A_5$ est \omterm{def:b3:groups:simple}{simple}~: un \omterm{def:b3:groups:normal}{sous-groupe normal} est
une réunion de \omterm{ex:b3:groups:actions}{classes de conjugaison}, contient $e$, et a un
cardinal divisant $60$.
(c) Montrer qu'un \omterm{def:b3:groups:simple}{groupe simple} d'ordre $60$ a nécessairement
$n_5 = 6$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:groups:12}
(Les \omterm{ex:b3:groups:actions}{normalisateurs} des \omterm{def:b3:groups:sylow}{sous-groupes de Sylow} sont
auto-normalisants) Soit $P$ un $p$-sous-groupe de Sylow d'un
groupe fini $G$ et $H = N_G(P)$.
(a) Montrer que $P$ est l'\emph{unique} $p$-sous-groupe de Sylow
de $H$.
(b) En déduire $N_G(H) = H$. \emph{(Pour $g \in N_G(H)$~:
$gPg^{-1}$ est un $p$-sous-groupe de Sylow de $H$, donc
$gPg^{-1} = P$.)}
(c) Conclure qu'aucun \omterm{ex:b3:groups:actions}{normalisateur} de Sylow n'est contenu dans
un \omterm{def:b3:groups:normal}{sous-groupe normal} propre de $G$, et qu'un sous-groupe maximal
contenant $N_G(P)$ est auto-normalisant.
\end{exercise}

\section{Problème~: les groupes d'ordre au plus 15}

\begin{problem}[{Problème du week-end --- classification des petits
groupes}]\label{pb:b3:groups:1}
Le but est une classification complète, avec démonstrations
entières, des groupes d'ordre $\leq 15$ à isomorphisme près. Les
ordres $1, 2, 3, 5, 7, 11, 13$ sont réglés par Lagrange
(cycliques), et les ordres $4$ et $9$ par
le~\cref{thm:b3:groups:pfixed} plus l'analyse ci-dessous de
$p^2$~: restent $6, 8, 10, 12, 14, 15$.

\medskip
\noindent\textbf{Partie I --- Outils.}
\begin{enumerate}
  \item Montrer qu'un groupe dans lequel tout élément vérifie $x^2
        = e$ est abélien~; en déduire qu'un tel groupe fini a un
        ordre $2^k$ et est isomorphe à $(\Z/2\Z)^k$. \emph{(Le
        voir comme un espace vectoriel sur $\mathbb F_2$.)}
  \item Montrer qu'un groupe d'ordre $p^2$ est isomorphe à
        $\Z/p^2\Z$ ou $(\Z/p\Z)^2$. Lister les groupes abéliens
        d'ordre $8$ à isomorphisme près~: $\Z/8\Z$, $\Z/4\Z
        \times \Z/2\Z$, $(\Z/2\Z)^3$ --- prouver que la liste est
        complète et sans redondance \emph{sans} le théorème de
        structure du~\cref{ch:b3:modules} (discuter selon l'ordre
        maximal d'un élément).
  \item Soient $\varphi, \varphi' \colon K \to
        \operatorname{Aut}(H)$ deux \omterm{def:b3:groups:action}{actions}. Montrer que si
        $\varphi' = \varphi \circ \alpha$ avec $\alpha \in
        \operatorname{Aut}(K)$, alors $H \rtimes_{\varphi} K
        \cong H \rtimes_{\varphi'} K$.
  \item Déterminer $\operatorname{Aut}(\Z/n\Z)$ pour $n = 3, 4,
        5, 7$ explicitement, et montrer
        $\operatorname{Aut}\bigl((\Z/2\Z)^2 \bigr) \cong S_3$.
\end{enumerate}

\medskip
\noindent\textbf{Partie II --- Ordres $2p$ ($6$, $10$, $14$) et
$pq$.}
\begin{enumerate}[resume]
  \item Soit $\abs G = 2p$ avec $p$ un premier impair. Montrer que
        $G$ a un \omterm{def:b3:groups:normal}{sous-groupe normal} $N = \langle r \rangle$ d'ordre
        $p$ et un élément $s$ d'ordre $2$ hors de $N$.
  \item En déduire $G \cong \Z/p\Z \rtimes_\varphi \Z/2\Z$, où
        $\varphi(1) \in \operatorname{Aut}(\Z/p\Z)$ est une
        involution, et conclure~: $G \cong \Z/2p\Z$ ou $G \cong
        D_p$~; vérifier que ces deux ne sont pas isomorphes. Cela
        règle les ordres $6$, $10$, $14$.
  \item Plus généralement, soit $\abs G = pq$ avec $p < q$
        premiers. Montrer que si $p \nmid q-1$ alors $G$ est
        cyclique (\cref{ex:b3:groups:pq}), et que si $p \mid q -
        1$ il y a, outre $\Z/pq\Z$, \emph{exactement un} groupe
        non abélien $\Z/q\Z \rtimes \Z/p\Z$ à isomorphisme près
        --- utiliser la question 3 et le fait que
        $\operatorname{Aut}(\Z/q\Z) \cong (\Z/q\Z)^\times$ est
        cyclique d'ordre $q - 1$, admis ici et démontré dans
        le~\cref{ch:b3:galois} (cyclicité de $\mathbb
        F_q^\times$). Conclure pour l'ordre $15$.
\end{enumerate}

\medskip
\noindent\textbf{Partie III --- Ordre $8$.} Soit $G$ non abélien
d'ordre $8$.
\begin{enumerate}[resume]
  \item Montrer que $G$ a un élément $r$ d'ordre $4$ (utiliser la
        question 1) et que $N = \langle r\rangle$ est \omterm{def:b3:groups:normal}{normal}.
  \item Soit $s \notin N$. Montrer que $s^2 \in N$
        (\cref{exo:b3:groups:1}(b)), que $srs^{-1} = r^{-1}$
        (examiner les images possibles de $r$ par conjugaison,
        qui doivent avoir ordre 4, et exclure $srs^{-1} = r$), et
        que $s^2 \in \{e, r^2\}$ \emph{(que se passe-t-il si $s^2
        = r$ ou $r^3$~? et pourquoi $s^2$ doit-il commuter avec
        $s$~?)}.
  \item Dans le cas $s^2 = e$, montrer $G \cong D_4$.
  \item Dans le cas $s^2 = r^2$, montrer que la table de
        multiplication est entièrement déterminée~; le groupe
        résultant est le \emph{groupe des
        quaternions}\index{groupe des quaternions} $Q_8 = \{\pm
        1, \pm \mathrm i, \pm \mathrm j, \pm \mathrm k\}$,
        $\mathrm i^2 = \mathrm j^2 = \mathrm k^2 = \mathrm i\,
        \mathrm j\,\mathrm k = -1$ (poser $r = \mathrm i$, $s =
        \mathrm j$). Vérifier que $Q_8$ existe, p.ex.\ dans
        $GL_2(\C)$ via
        \[
        \mathrm i \mapsto \begin{pmatrix} \iu & 0\\ 0 & -\iu
        \end{pmatrix},
        \qquad
        \mathrm j \mapsto \begin{pmatrix} 0 & 1\\ -1 & 0
        \end{pmatrix}.
        \]
  \item Montrer que tout sous-groupe non trivial de $Q_8$ contient
        $-1$~; en déduire que tout sous-groupe de $Q_8$ est
        \omterm{def:b3:groups:normal}{normal}, que $D_4 \not\cong Q_8$ (compter les éléments
        d'ordre $2$), et que $Q_8$ n'est pas un \omterm{def:b3:groups:semidirect}{produit
        semi-direct} de deux sous-groupes propres.
\end{enumerate}

\medskip
\noindent\textbf{Partie IV --- Ordre $12$.} Soit $\abs G = 12$,
$P_3 \in \mathrm{Syl}_3(G)$, $P_2 \in \mathrm{Syl}_2(G)$.
\begin{enumerate}[resume]
  \item Montrer $n_3 \in \{1, 4\}$, $n_2 \in \{1, 3\}$, et que
        $n_3 = 4$ force $n_2 = 1$ \emph{(compter les éléments
        d'ordre $3$)}.
  \item Supposer $n_3 = 4$. L'\omterm{def:b3:groups:action}{action} par conjugaison sur
        $\mathrm{Syl}_3$ donne $\rho \colon G \to S_4$. Montrer
        que $\ker \rho$, contenu dans chaque $N_G(P_3)$ et donc
        d'ordre divisant $3$, est trivial \emph{(pourquoi ne
        peut-il pas avoir ordre $3$~?)}~; que l'image, un
        sous-groupe d'ordre $12$ de $S_4$, est nécessairement
        $A_4$ \emph{(les sous-groupes d'indice 2 sont normaux et
        contiennent tous les carrés ---
        \cref{exo:b3:groups:1}~; compter les carrés dans $S_4$)}~;
        et conclure $G \cong A_4$.
  \item Supposer $n_3 = 1$, donc $G \cong \Z/3\Z \rtimes_\varphi
        P_2$ avec $\varphi \colon P_2 \to
        \operatorname{Aut}(\Z/3\Z) \cong \Z/2\Z$. Énumérer les
        cas~: $\varphi$ trivial donne $\Z/12\Z$ et $\Z/6\Z
        \times \Z/2\Z$~; $P_2 = \Z/4\Z$ avec $\varphi$ surjectif
        donne le \emph{groupe dicyclique} $\mathrm{Dic}_3 =
        \Z/3\Z \rtimes \Z/4\Z$~; $P_2 = (\Z/2\Z)^2$ avec
        $\varphi$ surjectif donne, à l'équivalence de la question
        3 près, un seul groupe --- montrer que c'est $D_6$,
        p.ex.\ en exhibant un élément d'ordre $6$ et une
        involution de type réflexion.
  \item Montrer que $\Z/12\Z$, $\Z/6\Z \times \Z/2\Z$, $D_6$,
        $A_4$, $\mathrm{Dic}_3$ sont deux à deux non isomorphes
        \emph{(compter les éléments d'ordre $2$, ou utiliser
        $n_3$)}. Cela règle l'ordre $12$.
\end{enumerate}

\medskip
\noindent\textbf{Partie V --- Synthèse.}
\begin{enumerate}[resume]
  \item Assembler le tableau de classification~: pour chaque
        ordre $n \leq 15$, la liste complète des groupes à
        isomorphisme près, avec les comptes $1, 1, 1, 2, 1, 2,
        1, 5, 2, 2, 1, 5, 1, 2, 1$.
\end{enumerate}

\medskip
\noindent\textbf{Partie VI --- Au-delà~: les groupes d'ordre
$p^3$, $p$ impair.} L'analyse de l'ordre $8$ de la partie III a
un bel analogue pour $p$ impair, avec un phénomène vraiment
nouveau. Soit $p$ un premier impair et $G$ non abélien d'ordre
$p^3$.
\begin{enumerate}[resume]
  \item Montrer que $\abs{Z(G)} = p$, que $G/Z(G) \cong
        (\Z/p\Z)^2$ \emph{(un quotient cyclique par le \omterm{ex:b3:groups:actions}{centre}
        force l'abélianité~: \cref{exo:b3:groups:2})}, et que
        $D(G) = Z(G)$ \emph{(pour $D(G) \subseteq Z(G)$, utiliser
        que $G/Z(G)$ est abélien~; pour l'égalité, $G$ est non
        abélien et $D(G) \neq \{e\}$)}. En déduire que tout
        \omterm{def:b3:groups:derived}{commutateur} $[x, y] = xyx^{-1}y^{-1}$ est central et
        d'ordre divisant $p$.
  \item (L'identité clé) Soient $x, y \in G$ et $z = [y, x]$,
        central. Démontrer par récurrence sur $k$~:
        \[
        (xy)^k = x^k y^k z^{k(k-1)/2} .
        \]
        \emph{(Faire passer chaque $y$ devant chaque $x$~; chaque
        croisement coûte un facteur central $z$.)}
  \item En déduire que pour $p$ impair l'application $\theta
        \colon x \mapsto x^p$ est un morphisme de groupes de $G$
        vers $Z(G)$ \emph{(pourquoi $x^p$ est-il central~?
        pourquoi $z^{p(p-1)/2} = e$ a-t-il besoin de $p$
        impair~?)}, et conclure que $G$ a exposant $p$ ou $p^2$,
        les deux cas se distinguant selon que $\theta$ est
        trivial.
  \item (Exposant $p$) Supposer que tout élément vérifie $x^p =
        e$. Choisir $x, y$ dont les classes engendrent $G/Z(G)$ et
        poser $z = [y, x]$. Montrer que $z \neq e$, que tout
        élément de $G$ s'écrit de façon unique $x^ay^bz^c$ ($0
        \leq a, b, c < p$), et que la multiplication est
        entièrement déterminée par les relations $x^p = y^p =
        z^p = e$, $z$ central, $[y, x] = z$. Vérifier que le
        \emph{groupe de Heisenberg}
        \[
        H_p = \left\{
        \begin{pmatrix} 1 & a & c\\ 0 & 1 & b\\ 0 & 0 & 1
        \end{pmatrix} : a, b, c \in \mathbb F_p \right\}
        \subseteq GL_3(\mathbb F_p)
        \]
        réalise ces relations et a exposant $p$ (calculer $(I +
        N)^p$ avec $N$ strictement triangulaire supérieure, en
        utilisant $N^3 = 0$ et $p \geq 3$)~: tout groupe non
        abélien d'exposant $p$ et d'ordre $p^3$ est isomorphe à
        $H_p$.
  \item (Exposant $p^2$) Supposer qu'un certain $r \in G$ a ordre
        $p^2$, et poser $N = \langle r\rangle$, \omterm{def:b3:groups:normal}{normal} (indice
        $p$~: \cref{exo:b3:groups:10}). Montrer qu'il existe $s
        \notin N$ avec $s^p = e$ \emph{(prendre un $t \notin N$
        quelconque~; en utilisant la question 20, le corriger~:
        $\theta(t) = t^p \in Z(G) \subseteq N$ --- justifier
        $Z(G) = \langle r^p\rangle$ --- et choisir $a$ avec $s =
        tr^{a}$ vérifiant $s^p = e$~; où utilise-t-on $p$
        impair~?)}. Montrer $srs^{-1} = r^{1+p}$ quitte à
        remplacer $s$ par une puissance, et conclure~: il y a
        exactement \emph{un} groupe non abélien d'ordre $p^3$ et
        d'exposant $p^2$, à savoir $\Z/p^2\Z \rtimes_\varphi
        \Z/p\Z$ avec $\varphi(1)\colon r \mapsto r^{1+p}$
        \emph{(utiliser la question 3~;
        $\operatorname{Aut}(\Z/p^2\Z)$ est cyclique d'ordre
        $p(p-1)$, admis ici, donc il a un unique sous-groupe
        d'ordre $p$)}.
  \item Conclure le compte~: pour $p$ impair il y a exactement $5$
        groupes d'ordre $p^3$ (trois abéliens, deux non
        abéliens), tout comme pour $p = 2$ --- mais les deux non
        abéliens ne sont plus $D_4$ et $Q_8$. Localiser exactement
        où l'argument pour $p$ impair casse pour $p = 2$~: dans
        l'identité de la question 19, $z^{k(k-1)/2}$ pour $k = p
        = 2$ vaut $z^1 \neq e$, donc l'élévation au carré n'est
        pas un morphisme --- et en effet $Q_8$ a un unique
        élément d'ordre $2$ tandis que $D_4$ d'exposant $4$ en a
        cinq.
\end{enumerate}

\medskip
\noindent\textbf{Partie VII --- Compléments.}
\begin{enumerate}[resume]
  \item Pour $p$ impair, compter les éléments d'ordre $p$ dans
        chacun des deux groupes non abéliens d'ordre $p^3$~:
        montrer que $H_p$ en a exactement $p^3 - 1$, tandis que
        $M_p = \Z/p^2\Z \rtimes \Z/p\Z$ en a exactement $p^2 - 1$
        \emph{(utiliser le morphisme $\theta$ de la question 20~:
        identifier son image, puis l'ordre de son noyau)}.
        Vérifier numériquement pour $p = 3$~: $26$ contre $8$.
        Expliquer pourquoi aucun argument de morphisme d'élévation
        au carré de ce type ne peut séparer $D_4$ de $Q_8$, et
        quel dénombrement les sépare.
  \item Appeler un entier $n \geq 1$ \emph{cyclique} si tout
        groupe d'ordre $n$ est cyclique. Montrer que si $p^2 \mid
        n$ pour un certain premier $p$, ou si $n$ a des diviseurs
        premiers $p < q$ avec $p \mid q - 1$, alors $n$ n'est pas
        cyclique \emph{(dans chaque cas exhiber un groupe non
        cyclique d'ordre $n$, en utilisant la partie II pour le
        second)}. En déduire que $n$ cyclique force $\gcd(n,
        \varphi(n)) = 1$, où $\varphi$ est l'indicatrice d'Euler,
        et vérifier sur le tableau de la question 17~: parmi $n
        \leq 15$, les ordres portant un seul groupe sont
        exactement $n \in \{1, 2, 3, 5, 7, 11, 13, 15\}$,
        précisément ceux avec $\gcd(n, \varphi(n)) = 1$.
\end{enumerate}
\end{problem}
