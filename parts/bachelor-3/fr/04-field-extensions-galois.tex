\chapter{Extensions de corps et théorie de Galois}\label{ch:b3:galois}

Toute équation se résout-elle par radicaux, comme le suggèrent la
formule quadratique et les formules cubiques de Cardan~? Peut-on
trisecter un angle à la règle et au compas~? Les deux questions,
ouvertes pendant des siècles, reçoivent --- négativement --- une
réponse d'une seule idée d'Évariste Galois~: attacher à tout
polynôme un \emph{groupe} fini de symétries de ses racines, et
lire la réponse sur le groupe. Ce chapitre construit le
dictionnaire~: \omterm{def:b3:galois:extension}{extensions de corps} et \omterm{def:b3:galois:extension}{degré}, \omterm{thm:b3:galois:splitting}{corps de
décomposition}, \omterm{thm:b3:galois:finitefields}{corps finis}, séparabilité, et la correspondance
fondamentale de Galois. On en tire les applications
classiques~: les corps cyclotomiques, l'impossibilité de la
trisection et de la duplication du cube, et le théorème que
l'équation générale de \omterm{def:b3:galois:extension}{degré} $n \geq 5$ n'est pas \omterm{def:b3:groups:derived}{résoluble} par
radicaux --- où la simplicité de $A_n$
(\cref{thm:b3:groups:ansimple}), démontrée dans le chapitre de
théorie des groupes, ferme le compte tricentenaire ouvert par
la formule du second \omterm{def:b3:galois:extension}{degré} et les tentatives de Cardan, Ferrari et leurs successeurs au seizième siècle.

\section{Extensions, degré, algébricité}

\begin{definition}\label{def:b3:galois:extension}
Une \emph{extension de corps}\index{extension de corps} $L/K$ est
un corps $L$ contenant $K$ comme sous-corps. Alors $L$ est un
$K$-espace vectoriel~; sa dimension, le \emph{degré}\index{degré
d'une extension} $[L:K]$, peut être finie ou infinie. Si $A
\subseteq L$, on note $K(A)$ le plus petit sous-corps de $L$
contenant $K$ et $A$~; pour $A = \{\alpha_1, \dots, \alpha_n\}$ on
écrit $K(\alpha_1, \dots, \alpha_n)$. Une extension est
\emph{\omterm{def:b3:groups:simple}{simple}} si $L = K(\alpha)$ pour un certain $\alpha$.
\end{definition}

\begin{theorem}[Tour des degrés]\label{thm:b3:galois:tower}
Si $K \subseteq M \subseteq L$ sont des corps, alors $[L:K] =
[L:M]\,[M:K]$ (avec la convention $\infty \cdot n = \infty$).
\end{theorem}

\begin{proof}
Si l'un des \omterm{def:b3:galois:extension}{degrés} est infini, le produit l'est aussi (un
$K$-espace de dimension finie ne peut contenir un sous-espace de
dimension infinie, ni un quotient de dimension infinie). Si les
deux sont finis, soit $(m_i)$ une $K$-base de $M$ et $(\ell_j)$
une $M$-base de $L$~: les produits $m_i\ell_j$ forment une
$K$-base de $L$ (engendrer~: tout $x \in L$ s'écrit $\sum
a_j\ell_j$ avec $a_j \in M$, et chaque $a_j = \sum b_{ij}m_i$~;
indépendance~: $\sum_{i,j} c_{ij}m_i\ell_j = 0$ se regroupe en
$\sum_j\bigl(\sum_i c_{ij}m_i\bigr)\ell_j = 0$, d'où chaque
coefficient $M$-linéaire s'annule, puis chaque $c_{ij}$).
\end{proof}

\begin{definition}\label{def:b3:galois:algebraic}
Soit $L/K$ une extension et $\alpha \in L$. L'élément $\alpha$
est \emph{algébrique}\index{élément algébrique} sur $K$ s'il
existe $P \in K[X]\setminus\{0\}$ avec $P(\alpha) = 0$~; sinon
il est \emph{transcendant}. L'extension $L/K$ est
\emph{algébrique} si tout élément de $L$ est algébrique sur $K$.
Le \emph{polynôme minimal}\index{polynôme minimal} $\mu_{\alpha,
K}$ (ou $\mu_\alpha$) d'un $\alpha$ algébrique est le monique
générateur de l'\omterm{def:b3:rings:ideal}{idéal} $\{P \in K[X] : P(\alpha) = 0\}$
(principal car $K[X]$ est un \omterm{def:b3:rings:pidufd}{PID})~; il est irréductible.
\end{definition}

\begin{theorem}[Extensions simples algébriques]\label{thm:b3:galois:simple}
Soit $\alpha$ \omterm{def:b3:galois:algebraic}{algébrique} sur $K$ de \omterm{def:b3:galois:extension}{degré} $n = \deg\mu_\alpha$.
Alors $K(\alpha) \cong K[X]/(\mu_\alpha)$ via $P \mapsto
P(\alpha)$, et $\{1, \alpha, \dots, \alpha^{n-1}\}$ est une
$K$-base de $K(\alpha)$~: $[K(\alpha):K] = n$. En particulier
$\alpha$ est \omterm{def:b3:galois:algebraic}{algébrique} ssi $[K(\alpha):K] < \infty$.
\end{theorem}

\begin{proof}
L'évaluation $K[X] \to K[\alpha]$, $P \mapsto P(\alpha)$, est un
morphisme d'anneaux surjectif de noyau $(\mu_\alpha)$~: le
premier théorème d'isomorphisme donne $K[\alpha] \cong
K[X]/(\mu_\alpha)$. Comme $\mu_\alpha$ est irréductible, l'\omterm{def:b3:rings:ideal}{idéal}
$(\mu_\alpha)$ est \omterm{def:b3:rings:primemaximal}{maximal} et $K[\alpha]$ est un corps~: donc
$K[\alpha] = K(\alpha)$. La base des puissances modulo un
polynôme de \omterm{def:b3:galois:extension}{degré} $n$ donne la $K$-base annoncée. Réciproquement
si $[K(\alpha):K] = n < \infty$, les $n+1$ éléments $1, \alpha,
\dots, \alpha^n$ sont $K$-liés, d'où un polynôme annulateur.
\end{proof}

\begin{corollary}\label{cor:b3:galois:algclosed}
(a) Si $[L:K] < \infty$, alors $L/K$ est \omterm{def:b3:galois:algebraic}{algébrique}.
(b) Les éléments de $L$ \omterm{def:b3:galois:algebraic}{algébriques} sur $K$ forment un sous-corps
(la \emph{\omterm{def:b3:galois:closure}{clôture algébrique} relative} de $K$ dans $L$).
(c) Si $K \subseteq M \subseteq L$ avec $M/K$ et $L/M$
\omterm{def:b3:galois:algebraic}{algébriques}, alors $L/K$ est \omterm{def:b3:galois:algebraic}{algébrique}.
\end{corollary}

\begin{proof}
(a) Pour $\alpha \in L$, $K(\alpha) \subseteq L$ a \omterm{def:b3:galois:extension}{degré} fini,
donc $\alpha$ est \omterm{def:b3:galois:algebraic}{algébrique} par le~\cref{thm:b3:galois:simple}.
(b) Si $\alpha, \beta$ sont \omterm{def:b3:galois:algebraic}{algébriques}, $K(\alpha, \beta) =
K(\alpha)(\beta)$ a \omterm{def:b3:galois:extension}{degré} fini par la tour
($[K(\alpha):K] < \infty$ et $[K(\alpha,\beta):K(\alpha)] \leq
\deg\mu_{\beta,K} < \infty$), donc tout élément de $K(\alpha,
\beta)$ --- en particulier $\alpha \pm \beta$, $\alpha\beta$,
$\alpha^{-1}$ --- est \omterm{def:b3:galois:algebraic}{algébrique}. (c) Pour $\alpha \in L$,
$\mu_{\alpha, M} = X^n + a_{n-1}X^{n-1} + \dots + a_0$ a
coefficients dans $M$, chacun \omterm{def:b3:galois:algebraic}{algébrique} sur $K$~; le corps $F =
K(a_0, \dots, a_{n-1}, \alpha)$ a \omterm{def:b3:galois:extension}{degré} fini sur $K$ (tour +
adjonction de $\alpha$), donc $\alpha$ est \omterm{def:b3:galois:algebraic}{algébrique} sur $K$.
\end{proof}

\begin{example}\label{ex:b3:galois:degrees}
$[\Q(\iu):\Q] = 2$ ($\mu = X^2 + 1$)~; $[\Q(\sqrt[3]{2}):\Q] =
3$ ($X^3 - 2$ irréductible par \omterm{thm:b3:rings:criteria}{Eisenstein})~; $[\Q(\sqrt2,
\sqrt3):\Q] = 4$ (tour~: $[\Q(\sqrt2):\Q] = 2$, et $X^2 - 3$
reste irréductible sur $\Q(\sqrt2)$ car $3$ n'y est pas un
carré). Un élément transcendant~: $\pi$ sur $\Q$ (théorème de
Lindemann, admis ici~: sa démonstration dépasse ce cours)~;
alors $\Q(\pi) \cong \Q(X)$, de \omterm{def:b3:galois:extension}{degré} infini.
\end{example}

\section{Corps de décomposition~; clôture algébrique}

\begin{theorem}[Corps de décomposition]\label{thm:b3:galois:splitting}
Soit $P \in K[X]$ non constant. Il existe une extension $L/K$,
unique à $K$-isomorphisme près, dans laquelle $P$ se scinde en
facteurs linéaires et qui est engendrée sur $K$ par les racines
de $P$~: le \emph{corps de décomposition}\index{corps de
décomposition} de $P$ sur $K$. De plus $[L:K] \leq (\deg P)!$.
\end{theorem}

\begin{proof}
\emph{Existence.} Récurrence sur $n = \deg P$. Factoriser $P =
P_1\cdots P_r$ en irréductibles~; adjoindre une racine $\alpha$
de $P_1$ via $K(\alpha) \cong K[X]/(P_1)$~; dans $K(\alpha)[X]$,
$P = (X - \alpha)Q$, et le \omterm{thm:b3:galois:splitting}{corps de décomposition} de $Q$ sur
$K(\alpha)$ (récurrence) est un \omterm{thm:b3:galois:splitting}{corps de décomposition} de $P$ sur
$K$.

\emph{Unicité.} Soient $L, L'$ deux \omterm{thm:b3:galois:splitting}{corps de décomposition}, $P =
c\prod(X - \alpha_i)$ dans $L$ et $P = c\prod(X - \alpha_i')$
dans $L'$. Un $K$-plongement $K(\alpha_1) \to L'$ envoyant
$\alpha_1$ sur une racine $\alpha_j'$ de $P$ dans $L'$ (il en
existe car $P$ a une racine dans $L'$) s'étend, par récurrence
sur le nombre de racines, en un $K$-isomorphisme $L \to L'$
(les deux sont engendrés par les racines, et le plongement envoie
racines sur racines).

\emph{Borne de \omterm{def:b3:galois:extension}{degré}.} À chaque étape d'adjonction d'une racine
d'un facteur irréductible de \omterm{def:b3:galois:extension}{degré} $d$, le \omterm{def:b3:galois:extension}{degré} croît d'un
facteur $d \leq n$~; en $n$ étapes, $[L:K] \leq n!$.
\end{proof}

\begin{definition}\label{def:b3:galois:closure}
Un corps $K$ est \emph{algébriquement clos}\index{corps
algébriquement clos} si tout $P \in K[X]$ non constant a une
racine dans $K$ (équivalemment~: se scinde en linéaires dans
$K[X]$). Une \emph{clôture algébrique}\index{clôture algébrique}
de $K$ est une extension \omterm{def:b3:galois:algebraic}{algébrique} algébriquement close de $K$.
\end{definition}

\begin{theorem}[Steinitz]\label{thm:b3:galois:steinitz}
Tout corps admet une \omterm{def:b3:galois:closure}{clôture algébrique}, unique à isomorphisme
près.\index{théorème de Steinitz}
\end{theorem}

\begin{proof}
\emph{Existence (construction d'Artin).} Soit $R = K[(X_f)_f]$
l'anneau des polynômes en une indéterminée $X_f$ par monique non
constant $f \in K[X]$, et $I$ l'\omterm{def:b3:rings:ideal}{idéal} engendré par tous les
$f(X_f)$. $I$ est propre~: une relation $1 = \sum_{i=1}^r g_i\,
f_i(X_{f_i})$ n'implique qu'un nombre fini de polynômes~; dans
un \omterm{thm:b3:galois:splitting}{corps de décomposition} commun $E$ de $f_1\cdots f_r$
choisir des racines $\alpha_i$ de $f_i$ et évaluer $X_{f_i}
\mapsto \alpha_i$ (les autres variables $\mapsto 0$)~: $1 =
0$, absurde. Soit $\mathfrak m \supseteq I$ \omterm{def:b3:rings:primemaximal}{maximal}
(\cref{thm:b3:rings:krull}~; Zorn) et $K_1 = R/\mathfrak m$~:
une \omterm{def:b3:galois:extension}{extension de corps} de $K$ dans laquelle tout $f \in K[X]$
non constant a une racine, à savoir $\bar X_f$, et qui est
\omterm{def:b3:galois:algebraic}{algébrique} sur $K$ (engendrée par les $\bar X_f$, chacun
\omterm{def:b3:galois:algebraic}{algébrique}). Itérer~: $K \subseteq K_1 \subseteq K_2 \subseteq
\cdots$, où $K_{n+1}$ fait à $K_n$ ce que $K_1$ a fait à $K$,
et poser $\Omega = \bigcup_n K_n$, un corps. Tout $g \in
\Omega[X]$ non constant a ses coefficients finiment nombreux
dans un certain $K_n$~; un facteur irréductible de $g$ sur
$K_n$ a une racine dans $K_{n+1} \subseteq \Omega$~: $\Omega$
est \omterm{def:b3:galois:closure}{algébriquement clos}, et \omterm{def:b3:galois:algebraic}{algébrique} sur $K$ (chaque $K_n$
l'est, par transitivité, \cref{cor:b3:galois:algclosed})~:
$\Omega$ est une \omterm{def:b3:galois:closure}{clôture algébrique}.

\emph{Unicité.} Soient $\Omega, \Omega'$ deux \omterm{def:b3:galois:closure}{clôtures
algébriques}. Considérer l'ensemble des paires $(E, \tau)$ où
$K \subseteq E \subseteq \Omega$ et $\tau\colon E \to
\Omega'$ est un $K$-plongement, ordonné par extension~; il est
non vide ($(K, \mathrm{id})$) et inductif (réunion d'une
chaîne), donc Zorn donne un \omterm{def:b3:rings:primemaximal}{maximal} $(E_0, \tau_0)$. Si $E_0
\neq \Omega$, choisir $\alpha \in \Omega\setminus E_0$~:
$\pi_{\alpha/E_0}$ s'envoie sur un polynôme sur $\tau_0(E_0)$
ayant une racine $\beta$ dans l'\omterm{def:b3:galois:closure}{algébriquement clos}
$\Omega'$, et $\tau_0$ s'étend à $E_0(\alpha) \to \Omega'$
($\alpha \mapsto \beta$), contredisant la maximalité. Donc il
existe un $K$-plongement $\tau \colon \Omega \to \Omega'$~;
son image, isomorphe à $\Omega$, est algébriquement close, et
$\Omega'$ est \omterm{def:b3:galois:algebraic}{algébrique} sur elle~: pour $x \in \Omega'$,
$\pi_{x/\tau(\Omega)}$ se scinde sur $\tau(\Omega)$, donc $x
\in \tau(\Omega)$. Ainsi $\tau$ est surjectif~: un
isomorphisme.
\end{proof}

\begin{remark}\label{rem:b3:galois:closureexamples}
$\C$ est \omterm{def:b3:galois:closure}{algébriquement clos} (d'Alembert--Gauss, démontré dans
le~\cref{ch:b3:holomorphic}). Une \omterm{def:b3:galois:closure}{clôture algébrique} $\overline
\Q$ de $\Q$ s'identifie au sous-corps de $\C$ des nombres
\omterm{def:b3:galois:algebraic}{algébriques}~; elle est dénombrable, de \omterm{def:b3:galois:extension}{degré} infini sur $\Q$.
On fixe désormais, quand c'est commode, une \omterm{def:b3:galois:closure}{clôture algébrique}
$\overline K$ de $K$ et l'on travaille à l'intérieur~: tous les
\omterm{thm:b3:galois:splitting}{corps de décomposition} et toutes les extensions \omterm{def:b3:galois:algebraic}{algébriques}
considérés seront implicitement des sous-corps de $\overline K$.
\end{remark}

\section{Corps finis}

\begin{theorem}\label{thm:b3:galois:finitefields}
Soit $p$ premier, $n \geq 1$, $q = p^n$.
\begin{enumerate}
  \item Un \omterm{thm:b3:galois:finitefields}{corps fini} a pour cardinal une puissance de premier,
        et pour chaque $q$ il existe exactement un corps
        $\mathbb F_q$ à $q$ éléments à isomorphisme près~: le
        \omterm{thm:b3:galois:splitting}{corps de décomposition} de $X^q - X$ sur $\mathbb
        F_p$.\index{corps fini}
  \item Le \emph{Frobenius}\index{endomorphisme de Frobenius}
        $F \colon x \mapsto x^p$ est un automorphisme de
        $\mathbb F_q$, et le groupe des automorphismes de
        $\mathbb F_q$ est cyclique d'ordre $n$, engendré par
        $F$.
  \item $\mathbb F_{p^m}$ se plonge dans $\mathbb F_{p^n}$ ssi
        $m \mid n$.
\end{enumerate}
\end{theorem}

\begin{proof}
(1) Un \omterm{thm:b3:galois:finitefields}{corps fini} $E$ a caractéristique $p > 0$ et est un
$\mathbb F_p$-espace vectoriel de dimension finie~: $\abs E =
p^n$. Son groupe multiplicatif a ordre $q - 1$, donc tout $x
\in E$ vérifie $x^q = x$~: $E$ consiste en $q$ racines de $X^q
- X$, d'où c'est un \omterm{thm:b3:galois:splitting}{corps de décomposition} de celui-ci sur
$\mathbb F_p$ --- déterminant $E$ à isomorphisme près
(\cref{thm:b3:galois:splitting}). Réciproquement, dans un
\omterm{thm:b3:galois:splitting}{corps de décomposition} $L$ de $X^q - X$, l'ensemble $E$ de ses
racines est un \emph{sous-corps}~: $(x + y)^q = x^q + y^q$ en
itérant le rêve du débutant $(a+b)^p = a^p + b^p$ ($p \mid
\binom pk$), et $(xy)^q = x^qy^q$, $(x^{-1})^q = (x^q)^{-1}$~;
il a exactement $q$ éléments car $X^q - X$ est \omterm{def:b3:galois:separable}{séparable}~: sa
dérivée est $qX^{q-1} - 1 = -1$ (car $p \mid q$), première
avec lui, donc pas de racines multiples. Ainsi $L = E$ a $q$
éléments.

(2) $F$ est un morphisme de corps (même rêve du débutant) et
injectif (noyau trivial), donc un automorphisme sur le fini
$\mathbb F_q$. Les points fixes de $F^k$ sont les racines de
$X^{p^k} - X$, au nombre de $p^k$~: $F$ a ordre exactement $n$
dans $\operatorname{Aut}(\mathbb F_q)$. Or $\abs{\operatorname
{Aut}(\mathbb F_q)} \leq n$ (un automorphisme est déterminé
par l'image d'un générateur primitif de l'extension de \omterm{def:b3:galois:extension}{degré}
$n$ sur $\mathbb F_p$)~: le groupe est exactement $\langle F
\rangle$, cyclique d'ordre $n$.

(3) $\mathbb F_{p^m} \subseteq \mathbb F_{p^n}$ ssi $X^{p^m} -
X$ divise $X^{p^n} - X$ ssi $m \mid n$ (critère de divisibilité
des polynômes $X^{p^a} - X$~: écrire $n = mq + r$ et comparer
les cardinaux des corps fixes de $F^m$ et $F^n$).
\end{proof}

\begin{theorem}[Cyclicité de $\mathbb F_q^\times$]\label{thm:b3:galois:cyclic}
Le groupe multiplicatif d'un \omterm{thm:b3:galois:finitefields}{corps fini} est cyclique.\index{corps fini, cyclicité}
\end{theorem}

\begin{proof}
Soit $G = \mathbb F_q^\times$, d'ordre $q - 1$. Par le théorème
de structure (\cref{cor:b3:modules:abelian}), $G \cong
\Z/d_1\Z \times \cdots \times \Z/d_s\Z$ avec $d_1 \mid \dots
\mid d_s$. L'exposant de $G$ est $d_s$~: tout élément vérifie
$x^{d_s} = 1$. Mais le polynôme $X^{d_s} - 1$ a au plus $d_s$
racines dans un corps, et $G$ en fournit $q - 1 = |G| =
d_1\cdots d_s$~; donc $d_1\cdots d_s \leq d_s$, forçant $s = 1$
et $G$ cyclique. (C'est le critère de \cref{exo:b3:modules:11}.)
\end{proof}

\begin{example}\label{ex:b3:galois:f8}
$\mathbb F_8 = \mathbb F_2[X]/(X^3 + X + 1)$ (le polynôme est
irréductible sur $\mathbb F_2$~: pas de racine). Si $\omega =
\bar X$, alors $\omega^3 = \omega + 1$, et $\mathbb F_8^\times =
\langle \omega \rangle$ d'ordre $7$. Les sous-corps de $\mathbb
F_{p^n}$ sont exactement les $\mathbb F_{p^d}$ pour $d \mid n$
(car $\mathbb F_{p^d} \subseteq \mathbb F_{p^n}$ ssi $X^{p^d} -
X$ divise $X^{p^n} - X$ ssi $d \mid n$).

Plus généralement, l'unique sous-corps de $\mathbb F_{q}$ de
cardinal $p^d$ (quand $d \mid n$ et $q = p^n$) est le corps fixe
de $\varphi^d$, où $\varphi$ est le \omterm{thm:b3:galois:finitefields}{Frobenius} $x \mapsto x^p$~;
le \omterm{def:b3:galois:galois}{groupe de Galois} $\operatorname{Gal}(\mathbb F_{p^n}/\mathbb
F_p) = \langle \varphi \rangle$ est cyclique d'ordre $n$, ce
qui rend la \omterm{thm:b3:galois:fundamental}{correspondance de Galois} particulièrement
transparente dans le cas fini~: un unique sous-corps par
diviseur de $n$.
\end{example}

\section{Séparabilité et plongements}

\begin{definition}\label{def:b3:galois:separable}
Un polynôme irréductible $P \in K[X]$ est
\emph{séparable}\index{polynôme séparable} s'il n'a pas de
racine multiple dans une \omterm{def:b3:galois:closure}{clôture algébrique} (équivalemment $P'
\neq 0$, ou $\gcd(P, P') = 1$). Un \omterm{def:b3:galois:algebraic}{élément algébrique} est
séparable si son \omterm{def:b3:galois:algebraic}{polynôme minimal} l'est~; une extension
\omterm{def:b3:galois:algebraic}{algébrique} est séparable si tout élément l'est.
\end{definition}

\begin{proposition}\label{prop:b3:galois:perfect}
Sur un corps de caractéristique $0$, tout irréductible est
\omterm{def:b3:galois:separable}{séparable}. Sur un \omterm{thm:b3:galois:finitefields}{corps fini}, aussi. Plus généralement, un corps
est \emph{parfait}\index{corps parfait} (tout irréductible est
\omterm{def:b3:galois:separable}{séparable}) ssi il est de caractéristique $0$ ou, en
caractéristique $p$, si le \omterm{thm:b3:galois:finitefields}{Frobenius} $x \mapsto x^p$ est
surjectif.
\end{proposition}

\begin{proof}
Si $P$ irréductible a une racine multiple, alors $\gcd(P, P') =
P$ (car $\gcd$ est non trivial et $P$ irréductible), donc $P'
= 0$. En caractéristique $0$, $P' = 0$ force $P$ constant. En
caractéristique $p$, $P' = 0$ ssi $P \in K[X^p]$~: $P =
\sum a_i X^{ip}$. Si le \omterm{thm:b3:galois:finitefields}{Frobenius} est surjectif, $a_i = b_i^p$
et $P = (\sum b_i X^i)^p$, contredisant l'irréductibilité sauf
si $\deg = 0$. Les \omterm{thm:b3:galois:finitefields}{corps finis} ont un \omterm{thm:b3:galois:finitefields}{Frobenius} bijectif (\omterm{thm:b3:galois:finitefields}{corps
fini}). Réciproquement, si $a \in K$ n'est pas une puissance
$p$-ième, $X^p - a$ est irréductible (ou une puissance d'un
irréductible inséparable)~: $K$ n'est pas \omterm{prop:b3:galois:perfect}{parfait}.
\end{proof}

\begin{proposition}[Nombre de plongements]\label{prop:b3:galois:embeddings}
Soit $L = K(\alpha_1, \dots, \alpha_n)$ une extension finie
\omterm{def:b3:galois:separable}{séparable} de $K$, et $\overline K$ une \omterm{def:b3:galois:closure}{clôture algébrique}. Le
nombre de $K$-plongements $L \hookrightarrow \overline K$ égale
$[L:K]$.
\end{proposition}

\begin{proof}
Récurrence sur $n$. Pour $L = K(\alpha)$ \omterm{def:b3:galois:separable}{séparable} de \omterm{def:b3:galois:extension}{degré} $d$,
les $K$-plongements $L \hookrightarrow \overline K$
correspondent bijectivement aux racines de $\mu_\alpha$ dans
$\overline K$~: il y en a $d = [L:K]$ (séparabilité~: $d$
racines distinctes, chacune définit un unique plongement par
$\alpha \mapsto$ cette racine). En général, $L = M(\alpha_n)$
avec $M = K(\alpha_1, \dots, \alpha_{n-1})$~; par récurrence
$M$ a $[M:K]$ plongements dans $\overline K$. Chacun de ces
plongements $\tau \colon M \to \overline K$ s'étend en
$[L:M]$ plongements de $L$~: appliquer le cas \omterm{def:b3:groups:simple}{simple} à
$\mu_{\alpha_n, M}$ sur le corps $\tau(M)$, qui reste
\omterm{def:b3:galois:separable}{séparable} (même \omterm{def:b3:galois:algebraic}{polynôme minimal}, ou conjugué). Total~:
$[L:M][M:K] = [L:K]$ par la \omterm{thm:b3:galois:tower}{tour des degrés}.
\end{proof}

\begin{theorem}[Élément primitif]\label{thm:b3:galois:primitive}
Toute extension finie \omterm{def:b3:galois:separable}{séparable} est \omterm{def:b3:groups:simple}{simple}~: $L = K(\alpha)$
pour un certain $\alpha$ (un \emph{élément
primitif}\index{élément primitif}).
\end{theorem}

\begin{proof}
Si $K$ est infini~: il suffit de traiter $L = K(\alpha, \beta)$.
Soit $\mu_\alpha, \mu_\beta$ les minimaux, de racines $\alpha =
\alpha_1, \dots, \alpha_r$ et $\beta = \beta_1, \dots,
\beta_s$ dans $\overline K$ (distinctes par séparabilité).
Choisir $c \in K$ évitant les finiement nombreuses valeurs
$(\alpha_i - \alpha_1)/(\beta_1 - \beta_j)$ pour $i \geq 1$, $j
\geq 2$~; poser $\gamma = \alpha + c\beta$. Alors $K(\gamma)
\subseteq K(\alpha, \beta)$, et $\beta$ est la seule racine
commune de $\mu_\beta(Y)$ et $\mu_\alpha(\gamma - cY)$ (toute
autre $\beta_j$ donnerait $\gamma - c\beta_j = \alpha_i$ pour un
$i$, d'où $c = (\alpha_i - \alpha)/(\beta - \beta_j)$ interdit)~:
donc $\mu_{\beta, K(\gamma)}$ divise un polynôme de \omterm{def:b3:galois:extension}{degré} $1$,
$\beta \in K(\gamma)$, $\alpha = \gamma - c\beta \in
K(\gamma)$. Si $K$ est fini, $L$ aussi, et $L^\times$ est
cyclique (\cref{thm:b3:galois:cyclic})~: un générateur $\alpha$
vérifie $L = K(\alpha)$.
\end{proof}

\section{La correspondance de Galois}

\begin{definition}\label{def:b3:galois:galois}
Une extension finie $L/K$ est \emph{galoisienne}\index{extension
galoisienne} (ou \emph{normale et \omterm{def:b3:galois:separable}{séparable}}) si elle est
\omterm{def:b3:galois:separable}{séparable} et si tout polynôme irréductible de $K[X]$ ayant une
racine dans $L$ se scinde entièrement dans $L$ (normalité). Le
\emph{groupe de Galois}\index{groupe de Galois} est
\[
\operatorname{Gal}(L/K) = \{\sigma \colon L \to L : \sigma
\text{ automorphisme, } \sigma|_K = \mathrm{id}\}.
\]
\end{definition}

\begin{proposition}\label{prop:b3:galois:galoisorder}
Pour $L/K$ finie \omterm{def:b3:galois:separable}{séparable}, les conditions suivantes sont
équivalentes~: (i) $L/K$ est \omterm{def:b3:galois:galois}{galoisienne}~; (ii) $L$ est le \omterm{thm:b3:galois:splitting}{corps
de décomposition} d'un \omterm{def:b3:galois:separable}{polynôme séparable} sur $K$~; (iii) $|\operatorname{Gal}(L/K)| = [L:K]$~; (iv) le corps fixe
$L^G = \{x \in L : \sigma(x) = x\ \forall \sigma \in G\}$ égale
$K$, où $G = \operatorname{Gal}(L/K)$.
\end{proposition}

\begin{proof}
(i)$\Rightarrow$(ii)~: par l'élément primitif, $L = K(\alpha)$~;
$\mu_\alpha$ est \omterm{def:b3:galois:separable}{séparable} et se scinde dans $L$ (normalité,
une racine $\alpha \in L$), et $L$ est engendré par les racines
(les conjugués de $\alpha$ engendrent la même extension).
(ii)$\Rightarrow$(iii)~: tout $K$-plongement $L \hookrightarrow
\overline K$ envoie racines sur racines, donc stabilise $L$~:
c'est un élément de $\operatorname{Gal}(L/K)$. Par
la~\cref{prop:b3:galois:embeddings} il y en a $[L:K]$.
(iii)$\Rightarrow$(iv)~: $K \subseteq L^G$ toujours~; $L/L^G$
est \omterm{def:b3:galois:galois}{galoisienne} de groupe $G$ (tout $\sigma \in G$ fixe $L^G$),
donc $|G| = [L:L^G]$ par le sens (ii)$\Rightarrow$(iii) déjà
prouvé pour le \omterm{thm:b3:galois:splitting}{corps de décomposition}~--- plus proprement~: par
Artin ci-dessous, $[L:L^G] = |G|$~; donc si $|G| = [L:K]$ alors
$L^G = K$.
(iv)$\Rightarrow$(i)~: séparabilité de $L/K$~: pour $\alpha \in
L$, le polynôme $\prod_{\sigma \in G/\operatorname{Stab}(\alpha)}
(X - \sigma\alpha)$ est dans $K[X]$ (coefficients fixes par $G$,
donc dans $L^G = K$), annule $\alpha$, et est \omterm{def:b3:galois:separable}{séparable}~; donc
$\mu_\alpha$ est \omterm{def:b3:galois:separable}{séparable}. Normalité~: un $K$-plongement de $L$
dans $\overline K$ envoie $\alpha$ sur un conjugué $\sigma\alpha
\in L$, donc stabilise $L$.
\end{proof}

\begin{lemma}[Artin]\label{lem:b3:galois:artin}
Soit $G$ un groupe fini d'automorphismes d'un corps $L$, et $K =
L^G$. Alors $L/K$ est \omterm{def:b3:galois:galois}{galoisienne} de groupe $G$, et $[L:K] =
|G|$.\index{lemme d'Artin}
\end{lemma}

\begin{proof}
Pour $\alpha \in L$, le polynôme $P = \prod_{\sigma \in
G/\operatorname{Stab}(\alpha)}(X - \sigma\alpha)$ est dans $K[X]$
(les coefficients sont $G$-invariants), monique, \omterm{def:b3:galois:separable}{séparable}, et
annule $\alpha$~: $L/K$ est \omterm{def:b3:galois:algebraic}{algébrique} \omterm{def:b3:galois:separable}{séparable}. De plus $L$ est
normale sur $K$ (tout $K$-plongement envoie $\alpha$ sur un
conjugué $\sigma\alpha \in L$). Ainsi $L/K$ est \omterm{def:b3:galois:galois}{galoisienne}. Or
$G \subseteq \operatorname{Gal}(L/K)$, donc $|G| \leq
|\operatorname{Gal}(L/K)| = [L:K]$.

Pour l'autre inégalité, on invoque l'indépendance des caractères
de Dedekind~: des morphismes de groupes distincts $\chi_i \colon
L^\times \to L^\times$ sont $L$-linéairement indépendants comme
applications $L \to L$. En particulier les $|G|$
automorphismes de $G$ sont $K$-linéairement indépendants, donc
$\dim_K L \geq |G|$. Combinant les deux inégalités~: $[L:K] =
|G|$ et $G = \operatorname{Gal}(L/K)$.
\end{proof}

\begin{theorem}[Correspondance fondamentale de Galois]\label{thm:b3:galois:fundamental}
Soit $L/K$ \omterm{def:b3:galois:galois}{galoisienne} finie de groupe $G = \operatorname{Gal}(L/K)$.
Il y a une bijection
\[
\{\text{sous-corps } K \subseteq M \subseteq L\}
\;\longleftrightarrow\;
\{\text{sous-groupes } H \leq G\},
\]
donnée par $M \mapsto \operatorname{Gal}(L/M)$ et $H \mapsto
L^H$, inverses l'une de l'autre. De plus~:
\begin{enumerate}
  \item $H_1 \subseteq H_2 \iff L^{H_1} \supseteq L^{H_2}$
        (renversement des inclusions)~;
  \item $[L:L^H] = |H|$ et $[L^H:K] = [G:H]$~;
  \item $H$ est \omterm{def:b3:groups:normal}{normal} dans $G$ ssi $L^H/K$ est \omterm{def:b3:galois:galois}{galoisienne}, et
        alors $\operatorname{Gal}(L^H/K) \cong G/H$.
\end{enumerate}
\index{correspondance de Galois}
\end{theorem}

\begin{proof}
Pour $H \leq G$, le lemme d'Artin donne $\operatorname{Gal}(L/L^H)
= H$ et $[L:L^H] = |H|$. Pour un sous-corps $M$, $L/M$ est
\omterm{def:b3:galois:galois}{galoisienne}~: \omterm{def:b3:galois:separable}{séparable} (sous-extension d'une \omterm{def:b3:galois:separable}{séparable}) et
normale (tout $M$-plongement de $L$ dans $\overline K$ est un
$K$-plongement, donc un élément de $G$, qui stabilise $L$). Le
groupe $\operatorname{Gal}(L/M) \leq G$ a pour corps fixe $M$
par la~\cref{prop:b3:galois:galoisorder}(iv). Les deux
applications sont donc mutuellement inverses.

(1) est clair par construction. (2)~: $[L:L^H] = |H|$ et la
\omterm{thm:b3:galois:tower}{tour des degrés} donnent $[L^H:K] = [L:K]/[L:L^H] = |G|/|H| =
[G:H]$. (3)~: pour $\sigma \in G$, le conjugué $\sigma(L^H) =
L^{\sigma H\sigma^{-1}}$. Donc $H$ est \omterm{def:b3:groups:normal}{normal} dans $G$ ssi
$\sigma(L^H) = L^H$ pour tout $\sigma$ ssi tout $K$-plongement
de $L^H$ (restriction d'un élément de $G$) stabilise $L^H$ ssi
$L^H/K$ est normale (la séparabilité est automatique). Alors
le morphisme de restriction $G \to \operatorname{Gal}(L^H/K)$
est surjectif (tout automorphisme de $L^H/K$ s'étend à $L$ car
$L/L^H$ est \omterm{def:b3:galois:galois}{galoisienne}) de noyau $\operatorname{Gal}(L/L^H) =
H$, d'où $\operatorname{Gal}(L^H/K) \cong G/H$.
\end{proof}

\section{Extensions cyclotomiques}

\begin{definition}\label{def:b3:galois:cyclotomic}
Soit $n \geq 1$. Le $n$-ième \emph{polynôme
cyclotomique}\index{polynôme cyclotomique} est
\[
\Phi_n(X) = \prod_{\substack{1 \leq k \leq n\\ \gcd(k,n)=1}}
\bigl(X - \eu^{2\iu\pi k/n}\bigr) \in \C[X].
\]
Le $n$-ième \emph{corps cyclotomique} est $\Q(\zeta_n)$ où
$\zeta_n = \eu^{2\iu\pi/n}$.
\end{definition}

\begin{theorem}\label{thm:b3:galois:cyclotomicirred}
$\Phi_n \in \Z[X]$ est irréductible sur $\Q$, de \omterm{def:b3:galois:extension}{degré}
$\varphi(n)$. L'extension $\Q(\zeta_n)/\Q$ est \omterm{def:b3:galois:galois}{galoisienne} de
groupe $(\Z/n\Z)^\times$, via $\sigma_k \colon \zeta_n \mapsto
\zeta_n^k$ pour $\gcd(k,n) = 1$.\index{extension cyclotomique}
\end{theorem}

\begin{proof}
$X^n - 1 = \prod_{d \mid n} \Phi_d(X)$ (partition des racines
selon l'ordre). Par récurrence de Möbius, $\Phi_n \in \Z[X]$
(diviser dans $\Z[X]$ par les $\Phi_d$ unitaires pour $d < n$).
Irréductibilité~: soit $P$ le facteur irréductible unitaire de
$\Phi_n$ dans $\Z[X]$ annulant $\zeta = \zeta_n$, et $q$ un
premier ne divisant pas $n$. On montre que $\zeta^q$ est aussi
racine de $P$. Sinon $P$ et $\Phi_n$ donneraient, modulo $q$, un
facteur commun avec $X^n - 1$ et sa dérivée $nX^{n-1}$ (via
$(\zeta^q$ racine de $\Phi_n/P$), d'où une racine multiple de
$X^n - 1$ mod $q$, impossible car $q \nmid n$. Donc tous les
$\zeta^k$ avec $\gcd(k,n)=1$ (produits de tels premiers) sont
racines de $P$~: $P = \Phi_n$.

Galois~: $\Q(\zeta_n)$ est le \omterm{thm:b3:galois:splitting}{corps de décomposition} de $X^n -
1$ (\omterm{def:b3:galois:separable}{séparable}, $nX^{n-1}$ premier à $X^n-1$ en caractéristique
$0$). Le \omterm{def:b3:galois:galois}{groupe de Galois} s'injecte dans $(\Z/n\Z)^\times$ via
$\sigma(\zeta_n) = \zeta_n^{k(\sigma)}$, et l'ordre
$[\Q(\zeta_n):\Q] = \deg\Phi_n = \varphi(n) =
|(\Z/n\Z)^\times|$ force la surjectivité.
\end{proof}

\section{Règle et compas}

\begin{definition}\label{def:b3:galois:constructible}
Identifier le plan à $\C$~; partir de $\{0, 1\}$. Un point est
\emph{constructible}\index{nombre constructible} s'il s'obtient
par un nombre fini d'intersections de droites passant par deux
points déjà construits et de cercles centrés en un point
construit de rayon une distance de deux points construits.
\end{definition}

\begin{theorem}[Wantzel]\label{thm:b3:galois:wantzel}
$z \in \C$ est \omterm{def:b3:galois:constructible}{constructible} ssi il existe une tour $\Q = F_0
\subseteq F_1 \subseteq \dots \subseteq F_r$ avec $[F_{i+1} :
F_i] = 2$ et $z \in F_r$. En particulier, un \omterm{def:b3:galois:constructible}{nombre
constructible} est \omterm{def:b3:galois:algebraic}{algébrique} de \omterm{def:b3:galois:extension}{degré} une puissance de $2$ sur
$\Q$.\index{théorème de Wantzel}
\end{theorem}

\begin{proof}
($\Rightarrow$) Les coordonnées de l'intersection de deux droites
passant par des points à coordonnées dans un sous-corps $F
\subseteq \R$ résolvent un système linéaire sur $F$~: elles
restent dans $F$. Les intersections droite--cercle et
cercle--cercle conduisent, après élimination de la partie
linéaire (soustraire les deux équations de cercles donne une
droite), à une équation quadratique sur $F$~: les nouvelles
coordonnées vivent dans $F$ ou dans $F(\sqrt d)$ pour un $d \in
F$, $d > 0$. Par récurrence, tout point construit a ses
coordonnées dans une tour d'extensions quadratiques de $\Q$~; et
$z = x + \iu y$ est dans une tour quadratique aussi (adjoindre
$\iu$~: une étape quadratique de plus). Conséquence sur le
\omterm{def:b3:galois:extension}{degré}~: $[\Q(z):\Q]$ divise $[F_r : \Q] = 2^r$ (\omterm{thm:b3:galois:tower}{loi de la tour}).

($\Leftarrow$) Les \omterm{def:b3:galois:constructible}{nombres constructibles} forment un corps~: sommes
et différences par parallélogrammes (les parallèles sont
\omterm{def:b3:galois:constructible}{constructibles}~: abaisser et relever deux perpendiculaires ---
la perpendiculaire classique par un point utilise un cercle et
deux arcs)~; produits et quotients par des configurations
d'intercept de Thalès (données les longueurs $a, b$, construire
$ab$ et $a/b$ avec des triangles semblables sur deux rayons).
Et le corps est stable par racines carrées~: pour $a > 0$, le
cercle de diamètre $1 + a$ et la perpendiculaire au point de
jonction se rencontrent à hauteur $\sqrt a$ (relation
altitude--moyenne géométrique dans un triangle rectangle)~; pour
un complexe $w = \rho\eu^{\iu\theta}$, construire $\sqrt\rho$ et
bissecter $\theta$ (la bissection d'angle est une construction
au compas). Les parties réelle et imaginaire des membres d'une
tour quadratique sont donc \omterm{def:b3:galois:constructible}{constructibles} par récurrence sur
la tour~: chaque étape adjoint des racines d'une quadratique,
exprimables par opérations de corps et une racine carrée d'un
nombre déjà construit (la formule quadratique~; en
caractéristique $0$).
\end{proof}

\begin{corollary}\label{cor:b3:galois:impossible}
Les trois problèmes classiques sont insolubles à la règle et au
compas~\index{règle et compas}~:
\begin{enumerate}
  \item \emph{Duplication du cube}~: $\sqrt[3]2$ a \omterm{def:b3:galois:extension}{degré} $3$,
        non puissance de $2$.
  \item \emph{Trisection de l'angle}~: trisecter $60^\circ$
        demande $\cos 20^\circ$, racine de l'irréductible $8X^3
        - 6X - 1$~: \omterm{def:b3:galois:extension}{degré} $3$.
  \item \emph{Quadrature du cercle}~: $\sqrt\pi$ est
        transcendant ($\pi$ l'est --- théorème de Lindemann,
        admis ici~: sa preuve relève d'un cours de théorie de
        la transcendance).
\end{enumerate}
De plus, le $n$-gone régulier est \omterm{def:b3:galois:constructible}{constructible} ssi
$\varphi(n)$ est une puissance de $2$ (Gauss--Wantzel~; le
«~si~» utilise la méthode du problème du week-end, le «~seulement
si~» est le~\cref{thm:b3:galois:wantzel} appliqué à $\zeta_n$,
de \omterm{def:b3:galois:extension}{degré} $\varphi(n)$). Pour $n = 7$~: $\varphi(7) = 6$~: le
heptagone régulier est impossible~; pour $n = 17$~: $\varphi(17)
= 16 = 2^4$~: \omterm{def:b3:galois:constructible}{constructible} --- le problème du week-end le
construit.
\end{corollary}

\begin{proof}
(1) $X^3 - 2$ est irréductible (\omterm{thm:b3:rings:criteria}{Eisenstein}). (2) De $\cos
3\theta = 4\cos^3\theta - 3\cos\theta$ avec $3\theta =
60^\circ$~: $8c^3 - 6c = 1$ pour $c = \cos 20^\circ$~; le cubique
$8X^3 - 6X - 1$ n'a pas de racine rationnelle (les candidats
$\pm1, \pm\frac12, \pm\frac14, \pm\frac18$ échouent), donc est
irréductible~: \omterm{def:b3:galois:extension}{degré} $3$. Un angle de $60^\circ$ général est
\omterm{def:b3:galois:constructible}{constructible}, donc un trisecteur construirait $c$. (3) Si
$\sqrt\pi$ était \omterm{def:b3:galois:constructible}{constructible}, il serait \omterm{def:b3:galois:algebraic}{algébrique}, donc aussi
$\pi$. L'énoncé sur le $n$-gone~: le \omterm{def:b3:galois:extension}{degré} de $\zeta_n$ est
$\varphi(n)$ (\cref{thm:b3:galois:cyclotomicirred})~; la
nécessité suit de Wantzel~; pour la suffisance, le \omterm{def:b3:galois:galois}{groupe de
Galois}, abélien d'ordre $2^m$, admet une chaîne de sous-groupes
d'indice $2$ (un $2$-groupe fini le fait~:
\cref{exo:b3:groups:10}), dont les corps fixes forment une tour
quadratique aboutissant à $\Q(\zeta_n)$
(\cref{thm:b3:galois:fundamental})~; conclure par
le~\cref{thm:b3:galois:wantzel}.
\end{proof}

\section{Résolubilité par radicaux}

\begin{definition}\label{def:b3:galois:radical}
Une extension $L/K$ est \emph{radicale}\index{extension radicale}
s'il existe une tour $K = K_0 \subseteq K_1 \subseteq \dots
\subseteq K_m = L$ avec $K_{i+1} = K_i(\sqrt[n_i]{a_i})$ pour
certains $a_i \in K_i$, $n_i \geq 1$. Un polynôme $P \in K[X]$
est \emph{\omterm{def:b3:groups:derived}{résoluble} par radicaux} si son \omterm{thm:b3:galois:splitting}{corps de décomposition}
est \omterm{def:b3:rings:content}{contenu} dans une extension radicale de $K$.
\end{definition}

\begin{lemma}\label{lem:b3:galois:cyclicsteps}
Soit $K$ contenant une racine primitive $n$-ième de l'unité
$\zeta$, i.e.\ $\zeta$ d'ordre $n$ dans $K^\times$, et $a \in
K^\times$. Alors $K(\sqrt[n]a)/K$ est \omterm{def:b3:galois:galois}{galoisienne} de groupe
\emph{cyclique}. Réciproquement --- non utilisée ci-dessous ---
toute extension cyclique de \omterm{def:b3:galois:extension}{degré} $n$ est de cette forme. De
plus $K(\zeta_n)/K$ est \omterm{def:b3:galois:galois}{galoisienne} de groupe \emph{abélien},
pour tout $K$ de caractéristique $0$.
\end{lemma}

\begin{proof}
$X^n - a$ est \omterm{def:b3:galois:separable}{séparable} ($\gcd$ avec $nX^{n-1}$~: $a \neq 0$)
et se scinde dans $K(\alpha)$, $\alpha^n = a$~: ses racines
sont les $\zeta^k\alpha \in K(\alpha)$. Donc $K(\alpha)/K$ est
\omterm{def:b3:galois:galois}{galoisienne}~; l'application $\sigma \mapsto
\sigma(\alpha)/\alpha \in \mu_n = \langle \zeta\rangle$ est un
morphisme injectif ($\sigma\tau(\alpha) =
\sigma(\tau(\alpha)/\alpha \cdot \alpha) =
(\sigma(\alpha)/\alpha)\,(\tau(\alpha)/\alpha)\,\alpha$), donc
$G$ s'injecte dans un groupe cyclique~: il est cyclique.
L'extension cyclotomique $K(\zeta_n)/K$ est le \omterm{thm:b3:galois:splitting}{corps de
décomposition} de $X^n - 1$, \omterm{def:b3:galois:separable}{séparable} en caractéristique $0$~;
un automorphisme envoie $\zeta_n$ sur $\zeta_n^k$ avec $\gcd(k,
n) = 1$, et la composition des morphismes $k$ est
multiplicative~: le groupe s'injecte dans $(\Z/n\Z)^\times$,
abélien.
\end{proof}

\begin{theorem}[Résolubilité par radicaux]\label{thm:b3:galois:solvable}
Soit $K$ de caractéristique $0$ et $P \in K[X]$. Si $P$ est
\omterm{def:b3:groups:derived}{résoluble} par radicaux, alors le \omterm{def:b3:galois:galois}{groupe de Galois} de $P$
(\omterm{def:b3:galois:galois}{groupe de Galois} de son \omterm{thm:b3:galois:splitting}{corps de décomposition} sur $K$) est un
\omterm{def:b3:groups:derived}{groupe résoluble}.\index{résolubilité par radicaux}
\end{theorem}

\begin{proof}
Soit $L$ le \omterm{thm:b3:galois:splitting}{corps de décomposition} de $P$, \omterm{def:b3:rings:content}{contenu} dans une
\omterm{def:b3:galois:radical}{extension radicale} $K = F_0 \subseteq \dots \subseteq F_r = R$
avec $F_{i+1} = F_i(\alpha_i)$, $\alpha_i^{n_i} \in F_i$. Poser
$N = \operatorname{lcm}(n_1, \dots, n_r)$ et adjoindre
d'abord $\zeta_N$~: $K' = K(\zeta_N)$ est \omterm{def:b3:galois:galois}{galoisienne} sur $K$
de groupe abélien (lemme), donc \omterm{def:b3:groups:derived}{résoluble}. Remplacer chaque
étape par $F_i' = F_i(\zeta_N)$~: on a une tour $K' =
F_0' \subseteq \dots \subseteq F_r' = R'$ où chaque étape
$F_{i+1}'/F_i'$ est cyclique par le lemme (les racines de
l'unité sont déjà dans $F_i'$). Passer à une clôture
\omterm{def:b3:galois:galois}{galoisienne} $N$ de $R'/K$ contenant $L$~: le groupe
$\operatorname{Gal}(N/K)$ admet une filtration dont les
quotients successifs sont abéliens (d'abord le quotient
cyclotomique, puis les quotients cycliques des étapes
\omterm{def:b3:galois:radical}{radicales}, éventuellement après avoir normalisé chaque étape
par les conjugués)~: $G = \operatorname{Gal}(N/K)$ est
\omterm{def:b3:groups:derived}{résoluble} (\cref{prop:b3:groups:derived}). Or
$\operatorname{Gal}(L/K)$ est un quotient de $G$ (restriction,
surjective car $L/K$ est normale), donc \omterm{def:b3:groups:derived}{résoluble}.
\end{proof}

\begin{corollary}\label{cor:b3:galois:quintic}
L'équation générale de \omterm{def:b3:galois:extension}{degré} $n \geq 5$ n'est pas \omterm{def:b3:groups:derived}{résoluble} par
radicaux. Plus précisément, il existe des polynômes de \omterm{def:b3:galois:extension}{degré}
$5$ sur $\Q$ (p.ex.\ $X^5 - 4X + 2$) dont le \omterm{def:b3:galois:galois}{groupe de Galois}
est $S_5$, non \omterm{def:b3:groups:derived}{résoluble}.
\end{corollary}

Le \omterm{def:b3:galois:galois}{groupe de Galois} de l'équation générale est $S_n$ (les
racines sont indéterminées, le corps de base est le corps des
fractions des polynômes symétriques)~; $S_n$ n'est pas
\omterm{def:b3:groups:derived}{résoluble} pour $n \geq 5$
(\cref{cor:b3:groups:snnotsolvable}). Pour un exemple concret
sur $\Q$~: $P = X^5 - 4X + 2$ est irréductible (\omterm{thm:b3:rings:criteria}{Eisenstein} en
$2$)~; il a exactement trois racines réelles (étude de $P'$,
comme dans \cref{exo:b3:galois:11})~; la conjugaison complexe
induit une transposition dans le \omterm{def:b3:galois:galois}{groupe de Galois} $G \leq
S_5$, et l'irréductibilité donne un $5$-cycle~: $G = S_5$.
Ainsi le~\cref{thm:b3:galois:solvable} interdit la
résolubilité par radicaux --- le même $S_5$ dont la
simplicité de $A_5$ (volume des groupes) a frappé sa cible.

\begin{remark}\label{rem:b3:galois:converse}
La réciproque du~\cref{thm:b3:galois:solvable} est vraie en
caractéristique $0$ (admis ici)~: si le \omterm{def:b3:galois:galois}{groupe de Galois} est
\omterm{def:b3:groups:derived}{résoluble}, le polynôme est \omterm{def:b3:groups:derived}{résoluble} par radicaux. La
démonstration construit une tour en inversant le
\cref{lem:b3:galois:cyclicsteps}~: une \omterm{thm:b3:groups:jordanholder}{suite de composition} à
quotients cycliques se réalise par des adjonctions de
radicaux (après avoir adjoint les racines de l'unité
nécessaires). C'est le sens «~\omterm{def:b3:groups:derived}{groupe résoluble} $\Rightarrow$
équation \omterm{def:b3:groups:derived}{résoluble}~» qui justifie le vocabulaire de
\cref{ex:b3:groups:solvableexamples}, et qui fait de la
simplicité de $A_5$ l'obstruction définitive pour le \omterm{def:b3:galois:extension}{degré}
cinq.
\end{remark}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:b3:galois:1}
Montrer $[\Q(\sqrt2, \sqrt3):\Q] = 4$, que $\Q(\sqrt2 + \sqrt3)
= \Q(\sqrt2, \sqrt3)$, et calculer le \omterm{def:b3:galois:algebraic}{polynôme minimal} de
$\sqrt2 + \sqrt3$ sur $\Q$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b3:galois:2}
Soit $\alpha = \sqrt[3]2$. Montrer que $\Q(\alpha)/\Q$ n'est
pas normale (exhiber un plongement $\Q(\alpha) \to \C$ dont
l'image n'est pas $\Q(\alpha)$), déterminer le \omterm{thm:b3:galois:splitting}{corps de
décomposition} $L$ de $X^3 - 2$ et $[L:\Q]$, et vérifier
$\operatorname{Aut}_\Q(\Q(\alpha)) = \{\mathrm{id}\}$~: pour
les extensions non \omterm{def:b3:galois:galois}{galoisiennes}, le groupe d'automorphismes
peut être bien plus petit que le \omterm{def:b3:galois:extension}{degré}.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b3:galois:3}
Construire $\mathbb F_9$ comme $\mathbb F_3[X]/(X^2+1)$ et
trouver un générateur de $\mathbb F_9^\times$. Lister les
polynômes moniques irréductibles de \omterm{def:b3:galois:extension}{degrés} $1, 2, 3$ sur
$\mathbb F_2$, et vérifier $X^8 - X = X(X+1)(X^3+X+1)(X^3+X^2+1)$
sur $\mathbb F_2$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:galois:4}
(a) Trouver toutes les racines primitives modulo $7$ et modulo
$11$ (i.e.\ générateurs de $\mathbb F_7^\times$, $\mathbb
F_{11}^\times$).
(b) Montrer que pour $p$ impair, $x \in \mathbb F_p^\times$ est
un carré ssi $x^{(p-1)/2} = 1$ (critère d'Euler), et retrouver
le critère pour $-1$ du~\cref{pb:b3:rings:1}.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:galois:5}
Montrer que $\mathbb F_{p^m} \cap \mathbb F_{p^n} = \mathbb
F_{p^{\gcd(m,n)}}$ et $\mathbb F_{p^m}\mathbb F_{p^n} = \mathbb
F_{p^{\operatorname{lcm}(m,n)}}$ dans une \omterm{def:b3:galois:closure}{clôture algébrique}
fixée $\bar{\mathbb F}_p$, et décrire
$\operatorname{Gal}(\mathbb F_{p^n}/\mathbb F_{p^m})$ pour $m
\mid n$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:galois:6}
Soit $I_d(q)$ le nombre de polynômes moniques irréductibles de
\omterm{def:b3:galois:extension}{degré} $d$ sur $\mathbb F_q$. Démontrer
\[
X^{q^n} - X \;=\; \prod_{d \mid n}\ \prod_{P \text{ irred.
monique, } \deg P = d} P ,
\qquad\text{d'où}\qquad
q^n = \sum_{d \mid n} d\, I_d(q).
\]
En déduire $I_1, I_2, I_3, I_4$ explicitement, et $I_d(q) \geq
1$ pour tout $d$ (donc les extensions $\mathbb F_{q^d}/\mathbb
F_q$ existent comme quotients $\mathbb F_q[X]/(P)$ pour tout
$d$).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:galois:7}
Déterminer $\operatorname{Gal}(\Q(\sqrt2,\sqrt3)/\Q)$ et le
treillis complet des corps intermédiaires. Même question pour
le \omterm{thm:b3:galois:splitting}{corps de décomposition} de $(X^2-2)(X^2-3)(X^2-6)$ --- que
remarque-t-on~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:galois:8}
(Le cubique, résolu par son groupe) Soit $P = X^3 + pX + q \in
\Q[X]$ irréductible de racines $x_1, x_2, x_3$ et de \omterm{thm:b3:galois:splitting}{corps de
décomposition} $L$. Soit $\delta = (x_1 - x_2)(x_1 - x_3)(x_2 -
x_3)$ et $\Delta = \delta^2 = -4p^3 - 27q^2$ (admettre cette
identité classique ou la vérifier en développant les fonctions
symétriques).
(a) Montrer $\operatorname{Gal}(L/\Q) \cong A_3$ ou $S_3$,
selon que $\Delta$ est ou n'est pas un carré dans $\Q$.
(b) Avec $j = \zeta_3$, définir les résolvantes de Lagrange $u
= x_1 + jx_2 + j^2x_3$ et $v = x_1 + j^2x_2 + jx_3$. Montrer
$u^3 + v^3 = -27q$ et $uv = -3p$, et résoudre pour $u^3, v^3$~:
les formules de Cardan sortent. Où a-t-on utilisé la
résolubilité de $S_3$~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:galois:9}
Dans $\Q(\zeta_5)$~: montrer que l'unique sous-corps
quadratique est $\Q(\sqrt5)$, via les sommes de Gauss $\eta_0
= \zeta_5 + \zeta_5^4$, $\eta_1 = \zeta_5^2 + \zeta_5^3$~:
calculer $\eta_0 + \eta_1$ et $\eta_0\eta_1$, et en déduire
$\cos\frac{2\pi}5 = \frac{\sqrt5 - 1}4$. Conclure que le
pentagone régulier est \omterm{def:b3:galois:constructible}{constructible}.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b3:galois:10}
Soit $K = \mathbb F_p(S, T)$ (fractions rationnelles en deux
indéterminées) et $L = K(S^{1/p}, T^{1/p})$.
(a) Montrer $[L:K] = p^2$ et que $\alpha^p \in K$ pour tout
$\alpha \in L$.
(b) En déduire que $L/K$ n'est \emph{pas} \omterm{def:b3:groups:simple}{simple}~: aucun
élément primitif n'existe --- l'inséparabilité est fatale au
\cref{thm:b3:galois:primitive}.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b3:galois:11}
Soit $P = X^5 - 4X + 2$ et $G$ son \omterm{def:b3:galois:galois}{groupe de Galois} sur $\Q$,
agissant sur les $5$ racines.
(a) Montrer que $P$ est irréductible, et en déduire $5 \mid
\abs G$~; conclure que $G$ contient un $5$-cycle (Cauchy,
\cref{thm:b3:groups:cauchy}).
(b) Montrer, en étudiant les variations de $x \mapsto x^5 - 4x
+ 2$, que $P$ a exactement $3$ racines réelles~; en déduire que
la conjugaison complexe se restreint en une transposition dans
$G$.
(c) Montrer qu'un sous-groupe de $S_5$ contenant une
transposition et un $5$-cycle est $S_5$ \emph{(conjuguer la
transposition par des puissances du cycle)}. Conclure $G \cong
S_5$ et, avec le~\cref{thm:b3:galois:solvable}, que $P$ n'est
pas \omterm{def:b3:groups:derived}{résoluble} par radicaux.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b3:galois:12}
(Le quartique diédral) Soit $\alpha = \sqrt[4]2$ et $L =
\Q(\alpha, \iu)$, le \omterm{thm:b3:galois:splitting}{corps de décomposition} de $X^4 - 2$ sur
$\Q$.
(a) Montrer $[L : \Q] = 8$ et que $G =
\operatorname{Gal}(L/\Q)$ est engendré par $\sigma\colon
\alpha \mapsto \iu\alpha,\ \iu \mapsto \iu$ et la conjugaison
complexe $\tau$, avec $\sigma^4 = \tau^2 = e$ et
$\tau\sigma\tau = \sigma^{-1}$~: $G \cong D_4$.
(b) Lister le treillis des sous-groupes de $D_4$ (dix
sous-groupes) et apparier chacun à son corps fixe~; vérifier
en particulier que $\Q(\sqrt2)$, $\Q(\iu)$, $\Q(\iu\sqrt2)$
sont les trois sous-corps quadratiques, et localiser
$\Q(\alpha)$, $\Q(\iu\alpha)$, $\Q(\sqrt2, \iu)$.
(c) Quels corps intermédiaires sont galoisiens sur $\Q$~?
Apparier la réponse aux sous-groupes normaux de $D_4$, et
expliquer pourquoi $\Q(\alpha)/\Q$ échoue tandis que
$\Q(\sqrt2)/\Q$ réussit.
\end{exercise}

\section{Problème~: Gauss et le 17-gone régulier}

\begin{problem}[{Problème du week-end --- constructibilité du
17-gone}]\label{pb:b3:galois:1}
Le 30 mars 1796, Gauss, âgé de dix-neuf ans, montra que le
$17$-gone régulier est \omterm{def:b3:galois:constructible}{constructible} --- le premier progrès sur
la question depuis l'Antiquité. On reconstruit son calcul avec
les outils de ce chapitre. Poser $\zeta = \eu^{2\iu\pi/17}$, $L
= \Q(\zeta)$, $G = \operatorname{Gal}(L/\Q)$.

\medskip
\noindent\textbf{Partie I --- Le groupe et sa filtration.}
\begin{enumerate}
  \item Justifier~: $[L:\Q] = 16$, $G \cong (\Z/17\Z)^\times$,
        cyclique d'ordre $16$. Vérifier que $3$ engendre
        $(\Z/17\Z)^\times$ \emph{(calculer les puissances de
        $3$ modulo $17$~: $3, 9, 10, 13, 5, 15, 11, 16,
        \dots$)}.
  \item Soit $\sigma \in G$ avec $\sigma(\zeta) = \zeta^3$, et
        $H_k = \langle \sigma^{2^k}\rangle$ pour $k = 0, \dots,
        4$. Montrer que $G = H_0 \supset H_1 \supset H_2
        \supset H_3 \supset H_4 = \{e\}$ avec chaque indice
        $[H_k : H_{k+1}] = 2$, et que les corps fixes $\Q =
        L_0 \subset L_1 \subset L_2 \subset L_3 \subset L_4 =
        L$ forment une tour d'extensions quadratiques.
  \item Conclure \emph{a priori}, en utilisant
        le~\cref{thm:b3:galois:wantzel}, que $\zeta$ --- donc
        le $17$-gone --- est \omterm{def:b3:galois:constructible}{constructible}. Le reste du
        problème rend la tour explicite.
\end{enumerate}

\medskip
\noindent\textbf{Partie II --- Les périodes de longueur $8$.}
Définir les \emph{périodes de Gauss}
\[
\eta_0 = \sum_{k \text{ pair}} \zeta^{3^k \bmod 17}
= \zeta^{1} + \zeta^{9} + \zeta^{13} + \zeta^{15} + \zeta^{16} +
\zeta^{8} + \zeta^{4} + \zeta^{2},
\qquad
\eta_1 = \sum_{k \text{ impair}} \zeta^{3^k \bmod 17}.
\]
\begin{enumerate}[resume]
  \item Montrer que $\eta_0, \eta_1$ sont fixés par $H_1$ et
        échangés par $\sigma$~; en déduire $\eta_0, \eta_1 \in
        L_1$ et qu'ils sont les deux racines d'une quadratique
        sur $\Q$.
  \item Calculer $\eta_0 + \eta_1 = -1$. Montrer $\eta_0\eta_1
        = -4$ \emph{(chaque produit $\zeta^a\zeta^b$ est un
        certain $\zeta^c$, $c \neq 0$~; compter combien de
        fois chaque $c$ apparaît, ou argumenter que le produit
        est un entier rationnel fixé par $G$, égal à la somme
        sur les $64$ produits, et utiliser que chaque résidu
        non nul apparaît également souvent)}.
  \item En déduire $\eta_0 = \frac{-1 + \sqrt{17}}2$, $\eta_1 =
        \frac{-1-\sqrt{17}}2$ (identifier lequel est lequel
        numériquement~: $\eta_0 \approx 1{,}56$), et $L_1 =
        \Q(\sqrt{17})$.
\end{enumerate}

\medskip
\noindent\textbf{Partie III --- Périodes de longueur $4$ et
$2$.}
Définir
\[
\beta_0 = \zeta + \zeta^{13} + \zeta^{16} + \zeta^{4},\quad
\beta_1 = \zeta^3 + \zeta^5 + \zeta^{14} + \zeta^{12},\quad
\beta_2 = \zeta^9 + \zeta^{15} + \zeta^{8} + \zeta^{2},\quad
\beta_3 = \zeta^{10} + \zeta^{11} + \zeta^{7} + \zeta^{6}.
\]
\begin{enumerate}[resume]
  \item Montrer $\beta_0 + \beta_2 = \eta_0$, $\beta_1 +
        \beta_3 = \eta_1$, et que $\beta_0, \beta_2$ sont
        fixés par $H_2$, échangés par $\sigma^2$.
  \item Calculer $\beta_0\beta_2 = -1$ et $\beta_1\beta_3 = -1$
        \emph{(développer~: les seize exposants obtenus
        couvrent $1, \dots, 16$ exactement une fois)}.
  \item En déduire $\beta_0 = \frac{\eta_0 +
        \sqrt{\eta_0^2 + 4}}2$ (vérifier le signe
        numériquement~: $\beta_0 \approx 2{,}05$) et la
        formule analogue pour $\beta_1$~; d'où $L_2 =
        \Q(\beta_0)$, quadratique sur $L_1$.
  \item Soit $\gamma_0 = \zeta + \zeta^{16} =
        2\cos\frac{2\pi}{17}$ et $\gamma_1 = \zeta^{13} +
        \zeta^4$. Montrer $\gamma_0 + \gamma_1 = \beta_0$ et
        $\gamma_0\gamma_1 = \beta_1$, de sorte que $\gamma_0 =
        \frac{\beta_0 + \sqrt{\beta_0^2 - 4\beta_1}}2$.
  \item Assembler la chaîne de formules exprimant
        $\cos\frac{2\pi}{17}$ par des racines carrées
        emboîtées, et donner un contrôle décimal
        ($\cos\frac{2\pi}{17} \approx 0{,}93247$).
\end{enumerate}

\medskip
\noindent\textbf{Partie IV --- Épilogue.}
\begin{enumerate}[resume]
  \item Où exactement l'argument a-t-il utilisé que $17$ est un
        \emph{premier de Fermat} ($17 = 2^{2^2} + 1$)~? Montrer
        que pour un premier $p$, le $p$-gone régulier est
        \omterm{def:b3:galois:constructible}{constructible} ssi $p = 2^{2^t} + 1$ pour un certain $t$
        \emph{(si $p - 1 = 2^m$, montrer que $m$ doit
        lui-même être une puissance de $2$)}.
  \item En déduire la liste complète des $n$-gones réguliers
        \omterm{def:b3:galois:constructible}{constructibles} pour $n \leq 20$, en utilisant le
        critère de Gauss--Wantzel
        du~\cref{cor:b3:galois:impossible}.
\end{enumerate}

\medskip
\noindent\textbf{Partie V --- Sommes de Gauss et réciprocité
quadratique.} Les périodes de la partie II cachent un trésor.
Pour un premier impair $p$, le \emph{symbole de Legendre}
$\bigl(\frac ap\bigr)$ vaut $+1$ si $a$ est un carré non nul
mod $p$, $-1$ sinon, $0$ si $p \mid a$~;
\cref{exo:b3:galois:4}(b) (critère d'Euler) donne
$\bigl(\frac ap\bigr) \equiv a^{(p-1)/2} \pmod p$, d'où la
multiplicitivité. Écrire $\zeta = \eu^{2\iu\pi/p}$, $p^* =
(-1)^{(p-1)/2}p$, et définir la \emph{somme de Gauss}
\[
g \;=\; \sum_{a=1}^{p-1}\Bigl(\frac ap\Bigr)\zeta^a .
\]
\begin{enumerate}[resume]
  \item Montrer $\sum_{a=1}^{p-1}\bigl(\frac ap\bigr) = 0$
        (autant de carrés que de non-carrés), et démontrer la
        forme alternative $g = \sum_{a=0}^{p-1}\zeta^{a^2}$
        \emph{(chaque carré non nul est atteint deux fois, et
        $\sum_{a}\zeta^a = 0$)}. Pour $p = 17$~: relier $g$
        aux périodes de la partie II --- montrer $g = \eta_0 -
        \eta_1$ \emph{(les carrés mod $17$ sont exactement les
        puissances paires du générateur $3$)}.
  \item Démontrer $g^2 = p^*$~: développer
        \[
        g^2 = \sum_{a,b\neq0}\Bigl(\frac{ab}p\Bigr)
        \zeta^{a+b}
        = \sum_{c}\ \sum_{a \neq 0}\Bigl(\frac{a(c -
        a)}p\Bigr)\zeta^{c}
        \]
        (poser $b = c - a$), substituer $c - a = at$ pour
        évaluer la somme intérieure comme
        $\bigl(\frac{-1}p\bigr)(p - 1)$ pour $c = 0$ et
        $-\bigl(\frac{-1}p\bigr)$ sinon, et conclure avec la
        question 14. Vérifier numériquement~: pour $p = 17$,
        $(\eta_0 - \eta_1)^2 = 17$ (partie II).
  \item En déduire $\sqrt{p^*} \in \Q(\zeta_p)$, et conclure
        que l'\emph{unique} sous-corps quadratique de
        $\Q(\zeta_p)$ est $\Q(\sqrt{p^*})$ --- unique car
        $\operatorname{Gal}(\Q(\zeta_p)/\Q)$ est cyclique
        (\cref{thm:b3:galois:cyclotomicirred}) et un groupe
        cyclique a exactement un sous-groupe d'indice $2$.
        (Tout corps quadratique se plonge dans un certain
        corps cyclotomique --- c'est le premier cas du
        théorème de Kronecker--Weber, dont la forme générale
        est bien plus lointaine.)
  \item Soit maintenant $q \neq p$ un autre premier impair. En
        travaillant dans l'anneau $\Z[\zeta]$ modulo $q$,
        démontrer
        \[
        g^q \equiv \Bigl(\frac qp\Bigr)\,g \pmod{q\Z[\zeta]}
        \]
        \emph{(rêve du collégien~: $(x + y)^q \equiv x^q +
        y^q$ mod $q$ dans tout anneau commutatif~; alors $g^q
        \equiv \sum_a\bigl(\frac ap\bigr)^q\zeta^{aq}$,
        réindexer $b = aq$ et extraire
        $\bigl(\frac{q^{-1}}p\bigr) = \bigl(\frac qp\bigr)$)}.
  \item D'autre part, $g^q = g\,(g^2)^{(q-1)/2} =
        g\,(p^*)^{(q-1)/2}$~; en utilisant le critère d'Euler
        mod $q$, en déduire $g^q \equiv \bigl(\frac{p^*}q\bigr)g
        \pmod{q\Z[\zeta]}$, puis --- en multipliant les deux
        expressions de $g^q$ par $g$ et en utilisant $g^2 =
        p^*$, inversible mod $q$ --- conclure
        \[
        \Bigl(\frac qp\Bigr) = \Bigl(\frac{p^*}q\Bigr) .
        \]
        \emph{(Pourquoi une congruence entre les entiers $\pm
        p^*$ modulo $q\Z[\zeta]$ implique-t-elle leur
        égalité~? Intersecter avec $\Z$.)}
  \item Déplier $\bigl(\frac{p^*}q\bigr) =
        \bigl(\frac{-1}q\bigr)^{(p-1)/2}\bigl(\frac pq\bigr)$
        et $\bigl(\frac{-1}q\bigr) = (-1)^{(q-1)/2}$ pour
        obtenir la \emph{loi de réciprocité quadratique}~:
        \[
        \Bigl(\frac pq\Bigr)\Bigl(\frac qp\Bigr)
        = (-1)^{\frac{p-1}2\cdot\frac{q-1}2} .
        \]
        La vérifier sur $(p, q) = (17, 3)$ en listant les
        carrés mod $17$ et mod $3$, et l'utiliser pour
        décider en trois lignes si $x^2 \equiv 219 \pmod{383}$
        est \omterm{def:b3:groups:derived}{résoluble} ($383$ est premier, $219 = 3\cdot73$).
\end{enumerate}

\medskip
\noindent\textbf{Partie VI --- Compter les polynômes
irréductibles~: le théorème des nombres premiers de $\mathbb
F_q[X]$.} Fixer une puissance de premier $q$ et soit $N_q(n)$
le nombre de polynômes moniques \emph{irréductibles} de \omterm{def:b3:galois:extension}{degré}
$n$ sur $\mathbb F_q$~; rappeler de
\cref{exo:b3:galois:6} la factorisation de $X^{q^n} - X$ et
l'identité $q^n = \sum_{d\mid n}d\,N_q(d)$, que l'on inverse,
réinterprète et exploite maintenant.
\begin{enumerate}[resume]
  \item (Mots) Appeler un mot $w \in \mathbb F_q^n$
        \emph{primitif} s'il n'est pas une puissance $u^{n/d} =
        u\cdots u$ d'un mot $u$ strictement plus court, et
        soit $A(d)$ le nombre de mots primitifs de longueur
        $d$. Montrer que tout mot de longueur $n$ est de façon
        unique une puissance d'un mot primitif d'une certaine
        longueur $d \mid n$, de sorte que $q^n = \sum_{d \mid
        n}A(d)$~; en comparant avec
        \cref{exo:b3:galois:6}, conclure $A(d) = d\,N_q(d)$
        pour tout $d$, et expliquer cette coïncidence par une
        bijection explicite~: un élément $\alpha \in
        \mathbb F_{q^d}$ de \omterm{def:b3:galois:extension}{degré} $d$ a pour \omterm{def:b3:groups:action}{orbite} de
        \omterm{thm:b3:galois:finitefields}{Frobenius} $(\alpha, \alpha^q, \dots,
        \alpha^{q^{d-1}})$ exactement $d$ éléments distincts,
        et les éléments de \omterm{def:b3:galois:extension}{degré} $d$ correspondent $d$-à-$1$
        aux irréductibles de \omterm{def:b3:galois:extension}{degré} $d$.
  \item Démontrer la \emph{formule d'inversion de Möbius}~: si
        $f(n) = \sum_{d\mid n}g(d)$ pour tout $n$, alors $g(n)
        = \sum_{d\mid n}\mu(d)\,f(n/d)$, où $\mu$ est la
        fonction de Möbius ($\mu(m) = (-1)^{\#\text{facteurs
        premiers}}$ si $m$ est sans carré, $0$ sinon)
        \emph{(lemme clé~: $\sum_{d \mid m}\mu(d) = 0$ pour $m
        > 1$ --- apparier les diviseurs avec et sans un
        premier fixé)}. En déduire
        \[
        N_q(n) = \frac1n\sum_{d \mid n}\mu(d)\,q^{n/d} .
        \]
  \item Montrer $N_q(n) \geq \frac1n\bigl(q^n - 2q^{n/2}\bigr)
        > 0$ pour tout $n \geq 1$~: une nouvelle démonstration
        que $\mathbb F_{q^n}$ existe pour tout $n$. Interpréter
        le terme dominant~: un polynôme monique aléatoire de
        \omterm{def:b3:galois:extension}{degré} $n$ est irréductible avec probabilité $\sim
        \frac1n$ --- l'analogue \omterm{prop:b3:galois:perfect}{parfait} du théorème des nombres
        premiers, avec $\log x$ échangé contre $n$~; vérifier
        numériquement pour $q = 2$, $n \leq 4$
        (\cref{exo:b3:galois:6} liste les comptes).
  \item Démontrer le compagnon multiplicatif de la question
        21~:
        \[
        \prod_{\substack{\pi \text{ monique irred.}\\ \deg\pi =
        n}}\pi \;=\; \prod_{d \mid n}
        \bigl(X^{q^d} - X\bigr)^{\mu(n/d)}
        \]
        \emph{(inversion de Möbius dans le groupe abélien des
        fractions rationnelles non nulles)}~; le vérifier à la
        main pour $q = 2$, $n = 2$~: $(X^4 - X)/(X^2 - X) =
        X^2 + X + 1$.
\end{enumerate}

\medskip
\noindent\textbf{Partie VII --- Deux codas.}
\begin{enumerate}[resume]
  \item (Le second supplément) La méthode de la partie V
        calcule aussi $\bigl(\frac2q\bigr)$. Soit $\omega =
        \eu^{2\iu\pi/8}$ et $g = \omega + \omega^{-1}$. Montrer
        $g^2 = 2$ \emph{($\omega^2 = \iu$)}~; puis, pour un
        premier impair $q$, démontrer dans $\Z[\omega]$ modulo
        $q$ que
        \[
        g^q \equiv \omega^q + \omega^{-q} \pmod{q\Z[\omega]},
        \]
        et que le membre de droite égale $g$ si $q \equiv \pm1
        \pmod 8$ et $-g$ si $q \equiv \pm3 \pmod 8$. En
        comparant avec $g^q = g\,(g^2)^{(q-1)/2} \equiv
        \bigl(\frac2q\bigr)g$ comme dans la question 18,
        conclure
        \[
        \Bigl(\frac2q\Bigr) = (-1)^{(q^2-1)/8},
        \]
        en vérifiant que $(q^2 - 1)/8$ est pair exactement quand
        $q \equiv \pm1 \pmod 8$. Vérifier~: $2$ est un carré
        mod $7$ et mod $17$ ($3^2$ et $6^2$), pas mod $3$ ni
        mod $5$.
  \item (La fonction zêta de $\mathbb F_q[X]$) Démontrer
        l'identité de séries formelles en $t$~:
        \[
        \prod_{n \geq 1}\bigl(1 - t^n\bigr)^{-N_q(n)}
        = \frac1{1 - qt}
        \]
        \emph{(factorisation unique en irréductibles moniques~:
        développer chaque facteur en série géométrique et
        compter les moniques de \omterm{def:b3:galois:extension}{degré} $n$)}. Retrouver
        l'identité $q^m = \sum_{d \mid m}d\,N_q(d)$ de
        \cref{exo:b3:galois:6} en prenant les logarithmes.
        Vérifier le coefficient de $t^2$ à la main pour $q =
        2$, et utiliser la formule de la question 21 pour
        calculer $N_2(6) = 9$, en vérifiant $2^6 = 1\cdot2 +
        2\cdot1 + 3\cdot2 + 6\cdot9$.
\end{enumerate}
\end{problem}
