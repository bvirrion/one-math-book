\chapter{Théorie de la mesure}\label{ch:b3:measure}

Quelle est la longueur d'un sous-ensemble de $\R$~? La réponse
naïve --- attribuer à tout ensemble une longueur invariante
par translation prolongeant celle des intervalles --- est
\emph{impossible}~: la construction de Vitali, en fin de
chapitre, produit un ensemble sans longueur cohérente. La
théorie de la \omterm{def:b3:measure:measure}{mesure} est la retraite disciplinée~: on se
restreint à une classe riche d'ensembles \emph{mesurables},
sur laquelle une longueur dénombrablement additive existe et
est unique. Les récompenses sont immenses --- l'intégrale de
Lebesgue (\cref{ch:b3:lebesgue}), les espaces $L^p$ de
l'analyse fonctionnelle, et toute la probabilité moderne
(\cref{ch:b3:probability}) reposent sur les trois théorèmes
prouvés ici~: le lemme d'unicité de Dynkin, le théorème
d'extension de Carathéodory, et l'existence de la \omterm{def:b3:measure:lebesgueouter}{mesure de
Lebesgue}.

\section{$\sigma$-algèbres}

\begin{definition}\label{def:b3:measure:sigmaalgebra}
Une \emph{$\sigma$-algèbre}\index{sigma-algebre@$\sigma$-algèbre}
sur un ensemble $X$ est une famille $\mathcal A$ de
sous-ensembles contenant $\varnothing$, stable par
complémentation et par unions \emph{dénombrables} (donc par
intersections dénombrables, différences, et elle contient
$X$). Le couple $(X, \mathcal A)$ est un \emph{espace
mesurable}~; les membres de $\mathcal A$ sont les
\emph{ensembles mesurables}. Pour toute famille $\mathcal E$
de sous-ensembles, $\sigma(\mathcal E)$ désigne la plus
petite $\sigma$-algèbre contenant $\mathcal E$ (l'intersection
de toutes --- une intersection de $\sigma$-algèbres en est
une).
\end{definition}

\begin{definition}\label{def:b3:measure:borel}
La \emph{$\sigma$-algèbre de Borel}\index{Borel
sigma-algebre@$\sigma$-algèbre de Borel} d'un espace
topologique est $\mathcal B(X) = \sigma(\{\text{ouverts}\})$.
Sur $\R$~: $\mathcal B(\R)$ est aussi engendrée par les
intervalles \omterm{def:b3:topology:topology}{ouverts}, par les intervalles fermés, par les
demi-droites $(-\infty, a]$, et par les demi-droites à
extrémités rationnelles (\cref{exo:b3:measure:1}) --- chaque
famille engendre les \omterm{def:b3:topology:topology}{ouverts} par opérations dénombrables,
p.~ex.\ tout \omterm{def:b3:topology:topology}{ouvert} de $\R$ est une union dénombrable
d'intervalles \omterm{def:b3:topology:topology}{ouverts} à données rationnelles.
\end{definition}

\begin{definition}\label{def:b3:measure:dynkin}
Un \emph{$\pi$-système}\index{pi-systeme@$\pi$-système} est une
famille stable par intersections finies. Un
\emph{$\lambda$-système}\index{lambda-systeme@$\lambda$-système}
(classe de Dynkin) est une famille $\mathcal D$ avec~: $X \in
\mathcal D$~; $A, B \in \mathcal D$, $A \subseteq B$
$\Rightarrow$ $B \setminus A \in \mathcal D$~; et $A_n
\uparrow A$, $A_n \in \mathcal D$ $\Rightarrow$ $A \in
\mathcal D$.
\end{definition}

\begin{theorem}[Lemme $\pi$--$\lambda$ de Dynkin]
\label{thm:b3:measure:dynkin}
Si un $\lambda$-système $\mathcal D$ contient un
$\pi$-système $\mathcal P$, alors $\mathcal D \supseteq
\sigma(\mathcal P)$.\index{lemme de Dynkin}
\end{theorem}

\begin{proof}
Soit $\mathcal D_0$ le plus petit $\lambda$-système contenant
$\mathcal P$ (intersection de tous ceux-là)~; il suffit de
montrer que $\mathcal D_0$ est une $\sigma$-algèbre, car alors
$\sigma(\mathcal P) \subseteq \mathcal D_0 \subseteq \mathcal
D$. Un $\lambda$-système stable par intersections finies
\emph{est} une $\sigma$-algèbre~: complémentaires ($X
\setminus A = X \setminus A$ avec $A \subseteq X$), unions
finies ($A \cup B = X \setminus ((X\setminus
A)\cap(X\setminus B))$), et unions dénombrables via
$\bigcup_{k \leq n}A_k \uparrow \bigcup_kA_k$. On prouve donc
que $\mathcal D_0$ est un $\pi$-système, en deux étapes. Soit
\[
\mathcal D_1 = \{A \in \mathcal D_0 : A \cap P \in \mathcal
D_0 \ \forall P \in \mathcal P\}.
\]
$\mathcal D_1$ est un $\lambda$-système (les trois axiomes se
vérifient en intersectant avec $P$~: p.~ex.\ $(B\setminus
A)\cap P = (B \cap P)\setminus(A \cap P)$, une différence
propre dans $\mathcal D_0$) et contient $\mathcal P$
($\pi$-système)~: $\mathcal D_1 = \mathcal D_0$. Soit
maintenant
\[
\mathcal D_2 = \{A \in \mathcal D_0 : A \cap D \in \mathcal
D_0\ \forall D \in \mathcal D_0\}.
\]
Par l'étape précédente, $\mathcal D_2 \supseteq \mathcal P$~;
et $\mathcal D_2$ est un $\lambda$-système par la même
vérification~: $\mathcal D_2 = \mathcal D_0$, ce qui dit
exactement que $\mathcal D_0$ est stable par intersections.
\end{proof}

\section{Mesures}

\begin{definition}\label{def:b3:measure:measure}
Une \emph{mesure}\index{mesure} sur $(X, \mathcal A)$ est une
application $\mu \colon \mathcal A \to [0, +\infty]$ avec
$\mu(\varnothing) = 0$ qui est \emph{$\sigma$-additive}~: pour
des $(A_n)_{n\in\N}$ deux à deux disjoints,
\[
\mu\Bigl(\bigsqcup_n A_n\Bigr) = \sum_n \mu(A_n).
\]
$(X, \mathcal A, \mu)$ est un \emph{espace mesuré}~; $\mu$ est
\emph{finie} si $\mu(X) < \infty$, une \emph{mesure de
probabilité} si $\mu(X) = 1$, \emph{$\sigma$-finie} si $X$ est
une union dénombrable d'ensembles de mesure finie. Exemples~:
la mesure de comptage sur $(\N, \mathcal P(\N))$~; la masse de
Dirac $\delta_a(A) = \mathbf 1_{a \in A}$~; et, objet de ce
chapitre, la \omterm{def:b3:measure:lebesgueouter}{mesure de Lebesgue}.
\end{definition}

\begin{proposition}\label{prop:b3:measure:basics}
Soit $\mu$ une \omterm{def:b3:measure:measure}{mesure}. (a) Monotonie~: $A \subseteq B
\Rightarrow \mu(A) \leq \mu(B)$. (b) Sous-additivité
dénombrable~: $\mu(\bigcup A_n) \leq \sum\mu(A_n)$. (c)
Continuité par en bas~: $A_n \uparrow A \Rightarrow \mu(A_n)
\to \mu(A)$. (d) Continuité par en haut~: $A_n \downarrow A$
\emph{avec $\mu(A_1) < \infty$} $\Rightarrow \mu(A_n) \to
\mu(A)$.
\end{proposition}

\begin{proof}
(a) $B = A \sqcup (B\setminus A)$. (b) Disjoindre~: $B_n = A_n
\setminus \bigcup_{k<n}A_k$ sont disjoints de même union, et
$\mu(B_n) \leq \mu(A_n)$. (c) $A = \bigsqcup_n (A_n \setminus
A_{n-1})$ ($A_0 = \varnothing$)~: les sommes partielles de
$\sum\mu(A_n\setminus A_{n-1})$ valent $\mu(A_n)$. (d)
Appliquer (c) à $A_1 \setminus A_n \uparrow A_1 \setminus A$
et soustraire de $\mu(A_1)$ --- la finitude rend la
soustraction légitime. Contre-exemple sans elle~: $A_n = [n,
\infty)$ pour la \omterm{def:b3:measure:lebesgueouter}{mesure de Lebesgue}~: $A_n \downarrow
\varnothing$ mais $\mu(A_n) = \infty$.
\end{proof}

\begin{theorem}[Unicité]\label{thm:b3:measure:uniqueness}
Soient $\mu, \nu$ des \omterm{def:b3:measure:measure}{mesures} sur $\sigma(\mathcal P)$,
$\mathcal P$ un $\pi$-système, avec $\mu = \nu$ sur $\mathcal
P$. S'il existe des $P_k \in \mathcal P$ avec $P_k \uparrow X$
et $\mu(P_k) < \infty$, alors $\mu = \nu$ sur tout
$\sigma(\mathcal P)$.
\end{theorem}

\begin{proof}
Fixons $k$ et considérons les \omterm{def:b3:measure:measure}{mesures} finies $\mu_k(A) =
\mu(A \cap P_k)$ et $\nu_k(A) = \nu(A \cap P_k)$ sur
$\sigma(\mathcal P)$~: elles coïncident sur $\mathcal P$, car
$P \cap P_k \in \mathcal P$ ($\pi$-système), et elles
attribuent à $X$ la même valeur finie $\mu(P_k)$. La classe
$\mathcal D = \{A : \mu_k(A) = \nu_k(A)\}$ est un
$\lambda$-système~: $X \in \mathcal D$~; différences propres
par soustraction (valeurs finies)~; limites croissantes par
continuité par en bas (\cref{prop:b3:measure:basics}(c)). Elle
contient le $\pi$-système $\mathcal P$, donc Dynkin
(\cref{thm:b3:measure:dynkin}) donne $\mathcal D \supseteq
\sigma(\mathcal P)$~: $\mu_k = \nu_k$ partout. Enfin, pour
tout $A \in \sigma(\mathcal P)$, la continuité par en bas le
long de $A \cap P_k \uparrow A$ donne $\mu(A) =
\lim_k\mu_k(A) = \lim_k\nu_k(A) = \nu(A)$.
\end{proof}

\section{Mesures extérieures et théorème de Carathéodory}

\begin{definition}\label{def:b3:measure:outer}
Une \emph{mesure extérieure}\index{mesure extérieure} sur $X$
est une application $\mu^* \colon \mathcal P(X) \to [0,
\infty]$ avec $\mu^*(\varnothing) = 0$, monotone, et
dénombrablement sous-additive. Un ensemble $A$ est
\emph{$\mu^*$-mesurable} (Carathéodory) s'il scinde
additivement tout ensemble~:
\[
\mu^*(E) = \mu^*(E \cap A) + \mu^*(E \setminus A)
\qquad \text{pour tout } E \subseteq X
\]
($\leq$ vaut toujours par sous-additivité~; le \omterm{def:b3:rings:content}{contenu} est
$\geq$).
\end{definition}

\begin{theorem}[Carathéodory]\label{thm:b3:measure:caratheodory}
Les ensembles $\mu^*$-mesurables forment une $\sigma$-algèbre
$\mathcal M$, et $\mu^*\restriction_{\mathcal M}$ est une
\omterm{def:b3:measure:measure}{mesure}. De plus tout ensemble avec $\mu^*(N) = 0$ appartient
à $\mathcal M$ (la \omterm{def:b3:measure:measure}{mesure} est
\emph{complète}).\index{théorème de Carathéodory}
\end{theorem}

\begin{proof}
$\mathcal M$ contient $\varnothing$ et est stable par
complémentation (la condition définissante est symétrique en
$A$, $X\setminus A$). \emph{Unions finies}~: soient $A, B \in
\mathcal M$ et $E$ arbitraire~; en scindant $E$ par $A$, puis
chaque morceau par $B$~:
\[
\mu^*(E) = \mu^*(E\cap A\cap B) + \mu^*(E\cap A\setminus B) +
\mu^*(E\cap B\setminus A) + \mu^*(E\setminus(A\cup B)).
\]
Les trois premiers morceaux recouvrent $E \cap (A \cup B)$,
donc la sous-additivité donne $\mu^*(E) \geq
\mu^*(E\cap(A\cup B)) + \mu^*(E\setminus(A\cup B))$~: $A \cup
B \in \mathcal M$. Par induction, unions finies~; avec les
complémentaires, les manipulations de disjonction finie sont
disponibles.

\emph{Additivité sur $\mathcal M$}~: pour $A, B \in \mathcal
M$ disjoints et $E$ quelconque~: $\mu^*(E\cap(A\sqcup B)) =
\mu^*(E\cap A) + \mu^*(E \cap B)$ (scinder par $A$)~; par
induction,
\[
\mu^*\Bigl(E \cap \bigsqcup_{k\leq n}A_k\Bigr)
= \sum_{k\leq n}\mu^*(E\cap A_k).
\tag{$*$}
\]

\emph{Unions dénombrables}~: soient $(A_k) \subseteq \mathcal
M$ disjoints (il suffit, par disjonction dans l'algèbre
$\mathcal M$), $A = \bigsqcup A_k$, $E$ arbitraire. En
utilisant $\bigsqcup_{k\leq n}A_k \in \mathcal M$ et la
monotonie~:
\[
\mu^*(E) = \mu^*\Bigl(E\cap\bigsqcup_{k\leq n}A_k\Bigr) +
\mu^*\Bigl(E\setminus\bigsqcup_{k\leq n}A_k\Bigr)
\geq \sum_{k \leq n}\mu^*(E\cap A_k) + \mu^*(E\setminus A)
\]
par ($*$). Faire $n \to \infty$ et utiliser la
sous-additivité dénombrable en sens inverse~:
\[
\mu^*(E) \geq \sum_{k}\mu^*(E\cap A_k) + \mu^*(E\setminus A)
\geq \mu^*(E \cap A) + \mu^*(E\setminus A) \geq \mu^*(E) :
\]
toutes les inégalités sont des égalités. Cela prouve à la
fois $A \in \mathcal M$ et, en prenant $E = A$,
l'additivité dénombrable de $\mu^*$ sur $\mathcal M$.

\emph{Ensembles nuls}~: si $\mu^*(N) = 0$, alors pour tout
$E$~: $\mu^*(E \cap N) + \mu^*(E\setminus N) \leq 0 +
\mu^*(E)$~: $N \in \mathcal M$.
\end{proof}

\section{Mesure de Lebesgue sur \texorpdfstring{$\R$}{R}}

\begin{definition}\label{def:b3:measure:lebesgueouter}
La \emph{mesure extérieure de Lebesgue}\index{mesure de
Lebesgue} de $A \subseteq \R$ est
\[
\lambda^*(A) = \inf\Bigl\{\sum_{n} (b_n - a_n) :
A \subseteq \bigcup_n \intoo{a_n}{b_n}\Bigr\}
\]
(recouvrements dénombrables par intervalles \omterm{def:b3:topology:topology}{ouverts}).
\end{definition}

\begin{lemma}\label{lem:b3:measure:outerbasics}
$\lambda^*$ est une \omterm{def:b3:measure:outer}{mesure extérieure}, invariante par
translations, et $\lambda^*(I) = \ell(I)$ (la longueur) pour
tout intervalle $I$.
\end{lemma}

\begin{proof}
\omterm{def:b3:measure:outer}{Mesure extérieure}~: $\varnothing$ est recouvert par des
intervalles arbitrairement petits~; la monotonie est claire~;
sous-additivité~: étant donné des recouvrements de chaque
$A_n$ à $\varepsilon 2^{-n}$ de l'infimum, leur union
recouvre $\bigcup A_n$ avec longueur totale $\leq \sum
\lambda^*(A_n) + \varepsilon$. Invariance par translation~:
translater les recouvrements.

Longueur~: il suffit de traiter $I = \intcc ab$ (les autres
types diffèrent par des extrémités, de \omterm{def:b3:measure:outer}{mesure extérieure}
$0$~: les recouvrir par de tout petits intervalles~; puis
encadrer $\intcc{a+\varepsilon}{b - \varepsilon} \subseteq
\intoo ab$ par des comparaisons). $\lambda^*(\intcc ab) \leq
b - a$~: recouvrir par $\intoo{a-\varepsilon}{b+\varepsilon}$.
Réciproquement soit $\intcc ab \subseteq
\bigcup_n\intoo{a_n}{b_n}$~: par \emph{compacité}
(Borel--Lebesgue, \cref{thm:b3:topology:metriccompact}), un
nombre fini d'intervalles suffit, disons $I_1, \dots, I_N$.
On montre $\sum_{k\leq N}(b_k - a_k) \geq b - a$ par
induction sur $N$~: choisir $I_{k_1} \ni a$~; si $b_{k_1} > b$
c'est fini ($b_{k_1} - a_{k_1} > b - a$)~; sinon le segment
$\intcc{b_{k_1}}b$ est recouvert par les $N - 1$ intervalles
restants, et l'induction donne $\sum_{k \neq k_1}(b_k - a_k)
\geq b - b_{k_1}$, tandis que $b_{k_1} - a_{k_1} > b_{k_1} -
a$~: sommer.
\end{proof}

\begin{theorem}[Mesure de Lebesgue]\label{thm:b3:measure:lebesgue}
Tout borélien de $\R$ est $\lambda^*$-mesurable. La
restriction $\lambda$ de $\lambda^*$ à la $\sigma$-algèbre
$\mathcal L = \mathcal M_{\lambda^*} \supseteq \mathcal
B(\R)$ (la \emph{$\sigma$-algèbre de Lebesgue}) est l'unique
\omterm{def:b3:measure:measure}{mesure} sur $\mathcal B(\R)$ attribuant à chaque intervalle sa
longueur~; elle est invariante par translation et
$\sigma$-finie.
\end{theorem}

\begin{proof}
Par le~\cref{thm:b3:measure:caratheodory} il suffit de
montrer que chaque demi-droite $A = \intoo{-\infty}c$ est
$\lambda^*$-mesurable (les demi-droites engendrent $\mathcal
B$, \cref{def:b3:measure:borel}). Soit $E \subseteq \R$ avec
$\lambda^*(E) < \infty$ et $\bigcup I_n \supseteq E$ un
recouvrement avec $\sum\ell(I_n) \leq \lambda^*(E) +
\varepsilon$. Chaque $I_n$ se scinde en les deux intervalles
$I_n' = I_n \cap A$ et $I_n'' = I_n\setminus A$ (un
intervalle moins une demi-droite est un intervalle) avec
$\ell(I_n') + \ell(I_n'') = \ell(I_n)$~; les $I_n'$
recouvrent $E \cap A$ et les $I_n''$ recouvrent $E \setminus
A$ (élargir chacun en un intervalle \omterm{def:b3:topology:topology}{ouvert} de longueur $\ell +
\varepsilon2^{-n}$ pour rester dans la définition), donc
\[
\lambda^*(E\cap A) + \lambda^*(E\setminus A)
\leq \sum_n\bigl(\ell(I_n') + \ell(I_n'')\bigr) + 2\varepsilon
\leq \lambda^*(E) + 3\varepsilon .
\]
Unicité~: deux \omterm{def:b3:measure:measure}{mesures} coïncidant avec la longueur sur le
$\pi$-système des intervalles $\intoc ab$ (finies sur ceux-ci)
coïncident sur $\sigma(\text{intervalles}) = \mathcal B$ par
le~\cref{thm:b3:measure:uniqueness} avec $P_k =
\intoc{-k}k$. $\sigma$-finitude~: $\R = \bigcup(-k, k]$.
\end{proof}

\begin{theorem}[Régularité]\label{thm:b3:measure:regularity}
Pour tout $A \in \mathcal L$~:\index{régularité de la mesure
de Lebesgue}
\[
\lambda(A) = \inf\{\lambda(U) : U \supseteq A \text{
ouvert}\}
= \sup\{\lambda(K) : K \subseteq A \text{ compact}\}.
\]
\end{theorem}

\begin{proof}
\emph{Extérieure}~: un recouvrement $\bigcup I_n$ avec
$\sum\ell(I_n) \leq \lambda(A) + \varepsilon$ est un \omterm{def:b3:topology:topology}{ouvert}
$U \supseteq A$ avec $\lambda(U) \leq \lambda(A) +
\varepsilon$ (sous-additivité)~; si $\lambda(A) = \infty$
l'énoncé est trivial. \emph{Intérieure}~: d'abord $A$ borné,
$A \subseteq [-M, M]$. Choisir un \omterm{def:b3:topology:topology}{ouvert} $U \supseteq
([-M,M]\setminus A)$ avec $\lambda(U) \leq
\lambda([-M,M]\setminus A) + \varepsilon$~; alors $K = [-M,
M]\setminus U$ est compact, $K \subseteq A$, et
\[
\lambda(K) = \lambda([-M,M]) - \lambda([-M,M]\cap U)
\geq \lambda([-M,M]) - \bigl(\lambda([-M,M]) - \lambda(A) +
\varepsilon\bigr) = \lambda(A) - \varepsilon .
\]
Pour $A$ général~: $\lambda(A) = \lim_M\lambda(A\cap[-M,M])$
(continuité par en bas) et appliquer le cas borné à
l'intérieur.
\end{proof}

\begin{example}\label{ex:b3:measure:cantor}
L'ensemble de Cantor (\cref{exo:b3:topology:10}) a $\lambda(C)
= 0$~: $C \subseteq C_n$, une union de $2^n$ intervalles de
longueur $3^{-n}$, donc $\lambda(C) \leq (2/3)^n \to 0$. Un
ensemble nul non dénombrable --- la cardinalité ne voit pas
la \omterm{def:b3:measure:measure}{mesure}. Réciproquement, les \emph{ensembles de Cantor
gras} (\cref{exo:b3:measure:5}) sont d'intérieur vide et de
\omterm{def:b3:measure:measure}{mesure} positive~: la \omterm{def:b3:topology:topology}{topologie} ne voit pas non plus la
\omterm{def:b3:measure:measure}{mesure}. Le problème du week-end pousse ce jeu jusqu'à sa
conclusion frappante~: il existe des ensembles
Lebesgue-mesurables qui ne sont pas boréliens.
\end{example}

\begin{theorem}[Vitali]\label{thm:b3:measure:vitali}
Il n'existe pas de \omterm{def:b3:measure:measure}{mesure} sur \emph{tous} les sous-ensembles
de $\R$ qui soit invariante par translation et attribue à
chaque intervalle sa longueur. En particulier $\mathcal L
\neq \mathcal P(\R)$~: des ensembles non mesurables
existent.\index{ensemble non mesurable de Vitali}
\end{theorem}

\begin{proof}
Supposons $\mu$ une telle \omterm{def:b3:measure:measure}{mesure}. Sur $\intcc01$,
considérons l'équivalence $x \sim y \iff x - y \in \Q$~; par
l'\emph{axiome du choix} on prend un représentant dans
$\intcc01$ par classe~: un ensemble $V$. Pour $q \in
\Q\cap\intcc{-1}1$, les translatés $V + q$ sont deux à deux
disjoints (deux points de $V$ différant d'un rationnel
seraient équivalents mais des représentants distincts) et
\[
\intcc01 \subseteq \bigsqcup_{q \in \Q\cap\intcc{-1}1}(V + q)
\subseteq \intcc{-1}2 :
\]
la première inclusion car tout $x \in \intcc01$ diffère de
son représentant $v$ d'un rationnel $q = x - v \in
\intcc{-1}1$. Monotonie et $\sigma$-additivité donnent
\[
1 \leq \sum_{q}\mu(V + q) \leq 3,
\qquad\text{avec } \mu(V + q) = \mu(V) \text{ pour tout } q .
\]
Une somme infinie de la constante $\mu(V)$ vaut $0$ ou
$\infty$~: les deux bornes ne peuvent coexister. Donc aucune
telle $\mu$ n'existe --- et $V \notin \mathcal L$, puisque
$\lambda$ sur $\mathcal L$ a toutes les propriétés utilisées.
\end{proof}

\begin{remark}\label{rem:b3:measure:banachtarski}
En $\R^3$ l'échec est plus spectaculaire~: le paradoxe de
Banach--Tarski décompose une boule en cinq morceaux qui se
réassemblent, par rotations et translations, en
\emph{deux} boules de même rayon --- donc même une \omterm{def:b3:measure:measure}{mesure} de
volume finiment additive invariante par rotation sur tous les
sous-ensembles de $\R^3$ n'existe pas. Les morceaux sont,
bien sûr, non mesurables. La mesurabilité n'est pas une
précaution bureaucratique~; c'est la frontière de la
cohérence.
\end{remark}

\begin{method}\label{met:b3:measure:goodsets}
Le \emph{principe des bons ensembles}~: pour prouver que tous
les ensembles de $\sigma(\mathcal E)$ ont une propriété,
montrer que les bons ensembles forment une $\sigma$-algèbre
(ou un $\lambda$-système, si la propriété est de nature
mesurique et $\mathcal E$ un $\pi$-système --- alors Dynkin)
contenant $\mathcal E$. Presque toute preuve de ce chapitre
et du suivant en est un cas particulier. Pour prouver deux
\omterm{def:b3:measure:measure}{mesures} égales~: les vérifier sur un $\pi$-système engendrant
plus la $\sigma$-finitude
(\cref{thm:b3:measure:uniqueness}). Pour construire une
\omterm{def:b3:measure:measure}{mesure}~: construire une \omterm{def:b3:measure:outer}{mesure extérieure} par
recouvrements et invoquer Carathéodory.
\end{method}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:b3:measure:1}
(a) Montrer que $\{A \subseteq X : A$ ou $X\setminus A$ est
dénombrable$\}$ est une $\sigma$-algèbre~: celle engendrée
par les singletons.
(b) Montrer que $\mathcal B(\R)$ est engendrée par chacune
de~: les intervalles \omterm{def:b3:topology:topology}{ouverts}~; les intervalles fermés~; les
demi-droites $\intoc{-\infty}a$~; les demi-droites avec $a
\in \Q$.
(c) La famille des \emph{unions disjointes finies
d'intervalles} $\intoc ab$ est-elle une $\sigma$-algèbre~?
Une algèbre (stable par complémentation et unions finies)~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b3:measure:2}
(a) Prouver l'inclusion--exclusion pour une \omterm{def:b3:measure:measure}{mesure} finie~:
$\mu(A\cup B) = \mu(A) + \mu(B) - \mu(A\cap B)$, et la
version à trois ensembles.
(b) Donner un exemple montrant que la continuité par en haut
(\cref{prop:b3:measure:basics}(d)) échoue sans l'hypothèse
de finitude.
(c) Montrer qu'un ensemble dénombrable a \omterm{def:b3:measure:lebesgueouter}{mesure de Lebesgue}
nulle. En déduire $\lambda(\Q) = 0$ et
$\lambda(\intcc01\setminus\Q) = 1$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:measure:3}
Soient $\mu, \nu$ des \omterm{def:b3:measure:measure}{mesures} de probabilité sur $\mathcal
B(\R)$ avec $\mu(\intoc{-\infty}t) = \nu(\intoc{-\infty}t)$
pour tout $t \in \R$. Montrer $\mu = \nu$. (Cela fait de la
\emph{fonction de répartition} $F(t) = \mu(\intoc{-\infty}t)$
un invariant \omterm{def:b3:complete:complete}{complet} --- le fondement
du~\cref{ch:b3:probability}.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:measure:4}
(Borel--Cantelli, version \omterm{def:b3:measure:measure}{mesure}) Soient $(A_n)$ mesurables
avec $\sum_n\mu(A_n) < \infty$, et $\limsup A_n =
\bigcap_N\bigcup_{n\geq N}A_n$ (les points appartenant à une
infinité de $A_n$). Montrer $\mu(\limsup A_n) = 0$.
Application~: pour presque tout $x \in \intcc01$, seul un
nombre fini de $n$ satisfont $\abs{x - p/q_n} \leq 4^{-n}$
pour le $n$-ième rationnel $p/q_n$ d'une énumération de
$\Q\cap\intcc01$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:measure:5}
(Ensemble de Cantor gras) Reprendre la construction de
Cantor sur $\intcc01$, mais à l'étape $n$ retirer de chacun
des $2^{n-1}$ intervalles un intervalle \omterm{def:b3:topology:topology}{ouvert}
\emph{centré} de longueur $4^{-n}$ seulement. Montrer que
le $K = \bigcap K_n$ résultant est compact, d'intérieur vide
(aucun intervalle ne survit), et
\[
\lambda(K) = 1 - \sum_{n\geq1}2^{n-1}4^{-n} = \tfrac12 :
\]
un ensemble nulle part \omterm{def:b3:topology:interior}{dense} de \omterm{def:b3:measure:measure}{mesure} $\frac12$. En déduire
un sous-ensemble \emph{\omterm{rem:b3:complete:meagre}{maigre}} de $\intcc01$ de \omterm{def:b3:measure:measure}{mesure} pleine
$1$, et un \omterm{def:b3:topology:topology}{ouvert} \omterm{def:b3:topology:interior}{dense} de \omterm{def:b3:measure:measure}{mesure} $< \varepsilon$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:measure:6}
Soit $\mu$ une \omterm{def:b3:measure:measure}{mesure} sur $\mathcal B(\R)$, invariante par
translations, avec $c = \mu(\intoc01) < \infty$. Montrer
$\mu = c\,\lambda$ sur $\mathcal B(\R)$.
\emph{(Calculer $\mu$ sur les intervalles dyadiques en
divisant $\intoc01$ en $2^n$ translatés, puis invoquer
le~\cref{thm:b3:measure:uniqueness}.)}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:measure:7}
(Approximation) Soient $A \in \mathcal L$ avec $\lambda(A) <
\infty$ et $\varepsilon > 0$. Montrer qu'il existe une union
\emph{finie} d'intervalles $B$ avec $\lambda(A\,\triangle\,B)
< \varepsilon$ ($\triangle$ = différence symétrique).
\emph{(Régularité~: encadrer $K \subseteq A \subseteq U$ et
utiliser la structure de l'\omterm{def:b3:topology:topology}{ouvert} $U$ comme union
dénombrable d'intervalles, plus la compacité de $K$.)}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b3:measure:8}
(Steinhaus) Soit $A \in \mathcal L$ avec $\lambda(A) > 0$.
Montrer que $A - A = \{x - y : x, y \in A\}$ contient un
intervalle autour de $0$.
\emph{(Se ramener à $\lambda(A) < \infty$~; par une
régularité du style de l'\cref{exo:b3:measure:7}, trouver un
intervalle $I$ avec $\lambda(A \cap I) > \frac34\ell(I)$~;
alors pour $\abs t < \frac12\ell(I)$, les ensembles $A\cap I$
et $(A\cap I) + t$ sont tous deux dans un intervalle de
longueur $\frac32\ell(I)$ et ont \omterm{def:b3:measure:measure}{mesure} totale $>
\frac32\ell(I)$~: ils doivent s'intersecter.)}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b3:measure:9}
Montrer que tout $A \in \mathcal L$ avec $\lambda(A) > 0$
contient un sous-ensemble non mesurable. \emph{(Intersecter
$A$ avec les translatés $V + q$ de l'ensemble de Vitali~: si
tous les $A \cap (V+q)$ étaient mesurables, chacun serait
nul par l'argument du~\cref{thm:b3:measure:vitali} ---
Steinhaus (\cref{exo:b3:measure:8}) aide~: un ensemble
mesurable de \omterm{def:b3:measure:measure}{mesure} positive dans $V + q$ donnerait $(V+q) -
(V+q) \supseteq$ un intervalle, contredisant que cet ensemble
différence ne rencontre $\Q$ qu'en $0$~; conclure par
sous-additivité.)}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:measure:10}
Montrer qu'un $A \subseteq \R$ avec $\lambda^*(A) < \infty$
est Lebesgue-mesurable ssi pour tout $\varepsilon > 0$ il
existe un \omterm{def:b3:topology:topology}{ouvert} $U \supseteq A$ avec $\lambda^*(U \setminus
A) < \varepsilon$, ssi il existe un $G_\delta$ $G \supseteq
A$ avec $\lambda^*(G\setminus A) = 0$. (Donc les ensembles
de Lebesgue sont des boréliens modulo les ensembles nuls.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:measure:11}
(Continuité le long des limites monotones, et sa netteté)
(a) Montrer que pour des ensembles mesurables,
$\mu(\liminf A_n) \leq \liminf\mu(A_n)$ (Fatou pour les
ensembles), et que si $\mu\bigl(\bigcup A_n\bigr) < \infty$,
aussi $\limsup\mu(A_n) \leq \mu(\limsup A_n)$.
(b) Exhiber, pour la \omterm{def:b3:measure:lebesgueouter}{mesure de Lebesgue} sur $\R$, une suite
avec $\mu(A_n) = 1$ pour tout $n$ mais $\mu(\limsup A_n) =
0$~: l'hypothèse de finitude dans la seconde inégalité n'est
pas décorative.
(c) En déduire~: si $\sum\mu(A_n) < \infty$ alors
$\mu(\limsup A_n) = 0$ (Borel--Cantelli encore), et si les
$A_n$ croissent ou décroissent (avec $\mu(A_1) < \infty$
dans le cas décroissant), $\mu(\lim A_n) = \lim\mu(A_n)$.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b3:measure:12}
(Théorème d'Egorov) Soient $\mu(X) < \infty$ et $f_n \to f$
ponctuellement, toutes mesurables (à valeurs réelles). Pour
$k, N \geq 1$ poser
\[
E_{k,N} = \bigcap_{n \geq N}\Bigl\{x : \abs{f_n(x) - f(x)}
\leq \tfrac1k\Bigr\} .
\]
(a) Montrer que pour $k$ fixé, $E_{k,N} \nearrow X$ lorsque
$N \to \infty$, et en déduire des $N_k$ avec $\mu(X
\setminus E_{k,N_k}) \leq \varepsilon2^{-k}$.
(b) Conclure le \emph{théorème d'Egorov}~: pour tout
$\varepsilon > 0$ il existe un mesurable $A$ avec
$\mu(X\setminus A) \leq \varepsilon$ tel que $f_n \to f$
\emph{uniformément sur $A$} --- la convergence ponctuelle
est une convergence uniforme hors d'un ensemble
arbitrairement petit.
(c) Montrer que le théorème échoue sur $(\R, \lambda)$~: les
bosses mobiles $f_n = \mathbf 1_{\intcc n{n+1}}$ convergent
ponctuellement vers $0$ mais uniformément sur aucun
complémentaire d'un ensemble de \omterm{def:b3:measure:measure}{mesure} finie. Où (a) a-t-il
utilisé $\mu(X) < \infty$~?
\end{exercise}

\section{Problème~: l'escalier de Cantor--Vitali et un
ensemble mesurable non borélien}

\begin{problem}[{Problème du week-end --- l'escalier du
diable, et $\mathcal B(\R) \subsetneq \mathcal L$}]
\label{pb:b3:measure:1}
On construit la \emph{fonction de Cantor--Vitali} (escalier
du diable), on l'utilise pour transporter pathologiquement
la \omterm{def:b3:measure:measure}{mesure}, et on conclut par un théorème qu'aucun argument
mou n'offre~: il existe des ensembles Lebesgue-mesurables
qui ne sont pas boréliens. Notations~: $C$ est l'ensemble de
Cantor, $C_n$ son $n$-ième étage ($2^n$ intervalles de
longueur $3^{-n}$), et tout $x \in C$ a des chiffres
ternaires $x = \sum 2b_n3^{-n}$, $b_n \in \{0,1\}$
(\cref{exo:b3:topology:10}).

\medskip
\noindent\textbf{Partie I --- L'escalier.} Définir $c_0(x) =
x$ et $c_{n+1}$ à partir de $c_n$ par
\[
c_{n+1}(x) = \begin{cases}
\tfrac12\,c_n(3x) & 0 \leq x \leq \tfrac13,\\[2pt]
\tfrac12 & \tfrac13 \leq x \leq \tfrac23,\\[2pt]
\tfrac12 + \tfrac12\,c_n(3x - 2) & \tfrac23 \leq x \leq 1.
\end{cases}
\]
\begin{enumerate}
  \item Montrer que chaque $c_n$ est \omterm{def:b3:topology:continuity}{continue}, croissante, avec
        $c_n(0) = 0$, $c_n(1) = 1$, et que
        $\norm{c_{n+1} - c_n}_\infty \leq
        \tfrac12\norm{c_n - c_{n-1}}_\infty$.
  \item En déduire que $(c_n)$ converge uniformément vers une
        $c$ \omterm{def:b3:topology:continuity}{continue} croissante avec $c(0) = 0$, $c(1) = 1$
        (\emph{la} fonction de Cantor--Vitali), qui satisfait
        les mêmes relations auto-similaires que les $c_{n+1}$
        ci-dessus.
  \item Montrer que $c$ est constante sur chaque \omterm{def:b3:topology:components}{composante
        connexe} de $\intcc01\setminus C$, et que pour $x =
        \sum_n 2b_n3^{-n} \in C$, $c(x) = \sum_n b_n2^{-n}$~:
        l'escalier lit les chiffres de Cantor en binaire
        (la fonction $g$ du~\cref{pb:b3:topology:1}, rendue
        monotone et globale).
  \item En déduire que $c$ est dérivable, avec $c' = 0$, en
        tout point de $\intcc01\setminus C$~: $c' = 0$
        \emph{$\lambda$-presque partout}
        (\cref{ex:b3:measure:cantor}). Conclure que le
        théorème fondamental du calcul échoue pour $c$~:
        \[
        c(1) - c(0) = 1 \neq 0 = \int_0^1 c'(t)\,\dd t
        \]
        (l'intégrale portant sur l'ensemble de \omterm{def:b3:measure:measure}{mesure} pleine
        où $c' = 0$~; en anticipant le~\cref{ch:b3:lebesgue},
        les ensembles nuls n'affectent pas les intégrales).
        Quelle hypothèse du théorème fondamental $\mathcal
        C^1$ est violée~?
  \item Montrer que $c(C) = \intcc01$~: l'ensemble nul $C$ est
        envoyé \emph{sur} un ensemble de \omterm{def:b3:measure:measure}{mesure} pleine.
\end{enumerate}

\medskip
\noindent\textbf{Partie II --- L'\omterm{def:b3:topology:continuity}{homéomorphisme} tordu.} Soit
$h(x) = \frac{c(x) + x}{2}$.
\begin{enumerate}[resume]
  \item Montrer que $h \colon \intcc01 \to \intcc01$ est un
        \omterm{def:b3:topology:continuity}{homéomorphisme} (strictement croissant, continu,
        surjectif).
  \item Montrer que $\lambda\bigl(h(\intcc01\setminus C)\bigr)
        = \tfrac12$~: sur chaque trou de longueur $\ell$, $h$
        agit comme une application affine de pente
        $\tfrac12$, et les trous ont longueur totale $1$.
  \item En déduire $\lambda\bigl(h(C)\bigr) = \tfrac12$~:
        l'image homéomorphe d'un ensemble nul peut avoir
        \omterm{def:b3:measure:measure}{mesure} positive. (Où cela contredit-il l'intuition
        naïve sur la «~taille~»~?)
\end{enumerate}

\medskip
\noindent\textbf{Partie III --- Un ensemble mesurable non
borélien.}
\begin{enumerate}[resume]
  \item Par l'\cref{exo:b3:measure:9}, choisir un $W
        \subseteq h(C)$ non mesurable. Montrer que $Z =
        h^{-1}(W) \subseteq C$ est Lebesgue-mesurable.
        \emph{(C'est un sous-ensemble d'un ensemble nul~;
        complétude, \cref{thm:b3:measure:caratheodory}.)}
  \item Montrer que l'image réciproque d'un borélien par une
        \omterm{def:b3:topology:continuity}{application continue} est borélienne. \emph{(Principe
        des bons ensembles~: $\{B : h^{-1}(B) \in \mathcal
        B\}$ est une $\sigma$-algèbre contenant les \omterm{def:b3:topology:topology}{ouverts}
        --- attention au sens de l'application.)}
  \item Conclure que $Z$ n'est \emph{pas} borélien~: s'il
        l'était, $W = (h^{-1})^{-1}(Z)$ serait borélien
        (appliquer la question 10 au $h^{-1}$ continu), donc
        mesurable --- contradiction. Par conséquent
        \[
        \boxed{\ \mathcal B(\R) \subsetneq \mathcal L\ }
        \]
        et la complétude agrandit véritablement le monde
        borélien.
  \item Exhiber une fonction Lebesgue-mesurable $g$ et une
        fonction \omterm{def:b3:topology:continuity}{continue} $\varphi$ telles que $g \circ
        \varphi$ n'est pas Lebesgue-mesurable~: la
        mesurabilité, à la différence de la continuité, ne se
        compose pas. \emph{(Prendre $g = \mathbf 1_Z$ et
        $\varphi = h^{-1}$, en anticipant la définition des
        fonctions mesurables du~\cref{ch:b3:lebesgue}~: les
        images réciproques de boréliens sont des ensembles de
        Lebesgue. Où faut-il être prudent sur la
        $\sigma$-algèbre utilisée sur le but~?)}
\end{enumerate}

\medskip
\noindent\textbf{Partie IV --- Épilogue.}
\begin{enumerate}[resume]
  \item Ordonner les classes suivantes par inclusion stricte
        et justifier chaque stricte par un exemple de ce
        chapitre et de son problème~: ensembles dénombrables~;
        boréliens nuls~; ensembles de Lebesgue nuls~;
        boréliens~; ensembles de Lebesgue~; ensembles
        arbitraires.
\end{enumerate}

\medskip
\noindent\textbf{Partie V --- La \omterm{def:b3:measure:measure}{mesure} de Cantor~: masse sur
un ensemble nul.} L'escalier est la fonction de répartition
d'une \omterm{def:b3:measure:measure}{mesure} remarquable, que l'on construit maintenant avec
les outils du chapitre.
\begin{enumerate}[resume]
  \item (Lebesgue--Stieltjes, existence) Soit $F\colon\R\to\R$
        croissante, \omterm{def:b3:topology:continuity}{continue}, bornée. Sur les intervalles
        \omterm{def:b3:topology:topology}{semi-ouverts} définir $\rho\bigl(\intoc ab\bigr) =
        F(b) - F(a)$ et, pour $A \subseteq \R$,
        \[
        \mu_F^*(A) = \inf\Bigl\{\sum_k\bigl(F(b_k) -
        F(a_k)\bigr) : A \subseteq
        \bigcup_k\intoc{a_k}{b_k}\Bigr\} .
        \]
        Montrer que $\mu_F^*$ est une \omterm{def:b3:measure:outer}{mesure extérieure} et
        que $\mu_F^*\bigl(\intoc ab\bigr) = F(b) - F(a)$
        \emph{(imiter l'argument de compacité
        du~\cref{thm:b3:measure:lebesgue}, en élargissant
        chaque $\intoc{a_k}{b_k}$ en un intervalle \omterm{def:b3:topology:topology}{ouvert} à
        $F$-coût $\leq \varepsilon2^{-k}$ --- où utilise-t-on
        la continuité de $F$~?)}.
  \item Montrer que tout borélien est $\mu_F^*$-mesurable au
        sens de Carathéodory \emph{(comme dans le cas
        Lebesgue, il suffit de tester les demi-droites~;
        suivre la preuve de l'application
        du~\cref{thm:b3:measure:caratheodory})}, de sorte que
        $\mu_F = \mu_F^*$ restreinte à $\mathcal B(\R)$ est
        une \omterm{def:b3:measure:measure}{mesure} avec $\mu_F(\intoc ab) = F(b) - F(a)$~: la
        \emph{\omterm{def:b3:measure:measure}{mesure} de Lebesgue--Stieltjes} de $F$.
  \item Appliquer ceci à l'escalier ($F = c$ étendue par $0$
        sur $\R_-$ et $1$ sur $\intco1\infty$)~: la
        \emph{\omterm{def:b3:measure:measure}{mesure} de Cantor} $\mu$. Montrer $\mu(\R) = 1$,
        que chaque trou de l'ensemble de Cantor est
        $\mu$-nul ($c$ y est constante), et conclure
        \[
        \mu(C) = 1, \qquad \lambda(C) = 0 :
        \]
        $\mu$ et $\lambda$ vivent sur des supports disjoints
        ($C$ et son complémentaire). Deux \omterm{def:b3:measure:measure}{mesures} dans cette
        position sont dites \emph{mutuellement singulières},
        noté $\mu \perp \lambda$.
  \item Montrer que $\mu$ n'a pas d'atomes~: $\mu(\{x\}) = 0$
        pour tout $x$ \emph{(continuité de $c$)}. Une \omterm{def:b3:measure:measure}{mesure}
        de probabilité sans atome portée par un compact
        Lebesgue-nul~: comparer avec les seules \omterm{def:b3:measure:measure}{mesures} vues
        jusqu'ici.
  \item (Pile ou face déguisé) Pour un mot $(\varepsilon_1,
        \dots, \varepsilon_m) \in \{0,1\}^m$, soit
        $C_{\varepsilon}$ l'ensemble des $x \in C$ dont les
        chiffres ternaires satisfont $b_i(x) = \varepsilon_i$
        pour $i \leq m$ (l'une des $2^m$ pièces de Cantor de
        profondeur $m$). Montrer $\mu(C_\varepsilon) =
        2^{-m}$ \emph{(l'escalier monte de $2^{-m}$ sur cette
        pièce~: utiliser la partie I, question 3)}. La \omterm{def:b3:measure:measure}{mesure}
        de Cantor est la loi d'une suite infinie de piles ou
        faces équitables lue en ternaire ---
        le~\cref{ch:b3:probability} rendra cela exact.
  \item Prouver l'auto-similarité~: pour tout borélien $A$,
        \[
        \mu(A) = \tfrac12\,\mu(3A) +
        \tfrac12\,\mu(3A - 2) ,
        \]
        où $3A - 2 = \{3x - 2 : x \in A\}$ \emph{(vérifier sur
        les intervalles engendrant $\intoc ab$ via les
        relations auto-similaires de $c$, puis invoquer
        l'unicité, \cref{thm:b3:measure:uniqueness})}.
  \item Montrer que la réflexion $s(x) = 1 - x$ préserve
        $\mu$~: $\mu(s(A)) = \mu(A)$ \emph{(via $c(1 - x) = 1
        - c(x)$, qui suit de la symétrie de la construction
        --- le prouver)}.
  \item Calculer les deux premiers moments de $\mu$, c.-à-d.\
        d'un point aléatoire $X$ de loi $\mu$ (les intégrales
        peuvent être traitées comme limites de sommes sur les
        pièces de profondeur $m$, en anticipant
        le~\cref{ch:b3:lebesgue})~: la symétrie donne $\int
        x\,\dd\mu = \frac12$, et l'auto-similarité donne
        \[
        \int x^2\,\dd\mu = \frac38,
        \qquad\text{d'où}\qquad
        \operatorname{Var}(X) = \frac18 .
        \]
        Comparer avec la loi uniforme sur $\intcc01$
        (variance $\frac1{12}$)~: la masse de Cantor,
        poussée vers les bords, s'étale \emph{davantage}.
  \item Montrer que le \emph{support} (topologique) de $\mu$
        --- le plus petit fermé de \omterm{def:b3:measure:measure}{mesure} pleine --- est
        exactement $C$.
  \item (Synthèse) L'escalier $c$ est continu et croissant
        mais échoue au théorème fondamental du calcul (partie
        I)~; la \omterm{def:b3:measure:measure}{mesure} $\mu_c$ est une probabilité, sans
        atome, singulière par rapport à $\lambda$. Expliquer
        en un court paragraphe comment ce sont deux faces d'un
        même phénomène, et énoncer la morale générale~: les
        fonctions croissantes correspondent aux \omterm{def:b3:measure:measure}{mesures} ($F
        \leftrightarrow \mu_F$), la dérivabilité p.p.\
        correspond à la «~partie absolument \omterm{def:b3:topology:continuity}{continue}~», et
        $c$ est le témoin standard qu'un $F$ continu peut
        ne porter \emph{aucune} partie absolument \omterm{def:b3:topology:continuity}{continue}.
  \item (Le \omterm{def:b3:modules:module}{module} de continuité optimal) Soit $s =
        \frac{\ln 2}{\ln 3}$. Montrer que $c$ est
        höldérienne d'exposant $s$~:
        \[
        \abs{c(x) - c(y)} \leq 4\,\abs{x - y}^{s}
        \qquad (x, y \in \intcc01),
        \]
        et qu'aucun exposant $t > s$ ne peut convenir, même
        localement. En déduire la forme mesurique~: pour tout
        $x$ et tout $r \in \intoc01$,
        \[
        \mu\bigl(\intcc{x - r}{x + r}\bigr) \leq 8\,r^{s} .
        \]
        \emph{(Comparer une grille triadique de profondeur
        $m$ avec l'échelle de $\abs{x - y}$~; la question 18
        donne la montée sur chaque pièce. L'exposant $s$ est
        la dimension de Hausdorff de $C$, comme le diront
        des cours ultérieurs.)}
  \item (L'auto-similarité caractérise $\mu$) Prouver la
        réciproque de la question 19~: si $\nu$ est une
        \omterm{def:b3:measure:measure}{mesure} de probabilité sur $\mathcal B(\R)$ portée par
        $\intcc01$ et satisfaisant
        \[
        \nu(A) = \tfrac12\,\nu(3A) + \tfrac12\,\nu(3A - 2)
        \qquad (A \in \mathcal B(\R)),
        \]
        alors $\nu = \mu$. \emph{(Itérer la relation $m$
        fois pour étaler $\nu$ sur les $2^m$ pièces de Cantor
        de profondeur $m$, estimer $\nu(\intoc ab)$ contre le
        nombre de pièces dans $\intoc ab$, et faire $m \to
        \infty$~; conclure avec
        le~\cref{thm:b3:measure:uniqueness}.)}
\end{enumerate}
\end{problem}
