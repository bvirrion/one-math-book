\chapter{Fonctions holomorphes}\label{ch:b3:holomorphic}

La différentiabilité complexe ressemble à une petite variation
sur la théorie réelle --- une limite, un quotient. C'est en
réalité un univers différent. Une fonction différentiable
\emph{une seule fois} sur un \omterm{def:b3:topology:topology}{ouvert} de $\C$ est automatiquement
indéfiniment différentiable, analytique, déterminée sur tout un
domaine par ses valeurs près d'un seul point, et contrainte par
des principes globaux rigides (Liouville, \omterm{def:b3:modules:module}{module} maximum). Tout
cela découle d'un unique miracle, le théorème de Cauchy~: l'intégrale
d'une \omterm{def:b3:holomorphic:holo}{fonction holomorphe} le long d'un chemin fermé dans un domaine
étoilé s'annule. Ce chapitre démontre le miracle (argument de
Goursat, sans régularité au-delà de la différentiabilité), en
récolte les conséquences classiques, et règle un théorème que
ce livre utilise à crédit depuis le~\cref{ch:b3:galois}~: tout
polynôme complexe non constant a une racine.

Tout au long, $\Omega \subseteq \C$ est \omterm{def:b3:topology:topology}{ouvert}, et $D(a, r)$
désigne le disque \omterm{def:b3:topology:topology}{ouvert}.

\section{Différentiabilité complexe}

\begin{definition}\label{def:b3:holomorphic:holo}
$f \colon \Omega \to \C$ est \emph{holomorphe}\index{fonction
holomorphe} sur $\Omega$ si pour tout $z_0 \in \Omega$
\[
f'(z_0) = \lim_{h\to0}\frac{f(z_0 + h) - f(z_0)}{h}
\]
existe ($h \in \C^*$). Sommes, produits, quotients (dénominateurs
non nuls), compositions de fonctions holomorphes sont
holomorphes, avec les formules usuelles (les preuves de
première--deuxième année sont verbatim~: elles n'utilisent que
les opérations de corps et les limites). $\mathcal H(\Omega)$
désigne l'ensemble des fonctions holomorphes sur $\Omega$.
\end{definition}

\begin{proposition}[Cauchy--Riemann]
\label{prop:b3:holomorphic:cauchyriemann}
Écrivons $f(x + \iu y) = P(x,y) + \iu Q(x,y)$. Alors $f$ est
\omterm{def:b3:holomorphic:holo}{holomorphe} en $z_0$ si et seulement si $f$ est $\R$-différentiable
en $z_0$ (comme application de deux variables réelles)
et\index{équations de Cauchy--Riemann}
\[
\frac{\partial P}{\partial x} = \frac{\partial Q}{\partial y},
\qquad
\frac{\partial P}{\partial y} = -\frac{\partial Q}{\partial x}
\qquad \text{en } z_0 ;
\]
de façon équivalente, la différentielle réelle est la
multiplication par le nombre complexe $f'(z_0)$.
\end{proposition}

\begin{proof}
La $\C$-différentiabilité dit $f(z_0 + h) = f(z_0) + ch +
o(\abs h)$ avec $c = f'(z_0)$~: une différentielle $\R$-linéaire
qui est la multiplication par $c = a + \iu b$, c'est-à-dire de
matrice $\bigl(\begin{smallmatrix} a & -b\\ b & a
\end{smallmatrix}\bigr)$ dans la \omterm{def:b3:topology:basis}{base} $(1, \iu)$ --- exactement
les relations affichées pour les dérivées partielles. Réciproquement
une telle différentielle est $\C$-linéaire, et les définitions
en $o(\abs h)$ coïncident.
\end{proof}

\begin{example}\label{ex:b3:holomorphic:examples}
Les polynômes en $z$, les fractions rationnelles hors de leurs
pôles, et --- par le théorème de dérivation terme à terme des
séries entières de deuxième année, dont la preuve fonctionne
identiquement sur $\C$ --- toute somme d'une série entière
$\sum a_n(z - a)^n$ à l'intérieur de son disque de convergence~:
\omterm{def:b3:holomorphic:holo}{holomorphes}, de dérivée $\sum na_n(z - a)^{n-1}$ (même rayon).
En particulier $\exp z = \sum z^n/n!$ est entière (\omterm{def:b3:holomorphic:holo}{holomorphe}
sur $\C$) avec $\exp' = \exp$. En revanche $z \mapsto \bar z$,
$\abs z$, $\operatorname{Re}z$ ne sont nulle part \omterm{def:b3:holomorphic:holo}{holomorphes}
(Cauchy--Riemann échoue partout)~: l'holomorphie est une rigidité
qui préserve orientation et angles, non une \omterm{def:b3:groups:simple}{simple} régularité.
\end{example}

\section{Intégrales de contour}

\begin{definition}\label{def:b3:holomorphic:contour}
Un \emph{chemin} est une application $\mathcal C^1$ par morceaux
$\gamma \colon \intcc ab \to \C$~; il est \emph{fermé} si
$\gamma(a) = \gamma(b)$. Pour $f$ \omterm{def:b3:topology:continuity}{continue} sur l'image de
$\gamma$~:\index{intégrale de contour}
\[
\int_\gamma f(z)\,\dd z =
\int_a^b f(\gamma(t))\,\gamma'(t)\,\dd t,
\qquad
\Bigl|\int_\gamma f\,\dd z\Bigr| \leq
\sup_{\gamma}\abs f\cdot\operatorname{length}(\gamma)
\]
(l'\emph{inégalité ML}~; longueur $= \int_a^b\abs{\gamma'}$).
L'intégrale est invariante par reparamétrisation $\mathcal C^1$
croissante et change de signe par renversement d'orientation.
\end{definition}

\begin{proposition}[Primitives]\label{prop:b3:holomorphic:primitive}
Pour $f$ \omterm{def:b3:topology:continuity}{continue} sur $\Omega$, les assertions suivantes sont
équivalentes~: (i) $f$ admet une \emph{primitive} $F \in
\mathcal H(\Omega)$ ($F' = f$)~; (ii) $\int_\gamma f\,\dd z =
0$ pour tout chemin fermé $\gamma$ dans $\Omega$. Dans ce cas
$\int_\gamma f\,\dd z = F(\gamma(b)) - F(\gamma(a))$ pour tout
chemin.
\end{proposition}

\begin{proof}
(i)$\Rightarrow$~: $\frac{\dd}{\dd t}F(\gamma(t)) =
F'(\gamma(t))\gamma'(t)$ (règle de la chaîne, valide par
morceaux), donc l'intégrale se téléscope en la différence des
extrémités~; les chemins fermés donnent $0$. (ii)$\Rightarrow$(i)~:
fixons $z_*$ dans une \omterm{def:b3:topology:components}{composante connexe}, définissons $F(z) =
\int_{\gamma_z}f\,\dd z$ le long d'un chemin quelconque de $z_*$
à $z$ (bien défini~: deux chemins diffèrent d'un chemin fermé)~;
pour $h$ petit, en prenant le segment de $z$ à $z + h$,
\[
\frac{F(z + h) - F(z)}{h} - f(z)
= \frac1h\int_{[z, z+h]}\bigl(f(w) - f(z)\bigr)\dd w
\longrightarrow 0
\]
par l'inégalité ML et la continuité de $f$ en $z$.
\end{proof}

\begin{definition}[Nombre de tours]\label{def:b3:holomorphic:index}
Pour un chemin fermé $\gamma$ et $z \notin
\operatorname{im}\gamma$, l'\emph{indice}\index{nombre de
tours}\index{indice d'un chemin} est
\[
\operatorname{Ind}_\gamma(z) = \frac1{2\iu\pi}
\int_\gamma\frac{\dd w}{w - z} .
\]
C'est un \emph{entier}~: en posant $\varphi(t) =
\int_a^t\frac{\gamma'(s)}{\gamma(s) - z}\dd s$, la fonction
$(\gamma(t) - z)\eu^{-\varphi(t)}$ a une dérivée nulle (par
morceaux), donc est constante~; en $t = b$, $\eu^{\varphi(b)} =
\frac{\gamma(b) - z}{\gamma(a) - z} = 1$, d'où $\varphi(b) \in
2\iu\pi\Z$. Comme fonction de $z$, l'indice est continu sur
$\C\setminus\operatorname{im}\gamma$ (convergence dominée),
donc constant sur chaque \omterm{def:b3:topology:components}{composante connexe}, et $0$ sur la
composante non bornée (ML~: l'intégrale tend vers $0$ quand
$z \to \infty$). Pour le cercle $\gamma(t) = a + r\eu^{\iu t}$,
$t \in \intcc0{2\pi}$~: $\operatorname{Ind}_\gamma(z) = 1$ pour
$z \in D(a,r)$ (calculer en $z = a$~: $\frac1{2\iu\pi}
\int_0^{2\pi}\frac{r\iu\eu^{\iu t}}{r\eu^{\iu t}}\dd t = 1$~;
la constance fait le reste).
\end{definition}

\section{Théorème de Cauchy}

\begin{theorem}[Goursat]\label{thm:b3:holomorphic:goursat}
Soit $f \in \mathcal H(\Omega)$ et $T \subseteq \Omega$ un
triangle plein fermé. Alors $\int_{\partial T}f\,\dd z = 0$
(bord parcouru une fois, orientation quelconque).\index{théorème
de Goursat}
\end{theorem}

\begin{proof}
Posons $I(T) = \int_{\partial T}f\,\dd z$. En joignant les
milieux des côtés, on découpe $T$ en quatre triangles de
demi-taille $T^{(1)}, \dots, T^{(4)}$, et les arêtes intérieures
s'annulent deux à deux~: $I(T) = \sum_iI(T^{(i)})$. Choisissons
$T_1$ parmi eux avec $\abs{I(T_1)} \geq \frac14\abs{I(T)}$, et
itérons~: une suite emboîtée $T \supseteq T_1 \supseteq T_2
\supseteq\cdots$ avec
\[
\abs{I(T_n)} \geq 4^{-n}\abs{I(T)},
\qquad
\operatorname{diam}T_n = 2^{-n}\operatorname{diam}T,
\quad
\operatorname{length}(\partial T_n) =
2^{-n}\operatorname{length}(\partial T).
\]
L'intersection $\bigcap T_n$ est un unique point $z_0$
(compacts emboîtés de diamètres tendant vers $0$,
le~\cref{thm:b3:topology:compactprops}(3)). Différentiabilité
en $z_0$~: donné $\varepsilon$, pour $n$ grand, sur $T_n$,
\[
f(z) = f(z_0) + f'(z_0)(z - z_0) + R(z),
\qquad \abs{R(z)} \leq \varepsilon\abs{z - z_0} \leq
\varepsilon\operatorname{diam}T_n .
\]
La partie affine a une primitive~: son intégrale sur le fermé
$\partial T_n$ s'annule (la~\cref{prop:b3:holomorphic:primitive}),
il reste
\[
\abs{I(T_n)} = \Bigl|\int_{\partial T_n}R\Bigr|
\leq \varepsilon\operatorname{diam}(T_n)\,
\operatorname{length}(\partial T_n)
= \varepsilon\,4^{-n}\operatorname{diam}(T)
\operatorname{length}(\partial T) .
\]
Comparant avec $\abs{I(T_n)} \geq 4^{-n}\abs{I(T)}$~:
$\abs{I(T)} \leq \varepsilon\cdot\text{const}$ pour tout
$\varepsilon$~: $I(T) = 0$.
\end{proof}

\begin{theorem}[Théorème de Cauchy, version étoilée]
\label{thm:b3:holomorphic:cauchy}
Soit $\Omega$ \emph{étoilé} par rapport à $c$ (tout segment
$[c, z]$, $z \in \Omega$, est \omterm{def:b3:rings:content}{contenu} dans $\Omega$) --- p.\
ex.\ convexe. Toute $f \in \mathcal H(\Omega)$ admet une
primitive sur $\Omega$~; par conséquent $\int_\gamma f\,\dd z
= 0$ pour \emph{tout} chemin fermé $\gamma$ dans
$\Omega$.\index{théorème de Cauchy}
\end{theorem}

\begin{proof}
Définissons $F(z) = \int_{[c,z]}f\,\dd w$. Pour $z, z + h \in
\Omega$ avec $[z, z+h] \subseteq \Omega$ (vrai pour $h$ petit),
le triangle de sommets $c, z, z+h$ est \omterm{def:b3:rings:content}{contenu} dans $\Omega$
(étoilé~: chacun de ses points est sur un segment $[c, w]$ avec
$w \in [z, z+h] \subseteq \Omega$)~: Goursat donne
\[
F(z + h) - F(z) = \int_{[z, z+h]}f\,\dd w,
\]
et le calcul du quotient différentiel de
la~\cref{prop:b3:holomorphic:primitive} fournit $F' = f$.
L'annulation des intégrales sur les chemins fermés suit de la
même proposition.
\end{proof}

\begin{theorem}[Formule intégrale de Cauchy]
\label{thm:b3:holomorphic:formula}
Soit $f \in \mathcal H(\Omega)$, $\bar D(a, r) \subseteq
\Omega$, et $C_r$ le cercle $\partial D(a,r)$ parcouru une fois
dans le sens trigonométrique. Alors pour tout $z \in D(a, r)$~:
\index{formule intégrale de Cauchy}
\[
f(z) = \frac1{2\iu\pi}\int_{C_r}\frac{f(w)}{w - z}\,\dd w .
\]
\end{theorem}

\begin{proof}
Fixons $z$ et définissons sur $\Omega$
\[
g(w) = \begin{cases}
\dfrac{f(w) - f(z)}{w - z} & w \neq z,\\[4pt]
f'(z) & w = z :
\end{cases}
\]
$g$ est \omterm{def:b3:topology:continuity}{continue} sur $\Omega$ et \omterm{def:b3:holomorphic:holo}{holomorphe} hors de $z$. Goursat
vaut pour $g$ sur tout triangle $T \subseteq \Omega'$, où
$\Omega'$ est un disque légèrement plus grand que $\bar D(a,r)$
dans $\Omega$~: si $z \notin T$, directement~; si $z \in T$,
découper $T$ en petits triangles ayant $z$ pour sommet plus des
triangles évitant $z$~; sur un triangle de sommet $z$ la borne
ML donne $\abs{\int_{\partial T'}g} \leq \sup_{T'}\abs
g\cdot\operatorname{length} \to 0$ quand le triangle se
rétrécit, et les morceaux restants s'annulent par Goursat ---
d'où $\int_{\partial T}g = 0$ dans tous les cas. La preuve
du~\cref{thm:b3:holomorphic:cauchy} n'utilisait que cette
propriété triangulaire~: $g$ a une primitive sur le convexe
$\Omega'$, donc $\int_{C_r}g = 0$, c'est-à-dire
\[
\frac1{2\iu\pi}\int_{C_r}\frac{f(w)}{w - z}\dd w
= f(z)\,\frac{1}{2\iu\pi}\int_{C_r}\frac{\dd w}{w - z}
= f(z)\operatorname{Ind}_{C_r}(z) = f(z) .
\]
\end{proof}

\section{Analyticité et sa cascade}

\begin{theorem}[Holomorphe $=$ analytique]
\label{thm:b3:holomorphic:analytic}
Soit $f \in \mathcal H(\Omega)$ et $D(a, R) \subseteq \Omega$.
Alors\index{analyticité des fonctions holomorphes}
\[
f(z) = \sum_{n\geq0}c_n\,(z - a)^n \quad \text{sur } D(a, R),
\qquad
c_n = \frac1{2\iu\pi}\int_{C_r}\frac{f(w)}{(w -
a)^{n+1}}\,\dd w \ \ (0 < r < R),
\]
les coefficients étant indépendants de $r$. Par conséquent $f$
est indéfiniment $\C$-différentiable, $c_n = f^{(n)}(a)/n!$, et
les \emph{estimations de Cauchy}\index{estimations de Cauchy}
valent~:
\[
\abs{c_n} \;\leq\; \frac{\sup_{\abs{w - a} =
r}\abs{f(w)}}{r^{n}} .
\]
\end{theorem}

\begin{proof}
Pour $\abs{z - a} < r$~: développer le noyau de Cauchy en série
géométrique
\[
\frac1{w - z} = \frac1{(w - a)\bigl(1 - \frac{z - a}{w -
a}\bigr)} = \sum_{n\geq0}\frac{(z - a)^n}{(w - a)^{n+1}},
\]
normalement convergente en $w$ sur $C_r$ ($\abs{\frac{z-a}{w-a}} =
\frac{\abs{z-a}}r < 1$)~: intégrer terme à terme contre
$\frac{f(w)}{2\iu\pi}$ (la convergence uniforme justifie
l'interversion) et appliquer le~\cref{thm:b3:holomorphic:formula}.
Une série entière est indéfiniment différentiable avec $c_n =
f^{(n)}(a)/n!$ (deuxième année), ce qui montre aussi que les
$c_n$ ne dépendent pas de $r$. Les estimations~: borner
l'intégrale des coefficients par ML.
\end{proof}

\begin{example}[Les singularités dictent les rayons]
\label{ex:b3:holomorphic:radius}
Pourquoi l'innocente fonction réelle $\frac1{1 + x^2}$ a-t-elle
une série de Taylor en $x = 3$ ne convergeant que pour $\abs{x
- 3} < \sqrt{10}$, alors que rien ne se passe de mal sur la
droite réelle~? Parce que le théorème ci-dessus rend le rayon
de convergence en $a$ égal à la distance de $a$ au point le
plus proche où l'holomorphie échoue. Ici $f(z) = \frac1{1 +
z^2}$ est \omterm{def:b3:holomorphic:holo}{holomorphe} exactement sur $\C\setminus\{\pm\iu\}$,
donc le développement en $a = 3$ converge sur le plus grand
disque évitant $\pm\iu$, de rayon $\abs{3 - \iu} = \sqrt{10}$
--- et ne peut converger sur un plus grand, car la somme
étendrait $f$ holomorphiquement à un \omterm{def:b3:topology:topology}{voisinage} de $\pm\iu$, où
$\abs f \to \infty$. La théorie réelle voit le mystérieux rayon
$\sqrt{10}$~; le plan complexe voit deux pôles. Règle pratique~:
\emph{pour trouver un rayon de convergence, localiser les
singularités} --- p.\ ex.\ la série de Taylor de $\tan$ en $0$
a pour rayon $\frac\pi2$ (zéros les plus proches de $\cos$), et
la fonction génératrice de Bernoulli $\frac z{\eu^z - 1}$
(le~\cref{pb:b3:holomorphic:1}, partie~VI) a pour rayon $2\pi$
(zéros non nuls les plus proches de $\eu^z - 1$~: $\pm2\iu\pi$).
\end{example}

\begin{corollary}[Liouville~; d'Alembert--Gauss]
\label{cor:b3:holomorphic:liouville}
Une fonction entière bornée est constante. Par conséquent tout
polynôme non constant sur $\C$ a une racine~: \emph{$\C$ est
\omterm{def:b3:galois:closure}{algébriquement clos}}.\index{théorème de Liouville}\index{théorème
fondamental de l'algèbre}
\end{corollary}

\begin{proof}
Si $\abs f \leq M$ sur $\C$~: pour tout $a$ et $r$, $\abs{c_1(a)}
= \abs{f'(a)} \leq M/r \to 0$~: $f' \equiv 0$, et $f$ est
constante (sur le \omterm{def:b3:topology:connected}{connexe} $\C$~: dérivée nulle implique
localement constante --- intégrer le long des segments). Si $P$
n'avait pas de racine, $1/P$ serait entière et bornée ($\abs{P(z)}
\to \infty$ quand $\abs z \to \infty$~: le terme dominant
domine, donc $\abs{1/P}$ est petit hors d'un grand disque et
continu sur le disque compact)~: constante --- absurde pour $P$
non constant. (Le problème de week-end donne une seconde preuve
élémentaire et les conséquences \omterm{def:b3:galois:algebraic}{algébriques}.)
\end{proof}

\begin{theorem}[Zéros isolés~; théorème d'identité]
\label{thm:b3:holomorphic:identity}
Soit $\Omega$ \emph{\omterm{def:b3:topology:connected}{connexe}} et $f \in \mathcal H(\Omega)$,
$f \not\equiv 0$. Alors tout zéro $a$ de $f$ a un \emph{ordre}
fini~: $f(z) = (z - a)^m\,g(z)$ avec $g \in \mathcal
H(\Omega)$, $g(a) \neq 0$, et les zéros de $f$ n'ont pas de
point d'accumulation dans $\Omega$. Par conséquent, si deux
\omterm{def:b3:holomorphic:holo}{fonctions holomorphes} sur $\Omega$ coïncident sur un ensemble
ayant un point d'accumulation dans $\Omega$, elles coïncident
partout.\index{théorème d'identité}\index{zéros isolés}
\end{theorem}

\begin{proof}
Soit $Z$ l'ensemble des points où toutes les dérivées de $f$
s'annulent. $Z$ est fermé (intersection de fermés) et \omterm{def:b3:topology:topology}{ouvert}~:
si tous les $c_n = 0$ en $a$, le développement en série entière
rend $f \equiv 0$ sur un disque autour de $a$. Connexité~: $Z =
\varnothing$ ou $Z = \Omega$~; ce dernier est exclu par $f
\not\equiv 0$. Donc en un zéro $a$, un coefficient est non nul~:
soit $m$ minimal avec $c_m \neq 0$~; alors $f(z) = (z -
a)^m\sum_{n\geq m}c_n(z-a)^{n-m}$ sur un disque, et la somme
définit $g$ \omterm{def:b3:holomorphic:holo}{holomorphe} près de $a$ avec $g(a) = c_m \neq 0$~;
étendre $g = f/(z-a)^m$ hors de $a$ (\omterm{def:b3:holomorphic:holo}{holomorphe} là). Comme
$g(a) \neq 0$ et $g$ est \omterm{def:b3:topology:continuity}{continue}, $f$ n'a pas d'autre zéro
dans un \omterm{def:b3:topology:topology}{voisinage} de $a$~: les zéros sont isolés, et un ensemble
de points isolés n'a pas de point d'accumulation dans $\Omega$
(un point d'accumulation de zéros est un zéro --- continuité ---
et ne serait pas isolé). Identité~: appliquer à la différence,
dont l'ensemble des zéros a un point d'accumulation, ce qui
force la branche $Z = \Omega$.
\end{proof}

\begin{theorem}[Valeur moyenne et module maximum]
\label{thm:b3:holomorphic:maximum}
Soit $f \in \mathcal H(\Omega)$.
\begin{enumerate}
  \item (Valeur moyenne) Pour $\bar D(a, r) \subseteq \Omega$~:
        $f(a) = \frac1{2\pi}\int_0^{2\pi}f(a +
        r\eu^{\iu t})\,\dd t$.
  \item (Principe du maximum) Si $\Omega$ est \omterm{def:b3:topology:connected}{connexe} et
        $\abs f$ atteint un maximum local en un point de
        $\Omega$, alors $f$ est constante. Par conséquent,
        pour $\Omega$ borné et $f$ \omterm{def:b3:topology:continuity}{continue} sur
        $\bar\Omega$~: $\sup_{\bar\Omega}\abs f =
        \sup_{\partial\Omega}\abs f$.\index{principe du maximum}
\end{enumerate}
\end{theorem}

\begin{proof}
(1) est la formule de Cauchy au \omterm{ex:b3:groups:actions}{centre}~: paramétrer $C_r$.
(2) Supposons $\abs f \leq \abs{f(a)}$ sur $\bar D(a, \rho)
\subseteq \Omega$. Si $f(a) = 0$, $f \equiv 0$ près de $a$.
Sinon, pour $0 < r \leq \rho$, la valeur moyenne donne
\[
\abs{f(a)} \leq \frac1{2\pi}\int_0^{2\pi}\abs{f(a +
r\eu^{\iu t})}\,\dd t \leq \abs{f(a)} :
\]
l'intégrande \omterm{def:b3:topology:continuity}{continue} non négative $\abs{f(a)} - \abs{f(a +
r\eu^{\iu t})}$ a moyenne nulle, donc s'annule~: $\abs f$ est
\emph{constante} $= \abs{f(a)} \ne 0$ sur le disque. Une
\omterm{def:b3:holomorphic:holo}{fonction holomorphe} de \omterm{def:b3:modules:module}{module} non nul constant sur un disque
est constante~: dériver $P^2 + Q^2 = \text{const}$ donne
$PP_x + QQ_x = 0$ et $PP_y + QQ_y = 0$~; en substituant les
relations de Cauchy--Riemann $P_y = -Q_x$, $Q_y = P_x$ dans la
seconde équation on obtient le système linéaire
\[
P\,P_x + Q\,Q_x = 0, \qquad -P\,Q_x + Q\,P_x = 0,
\]
dont le déterminant est $P^2 + Q^2 \neq 0$~: $P_x = Q_x = 0$,
donc $f' = P_x + \iu Q_x = 0$ sur le disque~: $f$ constante
là. Le théorème d'identité propage la constance à tout
$\Omega$. Forme au bord~: $\abs f$ atteint son sup sur le
compact $\bar\Omega$~; un maximum intérieur rend $f$ constante,
et le sup est atteint sur le bord dans tous les cas.
\end{proof}

\begin{theorem}[Théorème de convergence de Weierstrass]
\label{thm:b3:holomorphic:weierstrassconv}
Si $f_n \in \mathcal H(\Omega)$ convergent vers $f$ uniformément
sur tout sous-ensemble compact de $\Omega$, alors $f \in
\mathcal H(\Omega)$ et $f_n^{(k)} \to f^{(k)}$ uniformément
sur les compacts, pour tout $k$.\index{théorème de convergence
de Weierstrass}
\end{theorem}

\begin{proof}
$f$ est \omterm{def:b3:topology:continuity}{continue}. Pour tout triangle fermé $T \subseteq
\Omega$~: $\int_{\partial T}f = \lim\int_{\partial T}f_n = 0$
(convergence uniforme sur le compact $\partial T$~; Goursat
pour les $f_n$). Par l'argument du~\cref{thm:b3:holomorphic:cauchy},
$f$ a des primitives locales (les disques sont convexes~; seule
la propriété triangulaire a été utilisée), c'est-à-dire $f =
F'$ localement avec $F$ \omterm{def:b3:holomorphic:holo}{holomorphe}~; $F$ est analytique
(le~\cref{thm:b3:holomorphic:analytic}), donc aussi $f = F'$~:
\omterm{def:b3:holomorphic:holo}{holomorphe}. (C'est le \emph{théorème de
Morera}\index{théorème de Morera}~: \omterm{def:b3:topology:continuity}{continue} à intégrales
triangulaires nulles implique \omterm{def:b3:holomorphic:holo}{holomorphe}.) Dérivées~: pour
$\bar D(a, 2r) \subseteq \Omega$ et $z \in \bar D(a, r)$,
la formule de Cauchy pour les dérivées (dériver
le~\cref{thm:b3:holomorphic:formula} sous l'intégrale, ou
utiliser la formule des coefficients) donne
\[
\abs{f_n'(z) - f'(z)} =
\Bigl|\frac{1}{2\iu\pi}\int_{C_{2r}}\frac{f_n(w) - f(w)}{(w -
z)^2}\,\dd w\Bigr|
\leq \frac{2r\,\sup_{C_{2r}}\abs{f_n - f}}{r^2} \to 0
\]
uniformément sur $\bar D(a,r)$~; recouvrir un compact par un
nombre fini de tels disques, et itérer pour les $k$ plus
élevés.
\end{proof}

\begin{method}\label{met:b3:holomorphic:toolkit}
La boîte à outils quotidienne. Pour prouver qu'une fonction
est \omterm{def:b3:holomorphic:holo}{holomorphe}~: l'exhiber comme série entière, composition,
limite localement uniforme (le~\cref{thm:b3:holomorphic:weierstrassconv}),
ou intégrale à paramètre \omterm{def:b3:holomorphic:holo}{holomorphe} (l'\cref{exo:b3:holomorphic:7}
--- dériver sous $\int$ ou appliquer Morera--Fubini). Pour
prouver des identités~: les prouver sur un segment ou un
sous-domaine et invoquer le théorème d'identité. Pour borner~:
\omterm{thm:b3:holomorphic:analytic}{estimations de Cauchy} sur le plus grand cercle disponible. Pour
prouver constance/non-existence~: Liouville ou le principe du
maximum. Toujours savoir \emph{où} la fonction est \omterm{def:b3:holomorphic:holo}{holomorphe} et
\emph{quels} disques tiennent dans $\Omega$.
\end{method}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:b3:holomorphic:1}
(a) En quels points $z \mapsto \bar z$, $\abs z^2$,
$\operatorname{Re}z$ sont-elles complexe-différentiables~?
\omterm{def:b3:holomorphic:holo}{Holomorphes} sur un \omterm{def:b3:topology:topology}{ouvert}~?
(b) Montrer que $P(x, y) = x^2 - y^2$ est la partie réelle
d'une \omterm{def:b3:holomorphic:holo}{fonction holomorphe} sur $\C$, trouvée explicitement, et
déterminer \emph{toutes} ces fonctions.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b3:holomorphic:2}
Calculer à partir des définitions~: $\int_{C}z^n\,\dd z$ pour
tout $n \in \Z$, $C$ le cercle unité~; $\int_\gamma\bar z\,\dd
z$ le long du segment $[0, 1+\iu]$ et le long du chemin en
deux segments passant par $1$~: conclure que $\bar z$ n'a pas
de primitive sur aucun \omterm{def:b3:topology:topology}{voisinage} de ces chemins.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:holomorphic:3}
(a) Montrer que le logarithme principal $\log z = \ln\abs z +
\iu\arg z$ ($\arg \in \intoo{-\pi}\pi$) est \omterm{def:b3:holomorphic:holo}{holomorphe} sur
$\C\setminus\intoc{-\infty}0$ de dérivée $\frac1z$
\emph{(primitive de $\frac1z$ sur le plan découpé étoilé~:
le~\cref{thm:b3:holomorphic:cauchy}~; fixer la constante)}.
(b) Montrer qu'aucun logarithme continu n'existe sur $\C^*$
\emph{(son obstruction sans dérivée~: l'\omterm{def:b3:holomorphic:index}{indice} du cercle
unité)}.
(c) Développer $\log(1 + z)$ en série entière sur $D(0,1)$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:holomorphic:4}
(a) Soit $f$ entière avec $\abs{f(z)} \leq C(1 + \abs z)^{n}$.
Montrer que $f$ est un polynôme de \omterm{def:b3:galois:extension}{degré} $\leq n$
\emph{(\omterm{thm:b3:holomorphic:analytic}{estimations de Cauchy} sur de grands cercles)}.
(b) Soit $f$ entière avec $\operatorname{Re}f$ bornée
supérieurement. Montrer que $f$ est constante \emph{(considérer
$\eu^{f}$)}.
(c) En déduire le «~petit Picard pour les applications
affines~»~: une fonction entière omettant un demi-plan est
constante.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:holomorphic:5}
(a) Soit $f$ \omterm{def:b3:holomorphic:holo}{holomorphe} sur un $\Omega$ \omterm{def:b3:topology:connected}{connexe} $\ni 0$ avec
$f(\frac1n) = \frac1{n^2}$ pour tout $n$ assez grand.
Déterminer $f$.
(b) Existe-t-il une $f$ \omterm{def:b3:holomorphic:holo}{holomorphe} sur $\C^*$ satisfaisant
$f(\frac1n) = \frac{(-1)^n}{n}$ pour tout $n \geq 1$~?
Justifier.
(c) Exhiber deux \omterm{def:b3:holomorphic:holo}{fonctions holomorphes} distinctes sur
$D(0,1)\sqcup D(3,1)$ coïncidant sur $D(0,1)$~: où le théorème
d'identité utilise-t-il la connexité~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:holomorphic:6}
Soit $f$ \omterm{def:b3:holomorphic:holo}{holomorphe} sur le disque unité \omterm{def:b3:topology:topology}{ouvert} $\mathbb D$,
\omterm{def:b3:topology:continuity}{continue} sur $\bar{\mathbb D}$, avec $\abs f \equiv 1$ sur le
cercle bord.
(a) Si $f$ n'a pas de zéro dans $\mathbb D$, montrer que $f$
est constante \emph{(appliquer le principe du maximum à $f$ et
à $1/f$)}.
(b) Donner un exemple avec un zéro où $f$ n'est pas constante.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:holomorphic:7}
(Holomorphie sous l'intégrale) Soit $\mu$ une \omterm{def:b3:measure:measure}{mesure} finie sur
un espace $X$ et $g \colon X\times\Omega \to \C$ avec~: $g(x,
\cdot) \in \mathcal H(\Omega)$ pour chaque $x$, $g$ \omterm{def:b3:lebesgue:measurable}{mesurable}
en $x$, et $\abs g \leq h(x)$ avec $h$ \omterm{def:b3:lebesgue:l1}{intégrable}, localement
uniformément en $z$. Montrer que $G(z) = \int_Xg(x, z)\dd\mu(x)$
est \omterm{def:b3:holomorphic:holo}{holomorphe} sur $\Omega$. \emph{(Morera~: les intégrales
triangulaires s'annulent par Fubini et Goursat~; continuité par
convergence dominée. Puis appliquer à $\Gamma(z) =
\int_0^\infty t^{z-1}\eu^{-t}\dd t$ sur $\{\operatorname{Re}z >
0\}$.)}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b3:holomorphic:8}
(Gauss--Lucas) Soit $P \in \C[X]$ non constant. Montrer que
toute racine de $P'$ est dans l'enveloppe convexe des racines
de $P$. \emph{(Écrire $\frac{P'}{P} = \sum_k\frac{m_k}{z -
a_k}$ en un zéro $z$ de $P'$ qui n'est pas zéro de $P$,
prendre les conjugués, et lire une combinaison convexe.)}
Illustrer sur $P = z^3 - 1$.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b3:holomorphic:9}
Soit $f$ entière et doublement périodique~: $f(z + 1) = f(z +
\iu) = f(z)$ pour tout $z$. Montrer que $f$ est constante.
\emph{(Borner $f$ sur le carré fondamental compact, puis
partout~; Liouville.)} Morale~: les fonctions elliptiques non
constantes doivent avoir des pôles --- le thème
du~\cref{ch:b3:residues}.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:holomorphic:10}
(a) Montrer que $P = \operatorname{Re}f$ d'une $f$ \omterm{def:b3:holomorphic:holo}{holomorphe}
satisfait la propriété de la moyenne $P(a) =
\frac1{2\pi}\int_0^{2\pi}P(a + r\eu^{\iu t})\dd t$ et est
\emph{harmonique}~: $\partial^2_{xx}P + \partial^2_{yy}P = 0$
(dériver Cauchy--Riemann~; utiliser
le~\cref{thm:b3:holomorphic:analytic} pour la régularité
requise).
(b) En déduire le principe du maximum pour les parties réelles
de \omterm{def:b3:holomorphic:holo}{fonctions holomorphes} sur des domaines bornés.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b3:holomorphic:11}
(Réflexion de Schwarz) Soit $\Omega^+ = \{z : \abs z < 1,\
\operatorname{Im}z > 0\}$, $I = \intoo{-1}1$, et $f$
\omterm{def:b3:holomorphic:holo}{holomorphe} sur $\Omega^+$, \omterm{def:b3:topology:continuity}{continue} sur $\Omega^+\cup I$,
\emph{à valeurs réelles sur $I$}. Définir
\[
F(z) = \begin{cases} f(z) & z \in \Omega^+\cup I,\\
\overline{f(\bar z)} & \bar z \in \Omega^+ . \end{cases}
\]
(a) Montrer que $F$ est bien définie et \omterm{def:b3:topology:continuity}{continue} sur $\Omega
= \Omega^+\cup I\cup\Omega^-$, et \omterm{def:b3:holomorphic:holo}{holomorphe} sur $\Omega^\pm$
\emph{(pour $\Omega^-$~: vérifier Cauchy--Riemann pour
$\overline{f(\bar z)}$, ou développer $f$ en série entière
locale et conjuguer les coefficients)}.
(b) Montrer que $F$ est \omterm{def:b3:holomorphic:holo}{holomorphe} sur tout $\Omega$ par le
critère de Morera~: $\int_{\partial T}F = 0$ pour tout triangle
$T \subseteq \Omega$ \emph{(découper les triangles en $I$ et
pousser leurs côtés horizontaux hors de l'axe de $\varepsilon$,
en utilisant la continuité uniforme)}.
(c) En déduire~: une \omterm{def:b3:holomorphic:holo}{fonction holomorphe} sur le disque, réelle
sur un diamètre, satisfait $f(\bar z) = \overline{f(z)}$~; et
une \omterm{def:b3:holomorphic:holo}{fonction holomorphe} non constante ne peut être à valeurs
réelles sur aucun \omterm{def:b3:topology:topology}{ouvert} non vide de son domaine (\omterm{def:b3:topology:connected}{connexe}).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:holomorphic:12}
(L'équation de Pythagore complexe) Trouver tous les couples de
fonctions \emph{entières} avec $f^2 + g^2 = 1$.
(a) Montrer que $h = f + \iu g$ est entière et sans zéro, et
que toute fonction entière sans zéro est $\eu^{\varphi}$ pour
une $\varphi$ entière \emph{($h'/h$ est entière, donc a une
primitive sur le $\C$ étoilé~; ajuster la constante et montrer
que $h\eu^{-\varphi}$ est constante)}.
(b) Conclure $f = \cos\varphi$, $g = \sin\varphi$ avec
$\varphi$ entière, et vérifier la réciproque. Quelles sont les
solutions entières de $f^2 + g^2 = 0$~?
\end{exercise}

\section{Problème~: le théorème fondamental de l'algèbre, deux
fois}

\begin{problem}[{Problème de week-end --- $\C$ est
algébriquement clos~: preuve de d'Alembert, preuve de
Liouville, et la récolte}]\label{pb:b3:holomorphic:1}
Soit $P(z) = z^n + a_{n-1}z^{n-1} + \dots + a_0$, $n \geq 1$.
Nous prouvons deux fois que $P$ a une racine, puis recueillons
ce que l'algèbre attend depuis le~\cref{ch:b3:galois}.

\medskip
\noindent\textbf{Partie~I --- Coercivité et le minimum.}
\begin{enumerate}
  \item Montrer que $\abs{P(z)} \to +\infty$ quand $\abs z \to
        \infty$~: précisément, $\abs{P(z)} \geq \frac12\abs
        z^n$ pour $\abs z \geq R_0$ convenable.
  \item En déduire que $\abs P$ atteint un minimum global sur
        $\C$~: il existe $z_0$ avec $\abs{P(z_0)} =
        \inf_\C\abs P$ \emph{(compacité d'un grand disque
        fermé, le~\cref{cor:b3:topology:heineborel})}.
\end{enumerate}

\medskip
\noindent\textbf{Partie~II --- La descente de d'Alembert.}
Supposons, par l'absurde, $P(z_0) \neq 0$.
\begin{enumerate}[resume]
  \item Développer $Q(h) = P(z_0 + h)/P(z_0)$ comme polynôme
        en $h$~: $Q(h) = 1 + c_kh^k + h^{k+1}S(h)$ avec $c_k
        \neq 0$, $k \geq 1$, $S$ un polynôme.
  \item Choisir la direction de descente~: pour $t > 0$ petit,
        poser $h = t\,\omega$ où $\omega^k = -1/c_k$
        (pourquoi un tel $\omega$ existe-t-il~? ---
        \emph{prouver} l'existence de racines $k$-ièmes de
        tout nombre complexe via la forme polaire,
        indépendamment du théorème en cours de démonstration).
        Montrer
        \[
        \abs{Q(t\omega)} \leq 1 - t^k + C\,t^{k+1}
        \]
        pour $t$ petit, avec une constante $C$ explicite.
  \item Conclure $\abs{Q(t\omega)} < 1$ pour $t$ petit ---
        contredisant la minimalité de $\abs{P(z_0)}$. D'où
        $P(z_0) = 0$~: \emph{tout polynôme complexe non
        constant a une racine} (d'Alembert--Argand).
\end{enumerate}

\medskip
\noindent\textbf{Partie~III --- Le one-liner de Liouville, en
entier.}
\begin{enumerate}[resume]
  \item Rédiger soigneusement la preuve
        du~\cref{cor:b3:holomorphic:liouville}~: si $P$ n'a
        pas de racine, vérifier que $1/P$ est entière,
        bornée (quantifier, en utilisant la question~1),
        donc constante, et conclure. Comparer les deux
        preuves~: quels ingrédients chacune utilise-t-elle~?
        (La compacité apparaît dans les deux --- où~?)
\end{enumerate}

\medskip
\noindent\textbf{Partie~IV --- La récolte.}
\begin{enumerate}[resume]
  \item Montrer que tout $P \in \C[X]$ de \omterm{def:b3:galois:extension}{degré} $n$ se
        scinde~: $P = c\prod_{i}(X - \alpha_i)^{m_i}$ avec
        $\sum m_i = n$ \emph{(récurrence, division euclidienne
        par $(X - \alpha)$)}.
  \item Montrer que les polynômes irréductibles de $\R[X]$
        sont les linéaires et les quadratiques à discriminant
        négatif \emph{(apparier les racines conjuguées)}~; en
        déduire que tout polynôme réel de \omterm{def:b3:galois:extension}{degré} impair a une
        racine réelle, et donner une seconde preuve, d'ordre
        théorique, de ce dernier fait (théorème des valeurs
        intermédiaires) --- en vérifiant qu'elles
        s'accordent sur $X^3 - X - 1$.
  \item En déduire les dettes que ce livre peut maintenant
        rembourser~: (i) tout endomorphisme d'un espace
        vectoriel $\C$ non nul de dimension finie a une
        valeur propre, donc toute matrice complexe a une
        \omterm{thm:b3:modules:jordan}{forme de Jordan} (le~\cref{thm:b3:modules:jordan})~;
        (ii) le corps $\bar\Q$ des nombres \omterm{def:b3:galois:algebraic}{algébriques} utilisé
        dans la~\cref{rem:b3:galois:closureexamples} est bien
        une \omterm{def:b3:galois:closure}{clôture algébrique} de $\Q$.
  \item (Finale) Localiser où chaque preuve casserait sur un
        corps comme $\Q(\iu)$~: quelles étapes utilisent
        l'existence de racines $k$-ièmes (question~4), et
        lesquelles utilisent la compacité ou la complétude
        (questions~2 et~6)~? Conclure en cinq lignes~: le
        théorème est authentiquement \emph{analytique} ---
        toute preuve invoque quelque part la complétude ou la
        connexité de $\R$ --- même si son énoncé est purement
        \omterm{def:b3:galois:algebraic}{algébrique}.
\end{enumerate}

\medskip
\noindent\textbf{Partie~V --- L'échelle de rigidité des
fonctions entières.} Liouville est le premier barreau d'une
échelle~; nous grimpons.
\begin{enumerate}[resume]
  \item (\omterm{thm:b3:holomorphic:analytic}{Estimations de Cauchy}) À partir de la formule de
        Cauchy sur le cercle de rayon $r$ autour de $a$,
        prouver
        \[
        \bigl|f^{(n)}(a)\bigr| \leq
        \frac{n!\,\sup_{\abs{z-a}=r}\abs f}{r^n} ,
        \]
        et retrouver Liouville comme le cas $n = 1$, $r \to
        \infty$.
  \item (Croissance polynomiale) Montrer qu'une $f$ entière
        avec $\abs{f(z)} \leq A + B\abs z^m$ pour tout $z$ est
        un polynôme de \omterm{def:b3:galois:extension}{degré} $\leq m$ \emph{(tuer les
        coefficients de Taylor au-delà de $m$ avec la
        question~11)}.
  \item (Partie réelle bornée) Montrer qu'une $f$ entière avec
        $\operatorname{Re}f$ bornée supérieurement est
        constante \emph{(appliquer Liouville à $\eu^{f}$)}.
  \item (Double périodicité) Soit $f$ entière avec $f(z + 1)
        = f(z)$ et $f(z + \iu) = f(z)$ pour tout $z$. Montrer
        que $f$ est constante. Conclure~: une fonction
        «~elliptique~» non constante doit avoir des
        singularités --- la raison historique pour laquelle
        les pôles entrent en analyse complexe.
  \item (Image \omterm{def:b3:topology:interior}{dense}) Montrer que l'image d'une fonction
        entière non constante est \omterm{def:b3:topology:interior}{dense} dans $\C$~: si
        $f(\C)$ manque un disque $D(a, r)$, alors $\frac1{f -
        a}$ est entière et bornée. (Picard a prouvé que
        l'image manque au plus \emph{un point}~; la \omterm{ex:b3:lebesgue:gamma}{densité}
        est le niveau que nos outils atteignent.)
  \item (Propre $\Rightarrow$ polynôme) Supposons $f$ entière
        et $\abs{f(z)} \to \infty$ quand $\abs z \to \infty$.
        Montrer~: les zéros de $f$ sont en nombre fini
        ($z_1, \dots, z_p$, avec multiplicités $m_i$)~; le
        quotient $g = f/\prod(z - z_i)^{m_i}$ est entier et
        sans zéro~; $1/g$ a une croissance polynomiale, donc
        (question~12) est un polynôme, nécessairement
        constant (sans zéro)~; conclure que $f$ est un
        polynôme. Ainsi parmi les fonctions entières, les
        polynômes sont exactement les propres --- $\eu^z$
        échoue à la propreté le long de $\R_-$.
\end{enumerate}

\medskip
\noindent\textbf{Partie~VI --- Ombres harmoniques et une
moyenne de Gauss.}
\begin{enumerate}[resume]
  \item Soit $f = u + \iu v$ \omterm{def:b3:holomorphic:holo}{holomorphe} sur un \omterm{def:b3:topology:topology}{ouvert}.
        Vérifier que $u = \operatorname{Re}f$ satisfait la
        propriété de la moyenne
        \[
        u(a) = \frac1{2\pi}\int_0^{2\pi}
        u\bigl(a + r\eu^{\iu\theta}\bigr)\,\dd\theta
        \]
        (partie réelle de la formule de Cauchy), et en
        déduire le principe du maximum pour $u$ sur un
        domaine borné, avec la même preuve de connexité que
        pour $\abs f$.
  \item (Moyenne de Gauss) Pour $a \in \C$ et $r > 0$ avec
        $\abs a \neq r$, prouver
        \[
        \frac1{2\pi}\int_0^{2\pi}
        \log\bigl|a - r\eu^{\iu\theta}\bigr|\,\dd\theta
        = \log\max\bigl(\abs a, r\bigr)
        \]
        \emph{(si $\abs a > r$~: $z \mapsto \log\abs{a - z}$
        est la partie réelle d'un logarithme \omterm{def:b3:holomorphic:holo}{holomorphe} sur
        un \omterm{def:b3:topology:topology}{voisinage} du disque fermé --- pourquoi en
        existe-t-il un~? --- donc la question~17
        s'applique~; si $\abs a < r$~: factoriser $\abs{a -
        r\eu^{\iu\theta}} = r\,\abs{1 - \frac ar\eu^{-\iu
        \theta}}$ et réutiliser le premier cas)}.
  \item (\omterm{def:b3:measure:measure}{Mesure} de Mahler) Pour $P = c\prod_{i=1}^n(X -
        \alpha_i) \in \C[X]$, en déduire la \emph{formule de
        Jensen pour les polynômes}~:
        \[
        \frac1{2\pi}\int_0^{2\pi}\log\bigl|P(\eu^{\iu\theta})
        \bigr|\,\dd\theta
        = \log\Bigl(\abs c\prod_{i}\max(1,
        \abs{\alpha_i})\Bigr) :
        \]
        la moyenne géométrique de $\abs P$ sur le cercle
        unité lit les racines hors du disque. Vérifier sur
        $P = X^2 - X$ et sur $P = 2X - 1$.
  \item (Nombres de Bernoulli) Définir les coefficients $B_n$
        par $\frac{z}{\eu^z - 1} = \sum_{n\geq0}\frac{B_n}{n!}
        z^n$ près de $0$ (pourquoi le membre de gauche est-il
        analytique en $0$~?). Dériver la récurrence
        $\sum_{k=0}^{n}\binom{n+1}kB_k = 0$ ($n \geq 1$) de
        $(\eu^z - 1)\cdot\frac z{\eu^z-1} = z$, calculer
        $B_0, \dots, B_6$, et montrer $B_{2k+1} = 0$ pour $k
        \geq 1$ \emph{(la fonction $\frac z{\eu^z-1} +
        \frac z2$ est paire)}. Ces nombres tariferont chaque
        $\zeta(2k)$ dans le~\cref{ch:b3:residues}.
  \item (Réalité) Montrer qu'une fonction entière prenant des
        valeurs réelles sur $\R$ satisfait $f(\bar z) =
        \overline{f(z)}$ partout \emph{(comparer les
        coefficients de Taylor en $0$, ou appliquer le
        théorème d'identité à $z \mapsto
        \overline{f(\bar z)}$)}~; en déduire à nouveau que
        les racines non réelles des polynômes réels viennent
        par paires conjuguées (appariement de la question~8,
        reprouvé analytiquement).
  \item (La morale) Assembler l'échelle de rigidité~: bornée
        $\Rightarrow$ constante~; polynomialement bornée
        $\Rightarrow$ polynôme~; propre $\Rightarrow$
        polynôme~; manquant un disque $\Rightarrow$
        constante~; doublement périodique $\Rightarrow$
        constante. Contraster en un court paragraphe avec les
        fonctions réelles $\mathcal C^\infty$ (fonctions
        bosselées, le~\cref{thm:b3:lp:regularization})~:
        pourquoi l'holomorphie, condition purement locale,
        impose-t-elle une loi et un ordre globaux~?
\end{enumerate}

\medskip
\noindent\textbf{Partie~VII --- Dernière récolte.}
\begin{enumerate}[resume]
  \item (Inégalité de Landau) Pour $P = \sum_{k=0}^na_kX^k$,
        prouver la moyenne $\frac1{2\pi}\int_0^{2\pi}
        \abs{P(\eu^{\iu\theta})}^2\dd\theta =
        \sum_k\abs{a_k}^2$ (orthogonalité des
        $\eu^{\iu k\theta}$), puis, en utilisant la borne
        ponctuelle $\log t \leq t - 1$ pour comparer les
        moyennes de $\log\abs P^2$ et $\abs P^2$, déduire de
        la question~19 que
        \[
        \abs c\prod_{i}\max\bigl(1, \abs{\alpha_i}\bigr)
        \;\leq\;
        \Bigl(\sum_{k=0}^{n}\abs{a_k}^2\Bigr)^{1/2} :
        \]
        le produit des racines hors du disque unité est
        contrôlé par la taille $\ell^2$ des coefficients
        (traiter les racines \emph{sur} le cercle en
        appliquant l'inégalité à $P(rX)$ et en faisant $r \to
        1$). Vérifier sur $X^2 - X$.
  \item (Les nombres de Bernoulli croissent
        factoriellement) Montrer que le rayon de convergence
        de $\sum\frac{B_n}{n!}z^n$ est \emph{exactement}
        $2\pi$~: au moins $2\pi$ car $z/(\eu^z - 1)$ s'étend
        holomorphiquement à $D(0, 2\pi)$, au plus $2\pi$ car
        la somme resterait sinon bornée près de $2\pi\iu$, où
        $\abs{z/(\eu^z-1)} \to \infty$. En déduire
        \[
        \limsup_{k\to\infty}
        \Bigl(\frac{\abs{B_{2k}}}{(2k)!}\Bigr)^{1/2k}
        = \frac1{2\pi} :
        \]
        les nombres de Bernoulli croissent factoriellement.
        En admettant $B_{12} = -\frac{691}{2730}$ (la
        récurrence de la question~20, poussée plus loin),
        comparer $\abs{B_{12}} \approx 0{,}25311$ avec la
        prédiction plus fine $2\,(2k)!/(2\pi)^{2k} \approx
        0{,}25305$ en $k = 6$ --- quatre chiffres
        concordants d'une loi asymptotique
        que le~\cref{ch:b3:residues} prouvera exactement, via
        $\zeta(2k)$.
  \item (Les racines bougent continûment) Soit $(P_j)$ monique
        de \omterm{def:b3:galois:extension}{degré} $n$ avec coefficients convergeant vers ceux
        de (monique) $P$. Prouver la \emph{borne de Cauchy}~:
        toute racine d'un monique $Q = X^n +
        \sum_{k<n}q_kX^k$ satisfait $\abs\alpha \leq 1 +
        \max_k\abs{q_k}$~; en déduire que les racines des
        $P_j$ restent dans un compact fixe, et, en extrayant
        des sous-suites convergentes de vecteurs de racines
        et en passant à la limite dans la factorisation de
        la question~7, que les multi-ensembles de racines
        des $P_j$ convergent vers celui de $P$. Montrer enfin
        que la continuité est le mieux qu'on puisse dire~:
        pour $P_\varepsilon = X^2 - 2X + 1 + \varepsilon$,
        une perturbation de taille $\varepsilon = 10^{-4}$
        déplace la racine double $1$ de $10^{-2}$ ---
        exposant de Hölder $\frac1m$ en une racine $m$-uple,
        jamais lipschitzien~: numériquement, les racines
        multiples coûtent la moitié des chiffres.
\end{enumerate}
\end{problem}
