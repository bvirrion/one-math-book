\chapter{Anneaux et arithmétique}\label{ch:b3:rings}

Les entiers ordinaires se factorisent de façon unique en
premiers~; de même les polynômes sur un corps. Ces deux faits
n'en font-ils qu'un~? Ce chapitre répond oui, et trouve les
hypothèses exactes qui rendent une «~arithmétique~» possible dans
un anneau commutatif~: la chaîne
\[
\text{euclidien} \;\Longrightarrow\; \text{principal}
\;\Longrightarrow\; \text{factoriel (UFD)},
\]
avec toutes les implications démontrées et toutes les
réciproques réfutées. La théorie est ensuite mise à l'épreuve
là où elle gagne sa vie~: les entiers de Gauss $\Z[\iu]$ (qui
creveront le théorème des deux carrés de Fermat dans le problème
du week-end), les anneaux de polynômes en plusieurs variables
(lemme de Gauss, critère d'\omterm{thm:b3:rings:criteria}{Eisenstein}), et les anneaux
\omterm{def:b3:rings:noetherian}{noethériens}, culminant dans le théorème de la base de Hilbert.
Tout au long, \emph{anneau} signifie anneau commutatif unitaire
avec $1 \neq 0$~; les idéaux de $\Z$ et $K[X]$ du volume de L2
sont nos deux exemples guides.

\section{Idéaux, quotients et théorème d'isomorphisme}

\begin{definition}\label{def:b3:rings:ideal}
Un \emph{idéal}\index{idéal} $I$ d'un anneau $A$ est un
sous-groupe additif tel que $AI \subseteq I$. L'\emph{anneau
quotient}\index{anneau quotient} $A/I$ est le \omterm{thm:b3:groups:quotient}{groupe quotient}
$(A, +)/I$ muni de la multiplication $(a + I)(b + I) = ab + I$~:
bien définie, car changer $a$ en $a + x$, $b$ en $b + y$ ($x, y
\in I$) change $ab$ de $ay + xb + xy \in I$. La projection $\pi
\colon A \to A/I$ est un morphisme d'anneaux surjectif de noyau
$I$, et les noyaux des morphismes d'anneaux sont exactement les
idéaux.
\end{definition}

\begin{theorem}[Premier théorème d'isomorphisme]\label{thm:b3:rings:firstiso}
Si $f \colon A \to B$ est un morphisme d'anneaux, alors $\bar
f\colon A/\ker f \to \operatorname{im} f$, $a + \ker f \mapsto
f(a)$, est un isomorphisme d'anneaux. Plus généralement $f$ se
factorise à travers $A/I$ pour tout \omterm{def:b3:rings:ideal}{idéal} $I \subseteq \ker f$.
Les idéaux de $A/I$ sont les $J/I$ pour $J \supseteq I$ un \omterm{def:b3:rings:ideal}{idéal}
de $A$ (théorème de correspondance).
\end{theorem}

\begin{proof}
Comme pour les groupes (\cref{thm:b3:groups:firstiso,%
thm:b3:groups:correspondence}), en notant que toutes les
applications en vue respectent aussi les produits~: $\bar f$ est
bien définie, bijective sur l'image, et multiplicative~; la
correspondance $J \mapsto J/I$, $\bar J \mapsto \pi^{-1}(\bar
J)$ préserve les idéaux dans les deux sens car $\pi$ est un
morphisme d'anneaux surjectif.
\end{proof}

\begin{definition}\label{def:b3:rings:primemaximal}
Soit $I \subsetneq A$ un \omterm{def:b3:rings:ideal}{idéal} propre. $I$ est
\emph{premier}\index{idéal premier} si $ab \in I \Rightarrow a
\in I$ ou $b \in I$~; $I$ est \emph{maximal}\index{idéal maximal}
si aucun \omterm{def:b3:rings:ideal}{idéal} ne se place strictement entre $I$ et $A$.
\end{definition}

\begin{proposition}\label{prop:b3:rings:primemaximal}
$I$ est premier $\iff$ $A/I$ est un anneau intègre~; $I$ est
\omterm{def:b3:rings:primemaximal}{maximal} $\iff$ $A/I$ est un corps. En particulier les idéaux
maximaux sont premiers.
\end{proposition}

\begin{proof}
Noter $\bar a$ les classes dans $A/I$. «~$I$ premier~» se
traduit verbatim en «~$\bar a\bar b = 0 \Rightarrow \bar a = 0$
ou $\bar b = 0$~», et $A/I \neq 0$ en $I \neq A$~: c'est la
définition d'un anneau intègre. Pour la maximalité, utiliser le
théorème de correspondance~: aucun \omterm{def:b3:rings:ideal}{idéal} strictement entre $I$
et $A$ $\iff$ $A/I$ n'a d'\omterm{def:b3:rings:ideal}{idéal} autre que $0$ et lui-même $\iff$
$A/I$ est un corps --- pour la dernière étape~: dans un corps
les seuls idéaux sont $0$ et le tout (un \omterm{def:b3:rings:ideal}{idéal} contenant $x \ne
0$ contient $x^{-1}x = 1$)~; réciproquement si tout $x$ non nul
engendre l'\omterm{def:b3:rings:ideal}{idéal} unité, alors $xy = 1$ pour un certain $y$. Les
corps sont intègres, donc les idéaux maximaux sont premiers.
\end{proof}

\begin{example}\label{ex:b3:rings:primemaximal}
Dans $\Z$~: les idéaux premiers sont $(0)$ et les $(p)$, $p$
premier~; les maximaux sont les $(p)$ ($\Z/p\Z = \mathbb F_p$
est un corps, $\Z/(0) = \Z$ ne l'est pas). Dans $K[X, Y]$~: $(X)
\subsetneq (X, Y)$ sont tous deux premiers ($K[X,Y]/(X) \cong
K[Y]$, intègre~; $K[X,Y]/(X,Y) \cong K$, un corps), donc $(X)$
est premier mais non \omterm{def:b3:rings:primemaximal}{maximal}.
\end{example}

Pour garantir que des idéaux maximaux \emph{existent} en toute
généralité, on a besoin d'un principe ensembliste. Un ensemble
partiellement ordonné est \emph{inductif} si toute partie
totalement ordonnée (\emph{chaîne}) a un majorant.

\begin{theorem}[Lemme de Zorn]\label{thm:b3:rings:zorn}
Tout ensemble partiellement ordonné inductif non vide a un
élément \omterm{def:b3:rings:primemaximal}{maximal}.\index{lemme de Zorn}
\admitted
\end{theorem}

\begin{remark}\label{rem:b3:rings:zorn}
Ce n'est pas un théorème des mathématiques ordinaires mais un
\emph{axiome}~: il est équivalent, sur les axiomes de base de
Zermelo--Fraenkel de la théorie des ensembles, à l'axiome du
choix («~tout produit d'ensembles non vides est non vide~»), que
nous acceptons tout au long de ce livre. Nous signalons chaque
usage. L'analyse l'invoquera de nouveau (Hahn--Banach,
\cref{ch:b3:banach}).
\end{remark}

\begin{theorem}[Krull]\label{thm:b3:rings:krull}
Tout \omterm{def:b3:rings:ideal}{idéal} propre $I \subsetneq A$ est \omterm{def:b3:rings:content}{contenu} dans un \omterm{def:b3:rings:primemaximal}{idéal
maximal}.\index{théorème de Krull}
\end{theorem}

\begin{proof}
Ordonner par inclusion l'ensemble $\mathcal E$ des idéaux
propres contenant $I$~; il est non vide ($I \in \mathcal E$).
Une chaîne $(J_\lambda)$ dans $\mathcal E$ a pour majorant $J =
\bigcup J_\lambda$~: un \omterm{def:b3:rings:ideal}{idéal} (tout $a, b \in J$ vivent dans un
même $J_\lambda$ par totalité), propre ($1 \notin J_\lambda$
pour tout $\lambda$), contenant $I$. Le lemme de Zorn fournit un
élément \omterm{def:b3:rings:primemaximal}{maximal} de $\mathcal E$, qui est un \omterm{def:b3:rings:primemaximal}{idéal maximal}
contenant $I$ (un \omterm{def:b3:rings:ideal}{idéal} strictement au-dessus et propre
appartiendrait à $\mathcal E$).
\end{proof}

\begin{theorem}[Théorème des restes chinois]\label{thm:b3:rings:crt}
Soient $I_1, \dots, I_n$ des idéaux deux à deux
\emph{comaximaux}\index{idéaux comaximaux} de $A$ ($I_k + I_l =
A$ pour $k \neq l$). Alors
\[
A\Big/\bigcap_{k=1}^n I_k \;\xrightarrow{\;\sim\;}\;
\prod_{k=1}^n A/I_k,
\qquad
a \longmapsto (a + I_1, \dots, a + I_n),
\]
et de plus $\bigcap_k I_k = I_1 I_2 \cdots I_n$ (l'\omterm{def:b3:rings:ideal}{idéal}
engendré par les produits).\index{théorème des restes chinois}
\end{theorem}

\begin{proof}
L'application $f(a) = (a + I_k)_k$ est un morphisme d'anneaux de
noyau $\bigcap I_k$~; par le~\cref{thm:b3:rings:firstiso} il
suffit de prouver la surjectivité. Fixer $k$~; pour chaque $l
\neq k$ écrire $1 = u_l + v_l$ avec $u_l \in I_k$, $v_l \in I_l$
(comaximalité). Alors
\[
e_k = \prod_{l \neq k} v_l = \prod_{l\neq k}(1 - u_l)
\equiv 1 \pmod{I_k},
\qquad e_k \in I_l \ (l \neq k),
\]
donc $f(e_k) = (0, \dots, 1, \dots, 0)$~; étant donné une cible
$(a_k + I_k)_k$, l'élément $\sum_k a_k e_k$ s'y envoie.

Produits contre intersection~: $I_1\cdots I_n \subseteq \bigcap
I_k$ toujours. Réciproquement, par récurrence il suffit de
traiter $n = 2$ (on vérifie que $I_1$ et $I_2\cdots I_n$ sont
\omterm{thm:b3:rings:crt}{comaximaux}~: en multipliant $1 = u_l + v_l$ sur $l \geq 2$ on
obtient $1 \in I_1 + I_2\cdots I_n$). Pour $n = 2$~: écrire $1 =
u + v$, $u \in I_1$, $v \in I_2$~; pour $x \in I_1 \cap I_2$, $x
= xu + xv \in I_2I_1 + I_1I_2 = I_1I_2$.
\end{proof}

\begin{example}\label{ex:b3:rings:crtz}
Dans $\Z$ avec $I_k = (m_k)$, $m_k$ deux à deux premiers entre
eux~: $\Z/(m_1\cdots m_n)\Z \cong \prod \Z/m_k\Z$ --- le
théorème des restes chinois du volume de L2. En restreignant
aux unités~: $(\Z/mn\Z)^\times \cong (\Z/m\Z)^\times \times
(\Z/n\Z)^\times$ pour $\gcd(m,n)=1$, d'où la multiplicitivité de
l'indicatrice d'Euler $\varphi$ (\cref{exo:b3:rings:8}).
\end{example}

\section{Divisibilité~: euclidien, principal, factoriel}

\begin{definition}\label{def:b3:rings:divisibility}
Soit $A$ un anneau intègre, $a, b \in A$. On dit que $a$
\emph{divise} $b$ ($a \mid b$) si $b \in (a) = aA$. Des éléments
$a, b$ sont \emph{associés}\index{éléments associés} si $a = ub$
avec $u \in A^\times$ (de façon équivalente $(a) = (b)$). Un
élément non nul non inversible $p$ est~:
\begin{itemize}
  \item \emph{irréductible}\index{élément irréductible} si $p =
        ab$ force $a \in A^\times$ ou $b \in A^\times$~;
  \item \emph{premier}\index{élément premier} si $p \mid ab$
        force $p \mid a$ ou $p \mid b$ (i.e.\ l'\omterm{def:b3:rings:ideal}{idéal} $(p)$ est
        premier).
\end{itemize}
\end{definition}

\begin{proposition}\label{prop:b3:rings:primeirred}
Dans tout anneau intègre, premier $\Rightarrow$ \omterm{def:b3:rings:divisibility}{irréductible}. La
réciproque est fausse en général~: dans $\Z[\iu\sqrt 5] = \{a +
\iu b\sqrt5 : a, b \in \Z\}$, l'élément $2$ est \omterm{def:b3:rings:divisibility}{irréductible}
mais non premier.
\end{proposition}

\begin{proof}
Soit $p$ premier et $p = ab$. Alors $p \mid ab$, disons $p \mid
a$~: $a = pc$, donc $p = pcb$, et en simplifiant par $p$ (anneau
intègre~!) on obtient $cb = 1$~: $b \in A^\times$.

Dans $\Z[\iu\sqrt5]$, utiliser la norme $N(x + \iu y\sqrt 5) =
x^2 + 5y^2$, multiplicative (c'est $\abs z^2$). Si $2 = ab$ avec
$a, b$ non inversibles, alors $4 = N(a)N(b)$ avec $N(a), N(b)
\neq 1$ (les éléments de norme $1$ sont $\pm1$, les unités),
donc $N(a) = 2$~: impossible, $x^2 + 5y^2 = 2$ n'a pas de
solution entière. Donc $2$ est \omterm{def:b3:rings:divisibility}{irréductible}. Mais $2 \mid 6 = (1
+ \iu\sqrt5)(1 - \iu\sqrt5)$ tandis que $2$ ne divise aucun des
facteurs ($\frac12 \pm \frac{\iu\sqrt5}2 \notin
\Z[\iu\sqrt5]$)~: non premier.
\end{proof}

\begin{definition}\label{def:b3:rings:pidufd}
Un anneau intègre $A$ est~:
\begin{itemize}
  \item \emph{euclidien}\index{anneau euclidien} s'il existe une
        application $\nu \colon A \setminus \{0\} \to \N$ (une
        \emph{fonction euclidienne}) telle que pour tous $a, b$
        avec $b \ne 0$ il existe $q, r$ avec $a = bq + r$ et ($r
        = 0$ ou $\nu(r) < \nu(b)$)~;
  \item \emph{principal} (un \emph{PID}\index{anneau principal})
        si tout \omterm{def:b3:rings:ideal}{idéal} est de la forme $(a)$~;
  \item \emph{factoriel} (un \emph{UFD}\index{anneau factoriel})
        si tout non nul non inversible est un produit
        d'\omterm{def:b3:rings:divisibility}{irréductibles}, unique à l'ordre et aux \omterm{def:b3:rings:divisibility}{associés} près.
\end{itemize}
\end{definition}

\begin{theorem}\label{thm:b3:rings:euclideanpid}
\omterm{def:b3:rings:pidufd}{Euclidien} $\Rightarrow$ principal.
\end{theorem}

\begin{proof}
Soit $I \neq (0)$ un \omterm{def:b3:rings:ideal}{idéal} et $b \in I \setminus\{0\}$ avec
$\nu(b)$ minimal. Pour $a \in I$, diviser~: $a = bq + r$~; alors
$r = a - bq \in I$, et $\nu(r) < \nu(b)$ contredirait la
minimalité, donc $r = 0$ et $a \in (b)$~: $I = (b)$.
\end{proof}

\begin{example}\label{ex:b3:rings:euclidean}
$\Z$ (avec $\nu = \abs\cdot$) et $K[X]$ (avec $\nu = \deg$) sont
\omterm{def:b3:rings:pidufd}{euclidiens} --- le volume de L2 a démontré les deux divisions. De
même $\Z[\iu]$, avec $\nu = N$ la norme carrée
(\cref{exo:b3:rings:4})~; la géométrie de la démonstration est
dans la figure ci-dessous. Un \omterm{def:b3:rings:pidufd}{PID} non \omterm{def:b3:rings:pidufd}{euclidien} existe mais est
délicat à certifier (l'exemple standard est
$\Z\bigl[\frac{1+\iu\sqrt{19}}2\bigr]$)~; un \omterm{def:b3:rings:pidufd}{UFD} qui n'est pas
un \omterm{def:b3:rings:pidufd}{PID} est facile~: $K[X, Y]$ (\cref{exo:b3:rings:6}), ou
$\Z[X]$.
\end{example}

\begin{lemma}[Chaînes croissantes d'idéaux principaux]
\label{lem:b3:rings:acc}
Dans un \omterm{def:b3:rings:pidufd}{PID}, toute suite croissante d'idéaux $I_1 \subseteq I_2
\subseteq \cdots$ est éventuellement constante.
\end{lemma}

\begin{proof}
$I = \bigcup_n I_n$ est un \omterm{def:b3:rings:ideal}{idéal} (l'union est croissante), donc
$I = (a)$~; l'élément $a$ vit dans un certain $I_N$, et alors $I
= (a) \subseteq I_N \subseteq I_n \subseteq I$ pour $n \geq N$.
\end{proof}

\begin{lemma}[Bézout~; lemme d'Euclide]\label{lem:b3:rings:bezout}
Soit $A$ un \omterm{def:b3:rings:pidufd}{PID} et $a, b \in A$. Alors $(a) + (b) = (d)$ pour un
certain $d$, un \emph{plus grand commun
diviseur}\index{plus grand commun diviseur}~: $d \mid a$, $d
\mid b$, et tout diviseur commun de $a, b$ divise $d$~; de plus
$d = au + bv$ pour certains $u, v$ (Bézout). Par conséquent tout
\omterm{def:b3:rings:divisibility}{élément irréductible} d'un \omterm{def:b3:rings:pidufd}{PID} est premier.
\end{lemma}

\begin{proof}
$(a) + (b)$ est un \omterm{def:b3:rings:ideal}{idéal}, donc $(d)$~; $a, b \in (d)$ donnent $d
\mid a, b$~; et $d = au + bv \in (a) + (b)$. Un diviseur commun
$c$ de $a, b$ divise $au + bv = d$.

Euclide~: soit $p$ \omterm{def:b3:rings:divisibility}{irréductible}, $p \mid ab$, $p \nmid a$. Un
pgcd $d$ de $p$ et $a$ divise $p$, donc $d$ est une unité ou un
associé de $p$ (irréductibilité)~; associé est exclu par $p
\nmid a$. Donc $1 = pu + av$, d'où $b = pub + abv$, et $p$
divise les deux termes~: $p \mid b$.
\end{proof}

\begin{theorem}\label{thm:b3:rings:pidufd}
Principal $\Rightarrow$ factoriel.
\end{theorem}

\begin{proof}
\emph{Existence.} Supposons qu'un certain non nul non inversible
$a$ n'a pas de factorisation en \omterm{def:b3:rings:divisibility}{irréductibles}. Alors $a$ n'est
pas \omterm{def:b3:rings:divisibility}{irréductible}~: $a = a_1b_1$ avec les deux facteurs non
inversibles~; l'un d'eux au moins, disons $a_1$, n'a encore pas
de factorisation (un produit de deux éléments factorisables est
factorisable). En itérant, on obtient $a = a_0, a_1, a_2,
\dots$, chacun diviseur propre du précédent sans factorisation,
donc $(a_0) \subsetneq (a_1) \subsetneq (a_2) \subsetneq
\cdots$ --- les inclusions sont strictes car $a_n = a_{n+1}c$
avec $c$ non inversible signifie que $(a_n) = (a_{n+1})$
forcerait $c \in A^\times$ (simplifier dans un anneau intègre).
Cela contredit le~\cref{lem:b3:rings:acc}.

\emph{Unicité.} Soit $p_1 \cdots p_r = q_1 \cdots q_s$ avec tous
les facteurs \omterm{def:b3:rings:divisibility}{irréductibles}, $r \leq s$, par récurrence sur $r$.
Le premier (\cref{lem:b3:rings:bezout}) $p_1$ divise le membre
de droite, donc divise un certain $q_j$~; renuméroter $j = 1$.
Comme $q_1$ est \omterm{def:b3:rings:divisibility}{irréductible} et $p_1$ n'est pas une unité, $q_1
= u p_1$ avec $u \in A^\times$~: $p_1, q_1$ sont \omterm{def:b3:rings:divisibility}{associés}.
Simplifier par $p_1$~: $p_2 \cdots p_r = (u q_2) q_3\cdots q_s$
et conclure par récurrence ($r = 1$ force $s = 1$~: une unité
fois des \omterm{def:b3:rings:divisibility}{irréductibles} ne peut valoir $1$).
\end{proof}

\begin{remark}\label{rem:b3:rings:ufdgcd}
Dans un \omterm{def:b3:rings:pidufd}{UFD}, les pgcd existent (prendre les exposants minimaux
dans les factorisations) et le lemme d'Euclide vaut ---
\omterm{def:b3:rings:divisibility}{irréductible} $=$ premier (\cref{exo:b3:rings:2}) --- mais
Bézout peut échouer~: dans $\Z[X]$, $\gcd(2, X) = 1$ pourtant $1
\neq 2U + XV$ (évaluer en $X = 0$~: $1 = 2U(0)$, impossible).
Les identités de Bézout sont la propriété exclusive des \omterm{def:b3:rings:pidufd}{PID}.
\end{remark}

\begin{example}[Un anneau sans factorisation unique]
\label{ex:b3:rings:nonufd}
Aucune des implications \omterm{def:b3:rings:pidufd}{euclidien} $\Rightarrow$ \omterm{def:b3:rings:pidufd}{PID} $\Rightarrow$
\omterm{def:b3:rings:pidufd}{UFD} n'est une équivalence, et l'échec de la dernière mérite d'être
vu une fois en détail complet. Dans
\[
A = \Z[\iu\sqrt5] = \{a + \iu b\sqrt5 : a, b \in \Z\},
\qquad N(a + \iu b\sqrt5) = a^2 + 5b^2,
\]
la norme est multiplicative et $N(z) = 1$ ssi $z \in A^\times =
\{\pm1\}$. Considérer
\[
6 = 2 \cdot 3 = (1 + \iu\sqrt5)(1 - \iu\sqrt5).
\]
Les quatre facteurs sont \omterm{def:b3:rings:divisibility}{irréductibles}~: leurs normes sont $4,
9, 6, 6$, et une factorisation propre $z = z_1z_2$ forcerait
$N(z_1) \in \{2, 3\}$ --- or $a^2 + 5b^2$ ne vaut jamais $2$ ni
$3$ ($b = 0$ laisse les non-carrés $2, 3$~; $\abs b \geq 1$
donne $\geq 5$). Pourtant $2$ n'est associé ni à $1 +
\iu\sqrt5$ ni à $1 - \iu\sqrt5$ (normes $4 \neq 6$)~: deux
factorisations véritablement différentes de $6$ en
\omterm{def:b3:rings:divisibility}{irréductibles}. De façon équivalente, \omterm{def:b3:rings:divisibility}{irréductible} $\neq$
premier ici~: $2$ divise le produit $(1 + \iu\sqrt5)(1 -
\iu\sqrt5) = 6$ mais aucun des facteurs (normes encore). La
réparation idéelliste de cet échec --- factoriser les
\emph{idéaux} plutôt que les éléments --- est la naissance de
la théorie algébrique des nombres~; à notre niveau, l'exemple
calibre à quel point les anneaux \omterm{def:b3:rings:pidufd}{euclidiens} $\Z$, $K[X]$,
$\Z[\iu]$ de ce chapitre sont spéciaux.
\end{example}

\begin{method}\label{met:b3:rings:quotient}
Pour identifier un \omterm{def:b3:rings:ideal}{anneau quotient} $A/I$, chercher un morphisme
surjectif $f \colon A \to B$ de noyau $I$ et invoquer
le~\cref{thm:b3:rings:firstiso}~; quand $A = C[X]$ est un anneau
de polynômes, $f$ est en général une évaluation. Ainsi
$\Z[X]/(X^2+1) \cong \Z[\iu]$ (évaluer en $\iu$), $K[X,Y]/(Y -
X^2) \cong K[X]$ (évaluer $Y$ en $X^2$), $\R[X]/(X^2+1) \cong
\C$. Pour montrer que $I$ est premier ou \omterm{def:b3:rings:primemaximal}{maximal}, montrer que le
quotient est intègre ou un corps
(la~\cref{prop:b3:rings:primemaximal}).
\end{method}

\section{Polynômes sur un UFD~: Gauss et Eisenstein}

Tout au long de cette section $A$ est un \omterm{def:b3:rings:pidufd}{UFD} de corps des
fractions $K$ (construit comme le corps des quotients formels
$a/b$, $b \neq 0$, exactement comme $\Q$ à partir de $\Z$~; le
volume de L2 a fait cette construction pour $\Q$, et elle se
transfère verbatim). Notre but~: la factorialité passe de $A$ à
$A[X]$, et l'irréductibilité sur $A$ est essentiellement
l'irréductibilité sur le corps plus grand $K$.

\begin{definition}\label{def:b3:rings:content}
Le \emph{contenu}\index{contenu d'un polynôme} $c(P)$ d'un $P
\neq 0$ de $A[X]$ est un pgcd de ses coefficients (défini à une
unité près)~; $P$ est \emph{primitif} si $c(P) \in A^\times$.
Tout $P \in A[X]$ s'écrit $P = c(P)\,P_1$ avec $P_1$ primitif, et
tout $P \in K[X]\setminus\{0\}$ s'écrit $P = \lambda P_1$ avec
$\lambda \in K^\times$ et $P_1 \in A[X]$ primitif (chasser les
dénominateurs, puis extraire le contenu).
\end{definition}

\begin{lemma}[Gauss]\label{lem:b3:rings:gauss}
Le produit de deux polynômes primitifs de $A[X]$ est primitif~;
par conséquent $c(PQ) = c(P)c(Q)$ à unités près.\index{lemme de Gauss}
\end{lemma}

\begin{proof}
Soient $P, Q$ primitifs et supposons qu'un certain \omterm{def:b3:rings:divisibility}{irréductible}
($=$ premier, \omterm{def:b3:rings:pidufd}{UFD}) $p$ divise tous les coefficients de $PQ$.
Réduire modulo $p$~: dans $(A/(p))[X]$, $\bar P \bar Q = 0$.
Mais $A/(p)$ est intègre ($(p)$ premier), donc $(A/(p))[X]$ est
intègre (les coefficients dominants se multiplient), forçant
$\bar P = 0$ ou $\bar Q = 0$~: $p$ divise tous les coefficients
de $P$ ou tous ceux de $Q$, contredisant la primitivité. Pour la
conséquence, écrire $P = c(P)P_1$, $Q = c(Q)Q_1$~: $PQ =
c(P)c(Q) P_1Q_1$ avec $P_1Q_1$ primitif.
\end{proof}

\begin{theorem}\label{thm:b3:rings:gaussufd}
Soit $A$ un \omterm{def:b3:rings:pidufd}{UFD} de corps des fractions $K$.
\begin{enumerate}
  \item Un $P \in A[X]$ primitif de degré $\geq 1$ est
        \omterm{def:b3:rings:divisibility}{irréductible} dans $A[X]$ ssi il est \omterm{def:b3:rings:divisibility}{irréductible} dans
        $K[X]$.
  \item $A[X]$ est un \omterm{def:b3:rings:pidufd}{UFD}~; ses \omterm{def:b3:rings:divisibility}{irréductibles} sont les
        \omterm{def:b3:rings:divisibility}{irréductibles} de $A$ et les polynômes primitifs
        \omterm{def:b3:rings:divisibility}{irréductibles} sur $K$. En particulier $\Z[X]$, et par
        récurrence $K[X_1, \dots, X_n]$ et $\Z[X_1, \dots,
        X_n]$, sont des \omterm{def:b3:rings:pidufd}{UFD}.
\end{enumerate}
\end{theorem}

\begin{proof}
(1) ($\Leftarrow$) Si $P = QR$ dans $A[X]$ avec $Q, R$ non
inversibles, alors aucun des facteurs n'est constant (un facteur
constant d'un polynôme primitif est une unité), donc la
factorisation est propre dans $K[X]$. ($\Rightarrow$) Supposons
$P = QR$ avec $Q, R \in K[X]$ de degrés $\geq 1$. Écrire $Q =
\lambda Q_1$, $R = \mu R_1$ avec $Q_1, R_1 \in A[X]$
primitifs~: $P = \lambda\mu\, Q_1R_1$, et $Q_1R_1$ est primitif
par Gauss. En prenant les \omterm{def:b3:rings:content}{contenus}, $\lambda\mu \in A^\times$
(les deux côtés ont un \omterm{def:b3:rings:content}{contenu} unité~; formellement, $\lambda\mu
= c(P) \in A^\times$ à unités près, et en particulier $\lambda
\mu \in A$)~: $P = (\lambda\mu Q_1) R_1$ est une factorisation
propre dans $A[X]$.

(2) Existence~: étant donné $P \neq 0$ non inversible, factoriser
$P = c(P)P_1$, factoriser $c(P)$ en \omterm{def:b3:rings:divisibility}{irréductibles} de $A$, et
factoriser $P_1$ dans l'\omterm{def:b3:rings:pidufd}{UFD} $K[X]$ en $\prod Q_i$ avec $Q_i \in
K[X]$ \omterm{def:b3:rings:divisibility}{irréductibles}~; en écrivant $Q_i = \lambda_i R_i$ avec
$R_i \in A[X]$ primitifs (donc \omterm{def:b3:rings:divisibility}{irréductibles} sur $K$, donc dans
$A[X]$ par (1)), le produit $\prod \lambda_i$ est une unité de
$A$ comme ci-dessus, et $P_1 = u\prod R_i$. Unicité~: comparer
la partie constante et la partie polynomiale d'une
factorisation~; les constantes multiplient en $c(P)$ (Gauss),
unique par factorialité de $A$~; les parties polynomiales
donnent deux factorisations dans $K[X]$ du même polynôme, donc
elles s'apparient à des constantes de $K^\times$ près
(factorialité de $K[X]$, \cref{thm:b3:rings:pidufd}), et des
polynômes primitifs \omterm{def:b3:rings:divisibility}{associés} dans $K[X]$ sont \omterm{def:b3:rings:divisibility}{associés} dans
$A[X]$~: si $R = \lambda R'$ avec $R, R'$ primitifs et $\lambda
\in K^\times$, alors prendre les \omterm{def:b3:rings:content}{contenus} force $\lambda \in
A^\times$.
\end{proof}

\begin{theorem}[Critères d'irréductibilité]\label{thm:b3:rings:criteria}
Soit $A$ un \omterm{def:b3:rings:pidufd}{UFD}, $K$ son corps des fractions, et $P = a_nX^n +
\dots + a_0 \in A[X]$ primitif de degré $n \geq 1$.
\begin{enumerate}
  \item (\emph{Réduction}) Si $p \in A$ est premier, $p \nmid
        a_n$, et la réduction $\bar P$ est \omterm{def:b3:rings:divisibility}{irréductible} dans
        $(A/(p))[X]$, alors $P$ est \omterm{def:b3:rings:divisibility}{irréductible} dans $K[X]$
        (donc dans $A[X]$).
  \item (\emph{Eisenstein}\index{critère d'Eisenstein}) Si un
        certain premier $p$ vérifie $p \nmid a_n$, $p \mid a_i$
        pour $0 \leq i < n$, et $p^2 \nmid a_0$, alors $P$ est
        \omterm{def:b3:rings:divisibility}{irréductible} dans $K[X]$ (donc dans $A[X]$).
\end{enumerate}
\end{theorem}

\begin{proof}
Par le~\cref{thm:b3:rings:gaussufd}(1), une factorisation propre
sur $K$ donne $P = QR$ avec $Q, R \in A[X]$, $\deg Q, \deg R
\geq 1$ (les constantes sont exclues~: elles seraient des unités
ou gâcheraient la primitivité).

(1) Réduire mod $p$~: $\bar P = \bar Q\bar R$ dans
$(A/(p))[X]$. Comme $p \nmid a_n$ et $\deg$ ne peut que baisser
sous réduction, $\deg \bar Q = \deg Q \geq 1$ et $\deg\bar R =
\deg R \geq 1$ (leurs coefficients dominants multiplient en
$\bar a_n \neq 0$, donc aucun ne baisse)~: $\bar P$ se factorise
proprement --- contradiction.

(2) Réduire mod $p$~: $\bar Q \bar R = \bar P = \bar a_n X^n$
(tous les coefficients inférieurs meurent). Dans l'anneau
intègre $(A/(p))[X]$, les factorisations de $cX^n$ ($c \ne 0$)
sont en constantes et puissances pures $c'X^k$~: en effet si
$\bar Q\bar R = \bar a_nX^n$, et disons $\bar Q$ avait un
coefficient non nul en degré $< \deg\bar Q$, prendre les termes
non nuls de plus bas degré~: $\operatorname{val}(\bar Q\bar R) =
\operatorname{val}\bar Q + \operatorname{val}\bar R$ (anneau
intègre), qui doit valoir $n = \deg\bar Q + \deg\bar R$,
forçant $\operatorname{val} = \deg$ pour les deux~: tous deux
sont monômes. Comme ci-dessus, les degrés ne baissent pas, donc
$Q$ et $R$ ont leurs termes constants $Q(0), R(0)$ divisibles
par $p$ --- les deux, car les deux réductions sont des monômes
de degré $\geq 1$. Alors $p^2 \mid Q(0)R(0) = a_0$~:
contradiction.
\end{proof}

\begin{example}\label{ex:b3:rings:eisenstein}
$X^n - p$ est \omterm{def:b3:rings:divisibility}{irréductible} sur $\Q$ pour tout premier $p$ et $n
\geq 1$ (\omterm{thm:b3:rings:criteria}{Eisenstein} en $p$)~: il existe des polynômes
\omterm{def:b3:rings:divisibility}{irréductibles} de tout degré sur $\Q$ --- en contraste frappant
avec $\C$ (degré $1$, d'Alembert--Gauss, démontré dans
le~\cref{ch:b3:holomorphic}) et $\R$ (degrés $1, 2$). Le tour du
\emph{décalage} élargit la portée d'\omterm{thm:b3:rings:criteria}{Eisenstein}~: le $p$-ième
\emph{polynôme cyclotomique}\index{polynôme cyclotomique}
$\Phi_p = X^{p-1} + \dots + X + 1 = \frac{X^p - 1}{X - 1}$ a
\[
\Phi_p(X + 1) = \frac{(X+1)^p - 1}{X}
= X^{p-1} + \binom{p}{1}X^{p-2} + \dots + \binom{p}{p-1},
\]
\omterm{thm:b3:rings:criteria}{Eisenstein} en $p$ ($p \mid \binom pk$ pour $0 < k < p$, et
$\binom{p}{p-1} = p \not\equiv 0 \bmod p^2$)~: $\Phi_p(X+1)$,
donc $\Phi_p$, est \omterm{def:b3:rings:divisibility}{irréductible} sur $\Q$. C'est le cœur
algébrique de l'histoire du $17$-gone racontée dans
le~\cref{ch:b3:galois}.
\end{example}

\begin{method}\label{met:b3:rings:irreducibility}
Pour prouver qu'un $P \in \Z[X]$ est \omterm{def:b3:rings:divisibility}{irréductible} sur $\Q$~:
(i) rendre $P$ primitif~; (ii) essayer \omterm{thm:b3:rings:criteria}{Eisenstein}, sur $P(X)$ et
sur les décalages $P(X \pm 1)$~; (iii) essayer la réduction
modulo de petits premiers ne divisant pas le coefficient
dominant --- l'irréductibilité mod \emph{un} $p$ suffit, et sur
$\mathbb F_p$ l'irréductibilité est un contrôle fini (pas de
racines exclut les facteurs de degré $1$~; puis tester les
finiement nombreux facteurs de chaque degré $\leq \deg P/2$)~;
(iv) si tout le reste échoue, coefficients indéterminés.
Attention~: la réductibilité mod tout $p$ n'\emph{implique}
pas la réductibilité sur $\Q$ (\cref{exo:b3:rings:11}).
\end{method}

\section{Anneaux noethériens}

\begin{definition}\label{def:b3:rings:noetherian}
Un anneau $A$ est \emph{noethérien}\index{anneau noethérien} si
tout \omterm{def:b3:rings:ideal}{idéal} de $A$ est de type fini.
\end{definition}

\begin{proposition}\label{prop:b3:rings:noethacc}
$A$ est \omterm{def:b3:rings:noetherian}{noethérien} ssi toute suite croissante d'idéaux est
éventuellement constante (\emph{condition de chaîne
ascendante}), ssi toute famille non vide d'idéaux a un élément
\omterm{def:b3:rings:primemaximal}{maximal} (pour l'inclusion).
\end{proposition}

\begin{proof}
\emph{(TF $\Rightarrow$ CCA)}~: pour une chaîne $I_1 \subseteq
I_2 \subseteq \cdots$, l'union $I$ est un \omterm{def:b3:rings:ideal}{idéal}, engendré par
$x_1, \dots, x_r$~; tous les $x_i$ vivent dans un certain $I_N$,
donc $I = I_N = I_n$ pour $n \geq N$. \emph{(CCA $\Rightarrow$
éléments maximaux)}~: si une famille non vide $\mathcal F$
n'avait pas d'élément \omterm{def:b3:rings:primemaximal}{maximal}, choisir $I_1 \in \mathcal F$,
puis inductivement $I_{n+1} \supsetneq I_n$ dans $\mathcal F$
(possible car $I_n$ n'est pas \omterm{def:b3:rings:primemaximal}{maximal})~: une chaîne strictement
croissante infinie. (Cela utilise l'axiome des choix dépendants,
une forme faible du choix dont nous ne faisons pas d'histoire.)
\emph{(Éléments maximaux $\Rightarrow$ TF)}~: étant donné un
\omterm{def:b3:rings:ideal}{idéal} $I$, la famille des idéaux de type fini \omterm{def:b3:rings:content}{contenus} dans $I$
est non vide ($(0)$)~; un élément \omterm{def:b3:rings:primemaximal}{maximal} $J = (x_1, \dots,
x_r)$ doit valoir $I$~: sinon, ajouter $x \in I \setminus J$ aux
générateurs produit un membre strictement plus grand de la
famille.
\end{proof}

\begin{theorem}[Théorème de la base de Hilbert]\label{thm:b3:rings:hilbert}
Si $A$ est \omterm{def:b3:rings:noetherian}{noethérien}, alors $A[X]$ l'est aussi. D'où aussi
$A[X_1, \dots, X_n]$, et tout quotient de ceux-ci.\index{théorème de la base de Hilbert}
\end{theorem}

\begin{proof}
Soit $I$ un \omterm{def:b3:rings:ideal}{idéal} de $A[X]$, et supposons $I$ non de type fini.
Construire une suite~: $f_1 \in I \setminus \{0\}$ de degré
minimal, et inductivement $f_{k+1} \in I \setminus (f_1, \dots,
f_k)$ de degré minimal (l'ensemble est non vide par
hypothèse). Les degrés $d_k = \deg f_k$ sont non décroissants
(par minimalité de chaque choix~: $f_{k+1}$ était disponible à
l'étape $k+1$... précisément, $f_{k+1} \notin (f_1,\dots,f_k)
\supseteq (f_1, \dots, f_{k-1})$, donc $f_{k+1}$ a concouru à
l'étape $k$ et a perdu ou fait match nul~: $d_{k+1} \geq d_k$).
Soit $a_k \in A$ le coefficient dominant de $f_k$. La chaîne
d'idéaux $(a_1) \subseteq (a_1, a_2) \subseteq \cdots$ se
stabilise~: $a_{n+1} \in (a_1, \dots, a_n)$ pour un certain $n$,
disons $a_{n+1} = \sum_{k\leq n} u_k a_k$. Considérer
\[
g = f_{n+1} - \sum_{k=1}^{n} u_k X^{\,d_{n+1} - d_k} f_k .
\]
Alors $g \in I \setminus (f_1, \dots, f_n)$ (la somme vit dans
l'\omterm{def:b3:rings:ideal}{idéal}, $f_{n+1}$ non), pourtant le coefficient de degré
$d_{n+1}$ s'annule~: $\deg g < d_{n+1}$, contredisant la
minimalité de $d_{n+1} = \deg f_{n+1}$.

En itérant, $A[X_1, \dots, X_n] = (A[X_1, \dots, X_{n-1}])[X_n]$
est \omterm{def:b3:rings:noetherian}{noethérien}~; un quotient $A/I$ est \omterm{def:b3:rings:noetherian}{noethérien} car ses idéaux
$J/I$ se relèvent en idéaux de $A$ (correspondance), où un
nombre fini de générateurs se projettent sur des générateurs.
\end{proof}

\begin{remark}\label{rem:b3:rings:noethmeaning}
La noethérianité est l'axiome de finitude de la géométrie
algébrique~: tout système d'équations polynomiales en $n$
variables, si infini soit-il, est équivalent à un nombre fini
d'entre elles --- son ensemble de solutions est découpé par un
nombre fini de polynômes. Les \omterm{def:b3:rings:pidufd}{PID} sont \omterm{def:b3:rings:noetherian}{noethériens}
(trivialement)~; $\Z[X_1, X_2, \dots]$ en une infinité de
variables ne l'est pas ($(X_1) \subsetneq (X_1, X_2) \subsetneq
\cdots$). Les anneaux non \omterm{def:b3:rings:noetherian}{noethériens} apparaissent aussi
naturellement en analyse~: les fonctions continues sur
$\intcc01$ en forment un (\cref{exo:b3:rings:10}).
\end{remark}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:b3:rings:1}
Identifier les quotients~: (a) $\Z[X]/(X^2 + 1) \cong \Z[\iu]$~;
(b) $\R[X]/(X^2+1) \cong \C$~; (c) $\mathbb F_2[X]/(X^2 + X +
1)$ est un corps à $4$ éléments --- écrire sa table de
multiplication.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b3:rings:2}
(a) Montrer que dans un \omterm{def:b3:rings:pidufd}{UFD}, tout \omterm{def:b3:rings:divisibility}{élément irréductible} est
premier.
(b) Montrer qu'un anneau intègre fini est un corps.
(c) En déduire que dans un anneau fini, tout \omterm{def:b3:rings:primemaximal}{idéal premier} est
\omterm{def:b3:rings:primemaximal}{maximal}.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b3:rings:3}
Dans $\Z[\iu\sqrt5]$~: vérifier que $3$, $1 + \iu\sqrt5$ et $1
- \iu\sqrt5$ sont \omterm{def:b3:rings:divisibility}{irréductibles}, que $9 = 3\cdot 3 = (2 +
\iu\sqrt5)(2 - \iu\sqrt5)$, et conclure à nouveau (après
la~\cref{prop:b3:rings:primeirred}) que $\Z[\iu\sqrt5]$ n'est
pas un \omterm{def:b3:rings:pidufd}{UFD}. Où exactement l'unicité échoue-t-elle~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:rings:4}
(a) Montrer que $\Z[\iu]$ est \omterm{def:b3:rings:pidufd}{euclidien} pour la norme $N(x + \iu
y) = x^2 + y^2$~: étant donnés $a, b \neq 0$, choisir $q \in
\Z[\iu]$ le plus proche de $a/b \in \C$.
(b) Déterminer $\Z[\iu]^\times$.
(c) Mêmes questions pour $\Z[\iu\sqrt2]$ et $N(x + \iu y\sqrt2)
= x^2 + 2y^2$. Pourquoi le même argument échoue-t-il pour
$\Z[\iu\sqrt5]$~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:rings:5}
Soit $A$ un anneau. (a) Montrer que si $x$ est nilpotent ($x^n =
0$ pour un certain $n$) alors $1 + x \in A^\times$. (b) Montrer
que si $A$ est intègre, $A[X]^\times = A^\times$~; donner un
contre-exemple sur $\Z/4\Z$. (c) Montrer qu'un anneau intègre
n'a pas d'idempotents ($e^2 = e$) autres que $0, 1$, et aucun
nilpotent autre que $0$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:rings:6}
Dans $A = K[X, Y]$~: (a) montrer que l'\omterm{def:b3:rings:ideal}{idéal} $(X, Y)$ est
\omterm{def:b3:rings:primemaximal}{maximal} mais non principal --- donc $K[X,Y]$ est un \omterm{def:b3:rings:pidufd}{UFD}
(\cref{thm:b3:rings:gaussufd}) qui n'est pas un \omterm{def:b3:rings:pidufd}{PID}~; (b)
identifier $K[X, Y]/(Y - X^2)$ et $K[X,Y]/(XY - 1)$ comme
sous-anneaux de fonctions rationnelles~; (c) $(Y - X^2)$
est-il premier~? \omterm{def:b3:rings:primemaximal}{maximal}~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:rings:7}
\omterm{def:b3:rings:divisibility}{Irréductible} ou non sur $\Q$~: $X^5 - 12X^3 + 36X - 12$~;\quad
$X^4 + X + 1$ \emph{(réduire mod $2$)}~;\quad $X^4 + 4$~;\quad
$\Phi_8 = X^4 + 1$ \emph{(décaler de $1$)}~;\quad $X^3 - X -
1$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:rings:8}
(a) À partir du~\cref{thm:b3:rings:crt}, démontrer que la
fonction d'Euler est multiplicative sur les arguments premiers
entre eux et que $\varphi(p^k) = p^{k-1}(p-1)$~; retrouver
$\varphi(n) = n\prod_{p \mid n}(1 - \frac1p)$.
(b) Résoudre~: $x \equiv 2 \pmod 7$, $x \equiv 5 \pmod{11}$, $x
\equiv 1 \pmod{13}$, en exhibant les idempotents $e_k$ de la
démonstration du~\cref{thm:b3:rings:crt}.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b3:rings:9}
(Le nilradical) Soit $\operatorname{Nil}(A)$ l'ensemble des
éléments nilpotents. (a) Montrer que $\operatorname{Nil}(A)$ est
un \omterm{def:b3:rings:ideal}{idéal} \omterm{def:b3:rings:content}{contenu} dans tout \omterm{def:b3:rings:primemaximal}{idéal premier}. (b) Réciproquement,
soit $a$ non nilpotent~; en utilisant le lemme de Zorn sur les
idéaux évitant $S = \{a^n : n \in \N\}$, produire un \omterm{def:b3:rings:primemaximal}{idéal
premier} ne contenant pas $a$. Conclure~:
\[
\operatorname{Nil}(A) = \bigcap_{\mathfrak p \text{ premier}}
\mathfrak p .
\]
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b3:rings:10}
(a) Soit $A$ \omterm{def:b3:rings:noetherian}{noethérien} et $f \colon A \to A$ un morphisme
d'anneaux surjectif. Montrer que $f$ est injectif.
\emph{(Considérer $\ker f \subseteq \ker f^2 \subseteq
\cdots$.)}
(b) Montrer que l'anneau $\mathcal C(\intcc01, \R)$ des
fonctions continues n'est pas \omterm{def:b3:rings:noetherian}{noethérien}. \emph{(Considérer $I_n
= \{f : f = 0 \text{ sur } \intcc0{1/n}\}$.)}
(c) Montrer que dans un anneau intègre \emph{\omterm{def:b3:rings:noetherian}{noethérien}}, tout
non nul non inversible est un produit (fini) d'\omterm{def:b3:rings:divisibility}{irréductibles} ---
donc la non-factorialité de $\Z[\iu\sqrt 5]$ est un échec de
l'\emph{unicité} seulement.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b3:rings:11}
Soit $P = X^4 + 1$.
(a) Montrer que $P$ est \omterm{def:b3:rings:divisibility}{irréductible} sur $\Q$
(\cref{exo:b3:rings:7}).
(b) Montrer que $P$ est réductible modulo \emph{tout} premier
$p$~: traiter $p = 2$~; puis, pour $p$ impair, montrer que $8
\mid p^2 - 1$ et admettre pour l'instant (démontré dans
le~\cref{ch:b3:galois}) que le groupe multiplicatif du corps à
$p^2$ éléments est cyclique, pour conclure que $P$ se scinde en
deux facteurs quadratiques mod $p$~; les rendre explicites
quand l'un de $-1$, $2$, $-2$ est un carré mod $p$, et montrer
que l'un d'eux l'est toujours.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:rings:12}
(Les idempotents scindent les anneaux) Un élément $e$ d'un
anneau commutatif $A$ est \emph{idempotent} si $e^2 = e$.
(a) Montrer que si $e$ est idempotent, $1 - e$ l'est aussi, et
que l'application $x \mapsto (ex, (1-e)x)$ est un isomorphisme
d'anneaux $A \cong Ae \times A(1-e)$, où $Ae$ est un anneau
d'unité $e$.
(b) Trouver tous les idempotents d'un anneau intègre, et de
$\Z/12\Z$~; exhiber l'isomorphisme $\Z/12\Z \cong \Z/4\Z \times
\Z/3\Z$ en nommant ses deux idempotents non triviaux.
(c) Montrer que la décomposition CRT de $\Z/n\Z$
(\cref{ex:b3:rings:crtz}) correspond exactement aux idempotents
$e_i \equiv 1 \bmod p_i^{a_i}$, $e_i \equiv 0$ modulo les
autres puissances de premiers~: les anneaux se décomposent le
long de leurs idempotents comme les espaces le long des
projections.
\end{exercise}

\section{Problème~: le théorème des deux carrés de Fermat}

\begin{problem}[{Problème du week-end --- sommes de deux carrés,
via $\Z[\iu]$}]\label{pb:b3:rings:1}
Quels entiers sont sommes de deux carrés~? La réponse de Fermat
(1640) est l'un des joyaux de l'arithmétique~; les entiers de
Gauss transforment sa démonstration en théorie des anneaux. Tout
au long, $N(x + \iu y) = x^2 + y^2$ désigne la norme, $\Z[\iu]$
est \omterm{def:b3:rings:pidufd}{euclidien} (\cref{exo:b3:rings:4}), donc un \omterm{def:b3:rings:pidufd}{PID} et un \omterm{def:b3:rings:pidufd}{UFD}, et
\emph{premier de Gauss} signifie \omterm{def:b3:rings:divisibility}{élément premier} ($=$
\omterm{def:b3:rings:divisibility}{irréductible}) de $\Z[\iu]$.

\medskip
\noindent\textbf{Partie I --- Normes et premiers de Gauss.}
\begin{enumerate}
  \item Vérifier $N(zw) = N(z)N(w)$, en déduire à nouveau
        $\Z[\iu]^\times = \{\pm1, \pm\iu\}$, et démontrer
        l'\emph{identité de Brahmagupta}~: un produit de deux
        sommes de deux carrés est une somme de deux carrés.
  \item Montrer que si $N(z)$ est un nombre premier, alors $z$
        est un premier de Gauss.
  \item Montrer que tout premier de Gauss $\pi$ divise
        exactement un nombre premier $p$ \emph{(considérer
        $N(\pi) = \pi\bar\pi$)}, et qu'alors $N(\pi) \in \{p,
        p^2\}$.
  \item En déduire la dichotomie~: pour chaque premier $p$, soit
        $p$ reste premier dans $\Z[\iu]$ (et aucun premier de
        Gauss de norme $p$ n'existe), soit $p = \pi\bar\pi$ avec
        $\pi$ un premier de Gauss de norme $p$ --- et alors $p =
        a^2 + b^2$.
\end{enumerate}

\medskip
\noindent\textbf{Partie II --- Théorème de Wilson et $-1$
modulo $p$.}
\begin{enumerate}[resume]
  \item Démontrer le \emph{théorème de Wilson}~: pour $p$
        premier, $(p-1)! \equiv -1 \pmod p$. \emph{(Apparier
        chaque résidu avec son inverse~; lesquels sont
        auto-appariés~?)}
  \item Soit $p$ un premier impair et $m = \frac{p-1}2$. Montrer
        que $(m!)^2 \equiv (-1)^{m+1} \pmod p$ \emph{(dans
        $(p-1)!$, remplacer chaque facteur $k > m$ par $-(p -
        k)$)}.
  \item Conclure~: $-1$ est un carré modulo $p$ ssi $p = 2$ ou
        $p \equiv 1 \pmod 4$. \emph{(Pour le «~seulement si~»~:
        si $x^2 \equiv -1$, quel est l'ordre de $x$ dans
        $(\Z/p\Z)^\times$, et que dit Lagrange~?)}
\end{enumerate}

\medskip
\noindent\textbf{Partie III --- La loi de scindage.}
\begin{enumerate}[resume]
  \item Soit $p \equiv 1 \pmod 4$, et $x$ avec $p \mid x^2 + 1 =
        (x + \iu)(x - \iu)$. Montrer que $p$ n'est \emph{pas} un
        premier de Gauss, et conclure avec la partie I~: $p =
        a^2 + b^2$.
  \item Soit $p \equiv 3 \pmod 4$. Montrer directement que $p$
        n'est pas une somme de deux carrés \emph{(carrés mod
        $4$)}, et en déduire que $p$ reste un premier de Gauss.
  \item Régler $p = 2$~: exhiber la factorisation $2 =
        -\iu(1+\iu)^2$ et vérifier que $1 + \iu$ est un premier
        de Gauss. ($2$ est l'unique premier \emph{ramifié}~:
        divisible par le carré d'un premier de Gauss à une unité
        près.)
  \item Assembler la classification des premiers de Gauss, à
        unités près~: $1 + \iu$~; les entiers $p \equiv 3 \pmod
        4$~; les paires conjuguées $\pi, \bar\pi$ de norme $p
        \equiv 1 \pmod 4$. La vérifier sur $5 = (2+\iu)(2-\iu)$
        et sur $3$.
\end{enumerate}

\medskip
\noindent\textbf{Partie IV --- Le théorème des deux carrés.}
\begin{enumerate}[resume]
  \item Démontrer le sens direct~: si dans la factorisation $n =
        \prod p_i^{\alpha_i}$ tout premier $\equiv 3 \pmod 4$
        apparaît avec un exposant pair, alors $n$ est une somme
        de deux carrés. \emph{(Brahmagupta + parties II--III.)}
  \item Démontrer la réciproque~: si $n = a^2 + b^2 = N(a + \iu
        b)$ et $q \equiv 3 \pmod 4$ divise $n$, montrer que $q$,
        un premier de Gauss, divise $a + \iu b$ ou $a - \iu b$,
        qu'il divise en fait $a$ et $b$, et conclure par
        récurrence sur $n$ que l'exposant de $q$ dans $n$ est
        pair.
  \item Énoncer le théorème final. Parmi $2025$, $2026$, $2027$,
        lesquels sont sommes de deux carrés~? \emph{($2025 = 81
        \cdot 25$~; $2026 = 2 \cdot 1013$, $1013$ premier~;
        $2027$ premier.)}
  \item (Épilogue) Montrer qu'un premier $p \equiv 1 \pmod 4$
        est une somme de deux carrés de façon essentiellement
        unique~: si $p = a^2 + b^2 = c^2 + d^2$ (entiers
        positifs), alors $\{a, b\} = \{c, d\}$.
        \emph{(Unicité de la factorisation dans $\Z[\iu]$.)}
\end{enumerate}

\medskip
\noindent\textbf{Partie V --- Compter les représentations~:
formule de Jacobi et série de Leibniz.} Noter $r_2(n) =
\#\{(a, b) \in \Z^2 : a^2 + b^2 = n\}$ (couples ordonnés,
signes et zéros inclus), et soit $\chi$ le caractère non trivial
mod $4$~: $\chi(d) = +1$ si $d \equiv 1$, $-1$ si $d \equiv 3
\pmod4$, $0$ si $d$ pair.
\begin{enumerate}[resume]
  \item (Échauffement, par contraste) Quels entiers sont des
        \emph{différences} de deux carrés~? Montrer~: $n = a^2 -
        b^2$ avec $a, b \in \Z$ ssi $n \not\equiv 2 \pmod 4$ ---
        pas de théorie des anneaux nécessaire, et aucune
        structure comparable à ce qui suit.
  \item Montrer que $r_2(n)$ est le nombre de $z \in \Z[\iu]$
        avec $N(z) = n$. En écrivant $n = 2^{a}\prod_jp_j^{b_j}
        \prod_kq_k^{c_k}$ avec $p_j \equiv 1$, $q_k \equiv 3
        \pmod4$, utiliser la classification de la question 11 et
        la factorisation unique pour montrer~: de tels $z$
        existent ssi tous les $c_k$ sont pairs, et dans ce cas
        \[
        r_2(n) = 4\prod_j\,(b_j + 1) .
        \]
        \emph{(Compter~: $z = u\,(1+\iu)^{a}\prod_j\pi_j^{s_j}
        \bar\pi_j^{\,b_j - s_j}\prod_kq_k^{c_k/2}$ avec $u$ une
        unité et $0 \leq s_j \leq b_j$~; pourquoi cette liste
        est-elle exhaustive et sans répétition~?)}
  \item Montrer que $d \mapsto \chi(d)$ est complètement
        multiplicative, en déduire que $n \mapsto
        \sum_{d \mid n}\chi(d)$ est multiplicative, et la
        calculer sur les puissances de premiers~: elle vaut $1$
        sur $2^a$~; $b + 1$ sur $p^b$ ($p \equiv 1$)~; $1$ ou
        $0$ sur $q^c$ ($q \equiv 3$) selon que $c$ est pair ou
        impair.
  \item Conclure le \emph{théorème de Jacobi}~:
        \[
        r_2(n) = 4\sum_{d \mid n}\chi(d) =
        4\bigl(d_1(n) - d_3(n)\bigr),
        \]
        où $d_i(n)$ compte les diviseurs $\equiv i \pmod 4$.
        Vérifier sur $n = 3, 5, 9, 25$, et lister les $16$
        représentations de $65$.
  \item (Le cercle) Montrer que $\sum_{n \leq x}r_2(n)$ est le
        nombre de points de réseau de $\Z^2$ dans le disque
        fermé de rayon $\sqrt x$, et démontrer
        \[
        \sum_{n\leq x}r_2(n) = \pi x + O(\sqrt x)
        \]
        \emph{(chaque point de réseau possède un carré unité~;
        comparer les aires, l'erreur vivant dans un anneau de
        largeur $O(1)$)}.
  \item (Leibniz, lu arithmétiquement) Combiner les questions
        19--20~:
        \[
        \sum_{d \leq x}\chi(d)\Bigl\lfloor\frac
        xd\Bigr\rfloor = \frac{\pi x}4 + O(\sqrt x),
        \]
        et en déduire --- en enlevant les parties entières avec
        soin --- la série de Leibniz
        \[
        1 - \frac13 + \frac15 - \frac17 + \dots = \frac\pi4 .
        \]
        La série alternée des inverses impairs \emph{est}
        l'excès moyen de diviseurs $\equiv 1$ sur les diviseurs
        $\equiv 3$~: de l'analyse calculée par l'arithmétique.
  \item (Quelle rareté les sommes de deux carrés~?) Montrer
        qu'aucun entier $\equiv 3 \pmod 4$ n'est une somme de
        deux carrés (deux façons~: carrés mod $4$, ou le
        critère de parité de la question 17), donc au moins un
        quart de tous les entiers sont manqués~; et montrer que
        la moyenne $\frac1x\sum_{n\leq x}r_2(n) \to \pi$ de la
        question 20 est compatible avec des entiers
        représentables de densité $0$ --- exhiber des entiers
        avec un nombre anormalement grand de représentations
        (prendre des produits de nombreux premiers $\equiv 1
        \bmod 4$) pour expliquer comment une proportion
        s'évanouissant peut encore porter une moyenne positive.
        (Landau a démontré que la vraie densité décroît comme
        $1/\sqrt{\log x}$~; cela dépasse nos outils, mais le
        mécanisme est désormais visible.)
\end{enumerate}

\medskip
\noindent\textbf{Partie VI --- Compléments~: représentations
primitives et Pythagore.}
\begin{enumerate}[resume]
  \item Appeler une représentation $n = a^2 + b^2$
        \emph{primitive} si $\gcd(a, b) = 1$. Montrer que $n
        \geq 1$ admet une représentation primitive ssi $4 \nmid
        n$ et aucun premier $q \equiv 3 \pmod 4$ ne divise $n$.
        \emph{(Pour la nécessité, réutiliser la descente de la
        question 13 et les carrés mod $4$~; pour la suffisance,
        construire $z$ à partir de $1 + \iu$ et des $\pi_j$
        seulement --- pas de conjugués --- et expliquer
        pourquoi un facteur premier commun de $a$ et $b$
        forcerait à la fois $\pi_j$ et $\bar\pi_j$, ou
        $(1+\iu)^2$, dans $z$.)}
  \item (Triplets pythagoriciens) Soit $a^2 + b^2 = c^2$ avec
        $a, b, c$ positifs, $\gcd(a, b) = 1$ et $b$ pair.
        Montrer que $a + \iu b$ et $a - \iu b$ sont premiers
        entre eux dans $\Z[\iu]$ \emph{(un diviseur premier de
        Gauss commun diviserait $2a$ et $2b$, et $c$ est
        impair)}, en déduire de la factorisation unique que $a
        + \iu b = u(m + \iu n)^2$ pour une unité $u$, et
        conclure la paramétrisation classique~: quitte à
        échanger $a$ et $b$,
        \[
        a = m^2 - n^2, \qquad b = 2mn, \qquad c = m^2 + n^2,
        \]
        avec $m > n \geq 1$ premiers entre eux de parités
        opposées. Retrouver $(3, 4, 5)$ et $(21, 20, 29)$ à
        partir de $(m, n) = (2, 1)$ et $(5, 2)$.
  \item (Vérification numérique) Prendre $x = 25$. Calculer
        $r_2(n)$ pour $1 \leq n \leq 25$ à partir de la formule
        de Jacobi, vérifier que les valeurs non nulles
        apparaissent exactement en $n = 1, 2, 4, 5, 8, 9, 10,
        13, 16, 17, 18, 20, 25$, et que
        \[
        \sum_{n \leq 25} r_2(n) = 80
        = 4\sum_{d \leq 25}\chi(d)
        \Bigl\lfloor\frac{25}d\Bigr\rfloor .
        \]
        Vérifier que le disque fermé de rayon $5$ contient $81$
        points de réseau, et comparer avec $\pi x \approx
        78{,}5$~: l'erreur est bien dans le $O(\sqrt x)$ de la
        question 20.
\end{enumerate}
\end{problem}
