\chapter{Séries de Laurent et théorème des résidus}
\label{ch:b3:residues}

Que devient une \omterm{def:b3:holomorphic:holo}{fonction holomorphe} près d'un point où elle
n'est \emph{pas} définie~? La réponse est une trichotomie
complète --- point éliminable, pôle, ou singularité
essentielle --- lue sur une série entière à deux côtés, le
développement de Laurent. Un coefficient de ce développement,
le \emph{\omterm{def:b3:residues:singularities}{résidu}}, contrôle toute \omterm{def:b3:holomorphic:contour}{intégrale de contour} autour
de la singularité~: le théorème des \omterm{def:b3:residues:singularities}{résidus} convertit des
intégrales définies difficiles en algèbre finie, compte les
zéros des fonctions (principe de l'argument, Rouché), et
prouve le théorème de l'application ouverte. Nous commençons
par remonter le théorème de Cauchy des domaines étoilés à sa
forme définitive, sans homologie --- l'élégant argument de
Dixon --- de sorte que tous les contours d'\omterm{def:b3:holomorphic:index}{indice} zéro autour
du complémentaire deviennent disponibles.

\section{Le théorème de Cauchy global}

Un \emph{cycle}\index{cycle} $\Gamma$ est une somme formelle
finie de chemins fermés $\gamma_1, \dots, \gamma_m$~; les
intégrales et \omterm{def:b3:holomorphic:index}{indices} le long de $\Gamma$ sont les sommes
correspondantes, et $\operatorname{im}\Gamma =
\bigcup\operatorname{im}\gamma_j$.

\begin{theorem}[Cauchy, forme globale]
\label{thm:b3:residues:globalcauchy}
Soit $\Omega \subseteq \C$ \omterm{def:b3:topology:topology}{ouvert}, $f \in \mathcal H(\Omega)$,
et $\Gamma$ un cycle dans $\Omega$ tel que
\[
\operatorname{Ind}_\Gamma(w) = 0
\qquad\text{pour tout } w \notin \Omega .
\]
Alors, pour tout $z \in \Omega\setminus\operatorname{im}\Gamma$,
\[
\frac1{2\iu\pi}\int_\Gamma\frac{f(w)}{w - z}\,\dd w =
\operatorname{Ind}_\Gamma(z)\,f(z),
\qquad\text{et}\qquad
\int_\Gamma f(w)\,\dd w = 0 .
\]
\index{théorème de Cauchy (global)}
\end{theorem}

\begin{proof}[Preuve (Dixon)]
Définir $g \colon \Omega\times\Omega \to \C$ par
\[
g(z, w) = \begin{cases}
\dfrac{f(w) - f(z)}{w - z} & w \neq z,\\[4pt]
f'(z) & w = z .
\end{cases}
\]
\emph{$g$ est \omterm{def:b3:topology:continuity}{continue}}~: hors de la diagonale, clair. Près
d'un point diagonal $(a, a)$, développer $f$ en série entière
en $a$ (le~\cref{thm:b3:holomorphic:analytic})~: $f(w) - f(z)
= \sum_{n\geq1}c_n\bigl((w-a)^n - (z-a)^n\bigr)$, et diviser
chaque terme par $w - z$ (factorisation de $u^n - v^n$) donne,
pour $z, w \in D(a, r)$,
\[
g(z, w) =
\sum_{n\geq1}c_n\sum_{j=0}^{n-1}(w-a)^{\,j}(z-a)^{\,n-1-j},
\]
valable aussi sur la diagonale (chaque somme intérieure
devient $n(z-a)^{n-1}$, sommant à $f'(z)$). Pour $r$ petit la
série converge uniformément sur $D(a,r)^2$
($\abs{\text{terme}} \leq n\abs{c_n}r^{n-1}$, sommable dans
le rayon)~: la somme est \omterm{def:b3:topology:continuity}{continue}.

Poser $h(z) = \frac1{2\iu\pi}\int_\Gamma g(z, w)\,\dd w$ sur
$\Omega$~: \omterm{def:b3:topology:continuity}{continue} (continuité uniforme de $g$ sur les
compacts), et \omterm{def:b3:holomorphic:holo}{holomorphe} --- par Morera (critère
du~\cref{thm:b3:holomorphic:weierstrassconv})~: pour un
triangle $T \subseteq \Omega$, Fubini donne $\int_{\partial
T}h = \frac1{2\iu\pi}\int_\Gamma\bigl(\int_{\partial T}g(z,
w)\dd z\bigr)\dd w = 0$, l'intégrale intérieure s'annulant
car $z \mapsto g(z, w)$ est \omterm{def:b3:holomorphic:holo}{holomorphe} sur $\Omega$ (en $z =
w$ la singularité est éliminable~: $g$ est \omterm{def:b3:topology:continuity}{continue} là et
\omterm{def:b3:holomorphic:holo}{holomorphe} ailleurs --- l'argument d'extension de la preuve
du~\cref{thm:b3:holomorphic:formula}).

Sur l'\omterm{def:b3:topology:topology}{ouvert} $\Omega' = \{z \notin \operatorname{im}\Gamma
: \operatorname{Ind}_\Gamma(z) = 0\}$, définir $h_1(z) =
\frac1{2\iu\pi}\int_\Gamma\frac{f(w)}{w - z}\dd w$~:
\omterm{def:b3:holomorphic:holo}{holomorphe} sur $\Omega'$ (Morera ou dérivation sous
l'intégrale). Pour $z \in \Omega\cap\Omega'$~:
\[
h(z) = \frac1{2\iu\pi}\int_\Gamma\frac{f(w)}{w-z}\dd w
- f(z)\operatorname{Ind}_\Gamma(z) = h_1(z) .
\]
Par hypothèse $\Omega\cup\Omega' = \C$ ($w \notin \Omega
\Rightarrow \operatorname{Ind}_\Gamma(w) = 0$), donc $h$ et
$h_1$ se collent en une fonction entière $H$. Comme la
composante non bornée du complémentaire de
$\operatorname{im}\Gamma$ est dans $\Omega'$ et $h_1(z) \to
0$ quand $\abs z \to \infty$ (borne ML), $H$ est bornée et
tend vers $0$~: Liouville (le~\cref{cor:b3:holomorphic:liouville})
donne $H \equiv 0$. Ainsi $h \equiv 0$ sur $\Omega$, ce qui
est la formule intégrale. En l'appliquant, pour $a \in
\Omega\setminus\operatorname{im}\Gamma$ fixe, à $\tilde f(w)
= (w - a)f(w)$ en $z = a$~:
\[
\frac1{2\iu\pi}\int_\Gamma f(w)\dd w =
\frac1{2\iu\pi}\int_\Gamma \frac{\tilde f(w)}{w - a}\dd w =
\operatorname{Ind}_\Gamma(a)\,\tilde f(a) = 0 .
\]
\end{proof}

\section{Séries de Laurent et singularités isolées}

\begin{theorem}[Développement de Laurent]\label{thm:b3:residues:laurent}
Soit $f$ \omterm{def:b3:holomorphic:holo}{holomorphe} sur l'anneau $A = \{r < \abs{z - a} <
R\}$ ($0 \leq r < R \leq \infty$). Alors\index{série de
Laurent}
\[
f(z) = \sum_{n\in\Z}c_n\,(z - a)^n
\qquad\text{sur } A,
\qquad
c_n = \frac1{2\iu\pi}\int_{C_\rho}\frac{f(w)}{(w -
a)^{n+1}}\,\dd w
\]
pour tout $r < \rho < R$ (indépendant de $\rho$), les deux
demi-séries convergeant normalement sur les sous-anneaux
compacts. Le développement est unique.
\end{theorem}

\begin{proof}
Fixer $r < \rho_1 < \abs{z - a} < \rho_2 < R$ et poser
$\Gamma = C_{\rho_2} - C_{\rho_1}$ (extérieur
trigonométrique, intérieur horaire)~: un cycle dans $A$ avec
$\operatorname{Ind}_\Gamma(w) = 0$ pour tout $w \notin A$
(points dans le petit disque~: $1 - 1 = 0$~; hors du grand~:
$0 - 0$). Par le~\cref{thm:b3:residues:globalcauchy},
$\operatorname{Ind}_\Gamma(z) = 1 - 0 = 1$ donne
\[
f(z) = \frac1{2\iu\pi}\int_{C_{\rho_2}}\frac{f(w)}{w - z}\dd
w - \frac1{2\iu\pi}\int_{C_{\rho_1}}\frac{f(w)}{w - z}\dd w .
\]
Développer le premier noyau comme dans
le~\cref{thm:b3:holomorphic:analytic} (puissances de
$\frac{z - a}{w - a}$, \omterm{def:b3:modules:module}{module} $< 1$)~: la partie non négative
$\sum_{n\geq0}c_n(z-a)^n$. Dans le second, développer
l'autre sens~: $\frac{-1}{w - z} = \frac1{(z-a)(1 - \frac{w -
a}{z - a})} = \sum_{m\geq0}\frac{(w-a)^m}{(z - a)^{m+1}}$,
normalement convergente sur $C_{\rho_1}$~: la partie négative
$\sum_{n\leq-1}c_n(z-a)^n$ avec les coefficients indiqués
(\omterm{def:b3:holomorphic:index}{indice} $n = -m-1$). Indépendance de $\rho$~: les intégrales
de coefficients sur $C_{\rho}$ et $C_{\rho'}$ diffèrent d'une
$\int_\Gamma$ sur un cycle d'\omterm{def:b3:holomorphic:index}{indice} nul de la \omterm{def:b3:holomorphic:holo}{fonction
holomorphe} $\frac{f(w)}{(w-a)^{n+1}}$ dans $A$~: zéro, par
le~\cref{thm:b3:residues:globalcauchy} encore. Unicité~:
intégrer $\sum c_n(z-a)^n$ contre $(z - a)^{-m-1}$ sur
$C_\rho$ terme à terme (convergence normale)~: seul $n = m$
survit.
\end{proof}

\begin{definition}\label{def:b3:residues:singularities}
Si $f$ est \omterm{def:b3:holomorphic:holo}{holomorphe} sur un disque percé $D(a, R)\setminus
\{a\}$, développer par Laurent ($r = 0$). Trois cas
exclusifs~:\index{singularité isolée}
\begin{itemize}
  \item tous les $c_n = 0$ pour $n < 0$~: \emph{\omterm{thm:b3:residues:riemanncw}{singularité
        éliminable}} (la série non négative étend $f$
        holomorphiquement en $a$)~;
  \item $c_n \neq 0$ pour un nombre fini, au moins un, $n <
        0$~: un \emph{pôle} d'ordre $m = -\min\{n : c_n \neq
        0\}$~; de façon équivalente $f = g/(z-a)^m$, $g$
        \omterm{def:b3:holomorphic:holo}{holomorphe}, $g(a) \neq 0$~; de façon équivalente
        $\abs{f(z)} \to \infty$ quand $z\to a$~;
  \item infiniment beaucoup de $c_n$ négatifs $\neq 0$~:
        \emph{singularité essentielle}.
\end{itemize}
Le \emph{résidu}\index{résidu} est $\operatorname{Res}(f, a)
= c_{-1}$. Une \omterm{def:b3:holomorphic:holo}{fonction holomorphe} sur $\Omega$ moins un
ensemble de pôles est \emph{méromorphe}\index{fonction
méromorphe} sur $\Omega$.
\end{definition}

\begin{theorem}[Riemann~; Casorati--Weierstrass]
\label{thm:b3:residues:riemanncw}
Soit $f$ \omterm{def:b3:holomorphic:holo}{holomorphe} sur $D(a,R)\setminus\{a\}$.
\begin{enumerate}
  \item (Riemann) Si $f$ est \emph{bornée} près de $a$, la
        singularité est éliminable.\index{singularité
        éliminable}
  \item (Casorati--Weierstrass) Si $a$ est essentielle, alors
        $f\bigl(D(a,\varepsilon)\setminus\{a\}\bigr)$ est
        \omterm{def:b3:topology:interior}{dense} dans $\C$ pour tout $\varepsilon$.
        \index{théorème de Casorati--Weierstrass}
\end{enumerate}
\end{theorem}

\begin{proof}
(1) Pour $n < 0$ et $\rho \to 0$~: $\abs{c_n} \leq
M\rho^{-n-1}\cdot\rho\cdot\rho^{-\,n}\dots$ par ML sur
$C_\rho$~: $\abs{c_n} \leq \frac{1}{2\pi}\,2\pi\rho\cdot
M\rho^{-(n+1)} = M\rho^{-n} \to 0$ (car $-n > 0$)~: tous les
coefficients négatifs s'annulent.
(2) Si une valeur $b$ n'était pas approchée~: $\abs{f - b}
\geq \delta$ près de $a$, donc $g = 1/(f - b)$ est
\omterm{def:b3:holomorphic:holo}{holomorphe} et bornée près de $a$~: éliminable (1), $g$
s'étend avec valeur $c$. Si $c \neq 0$, $f = b + 1/g$ est
bornée près de $a$~: éliminable --- exclu. Si $c = 0$, $g$ a
un zéro d'ordre fini $m$ en $a$
(le~\cref{thm:b3:holomorphic:identity}~; $g \not\equiv 0$),
et $f = b + 1/g$ a un pôle d'ordre $m$~: exclu encore.
\end{proof}

\section{Le théorème des résidus}

\begin{theorem}[Théorème des résidus]\label{thm:b3:residues:residue}
Soit $\Omega$ \omterm{def:b3:topology:topology}{ouvert}, $S \subseteq \Omega$ fini, $f \in
\mathcal H(\Omega\setminus S)$, et $\Gamma$ un cycle dans
$\Omega\setminus S$ avec $\operatorname{Ind}_\Gamma(w) = 0$
pour tout $w \notin \Omega$. Alors\index{théorème des
résidus}
\[
\frac{1}{2\iu\pi}\int_\Gamma f(z)\,\dd z
= \sum_{a\in S}\operatorname{Ind}_\Gamma(a)\,
\operatorname{Res}(f, a) .
\]
\end{theorem}

\begin{proof}
Pour chaque $a \in S$, soit $P_a(z) =
\sum_{n\leq-1}c_n^{(a)}(z - a)^n$ la partie principale de
$f$ en $a$~: une série convergeant sur $\C\setminus\{a\}$
(son rayon en $1/(z-a)$ est infini~: la queue de Laurent
converge pour tout $\abs{z-a}$ petit, donc, étant une série
entière en $(z-a)^{-1}$, partout), et \omterm{def:b3:holomorphic:holo}{holomorphe} là. Alors
$g = f - \sum_{a\in S}P_a$ a des \omterm{thm:b3:residues:riemanncw}{singularités éliminables} en
chaque point de $S$ (son développement de Laurent en $a$ n'a
pas de partie négative~: les autres $P_{a'}$ sont
\omterm{def:b3:holomorphic:holo}{holomorphes} en $a$), donc $g$ s'étend holomorphiquement à
$\Omega$, et le~\cref{thm:b3:residues:globalcauchy} donne
$\int_\Gamma g = 0$. Il reste à intégrer chaque $P_a$~: terme
à terme (convergence normale sur le compact
$\operatorname{im}\Gamma$, qui évite $a$),
\[
\frac1{2\iu\pi}\int_\Gamma(z - a)^n\,\dd z = 0 \ (n \leq -2:
\text{primitive } \tfrac{(z-a)^{n+1}}{n+1}),
\qquad
\frac1{2\iu\pi}\int_\Gamma\frac{\dd z}{z - a} =
\operatorname{Ind}_\Gamma(a),
\]
donc $\frac1{2\iu\pi}\int_\Gamma P_a =
c_{-1}^{(a)}\operatorname{Ind}_\Gamma(a)$. Sommer sur $a$.
\end{proof}

\begin{method}[Calcul des résidus]\label{met:b3:residues:compute}
Pôle \omterm{def:b3:groups:simple}{simple}~: $\operatorname{Res}(f, a) = \lim_{z\to a}(z -
a)f(z)$~; pour $f = g/h$ avec $g(a) \neq 0$, $h(a) = 0$,
$h'(a) \neq 0$~: $\operatorname{Res} = g(a)/h'(a)$. Pôle
d'ordre $m$~: $\operatorname{Res}(f, a) =
\frac1{(m-1)!}\lim_{z\to a}\bigl((z-a)^mf(z)\bigr)^{(m-1)}$.
Singularités essentielles~: développer et lire $c_{-1}$ (p.\
ex.\ à partir de séries connues). Toujours vérifier quels
pôles le contour encercle réellement, et avec quel \omterm{def:b3:holomorphic:index}{indice}.
\end{method}

\begin{example}[Les quatre types classiques d'intégrales]
\label{ex:b3:residues:integrals}
(a) \emph{Rationnelle sur $\R$}~: pour $\int_\R\frac{\dd x}{1
+ x^4}$, fermer avec un grand demi-cercle $S_R$ dans le
demi-plan supérieur~: l'intégrande est $O(R^{-4})$ là, donc
$\int_{S_R} \to 0$ (ML), et le théorème des \omterm{def:b3:residues:singularities}{résidus} avec les
pôles $\eu^{\iu\pi/4}, \eu^{3\iu\pi/4}$ (\omterm{def:b3:groups:simple}{simples}, \omterm{def:b3:residues:singularities}{résidus}
$\frac1{4z^3} = \frac{z}{4z^4} = -\frac z4$ en un pôle)
donne
\[
\int_\R\frac{\dd x}{1 + x^4}
= 2\iu\pi\Bigl(-\frac{\eu^{\iu\pi/4}}4 -
\frac{\eu^{3\iu\pi/4}}4\Bigr)
= \frac{\pi}{\sqrt2} .
\]
(b) \emph{Type Fourier}~: pour $t \geq 0$,
$\int_\R\frac{\eu^{\iu tx}}{1 + x^2}\dd x =
2\iu\pi\operatorname{Res}\bigl(\tfrac{\eu^{\iu tz}}{1+z^2},
\iu\bigr) = 2\iu\pi\frac{\eu^{-t}}{2\iu} = \pi\eu^{-t}$ ---
le demi-cercle supérieur marche car $\abs{\eu^{\iu tz}} =
\eu^{-t\operatorname{Im}z} \leq 1$ là~; en prenant les
parties réelles~: $\int_\R\frac{\cos(tx)}{1+x^2}\dd x =
\pi\eu^{-\abs t}$, réglant la formule admise de
l'\cref{exo:b3:lebesgue:10}.
(c) \emph{Trigonométrique sur une période}~: substituer $z =
\eu^{\iu t}$, $\cos t = \frac{z + z^{-1}}2$, $\dd t =
\frac{\dd z}{\iu z}$~: $\int_0^{2\pi}\frac{\dd t}{a + \cos
t}$ ($a > 1$) devient un comptage de \omterm{def:b3:residues:singularities}{résidus} dans le cercle
unité (l'\cref{exo:b3:residues:2}).
(d) \emph{Séries}~: apparier $f$ avec $\pi\cot(\pi z)$, dont
les pôles sont les entiers de \omterm{def:b3:residues:singularities}{résidu} $1$~: le problème de
week-end somme $\sum n^{-2}$ et $\sum n^{-4}$ ainsi.
\end{example}

\section{Le principe de l'argument et le théorème de Rouché}

\begin{theorem}[Principe de l'argument]
\label{thm:b3:residues:argument}
Soit $f$ \omterm{def:b3:residues:singularities}{méromorphe} sur $\Omega$, avec zéros $z_j$ (ordres
$m_j$) et pôles $p_k$ (ordres $\mu_k$), et $\gamma$ un chemin
fermé dans $\Omega$ les évitant tous, avec
$\operatorname{Ind}_\gamma = 0$ hors de $\Omega$. Alors
\index{principe de l'argument}
\[
\frac1{2\iu\pi}\int_\gamma\frac{f'(z)}{f(z)}\,\dd z
= \sum_j m_j\operatorname{Ind}_\gamma(z_j)
- \sum_k \mu_k\operatorname{Ind}_\gamma(p_k)
\]
(un nombre fini de termes sont non nuls). Pour un contour
\omterm{def:b3:groups:simple}{simple} trigonométrique, l'intégrale compte les zéros moins
les pôles à l'intérieur, avec multiplicité --- et égale le
\omterm{def:b3:holomorphic:index}{nombre de tours} du chemin image $f\circ\gamma$ autour de $0$.
\end{theorem}

\begin{proof}
Près d'un zéro d'ordre $m$~: $f = (z-a)^mg$, $g(a) \neq 0$,
donc $\frac{f'}f = \frac m{z - a} + \frac{g'}g$ avec le
second terme \omterm{def:b3:holomorphic:holo}{holomorphe} près de $a$~: un pôle \omterm{def:b3:groups:simple}{simple} de
\omterm{def:b3:residues:singularities}{résidu} $m$. Près d'un pôle d'ordre $\mu$~: $f =
(z-a)^{-\mu}g$ donne le \omterm{def:b3:residues:singularities}{résidu} $-\mu$. Ailleurs $\frac{f'}f$
est \omterm{def:b3:holomorphic:holo}{holomorphe}. (Les zéros et pôles d'\omterm{def:b3:holomorphic:index}{indice} non nul sont
dans une région compacte encerclée par $\gamma$~; par le
théorème d'identité ils y sont en nombre fini, $f \not\equiv
0$.) Appliquer le~\cref{thm:b3:residues:residue}. La dernière
remarque~: $\frac1{2\iu\pi}\int_\gamma\frac{f'}f =
\frac1{2\iu\pi}\int_{f\circ\gamma}\frac{\dd w}w =
\operatorname{Ind}_{f\circ\gamma}(0)$ (substituer $w =
f(\gamma(t))$).
\end{proof}

\begin{theorem}[Rouché]\label{thm:b3:residues:rouche}
Soient $f, g$ \omterm{def:b3:holomorphic:holo}{holomorphes} sur $\Omega$, et $\gamma$ un chemin
fermé avec $\operatorname{Ind}_\gamma \in \{0,1\}$, zéro hors
de $\Omega$ (un contour \omterm{def:b3:groups:simple}{simple}). Si\index{théorème de Rouché}
\[
\abs{g(z)} < \abs{f(z)} \qquad \text{sur }
\operatorname{im}\gamma,
\]
alors $f$ et $f + g$ ont le même nombre de zéros (avec
multiplicité) dans la région $\{\operatorname{Ind}_\gamma =
1\}$.
\end{theorem}

\begin{proof}
Pour $t \in \intcc01$, $f_t = f + tg$ n'a pas de zéro sur
$\operatorname{im}\gamma$ ($\abs{f_t} \geq \abs f - \abs g >
0$), donc
\[
N(t) = \frac1{2\iu\pi}\int_\gamma
\frac{f_t'(z)}{f_t(z)}\,\dd z
\]
est bien défini~; il compte les zéros dans la région
encerclée (le~\cref{thm:b3:residues:argument}~; pas de pôles).
$N$ est \omterm{def:b3:topology:continuity}{continue} en $t$ (l'intégrande est conjointement
\omterm{def:b3:topology:continuity}{continue}, les dénominateurs uniformément minorés ---
convergence dominée) et à valeurs entières~: constante.
$N(0) = N(1)$.
\end{proof}

\begin{corollary}[Théorème de l'application ouverte]
\label{cor:b3:residues:openmapping}
Une \omterm{def:b3:holomorphic:holo}{fonction holomorphe} non constante sur un \omterm{def:b3:topology:topology}{ouvert} \omterm{def:b3:topology:connected}{connexe}
est une application ouverte. En particulier (encore) le
principe du maximum vaut, et une bijection \omterm{def:b3:holomorphic:holo}{holomorphe} a une
inverse \omterm{def:b3:holomorphic:holo}{holomorphe}.\index{théorème de l'application ouverte
(holomorphe)}
\end{corollary}

\begin{proof}
Soit $f(a) = b$~; $f - b$ a un zéro d'ordre fini $m \geq 1$
en $a$ (théorème d'identité~: $f \not\equiv b$). Choisir $r$
avec $f - b$ sans zéro sur $\bar D(a, r)\setminus\{a\}$
(\omterm{thm:b3:holomorphic:identity}{zéros isolés}) et soit $\delta = \min_{\abs{z - a} =
r}\abs{f(z) - b} > 0$. Pour $\abs{w - b} < \delta$~: sur le
cercle, $\abs{(b - w)} < \delta \leq \abs{f - b}$, donc
Rouché ($f - b$ contre la constante $b - w$) dit que $f - w$
a exactement $m$ zéros dans $D(a, r)$~: tout tel $w$ est
atteint --- $f(D(a,r)) \supseteq D(b, \delta)$~: \omterm{def:b3:topology:topology}{ouvert}.
Principe du maximum~: un maximum intérieur de $\abs f$ est
impossible pour $f$ non constante, son image autour de $f(a)$
contenant des points de plus grand \omterm{def:b3:modules:module}{module}. Inverse~: une
bijection \omterm{def:b3:holomorphic:holo}{holomorphe} $f$ est ouverte, donc $f^{-1}$ est
\omterm{def:b3:topology:continuity}{continue}~; le zéro de $f - f(a)$ en $a$ est \omterm{def:b3:groups:simple}{simple} ($m \geq
2$ donnerait $m$ préimages de valeurs proches --- distinctes,
car $f'$ ne s'annule qu'en des points isolés, donc près de
$a$ les $m$ zéros de $f - w$ sont \omterm{def:b3:groups:simple}{simples} et distincts pour
$w$ générique petit~: contradiction avec l'injectivité)~;
alors $f'(a) \neq 0$ et le quotient différentiel de $f^{-1}$
converge~: $\bigl(f^{-1}\bigr)'(b) = 1/f'(a)$.
\end{proof}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:b3:residues:1}
Classer la singularité en $0$ et calculer le \omterm{def:b3:residues:singularities}{résidu}~:
\[
\frac{\sin z}{z},\qquad
\frac{\eu^z - 1}{z^2},\qquad
\frac{1}{z(z-1)^2},\qquad
\frac{\cos z}{z^3},\qquad
\eu^{1/z},\qquad
\frac1{\sin z} .
\]
Donner aussi le \omterm{def:b3:residues:singularities}{résidu} du troisième en $z = 1$ et du dernier
en $z = \pi$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b3:residues:2}
Pour $a > 1$ calculer, via $z = \eu^{\iu t}$~:
\[
\int_0^{2\pi}\frac{\dd t}{a + \cos t}
= \frac{2\pi}{\sqrt{a^2 - 1}} .
\]
Vérifier les comportements limites $a \to 1^+$ et $a \to
\infty$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:residues:3}
Calculer avec des contours en demi-cercle, en justifiant les
estimations d'arc~:
\[
\int_\R\frac{x^2}{1 + x^6}\,\dd x = \frac\pi3,
\qquad
\int_\R\frac{\dd x}{(1 + x^2)^{2}} = \frac\pi2
\quad\text{(retrouver l'\cref{exo:b3:fouriertransform:3})}.
\]
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:residues:4}
Prouver, pour $t \geq 0$ et $a > 0$~:
\[
\int_\R\frac{\cos(tx)}{x^2 + a^2}\,\dd x =
\frac{\pi}{a}\,\eu^{-at},
\]
et en déduire la transformée de Fourier de $x \mapsto
\eu^{-a\abs x}$ par inversion --- en comparant avec
l'\cref{exo:b3:fouriertransform:1}.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:residues:5}
Développer $f(z) = \dfrac1{(z-1)(z-2)}$ en \omterm{thm:b3:residues:laurent}{série de Laurent}
dans chacune des trois régions $\abs z < 1$, $1 < \abs z <
2$, $\abs z > 2$. Pourquoi les trois développements
diffèrent-ils~? Expliquer pourquoi le coefficient de
$z^{-1}$ dans les second et troisième développements n'est
\emph{pas} un \omterm{def:b3:residues:singularities}{résidu} de $f$ en $0$ ($f$ n'a pas de
singularité là), et calculer les vrais \omterm{def:b3:residues:singularities}{résidus} de $f$, en
$1$ et en $2$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:residues:6}
(a) Montrer que $\eu^{1/z}$ a une singularité essentielle en
$0$ et vérifier Casorati--Weierstrass à la main~: résoudre
$\eu^{1/z} = w$ explicitement pour tout $w \neq 0$, en
exhibant des solutions arbitrairement proches de $0$.
(b) Montrer que $\abs{\eu^{1/z}}$ est non bornée sur tout
\omterm{def:b3:topology:topology}{voisinage} percé de $0$ pourtant $\eu^{1/z}$ n'a pas de
pôle~: quelle limite échoue~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:residues:7}
Compter avec Rouché~:
(a) les zéros de $z^7 - 4z^3 + z - 1$ dans $\abs z < 1$~;
(b) les zéros de $z^4 + 5z + 1$ dans $\abs z < 1$ et dans $1
< \abs z < 2$~;
(c) reprouver d'Alembert--Gauss~: un polynôme monique de
\omterm{def:b3:galois:extension}{degré} $n$ a $n$ zéros dans un grand disque \emph{(comparer
avec $z^n$)}.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b3:residues:8}
(Hurwitz) Soit $f_n \to f$ uniformément sur les compacts,
$f_n \in \mathcal H(\Omega)$, $\Omega$ \omterm{def:b3:topology:connected}{connexe}, $f
\not\equiv 0$.
(a) Montrer que si toutes les $f_n$ sont sans zéro, il en va
de même de $f$. \emph{(Si $f(a) = 0$~: principe de l'argument
sur un petit cercle autour de $a$, et
le~\cref{thm:b3:holomorphic:weierstrassconv} pour passer à
la limite dans $\int f_n'/f_n$.)}
(b) Montrer que si toutes les $f_n$ sont injectives, $f$ est
injective ou constante. \emph{(Appliquer (a) à $z \mapsto
f_n(z) - f_n(w)$ sur $\Omega\setminus\{w\}$.)}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b3:residues:9}
Pour $n \geq 2$, intégrer $\frac1{1 + z^n}$ sur le bord du
secteur $\{0 \leq \arg z \leq \frac{2\pi}n,\ \abs z \leq R\}$
et en déduire
\[
\int_0^{+\infty}\frac{\dd x}{1 + x^n} =
\frac{\pi}{n\,\sin(\pi/n)} .
\]
Vérifier $n = 2$ contre $\arctan$, et la limite $n \to
\infty$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:residues:10}
Soit $f$ une fraction rationnelle avec
$\deg(\text{dénominateur}) \geq \deg(\text{numérateur}) +
2$. Montrer que la somme de \emph{tous} les \omterm{def:b3:residues:singularities}{résidus} de $f$
est nulle \emph{(intégrer sur des cercles de plus en plus
grands)}. Utiliser cela pour recalculer la décomposition en
éléments \omterm{def:b3:groups:simple}{simples} de $\frac1{z(z-1)(z-2)}$ sans algèbre
linéaire.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b3:residues:11}
(Le trou de serrure~: l'intégrale de réflexion d'Euler) Pour
$0 < a < 1$, calculer
\[
I(a) = \int_0^{\infty}\frac{x^{a-1}}{1 + x}\,\dd x =
\frac{\pi}{\sin(\pi a)}
\]
en intégrant $f(z) = \frac{z^{a-1}}{1+z} =
\frac{\eu^{(a-1)\log z}}{1 + z}$ (logarithme coupé le long
de $\R_+$, $\arg z \in \intoo0{2\pi}$) sur le contour en
trou de serrure~: le long du haut de la coupure de
$\varepsilon$ à $R$, autour de $C_R$, sous la coupure en
retour, autour de $C_\varepsilon$. Justifier~: les deux
tronçons droits diffèrent du facteur $\eu^{2\iu\pi(a-1)}$,
les contributions des cercles s'annulent ($R^{a-1}\cdot R
\to 0$ et $\varepsilon^{a-1}\cdot\varepsilon \to 0$), et
l'unique pôle $z = -1$ a pour \omterm{def:b3:residues:singularities}{résidu} $\eu^{\iu\pi(a - 1)}$.
En déduire aussi $\Gamma(a)\Gamma(1 - a) =
\frac\pi{\sin\pi a}$ \emph{(écrire $\Gamma(a)\Gamma(1-a) =
B(a, 1-a)$ par le~\cref{pb:b3:lebesgue:1} et substituer $t =
\frac{x}{1+x}$)}.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:residues:12}
(Compter les zéros avec le principe de l'argument,
numériquement) Soit $P(z) = z^4 + 8z + 1$.
(a) Combien de zéros dans le disque unité~? \emph{(Rouché
contre $8z + 1$.)}
(b) Combien dans l'anneau $1 < \abs z < 3$~? \emph{(Rouché
contre $z^4$ sur $\abs z = 3$.)} Affiner~: montrer que tout
zéro a un \omterm{def:b3:modules:module}{module} $< 2{,}1$.
(c) Combien dans le demi-plan droit~? \emph{(Compter d'abord
sur $\abs z = 2$~; puis suivre l'image de l'axe imaginaire~:
$P(\iu t) = t^4 + 1 + 8\iu t$ a une partie réelle positive
partout, donc pas de zéro sur l'axe, et la variation
d'argument le long de celui-ci est calculable --- conclure
avec un grand demi-disque.)}
\end{exercise}

\section{Problème~: \texorpdfstring{$\zeta(2k)$}{zeta(2k)} par
la cotangente}

\begin{problem}[{Problème de week-end --- sommer $\sum
n^{-2k}$ avec les \omterm{def:b3:residues:singularities}{résidus}}]\label{pb:b3:residues:1}
Le théorème des \omterm{def:b3:residues:singularities}{résidus} somme les séries~: en appariant une
fraction rationnelle avec $\pi\cot(\pi z)$, dont les pôles
siègent aux entiers, on transforme $\sum_{n}f(n)$ en un
comptage de \omterm{def:b3:residues:singularities}{résidus}. Nous prouvons la méthode et calculons
$\zeta(2) = \frac{\pi^2}{6}$ et $\zeta(4) =
\frac{\pi^4}{90}$ --- les valeurs trouvées par séries de
Fourier en deuxième année et par traces d'opérateurs dans
le~\cref{ch:b3:spectral}, maintenant par intégration de
contour.

\medskip
\noindent\textbf{Partie~I --- Le noyau cotangente.}
\begin{enumerate}
  \item Montrer que $\pi\cot(\pi z)$ est \omterm{def:b3:residues:singularities}{méromorphe} sur $\C$
        avec des pôles \omterm{def:b3:groups:simple}{simples} exactement en $z = n \in \Z$,
        chacun de \omterm{def:b3:residues:singularities}{résidu} $1$ \emph{(calculer $\lim_{z\to
        n}(z-n)\pi\cot\pi z$)}.
  \item Calculer le début du développement de Laurent en
        $0$~:
        \[
        \pi\cot(\pi z) = \frac1z - \frac{\pi^2}{3}\,z -
        \frac{\pi^4}{45}\,z^3 + O(z^5) ,
        \]
        en divisant la série entière de $\cos$ par celle de
        $\sin$ (justifier la division~: $\frac{\sin\pi z}{\pi
        z}$ est \omterm{def:b3:holomorphic:holo}{holomorphe} et non nulle près de $0$, donc sa
        réciproque est \omterm{def:b3:holomorphic:holo}{holomorphe}~; identifier les
        coefficients jusqu'à l'ordre $3$).
  \item Soit $C_N$ le bord du carré de sommets
        $(\pm1\pm\iu)(N + \frac12)$. Montrer que
        $\abs{\cot(\pi z)} \leq 2$ sur $C_N$ pour tout $N
        \geq 1$. \emph{(Sur les côtés verticaux,
        $\cot(\pi(\pm(N + \frac12) + \iu y)) =
        \mp\tan(\iu\pi y)$, de \omterm{def:b3:modules:module}{module} $\abs{\tanh(\pi y)}
        \leq 1$~; sur les côtés horizontaux $\abs y = N +
        \frac12$, borner $\abs{\cot(\pi(x\pm\iu y))} \leq
        \coth(\pi y) \leq \coth(\pi/2) < 1{,}1$.)}
\end{enumerate}

\medskip
\noindent\textbf{Partie~II --- Le théorème de sommation.}
\begin{enumerate}[resume]
  \item Soit $f$ rationnelle, \omterm{def:b3:holomorphic:holo}{holomorphe} aux entiers, avec
        $\deg(\text{dén}) \geq \deg(\text{num}) + 2$. En
        utilisant le théorème des \omterm{def:b3:residues:singularities}{résidus} sur $C_N$ et la
        borne de la question~3, prouver~:
        \[
        \lim_{N\to\infty}\ \sum_{n = -N}^{N} f(n)
        = -\sum_{p\ \text{pôle de}\ f}
        \operatorname{Res}\bigl(\pi\cot(\pi z)f(z),\,p\bigr).
        \]
  \item Où l'argument a-t-il besoin de la condition de
        \omterm{def:b3:galois:extension}{degré}~? Montrer par un exemple (prendre $f(z) =
        1/(z + \frac12)$) que pour une décroissance plus
        lente la limite symétrique peut encore exister
        tandis que la série bilatère diverge --- et que la
        formule calcule alors la \emph{valeur principale}.
\end{enumerate}

\medskip
\noindent\textbf{Partie~III --- Les valeurs.}
\begin{enumerate}[resume]
  \item Appliquer la méthode à $f(z) = 1/z^2$~: ici $f$ a
        son pôle \emph{en} un entier, donc mener l'argument
        directement --- intégrer $g(z) =
        \frac{\pi\cot(\pi z)}{z^2}$ sur $C_N$, montrer que
        l'intégrale $\to 0$, et calculer
        $\operatorname{Res}(g, 0)$ à partir de la
        question~2. Conclure~:
        \[
        2\sum_{n\geq1}\frac1{n^2} = \frac{\pi^2}{3},
        \qquad \zeta(2) = \frac{\pi^2}6 .
        \]
  \item Idem avec $g(z) = \frac{\pi\cot(\pi z)}{z^4}$~:
        calculer $\operatorname{Res}(g, 0)$ et en déduire
        $\zeta(4) = \frac{\pi^4}{90}$.
  \item Expliquer le schéma général~: pour tout $k \geq 1$,
        $\zeta(2k)$ est $-\frac12$ fois le coefficient de
        $z^{2k-1}$ dans le développement de Laurent de
        $\pi\cot(\pi z)$ en $0$ --- un multiple rationnel de
        $\pi^{2k}$. Calculer $\zeta(6)$ en poussant la
        division de la question~2 d'un cran. Que dit la
        méthode sur $\zeta(3)$ --- et pourquoi ne dit-elle
        rien~?
\end{enumerate}

\medskip
\noindent\textbf{Partie~IV --- Le développement en éléments
\omterm{def:b3:groups:simple}{simples} de la cotangente.}
\begin{enumerate}[resume]
  \item Fixer $w \in \C\setminus\Z$ et appliquer la méthode
        de la partie~II à $f(z) = \dfrac{1}{(z - w)(z +
        w)}$ --- en notant que $\pi\cot(\pi z)f(z)$ a
        maintenant des pôles \omterm{def:b3:groups:simple}{simples} supplémentaires en
        $\pm w$, dont les \omterm{def:b3:residues:singularities}{résidus} doivent rejoindre le
        comptage. En déduire le développement en éléments
        \omterm{def:b3:groups:simple}{simples}
        \[
        \pi\cot(\pi w) = \frac1w +
        \sum_{n\geq1}\frac{2w}{w^2 - n^2},
        \]
        la série convergeant normalement sur les
        sous-ensembles compacts de $\C\setminus\Z$.
  \item Retrouver à partir de ce développement, en
        développant chaque terme en puissances de $w$
        (justifier l'interversion), les \emph{mêmes}
        coefficients de Laurent qu'à la question~2 --- le
        cercle se ferme~: la formule d'Euler $\sum\frac1{n^2}
        = \frac{\pi^2}6$ est le coefficient de $w$ dans les
        deux faces de la cotangente. Comparer avec la
        preuve par séries de Fourier (deuxième année) et la
        preuve par traces (le~\cref{pb:b3:spectral:1})~:
        trois théories, un nombre.
\end{enumerate}

\medskip
\noindent\textbf{Partie~V --- Le produit d'Euler pour le
sinus.} Le développement de la question~9 est la dérivée
logarithmique d'un produit infini~; nous prouvons maintenant
honnêtement la factorisation d'Euler de 1734.
\begin{enumerate}[resume]
  \item Pour $N \geq 1$ poser $P_N(z) =
        z\prod_{n=1}^{N}\bigl(1 - \frac{z^2}{n^2}\bigr)$.
        Montrer que $P_N$ converge, uniformément sur tout
        disque $\bar D(0, R)$, vers une fonction entière $P$
        dont les zéros sont exactement les entiers, tous
        \omterm{def:b3:groups:simple}{simples}. \emph{(Pour $n \geq 2R$ écrire le facteur
        comme $\exp\log(1 - z^2/n^2)$ avec le logarithme
        principal de l'\cref{exo:b3:holomorphic:3}, borner
        $\abs{\log(1+u)} \leq 2\abs u$ pour $\abs u \leq
        \frac12$ via la série, et exponentier la somme
        normalement convergente de logarithmes~; les
        facteurs restants en nombre fini sont un polynôme.
        Conclure avec le~\cref{thm:b3:holomorphic:weierstrassconv}.)}
  \item Montrer que sur $\C\setminus\Z$,
        \[
        \frac{P'(z)}{P(z)} = \frac1z +
        \sum_{n\geq1}\frac{2z}{z^2 - n^2} = \pi\cot(\pi z)
        \]
        \emph{(dériver les produits finis, passer à la
        limite en utilisant
        le~\cref{thm:b3:holomorphic:weierstrassconv} et
        l'absence de zéro de $P$ hors de $\Z$, et citer la
        question~9)}.
  \item Montrer que $Q = \sin(\pi z)/P(z)$ s'étend en une
        fonction entière sans zéro avec $Q' = 0$, et
        conclure le \emph{produit d'Euler}~:
        \[
        \sin(\pi z) = \pi z\prod_{n\geq1}
        \Bigl(1 - \frac{z^2}{n^2}\Bigr)
        \qquad (z \in \C) .
        \]
  \item (Wallis, 1655) Évaluer en $z = \frac12$~:
        \[
        \frac\pi2 = \prod_{n\geq1}\frac{4n^2}{4n^2 - 1}
        = \lim_{N\to\infty}
        \frac{2\cdot2\cdot4\cdot4\cdots(2N)(2N)}
        {1\cdot3\cdot3\cdot5\cdots(2N-1)(2N+1)} .
        \]
  \item Pour $\abs z < 1$, développer le logarithme du
        produit en série double (justifier le
        réarrangement) et retrouver $\zeta(2) =
        \frac{\pi^2}6$ en identifiant le coefficient de
        $z^3$ dans $\sin(\pi z) = \pi z - \frac{\pi^3}6z^3 +
        \cdots$ --- la face produit du nombre d'Euler.
\end{enumerate}

\medskip
\noindent\textbf{Partie~VI --- Noyaux frères.} La cotangente
a des frères~; chacun tarife sa propre famille de séries.
\begin{enumerate}[resume]
  \item Dériver terme à terme le développement de la
        question~9 (justifié par
        le~\cref{thm:b3:holomorphic:weierstrassconv}) pour
        obtenir, normalement sur les compacts de
        $\C\setminus\Z$,
        \[
        \frac{\pi^2}{\sin^2(\pi z)} =
        \sum_{n\in\Z}\frac1{(z - n)^2} .
        \]
  \item Évaluer en $z = \frac12$~:
        $\sum_{m\geq0}\frac1{(2m+1)^2} = \frac{\pi^2}8$~;
        retrouver $\zeta(2)$ encore une fois en scindant les
        entiers par parité.
  \item Vérifier l'identité de duplication $\tan\theta =
        \cot\theta - 2\cot(2\theta)$ et en déduire
        \[
        \pi\tan(\pi z) =
        \sum_{m\geq0}\frac{8z}{(2m+1)^2 - 4z^2} ,
        \]
        normalement sur les compacts évitant $\frac12 + \Z$.
  \item Développer autour de $0$ ($\abs z < \frac12$~;
        Fubini encore)~: avec $\lambda(s) =
        \sum_{m\geq0}(2m+1)^{-s}$,
        \[
        \pi\tan(\pi z) =
        \sum_{k\geq0}8\cdot4^k\,\lambda(2k+2)\,z^{2k+1} ;
        \]
        comparer avec $\tan u = u + \frac{u^3}3 + O(u^5)$
        pour retrouver $\lambda(2) = \frac{\pi^2}8$ et
        obtenir $\lambda(4) = \frac{\pi^4}{96}$, puis
        recouper $\zeta(4) = \frac{\pi^4}{90}$ via
        $\lambda(4) = (1 - 2^{-4})\,\zeta(4)$.
  \item Vérifier $\frac1{\sin\theta} = \cot\frac\theta2 -
        \cot\theta$ et en déduire
        \[
        \frac{\pi}{\sin(\pi z)} = \frac1z +
        \sum_{n\geq1}(-1)^n\,\frac{2z}{z^2 - n^2} .
        \]
        Vérifier les signes contre les \omterm{def:b3:residues:singularities}{résidus} de
        $\pi/\sin(\pi z)$ aux entiers.
  \item Lire le coefficient de $z$~: $\eta(2) =
        \sum_{n\geq1}\frac{(-1)^{n-1}}{n^2} =
        \frac{\pi^2}{12}$, et confirmer la cohérence
        $\eta(2) = (1 - 2^{1-2})\,\zeta(2)$.
  \item (Finale) Évaluer le développement de la question~20
        en $z = \frac12$ et en déduire la \emph{formule de
        Leibniz}
        \[
        \frac\pi4 = 1 - \frac13 + \frac15 - \frac17 +
        \cdots
        \]
        Clore par un court paragraphe~: un noyau par
        arithmétique --- quel noyau tarife quelle famille de
        séries, et pourquoi tous sont structurellement
        aveugles à $\zeta(3)$.

\end{enumerate}
\medskip
\noindent\textbf{Partie~VII --- La grille tarifaire
complète~: nombres de Bernoulli.}
\begin{enumerate}[resume]
  \item Combiner le développement en éléments \omterm{def:b3:groups:simple}{simples} de
        $\pi z\cot(\pi z)$ avec la fonction génératrice des
        nombres de Bernoulli ($\frac{w}{\eu^w - 1} =
        \sum_n\frac{B_n}{n!}w^n$, le~\cref{pb:b3:holomorphic:1},
        partie~VI)~: de
        \[
        \pi z\cot(\pi z) = \iu\pi z +
        \frac{2\iu\pi z}{\eu^{2\iu\pi z} - 1}
        \]
        (prouver d'abord cette identité), déduire la forme
        fermée
        \[
        \zeta(2k) = (-1)^{k+1}\,
        \frac{(2\pi)^{2k}\,B_{2k}}{2\,(2k)!}
        \qquad (k \geq 1).
        \]
  \item Vérifier la formule contre $B_2 = \frac16$, $B_4 =
        -\frac1{30}$, $B_6 = \frac1{42}$~: retrouver
        $\zeta(2) = \frac{\pi^2}6$, $\zeta(4) =
        \frac{\pi^4}{90}$, et calculer $\zeta(6) =
        \frac{\pi^6}{945}$.
  \item (Récursion d'Euler) Développer les deux membres de
        $\bigl(z\cot z\bigr)' = \cot z - z(1 + \cot^2z)$ ---
        ou élever au carré la série de cotangente
        directement --- pour prouver
        \[
        \Bigl(k + \frac12\Bigr)\zeta(2k) =
        \sum_{j=1}^{k-1}\zeta(2j)\,\zeta(2k - 2j)
        \qquad (k \geq 2),
        \]
        et vérifier qu'elle calcule $\zeta(4)$ à partir de
        $\zeta(2)$ et $\zeta(6)$ à partir de $\zeta(2),
        \zeta(4)$ --- toutes les valeurs zêta paires à
        partir de la seule graine $\frac{\pi^2}6$, sans
        nouvelle intégration.
\end{enumerate}
\end{problem}
