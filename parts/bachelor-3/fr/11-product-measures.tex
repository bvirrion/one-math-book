\chapter{Mesures produit, Fubini, changement de variables}
\label{ch:b3:product}

La théorie de Lebesgue unidimensionnelle devient le calcul
multidimensionnel à travers deux théorèmes. \emph{Tonelli--Fubini}
affirme que les intégrales sur des produits sont des intégrales
itérées --- le découpage en tranches est légitime, dans l'un ou
l'autre ordre, sous des hypothèses que l'on peut réellement
vérifier. La \emph{formule de changement de variables} transporte
les intégrales le long des difféomorphismes $\mathcal C^1$, avec le
déterminant jacobien comme taux de change du volume~; on la prouve
entièrement, en partant du cas linéaire où elle explique ce qu'est
le déterminant. Les applications en cascade~: la formule des
couches, la convolution, les \omterm{ex:b3:product:polar}{coordonnées polaires}, le volume de la
boule unité --- et, dans le problème du week-end, la formule de
Stirling avec une analyse d'erreur honnête.

\section{$\sigma$-algèbres produit et mesures produit}

\begin{definition}\label{def:b3:product:sigma}
Pour des espaces \omterm{def:b3:lebesgue:measurable}{mesurables} $(X, \mathcal A)$, $(Y, \mathcal B)$,
la \emph{$\sigma$-algèbre produit}\index{sigma-algebre
produit@$\sigma$-algèbre produit} $\mathcal A \otimes \mathcal B$
sur $X \times Y$ est engendrée par les \emph{rectangles} $A
\times B$ ($A \in \mathcal A$, $B \in \mathcal B$) --- un
$\pi$-système. Pour $E \subseteq X\times Y$ et $x \in X$, la
\emph{section} est $E_x = \{y : (x,y) \in E\}$~; pour une fonction
$f$ sur le produit, $f_x = f(x, \cdot)$.
\end{definition}

\begin{proposition}\label{prop:b3:product:sections}
(a) Si $E \in \mathcal A\otimes\mathcal B$, toute section $E_x
\in \mathcal B$ (et symétriquement)~; si $f$ est $\mathcal
A\otimes\mathcal B$-mesurable, toute $f_x$ est $\mathcal
B$-mesurable.
(b) $\mathcal B(\R^m)\otimes\mathcal B(\R^n) = \mathcal
B(\R^{m+n})$.
\end{proposition}

\begin{proof}
(a) Bons ensembles~: $\{E : E_x \in \mathcal B\ \forall x\}$ est
une $\sigma$-algèbre (les sections commutent avec les
complémentaires et les unions dénombrables) contenant les
rectangles. Pour $f$~: $(f_x)^{-1}(B) = (f^{-1}(B))_x$.
(b) ($\subseteq$) Rectangles de boréliens~: il suffit que les
boîtes \omterm{def:b3:topology:topology}{ouvert}$\times$\omterm{def:b3:topology:topology}{ouvert} soient boréliennes dans $\R^{m+n}$
(elles sont \omterm{def:b3:topology:topology}{ouvertes}) et que les rectangles boréliens généraux
soient des limites --- bons ensembles encore~: $\{A :
A\times\R^n \in \mathcal B(\R^{m+n})\}$ est une
$\sigma$-algèbre contenant les \omterm{def:b3:topology:topology}{ouverts}~; intersecter deux
telles. ($\supseteq$) Tout \omterm{def:b3:topology:topology}{ouvert} de $\R^{m+n}$ est une union
dénombrable de boîtes \omterm{def:b3:topology:topology}{ouvertes} rationnelles $U\times V$~:
\omterm{def:b3:rings:content}{contenu} dans la $\sigma$-algèbre produit.
\end{proof}

\begin{theorem}[Mesure produit]\label{thm:b3:product:existence}
Soient $(X, \mathcal A, \mu)$ et $(Y, \mathcal B, \nu)$
\emph{$\sigma$-finis}. Pour tout $E \in \mathcal
A\otimes\mathcal B$, la fonction $x \mapsto \nu(E_x)$ est
\omterm{def:b3:lebesgue:measurable}{mesurable}, et\index{mesure produit}
\[
(\mu\otimes\nu)(E) = \int_X \nu(E_x)\,\dd\mu(x)
\]
définit l'unique \omterm{def:b3:measure:measure}{mesure} sur $\mathcal A\otimes\mathcal B$ avec
$(\mu\otimes\nu)(A\times B) = \mu(A)\nu(B)$. Elle est
$\sigma$-finie, et symétrique~: la même \omterm{def:b3:measure:measure}{mesure} s'obtient en
intégrant les $x$-sections contre $\nu$.
\end{theorem}

\begin{proof}
\emph{Mesurabilité de $x \mapsto \nu(E_x)$.} D'abord $\nu$
finie. La classe $\mathcal D$ des $E$ pour lesquels l'application
est \omterm{def:b3:lebesgue:measurable}{mesurable} contient les rectangles ($\nu((A\times B)_x) =
\nu(B)\mathbf 1_A(x)$) et est un $\lambda$-système~: pour $E
\subseteq F$ dans $\mathcal D$, $\nu((F\setminus E)_x) =
\nu(F_x) - \nu(E_x)$ (finitude)~; pour $E_n \uparrow E$,
$\nu((E_n)_x) \uparrow \nu(E_x)$ (continuité par en bas), et
les limites monotones de \omterm{def:b3:lebesgue:measurable}{fonctions mesurables} sont \omterm{def:b3:lebesgue:measurable}{mesurables}.
Les rectangles forment un $\pi$-système~: Dynkin
(\cref{thm:b3:measure:dynkin}) donne $\mathcal D = \mathcal
A\otimes\mathcal B$. Si $\nu$ est $\sigma$-finie, écrire $Y =
\bigcup Y_k$, $Y_k \uparrow$, $\nu(Y_k) < \infty$~:
$\nu(E_x) = \lim_k\nu_k(E_x)$ avec $\nu_k = \nu(\cdot\cap Y_k)$
finie.

\emph{\omterm{def:b3:measure:measure}{Mesure}.} La $\sigma$-additivité de $E \mapsto
\int\nu(E_x)\dd\mu$ suit du~\cref{cor:b3:lebesgue:additivity}
(les sections d'ensembles disjoints sont disjointes). Sur les
rectangles elle donne $\mu(A)\nu(B)$. \emph{Unicité}~: deux
candidates coïncident sur le $\pi$-système des rectangles~; la
$\sigma$-finitude fournit des rectangles $X_k\times Y_k
\uparrow X\times Y$ de \omterm{def:b3:measure:measure}{mesure} finie~:
\cref{thm:b3:measure:uniqueness}. Symétrie~: la construction
dans l'autre ordre est aussi une \omterm{def:b3:measure:measure}{mesure} coïncidant sur les
rectangles --- unique, donc la même.
\end{proof}

\begin{definition}\label{def:b3:product:lebesgue}
La \emph{mesure de Lebesgue sur $\R^d$}\index{mesure de Lebesgue
sur $\R^d$} est $\lambda_d = \lambda\otimes\cdots\otimes\lambda$
($d$ facteurs~; l'associativité de la construction se vérifie
sur les boîtes et se propage par unicité). C'est l'unique \omterm{def:b3:measure:measure}{mesure}
borélienne donnant à chaque boîte $\prod\intoc{a_i}{b_i}$ son
volume $\prod(b_i - a_i)$~; elle est invariante par translation
(les translatés coïncident sur les boîtes), $\sigma$-finie, et
complète après complétion de Carathéodory --- on note
$\lambda_d$ la \omterm{def:b3:measure:measure}{mesure} complétée et on intègre en conséquence.
\end{definition}

\section{Tonelli et Fubini}

\begin{theorem}[Tonelli]\label{thm:b3:product:tonelli}
$\mu, \nu$ $\sigma$-finis, $f \colon X\times Y \to [0,
+\infty]$ \omterm{def:b3:lebesgue:measurable}{mesurable}. Alors $x \mapsto \int_Y f_x\,\dd\nu$ est
\omterm{def:b3:lebesgue:measurable}{mesurable} et\index{théorème de Tonelli}
\[
\int_{X\times Y}f\,\dd(\mu\otimes\nu)
= \int_X\Bigl(\int_Y f(x,y)\,\dd\nu(y)\Bigr)\dd\mu(x)
= \int_Y\Bigl(\int_X f(x,y)\,\dd\mu(x)\Bigr)\dd\nu(y).
\]
\end{theorem}

\begin{proof}
La machine standard. Pour $f = \mathbf 1_E$ c'est
le~\cref{thm:b3:product:existence} (et sa forme symétrique).
Par linéarité cela tient pour $f \geq 0$ \omterm{def:b3:groups:simple}{simple}. Pour $f \geq
0$ générale~: prendre des \omterm{def:b3:groups:simple}{simples} $s_n \nearrow f$
(\cref{thm:b3:lebesgue:approximation})~; alors
$\int_Y(s_n)_x \dd\nu \nearrow \int_Y f_x\dd\nu$ pour chaque
$x$ (TCM en $Y$), donc les membres de gauche convergent par
TCM en $X$, tandis que $\int s_n\,\dd(\mu\otimes\nu)
\nearrow \int f$ par TCM sur le produit.
\end{proof}

\begin{theorem}[Fubini]\label{thm:b3:product:fubini}
$\mu, \nu$ $\sigma$-finis, $f \in L^1(\mu\otimes\nu)$. Alors
pour $\mu$-p.t.\ $x$ la section $f_x$ est $\nu$-intégrable, la
fonction définie p.p.\ $x \mapsto \int f_x\dd\nu$ est
\omterm{def:b3:lebesgue:l1}{intégrable}, et les deux intégrales itérées valent toutes deux
$\int f\,\dd(\mu\otimes\nu)$.\index{théorème de Fubini}
\end{theorem}

\begin{proof}
Tonelli appliqué à $\abs f$ montre que $\varphi(x) =
\int\abs{f_x} \dd\nu$ a une intégrale finie, donc est finie
p.p.~: $f_x \in L^1(\nu)$ pour p.t.\ $x$. Scinder $f = f^+ -
f^-$ (cas réel~; complexe par composantes)~: Tonelli calcule
chaque intégrale itérée de $f^\pm$ comme $\int
f^\pm\dd(\mu\otimes\nu) < \infty$, et la différence définie
p.p.\ s'intègre en la différence. Symétriquement pour l'autre
ordre.
\end{proof}

\begin{method}\label{met:b3:product:fubini}
Pour intervertir deux intégrales (ou une intégrale et une
somme, ou deux sommes)~: si l'intégrande est
\emph{positive}, intervertir librement (Tonelli). Sinon,
appliquer d'abord Tonelli à $\abs f$ dans l'ordre le plus
facile à estimer~; si le résultat est fini, Fubini légitime
l'interversion. Ne jamais sauter le contrôle de $\abs f$~:
l'intégrande de l'\cref{exo:b3:product:4} a deux intégrales
itérées de \emph{valeurs différentes}.
\end{method}

\begin{proposition}[Formule des couches]\label{prop:b3:product:layercake}
Pour $f \geq 0$ \omterm{def:b3:lebesgue:measurable}{mesurable} sur $(X, \mathcal A, \mu)$
$\sigma$-fini~:\index{formule des couches}
\[
\int_X f\,\dd\mu = \int_0^{+\infty}\mu(\{f > t\})\,\dd t,
\qquad
\int_X f^p\,\dd\mu = p\int_0^{+\infty}t^{p-1}\mu(\{f >
t\})\,\dd t \quad (p \geq 1).
\]
\end{proposition}

\begin{proof}
Appliquer Tonelli à $\mathbf 1_{\{(x,t) : 0 < t < f(x)\}}$
sur $X \times \intoo0{+\infty}$ (\omterm{def:b3:lebesgue:measurable}{mesurable}~: c'est
$\{(x,t): f(x) - t > 0\}\cap\{t > 0\}$, une combinaison de
type borélien de la $(x,t)\mapsto f(x) - t$ \omterm{def:b3:lebesgue:measurable}{mesurable})~: en
intégrant d'abord en $t$ on obtient $\int f\,\dd\mu$~; d'abord
en $x$, $\int_0^\infty\mu(f > t)\dd t$. Pour $f^p$~:
substituer $t = s^p$ dans $\int\mu(f^p > t) \dd t$, c.-à-d.\
appliquer la première formule à $f^p$ et changer de variable
dans l'intégrale unidimensionnelle ($\{f^p > s^p\} = \{f >
s\}$).
\end{proof}

\begin{theorem}[Convolution sur $L^1$]\label{thm:b3:product:convolution}
Pour $f, g \in L^1(\R^d, \lambda_d)$, l'intégrale
\[
(f * g)(x) = \int_{\R^d} f(x - y)\,g(y)\,\dd y
\]
converge absolument pour p.t.\ $x$, définit $f * g \in
L^1(\R^d)$ avec $\norm{f*g}_1 \leq \norm f_1\norm g_1$, et
$*$ est commutative et associative.\index{convolution}
\end{theorem}

\begin{proof}
$(x, y) \mapsto f(x-y)g(y)$ est \omterm{def:b3:lebesgue:measurable}{mesurable} ($(x,y)\mapsto x -
y$ est \omterm{def:b3:topology:continuity}{continue}~; composer et multiplier). Tonelli~:
\[
\int\!\!\int \abs{f(x-y)}\abs{g(y)}\,\dd y\,\dd x
= \int\abs{g(y)}\Bigl(\int\abs{f(x - y)}\dd x\Bigr)\dd y
= \norm f_1\norm g_1 < \infty
\]
(invariance par translation de $\lambda_d$ dans l'intégrale
intérieure). Donc l'intégrale double est finie~; Fubini donne
la convergence absolue p.p.\ et la borne de norme
$\norm{f*g}_1 \leq \norm f_1\norm g_1$. Commutativité~:
substituer $y \mapsto x - y$ (invariance par translation et
réflexion --- l'invariance par réflexion tient sur les
boîtes, donc partout par unicité). Associativité~:
Tonelli--Fubini sur une intégrale triple.
\end{proof}

\section{Changement de variables}

\begin{theorem}[Changement de variables linéaire]
\label{thm:b3:product:linearchange}
Pour $T \in GL_d(\R)$ et $A \in \mathcal B(\R^d)$~:
$\lambda_d(T(A)) = \abs{\det T}\,\lambda_d(A)$~; par
conséquent $\int f(y)\dd y = \abs{\det T}\int f(Tx)\,\dd x$
pour $f \geq 0$ ou \omterm{def:b3:lebesgue:l1}{intégrable}.\index{changement de variables
linéaire}
\end{theorem}

\begin{proof}
La \omterm{def:b3:measure:measure}{mesure} $\mu_T(A) = \lambda_d(T(A))$ est une \omterm{def:b3:measure:measure}{mesure}
borélienne (les \omterm{def:b3:topology:continuity}{homéomorphismes} préservent les boréliens,
\cref{pb:b3:measure:1}), invariante par translation
($T(A + x) = T(A) + Tx$), finie sur la boîte unité~: par la
caractérisation de la \omterm{def:b3:measure:lebesgueouter}{mesure de Lebesgue}
(\cref{exo:b3:measure:6}, dont la preuve fonctionne
mot pour mot en $\R^d$ avec des cubes dyadiques), $\mu_T =
c(T)\lambda_d$ avec $c(T) = \lambda_d(T(\intco01^d))$.
L'application $T \mapsto c(T)$ est multiplicative ($c(ST) =
c(S)c(T)$, par composition), donc il suffit de calculer $c$
sur des générateurs de $GL_d$~: les matrices élémentaires.
Diagonale $\operatorname{diag}(a, 1, \dots, 1)$~: envoie le
cube unité sur une boîte de volume $\abs a$~: $c = \abs a =
\abs\det$. Transposition de coordonnées~: permute le cube~:
$c = 1 = \abs\det$. Transvection $T(x) = x + \alpha
x_2e_1$~: l'image du cube unité est un prisme cisaillé~; par
Tonelli sa \omterm{def:b3:measure:measure}{mesure} est $\int\lambda_1(\text{section})\dd x_2
\cdots \dd x_d = 1$, chaque $x_1$-section étant un intervalle
de longueur $1$~: $c = 1 = \abs{\det}$. Toute matrice
inversible est un produit de celles-ci (élimination de
Gauss), et $c$ comme $\abs\det$ sont multiplicatives~:
$c(T) = \abs{\det T}$. La formule d'intégrale suit par la
machine standard (indicateurs, \omterm{def:b3:groups:simple}{simples}, TCM).
\end{proof}

\begin{theorem}[Changement de variables]\label{thm:b3:product:changeofvar}
Soient $U, V \subseteq \R^d$ \omterm{def:b3:topology:topology}{ouverts} et $\Phi \colon U \to V$
un difféomorphisme $\mathcal C^1$. Pour toute $f \colon V
\to [0, +\infty]$ \omterm{def:b3:lebesgue:measurable}{mesurable} (ou $f \in L^1(V)$)~:\index{formule
de changement de variables}
\[
\int_V f(y)\,\dd y = \int_U
f\bigl(\Phi(x)\bigr)\,\abs{\det D\Phi(x)}\,\dd x .
\]
\end{theorem}

\begin{proof}
Écrire $J(x) = \abs{\det D\Phi(x)}$. Le cœur de la preuve est
l'inégalité
\[
\lambda_d\bigl(\Phi(A)\bigr) \leq \int_A J\,\dd\lambda_d
\qquad\text{pour tout borélien } A \subseteq U;
\tag{$*$}
\]
l'étape 4 ci-dessous promeut $(*)$ --- appliquée à $\Phi$ et
à $\Phi^{-1}$ --- en l'égalité du théorème. Noter que
$\Phi^{-1}$ est lui-même un difféomorphisme $\mathcal C^1$
avec jacobien $\abs{\det D\Phi^{-1}(y)} = J(\Phi^{-1}y)^{-1}$
(règle de la chaîne sur $\Phi\circ\Phi^{-1} = \mathrm{id}$).

\emph{Étape 1~: $(*)$ pour des cubes avec un facteur de
distorsion.} Fixer un cube fermé $Q \subseteq U$ de \omterm{ex:b3:groups:actions}{centre}
$x_0$ et de côté $2r$ (boule pour la norme sup). Affirmation~:
pour tout $\varepsilon > 0$, si $\Phi$ est dérivable sur $Q$
avec $\norm{D\Phi(x) - D\Phi(x_0)} \leq \varepsilon$ sur $Q$
(\omterm{def:b3:banach:operator}{norme d'opérateur} pour la norme sup), alors
\[
\Phi(Q) \subseteq \Phi(x_0) + D\Phi(x_0)\Bigl(\,\bigl(1 +
\varepsilon\norm{D\Phi(x_0)^{-1}}\bigr)\,(Q - x_0)\Bigr),
\]
car pour $x \in Q$, l'inégalité des accroissements finis
appliquée à $\Phi(x) - \Phi(x_0) - D\Phi(x_0)(x - x_0)$ donne
$\norm{\Phi(x) - \Phi(x_0) - D\Phi(x_0)(x-x_0)}_\infty \leq
\varepsilon\norm{x - x_0}_\infty \leq \varepsilon r$, et
$D\Phi(x_0)^{-1}$ tire ce défaut en un agrandissement de
$\varepsilon\norm{D\Phi(x_0)^{-1}}\,r$ du cube. Par
le~\cref{thm:b3:product:linearchange},
\[
\lambda_d(\Phi(Q)) \leq \abs{\det D\Phi(x_0)}\,
\bigl(1 + \varepsilon\,C\bigr)^{d}\,\lambda_d(Q),
\qquad C = \sup_{Q}\norm{D\Phi(\cdot)^{-1}} .
\]

\emph{Étape 2~: $(*)$ pour des cubes compacts, par
subdivision.} Soit $Q \subseteq U$ un cube compact et
$\varepsilon > 0$. Sur $Q$, $D\Phi$ est uniformément \omterm{def:b3:topology:continuity}{continue}
et $\norm{D\Phi^{-1}}$ bornée (compacité)~; subdiviser $Q$
en $2^{kd}$ sous-cubes $Q_i$ assez petits pour que
l'oscillation de $D\Phi$ sur chacun soit $\leq \varepsilon$.
Étape 1 sur chaque sous-cube (\omterm{ex:b3:groups:actions}{centre} $x_i$)~:
\[
\lambda_d(\Phi(Q)) \leq \sum_i\lambda_d(\Phi(Q_i))
\leq (1 + C\varepsilon)^d \sum_i \abs{\det
D\Phi(x_i)}\,\lambda_d(Q_i)
\leq (1 + C\varepsilon)^d\Bigl(\int_Q J + \varepsilon'\Bigr),
\]
la dernière étape car $\sum_i\abs{\det D\Phi(x_i)}\mathbf
1_{Q_i} \to J$ uniformément sur $Q$ (continuité de $\det
D\Phi$) --- comparaison avec sommes de Riemann. Faire
$\varepsilon \to 0$~: $(*)$ tient pour les cubes compacts.

\emph{Étape 3~: $(*)$ pour tout borélien $A$.} La fonction
d'ensembles $A \mapsto \lambda_d(\Phi(A))$, sur les
sous-ensembles boréliens de $U$, est une \omterm{def:b3:measure:measure}{mesure} ($\Phi$ est
une bijection sur $V$ préservant les boréliens et la
disjonction dénombrable), et de même $A \mapsto \int_AJ$.
Tout \omterm{def:b3:topology:topology}{ouvert} de $U$ est une union dénombrable de cubes
compacts dyadiques presque disjoints (décomposition dyadique
standard~: prendre les cubes dyadiques maximaux \omterm{def:b3:rings:content}{contenus} dans
l'\omterm{def:b3:topology:topology}{ouvert}), et les deux \omterm{def:b3:measure:measure}{mesures} sont additives à travers eux
(les bords des cubes sont $\lambda_d$-nuls, et leurs
$\Phi$-images sont nuls par l'étape 2 appliquée à des
recouvrements minces des faces)~: $(*)$ passe des cubes aux
\omterm{def:b3:topology:topology}{ouverts}. Borélien général $A$~: épuiser $U$ par des compacts
$K_m \uparrow U$ avec $K_m \subseteq \mathring K_{m+1}$, et
fixer $m$~; sur $\mathring K_{m+1}$, $J$ est bornée par un
certain $M_m$. Par régularité extérieure de $\lambda_d$
(preuve comme au~\cref{thm:b3:measure:regularity}, avec des
boîtes), choisir des \omterm{def:b3:topology:topology}{ouverts} $O_n$ avec $A \cap K_m \subseteq
O_n \subseteq \mathring K_{m+1}$ et $\lambda_d\bigl(O_n
\setminus (A\cap K_m)\bigr) \to 0$. Alors
\[
\lambda_d\bigl(\Phi(A\cap K_m)\bigr) \leq
\lambda_d\bigl(\Phi(O_n)\bigr) \leq \int_{O_n}J
\leq \int_{A\cap K_m}J + M_m\,\lambda_d\bigl(O_n\setminus(A\cap
K_m)\bigr) \xrightarrow[n\to\infty]{} \int_{A\cap K_m}J .
\]
Faire $m \to \infty$~: continuité par en bas à gauche, TCM à
droite. Cela établit $(*)$.

\emph{Étape 4~: égalité et formule d'intégrale.} D'abord
étendre $(*)$ des ensembles aux intégrales~: pour toute $g
\geq 0$ \omterm{def:b3:lebesgue:measurable}{mesurable} sur $V$,
\[
\int_V g(y)\,\dd y \leq \int_U g(\Phi(x))\,J(x)\,\dd x .
\tag{$**$}
\]
En effet, pour $g = \mathbf 1_B$ c'est $(*)$ avec $A =
\Phi^{-1}(B)$~; la linéarité l'étend aux $g$ \omterm{def:b3:groups:simple}{simples}, et le
TCM à toutes les $g \geq 0$ (la machine standard). Maintenant
appliquer $(**)$ deux fois~: d'abord à $g$, puis --- pour le
difféomorphisme $\Phi^{-1}$ --- à la fonction $x \mapsto
g(\Phi(x))J(x)$~:
\[
\int_Vg \leq \int_U g(\Phi(x))J(x)\dd x
\leq \int_V g(y)\,J(\Phi^{-1}y)\,\abs{\det
D\Phi^{-1}(y)}\,\dd y = \int_V g ,
\]
puisque $J(\Phi^{-1}y)\abs{\det D\Phi^{-1}(y)} = \abs{\det\bigl(
D\Phi(\Phi^{-1}y)\,D\Phi^{-1}(y)\bigr)} = 1$ (règle de la
chaîne sur $\Phi\circ\Phi^{-1} = \mathrm{id}$). Toutes les
inégalités sont des égalités~: la formule tient pour $g \geq
0$, et pour les fonctions $L^1$ par décomposition.
\end{proof}

\begin{example}[Coordonnées polaires~; la gaussienne
encore]\label{ex:b3:product:polar}
$\Phi(r, \theta) = (r\cos\theta, r\sin\theta)$ est un
difféomorphisme $\mathcal C^1$ de
$\intoo0{+\infty}\times\intoo0{2\pi}$ sur $\R^2$ privé d'une
demi-droite (ensemble nul), avec $\det D\Phi = r$~:\index{coordonnées
polaires}
\[
\int_{\R^2}f(x, y)\,\dd x\,\dd y
= \int_0^{2\pi}\!\!\int_0^{+\infty}
f(r\cos\theta, r\sin\theta)\,r\,\dd r\,\dd\theta .
\]
Pour $f = \eu^{-x^2-y^2}$, Tonelli et cette formule donnent
\[
G^2 = \Bigl(\int_\R \eu^{-x^2}\dd x\Bigr)^2
= \int_{\R^2}\eu^{-x^2-y^2}
= 2\pi\int_0^\infty r\eu^{-r^2}\dd r = \pi:
\]
la preuve classique en deux lignes de $G = \sqrt\pi$,
maintenant pleinement justifiée (comparer la preuve par
paramètre du~\cref{pb:b3:lebesgue:1}).
\end{example}

\begin{theorem}[Volume de la boule unité]
\label{thm:b3:product:ballvolume}
Soit $v_d = \lambda_d(B(0,1))$ dans $\R^d$. Alors\index{volume
de la boule unité}
\[
v_d = \frac{\pi^{d/2}}{\Gamma\bigl(\frac d2 + 1\bigr)} :
\qquad
v_1 = 2,\quad v_2 = \pi,\quad v_3 = \tfrac{4\pi}3,\quad
v_4 = \tfrac{\pi^2}2,\ \dots
\]
\end{theorem}

\begin{proof}
Calculer $I = \int_{\R^d}\eu^{-\norm x_2^2}\dd\lambda_d$ de
deux façons. Par Tonelli elle se factorise~: $I = G^d =
\pi^{d/2}$. Par la formule des couches
(\cref{prop:b3:product:layercake}) avec $f = \eu^{-
\norm x^2}$, dont les ensembles de niveau sont des boules~:
$\{f > t\} = B\bigl(0, \sqrt{-\ln t}\bigr)$ pour $0 < t < 1$,
de \omterm{def:b3:measure:measure}{mesure} $v_d(-\ln t)^{d/2}$ (la dilatation par $\rho$
échelle $\lambda_d$ par $\rho^d$~:
\cref{thm:b3:product:linearchange}), donc
\[
I = \int_0^1 v_d\,(-\ln t)^{d/2}\,\dd t
\overset{t = \eu^{-s}}{=}
v_d\int_0^\infty s^{d/2}\eu^{-s}\,\dd s
= v_d\,\Gamma\Bigl(\frac d2 + 1\Bigr).
\]
Égaler. (Les valeurs~: $\Gamma(\frac32) = \frac{\sqrt\pi}2$,
$\Gamma(2) = 1$, etc.) Noter $v_d \to 0$ lorsque $d \to
\infty$ --- le problème du week-end quantifie à quelle
vitesse, via Stirling.
\end{proof}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:b3:product:1}
Soit $\mu$ la \omterm{def:b3:measure:measure}{mesure} de comptage sur $(\intcc01, \mathcal
B(\intcc01))$ (non $\sigma$-finie) et $\lambda$ la \omterm{def:b3:measure:lebesgueouter}{mesure de
Lebesgue}, et soit $\Delta = \{(x,x)\}$ la diagonale dans
$\intcc01^2$. Montrer que $\Delta$ est \omterm{def:b3:lebesgue:measurable}{mesurable}, et calculer
les deux intégrales itérées de $\mathbf 1_\Delta$ contre
$\lambda$ et $\mu$~: elles diffèrent. Quelle hypothèse
du~\cref{thm:b3:product:tonelli} échoue~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b3:product:2}
Justifier l'interversion et retrouver l'intégrale de
Dirichlet~: pour $A > 0$,
\[
\int_0^A\frac{\sin x}x\,\dd x
= \int_0^A\!\!\int_0^{+\infty}\eu^{-xy}\sin x\,\dd y\,\dd x
= \int_0^{+\infty}\!\!\int_0^A \eu^{-xy}\sin x\,\dd x\,\dd y,
\]
calculer l'intégrale intérieure sous forme fermée, et faire
$A \to +\infty$ (dominer l'intégrale en $y$) pour obtenir
$\int_0^\infty\frac{\sin x}x\dd x = \frac\pi2$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:product:3}
(a) Prouver que pour $f \geq 0$ \omterm{def:b3:lebesgue:measurable}{mesurable} et $\mu$ finie~:
$\sum_{n\geq1}\mu(\{f \geq n\}) \leq \int f\,\dd\mu \leq
\mu(X) + \sum_{n\geq1}\mu(\{f\geq n\})$~: l'intégrabilité est
la sommabilité des \omterm{def:b3:measure:measure}{mesures} de queue.
(b) En déduire que $f \in L^1(\mu)$ ($\mu$ finie) ssi
$\sum_n\mu(\abs f \geq n) < \infty$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:product:4}
Pour $f(x, y) = \dfrac{x^2 - y^2}{(x^2 + y^2)^2}$ sur

\int_0^1\!\!\int_0^1 f\,\dd y\,\dd x = \frac\pi4,
\qquad
\int_0^1\!\!\int_0^1 f\,\dd x\,\dd y = -\frac\pi4
\]
\emph{(noter $f = \partial_y\bigl(\frac{y}{x^2+y^2}\bigr)$)},
et vérifier directement que $\int\!\int\abs f = +\infty$~:
l'hypothèse d'intégrabilité de Fubini n'est pas décorative.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:product:5}
(a) Calculer $\mathbf 1_{\intcc01} * \mathbf 1_{\intcc01}$

(b) Montrer $\operatorname{supp}(f * g) \subseteq

(c) Montrer que si $f \in L^1$ et $g$ est bornée et \omterm{def:b3:topology:continuity}{continue},
$f * g$ est \omterm{def:b3:topology:continuity}{continue}. \emph{(TCD via continuité de la
translation sur le $g$ borné.)}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:product:6}
(a) Montrer que le simplexe $\Delta_d = \{x \in

(b) Retrouver $v_2 = \pi$, $v_3 = \frac{4\pi}3$

\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:product:7}
Pour quels $s > 0$ les suivantes sont-elles finies~? Justifier
avec les \omterm{ex:b3:product:polar}{coordonnées polaires}~:

\int_{B(0,1)\subseteq\R^2}\frac{\dd x\,\dd y}{(x^2 +

\qquad
\int_{\R^2\setminus B(0,1)}\frac{\dd x\,\dd y}{(x^2 +

Généraliser à $\R^d$ (les seuils $s < d/2$ et $s > d/2$).
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b3:product:8}
(Bêta--Gamma) Pour $p, q > 0$, soit $B(p, q) =
\int_0^1t^{p-1}(1 - t)^{q-1}\dd t$. Partant de
$\Gamma(p)\Gamma(q)$ comme intégrale double, substituer $(x,
y) = (uv,\, u(1 - v))$ (un difféomorphisme du quadrant \omterm{def:b3:topology:topology}{ouvert}
sur $\intoo0\infty\times\intoo01$~; calculer son jacobien $=
u$) et conclure\index{fonction bêta}
\[
B(p, q) = \frac{\Gamma(p)\,\Gamma(q)}{\Gamma(p + q)} .
\]
En déduire $\int_0^{\pi/2}\sin^{2p-1}\theta\cos^{2q-1}\theta\,
\dd\theta = \frac12B(p,q)$ et la valeur des intégrales de
Wallis $W_n = \int_0^{\pi/2}\sin^n$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:product:9}
(Formule de transfert) Soit $T \colon (X, \mathcal A, \mu)
\to (Y, \mathcal B)$ \omterm{def:b3:lebesgue:measurable}{mesurable} et $T_*\mu(B) =
\mu(T^{-1}(B))$ la \emph{mesure image}\index{mesure image}
(pushforward). Montrer que pour toute $g \geq 0$ \omterm{def:b3:lebesgue:measurable}{mesurable}
sur $Y$~:
\[
\int_Y g\,\dd(T_*\mu) = \int_X g\circ T\,\dd\mu
\]
(machine standard). Puis comparer
avec~\cref{thm:b3:product:linearchange}~: quelle information
supplémentaire la formule de changement de variables porte-t-elle
que la formule de transfert abstraite ne porte pas~?
\emph{(La formule de transfert n'identifie jamais $T_*\mu$~;
le théorème de changement de variables calcule $T_*\lambda_d$
explicitement comme une \omterm{def:b3:measure:measure}{mesure} à \omterm{ex:b3:lebesgue:gamma}{densité}.)}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b3:product:10}
(Moments gaussiens) En utilisant les \omterm{ex:b3:product:polar}{coordonnées polaires} et
Fubini, calculer pour le poids gaussien standard sur $\R^d$~:
\[
\int_{\R^d}\norm x_2^2\;\eu^{-\norm x_2^2}\,\dd x
\qquad\text{et}\qquad
\int_{\R^d}x_1^2\,\eu^{-\norm x^2_2}\,\dd x,
\]
vérifier la cohérence ($\norm x^2 = \sum x_i^2$), et en
déduire le second moment de la \omterm{def:b3:measure:measure}{mesure}
$\pi^{-d/2}\eu^{-\norm x^2}\dd x$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:product:11}
(Graphe et hypographe) Soit $f \colon \R^d \to \intco0\infty$
\omterm{def:b3:lebesgue:measurable}{mesurable}.
(a) Montrer que l'\emph{hypographe} $H = \{(x, y) \in
\R^d\times\R : 0 < y < f(x)\}$ est \omterm{def:b3:lebesgue:measurable}{mesurable} dans
$\R^{d+1}$ avec
\[
\lambda_{d+1}(H) = \int_{\R^d}f\,\dd\lambda_d :
\]
«~l'intégrale est l'aire sous le graphe~», enfin un théorème.
\emph{(Sections~; Tonelli.)}
(b) Montrer que le graphe $\{(x, f(x)) : x \in \R^d\}$ est un
ensemble nul de $\R^{d+1}$.
(c) En déduire une preuve en deux lignes que la sphère
$S^{d-1}$ est Lebesgue-nulle dans $\R^d$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b3:product:12}
(Une double intégrale célèbre) En utilisant la série
géométrique et Tonelli sur $\intoo01^2$, prouver
\[
\int_0^1\!\!\int_0^1\frac{\dd x\,\dd y}{1 - xy}
= \sum_{n\geq1}\frac1{n^2} = \zeta(2),
\qquad
\int_0^1\!\!\int_0^1\frac{\dd x\,\dd y}{1 + xy}
= \sum_{n\geq1}\frac{(-1)^{n-1}}{n^2} = \frac{\zeta(2)}2 .
\]
(La seconde identité de série~: scinder indices pairs et
impairs.) Avec $\zeta(2) = \frac{\pi^2}6$
(\cref{ch:b3:spectral}), deux intégrales d'apparence
innocente valent $\frac{\pi^2}6$ et $\frac{\pi^2}{12}$~; où
exactement l'hypothèse de positivité de Tonelli fait-elle son
travail~?
\end{exercise}

\section{Problème~: la formule de Stirling}

\begin{problem}[{Problème du week-end --- $n! \sim
\sqrt{2\pi n}\,(n/\eu)^n$, par convergence dominée}]
\label{pb:b3:product:1}
La formule de Stirling gouverne tout décompte asymptotique de
ce livre --- volumes de boules, coefficients binomiaux, la
forme locale du théorème central limite. On la prouve à
partir de l'intégrale $\Gamma$
(\cref{ex:b3:lebesgue:gamma}) avec la \emph{méthode de
Laplace}, sous sa forme de convergence dominée la plus
propre, puis on collecte les dividendes.\index{formule de
Stirling}

\medskip
\noindent\textbf{Partie I --- La formule.} Pour $t > 0$,
$\Gamma(t + 1) = \int_0^\infty x^{t}\eu^{-x}\dd x$.
\begin{enumerate}
  \item Substituer $x = t + \sqrt t\,u$ et montrer
        \[
        \frac{\Gamma(t+1)}{t^{t}\eu^{-t}\sqrt t}
        = \int_{-\sqrt t}^{+\infty}
        \exp\Bigl(t\ln\Bigl(1 + \frac u{\sqrt
        t}\Bigr) - \sqrt t\,u\Bigr)\,\dd u
        \;=\;\int_\R g_t(u)\,\dd u,
        \]
        où $g_t(u) = \exp\bigl(t\ln(1 + u/\sqrt t) -
        \sqrt t\,u\bigr)\mathbf 1_{u > -\sqrt t}$.
  \item Montrer la limite ponctuelle~: pour tout $u$ fixé,
        $g_t(u) \to \eu^{-u^2/2}$ lorsque $t \to +\infty$
        \emph{(développer $\ln(1 + h)$ à l'ordre deux)}.
  \item Domination. Soit $\varphi(h) = \ln(1 + h) - h$, de
        sorte que $g_t(u) = \exp\bigl(t\,\varphi(u/\sqrt
        t)\bigr)$ pour $u > -\sqrt t$. Prouver les deux
        bornes
        \[
        \varphi(h) \leq -\frac{h^2}4 \quad (-1 < h \leq 1),
        \qquad
        \varphi(h) \leq -c\,h \quad (h \geq 1),\ \ c = 1 -
        \ln 2 > 0
        \]
        \emph{(étudier $\varphi(h) + \frac{h^2}4$ et
        $\varphi(h) + ch$~: calculer les dérivées et
        vérifier le signe sur chaque intervalle)}. En
        déduire, pour $t \geq 1$~:
        \[
        g_t(u) \leq \eu^{-u^2/4}\ \ (\abs u \leq \sqrt t),
        \qquad
        g_t(u) \leq \eu^{-cu}\ \ (u \geq \sqrt t),
        \]
        de sorte que $g_t(u) \leq \eu^{-u^2/4} +
        \eu^{-cu}\,\mathbf 1_{u > 0}$~: un dominateur
        \omterm{def:b3:lebesgue:l1}{intégrable} indépendant de $t \geq 1$.
  \item Conclure avec le théorème de convergence dominée et
        l'intégrale gaussienne (\cref{ex:b3:product:polar})~:
        \[
        \Gamma(t + 1) \;\sim\;
        \sqrt{2\pi t}\;\Bigl(\frac t\eu\Bigr)^{t}
        \qquad (t \to +\infty),
        \]
        et en particulier $n! \sim \sqrt{2\pi
        n}\,(n/\eu)^n$.
\end{enumerate}

\medskip
\noindent\textbf{Partie II --- Dividendes.}
\begin{enumerate}[resume]
  \item (Wallis) De l'\cref{exo:b3:product:8}, $W_{2n} =
        \frac\pi2\binom{2n}n4^{-n}$-type formules~: dériver
        $\binom{2n}{n} \sim \frac{4^n}{\sqrt{\pi n}}$ de
        Stirling, et le vérifier contre la récurrence $W_{n}
        = \frac{n-1}nW_{n-2}$.
  \item (Les volumes de boules s'effondrent) Montrer
        \[
        v_d = \frac{\pi^{d/2}}{\Gamma(\frac d2 + 1)}
        \;\sim\; \frac{1}{\sqrt{\pi d}}
        \Bigl(\frac{2\pi\eu}{d}\Bigr)^{d/2},
        \]
        donc $v_d \to 0$ plus vite que toute suite
        géométrique~; trouver la dimension maximisant $v_d$
        (numériquement~: $d = 5$).
  \item (Concentration du binomial --- un avant-goût
        du~\cref{ch:b3:clt}) En utilisant Stirling, montrer
        l'estimation locale, pour $k = n/2 + s\sqrt n/2$
        avec $s$ fixé et $n$ pair~:
        \[
        2^{-n}\binom{n}{k} \;\sim\;
        \sqrt{\frac{2}{\pi n}}\;\eu^{-s^2/2},
        \]
        le profil gaussien discret~: de Moivre--Laplace en
        germe.
  \item Où exactement la preuve de la partie I a-t-elle
        utilisé~: (i) TCM ou TCD~; (ii) l'intégrale
        gaussienne~; (iii) les propriétés d'invariance de
        la \omterm{def:b3:measure:lebesgueouter}{mesure de Lebesgue}~? Une phrase chacune.
\end{enumerate}

\medskip
\noindent\textbf{Partie III --- Le terme d'erreur~: Stirling
avec barres.} Poser $d_n = \ln n! - \bigl(n +
\tfrac12\bigr)\ln n + n - \ln\sqrt{2\pi}$, de sorte que la
partie I dit $d_n \to 0$.
\begin{enumerate}[resume]
  \item Montrer $d_n - d_{n+1} = \bigl(n +
        \tfrac12\bigr)\ln\bigl(1 + \tfrac1n\bigr) - 1$.
  \item Avec $t = \frac1{2n+1}$, vérifier $\frac{n+1}n =
        \frac{1+t}{1-t}$ et développer~:
        \[
        d_n - d_{n+1} = \frac{t^2}3 + \frac{t^4}5 +
        \frac{t^6}7 + \cdots,
        \]
        et en déduire les bornes bilatères
        \[
        \frac1{3(2n+1)^2} \;<\; d_n - d_{n+1} \;<\;
        \frac1{12n} - \frac1{12(n+1)} .
        \]
  \item Télescoper (en utilisant $d_m \to 0$) et vérifier
        l'identité \omterm{def:b3:galois:algebraic}{algébrique} agréable $\frac1{3(2m+1)^2} >
        \frac1{12m+1} - \frac1{12(m+1)+1}$ pour $m \geq 1$,
        pour obtenir l'encadrement classique
        \[
        \sqrt{2\pi n}\Bigl(\frac n\eu\Bigr)^n
        \eu^{1/(12n+1)} \;<\; n! \;<\;
        \sqrt{2\pi n}\Bigl(\frac n\eu\Bigr)^n\eu^{1/(12n)} .
        \]
  \item Deux conséquences~: (a) l'erreur relative de la
        formule de Stirling est $< 10^{-6}$ dès que $n \geq
        83\,334$~; (b) estimer $100!$ à quatre chiffres
        significatifs à la main depuis l'encadrement ($100!
        \approx 9{,}3326\cdot 10^{157}$), et s'émerveiller
        brièvement de la précision d'une formule
        asymptotique à un $n$ très fini.
\end{enumerate}

\medskip
\noindent\textbf{Partie IV --- La voie de Wallis~: Stirling
sans la gaussienne.} Historiquement la constante
$\sqrt{2\pi}$ est venue de Wallis, non de Gauss~; cette
partie reprouve Stirling indépendamment des parties I--II, et
par là reprouve l'intégrale gaussienne. Soit $W_n =
\int_0^{\pi/2}\sin^n\theta\, \dd\theta$.
\begin{enumerate}[resume]
  \item Établir $W_n = \frac{n-1}nW_{n-2}$ (intégrer par
        parties), les formes fermées
        \[
        W_{2n} = \frac\pi2\binom{2n}n4^{-n}, \qquad
        W_{2n+1} = \frac{4^n}{(2n+1)\binom{2n}n},
        \]
        et l'identité $W_nW_{n-1} = \frac\pi{2n}$.
  \item De la monotonie de $(W_n)$ déduire
        $W_{2n}/W_{2n+1} \to 1$, puis
        \[
        W_{2n} \sim \frac12\sqrt{\frac\pi n}
        \qquad\text{et}\qquad
        \binom{2n}n4^{-n}\sqrt n \longrightarrow
        \frac1{\sqrt\pi} :
        \]
        le théorème de Wallis, obtenu sans Stirling.
  \item Montrer, par le seul télescopage de la partie III
        (aucune valeur de la constante nécessaire), que
        $e_n = \ln n! - (n + \frac12)\ln n + n$ converge
        vers une certaine limite $\ell$~; équivalemment $n!
        \sim K\,n^{n+1/2}\eu^{-n}$ avec $K = \eu^\ell > 0$
        pas encore identifié.
  \item Insérer cette asymptotique dans
        $\binom{2n}n4^{-n}\sqrt n$ et identifier, via la
        question 14, la seule valeur possible~: $K =
        \sqrt{2\pi}$. Assembler la logique~: les parties
        III--IV ensemble donnent une seconde preuve
        complète de Stirling --- et donc, en courant la
        substitution de la partie I à rebours, une
        évaluation indépendante de
        $\int_\R\eu^{-u^2/2}\dd u = \sqrt{2\pi}$. Deux
        piliers, dont chacun soutient l'autre.
\end{enumerate}

\medskip
\noindent\textbf{Partie V --- Derniers dividendes.}
\begin{enumerate}[resume]
  \item (Le profil local \omterm{def:b3:complete:complete}{complet}) Pour des entiers $\abs j
        \leq K\sqrt n$ ($K$ fixé), montrer
        \[
        \frac{\binom{2n}{n+j}}{\binom{2n}{n}} =
        \prod_{i=1}^{\abs j}\frac{n - i + 1}{n + i}
        = \exp\Bigl(-\frac{j^2}n +
        O\Bigl(\frac1{\sqrt n}\Bigr)\Bigr),
        \]
        uniformément en $j$ \emph{(prendre les logarithmes et
        utiliser $\ln \frac{1-x}{1+y} = -(x + y) + O(x^2 +
        y^2)$)}. C'est la version bilatère de la question 7
        et l'estimation exacte citée dans le problème du
        week-end du~\cref{ch:b3:clt}.
  \item (Un avant-goût de Poisson) Montrer avec Stirling que
        $\eu^{-n}\dfrac{n^n}{n!} \sim \dfrac1{\sqrt{2\pi
        n}}$~: le mode d'une loi de Poisson de grande
        moyenne $n$ porte une masse $\approx (2\pi
        n)^{-1/2}$, exactement comme le théorème central
        limite le prédira.
  \item (Rapports de Gamma) Pour $a \in \intoo01$, prouver
        $\dfrac{\Gamma(n + a)}{\Gamma(n)\,n^a} \to 1$ en
        utilisant les bornes de pentes log-convexes
        du~\cref{pb:b3:lebesgue:1} (question 14 là-bas), et
        étendre à tout réel $a > 0$ par l'équation
        fonctionnelle. (C'est ce que signifie «~$\Gamma(t+1)
        \sim$ Stirling~» entre les entiers.)
  \item (Boules, bis) De $v_d = \pi^{d/2}/\Gamma(\frac d2 +
        1)$~: tabuler $v_1, \dots, v_7$ exactement, vérifier
        l'unimodalité via $\frac{v_d}{v_{d-2}} =
        \frac{2\pi}d$ (croissant tant que $d < 2\pi$,
        décroissant après), et prouver l'identité
        génératrice frappante
        \[
        \sum_{k\geq0}v_{2k}\,x^{2k} = \eu^{\pi x^2} :
        \]
        tous les volumes de boules en dimensions paires
        emballés dans une seule exponentielle.
  \item (Asymptotique d'entropie) Pour $\alpha \in \intoo01$
        fixé avec $\alpha n \in \N$, déduire de Stirling
        \[
        \binom{n}{\alpha n} \;\sim\;
        \frac{\eu^{n\,H(\alpha)}}
        {\sqrt{2\pi\,\alpha(1-\alpha)\,n}},
        \qquad
        H(\alpha) = -\alpha\ln\alpha -
        (1-\alpha)\ln(1-\alpha) :
        \]
        le taux de croissance exponentielle des coefficients
        binomiaux est l'\emph{entropie} $H$ --- vérifier que
        $\alpha = \frac12$ retrouve la question 5, et que
        $H(\alpha) < \ln2$ pour $\alpha \neq \frac12$
        (donc les binomiaux hors \omterm{ex:b3:groups:actions}{centre} sont
        exponentiellement négligeables dans $2^n$).
  \item (Aires de surface) L'aire de la sphère unité
        $S^{d-1}$ est $s_{d-1} = d\,v_d$ (prouvé comme
        l'\cref{exo:b3:forms:6} dans le chapitre des formes
        différentielles~; ici, le prendre comme définition).
        Tabuler $s_0, \dots, s_6$, localiser la maximale ($d
        - 1 = 6$, $s_6 = \frac{16\pi^3}{15} \approx
        33{,}07$), et montrer $s_{d-1} \to 0$
        super-géométriquement aussi --- les sphères de
        grande dimension sont, par toute toise euclidienne,
        infiniment petites.
  \item (Le premier terme correctif) Déduire de
        l'encadrement de la question 11 que $d_n =
        \frac1{12n} + O\bigl(\frac1{n^2}\bigr)$, d'où
        \[
        n! = \sqrt{2\pi n}\,\Bigl(\frac
        n\eu\Bigr)^{n}\Bigl(1 + \frac1{12n} +
        O\Bigl(\frac1{n^2}\Bigr)\Bigr).
        \]
        Vérifier en $n = 10$~: la formule nue donne
        $3\,598\,696$ (erreur relative $8{,}3\cdot10^{-3}$),
        la corrigée $3\,628\,685$ contre $10! =
        3\,628\,800$ (erreur relative $3{,}2\cdot10^{-5}$)
        --- un terme de la série achète deux chiffres et
        demi.
  \item (La médiane de $\Gamma$) Montrer que
        \[
        \frac{1}{\Gamma(t+1)}
        \int_0^{t} x^{t}\eu^{-x}\,\dd x
        \;\longrightarrow\; \frac12
        \qquad (t \to +\infty) :
        \]
        asymptotiquement, exactement la moitié de la masse
        de l'intégrande de $\Gamma$ se situe sous son mode
        $x = t$. \emph{(Courir la substitution de la partie
        I sur l'intégrale tronquée~; le dominateur de la
        question 3 est déjà en place.)}
  \item (Entropie, non asymptotiquement) Pour $\alpha \in
        \intoc0{\frac12}$ prouver la borne, valable pour
        \emph{tout} $n \geq 1$~:
        \[
        \sum_{k=0}^{\lfloor\alpha n\rfloor}\binom nk
        \;\leq\; \eu^{n\,H(\alpha)} ,
        \]
        en comparant la somme avec $\sum_k\binom
        nk\lambda^{k-\alpha n}$ pour le basculement $\lambda
        = \frac{\alpha}{1-\alpha} \leq 1$. Vérifier que ce
        choix de $\lambda$ est optimal, et réconcilier avec
        la question 21~: le taux exponentiel $H(\alpha)$ de
        l'énoncé asymptotique est atteint par une inégalité
        en une ligne sans aucune asymptotique.
\end{enumerate}
\end{problem}
