\chapter{Complete Spaces: Baire, Ascoli, Stone--Weierstrass}
\label{ch:b3:complete}

\omterm{def:b3:complete:complete}{Completeness} --- every Cauchy sequence converges --- is the
property that lets analysis \emph{produce} objects: fixed points
of contractions, sums of series, solutions of equations obtained
as limits. This chapter assembles the three great existence
machines of the metric theory. \emph{Baire's theorem} shows that
a \omterm{def:b3:complete:complete}{complete} space cannot be a countable union of negligible
pieces, and conjures objects (\omterm{def:b3:topology:continuity}{continuous} \omterm{thm:b3:complete:nowherediff}{nowhere differentiable
functions}!) out of pure cardinality-style reasoning.
\emph{Arzel\`a--Ascoli} identifies the \omterm{def:b3:topology:compact}{compact} subsets of
$\mathcal C(K)$ and is analysis' compactness workhorse --- the
weekend problem uses it to prove Peano's existence theorem for
differential equations. \emph{Stone--Weierstrass} shows
polynomials, and much else, are dense in $\mathcal C(K)$:
approximation becomes an \omterm{def:b3:galois:algebraic}{algebraic} verification. Along the way we
construct \omterm{thm:b3:complete:completion}{completions} and prove the extension theorem for
uniformly \omterm{def:b3:topology:continuity}{continuous maps}, the daily bread of
\cref{ch:b3:lp,ch:b3:hilbert,ch:b3:fouriertransform}.

\section{Complete spaces, completions, extensions}

\begin{definition}\label{def:b3:complete:complete}
A metric space is \emph{complete}\index{complete metric space} if
every Cauchy sequence converges (Year 2: $\R^n$ is complete;
$\mathcal C(\intcc01)$ with $\norm\cdot_\infty$ is complete). A
closed subset of a complete space is complete; a complete subset
of any metric space is closed.
\end{definition}

\begin{proof}
For the two statements: a Cauchy sequence of the closed $F$
converges in $X$, and its limit, adherent to $F$, lies in $F$; a
convergent-in-$X$ sequence of a \omterm{def:b3:complete:complete}{complete} $A$ is Cauchy, so
converges in $A$, and limits are unique.
\end{proof}

\begin{theorem}[Extension of uniformly continuous maps]
\label{thm:b3:complete:extension}
Let $D \subseteq X$ be dense, $Y$ \omterm{def:b3:complete:complete}{complete}, and $f \colon D \to
Y$ uniformly \omterm{def:b3:topology:continuity}{continuous}. Then $f$ extends \emph{uniquely} to a
\omterm{def:b3:topology:continuity}{continuous} $\bar f \colon X \to Y$, and $\bar f$ is uniformly
\omterm{def:b3:topology:continuity}{continuous}.\index{extension theorem (uniform continuity)}
\end{theorem}

\begin{proof}
Uniqueness: two \omterm{def:b3:topology:continuity}{continuous} extensions agree on the dense $D$,
hence everywhere (the agreement set $\{g = h\}$ is closed:
preimage of the closed diagonal under $x\mapsto(g(x), h(x))$).
Existence: for $x \in X$ pick $d_n \to x$, $d_n \in D$. The
sequence $(f(d_n))$ is Cauchy: given $\varepsilon$, uniform
\omterm{def:b3:topology:continuity}{continuity} provides $\delta$ with $d(u,v) < \delta \Rightarrow
d(f(u), f(v)) < \varepsilon$, and $(d_n)$ is Cauchy. Define
$\bar f(x) = \lim f(d_n)$; the limit does not depend on the
chosen sequence (interlace two of them). $\bar f$ extends $f$
(constant sequences) and inherits the modulus of \omterm{def:b3:topology:continuity}{continuity}: if
$d(x, x') < \delta$, approximating both by points of $D$ at
distance $< \frac{\delta - d(x,x')}2$ gives $d(\bar f(x), \bar
f(x')) \leq \varepsilon$ in the limit --- $\bar f$ is uniformly
\omterm{def:b3:topology:continuity}{continuous}.
\end{proof}

\begin{theorem}[Completion]\label{thm:b3:complete:completion}
Every metric space $X$ embeds isometrically as a dense subset of
a \omterm{def:b3:complete:complete}{complete metric space} $\hat X$, unique up to isometry fixing
$X$ pointwise: its \emph{completion}\index{completion of a
metric space}.
\end{theorem}

\begin{proof}
\emph{Existence.} Let $\mathcal X$ be the set of Cauchy
sequences of $X$, with the pseudo-distance
\[
D\bigl((x_n), (y_n)\bigr) = \lim_n d(x_n, y_n),
\]
the limit existing because $\abs{d(x_n, y_n) - d(x_m, y_m)}
\leq d(x_n, x_m) + d(y_n, y_m)$ makes the real sequence
Cauchy. Let $\hat X = \mathcal X/{\sim}$,
identifying sequences at $D$-distance $0$; $D$ descends to a
distance. Embed $X$ by constant sequences: an isometry, with
dense image (a Cauchy sequence is $D$-approximated by the
constants built on its own terms: $D\bigl((x_n), (x_k)_{\rm
const}\bigr) = \lim_n d(x_n, x_k) \to 0$ as $k \to \infty$ by
Cauchyness). \omterm{def:b3:complete:complete}{Completeness} of $\hat X$: let $(\xi^k)$ be Cauchy
in $\hat X$; by density pick $x_k \in X$ with $D(\xi^k, x_k)
\leq 2^{-k}$; then $(x_k)$ is Cauchy in $X$ (triangle
inequality through the $\xi$'s), defines a point $\xi \in \hat
X$, and $D(\xi^k, \xi) \leq 2^{-k} + D(x_k, \xi) \to 0$ (the
distance from the constant $x_k$ to the class of $(x_j)_j$ is
$\lim_j d(x_k, x_j)$, small for large $k$).

\emph{Uniqueness}: two \omterm{thm:b3:complete:completion}{completions} $\hat X_1, \hat X_2$ contain
$X$ densely; the identity of $X$, an isometry, is uniformly
\omterm{def:b3:topology:continuity}{continuous}, so extends to $\hat X_1 \to \hat X_2$
(\cref{thm:b3:complete:extension}), still an isometry on a \omterm{def:b3:topology:interior}{dense
set} hence everywhere; symmetrically in the other direction, and
the composites fix the dense $X$: they are the identities.
\end{proof}

\begin{theorem}[Banach fixed point]\label{thm:b3:complete:banach}
Let $X$ be \omterm{def:b3:complete:complete}{complete}, nonempty, and $f \colon X \to X$ a
\emph{contraction}: $d(f(x), f(y)) \leq k\,d(x,y)$ with $k < 1$.
Then $f$ has a unique fixed point $x^*$, and every \omterm{def:b3:groups:action}{orbit}
converges to it, with the explicit rate $d(x_n, x^*) \leq
\frac{k^n}{1-k}\,d(x_1, x_0)$.\index{Banach fixed point theorem}
\end{theorem}

\begin{proof}
(Year 2 proved this; we re-record the two-line argument for
self-containedness.) The \omterm{def:b3:groups:action}{orbit} $x_{n+1} = f(x_n)$ has $d(x_{n+1},
x_n) \leq k^nd(x_1, x_0)$, hence is Cauchy (geometric series);
its limit $x^*$ is fixed (\omterm{def:b3:topology:continuity}{continuity} of $f$), unique since two
fixed points satisfy $d \leq k\,d$. The rate: sum the geometric
tail.
\end{proof}

\begin{example}[Perturbing the identity]
\label{ex:b3:complete:perturbident}
Let $g \colon \R^d \to \R^d$ be $k$-Lipschitz with $k < 1$.
Then $\varphi = \mathrm{id} + g$ is a \emph{\omterm{def:b3:topology:continuity}{homeomorphism} of
$\R^d$ onto $\R^d$}. Injectivity, with a quantitative
modulus:
\[
\norm{\varphi(x) - \varphi(y)} \geq \norm{x - y} -
\norm{g(x) - g(y)} \geq (1 - k)\norm{x - y} .
\]
Surjectivity is the fixed point theorem: solving
$\varphi(x) = y$ means $x = y - g(x)$, and $x \mapsto y -
g(x)$ is a $k$-contraction of the \omterm{def:b3:complete:complete}{complete} $\R^d$ --- a
unique solution $x = \psi(y)$ exists for every $y$. The
displayed inequality makes the inverse $\psi$ Lipschitz with
constant $\frac1{1-k}$: a \omterm{def:b3:topology:continuity}{homeomorphism}, with explicit
bounds on both moduli. This innocuous-looking statement is
the engine inside the inverse function theorem
(\cref{ch:b3:submanifolds}): near a point where $Df$ is
invertible, $f$ \emph{is} an invertible linear map plus a
small Lipschitz perturbation, and today's example does the
rest. It also quantifies numerical robustness: a system
perturbed by less than the inverse's margin remains
\omterm{def:b3:groups:derived}{solvable}, with the solution moving by at most
$\frac{1}{1-k}$ times the perturbation.
\end{example}

\section{Baire's theorem}

\begin{theorem}[Baire]\label{thm:b3:complete:baire}
In a \omterm{def:b3:complete:complete}{complete metric space}, a countable intersection of dense
\omterm{def:b3:topology:topology}{open sets} is dense. Equivalently: if $X = \bigcup_{n}F_n$ with
each $F_n$ closed, then some $F_n$ has nonempty
\omterm{def:b3:topology:interior}{interior}.\index{Baire's theorem}
\end{theorem}

\begin{proof}
Let $(U_n)$ be dense opens and $B_0 = B(x_0, r_0)$ any open
ball; we find a point of $\bigcap U_n$ in $B_0$. Inductively:
$U_{n}$ being dense and open, it meets the open ball $B_{n-1}$
in an \omterm{def:b3:topology:topology}{open set}, which contains a closed ball $\bar B(x_n, r_n)$
with $0 < r_n \leq r_{n-1}/2$ and $\bar B(x_n, r_n) \subseteq
B_{n-1}\cap U_n$. The \omterm{ex:b3:groups:actions}{centers} form a Cauchy sequence ($x_m \in
B_n$ for $m \geq n$, radii $\to 0$); the limit $x$ lies in
every $\bar B(x_n, r_n)$ (closedness), hence in every $U_n$ and
in $B_0$. For the second form: if no $F_n$ has \omterm{def:b3:topology:interior}{interior}, the
$U_n = X \setminus F_n$ are open and dense, and a point of
$\bigcap U_n$ escapes $\bigcup F_n = X$: absurd.
\end{proof}

\begin{remark}\label{rem:b3:complete:meagre}
Vocabulary: a set is \emph{nowhere dense} if its \omterm{def:b3:topology:interior}{closure} has
empty \omterm{def:b3:topology:interior}{interior}, \emph{meagre}\index{meagre set} (first category)
if it is a countable union of nowhere \omterm{def:b3:topology:interior}{dense sets}. Baire:
\emph{a \omterm{def:b3:complete:complete}{complete metric space} is not \omterm{rem:b3:complete:meagre}{meagre} in itself}, and the
complement of a \omterm{rem:b3:complete:meagre}{meagre set} is dense. ``\omterm{rem:b3:complete:meagre}{Meagre}'' is a notion of
smallness orthogonal to measure (\cref{ch:b3:measure} will
produce \omterm{rem:b3:complete:meagre}{meagre sets} of full measure), and Baire arguments prove
\emph{existence by abundance}: to exhibit one object without
property P, show the P-objects form a \omterm{rem:b3:complete:meagre}{meagre set}.
\end{remark}

\begin{corollary}\label{cor:b3:complete:bairecors}
(a) $\R$ is uncountable. (b) $\Q$ is not a countable
intersection of open subsets of $\R$, and a nonempty \omterm{def:b3:complete:complete}{complete
metric space} without isolated points is uncountable.
\end{corollary}

\begin{proof}
(a) $\R = \bigcup_{x}\{x\}$ over a countable set would make some
singleton have \omterm{def:b3:topology:interior}{interior}. (b) If $\Q = \bigcap_n V_n$ with $V_n$
open (necessarily dense, as $\supseteq \Q$), then the sets
$V_n$ and the complements $\R\setminus\{q\}$, $q \in \Q$, form a
countable family of dense opens with empty intersection ---
contradicting Baire. If $X$ is \omterm{def:b3:complete:complete}{complete} without isolated points
and countable, $X = \bigcup_{x \in X}\{x\}$ exhibits it as a
countable union of closed sets with empty \omterm{def:b3:topology:interior}{interior} (no isolated
points): Baire again.
\end{proof}

\begin{theorem}[Weierstrass's monsters exist]
\label{thm:b3:complete:nowherediff}
There exist \omterm{def:b3:topology:continuity}{continuous} functions on $\intcc01$ differentiable at
no point. Indeed, the set of $f \in \mathcal C(\intcc01)$
having a (finite) derivative at even one point is \omterm{rem:b3:complete:meagre}{meagre} in
$(\mathcal C(\intcc01), \norm\cdot_\infty)$.\index{nowhere
differentiable function}
\end{theorem}

\begin{proof}
For $n \geq 1$ let
\[
F_n = \Bigl\{f : \exists x \in \intcc01,\
\forall h \neq 0 \text{ with } x + h \in \intcc01,\
\abs{f(x+h) - f(x)} \leq n\abs h \Bigr\}.
\]
If $f$ is differentiable at $x$, then $f \in F_n$ for some $n$:
the quotient $\abs{f(x+h)-f(x)}/\abs h$ is bounded for $\abs h
\leq \delta$ (differentiability: it tends to
$\abs{f'(x)}$) and bounded by $2\norm f_\infty/\delta$ for
$\abs h \geq \delta$. So $\bigcup F_n$ contains
all somewhere-differentiable functions, and it suffices to show
each $F_n$ is closed with empty \omterm{def:b3:topology:interior}{interior}.

\emph{Closed}: let $f_k \to f$ uniformly, $f_k \in F_n$ with
witnesses $x_k \to x$ (compactness, after extraction). For $h$
with $x + h \in \intcc01$: choose $h_k \to h$ with $x_k + h_k
\in \intcc01$ (e.g.\ $h_k = h + x - x_k$ clipped); then
$\abs{f(x + h) - f(x)} = \lim \abs{f_k(x_k + h_k) - f_k(x_k)}
\leq \lim n\abs{h_k} = n\abs h$, using uniform convergence and
\omterm{def:b3:topology:continuity}{continuity} of $f$ at the relevant points: $f \in F_n$.

\emph{Empty \omterm{def:b3:topology:interior}{interior}}: given $f \in F_n$ and $\varepsilon > 0$,
we find $g$ with $\norm{g - f}_\infty \leq \varepsilon$ and $g
\notin F_n$. First approximate $f$ within $\varepsilon/2$ by a
piecewise affine $\varphi$ (uniform \omterm{def:b3:topology:continuity}{continuity}: interpolate on a
fine grid), of slopes bounded by some $M$. Add a small sawtooth:
$g = \varphi + \frac\varepsilon2\,s_N$, where $s_N(x)$ is the
$\frac1N$-periodic zigzag of amplitude $1$ and slope $\pm 2N$.
At every $x$, on one side there is $h$ arbitrarily small with
the sawtooth contributing slope $\pm 2N\cdot\frac\varepsilon2 =
\pm\varepsilon N$ over $[x, x+h]$: the difference quotient of
$g$ exceeds $\varepsilon N - M > n$ for $N$ large. So $g \notin
F_n$, at any uniform distance $\leq \varepsilon$ from $f$.
Conclusion: $\bigcup F_n$ is \omterm{rem:b3:complete:meagre}{meagre}; by Baire its complement
--- made of \omterm{thm:b3:complete:nowherediff}{nowhere differentiable functions} --- is dense in
$\mathcal C(\intcc01)$: such functions exist \emph{in
abundance}.
\end{proof}

\section{Arzel\`a--Ascoli}

Throughout, $K$ is a \emph{\omterm{def:b3:topology:compact}{compact}} metric space and $\mathcal
C(K) = \mathcal C(K, \R^d)$, with $\norm f_\infty = \sup_K
\norm{f(x)}$: a \omterm{def:b3:complete:complete}{complete} space (uniform limits of \omterm{def:b3:topology:continuity}{continuous}
are \omterm{def:b3:topology:continuity}{continuous} --- Year 2).

\begin{definition}\label{def:b3:complete:equicontinuous}
A family $\mathcal F \subseteq \mathcal C(K)$ is
\emph{equicontinuous}\index{equicontinuity} if for every
$\varepsilon > 0$ there is $\delta > 0$ such that
\[
d(x, y) < \delta \implies \norm{f(x) - f(y)} < \varepsilon
\quad\text{for \emph{all} } f \in \mathcal F
\]
(one $\delta$ for the whole family --- e.g.\ any family with a
common Lipschitz constant, or a common H\"older modulus), and
\emph{pointwise bounded} if $\sup_{f}\norm{f(x)} < \infty$ for
each $x$.
\end{definition}

\begin{theorem}[Arzel\`a--Ascoli]\label{thm:b3:complete:ascoli}
A subset $\mathcal F \subseteq \mathcal C(K)$ is relatively
\omterm{def:b3:topology:compact}{compact} (has \omterm{def:b3:topology:compact}{compact} \omterm{def:b3:topology:interior}{closure}) iff it is \omterm{def:b3:complete:equicontinuous}{equicontinuous} and
pointwise bounded. In particular, every \omterm{def:b3:complete:equicontinuous}{equicontinuous},
pointwise bounded \emph{sequence} has a uniformly convergent
subsequence.\index{Arzel\`a--Ascoli theorem}
\end{theorem}

\begin{proof}
($\Leftarrow$) Let $(f_k)$ be a sequence of $\mathcal F$. $K$
\omterm{def:b3:topology:compact}{compact} metric is \emph{separable}: for each $n$, finitely many
balls of radius $\frac1n$ cover $K$
(\cref{thm:b3:topology:metriccompact}); their \omterm{ex:b3:groups:actions}{centers} form a
countable \omterm{def:b3:topology:interior}{dense set} $D = \{x_1, x_2, \dots\}$. By pointwise
boundedness and Bolzano--Weierstrass, extract successively
subsequences converging at $x_1$, then also at $x_2$, etc., and
take the \emph{diagonal} subsequence $(g_j)$: it converges at
every point of $D$. \omterm{def:b3:complete:equicontinuous}{Equicontinuity} upgrades this to uniform
Cauchy: given $\varepsilon$, take $\delta$ as in the
definition, cover $K$ by finitely many balls $B(x_i, \delta)$
with $x_i \in D$ ($i \leq m$), and pick $J$ so large that
$\norm{g_j(x_i) - g_{j'}(x_i)} < \varepsilon$ for $j, j' \geq
J$, $i \leq m$. For arbitrary $x \in B(x_i, \delta)$:
\[
\norm{g_j(x) - g_{j'}(x)}
\leq \norm{g_j(x) - g_j(x_i)} + \norm{g_j(x_i) - g_{j'}(x_i)}
+ \norm{g_{j'}(x_i) - g_{j'}(x)} < 3\varepsilon .
\]
So $(g_j)$ is uniformly Cauchy, and converges in the \omterm{def:b3:complete:complete}{complete}
$\mathcal C(K)$. Thus every sequence of $\mathcal F$ has a
convergent subsequence: $\bar{\mathcal F}$ is (sequentially,
hence by \cref{thm:b3:topology:metriccompact}) \omterm{def:b3:topology:compact}{compact}.

($\Rightarrow$) If $\bar{\mathcal F}$ is \omterm{def:b3:topology:compact}{compact}: pointwise
boundedness is clear (evaluation is \omterm{def:b3:topology:continuity}{continuous}). For
\omterm{def:b3:complete:equicontinuous}{equicontinuity}, cover $\mathcal F$ by finitely many balls
$B(f_i, \varepsilon)$ of $\mathcal C(K)$; each $f_i$ is
uniformly \omterm{def:b3:topology:continuity}{continuous} (Heine,
\cref{cor:b3:topology:heineborel}), giving a common $\delta$
for $i \leq m$; then for $f \in B(f_i, \varepsilon)$ and $d(x,y)
< \delta$: $\norm{f(x)-f(y)} \leq 2\varepsilon +
\norm{f_i(x)-f_i(y)} < 3\varepsilon$.
\end{proof}

\begin{example}\label{ex:b3:complete:ascoliexamples}
The closed unit ball of $\mathcal C(\intcc01)$ is \emph{not}
\omterm{def:b3:topology:compact}{compact} ($f_n(x) = x^n$ has no uniformly convergent
subsequence: the pointwise limit is discontinuous), and indeed
$(x^n)$ is not \omterm{def:b3:complete:equicontinuous}{equicontinuous} at $1$. By contrast $\{f : \norm
f_\infty \leq 1,\ \operatorname{Lip}(f) \leq 1\}$ is \omterm{def:b3:topology:compact}{compact}:
bounded and $1$-Lipschitz-equicontinuous, and closed. Ascoli
explains \emph{why} compactness fails in infinite dimension
(Riesz, Year 2) and what to add to restore it: a uniform
modulus of \omterm{def:b3:topology:continuity}{continuity}.
\end{example}

\section{Stone--Weierstrass}

\begin{lemma}[Dini]\label{lem:b3:complete:dini}
Let $K$ be \omterm{def:b3:topology:compact}{compact} and $(f_n)$ a \emph{monotone} sequence of
\omterm{def:b3:topology:continuity}{continuous} real functions converging \emph{pointwise} to a
\omterm{def:b3:topology:continuity}{continuous} $f$. Then the convergence is
uniform.\index{Dini's theorem}
\end{lemma}

\begin{proof}
Say $f_n \uparrow f$; let $g_n = f - f_n \downarrow 0$,
\omterm{def:b3:topology:continuity}{continuous}. Given $\varepsilon$, the \omterm{def:b3:topology:topology}{open sets} $U_n = \{g_n <
\varepsilon\}$ increase and cover $K$ (pointwise convergence);
extract a finite subcover: $K = U_{n_0}$ for some $n_0$
(increasing family), i.e.\ $0 \leq g_n < \varepsilon$
everywhere for $n \geq n_0$.
\end{proof}

\begin{lemma}\label{lem:b3:complete:sqrt}
There is a sequence of \emph{polynomials} $u_n$ with $u_n(t) \to
\sqrt t$ uniformly on $\intcc01$.
\end{lemma}

\begin{proof}
Define $u_0 = 0$, $u_{n+1}(t) = u_n(t) + \frac12\bigl(t -
u_n(t)^2\bigr)$: polynomials. By induction $0 \leq u_n(t) \leq
\sqrt t$ on $\intcc01$: granting it for $n$,
\[
\sqrt t - u_{n+1}(t) = \bigl(\sqrt t - u_n(t)\bigr)
\Bigl(1 - \tfrac12\bigl(\sqrt t + u_n(t)\bigr)\Bigr) \geq 0,
\]
since $\sqrt t + u_n \leq 2$; and $u_{n+1} \geq u_n \geq 0$.
So $(u_n(t))$ is nondecreasing, bounded by $\sqrt t$: it
converges pointwise, and the limit $\ell(t)$ satisfies $\ell =
\ell + \frac12(t - \ell^2)$: $\ell(t) = \sqrt t$, \omterm{def:b3:topology:continuity}{continuous}.
Dini (\cref{lem:b3:complete:dini}) upgrades to uniform.
\end{proof}

\begin{theorem}[Stone--Weierstrass, real version]
\label{thm:b3:complete:stoneweierstrass}
Let $K$ be a \omterm{def:b3:topology:compact}{compact} (\omterm{def:b3:topology:hausdorff}{Hausdorff}) space and $\mathcal A \subseteq
\mathcal C(K, \R)$ a \emph{subalgebra} (stable under sums,
products, scalar multiples) that \emph{contains the constants}
and \emph{separates points} (for $x \neq y$, some $f \in
\mathcal A$ has $f(x) \neq f(y)$). Then $\mathcal A$ is dense in
$(\mathcal C(K,\R), \norm\cdot_\infty)$.\index{Stone--Weierstrass
theorem}
\end{theorem}

\begin{proof}
Let $\bar{\mathcal A}$ be the \omterm{def:b3:topology:interior}{closure}, again an algebra
(products of uniform limits on bounded sets converge).

\emph{Step 1: $\bar{\mathcal A}$ is a lattice}, i.e.\ stable
under $\max$ and $\min$. Since $\max(f,g) = \frac{f + g +
\abs{f-g}}2$ and $\min$ likewise, it suffices that $f \in
\bar{\mathcal A} \Rightarrow \abs f \in \bar{\mathcal A}$: with
$M = \norm f_\infty > 0$, $\abs f = M\sqrt{(f/M)^2}$, and
\cref{lem:b3:complete:sqrt} gives polynomials $u_n$ with
$u_n\bigl((f/M)^2\bigr) \to \abs f/M$ uniformly; polynomials in
members of the algebra (with constant term: constants are
there) stay in $\bar{\mathcal A}$.

\emph{Step 2: two-point interpolation.} For $x \neq y$ and $a,
b \in \R$, some $g \in \mathcal A$ has $g(x) = a$, $g(y) = b$:
take $h$ separating $x, y$ and set $g = a + (b -
a)\frac{h - h(x)}{h(y) - h(x)}$.

\emph{Step 3.} Let $f \in \mathcal C(K)$, $\varepsilon > 0$.
For each pair $x, y$ pick $g_{x,y} \in \mathcal A$ with
$g_{x,y}(x) = f(x)$, $g_{x,y}(y) = f(y)$ (Step 2; for $x = y$
take the constant function $g_{x,x} = f(x)$).
Fix $x$: for each $y$, the \omterm{def:b3:topology:topology}{open set} $V_y = \{g_{x,y} < f +
\varepsilon\}$ contains $y$; compactness extracts $y_1, \dots,
y_m$ with $K = \bigcup V_{y_j}$, and $h_x = \min_j g_{x, y_j}
\in \bar{\mathcal A}$ (Step 1) satisfies $h_x < f +
\varepsilon$ everywhere, $h_x(x) = f(x)$. Now vary $x$: $W_x =
\{h_x > f - \varepsilon\}$ is open, contains $x$; extract $x_1,
\dots, x_l$ covering $K$, and $h = \max_i h_{x_i} \in
\bar{\mathcal A}$ satisfies $f - \varepsilon < h < f +
\varepsilon$: $\norm{f - h}_\infty \leq \varepsilon$. Hence
$f \in \bar{\mathcal A}$.
\end{proof}

\begin{corollary}\label{cor:b3:complete:weierstrass}
(a) (\emph{Weierstrass}) Polynomials are dense in $\mathcal
C(\intcc ab, \R)$; polynomials in $n$ variables are dense in
$\mathcal C(K, \R)$ for \omterm{def:b3:topology:compact}{compact} $K \subseteq \R^n$.
(b) (\emph{Complex version}) If $\mathcal A \subseteq \mathcal
C(K, \C)$ is a subalgebra containing constants, separating
points, and \emph{stable under conjugation}, it is dense.
(c) (\emph{Trigonometric version}) Trigonometric polynomials
$\sum_{\abs n \leq N}c_n\eu^{\iu n t}$ are dense in the space of
\omterm{def:b3:topology:continuity}{continuous} $2\pi$-periodic functions with
$\norm\cdot_\infty$.\index{Weierstrass approximation theorem}
\end{corollary}

\begin{proof}
(a) Polynomials form an algebra with constants; the coordinate
functions separate points of $\R^n$.
(b) The real and imaginary parts $\frac{f + \bar f}2$,
$\frac{f - \bar f}{2\iu}$ of members of $\mathcal A$ form a
real algebra $\mathcal A_\R \subseteq \mathcal C(K, \R)$ with
constants; it separates points ($f(x) \ne f(y)$ forces
$\operatorname{Re}f$ or $\operatorname{Im}f$ to separate).
Apply the real theorem and recombine.
(c) View $2\pi$-periodic functions as $\mathcal C(S^1, \C)$
($S^1 = \R/2\pi\Z$, \omterm{def:b3:topology:compact}{compact}:
\cref{exo:b3:topology:5}); the algebra generated by
$\eu^{\iu t}, \eu^{-\iu t}$ and constants is stable under
conjugation and separates points of the circle ($\eu^{\iu t}$
is injective on it). Apply (b).
\end{proof}

\begin{remark}\label{rem:b3:complete:swremark}
The trigonometric version repairs, and vastly generalizes, the
gap left in Year 2's Fourier chapter: density of trigonometric
polynomials in $(\mathcal C(S^1), \norm\cdot_2)$ follows a
fortiori ($\norm\cdot_2 \leq \norm\cdot_\infty$ up to the
normalizing constant), which will make the Fourier system an
orthonormal \emph{basis} in \cref{ch:b3:hilbert}, proving
Parseval in full generality at last.
\end{remark}

\section{Problem: Peano's existence theorem}

\begin{problem}[{Weekend problem --- existence of solutions of
$x' = f(t, x)$ without Lipschitz}]\label{pb:b3:complete:1}
Cauchy--Lipschitz (Year 2; re-proved in \cref{ch:b3:ode})
demands $f$ Lipschitz in $x$. Peano (1890): \emph{\omterm{def:b3:topology:continuity}{continuity} of
$f$ already yields existence} --- though not uniqueness. We
prove it with Euler polygons and Ascoli. Setting: $f \colon R
\to \R^d$ \omterm{def:b3:topology:continuity}{continuous} on the rectangle $R = \intcc{t_0 - a}{t_0
+ a}\times \bar B(x_0, b) \subseteq \R\times\R^d$, $M =
\sup_R\norm f$, and
\[
T = \min\bigl(a,\ b/M\bigr)
\qquad (\text{if } M = 0 \text{ the problem is trivial}).
\]

\medskip
\noindent\textbf{Part I --- Euler polygons.} For $n \geq 1$,
subdivide $[t_0, t_0 + T]$ by $t_k = t_0 + kT/n$ ($0 \leq k
\leq n$) and define $\varphi_n$ piecewise affinely:
$\varphi_n(t_0) = x_0$ and, on $[t_k, t_{k+1}]$,
\[
\varphi_n(t) = \varphi_n(t_k) + (t -
t_k)\,f\bigl(t_k, \varphi_n(t_k)\bigr).
\]
\begin{enumerate}
  \item Show by induction that $\varphi_n$ is well defined,
        with $\norm{\varphi_n(t) - x_0} \leq M(t - t_0) \leq b$
        on $[t_0, t_0 + T]$ --- so the evaluation points stay
        in $R$. \emph{(This is where $T \leq b/M$ enters.)}
  \item Show that each $\varphi_n$ is $M$-Lipschitz.
  \item Deduce from Arzel\`a--Ascoli
        (\cref{thm:b3:complete:ascoli}) that a subsequence
        $(\varphi_{n_j})$ converges uniformly on $[t_0, t_0 +
        T]$ to some $\varphi$, itself $M$-Lipschitz with
        $\varphi(t_0) = x_0$.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The limit solves the equation.}
Define the \emph{defect} $\Delta_n(t) = \varphi_n'(t) -
f\bigl(t, \varphi_n(t)\bigr)$ at non-grid points.
\begin{enumerate}[resume]
  \item Show that $f$ is uniformly \omterm{def:b3:topology:continuity}{continuous} on $R$, and
        deduce: for every $\varepsilon > 0$ there is $n_0$ such
        that for $n \geq n_0$ and every non-grid $t$,
        $\norm{\Delta_n(t)} \leq \varepsilon$. \emph{(On $(t_k,
        t_{k+1})$, $\varphi_n'(t) = f(t_k, \varphi_n(t_k))$,
        and $(t, \varphi_n(t))$ is within distance
        $(1 + M)\,T/n$ of $(t_k, \varphi_n(t_k))$.)}
  \item Establish the integral form: for all $t$,
        \[
        \varphi_n(t) = x_0 + \int_{t_0}^{t}
        f\bigl(s, \varphi_n(s)\bigr)\dd s +
        \int_{t_0}^t \Delta_n(s)\,\dd s,
        \]
        the middle integrand being piecewise \omterm{def:b3:topology:continuity}{continuous}.
  \item Pass to the limit along $(n_j)$: show
        $f(s, \varphi_{n_j}(s)) \to f(s, \varphi(s))$
        \emph{uniformly} (uniform \omterm{def:b3:topology:continuity}{continuity} of $f$ again), and
        conclude
        \[
        \varphi(t) = x_0 + \int_{t_0}^t
        f\bigl(s, \varphi(s)\bigr)\dd s .
        \]
  \item Deduce that $\varphi$ is $\mathcal C^1$ on $[t_0, t_0 +
        T]$ and solves $x' = f(t, x)$, $x(t_0) = x_0$; extend
        the construction to $[t_0 - T, t_0]$ (time reversal).
        \emph{This is Peano's theorem.}
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Uniqueness genuinely fails.}
Consider $x' = 2\sqrt{\abs x}$, $x(0) = 0$, on $\R$.
\begin{enumerate}[resume]
  \item Check that $f(x) = 2\sqrt{\abs x}$ is \omterm{def:b3:topology:continuity}{continuous} but
        not Lipschitz on any \omterm{def:b3:topology:topology}{neighborhood} of $0$.
  \item Verify that $x \equiv 0$ and, for every $c \geq 0$,
        \[
        x_c(t) = \begin{cases} 0 & t \leq c,\\ (t - c)^2 & t
        \geq c,\end{cases}
        \]
        are all solutions through $(0,0)$: a continuum of
        distinct solutions.
  \item Where does the Picard iteration argument (Banach fixed
        point) break down for this $f$?
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Limits of the method.}
\begin{enumerate}[resume]
  \item Show that Peano's theorem fails in infinite dimension:
        we admit (or you may take on faith) the classical
        example of Dieudonn\'e in the space $c_0$ of null
        sequences; instead, prove the finite-dimensional
        ingredient that fails there: the closed unit ball of
        $c_0$ (sup norm) is not \omterm{def:b3:topology:compact}{compact} --- exhibit a bounded
        sequence with no convergent subsequence, and explain
        which step of Part I breaks.
  \item Summarize: which hypotheses give existence? existence
        and uniqueness? State precisely the two theorems
        (Peano; Cauchy--Lipschitz) side by side.
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Osgood: uniqueness beyond
Lipschitz.} Let $\omega\colon\intoo0\infty\to\intoo0\infty$ be
\omterm{def:b3:topology:continuity}{continuous}, nondecreasing, with
\[
\int_0^1\frac{\dd r}{\omega(r)} = +\infty
\qquad(\text{divergence at } 0),
\]
and suppose $f$ satisfies $\norm{f(t, x) - f(t, y)} \leq
\omega\bigl(\norm{x - y}\bigr)$ on $R$.
\begin{enumerate}[resume]
  \item Check that $\omega(r) = Lr$ qualifies (Lipschitz), that
        $\omega(r) = r\log\frac1r$ (extended by \omterm{def:b3:topology:continuity}{continuity}, for
        $r$ small) qualifies although it is \emph{not} $O(r)$,
        and that $\omega(r) = 2\sqrt r$ does not. Compute the
        integral in each case.
  \item Let $x_1, x_2$ solve the equation on $[t_0, t_0 + T]$
        with $x_1(t_0) = x_2(t_0)$, and $\delta(t) =
        \norm{x_1(t) - x_2(t)}$. Show, from the integral forms
        alone, that for $t_0 \leq s \leq t$:
        \[
        \delta(t) \leq \delta(s) +
        \int_s^t\omega\bigl(\delta(v)\bigr)\dd v .
        \]
  \item (Osgood's theorem) Suppose $\delta(t_1) > 0$ for some
        $t_1$, and let $\tau = \sup\{t \leq t_1 : \delta(t) =
        0\}$. For $s \in \intoo\tau{t_1}$, set $u(t) = \delta(s)
        + \int_s^t\omega(\delta(v))\,\dd v$. Show $\delta \leq
        u$, $u' = \omega(\delta) \leq \omega(u)$, and deduce
        \[
        \int_{\delta(s)}^{u(t_1)}\frac{\dd r}{\omega(r)}
        \leq t_1 - s \leq t_1 - \tau .
        \]
        Let $s \downarrow \tau$ and derive a contradiction with
        the divergence of the integral. Conclude:
        \emph{solutions through a common initial condition
        coincide} --- uniqueness under Osgood's condition.
  \item Draw the consequences: Cauchy--Lipschitz uniqueness is
        the case $\omega(r) = Lr$; the equation $x' =
        x\log\frac1{\abs x}$ (extended by $0$ at $0$) has unique
        solutions although its right side is not Lipschitz at
        $0$; and for $x' = 2\sqrt{\abs x}$ the convergence of
        $\int_0\frac{\dd r}{2\sqrt r}$ is exactly what lets a
        solution leave $0$ in finite time --- match the value of
        the integral $\int_0^{h^2}\frac{\dd r}{2\sqrt r} = h$
        with the escape behaviour of $x_c$.
\end{enumerate}

\medskip
\noindent\textbf{Part VI --- Rates, schemes, funnels.}
\begin{enumerate}[resume]
  \item (Integral Gr\"onwall lemma) Let $e, \eta \geq 0$ be
        \omterm{def:b3:topology:continuity}{continuous} on $[t_0, t_0+T]$ with $e(t) \leq \int_{t_0}
        ^t\bigl(L\,e(s) + \eta(s)\bigr)\dd s$ for all $t$. Show
        \[
        e(t) \leq \frac{\sup\eta}{L}\bigl(\eu^{L(t - t_0)} -
        1\bigr)
        \]
        \emph{(set $G(t) = \int_{t_0}^t(Le + \eta)$, note $G'
        \leq LG + \sup\eta$, and differentiate
        $\eu^{-Lt}G(t)$)}.
  \item (Euler converges with a rate) Suppose now $f$ is
        $L$-Lipschitz in $x$ and $L'$-Lipschitz in $t$ on $R$.
        Combining the defect bound of question 4 (made
        quantitative: $\norm{\Delta_n} \leq (L' + LM)T/n$) with
        question 17, prove
        \[
        \sup_{[t_0, t_0+T]}\norm{\varphi_n - \varphi}
        \;\leq\; \frac{(L' + LM)\,T}{n}\cdot
        \frac{\eu^{LT} - 1}{L} = O\Bigl(\frac1n\Bigr),
        \]
        where $\varphi$ is \emph{the} solution: with Lipschitz
        data the whole sequence converges, with an explicit
        rate --- no subsequences needed. Why does uniqueness
        upgrade subsequential convergence to full convergence
        even without this computation?
  \item (The scheme chooses) For $x' = 2\sqrt{\abs x}$, $x(0) =
        0$: show that every Euler polygon is identically zero,
        so the scheme converges to the solution $x \equiv 0$;
        but started at $x(0) = \varepsilon > 0$ it converges
        (as $n \to \infty$, then $\varepsilon \to 0$) to $t
        \mapsto t^2$, a \emph{different} solution through the
        origin. Non-uniqueness resurfaces as sensitivity of the
        numerical scheme to perturbations.
  \item Show that the set $\mathcal S$ of \emph{all} solutions
        of $x' = f(t,x)$, $x(t_0) = x_0$ on $[t_0, t_0+T]$
        (values in $\bar B(x_0, b)$) is nonempty (Part II),
        uniformly $M$-Lipschitz, and closed in
        $\bigl(\mathcal C([t_0, t_0+T], \R^d),
        \norm\cdot_\infty\bigr)$; conclude by Ascoli that
        $\mathcal S$ is \emph{\omterm{def:b3:topology:compact}{compact}}. (Kneser's theorem adds
        that $\mathcal S$ is \omterm{def:b3:topology:connected}{connected}; we shall not prove
        it.)
  \item Verify Kneser's phenomenon on the example: for $x' =
        2\sqrt{\abs x}$, $x(0) = 0$, on $[0, 1]$, show
        $\mathcal S = \{x_c : c \in \intcc0\infty\}$ with
        $x_\infty \equiv 0$ \emph{(for any solution, let $c =
        \sup\{t : x(t) = 0\}$ and integrate $(\sqrt x)' = 1$ on
        $\{x > 0\}$)}, that $c \mapsto x_c$ is \omterm{def:b3:topology:continuity}{continuous} from
        $\intcc0\infty$ (\omterm{def:b3:topology:locallycompact}{one-point compactification}, i.e.\
        $c = \infty$ glued as limit) to $\mathcal C([0,1])$,
        and conclude that $\mathcal S$ is indeed \omterm{def:b3:topology:compact}{compact} and
        \omterm{def:b3:topology:connected}{connected} --- a segment-shaped funnel.
  \item (Reachable sets) Deduce from question 20 that for each
        fixed $t$, the \emph{reachable set} $\mathcal S(t) =
        \{x(t) : x \in \mathcal S\}$ is \omterm{def:b3:topology:compact}{compact}; compute it for
        the example of question 21 and check it is also
        \omterm{def:b3:topology:connected}{connected}: $\mathcal S(t) = \intcc0{t^2}$ --- every
        intermediate state is attained by some solution.
\end{enumerate}

\medskip
\noindent\textbf{Part VII --- Complements: dependence,
optimality, and a scheme computed by hand.}
\begin{enumerate}[resume]
  \item (\omterm{def:b3:topology:continuity}{Continuous} dependence) Suppose $f$ is $L$-Lipschitz
        in $x$ on $R$, and let $x, y$ be two solutions with
        initial values $x_0, y_0$ at $t_0$. Adapting the proof
        of question 17 to the inequality $e(t) \leq \norm{x_0
        - y_0} + \int_{t_0}^tL\,e(s)\dd s$, show
        \[
        \norm{x(t) - y(t)} \leq \norm{x_0 - y_0}\,
        \eu^{L(t - t_0)},
        \]
        and check on $x' = Lx$ that the bound is attained:
        Gr\"onwall is sharp. Deduce again uniqueness ($x_0 =
        y_0$), and that the flow map $x_0 \mapsto x(t; x_0)$
        is Lipschitz, with constant $\eu^{LT}$, wherever it is
        defined.
  \item (Osgood is optimal) Conversely, let $\omega$ be
        \omterm{def:b3:topology:continuity}{continuous}, nondecreasing, positive on
        $\intoo0\infty$, with $\omega(r) \to 0$ as $r \to 0^+$
        but
        \[
        \int_0^1\frac{\dd r}{\omega(r)} < +\infty,
        \]
        and extend $f(x) = \omega(\abs x)$ by $f(0) = 0$.
        Show that $\Omega(x) = \int_0^x\frac{\dd r}{\omega(r)}$
        is an increasing bijection from $\intoc0{1}$ onto
        $\intoc0{\Omega(1)}$, that its inverse $g$ solves $g'
        = \omega(g)$ with $g(0^+) = 0$, and that $g$, extended
        by $0$ for $t \leq 0$, is a $\mathcal C^1$ solution of
        $x' = f(x)$ through $(0, 0)$ distinct from $x \equiv
        0$ \emph{(for $g'(0) = 0$, bound $\frac{g(t)}t$ by
        $\omega(g(t))$)}. Conclude: the divergence hypothesis
        of question 15 is not a convenience but the exact
        frontier of uniqueness; recover Part III from
        $\omega(r) = 2\sqrt r$, $\Omega(x) = \sqrt x$.
  \item (Euler computed by hand) For $x' = x$, $x(0) = 1$ on
        $[0, 1]$: show that the Euler polygon satisfies
        $\varphi_n(1) = \bigl(1 + \frac1n\bigr)^n$. Prove the
        expansion
        \[
        \Bigl(1 + \frac1n\Bigr)^{n}
        = \eu\Bigl(1 - \frac1{2n} + O\Bigl(\frac1{n^2}\Bigr)
        \Bigr),
        \]
        so the error at $t = 1$ is $\frac{\eu}{2n} +
        O(n^{-2})$: question 18's $O(\frac1n)$ rate, with the
        exact constant. Check numerically for $n = 10$:
        $1.1^{10} = 2.59374$ against $\eu \approx 2.71828$,
        an error $0.12454$ to compare with $\frac{\eu}{20}
        \approx 0.13591$.
\end{enumerate}
\end{problem}
