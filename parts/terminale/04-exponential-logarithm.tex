\chapter{Exponential and Logarithm}\label{ch:tle:exp}

The exponential function is the unique function equal to its own derivative
and taking the value $1$ at $0$. It converts sums into products; its inverse,
the natural logarithm, converts products into sums. Together they describe
every phenomenon whose rate of change is proportional to its size:
radioactive decay, population growth, compound interest.

\section{The exponential function}

\begin{theorem}[Existence and uniqueness]\label{thm:tle:exp:existence}
There exists a unique differentiable function $f \colon \R \to \R$ such that
\[
f' = f \qquad\text{and}\qquad f(0) = 1 .
\]
It is called the \emph{exponential function}\index{exponential} and written
$\exp$, or $x \mapsto \eu^x$.
\end{theorem}

\begin{proof}[Proof of uniqueness]
First, such a function never vanishes. Indeed, let
$g(x) = f(x)f(-x)$; then
\[
g'(x) = f'(x)f(-x) - f(x)f'(-x) = f(x)f(-x) - f(x)f(-x) = 0,
\]
so $g$ is constant equal to $g(0) = 1$: for every $x$,
$f(x)f(-x) = 1$, and in particular $f(x) \neq 0$.

Now let $f_1, f_2$ be two solutions and $h = \dfrac{f_1}{f_2}$ (legitimate
since $f_2$ never vanishes). Then
\[
h' = \frac{f_1' f_2 - f_1 f_2'}{f_2^2} = \frac{f_1 f_2 - f_1 f_2}{f_2^2} = 0,
\]
so $h$ is constant equal to $h(0) = 1$, \emph{i.e.} $f_1 = f_2$.

\emph{Existence} is admitted at this level (it can be obtained via the
sequence of \cref{exo:tle:seq:10}, or as the inverse of the logarithm
constructed by integration in \cref{ch:tle:integ}).
\end{proof}

\begin{proposition}[Functional equation]\label{prop:tle:exp:funceq}
For all $x, y \in \R$ and $n \in \Z$:
\[
\eu^{x+y} = \eu^x\,\eu^y, \qquad
\eu^{-x} = \frac{1}{\eu^x}, \qquad
\eu^{x-y} = \frac{\eu^x}{\eu^y}, \qquad
\bigl(\eu^{x}\bigr)^n = \eu^{nx}.
\]
Moreover $\eu^x > 0$ for all $x$.
\end{proposition}

\begin{proof}
Fix $y$ and consider $\varphi(x) = \dfrac{\exp(x+y)}{\exp(x)\exp(y)}$. Its
numerator and denominator, as functions of $x$, are both solutions of
$f' = f$ up to the constant $\exp(y)$; differentiating $\varphi$ directly
(quotient rule) gives $\varphi' = 0$, so
$\varphi \equiv \varphi(0) = 1$, proving the first identity. Taking
$y = -x$ gives the second (with the value $\eu^0 = 1$), and the third
follows. The fourth is an induction from the first for $n \geq 0$, extended
to $n < 0$ by the second.

Positivity: $\eu^x = \left(\eu^{x/2}\right)^2 \geq 0$ and $\eu^x \neq 0$
(shown in \cref{thm:tle:exp:existence}), so $\eu^x > 0$.
\end{proof}

\begin{proposition}[Variations and limits]\label{prop:tle:exp:variations}
The exponential is strictly increasing, convex, and
\[
\lim_{x\to-\infty} \eu^x = 0, \qquad
\lim_{x\to+\infty} \eu^x = +\infty .
\]
\end{proposition}

\begin{proof}
$(\exp)' = \exp > 0$ gives strict increase; $(\exp)'' = \exp > 0$ gives
convexity. By convexity, $\eu^x \geq 1 + x$ (tangent at $0$, see
\cref{ex:tle:deriv:ineq}), so $\eu^x \to +\infty$ as $x \to +\infty$ by
comparison. Then $\eu^{x} = 1/\eu^{-x} \to 0$ as $x \to -\infty$.
\end{proof}

\begin{theorem}[Growth comparison]\label{thm:tle:exp:growth}
\index{growth comparison}
For every integer $n \geq 1$,
\[
\lim_{x\to+\infty} \frac{\eu^x}{x^n} = +\infty,
\qquad
\lim_{x\to-\infty} x^n \eu^x = 0 .
\]
In words: the exponential beats every power of $x$.
\end{theorem}

\begin{proof}
For $n = 1$: applying $\eu^t \geq 1 + t \geq t$ at $t = x/2$,
\[
\frac{\eu^x}{x} = \frac{\left(\eu^{x/2}\right)^2}{x}
\geq \frac{(x/2)^2}{x} = \frac{x}{4} \xrightarrow[x\to+\infty]{} +\infty .
\]
For general $n$, write
\[
\frac{\eu^x}{x^n}
= \frac{1}{n^n}\left(\frac{\eu^{x/n}}{x/n}\right)^{n}
\xrightarrow[x\to+\infty]{} +\infty
\]
by the case $n=1$ and composition. The limit at $-\infty$ follows by the
substitution $x \mapsto -x$:
$\abs{x^n \eu^x} = \frac{\abs{x}^n}{\eu^{\abs x}} \to 0$.
\end{proof}

\section{The natural logarithm}

\begin{definition}[Natural logarithm]\label{def:tle:exp:ln}
The exponential is continuous and strictly increasing from $\R$ onto
$\intoo{0}{+\infty}$; by the bijection theorem
(\cref{thm:tle:limcont:bijection}), for every $y > 0$ the equation
$\eu^x = y$ has a unique solution. This solution is the \emph{natural
logarithm}\index{logarithm} of $y$, written $\ln y$. Thus
\[
\text{for } x \in \R,\ y > 0: \qquad y = \eu^x \iff x = \ln y .
\]
In particular $\ln 1 = 0$, $\ln \eu = 1$, and
$\eu^{\ln y} = y$, $\ln(\eu^x) = x$.
\end{definition}

\begin{proposition}[Algebraic properties]\label{prop:tle:exp:lnalg}
For all $a, b > 0$ and $n \in \Z$:
\[
\ln(ab) = \ln a + \ln b, \qquad
\ln\frac{1}{a} = -\ln a, \qquad
\ln\frac{a}{b} = \ln a - \ln b, \qquad
\ln(a^n) = n \ln a, \qquad
\ln\sqrt{a} = \tfrac12 \ln a .
\]
\end{proposition}

\begin{proof}
$\eu^{\ln a + \ln b} = \eu^{\ln a}\eu^{\ln b} = ab$, and taking $\ln$ of both
sides gives the first identity. The others follow by the same mechanism from
the corresponding identities of \cref{prop:tle:exp:funceq}.
\end{proof}

\begin{proposition}[Analytic properties]\label{prop:tle:exp:lnanalytic}
The function $\ln$ is differentiable on $\intoo{0}{+\infty}$ with
\[
(\ln x)' = \frac{1}{x},
\]
strictly increasing, concave, and
$\lim\limits_{x\to0^+} \ln x = -\infty$,
$\lim\limits_{x\to+\infty} \ln x = +\infty$. Moreover, for every integer
$n \geq 1$,
\[
\lim_{x\to+\infty} \frac{\ln x}{x^{1/n}} = 0, \qquad
\lim_{x\to0^+} x \ln x = 0 .
\]
\end{proposition}

\begin{proof}
Differentiability of the inverse function is admitted at this level; granting
it, differentiate the identity $\eu^{\ln x} = x$ by the chain rule:
$(\ln x)'\,\eu^{\ln x} = 1$, so $(\ln x)' = \frac{1}{x} > 0$, whence strict
increase, and $(\ln x)'' = -\frac1{x^2} < 0$, whence concavity. The limits at
$0^+$ and $+\infty$ mirror those of $\exp$ through the bijection. For the
growth comparison, substitute $x = \eu^{t}$:
$\frac{\ln x}{x} = \frac{t}{\eu^t} \to 0$ as $t \to +\infty$ by
\cref{thm:tle:exp:growth}, and similarly for the other limits with
$x = \eu^{-t}$, giving $x\ln x = -t\eu^{-t} \to 0$.
\end{proof}

\begin{method}[Solving equations with $\exp$ and $\ln$]\label{met:tle:exp:equations}
Both functions are strictly increasing, so they can be applied to (or removed
from) both sides of an equation or inequality without changing its direction:
\[
\eu^{u} = \eu^{v} \iff u = v, \qquad
\eu^{u} \leq \eu^{v} \iff u \leq v,
\]
and likewise for $\ln$ on positive quantities. \emph{Always check domains
first} ($\ln$ requires positive arguments). For equations like
$a^x = b$ with $a>0$, $a \neq 1$, rewrite $a^x = \eu^{x\ln a}$ and solve
$x = \frac{\ln b}{\ln a}$.
\end{method}

\begin{example}\label{ex:tle:exp:halflife}
A radioactive sample decays following $N(t) = N_0\,\eu^{-\lambda t}$. Its
\emph{half-life} $T$ satisfies $N(T) = N_0/2$, \emph{i.e.}
$\eu^{-\lambda T} = \frac12$, so $T = \frac{\ln 2}{\lambda}$: the half-life
does not depend on the initial quantity.
\end{example}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:tle:exp:1}
Simplify
$\dfrac{\eu^{3x}\,\eu^{-x+1}}{\eu^{x}}$ and
$\ln\!\left(\dfrac{\eu^2\sqrt{\eu}}{\eu^{-3}}\right)$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:tle:exp:2}
Solve in $\R$:
\[
\text{(a) } \eu^{2x} - 3\eu^{x} + 2 = 0;
\qquad
\text{(b) } \ln(x - 1) + \ln(x + 2) = \ln 4 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:tle:exp:3}
Compute the limits:
\[
\lim_{x\to+\infty} \frac{\eu^x - x^2}{\eu^x + 1}, \qquad
\lim_{x\to+\infty} \frac{\ln x}{\sqrt x}, \qquad
\lim_{x\to0^+} x^2 \ln x, \qquad
\lim_{x\to+\infty} \bigl(x - \ln x\bigr).
\]
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:tle:exp:4}
Study the function $f(x) = x\,\eu^{-x}$ on $\R$: variations, limits,
extremum; show that its curve has an inflection point and give its
coordinates.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:tle:exp:5}
Show that for all $x > -1$, $\ln(1 + x) \leq x$, with equality only at
$x = 0$. Deduce that for all $n \geq 1$,
\[
\left(1 + \frac{1}{n}\right)^n \leq \eu \leq \left(1 - \frac{1}{n+1}\right)^{-(n+1)} .
\]
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:tle:exp:6}
A capital $C_0$ is invested at an annual rate of $3\%$, interest compounded
each year.
\begin{enumerate}
  \item Express the capital $C_n$ after $n$ years.
  \item After how many years does the capital double? Give the exact answer
        using $\ln$, then a numerical value.
  \item Compare with the approximation ``$70$ divided by the rate in
        percent'' used by bankers.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:tle:exp:7}
Solve the inequality $\eu^{2x} - \eu^{x+1} > 0$, then the inequality
$\ln(x^2 - 1) \leq \ln(x + 5)$.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:tle:exp:8}
Let $f(x) = \dfrac{\eu^x}{x}$ for $x > 0$.
\begin{enumerate}
  \item Study the variations of $f$ on $\intoo{0}{+\infty}$ and give its
        minimum.
  \item For which values of $k$ does the equation $\eu^x = kx$ have $0$, $1$
        or $2$ solutions in $\intoo{0}{+\infty}$?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:tle:exp:9}
For $n \geq 1$, let $u_n = \left(1 + \frac1n\right)^n$.
\begin{enumerate}
  \item Using \cref{exo:tle:exp:5}, show that $(u_n)$ is bounded above
        by~$\eu$.
  \item Show that $\ln u_n = n \ln\left(1 + \frac1n\right) \to 1$, and
        deduce that $u_n \to \eu$.
        (Hint: recognize a difference quotient of $\ln$ at $1$.)
\end{enumerate}
\end{exercise}
