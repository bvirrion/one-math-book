\chapter{Vectors}\label{ch:g10:vectors}

A \omterm{def:g10:vectors:vector}{vector} encodes a displacement: a direction and a length, regardless of
the starting point. \omterm{def:g10:vectors:vector}{Vectors} give clean proofs of geometric facts and, once
\omterm{def:g10:coordgeom:system}{coordinates} enter the picture, reduce them to small computations. They
will be a key tool in \cref{ch:g10:lines} and, later, for the scalar
product in \cref{ch:g11:scal}.

\section{Translations and vectors}

\begin{definition}[Vector]\label{def:g10:vectors:vector}
Given two points $A$ and $B$, the \emph{vector}\index{vector} $\vect{AB}$
represents the displacement from $A$ to $B$: it has a
\emph{direction} (the direction of the line $(AB)$, together with the way
of traveling along it, from $A$ towards $B$) and a
\emph{length} (the distance $AB$, also written $\norm{\vect{AB}}$).
Two vectors are \emph{equal} when they encode the same displacement:
$\vect{AB} = \vect{CD}$ means that applying the two displacements to any
point gives the same result.
\end{definition}

\begin{proposition}[Equal vectors and parallelograms]\label{prop:g10:vectors:parallelogram}
$\vect{AB} = \vect{CD}$ exactly when $ABDC$ (in this order!) is a
parallelogram, possibly a flattened one --- equivalently, when $[AD]$ and
$[BC]$ have the same \omterm{prop:g10:coordgeom:midpoint}{midpoint}.
\end{proposition}
\admitted

\begin{notation}
A \omterm{def:g10:vectors:vector}{vector} is often named by a single letter, $\vec u$ or $\vec v$, when the
endpoints do not matter. The \emph{zero \omterm{def:g10:vectors:vector}{vector}} $\vec 0 = \vect{AA}$ is
the displacement that does not move anything. The \emph{opposite} of
$\vect{AB}$ is $\vect{BA}$: same length, opposite way. We write
$\vect{BA} = -\vect{AB}$.
\end{notation}

\section{Adding vectors}

\begin{definition}[Sum of two vectors]\label{def:g10:vectors:sum}
The \emph{sum}\index{vector!sum} $\vec u + \vec v$ is the displacement
obtained by performing $\vec u$ then $\vec v$.
\end{definition}

\begin{theorem}[Chasles' relation]\label{thm:g10:vectors:chasles}
For any three points $A$, $B$, $C$:
\[
\vect{AB} + \vect{BC} = \vect{AC}.
\]
\end{theorem}

\begin{proof}
Going from $A$ to $B$, then from $B$ to $C$, is a displacement that takes
$A$ to $C$: it is the displacement encoded by $\vect{AC}$.
\end{proof}

\begin{remark}[Parallelogram rule]
To add two \omterm{def:g10:vectors:vector}{vectors} drawn from the same point, $\vect{AB} + \vect{AC}$,
complete the parallelogram $ABDC$: the \omterm{def:g10:vectors:sum}{sum} is the diagonal $\vect{AD}$.
Indeed $\vect{AC} = \vect{BD}$, so
$\vect{AB} + \vect{AC} = \vect{AB} + \vect{BD} = \vect{AD}$ by Chasles'
relation.
\end{remark}

\begin{example}[Simplifying with Chasles]\label{ex:g10:vectors:chasles}
Simplify $\vect{MN} + \vect{NP} + \vect{PQ}$: chaining the displacements
gives $\vect{MQ}$. Simplify $\vect{AB} - \vect{AC}$: rewrite the
difference as a \omterm{def:g10:vectors:sum}{sum} with the opposite \omterm{def:g10:vectors:vector}{vector},
\[
\vect{AB} - \vect{AC} = \vect{AB} + \vect{CA}
= \vect{CA} + \vect{AB} = \vect{CB}.
\]
\end{example}

\section{Multiplying a vector by a number}

\begin{definition}[Scalar multiple]\label{def:g10:vectors:scalar}
Let $\vec u$ be a nonzero \omterm{def:g10:vectors:vector}{vector} and $k$ a \omterm{def:g10:numbers:sets}{real number}. The \omterm{def:g10:vectors:vector}{vector}
$k\vec u$\index{vector!scalar multiple} has the same direction as
$\vec u$, length $\abs{k} \times \norm{\vec u}$, and points the same way
as $\vec u$ if $k > 0$, the opposite way if $k < 0$. For $k = 0$,
$0\vec u = \vec 0$.
\end{definition}

\begin{definition}[Collinearity]\label{def:g10:vectors:collinear}
Two \omterm{def:g10:vectors:vector}{vectors} $\vec u$ and $\vec v$ are
\emph{collinear}\index{collinear vectors} when they have the same
direction, i.e.\ when $\vec v = k \vec u$ for some real $k$ (or one of
them is $\vec 0$).
\end{definition}

\begin{remark}
Collinearity is the \omterm{def:g10:vectors:vector}{vector} language for parallelism and alignment:
\begin{itemize}
  \item the lines $(AB)$ and $(CD)$ are parallel exactly when $\vect{AB}$
        and $\vect{CD}$ are \omterm{def:g10:vectors:collinear}{collinear};
  \item the points $A$, $B$, $C$ are aligned exactly when $\vect{AB}$ and
        $\vect{AC}$ are \omterm{def:g10:vectors:collinear}{collinear}.
\end{itemize}
\end{remark}

\section{Coordinates of vectors}

\begin{definition}[Coordinates of a vector]\label{def:g10:vectors:coords}
In a \omterm{def:g10:coordgeom:system}{coordinate system}, the \emph{\omterm{def:g10:coordgeom:system}{coordinates}} of the \omterm{def:g10:vectors:vector}{vector} $\vect{AB}$
are the numbers describing the displacement:
\[
\vect{AB}\,(x_B - x_A,\ y_B - y_A).
\]
A \omterm{def:g10:vectors:vector}{vector} $\vec u\,(a, b)$ moves every point $a$ units horizontally and
$b$ units vertically.
\end{definition}

\begin{proposition}[Computing with coordinates]\label{prop:g10:vectors:coordrules}
Let $\vec u\,(a, b)$ and $\vec v\,(c, d)$, and $k \in \R$.
\begin{enumerate}
  \item $\vec u = \vec v$ exactly when $a = c$ and $b = d$;
  \item $\vec u + \vec v$ has \omterm{def:g10:coordgeom:system}{coordinates} $(a + c,\ b + d)$;
  \item $k\vec u$ has \omterm{def:g10:coordgeom:system}{coordinates} $(ka,\ kb)$;
  \item $\norm{\vec u} = \sqrt{a^2 + b^2}$ (\omterm{def:g10:coordgeom:system}{orthonormal system}).
\end{enumerate}
\end{proposition}

\begin{proof}
1--3 restate, coordinate by coordinate, what the displacements do: for
instance performing $\vec u$ then $\vec v$ moves a point horizontally by
$a$ then $c$, hence by $a + c$ in total. Point 4 is the distance formula
of \cref{thm:g10:coordgeom:distance} applied to a segment representing
$\vec u$.
\end{proof}

\begin{theorem}[Collinearity criterion]\label{thm:g10:vectors:det}
Two \omterm{def:g10:vectors:vector}{vectors} $\vec u\,(a, b)$ and $\vec v\,(c, d)$ are \omterm{def:g10:vectors:collinear}{collinear} if and
only if
\[
ad - bc = 0 .
\]
\end{theorem}

\begin{proof}
If $\vec v = k\vec u$, then $c = ka$ and $d = kb$, so
$ad - bc = a(kb) - b(ka) = 0$. The same holds if $\vec u = k \vec v$ or if
either \omterm{def:g10:vectors:vector}{vector} is zero.

Conversely, suppose $ad - bc = 0$ with $\vec u \neq \vec 0$, say
$a \neq 0$ (the case $b \neq 0$ is similar). Set $k = \frac{c}{a}$; then
$c = ka$, and $ad = bc = b(ka)$ gives, after dividing by $a \neq 0$,
$d = kb$. So $\vec v = k\vec u$.
\end{proof}

\begin{example}\label{ex:g10:vectors:det}
Are $\vec u\,(3, -2)$ and $\vec v\,(-6, 4)$ \omterm{def:g10:vectors:collinear}{collinear}? Compute
$ad - bc = 3 \times 4 - (-2)\times(-6) = 12 - 12 = 0$: yes, and indeed
$\vec v = -2\vec u$. For $\vec u\,(3, -2)$ and $\vec w\,(1, 5)$:
$3 \times 5 - (-2) \times 1 = 17 \neq 0$: not \omterm{def:g10:vectors:collinear}{collinear}.
\end{example}

\begin{method}[Alignment and parallelism]\label{met:g10:vectors:alignment}
To prove that three points $A$, $B$, $C$ are aligned:
\begin{enumerate}
  \item compute the \omterm{def:g10:coordgeom:system}{coordinates} of $\vect{AB}$ and $\vect{AC}$;
  \item check the criterion $ad - bc = 0$ of \cref{thm:g10:vectors:det};
  \item conclude: the \omterm{def:g10:vectors:vector}{vectors} are \omterm{def:g10:vectors:collinear}{collinear} and share the point $A$, so
        the three points are aligned.
\end{enumerate}
The same computation with $\vect{AB}$ and $\vect{CD}$ proves that the
lines $(AB)$ and $(CD)$ are parallel.
\end{method}

\begin{example}\label{ex:g10:vectors:alignment}
Let $A(1, 2)$, $B(3, 5)$ and $C(7, 11)$. Then $\vect{AB}\,(2, 3)$ and
$\vect{AC}\,(6, 9)$, and $2 \times 9 - 3 \times 6 = 18 - 18 = 0$: the
points are aligned (in fact $\vect{AC} = 3\vect{AB}$).
\end{example}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:g10:vectors:1}
Simplify using Chasles' relation:
\[
\vect{AB} + \vect{BD}, \qquad
\vect{KL} + \vect{LM} + \vect{MK}, \qquad
\vect{AC} - \vect{BC}, \qquad
\vect{AB} + \vect{CA}.
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:vectors:2}
Let $A(2, 1)$, $B(5, 3)$, $C(-1, 4)$. Compute the \omterm{def:g10:coordgeom:system}{coordinates} of
$\vect{AB}$, $\vect{BC}$, $\vect{AC}$, and check on the \omterm{def:g10:coordgeom:system}{coordinates} that
$\vect{AB} + \vect{BC} = \vect{AC}$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:vectors:3}
Let $\vec u\,(2, -3)$ and $\vec v\,(-1, 4)$. Give the \omterm{def:g10:coordgeom:system}{coordinates} of
$\vec u + \vec v$, $3\vec u$, $2\vec u - \vec v$, and compute
$\norm{\vec u}$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:vectors:4}
In each case, say whether $\vec u$ and $\vec v$ are \omterm{def:g10:vectors:collinear}{collinear}:
\[
\text{(a) } \vec u\,(4, 6),\ \vec v\,(6, 9);
\qquad
\text{(b) } \vec u\,(2, -5),\ \vec v\,(-4, 10);
\qquad
\text{(c) } \vec u\,(3, 1),\ \vec v\,(1, 3).
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:vectors:5}
Let $A(0, 2)$, $B(4, 0)$, $C(6, 3)$. Find the \omterm{def:g10:coordgeom:system}{coordinates} of the point
$D$ such that $ABCD$ is a parallelogram, i.e.\ such that
$\vect{AB} = \vect{DC}$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:vectors:6}
Let $A(-2, 1)$, $B(1, 3)$, $C(4, -1)$.
\begin{enumerate}
  \item Compute the \omterm{def:g10:coordgeom:system}{coordinates} of the point $M$ such that
        $\vect{AM} = \vect{AB} + \vect{AC}$.
  \item What is the quadrilateral $ABMC$? Justify.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:vectors:7}
Let $A(1, -1)$, $B(4, 1)$, $C(10, 5)$. Are the points $A$, $B$, $C$
aligned? Same question for $A(0, 3)$, $B(2, 2)$, $C(8, -1)$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:vectors:8}
Let $A(1, 2)$, $B(5, 4)$, $C(6, 1)$, $D(2, -1)$. Show that $(AB)$ and
$(CD)$ are parallel, then that $ABCD$ is a parallelogram. Which
verification implies the other?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:vectors:9}
Let $ABC$ be a triangle, and let $I$ be the point defined by
$\vect{AI} = \frac23 \vect{AB}$. With $A(0,0)$, $B(6, 3)$, $C(2, 5)$
(so that computations stay simple):
\begin{enumerate}
  \item compute the \omterm{def:g10:coordgeom:system}{coordinates} of $I$;
  \item let $J$ be such that $\vect{CJ} = \frac23\vect{CB}$; compute the
        \omterm{def:g10:coordgeom:system}{coordinates} of $J$;
  \item show that $(IJ)$ is parallel to $(AC)$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g10:vectors:10}
Let $ABCD$ be any quadrilateral, and let $I$, $J$, $K$, $L$ be the
\omterm{prop:g10:coordgeom:midpoint}{midpoints} of $[AB]$, $[BC]$, $[CD]$, $[DA]$. Using \omterm{def:g10:coordgeom:system}{coordinates}
$A(x_A, y_A)$, etc., show that $\vect{IJ} = \vect{LK}$, and conclude that
the \omterm{prop:g10:coordgeom:midpoint}{midpoints} of the sides of \emph{any} quadrilateral form a
parallelogram.
\end{exercise}

\section{Problem: The algebra of arrows}

\begin{problem}[{Weekend problem --- Chasles gymnastics, the
centroid re-proved in two lines, and forces in equilibrium: what
vectors are really for}]
\label{pb:g10:vectors:1}
Grade 8 proved that the medians of a triangle meet at two thirds
of their length --- with an ingenious parallelogram hidden inside
the triangle (the centroid weekend problem of the Middle School volume). \omterm{def:g10:vectors:vector}{Vectors} reprove it in
two lines, and hand out the concurrency for free. That is what
this new algebra is for: theorems become computations with
arrows. This problem trains the computation (Chasles above all),
delivers the two-line proof, extends Varignon
(\cref{exo:g10:vectors:10}), and ends where \omterm{def:g10:vectors:vector}{vectors} were born:
forces and velocities.

\medskip
\noindent\textbf{Part I --- Chasles gymnastics.}
\begin{enumerate}
  \item Simplify using Chasles' relation
        (\cref{thm:g10:vectors:chasles}):
        $\vect{AB} + \vect{BC} + \vect{CD}$; \quad
        $\vect{MA} - \vect{MB}$; \quad
        $\vect{AB} + \vect{CD} + \vect{BC} + \vect{DA}$.
  \item The origin trick: show that for \emph{any} point $O$,
        $\vect{AB} = \vect{OB} - \vect{OA}$.
  \item Prove the two characterizations of the \omterm{prop:g10:coordgeom:midpoint}{midpoint} $I$ of
        $[AB]$:
        \[
        \vect{IA} + \vect{IB} = \vec 0
        \qquad\text{and}\qquad
        \vect{OI} = \tfrac12\left(\vect{OA} + \vect{OB}\right)
        \ \text{for any } O .
        \]
  \item Using $\vect{AB} = \vect{DC}$, re-find the fourth vertex
        $D$ of the parallelogram $ABCD$ with $A(-1, 2)$,
        $B(3, 4)$, $C(6, 0)$ --- and compare with the
        midpoint-trick answer of \cref{pb:g10:coordgeom:1}.
  \item Adding a fixed \omterm{def:g10:vectors:vector}{vector} $\vec u$ to every point of the
        plane performs which transformation --- the one
        discovered with two mirrors in the two-mirrors weekend problem of the Middle School volume
        and two half-turns in the half-turns weekend problem of the Middle School volume? Compute
        the \omterm{def:g10:functions:function}{image} of $(1, 2)$ under $\vec u = (3, -1)$.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The centroid, third proof.}
Define the point $G$ of a triangle $ABC$ by the elegant
condition
\[
\vect{GA} + \vect{GB} + \vect{GC} = \vec 0 .
\]
\begin{enumerate}[resume]
  \item Using the origin trick, show that this condition
        determines $G$ uniquely:
        $\vect{OG} = \frac13\left(\vect{OA} + \vect{OB} +
        \vect{OC}\right)$ for any $O$ --- so the \omterm{def:g10:coordgeom:system}{coordinates} of
        $G$ are the \emph{averages} of the vertices'
        (the centroid weekend problem of the Middle School volume saw it numerically).
  \item Prove $\vect{AG} = \frac13\left(\vect{AB} +
        \vect{AC}\right)$, and, writing $K$ for the \omterm{prop:g10:coordgeom:midpoint}{midpoint} of
        $[BC]$, $\vect{AK} = \frac12\left(\vect{AB} +
        \vect{AC}\right)$. Conclude:
        $\vect{AG} = \frac23\,\vect{AK}$ --- the two-thirds
        theorem, re-proved.
  \item Why do the medians from $B$ and from $C$ pass through
        the \emph{same} $G$ with no further computation?
        Compare the three available proofs of this theorem ---
        grade 8's hidden parallelogram, \omterm{def:g10:coordgeom:system}{coordinates}, \omterm{def:g10:vectors:vector}{vectors} ---
        in one sentence each.
  \item Numerical check: $A(1, 1)$, $B(5, 2)$, $C(3, 6)$.
        Compute $G$, then $K$, then verify
        $\vect{AG} = \frac23 \vect{AK}$ with the collinearity
        criterion (\cref{thm:g10:vectors:det}).
  \item Physics reads $G$ as the balance point of three equal
        masses at the vertices. Double the mass at $A$ (masses
        $2, 1, 1$): the balance point becomes
        $\vect{OG'} = \frac{2\vect{OA} + \vect{OB} +
        \vect{OC}}{4}$. Compute $G'$ for the triangle of
        question 9 and compare its position with $G$.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Collinearity at work.}
\begin{enumerate}[resume]
  \item Are $\vec u(3, -2)$ and $\vec v(-6, 4)$ \omterm{def:g10:vectors:collinear}{collinear}? And
        $\vec w(2, 5)$, $\vec z(4, 9)$? Decide with the
        determinant criterion.
  \item Are the points $A(1, 2)$, $B(3, 5)$, $C(7, 11)$
        aligned?
  \item Beyond Varignon (\cref{exo:g10:vectors:10}): in any
        quadrilateral $ABCD$, the \emph{bimedians} join the
        \omterm{prop:g10:coordgeom:midpoint}{midpoints} of opposite sides ($[IK]$ and $[JL]$). Using
        position \omterm{def:g10:vectors:vector}{vectors} ($\vect{OI} = \frac12(\vect{OA} +
        \vect{OB})$, etc.), show that both bimedians have the
        same \omterm{prop:g10:coordgeom:midpoint}{midpoint}, of position \omterm{def:g10:vectors:vector}{vector}
        $\frac14\left(\vect{OA} + \vect{OB} + \vect{OC} +
        \vect{OD}\right)$: the centroid of the quadrilateral.
  \item Thales in one line: $M$ and $N$ are the points defined
        by $\vect{AM} = \frac13\vect{AB}$ and
        $\vect{AN} = \frac13\vect{AC}$. Compute
        $\vect{MN}$ in terms of $\vect{BC}$ and conclude
        parallelism and ratio.
  \item Generalize question 14 to an arbitrary ratio $k$, and
        state in one sentence how the \omterm{def:g10:vectors:vector}{vector} calculus absorbs
        the \omterm{prop:g10:coordgeom:midpoint}{midpoint} theorem (the \omterm{prop:g10:coordgeom:midpoint}{midpoint} theorem)
        and Thales (Thales' theorem) at once.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Forces and velocities.}
\begin{enumerate}[resume]
  \item A boat is pushed by a current of $3$ km/h due east
        while rowing at $4$ km/h due north. Give the resultant
        velocity \omterm{def:g10:vectors:vector}{vector}, its magnitude, and describe the actual
        course. (A very old triple appears.)
  \item Three forces act on a ring: $\vec u(2, 1)$,
        $\vec v(-3, 2)$, $\vec w(1, -3)$. Compute the resultant.
        What does the ring do? And why is the \omterm{def:g10:vectors:sum}{sum} of the arrows
        along any \emph{closed} polygon always $\vec 0$
        (Chasles)?
  \item Two forces $\vec u(4, -1)$ and $\vec v(-1, 3)$ act on a
        hook. What third force holds the hook in equilibrium?
  \item A swimmer crosses a river heading straight across at
        $2$ m/s; the current runs at $1.5$ m/s. Give the
        resultant velocity and its magnitude. The river is
        $50$ m wide: how long does the crossing take, and how
        far downstream does the swimmer land?
  \item Finale: one object, three costumes --- displacement
        arrow, coordinate pair, force or velocity. List the
        three classical theorems this problem re-proved with
        \omterm{def:g10:vectors:vector}{vector} algebra, and name the one geometric quantity the
        \omterm{def:g10:vectors:vector}{vectors} of this chapter still cannot measure --- the
        tool that fixes it arrives with the scalar product, in
        grade 11.
\end{enumerate}
\end{problem}
