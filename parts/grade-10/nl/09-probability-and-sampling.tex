\chapter{Kansrekening en steekproeven}\label{ch:g10:proba}

Kansrekening kent getallen toe aan het toeval. Uitgaande van experimenten
met even waarschijnlijke uitkomsten --- dobbelstenen, munten, kaarten ---
bouwt dit hoofdstuk de basisregels op voor het berekenen van kansen van
gebeurtenissen, introduceert boomschema's voor tweestapsexperimenten, en
eindigt met de link tussen kansen en waargenomen frequenties. De theorie
gaat verder in \cref{ch:g11:prob}.

\section{Toevalsexperimenten en gebeurtenissen}

\begin{definition}[Uitkomsten en gebeurtenissen]\label{def:g10:proba:events}
Een \emph{toevalsexperiment}\index{toevalsexperiment} is een experiment
waarvan het resultaat niet voorspelbaar is (een dobbelsteen gooien). De
verzameling van alle mogelijke resultaten is de
\emph{uitkomstenruimte}\index{uitkomstenruimte}
$\Omega = \{\omega_1, \dots, \omega_n\}$; haar elementen zijn de
\emph{uitkomsten}. Een \emph{gebeurtenis}\index{gebeurtenis} is een
verzameling uitkomsten; een gebeurtenis \emph{treedt op} wanneer het
resultaat van het experiment ertoe behoort.
\end{definition}

\begin{definition}[Kansverdeling]\label{def:g10:proba:distribution}
Aan elke uitkomst $\omega_i$ een getal $p_i \geq 0$ toekennen, met
$p_1 + \dots + p_n = 1$, definieert een \emph{kans}\index{kans}: de
kans $\P(A)$ van een gebeurtenis $A$ is de som van de $p_i$ van de
uitkomsten in $A$. In het bijzonder $\P(\Omega) = 1$ en de onmogelijke
gebeurtenis $\varnothing$ heeft kans $0$.
\end{definition}

\begin{proposition}[Even waarschijnlijke uitkomsten]\label{prop:g10:proba:uniform}
Als de $n$ uitkomsten even waarschijnlijk zijn (elke $p_i = \frac1n$),
dan voor elke gebeurtenis $A$:
\[
\P(A) = \frac{\text{aantal uitkomsten in } A}{n}
= \frac{\text{gunstige gevallen}}{\text{mogelijke gevallen}} .
\]
\end{proposition}

\begin{proof}
$A$ bevat een zeker aantal $k$ uitkomsten, elk met kans $\frac1n$, dus
$\P(A) = \frac kn$.
\end{proof}

\begin{example}\label{ex:g10:proba:die}
Gooi een eerlijke dobbelsteen: $\Omega = \{1, 2, 3, 4, 5, 6\}$, elke
uitkomst met kans $\frac16$. De gebeurtenis $A$: ``het resultaat is even''
is $A = \{2, 4, 6\}$, dus $\P(A) = \frac36 = \frac12$. De gebeurtenis
$B$: ``het resultaat is minstens $5$'' is $B = \{5, 6\}$, dus
$\P(B) = \frac26 = \frac13$.
\end{example}

\section{Gebeurtenissen combineren}

\begin{definition}[Doorsnede, vereniging, complement]\label{def:g10:proba:operations}
Laat $A$ en $B$ gebeurtenissen zijn.
\begin{itemize}
  \item $A \cap B$ (``$A$ en $B$'') treedt op wanneer beide optreden;
  \item $A \cup B$ (``$A$ of $B$'') treedt op wanneer minstens één
        optreedt;
  \item het \emph{complement}\index{complement} $\bar A$ (``niet $A$'')
        treedt precies op wanneer $A$ dat niet doet;
  \item $A$ en $B$ zijn \emph{onverenigbaar} wanneer ze niet samen
        kunnen optreden: $A \cap B = \varnothing$.
\end{itemize}
\end{definition}

\begin{theorem}[Optelregels]\label{thm:g10:proba:addition}
Voor alle gebeurtenissen $A$ en $B$:
\[
\P(A \cup B) = \P(A) + \P(B) - \P(A \cap B),
\qquad
\P(\bar A) = 1 - \P(A) .
\]
Voor onverenigbare gebeurtenissen vereenvoudigt de eerste regel tot
$\P(A \cup B) = \P(A) + \P(B)$.
\end{theorem}

\begin{proof}
Optellen van $\P(A) + \P(B)$ telt de kans van elke uitkomst van
$A \cup B$ één keer, \emph{behalve} die van $A \cap B$, die twee keer
geteld worden --- één in $A$ en één in $B$. Aftrekken van
$\P(A \cap B)$ herstelt de dubbeltelling. Voor het complement: $A$ en
$\bar A$ zijn onverenigbaar en $A \cup \bar A = \Omega$, dus
$\P(A) + \P(\bar A) = \P(\Omega) = 1$.
\end{proof}

\begin{example}\label{ex:g10:proba:cards}
Trek één kaart uit een standaard spels van $52$ kaarten. Laat $A$: ``de
kaart is een harten'' ($13$ kaarten) en $B$: ``de kaart is een heer''
($4$ kaarten). Dan is $A \cap B$ ``de hartenheer'' ($1$ kaart), en
\[
\P(A \cup B) = \frac{13}{52} + \frac{4}{52} - \frac{1}{52} = \frac{16}{52}
= \frac{4}{13}.
\]
De complementregel is vaak de snelste route: de kans dat de kaart
\emph{geen} harten is, is $1 - \frac{13}{52} = \frac34$.
\end{example}

\section{Boomschema's}

\begin{method}[Tweestapsexperimenten]\label{met:g10:proba:tree}
Voor een experiment in twee stappen:
\begin{enumerate}
  \item teken een \emph{boom}\index{boomschema}: één tak per mogelijk
        resultaat van de eerste stap, daarna vanaf elk daarvan één tak
        per resultaat van de tweede stap, met de kans op elke tak (de
        takken die een knoop verlaten moeten tot $1$ sommen);
  \item de kans van een \emph{pad} (een blad van de boom) is het product
        van de kansen langs zijn takken;
  \item de kans van een gebeurtenis is de som van de kansen van de paden
        die haar realiseren.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g10:proba:tree}
Een urn bevat $3$ rode ballen en $2$ blauwe ballen. Trek één bal, leg
haar terug, trek opnieuw. Elke trekking geeft rood (R) met kans
$\frac35$ en blauw (B) met kans $\frac25$.

sommen tot $1$.}

De kans om twee ballen van dezelfde kleur te krijgen is
$\P(\text{RR}) + \P(\text{BB}) = \frac{9}{25} + \frac{4}{25} =
\frac{13}{25}$; de kans op minstens één blauw is
$1 - \P(\text{RR}) = 1 - \frac{9}{25} = \frac{16}{25}$ (complementregel).
\end{example}

\begin{remark}
Zonder teruglegging hangen de kansen van de tweede trekking af van de
eerste: met de urn hierboven, na het trekken van een rode bal, blijven
nog $2$ rood en $2$ blauw, dus de tweede tak draagt $\frac24$ en
$\frac24$. De boommethode werkt ongewijzigd; alleen de getallen op de
takken van het tweede niveau veranderen.
\end{remark}

\section{Frequenties en steekproeven}

Wanneer een toevalsexperiment vele malen herhaald wordt, nadert de
waargenomen frequentie van een gebeurtenis haar kans --- daarom is
kansrekening nuttig om de echte wereld te beschrijven.

\begin{definition}[Steekproef]\label{def:g10:proba:sample}
Een \emph{steekproef}\index{steekproef} van grootte $n$ is het resultaat
van $n$ onafhankelijke herhalingen van hetzelfde experiment. De
\emph{waargenomen frequentie} van een gebeurtenis in de steekproef is het
aantal optredens gedeeld door $n$.
\end{definition}

\begin{example}\label{ex:g10:proba:sampling}
Een eerlijke munt geeft kop met kans $0.5$. Haar $100$ keer opgooien
geeft zelden precies $50$ keer kop: steekproeven \emph{fluctueren}.
Typische reeksen van $100$ worpen geven frequenties zoals $0.46$, $0.53$,
$0.49$ --- dicht bij $0.5$, maar niet gelijk. Grotere steekproeven
fluctueren minder: met $10\,000$ worpen liggen waargenomen frequenties
meestal binnen ongeveer $0.01$ van $0.5$.
\end{example}

\begin{remark}[Fluctuatie-interval]
Een nuttige vuistregel, gerechtvaardigd in \cref{ch:g11:binom}: voor een
steekproef van grootte $n$ van een gebeurtenis met kans $p$ (met $n$
redelijk groot), ligt de waargenomen frequentie in het interval
\[
\intcc{p - \frac{1}{\sqrt n}}{\,p + \frac{1}{\sqrt n}}
\]
in ongeveer $95\%$ van de steekproeven. Als een waargenomen frequentie
ver buiten dit interval valt, twijfel aan de waarde van $p$: dit is het
startpunt van statistisch toetsen.
\end{remark}

\begin{example}\label{ex:g10:proba:test}
Een fabriek beweert dat slechts $10\%$ van haar onderdelen defect is
($p = 0.1$). In een steekproef van $n = 400$ onderdelen zijn er $64$
defect: waargenomen frequentie $\frac{64}{400} = 0.16$. Het
fluctuatie-interval is
$\intcc{0.1 - \frac{1}{20}}{0.1 + \frac{1}{20}} = \intcc{0.05}{0.15}$,
en $0.16$ valt erbuiten: de steekproef twijfelt serieus aan de claim.
\end{example}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g10:proba:1}
Een eerlijke dobbelsteen wordt gegooid. Bereken de kans van elke
gebeurtenis: ``een $6$ gooien''; ``een oneven getal gooien''; ``ten
hoogste $4$ gooien''; ``een $7$ gooien''.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:proba:2}
Een zak bevat $5$ groene, $3$ gele en $2$ zwarte knikkers; één knikker
wordt willekeurig getrokken. Bereken de kans dat ze groen is; dat ze niet
zwart is; dat ze groen of geel is.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:proba:3}
In een klas van $30$ leerlingen studeren $18$ Spaans, $10$ Duits, en
$4$ beide. Een leerling wordt willekeurig gekozen. Bereken met de
optelregel de kans dat deze leerling minstens één van de twee talen
studeert, daarna de kans dat hij of zij geen van beide studeert.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:proba:4}
Een eerlijke munt wordt twee keer opgegooid. Teken de boom van het
experiment en bereken de kans op twee keer kop; precies één keer kop;
minstens één keer kop.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:proba:5}
Een urn bevat $4$ rode en $6$ witte ballen. Twee ballen worden
achtereenvolgens getrokken \emph{zonder} teruglegging. Teken de boom met
de juiste kansen op het tweede niveau, en bereken de kans op twee rode
ballen, daarna op twee ballen van verschillende kleuren.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:proba:6}
Twee eerlijke dobbelstenen worden gegooid en hun som genoteerd.
\begin{enumerate}
  \item Leg uit waarom de uitkomstenruimte genomen kan worden als de
        $36$ even waarschijnlijke paren $(1,1), (1,2), \dots, (6,6)$.
  \item Bereken de kans dat de som $7$ is; dat ze $12$ is; dat ze ten
        hoogste $4$ is.
  \item Welke som is het meest waarschijnlijk?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:proba:7}
Een meerkeuzevraag biedt $4$ antwoorden, waarvan één correct. Een
leerling beantwoordt $2$ onafhankelijke zulke vragen volledig
willekeurig. Bereken de kans om beide correct te beantwoorden, precies
één correct, geen correct. Controleer dat de drie kansen tot $1$ sommen.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:proba:8}
Een politicus claimt $50\%$ goedkeuring. In een peiling van $n = 900$
willekeurig gekozen kiezers keuren $396$ goed.
\begin{enumerate}
  \item Bereken de waargenomen frequentie en het fluctuatie-interval
        rond $p = 0.5$ voor $n = 900$.
  \item Twijfelt de peiling aan de claim?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g10:proba:9}
Een spel kost $2$ om te spelen: je gooit een eerlijke dobbelsteen en
wint $6$ bij een $6$, $3$ bij een $5$, niets anders.
\begin{enumerate}
  \item Bereken de kans van elk gewonnen bedrag ($6$, $3$, $0$).
  \item Over $600$ spellen, hoeveel van elke uitkomst verwacht je
        ongeveer? Bereken de totale verwachte winst en het verlies: is
        het spel de moeite waard?
\end{enumerate}
\end{exercise}
