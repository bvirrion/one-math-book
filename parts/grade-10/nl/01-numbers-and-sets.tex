\chapter{Getallen en verzamelingen van getallen}\label{ch:g10:numbers}

Wiskunde begint bij getallen, en niet alle getallen zijn van dezelfde
soort: telgetallen, negatieve getallen, breuken, en getallen zoals
$\sqrt 2$ of $\pi$ die geen enkele breuk kan uitdrukken. Dit hoofdstuk
ordent ze in geneste families, introduceert intervallen om stukken van de
getallenlijn te beschrijven, en gebruikt de \omterm{def:g10:numbers:abs}{absolute waarde} om afstanden
tussen getallen te meten.

\section{De families van getallen}

\begin{definition}[Verzamelingen van getallen]\label{def:g10:numbers:sets}
\begin{itemize}
  \item $\N$ is de verzameling van de \emph{natuurlijke
        getallen}\index{natuurlijk getal}: $0, 1, 2, 3, \dots$
  \item $\Z$ is de verzameling van de \emph{gehele
        getallen}\index{geheel getal}: $\dots, -2, -1, 0, 1, 2, \dots$
  \item $\Q$ is de verzameling van de \emph{rationale
        getallen}\index{rationaal getal}: alle quotiënten $\frac{p}{q}$
        met $p \in \Z$, $q \in \N$ en $q \neq 0$.
  \item $\R$ is de verzameling van de \emph{reële
        getallen}\index{reëel getal}: alle getallen die op de
        getallenlijn geplaatst kunnen worden.
\end{itemize}
\end{definition}

\begin{notation}
Het symbool $\in$ leest men als ``behoort tot'': $3 \in \N$, $-5 \in \Z$,
$\frac23 \in \Q$. Het symbool $\subset$ leest men als ``is bevat in'':
elk \omterm{def:g10:numbers:sets}{natuurlijk getal} is een \omterm{def:g10:numbers:sets}{geheel getal}, elk \omterm{def:g10:numbers:sets}{geheel getal} is een
\omterm{def:g10:numbers:sets}{rationaal getal} (bijv.\ $-5 = \frac{-5}{1}$), en elk \omterm{def:g10:numbers:sets}{rationaal getal} is
reëel, dus
\[
\N \subset \Z \subset \Q \subset \R .
\]
Een schuine streep ontkent een symbool: $\frac12 \notin \Z$.
\end{notation}

\begin{example}\label{ex:g10:numbers:classify}
Laten we enkele getallen plaatsen in de kleinste familie die ze bevat.
\begin{itemize}
  \item $\frac{28}{4} = 7$, dus $\frac{28}{4} \in \N$ ook al is het als
        breuk geschreven: altijd eerst vereenvoudigen.
  \item $-3.5 = -\frac{35}{10} = -\frac72$ is rationaal maar geen \omterm{def:g10:numbers:sets}{geheel
        getal}.
  \item $0.333\dots$ (het cijfer $3$ herhaalt zich oneindig) is gelijk aan
        $\frac13$, een \omterm{def:g10:numbers:sets}{rationaal getal}.
  \item $\sqrt 2$ en $\pi$ zijn reëel maar niet rationaal, zoals we zo
        dadelijk voor $\sqrt 2$ zullen zien; zulke getallen heten
        \emph{irrationaal}\index{irrationaal getal}.
\end{itemize}
\end{example}

\begin{theorem}[Irrationaliteit van $\sqrt 2$]\label{thm:g10:numbers:sqrt2}
Het getal $\sqrt 2$ is niet rationaal: geen enkele breuk van \omterm{def:g10:numbers:sets}{gehele
getallen} heeft kwadraat $2$.
\end{theorem}

\begin{proof}
We redeneren uit het ongerijmde, in kleine stappen.
\begin{enumerate}
  \item Stel $\sqrt 2 = \frac{p}{q}$ waarbij $p$ en $q$ positieve \omterm{def:g10:numbers:sets}{gehele
        getallen} zijn, en de breuk volledig vereenvoudigd is, zodat $p$ en
        $q$ niet beide even zijn.
  \item Kwadrateren van beide kanten geeft $2 = \frac{p^2}{q^2}$, dat wil
        zeggen $p^2 = 2q^2$. Dus $p^2$ is even.
  \item Als $p$ oneven was, zeg $p = 2k+1$, dan zou
        $p^2 = 4k^2 + 4k + 1$ oneven zijn. Omdat $p^2$ even is, moet $p$
        even zijn: $p = 2k$ voor een zeker \omterm{def:g10:numbers:sets}{geheel getal} $k$.
  \item Substitueren: $(2k)^2 = 2q^2$, dus $4k^2 = 2q^2$, dus
        $q^2 = 2k^2$. Met hetzelfde argument als in stap 3 is $q$ even.
  \item Nu zijn $p$ en $q$ beide even, in strijd met stap 1. De aanname
        was onmogelijk: $\sqrt 2$ is \omterm{ex:g10:numbers:classify}{irrationaal}.
\end{enumerate}
\end{proof}

\begin{remark}
\omterm{def:g10:numbers:sets}{Rationale getallen} zijn precies de getallen waarvan de decimale
ontwikkeling ofwel stopt (zoals $\frac72 = 3.5$), ofwel uiteindelijk
steeds hetzelfde blok herhaalt (zoals $\frac13 = 0.333\dots$ of
$\frac{1}{7} = 0.142857\,142857\dots$). Irrationale getallen zoals
$\sqrt2 = 1.41421356\dots$ hebben decimale cijfers die nooit in een
herhalend patroon vallen. We geven deze karakterisering op dit niveau toe.
\end{remark}

\section{De getallenlijn en intervallen}

\omterm{def:g10:numbers:sets}{Reële getallen} vullen een lijn: door een oorsprong $0$ en een
lengte-eenheid te kiezen, komt elk \omterm{def:g10:numbers:sets}{reëel getal} overeen met precies één
punt.

\begin{definition}[Interval]\label{def:g10:numbers:interval}
Laat $a$ en $b$ \omterm{def:g10:numbers:sets}{reële getallen} zijn met $a < b$. Een
\emph{interval}\index{interval} is de verzameling van alle \omterm{def:g10:numbers:sets}{reële getallen}
tussen twee grenzen. Een rechte haak betekent dat de grens is inbegrepen,
een ronde haak betekent dat ze is uitgesloten:
\begin{center}
\begin{tabular}{c|c}
notatie & beschrijving \\
\hline
$\intcc{a}{b}$ & $a \leq x \leq b$ (beide einden inbegrepen) \\
$\intoo{a}{b}$ & $a < x < b$ (beide einden uitgesloten) \\
$\intco{a}{b}$ & $a \leq x < b$ \\
$\intoc{a}{b}$ & $a < x \leq b$ \\
$\intco{a}{+\infty}$ & $x \geq a$ \\
$\intoo{-\infty}{b}$ & $x < b$ \\
\end{tabular}
\end{center}
De symbolen $-\infty$ en $+\infty$ (``oneindig'') zijn geen getallen,
alleen een manier om te zeggen dat het interval oneindig doorloopt; de
haak ernaast is altijd rond. De hele lijn $\R$ is het interval
$\intoo{-\infty}{+\infty}$.
\end{definition}

\begin{definition}[Doorsnede en vereniging]\label{def:g10:numbers:interunion}
Laat $I$ en $J$ twee verzamelingen van \omterm{def:g10:numbers:sets}{reële getallen} zijn.
De \emph{doorsnede}\index{doorsnede} $I \cap J$ (``$I$ en $J$'') is de
verzameling van getallen die tot beide behoren; de
\emph{vereniging}\index{vereniging} $I \cup J$ (``$I$ of $J$'') is de
verzameling van getallen die tot minstens één van beide behoren.
\end{definition}

\begin{example}\label{ex:g10:numbers:interunion}
Neem $I = \intcc{-1}{3}$ en $J = \intoo{1}{5}$. Teken beide op dezelfde
lijn: ze overlappen tussen $1$ en $3$. Daarom
\[
I \cap J = \intoc{1}{3},
\qquad
I \cup J = \intco{-1}{5}.
\]
Let op de haken: $1 \notin J$ dus $1 \notin I \cap J$, maar $3 \in I$ en
$3 \in J$, dus $3 \in I \cap J$.
\end{example}

\begin{method}[Werken met intervallen]\label{met:g10:numbers:intervals}
Om de \omterm{def:g10:numbers:interunion}{doorsnede} of \omterm{def:g10:numbers:interunion}{vereniging} van twee intervallen te vinden:
\begin{enumerate}
  \item teken de getallenlijn en markeer de vier grenzen;
  \item arceer het eerste \omterm{def:g10:numbers:interval}{interval} boven de lijn en het tweede eronder;
  \item de \omterm{def:g10:numbers:interunion}{doorsnede} is waar de arceringen overlappen, de \omterm{def:g10:numbers:interunion}{vereniging} is
        waar minstens één arcering aanwezig is;
  \item beslis elke haak door te testen of de grens zelf tot beide
        verzamelingen behoort (\omterm{def:g10:numbers:interunion}{doorsnede}) of tot minstens één (\omterm{def:g10:numbers:interunion}{vereniging}).
\end{enumerate}
\end{method}

\section{Absolute waarde en afstand}

\begin{definition}[Absolute waarde]\label{def:g10:numbers:abs}
De \emph{absolute waarde}\index{absolute waarde} van een \omterm{def:g10:numbers:sets}{reëel getal} $x$
is
\[
\abs{x} =
\begin{cases}
x & \text{als } x \geq 0, \\
-x & \text{als } x < 0.
\end{cases}
\]
Bijvoorbeeld $\abs{7} = 7$ en $\abs{-4} = 4$. In alle gevallen is
$\abs{x} \geq 0$.
\end{definition}

\begin{proposition}[Afstand op de lijn]\label{prop:g10:numbers:distance}
Voor alle \omterm{def:g10:numbers:sets}{reële getallen} $a$ en $b$ is de afstand tussen de punten $a$ en
$b$ op de getallenlijn gelijk aan $\abs{b - a}$. In het bijzonder is
$\abs{x}$ de afstand van $x$ tot $0$.
\end{proposition}

\begin{proof}
Als $b \geq a$, is de afstand van $a$ tot $b$ gelijk aan $b - a \geq 0$,
wat $\abs{b-a}$ is. Als $b < a$, is de afstand $a - b = -(b - a) > 0$,
wat opnieuw $\abs{b - a}$ is volgens de definitie van de \omterm{def:g10:numbers:abs}{absolute
waarde}.
\end{proof}

\begin{proposition}[Absolute waarde en intervallen]\label{prop:g10:numbers:absinterval}
Laat $a$ een \omterm{def:g10:numbers:sets}{reëel getal} zijn en $r > 0$. Dan
\[
\abs{x - a} \leq r
\quad\text{precies wanneer}\quad
x \in \intcc{a - r}{a + r}.
\]
\end{proposition}

\begin{proof}
$\abs{x-a} \leq r$ zegt dat de afstand van $x$ tot $a$ ten hoogste $r$
is, d.w.z.\ dat $x$ niet verder dan $r$ van $a$ ligt aan weerszijden. De
getallen die hieraan voldoen zijn precies die tussen $a - r$ en $a + r$,
grenzen inbegrepen.
\end{proof}

\begin{example}\label{ex:g10:numbers:abs}
Los $\abs{x - 3} \leq 2$ op. De oplossingen zijn de getallen op afstand
ten hoogste $2$ van $3$: het \omterm{def:g10:numbers:interval}{interval} $\intcc{1}{5}$. Omgekeerd heeft het
\omterm{def:g10:numbers:interval}{interval} $\intcc{-1}{7}$ middelpunt $\frac{-1+7}{2} = 3$ en straal
$\frac{7-(-1)}{2} = 4$, dus het wordt beschreven door
$\abs{x - 3} \leq 4$.
\end{example}

\section{Benaderingen}

Irrationale getallen, en zelfs de meeste breuken, kunnen niet exact met
eindig veel decimalen geschreven worden, dus benaderen we ze in de
praktijk.

\begin{definition}[Benadering tot een gegeven nauwkeurigheid]\label{def:g10:numbers:approx}
Een getal $d$ is een \emph{benadering}\index{benadering} van $x$ tot op
$10^{-n}$ wanneer $\abs{x - d} \leq 10^{-n}$. Afkappen of afronden van de
decimale ontwikkeling na het $n$-de cijfer geeft beide zulke
benaderingen.
\end{definition}

\begin{example}
Uit $\pi = 3.14159\,26\dots$: de afkapping $3.141$ en de afronding
$3.142$ zijn beide \omterm{def:g10:numbers:approx}{benaderingen} van $\pi$ tot op $10^{-3}$. De afronding
ligt op afstand kleiner dan $\frac12 \times 10^{-3}$, de afkapping
garandeert alleen $10^{-3}$. Schrijven $3.141 \leq \pi \leq 3.142$
\emph{omkadert} $\pi$ tussen twee decimale grenzen.
\end{example}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g10:numbers:1}
Geef voor elk getal de kleinste van de verzamelingen $\N$, $\Z$, $\Q$,
$\R$ waartoe het behoort:
\[
\frac{15}{3}, \qquad -7, \qquad \frac{22}{7}, \qquad \sqrt{9}, \qquad
\sqrt{10}, \qquad -2.4, \qquad 0 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:numbers:2}
Schrijf elke uitspraak in intervalnotatie en teken die op een
getallenlijn: (a) $-2 \leq x < 5$; (b) $x > 3$; (c) $x \leq -1$;
(d) de afstand van $x$ tot $2$ is ten hoogste $3$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:numbers:3}
Bereken $I \cap J$ en $I \cup J$ voor
\[
\text{(a) } I = \intcc{-3}{2},\ J = \intco{0}{4};
\qquad
\text{(b) } I = \intoo{-\infty}{1},\ J = \intco{-2}{+\infty}.
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:numbers:4}
Bereken zonder rekenmachine:
\[
\abs{-6}, \qquad \abs{4 - 9}, \qquad \abs{-3 - 5}, \qquad
\abs{\sqrt 2 - 1}, \qquad \abs{1 - \sqrt 2}.
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:numbers:5}
Los de vergelijkingen en ongelijkheden op en geef de
oplossingsverzamelingen:
\[
\abs{x} = 5, \qquad \abs{x - 1} = 3, \qquad \abs{x - 4} \leq 1, \qquad
\abs{x + 2} < 3 .
\]
(Merk op dat $\abs{x+2} = \abs{x - (-2)}$ een afstand tot $-2$ is.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:numbers:6}
Beschrijf elk \omterm{def:g10:numbers:interval}{interval} door een ongelijkheid van de vorm
$\abs{x - a} \leq r$ of $\abs{x - a} < r$:
\[
\intcc{2}{8}, \qquad \intoo{-5}{1}, \qquad \intcc{-7}{-3}.
\]
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:numbers:7}
Toon aan dat $0.272727\dots$ (het blok $27$ herhaalt zich oneindig)
rationaal is. (Hint: noem het $x$ en bereken $100x - x$.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:numbers:8}
Waar of onwaar? Rechtvaardig elk antwoord met een argument of een
tegenvoorbeeld.
\begin{enumerate}
  \item De som van twee \omterm{def:g10:numbers:sets}{gehele getallen} is een \omterm{def:g10:numbers:sets}{geheel getal}.
  \item Het quotiënt van twee \omterm{def:g10:numbers:sets}{gehele getallen} is een \omterm{def:g10:numbers:sets}{geheel getal}.
  \item De som van twee \omterm{def:g10:numbers:sets}{rationale getallen} is rationaal.
  \item De som van een \omterm{def:g10:numbers:sets}{rationaal getal} en een \omterm{ex:g10:numbers:classify}{irrationaal getal} is
        \omterm{ex:g10:numbers:classify}{irrationaal}.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:numbers:9}
Gebruik $1.414 \leq \sqrt 2 \leq 1.415$ om de getallen $2\sqrt2$,
$\sqrt2 + 3$ en $-\sqrt2$ tussen twee decimale grenzen te omkaderen.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g10:numbers:10}
Pas het bewijs van \cref{thm:g10:numbers:sqrt2} aan om te tonen dat
$\sqrt 3$ \omterm{ex:g10:numbers:classify}{irrationaal} is. (Vervang ``even'' door ``veelvoud van $3$'':
controleer eerst dat als $p^2$ een veelvoud van $3$ is, dan ook $p$, door
de resten $0$, $1$, $2$ van $p$ bij deling door $3$ te bekijken.)
\end{exercise}
