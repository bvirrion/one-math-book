\chapter{Vectoren}\label{ch:g10:vectors}

Een \omterm{def:g10:vectors:vector}{vector} codeert een verplaatsing: een richting en een lengte,
ongeacht het beginpunt. \omterm{def:g10:vectors:vector}{Vectoren} geven schone bewijzen van meetkundige
feiten en, zodra \omterm{def:g10:coordgeom:system}{coördinaten} in \omterm{def:g10:functions:function}{beeld} komen, herleiden die tot kleine
berekeningen. Ze zijn een sleutelinstrument in \cref{ch:g10:lines} en,
later, voor het inwendig product in \cref{ch:g11:scal}.

\section{Translaties en vectoren}

\begin{definition}[Vector]\label{def:g10:vectors:vector}
Gegeven twee punten $A$ en $B$, stelt de \emph{vector}\index{vector}
$\vect{AB}$ de verplaatsing van $A$ naar $B$ voor: hij heeft een
\emph{richting} (de richting van de lijn $(AB)$, samen met de zin van
bewegen, van $A$ naar $B$) en een \emph{lengte} (de afstand $AB$, ook
genoteerd $\norm{\vect{AB}}$). Twee vectoren zijn \emph{gelijk} wanneer
ze dezelfde verplaatsing coderen: $\vect{AB} = \vect{CD}$ betekent dat
het toepassen van de twee verplaatsingen op enig punt hetzelfde resultaat
geeft.
\end{definition}

\begin{proposition}[Gelijke vectoren en parallellogrammen]\label{prop:g10:vectors:parallelogram}
$\vect{AB} = \vect{CD}$ precies wanneer $ABDC$ (in deze volgorde!) een
parallellogram is, eventueel afgeplat --- equivalent, wanneer $[AD]$ en
$[BC]$ hetzelfde \omterm{prop:g10:coordgeom:midpoint}{midden} hebben.
\end{proposition}
\admitted

\begin{notation}
Een \omterm{def:g10:vectors:vector}{vector} wordt vaak met één letter genoemd, $\vec u$ of $\vec v$,
wanneer de eindpunten er niet toe doen. De \emph{nulvector}
$\vec 0 = \vect{AA}$ is de verplaatsing die niets verplaatst. De
\emph{tegengestelde} van $\vect{AB}$ is $\vect{BA}$: zelfde lengte,
tegengestelde zin. We schrijven $\vect{BA} = -\vect{AB}$.
\end{notation}

\section{Vectoren optellen}

\begin{definition}[Som van twee vectoren]\label{def:g10:vectors:sum}
De \emph{som}\index{vector!som} $\vec u + \vec v$ is de verplaatsing die
men krijgt door eerst $\vec u$ en dan $\vec v$ uit te voeren.
\end{definition}

\begin{theorem}[Relatie van Chasles]\label{thm:g10:vectors:chasles}
Voor willekeurige drie punten $A$, $B$, $C$:
\[
\vect{AB} + \vect{BC} = \vect{AC}.
\]
\end{theorem}

\begin{proof}
Van $A$ naar $B$ gaan, daarna van $B$ naar $C$, is een verplaatsing die
$A$ naar $C$ brengt: het is de verplaatsing gecodeerd door $\vect{AC}$.
\end{proof}

\begin{remark}[Parallellogramregel]
Om twee \omterm{def:g10:vectors:vector}{vectoren} vanuit hetzelfde punt op te tellen, $\vect{AB} +
\vect{AC}$, voltooi het parallellogram $ABDC$: de \omterm{def:g10:vectors:sum}{som} is de diagonaal
$\vect{AD}$. Inderdaad $\vect{AC} = \vect{BD}$, dus
$\vect{AB} + \vect{AC} = \vect{AB} + \vect{BD} = \vect{AD}$ volgens de
relatie van Chasles.
\end{remark}

\begin{example}[Vereenvoudigen met Chasles]\label{ex:g10:vectors:chasles}
Vereenvoudig $\vect{MN} + \vect{NP} + \vect{PQ}$: de verplaatsingen
aaneenschakelen geeft $\vect{MQ}$. Vereenvoudig $\vect{AB} - \vect{AC}$:
herschrijf het verschil als \omterm{def:g10:vectors:sum}{som} met de tegengestelde \omterm{def:g10:vectors:vector}{vector},
\[
\vect{AB} - \vect{AC} = \vect{AB} + \vect{CA}
= \vect{CA} + \vect{AB} = \vect{CB}.
\]
\end{example}

\section{Een vector vermenigvuldigen met een getal}

\begin{definition}[Scalaïr veelvoud]\label{def:g10:vectors:scalar}
Laat $\vec u$ een niet-nul \omterm{def:g10:vectors:vector}{vector} zijn en $k$ een \omterm{def:g10:numbers:sets}{reëel getal}. De \omterm{def:g10:vectors:vector}{vector}
$k\vec u$\index{vector!scalaïr veelvoud} heeft dezelfde richting als
$\vec u$, lengte $\abs{k} \times \norm{\vec u}$, en wijst dezelfde zin
als $\vec u$ als $k > 0$, de tegengestelde zin als $k < 0$. Voor $k = 0$
is $0\vec u = \vec 0$.
\end{definition}

\begin{definition}[Collineariteit]\label{def:g10:vectors:collinear}
Twee \omterm{def:g10:vectors:vector}{vectoren} $\vec u$ en $\vec v$ zijn
\emph{collineair}\index{collineaire vectoren} wanneer ze dezelfde
richting hebben, d.w.z.\ wanneer $\vec v = k \vec u$ voor een zeker
reëel $k$ (of één van hen $\vec 0$ is).
\end{definition}

\begin{remark}
Collineariteit is de vectorentaal voor evenwijdigheid en collineariteit
van punten:
\begin{itemize}
  \item de lijnen $(AB)$ en $(CD)$ zijn evenwijdig precies wanneer
        $\vect{AB}$ en $\vect{CD}$ \omterm{def:g10:vectors:collinear}{collineair} zijn;
  \item de punten $A$, $B$, $C$ zijn \omterm{def:g10:vectors:collinear}{collineair} precies wanneer
        $\vect{AB}$ en $\vect{AC}$ \omterm{def:g10:vectors:collinear}{collineair} zijn.
\end{itemize}
\end{remark}

\section{Coördinaten van vectoren}

\begin{definition}[Coördinaten van een vector]\label{def:g10:vectors:coords}
In een \omterm{def:g10:coordgeom:system}{coördinatenstelsel} zijn de \emph{\omterm{def:g10:coordgeom:system}{coördinaten}} van de \omterm{def:g10:vectors:vector}{vector}
$\vect{AB}$ de getallen die de verplaatsing beschrijven:
\[
\vect{AB}\,(x_B - x_A,\ y_B - y_A).
\]
Een \omterm{def:g10:vectors:vector}{vector} $\vec u\,(a, b)$ verplaatst elk punt $a$ eenheden horizontaal
en $b$ eenheden verticaal.
\end{definition}

\begin{proposition}[Rekenen met coördinaten]\label{prop:g10:vectors:coordrules}
Laat $\vec u\,(a, b)$ en $\vec v\,(c, d)$, en $k \in \R$.
\begin{enumerate}
  \item $\vec u = \vec v$ precies wanneer $a = c$ en $b = d$;
  \item $\vec u + \vec v$ heeft \omterm{def:g10:coordgeom:system}{coördinaten} $(a + c,\ b + d)$;
  \item $k\vec u$ heeft \omterm{def:g10:coordgeom:system}{coördinaten} $(ka,\ kb)$;
  \item $\norm{\vec u} = \sqrt{a^2 + b^2}$ (\omterm{def:g10:coordgeom:system}{orthonormaal stelsel}).
\end{enumerate}
\end{proposition}

\begin{proof}
1--3 herformuleren, coördinaat voor coördinaat, wat de verplaatsingen
doen: bijvoorbeeld $\vec u$ dan $\vec v$ uitvoeren verplaatst een punt
horizontaal met $a$ dan $c$, dus in totaal met $a + c$. Punt 4 is de
afstandsformule van \cref{thm:g10:coordgeom:distance} toegepast op een
lijnstuk dat $\vec u$ voorstelt.
\end{proof}

\begin{theorem}[Collineariteitscriterium]\label{thm:g10:vectors:det}
Twee \omterm{def:g10:vectors:vector}{vectoren} $\vec u\,(a, b)$ en $\vec v\,(c, d)$ zijn \omterm{def:g10:vectors:collinear}{collineair} dan en
slechts dan als
\[
ad - bc = 0 .
\]
\end{theorem}

\begin{proof}
Als $\vec v = k\vec u$, dan $c = ka$ en $d = kb$, dus
$ad - bc = a(kb) - b(ka) = 0$. Hetzelfde geldt als $\vec u = k \vec v$ of
als één van de \omterm{def:g10:vectors:vector}{vectoren} nul is.

Omgekeerd, stel $ad - bc = 0$ met $\vec u \neq \vec 0$, zeg $a \neq 0$
(het geval $b \neq 0$ is analoog). Stel $k = \frac{c}{a}$; dan $c = ka$,
en $ad = bc = b(ka)$ geeft, na delen door $a \neq 0$, $d = kb$. Dus
$\vec v = k\vec u$.
\end{proof}

\begin{example}\label{ex:g10:vectors:det}
Zijn $\vec u\,(3, -2)$ en $\vec v\,(-6, 4)$ \omterm{def:g10:vectors:collinear}{collineair}? Bereken
$ad - bc = 3 \times 4 - (-2)\times(-6) = 12 - 12 = 0$: ja, en inderdaad
$\vec v = -2\vec u$. Voor $\vec u\,(3, -2)$ en $\vec w\,(1, 5)$:
$3 \times 5 - (-2) \times 1 = 17 \neq 0$: niet \omterm{def:g10:vectors:collinear}{collineair}.
\end{example}

\begin{method}[Collineariteit en evenwijdigheid]\label{met:g10:vectors:alignment}
Om te bewijzen dat drie punten $A$, $B$, $C$ \omterm{def:g10:vectors:collinear}{collineair} zijn:
\begin{enumerate}
  \item bereken de \omterm{def:g10:coordgeom:system}{coördinaten} van $\vect{AB}$ en $\vect{AC}$;
  \item controleer het criterium $ad - bc = 0$ van
        \cref{thm:g10:vectors:det};
  \item concludeer: de \omterm{def:g10:vectors:vector}{vectoren} zijn \omterm{def:g10:vectors:collinear}{collineair} en delen het punt $A$,
        dus de drie punten zijn \omterm{def:g10:vectors:collinear}{collineair}.
\end{enumerate}
Dezelfde berekening met $\vect{AB}$ en $\vect{CD}$ bewijst dat de lijnen
$(AB)$ en $(CD)$ evenwijdig zijn.
\end{method}

\begin{example}\label{ex:g10:vectors:alignment}
Laat $A(1, 2)$, $B(3, 5)$ en $C(7, 11)$. Dan $\vect{AB}\,(2, 3)$ en
$\vect{AC}\,(6, 9)$, en $2 \times 9 - 3 \times 6 = 18 - 18 = 0$: de
punten zijn \omterm{def:g10:vectors:collinear}{collineair} (in feite $\vect{AC} = 3\vect{AB}$).
\end{example}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g10:vectors:1}
Vereenvoudig met de relatie van Chasles:
\[
\vect{AB} + \vect{BD}, \qquad
\vect{KL} + \vect{LM} + \vect{MK}, \qquad
\vect{AC} - \vect{BC}, \qquad
\vect{AB} + \vect{CA}.
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:vectors:2}
Laat $A(2, 1)$, $B(5, 3)$, $C(-1, 4)$. Bereken de \omterm{def:g10:coordgeom:system}{coördinaten} van
$\vect{AB}$, $\vect{BC}$, $\vect{AC}$, en controleer op de \omterm{def:g10:coordgeom:system}{coördinaten}
dat $\vect{AB} + \vect{BC} = \vect{AC}$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:vectors:3}
Laat $\vec u\,(2, -3)$ en $\vec v\,(-1, 4)$. Geef de \omterm{def:g10:coordgeom:system}{coördinaten} van
$\vec u + \vec v$, $3\vec u$, $2\vec u - \vec v$, en bereken
$\norm{\vec u}$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:vectors:4}
Zeg in elk geval of $\vec u$ en $\vec v$ \omterm{def:g10:vectors:collinear}{collineair} zijn:
\[
\text{(a) } \vec u\,(4, 6),\ \vec v\,(6, 9);
\qquad
\text{(b) } \vec u\,(2, -5),\ \vec v\,(-4, 10);
\qquad
\text{(c) } \vec u\,(3, 1),\ \vec v\,(1, 3).
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:vectors:5}
Laat $A(0, 2)$, $B(4, 0)$, $C(6, 3)$. Vind de \omterm{def:g10:coordgeom:system}{coördinaten} van het punt
$D$ zodat $ABCD$ een parallellogram is, d.w.z.\ zodat
$\vect{AB} = \vect{DC}$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:vectors:6}
Laat $A(-2, 1)$, $B(1, 3)$, $C(4, -1)$.
\begin{enumerate}
  \item Bereken de \omterm{def:g10:coordgeom:system}{coördinaten} van het punt $M$ zodat
        $\vect{AM} = \vect{AB} + \vect{AC}$.
  \item Wat is de vierhoek $ABMC$? Rechtvaardig.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:vectors:7}
Laat $A(1, -1)$, $B(4, 1)$, $C(10, 5)$. Zijn de punten $A$, $B$, $C$
\omterm{def:g10:vectors:collinear}{collineair}? Zelfde vraag voor $A(0, 3)$, $B(2, 2)$, $C(8, -1)$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:vectors:8}
Laat $A(1, 2)$, $B(5, 4)$, $C(6, 1)$, $D(2, -1)$. Toon aan dat $(AB)$ en
$(CD)$ evenwijdig zijn, en daarna dat $ABCD$ een parallellogram is.
Welke verificatie impliceert de andere?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:vectors:9}
Laat $ABC$ een driehoek zijn, en laat $I$ het punt zijn gedefinieerd door
$\vect{AI} = \frac23 \vect{AB}$. Met $A(0,0)$, $B(6, 3)$, $C(2, 5)$
(zodat de berekeningen eenvoudig blijven):
\begin{enumerate}
  \item bereken de \omterm{def:g10:coordgeom:system}{coördinaten} van $I$;
  \item laat $J$ zodat $\vect{CJ} = \frac23\vect{CB}$; bereken de
        \omterm{def:g10:coordgeom:system}{coördinaten} van $J$;
  \item toon aan dat $(IJ)$ evenwijdig is met $(AC)$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g10:vectors:10}
Laat $ABCD$ een willekeurige vierhoek zijn, en laat $I$, $J$, $K$, $L$
de \omterm{prop:g10:coordgeom:midpoint}{middens} van $[AB]$, $[BC]$, $[CD]$, $[DA]$ zijn. Toon met \omterm{def:g10:coordgeom:system}{coördinaten}
$A(x_A, y_A)$, enz., dat $\vect{IJ} = \vect{LK}$, en concludeer dat de
\omterm{prop:g10:coordgeom:midpoint}{middens} van de zijden van \emph{elke} vierhoek een parallellogram
vormen.
\end{exercise}
