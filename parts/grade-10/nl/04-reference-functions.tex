\chapter{Referentiefuncties}\label{ch:g10:reffunc}

Een handvol eenvoudige \omterm{def:g10:functions:function}{functies} --- affiene, kwadraat, omgekeerde,
vierkantswortel, derdemacht --- komen overal voor, alleen of gecombineerd.
Hun \omterm{def:g10:functions:graph}{grafieken} en variaties uit het hoofd kennen maakt van veel problemen
een snelle schets. Dit hoofdstuk bestudeert ze elk op hun beurt en
bewijst hun variaties met de vergelijkingstechniek van
\cref{ch:g10:functions}.

\section{Affiene functies}

\begin{definition}[Affiene functie]\label{def:g10:reffunc:affine}
Een \emph{affiene functie}\index{affiene functie} is een \omterm{def:g10:functions:function}{functie} van de
vorm
\[
f(x) = mx + p,
\]
waar $m$ en $p$ vaste \omterm{def:g10:numbers:sets}{reële getallen} zijn. Haar \omterm{def:g10:functions:graph}{grafiek} is een rechte
lijn: $m$ is de \emph{richtingscoëfficiënt}\index{richtingscoëfficiënt} en
$p$ het \emph{snijpunt met de
$y$-as}\index{snijpunt met de y-as@snijpunt met de $y$-as} (de lijn snijdt
de verticale as in $(0, p)$). Wanneer $p = 0$ is de \omterm{def:g10:functions:function}{functie}
\emph{lineair}: $f(x) = mx$ drukt evenredigheid uit.
\end{definition}

\begin{proposition}[Richtingscoëfficiënt en variaties]\label{prop:g10:reffunc:affine}
Laat $f(x) = mx + p$.
\begin{enumerate}
  \item Voor twee verschillende inputs $u \neq v$:
        $m = \dfrac{f(v) - f(u)}{v - u}$ (de \omterm{def:g10:reffunc:affine}{richtingscoëfficiënt} is de
        verandering van de output per eenheid verandering van de input).
  \item Als $m > 0$, is $f$ strikt \omterm{def:g10:functions:variations}{stijgend} op $\R$; als $m < 0$, strikt
        \omterm{def:g10:functions:variations}{dalend}; als $m = 0$, constant.
\end{enumerate}
\end{proposition}

\begin{proof}
1. Bereken $f(v) - f(u) = (mv + p) - (mu + p) = m(v - u)$, en deel door
$v - u \neq 0$.

2. Als $u < v$, dan heeft $f(v) - f(u) = m(v-u)$ het teken van $m$,
omdat $v - u > 0$: positieve $m$ geeft $f(u) < f(v)$ (\omterm{def:g10:functions:variations}{stijgend}), negatieve
$m$ geeft $f(u) > f(v)$ (\omterm{def:g10:functions:variations}{dalend}).
\end{proof}

\begin{example}\label{ex:g10:reffunc:affine}
Vind de \omterm{def:g10:reffunc:affine}{affiene functie} waarvan de \omterm{def:g10:functions:graph}{grafiek} door $A(1, 3)$ en $B(4, 9)$
gaat. De \omterm{def:g10:reffunc:affine}{richtingscoëfficiënt} is
\[
m = \frac{9 - 3}{4 - 1} = \frac{6}{3} = 2 .
\]
Dan $f(x) = 2x + p$, en $f(1) = 3$ geeft $2 + p = 3$, dus $p = 1$:
$f(x) = 2x + 1$. Controle met $B$: $f(4) = 9$.
\end{example}

\section{De kwadraatfunctie}

\begin{proposition}[Kwadraatfunctie]\label{prop:g10:reffunc:square}
De \omterm{def:g10:functions:function}{functie} $f(x) = x^2$, gedefinieerd op $\R$:
\begin{enumerate}
  \item is strikt \omterm{def:g10:functions:variations}{dalend} op $\intoc{-\infty}{0}$ en strikt \omterm{def:g10:functions:variations}{stijgend} op
        $\intco{0}{+\infty}$, met \omterm{def:g10:functions:extrema}{minimum} $0$ bij $x = 0$;
  \item voldoet aan $f(-x) = f(x)$ voor alle $x$: haar \omterm{def:g10:functions:graph}{grafiek}, een
        \emph{parabool}\index{parabool}, is symmetrisch t.o.v.\ de
        verticale as.
\end{enumerate}
\end{proposition}

\begin{proof}
1. Laat $0 \leq u < v$. Dan
\[
f(v) - f(u) = v^2 - u^2 = (v - u)(v + u) > 0,
\]
omdat $v - u > 0$ en $v + u > 0$: $f$ is strikt \omterm{def:g10:functions:variations}{stijgend} op
$\intco{0}{+\infty}$. Als $u < v \leq 0$, dan $v - u > 0$ maar
$v + u < 0$, dus $f(v) - f(u) < 0$: strikt \omterm{def:g10:functions:variations}{dalend}. Ten slotte is
$x^2 \geq 0 = f(0)$ voor alle $x$.

2. $(-x)^2 = x^2$; dus de punten $(x, x^2)$ en $(-x, x^2)$,
spiegelbeelden t.o.v.\ de verticale as, liggen beide op de \omterm{def:g10:functions:graph}{grafiek}.
\end{proof}

\begin{remark}[Kwadraten en volgorde]
Omdat de kwadraatfunctie daalt aan de negatieve kant, \emph{keert}
kwadrateren de volgorde van negatieve getallen om: $-3 < -2$ maar
$9 > 4$. Kwadrateer nooit beide kanten van een ongelijkheid zonder de
tekens te controleren.
\end{remark}

\section{De omgekeerde functie}

\begin{proposition}[Omgekeerde functie]\label{prop:g10:reffunc:inverse}
De \omterm{def:g10:functions:function}{functie} $f(x) = \dfrac1x$, gedefinieerd voor $x \neq 0$:
\begin{enumerate}
  \item is strikt \omterm{def:g10:functions:variations}{dalend} op $\intoo{-\infty}{0}$ en strikt \omterm{def:g10:functions:variations}{dalend} op
        $\intoo{0}{+\infty}$;
  \item voldoet aan $f(-x) = -f(x)$: haar \omterm{def:g10:functions:graph}{grafiek}, een
        \emph{hyperbool}\index{hyperbool}, is symmetrisch t.o.v.\ de
        oorsprong.
\end{enumerate}
\end{proposition}

\begin{proof}
1. Laat $0 < u < v$. Dan
\[
f(v) - f(u) = \frac1v - \frac1u = \frac{u - v}{uv} < 0,
\]
omdat $u - v < 0$ en $uv > 0$: strikt \omterm{def:g10:functions:variations}{dalend} op $\intoo{0}{+\infty}$.
Op $\intoo{-\infty}{0}$ heeft hetzelfde quotiënt $u - v < 0$ en
$uv > 0$ opnieuw (product van twee negatieven), dus $f$ daalt daar ook.

2. $\frac{1}{-x} = -\frac1x$.
\end{proof}

\begin{remark}
Let op: $\frac1x$ is \emph{niet} \omterm{def:g10:functions:variations}{dalend} op haar hele \omterm{def:g10:functions:function}{domein}. Van
$u = -1$ naar $v = 1$ springt de waarde van $-1$ omhoog naar $1$. De
twee takken moeten apart bestudeerd worden.
\end{remark}

\section{Vierkantswortel en derdemacht}

\begin{proposition}[Vierkantswortelfunctie]\label{prop:g10:reffunc:sqrt}
De \omterm{def:g10:functions:function}{functie} $f(x) = \sqrt{x}$, gedefinieerd op $\intco{0}{+\infty}$, is
strikt \omterm{def:g10:functions:variations}{stijgend}.
\end{proposition}

\begin{proof}
Laat $0 \leq u < v$. Vermenigvuldig en deel door de
\emph{toegevoegde}:
\[
\sqrt v - \sqrt u
= \frac{(\sqrt v - \sqrt u)(\sqrt v + \sqrt u)}{\sqrt v + \sqrt u}
= \frac{v - u}{\sqrt v + \sqrt u} > 0,
\]
omdat $v - u > 0$ en $\sqrt v + \sqrt u > 0$ (merk op $v > 0$, dus
$\sqrt v > 0$).
\end{proof}

\begin{proposition}[Derdemachtfunctie]\label{prop:g10:reffunc:cube}
De \omterm{def:g10:functions:function}{functie} $f(x) = x^3$, gedefinieerd op $\R$, is strikt \omterm{def:g10:functions:variations}{stijgend}, en
haar \omterm{def:g10:functions:graph}{grafiek} is symmetrisch t.o.v.\ de oorsprong.
\end{proposition}
\admitted

\begin{proposition}[Vergelijken van $x$, $x^2$ en $\sqrt x$]\label{prop:g10:reffunc:compare}
Voor $0 < x < 1$:\ \ $x^2 < x < \sqrt x$.\quad
Voor $x > 1$:\ \ $\sqrt x < x < x^2$.\quad
Bij $x = 0$ en $x = 1$ zijn de drie waarden gelijk.
\end{proposition}

\begin{proof}
Stel $0 < x < 1$. Vermenigvuldigen van $x < 1$ met $x > 0$ geeft
$x^2 < x$. Vervolgens $x < \sqrt x$: omdat beide kanten positief zijn en
kwadrateren \omterm{def:g10:functions:variations}{stijgend} is op positieve getallen, is deze ongelijkheid
equivalent met $x^2 < x$, wat we zojuist bewezen. Voor $x > 1$ geeft
vermenigvuldigen van $x > 1$ met $x$ dat $x^2 > x$, en $\sqrt x < x$
volgt met hetzelfde kwadrateringsargument.
\end{proof}

\begin{method}[Beelden vergelijken]\label{met:g10:reffunc:compare}
Om $f(a)$ en $f(b)$ te vergelijken voor een referentiefunctie $f$:
\begin{enumerate}
  \item plaats $a$ en $b$ op een \omterm{def:g10:numbers:interval}{interval} waar de variaties van $f$
        bekend zijn;
  \item als $f$ daar stijgt, staan de \omterm{def:g10:functions:function}{beelden} in dezelfde volgorde als
        de inputs; als $f$ daalt, is de volgorde omgekeerd;
  \item als $a$ en $b$ niet in hetzelfde \omterm{def:g10:numbers:interval}{monotonie-interval} liggen, werk
        eerst de tekens uit (bijv.\ voor het kwadraat: een negatieve en
        een positieve input).
\end{enumerate}
\end{method}

\begin{example}
Vergelijk $\dfrac{1}{2.3}$ en $\dfrac{1}{2.4}$: beide inputs liggen in
$\intoo{0}{+\infty}$, waar de omgekeerde \omterm{def:g10:functions:function}{functie} daalt, en
$2.3 < 2.4$, dus $\dfrac{1}{2.3} > \dfrac{1}{2.4}$. Vergelijk
$(-1.2)^2$ en $(-1.5)^2$: op $\intoc{-\infty}{0}$ daalt het kwadraat, en
$-1.5 < -1.2$, dus $(-1.5)^2 > (-1.2)^2$ --- inderdaad $2.25 > 1.44$.
\end{example}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g10:reffunc:1}
Geef voor elke \omterm{def:g10:reffunc:affine}{affiene functie} de \omterm{def:g10:reffunc:affine}{richtingscoëfficiënt}, het snijpunt met
de $y$-as, en zeg of ze \omterm{def:g10:functions:variations}{stijgend} of \omterm{def:g10:functions:variations}{dalend} is:
\[
f(x) = 3x - 2, \qquad
g(x) = -\tfrac12 x + 4, \qquad
h(x) = 7, \qquad
k(x) = -x .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:reffunc:2}
Vind de \omterm{def:g10:reffunc:affine}{affiene functie} waarvan de \omterm{def:g10:functions:graph}{grafiek} door $(2, 1)$ en $(5, 10)$
gaat; daarna die door $(-1, 4)$ en $(3, -4)$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:reffunc:3}
Vergelijk zonder rekenmachine:
\[
(3.1)^2 \text{ en } (3.2)^2; \qquad
(-2.7)^2 \text{ en } (-2.8)^2; \qquad
\frac{1}{5.1} \text{ en } \frac{1}{5.2}; \qquad
\sqrt{17} \text{ en } \sqrt{15}.
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:reffunc:4}
Los met de \omterm{def:g10:functions:graph}{grafiek} van de kwadraatfunctie $x^2 = 16$ op, daarna
$x^2 \leq 16$, daarna $x^2 > 9$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:reffunc:5}
Laat $x \in \intoo{0}{1}$. Orden de getallen $x$, $x^2$, $x^3$ en
$\sqrt x$ van klein naar groot, en controleer je antwoord met
$x = 0.25$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:reffunc:6}
Los grafisch, daarna algebraïsch op: $\dfrac1x = 2$, en
$\dfrac1x < 2$ voor $x > 0$. Wat verandert er als we ook $x < 0$
toestaan?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:reffunc:7}
Wetende dat de kwadraatfunctie daalt op $\intoc{-\infty}{0}$ en stijgt
op $\intco{0}{+\infty}$, omkader $x^2$ wanneer:
\[
\text{(a) } x \in \intcc{2}{5};
\qquad
\text{(b) } x \in \intcc{-3}{-1};
\qquad
\text{(c) } x \in \intcc{-2}{3}.
\]
(In geval (c): pas op: het \omterm{def:g10:functions:extrema}{minimum} van $x^2$ ligt niet in een eindpunt.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:reffunc:8}
Een taxibedrijf rekent een vast bedrag van $4$ plus $1.5$ per kilometer;
een ander rekent geen vast bedrag en $2$ per kilometer. Modelleer elke
prijs door een \omterm{def:g10:reffunc:affine}{affiene functie} van de afstand $x$, teken beide lijnen, en
vind vanaf welke afstand het eerste bedrijf goedkoper is.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:reffunc:9}
Toon aan dat voor alle $a, b \geq 0$: $\sqrt{a + b} \leq \sqrt a + \sqrt b$.
(Hint: beide kanten zijn niet-negatief, dus vergelijk hun kwadraten.)
Wanneer is er gelijkheid?
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g10:reffunc:10}
Laat $f(x) = \dfrac{2x + 1}{x - 1}$, gedefinieerd voor $x \neq 1$.
\begin{enumerate}
  \item Toon aan dat $f(x) = 2 + \dfrac{3}{x - 1}$ voor alle $x \neq 1$.
  \item Leid de variaties van $f$ op $\intoo{1}{+\infty}$ af uit die van
        de omgekeerde \omterm{def:g10:functions:function}{functie}.
\end{enumerate}
\end{exercise}
