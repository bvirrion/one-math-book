\chapter{Vergelijkingen van lijnen en lineaire stelsels}\label{ch:g10:lines}

Een rechte lijn in het coördinatenvlak wordt beschreven door een
vergelijking, en twee lijnen vormen samen een stelsel vergelijkingen
waarvan de oplossingen hun snijpunten zijn. Dit hoofdstuk verbindt drie
talen --- meetkundig (lijnen), algebraïsch (vergelijkingen) en
functioneel (affiene functies uit \cref{ch:g10:reffunc}) --- en leert de
twee standaardmethoden om een lineair stelsel op te lossen.

\section{De vergelijking van een lijn}

\begin{theorem}[Gereduceerde vergelijking]\label{thm:g10:lines:reduced}
In een coördinatenstelsel:
\begin{enumerate}
  \item elke niet-verticale lijn heeft een vergelijking van de vorm
        \[
        y = mx + p ,
        \]
        met $m$ (de \emph{richtingscoëfficiënt}\index{richtingscoëfficiënt})
        en $p$ (het \emph{snijpunt met de $y$-as}) uniek bepaald --- dit
        is de \emph{gereduceerde
        vergelijking}\index{gereduceerde vergelijking} van de lijn;
  \item elke verticale lijn heeft een vergelijking $x = c$;
  \item een punt ligt op de lijn precies wanneer zijn coördinaten aan de
        vergelijking voldoen.
\end{enumerate}
\end{theorem}
\admitted

\begin{proposition}[Richtingscoëfficiënt door twee punten]\label{prop:g10:lines:slope}
De richtingscoëfficiënt van de (niet-verticale) lijn door $A(x_A, y_A)$
en $B(x_B, y_B)$, met $x_A \neq x_B$, is
\[
m = \frac{y_B - y_A}{x_B - x_A}
\qquad
\text{(``verandering in $y$ over verandering in $x$'').}
\]
\end{proposition}

\begin{proof}
Beide punten voldoen aan $y = mx + p$, dus
$y_B - y_A = (mx_B + p) - (mx_A + p) = m(x_B - x_A)$; deel door
$x_B - x_A \neq 0$.
\end{proof}

\begin{method}[De vergelijking van een lijn door twee punten vinden]\label{met:g10:lines:through}
Gegeven $A$ en $B$ met $x_A \neq x_B$:
\begin{enumerate}
  \item bereken de richtingscoëfficiënt
        $m = \dfrac{y_B - y_A}{x_B - x_A}$;
  \item schrijf $y = mx + p$ en substitueer de coördinaten van $A$ (of
        $B$) om $p$ te vinden;
  \item controleer de eindvergelijking met het \emph{andere} punt.
\end{enumerate}
Als $x_A = x_B$, is de lijn verticaal: haar vergelijking is $x = x_A$.
\end{method}

\begin{example}\label{ex:g10:lines:through}
Lijn door $A(1, 5)$ en $B(3, 1)$. Richtingscoëfficiënt:
$m = \frac{1 - 5}{3 - 1} = \frac{-4}{2} = -2$. Dan $y = -2x + p$; het
punt $A$ geeft $5 = -2 \times 1 + p$, dus $p = 7$. Vergelijking:
$y = -2x + 7$. Controle met $B$: $-2 \times 3 + 7 = 1$.
\end{example}

\begin{proposition}[Evenwijdige lijnen]\label{prop:g10:lines:parallel}
Twee niet-verticale lijnen zijn evenwijdig dan en slechts dan als ze
dezelfde richtingscoëfficiënt hebben. Twee verschillende evenwijdige
lijnen hebben geen gemeenschappelijk punt; twee lijnen met verschillende
richtingscoëfficiënten hebben er precies één.
\end{proposition}

\begin{proof}
Laat $y = mx + p$ en $y = m'x + p'$ de twee lijnen zijn. Een
gemeenschappelijk punt lost $mx + p = m'x + p'$ op, d.w.z.\
$(m - m')x = p' - p$. Als $m \neq m'$, heeft dit precies één oplossing
$x = \frac{p' - p}{m - m'}$: één snijpunt. Als $m = m'$, luidt de
vergelijking $0 = p' - p$: geen oplossing wanneer $p \neq p'$
(verschillende evenwijdigen), elke $x$ wanneer $p = p'$ (zelfde lijn).
Evenwijdigheid (geen snijpunt, of samenvallen) is dus equivalent met
$m = m'$.
\end{proof}

\begin{remark}[Richtingsvector]
De lijn $y = mx + p$ gaat $m$ eenheden omhoog voor elke eenheid naar
rechts, dus de vector $\vec u\,(1, m)$ is een
\emph{richtingsvector}\index{richtingsvector} van de lijn: de lijn is
evenwijdig met $\vec u$. Twee lijnen zijn evenwijdig precies wanneer hun
richtingsvectoren collineair zijn (\cref{thm:g10:vectors:det}).
\end{remark}

\section{Lineaire stelsels}

\begin{definition}[Lineair stelsel]\label{def:g10:lines:system}
Een \emph{stelsel van twee lineaire vergelijkingen}\index{lineair stelsel}
in de onbekenden $x$ en $y$ is een paar vergelijkingen
\[
\begin{cases}
a x + b y = c \\
a' x + b' y = c'
\end{cases}
\]
die \emph{gelijktijdig} moeten gelden. Een \emph{oplossing} is een paar
$(x, y)$ dat aan beide voldoet. Elke vergelijking beschrijft een lijn,
dus het stelsel oplossen betekent het snijpunt van twee lijnen vinden.
\end{definition}

\begin{method}[Substitutie]\label{met:g10:lines:substitution}
\begin{enumerate}
  \item Los één vergelijking op naar één onbekende --- kies de
        eenvoudigste, bijv.\ druk $y$ uit in $x$;
  \item substitueer deze uitdrukking in de \emph{andere} vergelijking,
        die nu alleen $x$ bevat;
  \item los op naar $x$, bereken daarna $y$ uit stap 1;
  \item controleer het paar in beide oorspronkelijke vergelijkingen.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g10:lines:substitution}
Los $\begin{cases} x + 2y = 8 \\ 3x - y = 3 \end{cases}$ op door
substitutie. De tweede vergelijking geeft $y = 3x - 3$. Substitueer in
de eerste:
\[
x + 2(3x - 3) = 8
\ \Longleftrightarrow\
x + 6x - 6 = 8
\ \Longleftrightarrow\
7x = 14
\ \Longleftrightarrow\
x = 2 ,
\]
dan $y = 3 \times 2 - 3 = 3$. De oplossing is het paar $(2, 3)$.
Controle: $2 + 2 \times 3 = 8$ en $3 \times 2 - 3 = 3$.
\end{example}

\begin{method}[Eliminatie]\label{met:g10:lines:elimination}
\begin{enumerate}
  \item Vermenigvuldig elke vergelijking met een welgekozen getal zodat
        één onbekende tegengestelde coëfficiënten krijgt in de twee
        vergelijkingen;
  \item tel de vergelijkingen op: die onbekende verdwijnt;
  \item los de overblijvende vergelijking met één onbekende op, en
        substitueer terug om de andere onbekende te vinden;
  \item controleer het paar in beide oorspronkelijke vergelijkingen.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g10:lines:elimination}
Los $\begin{cases} 2x + 3y = 7 \\ 5x - 2y = 8 \end{cases}$ op door
eliminatie. Om $y$ te elimineren, vermenigvuldig de eerste vergelijking
met $2$ en de tweede met $3$:
\[
\begin{cases}
4x + 6y = 14 \\
15x - 6y = 24
\end{cases}
\]
Optellen: $19x = 38$, dus $x = 2$. Substitueren in $2x + 3y = 7$:
$4 + 3y = 7$, dus $y = 1$. Oplossing: $(2, 1)$. Controle in de tweede
vergelijking: $5 \times 2 - 2 \times 1 = 8$.
\end{example}

\begin{remark}[Hoeveel oplossingen?]
Net als twee lijnen heeft een lineair stelsel in het algemeen precies één
oplossing (niet-evenwijdige lijnen), en uitzonderlijk geen (verschillende
evenwijdige lijnen) of oneindig veel (twee keer dezelfde lijn). Het
criterium van \cref{thm:g10:vectors:det} beslist: het stelsel heeft een
unieke oplossing precies wanneer $ab' - a'b \neq 0$.
\end{remark}

\begin{example}[Modelleren met een stelsel]\label{ex:g10:lines:modeling}
Drie schriften en twee pennen kosten $8$; één schrift en vier pennen
kosten $6$. Laat $x$ de prijs van een schrift zijn en $y$ die van een
pen:
\[
\begin{cases}
3x + 2y = 8 \\
x + 4y = 6 .
\end{cases}
\]
Uit de tweede vergelijking, $x = 6 - 4y$; substitueren,
$3(6 - 4y) + 2y = 8$, dus $18 - 10y = 8$, wat $y = 1$ en dan $x = 2$
geeft. Een schrift kost $2$ en een pen kost $1$.
\end{example}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g10:lines:1}
Lees voor elke lijn de richtingscoëfficiënt en het snijpunt met de
$y$-as af, en schets haar:
\[
y = 2x - 3, \qquad
y = -\tfrac13 x + 1, \qquad
y = 4, \qquad
x = -2 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:lines:2}
Behoort het punt $A(2, 7)$ tot de lijn $y = 3x + 1$? En $B(-1, -1)$?
En het punt $C(4, 13)$?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:lines:3}
Vind de gereduceerde vergelijking van de lijn door:
\[
\text{(a) } A(0, 2) \text{ en } B(3, 8);
\qquad
\text{(b) } A(-1, 5) \text{ en } B(2, -4);
\qquad
\text{(c) } A(4, 1) \text{ en } B(4, 6).
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:lines:4}
Onder de lijnen $y = 3x - 1$, $y = -3x + 2$, $y = 3x + 5$,
$y = \frac{6x + 2}{2}$, welke zijn evenwijdig met elkaar? Welke zijn
eigenlijk gelijk?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:lines:5}
Los door substitutie op:
\[
\begin{cases}
y = 2x - 1 \\
3x + y = 9
\end{cases}
\qquad\text{daarna}\qquad
\begin{cases}
x - y = 4 \\
2x + 3y = 3 .
\end{cases}
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:lines:6}
Los door eliminatie op:
\[
\begin{cases}
x + y = 10 \\
x - y = 4
\end{cases}
\qquad\text{daarna}\qquad
\begin{cases}
3x + 2y = 12 \\
2x + 5y = 8 .
\end{cases}
\]
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:lines:7}
Bepaal, zonder ze op te lossen, hoeveel oplossingen elk stelsel heeft:
\[
\begin{cases}
2x - y = 3 \\
-4x + 2y = -6
\end{cases}
\qquad
\begin{cases}
2x - y = 3 \\
-4x + 2y = 1
\end{cases}
\qquad
\begin{cases}
2x - y = 3 \\
x + y = 0 .
\end{cases}
\]
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:lines:8}
Tweehonderd tickets werden verkocht voor een schoolvoorstelling, sommige
voor $5$ (kinderen) en sommige voor $9$ (volwassenen), voor een totaal
van $1352$. Hoeveel tickets van elke soort werden verkocht?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:lines:9}
Laat $d$ de lijn $y = 2x - 3$ zijn en $A(1, 4)$.
\begin{enumerate}
  \item Geef de vergelijking van de lijn $d'$ evenwijdig met $d$ door
        $A$.
  \item Bereken het snijpunt van $d'$ met de $x$-as.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g10:lines:10}
Laat $A(0, 0)$, $B(6, 2)$ en $C(2, 4)$.
\begin{enumerate}
  \item Vind de vergelijkingen van de zwaartelijnen van de driehoek $ABC$
        vanuit $A$ en vanuit $B$ (een zwaartelijn verbindt een hoekpunt
        met het midden van de overstaande zijde).
  \item Bereken hun snijpunt $G$, en controleer dat $G$ ook op de derde
        zwaartelijn ligt. (Je zou moeten vinden dat de coördinaten van
        $G$ de gemiddelden zijn van die van $A$, $B$, $C$.)
\end{enumerate}
\end{exercise}
