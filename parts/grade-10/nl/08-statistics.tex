\chapter{Beschrijvende statistiek}\label{ch:g10:stats}

Statistiek vat een grote verzameling getallen --- cijfers, salarissen,
temperaturen --- samen met een paar goed gekozen indicatoren. Dit
hoofdstuk introduceert de twee families van indicatoren: die de
\emph{centrum} van de data lokaliseren (gemiddelde, mediaan) en die de
\emph{spreiding} meten (variatiebreedte, kwartielen). Een diepere
behandeling, inclusief de standaardafwijking, komt in
\cref{ch:g11:stat}.

\section{Data en frequenties}

\begin{definition}[Statistische reeks]\label{def:g10:stats:series}
Een \emph{statistische reeks}\index{statistische reeks} is een lijst van
waarden $x_1, x_2, \dots, x_N$ waargenomen op een populatie van grootte
$N$. Wanneer een waarde $x$ $n$ keer voorkomt, is $n$ haar
\emph{aantal} en het quotiënt
\[
f = \frac{n}{N}
\]
haar \emph{frequentie}\index{frequentie} (vaak uitgedrukt als
percentage). De frequenties van alle waarden tellen op tot $1$.
\end{definition}

\begin{example}\label{ex:g10:stats:frequencies}
De cijfers van $20$ leerlingen op een quiz van $5$ punten:
\begin{center}
\renewcommand{\arraystretch}{1.2}
\begin{tabular}{c|cccccc}
cijfer $x$ & $0$ & $1$ & $2$ & $3$ & $4$ & $5$ \\
\hline
aantal $n$ & $1$ & $2$ & $4$ & $6$ & $5$ & $2$ \\
frequentie $f$ & $0.05$ & $0.10$ & $0.20$ & $0.30$ & $0.25$ & $0.10$ \\
\end{tabular}
\end{center}
Controle: $1 + 2 + 4 + 6 + 5 + 2 = 20$ en de frequenties sommen tot $1$.
\end{example}

\section{Centrummaten}

\begin{definition}[Gemiddelde]\label{def:g10:stats:mean}
Het \emph{gemiddelde}\index{gemiddelde} van de reeks is
\[
\bar x = \frac{x_1 + x_2 + \dots + x_N}{N}.
\]
Als de waarden $v_1, \dots, v_k$ voorkomen met aantallen
$n_1, \dots, n_k$, wordt hetzelfde getal berekend als het
\emph{gewogen gemiddelde}
\[
\bar x = \frac{n_1 v_1 + n_2 v_2 + \dots + n_k v_k}{n_1 + n_2 + \dots + n_k}.
\]
\end{definition}

\begin{example}\label{ex:g10:stats:mean}
Voor de quiz van \cref{ex:g10:stats:frequencies}:
\[
\bar x = \frac{1 \times 0 + 2 \times 1 + 4 \times 2 + 6 \times 3
+ 5 \times 4 + 2 \times 5}{20}
= \frac{0 + 2 + 8 + 18 + 20 + 10}{20}
= \frac{58}{20} = 2.9 .
\]
\end{example}

\begin{definition}[Mediaan]\label{def:g10:stats:median}
De \emph{mediaan}\index{mediaan} is een waarde die de \emph{gesorteerde}
reeks in twee helften deelt: minstens de helft van de waarden is
$\leq$ de mediaan, en minstens de helft is $\geq$ haar. In de praktijk,
sorteer de $N$ waarden;
\begin{itemize}
  \item als $N$ oneven is, is de mediaan de middelste waarde, in positie
        $\frac{N+1}{2}$;
  \item als $N$ even is, neem het midden van de twee middelste waarden, in
        posities $\frac N2$ en $\frac N2 + 1$.
\end{itemize}
\end{definition}

\begin{example}\label{ex:g10:stats:median}
Gesorteerde reeks van $7$ huizenprijzen (in duizenden): $120$, $150$,
$160$, $180$, $210$, $240$, $900$. De mediaan is de $4$de waarde, $180$.
Het gemiddelde is $\frac{1960}{7} = 280$ --- groter dan $6$ van de $7$
prijzen! Eén extreme waarde ($900$) trekt het gemiddelde veel sterker op
dan de mediaan: de mediaan is \emph{robuust}, het gemiddelde niet.
\end{example}

\section{Spreidingsmaten}

\begin{definition}[Variatiebreedte, kwartielen]\label{def:g10:stats:quartiles}
Voor een gesorteerde reeks:
\begin{itemize}
  \item de \emph{variatiebreedte}\index{variatiebreedte} is het verschil
        tussen de grootste en de kleinste waarde;
  \item het \emph{eerste kwartiel}\index{kwartiel} $Q_1$ is de kleinste
        waarde zodat minstens een kwart ($25\%$) van de waarden
        $\leq Q_1$ is; het \emph{derde kwartiel} $Q_3$ is de kleinste
        waarde zodat minstens drie kwart ($75\%$) van de waarden
        $\leq Q_3$ is;
  \item de \emph{interkwartielafstand} $Q_3 - Q_1$ meet de spreiding van
        de centrale helft van de data.
\end{itemize}
\end{definition}

\begin{method}[De kwartielen vinden]\label{met:g10:stats:quartiles}
Voor een reeks van $N$ gesorteerde waarden:
\begin{enumerate}
  \item bereken $\frac N4$; als het geen geheel getal is, rond
        \emph{omhoog} af naar het volgende geheel getal; $Q_1$ is de
        waarde in die positie;
  \item bereken $\frac{3N}{4}$; rond omhoog af indien nodig; $Q_3$ is de
        waarde in die positie.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g10:stats:quartiles}
Neem de $11$ gesorteerde waarden
\[
2,\ 3,\ 5,\ 5,\ 6,\ 7,\ 8,\ 9,\ 9,\ 10,\ 12 .
\]
Mediaan: positie $\frac{11+1}{2} = 6$, dus de mediaan is $7$.
$Q_1$: $\frac{11}{4} = 2.75$, omhoog afronden naar $3$; de $3$de waarde
is $5$.
$Q_3$: $\frac{33}{4} = 8.25$, omhoog afronden naar $9$; de $9$de waarde
is $9$.
Variatiebreedte: $12 - 2 = 10$; interkwartielafstand: $9 - 5 = 4$.
\end{example}

\begin{example}[Twee groepen vergelijken]\label{ex:g10:stats:comparison}
Twee klassen maken dezelfde toets (scores op $20$):
\begin{center}
\renewcommand{\arraystretch}{1.2}
\begin{tabular}{c|ccc}
 & mediaan & $Q_1$ & $Q_3$ \\
\hline
klas A & $12$ & $10$ & $14$ \\
klas B & $12$ & $6$ & $17$ \\
\end{tabular}
\end{center}
Zelfde mediaan, zeer verschillende spreidingen: klas A is homogeen (half
haar scores in $\intcc{10}{14}$), klas B is heterogeen (haar centrale
helft overspant $\intcc{6}{17}$). Alleen centrummaten vertellen nooit het
hele verhaal.
\end{example}

\begin{remark}
Noch het gemiddelde noch de mediaan is ``beter'': het gemiddelde gebruikt
elke waarde (en is nodig om totalen te berekenen), de mediaan weerstaan
extreme waarden. Een serieuze samenvatting van een reeks geeft minstens
één centrummaat \emph{en} één spreidingsmaat.
\end{remark}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g10:stats:1}
Hier zijn de aantallen boeken die $15$ leerlingen in een jaar lazen:
\[
0,\ 1,\ 1,\ 2,\ 2,\ 2,\ 3,\ 3,\ 4,\ 4,\ 5,\ 6,\ 7,\ 8,\ 12 .
\]
Bereken het gemiddelde, de mediaan en de variatiebreedte van deze reeks.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:stats:2}
Een dobbelsteen werd $50$ keer gegooid:
\begin{center}
\renewcommand{\arraystretch}{1.2}
\begin{tabular}{c|cccccc}
uitkomst & $1$ & $2$ & $3$ & $4$ & $5$ & $6$ \\
\hline
aantal & $7$ & $9$ & $8$ & $10$ & $6$ & $10$ \\
\end{tabular}
\end{center}
Bereken de frequentie van elke uitkomst en de gemiddelde uitkomst.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:stats:3}
Vind de mediaan, $Q_1$ en $Q_3$ van de gesorteerde reeks
\[
1,\ 2,\ 4,\ 4,\ 5,\ 6,\ 6,\ 7,\ 8,\ 9,\ 9,\ 11 \qquad (N = 12),
\]
en teken de bijbehorende boxplot.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:stats:4}
Het gemiddelde van $24$ toetscijfers is $11.5$. Een $25$ste leerling
maakt de toets en scoort $19$. Wat is het nieuwe gemiddelde van de klas?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:stats:5}
Een leerling heeft een gemiddelde van $12$ na $5$ toetsen, allemaal met
hetzelfde gewicht. Welk cijfer op de $6$de toets zou het gemiddelde
verhogen tot $13$? Is een gemiddelde van $14$ bereikbaar (cijfers op
$20$)?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:stats:6}
Verzin een reeks van $7$ niet-negatieve waarden waarvan de mediaan $10$
is en waarvan het gemiddelde groter is dan $20$; daarna één waarvan de
mediaan $10$ is en het gemiddelde kleiner dan $6$. Waarom kan het
gemiddelde van zo'n reeks nooit kleiner zijn dan $\frac{40}{7}$? Wat
tonen deze voorbeelden?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:stats:7}
In een bedrijf verdienen de $9$ werknemers elk $1800$ per maand en de
directeur verdient $12000$.
\begin{enumerate}
  \item Bereken het gemiddelde en de mediane salaris.
  \item Welke indicator beschrijft hier het best een ``typisch'' salaris?
        Waarom?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:stats:8}
Een klas van $12$ jongens heeft een gemiddelde lengte van $168$ cm; de
$18$ meisjes van dezelfde klas hebben een gemiddelde lengte van $163$ cm.
Bereken de gemiddelde lengte van de hele klas. (Let op: het is
\emph{niet} het midden van $168$ en $163$ --- weeg de gemiddelden met de
groepsgroottes.)
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g10:stats:9}
Een reeks van $N$ waarden $x_1, \dots, x_N$ heeft gemiddelde $\bar x$.
Elke waarde wordt getransformeerd tot $y_i = a x_i + b$ voor vaste
getallen $a$ en $b$.
\begin{enumerate}
  \item Toon aan dat het gemiddelde van de nieuwe reeks $a \bar x + b$
        is.
  \item De temperaturen van een week, in graden Celsius, hebben
        gemiddelde $20$. Wat is hun gemiddelde in graden Fahrenheit
        ($F = 1.8\,C + 32$)?
\end{enumerate}
\end{exercise}
