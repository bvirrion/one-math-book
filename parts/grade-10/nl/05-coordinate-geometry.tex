\chapter{Analytische meetkunde}\label{ch:g10:coordgeom}

Descartes' grote idee was punten te beschrijven door paren van getallen,
zodat meetkundige problemen berekeningen worden. Met slechts twee formules
--- het \omterm{prop:g10:coordgeom:midpoint}{midden} en de afstand --- kan men bewijzen dat een driehoek
gelijkbenig is, dat een vierhoek een parallellogram is, of dat drie
punten collineair zijn, zonder een enkele hulplijn te tekenen.

\section{Coördinaten in het vlak}

\begin{definition}[Coördinatenstelsel]\label{def:g10:coordgeom:system}
Een \emph{coördinatenstelsel}\index{coördinatenstelsel} van het vlak
bestaat uit een oorsprong $O$ en twee gegradueerde assen door $O$: de
horizontale $x$-as en de verticale $y$-as. Elk punt $M$ heeft dan een
uniek paar \emph{coördinaten}\index{coördinaten} $(x, y)$: zijn
\emph{abscis}\index{abscis} $x$ en zijn \emph{ordinaat}\index{ordinaat}
$y$. Het stelsel is \emph{orthonormaal}\index{orthonormaal stelsel}
wanneer de assen loodrecht op elkaar staan en dezelfde lengte-eenheid
dragen. Alle stelsels in dit hoofdstuk zijn orthonormaal.
\end{definition}

\section{Midden van een lijnstuk}

\begin{proposition}[Middenformule]\label{prop:g10:coordgeom:midpoint}
Laat $A(x_A, y_A)$ en $B(x_B, y_B)$. Het \emph{midden}\index{midden} $I$
van het lijnstuk $[AB]$ heeft \omterm{def:g10:coordgeom:system}{coördinaten}
\[
I\left(\frac{x_A + x_B}{2},\ \frac{y_A + y_B}{2}\right).
\]
\end{proposition}

\begin{proof}
Bekijk de horizontale \omterm{def:g10:coordgeom:system}{coördinaten}. Van $A$ naar $B$ verandert de \omterm{def:g10:coordgeom:system}{abscis}
met $x_B - x_A$; het punt halverwege heeft \omterm{def:g10:coordgeom:system}{abscis}
\[
x_A + \frac{x_B - x_A}{2} = \frac{2x_A + x_B - x_A}{2}
= \frac{x_A + x_B}{2},
\]
het gemiddelde van de twee abscissen. Dezelfde berekening geldt voor de
ordinaten.
\end{proof}

\begin{example}\label{ex:g10:coordgeom:midpoint}
Met $A(-1, 4)$ en $B(5, -2)$ is het \omterm{prop:g10:coordgeom:midpoint}{midden} van $[AB]$
\[
I\left(\frac{-1 + 5}{2},\ \frac{4 + (-2)}{2}\right) = I(2, 1).
\]
De formule loopt ook achterwaarts: als $A(-1, 4)$ en het \omterm{prop:g10:coordgeom:midpoint}{midden}
$I(2, 1)$ is, dan voldoet $B$ aan $\frac{-1 + x_B}{2} = 2$ en
$\frac{4 + y_B}{2} = 1$, dus $x_B = 5$ en $y_B = -2$: het punt $B$ is
het \emph{symmetrische} van $A$ t.o.v.\ $I$.
\end{example}

\section{Afstand tussen twee punten}

\begin{theorem}[Afstandsformule]\label{thm:g10:coordgeom:distance}
In een \omterm{def:g10:coordgeom:system}{orthonormaal stelsel} is de afstand tussen $A(x_A, y_A)$ en
$B(x_B, y_B)$
\[
AB = \sqrt{(x_B - x_A)^2 + (y_B - y_A)^2}.
\]
\end{theorem}

\begin{proof}
Laat $C$ het punt $(x_B, y_A)$ zijn: het heeft dezelfde hoogte als $A$ en
dezelfde \omterm{def:g10:coordgeom:system}{abscis} als $B$, dus de driehoek $ACB$ heeft een rechte hoek in
$C$. Zijn benen zijn horizontale en verticale lijnstukken, van lengtes
$AC = \abs{x_B - x_A}$ en $CB = \abs{y_B - y_A}$. Volgens de stelling van
Pythagoras
\[
AB^2 = AC^2 + CB^2 = (x_B - x_A)^2 + (y_B - y_A)^2,
\]
en de vierkantswortel nemen (beide kanten zijn niet-negatief) geeft de
formule.
\end{proof}

  \draw[-Stealth] (-0.6,0) -- (5.6,0) node[below, font=\small] {$x$};
  \draw[-Stealth] (0,-0.6) -- (0,3.6) node[left, font=\small] {$y$};

  \draw[omDef, thick] (A) -- (B);

    {$\abs{x_B - x_A}$};
  \draw[omThm, thick] (C) -- (B) node[midway, right, font=\small]

  \draw[thin] (C) ++(-0.25,0) -- ++(0,0.25) -- ++(0.25,0);
  \fill (A) circle (2pt) node[above left, font=\small] {$A$};
  \fill (B) circle (2pt) node[above, font=\small] {$B$};
  \fill (C) circle (2pt) node[below right, font=\small] {$C$};
\end{tikzpicture}

{\small De afstandsformule is de stelling van Pythagoras toegepast op de
rechthoekige driehoek $ACB$ gebouwd op een horizontaal en een verticaal
been.}
\end{omfigure}

\begin{example}\label{ex:g10:coordgeom:distance}
Met $A(1, 1)$ en $B(5, 3)$:
\[
AB = \sqrt{(5-1)^2 + (3-1)^2} = \sqrt{16 + 4} = \sqrt{20} = 2\sqrt 5 .
\]
Afstanden worden meestal in exacte (vierkantswortel-)vorm gehouden;
rond alleen helemaal aan het eind als een decimaal antwoord nodig is.
\end{example}

\begin{remark}
De volgorde van de punten speelt geen rol: $(x_A - x_B)^2 =
(x_B - x_A)^2$. Maar vergeet de kwadraten niet! De afstand is
\emph{niet} $\abs{x_B - x_A} + \abs{y_B - y_A}$.
\end{remark}

\section{De formules gebruiken in de meetkunde}

\begin{method}[Aard van een driehoek]\label{met:g10:coordgeom:triangle}
Gegeven drie punten $A$, $B$, $C$ door hun \omterm{def:g10:coordgeom:system}{coördinaten}:
\begin{enumerate}
  \item bereken de drie gekwadrateerde lengtes $AB^2$, $AC^2$, $BC^2$ met
        de afstandsformule (kwadraten houden vermijdt vierkantswortels);
  \item twee gelijke gekwadrateerde lengtes $\Rightarrow$ de driehoek is
        \emph{gelijkbenig};
  \item als de grootste gekwadrateerde lengte de som is van de twee
        andere, is de driehoek \emph{rechthoekig} in het hoekpunt
        tegenover de langste zijde, volgens de omgekeerde van de stelling
        van Pythagoras.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g10:coordgeom:triangle}
Laat $A(0, 1)$, $B(4, 3)$ en $C(2, -3)$. Dan
\begin{align*}
AB^2 &= (4-0)^2 + (3-1)^2 = 16 + 4 = 20, \\
AC^2 &= (2-0)^2 + (-3-1)^2 = 4 + 16 = 20, \\
BC^2 &= (2-4)^2 + (-3-3)^2 = 4 + 36 = 40 .
\end{align*}
Eerst, $AB^2 = AC^2$, dus $AB = AC$: de driehoek is gelijkbenig in $A$.
Bovendien $AB^2 + AC^2 = 40 = BC^2$, dus volgens de omgekeerde van de
stelling van Pythagoras is de driehoek ook rechthoekig in $A$.
\end{example}

\begin{method}[Een parallellogram herkennen]\label{met:g10:coordgeom:parallelogram}
Een vierhoek $ABCD$ is een parallellogram precies wanneer zijn
diagonalen $[AC]$ en $[BD]$ hetzelfde \omterm{prop:g10:coordgeom:midpoint}{midden} hebben. Dus: bereken beide
\omterm{prop:g10:coordgeom:midpoint}{middens} met de middenformule en vergelijk ze.
\end{method}

\begin{example}\label{ex:g10:coordgeom:parallelogram}
Laat $A(-1, 0)$, $B(2, 2)$, $C(5, 1)$ en $D(2, -1)$ gegeven zijn. Het
\omterm{prop:g10:coordgeom:midpoint}{midden} van $[AC]$ is
$\left(\frac{-1+5}{2}, \frac{0+1}{2}\right) = (2, \frac12)$; het
\omterm{prop:g10:coordgeom:midpoint}{midden} van $[BD]$ is
$\left(\frac{2+2}{2}, \frac{2-1}{2}\right) = (2, \frac12)$. Ze vallen
samen, dus $ABCD$ is een parallellogram.
\end{example}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g10:coordgeom:1}
Laat $A(2, 5)$ en $B(-4, 1)$. Bereken de \omterm{def:g10:coordgeom:system}{coördinaten} van het \omterm{prop:g10:coordgeom:midpoint}{midden} van
$[AB]$ en de afstand $AB$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:coordgeom:2}
Laat $A(3, -2)$ en $I(1, 2)$. Vind de \omterm{def:g10:coordgeom:system}{coördinaten} van het punt $B$ zodat
$I$ het \omterm{prop:g10:coordgeom:midpoint}{midden} is van $[AB]$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:coordgeom:3}
Welk van de punten $P(4, 1)$, $Q(-3, 2)$ en $R(0, -5)$ ligt het dichtst
bij de oorsprong $O(0,0)$? Antwoord met exacte afstanden.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:coordgeom:4}
Teken de punten $A(1, 2)$, $B(5, 4)$, $C(7, 0)$ op ruitjespapier, en toon
dan door berekening dat de driehoek $ABC$ gelijkbenig is. Is hij
rechthoekig?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:coordgeom:5}
Laat $A(-2, 1)$, $B(1, 3)$, $C(4, 1)$ en $D(1, -1)$.
\begin{enumerate}
  \item Toon aan dat $ABCD$ een parallellogram is.
  \item Bereken $AB$ en $BC$. Is $ABCD$ een ruit (alle zijden gelijk)?
  \item Bereken $AC$ en $BD$. Is $ABCD$ een rechthoek (gelijke
        diagonalen)?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:coordgeom:6}
De cirkel met middelpunt $\Omega(2, 1)$ en straal $5$ bestaat uit alle
punten op afstand $5$ van $\Omega$. Welke van de punten $A(5, 5)$,
$B(-1, -3)$, $C(6, 2)$ liggen op deze cirkel? Erbinnen? Erbuiten?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:coordgeom:7}
Laat $A(1, 1)$ en $B(7, 5)$. Vind alle punten $M(x, 0)$ van de $x$-as die
op gelijke afstand van $A$ en $B$ liggen. (Schrijf $MA^2 = MB^2$ en los
op naar $x$.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:coordgeom:8}
Laat $A(0, 3)$, $B(4, 1)$.
\begin{enumerate}
  \item Bereken de \omterm{def:g10:coordgeom:system}{coördinaten} van het \omterm{prop:g10:coordgeom:midpoint}{midden} $I$ van $[AB]$.
  \item Toon aan dat $M(2, 2) = I$, en verifieer door $MA$, $MB$ te
        berekenen dat $M$ op gelijke afstand van $A$ en $B$ ligt.
  \item Vind een tweede punt, op de $y$-as, op gelijke afstand van $A$ en
        $B$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:coordgeom:9}
De punten $A(-1, -1)$, $B(3, 1)$ en $C(11, 5)$ zijn gegeven. Bereken
$AB$, $BC$ en $AC$, en leid af dat $A$, $B$, $C$ collineair zijn. (Drie
punten zijn collineair precies wanneer de grootste van de drie afstanden
de som is van de twee andere.)
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g10:coordgeom:10}
Laat $A(a, 0)$ en $B(0, b)$ met $a, b > 0$, en laat $I$ het \omterm{prop:g10:coordgeom:midpoint}{midden} van
$[AB]$ zijn. Toon door berekening dat $OI = IA = IB$, waar $O$ de
oorsprong is. (Dit bewijst een klassieke stelling: in een rechthoekige
driehoek is het \omterm{prop:g10:coordgeom:midpoint}{midden} van de schuine zijde op gelijke afstand van de
drie hoekpunten.)
\end{exercise}
