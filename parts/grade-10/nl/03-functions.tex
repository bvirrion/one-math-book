\chapter{Functies}\label{ch:g10:functions}

Een \omterm{def:g10:functions:function}{functie} is een machine die een getal inneemt en een getal
uitspuugt, altijd dezelfde output voor dezelfde input. \omterm{def:g10:functions:function}{Functies} zijn de
centrale objecten van de analyse die je de komende jaren opbouwt
(\cref{ch:g11:quad} en verder); dit hoofdstuk zet de woordenschat op ---
\omterm{def:g10:functions:function}{domein}, \omterm{def:g10:functions:function}{beeld}, \omterm{def:g10:functions:graph}{grafiek}, variaties --- en traint de essentiële vaardigheid
om informatie van een \omterm{def:g10:functions:graph}{grafiek} af te lezen.

\section{Woordenschat: beelden en originelen}

\begin{definition}[Functie]\label{def:g10:functions:function}
Laat $D$ een verzameling \omterm{def:g10:numbers:sets}{reële getallen} zijn. Een
\emph{functie}\index{functie} $f$ gedefinieerd op $D$ koppelt aan elk
getal $x \in D$ precies één \omterm{def:g10:numbers:sets}{reëel getal}, genoteerd $f(x)$ en het
\emph{beeld}\index{beeld} van $x$ genoemd. De verzameling $D$ is het
\emph{domein}\index{domein} van $f$. We schrijven
\[
f : x \longmapsto f(x).
\]
Als $f(x) = y$, dan is $x$ een \emph{origineel}\index{origineel} van $y$.
\end{definition}

\begin{remark}
Elke $x$ in het \omterm{def:g10:functions:function}{domein} heeft precies één \omterm{def:g10:functions:function}{beeld}, maar een getal $y$ kan
meerdere originelen hebben, of geen. Voor $f(x) = x^2$: het \omterm{def:g10:functions:function}{beeld} van $3$
is $9$, en $9$ heeft twee originelen, $3$ en $-3$; het getal $-4$ heeft
geen \omterm{def:g10:functions:function}{origineel}.
\end{remark}

\begin{example}[Een domein vinden]\label{ex:g10:functions:domain}
Wanneer een \omterm{def:g10:functions:function}{functie} door een formule gegeven is, is haar \omterm{def:g10:functions:function}{domein} de
verzameling van $x$ waarvoor de formule zin heeft.
\begin{itemize}
  \item $f(x) = 3x^2 - 5x + 1$ heeft zin voor elke $x$: het \omterm{def:g10:functions:function}{domein} is
        $\R$.
  \item $g(x) = \dfrac{1}{x - 2}$ eist $x - 2 \neq 0$: het \omterm{def:g10:functions:function}{domein} is
        alle reëlen behalve $2$.
  \item $h(x) = \sqrt{x - 3}$ eist $x - 3 \geq 0$: het \omterm{def:g10:functions:function}{domein} is
        $\intco{3}{+\infty}$.
\end{itemize}
\end{example}

\begin{example}[Beelden en originelen berekenen]\label{ex:g10:functions:images}
Laat $f(x) = x^2 - 4x + 3$, gedefinieerd op $\R$.

\emph{\omterm{def:g10:functions:function}{Beeld} van $5$:} substitueer $x = 5$:
$f(5) = 25 - 20 + 3 = 8$.

\emph{Originelen van $3$:} los $f(x) = 3$ op:
\[
x^2 - 4x + 3 = 3
\quad\Longleftrightarrow\quad
x^2 - 4x = 0
\quad\Longleftrightarrow\quad
x(x - 4) = 0,
\]
dus de originelen van $3$ zijn $0$ en $4$.
\end{example}

\section{De grafiek van een functie}

\begin{definition}[Grafiek]\label{def:g10:functions:graph}
In een coördinatenstelsel is de \emph{grafiek}\index{grafiek} (of
\emph{kromme}) van $f$ de verzameling van alle punten $(x, f(x))$ voor
$x$ in het \omterm{def:g10:functions:function}{domein}. Met andere woorden: een punt $(x, y)$ ligt op de
grafiek precies wanneer $y = f(x)$.
\end{definition}

\begin{omfigure}
\begin{tikzpicture}
\begin{axis}[omaxis,
  width=8.6cm, height=6.0cm,
  xmin=-2.6, xmax=4.6, ymin=-3.4, ymax=4.4,
  xtick={-2,-1,1,2,3,4}, ytick={-3,-2,-1,1,2,3,4}]
  \addplot[omDef, thick, domain=-2.2:4.2, samples=120]
    {0.25*x^3 - 0.75*x^2 - x + 2};
  \draw[dashed, omThm] (axis cs:3,0) -- (axis cs:3,1.25)
    -- (axis cs:0,1.25);
  \fill[omThm] (axis cs:3,1.25) circle (1.5pt);
  \node[omThm, font=\small, above right] at (axis cs:3,1.2)
    {$(3,\,f(3))$};
\end{axis}
\end{tikzpicture}

{\small Een \omterm{def:g10:functions:function}{beeld} aflezen op een \omterm{def:g10:functions:graph}{grafiek}: start bij $x = 3$ op de
horizontale as, ga verticaal naar de kromme, dan horizontaal naar de
verticale as om $f(3) = 1.25$ te lezen.}
\end{omfigure}

\begin{method}[Een grafiek lezen]\label{met:g10:functions:reading}
Op de \omterm{def:g10:functions:graph}{grafiek} van $f$:
\begin{enumerate}
  \item \emph{\omterm{def:g10:functions:function}{beeld} van $a$}: ga verticaal van $x = a$ op de horizontale
        as naar de kromme, dan horizontaal naar de verticale as; de
        waarde die je daar leest is $f(a)$;
  \item \emph{originelen van $b$}: teken de horizontale lijn $y = b$; de
        originelen zijn de $x$-coördinaten van al haar snijpunten met de
        kromme;
  \item \emph{oplossingen van $f(x) = k$}: hetzelfde als de originelen
        van $k$;
  \item \emph{oplossingen van $f(x) \leq k$}: de $x$ waarvoor de kromme
        op of onder de lijn $y = k$ ligt.
\end{enumerate}
\end{method}

\begin{omfigure}
\begin{tikzpicture}
\begin{axis}[omaxis,
  width=8.6cm, height=6.0cm,
  xmin=-3.4, xmax=3.4, ymin=-2.4, ymax=4.4,
  xtick={-2,2}, xticklabels={$x_1$,$x_2$}, ytick={2}, yticklabels={$k$}]
  \addplot[omDef, thick, domain=-3:3, samples=120] {0.5*x^2};
  \addplot[omThm, thick, domain=-3.2:3.2] {2};
  \fill (axis cs:-2,2) circle (1.5pt);
  \fill (axis cs:2,2) circle (1.5pt);
  \draw[dashed] (axis cs:-2,2) -- (axis cs:-2,0);
  \draw[dashed] (axis cs:2,2) -- (axis cs:2,0);
\end{axis}
\end{tikzpicture}

{\small $f(x) = k$ grafisch oplossen: de oplossingen $x_1$ en $x_2$ zijn
de abscissen van de snijpunten van de kromme met de horizontale lijn
$y = k$. Hier geldt $f(x) \leq k$ op $\intcc{x_1}{x_2}$, waar de kromme
onder de lijn ligt.}
\end{omfigure}

\begin{example}\label{ex:g10:functions:membership}
Ligt het punt $A(2, 5)$ op de \omterm{def:g10:functions:graph}{grafiek} van $f(x) = x^2 + 1$? Bereken
$f(2) = 4 + 1 = 5$: ja, want $f(2)$ is gelijk aan de $y$-coördinaat van
$A$. Het punt $B(3, 8)$ ligt niet op de \omterm{def:g10:functions:graph}{grafiek}, omdat
$f(3) = 10 \neq 8$.
\end{example}

\section{Variaties van een functie}

\begin{definition}[Stijgend, dalend]\label{def:g10:functions:variations}
Laat $f$ gedefinieerd zijn op een \omterm{def:g10:numbers:interval}{interval} $I$.
\begin{itemize}
  \item $f$ is \emph{stijgend}\index{stijgende functie} op $I$ wanneer
        voor alle $u, v \in I$, als $u < v$ dan $f(u) \leq f(v)$: de
        outputs groeien met de inputs, en de \omterm{def:g10:functions:graph}{grafiek} klimt van links naar
        rechts.
  \item $f$ is \emph{dalend}\index{dalende functie} op $I$ wanneer
        $u < v$ impliceert $f(u) \geq f(v)$: de \omterm{def:g10:functions:graph}{grafiek} daalt van links
        naar rechts.
\end{itemize}
Met strikte ongelijkheden $f(u) < f(v)$ (resp.\ $>$) zeggen we
\emph{strikt} stijgend (resp.\ dalend).
\end{definition}

\begin{definition}[Veranderingstabel]\label{def:g10:functions:table}
Een \emph{veranderingstabel}\index{veranderingstabel} vat samen op welke
intervallen $f$ stijgt en daalt, met pijlen $\nearrow$ en $\searrow$, en
noteert de waarden van $f$ in de keerpunten.
\end{definition}

\begin{example}\label{ex:g10:functions:table}
Beschouw de hieronder getekende \omterm{def:g10:functions:function}{functie} op $\intcc{-2}{4}$.

\begin{omfigure}
\begin{tikzpicture}
\begin{axis}[omaxis,
  width=8.6cm, height=5.4cm,
  xmin=-2.8, xmax=4.6, ymin=-2.4, ymax=3.8,
  xtick={-2,1,4}, ytick={-1.5,3}, yticklabels={$-1.5$,$3$}]
  \addplot[omDef, thick, domain=-2:4, samples=120] {-0.5*(x-1)^2 + 3};
  \fill (axis cs:-2,-1.5) circle (1.5pt);
  \fill (axis cs:4,-1.5) circle (1.5pt);
  \fill (axis cs:1,3) circle (1.5pt);
\end{axis}
\end{tikzpicture}

{\small Een \omterm{def:g10:functions:function}{functie} gedefinieerd op $\intcc{-2}{4}$, eerst \omterm{def:g10:functions:variations}{stijgend} dan
\omterm{def:g10:functions:variations}{dalend}.}
\end{omfigure}

Aflezen van de \omterm{def:g10:functions:graph}{grafiek}: $f$ stijgt van $f(-2) = -1.5$ tot haar \omterm{def:g10:functions:extrema}{maximum}
$f(1) = 3$, en daalt daarna tot $f(4) = -1.5$. Haar \omterm{def:g10:functions:table}{veranderingstabel} is
\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{c|ccccc}
$x$ & $-2$ & & $1$ & & $4$ \\
\hline
$f$ & $-1.5$ & $\nearrow$ & $3$ & $\searrow$ & $-1.5$ \\
\end{tabular}
\end{center}
\end{example}

\begin{example}[Een variatie bewijzen]\label{ex:g10:functions:proof}
Toon aan dat $f(x) = 3x + 1$ strikt \omterm{def:g10:functions:variations}{stijgend} is op $\R$. Neem willekeurige
$u < v$ en vergelijk de \omterm{def:g10:functions:function}{beelden}:
\[
f(v) - f(u) = (3v + 1) - (3u + 1) = 3(v - u) > 0,
\]
omdat $v - u > 0$. Dus $f(u) < f(v)$: de \omterm{def:g10:functions:function}{functie} is strikt \omterm{def:g10:functions:variations}{stijgend}.
Dezelfde berekening met een negatieve helling, bijv.\
$g(x) = -2x + 5$, geeft $g(v) - g(u) = -2(v-u) < 0$: strikt \omterm{def:g10:functions:variations}{dalend}.
\end{example}

\section{Extrema}

\begin{definition}[Maximum, minimum]\label{def:g10:functions:extrema}
Laat $f$ gedefinieerd zijn op $D$ en $a \in D$. De waarde $f(a)$ is het
\emph{maximum}\index{maximum} van $f$ op $D$ wanneer $f(x) \leq f(a)$
voor alle $x \in D$; het is het \emph{minimum}\index{minimum} wanneer
$f(x) \geq f(a)$ voor alle $x \in D$. Op een \omterm{def:g10:functions:graph}{grafiek} zijn dat de hoogste
en laagste punten van de kromme.
\end{definition}

\begin{example}\label{ex:g10:functions:extremum}
Toon aan dat $f(x) = (x - 1)^2 + 2$ \omterm{def:g10:functions:extrema}{minimum} $2$ heeft op $\R$, bereikt bij
$x = 1$. Voor elke $x$ is het kwadraat $(x-1)^2$ groter of gelijk aan
$0$, dus $f(x) = (x-1)^2 + 2 \geq 2$; en $f(1) = 0 + 2 = 2$, dus de
waarde $2$ wordt daadwerkelijk bereikt. Beide feiten samen bewijzen dat
$2$ het \omterm{def:g10:functions:extrema}{minimum} is.
\end{example}

\begin{omfigure}
\begin{tikzpicture}
\begin{axis}[omaxis,
  width=7.6cm, height=5.2cm,
  xmin=-1.8, xmax=3.8, ymin=-0.9, ymax=6.4,
  xtick={1}, ytick={2}]
  \addplot[omDef, thick, domain=-1.0:3.0, samples=100] {(x-1)^2 + 2};
  \addplot[omThm, dashed, domain=-1.6:3.6] {2};
  \fill (axis cs:1,2) circle (1.5pt);
  \node[below right, font=\small] at (axis cs:1,2) {$(1,2)$};
\end{axis}
\end{tikzpicture}

{\small De kromme van $f(x) = (x-1)^2 + 2$ gaat nooit onder de gestippelde
lijn $y = 2$, en raakt die bij $x = 1$: het \omterm{def:g10:functions:extrema}{minimum} van $f$ is $2$.}
\end{omfigure}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g10:functions:1}
Laat $f(x) = 2x^2 - 3x + 1$. Bereken de \omterm{def:g10:functions:function}{beelden} van $0$, $2$, $-1$ en
$\frac12$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:functions:2}
Geef het \omterm{def:g10:functions:function}{domein} van elke \omterm{def:g10:functions:function}{functie}:
\[
f(x) = 5x - 2, \qquad
g(x) = \frac{3}{x + 4}, \qquad
h(x) = \sqrt{2x - 6}, \qquad
k(x) = \frac{1}{x^2 + 1}.
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:functions:3}
Laat $f(x) = x^2 - 2x$. Vind alle originelen van $0$, van $3$, en van
$-1$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:functions:4}
Behoort het punt $A(-1, 4)$ tot de \omterm{def:g10:functions:graph}{grafiek} van $f(x) = x^2 - 2x + 1$?
En het punt $B(2, 1)$? Rechtvaardig met een berekening.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:functions:5}
Een \omterm{def:g10:functions:function}{functie} $g$ gedefinieerd op $\intcc{-3}{5}$ heeft de
\omterm{def:g10:functions:table}{veranderingstabel}
\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{c|ccccccc}
$x$ & $-3$ & & $0$ & & $3$ & & $5$ \\
\hline
$g$ & $1$ & $\searrow$ & $-2$ & $\nearrow$ & $4$ & $\searrow$ & $0$ \\
\end{tabular}
\end{center}
\begin{enumerate}
  \item Wat zijn het \omterm{def:g10:functions:extrema}{maximum} en het \omterm{def:g10:functions:extrema}{minimum} van $g$ op $\intcc{-3}{5}$?
  \item Vergelijk $g(-1)$ en $g(-0.5)$ zonder hun waarden te kennen.
  \item Hoeveel oplossingen heeft de \omterm{def:g10:algebra:equation}{vergelijking} $g(x) = 0$ ten hoogste
        op elk \omterm{def:g10:numbers:interval}{interval} van de tabel?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:functions:6}
Laat $f(x) = -2x + 7$. Toon aan, door $f(u)$ en $f(v)$ te vergelijken
voor $u < v$, dat $f$ strikt \omterm{def:g10:functions:variations}{dalend} is op $\R$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:functions:7}
Toon aan dat de \omterm{def:g10:functions:function}{functie} $f(x) = x^2 + 6x + 11$ een \omterm{def:g10:functions:extrema}{minimum} heeft op $\R$,
en geef de waarde en waar die bereikt wordt. (Hint: schrijf
$f(x) = (x + 3)^2 + c$ voor de juiste constante $c$.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:functions:8}
Een rechthoekige tuin heeft omtrek $40$ m. Laat $x$ de breedte zijn, in
meters.
\begin{enumerate}
  \item Druk de lengte, daarna de oppervlakte $A(x)$, uit als \omterm{def:g10:functions:function}{functies}
        van $x$. Voor welke $x$ heeft dit zin?
  \item Bereken $A(4)$, $A(8)$, $A(10)$, $A(12)$ en $A(16)$. Wat
        vermoed je over de vorm met de grootste oppervlakte?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:functions:9}
Laat $f(x) = \dfrac{1}{x^2 + 1}$, gedefinieerd op $\R$.
\begin{enumerate}
  \item Toon aan dat $0 < f(x) \leq 1$ voor elk \omterm{def:g10:numbers:sets}{reëel getal} $x$.
  \item Heeft $f$ een \omterm{def:g10:functions:extrema}{maximum} op $\R$? Een \omterm{def:g10:functions:extrema}{minimum}? Rechtvaardig.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g10:functions:10}
Laat $f(x) = x^2$ op $\R$.
\begin{enumerate}
  \item Laat $0 \leq u < v$. Ontbind $f(v) - f(u)$ en leid af dat $f$
        strikt \omterm{def:g10:functions:variations}{stijgend} is op $\intco{0}{+\infty}$.
  \item Pas het argument aan om te tonen dat $f$ strikt \omterm{def:g10:functions:variations}{dalend} is op
        $\intoc{-\infty}{0}$.
\end{enumerate}
\end{exercise}
