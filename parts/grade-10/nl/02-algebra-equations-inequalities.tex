\chapter{Algebra: vergelijkingen en ongelijkheden}\label{ch:g10:algebra}

Algebra is de kunst van rekenen met letters. Dit hoofdstuk herhaalt en
scherpt de twee basishandelingen --- \omterm{def:g10:algebra:expand}{ontwikkelen} en \omterm{def:g10:algebra:expand}{ontbinden} --- en
gebruikt die om systematisch \omterm{def:g10:algebra:equation}{vergelijkingen} en ongelijkheden op te
lossen, met de \omterm{def:g10:algebra:signtable}{tekentabel} als centraal gereedschap.

\section{Ontwikkelen en ontbinden}

\begin{definition}[Ontwikkelen, ontbinden]\label{def:g10:algebra:expand}
\emph{Ontwikkelen}\index{ontwikkelen} van een uitdrukking betekent
producten omzetten in sommen; \emph{ontbinden}\index{ontbinden in
factoren} betekent het omgekeerde, een som omzetten in een product. De
basisregel is de distributiviteit:
\[
k(a + b) = ka + kb,
\qquad
(a + b)(c + d) = ac + ad + bc + bd .
\]
\end{definition}

\begin{example}\label{ex:g10:algebra:expand}
Ontwikkel $(2x + 3)(x - 5)$ stap voor stap:
\begin{align*}
(2x + 3)(x - 5)
&= 2x \times x + 2x \times (-5) + 3 \times x + 3 \times (-5) \\
&= 2x^2 - 10x + 3x - 15 \\
&= 2x^2 - 7x - 15 .
\end{align*}
\end{example}

\begin{theorem}[Merkwaardige producten]\label{thm:g10:algebra:identities}
Voor alle \omterm{def:g10:numbers:sets}{reële getallen} $a$ en $b$:
\[
(a+b)^2 = a^2 + 2ab + b^2,
\qquad
(a-b)^2 = a^2 - 2ab + b^2,
\qquad
(a+b)(a-b) = a^2 - b^2 .
\]
\end{theorem}

\begin{proof}
Elk is een directe ontwikkeling. Voor het eerste:
\[
(a+b)^2 = (a+b)(a+b) = a^2 + ab + ba + b^2 = a^2 + 2ab + b^2 .
\]
Het tweede volgt door $b$ te vervangen door $-b$: $(a-b)^2 = a^2 + 2a(-b) +
(-b)^2 = a^2 - 2ab + b^2$. Voor het derde:
$(a+b)(a-b) = a^2 - ab + ba - b^2 = a^2 - b^2$.
\end{proof}

\begin{example}\label{ex:g10:algebra:identities}
Voorwaarts gebruikt (\omterm{def:g10:algebra:expand}{ontwikkelen}) en achterwaarts (\omterm{def:g10:algebra:expand}{ontbinden}):
\begin{align*}
(3x + 2)^2 &= 9x^2 + 12x + 4,
& x^2 - 25 &= (x+5)(x-5), \\
(x - 4)^2 &= x^2 - 8x + 16,
& 4x^2 + 4x + 1 &= (2x + 1)^2 .
\end{align*}
Hoofdrekenen profiteert ook: $101 \times 99 = (100+1)(100-1) =
10000 - 1 = 9999$.
\end{example}

\begin{method}[Een uitdrukking ontbinden]\label{met:g10:algebra:factor}
Probeer, in deze volgorde:
\begin{enumerate}
  \item \emph{Gemeenschappelijke factor:} zoek een factor die in elke term
        voorkomt en haal die eruit, bijv.\ $6x^2 + 4x = 2x(3x + 2)$. De
        gemeenschappelijke factor kan een hele haak zijn:
        $(x+1)(2x-3) + (x+1)(x+7) = (x+1)(3x+4)$.
  \item \emph{Merkwaardig product:} herken $a^2 - b^2$, of
        $a^2 \pm 2ab + b^2$, bijv.\ $9x^2 - 16 = (3x)^2 - 4^2 =
        (3x+4)(3x-4)$.
  \item Combineer beide, en controleer het resultaat door opnieuw te
        \omterm{def:g10:algebra:expand}{ontwikkelen}.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g10:algebra:factorsteps}
Ontbind $E = (2x + 1)^2 - (x - 3)^2$. Dit is een verschil van twee
kwadraten $a^2 - b^2$ met $a = 2x+1$ en $b = x-3$:
\begin{align*}
E &= \bigl[(2x+1) + (x-3)\bigr] \times \bigl[(2x+1) - (x-3)\bigr] \\
  &= (3x - 2)(2x + 1 - x + 3) \\
  &= (3x - 2)(x + 4).
\end{align*}
Let op de tweede haak: $(x - 3)$ aftrekken betekent $x$ aftrekken
\emph{en} $3$ optellen.
\end{example}

\section{Vergelijkingen}

\begin{definition}[Vergelijking, oplossing]\label{def:g10:algebra:equation}
Een \emph{vergelijking}\index{vergelijking} is een gelijkheid die een
onbekend getal $x$ bevat; een \emph{oplossing} is een waarde van $x$ die
de gelijkheid waar maakt. De vergelijking oplossen betekent \emph{alle}
oplossingen vinden.
\end{definition}

Twee \omterm{def:g10:algebra:equation}{vergelijkingen} zijn \emph{equivalent} wanneer ze precies dezelfde
oplossingen hebben. Dezelfde getal bij beide kanten optellen, of beide
kanten vermenigvuldigen met hetzelfde \emph{niet-nul} getal, levert een
equivalente \omterm{def:g10:algebra:equation}{vergelijking}.

\begin{example}[Eerstegraadsvergelijking]\label{ex:g10:algebra:firstdegree}
Los $5x - 7 = 2x + 8$ op, één stap tegelijk:
\begin{align*}
5x - 7 &= 2x + 8 \\
5x - 2x &= 8 + 7 && \text{(tel $7 - 2x$ bij beide kanten op)} \\
3x &= 15 \\
x &= 5 && \text{(deel beide kanten door $3$).}
\end{align*}
De unieke oplossing is $5$. Controle: $5 \times 5 - 7 = 18$ en
$2 \times 5 + 8 = 18$.
\end{example}

\begin{theorem}[Product-nul-regel]\label{thm:g10:algebra:zeroproduct}
Een product van \omterm{def:g10:numbers:sets}{reële getallen} is nul dan en slechts dan als minstens één
van de factoren nul is:
\[
A \times B = 0
\quad\Longleftrightarrow\quad
A = 0 \ \text{ of } \ B = 0 .
\]
\end{theorem}

\begin{proof}
Als $A = 0$ of $B = 0$, is het product duidelijk $0$. Omgekeerd, stel
$AB = 0$ en $A \neq 0$. Delen van beide kanten door $A$ geeft $B = 0$.
\end{proof}

\begin{method}[Oplossen door ontbinden]\label{met:g10:algebra:zeroproduct}
Om een \omterm{def:g10:algebra:equation}{vergelijking} op te lossen waarvan het rechterlid $0$ is en waarvan
het linkerlid ontbonden kan worden:
\begin{enumerate}
  \item breng alles naar links zodat de \omterm{def:g10:algebra:equation}{vergelijking} $E = 0$ luidt;
  \item ontbind $E$ in een product van eerstegraadsfactoren;
  \item pas de product-nul-regel toe en los elke kleine \omterm{def:g10:algebra:equation}{vergelijking} op;
  \item verzamel alle gevonden oplossingen.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g10:algebra:zeroproduct}
Los $(x - 2)(3x + 6) = 0$ op: ofwel $x - 2 = 0$, wat $x = 2$ geeft, ofwel
$3x + 6 = 0$, wat $x = -2$ geeft. Oplossingen: $-2$ en $2$.

Los $x^2 = 9x$ op. Deel \emph{niet} door $x$ (die kan nul zijn!); breng
alles naar één kant en ontbind:
\[
x^2 - 9x = 0
\quad\Longleftrightarrow\quad
x(x - 9) = 0
\quad\Longleftrightarrow\quad
x = 0 \ \text{ of } \ x = 9 .
\]
\end{example}

\begin{example}[Quotiëntvergelijking]\label{ex:g10:algebra:quotient}
Los $\dfrac{x - 1}{x + 2} = 0$ op. Een quotiënt is nul precies wanneer de
teller nul is \emph{en} de noemer dat niet is. Hier geeft $x - 1 = 0$
$x = 1$, en $1 + 2 = 3 \neq 0$: de enige oplossing is $1$. De waarde
$x = -2$ is \emph{verboden} (deling door nul) en moet altijd eerst
worden uitgesloten.
\end{example}

\section{Ongelijkheden en tekentabellen}

\begin{proposition}[Regels voor ongelijkheden]\label{prop:g10:algebra:ineqrules}
Laat $a \leq b$.
\begin{enumerate}
  \item Voor elke $c$: $a + c \leq b + c$ (optellen behoudt de volgorde).
  \item Als $c > 0$: $ac \leq bc$ (vermenigvuldigen met een positief
        getal behoudt de volgorde).
  \item Als $c < 0$: $ac \geq bc$ (vermenigvuldigen met een negatief
        getal \emph{keert} de volgorde om).
\end{enumerate}
\end{proposition}

\begin{proof}
1. $(b + c) - (a + c) = b - a \geq 0$.
2. $bc - ac = (b-a)c$ is het product van twee niet-negatieve getallen,
dus $bc \geq ac$.
3. Nu is $(b-a)c \leq 0$ als product van een niet-negatief en een
negatief getal, dus $bc \leq ac$.
\end{proof}

\begin{example}\label{ex:g10:algebra:firstineq}
Los $-2x + 3 > 9$ op:
\begin{align*}
-2x + 3 &> 9 \\
-2x &> 6 && \text{(trek $3$ af)} \\
x &< -3 && \text{(deel door $-2 < 0$: keer de ongelijkheid om).}
\end{align*}
De oplossingsverzameling is $\intoo{-\infty}{-3}$.
\end{example}

\begin{definition}[Tekentabel]\label{def:g10:algebra:signtable}
Een \emph{tekentabel}\index{tekentabel} noteert, op één rij per factor,
het teken ($+$ of $-$) van elke factor van een uitdrukking terwijl $x$
over $\R$ loopt, met een $0$ bij elke waarde waar de factor verdwijnt;
een laatste rij geeft het teken van het product, kolom voor kolom met de
tekenregels.
\end{definition}

\begin{example}\label{ex:g10:algebra:signtable}
Teken van $P(x) = (x - 2)(3 - x)$. De factor $x - 2$ verdwijnt bij $2$ en
is positief erna; de factor $3 - x$ verdwijnt bij $3$ en is positief
ervoor:
\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{c|ccccc}
$x$ & $-\infty$ \quad & $2$ & & $3$ & \quad $+\infty$ \\
\hline
$x - 2$ & $-$ & $0$ & $+$ & & $+$ \\
$3 - x$ & $+$ & & $+$ & $0$ & $-$ \\
\hline
$P(x)$ & $-$ & $0$ & $+$ & $0$ & $-$ \\
\end{tabular}
\end{center}
Dus $P(x) > 0$ precies op $\intoo{2}{3}$, en $P(x) \leq 0$ op
$\intoc{-\infty}{2} \cup \intco{3}{+\infty}$.
\end{example}

\begin{method}[Een ongelijkheid oplossen met een tekentabel]\label{met:g10:algebra:signtable}
Om een ongelijkheid zoals $E(x) \geq 0$ of $E(x) < 0$ op te lossen:
\begin{enumerate}
  \item breng alles naar het linkerlid, zodat het rechterlid $0$ is
        (vermenigvuldig nooit beide kanten met een uitdrukking waarvan
        het teken onbekend is);
  \item ontbind het linkerlid, of zet het op een gemeenschappelijke noemer
        als het een quotiënt is;
  \item bouw de \omterm{def:g10:algebra:signtable}{tekentabel} met één rij per factor (voor een quotiënt,
        markeer verboden waarden met een dubbele streep in de laatste
        rij);
  \item lees de intervallen af waar het gevraagde teken geldt, en
        controleer of elke grens is inbegrepen.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g10:algebra:quotientineq}
Los $\dfrac{x + 1}{x - 3} \leq 0$ op. De teller verdwijnt bij $-1$, de
noemer bij $3$ (verboden waarde). Het quotiënt is negatief wanneer de
twee tegengestelde tekens hebben, wat gebeurt voor $x$ tussen $-1$ en
$3$. De grens waar de \emph{teller} verdwijnt is inbegrepen, de
verboden waarde is uitgesloten: de oplossingsverzameling is
$\intco{-1}{3}$.
\end{example}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g10:algebra:1}
Ontwikkel en vereenvoudig:
\[
3(2x - 5) - 2(x - 7), \qquad
(x + 4)(2x - 1), \qquad
(3x - 2)^2, \qquad
(5 - x)(5 + x).
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:algebra:2}
Ontbind:
\[
x^2 - 49, \qquad
x^2 + 10x + 25, \qquad
7x^2 - 14x, \qquad
16x^2 - 8x + 1 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:algebra:3}
Los op:
\[
4x + 9 = x - 3, \qquad
\frac{2x - 1}{3} = 5, \qquad
2(x - 3) = 2x + 1 .
\]
(Eén ervan heeft geen oplossing --- leg uit waarom.)
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:algebra:4}
Los op met de product-nul-regel:
\[
(x + 5)(2x - 8) = 0, \qquad
x^2 - 16 = 0, \qquad
x^2 + 6x + 9 = 0 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:algebra:5}
Los de ongelijkheden op en schrijf de oplossingsverzamelingen als
intervallen:
\[
3x - 5 \leq 7, \qquad
-4x + 1 > 9, \qquad
2(x - 1) \geq 5x + 4 .
\]
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:algebra:6}
Ontbind $E = (x - 1)(x + 3) - (x - 1)(2x - 5)$, en los daarna $E = 0$ op.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:algebra:7}
Los $x^2 = 5x$ op, daarna $(2x + 1)^2 = (x - 4)^2$. (Voor de tweede:
breng alles naar links en ontbind het verschil van kwadraten.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:algebra:8}
Los met een \omterm{def:g10:algebra:signtable}{tekentabel} op
\[
(x + 2)(4 - x) > 0,
\qquad
\frac{2x - 6}{x + 1} \geq 0 .
\]
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:algebra:9}
Een streamingdienst kost $9$ per maand, plus een eenmalige
inschrijfkosten van $15$; een concurrent kost $12$ per maand zonder
kosten. Na hoeveel maanden wordt de eerste dienst in totaal de
goedkopere keuze? Stel een ongelijkheid op en los die op.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:algebra:10}
Los $\dfrac{x+3}{x-1} = 2$ op. (Vermenigvuldig beide kanten met $x - 1$
na uitsluiting van de verboden waarde, of breng alles naar links over
een gemeenschappelijke noemer.)
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g10:algebra:11}
Laat $n$ een \omterm{def:g10:numbers:sets}{geheel getal} zijn. Toon aan dat $(n + 1)^2 - n^2$ altijd
oneven is, en dat het verschil tussen de kwadraten van twee opeenvolgende
oneven getallen altijd een veelvoud van $8$ is.
\end{exercise}
