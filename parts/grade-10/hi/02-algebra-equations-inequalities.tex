\chapter{बीजगणित: समीकरण और असमानताएँ}\label{ch:g10:algebra}

बीजगणित अक्षरों से गणना करने की कला है। यह अध्याय समीक्षा करता है और
दो बुनियादी चालों को तेज़ करता है --- का \omterm{def:g10:algebra:expand}{विस्तार} और \omterm{def:g10:algebra:expand}{फैक्टरिंग} --- और उनका उपयोग करता है
हल करना \omterm{def:g10:algebra:equation}{समीकरण} और असमानताओं को व्यवस्थित रूप से, के साथ \omterm{def:g10:algebra:signtable}{साइन टेबल} जैसे
केंद्रीय उपकरण.

\section{विस्तार और फैक्टरिंग}

\begin{definition}[विस्तार करना, फैक्टरिंग करना]\label{def:g10:algebra:expand}
\emph{विस्तार}\index{विस्तार} अभिव्यक्ति का अर्थ है परिवर्तन
उत्पादों को योग में; \emph{फैक्टरिंग}\index{फैक्टरिंग} मतलब उलटा,
किसी राशि को उत्पाद में बदलना। मूल नियम वितरणशीलता है:
\[
k(a + b) = ka + kb,
\qquad
(a + b)(c + d) = ac + ad + bc + bd .
\]
\end{definition}

\begin{example}\label{ex:g10:algebra:expand}
बढ़ाना $(2x + 3)(x - 5)$ क्रमशः:
\begin{align*}
(2x + 3)(x - 5)
&= 2x \times x + 2x \times (-5) + 3 \times x + 3 \times (-5) \\
&= 2x^2 - 10x + 3x - 15 \\
&= 2x^2 - 7x - 15 .
\end{align*}
\end{example}

\begin{theorem}[उल्लेखनीय पहचान]\label{thm:g10:algebra:identities}
सभी के लिए \omterm{def:g10:numbers:sets}{वास्तविक संख्या} $ a $ और $ b $:
\[
(a+b)^2 = a^2 + 2ab + b^2,
\qquad
(a-b)^2 = a^2 - 2ab + b^2,
\qquad
(a+b)(a-b) = a^2 - b^2 .
\]
\end{theorem}

\begin{proof}
प्रत्येक एक प्रत्यक्ष \omterm{def:g10:algebra:expand}{विस्तार} है। पहले के लिए:
\[
(a+b)^2 = (a+b)(a+b) = a^2 + ab + ba + b^2 = a^2 + 2ab + b^2 .
\]
दूसरा प्रतिस्थापित करके अनुसरण करता है $ b $ साथ $-b $: $(a-b)^2 = a^2 + 2a(-b) +
(-b)^2 = a^2 - 2ab + b^2$. तीसरे के लिए:
$(a+b)(a-b) = a^2 - ab + ba - b^2 = a^2 - b^2$.
\end{proof}

\begin{example}\label{ex:g10:algebra:identities}
आगे प्रयुक्त (का \omterm{def:g10:algebra:expand}{विस्तार}) और पिछड़ा (\omterm{def:g10:algebra:expand}{फैक्टरिंग}):
\begin{align*}
(3x + 2)^2 &= 9x^2 + 12x + 4,
& x^2 - 25 &= (x+5)(x-5), \\
(x - 4)^2 &= x^2 - 8x + 16,
& 4x^2 + 4x + 1 &= (2x + 1)^2 .
\end{align*}
मानसिक अंकगणितीय लाभ भी: $101 \times 99 = (100+1)(100-1) =
10000 - 1 = 9999$.
\end{example}

\begin{method}[एक अभिव्यक्ति का गुणनखंडन]\label{met:g10:algebra:factor}
इस क्रम में प्रयास करें:
\begin{enumerate}
  \item \emph{सामान्य कारक:} प्रत्येक पद में मौजूद एक कारक को पहचानें और खींचें
        यह बाहर, उदा.\ $6x^2 + 4x = 2x(3x + 2)$. सामान्य कारक हो सकता है a
        संपूर्ण ब्रैकेट: $(x+1)(2x-3) + (x+1)(x+7) = (x+1)(3x+4)$.
  \item \emph{उल्लेखनीय पहचान:} पहचानना $ a^2 - b^2$, या
        $ a^2 \pm 2ab + b^2$, उदा.\ $9x^2 - 16 = (3x)^2 - 4^2 =
        (3x+4)(3x-4)$.
  \item दोनों को मिलाएं, और परिणाम की जांच करें का \omterm{def:g10:algebra:expand}{विस्तार} यह वापस.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g10:algebra:factorsteps}
कारक $ E = (2x + 1)^2 - (x - 3)^2$. यह दो वर्गों का अंतर है
$ a^2 - b^2$ साथ $ a = 2x+1$ और $ b = x-3$:
\begin{align*}
E &= \bigl[(2x+1) + (x-3)\bigr] \times \bigl[(2x+1) - (x-3)\bigr] \\
  &= (3x - 2)(2x + 1 - x + 3) \\
  &= (3x - 2)(x + 4).
\end{align*}
दूसरे ब्रैकेट पर ध्यान दें: घटाना $ x - 3$ मतलब घटाना $ x $
\emph{और} जोड़ना $3$.
\end{example}

\section{समीकरण}

\begin{definition}[समीकरण, समाधान]\label{def:g10:algebra:equation}
एक \emph{समीकरण}\index{समीकरण} एक समानता है जिसमें अज्ञात है
संख्या $ x $; ए \emph{समाधान} का एक मान है $ x $ जो समानता बनाता है
सच. समीकरण को हल करने का अर्थ है खोजना \emph{सभी} इसके समाधान.
\end{definition}

दो \omterm{def:g10:algebra:equation}{समीकरण} हैं \emph{समकक्ष} जब उनके पास बिल्कुल वैसा ही हो
समाधान. दोनों पक्षों में समान संख्या जोड़ना, या दोनों पक्षों को गुणा करना
उसी के द्वारा \emph{अशून्य} संख्या, एक समतुल्य उत्पन्न करती है \omterm{def:g10:algebra:equation}{समीकरण}.

\begin{example}[प्रथम-डिग्री समीकरण]\label{ex:g10:algebra:firstdegree}
हल करना $5x - 7 = 2x + 8$, एक समय में एक चाल:
\begin{align*}
5x - 7 &= 2x + 8 \\
5x - 2x &= 8 + 7 && \text{(add $7 - 2x $ to both sides)} \\
3x &= 15 \\
x &= 5 && \text{(divide both sides by $3$).}
\end{align*}
अनोखा समाधान है $5$. जाँच करना: $5 \times 5 - 7 = 18$ और
$2 \times 5 + 8 = 18$.
\end{example}

\begin{theorem}[शून्य-उत्पाद नियम]\label{thm:g10:algebra:zeroproduct}
का एक उत्पाद \omterm{def:g10:numbers:sets}{वास्तविक संख्या} शून्य है यदि और केवल यदि इसका कम से कम एक
कारक शून्य है:
\[
A \times B = 0
\quad\Longleftrightarrow\quad
A = 0 \ \text{ or } \ B = 0 .
\]
\end{theorem}

\begin{proof}
अगर $ A = 0$ या $ B = 0$, उत्पाद स्पष्ट रूप से है $0$. इसके विपरीत, मान लीजिए
$ AB = 0$ और $ A \neq 0$. द्वारा दोनों पक्षों को विभाजित करना $ A $ देता है $ B = 0$.
\end{proof}

\begin{method}[गुणनखंडन द्वारा हल करना]\label{met:g10:algebra:zeroproduct}
एक को हल करने के लिए \omterm{def:g10:algebra:equation}{समीकरण} जिसका दाहिना भाग है $0$ और किसका बायां हाथ
पक्ष पर विचार किया जा सकता है:
\begin{enumerate}
  \item हर चीज़ को बायीं ओर लाएँ ताकि \omterm{def:g10:algebra:equation}{समीकरण} पढ़ता $ E = 0$;
  \item कारक $ E $ प्रथम-डिग्री कारकों के उत्पाद में;
  \item शून्य-उत्पाद नियम लागू करें और प्रत्येक छोटे को हल करें \omterm{def:g10:algebra:equation}{समीकरण};
  \item पाए गए सभी समाधान एकत्रित करें.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g10:algebra:zeroproduct}
हल करना $(x - 2)(3x + 6) = 0$: दोनों में से एक $ x - 2 = 0$, देना $ x = 2$, या
$3x + 6 = 0$, देना $ x = -2$. समाधान: $-2$ और $2$.

हल करना $ x^2 = 9x $. करना \emph{नहीं} द्वारा विभाजित करें $ x $ (यह शून्य हो सकता है!); लाओ
सब कुछ एक तरफ और कारक:
\[
x^2 - 9x = 0
\quad\Longleftrightarrow\quad
x(x - 9) = 0
\quad\Longleftrightarrow\quad
x = 0 \ \text{ or } \ x = 9 .
\]
\end{example}

\begin{example}[भागफल समीकरण]\label{ex:g10:algebra:quotient}
हल करना $\dfrac{x - 1}{x + 2} = 0$. एक भागफल ठीक उसी समय शून्य होता है जब वह होता है
अंश शून्य है \emph{और} इसका हर नहीं है. यहाँ $ x - 1 = 0$
देता है $ x = 1$, और $1 + 2 = 3 \neq 0$: एकमात्र समाधान है $1$. मूल्य
$ x = -2$ है \emph{निषिद्ध} (शून्य से विभाजन) और हमेशा होना चाहिए
पहले बहिष्कृत.
\end{example}

\section{असमानताएँ और संकेत तालिकाएँ}

\begin{proposition}[असमानताओं के लिए नियम]\label{prop:g10:algebra:ineqrules}
होने देना $ a \leq b $.
\begin{enumerate}
  \item किसी के लिए $ c $: $ a + c \leq b + c $ (जोड़ने से ऑर्डर सुरक्षित रहता है)।
  \item अगर $ c > 0$: $ ac \leq bc $ (एक धनात्मक संख्या से गुणा करना
        आदेश सुरक्षित रखता है)।
  \item अगर $ c < 0$: $ ac \geq bc $ (एक ऋणात्मक संख्या से गुणा करना
        \emph{पराजय} आदेश)।
\end{enumerate}
\end{proposition}

\begin{proof}
1. $(b + c) - (a + c) = b - a \geq 0$.
2. $ bc - ac = (b-a)c $ दो अऋणात्मक संख्याओं का गुणनफल है, इसलिए
$ bc \geq ac $.
3. अब $(b-a)c \leq 0$ एक गैर-नकारात्मक और एक नकारात्मक के उत्पाद के रूप में
संख्या, तो $ bc \leq ac $.
\end{proof}

\begin{example}\label{ex:g10:algebra:firstineq}
हल करना $-2x + 3 > 9$:
\begin{align*}
-2x + 3 &> 9 \\
-2x &> 6 && \text{(subtract 3)} \\
x &< -3 && \text{(divide by $-2 < 0$: reverse the inequality).}
\end{align*}
समाधान सेट है $\intoo{-\infty}{-3}$.
\end{example}

\begin{definition}[साइन टेबल]\label{def:g10:algebra:signtable}
ए \emph{साइन टेबल}\index{साइन टेबल} रिकॉर्ड, प्रति कारक एक पंक्ति पर,
चिह्न ($+$ या $-$) अभिव्यक्ति के प्रत्येक कारक के रूप में $ x $ पर चलाता है $\R $,
के साथ $0$ प्रत्येक मूल्य पर जहां कारक गायब हो जाता है; एक अंतिम पंक्ति देती है
कॉलम दर कॉलम चिह्नों के नियम का उपयोग करते हुए उत्पाद का चिह्न।
\end{definition}

\begin{example}\label{ex:g10:algebra:signtable}
हस्ताक्षर करना $ P(x) = (x - 2)(3 - x)$. कारक $ x - 2$ पर गायब हो जाता है $2$ और है
बाद में सकारात्मक; कारक $3 - x $ पर गायब हो जाता है $3$ और पहले सकारात्मक है:
\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{c|ccccc}
$ x $ & $-\infty $ \quad & $2$ & & $3$ & \quad $+\infty $ \\
\hline
$ x - 2$ & $-$ & $0$ & $+$ & & $+$ \\
$3 - x $ & $+$ & & $+$ & $0$ & $-$ \\
\hline
$ P(x)$ & $-$ & $0$ & $+$ & $0$ & $-$ \\
\end{tabular}
\end{center}
इसलिए $ P(x) > 0$ बिलकुल चालू $\intoo{2}{3}$, और $ P(x) \leq 0$ पर
$\intoc{-\infty}{2} \cup \intco{3}{+\infty}$.
\end{example}

\begin{method}[एक संकेत तालिका के साथ असमानता को हल करना]\label{met:g10:algebra:signtable}
जैसे एक असमानता को हल करने के लिए $ E(x) \geq 0$ या $ E(x) < 0$:
\begin{enumerate}
  \item हर चीज़ को बाईं ओर लाएँ, इसलिए दाहिनी ओर है
        $0$ (कभी भी दोनों पक्षों को ऐसे व्यंजक से गुणा न करें जिसका चिह्न है
        अज्ञात);
  \item बायीं ओर का गुणनखंड करें, या यदि इसे सामान्य हर के ऊपर रखें
        यह एक भागफल है;
  \item का निर्माण करें \omterm{def:g10:algebra:signtable}{साइन टेबल} प्रति कारक एक पंक्ति के साथ (भागफल के लिए, चिह्न लगाएं
        अंतिम पंक्ति में डबल बार के साथ निषिद्ध मान);
  \item पढ़ो \omterm{def:g10:numbers:interval}{अंतराल} अनुरोधित चिह्न कहां है, इसकी जांच की जा रही है
        क्या प्रत्येक सीमा शामिल है।
\end{enumerate}
\end{method}

\begin{example}\label{ex:g10:algebra:quotientineq}
हल करना $\dfrac{x + 1}{x - 3} \leq 0$. अंश-गणक गायब हो जाता है $-1$, द
हर पर $3$ (निषिद्ध मूल्य). भागफल ऋणात्मक होता है जब
दो के विपरीत लक्षण होते हैं, जो होता है $ x $ बीच में $-1$ और $3$.
जिसमें वह सीमा भी शामिल है जहां \emph{मीटर} गायब हो जाता है लेकिन छोड़कर
निषिद्ध मान: समाधान सेट है $\intco{-1}{3}$.
\end{example}

\begin{exercise}[$\star\star $]\label{exo:g10:algebra:7}
हल करना $ x^2 = 5x $, तब $(2x + 1)^2 = (x - 4)^2$. (दूसरे के लिए,
सभी चीज़ों को बायीं ओर लाएँ और वर्गों के अंतर का गुणनखंड करें।)
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:algebra:8}
ए का उपयोग करना \omterm{def:g10:algebra:signtable}{साइन टेबल}, हल करना
\[
(x + 2)(4 - x) > 0,
\qquad
\frac{2x - 6}{x + 1} \geq 0 .
\]
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:algebra:9}
एक स्ट्रीमिंग सेवा की लागत $9$ प्रति माह, साथ ही एकमुश्त साइनअप शुल्क
$15$; एक प्रतिद्वंद्वी लागत $12$ प्रति माह बिना किसी शुल्क के। कितने महीने बाद होता है
पहली सेवा कुल मिलाकर सस्ता विकल्प बन गई? सेट अप करें और हल करें
असमानता.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:algebra:10}
हल करना $\dfrac{x+3}{x-1} = 2$. (दोनों पक्षों को इससे गुणा करें $ x - 1$ के बाद
निषिद्ध मूल्य को छोड़कर, या सब कुछ बाईं ओर ले आओ
सामान्य भाजक.)
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g10:algebra:11}
होने देना $ n $ एक हो पूर्णांक. बताते हैं कि $(n + 1)^2 - n^2$ हमेशा अजीब होता है, और वह
दो क्रमागत विषम संख्याओं के वर्गों के बीच का अंतर है
हमेशा का एक गुणक $8$.
\end{exercise}

\section{समस्या: बेबीलोन से स्वर्णिम अनुपात तक}

\begin{problem}[{सप्ताहांत समस्या---योग-और-उत्पाद समस्याएँ, जिस तरह से अल-ख्वारिज्मी ने इसे बनाया, उसी तरह से वर्ग को पूरा करना, और समीकरण $ x^2 = x + 1$}]
\label{pb:g10:algebra:1}
चार हजार साल पहले, बेबीलोन के शास्त्रियों ने इसे मिट्टी पर हल किया था:
\emph{दो संख्याएँ ज्ञात करें, उनका योग और उनका गुणनफल ज्ञात करें}. उनके
तरकीब --- चारों ओर सममित रूप से संख्याओं को देखें
अर्ध-योग --- अभी भी प्रत्येक द्विघात में दरार डालता है \omterm{def:g10:algebra:equation}{समीकरण} आज, के अंतर्गत
इसका आधुनिक नाम, \emph{वर्ग पूरा करना}. यह समस्या
तरकीब सीखता है, अल-ख्वारिज्मी को सचमुच पूरा करते हुए देखता है
वर्ग, और कार्य करने की विधि को सबसे अधिक प्रसिद्ध बनाता है
उन सभी का अनुपात.

\medskip
\noindent\textbf{Part I --- The Babylonian problem.}
\begin{enumerate}
  \item जोश में आना (\cref{thm:g10:algebra:identities}): कारक
        $ x^2 - 10x + 25$, तब $9x^2 - 49$, फिर --- एक संयोजन
        वर्गों के अंतर से पहचान ---
        $ x^2 + 6x + 9 - 4$.
  \item हल करना $(x + 1)(x + 5) = 0$ और
        $(3x - 7)(3x + 7) = 0$
        (\cref{thm:g10:algebra:zeroproduct}).
  \item लेखक की समस्या: दो संख्याओं का योग होता है $10$ और
        उत्पाद $21$. बेबीलोन का अनुसरण करते हुए उन्हें इस प्रकार लिखें $5 - t $
        और $5 + t $ (आधे योग के चारों ओर सममित) और खोजें $ t $,
        फिर दो नंबर.
  \item वही विधि: योग $14$, उत्पाद $33$.
  \item सामान्य नुस्खा: कुल मिलाकर $ s $ और उत्पाद $ p $, दिखाओ
        कि दो नंबर हैं
        \[
        \frac s2 - t
        \quad\text{and}\quad
        \frac s2 + t,
        \qquad\text{where}\quad
        t^2 = \left(\frac s2\right)^2 - p ,
        \]
        और शर्त बताएं $ s $ और $ p $ समस्या के लिए
        एक समाधान है
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Completing the square.}
\begin{enumerate}[resume]
  \item बढ़ाना $(x - a)(x - b)$ और निष्कर्ष निकालें: के समाधान
        $ x^2 - sx + p = 0$ बिल्कुल संख्याओं के जोड़े हैं
        योग $ s $ और उत्पाद $ p $. (शास्त्री समाधान कर रहे थे
        द्विघात \omterm{def:g10:algebra:equation}{समीकरण} इसे जाने बिना।)
  \item सत्यापित करें कि $ x^2 - 10x + 21 = (x - 5)^2 - 4$, और हल करें
        $ x^2 - 10x + 21 = 0$ द्वारा \omterm{def:g10:algebra:expand}{फैक्टरिंग} दाहिनी ओर एक के रूप में
        वर्गों का अंतर. प्रश्न 3 से तुलना करें.
  \item हल करना $ x^2 + 6x - 7 = 0$ वर्ग पूरा करके.
  \item में वर्ग पूरा करें $ x^2 + 4x + 5$, और समझाएं क्यों
        द \omterm{def:g10:algebra:equation}{समीकरण} $ x^2 + 4x + 5 = 0$ है \emph{नहीं} असली
        समाधान. सामान्य मानदंड बताएं: एक बार इस प्रकार लिखा जाए
        $(x + h)^2 + k $, द्विघात अभिव्यक्ति कब होती है
        गायब हो जाओ?
  \item नाम शाब्दिक है. नौवीं सदी के बगदाद में,
        अल-ख़्वारिज़्मी ने आकर्षित किया $ x^2 + 6x = 7$ भुजा के एक वर्ग के रूप में $ x $
        दो के साथ $3 \times x $ इसके दो हिस्सों से आयतें चिपकी हुई हैं
        पक्ष, और \emph{वर्ग पूरा किया} लापता के साथ
        $3 \times 3$ कोना। चित्र बनाइये और समझाइये कि यह कैसा है
        बदल देता है \omterm{def:g10:algebra:equation}{समीकरण} में $(x + 3)^2 = 16$ --- तो हल करें.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Signs, tables and two old
promises.}
\begin{enumerate}[resume]
  \item का निर्माण करें \omterm{def:g10:algebra:signtable}{साइन टेबल} का $(x - 3)(x + 2)$
        (\cref{def:g10:algebra:signtable}) और हल करें
        $(x - 3)(x + 2) < 0$.
  \item हल करना $ x^2 \leq 10x - 21$ (सब कुछ एक तरफ ले आओ
        और प्रश्न 7 का पुन: उपयोग करें)।
  \item हल करना $\dfrac{x - 1}{x + 2} \geq 0$ के साथ \omterm{def:g10:algebra:signtable}{साइन टेबल},
        निषिद्ध मूल्य पर विचार करना।
  \item एक पुराना वादा निभाया गया: किसानों के लिए
        रानी डिडो की बाड़ समस्या, मिडिल स्कूल खंड में, के साथ $20$ बाड़ का मीटर, क्षेत्र
        एक का $ x \times (10 - x)$ कलम संतुष्ट करती है
        \[
        A(x) = x(10 - x) = 25 - (x - 5)^2 .
        \]
        पहचान सत्यापित करें, और इसे पढ़ें \emph{सबूत} का
        ग्रेड 6 केवल अनुमान लगा सकता है: अधिकतम क्षेत्रफल,
        और अद्वितीय $ x $ इसे हासिल करना.
  \item दूसरा: इसे साबित करो $ x + \dfrac1x \geq 2$ हरएक के लिए
        $ x > 0$, केवल समानता के साथ $ x = 1$. (गुणा करके
        $ x $, इकट्ठा करना, एक वर्ग पहचानना ---
        अंकगणित--ज्यामितीय असमानता
        एक नई पोशाक में मिडिल स्कूल वॉल्यूम की यूक्लिड-संबंध सप्ताहांत समस्या।)
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The golden section.}
लंबाई का एक खंड काटें $1$ एक बड़े हिस्से में $ x $ और एक छोटा
भाग $1 - x $ ताकि
\emph{पूर्ण का अर्थ बड़ा है और बड़ा का अर्थ छोटा है}:
$\frac{1}{x} = \frac{x}{1 - x}$.
\begin{enumerate}[resume]
  \item दिखाएँ कि शर्त पढ़ती है $ x^2 + x - 1 = 0$.
  \item वर्ग पूरा करके हल करें और मूल वही रखें
        एक लंबाई है: $ x = \frac{\sqrt5 - 1}{2} \approx 0.618$.
  \item  \emph{सुनहरा अनुपात} है
        $\varphi = \frac1x = \frac{2}{\sqrt5 - 1}$. युक्तिसंगत बनाना
        हर (ऊपर और नीचे से गुणा करें)।
        $\sqrt5 + 1$, काम पर एक पहचान) और दिखाओ
        $\varphi = \frac{\sqrt5 + 1}{2} \approx 1.618$.
  \item सत्यापित करें \emph{बिल्कुल} की दो जादुई पहचान
        $\varphi $:
        \[
        \varphi^2 = \varphi + 1,
        \qquad
        \frac{1}{\varphi} = \varphi - 1 .
        \]
  \item समापन: समझाएँ क्यों $\varphi $ है \omterm{ex:g10:numbers:classify}{तर्कहीन}
        (मिडिल स्कूल वॉल्यूम की अतार्किकता सप्ताहांत समस्या इस बारे में मुख्य तथ्य प्रदान करती है
        $\sqrt5$), फिर अनुपातों की गणना करें $\frac53$,
        $\frac85$, $\frac{13}{8}$, $\frac{21}{13}$ का
        क्रमागत हेमचन्द्र--फाइबोनैचि संख्याएँ
        (मिडिल स्कूल वॉल्यूम की लय-गिनती सप्ताहांत समस्या) तीन दशमलव तक। क्या करें?
        वे पीछा करते दिखते हैं? (इसका प्रमाण है कि उन्होंने इसे पकड़ लिया
        अनुक्रमों पर अध्यायों से संबंधित है।)
\end{enumerate}
\end{problem}
