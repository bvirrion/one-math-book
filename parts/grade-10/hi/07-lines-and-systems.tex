\chapter{रेखाओं और रैखिक प्रणालियों के समीकरण}\label{ch:g10:lines}

निर्देशांक तल में एक सीधी रेखा का वर्णन एक द्वारा किया जाता है \omterm{def:g10:algebra:equation}{समीकरण}, और
दो रेखाएँ मिलकर एक प्रणाली बनाती हैं \omterm{def:g10:algebra:equation}{समीकरण} जिनके समाधान उनके हैं
\omterm{def:g10:numbers:interunion}{चौराहा} अंक. यह अध्याय तीन भाषाओं को जोड़ता है --- ज्यामितीय
(रेखाएँ), बीजगणितीय (\omterm{def:g10:algebra:equation}{समीकरण}) और कार्यात्मक (एफ़िन फ़ंक्शंस से
\cref{ch:g10:reffunc}) --- और हल करने के दो मानक तरीके सिखाता है
रैखिक प्रणाली.

\section{एक रेखा का समीकरण}

\begin{theorem}[घटा हुआ समीकरण]\label{thm:g10:lines:reduced}
में एक \omterm{def:g10:coordgeom:system}{निर्देशांक तरीका}:
\begin{enumerate}
  \item प्रत्येक गैर-ऊर्ध्वाधर रेखा में एक होता है \omterm{def:g10:algebra:equation}{समीकरण} रूप का
        \[
        y = mx + p ,
        \]
        साथ $m $ (द \emph{ढलान}\index{ढलान}) और $ p $ (द
        \emph{$ y $संवाद}) विशिष्ट रूप से निर्धारित --- यह है
        \emph{घटा हुआ समीकरण}\index{घटा हुआ समीकरण} पंक्ति का;
  \item प्रत्येक ऊर्ध्वाधर रेखा में एक होता है \omterm{def:g10:algebra:equation}{समीकरण} $ x = c $;
  \item एक बिंदु ठीक उसी समय रेखा पर स्थित होता है जब वह होता है \omterm{def:g10:coordgeom:system}{COORDINATES} को संतुष्ट करें
        \omterm{def:g10:algebra:equation}{समीकरण}.
\end{enumerate}
\end{theorem}
\admitted

\begin{proposition}[दो बिंदुओं से होकर ढलान]\label{prop:g10:lines:slope}
 \omterm{def:g10:reffunc:affine}{ढलान} (गैर-ऊर्ध्वाधर) रेखा के माध्यम से $ A(x_A, y_A)$ और
$ B(x_B, y_B)$, साथ $ x_A \neq x_B $, है
\[
m = \frac{y_B - y_A}{x_B - x_A}
\qquad
\text{(``change in $ y $ over change in $ x $'').}
\]
\end{proposition}

\begin{proof}
दोनों बातें संतुष्ट करती हैं $ y = mx + p $, इसलिए $ y_B - y_A = (mx_B + p) - (mx_A + p)
= m(x_B - x_A)$; द्वारा विभाजित करें $ x_B - x_A \neq 0$.
\end{proof}

\begin{method}[दो बिंदुओं से होकर जाने वाली रेखा का समीकरण ज्ञात करना]\label{met:g10:lines:through}
दिया गया $ A $ और $ B $ साथ $ x_A \neq x_B $:
\begin{enumerate}
  \item की गणना करें \omterm{def:g10:reffunc:affine}{ढलान} $ m = \dfrac{y_B - y_A}{x_B - x_A}$;
  \item लिखना $ y = mx + p $ और स्थानापन्न करें \omterm{def:g10:coordgeom:system}{COORDINATES} का $ A $ (या $ B $)
        ढूँढ़ने के लिए $ p $;
  \item अंतिम जांचें \omterm{def:g10:algebra:equation}{समीकरण} साथ \emph{अन्य} बिंदु।
\end{enumerate}
अगर $ x_A = x_B $, रेखा लंबवत है: इसका \omterm{def:g10:algebra:equation}{समीकरण} है $ x = x_A $.
\end{method}

\begin{example}\label{ex:g10:lines:through}
के माध्यम से रेखा $ A(1, 5)$ और $ B(3, 1)$. \omterm{def:g10:reffunc:affine}{ढलान}:
$ m = \frac{1 - 5}{3 - 1} = \frac{-4}{2} = -2$. तब $ y = -2x + p $; द
बिंदु $ A $ देता है $5 = -2 \times 1 + p $, इसलिए $ p = 7$. \omterm{def:g10:algebra:equation}{समीकरण}:
$ y = -2x + 7$. के साथ जांच $ B $: $-2 \times 3 + 7 = 1$.
\end{example}

\begin{proposition}[समानांतर रेखाएँ]\label{prop:g10:lines:parallel}
दो गैर-ऊर्ध्वाधर रेखाएँ समानांतर होती हैं यदि और केवल यदि उनमें समानता हो
\omterm{def:g10:reffunc:affine}{ढलान}. दो अलग-अलग समानांतर रेखाओं में कोई उभयनिष्ठ बिंदु नहीं है; दो पंक्तियों के साथ
अलग \omterm{def:g10:reffunc:affine}{ढलान}ों बिल्कुल एक है.
\end{proposition}

\begin{proof}
होने देना $ y = mx + p $ और $ y = m'x + p'$ दो पंक्तियाँ हो. एक सामान्य बात
हल करता है $ mx + p = m'x + p'$, यानी\ $(m - m')x = p' - p $. अगर $ m \neq m'$,
इसका बिल्कुल एक ही समाधान है $ x = \frac{p' - p}{m - m'}$: एक
\omterm{def:g10:numbers:interunion}{चौराहा} बिंदु। अगर $ m = m'$, द \omterm{def:g10:algebra:equation}{समीकरण} पढ़ता $0 = p' - p $: नहीं
समाधान कब $ p \neq p'$ (विशिष्ट समानताएं), प्रत्येक $ x $ कब $ p = p'$
(वही पंक्ति). समांतरता (नहीं होना) \omterm{def:g10:numbers:interunion}{चौराहा}, या बराबर होना) इस प्रकार है
के बराबर $ m = m'$.
\end{proof}

\begin{remark}[दिशा सदिश]
रेखा $ y = mx + p $ ऊपर चला जाता है $ m $ दाईं ओर प्रत्येक इकाई के लिए इकाइयाँ, इसलिए
\omterm{def:g10:vectors:vector}{वेक्टर} $\vec u\,(1, m)$ एक है \emph{दिशा
वेक्टर}\index{दिशा
वेक्टर} रेखा के: रेखा के समानांतर है
$\vec u $. दो रेखाएँ बिल्कुल अपनी दिशा के समानान्तर होती हैं वैक्टर हैं
\omterm{def:g10:vectors:collinear}{समरेख} (\cref{thm:g10:vectors:det}).
\end{remark}

\section{रैखिक प्रणाली}

\begin{definition}[रैखिक प्रणाली]\label{def:g10:lines:system}
ए \emph{दो रैखिक समीकरणों की प्रणाली}\index{दो रैखिक समीकरणों की प्रणाली} में
अज्ञात $ x $ और $ y $ की एक \omterm{def:g10:vectors:sum}{जोड़}ी है \omterm{def:g10:algebra:equation}{समीकरण}
\[
\begin{cases}
a x + b y = c \\
a' x + b' y = c'
\end{cases}
\]
संतोष होना \emph{इसके साथ ही}. ए \emph{समाधान} एक \omterm{def:g10:vectors:sum}{जोड़}ी है
$(x, y)$ दोनों को संतुष्ट करना. प्रत्येक \omterm{def:g10:algebra:equation}{समीकरण} एक पंक्ति का वर्णन करता है, इसलिए हल करना
सिस्टम का मतलब है खोजना \omterm{def:g10:numbers:interunion}{चौराहा} दो पंक्तियों का.
\end{definition}

\begin{method}[प्रतिस्थापन]\label{met:g10:lines:substitution}
\begin{enumerate}
  \item एक को सुलझाओ \omterm{def:g10:algebra:equation}{समीकरण} किसी अज्ञात के लिए --- सबसे आसान चुनें, उदा.\
        व्यक्त करें $ y $ के अनुसार $ x $;
  \item इस अभिव्यक्ति को इसमें प्रतिस्थापित करें \emph{अन्य} \omterm{def:g10:algebra:equation}{समीकरण}, जो
        अब केवल शामिल है $ x $;
  \item के लिए हल करें $ x $, फिर गणना करें $ y $ चरण 1 से;
  \item दोनों मूल में \omterm{def:g10:vectors:sum}{जोड़}ी की जाँच करें \omterm{def:g10:algebra:equation}{समीकरण}.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g10:lines:substitution}
हल करना $\begin{cases} x + 2y = 8 \\ 3x - y = 3 \end{cases}$ द्वारा
प्रतिस्थापन. दूसरा \omterm{def:g10:algebra:equation}{समीकरण} देता है $ y = 3x - 3$. में प्रतिस्थापित करें
प्रथम:
\[
x + 2(3x - 3) = 8
\ \Longleftrightarrow\
x + 6x - 6 = 8
\ \Longleftrightarrow\
7x = 14
\ \Longleftrightarrow\
x = 2 ,
\]
तब $ y = 3 \times 2 - 3 = 3$. समाधान \omterm{def:g10:vectors:sum}{जोड़}ी है $(2, 3)$.
जाँच करना: $2 + 2 \times 3 = 8$ और $3 \times 2 - 3 = 3$.
\end{example}

\begin{method}[उन्मूलन]\label{met:g10:lines:elimination}
\begin{enumerate}
  \item प्रत्येक को गुणा करें \omterm{def:g10:algebra:equation}{समीकरण} एक अच्छी तरह से चुनी गई संख्या से ताकि एक अज्ञात हो
        दोनों में विपरीत गुणांक प्राप्त होता है \omterm{def:g10:algebra:equation}{समीकरण};
  \item \omterm{def:g10:vectors:sum}{जोड़}ें \omterm{def:g10:algebra:equation}{समीकरण}: वह अज्ञात मिट जाता है;
  \item शेष एक-अज्ञात को हल करें \omterm{def:g10:algebra:equation}{समीकरण}, फिर वापस प्रतिस्थापित करें
        दूसरे अज्ञात को खोजो;
  \item दोनों मूल में \omterm{def:g10:vectors:sum}{जोड़}ी की जाँच करें \omterm{def:g10:algebra:equation}{समीकरण}.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g10:lines:elimination}
हल करना $\begin{cases} 2x + 3y = 7 \\ 5x - 2y = 8 \end{cases}$ द्वारा
उन्मूलन. खत्म करने के लिए $ y $, पहले को गुणा करें \omterm{def:g10:algebra:equation}{समीकरण} द्वारा $2$ और
दूसरा द्वारा $3$:
\[
\begin{cases}
4x + 6y = 14 \\
15x - 6y = 24
\end{cases}
\]
उन्हें जोड़ना: $19x = 38$, इसलिए $ x = 2$. में प्रतिस्थापित करना $2x + 3y = 7$:
$4 + 3y = 7$, इसलिए $ y = 1$. समाधान: $(2, 1)$. दूसरे में जांचें
\omterm{def:g10:algebra:equation}{समीकरण}: $5 \times 2 - 2 \times 1 = 8$.
\end{example}

\begin{remark}[कितने समाधान?]
दो पंक्तियों की तरह, ए रैखिक प्रणाली सामान्यतः इसका एक ही समाधान है
(गैर-समानांतर रेखाएं), और असाधारण रूप से कोई नहीं (अलग-अलग समानांतर रेखाएं) या
अपरिमित रूप से अनेक (एक ही पंक्ति में दो बार)। की कसौटी
\cref{thm:g10:vectors:det} निर्णय लेता है: सिस्टम के पास एक अनूठा समाधान है
बिल्कुल कब $ ab' - a'b \neq 0$.
\end{remark}

\begin{example}[एक सिस्टम के साथ मॉडलिंग]\label{ex:g10:lines:modeling}
तीन नोटबुक और दो पेन की लागत $8$; एक नोटबुक और चार पेन की कीमत
$6$. होने देना $ x $ एक नोटबुक की कीमत हो और $ y $ वह कलम का:
\[
\begin{cases}
3x + 2y = 8 \\
x + 4y = 6 .
\end{cases}
\]
दूसरे से \omterm{def:g10:algebra:equation}{समीकरण}, $ x = 6 - 4y $; प्रतिस्थापित करना,
$3(6 - 4y) + 2y = 8$, इसलिए $18 - 10y = 8$, देना $ y = 1$ और तब
$ x = 2$. एक नोटबुक की कीमत है $2$ और एक पेन की कीमत $1$.
\end{example}

\section{अभ्यास}

\begin{exercise}[$\star $]\label{exo:g10:lines:1}
प्रत्येक पंक्ति के लिए, पढ़ें \omterm{def:g10:reffunc:affine}{ढलान} और यह $ y $-अवरोधन करें, और इसे स्केच करें:
\[
y = 2x - 3, \qquad
y = -\tfrac13 x + 1, \qquad
y = 4, \qquad
x = -2 .
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g10:lines:2}
क्या बात है $ A(2, 7)$ लाइन के हैं $ y = 3x + 1$? और $ B(-1, -1)$?
और बात $ C(4, 13)$?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g10:lines:3}
खोजें \omterm{thm:g10:lines:reduced}{घटा हुआ समीकरण} के माध्यम से लाइन का:
\[
\text{(a) } A(0, 2) \text{ and } B(3, 8);
\qquad
\text{(b) } A(-1, 5) \text{ and } B(2, -4);
\qquad
\text{(c) } A(4, 1) \text{ and } B(4, 6).
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g10:lines:4}
पंक्तियों के बीच $ y = 3x - 1$, $ y = -3x + 2$, $ y = 3x + 5$,

\begin{exercise}[$\star\star $]\label{exo:g10:lines:9}
होने देना $ d $ पंक्ति हो $ y = 2x - 3$ और $ A(1, 4)$.
\begin{enumerate}
  \item दे दो \omterm{def:g10:algebra:equation}{समीकरण} पंक्ति का $ d'$ के समानांतर $ d $ के माध्यम से गुजरते हुए $ A $.
  \item की गणना करें \omterm{def:g10:numbers:interunion}{चौराहा} का बिंदु $ d'$ साथ $ x $-अक्ष.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g10:lines:10}
होने देना $ A(0, 0)$, $ B(6, 2)$ और $ C(2, 4)$.
\begin{enumerate}
  \item खोजें \omterm{def:g10:algebra:equation}{समीकरण} त्रिभुज की माध्यिकाओं का $ ABC $ से जारी किया गया $ A $ और से $ B $ (एक माध्यिका एक शीर्ष से जुड़ती है \omterm{prop:g10:coordgeom:midpoint}{मध्य} का
        विपरीत पक्ष)।
  \item उनकी गणना करें \omterm{def:g10:numbers:interunion}{चौराहा} बिंदु $ G $, और उसकी जांच करें
        $ G $ तीसरे माध्यिका पर भी है। (आपको यह पता लगाना चाहिए कि
        \omterm{def:g10:coordgeom:system}{COORDINATES} का $ G $ का औसत है $ A $, $ B $, $ C $.)
\end{enumerate}
\end{exercise}

\section{समस्या: एक समाधान, कोई नहीं, या अनंत रूप से अनेक}

\begin{problem}[{सप्ताहांत समस्या--- दो पंक्तियों में तीन संभव हैं नियति, एक निर्धारक उनके और दुनिया के बीच निर्णय लेता है सबसे पुरानी पाठ्यपुस्तक पहले से ही ज्ञात थी}]
\label{pb:g10:lines:1}
दो रैखिक \omterm{def:g10:algebra:equation}{समीकरण}, दो अज्ञात: वे जो रेखाएँ खींचते हैं
एक बार पार कर सकता है, कभी नहीं, या एक ही रेखा हो सकता है --- और
\emph{प्रत्येक} रैखिक प्रणाली इन तीन नियति में से एक को विरासत में मिला है।
यह समस्या उन्हें वर्गीकृत करती है, उस संख्या को पूरा करती है जो निर्णय लेती है (ए)।
से पुराना परिचय है वैक्टर अध्याय), हल करता है
चीनियों के दो हज़ार साल पुराने तीतर और खरगोश
\emph{नौ अध्याय}, और क्षेत्रों के कोनों के साथ समाप्त होता है ---
प्रत्येक एयरलाइन और फ़ैक्टरी द्वारा उपयोग किए जाने वाले अनुकूलन का पहला चरण
आज.

\medskip
\noindent\textbf{Part I --- The line, fluently.}
\begin{enumerate}
  \item खोजें \omterm{thm:g10:lines:reduced}{घटा हुआ समीकरण} लाइन के माध्यम से
        $(1, 2)$ और $(3, 8)$
        (\cref{met:g10:lines:through}).
  \item के बीच $ y = 3x - 1$, $ y = 3x + 4$ और $ y = -x + 7$: जो
        दो रेखाएं समानांतर हैं
        (\cref{prop:g10:lines:parallel}), और कहाँ करते हैं
        गैर-समानांतर \omterm{def:g10:vectors:sum}{जोड़}े मिलते हैं? (एक गणना करें \omterm{def:g10:numbers:interunion}{चौराहा}.)
  \item के माध्यम से लाइन $(2, 1)$ और $(2, 5)$: ऐसा क्यों है
        नहीं \omterm{thm:g10:lines:reduced}{घटा हुआ समीकरण} $ y = mx + p $, और यह क्या है
        \omterm{def:g10:algebra:equation}{समीकरण}?
  \item मुद्दा यह है $(4, 11)$ लाइन पर $ y = 3x - 1$? और
        $(-1, -5)$?
  \item के माध्यम से लाइन $(0, 3)$ साथ \omterm{def:g10:reffunc:affine}{ढलान} $-2$: इसे दे दो
        \omterm{def:g10:algebra:equation}{समीकरण}, इसके दो अक्ष अंतःखंड, और का क्षेत्रफल
        त्रिभुज को यह पहले चतुर्थांश से काटता है।
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Three destinies.}
\begin{enumerate}[resume]
  \item प्रतिस्थापन द्वारा हल करें
        (\cref{met:g10:lines:substitution}):
        $\begin{cases} y = 2x - 3 \\ 3x + y = 7 \end{cases}$
  \item निराकरण द्वारा हल करें
        (\cref{met:g10:lines:elimination}):
        $\begin{cases} 2x + 3y = 8 \\ 5x - 3y = -1 \end{cases}$
  \item वर्गीकृत करें --- और पंक्तियों के साथ व्याख्या करें --- दोनों
        सिस्टम
        \[
        \begin{cases} 2x - y = 3 \\ 4x - 2y = 6 \end{cases}
        \qquad\text{and}\qquad
        \begin{cases} 2x - y = 3 \\ 4x - 2y = 1 . \end{cases}
        \]
        संपूर्ण ट्राइकोटॉमी बताएं: क्रॉसिंग, समानांतर,
        समरूप --- एक समाधान, कोई नहीं, अनंत रूप से अनेक।
  \item सामान्य व्यवस्था के लिए $ ax + by = e $, $ cx + dy = f $,
        निर्णायक संख्या है $ ad - bc $ --- वही
        के संरेखता परीक्षण के रूप में निर्धारक
        \cref{pb:g10:vectors:1}. संयोग स्पष्ट करें (जो
        दो वैक्टर हैं \omterm{def:g10:vectors:collinear}{समरेख} ठीक उसी समय जब रेखाएँ होती हैं
        समानांतर या समान?), और मूल्यांकन करें $ ad - bc $ के लिए
        प्रश्न 7 और 8 की तीन प्रणालियाँ।
  \item किस मूल्य के लिए $ m $ करता है
        $\begin{cases} x + my = 3 \\ 2x + 6y = 7 \end{cases}$
        कोई समाधान नहीं है? अन्य सभी मानों के लिए सिस्टम को हल करें
        का $ m $\,\dots\ या कम से कम समझाएं कि आप कैसे करेंगे।
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The Nine Chapters.}
\begin{enumerate}[resume]
  \item से \emph{गणितीय कला पर नौ अध्याय}
        (चीन, लगभग 2000 वर्ष पहले): एक पिंजरे में तीतर रहते हैं
        और खरगोश --- $35$ सिर और $94$ कुल मिलाकर पैर. कैसे
        प्रत्येक में से अनेक?
  \item एक जूस स्टैंड मिश्रण करता है $60\,\%$-फ्रूट सिरप के साथ
        $20\,\%$-फल पेय प्राप्त करने के लिए $30$ एल पर $35\,\%$
        फल। प्रत्येक कितने लीटर?
  \item दो अंकों की संख्या में अंकों का योग होता है $12$; इसकी अदला-बदली
        अंक इसे कम कर देते हैं $18$. अंक का उपयोग करके इसे खोजें
        सिस्टम स्थापित करने के लिए मिडिल स्कूल वॉल्यूम के अंक-बीजगणित सप्ताहांत समस्या का बीजगणित।
  \item बेकरी पर, $3$ कॉफ़ी और $2$ क्रोइसैन्ट लागत
        $9.10$ यूरो; $2$ कॉफ़ी और $3$ क्रोइसैन्ट लागत
        $8.90$ यूरो. सुरुचिपूर्ण ढंग से हल करें: क्या करें \emph{जोड़ना} दो \omterm{def:g10:algebra:equation}{समीकरण} और \emph{घटाने} उनमें से प्रत्येक तुम्हें बताता है
        सीधे?
  \item एक बाज़ार स्टॉल का दावा है: $2$ सेब $+ 1$ नाशपाती के लिए
        $5$ यूरो, और $4$ सेब $+ 2$ नाशपाती के लिए $9$ यूरो.
        हल करें --- या यों कहें कि बताएं कि सिस्टम की नियति क्या है
        स्टाल के अंकगणित के बारे में पता चलता है।
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Corners rule.}
\begin{enumerate}[resume]
  \item रेखा $ y = 3x - 1$ विमान को दो भागों में विभाजित करता है। जो
        ओर मूल है? सेट का वर्णन करें
        $ y > 3x - 1$ और कोई इसके विरुद्ध एक बिंदु का परीक्षण कैसे करता है।
  \item चार बाधाओं द्वारा परिभाषित क्षेत्र बनाएं
        \[
        x \geq 0, \qquad y \geq 0, \qquad x + y \leq 4,
        \qquad y \leq 2x + 1,
        \]
        और इसके चार कोने बिंदुओं की गणना करें।
  \item एक कार्यशाला का लाभ है $ P = 3x + 2y $ के क्षेत्र पर
        प्रश्न 17. मूल्यांकन करें $ P $ चारों कोनों पर. स्वीकार कर रहा हूँ
        वह \omterm{def:g10:functions:extrema}{अधिकतम} ऐसे पर ऐसी रैखिक मात्रा का
        क्षेत्र हमेशा एक कोने पर पहुंचता है (रेखाओं का चित्र बनाएं)।
        $3x + 2y = $ निरंतर स्वीपिंग), इष्टतम दें
        उत्पादन योजना.
  \item एक प्रणाली के रूप में मिडिल स्कूल वॉल्यूम की दो-थर्मामीटर सप्ताहांत समस्या की दो टैक्सियाँ:
        लिखो और हल करो
        $\begin{cases} y = 1.5x + 3 \\ y = 2x \end{cases}$ और
        समाधान की व्याख्या करें.
  \item समापन, शब्दकोश: एक \omterm{def:g10:algebra:equation}{समीकरण} $\leftrightarrow $
        एक लकीर; एक प्रणाली $\leftrightarrow $ दो पंक्तियों के साथ
        तीन नियति, द्वारा तय की गई $ ad - bc $; अनेक असमानताएँ
        $\leftrightarrow$ एक क्षेत्र जिसके कोने ले जाते हैं
        इष्टतम. इसे तीन वाक्यों में बताएं --- आपके पास बस है
        के प्रवेश कक्ष का भ्रमण किया \emph{लीनियर अलजेब्रा} और
        \emph{रैखिक प्रोग्रामिंग}, दोनों पूर्ण रूप से निर्मित हैं
        विश्वविद्यालय की मात्रा.
\end{enumerate}
\end{problem}
