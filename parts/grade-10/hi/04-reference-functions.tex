\chapter{संदर्भ कार्य}\label{ch:g10:reffunc}

एक मुट्ठी सरल कार्य --- एफ़िन, वर्ग, व्युत्क्रम, वर्गमूल,
घन --- हर जगह दिखाई देते हैं, अकेले या संयुक्त। उनका जानना रेखांकन और
विभिन्नताओं को हृदयंगम करने से अनेक समस्याएँ एक त्वरित रेखाचित्र में बदल जाती हैं। यह अध्याय
उनमें से प्रत्येक का बारी-बारी से अध्ययन करता है और उनकी विविधताओं को सिद्ध करता है
की तुलना तकनीक \cref{ch:g10:functions}.

\section{एफ़िन फ़ंक्शन}

\begin{definition}[एफ़िन फ़ंक्शन]\label{def:g10:reffunc:affine}
एक \emph{एफ़िन फ़ंक्शन}\index{एफ़िन फ़ंक्शन} एक है \omterm{def:g10:functions:function}{समारोह} रूप का
\[
f(x) = mx + p,
\]
कहाँ $m $ और $ p $ फिक्स किए गए हैं \omterm{def:g10:numbers:sets}{वास्तविक संख्या}. इसका \omterm{def:g10:functions:graph}{ग्राफ} एक सीधी रेखा है:
$ m $ है \emph{ढलान}\index{ढलान} और $ p $ 
\emph{$ y $संवाद}\index{$ y $संवाद} (रेखा पार हो जाती है
ऊर्ध्वाधर अक्ष पर $(0, p)$). कब $ p = 0$  \omterm{def:g10:functions:function}{समारोह} है
\emph{रेखीय}: $ f(x) = mx $ आनुपातिकता व्यक्त करता है.
\end{definition}

\begin{proposition}[ढलान और विविधताएँ]\label{prop:g10:reffunc:affine}
होने देना $ f(x) = mx + p $.
\begin{enumerate}
  \item किन्हीं दो भिन्न इनपुट के लिए $ u \neq v $:
        $ m = \dfrac{f(v) - f(u)}{v - u}$ (द \omterm{def:g10:reffunc:affine}{ढलान} का परिवर्तन है
        इनपुट के प्रति यूनिट परिवर्तन पर आउटपुट)।
  \item अगर $ m > 0$, $ f $ सख्ती से है \omterm{def:g10:functions:variations}{की बढ़ती} पर $\R $; अगर $ m < 0$,
        कठोरता से \omterm{def:g10:functions:variations}{घटते}; अगर $ m = 0$, स्थिर।
\end{enumerate}
\end{proposition}

\begin{proof}
1. गणना करें $ f(v) - f(u) = (mv + p) - (mu + p) = m(v - u)$, फिर से विभाजित करें
$ v - u \neq 0$.

2. यदि $ u < v $, तब $ f(v) - f(u) = m(v-u)$ का चिन्ह है $ m $, तब से
$ v - u > 0$: सकारात्मक $ m $ देता है $ f(u) < f(v)$ (\omterm{def:g10:functions:variations}{की बढ़ती}), नकारात्मक $ m $
देता है $ f(u) > f(v)$ (\omterm{def:g10:functions:variations}{घटते}).
\end{proof}

\begin{example}\label{ex:g10:reffunc:affine}
खोजें \omterm{def:g10:reffunc:affine}{एफ़िन फ़ंक्शन} किसका \omterm{def:g10:functions:graph}{ग्राफ} से गुजरता है $ A(1, 3)$ और
$ B(4, 9)$. \omterm{def:g10:reffunc:affine}{ढलान} है
\[
m = \frac{9 - 3}{4 - 1} = \frac{6}{3} = 2 .
\]
तब $ f(x) = 2x + p $, और $ f(1) = 3$ देता है $2 + p = 3$, इसलिए $ p = 1$:
$ f(x) = 2x + 1$. के साथ जांच $ B $: $ f(4) = 9$.
\end{example}

\section{वर्गाकार फलन}

\begin{proposition}[चौकोर समारोह]\label{prop:g10:reffunc:square}
 \omterm{def:g10:functions:function}{समारोह} $ f(x) = x^2$, पर परिभाषित किया गया $\R $:
\begin{enumerate}
  \item सख्ती से है \omterm{def:g10:functions:variations}{घटते} पर $\intoc{-\infty}{0}$ और सख्ती से
        \omterm{def:g10:functions:variations}{की बढ़ती} पर $\intco{0}{+\infty}$, साथ \omterm{def:g10:functions:extrema}{न्यूनतम} $0$ पर $ x = 0$;
  \item संतुष्ट $ f(-x) = f(x)$ सभी के लिए $ x $: इसका \omterm{def:g10:functions:graph}{ग्राफ}, ए
        \emph{परवलय}\index{परवलय}, ऊर्ध्वाधर के बारे में सममित है
        अक्ष.
\end{enumerate}
\end{proposition}

\begin{proof}
1. चलो $0 \leq u < v $. तब
\[
f(v) - f(u) = v^2 - u^2 = (v - u)(v + u) > 0,
\]
तब से $ v - u > 0$ और $ v + u > 0$: $ f $ सख्ती से है \omterm{def:g10:functions:variations}{की बढ़ती} पर
$\intco{0}{+\infty}$. अगर $ u < v \leq 0$, तब $ v - u > 0$ लेकिन
$ v + u < 0$, इसलिए $ f(v) - f(u) < 0$: कठोरता से \omterm{def:g10:functions:variations}{घटते}. अंत में
$ x^2 \geq 0 = f(0)$ सभी के लिए $ x $.

2. $(-x)^2 = x^2$; तो अंक $(x, x^2)$ और $(-x, x^2)$, आईना इमेजिस
ऊर्ध्वाधर अक्ष के पार, दोनों पर हैं \omterm{def:g10:functions:graph}{ग्राफ}.
\end{proof}

\begin{remark}[वर्ग और व्यवस्था]
क्योंकि वर्ग \omterm{def:g10:functions:function}{समारोह} वर्ग लेते हुए नकारात्मक पक्ष पर घटता है
\emph{पराजय} ऋणात्मक संख्याओं का क्रम: $-3 < -2$ लेकिन $9 > 4$.
चिह्नों की जांच किए बिना कभी भी असमानता के दोनों पक्षों का वर्ग न करें।
\end{remark}

\section{उलटा कार्य}

\begin{proposition}[उलटा कार्य]\label{prop:g10:reffunc:inverse}
 \omterm{def:g10:functions:function}{समारोह} $ f(x) = \dfrac1x $, के लिए परिभाषित $ x \neq 0$:
\begin{enumerate}
  \item सख्ती से है \omterm{def:g10:functions:variations}{घटते} पर $\intoo{-\infty}{0}$ और सख्ती से
        \omterm{def:g10:functions:variations}{घटते} पर $\intoo{0}{+\infty}$;
  \item संतुष्ट $ f(-x) = -f(x)$: इसका \omterm{def:g10:functions:graph}{ग्राफ}, ए
        \emph{अतिशयोक्ति}\index{अतिशयोक्ति}, मूल के बारे में सममित है।
\end{enumerate}
\end{proposition}

\begin{proof}
1. चलो $0 < u < v $. तब
\[
f(v) - f(u) = \frac1v - \frac1u = \frac{u - v}{uv} < 0,
\]
तब से $ u - v < 0$ और $ uv > 0$: कठोरता से \omterm{def:g10:functions:variations}{घटते} पर
$\intoo{0}{+\infty}$. पर $\intoo{-\infty}{0}$ वही भागफल है
$ u - v < 0$ और $ uv > 0$ फिर से (दो नकारात्मक का उत्पाद), तो $ f $
वहां भी घट जाती है.

2. $\frac{1}{-x} = -\frac1x $.
\end{proof}

\begin{remark}
सावधान: $\frac1x $ है \emph{नहीं} \omterm{def:g10:functions:variations}{घटते} पूरी तरह से \omterm{def:g10:functions:function}{कार्यक्षेत्र}. से
$ u = -1$ को $ v = 1$ मूल्य उछलता है $-1$ तक $1$. दो शाखाएं
अलग से अध्ययन किया जाना चाहिए.
\end{remark}

\section{वर्गमूल और घन}

\begin{proposition}[वर्गमूल फलन]\label{prop:g10:reffunc:sqrt}
 \omterm{def:g10:functions:function}{समारोह} $ f(x) = \sqrt{x}$, पर परिभाषित किया गया $\intco{0}{+\infty}$, है
सख्ती से \omterm{def:g10:functions:variations}{की बढ़ती}.
\end{proposition}

\begin{proof}
होने देना $0 \leq u < v $. से गुणा और भाग करें \emph{संयुग्मित}:
\[
\sqrt v - \sqrt u
= \frac{(\sqrt v - \sqrt u)(\sqrt v + \sqrt u)}{\sqrt v + \sqrt u}
= \frac{v - u}{\sqrt v + \sqrt u} > 0,
\]
तब से $ v - u > 0$ और $\sqrt v + \sqrt u > 0$ (टिप्पणी $ v > 0$, इसलिए
$\sqrt v > 0$).
\end{proof}

\begin{proposition}[घन फलन]\label{prop:g10:reffunc:cube}
 \omterm{def:g10:functions:function}{समारोह} $ f(x) = x^3$, पर परिभाषित किया गया $\R $, सख्ती से है \omterm{def:g10:functions:variations}{की बढ़ती}, और इसके \omterm{def:g10:functions:graph}{ग्राफ} मूल के बारे में सममित है।
\end{proposition}
\admitted

\begin{proposition}[की तुलना $ x $, $ x^2$ और $\sqrt x $]\label{prop:g10:reffunc:compare}
के लिए $0 < x < 1$:\ \ $ x^2 < x < \sqrt x $.\quad
के लिए $ x > 1$:\ \ $\sqrt x < x < x^2$.\quad
पर $ x = 0$ और $ x = 1$ तीन मान बराबर हैं.
\end{proposition}

\begin{proof}
कल्पना करना $0 < x < 1$. गुणा $ x < 1$ द्वारा $ x > 0$ देता है $ x^2 < x $.
अगला, $ x < \sqrt x $: चूँकि दोनों पक्ष सकारात्मक हैं और वर्ग है
\omterm{def:g10:functions:variations}{की बढ़ती} सकारात्मक संख्याओं पर, यह असमानता समतुल्य है
$ x^2 < x $, जिसे हमने अभी साबित किया है। के लिए $ x > 1$, गुणा करना $ x > 1$ द्वारा $ x $
देता है $ x^2 > x $, और $\sqrt x < x $ उसी वर्गमूलक तर्क का अनुसरण करता है।
\end{proof}

\begin{method}[छवियों की तुलना करना]\label{met:g10:reffunc:compare}
तुलना करने के लिए $ f(a)$ और $ f(b)$ एक संदर्भ के लिए \omterm{def:g10:functions:function}{समारोह} $ f $:
\begin{enumerate}
  \item जगह $ a $ और $ b $ एक पर \omterm{def:g10:numbers:interval}{अंतराल} जहां की विविधताएं $ f $ ज्ञात हैं;
  \item अगर $ f $ वहाँ बढ़ जाता है, इमेजिस के समान क्रम में हैं
        इनपुट; अगर $ f $ घटता है, क्रम उलट जाता है;
  \item अगर $ a $ और $ b $ एक जैसे नहीं हैं \omterm{def:g10:numbers:interval}{अंतराल} एकरसता का, सौदा
        पहले संकेतों के साथ (उदा.\ वर्ग के लिए: एक नकारात्मक और एक सकारात्मक
        इनपुट).
\end{enumerate}
\end{method}

\begin{example}
तुलना करना $\dfrac{1}{2.3}$ और $\dfrac{1}{2.4}$: दोनों इनपुट अंदर हैं
$\intoo{0}{+\infty}$, जहां उलटा है \omterm{def:g10:functions:function}{समारोह} घट जाती है, और
$2.3 < 2.4$, इसलिए $\dfrac{1}{2.3} > \dfrac{1}{2.4}$. तुलना करना $(-1.2)^2$
और $(-1.5)^2$: पर $\intoc{-\infty}{0}$ वर्ग घटता है, और
$-1.5 < -1.2$, इसलिए $(-1.5)^2 > (-1.2)^2$ --- वास्तव में $2.25 > 1.44$.
\end{example}

\section{अभ्यास}

\begin{exercise}[$\star $]\label{exo:g10:reffunc:1}
प्रत्येक के लिए \omterm{def:g10:reffunc:affine}{एफ़िन फ़ंक्शन}, उसे दे दो \omterm{def:g10:reffunc:affine}{ढलान}, इसका $ y $-रोकें, और कहें

\begin{exercise}[$\star\star $]\label{exo:g10:reffunc:7}
यह जानते हुए कि वर्ग \omterm{def:g10:functions:function}{समारोह} पर घट जाती है $\intoc{-\infty}{0}$ और
पर बढ़ता है $\intco{0}{+\infty}$, चौखटा $ x^2$ कब:
\[
\text{(a) } x \in \intcc{2}{5};
\qquad
\text{(b) } x \in \intcc{-3}{-1};
\qquad
\text{(c) } x \in \intcc{-2}{3}.
\]
(मामले (सी) में, सावधान रहें: \omterm{def:g10:functions:extrema}{न्यूनतम} का $ x^2$ अंतिम बिंदु पर नहीं है।)
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:reffunc:8}
एक टैक्सी कंपनी एक निश्चित शुल्क लेती है $4$ प्लस $1.5$ प्रति किलोमीटर;
दूसरा कोई शुल्क नहीं लेता और $2$ प्रति किलोमीटर. प्रत्येक मूल्य को एक द्वारा मॉडल करें
\omterm{def:g10:reffunc:affine}{एफ़िन फ़ंक्शन} दूरी का $ x $, दोनों पंक्तियों को आलेखित करें और किसमें से खोजें
दूरी पहली कंपनी सस्ती है.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:reffunc:9}
वह सभी को दिखाओ $ a, b \geq 0$: $\sqrt{a + b} \leq \sqrt a + \sqrt b $.
(संकेत: दोनों पक्ष अऋणात्मक हैं, इसलिए उनके वर्गों की तुलना करें।) कब है
वहाँ समानता है?
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g10:reffunc:10}
होने देना $ f(x) = \dfrac{2x + 1}{x - 1}$, के लिए परिभाषित $ x \neq 1$.
\begin{enumerate}
  \item बताते हैं कि $ f(x) = 2 + \dfrac{3}{x - 1}$ सभी के लिए $ x \neq 1$.
  \item की विविधताएँ निकालें $ f $ पर $\intoo{1}{+\infty}$ उनसे
        उलटे का \omterm{def:g10:functions:function}{समारोह}.
\end{enumerate}
\end{exercise}

\section{समस्या: प्रकृति के नियम संदर्भ कार्यों में बोलते हैं}

\begin{problem}[{सप्ताहांत समस्या --- जैसे-जैसे ब्रेकिंग दूरी बढ़ती है $ v^2$, लीवर आज्ञापालन करते हैं $1/d $, ग्रह अनुसरण करते हैं $ a^{3/2}$: पढ़ना संदर्भ के साथ भौतिकी कार्य}]
\label{pb:g10:reffunc:1}
एक भौतिकी की किताब खोलो और वही कुछ रेखांकन प्रत्येक पृष्ठ पर दिखाई दें:
परवलय, अतिपरवलय, मूल वक्र। प्रकृति इसे लिखती है
इस अध्याय के संदर्भ के साथ कानून कार्य --- और जानना
उनके आकार (\cref{prop:g10:reffunc:square},
\cref{prop:g10:reffunc:inverse}, \cref{prop:g10:reffunc:sqrt},
\cref{prop:g10:reffunc:cube}) कार को ब्रेक लगाने, संतुलन बनाने के लिए पर्याप्त है
लीवर, और ग्रहों का समय। ग्रैंड फिनाले केप्लर का तीसरा है
कानून, सौर मंडल के डेटा को सीधे पढ़ें।

\medskip
\noindent\textbf{Part I --- The square law of braking.}
सूखी सड़क पर कार की ब्रेकिंग दूरी के लिए सामान्य नियम:
$ d = \left(\frac{v}{10}\right)^{2}$ मीटर, गति से $ v $ किमी/घंटा.
\begin{enumerate}
  \item ब्रेकिंग दूरी की गणना करें $30$, $50$, $90$ और
        $130$ किमी/घंटा.
  \item गति दोगुनी करने से ब्रेकिंग दूरी कई गुना बढ़ जाती है
        कितना? वर्ग से स्पष्ट करें समारोहस्केलिंग है,
        और जाँच करें $50 \to 100$ किमी/घंटा.
  \item फोरेंसिक: स्किड मार्क्स माप $50$ एम। क्या गति होती है
        फॉर्मूला ड्राइवर को (किमी/घंटा तक) दोषी ठहराता है? जो
        संदर्भ \omterm{def:g10:functions:function}{समारोह} उत्तर दिया---और किस पर \omterm{def:g10:functions:function}{कार्यक्षेत्र} है
        पढ़ना असंदिग्ध
        (\cref{prop:g10:reffunc:sqrt})?
  \item इसके कारण उत्पन्न अतिरिक्त दूरी की तुलना करें \emph{एक} किमी/घंटा
        अधिक, कम और उच्च गति पर: गणना करें
        $ d(31) - d(30)$ और $ d(91) - d(90)$. किसकी संपत्ति
        परवलय (\cref{prop:g10:reffunc:square}) दो करो
        उत्तर स्पष्ट करें?
  \item वास्तविक रुकने से प्रतिक्रिया समय जुड़ जाता है --- लगभग एक सेकंड,
        जिस दौरान कार कवर हो जाती है $0.28v $ मीटर:
        $ D(v) = 0.28v + \left(\frac{v}{10}\right)^2$. गणना करना
        $ D(50)$ और $ D(130)$. दोनों में से कौन सा शब्द --- एफ़िन है
        या वर्ग --- शहर की गति पर हावी है, और जिस पर
        राजमार्ग?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The hyperbolas of daily life.}
\begin{enumerate}[resume]
  \item बल लगाने पर लीवर संतुलित होता है $\times $ दूरी है
        दोनों तरफ समान. ए $60$ किलो का बच्चा बैठता है $2$ मी से
        धुरी; दूरी पर संतुलन बल $ d $ दूसरे पर
        पक्ष है $ F(d) = \frac{120}{d}$ (किलो-समकक्ष में)।
        गणना करना $ F(0.5)$, $ F(1)$, $ F(3)$.
  \item व्युत्क्रम के भेद क्या कहते हैं? \omterm{def:g10:functions:function}{समारोह}
        (\cref{prop:g10:reffunc:inverse}) लीवर के बारे में कहें ---
        और आर्किमिडीज़ घमंड क्यों करता है (``मुझे जगह दो
        खड़े रहो और मैं पृथ्वी को हिला दूंगा'') में छिप जाओ
        अतिपरवलय की पूँछ? क्या मना करता है $ d = 0$?
  \item ध्वनि और प्रकाश इसके साथ फीके पड़ जाते हैं \emph{वर्ग} का
        दूरी: पर $1$ मी एक दीपक से तीव्रता है $100$
        इकाइयाँ; इसे दे दो $2$, $3$ और $10$ एम। स्पष्ट करें
        प्रतिपादक $2$ एक गोले के साथ: प्रकाश किस क्षेत्र पर है
        की दूरी पर फैल गया $ d $ (आर्किमिडीज़ की समाधि की समस्या, मिडिल स्कूल खंड में)?
  \item स्कूल यात्रा के लिए एक बस की लागत $600$ यूरो, विभाजन
        समान रूप से बीच में $ n $ प्रतिभागियों. प्रति लागत सारणीबद्ध करें
        के लिए सिर $ n = 10, 20, 30$, पता लगाएं कि कितने प्रतिभागी हैं
        इसे ले आओ $25$ यूरो या उससे कम, और बताएं क्या
        हाइपरबोला का आकार वादा करता है --- और इनकार करता है --- जैसे $ n $
        बढ़ता है.
  \item उत्पादन $ x $ पोस्टर की लागत $2x + 3$ यूरो ($3$ यूरो का
        सेटअप)। दिखाओ कि \emph{औसत} प्रति पोस्टर लागत है
        $ a(x) = 2 + \frac3x $, इसकी विविधताओं का वर्णन करें, और
        इसके क्षैतिज अनंतस्पर्शी की आर्थिक रूप से व्याख्या करें।
        (के बीजगणित की तुलना करें \cref{exo:g10:reffunc:10}.)
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Kepler's harmony.}
प्रत्येक ग्रह के लिए, चलो $ a $ सूर्य से इसकी औसत दूरी (में) हो
खगोलीय इकाइयाँ, पृथ्वी $= 1$) और $ T $ इसकी अवधि (वर्षों में)
डेटा: मंगल $ a = 1.52$, $ T = 1.88$; बृहस्पति $ a = 5.20$,
$ T = 11.86$; शनि ग्रह $ a = 9.54$, $ T = 29.4$.
\begin{enumerate}[resume]
  \item गणना करना $ a^3$ और $ T^2$ पृथ्वी, मंगल और बृहस्पति के लिए.
        1618 में केप्लर ने क्या देखा?
  \item कानून को इस प्रकार बताएं \omterm{def:g10:functions:function}{समारोह}: $ T = \sqrt{a^3} = a\sqrt a $.
        शनि ग्रह पर इसका परीक्षण करें।
  \item दो भविष्यवाणियाँ: एक क्षुद्रग्रह की परिक्रमा $ a = 4$ एयू ---
        इसकी अवधि ज्ञात करें (संख्याएँ पूर्ण आती हैं); एक धूमकेतु
        हर लौटाता है $27$ वर्ष---इसकी माध्य दूरी ज्ञात कीजिए
        (एक पूर्ण घन की तलाश करें)।
  \item कौन से तीन संदर्भ कार्य क्या कानून लंगड़ाता है
        एक साथ? और इसका स्केलिंग नियम: कब $ a $ गुणा किया जाता है
        द्वारा $4$, क्या होता है $ T $? (क्षुद्रग्रह से जांचें
        पृथ्वी के विरुद्ध।)
  \item केप्लर ने इस नियम को टाइको ब्राहे के सत्तर वर्षों के डेटा में पढ़ा
        न्यूटन के गुरुत्वाकर्षण द्वारा इसकी व्याख्या करने से पहले। एक में
        वाक्य: यह प्रकरण किसकी शक्ति के बारे में क्या कहता है?
        एक संदर्भ को पहचानना \omterm{def:g10:functions:function}{समारोह} संख्याओं की तालिका में?
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The tortoise and the hare.}
\begin{enumerate}[resume]
  \item आदेश $\sqrt x $, $ x $ और $ x^2$ पर $\intoo{0}{1}$ और पर
        $\intoo{1}{+\infty}$
        (\cref{prop:g10:reffunc:compare}), और दोनों को सत्यापित करें
        पर आदेश $ x = 0.25$ और $ x = 4$.
  \item हल करना $\sqrt x > x $ पूरी तरह से (के लिए) $ x \geq 0$, चौकोर
        ध्वनिपूर्वक और a के साथ समाप्त करें \omterm{def:g10:algebra:signtable}{साइन टेबल}).
  \item क्यूब लाओ: ऑर्डर करें $ x $, $ x^2$, $ x^3$ पर
        $ x = 0.5$ और कम से $ x = 2$, और उन सभी बिंदुओं को ढूंढें जहां
        वर्ग और घन कार्य पार करना।
  \item दो बचत योजनाओं के बाद भुगतान करें $ t $ साल, $100\sqrt t $
        यूरो (स्कीम ए) या $10 t^2$ यूरो (योजना बी)। वो दिखाओ
        जब B, A से आगे निकल जाता है $ t^3 = 100$, और क्रॉसओवर दें
        महीने का समय. शक्तियों के विरुद्ध जड़ों के बारे में नैतिक
        दीर्घकाल तक?
  \item समापन: इस समस्या के प्रत्येक नियम के लिए --- ब्रेक लगाना
        ($ v^2$), लीवर ($1/d $), चिराग ($1/d^2$), केपलर
        ($ a^{3/2}$), स्कीम ए ($\sqrt t$) --- एक उपवाक्य में कहें
        इसके संदर्भ की कौन सी विशेषता \omterm{def:g10:functions:graph}{ग्राफ} ले जाता है
        भौतिक अर्थ (तीव्र वृद्धि, ऊर्ध्वाधर अनंतस्पर्शी,
        फैला हुआ क्षेत्र, स्केलिंग घातांक, चपटा विकास)।
        निष्कर्ष: संदर्भ कार्य वर्णमाला हैं;
        प्रकृति उनके साथ लिखती है.
\end{enumerate}
\end{problem}
