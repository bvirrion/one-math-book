\chapter{समन्वय ज्यामिति}\label{ch:g10:coordgeom}

डेसकार्टेस का महान विचार अंकों को संख्याओं के जोड़े द्वारा, मोड़कर वर्णन करना था
संगणना में ज्यामिति की समस्याएं। केवल दो सूत्रों के साथ --- द
\omterm{prop:g10:coordgeom:midpoint}{मध्य} और दूरी --- कोई यह सिद्ध कर सकता है कि एक त्रिभुज समद्विबाहु है,
कि एक चतुर्भुज एक समांतर चतुर्भुज है, या कि तीन बिंदु संरेखित हैं,
एक भी सहायक रेखा खींचे बिना।

\section{विमान में निर्देशांक}

\begin{definition}[निर्देशांक तरीका]\label{def:g10:coordgeom:system}
ए \emph{निर्देशांक तरीका}\index{निर्देशांक तरीका} विमान के होते हैं
एक मूल का $O $ और दो स्नातक अक्षों के माध्यम से $ O $: क्षैतिज
$ x $-अक्ष और ऊर्ध्वाधर $ y $-अक्ष. हर बिंदु $ M $ फिर एक अनोखी जोड़ी है
का \emph{COORDINATES}\index{COORDINATES} $(x, y)$: इसका
\emph{सूच्याकार आकृति का भुज}\index{सूच्याकार आकृति का भुज} $ x $ और इसके
\emph{तालमेल}\index{तालमेल} $ y $. सिस्टम है
\emph{ऑर्थोनॉर्मल}\index{ऑर्थोनॉर्मल} जब कुल्हाड़ियाँ होती हैं
लंबवत और लंबाई की समान इकाई रखते हैं। इसमें सभी सिस्टम
अध्याय लम्बवत हैं।
\end{definition}

\section{किसी खंड का मध्यबिंदु}

\begin{proposition}[मध्यबिंदु सूत्र]\label{prop:g10:coordgeom:midpoint}
होने देना $ A(x_A, y_A)$ और $ B(x_B, y_B)$. \emph{मध्य}\index{मध्य}
$ I $ खंड का $[AB]$ है \omterm{def:g10:coordgeom:system}{COORDINATES}
\[
I\left(\frac{x_A + x_B}{2},\ \frac{y_A + y_B}{2}\right).
\]
\end{proposition}

\begin{proof}
क्षैतिज पर विचार करें \omterm{def:g10:coordgeom:system}{COORDINATES}. से जा रहा हूँ $ A $ को $ B $, द \omterm{def:g10:coordgeom:system}{सूच्याकार आकृति का भुज}
द्वारा परिवर्तन $ x_B - x_A $; बात आधी रह गई है \omterm{def:g10:coordgeom:system}{सूच्याकार आकृति का भुज}
\[
x_A + \frac{x_B - x_A}{2} = \frac{2x_A + x_B - x_A}{2}
= \frac{x_A + x_B}{2},
\]
दोनों का औसत भुज. यही गणना लागू होती है
\omterm{def:g10:coordgeom:system}{तालमेल}.
\end{proof}

\begin{example}\label{ex:g10:coordgeom:midpoint}
साथ $ A(-1, 4)$ और $ B(5, -2)$, द \omterm{prop:g10:coordgeom:midpoint}{मध्य} का $[AB]$ है
\[
I\left(\frac{-1 + 5}{2},\ \frac{4 + (-2)}{2}\right) = I(2, 1).
\]
सूत्र भी उल्टा चलता है: यदि $ A(-1, 4)$ और यह \omterm{prop:g10:coordgeom:midpoint}{मध्य} है
$ I(2, 1)$, तब $ B $ संतुष्ट $\frac{-1 + x_B}{2} = 2$ और
$\frac{4 + y_B}{2} = 1$, इसलिए $ x_B = 5$ और $ y_B = -2$: मुद्दा $ B $ है
\emph{सममित} का $ A $ के बारे में $ I $.
\end{example}

\section{दो बिंदुओं के बीच की दूरी}

\begin{theorem}[दूरी सूत्र]\label{thm:g10:coordgeom:distance}
एक में \omterm{def:g10:coordgeom:system}{ऑर्थोनॉर्मल} प्रणाली, के बीच की दूरी $ A(x_A, y_A)$ और
$ B(x_B, y_B)$ है
\[
AB = \sqrt{(x_B - x_A)^2 + (y_B - y_A)^2}.
\]
\end{theorem}

\begin{proof}
होने देना $ C $ मुद्दा यह है $(x_B, y_A)$: इसकी ऊंचाई भी उतनी ही है $ A $ और एक सा \omterm{def:g10:coordgeom:system}{सूच्याकार आकृति का भुज} जैसा $ B $, तो त्रिकोण $ ACB $ पर समकोण है $ C $. यह
पैर लंबाई के क्षैतिज और ऊर्ध्वाधर खंड हैं
$ AC = \abs{x_B - x_A}$ और $ CB = \abs{y_B - y_A}$. पाइथागोरस द्वारा
प्रमेय,
\[
AB^2 = AC^2 + CB^2 = (x_B - x_A)^2 + (y_B - y_A)^2,
\]
और वर्गमूल लेने पर (दोनों पक्ष अऋणात्मक हैं) सूत्र प्राप्त होता है।
\end{proof}

\begin{example}\label{ex:g10:coordgeom:distance}
साथ $ A(1, 1)$ और $ B(5, 3)$:
\[
AB = \sqrt{(5-1)^2 + (3-1)^2} = \sqrt{16 + 4} = \sqrt{20} = 2\sqrt 5 .
\]
दूरियाँ आमतौर पर सटीक (वर्गमूल) रूप में रखी जाती हैं; पर केवल दौर
यदि दशमलव उत्तर की आवश्यकता हो तो बहुत अंत।
\end{example}

\begin{remark}
अंकों का क्रम मायने नहीं रखता: $(x_A - x_B)^2 =
(x_B - x_A)^2$. लेकिन चौकों को मत भूलना! दूरी है
\emph{नहीं} $\abs{x_B - x_A} + \abs{y_B - y_A}$.
\end{remark}

\section{ज्यामिति में सूत्रों का उपयोग करना}

\begin{method}[त्रिभुज की प्रकृति]\label{met:g10:coordgeom:triangle}
तीन अंक दिये गये $ A $, $ B $, $ C $ द्वारा उनके \omterm{def:g10:coordgeom:system}{COORDINATES}:
\begin{enumerate}
  \item तीन वर्ग लंबाई की गणना करें $ AB^2$, $ AC^2$, $ BC^2$ के साथ
        दूरी सूत्र (वर्ग रखने से वर्गमूलों से बचा जाता है);
  \item दो समान वर्ग लंबाई $\Rightarrow $ त्रिकोण है
        \emph{समद्विबाहु};
  \item यदि सबसे बड़ी वर्ग लंबाई दो अन्य का योग है, तो
        त्रिकोण है \emph{समकोण का} सबसे लंबे विपरीत शीर्ष पर
        पक्ष, पाइथागोरस प्रमेय के व्युत्क्रम द्वारा।
\end{enumerate}
\end{method}

\begin{example}\label{ex:g10:coordgeom:triangle}
होने देना $ A(0, 1)$, $ B(4, 3)$ और $ C(2, -3)$. तब
\begin{align*}
AB^2 &= (4-0)^2 + (3-1)^2 = 16 + 4 = 20, \\
AC^2 &= (2-0)^2 + (-3-1)^2 = 4 + 16 = 20, \\
BC^2 &= (2-4)^2 + (-3-3)^2 = 4 + 36 = 40 .
\end{align*}
पहला, $ AB^2 = AC^2$, इसलिए $ AB = AC $: त्रिभुज समद्विबाहु है $ A $.
इसके अतिरिक्त $ AB^2 + AC^2 = 40 = BC^2$, तो पायथागॉरियन के विपरीत से
प्रमेय यह है कि त्रिभुज भी समकोण है $ A $.
\end{example}

\begin{method}[समांतर चतुर्भुज को पहचानना]\label{met:g10:coordgeom:parallelogram}
एक चतुर्भुज $ ABCD $ एक समांतर चतुर्भुज ठीक उसी समय होता है जब इसके विकर्ण होते हैं
$[AC]$ और $[BD]$ एक ही लें \omterm{prop:g10:coordgeom:midpoint}{मध्य}. तो: दोनों की गणना करें \omterm{prop:g10:coordgeom:midpoint}{मध्य} बिन्दुओं साथ \omterm{prop:g10:coordgeom:midpoint}{मध्य} सूत्र और उनकी तुलना करें।
\end{method}

\begin{example}\label{ex:g10:coordgeom:parallelogram}
होने देना $ A(-1, 0)$, $ B(2, 2)$, $ C(5, 1)$ और $ D(2, -1)$. \omterm{prop:g10:coordgeom:midpoint}{मध्य} का
$[AC]$ है $\left(\frac{-1+5}{2}, \frac{0+1}{2}\right) = (2, \frac12)$;
\omterm{prop:g10:coordgeom:midpoint}{मध्य} का $[BD]$ है $\left(\frac{2+2}{2}, \frac{2-1}{2}\right) =
(2, \frac12)$. वे मेल खाते हैं, इसलिए $ ABCD $ एक समांतर चतुर्भुज है.
\end{example}

\section{अभ्यास}

\begin{exercise}[$\star $]\label{exo:g10:coordgeom:1}
होने देना $ A(2, 5)$ और $ B(-4, 1)$. की गणना करें \omterm{def:g10:coordgeom:system}{COORDINATES} की \omterm{prop:g10:coordgeom:midpoint}{मध्य} का
$[AB]$ और दूरी $ AB $.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g10:coordgeom:2}
होने देना $ A(3, -2)$ और $ I(1, 2)$. खोजें \omterm{def:g10:coordgeom:system}{COORDINATES} मुद्दे की $ B $ ऐसा है कि $ I $ है \omterm{prop:g10:coordgeom:midpoint}{मध्य} का $[AB]$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g10:coordgeom:3}
कौन सा बिंदु $ P(4, 1)$, $ Q(-3, 2)$ और $ R(0, -5)$ के सबसे करीब है
उत्पत्ति $ O(0,0)$? सटीक दूरियों के साथ उत्तर दें।
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g10:coordgeom:4}
बिंदुओं को प्लॉट करें $ A(1, 2)$, $ B(5, 4)$, $ C(7, 0)$ पर \omterm{def:g10:functions:graph}{ग्राफ} कागज, फिर दिखाओ
गणना करके कि त्रिभुज $ ABC $ समद्विबाहु है. क्या यह समकोण है?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:coordgeom:5}
होने देना $ A(-2, 1)$, $ B(1, 3)$, $ C(4, 1)$ और $ D(1, -1)$.
\begin{enumerate}
  \item बताते हैं कि $ ABCD $ एक समांतर चतुर्भुज है.
  \item गणना करना $ AB $ और $ BC $. है $ ABCD $ एक समचतुर्भुज (सभी भुजाएँ बराबर)?
  \item गणना करना $ AC $ और $ BD $. है $ ABCD $ एक आयत (समान विकर्ण)?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:coordgeom:6}
केंद्र वाला वृत्त $\Omega(2, 1)$ और त्रिज्या $5$ सभी से मिलकर बनता है
दूरी पर बिंदु $5$ से $\Omega $. कौन सा बिंदु $ A(5, 5)$,
$ B(-1, -3)$, $ C(6, 2)$ इस घेरे पर झूठ बोलो? इसके अंदर? इसके बाहर?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:coordgeom:7}
होने देना $ A(1, 1)$ और $ B(7, 5)$. सभी बिंदु खोजें $ M(x, 0)$ की $ x $-अक्ष
जो कि समान दूरी पर हैं $ A $ और $ B $. (लिखना $ MA^2 = MB^2$ और के लिए हल करें
$ x $.)
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:coordgeom:8}
होने देना $ A(0, 3)$, $ B(4, 1)$.
\begin{enumerate}
  \item की गणना करें \omterm{def:g10:coordgeom:system}{COORDINATES} की \omterm{prop:g10:coordgeom:midpoint}{मध्य} $ I $ का $[AB]$.
  \item बताते हैं कि $ M(2, 2) = I $, फिर कंप्यूटिंग द्वारा सत्यापित करें $ MA $, $ MB $ वह
        $ M $ से समान दूरी पर है $ A $ और $ B $.
  \item पर दूसरा बिंदु खोजें $ y $-अक्ष, से समान दूरी पर $ A $ और
        $ B $.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:coordgeom:9}
बिन्दु $ A(-1, -1)$, $ B(3, 1)$ और $ C(11, 5)$ दिया जाता है। गणना करना $ AB $,
$ BC $ और $ AC $, और उसका निष्कर्ष निकालो $ A $, $ B $, $ C $ संरेखित हैं. (तीन अंक
बिल्कुल तब संरेखित होते हैं जब तीन दूरियों में से सबसे बड़ी दूरियों का योग होता है
दो अन्य.)
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g10:coordgeom:10}
होने देना $ A(a, 0)$ और $ B(0, b)$ साथ $ a, b > 0$, और चलो $ I $ हो \omterm{prop:g10:coordgeom:midpoint}{मध्य}
का $[AB]$. गणना करके दिखाओ कि $ OI = IA = IB $, कहाँ $ O $ है
उत्पत्ति. (यह एक शास्त्रीय प्रमेय सिद्ध करता है: एक समकोण त्रिभुज में, द
\omterm{prop:g10:coordgeom:midpoint}{मध्य} कर्ण तीनों शीर्षों से समान दूरी पर है।)
\end{exercise}

\section{समस्या: डेसकार्टेस ब्रिज --- वृत्त समीकरण और कैसे
यह जानने के लिए कि आप कहाँ हैं}

\begin{problem}[{सप्ताहान्त समस्या---वृत्त बन जाता है समीकरण, पुराने प्रमेय संगणना बन जाते हैं, और तीन बीकन एक बिंदु का पता लगाएं: बीजगणित द्वारा त्रिपुंजीकरण}]
\label{pb:g10:coordgeom:1}
1637 में रेन\'ई डेसकार्टेस ने दो महाद्वीपों के बीच एक पुल का निर्माण किया:
ज्यामिति का प्रत्येक वक्र एक बन गया \emph{\omterm{def:g10:algebra:equation}{समीकरण}}, हर
ज्यामितीय प्रश्न एक गणना. यह समस्या पुल पर चलती है
दोनों दिशाओं में --- वृत्त प्राप्त करना \omterm{def:g10:algebra:equation}{समीकरण} से
दूरी सूत्र (\cref{thm:g10:coordgeom:distance}), पुनः सिद्ध करना
बीजगणित की तीन पंक्तियों में शास्त्रीय प्रमेय, और कहां समाप्त होते हैं
पुल पर आज सबसे अधिक ट्रैफ़िक है: स्थिति की गणना करना
दूरी के संकेतों से, जीपीएस का सपाट-पृथ्वी हृदय।

\medskip
\noindent\textbf{Part I --- The circle gets an \omterm{def:g10:algebra:equation}{समीकरण}.}
\begin{enumerate}
  \item एक बिंदु $ M(x, y)$ केन्द्र के वृत्त पर स्थित है
        $\Omega(2, 1)$ और त्रिज्या $5$ बिल्कुल कब
        $\Omega M = 5$. प्राप्त करने के लिए इस शर्त का वर्ग करें
        सर्कल का \omterm{def:g10:algebra:equation}{समीकरण}:
        \[
        (x - 2)^2 + (y - 1)^2 = 25 .
        \]
  \item टेस्ट सदस्यता: कौन सा $ A(5, 5)$, $ B(6, 4)$,
        $ D(4, 4)$ वृत्त पर हैं, उसके अंदर हैं, उसके बाहर हैं?
  \item भेष में एक वृत्त: वर्गों को पूरा करें
        (\cref{pb:g10:algebra:1}) में
        \[
        x^2 + y^2 - 6x + 4y - 12 = 0
        \]
        और इसका केंद्र और त्रिज्या बताएं।
  \item इसके साथ भी ऐसा ही करें $ x^2 + y^2 - 6x + 4y + 14 = 0$. क्या
        क्या पूरा किया गया फॉर्म प्रकट करता है? मानदंड बताएं: कब
        करता है $(x - h)^2 + (y - k)^2 = c $ एक वृत्त का वर्णन करें, ए
        एकल बिंदु, या कुछ भी नहीं?
  \item प्रश्न 3 के वृत्त को क्षैतिज से प्रतिच्छेद करें
        लाइन $ y = 3$: स्थानापन्न करें और हल करें। कितने अंक, और
        ऐसी रेखा का ज्यामितीय नाम क्या है?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Old theorems in three lines.}
\begin{enumerate}[resume]
  \item मिडिल स्कूल वॉल्यूम के सर्कल प्रमेय का सर्कल प्रमेय (प्रत्येक आधे सर्कल में एक समकोण),
        बीजगणित द्वारा पुनः सिद्ध: चलो $ A(-r, 0)$ और $ B(r, 0)$ हो
        एक व्यास के सिरे और $ M(x, y)$ का कोई भी बिंदु
        वृत्त $ x^2 + y^2 = r^2$. गणना करना $ MA^2 + MB^2$ और
        पाइथागोरस की बातचीत के साथ समाप्त करें
        $\widehat{AMB}$ सही है।
  \item माध्यिका प्रमेय: साथ $ A(-a, 0)$, $ B(a, 0)$ (इतना
        \omterm{prop:g10:coordgeom:midpoint}{मध्य} का $[AB]$ मूल है $ I $), उसके लिए दिखाओ
        हर बिंदु $ M(x, y)$:
        \[
        MA^2 + MB^2 = 2\,MI^2 + \frac{AB^2}{2} .
        \]
  \item बिंदुओं का स्थान ज्ञात कीजिए $ M $ साथ
        $ MA^2 + MB^2 = 26$ कब $ AB = 6$: कौन सा वक्र, कौन सा
        केंद्र, क्या त्रिज्या?
  \item एक अलग ठिकाना: $ A(0,0)$, $ B(3,0)$; सभी बिंदु खोजें
        के साथ $ MA = 2\,MB $. (चौकोर करें, विस्तृत करें, पूरा करें
        वर्ग: एक प्रसिद्ध वृत्त प्रकट होता है --- अपोलोनियस इसे जानता था
        बिना \omterm{def:g10:coordgeom:system}{COORDINATES}.)
  \item एक या दो वाक्यों में: डेसकार्टेस ब्रिज क्या है
        के बारे में परिवर्तन \emph{कैसे} प्रमेय सिद्ध हो जाते हैं? तुलना करें
        प्रश्न 6 के आयत प्रमाण के साथ
        मिडिल स्कूल वॉल्यूम का सर्कल प्रमेय (प्रत्येक आधे सर्कल में एक समकोण)।
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Three beacons find you.}
आपका मत $ M(x, y)$ अज्ञात है. प्रकाश $ P(0, 0)$ उपाय
आपकी दूरी के रूप में $5$; प्रकाश $ Q(6, 0)$ उपाय भी करता है $5$.
\begin{enumerate}[resume]
  \item दो वृत्त लिखें \omterm{def:g10:algebra:equation}{समीकरण}, उन्हें घटाएं, और देखें
        वर्ग रद्द करें: क्या सरल है \omterm{def:g10:algebra:equation}{समीकरण} बचता है?
        खोजने के लिए एक वृत्त के साथ संयोजित करें \emph{दो} उम्मीदवार
        पद.
  \item एक तीसरा बीकन $ R(0, 8)$ के रूप में आपकी दूरी मापता है $5$.
        प्रत्येक उम्मीदवार से इसकी दूरी की गणना करें और तय करें कि कहां
        आप हैं.
  \item बताएं कि दो वृत्तों को क्यों घटाया जा रहा है \omterm{def:g10:algebra:equation}{समीकरण}
        \emph{हमेशा} की पैदावार होती है \omterm{def:g10:algebra:equation}{समीकरण} एक पंक्ति का (कौन से पद
        रद्द करें?), और वह रेखा ज्यामितीय रूप से क्या है जब
        वृत्त दो बिंदुओं पर प्रतिच्छेद करते हैं।
  \item वास्तविक उपग्रह पोजीशनिंग अंतरिक्ष में और उसके साथ काम करती है
        अपूर्ण घड़ियाँ: प्रत्येक उपग्रह (सिग्नल यात्रा के माध्यम से) देता है
        समय) एक दूरी, अत: एक गोला। कितने अज्ञात
        क्या रिसीवर के पास (स्थिति) है? \emph{और} इसकी अपनी घड़ी है
        त्रुटि), और जीपीएस कम से कम क्यों सुनता है
        \emph{चार} उपग्रह?
  \item परिशुद्धता: मान लीजिए बीकन $ P $की दूरी वास्तव में है
        $5.1$ के बजाय $5$ (बाकी सब अपरिवर्तित). पुनः करें
        नया खोजने के लिए प्रश्न 11 का घटाव $ x $, और
        अनुमान लगाएं कि अनुमान कितना आगे बढ़ गया। (त्रुटि की तुलना करें
        के बजट विचार \cref{pb:g10:numbers:1}.)
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The surveyor's toolkit.}
\begin{enumerate}[resume]
  \item सबसे छोटा जल मार्ग: एक शिविर $ A(1, 5)$, एक खलिहान
        पर $ B(7, 3)$, धुरी के अनुदिश एक सीधी नदी $ y = 0$.
        से जाना है $ A $ नदी तक और फिर $ B $ के साथ
        कम से कम चलना: प्रतिबिंबित करें $ B $ नदी के उस पार $ B'$,
        और खोजें कि खंड कहां है $[AB']$ नदी पार करता है.
        क्रॉसिंग पॉइंट और \omterm{def:g10:functions:extrema}{न्यूनतम} कुल लंबाई दें।
        (यह विचार मिडिल स्कूल वॉल्यूम का बिलियर्ड-बाउंस अभ्यास है, जो अब पूरी तरह से है
        गणना योग्य.)
  \item प्रतिद्वंद्वी पर न्यूनतमता की पुष्टि करें: कुल पथ की गणना करें
        के माध्यम से $ Q(3, 0)$ और जांचें कि यह आपके उत्तर से हार जाता है।
  \item चतुर्भुज का वर्गीकरण कीजिए $ A(1,1)$, $ B(4,2)$,
        $ C(5,5)$, $ D(2,4)$ पूरी तरह से: समांतर चतुर्भुज? रोम्बस?
        आयत? वर्ग?
        (\cref{met:g10:coordgeom:parallelogram},
        \cref{met:g10:coordgeom:triangle} --- विकर्णों की तुलना करें
        भी.)
  \item दिया गया $ A(-1, 2)$, $ B(3, 4)$, $ C(6, 0)$, खोजो $ D $ ऐसा है कि $ ABCD$ एक समांतर चतुर्भुज है ( विकर्ण-मध्यबिंदु
        मिडिल स्कूल वॉल्यूम की आधी-अधूरी सप्ताहांत समस्या की युक्ति, अब सूत्रों में)।
  \item समापन: तीन उपकरण --- \omterm{prop:g10:coordgeom:midpoint}{मध्य} सूत्र, दूरी
        सूत्र, वृत्त का घटाव \omterm{def:g10:algebra:equation}{समीकरण}. प्रत्येक के लिए, नाम
        इस समस्या में किस प्रकार का प्रश्न सुलझाया गया, और
        बताएं कि अगले दो अध्याय किट में क्या जोड़ते हैं
        (दिशाएं और \omterm{def:g10:reffunc:affine}{ढलान}ों: सदिश और रेखा \omterm{def:g10:algebra:equation}{समीकरण}).
\end{enumerate}
\end{problem}
