\chapter{संभाव्यता और नमूनाकरण}\label{ch:g10:proba}

\omterm{def:g10:proba:distribution}{संभावना} मौका देने के लिए नंबर निर्दिष्ट करता है। प्रयोगों से शुरू करते हुए
समान रूप से संभावित परिणाम --- पासा, सिक्के, कार्ड --- यह अध्याय बनाता है
संभावनाओं की गणना के लिए बुनियादी नियम घटनाएँ, परिचय देता है \omterm{met:g10:proba:tree}{पेड़}
आरेख दो-चरणीय प्रयोगों के लिए, और बीच के लिंक के साथ समाप्त होता है
संभाव्यताएँ और प्रेक्षित आवृत्तियाँ। सिद्धांत जारी है
\cref{ch:g11:prob}.

\section{यादृच्छिक प्रयोग और घटनाएँ}

\begin{definition}[परिणाम और घटनाएँ]\label{def:g10:proba:events}
ए \emph{यादृच्छिक प्रयोग}\index{यादृच्छिक प्रयोग} एक प्रयोग है जिसका
परिणाम की भविष्यवाणी नहीं की जा सकती (पासा पलटना)। सभी संभव का सेट
परिणाम है \emph{नमूना अंतरिक्ष}\index{नमूना अंतरिक्ष}
$\Omega = \{\omega_1, \dots, \omega_n\}$; इसके तत्व हैं
\emph{परणाम}. एक \emph{आयोजन}\index{आयोजन} परिणामों का एक सेट है; एक
घटना \emph{घटित होना} जब प्रयोग का परिणाम उसका होता है.
\end{definition}

\begin{definition}[प्रायिकता वितरण]\label{def:g10:proba:distribution}
प्रत्येक परिणाम को असाइन करना $\omega_i $ एक संख्या $ p_i \geq 0$, साथ
$ p_1 + \dots + p_n = 1$, परिभाषित करता है ए \emph{संभावना}\index{संभावना}:
संभावना $\P(A)$ की एक \omterm{def:g10:proba:events}{आयोजन} $ A $ का योग है $ p_i $ का
में परिणाम $ A $. विशेष रूप से $\P(\Omega) = 1$ और असंभव \omterm{def:g10:proba:events}{आयोजन}
$\varnothing $ संभावना है $0$.
\end{definition}

\begin{proposition}[समान रूप से संभावित परिणाम]\label{prop:g10:proba:uniform}
यदि $ n $ परिणाम समान रूप से संभावित हैं (प्रत्येक $ p_i = \frac1n $), फिर के लिए
प्रत्येक \omterm{def:g10:proba:events}{आयोजन} $ A $:
\[
\P(A) = \frac{\text{number of outcomes in } A}{n}
= \frac{\text{favorable cases}}{\text{possible cases}} .
\]
\end{proposition}

\begin{proof}
$ A $ कुछ संख्या शामिल है $ k $ परिणामों में से प्रत्येक \omterm{def:g10:proba:distribution}{संभावना} $\frac1n $,
इसलिए $\P(A) = \frac kn $.
\end{proof}

\begin{example}\label{ex:g10:proba:die}
एक निष्पक्ष पासा घुमाएँ: $\Omega = \{1, 2, 3, 4, 5, 6\}$, प्रत्येक परिणाम हो रहा है
\omterm{def:g10:proba:distribution}{संभावना} $\frac16$. \omterm{def:g10:proba:events}{आयोजन} $ A $: ``परिणाम सम है'' है
$ A = \{2, 4, 6\}$, इसलिए $\P(A) = \frac36 = \frac12$. \omterm{def:g10:proba:events}{आयोजन} $ B $: ``द
परिणाम कम से कम है $5$'' है $ B = \{5, 6\}$, इसलिए $\P(B) = \frac26 =
\frac13$.
\end{example}

\section{घटनाओं का संयोजन}

\begin{definition}[प्रतिच्छेदन, मिलन, पूरक]\label{def:g10:proba:operations}
होने देना $ A $ और $ B $ होना घटनाएँ.
\begin{itemize}
  \item $ A \cap B $ (``$ A $ और $ B $'') तब होता है जब दोनों घटित होते हैं;
  \item $ A \cup B $ (``$ A $ या $ B $'') तब होता है जब कम से कम एक घटित होता है;
  \item  \emph{पूरक}\index{पूरक} $\bar A $ (``नहीं $ A $'')
        बिल्कुल तब होता है जब $ A $ नहीं करता;
  \item $ A $ और $ B $ हैं \emph{असंगत}\index{असंगत} जब वे घटित नहीं हो सकते
        एक साथ: $ A \cap B = \varnothing $.
\end{itemize}
\end{definition}

\begin{theorem}[अतिरिक्त नियम]\label{thm:g10:proba:addition}
सभी के लिए घटनाएँ $ A $ और $ B $:
\[
\P(A \cup B) = \P(A) + \P(B) - \P(A \cap B),
\qquad
\P(\bar A) = 1 - \P(A) .
\]
के लिए \omterm{def:g10:proba:operations}{असंगत} घटनाएँ, पहला नियम सरल बनाता है
$\P(A \cup B) = \P(A) + \P(B)$.
\end{theorem}

\begin{proof}
जोड़ने पर $\P(A) + \P(B)$ जोड़ता है \omterm{def:g10:proba:distribution}{संभावना} के हर परिणाम का
$ A \cup B $ एक बार, \emph{के अलावा} उन लोगों के $ A \cap B $, जिन्हें गिना जाता है
दो बार --- एक बार अंदर $ A $ और एक बार अंदर $ B $. घटाने $\P(A \cap B)$ मरम्मत
दोहरी गिनती. के लिए \omterm{def:g10:proba:operations}{पूरक}: $ A $ और $\bar A $ हैं \omterm{def:g10:proba:operations}{असंगत}
और $ A \cup \bar A = \Omega $, इसलिए $\P(A) + \P(\bar A) = \P(\Omega) = 1$.
\end{proof}

  \node[omExo, font=\small] at (3.0,1.45) {$\Omega $};

{\small दो अतिव्यापी घटनाएँ: में $\P(A) + \P(B)$, छायांकित
\omterm{def:g10:numbers:interunion}{चौराहा} $ A \cap B $ दो बार गिना जाता है, जो घटाव की व्याख्या करता है
अतिरिक्त नियम में.}
\end{omfigure}

\begin{example}\label{ex:g10:proba:cards}
एक मानक से एक कार्ड बनाएं $52$-कार्ड डेक। होने देना $ A $: ``कार्ड एक है
दिल'' ($13$ कार्ड) और $ B $: ``कार्ड एक राजा है'' ($4$ कार्ड)। तब
$ A \cap B $ ``दिलों का राजा'' है ($1$ कार्ड), और
\[
\P(A \cup B) = \frac{13}{52} + \frac{4}{52} - \frac{1}{52} = \frac{16}{52}
= \frac{4}{13}.
\]
 \omterm{def:g10:proba:operations}{पूरक} नियम अक्सर सबसे तेज़ मार्ग होता है: \omterm{def:g10:proba:distribution}{संभावना} वह
कार्ड है \emph{नहीं} एक दिल है $1 - \frac{13}{52} = \frac34$.
\end{example}

\section{वृक्ष रेखाचित्र}

\begin{method}[दो-चरणीय प्रयोग]\label{met:g10:proba:tree}
दो चरणों में किए गए प्रयोग के लिए:
\begin{enumerate}
  \item चित्रित करो \emph{पेड़}\index{पेड़}: प्रति संभव एक शाखा
        पहले चरण का परिणाम, फिर इनमें से प्रत्येक से एक शाखा
        दूसरे चरण का परिणाम, लिखना \omterm{def:g10:proba:distribution}{संभावना} प्रत्येक शाखा पर
        (किसी भी नोड को छोड़ने वाली शाखाओं का योग होना चाहिए $1$);
  \item  \omterm{def:g10:proba:distribution}{संभावना} एक का \emph{पथ} (का एक पत्ता \omterm{met:g10:proba:tree}{पेड़}) है
        इसकी शाखाओं के साथ संभावनाओं का उत्पाद;
  \item  \omterm{def:g10:proba:distribution}{संभावना} की एक \omterm{def:g10:proba:events}{आयोजन} की संभावनाओं का योग है
        वे रास्ते जो इसे साकार करते हैं।
\end{enumerate}
\end{method}

\begin{example}\label{ex:g10:proba:tree}
एक कलश में शामिल है $3$ लाल गेंदें और $2$ नीली गेंदें. एक गेंद निकालो, डालो
वापस, फिर से ड्रा करें। प्रत्येक ड्रा के साथ लाल (R) मिलता है \omterm{def:g10:proba:distribution}{संभावना} $\frac35$ और
नीला (बी) के साथ \omterm{def:g10:proba:distribution}{संभावना} $\frac25$.

 \omterm{def:g10:proba:distribution}{संभावना} एक ही रंग की दो गेंदें प्राप्त करना है
$\P(\text{RR}) + \P(\text{BB}) = \frac{9}{25} + \frac{4}{25} =
\frac{13}{25}$; \omterm{def:g10:proba:distribution}{संभावना} कम से कम एक नीला है
$1 - \P(\text{RR}) = 1 - \frac{9}{25} = \frac{16}{25}$ (\omterm{def:g10:proba:operations}{पूरक} नियम)।
\end{example}

\begin{remark}
प्रतिस्थापन के बिना, दूसरे ड्रा की संभावनाएँ निर्भर करती हैं
पहला: ऊपर कलश के साथ, केवल एक लाल गेंद निकालने के बाद $2$ लाल और $2$
नीला रहता है, इसलिए दूसरी शाखा ले जाती है $\frac24$ और $\frac24$.
\omterm{met:g10:proba:tree}{पेड़} विधि अपरिवर्तित काम करती है; केवल दूसरे स्तर की शाखाओं की संख्याएँ
परिवर्तन.
\end{remark}

\section{आवृत्तियाँ और नमूनाकरण}

जब ए \omterm{def:g10:proba:events}{यादृच्छिक प्रयोग} कई बार दोहराया जाता है, देखा गया \omterm{def:g10:stats:series}{आवृत्ति}
की एक \omterm{def:g10:proba:events}{आयोजन} उसके पास पहुँचता है \omterm{def:g10:proba:distribution}{संभावना} --- इसलिए \omterm{def:g10:proba:distribution}{संभावना} है
वास्तविक दुनिया का वर्णन करने के लिए उपयोगी।

\begin{definition}[नमूना]\label{def:g10:proba:sample}
ए \emph{नमूना}\index{नमूना} आकार का $ n $ को दोहराने का परिणाम है
वही प्रयोग $ n $ कई बार स्वतंत्र रूप से. \emph{देखा \omterm{def:g10:stats:series}{आवृत्ति}} की एक \omterm{def:g10:proba:events}{आयोजन} नमूने में घटनाओं की संख्या को विभाजित किया गया है $ n $.
\end{definition}

\begin{example}\label{ex:g10:proba:sampling}
एक निष्पक्ष सिक्का चित देता है \omterm{def:g10:proba:distribution}{संभावना} $0.5$. इसे उछालना $100$ बार
शायद ही कभी सटीक देता है $50$ प्रमुख: नमूने \emph{उतार चढ़ाव}. के विशिष्ट रन
$100$ टॉस जैसी आवृत्तियाँ देते हैं $0.46$, $0.53$, $0.49$ --- के करीब
$0.5$, लेकिन बराबर नहीं. बड़ा नमूने उतार-चढ़ाव कम: साथ $10\,000$
टॉस, प्रेक्षित आवृत्तियाँ आमतौर पर लगभग के भीतर होती हैं $0.01$ का $0.5$.
\end{example}

\begin{remark}[उतार-चढ़ाव का अंतराल]
अंगूठे का एक उपयोगी नियम, उचित ठहराया गया \cref{ch:g11:binom}: एक के लिए \omterm{def:g10:proba:sample}{नमूना} का
आकार $ n $ की एक \omterm{def:g10:proba:events}{आयोजन} का \omterm{def:g10:proba:distribution}{संभावना} $ p $ (साथ $ n $ यथोचित रूप से बड़ा), द
अवलोकन किया \omterm{def:g10:stats:series}{आवृत्ति} इसमें स्थित है \omterm{def:g10:numbers:interval}{अंतराल}
\[
\intcc{p - \frac{1}{\sqrt n}}{\,p + \frac{1}{\sqrt n}}
\]
मोटे तौर पर $95\%$ का नमूने. यदि कोई ध्यान दे \omterm{def:g10:stats:series}{आवृत्ति} बहुत बाहर गिरता है
यह \omterm{def:g10:numbers:interval}{अंतराल}, के मूल्य पर संदेह करें $ p $: यह का प्रारंभिक बिंदु है
सांख्यिकीय परीक्षण.
\end{remark}

\begin{example}\label{ex:g10:proba:test}
एक फैक्ट्री का दावा तो यही है $10\%$ इसके कुछ हिस्से ख़राब हैं
($ p = 0.1$). में एक \omterm{def:g10:proba:sample}{नमूना} का $ n = 400$ भाग, $64$ दोषपूर्ण हैं: देखा गया
\omterm{def:g10:stats:series}{आवृत्ति} $\frac{64}{400} = 0.16$. उतार-चढ़ाव \omterm{def:g10:numbers:interval}{अंतराल} है
$\intcc{0.1 - \frac{1}{20}}{0.1 + \frac{1}{20}} = \intcc{0.05}{0.15}$,
और $0.16$ इसके बाहर गिरता है: द \omterm{def:g10:proba:sample}{नमूना} दावे पर गंभीर संदेह पैदा करता है.
\end{example}

\section{अभ्यास}

\begin{exercise}[$\star $]\label{exo:g10:proba:1}
एक निष्पक्ष पासा फेंका गया है। की गणना करें \omterm{def:g10:proba:distribution}{संभावना} प्रत्येक का \omterm{def:g10:proba:events}{आयोजन}: ``ए मिल रहा है
$6$''; ``एक विषम संख्या प्राप्त करना''; ``\omterm{def:g10:functions:extrema}{अधिकतम} प्राप्त करना $4$''; ``एक मिल रहा है
$7$''.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g10:proba:2}
एक बैग में शामिल है $5$ हरा, $3$ पीला और $2$ काले पत्थर; एक संगमरमर है
यादृच्छिक रूप से खींचा गया. की गणना करें \omterm{def:g10:proba:distribution}{संभावना} कि यह हरा है; कि यह नहीं है
काला; कि ये हरा है या पीला.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g10:proba:3}
की एक कक्षा में $30$ छात्र, $18$ स्पैनिश का अध्ययन करें, $10$ जर्मन का अध्ययन करें, और
$4$ दोनों का अध्ययन करें. एक छात्र को यादृच्छिक रूप से चुना जाता है। \omterm{def:g10:vectors:sum}{जोड़} नियम का उपयोग करते हुए,
की गणना करें \omterm{def:g10:proba:distribution}{संभावना} कि यह छात्र दोनों में से कम से कम एक का अध्ययन करता है
भाषाएँ, फिर \omterm{def:g10:proba:distribution}{संभावना} कि वे न तो अध्ययन करते हैं।
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g10:proba:4}
एक निष्पक्ष सिक्के को दो बार उछाला जाता है। खींचना \omterm{met:g10:proba:tree}{पेड़} प्रयोग और गणना का
द \omterm{def:g10:proba:distribution}{संभावना} दो सिर पाने का; बिल्कुल एक सिर; कम से कम एक
सिर.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:proba:5}
एक कलश में शामिल है $4$ लाल और $6$ सफ़ेद गेंदें. एक के बाद एक दो गेंदें निकाली जाती हैं
दूसरा \emph{बिना} प्रतिस्थापन। खींचना \omterm{met:g10:proba:tree}{पेड़} सही के साथ
दूसरे स्तर पर संभाव्यताएँ, और गणना करें \omterm{def:g10:proba:distribution}{संभावना} ड्राइंग का
दो लाल गेंदें, फिर अलग-अलग रंगों की दो गेंदें निकालना।
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:proba:6}
दो निष्पक्ष पासे फेंके जाते हैं और उनका योग दर्ज किया जाता है।
\begin{enumerate}
  \item स्पष्ट करें कि क्यों \omterm{def:g10:proba:events}{नमूना अंतरिक्ष} के रूप में लिया जा सकता है $36$ समान रूप से
        संभावित \omterm{def:g10:vectors:sum}{जोड़}े $(1,1), (1,2), \dots, (6,6)$.
  \item की गणना करें \omterm{def:g10:proba:distribution}{संभावना} वह योग है $7$; यह है कि $12$;
        कि यह \omterm{def:g10:functions:extrema}{अधिकतम} है $4$.
  \item कौन सा योग सबसे अधिक संभावित है?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:proba:7}
एक बहुविकल्पीय प्रश्न प्रदान करता है $4$ उत्तर, एक सही। एक विद्यार्थी
उत्तर $2$ स्वतंत्र ऐसे प्रश्न पूरी तरह से यादृच्छिक हैं। की गणना करें
\omterm{def:g10:proba:distribution}{संभावना} दोनों का सही उत्तर देना, बिल्कुल एक सही, कोई नहीं
सही ढंग से. जांचें कि तीनों संभावनाओं का योग कितना है $1$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:proba:8}
एक राजनेता का दावा है $50\%$ अनुमोदन। के एक सर्वेक्षण में $ n = 900$ बेतरतीब ढंग से
चुने गए मतदाता, $396$ मंज़ूरी देना।
\begin{enumerate}
  \item प्रेक्षित की गणना करें \omterm{def:g10:stats:series}{आवृत्ति} और उतार-चढ़ाव \omterm{def:g10:numbers:interval}{अंतराल}
        आस-पास $ p = 0.5$ के लिए $ n = 900$.
  \item क्या सर्वेक्षण में दावे पर संदेह जताया गया है?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g10:proba:9}
एक खेल की लागत $2$ खेलने के लिए: आप एक निष्पक्ष पासा फेंकते हैं और जीतते हैं $6$ यदि आप एक रोल करते हैं
$6$, $3$ यदि आप एक रोल करते हैं $5$, अन्यथा कुछ नहीं.
\begin{enumerate}
  \item की गणना करें \omterm{def:g10:proba:distribution}{संभावना} जीती गई प्रत्येक राशि का ($6$, $3$, $0$).
  \item ऊपर $600$ खेल, आप प्रत्येक से कितने परिणामों की अपेक्षा करते हैं?
        कुल अपेक्षित लाभ और हानि की गणना करें: क्या खेल सार्थक है
        खेल रहे हो?
\end{enumerate}
\end{exercise}

\section{समस्या: तीन दरवाजे जिन्होंने हजारों लोगों को मूर्ख बनाया
गणितज्ञ}

\begin{problem}[{सप्ताहांत समस्या--- मोंटी हॉल विरोधाभास और इसके चचेरे भाई: जानकारी कैसे संभावना बदलती है, और नमूने कैसे बदलते हैं इच्छाधारी सोच से अलग दावे}]
\label{pb:g10:proba:1}
1990 में स्तंभकार मर्लिन वोस सावंत ने इसके बारे में एक पहेली प्रकाशित की
एक गेम शो होस्ट, तीन दरवाजे, एक कार और दो बकरियां --- और
इसका सही उत्तर दिया. दस हजार पाठक, उनमें से सैकड़ों
गणित की पीएचडी, ने यह कहते हुए लिखा कि वह गलत थी। यह
समस्या आपका साथ निभाती है \omterm{met:g10:proba:tree}{पेड़} (\cref{met:g10:proba:tree}),
\omterm{def:g10:proba:operations}{पूरक} और अतिरिक्त नियम
(\cref{thm:g10:proba:addition}), आपको तर्क निपटाने देता है
स्वयं, और उस उपकरण के साथ समाप्त होता है जो इस तरह की प्रत्येक मध्यस्थता करता है
व्यवहार में विवाद: उतार-चढ़ाव \omterm{def:g10:proba:sample}{नमूना}.

\medskip
\noindent\textbf{Part I --- The rules, warmed up.}
\begin{enumerate}
  \item एक मानक से एक कार्ड $52$-कार्ड डेक: गणना करें
        $ P(\text{king or heart})$ \omterm{def:g10:vectors:sum}{जोड़} नियम के साथ ---
        और बताओ क्यों $\frac{4}{52} + \frac{13}{52}$ अकेला है
        गलत.
  \item तीन निष्पक्ष सिक्के उछालें: गणना करें $ P(\text{at least one
        head})$ से \omterm{def:g10:proba:operations}{पूरक}.
  \item एक बैग रखता है $3$ लाल और $2$ नीले पत्थर; दो खींचे गए हैं
        प्रतिस्थापन के बिना. के साथ \omterm{met:g10:proba:tree}{पेड़}, गणना करें
        $ P(\text{two reds})$ और $ P(\text{one of each color})$.
  \item प्रश्न 3 का वितरण पूरा करें
        $ P(\text{two blues})$ और तीन संभावनाओं की जाँच करें
        का योग $1$.
  \item कोई रिपोर्ट करता है, दो के लिए \emph{\omterm{def:g10:proba:operations}{असंगत}} घटनाएँ,
        $ P(A) = 0.6$ और $ P(B) = 0.5$. उन्हें एक के साथ दोषी ठहराओ
        अंकगणित की रेखा.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The three doors.}
खेल: एक कार तीन दरवाजों में से एक के पीछे छिपी है, बकरियाँ उसके पीछे
अन्य दो. आप एक दरवाज़ा चुनें (दरवाज़ा 1 कहें)। मेज़बान --- कौन
\emph{जानता है} कार कहाँ है --- अन्य दो में से एक को खोलता है
दरवाजे, हमेशा एक बकरी को प्रकट करते हैं, और आपको मौका प्रदान करते हैं
शेष बंद दरवाजे पर स्विच करें।
\begin{enumerate}[resume]
  \item किसी भी चीज़ की गणना करने से पहले: क्या स्विचिंग से मदद मिलती है, नुकसान होता है, या
        कोई फर्क नहीं पड़ता? अपना आंतरिक उत्तर ईमानदारी से लिखें।
  \item अब \omterm{met:g10:proba:tree}{पेड़}, कार की वास्तविक स्थिति पर मामलों द्वारा
        (प्रत्येक $\frac13$): यदि कार आपके दरवाजे के पीछे है,
        स्विचिंग खो देता है; यदि यह किसी अन्य दरवाजे के पीछे है, तो क्या
        मेज़बान को क्या करना चाहिए, और फिर स्विच करने से क्या लाभ होता है?
        गणना करना $ P(\text{win by switching})$.
  \item एक-पंक्ति संस्करण: स्विचिंग ठीक उसी समय जीतती है जब आपका
        पहला चयन ग़लत था. पुनः समापन करें.
  \item अब महत्वपूर्ण संस्करण --- ``मोंटी फॉल'': मेजबान
        करता है \emph{नहीं} जानें कि कार कहां है, इनमें से एक को खोलता है
        अन्य दरवाजे बेतरतीब ढंग से, और भाग्य से एक बकरी का पता चलता है। सूची
        छह समान रूप से संभावित परिदृश्य (कार की स्थिति)। $\times $
        दरवाज़ा खुला), जहां कार दिखाई दे उसे हटा दें,
        और गणना करें \omterm{def:g10:proba:distribution}{संभावना} वह स्विचिंग जीत जाती है
        \emph{बचे हुए परिदृश्यों के बीच}. क्या बदला, और
        मेज़बान का ज्ञान क्यों मायने रखता है?
  \item अंतर्ज्ञान पंप: एक सौ दरवाजे, आप एक चुनें
        जानने वाला मेजबान खुलता है $98$ बकरी के दरवाजे. बदलना? साथ क्या
        \omterm{def:g10:proba:distribution}{संभावना} जीतने का?
  \item आप एक संशयवादी व्यक्ति का सामना करते हैं, जो हजारों पत्र-लेखकों की तरह,
        ``50--50'' पर जोर देता है। कार्ड-आधारित सिमुलेशन का वर्णन करें
        (तीन कार्ड, कई राउंड, मेज़बान के रूप में एक मित्र) और कहें
        इसके बाद आप किस अनुपात में स्विचिंग की जीत की उम्मीद करते हैं $300$
        दौर --- उतार-चढ़ाव के साथ \omterm{def:g10:numbers:interval}{अंतराल} अध्याय का
        आपके मार्जिन के रूप में.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Cousins of the paradox.}
\begin{enumerate}[resume]
  \item एक परिवार में दो बच्चे हैं; तुम वह सीखो \emph{कम से कम
        एक लड़का है}. समान रूप से संभावित दो बच्चों की सूची बनाएं
        जानकारी के साथ संगत कॉन्फ़िगरेशन, और
        गणना करें $ P(\text{two boys})$.
  \item एक ही परिवार, लेकिन जानकारी अब है: \emph{ज्येष्ठ
        बच्चा लड़का है}. पुनर्गणना करें. दो उत्तर क्यों देते हैं?
        भिन्न हैं, और वे दोनों मोंटी के साथ क्या साझा करते हैं?
  \item तीन उचित सिक्के: गणना करें $ P(\text{all three alike})$.
        गणितज्ञ डी'अलेम्बर्ट ने 1754 में दूसरे के लिए तर्क दिया
        उत्तर: ``या तो वे सभी एक जैसे हैं या वे नहीं हैं ---
        ''दो में से एक मामला।'' सटीक त्रुटि का नाम बताएं
        \cref{prop:g10:proba:uniform} हाथ में.
  \item एक खेल की लागत $2$ यूरो: पासा घुमाओ, जीतो $6$ यूरो पर ए
        छह, $3$ पाँच पर यूरो, अन्यथा कुछ नहीं
        (\cref{exo:g10:proba:9}). ऊपर $600$ खेल, क्या
        प्रति गेम औसत लाभ की गणना क्या भविष्यवाणी करती है? एक
        बीमाकर्ता वही गणित खेलता है: a $1\,\%$ वार्षिक
        एक का मौका $10\,000$-यूरो दावा --- क्या है
        \emph{गोरा} प्रीमियम, और वास्तविक प्रीमियम क्यों होता है
        इससे अधिक?
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- नमूने as arbiters.}
\begin{enumerate}[resume]
  \item एक सिक्का उछाला जाता है $100$ बार. उतार-चढ़ाव का उपयोग करना
        \omterm{def:g10:numbers:interval}{अंतराल} $ p \pm \frac{1}{\sqrt n}$, तय करें कि कौन सा
        इन नतीजों पर भौंहें तननी चाहिए: $55$ सिर; $70$
        सिर.
  \item का एक सर्वेक्षण $1\,000$ मतदाता एक उम्मीदवार देता है $52\,\%$.
        उतार-चढ़ाव की गणना करें \omterm{def:g10:numbers:interval}{अंतराल} आस-पास $50\,\%$ और
        कहो क्या जनमत संग्रह \emph{को सिद्ध करता} उम्मीदवार है
        आगे.
  \item एक फैक्ट्री का दावा है $4\,\%$ दोषपूर्ण भाग. ए \omterm{def:g10:proba:sample}{नमूना} का
        $400$ दिखाता है $28$ दोषपूर्ण ($7\,\%$); एक बाद में \omterm{def:g10:proba:sample}{नमूना} का
        $2\,500$ दिखाता है $175$ ($7\,\%$ दोबारा)। दावे का परीक्षण करें
        प्रत्येक के विरुद्ध \omterm{def:g10:proba:sample}{नमूना}. फैसले अलग-अलग क्यों होते हैं?
  \item प्रत्येक सर्वेक्षण में दो त्रुटि स्रोत परेशान करते हैं: नमूनाकरण
        \emph{पक्षपात} (मिडिल स्कूल वॉल्यूम की कैसे-डेटा-झूठ सप्ताहांत समस्या) और नमूनाकरण
        \emph{अस्थिरता} (यह अध्याय). कौन सा सिकुड़ता है
        $ n $ बढ़ता है, और कौन सा कोई जीवित रहता है \omterm{def:g10:proba:sample}{नमूना} आकार? एक
        प्रत्येक वाक्य.
  \item समापन --- संभाव्यता की चेकलिस्ट, एक पंक्ति प्रति
        आइटम: गिनती से पहले परिणाम समान रूप से संभावित हैं
        (प्रश्न 15); \omterm{met:g10:proba:tree}{पेड़} और \omterm{def:g10:proba:operations}{पूरक} मल्टी-स्टेप और के लिए
        ``कम से कम'' (प्रश्न 2--3); \emph{जानकारी
        संभावनाएँ बदलती हैं, और इसे कैसे प्राप्त किया गया यह मायने रखता है}
        (प्रश्न 7-14); नमूने के बारे में उतार-चढ़ाव
        $\frac{1}{\sqrt n}$, और अनुकरण मध्यस्थता
        (प्रश्न 11,16--18).
\end{enumerate}
\end{problem}
