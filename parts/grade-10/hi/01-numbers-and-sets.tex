\chapter{संख्याएँ और संख्याओं का समूह}\label{ch:g10:numbers}

गणित संख्याओं से शुरू होता है, और सभी संख्याएँ एक ही प्रकार की नहीं होती हैं:
संख्याओं, ऋणात्मक संख्याओं, भिन्नों और संख्याओं की गिनती करना $\sqrt 2$
या $\pi $ जिसे कोई भी अंश व्यक्त नहीं कर सकता। यह अध्याय उन्हें व्यवस्थित करता है
नेस्टेड परिवार, परिचय \omterm{def:g10:numbers:interval}{अंतराल} संख्या के भागों का वर्णन करने के लिए
लाइन, और का उपयोग करता है \omterm{def:g10:numbers:abs}{निरपेक्ष मूल्य} संख्याओं के बीच की दूरी मापने के लिए.

\section{संख्याओं के परिवार}

\begin{definition}[संख्याओं का समूह]\label{def:g10:numbers:sets}
\begin{itemize}
  \item $\N $ का सेट है \emph{प्राकृतिक संख्या}\index{प्राकृतिक संख्या}:
        $0, 1, 2, 3, \dots $
  \item $\Z $ का सेट है \emph{पूर्णांकों}\index{पूर्णांकों}: $\dots, -2, -1,
        0, 1, 2, \dots $
  \item $\Q $ का सेट है \emph{भिन्नात्मक संख्याएं}\index{भिन्नात्मक संख्याएं}:
        सभी भागफल $\frac{p}{q}$ साथ $ p \in \Z $, $ q \in \N $ और
        $ q \neq 0$.
  \item $\R $ का सेट है \emph{वास्तविक संख्या}\index{वास्तविक संख्या}: सभी
        वे संख्याएँ जिन्हें संख्या रेखा पर रखा जा सकता है।
\end{itemize}
\end{definition}

\begin{notation}
प्रतीक $\in $ पढ़ता है ``इससे संबंधित है'': $3 \in \N $, $-5 \in \Z $,
$\frac23 \in \Q $. प्रतीक $\subset $ पढ़ता है ``इसमें शामिल है'': प्रत्येक
\omterm{def:g10:numbers:sets}{प्राकृतिक संख्या} एक पूर्णांक, प्रत्येक पूर्णांक एक है तर्कसंगत संख्या
(उदा.\ $-5 = \frac{-5}{1}$), और हर तर्कसंगत संख्या वास्तविक है, इसलिए
\[
\N \subset \Z \subset \Q \subset \R .
\]
एक स्लैश एक प्रतीक को नकारता है: $\frac12 \notin \Z $.
\end{notation}

\begin{example}\label{ex:g10:numbers:classify}
आइए हम उनमें से सबसे छोटे परिवार में कुछ संख्याएँ रखें।
\begin{itemize}
  \item $\frac{28}{4} = 7$, इसलिए $\frac{28}{4} \in \N $ भले ही यह है
        भिन्न के रूप में लिखा गया: हमेशा पहले सरलीकरण करें।
  \item $-3.5 = -\frac{35}{10} = -\frac72$ तर्कसंगत है लेकिन एक नहीं पूर्णांक.
  \item $0.333\dots $ (अंक $3$ हमेशा के लिए दोहराना) बराबर है $\frac13$,
        ए तर्कसंगत संख्या.
  \item $\sqrt 2$ और $\pi $ वास्तविक हैं लेकिन तर्कसंगत नहीं, जैसा कि हम करने वाले हैं
        के लिए देखें $\sqrt 2$; ऐसे नंबरों को कॉल किया जाता है
        \emph{तर्कहीन}\index{तर्कहीन}.
\end{itemize}
\end{example}

\begin{theorem}[की अतार्किकता $\sqrt 2$]\label{thm:g10:numbers:sqrt2}
जो नंबर $\sqrt 2$ तर्कसंगत नहीं है: का कोई अंश नहीं \omterm{def:g10:numbers:sets}{पूर्णांकों} वर्गाकार है
$2$.
\end{theorem}

\begin{proof}
हम छोटे-छोटे चरणों में विरोधाभास के आधार पर तर्क करते हैं।
\begin{enumerate}
  \item कल्पना करना $\sqrt 2 = \frac{p}{q}$ कहाँ $ p $ और $ q $ सकारात्मक हैं
        \omterm{def:g10:numbers:sets}{पूर्णांकों}, और अंश पूरी तरह से सरलीकृत है, इसलिए $ p $ और $ q $ हैं
        दोनों भी नहीं.
  \item दोनों पक्षों का वर्ग करने पर प्राप्त होता है $2 = \frac{p^2}{q^2}$, वह है
        $ p^2 = 2q^2$. इसलिए $ p^2$ सम है.
  \item अगर $ p $ अजीब थे, कहते हैं $ p = 2k+1$, तब
        $ p^2 = 4k^2 + 4k + 1$ अजीब होगा. तब से $ p^2$ सम है, $ p $ अवश्य
        सम हो: $ p = 2k $ कुछ के लिए पूर्णांक $ k $.
  \item प्रतिस्थापन: $(2k)^2 = 2q^2$, इसलिए $4k^2 = 2q^2$, इसलिए
        $ q^2 = 2k^2$. चरण 3 के समान तर्क से, $ q $ सम है.
  \item अब $ p $ और $ q $ दोनों सम हैं, चरण 1 का खंडन करते हुए
        अनुमान असंभव था: $\sqrt 2$ है \omterm{ex:g10:numbers:classify}{तर्कहीन}.
\end{enumerate}
\end{proof}

\begin{remark}
\omterm{def:g10:numbers:sets}{भिन्नात्मक संख्याएं} बिल्कुल वही संख्याएँ हैं जिनका दशमलव विस्तार या तो
रुकता है (जैसे $\frac72 = 3.5$) या अंततः उसी ब्लॉक को हमेशा के लिए दोहराता है
(जैसे $\frac13 = 0.333\dots $ या $\frac{1}{7} = 0.142857\,142857\dots $).
\omterm{ex:g10:numbers:classify}{तर्कहीन} संख्या पसंद $\sqrt2 = 1.41421356\dots $ दशमलव अंक हैं
जो कभी भी दोहराए जाने वाले पैटर्न में नहीं आता। हम इस लक्षण वर्णन को स्वीकार करते हैं
इस स्तर पर.
\end{remark}

\section{संख्या रेखा और अंतराल}

\omterm{def:g10:numbers:sets}{वास्तविक संख्या} एक पंक्ति भरें: एक मूल चुनना $0$ और लंबाई की एक इकाई,
प्रत्येक \omterm{def:g10:numbers:sets}{वास्तविक संख्या} बिल्कुल एक बिंदु से मेल खाता है.

\begin{definition}[अंतराल]\label{def:g10:numbers:interval}
होने देना $ a $ और $ b $ होना \omterm{def:g10:numbers:sets}{वास्तविक संख्या} साथ $ a < b $. एक
\emph{अंतराल}\index{अंतराल} सबका समुच्चय है \omterm{def:g10:numbers:sets}{वास्तविक संख्या} दो के बीच
सीमा. वर्गाकार कोष्ठक का अर्थ है कि सीमा सम्मिलित है, कोष्ठक का अर्थ है
इसे बाहर रखा गया है:
\begin{center}
\begin{tabular}{c|c}
notation & description \\
\hline
$\intcc{a}{b}$ & $ a \leq x \leq b $ (both ends included) \\
$\intoo{a}{b}$ & $ a < x < b $ (both ends excluded) \\
$\intco{a}{b}$ & $ a \leq x < b $ \\
$\intoc{a}{b}$ & $ a < x \leq b $ \\
$\intco{a}{+\infty}$ & $ x \geq a $ \\
$\intoo{-\infty}{b}$ & $ x < b $ \\
\end{tabular}
\end{center}
प्रतीक $-\infty $ और $+\infty $ (``अनन्तता'') संख्याएँ नहीं हैं, केवल a हैं
कहने का ढंग कि अंतराल सदैव चलता रहता है; बगल में ब्रैकेट
वे सदैव एक कोष्ठक होते हैं। पूरी लाइन $\R $ अंतराल है
$\intoo{-\infty}{+\infty}$.
\end{definition}

\begin{definition}[अंतर्विरोध और मिलन]\label{def:g10:numbers:interunion}
होने देना $ I $ और $ J $ के दो सेट हों \omterm{def:g10:numbers:sets}{वास्तविक संख्या}.
 \emph{चौराहा}\index{चौराहा} $ I \cap J $ (``$ I $ और $ J $'')
दोनों से संबंधित संख्याओं का समूह है;
\emph{मिलन}\index{मिलन} $ I \cup J $ (``$ I $ या $ J $'') का समुच्चय है
उनमें से कम से कम एक से संबंधित संख्याएँ।
\end{definition}

\begin{example}\label{ex:g10:numbers:interunion}
लेना $ I = \intcc{-1}{3}$ और $ J = \intoo{1}{5}$. दोनों को एक ही पर ड्रा करें
पंक्ति: वे बीच में ओवरलैप करते हैं $1$ और $3$. इसलिए
\[
I \cap J = \intoc{1}{3},
\qquad
I \cup J = \intco{-1}{5}.
\]
कोष्ठक पर ध्यान दें: $1 \notin J $ इसलिए $1 \notin I \cap J $, लेकिन $3 \in I $ और
$3 \in J $, इसलिए $3 \in I \cap J $.
\end{example}

\begin{method}[अंतराल के साथ कार्य करना]\label{met:g10:numbers:intervals}
खोजने के लिए \omterm{def:g10:numbers:interunion}{चौराहा} या \omterm{def:g10:numbers:interunion}{मिलन} दोनों में से \omterm{def:g10:numbers:interval}{अंतराल}:
\begin{enumerate}
  \item संख्या रेखा खींचें और चार सीमाएं चिह्नित करें;
  \item पहले छाया करो \omterm{def:g10:numbers:interval}{अंतराल} रेखा के ऊपर और दूसरा उसके नीचे;
  \item  \omterm{def:g10:numbers:interunion}{चौराहा} वह जगह है जहां शेडिंग ओवरलैप होती है \omterm{def:g10:numbers:interunion}{मिलन} कहाँ है
        कम से कम एक छायांकन मौजूद है;
  \item प्रत्येक कोष्ठक का परीक्षण करके निर्णय लें कि क्या सीमा स्वयं से संबंधित है
        दोनों सेट (\omterm{def:g10:numbers:interunion}{चौराहा}) या कम से कम एक (\omterm{def:g10:numbers:interunion}{मिलन}).
\end{enumerate}
\end{method}

\section{निरपेक्ष मान और दूरी}

\begin{definition}[पूर्ण मूल्य]\label{def:g10:numbers:abs}
 \emph{निरपेक्ष मूल्य}\index{निरपेक्ष मूल्य} एक का \omterm{def:g10:numbers:sets}{वास्तविक संख्या} $ x $ है
\[
\abs{x} =
\begin{cases}
x & \text{if } x \geq 0, \\
-x & \text{if } x < 0.
\end{cases}
\]
उदाहरण के लिए $\abs{7} = 7$ और $\abs{-4} = 4$. हर मामले में
$\abs{x} \geq 0$.
\end{definition}

\begin{proposition}[लाइन पर दूरी]\label{prop:g10:numbers:distance}
सभी के लिए \omterm{def:g10:numbers:sets}{वास्तविक संख्या} $ a $ और $ b $, बिंदुओं के बीच की दूरी $ a $ और
$ b $ संख्या रेखा पर है $\abs{b - a}$. विशेष रूप से $\abs{x}$ है
से दूरी $ x $ को $0$.
\end{proposition}

\begin{proof}
अगर $ b \geq a $, से दूरी $ a $ को $ b $ है $ b - a \geq 0$, जो
बराबर $\abs{b-a}$. अगर $ b < a $, दूरी है $ a - b = -(b - a) > 0$,
जो फिर से है $\abs{b - a}$ की परिभाषा के अनुसार \omterm{def:g10:numbers:abs}{निरपेक्ष मूल्य}.
\end{proof}

\begin{proposition}[निरपेक्ष मान और अंतराल]\label{prop:g10:numbers:absinterval}
होने देना $ a $ एक हो \omterm{def:g10:numbers:sets}{वास्तविक संख्या} और $ r > 0$. तब
\[
\abs{x - a} \leq r
\quad\text{exactly when}\quad
x \in \intcc{a - r}{a + r}.
\]
\end{proposition}

\begin{proof}
$\abs{x-a} \leq r $ का कहना है कि से दूरी $ x $ को $ a $ अधिकतम है $ r $,
यानी\ वह $ x $ इससे आगे कुछ नहीं है $ r $ से $ a $ दोनों तरफ़। द
इसे संतुष्ट करने वाली संख्याएँ बिल्कुल बीच की संख्याएँ हैं $ a - r $ और $ a + r $,
सीमाएं शामिल हैं.
\end{proof}

  \draw[omDef, very thick] (1.5,0.02) -- (5.5,0.02);

\end{omfigure}

\begin{example}\label{ex:g10:numbers:abs}
हल करना $\abs{x - 3} \leq 2$. समाधान दूरी पर स्थित संख्याएँ हैं
सबसे $2$ से $3$: द \omterm{def:g10:numbers:interval}{अंतराल} $\intcc{1}{5}$. इसके विपरीत, \omterm{def:g10:numbers:interval}{अंतराल}
$\intcc{-1}{7}$ केंद्र है $\frac{-1+7}{2} = 3$ और त्रिज्या
$\frac{7-(-1)}{2} = 4$, इसलिए इसका वर्णन किया गया है $\abs{x - 3} \leq 4$.
\end{example}

\section{अनुमान}

\omterm{ex:g10:numbers:classify}{तर्कहीन} संख्या, और यहां तक कि अधिकांश भिन्नों को भी सटीक रूप से नहीं लिखा जा सकता है
बहुत से दशमलव अंकों के साथ, इसलिए व्यवहार में हम उनका अनुमान लगाते हैं।

\begin{definition}[किसी दी गई सटीकता का अनुमान]\label{def:g10:numbers:approx}
एक संख्या $ d $ एक \emph{सन्निकटन}\index{सन्निकटन} का $ x $ को
भीतर $10^{-n}$ कब $\abs{x - d} \leq 10^{-n}$. काट-छाँट या गोलाई करना
के बाद दशमलव विस्तार $ n $-वें अंक दोनों ऐसा देते हैं
अनुमान.
\end{definition}

\begin{example}
से $\pi = 3.14159\,26\dots $: कटाव $3.141$ और गोलाई
$3.142$ दोनों अनुमान का $\pi $ भीतर तक $10^{-3}$. गोलाई
से कम दूरी पर है $\frac12 \times 10^{-3}$, केवल काट-छाँट
गारंटी देता है $10^{-3}$. लिखना $3.141 \leq \pi \leq 3.142$ \emph{फ़्रेम}
$\pi $ दो दशमलव सीमाओं के बीच.
\end{example}

\section{अभ्यास}

\begin{exercise}[$\star $]\label{exo:g10:numbers:1}
प्रत्येक संख्या के लिए, सबसे छोटा सेट दें $\N $, $\Z $, $\Q $, $\R $ को
यह किससे संबंधित है:
\[
\frac{15}{3}, \qquad -7, \qquad \frac{22}{7}, \qquad \sqrt{9}, \qquad
\sqrt{10}, \qquad -2.4, \qquad 0 .
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g10:numbers:2}
प्रत्येक कथन को इसके साथ लिखें \omterm{def:g10:numbers:interval}{अंतराल} अंकन, फिर इसे किसी संख्या पर बनाएं
पंक्ति: (ए) $-2 \leq x < 5$; (बी) $ x > 3$; (सी) $ x \leq -1$;
(डी) से दूरी $ x $ को $2$ अधिकतम है $3$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g10:numbers:3}
गणना करना $ I \cap J $ और $ I \cup J $ के लिए
\[
\text{(a) } I = \intcc{-3}{2},\ J = \intco{0}{4};
\qquad
\text{(b) } I = \intoo{-\infty}{1},\ J = \intco{-2}{+\infty}.
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g10:numbers:4}
बिना कैलकुलेटर के गणना करें:
\[
\abs{-6}, \qquad \abs{4 - 9}, \qquad \abs{-3 - 5}, \qquad
\abs{\sqrt 2 - 1}, \qquad \abs{1 - \sqrt 2}.
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g10:numbers:5}
समीकरणों और असमानताओं को हल करें और समाधान सेट दें:
\[
\abs{x} = 5, \qquad \abs{x - 1} = 3, \qquad \abs{x - 4} \leq 1, \qquad
\abs{x + 2} < 3 .
\]
(ध्यान दें कि $\abs{x+2} = \abs{x - (-2)}$ की दूरी है $-2$.)
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:numbers:6}
प्रत्येक का वर्णन करें \omterm{def:g10:numbers:interval}{अंतराल} रूप की असमानता से
$\abs{x - a} \leq r $ या $\abs{x - a} < r $:
\[
\intcc{2}{8}, \qquad \intoo{-5}{1}, \qquad \intcc{-7}{-3}.
\]
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:numbers:7}
बताते हैं कि $0.272727\dots $ (द ब्लॉक $27$ हमेशा के लिए दोहराना) तर्कसंगत है।
(संकेत: इसे कॉल करें $ x $ और गणना करें $100x - x $.)
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:numbers:8}
सच या झूठ? प्रत्येक उत्तर को तर्क या प्रतिउदाहरण के साथ उचित ठहराएँ।
\begin{enumerate}
  \item दो का योग \omterm{def:g10:numbers:sets}{पूर्णांकों} एक पूर्णांक.
  \item दो का भागफल \omterm{def:g10:numbers:sets}{पूर्णांकों} एक पूर्णांक.
  \item दो का योग \omterm{def:g10:numbers:sets}{भिन्नात्मक संख्याएं} तर्कसंगत है.
  \item ए का योग तर्कसंगत संख्या और एक अपरिमेय संख्या है
        \omterm{ex:g10:numbers:classify}{तर्कहीन}.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:numbers:9}
का उपयोग करते हुए $1.414 \leq \sqrt 2 \leq 1.415$, संख्याओं को फ्रेम करें $2\sqrt2$,
$\sqrt2 + 3$ और $-\sqrt2$ दो दशमलव सीमाओं के बीच.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g10:numbers:10}
का प्रमाण अपनाएँ \cref{thm:g10:numbers:sqrt2} यह दिखाने के लिए $\sqrt 3$ है
\omterm{ex:g10:numbers:classify}{तर्कहीन}. (``सम'' को ''मल्टीपल'' से बदलें $3$'': पहले यह जांचें कि क्या
$ p^2$ का गुणज है $3$, तो ऐसा ही है $ p $, अवशेषों की जांच करके
$0$, $1$, $2$ का $ p $ द्वारा विभाजित करने पर $3$.)
\end{exercise}

\section{समस्या: किन्हीं दो संख्याओं के बीच}

\begin{problem}[{सप्ताहांत समस्या --- तर्कसंगत और तर्कहीन इंटरलेस: प्रत्येक अंतराल, चाहे वह कितना ही छोटा क्यों न हो, उसमें अनंत रूप से अनेक शामिल होते हैं प्रत्येक का, और कोई भी माप उन्हें कभी अलग नहीं बता सकता}]
\label{pb:g10:numbers:1}
तर्कसंगत भीड़ की तरह दिखते हैं (सभी अंश!) और
अपरिमेय विदेशी अपवादों की तरह ($\sqrt2$, $\pi $). यह समस्या
असली तस्वीर उजागर करती है: दो परिवार \emph{जिल्द} तो
बारीकी से वह हर \omterm{def:g10:numbers:interval}{अंतराल} हालाँकि, संख्या रेखा का
सूक्ष्म, प्रत्येक के अनंत रूप से अनेक होते हैं --- अजीब के साथ
परिणाम, जैसे कि: हालाँकि, कोई भौतिक माप नहीं
सटीक, कभी भी यह तय कर सकता है कि लंबाई तर्कसंगत है या नहीं।

\medskip
\noindent\textbf{Part I --- The four kingdoms.}
\begin{enumerate}
  \item प्रत्येक संख्या के लिए, सबसे छोटे सेट का नाम बताएं $\N $,
        $\Z $, $\Q $, $\R $ इसमें शामिल है
        (\cref{def:g10:numbers:sets}): $-7$; \ $\frac{13}{4}$; \
        $\sqrt{16}$; \ $0.121212\ldots $ (याद करें
        मिडिल स्कूल वॉल्यूम की दोहराव-दशमलव सप्ताहांत समस्या); \ $\sqrt8$; \ $\pi $ (इसे स्वीकार करें
        अतार्किकता --- केवल विश्वविद्यालय में सिद्ध हुई
        वॉल्यूम)।
  \item साबित करें कि $\Q $ जोड़ के तहत स्थिर है और
        गुणन: यदि $ x = \frac pq $ और $ y = \frac rs $ हैं
        तर्कसंगत, लिखो $ x + y $ और $ xy $ एकल भिन्न के रूप में.
  \item विरोधाभास से निष्कर्ष निकालें: (ए) एक तर्कसंगत और एक का योग
        \omterm{ex:g10:numbers:classify}{तर्कहीन} है \omterm{ex:g10:numbers:classify}{तर्कहीन}; (बी) ए का उत्पाद
        \emph{अशून्य} तर्कसंगत और एक \omterm{ex:g10:numbers:classify}{तर्कहीन} है \omterm{ex:g10:numbers:classify}{तर्कहीन}.
  \item दिखाएँ कि का सेट अपरिमेय के अंतर्गत स्थिर है
        \emph{कोई भी नहीं} ऑपरेशन: दो प्रदर्शित करें अपरिमेय किसका
        योग तर्कसंगत है, और दो जिनका उत्पाद तर्कसंगत है।
  \item का पता लगाने $\sqrt2 + \sqrt8$ और $\sqrt2 \times \sqrt8$ में
        चार राज्य (सरल करें) $\sqrt8$ प्रथम,
        मिडिल स्कूल वॉल्यूम की विधि)।
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The rationals are dense.}
\begin{enumerate}[resume]
  \item के बीच सख्ती से एक तर्कसंगत खोजें $3.47$ और $3.48$; फिर
        एक दूसरा; फिर वर्णन करें कि अधिक से अधिक उत्पादन कैसे किया जाए
        वांछित (मिडिल स्कूल वॉल्यूम की नो-नेक्स्ट-नंबर सप्ताहांत समस्या का ज़ूम, अब एक के साथ
        सबूत सामने है)।
  \item सामान्य प्रमेय. होने देना $ a < b $ कोई दो वास्तविक हों, साथ
        \omterm{def:g10:numbers:interval}{अंतराल} $ g = b - a > 0$. चुनना $ n $ साथ
        $10^{-n} < g $, और के गुणकों पर विचार करें $10^{-n}$
        (चरण का दशमलव ग्रिड $10^{-n}$). स्पष्ट करें कि क्यों
        कम से कम एक ग्रिड बिंदु सख्ती से बीच में आता है $ a $ और $ b $,
        और निष्कर्ष निकालें: \emph{प्रत्येक \omterm{def:g10:numbers:interval}{अंतराल} सकारात्मक लंबाई का
        एक शामिल है तर्कसंगत संख्या}.
  \item निष्कर्ष को अपग्रेड करें: प्रत्येक ऐसा \omterm{def:g10:numbers:interval}{अंतराल} रोकना
        \emph{असीम रूप से अनेक} तर्कसंगत. (प्रश्न 7 लागू करें
        फिर से, एक छोटे से अंदर \omterm{def:g10:numbers:interval}{अंतराल}.)
  \item अब अपरिमेय: दिया गया $ a < b $, एक तर्कसंगत ले लो $ r $
        सख्ती से अंदर (प्रश्न 7) और संख्याओं पर विचार करें
        $ r + \frac{\sqrt2}{10^k}$. प्रश्न 3 का प्रयोग करते हुए उन्हें दिखाएँ
        हैं \omterm{ex:g10:numbers:classify}{तर्कहीन}, और वह के लिए $ k $ वे अभी भी काफी बड़े हैं
        में झूठ बोलना \omterm{def:g10:numbers:interval}{अंतराल}: \emph{प्रत्येक \omterm{def:g10:numbers:interval}{अंतराल} भी शामिल है
        असीम रूप से अनेक अपरिमेय}.
  \item दो क्लासिक पहेलियां सुलझीं: क्या कोई सबसे छोटा है
        सकारात्मक \omterm{def:g10:numbers:sets}{वास्तविक संख्या}? ए \omterm{def:g10:numbers:sets}{वास्तविक संख्या} ``बस बाद में
        $3$''? प्रश्न 7 के प्रमेय के साथ दोनों का उत्तर दें, और
        बचपन के संस्करण को सलाम
        (मिडिल स्कूल वॉल्यूम की नो-नेक्स्ट-नंबर सप्ताहांत समस्या)।
\end{enumerate}

\medskip
\noindent\textbf{Part III --- पूर्ण मूल्य, the geometry of
$\R $.}
\begin{enumerate}[resume]
  \item हल करें और समाधान को इस रूप में व्यक्त करें \omterm{def:g10:numbers:interval}{अंतराल} या यूनियन
        (\cref{prop:g10:numbers:absinterval}):
        $\abs{x - 5} = 2$; \quad $\abs{x - 5} < 2$; \quad
        $\abs{x + 1} \geq 3$.
  \item एक पिस्टन का निर्माण किया जाना चाहिए $80$ ए के भीतर मिमी
        की सहनशीलता $0.05$ मिमी. आवश्यकता को a के साथ लिखें
        \omterm{def:g10:numbers:abs}{निरपेक्ष मूल्य}, फिर एक के रूप में \omterm{def:g10:numbers:interval}{अंतराल}. दो पिस्टन माप
        $79.97$ और $80.06$ मिमी: फैसले?
  \item रेखा पर त्रिभुज असमानता:
        $\abs{a + b} \leq \abs a + \abs b $. इसकी जांच करें
        $(a, b) = (3, -5)$ और $(-2, -7)$, इसे कब साबित करें $ a $ और
        $ b $ एक ही चिन्ह हो और जब उनका चिन्ह विपरीत हो
        संकेत, और ठीक-ठीक बताएं कि समानता कब है।
  \item हल करना $\abs{x - 2} = \abs{x + 4}$ इसे एक के रूप में पढ़कर
        रेखा पर दूरियों की समानता. किस बिंदु का
        क्या आपने अभी-अभी मिडिल स्कूल वॉल्यूम की टू-मिरर सप्ताहांत समस्या की गणना की है, एक
        आयाम नीचे?
  \item सरल $\sqrt{x^2}$ --- सावधानी से। अपने सूत्र का परीक्षण करें
        पर $ x = 3$ और $ x = -3$, फिर इसे हल करने के लिए उपयोग करें
        $ x^2 < 9$ एक साथ \omterm{def:g10:numbers:abs}{निरपेक्ष मूल्य}.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- What no measurement can decide.}
\begin{enumerate}[resume]
  \item एक भौतिक विज्ञानी एक छड़ को मापता है: $1.41421 \pm 0.00001$ एम।
        क्या ऐसा कोई माप हो सकता है --- यह एक या भविष्य, और अधिक
        सटीक एक---कभी भी \emph{सिद्ध करना} कि छड़ की लंबाई
        है \omterm{ex:g10:numbers:classify}{तर्कहीन}? या कि यह तर्कसंगत है? प्रश्न 7 का प्रयोग करें
        और सहनशीलता पर 9 \omterm{def:g10:numbers:interval}{अंतराल}, और समझाएं कि केवल क्यों
        \emph{सबूत} (विकर्ण के संबंध में, मिडिल स्कूल वॉल्यूम की अतार्किकता सप्ताहांत समस्या)
        अतार्किकता को सुलझा सकता है.
  \item काट-छाँट $1.4$, $1.41$, $1.414$, $1.4142,\dots $
        का $\sqrt2$ सभी तर्कसंगत हैं. वो क्या दिखाते हैं
        कितना करीब $\Q $ हर किसी के खिलाफ दबाव डालता है \omterm{ex:g10:numbers:classify}{तर्कहीन}?
        सामान्य तथ्य निरूपित करें।
  \item \omterm{def:g10:numbers:interval}{अंतराल} अंकगणित: मापा गया मान $ a = 2.5 \pm 0.1$
        और $ b = 1.2 \pm 0.1$. ब्रैकेट $ a + b $ और $ a \times b $
        कुछ निश्चित सीमाओं के बीच. कौन सा ऑपरेशन ख़राब करता है
        परिशुद्धता अधिक?
  \item एक इंजीनियर का नियम स्वतंत्र की त्रुटियाँ कहता है
        योगों के लिए माप ``जोड़ें'': त्रिकोण के साथ व्यक्त करें
        असमानता (प्रश्न 13) की त्रुटि क्यों $ a + b $ पर है
        अधिकांश त्रुटियों का योग---असमानता है
        इंजीनियरिंग नियम.
  \item समापन, एक संक्षिप्त पैराग्राफ में: चित्र को इकट्ठा करें
        इस समस्या और इसके द्वारा स्थापित वास्तविक रेखा
        पूर्वज---कोई अगली संख्या नहीं
        (मिडिल स्कूल वॉल्यूम की नो-नेक्स्ट-नंबर सप्ताहांत समस्या), तर्कसंगत $=$ दोहराना
        दशमलव (मिडिल स्कूल वॉल्यूम की दोहराव-दशमलव सप्ताहांत समस्या), दोनों परिवार घने हैं
        (प्रश्न 8 और 9), माप हमेशा के लिए अनिर्णीत
        (प्रश्न 16). यह अध्याय टीज़र के साथ समाप्त नहीं हो सकता
        अभी तक साबित करें: सटीक अर्थ में वहाँ हैं \emph{बड़े पैमाने पर
        अधिक} अपरिमेय तर्कसंगत से --- विश्वविद्यालय
        मात्राएँ अनंत को गिनती हैं।
\end{enumerate}
\end{problem}
