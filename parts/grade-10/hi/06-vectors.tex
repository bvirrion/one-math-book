\chapter{वैक्टर}\label{ch:g10:vectors}

ए \omterm{def:g10:vectors:vector}{वेक्टर} एक विस्थापन को एनकोड करता है: एक दिशा और एक लंबाई, चाहे कुछ भी हो
प्रारंभिक बिंदु. वैक्टर ज्यामितीय तथ्यों का स्पष्ट प्रमाण दीजिए और, एक बार
\omterm{def:g10:coordgeom:system}{COORDINATES} चित्र दर्ज करें, उन्हें छोटी गणनाओं में घटाएँ। वे
में एक महत्वपूर्ण उपकरण होगा \cref{ch:g10:lines} और, बाद में, अदिश राशि के लिए
उत्पाद में \cref{ch:g11:scal}.

\section{अनुवाद और वेक्टर}

\begin{definition}[वेक्टर]\label{def:g10:vectors:vector}
दो अंक दिए गए $A $ और $ B $, द \emph{वेक्टर}\index{वेक्टर} $\vect{AB}$
से विस्थापन को दर्शाता है $ A $ को $ B $: यह है एक
\emph{दिशा} (रेखा की दिशा $(AB)$, रास्ते के साथ
इसके साथ यात्रा करने से, से $ A $ की ओर $ B $) और ए
\emph{लंबाई} (दूरी $ AB $, भी लिखा है $\norm{\vect{AB}}$).
दो वेक्टर हैं \emph{बराबर} जब वे समान विस्थापन को एन्कोड करते हैं:
$\vect{AB} = \vect{CD}$ इसका मतलब है कि दोनों विस्थापनों को किसी एक पर लागू करना
बिंदु वही परिणाम देता है.
\end{definition}

\begin{proposition}[समान सदिश और समांतर चतुर्भुज]\label{prop:g10:vectors:parallelogram}
$\vect{AB} = \vect{CD}$ बिल्कुल कब $ ABDC $ (इस क्रम में!) एक है
समांतर चतुर्भुज, संभवतः एक चपटा --- समकक्ष, जब $[AD]$ और
$[BC]$ एक ही लें \omterm{prop:g10:coordgeom:midpoint}{मध्य}.
\end{proposition}
\admitted

\begin{notation}
ए \omterm{def:g10:vectors:vector}{वेक्टर} अक्सर एक ही अक्षर से नाम दिया जाता है, $\vec u $ या $\vec v $, जब
अंतिमबिंदु कोई मायने नहीं रखते. \emph{शून्य \omterm{def:g10:vectors:vector}{वेक्टर}} $\vec 0 = \vect{AA}$ है
वह विस्थापन जो कुछ भी नहीं हिलाता। \emph{विलोम} का
$\vect{AB}$ है $\vect{BA}$: समान लंबाई, विपरीत दिशा। हम लिखते हैं
$\vect{BA} = -\vect{AB}$.
\end{notation}

\section{वेक्टर जोड़ना}

\begin{definition}[दो सदिशों का योग]\label{def:g10:vectors:sum}
 \emph{जोड़}\index{जोड़} $\vec u + \vec v $ विस्थापन है
प्रदर्शन करके प्राप्त किया गया $\vec u $ तब $\vec v $.
\end{definition}

\begin{theorem}[चैसल्स का रिश्ता]\label{thm:g10:vectors:chasles}
किन्हीं तीन बिंदुओं के लिए $ A $, $ B $, $ C $:
\[
\vect{AB} + \vect{BC} = \vect{AC}.
\]
\end{theorem}

\begin{proof}
से जा रहा हूँ $ A $ को $ B $, फिर से $ B $ को $ C $, एक विस्थापन है जो लेता है
$ A $ को $ C $: यह द्वारा एन्कोड किया गया विस्थापन है $\vect{AC}$.
\end{proof}

\begin{remark}[समांतर चतुर्भुज नियम]
दो जोड़ने के लिए वैक्टर उसी बिंदु से खींचा गया, $\vect{AB} + \vect{AC}$,
समांतर चतुर्भुज को पूरा करें $ ABDC $: द \omterm{def:g10:vectors:sum}{जोड़} विकर्ण है $\vect{AD}$.
वास्तव में $\vect{AC} = \vect{BD}$, इसलिए
$\vect{AB} + \vect{AC} = \vect{AB} + \vect{BD} = \vect{AD}$ चैसल्स द्वारा'
संबंध.
\end{remark}

\begin{example}[चैसल्स के साथ सरलीकरण]\label{ex:g10:vectors:chasles}
सरल $\vect{MN} + \vect{NP} + \vect{PQ}$: विस्थापनों को जंजीर में बांधना
देता है $\vect{MQ}$. सरल $\vect{AB} - \vect{AC}$: पुनः लिखें
एक के रूप में अंतर \omterm{def:g10:vectors:sum}{जोड़} विपरीत के साथ \omterm{def:g10:vectors:vector}{वेक्टर},
\[
\vect{AB} - \vect{AC} = \vect{AB} + \vect{CA}
= \vect{CA} + \vect{AB} = \vect{CB}.
\]
\end{example}

\section{किसी सदिश को किसी संख्या से गुणा करना}

\begin{definition}[अदिश गुणक]\label{def:g10:vectors:scalar}
होने देना $\vec u $ शून्येतर हो \omterm{def:g10:vectors:vector}{वेक्टर} और $ k $ ए \omterm{def:g10:numbers:sets}{वास्तविक संख्या}. \omterm{def:g10:vectors:vector}{वेक्टर}
$ k\vec u $\index{वेक्टर!अदिश गुणक} जैसी ही दिशा है
$\vec u $, लंबाई $\abs{k} \times \norm{\vec u}$, और उसी तरह इंगित करता है
जैसे $\vec u $ अगर $ k > 0$, विपरीत तरीका अगर $ k < 0$. के लिए $ k = 0$,
$0\vec u = \vec 0$.
\end{definition}

\begin{definition}[समरैखिकता]\label{def:g10:vectors:collinear}
दो वैक्टर $\vec u $ और $\vec v $ हैं
\emph{समरेख}\index{समरेख} जब उनके पास समान हो
दिशा, यानी\ कब $\vec v = k \vec u $ कुछ वास्तविक के लिए $ k $ (या इनमें से एक
वे है $\vec 0$).
\end{definition}

\begin{remark}
संरेखता है \omterm{def:g10:vectors:vector}{वेक्टर} समानता और संरेखण के लिए भाषा:
\begin{itemize}
  \item पंक्तियाँ $(AB)$ और $(CD)$ बिल्कुल समानांतर हैं जब $\vect{AB}$
        और $\vect{CD}$ हैं \omterm{def:g10:vectors:collinear}{समरेख};
  \item बिन्दु $ A $, $ B $, $ C $ ठीक उसी समय संरेखित होते हैं $\vect{AB}$ और
        $\vect{AC}$ हैं \omterm{def:g10:vectors:collinear}{समरेख}.
\end{itemize}
\end{remark}

\section{सदिशों के निर्देशांक}

\begin{definition}[एक वेक्टर के निर्देशांक]\label{def:g10:vectors:coords}
में एक \omterm{def:g10:coordgeom:system}{निर्देशांक तरीका}, द \emph{\omterm{def:g10:coordgeom:system}{COORDINATES}} की \omterm{def:g10:vectors:vector}{वेक्टर} $\vect{AB}$
विस्थापन का वर्णन करने वाली संख्याएँ हैं:
\[
\vect{AB}\,(x_B - x_A,\ y_B - y_A).
\]
ए \omterm{def:g10:vectors:vector}{वेक्टर} $\vec u\,(a, b)$ हर बिंदु को हिलाता है $ a $ इकाइयाँ क्षैतिज रूप से और
$ b $ इकाइयाँ लंबवत्।
\end{definition}

\begin{proposition}[निर्देशांक के साथ कंप्यूटिंग]\label{prop:g10:vectors:coordrules}
होने देना $\vec u\,(a, b)$ और $\vec v\,(c, d)$, और $ k \in \R $.
\begin{enumerate}
  \item $\vec u = \vec v $ बिल्कुल कब $ a = c $ और $ b = d $;
  \item $\vec u + \vec v $ है \omterm{def:g10:coordgeom:system}{COORDINATES} $(a + c,\ b + d)$;
  \item $ k\vec u $ है \omterm{def:g10:coordgeom:system}{COORDINATES} $(ka,\ kb)$;
  \item $\norm{\vec u} = \sqrt{a^2 + b^2}$ (\omterm{def:g10:coordgeom:system}{ऑर्थोनॉर्मल} प्रणाली).
\end{enumerate}
\end{proposition}

\begin{proof}
1--3 पुनर्स्थापन, निर्देशांक द्वारा समन्वय, विस्थापन क्या करते हैं: के लिए
उदाहरण प्रदर्शन $\vec u $ तब $\vec v $ एक बिंदु को क्षैतिज रूप से घुमाता है
$ a $ तब $ c $, इसलिए द्वारा $ a + c $ कुल मिलाकर। बिन्दु 4 दूरी का सूत्र है
का \cref{thm:g10:coordgeom:distance} प्रतिनिधित्व करने वाले एक खंड पर लागू किया गया
$\vec u $.
\end{proof}

\begin{theorem}[संरेखता मानदंड]\label{thm:g10:vectors:det}
दो वैक्टर $\vec u\,(a, b)$ और $\vec v\,(c, d)$ हैं \omterm{def:g10:vectors:collinear}{समरेख} अगर और केवल अगर
\[
ad - bc = 0 .
\]
\end{theorem}

\begin{proof}
अगर $\vec v = k\vec u $, तब $ c = ka $ और $ d = kb $, इसलिए
$ ad - bc = a(kb) - b(ka) = 0$. यदि यही बात लागू होती है $\vec u = k \vec v $ या यदि
या तो \omterm{def:g10:vectors:vector}{वेक्टर} शून्य है.

इसके विपरीत, मान लीजिए $ ad - bc = 0$ साथ $\vec u \neq \vec 0$, कहना
$ a \neq 0$ (मामला $ b \neq 0$ समान है)। तय करना $ k = \frac{c}{a}$; तब
$ c = ka $, और $ ad = bc = b(ka)$ से विभाजित करके देता है $ a \neq 0$,
$ d = kb $. इसलिए $\vec v = k\vec u $.
\end{proof}

\begin{example}\label{ex:g10:vectors:det}
हैं $\vec u\,(3, -2)$ और $\vec v\,(-6, 4)$ \omterm{def:g10:vectors:collinear}{समरेख}? गणना करना
$ ad - bc = 3 \times 4 - (-2)\times(-6) = 12 - 12 = 0$: हाँ, और वास्तव में
$\vec v = -2\vec u $. के लिए $\vec u\,(3, -2)$ और $\vec w\,(1, 5)$:
$3 \times 5 - (-2) \times 1 = 17 \neq 0$: नहीं \omterm{def:g10:vectors:collinear}{समरेख}.
\end{example}

\begin{method}[संरेखण और समानता]\label{met:g10:vectors:alignment}
इसे सिद्ध करने के लिए तीन बिंदु $ A $, $ B $, $ C $ संरेखित हैं:
\begin{enumerate}
  \item की गणना करें \omterm{def:g10:coordgeom:system}{COORDINATES} का $\vect{AB}$ और $\vect{AC}$;
  \item मानदंड की जाँच करें $ ad - bc = 0$ का \cref{thm:g10:vectors:det};
  \item निष्कर्ष: वैक्टर हैं \omterm{def:g10:vectors:collinear}{समरेख} और बात साझा करें $ A $, so
        तीन बिंदु संरेखित हैं।
\end{enumerate}
के साथ भी यही गणना $\vect{AB}$ और $\vect{CD}$ साबित करता है कि
पंक्तियाँ $(AB)$ और $(CD)$ समानांतर हैं.
\end{method}

\begin{example}\label{ex:g10:vectors:alignment}
होने देना $ A(1, 2)$, $ B(3, 5)$ और $ C(7, 11)$. तब $\vect{AB}\,(2, 3)$ और
$\vect{AC}\,(6, 9)$, और $2 \times 9 - 3 \times 6 = 18 - 18 = 0$: द
बिंदु संरेखित हैं (वास्तव में $\vect{AC} = 3\vect{AB}$).
\end{example}

\section{अभ्यास}

\begin{exercise}[$\star $]\label{exo:g10:vectors:1}
चैसल्स संबंध का उपयोग करके सरल बनाएं:
\[
\vect{AB} + \vect{BD}, \qquad
\vect{KL} + \vect{LM} + \vect{MK}, \qquad
\vect{AC} - \vect{BC}, \qquad
\vect{AB} + \vect{CA}.
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g10:vectors:2}
होने देना $ A(2, 1)$, $ B(5, 3)$, $ C(-1, 4)$. की गणना करें \omterm{def:g10:coordgeom:system}{COORDINATES} का
$\vect{AB}$, $\vect{BC}$, $\vect{AC}$, और जाँच करें \omterm{def:g10:coordgeom:system}{COORDINATES} वह
$\vect{AB} + \vect{BC} = \vect{AC}$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g10:vectors:3}
होने देना $\vec u\,(2, -3)$ और $\vec v\,(-1, 4)$. दे दो \omterm{def:g10:coordgeom:system}{COORDINATES} का
$\vec u + \vec v $, $3\vec u $, $2\vec u - \vec v $, और गणना करें
$\norm{\vec u}$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g10:vectors:4}
प्रत्येक मामले में, कहें कि क्या $\vec u $ और $\vec v $ हैं \omterm{def:g10:vectors:collinear}{समरेख}:
\[
\text{(a) } \vec u\,(4, 6),\ \vec v\,(6, 9);
\qquad
\text{(b) } \vec u\,(2, -5),\ \vec v\,(-4, 10);
\qquad
\text{(c) } \vec u\,(3, 1),\ \vec v\,(1, 3).
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g10:vectors:5}
होने देना $ A(0, 2)$, $ B(4, 0)$, $ C(6, 3)$. खोजें \omterm{def:g10:coordgeom:system}{COORDINATES} मुद्दे की
$ D $ ऐसा है कि $ ABCD $ एक समांतर चतुर्भुज है, अर्थात\ ऐसा है कि
$\vect{AB} = \vect{DC}$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:vectors:6}
होने देना $ A(-2, 1)$, $ B(1, 3)$, $ C(4, -1)$.
\begin{enumerate}
  \item की गणना करें \omterm{def:g10:coordgeom:system}{COORDINATES} मुद्दे की $ M $ ऐसा है कि
        $\vect{AM} = \vect{AB} + \vect{AC}$.
  \item चतुर्भुज क्या है $ ABMC $? औचित्य।
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:vectors:7}
होने देना $ A(1, -1)$, $ B(4, 1)$, $ C(10, 5)$. बिंदु हैं $ A $, $ B $, $ C $
संरेखित? के लिए वही प्रश्न $ A(0, 3)$, $ B(2, 2)$, $ C(8, -1)$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:vectors:8}
होने देना $ A(1, 2)$, $ B(5, 4)$, $ C(6, 1)$, $ D(2, -1)$. बताते हैं कि $(AB)$ और
$(CD)$ समानांतर हैं, तो वह $ ABCD $ एक समांतर चतुर्भुज है. जो
सत्यापन का तात्पर्य दूसरे से है?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:vectors:9}
होने देना $ ABC $ एक त्रिकोण बनो, और चलो $ I $ द्वारा परिभाषित बिंदु बनें
$\vect{AI} = \frac23 \vect{AB}$. साथ $ A(0,0)$, $ B(6, 3)$, $ C(2, 5)$
(ताकि गणनाएँ सरल रहें):
\begin{enumerate}
  \item की गणना करें \omterm{def:g10:coordgeom:system}{COORDINATES} का $ I $;
  \item होने देना $ J $ ऐसा हो कि $\vect{CJ} = \frac23\vect{CB}$; की गणना करें
        \omterm{def:g10:coordgeom:system}{COORDINATES} का $ J $;
  \item बताते हैं कि $(IJ)$ के समानांतर है $(AC)$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g10:vectors:10}
होने देना $ ABCD $ कोई भी चतुर्भुज हो, और रहने दो $ I $, $ J $, $ K $, $ L $ हो
\omterm{prop:g10:coordgeom:midpoint}{मध्य} बिन्दुओं का $[AB]$, $[BC]$, $[CD]$, $[DA]$. का उपयोग करते हुए \omterm{def:g10:coordgeom:system}{COORDINATES}
$ A(x_A, y_A)$, इत्यादि, वह दिखाओ $\vect{IJ} = \vect{LK}$, और निष्कर्ष निकालो
द \omterm{prop:g10:coordgeom:midpoint}{मध्य} बिन्दुओं के पक्षों का \emph{कोई} चतुर्भुज रूप ए
समांतर चतुर्भुज
\end{exercise}

\section{समस्या: तीरों का बीजगणित}

\begin{problem}[{सप्ताहांत समस्या --- चैसल्स जिम्नास्टिक, द केन्द्रक दो रेखाओं में पुनः सिद्ध हुआ, और बल संतुलन में: क्या वेक्टर वास्तव में हैं}]
\label{pb:g10:vectors:1}
ग्रेड 8 ने सिद्ध कर दिया कि त्रिभुज की माध्यिकाएँ दो तिहाई पर मिलती हैं
उनकी लंबाई --- अंदर छिपे एक सरल समांतर चतुर्भुज के साथ
त्रिकोण (मिडिल स्कूल खंड की केन्द्रक सप्ताहांत समस्या)। वैक्टर इसमें इसकी निंदा करें
दो पंक्तियाँ, और संगामिति मुफ़्त में बाँटें। वही तो है
यह नया बीजगणित इसके लिए है: प्रमेय संगणना बन जाते हैं
तीर. यह समस्या गणना को प्रशिक्षित करती है (सबसे ऊपर चैसल्स),
दो-पंक्ति प्रमाण प्रदान करता है, वेरिग्नन का \omterm{def:g10:algebra:expand}{विस्तार} करता है
(\cref{exo:g10:vectors:10}), और कहाँ समाप्त होता है वैक्टर पैदा हुए थे:
बल और वेग.

\medskip
\noindent\textbf{Part I --- Chasles gymnastics.}
\begin{enumerate}
  \item चैसल्स संबंध का उपयोग करके सरल बनाएं
        (\cref{thm:g10:vectors:chasles}):
        $\vect{AB} + \vect{BC} + \vect{CD}$; \quad
        $\vect{MA} - \vect{MB}$; \quad
        $\vect{AB} + \vect{CD} + \vect{BC} + \vect{DA}$.
  \item मूल युक्ति: उसके लिए दिखाएँ \emph{कोई} बिंदु $ O $,
        $\vect{AB} = \vect{OB} - \vect{OA}$.
  \item के दो लक्षण सिद्ध कीजिए \omterm{prop:g10:coordgeom:midpoint}{मध्य} $ I $ का
        $[AB]$:
        \[
        \vect{IA} + \vect{IB} = \vec 0
        \qquad\text{and}\qquad
        \vect{OI} = \tfrac12\left(\vect{OA} + \vect{OB}\right)
        \ \text{for any } O .
        \]
  \item का उपयोग करते हुए $\vect{AB} = \vect{DC}$, चौथे शीर्ष को पुनः खोजें
        $ D $ समांतर चतुर्भुज का $ ABCD $ साथ $ A(-1, 2)$,
        $ B(3, 4)$, $ C(6, 0)$ --- और तुलना करें
        का मध्यबिंदु-ट्रिक उत्तर \cref{pb:g10:coordgeom:1}.
  \item एक निश्चित \omterm{def:g10:vectors:sum}{जोड़}ा जा रहा है \omterm{def:g10:vectors:vector}{वेक्टर} $\vec u $ के हर बिंदु पर
        समतल कौन सा परिवर्तन करता है --- एक
        मिडिल स्कूल वॉल्यूम की दो-दर्पण सप्ताहांत समस्या में दो दर्पणों के साथ खोजा गया
        और मिडिल स्कूल वॉल्यूम की हाफ-टर्न सप्ताहांत समस्या में दो हाफ-टर्न? गणना करें
        द \omterm{def:g10:functions:function}{छवि} का $(1, 2)$ अंतर्गत $\vec u = (3, -1)$.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The centroid, third proof.}
बिंदु को परिभाषित करें $ G $ एक त्रिकोण का $ ABC $ सुरुचिपूर्ण द्वारा
शर्त
\[
\vect{GA} + \vect{GB} + \vect{GC} = \vec 0 .
\]
\begin{enumerate}[resume]
  \item मूल ट्रिक का उपयोग करके दिखाएँ कि यह स्थिति है
        निर्धारित करता है $ G $ विशिष्ट रूप से:
        $\vect{OG} = \frac13\left(\vect{OA} + \vect{OB} +
        \vect{OC}\right)$ किसी के लिए $ O $ --- इतना \omterm{def:g10:coordgeom:system}{COORDINATES} का
        $ G $ हैं \emph{औसत} शीर्षों का
        (मिडिल स्कूल वॉल्यूम की सेंट्रोइड सप्ताहांत समस्या ने इसे संख्यात्मक रूप से देखा)।
  \item सिद्ध करना $\vect{AG} = \frac13\left(\vect{AB} +
        \vect{AC}\right)$, और, लेखन $ K $ के लिए \omterm{prop:g10:coordgeom:midpoint}{मध्य} का
        $[BC]$, $\vect{AK} = \frac12\left(\vect{AB} +
        \vect{AC}\right)$. निष्कर्ष:
        $\vect{AG} = \frac23\,\vect{AK}$ --- दो-तिहाई
        प्रमेय, पुनः सिद्ध।
  \item मध्यस्थ क्यों से करते हैं? $ B $ और से $ C $ गुजरना
        द \emph{वही} $ G $ बिना किसी अतिरिक्त गणना के?
        इस प्रमेय के तीन उपलब्ध प्रमाणों की तुलना करें ---
        ग्रेड 8 का छिपा हुआ समांतर चतुर्भुज, \omterm{def:g10:coordgeom:system}{COORDINATES}, वैक्टर ---
        प्रत्येक एक वाक्य में.
  \item संख्यात्मक जाँच: $ A(1, 1)$, $ B(5, 2)$, $ C(3, 6)$.
        गणना करना $ G $, तब $ K $, फिर सत्यापित करें
        $\vect{AG} = \frac23 \vect{AK}$ संरेखता के साथ
        कसौटी (\cref{thm:g10:vectors:det}).
  \item फिजिक्स पढ़ता है $ G $ क्योंकि तीनों का संतुलन बिंदु बराबर है
        शीर्षों पर जनसमूह. पर द्रव्यमान दोगुना करें $ A $ (जनता
        $2, 1, 1$): संतुलन बिंदु बन जाता है
        $\vect{OG'} = \frac{2\vect{OA} + \vect{OB} +
        \vect{OC}}{4}$. गणना करना $ G'$ के त्रिकोण के लिए
        प्रश्न 9 और इसकी स्थिति की तुलना करें $ G $.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Collinearity at work.}
\begin{enumerate}[resume]
  \item हैं $\vec u(3, -2)$ और $\vec v(-6, 4)$ \omterm{def:g10:vectors:collinear}{समरेख}? और
        $\vec w(2, 5)$, $\vec z(4, 9)$? के साथ निर्णय लें
        निर्धारक मानदंड.
  \item बिंदु हैं $ A(1, 2)$, $ B(3, 5)$, $ C(7, 11)$
        संरेखित?
  \item वेरिग्नन से परे (\cref{exo:g10:vectors:10}): किसी में
        चतुर्भुज $ ABCD $, द \emph{bimedians} शामिल होना
        \omterm{prop:g10:coordgeom:midpoint}{मध्य} बिन्दुओं विपरीत भुजाओं की ($[IK]$ और $[JL]$). उपयोग करना
        स्थिति वैक्टर ($\vect{OI} = \frac12(\vect{OA} +
        \vect{OB})$, आदि), दिखाएँ कि दोनों द्विमाध्यमियों के पास है
        वही \omterm{prop:g10:coordgeom:midpoint}{मध्य}, पद का \omterm{def:g10:vectors:vector}{वेक्टर}
        $\frac14\left(\vect{OA} + \vect{OB} + \vect{OC} +
        \vect{OD}\right)$: चतुर्भुज का केन्द्रक.
  \item थेल्स एक पंक्ति में: $ M $ और $ N $ परिभाषित बिंदु हैं
        द्वारा $\vect{AM} = \frac13\vect{AB}$ और
        $\vect{AN} = \frac13\vect{AC}$. गणना करना
        $\vect{MN}$ के अनुसार $\vect{BC}$ और निष्कर्ष निकालो
        समानता और अनुपात.
  \item प्रश्न 14 को मनमाने अनुपात में सामान्यीकृत करें $ k $, और
        एक वाक्य में बताएं कि कैसे \omterm{def:g10:vectors:vector}{वेक्टर} कैलकुलस अवशोषित कर लेता है
        द \omterm{prop:g10:coordgeom:midpoint}{मध्य} प्रमेय ( \omterm{prop:g10:coordgeom:midpoint}{मध्य} प्रमेय)
        और थेल्स (थेल्स प्रमेय) एक ही बार में।
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Forces and velocities.}
\begin{enumerate}[resume]
  \item एक नाव को धारा द्वारा धकेला जाता है $3$ किमी/घंटा पूर्व की ओर
        नौकायन करते समय $4$ किमी/घंटा उत्तर की ओर। परिणाम दीजिए
        वेग \omterm{def:g10:vectors:vector}{वेक्टर}, इसकी भयावहता, और वास्तविक वर्णन करें
        पाठ्यक्रम. (एक बहुत पुराना ट्रिपल प्रकट होता है।)
  \item एक वलय पर तीन बल कार्य करते हैं: $\vec u(2, 1)$,
        $\vec v(-3, 2)$, $\vec w(1, -3)$. परिणामी की गणना करें.
        अंगूठी क्या करती है? और क्यों है \omterm{def:g10:vectors:sum}{जोड़} तीरों का
        किसी के साथ \emph{बंद किया हुआ} बहुभुज सदैव $\vec 0$
        (चेसल्स)?
  \item दो ताकतें $\vec u(4, -1)$ और $\vec v(-1, 3)$ ए पर कार्रवाई करें
        हुक. कौन सा तीसरा बल हुक को संतुलन में रखता है?
  \item एक तैराक एक नदी को सीधे पार करते हुए पार करता है
        $2$ एमएस; करंट चलता है $1.5$ एमएस। दे दो
        परिणामी वेग और उसका परिमाण. नदी है
        $50$ मी चौड़ा: क्रॉसिंग में कितना समय लगता है, और कैसे
        तैराक कितनी दूर तक नीचे की ओर उतरता है?
  \item समापन: एक वस्तु, तीन पोशाकें---विस्थापन
        तीर, निर्देशांक युग्म, बल या वेग। सूचीबद्ध करें
        तीन शास्त्रीय प्रमेयों से इस समस्या को पुनः सिद्ध किया गया
        \omterm{def:g10:vectors:vector}{वेक्टर} बीजगणित, और एक ज्यामितीय मात्रा का नाम बताइए
        वैक्टर इस अध्याय का अभी भी माप नहीं किया जा सकता ---
        इसे ठीक करने वाला उपकरण स्केलर उत्पाद के साथ आता है
        ग्रेड 11.
\end{enumerate}
\end{problem}
