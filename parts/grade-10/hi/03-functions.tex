\chapter{कार्य}\label{ch:g10:functions}

ए \omterm{def:g10:functions:function}{समारोह} एक मशीन है जो एक नंबर अंदर लेती है और एक नंबर बाहर निकालती है,
एक ही इनपुट के लिए हमेशा एक ही आउटपुट। कार्य केन्द्रीय हैं
विश्लेषण की वस्तुएं जो आप अगले वर्षों में बनाएंगे
(\cref{ch:g11:quad} आगे); यह अध्याय शब्दावली स्थापित करता है ---
\omterm{def:g10:functions:function}{कार्यक्षेत्र}, \omterm{def:g10:functions:function}{छवि}, \omterm{def:g10:functions:graph}{ग्राफ}, विविधताएँ --- और आवश्यक कौशल को प्रशिक्षित करती हैं
जानकारी पढ़ना a \omterm{def:g10:functions:graph}{ग्राफ}.

\section{शब्दावली: छवियाँ और पूर्व छवियाँ}

\begin{definition}[समारोह]\label{def:g10:functions:function}
होने देना $D $ का एक सेट हो \omterm{def:g10:numbers:sets}{वास्तविक संख्या}. ए \emph{समारोह}\index{समारोह} $ f $
पर परिभाषित $ D $ प्रत्येक नंबर से संबद्ध $ x \in D $ बिल्कुल एक असली
संख्या, लिखा हुआ $ f(x)$ और बुलाया \emph{छवि}\index{छवि} का $ x $.
सेट $ D $ है \emph{कार्यक्षेत्र}\index{कार्यक्षेत्र} का $ f $. हम लिखते हैं
\[
f : x \longmapsto f(x).
\]
अगर $ f(x) = y $, तब $ x $ एक है \emph{पूर्वछवि}\index{पूर्वछवि} का $ y $.
\end{definition}

\begin{remark}
प्रत्येक $ x $ में \omterm{def:g10:functions:function}{कार्यक्षेत्र} बिल्कुल एक है \omterm{def:g10:functions:function}{छवि}, लेकिन एक संख्या $ y $ हो सकता है
अनेक पूर्वछवियाँ, या कोई नहीं. के लिए $ f(x) = x^2$: द \omterm{def:g10:functions:function}{छवि} का $3$ है $9$, और
$9$ दो हैं पूर्वछवियाँ, $3$ और $-3$; जो नंबर $-4$ है कोई \omterm{def:g10:functions:function}{पूर्वछवि}.
\end{remark}

\begin{example}[एक डोमेन ढूँढना]\label{ex:g10:functions:domain}
जब ए \omterm{def:g10:functions:function}{समारोह} एक सूत्र द्वारा दिया गया है, इसका \omterm{def:g10:functions:function}{कार्यक्षेत्र} का सेट है $ x $ के लिए
जो सूत्र समझ में आता है.
\begin{itemize}
  \item $ f(x) = 3x^2 - 5x + 1$ हर किसी के लिए समझ में आता है $ x $: द \omterm{def:g10:functions:function}{कार्यक्षेत्र} है
        $\R $.
  \item $ g(x) = \dfrac{1}{x - 2}$ आवश्यक है $ x - 2 \neq 0$: द \omterm{def:g10:functions:function}{कार्यक्षेत्र} है
        सिवाय सभी वास्तविक के $2$.
  \item $ h(x) = \sqrt{x - 3}$ आवश्यक है $ x - 3 \geq 0$: द \omterm{def:g10:functions:function}{कार्यक्षेत्र} है
        $\intco{3}{+\infty}$.
\end{itemize}
\end{example}

\begin{example}[छवियों और पूर्वछवियों की गणना करना]\label{ex:g10:functions:images}
होने देना $ f(x) = x^2 - 4x + 3$, पर परिभाषित किया गया $\R $.

\emph{\omterm{def:g10:functions:function}{छवि} का $5$:} विकल्प $ x = 5$:
$ f(5) = 25 - 20 + 3 = 8$.

\emph{पूर्वछवियाँ का $3$:} हल करना $ f(x) = 3$:
\[
x^2 - 4x + 3 = 3
\quad\Longleftrightarrow\quad
x^2 - 4x = 0
\quad\Longleftrightarrow\quad
x(x - 4) = 0,
\]
इतना पूर्वछवियाँ का $3$ हैं $0$ और $4$.
\end{example}

\section{किसी फ़ंक्शन का ग्राफ़}

\begin{definition}[ग्राफ़]\label{def:g10:functions:graph}
एक समन्वय प्रणाली में, \emph{ग्राफ}\index{ग्राफ} (या \emph{वक्र}) का
$ f $ सभी बिंदुओं का समुच्चय है $(x, f(x))$ के लिए $ x $ में \omterm{def:g10:functions:function}{कार्यक्षेत्र}. दूसरे में
शब्द, एक बिंदु $(x, y)$ ठीक उसी समय ग्राफ़ पर स्थित होता है $ y = f(x)$.
\end{definition}

  \node[omThm, font=\small, above right] at (axis cs:3,1.2)

\end{axis}

अक्ष, ऊर्ध्वाधर रूप से वक्र पर जाएं, फिर क्षैतिज रूप से ऊर्ध्वाधर अक्ष पर जाएं
पढ़ने के लिए $ f(3) = 1.25$.}
\end{omfigure}

\begin{method}[एक ग्राफ़ पढ़ना]\label{met:g10:functions:reading}
पर \omterm{def:g10:functions:graph}{ग्राफ} का $ f $:
\begin{enumerate}
  \item \emph{\omterm{def:g10:functions:function}{छवि} का $ a $}: से लंबवत जाएं $ x = a $ क्षैतिज पर
        अक्ष से वक्र तक, फिर क्षैतिज रूप से ऊर्ध्वाधर अक्ष तक; द
        वहाँ पढ़ा गया मूल्य है $ f(a)$;
  \item \emph{पूर्वछवियाँ का $ b $}: क्षैतिज रेखा खींचें $ y = b $;
        पूर्वछवियाँ हैं $ x $-इसके सभी के निर्देशांक \omterm{def:g10:numbers:interunion}{चौराहा} अंक
        वक्र के साथ;
  \item \emph{के समाधान $ f(x) = k $}: के समान पूर्वछवियाँ का $ k $;
  \item \emph{के समाधान $ f(x) \leq k $}: द $ x $ जिसके लिए वक्र है
        रेखा पर या नीचे $ y = k $.
\end{enumerate}
\end{method}

का भुज \omterm{def:g10:numbers:interunion}{चौराहा} क्षैतिज के साथ वक्र के बिंदु
लाइन $ y = k $. यहाँ $ f(x) \leq k $ बनाए रखता है $\intcc{x_1}{x_2}$, जहां
वक्र रेखा के नीचे है.}
\end{omfigure}

\begin{example}\label{ex:g10:functions:membership}
मुद्दा यह है $ A(2, 5)$ पर \omterm{def:g10:functions:graph}{ग्राफ} का $ f(x) = x^2 + 1$? गणना करना
$ f(2) = 4 + 1 = 5$: हाँ, तब से $ f(2)$ के बराबर है $ y $-का समन्वय $ A $.
बात $ B(3, 8)$ पर नहीं है \omterm{def:g10:functions:graph}{ग्राफ}, क्योंकि $ f(3) = 10 \neq 8$.
\end{example}

\section{किसी फ़ंक्शन की विविधताएँ}

\begin{definition}[बढ़ रहा है, घट रहा है]\label{def:g10:functions:variations}
होने देना $ f $ एक पर परिभाषित किया जाए \omterm{def:g10:numbers:interval}{अंतराल} $ I $.
\begin{itemize}
  \item $ f $ है \emph{की बढ़ती}\index{की बढ़ती} पर $ I $ कब
        सबके लिए $ u, v \in I $, अगर $ u < v $ तब $ f(u) \leq f(v)$: द
        इनपुट के साथ आउटपुट बढ़ता है, और \omterm{def:g10:functions:graph}{ग्राफ} बायीं ओर से चढ़ता है
        ठीक है.
  \item $ f $ है \emph{घटते}\index{घटते} पर $ I $ कब
        $ u < v $ तात्पर्य $ f(u) \geq f(v)$: द \omterm{def:g10:functions:graph}{ग्राफ} बायीं ओर से गिरता है
        ठीक है.
\end{itemize}
सख्त असमानताओं के साथ $ f(u) < f(v)$ (सम्मान.\ $>$) हम कहते हैं
\emph{कठोरता से} बढ़ रहा है (सम्मान)\ घट रहा है)।
\end{definition}

\begin{definition}[विविधता तालिका]\label{def:g10:functions:table}
ए \emph{विविधता तालिका}\index{विविधता तालिका} जिस पर सारांश प्रस्तुत करता है
\omterm{def:g10:numbers:interval}{अंतराल} $ f $ तीरों से बढ़ता और घटता है $\nearrow $ और
$\searrow $, और के मूल्यों को रिकॉर्ड करता है $ f $ निर्णायक मोड़ पर.
\end{definition}

\begin{example}\label{ex:g10:functions:table}
पर विचार करें \omterm{def:g10:functions:function}{समारोह} नीचे रेखांकन किया गया है $\intcc{-2}{4}$.

  xtick={-2,1,4}, ytick={-1.5,3}, yticklabels={$-1.5$,$3$}]
  \addplot[omDef, thick, domain=-2:4, samples=120] {-0.5*(x-1)^2 + 3};

{\small ए \omterm{def:g10:functions:function}{समारोह} पर परिभाषित $\intcc{-2}{4}$, \omterm{def:g10:functions:variations}{की बढ़ती} तब \omterm{def:g10:functions:variations}{घटते}.}
\end{omfigure}

पढ़ रहा हूँ \omterm{def:g10:functions:graph}{ग्राफ}: $ f $ से बढ़ता है $ f(-2) = -1.5$ इसके ऊपर तक \omterm{def:g10:functions:extrema}{अधिकतम}
$ f(1) = 3$, फिर नीचे तक घट जाती है $ f(4) = -1.5$. इसका \omterm{def:g10:functions:table}{विविधता तालिका} है
\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{c|ccccc}
$ x $ & $-2$ & & $1$ & & $4$ \\
\hline
$ f $ & $-1.5$ & $\nearrow $ & $3$ & $\searrow $ & $-1.5$ \\
\end{tabular}
\end{center}
\end{example}

\begin{example}[भिन्नता सिद्ध करना]\label{ex:g10:functions:proof}
बताते हैं कि $ f(x) = 3x + 1$ सख्ती से है \omterm{def:g10:functions:variations}{की बढ़ती} पर $\R $. कोई भी ले जाओ
$ u < v $ और तुलना करें इमेजिस:
\[
f(v) - f(u) = (3v + 1) - (3u + 1) = 3(v - u) > 0,
\]
क्योंकि $ v - u > 0$. इसलिए $ f(u) < f(v)$: द \omterm{def:g10:functions:function}{समारोह} सख्ती से है
\omterm{def:g10:functions:variations}{की बढ़ती}. नकारात्मक ढलान के साथ समान गणना, उदा.\
$ g(x) = -2x + 5$, देता है $ g(v) - g(u) = -2(v-u) < 0$: कठोरता से \omterm{def:g10:functions:variations}{घटते}.
\end{example}

\section{एक्सट्रीमा}

\begin{definition}[अधिकतम न्यूनतम]\label{def:g10:functions:extrema}
होने देना $ f $ पर परिभाषित किया जाए $ D $ और $ a \in D $. मूल्य $ f(a)$ है
\emph{अधिकतम}\index{अधिकतम} का $ f $ पर $ D $ कब $ f(x) \leq f(a)$ सभी के लिए
$ x \in D $; यह है \emph{न्यूनतम}\index{न्यूनतम} कब
$ f(x) \geq f(a)$ सभी के लिए $ x \in D $. एक पर \omterm{def:g10:functions:graph}{ग्राफ}, वे उच्चतम हैं और
वक्र के निम्नतम बिंदु.
\end{definition}

\begin{example}\label{ex:g10:functions:extremum}
बताते हैं कि $ f(x) = (x - 1)^2 + 2$ है \omterm{def:g10:functions:extrema}{न्यूनतम} $2$ पर $\R $, पर प्राप्त किया गया
$ x = 1$. हरएक के लिए $ x $, वर्ग $(x-1)^2$ है $\geq 0$, इसलिए
$ f(x) = (x-1)^2 + 2 \geq 2$; और $ f(1) = 0 + 2 = 2$, तो मूल्य $2$ है
वास्तव में पहुंच गया. दोनों तथ्य मिलकर यह सिद्ध करते हैं $2$ है \omterm{def:g10:functions:extrema}{न्यूनतम}.
\end{example}

$ y = 2$, और इसे छूता है $ x = 1$: द \omterm{def:g10:functions:extrema}{न्यूनतम} का $ f $ है $2$.}
\end{omfigure}

\section{अभ्यास}

\begin{exercise}[$\star $]\label{exo:g10:functions:1}
होने देना $ f(x) = 2x^2 - 3x + 1$. की गणना करें इमेजिस का $0$, $2$, $-1$ और
$\frac12$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g10:functions:2}
दे दो \omterm{def:g10:functions:function}{कार्यक्षेत्र} प्रत्येक का \omterm{def:g10:functions:function}{समारोह}:
\[
f(x) = 5x - 2, \qquad
g(x) = \frac{3}{x + 4}, \qquad
h(x) = \sqrt{2x - 6}, \qquad
k(x) = \frac{1}{x^2 + 1}.
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g10:functions:3}
होने देना $ f(x) = x^2 - 2x $. सभी खोजें पूर्वछवियाँ का $0$, का $3$, और का $-1$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g10:functions:4}
क्या बात है $ A(-1, 4)$ के हैं \omterm{def:g10:functions:graph}{ग्राफ} का $ f(x) = x^2 - 2x + 1$?
और बात $ B(2, 1)$? गणना द्वारा औचित्य सिद्ध करें।
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g10:functions:5}
ए \omterm{def:g10:functions:function}{समारोह} $ g $ पर परिभाषित $\intcc{-3}{5}$ के पास है \omterm{def:g10:functions:table}{विविधता तालिका}
\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{c|ccccccc}
$ x $ & $-3$ & & $0$ & & $3$ & & $5$ \\
\hline
$ g $ & $1$ & $\searrow $ & $-2$ & $\nearrow $ & $4$ & $\searrow $ & $0$ \\
\end{tabular}
\end{center}
\begin{enumerate}
  \item क्या है \omterm{def:g10:functions:extrema}{अधिकतम} और यह \omterm{def:g10:functions:extrema}{न्यूनतम} का $ g $ पर $\intcc{-3}{5}$?
  \item तुलना करना $ g(-1)$ और $ g(-0.5)$ उनके मूल्यों को जाने बिना।
  \item कितने समाधान करता है \omterm{def:g10:algebra:equation}{समीकरण} $ g(x) = 0$ अधिक से अधिक चालू रखें
        प्रत्येक \omterm{def:g10:numbers:interval}{अंतराल} तालिका के?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:functions:6}
होने देना $ f(x) = -2x + 7$. तुलना करके दिखाओ $ f(u)$ और $ f(v)$ के लिए $ u < v $,
वह $ f $ सख्ती से है \omterm{def:g10:functions:variations}{घटते} पर $\R $.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:functions:7}
दिखाओ कि \omterm{def:g10:functions:function}{समारोह} $ f(x) = x^2 + 6x + 11$ एक \omterm{def:g10:functions:extrema}{न्यूनतम} पर $\R $, और
इसका मूल्य बताएं और यह कहां प्राप्त होता है। (संकेत: लिखें
$ f(x) = (x + 3)^2 + c $ सही स्थिरांक के लिए $ c $.)
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:functions:8}
एक आयताकार बगीचे का परिमाप है $40$ एम। होने देना $ x $ इसकी चौड़ाई हो, में
मीटर.
\begin{enumerate}
  \item लंबाई व्यक्त करें, फिर क्षेत्रफल $ A(x)$, जैसा कार्य का $ x $.
        जिसके लिए $ x $ क्या इसका कोई मतलब है?
  \item गणना करना $ A(4)$, $ A(8)$, $ A(10)$, $ A(12)$ और $ A(16)$. आप क्या करते हैं?
        सबसे बड़े क्षेत्रफल के आकार के बारे में अनुमान?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:functions:9}
होने देना $ f(x) = \dfrac{1}{x^2 + 1}$, पर परिभाषित किया गया $\R $.
\begin{enumerate}
  \item बताते हैं कि $0 < f(x) \leq 1$ हर वास्तविक के लिए $ x $.
  \item करता है $ f $ लीजिये \omterm{def:g10:functions:extrema}{अधिकतम} पर $\R $? ए \omterm{def:g10:functions:extrema}{न्यूनतम}? औचित्य।
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g10:functions:10}
होने देना $ f(x) = x^2$ पर $\R $.
\begin{enumerate}
  \item होने देना $0 \leq u < v $. कारक $ f(v) - f(u)$ और उसका निष्कर्ष निकालो $ f $ है
        सख्ती से \omterm{def:g10:functions:variations}{की बढ़ती} पर $\intco{0}{+\infty}$.
  \item यह दिखाने के लिए तर्क अपनाएँ $ f $ सख्ती से है \omterm{def:g10:functions:variations}{घटते} पर
        $\intoc{-\infty}{0}$.
\end{enumerate}
\end{exercise}

\section{समस्या: मशीनें अपना आउटपुट स्वयं देती हैं}

\begin{problem}[{सप्ताहांत समस्या --- निश्चित बिंदु, पुनरावृत्ति, और तीन प्रसिद्ध मशीनें: हेरॉन इम्प्रूवर, स्क्वैयर, और द अनसुलझे ओलावृष्टि}]
\label{pb:g10:functions:1}
ए \omterm{def:g10:functions:function}{समारोह} एक मशीन है: एक नंबर अंदर जाता है, एक नंबर बाहर आता है।
सबसे दिलचस्प प्रयोग मशीन को खिलाते हैं \emph{इसका अपना
आउटपुट}, बार-बार --- और फिर दो प्रश्न हावी हो जाते हैं:
क्या प्रक्रिया कहीं स्थिर हो जाती है (a \emph{\omterm{pb:g10:functions:1}{नियत बिन्दु}}), और
मशीन की विविधता इसे वहां कैसे ले जाती है? इसमें से एक
समस्या की मशीनें दो के लिए वर्गमूलों को पॉलिश कर रही हैं
हजार वर्ष; दूसरा सबसे प्रसिद्ध अनसुलझी समस्या की रखवाली करता है
प्रारंभिक गणित की पेशकश करनी होगी।

\medskip
\noindent\textbf{Part I --- Heron's machine.}
पर विचार करें \omterm{def:g10:functions:function}{समारोह}
\[
f(x) = \frac12\left(x + \frac2x\right).
\]
\begin{enumerate}
  \item दे दो \omterm{def:g10:functions:function}{कार्यक्षेत्र} का $ f $, और गणना करें $ f(1)$, $ f(2)$ और
        $ f\!\left(\frac32\right)$ सटीक भिन्नों के रूप में.
  \item सभी खोजें पूर्वछवियाँ का $\frac32$: हल करना
        $ f(x) = \frac32$ (हर को साफ़ करें और का उपयोग करें
        के तरीके \cref{pb:g10:algebra:1}).
  \item ए \emph{नियत बिन्दु}\index{नियत बिन्दु} का $ f $ एक है
        मशीन द्वारा संख्या अपरिवर्तित: $ f(x) = x $. दोनों को खोजें
        निश्चित बिंदु का $ f $. कौन सा पुराना परिचित पहरा देता है
        सकारात्मक?
  \item से शुरू $ x = 1$, मशीन को अपना आउटपुट फ़ीड करें
        तीन बार और सटीक परिणाम रिकॉर्ड करें। कौन सा क्रम
        मिडिल स्कूल वॉल्यूम की अतार्किकता सप्ताहांत समस्या का आपने पुनर्निर्माण किया है --- और क्या था
        ``बगुला का नुस्खा'', की भाषा में कार्य?
  \item एक चित्र पर, स्केच करें \omterm{def:g10:functions:graph}{ग्राफ} का $ f $ (के लिए $ x > 0$)
        और विकर्ण रेखा $ y = x $. कहाँ हैं निश्चित
        अंक इस चित्र पर? (तुलना करें
        मिडिल स्कूल वॉल्यूम की दो-थर्मामीटर सप्ताहांत समस्या: तापमान समान पढ़ रहा है
        दोनों पैमानों में एक ही विचार था।)
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Why the machine cannot miss.}
\begin{enumerate}[resume]
  \item के लिए $0 < u < v $, सामान्य में कमी करके दिखाएँ
        भाजक वह
        \[
        f(v) - f(u) = \frac{(v - u)(uv - 2)}{2uv} ,
        \]
        और की विविधताएँ निकालें $ f $ पर
        $\intoo{0}{+\infty}$: \omterm{def:g10:functions:variations}{घटते} फिर एक बिंदु तक
        \omterm{def:g10:functions:variations}{की बढ़ती} --- कौन सा बिंदु? (उसी तकनीक के समान
        \cref{exo:g10:functions:10}.)
  \item उस पर निष्कर्ष निकालें $\intoo{0}{+\infty}$  \omterm{def:g10:functions:function}{समारोह} $ f $ एक \omterm{def:g10:functions:extrema}{न्यूनतम}, बिल्कुल लायक $\sqrt2$, पर प्राप्त किया गया
        $\sqrt2$. ये किस बारे में कहता है \emph{प्रत्येक} आउटपुट
        मशीन का (सकारात्मक इनपुट)?
  \item दिखाओ अगर $ x > \sqrt2$ तब
        $\sqrt2 < f(x) < x $: बहुत बड़ा अनुमान हमेशा सुधारा जाता है,
        कभी भी अति न करें. (के लिए $ f(x) < x $, तुलना करना $\frac2x $ साथ
        $ x $.)
  \item प्रश्न 6-8 को एकत्रित करें \omterm{def:g10:functions:table}{विविधता तालिका} का $ f $
        पर $\intoo{0}{+\infty}$
        (\cref{def:g10:functions:table}), \omterm{def:g10:functions:extrema}{न्यूनतम} शामिल.
  \item दूसरी मशीन: $ g(x) = x^2$. इसका निश्चित बिंदु हैं
        $0$ और $1$ (जाँच करना)। दोहराएं $ g $ से चार बार
        $ x = 0.9$, फिर से चार बार $ x = 1.1$ (कैलकुलेटर,
        तीन दशमलव)। एक \omterm{pb:g10:functions:1}{नियत बिन्दु} आकर्षित करता है, दूसरा
        प्रतिकारक: कौन सा है?
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The hailstone machine.}
पूर्ण संख्याओं पर, परिभाषित करें: $ h(n) = \frac n2$ अगर $ n $ सम है,
और $ h(n) = 3n + 1$ अगर $ n $ अजीब है. संख्याएँ नीचे उछलती हैं $ h $ जैसे
तूफ़ानी बादल में ओले।
\begin{enumerate}[resume]
  \item की पूरी उड़ान की गणना करें $7$ अंतर्गत $ h $, नीचे तक $1$.
        यह कितने कदम चलता है और कितनी ऊंचाई तक उड़ता है?
  \item की उड़ान शुरू करें $27$ और दस चरणों की गणना करें। (इसके
        पूरी उड़ान चलती है $111$ चरण और शिखर $9\,232$ ---
        के एक प्रारंभिक बिंदु से $27$.) सरल के बारे में क्या पाठ
        नियम करते हैं $7$ और $27$ पढ़ाना?
  \item बताएं कि कोई भी उड़ान किस शक्ति तक पहुंचती है $2$
        सीधे दुर्घटनाग्रस्त हो जाता है $1$, और क्या होता है $1$
        (गणना करें $ h(1)$, $ h(4)$, $ h(2)$). करता है $ h $ कोई भी है क्या
        \omterm{pb:g10:functions:1}{नियत बिन्दु} धनात्मक पूर्ण संख्याओं में से? क्या करता है
        इसके बजाय प्रत्येक देखे गए उड़ान प्रपत्र का अंत?
  \item  \emph{कोलाट्ज़ अनुमान} हर शुरुआत का दावा करता है
        संख्या अंततः पहुंचती है $1$. कंप्यूटर ने सत्यापन कर लिया है
        यह बहुत आगे है $10^{20}$; किसी भी इंसान ने इसे साबित नहीं किया है. क्या
        दो प्रकार की खोज होगी \emph{असत्य सिद्ध करना} यह? और क्यों
        बड़े पैमाने पर सत्यापन के बाद भी मामला बंद नहीं होता है
        (मिडिल स्कूल वॉल्यूम की माचिस-ओरेकल सप्ताहांत समस्या ने इस घंटी को बजाया)?
  \item ``गणित अभी ऐसे प्रश्नों के लिए तैयार नहीं है,'' कहा
        पॉल एर्ड\H{o}इस अनुमान का एस. एक वाक्य में: क्या
        हेलस्टोन मशीन को हेरॉन से अलग करता है, जहां
        प्रश्न 6-8 सब कुछ तय हो गया?
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The machine user's manual.}
\begin{enumerate}[resume]
  \item डोमेन जैसा \omterm{def:g10:numbers:interval}{अंतराल} (\cref{pb:g10:numbers:1}का
        भाषा): दे दो डोमेन का
        $ x \mapsto \sqrt{x - 3}$, \
        $ x \mapsto \frac{1}{x^2 - 9}$, \
        $ x \mapsto \sqrt{9 - x^2}$.
  \item हेरोन्स को लौटें $ f $: सटीक खोजें पूर्वछवियाँ का $3$
        (हल करना $ f(x) = 3$ वर्ग पूरा करके)।
  \item एक पत्थर ऊपर की ओर फेंका जाता है: उसकी ऊंचाई होती है
        $ H(t) = 20t - 5t^2$ मीटर के बाद $ t $ सेकंड. जिस पर
        समय \omterm{def:g10:numbers:interval}{अंतराल} क्या सूत्र का भौतिक अर्थ निकलता है?
        \omterm{def:g10:functions:extrema}{अधिकतम} ऊंचाई और उसकी ऊंचाई ज्ञात करने के लिए वर्ग को पूरा करें
        समय (\cref{def:g10:functions:extrema}), और दे
        \omterm{def:g10:functions:table}{विविधता तालिका}.
  \item पत्थर कम से कम कितने समय का है $15$ मैं ऊँचा हूँ? (हल करना
        $ H(t) \geq 15$ के साथ \omterm{def:g10:algebra:signtable}{साइन टेबल},
        \cref{met:g10:algebra:signtable}.)
  \item समापन: इस समस्या से चार मशीनें चलीं --- हेरॉन की
        सुधारक, चौकोर, ओला, पत्थर
        ऊंचाई. दो या तीन वाक्यों में बताएं कि प्रत्येक क्या है
        \omterm{def:g10:functions:function}{समारोह} एक \omterm{def:g10:functions:function}{कार्यक्षेत्र}, एक नियम, ए \omterm{def:g10:functions:graph}{ग्राफ}) और दो
        इस समस्या ने प्रत्येक से महान प्रश्न पूछे (कहाँ करता है)।
        पुनरावृत्ति व्यवस्थित? कैसे करता है \omterm{def:g10:functions:function}{समारोह} भिन्न?) ---
        अगले अध्यायों में एक-एक करके प्रश्न पूछे जाते हैं।
\end{enumerate}
\end{problem}
