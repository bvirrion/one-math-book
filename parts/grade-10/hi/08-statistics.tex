\chapter{वर्णनात्मक आँकड़े}\label{ch:g10:stats}

सांख्यिकी संख्याओं के एक बड़े संग्रह का सारांश प्रस्तुत करती है --- ग्रेड, वेतन,
तापमान --- कुछ अच्छी तरह से चुने गए संकेतकों द्वारा। यह अध्याय परिचय देता है
संकेतकों के दो परिवार: वे जो पता लगा रहे हैं \emph{केंद्र} का
डेटा (\omterm{def:g10:stats:mean}{अर्थ}, \omterm{def:g10:stats:median}{MEDIAN}) और इसे मापने वाले \emph{फैलाना} (\omterm{def:g10:stats:quartiles}{श्रेणी},
चतुर्थक). मानक विचलन सहित एक गहरा उपचार आता है
\cref{ch:g11:stat}.

\section{डेटा और आवृत्तियाँ}

\begin{definition}[सांख्यिकीय श्रृंखला]\label{def:g10:stats:series}
ए \emph{सांख्यिकीय श्रृंखला}\index{सांख्यिकीय श्रृंखला} मानों की एक सूची है
$x_1, x_2, \dots, x_N $ आकार की जनसंख्या पर देखा गया $ N $. जब एक मान
$ x $ प्रकट होता है $ n $ समय, $ n $ इसका है \emph{गिनती करना} और भागफल
\[
f = \frac{n}{N}
\]
इसका है \emph{आवृत्ति}\index{आवृत्ति} (अक्सर एक के रूप में व्यक्त किया जाता है
प्रतिशत). सभी मानों की आवृत्तियों का योग बनता है $1$.
\end{definition}

\begin{example}\label{ex:g10:stats:frequencies}
के ग्रेड $20$ एक पर छात्र $5$-बिंदु प्रश्नोत्तरी:
\begin{center}
\renewcommand{\arraystretch}{1.2}
\begin{tabular}{c|cccccc}
grade $ x $ & $0$ & $1$ & $2$ & $3$ & $4$ & $5$ \\
\hline
count $ n $ & $1$ & $2$ & $4$ & $6$ & $5$ & $2$ \\
\omterm{def:g10:stats:series}{आवृत्ति} $ f $ & $0.05$ & $0.10$ & $0.20$ & $0.30$ & $0.25$ & $0.10$ \\
\end{tabular}
\end{center}
जाँच करना: $1 + 2 + 4 + 6 + 5 + 2 = 20$ और आवृत्तियों का योग है $1$.
\end{example}

\section{केंद्र के उपाय}

\begin{definition}[अर्थ]\label{def:g10:stats:mean}
 \emph{अर्थ}\index{अर्थ} श्रृंखला का है
\[
\bar x = \frac{x_1 + x_2 + \dots + x_N}{N}.
\]
यदि मान $ v_1, \dots, v_k $ गिनती के साथ प्रकट हों $ n_1, \dots, n_k $, द
उसी संख्या की गणना की जाती है \emph{भारित माध्य}
\[
\bar x = \frac{n_1 v_1 + n_2 v_2 + \dots + n_k v_k}{n_1 + n_2 + \dots + n_k}.
\]
\end{definition}

\begin{example}\label{ex:g10:stats:mean}
की प्रश्नोत्तरी के लिए \cref{ex:g10:stats:frequencies}:
\[
\bar x = \frac{1 \times 0 + 2 \times 1 + 4 \times 2 + 6 \times 3
+ 5 \times 4 + 2 \times 5}{20}
= \frac{0 + 2 + 8 + 18 + 20 + 10}{20}
= \frac{58}{20} = 2.9 .
\]
\end{example}

\begin{definition}[मंझला]\label{def:g10:stats:median}
 \emph{MEDIAN}\index{MEDIAN} एक मान है जो विभाजित करता है \emph{क्रमबद्ध}
दो हिस्सों में श्रृंखला: कम से कम आधे मान हैं $\leq $ माध्यिका, और
कम से कम आधे हैं $\geq $ यह। व्यवहार में, क्रमबद्ध करें $ N $ मूल्य;
\begin{itemize}
  \item अगर $ N $ विषम है, माध्य \omterm{prop:g10:coordgeom:midpoint}{मध्य} मान है, स्थिति में
        $\frac{N+1}{2}$;
  \item अगर $ N $ सम है, ले लो \omterm{prop:g10:coordgeom:midpoint}{मध्य} दो \omterm{prop:g10:coordgeom:midpoint}{मध्य} मानों में से, में
        पद $\frac N2$ और $\frac N2 + 1$.
\end{itemize}
\end{definition}

\begin{example}\label{ex:g10:stats:median}
की क्रमबद्ध श्रृंखला $7$ घर की कीमतें (हजारों में): $120$, $150$, $160$,
$180$, $210$, $240$, $900$. \omterm{def:g10:stats:median}{MEDIAN} है $4$ वें मान, $180$. \omterm{def:g10:stats:mean}{अर्थ}
है $\frac{1960}{7} = 280$ --- की तुलना में बड़ा है $6$ की $7$ कीमतें!
एक चरम मूल्य ($900$) खींचता है \omterm{def:g10:stats:mean}{अर्थ} से कहीं अधिक \omterm{def:g10:stats:median}{MEDIAN}: द
\omterm{def:g10:stats:median}{MEDIAN} है \emph{मज़बूत}, द \omterm{def:g10:stats:mean}{अर्थ} क्या नहीं है।
\end{example}

\section{प्रसार के उपाय}

\begin{definition}[रेंज, चतुर्थक]\label{def:g10:stats:quartiles}
क्रमबद्ध श्रृंखला के लिए:
\begin{itemize}
  \item  \emph{श्रेणी}\index{श्रेणी} के बीच का अंतर है
        सबसे बड़े और सबसे छोटे मूल्य;
  \item  \emph{प्रथम चतुर्थक}\index{प्रथम चतुर्थक} $ Q_1$ सबसे छोटा है
        मूल्य ऐसा कि कम से कम एक चौथाई ($25\%$) के मान हैं
        $\leq Q_1$; \emph{तृतीय चतुर्थक} $ Q_3$ सबसे छोटा मान है
        ऐसा कि कम से कम तीन चौथाई ($75\%$) के मान हैं
        $\leq Q_3$;
  \item  \emph{अंतरचतुर्थक सीमा}\index{अंतरचतुर्थक सीमा} $ Q_3 - Q_1$ के प्रसार को मापता है
        डेटा का \omterm{prop:g10:coordgeom:midpoint}{मध्य} भाग.
\end{itemize}
\end{definition}

\begin{method}[चतुर्थक ढूँढना]\label{met:g10:stats:quartiles}
की एक श्रृंखला के लिए $ N $ क्रमबद्ध मान:
\begin{enumerate}
  \item गणना $\frac N4$; यदि यह एक नहीं है पूर्णांक इसे गोल करो \emph{ऊपर} अगले इसपर पूर्णांक; $ Q_1$ उस स्थिति में मूल्य है;
  \item गणना $\frac{3N}{4}$; यदि आवश्यक हो तो गोल करें; $ Q_3$ में मूल्य है
        वह स्थिति.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g10:stats:quartiles}
ले लो $11$ क्रमबद्ध मान
\[
2,\ 3,\ 5,\ 5,\ 6,\ 7,\ 8,\ 9,\ 9,\ 10,\ 12 .
\]
मंझला: पद $\frac{11+1}{2} = 6$, इतना \omterm{def:g10:stats:median}{MEDIAN} है $7$.
$ Q_1$: $\frac{11}{4} = 2.75$, तक पूर्णांकित करें $3$; $3$ तीसरा मान है $5$.
$ Q_3$: $\frac{33}{4} = 8.25$, तक पूर्णांकित करें $9$; $9$ वां मान है $9$.
\omterm{def:g10:stats:quartiles}{श्रेणी}: $12 - 2 = 10$; \omterm{def:g10:stats:quartiles}{अंतरचतुर्थक सीमा}: $9 - 5 = 4$.
\end{example}

\begin{example}[दो समूहों की तुलना करना]\label{ex:g10:stats:comparison}
दो कक्षाएँ एक ही परीक्षा देती हैं (स्कोर बाहर)। $20$):
\begin{center}
\renewcommand{\arraystretch}{1.2}
\begin{tabular}{c|ccc}
 & \omterm{def:g10:stats:median}{MEDIAN} & $ Q_1$ & $ Q_3$ \\
\hline
class A & $12$ & $10$ & $14$ \\
class B & $12$ & $6$ & $17$ \\
\end{tabular}
\end{center}
वही \omterm{def:g10:stats:median}{MEDIAN}, बहुत भिन्न फैलाव: वर्ग ए सजातीय है (उसका आधा)।
में स्कोर $\intcc{10}{14}$), वर्ग बी विषमांगी है (इसका केंद्रीय भाग
फैला हुआ $\intcc{6}{17}$). अकेले केंद्र के संकेतक कभी भी संपूर्ण नहीं बताते
कहानी.
\end{example}

\begin{remark}
न ही \omterm{def:g10:stats:mean}{अर्थ} न ही \omterm{def:g10:stats:median}{MEDIAN} ``बेहतर'' है: \omterm{def:g10:stats:mean}{अर्थ} प्रत्येक मान का उपयोग करता है
(और कुल की गणना करने के लिए आवश्यक है), द \omterm{def:g10:stats:median}{MEDIAN} अतिवादी मूल्यों का विरोध करता है। ए
किसी श्रृंखला का गंभीर सारांश केंद्र का कम से कम एक संकेतक देता है
\emph{और} प्रसार में से एक.
\end{remark}

\section{अभ्यास}

\begin{exercise}[$\star $]\label{exo:g10:stats:1}
यहां एक वर्ष में पढ़ी गई पुस्तकों की संख्या दी गई है $15$ छात्र:
\[
0,\ 1,\ 1,\ 2,\ 2,\ 2,\ 3,\ 3,\ 4,\ 4,\ 5,\ 6,\ 7,\ 8,\ 12 .
\]
की गणना करें \omterm{def:g10:stats:mean}{अर्थ}, द \omterm{def:g10:stats:median}{MEDIAN} और यह \omterm{def:g10:stats:quartiles}{श्रेणी} इस श्रृंखला के.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g10:stats:2}
एक पासा फेंका गया $50$ समय:
\begin{center}
\renewcommand{\arraystretch}{1.2}
\begin{tabular}{c|cccccc}
outcome & $1$ & $2$ & $3$ & $4$ & $5$ & $6$ \\
\hline
count & $7$ & $9$ & $8$ & $10$ & $6$ & $10$ \\
\end{tabular}
\end{center}
की गणना करें \omterm{def:g10:stats:series}{आवृत्ति} प्रत्येक परिणाम का और \omterm{def:g10:stats:mean}{अर्थ} नतीजा।
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g10:stats:3}
खोजें \omterm{def:g10:stats:median}{MEDIAN}, $ Q_1$ और $ Q_3$ क्रमबद्ध श्रृंखला का
\[
1,\ 2,\ 4,\ 4,\ 5,\ 6,\ 6,\ 7,\ 8,\ 9,\ 9,\ 11 \qquad (N = 12),
\]
और संबंधित बॉक्स प्लॉट बनाएं।
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g10:stats:4}
 \omterm{def:g10:stats:mean}{अर्थ} का $24$ परीक्षण स्कोर है $11.5$. ए $25$ वें छात्र परीक्षा देता है
और स्कोर $19$. नया क्या है \omterm{def:g10:stats:mean}{अर्थ} कक्षा का?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:stats:5}
एक छात्र के पास है \omterm{def:g10:stats:mean}{अर्थ} का $12$ बाद $5$ परीक्षण, सभी एक ही वजन के साथ।
पर क्या स्कोर है $6$ वां परीक्षण बढ़ाएगा \omterm{def:g10:stats:mean}{अर्थ} को $13$? एक है \omterm{def:g10:stats:mean}{अर्थ} का
$14$ पहुंच योग्य (स्कोर बाहर हैं $20$)?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:stats:6}
की एक शृंखला का आविष्कार करें $7$ गैर-नकारात्मक मान जिसका \omterm{def:g10:stats:median}{MEDIAN} है $10$ और किसका
\omterm{def:g10:stats:mean}{अर्थ} से बड़ा है $20$; फिर एक किसका \omterm{def:g10:stats:median}{MEDIAN} है $10$ और किसका \omterm{def:g10:stats:mean}{अर्थ} की तुलना में छोटा है $6$. क्यों कर सकते हैं \omterm{def:g10:stats:mean}{अर्थ} ऐसी शृंखला कभी भी छोटी नहीं होगी
$\frac{40}{7}$? ये उदाहरण क्या दर्शाते हैं?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:stats:7}
एक कंपनी में, $9$ कर्मचारी कमाते हैं $1800$ प्रत्येक प्रति माह और
निर्देशक कमाता है $12000$.
\begin{enumerate}
  \item की गणना करें \omterm{def:g10:stats:mean}{अर्थ} और यह \omterm{def:g10:stats:median}{MEDIAN} वेतन।
  \item यहां कौन सा संकेतक ``सामान्य'' वेतन का सबसे अच्छा वर्णन करता है? क्यों?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:stats:8}
का एक वर्ग $12$ लड़कों के पास है \omterm{def:g10:stats:mean}{अर्थ} की ऊंचाई $168$ सेमी; $18$ की लड़कियाँ
एक ही कक्षा में एक है \omterm{def:g10:stats:mean}{अर्थ} की ऊंचाई $163$ सेमी। की गणना करें \omterm{def:g10:stats:mean}{अर्थ} की ऊंचाई
पूरी कक्षा. (सावधान: यह है \emph{नहीं}  \omterm{prop:g10:coordgeom:midpoint}{मध्य} का $168$ और $163$
--- समूह के आकार के आधार पर साधनों का वजन करें।)
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g10:stats:9}
की एक श्रृंखला $ N $ मान $ x_1, \dots, x_N $ है \omterm{def:g10:stats:mean}{अर्थ} $\bar x $. प्रत्येक मान है
में तब्दील हो गया $ y_i = a x_i + b $ निश्चित संख्या के लिए $ a $ और $ b $.
\begin{enumerate}
  \item दिखाओ कि \omterm{def:g10:stats:mean}{अर्थ} नई सीरीज का है $ a \bar x + b $.
  \item एक सप्ताह का तापमान, डिग्री सेल्सियस में, है \omterm{def:g10:stats:mean}{अर्थ} $20$.
        उनका क्या है \omterm{def:g10:stats:mean}{अर्थ} डिग्री फ़ारेनहाइट में
        ($ F = 1.8\,C + 32$)?
\end{enumerate}
\end{exercise}

\section{समस्या: औसत व्यक्ति का अस्तित्व नहीं है}

\begin{problem}[{सप्ताहांत समस्या --- मध्यिका के विरुद्ध माध्य: आउटलेर्स, ग्रेडिंग कर्व्स, सड़क पर गोदाम, और कॉकपिट डिज़ाइन किया गया किसी के लिए नहीं}]
\label{pb:g10:stats:1}
1950 में अमेरिकी वायु सेना ने माप लिया $4\,000$ दस शरीर पर पायलट
आयाम और के लिए कॉकपिट का निर्माण किया \emph{औसत आदमी}. यह
फिट --- गिनती के द्वारा खोजे गए एक लेफ्टिनेंट के रूप में --- अनिवार्य रूप से
कोई नहीं. आँकड़े एकल संख्याओं और प्रत्येक के साथ भीड़ का सारांश प्रस्तुत करते हैं
सारांश किसी चीज़ के बारे में सच बताता है और किसी चीज़ के बारे में झूठ
अन्यथा. यह समस्या लेती है \omterm{def:g10:stats:mean}{अर्थ} और यह \omterm{def:g10:stats:median}{MEDIAN}
(\cref{def:g10:stats:mean}, \cref{def:g10:stats:median}) अदालत में,
प्रत्येक को यह दिखाने देता है कि वह किसमें सर्वश्रेष्ठ है, और कॉकपिट केस को बंद कर देता है।

\medskip
\noindent\textbf{Part I --- Two summaries on trial.}
\begin{enumerate}
  \item एक सड़क पर सात घर बिकते हैं, हजारों यूरो में:
        $200$, $210$, $220$, $230$, $240$, $250$ --- और एक
        हवेली पर $1\,200$. की गणना करें \omterm{def:g10:stats:mean}{अर्थ} और यह \omterm{def:g10:stats:median}{MEDIAN}
        कीमत। कौन सी संख्या ``इस गली में एक घर'' का वर्णन करती है?
  \item हवेली हटाओ और दोनों का पुनः हिसाब करो। कितना किया
        प्रत्येक सारांश चाल? आउटलेर्स के बारे में नैतिक बातें बताएं।
  \item एक जनगणना तालिका: से बाहर $100$ परिवार, $20$ नहीं है
        बच्चा, $25$ एक लो, $30$ दो हैं, $15$ तीन हैं,
        $10$ चार हैं. की गणना करें \omterm{def:g10:stats:mean}{अर्थ} बच्चों की संख्या. नहीं
        परिवार ``है $1.7$ बच्चे'' --- तो वास्तविक मात्रा क्या है
        करता है \omterm{def:g10:stats:mean}{अर्थ} सांकेतिक शब्दों में बदलना? (इससे गुणा करें $100$.)
  \item पन्द्रह अंक के लिए
        $4, 6, 7, 8, 9, 10, 10, 11, 12, 12, 13, 14, 15, 17, 19$:
        खोजें \omterm{def:g10:stats:median}{MEDIAN} और यह चतुर्थक $ Q_1$, $ Q_3$
        (\cref{met:g10:stats:quartiles}).
  \item उसी शृंखला के लिए, दें \omterm{def:g10:stats:quartiles}{श्रेणी} और यह
        \omterm{def:g10:stats:quartiles}{अंतरचतुर्थक सीमा}. प्रत्येक क्या मापता है?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Grading curves and merged classes.}
\begin{enumerate}[resume]
  \item से \cref{exo:g10:stats:9}: प्रत्येक मान को इसके द्वारा स्थानांतरित करना
        $+b $ परिवर्तन \omterm{def:g10:stats:mean}{अर्थ}, \omterm{def:g10:stats:median}{MEDIAN} और चतुर्थक द्वारा $+b $; स्केलिंग
        द्वारा $ a $ उन्हें मापता है $ a $. क्या होता है \omterm{def:g10:stats:quartiles}{श्रेणी} और यह \omterm{def:g10:stats:quartiles}{अंतरचतुर्थक सीमा} एक बदलाव के तहत? स्केलिंग के तहत?
        औचित्य।
  \item एक परीक्षण हुआ है \omterm{def:g10:stats:mean}{अर्थ} $9$ (बाहर) $20$) और शिक्षक चाहता है
        \omterm{def:g10:stats:mean}{अर्थ} $12$. दो वक्र प्रस्तावित हैं: \omterm{def:g10:vectors:sum}{जोड़}ें $3$ की ओर इशारा करता है
        प्रत्येक, या प्रत्येक चिह्न को गुणा करें $\frac43$. गणना करें
        एक छात्र का भाग्य $6$ और एक पर $18$ प्रत्येक के अंतर्गत
        वक्र. कौन सा वक्र किसके पक्ष में है --- और कौन सा वक्र किसके पक्ष में है
        छत की समस्या?
  \item एक कक्षा ($20$ विद्यार्थियों) के पास है \omterm{def:g10:stats:mean}{अर्थ} $12$; कक्षा बी ($30$
        विद्यार्थियों) के पास है \omterm{def:g10:stats:mean}{अर्थ} $10$. की गणना करें \omterm{def:g10:stats:mean}{अर्थ} की $50$
        छात्र मिलकर बताएं कि ऐसा क्यों नहीं है $11$
        (भारित-औसत पाठ अनुकूल माध्य मिडिल स्कूल वॉल्यूम की सप्ताहांत समस्या, में
        कक्षा)
  \item बताते हैं कि माध्यिकाओं इस गेम को मना करें: गणना करें माध्यिकाओं
        का $ A = \{1, 2, 3\}$ और $ B = \{0, 10\}$, फिर
        \omterm{def:g10:stats:median}{MEDIAN} मर्ज की गई सूची का, और तुलना करें
        दोनों का आकार-भारित औसत माध्यिकाओं.
  \item फिगर स्केटिंग: पांच जजों का स्कोर $5.2$, $5.3$, $5.4$,
        $5.5$, $9.9$. की गणना करें \omterm{def:g10:stats:mean}{अर्थ}, द \omterm{def:g10:stats:median}{MEDIAN}, और यह
        \emph{छंटनी \omterm{def:g10:stats:mean}{अर्थ}} (\omterm{def:g10:functions:extrema}{न्यूनतम} और उच्चतम को गिराएं, औसत
        बाकी)। अधिकांश निर्णायक खेल ट्रिम क्यों करते हैं?
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Reading distributions.}
\begin{enumerate}[resume]
  \item दो कारखाने दोनों बोल्ट का उत्पादन करते हैं \omterm{def:g10:stats:mean}{अर्थ} लंबाई
        $50$ मिमी. फ़ैक्टरी ए: $ Q_1 = 49.9$, $ Q_3 = 50.1$; फ़ैक्टरी
        बी: $ Q_1 = 48$, $ Q_3 = 52$. आपको विनिमेय की आवश्यकता है
        बोल्ट: कौन सा कारखाना, और कौन सा आँकड़ा तय किया गया?
  \item एक क्लिनिक में, \omterm{def:g10:stats:median}{MEDIAN} इंतज़ार है $10$ मिनट लेकिन
        \omterm{def:g10:stats:mean}{अर्थ} इंतज़ार है $25$. यह \omterm{def:g10:vectors:sum}{जोड़}ी किस आकार का डेटा बनाती है
        विश्वासघात? बिल्कुल इसी के साथ पांच-मूल्य वाला डेटासेट बनाएं
        \omterm{def:g10:stats:median}{MEDIAN} और \omterm{def:g10:stats:mean}{अर्थ}.
  \item ``आपका बच्चा ऊंचाई के लिए 75वें प्रतिशतक पर है'' ---
        वह वाक्य क्या कहता है \omterm{def:g10:stats:mean}{अर्थ}? और क्या हर बच्चा ऐसा कर सकता है?
        शहर ``औसत से ऊपर'' होगा? के ऊपर \omterm{def:g10:stats:median}{MEDIAN}? स्पष्ट करें
        अंतर.
  \item कक्षा ए: \omterm{def:g10:functions:extrema}{न्यूनतम} $4$, $ Q_1\,8$, \omterm{def:g10:stats:median}{MEDIAN} $11$, $ Q_3\,13$, \omterm{def:g10:functions:extrema}{अधिकतम}
        $17$. कक्षा बी: \omterm{def:g10:functions:extrema}{न्यूनतम} $2$, $ Q_1\,9$, \omterm{def:g10:stats:median}{MEDIAN} $11$, $ Q_3\,12$,
        \omterm{def:g10:functions:extrema}{अधिकतम} $19$. तुलना करें: विशिष्ट चिह्न, की एकरूपता
        \omterm{prop:g10:coordgeom:midpoint}{मध्य} भाग, चरम सीमा का जंगलीपन।
  \item पत्रकार की चेकलिस्ट: एक शीर्षक कहता है ``औसत।''
        टेककॉर्प में वेतन: $5\,000$ यूरो''। तीनो की सूची बनाओ
        वे प्रश्न जो इस समस्या ने आपको पहले पूछना सिखाया है
        विश्वास करें कि शीर्षक किसी विशिष्ट चीज़ के बारे में कुछ भी कहता है
        कर्मचारी.
\end{enumerate}

\noindent\textbf{Part IV --- The cockpit case.}
\begin{enumerate}[resume]
  \item अपने स्वयं के साक्ष्य बनाएँ: पाँच मकानों की कीमतें बनाएँ
        के साथ \omterm{def:g10:stats:mean}{अर्थ} $300$ और \omterm{def:g10:stats:median}{MEDIAN} $200$ (हजारों यूरो).
  \item  \omterm{def:g10:stats:mean}{अर्थ} संतुलन बिंदु है: परिभाषा से सिद्ध करें
        कि से विचलन \omterm{def:g10:stats:mean}{अर्थ} हमेशा रद्द करें,
        $\sum (x_i - \bar x) = 0$, और इसे अपने डेटासेट पर सत्यापित करें
        प्रश्न 16 का.
  \item  \omterm{def:g10:stats:median}{MEDIAN} कार्\omterm{def:g10:functions:extrema}{य-न्यूनतम} है: दुकानें बैठती हैं
        किलोमीटर $1$, $3$, $4$, $9$, $13$ एक सड़क का, और एक
        गोदाम को \omterm{def:g10:functions:extrema}{न्यूनतम} करना होगा \emph{कुल दूरी} को
        पांच दुकानें. एक गोदाम के लिए उस कुल की गणना करें
        \omterm{def:g10:stats:median}{MEDIAN} ($4$), पर \omterm{def:g10:stats:mean}{अर्थ} ($6$), और कम से $5$. कौन जीतता है?
        (यह सामान्य है: द \omterm{def:g10:stats:median}{MEDIAN} कुल निरपेक्षता को \omterm{def:g10:functions:extrema}{न्यूनतम} करता है
        दूरी --- किसी अन्य स्थान का परीक्षण करें।)
  \item कॉकपिट, हल हो गया: मान लीजिए कि एक पायलट के पास एक है $30\,\%$
        एक शरीर पर ``औसत के करीब'' होने का मौका
        आयाम, स्वतंत्र रूप से पार $10$ आयाम.
        औसत के करीब होने की संभावना का अनुमान लगाएं
        \emph{सभी दस} --- और परिणाम का मिलान किससे करें
        लेफ्टिनेंट मिला $4\,000$ पायलट. हवा ने क्या किया
        औसत कॉकपिट के बजाय बल निर्माण?
  \item समापन --- उपयोक्ता मार्गदर्शिका, प्रत्येक पंक्ति एक: जब कुल
        क्या मायने रखता है (एक बजट, एक फसल), इसका उपयोग करें \dots;
        जब सामान्य व्यक्तिगत मायने रखता है (वेतन, प्रतीक्षा)।
        टाइम्स), का उपयोग करें \dots; जब प्रसार की निष्पक्षता मायने रखती है,
        रिपोर्ट करें \dots; जब संदूषण का खतरा हो, तो इसका उपयोग करें
        \dots. समस्या के शीर्षक के साथ केस को बंद करें.
\end{enumerate}
\end{problem}
