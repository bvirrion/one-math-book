\chapter{Funções de referência}\label{ch:g10:reffunc}

Um punhado de coisas simples funções --- afim, quadrado, inverso, raiz quadrada,
cubo --- aparecem em todos os lugares, sozinhos ou combinados. Conhecendo seus \omterm{def:g10:functions:graph}{gráficos} e
variações de cor transformam muitos problemas em um esboço rápido. Este capítulo
estuda cada um deles por vez e comprova suas variações com o
técnica de comparação de \cref{ch:g10:functions}.

\section{Funções afins}

\begin{definition}[Função afim]\label{def:g10:reffunc:affine}
Um \emph{função afim}\index{função afim} é um \omterm{def:g10:functions:function}{função} da forma
\[
f(x) = mx + p,
\]
onde $m $ e $ p $ são fixos \omterm{def:g10:numbers:sets}{números reais}. Isso é \omterm{def:g10:functions:graph}{gráfico} é uma linha reta:
$ m $ é o \emph{recusar}\index{recusar} e $ p $ o
\emph{$ y $-interceptar}\index{$ y $-interceptar} (a linha cruza
o eixo vertical em $(0, p)$). Quando $ p = 0$ o \omterm{def:g10:functions:function}{função} é
\emph{linear}: $ f(x) = mx $ expressa proporcionalidade.
\end{definition}

\begin{proposition}[Inclinação e variações]\label{prop:g10:reffunc:affine}
Deixar $ f(x) = mx + p $.
\begin{enumerate}
  \item Para quaisquer duas entradas distintas $ u \neq v $:
        $ m = \dfrac{f(v) - f(u)}{v - u}$ (o declive é a mudança do
        saída por variação unitária da entrada).
  \item Se $ m > 0$, $ f $ é estritamente \omterm{def:g10:functions:variations}{aumentando} sobre $\R $; se $ m < 0$,
        estritamente diminuindo; se $ m = 0$, constante.
\end{enumerate}
\end{proposition}

\begin{proof}
1. Calcular $ f(v) - f(u) = (mv + p) - (mu + p) = m(v - u)$, então divida por
$ v - u \neq 0$.

2. Se $ u < v $, então $ f(v) - f(u) = m(v-u)$ tem o sinal de $ m $, desde
$ v - u > 0$: positivo $ m $ dá $ f(u) < f(v)$ (\omterm{def:g10:functions:variations}{aumentando}), negativo $ m $
dá $ f(u) > f(v)$ (diminuindo).
\end{proof}

\begin{example}\label{ex:g10:reffunc:affine}
Encontre o \omterm{def:g10:reffunc:affine}{função afim} cujo \omterm{def:g10:functions:graph}{gráfico} passa $ A(1, 3)$ e
$ B(4, 9)$. O declive é
\[
m = \frac{9 - 3}{4 - 1} = \frac{6}{3} = 2 .
\]
Então $ f(x) = 2x + p $, e $ f(1) = 3$ dá $2 + p = 3$, então $ p = 1$:
$ f(x) = 2x + 1$. Verifique com $ B $: $ f(4) = 9$.
\end{example}

\section{A função quadrada}

\begin{proposition}[Função quadrada]\label{prop:g10:reffunc:square}
O \omterm{def:g10:functions:function}{função} $ f(x) = x^2$, definido em $\R $:
\begin{enumerate}
  \item é estritamente diminuindo sobre $\intoc{-\infty}{0}$ e estritamente
        \omterm{def:g10:functions:variations}{aumentando} sobre $\intco{0}{+\infty}$, com \omterm{def:g10:functions:extrema}{mínimo} $0$ no $ x = 0$;
  \item satisfaz $ f(-x) = f(x)$ para todos $ x $: isso é \omterm{def:g10:functions:graph}{gráfico}, um
        \emph{parábola}\index{parábola}, é simétrico em relação à vertical
        eixo.
\end{enumerate}
\end{proposition}

\begin{proof}
1. Deixe $0 \leq u < v $. Então
\[
f(v) - f(u) = v^2 - u^2 = (v - u)(v + u) > 0,
\]
desde $ v - u > 0$ e $ v + u > 0$: $ f $ é estritamente \omterm{def:g10:functions:variations}{aumentando} sobre
$\intco{0}{+\infty}$. Se $ u < v \leq 0$, então $ v - u > 0$ mas
$ v + u < 0$, então $ f(v) - f(u) < 0$: estritamente diminuindo. Finalmente
$ x^2 \geq 0 = f(0)$ para todos $ x $.

2. $(-x)^2 = x^2$; então os pontos $(x, x^2)$ e $(-x, x^2)$, espelho imagens
ao longo do eixo vertical, estão ambos no \omterm{def:g10:functions:graph}{gráfico}.
\end{proof}

\begin{remark}[Quadrados e ordem]
Porque a praça \omterm{def:g10:functions:function}{função} diminui no lado negativo, tomando quadrados
\emph{inverte} a ordem dos números negativos: $-3 < -2$ mas $9 > 4$.
Nunca eleve ambos os lados de uma inequação ao quadrado sem verificar os sinais.
\end{remark}

\section{A função inversa}

\begin{proposition}[Função inversa]\label{prop:g10:reffunc:inverse}
O \omterm{def:g10:functions:function}{função} $ f(x) = \dfrac1x $, definido para $ x \neq 0$:
\begin{enumerate}
  \item é estritamente diminuindo sobre $\intoo{-\infty}{0}$ e estritamente
        diminuindo sobre $\intoo{0}{+\infty}$;
  \item satisfaz $ f(-x) = -f(x)$: isso é \omterm{def:g10:functions:graph}{gráfico}, um
        \emph{hipérbole}\index{hipérbole}, é simétrico em relação à origem.
\end{enumerate}
\end{proposition}

\begin{proof}
1. Deixe $0 < u < v $. Então
\[
f(v) - f(u) = \frac1v - \frac1u = \frac{u - v}{uv} < 0,
\]
desde $ u - v < 0$ e $ uv > 0$: estritamente diminuindo sobre
$\intoo{0}{+\infty}$. Sobre $\intoo{-\infty}{0}$ o mesmo quociente tem
$ u - v < 0$ e $ uv > 0$ novamente (produto de dois negativos), então $ f $
diminui lá também.

2. $\frac{1}{-x} = -\frac1x $.
\end{proof}

\begin{remark}
Cuidadoso: $\frac1x $ é \emph{não} diminuindo em sua totalidade domínio. De
$ u = -1$ para $ v = 1$ o valor salta de $-1$ até $1$. Os dois ramos
devem ser estudados separadamente.
\end{remark}

\section{Raiz quadrada e cubo}

\begin{proposition}[Função raiz quadrada]\label{prop:g10:reffunc:sqrt}
O \omterm{def:g10:functions:function}{função} $ f(x) = \sqrt{x}$, definido em $\intco{0}{+\infty}$, é
estritamente \omterm{def:g10:functions:variations}{aumentando}.
\end{proposition}

\begin{proof}
Deixar $0 \leq u < v $. Multiplique e divida por \emph{conjugado}:
\[
\sqrt v - \sqrt u
= \frac{(\sqrt v - \sqrt u)(\sqrt v + \sqrt u)}{\sqrt v + \sqrt u}
= \frac{v - u}{\sqrt v + \sqrt u} > 0,
\]
desde $ v - u > 0$ e $\sqrt v + \sqrt u > 0$ (observação $ v > 0$, então
$\sqrt v > 0$).
\end{proof}

\begin{proposition}[Função cubo]\label{prop:g10:reffunc:cube}
O \omterm{def:g10:functions:function}{função} $ f(x) = x^3$, definido em $\R $, é estritamente aumentandoe
é \omterm{def:g10:functions:graph}{gráfico} é simétrico em relação à origem.
\end{proposition}
\admitted

\begin{proposition}[Comparando $ x $, $ x^2$ e $\sqrt x $]\label{prop:g10:reffunc:compare}
Para $0 < x < 1$:\ \ $ x^2 < x < \sqrt x $.\quad
Para $ x > 1$:\ \ $\sqrt x < x < x^2$.\quad
No $ x = 0$ e $ x = 1$ os três valores são iguais.
\end{proposition}

\begin{proof}
Suponha $0 < x < 1$. Multiplicando $ x < 1$ por $ x > 0$ dá $ x^2 < x $.
Próximo, $ x < \sqrt x $: como ambos os lados são positivos e a quadratura é
\omterm{def:g10:functions:variations}{aumentando} em números positivos, esta desigualdade é equivalente a
$ x^2 < x $, o que acabamos de provar. Para $ x > 1$, multiplicando $ x > 1$ por $ x $
dá $ x^2 > x $, e $\sqrt x < x $ segue pelo mesmo argumento de quadratura.
\end{proof}

\begin{method}[Comparando imagens]\label{met:g10:reffunc:compare}
Para comparar $ f(a)$ e $ f(b)$ para uma referência \omterm{def:g10:functions:function}{função} $ f $:
\begin{enumerate}
  \item lugar $ a $ e $ b $ em um \omterm{def:g10:numbers:interval}{intervalo} onde as variações de $ f $ são
        conhecido;
  \item se $ f $ aumenta aí, imagens estão na mesma ordem que
        insumos; se $ f $ diminui, a ordem é invertida;
  \item se $ a $ e $ b $ não estão no mesmo \omterm{def:g10:numbers:interval}{intervalo} de monotonicidade, acordo
        com sinais primeiro (por exemplo\ para o quadrado: um negativo e um positivo
        entrada).
\end{enumerate}
\end{method}

\begin{example}
Comparar $\dfrac{1}{2.3}$ e $\dfrac{1}{2.4}$: ambas as entradas estão em
$\intoo{0}{+\infty}$, onde o inverso \omterm{def:g10:functions:function}{função} diminui, e
$2.3 < 2.4$, então $\dfrac{1}{2.3} > \dfrac{1}{2.4}$. Comparar $(-1.2)^2$
e $(-1.5)^2$: sobre $\intoc{-\infty}{0}$ o quadrado diminui e
$-1.5 < -1.2$, então $(-1.5)^2 > (-1.2)^2$ --- de fato $2.25 > 1.44$.
\end{example}

f(x) = 3x - 2, \qquad

\end{exercise}

\begin{exercise}[$\star $]\label{exo:g10:reffunc:5}
Deixar $ x \in \intoo{0}{1}$. Ordene os números $ x $, $ x^2$, $ x^3$ e

\begin{exercise}[$\star\star $]\label{exo:g10:reffunc:6}

\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:reffunc:7}
Sabendo que o quadrado \omterm{def:g10:functions:function}{função} diminui em $\intoc{-\infty}{0}$ e

\[
\text{(a) } x \in \intcc{2}{5};
\qquad
\text{(b) } x \in \intcc{-3}{-1};
\qquad
\text{(c) } x \in \intcc{-2}{3}.
\]
(No caso (c), tenha cuidado: o \omterm{def:g10:functions:extrema}{mínimo} de $ x^2$ não está em um ponto final.)
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:reffunc:8}
Uma empresa de táxi cobra uma taxa fixa de $4$ mais $1.5$ por quilômetro;
another charges no fee and $2$ por quilômetro. Modele cada preço por um

\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:reffunc:9}

\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g10:reffunc:10}

\end{exercise}

\section{Problema: As leis da natureza falam em funções de referência}

\begin{problem}[{Problema de fim de semana --- as distâncias de frenagem aumentam conforme
$ v^2$, alavancas obedecem $1/d $, os planetas seguintes $ a^{3/2}$: lendo
física com referência funções}]
\label{pb:g10:reffunc:1}
Abra um livro de física e os mesmos \omterm{def:g10:functions:graph}{gráficos} aparecem em todas as páginas:
a parábola, a hipérbole, a curva raiz. A natureza escreve o seu
leis com a referência deste capítulo funções --- e sabendo
suas formas (\cref{prop:g10:reffunc:square},
\cref{prop:g10:reffunc:inverse}, \cref{prop:g10:reffunc:sqrt},
\cref{prop:g10:reffunc:cube}) é suficiente para frear um carro, equilibrar um
alavanca e cronometrar os planetas. A grande final é a terceira de Kepler
lei, leia diretamente os dados do sistema solar.

\medskip
\noindent\textbf{Part I --- The square law of braking.}
Uma regra prática para a distância de frenagem de um carro em estrada seca:
$ d = \left(\frac{v}{10}\right)^{2}$ metros, em velocidade $ v $ km/h.
\begin{enumerate}
  \item Calcule as distâncias de frenagem em $30$, $50$, $90$ e
        $130$ km/h.
  \item Dobrar a velocidade multiplica a distância de frenagem por
        quanto? Explique do quadrado funçãoestá escalando,
        e verifique $50 \to 100$ km/h.
  \item Perícia: medição de marcas de derrapagem $50$ m. Que velocidade faz
        a fórmula condena o motorista de (aos km/h)? Qual
        referência \omterm{def:g10:functions:function}{função} respondeu --- e em que domínio é
        a leitura inequívoca
        (\cref{prop:g10:reffunc:sqrt})?
  \item Compare a distância extra causada por \emph{um} km/h
        mais, em baixa e em alta velocidade: computar
        $ d(31) - d(30)$ e $ d(91) - d(90)$. Qual propriedade de
        a parábola (\cref{prop:g10:reffunc:square}) faça os dois
        as respostas ilustram?
  \item A parada real adiciona tempo de reação --- cerca de um segundo,
        durante o qual o carro cobre $0.28v $ metros:
        $ D(v) = 0.28v + \left(\frac{v}{10}\right)^2$. Calcular
        $ D(50)$ e $ D(130)$. Qual dos dois termos --- afim
        ou quadrado --- domina na velocidade da cidade, e que no
        rodovia?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The hyperbolas of daily life.}
\begin{enumerate}[resume]
  \item Uma alavanca se equilibra quando a força $\times $ a distância é a
        iguais em ambos os lados. UM $60$ criança kg sentada $2$ estou do
        pivô; a força de equilíbrio à distância $ d $ por outro
        lado é $ F(d) = \frac{120}{d}$ (em kg equivalentes).
        Calcular $ F(0.5)$, $ F(1)$, $ F(3)$.
  \item O que significam as variações do inverso \omterm{def:g10:functions:function}{função}
        (\cref{prop:g10:reffunc:inverse}) diz sobre alavancas ---
        e por que Arquimedes se vangloria (“dê-me um lugar para
        ficar de pé e eu moverei a Terra'') se esconder no
        cauda da hipérbole? O que proíbe $ d = 0$?
  \item O som e a luz desaparecem com o \emph{quadrado} do
        distância: em $1$ m de uma lâmpada a intensidade é $100$
        unidades; dê em $2$, $3$ e $10$ m. Explique o
        expoente $2$ com uma esfera: sobre qual área tem a luz
        espalhar à distância $ d $ (O problema da lápide de Arquimedes, no volume do Ensino Médio)?
  \item Um ônibus para a viagem escolar custa $600$ euros, dividir
        igualmente entre $ n $ participantes. Tabule o custo por
        vá para $ n = 10, 20, 30$, descubra quantos participantes
        traga-o para $25$ euros ou menos, e indicar qual é o valor
        a forma da hipérbole promete --- e recusa --- como $ n $
        cresce.
  \item Produzindo $ x $ custos de cartazes $2x + 3$ euros ($3$ euros de
        configuração). Mostre que o \emph{média} o custo por pôster é
        $ a(x) = 2 + \frac3x $, descreva suas variações e
        interpretar sua assíntota horizontal economicamente.
        (Compare a álgebra de \cref{exo:g10:reffunc:10}.)
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Kepler's harmony.}
Para cada planeta, deixe $ a $ seja a sua distância média ao Sol (em
unidades astronômicas, Terra $= 1$) e $ T $ seu período (em anos).
Dados: Marte $ a = 1.52$, $ T = 1.88$; Júpiter $ a = 5.20$,
$ T = 11.86$; Saturno $ a = 9.54$, $ T = 29.4$.
\begin{enumerate}[resume]
  \item Calcular $ a^3$ e $ T^2$ para a Terra, Marte e Júpiter.
        O que Kepler percebeu em 1618?
  \item Enuncie a lei como um \omterm{def:g10:functions:function}{função}: $ T = \sqrt{a^3} = a\sqrt a $.
        Teste-o em Saturno.
  \item Duas previsões: um asteróide orbita em $ a = 4$ UA ---
        encontre seu período (os números saem \omterm{def:g10:numbers:sets}{inteiros}); um cometa
        retorna a cada $27$ anos --- encontre sua distância média
        (procure um cubo perfeito).
  \item Quais três referências funções a lei trança
        juntos? E sua regra de escala: quando $ a $ é multiplicado
        por $4$, o que acontece com $ T $? (Verifique com o asteróide
        contra a Terra.)
  \item Kepler leu esta lei nos dados de Tycho Brahe setenta anos
        antes que a gravitação de Newton explicasse isso. Em um
        frase: o que este episódio diz sobre o poder de
        reconhecendo uma referência \omterm{def:g10:functions:function}{função} em uma tabela de números?
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The tortoise and the hare.}
\begin{enumerate}[resume]
  \item Ordem $\sqrt x $, $ x $ e $ x^2$ sobre $\intoo{0}{1}$ e assim por diante
        $\intoo{1}{+\infty}$
        (\cref{prop:g10:reffunc:compare}) e verifique os dois
        pedidos em $ x = 0.25$ e $ x = 4$.
  \item Resolver $\sqrt x > x $ completamente (por $ x \geq 0$, quadrado
        profundamente e terminar com um \omterm{def:g10:algebra:signtable}{mesa de sinalização}).
  \item Traga o cubo: ordem $ x $, $ x^2$, $ x^3$ no
        $ x = 0.5$ e em $ x = 2$, e encontre todos os pontos onde
        quadrado e o cubo funções cruzar.
  \item Dois regimes de poupança pagam, após $ t $ anos, $100\sqrt t $
        euros (regime A) ou $10 t^2$ euros (regime B). Mostre isso
        B ultrapassa A quando $ t^3 = 100$, e dê o cruzamento
        tempo para o mês. Moral sobre raízes contra poderes em
        a longo prazo?
  \item Finale: para cada lei deste problema --- frenagem
        ($ v^2$), a alavanca ($1/d $), a lâmpada ($1/d^2$), Kepler
        ($ a^{3/2}$), esquema A ($\sqrt t$) --- digamos em uma cláusula
        qual característica de sua referência \omterm{def:g10:functions:graph}{gráfico} carrega o
        significado físico (aumento acentuado, assíntota vertical,
        esfera de expansão, expoente de escala, crescimento achatado).
        Conclusão: a referência funções são o alfabeto;
        a natureza escreve com eles.
\end{enumerate}
\end{problem}
