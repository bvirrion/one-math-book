\chapter{Álgebra: Equações e Desigualdades}\label{ch:g10:algebra}

Álgebra é a arte de calcular com letras. Este capítulo analisa e
aguça os dois movimentos básicos --- expandindo e \omterm{def:g10:algebra:expand}{factoring} --- e os usa
resolver equações e desigualdades de forma sistemática, com a \omterm{def:g10:algebra:signtable}{mesa de sinalização} como
a ferramenta central.

\section{Expandindo e fatorando}

\begin{definition}[Expandindo, fatorando]\label{def:g10:algebra:expand}
\emph{Expandindo}\index{Expandindo} uma expressão significa transformar
produtos em somas; \emph{factoring}\index{factoring} significa o inverso,
transformar uma soma em um produto. A regra básica é a distributividade:
\[
k(a + b) = ka + kb,
\qquad
(a + b)(c + d) = ac + ad + bc + bd .
\]
\end{definition}

\begin{example}\label{ex:g10:algebra:expand}
Expandir $(2x + 3)(x - 5)$ passo a passo:
\begin{align*}
(2x + 3)(x - 5)
&= 2x \times x + 2x \times (-5) + 3 \times x + 3 \times (-5) \\
&= 2x^2 - 10x + 3x - 15 \\
&= 2x^2 - 7x - 15 .
\end{align*}
\end{example}

\begin{theorem}[Identidades notáveis]\label{thm:g10:algebra:identities}
Para todos \omterm{def:g10:numbers:sets}{números reais} $ a $ e $ b $:
\[
(a+b)^2 = a^2 + 2ab + b^2,
\qquad
(a-b)^2 = a^2 - 2ab + b^2,
\qquad
(a+b)(a-b) = a^2 - b^2 .
\]
\end{theorem}

\begin{proof}
Cada um é uma expansão direta. Para o primeiro:
\[
(a+b)^2 = (a+b)(a+b) = a^2 + ab + ba + b^2 = a^2 + 2ab + b^2 .
\]
O segundo segue substituindo $ b $ com $-b $: $(a-b)^2 = a^2 + 2a(-b) +
(-b)^2 = a^2 - 2ab + b^2$. Para o terceiro:
$(a+b)(a-b) = a^2 - ab + ba - b^2 = a^2 - b^2$.
\end{proof}

\begin{example}\label{ex:g10:algebra:identities}
Usado para frente (expandindo) e para trás (\omterm{def:g10:algebra:expand}{factoring}):
\begin{align*}
(3x + 2)^2 &= 9x^2 + 12x + 4,
& x^2 - 25 &= (x+5)(x-5), \\
(x - 4)^2 &= x^2 - 8x + 16,
& 4x^2 + 4x + 1 &= (2x + 1)^2 .
\end{align*}
Benefícios da aritmética mental também: $101 \times 99 = (100+1)(100-1) =
10000 - 1 = 9999$.
\end{example}

\begin{method}[Fatorando uma expressão]\label{met:g10:algebra:factor}
Experimente, nesta ordem:
\begin{enumerate}
  \item \emph{Fator comum:} identifique um fator presente em cada termo e puxe
        para fora, por exemplo.\ $6x^2 + 4x = 2x(3x + 2)$. O fator comum pode ser um
        colchete inteiro: $(x+1)(2x-3) + (x+1)(x+7) = (x+1)(3x+4)$.
  \item \emph{Identidade notável:} reconhecer $ a^2 - b^2$, ou
        $ a^2 \pm 2ab + b^2$, por exemplo\ $9x^2 - 16 = (3x)^2 - 4^2 =
        (3x+4)(3x-4)$.
  \item Combine ambos e verifique o resultado expandindo de volta.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g10:algebra:factorsteps}
Fator $ E = (2x + 1)^2 - (x - 3)^2$. Esta é uma diferença de dois quadrados
$ a^2 - b^2$ com $ a = 2x+1$ e $ b = x-3$:
\begin{align*}
E &= \bigl[(2x+1) + (x-3)\bigr] \times \bigl[(2x+1) - (x-3)\bigr] \\
  &= (3x - 2)(2x + 1 - x + 3) \\
  &= (3x - 2)(x + 4).
\end{align*}
Cuidado com o segundo colchete: subtraindo $ x - 3$ significa subtrair $ x $
\emph{e} adicionando $3$.
\end{example}

\section{Equações}

\begin{definition}[Equação, solução]\label{def:g10:algebra:equation}
Um \emph{pesquisa}\index{pesquisa} é uma igualdade contendo uma incógnita
número $ x $; um \emph{solução} é um valor de $ x $ que faz a igualdade
verdade. Resolver a equação significa encontrar \emph{todos} suas soluções.
\end{definition}

Dois equações são \emph{equivalente} quando eles têm exatamente o mesmo
soluções. Adicionando o mesmo número a ambos os lados ou multiplicando ambos os lados
pelo mesmo \emph{diferente de zero} número, produz um equivalente equação.

\begin{example}[Equação de primeiro grau]\label{ex:g10:algebra:firstdegree}
Resolver $5x - 7 = 2x + 8$, um movimento de cada vez:
\begin{align*}
5x - 7 &= 2x + 8 \\
5x - 2x &= 8 + 7 && \text{(add $7 - 2x $ to both sides)} \\
3x &= 15 \\
x &= 5 && \text{(divide both sides by $3$).}
\end{align*}
A solução única é $5$. Verificar: $5 \times 5 - 7 = 18$ e
$2 \times 5 + 8 = 18$.
\end{example}

\begin{theorem}[Regra de produto zero]\label{thm:g10:algebra:zeroproduct}
Um produto de \omterm{def:g10:numbers:sets}{números reais} é zero se e somente se pelo menos um de seus
fatores é zero:
\[
A \times B = 0
\quad\Longleftrightarrow\quad
A = 0 \ \text{ or } \ B = 0 .
\]
\end{theorem}

\begin{proof}
Se $ A = 0$ ou $ B = 0$, o produto é claramente $0$. Por outro lado, suponha
$ AB = 0$ e $ A \neq 0$. Dividindo ambos os lados por $ A $ dá $ B = 0$.
\end{proof}

\begin{method}[Resolvendo por fatoração]\label{met:g10:algebra:zeroproduct}
Para resolver um equação cujo lado direito é $0$ e cuja mão esquerda
lado pode ser fatorado:
\begin{enumerate}
  \item traga tudo para a esquerda para que o equação lê $ E = 0$;
  \item fator $ E $ num produto de fatores de primeiro grau;
  \item aplique a regra do produto zero e resolva cada pequeno equação;
  \item coletar todas as soluções encontradas.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g10:algebra:zeroproduct}
Resolver $(x - 2)(3x + 6) = 0$: qualquer $ x - 2 = 0$, dando $ x = 2$, ou
$3x + 6 = 0$, dando $ x = -2$. Soluções: $-2$ e $2$.

Resolver $ x^2 = 9x $. Fazer \emph{não} dividir por $ x $ (pode ser zero!); trazer
tudo de lado e fator:
\[
x^2 - 9x = 0
\quad\Longleftrightarrow\quad
x(x - 9) = 0
\quad\Longleftrightarrow\quad
x = 0 \ \text{ or } \ x = 9 .
\]
\end{example}

\begin{example}[Equação de quociente]\label{ex:g10:algebra:quotient}
Resolver $\dfrac{x - 1}{x + 2} = 0$. Um quociente é zero exatamente quando
numerador é zero \emph{e} seu denominador não é. Aqui $ x - 1 = 0$
dá $ x = 1$, e $1 + 2 = 3 \neq 0$: a única solução é $1$. O valor
$ x = -2$ é \emph{proibido} (divisão por zero) e deve ser sempre
excluído primeiro.
\end{example}

\section{Desigualdades e tabelas de sinais}

\begin{proposition}[Regras para desigualdades]\label{prop:g10:algebra:ineqrules}
Deixar $ a \leq b $.
\begin{enumerate}
  \item Para qualquer $ c $: $ a + c \leq b + c $ (adicionar preserva a ordem).
  \item Se $ c > 0$: $ ac \leq bc $ (multiplicando por um número positivo
        preserva a ordem).
  \item Se $ c < 0$: $ ac \geq bc $ (multiplicando por um número negativo
        \emph{inverte} a ordem).
\end{enumerate}
\end{proposition}

\begin{proof}
1. $(b + c) - (a + c) = b - a \geq 0$.
2. $ bc - ac = (b-a)c $ é o produto de dois números não negativos, então
$ bc \geq ac $.
3. Agora $(b-a)c \leq 0$ como o produto de um não negativo e um negativo
número, então $ bc \leq ac $.
\end{proof}

\begin{example}\label{ex:g10:algebra:firstineq}
Resolver $-2x + 3 > 9$:
\begin{align*}
-2x + 3 &> 9 \\
-2x &> 6 && \text{(subtract 3)} \\
x &< -3 && \text{(divide by $-2 < 0$: reverse the inequality).}
\end{align*}
O conjunto solução é $\intoo{-\infty}{-3}$.
\end{example}

\begin{definition}[Mesa de sinalização]\label{def:g10:algebra:signtable}
UM \emph{mesa de sinalização}\index{mesa de sinalização} registra, em uma linha por fator, o
sinal ($+$ ou $-$) de cada fator de uma expressão como $ x $ atropela $\R $,
com um $0$ em cada valor onde o fator desaparece; uma linha final dá o
sinal do produto, utilizando a regra dos sinais coluna por coluna.
\end{definition}

\begin{example}\label{ex:g10:algebra:signtable}
Sinal de $ P(x) = (x - 2)(3 - x)$. O fator $ x - 2$ desaparece em $2$ e é
positivo depois; o fator $3 - x $ desaparece em $3$ e é positivo antes:
\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{c|ccccc}
$ x $ & $-\infty $ \quad & $2$ & & $3$ & \quad $+\infty $ \\
\hline
$ x - 2$ & $-$ & $0$ & $+$ & & $+$ \\
$3 - x $ & $+$ & & $+$ & $0$ & $-$ \\
\hline
$ P(x)$ & $-$ & $0$ & $+$ & $0$ & $-$ \\
\end{tabular}
\end{center}
Então $ P(x) > 0$ exatamente em $\intoo{2}{3}$, e $ P(x) \leq 0$ sobre
$\intoc{-\infty}{2} \cup \intco{3}{+\infty}$.
\end{example}

\begin{method}[Resolvendo uma inequação com uma tabela de sinais]\label{met:g10:algebra:signtable}
Para resolver uma desigualdade como $ E(x) \geq 0$ ou $ E(x) < 0$:
\begin{enumerate}
  \item traga tudo para o lado esquerdo, então o lado direito é
        $0$ (nunca multiplique ambos os lados por uma expressão cujo sinal seja
        desconhecido);
  \item fatore o lado esquerdo ou coloque-o sobre um denominador comum se
        é um quociente;
  \item construir o \omterm{def:g10:algebra:signtable}{mesa de sinalização} com uma linha por fator (para um quociente, marque
        valores proibidos com barra dupla na última linha);
  \item leia o \omterm{def:g10:numbers:interval}{intervalos} onde se encontra o sinal solicitado, verificando
        se cada limite está incluído.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g10:algebra:quotientineq}
Resolver $\dfrac{x + 1}{x - 3} \leq 0$. O numerador desaparece em $-1$, o
denominador em $3$ (valor proibido). O quociente é negativo quando
dois têm sinais opostos, o que acontece para $ x $ entre $-1$ e $3$.
Incluindo o limite onde o \emph{numerador} desaparece, mas excluindo o
valor proibido: o conjunto solução é $\intco{-1}{3}$.
\end{example}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g10:algebra:1}
Expanda e simplifique:
\[
3(2x - 5) - 2(x - 7), \qquad
(x + 4)(2x - 1), \qquad
(3x - 2)^2, \qquad
(5 - x)(5 + x).
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g10:algebra:2}
Fator:
\[
x^2 - 49, \qquad
x^2 + 10x + 25, \qquad
7x^2 - 14x, \qquad
16x^2 - 8x + 1 .
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g10:algebra:3}
Resolver:
\[
4x + 9 = x - 3, \qquad
\frac{2x - 1}{3} = 5, \qquad
2(x - 3) = 2x + 1 .
\]
(Um deles não tem solução --- explique o porquê.)
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g10:algebra:4}
Resolva usando a regra do produto zero:
\[
(x + 5)(2x - 8) = 0, \qquad
x^2 - 16 = 0, \qquad
x^2 + 6x + 9 = 0 .
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g10:algebra:5}
Resolva as desigualdades e escreva os conjuntos solução como \omterm{def:g10:numbers:interval}{intervalos}:
\[
3x - 5 \leq 7, \qquad
-4x + 1 > 9, \qquad
2(x - 1) \geq 5x + 4 .
\]
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:algebra:6}
Fator $ E = (x - 1)(x + 3) - (x - 1)(2x - 5)$, então resolva $ E = 0$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:algebra:7}
Resolver $ x^2 = 5x $, então $(2x + 1)^2 = (x - 4)^2$. (Para o segundo,
traga tudo para a esquerda e fatore a diferença dos quadrados.)
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:algebra:8}
Usando um \omterm{def:g10:algebra:signtable}{mesa de sinalização}, resolver
\[
(x + 2)(4 - x) > 0,
\qquad
\frac{2x - 6}{x + 1} \geq 0 .
\]
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:algebra:9}
Um serviço de streaming custa $9$ por mês, mais uma taxa única de inscrição de
$15$; um rival custa $12$ por mês sem taxa. Depois de quantos meses
o primeiro serviço se tornou a escolha mais barata em geral? Configurar e resolver um
desigualdade.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:algebra:10}
Resolver $\dfrac{x+3}{x-1} = 2$. (Multiplique ambos os lados por $ x - 1$ depois
excluindo o valor proibido, ou trazer tudo para a esquerda durante um
denominador comum.)
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g10:algebra:11}
Deixar $ n $ feijão inteiro. Mostre isso $(n + 1)^2 - n^2$ é sempre estranho, e isso
a diferença entre os quadrados de dois números ímpares consecutivos é
sempre um múltiplo de $8$.
\end{exercise}

\section{Problema: Da Babilônia à proporção áurea}

\begin{problem}[{Problema de fim de semana --- problemas de soma e produto,
completando o quadrado da maneira que al-Khwarizmi o desenhou, e o
equação $ x^2 = x + 1$}]
\label{pb:g10:algebra:1}
Há quatro mil anos, os escribas babilônios resolveram isso no barro:
\emph{encontre dois números sabendo sua soma e seu produto}. Seu
truque --- procure os números simetricamente ao redor do
meia soma --- ainda quebra cada quadrático equação hoje, sob
seu nome moderno, \emph{completando o quadrado}. Este problema
aprende o truque, observa al-Khwarizmi literalmente completar um
square, and puts the method to work on the most celebrated
proporção de todos eles.

\medskip
\noindent\textbf{Part I --- The Babylonian problem.}
\begin{enumerate}
  \item Aquecimento (\cref{thm:g10:algebra:identities}): fator
        $ x^2 - 10x + 25$, então $9x^2 - 49$, então --- combinando um
        identidade com diferença de quadrados ---
        $ x^2 + 6x + 9 - 4$.
  \item Resolver $(x + 1)(x + 5) = 0$ e
        $(3x - 7)(3x + 7) = 0$
        (\cref{thm:g10:algebra:zeroproduct}).
  \item O problema do escriba: dois números têm soma $10$ e
        produto $21$. Seguindo Babilônia, escreva-os como $5 - t $
        e $5 + t $ (simétrico em torno da meia soma) e encontre $ t $,
        então os dois números.
  \item Mesmo método: soma $14$, produto $33$.
  \item A receita geral: para somar $ s $ e produto $ p $, mostrar
        que os dois números são
        \[
        \frac s2 - t
        \quad\text{and}\quad
        \frac s2 + t,
        \qquad\text{where}\quad
        t^2 = \left(\frac s2\right)^2 - p ,
        \]
        e indique a condição em $ s $ e $ p $ para que o problema
        tenha uma solução.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Completing the square.}
\begin{enumerate}[resume]
  \item Expandir $(x - a)(x - b)$ e concluo: as soluções de
        $ x^2 - sx + p = 0$ são exatamente os pares de números com
        soma $ s $ e produto $ p $. (Os escribas estavam resolvendo
        quadrático equações sem saber.)
  \item Verifique isso $ x^2 - 10x + 21 = (x - 5)^2 - 4$ e resolva
        $ x^2 - 10x + 21 = 0$ por \omterm{def:g10:algebra:expand}{factoring} o lado direito como um
        diferença de quadrados. Compare com a pergunta 3.
  \item Resolver $ x^2 + 6x - 7 = 0$ completando o quadrado.
  \item Complete o quadrado em $ x^2 + 4x + 5$ e explique por que
        o equação $ x^2 + 4x + 5 = 0$ tem \emph{não} verdadeiro
        solução. Enuncie o critério geral: uma vez escrito como
        $(x + h)^2 + k $, quando uma expressão quadrática
        desaparecer?
  \item O nome é literal. Na Bagdá do século IX,
        al-Khwarizmi desenhou $ x^2 + 6x = 7$ como um quadrado de lado $ x $
        com dois $3 \times x $ retângulos colados em dois de seus
        lados, e \emph{completou o quadrado} com a falta
        $3 \times 3$ canto. Desenhe a figura e explique como ela
        vira o equação em $(x + 3)^2 = 16$ --- então resolva.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Signs, tables and two old
promises.}
\begin{enumerate}[resume]
  \item Construa o \omterm{def:g10:algebra:signtable}{mesa de sinalização} de $(x - 3)(x + 2)$
        (\cref{def:g10:algebra:signtable}) e resolver
        $(x - 3)(x + 2) < 0$.
  \item Resolver $ x^2 \leq 10x - 21$ (traga tudo para o lado
        e reutilizar a questão 7).
  \item Resolver $\dfrac{x - 1}{x + 2} \geq 0$ com um \omterm{def:g10:algebra:signtable}{mesa de sinalização},
        cuidando do valor proibido.
  \item Uma velha promessa cumprida: para o agricultor de
        problema da cerca da rainha Dido, no volume do Ensino Médio, com $20$ m de cerca, a área
        de um $ x \times (10 - x)$ caneta satisfaz
        \[
        A(x) = x(10 - x) = 25 - (x - 5)^2 .
        \]
        Verifique a identidade e leia-a \emph{prova} de
        o que a 6ª série só poderia conjecturar: a área máxima,
        e o único $ x $ conseguindo isso.
  \item Outro: prove que $ x + \dfrac1x \geq 2$ para cada
        $ x > 0$, com igualdade apenas em $ x = 1$. (Multiplique por
        $ x $, reúna, reconheça um quadrado --- o
        aritmética - desigualdade geométrica de
        o problema do fim de semana das relações de Euclides do volume do ensino médio em uma nova roupagem.)
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The golden section.}
Corte um segmento de comprimento $1$ em grande parte $ x $ e um pequeno
parte $1 - x $ para que
\emph{o todo é muito grande assim como o grande é muito pequeno}:
$\frac{1}{x} = \frac{x}{1 - x}$.
\begin{enumerate}[resume]
  \item Mostre que a condição é $ x^2 + x - 1 = 0$.
  \item Resolva completando o quadrado e mantenha a raiz que
        é um comprimento: $ x = \frac{\sqrt5 - 1}{2} \approx 0.618$.
  \item O \emph{proporção áurea} é
        $\varphi = \frac1x = \frac{2}{\sqrt5 - 1}$. Racionalizar
        o denominador (multiplique o topo e o fundo por
        $\sqrt5 + 1$, uma identidade no trabalho) e mostrar
        $\varphi = \frac{\sqrt5 + 1}{2} \approx 1.618$.
  \item Verificar \emph{exatamente} as duas identidades mágicas de
        $\varphi $:
        \[
        \varphi^2 = \varphi + 1,
        \qquad
        \frac{1}{\varphi} = \varphi - 1 .
        \]
  \item Final: explique por que $\varphi $ é \omterm{ex:g10:numbers:classify}{irracional}
        (o problema da irracionalidade do fim de semana do volume do Ensino Médio fornece o fato-chave sobre
        $\sqrt5$), então calcule as proporções $\frac53$,
        $\frac85$, $\frac{13}{8}$, $\frac{21}{13}$ de
        números consecutivos de Hemachandra-Fibonacci
        (o problema de contagem rítmica do fim de semana do volume do Ensino Médio) com três casas decimais. O que fazer
        eles parecem perseguir? (A prova de que eles pegaram
        pertence aos capítulos sobre sequências.)
\end{enumerate}
\end{problem}
