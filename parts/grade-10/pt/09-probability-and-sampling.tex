\chapter{Probabilidade e Amostragem}\label{ch:g10:proba}

\omterm{def:g10:proba:distribution}{Probabilidade} assigns numbers to chance. Starting from experiments with
resultados igualmente prováveis --- dados, moedas, cartas --- este capítulo constrói o
basic rules for computing probabilities of \omterm{def:g10:proba:events}{eventos}, apresenta \omterm{met:g10:proba:tree}{árvore}
diagramas para experimentos em duas etapas e termina com a ligação entre
\omterm{def:g10:proba:distribution}{probabilidades} e \omterm{def:g10:stats:series}{frequências} observadas. A teoria continua em
\cref{ch:g11:prob}.

\section{Experimentos e eventos aleatórios}

\begin{definition}[Resultados e eventos]\label{def:g10:proba:events}
UM \emph{experimento aleatório}\index{experimento aleatório} é um experimento cujo
o resultado não pode ser previsto (lançamento de um dado). O conjunto de todos os possíveis
resultados é o \emph{espaço amostral}\index{espaço amostral}
$\Omega = \{\omega_1, \dots, \omega_n\}$; seus elementos são os
\emph{resultados}. Um \emph{evento}\index{evento} é um conjunto de resultados; um
evento \emph{ocorre} quando o resultado do experimento lhe pertence.
\end{definition}

\begin{definition}[Distribuição de probabilidade]\label{def:g10:proba:distribution}
Atribuindo a cada resultado $\omega_i $ um número $ p_i \geq 0$, com
$ p_1 + \dots + p_n = 1$, define um \emph{probabilidade}\index{probabilidade}:
a probabilidade $\P(A)$ de um \omterm{def:g10:proba:events}{evento} $ A $ é a soma dos $ p_i $ do
resultados em $ A $. Em particular $\P(\Omega) = 1$ e o impossível \omterm{def:g10:proba:events}{evento}
$\varnothing $ tem probabilidade $0$.
\end{definition}

\begin{proposition}[Resultados igualmente prováveis]\label{prop:g10:proba:uniform}
Se o $ n $ os resultados são igualmente prováveis (cada $ p_i = \frac1n $), então para
cada \omterm{def:g10:proba:events}{evento} $ A $:
\[
\P(A) = \frac{\text{number of outcomes in } A}{n}
= \frac{\text{favorable cases}}{\text{possible cases}} .
\]
\end{proposition}

\begin{proof}
$ A $ contém algum número $ k $ de resultados, cada um \omterm{def:g10:proba:distribution}{probabilidade} $\frac1n $,
então $\P(A) = \frac kn $.
\end{proof}

\begin{example}\label{ex:g10:proba:die}
Jogue um dado justo: $\Omega = \{1, 2, 3, 4, 5, 6\}$, cada resultado tendo
\omterm{def:g10:proba:distribution}{probabilidade} $\frac16$. O \omterm{def:g10:proba:events}{evento} $ A $: ``o resultado é par'' é
$ A = \{2, 4, 6\}$, então $\P(A) = \frac36 = \frac12$. O \omterm{def:g10:proba:events}{evento} $ B $: ``o
o resultado é pelo menos $5$'' é $ B = \{5, 6\}$, então $\P(B) = \frac26 =
\frac13$.
\end{example}

\section{Combinando eventos}

\begin{definition}[Intersecção, união, complemento]\label{def:g10:proba:operations}
Deixar $ A $ e $ B $ ser \omterm{def:g10:proba:events}{eventos}.
\begin{itemize}
  \item $ A \cap B $ (``$ A $ e $ B $'') ocorre quando ambos ocorrem;
  \item $ A \cup B $ (``$ A $ ou $ B $'') ocorre quando pelo menos um ocorre;
  \item o \emph{complemento}\index{complemento} $\bar A $ (``não $ A $'')
        ocorre exatamente quando $ A $ não;
  \item $ A $ e $ B $ são \emph{incompatível}\index{incompatível} quando não podem ocorrer
        juntos: $ A \cap B = \varnothing $.
\end{itemize}
\end{definition}

\begin{theorem}[Regras de adição]\label{thm:g10:proba:addition}
Para todos \omterm{def:g10:proba:events}{eventos} $ A $ e $ B $:
\[
\P(A \cup B) = \P(A) + \P(B) - \P(A \cap B),
\qquad
\P(\bar A) = 1 - \P(A) .
\]
Para \omterm{def:g10:proba:operations}{incompatível} \omterm{def:g10:proba:events}{eventos}, a primeira regra simplifica para
$\P(A \cup B) = \P(A) + \P(B)$.
\end{theorem}

\begin{proof}
Resumindo $\P(A) + \P(B)$ adiciona o \omterm{def:g10:proba:distribution}{probabilidade} de cada resultado de
$ A \cup B $ uma vez, \emph{exceto} aqueles de $ A \cap B $, que são contados
duas vezes --- uma vez $ A $ e uma vez em $ B $. Subtraindo $\P(A \cap B)$ reparos
a dupla contagem. Para o \omterm{def:g10:proba:operations}{complemento}: $ A $ e $\bar A $ são \omterm{def:g10:proba:operations}{incompatível}
e $ A \cup \bar A = \Omega $, então $\P(A) + \P(\bar A) = \P(\Omega) = 1$.
\end{proof}

\begin{example}\label{ex:g10:proba:cards}
Compre uma carta de um padrão $52$-baralho de cartas. Deixar $ A $: ``o cartão é um
coração'' ($13$ cartões) e $ B $: ``a carta é um rei'' ($4$ cartões). Então
$ A \cap B $ é ``o rei dos corações'' ($1$ cartão) e
\[
\P(A \cup B) = \frac{13}{52} + \frac{4}{52} - \frac{1}{52} = \frac{16}{52}
= \frac{4}{13}.
\]
O \omterm{def:g10:proba:operations}{complemento} regra é muitas vezes o caminho mais rápido: o \omterm{def:g10:proba:distribution}{probabilidade} que o
cartão é \emph{não} um coração é $1 - \frac{13}{52} = \frac34$.
\end{example}

\section{Diagramas de árvore}

\begin{method}[Experimentos em duas etapas]\label{met:g10:proba:tree}
Para um experimento realizado em duas etapas:
\begin{enumerate}
  \item desenhe um \emph{árvore}\index{árvore}: um ramo por possível
        resultado da primeira etapa, então de cada um deles, um ramo por
        resultado do segundo passo, escrevendo o \omterm{def:g10:proba:distribution}{probabilidade} em cada filial
        (os ramos que saem de qualquer nó devem somar $1$);
  \item o \omterm{def:g10:proba:distribution}{probabilidade} de um \emph{caminho} (uma folha do \omterm{met:g10:proba:tree}{árvore}) é o
        produto das \omterm{def:g10:proba:distribution}{probabilidades} ao longo dos seus ramos;
  \item o \omterm{def:g10:proba:distribution}{probabilidade} de um \omterm{def:g10:proba:events}{evento} é a soma das \omterm{def:g10:proba:distribution}{probabilidades} de
        os caminhos que o realizam.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g10:proba:tree}
Uma urna contém $3$ bolas vermelhas e $2$ bolas azuis. Desenhe uma bola, coloque-a
de volta, desenhe novamente. Cada sorteio dá vermelho (R) com \omterm{def:g10:proba:distribution}{probabilidade} $\frac35$ e
azul (B) com \omterm{def:g10:proba:distribution}{probabilidade} $\frac25$.

soma das \omterm{def:g10:proba:distribution}{probabilidades} $1$.}
\end{omfigure}

O \omterm{def:g10:proba:distribution}{probabilidade} de obter duas bolas da mesma cor é
$\P(\text{RR}) + \P(\text{BB}) = \frac{9}{25} + \frac{4}{25} =
\frac{13}{25}$; o \omterm{def:g10:proba:distribution}{probabilidade} de pelo menos um azul é
$1 - \P(\text{RR}) = 1 - \frac{9}{25} = \frac{16}{25}$ (\omterm{def:g10:proba:operations}{complemento} regra).
\end{example}

\begin{remark}
Sem reposição, as \omterm{def:g10:proba:distribution}{probabilidades} do segundo empate dependem do
primeiro: com a urna acima, depois de tirar uma bola vermelha, só $2$ vermelho e $2$
azul permanece, então o segundo ramo carrega $\frac24$ e $\frac24$. O
\omterm{met:g10:proba:tree}{árvore} o método funciona inalterado; apenas os números nas ramificações de segundo nível
mudar.
\end{remark}

\section{Frequências e amostragem}

Quando um \omterm{def:g10:proba:events}{experimento aleatório} é repetido muitas vezes, o observado freqüência
de um \omterm{def:g10:proba:events}{evento} se aproxima de seu \omterm{def:g10:proba:distribution}{probabilidade} --- é por isso \omterm{def:g10:proba:distribution}{probabilidade} é
útil para descrever o mundo real.

\begin{definition}[Amostra]\label{def:g10:proba:sample}
UM \emph{amostra}\index{amostra} de tamanho $ n $ é o resultado da repetição do
mesmo experimento $ n $ vezes de forma independente. O \emph{observado freqüência} de
um \omterm{def:g10:proba:events}{evento} na amostra é o número de ocorrências dividido por $ n $.
\end{definition}

\begin{example}\label{ex:g10:proba:sampling}
Uma moeda justa dá cara com \omterm{def:g10:proba:distribution}{probabilidade} $0.5$. Jogando $100$ vezes
raramente dá exatamente $50$ cabeças: \omterm{def:g10:proba:sample}{amostras} \emph{flutuar}. Corridas típicas de
$100$ lançamentos fornecem \omterm{def:g10:stats:series}{frequências} como $0.46$, $0.53$, $0.49$ --- perto de
$0.5$, mas não igual. Maior \omterm{def:g10:proba:sample}{amostras} flutuar menos: com $10\,000$
lançamentos, as \omterm{def:g10:stats:series}{frequências} observadas geralmente estão dentro de cerca de $0.01$ de $0.5$.
\end{example}

\begin{remark}[Intervalo de flutuação]
Uma regra prática útil, justificada em \cref{ch:g11:binom}: para um \omterm{def:g10:proba:sample}{amostra} de
tamanho $ n $ de um \omterm{def:g10:proba:events}{evento} de \omterm{def:g10:proba:distribution}{probabilidade} $ p $ (com $ n $ razoavelmente grande), o
observado freqüência encontra-se no \omterm{def:g10:numbers:interval}{intervalo}
\[
\intcc{p - \frac{1}{\sqrt n}}{\,p + \frac{1}{\sqrt n}}
\]
em aproximadamente $95\%$ de \omterm{def:g10:proba:sample}{amostras}. Se for observado freqüência cai longe lá fora
isso \omterm{def:g10:numbers:interval}{intervalo}, duvide do valor de $ p $: este é o ponto de partida
testes estatísticos.
\end{remark}

\begin{example}\label{ex:g10:proba:test}
Uma fábrica afirma que apenas $10\%$ de suas peças estão com defeito
($ p = 0.1$). Em um \omterm{def:g10:proba:sample}{amostra} de $ n = 400$ peças, $64$ estão com defeito: observados
freqüência $\frac{64}{400} = 0.16$. A flutuação \omterm{def:g10:numbers:interval}{intervalo} é
$\intcc{0.1 - \frac{1}{20}}{0.1 + \frac{1}{20}} = \intcc{0.05}{0.15}$,
e $0.16$ fica fora dele: o \omterm{def:g10:proba:sample}{amostra} lança sérias dúvidas sobre a afirmação.
\end{example}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g10:proba:1}
Um dado justo é lançado. Calcule o \omterm{def:g10:proba:distribution}{probabilidade} de cada \omterm{def:g10:proba:events}{evento}: ``conseguir um
$6$''; ``obter um número ímpar''; ``obtendo no \omterm{def:g10:functions:extrema}{máximo} $4$''; ``obter um
$7$''.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g10:proba:2}
Uma bolsa contém $5$ verde, $3$ amarelo e $2$ bolinhas de gude pretas; uma bola de gude é
sorteado aleatoriamente. Calcule o \omterm{def:g10:proba:distribution}{probabilidade} que é verde; que não é
preto; que é verde ou amarelo.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g10:proba:3}
Em uma aula de $30$ estudantes, $18$ estudar espanhol, $10$ estudar alemão e
$4$ estudar ambos. Um aluno é escolhido aleatoriamente. Usando a regra de adição,
calcular o \omterm{def:g10:proba:distribution}{probabilidade} que este aluno estude pelo menos um dos dois
línguas, então o \omterm{def:g10:proba:distribution}{probabilidade} que eles não estudam nenhum dos dois.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g10:proba:4}
Uma moeda justa é lançada duas vezes. Desenhe o \omterm{met:g10:proba:tree}{árvore} do experimento e calcular
o \omterm{def:g10:proba:distribution}{probabilidade} de conseguir duas cabeças; exatamente uma cabeça; pelo menos um
cabeça.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:proba:5}
Uma urna contém $4$ vermelho e $6$ bolas brancas. Duas bolas são sorteadas uma após
o outro \emph{sem} substituição. Desenhe o \omterm{met:g10:proba:tree}{árvore} com o correto
\omterm{def:g10:proba:distribution}{probabilidades} no segundo nível e calcular o \omterm{def:g10:proba:distribution}{probabilidade} de desenho
duas bolas vermelhas, depois de desenhar duas bolas de cores diferentes.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:proba:6}
Dois dados honestos são lançados e sua soma é registrada.
\begin{enumerate}
  \item Explique por que \omterm{def:g10:proba:events}{espaço amostral} pode ser tomado como o $36$ igualmente
        pares prováveis $(1,1), (1,2), \dots, (6,6)$.
  \item Calcule o \omterm{def:g10:proba:distribution}{probabilidade} que a soma é $7$; que é $12$;
        que é no \omterm{def:g10:functions:extrema}{máximo} $4$.
  \item Qual soma é a mais provável?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:proba:7}
Uma questão de múltipla escolha oferece $4$ respostas, uma correta. Um estudante
respostas $2$ independentes, tais questões são completamente aleatórias. Calcule o
\omterm{def:g10:proba:distribution}{probabilidade} de responder ambos corretamente, exatamente um corretamente, nenhum
corretamente. Verifique se a soma das três \omterm{def:g10:proba:distribution}{probabilidades} é $1$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:proba:8}
Um político afirma $50\%$ aprovação. Em uma enquete de $ n = 900$ aleatoriamente
eleitores escolhidos, $396$ aprovar.
\begin{enumerate}
  \item Calcule o observado freqüência e a flutuação \omterm{def:g10:numbers:interval}{intervalo}
        em volta $ p = 0.5$ para $ n = 900$.
  \item A \omterm{def:g10:algebra:equation}{pesquisa} lança dúvidas sobre a afirmação?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g10:proba:9}
Um jogo custa $2$ para jogar: você joga um dado justo e ganha $6$ se você rolar um
$6$, $3$ se você rolar um $5$, nada de outra forma.
\begin{enumerate}
  \item Calcule o \omterm{def:g10:proba:distribution}{probabilidade} de cada valor ganho ($6$, $3$, $0$).
  \item Sobre $600$ jogos, quantos de cada resultado você espera?
        Calcule o ganho e a perda totais esperados: o jogo vale a pena
        jogando?
\end{enumerate}
\end{exercise}

\section{Problema: As três portas que enganaram mil
matemáticos}

\begin{problem}[{Problema de fim de semana --- o paradoxo de Monty Hall e
seus primos: como a informação muda a probabilidade e como as amostras
separar afirmações de ilusões}]
\label{pb:g10:proba:1}
Em 1990, a colunista Marilyn vos Savant publicou um enigma sobre
um apresentador de game show, três portas, um carro e duas cabras --- e
respondeu corretamente. Dez mil leitores, entre eles centenas
dos doutores em matemática, escreveu para insistir que estava errada. Isto
o problema arma você com \omterm{met:g10:proba:tree}{árvores} (\cref{met:g10:proba:tree}),
\omterm{def:g10:proba:operations}{complementos} e as regras de adição
(\cref{thm:g10:proba:addition}), permite resolver a discussão
você mesmo e termina com a ferramenta que arbitra cada um desses
disputa na prática: a flutuação \omterm{def:g10:proba:sample}{amostra}.

\medskip
\noindent\textbf{Part I --- The rules, warmed up.}
\begin{enumerate}
  \item Um cartão de um padrão $52$- baralho de cartas: calcular
        $ P(\text{king or heart})$ com a regra de adição ---
        e diga por que $\frac{4}{52} + \frac{13}{52}$ sozinho é
        errado.
  \item Três lançamentos justos de moeda: calcular $ P(\text{at least one
        head})$ pelo \omterm{def:g10:proba:operations}{complemento}.
  \item Uma bolsa contém $3$ vermelho e $2$ bolinhas de gude azuis; dois são sorteados
        sem substituição. Com um \omterm{met:g10:proba:tree}{árvore}, calcular
        $ P(\text{two reds})$ e $ P(\text{one of each color})$.
  \item Complete a distribuição da questão 3 com
        $ P(\text{two blues})$ e verifique as três \omterm{def:g10:proba:distribution}{probabilidades}
        soma para $1$.
  \item Alguém relata, por dois \emph{\omterm{def:g10:proba:operations}{incompatível}} \omterm{def:g10:proba:events}{eventos},
        $ P(A) = 0.6$ e $ P(B) = 0.5$. Condene-os com um
        linha da aritmética.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The three doors.}
O jogo: um carro se esconde atrás de uma das três portas, cabras atrás
the other two. Você escolhe uma porta (digamos, porta 1). The host --- who
\emph{sabe} onde está o carro --- abre um dos outros dois
portas, sempre revelando uma cabra, e oferece a oportunidade de
mude para a porta fechada restante.
\begin{enumerate}[resume]
  \item Antes de calcular qualquer coisa: a troca ajuda, prejudica ou
        não importa? Escreva sua resposta instintiva --- honestamente.
  \item Agora o \omterm{met:g10:proba:tree}{árvore}, por casos sobre a verdadeira posição do carro
        (cada $\frac13$): se o carro estiver atrás da sua porta,
        a troca perde; se estiver atrás de qualquer outra porta, o que
        o host deve fazer, e o que a troca ganha?
        Calcular $ P(\text{win by switching})$.
  \item A versão de uma linha: a troca vence exatamente quando o seu
        a primeira escolha estava errada. Conclua novamente.
  \item Agora a variante crucial --- ``Monty Fall'': o anfitrião
        faz \emph{não} sabe onde o carro está, abre um dos
        outras portas aleatoriamente, e por sorte revela uma cabra. Lista
        os seis cenários igualmente prováveis (posição do carro $\times $
        porta aberta), descarte aqueles onde o carro é revelado,
        e calcular o \omterm{def:g10:proba:distribution}{probabilidade} que a mudança vence
        \emph{entre os cenários sobreviventes}. O que mudou e
        por que o conhecimento do anfitrião é importante?
  \item A bomba da intuição: cem portas, você escolhe uma, a
        sabendo que o host abre $98$ portas de cabra. Trocar? Com o que
        \omterm{def:g10:proba:distribution}{probabilidade} de vencer?
  \item Você enfrenta um cético que, como os milhares de escritores de cartas,
        insiste em “50--50”. Descreva uma simulação baseada em cartão
        (três cartas, muitas rodadas, um amigo como anfitrião) e diga
        que proporção de vitórias na troca você espera depois $300$
        rodadas --- com a flutuação \omterm{def:g10:numbers:interval}{intervalo} do capítulo
        como sua margem.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Cousins of the paradox.}
\begin{enumerate}[resume]
  \item Uma família tem dois filhos; você aprende isso \emph{pelo menos
        um é um menino}. Liste os dois filhos igualmente prováveis
        configurações compatíveis com as informações, e
        calcular $ P(\text{two boys})$.
  \item Mesma família, mas a informação agora é: \emph{o mais velho
        criança é um menino}. Recalcular. Por que as duas respostas
        diferem, e o que eles compartilham com os dois Montys?
  \item Três moedas justas: calcular $ P(\text{all three alike})$.
        O matemático d'Alembert defendeu em 1754 outra
        resposta: ``ou são todos iguais ou não são ---
        um caso em dois.'' Nomeie o erro exato, com
        \cref{prop:g10:proba:uniform} na mão.
  \item Um jogo custa $2$ euros: jogue um dado, ganhe $6$ euros em um
        seis, $3$ euros por cinco, nada de outra forma
        (\cref{exo:g10:proba:9}). Sobre $600$ jogos, o que
        ganho médio por jogo a contagem prevê? Um
        seguradora joga a mesma matemática: uma $1\,\%$ anualmente
        chance de um $10\,000$-reivindicação de euro --- qual é o
        \emph{justo} premium, e por que o prêmio real
        ultrapassá-lo?
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- \omterm{def:g10:proba:sample}{Amostras} as arbiters.}
\begin{enumerate}[resume]
  \item Uma moeda é lançada $100$ vezes. Usando a flutuação
        \omterm{def:g10:numbers:interval}{intervalo} $ p \pm \frac{1}{\sqrt n}$, decida qual
        esses resultados devem levantar sobrancelhas: $55$ cabeças; $70$
        cabeças.
  \item Uma enquete de $1\,000$ eleitores dão um candidato $52\,\%$.
        Calcule a flutuação \omterm{def:g10:numbers:interval}{intervalo} em volta $50\,\%$ e
        diga se a enquete \emph{prova} o candidato é
        adiante.
  \item Uma fábrica afirma $4\,\%$ peças defeituosas. UM \omterm{def:g10:proba:sample}{amostra} de
        $400$ mostra $28$ com defeito ($7\,\%$); mais tarde \omterm{def:g10:proba:sample}{amostra} de
        $2\,500$ mostra $175$ ($7\,\%$ de novo). Teste a afirmação
        contra cada um \omterm{def:g10:proba:sample}{amostra}. Por que os veredictos diferem?
  \item Duas fontes de erro assombram todas as \omterm{def:g10:algebra:equation}{pesquisas}: amostragem
        \emph{viés} (o problema de como estão os dados no fim de semana do volume do ensino médio) e amostragem
        \emph{flutuação} (este capítulo). Qual deles encolhe como
        $ n $ cresce, e qual sobrevive a qualquer \omterm{def:g10:proba:sample}{amostra} tamanho? Um
        sentença cada.
  \item Finale --- a lista de verificação do probabilista, uma linha por
        item: resultados igualmente prováveis antes da contagem
        (questão 15); \omterm{met:g10:proba:tree}{árvores} e \omterm{def:g10:proba:operations}{complementos} para vários passos e
        ``pelo menos'' (questões 2-3); \emph{informação
        muda as \omterm{def:g10:proba:distribution}{probabilidades}, e como foi obtido é importante}
        (questões 7 a 14); \omterm{def:g10:proba:sample}{amostras} flutuar em cerca de
        $\frac{1}{\sqrt n}$ e árbitros de simulação
        (questões 11, 16-18).
\end{enumerate}
\end{problem}
