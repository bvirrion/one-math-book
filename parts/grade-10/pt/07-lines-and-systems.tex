\chapter{Equações de Retas e Sistemas Lineares}\label{ch:g10:lines}

Uma linha reta no plano coordenado é descrita por um equaçãoe
duas linhas juntas formam um sistema de equações cujas soluções são suas
\omterm{def:g10:numbers:interunion}{interseção} pontos. Este capítulo conecta três linguagens --- geométrica
(linhas), algébrico (equações) e funcional (funções afins de
\cref{ch:g10:reffunc}) --- e ensina as duas maneiras padrão de resolver um
sistema linear.

\section{A equação de uma reta}

\begin{theorem}[Equação reduzida]\label{thm:g10:lines:reduced}
Em um sistema de \omterm{def:g10:coordgeom:system}{coordenadas}:
\begin{enumerate}
  \item toda linha não vertical tem um equação da forma
        \[
        y = mx + p ,
        \]
        com $m $ (o \emph{recusar}\index{recusar}) e $ p $ (o
        \emph{$ y $-interceptar}) determinado exclusivamente --- este é o
        \emph{alcance reduzido}\index{alcance reduzido} da linha;
  \item toda linha vertical tem um equação $ x = c $;
  \item um ponto está na reta exatamente quando é \omterm{def:g10:coordgeom:system}{coordenadas} satisfazer o
        equação.
\end{enumerate}
\end{theorem}
\admitted

\begin{proposition}[Inclinação através de dois pontos]\label{prop:g10:lines:slope}
O declive da linha (não vertical) que passa $ A(x_A, y_A)$ e
$ B(x_B, y_B)$, com $ x_A \neq x_B $, é
\[
m = \frac{y_B - y_A}{x_B - x_A}
\qquad
\text{(``change in $ y $ over change in $ x $'').}
\]
\end{proposition}

\begin{proof}
Ambos os pontos satisfazem $ y = mx + p $, então $ y_B - y_A = (mx_B + p) - (mx_A + p)
= m(x_B - x_A)$; dividir por $ x_B - x_A \neq 0$.
\end{proof}

\begin{method}[Encontrar a equação de uma reta que passa por dois pontos]\label{met:g10:lines:through}
Dado $ A $ e $ B $ com $ x_A \neq x_B $:
\begin{enumerate}
  \item calcular o declive $ m = \dfrac{y_B - y_A}{x_B - x_A}$;
  \item escrever $ y = mx + p $ e substitua o \omterm{def:g10:coordgeom:system}{coordenadas} de $ A $ (ou $ B $)
        encontrar $ p $;
  \item confira a final equação com o \emph{outro} apontar.
\end{enumerate}
Se $ x_A = x_B $, a linha é vertical: é equação é $ x = x_A $.
\end{method}

\begin{example}\label{ex:g10:lines:through}
Linha através $ A(1, 5)$ e $ B(3, 1)$. Declive:
$ m = \frac{1 - 5}{3 - 1} = \frac{-4}{2} = -2$. Então $ y = -2x + p $; o
ponto $ A $ dá $5 = -2 \times 1 + p $, então $ p = 7$. Equação:
$ y = -2x + 7$. Verifique com $ B $: $-2 \times 3 + 7 = 1$.
\end{example}

\begin{proposition}[Linhas paralelas]\label{prop:g10:lines:parallel}
Duas retas não verticais são paralelas se, e somente se, têm as mesmas
declive. Duas linhas paralelas distintas não têm ponto comum; duas linhas com
diferente encostas tem exatamente um.
\end{proposition}

\begin{proof}
Deixar $ y = mx + p $ e $ y = m'x + p'$ sejam as duas linhas. Um ponto comum
resolve $ mx + p = m'x + p'$, ou seja\ $(m - m')x = p' - p $. Se $ m \neq m'$,
isso tem exatamente uma solução $ x = \frac{p' - p}{m - m'}$: um
\omterm{def:g10:numbers:interunion}{interseção} apontar. Se $ m = m'$, o equação lê $0 = p' - p $: não
solução quando $ p \neq p'$ (paralelos distintos), cada $ x $ quando $ p = p'$
(mesma linha). Paralelismo (não tendo \omterm{def:g10:numbers:interunion}{interseção}, ou sendo igual) é assim
equivalente a $ m = m'$.
\end{proof}

\begin{remark}[Vetor de direção]
A linha $ y = mx + p $ sobe $ m $ unidades para cada unidade à direita, então o
vetor $\vec u\,(1, m)$ é um \emph{direção
vetores}\index{direção
vetores} da linha: a linha é paralela a
$\vec u $. Duas retas são paralelas exatamente quando sua \omterm{prop:g10:lines:parallel}{direção vetores} são
\omterm{def:g10:vectors:collinear}{colinear} (\cref{thm:g10:vectors:det}).
\end{remark}

\section{Sistemas lineares}

\begin{definition}[Sistema linear]\label{def:g10:lines:system}
UM \emph{sistema de duas equações lineares}\index{sistema de duas equações lineares} no
incógnitas $ x $ e $ y $ é um par de equações
\[
\begin{cases}
a x + b y = c \\
a' x + b' y = c'
\end{cases}
\]
estar satisfeito \emph{simultaneamente}. UM \emph{solução} é um par
$(x, y)$ satisfazendo ambos. Cada equação descreve uma linha, então resolvendo o
sistema significa encontrar o \omterm{def:g10:numbers:interunion}{interseção} de duas linhas.
\end{definition}

\begin{method}[Substituição]\label{met:g10:lines:substitution}
\begin{enumerate}
  \item Resolva um equação para um desconhecido --- escolha o mais fácil, por exemplo\
        expressar $ y $ em termos de $ x $;
  \item substitua esta expressão no \emph{outro} equação, que
        agora contém apenas $ x $;
  \item resolver para $ x $, então calcule $ y $ da etapa 1;
  \item verifique o par em ambos originais equações.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g10:lines:substitution}
Resolver $\begin{cases} x + 2y = 8 \\ 3x - y = 3 \end{cases}$ por
substituição. O segundo equação dá $ y = 3x - 3$. Substitua no
primeiro:
\[
x + 2(3x - 3) = 8
\ \Longleftrightarrow\
x + 6x - 6 = 8
\ \Longleftrightarrow\
7x = 14
\ \Longleftrightarrow\
x = 2 ,
\]
então $ y = 3 \times 2 - 3 = 3$. A solução é o par $(2, 3)$.
Verificar: $2 + 2 \times 3 = 8$ e $3 \times 2 - 3 = 3$.
\end{example}

\begin{method}[Eliminação]\label{met:g10:lines:elimination}
\begin{enumerate}
  \item Multiplique cada um equação por um número bem escolhido, de modo que um desconhecido
        obtém coeficientes opostos nos dois equações;
  \item adicione o equações: esse desconhecido desaparece;
  \item resolva o restante desconhecido equação, então substitua de volta por
        encontre o outro desconhecido;
  \item verifique o par em ambos originais equações.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g10:lines:elimination}
Resolver $\begin{cases} 2x + 3y = 7 \\ 5x - 2y = 8 \end{cases}$ por
eliminação. Para eliminar $ y $, multiplique o primeiro equação por $2$ e o
segundo por $3$:
\[
\begin{cases}
4x + 6y = 14 \\
15x - 6y = 24
\end{cases}
\]
Adicionando-os: $19x = 38$, então $ x = 2$. Substituindo em $2x + 3y = 7$:
$4 + 3y = 7$, então $ y = 1$. Solução: $(2, 1)$. Confira no segundo
equação: $5 \times 2 - 2 \times 1 = 8$.
\end{example}

\begin{remark}[Quantas soluções?]
Como duas linhas, uma sistema linear tem exatamente uma solução em geral
(linhas não paralelas) e excepcionalmente nenhuma (linhas paralelas distintas) ou
infinitamente muitos (duas vezes a mesma linha). O critério de
\cref{thm:g10:vectors:det} decide: o sistema tem uma solução única
exatamente quando $ ab' - a'b \neq 0$.
\end{remark}

\begin{example}[Modelagem com um sistema]\label{ex:g10:lines:modeling}
Custo de três cadernos e duas canetas $8$; custa um caderno e quatro canetas
$6$. Deixar $ x $ ser o preço de um notebook e $ y $ o de uma caneta:
\[
\begin{cases}
3x + 2y = 8 \\
x + 4y = 6 .
\end{cases}
\]
A partir do segundo equação, $ x = 6 - 4y $; substituindo,
$3(6 - 4y) + 2y = 8$, então $18 - 10y = 8$, dando $ y = 1$ e então
$ x = 2$. Um notebook custa $2$ e uma caneta custa $1$.
\end{example}

\begin{exercise}[$\star $]\label{exo:g10:lines:1}
Para cada linha, leia o declive e o $ y $-intercepte e esboce-o:
\[
y = 2x - 3, \qquad
y = -\tfrac13 x + 1, \qquad
y = 4, \qquad
x = -2 .
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g10:lines:2}
Será que o ponto $ A(2, 7)$ pertence à linha $ y = 3x + 1$? E $ B(-1, -1)$?
E o ponto $ C(4, 13)$?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g10:lines:3}
Encontre o equação reduzida da linha através de:
\[
\text{(a) } A(0, 2) \text{ and } B(3, 8);
\qquad
\text{(b) } A(-1, 5) \text{ and } B(2, -4);
\qquad
\text{(c) } A(4, 1) \text{ and } B(4, 6).
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g10:lines:4}
Entre as linhas $ y = 3x - 1$, $ y = -3x + 2$, $ y = 3x + 5$,
$ y = \frac{6x + 2}{2}$, que são paralelos entre si? Quais são
realmente igual?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g10:lines:5}
Resolva por substituição:
\[
\begin{cases}
y = 2x - 1 \\
3x + y = 9
\end{cases}
\qquad\text{then}\qquad
\begin{cases}
x - y = 4 \\
2x + 3y = 3 .

\end{exercise}

\begin{exercise}[$\star $]\label{exo:g10:lines:6}

x - y = 4
\end{cases}
\qquad\text{then}\qquad
\begin{cases}
3x + 2y = 12 \\
2x + 5y = 8 .
\end{cases}
\]
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:lines:7}

\[
\begin{cases}
2x - y = 3 \\
-4x + 2y = -6
\end{cases}
\qquad
\begin{cases}

-4x + 2y = 1

\qquad

\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:lines:8}
Foram vendidos duzentos ingressos para uma peça escolar, alguns em $5$ (crianças)
e alguns em $9$ (adultos), num total de $1352$. Quantos ingressos de cada

\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:lines:9}
Deixar $ d $ seja a linha $ y = 2x - 3$ e $ A(1, 4)$.
\begin{enumerate}
  \item Dê o equação da linha $ d'$ paralelo a $ d $ passando
        através $ A $.
  \item Calcule o \omterm{def:g10:numbers:interunion}{interseção} ponto de $ d'$ com o $ x $-eixo.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g10:lines:10}
Deixar $ A(0, 0)$, $ B(6, 2)$ e $ C(2, 4)$.
\begin{enumerate}
  \item Encontre o equações das medianas do triângulo $ ABC $ emitido
        de $ A $ e de $ B $ (uma mediana une um vértice ao \omterm{prop:g10:coordgeom:midpoint}{ponto médio} de
        o lado oposto).
  \item Calcule seus \omterm{def:g10:numbers:interunion}{interseção} apontar $ G $ e verifique isso
        $ G $ também está na terceira mediana. (Você deve descobrir que o
        \omterm{def:g10:coordgeom:system}{coordenadas} de $ G $ são as médias daqueles de $ A $, $ B $, $ C $.)
\end{enumerate}
\end{exercise}

\section{Problema: Uma solução, nenhuma ou infinitas}

\begin{problem}[{Problema de fim de semana --- duas linhas têm três possíveis
destinos, um determinante decide entre eles e o mundo
livro mais antigo já conhecia}]
\label{pb:g10:lines:1}
Dois lineares equações, duas incógnitas: o par de linhas que eles desenham
pode cruzar uma vez, nunca ou ser a mesma linha --- e
\emph{todo} sistema linear herda um desses três destinos.
Este problema os classifica, encontra o número que decide (um
velho conhecido do \omterm{def:g10:vectors:vector}{vetores} capítulo), resolve o
faisões e coelhos de dois mil anos dos chineses
\emph{Nove capítulos}, e termina com cantos de regiões --- o
primeiro passo da otimização usada por todas as companhias aéreas e fábricas
hoje.

\medskip
\noindent\textbf{Part I --- The line, fluently.}
\begin{enumerate}
  \item Encontre o equação reduzida da linha através
        $(1, 2)$ e $(3, 8)$
        (\cref{met:g10:lines:through}).
  \item Entre $ y = 3x - 1$, $ y = 3x + 4$ e $ y = -x + 7$: qual
        duas linhas são paralelas
        (\cref{prop:g10:lines:parallel}) e onde
        pares não paralelos se encontram? (Calcule um \omterm{def:g10:numbers:interunion}{interseção}.)
  \item A linha através $(2, 1)$ e $(2, 5)$: por que tem
        não equação reduzida $ y = mx + p $, e qual é o seu
        equação?
  \item É o ponto $(4, 11)$ na linha $ y = 3x - 1$? E
        $(-1, -5)$?
  \item A linha através $(0, 3)$ com declive $-2$: dê o seu
        equação, suas interceptações de dois eixos e a área do
        triângulo que corta do primeiro quadrante.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Three destinies.}
\begin{enumerate}[resume]
  \item Resolva por substituição
        (\cref{met:g10:lines:substitution}):
        $\begin{cases} y = 2x - 3 \\ 3x + y = 7 \end{cases}$
  \item Resolva por eliminação
        (\cref{met:g10:lines:elimination}):
        $\begin{cases} 2x + 3y = 8 \\ 5x - 3y = -1 \end{cases}$
  \item Classifique --- e interprete com linhas --- os dois
        sistemas
        \[
        \begin{cases} 2x - y = 3 \\ 4x - 2y = 6 \end{cases}
        \qquad\text{and}\qquad
        \begin{cases} 2x - y = 3 \\ 4x - 2y = 1 . \end{cases}
        \]
        Indique a tricotomia completa: cruzada, paralela,
        idêntico --- uma solução, nenhuma, infinitas.
  \item Para o sistema geral $ ax + by = e $, $ cx + dy = f $,
        o número decisivo é $ ad - bc $ --- o mesmo
        determinante como o teste de colinearidade de
        \cref{pb:g10:vectors:1}. Explique a coincidência (que
        dois \omterm{def:g10:vectors:vector}{vetores} são \omterm{def:g10:vectors:collinear}{colinear} exatamente quando as linhas estão
        paralelo ou idêntico?) e avaliar $ ad - bc $ para o
        três sistemas de questões 7 e 8.
  \item Para qual valor de $ m $ faz
        $\begin{cases} x + my = 3 \\ 2x + 6y = 7 \end{cases}$
        não tem solução? Resolva o sistema para todos os outros valores
        de $ m $\,\dots\ ou pelo menos explique como você faria.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The Nine Chapters.}
\begin{enumerate}[resume]
  \item Do \emph{Nove capítulos sobre a arte matemática}
        (China, cerca de 2.000 anos atrás): uma gaiola contém faisões
        e coelhos --- $35$ cabeças e $94$ pernas ao todo. Como
        muitos de cada?
  \item Uma barraca de suco mistura um $60\,\%$- xarope de frutas com
        $20\,\%$- bebida de frutas para obter $30$ eu em $35\,\%$
        fruta. Quantos litros de cada?
  \item Um número de dois dígitos tem soma de dígitos $12$; trocando seu
        dígitos diminui em $18$. Encontre-o, usando o dígito
        álgebra do problema de fim de semana de álgebra de dígitos do volume do ensino médio para configurar o sistema.
  \item Na padaria, $3$ cafés e $2$ custo de croissants
        $9.10$ euros; $2$ cafés e $3$ custo de croissants
        $8.90$ euros. Resolva com elegância: o que fazer \emph{adicionando} o
        dois equações e \emph{subtraindo} cada um deles diz a você
        diretamente?
  \item Uma banca de mercado afirma: $2$ maçãs $+ 1$ pêra para
        $5$ euros, e $4$ maçãs $+ 2$ peras para $9$ euros.
        Resolver --- ou melhor, explicar qual é o destino do sistema
        revela sobre a aritmética da barraca.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Corners rule.}
\begin{enumerate}[resume]
  \item A linha $ y = 3x - 1$ divide o avião em dois. Qual
        lado está a origem? Descreva o conjunto
        $ y > 3x - 1$ e como se testa um ponto contra ele.
  \item Desenhe a região definida pelas quatro restrições
        \[
        x \geq 0, \qquad y \geq 0, \qquad x + y \leq 4,
        \qquad y \leq 2x + 1,
        \]
        e calcule seus quatro pontos de canto.
  \item O lucro de uma oficina é $ P = 3x + 2y $ na região de
        questão 17. Avalie $ P $ nos quatro cantos. Admitindo
        que o \omterm{def:g10:functions:extrema}{máximo} de tal quantidade linear em tal
        região é sempre alcançada em um canto (imagine as linhas
        $3x + 2y = $ varredura constante), fornece o ideal
        plano de produção.
  \item Os dois táxis do problema do fim de semana dos dois termômetros do volume do ensino médio como sistema:
        escreva e resolva
        $\begin{cases} y = 1.5x + 3 \\ y = 2x \end{cases}$ e
        interpretar a solução.
  \item Finale, o dicionário: um equação $\leftrightarrow $
        uma linha; um sistema $\leftrightarrow $ duas linhas com
        três destinos, decididos por $ ad - bc $; muitas desigualdades
        $\leftrightarrow$ uma região cujos cantos carregam o
        ótimo. Declare isso em três frases --- você acabou de
        visitou o hall de entrada do \emph{álgebra linear} e
        \emph{programação linear}, ambos construídos integralmente no
        volumes universitários.
\end{enumerate}
\end{problem}
