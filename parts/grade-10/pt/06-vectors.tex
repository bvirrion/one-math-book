\chapter{Vetores}\label{ch:g10:vectors}

UM vetor codifica um deslocamento: uma direção e um comprimento, independentemente de
o ponto de partida. \omterm{def:g10:vectors:vector}{Vetores} fornecer provas claras de fatos geométricos e, uma vez
\omterm{def:g10:coordgeom:system}{coordenadas} entrar na \omterm{def:g10:functions:function}{imagem}, reduza-os a pequenos cálculos. Eles
será uma ferramenta fundamental \cref{ch:g10:lines} e, mais tarde, para o escalar
produto em \cref{ch:g11:scal}.

\section{Traduções e vetores}

\begin{definition}[Vetor]\label{def:g10:vectors:vector}
Dados dois pontos $A $ e $ B $, o \emph{vetores}\index{vetores} $\vect{AB}$
representa o deslocamento de $ A $ para $ B $: tem um
\emph{direção} (a direção da linha $(AB)$, juntamente com o caminho
de viajar ao longo dele, de $ A $ em direção a $ B $) e um
\emph{comprimento} (a distância $ AB $, também escrito $\norm{\vect{AB}}$).
Dois vetores são \emph{igual} quando eles codificam o mesmo deslocamento:
$\vect{AB} = \vect{CD}$ significa que aplicar os dois deslocamentos a qualquer
ponto dá o mesmo resultado.
\end{definition}

\begin{proposition}[Vetores iguais e paralelogramos]\label{prop:g10:vectors:parallelogram}
$\vect{AB} = \vect{CD}$ exatamente quando $ ABDC $ (nesta ordem!) é um
paralelogramo, possivelmente achatado --- equivalentemente, quando $[AD]$ e
$[BC]$ tem o mesmo \omterm{prop:g10:coordgeom:midpoint}{ponto médio}.
\end{proposition}
\admitted

\begin{notation}
UM vetor geralmente é nomeado por uma única letra, $\vec u $ ou $\vec v $, quando o
pontos finais não importam. O \emph{zero vetor} $\vec 0 = \vect{AA}$ é
o deslocamento que não move nada. O \emph{oposto} de
$\vect{AB}$ é $\vect{BA}$: mesmo comprimento, sentido oposto. Nós escrevemos
$\vect{BA} = -\vect{AB}$.
\end{notation}

\section{Adicionando vetores}

\begin{definition}[Soma de dois vetores]\label{def:g10:vectors:sum}
O \emph{soma}\index{soma} $\vec u + \vec v $ é o deslocamento
obtido realizando $\vec u $ então $\vec v $.
\end{definition}

\begin{theorem}[Relação de Chasles]\label{thm:g10:vectors:chasles}
Para quaisquer três pontos $ A $, $ B $, $ C $:
\[
\vect{AB} + \vect{BC} = \vect{AC}.
\]
\end{theorem}

\begin{proof}
Indo de $ A $ para $ B $, então de $ B $ para $ C $, é um deslocamento que leva
$ A $ para $ C $: é o deslocamento codificado por $\vect{AC}$.
\end{proof}

\begin{remark}[Regra do paralelogramo]
Para adicionar dois \omterm{def:g10:vectors:vector}{vetores} extraído do mesmo ponto, $\vect{AB} + \vect{AC}$,
complete o paralelogramo $ ABDC $: o \omterm{def:g10:vectors:sum}{soma} é a diagonal $\vect{AD}$.
De fato $\vect{AC} = \vect{BD}$, então
$\vect{AB} + \vect{AC} = \vect{AB} + \vect{BD} = \vect{AD}$ por Chasles'
relação.
\end{remark}

\begin{example}[Simplificando com Chasles]\label{ex:g10:vectors:chasles}
Simplificar $\vect{MN} + \vect{NP} + \vect{PQ}$: encadeando os deslocamentos
dá $\vect{MQ}$. Simplificar $\vect{AB} - \vect{AC}$: reescrever o
diferença como um \omterm{def:g10:vectors:sum}{soma} com o oposto vetor,
\[
\vect{AB} - \vect{AC} = \vect{AB} + \vect{CA}
= \vect{CA} + \vect{AB} = \vect{CB}.
\]
\end{example}

\section{Multiplicando um vetor por um número}

\begin{definition}[Múltiplo escalar]\label{def:g10:vectors:scalar}
Deixar $\vec u $ ser diferente de zero vetor e $ k $ um número real. O vetor
$ k\vec u $\index{vetor!múltiplo escalar} tem a mesma direção que
$\vec u $, comprimento $\abs{k} \times \norm{\vec u}$, e aponta da mesma maneira
como $\vec u $ se $ k > 0$, o caminho oposto se $ k < 0$. Para $ k = 0$,
$0\vec u = \vec 0$.
\end{definition}

\begin{definition}[Colinearidade]\label{def:g10:vectors:collinear}
Dois \omterm{def:g10:vectors:vector}{vetores} $\vec u $ e $\vec v $ são
\emph{colinear}\index{colinear} quando eles têm o mesmo
direção, ou seja, quando $\vec v = k \vec u $ por algum real $ k $ (ou um dos
eles são $\vec 0$).
\end{definition}

\begin{remark}
Colinearidade é o vetor linguagem para paralelismo e alinhamento:
\begin{itemize}
  \item as linhas $(AB)$ e $(CD)$ são paralelos exatamente quando $\vect{AB}$
        e $\vect{CD}$ são \omterm{def:g10:vectors:collinear}{colinear};
  \item os pontos $ A $, $ B $, $ C $ estão alinhados exatamente quando $\vect{AB}$ e
        $\vect{AC}$ são \omterm{def:g10:vectors:collinear}{colinear}.
\end{itemize}
\end{remark}

\section{Coordenadas de vetores}

\begin{definition}[Coordenadas de um vetor]\label{def:g10:vectors:coords}
Em um sistema de \omterm{def:g10:coordgeom:system}{coordenadas}, o \emph{\omterm{def:g10:coordgeom:system}{coordenadas}} do vetor $\vect{AB}$
são os números que descrevem o deslocamento:
\[
\vect{AB}\,(x_B - x_A,\ y_B - y_A).
\]
UM vetor $\vec u\,(a, b)$ move cada ponto $ a $ unidades horizontalmente e
$ b $ unidades verticalmente.
\end{definition}

\begin{proposition}[Computando com coordenadas]\label{prop:g10:vectors:coordrules}
Deixar $\vec u\,(a, b)$ e $\vec v\,(c, d)$, e $ k \in \R $.
\begin{enumerate}
  \item $\vec u = \vec v $ exatamente quando $ a = c $ e $ b = d $;
  \item $\vec u + \vec v $ tem \omterm{def:g10:coordgeom:system}{coordenadas} $(a + c,\ b + d)$;
  \item $ k\vec u $ tem \omterm{def:g10:coordgeom:system}{coordenadas} $(ka,\ kb)$;
  \item $\norm{\vec u} = \sqrt{a^2 + b^2}$ (sistema \omterm{def:g10:coordgeom:system}{ortonormal}).
\end{enumerate}
\end{proposition}

\begin{proof}
1--3 reafirmam, coordenada por coordenada, o que os deslocamentos fazem: para
desempenho da instância $\vec u $ então $\vec v $ move um ponto horizontalmente por
$ a $ então $ c $, portanto por $ a + c $ no total. O ponto 4 é a fórmula da distância
de \cref{thm:g10:coordgeom:distance} aplicado a um segmento que representa
$\vec u $.
\end{proof}

\begin{theorem}[Critério de colinearidade]\label{thm:g10:vectors:det}
Dois \omterm{def:g10:vectors:vector}{vetores} $\vec u\,(a, b)$ e $\vec v\,(c, d)$ são \omterm{def:g10:vectors:collinear}{colinear} se e
somente se
\[
ad - bc = 0 .
\]
\end{theorem}

\begin{proof}
Se $\vec v = k\vec u $, então $ c = ka $ e $ d = kb $, então
$ ad - bc = a(kb) - b(ka) = 0$. O mesmo vale se $\vec u = k \vec v $ ou se
ou vetor é zero.

Por outro lado, suponha $ ad - bc = 0$ com $\vec u \neq \vec 0$, dizer
$ a \neq 0$ (o caso $ b \neq 0$ é semelhante). Definir $ k = \frac{c}{a}$; então
$ c = ka $, e $ ad = bc = b(ka)$ dá, depois de dividir por $ a \neq 0$,
$ d = kb $. Então $\vec v = k\vec u $.
\end{proof}

\begin{example}\label{ex:g10:vectors:det}
São $\vec u\,(3, -2)$ e $\vec v\,(-6, 4)$ \omterm{def:g10:vectors:collinear}{colinear}? Calcular
$ ad - bc = 3 \times 4 - (-2)\times(-6) = 12 - 12 = 0$: sim, e de fato
$\vec v = -2\vec u $. Para $\vec u\,(3, -2)$ e $\vec w\,(1, 5)$:
$3 \times 5 - (-2) \times 1 = 17 \neq 0$: não \omterm{def:g10:vectors:collinear}{colinear}.
\end{example}

\begin{method}[Alinhamento e paralelismo]\label{met:g10:vectors:alignment}
Para provar que três pontos $ A $, $ B $, $ C $ estão alinhados:
\begin{enumerate}
  \item calcular o \omterm{def:g10:coordgeom:system}{coordenadas} de $\vect{AB}$ e $\vect{AC}$;
  \item verifique o critério $ ad - bc = 0$ de \cref{thm:g10:vectors:det};
  \item concluir: o \omterm{def:g10:vectors:vector}{vetores} são \omterm{def:g10:vectors:collinear}{colinear} e compartilhe o ponto $ A $, então
        os três pontos estão alinhados.
\end{enumerate}
O mesmo cálculo com $\vect{AB}$ e $\vect{CD}$ prova que o
linhas $(AB)$ e $(CD)$ são paralelos.
\end{method}

\begin{example}\label{ex:g10:vectors:alignment}
Deixar $ A(1, 2)$, $ B(3, 5)$ e $ C(7, 11)$. Então $\vect{AB}\,(2, 3)$ e
$\vect{AC}\,(6, 9)$, e $2 \times 9 - 3 \times 6 = 18 - 18 = 0$: o
os pontos estão alinhados (na verdade $\vect{AC} = 3\vect{AB}$).
\end{example}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g10:vectors:1}
Simplifique usando a relação de Chasles:
\[
\vect{AB} + \vect{BD}, \qquad
\vect{KL} + \vect{LM} + \vect{MK}, \qquad
\vect{AC} - \vect{BC}, \qquad
\vect{AB} + \vect{CA}.
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g10:vectors:2}
Deixar $ A(2, 1)$, $ B(5, 3)$, $ C(-1, 4)$. Calcule o \omterm{def:g10:coordgeom:system}{coordenadas} de
$\vect{AB}$, $\vect{BC}$, $\vect{AC}$, e verifique no \omterm{def:g10:coordgeom:system}{coordenadas} que
$\vect{AB} + \vect{BC} = \vect{AC}$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g10:vectors:3}
Deixar $\vec u\,(2, -3)$ e $\vec v\,(-1, 4)$. Dê o \omterm{def:g10:coordgeom:system}{coordenadas} de
$\vec u + \vec v $, $3\vec u $, $2\vec u - \vec v $ e calcular
$\norm{\vec u}$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g10:vectors:4}
Em cada caso, diga se $\vec u $ e $\vec v $ são \omterm{def:g10:vectors:collinear}{colinear}:
\[
\text{(a) } \vec u\,(4, 6),\ \vec v\,(6, 9);
\qquad
\text{(b) } \vec u\,(2, -5),\ \vec v\,(-4, 10);
\qquad
\text{(c) } \vec u\,(3, 1),\ \vec v\,(1, 3).
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g10:vectors:5}
Deixar $ A(0, 2)$, $ B(4, 0)$, $ C(6, 3)$. Encontre o \omterm{def:g10:coordgeom:system}{coordenadas} do ponto
$ D $ tal que $ ABCD $ é um paralelogramo, ou seja, tal que
$\vect{AB} = \vect{DC}$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:vectors:6}
Deixar $ A(-2, 1)$, $ B(1, 3)$, $ C(4, -1)$.
\begin{enumerate}
  \item Calcule o \omterm{def:g10:coordgeom:system}{coordenadas} do ponto $ M $ tal que
        $\vect{AM} = \vect{AB} + \vect{AC}$.
  \item Qual é o quadrilátero $ ABMC $? Justificar.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:vectors:7}
Deixar $ A(1, -1)$, $ B(4, 1)$, $ C(10, 5)$. São os pontos $ A $, $ B $, $ C $
alinhado? A mesma pergunta para $ A(0, 3)$, $ B(2, 2)$, $ C(8, -1)$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:vectors:8}
Deixar $ A(1, 2)$, $ B(5, 4)$, $ C(6, 1)$, $ D(2, -1)$. Mostre isso $(AB)$ e
$(CD)$ são paralelos, então isso $ ABCD $ é um paralelogramo. Qual
verificação implica o outro?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:vectors:9}
Deixar $ ABC $ seja um triângulo e seja $ I $ seja o ponto definido por
$\vect{AI} = \frac23 \vect{AB}$. Com $ A(0,0)$, $ B(6, 3)$, $ C(2, 5)$
(para que os cálculos permaneçam simples):
\begin{enumerate}
  \item calcular o \omterm{def:g10:coordgeom:system}{coordenadas} de $ I $;
  \item deixar $ J $ seja tal que $\vect{CJ} = \frac23\vect{CB}$; calcular o
        \omterm{def:g10:coordgeom:system}{coordenadas} de $ J $;
  \item mostre isso $(IJ)$ é paralelo a $(AC)$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g10:vectors:10}
Deixar $ ABCD $ seja qualquer quadrilátero, e seja $ I $, $ J $, $ K $, $ L $ seja o
pontos médios de $[AB]$, $[BC]$, $[CD]$, $[DA]$. Usando \omterm{def:g10:coordgeom:system}{coordenadas}
$ A(x_A, y_A)$, etc., mostre que $\vect{IJ} = \vect{LK}$, e concluir que
o pontos médios dos lados de \emph{qualquer} quadrilátero forma um
paralelogramo.
\end{exercise}

\section{Problema: A álgebra das setas}

\begin{problem}[{Problema de fim de semana --- Ginástica Chasles, o
centróide re-provado em duas linhas e forças em equilíbrio: o que
vetores são realmente para}]
\label{pb:g10:vectors:1}
A 8ª série provou que as medianas de um triângulo se encontram em dois terços
de seu comprimento --- com um paralelogramo engenhoso escondido dentro
o triângulo (o problema do fim de semana centróide do volume do Ensino Médio). \omterm{def:g10:vectors:vector}{Vetores} reprová-lo em
duas linhas e distribua a simultaneidade gratuitamente. Isso é o que
esta nova álgebra serve para: teoremas tornam-se cálculos com
setas. Este problema treina a computação (Chasles acima de tudo),
entrega a prova de duas linhas, estende Varignon
(\cref{exo:g10:vectors:10}) e termina onde \omterm{def:g10:vectors:vector}{vetores} nasceram:
forças e velocidades.

\medskip
\noindent\textbf{Part I --- Chasles gymnastics.}
\begin{enumerate}
  \item Simplifique usando a relação de Chasles
        (\cref{thm:g10:vectors:chasles}):
        $\vect{AB} + \vect{BC} + \vect{CD}$; \quad
        $\vect{MA} - \vect{MB}$; \quad
        $\vect{AB} + \vect{CD} + \vect{BC} + \vect{DA}$.
  \item O truque da origem: mostre que para \emph{qualquer} apontar $ O $,
        $\vect{AB} = \vect{OB} - \vect{OA}$.
  \item Prove as duas caracterizações do \omterm{prop:g10:coordgeom:midpoint}{ponto médio} $ I $ de
        $[AB]$:
        \[
        \vect{IA} + \vect{IB} = \vec 0
        \qquad\text{and}\qquad
        \vect{OI} = \tfrac12\left(\vect{OA} + \vect{OB}\right)
        \ \text{for any } O .
        \]
  \item Usando $\vect{AB} = \vect{DC}$, encontre novamente o quarto vértice
        $ D $ do paralelogramo $ ABCD $ com $ A(-1, 2)$,
        $ B(3, 4)$, $ C(6, 0)$ --- e compare com o
        resposta do truque do \omterm{prop:g10:coordgeom:midpoint}{ponto médio} de \cref{pb:g10:coordgeom:1}.
  \item Adicionando um fixo vetor $\vec u $ para todos os pontos do
        plano executa qual transformação --- aquela
        descoberto com dois espelhos no problema de fim de semana de dois espelhos do volume do Ensino Médio
        e duas meias voltas no problema de fim de semana de meia volta do volume do ensino médio? Calcular
        o \omterm{def:g10:functions:function}{imagem} de $(1, 2)$ sob $\vec u = (3, -1)$.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The centroid, third proof.}
Defina o ponto $ G $ de um triângulo $ ABC $ pelo elegante
condição
\[
\vect{GA} + \vect{GB} + \vect{GC} = \vec 0 .
\]
\begin{enumerate}[resume]
  \item Usando o truque da origem, mostre que esta condição
        determina $ G $ exclusivamente:
        $\vect{OG} = \frac13\left(\vect{OA} + \vect{OB} +
        \vect{OC}\right)$ para qualquer $ O $ --- então o \omterm{def:g10:coordgeom:system}{coordenadas} de
        $ G $ são os \emph{médias} dos vértices
        (o problema do fim de semana centróide do volume do Ensino Médio viu isso numericamente).
  \item Provar $\vect{AG} = \frac13\left(\vect{AB} +
        \vect{AC}\right)$, e, escrevendo $ K $ para o \omterm{prop:g10:coordgeom:midpoint}{ponto médio} de
        $[BC]$, $\vect{AK} = \frac12\left(\vect{AB} +
        \vect{AC}\right)$. Concluir:
        $\vect{AG} = \frac23\,\vect{AK}$ --- os dois terços
        teorema, reprovado.
  \item Por que as medianas de $ B $ e de $ C $ passar
        o \emph{mesmo} $ G $ sem mais cálculos?
        Compare as três provas disponíveis deste teorema ---
        paralelogramo oculto da 8ª série, \omterm{def:g10:coordgeom:system}{coordenadas}, \omterm{def:g10:vectors:vector}{vetores} ---
        em uma frase cada.
  \item Verificação numérica: $ A(1, 1)$, $ B(5, 2)$, $ C(3, 6)$.
        Calcular $ G $, então $ K $, então verifique
        $\vect{AG} = \frac23 \vect{AK}$ com a colinearidade
        critério (\cref{thm:g10:vectors:det}).
  \item Leituras de física $ G $ como o ponto de equilíbrio de três iguais
        massas nos vértices. Dobre a massa em $ A $ (massas
        $2, 1, 1$): o ponto de equilíbrio torna-se
        $\vect{OG'} = \frac{2\vect{OA} + \vect{OB} +
        \vect{OC}}{4}$. Calcular $ G'$ para o triângulo de
        questão 9 e compare sua posição com $ G $.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Collinearity at work.}
\begin{enumerate}[resume]
  \item São $\vec u(3, -2)$ e $\vec v(-6, 4)$ \omterm{def:g10:vectors:collinear}{colinear}? E
        $\vec w(2, 5)$, $\vec z(4, 9)$? Decida com o
        critério determinante.
  \item São os pontos $ A(1, 2)$, $ B(3, 5)$, $ C(7, 11)$
        alinhado?
  \item Além de Varignon (\cref{exo:g10:vectors:10}): em qualquer
        quadrilátero $ ABCD $, o \emph{bimedianos} junte-se ao
        pontos médios de lados opostos ($[IK]$ e $[JL]$). Usando
        posição \omterm{def:g10:vectors:vector}{vetores} ($\vect{OI} = \frac12(\vect{OA} +
        \vect{OB})$, etc.), mostram que ambos os bimedianos têm o
        mesmo \omterm{prop:g10:coordgeom:midpoint}{ponto médio}, de posição vetor
        $\frac14\left(\vect{OA} + \vect{OB} + \vect{OC} +
        \vect{OD}\right)$: o centróide do quadrilátero.
  \item Tales em uma linha: $ M $ e $ N $ são os pontos definidos
        por $\vect{AM} = \frac13\vect{AB}$ e
        $\vect{AN} = \frac13\vect{AC}$. Calcular
        $\vect{MN}$ em termos de $\vect{BC}$ e concluir
        paralelismo e razão.
  \item Generalize a questão 14 para uma proporção arbitrária $ k $ e
        descreva em uma frase como vetor cálculo absorve
        o \omterm{prop:g10:coordgeom:midpoint}{ponto médio} teorema (o \omterm{prop:g10:coordgeom:midpoint}{ponto médio} teorema)
        e Tales (teorema de Tales) de uma só vez.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Forces and velocities.}
\begin{enumerate}[resume]
  \item Um barco é empurrado por uma corrente de $3$ km/h para leste
        enquanto remava em $4$ km/h para norte. Dê a resultante
        velocidade vetor, sua magnitude e descrever o real
        claro. (Um triplo muito antigo aparece.)
  \item Três forças atuam em um anel: $\vec u(2, 1)$,
        $\vec v(-3, 2)$, $\vec w(1, -3)$. Calcule a resultante.
        O que o anel faz? E por que é que \omterm{def:g10:vectors:sum}{soma} das flechas
        ao longo de qualquer \emph{fechado} polígono sempre $\vec 0$
        (Chasles)?
  \item Duas forças $\vec u(4, -1)$ e $\vec v(-1, 3)$ agir em um
        gancho. Qual terceira força mantém o gancho em equilíbrio?
  \item Um nadador atravessa um rio em linha reta em
        $2$ EM; a corrente corre em $1.5$ EM. Dê o
        velocidade resultante e sua magnitude. O rio é
        $50$ m de largura: quanto tempo leva a travessia e quanto
        bem a jusante o nadador pousa?
  \item Final: um objeto, três fantasias --- deslocamento
        seta, par de \omterm{def:g10:coordgeom:system}{coordenadas}, força ou velocidade. Liste o
        três teoremas clássicos que este problema foi reprovado com
        vetor álgebra, e nomeie a única quantidade geométrica como
        \omterm{def:g10:vectors:vector}{vetores} deste capítulo ainda não consegue medir --- o
        ferramenta que corrige isso chega com o produto escalar, em
        11ª série.
\end{enumerate}
