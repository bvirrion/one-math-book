\chapter{Números e conjuntos de números}\label{ch:g10:numbers}

A matemática começa com números, e nem todos os números são do mesmo tipo:
contando números, números negativos, frações e números como $\sqrt 2$
ou $\pi $ que nenhuma fração pode expressar. Este capítulo os organiza em
famílias aninhadas, apresenta \omterm{def:g10:numbers:interval}{intervalos} para descrever partes do número
linha e usa o \omterm{def:g10:numbers:abs}{valor absoluto} para medir distâncias entre números.

\section{As famílias de números}

\begin{definition}[Conjuntos de números]\label{def:g10:numbers:sets}
\begin{itemize}
  \item $\N $ é o conjunto de \emph{números naturais}\index{números naturais}:
        $0, 1, 2, 3, \dots $
  \item $\Z $ é o conjunto de \emph{inteiros}\index{inteiros}: $\dots, -2, -1,
        0, 1, 2, \dots $
  \item $\Q $ é o conjunto de \emph{números racionais}\index{números racionais}:
        todos os quocientes $\frac{p}{q}$ com $ p \in \Z $, $ q \in \N $ e
        $ q \neq 0$.
  \item $\R $ é o conjunto de \emph{números reais}\index{números reais}: todos os
        números que podem ser colocados na reta numérica.
\end{itemize}
\end{definition}

\begin{notation}
O símbolo $\in $ lê ``pertence a'': $3 \in \N $, $-5 \in \Z $,
$\frac23 \in \Q $. O símbolo $\subset $ lê ``está incluído em'': cada
número natural é um inteiro, todo inteiro é um número racional
(por exemplo\ $-5 = \frac{-5}{1}$) e cada número racional é real, então
\[
\N \subset \Z \subset \Q \subset \R .
\]
Uma barra nega um símbolo: $\frac12 \notin \Z $.
\end{notation}

\begin{example}\label{ex:g10:numbers:classify}
Coloquemos alguns números na menor família que os contém.
\begin{itemize}
  \item $\frac{28}{4} = 7$, então $\frac{28}{4} \in \N $ mesmo que seja
        escrito como uma fração: sempre simplifique primeiro.
  \item $-3.5 = -\frac{35}{10} = -\frac72$ é racional, mas não inteiro.
  \item $0.333\dots $ (o dígito $3$ repetindo para sempre) é igual $\frac13$,
        um número racional.
  \item $\sqrt 2$ e $\pi $ são reais, mas não racionais, como estamos prestes a
        veja por $\sqrt 2$; esses números são chamados
        \emph{irracional}\index{irracional}.
\end{itemize}
\end{example}

\begin{theorem}[Irracionalidade de $\sqrt 2$]\label{thm:g10:numbers:sqrt2}
O número $\sqrt 2$ não é racional: nenhuma fração de \omterm{def:g10:numbers:sets}{inteiros} tem quadrado
$2$.
\end{theorem}

\begin{proof}
Raciocinamos por contradição, em pequenos passos.
\begin{enumerate}
  \item Suponha $\sqrt 2 = \frac{p}{q}$ onde $ p $ e $ q $ são positivos
        \omterm{def:g10:numbers:sets}{inteiros}, e a fração é totalmente simplificada, então $ p $ e $ q $ são
        nem ambos.
  \item Quadratura de ambos os lados dá $2 = \frac{p^2}{q^2}$, aquilo é
        $ p^2 = 2q^2$. Então $ p^2$ é par.
  \item Se $ p $ eram estranhos, digamos $ p = 2k+1$, então
        $ p^2 = 4k^2 + 4k + 1$ seria estranho. Desde $ p^2$ é par, $ p $ deve
        seja par: $ p = 2k $ para alguns inteiro $ k $.
  \item Substituindo: $(2k)^2 = 2q^2$, então $4k^2 = 2q^2$, então
        $ q^2 = 2k^2$. Pelo mesmo argumento do passo 3, $ q $ é par.
  \item Agora $ p $ e $ q $ são ambos pares, contradizendo o passo 1. O
        suposição era impossível: $\sqrt 2$ é \omterm{ex:g10:numbers:classify}{irracional}.
\end{enumerate}
\end{proof}

\begin{remark}
\omterm{def:g10:numbers:sets}{Números racionais} são exatamente os números cuja expansão decimal
pára (como $\frac72 = 3.5$) ou eventualmente repete o mesmo bloco para sempre
(como $\frac13 = 0.333\dots $ ou $\frac{1}{7} = 0.142857\,142857\dots $).
Números irracionais como $\sqrt2 = 1.41421356\dots $ tem dígitos decimais
que nunca caem em um padrão repetitivo. Admitimos esta caracterização
neste nível.
\end{remark}

\section{A reta numérica e os intervalos}

\omterm{def:g10:numbers:sets}{Números reais} preencher uma linha: escolhendo uma origem $0$ e uma unidade de comprimento,
cada número real corresponde exatamente a um ponto.

\begin{definition}[Intervalo]\label{def:g10:numbers:interval}
Deixar $ a $ e $ b $ ser \omterm{def:g10:numbers:sets}{números reais} com $ a < b $. Um
\emph{intervalo}\index{intervalo} é o conjunto de todos \omterm{def:g10:numbers:sets}{números reais} entre dois
limites. Um colchete significa que o limite está incluído, um parêntese significa
está excluído:
\begin{center}
\begin{tabular}{c|c}
notation & description \\
\hline
$\intcc{a}{b}$ & $ a \leq x \leq b $ (both ends included) \\
$\intoo{a}{b}$ & $ a < x < b $ (both ends excluded) \\
$\intco{a}{b}$ & $ a \leq x < b $ \\
$\intoc{a}{b}$ & $ a < x \leq b $ \\
$\intco{a}{+\infty}$ & $ x \geq a $ \\
$\intoo{-\infty}{b}$ & $ x < b $ \\
\end{tabular}
\end{center}
Os símbolos $-\infty $ e $+\infty $ (``infinito'') não são números, apenas um
maneira de dizer que o intervalo continua para sempre; o colchete ao lado
eles são sempre um parêntese. Toda a linha $\R $ é o intervalo
$\intoo{-\infty}{+\infty}$.
\end{definition}

\begin{definition}[Intersecção e união]\label{def:g10:numbers:interunion}
Deixar $ I $ e $ J $ ser dois conjuntos de \omterm{def:g10:numbers:sets}{números reais}.
O \emph{interseção}\index{interseção} $ I \cap J $ (``$ I $ e $ J $'')
é o conjunto de números pertencentes a ambos; o
\emph{união}\index{união} $ I \cup J $ (``$ I $ ou $ J $'') é o conjunto de
números pertencentes a pelo menos um deles.
\end{definition}

\begin{example}\label{ex:g10:numbers:interunion}
Pegar $ I = \intcc{-1}{3}$ e $ J = \intoo{1}{5}$. Desenhe os dois no mesmo
linha: eles se sobrepõem entre $1$ e $3$. Portanto
\[
I \cap J = \intoc{1}{3},
\qquad
I \cup J = \intco{-1}{5}.
\]
Observe os colchetes: $1 \notin J $ então $1 \notin I \cap J $, mas $3 \in I $ e
$3 \in J $, então $3 \in I \cap J $.
\end{example}

\begin{method}[Trabalhando com intervalos]\label{met:g10:numbers:intervals}
Para encontrar o \omterm{def:g10:numbers:interunion}{interseção} ou \omterm{def:g10:numbers:interunion}{união} de dois \omterm{def:g10:numbers:interval}{intervalos}:
\begin{enumerate}
  \item desenhe a reta numérica e marque os quatro limites;
  \item sombreie o primeiro \omterm{def:g10:numbers:interval}{intervalo} acima da linha e o segundo abaixo dela;
  \item o \omterm{def:g10:numbers:interunion}{interseção} é onde as sombras se sobrepõem, o \omterm{def:g10:numbers:interunion}{união} é onde
        pelo menos um sombreamento está presente;
  \item decida cada colchete testando se o próprio limite pertence a
        ambos os conjuntos (\omterm{def:g10:numbers:interunion}{interseção}) ou para pelo menos um (\omterm{def:g10:numbers:interunion}{união}).
\end{enumerate}
\end{method}

\section{Valor absoluto e distância}

\begin{definition}[Valor absoluto]\label{def:g10:numbers:abs}
O \emph{valor absoluto}\index{valor absoluto} de um número real $ x $ é
\[
\abs{x} =
\begin{cases}
x & \text{if } x \geq 0, \\
-x & \text{if } x < 0.
\end{cases}
\]
Por exemplo $\abs{7} = 7$ e $\abs{-4} = 4$. Em todos os casos
$\abs{x} \geq 0$.
\end{definition}

\begin{proposition}[Distância na linha]\label{prop:g10:numbers:distance}
Para todos \omterm{def:g10:numbers:sets}{números reais} $ a $ e $ b $, a distância entre os pontos $ a $ e
$ b $ na reta numérica é $\abs{b - a}$. Em particular $\abs{x}$ é o
distância de $ x $ para $0$.
\end{proposition}

\begin{proof}
Se $ b \geq a $, a distância de $ a $ para $ b $ é $ b - a \geq 0$, que
é igual $\abs{b-a}$. Se $ b < a $, a distância é $ a - b = -(b - a) > 0$,
o que é novamente $\abs{b - a}$ por definição do \omterm{def:g10:numbers:abs}{valor absoluto}.
\end{proof}

\begin{proposition}[Valor absoluto e intervalos]\label{prop:g10:numbers:absinterval}
Deixar $ a $ seja um número real e $ r > 0$. Então
\[
\abs{x - a} \leq r
\quad\text{exactly when}\quad
x \in \intcc{a - r}{a + r}.
\]
\end{proposition}

\begin{proof}
$\abs{x-a} \leq r $ diz que a distância $ x $ para $ a $ é no máximo $ r $,
ou seja, isso $ x $ não fica além $ r $ de $ a $ em ambos os lados. O
números que satisfazem isso são exatamente aqueles entre $ a - r $ e $ a + r $,
limites incluídos.
\end{proof}

\begin{example}\label{ex:g10:numbers:abs}
Resolver $\abs{x - 3} \leq 2$. As soluções são os números à distância em
mais $2$ de $3$: o \omterm{def:g10:numbers:interval}{intervalo} $\intcc{1}{5}$. Por outro lado, o \omterm{def:g10:numbers:interval}{intervalo}
$\intcc{-1}{7}$ tem centro $\frac{-1+7}{2} = 3$ e raio
$\frac{7-(-1)}{2} = 4$, então é descrito por $\abs{x - 3} \leq 4$.
\end{example}

\section{Aproximações}

Números irracionais, e mesmo a maioria das frações, não podem ser escritas exatamente
com um número finito de dígitos decimais, então, na prática, nós os aproximamos.

\begin{definition}[Aproximação para uma determinada precisão]\label{def:g10:numbers:approx}
Um número $ d $ é um \emph{aproximação}\index{aproximação} de $ x $ para
dentro $10^{-n}$ quando $\abs{x - d} \leq 10^{-n}$. Truncar ou arredondar
a expansão decimal após o $ n $-ésimo dígito ambos fornecem tal
aproximações.
\end{definition}

\begin{example}
De $\pi = 3.14159\,26\dots $: o truncamento $3.141$ e o arredondamento
$3.142$ são ambos aproximações de $\pi $ para dentro $10^{-3}$. O arredondamento
está a uma distância menor que $\frac12 \times 10^{-3}$, o truncamento apenas
garantias $10^{-3}$. Escrita $3.141 \leq \pi \leq 3.142$ \emph{quadros}
$\pi $ entre dois limites decimais.
\end{example}

\begin{exercise}[$\star $]\label{exo:g10:numbers:3}
Calcular $ I \cap J $ e $ I \cup J $ para

\text{(a) } I = \intcc{-3}{2},\ J = \intco{0}{4};

\[
\abs{-6}, \qquad \abs{4 - 9}, \qquad \abs{-3 - 5}, \qquad

\end{exercise}

\begin{exercise}[$\star $]\label{exo:g10:numbers:5}

\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:numbers:6}

\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:numbers:7}

\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:numbers:8}
Verdadeiro ou falso? Justifique cada resposta com um argumento ou contra-exemplo.
\begin{enumerate}
  \item A soma de dois \omterm{def:g10:numbers:sets}{inteiros} é um inteiro.
  \item O quociente de dois \omterm{def:g10:numbers:sets}{inteiros} é um inteiro.
  \item A soma de dois \omterm{def:g10:numbers:sets}{números racionais} é racional.
  \item A soma de um número racional e um número \omterm{ex:g10:numbers:classify}{irracional} é

\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:numbers:9}

\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g10:numbers:10}
Adapte a prova de \cref{thm:g10:numbers:sqrt2} para mostrar isso $\sqrt 3$ é

\end{exercise}

\section{Problema: Entre quaisquer dois números}

\begin{problem}[{Problema de fim de semana --- racionais e irracionais
entrelaçado: cada intervalo, por menor que seja, contém infinitos
de cada um, e nenhuma medida pode diferenciá-los}]
\label{pb:g10:numbers:1}
Os racionais parecem uma multidão (todas as frações!) e os
irracionais como exceções exóticas ($\sqrt2$, $\pi $). Este problema
revela a verdadeira imagem: as duas famílias \emph{entrelaçar} então
finamente que cada \omterm{def:g10:numbers:interval}{intervalo} da reta numérica, no entanto
microscópico, contém infinitamente muitos de cada --- com estranhos
consequências, como esta: nenhuma medição física, no entanto
preciso, pode decidir se um comprimento é racional.

\medskip
\noindent\textbf{Part I --- The four kingdoms.}
\begin{enumerate}
  \item Para cada número, nomeie o menor dos conjuntos $\N $,
        $\Z $, $\Q $, $\R $ contendo-o
        (\cref{def:g10:numbers:sets}): $-7$; \ $\frac{13}{4}$; \
        $\sqrt{16}$; \ $0.121212\ldots $ (lembre-se
        o problema de fim de semana de decimais repetidos do volume do Ensino Médio); \ $\sqrt8$; \ $\pi $ (admita seu
        irracionalidade --- provada apenas na universidade
        volumes).
  \item Prove isso $\Q $ é estável sob adição e
        multiplicação: se $ x = \frac pq $ e $ y = \frac rs $ são
        racional, escreva $ x + y $ e $ xy $ como frações simples.
  \item Deduza por contradição: (a) a soma de um racional e de um
        \omterm{ex:g10:numbers:classify}{irracional} é \omterm{ex:g10:numbers:classify}{irracional}; (b) o produto de um
        \emph{diferente de zero} racional e um \omterm{ex:g10:numbers:classify}{irracional} é \omterm{ex:g10:numbers:classify}{irracional}.
  \item Mostre que o conjunto de irracionais é estável sob
        \emph{nenhum} operação: exposição dois irracionais cujo
        soma é racional e dois cujo produto é racional.
  \item Localizar $\sqrt2 + \sqrt8$ e $\sqrt2 \times \sqrt8$ em
        os quatro reinos (simplificar $\sqrt8$ primeiro,
        o método do volume do Ensino Médio).
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The rationals are dense.}
\begin{enumerate}[resume]
  \item Encontre um racional estritamente entre $3.47$ e $3.48$; então
        um segundo; em seguida, descreva como produzir tantos quantos
        desejado (o zoom do problema do fim de semana sem próximo número do volume do Ensino Médio, agora com um
        prova à vista).
  \item O teorema geral. Deixar $ a < b $ ser quaisquer dois reais, com
        lacuna $ g = b - a > 0$. Escolher $ n $ com
        $10^{-n} < g $, e considere os múltiplos de $10^{-n}$
        (a grade decimal da etapa $10^{-n}$). Explique por que em
        pelo menos um ponto da grade fica estritamente entre $ a $ e $ b $,
        e concluo: \emph{todo \omterm{def:g10:numbers:interval}{intervalo} de comprimento positivo
        contém um número racional}.
  \item Atualize a conclusão: cada um desses \omterm{def:g10:numbers:interval}{intervalo} contém
        \emph{infinitamente muitos} racionais. (Aplique a questão 7
        novamente, dentro de um menor \omterm{def:g10:numbers:interval}{intervalo}.)
  \item Agora o irracionais: dado $ a < b $, tome um raciocínio $ r $
        estritamente dentro (questão 7) e considere os números
        $ r + \frac{\sqrt2}{10^k}$. Usando a questão 3, mostre que eles
        são \omterm{ex:g10:numbers:classify}{irracional}, e isso para $ k $ grandes o suficiente, eles ainda
        deitar no \omterm{def:g10:numbers:interval}{intervalo}: \emph{todo \omterm{def:g10:numbers:interval}{intervalo} também contém
        infinitamente muitos irracionais}.
  \item Dois enigmas clássicos resolvidos: existe um menor
        positivo número real? UM número real ``logo depois
        $3$''? Responda ambos com o teorema da questão 7, e
        saudar a versão infantil
        (o problema do fim de semana sem próximo número do volume do Ensino Médio).
\end{enumerate}

\medskip
\noindent\textbf{Part III --- \omterm{def:g10:numbers:abs}{Valor absoluto}, the geometry of
$\R $.}
\begin{enumerate}[resume]
  \item Resolva e expresse as soluções como \omterm{def:g10:numbers:interval}{intervalos} ou sindicatos
        (\cref{prop:g10:numbers:absinterval}):
        $\abs{x - 5} = 2$; \quad $\abs{x - 5} < 2$; \quad
        $\abs{x + 1} \geq 3$.
  \item Um pistão deve ser fabricado para $80$ mm dentro de um
        tolerância de $0.05$ mm. Escreva o requisito com um
        \omterm{def:g10:numbers:abs}{valor absoluto}, então como um \omterm{def:g10:numbers:interval}{intervalo}. Medida de dois pistões
        $79.97$ e $80.06$ mm: veredictos?
  \item A desigualdade triangular na reta:
        $\abs{a + b} \leq \abs a + \abs b $. Confira
        $(a, b) = (3, -5)$ e $(-2, -7)$, prove quando $ a $ e
        $ b $ têm o mesmo sinal e quando têm sinais opostos
        sinais e indicar exatamente quando a igualdade é válida.
  \item Resolver $\abs{x - 2} = \abs{x + 4}$ lendo-o como um
        igualdade de distâncias na linha. Qual ponto de
        o problema de fim de semana de dois espelhos do volume do ensino médio que você acabou de calcular, um
        dimensão para baixo?
  \item Simplificar $\sqrt{x^2}$ --- com cuidado. Teste sua fórmula
        ligado $ x = 3$ e $ x = -3$, então use-o para resolver
        $ x^2 < 9$ com um \omterm{def:g10:numbers:abs}{valor absoluto}.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- What no measurement can decide.}
\begin{enumerate}[resume]
  \item Um físico mede uma vara: $1.41421 \pm 0.00001$ m.
        Pode qualquer medição desse tipo --- esta ou uma futura, mais
        preciso --- sempre \emph{provar} que o comprimento da haste
        é \omterm{ex:g10:numbers:classify}{irracional}? Ou que é racional? Use as perguntas 7
        e 9 na tolerância intervaloe explique por que apenas
        \emph{prova} (quanto à diagonal, o problema de irracionalidade do fim de semana do volume do Ensino Médio)
        pode resolver a irracionalidade.
  \item Os truncamentos $1.4$, $1.41$, $1.414$, $1.4142,\dots $
        de $\sqrt2$ são todos racionais. O que eles mostram sobre
        quão perto $\Q $ pressiona contra todos \omterm{ex:g10:numbers:classify}{irracional}?
        Formule o fato geral.
  \item \omterm{def:g10:numbers:interval}{Intervalo} aritmética: valores medidos $ a = 2.5 \pm 0.1$
        e $ b = 1.2 \pm 0.1$. Suporte $ a + b $ e $ a \times b $
        entre certos limites. Qual operação degrada o
        precisão mais?
  \item A regra de um engenheiro diz que erros independentes
        medidas ``adicionar'' para somas: expresse com o triângulo
        desigualdade (questão 13) por que o erro de $ a + b $ está em
        mais a soma dos erros --- a desigualdade é o
        regra de engenharia.
  \item Finale, em um pequeno parágrafo: monte a imagem de
        a linha real estabelecida por este problema e sua
        ancestrais --- sem próximo número
        (o problema do fim de semana sem próximo número do volume do Ensino Médio), racionais $=$ repetindo
        decimais (o problema de fim de semana de decimais repetidos do volume do ensino médio), ambas as famílias densas
        (questões 8 e 9), medição sempre indecisa
        (questão 16). Termine com o teaser que este capítulo não pode
        ainda provar: num sentido preciso, existem \emph{vastamente
        mais} irracionais do que racionais --- a universidade
        os volumes contam os infinitos.
\end{enumerate}
\end{problem}
