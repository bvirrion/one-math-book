\chapter{Funções}\label{ch:g10:functions}

UM \omterm{def:g10:functions:function}{função} é uma máquina que recebe um número e produz um número,
sempre a mesma saída para a mesma entrada. Funções são os centrais
objetos de análise que você construirá nos próximos anos
(\cref{ch:g11:quad} em diante); este capítulo configura o vocabulário ---
domínio, \omterm{def:g10:functions:function}{imagem}, \omterm{def:g10:functions:graph}{gráfico}, variações --- e treina a habilidade essencial de
lendo informações de um \omterm{def:g10:functions:graph}{gráfico}.

\section{Vocabulário: imagens e pré-imagens}

\begin{definition}[Função]\label{def:g10:functions:function}
Deixar $D $ ser um conjunto de \omterm{def:g10:numbers:sets}{números reais}. UM \emph{função}\index{função} $ f $
definido em $ D $ associados a cada número $ x \in D $ exatamente um verdadeiro
número, escrito $ f(x)$ e chamou o \emph{imagem}\index{imagem} de $ x $.
O conjunto $ D $ é o \emph{domínio domínio}\index{domínio domínio} de $ f $. Nós escrevemos
\[
f : x \longmapsto f(x).
\]
Se $ f(x) = y $, então $ x $ é um \emph{pré-imagem}\index{pré-imagem} de $ y $.
\end{definition}

\begin{remark}
Cada $ x $ no domínio tem exatamente um \omterm{def:g10:functions:function}{imagem}, mas um número $ y $ pode ter
vários pré-imagens, ou nenhum. Para $ f(x) = x^2$: o \omterm{def:g10:functions:function}{imagem} de $3$ é $9$, e
$9$ tem dois pré-imagens, $3$ e $-3$; o número $-4$ não tem \omterm{def:g10:functions:function}{pré-imagem}.
\end{remark}

\begin{example}[Encontrando um domínio]\label{ex:g10:functions:domain}
Quando um \omterm{def:g10:functions:function}{função} é dado por uma fórmula, seu domínio é o conjunto de $ x $ para
qual a fórmula faz sentido.
\begin{itemize}
  \item $ f(x) = 3x^2 - 5x + 1$ faz sentido para cada $ x $: o domínio é
        $\R $.
  \item $ g(x) = \dfrac{1}{x - 2}$ requer $ x - 2 \neq 0$: o domínio é
        todos os reais, exceto $2$.
  \item $ h(x) = \sqrt{x - 3}$ requer $ x - 3 \geq 0$: o domínio é
        $\intco{3}{+\infty}$.
\end{itemize}
\end{example}

\begin{example}[Computando imagens e pré-imagens]\label{ex:g10:functions:images}
Deixar $ f(x) = x^2 - 4x + 3$, definido em $\R $.

\emph{\omterm{def:g10:functions:function}{Imagem} de $5$:} substituir $ x = 5$:
$ f(5) = 25 - 20 + 3 = 8$.

\emph{Pré-imagens de $3$:} resolver $ f(x) = 3$:
\[
x^2 - 4x + 3 = 3
\quad\Longleftrightarrow\quad
x^2 - 4x = 0
\quad\Longleftrightarrow\quad
x(x - 4) = 0,
\]
então o pré-imagens de $3$ são $0$ e $4$.
\end{example}

\section{O gráfico de uma função}

\begin{definition}[Gráfico]\label{def:g10:functions:graph}
Em um sistema de coordenadas, o \emph{gráfico}\index{gráfico} (ou \emph{curva}) de
$ f $ é o conjunto de todos os pontos $(x, f(x))$ para $ x $ no domínio. Em outro
palavras, um ponto $(x, y)$ está no gráfico exatamente quando $ y = f(x)$.
\end{definition}

\begin{method}[Lendo um gráfico]\label{met:g10:functions:reading}
No \omterm{def:g10:functions:graph}{gráfico} de $ f $:
\begin{enumerate}
  \item \emph{\omterm{def:g10:functions:function}{imagem} de $ a $}: vá verticalmente de $ x = a $ na horizontal
        eixo para a curva, depois horizontalmente para o eixo vertical; o
        valor lido existe $ f(a)$;
  \item \emph{pré-imagens de $ b $}: desenhe a linha horizontal $ y = b $; o
        pré-imagens são os $ x $-coordenadas de todos os seus \omterm{def:g10:numbers:interunion}{interseção} pontos
        com a curva;
  \item \emph{soluções de $ f(x) = k $}: igual ao pré-imagens de $ k $;
  \item \emph{soluções de $ f(x) \leq k $}: o $ x $ para a qual a curva é
        na linha ou abaixo dela $ y = k $.
\end{enumerate}
\end{method}

as abcissas do \omterm{def:g10:numbers:interunion}{interseção} pontos da curva com a horizontal
linha $ y = k $. Aqui $ f(x) \leq k $ aguenta $\intcc{x_1}{x_2}$, onde o
a curva está abaixo da linha.}
\end{omfigure}

\begin{example}\label{ex:g10:functions:membership}
É o ponto $ A(2, 5)$ no \omterm{def:g10:functions:graph}{gráfico} de $ f(x) = x^2 + 1$? Calcular
$ f(2) = 4 + 1 = 5$: sim, desde $ f(2)$ é igual ao $ y $-coordenada de $ A $.
O ponto $ B(3, 8)$ não está no \omterm{def:g10:functions:graph}{gráfico}, porque $ f(3) = 10 \neq 8$.
\end{example}

\section{Variações de uma função}

\begin{definition}[Aumentando, diminuindo]\label{def:g10:functions:variations}
Deixar $ f $ ser definido em um \omterm{def:g10:numbers:interval}{intervalo} $ I $.
\begin{itemize}
  \item $ f $ é \emph{aumentando}\index{aumentando} sobre $ I $ quando
        para todos $ u, v \in I $, se $ u < v $ então $ f(u) \leq f(v)$: o
        os resultados crescem com os insumos, e o \omterm{def:g10:functions:graph}{gráfico} sobe da esquerda para
        certo.
  \item $ f $ é \emph{conduzindo}\index{conduzindo} sobre $ I $ quando
        $ u < v $ implica $ f(u) \geq f(v)$: o \omterm{def:g10:functions:graph}{gráfico} cai da esquerda para
        certo.
\end{itemize}
Com desigualdades estritas $ f(u) < f(v)$ (resp.\ $>$) dizemos
\emph{estritamente} aumentando (resp.\ diminuindo).
\end{definition}

\begin{definition}[Tabela de variação]\label{def:g10:functions:table}
UM \emph{tabela de variação}\index{tabela de variação} resume sobre o que
\omterm{def:g10:numbers:interval}{intervalos} $ f $ aumenta e diminui, com setas $\nearrow $ e
$\searrow $, e registra os valores de $ f $ nos pontos de viragem.
\end{definition}

\begin{example}\label{ex:g10:functions:table}
Considere o \omterm{def:g10:functions:function}{função} \omterm{def:g10:functions:graph}{gráfico} abaixo em $\intcc{-2}{4}$.

{\small UM \omterm{def:g10:functions:function}{função} definido em $\intcc{-2}{4}$, \omterm{def:g10:functions:variations}{aumentando} então diminuindo.}
\end{omfigure}

Lendo o \omterm{def:g10:functions:graph}{gráfico}: $ f $ aumenta de $ f(-2) = -1.5$ até o seu \omterm{def:g10:functions:extrema}{máximo}
$ f(1) = 3$, depois diminui para $ f(4) = -1.5$. Isso é \omterm{def:g10:functions:table}{tabela de variação} é
\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{c|ccccc}
$ x $ & $-2$ & & $1$ & & $4$ \\
\hline
$ f $ & $-1.5$ & $\nearrow $ & $3$ & $\searrow $ & $-1.5$ \\
\end{tabular}
\end{center}
\end{example}

\begin{example}[Provando uma variação]\label{ex:g10:functions:proof}
Mostre isso $ f(x) = 3x + 1$ é estritamente \omterm{def:g10:functions:variations}{aumentando} sobre $\R $. Pegue qualquer
$ u < v $ e compare o imagens:
\[
f(v) - f(u) = (3v + 1) - (3u + 1) = 3(v - u) > 0,
\]
porque $ v - u > 0$. Então $ f(u) < f(v)$: o \omterm{def:g10:functions:function}{função} é estritamente
\omterm{def:g10:functions:variations}{aumentando}. O mesmo cálculo com uma inclinação negativa, por exemplo\
$ g(x) = -2x + 5$, dá $ g(v) - g(u) = -2(v-u) < 0$: estritamente diminuindo.
\end{example}

\section{Extrema}

\begin{definition}[Máximo, mínimo]\label{def:g10:functions:extrema}
Deixar $ f $ ser definido em $ D $ e $ a \in D $. O valor $ f(a)$ é o
\emph{máximo}\index{máximo} de $ f $ sobre $ D $ quando $ f(x) \leq f(a)$ para todos
$ x \in D $; é o \emph{mínimo}\index{mínimo} quando
$ f(x) \geq f(a)$ para todos $ x \in D $. Em um \omterm{def:g10:functions:graph}{gráfico}, eles são os mais altos e
pontos mais baixos da curva.
\end{definition}

\begin{example}\label{ex:g10:functions:extremum}
Mostre isso $ f(x) = (x - 1)^2 + 2$ tem \omterm{def:g10:functions:extrema}{mínimo} $2$ sobre $\R $, alcançado em
$ x = 1$. Para cada $ x $, o quadrado $(x-1)^2$ é $\geq 0$, então
$ f(x) = (x-1)^2 + 2 \geq 2$; e $ f(1) = 0 + 2 = 2$, então o valor $2$ é
realmente alcançado. Os dois fatos juntos provam que $2$ é o \omterm{def:g10:functions:extrema}{mínimo}.
\end{example}

\begin{omfigure}
\begin{tikzpicture}
\begin{axis}[omaxis,
  width=7.6cm, height=5.2cm,

$ y = 2$, e toca-o em $ x = 1$: o \omterm{def:g10:functions:extrema}{mínimo} de $ f $ é $2$.}
\end{omfigure}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g10:functions:1}
Deixar $ f(x) = 2x^2 - 3x + 1$. Calcule o imagens de $0$, $2$, $-1$ e
$\frac12$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g10:functions:2}
Dê o domínio de cada \omterm{def:g10:functions:function}{função}:
\[
f(x) = 5x - 2, \qquad
g(x) = \frac{3}{x + 4}, \qquad
h(x) = \sqrt{2x - 6}, \qquad
k(x) = \frac{1}{x^2 + 1}.
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g10:functions:3}
Deixar $ f(x) = x^2 - 2x $. Encontre tudo pré-imagens de $0$, de $3$, e de $-1$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g10:functions:4}
Será que o ponto $ A(-1, 4)$ pertence ao \omterm{def:g10:functions:graph}{gráfico} de $ f(x) = x^2 - 2x + 1$?
E o ponto $ B(2, 1)$? Justifique por um cálculo.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g10:functions:5}
UM \omterm{def:g10:functions:function}{função} $ g $ definido em $\intcc{-3}{5}$ tem o \omterm{def:g10:functions:table}{tabela de variação}
\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{c|ccccccc}
$ x $ & $-3$ & & $0$ & & $3$ & & $5$ \\
\hline
$ g $ & $1$ & $\searrow $ & $-2$ & $\nearrow $ & $4$ & $\searrow $ & $0$ \\
\end{tabular}
\end{center}
\begin{enumerate}
  \item Quais são os \omterm{def:g10:functions:extrema}{máximo} e o \omterm{def:g10:functions:extrema}{mínimo} de $ g $ sobre $\intcc{-3}{5}$?
  \item Comparar $ g(-1)$ e $ g(-0.5)$ sem conhecer seus valores.
  \item Quantas soluções o equação $ g(x) = 0$ tem no \omterm{def:g10:functions:extrema}{máximo} ligado
        cada \omterm{def:g10:numbers:interval}{intervalo} da mesa?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:functions:6}
Deixar $ f(x) = -2x + 7$. Mostre, comparando $ f(u)$ e $ f(v)$ para $ u < v $,
que $ f $ é estritamente diminuindo sobre $\R $.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:functions:7}
Mostre que o \omterm{def:g10:functions:function}{função} $ f(x) = x^2 + 6x + 11$ tem um \omterm{def:g10:functions:extrema}{mínimo} sobre $\R $ e
dar o seu valor e onde ele é alcançado. (Dica: escreva
$ f(x) = (x + 3)^2 + c $ para a constante certa $ c $.)
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:functions:8}
Um jardim retangular tem perímetro $40$ m. Deixar $ x $ seja sua largura, em
metros.
\begin{enumerate}
  \item Expresse o comprimento e depois a área $ A(x)$, como funções de $ x $.
        Para qual $ x $ isso faz sentido?
  \item Calcular $ A(4)$, $ A(8)$, $ A(10)$, $ A(12)$ e $ A(16)$. O que você
        conjectura sobre a forma da maior área?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:functions:9}
Deixar $ f(x) = \dfrac{1}{x^2 + 1}$, definido em $\R $.
\begin{enumerate}
  \item Mostre isso $0 < f(x) \leq 1$ por cada real $ x $.
  \item Faz $ f $ tenha um \omterm{def:g10:functions:extrema}{máximo} sobre $\R $? UM \omterm{def:g10:functions:extrema}{mínimo}? Justificar.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g10:functions:10}
Deixar $ f(x) = x^2$ sobre $\R $.
\begin{enumerate}
  \item Deixar $0 \leq u < v $. Fator $ f(v) - f(u)$ e deduza que $ f $ é
        estritamente \omterm{def:g10:functions:variations}{aumentando} sobre $\intco{0}{+\infty}$.
  \item Adapte o argumento para mostrar que $ f $ é estritamente diminuindo sobre
        $\intoc{-\infty}{0}$.
\end{enumerate}
\end{exercise}

\section{Problema: Máquinas alimentam sua própria produção}

\begin{problem}[{Problema de fim de semana --- pontos fixos, iteração e
três máquinas famosas: o melhorador de Heron, o esquadreador e o
granizo não resolvido}]
\label{pb:g10:functions:1}
UM \omterm{def:g10:functions:function}{função} é uma máquina: entra um número, sai um número.
Os experimentos mais interessantes alimentam a máquina \emph{próprio
saída}, repetidamente --- e então duas questões dominam:
o processo se estabelece em algum lugar (um \emph{\omterm{pb:g10:functions:1}{ponto fixo}}) e
como a variação da máquina a leva até lá? Um destes
as máquinas do problema estão polindo raízes quadradas há dois
mil anos; outro guarda o problema não resolvido mais famoso
matemática elementar tem a oferecer.

\medskip
\noindent\textbf{Part I --- Heron's machine.}
Considere o \omterm{def:g10:functions:function}{função}
\[
f(x) = \frac12\left(x + \frac2x\right).
\]
\begin{enumerate}
  \item Dê o domínio de $ f $ e calcular $ f(1)$, $ f(2)$ e
        $ f\!\left(\frac32\right)$ como frações exatas.
  \item Encontre tudo pré-imagens de $\frac32$: resolver
        $ f(x) = \frac32$ (limpe o denominador e use o
        métodos de \cref{pb:g10:algebra:1}).
  \item UM \emph{ponto fixo}\index{ponto fixo} de $ f $ é um
        número inalterado pela máquina: $ f(x) = x $. Encontre ambos
        pontos fixos de $ f $. Qual velho conhecido guarda o
        positivo?
  \item A partir de $ x = 1$, alimente a máquina com sua própria saída
        três vezes e registre os resultados exatos. Qual sequência
        do problema de irracionalidade do fim de semana do volume do ensino médio você reconstruiu --- e o que foi
        ``Receita da Garça'', na linguagem de funções?
  \item Em uma \omterm{def:g10:functions:function}{imagem}, esboce o \omterm{def:g10:functions:graph}{gráfico} de $ f $ (para $ x > 0$)
        e a linha diagonal $ y = x $. Onde estão os fixo
        pontos nesta foto? (Comparar
        o problema de fim de semana de dois termômetros do volume do ensino médio: a temperatura é a mesma
        em ambas as escalas foi a mesma ideia.)
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Why the machine cannot miss.}
\begin{enumerate}[resume]
  \item Para $0 < u < v $, mostre por redução a um comum
        denominador que
        \[
        f(v) - f(u) = \frac{(v - u)(uv - 2)}{2uv} ,
        \]
        e deduzir as variações de $ f $ sobre
        $\intoo{0}{+\infty}$: diminuindo até certo ponto, então
        \omterm{def:g10:functions:variations}{aumentando} --- qual ponto? (Mesma técnica que
        \cref{exo:g10:functions:10}.)
  \item Deduza isso em $\intoo{0}{+\infty}$ o \omterm{def:g10:functions:function}{função} $ f $ tem
        um \omterm{def:g10:functions:extrema}{mínimo}, vale exatamente $\sqrt2$, alcançado em
        $\sqrt2$. O que isso diz sobre \emph{todo} saída
        da máquina (entradas positivas)?
  \item Mostre que se $ x > \sqrt2$ então
        $\sqrt2 < f(x) < x $: um palpite muito grande é sempre melhorado,
        nunca ultrapassou. (Para $ f(x) < x $, comparar $\frac2x $ com
        $ x $.)
  \item Reúna as questões 6 a 8 no \omterm{def:g10:functions:table}{tabela de variação} de $ f $
        sobre $\intoo{0}{+\infty}$
        (\cref{def:g10:functions:table}), \omterm{def:g10:functions:extrema}{mínimo} incluído.
  \item Uma segunda máquina: $ g(x) = x^2$. Isso é pontos fixos são
        $0$ e $1$ (verificar). Iterar $ g $ quatro vezes de
        $ x = 0.9$, então quatro vezes a partir de $ x = 1.1$ (calculadora,
        três casas decimais). Um \omterm{pb:g10:functions:1}{ponto fixo} atrai, o outro
        repele: qual é qual?
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The hailstone machine.}
Nos números \omterm{def:g10:numbers:sets}{inteiros}, defina: $ h(n) = \frac n2$ se $ n $ é par,
e $ h(n) = 3n + 1$ se $ n $ é estranho. Os números saltam para baixo $ h $ gosto
granizo em uma nuvem de tempestade.
\begin{enumerate}[resume]
  \item Calcule o voo completo de $7$ sob $ h $, até $1$.
        Quantos passos ele dá e quão alto ele voa?
  \item Comece o vôo de $27$ e calcule dez passos. (É
        o voo completo dura $111$ passos e picos em $9\,232$ ---
        de um ponto de partida $27$.) Que lição sobre simples
        regras fazem $7$ e $27$ ensinar?
  \item Explique por que qualquer voo que atinja uma potência de $2$
        bate direto para $1$, e o que acontece em $1$
        (calcular $ h(1)$, $ h(4)$, $ h(2)$). Faz $ h $ tem algum
        \omterm{pb:g10:functions:1}{ponto fixo} entre os números \omterm{def:g10:numbers:sets}{inteiros} positivos? O que faz
        o final de cada formulário de voo observado?
  \item O \emph{Conjectura de Collatz} reivindica cada partida
        número eventualmente chega $1$. Os computadores verificaram
        isso muito além $10^{20}$; nenhum humano provou isso. O que
        dois tipos de descoberta seriam \emph{refutar} isto? E por que
        faz verificação massiva ainda não encerra o caso
        (o problema do fim de semana do oráculo do palito de fósforo do volume do ensino médio tocou a campainha)?
  \item “A matemática ainda não está madura para tais questões”, disse
        Paulo Erd\H{o}s desta conjectura. Em uma frase: o que
        separa a máquina de granizo da de Heron, onde
        as questões 6 a 8 resolveram tudo?
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The machine user's manual.}
\begin{enumerate}[resume]
  \item Domínios como \omterm{def:g10:numbers:interval}{intervalos} (\cref{pb:g10:numbers:1}de
        idioma): dê o domínios de
        $ x \mapsto \sqrt{x - 3}$, \
        $ x \mapsto \frac{1}{x^2 - 9}$, \
        $ x \mapsto \sqrt{9 - x^2}$.
  \item De volta ao Heron's $ f $: encontre o exato pré-imagens de $3$
        (resolver $ f(x) = 3$ completando o quadrado).
  \item Uma pedra é atirada para cima: sua altura é
        $ H(t) = 20t - 5t^2$ metros depois $ t $ segundos. Em que
        hora \omterm{def:g10:numbers:interval}{intervalo} a fórmula faz sentido físico?
        Complete o quadrado para encontrar a altura máxima e sua
        tempo (\cref{def:g10:functions:extrema}) e dê o
        \omterm{def:g10:functions:table}{tabela de variação}.
  \item Por quanto tempo a pedra dura pelo menos $15$ estou alto? (Resolver
        $ H(t) \geq 15$ com um \omterm{def:g10:algebra:signtable}{mesa de sinalização},
        \cref{met:g10:algebra:signtable}.)
  \item Final: este problema rodou quatro máquinas --- Heron's
        melhorador, o esquadro, o granizo, a pedra
        altura. Em duas ou três frases, indique o que cada
        \omterm{def:g10:functions:function}{função} tem (um domínio, uma regra, um \omterm{def:g10:functions:graph}{gráfico}) e os dois
        ótimas perguntas que esse problema fez a cada um (onde
        iteração resolvida? como é que \omterm{def:g10:functions:function}{função} variar?) ---
        perguntas que os próximos capítulos abordam um por um.
\end{enumerate}
\end{problem}
