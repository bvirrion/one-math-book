\chapter{Geometria coordenada}\label{ch:g10:coordgeom}

A grande ideia de Descartes foi descrever os pontos por pares de números, transformando
problemas de geometria em cálculos. Com apenas duas fórmulas --- a
\omterm{prop:g10:coordgeom:midpoint}{ponto médio} e a distância --- pode-se provar que um triângulo é isósceles,
que um quadrilátero é um paralelogramo, ou que três pontos estão alinhados,
sem desenhar uma única linha auxiliar.

\section{Coordenadas no avião}

\begin{definition}[Sistema de coordenadas]\label{def:g10:coordgeom:system}
UM \emph{sistema de coordenação}\index{sistema de coordenação} do avião consiste
de uma origem $O $ e dois eixos graduados através $ O $: a horizontal
$ x $-eixo e a vertical $ y $-eixo. Cada ponto $ M $ então tem um par único
de \emph{coordenadas}\index{coordenadas} $(x, y)$: isso é
\emph{abscissa}\index{abscissa} $ x $ e seu
\emph{ordenar}\index{ordenar} $ y $. O sistema é
\emph{ortonormal}\index{ortonormal} quando os eixos estão
perpendicular e tem a mesma unidade de comprimento. Todos os sistemas neste
capítulo são ortonormais.
\end{definition}

\section{Ponto médio de um segmento}

\begin{proposition}[Fórmula do ponto médio]\label{prop:g10:coordgeom:midpoint}
Deixar $ A(x_A, y_A)$ e $ B(x_B, y_B)$. O \emph{ponto médio}\index{ponto médio}
$ I $ do segmento $[AB]$ tem \omterm{def:g10:coordgeom:system}{coordenadas}
\[
I\left(\frac{x_A + x_B}{2},\ \frac{y_A + y_B}{2}\right).
\]
\end{proposition}

\begin{proof}
Considere a horizontal \omterm{def:g10:coordgeom:system}{coordenadas}. Indo de $ A $ para $ B $, o \omterm{def:g10:coordgeom:system}{abscissa}
mudanças por $ x_B - x_A $; o ponto no meio do caminho tem \omterm{def:g10:coordgeom:system}{abscissa}
\[
x_A + \frac{x_B - x_A}{2} = \frac{2x_A + x_B - x_A}{2}
= \frac{x_A + x_B}{2},
\]
a média dos dois \omterm{def:g10:coordgeom:system}{abscissas}. O mesmo cálculo se aplica ao
ordenadas.
\end{proof}

\begin{example}\label{ex:g10:coordgeom:midpoint}
Com $ A(-1, 4)$ e $ B(5, -2)$, o \omterm{prop:g10:coordgeom:midpoint}{ponto médio} de $[AB]$ é
\[
I\left(\frac{-1 + 5}{2},\ \frac{4 + (-2)}{2}\right) = I(2, 1).
\]
A fórmula também funciona ao contrário: se $ A(-1, 4)$ e o \omterm{prop:g10:coordgeom:midpoint}{ponto médio} é
$ I(2, 1)$, então $ B $ satisfaz $\frac{-1 + x_B}{2} = 2$ e
$\frac{4 + y_B}{2} = 1$, então $ x_B = 5$ e $ y_B = -2$: o ponto $ B $ é o
\emph{simétrico} de $ A $ sobre $ I $.
\end{example}

\section{Distância entre dois pontos}

\begin{theorem}[Fórmula de distância]\label{thm:g10:coordgeom:distance}
Em um sistema \omterm{def:g10:coordgeom:system}{ortonormal}, a distância entre $ A(x_A, y_A)$ e
$ B(x_B, y_B)$ é
\[
AB = \sqrt{(x_B - x_A)^2 + (y_B - y_A)^2}.
\]
\end{theorem}

\begin{proof}
Deixar $ C $ seja o ponto $(x_B, y_A)$: tem a mesma altura que $ A $ e o
mesmo \omterm{def:g10:coordgeom:system}{abscissa} como $ B $, então o triângulo $ ACB $ tem um ângulo reto em $ C $. É
pernas são segmentos horizontais e verticais, de comprimentos
$ AC = \abs{x_B - x_A}$ e $ CB = \abs{y_B - y_A}$. Pelo Pitagórico
teorema,
\[
AB^2 = AC^2 + CB^2 = (x_B - x_A)^2 + (y_B - y_A)^2,
\]
e tirar a raiz quadrada (ambos os lados não são negativos) dá a fórmula.
\end{proof}

\end{tikzpicture}

{\small A fórmula da distância é o teorema de Pitágoras aplicado ao
triângulo retângulo $ ACB $ construído em uma perna horizontal e vertical.}
\end{omfigure}

\begin{example}\label{ex:g10:coordgeom:distance}
Com $ A(1, 1)$ e $ B(5, 3)$:
\[
AB = \sqrt{(5-1)^2 + (3-1)^2} = \sqrt{16 + 4} = \sqrt{20} = 2\sqrt 5 .
\]
As distâncias são geralmente mantidas na forma exata (raiz quadrada); apenas rodada no
bem no final se uma resposta decimal for necessária.
\end{example}

\begin{remark}
A ordem dos pontos não importa: $(x_A - x_B)^2 =
(x_B - x_A)^2$. Mas não se esqueça dos quadrados! A distância é
\emph{não} $\abs{x_B - x_A} + \abs{y_B - y_A}$.
\end{remark}

\section{Usando as fórmulas em geometria}

\begin{method}[Natureza de um triângulo]\label{met:g10:coordgeom:triangle}
Dados três pontos $ A $, $ B $, $ C $ por seus \omterm{def:g10:coordgeom:system}{coordenadas}:
\begin{enumerate}
  \item calcular os três comprimentos quadrados $ AB^2$, $ AC^2$, $ BC^2$ com o
        fórmula de distância (manter quadrados evita raízes quadradas);
  \item dois comprimentos quadrados iguais $\Rightarrow $ o triângulo é
        \emph{isósceles};
  \item se o maior comprimento ao quadrado for a soma dos outros dois, o
        triângulo é \emph{em ângulo reto} no vértice oposto ao mais longo
        lado, pelo inverso do teorema de Pitágoras.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g10:coordgeom:triangle}
Deixar $ A(0, 1)$, $ B(4, 3)$ e $ C(2, -3)$. Então
\begin{align*}
AB^2 &= (4-0)^2 + (3-1)^2 = 16 + 4 = 20, \\
AC^2 &= (2-0)^2 + (-3-1)^2 = 4 + 16 = 20, \\
BC^2 &= (2-4)^2 + (-3-3)^2 = 4 + 36 = 40 .
\end{align*}
Primeiro, $ AB^2 = AC^2$, então $ AB = AC $: o triângulo é isósceles em $ A $.
Além disso $ AB^2 + AC^2 = 40 = BC^2$, então, pelo inverso do pitagórico
teorema, o triângulo também é retângulo em $ A $.
\end{example}

\begin{method}[Reconhecendo um paralelogramo]\label{met:g10:coordgeom:parallelogram}
Um quadrilátero $ ABCD $ é um paralelogramo exatamente quando suas diagonais
$[AC]$ e $[BD]$ tem o mesmo \omterm{prop:g10:coordgeom:midpoint}{ponto médio}. Então: calcule ambos pontos médios com
o \omterm{prop:g10:coordgeom:midpoint}{ponto médio} fórmula e compare-as.
\end{method}

\begin{example}\label{ex:g10:coordgeom:parallelogram}
Deixar $ A(-1, 0)$, $ B(2, 2)$, $ C(5, 1)$ e $ D(2, -1)$. O \omterm{prop:g10:coordgeom:midpoint}{ponto médio} de
$[AC]$ é $\left(\frac{-1+5}{2}, \frac{0+1}{2}\right) = (2, \frac12)$; o
\omterm{prop:g10:coordgeom:midpoint}{ponto médio} de $[BD]$ é $\left(\frac{2+2}{2}, \frac{2-1}{2}\right) =
(2, \frac12)$. Eles coincidem, então $ ABCD $ é um paralelogramo.
\end{example}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g10:coordgeom:1}
Deixar $ A(2, 5)$ e $ B(-4, 1)$. Calcule o \omterm{def:g10:coordgeom:system}{coordenadas} do \omterm{prop:g10:coordgeom:midpoint}{ponto médio} de
$[AB]$ e a distância $ AB $.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g10:coordgeom:2}
Deixar $ A(3, -2)$ e $ I(1, 2)$. Encontre o \omterm{def:g10:coordgeom:system}{coordenadas} do ponto $ B $ tal
isso $ I $ é o \omterm{prop:g10:coordgeom:midpoint}{ponto médio} de $[AB]$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g10:coordgeom:3}
Qual dos pontos $ P(4, 1)$, $ Q(-3, 2)$ e $ R(0, -5)$ está mais próximo do
origem $ O(0,0)$? Responda com distâncias exatas.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g10:coordgeom:4}
Plote os pontos $ A(1, 2)$, $ B(5, 4)$, $ C(7, 0)$ sobre \omterm{def:g10:functions:graph}{gráfico} papel e depois mostre
calculando que o triângulo $ ABC $ é isósceles. É em ângulo reto?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:coordgeom:5}
Deixar $ A(-2, 1)$, $ B(1, 3)$, $ C(4, 1)$ e $ D(1, -1)$.
\begin{enumerate}
  \item Mostre isso $ ABCD $ é um paralelogramo.
  \item Calcular $ AB $ e $ BC $. É $ ABCD $ um losango (todos os lados iguais)?
  \item Calcular $ AC $ e $ BD $. É $ ABCD $ um retângulo (diagonais iguais)?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:coordgeom:6}
O círculo com centro $\Omega(2, 1)$ e raio $5$ consiste em todos
pontos à distância $5$ de $\Omega $. Qual dos pontos $ A(5, 5)$,
$ B(-1, -3)$, $ C(6, 2)$ mentir neste círculo? Dentro dele? Fora disso?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:coordgeom:7}
Deixar $ A(1, 1)$ e $ B(7, 5)$. Encontre todos os pontos $ M(x, 0)$ do $ x $-eixo
que são equidistantes de $ A $ e $ B $. (Escrever $ MA^2 = MB^2$ e resolver para
$ x $.)
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:coordgeom:8}
Deixar $ A(0, 3)$, $ B(4, 1)$.
\begin{enumerate}
  \item Calcule o \omterm{def:g10:coordgeom:system}{coordenadas} do \omterm{prop:g10:coordgeom:midpoint}{ponto médio} $ I $ de $[AB]$.
  \item Mostre isso $ M(2, 2) = I $, então verifique calculando $ MA $, $ MB $ que
        $ M $ é equidistante de $ A $ e $ B $.
  \item Encontre um segundo ponto, no $ y $-eixo, equidistante de $ A $ e
        $ B $.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:coordgeom:9}
Os pontos $ A(-1, -1)$, $ B(3, 1)$ e $ C(11, 5)$ são dados. Calcular $ AB $,
$ BC $ e $ AC $, e deduza que $ A $, $ B $, $ C $ estão alinhados. (Três pontos
estão alinhados exatamente quando a maior das três distâncias é a soma de
os outros dois.)
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g10:coordgeom:10}
Deixar $ A(a, 0)$ e $ B(0, b)$ com $ a, b > 0$, e deixe $ I $ seja o \omterm{prop:g10:coordgeom:midpoint}{ponto médio}
de $[AB]$. Mostre por cálculo que $ OI = IA = IB $, onde $ O $ é o
origem. (Isso prova um teorema clássico: em um triângulo retângulo, o
\omterm{prop:g10:coordgeom:midpoint}{ponto médio} da hipotenusa é equidistante dos três vértices.)
\end{exercise}

\section{Problema: Ponte de Descartes --- equações circulares e como
para descobrir onde você está}

\begin{problem}[{Problema de fim de semana --- o círculo se torna um
equação, velhos teoremas tornam-se cálculos e três faróis
localizar um ponto: trilateração por álgebra}]
\label{pb:g10:coordgeom:1}
Em 1637 René Descartes construiu uma ponte entre dois continentes:
cada curva da geometria tornou-se um \emph{equação}, cada
questão geométrica um cálculo. Este problema caminha pela ponte
em ambas as direções --- derivando o círculo equação do
fórmula de distância (\cref{thm:g10:coordgeom:distance}), provando novamente
teoremas clássicos em três linhas de álgebra, e terminando onde
a ponte transporta mais tráfego hoje: calculando uma posição
dos sinais de distância, o coração plano do GPS.

\medskip
\noindent\textbf{Part I --- The circle gets an equação.}
\begin{enumerate}
  \item Um ponto $ M(x, y)$ encontra-se no círculo do centro
        $\Omega(2, 1)$ e raio $5$ exatamente quando
        $\Omega M = 5$. Eleve ao quadrado esta condição para obter o
        círculo equação:
        \[
        (x - 2)^2 + (y - 1)^2 = 25 .
        \]
  \item Associação de teste: qual dos $ A(5, 5)$, $ B(6, 4)$,
        $ D(4, 4)$ estão no círculo, dentro dele, fora dele?
  \item Um círculo disfarçado: complete os quadrados
        (\cref{pb:g10:algebra:1}) em
        \[
        x^2 + y^2 - 6x + 4y - 12 = 0
        \]
        e indique seu centro e raio.
  \item Faça o mesmo com $ x^2 + y^2 - 6x + 4y + 14 = 0$. O que
        o formulário preenchido revela? Indique o critério: quando
        faz $(x - h)^2 + (y - k)^2 = c $ descrever um círculo, um
        único ponto ou nada?
  \item Intersecte o círculo da questão 3 com a horizontal
        linha $ y = 3$: substitua e resolva. Quantos pontos e
        qual é o nome geométrico dessa linha?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Old theorems in three lines.}
\begin{enumerate}[resume]
  \item O teorema do círculo do teorema do círculo do volume do ensino médio (um ângulo reto em cada semicírculo),
        re-provado pela álgebra: deixe $ A(-r, 0)$ e $ B(r, 0)$ ser
        as extremidades de um diâmetro e $ M(x, y)$ qualquer ponto do
        círculo $ x^2 + y^2 = r^2$. Calcular $ MA^2 + MB^2$ e
        conclua com o inverso de Pitágoras que
        $\widehat{AMB}$ está certo.
  \item O teorema da mediana: com $ A(-a, 0)$, $ B(a, 0)$ (então o
        \omterm{prop:g10:coordgeom:midpoint}{ponto médio} de $[AB]$ é a origem $ I $), mostre que para
        cada ponto $ M(x, y)$:
        \[
        MA^2 + MB^2 = 2\,MI^2 + \frac{AB^2}{2} .
        \]
  \item Deduza o lugar geométrico dos pontos $ M $ com
        $ MA^2 + MB^2 = 26$ quando $ AB = 6$: que curva, qual
        centro, qual raio?
  \item Um locus diferente: $ A(0,0)$, $ B(3,0)$; encontre todos os pontos
        com $ MA = 2\,MB $. (Quadrar, expandir, completar o
        quadrados: um famoso círculo aparece --- Apolônio sabia disso
        sem \omterm{def:g10:coordgeom:system}{coordenadas}.)
  \item Em uma ou duas frases: o que significa a ponte de Descartes
        mudar sobre \emph{como} teoremas são provados? Comparar
        questão 6 com a prova retangular de
        o teorema do círculo do volume do ensino médio (um ângulo reto em cada semicírculo).
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Three beacons find you.}
Sua posição $ M(x, y)$ é desconhecido. Baliza $ P(0, 0)$ medidas
sua distância como $5$; baliza $ Q(6, 0)$ também mede $5$.
\begin{enumerate}[resume]
  \item Escreva os dois círculos equações, subtraia-os e observe
        os quadrados se cancelam: que simples equação sobrevive?
        Combine com um círculo para encontrar o \emph{dois} candidato
        posições.
  \item Um terceiro farol $ R(0, 8)$ mede sua distância como $5$.
        Calcule a distância até cada candidato e decida onde
        você é.
  \item Explique por que subtrair dois círculos equações
        \emph{sempre} produz o equação de uma linha (que termos
        cancelar?), e o que essa linha é geometricamente quando o
        círculos se cruzam em dois pontos.
  \item O posicionamento real por satélite funciona no espaço e com
        relógios imperfeitos: cada satélite fornece (por meio do deslocamento do sinal
        tempo) uma distância, portanto uma esfera. Quantas incógnitas
        um receptor tem (posição \emph{e} seu próprio relógio
        erro), e por que o GPS escuta pelo menos
        \emph{quatro} satélites?
  \item Precisão: suponha farol $ P $ a distância é realmente
        $5.1$ em vez de $5$ (todo o resto inalterado). Refaça o
        subtração da questão 11 para encontrar o novo $ x $ e
        quantificar até que ponto a estimativa se moveu. (Compare o erro
        ideias de orçamento de \cref{pb:g10:numbers:1}.)
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The surveyor's toolkit.}
\begin{enumerate}[resume]
  \item O caminho de irrigação mais curto: um acampamento em $ A(1, 5)$, um celeiro
        em $ B(7, 3)$, um rio reto ao longo do eixo $ y = 0$.
        Para ir de $ A $ para o rio e depois para $ B $ com o
        menos caminhar: refletir $ B $ do outro lado do rio em $ B'$,
        e descubra onde o segmento $[AB']$ atravessa o rio.
        Forneça o ponto de cruzamento e o comprimento total \omterm{def:g10:functions:extrema}{mínimo}.
        (A ideia é o exercício de salto de bilhar do volume do Ensino Médio, agora totalmente
        computável.)
  \item Confirme a minimalidade em um rival: calcule o caminho total
        através de $ Q(3, 0)$ e verifique se perde para sua resposta.
  \item Classifique o quadrilátero $ A(1,1)$, $ B(4,2)$,
        $ C(5,5)$, $ D(2,4)$ completamente: paralelogramo? losango?
        retângulo? quadrado?
        (\cref{met:g10:coordgeom:parallelogram},
        \cref{met:g10:coordgeom:triangle} --- comparar diagonais
        também.)
  \item Dado $ A(-1, 2)$, $ B(3, 4)$, $ C(6, 0)$, encontrar $ D $ tal
        isso $ ABCD$ é um paralelogramo (o \omterm{prop:g10:coordgeom:midpoint}{ponto médio} diagonal
        truque do problema de fim de semana de meia volta do volume do Ensino Médio, agora em fórmulas).
  \item Final: três ferramentas --- \omterm{prop:g10:coordgeom:midpoint}{ponto médio} fórmula, distância
        fórmula, subtração de círculo equações. Para cada um, nome
        o tipo de questão que resolveu neste problema, e
        indique o que os próximos dois capítulos acrescentam ao kit
        (direções e encostas: vetores e linha equações).
\end{enumerate}
\end{problem}
