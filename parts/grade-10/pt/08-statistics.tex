\chapter{Estatística Descritiva}\label{ch:g10:stats}

As estatísticas resumem uma grande coleção de números --- notas, salários,
temperaturas --- por alguns indicadores bem escolhidos. Este capítulo apresenta
as duas famílias de indicadores: aqueles que localizam o \emph{centro} do
dados (significar, \omterm{def:g10:stats:median}{mediana}) e aqueles que medem seu \emph{espalhar} (\omterm{def:g10:stats:quartiles}{faixa},
quartis). Um tratamento mais profundo, incluindo o desvio padrão, entra
\cref{ch:g11:stat}.

\section{Dados e frequências}

\begin{definition}[Série estatística]\label{def:g10:stats:series}
UM \emph{série estatística}\index{série estatística} é uma lista de valores
$x_1, x_2, \dots, x_N $ observado em uma população de tamanho $ N $. Quando um valor
$ x $ aparece $ n $ vezes, $ n $ é o seu \emph{contar} e o quociente
\[
f = \frac{n}{N}
\]
é o seu \emph{frequência}\index{frequência} (frequentemente expresso como um
porcentagem). As frequências de todos os valores somam $1$.
\end{definition}

\begin{example}\label{ex:g10:stats:frequencies}
As notas de $20$ estudantes em um $5$ teste de pontos:
\begin{center}
\renewcommand{\arraystretch}{1.2}
\begin{tabular}{c|cccccc}
grade $ x $ & $0$ & $1$ & $2$ & $3$ & $4$ & $5$ \\
\hline
count $ n $ & $1$ & $2$ & $4$ & $6$ & $5$ & $2$ \\
freqüência $ f $ & $0.05$ & $0.10$ & $0.20$ & $0.30$ & $0.25$ & $0.10$ \\
\end{tabular}
\end{center}
Verificar: $1 + 2 + 4 + 6 + 5 + 2 = 20$ e as \omterm{def:g10:stats:series}{frequências} somam $1$.
\end{example}

\section{Medidas de centro}

\begin{definition}[Significar]\label{def:g10:stats:mean}
O \emph{significativo}\index{significativo} da série é
\[
\bar x = \frac{x_1 + x_2 + \dots + x_N}{N}.
\]
Se os valores $ v_1, \dots, v_k $ aparecer com contagens $ n_1, \dots, n_k $, o
mesmo número é calculado como \emph{média ponderada}
\[
\bar x = \frac{n_1 v_1 + n_2 v_2 + \dots + n_k v_k}{n_1 + n_2 + \dots + n_k}.
\]
\end{definition}

\begin{example}\label{ex:g10:stats:mean}
Para o teste de \cref{ex:g10:stats:frequencies}:
\[
\bar x = \frac{1 \times 0 + 2 \times 1 + 4 \times 2 + 6 \times 3
+ 5 \times 4 + 2 \times 5}{20}
= \frac{0 + 2 + 8 + 18 + 20 + 10}{20}
= \frac{58}{20} = 2.9 .
\]
\end{example}

\begin{definition}[Mediana]\label{def:g10:stats:median}
O \emph{mediana}\index{mediana} é um valor que divide o \emph{classificado}
série em duas metades: pelo menos metade dos valores são $\leq $ a mediana, e
pelo menos metade são $\geq $ isto. Na prática, classifique o $ N $ valores;
\begin{itemize}
  \item se $ N $ é ímpar, a mediana é o valor do meio, na posição
        $\frac{N+1}{2}$;
  \item se $ N $ é par, pegue o \omterm{prop:g10:coordgeom:midpoint}{ponto médio} dos dois valores médios, em
        posições $\frac N2$ e $\frac N2 + 1$.
\end{itemize}
\end{definition}

\begin{example}\label{ex:g10:stats:median}
Série ordenada de $7$ preços das casas (em milhares): $120$, $150$, $160$,
$180$, $210$, $240$, $900$. O \omterm{def:g10:stats:median}{mediana} é o $4$ o valor, $180$. O significar
é $\frac{1960}{7} = 280$ --- maior que $6$ do $7$ preços!
Um valor extremo ($900$) puxa o significar muito mais do que \omterm{def:g10:stats:median}{mediana}: o
\omterm{def:g10:stats:median}{mediana} é \emph{robusto}, o significar não é.
\end{example}

\section{Medidas de propagação}

\begin{definition}[Intervalo, quartis]\label{def:g10:stats:quartiles}
Para uma série ordenada:
\begin{itemize}
  \item o \emph{faixa}\index{faixa} é a diferença entre
        valores maiores e menores;
  \item o \emph{primeiro quartil}\index{primeiro quartil} $ Q_1$ é o menor
        valor tal que pelo menos um quarto ($25\%$) dos valores são
        $\leq Q_1$; o \emph{terceiro quartil} $ Q_3$ é o menor valor
        tal que pelo menos três quartos ($75\%$) dos valores são
        $\leq Q_3$;
  \item o \emph{intervalo interquartil}\index{intervalo interquartil} $ Q_3 - Q_1$ mede a propagação de
        a metade central dos dados.
\end{itemize}
\end{definition}

\begin{method}[Encontrando os quartis]\label{met:g10:stats:quartiles}
Para uma série de $ N $ valores classificados:
\begin{enumerate}
  \item calcular $\frac N4$; se não for um inteiro arredondar \emph{acima} para
        o próximo inteiro; $ Q_1$ é o valor nessa posição;
  \item calcular $\frac{3N}{4}$; arredondar se necessário; $ Q_3$ é o valor em
        essa posição.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g10:stats:quartiles}
Pegue o $11$ valores classificados
\[
2,\ 3,\ 5,\ 5,\ 6,\ 7,\ 8,\ 9,\ 9,\ 10,\ 12 .
\]
\omterm{def:g10:stats:median}{Mediana}: posição $\frac{11+1}{2} = 6$, então o \omterm{def:g10:stats:median}{mediana} é $7$.
$ Q_1$: $\frac{11}{4} = 2.75$, arredondar para $3$; o $3$ O terceiro valor é $5$.
$ Q_3$: $\frac{33}{4} = 8.25$, arredondar para $9$; o $9$ o valor é $9$.
\omterm{def:g10:stats:quartiles}{Faixa}: $12 - 2 = 10$; \omterm{def:g10:stats:quartiles}{intervalo interquartil}: $9 - 5 = 4$.
\end{example}

\begin{example}[Comparando dois grupos]\label{ex:g10:stats:comparison}
Duas turmas fazem o mesmo teste (pontuações de $20$):
\begin{center}
\renewcommand{\arraystretch}{1.2}
\begin{tabular}{c|ccc}
 & \omterm{def:g10:stats:median}{mediana} & $ Q_1$ & $ Q_3$ \\
\hline
class A & $12$ & $10$ & $14$ \\
class B & $12$ & $6$ & $17$ \\
\end{tabular}
\end{center}
Mesmo \omterm{def:g10:stats:median}{mediana}, spreads muito diferentes: a classe A é homogênea (metade de sua
pontuações em $\intcc{10}{14}$), a classe B é heterogênea (sua metade central
vãos $\intcc{6}{17}$). Os indicadores do centro por si só nunca dizem tudo
história.
\end{example}

\begin{remark}
Nem o significar nem o \omterm{def:g10:stats:median}{mediana} é ``melhor'': o significar usa todos os valores
(e é necessário para calcular totais), o \omterm{def:g10:stats:median}{mediana} resiste a valores extremos. Um
O resumo sério de uma série fornece pelo menos um indicador de centro
\emph{e} um de propagação.
\end{remark}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g10:stats:1}
Aqui estão os números de livros lidos em um ano por $15$ alunos:
\[
0,\ 1,\ 1,\ 2,\ 2,\ 2,\ 3,\ 3,\ 4,\ 4,\ 5,\ 6,\ 7,\ 8,\ 12 .
\]
Calcule o significar, o \omterm{def:g10:stats:median}{mediana} e o \omterm{def:g10:stats:quartiles}{faixa} desta série.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g10:stats:2}
Um dado foi lançado $50$ vezes:
\begin{center}
\renewcommand{\arraystretch}{1.2}
\begin{tabular}{c|cccccc}
outcome & $1$ & $2$ & $3$ & $4$ & $5$ & $6$ \\
\hline
count & $7$ & $9$ & $8$ & $10$ & $6$ & $10$ \\
\end{tabular}
\end{center}
Calcule o freqüência de cada resultado e o significar resultado.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g10:stats:3}
Encontre o \omterm{def:g10:stats:median}{mediana}, $ Q_1$ e $ Q_3$ da série ordenada
\[
1,\ 2,\ 4,\ 4,\ 5,\ 6,\ 6,\ 7,\ 8,\ 9,\ 9,\ 11 \qquad (N = 12),
\]
e desenhe o box plot correspondente.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g10:stats:4}
O significar de $24$ pontuação dos testes é $11.5$. UM $25$ o aluno faz o teste
e pontuações $19$. Qual é o novo significar da aula?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:stats:5}
Um aluno tem significar de $12$ depois $5$ testes, todos com o mesmo peso.
Qual pontuação no $6$ o teste aumentaria o significar para $13$? É um significar de
$14$ alcançável (as pontuações estão fora de $20$)?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:stats:6}
Invente uma série de $7$ valores não negativos cujos \omterm{def:g10:stats:median}{mediana} é $10$ e cujo
significar é maior que $20$; então aquele cujo \omterm{def:g10:stats:median}{mediana} é $10$ e cujo significar é
menor que $6$. Por que pode o significar de tal série nunca será menor que
$\frac{40}{7}$? O que esses exemplos mostram?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:stats:7}
Numa empresa, o $9$ funcionários ganham $1800$ cada um por mês e o
diretor ganha $12000$.
\begin{enumerate}
  \item Calcule o significar e o \omterm{def:g10:stats:median}{mediana} salário.
  \item Qual indicador melhor descreve um salário “típico” aqui? Por que?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g10:stats:8}
Uma aula de $12$ meninos tem um significar altura de $168$ cm; o $18$ meninas do
mesma turma tem um significar altura de $163$ cm. Calcule o significar altura do
turma inteira. (Cuidado: é \emph{não} o \omterm{prop:g10:coordgeom:midpoint}{ponto médio} de $168$ e $163$
--- pondere as médias pelos tamanhos dos grupos.)
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g10:stats:9}
Uma série de $ N $ valores $ x_1, \dots, x_N $ tem significar $\bar x $. Cada valor é
transformado em $ y_i = a x_i + b $ para números fixos $ a $ e $ b $.
\begin{enumerate}
  \item Mostre que o significar da nova série é $ a \bar x + b $.
  \item As temperaturas de uma semana, em graus Celsius, significar $20$.
        Qual é o seu significar em graus Fahrenheit
        ($ F = 1.8\,C + 32$)?
\end{enumerate}
\end{exercise}

\section{Problema: A pessoa média não existe}

\begin{problem}[{Problema de fim de semana --- média em relação à mediana: valores discrepantes,
curvas de nivelamento, o armazém na estrada e o cockpit projetado
para ninguém}]
\label{pb:g10:stats:1}
Em 1950, a Força Aérea dos EUA mediu $4\,000$ pilotos em dez corpos
dimensões e construiu o cockpit para o \emph{homem comum}. Isso
equipado --- como um tenente descobriu contando --- essencialmente
ninguém. As estatísticas resumem multidões com números únicos, e cada
resumo diz a verdade sobre algo e mente sobre algo
mais. Este problema leva o significar e o \omterm{def:g10:stats:median}{mediana}
(\cref{def:g10:stats:mean}, \cref{def:g10:stats:median}) ao tribunal,
deixa cada um mostrar o que faz de melhor e fecha a caixa do cockpit.

\medskip
\noindent\textbf{Part I --- Two summaries on trial.}
\begin{enumerate}
  \item Uma rua vende sete casas, em milhares de euros:
        $200$, $210$, $220$, $230$, $240$, $250$ --- e um
        mansão em $1\,200$. Calcule o significar e o \omterm{def:g10:stats:median}{mediana}
        preço. Qual número descreve “uma casa nesta rua”?
  \item Remova a mansão e recalcule ambas. Por quanto
        cada movimento de resumo? Declare a moral sobre os valores discrepantes.
  \item Uma tabela de censo: fora de $100$ famílias, $20$ não tenho
        criança, $25$ tenha um, $30$ tem dois, $15$ tem três,
        $10$ tem quatro. Calcule o significar número de filhos. Não
        família ``tem $1.7$ crianças'' --- então qual é a quantidade real
        faz o significar codificar? (Multiplique por $100$.)
  \item Pelas quinze marcas
        $4, 6, 7, 8, 9, 10, 10, 11, 12, 12, 13, 14, 15, 17, 19$:
        encontre o \omterm{def:g10:stats:median}{mediana} e o quartis $ Q_1$, $ Q_3$
        (\cref{met:g10:stats:quartiles}).
  \item Para a mesma série, dê o \omterm{def:g10:stats:quartiles}{faixa} e o
        \omterm{def:g10:stats:quartiles}{intervalo interquartil}. O que cada um mede?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Grading curves and merged classes.}
\begin{enumerate}[resume]
  \item De \cref{exo:g10:stats:9}: mudando cada valor por
        $+b $ turnos significar, \omterm{def:g10:stats:median}{mediana} e quartis por $+b $; dimensionamento
        por $ a $ escala-los por $ a $. O que acontece com o \omterm{def:g10:stats:quartiles}{faixa} e
        o \omterm{def:g10:stats:quartiles}{intervalo interquartil} sob um turno? Sob uma escala?
        Justificar.
  \item Um teste tem significar $9$ (fora de $20$) e o professor quer
        significar $12$. Duas curvas são propostas: adicionar $3$ aponta para
        todos, ou multiplique cada marca por $\frac43$. Calcular
        o destino de um estudante em $6$ e um em $18$ sob cada
        curva. Qual curva favorece quem --- e qual tem uma
        problema no teto?
  \item Classe A ($20$ estudantes) tem significar $12$; classe B ($30$
        estudantes) tem significar $10$. Calcule o significar do $50$
        alunos juntos e explique por que não é $11$
        (a lição da média ponderada do média harmônica problema de fim de semana do volume do ensino médio, em
        a sala de aula).
  \item Mostre isso \omterm{def:g10:stats:median}{medianas} recuse este jogo: calcule o \omterm{def:g10:stats:median}{medianas}
        de $ A = \{1, 2, 3\}$ e $ B = \{0, 10\}$, então o
        \omterm{def:g10:stats:median}{mediana} da lista mesclada e compare com o
        média ponderada pelo tamanho dos dois \omterm{def:g10:stats:median}{medianas}.
  \item Patinação artística: pontuação de cinco juízes $5.2$, $5.3$, $5.4$,
        $5.5$, $9.9$. Calcule o significar, o \omterm{def:g10:stats:median}{mediana}, e o
        \emph{aparado significar} (descarte a média mais baixa e mais alta
        o resto). Por que os esportes mais julgados são cortados?
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Reading distributions.}
\begin{enumerate}[resume]
  \item Duas fábricas produzem parafusos de significar comprimento
        $50$ mm. Fábrica A: $ Q_1 = 49.9$, $ Q_3 = 50.1$; fábrica
        B: $ Q_1 = 48$, $ Q_3 = 52$. Você precisa de intercambiáveis
        parafusos: qual fábrica e qual estatística decidiu?
  \item Numa clínica, o \omterm{def:g10:stats:median}{mediana} espere é $10$ minutos, mas o
        significar espere é $25$. Que formato de dados esse par
        trair? Construa um conjunto de dados de cinco valores exatamente com isso
        \omterm{def:g10:stats:median}{mediana} e significar.
  \item ``Seu bebê está no percentil 75 de altura'' ---
        o que significa essa frase significar? E pode cada filho de um
        cidade esteja “acima da média”? Acima do \omterm{def:g10:stats:median}{mediana}? Explique o
        diferença.
  \item Classe A: \omterm{def:g10:functions:extrema}{mínimo} $4$, $ Q_1\,8$, \omterm{def:g10:stats:median}{mediana} $11$, $ Q_3\,13$, máx.
        $17$. Classe B: \omterm{def:g10:functions:extrema}{mínimo} $2$, $ Q_1\,9$, \omterm{def:g10:stats:median}{mediana} $11$, $ Q_3\,12$,
        máx. $19$. Compare: marca típica, homogeneidade do
        metade intermediária, selvageria dos extremos.
  \item A lista de verificação do jornalista: uma manchete diz “média
        salário na TechCorp: $5\,000$ euros''. Liste os três
        perguntas que este problema ensinou você a fazer antes
        acreditar que o título diz algo sobre um típico
        funcionário.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The cockpit case.}
\begin{enumerate}[resume]
  \item Construa sua própria evidência: construa cinco preços de casas
        com significar $300$ e \omterm{def:g10:stats:median}{mediana} $200$ (milhares de euros).
  \item O significar é o ponto de equilíbrio: prove a partir da definição
        que os desvios do significar sempre cancelar,
        $\sum (x_i - \bar x) = 0$ e verifique-o em seu conjunto de dados
        da questão 16.
  \item O \omterm{def:g10:stats:median}{mediana} é o minimizador de tarefas: as lojas ficam em
        quilômetros $1$, $3$, $4$, $9$, $13$ de uma estrada e um
        armazém deve minimizar o \emph{distância total} para o
        cinco lojas. Calcule esse total para um armazém no
        \omterm{def:g10:stats:median}{mediana} ($4$), no significar ($6$) e em $5$. Quem ganha?
        (Isto é geral: o \omterm{def:g10:stats:median}{mediana} minimiza total absoluto
        distância --- teste qualquer outro ponto.)
  \item A cabine, resolvido: suponha que um piloto tenha um $30\,\%$
        chance de estar “próximo da média” em um corpo
        dimensão, independentemente $10$ dimensões.
        Estime a chance de estar próximo da média em
        \emph{todos os dez} --- e conciliar o resultado com o que
        tenente encontrado em $4\,000$ pilotos. O que o ar
        Forçar a construção em vez do cockpit médio?
  \item Finale --- o guia do usuário, uma linha cada: quando os totais
        são o que importa (um orçamento, uma colheita), use o \dots;
        quando o indivíduo típico importa (salários, espera
        vezes), use o \dots; quando a justiça na distribuição é importante,
        relatar o \dots; quando houver ameaça de contaminação, use o
        \dots. Encerre o caso com o título do problema.
\end{enumerate}
\end{problem}
