\chapter{Geometría de coordenadas}\label{ch:g10:coordgeom}

La gran idea de Descartes fue describir puntos mediante pares de números, convirtiendo
problemas de geometría en cálculos. Con sólo dos fórmulas --- la
\omterm{prop:g10:coordgeom:midpoint}{punto medio} y la distancia --- se puede demostrar que un triángulo es isósceles,
que un cuadrilátero es un paralelogramo, o que tres puntos están alineados,
sin trazar una sola línea auxiliar.

\section{Coordenadas en el avión.}

\begin{definition}[sistema de coordenadas]\label{def:g10:coordgeom:system}
Un \emph{sistema de coordenadas}\index{sistema de coordenadas} del avión consta
de un origen $O$ y dos ejes graduados hasta $O$: el horizontal
Eje $x$ y el eje vertical $y$. Cada punto $M$ tiene un par único.
de \emph{coordenadas}\index{coordenadas} $(x, y)$: es
\emph{abscisa}\index{abscisa} $x$ y sus
\emph{ordenada}\index{ordenada} $y$. El sistema es
\emph{ortonormal}\index{sistema ortonormal} cuando los ejes están
perpendicular y lleva la misma unidad de longitud. Todos los sistemas en este
capítulo son ortonormales.
\end{definition}

\section{Punto medio de un segmento}

\begin{proposition}[Fórmula del punto medio]\label{prop:g10:coordgeom:midpoint}
Sean $A(x_A, y_A)$ y $B(x_B, y_B)$. El \emph{punto medio}\index{punto medio}
$I$ del segmento $[AB]$ tiene \omterm{def:g10:coordgeom:system}{coordenadas}
\[
I\left(\frac{x_A + x_B}{2},\ \frac{y_A + y_B}{2}\right).
\]
\end{proposition}

\begin{proof}
Considere el \omterm{def:g10:coordgeom:system}{coordenadas} horizontal. Pasando de $A$ a $B$, el \omterm{def:g10:coordgeom:system}{abscisa}
cambios por $x_B - x_A$; el punto a mitad de camino tiene \omterm{def:g10:coordgeom:system}{abscisa}
\[
x_A + \frac{x_B - x_A}{2} = \frac{2x_A + x_B - x_A}{2}
= \frac{x_A + x_B}{2},
\]
el promedio de los dos \omterm{def:g10:coordgeom:system}{abscisas}. El mismo cálculo se aplica a la
\omterm{def:g10:coordgeom:system}{ordenadas}.
\end{proof}

\begin{example}\label{ex:g10:coordgeom:midpoint}
Con $A(-1, 4)$ y $B(5, -2)$, el \omterm{prop:g10:coordgeom:midpoint}{punto medio} de $[AB]$ es
\[
I\left(\frac{-1 + 5}{2},\ \frac{4 + (-2)}{2}\right) = I(2, 1).
\]
La fórmula también se ejecuta al revés: si $A(-1, 4)$ y \omterm{prop:g10:coordgeom:midpoint}{punto medio} son
$I(2, 1)$, entonces $B$ satisface $\frac{-1 + x_B}{2} = 2$ y
$\frac{4 + y_B}{2} = 1$, entonces $x_B = 5$ y $y_B = -2$: el punto $B$ es el
\emph{simétrico} de $A$ sobre $I$.
\end{example}

\section{Distancia entre dos puntos}

\begin{theorem}[Fórmula de distancia]\label{thm:g10:coordgeom:distance}
En un \omterm{def:g10:coordgeom:system}{sistema ortonormal}, la distancia entre $A(x_A, y_A)$ y
$B(x_B, y_B)$ es
\[
AB = \sqrt{(x_B - x_A)^2 + (y_B - y_A)^2}.
\]
\end{theorem}

\begin{proof}
Sea $C$ el punto $(x_B, y_A)$: tiene la misma altura que $A$ y el
mismo \omterm{def:g10:coordgeom:system}{abscisa} que $B$, por lo que el triángulo $ACB$ tiene un ángulo recto en $C$. su
Los catetos son segmentos horizontales y verticales, de longitudes.
$AC = \abs{x_B - x_A}$ y $CB = \abs{y_B - y_A}$. Por el pitagórico
teorema,
\[
AB^2 = AC^2 + CB^2 = (x_B - x_A)^2 + (y_B - y_A)^2,
\]
y tomando la raíz cuadrada (ambos lados no negativos) se obtiene la fórmula.
\end{proof}

\begin{example}\label{ex:g10:coordgeom:distance}
Con $A(1, 1)$ y $B(5, 3)$:
\[
AB = \sqrt{(5-1)^2 + (3-1)^2} = \sqrt{16 + 4} = \sqrt{20} = 2\sqrt 5 .
\]
Las distancias generalmente se mantienen en forma exacta (raíz cuadrada); solo vuelta en el
final si se necesita una respuesta decimal.
\end{example}

\begin{remark}
No importa el orden de los puntos: $(x_A - x_B)^2 =
(x_B - x_A)^2$. ¡Pero no te olvides de los cuadrados! la distancia es
\emph{no} $\abs{x_B - x_A} + \abs{y_B - y_A}$.
\end{remark}

\section{Usando las fórmulas en geometría.}

\begin{method}[Naturaleza de un triangulo]\label{met:g10:coordgeom:triangle}
Dados tres puntos $A$, $B$, $C$ por su \omterm{def:g10:coordgeom:system}{coordenadas}:
\begin{enumerate}
  \item calcula las tres longitudes al cuadrado $AB^2$, $AC^2$, $BC^2$ con el
        fórmula de distancia (mantener cuadrados evita raíces cuadradas);
  \item dos longitudes iguales al cuadrado $\Rightarrow$ el triángulo es
        \emph{isósceles};
  \item si la longitud al cuadrado más grande es la suma de las otras dos, la
        triángulo es \emph{en ángulo recto} en el vértice opuesto al más largo
        lado, por el inverso del teorema de Pitágoras.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g10:coordgeom:triangle}
Sean $A(0, 1)$, $B(4, 3)$ y $C(2, -3)$. entonces
\begin{align*}
AB^2 &= (4-0)^2 + (3-1)^2 = 16 + 4 = 20, \\
AC^2 &= (2-0)^2 + (-3-1)^2 = 4 + 16 = 20, \\
BC^2 &= (2-4)^2 + (-3-3)^2 = 4 + 36 = 40 .
\end{align*}
Primero, $AB^2 = AC^2$, entonces $AB = AC$: el triángulo es isósceles en $A$.
Además $AB^2 + AC^2 = 40 = BC^2$, por lo contrario del pitagórico
teorema el triángulo también es rectángulo en $A$.
\end{example}

\begin{method}[Reconociendo un paralelogramo]\label{met:g10:coordgeom:parallelogram}
Un cuadrilátero $ABCD$ es un paralelogramo exactamente cuando sus diagonales
$[AC]$ y $[BD]$ tienen el mismo \omterm{prop:g10:coordgeom:midpoint}{punto medio}. Entonces: calcule tanto \omterm{prop:g10:coordgeom:midpoint}{puntos medios} con
la fórmula \omterm{prop:g10:coordgeom:midpoint}{punto medio} y compararlos.
\end{method}

\begin{example}\label{ex:g10:coordgeom:parallelogram}
Sean $A(-1, 0)$, $B(2, 2)$, $C(5, 1)$ y $D(2, -1)$. El \omterm{prop:g10:coordgeom:midpoint}{punto medio} de
$[AC]$ es $\left(\frac{-1+5}{2}, \frac{0+1}{2}\right) = (2, \frac12)$; el
\omterm{prop:g10:coordgeom:midpoint}{punto medio} de $[BD]$ es $\left(\frac{2+2}{2}, \frac{2-1}{2}\right) =
(2, \frac12)$. Coinciden, por lo que $ABCD$ es un paralelogramo.
\end{example}
