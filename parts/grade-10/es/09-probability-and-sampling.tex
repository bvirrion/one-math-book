\chapter{Probabilidad y muestreo}\label{ch:g10:proba}

\omterm{def:g10:proba:distribution}{Probabilidad} asigna números al azar. A partir de experimentos con
resultados igualmente probables --- dados, monedas, cartas --- este capítulo construye el
reglas básicas para calcular probabilidades de \omterm{def:g10:proba:events}{eventos}, presenta \omterm{met:g10:proba:tree}{árbol
diagramas} para experimentos de dos pasos y termina con el vínculo entre
probabilidades y frecuencias observadas. La teoría continúa en
\cref{ch:g11:prob}.

\section{Experimentos y eventos aleatorios.}

\begin{definition}[Resultados y eventos]\label{def:g10:proba:events}
Un \emph{experimento aleatorio}\index{experimento aleatorio} es un experimento cuyo
El resultado no se puede predecir (tirar un dado). El conjunto de todos los posibles.
El resultado es el \emph{espacio muestral}\index{espacio muestral}.
$\Omega = \{\omega_1, \dots, \omega_n\}$; sus elementos son los
\emph{resultados}. Un \emph{evento}\index{evento} es un conjunto de resultados; un
evento \emph{ocurre} cuando le pertenezca el resultado del experimento.
\end{definition}

\begin{definition}[Distribución de probabilidad]\label{def:g10:proba:distribution}
Asignando a cada resultado $\omega_i$ un número $p_i \geq 0$, con
$p_1 + \dots + p_n = 1$, define un \emph{probabilidad}\index{probabilidad}:
la probabilidad $\P(A)$ de un \omterm{def:g10:proba:events}{evento} $A$ es la suma de $p_i$ de los
resultados en $A$. En particular $\P(\Omega) = 1$ y lo imposible \omterm{def:g10:proba:events}{evento}
$\varnothing$ tiene probabilidad $0$.
\end{definition}

\begin{proposition}[Resultados igualmente probables]\label{prop:g10:proba:uniform}
Si los resultados $n$ son igualmente probables (cada $p_i = \frac1n$), entonces para
cada \omterm{def:g10:proba:events}{evento} $A$:
\[
\P(A) = \frac{\text{number of outcomes in } A}{n}
= \frac{\text{favorable cases}}{\text{possible cases}} .
\]
\end{proposition}

\begin{proof}
$A$ contiene un número $k$ de resultados, cada uno de \omterm{def:g10:proba:distribution}{probabilidad} $\frac1n$,
Entonces $\P(A) = \frac kn$.
\end{proof}

\begin{example}\label{ex:g10:proba:die}
Tira un dado justo: $\Omega = \{1, 2, 3, 4, 5, 6\}$, cada resultado tiene
\omterm{def:g10:proba:distribution}{probabilidad} $\frac16$. El \omterm{def:g10:proba:events}{evento} $A$: ``el resultado es par'' es
$A = \{2, 4, 6\}$, entonces $\P(A) = \frac36 = \frac12$. El \omterm{def:g10:proba:events}{evento} $B$: ``el
El resultado es al menos $5$'' es $B = \{5, 6\}$, por lo que $\P(B) = \frac26 =
\frac13$.
\end{example}

\section{Combinando eventos}

\begin{definition}[Intersección, unión, complemento]\label{def:g10:proba:operations}
Sean $A$ y $B$ \omterm{def:g10:proba:events}{eventos}.
\begin{itemize}
  \item $A \cap B$ ("$A$ y $B$") ocurre cuando ocurren ambos;
  \item $A \cup B$ ("$A$ o $B$") ocurre cuando ocurre al menos uno;
  \item o \emph{complementar}\index{complementar} $\bar A$ ("no $A$")
        ocurre exactamente cuando $A$ no;
  \item $A$ y $B$ son \emph{incompatible}\index{incompatible} cuando no pueden ocurrir
        juntos: $A \cap B = \varnothing$.
\end{itemize}
\end{definition}

\begin{theorem}[Reglas de adición]\label{thm:g10:proba:addition}
Para todos los \omterm{def:g10:proba:events}{eventos} $A$ y $B$:
\[
\P(A \cup B) = \P(A) + \P(B) - \P(A \cap B),
\qquad
\P(\bar A) = 1 - \P(A) .
\]
Para \omterm{def:g10:proba:operations}{incompatible} \omterm{def:g10:proba:events}{eventos}, la primera regla se simplifica a
$\P(A \cup B) = \P(A) + \P(B)$.
\end{theorem}

\begin{proof}
Sumando $\P(A) + \P(B)$ suma el \omterm{def:g10:proba:distribution}{probabilidad} de cada resultado de
$A \cup B$ una vez, \emph{excepto} las de $A \cap B$, que se cuentan
dos veces --- una vez en $A$ y una vez en $B$. Restando reparaciones $\P(A \cap B)$
el doble cómputo. Para \omterm{def:g10:proba:operations}{complementar}: $A$ y $\bar A$ son \omterm{def:g10:proba:operations}{incompatible}
y $A \cup \bar A = \Omega$, entonces $\P(A) + \P(\bar A) = \P(\Omega) = 1$.
\end{proof}

\begin{example}\label{ex:g10:proba:cards}
Roba una carta de una baraja de cartas estándar $52$. Sea $A$: ``la tarjeta es una
corazón'' (cartas $13$) y $B$: ``la carta es un rey'' (cartas $4$). entonces
$A \cap B$ es "el rey de corazones" (tarjeta $1$), y
\[
\P(A \cup B) = \frac{13}{52} + \frac{4}{52} - \frac{1}{52} = \frac{16}{52}
= \frac{4}{13}.
\]
La regla \omterm{def:g10:proba:operations}{complementar} suele ser la ruta más rápida: la \omterm{def:g10:proba:distribution}{probabilidad} que el
la tarjeta es \emph{no} y el corazón es $1 - \frac{13}{52} = \frac34$.
\end{example}

\section{Diagramas de árbol}

\begin{method}[Experimentos de dos pasos]\label{met:g10:proba:tree}
Para un experimento realizado en dos pasos:
\begin{enumerate}
  \item dibuja un \emph{árbol}\index{diagrama de arbol}: una rama por posible
        resultado del primer paso, luego de cada uno de estos, una rama por
        resultado del segundo paso, escribiendo el \omterm{def:g10:proba:distribution}{probabilidad} en cada rama
        (las ramas que salen de cualquier nodo deben sumar $1$);
  \item el \omterm{def:g10:proba:distribution}{probabilidad} de un \emph{camino} (una hoja del \omterm{met:g10:proba:tree}{árbol}) es el
        producto de las probabilidades a lo largo de sus ramas;
  \item el \omterm{def:g10:proba:distribution}{probabilidad} de un \omterm{def:g10:proba:events}{evento} es la suma de las probabilidades de
        los caminos que lo realizan.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g10:proba:tree}
Una urna contiene $3$ bolas rojas y $2$ bolas azules. Saca una bola, ponla.
Atrás, dibuja de nuevo. Cada sorteo da rojo (R) con \omterm{def:g10:proba:distribution}{probabilidad} $\frac35$ y
azul (B) con \omterm{def:g10:proba:distribution}{probabilidad} $\frac25$.

\section{Frecuencias y muestreo}

Cuando un \omterm{def:g10:proba:events}{experimento aleatorio} se repite muchas veces, el \omterm{def:g10:stats:series}{frecuencia} observado
de un \omterm{def:g10:proba:events}{evento} se acerca a su \omterm{def:g10:proba:distribution}{probabilidad} --- es por eso que \omterm{def:g10:proba:distribution}{probabilidad} es
útil para describir el mundo real.

\begin{definition}[muestra]\label{def:g10:proba:sample}
Un \emph{muestra}\index{muestra} de tamaño $n$ es el resultado de repetir el
mismo experimento $n$ veces de forma independiente. El \emph{observed \omterm{def:g10:stats:series}{frecuencia}} de
un \omterm{def:g10:proba:events}{evento} en la muestra es el número de apariciones dividido por $n$.
\end{definition}

\begin{example}\label{ex:g10:proba:sampling}
Una moneda justa da cara con \omterm{def:g10:proba:distribution}{probabilidad} $0.5$. Tirandolo $100$ veces
rara vez da exactamente cabezas $50$: \omterm{def:g10:proba:sample}{muestras} \emph{fluctuar}. Ejecuciones típicas de
Los lanzamientos de $100$ dan frecuencias como $0.46$, $0.53$, $0.49$ --- cercanas a
$0.5$, pero no iguales. Los \omterm{def:g10:proba:sample}{muestras} más grandes fluctúan menos: con $10\,000$
lanzamientos, las frecuencias observadas generalmente están dentro de aproximadamente $0.01$ de $0.5$.
\end{example}

\begin{remark}[Intervalo de fluctuación]
Una regla práctica útil, justificada en \cref{ch:g11:binom}: para un \omterm{def:g10:proba:sample}{muestra} de
tamaño $n$ de un \omterm{def:g10:proba:events}{evento} de \omterm{def:g10:proba:distribution}{probabilidad} $p$ (con $n$ razonablemente grande), el
observado \omterm{def:g10:stats:series}{frecuencia} se encuentra en el \omterm{def:g10:numbers:interval}{intervalo}
\[
\intcc{p - \frac{1}{\sqrt n}}{\,p + \frac{1}{\sqrt n}}
\]
en aproximadamente $95\%$ de \omterm{def:g10:proba:sample}{muestras}. Si un \omterm{def:g10:stats:series}{frecuencia} observado cae muy afuera
este \omterm{def:g10:numbers:interval}{intervalo}, duda del valor de $p$: este es el punto de partida de
pruebas estadísticas.
\end{remark}

\begin{example}\label{ex:g10:proba:test}
Una fábrica afirma que sólo $10\%$ de sus piezas están defectuosas
($p = 0.1$). En un \omterm{def:g10:proba:sample}{muestra} de piezas $n = 400$, $64$ están defectuosas: observado
\omterm{def:g10:stats:series}{frecuencia} $\frac{64}{400} = 0.16$. La fluctuación \omterm{def:g10:numbers:interval}{intervalo} es
$\intcc{0.1 - \frac{1}{20}}{0.1 + \frac{1}{20}} = \intcc{0.05}{0.15}$,
y $0.16$ queda fuera de él: el \omterm{def:g10:proba:sample}{muestra} arroja serias dudas sobre la afirmación.
\end{example}

\section{Ceremonias}

\begin{exercise}[$\star$]\label{exo:g10:proba:1}

Una bolsa contiene canicas verdes $5$, amarillas $3$ y negras $2$; una canica es

\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:proba:3}
En una clase de estudiantes $30$, $18$ estudia español, $10$ estudia alemán y

probabilidades en el segundo nivel y calcular el \omterm{def:g10:proba:distribution}{probabilidad} del dibujo

\section{Problema: Las tres puertas que engañaron a mil
matemáticos}

\begin{problem}[{Problema del fin de semana --- la paradoja de Monty Hall y
sus primos: cómo la información cambia la probabilidad y cómo las muestras
separar afirmaciones de ilusiones}]
\label{pb:g10:proba:1}
En 1990, la columnista Marilyn vos Savant publicó un acertijo sobre
un presentador de concurso, tres puertas, un coche y dos cabras --- y
Respondió correctamente. Diez mil lectores, entre ellos cientos
de doctorados en matemáticas, escribió para insistir en que estaba equivocada. esto
El problema te arma con \omterm{met:g10:proba:tree}{árboles} (\cref{met:g10:proba:tree}),
\omterm{def:g10:proba:operations}{complementos} y las reglas de suma
(\cref{thm:g10:proba:addition}), permite resolver la discusión
usted mismo, y termina con la herramienta que arbitra cada uno de esos
Disputa en la práctica: el fluctuante \omterm{def:g10:proba:sample}{muestra}.

\medskip
\noindent\textbf{Parte I --- The rules, warmed up.}
\begin{enumerate}
  \item Una carta de una baraja de cartas estándar $52$: calcular
        $P(\text{king or heart})$ con la regla de suma ---
        y decir por qué $\frac{4}{52} + \frac{13}{52}$ solo es
        mal.
  \item Tres lanzamientos de moneda justos: calcule $P(\text{at least one
        head})$ mediante \omterm{def:g10:proba:operations}{complementar}.
  \item Una bolsa contiene canicas rojas $3$ y azules $2$; dos estan dibujados
        sin reemplazo. Con un \omterm{met:g10:proba:tree}{árbol}, calcule
        $P(\text{two reds})$ y $P(\text{one of each color})$.
  \item Complete la distribución de la pregunta 3 con
        $P(\text{two blues})$ y comprueba las tres probabilidades.
        suma a $1$.
  \item Alguien informa, para dos \emph{\omterm{def:g10:proba:operations}{incompatible}} \omterm{def:g10:proba:events}{eventos},
        $P(A) = 0.6$ y $P(B) = 0.5$. Condenarlos con uno
        línea de aritmética.
\end{enumerate}

\medskip
\noindent\textbf{Parte II --- The three doors.}
El juego: un coche se esconde detrás de una de las tres puertas, detrás unas cabras
los otros dos. Eliges una puerta (digamos la puerta 1). El anfitrión --- quien
\emph{sabe} donde esta el auto --- abre uno de los otros dos
puertas, siempre revelando una cabra, y te ofrece la oportunidad de
cambie a la puerta cerrada restante.
\begin{enumerate}[resume]
  \item Antes de calcular algo: ¿cambiar ayuda, perjudica o
        no importa? Escriba su respuesta instintiva, honestamente.
  \item Ahora el \omterm{met:g10:proba:tree}{árbol}, por casos sobre la verdadera posición del auto.
        (cada $\frac13$): si el automóvil está detrás de su puerta,
        el cambio pierde; Si está detrás de cualquier otra puerta, ¿qué
        ¿Qué debe hacer el anfitrión y qué gana entonces el cambio?
        Calcule $P(\text{win by switching})$.
  \item La versión de una línea: cambiar gana exactamente cuando su
        La primera elección fue incorrecta. Concluir de nuevo.
  \item Ahora la variante crucial --- ``Monty Fall'': el anfitrión
        \emph{no} sabe dónde está el auto? Abre uno de los
        Otras puertas al azar y por suerte revela una cabra. Lista
        los seis escenarios igualmente probables (posición del automóvil $\times$
        puerta abierta), descartar aquellas en las que se revela el coche,
        y calcular el \omterm{def:g10:proba:distribution}{probabilidad} que el cambio gana
        \emph{entre los escenarios sobrevivientes}. ¿Qué cambió y
        ¿Por qué es importante el conocimiento del anfitrión?
  \item La bomba de la intuición: cien puertas, eliges una, la
        el anfitrión conocedor abre las puertas de la cabra $98$. ¿Cambiar? con que
        \omterm{def:g10:proba:distribution}{probabilidad} de ganar?
  \item Te enfrentas a un escéptico que, como los mil escritores de cartas,
        insiste en "50--50". Describir una simulación basada en tarjetas.
        (tres cartas, muchas rondas, un amigo como anfitrión) y di
        ¿Qué proporción de ganancias de cambio espera después de $300$?
        rondas --- con la fluctuación \omterm{def:g10:numbers:interval}{intervalo} del capítulo
        como su margen.
\end{enumerate}

\medskip
\noindent\textbf{Parte III --- Cousins of the paradox.}
\begin{enumerate}[resume]
  \item Una familia tiene dos hijos; aprendes que \emph{al menos
        uno es un niño}. Enumere los dos hijos igualmente probables
        configuraciones compatibles con la información, y
        calcular $P(\text{two boys})$.
  \item Misma familia, pero la información ahora es: \emph{el mayor
        el niño es un niño}. Vuelva a calcular. ¿Por qué las dos respuestas?
        Se diferencian y ¿qué comparten con los dos Monty?
  \item Tres monedas justas: calcule $P(\text{all three alike})$.
        El matemático d'Alembert defendió en 1754 otra
        respuesta: ``o son todos iguales o no lo son ---
        un caso de cada dos''. Nombra el error exacto, con
        \cref{prop:g10:proba:uniform} en mano.
  \item Un juego cuesta $2$ euros: tira un dado, gana $6$ euros en
        seis, $3$ euros por cinco, nada más
        (\cref{exo:g10:proba:9}). Más de juegos $600$, ¿qué
        ¿Qué ganancia promedio por juego predice el conteo? un
        aseguradora juega las mismas matemáticas: un $1\,\%$ anual
        probabilidad de una reclamación de $10\,000$ euros --- ¿cuál es el
        \emph{justo} prima, y ¿por qué la prima real
        ¿superarlo?
\end{enumerate}

\medskip
\noindent\textbf{Parte IV --- \omterm{def:g10:proba:sample}{Muestras} as arbiters.}
\begin{enumerate}[resume]
  \item Se lanza una moneda $100$ veces. Usando la fluctuación
        \omterm{def:g10:numbers:interval}{intervalo} $p \pm \frac{1}{\sqrt n}$, decide cuál de
        estos resultados deberían sorprender: $55$ cabezas; $70$
        cabezas.
  \item Una encuesta de votantes $1\,000$ arroja un candidato $52\,\%$.
        Calcule la fluctuación \omterm{def:g10:numbers:interval}{intervalo} alrededor de $50\,\%$ y
        decir si en la encuesta \emph{prueba} el candidato es
        adelante.
  \item Una fábrica reclama piezas defectuosas $4\,\%$. Un \omterm{def:g10:proba:sample}{muestra} de
        $400$ muestra $28$ defectuoso ($7\,\%$); un \omterm{def:g10:proba:sample}{muestra} posterior de
        $2\,500$ muestra $175$ ($7\,\%$ nuevamente). Pruebe el reclamo
        contra cada \omterm{def:g10:proba:sample}{muestra}. ¿Por qué difieren los veredictos?
  \item Dos fuentes de error acechan en todas las encuestas: el muestreo
        \emph{inclinación} (el problema de fin de semana de cómo se encuentran los datos del volumen de la escuela secundaria) y muestreo
        \emph{fluctuación} (este capítulo). ¿Cuál se encoge como
        $n$ crece y ¿cuál sobrevive a cualquier tamaño de \omterm{def:g10:proba:sample}{muestra}? uno
        frase cada uno.
  \item Finale --- la lista de verificación del probabilista, una línea por
        ítem: resultados igualmente probables antes de contar
        (pregunta 15); \omterm{met:g10:proba:tree}{árboles} y \omterm{def:g10:proba:operations}{complementos} para varios pasos y
        ``al menos'' (preguntas 2--3); \emph{información
        cambia las probabilidades, y cómo se obtuvo importa}
        (preguntas 7--14); \omterm{def:g10:proba:sample}{muestras} fluctúa aproximadamente
        $\frac{1}{\sqrt n}$ y arbitrajes de simulación.
        (preguntas 11, 16--18).
\end{enumerate}
\end{problem}
