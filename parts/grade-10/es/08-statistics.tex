\chapter{Estadísticas descriptivas}\label{ch:g10:stats}

Las estadísticas resumen una gran colección de números: grados, salarios,
temperaturas --- según unos pocos indicadores bien elegidos. Este capítulo presenta
las dos familias de indicadores: los que localizan el \emph{centro} del
datos (\omterm{def:g10:stats:mean}{significar}, \omterm{def:g10:stats:median}{mediana}) y aquellos que miden su \emph{desparramar} (\omterm{def:g10:stats:quartiles}{rango},
\omterm{def:g10:stats:quartiles}{cuartiles}). Entra un tratamiento más profundo, que incluye la desviación estándar.
\cref{ch:g11:stat}.

\section{Datos y frecuencias}

\begin{definition}[serie estadistica]\label{def:g10:stats:series}
A \emph{serie estadística}\index{serie estadística} es una lista de valores
$x_1, x_2, \dots, x_N$ observado en una población de tamaño $N$. cuando un valor
$x$ aparece $n$ veces, $n$ es su \emph{contar} y el cociente
\[
f = \frac{n}{N}
\]
es su \emph{frecuencia}\index{frecuencia} (a menudo expresado como
porcentaje). Las frecuencias de todos los valores suman $1$.
\end{definition}

\begin{example}\label{ex:g10:stats:frequencies}
Las calificaciones de los estudiantes $20$ en una prueba de puntos $5$:
\begin{center}
\renewcommand{\arraystretch}{1.2}
\begin{tabular}{c|cccccc}
grade $x$ & $0$ & $1$ & $2$ & $3$ & $4$ & $5$ \\
\hline
count $n$ & $1$ & $2$ & $4$ & $6$ & $5$ & $2$ \\
\omterm{def:g10:stats:series}{frecuencia} $f$ & $0.05$ & $0.10$ & $0.20$ & $0.30$ & $0.25$ & $0.10$ \\
\end{tabular}
\end{center}
Verifique: $1 + 2 + 4 + 6 + 5 + 2 = 20$ y las frecuencias suman $1$.
\end{example}

\section{medidas de centro}

\begin{definition}[Medio]\label{def:g10:stats:mean}
El \emph{significar}\index{significar} de la serie es
\[
\bar x = \frac{x_1 + x_2 + \dots + x_N}{N}.
\]
Si los valores $v_1, \dots, v_k$ aparecen con recuentos $n_1, \dots, n_k$, el
Se calcula el mismo número que \emph{media ponderada}.
\[
\bar x = \frac{n_1 v_1 + n_2 v_2 + \dots + n_k v_k}{n_1 + n_2 + \dots + n_k}.
\]
\end{definition}

\begin{example}\label{ex:g10:stats:mean}
Para el cuestionario de \cref{ex:g10:stats:frequencies}:
\[
\bar x = \frac{1 \times 0 + 2 \times 1 + 4 \times 2 + 6 \times 3
+ 5 \times 4 + 2 \times 5}{20}
= \frac{0 + 2 + 8 + 18 + 20 + 10}{20}
= \frac{58}{20} = 2.9 .
\]
\end{example}

\begin{definition}[mediana]\label{def:g10:stats:median}
El \emph{mediana}\index{mediana} es un valor que divide el \emph{ordenado}
serie en dos mitades: al menos la mitad de los valores son $\leq$ la mediana, y
al menos la mitad son $\geq$. En la práctica, ordene los valores $N$;
\begin{itemize}
  \item si $N$ es impar, la mediana es el valor medio, en posición
        $\frac{N+1}{2}$;
  \item si $N$ es par, tome el \omterm{prop:g10:coordgeom:midpoint}{punto medio} de los dos valores medios, en
        posiciones $\frac N2$ y $\frac N2 + 1$.
\end{itemize}
\end{definition}

\begin{example}\label{ex:g10:stats:median}
Serie ordenada de precios de viviendas $7$ (en miles): $120$, $150$, $160$,
$180$, $210$, $240$, $900$. El \omterm{def:g10:stats:median}{mediana} es el valor $4$, $180$. El \omterm{def:g10:stats:mean}{significar}
es $\frac{1960}{7} = 280$ --- ¡mayor que $6$ de los precios $7$!
Un valor extremo ($900$) tira del \omterm{def:g10:stats:mean}{significar} mucho más que del \omterm{def:g10:stats:median}{mediana}: el
\omterm{def:g10:stats:median}{mediana} es \emph{robusto}, \omterm{def:g10:stats:mean}{significar} no lo es.
\end{example}

\section{Medidas de propagación}

\begin{definition}[rango, cuartiles]\label{def:g10:stats:quartiles}
Para una serie ordenada:
\begin{itemize}
  \item el \emph{rango}\index{rango} es la diferencia entre el
        valores más grandes y más pequeños;
  \item el \emph{primer cuartil}\index{cuartilla} $Q_1$ es el más pequeño
        valor tal que al menos una cuarta parte ($25\%$) de los valores sean
        $\leq Q_1$; el \emph{tercer cuartil} $Q_3$ es el valor más pequeño
        tal que al menos tres cuartas partes ($75\%$) de los valores sean
        $\leq Q_3$;
  \item el \emph{rango intercuartil}\index{rango intercuartil} $Q_3 - Q_1$ mide la propagación de
        la mitad central de los datos.
\end{itemize}
\end{definition}

\begin{method}[Encontrar los cuartiles]\label{met:g10:stats:quartiles}
Para una serie de valores ordenados $N$:
\begin{enumerate}
  \item calcula $\frac N4$; si no es \omterm{def:g10:numbers:sets}{entero}, redondee \emph{up} a
        el siguiente \omterm{def:g10:numbers:sets}{entero}; $Q_1$ es el valor en esa posición;
  \item calcula $\frac{3N}{4}$; redondear si es necesario; $Q_3$ es el valor en
        esa posición.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g10:stats:quartiles}
Tome los valores ordenados $11$
\[
2,\ 3,\ 5,\ 5,\ 6,\ 7,\ 8,\ 9,\ 9,\ 10,\ 12 .
\]
\omterm{def:g10:stats:median}{Mediana}: posición $\frac{11+1}{2} = 6$, por lo que \omterm{def:g10:stats:median}{mediana} es $7$.
$Q_1$: $\frac{11}{4} = 2.75$, redondear hacia arriba a $3$; el valor $3$ es $5$.
$Q_3$: $\frac{33}{4} = 8.25$, redondear hacia arriba a $9$; el valor $9$ es $9$.
\omterm{def:g10:stats:quartiles}{Rango}: $12 - 2 = 10$; \omterm{def:g10:stats:quartiles}{rango intercuartil}: $9 - 5 = 4$.
\end{example}

\begin{example}[Comparando dos grupos]\label{ex:g10:stats:comparison}
Dos clases toman el mismo examen (puntuaciones de $20$):
\begin{center}
\renewcommand{\arraystretch}{1.2}
\begin{tabular}{c|ccc}
 & \omterm{def:g10:stats:median}{mediana} & $Q_1$ & $Q_3$ \\
\hline
class A & $12$ & $10$ & $14$ \\
class B & $12$ & $6$ & $17$ \\
\end{tabular}
\end{center}
Mismo \omterm{def:g10:stats:median}{mediana}, spreads muy diferentes: la clase A es homogénea (la mitad de su
puntuaciones en $\intcc{10}{14}$), la clase B es heterogénea (su mitad central
abarca $\intcc{6}{17}$). Los indicadores del centro por sí solos nunca dicen el todo
historia.
\end{example}

(y es necesario para calcular totales), el \omterm{def:g10:stats:median}{mediana} resiste valores extremos. un

\section{Problema: la persona promedio no existe}

\begin{problem}[{Problema de fin de semana --- media contra mediana: valores atípicos,
curvas niveladoras, el almacén en la carretera y la cabina diseñada
para nadie}]
\label{pb:g10:stats:1}
En 1950, la Fuerza Aérea de EE. UU. midió a $4\,000$ pilotos en diez cuerpos.
dimensiones y construyó la cabina para el \emph{hombre promedio}. eso
equipado --- como descubrió un teniente al contar --- esencialmente
nadie. Las estadísticas resumen las multitudes con números únicos, y cada
resumen dice la verdad sobre algo y miente sobre algo
más. Este problema toma el \omterm{def:g10:stats:mean}{significar} y el \omterm{def:g10:stats:median}{mediana}.
