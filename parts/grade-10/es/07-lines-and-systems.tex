\chapter{Ecuaciones de rectas y sistemas lineales}\label{ch:g10:lines}

Una línea recta en el plano de coordenadas se describe mediante \omterm{def:g10:algebra:equation}{ecuación}, y
dos líneas juntas forman un sistema de \omterm{def:g10:algebra:equation}{ecuaciones} cuyas soluciones son sus
\omterm{def:g10:numbers:interunion}{intersección} puntos. Este capítulo conecta tres lenguajes: geométrico.
(líneas), algebraico (\omterm{def:g10:algebra:equation}{ecuaciones}) y funcional (\omterm{def:g10:reffunc:affine}{funciones afines} de
\cref{ch:g10:reffunc}) --- y enseña las dos formas estándar de resolver un
\omterm{def:g10:lines:system}{sistema lineal}.

\section{La ecuación de una recta.}

\begin{theorem}[Ecuación reducida]\label{thm:g10:lines:reduced}
En un \omterm{def:g10:coordgeom:system}{sistema de coordenadas}:
\begin{enumerate}
  \item cada línea no vertical tiene un \omterm{def:g10:algebra:equation}{ecuación} de la forma
        \[
        y = mx + p ,
        \]
        con $m$ (el \emph{pendiente}\index{pendiente}) y $p$ (el
        \emph{$y$-intercepción}) determinado de forma única --- este es el
        \emph{ecuación reducida}\index{ecuación reducida} de la línea;
  \item cada línea vertical tiene un \omterm{def:g10:algebra:equation}{ecuación} $x = c$;
  \item un punto se encuentra en la recta exactamente cuando su \omterm{def:g10:coordgeom:system}{coordenadas} satisface la
        \omterm{def:g10:algebra:equation}{ecuación}.
\end{enumerate}
\end{theorem}
\admitted

\begin{proposition}[pendiente por dos puntos]\label{prop:g10:lines:slope}
El \omterm{def:g10:reffunc:affine}{pendiente} de la línea (no vertical) hasta $A(x_A, y_A)$ y
$B(x_B, y_B)$, con $x_A \neq x_B$, es
\[
m = \frac{y_B - y_A}{x_B - x_A}
\qquad
\text{(``change in $y$ over change in $x$'').}
\]
\end{proposition}

\begin{proof}
Ambos puntos satisfacen $y = mx + p$, por lo que $y_B - y_A = (mx_B + p) - (mx_A + p)
= m(x_B - x_A)$; dividir por $x_B - x_A \neq 0$.
\end{proof}

\begin{method}[Encontrar la ecuación de una recta que pasa por dos puntos.]\label{met:g10:lines:through}
Dado $A$ y $B$ con $x_A \neq x_B$:
\begin{enumerate}
  \item calcula el \omterm{def:g10:reffunc:affine}{pendiente} $m = \dfrac{y_B - y_A}{x_B - x_A}$;
  \item escribe $y = mx + p$ y sustituye el \omterm{def:g10:coordgeom:system}{coordenadas} por $A$ (o $B$)
        para encontrar $p$;
  \item comprueba el \omterm{def:g10:algebra:equation}{ecuación} final con el punto \emph{otro}.
\end{enumerate}
Si $x_A = x_B$, la línea es vertical: su \omterm{def:g10:algebra:equation}{ecuación} es $x = x_A$.
\end{method}

\begin{example}\label{ex:g10:lines:through}
Línea por $A(1, 5)$ y $B(3, 1)$. \omterm{def:g10:reffunc:affine}{pendiente}:
$m = \frac{1 - 5}{3 - 1} = \frac{-4}{2} = -2$. Entonces $y = -2x + p$; el
el punto $A$ da $5 = -2 \times 1 + p$, por lo que $p = 7$. \omterm{def:g10:algebra:equation}{Ecuación}:
$y = -2x + 7$. Consulte con $B$: $-2 \times 3 + 7 = 1$.
\end{example}

\begin{proposition}[Líneas paralelas]\label{prop:g10:lines:parallel}
Dos rectas no verticales son paralelas si y sólo si tienen las mismas
\omterm{def:g10:reffunc:affine}{pendiente}. Dos líneas paralelas distintas no tienen un punto común; dos líneas con
diferentes \omterm{def:g10:reffunc:affine}{pendientes} tienen exactamente uno.
\end{proposition}

\begin{proof}
Sean $y = mx + p$ y $y = m'x + p'$ las dos líneas. Un punto común
resuelve $mx + p = m'x + p'$, es decir\ $(m - m')x = p' - p$. Si $m \neq m'$,
esto tiene exactamente una solución $x = \frac{p' - p}{m - m'}$: una
\omterm{def:g10:numbers:interunion}{intersección} punto. Si $m = m'$, \omterm{def:g10:algebra:equation}{ecuación} dice $0 = p' - p$: no
solución cuando $p \neq p'$ (paralelos distintos), cada $x$ cuando $p = p'$
(misma línea). Por lo tanto, el paralelismo (no tener \omterm{def:g10:numbers:interunion}{intersección}, o ser igual) es
equivalente a $m = m'$.
\end{proof}

\begin{remark}[Vector de dirección]
La línea $y = mx + p$ sube unidades $m$ por cada unidad a la derecha, por lo que
\omterm{def:g10:vectors:vector}{vector} $\vec u\,(1, m)$ es un \emph{dirección
vector}\index{vector de direccion} de la recta: la recta es paralela a
$\vec u$. Dos rectas son paralelas exactamente cuando sus direcciones \omterm{def:g10:vectors:vector}{vectores} son
\omterm{def:g10:vectors:collinear}{colineal} (\cref{thm:g10:vectors:det}).
\end{remark}

\section{Sistemas lineales}

\begin{definition}[sistema lineal]\label{def:g10:lines:system}
Un \emph{sistema de dos ecuaciones lineales}\index{sistema lineal} en el
desconocidos $x$ y $y$ es un par de \omterm{def:g10:algebra:equation}{ecuaciones}
\[
\begin{cases}
a x + b y = c \\
a' x + b' y = c'
\end{cases}
\]
estar satisfecho \emph{simultáneamente}. Un \emph{solución} es un par
$(x, y)$ satisfaciendo a ambos. Cada \omterm{def:g10:algebra:equation}{ecuación} describe una línea, por lo que resolver la
El sistema significa encontrar el \omterm{def:g10:numbers:interunion}{intersección} de dos líneas.
\end{definition}

\begin{method}[Sustitución]\label{met:g10:lines:substitution}
\begin{enumerate}
  \item Resuelva un \omterm{def:g10:algebra:equation}{ecuación} para una incógnita --- elija la más fácil, por ejemplo\
        expresar $y$ en términos de $x$;
  \item sustituye esta expresión en \emph{otro} \omterm{def:g10:algebra:equation}{ecuación}, que
        ahora contiene sólo $x$;
  \item resuelva para $x$, luego calcule $y$ desde el paso 1;
  \item verifique el par en ambos \omterm{def:g10:algebra:equation}{ecuaciones} originales.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g10:lines:substitution}
Resolver $\begin{cases} x + 2y = 8 \\ 3x - y = 3 \end{cases}$ por
sustitución. El segundo \omterm{def:g10:algebra:equation}{ecuación} da $y = 3x - 3$. Sustituir en el
primero:
\[
x + 2(3x - 3) = 8
\ \Longleftrightarrow\
x + 6x - 6 = 8
\ \Longleftrightarrow\
7x = 14
\ \Longleftrightarrow\
x = 2 ,
\]
luego $y = 3 \times 2 - 3 = 3$. La solución es el par $(2, 3)$.
Verifique: $2 + 2 \times 3 = 8$ y $3 \times 2 - 3 = 3$.
\end{example}

\begin{method}[Eliminación]\label{met:g10:lines:elimination}
\begin{enumerate}
  \item Multiplica cada \omterm{def:g10:algebra:equation}{ecuación} por un número bien elegido para que uno desconocido
        obtiene coeficientes opuestos en los dos \omterm{def:g10:algebra:equation}{ecuaciones};
  \item añade el \omterm{def:g10:algebra:equation}{ecuaciones}: esa incógnita desaparece;
  \item resuelva la incógnita restante \omterm{def:g10:algebra:equation}{ecuación} y luego sustitúyala por
        encontrar al otro desconocido;
  \item verifique el par en ambos \omterm{def:g10:algebra:equation}{ecuaciones} originales.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g10:lines:elimination}
Resolver $\begin{cases} 2x + 3y = 7 \\ 5x - 2y = 8 \end{cases}$ por
eliminación. Para eliminar $y$, multiplique el primer \omterm{def:g10:algebra:equation}{ecuación} por $2$ y el
segundo por $3$:
\[
\begin{cases}
4x + 6y = 14 \\
15x - 6y = 24
\end{cases}
\]
Agregándolos: $19x = 38$, entonces $x = 2$. Sustituyendo en $2x + 3y = 7$:
$4 + 3y = 7$, entonces $y = 1$. Solución: $(2, 1)$. Consulta en el segundo
\omterm{def:g10:algebra:equation}{ecuación}: $5 \times 2 - 2 \times 1 = 8$.
\end{example}

\begin{remark}[¿Cuántas soluciones?]
Como dos líneas, un \omterm{def:g10:lines:system}{sistema lineal} tiene exactamente una solución en general
(líneas no paralelas), y excepcionalmente ninguna (líneas paralelas distintas) o
infinitamente muchos (dos veces la misma línea). El criterio de
\cref{thm:g10:vectors:det} decide: el sistema tiene una solución única
exactamente cuando $ab' - a'b \neq 0$.
\end{remark}

\begin{example}[Modelado con un sistema]\label{ex:g10:lines:modeling}
Tres cuadernos y dos bolígrafos cuestan $8$; cuesta un cuaderno y cuatro bolígrafos
$6$. Sea $x$ el precio de una libreta y $y$ el de un bolígrafo:
\[
\begin{cases}
3x + 2y = 8 \\
x + 4y = 6 .
\end{cases}
\]
Del segundo \omterm{def:g10:algebra:equation}{ecuación}, $x = 6 - 4y$; sustituyendo,
$3(6 - 4y) + 2y = 8$, entonces $18 - 10y = 8$, dando $y = 1$ y luego
$x = 2$. Una libreta cuesta $2$ y un bolígrafo cuesta $1$.
\end{example}

\section{Problema: una solución, ninguna o infinitas soluciones}

\begin{problem}[{Problema de fin de semana --- dos líneas tienen tres posibles
destinos, un determinante decide entre ellos y el mundo.
el libro de texto más antiguo ya lo sabía}]
\label{pb:g10:lines:1}
Dos lineales \omterm{def:g10:algebra:equation}{ecuaciones}, dos incógnitas: el par de rectas que dibujan
puede cruzarse una vez, nunca, o ser la misma línea --- y
\emph{cada} \omterm{def:g10:lines:system}{sistema lineal} hereda uno de estos tres destinos.
Este problema los clasifica, cumple con el número que decide (un
viejo conocido del capítulo \omterm{def:g10:vectors:vector}{vectores}), resuelve el
faisanes y conejos chinos de dos mil años de antigüedad
\emph{Nueve capítulos}, y termina con esquinas de regiones --- el
Primer paso de la optimización utilizada por cada aerolínea y fábrica.
hoy.

\medskip
\noindent\textbf{Parte I --- The line, fluently.}
\begin{enumerate}
  \item Encuentre el \omterm{thm:g10:lines:reduced}{ecuación reducida} de la línea a través
        $(1, 2)$ y $(3, 8)$
        (\cref{met:g10:lines:through}).
  \item Entre $y = 3x - 1$, $y = 3x + 4$ y $y = -x + 7$: cuál
        dos rectas son paralelas
        (\cref{prop:g10:lines:parallel}), y ¿dónde están los
        ¿Se encuentran pares no paralelos? (Calcule uno \omterm{def:g10:numbers:interunion}{intersección}.)
  \item La línea que pasa por $(2, 1)$ y $(2, 5)$: ¿por qué tiene
        no \omterm{thm:g10:lines:reduced}{ecuación reducida} $y = mx + p$, y cuál es su
        \omterm{def:g10:algebra:equation}{ecuación}?
  \item ¿Está el punto $(4, 11)$ en la línea $y = 3x - 1$? y
        $(-1, -5)$?
  \item La línea que pasa por $(0, 3)$ con \omterm{def:g10:reffunc:affine}{pendiente} $-2$: dé su
        \omterm{def:g10:algebra:equation}{ecuación}, sus dos ejes se interceptan y el área del
        triángulo que corta del primer cuadrante.
\end{enumerate}

\noindent\textbf{Parte II --- Three destinies.}
\begin{enumerate}[resume]
  \item Resolver por sustitución

  \item Resolver por eliminación

        \begin{cases} 2x - y = 3 \\ 4x - 2y = 1 . \end{cases}

        Indique la tricotomía completa: cruzada, paralela,

  \item ¿Para qué valor de $m$ no

\begin{enumerate}[resume]

\end{problem}
