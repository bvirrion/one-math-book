\chapter{Funciones de referencia}\label{ch:g10:reffunc}

Un puñado de \omterm{def:g10:functions:function}{funciones} simples --- afines, cuadrados, inversos, raíz cuadrada,
cubo --- aparecen en todas partes, solos o combinados. Conociendo su \omterm{def:g10:functions:graph}{graficos} y
Las variaciones de memoria convierten muchos problemas en un boceto rápido. este capitulo
estudia cada uno de ellos por turno y demuestra sus variaciones con el
Técnica de comparación de \cref{ch:g10:functions}.

\section{Funciones afines}

\begin{definition}[Función afín]\label{def:g10:reffunc:affine}
Un \emph{función afín}\index{función afín} es un \omterm{def:g10:functions:function}{función} de la forma
\[
f(x) = mx + p,
\]
donde $m$ y $p$ son \omterm{def:g10:numbers:sets}{números reales} fijos. Su \omterm{def:g10:functions:graph}{gráfico} es una línea recta:
$m$ es el \emph{pendiente}\index{pendiente} y $p$ el
\emph{$y$-intercepción}\index{intersección y@$y$-intersección} (la línea cruza
el eje vertical en $(0, p)$). Cuando $p = 0$ el \omterm{def:g10:functions:function}{función} es
\emph{lineal}: $f(x) = mx$ expresa proporcionalidad.
\end{definition}

\begin{proposition}[pendiente y variaciones]\label{prop:g10:reffunc:affine}
Deje $f(x) = mx + p$.
\begin{enumerate}
  \item Para dos entradas distintas $u \neq v$:
        $m = \dfrac{f(v) - f(u)}{v - u}$ (el \omterm{def:g10:reffunc:affine}{pendiente} es el cambio del
        salida por unidad de cambio de la entrada).
  \item Si $m > 0$, $f$ es estrictamente \omterm{def:g10:functions:variations}{creciente} en $\R$; si $m < 0$,
        estrictamente \omterm{def:g10:functions:variations}{decreciente}; si $m = 0$, constante.
\end{enumerate}
\end{proposition}

\begin{proof}
1. Calcule $f(v) - f(u) = (mv + p) - (mu + p) = m(v - u)$ y luego divida entre
$v - u \neq 0$.

2. Si $u < v$, entonces $f(v) - f(u) = m(v-u)$ tiene el signo de $m$, ya que
$v - u > 0$: positivo $m$ da $f(u) < f(v)$ (\omterm{def:g10:functions:variations}{creciente}), negativo $m$
da $f(u) > f(v)$ (\omterm{def:g10:functions:variations}{decreciente}).
\end{proof}

\begin{example}\label{ex:g10:reffunc:affine}
Encuentre el \omterm{def:g10:reffunc:affine}{función afín} cuyo \omterm{def:g10:functions:graph}{gráfico} pasa por $A(1, 3)$ y
$B(4, 9)$. El \omterm{def:g10:reffunc:affine}{pendiente} es
\[
m = \frac{9 - 3}{4 - 1} = \frac{6}{3} = 2 .
\]
Luego $f(x) = 2x + p$ y $f(1) = 3$ dan $2 + p = 3$, por lo que $p = 1$:
$f(x) = 2x + 1$. Consulte con $B$: $f(4) = 9$.
\end{example}

\section{La función cuadrada}

\begin{proposition}[Función cuadrada]\label{prop:g10:reffunc:square}
El \omterm{def:g10:functions:function}{función} $f(x) = x^2$, definido en $\R$:
\begin{enumerate}
  \item es estrictamente \omterm{def:g10:functions:variations}{decreciente} en $\intoc{-\infty}{0}$ y estrictamente
        \omterm{def:g10:functions:variations}{creciente} en $\intco{0}{+\infty}$, con \omterm{def:g10:functions:extrema}{mínimo} $0$ en $x = 0$;
  \item satisface $f(-x) = f(x)$ para todo $x$: es \omterm{def:g10:functions:graph}{gráfico}, un
        \emph{parábola}\index{parábola}, es simétrico con respecto a la vertical
        eje.
\end{enumerate}
\end{proposition}

\begin{proof}
1. Deje $0 \leq u < v$. entonces
\[
f(v) - f(u) = v^2 - u^2 = (v - u)(v + u) > 0,
\]
desde $v - u > 0$ y $v + u > 0$: $f$ es estrictamente \omterm{def:g10:functions:variations}{creciente} en
$\intco{0}{+\infty}$. Si $u < v \leq 0$, entonces $v - u > 0$ pero
$v + u < 0$, entonces $f(v) - f(u) < 0$: estrictamente \omterm{def:g10:functions:variations}{decreciente}. Finalmente
$x^2 \geq 0 = f(0)$ para todos los $x$.

2.$(-x)^2 = x^2$; entonces los puntos $(x, x^2)$ y $(-x, x^2)$, reflejan \omterm{def:g10:functions:function}{imágenes}
a lo largo del eje vertical, están ambos en el \omterm{def:g10:functions:graph}{gráfico}.
\end{proof}

\begin{remark}[Cuadrados y orden]
Debido a que el cuadrado \omterm{def:g10:functions:function}{función} disminuye en el lado negativo, tomando cuadrados
\emph{invierte} el orden de los números negativos: $-3 < -2$ pero $9 > 4$.
Nunca eleve ambos lados de una desigualdad sin verificar los signos.
\end{remark}

\section{La función inversa}

\begin{proposition}[Función inversa]\label{prop:g10:reffunc:inverse}
El \omterm{def:g10:functions:function}{función} $f(x) = \dfrac1x$, definido para $x \neq 0$:
\begin{enumerate}
  \item es estrictamente \omterm{def:g10:functions:variations}{decreciente} en $\intoo{-\infty}{0}$ y estrictamente
        \omterm{def:g10:functions:variations}{decreciente} en $\intoo{0}{+\infty}$;
  \item satisface $f(-x) = -f(x)$: es \omterm{def:g10:functions:graph}{gráfico}, un
        \emph{hipérbola}\index{hipérbola}, es simétrico con respecto al origen.
\end{enumerate}
\end{proposition}

\begin{proof}
1. Deje $0 < u < v$. entonces
\[
f(v) - f(u) = \frac1v - \frac1u = \frac{u - v}{uv} < 0,
\]
desde $u - v < 0$ y $uv > 0$: estrictamente \omterm{def:g10:functions:variations}{decreciente} en
$\intoo{0}{+\infty}$. En $\intoo{-\infty}{0}$ se tiene el mismo cociente
$u - v < 0$ y $uv > 0$ nuevamente (producto de dos negativos), entonces $f$
allí también disminuye.

2. $\frac{1}{-x} = -\frac1x$.
\end{proof}

\begin{remark}
Cuidado: $\frac1x$ es \emph{no} \omterm{def:g10:functions:variations}{decreciente} en su totalidad \omterm{def:g10:functions:function}{dominio}. De
$u = -1$ a $v = 1$ el valor salta de $-1$ hasta $1$. las dos ramas
debe estudiarse por separado.
\end{remark}

\section{Raíz cuadrada y cubo}

\begin{proposition}[Función de raíz cuadrada]\label{prop:g10:reffunc:sqrt}
El \omterm{def:g10:functions:function}{función} $f(x) = \sqrt{x}$, definido en $\intco{0}{+\infty}$, es
estrictamente \omterm{def:g10:functions:variations}{creciente}.
\end{proposition}

\begin{proof}
Deje $0 \leq u < v$. Multiplica y divide por \emph{conjugado}:
\[
\sqrt v - \sqrt u
= \frac{(\sqrt v - \sqrt u)(\sqrt v + \sqrt u)}{\sqrt v + \sqrt u}
= \frac{v - u}{\sqrt v + \sqrt u} > 0,
\]
desde $v - u > 0$ y $\sqrt v + \sqrt u > 0$ (nota $v > 0$, por lo que
$\sqrt v > 0$).
\end{proof}

\begin{proposition}[Función cubo]\label{prop:g10:reffunc:cube}
El \omterm{def:g10:functions:function}{función} $f(x) = x^3$, definido en $\R$, es estrictamente \omterm{def:g10:functions:variations}{creciente}, y
su \omterm{def:g10:functions:graph}{gráfico} es simétrico con respecto al origen.
\end{proposition}
\admitted

\begin{proposition}[Comparando $x$, $x^2$ y $\sqrt x$]\label{prop:g10:reffunc:compare}
Para $0 < x < 1$:\ \ $x^2 < x < \sqrt x$.\quad
Para $x > 1$:\ \ $\sqrt x < x < x^2$.\quad
En $x = 0$ y $x = 1$ los tres valores son iguales.
\end{proposition}

\begin{proof}
Supongamos $0 < x < 1$. Multiplicar $x < 1$ por $x > 0$ da $x^2 < x$.
A continuación, $x < \sqrt x$: como ambos lados son positivos y elevar al cuadrado es
\omterm{def:g10:functions:variations}{creciente} en números positivos, esta desigualdad es equivalente a
$x^2 < x$, que acabamos de demostrar. Para $x > 1$, multiplicar $x > 1$ por $x$
da $x^2 > x$, y $\sqrt x < x$ sigue el mismo argumento de elevación al cuadrado.
\end{proof}

\begin{method}[Comparando imágenes]\label{met:g10:reffunc:compare}
Para comparar $f(a)$ y $f(b)$ para obtener una referencia \omterm{def:g10:functions:function}{función} $f$:
\begin{enumerate}
  \item coloque $a$ y $b$ en un \omterm{def:g10:numbers:interval}{intervalo} donde se encuentran las variaciones de $f$.
        conocido;
  \item si $f$ aumenta allí, \omterm{def:g10:functions:function}{imágenes} están en el mismo orden que el
        entradas; si $f$ disminuye, se invierte el orden;
  \item si $a$ y $b$ no están en el mismo \omterm{def:g10:numbers:interval}{intervalo} de monotonicidad, trato
        con signos primero (por ejemplo,\ para el cuadrado: un negativo y un positivo
        entrada).
\end{enumerate}
\end{method}

\begin{example}
Compare $\dfrac{1}{2.3}$ y $\dfrac{1}{2.4}$: ambas entradas están en
$\intoo{0}{+\infty}$, donde la inversa \omterm{def:g10:functions:function}{función} disminuye, y
$2.3 < 2.4$, entonces $\dfrac{1}{2.3} > \dfrac{1}{2.4}$. Comparar $(-1.2)^2$
y $(-1.5)^2$: en $\intoc{-\infty}{0}$ el cuadrado disminuye, y
$-1.5 < -1.2$, entonces $(-1.5)^2 > (-1.2)^2$ --- de hecho $2.25 > 1.44$.
\end{example}

\section{Problema: las leyes de la naturaleza hablan en funciones de referencia.}

\begin{problem}[{Problema de fin de semana --- las distancias de frenado aumentan a medida que
$v^2$, las palancas obedecen a $1/d$, los planetas siguen a $a^{3/2}$: lectura
fisica con la referencia \omterm{def:g10:functions:function}{funciones}}]
\label{pb:g10:reffunc:1}
Abra un libro de física y los mismos \omterm{def:g10:functions:graph}{graficos} aparecen en cada página:
la parábola, la hipérbola, la raíz de la curva. La naturaleza escribe su
leyes con la referencia de este capítulo \omterm{def:g10:functions:function}{funciones} --- y sabiendo
sus formas (\cref{prop:g10:reffunc:square},
\cref{prop:g10:reffunc:inverse}, \cref{prop:g10:reffunc:sqrt},
\cref{prop:g10:reffunc:cube}) es suficiente para frenar un coche, equilibrar un
palanca y cronometrar los planetas. El gran final es el tercero de Kepler.
ley, lea directamente los datos del sistema solar.
