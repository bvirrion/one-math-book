\chapter{Vectores}\label{ch:g10:vectors}

Un \omterm{def:g10:vectors:vector}{vector} codifica un desplazamiento: una dirección y una longitud, independientemente de
el punto de partida. \omterm{def:g10:vectors:vector}{Vectores} dar pruebas limpias de hechos geométricos y, una vez
\omterm{def:g10:coordgeom:system}{coordenadas} entra en escena, redúcelos a pequeños cálculos. ellos
será una herramienta clave en \cref{ch:g10:lines} y, posteriormente, para el escalar
producto en \cref{ch:g11:scal}.

\section{Traducciones y vectores}

\begin{definition}[Vector]\label{def:g10:vectors:vector}
Dados dos puntos $A$ y $B$, el \emph{vector}\index{vector} $\vect{AB}$
representa el desplazamiento de $A$ a $B$: tiene un
\emph{dirección} (el sentido de la línea $(AB)$, junto con el camino
de recorrerlo, desde $A$ hacia $B$) y un
\emph{longitud} (la distancia $AB$, también escrita $\norm{\vect{AB}}$).
Dos vectores son \emph{igual} cuando codifican el mismo desplazamiento:
$\vect{AB} = \vect{CD}$ significa que aplicando los dos desplazamientos a cualquier
punto da el mismo resultado.
\end{definition}

\begin{proposition}[Vectores iguales y paralelogramos]\label{prop:g10:vectors:parallelogram}
$\vect{AB} = \vect{CD}$ exactamente cuando $ABDC$ (¡en este orden!) es un
paralelogramo, posiblemente uno aplanado --- equivalentemente, cuando $[AD]$ y
$[BC]$ tiene el mismo \omterm{prop:g10:coordgeom:midpoint}{punto medio}.
\end{proposition}
\admitted

\begin{notation}
Un \omterm{def:g10:vectors:vector}{vector} suele denominarse con una sola letra, $\vec u$ o $\vec v$, cuando el
los puntos finales no importan. El \emph{zero \omterm{def:g10:vectors:vector}{vector}} $\vec 0 = \vect{AA}$ es
el desplazamiento que no mueve nada. El \emph{opuesto} de
$\vect{AB}$ es $\vect{BA}$: misma longitud, en sentido opuesto. escribimos
$\vect{BA} = -\vect{AB}$.
\end{notation}

\section{Añadiendo vectores}

\begin{definition}[Suma de dos vectores]\label{def:g10:vectors:sum}
El \emph{suma}\index{vector!suma} $\vec u + \vec v$ es el desplazamiento
obtenido realizando $\vec u$ y luego $\vec v$.
\end{definition}

\begin{theorem}[Relación de Chasles]\label{thm:g10:vectors:chasles}
Para tres puntos cualesquiera $A$, $B$, $C$:
\[
\vect{AB} + \vect{BC} = \vect{AC}.
\]
\end{theorem}

\begin{proof}
Pasar de $A$ a $B$, luego de $B$ a $C$, es un desplazamiento que toma
$A$ a $C$: es el desplazamiento codificado por $\vect{AC}$.
\end{proof}

\begin{remark}[Regla del paralelogramo]
Para sumar dos \omterm{def:g10:vectors:vector}{vectores} extraídos del mismo punto, $\vect{AB} + \vect{AC}$,
Completa el paralelogramo $ABDC$: el \omterm{def:g10:vectors:sum}{suma} es la diagonal $\vect{AD}$.
De hecho $\vect{AC} = \vect{BD}$, entonces
$\vect{AB} + \vect{AC} = \vect{AB} + \vect{BD} = \vect{AD}$ por Chasles'
relación.
\end{remark}

\begin{example}[Simplificando con Chasles]\label{ex:g10:vectors:chasles}
Simplificar $\vect{MN} + \vect{NP} + \vect{PQ}$: encadenando los desplazamientos
da $\vect{MQ}$. Simplifique $\vect{AB} - \vect{AC}$: reescriba el
diferencia como un \omterm{def:g10:vectors:sum}{suma} con el opuesto \omterm{def:g10:vectors:vector}{vector},
\[
\vect{AB} - \vect{AC} = \vect{AB} + \vect{CA}
= \vect{CA} + \vect{AB} = \vect{CB}.
\]
\end{example}

\section{Multiplicar un vector por un número}

\begin{definition}[escalera múltiple]\label{def:g10:vectors:scalar}
Sea $\vec u$ un \omterm{def:g10:vectors:vector}{vector} distinto de cero y $k$ un \omterm{def:g10:numbers:sets}{numero real}. El \omterm{def:g10:vectors:vector}{vector}
$k\vec u$\index{vector!múltiplo escalar} tiene la misma dirección que
$\vec u$, longitud $\abs{k} \times \norm{\vec u}$ y apunta en la misma dirección
como $\vec u$ si $k > 0$, al revés si $k < 0$. Para $k = 0$,
$0\vec u = \vec 0$.
\end{definition}

\begin{definition}[Colinealidad]\label{def:g10:vectors:collinear}
Dos \omterm{def:g10:vectors:vector}{vectores} $\vec u$ y $\vec v$ son
\emph{colineal}\index{Vectores colineales} cuando tengan el mismo
dirección, es decir,\ cuando $\vec v = k \vec u$ para algún $k$ real (o uno de
ellos es $\vec 0$).
\end{definition}

\begin{remark}
La colinealidad es el lenguaje \omterm{def:g10:vectors:vector}{vector} para paralelismo y alineación:
\begin{itemize}
  \item las líneas $(AB)$ y $(CD)$ son paralelas exactamente cuando $\vect{AB}$
        y $\vect{CD}$ son \omterm{def:g10:vectors:collinear}{colineal};
  \item los puntos $A$, $B$, $C$ están alineados exactamente cuando $\vect{AB}$ y
        $\vect{AC}$ son \omterm{def:g10:vectors:collinear}{colineal}.
\end{itemize}
\end{remark}

\section{Coordenadas de vectores}

\begin{definition}[Coordenadas de un vector]\label{def:g10:vectors:coords}
En un \omterm{def:g10:coordgeom:system}{sistema de coordenadas}, el \emph{\omterm{def:g10:coordgeom:system}{coordenadas}} del \omterm{def:g10:vectors:vector}{vector} $\vect{AB}$
son los números que describen el desplazamiento:
\[
\vect{AB}\,(x_B - x_A,\ y_B - y_A).
\]
Un \omterm{def:g10:vectors:vector}{vector} $\vec u\,(a, b)$ mueve cada punto $a$ unidades horizontalmente y
$b$ unidades verticalmente.
\end{definition}

\begin{proposition}[Cálculo con coordenadas]\label{prop:g10:vectors:coordrules}
Dejemos $\vec u\,(a, b)$ y $\vec v\,(c, d)$ y $k \in \R$.
\begin{enumerate}
  \item $\vec u = \vec v$ exactamente cuando $a = c$ y $b = d$;
  \item $\vec u + \vec v$ tiene \omterm{def:g10:coordgeom:system}{coordenadas} $(a + c,\ b + d)$;
  \item $k\vec u$ tiene \omterm{def:g10:coordgeom:system}{coordenadas} $(ka,\ kb)$;
  \item $\norm{\vec u} = \sqrt{a^2 + b^2}$ (\omterm{def:g10:coordgeom:system}{sistema ortonormal}).
\end{enumerate}
\end{proposition}

\begin{proof}
1--3 reformula, coordenada por coordenada, lo que hacen los desplazamientos: por
instancia que realiza $\vec u$, luego $\vec v$ mueve un punto horizontalmente
$a$ luego $c$, por lo tanto, $a + c$ en total. El punto 4 es la fórmula de la distancia.
de \cref{thm:g10:coordgeom:distance} aplicado a un segmento que representa
$\vec u$.
\end{proof}

\begin{theorem}[Criterio de colinealidad]\label{thm:g10:vectors:det}
Dos \omterm{def:g10:vectors:vector}{vectores} $\vec u\,(a, b)$ y $\vec v\,(c, d)$ son \omterm{def:g10:vectors:collinear}{colineal} si y
solo si
\[
ad - bc = 0 .
\]
\end{theorem}

\begin{proof}
Si $\vec v = k\vec u$, entonces $c = ka$ y $d = kb$, entonces
$ad - bc = a(kb) - b(ka) = 0$. Lo mismo vale si $\vec u = k \vec v$ o si
\omterm{def:g10:vectors:vector}{vector} es cero.

Por el contrario, supongamos $ad - bc = 0$ con $\vec u \neq \vec 0$, digamos
$a \neq 0$ (el caso $b \neq 0$ es similar). Establecer $k = \frac{c}{a}$; entonces
$c = ka$ y $ad = bc = b(ka)$ dan, después de dividir por $a \neq 0$,
$d = kb$. Entonces $\vec v = k\vec u$.
\end{proof}

\begin{example}\label{ex:g10:vectors:det}
¿Son $\vec u\,(3, -2)$ y $\vec v\,(-6, 4)$ \omterm{def:g10:vectors:collinear}{colineal}? calcular
$ad - bc = 3 \times 4 - (-2)\times(-6) = 12 - 12 = 0$: si, y efectivamente
$\vec v = -2\vec u$. Para $\vec u\,(3, -2)$ y $\vec w\,(1, 5)$:
$3 \times 5 - (-2) \times 1 = 17 \neq 0$: no \omterm{def:g10:vectors:collinear}{colineal}.
\end{example}

\begin{method}[Alineación y paralelismo]\label{met:g10:vectors:alignment}
Para demostrar que tres puntos $A$, $B$, $C$ están alineados:
\begin{enumerate}
  \item calcula el \omterm{def:g10:coordgeom:system}{coordenadas} de $\vect{AB}$ y $\vect{AC}$;
  \item comprobar el criterio $ad - bc = 0$ de \cref{thm:g10:vectors:det};
  \item concluye: los \omterm{def:g10:vectors:vector}{vectores} son \omterm{def:g10:vectors:collinear}{colineal} y comparten el punto $A$, por lo que
        Los tres puntos están alineados.
\end{enumerate}
El mismo cálculo con $\vect{AB}$ y $\vect{CD}$ demuestra que el
Las líneas $(AB)$ y $(CD)$ son paralelas.
\end{method}

\begin{example}\label{ex:g10:vectors:alignment}
Sean $A(1, 2)$, $B(3, 5)$ y $C(7, 11)$. Luego $\vect{AB}\,(2, 3)$ y
$\vect{AC}\,(6, 9)$ y $2 \times 9 - 3 \times 6 = 18 - 18 = 0$: el
Los puntos están alineados (de hecho, $\vect{AC} = 3\vect{AB}$).
\end{example}

\section{Problema: el álgebra de las flechas}

\begin{problem}[{Problema de fin de semana --- Gimnasia Chasles, la
centroide reprobado en dos líneas, y fuerzas en equilibrio: ¿qué?
los vectores son realmente para}]
\label{pb:g10:vectors:1}
El octavo grado demostró que las medianas de un triángulo se encuentran en dos tercios.
de su longitud --- con un ingenioso paralelogramo escondido en su interior
el triángulo (el problema del centroide del fin de semana del volumen de la escuela secundaria). \omterm{def:g10:vectors:vector}{Vectores} reprenderlo en
dos líneas y reparte la concurrencia gratis. Eso es lo que
Esta nueva álgebra es para: los teoremas se convierten en cálculos con
flechas. Este problema entrena el cálculo (Chasles sobre todo),
ofrece la prueba de dos líneas, amplía Varignon
(\cref{exo:g10:vectors:10}), y termina donde nació \omterm{def:g10:vectors:vector}{vectores}:
fuerzas y velocidades.

\medskip
\noindent\textbf{Parte I --- Chasles gymnastics.}
\begin{enumerate}
  \item Simplifica usando la relación de Chasles
        (\cref{thm:g10:vectors:chasles}):
        $\vect{AB} + \vect{BC} + \vect{CD}$; \quad
        $\vect{MA} - \vect{MB}$; \quad
        $\vect{AB} + \vect{CD} + \vect{BC} + \vect{DA}$.
  \item El truco del origen: muestra que para \emph{cualquier} punto $O$,
        $\vect{AB} = \vect{OB} - \vect{OA}$.
  \item Probar las dos caracterizaciones del \omterm{prop:g10:coordgeom:midpoint}{punto medio} $I$ de
        $[AB]$:
        \[
        \vect{IA} + \vect{IB} = \vec 0
        \qquad\text{and}\qquad
        \vect{OI} = \tfrac12\left(\vect{OA} + \vect{OB}\right)
        \ \text{for any } O .
        \]
  \item Usando $\vect{AB} = \vect{DC}$, vuelva a encontrar el cuarto vértice
        $D$ del paralelogramo $ABCD$ con $A(-1, 2)$,
        $B(3, 4)$, $C(6, 0)$ --- y comparar con el
        respuesta del truco del punto medio de \cref{pb:g10:coordgeom:1}.
  \item Agregando un \omterm{def:g10:vectors:vector}{vector} $\vec u$ fijo a cada punto del
        plano realiza qué transformación --- la que
        descubierto con dos espejos en el problema de fin de semana de dos espejos del volumen de Escuela Secundaria
        y dos medias vueltas en el problema de fin de semana de medias vueltas del volumen de la escuela secundaria? calcular
        el \omterm{def:g10:functions:function}{imagen} de $(1, 2)$ bajo $\vec u = (3, -1)$.
\end{enumerate}

\medskip
\noindent\textbf{Parte II --- The centroid, third proof.}
Definir el punto $G$ de un triángulo $ABC$ mediante el elegante
condición
\[
\vect{GA} + \vect{GB} + \vect{GC} = \vec 0 .
\]
\begin{enumerate}[resume]
  \item Usando el truco del origen, demuestre que esta condición
        determina $G$ de forma única:
        $\vect{OG} = \frac13\left(\vect{OA} + \vect{OB} +
        \vect{OC}\right)$ para cualquier $O$ --- entonces el \omterm{def:g10:coordgeom:system}{coordenadas} de
        $G$ son los \emph{promedios} de los vértices
        (el problema del centroide del fin de semana del volumen de la escuela secundaria lo vio numéricamente).
  \item Pruebe $\vect{AG} = \frac13\left(\vect{AB} +
        \vect{AC}\right)$, y, escribiendo $K$ para el \omterm{prop:g10:coordgeom:midpoint}{punto medio} de
        $[BC]$, $\vect{AK} = \frac12\left(\vect{AB} +
        \vect{AC}\right)$. Concluir:
        $\vect{AG} = \frac23\,\vect{AK}$ --- los dos tercios
        teorema, re-probado.
  \item ¿Por qué pasan las medianas de $B$ y de $C$?
        el \emph{mismo} $G$ sin más cálculos?
        Compare las tres pruebas disponibles de este teorema ---
        paralelogramo oculto de grado 8, \omterm{def:g10:coordgeom:system}{coordenadas}, \omterm{def:g10:vectors:vector}{vectores} ---
        en una frase cada uno.
  \item Comprobación numérica: $A(1, 1)$, $B(5, 2)$, $C(3, 6)$.
        Calcule $G$, luego $K$, luego verifique
        $\vect{AG} = \frac23 \vect{AK}$ con la colinealidad
        criterio (\cref{thm:g10:vectors:det}).
  \item Física lee $G$ como el punto de equilibrio de tres iguales
        masas en los vértices. Duplicar la masa en $A$ (masas
        $2, 1, 1$): el punto de equilibrio pasa a ser
        $\vect{OG'} = \frac{2\vect{OA} + \vect{OB} +
        \vect{OC}}{4}$. Calcule $G'$ para el triángulo de
        pregunta 9 y compare su posición con $G$.
\end{enumerate}

\medskip
\noindent\textbf{Parte III --- Collinearity at work.}
\begin{enumerate}[resume]
  \item ¿Son $\vec u(3, -2)$ y $\vec v(-6, 4)$ \omterm{def:g10:vectors:collinear}{colineal}? y
        $\vec w(2, 5)$, $\vec z(4, 9)$? decidir con el
        criterio determinante.
  \item Son los puntos $A(1, 2)$, $B(3, 5)$, $C(7, 11)$
        ¿alineado?
  \item Más allá de Varignon (\cref{exo:g10:vectors:10}): en cualquier
        cuadrilátero $ABCD$, el \emph{bimedianas} se une al
        \omterm{prop:g10:coordgeom:midpoint}{puntos medios} de lados opuestos ($[IK]$ y $[JL]$). Usando
        posición \omterm{def:g10:vectors:vector}{vectores} ($\vect{OI} = \frac12(\vect{OA} +
        \vect{OB})$, etc.), muestran que ambas bimedianas tienen la
        mismo \omterm{prop:g10:coordgeom:midpoint}{punto medio}, de posición \omterm{def:g10:vectors:vector}{vector}
        $\frac14\left(\vect{OA} + \vect{OB} + \vect{OC} +
        \vect{OD}\right)$: el centroide del cuadrilátero.
  \item Tales en una línea: $M$ y $N$ son los puntos definidos

  \item Generalizar la pregunta 14 a una proporción arbitraria $k$, y

\begin{enumerate}[resume]
  \item Un barco es empujado por una corriente de $3$ km/h hacia el este
        mientras remaba a $4$ km/h hacia el norte. dar el resultante

        curso. (Aparece un triple muy antiguo.)

        ¿Qué tan abajo aterriza el nadador?

        \omterm{def:g10:vectors:vector}{vectores} de este capítulo aún no puede medir --- el
