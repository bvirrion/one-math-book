\chapter{Funciones}\label{ch:g10:functions}

Una \omterm{def:g10:functions:function}{función} es una máquina que toma un número y produce un número,
Siempre la misma salida para la misma entrada. \omterm{def:g10:functions:function}{Funciones} son los centrales
Objetos del análisis que construirás en los próximos años.
(\cref{ch:g11:quad} en adelante); este capítulo establece el vocabulario ---
\omterm{def:g10:functions:function}{dominio}, \omterm{def:g10:functions:function}{imagen}, \omterm{def:g10:functions:graph}{gráfico}, variaciones --- y entrena la habilidad esencial de
leyendo información de un \omterm{def:g10:functions:graph}{gráfico}.

\section{Vocabulario: imágenes y preimágenes}

\begin{definition}[Función]\label{def:g10:functions:function}
Sea $D$ un conjunto de \omterm{def:g10:numbers:sets}{números reales}. \emph{función}\index{función} $f$
definido en $D$ asocia a cada número $x \in D$ exactamente un \omterm{def:g10:numbers:sets}{real
numero}, escrito $f(x)$ y denominado \emph{imagen}\index{imagen} de $x$.
El conjunto $D$ es el \emph{dominio}\index{dominio} de $f$. escribimos
\[
f : x \longmapsto f(x).
\]
Si $f(x) = y$, entonces $x$ es un \emph{preimagen}\index{preimagen} de $y$.
\end{definition}

\begin{remark}
Cada $x$ en el \omterm{def:g10:functions:function}{dominio} tiene exactamente un \omterm{def:g10:functions:function}{imagen}, pero un número $y$ puede tener
varios \omterm{def:g10:functions:function}{preimágenes}, o ninguno. Para $f(x) = x^2$: el \omterm{def:g10:functions:function}{imagen} de $3$ es $9$, y
$9$ tiene dos \omterm{def:g10:functions:function}{preimágenes}, $3$ y $-3$; el número $-4$ no tiene \omterm{def:g10:functions:function}{preimagen}.
\end{remark}

\begin{example}[Encontrar un dominio]\label{ex:g10:functions:domain}
Cuando un \omterm{def:g10:functions:function}{función} viene dado por una fórmula, su \omterm{def:g10:functions:function}{dominio} es el conjunto de $x$ para
que la fórmula tiene sentido.
\begin{itemize}
  \item $f(x) = 3x^2 - 5x + 1$ tiene sentido para cada $x$: el \omterm{def:g10:functions:function}{dominio} es
        $\R$.
  \item $g(x) = \dfrac{1}{x - 2}$ requiere $x - 2 \neq 0$: el \omterm{def:g10:functions:function}{dominio} es
        todos los reales excepto $2$.
  \item $h(x) = \sqrt{x - 3}$ requiere $x - 3 \geq 0$: el \omterm{def:g10:functions:function}{dominio} es
        $\intco{3}{+\infty}$.
\end{itemize}
\end{example}

\begin{example}[Computación de imágenes y preimágenes]\label{ex:g10:functions:images}
Sea $f(x) = x^2 - 4x + 3$, definido en $\R$.

\emph{\omterm{def:g10:functions:function}{Imagen} of $5$:} sustituto $x = 5$:
$f(5) = 25 - 20 + 3 = 8$.

\emph{\omterm{def:g10:functions:function}{Preimágenes} of $3$:} resolver $f(x) = 3$:
\[
x^2 - 4x + 3 = 3
\quad\Longleftrightarrow\quad
x^2 - 4x = 0
\quad\Longleftrightarrow\quad
x(x - 4) = 0,
\]
entonces \omterm{def:g10:functions:function}{preimágenes} de $3$ son $0$ y $4$.
\end{example}

\section{La gráfica de una función.}

\begin{definition}[Gráfico]\label{def:g10:functions:graph}
En un sistema de coordenadas, el \emph{gráfico}\index{gráfico} (o \emph{curva}) de
$f$ es el conjunto de todos los puntos $(x, f(x))$ para $x$ en \omterm{def:g10:functions:function}{dominio}. en otros
En otras palabras, un punto $(x, y)$ se encuentra en el gráfico exactamente cuando $y = f(x)$.
\end{definition}

\begin{method}[Leyendo un gráfico]\label{met:g10:functions:reading}
En el \omterm{def:g10:functions:graph}{gráfico} de $f$:
\begin{enumerate}
  \item \emph{\omterm{def:g10:functions:function}{imagen} of $a$}: ir verticalmente desde $x = a$ en horizontal
        eje a la curva, luego horizontalmente al eje vertical; el
        el valor leído es $f(a)$;
  \item \emph{\omterm{def:g10:functions:function}{preimágenes} of $b$}: trazar la línea horizontal $y = b$; el
        \omterm{def:g10:functions:function}{preimágenes} son las coordenadas $x$ de todos sus puntos \omterm{def:g10:numbers:interunion}{intersección}
        con la curva;
  \item \emph{soluciones de $f(x) = k$}: igual que el \omterm{def:g10:functions:function}{preimágenes} de $k$;
  \item \emph{solutions of $f(x) \leq k$}: el $x$ para el cual la curva es
        en o debajo de la línea $y = k$.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g10:functions:membership}
¿Está el punto $A(2, 5)$ en el \omterm{def:g10:functions:graph}{gráfico} de $f(x) = x^2 + 1$? calcular
$f(2) = 4 + 1 = 5$: sí, ya que $f(2)$ es igual a la coordenada $y$ de $A$.
El punto $B(3, 8)$ no está en el \omterm{def:g10:functions:graph}{gráfico}, porque $f(3) = 10 \neq 8$.
\end{example}

\section{Variaciones de una función}

\begin{definition}[Creciente, decreciente]\label{def:g10:functions:variations}
Dejemos que $f$ se defina en un \omterm{def:g10:numbers:interval}{intervalo} $I$.
\begin{itemize}
  \item $f$ es \emph{creciente}\index{función creciente} en $I$ cuando
        para todos $u, v \in I$, si $u < v$ entonces $f(u) \leq f(v)$: el
        Las salidas crecen con las entradas, y el \omterm{def:g10:functions:graph}{gráfico} sube de izquierda a derecha.
        correcto.
  \item $f$ es \emph{decreciente}\index{función decreciente} en $I$ cuando
        $u < v$ implica $f(u) \geq f(v)$: el \omterm{def:g10:functions:graph}{gráfico} cae de izquierda a
        correcto.
\end{itemize}
Con desigualdades estrictas $f(u) < f(v)$ (resp.\ $>$) decimos
\emph{estrictamente} aumentando (resp.\ disminuyendo).
\end{definition}

\begin{definition}[Tabla de variaciones]\label{def:g10:functions:table}
A \emph{tabla de variaciones}\index{tabla de variaciones} resume en qué
\omterm{def:g10:numbers:interval}{intervalos} $f$ aumenta y disminuye, con las flechas $\nearrow$ y
$\searrow$, y registra los valores de $f$ en los puntos de inflexión.
\end{definition}

\begin{example}\label{ex:g10:functions:table}
Considere el \omterm{def:g10:functions:function}{función} que se muestra a continuación en $\intcc{-2}{4}$.

\begin{example}[Probando una variación]\label{ex:g10:functions:proof}
Demuestre que $f(x) = 3x + 1$ es estrictamente \omterm{def:g10:functions:variations}{creciente} en $\R$. Toma cualquier
$u < v$ y compare el \omterm{def:g10:functions:function}{imágenes}:
\[
f(v) - f(u) = (3v + 1) - (3u + 1) = 3(v - u) > 0,
\]
porque $v - u > 0$. Entonces $f(u) < f(v)$: el \omterm{def:g10:functions:function}{función} es estrictamente
\omterm{def:g10:functions:variations}{creciente}. El mismo cálculo con pendiente negativa, por ejemplo\
$g(x) = -2x + 5$, da $g(v) - g(u) = -2(v-u) < 0$: estrictamente \omterm{def:g10:functions:variations}{decreciente}.
\end{example}

\section{extremo}

\begin{definition}[Máximo, mínimo]\label{def:g10:functions:extrema}
Dejemos que $f$ se defina en $D$ y $a \in D$. El valor $f(a)$ es el
\emph{máximo}\index{máximo} de $f$ en $D$ cuando $f(x) \leq f(a)$ para todos
$x \in D$; es el \emph{mínimo}\index{mínimo} cuando
$f(x) \geq f(a)$ para todos los $x \in D$. En un \omterm{def:g10:functions:graph}{gráfico}, son los más altos y
puntos más bajos de la curva.
\end{definition}

\begin{example}\label{ex:g10:functions:extremum}
Demuestre que $f(x) = (x - 1)^2 + 2$ tiene \omterm{def:g10:functions:extrema}{mínimo} $2$ en $\R$, alcanzado en
$x = 1$. Por cada $x$, el cuadrado $(x-1)^2$ es $\geq 0$, entonces
$f(x) = (x-1)^2 + 2 \geq 2$; y $f(1) = 0 + 2 = 2$, por lo que el valor $2$ es
realmente alcanzado. Ambos hechos juntos prueban que $2$ es el \omterm{def:g10:functions:extrema}{mínimo}.
\end{example}

\end{omfigure}

\section{Problema: las máquinas alimentan su propia producción}

\begin{problem}[{Problema de fin de semana --- puntos fijos, iteración y
tres máquinas famosas: la mejoradora de Heron, la escuadradora y la
granizo sin resolver}]
\label{pb:g10:functions:1}
Una \omterm{def:g10:functions:function}{función} es una máquina: entra un número, sale un número.
Los experimentos más interesantes alimentan la máquina \emph{su propio
salida}, una y otra vez, y luego dominan dos preguntas:
¿El proceso se establece en algún lugar (un \emph{\omterm{pb:g10:functions:1}{punto fijo}}), y
¿Cómo la lleva la variación de la máquina hasta allí? uno de esto
La máquina del problema ha estado puliendo raíces cuadradas durante dos
mil años; otro guarda el problema sin resolver más famoso
las matemáticas elementales tienen para ofrecer.

\medskip
\noindent\textbf{Parte I --- Heron's machine.}
Considere el \omterm{def:g10:functions:function}{función}
\[
f(x) = \frac12\left(x + \frac2x\right).
\]
\begin{enumerate}
  \item Proporcione el \omterm{def:g10:functions:function}{dominio} de $f$ y calcule $f(1)$, $f(2)$ y
        $f\!\left(\frac32\right)$ como fracciones exactas.
  \item Encuentra todos los \omterm{def:g10:functions:function}{preimágenes} de $\frac32$: resolver
        $f(x) = \frac32$ (borre el denominador y use el
        métodos de \cref{pb:g10:algebra:1}).
  \item Un \emph{punto fijo}\index{punto fijo} de $f$ es un
        número no modificado por la máquina: $f(x) = x$. encontrar ambos
        \omterm{pb:g10:functions:1}{puntos fijos} de $f$. ¿Qué viejo conocido guarda el
        uno positivo?
  \item A partir de $x = 1$, alimente a la máquina su propia salida.
        tres veces y registrar los resultados exactos. ¿Qué secuencia
        del problema de la irracionalidad del fin de semana del volumen de la Escuela Secundaria, ¿has reconstruido --- y lo que fue
        ``La receta de la garza'', en el idioma de \omterm{def:g10:functions:function}{funciones}?
  \item En una imagen, dibuje el \omterm{def:g10:functions:graph}{gráfico} de $f$ (para $x > 0$)
        y la línea diagonal $y = x$. ¿Dónde están los \omterm{pb:g10:functions:1}{fijo
        puntos} en esta imagen? (Comparar
        El problema del fin de semana de dos termómetros del volumen de la escuela secundaria: la lectura de temperatura es la misma
        en ambas escalas fue la misma idea.)
\end{enumerate}

\medskip
\noindent\textbf{Parte II --- Why the machine cannot miss.}
\begin{enumerate}[resume]
  \item Para $0 < u < v$, mostrar por reducción a un común
        denominador que
        \[
        f(v) - f(u) = \frac{(v - u)(uv - 2)}{2uv} ,
        \]
        y deducir las variaciones de $f$ en
        $\intoo{0}{+\infty}$: \omterm{def:g10:functions:variations}{decreciente} hasta cierto punto, luego
        \omterm{def:g10:functions:variations}{creciente} --- ¿qué punto? (La misma técnica que
        \cref{exo:g10:functions:10}.)
  \item Deduce que en $\intoo{0}{+\infty}$ el \omterm{def:g10:functions:function}{función} $f$ tiene
        un \omterm{def:g10:functions:extrema}{mínimo}, con valor exacto de $\sqrt2$, obtenido en
        $\sqrt2$. ¿Qué dice esto sobre la salida \emph{cada}?
        de la máquina (entradas positivas)?
  \item Demuestre que si $x > \sqrt2$ entonces
        $\sqrt2 < f(x) < x$: una suposición demasiado grande siempre mejora,
        nunca se sobrepasó. (Para $f(x) < x$, compare $\frac2x$ con
        $x$.)
  \item Reúna las preguntas 6 a 8 en \omterm{def:g10:functions:table}{tabla de variaciones} de $f$
        en $\intoo{0}{+\infty}$
        (\cref{def:g10:functions:table}), \omterm{def:g10:functions:extrema}{mínimo} incluidos.
  \item Una segunda máquina: $g(x) = x^2$. Sus \omterm{pb:g10:functions:1}{puntos fijos} son
        $0$ y $1$ (consultar). Iterar $g$ cuatro veces desde
        $x = 0.9$, luego cuatro veces desde $x = 1.1$ (calculadora,
        tres decimales). Uno \omterm{pb:g10:functions:1}{punto fijo} atrae, el otro
        repele: ¿cuál es cuál?
\end{enumerate}

\medskip
\noindent\textbf{Parte III --- The hailstone machine.}
Para los números enteros, defina: $h(n) = \frac n2$ si $n$ es par,
y $h(n) = 3n + 1$ si $n$ es impar. Los números rebotan bajo $h$ como
Granizo en una nube de tormenta.
\begin{enumerate}[resume]
  \item Calcule el vuelo completo de $7$ en $h$, hasta $1$.
        ¿Cuántos pasos da y a qué altura vuela?
  \item Inicie el vuelo de $27$ y calcule diez pasos. (Su
        el vuelo completo dura $111$ pasos y alcanza su punto máximo en $9\,232$ ---
        desde un punto de partida de $27$.) ¿Qué lección sobre simple
        ¿Cuáles son las reglas que enseñan $7$ y $27$?
  \item Explique por qué cualquier vuelo que alcance una potencia de $2$
        se bloquea directamente en $1$ y qué sucede en $1$
        (calcular $h(1)$, $h(4)$, $h(2)$). ¿$h$ tiene alguna
        \omterm{pb:g10:functions:1}{punto fijo} entre los números enteros positivos? ¿Qué hace?
        ¿El final de cada vuelo observado?
  \item El \emph{Conjetura de Collatz} reclama cada salida
        El número finalmente llega a $1$. Las computadoras han verificado
        mucho más allá de $10^{20}$; ningún humano lo ha demostrado. que
        ¿Serían dos tipos de descubrimiento \emph{refutar}? y por que
        ¿La verificación masiva aún no cierra el caso?
        (¿El problema del fin de semana del oráculo de cerillas del volumen de la escuela secundaria hizo sonar esta campana)?
  \item ``Las matemáticas aún no están maduras para este tipo de preguntas'', dijo
        Paul Erd\H{o}s de esta conjetura. En una frase: ¿qué?
        separa la máquina de granizo de la de Heron, donde
        ¿Las preguntas 6 a 8 resolvieron todo?
\end{enumerate}

\begin{enumerate}[resume]
  \item \omterm{def:g10:functions:function}{Dominios} como \omterm{def:g10:numbers:interval}{intervalos} (\cref{pb:g10:numbers:1}

  \item Volver a $f$ de Heron: encuentre el \omterm{def:g10:functions:function}{preimágenes} exacto de $3$

  \item Se lanza una piedra hacia arriba: su altura es
        $H(t) = 20t - 5t^2$ metros después de $t$ segundos. en cual
        hora \omterm{def:g10:numbers:interval}{intervalo} ¿la fórmula tiene sentido físico?
        Completa el cuadrado para encontrar la altura máxima y su
        hora (\cref{def:g10:functions:extrema}) y déle el

  \item ¿Durante cuánto tiempo la piedra tiene al menos $15$ m de altura? (Resolver
        $H(t) \geq 15$ con un \omterm{def:g10:algebra:signtable}{mesa de signos},
        \cref{met:g10:algebra:signtable}.)
  \item Final: este problema ejecutó cuatro máquinas --- Heron's
        mejorador, el escuadrador, el granizo, la piedra
        altura. En dos o tres oraciones, indica lo que cada
        \omterm{def:g10:functions:function}{función} tiene (un \omterm{def:g10:functions:function}{dominio}, una regla, un \omterm{def:g10:functions:graph}{gráfico}) y los dos
        grandes preguntas que este problema planteó de cada uno (¿dónde

        preguntas que los próximos capítulos abordan una por una.
\end{enumerate}
