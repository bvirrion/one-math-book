\chapter{Álgebra: ecuaciones y desigualdades}\label{ch:g10:algebra}

El álgebra es el arte de calcular con letras. Este capítulo revisa y
afina los dos movimientos básicos --- \omterm{def:g10:algebra:expand}{en expansión} y \omterm{def:g10:algebra:expand}{factorización} --- y los usa
para resolver \omterm{def:g10:algebra:equation}{ecuaciones} y desigualdades sistemáticamente, con \omterm{def:g10:algebra:signtable}{mesa de signos} como
la herramienta central.

\section{Expandiendo y factorizando}

\begin{definition}[Expandiendo, factorizando]\label{def:g10:algebra:expand}
\emph{En expansión}\index{en expansión} una expresión significa transformar
productos en sumas; \emph{factorización}\index{factorización} significa lo contrario,
Transformar una suma en un producto. La regla básica es la distributividad:
\[
k(a + b) = ka + kb,
\qquad
(a + b)(c + d) = ac + ad + bc + bd .
\]
\end{definition}

\begin{example}\label{ex:g10:algebra:expand}
Expandir $(2x + 3)(x - 5)$ paso a paso:
\begin{align*}
(2x + 3)(x - 5)
&= 2x \times x + 2x \times (-5) + 3 \times x + 3 \times (-5) \\
&= 2x^2 - 10x + 3x - 15 \\
&= 2x^2 - 7x - 15 .
\end{align*}
\end{example}

\begin{theorem}[Identidades notables]\label{thm:g10:algebra:identities}
Para todos los \omterm{def:g10:numbers:sets}{números reales} $a$ y $b$:
\[
(a+b)^2 = a^2 + 2ab + b^2,
\qquad
(a-b)^2 = a^2 - 2ab + b^2,
\qquad
(a+b)(a-b) = a^2 - b^2 .
\]
\end{theorem}

\begin{proof}
Cada uno es una expansión directa. Para el primero:
\[
(a+b)^2 = (a+b)(a+b) = a^2 + ab + ba + b^2 = a^2 + 2ab + b^2 .
\]
El segundo sigue reemplazando $b$ con $-b$: $(a-b)^2 = a^2 + 2a(-b) +
(-b)^2 = a^2 - 2ab + b^2$. Para el tercero:
$(a+b)(a-b) = a^2 - ab + ba - b^2 = a^2 - b^2$.
\end{proof}

\begin{example}\label{ex:g10:algebra:identities}
Usado hacia adelante (\omterm{def:g10:algebra:expand}{en expansión}) y hacia atrás (\omterm{def:g10:algebra:expand}{factorización}):
\begin{align*}
(3x + 2)^2 &= 9x^2 + 12x + 4,
& x^2 - 25 &= (x+5)(x-5), \\
(x - 4)^2 &= x^2 - 8x + 16,
& 4x^2 + 4x + 1 &= (2x + 1)^2 .
\end{align*}
Beneficios de la aritmética mental también: $101 \times 99 = (100+1)(100-1) =
10000 - 1 = 9999$.
\end{example}

\begin{method}[Factorizar una expresión]\label{met:g10:algebra:factor}
Pruebe, en este orden:
\begin{enumerate}
  \item \emph{Factor común:} detecta un factor presente en cada término y extrae
        sáquelo, por ejemplo\ $6x^2 + 4x = 2x(3x + 2)$. El factor común puede ser un
        soporte completo: $(x+1)(2x-3) + (x+1)(x+7) = (x+1)(3x+4)$.
  \item \emph{Identidad notable:} reconoce $a^2 - b^2$, o
        $a^2 \pm 2ab + b^2$, por ejemplo\ $9x^2 - 16 = (3x)^2 - 4^2 =
        (3x+4)(3x-4)$.
  \item Combine ambos y verifique el resultado \omterm{def:g10:algebra:expand}{en expansión} nuevamente.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g10:algebra:factorsteps}
Factor $E = (2x + 1)^2 - (x - 3)^2$. Esta es una diferencia de dos cuadrados.
$a^2 - b^2$ con $a = 2x+1$ y $b = x-3$:
\begin{align*}
E &= \bigl[(2x+1) + (x-3)\bigr] \times \bigl[(2x+1) - (x-3)\bigr] \\
  &= (3x - 2)(2x + 1 - x + 3) \\
  &= (3x - 2)(x + 4).
\end{align*}
Tenga en cuenta el segundo paréntesis: restar $x - 3$ significa restar $x$
\emph{y} agregando $3$.
\end{example}

\section{Ecuaciones}

\begin{definition}[Ecuación, solución]\label{def:g10:algebra:equation}
Un \emph{ecuación}\index{ecuación} es una igualdad que contiene una incógnita.
número $x$; a \emph{solución} es un valor de $x$ que hace la igualdad
cierto. Resolver la ecuación significa encontrar \emph{todo} sus soluciones.
\end{definition}

Dos \omterm{def:g10:algebra:equation}{ecuaciones} son \emph{equivalente} cuando tienen exactamente lo mismo
soluciones. Sumar el mismo número a ambos lados o multiplicar ambos lados
por el mismo número \emph{distinto de cero}, produce un \omterm{def:g10:algebra:equation}{ecuación} equivalente.

\begin{example}[Ecuación de primer grado]\label{ex:g10:algebra:firstdegree}
Resuelva $5x - 7 = 2x + 8$, un movimiento a la vez:
\begin{align*}
5x - 7 &= 2x + 8 \\
5x - 2x &= 8 + 7 && \text{(add $7 - 2x$ to both sides)} \\
3x &= 15 \\
x &= 5 && \text{(divide both sides by $3$).}
\end{align*}
La solución única es $5$. Verificar: $5 \times 5 - 7 = 18$ y
$2 \times 5 + 8 = 18$.
\end{example}

\begin{theorem}[Regla de producto cero]\label{thm:g10:algebra:zeroproduct}
Un producto de \omterm{def:g10:numbers:sets}{números reales} es cero si y sólo si al menos uno de sus
factores es cero:
\[
A \times B = 0
\quad\Longleftrightarrow\quad
A = 0 \ \text{ or } \ B = 0 .
\]
\end{theorem}

\begin{proof}
Si $A = 0$ o $B = 0$, el producto es claramente $0$. Por el contrario, supongamos
$AB = 0$ y $A \neq 0$. Al dividir ambos lados por $A$ se obtiene $B = 0$.
\end{proof}

\begin{method}[Resolviendo por factorización]\label{met:g10:algebra:zeroproduct}
Para resolver un \omterm{def:g10:algebra:equation}{ecuación} cuyo lado derecho es $0$ y cuyo lado izquierdo
lado se puede factorizar:
\begin{enumerate}
  \item lleva todo a la izquierda para que \omterm{def:g10:algebra:equation}{ecuación} diga $E = 0$;
  \item factoriza $E$ en un producto de factores de primer grado;
  \item aplica la regla del producto cero y resuelve cada pequeño \omterm{def:g10:algebra:equation}{ecuación};
  \item recoge todas las soluciones encontradas.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g10:algebra:zeroproduct}
Resuelva $(x - 2)(3x + 6) = 0$: $x - 2 = 0$, dando $x = 2$, o
$3x + 6 = 0$, dando $x = -2$. Soluciones: $-2$ y $2$.

Resuelve $x^2 = 9x$. Divida \emph{no} por $x$ (¡podría ser cero!); traer
todo a un lado y factorizar:
\[
x^2 - 9x = 0
\quad\Longleftrightarrow\quad
x(x - 9) = 0
\quad\Longleftrightarrow\quad
x = 0 \ \text{ or } \ x = 9 .
\]
\end{example}

\begin{example}[Ecuación del cociente]\label{ex:g10:algebra:quotient}
Resuelve $\dfrac{x - 1}{x + 2} = 0$. Un cociente es cero exactamente cuando su
el numerador es cero \emph{y} su denominador no lo es. Aquí $x - 1 = 0$
da $x = 1$ y $1 + 2 = 3 \neq 0$: la única solución es $1$. el valor
$x = -2$ es \emph{prohibido} (división por cero) y siempre debe ser
excluido primero.
\end{example}

\section{Desigualdades y tablas de signos.}

\begin{proposition}[Reglas para desigualdades]\label{prop:g10:algebra:ineqrules}
Deje $a \leq b$.
\begin{enumerate}
  \item Para cualquier $c$: $a + c \leq b + c$ (agregar conserva el orden).
  \item Si $c > 0$: $ac \leq bc$ (multiplicando por un número positivo
        conserva el orden).
  \item Si $c < 0$: $ac \geq bc$ (multiplicando por un número negativo
        \emph{invierte} el pedido).
\end{enumerate}
\end{proposition}

\begin{proof}
1.$(b + c) - (a + c) = b - a \geq 0$.
2. $bc - ac = (b-a)c$ es el producto de dos números no negativos, entonces
$bc \geq ac$.
3. Ahora $(b-a)c \leq 0$ como producto de un no negativo y un negativo
número, entonces $bc \leq ac$.
\end{proof}

\begin{example}\label{ex:g10:algebra:firstineq}
Resuelva $-2x + 3 > 9$:
\begin{align*}
-2x + 3 &> 9 \\
-2x &> 6 && \text{(subtract 3)} \\
x &< -3 && \text{(divide by $-2 < 0$: reverse the inequality).}
\end{align*}
El conjunto de soluciones es $\intoo{-\infty}{-3}$.
\end{example}

\begin{definition}[Tabla de señales]\label{def:g10:algebra:signtable}
Un \emph{mesa de signos}\index{mesa de signos} registra, en una fila por factor, el
signo ($+$ o $-$) de cada factor de una expresión cuando $x$ se ejecuta sobre $\R$,
con un $0$ en cada valor donde el factor desaparece; una fila final da la
signo del producto, utilizando la regla de los signos columna por columna.
\end{definition}

\begin{example}\label{ex:g10:algebra:signtable}
Signo de $P(x) = (x - 2)(3 - x)$. El factor $x - 2$ desaparece en $2$ y es
positivo después; el factor $3 - x$ desaparece en $3$ y es positivo antes:
\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{c|ccccc}
$x$ & $-\infty$ \quad & $2$ & & $3$ & \quad $+\infty$ \\
\hline
$x - 2$ & $-$ & $0$ & $+$ & & $+$ \\
$3 - x$ & $+$ & & $+$ & $0$ & $-$ \\
\hline
$P(x)$ & $-$ & $0$ & $+$ & $0$ & $-$ \\
\end{tabular}
\end{center}
Entonces $P(x) > 0$ exactamente en $\intoo{2}{3}$ y $P(x) \leq 0$ en
$\intoc{-\infty}{2} \cup \intco{3}{+\infty}$.
\end{example}

\begin{method}[Resolver una desigualdad con una tabla de signos]\label{met:g10:algebra:signtable}
Para resolver una desigualdad como $E(x) \geq 0$ o $E(x) < 0$:
\begin{enumerate}
  \item lleva todo al lado izquierdo, de modo que el lado derecho quede
        $0$ (nunca multiplique ambos lados por una expresión cuyo signo sea
        desconocido);
  \item factoriza el lado izquierdo o ponlo sobre un denominador común si
        es un cociente;
  \item construye el \omterm{def:g10:algebra:signtable}{mesa de signos} con una fila por factor (para un cociente, marca
        valores prohibidos con doble barra en la última fila);
  \item lee el \omterm{def:g10:numbers:interval}{intervalos} donde se encuentra el signo solicitado, verificando
        si cada límite está incluido.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g10:algebra:quotientineq}
Resuelve $\dfrac{x + 1}{x - 3} \leq 0$. El numerador desaparece en $-1$, el
denominador en $3$ (valor prohibido). El cociente es negativo cuando
dos tienen signos opuestos, lo que ocurre para $x$ entre $-1$ y $3$.
Incluyendo el límite donde desaparece el \emph{numerador} pero excluyendo el
valor prohibido: el conjunto de soluciones es $\intco{-1}{3}$.
\end{example}

\section{Problema: De Babilonia a la proporción áurea}

\begin{problem}[{Problema de fin de semana --- problemas de suma y producto,
completando el cuadrado como lo dibujó al-Khwarizmi, y el
ecuación $x^2 = x + 1$}]
\label{pb:g10:algebra:1}
Hace cuatro mil años, los escribas babilónicos resolvieron esto en arcilla:
\emph{encontrar dos números sabiendo su suma y su producto}. Su
truco --- busque los números simétricamente alrededor del
media suma --- todavía descifra cada \omterm{def:g10:algebra:equation}{ecuación} cuadrática hoy, bajo
su nombre moderno, \emph{completando el cuadrado}. este problema
aprende el truco, observa a al-Khwarizmi literalmente completar un
cuadrado, y pone el método a trabajar en los más célebres
proporción de todos ellos.

\medskip
\noindent\textbf{Parte I --- The Babylonian problem.}
\begin{enumerate}
  \item Calentamiento (\cref{thm:g10:algebra:identities}): factor
        $x^2 - 10x + 25$, luego $9x^2 - 49$, luego --- combinando un
        identidad con diferencia de cuadrados ---
        $x^2 + 6x + 9 - 4$.
  \item Resuelve $(x + 1)(x + 5) = 0$ y
        $(3x - 7)(3x + 7) = 0$
        (\cref{thm:g10:algebra:zeroproduct}).
  \item El problema del escriba: dos números tienen suma $10$ y
        producto $21$. Siguiendo a Babylon, escríbelos como $5 - t$
        y $5 + t$ (simétrico alrededor de la mitad de la suma) y encuentre $t$,
        luego los dos números.
  \item Mismo método: suma $14$, producto $33$.
  \item La receta general: para la suma $s$ y el producto $p$, mostrar
        que los dos numeros son
        \[
        \frac s2 - t
        \quad\text{and}\quad
        \frac s2 + t,
        \qquad\text{where}\quad
        t^2 = \left(\frac s2\right)^2 - p ,
        \]
        e indique la condición en $s$ y $p$ para que el problema
        tener una solución.
\end{enumerate}

\medskip
\noindent\textbf{Parte II --- Completing the square.}
\begin{enumerate}[resume]
  \item Expanda $(x - a)(x - b)$ y concluya: las soluciones de
        $x^2 - sx + p = 0$ son exactamente los pares de números con
        suma $s$ y producto $p$. (Los escribas estaban resolviendo
        cuadrático \omterm{def:g10:algebra:equation}{ecuaciones} sin saberlo.)
  \item Verifique que $x^2 - 10x + 21 = (x - 5)^2 - 4$ y resuelva
        $x^2 - 10x + 21 = 0$ por \omterm{def:g10:algebra:expand}{factorización} el lado derecho como
        diferencia de cuadrados. Compárese con la pregunta 3.
  \item Resuelve $x^2 + 6x - 7 = 0$ completando el cuadrado.
  \item Complete el cuadrado en $x^2 + 4x + 5$ y explique por qué.
        el \omterm{def:g10:algebra:equation}{ecuación} $x^2 + 4x + 5 = 0$ tiene \emph{no} real
        solución. Indique el criterio general: una vez escrito como
        $(x + h)^2 + k$, ¿cuándo aparece una expresión cuadrática?
        ¿desaparecer?
  \item El nombre es literal. En la Bagdad del siglo IX,
        al-Khwarizmi dibujó $x^2 + 6x = 7$ como un cuadrado de lado $x$
        con dos rectángulos $3 \times x$ pegados a dos de sus
        lados, y \emph{completó la plaza} con los faltantes
        $3 \times 3$ esquina. Dibuja la figura y explica cómo
        convierte el \omterm{def:g10:algebra:equation}{ecuación} en $(x + 3)^2 = 16$ --- luego resuelve.
\end{enumerate}

\medskip
\noindent\textbf{Parte III --- Signs, tables and two old
promises.}
\begin{enumerate}[resume]
  \item Construye el \omterm{def:g10:algebra:signtable}{mesa de signos} de $(x - 3)(x + 2)$
        (\cref{def:g10:algebra:signtable}) y resuelve
        $(x - 3)(x + 2) < 0$.
  \item Resuelve $x^2 \leq 10x - 21$ (lleva todo a un lado
        y reutilizar pregunta 7).
  \item Resuelva $\dfrac{x - 1}{x + 2} \geq 0$ con un \omterm{def:g10:algebra:signtable}{mesa de signos},
        teniendo en cuenta el valor prohibido.
  \item Una vieja promesa cumplida: para el granjero de
        problema del cerco de la reina Dido, en el volumen de Secundaria, con $20$ m de cerco, el área
        de un bolígrafo $x \times (10 - x)$ satisface
        \[
        A(x) = x(10 - x) = 25 - (x - 5)^2 .
        \]
        Verifique la identidad y lea un \emph{prueba} de
        lo que el grado 6 sólo podía conjeturar: el área máxima,
        y el exclusivo $x$ lo logra.
  \item Otro: demostrar que $x + \dfrac1x \geq 2$ para cada
        $x > 0$, con igualdad sólo en $x = 1$. (Multiplicar por
        $x$, junta, reconoce un cuadrado --- el
        aritmética: desigualdad geométrica de
        el problema de fin de semana sobre las relaciones de Euclides del volumen de la Escuela Secundaria con un traje nuevo.)
\end{enumerate}

\medskip
\noindent\textbf{Parte IV --- The golden section.}
Cortar un segmento de longitud $1$ en una parte grande $x$ y una pequeña
parte $1 - x$ para que
\emph{el conjunto es demasiado grande y lo grande es demasiado pequeño}:
$\frac{1}{x} = \frac{x}{1 - x}$.
\begin{enumerate}[resume]
  \item Muestre que la condición dice $x^2 + x - 1 = 0$.
  \item Resuelve completando el cuadrado, y mantén la raíz que
        es una longitud: $x = \frac{\sqrt5 - 1}{2} \approx 0.618$.
  \item El \emph{proporción áurea} es
        $\varphi = \frac1x = \frac{2}{\sqrt5 - 1}$. racionalizar
        el denominador (multiplicar arriba y abajo por
        $\sqrt5 + 1$, una identidad en el trabajo) y mostrar
        $\varphi = \frac{\sqrt5 + 1}{2} \approx 1.618$.
  \item Verifica \emph{exactamente} las dos identidades mágicas de
        $\varphi$:
        \[
        \varphi^2 = \varphi + 1,
        \qquad
        \frac{1}{\varphi} = \varphi - 1 .
        \]
  \item Final: explica por qué $\varphi$ es \omterm{ex:g10:numbers:classify}{irracional}
        (El problema de la irracionalidad del fin de semana del volumen de la escuela secundaria proporciona el hecho clave acerca de
        $\sqrt5$), luego calcule las relaciones $\frac53$,
        $\frac85$, $\frac{13}{8}$, $\frac{21}{13}$ de
        Números consecutivos de Hemachandra-Fibonacci.
