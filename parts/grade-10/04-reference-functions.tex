\chapter{Reference Functions}\label{ch:g10:reffunc}

A handful of simple \omterm{def:g10:functions:function}{functions} --- affine, square, inverse, square root,
cube --- appear everywhere, alone or combined. Knowing their \omterm{def:g10:functions:graph}{graphs} and
variations by heart turns many problems into a quick sketch. This chapter
studies each of them in turn and proves their variations with the
comparison technique of \cref{ch:g10:functions}.

\section{Affine functions}

\begin{definition}[Affine function]\label{def:g10:reffunc:affine}
An \emph{affine function}\index{affine function} is a \omterm{def:g10:functions:function}{function} of the form
\[
f(x) = mx + p,
\]
where $m$ and $p$ are fixed \omterm{def:g10:numbers:sets}{real numbers}. Its \omterm{def:g10:functions:graph}{graph} is a straight line:
$m$ is the \emph{slope}\index{slope} and $p$ the
\emph{$y$-intercept}\index{y-intercept@$y$-intercept} (the line crosses
the vertical axis at $(0, p)$). When $p = 0$ the \omterm{def:g10:functions:function}{function} is
\emph{linear}: $f(x) = mx$ expresses proportionality.
\end{definition}

\begin{proposition}[Slope and variations]\label{prop:g10:reffunc:affine}
Let $f(x) = mx + p$.
\begin{enumerate}
  \item For any two distinct inputs $u \neq v$:
        $m = \dfrac{f(v) - f(u)}{v - u}$ (the \omterm{def:g10:reffunc:affine}{slope} is the change of the
        output per unit change of the input).
  \item If $m > 0$, $f$ is strictly \omterm{def:g10:functions:variations}{increasing} on $\R$; if $m < 0$,
        strictly \omterm{def:g10:functions:variations}{decreasing}; if $m = 0$, constant.
\end{enumerate}
\end{proposition}

\begin{proof}
1. Compute $f(v) - f(u) = (mv + p) - (mu + p) = m(v - u)$, then divide by
$v - u \neq 0$.

2. If $u < v$, then $f(v) - f(u) = m(v-u)$ has the sign of $m$, since
$v - u > 0$: positive $m$ gives $f(u) < f(v)$ (\omterm{def:g10:functions:variations}{increasing}), negative $m$
gives $f(u) > f(v)$ (\omterm{def:g10:functions:variations}{decreasing}).
\end{proof}

\begin{example}\label{ex:g10:reffunc:affine}
Find the \omterm{def:g10:reffunc:affine}{affine function} whose \omterm{def:g10:functions:graph}{graph} passes through $A(1, 3)$ and
$B(4, 9)$. The \omterm{def:g10:reffunc:affine}{slope} is
\[
m = \frac{9 - 3}{4 - 1} = \frac{6}{3} = 2 .
\]
Then $f(x) = 2x + p$, and $f(1) = 3$ gives $2 + p = 3$, so $p = 1$:
$f(x) = 2x + 1$. Check with $B$: $f(4) = 9$.
\end{example}

\section{The square function}

\begin{proposition}[Square function]\label{prop:g10:reffunc:square}
The \omterm{def:g10:functions:function}{function} $f(x) = x^2$, defined on $\R$:
\begin{enumerate}
  \item is strictly \omterm{def:g10:functions:variations}{decreasing} on $\intoc{-\infty}{0}$ and strictly
        \omterm{def:g10:functions:variations}{increasing} on $\intco{0}{+\infty}$, with \omterm{def:g10:functions:extrema}{minimum} $0$ at $x = 0$;
  \item satisfies $f(-x) = f(x)$ for all $x$: its \omterm{def:g10:functions:graph}{graph}, a
        \emph{parabola}\index{parabola}, is symmetric about the vertical
        axis.
\end{enumerate}
\end{proposition}

\begin{proof}
1. Let $0 \leq u < v$. Then
\[
f(v) - f(u) = v^2 - u^2 = (v - u)(v + u) > 0,
\]
since $v - u > 0$ and $v + u > 0$: $f$ is strictly \omterm{def:g10:functions:variations}{increasing} on
$\intco{0}{+\infty}$. If $u < v \leq 0$, then $v - u > 0$ but
$v + u < 0$, so $f(v) - f(u) < 0$: strictly \omterm{def:g10:functions:variations}{decreasing}. Finally
$x^2 \geq 0 = f(0)$ for all $x$.

2. $(-x)^2 = x^2$; so the points $(x, x^2)$ and $(-x, x^2)$, mirror \omterm{def:g10:functions:function}{images}
across the vertical axis, are both on the \omterm{def:g10:functions:graph}{graph}.
\end{proof}

\begin{remark}[Squares and order]
Because the square \omterm{def:g10:functions:function}{function} decreases on the negative side, taking squares
\emph{reverses} the order of negative numbers: $-3 < -2$ but $9 > 4$.
Never square both sides of an inequality without checking signs.
\end{remark}

\section{The inverse function}

\begin{proposition}[Inverse function]\label{prop:g10:reffunc:inverse}
The \omterm{def:g10:functions:function}{function} $f(x) = \dfrac1x$, defined for $x \neq 0$:
\begin{enumerate}
  \item is strictly \omterm{def:g10:functions:variations}{decreasing} on $\intoo{-\infty}{0}$ and strictly
        \omterm{def:g10:functions:variations}{decreasing} on $\intoo{0}{+\infty}$;
  \item satisfies $f(-x) = -f(x)$: its \omterm{def:g10:functions:graph}{graph}, a
        \emph{hyperbola}\index{hyperbola}, is symmetric about the origin.
\end{enumerate}
\end{proposition}

\begin{proof}
1. Let $0 < u < v$. Then
\[
f(v) - f(u) = \frac1v - \frac1u = \frac{u - v}{uv} < 0,
\]
since $u - v < 0$ and $uv > 0$: strictly \omterm{def:g10:functions:variations}{decreasing} on
$\intoo{0}{+\infty}$. On $\intoo{-\infty}{0}$ the same quotient has
$u - v < 0$ and $uv > 0$ again (product of two negatives), so $f$
decreases there too.

2. $\frac{1}{-x} = -\frac1x$.
\end{proof}

\begin{remark}
Careful: $\frac1x$ is \emph{not} \omterm{def:g10:functions:variations}{decreasing} on its whole \omterm{def:g10:functions:function}{domain}. From
$u = -1$ to $v = 1$ the value jumps from $-1$ up to $1$. The two branches
must be studied separately.
\end{remark}

\section{Square root and cube}

\begin{proposition}[Square root function]\label{prop:g10:reffunc:sqrt}
The \omterm{def:g10:functions:function}{function} $f(x) = \sqrt{x}$, defined on $\intco{0}{+\infty}$, is
strictly \omterm{def:g10:functions:variations}{increasing}.
\end{proposition}

\begin{proof}
Let $0 \leq u < v$. Multiply and divide by the \emph{conjugate}:
\[
\sqrt v - \sqrt u
= \frac{(\sqrt v - \sqrt u)(\sqrt v + \sqrt u)}{\sqrt v + \sqrt u}
= \frac{v - u}{\sqrt v + \sqrt u} > 0,
\]
since $v - u > 0$ and $\sqrt v + \sqrt u > 0$ (note $v > 0$, so
$\sqrt v > 0$).
\end{proof}

\begin{proposition}[Cube function]\label{prop:g10:reffunc:cube}
The \omterm{def:g10:functions:function}{function} $f(x) = x^3$, defined on $\R$, is strictly \omterm{def:g10:functions:variations}{increasing}, and
its \omterm{def:g10:functions:graph}{graph} is symmetric about the origin.
\end{proposition}
\admitted

\begin{proposition}[Comparing $x$, $x^2$ and $\sqrt x$]\label{prop:g10:reffunc:compare}
For $0 < x < 1$:\ \ $x^2 < x < \sqrt x$.\quad
For $x > 1$:\ \ $\sqrt x < x < x^2$.\quad
At $x = 0$ and $x = 1$ the three values are equal.
\end{proposition}

\begin{proof}
Suppose $0 < x < 1$. Multiplying $x < 1$ by $x > 0$ gives $x^2 < x$.
Next, $x < \sqrt x$: since both sides are positive and squaring is
\omterm{def:g10:functions:variations}{increasing} on positive numbers, this inequality is equivalent to
$x^2 < x$, which we just proved. For $x > 1$, multiplying $x > 1$ by $x$
gives $x^2 > x$, and $\sqrt x < x$ follows by the same squaring argument.
\end{proof}

\begin{method}[Comparing images]\label{met:g10:reffunc:compare}
To compare $f(a)$ and $f(b)$ for a reference \omterm{def:g10:functions:function}{function} $f$:
\begin{enumerate}
  \item place $a$ and $b$ on an \omterm{def:g10:numbers:interval}{interval} where the variations of $f$ are
        known;
  \item if $f$ increases there, \omterm{def:g10:functions:function}{images} are in the same order as the
        inputs; if $f$ decreases, the order is reversed;
  \item if $a$ and $b$ are not in the same \omterm{def:g10:numbers:interval}{interval} of monotonicity, deal
        with signs first (e.g.\ for the square: a negative and a positive
        input).
\end{enumerate}
\end{method}

\begin{example}
Compare $\dfrac{1}{2.3}$ and $\dfrac{1}{2.4}$: both inputs are in
$\intoo{0}{+\infty}$, where the inverse \omterm{def:g10:functions:function}{function} decreases, and
$2.3 < 2.4$, so $\dfrac{1}{2.3} > \dfrac{1}{2.4}$. Compare $(-1.2)^2$
and $(-1.5)^2$: on $\intoc{-\infty}{0}$ the square decreases, and
$-1.5 < -1.2$, so $(-1.5)^2 > (-1.2)^2$ --- indeed $2.25 > 1.44$.
\end{example}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:g10:reffunc:1}
For each \omterm{def:g10:reffunc:affine}{affine function}, give its \omterm{def:g10:reffunc:affine}{slope}, its $y$-intercept, and say
whether it is \omterm{def:g10:functions:variations}{increasing} or \omterm{def:g10:functions:variations}{decreasing}:
\[
f(x) = 3x - 2, \qquad
g(x) = -\tfrac12 x + 4, \qquad
h(x) = 7, \qquad
k(x) = -x .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:reffunc:2}
Find the \omterm{def:g10:reffunc:affine}{affine function} whose \omterm{def:g10:functions:graph}{graph} passes through $(2, 1)$ and
$(5, 10)$; then the one through $(-1, 4)$ and $(3, -4)$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:reffunc:3}
Without a calculator, compare:
\[
(3.1)^2 \text{ and } (3.2)^2; \qquad
(-2.7)^2 \text{ and } (-2.8)^2; \qquad
\frac{1}{5.1} \text{ and } \frac{1}{5.2}; \qquad
\sqrt{17} \text{ and } \sqrt{15}.
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:reffunc:4}
Using the \omterm{def:g10:functions:graph}{graph} of the square \omterm{def:g10:functions:function}{function}, solve $x^2 = 16$, then
$x^2 \leq 16$, then $x^2 > 9$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:reffunc:5}
Let $x \in \intoo{0}{1}$. Order the numbers $x$, $x^2$, $x^3$ and
$\sqrt x$ from smallest to largest, and check your answer with
$x = 0.25$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:reffunc:6}
Solve graphically, then algebraically: $\dfrac1x = 2$, and
$\dfrac1x < 2$ for $x > 0$. What changes if we also allow $x < 0$?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:reffunc:7}
Knowing that the square \omterm{def:g10:functions:function}{function} decreases on $\intoc{-\infty}{0}$ and
increases on $\intco{0}{+\infty}$, frame $x^2$ when:
\[
\text{(a) } x \in \intcc{2}{5};
\qquad
\text{(b) } x \in \intcc{-3}{-1};
\qquad
\text{(c) } x \in \intcc{-2}{3}.
\]
(In case (c), be careful: the \omterm{def:g10:functions:extrema}{minimum} of $x^2$ is not at an endpoint.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:reffunc:8}
A taxi company charges a fixed fee of $4$ plus $1.5$ per kilometer;
another charges no fee and $2$ per kilometer. Model each price by an
\omterm{def:g10:reffunc:affine}{affine function} of the distance $x$, plot both lines, and find from which
distance the first company is cheaper.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:reffunc:9}
Show that for all $a, b \geq 0$: $\sqrt{a + b} \leq \sqrt a + \sqrt b$.
(Hint: both sides are nonnegative, so compare their squares.) When is
there equality?
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g10:reffunc:10}
Let $f(x) = \dfrac{2x + 1}{x - 1}$, defined for $x \neq 1$.
\begin{enumerate}
  \item Show that $f(x) = 2 + \dfrac{3}{x - 1}$ for all $x \neq 1$.
  \item Deduce the variations of $f$ on $\intoo{1}{+\infty}$ from those
        of the inverse \omterm{def:g10:functions:function}{function}.
\end{enumerate}
\end{exercise}

\section{Problem: The laws of nature speak in reference functions}

\begin{problem}[{Weekend problem --- braking distances grow as
$v^2$, levers obey $1/d$, planets follow $a^{3/2}$: reading
physics with the reference \omterm{def:g10:functions:function}{functions}}]
\label{pb:g10:reffunc:1}
Open a physics book and the same few \omterm{def:g10:functions:graph}{graphs} appear on every page:
the parabola, the hyperbola, the root curve. Nature writes its
laws with this chapter's reference \omterm{def:g10:functions:function}{functions} --- and knowing
their shapes (\cref{prop:g10:reffunc:square},
\cref{prop:g10:reffunc:inverse}, \cref{prop:g10:reffunc:sqrt},
\cref{prop:g10:reffunc:cube}) is enough to brake a car, balance a
lever, and time the planets. The grand finale is Kepler's third
law, read straight off the solar system's data.

\medskip
\noindent\textbf{Part I --- The square law of braking.}
A rule of thumb for a car's braking distance on dry road:
$d = \left(\frac{v}{10}\right)^{2}$ meters, at speed $v$ km/h.
\begin{enumerate}
  \item Compute the braking distances at $30$, $50$, $90$ and
        $130$ km/h.
  \item Doubling the speed multiplies the braking distance by
        how much? Explain from the square \omterm{def:g10:functions:function}{function}'s scaling,
        and check on $50 \to 100$ km/h.
  \item Forensics: skid marks measure $50$ m. What speed does
        the formula convict the driver of (to the km/h)? Which
        reference \omterm{def:g10:functions:function}{function} answered --- and on which \omterm{def:g10:functions:function}{domain} is
        the reading unambiguous
        (\cref{prop:g10:reffunc:sqrt})?
  \item Compare the extra distance caused by \emph{one} km/h
        more, at low and at high speed: compute
        $d(31) - d(30)$ and $d(91) - d(90)$. Which property of
        the parabola (\cref{prop:g10:reffunc:square}) do the two
        answers illustrate?
  \item Real stopping adds reaction time --- about one second,
        during which the car covers $0.28v$ meters:
        $D(v) = 0.28v + \left(\frac{v}{10}\right)^2$. Compute
        $D(50)$ and $D(130)$. Which of the two terms --- affine
        or square --- dominates at city speed, and which on the
        highway?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The hyperbolas of daily life.}
\begin{enumerate}[resume]
  \item A lever balances when force $\times$ distance is the
        same on both sides. A $60$ kg child sits $2$ m from the
        pivot; the balancing force at distance $d$ on the other
        side is $F(d) = \frac{120}{d}$ (in kg-equivalents).
        Compute $F(0.5)$, $F(1)$, $F(3)$.
  \item What do the variations of the inverse \omterm{def:g10:functions:function}{function}
        (\cref{prop:g10:reffunc:inverse}) say about levers ---
        and why does Archimedes' boast (``give me a place to
        stand and I shall move the Earth'') hide in the
        hyperbola's tail? What forbids $d = 0$?
  \item Sound and light fade with the \emph{square} of the
        distance: at $1$ m from a lamp the intensity is $100$
        units; give it at $2$, $3$ and $10$ m. Explain the
        exponent $2$ with a sphere: over what area has the light
        spread at distance $d$ (Archimedes' tombstone problem, in the Middle School volume)?
  \item A bus for the school trip costs $600$ euros, split
        equally among $n$ participants. Tabulate the cost per
        head for $n = 10, 20, 30$, find how many participants
        bring it to $25$ euros or less, and state what the
        hyperbola's shape promises --- and refuses --- as $n$
        grows.
  \item Producing $x$ posters costs $2x + 3$ euros ($3$ euros of
        setup). Show that the \emph{average} cost per poster is
        $a(x) = 2 + \frac3x$, describe its variations, and
        interpret its horizontal asymptote economically.
        (Compare the algebra of \cref{exo:g10:reffunc:10}.)
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Kepler's harmony.}
For each planet, let $a$ be its mean distance to the Sun (in
astronomical units, Earth $= 1$) and $T$ its period (in years).
Data: Mars $a = 1.52$, $T = 1.88$; Jupiter $a = 5.20$,
$T = 11.86$; Saturn $a = 9.54$, $T = 29.4$.
\begin{enumerate}[resume]
  \item Compute $a^3$ and $T^2$ for the Earth, Mars and Jupiter.
        What did Kepler notice in 1618?
  \item State the law as a \omterm{def:g10:functions:function}{function}: $T = \sqrt{a^3} = a\sqrt a$.
        Test it on Saturn.
  \item Two predictions: an asteroid orbits at $a = 4$ AU ---
        find its period (the numbers come out whole); a comet
        returns every $27$ years --- find its mean distance
        (look for a perfect cube).
  \item Which three reference \omterm{def:g10:functions:function}{functions} does the law braid
        together? And its scaling rule: when $a$ is multiplied
        by $4$, what happens to $T$? (Check with the asteroid
        against the Earth.)
  \item Kepler read this law in Tycho Brahe's data seventy years
        before Newton's gravitation explained it. In one
        sentence: what does this episode say about the power of
        recognizing a reference \omterm{def:g10:functions:function}{function} in a table of numbers?
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The tortoise and the hare.}
\begin{enumerate}[resume]
  \item Order $\sqrt x$, $x$ and $x^2$ on $\intoo{0}{1}$ and on
        $\intoo{1}{+\infty}$
        (\cref{prop:g10:reffunc:compare}), and verify the two
        orderings at $x = 0.25$ and $x = 4$.
  \item Solve $\sqrt x > x$ completely (for $x \geq 0$, square
        soundly and finish with a \omterm{def:g10:algebra:signtable}{sign table}).
  \item Bring in the cube: order $x$, $x^2$, $x^3$ at
        $x = 0.5$ and at $x = 2$, and find all points where the
        square and the cube \omterm{def:g10:functions:function}{functions} cross.
  \item Two savings schemes pay, after $t$ years, $100\sqrt t$
        euros (scheme A) or $10 t^2$ euros (scheme B). Show that
        B overtakes A when $t^3 = 100$, and give the crossover
        time to the month. Moral about roots against powers in
        the long run?
  \item Finale: for each law of this problem --- braking
        ($v^2$), the lever ($1/d$), the lamp ($1/d^2$), Kepler
        ($a^{3/2}$), scheme A ($\sqrt t$) --- say in one clause
        which feature of its reference \omterm{def:g10:functions:graph}{graph} carries the
        physical meaning (steepening rise, vertical asymptote,
        spreading sphere, scaling exponent, flattening growth).
        Conclusion: the reference \omterm{def:g10:functions:function}{functions} are the alphabet;
        nature writes with them.
\end{enumerate}
\end{problem}
