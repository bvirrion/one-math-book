\chapter{Functions}\label{ch:g10:functions}

A \omterm{def:g10:functions:function}{function} is a machine that takes a number in and produces a number out,
always the same output for the same input. \omterm{def:g10:functions:function}{Functions} are the central
objects of the analysis you will build over the next years
(\cref{ch:g11:quad} onward); this chapter sets up the vocabulary ---
\omterm{def:g10:functions:function}{domain}, \omterm{def:g10:functions:function}{image}, \omterm{def:g10:functions:graph}{graph}, variations --- and trains the essential skill of
reading information off a \omterm{def:g10:functions:graph}{graph}.

\section{Vocabulary: images and preimages}

\begin{definition}[Function]\label{def:g10:functions:function}
Let $D$ be a set of \omterm{def:g10:numbers:sets}{real numbers}. A \emph{function}\index{function} $f$
defined on $D$ associates to each number $x \in D$ exactly one \omterm{def:g10:numbers:sets}{real
number}, written $f(x)$ and called the \emph{image}\index{image} of $x$.
The set $D$ is the \emph{domain}\index{domain} of $f$. We write
\[
f : x \longmapsto f(x).
\]
If $f(x) = y$, then $x$ is a \emph{preimage}\index{preimage} of $y$.
\end{definition}

\begin{remark}
Each $x$ in the \omterm{def:g10:functions:function}{domain} has exactly one \omterm{def:g10:functions:function}{image}, but a number $y$ can have
several \omterm{def:g10:functions:function}{preimages}, or none. For $f(x) = x^2$: the \omterm{def:g10:functions:function}{image} of $3$ is $9$, and
$9$ has two \omterm{def:g10:functions:function}{preimages}, $3$ and $-3$; the number $-4$ has no \omterm{def:g10:functions:function}{preimage}.
\end{remark}

\begin{example}[Finding a domain]\label{ex:g10:functions:domain}
When a \omterm{def:g10:functions:function}{function} is given by a formula, its \omterm{def:g10:functions:function}{domain} is the set of $x$ for
which the formula makes sense.
\begin{itemize}
  \item $f(x) = 3x^2 - 5x + 1$ makes sense for every $x$: the \omterm{def:g10:functions:function}{domain} is
        $\R$.
  \item $g(x) = \dfrac{1}{x - 2}$ requires $x - 2 \neq 0$: the \omterm{def:g10:functions:function}{domain} is
        all reals except $2$.
  \item $h(x) = \sqrt{x - 3}$ requires $x - 3 \geq 0$: the \omterm{def:g10:functions:function}{domain} is
        $\intco{3}{+\infty}$.
\end{itemize}
\end{example}

\begin{example}[Computing images and preimages]\label{ex:g10:functions:images}
Let $f(x) = x^2 - 4x + 3$, defined on $\R$.

\emph{\omterm{def:g10:functions:function}{Image} of $5$:} substitute $x = 5$:
$f(5) = 25 - 20 + 3 = 8$.

\emph{\omterm{def:g10:functions:function}{Preimages} of $3$:} solve $f(x) = 3$:
\[
x^2 - 4x + 3 = 3
\quad\Longleftrightarrow\quad
x^2 - 4x = 0
\quad\Longleftrightarrow\quad
x(x - 4) = 0,
\]
so the \omterm{def:g10:functions:function}{preimages} of $3$ are $0$ and $4$.
\end{example}

\section{The graph of a function}

\begin{definition}[Graph]\label{def:g10:functions:graph}
In a coordinate system, the \emph{graph}\index{graph} (or \emph{curve}) of
$f$ is the set of all points $(x, f(x))$ for $x$ in the \omterm{def:g10:functions:function}{domain}. In other
words, a point $(x, y)$ lies on the graph exactly when $y = f(x)$.
\end{definition}

\begin{method}[Reading a graph]\label{met:g10:functions:reading}
On the \omterm{def:g10:functions:graph}{graph} of $f$:
\begin{enumerate}
  \item \emph{\omterm{def:g10:functions:function}{image} of $a$}: go vertically from $x = a$ on the horizontal
        axis to the curve, then horizontally to the vertical axis; the
        value read there is $f(a)$;
  \item \emph{\omterm{def:g10:functions:function}{preimages} of $b$}: draw the horizontal line $y = b$; the
        \omterm{def:g10:functions:function}{preimages} are the $x$-coordinates of all its \omterm{def:g10:numbers:interunion}{intersection} points
        with the curve;
  \item \emph{solutions of $f(x) = k$}: same as the \omterm{def:g10:functions:function}{preimages} of $k$;
  \item \emph{solutions of $f(x) \leq k$}: the $x$ for which the curve is
        on or below the line $y = k$.
\end{enumerate}
\end{method}

\begin{tikzpicture}
\begin{axis}[omaxis,

  \fill (axis cs:-2,2) circle (1.5pt);
  \fill (axis cs:2,2) circle (1.5pt);

  \draw[dashed] (axis cs:2,2) -- (axis cs:2,0);
\end{axis}
\end{tikzpicture}

{\small Solving $f(x) = k$ graphically: the solutions $x_1$ and $x_2$ are
the abscissas of the \omterm{def:g10:numbers:interunion}{intersection} points of the curve with the horizontal
line $y = k$. Here $f(x) \leq k$ holds on $\intcc{x_1}{x_2}$, where the
curve is below the line.}
\end{omfigure}

\begin{example}\label{ex:g10:functions:membership}
Is the point $A(2, 5)$ on the \omterm{def:g10:functions:graph}{graph} of $f(x) = x^2 + 1$? Compute
$f(2) = 4 + 1 = 5$: yes, since $f(2)$ equals the $y$-coordinate of $A$.
The point $B(3, 8)$ is not on the \omterm{def:g10:functions:graph}{graph}, because $f(3) = 10 \neq 8$.
\end{example}

\section{Variations of a function}

\begin{definition}[Increasing, decreasing]\label{def:g10:functions:variations}
Let $f$ be defined on an \omterm{def:g10:numbers:interval}{interval} $I$.
\begin{itemize}
  \item $f$ is \emph{increasing}\index{increasing function} on $I$ when
        for all $u, v \in I$, if $u < v$ then $f(u) \leq f(v)$: the
        outputs grow with the inputs, and the \omterm{def:g10:functions:graph}{graph} climbs from left to
        right.
  \item $f$ is \emph{decreasing}\index{decreasing function} on $I$ when
        $u < v$ implies $f(u) \geq f(v)$: the \omterm{def:g10:functions:graph}{graph} falls from left to
        right.
\end{itemize}
With strict inequalities $f(u) < f(v)$ (resp.\ $>$) we say
\emph{strictly} increasing (resp.\ decreasing).
\end{definition}

\begin{definition}[Variation table]\label{def:g10:functions:table}
A \emph{variation table}\index{variation table} summarizes on which
\omterm{def:g10:numbers:interval}{intervals} $f$ increases and decreases, with arrows $\nearrow$ and
$\searrow$, and records the values of $f$ at the turning points.
\end{definition}

\begin{example}\label{ex:g10:functions:table}
Consider the \omterm{def:g10:functions:function}{function} graphed below on $\intcc{-2}{4}$.

\begin{tikzpicture}
\begin{axis}[omaxis,

  xmin=-2.8, xmax=4.6, ymin=-2.4, ymax=3.8,
  xtick={-2,1,4}, ytick={-1.5,3}, yticklabels={$-1.5$,$3$}]
  \addplot[omDef, thick, domain=-2:4, samples=120] {-0.5*(x-1)^2 + 3};
  \fill (axis cs:-2,-1.5) circle (1.5pt);
  \fill (axis cs:4,-1.5) circle (1.5pt);
  \fill (axis cs:1,3) circle (1.5pt);
\end{axis}
\end{tikzpicture}

{\small A \omterm{def:g10:functions:function}{function} defined on $\intcc{-2}{4}$, \omterm{def:g10:functions:variations}{increasing} then \omterm{def:g10:functions:variations}{decreasing}.}
\end{omfigure}

Reading the \omterm{def:g10:functions:graph}{graph}: $f$ increases from $f(-2) = -1.5$ up to its \omterm{def:g10:functions:extrema}{maximum}
$f(1) = 3$, then decreases down to $f(4) = -1.5$. Its \omterm{def:g10:functions:table}{variation table} is
\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{c|ccccc}
$x$ & $-2$ & & $1$ & & $4$ \\
\hline
$f$ & $-1.5$ & $\nearrow$ & $3$ & $\searrow$ & $-1.5$ \\
\end{tabular}
\end{center}
\end{example}

\begin{example}[Proving a variation]\label{ex:g10:functions:proof}
Show that $f(x) = 3x + 1$ is strictly \omterm{def:g10:functions:variations}{increasing} on $\R$. Take any
$u < v$ and compare the \omterm{def:g10:functions:function}{images}:
\[
f(v) - f(u) = (3v + 1) - (3u + 1) = 3(v - u) > 0,
\]
because $v - u > 0$. So $f(u) < f(v)$: the \omterm{def:g10:functions:function}{function} is strictly
\omterm{def:g10:functions:variations}{increasing}. The same computation with a negative slope, e.g.\
$g(x) = -2x + 5$, gives $g(v) - g(u) = -2(v-u) < 0$: strictly \omterm{def:g10:functions:variations}{decreasing}.
\end{example}

\section{Extrema}

\begin{definition}[Maximum, minimum]\label{def:g10:functions:extrema}
Let $f$ be defined on $D$ and $a \in D$. The value $f(a)$ is the
\emph{maximum}\index{maximum} of $f$ on $D$ when $f(x) \leq f(a)$ for all
$x \in D$; it is the \emph{minimum}\index{minimum} when
$f(x) \geq f(a)$ for all $x \in D$. On a \omterm{def:g10:functions:graph}{graph}, they are the highest and
lowest points of the curve.
\end{definition}

\begin{example}\label{ex:g10:functions:extremum}
Show that $f(x) = (x - 1)^2 + 2$ has \omterm{def:g10:functions:extrema}{minimum} $2$ on $\R$, attained at
$x = 1$. For every $x$, the square $(x-1)^2$ is $\geq 0$, so
$f(x) = (x-1)^2 + 2 \geq 2$; and $f(1) = 0 + 2 = 2$, so the value $2$ is
actually reached. Both facts together prove that $2$ is the \omterm{def:g10:functions:extrema}{minimum}.
\end{example}

\begin{omfigure}
\begin{tikzpicture}
\begin{axis}[omaxis,
  width=7.6cm, height=5.2cm,
  xmin=-1.8, xmax=3.8, ymin=-0.9, ymax=6.4,
  xtick={1}, ytick={2}]
  \addplot[omDef, thick, domain=-1.0:3.0, samples=100] {(x-1)^2 + 2};
  \addplot[omThm, dashed, domain=-1.6:3.6] {2};
  \fill (axis cs:1,2) circle (1.5pt);
  \node[below right, font=\small] at (axis cs:1,2) {$(1,2)$};
\end{axis}
\end{tikzpicture}

{\small The curve of $f(x) = (x-1)^2 + 2$ never goes below the dashed line
$y = 2$, and touches it at $x = 1$: the \omterm{def:g10:functions:extrema}{minimum} of $f$ is $2$.}
\end{omfigure}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:g10:functions:1}
Let $f(x) = 2x^2 - 3x + 1$. Compute the \omterm{def:g10:functions:function}{images} of $0$, $2$, $-1$ and
$\frac12$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:functions:2}
Give the \omterm{def:g10:functions:function}{domain} of each \omterm{def:g10:functions:function}{function}:
\[
f(x) = 5x - 2, \qquad
g(x) = \frac{3}{x + 4}, \qquad
h(x) = \sqrt{2x - 6}, \qquad
k(x) = \frac{1}{x^2 + 1}.
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:functions:3}
Let $f(x) = x^2 - 2x$. Find all \omterm{def:g10:functions:function}{preimages} of $0$, of $3$, and of $-1$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:functions:4}
Does the point $A(-1, 4)$ belong to the \omterm{def:g10:functions:graph}{graph} of $f(x) = x^2 - 2x + 1$?
And the point $B(2, 1)$? Justify by a computation.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:functions:5}
A \omterm{def:g10:functions:function}{function} $g$ defined on $\intcc{-3}{5}$ has the \omterm{def:g10:functions:table}{variation table}
\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{c|ccccccc}
$x$ & $-3$ & & $0$ & & $3$ & & $5$ \\
\hline
$g$ & $1$ & $\searrow$ & $-2$ & $\nearrow$ & $4$ & $\searrow$ & $0$ \\
\end{tabular}
\end{center}
\begin{enumerate}
  \item What are the \omterm{def:g10:functions:extrema}{maximum} and the \omterm{def:g10:functions:extrema}{minimum} of $g$ on $\intcc{-3}{5}$?
  \item Compare $g(-1)$ and $g(-0.5)$ without knowing their values.
  \item How many solutions does the \omterm{def:g10:algebra:equation}{equation} $g(x) = 0$ have at most on
        each \omterm{def:g10:numbers:interval}{interval} of the table?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:functions:6}
Let $f(x) = -2x + 7$. Show, by comparing $f(u)$ and $f(v)$ for $u < v$,
that $f$ is strictly \omterm{def:g10:functions:variations}{decreasing} on $\R$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:functions:7}
Show that the \omterm{def:g10:functions:function}{function} $f(x) = x^2 + 6x + 11$ has a \omterm{def:g10:functions:extrema}{minimum} on $\R$, and
give its value and where it is attained. (Hint: write
$f(x) = (x + 3)^2 + c$ for the right constant $c$.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:functions:8}
A rectangular garden has perimeter $40$ m. Let $x$ be its width, in
meters.
\begin{enumerate}
  \item Express the length, then the area $A(x)$, as \omterm{def:g10:functions:function}{functions} of $x$.
        For which $x$ does this make sense?
  \item Compute $A(4)$, $A(8)$, $A(10)$, $A(12)$ and $A(16)$. What do you
        conjecture about the shape of largest area?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:functions:9}
Let $f(x) = \dfrac{1}{x^2 + 1}$, defined on $\R$.
\begin{enumerate}
  \item Show that $0 < f(x) \leq 1$ for every real $x$.
  \item Does $f$ have a \omterm{def:g10:functions:extrema}{maximum} on $\R$? A \omterm{def:g10:functions:extrema}{minimum}? Justify.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g10:functions:10}
Let $f(x) = x^2$ on $\R$.
\begin{enumerate}
  \item Let $0 \leq u < v$. Factor $f(v) - f(u)$ and deduce that $f$ is
        strictly \omterm{def:g10:functions:variations}{increasing} on $\intco{0}{+\infty}$.
  \item Adapt the argument to show that $f$ is strictly \omterm{def:g10:functions:variations}{decreasing} on
        $\intoc{-\infty}{0}$.
\end{enumerate}
\end{exercise}

\section{Problem: Machines fed their own output}

\begin{problem}[{Weekend problem --- fixed points, iteration, and
three famous machines: Heron's improver, the squarer, and the
unsolved hailstone}]
\label{pb:g10:functions:1}
A \omterm{def:g10:functions:function}{function} is a machine: a number goes in, a number comes out.
The most interesting experiments feed the machine \emph{its own
output}, again and again --- and then two questions dominate:
does the process settle somewhere (a \emph{\omterm{pb:g10:functions:1}{fixed point}}), and
how does the machine's variation drive it there? One of this
problem's machines has been polishing square roots for two
thousand years; another guards the most famous unsolved problem
elementary mathematics has to offer.

\medskip
\noindent\textbf{Part I --- Heron's machine.}
Consider the \omterm{def:g10:functions:function}{function}
\[
f(x) = \frac12\left(x + \frac2x\right).
\]
\begin{enumerate}
  \item Give the \omterm{def:g10:functions:function}{domain} of $f$, and compute $f(1)$, $f(2)$ and
        $f\!\left(\frac32\right)$ as exact fractions.
  \item Find all \omterm{def:g10:functions:function}{preimages} of $\frac32$: solve
        $f(x) = \frac32$ (clear the denominator and use the
        methods of \cref{pb:g10:algebra:1}).
  \item A \emph{fixed point}\index{fixed point} of $f$ is a
        number unchanged by the machine: $f(x) = x$. Find both
        \omterm{pb:g10:functions:1}{fixed points} of $f$. Which old acquaintance guards the
        positive one?
  \item Starting from $x = 1$, feed the machine its own output
        three times and record the exact results. Which sequence
        of the irrationality weekend problem of the Middle School volume have you rebuilt --- and what was
        ``Heron's recipe'', in the language of \omterm{def:g10:functions:function}{functions}?
  \item On one picture, sketch the \omterm{def:g10:functions:graph}{graph} of $f$ (for $x > 0$)
        and the diagonal line $y = x$. Where are the \omterm{pb:g10:functions:1}{fixed
        points} on this picture? (Compare
        the two-thermometers weekend problem of the Middle School volume: the temperature reading the same
        in both scales was the same idea.)
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Why the machine cannot miss.}
\begin{enumerate}[resume]
  \item For $0 < u < v$, show by reduction to a common
        denominator that
        \[
        f(v) - f(u) = \frac{(v - u)(uv - 2)}{2uv} ,
        \]
        and deduce the variations of $f$ on
        $\intoo{0}{+\infty}$: \omterm{def:g10:functions:variations}{decreasing} up to a point, then
        \omterm{def:g10:functions:variations}{increasing} --- which point? (Same technique as
        \cref{exo:g10:functions:10}.)
  \item Deduce that on $\intoo{0}{+\infty}$ the \omterm{def:g10:functions:function}{function} $f$ has
        a \omterm{def:g10:functions:extrema}{minimum}, worth exactly $\sqrt2$, attained at
        $\sqrt2$. What does this say about \emph{every} output
        of the machine (positive inputs)?
  \item Show that if $x > \sqrt2$ then
        $\sqrt2 < f(x) < x$: a too-big guess is always improved,
        never overshot. (For $f(x) < x$, compare $\frac2x$ with
        $x$.)
  \item Assemble questions 6--8 into the \omterm{def:g10:functions:table}{variation table} of $f$
        on $\intoo{0}{+\infty}$
        (\cref{def:g10:functions:table}), \omterm{def:g10:functions:extrema}{minimum} included.
  \item A second machine: $g(x) = x^2$. Its \omterm{pb:g10:functions:1}{fixed points} are
        $0$ and $1$ (check). Iterate $g$ four times from
        $x = 0.9$, then four times from $x = 1.1$ (calculator,
        three decimals). One \omterm{pb:g10:functions:1}{fixed point} attracts, the other
        repels: which is which?
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The hailstone machine.}
On the whole numbers, define: $h(n) = \frac n2$ if $n$ is even,
and $h(n) = 3n + 1$ if $n$ is odd. Numbers bounce under $h$ like
hailstones in a storm cloud.
\begin{enumerate}[resume]
  \item Compute the full flight of $7$ under $h$, down to $1$.
        How many steps does it take, and how high does it fly?
  \item Start the flight of $27$ and compute ten steps. (Its
        full flight lasts $111$ steps and peaks at $9\,232$ ---
        from a starting point of $27$.) What lesson about simple
        rules do $7$ and $27$ teach?
  \item Explain why any flight that reaches a power of $2$
        crashes straight to $1$, and what happens at $1$
        (compute $h(1)$, $h(4)$, $h(2)$). Does $h$ have any
        \omterm{pb:g10:functions:1}{fixed point} among the positive whole numbers? What does
        the end of every observed flight form instead?
  \item The \emph{Collatz conjecture} claims every starting
        number eventually reaches $1$. Computers have verified
        it far beyond $10^{20}$; no human has proved it. What
        two kinds of discovery would \emph{disprove} it? And why
        does massive verification still not close the case
        (the matchstick-oracle weekend problem of the Middle School volume rang this bell)?
  \item ``Mathematics is not yet ripe for such questions,'' said
        Paul Erd\H{o}s of this conjecture. In one sentence: what
        separates the hailstone machine from Heron's, where
        questions 6--8 settled everything?
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The machine user's manual.}
\begin{enumerate}[resume]
  \item \omterm{def:g10:functions:function}{Domains} as \omterm{def:g10:numbers:interval}{intervals} (\cref{pb:g10:numbers:1}'s
        language): give the \omterm{def:g10:functions:function}{domains} of
        $x \mapsto \sqrt{x - 3}$, \
        $x \mapsto \frac{1}{x^2 - 9}$, \
        $x \mapsto \sqrt{9 - x^2}$.
  \item Back to Heron's $f$: find the exact \omterm{def:g10:functions:function}{preimages} of $3$
        (solve $f(x) = 3$ by completing the square).
  \item A stone is thrown upward: its height is
        $H(t) = 20t - 5t^2$ meters after $t$ seconds. On which
        time \omterm{def:g10:numbers:interval}{interval} does the formula make physical sense?
        Complete the square to find the maximal height and its
        time (\cref{def:g10:functions:extrema}), and give the
        \omterm{def:g10:functions:table}{variation table}.
  \item For how long is the stone at least $15$ m high? (Solve
        $H(t) \geq 15$ with a \omterm{def:g10:algebra:signtable}{sign table},
        \cref{met:g10:algebra:signtable}.)
  \item Finale: this problem ran four machines --- Heron's
        improver, the squarer, the hailstone, the stone's
        height. In two or three sentences, state what every
        \omterm{def:g10:functions:function}{function} has (a \omterm{def:g10:functions:function}{domain}, a rule, a \omterm{def:g10:functions:graph}{graph}) and the two
        great questions this problem asked of each (where does
        iteration settle? how does the \omterm{def:g10:functions:function}{function} vary?) ---
        questions the next chapters take up one by one.
\end{enumerate}
\end{problem}
