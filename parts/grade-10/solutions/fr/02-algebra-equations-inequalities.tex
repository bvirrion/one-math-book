\section*{Chapitre \ref{ch:g10:algebra} --- Algèbre~: équations et inéquations}

\begin{solution}{exo:g10:algebra:1}
$3(2x-5) - 2(x-7) = 6x - 15 - 2x + 14 = 4x - 1$.

$(x+4)(2x-1) = 2x^2 - x + 8x - 4 = 2x^2 + 7x - 4$.

$(3x-2)^2 = 9x^2 - 12x + 4$.

$(5-x)(5+x) = 25 - x^2$.
\end{solution}

\begin{solution}{exo:g10:algebra:2}
$x^2 - 49 = (x+7)(x-7)$ (différence de carrés).

$x^2 + 10x + 25 = (x+5)^2$.

$7x^2 - 14x = 7x(x - 2)$ (facteur commun).

$16x^2 - 8x + 1 = (4x - 1)^2$.
\end{solution}

\begin{solution}{exo:g10:algebra:3}
$4x + 9 = x - 3$~: $3x = -12$, donc $x = -4$.

$\frac{2x-1}{3} = 5$~: $2x - 1 = 15$, donc $x = 8$.

$2(x-3) = 2x + 1$ se développe en $2x - 6 = 2x + 1$, c'est-à-dire
$-6 = 1$~: faux pour tout $x$. Aucune solution --- les deux membres
définissent des droites parallèles qui ne se rencontrent jamais.
\end{solution}

\begin{solution}{exo:g10:algebra:4}
$(x+5)(2x-8) = 0$~: $x = -5$ ou $x = 4$.

$x^2 - 16 = (x-4)(x+4) = 0$~: $x = 4$ ou $x = -4$.

$x^2 + 6x + 9 = (x+3)^2 = 0$~: l'unique solution $x = -3$.
\end{solution}

\begin{solution}{exo:g10:algebra:5}
$3x - 5 \leq 7$~: $3x \leq 12$, donc $x \leq 4$~: ensemble des solutions
$\intoc{-\infty}{4}$.

$-4x + 1 > 9$~: $-4x > 8$, diviser par $-4$ inverse l'inégalité~:
$x < -2$~: ensemble des solutions $\intoo{-\infty}{-2}$.

$2(x-1) \geq 5x + 4$~: $2x - 2 \geq 5x + 4$, donc $-3x \geq 6$, donc
$x \leq -2$~: ensemble des solutions $\intoc{-\infty}{-2}$.
\end{solution}

\begin{solution}{exo:g10:algebra:6}
Le facteur commun est $(x-1)$~:
\[
E = (x-1)\bigl[(x+3) - (2x - 5)\bigr] = (x-1)(x + 3 - 2x + 5)
= (x-1)(8 - x).
\]
Alors $E = 0$ lorsque $x = 1$ ou $x = 8$ (règle du produit nul).
\end{solution}

\begin{solution}{exo:g10:algebra:7}
$x^2 = 5x$~: $x^2 - 5x = x(x - 5) = 0$, donc $x = 0$ ou $x = 5$ (ne
jamais diviser par $x$~: la solution $0$ serait perdue).

$(2x+1)^2 = (x-4)^2$~: tout ramener à gauche et \omterm{def:g10:algebra:expand}{factoriser} la
différence de carrés~:
\[
(2x+1)^2 - (x-4)^2
= \bigl[(2x+1) + (x-4)\bigr]\bigl[(2x+1) - (x-4)\bigr]
= (3x - 3)(x + 5).
\]
Règle du produit nul~: $x = 1$ ou $x = -5$.
\end{solution}

\begin{solution}{exo:g10:algebra:8}
$(x+2)(4-x) > 0$~: le facteur $x + 2$ s'annule en $-2$ (positif après),
$4 - x$ s'annule en $4$ (positif avant). Le produit est positif
exactement lorsque les deux facteurs sont positifs, c'est-à-dire sur
$\intoo{-2}{4}$.

$\frac{2x-6}{x+1} \geq 0$~: numérateur nul en $3$, dénominateur nul en
$-1$ (interdit). \omterm{def:g10:algebra:signtable}{Tableau de signes}~: le quotient est positif lorsque les
deux parties ont le même signe, c'est-à-dire pour $x < -1$ ou $x > 3$~;
il s'annule en $x = 3$. Ensemble des solutions~:
$\intoo{-\infty}{-1} \cup \intco{3}{+\infty}$.
\end{solution}

\begin{solution}{exo:g10:algebra:9}
Après $x$ mois le premier service coûte $9x + 15$ et le second $12x$. Le
premier est moins cher lorsque
\[
9x + 15 < 12x
\ \Longleftrightarrow\
15 < 3x
\ \Longleftrightarrow\
x > 5 .
\]
À partir du $6$-ième mois, le premier service est le choix le moins cher
au total.
\end{solution}

\begin{solution}{exo:g10:algebra:10}
Valeur interdite~: $x = 1$. Pour $x \neq 1$, multiplier les deux
membres par $x - 1$~:
\[
x + 3 = 2(x - 1) = 2x - 2
\ \Longleftrightarrow\
x = 5 .
\]
Comme $5 \neq 1$, la solution est $x = 5$. Vérification~:
$\frac{5+3}{5-1} = \frac84 = 2$.
\end{solution}

\begin{solution}{exo:g10:algebra:11}
$(n+1)^2 - n^2 = n^2 + 2n + 1 - n^2 = 2n + 1$, qui est impair pour tout
entier $n$.

Deux nombres impairs consécutifs s'écrivent $2n - 1$ et $2n + 1$. Alors
\[
(2n+1)^2 - (2n-1)^2
= \bigl[(2n+1)+(2n-1)\bigr]\bigl[(2n+1)-(2n-1)\bigr]
= 4n \times 2 = 8n,
\]
un multiple de $8$.
\end{solution}
