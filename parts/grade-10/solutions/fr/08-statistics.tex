\section*{Chapitre \ref{ch:g10:stats} --- Statistique descriptive}

\begin{solution}{exo:g10:stats:1}
Somme~: $0+1+1+2+2+2+3+3+4+4+5+6+7+8+12 = 60$, donc la moyenne est
$\frac{60}{15} = 4$. La série est déjà triée et $N = 15$ est impair~: la
médiane est la $8$-ième valeur, $3$. Étendue~: $12 - 0 = 12$.
\end{solution}

\begin{solution}{exo:g10:stats:2}
Fréquences~: $\frac{7}{50} = 0.14$, $\frac{9}{50} = 0.18$,
$\frac{8}{50} = 0.16$, $\frac{10}{50} = 0.20$, $\frac{6}{50} = 0.12$,
$\frac{10}{50} = 0.20$ (somme $1$). Issue moyenne~:
\[
\bar x = \frac{7 \times 1 + 9 \times 2 + 8 \times 3 + 10 \times 4
+ 6 \times 5 + 10 \times 6}{50}
= \frac{7 + 18 + 24 + 40 + 30 + 60}{50}
= \frac{179}{50} = 3.58 .
\]
\end{solution}

\begin{solution}{exo:g10:stats:3}
$N = 12$ (pair)~: la médiane est le milieu des $6$-ième et $7$-ième
valeurs, toutes deux égales à $6$~: médiane $6$.

$Q_1$~: $\frac{12}{4} = 3$, donc $Q_1$ est la $3$-ième valeur~: $4$.

$Q_3$~: $\frac{3 \times 12}{4} = 9$, donc $Q_3$ est la $9$-ième
valeur~: $8$.

Diagramme en boîte~: boîte de $4$ à $8$ avec la barre de médiane en $6$,
moustaches de $1$ à $11$.
\end{solution}

\begin{solution}{exo:g10:stats:4}
Les $24$ notes totalisent $24 \times 11.5 = 276$. Avec la nouvelle note
le total est $276 + 19 = 295$ pour $25$ élèves~: nouvelle moyenne
$\frac{295}{25} = 11.8$.
\end{solution}

\begin{solution}{exo:g10:stats:5}
Total actuel~: $5 \times 12 = 60$. Une moyenne de $13$ sur $6$ tests
exige un total de $78$, donc la $6$-ième note doit être $78 - 60 = 18$.
Une moyenne de $14$ exigerait $84 - 60 = 24 > 20$~: non atteignable.
\end{solution}

\begin{solution}{exo:g10:stats:6}
Moyenne plus grande que $20$~: par exemple
$1,\ 2,\ 3,\ 10,\ 40,\ 50,\ 60$ --- médiane $10$ (la $4$-ième valeur
triée), moyenne $\frac{166}{7} \approx 23.7$.

Moyenne plus petite que $6$~: par exemple
$0,\ 0,\ 0,\ 10,\ 10,\ 10,\ 10$ --- médiane $10$, moyenne
$\frac{40}{7} \approx 5.7$.

La moyenne ne peut pas descendre sous $\frac{40}{7}$~: pour que la
médiane (la $4$-ième valeur triée) soit $10$, les valeurs en positions
$4$ à $7$ doivent toutes être $\geq 10$, donc le total est au moins
$40$ et la moyenne au moins $\frac{40}{7}$ --- le second exemple est
extrémal.

Ces exemples montrent que la moyenne et la médiane sont largement
indépendantes~: connaître l'une dit peu sur l'autre, d'où l'intérêt d'un
bon résumé qui rapporte les deux.
\end{solution}

\begin{solution}{exo:g10:stats:7}
\emph{1.} Moyenne~:
$\frac{9 \times 1800 + 12000}{10} = \frac{16200 + 12000}{10} = 2820$.
Médiane~: triés, les $10$ salaires sont neuf valeurs $1800$ puis
$12000$~; la médiane est le milieu des $5$-ième et $6$-ième valeurs,
toutes deux $1800$~: médiane $1800$.

\emph{2.} La médiane ($1800$) décrit le salaire typique~: c'est ce que
gagnent réellement $9$ employés sur $10$. La moyenne ($2820$) est tirée
vers le haut par l'unique haut salaire et ne correspond à la fiche de
paie de personne.
\end{solution}

\begin{solution}{exo:g10:stats:8}
Taille totale~: $12 \times 168 + 18 \times 163 = 2016 + 2934 = 4950$~cm
pour $30$ élèves, donc la moyenne est $\frac{4950}{30} = 165$~cm. Elle
est plus proche de la moyenne des filles car elles sont plus nombreuses
(moyenne pondérée, poids $12$ et $18$).
\end{solution}

\begin{solution}{exo:g10:stats:9}
\emph{1.} La nouvelle moyenne est
\[
\bar y = \frac{y_1 + \dots + y_N}{N}
= \frac{(a x_1 + b) + \dots + (a x_N + b)}{N}
= \frac{a(x_1 + \dots + x_N) + Nb}{N}
= a \bar x + b .
\]

\emph{2.} Avec $a = 1.8$ et $b = 32$~: moyenne
$1.8 \times 20 + 32 = 68$ degrés Fahrenheit.
\end{solution}
