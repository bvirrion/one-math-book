\section*{Chapitre \ref{ch:g10:reffunc} --- Fonctions de référence}

\begin{solution}{exo:g10:reffunc:1}
$f$~: coefficient $3$, \omterm{def:g10:reffunc:affine}{ordonnée à l'origine} $-2$, \omterm{def:g10:functions:variations}{croissante}. \quad
$g$~: coefficient $-\frac12$, \omterm{def:g10:reffunc:affine}{ordonnée à l'origine} $4$, \omterm{def:g10:functions:variations}{décroissante}.
\quad $h$~: coefficient $0$, \omterm{def:g10:reffunc:affine}{ordonnée à l'origine} $7$, constante.
\quad $k$~: coefficient $-1$, \omterm{def:g10:reffunc:affine}{ordonnée à l'origine} $0$, \omterm{def:g10:functions:variations}{décroissante}
(et linéaire).
\end{solution}

\begin{solution}{exo:g10:reffunc:2}
Par $(2,1)$ et $(5,10)$~: coefficient $m = \frac{10 - 1}{5 - 2} = 3$~;
puis $1 = 3 \times 2 + p$ donne $p = -5$~: $f(x) = 3x - 5$.

Par $(-1,4)$ et $(3,-4)$~: coefficient
$m = \frac{-4 - 4}{3 - (-1)} = \frac{-8}{4} = -2$~; puis
$4 = -2 \times (-1) + p$ donne $p = 2$~: $f(x) = -2x + 2$.
\end{solution}

\begin{solution}{exo:g10:reffunc:3}
$(3.1)^2 < (3.2)^2$~: le carré croît sur les positifs.

$(-2.7)^2 < (-2.8)^2$~: le carré décroît sur les négatifs et
$-2.8 < -2.7$.

$\frac{1}{5.1} > \frac{1}{5.2}$~: l'inverse décroît sur les positifs.

$\sqrt{17} > \sqrt{15}$~: la racine carrée croît.
\end{solution}

\begin{solution}{exo:g10:reffunc:4}
$x^2 = 16$~: deux solutions, $x = 4$ et $x = -4$ (la droite horizontale
$y = 16$ coupe la parabole deux fois).

$x^2 \leq 16$~: la parabole est sous la droite entre les deux points
d'\omterm{def:g10:numbers:interunion}{intersection}~: $x \in \intcc{-4}{4}$.

$x^2 > 9$~: la parabole est strictement au-dessus de $y = 9$ à
l'extérieur de $\intcc{-3}{3}$~:
$x \in \intoo{-\infty}{-3} \cup \intoo{3}{+\infty}$.
\end{solution}

\begin{solution}{exo:g10:reffunc:5}
Pour $0 < x < 1$~: multiplier $x < 1$ plusieurs fois par $x$ donne
$x^3 < x^2 < x$, et le \cref{prop:g10:reffunc:compare} donne
$x < \sqrt x$, donc
\[
x^3 < x^2 < x < \sqrt x .
\]
Vérification avec $x = 0.25$~: $x^3 = 0.015625$, $x^2 = 0.0625$,
$x = 0.25$, $\sqrt x = 0.5$.
\end{solution}

\begin{solution}{exo:g10:reffunc:6}
$\frac1x = 2$ a l'unique solution $x = \frac12$ (l'hyperbole rencontre
la droite horizontale $y = 2$ une fois, sur la branche positive).

Pour $x > 0$~: $\frac1x < 2$. Multiplier par $x > 0$~: $1 < 2x$, donc
$x > \frac12$~: ensemble des solutions $\intoo{\frac12}{+\infty}$.

Si $x < 0$ est autorisé~: tout $x$ négatif satisfait
$\frac1x < 0 < 2$, donc l'ensemble complet des solutions est
$\intoo{-\infty}{0} \cup \intoo{\frac12}{+\infty}$.
\end{solution}

\begin{solution}{exo:g10:reffunc:7}
(a)~Sur $\intcc{2}{5}$, toutes les entrées sont positives et le carré
croît~: $4 \leq x^2 \leq 25$.

(b)~Sur $\intcc{-3}{-1}$, le carré décroît~: le plus grand carré vient
de $-3$~: $1 \leq x^2 \leq 9$.

(c)~Sur $\intcc{-2}{3}$, l'\omterm{def:g10:numbers:interval}{intervalle} contient $0$, où le carré atteint
son \omterm{def:g10:functions:extrema}{minimum} $0$~; le \omterm{def:g10:functions:extrema}{maximum} est le plus grand de $(-2)^2 = 4$ et
$3^2 = 9$. Donc $0 \leq x^2 \leq 9$.
\end{solution}

\begin{solution}{exo:g10:reffunc:8}
Première compagnie~: $f(x) = 1.5x + 4$~; seconde~: $g(x) = 2x$. La
première est moins chère lorsque
\[
1.5x + 4 < 2x
\ \Longleftrightarrow\
4 < 0.5x
\ \Longleftrightarrow\
x > 8 .
\]
Au-delà de $8$~km, la première compagnie est moins chère~; à exactement
$8$~km les deux facturent $16$.
\end{solution}

\begin{solution}{exo:g10:reffunc:9}
Les deux membres sont positifs ou nuls, donc l'inégalité équivaut à
celle entre leurs carrés~:
\[
\left(\sqrt{a+b}\right)^2 = a + b
\qquad\text{et}\qquad
\left(\sqrt a + \sqrt b\right)^2 = a + 2\sqrt a \sqrt b + b .
\]
Comme $2\sqrt a\sqrt b \geq 0$, le second carré est au moins égal au
premier, ce qui prouve $\sqrt{a+b} \leq \sqrt a + \sqrt b$. L'égalité a
lieu exactement lorsque $\sqrt a \sqrt b = 0$, c'est-à-dire lorsque
$a = 0$ ou $b = 0$.
\end{solution}

\begin{solution}{exo:g10:reffunc:10}
\emph{1.} Mettre le second membre sur un dénominateur commun~:
\[
2 + \frac{3}{x-1} = \frac{2(x-1) + 3}{x - 1} = \frac{2x + 1}{x - 1}
= f(x).
\]

\emph{2.} Sur $\intoo{1}{+\infty}$, lorsque $x$ croît, $x - 1$ croît et
reste positif, donc $\frac{3}{x-1}$ décroît (\omterm{def:g10:functions:function}{fonction} inverse, multipliée
par la constante positive $3$), donc $f(x) = 2 + \frac{3}{x-1}$
décroît~: $f$ est strictement \omterm{def:g10:functions:variations}{décroissante} sur $\intoo{1}{+\infty}$.
\end{solution}
