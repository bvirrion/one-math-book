\section*{Chapitre \ref{ch:g10:proba} --- Probabilités et échantillonnage}

\begin{solution}{exo:g10:proba:1}
Chacune des $6$ issues a pour \omterm{def:g10:proba:distribution}{probabilité} $\frac16$.
«~Obtenir un $6$~»~: $\frac16$.
«~Nombre impair~» $= \{1, 3, 5\}$~: $\frac36 = \frac12$.
«~Au plus $4$~» $= \{1,2,3,4\}$~: $\frac46 = \frac23$.
«~Obtenir un $7$~»~: \omterm{def:g10:proba:events}{événement} impossible, \omterm{def:g10:proba:distribution}{probabilité} $0$.
\end{solution}

\begin{solution}{exo:g10:proba:2}
Il y a $10$ billes équiprobables.
$\P(\text{verte}) = \frac{5}{10} = \frac12$.
$\P(\text{non noire}) = 1 - \frac{2}{10} = \frac{8}{10} = \frac45$.
Verte et jaune sont \omterm{def:g10:proba:operations}{incompatibles}~:
$\P(\text{verte ou jaune}) = \frac{5}{10} + \frac{3}{10} = \frac{8}{10}
= \frac45$.
\end{solution}

\begin{solution}{exo:g10:proba:3}
Soit $S$~: «~étudie l'espagnol~», $G$~: «~étudie l'allemand~». Alors
$\P(S) = \frac{18}{30}$, $\P(G) = \frac{10}{30}$,
$\P(S \cap G) = \frac{4}{30}$. Règle d'addition~:
\[
\P(S \cup G) = \frac{18}{30} + \frac{10}{30} - \frac{4}{30}
= \frac{24}{30} = \frac45 .
\]
\omterm{def:g10:proba:operations}{Complémentaire}~: $\P(\text{ni l'un ni l'autre}) = 1 - \frac45 =
\frac15$.
\end{solution}

\begin{solution}{exo:g10:proba:4}
L'\omterm{met:g10:proba:tree}{arbre} a quatre chemins FF, FP, PF, PP, chacun de \omterm{def:g10:proba:distribution}{probabilité}
$\frac12 \times \frac12 = \frac14$.
Deux faces~: $\frac14$. Exactement une face (FP ou PF)~:
$\frac24 = \frac12$.
Au moins une face~: $1 - \P(\text{PP}) = 1 - \frac14 = \frac34$.
\end{solution}

\begin{solution}{exo:g10:proba:5}
Premier tirage~: rouge avec \omterm{def:g10:proba:distribution}{probabilité} $\frac{4}{10}$, blanche
$\frac{6}{10}$.
Second tirage \emph{sans} remise~: après une rouge, $3$ rouges et $6$
blanches restent parmi $9$~; après une blanche, $4$ rouges et $5$
blanches parmi $9$.

Deux rouges~: $\frac{4}{10} \times \frac39 = \frac{12}{90} =
\frac{2}{15}$.

Couleurs différentes~: chemins RB et BR~:
\[
\frac{4}{10} \times \frac69 + \frac{6}{10} \times \frac49
= \frac{24}{90} + \frac{24}{90} = \frac{48}{90} = \frac{8}{15}.
\]
\end{solution}

\begin{solution}{exo:g10:proba:6}
\emph{1.} Chaque dé montre indépendamment chaque face avec \omterm{def:g10:proba:distribution}{probabilité}
$\frac16$, donc les $36$ couples ordonnés sont équiprobables~; la somme
est déterminée par le couple.

\emph{2.} Somme $7$~: les couples $(1,6), (2,5), (3,4), (4,3), (5,2),
(6,1)$, donc $\P = \frac{6}{36} = \frac16$. Somme $12$~: seulement
$(6,6)$, $\P = \frac{1}{36}$. Somme au plus $4$~: couples sommant à
$2$, $3$ ou $4$~: $(1,1)$~; $(1,2), (2,1)$~; $(1,3), (2,2), (3,1)$ ---
six couples, donc $\P = \frac{6}{36} = \frac16$.

\emph{3.} Le nombre de couples réalisant chaque somme croît de $1$
(somme $2$) à $6$ (somme $7$) puis décroît~: la somme la plus probable
est $7$.
\end{solution}

\begin{solution}{exo:g10:proba:7}
Chaque question est correctement répondue avec \omterm{def:g10:proba:distribution}{probabilité} $\frac14$,
indépendamment. \omterm{met:g10:proba:tree}{Arbre} à deux niveaux~:

Les deux correctes~: $\frac14 \times \frac14 = \frac{1}{16}$.

Exactement une correcte~: chemins correct--faux et faux--correct~:
$\frac14 \times \frac34 + \frac34 \times \frac14 = \frac{6}{16}
= \frac38$.

Aucune correcte~: $\frac34 \times \frac34 = \frac{9}{16}$.

Vérification~: $\frac{1}{16} + \frac{6}{16} + \frac{9}{16} = 1$.
\end{solution}

\begin{solution}{exo:g10:proba:8}
\emph{1.} \omterm{def:g10:stats:series}{Fréquence} observée~: $\frac{396}{900} = 0.44$. Avec
$n = 900$~: $\frac{1}{\sqrt{900}} = \frac{1}{30} \approx 0.033$, donc
l'\omterm{def:g10:numbers:interval}{intervalle} de fluctuation est environ $\intcc{0.467}{0.533}$.

\emph{2.} La \omterm{def:g10:stats:series}{fréquence} observée $0.44$ est hors de l'\omterm{def:g10:numbers:interval}{intervalle}~: des
\omterm{def:g10:proba:sample}{échantillons} de $900$ électeurs d'une population à $50\,\%$ d'approbation
s'écartent rarement autant. Le sondage jette un doute sérieux sur les
$50\,\%$ affirmés.
\end{solution}

\begin{solution}{exo:g10:proba:9}
\emph{1.} $\P(\text{gagner } 6) = \frac16$ (lancer un $6$),
$\P(\text{gagner } 3) = \frac16$ (lancer un $5$),
$\P(\text{gagner } 0) = \frac46 = \frac23$.

\emph{2.} Sur $600$ parties, on attend environ $100$ six, $100$ cinq,
$400$ autres lancers. Gains~: environ
$100 \times 6 + 100 \times 3 = 900$. Coût~: $600 \times 2 = 1200$.
Résultat net attendu~: $900 - 1200 = -300$, c'est-à-dire une perte
d'environ $0.5$ par partie en \omterm{def:g10:stats:mean}{moyenne} --- le jeu ne vaut pas la peine
d'être joué.
\end{solution}
