\section*{Chapitre \ref{ch:g10:vectors} --- Vecteurs}

\begin{solution}{exo:g10:vectors:1}
$\vect{AB} + \vect{BD} = \vect{AD}$ (Chasles).

$\vect{KL} + \vect{LM} + \vect{MK} = \vect{KK} = \vec 0$.

$\vect{AC} - \vect{BC} = \vect{AC} + \vect{CB} = \vect{AB}$.

$\vect{AB} + \vect{CA} = \vect{CA} + \vect{AB} = \vect{CB}$.
\end{solution}

\begin{solution}{exo:g10:vectors:2}
$\vect{AB}\,(5-2,\ 3-1) = (3, 2)$~; \quad
$\vect{BC}\,(-1-5,\ 4-3) = (-6, 1)$~; \quad
$\vect{AC}\,(-1-2,\ 4-1) = (-3, 3)$.
Vérification~: $(3, 2) + (-6, 1) = (-3, 3)$, coordonnée par
coordonnée.
\end{solution}

\begin{solution}{exo:g10:vectors:3}
$\vec u + \vec v = (2 + (-1),\ -3 + 4) = (1, 1)$.

$3\vec u = (6, -9)$.

$2\vec u - \vec v = (4 - (-1),\ -6 - 4) = (5, -10)$.

$\norm{\vec u} = \sqrt{2^2 + (-3)^2} = \sqrt{13}$.
\end{solution}

\begin{solution}{exo:g10:vectors:4}
(a)~$4 \times 9 - 6 \times 6 = 36 - 36 = 0$~: colinéaires
($\vec v = \frac32 \vec u$).

(b)~$2 \times 10 - (-5) \times (-4) = 20 - 20 = 0$~: colinéaires
($\vec v = -2\vec u$).

(c)~$3 \times 3 - 1 \times 1 = 8 \neq 0$~: non colinéaires.
\end{solution}

\begin{solution}{exo:g10:vectors:5}
$\vect{AB}\,(4, -2)$, et $\vect{DC}\,(6 - x_D,\ 3 - y_D)$. La condition
$\vect{AB} = \vect{DC}$ donne $6 - x_D = 4$ et $3 - y_D = -2$, donc
$D(2, 5)$.
\end{solution}

\begin{solution}{exo:g10:vectors:6}
\emph{1.} $\vect{AB}\,(3, 2)$ et $\vect{AC}\,(6, -2)$, donc
$\vect{AB} + \vect{AC} = (9, 0)$ et
$M = (-2 + 9,\ 1 + 0) = (7, 1)$.

\emph{2.} Par la règle du parallélogramme,
$\vect{AM} = \vect{AB} + \vect{AC}$ signifie que $ABMC$ est un
parallélogramme ($M$ est le quatrième sommet opposé à $A$). On peut
vérifier~: $\vect{AB}\,(3,2)$ et $\vect{CM}\,(7-4,\ 1-(-1)) = (3, 2)$
sont égaux.
\end{solution}

\begin{solution}{exo:g10:vectors:7}
Premier triplet~: $\vect{AB}\,(3, 2)$, $\vect{AC}\,(9, 6)$~;
$3 \times 6 - 2 \times 9 = 0$~: alignés ($\vect{AC} = 3\vect{AB}$).

Second triplet~: $\vect{AB}\,(2, -1)$, $\vect{AC}\,(8, -4)$~;
$2 \times (-4) - (-1) \times 8 = -8 + 8 = 0$~: alignés aussi
($\vect{AC} = 4\vect{AB}$).
\end{solution}

\begin{solution}{exo:g10:vectors:8}
$\vect{AB}\,(4, 2)$ et $\vect{DC}\,(6-2,\ 1-(-1)) = (4, 2)$~: les deux
vecteurs sont \emph{égaux}, donc $ABCD$ est un parallélogramme, et en
particulier $(AB) \parallel (CD)$. La propriété de parallélogramme est
la plus forte~: elle implique le parallélisme (colinéarité des
vecteurs), pas l'inverse.
\end{solution}

\begin{solution}{exo:g10:vectors:9}
\emph{1.} $\vect{AB}\,(6, 3)$, donc
$\vect{AI} = \frac23\vect{AB} = (4, 2)$ et $I(4, 2)$.

\emph{2.} $\vect{CB}\,(6-2,\ 3-5) = (4, -2)$, donc
$\vect{CJ} = \frac23\vect{CB} = \left(\frac83, -\frac43\right)$ et
$J\left(2 + \frac83,\ 5 - \frac43\right) = \left(\frac{14}{3},
\frac{11}{3}\right)$.

\emph{3.} $\vect{IJ}\,\left(\frac{14}{3} - 4,\ \frac{11}{3} - 2\right) =
\left(\frac23, \frac53\right)$ et $\vect{AC}\,(2, 5)$. Critère de
colinéarité~:
\[
\frac23 \times 5 - \frac53 \times 2 = \frac{10}{3} - \frac{10}{3} = 0 :
\]
les vecteurs sont colinéaires, donc $(IJ)$ est parallèle à $(AC)$.
(Les deux points $I$ et $J$ sont situés aux deux tiers du côté vers
$B$~: le petit triangle $IBJ$ est une réduction de $ABC$.)
\end{solution}

\begin{solution}{exo:g10:vectors:10}
Les milieux sont
\[
I\left(\tfrac{x_A + x_B}{2}, \tfrac{y_A + y_B}{2}\right), \quad
J\left(\tfrac{x_B + x_C}{2}, \tfrac{y_B + y_C}{2}\right), \quad
K\left(\tfrac{x_C + x_D}{2}, \tfrac{y_C + y_D}{2}\right), \quad
L\left(\tfrac{x_D + x_A}{2}, \tfrac{y_D + y_A}{2}\right).
\]
Premières coordonnées des deux vecteurs~:
\[
x_J - x_I = \frac{x_B + x_C}{2} - \frac{x_A + x_B}{2}
= \frac{x_C - x_A}{2},
\qquad
x_K - x_L = \frac{x_C + x_D}{2} - \frac{x_D + x_A}{2}
= \frac{x_C - x_A}{2},
\]
et le même calcul fonctionne pour les secondes coordonnées. Donc
$\vect{IJ} = \vect{LK}$ (tous deux égaux à $\frac12\vect{AC}$), et
$IJKL$ est un parallélogramme, quelle que soit l'allure du
quadrilatère $ABCD$.
\end{solution}
