\section*{Capítulo \ref{ch:g10:algebra} --- Álgebra: Equações e Desigualdades}

\begin{solution}{exo:g10:algebra:1}
$3(2x-5) - 2(x-7) = 6x - 15 - 2x + 14 = 4x - 1$.

$(x+4)(2x-1) = 2x^2 - x + 8x - 4 = 2x^2 + 7x - 4$.

$(3x-2)^2 = 9x^2 - 12x + 4$.

$(5-x)(5+x) = 25 - x^2$.
\end{solution}

\begin{solution}{exo:g10:algebra:2}
$ x^2 - 49 = (x+7)(x-7)$ (diferença de quadrados).

$ x^2 + 10x + 25 = (x+5)^2$.

$7x^2 - 14x = 7x(x - 2)$ (fator comum).

$16x^2 - 8x + 1 = (4x - 1)^2$.
\end{solution}

\begin{solution}{exo:g10:algebra:3}
$4x + 9 = x - 3$: $3x = -12$, então $ x = -4$.

$\frac{2x-1}{3} = 5$: $2x - 1 = 15$, então $ x = 8$.

$2(x-3) = 2x + 1$ expande para $2x - 6 = 2x + 1$, ou seja\ $-6 = 1$: falso para
cada $ x $. Nenhuma solução --- os dois lados definem linhas paralelas que nunca
conhecer.
\end{solution}

\begin{solution}{exo:g10:algebra:4}
$(x+5)(2x-8) = 0$: $ x = -5$ ou $ x = 4$.

$ x^2 - 16 = (x-4)(x+4) = 0$: $ x = 4$ ou $ x = -4$.

$ x^2 + 6x + 9 = (x+3)^2 = 0$: a única solução $ x = -3$.
\end{solution}

\begin{solution}{exo:g10:algebra:5}
$3x - 5 \leq 7$: $3x \leq 12$, então $ x \leq 4$: conjunto de soluções
$\intoc{-\infty}{4}$.

$-4x + 1 > 9$: $-4x > 8$, dividindo por $-4$ inverte a desigualdade:
$ x < -2$: conjunto de soluções $\intoo{-\infty}{-2}$.

$2(x-1) \geq 5x + 4$: $2x - 2 \geq 5x + 4$, então $-3x \geq 6$, então
$ x \leq -2$: conjunto de soluções $\intoc{-\infty}{-2}$.
\end{solution}

\begin{solution}{exo:g10:algebra:6}
O fator comum é $(x-1)$:
\[
E = (x-1)\bigl[(x+3) - (2x - 5)\bigr] = (x-1)(x + 3 - 2x + 5)
= (x-1)(8 - x).
\]
Então $ E = 0$ quando $ x = 1$ ou $ x = 8$ (regra do produto zero).
\end{solution}

\begin{solution}{exo:g10:algebra:7}
$ x^2 = 5x $: $ x^2 - 5x = x(x - 5) = 0$, então $ x = 0$ ou $ x = 5$ (nunca
dividir por $ x $: a solução $0$ estaria perdido).

$(2x+1)^2 = (x-4)^2$: traga para a esquerda e fatore a diferença de
quadrados:
\[
(2x+1)^2 - (x-4)^2
= \bigl[(2x+1) + (x-4)\bigr]\bigl[(2x+1) - (x-4)\bigr]
= (3x - 3)(x + 5).
\]
Regra de produto zero: $ x = 1$ ou $ x = -5$.
\end{solution}

\begin{solution}{exo:g10:algebra:8}
$(x+2)(4-x) > 0$: o fator $ x + 2$ desaparece em $-2$ (positivo depois),
$4 - x $ desaparece em $4$ (positivo antes). O produto é positivo
exatamente quando ambos os fatores são positivos, ou seja,\ em $\intoo{-2}{4}$.

$\frac{2x-6}{x+1} \geq 0$: numerador zero em $3$, denominador zero em
$-1$ (proibido). \omterm{def:g10:algebra:signtable}{Mesa de sinalização}: o quociente é positivo quando ambas as partes
têm o mesmo sinal, ou seja,\ para $ x < -1$ ou $ x > 3$; ele desaparece em $ x = 3$.
Conjunto de soluções: $\intoo{-\infty}{-1} \cup \intco{3}{+\infty}$.
\end{solution}

\begin{solution}{exo:g10:algebra:9}
Depois $ x $ meses o primeiro serviço custa $9x + 15$ e o segundo $12x $.
O primeiro é mais barato quando
\[
9x + 15 < 12x
\ \Longleftrightarrow\
15 < 3x
\ \Longleftrightarrow\
x > 5 .
\]
Do $6$ Após o mês, o primeiro serviço é a escolha mais barata em geral.
\end{solution}

\begin{solution}{exo:g10:algebra:10}
Valor proibido: $ x = 1$. Para $ x \neq 1$, multiplique ambos os lados por $ x - 1$:
\[
x + 3 = 2(x - 1) = 2x - 2
\ \Longleftrightarrow\
x = 5 .
\]
Desde $5 \neq 1$, a solução é $ x = 5$. Verificar:
$\frac{5+3}{5-1} = \frac84 = 2$.
\end{solution}

\begin{solution}{exo:g10:algebra:11}
$(n+1)^2 - n^2 = n^2 + 2n + 1 - n^2 = 2n + 1$, o que é estranho para todo
inteiro $ n $.

Dois números ímpares consecutivos podem ser escritos $2n - 1$ e $2n + 1$. Então
\[
(2n+1)^2 - (2n-1)^2
= \bigl[(2n+1)+(2n-1)\bigr]\bigl[(2n+1)-(2n-1)\bigr]
= 4n \times 2 = 8n,
\]
um múltiplo de $8$.
\end{solution}

\begin{solution}{pb:g10:algebra:1}
\textbf{1.} $ x^2 - 10x + 25 = (x - 5)^2$;
$9x^2 - 49 = (3x - 7)(3x + 7)$;
$ x^2 + 6x + 9 - 4 = (x + 3)^2 - 2^2 = (x + 1)(x + 5)$.

\textbf{2.} $ x = -1$ ou $ x = -5$; \quad $ x = \frac73$ ou
$ x = -\frac73$.

\textbf{3.} $(5 - t)(5 + t) = 25 - t^2 = 21$ dá $ t^2 = 4$,
$ t = 2$: os números são $3$ e $7$ (soma $10$, produto $21$:
verificar).

\textbf{4.} $(7 - t)(7 + t) = 49 - t^2 = 33$: $ t = 4$, números
$3$ e $11$.

\textbf{5.} $\left(\frac s2 - t\right)\left(\frac s2 + t\right)
= \left(\frac s2\right)^2 - t^2 = p $ forças
$ t^2 = \left(\frac s2\right)^2 - p $. Existe uma solução exatamente
quando isso é $\geq 0$, ou seja,\ quando
$ p \leq \left(\frac s2\right)^2$ --- o produto de dois números
de determinada soma nunca pode vencer o quadrado da meia soma
(problema da cerca da rainha Dido, conhecido no volume do Ensino Médio).

\textbf{6.} $(x - a)(x - b) = x^2 - (a + b)x + ab $: o
equação $ x^2 - sx + p = 0$ diz precisamente ``$ x $ é um dos dois
números com soma $ s $ e produto $ p $''.

\textbf{7.} $(x-5)^2 - 4 = x^2 - 10x + 25 - 4 = x^2 - 10x +
21$. Diferença de quadrados:
$(x - 5 - 2)(x - 5 + 2) = (x - 7)(x - 3) = 0$: soluções $3$
e $7$ --- os números do escriba da pergunta 3.

\textbf{8.} $ x^2 + 6x - 7 = (x + 3)^2 - 16 = 0$ dá
$(x + 3)^2 = 16$, $ x + 3 = \pm 4$: $ x = 1$ ou $ x = -7$.

\textbf{9.} $ x^2 + 4x + 5 = (x + 2)^2 + 1 \geq 1 > 0$ para cada
$ x $: nenhuma solução real. No formulário $(x + h)^2 + k $: soluções
existe exatamente quando $ k \leq 0$ (então $(x+h)^2 = -k $ tem os dois
raízes $-h \pm \sqrt{-k}$); quando $ k > 0$ a expressão permanece
positivo.

\textbf{10.} O quadrado do lado $ x $ (área $ x^2$) com dois
$3 \times x $ retângulos ($6x $ ao todo) forma um L de área
$ x^2 + 6x = 7$; o desaparecido $3 \times 3$ canto (área $9$)
completa um grande quadrado de lado $ x + 3$. Adicionando $9$ para ambos
lados: $(x + 3)^2 = 16$, então $ x + 3 = 4$ (comprimentos são positivos)
e $ x = 1$. A álgebra e a imagem são o mesmo ato.

\textbf{11.} $(x-3)(x+2)$: negativo entre as raízes. Solução
de $< 0$: $ x \in \intoo{-2}{3}$.

\textbf{12.} $ x^2 - 10x + 21 \leq 0$, ou seja\
$(x - 3)(x - 7) \leq 0$: entre as raízes, pontas incluídas:
$ x \in \intcc{3}{7}$.

\textbf{13.} Zero em $ x = 1$, proibido em $ x = -2$; o
quociente é positivo fora das raízes:
$ x \in \intoo{-\infty}{-2} \cup \intco{1}{+\infty}$.

\textbf{14.} $25 - (x - 5)^2 = 25 - x^2 + 10x - 25 =
x(10 - x)$: identidade verificada. Desde $(x-5)^2 \geq 0$ sempre,
$ A(x) \leq 25$, com igualdade exatamente quando $ x = 5$: o
$5 \times 5$ caneta quadrada, máximo $25$ eu $^2$ --- a nota 6
tabela, agora uma prova de duas linhas.

\textbf{15.} Para $ x > 0$, multiplique a afirmação por $ x $:
$ x^2 + 1 \geq 2x $, ou seja\ $ x^2 - 2x + 1 = (x - 1)^2 \geq 0$ ---
verdade, com igualdade apenas em $ x = 1$. Dividindo por $ x > 0$
preserva tudo (\cref{prop:g10:algebra:ineqrules}).

\textbf{16.} Multiplicação cruzada $\frac1x = \frac{x}{1 - x}$
(todos os comprimentos positivos): $ x^2 = 1 - x $, ou seja\
$ x^2 + x - 1 = 0$.

\textbf{17.} $ x^2 + x - 1 = \left(x + \frac12\right)^2 -
\frac54 = 0$ dá $ x + \frac12 = \pm\frac{\sqrt5}{2}$. O
raiz positiva: $ x = \frac{\sqrt5 - 1}{2} \approx 0.618$.

\textbf{18.} $\varphi = \frac{2}{\sqrt5 - 1} =
\frac{2(\sqrt5 + 1)}{(\sqrt5 - 1)(\sqrt5 + 1)}
= \frac{2(\sqrt5 + 1)}{5 - 1} = \frac{\sqrt5 + 1}{2}
\approx 1.618$.

\textbf{19.} $\varphi^2 = \frac{(\sqrt5 + 1)^2}{4} =
\frac{6 + 2\sqrt5}{4} = \frac{3 + \sqrt5}{2}$, e
$\varphi + 1 = \frac{\sqrt5 + 3}{2}$: igual. E
$\varphi - 1 = \frac{\sqrt5 - 1}{2} = x = \frac1\varphi $ (que
é como $\varphi $ foi definido). A proporção áurea é o número
cujo quadrado soma um e cujo inverso subtrai um.

\textbf{20.} $\sqrt5$ é \omterm{ex:g10:numbers:classify}{irracional} (o problema da irracionalidade do fim de semana do volume do Ensino Médio: $5$
não é um quadrado perfeito) e adicionando $1$ então reduzir pela metade as conservas
irracionalidade (operações racionais, \cref{pb:g10:numbers:1}).
Proporções: $\frac53 \approx 1.667$, $\frac85 = 1.600$,
$\frac{13}{8} = 1.625$, $\frac{21}{13} \approx 1.615$:
alternadamente acima e abaixo, fechando
$\varphi = 1.618\ldots$ --- os números de contagem rítmica de
o problema de contagem rítmica do fim de semana do volume do Ensino Médio persegue secretamente a seção áurea; o
capítulos sobre sequências irão pegá-los em flagrante.
\end{solution}
