\begin{solution}{exo:g10:functions:2}
$ f $: sem restrição, domínio $\R $.

$ g $: requer $ x + 4 \neq 0$, domínio todos os reais, exceto $-4$.

$ h $: requer $2x - 6 \geq 0$, ou seja\ $ x \geq 3$: domínio
$\intco{3}{+\infty}$.

$ k $: $ x^2 + 1 \geq 1 > 0$ nunca desaparece, domínio $\R $.
\end{solution}

\begin{solution}{exo:g10:functions:3}
\emph{Pré-imagens de $0$:} $ x^2 - 2x = x(x-2) = 0$, então $0$ e $2$.

\emph{Pré-imagens de $3$:} $ x^2 - 2x = 3$, ou seja\ $ x^2 - 2x - 3 = 0$, ou seja\
$(x-3)(x+1) = 0$ (verifique por expandindo), então $3$ e $-1$.

\emph{Pré-imagens de $-1$:} $ x^2 - 2x + 1 = (x - 1)^2 = 0$, então o único
\omterm{def:g10:functions:function}{pré-imagem} $1$.
\end{solution}

\begin{solution}{exo:g10:functions:4}

$ f(2) = 4 - 4 + 1 = 1$: sim, $ B(2, 1)$ está no \omterm{def:g10:functions:graph}{gráfico} também.

\begin{solution}{exo:g10:functions:5}
\emph{1.} O maior valor da tabela é $4 = g(3)$ (\omterm{def:g10:functions:extrema}{máximo}), o
o menor é $-2 = g(0)$ (\omterm{def:g10:functions:extrema}{mínimo}).

\emph{2.} Sobre $\intcc{-3}{0}$ o \omterm{def:g10:functions:function}{função} diminui, e
$-1 < -0.5$, então $ g(-1) \geq g(-0.5)$.

\emph{3.} Em cada \omterm{def:g10:numbers:interval}{intervalo} da monotonicidade, $ g $ pega cada valor no \omterm{def:g10:functions:extrema}{máximo}
uma vez, então $ g(x) = 0$ tem no \omterm{def:g10:functions:extrema}{máximo} uma solução por \omterm{def:g10:numbers:interval}{intervalo}: no \omterm{def:g10:functions:extrema}{máximo} $3$
soluções no total. (Aqui $0$ fica entre $-2$ e $1$, entre $-2$ e
$4$, e entre $0$ e $4$, então há exatamente três.)
\end{solution}

\begin{solution}{exo:g10:functions:6}
Deixar $ u < v $. Então
\[
f(v) - f(u) = (-2v + 7) - (-2u + 7) = -2(v - u) < 0,
\]
desde $ v - u > 0$. Então $ f(v) < f(u)$: $ f $ é estritamente diminuindo sobre $\R $.
\end{solution}

\begin{solution}{exo:g10:functions:7}
Complete o quadrado: $(x+3)^2 = x^2 + 6x + 9$, então
\[
f(x) = x^2 + 6x + 11 = (x + 3)^2 + 2 .
\]
Para cada $ x $, $(x+3)^2 \geq 0$, então $ f(x) \geq 2$; e
$ f(-3) = 0 + 2 = 2$. O \omterm{def:g10:functions:extrema}{mínimo} de $ f $ é $2$, alcançado em $ x = -3$.
\end{solution}

\begin{solution}{exo:g10:functions:8}
\emph{1.} Comprimento $+$ largura $= 20$, então o comprimento é $20 - x $ e
\[
A(x) = x(20 - x).
\]
Ambas as dimensões devem ser positivas: $0 < x < 20$.

\emph{2.} $ A(4) = 64$, $ A(8) = 96$, $ A(10) = 100$, $ A(12) = 96$,
$ A(16) = 64$. Os valores sobem até $ x = 10$ então caia simetricamente:
a área parece maior para $ x = 10$, ou seja,\ para um \emph{quadrado} jardim.
(\cref{ch:g11:quad} prova esta conjectura.)
\end{solution}

\begin{solution}{exo:g10:functions:9}
\emph{1.} Para cada $ x $: $ x^2 + 1 \geq 1 > 0$, então $ f(x) = \frac{1}{x^2+1}$
é positivo, e como $ x^2 + 1 \geq 1$, tomando inversos (ambos os lados
positivo) dá $ f(x) \leq 1$.

\emph{2.} \omterm{def:g10:functions:extrema}{Máximo}: $ f(0) = 1$ e $ f(x) \leq 1$ para todos $ x $, então $1$ é o
\omterm{def:g10:functions:extrema}{máximo}, alcançado em $0$. \omterm{def:g10:functions:extrema}{Mínimo}: $ f(x) > 0$ sempre, mas nenhum valor de $ x $
alcança $0$ (um quociente de números diferentes de zero é diferente de zero), e $ f $ leva
valores tão próximos $0$ como gostamos para grandes $ x $; então $ f $ não tem \omterm{def:g10:functions:extrema}{mínimo}.
\end{solution}

\begin{solution}{exo:g10:functions:10}
\emph{1.} $ f(v) - f(u) = v^2 - u^2 = (v-u)(v+u)$. Se $0 \leq u < v $, então
$ v - u > 0$ e $ v + u > 0$ (soma de um número não negativo e um número positivo),
então $ f(v) - f(u) > 0$: $ f $ é estritamente \omterm{def:g10:functions:variations}{aumentando} sobre $\intco{0}{+\infty}$.

\emph{2.} Se $ u < v \leq 0$, então $ v - u > 0$ ainda, mas agora
$ v + u < 0$ (soma de um número não positivo e um número negativo), então
$ f(v) - f(u) < 0$: $ f $ é estritamente diminuindo sobre $\intoc{-\infty}{0}$.
\end{solution}

\begin{solution}{pb:g10:functions:1}
\textbf{1.} Domínio: $ x \neq 0$ (divisão por $ x $). $ f(1) =
\frac12(1 + 2) = \frac32$; $ f(2) = \frac12(2 + 1) = \frac32$;
$ f\!\left(\frac32\right) = \frac12\left(\frac32 +
\frac43\right) = \frac{17}{12}$.

\textbf{2.} $\frac12\left(x + \frac2x\right) = \frac32$ dá
$ x^2 + 2 = 3x $, ou seja\ $ x^2 - 3x + 2 = (x - 1)(x - 2) = 0$: o
pré-imagens de $\frac32$ são $1$ e $2$ --- como pergunta 1
anunciado.

\textbf{3.} $ f(x) = x $ dá $ x + \frac2x = 2x $, então
$\frac2x = x $, $ x^2 = 2$: pontos fixos $\sqrt2$ e $-\sqrt2$.
O positivo é a diagonal do quadrado unitário, \omterm{ex:g10:numbers:classify}{irracional}
pelo problema da irracionalidade do fim de semana do volume do Ensino Médio.

\textbf{4.} $1 \to \frac32 \to \frac{17}{12} \to
\frac{577}{408}$: Garça aproximações de $\sqrt2$. Seu
receita ``média do palpite com $2/$ acho que'' é exatamente
\emph{iterando o \omterm{def:g10:functions:function}{função} $ f $} --- e o número
perseguições de processo é $ f $ de \omterm{pb:g10:functions:1}{ponto fixo}.

\textbf{5.} O pontos fixos são as travessias do \omterm{def:g10:functions:graph}{gráfico}
com a diagonal $ y = x $: alimentando as saídas de volta caminha o
apontar ao longo do \omterm{def:g10:functions:graph}{gráfico} em direção a essa travessia. Em
o problema de fim de semana dos dois termômetros do volume do ensino médio, a leitura $-40^\circ $ era o mesmo
ideia de cruzar a diagonal para a conversão \omterm{def:g10:functions:function}{função}.

\textbf{6.} $ f(v) - f(u) = \frac{v - u}{2} + \frac1v - \frac1u
= \frac{v - u}{2} + \frac{u - v}{uv}
= (v - u)\left(\frac12 - \frac{1}{uv}\right)
= \frac{(v - u)(uv - 2)}{2uv}$. Para $0 < u < v \leq \sqrt2$:
$ uv < 2$, o colchete é negativo, $ f(v) < f(u)$: diminuindo.
Para $\sqrt2 \leq u < v $: $ uv > 2$: \omterm{def:g10:functions:variations}{aumentando}. Ponto de viragem:
$\sqrt2$.

\textbf{7.} Diminuindo antes $\sqrt2$, \omterm{def:g10:functions:variations}{aumentando} depois: $ f $
atinge o seu \omterm{def:g10:functions:extrema}{mínimo} no $\sqrt2$, valor
$ f(\sqrt2) = \frac12\left(\sqrt2 + \frac{2}{\sqrt2}\right)
= \sqrt2$. Consequência: cada saída da máquina (positiva
entrada) é pelo menos $\sqrt2$ --- depois de uma volta da manivela,
palpites moram em $\intco{\sqrt2}{+\infty}$.

\textbf{8.} Se $ x > \sqrt2$, então $ x^2 > 2$, então
$\frac2x < x $ e a média $ f(x)$ de $ x $ e $\frac2x $ é
menos que $ x $; e $ f(x) \geq \sqrt2$ pela questão 7, com
igualdade apenas em $ x = \sqrt2$. Então
$\sqrt2 < f(x) < x $: cada iteração melhora estritamente sem
ultrapassando --- a sequência da pergunta 4 desliza para baixo
$\sqrt2$.

\textbf{9.} Sobre $\intoo{0}{+\infty}$: diminuindo de
$+\infty $ (aproximar $0$) até o \omterm{def:g10:functions:extrema}{mínimo} $\sqrt2$ no
$ x = \sqrt2$, então \omterm{def:g10:functions:variations}{aumentando} sem limite.

\textbf{10.} $ g(x) = x $ dá $ x^2 = x $, $ x(x - 1) = 0$: fixo
pontos $0$ e $1$. De $0.9$: $0.81$, $0.656$, $0.430$,
$0.185$ --- deslizando em direção $0$. De $1.1$: $1.21$, $1.464$,
$2.144$, $4.595$ --- fugindo para o infinito. O \omterm{pb:g10:functions:1}{ponto fixo} $0$
atrai, $1$ repele: a diferença de um fio de cabelo no início, oposto
destinos.

\textbf{11.} $7 \to 22 \to 11 \to 34 \to 17 \to 52 \to 26 \to
13 \to 40 \to 20 \to 10 \to 5 \to 16 \to 8 \to 4 \to 2 \to 1$:
dezesseis passos, chegando a $52$.

\textbf{12.} $27 \to 82 \to 41 \to 124 \to 62 \to 31 \to 94 \to
47 \to 142 \to 71 \to 214 \to \dots $ --- ainda subindo depois
dez passos, a caminho de um pico de $9\,232$ e um $111$-passo
voo. Lição: uma regra de duas linhas pode produzir um comportamento que ninguém pode
prever rapidamente --- números da porta ao lado ($26$, $27$, $28$)
têm vôos totalmente diferentes.

\textbf{13.} Um poder de $2$ divide toda a escada:
$2^k \to 2^{k-1} \to \dots \to 2 \to 1$. Na parte inferior,
$ h(1) = 4$, $ h(4) = 2$, $ h(2) = 1$: o voo entra no ciclo
$4 \to 2 \to 1 \to 4$. Pontos fixos precisaria $\frac n2 = n $
(apenas $ n = 0$) ou $3n + 1 = n $ (negativo): nenhum entre os
positivo \omterm{def:g10:numbers:sets}{inteiros} --- os voos observados não terminam em um fixo
ponto mas naquele pouco \emph{ciclo}.

\textbf{14.} Um contra-exemplo seria um voo que
cresce para sempre (nunca voltando para $1$) ou um segundo ciclo
separado de $4 \to 2 \to 1$. Verificação até $10^{20}$
deixa infinitos números não testados --- exatamente o
lição de regiões circulares do problema do fim de semana do oráculo do palito de fósforo do volume do Ensino Médio: acordo sobre
um número finito de casos não prova nada sobre todos eles.

\textbf{15.} Para a máquina de Heron tivemos \emph{estrutura} --- um
diferença fatorada (questão 6) revelando variações, uma
\omterm{def:g10:functions:extrema}{mínimo}, um aperto --- enquanto a mudança de paridade do granizo
regra não oferece tal solução: ninguém encontrou a estrutura que
doma isso.

\textbf{16.} $\sqrt{x - 3}$: precisa $ x - 3 \geq 0$:
$\intco{3}{+\infty}$. $\frac{1}{x^2 - 9}$: precisa
$ x \neq \pm 3$: $\R $ privado de $-3$ e $3$.
$\sqrt{9 - x^2}$: precisa $ x^2 \leq 9$: $\intcc{-3}{3}$.

\textbf{17.} $ f(x) = 3$ dá $ x + \frac2x = 6$, então
$ x^2 - 6x + 2 = 0$, ou seja\ $(x - 3)^2 = 7$:
$ x = 3 - \sqrt7$ ou $ x = 3 + \sqrt7$ (ambos positivos: dois exatos
pré-imagens).

\textbf{18.} Sentido físico enquanto a pedra está no alto:
$ H(t) = 5t(4 - t) \geq 0$ para $ t \in \intcc{0}{4}$. Completando:
$ H(t) = 20 - 5(t - 2)^2$: \omterm{def:g10:functions:extrema}{máximo} $20$ estou em $ t = 2$ S.
Variações: \omterm{def:g10:functions:variations}{aumentando} sobre $\intcc{0}{2}$ de $0$ para $20$,
diminuindo sobre $\intcc{2}{4}$ de volta para $0$.

\textbf{19.} $20t - 5t^2 \geq 15 \iff t^2 - 4t + 3 \leq 0 \iff
(t - 1)(t - 3) \leq 0 \iff t \in \intcc{1}{3}$: dois completos
segundos acima $15$ m.

\textbf{20.} Cada máquina tem um domínio (onde a regra é
permitido), uma regra (a fórmula ou a mudança de paridade) e um
\omterm{def:g10:functions:graph}{gráfico} ou tabela exibindo tudo. De cada um perguntamos aos dois
ótimas perguntas: \emph{onde a repetição leva} (fixo
pontos, ciclos, atração --- resolvido para Heron, aberto para o
granizo) e \emph{como a saída se move com a entrada}
(variações, extremos --- o cume da pedra, o da Garça \omterm{def:g10:functions:extrema}{mínimo}).
Os capítulos a seguir aprimoram ambos: referência funções próximo,
então, nos anos posteriores, derivativos para medir a variação e
limites para certificar onde as iterações chegam.
\end{solution}
