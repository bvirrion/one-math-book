\begin{solution}{exo:g10:vectors:1}
$\vect{AB} + \vect{BD} = \vect{AD}$ (Chasles).

$\vect{KL} + \vect{LM} + \vect{MK} = \vect{KK} = \vec 0$.

$\vect{AC} - \vect{BC} = \vect{AC} + \vect{CB} = \vect{AB}$.

$\vect{AB} + \vect{CA} = \vect{CA} + \vect{AB} = \vect{CB}$.
\end{solution}

\begin{solution}{exo:g10:vectors:2}

$\vect{BC}\,(-1-5,\ 4-3) = (-6, 1)$; \quad
$\vect{AC}\,(-1-2,\ 4-1) = (-3, 3)$.
Verificar: $(3, 2) + (-6, 1) = (-3, 3)$, coordenada por coordenada.
\end{solution}

\begin{solution}{exo:g10:vectors:3}
$\vec u + \vec v = (2 + (-1),\ -3 + 4) = (1, 1)$.

$3\vec u = (6, -9)$.

$2\vec u - \vec v = (4 - (-1),\ -6 - 4) = (5, -10)$.

$\norm{\vec u} = \sqrt{2^2 + (-3)^2} = \sqrt{13}$.
\end{solution}

\begin{solution}{exo:g10:vectors:4}
(um) $4 \times 9 - 6 \times 6 = 36 - 36 = 0$: \omterm{def:g10:vectors:collinear}{colinear}
($\vec v = \frac32 \vec u $).

(b) $2 \times 10 - (-5) \times (-4) = 20 - 20 = 0$: \omterm{def:g10:vectors:collinear}{colinear}
($\vec v = -2\vec u $).

(c) $3 \times 3 - 1 \times 1 = 8 \neq 0$: não \omterm{def:g10:vectors:collinear}{colinear}.
\end{solution}

\begin{solution}{exo:g10:vectors:5}
$\vect{AB}\,(4, -2)$, e $\vect{DC}\,(6 - x_D,\ 3 - y_D)$. A condição
$\vect{AB} = \vect{DC}$ dá $6 - x_D = 4$ e $3 - y_D = -2$, então

\end{solution}

\begin{solution}{exo:g10:vectors:6}
\emph{1.} $\vect{AB}\,(3, 2)$ e $\vect{AC}\,(6, -2)$, então
$\vect{AB} + \vect{AC} = (9, 0)$ e
$ M = (-2 + 9,\ 1 + 0) = (7, 1)$.

\emph{2.} Pela regra do paralelogramo, $\vect{AM} = \vect{AB} + \vect{AC}$
significa que $ ABMC $ é um paralelogramo ($ M $ é o quarto vértice oposto
$ A $). Pode-se verificar: $\vect{AB}\,(3,2)$ e $\vect{CM}\,(7-4,\ 1-(-1)) =
(3, 2)$ são iguais.
\end{solution}

\begin{solution}{exo:g10:vectors:7}
Primeiro triplo: $\vect{AB}\,(3, 2)$, $\vect{AC}\,(9, 6)$;
$3 \times 6 - 2 \times 9 = 0$: alinhado ($\vect{AC} = 3\vect{AB}$).

Segundo triplo: $\vect{AB}\,(2, -1)$, $\vect{AC}\,(8, -4)$;
$2 \times (-4) - (-1) \times 8 = -8 + 8 = 0$: alinhado também
($\vect{AC} = 4\vect{AB}$).
\end{solution}

\begin{solution}{exo:g10:vectors:8}
$\vect{AB}\,(4, 2)$ e $\vect{DC}\,(6-2,\ 1-(-1)) = (4, 2)$: os dois
\omterm{def:g10:vectors:vector}{vetores} são \emph{igual}, então $ ABCD $ é um paralelogramo e, em particular
$(AB) \parallel (CD)$. A propriedade do paralelogramo é a mais forte:
implica o paralelismo (colinearidade do \omterm{def:g10:vectors:vector}{vetores}), e não o contrário.
\end{solution}

\begin{solution}{exo:g10:vectors:9}
\emph{1.} $\vect{AB}\,(6, 3)$, então
$\vect{AI} = \frac23\vect{AB} = (4, 2)$ e $ I(4, 2)$.

\emph{2.} $\vect{CB}\,(6-2,\ 3-5) = (4, -2)$, então
$\vect{CJ} = \frac23\vect{CB} = \left(\frac83, -\frac43\right)$ e
$ J\left(2 + \frac83,\ 5 - \frac43\right) = \left(\frac{14}{3},
\frac{11}{3}\right)$.

\emph{3.} $\vect{IJ}\,\left(\frac{14}{3} - 4,\ \frac{11}{3} - 2\right) =
\left(\frac23, \frac53\right)$ e $\vect{AC}\,(2, 5)$. Colinearidade
critério:
\[
\frac23 \times 5 - \frac53 \times 2 = \frac{10}{3} - \frac{10}{3} = 0 :
\]
o \omterm{def:g10:vectors:vector}{vetores} são \omterm{def:g10:vectors:collinear}{colinear}, então $(IJ)$ é paralelo a $(AC)$. (Ambos $ I $ e
$ J $ sente-se a um terço do caminho do lado para $ B $: o pequeno
triângulo $ IBJ $ é uma redução de $ ABC $.)
\end{solution}

\begin{solution}{exo:g10:vectors:10}
O pontos médios são
\[
I\left(\tfrac{x_A + x_B}{2}, \tfrac{y_A + y_B}{2}\right), \quad
J\left(\tfrac{x_B + x_C}{2}, \tfrac{y_B + y_C}{2}\right), \quad
K\left(\tfrac{x_C + x_D}{2}, \tfrac{y_C + y_D}{2}\right), \quad
L\left(\tfrac{x_D + x_A}{2}, \tfrac{y_D + y_A}{2}\right).
\]
Primeiro \omterm{def:g10:coordgeom:system}{coordenadas} dos dois \omterm{def:g10:vectors:vector}{vetores}:
\[
x_J - x_I = \frac{x_B + x_C}{2} - \frac{x_A + x_B}{2}
= \frac{x_C - x_A}{2},
\qquad
x_K - x_L = \frac{x_C + x_D}{2} - \frac{x_D + x_A}{2}
= \frac{x_C - x_A}{2},
\]
e o mesmo cálculo funciona para o segundo \omterm{def:g10:coordgeom:system}{coordenadas}. Então
$\vect{IJ} = \vect{LK}$ (ambos iguais $\frac12\vect{AC}$), e $ IJKL $ é um
paralelogramo, qualquer que seja o quadrilátero $ ABCD $ parece.
\end{solution}

\begin{solution}{pb:g10:vectors:1}
\textbf{1.} $\vect{AB} + \vect{BC} + \vect{CD} = \vect{AD}$;
\quad $\vect{MA} - \vect{MB} = \vect{MA} + \vect{BM} =
\vect{BA}$; \quad $\vect{AB} + \vect{BC} + \vect{CD} +
\vect{DA} = \vect{AA} = \vec 0$ (reordenando o terceiro \omterm{def:g10:vectors:sum}{soma}).

\textbf{2.} $\vect{OB} - \vect{OA} = \vect{AO} + \vect{OB} =
\vect{AB}$ por Chasles.

\textbf{3.} $ I $ é o \omterm{prop:g10:coordgeom:midpoint}{ponto médio} de $[AB]$ exatamente quando
$\vect{IA} = -\vect{IB}$, ou seja\
$\vect{IA} + \vect{IB} = \vec 0$. Então, para qualquer $ O $:
$\vec 0 = \vect{IO} + \vect{OA} + \vect{IO} + \vect{OB}$, então
$2\,\vect{OI} = \vect{OA} + \vect{OB}$ --- e inversamente.

\textbf{4.} $\vect{AB} = (4, 2)$, e $ D $ satisfaz
$\vect{DC} = \vect{AB}$: $ D = C - (4, 2) = (2, -2)$ --- o mesmo $ D $ como \cref{pb:g10:coordgeom:1}, uma linha mais rápida.

\textbf{5.} A tradução de vetor $\vec u $ --- exatamente o
transformação produzida por dois espelhos paralelos ou dois
meias voltas. \omterm{def:g10:functions:function}{Imagem} de $(1, 2)$: $(4, 1)$.

\textbf{6.} \omterm{def:g10:algebra:expand}{Expandindo} cada $\vect{GX}$ como
$\vect{GO} + \vect{OX}$:
$3\,\vect{GO} + \vect{OA} + \vect{OB} + \vect{OC} = \vec 0$,
então $\vect{OG} = \frac13(\vect{OA} + \vect{OB} + \vect{OC})$:
um e apenas um desses pontos, cujo \omterm{def:g10:coordgeom:system}{coordenadas} são as médias
dos três vértices'.

\textbf{7.} $\vect{AB} + \vect{AC} =
(\vect{AG} + \vect{GB}) + (\vect{AG} + \vect{GC})
= 2\vect{AG} + (\vect{GB} + \vect{GC})
= 2\vect{AG} - \vect{GA} = 3\vect{AG}$, de onde
$\vect{AG} = \frac13(\vect{AB} + \vect{AC})$. O \omterm{prop:g10:coordgeom:midpoint}{ponto médio}
fórmula (questão 3, vista de $ A $) dá
$\vect{AK} = \frac12(\vect{AB} + \vect{AC})$. Comparando:
$\vect{AG} = \frac23\vect{AK}$ --- $ G $ encontra-se na mediana
$[AK]$, a dois terços do caminho $ A $.

\textbf{8.} A condição definidora trata $ A $, $ B $, $ C $
de forma idêntica, então o mesmo cálculo de $ B $ ou de $ C $
coloca o \emph{mesmo} $ G $ em dois terços dessas medianas:
a simultaneidade não custa nada. A prova do paralelogramo foi
engenhoso, mas ad hoc; \omterm{def:g10:coordgeom:system}{coordenadas} computar, mas obscuro;
\omterm{def:g10:vectors:vector}{vetores} torne a simetria visível e a prova duas linhas.

\textbf{9.} $ G\left(\frac{1+5+3}{3}, \frac{1+2+6}{3}\right) =
(3, 3)$; $ K(4, 4)$; $\vect{AG} = (2, 2)$ e
$\vect{AK} = (3, 3)$: determinante
$2 \times 3 - 2 \times 3 = 0$, \omterm{def:g10:vectors:collinear}{colinear}, e de fato
$\vect{AG} = \frac23\vect{AK}$.

\textbf{10.} $ G'\left(\frac{2 + 5 + 3}{4}, \frac{2 + 2 +
6}{4}\right) = (2.5,\ 2.5)$: retirado de $ G(3,3)$ em direção ao
vértice duplo $ A(1,1)$, como deveria ser um ponto de equilíbrio.

\textbf{11.} $3 \times 4 - (-2)(-6) = 12 - 12 = 0$: \omterm{def:g10:vectors:collinear}{colinear}
($\vec v = -2\vec u $). $2 \times 9 - 5 \times 4 = -2 \neq 0$:
não \omterm{def:g10:vectors:collinear}{colinear}.

\textbf{12.} $\vect{AB} = (2, 3)$, $\vect{AC} = (6, 9)$:
determinante $2 \times 9 - 3 \times 6 = 0$: alinhado.

\textbf{13.} $\vect{OI} = \frac12(\vect{OA} + \vect{OB})$ e
$\vect{OK} = \frac12(\vect{OC} + \vect{OD})$, então o \omterm{prop:g10:coordgeom:midpoint}{ponto médio}
de $[IK]$ tem posição vetor
$\frac12(\vect{OI} + \vect{OK}) = \frac14(\vect{OA} +
\vect{OB} + \vect{OC} + \vect{OD})$ --- e o cálculo
para $[JL]$ dá o mesmo vetor. Os dois bimedianos (e,
de Varignon, o paralelogramo $ IJKL $ diagonais) todos compartilham
este centro: o centróide do quadrilátero.

\textbf{14.} $\vect{MN} = \vect{AN} - \vect{AM} =
\frac13\vect{AC} - \frac13\vect{AB} = \frac13\vect{BC}$:
então $(MN) \parallel (BC)$ e $ MN = \frac13 BC $.

\textbf{15.} Com $\vect{AM} = k\,\vect{AB}$ e
$\vect{AN} = k\,\vect{AC}$: $\vect{MN} = k\,\vect{BC}$ --- um
linha contendo o \omterm{prop:g10:coordgeom:midpoint}{ponto médio} teorema ($ k = \frac12$) e
Tales (qualquer $ k $): a razão passa pela subtração
intocado.

\textbf{16.} Resultante $(3, 4)$, magnitude
$\sqrt{9 + 16} = 5$ km/h: o barco realmente viaja
nordeste por norte ao longo do $3$--$4$--$5$ triângulo
hipotenusa.

\textbf{17.} $\vec u + \vec v + \vec w = (0, 0)$: o anel
permanece parado --- equilíbrio. Ao longo de um polígono fechado
$ A_1 A_2 \dots A_n A_1$, Chasles telescópica o \omterm{def:g10:vectors:sum}{soma} para
$\vect{A_1 A_1} = \vec 0$: andar por aí traz você de volta.

\textbf{18.} $\vec w = -(\vec u + \vec v) = -(3, 2) =
(-3, -2)$.

\textbf{19.} Resultante $(1.5,\ 2)$, magnitude
$\sqrt{2.25 + 4} = 2.5$ EM. Cruzamento: $50$ estou em $2$ m/s
através de: $25$ é; deriva: $1.5 \times 25 = 37.5$ estou a jusante.

\textbf{20.} Reprovado neste fim de semana: os dois terços do centróide
teorema (questão 7), o centro bimediano/Varignon (questão
13), Tales – \omterm{prop:g10:coordgeom:midpoint}{ponto médio} (questões 14-15). Que flechas ainda
não posso fazer: medir \emph{ângulos} (e, portanto, comprimentos em
direções arbitrárias) --- o produto escalar da nota 11 é
precisamente o instrumento que faltava.
\end{solution}
