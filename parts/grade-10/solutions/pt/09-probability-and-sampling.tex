\section*{Capítulo \ref{ch:g10:proba} --- Probabilidade e Amostragem}

\begin{solution}{exo:g10:proba:1}
Cada um dos $6$ resultados tem \omterm{def:g10:proba:distribution}{probabilidade} $\frac16$.
``Conseguir um $6$'': $\frac16$.
``Número ímpar'' $= \{1, 3, 5\}$: $\frac36 = \frac12$.
``No \omterm{def:g10:functions:extrema}{máximo} $4$'' $= \{1,2,3,4\}$: $\frac46 = \frac23$.
``Conseguir um $7$'': impossível \omterm{def:g10:proba:events}{evento}, \omterm{def:g10:proba:distribution}{probabilidade} $0$.
\end{solution}

\begin{solution}{exo:g10:proba:2}
Há $10$ bolinhas de gude igualmente prováveis.
$\P(\text{green}) = \frac{5}{10} = \frac12$.
$\P(\text{not black}) = 1 - \frac{2}{10} = \frac{8}{10} = \frac45$.
Verde e amarelo são \omterm{def:g10:proba:operations}{incompatível}:
$\P(\text{green or yellow}) = \frac{5}{10} + \frac{3}{10} = \frac{8}{10}
= \frac45$.
\end{solution}

\begin{solution}{exo:g10:proba:3}
Deixar $ S $: ``estuda espanhol'', $ G $: ``estuda alemão''. Então
$\P(S) = \frac{18}{30}$, $\P(G) = \frac{10}{30}$,
$\P(S \cap G) = \frac{4}{30}$. Regra de adição:
\[
\P(S \cup G) = \frac{18}{30} + \frac{10}{30} - \frac{4}{30}
= \frac{24}{30} = \frac45 .
\]
\omterm{def:g10:proba:operations}{Complemento}: $\P(\text{neither}) = 1 - \frac45 = \frac15$.
\end{solution}

\begin{solution}{exo:g10:proba:4}
O \omterm{met:g10:proba:tree}{árvore} tem quatro caminhos HH, HT, TH, TT, cada um dos \omterm{def:g10:proba:distribution}{probabilidade}
$\frac12 \times \frac12 = \frac14$.
Duas cabeças: $\frac14$. Exatamente uma cabeça (HT ou TH): $\frac24 = \frac12$.
Pelo menos uma cabeça: $1 - \P(\text{TT}) = 1 - \frac14 = \frac34$.
\end{solution}

\begin{solution}{exo:g10:proba:5}
Primeiro sorteio: vermelho com \omterm{def:g10:proba:distribution}{probabilidade} $\frac{4}{10}$, branco $\frac{6}{10}$.
Segundo sorteio \emph{sem} substituição: depois de um vermelho, $3$ vermelho e $6$
brancos permanecem entre $9$; depois de um branco, $4$ vermelho e $5$ branco entre $9$.

Dois tintos: $\frac{4}{10} \times \frac39 = \frac{12}{90} = \frac{2}{15}$.

Cores diferentes: caminhos RW e WR:
\[
\frac{4}{10} \times \frac69 + \frac{6}{10} \times \frac49
= \frac{24}{90} + \frac{24}{90} = \frac{48}{90} = \frac{8}{15}.
\]
\end{solution}

\begin{solution}{exo:g10:proba:6}
\emph{1.} Cada dado mostra independentemente cada face com \omterm{def:g10:proba:distribution}{probabilidade}
$\frac16$, então o $36$ pares ordenados são igualmente prováveis; a soma é
determinado pela dupla.

\emph{2.} Soma $7$: os pares $(1,6), (2,5), (3,4), (4,3), (5,2), (6,1)$,
então $\P = \frac{6}{36} = \frac16$. Soma $12$: apenas $(6,6)$,
$\P = \frac{1}{36}$. Sum at most $4$: pares somando para $2$, $3$ ou $4$:
$(1,1)$; $(1,2), (2,1)$; $(1,3), (2,2), (3,1)$ --- seis pares, então
$\P = \frac{6}{36} = \frac16$.

\emph{3.} O número de pares que alcançam cada soma cresce de $1$ (soma $2$)
para $6$ (soma $7$) então diminui: a soma mais provável é $7$.
\end{solution}

\begin{solution}{exo:g10:proba:7}
Cada pergunta é respondida corretamente com \omterm{def:g10:proba:distribution}{probabilidade} $\frac14$,
de forma independente. \omterm{met:g10:proba:tree}{Árvore} com dois níveis:

Ambos corretos: $\frac14 \times \frac14 = \frac{1}{16}$.

Exatamente um correto: caminhos corretos – errado e errado – correto:
$\frac14 \times \frac34 + \frac34 \times \frac14 = \frac{6}{16}
= \frac38$.

Nenhum correto: $\frac34 \times \frac34 = \frac{9}{16}$.

Verificar: $\frac{1}{16} + \frac{6}{16} + \frac{9}{16} = 1$.
\end{solution}

\begin{solution}{exo:g10:proba:8}
\emph{1.} Observado freqüência: $\frac{396}{900} = 0.44$. Com $ n = 900$:
$\frac{1}{\sqrt{900}} = \frac{1}{30} \approx 0.033$, então a flutuação
\omterm{def:g10:numbers:interval}{intervalo} é sobre $\intcc{0.467}{0.533}$.

\emph{2.} O observado $0.44$ fica fora do \omterm{def:g10:numbers:interval}{intervalo}: \omterm{def:g10:proba:sample}{amostras} de $900$
eleitores de uma população com $50\%$ a aprovação raramente vai tão longe. O
\omterm{def:g10:algebra:equation}{pesquisa} lança sérias dúvidas sobre o alegado $50\%$.
\end{solution}

\begin{solution}{exo:g10:proba:9}
\emph{1.} $\P(\text{win } 6) = \frac16$ (role um $6$),
$\P(\text{win } 3) = \frac16$ (role um $5$),
$\P(\text{win } 0) = \frac46 = \frac23$.

\emph{2.} Sobre $600$ jogos, espere cerca de $100$ seis, $100$ cincos, $400$
outros rolos. Ganhos: cerca de $100 \times 6 + 100 \times 3 = 900$. Custo:
$600 \times 2 = 1200$. Resultado líquido esperado: $900 - 1200 = -300$, ou seja\ um
perda de cerca $0.5$ por jogo em média --- o jogo não vale
jogando.
\end{solution}

\begin{solution}{pb:g10:proba:1}
\textbf{1.} $ P(\text{king or heart}) = \frac{4}{52} +
\frac{13}{52} - \frac{1}{52} = \frac{16}{52} = \frac{4}{13}$:
o rei dos corações está sentado em ambos \omterm{def:g10:proba:events}{eventos} e não deve ser contado
duas vezes --- daí a subtração
(\cref{thm:g10:proba:addition}).

\textbf{2.} $ P(\text{no head}) = \left(\frac12\right)^3 =
\frac18$, então $ P(\text{at least one}) = \frac78$.

\textbf{3.} $ P(RR) = \frac35 \times \frac24 = \frac{3}{10}$;
$ P(\text{one of each}) = \frac35 \times \frac24 + \frac25
\times \frac34 = \frac{6}{20} + \frac{6}{20} = \frac35$.

\textbf{4.} $ P(BB) = \frac25 \times \frac14 = \frac1{10}$; e
$\frac{3}{10} + \frac{6}{10} + \frac{1}{10} = 1$: o árvorede
sai sempre total $1$.

\textbf{5.} \omterm{def:g10:proba:operations}{Incompatível} \omterm{def:g10:proba:events}{eventos} obedecer $ P(A \cup B) = P(A) +
P(B) = 1.1 > 1$: impossível --- alguns \omterm{def:g10:proba:distribution}{probabilidade} foi inventado.

\textbf{6.} (A maioria das pessoas, incluindo os milhares de escritores de cartas,
responda ``não importa, 50--50''. Guarde sua resposta para
questão 8.)

\textbf{7.} Carro atrás da porta 1 (\omterm{def:g10:proba:distribution}{probabilidade} $\frac13$): o anfitrião
abre qualquer porta de cabra; a troca sai do carro: \emph{perder}.
Carro atrás da porta 2 ($\frac13$): o host \emph{deve} abra a porta 3;
a troca cai na porta 2: \emph{ganhar}. Carro atrás da porta 3:
simétrico: \emph{ganhar}. Total:
$ P(\text{switch wins}) = \frac13 + \frac13 = \frac23$.

\textbf{8.} Sua primeira escolha está errada com \omterm{def:g10:proba:distribution}{probabilidade}
$\frac23$ --- e precisamente então, a revelação forçada do anfitrião
deixa o carro atrás da porta restante: a troca converte
cada erro inicial em uma vitória. $\frac23$, seja qual for o instinto
disse.

\textbf{9.} Seis cenários igualmente prováveis: carro atrás $1$, $2$
ou $3$, o anfitrião abre a porta $2$ ou $3$ (nunca seu). A revelação
mostra o carro em dois deles (car@2/opens 2, car@3/opens 3):
descartado. Entre os quatro sobreviventes --- car@1/opens 2,
carro@1/abre 3, carro@2/abre 3, carro@3/abre 2 --- trocando vitórias
nos dois últimos: $\frac24 = \frac12$. Com um anfitrião ignorante,
50--50 \emph{é} correto: o paradoxo reside inteiramente no
conhecimento do anfitrião, que filtra os cenários de forma desigual.

\textbf{10.} Sua escolha segura o carro com \omterm{def:g10:proba:distribution}{probabilidade}
$\frac{1}{100}$; o anfitrião conhecedor $98$ revela funil o
restante $\frac{99}{100}$ para a única porta fechada. Mudar,
e vencer $99$ vezes em $100$.

\textbf{11.} Três cartas (um ás, dois curingas), um amigo que
espia e sempre descarta um curinga dos dois que você não
escolher; jogar $300$ rodadas, sempre trocando e registrando vitórias.
Esperado freqüência $\frac23 \approx 0.667$, com flutuação
$\pm\frac{1}{\sqrt{300}} \approx 0.058$: espere entre cerca de
$61\,\%$ e $72\,\%$ trocando vitórias --- longe de qualquer
``50--50''.

\textbf{12.} Configurações igualmente prováveis (mais velho - mais jovem):
BB, BG, GB, GG. ``Pelo menos um menino'' mantém BB, BG, GB:
$ P(\text{BB}) = \frac13$.

\textbf{13.} ``O mais velho é um menino'' mantém BB, BG:
$ P(\text{BB}) = \frac12$. Diferentes filtros de informação
cenários diferentes --- exatamente como os dois Montys fizeram: o que
você aprende assuntos e \emph{como} você aprendeu que isso importa apenas
tanto.

\textbf{14.} Todos iguais: HHH, TTT: $\frac28 = \frac14$.
O erro de D'Alembert: seus dois “casos” não são igualmente prováveis
--- ``nem todos iguais'' agrupa seis dos oito resultados.
\Cref{prop:g10:proba:uniform} conta apenas sobre igualmente provável
resultados.

\textbf{15.} Ganhos médios por jogo: $6 \times \frac16 + 3
\times \frac16 = 1.50$ euros, contra um $2$-participação em euros: uma
perda média de $0.50$ por jogo, cerca de $300$ euros acima $600$
jogos. O prêmio justo equaliza o pagamento médio:
$1\,\% \times 10\,000 = 100$ euros; prêmios reais adicionam custos
e uma margem de segurança --- a seguradora deve sobreviver a anos ruins,
não apenas os médios.

\textbf{16.} \omterm{def:g10:numbers:interval}{Intervalo}: $0.5 \pm \frac{1}{\sqrt{100}} =
\intcc{0.4}{0.6}$. $55$ cabeças ($0.55$): por dentro, normal.
$70$ cabeças ($0.70$): muito lá fora --- duvide da moeda.

\textbf{17.} $\frac{1}{\sqrt{1000}} \approx 0.032$: uma feira
$50\,\%$ candidato normalmente pesquisaria em
$\intcc{0.468}{0.532}$ --- e $0.52$ senta-se confortavelmente por dentro.
A enquete \emph{sugere} uma vantagem, mas não prova nada: a margem
engole.

\textbf{18.} Primeiro \omterm{def:g10:proba:sample}{amostra}: \omterm{def:g10:numbers:interval}{intervalo} $0.04 \pm \frac{1}{20} =
\intcc{0}{0.09}$; o observado $0.07$ está dentro: nenhum veredicto.
Segundo: $0.04 \pm \frac{1}{50} = \intcc{0.02}{0.06}$; o
observado $0.07$ cai fora: a reclamação agora é grave
dúvida. Mesmo freqüência, instrumento mais nítido: a precisão cresce como
$\sqrt n $.

\textbf{19.} A flutuação diminui como $\frac{1}{\sqrt n}$:
quadruplicar o \omterm{def:g10:proba:sample}{amostra}, reduza pela metade o ruído. Preconceito --- perguntando ao
multidão errada --- sobrevive a qualquer $ n $: dois milhões mal escolhidos
as cédulas continuaram erradas (o problema de como estão os dados no fim de semana do volume do ensino médio).

\textbf{20.} (i) Os resultados da verificação são igualmente prováveis antes
contando --- d'Alembert esqueceu. (ii) \omterm{met:g10:proba:tree}{Árvores} multiplique junto
filiais; ``pelo menos um'' se rende ao \omterm{def:g10:proba:operations}{complemento}.
(iii) Cenários de filtros de informação, e sua \emph{fonte}
muda o filtro --- conhecendo Monty contra Monty caído,
menino mais velho contra algum menino. (iv) Abordagem de \omterm{def:g10:stats:series}{frequências}
\omterm{def:g10:proba:distribution}{probabilidades} em velocidade $\frac{1}{\sqrt n}$, então um paciente
simulação resolve qualquer argumento --- incluindo um com dez
mil escritores de cartas furiosos.
\end{solution}
