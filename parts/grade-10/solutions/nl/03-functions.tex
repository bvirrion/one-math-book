\begin{solution}{exo:g10:functions:1}
$f(0) = 1$; \quad $f(2) = 8 - 6 + 1 = 3$; \quad
$f(-1) = 2 + 3 + 1 = 6$; \quad
$f\!\left(\tfrac12\right) = 2 \times \tfrac14 - \tfrac32 + 1 = \tfrac12 -
\tfrac32 + 1 = 0$.
\end{solution}

\begin{solution}{exo:g10:functions:2}

$h$: eist $2x - 6 \geq 0$, d.w.z.\ $x \geq 3$: domein

\begin{solution}{exo:g10:functions:3}
\emph{Originelen van $0$:} $x^2 - 2x = x(x-2) = 0$, dus $0$ en $2$.

\emph{Originelen van $3$:} $x^2 - 2x = 3$, d.w.z.\ $x^2 - 2x - 3 = 0$,
d.w.z.\ $(x-3)(x+1) = 0$ (controleer door te ontwikkelen), dus $3$ en
$-1$.

\emph{Originelen van $-1$:} $x^2 - 2x + 1 = (x - 1)^2 = 0$, dus het
enige origineel $1$.
\end{solution}

\begin{solution}{exo:g10:functions:5}
\emph{1.} De grootste waarde in de tabel is $4 = g(3)$ (maximum), de
kleinste is $-2 = g(0)$ (minimum).

\emph{2.} Op $\intcc{-3}{0}$ is de functie dalend, en $-1 < -0.5$, dus
$g(-1) \geq g(-0.5)$.

\emph{3.} Op elk monotonie-interval neemt $g$ elke waarde ten hoogste
één keer aan, dus $g(x) = 0$ heeft ten hoogste één oplossing per
interval: ten hoogste $3$ oplossingen in totaal. (Hier ligt $0$ tussen
$-2$ en $1$, tussen $-2$ en $4$, en tussen $0$ en $4$, dus er zijn er
precies drie.)
\end{solution}

\begin{solution}{exo:g10:functions:6}
Laat $u < v$. Dan
\[
f(v) - f(u) = (-2v + 7) - (-2u + 7) = -2(v - u) < 0,
\]
omdat $v - u > 0$. Dus $f(v) < f(u)$: $f$ is strikt dalend op $\R$.
\end{solution}

\begin{solution}{exo:g10:functions:7}
Voltooi het kwadraat: $(x+3)^2 = x^2 + 6x + 9$, dus
\[
f(x) = x^2 + 6x + 11 = (x + 3)^2 + 2 .
\]
Voor elke $x$ is $(x+3)^2 \geq 0$, dus $f(x) \geq 2$; en
$f(-3) = 0 + 2 = 2$. Het minimum van $f$ is $2$, bereikt bij $x = -3$.
\end{solution}

\begin{solution}{exo:g10:functions:8}
\emph{1.} Lengte $+$ breedte $= 20$, dus de lengte is $20 - x$ en
\[
A(x) = x(20 - x).
\]
Beide afmetingen moeten positief zijn: $0 < x < 20$.

\emph{2.} $A(4) = 64$, $A(8) = 96$, $A(10) = 100$, $A(12) = 96$,
$A(16) = 64$. De waarden stijgen tot $x = 10$ en dalen dan symmetrisch:
de oppervlakte lijkt het grootst voor $x = 10$, d.w.z.\ voor een
\emph{vierkante} tuin. (\cref{ch:g11:quad} bewijst dit vermoeden.)
\end{solution}

\begin{solution}{exo:g10:functions:9}
\emph{1.} Voor elke $x$: $x^2 + 1 \geq 1 > 0$, dus $f(x) = \frac{1}{x^2+1}$
is positief, en omdat $x^2 + 1 \geq 1$, geeft omkeren (beide kanten
positief) $f(x) \leq 1$.

\emph{2.} Maximum: $f(0) = 1$ en $f(x) \leq 1$ voor alle $x$, dus $1$ is
het maximum, bereikt bij $0$. Minimum: $f(x) > 0$ altijd, maar geen
waarde van $x$ bereikt $0$ (een quotiënt van niet-nul getallen is
niet-nul), en $f$ neemt waarden willekeurig dicht bij $0$ voor grote
$x$; dus $f$ heeft geen minimum.
\end{solution}

\begin{solution}{exo:g10:functions:10}
\emph{1.} $f(v) - f(u) = v^2 - u^2 = (v-u)(v+u)$. Als $0 \leq u < v$, dan
$v - u > 0$ en $v + u > 0$ (som van een niet-negatief en een positief
getal), dus $f(v) - f(u) > 0$: $f$ is strikt stijgend op
$\intco{0}{+\infty}$.

\emph{2.} Als $u < v \leq 0$, dan is $v - u > 0$ nog steeds, maar nu
$v + u < 0$ (som van een niet-positief en een negatief getal), dus
$f(v) - f(u) < 0$: $f$ is strikt dalend op $\intoc{-\infty}{0}$.
\end{solution}
