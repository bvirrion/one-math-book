\section*{Hoofdstuk \ref{ch:g10:numbers} --- Getallen en verzamelingen van getallen}

\begin{solution}{exo:g10:numbers:1}
$\frac{15}{3} = 5 \in \N$. \quad $-7 \in \Z$. \quad
$\frac{22}{7} \in \Q$ (het is geen \omterm{def:g10:numbers:sets}{geheel getal}: $22 = 7 \times 3 + 1$).
\quad $\sqrt 9 = 3 \in \N$. \quad $\sqrt{10} \in \R$ (\omterm{ex:g10:numbers:classify}{irrationaal}, omdat
$10$ niet het kwadraat van een rationaal is --- hier toegegeven, in de
geest van \cref{thm:g10:numbers:sqrt2}). \quad
$-2.4 = -\frac{24}{10} \in \Q$. \quad $0 \in \N$.
\end{solution}

\begin{solution}{exo:g10:numbers:2}
(a) $\intco{-2}{5}$: gevulde stip bij $-2$, holle stip bij $5$.
(b) $\intoo{3}{+\infty}$: holle stip bij $3$, arcering naar rechts.
(c) $\intoc{-\infty}{-1}$: arcering van links tot een gevulde stip bij
$-1$.
(d) ``afstand van $x$ tot $2$ ten hoogste $3$'' betekent
$\abs{x - 2} \leq 3$, d.w.z.\ $x \in \intcc{-1}{5}$: gevulde stippen bij
$-1$ en $5$.
\end{solution}

\begin{solution}{exo:g10:numbers:3}
(a) De twee intervallen overlappen tussen $0$ en $2$:
$I \cap J = \intcc{0}{2}$ ($0 \in J$ en $2 \in I$, en beide behoren ook
tot de andere verzameling), en $I \cup J = \intco{-3}{4}$.

(b) $I \cap J$ is de verzameling van $x$ met $-2 \leq x$ en $x < 1$:
$\intco{-2}{1}$. De \omterm{def:g10:numbers:interunion}{vereniging} dekt alles: $I \cup J = \R$.
\end{solution}

\begin{solution}{exo:g10:numbers:4}
$\abs{-6} = 6$; \quad $\abs{4 - 9} = \abs{-5} = 5$; \quad
$\abs{-3 - 5} = \abs{-8} = 8$; \quad
$\sqrt 2 > 1$ dus $\abs{\sqrt2 - 1} = \sqrt2 - 1$; \quad
$1 - \sqrt2 < 0$ dus $\abs{1 - \sqrt2} = \sqrt2 - 1$ eveneens (een getal
en zijn tegengestelde hebben dezelfde \omterm{def:g10:numbers:abs}{absolute waarde}).
\end{solution}

\begin{solution}{exo:g10:numbers:5}
$\abs{x} = 5$: afstand $5$ tot $0$, dus $x = 5$ of $x = -5$.

$\abs{x - 1} = 3$: afstand $3$ tot $1$, dus $x = 4$ of $x = -2$.

$\abs{x - 4} \leq 1$: afstand ten hoogste $1$ tot $4$, dus
$x \in \intcc{3}{5}$.

$\abs{x + 2} < 3$: afstand strikt kleiner dan $3$ tot $-2$, dus
$x \in \intoo{-5}{1}$.
\end{solution}

\begin{solution}{exo:g10:numbers:6}
Elk \omterm{def:g10:numbers:interval}{interval} wordt beschreven door zijn middelpunt $a$ (midden van de
grenzen) en straal $r$ (halve lengte).

$\intcc{2}{8}$: $a = 5$, $r = 3$, dus $\abs{x - 5} \leq 3$.

$\intoo{-5}{1}$: $a = -2$, $r = 3$, open grenzen, dus
$\abs{x + 2} < 3$.

$\intcc{-7}{-3}$: $a = -5$, $r = 2$, dus $\abs{x + 5} \leq 2$.
\end{solution}

\begin{solution}{exo:g10:numbers:7}
Laat $x = 0.272727\dots$ Dan $100x = 27.2727\dots$, en aftrekken:
\[
100x - x = 27.2727\dots - 0.2727\dots = 27,
\]
dus $99x = 27$ en $x = \frac{27}{99} = \frac{3}{11}$, een quotiënt van
\omterm{def:g10:numbers:sets}{gehele getallen}: $x$ is rationaal.
\end{solution}

\begin{solution}{exo:g10:numbers:8}
\emph{1. Waar:} optellen van \omterm{def:g10:numbers:sets}{gehele getallen} (positieve of negatieve
hele getallen) levert altijd een \omterm{def:g10:numbers:sets}{geheel getal}.

\emph{2. Onwaar:} $\frac{1}{2}$ is een quotiënt van de \omterm{def:g10:numbers:sets}{gehele getallen}
$1$ en $2$ en is geen \omterm{def:g10:numbers:sets}{geheel getal}.

\emph{3. Waar:} $\frac pq + \frac{p'}{q'} = \frac{pq' + p'q}{qq'}$ is
opnieuw een quotiënt van \omterm{def:g10:numbers:sets}{gehele getallen} (met niet-nul noemer).

\emph{4. Waar:} stel $r$ rationaal, $t$ \omterm{ex:g10:numbers:classify}{irrationaal}, en $r + t = s$
rationaal. Dan zou $t = s - r$ een verschil van twee rationalen zijn,
dus rationaal (via 3, met $-r$) --- tegenspraak. Dus $r + t$ is
\omterm{ex:g10:numbers:classify}{irrationaal}.
\end{solution}

\begin{solution}{exo:g10:numbers:9}
Vermenigvuldigen van $1.414 \leq \sqrt2 \leq 1.415$ met $2 > 0$:
$2.828 \leq 2\sqrt2 \leq 2.830$.

Optellen van $3$: $4.414 \leq \sqrt2 + 3 \leq 4.415$.

Vermenigvuldigen met $-1 < 0$ keert de ongelijkheden om:
$-1.415 \leq -\sqrt2 \leq -1.414$.
\end{solution}

\begin{solution}{exo:g10:numbers:10}
Eerst het hulpfeit. Deel $p$ door $3$: de rest is $0$, $1$ of $2$, d.w.z.\
$p = 3k$, $p = 3k+1$ of $p = 3k+2$. Kwadrateren:
\[
(3k)^2 = 3(3k^2), \quad
(3k+1)^2 = 3(3k^2 + 2k) + 1, \quad
(3k+2)^2 = 3(3k^2 + 4k + 1) + 1 .
\]
Alleen de eerste is een veelvoud van $3$: als $p^2$ een veelvoud van $3$
is, dan is $p$ dat ook.

Stel nu $\sqrt3 = \frac pq$ volledig vereenvoudigd. Kwadrateren geeft
$p^2 = 3q^2$, dus $p^2$ is een veelvoud van $3$, dus $p = 3k$. Dan
$9k^2 = 3q^2$, dus $q^2 = 3k^2$ en $q$ is ook een veelvoud van $3$ ---
maar dan was de breuk $\frac pq$ niet volledig vereenvoudigd:
tegenspraak. Dus $\sqrt3$ is \omterm{ex:g10:numbers:classify}{irrationaal}.
\end{solution}
