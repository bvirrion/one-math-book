\section*{Hoofdstuk \ref{ch:g10:coordgeom} --- Analytische meetkunde}

\begin{solution}{exo:g10:coordgeom:1}
\omterm{prop:g10:coordgeom:midpoint}{Midden}: $\left(\frac{2 + (-4)}{2}, \frac{5 + 1}{2}\right) = (-1, 3)$.
Afstand:
\[
AB = \sqrt{(-4 - 2)^2 + (1 - 5)^2} = \sqrt{36 + 16} = \sqrt{52}
= 2\sqrt{13}.
\]
\end{solution}

\begin{solution}{exo:g10:coordgeom:2}
$I$ is het \omterm{prop:g10:coordgeom:midpoint}{midden} van $[AB]$ betekent $\frac{3 + x_B}{2} = 1$ en
$\frac{-2 + y_B}{2} = 2$, dus $x_B = -1$ en $y_B = 6$: $B(-1, 6)$.
\end{solution}

\begin{solution}{exo:g10:coordgeom:3}
$OP = \sqrt{16 + 1} = \sqrt{17}$; $OQ = \sqrt{9 + 4} = \sqrt{13}$;
$OR = \sqrt{0 + 25} = 5 = \sqrt{25}$. Omdat $13 < 17 < 25$ en de
vierkantswortel \omterm{def:g10:functions:variations}{stijgend} is, ligt $Q$ het dichtst bij de oorsprong.
\end{solution}

\begin{solution}{exo:g10:coordgeom:4}
Gekwadrateerde lengtes:
\begin{align*}
AB^2 &= (5-1)^2 + (4-2)^2 = 16 + 4 = 20, \\
BC^2 &= (7-5)^2 + (0-4)^2 = 4 + 16 = 20, \\
AC^2 &= (7-1)^2 + (0-2)^2 = 36 + 4 = 40 .
\end{align*}
$AB = BC$: de driehoek is gelijkbenig in $B$. Bovendien
$AB^2 + BC^2 = 40 = AC^2$: volgens de omgekeerde van de stelling van
Pythagoras is hij ook rechthoekig in $B$.
\end{solution}

\begin{solution}{exo:g10:coordgeom:5}
\emph{1.} \omterm{prop:g10:coordgeom:midpoint}{Midden} van $[AC]$:
$\left(\frac{-2+4}{2}, \frac{1+1}{2}\right) = (1, 1)$; \omterm{prop:g10:coordgeom:midpoint}{midden} van
$[BD]$: $\left(\frac{1+1}{2}, \frac{3+(-1)}{2}\right) = (1, 1)$.
Zelfde \omterm{prop:g10:coordgeom:midpoint}{midden}: $ABCD$ is een parallellogram.

\emph{2.} $AB^2 = 3^2 + 2^2 = 13$ en $BC^2 = 3^2 + (-4)^2 = 25$:
$AB \neq BC$, geen ruit.

\emph{3.} $AC^2 = 6^2 + 0^2 = 36$ en $BD^2 = 0^2 + (-4)^2 = 16$:
$AC \neq BD$, geen rechthoek. $ABCD$ is een gewoon parallellogram.
\end{solution}

\begin{solution}{exo:g10:coordgeom:6}
Bereken gekwadrateerde afstanden tot $\Omega(2,1)$ en vergelijk met
$5^2 = 25$:
\[
\Omega A^2 = 3^2 + 4^2 = 25, \qquad
\Omega B^2 = (-3)^2 + (-4)^2 = 25, \qquad
\Omega C^2 = 4^2 + 1^2 = 17 .
\]
$A$ en $B$ liggen op de cirkel; $C$ ligt erin ($17 < 25$).
\end{solution}

\begin{solution}{exo:g10:coordgeom:7}
$MA^2 = (x - 1)^2 + 1$ en $MB^2 = (x - 7)^2 + 25$. Stel
$MA^2 = MB^2$:
\[
x^2 - 2x + 2 = x^2 - 14x + 74
\ \Longleftrightarrow\
12x = 72
\ \Longleftrightarrow\
x = 6 .
\]
Het unieke zulke punt is $M(6, 0)$.
\end{solution}

\begin{solution}{exo:g10:coordgeom:8}
\emph{1.} $I = \left(\frac{0+4}{2}, \frac{3+1}{2}\right) = (2, 2)$.

\emph{2.} $M(2,2)$ is inderdaad $I$. En
$MA^2 = (0-2)^2 + (3-2)^2 = 5$, $MB^2 = (4-2)^2 + (1-2)^2 = 5$: het
\omterm{prop:g10:coordgeom:midpoint}{midden} ligt op gelijke afstand van de twee eindpunten, zoals verwacht.

\emph{3.} Zoek $N(0, y)$ met $NA^2 = NB^2$:
\[
(3 - y)^2 = 16 + (1 - y)^2
\ \Longleftrightarrow\
9 - 6y = 16 + 1 - 2y
\ \Longleftrightarrow\
-4y = 8,
\]
dus $y = -2$: het punt $N(0, -2)$.
\end{solution}

\begin{solution}{exo:g10:coordgeom:9}
\begin{align*}
AB &= \sqrt{4^2 + 2^2} = \sqrt{20} = 2\sqrt5, \\
BC &= \sqrt{8^2 + 4^2} = \sqrt{80} = 4\sqrt5, \\
AC &= \sqrt{12^2 + 6^2} = \sqrt{180} = 6\sqrt5 .
\end{align*}
Dan $AB + BC = 2\sqrt5 + 4\sqrt5 = 6\sqrt5 = AC$: de driehoeksongelijkheid
is een gelijkheid, dus $B$ ligt op het lijnstuk $[AC]$ --- de drie
punten zijn collineair.
\end{solution}

\begin{solution}{exo:g10:coordgeom:10}
Het \omterm{prop:g10:coordgeom:midpoint}{midden} van $[AB]$ is $I\left(\frac a2, \frac b2\right)$. Dan
\[
OI^2 = \frac{a^2}{4} + \frac{b^2}{4},
\qquad
IA^2 = \left(a - \frac a2\right)^2 + \left(0 - \frac b2\right)^2
= \frac{a^2}{4} + \frac{b^2}{4},
\]
en $IB^2 = \left(0 - \frac a2\right)^2 + \left(b - \frac b2\right)^2 =
\frac{a^2}{4} + \frac{b^2}{4}$ eveneens. De drie gekwadrateerde afstanden
zijn gelijk, dus $OI = IA = IB$: het \omterm{prop:g10:coordgeom:midpoint}{midden} van de schuine zijde van de
rechthoekige driehoek $OAB$ ligt op gelijke afstand van alle drie
hoekpunten (het is het middelpunt van de omgeschreven cirkel).
\end{solution}
