\section*{Hoofdstuk \ref{ch:g10:lines} --- Vergelijkingen van lijnen en lineaire stelsels}

\begin{solution}{exo:g10:lines:1}
$y = 2x - 3$: \omterm{def:g10:reffunc:affine}{richtingscoëfficiënt} $2$, snijpunt $-3$.
$y = -\frac13 x + 1$: \omterm{def:g10:reffunc:affine}{richtingscoëfficiënt} $-\frac13$, snijpunt $1$.
$y = 4$: \omterm{def:g10:reffunc:affine}{richtingscoëfficiënt} $0$, snijpunt $4$ (horizontale lijn).
$x = -2$: verticale lijn, geen \omterm{def:g10:reffunc:affine}{richtingscoëfficiënt} en geen
\omterm{thm:g10:lines:reduced}{gereduceerde vergelijking}.
\end{solution}

\begin{solution}{exo:g10:lines:2}
$3 \times 2 + 1 = 7$: ja, $A$ ligt op de lijn.
$3 \times (-1) + 1 = -2 \neq -1$: $B$ niet.
$3 \times 4 + 1 = 13$: ja, $C$ ligt op de lijn.
\end{solution}

\begin{solution}{exo:g10:lines:3}
(a) $m = \frac{8 - 2}{3 - 0} = 2$ en $p = y_A = 2$ (het punt $A$ ligt op
de $y$-as): $y = 2x + 2$.

(b) $m = \frac{-4 - 5}{2 - (-1)} = \frac{-9}{3} = -3$; dan
$5 = -3 \times (-1) + p$ geeft $p = 2$: $y = -3x + 2$.

(c) $x_A = x_B = 4$: verticale lijn $x = 4$.
\end{solution}

\begin{solution}{exo:g10:lines:4}
$y = \frac{6x+2}{2} = 3x + 1$. De lijnen met \omterm{def:g10:reffunc:affine}{richtingscoëfficiënt} $3$
zijn $y = 3x - 1$, $y = 3x + 5$ en $y = 3x + 1$: die drie zijn
evenwijdig met elkaar; geen twee zijn gelijk (verschillende snijpunten),
en geen is evenwijdig met $y = -3x + 2$.
\end{solution}

\begin{solution}{exo:g10:lines:5}
Eerste stelsel: substitueer $y = 2x - 1$ in $3x + y = 9$:
$3x + 2x - 1 = 9$, dus $5x = 10$, $x = 2$, dan $y = 3$. Oplossing:
$(2, 3)$.

Tweede stelsel: uit $x - y = 4$, $x = y + 4$. Dan
$2(y + 4) + 3y = 3$, dus $5y = -5$, $y = -1$, dan $x = 3$. Oplossing:
$(3, -1)$.
\end{solution}

\begin{solution}{exo:g10:lines:6}
Eerste stelsel: optellen elimineert $y$: $2x = 14$, $x = 7$;
dan $7 + y = 10$ geeft $y = 3$. Oplossing: $(7, 3)$.

Tweede stelsel: vermenigvuldig de eerste \omterm{def:g10:algebra:equation}{vergelijking} met $5$ en de
tweede met $-2$:
\[
\begin{cases}
15x + 10y = 60 \\
-4x - 10y = -16
\end{cases}
\]
Optellen: $11x = 44$, dus $x = 4$; dan $3 \times 4 + 2y = 12$ geeft
$y = 0$. Oplossing: $(4, 0)$.
\end{solution}

\begin{solution}{exo:g10:lines:7}
Bereken $ab' - a'b$ in elk geval.

Eerste: $2 \times 2 - (-4) \times (-1) = 4 - 4 = 0$, en de tweede
\omterm{def:g10:algebra:equation}{vergelijking} is $-2$ maal de eerste: twee keer dezelfde lijn, oneindig
veel oplossingen.

Tweede: $ab' - a'b = 0$ opnieuw, maar de rechterleden staan niet in de
verhouding $-2$ ($1 \neq -6$): twee verschillende evenwijdige lijnen,
geen oplossing.

Derde: $2 \times 1 - 1 \times (-1) = 3 \neq 0$: precies één oplossing.
\end{solution}

\begin{solution}{exo:g10:lines:8}
Laat $x$ het aantal kindertickets zijn en $y$ het aantal
volwassenentickets:
\[
\begin{cases}
x + y = 200 \\
5x + 9y = 1352 .
\end{cases}
\]
Uit de eerste \omterm{def:g10:algebra:equation}{vergelijking} $x = 200 - y$; substitueren,
$5(200 - y) + 9y = 1352$, dus $1000 + 4y = 1352$, $y = 88$, dan
$x = 112$. Dus $112$ kindertickets en $88$ volwassenentickets. Controle:
$5 \times 112 + 9 \times 88 = 560 + 792 = 1352$.
\end{solution}

\begin{solution}{exo:g10:lines:9}
\emph{1.} Evenwijdig betekent dezelfde \omterm{def:g10:reffunc:affine}{richtingscoëfficiënt} $2$:
$y = 2x + p$ met $4 = 2 \times 1 + p$, dus $p = 2$:
$d'\colon y = 2x + 2$.

\emph{2.} Op de $x$-as, $y = 0$: $2x + 2 = 0$ geeft $x = -1$. Het
snijpunt is $(-1, 0)$.
\end{solution}

\begin{solution}{exo:g10:lines:10}
\emph{1.} \omterm{prop:g10:coordgeom:midpoint}{Midden} van $[BC]$: $A' = \left(\frac{6+2}{2},
\frac{2+4}{2}\right) = (4, 3)$. De zwaartelijn vanuit $A(0,0)$ heeft
\omterm{def:g10:reffunc:affine}{richtingscoëfficiënt} $\frac{3 - 0}{4 - 0} = \frac34$: \omterm{def:g10:algebra:equation}{vergelijking}
$y = \frac34 x$.

\omterm{prop:g10:coordgeom:midpoint}{Midden} van $[AC]$: $B' = (1, 2)$. De zwaartelijn vanuit $B(6, 2)$ heeft
\omterm{def:g10:reffunc:affine}{richtingscoëfficiënt} $\frac{2 - 2}{1 - 6} = 0$: het is de horizontale
lijn $y = 2$.

\emph{2.} Snijpunt: $\frac34 x = 2$ geeft $x = \frac83$, dus
$G\left(\frac83, 2\right)$. De derde zwaartelijn verbindt $C(2,4)$ met
het \omterm{prop:g10:coordgeom:midpoint}{midden} van $[AB]$, $C' = (3, 1)$; haar \omterm{def:g10:reffunc:affine}{richtingscoëfficiënt} is
$\frac{1 - 4}{3 - 2} = -3$, \omterm{def:g10:algebra:equation}{vergelijking} $y = -3x + 10$. Bij
$x = \frac83$: $y = -8 + 10 = 2$: $G$ ligt erop. En inderdaad
$G = \left(\frac{0 + 6 + 2}{3}, \frac{0 + 2 + 4}{3}\right)$: het
zwaartepunt middelelt de \omterm{def:g10:coordgeom:system}{coördinaten} van de hoekpunten.
\end{solution}
