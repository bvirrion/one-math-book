\begin{solution}{exo:g10:functions:3}

\begin{solution}{exo:g10:functions:5}
\emph{1.} El valor más grande en la tabla es $4 = g(3)$ (\omterm{def:g10:functions:extrema}{máximo}), el
el más pequeño es $-2 = g(0)$ (\omterm{def:g10:functions:extrema}{mínimo}).

\emph{2.} En $\intcc{-3}{0}$ el \omterm{def:g10:functions:function}{función} disminuye, y
$-1 < -0.5$, entonces $g(-1) \geq g(-0.5)$.

\emph{3.} En cada \omterm{def:g10:numbers:interval}{intervalo} de monotonicidad, $g$ toma cada valor como máximo
una vez, por lo que $g(x) = 0$ tiene como máximo una solución por \omterm{def:g10:numbers:interval}{intervalo}: como máximo $3$
soluciones en total. (Aquí $0$ se encuentra entre $-2$ y $1$, entre $-2$ y
$4$, y entre $0$ y $4$, por lo que hay exactamente tres).
\end{solution}

\begin{solution}{exo:g10:functions:6}
Deje $u < v$. entonces
\[
f(v) - f(u) = (-2v + 7) - (-2u + 7) = -2(v - u) < 0,
\]
desde $v - u > 0$. Entonces $f(v) < f(u)$: $f$ es estrictamente \omterm{def:g10:functions:variations}{decreciente} en $\R$.
\end{solution}

\begin{solution}{exo:g10:functions:7}
Completa el cuadrado: $(x+3)^2 = x^2 + 6x + 9$, entonces
\[
f(x) = x^2 + 6x + 11 = (x + 3)^2 + 2 .
\]
Por cada $x$, $(x+3)^2 \geq 0$, entonces $f(x) \geq 2$; y
$f(-3) = 0 + 2 = 2$. El \omterm{def:g10:functions:extrema}{mínimo} de $f$ es $2$, obtenido en $x = -3$.
\end{solution}

\begin{solution}{exo:g10:functions:8}

A(x) = x(20 - x).
\]
Ambas dimensiones deben ser positivas: $0 < x < 20$.

\emph{2.} $A(4) = 64$, $A(8) = 96$, $A(10) = 100$, $A(12) = 96$,

\begin{solution}{exo:g10:functions:9}
\emph{1.} Por cada $x$: $x^2 + 1 \geq 1 > 0$, entonces $f(x) = \frac{1}{x^2+1}$
es positivo, y desde $x^2 + 1 \geq 1$, tomando inversas (ambos lados
positivo) da $f(x) \leq 1$.

\emph{2.} \omterm{def:g10:functions:extrema}{Máximo}: $f(0) = 1$ y $f(x) \leq 1$ para todos los $x$, por lo que $1$ es el
\omterm{def:g10:functions:extrema}{máximo}, obtenido en $0$. \omterm{def:g10:functions:extrema}{Mínimo}: $f(x) > 0$ siempre, pero sin valor de $x$
logra $0$ (un cociente de números distintos de cero es distinto de cero), y $f$ toma
valores tan cercanos a $0$ como queramos para $x$ grande; entonces $f$ no tiene \omterm{def:g10:functions:extrema}{mínimo}.
\end{solution}

\begin{solution}{exo:g10:functions:10}
\emph{1.} $f(v) - f(u) = v^2 - u^2 = (v-u)(v+u)$. Si $0 \leq u < v$, entonces
$v - u > 0$ y $v + u > 0$ (suma de un número positivo y no negativo),
entonces $f(v) - f(u) > 0$: $f$ es estrictamente \omterm{def:g10:functions:variations}{creciente} en $\intco{0}{+\infty}$.

\emph{2.} Si $u < v \leq 0$, entonces $v - u > 0$ todavía, pero ahora
$v + u < 0$ (suma de un número no positivo y uno negativo), entonces
$f(v) - f(u) < 0$: $f$ es estrictamente \omterm{def:g10:functions:variations}{decreciente} en $\intoc{-\infty}{0}$.
\end{solution}

\begin{solution}{pb:g10:functions:1}
\textbf{1.} \omterm{def:g10:functions:function}{Dominio}: $x \neq 0$ (división por $x$). $f(1) =

$f\!\left(\frac32\right) = \frac12\left(\frac32 +
\frac43\right) = \frac{17}{12}$.

\textbf{2.} $\frac12\left(x + \frac2x\right) = \frac32$ da

\omterm{def:g10:functions:function}{preimágenes} de $\frac32$ son $1$ y $2$ --- como pregunta 1

\textbf{3.} $f(x) = x$ da $x + \frac2x = 2x$, entonces

El positivo es la diagonal del cuadrado unitario, \omterm{ex:g10:numbers:classify}{irracional}
por el problema de irracionalidad del fin de semana del volumen de la escuela secundaria.

\textbf{6.}$f(v) - f(u) = \frac{v - u}{2} + \frac1v - \frac1u
= \frac{v - u}{2} + \frac{u - v}{uv}
= (v - u)\left(\frac12 - \frac{1}{uv}\right)
= \frac{(v - u)(uv - 2)}{2uv}$. Para $0 < u < v \leq \sqrt2$:
$uv < 2$, el corchete es negativo, $f(v) < f(u)$: \omterm{def:g10:functions:variations}{decreciente}.
Para $\sqrt2 \leq u < v$: $uv > 2$: \omterm{def:g10:functions:variations}{creciente}. Punto de inflexión:
$\sqrt2$.

\textbf{7.} \omterm{def:g10:functions:variations}{Decreciente} antes de $\sqrt2$, \omterm{def:g10:functions:variations}{creciente} después: $f$
alcanza su \omterm{def:g10:functions:extrema}{mínimo} en $\sqrt2$, valor
$f(\sqrt2) = \frac12\left(\sqrt2 + \frac{2}{\sqrt2}\right)
= \sqrt2$. Consecuencia: cada salida de la máquina (positiva
entrada) es al menos $\sqrt2$ --- después de una vuelta de la manivela,
Las conjeturas viven en $\intco{\sqrt2}{+\infty}$.

\textbf{8.} Si $x > \sqrt2$, entonces $x^2 > 2$, entonces
$\frac2x < x$ y el promedio $f(x)$ de $x$ y $\frac2x$ es
menos de $x$; y $f(x) \geq \sqrt2$ por la pregunta 7, con
igualdad sólo en $x = \sqrt2$. entonces
$\sqrt2 < f(x) < x$: cada iteración mejora estrictamente sin
sobrepasando --- la secuencia de la pregunta 4 se desliza hacia abajo

\textbf{9.} En $\intoo{0}{+\infty}$: \omterm{def:g10:functions:variations}{decreciente} de
$+\infty$ (cerca de $0$) hasta \omterm{def:g10:functions:extrema}{mínimo} $\sqrt2$ en
$x = \sqrt2$, luego \omterm{def:g10:functions:variations}{creciente} sin límite.

\textbf{10.} $g(x) = x$ da $x^2 = x$, $x(x - 1) = 0$: \omterm{pb:g10:functions:1}{fijo
puntos} $0$ y $1$. De $0.9$: $0.81$, $0.656$, $0.430$,
$0.185$ --- deslizándose hacia $0$. De $1.1$: $1.21$, $1.464$,
$2.144$, $4.595$ --- huyendo al infinito. El \omterm{pb:g10:functions:1}{punto fijo} $0$
atrae, $1$ repele: un pelo de diferencia al principio, opuesto
destinos.

\textbf{11.} $7 \to 22 \to 11 \to 34 \to 17 \to 52 \to 26 \to
13 \to 40 \to 20 \to 10 \to 5 \to 16 \to 8 \to 4 \to 2 \to 1$:
dieciséis pasos, alcanzando un máximo en $52$.

\textbf{12.} $27 \to 82 \to 41 \to 124 \to 62 \to 31 \to 94 \to
47 \to 142 \to 71 \to 214 \to \dots$ --- sigue subiendo después
diez pasos, en camino a un pico de $9\,232$ y un paso $111$
vuelo. Lección: una regla de dos líneas puede producir un comportamiento que nadie puede
predecir de un vistazo --- números de al lado ($26$, $27$, $28$)
tienen vuelos tremendamente diferentes.

\textbf{13.} Una potencia de $2$ reduce a la mitad toda su escalera:
$2^k \to 2^{k-1} \to \dots \to 2 \to 1$. En la parte inferior,
$h(1) = 4$, $h(4) = 2$, $h(2) = 1$: el vuelo entra en ciclo
$4 \to 2 \to 1 \to 4$. \omterm{pb:g10:functions:1}{Puntos fijos} necesitaría $\frac n2 = n$
(sólo $n = 0$) o $3n + 1 = n$ (negativo): ninguno entre los
positivo \omterm{def:g10:numbers:sets}{números enteros} --- los vuelos observados terminan no en un \omterm{pb:g10:functions:1}{fijo
punto} sino en ese pequeño \emph{ciclo}.

\textbf{16.} $\sqrt{x - 3}$: necesita $x - 3 \geq 0$:
$\intco{3}{+\infty}$. $\frac{1}{x^2 - 9}$: necesidades
$x \neq \pm 3$: $\R$ privado de $-3$ y $3$.

\textbf{17.} $f(x) = 3$ da $x + \frac2x = 6$, entonces
$x^2 - 6x + 2 = 0$, es decir\ $(x - 3)^2 = 7$:
$x = 3 - \sqrt7$ o $x = 3 + \sqrt7$ (ambos positivos: dos exactos

\textbf{18.} Sentido físico mientras la piedra está en el aire:
$H(t) = 5t(4 - t) \geq 0$ para $t \in \intcc{0}{4}$. Completando:
$H(t) = 20 - 5(t - 2)^2$: \omterm{def:g10:functions:extrema}{máximo} $20$ m en $t = 2$ s.
Variaciones: \omterm{def:g10:functions:variations}{creciente} en $\intcc{0}{2}$ de $0$ a $20$,
\omterm{def:g10:functions:variations}{decreciente} en $\intcc{2}{4}$ de regreso a $0$.

\textbf{19.} $20t - 5t^2 \geq 15 \iff t^2 - 4t + 3 \leq 0 \iff
(t - 1)(t - 3) \leq 0 \iff t \in \intcc{1}{3}$: dos completos
segundos por encima de $15$ m.

\textbf{20.} Cada máquina tiene un \omterm{def:g10:functions:function}{dominio} (donde la regla es
permitido), una regla (la fórmula o el cambio de paridad) y una
\omterm{def:g10:functions:graph}{gráfico} o tabla que lo muestra completo. De cada uno le preguntamos a los dos
grandes preguntas: \emph{¿A dónde lleva la repetición?} (\omterm{pb:g10:functions:1}{fijo
puntos}, ciclos, atracción --- resuelto para Heron, abierto para
granizo) y \emph{¿Cómo se mueve la salida con la entrada?}
(variaciones, extremos --- cumbre de la piedra, Heron's \omterm{def:g10:functions:extrema}{mínimo}).
Los capítulos siguientes agudizan ambos: referencia \omterm{def:g10:functions:function}{funciones} a continuación,
luego, en años posteriores, derivados para medir la variación y
límites para certificar dónde aterrizan las iteraciones.
