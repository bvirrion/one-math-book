\section*{Capítulo \ref{ch:g10:algebra} --- Álgebra: ecuaciones y desigualdades}

\begin{solution}{exo:g10:algebra:1}
$3(2x-5) - 2(x-7) = 6x - 15 - 2x + 14 = 4x - 1$.

$(x+4)(2x-1) = 2x^2 - x + 8x - 4 = 2x^2 + 7x - 4$.

$(3x-2)^2 = 9x^2 - 12x + 4$.

$(5-x)(5+x) = 25 - x^2$.
\end{solution}

\begin{solution}{exo:g10:algebra:2}
$x^2 - 49 = (x+7)(x-7)$ (diferencia de cuadrados).

$x^2 + 10x + 25 = (x+5)^2$.

$7x^2 - 14x = 7x(x - 2)$ (factor común).

$16x^2 - 8x + 1 = (4x - 1)^2$.
\end{solution}

\begin{solution}{exo:g10:algebra:3}
$4x + 9 = x - 3$: $3x = -12$, so $x = -4$.

$\frac{2x-1}{3} = 5$: $2x - 1 = 15$, so $x = 8$.

$2(x-3) = 2x + 1$ se expande a $2x - 6 = 2x + 1$, es decir\ $-6 = 1$: falso para
cada $x$. No hay solución --- los dos lados definen líneas paralelas que nunca
reunirse.
\end{solution}

\begin{solution}{exo:g10:algebra:4}
$(x+5)(2x-8) = 0$: $x = -5$ or $x = 4$.

$x^2 - 16 = (x-4)(x+4) = 0$: $x = 4$ or $x = -4$.

$x^2 + 6x + 9 = (x+3)^2 = 0$: la solución única $x = -3$.
\end{solution}

\begin{solution}{exo:g10:algebra:5}
$3x - 5 \leq 7$: $3x \leq 12$, entonces $x \leq 4$: conjunto de soluciones
$\intoc{-\infty}{4}$.

$-4x + 1 > 9$: $-4x > 8$, dividir por $-4$ invierte la desigualdad:
$x < -2$: conjunto de soluciones $\intoo{-\infty}{-2}$.

$2(x-1) \geq 5x + 4$: $2x - 2 \geq 5x + 4$, entonces $-3x \geq 6$, entonces
$x \leq -2$: conjunto de soluciones $\intoc{-\infty}{-2}$.
\end{solution}

\begin{solution}{exo:g10:algebra:6}
El factor común es $(x-1)$:
\[
E = (x-1)\bigl[(x+3) - (2x - 5)\bigr] = (x-1)(x + 3 - 2x + 5)
= (x-1)(8 - x).
\]
Luego $E = 0$ cuando $x = 1$ o $x = 8$ (regla de producto cero).
\end{solution}

\begin{solution}{exo:g10:algebra:7}
$x^2 = 5x$: $x^2 - 5x = x(x - 5) = 0$, entonces $x = 0$ o $x = 5$ (nunca
dividir por $x$: se perdería la solución $0$).

$(2x+1)^2 = (x-4)^2$: llevar a la izquierda y factorizar la diferencia de
cuadrados:
\[
(2x+1)^2 - (x-4)^2
= \bigl[(2x+1) + (x-4)\bigr]\bigl[(2x+1) - (x-4)\bigr]
= (3x - 3)(x + 5).
\]
Regla de producto cero: $x = 1$ o $x = -5$.
\end{solution}

\begin{solution}{exo:g10:algebra:8}
$(x+2)(4-x) > 0$: el factor $x + 2$ desaparece en $-2$ (positivo después),
$4 - x$ desaparece en $4$ (positivo antes). El producto es positivo.
exactamente cuando ambos factores son positivos, es decir\ en $\intoo{-2}{4}$.

$\frac{2x-6}{x+1} \geq 0$: numerador cero en $3$, denominador cero en
$-1$ (prohibido). \omterm{def:g10:algebra:signtable}{mesa de firmas}: el cociente es positivo cuando ambas partes
tener el mismo signo, es decir\ para $x < -1$ o $x > 3$; desaparece en $x = 3$.
Conjunto de soluciones: $\intoo{-\infty}{-1} \cup \intco{3}{+\infty}$.
\end{solution}

\begin{solution}{exo:g10:algebra:9}
Después de $x$ meses el primer servicio cuesta $9x + 15$ y el segundo $12x$.
El primero es más barato cuando
\[
9x + 15 < 12x
\ \Longleftrightarrow\
15 < 3x
\ \Longleftrightarrow\
x > 5 .
\]
A partir del mes $6$, el primer servicio es la opción más económica en general.
\end{solution}

\begin{solution}{exo:g10:algebra:10}
Valor prohibido: $x = 1$. Para $x \neq 1$, multiplique ambos lados por $x - 1$:
\[
x + 3 = 2(x - 1) = 2x - 2
\ \Longleftrightarrow\
x = 5 .
\]
Desde $5 \neq 1$, la solución es $x = 5$. Comprobar:
$\frac{5+3}{5-1} = \frac84 = 2$.
\end{solution}

\begin{solution}{exo:g10:algebra:11}
$(n+1)^2 - n^2 = n^2 + 2n + 1 - n^2 = 2n + 1$, que es impar para cada
\omterm{def:g10:numbers:sets}{entero} $n$.

Se pueden escribir dos números impares consecutivos $2n - 1$ y $2n + 1$. entonces
\[
(2n+1)^2 - (2n-1)^2
= \bigl[(2n+1)+(2n-1)\bigr]\bigl[(2n+1)-(2n-1)\bigr]
= 4n \times 2 = 8n,
\]
un múltiplo de $8$.
\end{solution}

\begin{solution}{pb:g10:algebra:1}
\textbf{1.} $x^2 - 10x + 25 = (x - 5)^2$;
$9x^2 - 49 = (3x - 7)(3x + 7)$;
$x^2 + 6x + 9 - 4 = (x + 3)^2 - 2^2 = (x + 1)(x + 5)$.

\textbf{2.} $x = -1$ or $x = -5$; \quad $x = \frac73$ or
$x = -\frac73$.

\textbf{3.} $(5 - t)(5 + t) = 25 - t^2 = 21$ da $t^2 = 4$,
$t = 2$: los números son $3$ y $7$ (suma $10$, producto $21$:
comprobar).

\textbf{4.} $(7 - t)(7 + t) = 49 - t^2 = 33$: $t = 4$, números
$3$ y $11$.

\textbf{5.} $\left(\frac s2 - t\right)\left(\frac s2 + t\right)
= \left(\frac s2\right)^2 - t^2 = p$ fuerzas
$t^2 = \left(\frac s2\right)^2 - p$. Existe una solución exactamente
cuando esto es $\geq 0$, es decir,\ cuando
$p \leq \left(\frac s2\right)^2$ --- el producto de dos números
de una suma dada nunca puede superar el cuadrado de la mitad de la suma
(El problema de la cerca de la reina Dido, en el volumen de la escuela secundaria lo conocía).

\textbf{6.} $(x - a)(x - b) = x^2 - (a + b)x + ab$: el
\omterm{def:g10:algebra:equation}{ecuación} $x^2 - sx + p = 0$ dice precisamente ``$x$ es uno de dos
números con suma $s$ y producto $p$''.

\textbf{7.} $(x-5)^2 - 4 = x^2 - 10x + 25 - 4 = x^2 - 10x +
21$. Diferencia de cuadrados:
$(x - 5 - 2)(x - 5 + 2) = (x - 7)(x - 3) = 0$: soluciones $3$
y $7$ --- los números del escriba de la pregunta 3.

\textbf{8.} $x^2 + 6x - 7 = (x + 3)^2 - 16 = 0$ da
$(x + 3)^2 = 16$, $x + 3 = \pm 4$: $x = 1$ o $x = -7$.

\textbf{9.} $x^2 + 4x + 5 = (x + 2)^2 + 1 \geq 1 > 0$ para cada
$x$: no hay una solución real. En el formulario $(x + h)^2 + k$: soluciones
existe exactamente cuando $k \leq 0$ (entonces $(x+h)^2 = -k$ tiene los dos
raíces $-h \pm \sqrt{-k}$); cuando $k > 0$ la expresión permanece
positivo.

\textbf{10.} El cuadrado del lado $x$ (área $x^2$) con dos
Los rectángulos $3 \times x$ ($6x$ en total) forman una L de área
$x^2 + 6x = 7$; la esquina $3 \times 3$ que falta (área $9$)
completa un gran cuadrado de lado $x + 3$. Agregando $9$ a ambos
lados: $(x + 3)^2 = 16$, entonces $x + 3 = 4$ (las longitudes son positivas)
y $x = 1$. El álgebra y la imagen son el mismo acto.

\textbf{11.} $(x-3)(x+2)$: negativo entre las raíces. Solución
de $< 0$: $x \in \intoo{-2}{3}$.

\textbf{12.} $x^2 - 10x + 21 \leq 0$, es decir\
$(x - 3)(x - 7) \leq 0$: entre las raíces, extremos incluidos:
$x \in \intcc{3}{7}$.

\textbf{13.} Cero en $x = 1$, prohibido en $x = -2$; el
el cociente es positivo fuera de las raíces:
$x \in \intoo{-\infty}{-2} \cup \intco{1}{+\infty}$.

\textbf{14.} $25 - (x - 5)^2 = 25 - x^2 + 10x - 25 =
x(10 - x)$: identidad verificada. Desde $(x-5)^2 \geq 0$ siempre,
$A(x) \leq 25$, con igualdad exactamente cuando $x = 5$: el
$5 \times 5$ bolígrafo cuadrado, máximo $25$ m$^2$ --- el grado 6
tabla, ahora una prueba de dos líneas.

\textbf{15.} Para $x > 0$, multiplique el reclamo por $x$:
$x^2 + 1 \geq 2x$, es decir\ $x^2 - 2x + 1 = (x - 1)^2 \geq 0$ ---
cierto, con igualdad solo en $x = 1$. Dividiendo hacia atrás por $x > 0$
conserva todo (\cref{prop:g10:algebra:ineqrules}).

\textbf{16.} Multiplicación cruzada $\frac1x = \frac{x}{1 - x}$
(todas las longitudes positivas): $x^2 = 1 - x$, es decir\
$x^2 + x - 1 = 0$.

\textbf{17.} $x^2 + x - 1 = \left(x + \frac12\right)^2 -
\frac54 = 0$ da $x + \frac12 = \pm\frac{\sqrt5}{2}$. el
raíz positiva: $x = \frac{\sqrt5 - 1}{2} \approx 0.618$.

\textbf{18.} $\varphi = \frac{2}{\sqrt5 - 1} =
\frac{2(\sqrt5 + 1)}{(\sqrt5 - 1)(\sqrt5 + 1)}
= \frac{2(\sqrt5 + 1)}{5 - 1} = \frac{\sqrt5 + 1}{2}
\approx 1.618$.

\textbf{19.} $\varphi^2 = \frac{(\sqrt5 + 1)^2}{4} =
\frac{6 + 2\sqrt5}{4} = \frac{3 + \sqrt5}{2}$, y
$\varphi + 1 = \frac{\sqrt5 + 3}{2}$: igual. y
$\varphi - 1 = \frac{\sqrt5 - 1}{2} = x = \frac1\varphi$ (que
así se definió $\varphi$). La proporción áurea es el número.
cuyo cuadrado suma uno y cuyo recíproco resta uno.

\textbf{20.} $\sqrt5$ es \omterm{ex:g10:numbers:classify}{irracional} (el problema de irracionalidad del fin de semana del volumen de la escuela secundaria: $5$
no es un cuadrado perfecto), y al agregar $1$ y luego dividir a la mitad se conserva
irracionalidad (operaciones racionales, \cref{pb:g10:numbers:1}).
Relaciones: $\frac53 \approx 1.667$, $\frac85 = 1.600$,
$\frac{13}{8} = 1.625$, $\frac{21}{13} \approx 1.615$:
alternativamente arriba y abajo, acercándose a
$\varphi = 1.618\ldots$ --- los números de conteo de ritmo de
el problema del fin de semana de contar el ritmo del volumen de la escuela secundaria persigue en secreto la sección áurea; el
Los capítulos sobre secuencias los atraparán en el acto.
\end{solution}
