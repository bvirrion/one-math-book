\begin{solution}{exo:g10:vectors:1}
$\vect{AB} + \vect{BD} = \vect{AD}$ (Chasles).

$\vect{KL} + \vect{LM} + \vect{MK} = \vect{KK} = \vec 0$.

$\vect{AC} - \vect{BC} = \vect{AC} + \vect{CB} = \vect{AB}$.

$\vect{AB} + \vect{CA} = \vect{CA} + \vect{AB} = \vect{CB}$.
\end{solution}

\begin{solution}{exo:g10:vectors:3}

$3\vec u = (6, -9)$.

$2\vec u - \vec v = (4 - (-1),\ -6 - 4) = (5, -10)$.

\end{solution}

\begin{solution}{exo:g10:vectors:4}
(a) $4 \times 9 - 6 \times 6 = 36 - 36 = 0$: \omterm{def:g10:vectors:collinear}{colineal}
($\vec v = \frac32 \vec u$).

(b) $2 \times 10 - (-5) \times (-4) = 20 - 20 = 0$: \omterm{def:g10:vectors:collinear}{colineal}
($\vec v = -2\vec u$).

(c) $3 \times 3 - 1 \times 1 = 8 \neq 0$: no \omterm{def:g10:vectors:collinear}{colineal}.
\end{solution}

\begin{solution}{exo:g10:vectors:5}
$\vect{AB}\,(4, -2)$ y $\vect{DC}\,(6 - x_D,\ 3 - y_D)$. la condición
$\vect{AB} = \vect{DC}$ da $6 - x_D = 4$ y $3 - y_D = -2$, por lo que

\end{solution}

\begin{solution}{exo:g10:vectors:6}
\emph{1.} $\vect{AB}\,(3, 2)$ y $\vect{AC}\,(6, -2)$, entonces
$\vect{AB} + \vect{AC} = (9, 0)$ y
$M = (-2 + 9,\ 1 + 0) = (7, 1)$.

\emph{2.} Por la regla del paralelogramo, $\vect{AM} = \vect{AB} + \vect{AC}$
significa que $ABMC$ es un paralelogramo ($M$ es el cuarto vértice opuesto
$A$). Se puede comprobar: $\vect{AB}\,(3,2)$ y $\vect{CM}\,(7-4,\ 1-(-1)) =
(3, 2)$ son iguales.
\end{solution}

\begin{solution}{exo:g10:vectors:7}
Primer triple: $\vect{AB}\,(3, 2)$, $\vect{AC}\,(9, 6)$;
$3 \times 6 - 2 \times 9 = 0$: alineado ($\vect{AC} = 3\vect{AB}$).

Segundo triple: $\vect{AB}\,(2, -1)$, $\vect{AC}\,(8, -4)$;
$2 \times (-4) - (-1) \times 8 = -8 + 8 = 0$: alineado también
($\vect{AC} = 4\vect{AB}$).
\end{solution}

\begin{solution}{exo:g10:vectors:8}
$\vect{AB}\,(4, 2)$ y $\vect{DC}\,(6-2,\ 1-(-1)) = (4, 2)$: los dos
\omterm{def:g10:vectors:vector}{vectores} son \emph{igual}, por lo que $ABCD$ es un paralelogramo y, en particular,
$(AB) \parallel (CD)$. La propiedad del paralelogramo es la más fuerte:
implica el paralelismo (colinealidad del \omterm{def:g10:vectors:vector}{vectores}), no al revés.
\end{solution}

\begin{solution}{exo:g10:vectors:9}
\emph{1.} $\vect{AB}\,(6, 3)$, entonces
$\vect{AI} = \frac23\vect{AB} = (4, 2)$ y $I(4, 2)$.

\emph{2.} $\vect{CB}\,(6-2,\ 3-5) = (4, -2)$, entonces
$\vect{CJ} = \frac23\vect{CB} = \left(\frac83, -\frac43\right)$ y
$J\left(2 + \frac83,\ 5 - \frac43\right) = \left(\frac{14}{3},
\frac{11}{3}\right)$.

\emph{3.} $\vect{IJ}\,\left(\frac{14}{3} - 4,\ \frac{11}{3} - 2\right) =
\left(\frac23, \frac53\right)$ y $\vect{AC}\,(2, 5)$. Colinealidad
criterio:
\[
\frac23 \times 5 - \frac53 \times 2 = \frac{10}{3} - \frac{10}{3} = 0 :
\]
los \omterm{def:g10:vectors:vector}{vectores} son \omterm{def:g10:vectors:collinear}{colineal}, por lo que $(IJ)$ es paralelo a $(AC)$. (Tanto $I$ como
$J$ siéntate a un tercio del camino desde el costado hacia $B$: el pequeño
El triángulo $IBJ$ es una reducción de $ABC$).
\end{solution}

\begin{solution}{exo:g10:vectors:10}
Los \omterm{prop:g10:coordgeom:midpoint}{puntos medios} son
\[
I\left(\tfrac{x_A + x_B}{2}, \tfrac{y_A + y_B}{2}\right), \quad
J\left(\tfrac{x_B + x_C}{2}, \tfrac{y_B + y_C}{2}\right), \quad
K\left(\tfrac{x_C + x_D}{2}, \tfrac{y_C + y_D}{2}\right), \quad
L\left(\tfrac{x_D + x_A}{2}, \tfrac{y_D + y_A}{2}\right).
\]
Primer \omterm{def:g10:coordgeom:system}{coordenadas} de los dos \omterm{def:g10:vectors:vector}{vectores}:
\[
x_J - x_I = \frac{x_B + x_C}{2} - \frac{x_A + x_B}{2}
= \frac{x_C - x_A}{2},
\qquad
x_K - x_L = \frac{x_C + x_D}{2} - \frac{x_D + x_A}{2}
= \frac{x_C - x_A}{2},
\]
y el mismo cálculo funciona para el segundo \omterm{def:g10:coordgeom:system}{coordenadas}. entonces
$\vect{IJ} = \vect{LK}$ (ambos iguales a $\frac12\vect{AC}$) y $IJKL$ es un
paralelogramo, cualquiera que sea el aspecto del cuadrilátero $ABCD$.
\end{solution}

\begin{solution}{pb:g10:vectors:1}
\textbf{1.}$\vect{AB} + \vect{BC} + \vect{CD} = \vect{AD}$;
\quad$\vect{MA} - \vect{MB} = \vect{MA} + \vect{BM} =
\vect{BA}$; \quad $\vect{AB} + \vect{BC} + \vect{CD} +
\vect{DA} = \vect{AA} = \vec 0$ (reordenando el tercer \omterm{def:g10:vectors:sum}{suma}).

\textbf{2.} $\vect{OB} - \vect{OA} = \vect{AO} + \vect{OB} =
\vect{AB}$ de Chasles.

\textbf{3.} $I$ es el \omterm{prop:g10:coordgeom:midpoint}{punto medio} de $[AB]$ exactamente cuando
$\vect{IA} = -\vect{IB}$, es decir\
$\vect{IA} + \vect{IB} = \vec 0$. Luego, para cualquier $O$:
$\vec 0 = \vect{IO} + \vect{OA} + \vect{IO} + \vect{OB}$, entonces
$2\,\vect{OI} = \vect{OA} + \vect{OB}$ --- y viceversa.

\textbf{4.} $\vect{AB} = (4, 2)$ y $D$ satisfacen
$\vect{DC} = \vect{AB}$: $D = C - (4, 2) = (2, -2)$ --- el
mismo $D$ que \cref{pb:g10:coordgeom:1}, una línea más rápida.

\textbf{5.} La traducción de \omterm{def:g10:vectors:vector}{vector} $\vec u$ --- exactamente la
transformación producida por dos espejos paralelos o dos
medias vueltas. \omterm{def:g10:functions:function}{Imagen} de $(1, 2)$: $(4, 1)$.

\textbf{6.} \omterm{def:g10:algebra:expand}{En expansión} cada $\vect{GX}$ como
$\vect{GO} + \vect{OX}$:
$3\,\vect{GO} + \vect{OA} + \vect{OB} + \vect{OC} = \vec 0$,
entonces $\vect{OG} = \frac13(\vect{OA} + \vect{OB} + \vect{OC})$:
uno y sólo uno de esos puntos, cuyos \omterm{def:g10:coordgeom:system}{coordenadas} son los promedios
de los tres vértices'.

\textbf{7.} $\vect{AB} + \vect{AC} =
(\vect{AG} + \vect{GB}) + (\vect{AG} + \vect{GC})
= 2\vect{AG} + (\vect{GB} + \vect{GC})
= 2\vect{AG} - \vect{GA} = 3\vect{AG}$, de donde
$\vect{AG} = \frac13(\vect{AB} + \vect{AC})$. El \omterm{prop:g10:coordgeom:midpoint}{punto medio}
La fórmula (pregunta 3, vista desde $A$) da
$\vect{AK} = \frac12(\vect{AB} + \vect{AC})$. Comparando:
$\vect{AG} = \frac23\vect{AK}$ --- $G$ se encuentra en la mediana
$[AK]$, a dos tercios del camino desde $A$.

\textbf{8.} La condición definitoria trata $A$, $B$, $C$
idénticamente, por lo que el mismo cálculo de $B$ o de $C$
coloca el \emph{mismo} $G$ en dos tercios de esas medianas:
la concurrencia no cuesta nada. La prueba del paralelogramo fue
ingenioso pero ad hoc; \omterm{def:g10:coordgeom:system}{coordenadas} cálculo pero oscuro;
\omterm{def:g10:vectors:vector}{vectores} hace visible la simetría y la prueba de dos líneas.

\textbf{9.}$G\left(\frac{1+5+3}{3}, \frac{1+2+6}{3}\right) =
(3, 3)$; $K(4, 4)$; $\vect{AG} = (2, 2)$ y
$\vect{AK} = (3, 3)$: determinante
$2 \times 3 - 2 \times 3 = 0$, \omterm{def:g10:vectors:collinear}{colineal} y, de hecho,
$\vect{AG} = \frac23\vect{AK}$.

\textbf{10.} $G'\left(\frac{2 + 5 + 3}{4}, \frac{2 + 2 +
6}{4}\right) = (2.5,\ 2.5)$: tirado de $G(3,3)$ hacia el
vértice duplicado $A(1,1)$, como debe ser un punto de equilibrio.

\textbf{11.} $3 \times 4 - (-2)(-6) = 12 - 12 = 0$: \omterm{def:g10:vectors:collinear}{colineal}
($\vec v = -2\vec u$). $2 \times 9 - 5 \times 4 = -2 \neq 0$:
no \omterm{def:g10:vectors:collinear}{colineal}.

\textbf{12.} $\vect{AB} = (2, 3)$, $\vect{AC} = (6, 9)$:
determinante $2 \times 9 - 3 \times 6 = 0$: alineado.

\textbf{13.} $\vect{OI} = \frac12(\vect{OA} + \vect{OB})$ y
$\vect{OK} = \frac12(\vect{OC} + \vect{OD})$, entonces el \omterm{prop:g10:coordgeom:midpoint}{punto medio}
de $[IK]$ tiene la posición \omterm{def:g10:vectors:vector}{vector}
$\frac12(\vect{OI} + \vect{OK}) = \frac14(\vect{OA} +
\vect{OB} + \vect{OC} + \vect{OD})$ --- y el cálculo
para $[JL]$ da el mismo \omterm{def:g10:vectors:vector}{vector}. Las dos bimedianas (y,
por Varignon, las diagonales del paralelogramo $IJKL$) todas comparten
este centro: el centroide del cuadrilátero.

\textbf{14.}$\vect{MN} = \vect{AN} - \vect{AM} =
\frac13\vect{AC} - \frac13\vect{AB} = \frac13\vect{BC}$:
entonces $(MN) \parallel (BC)$ y $MN = \frac13 BC$.

\textbf{15.} Con $\vect{AM} = k\,\vect{AB}$ y
$\vect{AN} = k\,\vect{AC}$: $\vect{MN} = k\,\vect{BC}$ --- uno
línea que contiene el teorema \omterm{prop:g10:coordgeom:midpoint}{punto medio} ($k = \frac12$) y
Tales (cualquier $k$): la razón pasa por la resta
intacto.

\textbf{16.} Resultante $(3, 4)$, magnitud
$\sqrt{9 + 16} = 5$ km/h: el barco realmente viaja
noreste a norte a lo largo del triángulo $3$--$4$--$5$
hipotenusa.

\textbf{17.} $\vec u + \vec v + \vec w = (0, 0)$: el anillo
permanece quieto --- equilibrio. A lo largo de un polígono cerrado
$A_1 A_2 \dots A_n A_1$, Chasles telescópicamente el \omterm{def:g10:vectors:sum}{suma} para
$\vect{A_1 A_1} = \vec 0$: dar vueltas te trae de vuelta.

\textbf{18.} $\vec w = -(\vec u + \vec v) = -(3, 2) =
(-3, -2)$.

\textbf{19.} Resultante $(1.5,\ 2)$, magnitud
$\sqrt{2.25 + 4} = 2.5$ m/s. Cruce: $50$ m a $2$ m/s
a través de: $25$ s; deriva: $1.5 \times 25 = 37.5$ m aguas abajo.

\textbf{20.} Reprobado este fin de semana: los dos tercios del centroide
teorema (pregunta 7), el centro bimediano/Varignon (pregunta
13), Tales--punto medio (preguntas 14--15). ¿Qué flechas todavía?
no puedo hacer: medir \emph{anglos} (y por lo tanto las longitudes en
direcciones arbitrarias) --- el producto escalar de grado 11 es
Precisamente el instrumento que falta.
\end{solution}
