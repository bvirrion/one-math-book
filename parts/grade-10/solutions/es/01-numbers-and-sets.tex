\section*{Capítulo \ref{ch:g10:numbers} --- Números y conjuntos de números}

\begin{solution}{exo:g10:numbers:1}
$\frac{15}{3} = 5 \in \N$. \quad $-7 \in \Z$. \quad
$\frac{22}{7} \in \Q$ (no es un \omterm{def:g10:numbers:sets}{entero}: $22 = 7 \times 3 + 1$).
\quad $\sqrt 9 = 3 \in \N$. \quad $\sqrt{10} \in \R$
(\omterm{ex:g10:numbers:classify}{irracional}, pues $10$ no es el cuadrado de un racional ---
admitido aquí, en el espíritu del \cref{thm:g10:numbers:sqrt2}).
\quad $-2.4 = -\frac{24}{10} \in \Q$.
\quad $0 \in \N$.
\end{solution}

\begin{solution}{exo:g10:numbers:2}
(a) $\intco{-2}{5}$: punto lleno en $-2$, punto hueco en $5$.
(b) $\intoo{3}{+\infty}$: punto hueco en $3$, sombreado hacia la
derecha.
(c) $\intoc{-\infty}{-1}$: sombreado desde la izquierda hasta un
punto lleno en $-1$.
(d) «distancia de $x$ a $2$ a lo sumo $3$» significa
$\abs{x - 2} \leq 3$, es decir $x \in \intcc{-1}{5}$: puntos llenos
en $-1$ y $5$.
\end{solution}

\begin{solution}{exo:g10:numbers:3}
(a) Los dos \omterm{def:g10:numbers:interval}{intervalos} se solapan entre $0$ y $2$:
$I \cap J = \intcc{0}{2}$ ($0 \in J$ y $2 \in I$, y ambos
pertenecen también al otro conjunto), e
$I \cup J = \intco{-3}{4}$.

(b) $I \cap J$ es el conjunto de $x$ con $-2 \leq x$ y $x < 1$:
$\intco{-2}{1}$. La \omterm{def:g10:numbers:interunion}{unión} cubre todo: $I \cup J = \R$.
\end{solution}

\begin{solution}{exo:g10:numbers:4}
$\abs{-6} = 6$; \quad $\abs{4 - 9} = \abs{-5} = 5$; \quad
$\abs{-3 - 5} = \abs{-8} = 8$; \quad
$\sqrt 2 > 1$ así que $\abs{\sqrt2 - 1} = \sqrt2 - 1$; \quad
$1 - \sqrt2 < 0$ así que $\abs{1 - \sqrt2} = \sqrt2 - 1$ también
(un número y su opuesto tienen el mismo
\omterm{def:g10:numbers:abs}{valor absoluto}).
\end{solution}

\begin{solution}{exo:g10:numbers:5}
$\abs{x} = 5$: distancia $5$ a $0$, luego $x = 5$ o $x = -5$.

$\abs{x - 1} = 3$: distancia $3$ a $1$, luego $x = 4$ o $x = -2$.

$\abs{x - 4} \leq 1$: distancia a lo sumo $1$ a $4$, luego
$x \in \intcc{3}{5}$.

$\abs{x + 2} < 3$: distancia estrictamente menor que $3$ a $-2$,
luego $x \in \intoo{-5}{1}$.
\end{solution}

\begin{solution}{exo:g10:numbers:6}
Cada \omterm{def:g10:numbers:interval}{intervalo} se describe por su centro $a$ (punto medio de
las cotas) y radio $r$ (mitad de la longitud).

$\intcc{2}{8}$: $a = 5$, $r = 3$, luego $\abs{x - 5} \leq 3$.

$\intoo{-5}{1}$: $a = -2$, $r = 3$, cotas abiertas, luego
$\abs{x + 2} < 3$.

$\intcc{-7}{-3}$: $a = -5$, $r = 2$, luego $\abs{x + 5} \leq 2$.
\end{solution}

\begin{solution}{exo:g10:numbers:7}
Sea $x = 0.272727\dots$ Entonces $100x = 27.2727\dots$, y al restar:
\[
100x - x = 27.2727\dots - 0.2727\dots = 27,
\]
luego $99x = 27$ y $x = \frac{27}{99} = \frac{3}{11}$, un cociente de
\omterm{def:g10:numbers:sets}{enteros}: $x$ es racional.
\end{solution}

\begin{solution}{exo:g10:numbers:8}
\emph{1. Verdadero:} sumar \omterm{def:g10:numbers:sets}{enteros} (números enteros positivos o
negativos) siempre produce un \omterm{def:g10:numbers:sets}{entero}.

\emph{2. Falso:} $\frac{1}{2}$ es un cociente de los
\omterm{def:g10:numbers:sets}{enteros} $1$ y $2$ y no es un \omterm{def:g10:numbers:sets}{entero}.

\emph{3. Verdadero:} $\frac pq + \frac{p'}{q'} = \frac{pq' + p'q}{qq'}$
es de nuevo un cociente de \omterm{def:g10:numbers:sets}{enteros} (con denominador no nulo).

\emph{4. Verdadero:} supóngase $r$ racional, $t$
\omterm{ex:g10:numbers:classify}{irracional}, y $r + t = s$ racional. Entonces $t = s - r$
sería diferencia de dos racionales, luego racional (por 3, con $-r$)
--- contradicción. Así $r + t$ es \omterm{ex:g10:numbers:classify}{irracional}.
\end{solution}

\begin{solution}{exo:g10:numbers:9}
Multiplicando $1.414 \leq \sqrt2 \leq 1.415$ por $2 > 0$:
$2.828 \leq 2\sqrt2 \leq 2.830$.

Sumando $3$: $4.414 \leq \sqrt2 + 3 \leq 4.415$.

Multiplicando por $-1 < 0$ se invierten las desigualdades:
$-1.415 \leq -\sqrt2 \leq -1.414$.
\end{solution}

\begin{solution}{exo:g10:numbers:10}
Primero el hecho auxiliar. Divida $p$ por $3$: el resto es $0$, $1$ o
$2$, es decir $p = 3k$, $p = 3k+1$ o $p = 3k+2$. Elevando al cuadrado:
\[
(3k)^2 = 3(3k^2), \quad
(3k+1)^2 = 3(3k^2 + 2k) + 1, \quad
(3k+2)^2 = 3(3k^2 + 4k + 1) + 1 .
\]
Solo el primero es múltiplo de $3$: si $p^2$ es múltiplo de $3$,
también lo es $p$.

Supóngase ahora $\sqrt3 = \frac pq$ en forma simplificada. Elevando al
cuadrado: $p^2 = 3q^2$, así $p^2$ es múltiplo de $3$, luego $p = 3k$.
Entonces $9k^2 = 3q^2$, luego $q^2 = 3k^2$ y $q$ también es múltiplo de
$3$ --- pero entonces la fracción $\frac pq$ no estaba simplificada:
contradicción. Así $\sqrt3$ es \omterm{ex:g10:numbers:classify}{irracional}.
\end{solution}

\begin{solution}{pb:g10:numbers:1}
\textbf{1.} $-7 \in \Z$; \ $\frac{13}{4} \in \Q$; \
$\sqrt{16} = 4 \in \N$; \ $0.121212\ldots = \frac{12}{99} =
\frac{4}{33} \in \Q$ (problema de fin de semana de los decimales
periódicos del volumen de Primaria--Secundaria); \
$\sqrt8 = 2\sqrt2$ es \omterm{ex:g10:numbers:classify}{irracional}: conjunto más pequeño $\R$; \
$\pi$: $\R$.

\textbf{2.} $\frac pq + \frac rs = \frac{ps + qr}{qs}$ y
$\frac pq \times \frac rs = \frac{pr}{qs}$: cocientes de
\omterm{def:g10:numbers:sets}{enteros} con denominadores no nulos --- racionales.

\textbf{3.} (a) Si $r$ es racional, $x$ \omterm{ex:g10:numbers:classify}{irracional} y
$r + x = q$ fuera racional, entonces $x = q - r$ sería diferencia de
racionales, luego racional (pregunta~2): contradicción. (b) Si
$r \neq 0$ y $r x = q$ racional, entonces $x = \frac qr$ es racional:
contradicción.

\textbf{4.} $\sqrt2$ y $-\sqrt2$ son \omterm{ex:g10:numbers:classify}{irracionales} con suma $0$;
$\sqrt2$ y $\sqrt2$ tienen producto $2$. Resultados racionales a
partir de ingredientes \omterm{ex:g10:numbers:classify}{irracionales}: los
\omterm{ex:g10:numbers:classify}{irracionales} no forman un reino estable.

\textbf{5.} $\sqrt8 = 2\sqrt2$, luego
$\sqrt2 + \sqrt8 = 3\sqrt2$: \omterm{ex:g10:numbers:classify}{irracional} (pregunta~3b,
multiplicador $3$). Y $\sqrt2 \times \sqrt8 = \sqrt{16} = 4 \in \N$:
el producto de dos \omterm{ex:g10:numbers:classify}{irracionales} cae en el reino más pequeño de
todos.

\textbf{6.} $3.475$, luego $3.471$ (o $3.4701$, $3.47001$,
\dots): añadir cifras decimales produce racionales frescos entre los
dos para siempre.

\textbf{7.} Los múltiplos de $10^{-n}$ marchan por la recta en pasos
de $10^{-n} < g$. El primer múltiplo estrictamente mayor que $a$ ---
existe, pues los múltiplos acaban superando $a$ --- está a lo sumo un
paso más allá de $a$, luego antes de $a + g = b$: estrictamente entre
$a$ y $b$. Un punto de la malla es un decimal, luego un racional: todo
\omterm{def:g10:numbers:interval}{intervalo} de longitud positiva contiene uno.

\textbf{8.} Entre $a$ y el racional $r_1$ hallado en la pregunta~7 hay
(pregunta~7 de nuevo) un racional $r_2$; entre $a$ y $r_2$ un
racional $r_3$; y así sucesivamente: infinitos, todos distintos,
todos en el \omterm{def:g10:numbers:interval}{intervalo} original.

\textbf{9.} $r + \frac{\sqrt2}{10^k}$ es la suma de un racional y un
\omterm{ex:g10:numbers:classify}{irracional} ($\frac{\sqrt2}{10^k}$ es \omterm{ex:g10:numbers:classify}{irracional} por la
pregunta~3b), luego \omterm{ex:g10:numbers:classify}{irracional}. Como
$\frac{\sqrt2}{10^k} < \frac{2}{10^k}$ se encoge por debajo de
cualquier cota, para $k$ bastante grande
$r + \frac{\sqrt2}{10^k}$ sigue antes de $b$: un
\omterm{ex:g10:numbers:classify}{irracional} dentro del \omterm{def:g10:numbers:interval}{intervalo} --- y al variar $k$ se
obtienen infinitos.

\textbf{10.} No hay real positivo mínimo: si $s > 0$ lo fuera, el
\omterm{def:g10:numbers:interval}{intervalo} $(0, s)$ aún contendría un racional (pregunta~7),
positivo y menor que $s$. No hay número justo después de $3$:
cualquier candidato $c > 3$ deja el \omterm{def:g10:numbers:interval}{intervalo} $(3, c)$, que no
está vacío --- el juego del problema «no hay siguiente número» del
volumen de Primaria--Secundaria, ahora un teorema sobre $\R$.

\textbf{11.} $\abs{x - 5} = 2$: $x = 3$ o $x = 7$.
$\abs{x - 5} < 2$: $x \in \intoo{3}{7}$.
$\abs{x + 1} \geq 3$: distancia a $-1$ al menos $3$:
$x \in \intoc{-\infty}{-4} \cup \intco{2}{+\infty}$.

\textbf{12.} $\abs{d - 80} \leq 0.05$, es decir
$d \in \intcc{79.95}{80.05}$. El pistón de $79.97$ pasa; el de
$80.06$ falla por una centésima de milímetro.

\textbf{13.} $(3, -5)$: $\abs{-2} = 2 \leq 8$. $(-2, -7)$:
$\abs{-9} = 9 = 2 + 7$: igualdad. Mismo signo: $\abs{a + b}$ es la
suma de las distancias, igual a $\abs a + \abs b$. Signos opuestos:
el recorrido da la vuelta, $\abs{a + b}$ es la \emph{diferencia} de
las distancias, estrictamente menor que su suma (salvo que una sea
$0$). Igualdad exactamente cuando $a$ y $b$ tienen el mismo signo o
uno es cero.

\textbf{14.} Las soluciones son los puntos equidistantes de $2$ y
$-4$: el punto medio, $x = -1$. Es la mediatriz del problema de los
dos espejos del volumen de Primaria--Secundaria, colapsada a una
dimensión: un solo punto.

\textbf{15.} $\sqrt{x^2} = \abs{x}$, no $x$: para $x = -3$,
$\sqrt{9} = 3 = \abs{-3} \neq -3$. Entonces $x^2 < 9$ se lee
$\sqrt{x^2} < 3$, es decir $\abs x < 3$:
$x \in \intoo{-3}{3}$.

\textbf{16.} La medición solo afirma que la longitud está en el
\omterm{def:g10:numbers:interval}{intervalo} $\intcc{1.41420}{1.41422}$ --- y por las
preguntas~7 y~9 \emph{ese \omterm{def:g10:numbers:interval}{intervalo} contiene infinitos
racionales e infinitos \omterm{ex:g10:numbers:classify}{irracionales}}. Lo mismo vale para
toda tolerancia futura, por pequeña que sea. Ninguna medición puede
separar las dos familias; solo una demostración sobre la longitud
exacta --- como la de la diagonal del cuadrado unidad (problema de
irracionalidad del volumen de Primaria--Secundaria) --- puede.

\textbf{17.} Se acercan a $\sqrt2$ con errores por debajo de
$10^{-1}, 10^{-2}, 10^{-3}, \dots$: los racionales se aprietan
contra $\sqrt2$ a todas las escalas. El hecho general: todo
\omterm{def:g10:numbers:sets}{número real} se aproxima tanto como se quiera por racionales
(sus truncamientos decimales) --- de nuevo la densidad, vista desde
el lado del objetivo.

\textbf{18.} $a + b \in \intcc{2.4 + 1.1}{2.6 + 1.3} =
\intcc{3.5}{3.9}$: incertidumbre $\pm 0.2$ (errores sumados).
$a \times b \in \intcc{2.4 \times 1.1}{2.6 \times 1.3} =
\intcc{2.64}{3.38}$: una incertidumbre de unos $\pm 0.37$ en torno
a $3$ --- los errores relativos se suman en productos, así que la
precisión se degrada más rápido.

\textbf{19.} Escriba los valores verdaderos $a = 2.5 + e_1$,
$b = 1.2 + e_2$ con $\abs{e_1}, \abs{e_2} \leq 0.1$. Entonces el
error de la suma es $\abs{e_1 + e_2} \leq \abs{e_1} +
\abs{e_2} \leq 0.2$: la regla del ingeniero \emph{es} la
desigualdad triangular de la pregunta~13.

\textbf{20.} La recta real no tiene huecos ni vecinos: después de
cualquier número, no hay un «siguiente» (pregunta~10). Sus puntos se
parten en los racionales --- exactamente los decimales que terminan
o se repiten (problema del volumen de Primaria--Secundaria) --- y los
\omterm{ex:g10:numbers:classify}{irracionales}, y las dos familias están tan entrelazadas que
todo \omterm{def:g10:numbers:interval}{intervalo} alberga infinitos de cada una (preguntas~8
y~9), por lo que ninguna medición, solo una demostración, puede
distinguirlas (pregunta~16). Y la sorpresa final, dejada como
promesa: los \omterm{ex:g10:numbers:classify}{irracionales} superan en número a los racionales
--- no contando uno a uno, sino en el sentido preciso de comparar
infinitos, una teoría construida en los volúmenes universitarios.
\end{solution}
