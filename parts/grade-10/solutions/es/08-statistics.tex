\section*{Capítulo \ref{ch:g10:stats} --- Estadísticas descriptivas}

\begin{solution}{exo:g10:stats:1}
Suma: $0+1+1+2+2+2+3+3+4+4+5+6+7+8+12 = 60$, por lo que \omterm{def:g10:stats:mean}{significar} es
$\frac{60}{15} = 4$. La serie ya está ordenada y $N = 15$ es impar:
el \omterm{def:g10:stats:median}{mediana} es el valor $8$, $3$. \omterm{def:g10:stats:quartiles}{Rango}: $12 - 0 = 12$.
\end{solution}

\begin{solution}{exo:g10:stats:2}
Frecuencias: $\frac{7}{50} = 0.14$, $\frac{9}{50} = 0.18$,
$\frac{8}{50} = 0.16$, $\frac{10}{50} = 0.20$, $\frac{6}{50} = 0.12$,
$\frac{10}{50} = 0.20$ (suma $1$). \omterm{def:g10:stats:mean}{Significar} resultado:
\[
\bar x = \frac{7 \times 1 + 9 \times 2 + 8 \times 3 + 10 \times 4
+ 6 \times 5 + 10 \times 6}{50}
= \frac{7 + 18 + 24 + 40 + 30 + 60}{50}
= \frac{179}{50} = 3.58 .
\]
\end{solution}

\begin{solution}{exo:g10:stats:3}
$N = 12$ (par): el \omterm{def:g10:stats:median}{mediana} es el \omterm{prop:g10:coordgeom:midpoint}{punto medio} del $6$th y $7$th
valores, ambos iguales a $6$: \omterm{def:g10:stats:median}{mediana} $6$.

$Q_1$: $\frac{12}{4} = 3$, por lo que $Q_1$ es el tercer valor $3$: $4$.

$Q_3$: $\frac{3 \times 12}{4} = 9$, por lo que $Q_3$ es el valor $9$: $8$.

Diagrama de caja: caja de $4$ a $8$ con la barra \omterm{def:g10:stats:median}{mediana} en $6$, bigotes de
$1$ a $11$.
\end{solution}

\begin{solution}{exo:g10:stats:4}
Las puntuaciones $24$ suman $24 \times 11.5 = 276$. Con la nueva puntuación el
el total es $276 + 19 = 295$ para estudiantes $25$: nuevo \omterm{def:g10:stats:mean}{significar}
$\frac{295}{25} = 11.8$.
\end{solution}

\begin{solution}{exo:g10:stats:5}
Total actual: $5 \times 12 = 60$. Se requiere una prueba \omterm{def:g10:stats:mean}{significar} de $13$ sobre $6$
un total de $78$, por lo que la puntuación $6$ debe ser $78 - 60 = 18$. Un \omterm{def:g10:stats:mean}{significar} de
$14$ requeriría $84 - 60 = 24 > 20$: no accesible.
\end{solution}

\begin{solution}{exo:g10:stats:6}
\omterm{def:g10:stats:mean}{Significar} mayor que $20$: por ejemplo
$1,\ 2,\ 3,\ 10,\ 40,\ 50,\ 60$ --- \omterm{def:g10:stats:median}{mediana} $10$ (el valor ordenado $4$),
\omterm{def:g10:stats:mean}{significar} $\frac{166}{7} \approx 23.7$.

\omterm{def:g10:stats:mean}{Significar} menor que $6$: por ejemplo
$0,\ 0,\ 0,\ 10,\ 10,\ 10,\ 10$ --- \omterm{def:g10:stats:median}{mediana} $10$, \omterm{def:g10:stats:mean}{significar}
$\frac{40}{7} \approx 5.7$.

El \omterm{def:g10:stats:mean}{significar} no puede ir por debajo de $\frac{40}{7}$: para el \omterm{def:g10:stats:median}{mediana} (el $4$th ordenado
valor) para que sea $10$, todos los valores en las posiciones $4$ a $7$ deben ser
$\geq 10$, por lo que el total es al menos $40$ y el \omterm{def:g10:stats:mean}{significar} al menos
$\frac{40}{7}$ --- el segundo ejemplo es extremo.

Estos ejemplos muestran que \omterm{def:g10:stats:mean}{significar} y \omterm{def:g10:stats:median}{mediana} son en gran medida independientes:
conocer uno dice poco del otro, por eso un buen resumen
informa ambos.
\end{solution}

\begin{solution}{exo:g10:stats:7}
\emph{1.} \omterm{def:g10:stats:mean}{Significar}:
$\frac{9 \times 1800 + 12000}{10} = \frac{16200 + 12000}{10} = 2820$.
\omterm{def:g10:stats:median}{Mediana}: ordenado, los salarios $10$ son nueve valores $1800$ y luego $12000$;
el \omterm{def:g10:stats:median}{mediana} es el \omterm{prop:g10:coordgeom:midpoint}{punto medio} de los valores $5$th y $6$th, ambos $1800$:
\omterm{def:g10:stats:median}{mediana} $1800$.

\emph{2.} El \omterm{def:g10:stats:median}{mediana} ($1800$) describe el salario típico: es lo que
Los empleados de $9$ de $10$ realmente ganan. El \omterm{def:g10:stats:mean}{significar} ($2820$) está levantado.
por el salario único alto y no iguala la nómina de nadie.
\end{solution}

\begin{solution}{exo:g10:stats:8}
Altura total: $12 \times 168 + 18 \times 163 = 2016 + 2934 = 4950$ cm para
$30$ estudiantes, por lo que \omterm{def:g10:stats:mean}{significar} es $\frac{4950}{30} = 165$ cm. esta mas cerca de
\omterm{def:g10:stats:mean}{significar} de las niñas porque son más numerosas (ponderado \omterm{def:g10:stats:mean}{significar}, pesos
$12$ y $18$).
\end{solution}

\begin{solution}{exo:g10:stats:9}
\emph{1.} El nuevo \omterm{def:g10:stats:mean}{significar} es
\[
\bar y = \frac{y_1 + \dots + y_N}{N}
= \frac{(a x_1 + b) + \dots + (a x_N + b)}{N}
= \frac{a(x_1 + \dots + x_N) + Nb}{N}
= a \bar x + b .
\]

\emph{2.} Con $a = 1.8$ y $b = 32$: \omterm{def:g10:stats:mean}{significar}
$1.8 \times 20 + 32 = 68$ grados Fahrenheit.
\end{solution}

\begin{solution}{pb:g10:stats:1}
\textbf{1.} \omterm{def:g10:stats:mean}{Significar}: $\frac{2\,550}{7} \approx 364$ mil
euros; \omterm{def:g10:stats:median}{mediana}: el cuarto de los siete precios pedidos, $230$.
El \omterm{def:g10:stats:median}{mediana} describe la calle --- seis de las siete casas
no cuesta nada cerca de $364$.

\textbf{2.} Sin la mansión: \omterm{def:g10:stats:mean}{significar} $\frac{1\,350}{6} = 225$,
\omterm{def:g10:stats:median}{mediana} $\frac{220 + 230}{2} = 225$. El \omterm{def:g10:stats:mean}{significar} cayó aproximadamente
$139$; el \omterm{def:g10:stats:median}{mediana} por $5$. Un valor extremo puede arrastrar al \omterm{def:g10:stats:mean}{significar}
en cualquier lugar; el \omterm{def:g10:stats:median}{mediana} apenas se da cuenta --- es \emph{robusto}.

\textbf{3.} \omterm{def:g10:stats:mean}{Significar} $= \frac{0 \times 20 + 1 \times 25 + 2 \times
30 + 3 \times 15 + 4 \times 10}{100} = \frac{170}{100} = 1.7$.
Ninguna familia tiene hijos $1.7$, pero $1.7 \times 100 = 170$
\emph{is} el número total exacto de niños: el \omterm{def:g10:stats:mean}{significar} codifica
el total, redistribuido equitativamente --- su verdadero talento.

\textbf{4.} $N = 15$: \omterm{def:g10:stats:median}{mediana} $=$ 8vo valor $= 11$;
$Q_1 =$ 4to valor $= 8$; $Q_3 =$ Valor 12 $= 14$.

\textbf{5.} \omterm{def:g10:stats:quartiles}{Rango} $19 - 4 = 15$: la extensión completa, extremos
incluido. \omterm{def:g10:stats:quartiles}{rango intercuartil} $14 - 8 = 6$: el ancho del
mitad media --- extendido sin el ruido de los extremos.

\textbf{6.} Un desplazamiento $+b$ mueve el valor \emph{cada}, por lo que
diferencias de valores --- \omterm{def:g10:stats:quartiles}{rango}, IQR --- no cambian:
$(x_i + b) - (x_j + b) = x_i - x_j$. Una escala de $a$
multiplica todas las diferencias por $a$: \omterm{def:g10:stats:quartiles}{rango} y IQR se escalan por
$\abs a$.

\textbf{7.} Curva aditiva: $6 \to 9$ y $18 \to 21$ --- el
el estudiante débil gana $50\,\%$, el fuerte $17\,\%$ y $21$
supera ligeramente la escala de puntos $20$. Multiplicativo:
$6 \to 8$ y $18 \to 24$ --- proporciones conservadas, pero $24$ es
imposible en una escala $20$: el problema del techo. aditivo
las curvas aplanan la parte inferior; los multiplicativos explotan el
arriba.

\textbf{8.} Puntuación total: $20 \times 12 + 30 \times 10 = 540$,
entonces el \omterm{def:g10:stats:mean}{significar} fusionado es $\frac{540}{50} = 10.8$: treinta de clase B
los estudiantes superan a los veinte de la clase A --- significa fusionarse por
\emph{ponderado} promedio, nunca por promedio simple.

\textbf{9.} \omterm{def:g10:stats:median}{Mediana} de $A$: $2$; de $B$: $5$. Lista fusionada
$\{0, 1, 2, 3, 10\}$: \omterm{def:g10:stats:median}{mediana} $2$. El promedio ponderado por tamaño de
el \omterm{def:g10:stats:median}{medianas} sería $\frac{3 \times 2 + 2 \times 5}{5} = 3.2
\neq 2$: \omterm{def:g10:stats:median}{medianas} no lleva total, por lo que no existe una fórmula de fusión
--- hay que reordenar toda la lista.

\textbf{10.} \omterm{def:g10:stats:mean}{Significar}: $\frac{31.3}{5} = 6.26$. \omterm{def:g10:stats:median}{Mediana}: $5.4$.
Recortado \omterm{def:g10:stats:mean}{significar}: $\frac{5.3 + 5.4 + 5.5}{3} = 5.4$. uno
Un juez entusiasta (o corrupto) movió el \omterm{def:g10:stats:mean}{significar} por casi un
punto; el \omterm{def:g10:stats:median}{mediana} y el \omterm{def:g10:stats:mean}{significar} recortado se encogieron de hombros --- de ahí el
Reglas de recorte de deportes juzgados.

\textbf{11.} Fábrica A: mitad media dentro de $0.2$ mm; fábrica B:
dentro de $4$ mm. Mismo \omterm{def:g10:stats:mean}{significar}, confiabilidad opuesta: compre en A. El
\omterm{def:g10:stats:quartiles}{rango intercuartil} decidió --- significa solo comparar centros,
nunca coherencia.

\textbf{12.} \omterm{def:g10:stats:median}{Mediana} muy por debajo de \omterm{def:g10:stats:mean}{significar} delata un \emph{largo a la derecha
cola}: la mayoría espera corta, algunas catastróficas. Ejemplo:
$\{5, 5, 10, 10, 95\}$ --- \omterm{def:g10:stats:median}{mediana} $10$, \omterm{def:g10:stats:mean}{significar}
$\frac{125}{5} = 25$.

\textbf{13.} Significa que $75\,\%$ de bebés de esa edad son
más bajo (y $25\,\%$ más alto): un rango, no una talla. ``Cada
niño por encima de \omterm{def:g10:stats:mean}{significar}'' es imposible --- si todos los valores excedieron
el \omterm{def:g10:stats:mean}{significar}, su promedio superaría al \omterm{def:g10:stats:mean}{significar}, que es a su vez:
contradicción. Pero \emph{casi} todo puede ser: uno muy corto
El niño puede sostener el \omterm{def:g10:stats:mean}{significar} debajo de todos los demás. Por encima del
\emph{\omterm{def:g10:stats:median}{mediana}}: nunca más de la mitad, por definición. Rangos
no puede ser halagado; los promedios pueden.

\textbf{14.} Marca típica: igual, \omterm{def:g10:stats:median}{mediana} $11$ para ambos.
Homogeneidad del medio: B más ajustado (IQR $3$ contra $5$).
Extremos: B wilder (\omterm{def:g10:stats:quartiles}{rango} $17$ contra $13$). B es una clase de
estudiantes similares más algunos valores atípicos; A está distribuido uniformemente.

\textbf{15.} (1) ¿Es $5\,000$ el \omterm{def:g10:stats:mean}{significar} o el \omterm{def:g10:stats:median}{mediana} --- y puede
¿Tengo el \omterm{def:g10:stats:median}{mediana}? (2) ¿Cuál es el diferencial (\omterm{def:g10:stats:quartiles}{cuartiles}), así que
¿Sabes lo típico que es lo "típico"? (3) Quién fue contado --- muestra
tamaño y selección (el problema del fin de semana de cómo se encuentran los datos de la lección de encuesta del volumen de la escuela secundaria)?

\textbf{16.} Por ejemplo $100, 150, 200, 250, 800$: \omterm{def:g10:stats:median}{mediana}
$200$, \omterm{def:g10:stats:mean}{significar} $\frac{1\,500}{5} = 300$.

\textbf{17.}$\sum (x_i - \bar x) = \sum x_i - N \bar x =
N\bar x - N\bar x = 0$. Verificar: $-200 - 150 - 100 - 50 + 500 =
0$. El \omterm{def:g10:stats:mean}{significar} es el punto donde se equilibran las desviaciones --- el
centro de gravedad de los datos (el problema del fin de semana del centroide del volumen $G$ de la escuela secundaria, en
una dimensión).

\textbf{18.} En el \omterm{def:g10:stats:median}{mediana} $4$: $3 + 1 + 0 + 5 + 9 = 18$ km. en
el \omterm{def:g10:stats:mean}{significar} $6$: $5 + 3 + 2 + 3 + 7 = 20$ km. En $5$:
$4 + 2 + 1 + 4 + 8 = 19$ kilómetros. El \omterm{def:g10:stats:median}{mediana} gana --- alejándose
desde allí siempre pasa más tiendas de las que se acerca, por lo que el
el total sólo puede crecer.

\textbf{19.} $0.3^{10} \approx 6 \times 10^{-6}$: alrededor de seis
pilotos \emph{por millón}. En $4\,000$ pilotos lo esperado
El recuento es $0.02$ --- cero, como descubrió el teniente Daniels. el aire
Force dejó de diseñar para el hombre promedio y creó
todo ajustable: asientos, pedales, correas --- diseño para el
\emph{desparramar}, no el centro.

\textbf{20.} Totales: el \emph{\omterm{def:g10:stats:mean}{significar}}. Individuo típico: el
\emph{\omterm{def:g10:stats:median}{mediana}}. Equidad de propagación: el \emph{\omterm{def:g10:stats:quartiles}{cuartiles}} (y
RIQ). Contaminación: el \emph{trimmed \omterm{def:g10:stats:mean}{significar}}. y el cierre
línea: la persona promedio no existe --- pero el \omterm{def:g10:stats:median}{mediana}
la persona casi lo hace.
\end{solution}
