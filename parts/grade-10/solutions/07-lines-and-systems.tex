\section*{Chapter \ref{ch:g10:lines} --- Equations of Lines and Linear Systems}

\begin{solution}{exo:g10:lines:1}
$y = 2x - 3$: \omterm{def:g10:reffunc:affine}{slope} $2$, intercept $-3$.
$y = -\frac13 x + 1$: \omterm{def:g10:reffunc:affine}{slope} $-\frac13$, intercept $1$.
$y = 4$: \omterm{def:g10:reffunc:affine}{slope} $0$, intercept $4$ (horizontal line).
$x = -2$: vertical line, no \omterm{def:g10:reffunc:affine}{slope} and no \omterm{thm:g10:lines:reduced}{reduced equation}.
\end{solution}

\begin{solution}{exo:g10:lines:2}
$3 \times 2 + 1 = 7$: yes, $A$ is on the line.
$3 \times (-1) + 1 = -2 \neq -1$: $B$ is not.
$3 \times 4 + 1 = 13$: yes, $C$ is on the line.
\end{solution}

\begin{solution}{exo:g10:lines:3}
(a) $m = \frac{8 - 2}{3 - 0} = 2$ and $p = y_A = 2$ (the point $A$ is on
the $y$-axis): $y = 2x + 2$.

(b) $m = \frac{-4 - 5}{2 - (-1)} = \frac{-9}{3} = -3$; then
$5 = -3 \times (-1) + p$ gives $p = 2$: $y = -3x + 2$.

(c) $x_A = x_B = 4$: vertical line $x = 4$.
\end{solution}

\begin{solution}{exo:g10:lines:4}
$y = \frac{6x+2}{2} = 3x + 1$. The lines with \omterm{def:g10:reffunc:affine}{slope} $3$ are $y = 3x - 1$,
$y = 3x + 5$ and $y = 3x + 1$: those three are parallel to each other;
no two of them are equal (different intercepts), and none is parallel to
$y = -3x + 2$.
\end{solution}

\begin{solution}{exo:g10:lines:5}
First system: substitute $y = 2x - 1$ into $3x + y = 9$:
$3x + 2x - 1 = 9$, so $5x = 10$, $x = 2$, then $y = 3$. Solution:
$(2, 3)$.

Second system: from $x - y = 4$, $x = y + 4$. Then
$2(y + 4) + 3y = 3$, so $5y = -5$, $y = -1$, then $x = 3$. Solution:
$(3, -1)$.
\end{solution}

\begin{solution}{exo:g10:lines:6}
First system: adding the \omterm{def:g10:algebra:equation}{equations} eliminates $y$: $2x = 14$, $x = 7$;
then $7 + y = 10$ gives $y = 3$. Solution: $(7, 3)$.

Second system: multiply the first \omterm{def:g10:algebra:equation}{equation} by $5$ and the second by $-2$:
\[
\begin{cases}
15x + 10y = 60 \\
-4x - 10y = -16
\end{cases}
\]
Adding: $11x = 44$, so $x = 4$; then $3 \times 4 + 2y = 12$ gives
$y = 0$. Solution: $(4, 0)$.
\end{solution}

\begin{solution}{exo:g10:lines:7}
Compute $ab' - a'b$ in each case.

First: $2 \times 2 - (-4) \times (-1) = 4 - 4 = 0$, and the second
\omterm{def:g10:algebra:equation}{equation} is $-2$ times the first: twice the same line, infinitely many
solutions.

Second: $ab' - a'b = 0$ again, but the right-hand sides are not in the
ratio $-2$ ($1 \neq -6$): two distinct parallel lines, no solution.

Third: $2 \times 1 - 1 \times (-1) = 3 \neq 0$: exactly one solution.
\end{solution}

\begin{solution}{exo:g10:lines:8}
Let $x$ be the number of child tickets and $y$ the number of adult
tickets:
\[
\begin{cases}
x + y = 200 \\
5x + 9y = 1352 .
\end{cases}
\]
From the first \omterm{def:g10:algebra:equation}{equation} $x = 200 - y$; substituting,
$5(200 - y) + 9y = 1352$, so $1000 + 4y = 1352$, $y = 88$, then
$x = 112$. So $112$ child tickets and $88$ adult tickets. Check:
$5 \times 112 + 9 \times 88 = 560 + 792 = 1352$.
\end{solution}

\begin{solution}{exo:g10:lines:9}
\emph{1.} Parallel means same \omterm{def:g10:reffunc:affine}{slope} $2$: $y = 2x + p$ with
$4 = 2 \times 1 + p$, so $p = 2$: $d'\colon y = 2x + 2$.

\emph{2.} On the $x$-axis, $y = 0$: $2x + 2 = 0$ gives $x = -1$. The
\omterm{def:g10:numbers:interunion}{intersection} point is $(-1, 0)$.
\end{solution}

\begin{solution}{exo:g10:lines:10}
\emph{1.} \omterm{prop:g10:coordgeom:midpoint}{Midpoint} of $[BC]$: $A' = \left(\frac{6+2}{2},
\frac{2+4}{2}\right) = (4, 3)$. The median from $A(0,0)$ has \omterm{def:g10:reffunc:affine}{slope}
$\frac{3 - 0}{4 - 0} = \frac34$: \omterm{def:g10:algebra:equation}{equation} $y = \frac34 x$.

\omterm{prop:g10:coordgeom:midpoint}{Midpoint} of $[AC]$: $B' = (1, 2)$. The median from $B(6, 2)$ has \omterm{def:g10:reffunc:affine}{slope}
$\frac{2 - 2}{1 - 6} = 0$: it is the horizontal line $y = 2$.

\emph{2.} \omterm{def:g10:numbers:interunion}{Intersection}: $\frac34 x = 2$ gives $x = \frac83$, so
$G\left(\frac83, 2\right)$. The third median joins $C(2,4)$ to the
\omterm{prop:g10:coordgeom:midpoint}{midpoint} of $[AB]$, $C' = (3, 1)$; its \omterm{def:g10:reffunc:affine}{slope} is
$\frac{1 - 4}{3 - 2} = -3$, \omterm{def:g10:algebra:equation}{equation} $y = -3x + 10$. At $x = \frac83$:
$y = -8 + 10 = 2$: $G$ is on it. And indeed
$G = \left(\frac{0 + 6 + 2}{3}, \frac{0 + 2 + 4}{3}\right)$: the
centroid averages the \omterm{def:g10:coordgeom:system}{coordinates} of the vertices.
\end{solution}

\begin{solution}{pb:g10:lines:1}
\textbf{1.} \omterm{def:g10:reffunc:affine}{Slope} $\frac{8 - 2}{3 - 1} = 3$; through $(1, 2)$:
$y = 3x - 1$.

\textbf{2.} $y = 3x - 1$ and $y = 3x + 4$: same \omterm{def:g10:reffunc:affine}{slope}, parallel
(and distinct). Crossing $y = 3x - 1$ with $y = -x + 7$:
$3x - 1 = -x + 7$ gives $x = 2$, $y = 5$: point $(2, 5)$.

\textbf{3.} Both points share $x = 2$: the line is vertical,
\omterm{def:g10:algebra:equation}{equation} $x = 2$. A reduced form $y = mx + p$ answers ``one $y$
per $x$'', which a vertical line refuses: its \omterm{def:g10:reffunc:affine}{slope} would be
infinite.

\textbf{4.} $3 \times 4 - 1 = 11$: yes, $(4, 11)$ is on the
line. $3 \times (-1) - 1 = -4 \neq -5$: no.

\textbf{5.} $y = -2x + 3$; crosses the axes at $(0, 3)$ and
$\left(\frac32, 0\right)$. Triangle area:
$\frac12 \times \frac32 \times 3 = \frac94 = 2.25$.

\textbf{6.} Substituting: $3x + 2x - 3 = 7$, $x = 2$,
$y = 1$: solution $(2, 1)$.

\textbf{7.} Adding: $7x = 7$, $x = 1$; then $3y = 6$, $y = 2$:
solution $(1, 2)$.

\textbf{8.} First system: the second \omterm{def:g10:algebra:equation}{equation} is twice the
first: one line counted twice --- infinitely many solutions
(all points of $2x - y = 3$). Second system: same left sides
proportional, right sides not ($2 \times 3 = 6 \neq 1$): two
parallel lines --- no solution. Trichotomy: distinct \omterm{def:g10:reffunc:affine}{slopes}
$\to$ one crossing; equal \omterm{def:g10:reffunc:affine}{slopes}, different intercepts $\to$
parallel, none; equal everything $\to$ same line, infinitely
many.

\textbf{9.} The lines' direction \omterm{def:g10:vectors:vector}{vectors} are $(-b, a)$ and
$(-d, c)$ (or: \omterm{def:g10:reffunc:affine}{slopes} $-\frac ab$ and $-\frac cd$), \omterm{def:g10:vectors:collinear}{collinear}
exactly when $ad - bc = 0$ --- the determinant of
\cref{pb:g10:vectors:1} in a new job. Values: question 7:
$2 \times (-3) - 3 \times 5 = -21 \neq 0$: unique solution.
Question 8: $2 \times (-2) - (-1) \times 4 = 0$ for both:
parallel-or-identical, and the constants separate the two
cases.

\textbf{10.} $ad - bc = 6 - 2m$: zero for $m = 3$. There
$x + 3y = 3$ against $2x + 6y = 7$: doubling the first gives
$2x + 6y = 6 \neq 7$: parallel, \emph{no solution}. For
$m \neq 3$, elimination gives the unique solution
$y = \frac{1}{6 - 2m}$, then $x = 3 - \frac{m}{6 - 2m}$.

\textbf{11.} $p + r = 35$, $2p + 4r = 94$: halving,
$p + 2r = 47$, so $r = 12$ and $p = 23$: twenty-three pheasants,
twelve rabbits --- the \emph{Nine Chapters}' own answer.

\textbf{12.} $x + y = 30$ and $0.6x + 0.2y = 0.35 \times 30 =
10.5$: subtracting $0.2 \times$ the first, $0.4x = 4.5$:
$x = 11.25$ L of syrup, $y = 18.75$ L of drink.

\textbf{13.} Digits $a$, $b$: $a + b = 12$ and
$(10a + b) - (10b + a) = 9(a - b) = 18$, so $a - b = 2$:
$a = 7$, $b = 5$: the number is $75$.

\textbf{14.} Adding: $5c + 5r = 18$, so $c + r = 3.60$ --- one
coffee plus one croissant. Subtracting: $c - r = 0.20$. Hence
$c = 1.90$ and $r = 1.70$ euros: the symmetric shortcut solved
it without any substitution.

\textbf{15.} Doubling the first claim gives $4$ apples $+ 2$
pears $= 10$ euros, but the stall charges $9$: the system is
inconsistent --- parallel lines, empty solution set. Verdict:
the two price claims cannot both be right; either a discount
is hiding, or the arithmetic is.

\textbf{16.} At the origin: $0 > 3 \times 0 - 1 = -1$: true,
so $(0,0)$ lies in the region $y > 3x - 1$ (above the line).
Test any point by plugging it in: above/below according to the
inequality's truth.

\textbf{17.} Corners: $(0, 0)$; $(4, 0)$ (axes and
$x + y = 4$); $(0, 1)$ (axis and $y = 2x + 1$); and
$x + y = 4$ with $y = 2x + 1$: $x = 1$, $y = 3$: $(1, 3)$.

\textbf{18.} $P(0,0) = 0$; $P(4, 0) = 12$; $P(0, 1) = 2$;
$P(1, 3) = 9$. \omterm{def:g10:functions:extrema}{Maximum} at the corner $(4, 0)$: the plan
$x = 4$, $y = 0$ earns $12$ --- corners rule, as the sweeping
parallel lines $3x + 2y = c$ make visible.

\textbf{19.} $1.5x + 3 = 2x$ gives $x = 6$, $y = 12$: at six
kilometers both taxis charge twelve euros --- the break-even
crossing of the two-thermometers weekend problem of the Middle School volume, now a system's unique
solution.

\textbf{20.} A linear \omterm{def:g10:algebra:equation}{equation}'s solutions draw a line; a
system superposes two lines, and $ad - bc$ tells at a glance
whether they cross once (nonzero) or fall into the
parallel-or-identical cases (zero). Pile up inequalities
instead and the solutions form a polygonal region whose corners
carry any linear optimum. Linear algebra generalizes the
determinant to any number of unknowns; linear programming
industrializes the corners.
\end{solution}
