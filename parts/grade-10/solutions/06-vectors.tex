\section*{Chapter \ref{ch:g10:vectors} --- Vectors}

\begin{solution}{exo:g10:vectors:1}
$\vect{AB} + \vect{BD} = \vect{AD}$ (Chasles).

$\vect{KL} + \vect{LM} + \vect{MK} = \vect{KK} = \vec 0$.

$\vect{AC} - \vect{BC} = \vect{AC} + \vect{CB} = \vect{AB}$.

$\vect{AB} + \vect{CA} = \vect{CA} + \vect{AB} = \vect{CB}$.
\end{solution}

\begin{solution}{exo:g10:vectors:2}
$\vect{AB}\,(5-2,\ 3-1) = (3, 2)$; \quad
$\vect{BC}\,(-1-5,\ 4-3) = (-6, 1)$; \quad
$\vect{AC}\,(-1-2,\ 4-1) = (-3, 3)$.
Check: $(3, 2) + (-6, 1) = (-3, 3)$, coordinate by coordinate.
\end{solution}

\begin{solution}{exo:g10:vectors:3}
$\vec u + \vec v = (2 + (-1),\ -3 + 4) = (1, 1)$.

$3\vec u = (6, -9)$.

$2\vec u - \vec v = (4 - (-1),\ -6 - 4) = (5, -10)$.

$\norm{\vec u} = \sqrt{2^2 + (-3)^2} = \sqrt{13}$.
\end{solution}

\begin{solution}{exo:g10:vectors:4}
(a) $4 \times 9 - 6 \times 6 = 36 - 36 = 0$: \omterm{def:g10:vectors:collinear}{collinear}
($\vec v = \frac32 \vec u$).

(b) $2 \times 10 - (-5) \times (-4) = 20 - 20 = 0$: \omterm{def:g10:vectors:collinear}{collinear}
($\vec v = -2\vec u$).

(c) $3 \times 3 - 1 \times 1 = 8 \neq 0$: not \omterm{def:g10:vectors:collinear}{collinear}.
\end{solution}

\begin{solution}{exo:g10:vectors:5}
$\vect{AB}\,(4, -2)$, and $\vect{DC}\,(6 - x_D,\ 3 - y_D)$. The condition
$\vect{AB} = \vect{DC}$ gives $6 - x_D = 4$ and $3 - y_D = -2$, so
$D(2, 5)$.
\end{solution}

\begin{solution}{exo:g10:vectors:6}
\emph{1.} $\vect{AB}\,(3, 2)$ and $\vect{AC}\,(6, -2)$, so
$\vect{AB} + \vect{AC} = (9, 0)$ and
$M = (-2 + 9,\ 1 + 0) = (7, 1)$.

\emph{2.} By the parallelogram rule, $\vect{AM} = \vect{AB} + \vect{AC}$
means that $ABMC$ is a parallelogram ($M$ is the fourth vertex opposite
$A$). One can check: $\vect{AB}\,(3,2)$ and $\vect{CM}\,(7-4,\ 1-(-1)) =
(3, 2)$ are equal.
\end{solution}

\begin{solution}{exo:g10:vectors:7}
First triple: $\vect{AB}\,(3, 2)$, $\vect{AC}\,(9, 6)$;
$3 \times 6 - 2 \times 9 = 0$: aligned ($\vect{AC} = 3\vect{AB}$).

Second triple: $\vect{AB}\,(2, -1)$, $\vect{AC}\,(8, -4)$;
$2 \times (-4) - (-1) \times 8 = -8 + 8 = 0$: aligned as well
($\vect{AC} = 4\vect{AB}$).
\end{solution}

\begin{solution}{exo:g10:vectors:8}
$\vect{AB}\,(4, 2)$ and $\vect{DC}\,(6-2,\ 1-(-1)) = (4, 2)$: the two
\omterm{def:g10:vectors:vector}{vectors} are \emph{equal}, so $ABCD$ is a parallelogram, and in particular
$(AB) \parallel (CD)$. The parallelogram property is the stronger one: it
implies the parallelism (collinearity of the \omterm{def:g10:vectors:vector}{vectors}), not conversely.
\end{solution}

\begin{solution}{exo:g10:vectors:9}
\emph{1.} $\vect{AB}\,(6, 3)$, so
$\vect{AI} = \frac23\vect{AB} = (4, 2)$ and $I(4, 2)$.

\emph{2.} $\vect{CB}\,(6-2,\ 3-5) = (4, -2)$, so
$\vect{CJ} = \frac23\vect{CB} = \left(\frac83, -\frac43\right)$ and
$J\left(2 + \frac83,\ 5 - \frac43\right) = \left(\frac{14}{3},
\frac{11}{3}\right)$.

\emph{3.} $\vect{IJ}\,\left(\frac{14}{3} - 4,\ \frac{11}{3} - 2\right) =
\left(\frac23, \frac53\right)$ and $\vect{AC}\,(2, 5)$. Collinearity
criterion:
\[
\frac23 \times 5 - \frac53 \times 2 = \frac{10}{3} - \frac{10}{3} = 0 :
\]
the \omterm{def:g10:vectors:vector}{vectors} are \omterm{def:g10:vectors:collinear}{collinear}, so $(IJ)$ is parallel to $(AC)$. (Both $I$ and
$J$ sit one third of the way from the side towards $B$: the small
triangle $IBJ$ is a reduction of $ABC$.)
\end{solution}

\begin{solution}{exo:g10:vectors:10}
The \omterm{prop:g10:coordgeom:midpoint}{midpoints} are
\[
I\left(\tfrac{x_A + x_B}{2}, \tfrac{y_A + y_B}{2}\right), \quad
J\left(\tfrac{x_B + x_C}{2}, \tfrac{y_B + y_C}{2}\right), \quad
K\left(\tfrac{x_C + x_D}{2}, \tfrac{y_C + y_D}{2}\right), \quad
L\left(\tfrac{x_D + x_A}{2}, \tfrac{y_D + y_A}{2}\right).
\]
First \omterm{def:g10:coordgeom:system}{coordinates} of the two \omterm{def:g10:vectors:vector}{vectors}:
\[
x_J - x_I = \frac{x_B + x_C}{2} - \frac{x_A + x_B}{2}
= \frac{x_C - x_A}{2},
\qquad
x_K - x_L = \frac{x_C + x_D}{2} - \frac{x_D + x_A}{2}
= \frac{x_C - x_A}{2},
\]
and the same computation works for the second \omterm{def:g10:coordgeom:system}{coordinates}. So
$\vect{IJ} = \vect{LK}$ (both equal $\frac12\vect{AC}$), and $IJKL$ is a
parallelogram, whatever the quadrilateral $ABCD$ looks like.
\end{solution}

\begin{solution}{pb:g10:vectors:1}
\textbf{1.} $\vect{AB} + \vect{BC} + \vect{CD} = \vect{AD}$;
\quad $\vect{MA} - \vect{MB} = \vect{MA} + \vect{BM} =
\vect{BA}$; \quad $\vect{AB} + \vect{BC} + \vect{CD} +
\vect{DA} = \vect{AA} = \vec 0$ (reordering the third \omterm{def:g10:vectors:sum}{sum}).

\textbf{2.} $\vect{OB} - \vect{OA} = \vect{AO} + \vect{OB} =
\vect{AB}$ by Chasles.

\textbf{3.} $I$ is the \omterm{prop:g10:coordgeom:midpoint}{midpoint} of $[AB]$ exactly when
$\vect{IA} = -\vect{IB}$, i.e.\
$\vect{IA} + \vect{IB} = \vec 0$. Then, for any $O$:
$\vec 0 = \vect{IO} + \vect{OA} + \vect{IO} + \vect{OB}$, so
$2\,\vect{OI} = \vect{OA} + \vect{OB}$ --- and conversely.

\textbf{4.} $\vect{AB} = (4, 2)$, and $D$ satisfies
$\vect{DC} = \vect{AB}$: $D = C - (4, 2) = (2, -2)$ --- the
same $D$ as \cref{pb:g10:coordgeom:1}, one line faster.

\textbf{5.} The translation of \omterm{def:g10:vectors:vector}{vector} $\vec u$ --- exactly the
transformation produced by two parallel mirrors or two
half-turns. \omterm{def:g10:functions:function}{Image} of $(1, 2)$: $(4, 1)$.

\textbf{6.} \omterm{def:g10:algebra:expand}{Expanding} each $\vect{GX}$ as
$\vect{GO} + \vect{OX}$:
$3\,\vect{GO} + \vect{OA} + \vect{OB} + \vect{OC} = \vec 0$,
so $\vect{OG} = \frac13(\vect{OA} + \vect{OB} + \vect{OC})$:
one and only one such point, whose \omterm{def:g10:coordgeom:system}{coordinates} are the averages
of the three vertices'.

\textbf{7.} $\vect{AB} + \vect{AC} =
(\vect{AG} + \vect{GB}) + (\vect{AG} + \vect{GC})
= 2\vect{AG} + (\vect{GB} + \vect{GC})
= 2\vect{AG} - \vect{GA} = 3\vect{AG}$, whence
$\vect{AG} = \frac13(\vect{AB} + \vect{AC})$. The \omterm{prop:g10:coordgeom:midpoint}{midpoint}
formula (question 3, seen from $A$) gives
$\vect{AK} = \frac12(\vect{AB} + \vect{AC})$. Comparing:
$\vect{AG} = \frac23\vect{AK}$ --- $G$ lies on the median
$[AK]$, two thirds of the way from $A$.

\textbf{8.} The defining condition treats $A$, $B$, $C$
identically, so the same computation from $B$ or from $C$
places the \emph{same} $G$ at two thirds of those medians:
concurrency costs nothing. The parallelogram proof was
ingenious but ad hoc; \omterm{def:g10:coordgeom:system}{coordinates} compute but obscure;
\omterm{def:g10:vectors:vector}{vectors} make the symmetry visible and the proof two lines.

\textbf{9.} $G\left(\frac{1+5+3}{3}, \frac{1+2+6}{3}\right) =
(3, 3)$; $K(4, 4)$; $\vect{AG} = (2, 2)$ and
$\vect{AK} = (3, 3)$: determinant
$2 \times 3 - 2 \times 3 = 0$, \omterm{def:g10:vectors:collinear}{collinear}, and indeed
$\vect{AG} = \frac23\vect{AK}$.

\textbf{10.} $G'\left(\frac{2 + 5 + 3}{4}, \frac{2 + 2 +
6}{4}\right) = (2.5,\ 2.5)$: pulled from $G(3,3)$ towards the
doubled vertex $A(1,1)$, as a balance point should be.

\textbf{11.} $3 \times 4 - (-2)(-6) = 12 - 12 = 0$: \omterm{def:g10:vectors:collinear}{collinear}
($\vec v = -2\vec u$). $2 \times 9 - 5 \times 4 = -2 \neq 0$:
not \omterm{def:g10:vectors:collinear}{collinear}.

\textbf{12.} $\vect{AB} = (2, 3)$, $\vect{AC} = (6, 9)$:
determinant $2 \times 9 - 3 \times 6 = 0$: aligned.

\textbf{13.} $\vect{OI} = \frac12(\vect{OA} + \vect{OB})$ and
$\vect{OK} = \frac12(\vect{OC} + \vect{OD})$, so the \omterm{prop:g10:coordgeom:midpoint}{midpoint}
of $[IK]$ has position \omterm{def:g10:vectors:vector}{vector}
$\frac12(\vect{OI} + \vect{OK}) = \frac14(\vect{OA} +
\vect{OB} + \vect{OC} + \vect{OD})$ --- and the computation
for $[JL]$ gives the very same \omterm{def:g10:vectors:vector}{vector}. The two bimedians (and,
by Varignon, the parallelogram $IJKL$'s diagonals) all share
this center: the quadrilateral's centroid.

\textbf{14.} $\vect{MN} = \vect{AN} - \vect{AM} =
\frac13\vect{AC} - \frac13\vect{AB} = \frac13\vect{BC}$:
so $(MN) \parallel (BC)$ and $MN = \frac13 BC$.

\textbf{15.} With $\vect{AM} = k\,\vect{AB}$ and
$\vect{AN} = k\,\vect{AC}$: $\vect{MN} = k\,\vect{BC}$ --- one
line containing the \omterm{prop:g10:coordgeom:midpoint}{midpoint} theorem ($k = \frac12$) and
Thales (any $k$): the ratio passes through the subtraction
untouched.

\textbf{16.} Resultant $(3, 4)$, magnitude
$\sqrt{9 + 16} = 5$ km/h: the boat actually travels
north-east-by-north along the $3$--$4$--$5$ triangle's
hypotenuse.

\textbf{17.} $\vec u + \vec v + \vec w = (0, 0)$: the ring
stays put --- equilibrium. Along a closed polygon
$A_1 A_2 \dots A_n A_1$, Chasles telescopes the \omterm{def:g10:vectors:sum}{sum} to
$\vect{A_1 A_1} = \vec 0$: going around brings you back.

\textbf{18.} $\vec w = -(\vec u + \vec v) = -(3, 2) =
(-3, -2)$.

\textbf{19.} Resultant $(1.5,\ 2)$, magnitude
$\sqrt{2.25 + 4} = 2.5$ m/s. Crossing: $50$ m at $2$ m/s
across: $25$ s; drift: $1.5 \times 25 = 37.5$ m downstream.

\textbf{20.} Re-proved this weekend: the centroid's two-thirds
theorem (question 7), the bimedian/Varignon center (question
13), Thales--midpoint (questions 14--15). What arrows still
cannot do: measure \emph{angles} (and hence lengths in
arbitrary directions) --- the scalar product of grade 11 is
precisely the missing instrument.
\end{solution}
