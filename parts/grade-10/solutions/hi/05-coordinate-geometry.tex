\begin{solution}{exo:g10:coordgeom:1}
\omterm{prop:g10:coordgeom:midpoint}{मध्य}: $\left(\frac{2 + (-4)}{2}, \frac{5 + 1}{2}\right) = (-1, 3)$.
दूरी:
\[
AB = \sqrt{(-4 - 2)^2 + (1 - 5)^2} = \sqrt{36 + 16} = \sqrt{52}
= 2\sqrt{13}.
\]
\end{solution}

\begin{solution}{exo:g10:coordgeom:2}

$\frac{-2 + y_B}{2} = 2$, इसलिए $ x_B = -1$ और $ y_B = 6$: $ B(-1, 6)$.
\end{solution}

\begin{solution}{exo:g10:coordgeom:3}
$ OP = \sqrt{16 + 1} = \sqrt{17}$; $ OQ = \sqrt{9 + 4} = \sqrt{13}$;
$ OR = \sqrt{0 + 25} = 5 = \sqrt{25}$. तब से $13 < 17 < 25$, वर्ग
जड़ होना \omterm{def:g10:functions:variations}{की बढ़ती}, $ Q $ मूल के सबसे निकट है.
\end{solution}

\begin{solution}{exo:g10:coordgeom:4}
वर्ग लंबाई:
\begin{align*}
AB^2 &= (5-1)^2 + (4-2)^2 = 16 + 4 = 20, \\
BC^2 &= (7-5)^2 + (0-4)^2 = 4 + 16 = 20, \\
AC^2 &= (7-1)^2 + (0-2)^2 = 36 + 4 = 40 .
\end{align*}
$ AB = BC $: त्रिभुज समद्विबाहु है $ B $. इसके अतिरिक्त
$ AB^2 + BC^2 = 40 = AC^2$: पाइथागोरस प्रमेय के व्युत्क्रम द्वारा
पर समकोण है $ B $ भी।
\end{solution}

\begin{solution}{exo:g10:coordgeom:5}
\emph{1.} \omterm{prop:g10:coordgeom:midpoint}{मध्य} का $[AC]$:
$\left(\frac{-2+4}{2}, \frac{1+1}{2}\right) = (1, 1)$; \omterm{prop:g10:coordgeom:midpoint}{मध्य} का
$[BD]$: $\left(\frac{1+1}{2}, \frac{3+(-1)}{2}\right) = (1, 1)$. वही
\omterm{prop:g10:coordgeom:midpoint}{मध्य}: $ ABCD $ एक समांतर चतुर्भुज है.

\emph{2.} $ AB^2 = 3^2 + 2^2 = 13$ और $ BC^2 = 3^2 + (-4)^2 = 25$:
$ AB \neq BC $, एक समचतुर्भुज नहीं.

\emph{3.} $ AC^2 = 6^2 + 0^2 = 36$ और $ BD^2 = 0^2 + (-4)^2 = 16$:
$ AC \neq BD $, कोई आयत नहीं. $ ABCD $ एक सादा समांतर चतुर्भुज है.
\end{solution}

\begin{solution}{exo:g10:coordgeom:6}
वर्ग दूरी की गणना करें $\Omega(2,1)$ और तुलना करें $5^2 = 25$:
\[
\Omega A^2 = 3^2 + 4^2 = 25, \qquad
\Omega B^2 = (-3)^2 + (-4)^2 = 25, \qquad
\Omega C^2 = 4^2 + 1^2 = 17 .
\]
$ A $ और $ B $ घेरे पर हैं; $ C $ इसके अंदर है ($17 < 25$).
\end{solution}

\begin{solution}{exo:g10:coordgeom:7}
$ MA^2 = (x - 1)^2 + 1$ और $ MB^2 = (x - 7)^2 + 25$. सेटिंग
$ MA^2 = MB^2$:
\[
x^2 - 2x + 2 = x^2 - 14x + 74
\ \Longleftrightarrow\
12x = 72
\ \Longleftrightarrow\
x = 6 .
\]
ऐसा अनोखा बिंदु है $ M(6, 0)$.
\end{solution}

\begin{solution}{exo:g10:coordgeom:8}
\emph{1.} $ I = \left(\frac{0+4}{2}, \frac{3+1}{2}\right) = (2, 2)$.

\emph{2.} $ M(2,2)$ वास्तव में है $ I $. और
$ MA^2 = (0-2)^2 + (3-2)^2 = 5$, $ MB^2 = (4-2)^2 + (1-2)^2 = 5$: द
\omterm{prop:g10:coordgeom:midpoint}{मध्य} जैसा कि अपेक्षित था, दोनों अंतिम बिंदुओं से समान दूरी पर है।

\emph{3.} देखो के लिए $ N(0, y)$ साथ $ NA^2 = NB^2$:
\[
(3 - y)^2 = 16 + (1 - y)^2
\ \Longleftrightarrow\
9 - 6y = 16 + 1 - 2y
\ \Longleftrightarrow\
-4y = 8,
\]
इसलिए $ y = -2$: मुद्दा $ N(0, -2)$.
\end{solution}

\begin{solution}{exo:g10:coordgeom:9}
\begin{align*}
AB &= \sqrt{4^2 + 2^2} = \sqrt{20} = 2\sqrt5, \\
BC &= \sqrt{8^2 + 4^2} = \sqrt{80} = 4\sqrt5, \\
AC &= \sqrt{12^2 + 6^2} = \sqrt{180} = 6\sqrt5 .
\end{align*}
तब $ AB + BC = 2\sqrt5 + 4\sqrt5 = 6\sqrt5 = AC $: त्रिकोण
असमानता एक समानता है, इसलिए $ B $ खंड पर स्थित है $[AC]$ --- द
तीन बिंदु संरेखित हैं.
\end{solution}

\begin{solution}{exo:g10:coordgeom:10}
 \omterm{prop:g10:coordgeom:midpoint}{मध्य} का $[AB]$ है $ I\left(\frac a2, \frac b2\right)$. तब
\[
OI^2 = \frac{a^2}{4} + \frac{b^2}{4},
\qquad
IA^2 = \left(a - \frac a2\right)^2 + \left(0 - \frac b2\right)^2
= \frac{a^2}{4} + \frac{b^2}{4},
\]
और $ IB^2 = \left(0 - \frac a2\right)^2 + \left(b - \frac b2\right)^2 =
\frac{a^2}{4} + \frac{b^2}{4}$ भी। तीन वर्ग दूरियां हैं
बराबर, तो $ OI = IA = IB $: द \omterm{prop:g10:coordgeom:midpoint}{मध्य} दाईं ओर के कर्ण का
त्रिकोण $ OAB $ तीनों शीर्षों से समान दूरी पर है (यह केंद्र है)।
इसके परिबद्ध वृत्त का)।
\end{solution}

\begin{solution}{pb:g10:coordgeom:1}
\textbf{1.} $\Omega M = 5 \iff \Omega M^2 = 25 \iff
(x - 2)^2 + (y - 1)^2 = 25$ दूरी सूत्र द्वारा
(\cref{thm:g10:coordgeom:distance}) --- वर्ग करना हानिरहित है,
दोनों पक्ष सकारात्मक हैं।

\textbf{2.} $ A $: $(5-2)^2 + (5-1)^2 = 9 + 16 = 25$: पर
वृत्त. $ B $: $16 + 9 = 25$: उस पर भी. $ D $: $4 + 9 = 13 < 25$:
अंदर।

\textbf{3.} $ x^2 - 6x = (x - 3)^2 - 9$ और
$ y^2 + 4y = (y + 2)^2 - 4$, इतना \omterm{def:g10:algebra:equation}{समीकरण} पढ़ता
$(x - 3)^2 + (y + 2)^2 = 25$: केंद्र $(3, -2)$, त्रिज्या $5$.

\textbf{4.} $(x - 3)^2 + (y + 2)^2 = -1$: वर्गों का योग है
कभी भी नकारात्मक नहीं --- कोई भी बिंदु इसे संतुष्ट नहीं करता: खाली सेट। में
सामान्य $(x - h)^2 + (y - k)^2 = c $ त्रिज्या का एक वृत्त है
$\sqrt c $ अगर $ c > 0$, एकल बिंदु $(h, k)$ अगर $ c = 0$, और
खाली अगर $ c < 0$.

\textbf{5.} साथ $ y = 3$: $(x - 3)^2 + 25 = 25$, इसलिए
$(x - 3)^2 = 0$: एकल बिंदु $(3, 3)$. एक संपर्क बिंदु:
लाइन है \emph{स्पर्शरेखा} वृत्त को.

\textbf{6.} $ MA^2 + MB^2 = (x + r)^2 + y^2 + (x - r)^2 + y^2
= 2x^2 + 2y^2 + 2r^2 = 2r^2 + 2r^2 = 4r^2$ (का उपयोग करते हुए
$ x^2 + y^2 = r^2$). तब से $ AB^2 = (2r)^2 = 4r^2$, त्रिकोण
$ AMB $ संतुष्ट $ MA^2 + MB^2 = AB^2$: समकोण पर $ M $
(पाइथागोरस का व्युत्क्रम) --- तीन पंक्तियाँ, जैसा कि वादा किया गया था।

\textbf{7.} $ MA^2 + MB^2 = (x + a)^2 + y^2 + (x - a)^2 + y^2
= 2x^2 + 2y^2 + 2a^2 = 2\,MI^2 + 2a^2$, और
$\frac{AB^2}{2} = \frac{(2a)^2}{2} = 2a^2$. पहचान साबित हुई.

\textbf{8.} $2\,MI^2 + \frac{36}{2} = 26$ देता है $ MI^2 = 4$: द
केंद्र का वृत्त \omterm{prop:g10:coordgeom:midpoint}{मध्य} $ I $ और त्रिज्या $2$.

\textbf{9.} $ MA^2 = 4\,MB^2$:
$ x^2 + y^2 = 4\left((x - 3)^2 + y^2\right)$, अर्थात\
$3x^2 + 3y^2 - 24x + 36 = 0$, या
$ x^2 + y^2 - 8x + 12 = 0$: पूरा करना,
$(x - 4)^2 + y^2 = 4$ --- केंद्र का वृत्त $(4, 0)$ और
त्रिज्या $2$ (द \emph{अपोलोनियस का चक्र} अनुपात का $2$).

\textbf{10.} पुल निर्माणों को परिवर्तित करता है
संगणनाएँ: कोई सहायक आयत नहीं, कोई केस विश्लेषण नहीं --- एक
\omterm{def:g10:algebra:expand}{विस्तार} और प्रमेय समाप्त हो जाता है। कीमत: गणना
कम ज्यामितीय अंतर्दृष्टि रखें; शास्त्रीय प्रमाण दिखाता है
\emph{क्यों}, बीजगणितीय दिखाता है \emph{वह} --- एक कामकाजी
गणितज्ञ दोनों रखता है.

\textbf{11.} $ x^2 + y^2 = 25$ और $(x - 6)^2 + y^2 = 25$.
घटाना: $ x^2 - (x - 6)^2 = 0$, यानी\ $12x - 36 = 0$:
$ x = 3$. तब $9 + y^2 = 25$: $ y = \pm 4$. उम्मीदवार
$(3, 4)$ और $(3, -4)$.

\textbf{12.} से $(3, 4)$ को $ R(0, 8)$:
$\sqrt{9 + 16} = 5$ --- मिलान. से $(3, -4)$:
$\sqrt{9 + 144} \approx 12.4$ --- गलत। आप यहां पर है $(3, 4)$.

\textbf{13.} दोनों \omterm{def:g10:algebra:equation}{समीकरण} वही ले जाओ $ x^2 + y^2$ (या साथ
\omterm{def:g10:algebra:expand}{विस्तार} के बाद समान गुणांक): घटाव रद्द कर देता है
वर्ग और पत्तियां ए \omterm{def:g10:algebra:equation}{समीकरण} डिग्री एक की --- एक पंक्ति। कब
वृत्त दो बार मिलते हैं, वह रेखा दोनों \omterm{def:g10:numbers:interunion}{मिलन} से होकर गुजरती है
अंक: यह सामान्य राग की रेखा है।

\textbf{14.} चार अज्ञात: तीन \omterm{def:g10:coordgeom:system}{COORDINATES} इसके अलावा
रिसीवर की घड़ी की त्रुटि (एक सस्ती घड़ी, जिसे ठीक किया गया है)
गणित) प्रत्येक उपग्रह एक आपूर्ति करता है \omterm{def:g10:algebra:equation}{समीकरण}, तो चार
उपग्रह चार देते हैं \omterm{def:g10:algebra:equation}{समीकरण} चार अज्ञातों के लिए --- और अधिक के साथ
उपग्रह परिशुद्धता में सुधार कर रहे हैं।

\textbf{15.} घटाव अब देता है
$ x = \frac{5.1^2 - 5^2 + 36}{12} = \frac{37.01}{12} \approx
3.084$: ए $10$ एक दूरी में सेमी त्रुटि ने अनुमान को आगे बढ़ा दिया
के बारे में $8$ यहां सेमी. त्रुटियाँ हर सूत्र के माध्यम से फैलती हैं ---
की अंतराल-अंकगणितीय सावधानी \cref{pb:g10:numbers:1},
अब उसके पास \omterm{def:g10:coordgeom:system}{COORDINATES}.

\textbf{16.} $ B' (7, -3)$. रेखा $(AB')$ है \omterm{def:g10:reffunc:affine}{ढलान}
$\frac{-3 - 5}{7 - 1} = -\frac43$ और पार $ y = 0$ पर
$ x = 1 + \frac{5}{8} \times 6 = 4.75$: पर पियें
$ P(4.75,\ 0)$. \omterm{def:g10:functions:extrema}{न्यूनतम} लंबाई: $ AP + PB = AP + PB' = AB' =
\sqrt{6^2 + 8^2} = 10$.

\textbf{17.} के जरिए $ Q(3, 0)$:
$ AQ + QB = \sqrt{4 + 25} + \sqrt{16 + 9} = \sqrt{29} + 5
\approx 10.39 > 10$. परावर्तित सीधी रेखा जीतती है।

\textbf{18.} विकर्ण \omterm{prop:g10:coordgeom:midpoint}{मध्य} बिन्दुओं: $[AC]$: $(3, 3)$; $[BD]$:
$(3, 3)$ --- बराबर: समांतर चतुर्भुज. पक्ष:
$ AB = \sqrt{10} = AD $: रोम्बस. विकर्ण:
$ AC = \sqrt{32} \neq BD = \sqrt{8}$: इसलिए, यह एक आयत नहीं है
एक वर्ग नहीं. फैसला: एक समचतुर्भुज, और कुछ नहीं।

\textbf{19.} $ D = A + C - B = (-1 + 6 - 3,\ 2 + 0 - 4) =
(2, -2)$: तब $[AC]$ और $[BD]$ बांटो \omterm{prop:g10:coordgeom:midpoint}{मध्य}
$\left(\frac52, 1\right)$.

\textbf{20.} \omterm{prop:g10:coordgeom:midpoint}{मध्य} सूत्र: समांतर चतुर्भुज और केंद्र
(प्रश्न 18,19). दूरी सूत्र: वृत्त, लोकी, त्रिकोण
प्रकृति, सबसे छोटे रास्ते (प्रश्न 1--9, 16)। का घटाव
वृत्त \omterm{def:g10:algebra:equation}{समीकरण}: वह रेखा जो किसी स्थिति का पता लगाती है (प्रश्न)।
11--15). किट से गायब: का एक साफ़ बीजगणित
\emph{दिशा-निर्देश} --- वैक्टर --- और का \emph{\omterm{def:g10:reffunc:affine}{ढलान}ों} ---
रेखा \omterm{def:g10:algebra:equation}{समीकरण} और सिस्टम: अगले दो अध्याय आपूर्ति करते हैं
बिलकुल वैसा ही.
\end{solution}
