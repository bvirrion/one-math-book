\begin{solution}{exo:g10:vectors:1}
$\vect{AB} + \vect{BD} = \vect{AD}$ (चेसल्स)।

$\vect{KL} + \vect{LM} + \vect{MK} = \vect{KK} = \vec 0$.

$\vect{AC} - \vect{BC} = \vect{AC} + \vect{CB} = \vect{AB}$.

$\vect{AB} + \vect{CA} = \vect{CA} + \vect{AB} = \vect{CB}$.
\end{solution}

$\vect{BC}\,(-1-5,\ 4-3) = (-6, 1)$; \quad
$\vect{AC}\,(-1-2,\ 4-1) = (-3, 3)$.

\end{solution}

\begin{solution}{exo:g10:vectors:3}
$\vec u + \vec v = (2 + (-1),\ -3 + 4) = (1, 1)$.

$3\vec u = (6, -9)$.

$2\vec u - \vec v = (4 - (-1),\ -6 - 4) = (5, -10)$.

$\norm{\vec u} = \sqrt{2^2 + (-3)^2} = \sqrt{13}$.
\end{solution}

\begin{solution}{exo:g10:vectors:4}
(ए) $4 \times 9 - 6 \times 6 = 36 - 36 = 0$: \omterm{def:g10:vectors:collinear}{समरेख}
($\vec v = \frac32 \vec u $).

(बी) $2 \times 10 - (-5) \times (-4) = 20 - 20 = 0$: \omterm{def:g10:vectors:collinear}{समरेख}
($\vec v = -2\vec u $).

(सी) $3 \times 3 - 1 \times 1 = 8 \neq 0$: नहीं \omterm{def:g10:vectors:collinear}{समरेख}.
\end{solution}

\begin{solution}{exo:g10:vectors:5}
$\vect{AB}\,(4, -2)$, और $\vect{DC}\,(6 - x_D,\ 3 - y_D)$. स्थिति
$\vect{AB} = \vect{DC}$ देता है $6 - x_D = 4$ और $3 - y_D = -2$, इसलिए

\end{solution}

\begin{solution}{exo:g10:vectors:6}
\emph{1.} $\vect{AB}\,(3, 2)$ और $\vect{AC}\,(6, -2)$, इसलिए
$\vect{AB} + \vect{AC} = (9, 0)$ और
$ M = (-2 + 9,\ 1 + 0) = (7, 1)$.

\emph{2.} समांतर चतुर्भुज नियम से, $\vect{AM} = \vect{AB} + \vect{AC}$
मतलब कि $ ABMC $ एक समांतर चतुर्भुज है ($ M $ विपरीत चौथा शीर्ष है
$ A $). कोई जाँच सकता है: $\vect{AB}\,(3,2)$ और $\vect{CM}\,(7-4,\ 1-(-1)) =
(3, 2)$ बराबर हैं.
\end{solution}

\begin{solution}{exo:g10:vectors:7}
पहला त्रिक: $\vect{AB}\,(3, 2)$, $\vect{AC}\,(9, 6)$;
$3 \times 6 - 2 \times 9 = 0$: संरेखित ($\vect{AC} = 3\vect{AB}$).

दूसरा त्रिक: $\vect{AB}\,(2, -1)$, $\vect{AC}\,(8, -4)$;
$2 \times (-4) - (-1) \times 8 = -8 + 8 = 0$: भी संरेखित
($\vect{AC} = 4\vect{AB}$).
\end{solution}

\begin{solution}{exo:g10:vectors:8}
$\vect{AB}\,(4, 2)$ और $\vect{DC}\,(6-2,\ 1-(-1)) = (4, 2)$: दो
वैक्टर हैं \emph{बराबर}, इसलिए $ ABCD $ एक समांतर चतुर्भुज है, और विशेष रूप से
$(AB) \parallel (CD)$. समांतर चतुर्भुज गुण अधिक मजबूत है: यह
इसका तात्पर्य समानता (समानता) से है वैक्टर), इसके विपरीत नहीं.
\end{solution}

\begin{solution}{exo:g10:vectors:9}
\emph{1.} $\vect{AB}\,(6, 3)$, इसलिए
$\vect{AI} = \frac23\vect{AB} = (4, 2)$ और $ I(4, 2)$.

\emph{2.} $\vect{CB}\,(6-2,\ 3-5) = (4, -2)$, इसलिए
$\vect{CJ} = \frac23\vect{CB} = \left(\frac83, -\frac43\right)$ और
$ J\left(2 + \frac83,\ 5 - \frac43\right) = \left(\frac{14}{3},
\frac{11}{3}\right)$.

\emph{3.} $\vect{IJ}\,\left(\frac{14}{3} - 4,\ \frac{11}{3} - 2\right) =
\left(\frac23, \frac53\right)$ और $\vect{AC}\,(2, 5)$. संरेखता
कसौटी:
\[
\frac23 \times 5 - \frac53 \times 2 = \frac{10}{3} - \frac{10}{3} = 0 :
\]
 वैक्टर हैं \omterm{def:g10:vectors:collinear}{समरेख}, इसलिए $(IJ)$ के समानांतर है $(AC)$. (दोनों $ I $ और
$ J $ बगल से एक तिहाई दूरी पर बैठें $ B $: छोटा
त्रिकोण $ IBJ $ की कमी है $ ABC $.)
\end{solution}

\begin{solution}{exo:g10:vectors:10}
 \omterm{prop:g10:coordgeom:midpoint}{मध्य} बिन्दुओं हैं
\[
I\left(\tfrac{x_A + x_B}{2}, \tfrac{y_A + y_B}{2}\right), \quad
J\left(\tfrac{x_B + x_C}{2}, \tfrac{y_B + y_C}{2}\right), \quad
K\left(\tfrac{x_C + x_D}{2}, \tfrac{y_C + y_D}{2}\right), \quad
L\left(\tfrac{x_D + x_A}{2}, \tfrac{y_D + y_A}{2}\right).
\]
पहला \omterm{def:g10:coordgeom:system}{COORDINATES} दोनों में से वैक्टर:
\[
x_J - x_I = \frac{x_B + x_C}{2} - \frac{x_A + x_B}{2}
= \frac{x_C - x_A}{2},
\qquad
x_K - x_L = \frac{x_C + x_D}{2} - \frac{x_D + x_A}{2}
= \frac{x_C - x_A}{2},
\]
और वही गणना दूसरे के लिए भी काम करती है \omterm{def:g10:coordgeom:system}{COORDINATES}. इसलिए
$\vect{IJ} = \vect{LK}$ (दोनों बराबर $\frac12\vect{AC}$), और $ IJKL $ एक है
समांतर चतुर्भुज, चाहे चतुर्भुज कोई भी हो $ ABCD $ की तरह लगता है।
\end{solution}

\begin{solution}{pb:g10:vectors:1}
\textbf{1.} $\vect{AB} + \vect{BC} + \vect{CD} = \vect{AD}$;
\quad $\vect{MA} - \vect{MB} = \vect{MA} + \vect{BM} =
\vect{BA}$; \quad $\vect{AB} + \vect{BC} + \vect{CD} +
\vect{DA} = \vect{AA} = \vec 0$ (तीसरे को पुनः व्यवस्थित करना \omterm{def:g10:vectors:sum}{जोड़}).

\textbf{2.} $\vect{OB} - \vect{OA} = \vect{AO} + \vect{OB} =
\vect{AB}$ चैसल्स द्वारा.

\textbf{3.} $ I $ है \omterm{prop:g10:coordgeom:midpoint}{मध्य} का $[AB]$ बिल्कुल कब
$\vect{IA} = -\vect{IB}$, अर्थात\
$\vect{IA} + \vect{IB} = \vec 0$. फिर, किसी के लिए $ O $:
$\vec 0 = \vect{IO} + \vect{OA} + \vect{IO} + \vect{OB}$, इसलिए
$2\,\vect{OI} = \vect{OA} + \vect{OB}$ --- और इसके विपरीत.

\textbf{4.} $\vect{AB} = (4, 2)$, और $ D $ संतुष्ट
$\vect{DC} = \vect{AB}$: $ D = C - (4, 2) = (2, -2)$ --- जो उसी $ D $ जैसा \cref{pb:g10:coordgeom:1}, एक पंक्ति तेज़।

\textbf{5.} का अनुवाद \omterm{def:g10:vectors:vector}{वेक्टर} $\vec u $ --- बिलकुल
दो या दो समानांतर दर्पणों द्वारा उत्पन्न परिवर्तन
आधा मोड़. \omterm{def:g10:functions:function}{छवि} का $(1, 2)$: $(4, 1)$.

\textbf{6.} \omterm{def:g10:algebra:expand}{विस्तार} प्रत्येक $\vect{GX}$ जैसा
$\vect{GO} + \vect{OX}$:
$3\,\vect{GO} + \vect{OA} + \vect{OB} + \vect{OC} = \vec 0$,
इसलिए $\vect{OG} = \frac13(\vect{OA} + \vect{OB} + \vect{OC})$:
एक और केवल एक ही ऐसा बिंदु, जिसका \omterm{def:g10:coordgeom:system}{COORDINATES} औसत हैं
तीन शीर्षों में से'.

\textbf{7.} $\vect{AB} + \vect{AC} =
(\vect{AG} + \vect{GB}) + (\vect{AG} + \vect{GC})
= 2\vect{AG} + (\vect{GB} + \vect{GC})
= 2\vect{AG} - \vect{GA} = 3\vect{AG}$, कहाँ से
$\vect{AG} = \frac13(\vect{AB} + \vect{AC})$. \omterm{prop:g10:coordgeom:midpoint}{मध्य}
सूत्र (प्रश्न 3, से देखा गया $ A $) देता है
$\vect{AK} = \frac12(\vect{AB} + \vect{AC})$. तुलना:
$\vect{AG} = \frac23\vect{AK}$ --- $ G $ माध्यिका पर स्थित है
$[AK]$, से दो तिहाई रास्ता $ A $.

\textbf{8.} परिभाषित स्थिति व्यवहार करती है $ A $, $ B $, $ C $
समान रूप से, इसलिए समान गणना से $ B $ या से $ C $
स्थानों \emph{वही} $ G $ उन मध्यस्थों के दो तिहाई पर:
समवर्ती लागत कुछ भी नहीं है. समांतर चतुर्भुज प्रमाण था
सरल लेकिन तदर्थ; \omterm{def:g10:coordgeom:system}{COORDINATES} गणना करें लेकिन अस्पष्ट;
वैक्टर समरूपता को दृश्यमान बनाएं और प्रमाण को दो रेखाएं बनाएं।

\textbf{9.} $ G\left(\frac{1+5+3}{3}, \frac{1+2+6}{3}\right) =
(3, 3)$; $ K(4, 4)$; $\vect{AG} = (2, 2)$ और
$\vect{AK} = (3, 3)$: निर्धारक
$2 \times 3 - 2 \times 3 = 0$, \omterm{def:g10:vectors:collinear}{समरेख}, सचमुच
$\vect{AG} = \frac23\vect{AK}$.

\textbf{10.} $ G'\left(\frac{2 + 5 + 3}{4}, \frac{2 + 2 +
6}{4}\right) = (2.5,\ 2.5)$: द्वारा लिया गया $ G(3,3)$ की ओर
दोगुना शीर्ष $ A(1,1)$, जैसा कि एक संतुलन बिंदु होना चाहिए।

\textbf{11.} $3 \times 4 - (-2)(-6) = 12 - 12 = 0$: \omterm{def:g10:vectors:collinear}{समरेख}
($\vec v = -2\vec u $). $2 \times 9 - 5 \times 4 = -2 \neq 0$:
नहीं \omterm{def:g10:vectors:collinear}{समरेख}.

\textbf{12.} $\vect{AB} = (2, 3)$, $\vect{AC} = (6, 9)$:
निर्धारक $2 \times 9 - 3 \times 6 = 0$: संरेखित.

\textbf{13.} $\vect{OI} = \frac12(\vect{OA} + \vect{OB})$ और
$\vect{OK} = \frac12(\vect{OC} + \vect{OD})$, इतना \omterm{prop:g10:coordgeom:midpoint}{मध्य}
का $[IK]$ पद है \omterm{def:g10:vectors:vector}{वेक्टर}
$\frac12(\vect{OI} + \vect{OK}) = \frac14(\vect{OA} +
\vect{OB} + \vect{OC} + \vect{OD})$ --- और गणना
के लिए $[JL]$ वही देता है \omterm{def:g10:vectors:vector}{वेक्टर}. दो द्विमाध्यमिक (और,
वैरिग्नन द्वारा, समांतर चतुर्भुज $ IJKL $के विकर्ण) सभी साझा करते हैं
यह केंद्र: चतुर्भुज का केन्द्रक।

\textbf{14.} $\vect{MN} = \vect{AN} - \vect{AM} =
\frac13\vect{AC} - \frac13\vect{AB} = \frac13\vect{BC}$:
इसलिए $(MN) \parallel (BC)$ और $ MN = \frac13 BC $.

\textbf{15.} साथ $\vect{AM} = k\,\vect{AB}$ और
$\vect{AN} = k\,\vect{AC}$: $\vect{MN} = k\,\vect{BC}$ --- एक
युक्त पंक्ति \omterm{prop:g10:coordgeom:midpoint}{मध्य} प्रमेय ($ k = \frac12$) और
थेल्स (कोई भी) $ k $): अनुपात घटाव से होकर गुजरता है
अछूता.

\textbf{16.} परिणामी $(3, 4)$, परिमाण
$\sqrt{9 + 16} = 5$ किमी/घंटा: नाव वास्तव में यात्रा करती है
उत्तर-पूर्व-उत्तर की ओर $3$--$4$--$5$ त्रिकोण का
कर्ण.

\textbf{17.} $\vec u + \vec v + \vec w = (0, 0)$: अंगूठी
संतुलन बना रहता है। एक बंद बहुभुज के साथ
$ A_1 A_2 \dots A_n A_1$, चैसल्स दूरबीन \omterm{def:g10:vectors:sum}{जोड़} को
$\vect{A_1 A_1} = \vec 0$: घूमना आपको वापस लाता है।

\textbf{18.} $\vec w = -(\vec u + \vec v) = -(3, 2) =
(-3, -2)$.

\textbf{19.} परिणामी $(1.5,\ 2)$, परिमाण
$\sqrt{2.25 + 4} = 2.5$ एमएस। पार करना: $50$ एम एट $2$ एम/एस
पार: $25$ एस; बहाव: $1.5 \times 25 = 37.5$ एम डाउनस्ट्रीम.

\textbf{20.} इस सप्ताह के अंत में पुनः सिद्ध हुआ: केन्द्रक का दो-तिहाई
प्रमेय (प्रश्न 7), द्विमीडियन/वेरिग्नन केंद्र (प्रश्न
13), थेल्स--मध्यबिंदु (प्रश्न 14--15)। अभी भी कौन से तीर हैं
नहीं कर सकते: मापें \emph{एंगल्स} (और इसलिए लंबाई में
मनमानी दिशाएँ) --- ग्रेड 11 का अदिश गुणनफल है
बिल्कुल गायब उपकरण।
\end{solution}
