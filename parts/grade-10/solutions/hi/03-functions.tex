\begin{solution}{exo:g10:functions:10}
\emph{1.} $ f(v) - f(u) = v^2 - u^2 = (v-u)(v+u)$. अगर $0 \leq u < v $, तब
$ v - u > 0$ और $ v + u > 0$ (एक अऋणात्मक और एक धनात्मक संख्या का योग),
तो $ f(v) - f(u) > 0$: $ f $ सख्ती से है \omterm{def:g10:functions:variations}{की बढ़ती} पर $\intco{0}{+\infty}$.

\emph{2.} अगर $ u < v \leq 0$, तब $ v - u > 0$ अभी भी, लेकिन अभी
$ v + u < 0$ (एक गैर-धनात्मक और एक ऋणात्मक संख्या का योग), इसलिए
$ f(v) - f(u) < 0$: $ f $ सख्ती से है \omterm{def:g10:functions:variations}{घटते} पर $\intoc{-\infty}{0}$.
\end{solution}

\begin{solution}{pb:g10:functions:1}
\textbf{1.} \omterm{def:g10:functions:function}{कार्यक्षेत्र}: $ x \neq 0$ (द्वारा विभाजन $ x $). $ f(1) =
\frac12(1 + 2) = \frac32$; $ f(2) = \frac12(2 + 1) = \frac32$;
$ f\!\left(\frac32\right) = \frac12\left(\frac32 +
\frac43\right) = \frac{17}{12}$.

\textbf{2.} $\frac12\left(x + \frac2x\right) = \frac32$ देता है
$ x^2 + 2 = 3x $, यानी\ $ x^2 - 3x + 2 = (x - 1)(x - 2) = 0$: द
पूर्वछवियाँ का $\frac32$ हैं $1$ और $2$ --- प्रश्न 1 के रूप में
घोषणा की.

\textbf{3.} $ f(x) = x $ देता है $ x + \frac2x = 2x $, इसलिए
$\frac2x = x $, $ x^2 = 2$: निश्चित बिंदु $\sqrt2$ और $-\sqrt2$.
धनात्मक इकाई वर्ग का विकर्ण है, \omterm{ex:g10:numbers:classify}{तर्कहीन}
मिडिल स्कूल वॉल्यूम की अतार्किकता सप्ताहांत समस्या से।

\textbf{4.} $1 \to \frac32 \to \frac{17}{12} \to
\frac{577}{408}$: बगुला का अनुमान का $\sqrt2$. उसका
नुस्खा ``औसत अनुमान के साथ $2/$ अनुमान बिल्कुल सही है
\emph{की पुनरावृति \omterm{def:g10:functions:function}{समारोह} $f $} --- और संख्या
प्रक्रिया का पीछा है $ f $'एस \omterm{pb:g10:functions:1}{नियत बिन्दु}.

\textbf{5.}  निश्चित बिंदु के क्रॉसिंग हैं \omterm{def:g10:functions:graph}{ग्राफ}
विकर्ण के साथ $ y = x $: आउटपुट को वापस वॉक में फीड करना
के साथ इंगित करें \omterm{def:g10:functions:graph}{ग्राफ} उस पार की ओर. में
मिडिल स्कूल की मात्रा, पढ़ने की दो-थर्मामीटर सप्ताहांत समस्या $-40^\circ $ वैसा ही था
रूपांतरण के लिए विकर्ण को पार करने का विचार \omterm{def:g10:functions:function}{समारोह}.

\textbf{6.} $ f(v) - f(u) = \frac{v - u}{2} + \frac1v - \frac1u
= \frac{v - u}{2} + \frac{u - v}{uv}
= (v - u)\left(\frac12 - \frac{1}{uv}\right)
= \frac{(v - u)(uv - 2)}{2uv}$. के लिए $0 < u < v \leq \sqrt2$:
$ uv < 2$, कोष्ठक ऋणात्मक है, $ f(v) < f(u)$: \omterm{def:g10:functions:variations}{घटते}.
के लिए $\sqrt2 \leq u < v $: $ uv > 2$: \omterm{def:g10:functions:variations}{की बढ़ती}. मोड़:
$\sqrt2$.

\textbf{7.} घटाना पहले $\sqrt2$, \omterm{def:g10:functions:variations}{की बढ़ती} बाद में: $ f $
उसे प्राप्त कर लेता है \omterm{def:g10:functions:extrema}{न्यूनतम} पर $\sqrt2$, लायक
$ f(\sqrt2) = \frac12\left(\sqrt2 + \frac{2}{\sqrt2}\right)
= \sqrt2$. परिणाम: मशीन का प्रत्येक आउटपुट (सकारात्मक)।
इनपुट) कम से कम है $\sqrt2$ --- क्रैंक के एक मोड़ के बाद,
अनुमान रहते हैं $\intco{\sqrt2}{+\infty}$.

\textbf{8.} अगर $ x > \sqrt2$, तब $ x^2 > 2$, इसलिए
$\frac2x < x $ और औसत $ f(x)$ का $ x $ और $\frac2x $ मै रुक जाना $ x $; और $ f(x) \geq \sqrt2$ प्रश्न 7 द्वारा, साथ
समानता केवल पर $ x = \sqrt2$. इसलिए
$\sqrt2 < f(x) < x $: प्रत्येक पुनरावृत्ति बिना सख्ती से सुधार करती है
ओवरशूटिंग --- प्रश्न 4 का क्रम नीचे की ओर खिसकता है
$\sqrt2$.

\textbf{9.} पर $\intoo{0}{+\infty}$: \omterm{def:g10:functions:variations}{घटते} से
$+\infty $ (पास में $0$) नीचे तक \omterm{def:g10:functions:extrema}{न्यूनतम} $\sqrt2$ पर
$ x = \sqrt2$, तब \omterm{def:g10:functions:variations}{की बढ़ती} बिना बंधन के.

\textbf{10.} $ g(x) = x $ देता है $ x^2 = x $, $ x(x - 1) = 0$: निश्चित
अंक $0$ और $1$. से $0.9$: $0.81$, $0.656$, $0.430$,
$0.185$ --- की ओर सरकना $0$. से $1.1$: $1.21$, $1.464$,
$2.144$, $4.595$ --- अनंत की ओर भागना. \omterm{pb:g10:functions:1}{नियत बिन्दु} $0$
आकर्षित करता है, $1$ विकर्षक: शुरुआत में एक बाल का अंतर, विपरीत
नियति

\textbf{13.} की एक शक्ति $2$ इसकी पूरी सीढ़ी को आधा नीचे गिरा दिया गया है:
$2^k \to 2^{k-1} \to \dots \to 2 \to 1$. तल पर,
$ h(1) = 4$, $ h(4) = 2$, $ h(2) = 1$: उड़ान चक्र में प्रवेश करती है
$4 \to 2 \to 1 \to 4$. निश्चित अंक जरूरत पड़ेगी $\frac n2 = n $
(केवल $ n = 0$) या $3n + 1 = n $ (नकारात्मक): इनमें से कोई नहीं
सकारात्मक \omterm{def:g10:numbers:sets}{पूर्णांकों} --- देखी गई उड़ानें एक पर समाप्त नहीं होतीं \omterm{pb:g10:functions:1}{नियत बिन्दु} लेकिन उस थोड़े में \emph{चक्र}.

\textbf{14.} एक प्रति-उदाहरण या तो एक उड़ान होगी
हमेशा के लिए बढ़ता है (कभी वापस नहीं लौटता)। $1$) या दूसरा चक्र
से अलग करना $4 \to 2 \to 1$. तक सत्यापन $10^{20}$
अपरिमित रूप से कई अपरीक्षित संख्याएँ छोड़ता है --- बिल्कुल
मिडिल स्कूल वॉल्यूम की माचिस-ओरेकल सप्ताहांत समस्या का सर्कल-क्षेत्र पाठ: पर सहमति
बहुत से मामले उन सभी के बारे में कुछ भी साबित नहीं करते हैं।

\textbf{15.} हेरॉन की मशीन के लिए हमारे पास था \emph{संरचना} --- ए
गुणनखंडित अंतर (प्रश्न 6) विविधताओं को प्रकट करता है, a
\omterm{def:g10:functions:extrema}{न्यूनतम}, एक निचोड़ --- जबकि ओले की समता-स्विचिंग
नियम ऐसा कोई हैंडल प्रदान नहीं करता है: किसी को भी वह संरचना नहीं मिली है
इसे वश में करता है.

\textbf{16.} $\sqrt{x - 3}$: आवश्यकताएँ $ x - 3 \geq 0$:
$\intco{3}{+\infty}$. $\frac{1}{x^2 - 9}$: आवश्यकताएँ
$ x \neq \pm 3$: $\R $ से वंचित $-3$ और $3$.
$\sqrt{9 - x^2}$: आवश्यकताएँ $ x^2 \leq 9$: $\intcc{-3}{3}$.

\textbf{17.} $ f(x) = 3$ देता है $ x + \frac2x = 6$, इसलिए
$ x^2 - 6x + 2 = 0$, यानी\ $(x - 3)^2 = 7$:
$ x = 3 - \sqrt7$ या $ x = 3 + \sqrt7$ (दोनों सकारात्मक: दो सटीक
पूर्वछवियाँ).

\textbf{18.} पत्थर ऊपर होने पर भौतिक बोध:
$ H(t) = 5t(4 - t) \geq 0$ के लिए $ t \in \intcc{0}{4}$. पूरा करना:
$ H(t) = 20 - 5(t - 2)^2$: \omterm{def:g10:functions:extrema}{अधिकतम} $20$ एम एट $ t = 2$ एस।
विविधताएँ: \omterm{def:g10:functions:variations}{की बढ़ती} पर $\intcc{0}{2}$ से $0$ को $20$,
\omterm{def:g10:functions:variations}{घटते} पर $\intcc{2}{4}$ वापस $0$.

\textbf{19.} $20t - 5t^2 \geq 15 \iff t^2 - 4t + 3 \leq 0 \iff
(t - 1)(t - 3) \leq 0 \iff t \in \intcc{1}{3}$: दो पूर्ण
ऊपर सेकंड $15$ एम।

\textbf{20.} प्रत्येक मशीन में एक है \omterm{def:g10:functions:function}{कार्यक्षेत्र} (जहां नियम है
अनुमति), एक नियम (सूत्र या समता स्विच), और ए
\omterm{def:g10:functions:graph}{ग्राफ} या इसे संपूर्ण प्रदर्शित करने वाली तालिका। प्रत्येक में से हमने दो से पूछा
महान प्रश्न: \emph{दोहराव कहाँ ले जाता है} (निश्चित
अंक, चक्र, आकर्षण --- बगुला के लिए हल, के लिए खुला
ओले) और \emph{आउटपुट इनपुट के साथ कैसे चलता है}
(विविधताएँ, एक्स्ट्रेमा --- पत्थर का शिखर, बगुला का \omterm{def:g10:functions:extrema}{न्यूनतम}).
आगे के अध्याय दोनों को तीक्ष्ण बनाते हैं: संदर्भ कार्य अगला,
फिर, बाद के वर्षों में, भिन्नता को मापने के लिए व्युत्पन्न और
यह प्रमाणित करने की सीमाएँ कि पुनरावृत्तियाँ कहाँ पहुँचती हैं।
\end{solution}
