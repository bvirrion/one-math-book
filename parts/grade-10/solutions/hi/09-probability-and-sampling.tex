\section*{अध्याय \ref{ch:g10:proba} --- संभाव्यता और नमूनाकरण}

\begin{solution}{exo:g10:proba:1}
हरेक $6$ परिणाम है \omterm{def:g10:proba:distribution}{संभावना} $\frac16$.
``एक मिल रहा है $6$'': $\frac16$.
``विषम संख्या'' $= \{1, 3, 5\}$: $\frac36 = \frac12$.
``ज्यादा से ज्यादा $4$'' $= \{1,2,3,4\}$: $\frac46 = \frac23$.
``एक मिल रहा है $7$'': असंभव \omterm{def:g10:proba:events}{आयोजन}, \omterm{def:g10:proba:distribution}{संभावना} $0$.
\end{solution}

\begin{solution}{exo:g10:proba:2}
वहाँ हैं $10$ समान रूप से संभावित कंचे।
$\P(\text{green}) = \frac{5}{10} = \frac12$.
$\P(\text{not black}) = 1 - \frac{2}{10} = \frac{8}{10} = \frac45$.
हरे और पीले हैं \omterm{def:g10:proba:operations}{असंगत}:
$\P(\text{green or yellow}) = \frac{5}{10} + \frac{3}{10} = \frac{8}{10}
= \frac45$.
\end{solution}

\begin{solution}{exo:g10:proba:3}
होने देना $ S $: ``स्पेनिश का अध्ययन करता है'', $ G $: ``जर्मन पढ़ता है''। तब
$\P(S) = \frac{18}{30}$, $\P(G) = \frac{10}{30}$,
$\P(S \cap G) = \frac{4}{30}$. अतिरिक्त नियम:
\[
\P(S \cup G) = \frac{18}{30} + \frac{10}{30} - \frac{4}{30}
= \frac{24}{30} = \frac45 .
\]
\omterm{def:g10:proba:operations}{पूरक}: $\P(\text{neither}) = 1 - \frac45 = \frac15$.
\end{solution}

\begin{solution}{exo:g10:proba:4}
 \omterm{met:g10:proba:tree}{पेड़} चार पथ हैं HH, HT, TH, TT, प्रत्येक \omterm{def:g10:proba:distribution}{संभावना}
$\frac12 \times \frac12 = \frac14$.
दो सिर: $\frac14$. बिल्कुल एक शीर्ष (HT या TH): $\frac24 = \frac12$.
कम से कम एक सिर: $1 - \P(\text{TT}) = 1 - \frac14 = \frac34$.
\end{solution}

\begin{solution}{exo:g10:proba:5}
पहला ड्रा: लाल के साथ \omterm{def:g10:proba:distribution}{संभावना} $\frac{4}{10}$, सफ़ेद $\frac{6}{10}$.
दूसरा ड्रा \emph{बिना} प्रतिस्थापन: एक लाल रंग के बाद, $3$ लाल और $6$
सफेद बीच में रहते हैं $9$; सफ़ेद होने के बाद, $4$ लाल और $5$ बीच में सफेद $9$.

दो लाल: $\frac{4}{10} \times \frac39 = \frac{12}{90} = \frac{2}{15}$.

अलग-अलग रंग: पथ आरडब्ल्यू और डब्ल्यूआर:
\[
\frac{4}{10} \times \frac69 + \frac{6}{10} \times \frac49
= \frac{24}{90} + \frac{24}{90} = \frac{48}{90} = \frac{8}{15}.
\]
\end{solution}

\begin{solution}{exo:g10:proba:6}
\emph{1.} प्रत्येक पासा स्वतंत्र रूप से प्रत्येक चेहरे को दर्शाता है \omterm{def:g10:proba:distribution}{संभावना}
$\frac16$, इतना $36$ क्रमित \omterm{def:g10:vectors:sum}{जोड़}े समान रूप से संभावित हैं; योग है
\omterm{def:g10:vectors:sum}{जोड़}ी द्वारा निर्धारित.

\emph{2.} \omterm{def:g10:vectors:sum}{जोड़} $7$: \omterm{def:g10:vectors:sum}{जोड़}े $(1,6), (2,5), (3,4), (4,3), (5,2), (6,1)$,
इसलिए $\P = \frac{6}{36} = \frac16$. \omterm{def:g10:vectors:sum}{जोड़} $12$: केवल $(6,6)$,
$\P = \frac{1}{36}$. \omterm{def:g10:functions:extrema}{अधिकतम} योग $4$: \omterm{def:g10:vectors:sum}{जोड़}ों का योग $2$, $3$ या $4$:
$(1,1)$; $(1,2), (2,1)$; $(1,3), (2,2), (3,1)$ --- छह \omterm{def:g10:vectors:sum}{जोड़}े, तो
$\P = \frac{6}{36} = \frac16$.

\emph{3.} प्रत्येक योग को प्राप्त करने वाले \omterm{def:g10:vectors:sum}{जोड़}ियों की संख्या बढ़ती जाती है $1$ (\omterm{def:g10:vectors:sum}{जोड़} $2$)
को $6$ (\omterm{def:g10:vectors:sum}{जोड़} $7$) फिर घट जाती है: सबसे संभावित योग है $7$.
\end{solution}

\begin{solution}{exo:g10:proba:7}
प्रत्येक प्रश्न का सही उत्तर दिया गया है \omterm{def:g10:proba:distribution}{संभावना} $\frac14$,
स्वतंत्र रूप से। \omterm{met:g10:proba:tree}{पेड़} दो स्तरों के साथ:

दोनों सही: $\frac14 \times \frac14 = \frac{1}{16}$.

बिल्कुल एक सही: रास्ते सही--गलत और गलत--सही:
$\frac14 \times \frac34 + \frac34 \times \frac14 = \frac{6}{16}
= \frac38$.

कोई भी सही नहीं: $\frac34 \times \frac34 = \frac{9}{16}$.

जाँच करना: $\frac{1}{16} + \frac{6}{16} + \frac{9}{16} = 1$.
\end{solution}

\begin{solution}{exo:g10:proba:8}
\emph{1.} देखा \omterm{def:g10:stats:series}{आवृत्ति}: $\frac{396}{900} = 0.44$. साथ $ n = 900$:
$\frac{1}{\sqrt{900}} = \frac{1}{30} \approx 0.033$, तो उतार-चढ़ाव
\omterm{def:g10:numbers:interval}{अंतराल} के बारे में है $\intcc{0.467}{0.533}$.

\emph{2.} अवलोकन किया गया $0.44$ के बाहर स्थित है \omterm{def:g10:numbers:interval}{अंतराल}: नमूने का $900$
के साथ जनसंख्या के मतदाता $50\%$ अनुमोदन शायद ही कभी इतनी दूर भटकता हो। द
सर्वेक्षण में दावे पर गंभीर संदेह जताया गया है $50\%$.
\end{solution}

\begin{solution}{exo:g10:proba:9}
\emph{1.} $\P(\text{win } 6) = \frac16$ (रोल ए $6$),
$\P(\text{win } 3) = \frac16$ (रोल ए $5$),
$\P(\text{win } 0) = \frac46 = \frac23$.

\emph{2.} ऊपर $600$ खेल, के बारे में उम्मीद है $100$ छक्के, $100$ फाइव्स, $400$
अन्य रोल. जीत: के बारे में $100 \times 6 + 100 \times 3 = 900$. लागत:
$600 \times 2 = 1200$. अपेक्षित शुद्ध परिणाम: $900 - 1200 = -300$, यानी\ ए
के बारे में हानि $0.5$ प्रति गेम औसतन --- गेम सार्थक नहीं है
खेल रहे हैं.
\end{solution}

\begin{solution}{pb:g10:proba:1}
\textbf{1.} $ P(\text{king or heart}) = \frac{4}{52} +
\frac{13}{52} - \frac{1}{52} = \frac{16}{52} = \frac{4}{13}$:
दिल का राजा दोनों में बैठता है घटनाएँ और इसकी गिनती नहीं की जानी चाहिए
दो बार --- इसलिए घटाव
(\cref{thm:g10:proba:addition}).

\textbf{2.} $ P(\text{no head}) = \left(\frac12\right)^3 =
\frac18$, इसलिए $ P(\text{at least one}) = \frac78$.

\textbf{3.} $ P(RR) = \frac35 \times \frac24 = \frac{3}{10}$;
$ P(\text{one of each}) = \frac35 \times \frac24 + \frac25
\times \frac34 = \frac{6}{20} + \frac{6}{20} = \frac35$.

\textbf{4.} $ P(BB) = \frac25 \times \frac14 = \frac1{10}$; और
$\frac{3}{10} + \frac{6}{10} + \frac{1}{10} = 1$: द पेड़का
पत्तियां हमेशा पूरी होती हैं $1$.

\textbf{5.} \omterm{def:g10:proba:operations}{असंगत} घटनाएँ आज्ञा का पालन करना $ P(A \cup B) = P(A) +
P(B) = 1.1 > 1$: असंभव --- कुछ \omterm{def:g10:proba:distribution}{संभावना} का आविष्कार किया गया था.

\textbf{6.} (हजारों पत्र-लेखकों सहित अधिकांश लोग,
उत्तर ``इससे कोई फर्क नहीं पड़ता, 50--50''। अपना उत्तर अपने पास रखें
प्रश्न 8.)

\textbf{7.} दरवाज़े 1 के पीछे कार (\omterm{def:g10:proba:distribution}{संभावना} $\frac13$): मेज़बान
या तो बकरी का दरवाज़ा खोलता है; स्विचिंग कार छोड़ देता है: \emph{खोना}.
दरवाजे 2 के पीछे कार ($\frac13$): मेजबान \emph{अवश्य} खुला दरवाज़ा 3;
दरवाज़ा 2 पर ज़मीन बदलना: \emph{जीतना}. दरवाज़े 3 के पीछे कार:
सममित: \emph{जीतना}. कुल:
$ P(\text{switch wins}) = \frac13 + \frac13 = \frac23$.

\textbf{8.} आपका पहला चयन ग़लत है \omterm{def:g10:proba:distribution}{संभावना}
$\frac23$ --- और ठीक तभी, मेज़बान का जबरन खुलासा
कार को शेष दरवाजे के पीछे छोड़ देता है: स्विचिंग कन्वर्ट
हर प्रारंभिक त्रुटि जीत में बदल जाती है। $\frac23$, जो भी पेट हो
कहा.

\textbf{9.} छह समान रूप से संभावित परिदृश्य: कार पीछे $1$, $2$
या $3$, मेज़बान दरवाज़ा खोलता है $2$ या $3$ (तुम्हारा कभी नहीं)। खुलासा
उनमें से दो में कार दिखाता है (car@2/खुलता 2, कार@3/खुलता 3):
त्याग दिया गया। जीवित बचे चार लोगों में --- कार@1/ओपन्स 2,
कार@1/खुलती 3, कार@2/खुलती 3, कार@3/खुलती 2 --- स्विचिंग जीत
पिछले दो में: $\frac24 = \frac12$. एक अज्ञानी मेज़बान के साथ,
50--50 \emph{है} सही: विरोधाभास पूरी तरह से रहता है
मेजबान का ज्ञान, जो परिदृश्यों को असमान रूप से फ़िल्टर करता है।

\textbf{10.} आपकी पसंद कार को अपने पास रखती है \omterm{def:g10:proba:distribution}{संभावना}
$\frac{1}{100}$; जानने वाले मेज़बान का $98$ फ़नल का पता चलता है
शेष $\frac{99}{100}$ एक बंद दरवाजे पर. स्विच,
और जीतो $99$ कई बार $100$.

\textbf{11.} तीन कार्ड (एक इक्का, दो जोकर), एक दोस्त जो
झांकता है और हमेशा उन दोनों में से एक जोकर को त्याग देता है जो आपने नहीं किया
चुनना; खेल $300$ राउंड, हमेशा स्विचिंग, और रिकॉर्ड जीत।
अपेक्षित \omterm{def:g10:stats:series}{आवृत्ति} $\frac23 \approx 0.667$, उतार-चढ़ाव के साथ
$\pm\frac{1}{\sqrt{300}} \approx 0.058$: के बीच की अपेक्षा करें
$61\,\%$ और $72\,\%$ स्विचिंग जीत --- किसी से बहुत दूर
``50--50''।

\textbf{12.} समान रूप से संभावित कॉन्फ़िगरेशन (बड़े - छोटे):
बीबी, बीजी, जीबी, जीजी। ``कम से कम एक लड़का'' बीबी, बीजी, जीबी रखता है:
$ P(\text{BB}) = \frac13$.

\textbf{13.} ``बड़ा एक लड़का है'' बीबी कहती है, बीजी:
$ P(\text{BB}) = \frac12$. विभिन्न सूचना फ़िल्टर
अलग-अलग परिदृश्य सेट --- बिल्कुल वैसे ही जैसे दो मोंटी ने किया था: क्या
आप मामलों को सीखते हैं, और \emph{कैसे} आपने सीखा कि यह बस मायने रखता है
उतना ही.

\textbf{14.} सभी एक जैसे: एचएचएच, टीटीटी: $\frac28 = \frac14$.
डी'अलेम्बर्ट की त्रुटि: उसके दो ``मामले'' समान रूप से संभावित नहीं हैं
--- ``सभी एक जैसे नहीं'' आठ में से छह परिणामों को एकत्रित करता है।
\Cref{prop:g10:proba:uniform} केवल समान \omterm{def:g10:proba:distribution}{संभावना} से अधिक गिना जाता है
परिणाम.

\textbf{15.} प्रति गेम औसत जीत: $6 \times \frac16 + 3
\times \frac16 = 1.50$ यूरो, एक के खिलाफ $2$-यूरो हिस्सेदारी: एक
का औसत नुकसान $0.50$ प्रति खेल, के बारे में $300$ यूरो खत्म $600$
खेल. उचित प्रीमियम औसत भुगतान के बराबर होता है:
$1\,\% \times 10\,000 = 100$ यूरो; वास्तविक प्रीमियम लागत जोड़ते हैं
और एक सुरक्षा मार्जिन --- बीमाकर्ता को बुरे वर्षों में जीवित रहना चाहिए,
सिर्फ औसत वाले नहीं।

\textbf{16.} \omterm{def:g10:numbers:interval}{अंतराल}: $0.5 \pm \frac{1}{\sqrt{100}} =
\intcc{0.4}{0.6}$. $55$ सिर ($0.55$): अंदर, अचूक.
$70$ सिर ($0.70$): बहुत बाहर --- सिक्के पर संदेह करें।

\textbf{17.} $\frac{1}{\sqrt{1000}} \approx 0.032$: एक निष्पक्ष
$50\,\%$ उम्मीदवार आम तौर पर मतदान करेगा
$\intcc{0.468}{0.532}$ --- और $0.52$ अंदर आराम से बैठता है.
सर्वेक्षण \emph{का सुझाव} एक बढ़त लेकिन कुछ भी साबित नहीं होता: मार्जिन
इसे निगल जाता है.

\textbf{18.} पहला \omterm{def:g10:proba:sample}{नमूना}: \omterm{def:g10:numbers:interval}{अंतराल} $0.04 \pm \frac{1}{20} =
\intcc{0}{0.09}$; अवलोकन किया गया $0.07$ अंदर है: कोई फैसला नहीं.
दूसरा: $0.04 \pm \frac{1}{50} = \intcc{0.02}{0.06}$; द
अवलोकन किया $0.07$ बाहर गिर जाता है: दावा अब गंभीर है
संदेह. वही \omterm{def:g10:stats:series}{आवृत्ति}, तेज उपकरण: परिशुद्धता की तरह बढ़ता है
$\sqrt n $.

\textbf{19.} उतार-चढ़ाव जैसे सिकुड़ जाता है $\frac{1}{\sqrt n}$:
चौगुना \omterm{def:g10:proba:sample}{नमूना}, शोर आधा कर दो। पूर्वाग्रह --- पूछ रहा हूँ
गलत भीड़--- कोई भी बच जाता है $ n $: दो मिलियन बुरी तरह से चुने गए
मतपत्र गलत रहे (मिडिल स्कूल वॉल्यूम की कैसे-डेटा-झूठ सप्ताहांत समस्या)।

\textbf{20.} (i) जाँच के परिणाम पहले की तरह ही संभावित हैं
गिनती --- डी'अलेम्बर्ट भूल गया। (ii) \omterm{met:g10:proba:tree}{पेड़} साथ में गुणा करें
शाखाएँ; ``कम से कम एक'' समर्पण करता है \omterm{def:g10:proba:operations}{पूरक}.
(iii) सूचना फ़िल्टर परिदृश्य, और इसके \emph{स्रोत}
फ़िल्टर बदलता है --- मोंटी को गिरे हुए मोंटी के विरुद्ध जानना,
किसी लड़के के विरुद्ध बड़ा लड़का। (iv) \omterm{def:g10:stats:series}{आवृत्ति} दृष्टिकोण
गति से संभावनाएँ $\frac{1}{\sqrt n}$, तो एक रोगी
अनुकरण किसी भी तर्क का समाधान करता है --- जिसमें दस के साथ एक भी शामिल है
हजारों नाराज पत्र-लेखक.
\end{solution}
