\section*{अध्याय \ref{ch:g10:numbers} --- संख्याएँ और संख्याओं का समूह}

\begin{solution}{exo:g10:numbers:1}
$\frac{15}{3} = 5 \in \N $. \quad $-7 \in \Z $. \quad
$\frac{22}{7} \in \Q $ (यह एक नहीं है पूर्णांक: $22 = 7 \times 3 + 1$).
\quad $\sqrt 9 = 3 \in \N $. \quad $\sqrt{10} \in \R $ (\omterm{ex:g10:numbers:classify}{तर्कहीन}, तब से
$10$ तर्कसंगत का वर्ग नहीं है --- यहाँ, की भावना से स्वीकार किया गया है

\quad $0 \in \N $.
\end{solution}

\begin{solution}{exo:g10:numbers:2}
(ए) $\intco{-2}{5}$: भरा हुआ बिंदु $-2$, खोखला बिंदु पर $5$.
(बी) $\intoo{3}{+\infty}$: खोखला बिंदु पर $3$, दाईं ओर छायांकन।
(सी) $\intoc{-\infty}{-1}$: बाईं ओर से भरे हुए बिंदु तक छायांकन
$-1$.
(डी) ``से दूरी $ x $ को $2$ अधिक से अधिक $3$'' मतलब $\abs{x - 2} \leq 3$,
यानी\ $ x \in \intcc{-1}{5}$: पर भरे हुए बिंदु $-1$ और $5$.
\end{solution}

\begin{solution}{exo:g10:numbers:3}
(ए) दो \omterm{def:g10:numbers:interval}{अंतराल} के बीच ओवरलैप करें $0$ और $2$:
$ I \cap J = \intcc{0}{2}$ ($0 \in J $ और $2 \in I $, और दोनों का संबंध है
अन्य सेट भी), और $ I \cup J = \intco{-3}{4}$.

(बी) $ I \cap J $ का सेट है $ x $ साथ $-2 \leq x $ और $ x < 1$:
$\intco{-2}{1}$. \omterm{def:g10:numbers:interunion}{मिलन} सब कुछ कवर करता है: $ I \cup J = \R $.
\end{solution}

\begin{solution}{exo:g10:numbers:4}
$\abs{-6} = 6$; \quad $\abs{4 - 9} = \abs{-5} = 5$; \quad
$\abs{-3 - 5} = \abs{-8} = 8$; \quad
$\sqrt 2 > 1$ इसलिए $\abs{\sqrt2 - 1} = \sqrt2 - 1$; \quad
$1 - \sqrt2 < 0$ इसलिए $\abs{1 - \sqrt2} = \sqrt2 - 1$ साथ ही (एक संख्या और
इसके विपरीत में समान है \omterm{def:g10:numbers:abs}{निरपेक्ष मूल्य}).
\end{solution}

\begin{solution}{exo:g10:numbers:5}
$\abs{x} = 5$: दूरी $5$ से $0$, इसलिए $ x = 5$ या $ x = -5$.

$\abs{x - 1} = 3$: दूरी $3$ से $1$, इसलिए $ x = 4$ या $ x = -2$.

$\abs{x - 4} \leq 1$: अधिकतम दूरी $1$ से $4$, इसलिए
$ x \in \intcc{3}{5}$.

$\abs{x + 2} < 3$: दूरी बिल्कुल कम $3$ से $-2$, इसलिए
$ x \in \intoo{-5}{1}$.
\end{solution}

\begin{solution}{exo:g10:numbers:6}
प्रत्येक \omterm{def:g10:numbers:interval}{अंतराल} इसका वर्णन इसके केंद्र द्वारा किया गया है $ a $ (सीमा का मध्यबिंदु) और
त्रिज्या $ r $ (आधी लंबाई).

$\intcc{2}{8}$: $ a = 5$, $ r = 3$, इसलिए $\abs{x - 5} \leq 3$.

$\intoo{-5}{1}$: $ a = -2$, $ r = 3$, खुली सीमाएँ, इसलिए $\abs{x + 2} < 3$.

$\intcc{-7}{-3}$: $ a = -5$, $ r = 2$, इसलिए $\abs{x + 5} \leq 2$.
\end{solution}

\begin{solution}{exo:g10:numbers:7}
होने देना $ x = 0.272727\dots $ तब $100x = 27.2727\dots $, और घटाना:
\[
100x - x = 27.2727\dots - 0.2727\dots = 27,
\]
इसलिए $99x = 27$ और $ x = \frac{27}{99} = \frac{3}{11}$, का एक भागफल
\omterm{def:g10:numbers:sets}{पूर्णांकों}: $ x $ तर्कसंगत है.
\end{solution}

\begin{solution}{exo:g10:numbers:8}
\emph{1. सच:} जोड़ना \omterm{def:g10:numbers:sets}{पूर्णांकों} (धनात्मक या ऋणात्मक पूर्णांक)
हमेशा एक पैदा करता है पूर्णांक.

\emph{2. मिथ्या:} $\frac{1}{2}$ का एक भागफल है \omterm{def:g10:numbers:sets}{पूर्णांकों} $1$ और $2$
और एक नहीं है पूर्णांक.

\emph{3. सच:} $\frac pq + \frac{p'}{q'} = \frac{pq' + p'q}{qq'}$ है
फिर से एक भागफल \omterm{def:g10:numbers:sets}{पूर्णांकों} (शून्येतर हर के साथ)।

\emph{4. सच:} कल्पना करना $ r $ तर्कसंगत है, $ t $ \omterm{ex:g10:numbers:classify}{तर्कहीन}, और $ r + t = s $
तर्कसंगत थे. तब $ t = s - r $ दो तर्कों का अंतर होगा,
इसलिए तर्कसंगत (3 से, के साथ लागू किया गया $-r $) --- विरोधाभास। इसलिए $ r + t $ है
\omterm{ex:g10:numbers:classify}{तर्कहीन}.
\end{solution}

\begin{solution}{exo:g10:numbers:9}
गुणा $1.414 \leq \sqrt2 \leq 1.415$ द्वारा $2 > 0$:
$2.828 \leq 2\sqrt2 \leq 2.830$.

जोड़ा जा रहा है $3$: $4.414 \leq \sqrt2 + 3 \leq 4.415$.

से गुणा करना $-1 < 0$ असमानताओं को उलट देता है:
$-1.415 \leq -\sqrt2 \leq -1.414$.
\end{solution}

\begin{solution}{exo:g10:numbers:10}
सबसे पहले सहायक तथ्य. विभाजित करना $ p $ द्वारा $3$:शेष है $0$, $1$ या
$2$, यानी\ $ p = 3k $, $ p = 3k+1$ या $ p = 3k+2$. वर्ग बनाना:
\[
(3k)^2 = 3(3k^2), \quad
(3k+1)^2 = 3(3k^2 + 2k) + 1, \quad
(3k+2)^2 = 3(3k^2 + 4k + 1) + 1 .
\]
केवल प्रथम का गुणज है $3$: अगर $ p^2$ का गुणज है $3$, तब
$ p $ भी है।

अब मान लीजिए $\sqrt3 = \frac pq $ पूरी तरह से सरलीकृत. चुकता करने से मिलता है
$ p^2 = 3q^2$, इसलिए $ p^2$ का गुणज है $3$, इसलिए $ p = 3k $. तब
$9k^2 = 3q^2$, इसलिए $ q^2 = 3k^2$ और $ q $ का गुणज भी है $3$ --- लेकिन
फिर अंश $\frac pq $ पूरी तरह से सरलीकृत नहीं किया गया था: विरोधाभास. इसलिए
$\sqrt3$ है \omterm{ex:g10:numbers:classify}{तर्कहीन}.
\end{solution}

\begin{solution}{pb:g10:numbers:1}
\textbf{1.} $-7 \in \Z $; \ $\frac{13}{4} \in \Q $; \
$\sqrt{16} = 4 \in \N $; \ $0.121212\ldots = \frac{12}{99} =
\frac{4}{33} \in \Q $ (मिडिल स्कूल वॉल्यूम की दोहराव-दशमलव सप्ताहांत समस्या); \
$\sqrt8 = 2\sqrt2$ है \omterm{ex:g10:numbers:classify}{तर्कहीन}: सबसे छोटा सेट $\R $; \ $\pi $:
$\R $.

\textbf{2.} $\frac pq + \frac rs = \frac{ps + qr}{qs}$ और
$\frac pq \times \frac rs = \frac{pr}{qs}$: के भागफल
\omterm{def:g10:numbers:sets}{पूर्णांकों} शून्येतर हर के साथ --- तर्कसंगत।

\textbf{3.} (ए) यदि $ r $ तर्कसंगत है, $ x $ \omterm{ex:g10:numbers:classify}{तर्कहीन} और
$ r + x = q $ तब तर्कसंगत थे $ x = q - r $ एक होगा
परिमेय का अंतर, अत: परिमेय (प्रश्न 2):
विरोधाभास. (बी) यदि $ r \neq 0$ और $ r x = q $ तर्कसंगत, तो
$ x = \frac qr $ तर्कसंगत है: विरोधाभास.

\textbf{4.} $\sqrt2$ और $-\sqrt2$ हैं \omterm{ex:g10:numbers:classify}{तर्कहीन} योग के साथ $0$;
$\sqrt2$ और $\sqrt2$ उत्पाद है $2$. से तर्कसंगत परिणाम
\omterm{ex:g10:numbers:classify}{तर्कहीन} सामग्री: अपरिमेय कोई स्थिर साम्राज्य नहीं।

\textbf{5.} $\sqrt8 = 2\sqrt2$, इसलिए
$\sqrt2 + \sqrt8 = 3\sqrt2$: \omterm{ex:g10:numbers:classify}{तर्कहीन} (प्रश्न 3बी, गुणक
$3$). और $\sqrt2 \times \sqrt8 = \sqrt{16} = 4 \in \N $: द
दो का उत्पाद अपरिमेय के सबसे छोटे राज्य में भूमि
सब.

\textbf{6.} $3.475$, तब $3.471$ (या $3.4701$, $3.47001$,
\dots): दशमलव अंकों को जोड़ने से ताज़ा परिमेय उत्पन्न होते हैं
दोनों के बीच हमेशा के लिए.

\textbf{7.} के गुणज $10^{-n}$ लाइन के साथ मार्च करें
के चरण $10^{-n} < g $. पहला गुणक सख्ती से बड़ा है
से $ a $ --- यह अस्तित्व में है, क्योंकि गुणज अंततः अधिक हो जाते हैं
$ a $ --- अधिक से अधिक एक कदम आगे है $ a $, इसलिए पहले
$ a + g = b $: सख्ती से बीच में $ a $ और $ b $. एक ग्रिड बिंदु एक है
दशमलव, इसलिए एक तर्कसंगत: प्रत्येक \omterm{def:g10:numbers:interval}{अंतराल} सकारात्मक लंबाई का
एक शामिल है.

\textbf{8.} बीच में $ a $ और तर्कसंगत $ r_1$ में पाया गया
प्रश्न 7 (प्रश्न 7 फिर से) एक तर्कसंगत है $ r_2$; बीच में
$ a $ और $ r_2$ एक तर्कसंगत $ r_3$; और इसी तरह: अनंत रूप से अनेक, सभी
विशिष्ट, सभी मूल में \omterm{def:g10:numbers:interval}{अंतराल}.

\textbf{9.} $ r + \frac{\sqrt2}{10^k}$ एक तर्कसंगत का योग है
और एक \omterm{ex:g10:numbers:classify}{तर्कहीन} ($\frac{\sqrt2}{10^k}$ है \omterm{ex:g10:numbers:classify}{तर्कहीन} द्वारा
प्रश्न 3बी), इसलिए \omterm{ex:g10:numbers:classify}{तर्कहीन}. तब से
$\frac{\sqrt2}{10^k} < \frac{2}{10^k}$ किसी भी सीमा से नीचे सिकुड़ जाता है,
के लिए $ k $ बहुत पर्याप्त $ r + \frac{\sqrt2}{10^k}$ अभी भी झूठ बोलता है
पहले $ b $: एक \omterm{ex:g10:numbers:classify}{तर्कहीन} के अंदर \omterm{def:g10:numbers:interval}{अंतराल} --- और भिन्न-भिन्न
$ k $ अनन्त रूप से अनेक देता है।

\textbf{10.} कोई लघुत्तम सकारात्मक यथार्थ नहीं: यदि $ s > 0$ क्या यह था,
\omterm{def:g10:numbers:interval}{अंतराल} $(0, s)$ इसमें अभी भी एक तर्कसंगत (प्रश्न 7) होगा,
सकारात्मक और उससे छोटा $ s $. ठीक बाद में कोई नंबर नहीं $3$: कोई भी
उम्मीदवार $ c > 3$ छोड़ देता है \omterm{def:g10:numbers:interval}{अंतराल} $(3, c)$, जो नहीं है
खाली --- मिडिल स्कूल वॉल्यूम के गेम की नो-नेक्स्ट-नंबर सप्ताहांत समस्या, अब इसके बारे में एक प्रमेय
$\R $.

\textbf{11.} $\abs{x - 5} = 2$: $ x = 3$ या $ x = 7$.
$\abs{x - 5} < 2$: $ x \in \intoo{3}{7}$.
$\abs{x + 1} \geq 3$: की दूरी $-1$ कम से कम $3$:
$ x \in \intoc{-\infty}{-4} \cup \intco{2}{+\infty}$.

\textbf{12.} $\abs{d - 80} \leq 0.05$, अर्थात\
$ d \in \intcc{79.95}{80.05}$. पिस्टन पर $79.97$ गुजरता है;
एक पर $80.06$ एक मिलीमीटर के सौवें हिस्से से विफल रहता है।

\textbf{13.} $(3, -5)$: $\abs{-2} = 2 \leq 8$. $(-2, -7)$:
$\abs{-9} = 9 = 2 + 7$: समानता. वही संकेत: $\abs{a + b}$ है
दूरियों का योग, के बराबर $\abs a + \abs b $. विपरीत
संकेत: चलना दोगुना हो जाता है, $\abs{a + b}$ है
\emph{अंतर} दूरियों की, उनसे बिल्कुल कम
योग (जब तक कि उनमें से एक न हो $0$). समानता बिल्कुल कब $ a $ और
$ b $ एक ही चिन्ह है या एक शून्य है।

\textbf{14.} समाधान वे बिंदु हैं जो समान दूरी पर हैं $2$
और $-4$: मध्यबिंदु, $ x = -1$. यह लम्बवत है
मिडिल स्कूल वॉल्यूम की दो-दर्पण सप्ताहांत समस्या का समद्विभाजक एक पर सिमट गया
आयाम: एक बिंदु.

\textbf{15.} $\sqrt{x^2} = \abs{x}$, नहीं $ x $: के लिए $ x = -3$,
$\sqrt{9} = 3 = \abs{-3} \neq -3$. तब $ x^2 < 9$ पढ़ता
$\sqrt{x^2} < 3$, यानी\ $\abs x < 3$:
$ x \in \intoo{-3}{3}$.

\textbf{16.} माप केवल यह दावा करता है कि लंबाई निहित है
में \omterm{def:g10:numbers:interval}{अंतराल} $\intcc{1.41420}{1.41422}$ --- और सवालों से
7 और 9 \emph{वह \omterm{def:g10:numbers:interval}{अंतराल} इसमें अपरिमित रूप से अनेक तर्कसंगतताएं शामिल हैं
और अनंत रूप से अनेक अपरिमेय}. प्रत्येक के लिए यही बात लागू होती है
भविष्य की सहनशीलता, चाहे कितनी ही छोटी क्यों न हो। कोई भी माप कभी नहीं हो सकता
दोनों परिवारों को अलग करें; सटीक लंबाई के बारे में केवल एक प्रमाण
--- इकाई वर्ग के विकर्ण की तरह
(मिडिल स्कूल वॉल्यूम की अतार्किकता सप्ताहांत समस्या) --- कर सकते हैं।

\textbf{17.} वे पास आते हैं $\sqrt2$ नीचे त्रुटियों के साथ
$10^{-1}, 10^{-2}, 10^{-3}, \dots $:तर्कसंगत लोग इसके विरुद्ध दबाव डालते हैं
$\sqrt2$ हर पैमाने पर. सामान्य तथ्य: प्रत्येक \omterm{def:g10:numbers:sets}{वास्तविक संख्या} है
तर्कसंगत (इसका दशमलव) द्वारा वांछित के रूप में निकटता से संपर्क किया गया
काट-छाँट) --- घनत्व एक बार फिर, लक्ष्य से देखा गया
ओर.

\textbf{18.} $ a + b \in \intcc{2.4 + 1.1}{2.6 + 1.3} =
\intcc{3.5}{3.9}$: अनिश्चितता $\pm 0.2$ (त्रुटियाँ जोड़ी गईं)।
$ a \times b \in \intcc{2.4 \times 1.1}{2.6 \times 1.3} =
\intcc{2.64}{3.38}$: के बारे में एक अनिश्चितता $\pm 0.37$ आस-पास
$3$ --- उत्पादों में सापेक्ष त्रुटियाँ जुड़ती हैं, इसलिए परिशुद्धता कम हो जाती है
तेज़.

\textbf{19.} सच्चे मूल्य लिखें $ a = 2.5 + e_1$,
$ b = 1.2 + e_2$ साथ $\abs{e_1}, \abs{e_2} \leq 0.1$. फिर
योग की त्रुटि है $\abs{e_1 + e_2} \leq \abs{e_1} +
\abs{e_2} \leq 0.2$: इंजीनियर का नियम \emph{है} त्रिकोण
प्रश्न 13 की असमानता.

\textbf{20.} वास्तविक रेखा में कोई \omterm{def:g10:numbers:interval}{अंतराल} नहीं है और कोई पड़ोसी नहीं है: बाद में
कोई भी संख्या, कोई ``अगला'' नहीं (प्रश्न 10)। इसके अंक बंट गए
तर्कसंगत में --- बिल्कुल रोकना या दोहराना
दशमलव (मिडिल स्कूल वॉल्यूम की दोहराव-दशमलव सप्ताहांत समस्या) --- और अपरिमेय,
और दोनों परिवार आपस में इतनी मजबूती से जुड़े हुए हैं कि हर कोई
\omterm{def:g10:numbers:interval}{अंतराल} प्रत्येक के अपरिमित रूप से अनेक धारण करता है (प्रश्न 8 और 9),
यही कारण है कि कोई भी माप, केवल प्रमाण, उन्हें अलग नहीं बता सकता
(प्रश्न 16). और अंतिम आश्चर्य, एक वादे के रूप में छोड़ा गया:
अपरिमेय परिमेय संख्या से अधिक संख्या --- एक-एक करके गिनने से नहीं
एक, लेकिन अनन्तताओं की तुलना के सटीक अर्थ में, ए
विश्वविद्यालय के संस्करणों में निर्मित सिद्धांत।
\end{solution}
