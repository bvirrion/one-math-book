\begin{solution}{exo:g10:stats:1}
\omterm{def:g10:vectors:sum}{जोड़}: $0+1+1+2+2+2+3+3+4+4+5+6+7+8+12 = 60$, इतना \omterm{def:g10:stats:mean}{अर्थ} है
$\frac{60}{15} = 4$. श्रृंखला पहले से ही क्रमबद्ध है और $ N = 15$ अजीब है:
द \omterm{def:g10:stats:median}{MEDIAN} है $8$ वें मान, $3$. \omterm{def:g10:stats:quartiles}{श्रेणी}: $12 - 0 = 12$.
\end{solution}

\begin{solution}{exo:g10:stats:2}
आवृत्तियाँ: $\frac{7}{50} = 0.14$, $\frac{9}{50} = 0.18$,
$\frac{8}{50} = 0.16$, $\frac{10}{50} = 0.20$, $\frac{6}{50} = 0.12$,
$\frac{10}{50} = 0.20$ (\omterm{def:g10:vectors:sum}{जोड़} $1$). \omterm{def:g10:stats:mean}{अर्थ} नतीजा:
\[
\bar x = \frac{7 \times 1 + 9 \times 2 + 8 \times 3 + 10 \times 4
+ 6 \times 5 + 10 \times 6}{50}
= \frac{7 + 18 + 24 + 40 + 30 + 60}{50}
= \frac{179}{50} = 3.58 .
\]
\end{solution}

\begin{solution}{exo:g10:stats:5}
वर्तमान कुल: $5 \times 12 = 60$. ए \omterm{def:g10:stats:mean}{अर्थ} का $13$ ऊपर $6$ परीक्षणों की आवश्यकता है
कुल $78$, इतना $6$ वां स्कोर होना चाहिए $78 - 60 = 18$. ए \omterm{def:g10:stats:mean}{अर्थ} का
$14$ आवश्यकता होगी $84 - 60 = 24 > 20$: पहुंच योग्य नहीं है।
\end{solution}

\begin{solution}{exo:g10:stats:6}
\omterm{def:g10:stats:mean}{अर्थ} की तुलना में बड़ा है $20$: उदाहरण के लिए
$1,\ 2,\ 3,\ 10,\ 40,\ 50,\ 60$ --- \omterm{def:g10:stats:median}{MEDIAN} $10$ (द $4$ वें क्रमबद्ध मान),
\omterm{def:g10:stats:mean}{अर्थ} $\frac{166}{7} \approx 23.7$.

\omterm{def:g10:stats:mean}{अर्थ} तुलना में छोटा $6$: उदाहरण के लिए
$0,\ 0,\ 0,\ 10,\ 10,\ 10,\ 10$ --- \omterm{def:g10:stats:median}{MEDIAN} $10$, \omterm{def:g10:stats:mean}{अर्थ}
$\frac{40}{7} \approx 5.7$.

 \omterm{def:g10:stats:mean}{अर्थ} नीचे नहीं जा सकते $\frac{40}{7}$: के लिए \omterm{def:g10:stats:median}{MEDIAN} (द $4$ वें क्रमबद्ध
मूल्य) होना $10$, पदों में मान $4$ को $7$ सब होना चाहिए
$\geq 10$, तो कुल कम से कम है $40$ और यह \omterm{def:g10:stats:mean}{अर्थ} कम से कम
$\frac{40}{7}$ --- दूसरा उदाहरण चरम है.

ये उदाहरण दर्शाते हैं कि \omterm{def:g10:stats:mean}{अर्थ} और यह \omterm{def:g10:stats:median}{MEDIAN} काफी हद तक स्वतंत्र हैं:
एक को जानना दूसरे के बारे में बहुत कम कहता है, यही कारण है कि एक अच्छा सारांश है
दोनों की रिपोर्ट.
\end{solution}

\begin{solution}{exo:g10:stats:7}
\emph{1.} \omterm{def:g10:stats:mean}{अर्थ}:
$\frac{9 \times 1800 + 12000}{10} = \frac{16200 + 12000}{10} = 2820$.
मंझला: क्रमबद्ध, द $10$ वेतन नौ मान हैं $1800$ तब $12000$;
 \omterm{def:g10:stats:median}{MEDIAN} है \omterm{prop:g10:coordgeom:midpoint}{मध्य} की $5$ वें और $6$ वें मान, दोनों $1800$:
\omterm{def:g10:stats:median}{MEDIAN} $1800$.

\emph{2.}  \omterm{def:g10:stats:median}{MEDIAN} ($1800$) विशिष्ट वेतन का वर्णन करता है: यह क्या है
$9$ कर्मचारी बाहर $10$ वास्तव में कमाओ. \omterm{def:g10:stats:mean}{अर्थ} ($2820$) ऊपर खींच लिया जाता है
एकल उच्च वेतन से और किसी की वेतन पर्ची से मेल नहीं खाता।
\end{solution}

\begin{solution}{exo:g10:stats:9}
\emph{1.} नई \omterm{def:g10:stats:mean}{अर्थ} है
\[
\bar y = \frac{y_1 + \dots + y_N}{N}
= \frac{(a x_1 + b) + \dots + (a x_N + b)}{N}
= \frac{a(x_1 + \dots + x_N) + Nb}{N}
= a \bar x + b .
\]

\emph{2.} साथ $ a = 1.8$ और $ b = 32$: \omterm{def:g10:stats:mean}{अर्थ}
$1.8 \times 20 + 32 = 68$ डिग्री फ़ारेनहाइट.
\end{solution}

\begin{solution}{pb:g10:stats:1}
\textbf{1.} \omterm{def:g10:stats:mean}{अर्थ}: $\frac{2\,550}{7} \approx 364$ हजार
यूरो; \omterm{def:g10:stats:median}{MEDIAN}: सात आदेशित कीमतों में से चौथा, $230$.
 \omterm{def:g10:stats:median}{MEDIAN} सड़क का वर्णन करता है --- सात में से छह घर
लागत कहीं नहीं है $364$.

\textbf{2.} हवेली के बिना: \omterm{def:g10:stats:mean}{अर्थ} $\frac{1\,350}{6} = 225$,
\omterm{def:g10:stats:median}{MEDIAN} $\frac{220 + 230}{2} = 225$. \omterm{def:g10:stats:mean}{अर्थ} के बारे में गिर गया
$139$; \omterm{def:g10:stats:median}{MEDIAN} द्वारा $5$. एक चरम मूल्य खींच सकता है \omterm{def:g10:stats:mean}{अर्थ}
कहीं भी; \omterm{def:g10:stats:median}{MEDIAN} बमुश्किल नोटिस --- यह है \emph{मज़बूत}.

\textbf{3.} \omterm{def:g10:stats:mean}{अर्थ} $= \frac{0 \times 20 + 1 \times 25 + 2 \times
30 + 3 \times 15 + 4 \times 10}{100} = \frac{170}{100} = 1.7$.
किसी परिवार के पास नहीं है $1.7$ बच्चे, लेकिन $1.7 \times 100 = 170$
\emph{है} बच्चों की सटीक कुल संख्या: \omterm{def:g10:stats:mean}{अर्थ} एन्कोड करता है
कुल, समान रूप से पुनर्वितरित --- यह असली प्रतिभा है।

\textbf{4.} $ N = 15$: \omterm{def:g10:stats:median}{MEDIAN} $=$ आठवाँ मान $= 11$;
$ Q_1 =$ चौथा मान $= 8$; $ Q_3 =$ 12वाँ मान $= 14$.

\textbf{5.} \omterm{def:g10:stats:quartiles}{श्रेणी} $19 - 4 = 15$: पूर्ण प्रसार, चरम सीमाएँ
शामिल. \omterm{def:g10:stats:quartiles}{अंतरचतुर्थक सीमा} $14 - 8 = 6$: की चौड़ाई
\omterm{prop:g10:coordgeom:midpoint}{मध्य} भाग --- चरम शोर के बिना फैल गया।

\textbf{6.} अवस्था बदलना $+b $ चाल \emph{प्रत्येक} मूल्य, तो
मूल्यों का अंतर --- \omterm{def:g10:stats:quartiles}{श्रेणी}, IQR --- अपरिवर्तित हैं:
$(x_i + b) - (x_j + b) = x_i - x_j $. द्वारा एक स्केलिंग $ a $
सभी अंतरों को गुणा करता है $ a $: \omterm{def:g10:stats:quartiles}{श्रेणी} और IQR द्वारा मापे जाते हैं
$\abs a $.

\textbf{7.} योगात्मक वक्र: $6 \to 9$ और $18 \to 21$ --- द
कमजोर छात्र लाभ $50\,\%$, मजबूत एक $17\,\%$, और $21$
को फोड़ देता है $20$-पॉइंट स्केल केवल हल्के ढंग से। गुणक:
$6 \to 8$ और $18 \to 24$ --- अनुपात संरक्षित, लेकिन $24$ है
एक पर असंभव $20$-स्केल: छत की समस्या। योजक
वक्र नीचे की ओर चापलूसी करते हैं; गुणक वाले विस्फोट करते हैं
शीर्ष.

\textbf{8.} कुल मार्क: $20 \times 12 + 30 \times 10 = 540$,
तो विलय हो गया \omterm{def:g10:stats:mean}{अर्थ} है $\frac{540}{50} = 10.8$: कक्षा बी के तीस
कक्षा ए के छात्रों की संख्या बीस से अधिक है --- इसका मतलब है विलय
\emph{भारित} औसत, साधारण औसत से कभी नहीं।

\textbf{9.} मंझला का $ A $: $2$; का $ B $: $5$. मर्ज की गई सूची
$\{0, 1, 2, 3, 10\}$: \omterm{def:g10:stats:median}{MEDIAN} $2$. का आकार-भारित औसत
द माध्यिकाओं होगा $\frac{3 \times 2 + 2 \times 5}{5} = 3.2
\neq 2$: माध्यिकाओं कोई कुल नहीं है, इसलिए कोई विलय सूत्र मौजूद नहीं है
--- पूरी सूची को दोबारा क्रमबद्ध करना होगा।

\textbf{10.} \omterm{def:g10:stats:mean}{अर्थ}: $\frac{31.3}{5} = 6.26$. मंझला: $5.4$.
छंटनी \omterm{def:g10:stats:mean}{अर्थ}: $\frac{5.3 + 5.4 + 5.5}{3} = 5.4$. एक
उत्साही (या भ्रष्ट) न्यायाधीश ने यह कदम उठाया \omterm{def:g10:stats:mean}{अर्थ} लगभग एक द्वारा
बिंदु; \omterm{def:g10:stats:median}{MEDIAN} और छंटनी की गई \omterm{def:g10:stats:mean}{अर्थ} कंधे उचकाए --- इसलिए
जज किए गए खेलों के नियमों में कटौती।

\textbf{11.} फ़ैक्टरी ए: \omterm{prop:g10:coordgeom:midpoint}{मध्य} आधा भीतर $0.2$ मिमी; फ़ैक्टरी बी:
भीतर $4$ मिमी. वही \omterm{def:g10:stats:mean}{अर्थ}, विपरीत विश्वसनीयता: ए से खरीदें
\omterm{def:g10:stats:quartiles}{अंतरचतुर्थक सीमा} निर्णय लिया --- अकेले केंद्रों की तुलना का मतलब है,
कभी एकरूपता नहीं.

\textbf{12.} मंझला दूर से \omterm{def:g10:stats:mean}{अर्थ} धोखा देता है ए \emph{लंबे समय तक सही
पूंछ}: अधिकांश का इंतजार छोटा होता है, कुछ का विनाशकारी। उदाहरण:
$\{5, 5, 10, 10, 95\}$ --- \omterm{def:g10:stats:median}{MEDIAN} $10$, \omterm{def:g10:stats:mean}{अर्थ}
$\frac{125}{5} = 25$.

\textbf{13.} इसका मतलब है $75\,\%$ उस उम्र के बच्चे हैं
छोटा (और $25\,\%$ लंबा): एक रैंक, आकार नहीं। ``प्रत्येक
ऊपर का बच्चा \omterm{def:g10:stats:mean}{अर्थ}'' असंभव है --- यदि सभी मान पार हो गए हैं
द \omterm{def:g10:stats:mean}{अर्थ}, उनका औसत इससे अधिक होगा \omterm{def:g10:stats:mean}{अर्थ}, जो स्वयं है:
विरोधाभास. लेकिन \emph{लगभग} सभी हो सकते हैं: एक बहुत छोटा
बच्चा पकड़ सकता है \omterm{def:g10:stats:mean}{अर्थ} बाकी सभी से नीचे. के ऊपर
\emph{\omterm{def:g10:stats:median}{MEDIAN}}: परिभाषा के अनुसार, कभी भी आधे से अधिक नहीं। रैंक
चापलूसी नहीं की जा सकती; औसत कर सकते हैं.
