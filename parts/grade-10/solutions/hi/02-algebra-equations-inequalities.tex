\section*{अध्याय \ref{ch:g10:algebra} --- बीजगणित: समीकरण और असमानताएँ}

\begin{solution}{exo:g10:algebra:1}
$3(2x-5) - 2(x-7) = 6x - 15 - 2x + 14 = 4x - 1$.

$(x+4)(2x-1) = 2x^2 - x + 8x - 4 = 2x^2 + 7x - 4$.

$(3x-2)^2 = 9x^2 - 12x + 4$.

$(5-x)(5+x) = 25 - x^2$.
\end{solution}

\begin{solution}{exo:g10:algebra:2}
$ x^2 - 49 = (x+7)(x-7)$ (वर्गों का अंतर).

$ x^2 + 10x + 25 = (x+5)^2$.

$7x^2 - 14x = 7x(x - 2)$ (सामान्य कारक)।

$16x^2 - 8x + 1 = (4x - 1)^2$.
\end{solution}

\begin{solution}{exo:g10:algebra:3}
$4x + 9 = x - 3$: $3x = -12$, इसलिए $ x = -4$.

$\frac{2x-1}{3} = 5$: $2x - 1 = 15$, इसलिए $ x = 8$.

$2(x-3) = 2x + 1$ तक फैलता है $2x - 6 = 2x + 1$, यानी\ $-6 = 1$: के लिए झूठा
प्रत्येक $ x $. कोई समाधान नहीं --- दोनों पक्ष समानांतर रेखाओं को परिभाषित करते हैं जो कभी नहीं
\omterm{def:g10:numbers:interunion}{मिलन}ा.
\end{solution}

\begin{solution}{exo:g10:algebra:4}
$(x+5)(2x-8) = 0$: $ x = -5$ या $ x = 4$.

$ x^2 - 16 = (x-4)(x+4) = 0$: $ x = 4$ या $ x = -4$.

$ x^2 + 6x + 9 = (x+3)^2 = 0$: एकल समाधान $ x = -3$.
\end{solution}

\begin{solution}{exo:g10:algebra:5}
$3x - 5 \leq 7$: $3x \leq 12$, इसलिए $ x \leq 4$: समाधान सेट
$\intoc{-\infty}{4}$.

$-4x + 1 > 9$: $-4x > 8$, से विभाजित करना $-4$ असमानता को उलट देता है:
$ x < -2$: समाधान सेट $\intoo{-\infty}{-2}$.

$2(x-1) \geq 5x + 4$: $2x - 2 \geq 5x + 4$, इसलिए $-3x \geq 6$, इसलिए
$ x \leq -2$: समाधान सेट $\intoc{-\infty}{-2}$.
\end{solution}

\begin{solution}{exo:g10:algebra:6}
सामान्य कारक है $(x-1)$:
\[
E = (x-1)\bigl[(x+3) - (2x - 5)\bigr] = (x-1)(x + 3 - 2x + 5)
= (x-1)(8 - x).
\]
तब $ E = 0$ कब $ x = 1$ या $ x = 8$ (शून्य-उत्पाद नियम).
\end{solution}

\begin{solution}{exo:g10:algebra:7}
$ x^2 = 5x $: $ x^2 - 5x = x(x - 5) = 0$, इसलिए $ x = 0$ या $ x = 5$ (कभी नहीं
द्वारा विभाजित करें $ x $: समाधान $0$ खो जाएगा)।

$(2x+1)^2 = (x-4)^2$: बाईं ओर लाएँ और अंतर का गुणनखंड करें
वर्ग:
\[
(2x+1)^2 - (x-4)^2
= \bigl[(2x+1) + (x-4)\bigr]\bigl[(2x+1) - (x-4)\bigr]
= (3x - 3)(x + 5).
\]
शून्य-उत्पाद नियम: $ x = 1$ या $ x = -5$.
\end{solution}

\begin{solution}{exo:g10:algebra:8}
$(x+2)(4-x) > 0$: कारक $ x + 2$ पर गायब हो जाता है $-2$ (बाद में सकारात्मक),
$4 - x $ पर गायब हो जाता है $4$ (पहले सकारात्मक)। उत्पाद सकारात्मक है
बिल्कुल जब दोनों कारक सकारात्मक हों, यानी\ पर $\intoo{-2}{4}$.

$\frac{2x-6}{x+1} \geq 0$: अंश शून्य पर $3$, हर शून्य पर
$-1$ (निषिद्ध)। \omterm{def:g10:algebra:signtable}{साइन टेबल}: दोनों भाग होने पर भागफल सकारात्मक होता है
एक ही चिन्ह है, अर्थात\ के लिए $ x < -1$ या $ x > 3$; यह गायब हो जाता है $ x = 3$.
समाधान सेट: $\intoo{-\infty}{-1} \cup \intco{3}{+\infty}$.
\end{solution}

\begin{solution}{exo:g10:algebra:9}
बाद $ x $ पहली सेवा लागत के महीने $9x + 15$ और दूसरा $12x $.
पहला सस्ता है जब
\[
9x + 15 < 12x
\ \Longleftrightarrow\
15 < 3x
\ \Longleftrightarrow\
x > 5 .
\]
से $6$ आठवें महीने में, पहली सेवा कुल मिलाकर सस्ता विकल्प है।
\end{solution}

\begin{solution}{exo:g10:algebra:10}
निषिद्ध मूल्य: $x = 1$. के लिए $ x \neq 1$, दोनों पक्षों को गुणा करें $ x - 1$:
\[
x + 3 = 2(x - 1) = 2x - 2
\ \Longleftrightarrow\
x = 5 .
\]
तब से $5 \neq 1$, समाधान है $ x = 5$. जाँच करना:
$\frac{5+3}{5-1} = \frac84 = 2$.
\end{solution}

\begin{solution}{exo:g10:algebra:11}
$(n+1)^2 - n^2 = n^2 + 2n + 1 - n^2 = 2n + 1$, जो हर किसी के लिए अजीब है
पूर्णांक $ n $.

दो क्रमागत विषम संख्याएँ लिखी जा सकती हैं $2n - 1$ और $2n + 1$. तब
\[
(2n+1)^2 - (2n-1)^2
= \bigl[(2n+1)+(2n-1)\bigr]\bigl[(2n+1)-(2n-1)\bigr]
= 4n \times 2 = 8n,
\]
का एक गुणक $8$.
\end{solution}

\begin{solution}{pb:g10:algebra:1}
\textbf{1.} $ x^2 - 10x + 25 = (x - 5)^2$;
$9x^2 - 49 = (3x - 7)(3x + 7)$;
$ x^2 + 6x + 9 - 4 = (x + 3)^2 - 2^2 = (x + 1)(x + 5)$.

\textbf{2.} $ x = -1$ या $ x = -5$; \quad $ x = \frac73$ या
$ x = -\frac73$.

\textbf{3.} $(5 - t)(5 + t) = 25 - t^2 = 21$ देता है $ t^2 = 4$,
$ t = 2$: संख्याएं हैं $3$ और $7$ (जोड़ $10$, उत्पाद $21$:
जाँच करना)।

\textbf{4.} $(7 - t)(7 + t) = 49 - t^2 = 33$: $ t = 4$, संख्याएँ
$3$ और $11$.

\textbf{5.} $\left(\frac s2 - t\right)\left(\frac s2 + t\right)
= \left(\frac s2\right)^2 - t^2 = p $ बल
$ t^2 = \left(\frac s2\right)^2 - p $. एक समाधान बिल्कुल मौजूद है
जब यह है $\geq 0$, यानी\ कब
$ p \leq \left(\frac s2\right)^2$ --- दो संख्याओं का गुणनफल
दिए गए योग का योग आधे योग के वर्ग को कभी नहीं हरा सकता
(रानी डिडो की बाड़ समस्या, मिडिल स्कूल वॉल्यूम में पता था)।

\textbf{6.} $(x - a)(x - b) = x^2 - (a + b)x + ab $: द
\omterm{def:g10:algebra:equation}{समीकरण} $ x^2 - sx + p = 0$ सटीक रूप से कहता है ``$ x $ दो में से एक है
योग के साथ संख्याएँ $ s $ और उत्पाद $ p $''.

\textbf{7.} $(x-5)^2 - 4 = x^2 - 10x + 25 - 4 = x^2 - 10x +
21$. वर्गों का अंतर:
$(x - 5 - 2)(x - 5 + 2) = (x - 7)(x - 3) = 0$: समाधान $3$
और $7$ ---प्रश्न 3 के लेखक के अंक.

\textbf{8.} $ x^2 + 6x - 7 = (x + 3)^2 - 16 = 0$ देता है
$(x + 3)^2 = 16$, $ x + 3 = \pm 4$: $ x = 1$ या $ x = -7$.

\textbf{9.} $ x^2 + 4x + 5 = (x + 2)^2 + 1 \geq 1 > 0$ हरएक के लिए
$ x $: कोई वास्तविक समाधान नहीं. प्रपत्र में $(x + h)^2 + k $: समाधान
ठीक उसी समय अस्तित्व में है $ k \leq 0$ (तब $(x+h)^2 = -k $ दो है
जड़ें $-h \pm \sqrt{-k}$); कब $ k > 0$ अभिव्यक्ति बनी रहती है
सकारात्मक.

\textbf{10.} भुजा का वर्ग $ x $ (क्षेत्र $ x^2$) दो के साथ
$3 \times x $ आयत ($6x $ कुल मिलाकर) क्षेत्रफल का L बनता है
$ x^2 + 6x = 7$; लापता $3 \times 3$ कोने (क्षेत्र $9$)
भुजा का एक बड़ा वर्ग पूरा करता है $ x + 3$. जोड़ा जा रहा है $9$ दोनों को
पक्ष: $(x + 3)^2 = 16$, इसलिए $ x + 3 = 4$ (लंबाई सकारात्मक हैं)
और $ x = 1$. बीजगणित और चित्र एक ही क्रिया हैं।

\textbf{11.} $(x-3)(x+2)$: जड़ों के बीच नकारात्मक. समाधान
का $< 0$: $ x \in \intoo{-2}{3}$.

\textbf{12.} $ x^2 - 10x + 21 \leq 0$, अर्थात\
$(x - 3)(x - 7) \leq 0$: जड़ों के बीच, सिरे शामिल हैं:
$ x \in \intcc{3}{7}$.

\textbf{13.} शून्य पर $ x = 1$, पर वर्जित है $ x = -2$; द
मूल के बाहर भागफल धनात्मक है:
$ x \in \intoo{-\infty}{-2} \cup \intco{1}{+\infty}$.

\textbf{14.} $25 - (x - 5)^2 = 25 - x^2 + 10x - 25 =
x(10 - x)$: पहचान सत्यापित. तब से $(x-5)^2 \geq 0$ हमेशा,
$ A(x) \leq 25$, बिल्कुल समानता के साथ जब $ x = 5$: द
$5 \times 5$ वर्गाकार कलम, अधिकतम $25$ एम $^2$ --- ग्रेड-6
तालिका, अब दो-पंक्ति का प्रमाण।

\textbf{15.} के लिए $ x > 0$, दावे को इससे गुणा करें $ x $:
$ x^2 + 1 \geq 2x $, यानी\ $ x^2 - 2x + 1 = (x - 1)^2 \geq 0$ ---
सत्य, केवल समानता के साथ $ x = 1$. द्वारा वापस विभाजित करना $ x > 0$
सब कुछ सुरक्षित रखता है (\cref{prop:g10:algebra:ineqrules}).

\textbf{16.} पार गुणा $\frac1x = \frac{x}{1 - x}$
(सभी लंबाई सकारात्मक): $ x^2 = 1 - x $, अर्थात\
$ x^2 + x - 1 = 0$.

\textbf{17.} $ x^2 + x - 1 = \left(x + \frac12\right)^2 -
\frac54 = 0$ देता है $ x + \frac12 = \pm\frac{\sqrt5}{2}$. द
सकारात्मक जड़: $ x = \frac{\sqrt5 - 1}{2} \approx 0.618$.

\textbf{18.} $\varphi = \frac{2}{\sqrt5 - 1} =
\frac{2(\sqrt5 + 1)}{(\sqrt5 - 1)(\sqrt5 + 1)}
= \frac{2(\sqrt5 + 1)}{5 - 1} = \frac{\sqrt5 + 1}{2}
\approx 1.618$.

\textbf{19.} $\varphi^2 = \frac{(\sqrt5 + 1)^2}{4} =
\frac{6 + 2\sqrt5}{4} = \frac{3 + \sqrt5}{2}$, और
$\varphi + 1 = \frac{\sqrt5 + 3}{2}$: बराबर। और
$\varphi - 1 = \frac{\sqrt5 - 1}{2} = x = \frac1\varphi $ (वो कैसे $\varphi $ परिभाषित किया गया था)। स्वर्णिम अनुपात संख्या है
जिसका वर्ग एक जोड़ता है और जिसका व्युत्क्रम एक घटाता है।

\textbf{20.} $\sqrt5$ है \omterm{ex:g10:numbers:classify}{तर्कहीन} (मिडिल स्कूल वॉल्यूम की अतार्किकता सप्ताहांत समस्या: $5$
कोई पूर्ण वर्ग नहीं है), और जोड़ना $1$ फिर आधा करके सुरक्षित रखता है
अतार्किकता (तर्कसंगत संचालन, \cref{pb:g10:numbers:1}).
अनुपात: $\frac53 \approx 1.667$, $\frac85 = 1.600$,
$\frac{13}{8} = 1.625$, $\frac{21}{13} \approx 1.615$:
बारी-बारी से ऊपर और नीचे, बंद करते हुए
$\varphi = 1.618\ldots$ --- लय-गिनती संख्या
मिडिल स्कूल वॉल्यूम की लय-गिनती सप्ताहांत समस्या गुप्त रूप से सुनहरे खंड का पीछा करती है; द
अनुक्रमों पर अध्याय उन्हें कार्य में पकड़ लेंगे।
\end{solution}
