\section*{Chapter \ref{ch:g10:stats} --- Descriptive Statistics}

\begin{solution}{exo:g10:stats:1}
Sum: $0+1+1+2+2+2+3+3+4+4+5+6+7+8+12 = 60$, so the \omterm{def:g10:stats:mean}{mean} is
$\frac{60}{15} = 4$. The series is already sorted and $N = 15$ is odd:
the \omterm{def:g10:stats:median}{median} is the $8$th value, $3$. \omterm{def:g10:stats:quartiles}{Range}: $12 - 0 = 12$.
\end{solution}

\begin{solution}{exo:g10:stats:2}
Frequencies: $\frac{7}{50} = 0.14$, $\frac{9}{50} = 0.18$,
$\frac{8}{50} = 0.16$, $\frac{10}{50} = 0.20$, $\frac{6}{50} = 0.12$,
$\frac{10}{50} = 0.20$ (sum $1$). \omterm{def:g10:stats:mean}{Mean} outcome:
\[
\bar x = \frac{7 \times 1 + 9 \times 2 + 8 \times 3 + 10 \times 4
+ 6 \times 5 + 10 \times 6}{50}
= \frac{7 + 18 + 24 + 40 + 30 + 60}{50}
= \frac{179}{50} = 3.58 .
\]
\end{solution}

\begin{solution}{exo:g10:stats:3}
$N = 12$ (even): the \omterm{def:g10:stats:median}{median} is the \omterm{prop:g10:coordgeom:midpoint}{midpoint} of the $6$th and $7$th
values, both equal to $6$: \omterm{def:g10:stats:median}{median} $6$.

$Q_1$: $\frac{12}{4} = 3$, so $Q_1$ is the $3$rd value: $4$.

$Q_3$: $\frac{3 \times 12}{4} = 9$, so $Q_3$ is the $9$th value: $8$.

Box plot: box from $4$ to $8$ with the \omterm{def:g10:stats:median}{median} bar at $6$, whiskers from
$1$ to $11$.
\end{solution}

\begin{solution}{exo:g10:stats:4}
The $24$ scores total $24 \times 11.5 = 276$. With the new score the
total is $276 + 19 = 295$ for $25$ students: new \omterm{def:g10:stats:mean}{mean}
$\frac{295}{25} = 11.8$.
\end{solution}

\begin{solution}{exo:g10:stats:5}
Current total: $5 \times 12 = 60$. A \omterm{def:g10:stats:mean}{mean} of $13$ over $6$ tests requires
a total of $78$, so the $6$th score must be $78 - 60 = 18$. A \omterm{def:g10:stats:mean}{mean} of
$14$ would require $84 - 60 = 24 > 20$: not reachable.
\end{solution}

\begin{solution}{exo:g10:stats:6}
\omterm{def:g10:stats:mean}{Mean} larger than $20$: for instance
$1,\ 2,\ 3,\ 10,\ 40,\ 50,\ 60$ --- \omterm{def:g10:stats:median}{median} $10$ (the $4$th sorted value),
\omterm{def:g10:stats:mean}{mean} $\frac{166}{7} \approx 23.7$.

\omterm{def:g10:stats:mean}{Mean} smaller than $6$: for instance
$0,\ 0,\ 0,\ 10,\ 10,\ 10,\ 10$ --- \omterm{def:g10:stats:median}{median} $10$, \omterm{def:g10:stats:mean}{mean}
$\frac{40}{7} \approx 5.7$.

The \omterm{def:g10:stats:mean}{mean} cannot go below $\frac{40}{7}$: for the \omterm{def:g10:stats:median}{median} (the $4$th sorted
value) to be $10$, the values in positions $4$ to $7$ must all be
$\geq 10$, so the total is at least $40$ and the \omterm{def:g10:stats:mean}{mean} at least
$\frac{40}{7}$ --- the second example is extremal.

These examples show that the \omterm{def:g10:stats:mean}{mean} and the \omterm{def:g10:stats:median}{median} are largely independent:
knowing one says little about the other, which is why a good summary
reports both.
\end{solution}

\begin{solution}{exo:g10:stats:7}
\emph{1.} \omterm{def:g10:stats:mean}{Mean}:
$\frac{9 \times 1800 + 12000}{10} = \frac{16200 + 12000}{10} = 2820$.
\omterm{def:g10:stats:median}{Median}: sorted, the $10$ salaries are nine values $1800$ then $12000$;
the \omterm{def:g10:stats:median}{median} is the \omterm{prop:g10:coordgeom:midpoint}{midpoint} of the $5$th and $6$th values, both $1800$:
\omterm{def:g10:stats:median}{median} $1800$.

\emph{2.} The \omterm{def:g10:stats:median}{median} ($1800$) describes the typical salary: it is what
$9$ employees out of $10$ actually earn. The \omterm{def:g10:stats:mean}{mean} ($2820$) is pulled up
by the single high salary and matches nobody's payslip.
\end{solution}

\begin{solution}{exo:g10:stats:8}
Total height: $12 \times 168 + 18 \times 163 = 2016 + 2934 = 4950$ cm for
$30$ students, so the \omterm{def:g10:stats:mean}{mean} is $\frac{4950}{30} = 165$ cm. It is closer to
the girls' \omterm{def:g10:stats:mean}{mean} because they are more numerous (weighted \omterm{def:g10:stats:mean}{mean}, weights
$12$ and $18$).
\end{solution}

\begin{solution}{exo:g10:stats:9}
\emph{1.} The new \omterm{def:g10:stats:mean}{mean} is
\[
\bar y = \frac{y_1 + \dots + y_N}{N}
= \frac{(a x_1 + b) + \dots + (a x_N + b)}{N}
= \frac{a(x_1 + \dots + x_N) + Nb}{N}
= a \bar x + b .
\]

\emph{2.} With $a = 1.8$ and $b = 32$: \omterm{def:g10:stats:mean}{mean}
$1.8 \times 20 + 32 = 68$ degrees Fahrenheit.
\end{solution}

\begin{solution}{pb:g10:stats:1}
\textbf{1.} \omterm{def:g10:stats:mean}{Mean}: $\frac{2\,550}{7} \approx 364$ thousand
euros; \omterm{def:g10:stats:median}{median}: the fourth of the seven ordered prices, $230$.
The \omterm{def:g10:stats:median}{median} describes the street --- six of the seven houses
cost nowhere near $364$.

\textbf{2.} Without the mansion: \omterm{def:g10:stats:mean}{mean} $\frac{1\,350}{6} = 225$,
\omterm{def:g10:stats:median}{median} $\frac{220 + 230}{2} = 225$. The \omterm{def:g10:stats:mean}{mean} fell by about
$139$; the \omterm{def:g10:stats:median}{median} by $5$. One extreme value can drag the \omterm{def:g10:stats:mean}{mean}
anywhere; the \omterm{def:g10:stats:median}{median} barely notices --- it is \emph{robust}.

\textbf{3.} \omterm{def:g10:stats:mean}{Mean} $= \frac{0 \times 20 + 1 \times 25 + 2 \times
30 + 3 \times 15 + 4 \times 10}{100} = \frac{170}{100} = 1.7$.
No family has $1.7$ children, but $1.7 \times 100 = 170$
\emph{is} the exact total number of children: the \omterm{def:g10:stats:mean}{mean} encodes
the total, redistributed equally --- its true talent.

\textbf{4.} $N = 15$: \omterm{def:g10:stats:median}{median} $=$ 8th value $= 11$;
$Q_1 =$ 4th value $= 8$; $Q_3 =$ 12th value $= 14$.

\textbf{5.} \omterm{def:g10:stats:quartiles}{Range} $19 - 4 = 15$: the full spread, extremes
included. \omterm{def:g10:stats:quartiles}{Interquartile range} $14 - 8 = 6$: the width of the
middle half --- spread without the extremes' noise.

\textbf{6.} A shift $+b$ moves \emph{every} value, so
differences of values --- \omterm{def:g10:stats:quartiles}{range}, IQR --- are unchanged:
$(x_i + b) - (x_j + b) = x_i - x_j$. A scaling by $a$
multiplies all differences by $a$: \omterm{def:g10:stats:quartiles}{range} and IQR are scaled by
$\abs a$.

\textbf{7.} Additive curve: $6 \to 9$ and $18 \to 21$ --- the
weak student gains $50\,\%$, the strong one $17\,\%$, and $21$
bursts the $20$-point scale only mildly. Multiplicative:
$6 \to 8$ and $18 \to 24$ --- ratios preserved, but $24$ is
impossible on a $20$-scale: the ceiling problem. Additive
curves flatter the bottom; multiplicative ones explode the
top.

\textbf{8.} Total marks: $20 \times 12 + 30 \times 10 = 540$,
so the merged \omterm{def:g10:stats:mean}{mean} is $\frac{540}{50} = 10.8$: class B's thirty
students outweigh class A's twenty --- means merge by
\emph{weighted} average, never by simple average.

\textbf{9.} \omterm{def:g10:stats:median}{Median} of $A$: $2$; of $B$: $5$. Merged list
$\{0, 1, 2, 3, 10\}$: \omterm{def:g10:stats:median}{median} $2$. The size-weighted average of
the \omterm{def:g10:stats:median}{medians} would be $\frac{3 \times 2 + 2 \times 5}{5} = 3.2
\neq 2$: \omterm{def:g10:stats:median}{medians} carry no total, so no merging formula exists
--- one must re-sort the whole list.

\textbf{10.} \omterm{def:g10:stats:mean}{Mean}: $\frac{31.3}{5} = 6.26$. \omterm{def:g10:stats:median}{Median}: $5.4$.
Trimmed \omterm{def:g10:stats:mean}{mean}: $\frac{5.3 + 5.4 + 5.5}{3} = 5.4$. One
enthusiastic (or corrupt) judge moved the \omterm{def:g10:stats:mean}{mean} by almost a
point; the \omterm{def:g10:stats:median}{median} and the trimmed \omterm{def:g10:stats:mean}{mean} shrugged --- hence the
trimming rules of judged sports.

\textbf{11.} Factory A: middle half within $0.2$ mm; factory B:
within $4$ mm. Same \omterm{def:g10:stats:mean}{mean}, opposite reliability: buy from A. The
\omterm{def:g10:stats:quartiles}{interquartile range} decided --- means alone compare centers,
never consistency.

\textbf{12.} \omterm{def:g10:stats:median}{Median} far below \omterm{def:g10:stats:mean}{mean} betrays a \emph{long right
tail}: most waits short, a few catastrophic. Example:
$\{5, 5, 10, 10, 95\}$ --- \omterm{def:g10:stats:median}{median} $10$, \omterm{def:g10:stats:mean}{mean}
$\frac{125}{5} = 25$.

\textbf{13.} It means $75\,\%$ of babies of that age are
shorter (and $25\,\%$ taller): a rank, not a size. ``Every
child above the \omterm{def:g10:stats:mean}{mean}'' is impossible --- if all values exceeded
the \omterm{def:g10:stats:mean}{mean}, their average would exceed the \omterm{def:g10:stats:mean}{mean}, which is itself:
contradiction. But \emph{almost} all can be: one very short
child can hold the \omterm{def:g10:stats:mean}{mean} below everyone else. Above the
\emph{\omterm{def:g10:stats:median}{median}}: never more than half, by definition. Ranks
cannot be flattered; averages can.

\textbf{14.} Typical mark: same, \omterm{def:g10:stats:median}{median} $11$ for both.
Homogeneity of the middle: B tighter (IQR $3$ against $5$).
Extremes: B wilder (\omterm{def:g10:stats:quartiles}{range} $17$ against $13$). B is a class of
similar students plus a few outliers; A is evenly spread.

\textbf{15.} (1) Is $5\,000$ the \omterm{def:g10:stats:mean}{mean} or the \omterm{def:g10:stats:median}{median} --- and may
I have the \omterm{def:g10:stats:median}{median}? (2) What is the spread (\omterm{def:g10:stats:quartiles}{quartiles}), so I
know how typical ``typical'' is? (3) Who was counted --- sample
size and selection (the how-data-lies weekend problem of the Middle School volume's poll lesson)?

\textbf{16.} For instance $100, 150, 200, 250, 800$: \omterm{def:g10:stats:median}{median}
$200$, \omterm{def:g10:stats:mean}{mean} $\frac{1\,500}{5} = 300$.

\textbf{17.} $\sum (x_i - \bar x) = \sum x_i - N \bar x =
N\bar x - N\bar x = 0$. Check: $-200 - 150 - 100 - 50 + 500 =
0$. The \omterm{def:g10:stats:mean}{mean} is the point where the deviations balance --- the
data's center of gravity (the centroid weekend problem of the Middle School volume's $G$, in
one dimension).

\textbf{18.} At the \omterm{def:g10:stats:median}{median} $4$: $3 + 1 + 0 + 5 + 9 = 18$ km. At
the \omterm{def:g10:stats:mean}{mean} $6$: $5 + 3 + 2 + 3 + 7 = 20$ km. At $5$:
$4 + 2 + 1 + 4 + 8 = 19$ km. The \omterm{def:g10:stats:median}{median} wins --- moving away
from it always passes more shops than it approaches, so the
total can only grow.

\textbf{19.} $0.3^{10} \approx 6 \times 10^{-6}$: about six
pilots \emph{per million}. In $4\,000$ pilots the expected
count is $0.02$ --- zero, as lieutenant Daniels found. The Air
Force stopped designing for the average man and made
everything adjustable: seats, pedals, straps --- design for the
\emph{spread}, not the center.

\textbf{20.} Totals: the \emph{\omterm{def:g10:stats:mean}{mean}}. Typical individual: the
\emph{\omterm{def:g10:stats:median}{median}}. Fairness of spread: the \emph{\omterm{def:g10:stats:quartiles}{quartiles}} (and
IQR). Contamination: the \emph{trimmed \omterm{def:g10:stats:mean}{mean}}. And the closing
line: the average person does not exist --- but the \omterm{def:g10:stats:median}{median}
person almost does.
\end{solution}
