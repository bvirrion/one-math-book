\section*{Chapter \ref{ch:g10:proba} --- Probability and Sampling}

\begin{solution}{exo:g10:proba:1}
Each of the $6$ outcomes has \omterm{def:g10:proba:distribution}{probability} $\frac16$.
``Getting a $6$'': $\frac16$.
``Odd number'' $= \{1, 3, 5\}$: $\frac36 = \frac12$.
``At most $4$'' $= \{1,2,3,4\}$: $\frac46 = \frac23$.
``Getting a $7$'': impossible \omterm{def:g10:proba:events}{event}, \omterm{def:g10:proba:distribution}{probability} $0$.
\end{solution}

\begin{solution}{exo:g10:proba:2}
There are $10$ equally likely marbles.
$\P(\text{green}) = \frac{5}{10} = \frac12$.
$\P(\text{not black}) = 1 - \frac{2}{10} = \frac{8}{10} = \frac45$.
Green and yellow are \omterm{def:g10:proba:operations}{incompatible}:
$\P(\text{green or yellow}) = \frac{5}{10} + \frac{3}{10} = \frac{8}{10}
= \frac45$.
\end{solution}

\begin{solution}{exo:g10:proba:3}
Let $S$: ``studies Spanish'', $G$: ``studies German''. Then
$\P(S) = \frac{18}{30}$, $\P(G) = \frac{10}{30}$,
$\P(S \cap G) = \frac{4}{30}$. Addition rule:
\[
\P(S \cup G) = \frac{18}{30} + \frac{10}{30} - \frac{4}{30}
= \frac{24}{30} = \frac45 .
\]
\omterm{def:g10:proba:operations}{Complement}: $\P(\text{neither}) = 1 - \frac45 = \frac15$.
\end{solution}

\begin{solution}{exo:g10:proba:4}
The \omterm{met:g10:proba:tree}{tree} has four paths HH, HT, TH, TT, each of \omterm{def:g10:proba:distribution}{probability}
$\frac12 \times \frac12 = \frac14$.
Two heads: $\frac14$. Exactly one head (HT or TH): $\frac24 = \frac12$.
At least one head: $1 - \P(\text{TT}) = 1 - \frac14 = \frac34$.
\end{solution}

\begin{solution}{exo:g10:proba:5}
First draw: red with \omterm{def:g10:proba:distribution}{probability} $\frac{4}{10}$, white $\frac{6}{10}$.
Second draw \emph{without} replacement: after a red, $3$ red and $6$
white remain among $9$; after a white, $4$ red and $5$ white among $9$.

Two reds: $\frac{4}{10} \times \frac39 = \frac{12}{90} = \frac{2}{15}$.

Different colors: paths RW and WR:
\[
\frac{4}{10} \times \frac69 + \frac{6}{10} \times \frac49
= \frac{24}{90} + \frac{24}{90} = \frac{48}{90} = \frac{8}{15}.
\]
\end{solution}

\begin{solution}{exo:g10:proba:6}
\emph{1.} Each die independently shows each face with \omterm{def:g10:proba:distribution}{probability}
$\frac16$, so the $36$ ordered pairs are equally likely; the sum is
determined by the pair.

\emph{2.} Sum $7$: the pairs $(1,6), (2,5), (3,4), (4,3), (5,2), (6,1)$,
so $\P = \frac{6}{36} = \frac16$. Sum $12$: only $(6,6)$,
$\P = \frac{1}{36}$. Sum at most $4$: pairs summing to $2$, $3$ or $4$:
$(1,1)$; $(1,2), (2,1)$; $(1,3), (2,2), (3,1)$ --- six pairs, so
$\P = \frac{6}{36} = \frac16$.

\emph{3.} The number of pairs achieving each sum grows from $1$ (sum $2$)
to $6$ (sum $7$) then decreases: the most likely sum is $7$.
\end{solution}

\begin{solution}{exo:g10:proba:7}
Each question is answered correctly with \omterm{def:g10:proba:distribution}{probability} $\frac14$,
independently. \omterm{met:g10:proba:tree}{Tree} with two levels:

Both correct: $\frac14 \times \frac14 = \frac{1}{16}$.

Exactly one correct: paths correct--wrong and wrong--correct:
$\frac14 \times \frac34 + \frac34 \times \frac14 = \frac{6}{16}
= \frac38$.

None correct: $\frac34 \times \frac34 = \frac{9}{16}$.

Check: $\frac{1}{16} + \frac{6}{16} + \frac{9}{16} = 1$.
\end{solution}

\begin{solution}{exo:g10:proba:8}
\emph{1.} Observed \omterm{def:g10:stats:series}{frequency}: $\frac{396}{900} = 0.44$. With $n = 900$:
$\frac{1}{\sqrt{900}} = \frac{1}{30} \approx 0.033$, so the fluctuation
\omterm{def:g10:numbers:interval}{interval} is about $\intcc{0.467}{0.533}$.

\emph{2.} The observed $0.44$ lies outside the \omterm{def:g10:numbers:interval}{interval}: \omterm{def:g10:proba:sample}{samples} of $900$
voters from a population with $50\%$ approval rarely stray that far. The
poll casts serious doubt on the claimed $50\%$.
\end{solution}

\begin{solution}{exo:g10:proba:9}
\emph{1.} $\P(\text{win } 6) = \frac16$ (roll a $6$),
$\P(\text{win } 3) = \frac16$ (roll a $5$),
$\P(\text{win } 0) = \frac46 = \frac23$.

\emph{2.} Over $600$ games, expect about $100$ sixes, $100$ fives, $400$
other rolls. Winnings: about $100 \times 6 + 100 \times 3 = 900$. Cost:
$600 \times 2 = 1200$. Expected net result: $900 - 1200 = -300$, i.e.\ a
loss of about $0.5$ per game on average --- the game is not worth
playing.
\end{solution}

\begin{solution}{pb:g10:proba:1}
\textbf{1.} $P(\text{king or heart}) = \frac{4}{52} +
\frac{13}{52} - \frac{1}{52} = \frac{16}{52} = \frac{4}{13}$:
the king of hearts sits in both \omterm{def:g10:proba:events}{events} and must not be counted
twice --- hence the subtraction
(\cref{thm:g10:proba:addition}).

\textbf{2.} $P(\text{no head}) = \left(\frac12\right)^3 =
\frac18$, so $P(\text{at least one}) = \frac78$.

\textbf{3.} $P(RR) = \frac35 \times \frac24 = \frac{3}{10}$;
$P(\text{one of each}) = \frac35 \times \frac24 + \frac25
\times \frac34 = \frac{6}{20} + \frac{6}{20} = \frac35$.

\textbf{4.} $P(BB) = \frac25 \times \frac14 = \frac1{10}$; and
$\frac{3}{10} + \frac{6}{10} + \frac{1}{10} = 1$: the \omterm{met:g10:proba:tree}{tree}'s
leaves always total $1$.

\textbf{5.} \omterm{def:g10:proba:operations}{Incompatible} \omterm{def:g10:proba:events}{events} obey $P(A \cup B) = P(A) +
P(B) = 1.1 > 1$: impossible --- some \omterm{def:g10:proba:distribution}{probability} was invented.

\textbf{6.} (Most people, including the thousand letter-writers,
answer ``it doesn't matter, 50--50''. Keep your answer for
question 8.)

\textbf{7.} Car behind door 1 (\omterm{def:g10:proba:distribution}{probability} $\frac13$): the host
opens either goat door; switching leaves the car: \emph{lose}.
Car behind door 2 ($\frac13$): the host \emph{must} open door 3;
switching lands on door 2: \emph{win}. Car behind door 3:
symmetric: \emph{win}. Total:
$P(\text{switch wins}) = \frac13 + \frac13 = \frac23$.

\textbf{8.} Your first pick is wrong with \omterm{def:g10:proba:distribution}{probability}
$\frac23$ --- and precisely then, the host's forced reveal
leaves the car behind the remaining door: switching converts
every initial error into a win. $\frac23$, whatever the gut
said.

\textbf{9.} Six equally likely scenarios: car behind $1$, $2$
or $3$, host opens door $2$ or $3$ (never yours). The reveal
shows the car in two of them (car@2/opens 2, car@3/opens 3):
discarded. Among the four survivors --- car@1/opens 2,
car@1/opens 3, car@2/opens 3, car@3/opens 2 --- switching wins
in the last two: $\frac24 = \frac12$. With an ignorant host,
50--50 \emph{is} correct: the paradox lives entirely in the
host's knowledge, which filters the scenarios unevenly.

\textbf{10.} Your pick holds the car with \omterm{def:g10:proba:distribution}{probability}
$\frac{1}{100}$; the knowing host's $98$ reveals funnel the
remaining $\frac{99}{100}$ onto the one closed door. Switch,
and win $99$ times in $100$.

\textbf{11.} Three cards (one ace, two jokers), a friend who
peeks and always discards a joker from the two you did not
pick; play $300$ rounds, always switching, and record wins.
Expected \omterm{def:g10:stats:series}{frequency} $\frac23 \approx 0.667$, with fluctuation
$\pm\frac{1}{\sqrt{300}} \approx 0.058$: expect between about
$61\,\%$ and $72\,\%$ switching wins --- far from any
``50--50''.

\textbf{12.} Equally likely configurations (elder--younger):
BB, BG, GB, GG. ``At least one boy'' keeps BB, BG, GB:
$P(\text{BB}) = \frac13$.

\textbf{13.} ``The elder is a boy'' keeps BB, BG:
$P(\text{BB}) = \frac12$. Different information filters
different scenario sets --- exactly as the two Montys did: what
you learn matters, and \emph{how} you learned it matters just
as much.

\textbf{14.} All alike: HHH, TTT: $\frac28 = \frac14$.
D'Alembert's error: his two ``cases'' are not equally likely
--- ``not all alike'' bundles six of the eight outcomes.
\Cref{prop:g10:proba:uniform} counts only over equally likely
outcomes.

\textbf{15.} Average winnings per game: $6 \times \frac16 + 3
\times \frac16 = 1.50$ euros, against a $2$-euro stake: an
average loss of $0.50$ per game, about $300$ euros over $600$
games. The fair premium equalizes average payout:
$1\,\% \times 10\,000 = 100$ euros; real premiums add costs
and a safety margin --- the insurer must survive bad years,
not just average ones.

\textbf{16.} \omterm{def:g10:numbers:interval}{Interval}: $0.5 \pm \frac{1}{\sqrt{100}} =
\intcc{0.4}{0.6}$. $55$ heads ($0.55$): inside, unremarkable.
$70$ heads ($0.70$): far outside --- doubt the coin.

\textbf{17.} $\frac{1}{\sqrt{1000}} \approx 0.032$: a fair
$50\,\%$ candidate would typically poll in
$\intcc{0.468}{0.532}$ --- and $0.52$ sits comfortably inside.
The poll \emph{suggests} a lead but proves nothing: the margin
swallows it.

\textbf{18.} First \omterm{def:g10:proba:sample}{sample}: \omterm{def:g10:numbers:interval}{interval} $0.04 \pm \frac{1}{20} =
\intcc{0}{0.09}$; the observed $0.07$ is inside: no verdict.
Second: $0.04 \pm \frac{1}{50} = \intcc{0.02}{0.06}$; the
observed $0.07$ falls outside: the claim is now in serious
doubt. Same \omterm{def:g10:stats:series}{frequency}, sharper instrument: precision grows like
$\sqrt n$.

\textbf{19.} Fluctuation shrinks like $\frac{1}{\sqrt n}$:
quadruple the \omterm{def:g10:proba:sample}{sample}, halve the noise. Bias --- asking the
wrong crowd --- survives any $n$: two million badly chosen
ballots stayed wrong (the how-data-lies weekend problem of the Middle School volume).

\textbf{20.} (i) Check outcomes are equally likely before
counting --- d'Alembert forgot. (ii) \omterm{met:g10:proba:tree}{Trees} multiply along
branches; ``at least one'' surrenders to the \omterm{def:g10:proba:operations}{complement}.
(iii) Information filters scenarios, and its \emph{source}
changes the filter --- knowing Monty against fallen Monty,
elder-boy against some-boy. (iv) Frequencies approach
probabilities at speed $\frac{1}{\sqrt n}$, so a patient
simulation settles any argument --- including one with ten
thousand angry letter-writers.
\end{solution}
