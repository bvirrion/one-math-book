\begin{solution}{exo:g10:coordgeom:2}

\begin{solution}{exo:g10:coordgeom:3}
$OP = \sqrt{16 + 1} = \sqrt{17}$; $OQ = \sqrt{9 + 4} = \sqrt{13}$;
$OR = \sqrt{0 + 25} = 5 = \sqrt{25}$. Since $13 < 17 < 25$, the square

\end{solution}

\begin{solution}{exo:g10:coordgeom:4}
Squared lengths:
\begin{align*}
AB^2 &= (5-1)^2 + (4-2)^2 = 16 + 4 = 20, \\
BC^2 &= (7-5)^2 + (0-4)^2 = 4 + 16 = 20, \\
AC^2 &= (7-1)^2 + (0-2)^2 = 36 + 4 = 40 .
\end{align*}
$AB = BC$: the triangle is isosceles at $B$. Moreover
$AB^2 + BC^2 = 40 = AC^2$: by the converse of the Pythagorean theorem it
is right-angled at $B$ as well.
\end{solution}

\begin{solution}{exo:g10:coordgeom:5}
\emph{1.} \omterm{prop:g10:coordgeom:midpoint}{Midpoint} of $[AC]$:
$\left(\frac{-2+4}{2}, \frac{1+1}{2}\right) = (1, 1)$; \omterm{prop:g10:coordgeom:midpoint}{midpoint} of
$[BD]$: $\left(\frac{1+1}{2}, \frac{3+(-1)}{2}\right) = (1, 1)$. Same
\omterm{prop:g10:coordgeom:midpoint}{midpoint}: $ABCD$ is a parallelogram.

\emph{2.} $AB^2 = 3^2 + 2^2 = 13$ and $BC^2 = 3^2 + (-4)^2 = 25$:
$AB \neq BC$, not a rhombus.

\emph{3.} $AC^2 = 6^2 + 0^2 = 36$ and $BD^2 = 0^2 + (-4)^2 = 16$:
$AC \neq BD$, not a rectangle. $ABCD$ is a plain parallelogram.
\end{solution}

\begin{solution}{exo:g10:coordgeom:6}
Compute squared distances to $\Omega(2,1)$ and compare with $5^2 = 25$:
\[
\Omega A^2 = 3^2 + 4^2 = 25, \qquad
\Omega B^2 = (-3)^2 + (-4)^2 = 25, \qquad
\Omega C^2 = 4^2 + 1^2 = 17 .
\]
$A$ and $B$ are on the circle; $C$ is inside it ($17 < 25$).
\end{solution}

\begin{solution}{exo:g10:coordgeom:7}
$MA^2 = (x - 1)^2 + 1$ and $MB^2 = (x - 7)^2 + 25$. Setting
$MA^2 = MB^2$:
\[
x^2 - 2x + 2 = x^2 - 14x + 74
\ \Longleftrightarrow\
12x = 72
\ \Longleftrightarrow\
x = 6 .
\]
The unique such point is $M(6, 0)$.
\end{solution}

\begin{solution}{exo:g10:coordgeom:8}
\emph{1.} $I = \left(\frac{0+4}{2}, \frac{3+1}{2}\right) = (2, 2)$.

\emph{2.} $M(2,2)$ is indeed $I$. And
$MA^2 = (0-2)^2 + (3-2)^2 = 5$, $MB^2 = (4-2)^2 + (1-2)^2 = 5$: the
\omterm{prop:g10:coordgeom:midpoint}{midpoint} is equidistant from the two endpoints, as expected.

\emph{3.} Look for $N(0, y)$ with $NA^2 = NB^2$:
\[
(3 - y)^2 = 16 + (1 - y)^2
\ \Longleftrightarrow\
9 - 6y = 16 + 1 - 2y
\ \Longleftrightarrow\
-4y = 8,
\]
so $y = -2$: the point $N(0, -2)$.
\end{solution}

\begin{solution}{exo:g10:coordgeom:9}
\begin{align*}
AB &= \sqrt{4^2 + 2^2} = \sqrt{20} = 2\sqrt5, \\
BC &= \sqrt{8^2 + 4^2} = \sqrt{80} = 4\sqrt5, \\
AC &= \sqrt{12^2 + 6^2} = \sqrt{180} = 6\sqrt5 .
\end{align*}
Then $AB + BC = 2\sqrt5 + 4\sqrt5 = 6\sqrt5 = AC$: the triangle
inequality is an equality, so $B$ lies on the segment $[AC]$ --- the
three points are aligned.
\end{solution}

\begin{solution}{exo:g10:coordgeom:10}
The \omterm{prop:g10:coordgeom:midpoint}{midpoint} of $[AB]$ is $I\left(\frac a2, \frac b2\right)$. Then
\[
OI^2 = \frac{a^2}{4} + \frac{b^2}{4},
\qquad
IA^2 = \left(a - \frac a2\right)^2 + \left(0 - \frac b2\right)^2
= \frac{a^2}{4} + \frac{b^2}{4},
\]
and $IB^2 = \left(0 - \frac a2\right)^2 + \left(b - \frac b2\right)^2 =
\frac{a^2}{4} + \frac{b^2}{4}$ as well. The three squared distances are
equal, so $OI = IA = IB$: the \omterm{prop:g10:coordgeom:midpoint}{midpoint} of the hypotenuse of the right
triangle $OAB$ is equidistant from all three vertices (it is the center
of its circumscribed circle).
\end{solution}

\begin{solution}{pb:g10:coordgeom:1}
\textbf{1.} $\Omega M = 5 \iff \Omega M^2 = 25 \iff
(x - 2)^2 + (y - 1)^2 = 25$ by the distance formula
(\cref{thm:g10:coordgeom:distance}) --- squaring is harmless,
both sides being positive.

\textbf{2.} $A$: $(5-2)^2 + (5-1)^2 = 9 + 16 = 25$: on the
circle. $B$: $16 + 9 = 25$: on it too. $D$: $4 + 9 = 13 < 25$:
inside.

\textbf{3.} $x^2 - 6x = (x - 3)^2 - 9$ and
$y^2 + 4y = (y + 2)^2 - 4$, so the \omterm{def:g10:algebra:equation}{equation} reads
$(x - 3)^2 + (y + 2)^2 = 25$: center $(3, -2)$, radius $5$.

\textbf{4.} $(x - 3)^2 + (y + 2)^2 = -1$: a sum of squares is
never negative --- no point satisfies it: the empty set. In
general $(x - h)^2 + (y - k)^2 = c$ is a circle of radius
$\sqrt c$ if $c > 0$, the single point $(h, k)$ if $c = 0$, and
empty if $c < 0$.

\textbf{5.} With $y = 3$: $(x - 3)^2 + 25 = 25$, so
$(x - 3)^2 = 0$: the single point $(3, 3)$. One contact point:
the line is \emph{tangent} to the circle.

\textbf{6.} $MA^2 + MB^2 = (x + r)^2 + y^2 + (x - r)^2 + y^2
= 2x^2 + 2y^2 + 2r^2 = 2r^2 + 2r^2 = 4r^2$ (using
$x^2 + y^2 = r^2$). Since $AB^2 = (2r)^2 = 4r^2$, the triangle
$AMB$ satisfies $MA^2 + MB^2 = AB^2$: right-angled at $M$
(converse of Pythagoras) --- three lines, as promised.

\textbf{7.} $MA^2 + MB^2 = (x + a)^2 + y^2 + (x - a)^2 + y^2
= 2x^2 + 2y^2 + 2a^2 = 2\,MI^2 + 2a^2$, and
$\frac{AB^2}{2} = \frac{(2a)^2}{2} = 2a^2$. Identity proved.

\textbf{8.} $2\,MI^2 + \frac{36}{2} = 26$ gives $MI^2 = 4$: the
circle of center the \omterm{prop:g10:coordgeom:midpoint}{midpoint} $I$ and radius $2$.

\textbf{9.} $MA^2 = 4\,MB^2$:
$x^2 + y^2 = 4\left((x - 3)^2 + y^2\right)$, i.e.\
$3x^2 + 3y^2 - 24x + 36 = 0$, or
$x^2 + y^2 - 8x + 12 = 0$: completing,
$(x - 4)^2 + y^2 = 4$ --- the circle of center $(4, 0)$ and
radius $2$ (the \emph{circle of Apollonius} of ratio $2$).

\textbf{10.} The bridge converts constructions into
computations: no auxiliary rectangle, no case analysis --- one
expansion and the theorem falls out. The price: computations
carry less geometric insight; the classical proof shows
\emph{why}, the algebraic one shows \emph{that} --- a working
mathematician keeps both.

\textbf{11.} $x^2 + y^2 = 25$ and $(x - 6)^2 + y^2 = 25$.
Subtracting: $x^2 - (x - 6)^2 = 0$, i.e.\ $12x - 36 = 0$:
$x = 3$. Then $9 + y^2 = 25$: $y = \pm 4$. Candidates
$(3, 4)$ and $(3, -4)$.

\textbf{12.} From $(3, 4)$ to $R(0, 8)$:
$\sqrt{9 + 16} = 5$ --- matches. From $(3, -4)$:
$\sqrt{9 + 144} \approx 12.4$ --- wrong. You are at $(3, 4)$.

\textbf{13.} Both \omterm{def:g10:algebra:equation}{equations} carry the same $x^2 + y^2$ (or with
equal coefficients after expansion): subtraction cancels the
squares and leaves an \omterm{def:g10:algebra:equation}{equation} of degree one --- a line. When
the circles meet twice, that line passes through both meeting
points: it is the line of the common chord.

\textbf{14.} Four unknowns: three \omterm{def:g10:coordgeom:system}{coordinates} plus the
receiver's clock error (a cheap watch, corrected by the
mathematics). Each satellite supplies one \omterm{def:g10:algebra:equation}{equation}, so four
satellites give four \omterm{def:g10:algebra:equation}{equations} for four unknowns --- with more
satellites improving the precision.

\textbf{15.} Subtraction now gives
$x = \frac{5.1^2 - 5^2 + 36}{12} = \frac{37.01}{12} \approx
3.084$: a $10$ cm error in one distance moved the estimate
about $8$ cm here. Errors propagate through every formula ---
the interval-arithmetic caution of \cref{pb:g10:numbers:1},
now with \omterm{def:g10:coordgeom:system}{coordinates}.

\textbf{16.} $B' (7, -3)$. The line $(AB')$ has \omterm{def:g10:reffunc:affine}{slope}
$\frac{-3 - 5}{7 - 1} = -\frac43$ and crosses $y = 0$ at
$x = 1 + \frac{5}{8} \times 6 = 4.75$: drink at
$P(4.75,\ 0)$. Minimal length: $AP + PB = AP + PB' = AB' =
\sqrt{6^2 + 8^2} = 10$.

\textbf{17.} Via $Q(3, 0)$:
$AQ + QB = \sqrt{4 + 25} + \sqrt{16 + 9} = \sqrt{29} + 5
\approx 10.39 > 10$. The reflected straight line wins.

\textbf{18.} Diagonal \omterm{prop:g10:coordgeom:midpoint}{midpoints}: $[AC]$: $(3, 3)$; $[BD]$:
$(3, 3)$ --- equal: parallelogram. Sides:
$AB = \sqrt{10} = AD$: rhombus. Diagonals:
$AC = \sqrt{32} \neq BD = \sqrt{8}$: not a rectangle, hence
not a square. Verdict: a rhombus, and nothing more.

\textbf{19.} $D = A + C - B = (-1 + 6 - 3,\ 2 + 0 - 4) =
(2, -2)$: then $[AC]$ and $[BD]$ share the \omterm{prop:g10:coordgeom:midpoint}{midpoint}
$\left(\frac52, 1\right)$.

\textbf{20.} \omterm{prop:g10:coordgeom:midpoint}{Midpoint} formula: parallelograms and centers
(questions 18, 19). Distance formula: circles, loci, triangle
natures, shortest paths (questions 1--9, 16). Subtraction of
circle \omterm{def:g10:algebra:equation}{equations}: the line that locates a position (questions
11--15). Missing from the kit: a clean algebra of
\emph{directions} --- vectors --- and of \emph{\omterm{def:g10:reffunc:affine}{slopes}} ---
line \omterm{def:g10:algebra:equation}{equations} and systems: the next two chapters supply
exactly that.
\end{solution}
