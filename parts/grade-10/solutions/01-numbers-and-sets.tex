\section*{Chapter \ref{ch:g10:numbers} --- Numbers and Sets of Numbers}

\begin{solution}{exo:g10:numbers:1}
$\frac{15}{3} = 5 \in \N$. \quad $-7 \in \Z$. \quad
$\frac{22}{7} \in \Q$ (it is not an \omterm{def:g10:numbers:sets}{integer}: $22 = 7 \times 3 + 1$).
\quad $\sqrt 9 = 3 \in \N$. \quad $\sqrt{10} \in \R$ (\omterm{ex:g10:numbers:classify}{irrational}, since
$10$ is not the square of a rational --- admitted here, in the spirit of
\cref{thm:g10:numbers:sqrt2}). \quad $-2.4 = -\frac{24}{10} \in \Q$.
\quad $0 \in \N$.
\end{solution}

\begin{solution}{exo:g10:numbers:2}
(a) $\intco{-2}{5}$: filled dot at $-2$, hollow dot at $5$.
(b) $\intoo{3}{+\infty}$: hollow dot at $3$, shading to the right.
(c) $\intoc{-\infty}{-1}$: shading from the left up to a filled dot at
$-1$.
(d) ``distance from $x$ to $2$ at most $3$'' means $\abs{x - 2} \leq 3$,
i.e.\ $x \in \intcc{-1}{5}$: filled dots at $-1$ and $5$.
\end{solution}

\begin{solution}{exo:g10:numbers:3}
(a) The two \omterm{def:g10:numbers:interval}{intervals} overlap between $0$ and $2$:
$I \cap J = \intcc{0}{2}$ ($0 \in J$ and $2 \in I$, and both belong to the
other set as well), and $I \cup J = \intco{-3}{4}$.

(b) $I \cap J$ is the set of $x$ with $-2 \leq x$ and $x < 1$:
$\intco{-2}{1}$. The \omterm{def:g10:numbers:interunion}{union} covers everything: $I \cup J = \R$.
\end{solution}

\begin{solution}{exo:g10:numbers:4}
$\abs{-6} = 6$; \quad $\abs{4 - 9} = \abs{-5} = 5$; \quad
$\abs{-3 - 5} = \abs{-8} = 8$; \quad
$\sqrt 2 > 1$ so $\abs{\sqrt2 - 1} = \sqrt2 - 1$; \quad
$1 - \sqrt2 < 0$ so $\abs{1 - \sqrt2} = \sqrt2 - 1$ as well (a number and
its opposite have the same \omterm{def:g10:numbers:abs}{absolute value}).
\end{solution}

\begin{solution}{exo:g10:numbers:5}
$\abs{x} = 5$: distance $5$ from $0$, so $x = 5$ or $x = -5$.

$\abs{x - 1} = 3$: distance $3$ from $1$, so $x = 4$ or $x = -2$.

$\abs{x - 4} \leq 1$: distance at most $1$ from $4$, so
$x \in \intcc{3}{5}$.

$\abs{x + 2} < 3$: distance strictly less than $3$ from $-2$, so
$x \in \intoo{-5}{1}$.
\end{solution}

\begin{solution}{exo:g10:numbers:6}
Each \omterm{def:g10:numbers:interval}{interval} is described by its center $a$ (midpoint of the bounds) and
radius $r$ (half the length).

$\intcc{2}{8}$: $a = 5$, $r = 3$, so $\abs{x - 5} \leq 3$.

$\intoo{-5}{1}$: $a = -2$, $r = 3$, open bounds, so $\abs{x + 2} < 3$.

$\intcc{-7}{-3}$: $a = -5$, $r = 2$, so $\abs{x + 5} \leq 2$.
\end{solution}

\begin{solution}{exo:g10:numbers:7}
Let $x = 0.272727\dots$ Then $100x = 27.2727\dots$, and subtracting:
\[
100x - x = 27.2727\dots - 0.2727\dots = 27,
\]
so $99x = 27$ and $x = \frac{27}{99} = \frac{3}{11}$, a quotient of
\omterm{def:g10:numbers:sets}{integers}: $x$ is rational.
\end{solution}

\begin{solution}{exo:g10:numbers:8}
\emph{1. True:} adding \omterm{def:g10:numbers:sets}{integers} (positive or negative whole numbers)
always produces an \omterm{def:g10:numbers:sets}{integer}.

\emph{2. False:} $\frac{1}{2}$ is a quotient of the \omterm{def:g10:numbers:sets}{integers} $1$ and $2$
and is not an \omterm{def:g10:numbers:sets}{integer}.

\emph{3. True:} $\frac pq + \frac{p'}{q'} = \frac{pq' + p'q}{qq'}$ is
again a quotient of \omterm{def:g10:numbers:sets}{integers} (with nonzero denominator).

\emph{4. True:} suppose $r$ is rational, $t$ \omterm{ex:g10:numbers:classify}{irrational}, and $r + t = s$
were rational. Then $t = s - r$ would be a difference of two rationals,
hence rational (by 3, applied with $-r$) --- contradiction. So $r + t$ is
\omterm{ex:g10:numbers:classify}{irrational}.
\end{solution}

\begin{solution}{exo:g10:numbers:9}
Multiplying $1.414 \leq \sqrt2 \leq 1.415$ by $2 > 0$:
$2.828 \leq 2\sqrt2 \leq 2.830$.

Adding $3$: $4.414 \leq \sqrt2 + 3 \leq 4.415$.

Multiplying by $-1 < 0$ reverses the inequalities:
$-1.415 \leq -\sqrt2 \leq -1.414$.
\end{solution}

\begin{solution}{exo:g10:numbers:10}
First the auxiliary fact. Divide $p$ by $3$: the remainder is $0$, $1$ or
$2$, i.e.\ $p = 3k$, $p = 3k+1$ or $p = 3k+2$. Squaring:
\[
(3k)^2 = 3(3k^2), \quad
(3k+1)^2 = 3(3k^2 + 2k) + 1, \quad
(3k+2)^2 = 3(3k^2 + 4k + 1) + 1 .
\]
Only the first is a multiple of $3$: if $p^2$ is a multiple of $3$, then
$p$ is too.

Now suppose $\sqrt3 = \frac pq$ fully simplified. Squaring gives
$p^2 = 3q^2$, so $p^2$ is a multiple of $3$, so $p = 3k$. Then
$9k^2 = 3q^2$, so $q^2 = 3k^2$ and $q$ is also a multiple of $3$ --- but
then the fraction $\frac pq$ was not fully simplified: contradiction. So
$\sqrt3$ is \omterm{ex:g10:numbers:classify}{irrational}.
\end{solution}
