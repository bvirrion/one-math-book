\chapter{Vecteurs}\label{ch:g10:vectors}

Un \omterm{def:g10:vectors:vector}{vecteur} encode un déplacement~: une direction et une longueur,
indépendamment du point de départ. Les \omterm{def:g10:vectors:vector}{vecteurs} donnent des preuves
propres de faits géométriques et, une fois les \omterm{def:g10:coordgeom:system}{coordonnées} en jeu, les
réduisent à de petits calculs. Ils seront un outil clé dans le
\cref{ch:g10:lines} et, plus tard, pour le produit scalaire dans le
\cref{ch:g11:scal}.

\section{Translations et vecteurs}

\begin{definition}[Vecteur]\label{def:g10:vectors:vector}
Étant donnés deux points $A$ et $B$, le \emph{vecteur}\index{vecteur}
$\vect{AB}$ représente le déplacement de $A$ vers $B$~: il a une
\emph{direction} (la direction de la droite $(AB)$, avec le sens de
parcours, de $A$ vers $B$) et une \emph{longueur} (la distance $AB$,
aussi notée $\norm{\vect{AB}}$). Deux vecteurs sont \emph{égaux}
lorsqu'ils encodent le même déplacement~: $\vect{AB} = \vect{CD}$
signifie qu'appliquer les deux déplacements à un point quelconque donne
le même résultat.
\end{definition}

\begin{proposition}[Vecteurs égaux et parallélogrammes]\label{prop:g10:vectors:parallelogram}
$\vect{AB} = \vect{CD}$ exactement lorsque $ABDC$ (dans cet ordre~!) est
un parallélogramme, éventuellement aplati --- de façon équivalente,
lorsque $[AD]$ et $[BC]$ ont le même \omterm{prop:g10:coordgeom:midpoint}{milieu}.
\end{proposition}
\admitted

\begin{notation}
Un \omterm{def:g10:vectors:vector}{vecteur} est souvent nommé par une seule lettre, $\vec u$ ou $\vec v$,
lorsque les extrémités n'importent pas. Le \emph{\omterm{def:g10:vectors:vector}{vecteur} nul}
$\vec 0 = \vect{AA}$ est le déplacement qui ne déplace rien.
L'\emph{opposé} de $\vect{AB}$ est $\vect{BA}$~: même longueur, sens
opposé. On écrit $\vect{BA} = -\vect{AB}$.
\end{notation}

\section{Addition de vecteurs}

\begin{definition}[Somme de deux vecteurs]\label{def:g10:vectors:sum}
La \emph{somme}\index{vecteur!somme} $\vec u + \vec v$ est le
déplacement obtenu en effectuant $\vec u$ puis $\vec v$.
\end{definition}

\begin{theorem}[Relation de Chasles]\label{thm:g10:vectors:chasles}
Pour trois points $A$, $B$, $C$ quelconques~:
\[
\vect{AB} + \vect{BC} = \vect{AC}.
\]
\end{theorem}

\begin{proof}
Aller de $A$ à $B$, puis de $B$ à $C$, est un déplacement qui amène $A$
en $C$~: c'est le déplacement encodé par $\vect{AC}$.
\end{proof}

\begin{remark}[Règle du parallélogramme]
Pour additionner deux \omterm{def:g10:vectors:vector}{vecteurs} issus du même point, $\vect{AB} +
\vect{AC}$, compléter le parallélogramme $ABDC$~: la \omterm{def:g10:vectors:sum}{somme} est la
diagonale $\vect{AD}$. En effet $\vect{AC} = \vect{BD}$, donc
$\vect{AB} + \vect{AC} = \vect{AB} + \vect{BD} = \vect{AD}$ par la
relation de Chasles.
\end{remark}

\begin{example}[Simplifier avec Chasles]\label{ex:g10:vectors:chasles}
Simplifier $\vect{MN} + \vect{NP} + \vect{PQ}$~: enchaîner les
déplacements donne $\vect{MQ}$. Simplifier $\vect{AB} - \vect{AC}$~:
réécrire la différence comme une \omterm{def:g10:vectors:sum}{somme} avec le \omterm{def:g10:vectors:vector}{vecteur} opposé,
\[
\vect{AB} - \vect{AC} = \vect{AB} + \vect{CA}
= \vect{CA} + \vect{AB} = \vect{CB}.
\]
\end{example}

\section{Multiplier un vecteur par un nombre}

\begin{definition}[Produit par un scalaire]\label{def:g10:vectors:scalar}
Soit $\vec u$ un \omterm{def:g10:vectors:vector}{vecteur} non nul et $k$ un \omterm{def:g10:numbers:sets}{nombre réel}. Le \omterm{def:g10:vectors:vector}{vecteur}
$k\vec u$\index{vecteur!produit par un scalaire} a la même direction
que $\vec u$, la longueur $\abs{k} \times \norm{\vec u}$, et le même sens
que $\vec u$ si $k > 0$, le sens opposé si $k < 0$. Pour $k = 0$,
$0\vec u = \vec 0$.
\end{definition}

\begin{definition}[Colinéarité]\label{def:g10:vectors:collinear}
Deux \omterm{def:g10:vectors:vector}{vecteurs} $\vec u$ et $\vec v$ sont
\emph{colinéaires}\index{vecteurs colinéaires} lorsqu'ils ont la même
direction, c'est-à-dire lorsque $\vec v = k \vec u$ pour un certain réel
$k$ (ou que l'un d'eux est $\vec 0$).
\end{definition}

\begin{remark}
La colinéarité est le langage vectoriel du parallélisme et de
l'alignement~:
\begin{itemize}
  \item les droites $(AB)$ et $(CD)$ sont parallèles exactement lorsque
        $\vect{AB}$ et $\vect{CD}$ sont \omterm{def:g10:vectors:collinear}{colinéaires}~;
  \item les points $A$, $B$, $C$ sont alignés exactement lorsque
        $\vect{AB}$ et $\vect{AC}$ sont \omterm{def:g10:vectors:collinear}{colinéaires}.
\end{itemize}
\end{remark}

\section{Coordonnées de vecteurs}

\begin{definition}[Coordonnées d'un vecteur]\label{def:g10:vectors:coords}
Dans un \omterm{def:g10:coordgeom:system}{repère}, les \emph{\omterm{def:g10:coordgeom:system}{coordonnées}} du \omterm{def:g10:vectors:vector}{vecteur} $\vect{AB}$ sont les
nombres décrivant le déplacement~:
\[
\vect{AB}\,(x_B - x_A,\ y_B - y_A).
\]
Un \omterm{def:g10:vectors:vector}{vecteur} $\vec u\,(a, b)$ déplace tout point de $a$ unités
horizontalement et $b$ unités verticalement.
\end{definition}

\begin{proposition}[Calculer avec les coordonnées]\label{prop:g10:vectors:coordrules}
Soient $\vec u\,(a, b)$ et $\vec v\,(c, d)$, et $k \in \R$.
\begin{enumerate}
  \item $\vec u = \vec v$ exactement lorsque $a = c$ et $b = d$~;
  \item $\vec u + \vec v$ a pour \omterm{def:g10:coordgeom:system}{coordonnées} $(a + c,\ b + d)$~;
  \item $k\vec u$ a pour \omterm{def:g10:coordgeom:system}{coordonnées} $(ka,\ kb)$~;
  \item $\norm{\vec u} = \sqrt{a^2 + b^2}$ (\omterm{def:g10:coordgeom:system}{repère orthonormal}).
\end{enumerate}
\end{proposition}

\begin{proof}
1--3 reformulent, coordonnée par coordonnée, ce que font les
déplacements~: par exemple effectuer $\vec u$ puis $\vec v$ déplace un
point horizontalement de $a$ puis de $c$, donc de $a + c$ au total. Le
point~4 est la formule de la distance du \cref{thm:g10:coordgeom:distance}
appliquée à un segment représentant $\vec u$.
\end{proof}

\begin{theorem}[Critère de colinéarité]\label{thm:g10:vectors:det}
Deux \omterm{def:g10:vectors:vector}{vecteurs} $\vec u\,(a, b)$ et $\vec v\,(c, d)$ sont \omterm{def:g10:vectors:collinear}{colinéaires} si et
seulement si
\[
ad - bc = 0 .
\]
\end{theorem}

\begin{proof}
Si $\vec v = k\vec u$, alors $c = ka$ et $d = kb$, donc
$ad - bc = a(kb) - b(ka) = 0$. De même si $\vec u = k \vec v$ ou si l'un
des \omterm{def:g10:vectors:vector}{vecteurs} est nul.

Réciproquement, supposons $ad - bc = 0$ avec $\vec u \neq \vec 0$,
disons $a \neq 0$ (le cas $b \neq 0$ est analogue). Poser
$k = \frac{c}{a}$~; alors $c = ka$, et $ad = bc = b(ka)$ donne, après
division par $a \neq 0$, $d = kb$. Donc $\vec v = k\vec u$.
\end{proof}

\begin{example}\label{ex:g10:vectors:det}
Les \omterm{def:g10:vectors:vector}{vecteurs} $\vec u\,(3, -2)$ et $\vec v\,(-6, 4)$ sont-ils
\omterm{def:g10:vectors:collinear}{colinéaires}~? Calculer
$ad - bc = 3 \times 4 - (-2)\times(-6) = 12 - 12 = 0$~: oui, et en effet
$\vec v = -2\vec u$. Pour $\vec u\,(3, -2)$ et $\vec w\,(1, 5)$~:
$3 \times 5 - (-2) \times 1 = 17 \neq 0$~: non \omterm{def:g10:vectors:collinear}{colinéaires}.
\end{example}

\begin{method}[Alignement et parallélisme]\label{met:g10:vectors:alignment}
Pour prouver que trois points $A$, $B$, $C$ sont alignés~:
\begin{enumerate}
  \item calculer les \omterm{def:g10:coordgeom:system}{coordonnées} de $\vect{AB}$ et $\vect{AC}$~;
  \item vérifier le critère $ad - bc = 0$ du \cref{thm:g10:vectors:det}~;
  \item conclure~: les \omterm{def:g10:vectors:vector}{vecteurs} sont \omterm{def:g10:vectors:collinear}{colinéaires} et partagent le point
        $A$, donc les trois points sont alignés.
\end{enumerate}
Le même calcul avec $\vect{AB}$ et $\vect{CD}$ prouve que les droites
$(AB)$ et $(CD)$ sont parallèles.
\end{method}

\begin{example}\label{ex:g10:vectors:alignment}
Soient $A(1, 2)$, $B(3, 5)$ et $C(7, 11)$. Alors $\vect{AB}\,(2, 3)$ et
$\vect{AC}\,(6, 9)$, et $2 \times 9 - 3 \times 6 = 18 - 18 = 0$~: les
points sont alignés (en fait $\vect{AC} = 3\vect{AB}$).
\end{example}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:g10:vectors:1}
Simplifier en utilisant la relation de Chasles~:
\[
\vect{AB} + \vect{BD}, \qquad
\vect{KL} + \vect{LM} + \vect{MK}, \qquad
\vect{AC} - \vect{BC}, \qquad
\vect{AB} + \vect{CA}.
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:vectors:2}
Soient $A(2, 1)$, $B(5, 3)$, $C(-1, 4)$. Calculer les \omterm{def:g10:coordgeom:system}{coordonnées} de
$\vect{AB}$, $\vect{BC}$, $\vect{AC}$, et vérifier sur les \omterm{def:g10:coordgeom:system}{coordonnées}
que $\vect{AB} + \vect{BC} = \vect{AC}$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:vectors:3}
Soient $\vec u\,(2, -3)$ et $\vec v\,(-1, 4)$. Donner les \omterm{def:g10:coordgeom:system}{coordonnées} de
$\vec u + \vec v$, $3\vec u$, $2\vec u - \vec v$, et calculer
$\norm{\vec u}$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:vectors:4}
Dans chaque cas, dire si $\vec u$ et $\vec v$ sont \omterm{def:g10:vectors:collinear}{colinéaires}~:
\[
\text{(a) } \vec u\,(4, 6),\ \vec v\,(6, 9);
\qquad
\text{(b) } \vec u\,(2, -5),\ \vec v\,(-4, 10);
\qquad
\text{(c) } \vec u\,(3, 1),\ \vec v\,(1, 3).
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:vectors:5}
Soient $A(0, 2)$, $B(4, 0)$, $C(6, 3)$. Trouver les \omterm{def:g10:coordgeom:system}{coordonnées} du
point $D$ tel que $ABCD$ soit un parallélogramme, c'est-à-dire tel que
$\vect{AB} = \vect{DC}$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:vectors:6}
Soient $A(-2, 1)$, $B(1, 3)$, $C(4, -1)$.
\begin{enumerate}
  \item Calculer les \omterm{def:g10:coordgeom:system}{coordonnées} du point $M$ tel que
        $\vect{AM} = \vect{AB} + \vect{AC}$.
  \item Quel est le quadrilatère $ABMC$~? Justifier.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:vectors:7}
Soient $A(1, -1)$, $B(4, 1)$, $C(10, 5)$. Les points $A$, $B$, $C$
sont-ils alignés~? Même question pour $A(0, 3)$, $B(2, 2)$, $C(8, -1)$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:vectors:8}
Soient $A(1, 2)$, $B(5, 4)$, $C(6, 1)$, $D(2, -1)$. Montrer que $(AB)$
et $(CD)$ sont parallèles, puis que $ABCD$ est un parallélogramme.
Laquelle des deux vérifications implique l'autre~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:vectors:9}
Soit $ABC$ un triangle, et soit $I$ le point défini par
$\vect{AI} = \frac23 \vect{AB}$. Avec $A(0,0)$, $B(6, 3)$, $C(2, 5)$
(pour que les calculs restent simples)~:
\begin{enumerate}
  \item calculer les \omterm{def:g10:coordgeom:system}{coordonnées} de $I$~;
  \item soit $J$ tel que $\vect{CJ} = \frac23\vect{CB}$~; calculer les
        \omterm{def:g10:coordgeom:system}{coordonnées} de $J$~;
  \item montrer que $(IJ)$ est parallèle à $(AC)$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g10:vectors:10}
Soit $ABCD$ un quadrilatère quelconque, et soient $I$, $J$, $K$, $L$
les milieux de $[AB]$, $[BC]$, $[CD]$, $[DA]$. En utilisant les
\omterm{def:g10:coordgeom:system}{coordonnées} $A(x_A, y_A)$, etc., montrer que $\vect{IJ} = \vect{LK}$,
et conclure que les milieux des côtés de \emph{n'importe quel}
quadrilatère forment un parallélogramme.
\end{exercise}
