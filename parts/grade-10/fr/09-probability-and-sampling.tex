\chapter{Probabilités et échantillonnage}\label{ch:g10:proba}

La \omterm{def:g10:proba:distribution}{probabilité} attribue des nombres au hasard. En partant d'expériences
à issues équiprobables --- dés, pièces, cartes --- ce chapitre construit
les règles de base pour calculer des \omterm{def:g10:proba:distribution}{probabilités} d'\omterm{def:g10:proba:events}{événements}, introduit
les \omterm{met:g10:proba:tree}{arbres} pour les expériences à deux étapes, et se termine par le lien
entre \omterm{def:g10:proba:distribution}{probabilités} et \omterm{def:g10:stats:series}{fréquences} observées. La théorie se poursuit dans
le \cref{ch:g11:prob}.

\section{Expériences aléatoires et événements}

\begin{definition}[Issues et événements]\label{def:g10:proba:events}
Une \emph{expérience aléatoire}\index{expérience aléatoire} est une
expérience dont le résultat ne peut être prédit (lancer un dé).
L'ensemble de tous les résultats possibles est l'\emph{univers}\index{univers}
$\Omega = \{\omega_1, \dots, \omega_n\}$~; ses éléments sont les
\emph{issues}. Un \emph{événement}\index{événement} est un ensemble
d'issues~; un événement \emph{se réalise} lorsque le résultat de
l'expérience lui appartient.
\end{definition}

\begin{definition}[Loi de probabilité]\label{def:g10:proba:distribution}
Attribuer à chaque issue $\omega_i$ un nombre $p_i \geq 0$, avec
$p_1 + \dots + p_n = 1$, définit une
\emph{probabilité}\index{probabilité}~: la probabilité $\P(A)$ d'un
\omterm{def:g10:proba:events}{événement} $A$ est la somme des $p_i$ des issues dans $A$. En particulier
$\P(\Omega) = 1$ et l'\omterm{def:g10:proba:events}{événement} impossible $\varnothing$ a pour
probabilité $0$.
\end{definition}

\begin{proposition}[Issues équiprobables]\label{prop:g10:proba:uniform}
Si les $n$ issues sont équiprobables (chaque $p_i = \frac1n$), alors pour
tout \omterm{def:g10:proba:events}{événement} $A$~:
\[
\P(A) = \frac{\text{nombre d'issues dans } A}{n}
= \frac{\text{cas favorables}}{\text{cas possibles}} .
\]
\end{proposition}

\begin{proof}
$A$ contient un certain nombre $k$ d'issues, chacune de \omterm{def:g10:proba:distribution}{probabilité}
$\frac1n$, donc $\P(A) = \frac kn$.
\end{proof}

\begin{example}\label{ex:g10:proba:die}
Lancer un dé équilibré~: $\Omega = \{1, 2, 3, 4, 5, 6\}$, chaque issue
ayant la \omterm{def:g10:proba:distribution}{probabilité} $\frac16$. L'\omterm{def:g10:proba:events}{événement} $A$~: «~le résultat est
pair~» est $A = \{2, 4, 6\}$, donc $\P(A) = \frac36 = \frac12$.
L'\omterm{def:g10:proba:events}{événement} $B$~: «~le résultat est au moins $5$~» est $B = \{5, 6\}$,
donc $\P(B) = \frac26 = \frac13$.
\end{example}

\section{Combiner des événements}

\begin{definition}[Intersection, réunion, complémentaire]\label{def:g10:proba:operations}
Soient $A$ et $B$ des \omterm{def:g10:proba:events}{événements}.
\begin{itemize}
  \item $A \cap B$ («~$A$ et $B$~») se réalise lorsque les deux se
        réalisent~;
  \item $A \cup B$ («~$A$ ou $B$~») se réalise lorsque l'un au moins se
        réalise~;
  \item le \emph{complémentaire}\index{complémentaire} $\bar A$
        («~non $A$~») se réalise exactement lorsque $A$ ne se réalise
        pas~;
  \item $A$ et $B$ sont \emph{incompatibles}\index{incompatibles} lorsqu'ils ne peuvent se
        réaliser ensemble~: $A \cap B = \varnothing$.
\end{itemize}
\end{definition}

\begin{theorem}[Règles d'addition]\label{thm:g10:proba:addition}
Pour tous \omterm{def:g10:proba:events}{événements} $A$ et $B$~:
\[
\P(A \cup B) = \P(A) + \P(B) - \P(A \cap B),
\qquad
\P(\bar A) = 1 - \P(A) .
\]
Pour des \omterm{def:g10:proba:events}{événements} \omterm{def:g10:proba:operations}{incompatibles}, la première règle se simplifie en
$\P(A \cup B) = \P(A) + \P(B)$.
\end{theorem}

\begin{proof}
Sommer $\P(A) + \P(B)$ ajoute la \omterm{def:g10:proba:distribution}{probabilité} de chaque issue de
$A \cup B$ une fois, \emph{sauf} celles de $A \cap B$, qui sont comptées
deux fois --- une fois dans $A$ et une fois dans $B$. Soustraire
$\P(A \cap B)$ répare le double comptage. Pour le \omterm{def:g10:proba:operations}{complémentaire}~: $A$ et
$\bar A$ sont \omterm{def:g10:proba:operations}{incompatibles} et $A \cup \bar A = \Omega$, donc
$\P(A) + \P(\bar A) = \P(\Omega) = 1$.
\end{proof}

la soustraction dans la règle d'addition.}
\end{omfigure}

\begin{example}\label{ex:g10:proba:cards}
Tirer une carte d'un jeu standard de $52$ cartes. Soit $A$~: «~la carte
est un cœur~» ($13$ cartes) et $B$~: «~la carte est un roi~» ($4$
cartes). Alors $A \cap B$ est «~le roi de cœur~» ($1$ carte), et
\[
\P(A \cup B) = \frac{13}{52} + \frac{4}{52} - \frac{1}{52} = \frac{16}{52}
= \frac{4}{13}.
\]
La règle du \omterm{def:g10:proba:operations}{complémentaire} est souvent la voie la plus rapide~: la
\omterm{def:g10:proba:distribution}{probabilité} que la carte ne soit \emph{pas} un cœur est
$1 - \frac{13}{52} = \frac34$.
\end{example}

\section{Arbres de probabilités}

\begin{method}[Expériences à deux étapes]\label{met:g10:proba:tree}
Pour une expérience réalisée en deux étapes~:
\begin{enumerate}
  \item dessiner un \emph{arbre}\index{arbre de probabilités}~: une
        branche par résultat possible de la première étape, puis depuis
        chacune, une branche par résultat de la seconde étape, en
        inscrivant la \omterm{def:g10:proba:distribution}{probabilité} sur chaque branche (les branches
        sortant d'un nœud doivent sommer à $1$)~;
  \item la \omterm{def:g10:proba:distribution}{probabilité} d'un \emph{chemin} (une feuille de l'\omterm{met:g10:proba:tree}{arbre}) est le
        produit des \omterm{def:g10:proba:distribution}{probabilités} le long de ses branches~;
  \item la \omterm{def:g10:proba:distribution}{probabilité} d'un \omterm{def:g10:proba:events}{événement} est la somme des \omterm{def:g10:proba:distribution}{probabilités} des
        chemins qui le réalisent.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g10:proba:tree}
Une urne contient $3$ boules rouges et $2$ boules bleues. Tirer une
boule, la remettre, tirer à nouveau. Chaque tirage donne rouge (R) avec
\omterm{def:g10:proba:distribution}{probabilité} $\frac35$ et bleu (B) avec \omterm{def:g10:proba:distribution}{probabilité} $\frac25$.

    {RR:\ $\frac35 \times \frac35 = \frac{9}{25}$};

{\small L'\omterm{met:g10:proba:tree}{arbre} de deux tirages avec remise. Les quatre \omterm{def:g10:proba:distribution}{probabilités} de
chemin somment à $1$.}
\end{omfigure}

La \omterm{def:g10:proba:distribution}{probabilité} d'obtenir deux boules de la même couleur est
$\P(\text{RR}) + \P(\text{BB}) = \frac{9}{25} + \frac{4}{25} =
\frac{13}{25}$~; la \omterm{def:g10:proba:distribution}{probabilité} d'au moins une bleue est
$1 - \P(\text{RR}) = 1 - \frac{9}{25} = \frac{16}{25}$ (règle du
\omterm{def:g10:proba:operations}{complémentaire}).
\end{example}

\begin{remark}
Sans remise, les \omterm{def:g10:proba:distribution}{probabilités} du second tirage dépendent du premier~:
avec l'urne ci-dessus, après avoir tiré une boule rouge, il reste $2$
rouges et $2$ bleues, donc la seconde branche porte $\frac24$ et
$\frac24$. La méthode de l'\omterm{met:g10:proba:tree}{arbre} fonctionne sans changement~; seuls les
nombres sur les branches du second niveau changent.
\end{remark}

\section{Fréquences et échantillonnage}

Lorsqu'une \omterm{def:g10:proba:events}{expérience aléatoire} est répétée de nombreuses fois, la
\omterm{def:g10:stats:series}{fréquence} observée d'un \omterm{def:g10:proba:events}{événement} se rapproche de sa \omterm{def:g10:proba:distribution}{probabilité} --- c'est
pourquoi la \omterm{def:g10:proba:distribution}{probabilité} est utile pour décrire le monde réel.

\begin{definition}[Échantillon]\label{def:g10:proba:sample}
Un \emph{échantillon}\index{échantillon} de taille $n$ est le résultat de
la répétition de la même expérience $n$ fois indépendamment. La
\emph{\omterm{def:g10:stats:series}{fréquence} observée} d'un \omterm{def:g10:proba:events}{événement} dans l'échantillon est le nombre
d'occurrences divisé par $n$.
\end{definition}

\begin{example}\label{ex:g10:proba:sampling}
Une pièce équilibrée donne pile avec \omterm{def:g10:proba:distribution}{probabilité} $0.5$. La lancer $100$
fois donne rarement exactement $50$ piles~: les \omterm{def:g10:proba:sample}{échantillons}
\emph{fluctuent}. Des séries typiques de $100$ lancers donnent des
\omterm{def:g10:stats:series}{fréquences} comme $0.46$, $0.53$, $0.49$ --- proches de $0.5$, mais non
égales. Les grands \omterm{def:g10:proba:sample}{échantillons} fluctuent moins~: avec $10\,000$
lancers, les \omterm{def:g10:stats:series}{fréquences} observées sont habituellement à moins de $0.01$
de $0.5$.
\end{example}

\begin{remark}[Intervalle de fluctuation]
Une règle pratique utile, justifiée dans le \cref{ch:g11:binom}~: pour un
\omterm{def:g10:proba:sample}{échantillon} de taille $n$ d'un \omterm{def:g10:proba:events}{événement} de \omterm{def:g10:proba:distribution}{probabilité} $p$ (avec $n$
raisonnablement grand), la \omterm{def:g10:stats:series}{fréquence} observée se trouve dans l'\omterm{def:g10:numbers:interval}{intervalle}
\[
\intcc{p - \frac{1}{\sqrt n}}{\,p + \frac{1}{\sqrt n}}
\]
dans environ $95\,\%$ des \omterm{def:g10:proba:sample}{échantillons}. Si une \omterm{def:g10:stats:series}{fréquence} observée tombe
loin en dehors de cet \omterm{def:g10:numbers:interval}{intervalle}, douter de la valeur de $p$~: c'est le
point de départ des tests statistiques.
\end{remark}

\begin{example}\label{ex:g10:proba:test}
Une usine prétend que seulement $10\,\%$ de ses pièces sont défectueuses
($p = 0.1$). Dans un \omterm{def:g10:proba:sample}{échantillon} de $n = 400$ pièces, $64$ sont
défectueuses~: \omterm{def:g10:stats:series}{fréquence} observée $\frac{64}{400} = 0.16$. L'\omterm{def:g10:numbers:interval}{intervalle}
de fluctuation est
$\intcc{0.1 - \frac{1}{20}}{0.1 + \frac{1}{20}} = \intcc{0.05}{0.15}$,
et $0.16$ tombe en dehors~: l'\omterm{def:g10:proba:sample}{échantillon} jette un doute sérieux sur
l'affirmation.
\end{example}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:g10:proba:1}
Un dé équilibré est lancé. Calculer la \omterm{def:g10:proba:distribution}{probabilité} de chaque \omterm{def:g10:proba:events}{événement}~:
«~obtenir un $6$~»~; «~obtenir un nombre impair~»~; «~obtenir au plus
$4$~»~; «~obtenir un $7$~».
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:proba:2}
Un sac contient $5$ billes vertes, $3$ jaunes et $2$ noires~; une bille
est tirée au hasard. Calculer la \omterm{def:g10:proba:distribution}{probabilité} qu'elle soit verte~; qu'elle
ne soit pas noire~; qu'elle soit verte ou jaune.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:proba:3}
Dans une classe de $30$ élèves, $18$ étudient l'espagnol, $10$
l'allemand, et $4$ les deux. Un élève est choisi au hasard. En utilisant
la règle d'addition, calculer la \omterm{def:g10:proba:distribution}{probabilité} que cet élève étudie au
moins l'une des deux langues, puis la \omterm{def:g10:proba:distribution}{probabilité} qu'il n'étudie ni
l'une ni l'autre.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:proba:4}
Une pièce équilibrée est lancée deux fois. Dessiner l'\omterm{met:g10:proba:tree}{arbre} de
l'expérience et calculer la \omterm{def:g10:proba:distribution}{probabilité} d'obtenir deux faces~; exactement
une face~; au moins une face.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:proba:5}
Une urne contient $4$ boules rouges et $6$ blanches. Deux boules sont
tirées l'une après l'autre \emph{sans} remise. Dessiner l'\omterm{met:g10:proba:tree}{arbre} avec les
bonnes \omterm{def:g10:proba:distribution}{probabilités} au second niveau, et calculer la \omterm{def:g10:proba:distribution}{probabilité} de tirer
deux boules rouges, puis de tirer deux boules de couleurs différentes.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:proba:6}
Deux dés équilibrés sont lancés et leur somme est enregistrée.
\begin{enumerate}
  \item Expliquer pourquoi l'\omterm{def:g10:proba:events}{univers} peut être pris comme les $36$
        couples équiprobables $(1,1), (1,2), \dots, (6,6)$.
  \item Calculer la \omterm{def:g10:proba:distribution}{probabilité} que la somme soit $7$~; qu'elle soit
        $12$~; qu'elle soit au plus $4$.
  \item Quelle somme est la plus probable~?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:proba:7}
Une question à choix multiples propose $4$ réponses, une seule correcte.
Un élève répond à $2$ telles questions indépendantes complètement au
hasard. Calculer la \omterm{def:g10:proba:distribution}{probabilité} de répondre correctement aux deux, à
exactement une, à aucune. Vérifier que les trois \omterm{def:g10:proba:distribution}{probabilités} somment à
$1$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:proba:8}
Un politicien prétend avoir $50\,\%$ d'approbation. Dans un sondage de
$n = 900$ électeurs choisis au hasard, $396$ approuvent.
\begin{enumerate}
  \item Calculer la \omterm{def:g10:stats:series}{fréquence} observée et l'\omterm{def:g10:numbers:interval}{intervalle} de fluctuation
        autour de $p = 0.5$ pour $n = 900$.
  \item Le sondage jette-t-il un doute sur l'affirmation~?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g10:proba:9}
Un jeu coûte $2$ à jouer~: on lance un dé équilibré et on gagne $6$ si
l'on obtient un $6$, $3$ si l'on obtient un $5$, rien sinon.
\begin{enumerate}
  \item Calculer la \omterm{def:g10:proba:distribution}{probabilité} de chaque montant gagné ($6$, $3$, $0$).
  \item Sur $600$ parties, environ combien de chaque issue attend-on~?
        Calculer le gain et la perte totaux attendus~: le jeu vaut-il la
        peine d'être joué~?
\end{enumerate}
\end{exercise}
