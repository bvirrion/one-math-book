\chapter{Fonctions}\label{ch:g10:functions}

Une \omterm{def:g10:functions:function}{fonction} est une machine qui prend un nombre en entrĂŠe et produit un
nombre en sortie, toujours la mĂŞme sortie pour la mĂŞme entrĂŠe. Les
\omterm{def:g10:functions:function}{fonctions} sont les objets centraux de l'analyse que l'on construira au
cours des prochaines annĂŠes (\cref{ch:g11:quad} et suivants)~; ce chapitre
met en place le vocabulaire --- \omterm{def:g10:functions:function}{ensemble de dĂŠfinition}, \omterm{def:g10:functions:function}{image}, \omterm{def:g10:functions:graph}{graphique},
variations --- et entraĂŽne la compĂŠtence essentielle de lire des
informations sur un \omterm{def:g10:functions:graph}{graphique}.

\section{Vocabulaire~: images et antĂŠcĂŠdents}

\begin{definition}[Fonction]\label{def:g10:functions:function}
Soit $D$ un ensemble de \omterm{def:g10:numbers:sets}{nombres rĂŠels}. Une \emph{fonction}\index{fonction}
$f$ dĂŠfinie sur $D$ associe Ă  chaque nombre $x \in D$ exactement un
\omterm{def:g10:numbers:sets}{nombre rĂŠel}, notĂŠ $f(x)$ et appelĂŠ l'\emph{image}\index{image} de $x$.
L'ensemble $D$ est l'\emph{ensemble de dĂŠfinition}\index{ensemble de
dĂŠfinition} de $f$. On ĂŠcrit
\[
f : x \longmapsto f(x).
\]
Si $f(x) = y$, alors $x$ est un \emph{antĂŠcĂŠdent}\index{antĂŠcĂŠdent} de
$y$.
\end{definition}

\begin{remark}
Chaque $x$ dans l'\omterm{def:g10:functions:function}{ensemble de dĂŠfinition} a exactement une \omterm{def:g10:functions:function}{image}, mais un
nombre $y$ peut avoir plusieurs \omterm{def:g10:functions:function}{antĂŠcĂŠdents}, ou aucun. Pour
$f(x) = x^2$~: l'\omterm{def:g10:functions:function}{image} de $3$ est $9$, et $9$ a deux \omterm{def:g10:functions:function}{antĂŠcĂŠdents}, $3$ et
$-3$~; le nombre $-4$ n'a pas d'\omterm{def:g10:functions:function}{antĂŠcĂŠdent}.
\end{remark}

\begin{example}[Trouver un ensemble de dĂŠfinition]\label{ex:g10:functions:domain}
Lorsqu'une \omterm{def:g10:functions:function}{fonction} est donnĂŠe par une formule, son \omterm{def:g10:functions:function}{ensemble de
dĂŠfinition} est l'ensemble des $x$ pour lesquels la formule a un sens.
\begin{itemize}
  \item $f(x) = 3x^2 - 5x + 1$ a un sens pour tout $x$~: l'\omterm{def:g10:functions:function}{ensemble de
        dĂŠfinition} est $\R$.
  \item $g(x) = \dfrac{1}{x - 2}$ exige $x - 2 \neq 0$~: l'\omterm{def:g10:functions:function}{ensemble de
        dĂŠfinition} est tous les rĂŠels sauf $2$.
  \item $h(x) = \sqrt{x - 3}$ exige $x - 3 \geq 0$~: l'\omterm{def:g10:functions:function}{ensemble de
        dĂŠfinition} est $\intco{3}{+\infty}$.
\end{itemize}
\end{example}

\begin{example}[Calculer images et antĂŠcĂŠdents]\label{ex:g10:functions:images}
Soit $f(x) = x^2 - 4x + 3$, dĂŠfinie sur $\R$.

\emph{\omterm{def:g10:functions:function}{Image} de $5$~:} substituer $x = 5$~:
$f(5) = 25 - 20 + 3 = 8$.

\emph{\omterm{def:g10:functions:function}{AntĂŠcĂŠdents} de $3$~:} rĂŠsoudre $f(x) = 3$~:
\[
x^2 - 4x + 3 = 3
\quad\Longleftrightarrow\quad
x^2 - 4x = 0
\quad\Longleftrightarrow\quad
x(x - 4) = 0,
\]
donc les \omterm{def:g10:functions:function}{antĂŠcĂŠdents} de $3$ sont $0$ et $4$.
\end{example}

\section{Le graphique d'une fonction}

\begin{definition}[Graphique]\label{def:g10:functions:graph}
Dans un repĂ¨re, le \emph{graphique}\index{graphique} (ou la
\emph{courbe}) de $f$ est l'ensemble de tous les points $(x, f(x))$ pour
$x$ dans l'\omterm{def:g10:functions:function}{ensemble de dĂŠfinition}. Autrement dit, un point $(x, y)$
appartient au graphique exactement lorsque $y = f(x)$.
\end{definition}

\begin{method}[Lire un graphique]\label{met:g10:functions:reading}
Sur le \omterm{def:g10:functions:graph}{graphique} de $f$~:
\begin{enumerate}
  \item \emph{\omterm{def:g10:functions:function}{image} de $a$~:} monter verticalement depuis $x = a$ sur
        l'axe horizontal jusqu'Ă  la courbe, puis horizontalement jusqu'Ă 
        l'axe vertical~; la valeur lue est $f(a)$~;
  \item \emph{\omterm{def:g10:functions:function}{antĂŠcĂŠdents} de $b$~:} tracer la droite horizontale $y = b$~;
        les \omterm{def:g10:functions:function}{antĂŠcĂŠdents} sont les abscisses de tous ses points
        d'\omterm{def:g10:numbers:interunion}{intersection} avec la courbe~;
  \item \emph{solutions de $f(x) = k$~:} mĂŞme chose que les \omterm{def:g10:functions:function}{antĂŠcĂŠdents} de
        $k$~;
  \item \emph{solutions de $f(x) \leq k$~:} les $x$ pour lesquels la
        courbe est sur ou sous la droite $y = k$.
\end{enumerate}
\end{method}

\end{axis}
\end{tikzpicture}

{\small RĂŠsoudre $f(x) = k$ graphiquement~: les solutions $x_1$ et $x_2$
sont les abscisses des points d'\omterm{def:g10:numbers:interunion}{intersection} de la courbe avec la droite
horizontale $y = k$. Ici $f(x) \leq k$ est rĂŠalisĂŠ sur
$\intcc{x_1}{x_2}$, oĂš la courbe est sous la droite.}
\end{omfigure}

\begin{example}\label{ex:g10:functions:membership}
Le point $A(2, 5)$ appartient-il au \omterm{def:g10:functions:graph}{graphique} de $f(x) = x^2 + 1$~?
Calculer $f(2) = 4 + 1 = 5$~: oui, puisque $f(2)$ ĂŠgale l'ordonnĂŠe de
$A$. Le point $B(3, 8)$ n'est pas sur le \omterm{def:g10:functions:graph}{graphique}, car
$f(3) = 10 \neq 8$.
\end{example}

\section{Variations d'une fonction}

\begin{definition}[Croissante, dĂŠcroissante]\label{def:g10:functions:variations}
Soit $f$ dĂŠfinie sur un \omterm{def:g10:numbers:interval}{intervalle} $I$.
\begin{itemize}
  \item $f$ est \emph{croissante}\index{fonction croissante} sur $I$
        lorsque pour tous $u, v \in I$, si $u < v$ alors
        $f(u) \leq f(v)$~: les sorties croissent avec les entrĂŠes, et le
        \omterm{def:g10:functions:graph}{graphique} monte de gauche Ă  droite.
  \item $f$ est \emph{dĂŠcroissante}\index{fonction dĂŠcroissante} sur $I$
        lorsque $u < v$ implique $f(u) \geq f(v)$~: le \omterm{def:g10:functions:graph}{graphique} descend
        de gauche Ă  droite.
\end{itemize}
Avec des inĂŠgalitĂŠs strictes $f(u) < f(v)$ (resp.\ $>$) on dit
\emph{strictement} croissante (resp.\ dĂŠcroissante).
\end{definition}

\begin{definition}[Tableau de variations]\label{def:g10:functions:table}
Un \emph{tableau de variations}\index{tableau de variations} rĂŠsume sur
quels \omterm{def:g10:numbers:interval}{intervalles} $f$ croĂŽt et dĂŠcroĂŽt, avec des flĂ¨ches $\nearrow$ et
$\searrow$, et enregistre les valeurs de $f$ aux points de retournement.
\end{definition}

ConsidĂŠrons la \omterm{def:g10:functions:function}{fonction} grapquĂŠe ci-dessous sur $\intcc{-2}{4}$.

\begin{example}[DĂŠmontrer une variation]\label{ex:g10:functions:proof}
Montrer que $f(x) = 3x + 1$ est strictement \omterm{def:g10:functions:variations}{croissante} sur $\R$.
Prendre $u < v$ quelconques et comparer les \omterm{def:g10:functions:function}{images}~:
\[
f(v) - f(u) = (3v + 1) - (3u + 1) = 3(v - u) > 0,
\]
car $v - u > 0$. Donc $f(u) < f(v)$~: la \omterm{def:g10:functions:function}{fonction} est strictement
\omterm{def:g10:functions:variations}{croissante}. Le mĂŞme calcul avec un coefficient nĂŠgatif, par exemple
$g(x) = -2x + 5$, donne $g(v) - g(u) = -2(v-u) < 0$~: strictement
\omterm{def:g10:functions:variations}{dĂŠcroissante}.
\end{example}

\section{Extrema}

\begin{definition}[Maximum, minimum]\label{def:g10:functions:extrema}
Soit $f$ dĂŠfinie sur $D$ et $a \in D$. La valeur $f(a)$ est le
\emph{maximum}\index{maximum} de $f$ sur $D$ lorsque $f(x) \leq f(a)$
pour tout $x \in D$~; c'est le \emph{minimum}\index{minimum} lorsque
$f(x) \geq f(a)$ pour tout $x \in D$. Sur un \omterm{def:g10:functions:graph}{graphique}, ce sont les points
les plus hauts et les plus bas de la courbe.
\end{definition}

\begin{example}\label{ex:g10:functions:extremum}
Montrer que $f(x) = (x - 1)^2 + 2$ a pour \omterm{def:g10:functions:extrema}{minimum} $2$ sur $\R$, atteint
en $x = 1$. Pour tout $x$, le carrĂŠ $(x-1)^2$ est $\geq 0$, donc
$f(x) = (x-1)^2 + 2 \geq 2$~; et $f(1) = 0 + 2 = 2$, donc la valeur $2$ est
effectivement atteinte. Ces deux faits ensemble prouvent que $2$ est le
\omterm{def:g10:functions:extrema}{minimum}.
\end{example}

$2$.}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:g10:functions:1}
Soit $f(x) = 2x^2 - 3x + 1$. Calculer les \omterm{def:g10:functions:function}{images} de $0$, $2$, $-1$ et
$\frac12$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:functions:2}
Donner l'\omterm{def:g10:functions:function}{ensemble de dĂŠfinition} de chaque \omterm{def:g10:functions:function}{fonction}~:
\[
f(x) = 5x - 2, \qquad
g(x) = \frac{3}{x + 4}, \qquad
h(x) = \sqrt{2x - 6}, \qquad
k(x) = \frac{1}{x^2 + 1}.
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:functions:3}
Soit $f(x) = x^2 - 2x$. Trouver tous les \omterm{def:g10:functions:function}{antĂŠcĂŠdents} de $0$, de $3$, et
de $-1$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:functions:4}
Le point $A(-1, 4)$ appartient-il au \omterm{def:g10:functions:graph}{graphique} de
$f(x) = x^2 - 2x + 1$~? Et le point $B(2, 1)$~? Justifier par un calcul.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:functions:5}
Une \omterm{def:g10:functions:function}{fonction} $g$ dĂŠfinie sur $\intcc{-3}{5}$ a le \omterm{def:g10:functions:table}{tableau de variations}
\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{c|ccccccc}
$x$ & $-3$ & & $0$ & & $3$ & & $5$ \\
\hline
$g$ & $1$ & $\searrow$ & $-2$ & $\nearrow$ & $4$ & $\searrow$ & $0$ \\
\end{tabular}
\end{center}
\begin{enumerate}
  \item Quels sont le \omterm{def:g10:functions:extrema}{maximum} et le \omterm{def:g10:functions:extrema}{minimum} de $g$ sur $\intcc{-3}{5}$~?
  \item Comparer $g(-1)$ et $g(-0.5)$ sans connaĂŽtre leurs valeurs.
  \item Combien de solutions l'\omterm{def:g10:algebra:equation}{ĂŠquation} $g(x) = 0$ a-t-elle au plus sur
        chaque \omterm{def:g10:numbers:interval}{intervalle} du tableau~?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:functions:6}
Soit $f(x) = -2x + 7$. Montrer, en comparant $f(u)$ et $f(v)$ pour
$u < v$, que $f$ est strictement \omterm{def:g10:functions:variations}{dĂŠcroissante} sur $\R$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:functions:7}
Montrer que la \omterm{def:g10:functions:function}{fonction} $f(x) = x^2 + 6x + 11$ a un \omterm{def:g10:functions:extrema}{minimum} sur $\R$, et
donner sa valeur et oĂš il est atteint. (Indication~: ĂŠcrire
$f(x) = (x + 3)^2 + c$ pour la bonne constante $c$.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:functions:8}
Un jardin rectangulaire a un pĂŠrimĂ¨tre de $40$~m. Soit $x$ sa largeur, en
mĂ¨tres.
\begin{enumerate}
  \item Exprimer la longueur, puis l'aire $A(x)$, en \omterm{def:g10:functions:function}{fonction} de $x$. Pour
        quels $x$ cela a-t-il un sens~?
  \item Calculer $A(4)$, $A(8)$, $A(10)$, $A(12)$ et $A(16)$. Que
        conjecturer sur la forme d'aire maximale~?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:functions:9}
Soit $f(x) = \dfrac{1}{x^2 + 1}$, dĂŠfinie sur $\R$.
\begin{enumerate}
  \item Montrer que $0 < f(x) \leq 1$ pour tout rĂŠel $x$.
  \item $f$ a-t-elle un \omterm{def:g10:functions:extrema}{maximum} sur $\R$~? Un \omterm{def:g10:functions:extrema}{minimum}~? Justifier.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g10:functions:10}
Soit $f(x) = x^2$ sur $\R$.
\begin{enumerate}
  \item Soit $0 \leq u < v$. Factoriser $f(v) - f(u)$ et en dĂŠduire que
        $f$ est strictement \omterm{def:g10:functions:variations}{croissante} sur $\intco{0}{+\infty}$.
  \item Adapter l'argument pour montrer que $f$ est strictement
        \omterm{def:g10:functions:variations}{dĂŠcroissante} sur $\intoc{-\infty}{0}$.
\end{enumerate}
\end{exercise}
