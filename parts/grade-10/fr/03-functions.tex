\chapter{Fonctions}\label{ch:g10:functions}

Une fonction est une machine qui prend un nombre en entrée et produit un
nombre en sortie, toujours la même sortie pour la même entrée. Les
fonctions sont les objets centraux de l'analyse que l'on construira au
cours des prochaines années (\cref{ch:g11:quad} et suivants)~; ce chapitre
met en place le vocabulaire --- ensemble de définition, image, graphique,
variations --- et entraîne la compétence essentielle de lire des
informations sur un graphique.

\section{Vocabulaire~: images et antécédents}

\begin{definition}[Fonction]\label{def:g10:functions:function}
Soit $D$ un ensemble de nombres réels. Une \emph{fonction}\index{fonction}
$f$ définie sur $D$ associe à chaque nombre $x \in D$ exactement un
nombre réel, noté $f(x)$ et appelé l'\emph{image}\index{image} de $x$.
L'ensemble $D$ est l'\emph{ensemble de définition}\index{ensemble de
définition} de $f$. On écrit
\[
f : x \longmapsto f(x).
\]
Si $f(x) = y$, alors $x$ est un \emph{antécédent}\index{antécédent} de
$y$.
\end{definition}

\begin{remark}
Chaque $x$ dans l'ensemble de définition a exactement une image, mais un
nombre $y$ peut avoir plusieurs antécédents, ou aucun. Pour
$f(x) = x^2$~: l'image de $3$ est $9$, et $9$ a deux antécédents, $3$ et
$-3$~; le nombre $-4$ n'a pas d'antécédent.
\end{remark}

\begin{example}[Trouver un ensemble de définition]\label{ex:g10:functions:domain}
Lorsqu'une fonction est donnée par une formule, son ensemble de
définition est l'ensemble des $x$ pour lesquels la formule a un sens.
\begin{itemize}
  \item $f(x) = 3x^2 - 5x + 1$ a un sens pour tout $x$~: l'ensemble de
        définition est $\R$.
  \item $g(x) = \dfrac{1}{x - 2}$ exige $x - 2 \neq 0$~: l'ensemble de
        définition est tous les réels sauf $2$.
  \item $h(x) = \sqrt{x - 3}$ exige $x - 3 \geq 0$~: l'ensemble de
        définition est $\intco{3}{+\infty}$.
\end{itemize}
\end{example}

\begin{example}[Calculer images et antécédents]\label{ex:g10:functions:images}
Soit $f(x) = x^2 - 4x + 3$, définie sur $\R$.

\emph{Image de $5$~:} substituer $x = 5$~:
$f(5) = 25 - 20 + 3 = 8$.

\emph{Antécédents de $3$~:} résoudre $f(x) = 3$~:
\[
x^2 - 4x + 3 = 3
\quad\Longleftrightarrow\quad
x^2 - 4x = 0
\quad\Longleftrightarrow\quad
x(x - 4) = 0,
\]
donc les antécédents de $3$ sont $0$ et $4$.
\end{example}

\section{Le graphique d'une fonction}

\begin{definition}[Graphique]\label{def:g10:functions:graph}
Dans un repère, le \emph{graphique}\index{graphique} (ou la
\emph{courbe}) de $f$ est l'ensemble de tous les points $(x, f(x))$ pour
$x$ dans l'ensemble de définition. Autrement dit, un point $(x, y)$
appartient au graphique exactement lorsque $y = f(x)$.
\end{definition}

  \addplot[omDef, thick, domain=-2.2:4.2, samples=120]
    {0.25*x^3 - 0.75*x^2 - x + 2};
  \draw[dashed, omThm] (axis cs:3,0) -- (axis cs:3,1.25)
    -- (axis cs:0,1.25);
  \fill[omThm] (axis cs:3,1.25) circle (1.5pt);
  \node[omThm, font=\small, above right] at (axis cs:3,1.2)
    {$(3,\,f(3))$};
\end{axis}
\end{tikzpicture}

{\small Lire une image sur un graphique~: partir de $x = 3$ sur l'axe
horizontal, monter verticalement jusqu'à la courbe, puis horizontalement
jusqu'à l'axe vertical pour lire $f(3) = 1.25$.}
\end{omfigure}

\begin{method}[Lire un graphique]\label{met:g10:functions:reading}
Sur le graphique de $f$~:
\begin{enumerate}
  \item \emph{image de $a$~:} monter verticalement depuis $x = a$ sur
        l'axe horizontal jusqu'à la courbe, puis horizontalement jusqu'à
        l'axe vertical~; la valeur lue est $f(a)$~;
  \item \emph{antécédents de $b$~:} tracer la droite horizontale $y = b$~;
        les antécédents sont les abscisses de tous ses points
        d'intersection avec la courbe~;
  \item \emph{solutions de $f(x) = k$~:} même chose que les antécédents de
        $k$~;
  \item \emph{solutions de $f(x) \leq k$~:} les $x$ pour lesquels la
        courbe est sur ou sous la droite $y = k$.
\end{enumerate}
\end{method}

  \addplot[omThm, thick, domain=-3.2:3.2] {2};
  \fill (axis cs:-2,2) circle (1.5pt);
  \fill (axis cs:2,2) circle (1.5pt);
  \draw[dashed] (axis cs:-2,2) -- (axis cs:-2,0);
  \draw[dashed] (axis cs:2,2) -- (axis cs:2,0);
\end{axis}
\end{tikzpicture}

{\small Résoudre $f(x) = k$ graphiquement~: les solutions $x_1$ et $x_2$
sont les abscisses des points d'intersection de la courbe avec la droite
horizontale $y = k$. Ici $f(x) \leq k$ est réalisé sur
$\intcc{x_1}{x_2}$, où la courbe est sous la droite.}
\end{omfigure}

\begin{example}\label{ex:g10:functions:membership}
Le point $A(2, 5)$ appartient-il au graphique de $f(x) = x^2 + 1$~?
Calculer $f(2) = 4 + 1 = 5$~: oui, puisque $f(2)$ égale l'ordonnée de
$A$. Le point $B(3, 8)$ n'est pas sur le graphique, car
$f(3) = 10 \neq 8$.
\end{example}

\section{Variations d'une fonction}

\begin{definition}[Croissante, décroissante]\label{def:g10:functions:variations}
Soit $f$ définie sur un intervalle $I$.
\begin{itemize}
  \item $f$ est \emph{croissante}\index{fonction croissante} sur $I$
        lorsque pour tous $u, v \in I$, si $u < v$ alors
        $f(u) \leq f(v)$~: les sorties croissent avec les entrées, et le
        graphique monte de gauche à droite.
  \item $f$ est \emph{décroissante}\index{fonction décroissante} sur $I$
        lorsque $u < v$ implique $f(u) \geq f(v)$~: le graphique descend
        de gauche à droite.
\end{itemize}
Avec des inégalités strictes $f(u) < f(v)$ (resp.\ $>$) on dit
\emph{strictement} croissante (resp.\ décroissante).
\end{definition}

\begin{definition}[Tableau de variations]\label{def:g10:functions:table}
Un \emph{tableau de variations}\index{tableau de variations} résume sur
quels intervalles $f$ croît et décroît, avec des flèches $\nearrow$ et
$\searrow$, et enregistre les valeurs de $f$ aux points de retournement.
\end{definition}

\begin{example}\label{ex:g10:functions:table}
Considérons la fonction grapquée ci-dessous sur $\intcc{-2}{4}$.

\begin{omfigure}
\begin{tikzpicture}
\begin{axis}[omaxis,
  width=8.6cm, height=5.4cm,
  xmin=-2.8, xmax=4.6, ymin=-2.4, ymax=3.8,
  xtick={-2,1,4}, ytick={-1.5,3}, yticklabels={$-1.5$,$3$}]
  \addplot[omDef, thick, domain=-2:4, samples=120] {-0.5*(x-1)^2 + 3};
  \fill (axis cs:-2,-1.5) circle (1.5pt);
  \fill (axis cs:4,-1.5) circle (1.5pt);
  \fill (axis cs:1,3) circle (1.5pt);
\end{axis}
\end{tikzpicture}

{\small Une fonction définie sur $\intcc{-2}{4}$, croissante puis
décroissante.}
\end{omfigure}

Lecture du graphique~: $f$ croît de $f(-2) = -1.5$ jusqu'à son maximum
$f(1) = 3$, puis décroît jusqu'à $f(4) = -1.5$. Son tableau de variations
est
\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{c|ccccc}
$x$ & $-2$ & & $1$ & & $4$ \\
\hline
$f$ & $-1.5$ & $\nearrow$ & $3$ & $\searrow$ & $-1.5$ \\
\end{tabular}
\end{center}
\end{example}

\begin{example}[Démontrer une variation]\label{ex:g10:functions:proof}
Montrer que $f(x) = 3x + 1$ est strictement croissante sur $\R$.
Prendre $u < v$ quelconques et comparer les images~:
\[
f(v) - f(u) = (3v + 1) - (3u + 1) = 3(v - u) > 0,
\]
car $v - u > 0$. Donc $f(u) < f(v)$~: la fonction est strictement
croissante. Le même calcul avec un coefficient négatif, par exemple
$g(x) = -2x + 5$, donne $g(v) - g(u) = -2(v-u) < 0$~: strictement
décroissante.
\end{example}

\section{Extrema}

\begin{definition}[Maximum, minimum]\label{def:g10:functions:extrema}
Soit $f$ définie sur $D$ et $a \in D$. La valeur $f(a)$ est le
\emph{maximum}\index{maximum} de $f$ sur $D$ lorsque $f(x) \leq f(a)$
pour tout $x \in D$~; c'est le \emph{minimum}\index{minimum} lorsque
$f(x) \geq f(a)$ pour tout $x \in D$. Sur un graphique, ce sont les points
les plus hauts et les plus bas de la courbe.
\end{definition}

\begin{example}\label{ex:g10:functions:extremum}
Montrer que $f(x) = (x - 1)^2 + 2$ a pour minimum $2$ sur $\R$, atteint
en $x = 1$. Pour tout $x$, le carré $(x-1)^2$ est $\geq 0$, donc
$f(x) = (x-1)^2 + 2 \geq 2$~; et $f(1) = 0 + 2 = 2$, donc la valeur $2$ est
effectivement atteinte. Ces deux faits ensemble prouvent que $2$ est le
minimum.
\end{example}

\begin{omfigure}
\begin{tikzpicture}
\begin{axis}[omaxis,
  width=7.6cm, height=5.2cm,
  xmin=-1.8, xmax=3.8, ymin=-0.9, ymax=6.4,
  xtick={1}, ytick={2}]
  \addplot[omDef, thick, domain=-1.0:3.0, samples=100] {(x-1)^2 + 2};

  \fill (axis cs:1,2) circle (1.5pt);
  \node[below right, font=\small] at (axis cs:1,2) {$(1,2)$};
\end{axis}
\end{tikzpicture}

{\small La courbe de $f(x) = (x-1)^2 + 2$ ne descend jamais sous la droite
en pointillés $y = 2$, et la touche en $x = 1$~: le minimum de $f$ est
$2$.}
\end{omfigure}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:g10:functions:1}
Soit $f(x) = 2x^2 - 3x + 1$. Calculer les images de $0$, $2$, $-1$ et
$\frac12$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:functions:2}
Donner l'ensemble de définition de chaque fonction~:
\[
f(x) = 5x - 2, \qquad
g(x) = \frac{3}{x + 4}, \qquad
h(x) = \sqrt{2x - 6}, \qquad
k(x) = \frac{1}{x^2 + 1}.
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:functions:3}
Soit $f(x) = x^2 - 2x$. Trouver tous les antécédents de $0$, de $3$, et
de $-1$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:functions:4}
Le point $A(-1, 4)$ appartient-il au graphique de
$f(x) = x^2 - 2x + 1$~? Et le point $B(2, 1)$~? Justifier par un calcul.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:functions:5}
Une fonction $g$ définie sur $\intcc{-3}{5}$ a le tableau de variations
\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{c|ccccccc}
$x$ & $-3$ & & $0$ & & $3$ & & $5$ \\
\hline
$g$ & $1$ & $\searrow$ & $-2$ & $\nearrow$ & $4$ & $\searrow$ & $0$ \\
\end{tabular}
\end{center}
\begin{enumerate}
  \item Quels sont le maximum et le minimum de $g$ sur $\intcc{-3}{5}$~?
  \item Comparer $g(-1)$ et $g(-0.5)$ sans connaître leurs valeurs.
  \item Combien de solutions l'équation $g(x) = 0$ a-t-elle au plus sur
        chaque intervalle du tableau~?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:functions:6}
Soit $f(x) = -2x + 7$. Montrer, en comparant $f(u)$ et $f(v)$ pour
$u < v$, que $f$ est strictement décroissante sur $\R$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:functions:7}
Montrer que la fonction $f(x) = x^2 + 6x + 11$ a un minimum sur $\R$, et
donner sa valeur et où il est atteint. (Indication~: écrire
$f(x) = (x + 3)^2 + c$ pour la bonne constante $c$.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:functions:8}
Un jardin rectangulaire a un périmètre de $40$~m. Soit $x$ sa largeur, en
mètres.
\begin{enumerate}
  \item Exprimer la longueur, puis l'aire $A(x)$, en fonction de $x$. Pour
        quels $x$ cela a-t-il un sens~?
  \item Calculer $A(4)$, $A(8)$, $A(10)$, $A(12)$ et $A(16)$. Que
        conjecturer sur la forme d'aire maximale~?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:functions:9}
Soit $f(x) = \dfrac{1}{x^2 + 1}$, définie sur $\R$.
\begin{enumerate}
  \item Montrer que $0 < f(x) \leq 1$ pour tout réel $x$.
  \item $f$ a-t-elle un maximum sur $\R$~? Un minimum~? Justifier.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g10:functions:10}
Soit $f(x) = x^2$ sur $\R$.
\begin{enumerate}
  \item Soit $0 \leq u < v$. Factoriser $f(v) - f(u)$ et en déduire que
        $f$ est strictement croissante sur $\intco{0}{+\infty}$.
  \item Adapter l'argument pour montrer que $f$ est strictement
        décroissante sur $\intoc{-\infty}{0}$.
\end{enumerate}
\end{exercise}
