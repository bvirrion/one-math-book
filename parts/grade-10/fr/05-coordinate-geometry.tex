\chapter{Géométrie repérée}\label{ch:g10:coordgeom}

La grande idée de Descartes fut de décrire les points par des paires de
nombres, transformant les problèmes de géométrie en calculs. Avec
seulement deux formules --- le milieu et la distance --- on peut prouver
qu'un triangle est isocèle, qu'un quadrilatère est un parallélogramme, ou
que trois points sont alignés, sans tracer une seule ligne auxiliaire.

\section{Coordonnées dans le plan}

\begin{definition}[Repère]\label{def:g10:coordgeom:system}
Un \emph{repère}\index{repère} du plan est constitué d'une origine $O$ et
de deux axes gradués passant par $O$~: l'axe horizontal des $x$ et l'axe
vertical des $y$. Tout point $M$ a alors une unique paire de
\emph{coordonnées}\index{coordonnées} $(x, y)$~: son
\emph{abscisse}\index{abscisse} $x$ et son
\emph{ordonnée}\index{ordonnée} $y$. Le repère est
\emph{orthonormal}\index{repère orthonormal} lorsque les axes sont
perpendiculaires et portent la même unité de longueur. Tous les repères
de ce chapitre sont orthonormaux.
\end{definition}

\section{Milieu d'un segment}

\begin{proposition}[Formule du milieu]\label{prop:g10:coordgeom:midpoint}
Soient $A(x_A, y_A)$ et $B(x_B, y_B)$. Le \emph{milieu}\index{milieu}
$I$ du segment $[AB]$ a pour coordonnées
\[
I\left(\frac{x_A + x_B}{2},\ \frac{y_A + y_B}{2}\right).
\]
\end{proposition}

\begin{proof}
Considérons les coordonnées horizontales. En allant de $A$ à $B$,
l'abscisse change de $x_B - x_A$~; le point à mi-chemin a pour abscisse
\[
x_A + \frac{x_B - x_A}{2} = \frac{2x_A + x_B - x_A}{2}
= \frac{x_A + x_B}{2},
\]
la moyenne des deux abscisses. Le même calcul s'applique aux ordonnées.
\end{proof}

\begin{example}\label{ex:g10:coordgeom:midpoint}
Avec $A(-1, 4)$ et $B(5, -2)$, le milieu de $[AB]$ est
\[
I\left(\frac{-1 + 5}{2},\ \frac{4 + (-2)}{2}\right) = I(2, 1).
\]
La formule fonctionne aussi à l'envers~: si $A(-1, 4)$ et le milieu est
$I(2, 1)$, alors $B$ satisfait $\frac{-1 + x_B}{2} = 2$ et
$\frac{4 + y_B}{2} = 1$, donc $x_B = 5$ et $y_B = -2$~: le point $B$ est
le \emph{symétrique} de $A$ par rapport à $I$.
\end{example}

\section{Distance entre deux points}

\begin{theorem}[Formule de la distance]\label{thm:g10:coordgeom:distance}
Dans un repère orthonormal, la distance entre $A(x_A, y_A)$ et
$B(x_B, y_B)$ est
\[
AB = \sqrt{(x_B - x_A)^2 + (y_B - y_A)^2}.
\]
\end{theorem}

\begin{proof}
Soit $C$ le point $(x_B, y_A)$~: il a la même hauteur que $A$ et la même
abscisse que $B$, donc le triangle $ACB$ a un angle droit en $C$. Ses
côtés adjacents à l'angle droit sont des segments horizontal et vertical,
de longueurs $AC = \abs{x_B - x_A}$ et $CB = \abs{y_B - y_A}$. Par le
théorème de Pythagore,
\[
AB^2 = AC^2 + CB^2 = (x_B - x_A)^2 + (y_B - y_A)^2,
\]
et en prenant la racine carrée (les deux membres sont positifs ou nuls)
on obtient la formule.
\end{proof}

\end{tikzpicture}

{\small La formule de la distance est le théorème de Pythagore appliqué
au triangle rectangle $ACB$ construit sur un côté horizontal et un côté
vertical.}
\end{omfigure}

\begin{example}\label{ex:g10:coordgeom:distance}
Avec $A(1, 1)$ et $B(5, 3)$~:
\[
AB = \sqrt{(5-1)^2 + (3-1)^2} = \sqrt{16 + 4} = \sqrt{20} = 2\sqrt 5 .
\]
Les distances sont en général gardées sous forme exacte (racine carrée)~;
n'arrondir qu'à la toute fin si une réponse décimale est nécessaire.
\end{example}

\begin{remark}
L'ordre des points n'importe pas~: $(x_A - x_B)^2 = (x_B - x_A)^2$. Mais
ne pas oublier les carrés~! La distance n'est \emph{pas}
$\abs{x_B - x_A} + \abs{y_B - y_A}$.
\end{remark}

\section{Utiliser les formules en géométrie}

\begin{method}[Nature d'un triangle]\label{met:g10:coordgeom:triangle}
Étant donnés trois points $A$, $B$, $C$ par leurs coordonnées~:
\begin{enumerate}
  \item calculer les trois longueurs au carré $AB^2$, $AC^2$, $BC^2$ avec
        la formule de la distance (garder les carrés évite les racines)~;
  \item deux longueurs au carré égales $\Rightarrow$ le triangle est
        \emph{isocèle}~;
  \item si la plus grande longueur au carré est la somme des deux
        autres, le triangle est \emph{rectangle} au sommet opposé au plus
        long côté, par la réciproque du théorème de Pythagore.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g10:coordgeom:triangle}
Soient $A(0, 1)$, $B(4, 3)$ et $C(2, -3)$. Alors
\begin{align*}
AB^2 &= (4-0)^2 + (3-1)^2 = 16 + 4 = 20, \\
AC^2 &= (2-0)^2 + (-3-1)^2 = 4 + 16 = 20, \\
BC^2 &= (2-4)^2 + (-3-3)^2 = 4 + 36 = 40 .
\end{align*}
D'abord, $AB^2 = AC^2$, donc $AB = AC$~: le triangle est isocèle en $A$.
De plus $AB^2 + AC^2 = 40 = BC^2$, donc par la réciproque du théorème de
Pythagore le triangle est aussi rectangle en $A$.
\end{example}

\begin{method}[Reconnaître un parallélogramme]\label{met:g10:coordgeom:parallelogram}
Un quadrilatère $ABCD$ est un parallélogramme exactement lorsque ses
diagonales $[AC]$ et $[BD]$ ont le même milieu. Donc~: calculer les deux
milieux avec la formule du milieu et les comparer.
\end{method}

\begin{example}\label{ex:g10:coordgeom:parallelogram}
Soient $A(-1, 0)$, $B(2, 2)$, $C(5, 1)$ et $D(2, -1)$. Le milieu de
$[AC]$ est $\left(\frac{-1+5}{2}, \frac{0+1}{2}\right) = (2, \frac12)$~;
le milieu de $[BD]$ est
$\left(\frac{2+2}{2}, \frac{2-1}{2}\right) = (2, \frac12)$. Ils
coïncident, donc $ABCD$ est un parallélogramme.
\end{example}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:g10:coordgeom:1}
Soient $A(2, 5)$ et $B(-4, 1)$. Calculer les coordonnées du milieu de
$[AB]$ et la distance $AB$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:coordgeom:2}
Soient $A(3, -2)$ et $I(1, 2)$. Trouver les coordonnées du point $B$ tel
que $I$ soit le milieu de $[AB]$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:coordgeom:3}
Lequel des points $P(4, 1)$, $Q(-3, 2)$ et $R(0, -5)$ est le plus proche
de l'origine $O(0,0)$~? Répondre avec des distances exactes.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:coordgeom:4}
Placer les points $A(1, 2)$, $B(5, 4)$, $C(7, 0)$ sur du papier
millimétré, puis montrer par le calcul que le triangle $ABC$ est isocèle.
Est-il rectangle~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:coordgeom:5}
Soient $A(-2, 1)$, $B(1, 3)$, $C(4, 1)$ et $D(1, -1)$.
\begin{enumerate}
  \item Montrer que $ABCD$ est un parallélogramme.
  \item Calculer $AB$ et $BC$. $ABCD$ est-il un losange (tous les côtés
        égaux)~?
  \item Calculer $AC$ et $BD$. $ABCD$ est-il un rectangle (diagonales
        égales)~?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:coordgeom:6}
Le cercle de centre $\Omega(2, 1)$ et de rayon $5$ est l'ensemble de
tous les points à distance $5$ de $\Omega$. Lesquels des points
$A(5, 5)$, $B(-1, -3)$, $C(6, 2)$ sont sur ce cercle~? À l'intérieur~?
À l'extérieur~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:coordgeom:7}
Soient $A(1, 1)$ et $B(7, 5)$. Trouver tous les points $M(x, 0)$ de
l'axe des $x$ équidistants de $A$ et de $B$. (Écrire $MA^2 = MB^2$ et
résoudre en $x$.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:coordgeom:8}
Soient $A(0, 3)$, $B(4, 1)$.
\begin{enumerate}
  \item Calculer les coordonnées du milieu $I$ de $[AB]$.
  \item Montrer que $M(2, 2) = I$, puis vérifier en calculant $MA$, $MB$
        que $M$ est équidistant de $A$ et de $B$.
  \item Trouver un second point, sur l'axe des $y$, équidistant de $A$ et
        de $B$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:coordgeom:9}
Les points $A(-1, -1)$, $B(3, 1)$ et $C(11, 5)$ sont donnés. Calculer
$AB$, $BC$ et $AC$, et en déduire que $A$, $B$, $C$ sont alignés.
(Trois points sont alignés exactement lorsque la plus grande des trois
distances est la somme des deux autres.)
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g10:coordgeom:10}
Soient $A(a, 0)$ et $B(0, b)$ avec $a, b > 0$, et soit $I$ le milieu de
$[AB]$. Montrer par le calcul que $OI = IA = IB$, où $O$ est l'origine.
(Cela prouve un théorème classique~: dans un triangle rectangle, le
milieu de l'hypoténuse est équidistant des trois sommets.)
\end{exercise}
