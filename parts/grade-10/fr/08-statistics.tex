\chapter{Statistique descriptive}\label{ch:g10:stats}

La statistique résume une grande collection de nombres --- notes,
salaires, températures --- par quelques indicateurs bien choisis. Ce
chapitre introduit les deux familles d'indicateurs~: ceux qui localisent
le \emph{centre} des données (moyenne, médiane) et ceux qui mesurent leur
\emph{dispersion} (étendue, quartiles). Un traitement plus approfondi,
incluant l'écart type, vient dans le \cref{ch:g11:stat}.

\section{Données et fréquences}

\begin{definition}[Série statistique]\label{def:g10:stats:series}
Une \emph{série statistique}\index{série statistique} est une liste de
valeurs $x_1, x_2, \dots, x_N$ observées sur une population de taille
$N$. Lorsqu'une valeur $x$ apparaît $n$ fois, $n$ est son
\emph{effectif} et le quotient
\[
f = \frac{n}{N}
\]
est sa \emph{fréquence}\index{fréquence} (souvent exprimée en
pourcentage). Les fréquences de toutes les valeurs s'additionnent à $1$.
\end{definition}

\begin{example}\label{ex:g10:stats:frequencies}
Les notes de $20$ élèves à un quiz sur $5$ points~:
\begin{center}
\renewcommand{\arraystretch}{1.2}
\begin{tabular}{c|cccccc}
note $x$ & $0$ & $1$ & $2$ & $3$ & $4$ & $5$ \\
\hline
effectif $n$ & $1$ & $2$ & $4$ & $6$ & $5$ & $2$ \\
fréquence $f$ & $0.05$ & $0.10$ & $0.20$ & $0.30$ & $0.25$ & $0.10$ \\
\end{tabular}
\end{center}
Vérification~: $1 + 2 + 4 + 6 + 5 + 2 = 20$ et les fréquences somment à
$1$.
\end{example}

\section{Indicateurs de position}

\begin{definition}[Moyenne]\label{def:g10:stats:mean}
La \emph{moyenne}\index{moyenne} de la série est
\[
\bar x = \frac{x_1 + x_2 + \dots + x_N}{N}.
\]
Si les valeurs $v_1, \dots, v_k$ apparaissent avec les effectifs
$n_1, \dots, n_k$, le même nombre se calcule comme la \emph{moyenne
pondérée}
\[
\bar x = \frac{n_1 v_1 + n_2 v_2 + \dots + n_k v_k}{n_1 + n_2 + \dots + n_k}.
\]
\end{definition}

\begin{example}\label{ex:g10:stats:mean}
Pour le quiz de l'\cref{ex:g10:stats:frequencies}~:
\[
\bar x = \frac{1 \times 0 + 2 \times 1 + 4 \times 2 + 6 \times 3
+ 5 \times 4 + 2 \times 5}{20}
= \frac{0 + 2 + 8 + 18 + 20 + 10}{20}
= \frac{58}{20} = 2.9 .
\]
\end{example}

\begin{definition}[Médiane]\label{def:g10:stats:median}
La \emph{médiane}\index{médiane} est une valeur qui partage la série
\emph{triée} en deux moitiés~: au moins la moitié des valeurs sont
$\leq$ la médiane, et au moins la moitié sont $\geq$ à elle. En
pratique, trier les $N$ valeurs~;
\begin{itemize}
  \item si $N$ est impair, la médiane est la valeur du milieu, en
        position $\frac{N+1}{2}$~;
  \item si $N$ est pair, prendre le milieu des deux valeurs centrales, en
        positions $\frac N2$ et $\frac N2 + 1$.
\end{itemize}
\end{definition}

\begin{example}\label{ex:g10:stats:median}
Série triée de $7$ prix de maisons (en milliers)~: $120$, $150$, $160$,
$180$, $210$, $240$, $900$. La médiane est la $4$-ième valeur, $180$. La
moyenne est $\frac{1960}{7} = 280$ --- plus grande que $6$ des $7$
prix~! Une valeur extrême ($900$) tire la moyenne bien plus que la
médiane~: la médiane est \emph{robuste}, la moyenne ne l'est pas.
\end{example}

\section{Indicateurs de dispersion}

\begin{definition}[Étendue, quartiles]\label{def:g10:stats:quartiles}
Pour une série triée~:
\begin{itemize}
  \item l'\emph{étendue}\index{étendue} est la différence entre la plus
        grande et la plus petite valeur~;
  \item le \emph{premier quartile}\index{quartile} $Q_1$ est la plus
        petite valeur telle qu'au moins un quart ($25\,\%$) des valeurs
        sont $\leq Q_1$~; le \emph{troisième quartile} $Q_3$ est la plus
        petite valeur telle qu'au moins les trois quarts ($75\,\%$) des
        valeurs sont $\leq Q_3$~;
  \item l'\emph{écart interquartile} $Q_3 - Q_1$ mesure la dispersion de
        la moitié centrale des données.
\end{itemize}
\end{definition}

\begin{method}[Trouver les quartiles]\label{met:g10:stats:quartiles}
Pour une série de $N$ valeurs triées~:
\begin{enumerate}
  \item calculer $\frac N4$~; s'il n'est pas entier, l'arrondir
        \emph{au-dessus} à l'entier suivant~; $Q_1$ est la valeur en
        cette position~;
  \item calculer $\frac{3N}{4}$~; arrondir au-dessus si besoin~; $Q_3$
        est la valeur en cette position.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g10:stats:quartiles}
Prendre les $11$ valeurs triées
\[
2,\ 3,\ 5,\ 5,\ 6,\ 7,\ 8,\ 9,\ 9,\ 10,\ 12 .
\]
Médiane~: position $\frac{11+1}{2} = 6$, donc la médiane est $7$.
$Q_1$~: $\frac{11}{4} = 2.75$, arrondir à $3$~; la $3$-ième valeur est
$5$.
$Q_3$~: $\frac{33}{4} = 8.25$, arrondir à $9$~; la $9$-ième valeur est
$9$.
Étendue~: $12 - 2 = 10$~; écart interquartile~: $9 - 5 = 4$.
\end{example}

\begin{example}[Comparer deux groupes]\label{ex:g10:stats:comparison}
Deux classes passent le même test (notes sur $20$)~:
\begin{center}
\renewcommand{\arraystretch}{1.2}
\begin{tabular}{c|ccc}
 & médiane & $Q_1$ & $Q_3$ \\
\hline
classe A & $12$ & $10$ & $14$ \\
classe B & $12$ & $6$ & $17$ \\
\end{tabular}
\end{center}
Même médiane, dispersions très différentes~: la classe A est homogène
(la moitié de ses notes dans $\intcc{10}{14}$), la classe B est
hétérogène (sa moitié centrale s'étend sur $\intcc{6}{17}$). Les
indicateurs de position seuls ne disent jamais toute l'histoire.
\end{example}

\begin{remark}
Ni la moyenne ni la médiane n'est «~meilleure~»~: la moyenne utilise
chaque valeur (et est nécessaire pour calculer des totaux), la médiane
résiste aux valeurs extrêmes. Un résumé sérieux d'une série donne au
moins un indicateur de position \emph{et} un de dispersion.
\end{remark}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:g10:stats:1}
Voici le nombre de livres lus en un an par $15$ élèves~:
\[
0,\ 1,\ 1,\ 2,\ 2,\ 2,\ 3,\ 3,\ 4,\ 4,\ 5,\ 6,\ 7,\ 8,\ 12 .
\]
Calculer la moyenne, la médiane et l'étendue de cette série.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:stats:2}
Un dé a été lancé $50$ fois~:
\begin{center}
\renewcommand{\arraystretch}{1.2}
\begin{tabular}{c|cccccc}
issue & $1$ & $2$ & $3$ & $4$ & $5$ & $6$ \\
\hline
effectif & $7$ & $9$ & $8$ & $10$ & $6$ & $10$ \\
\end{tabular}
\end{center}
Calculer la fréquence de chaque issue et la moyenne des issues.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:stats:3}
Trouver la médiane, $Q_1$ et $Q_3$ de la série triée
\[
1,\ 2,\ 4,\ 4,\ 5,\ 6,\ 6,\ 7,\ 8,\ 9,\ 9,\ 11 \qquad (N = 12),
\]
et dessiner le diagramme en boîte correspondant.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:stats:4}
La moyenne de $24$ notes de test est $11.5$. Un $25$-ième élève passe le
test et obtient $19$. Quelle est la nouvelle moyenne de la classe~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:stats:5}
Un élève a une moyenne de $12$ après $5$ tests, tous de même poids.
Quelle note au $6$-ième test porterait la moyenne à $13$~? Une moyenne de
$14$ est-elle atteignable (notes sur $20$)~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:stats:6}
Inventer une série de $7$ valeurs positives ou nulles dont la médiane est
$10$ et dont la moyenne est plus grande que $20$~; puis une dont la
médiane est $10$ et dont la moyenne est plus petite que $6$. Pourquoi la
moyenne d'une telle série ne peut-elle jamais être plus petite que
$\frac{40}{7}$~? Que montrent ces exemples~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:stats:7}
Dans une entreprise, les $9$ employés gagnent $1800$ chacun par mois et
le directeur gagne $12000$.
\begin{enumerate}
  \item Calculer le salaire moyen et le salaire médian.
  \item Quel indicateur décrit le mieux un salaire «~typique~» ici~?
        Pourquoi~?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:stats:8}
Une classe de $12$ garçons a une taille moyenne de $168$~cm~; les $18$
filles de la même classe ont une taille moyenne de $163$~cm. Calculer la
taille moyenne de toute la classe. (Attention~: ce n'est \emph{pas} le
milieu de $168$ et $163$ --- pondérer les moyennes par les tailles des
groupes.)
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g10:stats:9}
Une série de $N$ valeurs $x_1, \dots, x_N$ a pour moyenne $\bar x$.
Chaque valeur est transformée en $y_i = a x_i + b$ pour des nombres
fixes $a$ et $b$.
\begin{enumerate}
  \item Montrer que la moyenne de la nouvelle série est $a \bar x + b$.
  \item Les températures d'une semaine, en degrés Celsius, ont pour
        moyenne $20$. Quelle est leur moyenne en degrés Fahrenheit
        ($F = 1.8\,C + 32$)~?
\end{enumerate}
\end{exercise}
