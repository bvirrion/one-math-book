\chapter{Équations de droites et systèmes linéaires}\label{ch:g10:lines}

Une droite dans le plan repéré est décrite par une \omterm{def:g10:algebra:equation}{équation}, et deux
droites ensemble forment un système d'\omterm{def:g10:algebra:equation}{équations} dont les solutions sont
leurs points d'\omterm{def:g10:numbers:interunion}{intersection}. Ce chapitre relie trois langages ---
géométrique (droites), algébrique (\omterm{def:g10:algebra:equation}{équations}) et fonctionnel (\omterm{def:g10:reffunc:affine}{fonctions
affines} du \cref{ch:g10:reffunc}) --- et enseigne les deux méthodes
standard de résolution d'un \omterm{def:g10:lines:system}{système linéaire}.

\section{L'équation d'une droite}

\begin{theorem}[Équation réduite]\label{thm:g10:lines:reduced}
Dans un \omterm{def:g10:coordgeom:system}{repère}~:
\begin{enumerate}
  \item toute droite non verticale a une \omterm{def:g10:algebra:equation}{équation} de la forme
        \[
        y = mx + p ,
        \]
        avec $m$ (le \emph{coefficient directeur}\index{coefficient
        directeur}) et $p$ (l'\emph{\omterm{def:g10:reffunc:affine}{ordonnée à l'origine}}) uniquement
        déterminés --- c'est l'\emph{équation réduite}\index{équation
        réduite} de la droite~;
  \item toute droite verticale a une \omterm{def:g10:algebra:equation}{équation} $x = c$~;
  \item un point appartient à la droite exactement lorsque ses
        \omterm{def:g10:coordgeom:system}{coordonnées} satisfont l'\omterm{def:g10:algebra:equation}{équation}.
\end{enumerate}
\end{theorem}
\admitted

\begin{proposition}[Coefficient directeur par deux points]\label{prop:g10:lines:slope}
Le \omterm{def:g10:reffunc:affine}{coefficient directeur} de la droite (non verticale) passant par
$A(x_A, y_A)$ et $B(x_B, y_B)$, avec $x_A \neq x_B$, est
\[
m = \frac{y_B - y_A}{x_B - x_A}
\qquad
\text{(«~variation de $y$ sur variation de $x$~»).}
\]
\end{proposition}

\begin{proof}
Les deux points satisfont $y = mx + p$, donc
$y_B - y_A = (mx_B + p) - (mx_A + p) = m(x_B - x_A)$~; diviser par
$x_B - x_A \neq 0$.
\end{proof}

\begin{method}[Trouver l'équation d'une droite par deux points]\label{met:g10:lines:through}
Étant donnés $A$ et $B$ avec $x_A \neq x_B$~:
\begin{enumerate}
  \item calculer le \omterm{def:g10:reffunc:affine}{coefficient directeur}
        $m = \dfrac{y_B - y_A}{x_B - x_A}$~;
  \item écrire $y = mx + p$ et substituer les \omterm{def:g10:coordgeom:system}{coordonnées} de $A$ (ou $B$)
        pour trouver $p$~;
  \item vérifier l'\omterm{def:g10:algebra:equation}{équation} finale avec l'\emph{autre} point.
\end{enumerate}
Si $x_A = x_B$, la droite est verticale~: son \omterm{def:g10:algebra:equation}{équation} est $x = x_A$.
\end{method}

\begin{example}\label{ex:g10:lines:through}
Droite passant par $A(1, 5)$ et $B(3, 1)$. \omterm{def:g10:reffunc:affine}{Coefficient directeur}~:
$m = \frac{1 - 5}{3 - 1} = \frac{-4}{2} = -2$. Puis $y = -2x + p$~; le
point $A$ donne $5 = -2 \times 1 + p$, donc $p = 7$. \omterm{def:g10:algebra:equation}{Équation}~:
$y = -2x + 7$. Vérification avec $B$~: $-2 \times 3 + 7 = 1$.
\end{example}

\begin{proposition}[Droites parallèles]\label{prop:g10:lines:parallel}
Deux droites non verticales sont parallèles si et seulement si elles ont
le même \omterm{def:g10:reffunc:affine}{coefficient directeur}. Deux droites parallèles distinctes n'ont
aucun point commun~; deux droites de \omterm{def:g10:reffunc:affine}{coefficients directeurs} différents
en ont exactement un.
\end{proposition}

\begin{proof}
Soient $y = mx + p$ et $y = m'x + p'$ les deux droites. Un point commun
résout $mx + p = m'x + p'$, c'est-à-dire $(m - m')x = p' - p$. Si
$m \neq m'$, cela a exactement une solution
$x = \frac{p' - p}{m - m'}$~: un point d'\omterm{def:g10:numbers:interunion}{intersection}. Si $m = m'$,
l'\omterm{def:g10:algebra:equation}{équation} s'écrit $0 = p' - p$~: aucune solution lorsque $p \neq p'$
(parallèles distinctes), tout $x$ lorsque $p = p'$ (même droite). Le
parallélisme (n'avoir aucune \omterm{def:g10:numbers:interunion}{intersection}, ou être confondues) est donc
équivalent à $m = m'$.
\end{proof}

\begin{remark}[Vecteur directeur]
La droite $y = mx + p$ monte de $m$ unités pour chaque unité vers la
droite, donc le \omterm{def:g10:vectors:vector}{vecteur} $\vec u\,(1, m)$ est un \emph{vecteur
directeur}\index{vecteur directeur} de la droite~: la droite est
parallèle à $\vec u$. Deux droites sont parallèles exactement lorsque
leurs \omterm{def:g10:vectors:vector}{vecteurs} directeurs sont \omterm{def:g10:vectors:collinear}{colinéaires}
(\cref{thm:g10:vectors:det}).
\end{remark}

\section{Systèmes linéaires}

\begin{definition}[Système linéaire]\label{def:g10:lines:system}
Un \emph{système de deux équations linéaires}\index{système linéaire} aux
inconnues $x$ et $y$ est une paire d'\omterm{def:g10:algebra:equation}{équations}
\[
\begin{cases}
a x + b y = c \\
a' x + b' y = c'
\end{cases}
\]
à satisfaire \emph{simultanément}. Une \emph{solution} est un couple
$(x, y)$ satisfaisant les deux. Chaque \omterm{def:g10:algebra:equation}{équation} décrit une droite, donc
résoudre le système, c'est trouver l'\omterm{def:g10:numbers:interunion}{intersection} de deux droites.
\end{definition}

\begin{method}[Substitution]\label{met:g10:lines:substitution}
\begin{enumerate}
  \item Résoudre une \omterm{def:g10:algebra:equation}{équation} en une inconnue --- choisir la plus
        simple, par exemple exprimer $y$ en \omterm{def:g10:functions:function}{fonction} de $x$~;
  \item substituer cette expression dans l'\emph{autre} \omterm{def:g10:algebra:equation}{équation}, qui ne
        contient plus que $x$~;
  \item résoudre en $x$, puis calculer $y$ à partir de l'étape~1~;
  \item vérifier le couple dans les deux \omterm{def:g10:algebra:equation}{équations} d'origine.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g10:lines:substitution}
Résoudre $\begin{cases} x + 2y = 8 \\ 3x - y = 3 \end{cases}$ par
substitution. La seconde \omterm{def:g10:algebra:equation}{équation} donne $y = 3x - 3$. Substituer dans la
première~:
\[
x + 2(3x - 3) = 8
\ \Longleftrightarrow\
x + 6x - 6 = 8
\ \Longleftrightarrow\
7x = 14
\ \Longleftrightarrow\
x = 2 ,
\]
puis $y = 3 \times 2 - 3 = 3$. La solution est le couple $(2, 3)$.
Vérification~: $2 + 2 \times 3 = 8$ et $3 \times 2 - 3 = 3$.
\end{example}

\begin{method}[Combinaison]\label{met:g10:lines:elimination}
\begin{enumerate}
  \item Multiplier chaque \omterm{def:g10:algebra:equation}{équation} par un nombre bien choisi pour qu'une
        inconnue ait des coefficients opposés dans les deux \omterm{def:g10:algebra:equation}{équations}~;
  \item additionner les \omterm{def:g10:algebra:equation}{équations}~: cette inconnue disparaît~;
  \item résoudre l'\omterm{def:g10:algebra:equation}{équation} à une inconnue restante, puis substituer pour
        trouver l'autre inconnue~;
  \item vérifier le couple dans les deux \omterm{def:g10:algebra:equation}{équations} d'origine.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g10:lines:elimination}
Résoudre $\begin{cases} 2x + 3y = 7 \\ 5x - 2y = 8 \end{cases}$ par
combinaison. Pour éliminer $y$, multiplier la première \omterm{def:g10:algebra:equation}{équation} par $2$
et la seconde par $3$~:
\[
\begin{cases}
4x + 6y = 14 \\
15x - 6y = 24
\end{cases}
\]
En les additionnant~: $19x = 38$, donc $x = 2$. En substituant dans
$2x + 3y = 7$~: $4 + 3y = 7$, donc $y = 1$. Solution~: $(2, 1)$.
Vérification dans la seconde \omterm{def:g10:algebra:equation}{équation}~: $5 \times 2 - 2 \times 1 = 8$.
\end{example}

\begin{remark}[Combien de solutions~?]
Comme deux droites, un \omterm{def:g10:lines:system}{système linéaire} a en général exactement une
solution (droites non parallèles), et exceptionnellement aucune (droites
parallèles distinctes) ou une infinité (deux fois la même droite). Le
critère du \cref{thm:g10:vectors:det} décide~: le système a une unique
solution exactement lorsque $ab' - a'b \neq 0$.
\end{remark}

\begin{example}[Modéliser avec un système]\label{ex:g10:lines:modeling}
Trois cahiers et deux stylos coûtent $8$~; un cahier et quatre stylos
coûtent $6$. Soit $x$ le prix d'un cahier et $y$ celui d'un stylo~:
\[
\begin{cases}
3x + 2y = 8 \\
x + 4y = 6 .
\end{cases}
\]
De la seconde \omterm{def:g10:algebra:equation}{équation}, $x = 6 - 4y$~; en substituant,
$3(6 - 4y) + 2y = 8$, donc $18 - 10y = 8$, d'où $y = 1$ puis $x = 2$. Un
cahier coûte $2$ et un stylo $1$.
\end{example}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:g10:lines:1}
Pour chaque droite, lire le \omterm{def:g10:reffunc:affine}{coefficient directeur} et l'\omterm{def:g10:reffunc:affine}{ordonnée à
l'origine}, et la croquer~:
\[
y = 2x - 3, \qquad
y = -\tfrac13 x + 1, \qquad
y = 4, \qquad
x = -2 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:lines:2}
Le point $A(2, 7)$ appartient-il à la droite $y = 3x + 1$~? Et
$B(-1, -1)$~? Et le point $C(4, 13)$~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:lines:3}
Trouver l'\omterm{thm:g10:lines:reduced}{équation réduite} de la droite passant par~:
\[
\text{(a) } A(0, 2) \text{ et } B(3, 8);
\qquad
\text{(b) } A(-1, 5) \text{ et } B(2, -4);
\qquad
\text{(c) } A(4, 1) \text{ et } B(4, 6).
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:lines:4}
Parmi les droites $y = 3x - 1$, $y = -3x + 2$, $y = 3x + 5$,
$y = \frac{6x + 2}{2}$, lesquelles sont parallèles entre elles~?
Lesquelles sont en fait confondues~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:lines:5}
Résoudre par substitution~:
\[
\begin{cases}
y = 2x - 1 \\
3x + y = 9
\end{cases}
\qquad\text{puis}\qquad
\begin{cases}
x - y = 4 \\
2x + 3y = 3 .
\end{cases}
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:lines:6}
Résoudre par combinaison~:
\[
\begin{cases}
x + y = 10 \\
x - y = 4
\end{cases}
\qquad\text{puis}\qquad
\begin{cases}
3x + 2y = 12 \\
2x + 5y = 8 .
\end{cases}
\]
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:lines:7}
Déterminer, sans les résoudre, combien de solutions chaque système
a~:
\[
\begin{cases}
2x - y = 3 \\
-4x + 2y = -6
\end{cases}
\qquad
\begin{cases}
2x - y = 3 \\
-4x + 2y = 1
\end{cases}
\qquad
\begin{cases}
2x - y = 3 \\
x + y = 0 .
\end{cases}
\]
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:lines:8}
Deux cents billets ont été vendus pour une pièce de théâtre scolaire,
certains à $5$ (enfants) et d'autres à $9$ (adultes), pour un total de
$1352$. Combien de billets de chaque type ont été vendus~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:lines:9}
Soit $d$ la droite $y = 2x - 3$ et $A(1, 4)$.
\begin{enumerate}
  \item Donner l'\omterm{def:g10:algebra:equation}{équation} de la droite $d'$ parallèle à $d$ passant par
        $A$.
  \item Calculer le point d'\omterm{def:g10:numbers:interunion}{intersection} de $d'$ avec l'axe des $x$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g10:lines:10}
Soient $A(0, 0)$, $B(6, 2)$ et $C(2, 4)$.
\begin{enumerate}
  \item Trouver les \omterm{def:g10:algebra:equation}{équations} des médianes du triangle $ABC$ issues de
        $A$ et de $B$ (une médiane joint un sommet au \omterm{prop:g10:coordgeom:midpoint}{milieu} du côté
        opposé).
  \item Calculer leur point d'\omterm{def:g10:numbers:interunion}{intersection} $G$, et vérifier que $G$ est
        aussi sur la troisième médiane. (On doit trouver que les
        \omterm{def:g10:coordgeom:system}{coordonnées} de $G$ sont les moyennes de celles de $A$, $B$,
        $C$.)
\end{enumerate}
\end{exercise}
