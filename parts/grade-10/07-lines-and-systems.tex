\chapter{Equations of Lines and Linear Systems}\label{ch:g10:lines}

A straight line in the coordinate plane is described by an equation, and
two lines together form a system of equations whose solutions are their
intersection points. This chapter connects three languages --- geometric
(lines), algebraic (equations) and functional (affine functions from
\cref{ch:g10:reffunc}) --- and teaches the two standard ways of solving a
linear system.

\section{The equation of a line}

\begin{theorem}[Reduced equation]\label{thm:g10:lines:reduced}
In a coordinate system:
\begin{enumerate}
  \item every non-vertical line has an equation of the form
        \[
        y = mx + p ,
        \]
        with $m$ (the \emph{slope}\index{slope}) and $p$ (the
        \emph{$y$-intercept}) uniquely determined --- this is the
        \emph{reduced equation}\index{reduced equation} of the line;
  \item every vertical line has an equation $x = c$;
  \item a point lies on the line exactly when its coordinates satisfy the
        equation.
\end{enumerate}
\end{theorem}
\admitted

\begin{proposition}[Slope through two points]\label{prop:g10:lines:slope}
The slope of the (non-vertical) line through $A(x_A, y_A)$ and
$B(x_B, y_B)$, with $x_A \neq x_B$, is
\[
m = \frac{y_B - y_A}{x_B - x_A}
\qquad
\text{(``change in $y$ over change in $x$'').}
\]
\end{proposition}

\begin{proof}
Both points satisfy $y = mx + p$, so $y_B - y_A = (mx_B + p) - (mx_A + p)
= m(x_B - x_A)$; divide by $x_B - x_A \neq 0$.
\end{proof}

\begin{method}[Finding the equation of a line through two points]\label{met:g10:lines:through}
Given $A$ and $B$ with $x_A \neq x_B$:
\begin{enumerate}
  \item compute the slope $m = \dfrac{y_B - y_A}{x_B - x_A}$;
  \item write $y = mx + p$ and substitute the coordinates of $A$ (or $B$)
        to find $p$;
  \item check the final equation with the \emph{other} point.
\end{enumerate}
If $x_A = x_B$, the line is vertical: its equation is $x = x_A$.
\end{method}

\begin{example}\label{ex:g10:lines:through}
Line through $A(1, 5)$ and $B(3, 1)$. Slope:
$m = \frac{1 - 5}{3 - 1} = \frac{-4}{2} = -2$. Then $y = -2x + p$; the
point $A$ gives $5 = -2 \times 1 + p$, so $p = 7$. Equation:
$y = -2x + 7$. Check with $B$: $-2 \times 3 + 7 = 1$.
\end{example}

\begin{proposition}[Parallel lines]\label{prop:g10:lines:parallel}
Two non-vertical lines are parallel if and only if they have the same
slope. Two distinct parallel lines have no common point; two lines with
different slopes have exactly one.
\end{proposition}

\begin{proof}
Let $y = mx + p$ and $y = m'x + p'$ be the two lines. A common point
solves $mx + p = m'x + p'$, i.e.\ $(m - m')x = p' - p$. If $m \neq m'$,
this has exactly one solution $x = \frac{p' - p}{m - m'}$: one
intersection point. If $m = m'$, the equation reads $0 = p' - p$: no
solution when $p \neq p'$ (distinct parallels), every $x$ when $p = p'$
(same line). Parallelism (having no intersection, or being equal) is thus
equivalent to $m = m'$.
\end{proof}

\begin{remark}[Direction vector]
The line $y = mx + p$ goes up $m$ units for each unit to the right, so the
vector $\vec u\,(1, m)$ is a \emph{direction
vector}\index{direction vector} of the line: the line is parallel to
$\vec u$. Two lines are parallel exactly when their direction vectors are
collinear (\cref{thm:g10:vectors:det}).
\end{remark}

\section{Linear systems}

\begin{definition}[Linear system]\label{def:g10:lines:system}
A \emph{system of two linear equations}\index{linear system} in the
unknowns $x$ and $y$ is a pair of equations
\[
\begin{cases}
a x + b y = c \\
a' x + b' y = c'
\end{cases}
\]
to be satisfied \emph{simultaneously}. A \emph{solution} is a pair
$(x, y)$ satisfying both. Each equation describes a line, so solving the
system means finding the intersection of two lines.
\end{definition}

\begin{method}[Substitution]\label{met:g10:lines:substitution}
\begin{enumerate}
  \item Solve one equation for one unknown --- pick the easiest, e.g.\
        express $y$ in terms of $x$;
  \item substitute this expression into the \emph{other} equation, which
        now contains only $x$;
  \item solve for $x$, then compute $y$ from step 1;
  \item check the pair in both original equations.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g10:lines:substitution}
Solve $\begin{cases} x + 2y = 8 \\ 3x - y = 3 \end{cases}$ by
substitution. The second equation gives $y = 3x - 3$. Substitute into the
first:
\[
x + 2(3x - 3) = 8
\ \Longleftrightarrow\
x + 6x - 6 = 8
\ \Longleftrightarrow\
7x = 14
\ \Longleftrightarrow\
x = 2 ,
\]
then $y = 3 \times 2 - 3 = 3$. The solution is the pair $(2, 3)$.
Check: $2 + 2 \times 3 = 8$ and $3 \times 2 - 3 = 3$.
\end{example}

\begin{method}[Elimination]\label{met:g10:lines:elimination}
\begin{enumerate}
  \item Multiply each equation by a well-chosen number so that one unknown
        gets opposite coefficients in the two equations;
  \item add the equations: that unknown disappears;
  \item solve the remaining one-unknown equation, then substitute back to
        find the other unknown;
  \item check the pair in both original equations.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g10:lines:elimination}
Solve $\begin{cases} 2x + 3y = 7 \\ 5x - 2y = 8 \end{cases}$ by
elimination. To eliminate $y$, multiply the first equation by $2$ and the
second by $3$:
\[
\begin{cases}
4x + 6y = 14 \\
15x - 6y = 24
\end{cases}
\]
Adding them: $19x = 38$, so $x = 2$. Substituting into $2x + 3y = 7$:
$4 + 3y = 7$, so $y = 1$. Solution: $(2, 1)$. Check in the second
equation: $5 \times 2 - 2 \times 1 = 8$.
\end{example}

\begin{remark}[How many solutions?]
Like two lines, a linear system has exactly one solution in general
(non-parallel lines), and exceptionally none (distinct parallel lines) or
infinitely many (twice the same line). The criterion of
\cref{thm:g10:vectors:det} decides: the system has a unique solution
exactly when $ab' - a'b \neq 0$.
\end{remark}

\begin{example}[Modeling with a system]\label{ex:g10:lines:modeling}
Three notebooks and two pens cost $8$; one notebook and four pens cost
$6$. Let $x$ be the price of a notebook and $y$ that of a pen:
\[
\begin{cases}
3x + 2y = 8 \\
x + 4y = 6 .
\end{cases}
\]
From the second equation, $x = 6 - 4y$; substituting,
$3(6 - 4y) + 2y = 8$, so $18 - 10y = 8$, giving $y = 1$ and then
$x = 2$. A notebook costs $2$ and a pen costs $1$.
\end{example}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:g10:lines:1}
For each line, read off the slope and the $y$-intercept, and sketch it:
\[
y = 2x - 3, \qquad
y = -\tfrac13 x + 1, \qquad
y = 4, \qquad
x = -2 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:lines:2}
Does the point $A(2, 7)$ belong to the line $y = 3x + 1$? And $B(-1, -1)$?
And the point $C(4, 13)$?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:lines:3}
Find the reduced equation of the line through:
\[
\text{(a) } A(0, 2) \text{ and } B(3, 8);
\qquad
\text{(b) } A(-1, 5) \text{ and } B(2, -4);
\qquad
\text{(c) } A(4, 1) \text{ and } B(4, 6).
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:lines:4}
Among the lines $y = 3x - 1$, $y = -3x + 2$, $y = 3x + 5$,
$y = \frac{6x + 2}{2}$, which are parallel to each other? Which are
actually equal?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:lines:5}
Solve by substitution:
\[
\begin{cases}
y = 2x - 1 \\
3x + y = 9
\end{cases}
\qquad\text{then}\qquad
\begin{cases}
x - y = 4 \\
2x + 3y = 3 .
\end{cases}
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:lines:6}
Solve by elimination:
\[
\begin{cases}
x + y = 10 \\
x - y = 4
\end{cases}
\qquad\text{then}\qquad
\begin{cases}
3x + 2y = 12 \\
2x + 5y = 8 .
\end{cases}
\]
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:lines:7}
Determine, without solving them, how many solutions each system has:
\[
\begin{cases}
2x - y = 3 \\
-4x + 2y = -6
\end{cases}
\qquad
\begin{cases}
2x - y = 3 \\
-4x + 2y = 1
\end{cases}
\qquad
\begin{cases}
2x - y = 3 \\
x + y = 0 .
\end{cases}
\]
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:lines:8}
Two hundred tickets were sold for a school play, some at $5$ (children)
and some at $9$ (adults), for a total of $1352$. How many tickets of each
kind were sold?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:lines:9}
Let $d$ be the line $y = 2x - 3$ and $A(1, 4)$.
\begin{enumerate}
  \item Give the equation of the line $d'$ parallel to $d$ passing
        through $A$.
  \item Compute the intersection point of $d'$ with the $x$-axis.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g10:lines:10}
Let $A(0, 0)$, $B(6, 2)$ and $C(2, 4)$.
\begin{enumerate}
  \item Find the equations of the medians of the triangle $ABC$ issued
        from $A$ and from $B$ (a median joins a vertex to the midpoint of
        the opposite side).
  \item Compute their intersection point $G$, and check that
        $G$ is also on the third median. (You should find that the
        coordinates of $G$ are the averages of those of $A$, $B$, $C$.)
\end{enumerate}
\end{exercise}
