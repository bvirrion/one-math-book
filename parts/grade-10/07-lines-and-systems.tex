\chapter{Equations of Lines and Linear Systems}\label{ch:g10:lines}

A straight line in the coordinate plane is described by an \omterm{def:g10:algebra:equation}{equation}, and
two lines together form a system of \omterm{def:g10:algebra:equation}{equations} whose solutions are their
\omterm{def:g10:numbers:interunion}{intersection} points. This chapter connects three languages --- geometric
(lines), algebraic (\omterm{def:g10:algebra:equation}{equations}) and functional (\omterm{def:g10:reffunc:affine}{affine functions} from
\cref{ch:g10:reffunc}) --- and teaches the two standard ways of solving a
\omterm{def:g10:lines:system}{linear system}.

\section{The equation of a line}

\begin{theorem}[Reduced equation]\label{thm:g10:lines:reduced}
In a \omterm{def:g10:coordgeom:system}{coordinate system}:
\begin{enumerate}
  \item every non-vertical line has an \omterm{def:g10:algebra:equation}{equation} of the form
        \[
        y = mx + p ,
        \]
        with $m$ (the \emph{slope}\index{slope}) and $p$ (the
        \emph{$y$-intercept}) uniquely determined --- this is the
        \emph{reduced equation}\index{reduced equation} of the line;
  \item every vertical line has an \omterm{def:g10:algebra:equation}{equation} $x = c$;
  \item a point lies on the line exactly when its \omterm{def:g10:coordgeom:system}{coordinates} satisfy the
        \omterm{def:g10:algebra:equation}{equation}.
\end{enumerate}
\end{theorem}
\admitted

\begin{proposition}[Slope through two points]\label{prop:g10:lines:slope}
The \omterm{def:g10:reffunc:affine}{slope} of the (non-vertical) line through $A(x_A, y_A)$ and
$B(x_B, y_B)$, with $x_A \neq x_B$, is
\[
m = \frac{y_B - y_A}{x_B - x_A}
\qquad
\text{(``change in $y$ over change in $x$'').}
\]
\end{proposition}

\begin{proof}
Both points satisfy $y = mx + p$, so $y_B - y_A = (mx_B + p) - (mx_A + p)
= m(x_B - x_A)$; divide by $x_B - x_A \neq 0$.
\end{proof}

\begin{method}[Finding the equation of a line through two points]\label{met:g10:lines:through}
Given $A$ and $B$ with $x_A \neq x_B$:
\begin{enumerate}
  \item compute the \omterm{def:g10:reffunc:affine}{slope} $m = \dfrac{y_B - y_A}{x_B - x_A}$;
  \item write $y = mx + p$ and substitute the \omterm{def:g10:coordgeom:system}{coordinates} of $A$ (or $B$)
        to find $p$;
  \item check the final \omterm{def:g10:algebra:equation}{equation} with the \emph{other} point.
\end{enumerate}
If $x_A = x_B$, the line is vertical: its \omterm{def:g10:algebra:equation}{equation} is $x = x_A$.
\end{method}

\begin{example}\label{ex:g10:lines:through}
Line through $A(1, 5)$ and $B(3, 1)$. \omterm{def:g10:reffunc:affine}{Slope}:
$m = \frac{1 - 5}{3 - 1} = \frac{-4}{2} = -2$. Then $y = -2x + p$; the
point $A$ gives $5 = -2 \times 1 + p$, so $p = 7$. \omterm{def:g10:algebra:equation}{Equation}:
$y = -2x + 7$. Check with $B$: $-2 \times 3 + 7 = 1$.
\end{example}

\begin{proposition}[Parallel lines]\label{prop:g10:lines:parallel}
Two non-vertical lines are parallel if and only if they have the same
\omterm{def:g10:reffunc:affine}{slope}. Two distinct parallel lines have no common point; two lines with
different \omterm{def:g10:reffunc:affine}{slopes} have exactly one.
\end{proposition}

\begin{proof}
Let $y = mx + p$ and $y = m'x + p'$ be the two lines. A common point
solves $mx + p = m'x + p'$, i.e.\ $(m - m')x = p' - p$. If $m \neq m'$,
this has exactly one solution $x = \frac{p' - p}{m - m'}$: one
\omterm{def:g10:numbers:interunion}{intersection} point. If $m = m'$, the \omterm{def:g10:algebra:equation}{equation} reads $0 = p' - p$: no
solution when $p \neq p'$ (distinct parallels), every $x$ when $p = p'$
(same line). Parallelism (having no \omterm{def:g10:numbers:interunion}{intersection}, or being equal) is thus
equivalent to $m = m'$.
\end{proof}

\begin{remark}[Direction vector]
The line $y = mx + p$ goes up $m$ units for each unit to the right, so the
\omterm{def:g10:vectors:vector}{vector} $\vec u\,(1, m)$ is a \emph{direction
vector}\index{direction vector} of the line: the line is parallel to
$\vec u$. Two lines are parallel exactly when their direction \omterm{def:g10:vectors:vector}{vectors} are
\omterm{def:g10:vectors:collinear}{collinear} (\cref{thm:g10:vectors:det}).
\end{remark}

\section{Linear systems}

\begin{definition}[Linear system]\label{def:g10:lines:system}
A \emph{system of two linear equations}\index{linear system} in the
unknowns $x$ and $y$ is a pair of \omterm{def:g10:algebra:equation}{equations}
\[
\begin{cases}
a x + b y = c \\
a' x + b' y = c'
\end{cases}
\]
to be satisfied \emph{simultaneously}. A \emph{solution} is a pair
$(x, y)$ satisfying both. Each \omterm{def:g10:algebra:equation}{equation} describes a line, so solving the
system means finding the \omterm{def:g10:numbers:interunion}{intersection} of two lines.
\end{definition}

\begin{method}[Substitution]\label{met:g10:lines:substitution}
\begin{enumerate}
  \item Solve one \omterm{def:g10:algebra:equation}{equation} for one unknown --- pick the easiest, e.g.\
        express $y$ in terms of $x$;
  \item substitute this expression into the \emph{other} \omterm{def:g10:algebra:equation}{equation}, which
        now contains only $x$;
  \item solve for $x$, then compute $y$ from step 1;
  \item check the pair in both original \omterm{def:g10:algebra:equation}{equations}.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g10:lines:substitution}
Solve $\begin{cases} x + 2y = 8 \\ 3x - y = 3 \end{cases}$ by
substitution. The second \omterm{def:g10:algebra:equation}{equation} gives $y = 3x - 3$. Substitute into the
first:
\[
x + 2(3x - 3) = 8
\ \Longleftrightarrow\
x + 6x - 6 = 8
\ \Longleftrightarrow\
7x = 14
\ \Longleftrightarrow\
x = 2 ,
\]
then $y = 3 \times 2 - 3 = 3$. The solution is the pair $(2, 3)$.
Check: $2 + 2 \times 3 = 8$ and $3 \times 2 - 3 = 3$.
\end{example}

\begin{method}[Elimination]\label{met:g10:lines:elimination}
\begin{enumerate}
  \item Multiply each \omterm{def:g10:algebra:equation}{equation} by a well-chosen number so that one unknown
        gets opposite coefficients in the two \omterm{def:g10:algebra:equation}{equations};
  \item add the \omterm{def:g10:algebra:equation}{equations}: that unknown disappears;
  \item solve the remaining one-unknown \omterm{def:g10:algebra:equation}{equation}, then substitute back to
        find the other unknown;
  \item check the pair in both original \omterm{def:g10:algebra:equation}{equations}.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g10:lines:elimination}
Solve $\begin{cases} 2x + 3y = 7 \\ 5x - 2y = 8 \end{cases}$ by
elimination. To eliminate $y$, multiply the first \omterm{def:g10:algebra:equation}{equation} by $2$ and the
second by $3$:
\[
\begin{cases}
4x + 6y = 14 \\
15x - 6y = 24
\end{cases}
\]
Adding them: $19x = 38$, so $x = 2$. Substituting into $2x + 3y = 7$:
$4 + 3y = 7$, so $y = 1$. Solution: $(2, 1)$. Check in the second
\omterm{def:g10:algebra:equation}{equation}: $5 \times 2 - 2 \times 1 = 8$.
\end{example}

\begin{remark}[How many solutions?]
Like two lines, a \omterm{def:g10:lines:system}{linear system} has exactly one solution in general
(non-parallel lines), and exceptionally none (distinct parallel lines) or
infinitely many (twice the same line). The criterion of
\cref{thm:g10:vectors:det} decides: the system has a unique solution
exactly when $ab' - a'b \neq 0$.
\end{remark}

\begin{example}[Modeling with a system]\label{ex:g10:lines:modeling}
Three notebooks and two pens cost $8$; one notebook and four pens cost
$6$. Let $x$ be the price of a notebook and $y$ that of a pen:
\[
\begin{cases}
3x + 2y = 8 \\
x + 4y = 6 .
\end{cases}
\]
From the second \omterm{def:g10:algebra:equation}{equation}, $x = 6 - 4y$; substituting,
$3(6 - 4y) + 2y = 8$, so $18 - 10y = 8$, giving $y = 1$ and then
$x = 2$. A notebook costs $2$ and a pen costs $1$.
\end{example}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:g10:lines:1}
For each line, read off the \omterm{def:g10:reffunc:affine}{slope} and the $y$-intercept, and sketch it:
\[
y = 2x - 3, \qquad
y = -\tfrac13 x + 1, \qquad
y = 4, \qquad
x = -2 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:lines:2}
Does the point $A(2, 7)$ belong to the line $y = 3x + 1$? And $B(-1, -1)$?
And the point $C(4, 13)$?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:lines:3}
Find the \omterm{thm:g10:lines:reduced}{reduced equation} of the line through:
\[
\text{(a) } A(0, 2) \text{ and } B(3, 8);
\qquad
\text{(b) } A(-1, 5) \text{ and } B(2, -4);
\qquad
\text{(c) } A(4, 1) \text{ and } B(4, 6).
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:lines:4}
Among the lines $y = 3x - 1$, $y = -3x + 2$, $y = 3x + 5$,
$y = \frac{6x + 2}{2}$, which are parallel to each other? Which are
actually equal?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:lines:5}
Solve by substitution:
\[
\begin{cases}
y = 2x - 1 \\
3x + y = 9
\end{cases}
\qquad\text{then}\qquad
\begin{cases}
x - y = 4 \\
2x + 3y = 3 .
\end{cases}
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g10:lines:6}
Solve by elimination:
\[
\begin{cases}
x + y = 10 \\
x - y = 4
\end{cases}
\qquad\text{then}\qquad
\begin{cases}
3x + 2y = 12 \\
2x + 5y = 8 .
\end{cases}
\]
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:lines:7}
Determine, without solving them, how many solutions each system has:
\[
\begin{cases}
2x - y = 3 \\
-4x + 2y = -6
\end{cases}
\qquad
\begin{cases}
2x - y = 3 \\
-4x + 2y = 1
\end{cases}
\qquad
\begin{cases}
2x - y = 3 \\
x + y = 0 .
\end{cases}
\]
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:lines:8}
Two hundred tickets were sold for a school play, some at $5$ (children)
and some at $9$ (adults), for a total of $1352$. How many tickets of each
kind were sold?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g10:lines:9}
Let $d$ be the line $y = 2x - 3$ and $A(1, 4)$.
\begin{enumerate}
  \item Give the \omterm{def:g10:algebra:equation}{equation} of the line $d'$ parallel to $d$ passing
        through $A$.
  \item Compute the \omterm{def:g10:numbers:interunion}{intersection} point of $d'$ with the $x$-axis.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g10:lines:10}
Let $A(0, 0)$, $B(6, 2)$ and $C(2, 4)$.
\begin{enumerate}
  \item Find the \omterm{def:g10:algebra:equation}{equations} of the medians of the triangle $ABC$ issued
        from $A$ and from $B$ (a median joins a vertex to the \omterm{prop:g10:coordgeom:midpoint}{midpoint} of
        the opposite side).
  \item Compute their \omterm{def:g10:numbers:interunion}{intersection} point $G$, and check that
        $G$ is also on the third median. (You should find that the
        \omterm{def:g10:coordgeom:system}{coordinates} of $G$ are the averages of those of $A$, $B$, $C$.)
\end{enumerate}
\end{exercise}

\section{Problem: One solution, none, or infinitely many}

\begin{problem}[{Weekend problem --- two lines have three possible
destinies, a determinant decides between them, and the world's
oldest textbook already knew}]
\label{pb:g10:lines:1}
Two linear \omterm{def:g10:algebra:equation}{equations}, two unknowns: the pair of lines they draw
can cross once, never, or be one and the same line --- and
\emph{every} \omterm{def:g10:lines:system}{linear system} inherits one of these three destinies.
This problem classifies them, meets the number that decides (an
old acquaintance from the \omterm{def:g10:vectors:vector}{vectors} chapter), solves the
two-thousand-year-old pheasants-and-rabbits of the Chinese
\emph{Nine Chapters}, and ends with corners of regions --- the
first step of the optimization used by every airline and factory
today.

\medskip
\noindent\textbf{Part I --- The line, fluently.}
\begin{enumerate}
  \item Find the \omterm{thm:g10:lines:reduced}{reduced equation} of the line through
        $(1, 2)$ and $(3, 8)$
        (\cref{met:g10:lines:through}).
  \item Among $y = 3x - 1$, $y = 3x + 4$ and $y = -x + 7$: which
        two lines are parallel
        (\cref{prop:g10:lines:parallel}), and where do the
        non-parallel pairs meet? (Compute one \omterm{def:g10:numbers:interunion}{intersection}.)
  \item The line through $(2, 1)$ and $(2, 5)$: why does it have
        no \omterm{thm:g10:lines:reduced}{reduced equation} $y = mx + p$, and what is its
        \omterm{def:g10:algebra:equation}{equation}?
  \item Is the point $(4, 11)$ on the line $y = 3x - 1$? And
        $(-1, -5)$?
  \item The line through $(0, 3)$ with \omterm{def:g10:reffunc:affine}{slope} $-2$: give its
        \omterm{def:g10:algebra:equation}{equation}, its two axis intercepts, and the area of the
        triangle it cuts from the first quadrant.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Three destinies.}
\begin{enumerate}[resume]
  \item Solve by substitution
        (\cref{met:g10:lines:substitution}):
        $\begin{cases} y = 2x - 3 \\ 3x + y = 7 \end{cases}$
  \item Solve by elimination
        (\cref{met:g10:lines:elimination}):
        $\begin{cases} 2x + 3y = 8 \\ 5x - 3y = -1 \end{cases}$
  \item Classify --- and interpret with lines --- the two
        systems
        \[
        \begin{cases} 2x - y = 3 \\ 4x - 2y = 6 \end{cases}
        \qquad\text{and}\qquad
        \begin{cases} 2x - y = 3 \\ 4x - 2y = 1 . \end{cases}
        \]
        State the complete trichotomy: crossing, parallel,
        identical --- one solution, none, infinitely many.
  \item For the general system $ax + by = e$, $cx + dy = f$,
        the deciding number is $ad - bc$ --- the same
        determinant as the collinearity test of
        \cref{pb:g10:vectors:1}. Explain the coincidence (which
        two \omterm{def:g10:vectors:vector}{vectors} are \omterm{def:g10:vectors:collinear}{collinear} exactly when the lines are
        parallel or identical?), and evaluate $ad - bc$ for the
        three systems of questions 7 and 8.
  \item For which value of $m$ does
        $\begin{cases} x + my = 3 \\ 2x + 6y = 7 \end{cases}$
        have no solution? Solve the system for all other values
        of $m$\,\dots\ or at least explain how you would.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The Nine Chapters.}
\begin{enumerate}[resume]
  \item From the \emph{Nine Chapters on the Mathematical Art}
        (China, about 2000 years ago): a cage holds pheasants
        and rabbits --- $35$ heads and $94$ legs in all. How
        many of each?
  \item A juice stand blends a $60\,\%$-fruit syrup with a
        $20\,\%$-fruit drink to obtain $30$ L at $35\,\%$
        fruit. How many liters of each?
  \item A two-digit number has digit sum $12$; swapping its
        digits decreases it by $18$. Find it, using the digit
        algebra of the digit-algebra weekend problem of the Middle School volume to set up the system.
  \item At the bakery, $3$ coffees and $2$ croissants cost
        $9.10$ euros; $2$ coffees and $3$ croissants cost
        $8.90$ euros. Solve elegantly: what do \emph{adding} the
        two \omterm{def:g10:algebra:equation}{equations} and \emph{subtracting} them each tell you
        directly?
  \item A market stall claims: $2$ apples $+ 1$ pear for
        $5$ euros, and $4$ apples $+ 2$ pears for $9$ euros.
        Solve --- or rather, explain what the system's destiny
        reveals about the stall's arithmetic.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Corners rule.}
\begin{enumerate}[resume]
  \item The line $y = 3x - 1$ splits the plane in two. Which
        side is the origin on? Describe the set
        $y > 3x - 1$ and how one tests a point against it.
  \item Draw the region defined by the four constraints
        \[
        x \geq 0, \qquad y \geq 0, \qquad x + y \leq 4,
        \qquad y \leq 2x + 1,
        \]
        and compute its four corner points.
  \item A workshop's profit is $P = 3x + 2y$ on the region of
        question 17. Evaluate $P$ at the four corners. Admitting
        that the \omterm{def:g10:functions:extrema}{maximum} of such a linear quantity on such a
        region is always reached at a corner (picture the lines
        $3x + 2y = $ constant sweeping across), give the optimal
        production plan.
  \item The two taxis of the two-thermometers weekend problem of the Middle School volume as a system:
        write and solve
        $\begin{cases} y = 1.5x + 3 \\ y = 2x \end{cases}$ and
        interpret the solution.
  \item Finale, the dictionary: one \omterm{def:g10:algebra:equation}{equation} $\leftrightarrow$
        one line; one system $\leftrightarrow$ two lines with
        three destinies, decided by $ad - bc$; many inequalities
        $\leftrightarrow$ a region whose corners carry the
        optimum. State it in three sentences --- you have just
        toured the entrance hall of \emph{linear algebra} and
        \emph{linear programming}, both built in full in the
        university volumes.
\end{enumerate}
\end{problem}
