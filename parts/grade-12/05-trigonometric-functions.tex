\chapter{Trigonometric Functions}\label{ch:g12:trigo}

The functions $\cos$ and $\sin$ turn the geometry of the circle into
analysis. Defined by winding the real line around the unit circle, they are
the archetype of periodic functions, and their derivatives make them
solutions of the equation of oscillations $y'' = -y$.

\section{The unit circle}

\begin{definition}[Cosine and sine]\label{def:g12:trigo:cossin}
Let $\mathcal{C}$ be the circle of radius 1 centered at the origin of an
orthonormal coordinate system. To each real $t$, associate the point $M(t)$
obtained by winding a length $\abs{t}$ along $\mathcal{C}$ from the point
$I(1,0)$, counterclockwise if $t \geq 0$ and clockwise otherwise. Then
\emph{$\cos t$ and $\sin t$ are the coordinates of
$M(t)$}\index{cosine}\index{sine}:
\[
M(t) = (\cos t, \sin t).
\]
\end{definition}

Since the circle has circumference $2\pi$, the point $M(t)$ is unchanged when
$t$ increases by $2\pi$; and since $M(t)$ lies on the unit circle:

\begin{proposition}[Fundamental identities]\label{prop:g12:trigo:identities}
For all $t \in \R$ and $k \in \Z$:
\[
\cos^2 t + \sin^2 t = 1, \qquad
\cos(t + 2k\pi) = \cos t, \qquad
\sin(t + 2k\pi) = \sin t,
\]
\[
\cos(-t) = \cos t, \qquad \sin(-t) = -\sin t,
\]
\[
\cos(\pi - t) = -\cos t, \quad \sin(\pi - t) = \sin t, \quad
\cos\left(\tfrac{\pi}{2} - t\right) = \sin t, \quad
\sin\left(\tfrac{\pi}{2} - t\right) = \cos t .
\]
\end{proposition}

\begin{proof}
The first identity is the equation of the unit circle; the others express the
symmetries of the circle: $M(-t)$ is the reflection of $M(t)$ in the
$x$-axis, $M(\pi - t)$ in the $y$-axis, and $M(\frac\pi2 - t)$ in the line
$y = x$.
\end{proof}

$\sin t$ & $0$ & $\dfrac12$ & $\dfrac{\sqrt2}{2}$ & $\dfrac{\sqrt3}{2}$ & $1$

\begin{proposition}[Addition formulas]\label{prop:g12:trigo:addition}
For all $a, b \in \R$:
\begin{align*}
\cos(a+b) &= \cos a \cos b - \sin a \sin b, &
\sin(a+b) &= \sin a \cos b + \cos a \sin b,\\
\cos(a-b) &= \cos a \cos b + \sin a \sin b, &
\sin(a-b) &= \sin a \cos b - \cos a \sin b.
\end{align*}
In particular $\cos 2a = 2\cos^2 a - 1 = 1 - 2\sin^2 a$ and
$\sin 2a = 2 \sin a \cos a$.
\end{proposition}

\begin{proof}
The vectors $\vect{u} = (\cos a, \sin a)$ and $\vect{v} = (\cos b, \sin b)$
are unit vectors making an angle $a - b$; their scalar product equals both
$\cos a\cos b + \sin a \sin b$ (coordinates) and
$\norm{\vect u}\,\norm{\vect v}\cos(a-b) = \cos(a - b)$ (geometry), which is
the third formula. Replacing $b$ by $-b$ gives the first; the sine formulas
follow using $\sin x = \cos(\frac\pi2 - x)$. The duplication formulas are the
case $b = a$ combined with $\cos^2 + \sin^2 = 1$.
\end{proof}

\section{Analytic study}

\begin{lemma}[Fundamental limit]\label{lem:g12:trigo:sinxx}
\[
\lim_{t \to 0} \frac{\sin t}{t} = 1
\qquad\text{and}\qquad
\lim_{t \to 0} \frac{\cos t - 1}{t} = 0 .
\]
\end{lemma}

\begin{proof}
For $0 < t < \frac{\pi}{2}$, compare three areas in the unit circle: the
triangle $OIM(t)$, the circular sector $OIM(t)$, and the right triangle with
base $OI$ and height $\tan t$:
\[
\frac{\sin t}{2} \leq \frac{t}{2} \leq \frac{\tan t}{2}.
\]

The first inequality gives $\frac{\sin t}{t} \leq 1$; the second gives
$\cos t \leq \frac{\sin t}{t}$. As $t \to 0^+$, $\cos t \to 1$ (continuity),
and the squeeze theorem yields $\frac{\sin t}{t} \to 1$; parity extends this
to $t \to 0$. For the second limit,
\[
\frac{\cos t - 1}{t} = \frac{\cos^2 t - 1}{t(\cos t + 1)}
= -\frac{\sin t}{t}\cdot\frac{\sin t}{\cos t + 1}
\xrightarrow[t\to0]{} -1 \cdot \frac{0}{2} = 0. \qedhere
\]
\end{proof}

\begin{theorem}[Derivatives of sine and cosine]\label{thm:g12:trigo:derivatives}
The functions $\sin$ and $\cos$ are differentiable on $\R$, with
\[
(\sin)' = \cos, \qquad (\cos)' = -\sin .
\]
More generally, $(\sin(ax+b))' = a\cos(ax+b)$ and
$(\cos(ax+b))' = -a\sin(ax+b)$.
\end{theorem}

\begin{proof}
By the addition formulas,
\[
\frac{\sin(x+h) - \sin x}{h}
= \sin x\,\frac{\cos h - 1}{h} + \cos x\,\frac{\sin h}{h}
\xrightarrow[h\to0]{} \sin x \cdot 0 + \cos x \cdot 1 = \cos x,
\]
using \cref{lem:g12:trigo:sinxx}. The computation for $\cos$ is identical,
and the general form follows from the chain rule.
\end{proof}

\begin{proposition}[Variations]\label{prop:g12:trigo:variations}
On $\intcc{0}{\pi}$, $\cos$ decreases from $1$ to $-1$. On
$\intcc{-\frac\pi2}{\frac\pi2}$, $\sin$ increases from $-1$ to $1$. Both
functions are $2\pi$-periodic and have ranges $\intcc{-1}{1}$; $\cos$ is
even, $\sin$ is odd.
\end{proposition}

\begin{proof}
On $\intoo{0}{\pi}$, $(\cos)' = -\sin < 0$ since the point $M(t)$ has
positive ordinate there; similarly $(\sin)' = \cos > 0$ on
$\intoo{-\frac\pi2}{\frac\pi2}$. Periodicity and parity come from
\cref{prop:g12:trigo:identities}.
\end{proof}

\begin{method}[Solving $\cos x = a$ and $\sin x = a$]\label{met:g12:trigo:equations}
For $a \in \intcc{-1}{1}$, find one particular solution $\alpha$ (from the
table of values or a calculator). Then all solutions are:
\[
\cos x = \cos\alpha \iff x = \alpha + 2k\pi \text{ or } x = -\alpha + 2k\pi,
\quad k \in \Z;
\]
\[
\sin x = \sin\alpha \iff x = \alpha + 2k\pi \text{ or } x = \pi - \alpha + 2k\pi,
\quad k \in \Z.
\]
For inequalities, locate the solution arcs on the unit circle and read off
the intervals within one period.
\end{method}

\begin{example}\label{ex:g12:trigo:equation}
Solve $\cos x = \frac12$ in $\intoc{-\pi}{\pi}$: a particular solution is
$\frac\pi3$, so the solutions are $x = \frac\pi3$ and $x = -\frac\pi3$.
In all of $\R$: $x = \pm\frac\pi3 + 2k\pi$, $k \in \Z$.
\end{example}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:g12:trigo:1}
Compute $\cos\frac{2\pi}{3}$, $\sin\frac{5\pi}{6}$, $\cos\frac{7\pi}{4}$ and
$\sin\left(-\frac{\pi}{3}\right)$ using the identities of
\cref{prop:g12:trigo:identities}.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:trigo:2}
Differentiate $f(x) = \sin^2 x$, $g(x) = \cos(3x) + x\sin x$ and
$h(x) = \dfrac{\sin x}{2 + \cos x}$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:trigo:3}
Solve in $\intoc{-\pi}{\pi}$, then in $\R$:
\[
\text{(a) } \sin x = \frac{\sqrt3}{2};
\qquad
\text{(b) } \cos\left(2x\right) = \frac{\sqrt2}{2} .
\]
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:trigo:4}
Using the addition formulas, compute the exact value of
$\cos\frac{\pi}{12}$ and $\sin\frac{\pi}{12}$.
(Hint: $\frac{\pi}{12} = \frac{\pi}{3} - \frac{\pi}{4}$.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:trigo:5}
Solve in $\intcc{0}{2\pi}$ the inequality $2\sin^2 x - \sin x - 1 \geq 0$.
(Hint: factor the quadratic in $\sin x$.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:trigo:6}
Compute the limits
\[
\lim_{x\to0} \frac{\sin 3x}{x}, \qquad
\lim_{x\to0} \frac{1 - \cos x}{x^2}, \qquad
\lim_{x\to+\infty} x \sin\frac{1}{x}.
\]
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:trigo:7}
Study the function $f(x) = x + \cos x$ on $\intcc{0}{2\pi}$: variations and
extrema. Show that $f$ is increasing on $\R$ although $f'$ vanishes at
infinitely many points.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:trigo:8}
Let $f(x) = A\cos x + B \sin x$ with $(A,B) \neq (0,0)$.
\begin{enumerate}
  \item Show that $f$ can be written $f(x) = R\cos(x - \varphi)$ with
        $R = \sqrt{A^2 + B^2}$ and a suitable $\varphi$.
  \item Deduce the maximum and minimum of $f$, and solve
        $\cos x + \sin x = 1$ in $\intoc{-\pi}{\pi}$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:trigo:9}
Show that $\sin$ and $\cos$ both satisfy the differential equation
$y'' = -y$. Conversely, let $y$ be a solution of $y'' = -y$ with
$y(0) = a$ and $y'(0) = b$; show that $y = a\cos + b\sin$.
(Hint: consider $g = (y - a\cos - b\sin)$ and the ``energy''
$E = g^2 + (g')^2$.)
\end{exercise}
