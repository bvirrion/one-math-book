\section*{Hoofdstuk \ref{ch:g12:contdist} --- Continue stochastische variabelen}

\begin{solution}{exo:g12:contdist:1}
$\P(X \leq 5) = \frac{5}{15} = \frac13$;
$\P(4 \leq X \leq 10) = \frac{6}{15} = \frac25$;
$\E(X) = \frac{15}{2} = 7.5$ min;
$\sigma(X) = \frac{15}{\sqrt{12}} = \frac{5\sqrt3}{2} \approx 4.3$ min.
\end{solution}

\begin{solution}{exo:g12:contdist:2}
$f \geq 0$ en
$\int_0^2 \frac38 t^2\,\dd t = \frac38\left[\frac{t^3}{3}\right]_0^2
= \frac38 \times \frac83 = 1$: een \omterm{def:g12:contdist:density}{dichtheid}.
\[
\E(X) = \int_0^2 \frac38 t^3\,\dd t = \frac38 \times \frac{16}{4} = \frac32,
\qquad
\P(X \geq 1) = \int_1^2 \frac38 t^2\,\dd t = \frac{8 - 1}{8} = \frac78 .
\]
\end{solution}

\begin{solution}{exo:g12:contdist:3}
\emph{1.} $\E(X) = \frac{1}{0.2} = 5$ jaar;
$\P(X > 5) = \eu^{-0.2\times5} = \eu^{-1} \approx 0.37$.

\emph{2.} Door geheugenloosheid (\cref{thm:g12:contdist:memoryless}),
$\pcond{X > 3}{X > 8} = \P(X > 5) = \eu^{-1} \approx 0.37$: de drie jaar
dienst veranderen niets.
\end{solution}

\begin{solution}{exo:g12:contdist:4}
$\P(X > m) = \eu^{-\lambda m} = \frac12$ geeft
$m = \frac{\ln 2}{\lambda} \approx \frac{0.69}{\lambda}$, kleiner dan
$\E(X) = \frac1\lambda$. De exponentiële \omterm{def:g12:contdist:density}{dichtheid} heeft een lange
rechterstaart: enkele atypisch lange levensduren trekken het
\emph{\omterm{def:g11:stat:mean}{gemiddelde}} boven de \emph{\omterm{def:g11:stat:median}{mediaan}}, de waarde die de helft van de
populatie overschrijdt. (Dit is de halfwaardetijd van
\cref{ex:g12:exp:halflife}: de helft van de atomen overleeft haar, hoewel
de \omterm{def:g11:stat:mean}{gemiddelde} levensduur langer is.)
\end{solution}

\begin{solution}{exo:g12:contdist:5}
$168 = \mu - \sigma$ en $182 = \mu + \sigma$: ongeveer $68\%$.
$189 = \mu + 2\sigma$: erboven ligt de helft van de resterende
$100 - 95.4 = 4.6\%$, dus ongeveer $2.3\%$.
$154 = \mu - 3\sigma$: ongeveer $\frac{100 - 99.7}{2} = 0.15\%$.
\end{solution}

\begin{solution}{exo:g12:contdist:6}
$\P(X < 500) \leq 2.5\%$ betekent dat $500$ minstens twee
\omterm{def:g11:stat:variance}{standaardafwijkingen} onder het \omterm{def:g11:stat:mean}{gemiddelde} moet liggen (de normaal laat
$2.3\% \approx 2.5\%$ onder $\mu - 2\sigma$; met de conventionele $1.96$
is de redenering identiek):
\[
\mu - 2\sigma \geq 500 \iff \mu \geq 500 + 8 = 508 \text{ g}.
\]
De machine moet op gemiddeld $508$ g worden ingesteld --- de prijs van de
garantie is $8$ g ``gratis'' product per zak.
\end{solution}

\begin{solution}{exo:g12:contdist:7}
\emph{1.} Met $p = \frac16$, $n = 1200$:
$\sqrt{\frac{p(1-p)}{n}} = \sqrt{\frac{5/36}{1200}} \approx 0.0108$, dus
het $95\%$-fluctuatie-interval is
\[
\frac16 \pm 1.96 \times 0.0108 \approx \intcc{0.146}{0.188}.
\]

\emph{2.} De waargenomen \omterm{def:g10:stats:series}{frequentie} $\frac{260}{1200} \approx 0.217$ ligt
ver buiten: de eerlijkheidshypothese wordt verworpen op het $5\%$-niveau.
De afwijking is ongeveer $4.6$ \omterm{def:g11:stat:variance}{standaardafwijkingen} --- voor een
normaalbenadering een \omterm{def:g11:prob:model}{kans} van orde $10^{-6}$, veel conclusiever dan de
$\leq 4.6\%$-grens van Bienaymé--Tsjebysjev
(\cref{exo:g12:sums:7}).
\end{solution}

\begin{solution}{exo:g12:contdist:8}
\emph{1.} Marge $\frac{1}{\sqrt n} \leq 0.02$ eist
$n \geq 2500$ personen.

\emph{2.} Om scores $1$ punt uit elkaar te onderscheiden, moet de marge
goed onder $0.5$ punt liggen, wat $n \geq \left(\frac{1}{0.005}\right)^2 =
40\,000$ eist volgens de vereenvoudigde formule --- al onpraktisch voor de
meeste peilingen. Bovendien rekent de statistische marge alleen
\emph{steekproeffout}; systematische biases (niet-representatieve
steekproeven, non-respons, last-minute swings) krimpen niet als $n$ groeit
en overschrijden typisch één punt. Een race van $1$ punt is echt te
dicht om te roepen.
\end{solution}

\begin{solution}{exo:g12:contdist:9}
\emph{1.} Omdat $\alpha > 0$, $c \leq \alpha X + \beta \leq d
\iff \frac{c - \beta}{\alpha} \leq X \leq \frac{d - \beta}{\alpha}$, dus
\[
\P(c \leq Y \leq d)
= \int_{(c-\beta)/\alpha}^{(d-\beta)/\alpha} f(t)\,\dd t .
\]
De substitutie $y = \alpha t + \beta$ (d.w.z.\ de oppervlakte lezen in de
$y$-variabele, $t = \frac{y - \beta}{\alpha}$, $\dd t = \frac{\dd y}{\alpha}$)
maakt dit tot
$\int_c^d \frac{1}{\alpha} f\left(\frac{y-\beta}{\alpha}\right)\dd y$: de
\omterm{def:g11:func:function}{functie} $g(y) = \frac1\alpha f\left(\frac{y-\beta}{\alpha}\right)$ is een
\omterm{def:g12:contdist:density}{dichtheid} van $Y$.

\emph{2.} Met $f = \varphi$, $\alpha = \sigma$, $\beta = \mu$:
\[
g(y) = \frac{1}{\sigma}\,\varphi\!\left(\frac{y - \mu}{\sigma}\right)
= \frac{1}{\sigma\sqrt{2\pi}}\,
\exp\left(-\frac{(y-\mu)^2}{2\sigma^2}\right),
\]
wat daarom de \omterm{def:g12:contdist:density}{dichtheid} is van
$\mathcal N(\mu, \sigma^2) = \mu + \sigma\,\mathcal N(0,1)$.
\end{solution}
