\section*{Hoofdstuk \ref{ch:g12:condprob} --- Voorwaardelijke kans en onafhankelijkheid}

\begin{solution}{exo:g12:condprob:1}
Er zijn $12$ plaatjes, waarvan $4$ heren:
$\pcond{\text{plaatje}}{\text{heer}} = \frac{4}{12} = \frac13$.

Onafhankelijkheid: $\P(\text{heer} \cap \text{harten}) = \frac{1}{52}$
(de hartenheer), en
$\P(\text{heer})\,\P(\text{harten}) = \frac{4}{52}\times\frac{13}{52}
= \frac{1}{52}$. Gelijk: de gebeurtenissen zijn onafhankelijk.
\end{solution}

\begin{solution}{exo:g12:condprob:2}
\emph{1.} Eerste tak: rood $\frac58$, blauw $\frac38$; tweede takken
(zonder teruglegging): na rood, rood $\frac47$ / blauw $\frac37$; na
blauw, rood $\frac57$ / blauw $\frac27$.

\emph{2.} $\P(RR) = \frac58 \times \frac47 = \frac{5}{14}$. Met totale
kans,
\[
\P(\text{2e rood}) = \frac58\cdot\frac47 + \frac38\cdot\frac57
= \frac{20 + 15}{56} = \frac58 .
\]
(Dezelfde als de eerste trekking --- door symmetrie is de tweede bal een
uniform willekeurige bal van de urn.)
\end{solution}

\begin{solution}{exo:g12:condprob:3}
$\P(A) = \frac{6}{36} = \frac16$ (zes paren sommeren tot $7$),
$\P(B) = \frac16$, en $A \cap B = \{(3,4)\}$ heeft kans
$\frac{1}{36} = \P(A)\P(B)$: $A$ en $B$ zijn onafhankelijk. (De som $7$ is
bijzonder: wat de eerste steen ook toont, precies één waarde van de tweede
geeft hem.)

$\P(C) = \frac{5}{36}$ en $B \cap C = \{(3,3)\}$:
$\P(B \cap C) = \frac{1}{36} \neq \frac16 \times \frac{5}{36}
= \frac{5}{216}$. Niet onafhankelijk.
\end{solution}

\begin{solution}{exo:g12:condprob:4}
\emph{1.} Totale kans met de partitie per machine:
\[
\P(D) = 0.5\times0.01 + 0.3\times0.02 + 0.2\times0.04
= 0.005 + 0.006 + 0.008 = 0.019 = 1.9\% .
\]

\emph{2.} Bayes:
$\pcond{D}{M_3} = \dfrac{0.008}{0.019} = \dfrac{8}{19} \approx 0.42$. De
machine die slechts een vijfde van de output produceert, is verantwoordelijk
voor meer dan $40\%$ van de defecten.
\end{solution}

\begin{solution}{exo:g12:condprob:5}
De voorwaarde is
\[
\frac{0.99\,p}{0.99\,p + 0.05\,(1-p)} \geq 0.9 .
\]
De noemer is positief, dus dit leest
$0.99p \geq 0.891p + 0.045 - 0.045p$, \emph{d.w.z.}\ $0.144\,p \geq 0.045$,
\emph{d.w.z.}\
\[
p \geq \frac{0.045}{0.144} = 0.3125 .
\]
De test alleen bereikt $90\%$ zekerheid alleen als de ziekte al meer dan
$31\%$ van de geteste populatie treft --- daarom vereisen massascreenings
van zeldzame ziekten bevestigingstests.
\end{solution}

\begin{solution}{exo:g12:condprob:6}
\emph{1.} Drie reeksen realiseren $E$ (KKM, KMK, MKK), elk van kans
$p^2(1-p)$ door onafhankelijkheid: $\P(E) = 3p^2(1 - p)$.

\emph{2.} $f(p) = 3p^2 - 3p^3$ heeft $f'(p) = 6p - 9p^2 = 3p(2 - 3p)$,
positief dan negatief op $\intoo{0}{1}$: maximum bij $p = \frac23$, waar
$\P(E) = 3\cdot\frac49\cdot\frac13 = \frac49$.
\end{solution}

\begin{solution}{exo:g12:condprob:7}
Stel u koos deur 1 en de gastheer opende deur 3 (gebeurtenis $H_3$). Met
$B_i$ = ``prijs achter deur $i$'', $\P(B_i) = \frac13$ en
\[
\pcond{B_1}{H_3} = \frac12, \qquad
\pcond{B_2}{H_3} = 1, \qquad
\pcond{B_3}{H_3} = 0
\]
(als de prijs achter uw deur is, kiest de gastheer willekeurig tussen deuren
$2$ en $3$; als zij achter deur $2$ is, is hij gedwongen $3$ te openen).
Bayes:
\[
\pcond{H_3}{B_1}
= \frac{\frac13\cdot\frac12}{\frac13\cdot\frac12 + \frac13\cdot1 + 0}
= \frac{1/6}{1/2} = \frac13,
\qquad
\pcond{H_3}{B_2} = \frac{\frac13}{\frac12} = \frac23 .
\]
Blijven wint met kans $\frac13$, wisselen met kans $\frac23$.
\end{solution}

\begin{solution}{exo:g12:condprob:8}
\emph{1.} De meerderheid is fout wanneer minstens twee van de drie kopieën
omgeklapt zijn:
\[
3\varepsilon^2(1-\varepsilon) + \varepsilon^3
= 3(0.01)(0.9) + 0.001 = 0.028,
\]
veel kleiner dan $\varepsilon = 0.1$: herhalingscodering werkt.

\emph{2.} Als $1$ werd verzonden (als $111$), vereist ontvangen $101$
precies één flip: kans $\varepsilon(1-\varepsilon)^2 = 0.081$. Als $0$
werd verzonden ($000$), zijn twee flips nodig:
$\varepsilon^2(1-\varepsilon) = 0.009$. Bayes met gelijke priors:
\[
\pcond{101}{\,\text{verzonden }1} = \frac{0.081}{0.081 + 0.009} = 0.9 .
\]
De meerderheidsdecodering ($101 \mapsto 1$) is inderdaad de meer
waarschijnlijke gok.
\end{solution}
