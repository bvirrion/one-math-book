\section*{Hoofdstuk \ref{ch:g12:arith} --- Rekenkunde}

\begin{solution}{exo:g12:arith:1}
$17 \times 119 = 2023$, dus $2026 = 17 \times 119 + 3$: quotiënt $119$,
rest $3$. Voor $-2026$: $-2026 = 17\times(-120) + 14$ (inderdaad
$17 \times 120 = 2040$ en $2040 - 2026 = 14$): quotiënt $-120$, rest
$14$ (de rest moet in $\intco{0}{17}$ liggen, dus is zij \emph{niet}
$-3$).
\end{solution}

\begin{solution}{exo:g12:arith:2}
Modulo $10$: $7^2 = 49 \equiv 9 \equiv -1$. Dus
$7^{100} = \left(7^2\right)^{50} \equiv (-1)^{50} = 1 \pmod{10}$: het
laatste cijfer van $7^{100}$ is $1$.
\end{solution}

\begin{solution}{exo:g12:arith:3}
Euclides: $1071 = 2\times462 + 147$; $462 = 3\times147 + 21$;
$147 = 7\times21 + 0$. Dus $\gcd = 21$.

Terugsubstitutie: $21 = 462 - 3\times147 = 462 - 3(1071 - 2\times462)
= 7\times462 - 3\times1071$. Dus $u = -3$, $v = 7$:
$1071\times(-3) + 462\times7 = 21$.
\end{solution}

\begin{solution}{exo:g12:arith:4}
Elk geheel getal is $\equiv 0, 1, 2$ of $3 \pmod 4$, en kwadrateren:
$0^2 \equiv 0$, $1^2 \equiv 1$, $2^2 = 4 \equiv 0$, $3^2 = 9 \equiv 1$. Dus
$n^2 \equiv 0$ of $1 \pmod 4$. Een som van twee kwadraten is dan congruent
met $0 + 0$, $0 + 1$ of $1 + 1$, \emph{d.w.z.}\ met $0$, $1$ of
$2 \pmod 4$ --- nooit met $3$.
\end{solution}

\begin{solution}{exo:g12:arith:5}
Deelbaarheid door $2$: onder $n$ en $n + 1$ is er één even. Deelbaarheid
door $3$: als $n \equiv 0$, dan $3 \mid n$; als $n \equiv 1 \pmod 3$, dan
$2n + 1 \equiv 3 \equiv 0$; als $n \equiv 2$, dan $n + 1 \equiv 0$. In
alle gevallen deelt $3$ het product. Omdat $\gcd(2,3) = 1$, geeft
\cref{cor:g12:arith:coprimeprod} $6 \mid n(n+1)(2n+1)$.
(Dit bewijst ook opnieuw dat $\frac{n(n+1)(2n+1)}{6}$, de som van
kwadraten van \cref{exo:g12:seq:1}, een geheel getal is.)
\end{solution}

\begin{solution}{exo:g12:arith:6}
We zoeken de inverse van $5$ modulo $11$: testen (of Bézout),
$5 \times 9 = 45 = 44 + 1 \equiv 1 \pmod{11}$. De congruentie met $9$
vermenigvuldigen:
\[
x \equiv 9 \times 3 = 27 \equiv 5 \pmod{11}.
\]
De oplossingen zijn de gehele getallen $x = 5 + 11k$, $k \in \Z$.
(Controle: $5\times5 = 25 \equiv 3 \pmod{11}$.)
\end{solution}

\begin{solution}{exo:g12:arith:7}
$\gcd(17, 40) = 1$, dus oplossingen bestaan. Euclides:
$40 = 2\times17 + 6$; $17 = 2\times6 + 5$; $6 = 5 + 1$.
Terugsubstitueren:
$1 = 6 - 5 = 6 - (17 - 2\times6) = 3\times6 - 17
= 3(40 - 2\times17) - 17 = 3\times40 - 7\times17$. Dus
$17\times(-7) - 40\times(-3) = 1$: de bijzondere oplossing
$(x_0, y_0) = (-7, -3)$.

Algemene oplossing van $17x - 40y = 1$: de bijzondere relatie aftrekken,
$17(x + 7) = 40(y + 3)$; omdat $\gcd(17, 40) = 1$, geeft Gauss
$40 \mid x + 7$, dus $x = -7 + 40k$ en dan $y = -3 + 17k$, $k \in \Z$
(alle controleren).

Voor $17x - 40y = 6$, vermenigvuldig de bijzondere oplossing met $6$:
$(x_1, y_1) = (-42, -18)$, en dezelfde redenering geeft
\[
x = -42 + 40k, \qquad y = -18 + 17k, \qquad k \in \Z .
\]
(Bijv.\ $k = 2$: $x = 38$, $y = 16$; inderdaad $17\times38 - 40\times16
= 646 - 640 = 6$.)
\end{solution}

\begin{solution}{exo:g12:arith:8}
Stel $\sqrt2 = \frac ab$ met $a, b \in \N^*$; dan $a^2 = 2b^2$. In de
priemfactorisatie van een kwadraat is elke exponent even; dus de exponent
van $2$ in $a^2$ is even, terwijl in $2b^2$ hij oneven is (één meer dan
een even getal). Twee factorisaties van hetzelfde gehele getal met
verschillende exponenten van $2$ weerspreken de uniciteit in
\cref{thm:g12:arith:fta}. Dus bestaat er geen zo'n breuk:
$\sqrt2 \notin \Q$.
\end{solution}

\begin{solution}{exo:g12:arith:9}
\emph{1.} Uit $k\binom pk = p \binom{p-1}{k-1}$ deelt $p$
$k\binom pk$. Voor $1 \leq k \leq p-1$, $p \nmid k$ en $p$ priem geven
$\gcd(p, k) = 1$, dus het lemma van Gauss levert $p \mid \binom pk$.

\emph{2.} Inductie op $a$. Voor $a = 0$: $0^p \equiv 0$. Neem
$a^p \equiv a \pmod p$ aan. Met de binomiaalstelling,
\[
(a+1)^p = \sum_{k=0}^{p} \binom pk a^k
\equiv a^p + 1 \pmod p,
\]
alle middentermen verdwijnen modulo $p$ door punt 1. Door de
inductiehypothese, $(a+1)^p \equiv a + 1 \pmod p$. Dit bewijst
$a^p \equiv a$ voor alle $a \in \N$, en het geval $a < 0$ volgt door
$a \equiv a + kp$ te schrijven voor een geschikte positieve
representant.
\end{solution}

\begin{solution}{exo:g12:arith:10}
$n \equiv 2 \pmod 3$ en $n \equiv 3 \pmod 5$: schrijf $n = 2 + 3s$; dan
$2 + 3s \equiv 3 \pmod 5$, \emph{d.w.z.}\ $3s \equiv 1 \pmod 5$. De
inverse van $3$ modulo $5$ is $2$ ($3\times2 = 6 \equiv 1$), dus
$s \equiv 2 \pmod 5$, zeg $s = 2 + 5t$, en $n = 8 + 15t$: de eerste twee
voorwaarden betekenen $n \equiv 8 \pmod{15}$.

$n \equiv 2 \pmod 7$ toevoegen: $8 + 15t \equiv 2 \pmod 7$, en
$15 \equiv 1 \pmod 7$, dus $t \equiv -6 \equiv 1 \pmod 7$, zeg
$t = 1 + 7u$. Dus $n = 23 + 105u$:
\[
n \equiv 23 \pmod{105}.
\]
(Controle: $23 = 3\times7 + 2 = 5\times4 + 3 = 7\times3 + 2$.)
\end{solution}
