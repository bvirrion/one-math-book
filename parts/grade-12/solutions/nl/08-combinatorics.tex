\section*{Hoofdstuk \ref{ch:g12:comb} --- Combinatoriek en tellen}

\begin{solution}{exo:g12:comb:1}
Vermenigvuldigingsprincipe:
$26^2 \times 10^3 \times 26^2 = 26^4 \times 10^3 = 456\,976\,000$.

Zonder herhaald teken moeten de vier letters verschillend zijn
($26 \times 25 \times 24 \times 23$ manieren, letterposities in volgorde
vullen) en de drie cijfers verschillend ($10 \times 9 \times 8$):
\[
26 \times 25 \times 24 \times 23 \times 10 \times 9 \times 8
= 358\,800 \times 720 = 258\,336\,000 .
\]
\end{solution}

\begin{solution}{exo:g12:comb:2}
$\dbinom83 = \dfrac{8 \times 7 \times 6}{3!} = 56$;
$\dbinom{10}{8} = \dbinom{10}{2} = \dfrac{10 \times 9}{2} = 45$;
\[
\frac{\binom n2}{\binom{n+1}2}
= \frac{n(n-1)/2}{(n+1)n/2} = \frac{n-1}{n+1}.
\]
\end{solution}

\begin{solution}{exo:g12:comb:3}
Kies de commissie: $\binom{30}{4}$ manieren. Kies daarna voorzitter en
penningmeester onder de 4, in volgorde: $4 \times 3 = 12$ manieren.
Totaal
\[
\binom{30}{4} \times 12 = 27\,405 \times 12 = 328\,860 .
\]
\end{solution}

\begin{solution}{exo:g12:comb:4}
\[
(x+2)^5 = x^5 + 10x^4 + 40x^3 + 80x^2 + 80x + 32 ,
\]
\[
(1-x)^6 = 1 - 6x + 15x^2 - 20x^3 + 15x^4 - 6x^5 + x^6 .
\]
In $(2x+3)^7$ is de term in $x^3$
$\binom{7}{3}(2x)^3\,3^4 = 35 \times 8 \times 81\, x^3$: de coëfficiënt
is $22\,680$.
\end{solution}

\begin{solution}{exo:g12:comb:5}
\emph{1.} $\dbinom{52}{5} = 2\,598\,960$.

\emph{2.} Precies één aas: kies die ($4$ manieren) en vul aan met $4$
niet-azen: $4 \times \binom{48}{4} = 4 \times 194\,580 = 778\,320$.
Minstens één aas: complementair tellen,
$\binom{52}{5} - \binom{48}{5} = 2\,598\,960 - 1\,712\,304 = 886\,656$.

\emph{3.} Kies de waarde van de three-of-a-kind ($13$), haar kleuren
($\binom43 = 4$), de waarde van het paar ($12$ over), haar kleuren
($\binom42 = 6$):
$13 \times 4 \times 12 \times 6 = 3744$.
\end{solution}

\begin{solution}{exo:g12:comb:6}
MATH heeft 4 verschillende letters: $4! = 24$ anagrammen.

BANANA heeft 6 letters: drie A, twee N, één B. Kies de posities van de A
($\binom63$), daarna de N onder de rest ($\binom32$), de B neemt de
laatste plaats:
\[
\binom{6}{3}\binom{3}{2} = 20 \times 3 = 60 .
\]
(Equivalent $\frac{6!}{3!\,2!\,1!} = 60$.)
\end{solution}

\begin{solution}{exo:g12:comb:7}
\emph{Algebraïsch:}
\[
k\binom nk = \frac{k\,n!}{k!(n-k)!} = \frac{n!}{(k-1)!\,(n-k)!}
= n\,\frac{(n-1)!}{(k-1)!\bigl((n-1)-(k-1)\bigr)!} = n\binom{n-1}{k-1}.
\]
\emph{Door dubbel tellen:} tel de paren (commissie van $k$, voorzitter
daarin). Of kies de commissie ($\binom nk$) dan haar voorzitter ($k$):
$k\binom nk$ paren. Of kies eerst de voorzitter ($n$ opties) dan de
$k-1$ andere leden onder de $n-1$ rest: $n\binom{n-1}{k-1}$ paren.
\end{solution}

\begin{solution}{exo:g12:comb:8}
Een pad bestaat uit precies $m + n$ stappen, waarvan $m$ Oost en $n$ Noord
zijn; het is volledig bepaald door de verzameling momenten (onder de
$m+n$) waarop men oostwaarts gaat. Er zijn $\binom{m+n}{m}$ zulke
keuzes.
\end{solution}

\begin{solution}{exo:g12:comb:9}
Splits een verzameling van $m + n$ personen in een groep $A$ van $m$ en
een groep $B$ van $n$. Een deelverzameling van $k$ elementen bevat een
zeker aantal $j$ leden van $A$ ($0 \leq j \leq k$) en $k - j$ leden van
$B$; voor vaste $j$ zijn er $\binom mj \binom{n}{k-j}$ zulke delen, en
het optelprincipe over $j$ geeft de identiteit van Vandermonde.

Met $m = n = k$:
\[
\binom{2n}{n} = \sum_{j=0}^n \binom nj \binom{n}{n-j}
= \sum_{j=0}^n \binom nj^{2},
\]
met de symmetrie $\binom{n}{n-j} = \binom nj$.
\end{solution}

\begin{solution}{exo:g12:comb:10}
\emph{Via \cref{exo:g12:comb:7}:}
\[
\sum_{k=1}^n k\binom nk = \sum_{k=1}^n n\binom{n-1}{k-1}
= n\sum_{j=0}^{n-1}\binom{n-1}{j} = n\,2^{n-1},
\]
volgens \cref{cor:g12:comb:sums}. \emph{Via differentiatie:}
differentiëren van $(1+x)^n = \sum_k \binom nk x^k$ geeft
$n(1+x)^{n-1} = \sum_k k \binom nk x^{k-1}$; evalueer in $x = 1$.
\end{solution}
