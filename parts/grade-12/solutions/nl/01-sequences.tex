\section*{Hoofdstuk \ref{ch:g12:seq} --- Rijen}

\begin{solution}{exo:g12:seq:1}
Laat $P(n)$ de uitspraak $\sum_{k=1}^{n} k^2 = \frac{n(n+1)(2n+1)}{6}$
zijn.
\emph{Basisstap:} voor $n = 0$ zijn beide kanten $0$ (lege som).
\emph{Inductiestap:} neem $P(n)$ aan. Dan
\[
\sum_{k=1}^{n+1} k^2 = \frac{n(n+1)(2n+1)}{6} + (n+1)^2
= \frac{(n+1)\bigl(n(2n+1) + 6(n+1)\bigr)}{6}
= \frac{(n+1)(2n^2 + 7n + 6)}{6}.
\]
Omdat $2n^2 + 7n + 6 = (n+2)(2n+3)$, is dit
$\frac{(n+1)(n+2)(2(n+1)+1)}{6}$, wat $P(n+1)$ is. Door inductie geldt
$P(n)$ voor alle $n$.
\end{solution}

\begin{solution}{exo:g12:seq:2}
$a_{n+1} - a_n = \frac{n+2}{n+1} - \frac{n+1}{n}
= \frac{n(n+2) - (n+1)^2}{n(n+1)} = \frac{-1}{n(n+1)} < 0$: $(a_n)$ is
strikt dalend.

$(b_n)$ heeft positieve termen en
$\frac{b_{n+1}}{b_n} = \frac{2^{n+1}}{n+1}\cdot\frac{n}{2^n} =
\frac{2n}{n+1} \geq 1 \iff 2n \geq n+1 \iff n \geq 1$: $(b_n)$ is stijgend
(strikt voor $n \geq 2$).

$c_{n+1} - c_n = (n+1)^2 - 10(n+1) - n^2 + 10n = 2n - 9$, wat negatief
is voor $n \leq 4$ en positief voor $n \geq 5$: $(c_n)$ daalt tot
$c_5 = -25$, haar minimum, en stijgt daarna. Ze is niet monotoon.
\end{solution}

\begin{solution}{exo:g12:seq:3}
Factoriseer dominante termen:
\[
u_n = \frac{n^2(3 - 1/n + 1/n^2)}{n^2(2 + 5/n^2)} \xrightarrow[n\to+\infty]{} \frac{3}{2}.
\]
Vermenigvuldig met de toegevoegde:
\[
v_n = \frac{(n+1) - n}{\sqrt{n+1} + \sqrt{n}} = \frac{1}{\sqrt{n+1}+\sqrt{n}}
\xrightarrow[n\to+\infty]{} 0.
\]
Deel teller en noemer door $3^n$:
\[
w_n = \frac{(2/3)^n - 1}{1 + (1/3)^n} \xrightarrow[n\to+\infty]{} \frac{0-1}{1+0} = -1,
\]
met $\lim q^n = 0$ voor $\abs{q} < 1$.
\end{solution}

\begin{solution}{exo:g12:seq:4}
$u_n = 5 + 3n \to +\infty$ en $v_n = 8 \cdot (1/2)^n = 2^{3-n} \to 0$.
De sommen zijn
\[
\sum_{k=0}^{n} u_k = (n+1)\,\frac{5 + (5+3n)}{2} = \frac{(n+1)(10+3n)}{2}
\xrightarrow[n\to+\infty]{} +\infty,
\]
\[
\sum_{k=0}^{n} v_k = 8\,\frac{1 - (1/2)^{n+1}}{1 - 1/2}
= 16\left(1 - \left(\tfrac{1}{2}\right)^{n+1}\right)
\xrightarrow[n\to+\infty]{} 16 .
\]
\end{solution}

\begin{solution}{exo:g12:seq:5}
Omdat $-1 \leq \cos n \leq 1$,
\[
\frac{n-1}{n+1} \leq \frac{n + \cos n}{n+1} \leq 1,
\]
en $\frac{n-1}{n+1} \to 1$, dus de limiet is $1$ door de insluitstelling.

Voor de tweede limiet, schrijf
\[
0 \leq \frac{n!}{n^n}
= \frac{1}{n}\cdot\frac{2}{n}\cdots\frac{n}{n}
\leq \frac{1}{n},
\]
omdat elke factor $\frac{k}{n}$ met $2 \leq k \leq n$ ten hoogste $1$ is.
Omdat $\frac1n \to 0$, geeft de insluitstelling $\frac{n!}{n^n} \to 0$.
(De gesuggereerde meetkundige grens werkt ook: elke factor met
$k \leq n/2$ is ten hoogste $\frac12$, wat de sterkere grens
$(1/2)^{\floor{n/2}}$ geeft.)
\end{solution}

\begin{solution}{exo:g12:seq:6}
\emph{1.} $u_0 = 0 \in \intcc{0}{2}$. Als $0 \leq u_n \leq 2$, dan
$2 \leq u_n + 2 \leq 4$, dus $\sqrt{2} \leq u_{n+1} \leq 2$; in het
bijzonder $0 \leq u_{n+1} \leq 2$. Door inductie geldt de eigenschap
voor alle $n$.

\emph{2.} $u_{n+1} - u_n = \sqrt{u_n + 2} - u_n$. Voor
$x \in \intcc{0}{2}$,
$\sqrt{x+2} \geq x \iff x + 2 \geq x^2 \iff (2-x)(x+1) \geq 0$, wat
waar is. Dus is $(u_n)$ stijgend.

\emph{3.} Stijgend en van boven begrensd door $2$, convergeert $(u_n)$
naar zekere $\ell \in \intcc{0}{2}$. Naar de limiet gaan in
$u_{n+1} = \sqrt{u_n + 2}$ (de afbeelding $x \mapsto \sqrt{x+2}$ is
continu) geeft $\ell = \sqrt{\ell + 2}$, dus $\ell^2 - \ell - 2 = 0$,
d.w.z.\ $\ell \in \{-1, 2\}$. Omdat $\ell \geq 0$, is $\lim u_n = 2$.
\end{solution}

\begin{solution}{exo:g12:seq:7}
\emph{1.} Tussen twee doses wordt $40\%$ van het medicijn geëlimineerd,
dus wordt de hoeveelheid $u_n$ tot $0.6\,u_n$; de volgende dosis voegt
$1$ eenheid toe: $u_{n+1} = 0.6\,u_n + 1$.

\emph{2.} $v_{n+1} = u_{n+1} - 2.5 = 0.6\,u_n + 1 - 2.5 = 0.6(u_n - 2.5)
= 0.6\,v_n$: $(v_n)$ is meetkundig met verhouding $0.6$ en eerste term
$v_0 = 1 - 2.5 = -1.5$. Dus $v_n = -1.5 \times 0.6^n$ en
\[
u_n = 2.5 - 1.5 \times 0.6^n .
\]

\emph{3.} Omdat $0.6^n \to 0$, $u_n \to 2.5$: de hoeveelheid medicijn
stabiliseert op $2.5$ eenheden.
\end{solution}

\begin{solution}{exo:g12:seq:8}
\emph{1.} $u_0 = 3 > 1$. Als $u_n > 1$, dan $u_n + 2 > 0$ en
\[
u_{n+1} - 1 = \frac{4u_n - 1 - u_n - 2}{u_n + 2} = \frac{3(u_n - 1)}{u_n + 2} > 0 .
\]
Door inductie is $u_n > 1$ voor alle $n$ (en in het bijzonder
$u_n + 2 \neq 0$, dus de rij is welgedefinieerd).

\emph{2.} Met de identiteit hierboven,
\[
v_{n+1} = \frac{1}{u_{n+1} - 1} = \frac{u_n + 2}{3(u_n - 1)}
= \frac{(u_n - 1) + 3}{3(u_n - 1)} = \frac{1}{3} + v_n .
\]
Dus is $(v_n)$ rekenkundig met gemeenschappelijk verschil $\frac13$ en
$v_0 = \frac{1}{u_0 - 1} = \frac12$.

\emph{3.} $v_n = \frac12 + \frac{n}{3}$, dus
$u_n = 1 + \frac{1}{v_n} = 1 + \frac{6}{3 + 2n}$. Omdat
$v_n \to +\infty$, is $u_n \to 1$.
\end{solution}

\begin{solution}{exo:g12:seq:9}
\emph{1.} $H_{2n} - H_n = \sum_{k=n+1}^{2n} \frac{1}{k}$ is een som van
$n$ termen, elk minstens $\frac{1}{2n}$; dus
$H_{2n} - H_n \geq n \cdot \frac{1}{2n} = \frac12$.

\emph{2.} Door inductie op $k$: $H_{2^0} = H_1 = 1 \geq 1$. Als
$H_{2^k} \geq 1 + \frac{k}{2}$, dan met punt 1 met $n = 2^k$,
\[
H_{2^{k+1}} \geq H_{2^k} + \frac12 \geq 1 + \frac{k+1}{2}.
\]
De rij $(H_n)$ is stijgend (elke stap voegt $\frac{1}{n+1} > 0$ toe) en
de deelrij $H_{2^k}$ is onbegrensd, dus is $(H_n)$ niet van boven
begrensd. Stijgend en onbegrensd, neigt ze naar $+\infty$
(\cref{thm:g12:seq:monotone}).
\end{solution}

\begin{solution}{exo:g12:seq:10}
\emph{1.} De rij $d_n = b_n - a_n$ voldoet aan
$d_{n+1} - d_n = (b_{n+1} - b_n) - (a_{n+1} - a_n) \leq 0$, dus is
$(d_n)$ dalend; omdat $d_n \to 0$, krijgen we $d_n \geq 0$ voor alle $n$
(een dalende rij met een negatieve term zou er voor altijd onder blijven,
wat limiet $0$ verhindert). Dus $a_n \leq b_n$.

\emph{2.} Uit $a_n \leq b_n \leq b_0$ is de stijgende rij $(a_n)$ van
boven begrensd, dus convergeert ze naar zekere $\ell$. Evenzo convergeert
$(b_n)$, dalend en van onder begrensd door $a_0$, naar zekere $\ell'$.
Dan $\ell' - \ell = \lim (b_n - a_n) = 0$, dus $\ell = \ell'$.

\emph{3.} $(a_n)$ is (strikt) stijgend omdat
$a_{n+1} - a_n = \frac{1}{(n+1)!} > 0$. Voor $(b_n)$,
\[
b_{n+1} - b_n = \frac{1}{(n+1)!} + \frac{1}{(n+1)(n+1)!} - \frac{1}{n\,n!}
= \frac{n(n+1) + n - (n+1)^2}{n(n+1)(n+1)!}
= \frac{-1}{n(n+1)(n+1)!} < 0 ,
\]
dus is $(b_n)$ dalend. Ten slotte $b_n - a_n = \frac{1}{n\,n!} \to 0$.
De twee rijen zijn aangrenzend, dus convergeren ze naar een
gemeenschappelijke limiet.
\end{solution}
