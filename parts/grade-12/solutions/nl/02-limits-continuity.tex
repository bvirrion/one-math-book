\section*{Hoofdstuk \ref{ch:g12:limcont} --- Limieten en continuïteit}

\begin{solution}{exo:g12:limcont:1}
Delen door $x^3$:
$\dfrac{2x^3 - x + 1}{5x^3 + x^2} = \dfrac{2 - 1/x^2 + 1/x^3}{5 + 1/x}
\to \dfrac{2}{5}$.

$x^2 + 3x - 1 = x^2(1 + 3/x - 1/x^2) \to +\infty$ als $x \to -\infty$,
omdat $x^2 \to +\infty$ en de haak naar $1$ neigt.

Als $x \to 2^+$, neigt de teller naar $3 > 0$ en de noemer naar $0^+$,
dus $\dfrac{x+1}{x-2} \to +\infty$.

Voor $x > 0$, $-\dfrac1x \leq \dfrac{\cos x}{x} \leq \dfrac1x$, dus de
limiet is $0$ door de insluitstelling.
\end{solution}

\begin{solution}{exo:g12:limcont:2}
\emph{1.} $2x^2 - 3x + 1 = (x-1)(2x-1)$ (de \omterm{def:g11:quad:discriminant}{wortel} $x=1$ is duidelijk
uit $2 - 3 + 1 = 0$). Voor $x \neq 1$, schrap $x - 1$: $f(x) = 2x - 1$.

\emph{2.} Dus $\lim_{x\to1} f(x) = 1$. Stellen $f(1) = 1$ breidt $f$ uit
tot de polynoomfunctie $x \mapsto 2x-1$ op $\R$, die \omterm{def:g12:limcont:continuity}{continu} is in $1$.
\end{solution}

\begin{solution}{exo:g12:limcont:3}
Schrijf $f(x) = \dfrac{3x - 1}{x + 2} = \dfrac{3(x+2) - 7}{x+2}
= 3 - \dfrac{7}{x+2}$. Als $x \to \pm\infty$, $f(x) \to 3$: de lijn
$y = 3$ is een \omterm{def:g12:limcont:asymptote}{horizontale asymptoot} in beide oneindigheden. Als
$x \to (-2)^\pm$, $f(x) \to \mp\infty$: de lijn $x = -2$ is een
\omterm{def:g12:limcont:asymptote}{verticale asymptoot}. De kromme is een \omterm{def:g11:func:reference}{hyperbool}, onder $y=3$ voor
$x > -2$ en erboven voor $x < -2$.
\end{solution}

\begin{solution}{exo:g12:limcont:4}
Vermenigvuldig met de toegevoegde:
\[
\sqrt{x^2 + x} - x = \frac{x^2 + x - x^2}{\sqrt{x^2+x} + x}
= \frac{x}{x\left(\sqrt{1 + 1/x} + 1\right)}
= \frac{1}{\sqrt{1 + 1/x} + 1}
\xrightarrow[x\to+\infty]{} \frac{1}{2}.
\]
Evenzo,
\[
\frac{\sqrt{1+x} - 1}{x} = \frac{(1+x) - 1}{x\left(\sqrt{1+x} + 1\right)}
= \frac{1}{\sqrt{1+x} + 1} \xrightarrow[x\to0]{} \frac{1}{2}.
\]
\end{solution}

\begin{solution}{exo:g12:limcont:5}
\emph{1.} Omdat $\sin x \geq -1$, is $f(x) \geq x - 1 \to +\infty$, en we
concluderen door \omterm{def:g10:algebra:equation}{vergelijking}. Toch verdwijnt $f'(x) = 1 + \cos x$ in
elk oneven veelvoud van $\pi$, en op elk \omterm{def:g10:numbers:interval}{interval} $\intco{A}{+\infty}$
wisselt de \omterm{def:g11:func:function}{functie} tussen intervallen van strikte stijging; ze is
\omterm{def:g12:seq:monotonic}{stijgend} maar niet strikt, en belangrijker is haar groei niet nodig:
\omterm{def:g10:algebra:equation}{vergelijking} volstaat.

\emph{2.} Voor $x > 1$, deel door $x$:
\[
g(x) = \frac{1 + \frac{\sin x}{x}}{1 - \frac{\sin x}{x}},
\]
en $\frac{\sin x}{x} \to 0$ door de insluitstelling
($\abs{\sin x} \leq 1$). Dus $g(x) \to \frac{1+0}{1-0} = 1$.
\end{solution}

\begin{solution}{exo:g12:limcont:6}
$f(x) = x^3 - 3x + 1$ is \omterm{def:g12:limcont:continuity}{continu}, met $f'(x) = 3x^2 - 3 = 3(x-1)(x+1)$:
$f$ stijgt op $\intoc{-\infty}{-1}$, daalt op $\intcc{-1}{1}$, stijgt op
$\intco{1}{+\infty}$, met lokaal \omterm{def:g10:functions:extrema}{maximum} $f(-1) = 3 > 0$ en lokaal
\omterm{def:g10:functions:extrema}{minimum} $f(1) = -1 < 0$, $\lim_{-\infty} f = -\infty$,
$\lim_{+\infty} f = +\infty$.

Op elk van de drie monotonie-intervallen is $f$ \omterm{def:g12:limcont:continuity}{continu}, strikt monotoon,
en ligt $0$ strikt tussen de eindpuntwaarden/limieten, dus geeft de
bijectiestelling precies één nulpunt per \omterm{def:g10:numbers:interval}{interval}: drie nulpunten in
totaal. Tekencontroles: $f(-2) = -1 < 0 < f(-1) = 3$, dus één nulpunt in
$\intoo{-2}{-1}$; $f(0) = 1 > 0 > f(1) = -1$, één nulpunt in
$\intoo{0}{1}$; $f(1) = -1 < 0 < f(2) = 3$, één nulpunt in
$\intoo{1}{2}$.
\end{solution}

\begin{solution}{exo:g12:limcont:7}
Laat $g(x) = f(x) - x$, \omterm{def:g12:limcont:continuity}{continu} op $\intcc{0}{1}$. Omdat
$f(0) \in \intcc{0}{1}$, is $g(0) = f(0) \geq 0$; omdat
$f(1) \in \intcc{0}{1}$, is $g(1) = f(1) - 1 \leq 0$. Volgens de
tussenwaardestelling verdwijnt $g$ in zekere $c \in \intcc{0}{1}$,
d.w.z.\ $f(c) = c$.
\end{solution}

\begin{solution}{exo:g12:limcont:8}
Laat $u(t)$ en $d(t)$ de posities zijn (gemeten als afstand vanaf de
voet van het pad) op tijdstip $t \in \intcc{8}{20}$ op de heenweg en op
de terugweg. Beide zijn \omterm{def:g12:limcont:continuity}{continu}, $u(8) = 0$, $u(20) = L$ (de lengte van
het pad), $d(8) = L$, $d(20) = 0$. De \omterm{def:g11:func:function}{functie} $g = u - d$ is \omterm{def:g12:limcont:continuity}{continu} met
$g(8) = -L \leq 0$ en $g(20) = L \geq 0$, dus $g(c) = 0$ voor zekere tijd
$c$ volgens de tussenwaardestelling: op tijdstip $c$ vallen de posities
van beide dagen samen.
\end{solution}

\begin{solution}{exo:g12:limcont:9}
\emph{1.} $f'(x) = 5x^4 + 3x^2 + 1 > 0$, dus $f$ is strikt \omterm{def:g12:seq:monotonic}{stijgend} op
$\R$; met $\lim_{\pm\infty} f = \pm\infty$ en \omterm{def:g12:limcont:continuity}{continuïteit} geeft de
bijectiestelling een uniek reëel nulpunt $\alpha$. Omdat
$f(0) = -1 < 0$ en $f(1) = 2 > 0$, is $\alpha \in \intoo{0}{1}$.

\emph{2.} Houd een \omterm{def:g10:numbers:interval}{interval} $\intcc{a_n}{b_n}$ bij met
$f(a_n) < 0 < f(b_n)$, startend van $\intcc{0}{1}$; evalueer bij elke
stap $f$ in het \omterm{prop:g10:coordgeom:midpoint}{midden} $m$ en houd de half waarin de tekenwisseling
blijft. Na $n$ stappen heeft het \omterm{def:g10:numbers:interval}{interval} lengte $2^{-n}$, dus benadert
het \omterm{prop:g10:coordgeom:midpoint}{midden} $\alpha$ tot op $2^{-(n+1)}$. We hebben
$2^{-(n+1)} \leq 10^{-6}$ nodig, d.w.z.\
$n + 1 \geq \frac{6}{\log_{10} 2} \approx 19.93$: $n = 20$ iteraties
volstaan.
\end{solution}
