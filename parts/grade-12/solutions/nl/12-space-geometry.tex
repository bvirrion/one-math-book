\section*{Hoofdstuk \ref{ch:g12:space} --- Vectoren, lijnen en vlakken in de ruimte}

\begin{solution}{exo:g12:space:1}
$\vect{AB}(2, 1, -1)$ en $\vect{AC}(1, -1, 1)$ zijn niet evenredig
($\frac21 \neq \frac{1}{-1}$), dus de punten zijn niet \omterm{def:g11:vect:collinear}{collineair}. Lijn
$(AB)$:
\[
(x, y, z) = (1 + 2t,\; t,\; 2 - t), \qquad t \in \R .
\]
\end{solution}

\begin{solution}{exo:g12:space:2}
$2(x - 1) - (y - 1) + 3(z - 0) = 0$, \emph{d.w.z.}\
\[
2x - y + 3z - 1 = 0 .
\]
Voor $B(0,2,1)$: $0 - 2 + 3 - 1 = 0$: ja, $B$ ligt op het vlak.
\end{solution}

\begin{solution}{exo:g12:space:3}
$\vect{AB}(1,1,0)$, $\vect{AC}(1,0,1)$:
\[
\cos\widehat{A}
= \frac{\vect{AB}\cdot\vect{AC}}{\norm{\vect{AB}}\,\norm{\vect{AC}}}
= \frac{1}{\sqrt2\,\sqrt2} = \frac12,
\]
dus $\widehat A = 60^\circ$.
\end{solution}

\begin{solution}{exo:g12:space:4}
Punten van $\mathcal D$: $(1 + t,\, 2 - t,\, 2t)$. Substitueren in de
\omterm{def:g10:algebra:equation}{vergelijking} van $\mathcal P$:
\[
2(1+t) + (2 - t) - 2t + 1 = 5 - t = 0 \iff t = 5 .
\]
Uniek snijpunt: $(6, -3, 10)$.
\end{solution}

\begin{solution}{exo:g12:space:5}
De \omterm{def:g12:space:normal}{normaalvectoren} $(1, -1, 2)$ en $(2, 1, 1)$ zijn niet \omterm{def:g11:vect:collinear}{collineair}, dus
de vlakken snijden in een lijn. De twee \omterm{def:g10:algebra:equation}{vergelijkingen} optellen:
$3x + 3z = 5$, dus $x = \frac53 - z$; daarna geeft de eerste \omterm{def:g10:algebra:equation}{vergelijking}
$y = x + 2z - 1 = \frac23 + z$. Met $z = t$:
\[
(x, y, z) = \left(\frac53 - t,\; \frac23 + t,\; t\right), \qquad t \in \R,
\]
een lijn met \omterm{def:g11:vect:direction}{richtingsvector} $(-1, 1, 1)$. (Beide \omterm{def:g10:algebra:equation}{vergelijkingen}
controleren identiek.)
\end{solution}

\begin{solution}{exo:g12:space:6}
Richtingen $\vec u_1(1, 2, -1)$ en $\vec u_2(1, 1, 2)$ zijn niet
\omterm{def:g11:vect:collinear}{collineair}, dus de lijnen zijn niet evenwijdig. Snijden zou vereisen
\[
1 + t = 2 + s, \qquad 2t = 1 + s, \qquad 3 - t = 1 + 2s .
\]
De eerste twee geven $t = 1 + s$ en $2(1+s) = 1 + s$, dus $s = -1$,
$t = 0$; maar dan leest de derde $3 = -1$, absurd. Geen gemeenschappelijk
punt en niet evenwijdig: de lijnen zijn kruisend.
\end{solution}

\begin{solution}{exo:g12:space:7}
\emph{1.} $\operatorname{dist} = \dfrac{\abs{2\cdot3 - 1 + 2\cdot1 - 3}}
{\sqrt{4 + 1 + 4}} = \dfrac{4}{3}$.

\emph{2.} $H = M_0 + t\,\vec n$ met $\vec n(2, -1, 2)$, $t$ zo gekozen dat
$H \in \mathcal P$:
\[
2(3 + 2t) - (1 - t) + 2(1 + 2t) - 3 = 4 + 9t = 0 \iff t = -\frac49 .
\]
Dus $H\left(\frac{19}{9}, \frac{13}{9}, \frac19\right)$, en
\[
M_0H = \norm{t\,\vec n} = \frac49 \times 3 = \frac43,
\]
in overeenstemming met de afstandsformule.
\end{solution}

\begin{solution}{exo:g12:space:8}
\omterm{def:g10:coordgeom:system}{Coördinaten}: $G(1,1,1)$, en het vlak $(BDE)$ gaat door $B(1,0,0)$,
$D(0,1,0)$, $E(0,0,1)$.

\emph{1.} Het vlak $(BDE)$ heeft \omterm{def:g10:algebra:equation}{vergelijking} $x + y + z = 1$ (voldaan
door de drie punten, en zij zijn niet \omterm{def:g11:vect:collinear}{collineair}), dus $\vec n(1,1,1)$ is
normaal erop. Omdat $\vect{AG} = (1,1,1) = \vec n$, is de diagonaal
$(AG)$ loodrecht op het vlak $(BDE)$.

\emph{2.} Afstand van $A(0,0,0)$:
$\frac{\abs{0 + 0 + 0 - 1}}{\sqrt3} = \frac{1}{\sqrt3} = \frac{\sqrt3}{3}$.
De lijn $(AG)$ is $(t, t, t)$; zij snijdt het vlak wanneer $3t = 1$: in het
punt $\left(\frac13, \frac13, \frac13\right)$, het zwaartepunt van de
driehoek $BDE$.
\end{solution}

\begin{solution}{exo:g12:space:9}
\emph{1.} $\vect{AB}(1, 0, -1)$, $\vect{AC}(-1, 1, 0)$,
$\vect{AD}(1, 1, 1)$. Als $\vect{AD} = a\vect{AB} + b\vect{AC}$, geven de
\omterm{def:g10:coordgeom:system}{coördinaten} $a - b = 1$, $b = 1$, $-a = 1$: de eerste twee forceren
$a = 2$, in strijd met $a = -1$. Niet \omterm{def:g12:space:collinear}{coplanair}.

\emph{2.} $\vect{BC}(-2, 1, 1)$, $\vect{BD}(0, 1, 2)$. Oplossen
$\vec n \cdot \vect{BC} = \vec n \cdot \vect{BD} = 0$: uit
$b + 2c = 0$, $b = -2c$; dan $-2a - 2c + c = 0$ geeft $c = -2a$. Met
$a = 1$: $\vec n(1, 4, -2)$. Vlak door $B(2,1,0)$:
\[
x + 4y - 2z - 6 = 0 .
\]

\emph{3.} Afstand van $A(1,1,1)$:
$\dfrac{\abs{1 + 4 - 2 - 6}}{\sqrt{21}} = \dfrac{3}{\sqrt{21}}$.
Oppervlakte van $BCD$: $\norm{\vect{BC}}^2 = 6$, $\norm{\vect{BD}}^2 = 5$,
$\vect{BC}\cdot\vect{BD} = 3$, dus
\[
\text{Opp.} = \frac12\sqrt{6 \times 5 - 9} = \frac{\sqrt{21}}{2}.
\]

\emph{4.}
\[
V = \frac13 \times \frac{\sqrt{21}}{2} \times \frac{3}{\sqrt{21}}
= \frac12 .
\]
\end{solution}
