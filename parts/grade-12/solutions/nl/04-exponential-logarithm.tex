\begin{solution}{exo:g12:exp:2}
\emph{(a)} Stel $X = \eu^x > 0$: $X^2 - 3X + 2 = 0$ geeft $X = 1$ of
$X = 2$, beide positief, dus $x = 0$ of $x = \ln 2$.

\emph{(b)} Definitiedomein: $x - 1 > 0$ en $x + 2 > 0$, dus $x > 1$. De
vergelijking wordt $\ln\bigl((x-1)(x+2)\bigr) = \ln 4$, vandaar
$(x-1)(x+2) = 4$, d.w.z.\ $x^2 + x - 6 = 0$, dus $x = 2$ of $x = -3$.
Alleen $x = 2$ ligt in het domein: $S = \{2\}$.
\end{solution}

\begin{solution}{exo:g12:exp:3}
Deel door $\eu^x$:
$\dfrac{1 - x^2\eu^{-x}}{1 + \eu^{-x}} \to \dfrac{1-0}{1+0} = 1$, met
$x^2 \eu^{-x} \to 0$ (\cref{thm:g12:exp:growth}).

$\dfrac{\ln x}{\sqrt x} \to 0$ volgens \cref{prop:g12:exp:lnanalytic}
(geval $n = 2$).

$x^2 \ln x = x \cdot (x \ln x) \to 0 \times 0 = 0$.

$x - \ln x = x\left(1 - \frac{\ln x}{x}\right) \to +\infty$ omdat
$\frac{\ln x}{x} \to 0$.
\end{solution}

\begin{solution}{exo:g12:exp:4}
$f'(x) = \eu^{-x} - x\eu^{-x} = (1 - x)\eu^{-x}$, met het teken van
$1 - x$: $f$ stijgt op $\intoc{-\infty}{1}$, daalt op $\intco{1}{+\infty}$,
met globaal maximum $f(1) = \eu^{-1}$.

Limieten: als $x \to +\infty$, $f(x) = \frac{x}{\eu^x} \to 0$
(\cref{thm:g12:exp:growth}); als $x \to -\infty$, $x \to -\infty$ en
$\eu^{-x} \to +\infty$, dus $f(x) \to -\infty$.

$f''(x) = -\eu^{-x} - (1-x)\eu^{-x} = (x - 2)\eu^{-x}$, die van teken
wisselt in $x = 2$: buigpunt in $\bigl(2,\, 2\eu^{-2}\bigr)$.
\end{solution}

\begin{solution}{exo:g12:exp:5}
Laat $g(x) = x - \ln(1+x)$ op $\intoo{-1}{+\infty}$. Dan
$g'(x) = 1 - \frac{1}{1+x} = \frac{x}{1+x}$, negatief op $\intoo{-1}{0}$
en positief op $\intoo{0}{+\infty}$: $g$ bereikt haar minimum $g(0) = 0$,
dus $g \geq 0$ met gelijkheid alleen in $0$. Vandaar $\ln(1+x) \leq x$.

Toepassen met $x = \frac1n$:
$n\ln\left(1 + \frac1n\right) \leq 1$, dus
$\left(1+\frac1n\right)^n = \eu^{\,n\ln(1+1/n)} \leq \eu$.
Toepassen met $x = -\frac{1}{n+1} > -1$:
$\ln\left(1 - \frac{1}{n+1}\right) \leq -\frac{1}{n+1}$, dus
$-(n+1)\ln\left(1 - \frac{1}{n+1}\right) \geq 1$ en
$\left(1 - \frac{1}{n+1}\right)^{-(n+1)} \geq \eu$.
\end{solution}

\begin{solution}{exo:g12:exp:6}
\emph{1.} Elk jaar vermenigvuldigt het kapitaal met $1.03$:
$C_n = C_0 (1.03)^n$.

\emph{2.} $C_n \geq 2C_0 \iff (1.03)^n \geq 2 \iff n \ln 1.03 \geq \ln 2
\iff n \geq \frac{\ln 2}{\ln 1.03} \approx 23.45$: het kapitaal verdubbelt
na $24$ jaren.

\emph{3.} $\frac{70}{3} \approx 23.3$, dicht bij de exacte
$\frac{\ln 2}{\ln 1.03}$. De regel werkt omdat
$\ln(1 + r) \approx r$ voor kleine $r$, dus
$\frac{\ln 2}{\ln(1+r)} \approx \frac{0.693}{r} \approx \frac{70}{100r}$.
\end{solution}

\begin{solution}{exo:g12:exp:7}
$\eu^{2x} > \eu^{x+1} \iff 2x > x + 1 \iff x > 1$ (de exponentiÃŦle is
strikt stijgend): $S = \intoo{1}{+\infty}$.

Definitiedomein van de tweede ongelijkheid: $x^2 - 1 > 0$ en $x + 5 > 0$,
d.w.z.\ $x \in \intoo{-5}{-1} \cup \intoo{1}{+\infty}$. Op dit domein,
omdat $\ln$ strikt stijgend is,
\[
\ln(x^2-1) \leq \ln(x+5) \iff x^2 - 1 \leq x + 5 \iff x^2 - x - 6 \leq 0
\iff x \in \intcc{-2}{3}.
\]
Doorsnede met het domein: $S = \intco{-2}{-1} \cup \intoc{1}{3}$.
\end{solution}

\begin{solution}{exo:g12:exp:8}
\emph{1.} $f'(x) = \dfrac{\eu^x x - \eu^x}{x^2} = \dfrac{(x-1)\eu^x}{x^2}$,
