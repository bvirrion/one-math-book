\section*{Hoofdstuk \ref{ch:g12:randvar} --- Stochastische variabelen en de binomiale verdeling}

\begin{solution}{exo:g12:randvar:1}
$\E(X) = -0.3 + 0 + 0.8 + 0.5 = 1$.
$\E(X^2) = 0.3 \times 1 + 0 + 0.4\times4 + 0.1\times25 = 4.4$, dus
volgens König--Huygens $\V(X) = 4.4 - 1 = 3.4$ en
$\sigma(X) = \sqrt{3.4} \approx 1.84$.
\end{solution}

\begin{solution}{exo:g12:randvar:2}
\emph{1.} De $10$ vragen zijn onafhankelijke Bernoulli-proeven met
$p = \frac14$: $X \sim \mathcal B\left(10, \frac14\right)$,
$\E(X) = 2.5$,
$\sigma(X) = \sqrt{10 \times \frac14 \times \frac34} = \sqrt{1.875}
\approx 1.37$.

\emph{2.} $\P(X = 0) = \left(\frac34\right)^{10} \approx 0.056$;
\[
\P(X = 5) = \binom{10}{5}\left(\frac14\right)^5\left(\frac34\right)^5
\approx 0.058;
\quad
\P(X \geq 1) = 1 - \left(\tfrac34\right)^{10} \approx 0.944 .
\]
\end{solution}

\begin{solution}{exo:g12:randvar:3}
Onafhankelijke identieke proeven: $X \sim \mathcal B(400,\ 0.05)$, dus
$\E(X) = 20$ claims en
\[
\sigma(X) = \sqrt{400 \times 0.05 \times 0.95} = \sqrt{19} \approx 4.4 .
\]
\end{solution}

\begin{solution}{exo:g12:randvar:4}
De betaling $P$ die ontvangen wordt voldoet aan
$\E(P) = \frac{5 + 6}{6} = \frac{11}{6}$, dus de eerlijke prijs is
$c = \frac{11}{6} \approx 1.83$ euro. De eerlijke winst $G = P - \frac{11}6$
neemt waarden $-\frac{11}{6}, \frac{19}{6}, \frac{25}{6}$ aan met kansen
$\frac46, \frac16, \frac16$:
\[
\V(G) = \E(G^2)
= \frac{4 \times 121 + 361 + 625}{6 \times 36} = \frac{1470}{216}
= \frac{245}{36} \approx 6.8,
\qquad \sigma(G) \approx 2.6 .
\]
Een eerlijk spel heeft gemiddelde winst nul maar zijn uitkomsten
fluctueren nog: eerlijk is niet risicoloos.
\end{solution}

\begin{solution}{exo:g12:randvar:5}
$X \sim \mathcal B(8,\ 0.7)$.
\[
\P(X = 6) = \binom86 (0.7)^6(0.3)^2 \approx 0.296 ;
\]
\[
\P(X \geq 6) = \P(6) + \P(7) + \P(8)
\approx 0.296 + 8(0.7)^7(0.3) + (0.7)^8
\approx 0.296 + 0.198 + 0.058 = 0.552 ;
\]
$\P(X \geq 1) = 1 - (0.3)^8 \approx 0.99993$. Modus:
$(n+1)p = 6.3$, dus de meest waarschijnlijke waarde is $k^* = 6$
(\cref{exo:g12:randvar:7}).
\end{solution}

\begin{solution}{exo:g12:randvar:6}
Het aantal passagiers dat verschijnt is $X \sim \mathcal B(104,\ 0.9)$;
de vlucht is overboekt wanneer $X \geq 101$:
\[
\P(X \geq 101) = \sum_{k=101}^{104}\binom{104}{k}(0.9)^k(0.1)^{104-k}
\approx 0.006 .
\]
$4\%$ meer tickets verkopen dan stoelen veroorzaakt een incident op
slechts ongeveer $0.6\%$ van de vluchten --- de economie achter overboeking.
\end{solution}

\begin{solution}{exo:g12:randvar:7}
\[
\frac{\P(X = k+1)}{\P(X = k)}
= \frac{\binom{n}{k+1}}{\binom nk}\cdot\frac{p}{1-p}
= \frac{n - k}{k + 1}\cdot\frac{p}{1-p},
\]
met $\binom{n}{k+1} = \binom nk \frac{n-k}{k+1}$. Deze verhouding is
$\geq 1$ iff $(n-k)p \geq (k+1)(1-p)$ iff $np - k p \geq k - kp + 1 - p$
iff $k \leq (n+1)p - 1$. Dus de kansen nemen strikt toe zolang
$k + 1 \leq (n+1)p$ en daarna af: het maximum wordt bereikt in
$k^* = \floor{(n+1)p}$ (gedeeld met $k^* - 1$ wanneer $(n+1)p$ een geheel
getal is).
\end{solution}

\begin{solution}{exo:g12:randvar:8}
\emph{1.} Kop eerst bij worp $k$ betekent $k-1$ munt dan kop: kans
$\left(\frac12\right)^{k-1}\cdot\frac12 = 2^{-k}$, voor
$1 \leq k \leq 10$. Totale winstkans
$\sum_{k=1}^{10} 2^{-k} = 1 - 2^{-10} = \frac{1023}{1024}$ (het spel is
alleen verloren bij tien opeenvolgende munt).

\emph{2.} Verwachte uitbetaling:
\[
\sum_{k=1}^{10} 2^k \cdot 2^{-k} = \sum_{k=1}^{10} 1 = 10 \text{ euro}.
\]
Zonder de cap divergeert de som $\sum_{k\geq1} 1$: de verwachte uitbetaling
is oneindig, hoewel het spel bijna altijd een klein bedrag betaalt --- de
beroemde \emph{Sint-Petersburg-paradox}, die toont dat verwachting alleen
de waarde van een spel niet meet.
\end{solution}
