\section*{Hoofdstuk \ref{ch:g12:sums} --- Sommen van stochastische variabelen en de wet van de grote aantallen}

\begin{solution}{exo:g12:sums:1}
Elke steen heeft \omterm{def:g12:randvar:exp}{verwachting}
$\frac{1 + 2 + \dots + 6}{6} = \frac{7}{2}$, dus door lineariteit
$\E(S) = \frac72 + \frac72 = 7$. De stenen zijn \omterm{def:g12:sums:indep}{onafhankelijk}, dus
$\V(S) = \frac{35}{12} + \frac{35}{12} = \frac{35}{6} \approx 5.83$.
\end{solution}

\begin{solution}{exo:g12:sums:2}
$\E(X) = 50$ en $\V(X) = 100 \times 0.25 = 25$.

Markov ($X \geq 0$): $\P(X \geq 75) \leq \frac{50}{75} = \frac23$.

Bienaymé--Tsjebysjev:
$\P(\abs{X - 50} \geq 25) \leq \frac{25}{25^2} = 0.04$ --- en dit begrensd
zelfs de \emph{tweezijdige} \omterm{def:g11:prob:model}{gebeurtenis}, waarvan $X \geq 75$ de helft is.
Tsjebysjev is hier veel scherper omdat hij de \omterm{def:g12:randvar:exp}{variantie} benut, niet alleen
het \omterm{def:g11:stat:mean}{gemiddelde}. (De ware waarde van $\P(X \geq 75)$ is kleiner dan
$10^{-6}$: beide grenzen zijn grof.)
\end{solution}

\begin{solution}{exo:g12:sums:3}
$F_n$ is het \omterm{prop:g12:sums:mean}{steekproefgemiddelde} van $n$ Bernoulli$\left(\frac12\right)$-variabelen,
van \omterm{def:g12:randvar:exp}{variantie} $\sigma^2 = \frac14$. De grens
\[
\P\left(\abs{F_n - 0.5} \geq 0.05\right)
\leq \frac{1/4}{n \times 0.05^2} = \frac{100}{n}
\]
is $\leq 0.05$ zodra $n \geq 2000$.
\end{solution}

\begin{solution}{exo:g12:sums:4}
Volgens \cref{prop:g12:sums:mean}, $\E(M_n) = 5\%$ (diversificatie
verandert de verwachte opbrengst niet) en
$\sigma(M_n) = \frac{20\%}{\sqrt n}$. Eisen
$\frac{20}{\sqrt n} < 4$ geeft $\sqrt n > 5$, \emph{d.w.z.}\ $n \geq 26$.
Dit is het principe van \emph{diversificatie}: kapitaal spreiden over
onafhankelijke risico's deelt het risico (\omterm{def:g11:stat:variance}{standaardafwijking}) door
$\sqrt n$ zonder de verwachte opbrengst te verlagen.
\end{solution}

\begin{solution}{exo:g12:sums:5}
\emph{1.} De $9$ even waarschijnlijke paren tellen: $S$ neemt de waarden
$2, 3, 4, 5, 6$ aan met kansen
$\frac19, \frac29, \frac39, \frac29, \frac19$. Dus
$\E(S) = \frac{2 + 6 + 12 + 10 + 6}{9} = 4$ en
$\E(S^2) = \frac{4 + 18 + 48 + 50 + 36}{9} = \frac{156}{9} = \frac{52}{3}$,
dus $\V(S) = \frac{52}{3} - 16 = \frac43$.

\emph{2.} Eén variabele: $\E(X) = 2$,
$\E(X^2) = \frac{1 + 4 + 9}{3} = \frac{14}{3}$,
$\V(X) = \frac{14}{3} - 4 = \frac23$. Dan $\E(S) = 2 + 2 = 4$ en, door
\omterm{def:g12:condprob:indep}{onafhankelijkheid}, $\V(S) = \frac23 + \frac23 = \frac43$. Dezelfde
waarden.
\end{solution}

\begin{solution}{exo:g12:sums:6}
Met $Y = X$: $\V(X + Y) = \V(2X) = 4\V(X)$, terwijl
$\V(X) + \V(Y) = 2\V(X)$. De twee komen alleen overeen wanneer
$\V(X) = 0$, \emph{d.w.z.}\ wanneer $X$ constant is --- dus additiviteit
eist echt \omterm{def:g12:condprob:indep}{onafhankelijkheid} (hier is $X$ maximaal afhankelijk van zichzelf).
\end{solution}

\begin{solution}{exo:g12:sums:7}
Onder eerlijkheid is $F_n$ het \omterm{def:g11:stat:mean}{gemiddelde} van $n = 1200$
Bernoulli-variabelen met $p = \frac16$,
$\sigma^2 = \frac16\cdot\frac56 = \frac{5}{36}$:
\[
\P\left(\abs{F_n - \tfrac16} \geq 0.05\right)
\leq \frac{5/36}{1200 \times 0.0025} = \frac{5}{108} \approx 0.046 .
\]
De waargenomen afwijking is precies $0.05$: een \omterm{def:g11:prob:model}{gebeurtenis} die een
eerlijke steen produceert met \omterm{def:g11:prob:model}{kans} hoogstens $4.6\%$ --- en omdat
Tsjebysjev zeer conservatief is, is de ware \omterm{def:g11:prob:model}{kans} veel kleiner. De
eerlijkheidshypothese is niet aannemelijk; de steen is zeer waarschijnlijk
oneerlijk.
\end{solution}

\begin{solution}{exo:g12:sums:8}
\emph{1.} $p(1-p) = \frac14 - \left(p - \frac12\right)^2 \leq \frac14$.

\emph{2.} $F_n$ heeft \omterm{def:g12:randvar:exp}{verwachting} $p$ en \omterm{def:g12:randvar:exp}{variantie}
$\frac{p(1-p)}{n} \leq \frac{1}{4n}$; Bienaymé--Tsjebysjev geeft
\[
\P\bigl(\abs{F_n - p} \geq \varepsilon\bigr)
\leq \frac{p(1-p)}{n\varepsilon^2} \leq \frac{1}{4n\varepsilon^2},
\]
geldig wat de onbekende $p$ ook is.

\emph{3.} Met $\varepsilon = 0.03$ en niveau $0.05$:
\[
\frac{1}{4n(0.03)^2} \leq 0.05
\iff n \geq \frac{1}{4 \times 0.0009 \times 0.05} \approx 5556 .
\]
Ongeveer $5600$ mensen volstaan volgens deze grove grens (het klassieke
normaal-benaderingsantwoord ligt dichter bij $1100$, zie
\cref{ch:g12:contdist}).
\end{solution}
