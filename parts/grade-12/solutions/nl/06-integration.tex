\section*{Hoofdstuk \ref{ch:g12:integ} --- Integratie}

\begin{solution}{exo:g12:integ:1}
\[
\int_1^2 \left(3x^2 - \frac{1}{x^2}\right)\dd x
= \left[x^3 + \frac1x\right]_1^2 = \left(8 + \tfrac12\right) - 2 = \frac{13}{2}.
\]
\[
\int_0^{\pi/2} \cos t\,\dd t = \bigl[\sin t\bigr]_0^{\pi/2} = 1 .
\]
Met $u = 2t+1$, $\frac{1}{2t+1} = \frac12\,\frac{u'}{u}$:
\[
\int_0^1 \frac{\dd t}{2t+1} = \frac12\bigl[\ln(2t+1)\bigr]_0^1
= \frac{\ln 3}{2}.
\]
\end{solution}

\begin{solution}{exo:g12:integ:2}
$x\,\eu^{x^2} = \frac12\,(x^2)'\,\eu^{x^2}$ heeft \omterm{def:g12:integ:primitive}{primitieven}
$\frac12\eu^{x^2} + c$. De voorwaarde $F(0) = 1$ geeft $\frac12 + c = 1$,
dus
\[
F(x) = \frac{\eu^{x^2} + 1}{2}.
\]
\end{solution}

\begin{solution}{exo:g12:integ:3}
\[
\mu = \frac{1}{\pi}\int_0^\pi \sin t\,\dd t
= \frac{1}{\pi}\bigl[-\cos t\bigr]_0^\pi = \frac{2}{\pi} \approx 0.64 .
\]
Grafisch heeft de rechthoek met basis $\intcc{0}{\pi}$ en hoogte
$\frac{2}{\pi}$ dezelfde oppervlakte als de boog van de sinuskromme.
\end{solution}

\begin{solution}{exo:g12:integ:4}
Neem $u' = 1$, $v = \ln t$, dus $u = t$, $v' = \frac1t$:
\[
\int_1^{\eu} \ln t\,\dd t = \bigl[t\ln t\bigr]_1^{\eu} - \int_1^{\eu} 1\,\dd t
= \eu - (\eu - 1) = 1 .
\]
Neem $u' = \sin t$, $v = t$, dus $u = -\cos t$, $v' = 1$:
\[
\int_0^\pi t\sin t\,\dd t = \bigl[-t\cos t\bigr]_0^\pi + \int_0^\pi \cos t\,\dd t
= \pi + 0 = \pi .
\]
\end{solution}

\begin{solution}{exo:g12:integ:5}
De krommen snijden waar $x^2 = x + 2$, d.w.z.\ $x \in \{-1, 2\}$, en op
$\intcc{-1}{2}$ ligt de lijn boven de \omterm{def:g11:quad:parabola}{parabool}. De oppervlakte is
\[
\int_{-1}^{2} \bigl(x + 2 - x^2\bigr)\dd x
= \left[\frac{x^2}{2} + 2x - \frac{x^3}{3}\right]_{-1}^{2}
= \frac{10}{3} - \left(-\frac{7}{6}\right) = \frac{9}{2}.
\]
\end{solution}

\begin{solution}{exo:g12:integ:6}
\emph{1.} $I = \bigl[\ln(1+t)\bigr]_0^1 = \ln 2$.

\emph{2.}
\[
I_n + I_{n+1} = \int_0^1 \frac{t^n(1 + t)}{1+t}\,\dd t
= \int_0^1 t^n \,\dd t = \frac{1}{n+1}.
\]
Op $\intcc{0}{1}$, $0 \leq \dfrac{t^n}{1+t} \leq t^n$, dus door
\omterm{def:g10:algebra:equation}{vergelijking} $0 \leq I_n \leq \frac{1}{n+1}$.

\emph{3.} Uit de recurrentie, door een telescoperende inductie,
\[
I_n = (-1)^n\left(I_0 - \left(1 - \frac12 + \dots + \frac{(-1)^{n-1}}{n}\right)\right),
\]
d.w.z.\
$\displaystyle\sum_{k=1}^{n} \frac{(-1)^{k-1}}{k} = \ln 2 - (-1)^n I_n$
(merk op $I_0 = \ln 2$). Omdat $\abs{I_n} \leq \frac{1}{n+1} \to 0$,
neigt de alternerende harmonische som naar $\ln 2$.
\end{solution}

\begin{solution}{exo:g12:integ:7}
Afstand:
\[
d = \int_0^{100} 30\left(1 - \eu^{-t/50}\right)\dd t
= 30\left[t + 50\,\eu^{-t/50}\right]_0^{100}
= 30\left(100 + 50\eu^{-2} - 50\right)
= 1500 + 1500\,\eu^{-2}.
\]
Numeriek $d \approx 1500 + 203 = 1703$ m. \omterm{def:g11:stat:mean}{Gemiddelde} snelheid:
$\frac{d}{100} = 15\bigl(1 + \eu^{-2}\bigr) \approx 17.0$ m/s.
\end{solution}

\begin{solution}{exo:g12:integ:8}
\emph{1.} $S_n$ is de totale oppervlakte van $n$ rechthoeken van breedte
$\frac1n$ en hoogtes $f\left(\frac kn\right)$, $k = 1, \dots, n$, waar
$f(t) = \frac{1}{1+t}$: het is de ``rechtereindpunt''-rechthoeksbenadering
van de oppervlakte onder $f$ op $\intcc{0}{1}$.

\emph{2.} $f$ is \omterm{def:g10:functions:variations}{dalend}, dus op elk \omterm{def:g10:numbers:interval}{interval}
$\intcc{\frac{k-1}{n}}{\frac kn}$,
$f\left(\frac kn\right) \leq f(t) \leq f\left(\frac{k-1}{n}\right)$;
integreren en sommeren over $k$:
\[
S_n \leq \int_0^1 f(t)\,\dd t \leq S_n + \frac1n\bigl(f(0) - f(1)\bigr)
= S_n + \frac{1}{2n}.
\]
(De middelste term van de oefening, met $f(0) - f(1) = 1 - \frac12$, is
precies deze grens.) Dus $0 \leq \ln 2 - S_n \leq \frac{1}{2n} \to 0$,
vandaar $S_n \to \ln 2$.
\end{solution}

\begin{solution}{exo:g12:integ:9}
\emph{1.} $W_0 = \int_0^{\pi/2} \dd t = \frac{\pi}{2}$ en
$W_1 = \bigl[-\cos t\bigr]_0^{\pi/2} = 1$.

\emph{2.} \omterm{thm:g12:integ:ibp}{Partieel integreren} met $u' = \sin t$, $v = \sin^{n+1} t$, dus
$u = -\cos t$, $v' = (n+1)\sin^n t \cos t$:
\[
W_{n+2} = \bigl[-\cos t \sin^{n+1} t\bigr]_0^{\pi/2}
+ (n+1)\int_0^{\pi/2} \cos^2 t\,\sin^n t\,\dd t
= (n+1)\int_0^{\pi/2} (1 - \sin^2 t)\sin^n t\,\dd t,
\]
de haak verdwijnt aan beide einden. Dus
$W_{n+2} = (n+1)(W_n - W_{n+2})$, d.w.z.\
$W_{n+2} = \frac{n+1}{n+2} W_n$.

\emph{3.} $W_2 = \frac12 W_0 = \frac{\pi}{4}$,
$W_3 = \frac23 W_1 = \frac23$,
$W_4 = \frac34 W_2 = \frac{3\pi}{16}$.
Op $\intoo{0}{\frac\pi2}$, $0 < \sin t < 1$, dus
$\sin^{n+1} t \leq \sin^n t$ met strikte ongelijkheid in het inwendige;
integreren, $0 < W_{n+1} \leq W_n$ (in feite $<$): $(W_n)$ is \omterm{def:g10:functions:variations}{dalend} en
positief.
\end{solution}
