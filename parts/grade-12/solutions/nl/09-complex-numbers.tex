\section*{Hoofdstuk \ref{ch:g12:complex} --- Complexe getallen}

\begin{solution}{exo:g12:complex:1}
$(3-2\iu)(1+4\iu) = 3 + 12\iu - 2\iu - 8\iu^2 = 11 + 10\iu$.

$\dfrac{2+\iu}{1-\iu} = \dfrac{(2+\iu)(1+\iu)}{(1-\iu)(1+\iu)}
= \dfrac{1 + 3\iu}{2} = \dfrac12 + \dfrac32\iu$.

$\iu^{2027}$: omdat $\iu^4 = 1$ en $2027 = 4 \times 506 + 3$,
$\iu^{2027} = \iu^3 = -\iu$.
\end{solution}

\begin{solution}{exo:g12:complex:2}
\emph{(a)} $\Delta = 16 - 52 = -36$:
$z = \dfrac{4 \pm 6\iu}{2} = 2 \pm 3\iu$. Geconjugeerd paar; product
$= 4 + 9 = 13 = \frac ca$. \emph{(b)} $\Delta = 1 - 4 = -3$:
$z = \dfrac{-1 \pm \iu\sqrt3}{2}$ (de derdemachtswortels van de eenheid $j$
en $\conj j$); product $= \frac{1 + 3}{4} = 1$.
\end{solution}

\begin{solution}{exo:g12:complex:3}
$\abs{z_1} = \sqrt2$, argument $\frac{3\pi}{4}$ (tweede kwadrant):
$z_1 = \sqrt2\,\eu^{3\iu\pi/4}$.
$\abs{z_2} = 2$, argument $-\frac\pi6$:
$z_2 = 2\,\eu^{-\iu\pi/6}$. Dus
\[
\frac{z_1}{z_2} = \frac{\sqrt2}{2}\,\eu^{\iu\left(\frac{3\pi}{4} + \frac{\pi}{6}\right)}
= \frac{\sqrt2}{2}\,\eu^{11\iu\pi/12}.
\]
Algebraïsch,
\[
\frac{z_1}{z_2} = \frac{-1+\iu}{\sqrt3 - \iu}
= \frac{(-1+\iu)(\sqrt3+\iu)}{4}
= \frac{-\sqrt3 - 1 + \iu(\sqrt3 - 1)}{4}.
\]
Identificeren van reële delen:
$\frac{\sqrt2}{2}\cos\frac{11\pi}{12} = \frac{-\sqrt3-1}{4}$, dus
\[
\cos\frac{11\pi}{12} = -\frac{\sqrt6 + \sqrt2}{4}.
\]
\end{solution}

\begin{solution}{exo:g12:complex:4}
\emph{(a)} $\abs{z - 2} = \abs{z - (-\iu)}$ betekent dat $M$ equidistant is
van $A(2, 0)$ en $B(0, -1)$: de \emph{middelloodlijn} van $[AB]$.

\emph{(b)} Afstand $3$ tot het punt $(1,1)$: de \emph{cirkel} met
middelpunt $1 + \iu$ en straal $3$.

\emph{(c)} Volgens \cref{prop:g12:complex:geometry}(4) betekent
$\frac{z-1}{z+1}$ zuiver imaginair dat de vectoren van $A(1,0)$ naar $M$
en van $B(-1,0)$ naar $M$ loodrecht zijn: $M$ ziet het segment $[AB]$ onder
een rechte hoek. De verzameling is de \emph{cirkel met diameter $[AB]$}
(de eenheidscirkel) \emph{minus de punten $A$ en $B$} zelf.
\end{solution}

\begin{solution}{exo:g12:complex:5}
De Moivre: $\cos3\theta + \iu\sin3\theta = (\cos\theta + \iu\sin\theta)^3$.
Ontwikkelen met de binomiaalstelling met $c = \cos\theta$, $s = \sin\theta$:
\[
(c + \iu s)^3 = c^3 + 3\iu c^2 s - 3 c s^2 - \iu s^3
= (c^3 - 3cs^2) + \iu(3c^2 s - s^3).
\]
Delen identificeren en $s^2 = 1 - c^2$, $c^2 = 1 - s^2$ gebruiken:
\[
\cos3\theta = 4\cos^3\theta - 3\cos\theta,
\qquad
\sin3\theta = 3\sin\theta - 4\sin^3\theta .
\]
\end{solution}

\begin{solution}{exo:g12:complex:6}
\[
\cos^3\theta = \left(\frac{\eu^{\iu\theta} + \eu^{-\iu\theta}}{2}\right)^{\!3}
= \frac{\eu^{3\iu\theta} + 3\eu^{\iu\theta} + 3\eu^{-\iu\theta} + \eu^{-3\iu\theta}}{8}
= \frac{\cos3\theta + 3\cos\theta}{4}.
\]
Dus
\[
\int_0^{\pi/2}\cos^3\theta\,\dd\theta
= \frac14\left[\frac{\sin3\theta}{3} + 3\sin\theta\right]_0^{\pi/2}
= \frac14\left(-\frac13 + 3\right) = \frac{2}{3}.
\]
(Consistent met $W_3 = \frac23$ in \cref{exo:g12:integ:9}, omdat
$\int_0^{\pi/2}\cos^3 = \int_0^{\pi/2}\sin^3$ door de symmetrie
$\theta \mapsto \frac\pi2 - \theta$.)
\end{solution}

\begin{solution}{exo:g12:complex:7}
$c = a + \eu^{\pm\iu\pi/3}(b - a)$ met $b - a = 2 + \iu$ en
$\eu^{\pm\iu\pi/3} = \frac12 \pm \iu\frac{\sqrt3}{2}$:
\[
\eu^{\iu\pi/3}(2+\iu) = \left(\tfrac12 + \iu\tfrac{\sqrt3}{2}\right)(2+\iu)
= \left(1 - \tfrac{\sqrt3}{2}\right) + \iu\left(\tfrac12 + \sqrt3\right),
\]
dus $c_1 = \left(2 - \frac{\sqrt3}{2}\right) +
\iu\left(\frac32 + \sqrt3\right)$, en evenzo met de hoek
$-\frac\pi3$:
$c_2 = \left(2 + \frac{\sqrt3}{2}\right) + \iu\left(\frac32 - \sqrt3\right)$.
\end{solution}

\begin{solution}{exo:g12:complex:8}
$-4 = 4\,\eu^{\iu\pi}$, dus de oplossingen van $z^4 = -4$ zijn
\[
z_k = \sqrt2\,\eu^{\iu\left(\frac\pi4 + \frac{k\pi}{2}\right)},
\quad k = 0, 1, 2, 3,
\]
\emph{d.w.z.}\ $1 + \iu$, $-1 + \iu$, $-1 - \iu$, $1 - \iu$: de
hoekpunten van een vierkant met omgeschreven straal $\sqrt 2$.
Geconjugeerde wortels groeperen:
\[
z^4 + 4 = \bigl(z - (1{+}\iu)\bigr)\bigl(z - (1{-}\iu)\bigr)
\bigl(z + (1{-}\iu)\bigr)\bigl(z + (1{+}\iu)\bigr)
= (z^2 - 2z + 2)(z^2 + 2z + 2).
\]
\end{solution}

\begin{solution}{exo:g12:complex:9}
Met $q = \eu^{\iu\theta} \neq 1$:
\[
\sum_{k=0}^{n} \eu^{\iu k\theta} = \frac{\eu^{\iu(n+1)\theta} - 1}{\eu^{\iu\theta} - 1}
= \frac{\eu^{\iu(n+1)\theta/2}}{\eu^{\iu\theta/2}}
\cdot\frac{\eu^{\iu(n+1)\theta/2} - \eu^{-\iu(n+1)\theta/2}}{\eu^{\iu\theta/2} - \eu^{-\iu\theta/2}}
= \eu^{\iu n\theta/2}\,
\frac{\sin\frac{(n+1)\theta}{2}}{\sin\frac{\theta}{2}},
\]
met $\eu^{\iu\alpha} - \eu^{-\iu\alpha} = 2\iu\sin\alpha$ (de
\emph{halve-hoekfactorisatie}). Reële delen nemen:
\[
S = \frac{\sin\frac{(n+1)\theta}{2}}{\sin\frac{\theta}{2}}\,
\cos\frac{n\theta}{2}.
\]
\end{solution}

\begin{solution}{exo:g12:complex:10}
\emph{1.} $\omega$ is een $5$-de eenheidswortel verschillend van $1$, dus
de som van alle vijf wortels verdwijnt
(\cref{thm:g12:complex:rootsofunity}), wat
$1 + \omega + \omega^2 + \omega^3 + \omega^4 = 0$ luidt.

\emph{2.} $u + v = \omega + \omega^2 + \omega^3 + \omega^4 = -1$ door
punt 1. Voor het product, met $\omega^5 = 1$:
\[
uv = (\omega + \omega^4)(\omega^2 + \omega^3)
= \omega^3 + \omega^4 + \omega^6 + \omega^7
= \omega^3 + \omega^4 + \omega + \omega^2 = -1 .
\]
Dus $u, v$ zijn de wortels van $X^2 + X - 1 = 0$:
$\left\{\frac{-1+\sqrt5}{2}, \frac{-1-\sqrt5}{2}\right\}$. Nu
$u = \omega + \conj\omega = 2\cos\frac{2\pi}{5} > 0$ (omdat
$\frac{2\pi}{5} < \frac{\pi}{2}$), dus $u = \frac{-1 + \sqrt5}{2}$.

\emph{3.} $\cos\frac{2\pi}{5} = \frac u2 = \frac{\sqrt5 - 1}{4}$.
\end{solution}
