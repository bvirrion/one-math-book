\section*{Chapter \ref{ch:g12:randvar} --- Random Variables and the Binomial Distribution}

\begin{solution}{exo:g12:randvar:1}
$\E(X) = -0.3 + 0 + 0.8 + 0.5 = 1$.
$\E(X^2) = 0.3 \times 1 + 0 + 0.4\times4 + 0.1\times25 = 4.4$, so by
König--Huygens $\V(X) = 4.4 - 1 = 3.4$ and
$\sigma(X) = \sqrt{3.4} \approx 1.84$.
\end{solution}

\begin{solution}{exo:g12:randvar:2}
\emph{1.} The $10$ questions are \omterm{def:g12:condprob:indep}{independent} \omterm{def:g12:randvar:bernoulli}{Bernoulli trials} with
$p = \frac14$: $X \sim \mathcal B\left(10, \frac14\right)$,
$\E(X) = 2.5$,
$\sigma(X) = \sqrt{10 \times \frac14 \times \frac34} = \sqrt{1.875}
\approx 1.37$.

\emph{2.} $\P(X = 0) = \left(\frac34\right)^{10} \approx 0.056$;
\[
\P(X = 5) = \binom{10}{5}\left(\frac14\right)^5\left(\frac34\right)^5
\approx 0.058;
\quad
\P(X \geq 1) = 1 - \left(\tfrac34\right)^{10} \approx 0.944 .
\]
\end{solution}

\begin{solution}{exo:g12:randvar:3}
\omterm{def:g12:condprob:indep}{Independent} identical trials: $X \sim \mathcal B(400,\ 0.05)$, so
$\E(X) = 20$ claims and
\[
\sigma(X) = \sqrt{400 \times 0.05 \times 0.95} = \sqrt{19} \approx 4.4 .
\]
\end{solution}

\begin{solution}{exo:g12:randvar:4}
The payment $P$ received satisfies
$\E(P) = \frac{5 + 6}{6} = \frac{11}{6}$, so the fair price is
$c = \frac{11}{6} \approx 1.83$ euros. The fair gain $G = P - \frac{11}6$
takes values $-\frac{11}{6}, \frac{19}{6}, \frac{25}{6}$ with probabilities
$\frac46, \frac16, \frac16$:
\[
\V(G) = \E(G^2)
= \frac{4 \times 121 + 361 + 625}{6 \times 36} = \frac{1470}{216}
= \frac{245}{36} \approx 6.8,
\qquad \sigma(G) \approx 2.6 .
\]
A fair game has zero \emph{average} gain but its outcomes still fluctuate:
fair is not riskless.
\end{solution}

\begin{solution}{exo:g12:randvar:5}
$X \sim \mathcal B(8,\ 0.7)$.
\[
\P(X = 6) = \binom86 (0.7)^6(0.3)^2 \approx 0.296 ;
\]
\[
\P(X \geq 6) = \P(6) + \P(7) + \P(8)
\approx 0.296 + 8(0.7)^7(0.3) + (0.7)^8
\approx 0.296 + 0.198 + 0.058 = 0.552 ;
\]
$\P(X \geq 1) = 1 - (0.3)^8 \approx 0.99993$. Mode:
$(n+1)p = 6.3$, so the most probable value is $k^* = 6$
(\cref{exo:g12:randvar:7}).
\end{solution}

\begin{solution}{exo:g12:randvar:6}
The number of passengers showing up is $X \sim \mathcal B(104,\ 0.9)$; the
flight is overbooked when $X \geq 101$:
\[
\P(X \geq 101) = \sum_{k=101}^{104}\binom{104}{k}(0.9)^k(0.1)^{104-k}
\approx 0.006 .
\]
Selling $4\%$ more tickets than seats causes an incident on only about
$0.6\%$ of flights --- the economics behind overbooking.
\end{solution}

\begin{solution}{exo:g12:randvar:7}
\[
\frac{\P(X = k+1)}{\P(X = k)}
= \frac{\binom{n}{k+1}}{\binom nk}\cdot\frac{p}{1-p}
= \frac{n - k}{k + 1}\cdot\frac{p}{1-p},
\]
using $\binom{n}{k+1} = \binom nk \frac{n-k}{k+1}$. This ratio is $\geq 1$
iff $(n-k)p \geq (k+1)(1-p)$ iff $np - k p \geq k - kp + 1 - p$ iff
$k \leq (n+1)p - 1$. So the probabilities increase strictly while
$k + 1 \leq (n+1)p$ and decrease afterwards: the \omterm{def:g10:functions:extrema}{maximum} is attained at
$k^* = \floor{(n+1)p}$ (shared with $k^* - 1$ when $(n+1)p$ is an
\omterm{def:g10:numbers:sets}{integer}).
\end{solution}

\begin{solution}{exo:g12:randvar:8}
\emph{1.} Heads first at toss $k$ means $k-1$ tails then heads:
\omterm{def:g10:proba:distribution}{probability} $\left(\frac12\right)^{k-1}\cdot\frac12 = 2^{-k}$, for
$1 \leq k \leq 10$. Total winning \omterm{def:g10:proba:distribution}{probability}
$\sum_{k=1}^{10} 2^{-k} = 1 - 2^{-10} = \frac{1023}{1024}$ (the game is
lost only on ten consecutive tails).

\emph{2.} Expected payoff:
\[
\sum_{k=1}^{10} 2^k \cdot 2^{-k} = \sum_{k=1}^{10} 1 = 10 \text{ euros}.
\]
Without the cap, the sum $\sum_{k\geq1} 1$ diverges: the expected payoff is
infinite, although the game almost always pays a small amount --- the
famous \emph{Saint Petersburg paradox}, showing that \omterm{def:g12:randvar:exp}{expectation} alone does
not measure the value of a game.
\end{solution}

\begin{solution}{pb:g12:randvar:1}
\textbf{1.} $X \sim \mathcal B\!\left(3, \frac16\right)$:
$\P(X = 0) = \frac{125}{216}$,
$\P(X = 1) = \frac{75}{216}$,
$\P(X = 2) = \frac{15}{216}$,
$\P(X = 3) = \frac{1}{216}$.

\textbf{2.} $\E(X) = 3 \times \frac16 = \frac12$;
$V(X) = 3 \times \frac16 \times \frac56 = \frac{5}{12}$.

\textbf{3.} Expected payout $2\,\E(X) = 1$ euro against a
$1$-euro stake: gain $0$ --- exactly fair (a rarity).

\textbf{4.} Without replacement the draws are dependent (the
second card's chances depend on the first): not binomial ---
the checklist's \omterm{def:g12:condprob:indep}{independence} box fails. With replacement:
fixed $n = 5$, constant $p = \frac14$, \omterm{def:g12:condprob:indep}{independent} draws:
binomial.

\textbf{5.} $\E(X) = 6$, $\sigma(X) = \sqrt{4.2} \approx
2.05$. $\E(S) = 5 \times 6 - 10 = 20$;
$\sigma(S) = 5\sigma(X) \approx 10.25$.

\textbf{6.} Fixed $n = 105$ trials (the tickets), each a
show/no-show with the same $p = 0.9$, assumed \omterm{def:g12:condprob:indep}{independent}:
binomial. The confession: families miss flights
\emph{together} and storms empty whole planes --- \omterm{def:g12:condprob:indep}{independence}
is the model's leap of faith, and correlated no-shows make
the tails fatter than the binomial promises.

\textbf{7.} $\E(X) = 94.5$;
$\sigma(X) = \sqrt{105 \times 0.9 \times 0.1} =
\sqrt{9.45} \approx 3.07$.

\textbf{8.} $\P(X \geq 101) = \sum_{k=101}^{105}
\binom{105}{k} 0.9^k\, 0.1^{105 - k}$.

\textbf{9.} $\approx 0.0167$: about one flight in sixty
bumps anyone at all.

\textbf{10.} $\E(\text{bumped}) \approx 0.023$ passengers per
flight: expected compensation $\approx 19$ euros --- against
$1\,000$ euros of extra revenue. Overbooking by five is
overwhelmingly profitable; hence every airline does it.

\textbf{11.} From the table of tail probabilities:
$n = 105$: $1.7\,\%$; $n = 106$: $4.0\,\%$; $n = 107$:
$8.1\,\%$. The largest compliant sale is $n = 106$ tickets.

\textbf{12.} $z = \frac{100.5 - 94.5}{3.07} \approx 1.95$:
the threshold sits two \omterm{def:g11:stat:variance}{standard deviations} above the \omterm{def:g11:stat:mean}{mean},
and the bell-curve rule of thumb (``upper $2\sigma$ tail
$\approx 2.5\,\%$'') lands close to the exact $1.7\,\%$ ---
the smooth curve shadowing the binomial is the coming
chapters' protagonist.

\textbf{13.} $0.98^{20} + 20 \times 0.02 \times 0.98^{19}
\approx 0.94$: the honest lot passes $94\,\%$ of the time.

\textbf{14.} $0.9^{20} + 20 \times 0.1 \times 0.9^{19}
\approx 0.39$: the bad lot still sneaks through $39\,\%$ of
the time.

\textbf{15.} Producer's risk: a good lot rejected
($\approx 6\,\%$); consumer's risk: a bad lot accepted
($\approx 39\,\%$). A larger \omterm{def:g10:proba:sample}{sample} (with a proportional
acceptance threshold) shrinks both --- at the cost of more
inspection: quality has a budget line.

\textbf{16.} $V\!\left(\frac Xn\right) = \frac{V(X)}{n^2} =
\frac{np(1-p)}{n^2} = \frac{p(1-p)}{n}$: \omterm{def:g11:stat:variance}{standard deviation}
$\sqrt{\frac{p(1-p)}{n}}$. Quadruple the \omterm{def:g10:proba:sample}{sample}, halve the
noise: the $\frac{1}{\sqrt n}$ law of grade 10's fluctuation
\omterm{def:g10:numbers:interval}{intervals}, now derived.

\textbf{17.} The cap bounds the payout at $2^{10} = 1024$
euros, so each of the ten rounds contributes exactly $1$ euro
of \omterm{def:g12:randvar:exp}{expectation}: $\E = 10$. The uncapped game's infinite
\omterm{def:g12:randvar:exp}{expectation} came entirely from astronomically rare,
astronomically large payoffs; capping confesses that no bank
pays $2^{50}$ euros. A fair ticket price for the capped game:
$10$ euros --- and suddenly nobody is paradoxed.

\textbf{18.} Expected prize money per ticket:
$\frac{100 + 2 \times 50}{200} = 1$ euro against a $2$-euro
ticket: margin $50\,\%$ --- eighteen times roulette's
$2.7\,\%$. The raffle survives because its players are
knowingly donating: the ``loss'' is the point.

\textbf{19.} Expected claims: $1\,000 \times 0.01 \times
10\,000 = 100\,000$; premiums $130\,000$: expected profit
$30\,000$ euros. \omterm{def:g11:stat:variance}{Standard deviation} of total claims:
$10\,000 \times \sqrt{1\,000 \times 0.01 \times 0.99}
\approx 31\,500$ euros --- one ordinary bad year devours the
whole expected profit. Hence capital reserves, reinsurance,
and portfolios far larger than a thousand policies: insurers
live off the law of large numbers and keep reserves against
its slowness.

\textbf{20.} Checklist first --- fixed $n$, same $p$,
\omterm{def:g12:condprob:indep}{independence} \emph{claimed} --- or no binomial at all. Then
navigate by $\E \pm \sigma$: means locate, deviations warn.
Then price the tails: bumping, bad lots, ruinous years all
live beyond $2\sigma$. Caveats: \omterm{def:g12:condprob:indep}{independence} is the modeler's
promise, not the world's; and \omterm{def:g12:randvar:exp}{expectation} ignores exactly
what destroys you. The rulebook's missing pages --- how fast
frequencies settle, and what shape the fluctuations take ---
are the next two chapters.
\end{solution}
