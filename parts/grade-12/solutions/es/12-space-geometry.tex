\begin{solution}{exo:g12:space:5}
\omterm{def:g12:space:normal}{vectores normales} $(1, -1, 2)$ y $(2, 1, 1)$ no son \omterm{def:g12:space:collinear}{colineal}, por lo que
los planos se cruzan en una recta. Sumando los dos \omterm{def:g10:algebra:equation}{ecuaciones}: $3x + 3z = 5$, entonces
$x = \frac53 - z$; entonces el primer \omterm{def:g10:algebra:equation}{ecuación} da
$y = x + 2z - 1 = \frac23 + z$. Con $z = t$:
\[
(x, y, z) = \left(\frac53 - t,\; \frac23 + t,\; t\right), \qquad t \in \R,
\]
una línea con \omterm{def:g11:vect:direction}{vector de direccion} $(-1, 1, 1)$. (Ambos \omterm{def:g10:algebra:equation}{ecuaciones} comprueban
idénticamente.)
\end{solution}

\begin{solution}{exo:g12:space:6}
Las direcciones $\vec u_1(1, 2, -1)$ y $\vec u_2(1, 1, 2)$ no son \omterm{def:g12:space:collinear}{colineal}.
entonces las rectas no son paralelas. La intersección requeriría
\[
1 + t = 2 + s, \qquad 2t = 1 + s, \qquad 3 - t = 1 + 2s .
\]
Los dos primeros dan $t = 1 + s$ y $2(1+s) = 1 + s$, por lo que $s = -1$, $t = 0$;
pero luego el tercero dice $3 = -1$, absurdo. Ningún punto en común y no
paralelas: las rectas están sesgadas.
\end{solution}

\begin{solution}{exo:g12:space:7}
\emph{1.} $\operatorname{dist} = \dfrac{\abs{2\cdot3 - 1 + 2\cdot1 - 3}}
{\sqrt{4 + 1 + 4}} = \dfrac{4}{3}$.

\emph{2.} $H = M_0 + t\,\vec n$ con $\vec n(2, -1, 2)$, eligiendo $t$ para
que $H \in \mathcal P$:
\[
2(3 + 2t) - (1 - t) + 2(1 + 2t) - 3 = 4 + 9t = 0 \iff t = -\frac49 .
\]
Por lo tanto $H\left(\frac{19}{9}, \frac{13}{9}, \frac19\right)$, y
\[
M_0H = \norm{t\,\vec n} = \frac49 \times 3 = \frac43,
\]
coincide con la fórmula de la distancia.
\end{solution}

\begin{solution}{exo:g12:space:8}
\omterm{def:g10:coordgeom:system}{Coordenadas}: $G(1,1,1)$, y el avión $(BDE)$ pasa por $B(1,0,0)$,
$D(0,1,0)$, $E(0,0,1)$.

\emph{1.} El avión $(BDE)$ tiene \omterm{def:g10:algebra:equation}{ecuación} $x + y + z = 1$ (satisfecho con el
tres puntos y no están alineados), por lo que $\vec n(1,1,1)$ es normal
eso. Desde $\vect{AG} = (1,1,1) = \vec n$, la diagonal $(AG)$ es \omterm{def:g11:scal:orthogonal}{ortogonal}
al avión $(BDE)$.

\emph{2.} Distancia desde $A(0,0,0)$:
$\frac{\abs{0 + 0 + 0 - 1}}{\sqrt3} = \frac{1}{\sqrt3} = \frac{\sqrt3}{3}$.
La línea $(AG)$ es $(t, t, t)$; se encuentra con el avión cuando $3t = 1$: en el
punto $\left(\frac13, \frac13, \frac13\right)$, el centroide del
triángulo $BDE$.
\end{solution}

\begin{solution}{exo:g12:space:9}
\emph{1.} $\vect{AB}(1, 0, -1)$, $\vect{AC}(-1, 1, 0)$,
$\vect{AD}(1, 1, 1)$. Si $\vect{AD} = a\vect{AB} + b\vect{AC}$, el
\omterm{def:g10:coordgeom:system}{coordenadas} dan $a - b = 1$, $b = 1$, $-a = 1$: los dos primeros fuerzan
$a = 2$, contradiciendo $a = -1$. No \omterm{def:g12:space:collinear}{coplanar}.

\emph{2.} $\vect{BC}(-2, 1, 1)$, $\vect{BD}(0, 1, 2)$. Resolviendo
$\vec n \cdot \vect{BC} = \vec n \cdot \vect{BD} = 0$: desde
$b + 2c = 0$, $b = -2c$; entonces $-2a - 2c + c = 0$ da $c = -2a$. con
$a = 1$: $\vec n(1, 4, -2)$. Avión a través de $B(2,1,0)$:
\[
x + 4y - 2z - 6 = 0 .
\]

\emph{3.} Distancia desde $A(1,1,1)$:
$\dfrac{\abs{1 + 4 - 2 - 6}}{\sqrt{21}} = \dfrac{3}{\sqrt{21}}$.
Área de $BCD$: $\norm{\vect{BC}}^2 = 6$, $\norm{\vect{BD}}^2 = 5$,
$\vect{BC}\cdot\vect{BD} = 3$, entonces
\[
\text{Area} = \frac12\sqrt{6 \times 5 - 9} = \frac{\sqrt{21}}{2}.
\]

\emph{4.}
\[
V = \frac13 \times \frac{\sqrt{21}}{2} \times \frac{3}{\sqrt{21}}
= \frac12 .
\]
\end{solution}

\begin{solution}{pb:g12:space:1}
\textbf{1.} Avión: $x + y + z = 1$. Línea: $(t, t, t)$.
\omterm{def:g10:numbers:interunion}{Intersección}: $3t = 1$:
$\left(\frac13, \frac13, \frac13\right)$.

\textbf{2.} $\dfrac{\abs{0 + 0 + 0 - 1}}{\sqrt3} =
\dfrac{1}{\sqrt3} = \dfrac{\sqrt3}{3}$.

\textbf{3.} $1 - 1 + 0 = 0$: \omterm{def:g11:scal:orthogonal}{ortogonal}.

\textbf{4.} Lo mismo $(1,1,1)$ normal, pero $\frac52 \neq 1$:
planos estrictamente paralelos. La línea: dirección
$(1,1,0) \cdot (1,1,1) = 2 \neq 0$: no paralelo al plano,
entonces lo perfora en un punto.

\textbf{5.} \omterm{prop:g10:coordgeom:midpoint}{Puntos medios}: $\left(1, \frac12, 0\right)$,
$\left(\frac12, 1, 0\right)$, $\left(0, 1, \frac12\right)$,
$\left(0, \frac12, 1\right)$, $\left(\frac12, 0, 1\right)$,
$\left(1, 0, \frac12\right)$: cada uno tiene suma de coordenadas
$\frac32$.

\textbf{6.} $P$ es normal $(1,1,1)$, la dirección de
$(AG)$: perpendicular. el centro
$\left(\frac12,\frac12,\frac12\right)$ tiene suma de coordenadas
$\frac32$: en $P$.

\textbf{7.} \omterm{prop:g10:coordgeom:midpoint}{puntos medios} consecutivos difieren por \omterm{def:g11:vect:vector}{vectores} como
$\left(-\frac12, \frac12, 0\right)$: longitud
$\frac{\sqrt2}{2}$ cada vez --- seis lados iguales.

\textbf{8.} En $\left(\frac12, 1, 0\right)$: \omterm{def:g11:vect:vector}{vectores-de-borde}
$\left(\frac12, -\frac12, 0\right)$ y
$\left(-\frac12, 0, \frac12\right)$: producto escalar $-\frac14$,
\omterm{def:g11:scal:dot}{normas} $\frac{\sqrt2}{2}$: $\cos = -\frac12$: ángulo
$120^\circ$. Lados iguales, ángulos iguales $120^\circ$: un regular
hexágono.

\textbf{9.} Perímetro $6 \times \frac{\sqrt2}{2} = 3\sqrt2
\approx 4.24$. Área $6 \times \frac{\sqrt3}{4} \times
\frac12 = \frac{3\sqrt3}{4} \approx 1.30$: más grande que una cara
(área $1$) --- una porción más grande que cualquier lado de la caja; solo
el rectángulo diagonal ($\sqrt2 \approx 1.41$) lo supera.

\textbf{10.} $(t,t,t)$ se encuentra con $P$ en $t = \frac12$: el centro
--- que es el \omterm{prop:g10:coordgeom:midpoint}{punto medio} de $[AG]$ ($A$ en $t = 0$, $G$ en
$t = 1$).

\textbf{11.} Para $c = \frac12$: el plano corta los tres bordes
en la esquina de $A$, en $\left(\frac12,0,0\right)$,
$\left(0,\frac12,0\right)$, $\left(0,0,\frac12\right)$: un
triángulo equilátero de lado $\frac{\sqrt2}{2}$. A medida que $c$ crece,
el triángulo crece, sus esquinas se truncan en un hexágono
(normal exactamente en $c = \frac32$), luego la imagen se reduce
simétricamente de regreso a un triángulo en la esquina de $G$: el
transformación tallada en cristal.

\textbf{12.} Cada lado de la sección es el \omterm{def:g10:numbers:interunion}{intersección} de
el plano de corte con un \emph{rostro} del cubo, y un
El plano se encuentra con una cara (un polígono plano) en como máximo un segmento:
como máximo $6$ lados. Siete es imposible; tres a seis todos
ocurren (las preguntas 8 y 11 muestran dos de ellos).

\textbf{13.} Base $ABD$: triángulo rectángulo del área $\frac12$;
altura $AE = 1$: volumen
$\frac13 \times \frac12 \times 1 = \frac16$.

\textbf{14.} Cada par entre $B, D, E, G$ difiere exactamente en
dos \omterm{def:g10:coordgeom:system}{coordenadas} por $1$: las seis distancias son iguales $\sqrt2$ ---
regulares. El cubo se divide en $BDEG$ más cuatro esquinas.
tetraedros congruentes con $ABDE$: volumen
$1 - 4 \times \frac16 = \frac13$.

\textbf{15.} Avión $(BDE)$: $x + y + z = 1$; centro
$\left(\frac12,\frac12,\frac12\right)$: distancia
$\frac{\abs{\frac32 - 1}}{\sqrt3} = \frac{1}{2\sqrt3} =
\frac{\sqrt3}{6}$.

\textbf{16.}$\vect{AG} = (1,1,1)$,
$\vect{BH} = (-1,1,1)$: $\cos\theta = \frac{-1 + 1 + 1}{3} =
\frac13$: $\theta \approx 70.5^\circ$, suplemento
$\approx 109.5^\circ$. Cuatro pares de electrones se repelen
y esparcidos lo más lejos posible alrededor del carbono: el
óptimo son las esquinas del tetraedro regular, cuyas
las direcciones del centro al vértice se encuentran exactamente en este ángulo ---
el diamante es la pregunta 14 cristalizado.

\textbf{17.} $(1, t, t)$ en $P$: $1 + 2t = \frac32$:
$t = \frac14$: el punto $\left(1, \frac14, \frac14\right)$.

\textbf{18.} $(AB)$: puntos $(t, 0, 0)$; $(EG)$: puntos
$(s, s, 1)$. La reunión necesitaría $0 = 1$ en el tercero.
coordinar: nunca. Paralelo necesitaría $(1,0,0)$ y
$(1,1,0)$ \omterm{def:g12:space:collinear}{colineal}: no. Ni encuentro ni paralelo: sesgo
--- dos pasillos en diferentes pisos.

\textbf{19.} $B + t(1,1,1) = (1 + t, t, t)$ en $P$:
$1 + 3t = \frac32$: $t = \frac16$: proyección
$\left(\frac76, \frac16, \frac16\right)$, cuya suma de coordenadas
es $\frac32$: en $P$, según sea necesario.

\textbf{20.} Perforación de líneas paramétricas (preguntas 1, 10, 17);
\omterm{def:g12:space:normal}{vectores normales} medir ángulos y distancias (preguntas 2, 8,
15, 16); plano \omterm{def:g10:algebra:equation}{ecuaciones} tallar secciones (el triángulo a
transformación hexagonal). Y el parque infantil retribuyó la visita: un
hexágono regular en una caja de cuadrados, un tetraedro regular en
las esquinas, el ángulo del diamante y dos líneas que ignoran
entre sí --- la geometría que sólo el espacio puede ofrecer.
\end{solution}
