\section*{Capítulo \ref{ch:g12:limcont} --- Límites y continuidad}

\begin{solution}{exo:g12:limcont:1}
Dividiendo por $x^3$:
$\dfrac{2x^3 - x + 1}{5x^3 + x^2} = \dfrac{2 - 1/x^2 + 1/x^3}{5 + 1/x}
\to \dfrac{2}{5}$.

$x^2 + 3x - 1 = x^2(1 + 3/x - 1/x^2) \to +\infty$ como $x \to -\infty$, desde
$x^2 \to +\infty$ y el corchete tiende a $1$.

Como $x \to 2^+$, el numerador tiende a $3 > 0$ y el denominador a $0^+$,
entonces $\dfrac{x+1}{x-2} \to +\infty$.

Para $x > 0$, $-\dfrac1x \leq \dfrac{\cos x}{x} \leq \dfrac1x$, entonces el límite
es $0$ según el teorema de compresión.
\end{solution}

\begin{solution}{exo:g12:limcont:2}
\emph{1.} $2x^2 - 3x + 1 = (x-1)(2x-1)$ (el \omterm{def:g11:quad:discriminant}{raíz} $x=1$ es evidente en
$2 - 3 + 1 = 0$). Para $x \neq 1$, cancele $x - 1$: $f(x) = 2x - 1$.

\emph{2.} Por lo tanto $\lim_{x\to1} f(x) = 1$. La configuración $f(1) = 1$ extiende $f$ a
el polinomio \omterm{def:g11:func:function}{función} $x \mapsto 2x-1$ en $\R$, que es \omterm{def:g12:limcont:continuity}{continuo} en $1$.
\end{solution}

\begin{solution}{exo:g12:limcont:3}
Escribe $f(x) = \dfrac{3x - 1}{x + 2} = \dfrac{3(x+2) - 7}{x+2}
= 3 - \dfrac{7}{x+2}$. Como $x \to \pm\infty$, $f(x) \to 3$: la línea $y = 3$
es un \omterm{def:g12:limcont:asymptote}{asíntota horizontal} en ambos infinitos. Como $x \to (-2)^\pm$,
$f(x) \to \mp\infty$: la línea $x = -2$ es una \omterm{def:g12:limcont:asymptote}{asíntota vertical}. la curva es
a \omterm{def:g11:func:reference}{hipérbola}, debajo de $y=3$ para $x > -2$ y por encima de él para $x < -2$.
\end{solution}

\begin{solution}{exo:g12:limcont:4}
Multiplicar por el conjugado:
\[
\sqrt{x^2 + x} - x = \frac{x^2 + x - x^2}{\sqrt{x^2+x} + x}
= \frac{x}{x\left(\sqrt{1 + 1/x} + 1\right)}
= \frac{1}{\sqrt{1 + 1/x} + 1}
\xrightarrow[x\to+\infty]{} \frac{1}{2}.
\]
De manera similar,
\[
\frac{\sqrt{1+x} - 1}{x} = \frac{(1+x) - 1}{x\left(\sqrt{1+x} + 1\right)}
= \frac{1}{\sqrt{1+x} + 1} \xrightarrow[x\to0]{} \frac{1}{2}.
\]
\end{solution}

\begin{solution}{exo:g12:limcont:5}
\emph{1.} Desde $\sin x \geq -1$, $f(x) \geq x - 1 \to +\infty$, y nosotros
concluir por comparación. Sin embargo, $f'(x) = 1 + \cos x$ desaparece en cada momento.
múltiplo de $\pi$, y en cada \omterm{def:g10:numbers:interval}{intervalo} $\intco{A}{+\infty}$ el \omterm{def:g11:func:function}{función}
alterna entre \omterm{def:g10:numbers:interval}{intervalos} de aumento estricto; es \omterm{def:g12:seq:monotonic}{creciente} pero no
estrictamente, y lo que es más importante, su crecimiento no es necesario: basta con comparar.

\emph{2.} Para $x > 1$, divida por $x$:
\[
g(x) = \frac{1 + \frac{\sin x}{x}}{1 - \frac{\sin x}{x}},
\]
y $\frac{\sin x}{x} \to 0$ por el teorema de compresión
($\abs{\sin x} \leq 1$). Por lo tanto $g(x) \to \frac{1+0}{1-0} = 1$.
\end{solution}

\begin{solution}{exo:g12:limcont:6}
$f(x) = x^3 - 3x + 1$ es \omterm{def:g12:limcont:continuity}{continuo}, con $f'(x) = 3x^2 - 3 = 3(x-1)(x+1)$:
$f$ aumenta en $\intoc{-\infty}{-1}$, disminuye en $\intcc{-1}{1}$,
aumenta en $\intco{1}{+\infty}$, con \omterm{def:g10:functions:extrema}{máximo} $f(-1) = 3 > 0$ local y
locales \omterm{def:g10:functions:extrema}{mínimo} $f(1) = -1 < 0$, $\lim_{-\infty} f = -\infty$,
$\lim_{+\infty} f = +\infty$.

En cada una de las tres monotonías \omterm{def:g10:numbers:interval}{intervalos}, $f$ es \omterm{def:g12:limcont:continuity}{continuo}, estrictamente
\omterm{def:g12:seq:monotonic}{monótono} y $0$ se encuentran estrictamente entre los valores/límites del punto final, por lo que el
El teorema de biyección da exactamente un cero por \omterm{def:g10:numbers:interval}{intervalo}: tres ceros en total.
Firmar cheques: $f(-2) = -1 < 0 < f(-1) = 3$, por lo que un cero adentro
$\intoo{-2}{-1}$; $f(0) = 1 > 0 > f(1) = -1$, un cero en $\intoo{0}{1}$;
$f(1) = -1 < 0 < f(2) = 3$, un cero en $\intoo{1}{2}$.
\end{solution}

\begin{solution}{exo:g12:limcont:7}
Deje $g(x) = f(x) - x$, \omterm{def:g12:limcont:continuity}{continuo} en $\intcc{0}{1}$. desde
$f(0) \in \intcc{0}{1}$, $g(0) = f(0) \geq 0$; desde $f(1) \in \intcc{0}{1}$,
$g(1) = f(1) - 1 \leq 0$. Según el teorema del valor intermedio, $g$ desaparece en
algunos $c \in \intcc{0}{1}$, es decir, \ $f(c) = c$.
\end{solution}

\begin{solution}{exo:g12:limcont:8}
Sean $u(t)$ y $d(t)$ las posiciones (medidas como distancia desde el fondo
del sendero) en el momento $t \in \intcc{8}{20}$ en el camino de subida y en el camino
abajo. Ambos son \omterm{def:g12:limcont:continuity}{continuo}, $u(8) = 0$, $u(20) = L$ (la longitud del sendero),
$d(8) = L$, $d(20) = 0$. El \omterm{def:g11:func:function}{función} $g = u - d$ es \omterm{def:g12:limcont:continuity}{continuo} con
$g(8) = -L \leq 0$ y $g(20) = L \geq 0$, por lo que $g(c) = 0$ por algún tiempo $c$ por
el teorema del valor intermedio: en el momento $c$, las posiciones de ambos días coinciden.
\end{solution}

\begin{solution}{exo:g12:limcont:9}
\emph{1.} $f'(x) = 5x^4 + 3x^2 + 1 > 0$, por lo que $f$ es estrictamente \omterm{def:g12:seq:monotonic}{creciente} en
$\R$; con $\lim_{\pm\infty} f = \pm\infty$ y \omterm{def:g12:limcont:continuity}{continuidad}, la biyección
El teorema da un cero real único $\alpha$. Desde $f(0) = -1 < 0$ y
$f(1) = 2 > 0$, $\alpha \in \intoo{0}{1}$.

\emph{2.} Mantener un \omterm{def:g10:numbers:interval}{intervalo} $\intcc{a_n}{b_n}$ con
$f(a_n) < 0 < f(b_n)$, a partir de $\intcc{0}{1}$; en cada paso evaluar
$f$ en el \omterm{prop:g10:coordgeom:midpoint}{punto medio} $m$ y conservar el \omterm{def:g10:numbers:interval}{medio intervalo} en el que cambia el signo.
persiste. Después de los pasos $n$, el \omterm{def:g10:numbers:interval}{intervalo} tiene una longitud $2^{-n}$, por lo que el \omterm{prop:g10:coordgeom:midpoint}{punto medio}
se aproxima a $\alpha$ dentro de $2^{-(n+1)}$. necesitamos
$2^{-(n+1)} \leq 10^{-6}$, es decir\ $n + 1 \geq \frac{6}{\log_{10} 2} \approx 19.93$:
$n = 20$ iteraciones son suficientes.
\end{solution}

\begin{solution}{pb:g12:limcont:1}
\textbf{1.} Términos dominantes: $\frac{3x^2}{x^2} \to 3$;
$\frac{2x}{x^2} = \frac2x \to 0$;
$x^3 - x^2 = x^3\left(1 - \frac1x\right) \to +\infty$.

\textbf{2.} $\lim_{x \to \pm\infty} f = 3$: \omterm{def:g12:limcont:asymptote}{horizontales
asíntota} $y = 3$; $\lim_{x \to (-2)^\pm} f = \mp\infty$:
\omterm{def:g12:limcont:asymptote}{asíntota vertical} $x = -2$.

\textbf{3.}$\frac{(x-3)(x+3)}{x - 3} = x + 3 \to 6$. y
$\frac{\sqrt{x+1} - 1}{x} = \frac{x}{x(\sqrt{x+1} + 1)} =
\frac{1}{\sqrt{x+1} + 1} \to \frac12$.

\textbf{4.} $\abs{x \sin\frac1x} \leq \abs x \to 0$, por lo que el
El límite es $0$. y
$\abs{\frac{\sin x}{x}} \leq \frac1x \to 0$: límite $0$.

\textbf{5.} $\frac{\sqrt{4x^2 + x}}{x} =
\sqrt{4 + \frac1x} \to \sqrt4 = 2$ (\omterm{def:g12:limcont:continuity}{continuidad} del cuadrado
raíz, composición).

\textbf{6.} $f(x) = x^3 + x$ es \omterm{def:g12:limcont:continuity}{continuo} con
$f(1) = 2 < 5 < 10 = f(2)$: por el IVT, $f(c) = 5$ para algunos
$c \in \intoo{1}{2}$.

\textbf{7.} $f'(x) = 3x^2 + 1 > 0$: estrictamente \omterm{def:g12:seq:monotonic}{creciente} en
$\R$, \omterm{def:g12:limcont:continuity}{continuo}, con límites $\mp\infty$: por la biyección
teorema, cada valor --- en particular $5$ --- se alcanza
exactamente una vez.

\textbf{8.} $p(x) = x^3\left(1 + \frac ax + \frac{b}{x^2} +
\frac{c}{x^3}\right)$: como $x \to +\infty$, $p \to +\infty$; como
$x \to -\infty$, $p \to -\infty$. Así que en un enorme \omterm{def:g10:numbers:interval}{intervalo}
$\intcc{-M}{M}$, $p(-M) < 0 < p(M)$: el IVT entrega un \omterm{def:g11:quad:discriminant}{raíz}.
El mismo argumento del término dominante funciona palabra por palabra para cualquier
grado --- los poderes impares mantienen signos opuestos en los dos
infinitos.

\textbf{9.} Reducir a la mitad $n$ veces deja un \omterm{def:g10:numbers:interval}{intervalo} de longitud
$2^{-n}$; $2^{-20} \approx 9.5 \times 10^{-7} < 10^{-6}$:
veinte pasos. Newton duplica los dígitos correctos en cada paso: desde
un dígito correcto, aproximadamente $3$--$4$ los pasos llegan a seis --- el
la tortuga y el galgo de la búsqueda de raíces.

\textbf{10.} $g$ no es \omterm{def:g12:limcont:continuity}{continuo} en $\intcc{-1}{1}$:
explota en $0$, que contiene el \omterm{def:g10:numbers:interval}{intervalo}. Los IVT
la conclusión se evapora con su hipótesis --- siempre verifique
\omterm{def:g12:limcont:continuity}{continuidad} \emph{on the whole \omterm{def:g10:numbers:interval}{intervalo}}, no sólo en el
puntos finales.

\textbf{11.} $g(x) = f(x) - x$ es \omterm{def:g12:limcont:continuity}{continuo};
$g(0) = f(0) \geq 0$ (los valores se encuentran en $\intcc{0}{1}$) y
$g(1) = f(1) - 1 \leq 0$. Por el IVT, $g(c) = 0$ para algunos $c$:
$f(c) = c$.

\textbf{12.} Idealizando al país como segmento
$\intcc{0}{1}$: $f$ envía cada lugar al lugar mostrado
directamente debajo de él en el mapa arrugado; arrugar mantiene el
mapa sobre el país, por lo que $f$ aterriza en $\intcc{0}{1}$, y
es \omterm{def:g12:limcont:continuity}{continuo} (el papel no se rompe). El \omterm{pb:g10:functions:1}{punto fijo} $c$ es
el lugar que se encuentra precisamente encima de su propio \omterm{def:g10:functions:function}{imagen} --- Brouwer
El teorema maneja el mapa bidimensional honesto.

\textbf{13.} $g(0) + g(\pi) = \left(T(0) - T(\pi)\right) +
\left(T(\pi) - T(2\pi)\right) = T(0) - T(2\pi) = 0$. entonces
$g(0)$ y $g(\pi)$ son opuestos (o ambos cero): en cualquier sentido
$g$, \omterm{def:g12:limcont:continuity}{continuo}, desaparece en algún $t^* \in \intcc{0}{\pi}$:
$T(t^*) = T(t^* + \pi)$ --- en algún lugar del ecuador, dos
Los puntos antípodas coinciden en la temperatura, en cada uno de ellos.
momento.

\textbf{14.} Rotación de la mesa un cuarto de vuelta.
los roles de las dos diagonales: el par de patas que presionaban
el suelo y la pareja que (uno de ellos) flotaba se intercambian
situaciones, por lo que la altura de vuelo --- contaba con su signo

Plantilla \textbf{15.}: construye el \emph{diferencia} de los dos
cantidades que se desean que coincidan ($f(x) - x$;

\textbf{16.} Preguntas abiertas: \emph{cuántos} soluciones
(respuesta por monotonicidad estricta y el teorema de biyección,
pregunta 7) y \emph{dónde} son (respondida por \omterm{thm:g12:limcont:ivt}{dicotomía}
o Newton, pregunta 9). El IVT es un oráculo de existencia pura.

\textbf{17.} $3 - \frac{7}{x + 2}$: \omterm{def:g12:limcont:asymptote}{asíntota vertical}
$x = -2$, horizontal $y = 3$; a cada lado de $-2$ el
\omterm{def:g11:func:function}{función} es \omterm{def:g12:seq:monotonic}{creciente} ($-\frac{7}{x+2}$ aumenta donde
definido); la curva es la referencia \omterm{def:g11:func:reference}{hipérbola}
$-\frac7x$ se desplazó hacia la izquierda $2$ y hacia arriba $3$ --- dos ramas
abrazando la cruz \omterm{def:g12:limcont:asymptote}{asíntota}.

\textbf{18.} Para $x \neq 1$: $f(x) = x + 1$, entonces
$\lim_{x \to 1} f = 2$: configuración de restauraciones $f(1) = 2$
\omterm{def:g12:limcont:continuity}{continuidad}. La discontinuidad fue un pinchazo, no un salto ---
llene el único punto que falta y la curva sanará:
\emph{desmontable}.

\textbf{19.} \omterm{def:g12:limcont:continuity}{Continuo} en cada \omterm{def:g10:numbers:interval}{intervalo} entre consecutivos
\omterm{def:g10:numbers:sets}{números enteros}; en cada \omterm{def:g10:numbers:sets}{entero} $n$ salta de $n - 1$ a $n$.
Con $a = 0.9$, $b = 1.1$: $\lfloor a \rfloor = 0 < \frac12 <
1 = \lfloor b \rfloor$, sin embargo $\lfloor x \rfloor = \frac12$ tiene
no hay solución --- el salto se teletransporta sobre $\frac12$, que es
precisamente lo que el IVT prohíbe hacer a \omterm{def:g12:limcont:continuity}{continuo} \omterm{def:g11:func:function}{funciones}.

\textbf{20.} Cara local: valor $=$ límite en cada punto ---
la punción removible (pregunta 18) lo violó de forma curable, la
saltos del suelo (pregunta 19) de forma irremediable. Cara global: en un
\omterm{def:g10:numbers:interval}{intervalo}, se visitan todos los valores intermedios --- el \omterm{def:g11:quad:discriminant}{raíz}
certificados (6--8), el \omterm{pb:g10:functions:1}{punto fijo}, el ecuador, la tabla
y el excursionista todos viven aquí. El nombre oficial del eslogan: el
\emph{teorema del valor intermedio}.
\end{solution}
