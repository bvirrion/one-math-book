\begin{solution}{exo:g12:complex:3}
$\abs{z_1} = \sqrt2$, \omterm{def:g12:complex:argument}{argumento} $\frac{3\pi}{4}$ (segundo cuadrante):
$z_1 = \sqrt2\,\eu^{3\iu\pi/4}$.
$\abs{z_2} = 2$, \omterm{def:g12:complex:argument}{argumento} $-\frac\pi6$:
$z_2 = 2\,\eu^{-\iu\pi/6}$. Por lo tanto
\[
\frac{z_1}{z_2} = \frac{\sqrt2}{2}\,\eu^{\iu\left(\frac{3\pi}{4} + \frac{\pi}{6}\right)}
= \frac{\sqrt2}{2}\,\eu^{11\iu\pi/12}.
\]
Algebraicamente,
\[
\frac{z_1}{z_2} = \frac{-1+\iu}{\sqrt3 - \iu}
= \frac{(-1+\iu)(\sqrt3+\iu)}{4}
= \frac{-\sqrt3 - 1 + \iu(\sqrt3 - 1)}{4}.
\]
Identificando \omterm{def:g12:complex:def}{piezas reales}:
$\frac{\sqrt2}{2}\cos\frac{11\pi}{12} = \frac{-\sqrt3-1}{4}$, entonces
\[
\cos\frac{11\pi}{12} = -\frac{\sqrt6 + \sqrt2}{4}.
\]
\end{solution}

\begin{solution}{exo:g12:complex:4}
\emph{(a)} $\abs{z - 2} = \abs{z - (-\iu)}$ significa que $M$ está equidistante de
$A(2, 0)$ y $B(0, -1)$: el \emph{bisectriz perpendicular} de $[AB]$.

\emph{(b)} Distancia $3$ del punto $(1,1)$: el \emph{círculo} del centro
$1 + \iu$ y radio $3$.

\emph{(c)} Por \cref{prop:g12:complex:geometry}(4), $\frac{z-1}{z+1}$ puramente
imaginario significa el \omterm{def:g11:vect:vector}{vectores} de $A(1,0)$ a $M$ y de $B(-1,0)$ a $M$
son \omterm{def:g11:scal:orthogonal}{ortogonal}: $M$ ve el segmento $[AB]$ bajo un ángulo recto. el conjunto es
el \emph{círculo de diámetro $[AB]$} (el círculo unitario) \emph{menos el
puntos $A$ y $B$}.
\end{solution}

\begin{solution}{exo:g12:complex:5}
De Moivre: $\cos3\theta + \iu\sin3\theta = (\cos\theta + \iu\sin\theta)^3$.
\omterm{def:g10:algebra:expand}{En expansión} por el teorema del binomio con $c = \cos\theta$, $s = \sin\theta$:
\[
(c + \iu s)^3 = c^3 + 3\iu c^2 s - 3 c s^2 - \iu s^3
= (c^3 - 3cs^2) + \iu(3c^2 s - s^3).
\]
Identificación de piezas y uso de $s^2 = 1 - c^2$, $c^2 = 1 - s^2$:
\[
\cos3\theta = 4\cos^3\theta - 3\cos\theta,
\qquad
\sin3\theta = 3\sin\theta - 4\sin^3\theta .
\]
\end{solution}

\begin{solution}{exo:g12:complex:6}
\[
\cos^3\theta = \left(\frac{\eu^{\iu\theta} + \eu^{-\iu\theta}}{2}\right)^{\!3}
= \frac{\eu^{3\iu\theta} + 3\eu^{\iu\theta} + 3\eu^{-\iu\theta} + \eu^{-3\iu\theta}}{8}
= \frac{\cos3\theta + 3\cos\theta}{4}.
\]
Por lo tanto
\[
\int_0^{\pi/2}\cos^3\theta\,\dd\theta
= \frac14\left[\frac{\sin3\theta}{3} + 3\sin\theta\right]_0^{\pi/2}
= \frac14\left(-\frac13 + 3\right) = \frac{2}{3}.
\]
(Consistente con $W_3 = \frac23$ en \cref{exo:g12:integ:9}, ya que
$\int_0^{\pi/2}\cos^3 = \int_0^{\pi/2}\sin^3$ por la simetría
$\theta \mapsto \frac\pi2 - \theta$.)
\end{solution}

\begin{solution}{exo:g12:complex:7}
$c = a + \eu^{\pm\iu\pi/3}(b - a)$ con $b - a = 2 + \iu$ y
$\eu^{\pm\iu\pi/3} = \frac12 \pm \iu\frac{\sqrt3}{2}$:
\[
\eu^{\iu\pi/3}(2+\iu) = \left(\tfrac12 + \iu\tfrac{\sqrt3}{2}\right)(2+\iu)
= \left(1 - \tfrac{\sqrt3}{2}\right) + \iu\left(\tfrac12 + \sqrt3\right),
\]
entonces $c_1 = \left(2 - \frac{\sqrt3}{2}\right) +
\iu\left(\frac32 + \sqrt3\right)$, y de manera similar con el ángulo
$-\frac\pi3$:
$c_2 = \left(2 + \frac{\sqrt3}{2}\right) + \iu\left(\frac32 - \sqrt3\right)$.
\end{solution}

\begin{solution}{exo:g12:complex:8}
$-4 = 4\,\eu^{\iu\pi}$, por lo que las soluciones de $z^4 = -4$ son
\[
z_k = \sqrt2\,\eu^{\iu\left(\frac\pi4 + \frac{k\pi}{2}\right)},
\quad k = 0, 1, 2, 3,
\]
\emph{i.e.} $1 + \iu$, $-1 + \iu$, $-1 - \iu$, $1 - \iu$: los vértices de una
cuadrado de circunradio $\sqrt 2$. Agrupación \omterm{def:g12:complex:conjugate}{conjugado} \omterm{def:g11:quad:discriminant}{raíces},
\[
z^4 + 4 = \bigl(z - (1{+}\iu)\bigr)\bigl(z - (1{-}\iu)\bigr)
\bigl(z + (1{-}\iu)\bigr)\bigl(z + (1{+}\iu)\bigr)
= (z^2 - 2z + 2)(z^2 + 2z + 2).
\]
\end{solution}

\begin{solution}{exo:g12:complex:9}
Con $q = \eu^{\iu\theta} \neq 1$:
\[
\sum_{k=0}^{n} \eu^{\iu k\theta} = \frac{\eu^{\iu(n+1)\theta} - 1}{\eu^{\iu\theta} - 1}
= \frac{\eu^{\iu(n+1)\theta/2}}{\eu^{\iu\theta/2}}
\cdot\frac{\eu^{\iu(n+1)\theta/2} - \eu^{-\iu(n+1)\theta/2}}{\eu^{\iu\theta/2} - \eu^{-\iu\theta/2}}
= \eu^{\iu n\theta/2}\,
\frac{\sin\frac{(n+1)\theta}{2}}{\sin\frac{\theta}{2}},
\]
usando $\eu^{\iu\alpha} - \eu^{-\iu\alpha} = 2\iu\sin\alpha$ (el
\emph{half-angle \omterm{def:g10:algebra:expand}{factorización}}). Tomando \omterm{def:g12:complex:def}{piezas reales}:
\[
S = \frac{\sin\frac{(n+1)\theta}{2}}{\sin\frac{\theta}{2}}\,
\cos\frac{n\theta}{2}.
\]
\end{solution}

\begin{solution}{exo:g12:complex:10}
\emph{1.} $\omega$ es un $5$-ésimo \omterm{def:g11:quad:discriminant}{raíz} de unidad diferente de $1$, por lo que la suma
de los cinco \omterm{def:g11:quad:discriminant}{raíces} desaparece (\cref{thm:g12:complex:rootsofunity}), que dice
$1 + \omega + \omega^2 + \omega^3 + \omega^4 = 0$.

\emph{2.} $u + v = \omega + \omega^2 + \omega^3 + \omega^4 = -1$ por el punto 1.
Para el producto, utilizando $\omega^5 = 1$:
\[
uv = (\omega + \omega^4)(\omega^2 + \omega^3)
= \omega^3 + \omega^4 + \omega^6 + \omega^7
= \omega^3 + \omega^4 + \omega + \omega^2 = -1 .
\]
Entonces $u, v$ son los \omterm{def:g11:quad:discriminant}{raíces} de $X^2 + X - 1 = 0$:
$\left\{\frac{-1+\sqrt5}{2}, \frac{-1-\sqrt5}{2}\right\}$. ahora
$u = \omega + \conj\omega = 2\cos\frac{2\pi}{5} > 0$ (desde
$\frac{2\pi}{5} < \frac{\pi}{2}$), entonces $u = \frac{-1 + \sqrt5}{2}$.

\emph{3.} $\cos\frac{2\pi}{5} = \frac u2 = \frac{\sqrt5 - 1}{4}$.
\end{solution}

\begin{solution}{pb:g12:complex:1}
\textbf{1.} $(2 + \iu)(3 - \iu) = 6 - 2\iu + 3\iu + 1 =
7 + \iu$. \quad $\frac{1 + \iu}{1 - \iu} =
\frac{(1 + \iu)^2}{2} = \frac{2\iu}{2} = \iu$. \quad
$\abs{3 + 4\iu} = 5$.

\textbf{2.} $1 + \iu = \sqrt2\,\eu^{\iu\pi/4}$;
$-2 = 2\eu^{\iu\pi}$;
$1 - \iu\sqrt3 = 2\eu^{-\iu\pi/3}$.

\textbf{3.} $(1 + \iu)^8 = \left(\sqrt2\right)^8
\eu^{8\iu\pi/4} = 16\,\eu^{2\iu\pi} = 16$.

\textbf{4.}$z^2 = \eu^{\iu\pi/2}$:
$z = \pm\eu^{\iu\pi/4} = \pm\frac{\sqrt2}{2}(1 + \iu)$. y
$\Delta = 4 - 20 = -16$: $z = 1 \pm 2\iu$.

\textbf{5.} $z \mapsto \iu z$: rotación de un cuarto de vuelta
sobre el origen. $\abs{z - (1 + \iu)} = 2$: el círculo de
centro $1 + \iu$ y radio $2$.

\textbf{6.}$\left(\omega^k\right)^5 = \eu^{2\iu\pi k} = 1$,
y los cinco puntos $\eu^{2\iu k\pi/5}$ son distintos: los cinco
soluciones de $z^5 = 1$. Se sientan en el círculo unitario a iguales
ángulos de $72^\circ$: un pentágono regular con un vértice en
$1$.

\textbf{7.} $S = 1 + \omega + \dots + \omega^4$ satisface
$(1 - \omega)S = 1 - \omega^5 = 0$ y $\omega \neq 1$:
$S = 0$. (El punto de equilibrio del pentágono es su centro).

\textbf{8.} $\omega^4 = \overline\omega$ y
$\omega^3 = \overline{\omega^2}$: emparejamiento \omterm{def:g12:complex:conjugate}{conjugados}, se lee el \omterm{def:g12:complex:def}{real
parte} de la suma
$1 + 2\cos\frac{2\pi}{5} + 2\cos\frac{4\pi}{5} = 0$.

\textbf{9.} $\cos\frac{4\pi}{5} = 2c^2 - 1$, entonces
$1 + 2c + 4c^2 - 2 = 0$, es decir\ $4c^2 + 2c - 1 = 0$:
$c = \frac{-1 + \sqrt5}{4}$ (el positivo \omterm{def:g11:quad:discriminant}{raíz}, ya que
$72^\circ$ es agudo):
$\cos 72^\circ = \frac{\sqrt5 - 1}{4}$.

\textbf{10.} Numéricamente $c \approx 0.30902$: coincide con el
calculadora. Y $\frac{1}{2\varphi} = \frac{1}{1 + \sqrt5} =
\frac{\sqrt5 - 1}{4}$ (racionalizar): $c = \frac{1}{2\varphi}$.
El pentágono es dorado de principio a fin: sus cortes diagonales
su lado en la proporción $\varphi$ --- el infinito del pentagrama
autosimilitud.

\textbf{11.} Las raíces cuadradas se pueden construir con un compás (las
máquina de semicírculo de la geometría del volumen de la escuela secundaria
\omterm{def:g11:stat:mean}{significar}), por lo que un número construido a partir de racionales sólo mediante raíces cuadradas
--- como $\cos 72^\circ$ --- es construible; las raíces cúbicas son
no, y $\cos 20^\circ$ necesita uno: el pentágono cede,
la trisección resiste. El criterio de Gauss resuelve todo regular.
polígono --- su 17-gon a los diecinueve lo hizo elegir
matemáticas.

\textbf{12.} $-4 = 4\eu^{\iu\pi}$: el cuarto \omterm{def:g11:quad:discriminant}{raíces} tiene
\omterm{def:g12:complex:modulus}{módulo} $\sqrt2$ y \omterm{def:g12:complex:argument}{argumentos}
$\frac{\pi}{4} + k\frac{\pi}{2}$: los cuatro números
$\pm 1 \pm \iu$. Emparejamiento \omterm{def:g12:complex:conjugate}{conjugados}:
$(z - 1 - \iu)(z - 1 + \iu) = z^2 - 2z + 2$ y
$(z + 1 - \iu)(z + 1 + \iu) = z^2 + 2z + 2$: el
factorización de \cref{exo:g12:complex:8}.

\textbf{13.}$(\cos\theta + \iu\sin\theta)^3 = \cos^3\theta +
3\iu\cos^2\theta\sin\theta - 3\cos\theta\sin^2\theta -
\iu\sin^3\theta$; por de Moivre esto es igual
$\cos 3\theta + \iu \sin 3\theta$. \omterm{def:g12:complex:def}{partes reales}:
$\cos 3\theta = \cos^3\theta - 3\cos\theta(1 - \cos^2\theta)
= 4\cos^3\theta - 3\cos\theta$ --- y el \omterm{def:g12:complex:def}{partes imaginarias}
entrega $\sin 3\theta$ gratis.

\textbf{14.} Es el \omterm{def:g12:complex:def}{parte real} de
$1 + \omega + \dots + \omega^4 = 0$: la suma es igual a $0$ ---
exactamente la pregunta 8, vista a través
El núcleo de \cref{exo:g12:complex:9}.

\textbf{15.} $8\iu = 8\eu^{\iu\pi/2}$: \omterm{def:g11:quad:discriminant}{raíces}
$2\eu^{\iu\pi/6} = \sqrt3 + \iu$;
$2\eu^{5\iu\pi/6} = -\sqrt3 + \iu$;
$2\eu^{3\iu\pi/2} = -2\iu$.

\textbf{16.} $1 + \iu = \sqrt2\,\eu^{\iu\pi/4}$: el mapa
gira el avión en $45^\circ$ y lo escala en $\sqrt2$. el
el cuadrado unitario se convierte en un cuadrado inclinado de lado $\sqrt2$ ---
duplicó su área, giró un octavo de vuelta.

\textbf{17.} $\abs{z_1 z_2}^2 = z_1 z_2 \overline{z_1 z_2} =
z_1\overline{z_1}\, z_2\overline{z_2} =
\abs{z_1}^2\abs{z_2}^2$: sacar raíces cuadradas. poderes de
$1 + \iu$: $1 + \iu$, $2\iu$, $-2 + 2\iu$, $-4$: cada uno
$45^\circ$ gira más y $\sqrt2$ veces más --- un
espiral hacia afuera a través de módulos
$\sqrt2, 2, 2\sqrt2, 4, \dots$

\textbf{18.} $\mathrm{j}$ es un tercer \omterm{def:g11:quad:discriminant}{raíz} de la unidad
distinto de $1$, por lo que (pregunta 7 \omterm{def:g12:complex:argument}{argumento} con $n = 3$)
$1 + \mathrm{j} + \mathrm{j}^2 = 0$. Criterio sobre la norma.
triángulo: $a + \mathrm{j}b + \mathrm{j}^2 c = 1 +
\mathrm{j}\cdot\mathrm{j} + \mathrm{j}^2 \cdot \mathrm{j}^2 =
1 + \mathrm{j}^2 + \mathrm{j}^4 = 1 + \mathrm{j}^2 +
\mathrm{j} = 0$: satisfecho, como el triángulo equilátero
$\left(1, \mathrm{j}, \mathrm{j}^2\right)$ demandas.

\textbf{19.} Durante $\C$, un polinomio real se divide en lineal
factores; su \omterm{def:g11:quad:discriminant}{raíces} no real viene en pares \omterm{def:g12:complex:conjugate}{conjugado} (\omterm{def:g12:complex:conjugate}{conjugado}
el \omterm{def:g10:algebra:equation}{ecuación}), y cada par se multiplica en una cuadrática real
$z^2 - 2\operatorname{Re}(z_0)\,z + \abs{z_0}^2$: de ahí el
factorización real en piezas de grado-$1$ y grado-$2$ ---
La pregunta 12 la realizó en $z^4 + 4$.

\textbf{20.} Puntos; rotación y escala; polígonos regulares;
identidades trigonométricas por expansión; y un mundo cerrado
donde cada \omterm{def:g10:algebra:equation}{ecuación} de grado $n$ tiene su $n$ \omterm{def:g11:quad:discriminant}{raíces}. Coda:
en la pregunta 10, el pentágono de $\pi$ estrechó la mano del dorado
ratio --- los dos invitados más famosos de Euclides, presentados por fin
por una suma de cinco flechas igual a cero.
\end{solution}
