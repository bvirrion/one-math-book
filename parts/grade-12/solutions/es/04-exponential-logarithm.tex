\begin{solution}{exo:g12:exp:1}
$\dfrac{\eu^{3x}\,\eu^{-x+1}}{\eu^{x}} = \eu^{3x - x + 1 - x} = \eu^{x+1}$.

$\ln\!\left(\dfrac{\eu^2\sqrt{\eu}}{\eu^{-3}}\right)
= 2 + \tfrac12 + 3 = \tfrac{11}{2}$.
\end{solution}

\begin{solution}{exo:g12:exp:2}
\emph{(a)} Establecer $X = \eu^x > 0$: $X^2 - 3X + 2 = 0$ da $X = 1$ o $X = 2$,
ambos positivos, entonces $x = 0$ o $x = \ln 2$.

\emph{(b)} \omterm{def:g11:func:function}{Dominio}: $x - 1 > 0$ y $x + 2 > 0$, entonces $x > 1$. El \omterm{def:g10:algebra:equation}{ecuación}
se convierte en $\ln\bigl((x-1)(x+2)\bigr) = \ln 4$, por lo tanto $(x-1)(x+2) = 4$,
\emph{i.e.} $x^2 + x - 6 = 0$, entonces $x = 2$ o $x = -3$. Sólo $x = 2$ está en el
\omterm{def:g11:func:function}{dominio}: $S = \{2\}$.
\end{solution}

\begin{solution}{exo:g12:exp:3}
Dividir por $\eu^x$:
$\dfrac{1 - x^2\eu^{-x}}{1 + \eu^{-x}} \to \dfrac{1-0}{1+0} = 1$, usando
$x^2 \eu^{-x} \to 0$ (\cref{thm:g12:exp:growth}).

$\dfrac{\ln x}{\sqrt x} \to 0$ por \cref{prop:g12:exp:lnanalytic} (caso
$n = 2$).

$x^2 \ln x = x \cdot (x \ln x) \to 0 \times 0 = 0$.

$x - \ln x = x\left(1 - \frac{\ln x}{x}\right) \to +\infty$ desde
$\frac{\ln x}{x} \to 0$.
\end{solution}

\begin{solution}{exo:g12:exp:4}
$f'(x) = \eu^{-x} - x\eu^{-x} = (1 - x)\eu^{-x}$, del signo de $1 - x$:
$f$ aumenta en $\intoc{-\infty}{1}$, disminuye en $\intco{1}{+\infty}$,
con un \omterm{def:g10:functions:extrema}{máximo} $f(1) = \eu^{-1}$ global.

Límites: como $x \to +\infty$, $f(x) = \frac{x}{\eu^x} \to 0$
(\cref{thm:g12:exp:growth}); como $x \to -\infty$, $x \to -\infty$ y
$\eu^{-x} \to +\infty$, entonces $f(x) \to -\infty$.

$f''(x) = -\eu^{-x} - (1-x)\eu^{-x} = (x - 2)\eu^{-x}$, que cambia de signo en
$x = 2$: \omterm{def:g12:deriv:inflection}{punto de inflexión} en $\bigl(2,\, 2\eu^{-2}\bigr)$.
\end{solution}

\begin{solution}{exo:g12:exp:5}
Deje $g(x) = x - \ln(1+x)$ en $\intoo{-1}{+\infty}$. entonces
$g'(x) = 1 - \frac{1}{1+x} = \frac{x}{1+x}$, negativo en $\intoo{-1}{0}$ y
positivo en $\intoo{0}{+\infty}$: $g$ alcanza su \omterm{def:g10:functions:extrema}{mínimo} $g(0) = 0$, por lo que
$g \geq 0$ con igualdad solo en $0$. Por lo tanto $\ln(1+x) \leq x$.

Aplicar esto con $x = \frac1n$:
$n\ln\left(1 + \frac1n\right) \leq 1$, entonces
$\left(1+\frac1n\right)^n = \eu^{\,n\ln(1+1/n)} \leq \eu$.
Aplicarlo con $x = -\frac{1}{n+1} > -1$:
$\ln\left(1 - \frac{1}{n+1}\right) \leq -\frac{1}{n+1}$, entonces
$-(n+1)\ln\left(1 - \frac{1}{n+1}\right) \geq 1$ y
$\left(1 - \frac{1}{n+1}\right)^{-(n+1)} \geq \eu$.
\end{solution}

\begin{solution}{exo:g12:exp:6}
\emph{1.} Cada año se multiplica el capital por $1.03$: $C_n = C_0 (1.03)^n$.

\emph{2.} $C_n \geq 2C_0 \iff (1.03)^n \geq 2 \iff n \ln 1.03 \geq \ln 2
\iff n \geq \frac{\ln 2}{\ln 1.03} \approx 23.45$: el capital se duplica después
$24$ años.

\emph{3.} $\frac{70}{3} \approx 23.3$, cerca del exacto
$\frac{\ln 2}{\ln 1.03}$. La regla funciona porque
$\ln(1 + r) \approx r$ para $r$ pequeño, por lo que
$\frac{\ln 2}{\ln(1+r)} \approx \frac{0.693}{r} \approx \frac{70}{100r}$.
\end{solution}

\begin{solution}{exo:g12:exp:7}
$\eu^{2x} > \eu^{x+1} \iff 2x > x + 1 \iff x > 1$ (la exponencial es
estrictamente \omterm{def:g12:seq:monotonic}{creciente}): $S = \intoo{1}{+\infty}$.

\omterm{def:g11:func:function}{Dominio} de la segunda desigualdad: $x^2 - 1 > 0$ y $x + 5 > 0$, \emph{i.e.}
$x \in \intoo{-5}{-1} \cup \intoo{1}{+\infty}$. En este \omterm{def:g11:func:function}{dominio}, siendo $\ln$
estrictamente \omterm{def:g12:seq:monotonic}{creciente},
\[
\ln(x^2-1) \leq \ln(x+5) \iff x^2 - 1 \leq x + 5 \iff x^2 - x - 6 \leq 0
\iff x \in \intcc{-2}{3}.
\]
Intersección con el \omterm{def:g11:func:function}{dominio}: $S = \intco{-2}{-1} \cup \intoc{1}{3}$.
\end{solution}

\begin{solution}{exo:g12:exp:8}
\emph{1.} $f'(x) = \dfrac{\eu^x x - \eu^x}{x^2} = \dfrac{(x-1)\eu^x}{x^2}$,
negativo en $\intoo{0}{1}$, positivo en $\intoo{1}{+\infty}$: \omterm{def:g10:functions:extrema}{mínimo}
$f(1) = \eu$. Límites: $f \to +\infty$ en $0^+$ (numerador $\to 1$,
denominador $\to 0^+$) y en $+\infty$ (\cref{thm:g12:exp:growth}).

\emph{2.} Para $x > 0$, $\eu^x = kx \iff f(x) = k$. Del \omterm{def:g10:functions:table}{tabla de variaciones}
($+\infty \searrow \eu \nearrow +\infty$, \omterm{def:g12:limcont:continuity}{continuo} y estrictamente \omterm{def:g12:seq:monotonic}{monótono}
en cada lado): no hay solución para $k < \eu$; exactamente uno ($x = 1$) para
$k = \eu$; exactamente dos para $k > \eu$ (uno en $\intoo{0}{1}$, uno en
$\intoo{1}{+\infty}$, por el teorema de biyección en cada \omterm{def:g10:numbers:interval}{intervalo}).
\end{solution}

\begin{solution}{exo:g12:exp:9}
\emph{1.} Directo de la primera desigualdad de \cref{exo:g12:exp:5}:
$u_n = \eu^{\,n\ln(1+1/n)} \leq \eu^1 = \eu$.

\emph{2.} Escribir
\[
\ln u_n = n \ln\!\left(1 + \frac1n\right)
= \frac{\ln\!\left(1 + \frac1n\right) - \ln 1}{\frac1n},
\]
un cociente de diferencias de $\ln$ en el punto $1$ con incremento
$h = \frac1n \to 0$. Dado que $\ln$ es \omterm{def:g12:deriv:derivative}{diferenciable} en $1$ con \omterm{def:g12:deriv:derivative}{derivado}
$1$, $\ln u_n \to 1$. Por \omterm{def:g12:limcont:continuity}{continuidad} de $\exp$
(\cref{prop:g12:limcont:seqcont}), $u_n = \eu^{\ln u_n} \to \eu^1 = \eu$.
\end{solution}

\begin{solution}{pb:g12:exp:1}
\textbf{1.}$x = \frac{\ln 5}{2}$. \quad
$3x - 1 = \eu^2$: $x = \frac{\eu^2 + 1}{3}$. \quad Con
$u = \eu^x$: $u^2 - 3u + 2 = (u - 1)(u - 2) = 0$: $x = 0$ o
$x = \ln 2$.

\textbf{2.} $\frac{3\ln 2}{\ln 2} = 3$; \quad
$3 + \frac12 = \frac72$; \quad $\frac52$.

\textbf{3.} $f'(x) = \eu^x - 1$: negativo antes de $0$, positivo
después: \omterm{def:g10:functions:extrema}{mínimo} $f(0) = 0$, entonces $f \geq 0$ en todas partes:
$\eu^x \geq 1 + x$. Sustituyendo $x = \ln(1 + u)$:
$1 + u \geq 1 + \ln(1 + u)$, es decir$\ln(1 + u) \leq u$.

\textbf{4.} $x^2\eu^{-x} = \frac{x^2}{\eu^x} \to 0$ y
$\frac{\ln x}{x} \to 0$: exponenciales aplastan potencias, potencias
aplastar \omterm{def:g12:exp:ln}{logaritmos}. Y $\frac{\eu^x - 1}{x} =
\frac{\eu^x - \eu^0}{x - 0} \to \exp'(0) = 1$.

\textbf{5.} $f'(x) = \ln x + 1$: cero en $\eu^{-1}$, negativo
antes, positivo después: \omterm{def:g10:functions:extrema}{mínimo}
$f\!\left(\frac1\eu\right) = -\frac1\eu$; y
$x \ln x \to 0$ en $0^+$: la curva sale del origen, desciende hasta
$-\frac1\eu$, y se aleja.

\textbf{6.} $n = \frac{\ln 2}{\ln 1.03} \approx 23.4$ años.

\textbf{7.} Exacto: $23.4$, $11.9$, $8.0$ años; regla del 72:
$24$, $12$, $8$. Desde $\ln(1+r) \approx r$,
$\frac{\ln 2}{\ln(1+r)} \approx \frac{0.693}{r}$, es decir\
$\frac{69.3}{\text{rate in }\%}$; los banqueros redondean hasta $72$
porque divide maravillosamente por $2, 3, 4, 6, 8, 9, 12$ ---
la aritmética mental supera a un decimal de precisión.

\textbf{8.}$2^{t/20} = \eu^{t \ln 2 / 20}$:
$\lambda = \frac{\ln 2}{20}$ por minuto. después
$24 \times 60 = 1440$ minutos: $2^{72} \approx 4.7 \times

y el espacio al final: el crecimiento real se curva hacia la curva S logística

\textbf{11.} Si los casos crecen exponencialmente, $c(t) = c_0
\eu^{\lambda t}$, entonces $\ln c(t) = \ln c_0 + \lambda t$: a

La posición del joystick para $b$ agrega las longitudes $\ln a + \ln b$, y

\textbf{17.} El \omterm{def:g10:functions:extrema}{mínimo} de $\frac{\eu^x}{x}$ en

\textbf{18.} $n \ln\!\left(1 - \frac1n\right) =

para que el éxito se concentre en esa misma constante.
$\frac1\eu \approx 0.368$.

\omterm{def:g12:deriv:tangent}{tangente} en $0$ tiene \omterm{def:g10:reffunc:affine}{pendiente} $1$ (por eso al cálculo le encanta);
