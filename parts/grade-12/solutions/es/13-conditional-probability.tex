\begin{solution}{exo:g12:condprob:1}
Hay figuras $12$, de las cuales $4$ son reyes:
$\pcond{\text{face}}{\text{king}} = \frac{4}{12} = \frac13$.

\omterm{def:g12:condprob:indep}{Independencia}: $\P(\text{king} \cap \text{heart}) = \frac{1}{52}$ (el rey
de corazones), y
$\P(\text{king})\,\P(\text{heart}) = \frac{4}{52}\times\frac{13}{52}
= \frac{1}{52}$. Iguales: los \omterm{def:g11:prob:model}{eventos} son \omterm{def:g12:condprob:indep}{independiente}.
\end{solution}

\begin{solution}{exo:g12:condprob:2}
\emph{1.} Primera rama: rojo $\frac58$, azul $\frac38$; segundas ramas
(sin reemplazo): después del rojo, rojo $\frac47$ / azul $\frac37$; después
azul, rojo $\frac57$ / azul $\frac27$.

\emph{2.} $\P(RR) = \frac58 \times \frac47 = \frac{5}{14}$. Por total
\omterm{def:g10:proba:distribution}{probabilidad},
\[
\P(\text{2nd red}) = \frac58\cdot\frac47 + \frac38\cdot\frac57
= \frac{20 + 15}{56} = \frac58 .
\]
(Igual que el primer sorteo --- por simetría, la segunda bola es uniformemente
bola aleatoria de la urna.)
\end{solution}

\begin{solution}{exo:g12:condprob:3}
$\P(A) = \frac{6}{36} = \frac16$ (seis pares suman $7$),
$\P(B) = \frac16$ y $A \cap B = \{(3,4)\}$ tienen \omterm{def:g10:proba:distribution}{probabilidad}
$\frac{1}{36} = \P(A)\P(B)$: $A$ y $B$ son \omterm{def:g12:condprob:indep}{independiente}. (La suma $7$ es
especial: cualquiera que sea el primer dado, lo da exactamente un valor del segundo.)

$\P(C) = \frac{5}{36}$ y $B \cap C = \{(3,3)\}$:
$\P(B \cap C) = \frac{1}{36} \neq \frac16 \times \frac{5}{36}
= \frac{5}{216}$. No \omterm{def:g12:condprob:indep}{independiente}.
\end{solution}

\begin{solution}{exo:g12:condprob:4}
\emph{1.} Total \omterm{def:g10:proba:distribution}{probabilidad} con la partición por máquina:
\[
\P(D) = 0.5\times0.01 + 0.3\times0.02 + 0.2\times0.04
= 0.005 + 0.006 + 0.008 = 0.019 = 1.9\% .
\]

\emph{2.} Bayes:
$\pcond{D}{M_3} = \dfrac{0.008}{0.019} = \dfrac{8}{19} \approx 0.42$. el
La máquina que produce sólo una quinta parte de la producción representa más de $40\%$ de
los defectos.
\end{solution}

\begin{solution}{exo:g12:condprob:5}
La condición es
\[
\frac{0.99\,p}{0.99\,p + 0.05\,(1-p)} \geq 0.9 .
\]
El denominador es positivo, por lo que se lee
$0.99p \geq 0.891p + 0.045 - 0.045p$, \emph{i.e.}$0.144\,p \geq 0.045$,
\emph{i.e.}
\[
p \geq \frac{0.045}{0.144} = 0.3125 .
\]
La prueba por sí sola alcanza la certeza $90\%$ sólo si la enfermedad ya afecta
más de $31\%$ de la población analizada, razón por la cual se realizan pruebas de detección masivas
de enfermedades raras requiere pruebas de confirmación.
\end{solution}

\begin{solution}{exo:g12:condprob:6}
\emph{1.} Tres \omterm{def:g12:seq:sequence}{secuencias} realizan $E$ (HHT, HTH, THH), cada uno de \omterm{def:g10:proba:distribution}{probabilidad}
$p^2(1-p)$ por \omterm{def:g12:condprob:indep}{independencia}: $\P(E) = 3p^2(1 - p)$.

\emph{2.} $f(p) = 3p^2 - 3p^3$ tiene $f'(p) = 6p - 9p^2 = 3p(2 - 3p)$,
positivo luego negativo en $\intoo{0}{1}$: \omterm{def:g10:functions:extrema}{máximo} en $p = \frac23$, donde
$\P(E) = 3\cdot\frac49\cdot\frac13 = \frac49$.
\end{solution}

\begin{solution}{exo:g12:condprob:7}
Supongamos que eligió la puerta 1 y el anfitrión abrió la puerta 3 (\omterm{def:g11:prob:model}{evento} $H_3$). con
$B_i$ = ``premio detrás de la puerta $i$'', $\P(B_i) = \frac13$ y
\[
\pcond{B_1}{H_3} = \frac12, \qquad
\pcond{B_2}{H_3} = 1, \qquad
\pcond{B_3}{H_3} = 0
\]
(si el premio está detrás de tu puerta, el anfitrión elige entre las puertas 2 y 3
al azar; si está detrás de la puerta 2, se ve obligado a abrir la 3). Bayes:
\[
\pcond{H_3}{B_1}
= \frac{\frac13\cdot\frac12}{\frac13\cdot\frac12 + \frac13\cdot1 + 0}
= \frac{1/6}{1/2} = \frac13,
\qquad
\pcond{H_3}{B_2} = \frac{\frac13}{\frac12} = \frac23 .
\]
Quedarse gana con \omterm{def:g10:proba:distribution}{probabilidad} $\frac13$, cambiar con \omterm{def:g10:proba:distribution}{probabilidad}
$\frac23$.
\end{solution}

\begin{solution}{exo:g12:condprob:8}
\emph{1.} La mayoría se equivoca cuando al menos dos de las tres copias son
volteado:
\[
3\varepsilon^2(1-\varepsilon) + \varepsilon^3
= 3(0.01)(0.9) + 0.001 = 0.028,
\]
mucho más pequeño que $\varepsilon = 0.1$: la codificación de repetición funciona.

\emph{2.} Si se envió $1$ (como $111$), para recibir $101$ se requiere exactamente un
voltear: \omterm{def:g10:proba:distribution}{probabilidad} $\varepsilon(1-\varepsilon)^2 = 0.081$. Si se envió $0$
($000$), se necesitan dos vueltas: $\varepsilon^2(1-\varepsilon) = 0.009$.
Bayes con anteriores iguales:
\[
\pcond{101}{\,\text{sent }1} = \frac{0.081}{0.081 + 0.009} = 0.9 .
\]
La decodificación mayoritaria ($101 \mapsto 1$) es, de hecho, la suposición más probable.
\end{solution}

\begin{solution}{pb:g12:condprob:1}
\textbf{1.} $\P(\text{red}) = \frac12 \cdot \frac23 + \frac12
\cdot \frac14 = \frac13 + \frac18 = \frac{11}{24}$.

\textbf{2.} $\P(\text{I} \mid \text{red}) =
\dfrac{\frac12 \cdot \frac23}{\frac{11}{24}} =
\dfrac{1/3}{11/24} = \dfrac{8}{11}$.

\textbf{3.} $\P(\text{even} \cap \leq 4) = \P(\{2, 4\}) =
\frac13$ y $\P(\text{even})\P(\leq 4) = \frac12 \cdot
\frac23 = \frac13$: \omterm{def:g12:condprob:indep}{independiente} (¡inesperadamente!). Para
$\leq 3$: $\P(\{2\}) = \frac16$ en contra
$\frac12 \cdot \frac12 = \frac14$: dependiente.

\textbf{4.} El \omterm{def:g10:proba:operations}{complementos} de
\omterm{def:g12:condprob:indep}{independiente}
\omterm{def:g11:prob:model}{eventos} son ellos mismos
\omterm{def:g12:condprob:indep}{independiente}, entonces
\[
\P(\text{none}) = 0.1 \times 0.1 = 0.01,
\qquad
\P(\text{at least one}) = 0.99 .
\]

\textbf{5.} $\frac{4}{52} \times \frac{3}{51} =
\frac{1}{221}$.

\textbf{6.} Enfermo: $100$; saludable: $99\,900$. Positivos:

\textbf{7.} $\P(\text{sick} \mid +) = \frac{99}{5\,094}

\textbf{9.} Confundido: $\P(+ \mid \text{sick}) = 99\,\%$

0.981 \times 0.05} \approx 28\,\%$. Cada \omterm{def:g12:condprob:indep}{independiente} positivo

y un tercer positivo superaría $90\,\%$ --- que es
por qué existen las pruebas confirmatorias.

\textbf{14.} Cuadratura ingenua: $\left(\frac{1}{8500}\right)^2
\approx \frac{1}{72 \text{ million}}$. con el segundo
la tragedia es cinco veces más probable dada la primera:
$\frac{1}{8500} \times \frac{5}{8500} \approx

Todavía ignorando que la hipótesis \emph{alternativa} debe ser

\textbf{15.} Bayes multiplica los ratios de verosimilitud de
\emph{todo} la evidencia en un posterior: el partido, el
coartada, las fibras, cada una cambiando las probabilidades. el jurado
frase: \emph{preguntar qué tan probable es la evidencia bajo ambos
hipótesis, y nunca confundas la condicional con su
revertir}.

La evidencia es más débil de lo que pretende el producto. el filtro

Clínica \textbf{20.}: una rara enfermedad se convierte en una fuerte prueba
en un veredicto débil --- actualización desde la tasa base, luego
