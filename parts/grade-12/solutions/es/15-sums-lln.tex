\section*{Capítulo \ref{ch:g12:sums} --- Sumas de variables aleatorias y la ley de los números grandes}

\begin{solution}{exo:g12:sums:1}
Cada dado tiene \omterm{def:g12:randvar:exp}{expectativa}
$\frac{1 + 2 + \dots + 6}{6} = \frac{7}{2}$, entonces por linealidad
$\E(S) = \frac72 + \frac72 = 7$. Los dados son \omterm{def:g12:sums:indep}{independiente}, entonces
$\V(S) = \frac{35}{12} + \frac{35}{12} = \frac{35}{6} \approx 5.83$.
\end{solution}

\begin{solution}{exo:g12:sums:2}
$\E(X) = 50$ y $\V(X) = 100 \times 0.25 = 25$.

Markov ($X \geq 0$): $\P(X \geq 75) \leq \frac{50}{75} = \frac23$.

Bienaymé--Chebyshev:
$\P(\abs{X - 50} \geq 25) \leq \frac{25}{25^2} = 0.04$ --- y esto incluso
limita el \emph{bilateral} \omterm{def:g11:prob:model}{evento}, del cual $X \geq 75$ es la mitad. Chebyshev
es mucho más nítido aquí porque explota el \omterm{def:g12:randvar:exp}{diferencia}, no sólo el \omterm{def:g11:stat:mean}{significar}.
(El valor verdadero de $\P(X \geq 75)$ es menor que $10^{-6}$: ambos límites
son toscos.)
\end{solution}

\begin{solution}{exo:g12:sums:3}
$F_n$ es el \omterm{prop:g12:sums:mean}{media muestral} de las variables $n$ Bernoulli$\left(\frac12\right)$,
de \omterm{def:g12:randvar:exp}{diferencia} $\sigma^2 = \frac14$. el atado
\[
\P\left(\abs{F_n - 0.5} \geq 0.05\right)
\leq \frac{1/4}{n \times 0.05^2} = \frac{100}{n}
\]
es $\leq 0.05$ tan pronto como $n \geq 2000$.
\end{solution}

\begin{solution}{exo:g12:sums:4}
Por \cref{prop:g12:sums:mean}, $\E(M_n) = 5\%$ (la diversificación no
cambiar el rendimiento esperado) y
$\sigma(M_n) = \frac{20\%}{\sqrt n}$. Requerir
$\frac{20}{\sqrt n} < 4$ da $\sqrt n > 5$, \emph{i.e.} $n \geq 26$.
Este es el principio de \emph{diversificación}: distribuir el capital entre
\omterm{def:g12:sums:indep}{independiente} riesgos \omterm{def:g12:arith:divides}{divide} el riesgo (\omterm{def:g11:stat:variance}{desviación estándar}) por $\sqrt n$
sin reducir el rendimiento esperado.
\end{solution}

\begin{solution}{exo:g12:sums:5}
\emph{1.} Contando los pares igualmente probables $9$: $S$ toma los valores
$2, 3, 4, 5, 6$ con probabilidades
$\frac19, \frac29, \frac39, \frac29, \frac19$. Por lo tanto
$\E(S) = \frac{2 + 6 + 12 + 10 + 6}{9} = 4$ y
$\E(S^2) = \frac{4 + 18 + 48 + 50 + 36}{9} = \frac{156}{9} = \frac{52}{3}$,
entonces $\V(S) = \frac{52}{3} - 16 = \frac43$.

\emph{2.} Una variable: $\E(X) = 2$,
$\E(X^2) = \frac{1 + 4 + 9}{3} = \frac{14}{3}$,
$\V(X) = \frac{14}{3} - 4 = \frac23$. Luego $\E(S) = 2 + 2 = 4$ y, por
\omterm{def:g12:condprob:indep}{independencia}, $\V(S) = \frac23 + \frac23 = \frac43$. Mismos valores.
\end{solution}

\begin{solution}{exo:g12:sums:6}
Con $Y = X$: $\V(X + Y) = \V(2X) = 4\V(X)$, mientras
$\V(X) + \V(Y) = 2\V(X)$. Los dos están de acuerdo sólo cuando $\V(X) = 0$, \emph{i.e.}
cuando $X$ es constante --- entonces la aditividad realmente requiere \omterm{def:g12:condprob:indep}{independencia}
(aquí $X$ depende al máximo de sí mismo).
\end{solution}

\begin{solution}{exo:g12:sums:7}
Para ser justos, $F_n$ es el \omterm{def:g11:stat:mean}{significar} de $n = 1200$ variables de Bernoulli con
$p = \frac16$, $\sigma^2 = \frac16\cdot\frac56 = \frac{5}{36}$:
\[
\P\left(\abs{F_n - \tfrac16} \geq 0.05\right)
\leq \frac{5/36}{1200 \times 0.0025} = \frac{5}{108} \approx 0.046 .
\]
La desviación observada es exactamente $0.05$: una \omterm{def:g11:prob:model}{evento} que produce un dado justo
con \omterm{def:g10:proba:distribution}{probabilidad} como máximo $4.6\%$ --- y como Chebyshev es muy
conservador, el verdadero \omterm{def:g10:proba:distribution}{probabilidad} es mucho más pequeño. La hipótesis de la justicia
no es plausible; Es muy probable que el dado esté cargado.
\end{solution}

\begin{solution}{exo:g12:sums:8}
\emph{1.} $p(1-p) = \frac14 - \left(p - \frac12\right)^2 \leq \frac14$.

\emph{2.} $F_n$ tiene \omterm{def:g12:randvar:exp}{expectativa} $p$ y \omterm{def:g12:randvar:exp}{diferencia}
$\frac{p(1-p)}{n} \leq \frac{1}{4n}$; Bienaymé--Chebyshev da
\[
\P\bigl(\abs{F_n - p} \geq \varepsilon\bigr)
\leq \frac{p(1-p)}{n\varepsilon^2} \leq \frac{1}{4n\varepsilon^2},
\]
válido cualquiera que sea el desconocido $p$.

\emph{3.} Con $\varepsilon = 0.03$ y nivel $0.05$:
\[
\frac{1}{4n(0.03)^2} \leq 0.05
\iff n \geq \frac{1}{4 \times 0.0009 \times 0.05} \approx 5556 .
\]
Aproximadamente $5600$ personas son suficientes para este tosco límite (el clásico
\omterm{def:g10:numbers:approx}{aproximación normal} la respuesta está más cerca de $1100$, consulte
\cref{ch:g12:contdist}).
\end{solution}

\begin{solution}{pb:g12:sums:1}
\textbf{1.} $V(X + Y) = V(X) + V(Y) = \frac{35}{12} +
\frac{35}{12} = \frac{35}{6}$ (\omterm{def:g12:condprob:indep}{independencia}).

\textbf{2.}$\E(\bar X) = 3.5$;
$\sigma(\bar X) = \frac{1.71}{\sqrt{100}} \approx 0.17$: el
promedio de cien dados abrazos $3.5$ dentro de una quinta parte de un punto.

\textbf{3.} $\P(X \geq 10) \leq \frac{2}{10} = 0.2$.

\textbf{4.} $\P \leq \frac{V(\bar X)}{0.5^2} =
\frac{35/1200}{0.25} \approx 0.117$: como máximo aproximadamente
$12\,\%$.

\textbf{5.} Con $Y = -X$: $V(X + Y) = V(0) = 0$, mientras
$V(X) + V(Y) = 2V(X) > 0$: variables perfectamente anticorrelacionadas
cancelar --- La adición de \omterm{def:g12:randvar:exp}{diferencia} es un privilegio \omterm{def:g12:condprob:indep}{independencia}.

\textbf{6.} $\E = \frac{18}{37} - \frac{19}{37} =
-\frac{1}{37} \approx -0.027$;
$\sigma = \sqrt{1 - \left(\frac{1}{37}\right)^2} \approx
1.00$.

\textbf{7.} $\E(G) = -2.70$, $\sigma(G) = 10 \times 1.00 =
10$: la deriva es solo $0.27\sigma$. El ruido de la noche eclipsa
la ventaja --- una gran minoría de jugadores (aproximadamente $40\,\%$,
dice el timbre) salen adelante, que es precisamente lo que mantiene
que regresen.

\textbf{8.} Ganancia del casino: \omterm{def:g12:randvar:exp}{expectativa} $+27\,027$ euros,
\omterm{def:g11:stat:variance}{desviación estándar} $\approx 1\,000$: el beneficio se sitúa en $27$
\omterm{def:g11:stat:variance}{desviaciones estándar} por encima de cero. En $27\sigma$, "el casino
"Podemos perder este año" no es un riesgo, es un error de redondeo.

\textbf{9.} $\P(\text{loss}) = \P(G \geq 27\,027$ para el
gamblers$)$ $\leq \frac{10^6 \times 1}{27\,027^2} \approx
0.0014$: hasta la desigualdad más contundente del libro garantiza
la casa en $99.86\,\%$ --- la verdad es astronómicamente
más fuerte.

\textbf{10.} Cada euro apostado, cuando y como se elija,
ha esperado el retorno $-\frac{1}{37}$ de sí mismo; por linealidad
la ganancia total esperada es $-\frac{1}{37} \times
(\text{total staked})$, negativa para cada estrategia que
apuesta cualquier cosa. Un sistema ganador violaría la linealidad
de \omterm{def:g12:randvar:exp}{expectativa} --- ninguna secuenciación inteligente de malas apuestas constituye un
bueno. (Los sistemas de doblar simplemente intercambian muchas pequeñas ganancias por
raras pérdidas catastróficas.)

\textbf{11.} El teorema: el \emph{\omterm{def:g10:stats:series}{frecuencia}} del rojo sobre
$n$ hace girar \omterm{def:g12:seq:limit}{converge} (en \omterm{def:g10:proba:distribution}{probabilidad}) a $\frac{18}{37}$. eso
no dice nada sobre el giro $n + 1$: la rueda no tiene memoria y
Después de diez rojos, la posibilidad de que salga rojo sigue siendo $\frac{18}{37}$.
Creer lo contrario es el \emph{falacia del jugador}, y
La pregunta 12 muestra lo que realmente sucede con las rachas.

\textbf{12.} El excedente de tintos $100$ no se devuelve ---
Los giros futuros son copias limpias, el excedente esperado permanece $100$.
Pero $\frac{100}{10^6} = 0.0001$: una centésima de punto.
El error de la falacia en una frase: \emph{la ley de los grandes
las cifras diluyen los accidentes pasados en un océano de nuevos juicios; eso
Nunca envía la rueda a cobrar deudas.}

\textbf{15.} Margen $\propto \frac{1}{\sqrt n}$: reducirlo a la mitad
cuadruplica el \omterm{def:g10:proba:sample}{muestra} --- alrededor de $4\,400$ personas para
$\pm 1.5$ puntos. La precisión se compra a precios cuadráticos.

\textbf{16.} El modelo es $n$ \omterm{def:g12:sums:indep}{independiente} dibuja con éxito.
\omterm{def:g10:proba:distribution}{probabilidad} $p$ --- el tamaño de población $N$ nunca aparece
(muestreo de una pequeña fracción de una población grande y bien mezclada
es lo que codifica \omterm{def:g12:condprob:indep}{independencia}). La analogía de los encuestadores:
prueba la sopa, una cucharada bien revuelta es suficiente ---
si el bote sirve para diez o diez mil. El truco es el
agitación --- un marco de muestreo sesgado, como en la escuela secundaria
encuesta de volumen que engañó a un país, no al bote.

\textbf{17.} $315\sqrt n < 30n \iff \sqrt n > 10.5 \iff
n > 110$: más allá de cien políticas, la deriva supera a la
ruido, y cada nueva política amplía la brecha en
Ley $\sqrt n$ --- agrupación \emph{is} del modelo de negocio.

\textbf{18.} \omterm{def:g12:sums:indep}{Independiente} partes iguales: $\sigma_{\text{pf}} =
\frac{20\,\%}{\sqrt{25}} = 4\,\%$: mismo rendimiento esperado,
una quinta parte de la fluctuación. El almuerzo: reducción del riesgo a cero
costo en \omterm{def:g12:randvar:exp}{expectativa} --- promediando la aleatoriedad \omterm{def:g12:sums:indep}{independiente}.

\textbf{19.} Perfectamente correlacionada: la cartera es una acción
en trajes $25$: $\sigma = 20\,\%$, sin reducción alguna.
Rentas de diversificación \emph{\omterm{def:g12:condprob:indep}{independencia}}; cuando una crisis
correlaciona todo (2008: todas las apuestas inmobiliarias fueron una apuesta),
la protección $\sqrt n$ se evapora exactamente cuando se
necesario.

\textbf{20.} Las sumas varían como $n$ y fluctúan como
$\sqrt n$, significa resolver como $\frac{1}{\sqrt n}$: el
el casino acumula más de un millón de giros; el encuestador
compra $\frac{1}{\sqrt{1100}}$ de ruido por un presupuesto de teléfono;
la aseguradora supera sus propias reclamaciones en $n > 110$; el
riesgo del inversor \omterm{def:g12:arith:divides}{divide} por $\sqrt{25}$. Abusadores: el jugador
quien cree en las deudas (sólo hay dilución), y el
financiero que cree en \omterm{def:g12:condprob:indep}{independencia} (a veces hay
sólo una apuesta). La forma universal de las fluctuaciones --- la
bell --- es la estrella del capítulo \omterm{def:g12:limcont:continuity}{continuo}, y su teorema,
el teorema del límite central, es la cumbre de la universidad
volúmenes '\omterm{def:g10:proba:distribution}{probabilidad}.
\end{solution}
