\section*{Capítulo \ref{ch:g12:diffeq} --- Ecuaciones diferenciales}

\begin{solution}{exo:g12:diffeq:1}
\emph{(a)}$y = C\eu^{3x}$; $y(0) = 2$ da $y = 2\eu^{3x}$.

\emph{(b)} $y' = -\frac12 y$, entonces $y = C\eu^{-x/2}$; $y(0) = -1$ da
$y = -\eu^{-x/2}$.

\emph{(c)} Equilibrio $y_p = 5$; $y = C\eu^{-x} + 5$; $y(0) = 0$ da
$C = -5$: $y = 5\left(1 - \eu^{-x}\right)$.
\end{solution}

\begin{solution}{exo:g12:diffeq:2}
$N' = kN$, entonces $N(t) = N_0 \eu^{kt}$. Duplicar en $3$ horas significa
$\eu^{3k} = 2$, \emph{i.e.}
\[
k = \frac{\ln 2}{3} \approx 0.231\ \text{h}^{-1}.
\]
\end{solution}

\begin{solution}{exo:g12:diffeq:3}
$y_p'(x) = \eu^x + x\eu^x = y_p(x) + \eu^x$: $y_p$ es una solución particular
(este es el caso resonante de \cref{met:g12:diffeq:particular}). Por
\cref{prop:g12:diffeq:general}, las soluciones son
$y = C\eu^{x} + x\eu^{x} = (C + x)\,\eu^x$, $C \in \R$.
\end{solution}

\begin{solution}{exo:g12:diffeq:4}
Prueba $y_p = \beta\eu^x$: $\beta\eu^x = -2\beta\eu^x + \eu^x$ te da
$3\beta = 1$, entonces $y_p = \frac13\eu^x$. solución general
$y = C\eu^{-2x} + \frac13\eu^{x}$; la condición $y(0) = 1$ da
$C = \frac23$:
\[
y(x) = \frac{2}{3}\,\eu^{-2x} + \frac{1}{3}\,\eu^{x}.
\]
\end{solution}

\begin{solution}{exo:g12:diffeq:5}
$N(t) = N_0\,\eu^{-\lambda t}$ con $\lambda = \frac{\ln 2}{5730}$
(\cref{ex:g12:exp:halflife}). Resolvemos $\eu^{-\lambda t} = 0.6$:
\[
t = \frac{\ln(1/0.6)}{\lambda} = 5730\,\frac{\ln(5/3)}{\ln 2}
\approx 5730 \times \frac{0.5108}{0.6931} \approx 4220 \text{ years}.
\]
\end{solution}

\begin{solution}{exo:g12:diffeq:6}
\emph{1.} La sal fluye a $0.2 \times 5 = 1$ kg/min. La salida lleva
concentración $\frac{m}{100}$ kg/L a $5$ L/min, \emph{i.e.} $\frac{m}{20}$
kg/min. Por lo tanto $m' = 1 - \frac{m}{20}$.

\emph{2.} Equilibrio $m_p = 20$; $m(t) = 20 + C\eu^{-t/20}$, y
$m(0) = 0$ da $C = -20$:
\[
m(t) = 20\left(1 - \eu^{-t/20}\right) \xrightarrow[t\to+\infty]{} 20 \text{ kg}.
\]
A la larga, la concentración del tanque es igual a la de la salmuera entrante:
$0.2$ kg/L $\times$ $100$ L $= 20$ kg.
\end{solution}

\begin{solution}{exo:g12:diffeq:7}
\emph{1.} $v' = g - \frac{k}{m} v$ con $\frac km = \frac{16}{80} = 0.2$.
Equilibrio $v_\infty = \frac{mg}{k} = \frac{9.8}{0.2} = 49$ m/s;
$v(t) = 49 + C\eu^{-0.2t}$ y $v(0) = 0$ dan
\[
v(t) = 49\left(1 - \eu^{-0.2 t}\right).
\]

\emph{2.} Velocidad terminal $49$ m/s ($\approx 176$ km/h). queremos
$1 - \eu^{-0.2t} = 0.95$, \emph{i.e.} $\eu^{-0.2t} = 0.05$:
\[
t = \frac{\ln 20}{0.2} \approx \frac{3.00}{0.2} \approx 15 \text{ s}.
\]
\end{solution}

\begin{solution}{exo:g12:diffeq:8}
\emph{1.} $z = \frac1y$ da $z' = -\frac{y'}{y^2}
= -\frac{y(1-y)}{y^2} = -\frac{1-y}{y} = -\frac1y + 1 = -z + 1$.

\emph{2.} Equilibrio $z_p = 1$, entonces $z(t) = 1 + C\eu^{-t}$. De
$y(0) = \frac{1}{10}$, $z(0) = 10$, entonces $C = 9$ y
\[
y(t) = \frac{1}{1 + 9\,\eu^{-t}} .
\]

\emph{3.} Como $t \to +\infty$, $9\eu^{-t} \to 0$ y $y(t) \to 1$. la curva
es la clásica curva logística en forma de S (\emph{sigmoideo}): crecimiento lento en
primero ($y$ pequeño, $y' \approx y$), crecimiento más rápido cuando $y = \frac12$
(donde $y' = y(1-y)$ es máximo), luego la saturación hacia el transporte
capacidad $1$.
\end{solution}

\begin{solution}{exo:g12:diffeq:9}
Por \cref{thm:g12:diffeq:affine}, $y(x) = C\eu^{ax} - \frac ba$. Por lo tanto
\[
u_{n+1} = C\eu^{a(n+1)} - \frac ba
= \eu^{a}\left(C\eu^{an} - \frac ba\right) + \frac ba\left(\eu^a - 1\right)
= q\,u_n + r
\]
con $q = \eu^{a}$ y $r = \frac{b}{a}\left(\eu^{a} - 1\right)$.
La condición $\abs q < 1$ significa $\eu^a < 1$, \emph{i.e.} $a < 0$: exactamente
la condición bajo la cual las soluciones del \omterm{def:g12:diffeq:ode}{ecuación diferencial}
convergen al equilibrio $-\frac ba$ --- y de hecho al \omterm{pb:g10:functions:1}{punto fijo} de
la recurrencia es $\frac{r}{1 - q} = -\frac{b}{a}$.
\end{solution}

\begin{solution}{pb:g12:diffeq:1}
\textbf{1.}$y = 2\eu^{3t}$. Para el segundo: equilibrio $3$,
entonces $y = 3 + C\eu^{-2t}$ con $y(0) = 0$: $C = -3$:
$y = 3\left(1 - \eu^{-2t}\right)$.

\textbf{2.} $\left(y\eu^{-at}\right)' = y'\eu^{-at} -
ay\eu^{-at} = (y' - ay)\eu^{-at} = 0$: el producto es un
constante $C$, por lo que $y = C\eu^{at}$ --- no se escapa ninguna solución.

\textbf{3.} Equilibrio: $y \equiv 3$. Cualquier otra solución es
$3 + C\eu^{-2t}$ con $C \neq 0$: los dados exponenciales y el
La solución se desliza hasta $3$ --- un \omterm{pb:g10:functions:1}{punto fijo} estable, el
\omterm{def:g12:limcont:continuity}{continuo} gemelo de la recurrencia \omterm{pb:g10:functions:1}{puntos fijos} de
\cref{pb:g11:seq:1}.

\textbf{4.} $y = y_0\eu^{-t/\tau}$ mitades cuando
$\eu^{-t/\tau} = \frac12$: $t = \tau\ln 2$.

\textbf{5.} Todas las soluciones son traducciones verticales de la decadencia.
hacia $3$: los que empiezan arriba caen a $3$, los de abajo
asciende a $3$, y la solución constante $3$ se queda quieta ---
embudo alrededor del equilibrio.

\textbf{6.} $\lambda = \frac{\ln 2}{5730} \approx 1.21 \times
10^{-4}$ por año.

\textbf{7.}$\eu^{-\lambda t} = 0.2$:
$t = \frac{\ln 5}{\lambda} \approx 13\,300$ años.

\textbf{8.}$t = \frac{\ln(1/0.15)}{\lambda} \approx 15\,700$
Años: los toros de Lascaux son de finales de la Edad del Hielo.

\textbf{9.}$t = \frac{\ln(1/0.53)}{\lambda} \approx 5\,250$
años: dentro de una vida del valor de los arqueólogos ---
el reloj funciona.

\textbf{10.} Diez vidas medias salen de $2^{-10} \approx 0.1\,\%$:
en el borde de la medición. Después de $65$ millones de años,
la fracción es $2^{-65\,000\,000/5\,730} \approx 2^{-11\,344}$:
No queda ningún átomo del stock original en ningún fósil.
los dinosaurios se datan mediante relojes más lentos (potasio-argón y

\textbf{11.} $52\,\%$ da $\approx 5\,405$ años,
$54\,\%$ regala $\approx 5\,095$: el $\pm 1\,\%$ de la química
se convierte aproximadamente en $\pm 155$ años de historia --- precisión en
el laboratorio es precisión en la etiqueta del museo.

\textbf{12.} $z = T - T_a$ satisface $z' = -kz$:
$z = (T_0 - T_a)\eu^{-kt}$ y $T = T_a + z$.

\textbf{13.} La brecha inicial es $T_0 - T_a = 70$; después de las cinco

$55 - 20 = 35 = 70\eu^{-kt}$:
$t = \frac{\ln 2}{k} \approx 10.3$ minutos.

\textbf{15.} Brechas desde el ambiente: a medianoche $10$, una hora
después $8$: $\eu^{-k} = 0.8$:
$k = \ln\frac{10}{8} \approx 0.223$ por hora. Al morir la brecha

\textbf{16.} Una habitación con calefacción o corrientes de aire ($T_a$ no constante)
dobla el reloj en cualquier sentido; cambios de ropa o masa corporal $k$
(las tablas del forense se ajustan a ello) --- una báscula $k$ incorrecta
todo el \omterm{def:g10:numbers:interval}{intervalo}; y un cuerpo movido de otro lado se reinicia
$T_a$ en plena decadencia, fingiendo una muerte anterior o posterior. el

\textbf{17.} $z = \frac1y$ obedece a $z' = 1 - z$:
$z = 1 + 9\eu^{-t}$ (de $z(0) = 10$), entonces
$y = \frac{1}{1 + 9\eu^{-t}}$. Inflexión en $y = \frac12$:
$9\eu^{-t} = 1$: $t = \ln 9 \approx 2.2$ --- la epidemia
fecha de giro, directamente de la fórmula.

\textbf{18.} Velocidad terminal: $v' = 0$ a $v = 20$ m/s.
Solución: $v(t) = 20\left(1 - \eu^{-t/2}\right)$. entonces
$0.95$: $\eu^{-t/2} = 0.05$: $t = 2\ln 20 \approx 6$ s ---
Las gotas de lluvia alcanzan su velocidad de crucero en segundos, lo cual es
por qué la lluvia no mata.

\textbf{19.} Estado estacionario: $c = 8$. La mitad:
$4 = 8\left(1 - \eu^{-t/2}\right)$: $\eu^{-t/2} = \frac12$:
$t = 2\ln 2 \approx 1.4$ horas. El goteo es el \omterm{def:g12:limcont:continuity}{continuo}
gemelo del préstamo: entrada constante, salida proporcional ---
\cref{exo:g12:diffeq:9} hace que el diccionario sea exacto.

\textbf{20.} Decaimiento ($a < 0$, $b = 0$), enfriamiento ($y = T - T_a$),
cayendo ($a < 0$, $b > 0$: aproximación al equilibrio desde
abajo), dosificación (igual, en vena): un \omterm{def:g10:algebra:equation}{ecuación}, cuatro mundos.
El bucle: enunciar la ley del cambio; escribe el \omterm{def:g10:algebra:equation}{ecuación}; resolver
(\cref{met:g12:diffeq:solve}); calibrar constantes en
medidas; predecir; luego audite los supuestos (pregunta
16). Necesidades de oscilación $y'' = -\omega^2 y$ --- cumplidas, resueltas y
establecido balanceándose en \cref{pb:g12:trigo:1}.
\end{solution}
