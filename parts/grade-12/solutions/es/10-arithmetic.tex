\section*{Capítulo \ref{ch:g12:arith} --- Arithmetic}

\begin{solution}{exo:g12:arith:1}
$17 \times 119 = 2023$, entonces $2026 = 17 \times 119 + 3$: cociente $119$,
resto $3$. Para $-2026$: $-2026 = 17\times(-120) + 14$ (de hecho
$17 \times 120 = 2040$ y $2040 - 2026 = 14$): cociente $-120$, resto
$14$ (el resto debe estar en $\intco{0}{17}$, por lo que es \emph{no} $-3$).
\end{solution}

\begin{solution}{exo:g12:arith:2}
Módulo $10$: $7^2 = 49 \equiv 9 \equiv -1$. Por lo tanto
$7^{100} = \left(7^2\right)^{50} \equiv (-1)^{50} = 1 \pmod{10}$: el último
El dígito de $7^{100}$ es $1$.
\end{solution}

\begin{solution}{exo:g12:arith:3}
Euclides: $1071 = 2\times462 + 147$; $462 = 3\times147 + 21$;
$147 = 7\times21 + 0$. Entonces $\gcd = 21$.

Sustitución hacia atrás: $21 = 462 - 3\times147 = 462 - 3(1071 - 2\times462)
= 7\times462 - 3\times1071$. Así $u = -3$, $v = 7$:
$1071\times(-3) + 462\times7 = 21$.
\end{solution}

\begin{solution}{exo:g12:arith:4}
Cada \omterm{def:g10:numbers:sets}{entero} es $\equiv 0, 1, 2$ o $3 \pmod 4$, y elevando al cuadrado:
$0^2 \equiv 0$, $1^2 \equiv 1$, $2^2 = 4 \equiv 0$, $3^2 = 9 \equiv 1$. entonces
$n^2 \equiv 0$ o $1 \pmod 4$. Entonces una suma de dos cuadrados es congruente con
$0 + 0$, $0 + 1$ o $1 + 1$, \emph{i.e.} a $0$, $1$ o $2 \pmod 4$ ---
nunca a $3$.
\end{solution}

\begin{solution}{exo:g12:arith:5}
\omterm{def:g12:arith:divides}{Divisibilidad} por $2$: entre $n$ y $n + 1$, uno está par. \omterm{def:g12:arith:divides}{Divisibilidad} por
$3$: si $n \equiv 0$, entonces $3 \mid n$; si $n \equiv 1 \pmod 3$, entonces
$2n + 1 \equiv 3 \equiv 0$; si $n \equiv 2$, entonces $n + 1 \equiv 0$. en total
casos $3$ \omterm{def:g12:arith:divides}{divide} el producto. Desde $\gcd(2,3) = 1$,
\cref{cor:g12:arith:coprimeprod} da $6 \mid n(n+1)(2n+1)$.
(Esto también vuelve a probar que $\frac{n(n+1)(2n+1)}{6}$, la suma de los cuadrados de
\cref{exo:g12:seq:1}, es un \omterm{def:g10:numbers:sets}{entero}).
\end{solution}

\begin{solution}{exo:g12:arith:6}
Buscamos el inverso de $5$ módulo $11$: testing (o Bézout),
$5 \times 9 = 45 = 44 + 1 \equiv 1 \pmod{11}$. Multiplicando el \omterm{def:g12:arith:congruence}{congruencia} por
$9$:
\[
x \equiv 9 \times 3 = 27 \equiv 5 \pmod{11}.
\]
Las soluciones son \omterm{def:g10:numbers:sets}{números enteros} $x = 5 + 11k$, $k \in \Z$. (Compruebe:
$5\times5 = 25 \equiv 3 \pmod{11}$.)
\end{solution}

\begin{solution}{exo:g12:arith:7}
$\gcd(17, 40) = 1$, por lo que existen soluciones. Euclides:
$40 = 2\times17 + 6$; $17 = 2\times6 + 5$; $6 = 5 + 1$. Sustitución hacia atrás:
$1 = 6 - 5 = 6 - (17 - 2\times6) = 3\times6 - 17
= 3(40 - 2\times17) - 17 = 3\times40 - 7\times17$. Por lo tanto
$17\times(-7) - 40\times(-3) = 1$: la solución particular
$(x_0, y_0) = (-7, -3)$.

Solución general de $17x - 40y = 1$: restando la relación particular,
$17(x + 7) = 40(y + 3)$; desde $\gcd(17, 40) = 1$, Gauss da
$40 \mid x + 7$, entonces $x = -7 + 40k$ y luego $y = -3 + 17k$, $k \in \Z$
(todos los cuales verifican).

Para $17x - 40y = 6$, multiplique la solución particular por $6$:
$(x_1, y_1) = (-42, -18)$, y el mismo razonamiento da
\[
x = -42 + 40k, \qquad y = -18 + 17k, \qquad k \in \Z .
\]
(Por ejemplo,\ $k = 2$: $x = 38$, $y = 16$; de hecho, $17\times38 - 40\times16
= 646 - 640 = 6$.)
\end{solution}

\begin{solution}{exo:g12:arith:8}
Supongamos $\sqrt2 = \frac ab$ con $a, b \in \N^*$; luego $a^2 = 2b^2$. en el
\omterm{def:g12:arith:prime}{principal} factorización de un cuadrado, todo exponente es par; entonces el exponente de
$2$ en $a^2$ es par, mientras que en $2b^2$ es impar (uno más que un par
número). Dos factorizaciones del mismo \omterm{def:g10:numbers:sets}{entero} con diferentes exponentes de
$2$ contradice la unicidad en \cref{thm:g12:arith:fta}. Por lo tanto no hay tal
existe fracción: $\sqrt2 \notin \Q$.
\end{solution}

\begin{solution}{exo:g12:arith:9}
\emph{1.} De $k\binom pk = p \binom{p-1}{k-1}$, $p$ \omterm{def:g12:arith:divides}{divide}
$k\binom pk$. Para $1 \leq k \leq p-1$, $p \nmid k$ y $p$ \omterm{def:g12:arith:prime}{principal} proporcione
$\gcd(p, k) = 1$, por lo que el lema de Gauss produce $p \mid \binom pk$.

\emph{2.} Inducción en $a$. Para $a = 0$: $0^p \equiv 0$. asumir
$a^p \equiv a \pmod p$. Por el teorema del binomio,
\[
(a+1)^p = \sum_{k=0}^{p} \binom pk a^k
\equiv a^p + 1 \pmod p,
\]
todos los términos medios desaparecen módulo $p$ por el punto 1. Por inducción
hipótesis, $(a+1)^p \equiv a + 1 \pmod p$. Esto prueba
$a^p \equiv a$ para todo $a \in \N$, y el caso $a < 0$ sigue por escrito
$a \equiv a + kp$ para un representante positivo adecuado.
\end{solution}

\begin{solution}{exo:g12:arith:10}
$n \equiv 2 \pmod 3$ y $n \equiv 3 \pmod 5$: escriba $n = 2 + 3s$; entonces
$2 + 3s \equiv 3 \pmod 5$, \emph{i.e.} $3s \equiv 1 \pmod 5$. lo inverso de
$3$ módulo $5$ es $2$ ($3\times2 = 6 \equiv 1$), por lo que $s \equiv 2 \pmod 5$,
digamos $s = 2 + 5t$ y $n = 8 + 15t$: las dos primeras condiciones \omterm{def:g11:stat:mean}{significar}
$n \equiv 8 \pmod{15}$.

Agregando $n \equiv 2 \pmod 7$: $8 + 15t \equiv 2 \pmod 7$, y
$15 \equiv 1 \pmod 7$, entonces $t \equiv -6 \equiv 1 \pmod 7$, digamos
$t = 1 + 7u$. Por lo tanto $n = 23 + 105u$:
\[
n \equiv 23 \pmod{105}.
\]
(Compruebe: $23 = 3\times7 + 2 = 5\times4 + 3 = 7\times3 + 2$.)
\end{solution}

\begin{solution}{pb:g12:arith:1}
\textbf{1.} $2026 = 289 \times 7 + 3$: $2026 \equiv 3
\pmod 7$. Potencias de $7$ mod $10$: $7, 9, 3, 1$, ciclo de
longitud $4$; $100 \equiv 0 \pmod 4$: último dígito de
$7^{100}$ es $1$.

\textbf{2.} $5^2 = 25 \equiv -1 \pmod{13}$, entonces
$5^{116} = \left(5^2\right)^{58} \equiv (-1)^{58} = 1$ y
$5^{117} \equiv 5 \pmod{13}$.

\textbf{3.} El inverso de $3$ módulo $7$ es $5$
($15 \equiv 1$): $x \equiv 5 \times 5 = 25 \equiv 4 \pmod 7$.

\textbf{4.}$97 = 2 \times 35 + 27$; $35 = 27 + 8$;
$27 = 3 \times 8 + 3$; $8 = 2 \times 3 + 2$; $3 = 2 + 1$.
Sustitución hacia atrás: $1 = 97 \times 13 + 35 \times (-36)$. entonces
$35 \times (-36) \equiv 1 \pmod{97}$: el inverso de $35$ es
$-36 \equiv 61 \pmod{97}$.

\textbf{5.} $a$ es módulo invertible $n$ exactamente cuando
$\gcd(a, n) = 1$: Bézout proporciona $au + nv = 1$, es decir\
$au \equiv 1$; por el contrario, una inversa obliga al \omterm{def:g12:arith:gcd}{mcd} a dividirse
$1$.

\textbf{6.} $0{\cdot}10 + 3{\cdot}9 + 0{\cdot}8 + 6{\cdot}7 +
4{\cdot}6 + 0{\cdot}5 + 6{\cdot}4 + 1{\cdot}3 + 5{\cdot}2 +
2{\cdot}1 = 132 = 12 \times 11 \equiv 0 \pmod{11}$: válido.

\textbf{7.} La suma cambia en $wd$ con
$1 \leq w \leq 10$ y $1 \leq \abs d \leq 9$: ya que $11$ es
\omterm{def:g12:arith:prime}{principal} y \omterm{def:g12:arith:divides}{divide} ningún factor, no puede dividir el producto
(\cref{thm:g12:arith:gauss} / \cref{prop:g12:arith:primedivides}):
la suma modificada nunca más será $\equiv 0$: cada dígito
error activa la alarma.

\textbf{8.} Intercambio de dígitos adyacentes $a, b$ (pesos
$w + 1, w$) cambia la suma por
$(w+1)b + wa - (w+1)a - wb = b - a \not\equiv 0$ para
$a \neq b$: detectado. El secreto es el \emph{primalidad} de
$11$: módulo $10$, productos como $5 \times 2$ desaparecen con
ni factor cero, por lo que un error de peso-$5$ de $\pm 2$ (o un
transposición desafortunada) podría pasar desapercibida.

\textbf{9.} Suma ponderada de los doce dígitos: $119$; el
dígito de control debe completarlo a un múltiplo de $10$: $1$ (completo
código $978\,2940199\,051$). EAN pierde transposiciones adyacentes
con $2(a - b) \equiv 0 \pmod{10}$, es decir\
$\abs{a - b} = 5$: intercambiar un $2$ y un $7$, digamos, pasa
invisible --- el precio del amigable \omterm{def:g12:complex:modulus}{módulo} $10$.

\textbf{10.} Duplicar cada segundo dígito desde la derecha y
plegado ($16 \to 7$, etc.), la suma llega a $80 \equiv 0
\pmod{10}$: la tarjeta de prueba valida.

\textbf{11.} Con un \omterm{def:g12:arith:prime}{principal} \omterm{def:g12:complex:modulus}{módulo} cada peso es invertible,
entonces \emph{todo} errores únicos y \emph{todo} adyacentes
se atrapan las transposiciones --- lujo del ISBN; esquemas mod-$10$
mantenga dígitos amigables para los humanos y acepte un punto ciego corto.

\textbf{12.} $2^{10} = 1024 = 3 \times 341 + 1 \equiv 1
\pmod{341}$, por lo tanto $2^{340} = \left(2^{10}\right)^{34} \equiv
1$. Sin embargo, $341 = 11 \times 31$ es compuesto: pasa la prueba de Fermat.
prueba en la base $2$ sin ser \omterm{def:g12:arith:prime}{principal}. Moraleja: Fermat
\omterm{def:g12:arith:congruence}{congruencia} es necesario, no suficiente --- pruebas de primalidad
necesita herramientas más afiladas (y las consigue, en la universidad)
volúmenes).

\textbf{13.} $3d \equiv 1 \pmod{20}$: $d = 7$
($21 = 20 + 1$).

\textbf{14.} $c = 4^3 = 64 \equiv 31 \pmod{33}$.

\textbf{15.} $31 \equiv -2$: $(-2)^7 = -128$, y
$-128 + 4 \times 33 = 4$: el texto cifrado se descifra a
$m = 4$. La cerradura gira.

\textbf{16.} Módulo $3$: si $3 \nmid m$,
$m^2 \equiv 1$ (Fermat), entonces
$m^{21} = m \cdot \left(m^2\right)^{10} \equiv m$; si
$3 \mid m$, ambos lados son $\equiv 0$. Módulo $11$:
$m^{10} \equiv 1$ o $11 \mid m$, y
$m^{21} = m \cdot \left(m^{10}\right)^2 \equiv m$. Ambos $3$
y $11$ dividen $m^{21} - m$, y siendo \omterm{def:g12:arith:gcd}{coprimo} su producto
$33$ también lo hace (Gauss): $m^{21} \equiv m \pmod{33}$. el
exponente $ed = 21 = 1 + 20k$ fue construido de manera que tanto Fermat
los exponentes ($2$ y $10$, dividiendo $20$) desaparecen.

\textbf{17.} Multiplicar dos dígitos $300$ \omterm{def:g12:arith:prime}{primos} toma un
microsegundo; recuperarlos de su producto derrota todos los
algoritmo conocido y todas las computadoras del mundo --- la cerradura es
una calle de un solo sentido. (Nuestro $n = 33$ es la calle a escala de juguete,
transitable en ambas direcciones).

\textbf{18.} Residuos de prueba (o edificio con Bézout):
$n \equiv 8 \pmod{15}$: los conteos $8, 23, 38, 53, \dots$
Módulo de unicidad $15$: dos soluciones se diferencian por un múltiplo de
$3$ y de $5$, por tanto de $15$ ($3$ y $5$ \omterm{def:g12:arith:gcd}{coprimo}, Gauss).
El general con soldados $1000$ anuncia ``$8$'' a las tres
alineaciones rápidas: el antiguo truco del recuento.

\textbf{19.} $10 \equiv 1 \pmod 9$ da $10^k \equiv 1$, entonces
$\sum d_k 10^k \equiv \sum d_k$: un número y la suma de sus dígitos
son congruentes módulo $9$ (y módulo $3$). y
$10 \equiv -1 \pmod{11}$ da
$\sum d_k 10^k \equiv \sum (-1)^k d_k$: la regla de alternancia.
Dos reglas infantiles, una línea cada una.

\textbf{20.} \omterm{def:g12:arith:congruence}{Congruencias} convirtió los restos en una aritmética
(Parte I); Bézout acuñó las inversas que resuelven linealmente
\omterm{def:g12:arith:congruence}{congruencias} y $d$ de RSA (preguntas 4, 13); El pequeño de Fermat
teorema abrió y cerró la cerradura (preguntas 15-16); pegando
\omterm{def:g12:arith:gcd}{coprimo} módulos contaron soldados y terminaron la prueba.
(preguntas 16, 18). Veredicto sobre Hardy: el teorema más puro que él
sabía ahora guarda cada compra --- la pureza, con el tiempo, es la
lo más aplicable que existe.
\end{solution}
