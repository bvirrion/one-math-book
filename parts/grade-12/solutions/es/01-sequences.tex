\section*{Capítulo \ref{ch:g12:seq} --- Sequences}

\begin{solution}{exo:g12:seq:1}
Sea $P(n)$ la declaración $\sum_{k=1}^{n} k^2 = \frac{n(n+1)(2n+1)}{6}$.
\emph{Caso base:} para $n = 0$ ambos lados son $0$ (suma vacía).
\emph{Paso inductivo:} supone $P(n)$. entonces
\[
\sum_{k=1}^{n+1} k^2 = \frac{n(n+1)(2n+1)}{6} + (n+1)^2
= \frac{(n+1)\bigl(n(2n+1) + 6(n+1)\bigr)}{6}
= \frac{(n+1)(2n^2 + 7n + 6)}{6}.
\]
Desde $2n^2 + 7n + 6 = (n+2)(2n+3)$, esto es
$\frac{(n+1)(n+2)(2(n+1)+1)}{6}$, que es $P(n+1)$. Por inducción, $P(n)$
es válido para todos los $n$.
\end{solution}

\begin{solution}{exo:g12:seq:2}
$a_{n+1} - a_n = \frac{n+2}{n+1} - \frac{n+1}{n}
= \frac{n(n+2) - (n+1)^2}{n(n+1)} = \frac{-1}{n(n+1)} < 0$: $(a_n)$ es
estrictamente \omterm{def:g10:functions:variations}{decreciente}.

$(b_n)$ tiene términos positivos y
$\frac{b_{n+1}}{b_n} = \frac{2^{n+1}}{n+1}\cdot\frac{n}{2^n} =
\frac{2n}{n+1} \geq 1 \iff 2n \geq n+1 \iff n \geq 1$: $(b_n)$ es \omterm{def:g12:seq:monotonic}{creciente}
(estrictamente para $n \geq 2$).

$c_{n+1} - c_n = (n+1)^2 - 10(n+1) - n^2 + 10n = 2n - 9$, que es negativo
para $n \leq 4$ y positivo para $n \geq 5$: $(c_n)$ disminuye hasta
$c_5 = -25$, es \omterm{def:g10:functions:extrema}{mínimo}, luego aumenta. No es \omterm{def:g12:seq:monotonic}{monótono}.
\end{solution}

\begin{solution}{exo:g12:seq:3}
Términos factoriales dominantes:
\[
u_n = \frac{n^2(3 - 1/n + 1/n^2)}{n^2(2 + 5/n^2)} \xrightarrow[n\to+\infty]{} \frac{3}{2}.
\]
Multiplicar por el conjugado:
\[
v_n = \frac{(n+1) - n}{\sqrt{n+1} + \sqrt{n}} = \frac{1}{\sqrt{n+1}+\sqrt{n}}
\xrightarrow[n\to+\infty]{} 0.
\]
Divida el numerador y el denominador por $3^n$:
\[
w_n = \frac{(2/3)^n - 1}{1 + (1/3)^n} \xrightarrow[n\to+\infty]{} \frac{0-1}{1+0} = -1,
\]
usando $\lim q^n = 0$ para $\abs{q} < 1$.
\end{solution}

\begin{solution}{exo:g12:seq:4}
$u_n = 5 + 3n \to +\infty$ y $v_n = 8 \cdot (1/2)^n = 2^{3-n} \to 0$.
las sumas son
\[
\sum_{k=0}^{n} u_k = (n+1)\,\frac{5 + (5+3n)}{2} = \frac{(n+1)(10+3n)}{2}
\xrightarrow[n\to+\infty]{} +\infty,
\]
\[
\sum_{k=0}^{n} v_k = 8\,\frac{1 - (1/2)^{n+1}}{1 - 1/2}
= 16\left(1 - \left(\tfrac{1}{2}\right)^{n+1}\right)
\xrightarrow[n\to+\infty]{} 16 .
\]
\end{solution}

\begin{solution}{exo:g12:seq:5}
Desde $-1 \leq \cos n \leq 1$,
\[
\frac{n-1}{n+1} \leq \frac{n + \cos n}{n+1} \leq 1,
\]
y $\frac{n-1}{n+1} \to 1$, por lo que el límite es $1$ según el teorema de compresión.

Para el segundo límite, escriba
\[
0 \leq \frac{n!}{n^n}
= \frac{1}{n}\cdot\frac{2}{n}\cdots\frac{n}{n}
\leq \frac{1}{n},
\]
porque cada factor $\frac{k}{n}$ con $2 \leq k \leq n$ es como máximo $1$.
Desde $\frac1n \to 0$, el teorema de compresión da $\frac{n!}{n^n} \to 0$. (El
El límite \omterm{def:g12:seq:arith-geom}{geométrico} sugerido también funciona: cada factor con $k \leq n/2$ es como máximo
$\frac12$, dando el límite más fuerte $(1/2)^{\floor{n/2}}$.)
\end{solution}

\begin{solution}{exo:g12:seq:6}
\emph{1.} $u_0 = 0 \in \intcc{0}{2}$. Si $0 \leq u_n \leq 2$, entonces
$2 \leq u_n + 2 \leq 4$, entonces $\sqrt{2} \leq u_{n+1} \leq 2$; en particular
$0 \leq u_{n+1} \leq 2$. Por inducción la propiedad es válida para todos los $n$.

\emph{2.} $u_{n+1} - u_n = \sqrt{u_n + 2} - u_n$. Para $x \in \intcc{0}{2}$,
$\sqrt{x+2} \geq x \iff x + 2 \geq x^2 \iff (2-x)(x+1) \geq 0$, lo cual es cierto.
Por lo tanto, $(u_n)$ es \omterm{def:g12:seq:monotonic}{creciente}.

\emph{3.} \omterm{def:g12:seq:monotonic}{Creciente} y \omterm{def:g12:seq:bounded}{delimitado arriba} por $2$, $(u_n)$ \omterm{def:g12:seq:limit}{converge} para algunos
$\ell \in \intcc{0}{2}$. Pasando al límite en $u_{n+1} = \sqrt{u_n + 2}$
(el mapa $x \mapsto \sqrt{x+2}$ es continuo) da $\ell = \sqrt{\ell + 2}$,
entonces $\ell^2 - \ell - 2 = 0$, es decir\ $\ell \in \{-1, 2\}$. Desde $\ell \geq 0$,
$\lim u_n = 2$.
\end{solution}

\begin{solution}{exo:g12:seq:7}
\emph{1.} Entre dos dosis se elimina $40\%$ del fármaco, por lo que el
la cantidad $u_n$ se convierte en $0.6\,u_n$; la siguiente dosis agrega la unidad $1$:
$u_{n+1} = 0.6\,u_n + 1$.

\emph{2.} $v_{n+1} = u_{n+1} - 2.5 = 0.6\,u_n + 1 - 2.5 = 0.6(u_n - 2.5)
= 0.6\,v_n$: $(v_n)$ es \omterm{def:g12:seq:arith-geom}{geométrico} con relación $0.6$ y primer término
$v_0 = 1 - 2.5 = -1.5$. Por lo tanto $v_n = -1.5 \times 0.6^n$ y
\[
u_n = 2.5 - 1.5 \times 0.6^n .
\]

\emph{3.} Desde $0.6^n \to 0$, $u_n \to 2.5$: la cantidad de medicamento se estabiliza
en unidades $2.5$.
\end{solution}

\begin{solution}{exo:g12:seq:8}
\emph{1.}$u_0 = 3 > 1$. Si $u_n > 1$, entonces $u_n + 2 > 0$ y
\[
u_{n+1} - 1 = \frac{4u_n - 1 - u_n - 2}{u_n + 2} = \frac{3(u_n - 1)}{u_n + 2} > 0 .
\]
Por inducción, $u_n > 1$ para todos los $n$ (y en particular $u_n + 2 \neq 0$, por lo que
el \omterm{def:g12:seq:sequence}{secuencia} está bien definido).

\emph{2.} Usando la identidad anterior,
\[
v_{n+1} = \frac{1}{u_{n+1} - 1} = \frac{u_n + 2}{3(u_n - 1)}
= \frac{(u_n - 1) + 3}{3(u_n - 1)} = \frac{1}{3} + v_n .
\]
Entonces $(v_n)$ es \omterm{def:g12:seq:arith-geom}{aritmética} con \omterm{def:g11:seq:arithmetic}{diferencia común} $\frac13$ y
$v_0 = \frac{1}{u_0 - 1} = \frac12$.

\emph{3.} $v_n = \frac12 + \frac{n}{3}$, por lo tanto
$u_n = 1 + \frac{1}{v_n} = 1 + \frac{6}{3 + 2n}$. Desde $v_n \to +\infty$,
$u_n \to 1$.
\end{solution}

\begin{solution}{exo:g12:seq:9}
\emph{1.} $H_{2n} - H_n = \sum_{k=n+1}^{2n} \frac{1}{k}$ es una suma de $n$
términos, cada uno al menos $\frac{1}{2n}$; por lo tanto
$H_{2n} - H_n \geq n \cdot \frac{1}{2n} = \frac12$.

\emph{2.} Por inducción en $k$: $H_{2^0} = H_1 = 1 \geq 1$. si
$H_{2^k} \geq 1 + \frac{k}{2}$, luego aplicando el punto 1 con $n = 2^k$,
\[
H_{2^{k+1}} \geq H_{2^k} + \frac12 \geq 1 + \frac{k+1}{2}.
\]
El \omterm{def:g12:seq:sequence}{secuencia} $(H_n)$ es \omterm{def:g12:seq:monotonic}{creciente} (cada paso agrega $\frac{1}{n+1} > 0$) y
la subsecuencia $H_{2^k}$ no está limitada, por lo que $(H_n)$ no es \omterm{def:g12:seq:bounded}{delimitado arriba}.
\omterm{def:g12:seq:monotonic}{Creciente} y ilimitado, tiende a $+\infty$
(\cref{thm:g12:seq:monotone}).
\end{solution}

\begin{solution}{exo:g12:seq:10}
\emph{1.} El \omterm{def:g12:seq:sequence}{secuencia} $d_n = b_n - a_n$ satisface
$d_{n+1} - d_n = (b_{n+1} - b_n) - (a_{n+1} - a_n) \leq 0$, entonces $(d_n)$ es
\omterm{def:g10:functions:variations}{decreciente}; desde $d_n \to 0$, obtenemos $d_n \geq 0$ para todos los $n$ (un \omterm{def:g10:functions:variations}{decreciente}
\omterm{def:g12:seq:sequence}{secuencia} con un término negativo permanecería por debajo de él para siempre, evitando
límite $0$). Por lo tanto $a_n \leq b_n$.

\emph{2.} De $a_n \leq b_n \leq b_0$, el \omterm{def:g12:seq:monotonic}{creciente} \omterm{def:g12:seq:sequence}{secuencia} $(a_n)$ es
\omterm{def:g12:seq:bounded}{delimitado arriba}, por lo que \omterm{def:g12:seq:limit}{converge} es $\ell$. Del mismo modo $(b_n)$, \omterm{def:g10:functions:variations}{decreciente}
y \omterm{def:g12:seq:bounded}{delimitado por debajo} por $a_0$, \omterm{def:g12:seq:limit}{converge} a algunos $\ell'$. entonces
$\ell' - \ell = \lim (b_n - a_n) = 0$, entonces $\ell = \ell'$.

\emph{3.} $(a_n)$ es (estrictamente) \omterm{def:g12:seq:monotonic}{creciente} desde
$a_{n+1} - a_n = \frac{1}{(n+1)!} > 0$. Para $(b_n)$,
\[
b_{n+1} - b_n = \frac{1}{(n+1)!} + \frac{1}{(n+1)(n+1)!} - \frac{1}{n\,n!}
= \frac{n(n+1) + n - (n+1)^2}{n(n+1)(n+1)!}
= \frac{-1}{n(n+1)(n+1)!} < 0 ,
\]
entonces $(b_n)$ es \omterm{def:g10:functions:variations}{decreciente}. Finalmente $b_n - a_n = \frac{1}{n\,n!} \to 0$. el
dos \omterm{def:g12:seq:sequence}{secuencias} son adyacentes, por lo tanto convergen a un límite común.
\end{solution}

\begin{solution}{pb:g12:seq:1}
\textbf{1.} Verdadero para $n = 1$ ($1 = 1^2$). si
$1 + 3 + \dots + (2n - 1) = n^2$, luego sumando el siguiente impar
número: $n^2 + (2n + 1) = (n + 1)^2$: herencia. Por inducción,
cierto para todos los $n \geq 1$.

\textbf{2.} $2^0 = 1 > 0$. Si $2^n > n$, entonces
$2^{n+1} = 2 \cdot 2^n > 2n \geq n + 1$ para $n \geq 1$ (y
$n = 0$ comprueba directamente): herencia, hecho.

\textbf{3.} $n = 0$: $1 \geq 1$. si
$(1 + x)^n \geq 1 + nx$, multiplicar por $1 + x \geq 1 > 0$:
$(1 + x)^{n+1} \geq (1 + nx)(1 + x) = 1 + (n + 1)x + nx^2
\geq 1 + (n + 1)x$.

\textbf{4.} El paso de $n = 1$ a $n = 2$: entre dos
canicas, ``la primera $n$'' y ``la última $n$'' son dos
\emph{desarticular} canicas individuales --- ninguna canica común une el
dos grupos, por lo que nada obliga a sus colores a coincidir. el
El argumento de la herencia requiere silenciosamente que los dos grupos se superpongan,
verdadero sólo desde $n \geq 2$; con la caja base $n = 1$ la cadena
nunca comienza.

\textbf{5.}$4^0 - 1 = 0 = 3 \times 0$. Si $4^n - 1 = 3k$,
luego $4^{n+1} - 1 = 4(4^n - 1) + 3 = 3(4k + 1)$: herencia.

\textbf{6.} $x_1 = \frac32$, $x_2 = \frac{17}{12}$,
$x_3 = \frac{577}{408}$.

\textbf{7.} $x_{n+1}^2 - 2 = \frac{(x_n^2 + 2)^2 - 8x_n^2}
{4x_n^2} = \frac{(x_n^2 - 2)^2}{4 x_n^2}$: un cuadrado sobre un
positivo, por lo tanto $\geq 0$ y $> 0$ siempre que $x_n^2 \neq 2$.
Inducción: $x_0 = 2 > 0$ con $x_0^2 = 4 > 2$; si $x_n > 0$
y $x_n^2 > 2$, entonces $x_{n+1}$ (un promedio de positivos) es
positivo y $x_{n+1}^2 - 2 > 0$.

\textbf{8.} $x_{n+1} - x_n = \frac{2 - x_n^2}{2x_n} < 0$ por
pregunta 7: estrictamente \omterm{def:g10:functions:variations}{decreciente}.

\textbf{9.} \omterm{def:g10:functions:variations}{Decreciente} y \omterm{def:g12:seq:bounded}{delimitado por debajo} (por $1$, desde
$x_n^2 > 2 > 1$ y $x_n > 0$): por la convergencia monótona
teorema, $(x_n)$ \omterm{def:g12:seq:limit}{converge} a algunos $L \geq 1$.

\textbf{10.} Los límites respetan el álgebra: desde
$x_{n+1} = \frac12\left(x_n + \frac{2}{x_n}\right)$ y
$x_n \to L \geq 1 > 0$:
$L = \frac12\left(L + \frac2L\right)$, entonces $L^2 = 2$ y $L$
siendo positivo, $L = \sqrt2$. Veredicto: convergencia demostrada,
límite identificado --- Heron absuelto con honores.

\textbf{11.} $x_{n+1} - \sqrt2 = \frac{x_n^2 - 2\sqrt2\,x_n +
2}{2x_n} = \frac{(x_n - \sqrt2)^2}{2x_n}$: exactamente
$e_{n+1} = \frac{e_n^2}{2x_n}$ y $x_n > \sqrt2$ dan
$e_{n+1} \leq \frac{e_n^2}{2\sqrt2}$. Error al cuadrado: cada paso
duplica el número de decimales correctos, como se observa desde
grado 9.

\textbf{12.} $e_0 \approx 0.5858$, $e_1 \approx 0.0858$,
$e_2 \approx 0.00245$, $e_3 \approx 2.1 \times 10^{-6}$.
Relaciones $\frac{e_1}{e_0^2} \approx 0.25 = \frac{1}{2x_0}$;
$\frac{e_2}{e_1^2} \approx 0.333 = \frac{1}{2x_1}$;
$\frac{e_3}{e_2^2} \approx 0.353 \approx \frac{1}{2x_2}$: el
teorema en acción.

\textbf{13.} $H_{2^{18}} \geq 1 + 9 = 10$: acerca de
Términos $260\,000$ ($2^{18} = 262\,144$) simplemente para pasar $10$ ---
divergencia a paso de tortuga (y $H_n > 100$ necesitaría más términos
que los átomos en cualquier biblioteca).

\textbf{14.} $S_n = 2 - \frac{1}{2^n}$ (suma \omterm{def:g12:seq:arith-geom}{geométrico}), y
$\frac{1}{2^n} \to 0$ (\cref{thm:g12:seq:geometric}), entonces
$S_n \to 2$: la barra de chocolate mordida sin cesar tiende a,
sin llegar nunca, al conjunto --- ahora en el oficial
lenguaje de límites.

\textbf{15.} Para $k \geq 2$:
$\frac{1}{k^2} \leq \frac{1}{k(k-1)} = \frac{1}{k-1} -
\frac1k$, entonces
$S_n \leq 1 + \left(1 - \frac1n\right) < 2$: \omterm{def:g12:seq:monotonic}{creciente} y
\omterm{def:g12:seq:bounded}{delimitado arriba}, por lo tanto convergente (convergencia monótona). Euler
Más tarde nombró el límite: $\frac{\pi^2}{6}$.

\textbf{16.} Los términos tendientes a $0$ son \emph{necesario} para el
sumas por saldar pero no decide nada: los términos armónicos
$\frac1n \to 0$ pero las sumas explotan; los términos
$\frac{1}{n^2} \to 0$ y las sumas convergen. Cómo \emph{rápido}
los términos morir es toda la cuestión --- la teoría de las series,
construido en los volúmenes universitarios.

\textbf{17.} $a_1 = \sqrt2 \approx 1.41421$, $b_1 = 1.5$;
$a_2 \approx 1.45648$, $b_2 \approx 1.45711$: dos iteraciones
ya estamos de acuerdo con tres decimales --- velocidad asombrosa.

\textbf{18.} $b_{n+1} - a_{n+1} = \frac{a_n + b_n}{2} -
\sqrt{a_n b_n} = \frac{(\sqrt{b_n} - \sqrt{a_n})^2}{2} \geq
0$: significa permanecer ordenado. $(a_n)$ aumenta:
$a_{n+1} = \sqrt{a_n b_n} \geq \sqrt{a_n \cdot a_n} = a_n$;
$(b_n)$ disminuye simétricamente.

\textbf{19.} $\frac{b_{n+1} - a_{n+1}}{b_n - a_n} =
\frac{(\sqrt{b_n} - \sqrt{a_n})^2}
{2(\sqrt{b_n} - \sqrt{a_n})(\sqrt{b_n} + \sqrt{a_n})}
= \frac{\sqrt{b_n} - \sqrt{a_n}}{2(\sqrt{b_n} + \sqrt{a_n})}
\leq \frac12$: la brecha se reduce al menos a la mitad, por lo que $b_n - a_n \to 0$;
con la pregunta 18, los \omterm{def:g12:seq:sequence}{secuencias} son adyacentes y comparten un límite
$M(1, 2)$.

\textbf{20.} Iteración tres da
$a_3 \approx b_3 \approx 1.456791$: $M(1, 2) \approx
1.456791$ en tres vueltas de manivela (la separación es aproximadamente
\emph{cuadrícula}, como el de Heron). Clasificación de velocidad de los problemas.
\omterm{def:g12:seq:sequence}{secuencias}, de más lento a más rápido: las sumas armónicas (glaciales
divergencia), las sumas \omterm{def:g12:seq:arith-geom}{geométrico} (el error se redujo a la mitad en cada paso),
Heron y la Asamblea General Anual (error al cuadrado en cada paso) --- y fue el
La velocidad sobrenatural de AGM que le dijo a Gauss que había encontrado una nueva veta
de análisis.
\end{solution}
