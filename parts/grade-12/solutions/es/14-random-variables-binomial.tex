\section*{Capítulo \ref{ch:g12:randvar} --- Variables aleatorias y distribución binomial}

\begin{solution}{exo:g12:randvar:1}
$\E(X) = -0.3 + 0 + 0.8 + 0.5 = 1$.
$\E(X^2) = 0.3 \times 1 + 0 + 0.4\times4 + 0.1\times25 = 4.4$, entonces por
König--Huygens $\V(X) = 4.4 - 1 = 3.4$ y
$\sigma(X) = \sqrt{3.4} \approx 1.84$.
\end{solution}

\begin{solution}{exo:g12:randvar:2}
\emph{1.} Las preguntas $10$ son \omterm{def:g12:condprob:indep}{independiente} \omterm{def:g12:randvar:bernoulli}{juicios de Bernoulli} con
$p = \frac14$: $X \sim \mathcal B\left(10, \frac14\right)$,
$\E(X) = 2.5$,
$\sigma(X) = \sqrt{10 \times \frac14 \times \frac34} = \sqrt{1.875}
\approx 1.37$.

\emph{2.} $\P(X = 0) = \left(\frac34\right)^{10} \approx 0.056$;
\[
\P(X = 5) = \binom{10}{5}\left(\frac14\right)^5\left(\frac34\right)^5
\approx 0.058;
\quad
\P(X \geq 1) = 1 - \left(\tfrac34\right)^{10} \approx 0.944 .
\]
\end{solution}

\begin{solution}{exo:g12:randvar:3}
\omterm{def:g12:condprob:indep}{Independiente} pruebas idénticas: $X \sim \mathcal B(400,\ 0.05)$, entonces
$\E(X) = 20$ reclamaciones y
\[
\sigma(X) = \sqrt{400 \times 0.05 \times 0.95} = \sqrt{19} \approx 4.4 .
\]
\end{solution}

\begin{solution}{exo:g12:randvar:4}
El pago recibido $P$ satisface
$\E(P) = \frac{5 + 6}{6} = \frac{11}{6}$, por lo que el precio justo es
$c = \frac{11}{6} \approx 1.83$ euros. La ganancia justa $G = P - \frac{11}6$
toma valores $-\frac{11}{6}, \frac{19}{6}, \frac{25}{6}$ con probabilidades
$\frac46, \frac16, \frac16$:
\[
\V(G) = \E(G^2)
= \frac{4 \times 121 + 361 + 625}{6 \times 36} = \frac{1470}{216}
= \frac{245}{36} \approx 6.8,
\qquad \sigma(G) \approx 2.6 .
\]
Un juego limpio tiene cero ganancias \emph{promedio} pero sus resultados aún fluctúan:
Lo justo no está exento de riesgos.
\end{solution}

\begin{solution}{exo:g12:randvar:5}
$X \sim \mathcal B(8,\ 0.7)$.
\[
\P(X = 6) = \binom86 (0.7)^6(0.3)^2 \approx 0.296 ;
\]
\[
\P(X \geq 6) = \P(6) + \P(7) + \P(8)
\approx 0.296 + 8(0.7)^7(0.3) + (0.7)^8
\approx 0.296 + 0.198 + 0.058 = 0.552 ;
\]
$\P(X \geq 1) = 1 - (0.3)^8 \approx 0.99993$. Modo:
$(n+1)p = 6.3$, por lo que el valor más probable es $k^* = 6$
(\cref{exo:g12:randvar:7}).
\end{solution}

\begin{solution}{exo:g12:randvar:6}
El número de pasajeros que se presentan es $X \sim \mathcal B(104,\ 0.9)$; el
el vuelo está sobrevendido cuando $X \geq 101$:
\[
\P(X \geq 101) = \sum_{k=101}^{104}\binom{104}{k}(0.9)^k(0.1)^{104-k}
\approx 0.006 .
\]
Vender $4\%$ más entradas que asientos provoca un incidente sólo en aproximadamente
$0.6\%$ de vuelos: la economía detrás del overbooking.
\end{solution}

\begin{solution}{exo:g12:randvar:7}
\[
\frac{\P(X = k+1)}{\P(X = k)}
= \frac{\binom{n}{k+1}}{\binom nk}\cdot\frac{p}{1-p}
= \frac{n - k}{k + 1}\cdot\frac{p}{1-p},
\]
utilizando $\binom{n}{k+1} = \binom nk \frac{n-k}{k+1}$. Esta relación es $\geq 1$
si $(n-k)p \geq (k+1)(1-p)$ si $np - k p \geq k - kp + 1 - p$ si
$k \leq (n+1)p - 1$. Entonces las probabilidades aumentan estrictamente mientras
$k + 1 \leq (n+1)p$ y disminuir después: el \omterm{def:g10:functions:extrema}{máximo} se alcanza en
$k^* = \floor{(n+1)p}$ (compartido con $k^* - 1$ cuando $(n+1)p$ es un
\omterm{def:g10:numbers:sets}{entero}).
\end{solution}

\begin{solution}{exo:g12:randvar:8}
\emph{1.} Cara primero en el lanzamiento $k$ significa $k-1$ cruz y luego cara:
\omterm{def:g10:proba:distribution}{probabilidad} $\left(\frac12\right)^{k-1}\cdot\frac12 = 2^{-k}$, para
$1 \leq k \leq 10$. Ganancia total \omterm{def:g10:proba:distribution}{probabilidad}
$\sum_{k=1}^{10} 2^{-k} = 1 - 2^{-10} = \frac{1023}{1024}$ (el juego es
perdió sólo en diez cruces consecutivas).

\emph{2.} Beneficio esperado:
\[
\sum_{k=1}^{10} 2^k \cdot 2^{-k} = \sum_{k=1}^{10} 1 = 10 \text{ euros}.
\]
Sin el límite, la suma $\sum_{k\geq1} 1$ diverge: el pago esperado es
infinito, aunque el juego casi siempre paga una pequeña cantidad --- el
famoso \emph{Paradoja de San Petersburgo}, demostrando que solo \omterm{def:g12:randvar:exp}{expectativa} lo hace
No medir el valor de un juego.
\end{solution}

\begin{solution}{pb:g12:randvar:1}
\textbf{1.} $X \sim \mathcal B\!\left(3, \frac16\right)$:
$\P(X = 0) = \frac{125}{216}$,
$\P(X = 1) = \frac{75}{216}$,
$\P(X = 2) = \frac{15}{216}$,
$\P(X = 3) = \frac{1}{216}$.

\textbf{2.} $\E(X) = 3 \times \frac16 = \frac12$;
$V(X) = 3 \times \frac16 \times \frac56 = \frac{5}{12}$.

\textbf{3.} Pago esperado $2\,\E(X) = 1$ euro contra un
$1$ apuesta en euros: ganancia $0$ --- exactamente justa (una rareza).

\textbf{4.} Sin reposición los sorteos son dependientes (el
las posibilidades de la segunda carta dependen de la primera): no binomial ---
La casilla \omterm{def:g12:condprob:indep}{independencia} de la lista de verificación falla. Con reemplazo:
fijo $n = 5$, constante $p = \frac14$, \omterm{def:g12:condprob:indep}{independiente} dibuja:
binomio.

\textbf{5.} $\E(X) = 6$, $\sigma(X) = \sqrt{4.2} \approx
2.05$. $\E(S) = 5 \times 6 - 10 = 20$;
$\sigma(S) = 5\sigma(X) \approx 10.25$.

\textbf{6.} Se corrigieron $n = 105$ pruebas (los boletos), cada una
show/no-show con el mismo $p = 0.9$, se supone \omterm{def:g12:condprob:indep}{independiente}:
binomio. La confesión: las familias pierden vuelos
\emph{juntos} y las tormentas vacían aviones enteros --- \omterm{def:g12:condprob:indep}{independencia}
es el acto de fe del modelo, y las ausencias correlacionadas hacen
las colas son más gordas que las promesas binomiales.

\textbf{7.} $\E(X) = 94.5$;
$\sigma(X) = \sqrt{105 \times 0.9 \times 0.1} =
\sqrt{9.45} \approx 3.07$.

\textbf{8.} $\P(X \geq 101) = \sum_{k=101}^{105}
\binom{105}{k} 0.9^k\, 0.1^{105 - k}$.

\textbf{9.} $\approx 0.0167$: aproximadamente un vuelo de cada sesenta
choca con cualquiera.

\textbf{10.} $\E(\text{bumped}) \approx 0.023$ pasajeros por
vuelo: compensación esperada $\approx 19$ euros --- contra
$1\,000$ euros de ingresos extra. El overbooking a las cinco es
abrumadoramente rentable; de ahí que todas las aerolíneas lo hagan.

\textbf{11.} De la tabla de probabilidades de cola:
$n = 105$: $1.7\,\%$; $n = 106$: $4.0\,\%$; $n = 107$:
$8.1\,\%$. La venta más grande que cumple con la norma es la de boletos $n = 106$.

\textbf{12.}$z = \frac{100.5 - 94.5}{3.07} \approx 1.95$:
el umbral se encuentra dos \omterm{def:g11:stat:variance}{desviaciones estándar} por encima del \omterm{def:g11:stat:mean}{significar},
y la regla general de la curva de campana (``superior $2\sigma$ cola
$\approx 2.5\,\%$'') aterriza cerca del $1.7\,\%$ exacto ---
la suave curva que ensombrece el binomio es la próxima
protagonista de los capítulos.

\textbf{13.} $0.98^{20} + 20 \times 0.02 \times 0.98^{19}
\approx 0.94$: el lote honesto pasa $94\,\%$ del tiempo.

\textbf{14.} $0.9^{20} + 20 \times 0.1 \times 0.9^{19}
\approx 0.39$: los malos todavía se cuelan por $39\,\%$ de
el tiempo.

\textbf{15.} Riesgo del productor: un buen lote rechazado
($\approx 6\,\%$); Riesgo del consumidor: un mal lote aceptado
($\approx 39\,\%$). Un \omterm{def:g10:proba:sample}{muestra} más grande (con un
umbral de aceptación) reduce ambos --- a costa de más
Inspección: la calidad tiene una línea presupuestaria.

\textbf{16.} $V\!\left(\frac Xn\right) = \frac{V(X)}{n^2} =
\frac{np(1-p)}{n^2} = \frac{p(1-p)}{n}$: \omterm{def:g11:stat:variance}{desviación estándar}
$\sqrt{\frac{p(1-p)}{n}}$. Cuádruple el \omterm{def:g10:proba:sample}{muestra}, reduzca a la mitad el
ruido: la ley $\frac{1}{\sqrt n}$ de la fluctuación de grado 10
\omterm{def:g10:numbers:interval}{intervalos}, ahora derivado.

\textbf{17.} El límite limita el pago en $2^{10} = 1024$
euros, por lo que cada una de las diez rondas aporta exactamente $1$ euro
de \omterm{def:g12:randvar:exp}{expectativa}: $\E = 10$. El juego sin límites es infinito.
\omterm{def:g12:randvar:exp}{expectativa} provino en su totalidad de astronómicamente raros,
pagos astronómicamente grandes; capping confiesa que no hay banco
paga $2^{50}$ euros. Un precio justo de entrada para el partido limitado:
$10$ euros --- y de repente nadie se lleva la paradoja.

\textbf{18.} Premio en metálico esperado por boleto:
$\frac{100 + 2 \times 50}{200} = 1$ euro frente a $2$ euro
billete: margen $50\,\%$ --- dieciocho veces la ruleta
$2.7\,\%$. El sorteo sobrevive porque sus jugadores son
donar a sabiendas: la "pérdida" es el punto.

\textbf{19.} Reclamaciones esperadas: $1\,000 \times 0.01 \times
10\,000 = 100\,000$; primas $130\,000$: beneficio esperado
$30\,000$ euros. \omterm{def:g11:stat:variance}{Desviación estándar} del total de reclamaciones:
$10\,000 \times \sqrt{1\,000 \times 0.01 \times 0.99}
\approx 31\,500$ euros --- un mal año cualquiera devora el
beneficio total esperado. De ahí las reservas de capital, los reaseguros,
y carteras mucho mayores que mil pólizas: aseguradoras
vivir de la ley de los grandes números y mantener reservas contra
su lentitud.

\textbf{20.} Lista de verificación primero --- arreglado $n$, mismo $p$,
\omterm{def:g12:condprob:indep}{independencia} \emph{reclamado} --- o ningún binomio. entonces
navegar por $\E \pm \sigma$: significa localizar, avisar desvíos.
Luego ponle precio a las colas: golpes, lotes malos, años ruinosos, todo
vivir más allá de $2\sigma$. Advertencias: \omterm{def:g12:condprob:indep}{independencia} es del modelador.
promesa, no la del mundo; y \omterm{def:g12:randvar:exp}{expectativa} ignora exactamente
lo que te destruye. Las páginas que faltan en el libro de reglas: ¿qué tan rápido?
las frecuencias se estabilizan y qué forma toman las fluctuaciones ---
son los siguientes dos capítulos.
\end{solution}
