\begin{solution}{exo:g12:contdist:9}
\emph{1.} Desde $\alpha > 0$, $c \leq \alpha X + \beta \leq d
\iff \frac{c - \beta}{\alpha} \leq X \leq \frac{d - \beta}{\alpha}$, entonces
\[
\P(c \leq Y \leq d)
= \int_{(c-\beta)/\alpha}^{(d-\beta)/\alpha} f(t)\,\dd t .
\]
La sustitución $y = \alpha t + \beta$ (es decir, \ leer el área en $y$
variable, $t = \frac{y - \beta}{\alpha}$, $\dd t = \frac{\dd y}{\alpha}$)
convierte esto en
$\int_c^d \frac{1}{\alpha} f\left(\frac{y-\beta}{\alpha}\right)\dd y$: el
\omterm{def:g11:func:function}{función} $g(y) = \frac1\alpha f\left(\frac{y-\beta}{\alpha}\right)$ es un
\omterm{def:g12:contdist:density}{densidad} de $Y$.

\emph{2.} Con $f = \varphi$, $\alpha = \sigma$, $\beta = \mu$:
\[
g(y) = \frac{1}{\sigma}\,\varphi\!\left(\frac{y - \mu}{\sigma}\right)
= \frac{1}{\sigma\sqrt{2\pi}}\,
\exp\left(-\frac{(y-\mu)^2}{2\sigma^2}\right),
\]
que es por tanto el \omterm{def:g12:contdist:density}{densidad} de
$\mathcal N(\mu, \sigma^2) = \mu + \sigma\,\mathcal N(0,1)$.
\end{solution}
