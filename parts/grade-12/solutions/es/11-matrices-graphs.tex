\begin{solution}{exo:g12:matrix:4}
\emph{1.} $N^2 = \begin{pmatrix} 0&1\\0&0\end{pmatrix}
\begin{pmatrix} 0&1\\0&0\end{pmatrix} = 0$ y $I$ conmutan con cada
\omterm{def:g12:matrix:matrix}{matriz}.

\emph{2.} Dado que los dos términos se conmutan, se aplica el teorema del binomio y todos
Los términos que contienen $N^2$ desaparecen:
\[
A^k = (2I + N)^k = 2^k I + k\,2^{k-1} N
= \begin{pmatrix} 2^k & k\,2^{k-1}\\ 0 & 2^k \end{pmatrix}.
\]
Compruebe si hay $k = 2$:
$A^2 = \begin{pmatrix} 2&1\\0&2\end{pmatrix}^2
= \begin{pmatrix} 4&4\\0&4\end{pmatrix}$, y la fórmula da
$2^2 = 4$, $2 \times 2 = 4$. \checkmark
\end{solution}

\begin{solution}{exo:g12:matrix:5}
\emph{Inducción.} Para $n = 1$:
$A^1 = \begin{pmatrix} 0&1\\1&1\end{pmatrix}
= \begin{pmatrix} F_0 & F_1\\ F_1 & F_2\end{pmatrix}$. Supongamos la fórmula
para $n$; entonces
\[
A^{n+1} = A^n A
= \begin{pmatrix} F_{n-1} & F_n\\ F_n & F_{n+1}\end{pmatrix}
\begin{pmatrix} 0 & 1\\ 1 & 1\end{pmatrix}
= \begin{pmatrix} F_n & F_{n-1} + F_n\\ F_{n+1} & F_n + F_{n+1}\end{pmatrix}
= \begin{pmatrix} F_n & F_{n+1}\\ F_{n+1} & F_{n+2}\end{pmatrix}.
\]
\emph{Identidad.} Para matrices $2\times2$, se muestra \omterm{def:g10:algebra:expand}{en expansión}
$\det(MN) = \det M \det N$; por lo tanto
$\det(A^n) = (\det A)^n = (-1)^n$, y
$\det A^n = F_{n-1}F_{n+1} - F_n^2$. (Este es \emph{La identidad de Cassini}.)
\end{solution}

\begin{solution}{exo:g12:matrix:6}
\emph{1.} Ordenar vértices $1, 2, 3$:
\[
M = \begin{pmatrix} 0&1&1\\ 0&0&1\\ 1&0&0\end{pmatrix}, \quad
M^2 = \begin{pmatrix} 1&0&1\\ 1&0&0\\ 0&1&1\end{pmatrix}, \quad
M^3 = \begin{pmatrix} 1&1&1\\ 1&0&1\\ 1&0&1 \end{pmatrix}.
\]

\emph{2.} $\bigl(M^3\bigr)_{11} = 1$: exactamente un paseo cerrado de longitud $3$
en el vértice $1$, es decir, $1 \to 2 \to 3 \to 1$. (El paseo
$1 \to 3 \to 1$ tiene longitud $2$ solamente, y $1 \to 3$ luego $3\to1$ luego
$1\to3$ termina en $3$.)
\end{solution}

\begin{solution}{exo:g12:matrix:7}
\emph{1.} $a_{n+1} = 0.8a_n + 0.3b_n$, $b_{n+1} = 0.2a_n + 0.7b_n$:
$M = \begin{pmatrix} 0.8 & 0.3\\ 0.2 & 0.7\end{pmatrix}$.

\emph{2.} $MX = X$ da $0.8a + 0.3b = a$, \emph{i.e.} $0.3b = 0.2a$, entonces
$b = \frac23 a$; con $a + b = 1$: $a = 0.6$, $b = 0.4$.

\emph{3.} Usando $b_n = 1 - a_n$:
$a_{n+1} = 0.8a_n + 0.3(1 - a_n) = 0.5a_n + 0.3$, entonces
\[
c_{n+1} = a_{n+1} - 0.6 = 0.5a_n + 0.3 - 0.6 = 0.5(a_n - 0.6) = 0.5\,c_n .
\]
Por lo tanto $c_n = 0.5^n c_0 \to 0$: $a_n \to 0.6$ y $b_n \to 0.4$, cualquiera que sea el
inicial \omterm{def:g11:prob:rv}{distribución}.
\end{solution}

\begin{solution}{exo:g12:matrix:8}
\emph{1.} $\det P = -2$, entonces
$P^{-1} = -\frac12\begin{pmatrix} -1 & -1\\ -1 & 1\end{pmatrix}
= \frac12\begin{pmatrix} 1 & 1\\ 1 & -1\end{pmatrix}$. entonces
\[
AP = \begin{pmatrix} 4 & 2\\ 4 & -2 \end{pmatrix}, \qquad
D = P^{-1}AP = \frac12\begin{pmatrix} 1&1\\1&-1\end{pmatrix}
\begin{pmatrix} 4&2\\4&-2\end{pmatrix}
= \begin{pmatrix} 4 & 0\\ 0 & 2\end{pmatrix}.
\]

\emph{2.} A partir de $A = PDP^{-1}$, una inducción inmediata da
$A^n = PD^nP^{-1}$ con
$D^n = \begin{pmatrix} 4^n & 0\\ 0 & 2^n\end{pmatrix}$, entonces
\[
A^n = P D^n P^{-1}
= \begin{pmatrix} 4^n & 2^n\\ 4^n & -2^n\end{pmatrix}\cdot
\frac12\begin{pmatrix} 1&1\\1&-1\end{pmatrix}
= \frac12\begin{pmatrix} 4^n + 2^n & 4^n - 2^n\\
4^n - 2^n & 4^n + 2^n\end{pmatrix}.
\]
Aplicando $A^n$ a $X_0 = \begin{pmatrix} u_0\\v_0\end{pmatrix}$ se reproduce
exactamente las fórmulas de \cref{ex:g12:matrix:coupled}.
\end{solution}

\begin{solution}{pb:g12:matrix:1}
\textbf{1.}$AB = \begin{pmatrix} 2 & 1\\ 4 & 3\end{pmatrix}$
y $BA = \begin{pmatrix} 3 & 4\\ 1 & 2\end{pmatrix}$: \omterm{def:g12:matrix:matrix}{matriz}
la multiplicación no es conmutativa --- $B$ intercambia columnas en el
derecha, filas a la izquierda.

\textbf{2.} \omterm{def:g11:vect:det}{Determinante} $6 - 5 = 1$: inversa
$\begin{pmatrix} 3 & -1\\ -5 & 2\end{pmatrix}$. Aplicándolo a
$\binom{4}{7}$: $x = 12 - 7 = 5$, $y = -20 + 14 = -6$.

\textbf{3.} $N^2 = 0$. Luego $(I + N)^n = I + nN$ por inducción:
$(I + nN)(I + N) = I + (n+1)N + nN^2 = I + (n+1)N$.

\textbf{4.} $A = \begin{pmatrix} 0&1&1\\ 1&0&1\\ 1&1&0
\end{pmatrix}$ y $A^3$ tienen entradas diagonales $2$: de cada
vértice, exactamente dos caminos cerrados de longitud $3$ (el triángulo
caminó en el sentido de las agujas del reloj o en el sentido contrario a las agujas del reloj) --- el teorema de conteo
en acción.

\textbf{5.} $D^n = \begin{pmatrix} 2^n & 0\\ 0 & 2^{-n}
\end{pmatrix}$: una dirección explota, la otra muere ---
los destinos diagonales son independientes \omterm{def:g11:seq:geometric}{secuencias geométricas}.

\textbf{6.}$F^2 = \begin{pmatrix} 2 & 1\\ 1 & 1\end{pmatrix}$,
$F^3 = \begin{pmatrix} 3 & 2\\ 2 & 1\end{pmatrix}$,
$F^4 = \begin{pmatrix} 5 & 3\\ 3 & 2\end{pmatrix}$: Fibonacci
en todas partes; conjetura como se dijo.

\textbf{7.} Si $F^n = \begin{pmatrix} F_{n+1} & F_n\\ F_n &
F_{n-1}\end{pmatrix}$, entonces
\[
F^{n+1} = F^n F =
\begin{pmatrix} F_{n+1} + F_n & F_{n+1}\\
F_n + F_{n-1} & F_n \end{pmatrix}
= \begin{pmatrix} F_{n+2} & F_{n+1}\\ F_{n+1} & F_n
\end{pmatrix} :
\]
herencia; el caso base $n = 1$ es el propio $F$, utilizando el
convención $F_0 = 0$ (que extiende la recurrencia hacia atrás).

\textbf{8.} $\det F = -1$, entonces
$\det(F^n) = (\det F)^n = (-1)^n$; y directamente
$\det(F^n) = F_{n+1}F_{n-1} - F_n^2$: Cassini, una línea. (El
regla de producto para $2 \times 2$ \omterm{def:g11:vect:det}{determinantes} es una agradable
expansión de cinco minutos.)

\textbf{9.} Arriba a la derecha de $F^m F^n$:
$F_{m+1}F_n + F_m F_{n-1}$; arriba a la derecha de $F^{m+n}$:
$F_{m+n}$. Para $m = n = 3$:
$F_4 F_3 + F_3 F_2 = 3 \times 2 + 2 \times 1 = 8 = F_6$.

\textbf{10.} Para $k = 1$: trivial. Si $F_n \mid F_{kn}$, el
fórmula de suma con $m = kn$:
$F_{(k+1)n} = F_{kn+1}F_n + F_{kn}F_{n-1}$: ambos términos son
múltiplos de $F_n$. Entonces $F_n \mid F_{kn}$ para todos los $k$: verificar
$F_3 = 2$ \omterm{def:g12:arith:divides}{divide} $F_6 = 8$ y $F_9 = 34$.

\textbf{11.} $F^{100} = F^{64} F^{32} F^4$: siete cuadraturas
($F^2, F^4, \dots, F^{64}$) más dos combinaciones --- nueve
multiplicaciones en lugar de noventa y nueve. es el
truco de la mesa de doblar de los escribas egipcios, tomado de
números a matrices: escriba $100$ en binario, multiplique el
duplicaciones que necesitas.

\textbf{12.} $0.8 + 0.2 = 1$ y $0.4 + 0.6 = 1$: mañana
debe ser \emph{alguno} clima --- cada columna es una información completa
\omterm{def:g10:proba:distribution}{probabilidad} \omterm{def:g11:prob:rv}{distribución}, por lo que las probabilidades se conservan.

\textbf{13.} Mañana: $\binom{0.8}{0.2}$. Día después:
