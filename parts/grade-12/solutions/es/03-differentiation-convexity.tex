\section*{Capítulo \ref{ch:g12:deriv} --- Diferenciación y convexidad}

\begin{solution}{exo:g12:deriv:1}
$f'(x) = 3x^2 - 5$.

Regla del cociente:
$g'(x) = \dfrac{(x^2+1) - x\cdot 2x}{(x^2+1)^2} = \dfrac{1 - x^2}{(x^2+1)^2}$.

Regla de la cadena con $u = 2x+1$: $h'(x) = 5 \cdot 2 \cdot (2x+1)^4 = 10(2x+1)^4$.

Regla de la cadena con $u = x^2 + 1$:
$k'(x) = \dfrac{2x}{2\sqrt{x^2+1}} = \dfrac{x}{\sqrt{x^2+1}}$.
\end{solution}

\begin{solution}{exo:g12:deriv:2}
$f'(x) = 2x - 3$, entonces $f(2) = -1$ y $f'(2) = 1$: el \omterm{def:g12:deriv:tangent}{tangente} es
$y = -1 + (x - 2) = x - 3$. La brecha es
\[
f(x) - (x - 3) = x^2 - 4x + 4 = (x-2)^2 \geq 0 ,
\]
por lo que la curva queda por encima de \omterm{def:g12:deriv:tangent}{tangente} en todas partes, tocándola sólo en $x = 2$
(como se esperaba: $f$ es \omterm{def:g12:deriv:convex}{convexo}).
\end{solution}

\begin{solution}{exo:g12:deriv:3}
Ambos son cocientes de diferencias. Con $f(x) = (1+x)^{100}$ en $0$:
$f'(x) = 100(1+x)^{99}$, por lo que el límite es $f'(0) = 100$.
Con $g(x) = \sqrt{x}$ en $4$: $g'(x) = \frac{1}{2\sqrt{x}}$, entonces el límite es
$g'(4) = \frac14$.
\end{solution}

\begin{solution}{exo:g12:deriv:4}
\omterm{def:g11:func:function}{Dominio} $\R \setminus \{1\}$. Límites: en $\pm\infty$, $f(x) \sim x \to
\pm\infty$; en $1^\pm$, numerador $\to 4 > 0$ y denominador $\to 0^\pm$,
entonces $f \to \pm\infty$; la línea $x = 1$ es una \omterm{def:g12:limcont:asymptote}{asíntota vertical}.

\[
f'(x) = \frac{2x(x-1) - (x^2+3)}{(x-1)^2} = \frac{x^2 - 2x - 3}{(x-1)^2}
= \frac{(x-3)(x+1)}{(x-1)^2}.
\]
Por lo tanto $f' > 0$ en $\intoo{-\infty}{-1}$ y $\intoo{3}{+\infty}$, $f' < 0$ en
$\intoo{-1}{1}$ y $\intoo{1}{3}$. El \omterm{def:g11:func:function}{función} aumenta a un \omterm{def:g12:deriv:extremum}{máximo local}
$f(-1) = -2$, disminuye a $-\infty$, salta a $+\infty$ después de \omterm{def:g12:limcont:asymptote}{asíntota},
disminuye a un \omterm{def:g10:functions:extrema}{mínimo} $f(3) = 6$ local y luego aumenta.
\end{solution}

\begin{solution}{exo:g12:deriv:5}
Para $x \in \intoo{0}{15}$, la caja tiene una base cuadrada de lado $30 - 2x$ y
altura $x$, entonces
\[
V(x) = x(30 - 2x)^2 .
\]
$V'(x) = (30-2x)^2 + x \cdot 2(30-2x)(-2) = (30-2x)(30 - 2x - 4x)
= (30-2x)(30-6x)$. En $\intoo{0}{15}$, $30 - 2x > 0$, entonces $V'$ tiene el signo de
$30 - 6x$: positivo antes de $x = 5$, negativo después. El volumen es máximo para
$x = 5$ cm, dando $V(5) = 5 \times 20^2 = 2000\ \text{cm}^3$.
\end{solution}

\begin{solution}{exo:g12:deriv:6}
\emph{1.} $f'(x) = 4x^3 - 12x^2$ y $f''(x) = 12x^2 - 24x = 12x(x-2)$.
Así, $f'' > 0$ en $\intoo{-\infty}{0}$ y en $\intoo{2}{+\infty}$ (\omterm{def:g12:deriv:convex}{convexo}),
$f'' < 0$ en $\intoo{0}{2}$ (\omterm{def:g12:deriv:convex}{cóncavo}). $f''$ cambia de signo en $0$ y en $2$:
ambos son \omterm{def:g12:deriv:inflection}{puntos de inflexión}.

\emph{2.} En $a = 2$: $f(2) = 16 - 32 + 10 = -6$, $f'(2) = 32 - 48 = -16$,
\omterm{def:g12:deriv:tangent}{tangente} $y = -6 - 16(x-2) = -16x + 26$. La brecha es
\[
g(x) = f(x) + 16x - 26 = x^4 - 4x^3 + 16x - 16 .
\]
Dado que $g(2) = g'(2) = g''(2) = 0$, $(x-2)^3$ divide $g$; la división da
$g(x) = (x-2)^3(x+2)$. Cerca de $x = 2$, $x + 2 > 0$, por lo que $g$ tiene el signo de
$(x-2)^3$: negativo antes de $2$, positivo después. La curva cruza el
\omterm{def:g12:deriv:tangent}{tangente}: $2$ es de hecho un \omterm{def:g12:deriv:inflection}{punto de inflexión}.
\end{solution}

\begin{solution}{exo:g12:deriv:7}
El \omterm{def:g11:func:function}{función} $f(x) = \sqrt{x}$ satisface
$f''(x) = -\frac{1}{4}x^{-3/2} < 0$ en $\intoo{0}{+\infty}$: es \omterm{def:g12:deriv:convex}{cóncavo},
entonces su \omterm{def:g10:functions:graph}{gráfico} se encuentra \emph{abajo} cada \omterm{def:g12:deriv:tangent}{tangente}. El \omterm{def:g12:deriv:tangent}{tangente} en $a = 1$ es
\[
y = f(1) + f'(1)(x - 1) = 1 + \tfrac12 (x-1) = \frac{x+1}{2},
\]
de donde $\sqrt{x} \leq \frac{x+1}{2}$ para todos los $x > 0$. La igualdad significa
la curva toca el \omterm{def:g12:deriv:tangent}{tangente}, lo que ocurre sólo en el punto de tangencia
$x = 1$. (Equivalentemente: $\left(\sqrt x - 1\right)^2 \geq 0$.)
\end{solution}

\begin{solution}{exo:g12:deriv:8}
Por \cref{thm:g12:deriv:convexity}, el \omterm{def:g10:functions:graph}{gráfico} de un \omterm{def:g12:deriv:derivative}{diferenciable} \omterm{def:g12:deriv:convex}{convexo}
\omterm{def:g11:func:function}{función} se encuentra encima de cada uno de sus \omterm{def:g12:deriv:tangent}{tangentes}. El \omterm{def:g12:deriv:tangent}{tangente} en $a$ es horizontal
($f'(a) = 0$), de \omterm{def:g10:algebra:equation}{ecuación} $y = f(a)$. Por lo tanto $f(x) \geq f(a)$ para todos
$x \in \R$: $f(a)$ es el \omterm{def:g10:functions:extrema}{mínimo} global.
\end{solution}

\begin{solution}{exo:g12:deriv:9}
\emph{Perímetro fijo.} Un rectángulo con lados $x$ y $\frac{p}{2} - x$
($0 < x < \frac p2$) tiene el área $S(x) = x\left(\frac p2 - x\right)$. entonces
$S'(x) = \frac p2 - 2x$ desaparece en $x = \frac p4$, positivo antes y
negativo después: el \omterm{def:g10:functions:extrema}{máximo} está en $x = \frac{p}{4}$, donde ambos lados son iguales
$\frac p4$ --- un cuadrado.

\emph{Área fija.} Los lados $x$ y $\frac Ax$ dan perímetro
$P(x) = 2\left(x + \frac Ax\right)$, con
$P'(x) = 2\left(1 - \frac{A}{x^2}\right)$, negativo para $x < \sqrt A$ y
positivo después: \omterm{def:g10:functions:extrema}{mínimo} en $x = \sqrt{A}$, un cuadrado nuevamente.

\emph{Relación.} Las dos declaraciones son duales. Supongamos un rectángulo no cuadrado
minimizó el perímetro en el área fija $A$; el cuadrado del mismo perímetro
tendría el área $> A$ por la primera declaración, y la reduciría homotéticamente
al área $A$ disminuiría estrictamente su perímetro --- contradiciendo
minimalidad. Por tanto, cada afirmación implica la otra.
\end{solution}
