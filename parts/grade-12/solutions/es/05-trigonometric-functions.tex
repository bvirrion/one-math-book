\section*{Capítulo \ref{ch:g12:trigo} --- Funciones trigonométricas}

\begin{solution}{exo:g12:trigo:1}
$\cos\frac{2\pi}{3} = \cos\left(\pi - \frac\pi3\right) = -\cos\frac\pi3
= -\frac12$.

$\sin\frac{5\pi}{6} = \sin\left(\pi - \frac\pi6\right) = \sin\frac\pi6
= \frac12$.

$\cos\frac{7\pi}{4} = \cos\left(2\pi - \frac{\pi}{4}\right)
= \cos\left(-\frac\pi4\right) = \frac{\sqrt2}{2}$.

$\sin\left(-\frac\pi3\right) = -\sin\frac\pi3 = -\frac{\sqrt3}{2}$.
\end{solution}

\begin{solution}{exo:g12:trigo:2}
$f'(x) = 2\sin x\cos x = \sin 2x$ (regla de la cadena).

$g'(x) = -3\sin(3x) + \sin x + x\cos x$ (reglas de cadena y producto).

Regla del cociente:
\[
h'(x) = \frac{\cos x\,(2 + \cos x) - \sin x\,(-\sin x)}{(2+\cos x)^2}
= \frac{2\cos x + \cos^2 x + \sin^2 x}{(2+\cos x)^2}
= \frac{2\cos x + 1}{(2+\cos x)^2}.
\]
\end{solution}

\begin{solution}{exo:g12:trigo:3}
\emph{(a)} Una solución particular de $\sin x = \frac{\sqrt3}{2}$ es
$\frac\pi3$. Soluciones generales: $x = \frac\pi3 + 2k\pi$ o
$x = \pi - \frac\pi3 + 2k\pi = \frac{2\pi}{3} + 2k\pi$. en
$\intoc{-\pi}{\pi}$: $x \in \left\{\frac\pi3, \frac{2\pi}{3}\right\}$.

\emph{(b)} $\cos 2x = \frac{\sqrt2}{2} = \cos\frac\pi4$ da
$2x = \pm\frac\pi4 + 2k\pi$, entonces $x = \pm\frac\pi8 + k\pi$. en
$\intoc{-\pi}{\pi}$:
$x \in \left\{-\frac{7\pi}{8}, -\frac\pi8, \frac\pi8, \frac{7\pi}{8}\right\}$.
\end{solution}

\begin{solution}{exo:g12:trigo:4}
Con $a = \frac\pi3$, $b = \frac\pi4$:
\[
\cos\frac{\pi}{12} = \cos a\cos b + \sin a \sin b
= \frac12\cdot\frac{\sqrt2}{2} + \frac{\sqrt3}{2}\cdot\frac{\sqrt2}{2}
= \frac{\sqrt2 + \sqrt6}{4},
\]
\[
\sin\frac{\pi}{12} = \sin a\cos b - \cos a \sin b
= \frac{\sqrt3}{2}\cdot\frac{\sqrt2}{2} - \frac12\cdot\frac{\sqrt2}{2}
= \frac{\sqrt6 - \sqrt2}{4}.
\]
\end{solution}

\begin{solution}{exo:g12:trigo:5}
Establecer $s = \sin x$: $2s^2 - s - 1 = (2s + 1)(s - 1) \geq 0$ si y así
$s \leq -\frac12$ o $s = 1$.

$\sin x = 1$ da $x = \frac\pi2$. $\sin x \leq -\frac12$: en la unidad
círculo, el arco debajo de la línea horizontal $y = -\frac12$, que en
$\intcc{0}{2\pi}$ es $x \in \intcc{\frac{7\pi}{6}}{\frac{11\pi}{6}}$.
Por lo tanto
\[
S = \left\{\frac{\pi}{2}\right\} \cup
\intcc{\frac{7\pi}{6}}{\frac{11\pi}{6}} .
\]
\end{solution}

\begin{solution}{exo:g12:trigo:6}
$\dfrac{\sin 3x}{x} = 3\,\dfrac{\sin 3x}{3x} \to 3 \cdot 1 = 3$
(composición con $u = 3x \to 0$).

Usando $1 - \cos x = \dfrac{\sin^2 x}{1 + \cos x}$:
\[
\frac{1 - \cos x}{x^2}
= \left(\frac{\sin x}{x}\right)^{\!2} \frac{1}{1 + \cos x}
\xrightarrow[x\to0]{} 1 \cdot \frac12 = \frac12 .
\]

Sustituyendo $u = \frac1x \to 0^+$:
$x \sin\frac1x = \frac{\sin u}{u} \to 1$.
\end{solution}

\begin{solution}{exo:g12:trigo:7}
$f'(x) = 1 - \sin x \geq 0$ para todos los $x$, con igualdad exactamente cuando
$\sin x = 1$, \emph{i.e.} $x = \frac\pi2 + 2k\pi$. En $\intcc{0}{2\pi}$, $f$
es \omterm{def:g12:seq:monotonic}{creciente} de $f(0) = 1$ a $f(2\pi) = 2\pi + 1$ (el cero de $f'$ en
$\frac{\pi}{2}$ está aislado, por lo que el aumento es incluso estricto); \omterm{def:g10:functions:extrema}{mínimo} $1$ en
$0$, \omterm{def:g10:functions:extrema}{máximo} $2\pi + 1$ y $2\pi$. En $\R$, $f' \geq 0$ en todas partes con solo
ceros aislados, por lo que $f$ es (estrictamente) \omterm{def:g12:seq:monotonic}{creciente} por
\cref{thm:g12:deriv:variations}, aunque $f'$ desaparece en cada
$\frac\pi2 + 2k\pi$.
\end{solution}

\begin{solution}{exo:g12:trigo:8}
\emph{1.} Vamos $R = \sqrt{A^2 + B^2} > 0$. el punto
$\left(\frac AR, \frac BR\right)$ se encuentra en el círculo unitario, por lo que existe
$\varphi$ con $\cos\varphi = \frac AR$ y $\sin\varphi = \frac BR$. entonces
\[
R\cos(x - \varphi) = R\bigl(\cos x\cos\varphi + \sin x \sin\varphi\bigr)
= A\cos x + B\sin x = f(x).
\]

\emph{2.} Dado que $\cos$ tiene el rango $\intcc{-1}{1}$, $f$ tiene \omterm{def:g10:functions:extrema}{máximo} $R$ y
\omterm{def:g10:functions:extrema}{mínimo}$-R$. Para $\cos x + \sin x$: $A = B = 1$, $R = \sqrt2$,
$\varphi = \frac\pi4$, por lo que \omterm{def:g10:algebra:equation}{ecuación} se convierte en
$\sqrt2\cos\left(x - \frac\pi4\right) = 1$, \emph{i.e.}
$\cos\left(x - \frac\pi4\right) = \frac{\sqrt2}{2}$, dando
$x - \frac\pi4 = \pm\frac\pi4 + 2k\pi$: en $\intoc{-\pi}{\pi}$,
$x \in \left\{0, \frac\pi2\right\}$.
\end{solution}

\begin{solution}{exo:g12:trigo:9}
$(\sin)'' = (\cos)' = -\sin$ y $(\cos)'' = (-\sin)' = -\cos$: ambos resuelven
$y'' = -y$.

Deje $g = y - a\cos - b\sin$. Entonces $g'' = -g$ (linealidad de diferenciación),
$g(0) = y(0) - a = 0$ y $g'(0) = y'(0) - b = 0$. conjunto
$E = g^2 + (g')^2$. entonces
\[
E' = 2g g' + 2g' g'' = 2g g' - 2g' g = 0,
\]
entonces $E$ es una constante igual a $E(0) = 0$. Una suma de dos cuadrados que desaparece.
fuerza de manera idéntica a $g = 0$, por lo tanto $y = a\cos + b\sin$.
\end{solution}

\begin{solution}{pb:g12:trigo:1}
\textbf{1.} $0.99833\ldots$ y $0.99998\ldots$: arrastrándose hacia
$1$.

\textbf{2.} Triángulo $OAM$: base $1$, altura $\sin x$: área
$\frac{\sin x}{2}$. Sector $OAM$: fracción
$\frac{x}{2\pi}$ del disco unitario: área $\frac x2$. Triangulo
$OAT$: base $1$, altura $\tan x$: área $\frac{\tan x}{2}$.

\textbf{3.} El triángulo se encuentra dentro del sector dentro del gran
triángulo: $\frac{\sin x}{2} < \frac x2 < \frac{\tan x}{2}$,
es decir,\ $\sin x < x < \tan x$. De $\sin x < x$:
$\frac{\sin x}{x} < 1$; y de
$x < \tan x = \frac{\sin x}{\cos x}$, multiplicando por
$\frac{\cos x}{x} > 0$: $\cos x < \frac{\sin x}{x}$. como
$x \to 0^+$, $\cos x \to 1$: exprimido,
$\frac{\sin x}{x} \to 1$; y $\frac{\sin x}{x}$ es par, entonces
el límite bilateral es $1$.

\textbf{4.} $\frac{1 - \cos x}{x^2} =
\frac{1 - \cos^2 x}{x^2 (1 + \cos x)} =
\left(\frac{\sin x}{x}\right)^2 \frac{1}{1 + \cos x}
\to 1 \times \frac12$.

\textbf{5.} $\sin'(0) = \lim_{h \to 0} \frac{\sin h - 0}{h} =
1$: toda la diferenciación de \omterm{def:g12:trigo:cossin}{seno} y \omterm{def:g12:trigo:cossin}{coseno} fluye de
éste límite. Medido en grados, el límite sería
$\frac{\pi}{180}$, y cada \omterm{def:g12:deriv:derivative}{derivado} arrastraría eso
constante: \omterm{def:g11:trigo:radian}{radianes} son exactamente la unidad que hace el cálculo
de oscilación limpia.

\textbf{6.} $y' = -R\omega\sin(\omega t - \varphi)$,
$y'' = -R\omega^2\cos(\omega t - \varphi) = -\omega^2 y$.

\textbf{7.} $\omega = 2$. $y(0) = A = 3$;
$y'(0) = 2B = 8$: $B = 4$: $y = 3\cos 2t + 4\sin 2t$.

\textbf{8.}$R = \sqrt{9 + 16} = 5$; período
$\frac{2\pi}{2} = \pi$; fase
$\varphi = \arctan\frac43 \approx 0.927$:
$y = 5\cos(2t - 0.927)$.

\textbf{9.} $5\cos(2t - \varphi) = 0$ primero cuando
$2t - \varphi = \frac\pi2$:
$t = \frac{\pi/2 + 0.927}{2} \approx 1.25$ s.

\textbf{10.} Con $\sin\theta \approx \theta$:
$\theta'' = -\frac gL \theta$: armónico con
$\omega = \sqrt{\frac gL}$, punto
$T = \frac{2\pi}{\omega} = 2\pi\sqrt{\frac Lg}$. Para
$L = 1$: $T \approx 2.006$ s. El período no depende de
la amplitud (para pequeñas oscilaciones) --- isocronismo de Galileo,
el hecho que hace posibles los relojes de péndulo.

\textbf{11.} $L = g\left(\frac{T}{2\pi}\right)^2 =
\frac{9.81}{\pi^2} \approx 0.994$ m: el péndulo de segundos está
aproximadamente $6$ mm por debajo del metro meridiano. La Academia eligió
el meridiano (la gravedad varía de un lugar a otro, lo que delata
el péndulo); Si $g$ hubiera sido un poco más grande, nuestro medidor sería
garrapata.

\textbf{12.}$\cos\left(x + \frac\pi2\right) =
\cos x \cos\frac\pi2 - \sin x \sin\frac\pi2 = -\sin x$;
$\sin\left(x + \frac\pi2\right) = \cos x$. tan diferenciador
\omterm{def:g12:trigo:cossin}{seno} le da a \omterm{def:g12:trigo:cossin}{seno} un cuarto de vuelta ($\cos$), y nuevamente
($-\sin$), y otra vez, y a casa en cuatro pasos: diferenciación
gira el círculo.

\textbf{13.} Agregando las dos expansiones de $\cos(a \mp b)$:
$\cos(a-b) + \cos(a+b) = 2\cos a\cos b$. Con $a = b = x$:
$1 + \cos 2x = 2\cos^2 x$, es decir\
$\cos^2 x = \frac{1 + \cos 2x}{2}$.

\textbf{14.} Con $p = \frac{p+q}{2} + \frac{p-q}{2}$ y
$q$ su espejo, \omterm{def:g10:algebra:expand}{en expansión} y agregando:
