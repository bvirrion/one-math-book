\section*{Capítulo \ref{ch:g12:diffeq} --- Equações Diferenciais}

\begin{solution}{exo:g12:diffeq:1}
\emph{(um)} $y = C\eu^{3x}$; $ y(0) = 2$ dá $ y = 2\eu^{3x}$.

\emph{(b)} $ y' = -\frac12 y $, então $ y = C\eu^{-x/2}$; $ y(0) = -1$ dá
$ y = -\eu^{-x/2}$.

\emph{(c)} Equilíbrio $ y_p = 5$; $ y = C\eu^{-x} + 5$; $ y(0) = 0$ dá
$ C = -5$: $ y = 5\left(1 - \eu^{-x}\right)$.
\end{solution}

\begin{solution}{exo:g12:diffeq:2}
$ N' = kN $, então $ N(t) = N_0 \eu^{kt}$. Dobrando $3$ horas significa
$\eu^{3k} = 2$, \emph{ou seja}
\[
k = \frac{\ln 2}{3} \approx 0.231\ \text{h}^{-1}.
\]
\end{solution}

\begin{solution}{exo:g12:diffeq:3}
$ y_p'(x) = \eu^x + x\eu^x = y_p(x) + \eu^x $: $ y_p $ é uma solução particular
(este é o caso ressonante de \cref{met:g12:diffeq:particular}). Por
\cref{prop:g12:diffeq:general}, as soluções são
$ y = C\eu^{x} + x\eu^{x} = (C + x)\,\eu^x $, $ C \in \R $.
\end{solution}

\begin{solution}{exo:g12:diffeq:4}
Tentar $ y_p = \beta\eu^x $: $\beta\eu^x = -2\beta\eu^x + \eu^x $ dá
$3\beta = 1$, então $ y_p = \frac13\eu^x $. Solução geral
$ y = C\eu^{-2x} + \frac13\eu^{x}$; a condição $ y(0) = 1$ dá
$ C = \frac23$:
\[
y(x) = \frac{2}{3}\,\eu^{-2x} + \frac{1}{3}\,\eu^{x}.
\]
\end{solution}

\begin{solution}{exo:g12:diffeq:5}
$ N(t) = N_0\,\eu^{-\lambda t}$ com $\lambda = \frac{\ln 2}{5730}$
(\cref{ex:g12:exp:halflife}). Nós resolvemos $\eu^{-\lambda t} = 0.6$:
\[
t = \frac{\ln(1/0.6)}{\lambda} = 5730\,\frac{\ln(5/3)}{\ln 2}
\approx 5730 \times \frac{0.5108}{0.6931} \approx 4220 \text{ years}.
\]
\end{solution}

\begin{solution}{exo:g12:diffeq:6}
\emph{1.} O sal entra em $0.2 \times 5 = 1$ kg/min. A saída carrega
concentração $\frac{m}{100}$ kg/L em $5$ L/min, \emph{ou seja} $\frac{m}{20}$
kg/min. Por isso $ m' = 1 - \frac{m}{20}$.

\emph{2.} Equilíbrio $ m_p = 20$; $ m(t) = 20 + C\eu^{-t/20}$, e
$ m(0) = 0$ dá $ C = -20$:
\[
m(t) = 20\left(1 - \eu^{-t/20}\right) \xrightarrow[t\to+\infty]{} 20 \text{ kg}.
\]
No longo prazo, a concentração do tanque é igual à da salmoura que entra:
$0.2$ kg/L $\times $ $100$ eu $= 20$ kg.
\end{solution}

\begin{solution}{exo:g12:diffeq:7}
\emph{1.} $ v' = g - \frac{k}{m} v $ com $\frac km = \frac{16}{80} = 0.2$.
Equilíbrio $ v_\infty = \frac{mg}{k} = \frac{9.8}{0.2} = 49$ EM;
$ v(t) = 49 + C\eu^{-0.2t}$, e $ v(0) = 0$ dá
\[
v(t) = 49\left(1 - \eu^{-0.2 t}\right).
\]

\emph{2.} Velocidade terminal $49$ EM ($\approx 176$ km/h). Nós queremos
$1 - \eu^{-0.2t} = 0.95$, \emph{ou seja} $\eu^{-0.2t} = 0.05$:
\[
t = \frac{\ln 20}{0.2} \approx \frac{3.00}{0.2} \approx 15 \text{ s}.
\]
\end{solution}

\begin{solution}{exo:g12:diffeq:8}
\emph{1.} $ z = \frac1y $ dá $ z' = -\frac{y'}{y^2}
= -\frac{y(1-y)}{y^2} = -\frac{1-y}{y} = -\frac1y + 1 = -z + 1$.

\emph{2.} Equilíbrio $ z_p = 1$, então $ z(t) = 1 + C\eu^{-t}$. De
$ y(0) = \frac{1}{10}$, $ z(0) = 10$, então $ C = 9$ e
\[
y(t) = \frac{1}{1 + 9\,\eu^{-t}} .
\]

\emph{3.} Como $ t \to +\infty $, $9\eu^{-t} \to 0$ e $ y(t) \to 1$. A curva
é o clássico em forma de S (\emph{sigmóide}) curva logística: crescimento lento em
primeiro ($ y $ pequeno, $ y' \approx y $), crescimento mais rápido quando $ y = \frac12$
(onde $ y' = y(1-y)$ é \omterm{def:g10:functions:extrema}{máximo}), então a saturação em direção ao transporte
capacidade $1$.
\end{solution}

\begin{solution}{exo:g12:diffeq:9}
Por \cref{thm:g12:diffeq:affine}, $ y(x) = C\eu^{ax} - \frac ba $. Por isso
\[
u_{n+1} = C\eu^{a(n+1)} - \frac ba
= \eu^{a}\left(C\eu^{an} - \frac ba\right) + \frac ba\left(\eu^a - 1\right)
= q\,u_n + r
\]
com $ q = \eu^{a}$ e $ r = \frac{b}{a}\left(\eu^{a} - 1\right)$.
A condição $\abs q < 1$ significa $\eu^a < 1$, \emph{ou seja} $ a < 0$: exatamente
a condição sob a qual as soluções do equação diferencial
convergem para o equilíbrio $-\frac ba $ --- e de fato o \omterm{pb:g10:functions:1}{ponto fixo} de
a recorrência é $\frac{r}{1 - q} = -\frac{b}{a}$.
\end{solution}

\begin{solution}{pb:g12:diffeq:1}
\textbf{1.} $ y = 2\eu^{3t}$. Para o segundo: equilíbrio $3$,
então $ y = 3 + C\eu^{-2t}$ com $ y(0) = 0$: $ C = -3$:
$ y = 3\left(1 - \eu^{-2t}\right)$.

\textbf{2.} $\left(y\eu^{-at}\right)' = y'\eu^{-at} -
ay\eu^{-at} = (y' - ay)\eu^{-at} = 0$: o produto é um
constante $ C $, então $ y = C\eu^{at}$ --- nenhuma solução escapa.

\textbf{3.} Equilíbrio: $ y \equiv 3$. Qualquer outra solução é
$3 + C\eu^{-2t}$ com $ C \neq 0$: os dados exponenciais e o
solução desliza para $3$ --- um estábulo \omterm{pb:g10:functions:1}{ponto fixo}, o
contínuo gêmeo da recorrência pontos fixos de
\cref{pb:g11:seq:1}.

\textbf{4.} $ y = y_0\eu^{-t/\tau}$ metades quando
$\eu^{-t/\tau} = \frac12$: $ t = \tau\ln 2$.

\textbf{5.} Todas as soluções são translações verticais do decaimento
em direção $3$: aqueles que começam acima caem para $3$, aqueles abaixo
subir para $3$, e a solução constante $3$ fica quieto --- um
funil em torno do equilíbrio.

\textbf{6.} $\lambda = \frac{\ln 2}{5730} \approx 1.21 \times
10^{-4}$ por ano.

\textbf{7.} $\eu^{-\lambda t} = 0.2$:
$ t = \frac{\ln 5}{\lambda} \approx 13\,300$ anos.

\textbf{8.} $ t = \frac{\ln(1/0.15)}{\lambda} \approx 15\,700$
anos: os touros de Lascaux estão no final da Idade do Gelo.

\textbf{9.} $ t = \frac{\ln(1/0.53)}{\lambda} \approx 5\,250$
anos: dentro de uma vida inteira do valor dos arqueólogos ---
o relógio funciona.

\textbf{10.} Dez meias-vidas partem $2^{-10} \approx 0.1\,\%$:
na borda da medição. Depois $65$ milhões de anos o
fração é $2^{-65\,000\,000/5\,730} \approx 2^{-11\,344}$:
nenhum átomo do estoque original permanece em qualquer fóssil ---
os dinossauros são datados por relógios mais lentos (potássio - argônio e
parentes).

\textbf{11.} $52\,\%$ dá $\approx 5\,405$ anos,
$54\,\%$ dá $\approx 5\,095$: o $\pm 1\,\%$ de química
torna-se aproximadamente $\pm 155$ anos de história --- precisão em
o laboratório é preciso no rótulo do museu.

\textbf{12.} $ z = T - T_a $ satisfaz $ z' = -kz $:
$ z = (T_0 - T_a)\eu^{-kt}$, e $ T = T_a + z $.

\textbf{13.} A lacuna inicial é $ T_0 - T_a = 70$; depois das cinco
minutos a diferença é $70 - 20 = 50$, então $50 = 70\eu^{-5k}$:
$ k = \frac{\ln(70/50)}{5} \approx 0.067$ por minuto. Potável:
$55 - 20 = 35 = 70\eu^{-kt}$:
$ t = \frac{\ln 2}{k} \approx 10.3$ minutos.

\textbf{14.} A taxa de perda é proporcional à lacuna
$ T - T_a $: o café bem quente sangra o calor mais rápido. Adicionando o
leite \emph{agora} reduz a lacuna imediatamente, então menos calor é
perdeu ao longo dos dez minutos; adicioná-lo no final permite que o
o café esfrie primeiro na velocidade máxima. Para a xícara mais quente: leite
primeiro. (Anfitriões impacientes têm a termodinâmica invertida.)

\textbf{15.} Lacunas do ambiente: à meia-noite $10$, uma hora
mais tarde $8$: $\eu^{-k} = 0.8$:
$ k = \ln\frac{10}{8} \approx 0.223$ por hora. Na morte a lacuna
foi $17$; decaiu para $10$ à meia-noite:
$ s = \frac{\ln(17/10)}{k} \approx 2.4$ horas. Hora da morte:
sobre $21{:}40$.

\textbf{16.} Uma sala aquecida ou com corrente de ar ($ T_a $ não constante)
inclina o relógio para qualquer lado; mudanças de roupas ou massa corporal $ k $
(as tabelas do legista se ajustam a isso) --- um erro $ k $ escalas
o todo \omterm{def:g10:numbers:interval}{intervalo}; e um corpo movido de outro lugar é reiniciado
$ T_a $ no meio da decadência, fingindo uma morte mais cedo ou mais tarde. O
equação é honesto; as suposições carregam o risco.

\textbf{17.} $ z = \frac1y $ obedece $ z' = 1 - z $:
$ z = 1 + 9\eu^{-t}$ (de $ z(0) = 10$), então
$ y = \frac{1}{1 + 9\eu^{-t}}$. Inflexão em $ y = \frac12$:
$9\eu^{-t} = 1$: $ t = \ln 9 \approx 2.2$ --- a epidemia
data de virada, direto da fórmula.

\textbf{18.} Velocidade terminal: $ v' = 0$ no $ v = 20$ EM.
Solução: $ v(t) = 20\left(1 - \eu^{-t/2}\right)$. Então
$0.95$: $\eu^{-t/2} = 0.05$: $ t = 2\ln 20 \approx 6$ ---
as gotas de chuva atingem sua velocidade de cruzeiro em segundos, o que é
por que a chuva não mata.

\textbf{19.} Curso estável: $ c = 8$. Metade disso:
$4 = 8\left(1 - \eu^{-t/2}\right)$: $\eu^{-t/2} = \frac12$:
$ t = 2\ln 2 \approx 1.4$ horas. O gotejamento é o contínuo
gêmeo do empréstimo: entrada constante, saída proporcional ---
\cref{exo:g12:diffeq:9} torna o dicionário exato.

\textbf{20.} Decadência ($ a < 0$, $ b = 0$), resfriamento ($ y = T - T_a $),
caindo ($ a < 0$, $ b > 0$: abordagem ao equilíbrio a partir de
abaixo), dosagem (igual, na veia): um equação, quatro mundos.
O ciclo: declare a lei da mudança; escreva o equação; resolver
(\cref{met:g12:diffeq:solve}); calibrar constantes em
medições; prever; em seguida, audite as suposições (pergunta
16). Necessidades de oscilação $ y'' = -\omega^2 y$ --- encontrado, resolvido e
conjunto balançando \cref{pb:g12:trigo:1}.
\end{solution}
