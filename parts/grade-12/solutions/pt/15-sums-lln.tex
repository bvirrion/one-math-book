\section*{Capítulo \ref{ch:g12:sums} --- Somas de variáveis aleatórias e a lei dos grandes números}

\begin{solution}{exo:g12:sums:1}
Cada dado tem \omterm{def:g12:randvar:exp}{expectativa}
$\frac{1 + 2 + \dots + 6}{6} = \frac{7}{2}$, então por linearidade
$\E(S) = \frac72 + \frac72 = 7$. Os dados são \omterm{def:g12:sums:indep}{independente}, então
$\V(S) = \frac{35}{12} + \frac{35}{12} = \frac{35}{6} \approx 5.83$.
\end{solution}

\begin{solution}{exo:g12:sums:2}
$\E(X) = 50$ e $\V(X) = 100 \times 0.25 = 25$.

Markov ($ X \geq 0$): $\P(X \geq 75) \leq \frac{50}{75} = \frac23$.

Bienaymé-Chebyshev:
$\P(\abs{X - 50} \geq 25) \leq \frac{25}{25^2} = 0.04$ --- e isso mesmo
limita o \emph{bilateral} \omterm{def:g11:prob:model}{evento}, dos quais $ X \geq 75$ é metade. Tchebyshev
é muito mais nítido aqui porque explora o \omterm{def:g12:randvar:exp}{variação}, não apenas o significar.
(O verdadeiro valor de $\P(X \geq 75)$ é menor que $10^{-6}$: ambos os limites
são brutos.)
\end{solution}

\begin{solution}{exo:g12:sums:3}
$ F_n $ é o média amostral de $ n $ Bernoulli $\left(\frac12\right)$ variáveis,
de \omterm{def:g12:randvar:exp}{variação} $\sigma^2 = \frac14$. O limite
\[
\P\left(\abs{F_n - 0.5} \geq 0.05\right)
\leq \frac{1/4}{n \times 0.05^2} = \frac{100}{n}
\]
é $\leq 0.05$ assim que $ n \geq 2000$.
\end{solution}

\begin{solution}{exo:g12:sums:4}
Por \cref{prop:g12:sums:mean}, $\E(M_n) = 5\%$ (a diversificação não
alterar o retorno esperado) e
$\sigma(M_n) = \frac{20\%}{\sqrt n}$. Exigindo
$\frac{20}{\sqrt n} < 4$ dá $\sqrt n > 5$, \emph{ou seja} $ n \geq 26$.
Este é o princípio \emph{diversificação}: espalhando capital
\omterm{def:g12:sums:indep}{independente} riscos \omterm{def:g12:arith:divides}{divide} o risco (\omterm{def:g11:stat:variance}{desvio padrão}) por $\sqrt n $
sem reduzir o retorno esperado.
\end{solution}

\begin{solution}{exo:g12:sums:5}
\emph{1.} Contando o $9$ pares igualmente prováveis: $ S $ leva os valores
$2, 3, 4, 5, 6$ com \omterm{def:g10:proba:distribution}{probabilidades}
$\frac19, \frac29, \frac39, \frac29, \frac19$. Por isso
$\E(S) = \frac{2 + 6 + 12 + 10 + 6}{9} = 4$ e
$\E(S^2) = \frac{4 + 18 + 48 + 50 + 36}{9} = \frac{156}{9} = \frac{52}{3}$,
então $\V(S) = \frac{52}{3} - 16 = \frac43$.

\emph{2.} Uma variável: $\E(X) = 2$,
$\E(X^2) = \frac{1 + 4 + 9}{3} = \frac{14}{3}$,
$\V(X) = \frac{14}{3} - 4 = \frac23$. Então $\E(S) = 2 + 2 = 4$ e, por
independência, $\V(S) = \frac23 + \frac23 = \frac43$. Mesmos valores.
\end{solution}

\begin{solution}{exo:g12:sums:6}
Com $ Y = X $: $\V(X + Y) = \V(2X) = 4\V(X)$, enquanto
$\V(X) + \V(Y) = 2\V(X)$. Os dois concordam apenas quando $\V(X) = 0$, \emph{ou seja}
quando $ X $ é constante --- então a aditividade requer genuinamente independência
(aqui $ X $ é maximamente dependente de si mesmo).
\end{solution}

\begin{solution}{exo:g12:sums:7}
Sob justiça, $ F_n $ é o significar de $ n = 1200$ Variáveis de Bernoulli com
$ p = \frac16$, $\sigma^2 = \frac16\cdot\frac56 = \frac{5}{36}$:
\[
\P\left(\abs{F_n - \tfrac16} \geq 0.05\right)
\leq \frac{5/36}{1200 \times 0.0025} = \frac{5}{108} \approx 0.046 .
\]
O desvio observado é exatamente $0.05$: um \omterm{def:g11:prob:model}{evento} que um dado justo produz
com \omterm{def:g10:proba:distribution}{probabilidade} no \omterm{def:g10:functions:extrema}{máximo} $4.6\%$ --- e como Chebyshev é muito
conservador, o verdadeiro \omterm{def:g10:proba:distribution}{probabilidade} é muito menor. A hipótese da justiça
não é plausível; o dado provavelmente está carregado.
\end{solution}

\begin{solution}{exo:g12:sums:8}
\emph{1.} $ p(1-p) = \frac14 - \left(p - \frac12\right)^2 \leq \frac14$.

\emph{2.} $ F_n $ tem \omterm{def:g12:randvar:exp}{expectativa} $ p $ e \omterm{def:g12:randvar:exp}{variação}
$\frac{p(1-p)}{n} \leq \frac{1}{4n}$; Bienaymé--Chebyshev dá
\[
\P\bigl(\abs{F_n - p} \geq \varepsilon\bigr)
\leq \frac{p(1-p)}{n\varepsilon^2} \leq \frac{1}{4n\varepsilon^2},
\]
válido seja qual for o desconhecido $ p $.

\emph{3.} Com $\varepsilon = 0.03$ e nível $0.05$:
\[
\frac{1}{4n(0.03)^2} \leq 0.05
\iff n \geq \frac{1}{4 \times 0.0009 \times 0.05} \approx 5556 .
\]
Sobre $5600$ pessoas são suficientes por esse limite bruto (o clássico
\omterm{def:g10:numbers:approx}{aproximação} normal a resposta está mais próxima $1100$, ver
\cref{ch:g12:contdist}).
\end{solution}

\begin{solution}{pb:g12:sums:1}
\textbf{1.} $ V(X + Y) = V(X) + V(Y) = \frac{35}{12} +
\frac{35}{12} = \frac{35}{6}$ (independência).

\textbf{2.} $\E(\bar X) = 3.5$;
$\sigma(\bar X) = \frac{1.71}{\sqrt{100}} \approx 0.17$: o
abraços médios de cem dados $3.5$ dentro de um quinto de um ponto.

\textbf{3.} $\P(X \geq 10) \leq \frac{2}{10} = 0.2$.

\textbf{4.} $\P \leq \frac{V(\bar X)}{0.5^2} =
\frac{35/1200}{0.25} \approx 0.117$: no \omterm{def:g10:functions:extrema}{máximo} cerca de
$12\,\%$.

\textbf{5.} Com $ Y = -X $: $ V(X + Y) = V(0) = 0$, enquanto
$ V(X) + V(Y) = 2V(X) > 0$: variáveis perfeitamente anti-correlacionadas
cancelar --- \omterm{def:g12:randvar:exp}{variação} adição é um independência privilégio.

\textbf{6.} $\E = \frac{18}{37} - \frac{19}{37} =
-\frac{1}{37} \approx -0.027$;
$\sigma = \sqrt{1 - \left(\frac{1}{37}\right)^2} \approx
1.00$.

\textbf{7.} $\E(G) = -2.70$, $\sigma(G) = 10 \times 1.00 =
10$: a deriva é apenas $0.27\sigma $. O barulho da noite diminui
o limite --- uma grande minoria de jogadores (aproximadamente $40\,\%$,
diz a campainha) ande na frente, que é exatamente o que mantém
eles voltando.

\textbf{8.} A opinião do cassino: \omterm{def:g12:randvar:exp}{expectativa} $+27\,027$ euros,
\omterm{def:g11:stat:variance}{desvio padrão} $\approx 1\,000$: o lucro fica $27$
desvios padrão acima de zero. No $27\sigma $, ``o cassino
poderá perder este ano'' não é um risco, é um erro de arredondamento.

\textbf{9.} $\P(\text{loss}) = \P(G \geq 27\,027$ para o
jogadores $)$ $\leq \frac{10^6 \times 1}{27\,027^2} \approx
0.0014$: mesmo a desigualdade mais flagrante do livro garante
a casa em $99.86\,\%$ --- a verdade é astronomicamente
mais forte.

\textbf{10.} Cada euro apostado, quando e como for escolhido,
tem retorno esperado $-\frac{1}{37}$ por si só; por linearidade
o ganho total esperado é $-\frac{1}{37} \times
(\text{total staked})$, negativo para toda estratégia que
aposta qualquer coisa. Um sistema vencedor violaria a linearidade
de \omterm{def:g12:randvar:exp}{expectativa} --- nenhuma \omterm{def:g12:seq:sequence}{sequência} inteligente de apostas ruins faz
boa. (Os sistemas de duplicação apenas trocam muitos pequenos ganhos por
perdas catastróficas raras.)

\textbf{11.} O teorema: o \emph{freqüência} de vermelho por cima
$ n $ gira converge (em \omterm{def:g10:proba:distribution}{probabilidade}) para $\frac{18}{37}$. Isso
não diz nada sobre spin $ n + 1$: a roda não tem memória, e
depois de dez vermelhos a \omterm{def:g11:func:parity}{chance} de vermelho ainda é $\frac{18}{37}$.
Acreditar no contrário é o \emph{falácia do jogador}e
a questão 12 mostra o que realmente acontece com as \omterm{def:g12:seq:sequence}{sequências}.

\textbf{12.} O excedente de $100$ vermelhos não são reembolsados ---
as rodadas futuras são cópias justas, o excedente esperado permanece $100$.
Mas $\frac{100}{10^6} = 0.0001$: um centésimo de ponto.
O erro da falácia em uma frase: \emph{a lei dos grandes
os números diluem os acidentes passados num oceano de novas provações; isso
nunca envia a roda para cobrar dívidas.}

\textbf{13.} $ n \geq \frac{1}{4 \times 0.05 \times 0.03^2}
\approx 5\,556$ pessoas.

\textbf{14.} $\frac{1.96^2 \times 0.25}{0.03^2} \approx
1\,067$: cinco vezes menos. Chebyshev defende \emph{todo}
\omterm{def:g12:randvar:rv}{distribuição} e paga pela sua universalidade com folga; o
constante dos sondadores $1.96$ vem do formato real do sino
das flutuações, que se concentra muito mais.

\textbf{15.} Margem $\propto \frac{1}{\sqrt n}$: dividindo pela metade
quadruplica o \omterm{def:g10:proba:sample}{amostra} --- sobre $4\,400$ pessoas para
$\pm 1.5$ pontos. A precisão é comprada a preços quadráticos.

\textbf{16.} O modelo é $ n $ \omterm{def:g12:sums:indep}{independente} desenha com sucesso
\omterm{def:g10:proba:distribution}{probabilidade} $ p $ --- o tamanho da população $ N $ nunca aparece
(amostrar uma pequena fração de uma população grande e bem misturada
é o que independência codifica). A analogia dos pesquisadores: para
experimente a sopa, uma colher bem mexida é suficiente ---
se o pote serve dez ou dez mil. O problema é o
agitado --- uma base de amostragem tendenciosa, como no ensino médio
a \omterm{def:g10:algebra:equation}{pesquisa} do volume que enganou um país - não a maconha.

\textbf{17.} $315\sqrt n < 30n \iff \sqrt n > 10.5 \iff
n > 110$: além de uma centena de políticas, o desvio ultrapassa o
ruído, e cada nova política aumenta a lacuna em
$\sqrt n $ lei --- pooling \emph{é} o modelo de negócios.

\textbf{18.} \omterm{def:g12:sums:indep}{Independente} partes iguais: $\sigma_{\text{pf}} =
\frac{20\,\%}{\sqrt{25}} = 4\,\%$: mesmo retorno esperado,
um quinto da flutuação. O almoço: redução de risco a zero
custo em \omterm{def:g12:randvar:exp}{expectativa} --- média \omterm{def:g12:sums:indep}{independente} aleatoriedade.

\textbf{19.} Perfeitamente correlacionado: o portfólio é uma ação
em $25$ trajes: $\sigma = 20\,\%$, nenhuma redução.
Rendas de diversificação \emph{independência}; quando uma crise
correlaciona tudo (2008: todas as apostas habitacionais foram uma aposta),
o $\sqrt n $ proteção evapora exatamente quando é
necessário.

\textbf{20.} As somas flutuam como $ n $ e flutuar como
$\sqrt n $, então significa resolver como $\frac{1}{\sqrt n}$: o
o cassino deposita mais de um milhão de rodadas; o pesquisador
compra $\frac{1}{\sqrt{1100}}$ de ruído para um orçamento telefônico;
a seguradora supera suas próprias reivindicações em $ n > 110$; o
investidor \omterm{def:g12:arith:divides}{divide} risco por $\sqrt{25}$. Abusadores: o jogador
que acredita em dívidas (só há diluição), e o
financista que acredita em independência (às vezes há
apenas uma aposta). A forma universal das flutuações --- o
sino --- é o contínuo estrela do capítulo e seu teorema,
o teorema do limite central, é o ápice da universidade
volumes' \omterm{def:g10:proba:distribution}{probabilidade}.
\end{solution}
