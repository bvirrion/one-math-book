\begin{solution}{exo:g12:complex:1}
$(3-2\iu)(1+4\iu) = 3 + 12\iu - 2\iu - 8\iu^2 = 11 + 10\iu $.

$\dfrac{2+\iu}{1-\iu} = \dfrac{(2+\iu)(1+\iu)}{(1-\iu)(1+\iu)}
= \dfrac{1 + 3\iu}{2} = \dfrac12 + \dfrac32\iu $.

$\iu^{2027}$: desde $\iu^4 = 1$ e $2027 = 4 \times 506 + 3$,
$\iu^{2027} = \iu^3 = -\iu $.
\end{solution}

\begin{solution}{exo:g12:complex:2}
\emph{(um)} $\Delta = 16 - 52 = -36$:
$ z = \dfrac{4 \pm 6\iu}{2} = 2 \pm 3\iu $. \omterm{def:g12:complex:conjugate}{Conjugado} par; produto
$= 4 + 9 = 13 = \frac ca $. \emph{(b)} $\Delta = 1 - 4 = -3$:
$ z = \dfrac{-1 \pm \iu\sqrt3}{2}$ (as raízes cúbicas da unidade $ j $ e
$\conj j $); produto $= \frac{1 + 3}{4} = 1$.
\end{solution}

\begin{solution}{exo:g12:complex:3}
$\abs{z_1} = \sqrt2$, \omterm{def:g12:complex:argument}{argumento} $\frac{3\pi}{4}$ (segundo quadrante):
$ z_1 = \sqrt2\,\eu^{3\iu\pi/4}$.
$\abs{z_2} = 2$, \omterm{def:g12:complex:argument}{argumento} $-\frac\pi6$:
$ z_2 = 2\,\eu^{-\iu\pi/6}$. Por isso
\[
\frac{z_1}{z_2} = \frac{\sqrt2}{2}\,\eu^{\iu\left(\frac{3\pi}{4} + \frac{\pi}{6}\right)}
= \frac{\sqrt2}{2}\,\eu^{11\iu\pi/12}.
\]
Algebricamente,
\[
\frac{z_1}{z_2} = \frac{-1+\iu}{\sqrt3 - \iu}
= \frac{(-1+\iu)(\sqrt3+\iu)}{4}
= \frac{-\sqrt3 - 1 + \iu(\sqrt3 - 1)}{4}.
\]
Identificando peças reais:
$\frac{\sqrt2}{2}\cos\frac{11\pi}{12} = \frac{-\sqrt3-1}{4}$, então
\[
\cos\frac{11\pi}{12} = -\frac{\sqrt6 + \sqrt2}{4}.
\]
\end{solution}

\begin{solution}{exo:g12:complex:4}
\emph{(um)} $\abs{z - 2} = \abs{z - (-\iu)}$ significa $ M $ é equidistante de
$ A(2, 0)$ e $ B(0, -1)$: o \emph{bissetriz perpendicular} de $[AB]$.

\emph{(b)} Distância $3$ do ponto $(1,1)$: o \emph{círculo} do centro
$1 + \iu $ e raio $3$.

\emph{(c)} Por \cref{prop:g12:complex:geometry}(4), $\frac{z-1}{z+1}$ puramente
imaginário significa o \omterm{def:g11:vect:vector}{vetores} de $ A(1,0)$ para $ M $ e de $ B(-1,0)$ para $ M $
são \omterm{def:g11:scal:orthogonal}{ortogonal}: $ M $ vê o segmento $[AB]$ sob um ângulo reto. O conjunto é
o \emph{círculo de diâmetro $[AB]$} (o círculo unitário) \emph{menos o
pontos $ A $ e $ B $} eles mesmos.
\end{solution}

\begin{solution}{exo:g12:complex:5}
De Moivre: $\cos3\theta + \iu\sin3\theta = (\cos\theta + \iu\sin\theta)^3$.
\omterm{def:g10:algebra:expand}{Expandindo} pelo teorema binomial com $ c = \cos\theta $, $ s = \sin\theta $:
\[
(c + \iu s)^3 = c^3 + 3\iu c^2 s - 3 c s^2 - \iu s^3
= (c^3 - 3cs^2) + \iu(3c^2 s - s^3).
\]
Identificando peças e usando $ s^2 = 1 - c^2$, $ c^2 = 1 - s^2$:
\[
\cos3\theta = 4\cos^3\theta - 3\cos\theta,
\qquad
\sin3\theta = 3\sin\theta - 4\sin^3\theta .
\]
\end{solution}

\begin{solution}{exo:g12:complex:6}
\[
\cos^3\theta = \left(\frac{\eu^{\iu\theta} + \eu^{-\iu\theta}}{2}\right)^{\!3}
= \frac{\eu^{3\iu\theta} + 3\eu^{\iu\theta} + 3\eu^{-\iu\theta} + \eu^{-3\iu\theta}}{8}
= \frac{\cos3\theta + 3\cos\theta}{4}.
\]
Por isso
\[
\int_0^{\pi/2}\cos^3\theta\,\dd\theta
= \frac14\left[\frac{\sin3\theta}{3} + 3\sin\theta\right]_0^{\pi/2}
= \frac14\left(-\frac13 + 3\right) = \frac{2}{3}.
\]
(Consistente com $ W_3 = \frac23$ em \cref{exo:g12:integ:9}, desde
$\int_0^{\pi/2}\cos^3 = \int_0^{\pi/2}\sin^3$ pela simetria
$\theta \mapsto \frac\pi2 - \theta $.)
\end{solution}

\begin{solution}{exo:g12:complex:7}
$ c = a + \eu^{\pm\iu\pi/3}(b - a)$ com $ b - a = 2 + \iu $ e
$\eu^{\pm\iu\pi/3} = \frac12 \pm \iu\frac{\sqrt3}{2}$:
\[
\eu^{\iu\pi/3}(2+\iu) = \left(\tfrac12 + \iu\tfrac{\sqrt3}{2}\right)(2+\iu)
= \left(1 - \tfrac{\sqrt3}{2}\right) + \iu\left(\tfrac12 + \sqrt3\right),
\]
então $ c_1 = \left(2 - \frac{\sqrt3}{2}\right) +
\iu\left(\frac32 + \sqrt3\right)$, e da mesma forma com o ângulo
$-\frac\pi3$:
$ c_2 = \left(2 + \frac{\sqrt3}{2}\right) + \iu\left(\frac32 - \sqrt3\right)$.
\end{solution}

\begin{solution}{exo:g12:complex:8}
$-4 = 4\,\eu^{\iu\pi}$, então as soluções de $ z^4 = -4$ são
\[
z_k = \sqrt2\,\eu^{\iu\left(\frac\pi4 + \frac{k\pi}{2}\right)},
\quad k = 0, 1, 2, 3,
\]
\emph{ou seja} $1 + \iu $, $-1 + \iu $, $-1 - \iu $, $1 - \iu $: os vértices de um
quadrado do circunraio $\sqrt 2$. Agrupamento \omterm{def:g12:complex:conjugate}{conjugado} raízes,
\[
z^4 + 4 = \bigl(z - (1{+}\iu)\bigr)\bigl(z - (1{-}\iu)\bigr)
\bigl(z + (1{-}\iu)\bigr)\bigl(z + (1{+}\iu)\bigr)
= (z^2 - 2z + 2)(z^2 + 2z + 2).
\]
\end{solution}

\begin{solution}{exo:g12:complex:9}
Com $ q = \eu^{\iu\theta} \neq 1$:
\[
\sum_{k=0}^{n} \eu^{\iu k\theta} = \frac{\eu^{\iu(n+1)\theta} - 1}{\eu^{\iu\theta} - 1}
= \frac{\eu^{\iu(n+1)\theta/2}}{\eu^{\iu\theta/2}}
\cdot\frac{\eu^{\iu(n+1)\theta/2} - \eu^{-\iu(n+1)\theta/2}}{\eu^{\iu\theta/2} - \eu^{-\iu\theta/2}}
= \eu^{\iu n\theta/2}\,
\frac{\sin\frac{(n+1)\theta}{2}}{\sin\frac{\theta}{2}},
\]
usando $\eu^{\iu\alpha} - \eu^{-\iu\alpha} = 2\iu\sin\alpha $ (o
\emph{meio ângulo \omterm{def:g10:algebra:expand}{factoring}}). Tirando peças reais:
\[
S = \frac{\sin\frac{(n+1)\theta}{2}}{\sin\frac{\theta}{2}}\,
\cos\frac{n\theta}{2}.
\]
\end{solution}

\begin{solution}{exo:g12:complex:10}
\emph{1.} $\omega $ é um $5$-º \omterm{def:g11:quad:discriminant}{raiz} de unidade diferente de $1$, então a soma
de todos os cinco raízes desaparece (\cref{thm:g12:complex:rootsofunity}), onde se lê
$1 + \omega + \omega^2 + \omega^3 + \omega^4 = 0$.

\emph{2.} $ u + v = \omega + \omega^2 + \omega^3 + \omega^4 = -1$ pelo ponto 1.
Para o produto, usando $\omega^5 = 1$:
\[
uv = (\omega + \omega^4)(\omega^2 + \omega^3)
= \omega^3 + \omega^4 + \omega^6 + \omega^7
= \omega^3 + \omega^4 + \omega + \omega^2 = -1 .
\]
Então $ u, v $ são os raízes de $ X^2 + X - 1 = 0$:
$\left\{\frac{-1+\sqrt5}{2}, \frac{-1-\sqrt5}{2}\right\}$. Agora
$ u = \omega + \conj\omega = 2\cos\frac{2\pi}{5} > 0$ (desde
$\frac{2\pi}{5} < \frac{\pi}{2}$), então $ u = \frac{-1 + \sqrt5}{2}$.

\emph{3.} $\cos\frac{2\pi}{5} = \frac u2 = \frac{\sqrt5 - 1}{4}$.
\end{solution}

\begin{solution}{pb:g12:complex:1}
\textbf{1.} $(2 + \iu)(3 - \iu) = 6 - 2\iu + 3\iu + 1 =
7 + \iu $. \quad $\frac{1 + \iu}{1 - \iu} =
\frac{(1 + \iu)^2}{2} = \frac{2\iu}{2} = \iu $. \quad
$\abs{3 + 4\iu} = 5$.

\textbf{2.} $1 + \iu = \sqrt2\,\eu^{\iu\pi/4}$;
$-2 = 2\eu^{\iu\pi}$;
$1 - \iu\sqrt3 = 2\eu^{-\iu\pi/3}$.

\textbf{3.} $(1 + \iu)^8 = \left(\sqrt2\right)^8
\eu^{8\iu\pi/4} = 16\,\eu^{2\iu\pi} = 16$.

\textbf{4.} $ z^2 = \eu^{\iu\pi/2}$:
$ z = \pm\eu^{\iu\pi/4} = \pm\frac{\sqrt2}{2}(1 + \iu)$. E
$\Delta = 4 - 20 = -16$: $ z = 1 \pm 2\iu $.

\textbf{5.} $ z \mapsto \iu z $: rotação de um quarto de volta
sobre a origem. $\abs{z - (1 + \iu)} = 2$: o círculo de
centro $1 + \iu $ e raio $2$.

\textbf{6.} $\left(\omega^k\right)^5 = \eu^{2\iu\pi k} = 1$,
e os cinco pontos $\eu^{2\iu k\pi/5}$ são distintos: todos os cinco
soluções de $ z^5 = 1$. Eles se sentam no círculo unitário em igualdade
ângulos de $72^\circ $: um pentágono regular com um vértice em
$1$.

\textbf{7.} $ S = 1 + \omega + \dots + \omega^4$ satisfaz
$(1 - \omega)S = 1 - \omega^5 = 0$, e $\omega \neq 1$:
$ S = 0$. (O ponto de equilíbrio do pentágono é o seu centro.)

\textbf{8.} $\omega^4 = \overline\omega $ e
$\omega^3 = \overline{\omega^2}$: emparelhamento \omterm{def:g12:complex:conjugate}{conjugados}, o verdadeiro
parte da soma lê
$1 + 2\cos\frac{2\pi}{5} + 2\cos\frac{4\pi}{5} = 0$.

\textbf{9.} $\cos\frac{4\pi}{5} = 2c^2 - 1$, então
$1 + 2c + 4c^2 - 2 = 0$, ou seja\ $4c^2 + 2c - 1 = 0$:
$ c = \frac{-1 + \sqrt5}{4}$ (o positivo \omterm{def:g11:quad:discriminant}{raiz}, desde
$72^\circ $ é agudo):
$\cos 72^\circ = \frac{\sqrt5 - 1}{4}$.

\textbf{10.} Numericamente $ c \approx 0.30902$: corresponde ao
calculadora. E $\frac{1}{2\varphi} = \frac{1}{1 + \sqrt5} =
\frac{\sqrt5 - 1}{4}$ (racionalizar): $ c = \frac{1}{2\varphi}$.
O pentágono é totalmente dourado: seus cortes diagonais
seu lado na proporção $\varphi $ --- o pentagrama é infinito
auto-similaridade.

\textbf{11.} As raízes quadradas são construtíveis com um compasso (o
máquina de semicírculo da geometria do volume do Ensino Médio
significar), então um número construído a partir de racionais apenas por raízes quadradas
--- gosto $\cos 72^\circ $ --- é construtível; raízes cúbicas são
não, e $\cos 20^\circ $ precisa de um: o pentágono cede,
a trissecção resiste. O critério de Gauss resolve todos os regulares
polígono --- seus 17 anos aos dezenove o fizeram escolher
matemática.

\textbf{12.} $-4 = 4\eu^{\iu\pi}$: o quarto raízes ter
\omterm{def:g12:complex:modulus}{módulo} $\sqrt2$ e \omterm{def:g12:complex:argument}{argumentos}
$\frac{\pi}{4} + k\frac{\pi}{2}$: os quatro números
$\pm 1 \pm \iu $. Emparelhamento \omterm{def:g12:complex:conjugate}{conjugados}:
$(z - 1 - \iu)(z - 1 + \iu) = z^2 - 2z + 2$ e
$(z + 1 - \iu)(z + 1 + \iu) = z^2 + 2z + 2$: o
fatoração de \cref{exo:g12:complex:8}.

\textbf{13.} $(\cos\theta + \iu\sin\theta)^3 = \cos^3\theta +
3\iu\cos^2\theta\sin\theta - 3\cos\theta\sin^2\theta -
\iu\sin^3\theta $; por de Moivre isso é igual
$\cos 3\theta + \iu \sin 3\theta $. Peças reais:
$\cos 3\theta = \cos^3\theta - 3\cos\theta(1 - \cos^2\theta)
= 4\cos^3\theta - 3\cos\theta $ --- e o partes imaginárias
entregar $\sin 3\theta $ de graça.

\textbf{14.} É o \omterm{def:g12:complex:def}{parte real} de
$1 + \omega + \dots + \omega^4 = 0$: a soma é igual $0$ ---
exatamente a questão 8, vista através
\cref{exo:g12:complex:9}núcleo.

\textbf{15.} $8\iu = 8\eu^{\iu\pi/2}$: raízes
$2\eu^{\iu\pi/6} = \sqrt3 + \iu $;
$2\eu^{5\iu\pi/6} = -\sqrt3 + \iu $;
$2\eu^{3\iu\pi/2} = -2\iu $.

\textbf{16.} $1 + \iu = \sqrt2\,\eu^{\iu\pi/4}$: o mapa
gira o avião em $45^\circ $ e dimensiona-o por $\sqrt2$. O
o quadrado unitário torna-se um quadrado inclinado de lado $\sqrt2$ ---
dobrou em área, girou um oitavo de volta.

\textbf{17.} $\abs{z_1 z_2}^2 = z_1 z_2 \overline{z_1 z_2} =
z_1\overline{z_1}\, z_2\overline{z_2} =
\abs{z_1}^2\abs{z_2}^2$: tire raízes quadradas. Poderes de
$1 + \iu $: $1 + \iu $, $2\iu $, $-2 + 2\iu $, $-4$: cada um
$45^\circ $ vire mais e $\sqrt2$ vezes mais --- um
espiral externa através de \omterm{def:g12:complex:modulus}{módulos}
$\sqrt2, 2, 2\sqrt2, 4, \dots $

\textbf{18.} $\mathrm{j}$ é um terço \omterm{def:g11:quad:discriminant}{raiz} da unidade
distinto de $1$, então (pergunta 7 \omterm{def:g12:complex:argument}{argumento} com $ n = 3$)
$1 + \mathrm{j} + \mathrm{j}^2 = 0$. Critério no padrão
triângulo: $ a + \mathrm{j}b + \mathrm{j}^2 c = 1 +
\mathrm{j}\cdot\mathrm{j} + \mathrm{j}^2 \cdot \mathrm{j}^2 =
1 + \mathrm{j}^2 + \mathrm{j}^4 = 1 + \mathrm{j}^2 +
\mathrm{j} = 0$: satisfeito, como o triângulo equilátero
$\left(1, \mathrm{j}, \mathrm{j}^2\right)$ demandas.

\textbf{19.} Sobre $\C $, um polinômio real se divide em linear
fatores; não é real raízes entre \omterm{def:g12:complex:conjugate}{conjugado} pares (\omterm{def:g12:complex:conjugate}{conjugado}
o equação), e cada par se multiplica em um quadrático real
$ z^2 - 2\operatorname{Re}(z_0)\,z + \abs{z_0}^2$: daí o
fatoração real em graus $1$ e grau-$2$ peças ---
questão 12 executou em $ z^4 + 4$.

\textbf{20.} Pontos; rotação e escala; polígonos regulares;
identidades trigonométricas por expansão; e um mundo fechado
onde cada equação de grau $ n $ tem seu $ n $ raízes. Código:
na questão 10, $\pi$ O pentágono apertou a mão do dourado
proporção --- Os dois convidados mais famosos de Euclides, finalmente apresentados
por uma soma de cinco setas igual a zero.
\end{solution}
