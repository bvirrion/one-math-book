\section*{Capítulo \ref{ch:g12:deriv} --- Diferenciação e Convexidade}

\begin{solution}{exo:g12:deriv:1}
$f'(x) = 3x^2 - 5$.

Regra do quociente:
$ g'(x) = \dfrac{(x^2+1) - x\cdot 2x}{(x^2+1)^2} = \dfrac{1 - x^2}{(x^2+1)^2}$.

Regra da cadeia com $ u = 2x+1$: $ h'(x) = 5 \cdot 2 \cdot (2x+1)^4 = 10(2x+1)^4$.

Regra da cadeia com $ u = x^2 + 1$:
$ k'(x) = \dfrac{2x}{2\sqrt{x^2+1}} = \dfrac{x}{\sqrt{x^2+1}}$.
\end{solution}

\begin{solution}{exo:g12:deriv:2}
$ f'(x) = 2x - 3$, então $ f(2) = -1$ e $ f'(2) = 1$: o \omterm{def:g12:deriv:tangent}{tangente} é
$ y = -1 + (x - 2) = x - 3$. A lacuna é
\[
f(x) - (x - 3) = x^2 - 4x + 4 = (x-2)^2 \geq 0 ,
\]
então a curva fica acima do \omterm{def:g12:deriv:tangent}{tangente} em todos os lugares, tocando-o apenas em $ x = 2$
(como esperado: $ f $ é \omterm{def:g12:deriv:convex}{convexo}).
\end{solution}

\begin{solution}{exo:g12:deriv:3}
Ambos são quocientes de diferença. Com $ f(x) = (1+x)^{100}$ no $0$:
$ f'(x) = 100(1+x)^{99}$, então o limite é $ f'(0) = 100$.
Com $ g(x) = \sqrt{x}$ no $4$: $ g'(x) = \frac{1}{2\sqrt{x}}$, então o limite é
$ g'(4) = \frac14$.
\end{solution}

\begin{solution}{exo:g12:deriv:4}
Domínio $\R \setminus \{1\}$. Limites: em $\pm\infty $, $ f(x) \sim x \to
\pm\infty $; no $1^\pm $, numerador $\to 4 > 0$ e denominador $\to 0^\pm $,
então $ f \to \pm\infty $; a linha $ x = 1$ é um \omterm{def:g12:limcont:asymptote}{assíntota vertical}.

\[
f'(x) = \frac{2x(x-1) - (x^2+3)}{(x-1)^2} = \frac{x^2 - 2x - 3}{(x-1)^2}
= \frac{(x-3)(x+1)}{(x-1)^2}.
\]
Por isso $ f' > 0$ sobre $\intoo{-\infty}{-1}$ e $\intoo{3}{+\infty}$, $ f' < 0$ sobre
$\intoo{-1}{1}$ e $\intoo{1}{3}$. O \omterm{def:g11:func:function}{função} aumenta para um \omterm{def:g12:deriv:extremum}{máximo local}
$ f(-1) = -2$, diminui para $-\infty $, salta para $+\infty $ depois do assíntota,
diminui para um local \omterm{def:g10:functions:extrema}{mínimo} $ f(3) = 6$, então aumenta.
\end{solution}

\begin{solution}{exo:g12:deriv:5}
Para $ x \in \intoo{0}{15}$, a caixa tem uma base quadrada de lado $30 - 2x $ e
altura $ x $, então
\[
V(x) = x(30 - 2x)^2 .
\]
$ V'(x) = (30-2x)^2 + x \cdot 2(30-2x)(-2) = (30-2x)(30 - 2x - 4x)
= (30-2x)(30-6x)$. Sobre $\intoo{0}{15}$, $30 - 2x > 0$, então $ V'$ tem o sinal de
$30 - 6x $: positivo antes $ x = 5$, negativo depois. O volume é \omterm{def:g10:functions:extrema}{máximo} para
$ x = 5$ cm, dando $ V(5) = 5 \times 20^2 = 2000\ \text{cm}^3$.
\end{solution}

\begin{solution}{exo:g12:deriv:6}
\emph{1.} $ f'(x) = 4x^3 - 12x^2$ e $ f''(x) = 12x^2 - 24x = 12x(x-2)$.
Por isso $ f'' > 0$ sobre $\intoo{-\infty}{0}$ e assim por diante $\intoo{2}{+\infty}$ (\omterm{def:g12:deriv:convex}{convexo}),
$ f'' < 0$ sobre $\intoo{0}{2}$ (\omterm{def:g12:deriv:convex}{côncavo}). $ f''$ muda o sinal em $0$ e em $2$:
ambos são pontos de inflexão.

\emph{2.} No $ a = 2$: $ f(2) = 16 - 32 + 10 = -6$, $ f'(2) = 32 - 48 = -16$,
\omterm{def:g12:deriv:tangent}{tangente} $ y = -6 - 16(x-2) = -16x + 26$. A lacuna é
\[
g(x) = f(x) + 16x - 26 = x^4 - 4x^3 + 16x - 16 .
\]
Desde $ g(2) = g'(2) = g''(2) = 0$, $(x-2)^3$ divide $ g $; divisão dá
$ g(x) = (x-2)^3(x+2)$. Aproximar $ x = 2$, $ x + 2 > 0$, então $ g $ tem o sinal de
$(x-2)^3$: negativo antes $2$, positivo depois. A curva cruza o
\omterm{def:g12:deriv:tangent}{tangente}: $2$ é de fato um ponto de inflexão.
\end{solution}

\begin{solution}{exo:g12:deriv:7}
O \omterm{def:g11:func:function}{função} $ f(x) = \sqrt{x}$ satisfaz
$ f''(x) = -\frac{1}{4}x^{-3/2} < 0$ sobre $\intoo{0}{+\infty}$: isso é \omterm{def:g12:deriv:convex}{côncavo},
então é \omterm{def:g10:functions:graph}{gráfico} mentiras \emph{abaixo} cada \omterm{def:g12:deriv:tangent}{tangente}. O \omterm{def:g12:deriv:tangent}{tangente} no $ a = 1$ é
\[
y = f(1) + f'(1)(x - 1) = 1 + \tfrac12 (x-1) = \frac{x+1}{2},
\]
de onde $\sqrt{x} \leq \frac{x+1}{2}$ para todos $ x > 0$. Igualdade significa
curva toca o \omterm{def:g12:deriv:tangent}{tangente}, o que acontece apenas no ponto de tangência
$ x = 1$. (Equivalentemente: $\left(\sqrt x - 1\right)^2 \geq 0$.)
\end{solution}

\begin{solution}{exo:g12:deriv:8}
Por \cref{thm:g12:deriv:convexity}, o \omterm{def:g10:functions:graph}{gráfico} de um \omterm{def:g12:deriv:derivative}{diferenciável} \omterm{def:g12:deriv:convex}{convexo}
\omterm{def:g11:func:function}{função} está acima de cada um de seus \omterm{def:g12:deriv:tangent}{tangentes}. O \omterm{def:g12:deriv:tangent}{tangente} no $ a $ é horizontal
($ f'(a) = 0$), de equação $ y = f(a)$. Por isso $ f(x) \geq f(a)$ para todos
$ x \in \R $: $ f(a)$ é o mundial \omterm{def:g10:functions:extrema}{mínimo}.
\end{solution}

\begin{solution}{exo:g12:deriv:9}
\emph{Perímetro fixo.} Um retângulo com lados $ x $ e $\frac{p}{2} - x $
($0 < x < \frac p2$) tem área $ S(x) = x\left(\frac p2 - x\right)$. Então
$ S'(x) = \frac p2 - 2x $ desaparece em $ x = \frac p4$, positivo antes e
negativo depois: o \omterm{def:g10:functions:extrema}{máximo} está em $ x = \frac{p}{4}$, onde ambos os lados são iguais
$\frac p4$ --- um quadrado.

\emph{Área fixa.} Lados $ x $ e $\frac Ax $ dar perímetro
$ P(x) = 2\left(x + \frac Ax\right)$, com
$ P'(x) = 2\left(1 - \frac{A}{x^2}\right)$, negativo para $ x < \sqrt A $ e
positivo depois: \omterm{def:g10:functions:extrema}{mínimo} no $ x = \sqrt{A}$, um quadrado novamente.

\emph{Relação.} As duas afirmações são duais. Suponha um retângulo não quadrado
minimizou o perímetro na área fixa $ A $; o quadrado do mesmo perímetro
teria área $> A $ pela primeira afirmação, e encolhendo-a homoteticamente
para a área $ A $ diminuiria estritamente seu perímetro --- contradizendo
minimalidade. Cada afirmação implica, portanto, a outra.
\end{solution}

\begin{solution}{pb:g12:deriv:1}
\textbf{1.} $30x\left(3x^2 + 1\right)^4$; \quad
$\dfrac{x}{\sqrt{x^2 + 9}}$; \quad
$-\dfrac{2x}{\left(x^2 + 1\right)^2}$.

\textbf{2.} $ y(4) = 5$, declive $\frac45$: \omterm{def:g12:deriv:tangent}{tangente}
$ y = 5 + \frac45(x - 4)$, ou seja\ $ y = \frac45 x + \frac95$.

\textbf{3.} $ f'(x) = \frac{(x^2 + 1) - x \cdot 2x}{(x^2+1)^2}
= \frac{1 - x^2}{(x^2 + 1)^2}$: negativo, positivo, negativo
através $-1$ e $1$: \omterm{def:g10:functions:extrema}{mínimo} $ f(-1) = -\frac12$, \omterm{def:g10:functions:extrema}{máximo}
$ f(1) = \frac12$, com \omterm{def:g12:limcont:asymptote}{assíntota horizontal} $0$ em ambos os lados.

\textbf{4.} $ g'(x) = 3x^2 - 6x = 3x(x - 2)$: \omterm{def:g12:seq:monotonic}{aumentando},
diminuindo, \omterm{def:g12:seq:monotonic}{aumentando} entre $0$ e $2$; \omterm{def:g12:deriv:extremum}{máximo local}
$ g(0) = 4$, \omterm{def:g10:functions:extrema}{mínimo} local $ g(2) = 0$. $ g''(x) = 6x - 6$: \omterm{def:g12:deriv:convex}{côncavo}
antes $1$, \omterm{def:g12:deriv:convex}{convexo} depois: inflexão em $(1, 2)$.

\textbf{5.} \omterm{def:g12:deriv:tangent}{Tangente} no $1$: declive $ g'(1) = -3$:
$ y = 2 - 3(x - 1) = -3x + 5$. Diferença:
$ x^3 - 3x^2 + 4 - (-3x + 5) = x^3 - 3x^2 + 3x - 1 =
(x - 1)^3$, que muda de sinal em $1$: o \omterm{def:g12:deriv:tangent}{tangente}
\emph{cruzes} a curva --- o comportamento da assinatura em um
ponto de inflexão, onde a curva muda de lado.

\textbf{6.} $ T(x) = \dfrac{\sqrt{x^2 + 900}}{5} +
\dfrac{\sqrt{(40 - x)^2 + 400}}{2}$. Entrando antes $0$ ou
além $40$ alonga \emph{ambos} pernas: inútil.

\textbf{7.} $ T'(x) = \dfrac{x}{5\sqrt{x^2 + 900}} -
\dfrac{40 - x}{2\sqrt{(40 - x)^2 + 400}}$.

\textbf{8.} No triângulo em movimento, o lado ao longo da costa
é $ x $ e a hipotenusa $\sqrt{x^2 + 900}$: o \omterm{def:g11:trigo:cossin}{seno} do
ângulo com a perpendicular é precisamente o seu quociente
(oposto à hipotenusa); da mesma maneira
$\sin\theta_2 = \frac{40 - x}{\sqrt{(40-x)^2 + 400}}$. Então
$ T'(x) = 0$ lê
$\frac{\sin\theta_1}{5} = \frac{\sin\theta_2}{2}$ ---Snell's
lei com velocidades $5$ e $2$.

\textbf{9.} $ T(24) \approx 20.5$ é; $ T(40) = 10 + 10 = 20$ é;
$ T(0) = 6 + \sqrt{2000}/2 \approx 28.4$ S. A linha reta
perde até para ``correr até o fim e depois nadar em linha reta'' --- e
ambos perdem para o caminho refratado.

\textbf{10.} \omterm{thm:g12:limcont:ivt}{Dicotomia} sobre $ T'$ (negativo em $24$, positivo em
$40$) converge para $ x^* \approx 34$ m, com
$ T(x^*) \approx 19.5$ s: cerca de um segundo mais rápido que o
linha reta --- a diferença entre um resgate e um
tragédia.

\textbf{11.} Cada tempo de perna é um \omterm{def:g12:deriv:convex}{convexo} \omterm{def:g11:func:function}{função} de $ x $
(um $\sqrt{\text{quadratic}}$ filial), então $ T $ é \omterm{def:g12:deriv:convex}{convexo}, e um
ponto crítico de um \omterm{def:g12:deriv:convex}{convexo} \omterm{def:g12:deriv:derivative}{diferenciável} \omterm{def:g11:func:function}{função} é o seu
mundial \omterm{def:g10:functions:extrema}{mínimo} (\cref{exo:g12:deriv:8}): $ x^*$ não é meramente
estacionário, é imbatível.

\textbf{12.} A luz na água é mais lenta; pelo princípio de Fermat
o caminho ar-água mais rápido se curva na superfície exatamente
Lei de Snell --- então os raios de um lápis submerso atingem o olho
curvado, e o cérebro, extrapolando linhas retas, vê o
lápis quebrado na linha d'água.

\textbf{13.} $(x^2)'' = 2 > 0$: \omterm{def:g12:deriv:convex}{convexo}. Convexidade no
\omterm{prop:g10:coordgeom:midpoint}{ponto médio}: $ f\left(\frac{a+b}{2}\right) \leq
\frac{f(a) + f(b)}{2}$, que é a afirmação. À mão:
$\frac{a^2 + b^2}{2} - \left(\frac{a+b}{2}\right)^2 =
\frac{(a - b)^2}{4} \geq 0$.

\textbf{14.} Tirando raízes quadradas (\omterm{def:g12:seq:monotonic}{aumentando}) na questão 13:
$\frac{a + b}{2} \leq \sqrt{\frac{a^2 + b^2}{2}}$. Cadeia completa
ligado $2, 8$: harmônico $\frac{2 \cdot 16}{10} = 3.2$; \omterm{def:g12:seq:arith-geom}{geométrico}
$\sqrt{16} = 4$; \omterm{def:g12:seq:arith-geom}{aritmética} $5$; quadrático
$\sqrt{34} \approx 5.83$: cada significar se curva para o próximo.

\textbf{15.} \omterm{def:g12:deriv:tangent}{Tangente} para $\frac1x $ no $1$: valor $1$, declive
$-1$: $ y = 2 - x $. \omterm{def:g12:deriv:convex}{Convexo} curvas ficam acima de seus \omterm{def:g12:deriv:tangent}{tangentes}
(\cref{thm:g12:deriv:convexity}): $\frac1x \geq 2 - x $ para todos
$ x > 0$, com igualdade exatamente no ponto de contato $ x = 1$.

\textbf{16.} Na inflexão, o \emph{diário} contagem de casos
(o derivado) picos: o crescimento para de acelerar e começa
desacelerar --- o primeiro sinal matemático de que a onda
está mudando, muito antes de as contagens caírem. Antes disso,
$ f'' > 0$ (cada dia pior que o anterior); depois disso,
$ f'' < 0$ (ainda crescendo, mas mais lentamente).

\textbf{17.} $ S'(r) = 4\pi r - \frac{2V}{r^2} = 0$ dá
$ r^3 = \frac{V}{2\pi} = \frac{330}{2\pi}$:
$ r \approx 3.74$ cm, e
$ h = \frac{330}{\pi r^2} \approx 7.49$ cm.

\textbf{18.} $ h = \frac{V}{\pi r^2} = \frac{2\pi r^3}{\pi r^2}
= 2r $ no ideal: a altura é igual ao diâmetro, qualquer que seja o
volume.

\textbf{19.} $ S''(r) = 4\pi + \frac{4V}{r^3} > 0$: \omterm{def:g12:deriv:convex}{convexo}, então
o raio crítico é o global \omterm{def:g10:functions:extrema}{mínimo}. Suporte de latas de verdade
mais alto para aderência, empilhamento, presença de prateleira e mais grosso
metal superior/inferior --- a otimização sempre minimiza exatamente
o que você escreveu, não o que você quis dizer.

\textbf{20.} Salva-vidas: modelo $ T(x)$, diferenciar, Snell,
convexidade, um segundo salvo. Pode: modelo $ S(r)$, diferenciar,
$ h = 2r $, convexidade, metal salvo. Feixe
(\cref{pb:g11:deriv:1}): modelo $ S(w)$, diferenciar,
$ d = w\sqrt2$, mesa, rigidez adquirida. E por Fermat
princípio, a própria luz percorre o gasoduto em todas as superfícies que ela
atende --- a natureza foi o primeiro otimizador.
\end{solution}
