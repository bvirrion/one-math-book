\section*{Capítulo \ref{ch:g12:randvar} --- Variáveis Aleatórias e a Distribuição Binomial}

\begin{solution}{exo:g12:randvar:1}
$\E(X) = -0.3 + 0 + 0.8 + 0.5 = 1$.
$\E(X^2) = 0.3 \times 1 + 0 + 0.4\times4 + 0.1\times25 = 4.4$, então por
König-Huygens $\V(X) = 4.4 - 1 = 3.4$ e
$\sigma(X) = \sqrt{3.4} \approx 1.84$.
\end{solution}

\begin{solution}{exo:g12:randvar:2}
\emph{1.} O $10$ as perguntas são \omterm{def:g12:condprob:indep}{independente} Ensaios de Bernoulli com
$ p = \frac14$: $ X \sim \mathcal B\left(10, \frac14\right)$,
$\E(X) = 2.5$,
$\sigma(X) = \sqrt{10 \times \frac14 \times \frac34} = \sqrt{1.875}
\approx 1.37$.

\emph{2.} $\P(X = 0) = \left(\frac34\right)^{10} \approx 0.056$;
\[
\P(X = 5) = \binom{10}{5}\left(\frac14\right)^5\left(\frac34\right)^5
\approx 0.058;
\quad
\P(X \geq 1) = 1 - \left(\tfrac34\right)^{10} \approx 0.944 .
\]
\end{solution}

\begin{solution}{exo:g12:randvar:3}
\omterm{def:g12:condprob:indep}{Independente} ensaios idênticos: $ X \sim \mathcal B(400,\ 0.05)$, então
$\E(X) = 20$ reivindicações e
\[
\sigma(X) = \sqrt{400 \times 0.05 \times 0.95} = \sqrt{19} \approx 4.4 .
\]
\end{solution}

\begin{solution}{exo:g12:randvar:4}
O pagamento $ P $ recebido satisfaz
$\E(P) = \frac{5 + 6}{6} = \frac{11}{6}$, então o preço justo é
$ c = \frac{11}{6} \approx 1.83$ euros. O ganho justo $ G = P - \frac{11}6$
leva valores $-\frac{11}{6}, \frac{19}{6}, \frac{25}{6}$ com \omterm{def:g10:proba:distribution}{probabilidades}
$\frac46, \frac16, \frac16$:
\[
\V(G) = \E(G^2)
= \frac{4 \times 121 + 361 + 625}{6 \times 36} = \frac{1470}{216}
= \frac{245}{36} \approx 6.8,
\qquad \sigma(G) \approx 2.6 .
\]
Um jogo justo tem zero \emph{média} ganha, mas seus resultados ainda flutuam:
justo não é isento de riscos.
\end{solution}

\begin{solution}{exo:g12:randvar:5}
$ X \sim \mathcal B(8,\ 0.7)$.
\[
\P(X = 6) = \binom86 (0.7)^6(0.3)^2 \approx 0.296 ;
\]
\[
\P(X \geq 6) = \P(6) + \P(7) + \P(8)
\approx 0.296 + 8(0.7)^7(0.3) + (0.7)^8
\approx 0.296 + 0.198 + 0.058 = 0.552 ;
\]
$\P(X \geq 1) = 1 - (0.3)^8 \approx 0.99993$. Modo:
$(n+1)p = 6.3$, então o valor mais provável é $ k^* = 6$
(\cref{exo:g12:randvar:7}).
\end{solution}

\begin{solution}{exo:g12:randvar:6}
O número de passageiros que compareceram é $ X \sim \mathcal B(104,\ 0.9)$; o
o voo está com overbook quando $ X \geq 101$:
\[
\P(X \geq 101) = \sum_{k=101}^{104}\binom{104}{k}(0.9)^k(0.1)^{104-k}
\approx 0.006 .
\]
Vendendo $4\%$ mais ingressos do que assentos causa um incidente em apenas cerca de
$0.6\%$ de voos --- a economia por trás do overbooking.
\end{solution}

\begin{solution}{exo:g12:randvar:7}
\[
\frac{\P(X = k+1)}{\P(X = k)}
= \frac{\binom{n}{k+1}}{\binom nk}\cdot\frac{p}{1-p}
= \frac{n - k}{k + 1}\cdot\frac{p}{1-p},
\]
usando $\binom{n}{k+1} = \binom nk \frac{n-k}{k+1}$. Esta proporção é $\geq 1$
se $(n-k)p \geq (k+1)(1-p)$ se $ np - k p \geq k - kp + 1 - p $ se
$ k \leq (n+1)p - 1$. Portanto, as \omterm{def:g10:proba:distribution}{probabilidades} aumentam estritamente enquanto
$ k + 1 \leq (n+1)p $ e diminuir depois: o \omterm{def:g10:functions:extrema}{máximo} é alcançado em
$ k^* = \floor{(n+1)p}$ (compartilhado com $ k^* - 1$ quando $(n+1)p $ é um
inteiro).
\end{solution}

\begin{solution}{exo:g12:randvar:8}
\emph{1.} Cabeça primeiro no sorteio $ k $ significa $ k-1$ coroa e depois cara:
\omterm{def:g10:proba:distribution}{probabilidade} $\left(\frac12\right)^{k-1}\cdot\frac12 = 2^{-k}$, para
$1 \leq k \leq 10$. Vitória total \omterm{def:g10:proba:distribution}{probabilidade}
$\sum_{k=1}^{10} 2^{-k} = 1 - 2^{-10} = \frac{1023}{1024}$ (o jogo é
perdeu apenas em dez derrotas consecutivas).

\emph{2.} Pagamento esperado:
\[
\sum_{k=1}^{10} 2^k \cdot 2^{-k} = \sum_{k=1}^{10} 1 = 10 \text{ euros}.
\]
Sem o limite, a soma $\sum_{k\geq1} 1$ diverge: o retorno esperado é
infinito, embora o jogo quase sempre pague uma pequena quantia --- o
famoso \emph{Paradoxo de São Petersburgo}, mostrando que \omterm{def:g12:randvar:exp}{expectativa} sozinho faz
não medir o valor de um jogo.
\end{solution}

\begin{solution}{pb:g12:randvar:1}
\textbf{1.} $ X \sim \mathcal B\!\left(3, \frac16\right)$:
$\P(X = 0) = \frac{125}{216}$,
$\P(X = 1) = \frac{75}{216}$,
$\P(X = 2) = \frac{15}{216}$,
$\P(X = 3) = \frac{1}{216}$.

\textbf{2.} $\E(X) = 3 \times \frac16 = \frac12$;
$ V(X) = 3 \times \frac16 \times \frac56 = \frac{5}{12}$.

\textbf{3.} Pagamento esperado $2\,\E(X) = 1$ euro contra um
$1$-aposta em euros: ganho $0$ --- exatamente justo (uma raridade).

\textbf{4.} Sem reposição os empates ficam dependentes (o
as \omterm{def:g11:func:parity}{chances} da segunda carta dependem da primeira): não binomial ---
a lista de verificação independência caixa falha. Com substituição:
fixo $ n = 5$, constante $ p = \frac14$, \omterm{def:g12:condprob:indep}{independente} sorteios:
binômio.

\textbf{5.} $\E(X) = 6$, $\sigma(X) = \sqrt{4.2} \approx
2.05$. $\E(S) = 5 \times 6 - 10 = 20$;
$\sigma(S) = 5\sigma(X) \approx 10.25$.

\textbf{6.} Fixo $ n = 105$ testes (os ingressos), cada um
comparecimento/não comparecimento com o mesmo $ p = 0.9$, assumido \omterm{def:g12:condprob:indep}{independente}:
binômio. A confissão: famílias perdem voos
\emph{junto} e tempestades esvaziam aviões \omterm{def:g10:numbers:sets}{inteiros} --- independência
é o salto de fé do modelo, e o não comparecimento correlacionado faz
as caudas são mais gordas do que as promessas do binômio.

\textbf{7.} $\E(X) = 94.5$;
$\sigma(X) = \sqrt{105 \times 0.9 \times 0.1} =
\sqrt{9.45} \approx 3.07$.

\textbf{8.} $\P(X \geq 101) = \sum_{k=101}^{105}
\binom{105}{k} 0.9^k\, 0.1^{105 - k}$.

\textbf{9.} $\approx 0.0167$: cerca de um voo em sessenta
esbarra em qualquer um.

\textbf{10.} $\E(\text{bumped}) \approx 0.023$ passageiros por
voo: compensação esperada $\approx 19$ euros --- contra
$1\,000$ euros de receita extra. Overbooking por cinco é
esmagadoramente lucrativo; portanto, todas as companhias aéreas fazem isso.

\textbf{11.} Da tabela de \omterm{def:g10:proba:distribution}{probabilidades} de cauda:
$ n = 105$: $1.7\,\%$; $ n = 106$: $4.0\,\%$; $ n = 107$:
$8.1\,\%$. A maior venda compatível é $ n = 106$ ingressos.

\textbf{12.} $ z = \frac{100.5 - 94.5}{3.07} \approx 1.95$:
o limite fica dois desvios padrão acima do significar,
e a regra prática da curva em sino (``superior $2\sigma $ cauda
$\approx 2.5\,\%$'') pousa perto do exato $1.7\,\%$ ---
a curva suave que sombreia o binômio é a vinda
protagonista dos capítulos.

\textbf{13.} $0.98^{20} + 20 \times 0.02 \times 0.98^{19}
\approx 0.94$: o lote honesto passa $94\,\%$ da época.

\textbf{14.} $0.9^{20} + 20 \times 0.1 \times 0.9^{19}
\approx 0.39$: o lote ruim ainda se esgueira $39\,\%$ de
a hora.

\textbf{15.} Risco do produtor: um bom lote rejeitado
($\approx 6\,\%$); risco do consumidor: um lote ruim aceito
($\approx 39\,\%$). Um maior \omterm{def:g10:proba:sample}{amostra} (com uma proporção
limite de aceitação) diminui ambos --- ao custo de mais
inspeção: a qualidade tem uma rubrica orçamentária.

\textbf{16.} $ V\!\left(\frac Xn\right) = \frac{V(X)}{n^2} =
\frac{np(1-p)}{n^2} = \frac{p(1-p)}{n}$: \omterm{def:g11:stat:variance}{desvio padrão}
$\sqrt{\frac{p(1-p)}{n}}$. Quadruplicar o \omterm{def:g10:proba:sample}{amostra}, reduza pela metade
ruído: o $\frac{1}{\sqrt n}$ lei da flutuação do 10º ano
\omterm{def:g10:numbers:interval}{intervalos}, agora derivado.

\textbf{17.} O limite limita o pagamento em $2^{10} = 1024$
euros, então cada uma das dez rodadas contribui exatamente $1$ euro
de \omterm{def:g12:randvar:exp}{expectativa}: $\E = 10$. O jogo ilimitado é infinito
\omterm{def:g12:randvar:exp}{expectativa} veio inteiramente de astronomicamente raro,
recompensas astronomicamente grandes; capping confessa que nenhum banco
paga $2^{50}$ euros. Um preço de ingresso justo para o jogo limitado:
$10$ euros --- e de repente ninguém fica paradoxal.

\textbf{18.} Prêmio em dinheiro esperado por ingresso:
$\frac{100 + 2 \times 50}{200} = 1$ euro contra um $2$-euro
bilhete: margem $50\,\%$ --- dezoito vezes na roleta
$2.7\,\%$. O sorteio sobrevive porque seus jogadores são
doar conscientemente: a “perda” é o ponto.

\textbf{19.} Reivindicações esperadas: $1\,000 \times 0.01 \times
10\,000 = 100\,000$; prêmios $130\,000$: lucro esperado
$30\,000$ euros. \omterm{def:g11:stat:variance}{Desvio padrão} do total de reclamações:
$10\,000 \times \sqrt{1\,000 \times 0.01 \times 0.99}
\approx 31\,500$ euros --- um ano ruim comum devora o
lucro total esperado. Daí as reservas de capital, resseguros,
e carteiras muito maiores que mil apólices: seguradoras
viver da lei dos grandes números e manter reservas contra
sua lentidão.

\textbf{20.} Lista de verificação primeiro --- corrigida $ n $, mesmo $ p $,
independência \emph{reivindicado} --- ou nenhum binômio. Então
navegar por $\E \pm \sigma $: significa localizar, alertar sobre desvios.
Então avalie as caudas: esbarrões, lotes ruins, anos ruinosos, tudo
viver além $2\sigma$. Advertências: independência é do modelador
promessa, não a do mundo; e \omterm{def:g12:randvar:exp}{expectativa} ignora exatamente
o que destrói você. As páginas que faltam no livro de regras --- quão rápido
as \omterm{def:g10:stats:series}{frequências} se estabilizam e qual a forma que as flutuações assumem ---
são os próximos dois capítulos.
\end{solution}
