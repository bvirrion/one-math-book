\section*{Capítulo \ref{ch:g12:integ} --- Integração}

\begin{solution}{exo:g12:integ:1}
\[
\int_1^2 \left(3x^2 - \frac{1}{x^2}\right)\dd x
= \left[x^3 + \frac1x\right]_1^2 = \left(8 + \tfrac12\right) - 2 = \frac{13}{2}.
\]
\[
\int_0^{\pi/2} \cos t\,\dd t = \bigl[\sin t\bigr]_0^{\pi/2} = 1 .
\]
Com $u = 2t+1$, $\frac{1}{2t+1} = \frac12\,\frac{u'}{u}$:
\[
\int_0^1 \frac{\dd t}{2t+1} = \frac12\bigl[\ln(2t+1)\bigr]_0^1
= \frac{\ln 3}{2}.
\]
\end{solution}

\begin{solution}{exo:g12:integ:2}
$ x\,\eu^{x^2} = \frac12\,(x^2)'\,\eu^{x^2}$ tem \omterm{def:g12:integ:primitive}{primitivos}
$\frac12\eu^{x^2} + c $. A condição $ F(0) = 1$ dá $\frac12 + c = 1$,
então
\[
F(x) = \frac{\eu^{x^2} + 1}{2}.
\]
\end{solution}

\begin{solution}{exo:g12:integ:3}
\[
\mu = \frac{1}{\pi}\int_0^\pi \sin t\,\dd t
= \frac{1}{\pi}\bigl[-\cos t\bigr]_0^\pi = \frac{2}{\pi} \approx 0.64 .
\]
Graficamente, o retângulo da base $\intcc{0}{\pi}$ e altura
$\frac{2}{\pi}$ tem a mesma área do arco do \omterm{def:g12:trigo:cossin}{seno} curva.
\end{solution}

\begin{solution}{exo:g12:integ:4}
Pegar $ u' = 1$, $ v = \ln t $, então $ u = t $, $ v' = \frac1t $:
\[
\int_1^{\eu} \ln t\,\dd t = \bigl[t\ln t\bigr]_1^{\eu} - \int_1^{\eu} 1\,\dd t
= \eu - (\eu - 1) = 1 .
\]
Pegar $ u' = \sin t $, $ v = t $, então $ u = -\cos t $, $ v' = 1$:
\[
\int_0^\pi t\sin t\,\dd t = \bigl[-t\cos t\bigr]_0^\pi + \int_0^\pi \cos t\,\dd t
= \pi + 0 = \pi .
\]
\end{solution}

\begin{solution}{exo:g12:integ:5}
As curvas se cruzam onde $ x^2 = x + 2$, \emph{ou seja} $ x \in \{-1, 2\}$ e
ligado $\intcc{-1}{2}$ a linha está acima do \omterm{def:g11:quad:parabola}{parábola}. A área é
\[
\int_{-1}^{2} \bigl(x + 2 - x^2\bigr)\dd x
= \left[\frac{x^2}{2} + 2x - \frac{x^3}{3}\right]_{-1}^{2}
= \frac{10}{3} - \left(-\frac{7}{6}\right) = \frac{9}{2}.
\]
\end{solution}

\begin{solution}{exo:g12:integ:6}
\emph{1.} $ I = \bigl[\ln(1+t)\bigr]_0^1 = \ln 2$.

\emph{2.}
\[
I_n + I_{n+1} = \int_0^1 \frac{t^n(1 + t)}{1+t}\,\dd t
= \int_0^1 t^n \,\dd t = \frac{1}{n+1}.
\]
Sobre $\intcc{0}{1}$, $0 \leq \dfrac{t^n}{1+t} \leq t^n $, então por comparação
$0 \leq I_n \leq \frac{1}{n+1}$.

\emph{3.} Da recorrência, por uma indução telescópica,
\[
I_n = (-1)^n\left(I_0 - \left(1 - \frac12 + \dots + \frac{(-1)^{n-1}}{n}\right)\right),
\]
\emph{ou seja}
$\displaystyle\sum_{k=1}^{n} \frac{(-1)^{k-1}}{k} = \ln 2 - (-1)^n I_n $
(lembrar $ I_0 = \ln 2$). Desde $\abs{I_n} \leq \frac{1}{n+1} \to 0$, o
soma harmônica alternada tende a $\ln 2$.
\end{solution}

\begin{solution}{exo:g12:integ:7}
Distância:
\[
d = \int_0^{100} 30\left(1 - \eu^{-t/50}\right)\dd t
= 30\left[t + 50\,\eu^{-t/50}\right]_0^{100}
= 30\left(100 + 50\eu^{-2} - 50\right)
= 1500 + 1500\,\eu^{-2}.
\]
Numericamente $ d \approx 1500 + 203 = 1703$ m. Velocidade média:
$\frac{d}{100} = 15\bigl(1 + \eu^{-2}\bigr) \approx 17.0$ EM.
\end{solution}

\begin{solution}{exo:g12:integ:8}
\emph{1.} $ S_n $ é a área total de $ n $ retângulos de largura $\frac1n $ e
alturas $ f\left(\frac kn\right)$, $ k = 1, \dots, n $, onde
$ f(t) = \frac{1}{1+t}$: é o retângulo ``ponto final'' \omterm{def:g10:numbers:approx}{aproximação}
da área sob $ f $ sobre $\intcc{0}{1}$.

\emph{2.} $ f $ é diminuindo, então em cada \omterm{def:g10:numbers:interval}{intervalo}
$\intcc{\frac{k-1}{n}}{\frac kn}$,
$ f\left(\frac kn\right) \leq f(t) \leq f\left(\frac{k-1}{n}\right)$;
integrando e somando $ k $:
\[
S_n \leq \int_0^1 f(t)\,\dd t \leq S_n + \frac1n\bigl(f(0) - f(1)\bigr)
= S_n + \frac{1}{2n}.
\]
(O termo intermediário da exibição do exercício, com
$ f(0) - f(1) = 1 - \frac12$, é exatamente esse limite.) Portanto
$0 \leq \ln 2 - S_n \leq \frac{1}{2n} \to 0$, então $ S_n \to \ln 2$.
\end{solution}

\begin{solution}{exo:g12:integ:9}
\emph{1.} $ W_0 = \int_0^{\pi/2} \dd t = \frac{\pi}{2}$ e
$ W_1 = \bigl[-\cos t\bigr]_0^{\pi/2} = 1$.

\emph{2.} Integrar por partes com $ u' = \sin t $, $ v = \sin^{n+1} t $, então
$ u = -\cos t $, $ v' = (n+1)\sin^n t \cos t $:
\[
W_{n+2} = \bigl[-\cos t \sin^{n+1} t\bigr]_0^{\pi/2}
+ (n+1)\int_0^{\pi/2} \cos^2 t\,\sin^n t\,\dd t
= (n+1)\int_0^{\pi/2} (1 - \sin^2 t)\sin^n t\,\dd t,
\]
o colchete desaparece em ambas as extremidades. Por isso
$ W_{n+2} = (n+1)(W_n - W_{n+2})$, \emph{ou seja}
$ W_{n+2} = \frac{n+1}{n+2} W_n $.

\emph{3.} $ W_2 = \frac12 W_0 = \frac{\pi}{4}$,
$ W_3 = \frac23 W_1 = \frac23$,
$ W_4 = \frac34 W_2 = \frac{3\pi}{16}$.
Sobre $\intoo{0}{\frac\pi2}$, $0 < \sin t < 1$, então
$\sin^{n+1} t \leq \sin^n t $ com estrita desigualdade no interior;
integrando, $0 < W_{n+1} \leq W_n $ (na verdade $<$): $(W_n)$ é diminuindo e
positivo.
\end{solution}

\begin{solution}{pb:g12:integ:1}
\textbf{1.} $\left[x^3 - x^2 + x\right]_0^1 = 1$; \quad
$\left[\ln x\right]_1^{\eu} = 1$; \quad
$\left[\sin x\right]_0^{\pi/2} = 1$.

\textbf{2.} $\left[\frac12 \eu^{x^2}\right]_0^1 =
\frac{\eu - 1}{2}$; \quad
$\left[\ln(x^2 + 1)\right]_0^1 = \ln 2$.

\textbf{3.} $\int_0^1 x\eu^x = \left[x\eu^x\right]_0^1 -
\int_0^1 \eu^x = \eu - (\eu - 1) = 1$. E
$\int_1^{\eu} \ln x = \left[x\ln x\right]_1^{\eu} -
\int_1^{\eu} 1 = \eu - (\eu - 1) = 1$.

\textbf{4.} $\frac{1}{\pi}\int_0^\pi \sin x\,\dd x =
\frac{2}{\pi} \approx 0.64$: a altura média do \omterm{def:g12:trigo:cossin}{seno}
arco é $\frac2\pi $, visivelmente mais da metade --- o
o arco é grosso perto do topo.

\textbf{5.} $\int_0^1 (x - x^2)\,\dd x = \frac12 - \frac13 =
\frac16$.

\textbf{6.} $ L_n = \frac1n \sum_{k=0}^{n-1}
\left(\frac kn\right)^2$ e
$ U_n = \frac1n \sum_{k=1}^{n} \left(\frac kn\right)^2$:
retângulos de largura $\frac1n $ com alturas lidas à esquerda
(abaixo da curva) ou extremidades direitas (acima).

\textbf{7.} Verdadeiro para $ n = 1$ ($1 = \frac{1 \cdot 2 \cdot
3}{6}$). Se valer por $ n $, então adicionando $(n+1)^2$:
\[
\frac{n(n+1)(2n+1)}{6} + (n+1)^2
= \frac{(n+1)\left(2n^2 + n + 6n + 6\right)}{6}
= \frac{(n+1)(n+2)(2n+3)}{6},
\]
a fórmula em $ n + 1$: hereditariedade, feito.

\textbf{8.} $ U_n = \frac{1}{n^3} \cdot
\frac{n(n+1)(2n+1)}{6} = \frac{(1 + \frac1n)(2 + \frac1n)}{6}
\to \frac26 = \frac13$, e $ L_n = U_n - \frac1n \to \frac13$:
espremida entre suas escadarias, a área é exatamente
$\frac13$.

\textbf{9.} $\int_0^1 x^2\,\dd x =
\left[\frac{x^3}{3}\right]_0^1 = \frac13$: uma linha. O
teorema fundamental converte um aperto infinito em um único
anti-derivado avaliação --- é por isso que é chamado
fundamental.

\textbf{10.} Segmento: $\int_{-1}^{1}(1 - x^2)\,\dd x =
2 - \frac23 = \frac43$. Triângulo com base na corda (comprimento
$2$, em altura $1$) e ápice $(0,0)$: área
$\frac12 \times 2 \times 1 = 1$. Razão: $\frac43$ --- exatamente
Teorema de Arquimedes, verificado pela máquina que lhe faltava.

\textbf{11.} $ T\left(1 + \frac14 + \frac{1}{16} + \dots\right)
= T \cdot \frac{1}{1 - \frac14} = \frac43 T $ (série geométrica
de proporção $\frac14$): com $ T = 1$, a área do segmento
$\frac43$ de novo. Arquimedes resumiu a série de uma forma puramente
argumento \omterm{def:g12:seq:arith-geom}{geométrico} --- os quartos da barra de chocolate
tipo, dois milênios antes dos limites serem palavras.

\textbf{12.} A exaustão precisa de uma geometria inteligente
autosimilaridade e fornece áreas exatas caso a caso; Riemann
precisa apenas de monotonicidade e dá uma definição universal em
o preço das somas computacionais; o teorema fundamental precisa do
todo o edifício de derivados e retribui fazendo áreas
cálculos de uma linha.

\textbf{13.} $ A(1) = 0$ e $ A'(x) = \frac1x $ (fundamental
teorema): $ A $ é o \omterm{def:g12:integ:primitive}{primitivo} de $\frac1x $ sobre
$\intoo{0}{+\infty}$ desaparecendo em $1$ --- o que é precisamente
o logaritmo natural.

\textbf{14.} $ g'(x) = \frac{a}{ax} - \frac1x = 0$: $ g $ é
constante; no $ x = 1$: $ g(1) = A(a) - A(1) = A(a)$. Por isso
$ A(ax) = A(a) + A(x)$ para todos $ x $, e com $ x = b $:
$\ln(ab) = \ln a + \ln b $. O hipérbolea área não se importa
sobre escala horizontal: alongamento $ t $ por $ a $ abóboras
$\frac1t $ por $ a $, e a área sobrevive --- adição de
logaritmos é uma simetria de escala.

\textbf{15.} Iterando: $\ln(a^n) = n\ln a $ (indução em
$ n $). E $0 = \ln 1 = \ln\left(a \cdot \frac1a\right) =
\ln a + \ln\frac1a $: $\ln\frac1a = -\ln a $.

\textbf{16.} $\frac{1}{11} + \frac{1}{12} + \dots +
\frac{1}{20} \approx 0.669$, contra $\ln 2 \approx 0.693$:
dez retângulos abaixo $\frac{1}{1+t}$, já perto. O
a série harmônica diverge, mas sua \emph{fatias} de $ n $ para
$2n $ resolver $\ln 2$: divergência em um crawl, medida por um
área.

\textbf{17.} Distância: $\int_0^{10} 3t^2\,\dd t = 1000$ m.
Significar velocidade: $100$ EM. Instantâneo com $ v = 100$:
$3t^2 = 100$: $ t = \sqrt{\frac{100}{3}} \approx 5.77$ S.

\textbf{18.} Massa: $\int_0^4 (2 + x)\,\dd x = 8 + 8 = 16$ kg.
Momento: $\int_0^4 (2x + x^2)\,\dd x = 16 + \frac{64}{3} =
\frac{112}{3}$. Centro de massa:
$\frac{112/3}{16} = \frac73 \approx 2.33$ m --- empurrado além
o meio pela extremidade direita mais pesada, como exige a intuição.

\textbf{19.} $ W_0 = \frac\pi2$; $ W_1 = \left[-\cos
t\right]_0^{\pi/2} = 1$; e
$ W_2 = \int_0^{\pi/2} \frac{1 - \cos 2t}{2}\,\dd t =
\frac\pi4$.

\textbf{20.} Área: a definição, espremida por escadas.
Acumulação: distâncias de velocidades, massas de densidades ---
o integrante como um total corrente. Anti-derivado: o
teorema fundamental, o atalho de Newton e Leibniz através
A paciência de Arquimedes. A pérola: a $\ln $ botão é a área
sob $\frac1t $, e a simetria de escala dessa área \emph{é}
a identidade $\ln(ab) = \ln a + \ln b$.
\end{solution}
