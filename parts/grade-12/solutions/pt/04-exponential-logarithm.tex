\section*{Capítulo \ref{ch:g12:exp} --- Exponencial e Logaritmo}

\begin{solution}{exo:g12:exp:1}
$\dfrac{\eu^{3x}\,\eu^{-x+1}}{\eu^{x}} = \eu^{3x - x + 1 - x} = \eu^{x+1}$.

$\ln\!\left(\dfrac{\eu^2\sqrt{\eu}}{\eu^{-3}}\right)
= 2 + \tfrac12 + 3 = \tfrac{11}{2}$.
\end{solution}

\begin{solution}{exo:g12:exp:2}
\emph{(um)} Definir $ X = \eu^x > 0$: $ X^2 - 3X + 2 = 0$ dá $ X = 1$ ou $ X = 2$,
ambos positivos, então $ x = 0$ ou $ x = \ln 2$.

\emph{(b)} Domínio: $ x - 1 > 0$ e $ x + 2 > 0$, então $ x > 1$. O equação
torna-se $\ln\bigl((x-1)(x+2)\bigr) = \ln 4$, por isso $(x-1)(x+2) = 4$,
\emph{ou seja} $ x^2 + x - 6 = 0$, então $ x = 2$ ou $ x = -3$. Apenas $ x = 2$ está no
domínio: $ S = \{2\}$.
\end{solution}

\begin{solution}{exo:g12:exp:3}
Dividir por $\eu^x $:
$\dfrac{1 - x^2\eu^{-x}}{1 + \eu^{-x}} \to \dfrac{1-0}{1+0} = 1$, usando
$ x^2 \eu^{-x} \to 0$ (\cref{thm:g12:exp:growth}).

$\dfrac{\ln x}{\sqrt x} \to 0$ por \cref{prop:g12:exp:lnanalytic} (caso
$ n = 2$).

$ x^2 \ln x = x \cdot (x \ln x) \to 0 \times 0 = 0$.

$ x - \ln x = x\left(1 - \frac{\ln x}{x}\right) \to +\infty $ desde
$\frac{\ln x}{x} \to 0$.
\end{solution}

\begin{solution}{exo:g12:exp:4}
$ f'(x) = \eu^{-x} - x\eu^{-x} = (1 - x)\eu^{-x}$, do sinal de $1 - x $:
$ f $ aumenta em $\intoc{-\infty}{1}$, diminui em $\intco{1}{+\infty}$,
com um mundo \omterm{def:g10:functions:extrema}{máximo} $ f(1) = \eu^{-1}$.

Limites: como $ x \to +\infty $, $ f(x) = \frac{x}{\eu^x} \to 0$
(\cref{thm:g12:exp:growth}); como $ x \to -\infty $, $ x \to -\infty $ e
$\eu^{-x} \to +\infty $, então $ f(x) \to -\infty $.

$ f''(x) = -\eu^{-x} - (1-x)\eu^{-x} = (x - 2)\eu^{-x}$, que muda de sinal em
$ x = 2$: ponto de inflexão no $\bigl(2,\, 2\eu^{-2}\bigr)$.
\end{solution}

\begin{solution}{exo:g12:exp:5}
Deixar $ g(x) = x - \ln(1+x)$ sobre $\intoo{-1}{+\infty}$. Então
$ g'(x) = 1 - \frac{1}{1+x} = \frac{x}{1+x}$, negativo em $\intoo{-1}{0}$ e
positivo em $\intoo{0}{+\infty}$: $ g $ atinge o seu \omterm{def:g10:functions:extrema}{mínimo} $ g(0) = 0$, então
$ g \geq 0$ com igualdade apenas em $0$. Por isso $\ln(1+x) \leq x $.

Aplique isso com $ x = \frac1n $:
$ n\ln\left(1 + \frac1n\right) \leq 1$, então
$\left(1+\frac1n\right)^n = \eu^{\,n\ln(1+1/n)} \leq \eu $.
Aplique com $ x = -\frac{1}{n+1} > -1$:
$\ln\left(1 - \frac{1}{n+1}\right) \leq -\frac{1}{n+1}$, então
$-(n+1)\ln\left(1 - \frac{1}{n+1}\right) \geq 1$ e
$\left(1 - \frac{1}{n+1}\right)^{-(n+1)} \geq \eu $.
\end{solution}

\begin{solution}{exo:g12:exp:6}
\emph{1.} Cada ano multiplica o capital por $1.03$: $ C_n = C_0 (1.03)^n $.

\emph{2.} $ C_n \geq 2C_0 \iff (1.03)^n \geq 2 \iff n \ln 1.03 \geq \ln 2
\iff n \geq \frac{\ln 2}{\ln 1.03} \approx 23.45$: o capital dobra depois
$24$ anos.

\emph{3.} $\frac{70}{3} \approx 23.3$, perto do exato
$\frac{\ln 2}{\ln 1.03}$. A regra funciona porque
$\ln(1 + r) \approx r $ para pequeno $ r $, então
$\frac{\ln 2}{\ln(1+r)} \approx \frac{0.693}{r} \approx \frac{70}{100r}$.
\end{solution}

\begin{solution}{exo:g12:exp:7}
$\eu^{2x} > \eu^{x+1} \iff 2x > x + 1 \iff x > 1$ (a exponencial é
estritamente \omterm{def:g12:seq:monotonic}{aumentando}): $ S = \intoo{1}{+\infty}$.

Domínio da segunda desigualdade: $ x^2 - 1 > 0$ e $ x + 5 > 0$, \emph{ou seja}
$ x \in \intoo{-5}{-1} \cup \intoo{1}{+\infty}$. Nesta domínio, $\ln $ sendo
estritamente \omterm{def:g12:seq:monotonic}{aumentando},
\[
\ln(x^2-1) \leq \ln(x+5) \iff x^2 - 1 \leq x + 5 \iff x^2 - x - 6 \leq 0
\iff x \in \intcc{-2}{3}.
\]
Cruzando com o domínio: $ S = \intco{-2}{-1} \cup \intoc{1}{3}$.
\end{solution}

\begin{solution}{exo:g12:exp:8}
\emph{1.} $ f'(x) = \dfrac{\eu^x x - \eu^x}{x^2} = \dfrac{(x-1)\eu^x}{x^2}$,
negativo ativado $\intoo{0}{1}$, positivo em $\intoo{1}{+\infty}$: \omterm{def:g10:functions:extrema}{mínimo}
$ f(1) = \eu $. Limites: $ f \to +\infty $ no $0^+$ (numerador $\to 1$,
denominador $\to 0^+$) e em $+\infty $ (\cref{thm:g12:exp:growth}).

\emph{2.} Para $ x > 0$, $\eu^x = kx \iff f(x) = k $. Do \omterm{def:g10:functions:table}{tabela de variação}
($+\infty \searrow \eu \nearrow +\infty $, contínuo e estritamente monótono
de cada lado): nenhuma solução para $ k < \eu $; exatamente um ($ x = 1$) para
$ k = \eu $; exatamente dois para $ k > \eu $ (um em $\intoo{0}{1}$, um em
$\intoo{1}{+\infty}$, pelo teorema da bijeção em cada \omterm{def:g10:numbers:interval}{intervalo}).
\end{solution}

\begin{solution}{exo:g12:exp:9}
\emph{1.} Direto da primeira desigualdade de \cref{exo:g12:exp:5}:
$ u_n = \eu^{\,n\ln(1+1/n)} \leq \eu^1 = \eu $.

\emph{2.} Escrever
\[
\ln u_n = n \ln\!\left(1 + \frac1n\right)
= \frac{\ln\!\left(1 + \frac1n\right) - \ln 1}{\frac1n},
\]
um quociente de diferença de $\ln $ no ponto $1$ com incremento
$ h = \frac1n \to 0$. Desde $\ln $ é \omterm{def:g12:deriv:derivative}{diferenciável} no $1$ com derivado
$1$, $\ln u_n \to 1$. Por continuidade de $\exp $
(\cref{prop:g12:limcont:seqcont}), $ u_n = \eu^{\ln u_n} \to \eu^1 = \eu $.
\end{solution}

\begin{solution}{pb:g12:exp:1}
\textbf{1.} $ x = \frac{\ln 5}{2}$. \quad
$3x - 1 = \eu^2$: $ x = \frac{\eu^2 + 1}{3}$. \quad Com
$ u = \eu^x $: $ u^2 - 3u + 2 = (u - 1)(u - 2) = 0$: $ x = 0$ ou
$ x = \ln 2$.

\textbf{2.} $\frac{3\ln 2}{\ln 2} = 3$; \quad
$3 + \frac12 = \frac72$; \quad $\frac52$.

\textbf{3.} $ f'(x) = \eu^x - 1$: negativo antes $0$, positivo
depois: \omterm{def:g10:functions:extrema}{mínimo} $ f(0) = 0$, então $ f \geq 0$ em todos os lugares:
$\eu^x \geq 1 + x $. Substituindo $ x = \ln(1 + u)$:
$1 + u \geq 1 + \ln(1 + u)$, ou seja\ $\ln(1 + u) \leq u $.

\textbf{4.} $ x^2\eu^{-x} = \frac{x^2}{\eu^x} \to 0$ e
$\frac{\ln x}{x} \to 0$: exponenciais esmagam poderes, poderes
esmagar logaritmos. E $\frac{\eu^x - 1}{x} =
\frac{\eu^x - \eu^0}{x - 0} \to \exp'(0) = 1$.

\textbf{5.} $ f'(x) = \ln x + 1$: zero em $\eu^{-1}$, negativo
antes, positivo depois: \omterm{def:g10:functions:extrema}{mínimo}
$ f\!\left(\frac1\eu\right) = -\frac1\eu $; e
$ x \ln x \to 0$ no $0^+$: a curva sai da origem, mergulha para
$-\frac1\eu $, e sobe para longe.

\textbf{6.} $ n = \frac{\ln 2}{\ln 1.03} \approx 23.4$ anos.

\textbf{7.} Exato: $23.4$, $11.9$, $8.0$ anos; regra de 72:
$24$, $12$, $8$. Desde $\ln(1+r) \approx r $,
$\frac{\ln 2}{\ln(1+r)} \approx \frac{0.693}{r}$, ou seja\
$\frac{69.3}{\text{rate in }\%}$; banqueiros arredondam para $72$
porque se divide lindamente por $2, 3, 4, 6, 8, 9, 12$ ---
a \omterm{def:g12:seq:arith-geom}{aritmética} mental supera uma casa decimal em precisão.

\textbf{8.} $2^{t/20} = \eu^{t \ln 2 / 20}$:
$\lambda = \frac{\ln 2}{20}$ por minuto. Depois
$24 \times 60 = 1440$ minutos: $2^{72} \approx 4.7 \times
10^{21}$ bactérias --- mais do que os grãos de areia da Terra,
de uma célula em um dia. O modelo é honesto apenas enquanto comida
e o espaço por último: o crescimento real curva-se na curva S logística
(a curva do epidemiologista de \cref{pb:g12:deriv:1}).

\textbf{9.} Às 23h ($8$ horas):
$100 \times 2^{-8/5} \approx 33$ mg. Abaixo $10$ mg:
$2^{-t/5} < 0.1$ dá $ t > 5\,\frac{\ln 10}{\ln 2} \approx
16.6$ h: por volta das 7h40 da manhã seguinte --- o café expresso às
três tem uma cauda longa.

\textbf{10.} Anual: $2$. Mensal:
$\left(1 + \frac{1}{12}\right)^{12} \approx 2.613$. Diário:
$\approx 2.715$. O teto: $\eu = 2.71828\ldots $
(\cref{exo:g12:exp:9}) --- composição continuamente, ambição
converge.

\textbf{11.} Se os casos crescerem exponencialmente, $ c(t) = c_0
\eu^{\lambda t}$, então $\ln c(t) = \ln c_0 + \lambda t $: um
linha reta de declive $\lambda $ --- a taxa de crescimento. Um
trecho reto no \omterm{def:g10:functions:graph}{gráfico} de toras \emph{é} o exponencial
fase, e a sua inclinação é o ritmo da epidemia.

\textbf{12.} $120 - 60 = 60$ dB significa
$10\log_{10}(I_2/I_1) = 60$: o concerto é
$10^6$ --- um milhão de vezes --- mais intenso que o
conversa.

\textbf{13.} Amplitude: $10^{7-5} = 100$. Energia:
$100^{3/2} = 1000$: dois pontos de magnitude escondem três ordens de
magnitude em energia.

\textbf{14.} $10^{6.5 - 2} = 10^{4.5} \approx 32\,000$: limão
o suco é trinta mil vezes mais ácido que o leite --- pH
comprime os abismos da química numa balança de bolso.

\textbf{15.} $\log_2\!\frac{4186}{27.5} =
\frac{\ln(4186/27.5)}{\ln 2} \approx 7.25$: um piano abrange apenas
mais de sete oitavas. Os sentidos respondem a \emph{proporções} de estímulos
--- dobrar a intensidade parece um passo, seja qual for o
nível inicial --- então a percepção é construída em um logarítmico
escala, assim como as unidades que inventamos para ela.

\textbf{16.} Colocando o bastão para $ a $ de ponta a ponta com
posição da vara para $ b $ adiciona os comprimentos $\ln a + \ln b $ e
a formatura sentada nesse comprimento total diz
$\eu^{\ln a + \ln b} = ab $: a régua de cálculo calcula produtos
adicionando logaritmos --- $\ln(ab) = \ln a + \ln b $
(\cref{prop:g12:exp:lnalg}) esculpido em buxo.

\textbf{17.} O \omterm{def:g10:functions:extrema}{mínimo} de $\frac{\eu^x}{x}$ sobre
$\intoo{0}{+\infty}$ é $\eu $, no $ x = 1$: nenhuma solução para
$ k < \eu $, exatamente um para $ k = \eu $, dois para $ k > \eu $.
\omterm{def:g12:deriv:tangent}{Tangente} verifique: em $ a = 1$ o \omterm{def:g12:deriv:tangent}{tangente} para $\eu^x $ é
$ y = \eu + \eu(x - 1) = \eu x $: passa pela origem
--- a linha limite, passando pela curva em $(1, \eu)$.

\textbf{18.} $ n \ln\!\left(1 - \frac1n\right) =
\frac{\ln(1 - 1/n)}{-1/n} \times (-1) \to -1$, então
$\left(1 - \frac1n\right)^n \to \eu^{-1} \approx 0.368$: o
$0.999^{1000} \approx 0.368$ de \cref{pb:g11:binom:1},
explicado --- a constante de quase acidente da loteria é
$\frac1\eu $.

\textbf{19.} Ambos são a forma limite de “muitos
\omterm{def:g11:prob:model}{eventos}, cada um individualmente improvável'': a \omterm{def:g11:func:parity}{chance} de que
\emph{nenhum} de $ n $ \omterm{def:g11:func:parity}{chances} de tamanho $\frac1n $ incêndios tendem a
$\eu^{-1}$, e a regra do secretário ajusta sua janela de rejeição
para que o sucesso se concentre nessa mesma constante.
$\frac1\eu \approx 0.368$.

\textbf{20.} O teto da composição; a base única cuja
\omterm{def:g12:deriv:tangent}{tangente} no $0$ tem declive $1$ (é por isso que o cálculo adora);
o crescimento que ultrapassa qualquer poder; a constante onde
quase-acidentes e parada ideal; e o logaritmo ---
multiplicação torna-se adição, as \omterm{def:g10:stats:quartiles}{faixas} mais selvagens do mundo
dobrado em uma régua.
\end{solution}
