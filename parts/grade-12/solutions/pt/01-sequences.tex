\section*{Capítulo \ref{ch:g12:seq} --- Sequências}

\begin{solution}{exo:g12:seq:1}
Deixar $P(n)$ seja a declaração $\sum_{k=1}^{n} k^2 = \frac{n(n+1)(2n+1)}{6}$.
\emph{Caso básico:} para $ n = 0$ ambos os lados estão $0$ (soma vazia).
\emph{Etapa indutiva:} assumir $ P(n)$. Então
\[
\sum_{k=1}^{n+1} k^2 = \frac{n(n+1)(2n+1)}{6} + (n+1)^2
= \frac{(n+1)\bigl(n(2n+1) + 6(n+1)\bigr)}{6}
= \frac{(n+1)(2n^2 + 7n + 6)}{6}.
\]
Desde $2n^2 + 7n + 6 = (n+2)(2n+3)$, isso é
$\frac{(n+1)(n+2)(2(n+1)+1)}{6}$, que é $ P(n+1)$. Por indução, $ P(n)$
vale para todos $ n $.
\end{solution}

\begin{solution}{exo:g12:seq:2}
$ a_{n+1} - a_n = \frac{n+2}{n+1} - \frac{n+1}{n}
= \frac{n(n+2) - (n+1)^2}{n(n+1)} = \frac{-1}{n(n+1)} < 0$: $(a_n)$ é
estritamente diminuindo.

$(b_n)$ tem termos positivos e
$\frac{b_{n+1}}{b_n} = \frac{2^{n+1}}{n+1}\cdot\frac{n}{2^n} =
\frac{2n}{n+1} \geq 1 \iff 2n \geq n+1 \iff n \geq 1$: $(b_n)$ é \omterm{def:g12:seq:monotonic}{aumentando}
(estritamente para $ n \geq 2$).

$ c_{n+1} - c_n = (n+1)^2 - 10(n+1) - n^2 + 10n = 2n - 9$, o que é negativo
para $ n \leq 4$ e positivo para $ n \geq 5$: $(c_n)$ diminui até
$ c_5 = -25$, isso é \omterm{def:g10:functions:extrema}{mínimo}, então aumenta. Não é monótono.
\end{solution}

\begin{solution}{exo:g12:seq:3}
Termos dominantes de fator:
\[
u_n = \frac{n^2(3 - 1/n + 1/n^2)}{n^2(2 + 5/n^2)} \xrightarrow[n\to+\infty]{} \frac{3}{2}.
\]
Multiplique pelo conjugado:
\[
v_n = \frac{(n+1) - n}{\sqrt{n+1} + \sqrt{n}} = \frac{1}{\sqrt{n+1}+\sqrt{n}}
\xrightarrow[n\to+\infty]{} 0.
\]
Divida o numerador e o denominador por $3^n $:
\[
w_n = \frac{(2/3)^n - 1}{1 + (1/3)^n} \xrightarrow[n\to+\infty]{} \frac{0-1}{1+0} = -1,
\]
usando $\lim q^n = 0$ para $\abs{q} < 1$.
\end{solution}

\begin{solution}{exo:g12:seq:4}
$ u_n = 5 + 3n \to +\infty $ e $ v_n = 8 \cdot (1/2)^n = 2^{3-n} \to 0$.
As somas são
\[
\sum_{k=0}^{n} u_k = (n+1)\,\frac{5 + (5+3n)}{2} = \frac{(n+1)(10+3n)}{2}
\xrightarrow[n\to+\infty]{} +\infty,
\]
\[
\sum_{k=0}^{n} v_k = 8\,\frac{1 - (1/2)^{n+1}}{1 - 1/2}
= 16\left(1 - \left(\tfrac{1}{2}\right)^{n+1}\right)
\xrightarrow[n\to+\infty]{} 16 .
\]
\end{solution}

\begin{solution}{exo:g12:seq:5}
Desde $-1 \leq \cos n \leq 1$,
\[
\frac{n-1}{n+1} \leq \frac{n + \cos n}{n+1} \leq 1,
\]
e $\frac{n-1}{n+1} \to 1$, então o limite é $1$ pelo teorema do aperto.

Para o segundo limite, escreva
\[
0 \leq \frac{n!}{n^n}
= \frac{1}{n}\cdot\frac{2}{n}\cdots\frac{n}{n}
\leq \frac{1}{n},
\]
porque cada fator $\frac{k}{n}$ com $2 \leq k \leq n $ é no \omterm{def:g10:functions:extrema}{máximo} $1$.
Desde $\frac1n \to 0$, o teorema da compressão dá $\frac{n!}{n^n} \to 0$. (O
sugerido \omterm{def:g12:seq:arith-geom}{geométrico} vinculado também funciona: cada fator com $ k \leq n/2$ é no \omterm{def:g10:functions:extrema}{máximo}
$\frac12$, dando o limite mais forte $(1/2)^{\floor{n/2}}$.)
\end{solution}

\begin{solution}{exo:g12:seq:6}
\emph{1.} $ u_0 = 0 \in \intcc{0}{2}$. Se $0 \leq u_n \leq 2$, então
$2 \leq u_n + 2 \leq 4$, então $\sqrt{2} \leq u_{n+1} \leq 2$; em particular
$0 \leq u_{n+1} \leq 2$. Por indução a propriedade vale para todos $ n $.

\emph{2.} $ u_{n+1} - u_n = \sqrt{u_n + 2} - u_n $. Para $ x \in \intcc{0}{2}$,
$\sqrt{x+2} \geq x \iff x + 2 \geq x^2 \iff (2-x)(x+1) \geq 0$, o que é verdade.
Por isso $(u_n)$ é \omterm{def:g12:seq:monotonic}{aumentando}.

\emph{3.} \omterm{def:g12:seq:monotonic}{Aumentando} e \omterm{def:g12:seq:bounded}{limitado acima} por $2$, $(u_n)$ converge para alguns
$\ell \in \intcc{0}{2}$. Passando ao limite em $ u_{n+1} = \sqrt{u_n + 2}$
(o mapa $ x \mapsto \sqrt{x+2}$ é contínuo) dá $\ell = \sqrt{\ell + 2}$,
então $\ell^2 - \ell - 2 = 0$, ou seja\ $\ell \in \{-1, 2\}$. Desde $\ell \geq 0$,
$\lim u_n = 2$.
\end{solution}

\begin{solution}{exo:g12:seq:7}
\emph{1.} Entre duas doses, $40\%$ da droga é eliminada, então o
quantidade $ u_n $ torna-se $0.6\,u_n $; a próxima dose adiciona $1$ unidade:
$ u_{n+1} = 0.6\,u_n + 1$.

\emph{2.} $ v_{n+1} = u_{n+1} - 2.5 = 0.6\,u_n + 1 - 2.5 = 0.6(u_n - 2.5)
= 0.6\,v_n $: $(v_n)$ é \omterm{def:g12:seq:arith-geom}{geométrico} com proporção $0.6$ e primeiro mandato
$ v_0 = 1 - 2.5 = -1.5$. Por isso $ v_n = -1.5 \times 0.6^n $ e
\[
u_n = 2.5 - 1.5 \times 0.6^n .
\]

\emph{3.} Desde $0.6^n \to 0$, $ u_n \to 2.5$: a quantidade de droga estabiliza
em $2.5$ unidades.
\end{solution}

\begin{solution}{exo:g12:seq:8}
\emph{1.} $ u_0 = 3 > 1$. Se $ u_n > 1$, então $ u_n + 2 > 0$ e
\[
u_{n+1} - 1 = \frac{4u_n - 1 - u_n - 2}{u_n + 2} = \frac{3(u_n - 1)}{u_n + 2} > 0 .
\]
Por indução, $ u_n > 1$ para todos $ n $ (e em particular $ u_n + 2 \neq 0$, então
o \omterm{def:g12:seq:sequence}{sequência} está bem definido).

\emph{2.} Usando a identidade acima,
\[
v_{n+1} = \frac{1}{u_{n+1} - 1} = \frac{u_n + 2}{3(u_n - 1)}
= \frac{(u_n - 1) + 3}{3(u_n - 1)} = \frac{1}{3} + v_n .
\]
Então $(v_n)$ é \omterm{def:g12:seq:arith-geom}{aritmética} com \omterm{def:g11:seq:arithmetic}{diferença comum} $\frac13$ e
$ v_0 = \frac{1}{u_0 - 1} = \frac12$.

\emph{3.} $ v_n = \frac12 + \frac{n}{3}$, por isso
$ u_n = 1 + \frac{1}{v_n} = 1 + \frac{6}{3 + 2n}$. Desde $ v_n \to +\infty $,
$ u_n \to 1$.
\end{solution}

\begin{solution}{exo:g12:seq:9}
\emph{1.} $ H_{2n} - H_n = \sum_{k=n+1}^{2n} \frac{1}{k}$ é uma soma de $ n $
termos, cada um pelo menos $\frac{1}{2n}$; por isso
$ H_{2n} - H_n \geq n \cdot \frac{1}{2n} = \frac12$.

\emph{2.} Por indução em $ k $: $ H_{2^0} = H_1 = 1 \geq 1$. Se
$ H_{2^k} \geq 1 + \frac{k}{2}$, então aplicando o ponto 1 com $ n = 2^k $,
\[
H_{2^{k+1}} \geq H_{2^k} + \frac12 \geq 1 + \frac{k+1}{2}.
\]
O \omterm{def:g12:seq:sequence}{sequência} $(H_n)$ é \omterm{def:g12:seq:monotonic}{aumentando} (cada passo adiciona $\frac{1}{n+1} > 0$) e
a subsequência $ H_{2^k}$ é ilimitado, então $(H_n)$ não é \omterm{def:g12:seq:bounded}{limitado acima}.
\omterm{def:g12:seq:monotonic}{Aumentando} e ilimitado, tende a $+\infty $
(\cref{thm:g12:seq:monotone}).
\end{solution}

\begin{solution}{exo:g12:seq:10}
\emph{1.} O \omterm{def:g12:seq:sequence}{sequência} $ d_n = b_n - a_n $ satisfaz
$ d_{n+1} - d_n = (b_{n+1} - b_n) - (a_{n+1} - a_n) \leq 0$, então $(d_n)$ é
diminuindo; desde $ d_n \to 0$, obtemos $ d_n \geq 0$ para todos $ n $ (um diminuindo
\omterm{def:g12:seq:sequence}{sequência} com um termo negativo ficaria abaixo dele para sempre, evitando que o
limite $0$). Por isso $ a_n \leq b_n $.

\emph{2.} De $ a_n \leq b_n \leq b_0$, o \omterm{def:g12:seq:monotonic}{aumentando} \omterm{def:g12:seq:sequence}{sequência} $(a_n)$ é
\omterm{def:g12:seq:bounded}{limitado acima}, então converge para alguns $\ell $. De forma similar $(b_n)$, diminuindo
e limitado abaixo por $ a_0$, converge para alguns $\ell'$. Então
$\ell' - \ell = \lim (b_n - a_n) = 0$, então $\ell = \ell'$.

\emph{3.} $(a_n)$ é (estritamente) \omterm{def:g12:seq:monotonic}{aumentando} desde
$ a_{n+1} - a_n = \frac{1}{(n+1)!} > 0$. Para $(b_n)$,
\[
b_{n+1} - b_n = \frac{1}{(n+1)!} + \frac{1}{(n+1)(n+1)!} - \frac{1}{n\,n!}
= \frac{n(n+1) + n - (n+1)^2}{n(n+1)(n+1)!}
= \frac{-1}{n(n+1)(n+1)!} < 0 ,
\]
então $(b_n)$ é diminuindo. Finalmente $ b_n - a_n = \frac{1}{n\,n!} \to 0$. O
dois \omterm{def:g12:seq:sequence}{sequências} são adjacentes e, portanto, convergem para um limite comum.
\end{solution}

\begin{solution}{pb:g12:seq:1}
\textbf{1.} Verdadeiro para $ n = 1$ ($1 = 1^2$). Se
$1 + 3 + \dots + (2n - 1) = n^2$, então adicionando a próxima ímpar
número: $ n^2 + (2n + 1) = (n + 1)^2$: hereditariedade. Por indução,
verdade para todos $ n \geq 1$.

\textbf{2.} $2^0 = 1 > 0$. Se $2^n > n $, então
$2^{n+1} = 2 \cdot 2^n > 2n \geq n + 1$ para $ n \geq 1$ (e
$ n = 0$ verifica diretamente): hereditariedade, pronto.

\textbf{3.} $ n = 0$: $1 \geq 1$. Se
$(1 + x)^n \geq 1 + nx $, multiplique por $1 + x \geq 1 > 0$:
$(1 + x)^{n+1} \geq (1 + nx)(1 + x) = 1 + (n + 1)x + nx^2
\geq 1 + (n + 1)x $.

\textbf{4.} O passo de $ n = 1$ para $ n = 2$: entre dois
mármores, ``o primeiro $ n $'' e ``o último $ n $''são dois
\emph{disjunto} bolinhas de gude únicas --- nenhuma bolinha de gude comum conecta o
dois grupos, então nada obriga suas cores a concordarem. O
o argumento da hereditariedade exige silenciosamente que os dois grupos se sobreponham,
verdadeiro apenas de $ n \geq 2$; com o caso base $ n = 1$ a corrente
nunca começa.

\textbf{5.} $4^0 - 1 = 0 = 3 \times 0$. Se $4^n - 1 = 3k $,
então $4^{n+1} - 1 = 4(4^n - 1) + 3 = 3(4k + 1)$: hereditariedade.

\textbf{6.} $ x_1 = \frac32$, $ x_2 = \frac{17}{12}$,
$ x_3 = \frac{577}{408}$.

\textbf{7.} $ x_{n+1}^2 - 2 = \frac{(x_n^2 + 2)^2 - 8x_n^2}
{4x_n^2} = \frac{(x_n^2 - 2)^2}{4 x_n^2}$: um quadrado sobre um
positivo, portanto $\geq 0$, e $> 0$ em qualquer momento $ x_n^2 \neq 2$.
Indução: $ x_0 = 2 > 0$ com $ x_0^2 = 4 > 2$; se $ x_n > 0$
e $ x_n^2 > 2$, então $ x_{n+1}$ (uma média de positivos) é
positivo e $ x_{n+1}^2 - 2 > 0$.

\textbf{8.} $ x_{n+1} - x_n = \frac{2 - x_n^2}{2x_n} < 0$ por
questão 7: estritamente diminuindo.

\textbf{9.} Diminuindo e limitado abaixo (por $1$, desde
$ x_n^2 > 2 > 1$ e $ x_n > 0$): pela convergência monótona
teorema, $(x_n)$ converge para alguns $ L \geq 1$.

\textbf{10.} Os limites respeitam a álgebra: de
$ x_{n+1} = \frac12\left(x_n + \frac{2}{x_n}\right)$ e
$ x_n \to L \geq 1 > 0$:
$ L = \frac12\left(L + \frac2L\right)$, então $ L^2 = 2$ e, $ L $
sendo positivo, $ L = \sqrt2$. Veredicto: convergência comprovada,
limite identificado --- Heron absolvido com honras.

\textbf{11.} $ x_{n+1} - \sqrt2 = \frac{x_n^2 - 2\sqrt2\,x_n +
2}{2x_n} = \frac{(x_n - \sqrt2)^2}{2x_n}$: exatamente
$ e_{n+1} = \frac{e_n^2}{2x_n}$, e $ x_n > \sqrt2$ dá
$ e_{n+1} \leq \frac{e_n^2}{2\sqrt2}$. Erro quadrático: cada etapa
dobra o número de decimais corretos, conforme observado desde
9ª série.

\textbf{12.} $ e_0 \approx 0.5858$, $ e_1 \approx 0.0858$,
$ e_2 \approx 0.00245$, $ e_3 \approx 2.1 \times 10^{-6}$.
Razões $\frac{e_1}{e_0^2} \approx 0.25 = \frac{1}{2x_0}$;
$\frac{e_2}{e_1^2} \approx 0.333 = \frac{1}{2x_1}$;
$\frac{e_3}{e_2^2} \approx 0.353 \approx \frac{1}{2x_2}$: o
teorema em ação.

\textbf{13.} $ H_{2^{18}} \geq 1 + 9 = 10$: sobre
$260\,000$ termos ($2^{18} = 262\,144$) apenas para passar $10$ ---
divergência em um rastreamento (e $ H_n > 100$ precisaria de mais termos
do que átomos em qualquer biblioteca).

\textbf{14.} $ S_n = 2 - \frac{1}{2^n}$ (\omterm{def:g12:seq:arith-geom}{geométrico} soma) e
$\frac{1}{2^n} \to 0$ (\cref{thm:g12:seq:geometric}), então
$ S_n \to 2$: a barra de chocolate infinitamente mordida tende a,
sem nunca chegar, o todo --- agora no oficial
linguagem dos limites.

\textbf{15.} Para $ k \geq 2$:
$\frac{1}{k^2} \leq \frac{1}{k(k-1)} = \frac{1}{k-1} -
\frac1k $, então
$ S_n \leq 1 + \left(1 - \frac1n\right) < 2$: \omterm{def:g12:seq:monotonic}{aumentando} e
\omterm{def:g12:seq:bounded}{limitado acima}, portanto convergente (convergência monótona). Euler
mais tarde nomeou o limite: $\frac{\pi^2}{6}$.

\textbf{16.} Termos que tendem a $0$ é \emph{necessário} para o
somas para acertar, mas não decide nada: os termos harmônicos
$\frac1n \to 0$ ainda assim, as somas explodem; os termos
$\frac{1}{n^2} \to 0$ e as somas convergem. Como \emph{rápido}
os termos morrem é toda a questão --- a teoria das séries,
construído nos volumes universitários.

\textbf{17.} $ a_1 = \sqrt2 \approx 1.41421$, $ b_1 = 1.5$;
$ a_2 \approx 1.45648$, $ b_2 \approx 1.45711$: duas iterações
já concordo com três casas decimais --- velocidade impressionante.

\textbf{18.} $ b_{n+1} - a_{n+1} = \frac{a_n + b_n}{2} -
\sqrt{a_n b_n} = \frac{(\sqrt{b_n} - \sqrt{a_n})^2}{2} \geq
0$: significa permanecer ordenado. $(a_n)$ aumenta:
$ a_{n+1} = \sqrt{a_n b_n} \geq \sqrt{a_n \cdot a_n} = a_n $;
$(b_n)$ diminui simetricamente.

\textbf{19.} $\frac{b_{n+1} - a_{n+1}}{b_n - a_n} =
\frac{(\sqrt{b_n} - \sqrt{a_n})^2}
{2(\sqrt{b_n} - \sqrt{a_n})(\sqrt{b_n} + \sqrt{a_n})}
= \frac{\sqrt{b_n} - \sqrt{a_n}}{2(\sqrt{b_n} + \sqrt{a_n})}
\leq \frac12$: a diferença é pelo menos reduzida pela metade, então $ b_n - a_n \to 0$;
com a questão 18, o \omterm{def:g12:seq:sequence}{sequências} são adjacentes e compartilham um limite
$ M(1, 2)$.

\textbf{20.} A iteração três dá
$ a_3 \approx b_3 \approx 1.456791$: $ M(1, 2) \approx
1.456791$ em três voltas da manivela (a folga aproximadamente
\emph{quadrados}, como o de Heron). Classificação de velocidade do problema
\omterm{def:g12:seq:sequence}{sequências}, do mais lento para o mais rápido: as somas harmônicas (glacial
divergência), o \omterm{def:g12:seq:arith-geom}{geométrico} somas (erro reduzido pela metade em cada etapa),
Heron e a AGM (erro ao quadrado de cada passo) --- e foi o
A velocidade sobrenatural da AGM que disse a Gauss que ele havia encontrado uma nova veia
de análise.
\end{solution}
