\section*{Capítulo \ref{ch:g12:trigo} --- Funções trigonométricas}

\begin{solution}{exo:g12:trigo:1}
$\cos\frac{2\pi}{3} = \cos\left(\pi - \frac\pi3\right) = -\cos\frac\pi3
= -\frac12$.

$\sin\frac{5\pi}{6} = \sin\left(\pi - \frac\pi6\right) = \sin\frac\pi6
= \frac12$.

$\cos\frac{7\pi}{4} = \cos\left(2\pi - \frac{\pi}{4}\right)
= \cos\left(-\frac\pi4\right) = \frac{\sqrt2}{2}$.

$\sin\left(-\frac\pi3\right) = -\sin\frac\pi3 = -\frac{\sqrt3}{2}$.
\end{solution}

\begin{solution}{exo:g12:trigo:2}
$ f'(x) = 2\sin x\cos x = \sin 2x $ (regra da cadeia).

$ g'(x) = -3\sin(3x) + \sin x + x\cos x $ (regras de cadeia e produto).

Regra do quociente:
\[
h'(x) = \frac{\cos x\,(2 + \cos x) - \sin x\,(-\sin x)}{(2+\cos x)^2}
= \frac{2\cos x + \cos^2 x + \sin^2 x}{(2+\cos x)^2}
= \frac{2\cos x + 1}{(2+\cos x)^2}.
\]
\end{solution}

\begin{solution}{exo:g12:trigo:3}
\emph{(um)} Uma solução específica de $\sin x = \frac{\sqrt3}{2}$ é
$\frac\pi3$. Soluções gerais: $ x = \frac\pi3 + 2k\pi $ ou
$ x = \pi - \frac\pi3 + 2k\pi = \frac{2\pi}{3} + 2k\pi $. Em
$\intoc{-\pi}{\pi}$: $ x \in \left\{\frac\pi3, \frac{2\pi}{3}\right\}$.

\emph{(b)} $\cos 2x = \frac{\sqrt2}{2} = \cos\frac\pi4$ dá
$2x = \pm\frac\pi4 + 2k\pi $, então $ x = \pm\frac\pi8 + k\pi $. Em
$\intoc{-\pi}{\pi}$:
$ x \in \left\{-\frac{7\pi}{8}, -\frac\pi8, \frac\pi8, \frac{7\pi}{8}\right\}$.
\end{solution}

\begin{solution}{exo:g12:trigo:4}
Com $ a = \frac\pi3$, $ b = \frac\pi4$:
\[
\cos\frac{\pi}{12} = \cos a\cos b + \sin a \sin b
= \frac12\cdot\frac{\sqrt2}{2} + \frac{\sqrt3}{2}\cdot\frac{\sqrt2}{2}
= \frac{\sqrt2 + \sqrt6}{4},
\]
\[
\sin\frac{\pi}{12} = \sin a\cos b - \cos a \sin b
= \frac{\sqrt3}{2}\cdot\frac{\sqrt2}{2} - \frac12\cdot\frac{\sqrt2}{2}
= \frac{\sqrt6 - \sqrt2}{4}.
\]
\end{solution}

\begin{solution}{exo:g12:trigo:5}
Definir $ s = \sin x $: $2s^2 - s - 1 = (2s + 1)(s - 1) \geq 0$ se
$ s \leq -\frac12$ ou $ s = 1$.

$\sin x = 1$ dá $ x = \frac\pi2$. $\sin x \leq -\frac12$: na unidade
círculo, o arco abaixo da linha horizontal $ y = -\frac12$, que em
$\intcc{0}{2\pi}$ é $ x \in \intcc{\frac{7\pi}{6}}{\frac{11\pi}{6}}$.
Por isso
\[
S = \left\{\frac{\pi}{2}\right\} \cup
\intcc{\frac{7\pi}{6}}{\frac{11\pi}{6}} .
\]
\end{solution}

\begin{solution}{exo:g12:trigo:6}
$\dfrac{\sin 3x}{x} = 3\,\dfrac{\sin 3x}{3x} \to 3 \cdot 1 = 3$
(composição com $ u = 3x \to 0$).

Usando $1 - \cos x = \dfrac{\sin^2 x}{1 + \cos x}$:
\[
\frac{1 - \cos x}{x^2}
= \left(\frac{\sin x}{x}\right)^{\!2} \frac{1}{1 + \cos x}
\xrightarrow[x\to0]{} 1 \cdot \frac12 = \frac12 .
\]

Substituindo $ u = \frac1x \to 0^+$:
$ x \sin\frac1x = \frac{\sin u}{u} \to 1$.
\end{solution}

\begin{solution}{exo:g12:trigo:7}
$ f'(x) = 1 - \sin x \geq 0$ para todos $ x $, com igualdade exatamente quando
$\sin x = 1$, \emph{ou seja} $ x = \frac\pi2 + 2k\pi $. Sobre $\intcc{0}{2\pi}$, $ f $
é \omterm{def:g12:seq:monotonic}{aumentando} de $ f(0) = 1$ para $ f(2\pi) = 2\pi + 1$ (o zero de $ f'$ no
$\frac{\pi}{2}$ é isolado, então o aumento é até rigoroso); \omterm{def:g10:functions:extrema}{mínimo} $1$ no
$0$, \omterm{def:g10:functions:extrema}{máximo} $2\pi + 1$ no $2\pi $. Sobre $\R $, $ f' \geq 0$ em todos os lugares com apenas
zeros isolados, então $ f $ é (estritamente) \omterm{def:g12:seq:monotonic}{aumentando} por
\cref{thm:g12:deriv:variations}, embora $ f'$ desaparece a cada
$\frac\pi2 + 2k\pi $.
\end{solution}

\begin{solution}{exo:g12:trigo:8}
\emph{1.} Deixar $ R = \sqrt{A^2 + B^2} > 0$. O ponto
$\left(\frac AR, \frac BR\right)$ está no círculo unitário, então existe
$\varphi $ com $\cos\varphi = \frac AR $ e $\sin\varphi = \frac BR $. Então
\[
R\cos(x - \varphi) = R\bigl(\cos x\cos\varphi + \sin x \sin\varphi\bigr)
= A\cos x + B\sin x = f(x).
\]

\emph{2.} Desde $\cos $ tem alcance $\intcc{-1}{1}$, $ f $ tem \omterm{def:g10:functions:extrema}{máximo} $ R $ e
\omterm{def:g10:functions:extrema}{mínimo} $-R $. Para $\cos x + \sin x $: $ A = B = 1$, $ R = \sqrt2$,
$\varphi = \frac\pi4$, então o equação torna-se
$\sqrt2\cos\left(x - \frac\pi4\right) = 1$, \emph{ou seja}
$\cos\left(x - \frac\pi4\right) = \frac{\sqrt2}{2}$, dando
$ x - \frac\pi4 = \pm\frac\pi4 + 2k\pi $: em $\intoc{-\pi}{\pi}$,
$ x \in \left\{0, \frac\pi2\right\}$.
\end{solution}

\begin{solution}{exo:g12:trigo:9}
$(\sin)'' = (\cos)' = -\sin $ e $(\cos)'' = (-\sin)' = -\cos $: ambos resolvem
$ y'' = -y $.

Deixar $ g = y - a\cos - b\sin $. Então $ g'' = -g $ (linearidade de diferenciação),
$ g(0) = y(0) - a = 0$ e $ g'(0) = y'(0) - b = 0$. Definir
$ E = g^2 + (g')^2$. Então
\[
E' = 2g g' + 2g' g'' = 2g g' - 2g' g = 0,
\]
então $ E $ é constante igual a $ E(0) = 0$. Uma soma de dois quadrados desaparecendo
forças idênticas $ g = 0$, por isso $ y = a\cos + b\sin $.
\end{solution}

\begin{solution}{pb:g12:trigo:1}
\textbf{1.} $0.99833\ldots $ e $0.99998\ldots $: rastejando para
$1$.

\textbf{2.} Triângulo $ OAM $: básico $1$, altura $\sin x $: área
$\frac{\sin x}{2}$. Setor $ OAM $: fração
$\frac{x}{2\pi}$ do disco unitário: área $\frac x2$. Triângulo
$ OAT $: básico $1$, altura $\tan x $: área $\frac{\tan x}{2}$.

\textbf{3.} O triângulo fica dentro do setor dentro do grande
triângulo: $\frac{\sin x}{2} < \frac x2 < \frac{\tan x}{2}$,
ou seja\ $\sin x < x < \tan x $. De $\sin x < x $:
$\frac{\sin x}{x} < 1$; e de
$ x < \tan x = \frac{\sin x}{\cos x}$, multiplicando por
$\frac{\cos x}{x} > 0$: $\cos x < \frac{\sin x}{x}$. Como
$ x \to 0^+$, $\cos x \to 1$: espremido,
$\frac{\sin x}{x} \to 1$; e $\frac{\sin x}{x}$ é par, então
o limite bilateral é $1$.

\textbf{4.} $\frac{1 - \cos x}{x^2} =
\frac{1 - \cos^2 x}{x^2 (1 + \cos x)} =
\left(\frac{\sin x}{x}\right)^2 \frac{1}{1 + \cos x}
\to 1 \times \frac12$.

\textbf{5.} $\sin'(0) = \lim_{h \to 0} \frac{\sin h - 0}{h} =
1$: toda a diferenciação de \omterm{def:g12:trigo:cossin}{seno} e \omterm{def:g11:trigo:cossin}{cosseno} flui de
este limite. Medido em graus, o limite seria
$\frac{\pi}{180}$, e cada derivado arrastaria isso
constante: \omterm{def:g11:trigo:radian}{radianos} são exatamente a unidade que faz o cálculo
de oscilação limpa.

\textbf{6.} $ y' = -R\omega\sin(\omega t - \varphi)$,
$ y'' = -R\omega^2\cos(\omega t - \varphi) = -\omega^2 y $.

\textbf{7.} $\omega = 2$. $ y(0) = A = 3$;
$ y'(0) = 2B = 8$: $ B = 4$: $ y = 3\cos 2t + 4\sin 2t $.

\textbf{8.} $ R = \sqrt{9 + 16} = 5$; período
$\frac{2\pi}{2} = \pi $; fase
$\varphi = \arctan\frac43 \approx 0.927$:
$ y = 5\cos(2t - 0.927)$.

\textbf{9.} $5\cos(2t - \varphi) = 0$ primeiro quando
$2t - \varphi = \frac\pi2$:
$ t = \frac{\pi/2 + 0.927}{2} \approx 1.25$ S.

\textbf{10.} Com $\sin\theta \approx \theta $:
$\theta'' = -\frac gL \theta $: harmônico com
$\omega = \sqrt{\frac gL}$, período
$ T = \frac{2\pi}{\omega} = 2\pi\sqrt{\frac Lg}$. Para
$ L = 1$: $ T \approx 2.006$ S. O período não depende
a amplitude (para pequenas oscilações) --- isocronismo de Galileu,
o fato que torna possíveis os relógios de pêndulo.

\textbf{11.} $ L = g\left(\frac{T}{2\pi}\right)^2 =
\frac{9.81}{\pi^2} \approx 0.994$ m: o pêndulo dos segundos é
sobre $6$ mm aquém do metro meridiano. A Academia escolheu
o meridiano (a gravidade varia de lugar para lugar, traindo
o pêndulo); tive $ g $ fosse um pouco maior, nosso medidor seria
carrapato.

\textbf{12.} $\cos\left(x + \frac\pi2\right) =
\cos x \cos\frac\pi2 - \sin x \sin\frac\pi2 = -\sin x $;
$\sin\left(x + \frac\pi2\right) = \cos x $. Tão diferenciador
\omterm{def:g12:trigo:cossin}{seno} dá \omterm{def:g12:trigo:cossin}{seno} mudou um quarto de volta ($\cos $) e novamente
($-\sin $), e novamente, e para casa em quatro etapas: diferenciação
gira o círculo.

\textbf{13.} Adicionando as duas expansões de $\cos(a \mp b)$:
$\cos(a-b) + \cos(a+b) = 2\cos a\cos b $. Com $ a = b = x $:
$1 + \cos 2x = 2\cos^2 x $, ou seja\
$\cos^2 x = \frac{1 + \cos 2x}{2}$.

\textbf{14.} Com $ p = \frac{p+q}{2} + \frac{p-q}{2}$ e
$ q $ seu espelho, expandindo e adicionando:
$\sin p + \sin q = 2\sin\frac{p+q}{2}\cos\frac{p-q}{2}$. Para
$440$ e $444$ Hz: um tom em $442$ Hz cujo volume é
modulado por $\cos(2\pi \cdot 2t)$ --- os picos de volume $4$
vezes por segundo (o envelope \omterm{def:g10:numbers:abs}{valor absoluto}): quatro batidas
um segundo, desaparecendo conforme os garfos concordam.

\textbf{15.} $\cos 3x = \cos 2x\cos x - \sin 2x \sin x =
(2\cos^2 x - 1)\cos x - 2\sin^2 x\cos x = 4\cos^3 x -
3\cos x $. No $ x = 20^\circ $: $\cos 60^\circ = \frac12$, então
$ c = \cos 20^\circ $ satisfaz $8c^3 - 6c - 1 = 0$ --- um cúbico
sem \omterm{def:g11:quad:discriminant}{raiz} quadrada solução: construir $20^\circ $, ou seja\
trissecção $60^\circ $, está além de régua e compasso.

\textbf{16.} $ R = \sqrt{9 + 27} = 6$;
$\tan\varphi = \frac{3\sqrt3}{3} = \sqrt3$:
$\varphi = \frac\pi3$: $ I(t) = 6\cos\left(100\pi t -
\frac\pi3\right)$: pico $6$ amperes, atraso de fase $60^\circ $.

\textbf{17.} $\sin^2 t = \frac{1 - \cos 2t}{2}$: período
$\pi $. $\sin 2t $ tem período $\pi $, $\sin 3t $ tem
$\frac{2\pi}{3}$: a soma se repete após o menor comum
múltiplo, $2\pi $.

\textbf{18.} Uma oscilação de período $1$ dentro do encolhimento
envelope $\pm\eu^{-t/10}$: cada balanço sobre $10\,\%$ mais baixo. No
$ t = 10$ o envelope é $\eu^{-1} \approx 0.368$ --- o
constante de \cref{pb:g12:exp:1} mais uma vez. Equações com tal
soluções amortecidas ($ y'' + a y' + b y = 0$) são resolvidos no
diferencial equações capítulo.

\textbf{19.} Uma força que recua proporcionalmente à
deslocamento dá $ y'' = -\omega^2 y $; e
\cref{exo:g12:trigo:9} mostra as soluções disso equação
são \emph{exatamente} as combinações de $\cos $ e $\sin $ ---
existência pela exposição, singularidade pelo exercício: vibração
não tem escolha a não ser ser sinusoidal.

\textbf{20.} Um limite ($\frac{\sin x}{x} \to 1$, por
áreas imprensadas), dois derivados ($\sin' = \cos $,
$\cos' = -\sin $), um equação ($ y'' = -\omega^2 y $), um mundo
de relógios, cordas, correntes e marés. Asterisco: para largura
balanços, $ T = 4\sqrt{\frac Lg} \cdot
\frac{\pi/2}{M(1, \cos(\theta_0/2))}$ --- o
\omterm{def:g12:seq:arith-geom}{aritmética} - geométrica significar de \cref{pb:g12:seq:1} calcula o
integral elíptica em algumas iterações: o diário de Gauss
entrada e o pêndulo de Galileu, com uma fórmula à parte.
\end{solution}
