\begin{solution}{exo:g12:matrix:4}
\emph{1.} $ N^2 = \begin{pmatrix} 0&1\\0&0\end{pmatrix}
\begin{pmatrix} 0&1\\0&0\end{pmatrix} = 0$, e $ I $ viaja com cada
matriz.

\emph{2.} Como os dois termos comutam, o teorema binomial se aplica e todos
termos contendo $ N^2$ desaparecer:
\[
A^k = (2I + N)^k = 2^k I + k\,2^{k-1} N
= \begin{pmatrix} 2^k & k\,2^{k-1}\\ 0 & 2^k \end{pmatrix}.
\]
Verifique se há $ k = 2$:
$ A^2 = \begin{pmatrix} 2&1\\0&2\end{pmatrix}^2
= \begin{pmatrix} 4&4\\0&4\end{pmatrix}$, e a fórmula dá
$2^2 = 4$, $2 \times 2 = 4$. \checkmark
\end{solution}

\begin{solution}{exo:g12:matrix:5}
\emph{Indução.} Para $ n = 1$:
$ A^1 = \begin{pmatrix} 0&1\\1&1\end{pmatrix}
= \begin{pmatrix} F_0 & F_1\\ F_1 & F_2\end{pmatrix}$. Suponha a fórmula
para $ n $; então
\[
A^{n+1} = A^n A
= \begin{pmatrix} F_{n-1} & F_n\\ F_n & F_{n+1}\end{pmatrix}
\begin{pmatrix} 0 & 1\\ 1 & 1\end{pmatrix}
= \begin{pmatrix} F_n & F_{n-1} + F_n\\ F_{n+1} & F_n + F_{n+1}\end{pmatrix}
= \begin{pmatrix} F_n & F_{n+1}\\ F_{n+1} & F_{n+2}\end{pmatrix}.
\]
\emph{Identidade.} Para $2\times2$ matrizes, expandindo mostra
$\det(MN) = \det M \det N $; por isso
$\det(A^n) = (\det A)^n = (-1)^n $, e
$\det A^n = F_{n-1}F_{n+1} - F_n^2$. (Isso é \emph{Identidade da Cassini}.)
\end{solution}

\begin{solution}{exo:g12:matrix:6}
\emph{1.} Ordenando vértices $1, 2, 3$:
\[
M = \begin{pmatrix} 0&1&1\\ 0&0&1\\ 1&0&0\end{pmatrix}, \quad
M^2 = \begin{pmatrix} 1&0&1\\ 1&0&0\\ 0&1&1\end{pmatrix}, \quad
M^3 = \begin{pmatrix} 1&1&1\\ 1&0&1\\ 1&0&1 \end{pmatrix}.
\]

\emph{2.} $\bigl(M^3\bigr)_{11} = 1$: exatamente um passeio fechado de comprimento $3$
no vértice $1$, nomeadamente $1 \to 2 \to 3 \to 1$. (A caminhada
$1 \to 3 \to 1$ tem comprimento $2$ apenas, e $1 \to 3$ então $3\to1$ então
$1\to3$ termina em $3$.)
\end{solution}

\begin{solution}{exo:g12:matrix:7}
\emph{1.} $ a_{n+1} = 0.8a_n + 0.3b_n $, $ b_{n+1} = 0.2a_n + 0.7b_n $:
$ M = \begin{pmatrix} 0.8 & 0.3\\ 0.2 & 0.7\end{pmatrix}$.

\emph{2.} $ MX = X $ dá $0.8a + 0.3b = a $, \emph{ou seja} $0.3b = 0.2a $, então
$ b = \frac23 a $; com $ a + b = 1$: $ a = 0.6$, $ b = 0.4$.

\emph{3.} Usando $ b_n = 1 - a_n $:
$ a_{n+1} = 0.8a_n + 0.3(1 - a_n) = 0.5a_n + 0.3$, então
\[
c_{n+1} = a_{n+1} - 0.6 = 0.5a_n + 0.3 - 0.6 = 0.5(a_n - 0.6) = 0.5\,c_n .
\]
Por isso $ c_n = 0.5^n c_0 \to 0$: $ a_n \to 0.6$ e $ b_n \to 0.4$, seja qual for o
inicial \omterm{def:g11:prob:rv}{distribuição}.
\end{solution}

\begin{solution}{exo:g12:matrix:8}
\emph{1.} $\det P = -2$, então
$ P^{-1} = -\frac12\begin{pmatrix} -1 & -1\\ -1 & 1\end{pmatrix}
= \frac12\begin{pmatrix} 1 & 1\\ 1 & -1\end{pmatrix}$. Então
\[
AP = \begin{pmatrix} 4 & 2\\ 4 & -2 \end{pmatrix}, \qquad
D = P^{-1}AP = \frac12\begin{pmatrix} 1&1\\1&-1\end{pmatrix}
\begin{pmatrix} 4&2\\4&-2\end{pmatrix}
= \begin{pmatrix} 4 & 0\\ 0 & 2\end{pmatrix}.
\]

\emph{2.} De $ A = PDP^{-1}$, uma indução imediata dá
$ A^n = PD^nP^{-1}$ com
$ D^n = \begin{pmatrix} 4^n & 0\\ 0 & 2^n\end{pmatrix}$, então
\[
A^n = P D^n P^{-1}
= \begin{pmatrix} 4^n & 2^n\\ 4^n & -2^n\end{pmatrix}\cdot
\frac12\begin{pmatrix} 1&1\\1&-1\end{pmatrix}
= \frac12\begin{pmatrix} 4^n + 2^n & 4^n - 2^n\\
4^n - 2^n & 4^n + 2^n\end{pmatrix}.
\]
Candidatura $ A^n $ para $ X_0 = \begin{pmatrix} u_0\\v_0\end{pmatrix}$ reproduz
exatamente as fórmulas de \cref{ex:g12:matrix:coupled}.
\end{solution}

\begin{solution}{pb:g12:matrix:1}
\textbf{1.} $ AB = \begin{pmatrix} 2 & 1\\ 4 & 3\end{pmatrix}$
e $ BA = \begin{pmatrix} 3 & 4\\ 1 & 2\end{pmatrix}$: matriz
multiplicação não é comutativa --- $ B $ troca colunas no
direita, linhas à esquerda.

\textbf{2.} \omterm{def:g11:vect:det}{Determinante} $6 - 5 = 1$: inverso
$\begin{pmatrix} 3 & -1\\ -5 & 2\end{pmatrix}$. Aplicando-o a
$\binom{4}{7}$: $ x = 12 - 7 = 5$, $ y = -20 + 14 = -6$.

\textbf{3.} $ N^2 = 0$. Então $(I + N)^n = I + nN $ por indução:
$(I + nN)(I + N) = I + (n+1)N + nN^2 = I + (n+1)N $.

\textbf{4.} $ A = \begin{pmatrix} 0&1&1\\ 1&0&1\\ 1&1&0
\end{pmatrix}$, e $ A^3$ tem entradas diagonais $2$: de cada
vértice, exatamente dois caminhos fechados de comprimento $3$ (o triângulo
andou no sentido horário ou anti-horário) --- o teorema da contagem
em ação.

\textbf{5.} $ D^n = \begin{pmatrix} 2^n & 0\\ 0 & 2^{-n}
\end{pmatrix}$: uma direção explode, a outra morre ---
destinos diagonais são independentes \omterm{def:g12:seq:sequence}{sequências} geométricas.

\textbf{6.} $ F^2 = \begin{pmatrix} 2 & 1\\ 1 & 1\end{pmatrix}$,
$ F^3 = \begin{pmatrix} 3 & 2\\ 2 & 1\end{pmatrix}$,
$ F^4 = \begin{pmatrix} 5 & 3\\ 3 & 2\end{pmatrix}$: Fibonacci
em todos os lugares; conjectura conforme declarado.

\textbf{7.} Se $ F^n = \begin{pmatrix} F_{n+1} & F_n\\ F_n &
F_{n-1}\end{pmatrix}$, então
\[
F^{n+1} = F^n F =
\begin{pmatrix} F_{n+1} + F_n & F_{n+1}\\
F_n + F_{n-1} & F_n \end{pmatrix}
= \begin{pmatrix} F_{n+2} & F_{n+1}\\ F_{n+1} & F_n
\end{pmatrix} :
\]
hereditariedade; o caso base $ n = 1$ é $ F $ em si, usando o
convenção $ F_0 = 0$ (que estende a recorrência para trás).

\textbf{8.} $\det F = -1$, então
$\det(F^n) = (\det F)^n = (-1)^n $; e diretamente
$\det(F^n) = F_{n+1}F_{n-1} - F_n^2$: Cassini, uma linha. (O
regra do produto para $2 \times 2$ \omterm{def:g11:vect:det}{determinantes} é um agradável
expansão de cinco minutos.)

\textbf{9.} Canto superior direito de $ F^m F^n $:
$ F_{m+1}F_n + F_m F_{n-1}$; canto superior direito de $ F^{m+n}$:
$ F_{m+n}$. Para $ m = n = 3$:
$ F_4 F_3 + F_3 F_2 = 3 \times 2 + 2 \times 1 = 8 = F_6$.

\textbf{10.} Para $ k = 1$: trivial. Se $ F_n \mid F_{kn}$, o
fórmula de adição com $ m = kn $:
$ F_{(k+1)n} = F_{kn+1}F_n + F_{kn}F_{n-1}$: ambos os termos são
múltiplos de $ F_n $. Então $ F_n \mid F_{kn}$ para todos $ k $: verificar
$ F_3 = 2$ \omterm{def:g12:arith:divides}{divide} $ F_6 = 8$ e $ F_9 = 34$.

\textbf{11.} $ F^{100} = F^{64} F^{32} F^4$: sete quadraturas
($ F^2, F^4, \dots, F^{64}$) mais duas combinações --- nove
multiplicações em vez de noventa e nove. É o
truque da mesa duplicada dos escribas egípcios, tirado de
números em matrizes: escreva $100$ em binário, multiplique o
duplicações que você precisa.

\textbf{12.} $0.8 + 0.2 = 1$ e $0.4 + 0.6 = 1$: amanhã
deve ser \emph{alguns} clima --- cada coluna é completa
\omterm{def:g10:proba:distribution}{probabilidade} \omterm{def:g11:prob:rv}{distribuição}, então as \omterm{def:g10:proba:distribution}{probabilidades} são conservadas.

\textbf{13.} Amanhã: $\binom{0.8}{0.2}$. Dia seguinte:
$ M\binom{0.8}{0.2} = \binom{0.72}{0.28}$.

\textbf{14.} $ Mv = v $ com $ v = \binom{s}{r}$, $ s + r = 1$:
$0.8s + 0.4r = s $ dá $0.4r = 0.2s $, $ s = 2r $:
$ v = \binom{2/3}{1/3}$. A longo prazo, dois dias em cada três são
ensolarado --- seja qual for a aparência de hoje.

\textbf{15.} De $\binom01$: componentes ensolarados $0.4$, $0.56$,
$0.624$, $0.6496$; lacunas para $\frac23$: $0.267$, $0.107$,
$0.043$, $0.017$ --- cada etapa multiplica a lacuna por exatamente
$0.4$ (o segundo autovalor da máquina): \omterm{def:g12:seq:arith-geom}{geométrico}
convergência para o estado estacionário.

\textbf{16.} Colunas (de $ A $, $ B $, $ C $):
$ P = \begin{pmatrix} 0 & 0 & 1\\ \frac12 & 0 & 0\\
\frac12 & 1 & 0\end{pmatrix}$. Curso estável: $ v_A = v_C $,
$ v_B = \frac{v_A}{2}$, $ v_C = \frac{v_A}{2} + v_B $; resumindo para
$1$: $ v = \left(\frac25, \frac15, \frac25\right)$. Classificação:
$ A $ e $ C $ empate em primeiro lugar, $ B $ durar.

\textbf{17.} $ C $ recebe \emph{todos} de $ B $ é tráfego e metade
de $ A $'s, e canaliza tudo de volta para $ A $: o constante
medidas estaduais onde o surfista \emph{passa tempo}, não como
muitos links apontam para --- um link de uma página popular supera
vários dos desertos. Essa ponderação recursiva é
precisamente a ideia fundadora do Google; amortecimento alças armadilhas para aranhas
e becos sem saída.

\textbf{18.} $ A^n = P D^n P^{-1}$ dá
$ u_n = \frac{4^n + 2^n}{2}$ (e
$ v_n = \frac{4^n - 2^n}{2}$). Verificar: $ u_1 = 3$, $ u_2 = 10$,
$ u_3 = 36$; diretamente: $(1,0) \to (3,1) \to (10,6) \to
(36, 28)$: correspondência.

\textbf{19.} A diagonalização muda para \omterm{def:g10:coordgeom:system}{coordenadas} em que
o sistema acoplado se desintegra em independente \omterm{def:g12:seq:arith-geom}{geométrico}
\omterm{def:g12:seq:sequence}{sequências} --- cada autovalor executa sua própria corrida. O Markov
estado estacionário é o autovetor do autovalor $1$, e o
taxa de convergência da questão 15 é o próximo autovalor: o
a máquina meteorológica sempre foi uma história própria.

\textbf{20.} Escrituração contábil: um sistema é um só matriz equação,
resolvido por um inverso. Combinatória: poderes do adjacência
matriz conte caminhadas, links, conexões. Evolução: poderes de
a máquina carrega estados para seus destinos, e o
direções próprias (direção áurea de Fibonacci,
estado estacionário, a classificação da web vetor) são os destinos.
A álgebra linear, nos volumes universitários, é a ciência da
exatamente isso.
\end{solution}
