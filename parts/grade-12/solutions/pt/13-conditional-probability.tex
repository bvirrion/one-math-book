\section*{Capítulo \ref{ch:g12:condprob} --- Probabilidade Condicional e Independência}

\begin{solution}{exo:g12:condprob:1}
Há $12$ cartas de figuras, das quais $4$ são reis:
$\pcond{\text{face}}{\text{king}} = \frac{4}{12} = \frac13$.

Independência: $\P(\text{king} \cap \text{heart}) = \frac{1}{52}$ (o rei
de corações), e
$\P(\text{king})\,\P(\text{heart}) = \frac{4}{52}\times\frac{13}{52}
= \frac{1}{52}$. Igual: o \omterm{def:g11:prob:model}{eventos} são \omterm{def:g12:condprob:indep}{independente}.
\end{solution}

\begin{solution}{exo:g12:condprob:2}
\emph{1.} Primeiro ramo: vermelho $\frac58$, azul $\frac38$; segundos ramos
(sem reposição): depois do vermelho, vermelho $\frac47$ / azul $\frac37$; depois
azul, vermelho $\frac57$ / azul $\frac27$.

\emph{2.} $\P(RR) = \frac58 \times \frac47 = \frac{5}{14}$. Por total
\omterm{def:g10:proba:distribution}{probabilidade},
\[
\P(\text{2nd red}) = \frac58\cdot\frac47 + \frac38\cdot\frac57
= \frac{20 + 15}{56} = \frac58 .
\]
(O mesmo que no primeiro sorteio --- por simetria, a segunda bola é uniformemente
bola aleatória da urna.)
\end{solution}

\begin{solution}{exo:g12:condprob:3}
$\P(A) = \frac{6}{36} = \frac16$ (seis pares somam $7$),
$\P(B) = \frac16$, e $ A \cap B = \{(3,4)\}$ tem \omterm{def:g10:proba:distribution}{probabilidade}
$\frac{1}{36} = \P(A)\P(B)$: $ A $ e $ B $ são \omterm{def:g12:condprob:indep}{independente}. (A soma $7$ é
especial: seja qual for o primeiro dado, exatamente um valor do segundo o fornece.)

$\P(C) = \frac{5}{36}$ e $ B \cap C = \{(3,3)\}$:
$\P(B \cap C) = \frac{1}{36} \neq \frac16 \times \frac{5}{36}
= \frac{5}{216}$. Não \omterm{def:g12:condprob:indep}{independente}.
\end{solution}

\begin{solution}{exo:g12:condprob:4}
\emph{1.} Total \omterm{def:g10:proba:distribution}{probabilidade} com a partição por máquina:
\[
\P(D) = 0.5\times0.01 + 0.3\times0.02 + 0.2\times0.04
= 0.005 + 0.006 + 0.008 = 0.019 = 1.9\% .
\]

\emph{2.} Bayes:
$\pcond{D}{M_3} = \dfrac{0.008}{0.019} = \dfrac{8}{19} \approx 0.42$. O
máquina que produz apenas um quinto da produção é responsável por mais de $40\%$ de
os defeitos.
\end{solution}

\begin{solution}{exo:g12:condprob:5}
A condição é
\[
\frac{0.99\,p}{0.99\,p + 0.05\,(1-p)} \geq 0.9 .
\]
O denominador é positivo, então isso diz
$0.99p \geq 0.891p + 0.045 - 0.045p $, \emph{ou seja} $0.144\,p \geq 0.045$,
\emph{ou seja}
\[
p \geq \frac{0.045}{0.144} = 0.3125 .
\]
O teste por si só atinge $90\%$ certeza apenas se a doença já afetar
mais do que $31\%$ da população testada --- é por isso que o rastreio em massa
das doenças raras exige testes de confirmação.
\end{solution}

\begin{solution}{exo:g12:condprob:6}
\emph{1.} Três \omterm{def:g12:seq:sequence}{sequências} entender $ E $ (HHT, HTH, THH), cada um dos \omterm{def:g10:proba:distribution}{probabilidade}
$ p^2(1-p)$ por independência: $\P(E) = 3p^2(1 - p)$.

\emph{2.} $ f(p) = 3p^2 - 3p^3$ tem $ f'(p) = 6p - 9p^2 = 3p(2 - 3p)$,
positivo e depois negativo $\intoo{0}{1}$: \omterm{def:g10:functions:extrema}{máximo} no $ p = \frac23$, onde
$\P(E) = 3\cdot\frac49\cdot\frac13 = \frac49$.
\end{solution}

\begin{solution}{exo:g12:condprob:7}
Digamos que você escolheu a porta 1 e o anfitrião abriu a porta 3 (\omterm{def:g11:prob:model}{evento} $ H_3$). Com
$ B_i $ = ``prêmio atrás da porta $ i $'', $\P(B_i) = \frac13$ e
\[
\pcond{B_1}{H_3} = \frac12, \qquad
\pcond{B_2}{H_3} = 1, \qquad
\pcond{B_3}{H_3} = 0
\]
(se o prêmio estiver atrás da sua porta, o anfitrião escolhe entre as portas 2 e 3
aleatoriamente; se estiver atrás da porta 2, ele é forçado a abrir 3). Bayes:
\[
\pcond{H_3}{B_1}
= \frac{\frac13\cdot\frac12}{\frac13\cdot\frac12 + \frac13\cdot1 + 0}
= \frac{1/6}{1/2} = \frac13,
\qquad
\pcond{H_3}{B_2} = \frac{\frac13}{\frac12} = \frac23 .
\]
Ficar ganha com \omterm{def:g10:proba:distribution}{probabilidade} $\frac13$, alternando com \omterm{def:g10:proba:distribution}{probabilidade}
$\frac23$.
\end{solution}

\begin{solution}{exo:g12:condprob:8}
\emph{1.} A maioria erra quando pelo menos duas das três cópias são
invertido:
\[
3\varepsilon^2(1-\varepsilon) + \varepsilon^3
= 3(0.01)(0.9) + 0.001 = 0.028,
\]
muito menor que $\varepsilon = 0.1$: codificação de repetição funciona.

\emph{2.} Se $1$ foi enviado (como $111$), recebendo $101$ requer exatamente um
virar: \omterm{def:g10:proba:distribution}{probabilidade} $\varepsilon(1-\varepsilon)^2 = 0.081$. Se $0$ foi enviado
($000$), são necessários dois lançamentos: $\varepsilon^2(1-\varepsilon) = 0.009$.
Bayes com anteriores iguais:
\[
\pcond{101}{\,\text{sent }1} = \frac{0.081}{0.081 + 0.009} = 0.9 .
\]
A decodificação majoritária ($101 \mapsto 1$) é de fato o palpite mais provável.
\end{solution}

\begin{solution}{pb:g12:condprob:1}
\textbf{1.} $\P(\text{red}) = \frac12 \cdot \frac23 + \frac12
\cdot \frac14 = \frac13 + \frac18 = \frac{11}{24}$.

\textbf{2.} $\P(\text{I} \mid \text{red}) =
\dfrac{\frac12 \cdot \frac23}{\frac{11}{24}} =
\dfrac{1/3}{11/24} = \dfrac{8}{11}$.

\textbf{3.} $\P(\text{even} \cap \leq 4) = \P(\{2, 4\}) =
\frac13$ e $\P(\text{even})\P(\leq 4) = \frac12 \cdot
\frac23 = \frac13$: \omterm{def:g12:condprob:indep}{independente} (inesperadamente!). Para
$\leq 3$: $\P(\{2\}) = \frac16$ contra
$\frac12 \cdot \frac12 = \frac14$: dependente.

\textbf{4.} O \omterm{def:g10:proba:operations}{complementos} de
\omterm{def:g12:condprob:indep}{independente}
\omterm{def:g11:prob:model}{eventos} são eles mesmos
\omterm{def:g12:condprob:indep}{independente}, então
\[
\P(\text{none}) = 0.1 \times 0.1 = 0.01,
\qquad
\P(\text{at least one}) = 0.99 .
\]

\textbf{5.} $\frac{4}{52} \times \frac{3}{51} =
\frac{1}{221}$.

\textbf{6.} Doente: $100$; saudável: $99\,900$. Positivos:
$99$ verdadeiro (99\,\% de $100$) e $4\,995$ falso (5\,\% de
$99\,900$): $5\,094$ positivos em tudo.

\textbf{7.} $\P(\text{sick} \mid +) = \frac{99}{5\,094}
\approx 1.9\,\%$: das pessoas sinalizadas pelo teste, menos de
um em cada cinquenta está realmente doente.

\textbf{8.} $\dfrac{0.001 \times 0.99}{0.001 \times 0.99 +
0.999 \times 0.05} = \dfrac{0.00099}{0.05094} \approx 0.019$:
as contagens e a fórmula concordam.

\textbf{9.} Confuso: $\P(+ \mid \text{sick}) = 99\,\%$
(a qualidade anunciada do teste) com
$\P(\text{sick} \mid +)$ (a pergunta real do paciente). Com
uma taxa básica de $\frac{1}{1000}$, os doentes são tão raros que
até mesmo um $5\,\%$ vazamento da enorme maioria saudável inunda o
verdadeiros positivos cinquenta para um.

\textbf{10.} $\dfrac{0.019 \times 0.99}{0.019 \times 0.99 +
0.981 \times 0.05} \approx 28\,\%$. Cada \omterm{def:g12:condprob:indep}{independente} positivo
multiplica as \omterm{def:g10:proba:distribution}{probabilidades} pelo mesmo fator: as evidências se acumulam,
e um terceiro positivo ultrapassaria $90\,\%$ --- o que é
por que existem testes confirmatórios.

\textbf{11.} $\dfrac{0.05 \times 0.99}{0.05 \times 0.99 +
0.95 \times 0.05} \approx 51\,\%$: em um grupo de alto risco o
mesmo teste é informativo. Moral: tela onde está a taxa básica
é substancial; triagem em massa uma condição rara fabrica
alarmes falsos aos milhares.

\textbf{12.} Partidas inocentes esperadas:
$10^7 \times 10^{-6} = 10$. Entre os aproximadamente $11$ correspondência
pessoas ($10$ inocente, $1$ culpado --- se o culpado estiver em
cidade), o suspeito encontrou \emph{pela partida sozinho} é
inocente com \omterm{def:g10:proba:distribution}{probabilidade} sobre $\frac{10}{11} \approx
91\,\%$.

\textbf{13.} O promotor citou
$\P(\text{match} \mid \text{innocent}) = 10^{-6}$; o tribunal
necessário $\P(\text{innocent} \mid \text{match})$ --- pergunta
A confusão de 9 sob juramento. (E no caso real isso
popularizou esses erros, um segundo erro agravou-os:
veja a pergunta 14.)

\textbf{14.} Quadratura ingênua: $\left(\frac{1}{8500}\right)^2
\approx \frac{1}{72 \text{ million}}$. Com o segundo
tragédia cinco vezes mais provável dada a primeira:
$\frac{1}{8500} \times \frac{5}{8500} \approx
\frac{1}{14.5 \text{ million}}$ --- cinco vezes maior, e
ainda ignorando que \emph{alternativa} hipótese deve ser
pesado por Bayes também. Lição: multiplicando \omterm{def:g10:proba:distribution}{probabilidades}
requer independênciae causas compartilhadas (genética,
ambiente) destruí-lo.

\textbf{15.} Bayes multiplica as razões de verossimilhança de
\emph{todos} as evidências em um posterior: a partida, o
álibi, as fibras, cada uma mudando as \omterm{def:g10:proba:distribution}{probabilidades}. O jurado
sentença: \emph{pergunte quão provável é a evidência sob ambos
hipóteses, e nunca confunda aquela condicional com sua
reverter}.

\textbf{16.} $\dfrac{0.4 \times 0.15}{0.4 \times 0.15 +
0.6 \times 0.005} = \dfrac{0.060}{0.063} \approx 95\,\%$
spam: uma palavra quase resolve tudo.

\textbf{17.} Cinco cópias falham quando $3$, $4$ ou $5$ virar:
$\binom53 \varepsilon^3(1-\varepsilon)^2 + \binom54
\varepsilon^4 (1 - \varepsilon) + \varepsilon^5 \approx
0.0086$ --- contra $0.028$ para três cópias: repetição
compra confiabilidade ao preço da largura de banda.

\textbf{18.} Até vira: $\binom40 0.9^4 + \binom42 0.9^2
0.1^2 + \binom44 0.1^4 = 0.6561 + 0.0486 + 0.0001 = 0.7048$;
e $\frac{1 + 0.8^4}{2} = \frac{1 + 0.4096}{2} = 0.7048$:
a fórmula do truque rastreia o preconceito $(1 - 2\varepsilon)$
através de cada relé.

\textbf{19.} Palavras com spam viajam em pacotes --- uma mensagem com
``lottery'' muito provavelmente contém ``win'', então seu conjunto
a evidência é mais fraca do que o produto finge. O filtro
ainda funciona porque o exagero geralmente empurra o
\emph{direção certa}: a classificação precisa do lado correto
de $50\,\%$, não o posterior correto.

\textbf{20.} Clínica: uma condição rara vira um teste forte
em um veredicto fraco --- atualização da taxa básica, então
acumular. Tribunal: uma coincidência rara em uma grande cidade indicia
a cidade, não o homem --- \omterm{def:g10:proba:distribution}{probabilidades} não são veredictos.
Caixa de entrada: muitas pistas fracas, multiplicadas, classifique a correspondência --- até
ingenuamente. As três armadilhas: esquecer o anterior e se deixar enganar;
falso independência e seja muito enganado; leia as evidências aos poucos
e nunca sabe o que você sabe.
\end{solution}
