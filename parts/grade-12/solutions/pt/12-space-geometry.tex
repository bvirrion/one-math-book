\begin{solution}{exo:g12:space:1}
$\vect{AB}(2, 1, -1)$ e $\vect{AC}(1, -1, 1)$ não são proporcionais
($\frac21 \neq \frac{1}{-1}$), então os pontos não estão alinhados. Linha $(AB)$:
\[
(x, y, z) = (1 + 2t,\; t,\; 2 - t), \qquad t \in \R .
\]
\end{solution}

\begin{solution}{exo:g12:space:2}
$2(x - 1) - (y - 1) + 3(z - 0) = 0$, \emph{ou seja}
\[
2x - y + 3z - 1 = 0 .
\]
Para $ B(0,2,1)$: $0 - 2 + 3 - 1 = 0$: sim, $ B $ encontra-se no avião.
\end{solution}

\begin{solution}{exo:g12:space:3}
$\vect{AB}(1,1,0)$, $\vect{AC}(1,0,1)$:
\[
\cos\widehat{A}
= \frac{\vect{AB}\cdot\vect{AC}}{\norm{\vect{AB}}\,\norm{\vect{AC}}}
= \frac{1}{\sqrt2\,\sqrt2} = \frac12,
\]
então $\widehat A = 60^\circ $.
\end{solution}

\begin{solution}{exo:g12:space:4}
Pontos de $\mathcal D $: $(1 + t,\, 2 - t,\, 2t)$. Substituindo no
equação de $\mathcal P $:
\[
2(1+t) + (2 - t) - 2t + 1 = 5 - t = 0 \iff t = 5 .
\]
Exclusivo \omterm{def:g10:numbers:interunion}{interseção} apontar: $(6, -3, 10)$.
\end{solution}

\begin{solution}{exo:g12:space:5}
O \omterm{def:g11:vect:vector}{vetores} normais $(1, -1, 2)$ e $(2, 1, 1)$ não são \omterm{def:g11:vect:collinear}{colinear}, então o
planos se cruzam em uma linha. Adicionando os dois equações: $3x + 3z = 5$, então
$ x = \frac53 - z $; então o primeiro equação dá
$ y = x + 2z - 1 = \frac23 + z $. Com $ z = t $:
\[
(x, y, z) = \left(\frac53 - t,\; \frac23 + t,\; t\right), \qquad t \in \R,
\]
uma linha com \omterm{def:g11:vect:direction}{vetor de direção} $(-1, 1, 1)$. (Ambos equações verificar
de forma idêntica.)
\end{solution}

\begin{solution}{exo:g12:space:6}
Instruções $\vec u_1(1, 2, -1)$ e $\vec u_2(1, 1, 2)$ não são \omterm{def:g11:vect:collinear}{colinear},
então as linhas não são paralelas. A \omterm{def:g10:numbers:interunion}{interseção} exigiria
\[
1 + t = 2 + s, \qquad 2t = 1 + s, \qquad 3 - t = 1 + 2s .
\]
Os dois primeiros dão $ t = 1 + s $ e $2(1+s) = 1 + s $, então $ s = -1$, $ t = 0$;
mas então o terceiro lê $3 = -1$, absurdo. Nenhum ponto comum e não
paralelo: as linhas estão distorcidas.
\end{solution}

\begin{solution}{exo:g12:space:7}
\emph{1.} $\operatorname{dist} = \dfrac{\abs{2\cdot3 - 1 + 2\cdot1 - 3}}
{\sqrt{4 + 1 + 4}} = \dfrac{4}{3}$.

\emph{2.} $ H = M_0 + t\,\vec n $ com $\vec n(2, -1, 2)$, escolhendo $ t $ para que $ H \in \mathcal P $:
\[
2(3 + 2t) - (1 - t) + 2(1 + 2t) - 3 = 4 + 9t = 0 \iff t = -\frac49 .
\]
Por isso $ H\left(\frac{19}{9}, \frac{13}{9}, \frac19\right)$, e
\[
M_0H = \norm{t\,\vec n} = \frac49 \times 3 = \frac43,
\]
correspondendo à fórmula da distância.
\end{solution}

\begin{solution}{exo:g12:space:8}
\omterm{def:g10:coordgeom:system}{Coordenadas}: $ G(1,1,1)$, e o avião $(BDE)$ passa $ B(1,0,0)$,
$ D(0,1,0)$, $ E(0,0,1)$.

\emph{1.} O avião $(BDE)$ tem equação $ x + y + z = 1$ (satisfeito com o
três pontos, e eles não estão alinhados), então $\vec n(1,1,1)$ é normal
isso. Desde $\vect{AG} = (1,1,1) = \vec n $, a diagonal $(AG)$ é \omterm{def:g11:scal:orthogonal}{ortogonal}
para o avião $(BDE)$.

\emph{2.} Distância de $ A(0,0,0)$:
$\frac{\abs{0 + 0 + 0 - 1}}{\sqrt3} = \frac{1}{\sqrt3} = \frac{\sqrt3}{3}$.
A linha $(AG)$ é $(t, t, t)$; ele encontra o avião quando $3t = 1$: no
ponto $\left(\frac13, \frac13, \frac13\right)$, o centróide do
triângulo $ BDE $.
\end{solution}

\begin{solution}{exo:g12:space:9}
\emph{1.} $\vect{AB}(1, 0, -1)$, $\vect{AC}(-1, 1, 0)$,
$\vect{AD}(1, 1, 1)$. Se $\vect{AD} = a\vect{AB} + b\vect{AC}$, o
\omterm{def:g10:coordgeom:system}{coordenadas} dar $ a - b = 1$, $ b = 1$, $-a = 1$: as duas primeiras forças
$ a = 2$, contradizendo $ a = -1$. Não \omterm{def:g12:space:collinear}{coplanar}.

\emph{2.} $\vect{BC}(-2, 1, 1)$, $\vect{BD}(0, 1, 2)$. Resolvendo
$\vec n \cdot \vect{BC} = \vec n \cdot \vect{BD} = 0$: de
$ b + 2c = 0$, $ b = -2c $; então $-2a - 2c + c = 0$ dá $ c = -2a $. Com
$ a = 1$: $\vec n(1, 4, -2)$. Avião através $ B(2,1,0)$:
\[
x + 4y - 2z - 6 = 0 .
\]

\emph{3.} Distância de $ A(1,1,1)$:
$\dfrac{\abs{1 + 4 - 2 - 6}}{\sqrt{21}} = \dfrac{3}{\sqrt{21}}$.
Área de $ BCD $: $\norm{\vect{BC}}^2 = 6$, $\norm{\vect{BD}}^2 = 5$,
$\vect{BC}\cdot\vect{BD} = 3$, então
\[
\text{Area} = \frac12\sqrt{6 \times 5 - 9} = \frac{\sqrt{21}}{2}.
\]

\emph{4.}
\[
V = \frac13 \times \frac{\sqrt{21}}{2} \times \frac{3}{\sqrt{21}}
= \frac12 .
\]
\end{solution}

\begin{solution}{pb:g12:space:1}
\textbf{1.} Avião: $ x + y + z = 1$. Linha: $(t, t, t)$.
\omterm{def:g10:numbers:interunion}{Interseção}: $3t = 1$:
$\left(\frac13, \frac13, \frac13\right)$.

\textbf{2.} $\dfrac{\abs{0 + 0 + 0 - 1}}{\sqrt3} =
\dfrac{1}{\sqrt3} = \dfrac{\sqrt3}{3}$.

\textbf{3.} $1 - 1 + 0 = 0$: \omterm{def:g11:scal:orthogonal}{ortogonal}.

\textbf{4.} Mesmo normal $(1,1,1)$, mas $\frac52 \neq 1$:
planos estritamente paralelos. A linha: direção
$(1,1,0) \cdot (1,1,1) = 2 \neq 0$: não paralelo ao plano,
então ele perfura em um ponto.

\textbf{5.} Pontos médios: $\left(1, \frac12, 0\right)$,
$\left(\frac12, 1, 0\right)$, $\left(0, 1, \frac12\right)$,
$\left(0, \frac12, 1\right)$, $\left(\frac12, 0, 1\right)$,
$\left(1, 0, \frac12\right)$: cada um tem soma de \omterm{def:g10:coordgeom:system}{coordenadas}
$\frac32$.

\textbf{6.} $ P $ normal é $(1,1,1)$, a direção de
$(AG)$: perpendicular. O centro
$\left(\frac12,\frac12,\frac12\right)$ tem soma de \omterm{def:g10:coordgeom:system}{coordenadas}
$\frac32$: sobre $ P $.

\textbf{7.} Consecutivo pontos médios diferir por \omterm{def:g11:vect:vector}{vetores} como
$\left(-\frac12, \frac12, 0\right)$: comprimento
$\frac{\sqrt2}{2}$ cada vez --- seis lados iguais.

\textbf{8.} No $\left(\frac12, 1, 0\right)$: \omterm{def:g11:vect:vector}{vetores} de borda
$\left(\frac12, -\frac12, 0\right)$ e
$\left(-\frac12, 0, \frac12\right)$: \omterm{def:g12:space:dot}{produto escalar} $-\frac14$,
\omterm{def:g11:scal:dot}{normas} $\frac{\sqrt2}{2}$: $\cos = -\frac12$: ângulo
$120^\circ $. Lados iguais, iguais $120^\circ $ ângulos: um regular
hexágono.

\textbf{9.} Perímetro $6 \times \frac{\sqrt2}{2} = 3\sqrt2
\approx 4.24$. Área $6 \times \frac{\sqrt3}{4} \times
\frac12 = \frac{3\sqrt3}{4} \approx 1.30$: maior que um rosto
(área $1$) --- uma fatia maior que qualquer lado da caixa; apenas
o retângulo diagonal ($\sqrt2 \approx 1.41$) supera.

\textbf{10.} $(t,t,t)$ conhece $ P $ no $ t = \frac12$: o centro
--- qual é o \omterm{prop:g10:coordgeom:midpoint}{ponto médio} de $[AG]$ ($ A $ no $ t = 0$, $ G $ no
$ t = 1$).

\textbf{11.} Para $ c = \frac12$: o avião corta as três arestas
em $ A $ esquina, em $\left(\frac12,0,0\right)$,
$\left(0,\frac12,0\right)$, $\left(0,0,\frac12\right)$: um
triângulo equilátero de lado $\frac{\sqrt2}{2}$. Como $ c $ cresce,
o triângulo cresce, seus cantos ficam truncados em um hexágono
(regular exatamente em $ c = \frac32$), então a \omterm{def:g10:functions:function}{imagem} encolhe
simetricamente de volta a um triângulo em $ G $ canto: o
transformação de corte de cristal.

\textbf{12.} Cada lado da seção é o \omterm{def:g10:numbers:interunion}{interseção} de
o plano de corte com um \emph{face} do cubo e um
plano encontra uma face (um polígono plano) em no \omterm{def:g10:functions:extrema}{máximo} um segmento:
no \omterm{def:g10:functions:extrema}{máximo} $6$ lados. Sete é impossível; três a seis todos
ocorrem (as questões 8 e 11 apresentam duas delas).

\textbf{13.} Base $ ABD $: triângulo retângulo de área $\frac12$;
altura $ AE = 1$: volume
$\frac13 \times \frac12 \times 1 = \frac16$.

\textbf{14.} Cada par entre $ B, D, E, G $ difere exatamente
dois \omterm{def:g10:coordgeom:system}{coordenadas} por $1$: todas as seis distâncias iguais $\sqrt2$ ---
normal. O cubo se \omterm{def:g12:arith:divides}{divide} em $ BDEG $ mais quatro cantos
tetraedros congruentes com $ ABDE $: volume
$1 - 4 \times \frac16 = \frac13$.

\textbf{15.} Avião $(BDE)$: $ x + y + z = 1$; centro
$\left(\frac12,\frac12,\frac12\right)$: distância
$\frac{\abs{\frac32 - 1}}{\sqrt3} = \frac{1}{2\sqrt3} =
\frac{\sqrt3}{6}$.

\textbf{16.} $\vect{AG} = (1,1,1)$,
$\vect{BH} = (-1,1,1)$: $\cos\theta = \frac{-1 + 1 + 1}{3} =
\frac13$: $\theta \approx 70.5^\circ $, suplemento
$\approx 109.5^\circ $. Quatro pares de elétrons se repelem
e espalhados o mais longe possível em torno do carbono: o
ótimo são os cantos do tetraedro regular, cujo
as direções centro-vértice se encontram exatamente neste ângulo ---
o diamante é a questão 14 cristalizado.

\textbf{17.} $(1, t, t)$ sobre $ P $: $1 + 2t = \frac32$:
$ t = \frac14$: o ponto $\left(1, \frac14, \frac14\right)$.

\textbf{18.} $(AB)$: pontos $(t, 0, 0)$; $(EG)$: pontos
$(s, s, 1)$. A reunião precisaria $0 = 1$ no terceiro
coordenada: nunca. Paralelo precisaria $(1,0,0)$ e
$(1,1,0)$ \omterm{def:g11:vect:collinear}{colinear}: não. Nem encontro nem paralelo: distorção
--- dois corredores em pisos diferentes.

\textbf{19.} $ B + t(1,1,1) = (1 + t, t, t)$ sobre $ P $:
$1 + 3t = \frac32$: $ t = \frac16$: projeção
$\left(\frac76, \frac16, \frac16\right)$, cuja soma de \omterm{def:g10:coordgeom:system}{coordenadas}
é $\frac32$: sobre $ P$, conforme necessário.

\textbf{20.} Perfuração de linhas paramétricas (questões 1, 10, 17);
\omterm{def:g11:vect:vector}{vetores} normais medir ângulos e distâncias (questões 2, 8,
15, 16); avião equações esculpir seções (o triângulo para
metamorfose hexágono). E o parque infantil retribuiu a visita: um
hexágono regular em uma caixa de quadrados, um tetraedro regular em
os cantos, o ângulo do diamante e duas linhas que ignoram
uns aos outros --- geometria que só o espaço pode oferecer.
\end{solution}
