\section*{Capítulo \ref{ch:g12:limcont} --- Limites e Continuidade}

\begin{solution}{exo:g12:limcont:1}
Dividindo por $x^3$:
$\dfrac{2x^3 - x + 1}{5x^3 + x^2} = \dfrac{2 - 1/x^2 + 1/x^3}{5 + 1/x}
\to \dfrac{2}{5}$.

$ x^2 + 3x - 1 = x^2(1 + 3/x - 1/x^2) \to +\infty $ como $ x \to -\infty $, desde
$ x^2 \to +\infty $ e o colchete tende a $1$.

Como $ x \to 2^+$, o numerador tende a $3 > 0$ e o denominador para $0^+$,
então $\dfrac{x+1}{x-2} \to +\infty $.

Para $ x > 0$, $-\dfrac1x \leq \dfrac{\cos x}{x} \leq \dfrac1x $, então o limite
é $0$ pelo teorema do aperto.
\end{solution}

\begin{solution}{exo:g12:limcont:2}
\emph{1.} $2x^2 - 3x + 1 = (x-1)(2x-1)$ (o \omterm{def:g11:quad:discriminant}{raiz} $ x=1$ é aparente de
$2 - 3 + 1 = 0$). Para $ x \neq 1$, cancelar $ x - 1$: $ f(x) = 2x - 1$.

\emph{2.} Por isso $\lim_{x\to1} f(x) = 1$. Contexto $ f(1) = 1$ estende $ f $ para
o polinômio \omterm{def:g11:func:function}{função} $ x \mapsto 2x-1$ sobre $\R $, que é contínuo no $1$.
\end{solution}

\begin{solution}{exo:g12:limcont:3}
Escrever $ f(x) = \dfrac{3x - 1}{x + 2} = \dfrac{3(x+2) - 7}{x+2}
= 3 - \dfrac{7}{x+2}$. Como $ x \to \pm\infty $, $ f(x) \to 3$: a linha $ y = 3$
é um \omterm{def:g12:limcont:asymptote}{assíntota horizontal} em ambos os infinitos. Como $ x \to (-2)^\pm $,
$ f(x) \to \mp\infty $: a linha $ x = -2$ é um \omterm{def:g12:limcont:asymptote}{assíntota vertical}. A curva é
um \omterm{def:g11:func:reference}{hipérbole}, abaixo $ y=3$ para $ x > -2$ e acima dele para $ x < -2$.
\end{solution}

\begin{solution}{exo:g12:limcont:4}
Multiplique pelo conjugado:
\[
\sqrt{x^2 + x} - x = \frac{x^2 + x - x^2}{\sqrt{x^2+x} + x}
= \frac{x}{x\left(\sqrt{1 + 1/x} + 1\right)}
= \frac{1}{\sqrt{1 + 1/x} + 1}
\xrightarrow[x\to+\infty]{} \frac{1}{2}.
\]
De forma similar,
\[
\frac{\sqrt{1+x} - 1}{x} = \frac{(1+x) - 1}{x\left(\sqrt{1+x} + 1\right)}
= \frac{1}{\sqrt{1+x} + 1} \xrightarrow[x\to0]{} \frac{1}{2}.
\]
\end{solution}

\begin{solution}{exo:g12:limcont:5}
\emph{1.} Desde $\sin x \geq -1$, $ f(x) \geq x - 1 \to +\infty $, e nós
concluir por comparação. Ainda $ f'(x) = 1 + \cos x $ desaparece a cada ímpar
múltiplo de $\pi $, e em cada \omterm{def:g10:numbers:interval}{intervalo} $\intco{A}{+\infty}$ o \omterm{def:g11:func:function}{função}
alterna entre \omterm{def:g10:numbers:interval}{intervalos} de aumento estrito; isso é \omterm{def:g12:seq:monotonic}{aumentando} mas não
estritamente e, mais importante ainda, o seu crescimento não é necessário: a comparação é suficiente.

\emph{2.} Para $ x > 1$, divida por $ x $:
\[
g(x) = \frac{1 + \frac{\sin x}{x}}{1 - \frac{\sin x}{x}},
\]
e $\frac{\sin x}{x} \to 0$ pelo teorema do aperto
($\abs{\sin x} \leq 1$). Por isso $ g(x) \to \frac{1+0}{1-0} = 1$.
\end{solution}

\begin{solution}{exo:g12:limcont:6}
$ f(x) = x^3 - 3x + 1$ é contínuo, com $ f'(x) = 3x^2 - 3 = 3(x-1)(x+1)$:
$ f $ aumenta em $\intoc{-\infty}{-1}$, diminui em $\intcc{-1}{1}$,
aumenta em $\intco{1}{+\infty}$, com locais \omterm{def:g10:functions:extrema}{máximo} $ f(-1) = 3 > 0$ e
locais \omterm{def:g10:functions:extrema}{mínimo} $ f(1) = -1 < 0$, $\lim_{-\infty} f = -\infty $,
$\lim_{+\infty} f = +\infty $.

Em cada uma das três monotonicidade \omterm{def:g10:numbers:interval}{intervalos}, $ f $ é contínuo, estritamente
monótono, e $0$ fica estritamente entre os valores/limites do ponto final, então o
teorema da bijeção dá exatamente um zero por \omterm{def:g10:numbers:interval}{intervalo}: três zeros no total.
Assinar cheques: $ f(-2) = -1 < 0 < f(-1) = 3$, então um zero em
$\intoo{-2}{-1}$; $ f(0) = 1 > 0 > f(1) = -1$, um zero em $\intoo{0}{1}$;
$ f(1) = -1 < 0 < f(2) = 3$, um zero em $\intoo{1}{2}$.
\end{solution}

\begin{solution}{exo:g12:limcont:7}
Deixar $ g(x) = f(x) - x $, contínuo sobre $\intcc{0}{1}$. Desde
$ f(0) \in \intcc{0}{1}$, $ g(0) = f(0) \geq 0$; desde $ f(1) \in \intcc{0}{1}$,
$ g(1) = f(1) - 1 \leq 0$. Pelo teorema do valor intermediário, $ g $ desaparece em
alguns $ c \in \intcc{0}{1}$, ou seja\ $ f(c) = c $.
\end{solution}

\begin{solution}{exo:g12:limcont:8}
Deixar $ u(t)$ e $ d(t)$ serão as posições (medidas como a distância da parte inferior
da trilha) no horário $ t \in \intcc{8}{20}$ no caminho para cima e no caminho
para baixo. Ambos são contínuo, $ u(8) = 0$, $ u(20) = L $ (o comprimento da trilha),
$ d(8) = L $, $ d(20) = 0$. O \omterm{def:g11:func:function}{função} $ g = u - d $ é contínuo com
$ g(8) = -L \leq 0$ e $ g(20) = L \geq 0$, então $ g(c) = 0$ por algum tempo $ c $ por
o teorema do valor intermediário: no tempo $ c $, as posições de ambos os dias coincidem.
\end{solution}

\begin{solution}{exo:g12:limcont:9}
\emph{1.} $ f'(x) = 5x^4 + 3x^2 + 1 > 0$, então $ f $ é estritamente \omterm{def:g12:seq:monotonic}{aumentando} sobre
$\R $; com $\lim_{\pm\infty} f = \pm\infty $ e continuidade, a bijeção
teorema dá um zero real único $\alpha $. Desde $ f(0) = -1 < 0$ e
$ f(1) = 2 > 0$, $\alpha \in \intoo{0}{1}$.

\emph{2.} Manter um \omterm{def:g10:numbers:interval}{intervalo} $\intcc{a_n}{b_n}$ com
$ f(a_n) < 0 < f(b_n)$, a partir de $\intcc{0}{1}$; em cada etapa avaliar
$ f $ no \omterm{prop:g10:coordgeom:midpoint}{ponto médio} $ m $ e mantenha o meio \omterm{def:g10:numbers:interval}{intervalo} em que o sinal muda
persiste. Depois $ n $ passos o \omterm{def:g10:numbers:interval}{intervalo} tem comprimento $2^{-n}$, então o \omterm{prop:g10:coordgeom:midpoint}{ponto médio}
aproxima $\alpha $ para dentro $2^{-(n+1)}$. Nós precisamos
$2^{-(n+1)} \leq 10^{-6}$, ou seja\ $ n + 1 \geq \frac{6}{\log_{10} 2} \approx 19.93$:
$ n = 20$ iterações são suficientes.
\end{solution}

\begin{solution}{pb:g12:limcont:1}
\textbf{1.} Termos dominantes: $\frac{3x^2}{x^2} \to 3$;
$\frac{2x}{x^2} = \frac2x \to 0$;
$ x^3 - x^2 = x^3\left(1 - \frac1x\right) \to +\infty $.

\textbf{2.} $\lim_{x \to \pm\infty} f = 3$: horizontais
assíntota $ y = 3$; $\lim_{x \to (-2)^\pm} f = \mp\infty $:
\omterm{def:g12:limcont:asymptote}{assíntota vertical} $ x = -2$.

\textbf{3.} $\frac{(x-3)(x+3)}{x - 3} = x + 3 \to 6$. E
$\frac{\sqrt{x+1} - 1}{x} = \frac{x}{x(\sqrt{x+1} + 1)} =
\frac{1}{\sqrt{x+1} + 1} \to \frac12$.

\textbf{4.} $\abs{x \sin\frac1x} \leq \abs x \to 0$, então o
limite é $0$. E
$\abs{\frac{\sin x}{x}} \leq \frac1x \to 0$: limite $0$.

\textbf{5.} $\frac{\sqrt{4x^2 + x}}{x} =
\sqrt{4 + \frac1x} \to \sqrt4 = 2$ (continuidade da praça
\omterm{def:g11:quad:discriminant}{raiz}, composição).

\textbf{6.} $ f(x) = x^3 + x $ é contínuo com
$ f(1) = 2 < 5 < 10 = f(2)$: pelo IVT, $ f(c) = 5$ para alguns
$ c \in \intoo{1}{2}$.

\textbf{7.} $ f'(x) = 3x^2 + 1 > 0$: estritamente \omterm{def:g12:seq:monotonic}{aumentando} sobre
$\R $, contínuo, com limites $\mp\infty $: pela bijeção
teorema, cada valor --- em particular $5$ --- é alcançado
exatamente uma vez.

\textbf{8.} $ p(x) = x^3\left(1 + \frac ax + \frac{b}{x^2} +
\frac{c}{x^3}\right)$: como $ x \to +\infty $, $ p \to +\infty $; como
$ x \to -\infty $, $ p \to -\infty $. Então, em alguns grandes \omterm{def:g10:numbers:interval}{intervalo}
$\intcc{-M}{M}$, $ p(-M) < 0 < p(M)$: o IVT fornece um \omterm{def:g11:quad:discriminant}{raiz}.
O mesmo argumento do termo dominante funciona literalmente para qualquer situação ímpar.
grau --- potências ímpares mantêm sinais opostos nos dois
infinitos.

\textbf{9.} Reduzir pela metade $ n $ vezes deixa um \omterm{def:g10:numbers:interval}{intervalo} de comprimento
$2^{-n}$; $2^{-20} \approx 9.5 \times 10^{-7} < 10^{-6}$:
vinte passos. Newton dobra os dígitos corretos a cada passo: de
um dígito correto, cerca de $3$--$4$ passos chegam a seis --- o
tartaruga e o galgo da descoberta de raízes.

\textbf{10.} $ g $ não é contínuo sobre $\intcc{-1}{1}$: isso
explode em $0$, que o \omterm{def:g10:numbers:interval}{intervalo} contém. O IVT
a conclusão evapora com sua hipótese --- sempre verifique
continuidade \emph{no geral \omterm{def:g10:numbers:interval}{intervalo}}, não apenas no
pontos finais.

\textbf{11.} $ g(x) = f(x) - x $ é contínuo;
$ g(0) = f(0) \geq 0$ (os valores estão em $\intcc{0}{1}$) e
$ g(1) = f(1) - 1 \leq 0$. Pelo IVT, $ g(c) = 0$ para alguns $ c $:
$ f(c) = c $.

\textbf{12.} Idealizando o país como segmento
$\intcc{0}{1}$: $ f $ envia cada lugar para o lugar mostrado
diretamente abaixo dele, no mapa amassado; o amassamento mantém o
mapear o país, então $ f $ pousa em $\intcc{0}{1}$, e isso
é contínuo (o papel não rasga). O \omterm{pb:g10:functions:1}{ponto fixo} $ c $ é
o ponto situado precisamente acima do seu próprio \omterm{def:g10:functions:function}{imagem} --- Brouwer's
teorema lida com o mapa bidimensional honesto.

\textbf{13.} $ g(0) + g(\pi) = \left(T(0) - T(\pi)\right) +
\left(T(\pi) - T(2\pi)\right) = T(0) - T(2\pi) = 0$. Então
$ g(0)$ e $ g(\pi)$ são opostos (ou ambos zero): de qualquer maneira
$ g $, contínuo, desaparece em alguns $ t^* \in \intcc{0}{\pi}$:
$ T(t^*) = T(t^* + \pi)$ --- em algum lugar no equador, dois
pontos antípodas concordam com a temperatura, em cada
momento.

\textbf{14.} Girar a mesa um quarto de volta troca
os papéis das duas diagonais: o par de pernas que pressionou
o chão e o par que (um deles) pairou trocam
situações, então a altura pairando --- contada com seu sinal
como ``lacuna da diagonal distinta'' --- toma o oposto
sinais em $\theta $ e $\theta + \frac\pi2$. Contínuo chão
faz $ h $ contínuo, e o IVT transfere um ângulo
$\theta^*$ com $ h(\theta^*) = 0$: quatro pernas para baixo. (Não existe tal
sorte com uma mesa retangular ou piso escalonado.)

\textbf{15.} Modelo: construa o \emph{diferença} dos dois
quantidades que você deseja coincidir ($ f(x) - x $;
$ T(t) - T(t + \pi)$; posição para cima menos posição para baixo para o
caminhante; a lacuna assinada para a tabela); verifique se leva valores de
ambos os sinais (muitas vezes através de uma simetria das extremidades); invocar o IVT
em um contínuo \omterm{def:g11:func:function}{função}. Coincidência forçada.

\textbf{16.} Perguntas abertas: \emph{quantos} soluções
(respondido pela monotonicidade estrita e pelo teorema da bijeção,
questão 7) e \emph{onde} eles são (respondidos por \omterm{thm:g12:limcont:ivt}{dicotomia}
ou Newton, questão 9). O IVT é um oráculo de pura existência.

\textbf{17.} $3 - \frac{7}{x + 2}$: \omterm{def:g12:limcont:asymptote}{assíntota vertical}
$ x = -2$, horizontais $ y = 3$; em cada lado de $-2$ o
\omterm{def:g11:func:function}{função} é \omterm{def:g12:seq:monotonic}{aumentando} ($-\frac{7}{x+2}$ aumenta onde
definido); a curva é a referência \omterm{def:g11:func:reference}{hipérbole}
$-\frac7x $ mudou para a esquerda $2$ e acima $3$ --- duas filiais
abraçando o assíntota cruzar.

\textbf{18.} Para $ x \neq 1$: $ f(x) = x + 1$, então
$\lim_{x \to 1} f = 2$: contexto $ f(1) = 2$ restaura
continuidade. A descontinuidade foi um furo, não um salto ---
preencha o único ponto que falta e a curva cicatriza:
\emph{removível}.

\textbf{19.} Contínuo em cada \omterm{def:g10:numbers:interval}{intervalo} entre consecutivos
\omterm{def:g10:numbers:sets}{inteiros}; em cada inteiro $ n $ ele salta de $ n - 1$ para $ n $.
Com $ a = 0.9$, $ b = 1.1$: $\lfloor a \rfloor = 0 < \frac12 <
1 = \lfloor b \rfloor $, ainda $\lfloor x \rfloor = \frac12$ tem
sem solução --- o salto se teletransporta $\frac12$, que é
precisamente o que o IVT proíbe contínuo funções pendência.

\textbf{20.} Face local: valor $=$ limite em cada ponto ---
a punção removível (questão 18) violou-a de forma curável, o
saltos do chão (questão 19) incuravelmente. Rosto global: em um
\omterm{def:g10:numbers:interval}{intervalo}, todos os valores intermediários são visitados --- o \omterm{def:g11:quad:discriminant}{raiz}
certificados (6--8), o \omterm{pb:g10:functions:1}{ponto fixo}, o equador, a mesa
e o caminhante moram aqui. O nome oficial do slogan: o
\emph{teorema do valor intermediário}.
\end{solution}
