\section*{Capítulo \ref{ch:g12:arith} --- Aritmética}

\begin{solution}{exo:g12:arith:1}
$17 \times 119 = 2023$, então $2026 = 17 \times 119 + 3$: quociente $119$,
restante $3$. Para $-2026$: $-2026 = 17\times(-120) + 14$ (de fato
$17 \times 120 = 2040$ e $2040 - 2026 = 14$): quociente $-120$, restante
$14$ (o restante deve estar em $\intco{0}{17}$, então é \emph{não} $-3$).
\end{solution}

\begin{solution}{exo:g12:arith:2}
\omterm{def:g12:complex:modulus}{Módulo} $10$: $7^2 = 49 \equiv 9 \equiv -1$. Por isso
$7^{100} = \left(7^2\right)^{50} \equiv (-1)^{50} = 1 \pmod{10}$: o último
dígito de $7^{100}$ é $1$.
\end{solution}

\begin{solution}{exo:g12:arith:3}
Euclides: $1071 = 2\times462 + 147$; $462 = 3\times147 + 21$;
$147 = 7\times21 + 0$. Então $\gcd = 21$.

Substituição reversa: $21 = 462 - 3\times147 = 462 - 3(1071 - 2\times462)
= 7\times462 - 3\times1071$. Por isso $ u = -3$, $ v = 7$:
$1071\times(-3) + 462\times7 = 21$.
\end{solution}

\begin{solution}{exo:g12:arith:4}
Todo inteiro é $\equiv 0, 1, 2$ ou $3 \pmod 4$ e quadratura:
$0^2 \equiv 0$, $1^2 \equiv 1$, $2^2 = 4 \equiv 0$, $3^2 = 9 \equiv 1$. Então
$ n^2 \equiv 0$ ou $1 \pmod 4$. A soma de dois quadrados é então congruente com
$0 + 0$, $0 + 1$ ou $1 + 1$, \emph{ou seja} para $0$, $1$ ou $2 \pmod 4$ ---
nunca para $3$.
\end{solution}

\begin{solution}{exo:g12:arith:5}
\omterm{def:g12:arith:divides}{Divisibilidade} por $2$: entre $ n $ e $ n + 1$, um é par. \omterm{def:g12:arith:divides}{Divisibilidade} por
$3$: se $ n \equiv 0$, então $3 \mid n $; se $ n \equiv 1 \pmod 3$, então
$2n + 1 \equiv 3 \equiv 0$; se $ n \equiv 2$, então $ n + 1 \equiv 0$. Em tudo
casos $3$ \omterm{def:g12:arith:divides}{divide} o produto. Desde $\gcd(2,3) = 1$,
\cref{cor:g12:arith:coprimeprod} dá $6 \mid n(n+1)(2n+1)$.
(Isso também prova novamente que $\frac{n(n+1)(2n+1)}{6}$, a soma dos quadrados de
\cref{exo:g12:seq:1}, é um inteiro.)
\end{solution}

\begin{solution}{exo:g12:arith:6}
Procuramos o inverso de $5$ \omterm{def:g12:complex:modulus}{módulo} $11$: teste (ou Bézout),
$5 \times 9 = 45 = 44 + 1 \equiv 1 \pmod{11}$. Multiplicando o congruência por
$9$:
\[
x \equiv 9 \times 3 = 27 \equiv 5 \pmod{11}.
\]
As soluções são as \omterm{def:g10:numbers:sets}{inteiros} $ x = 5 + 11k $, $ k \in \Z $. (Verificar:
$5\times5 = 25 \equiv 3 \pmod{11}$.)
\end{solution}

\begin{solution}{exo:g12:arith:7}
$\gcd(17, 40) = 1$, então existem soluções. Euclides:
$40 = 2\times17 + 6$; $17 = 2\times6 + 5$; $6 = 5 + 1$. Substituição reversa:
$1 = 6 - 5 = 6 - (17 - 2\times6) = 3\times6 - 17
= 3(40 - 2\times17) - 17 = 3\times40 - 7\times17$. Por isso
$17\times(-7) - 40\times(-3) = 1$: a solução particular
$(x_0, y_0) = (-7, -3)$.

Solução geral de $17x - 40y = 1$: subtraindo a relação particular,
$17(x + 7) = 40(y + 3)$; desde $\gcd(17, 40) = 1$, Gauss dá
$40 \mid x + 7$, então $ x = -7 + 40k $ e então $ y = -3 + 17k $, $ k \in \Z $
(todos os quais verificam).

Para $17x - 40y = 6$, multiplique a solução particular por $6$:
$(x_1, y_1) = (-42, -18)$, e o mesmo raciocínio dá
\[
x = -42 + 40k, \qquad y = -18 + 17k, \qquad k \in \Z .
\]
(Ex.\ $ k = 2$: $ x = 38$, $ y = 16$; de fato $17\times38 - 40\times16
= 646 - 640 = 6$.)
\end{solution}

\begin{solution}{exo:g12:arith:8}
Suponha $\sqrt2 = \frac ab $ com $ a, b \in \N^*$; então $ a^2 = 2b^2$. No
\omterm{def:g12:arith:prime}{melhor} fatoração de um quadrado, todo expoente é par; então o expoente de
$2$ em $ a^2$ é par, enquanto em $2b^2$ é ímpar (um a mais que um par
número). Duas fatorações do mesmo inteiro com diferentes expoentes de
$2$ contradiz a singularidade em \cref{thm:g12:arith:fta}. Portanto, não existe tal
existe fração: $\sqrt2 \notin \Q $.
\end{solution}

\begin{solution}{exo:g12:arith:9}
\emph{1.} De $ k\binom pk = p \binom{p-1}{k-1}$, $ p $ \omterm{def:g12:arith:divides}{divide}
$ k\binom pk $. Para $1 \leq k \leq p-1$, $ p \nmid k $ e $ p $ \omterm{def:g12:arith:prime}{melhor} dar
$\gcd(p, k) = 1$, então o lema de Gauss produz $ p \mid \binom pk $.

\emph{2.} Indução ativada $ a $. Para $ a = 0$: $0^p \equiv 0$. Assumir
$ a^p \equiv a \pmod p $. Pelo teorema binomial,
\[
(a+1)^p = \sum_{k=0}^{p} \binom pk a^k
\equiv a^p + 1 \pmod p,
\]
todos os termos médios desaparecendo \omterm{def:g12:complex:modulus}{módulo} $ p $ pelo ponto 1. Pela indução
hipótese, $(a+1)^p \equiv a + 1 \pmod p $. Isso prova
$ a^p \equiv a $ para todos $ a \in \N $, e o caso $ a < 0$ segue escrevendo
$ a \equiv a + kp $ para um representante positivo adequado.
\end{solution}

\begin{solution}{exo:g12:arith:10}
$ n \equiv 2 \pmod 3$ e $ n \equiv 3 \pmod 5$: escrever $ n = 2 + 3s $; então
$2 + 3s \equiv 3 \pmod 5$, \emph{ou seja} $3s \equiv 1 \pmod 5$. O inverso de
$3$ \omterm{def:g12:complex:modulus}{módulo} $5$ é $2$ ($3\times2 = 6 \equiv 1$), então $ s \equiv 2 \pmod 5$,
dizer $ s = 2 + 5t $, e $ n = 8 + 15t $: as duas primeiras condições significar
$ n \equiv 8 \pmod{15}$.

Adicionando $ n \equiv 2 \pmod 7$: $8 + 15t \equiv 2 \pmod 7$, e
$15 \equiv 1 \pmod 7$, então $ t \equiv -6 \equiv 1 \pmod 7$, dizer
$ t = 1 + 7u $. Por isso $ n = 23 + 105u $:
\[
n \equiv 23 \pmod{105}.
\]
(Verificar: $23 = 3\times7 + 2 = 5\times4 + 3 = 7\times3 + 2$.)
\end{solution}

\begin{solution}{pb:g12:arith:1}
\textbf{1.} $2026 = 289 \times 7 + 3$: $2026 \equiv 3
\pmod 7$. Poderes de $7$ moda $10$: $7, 9, 3, 1$, ciclo de
comprimento $4$; $100 \equiv 0 \pmod 4$: último dígito de
$7^{100}$ é $1$.

\textbf{2.} $5^2 = 25 \equiv -1 \pmod{13}$, então
$5^{116} = \left(5^2\right)^{58} \equiv (-1)^{58} = 1$ e
$5^{117} \equiv 5 \pmod{13}$.

\textbf{3.} O inverso de $3$ \omterm{def:g12:complex:modulus}{módulo} $7$ é $5$
($15 \equiv 1$): $ x \equiv 5 \times 5 = 25 \equiv 4 \pmod 7$.

\textbf{4.} $97 = 2 \times 35 + 27$; $35 = 27 + 8$;
$27 = 3 \times 8 + 3$; $8 = 2 \times 3 + 2$; $3 = 2 + 1$.
Substituição reversa: $1 = 97 \times 13 + 35 \times (-36)$. Então
$35 \times (-36) \equiv 1 \pmod{97}$: o inverso de $35$ é
$-36 \equiv 61 \pmod{97}$.

\textbf{5.} $ a $ é \omterm{def:g12:complex:modulus}{módulo} invertível $ n $ exatamente quando
$\gcd(a, n) = 1$:Bézout fornece $ au + nv = 1$, ou seja\
$ au \equiv 1$; inversamente, um inverso força o mdc dividir
$1$.

\textbf{6.} $0{\cdot}10 + 3{\cdot}9 + 0{\cdot}8 + 6{\cdot}7 +
4{\cdot}6 + 0{\cdot}5 + 6{\cdot}4 + 1{\cdot}3 + 5{\cdot}2 +
2{\cdot}1 = 132 = 12 \times 11 \equiv 0 \pmod{11}$: válido.

\textbf{7.} A soma varia em $ wd $ com
$1 \leq w \leq 10$ e $1 \leq \abs d \leq 9$: desde $11$ é
\omterm{def:g12:arith:prime}{melhor} e \omterm{def:g12:arith:divides}{divide} nenhum fator, não pode dividir o produto
(\cref{thm:g12:arith:gauss} / \cref{prop:g12:arith:primedivides}):
a soma alterada nunca mais será $\equiv 0$: cada dígito
erro dispara o alarme.

\textbf{8.} Trocando dígitos adjacentes $ a, b $ (pesos
$ w + 1, w $) altera a soma em
$(w+1)b + wa - (w+1)a - wb = b - a \not\equiv 0$ para
$ a \neq b $: detectado. O segredo é o \emph{primalidade} de
$11$: \omterm{def:g12:complex:modulus}{módulo} $10$, produtos como $5 \times 2$ desaparecer com
nem fator zero, então um peso $5$ erro de $\pm 2$ (ou um
transposição infeliz) poderia escapar.

\textbf{9.} Soma ponderada dos doze dígitos: $119$; o
dígito de verificação deve completá-lo para um múltiplo de $10$: $1$ (completo
código $978\,2940199\,051$). EAN perde transposições adjacentes
com $2(a - b) \equiv 0 \pmod{10}$, ou seja\
$\abs{a - b} = 5$: trocando um $2$ e um $7$, digamos, passa
invisível --- o preço do amigável \omterm{def:g12:complex:modulus}{módulo} $10$.

\textbf{10.} Dobrando cada segundo dígito da direita e
dobrável ($16 \to 7$, etc.), a soma chega a $80 \equiv 0
\pmod{10}$: o cartão de teste é validado.

\textbf{11.} Com um \omterm{def:g12:arith:prime}{melhor} \omterm{def:g12:complex:modulus}{módulo} todo peso é invertível,
então \emph{todos} erros únicos e \emph{todos} adjacente
transposições são capturadas --- luxo do ISBN; mod-$10$ esquemas
mantenha dígitos amigáveis e aceite um pequeno ponto cego.

\textbf{12.} $2^{10} = 1024 = 3 \times 341 + 1 \equiv 1
\pmod{341}$, por isso $2^{340} = \left(2^{10}\right)^{34} \equiv
1$. Ainda $341 = 11 \times 31$ é composto: passa pelo teste de Fermat
teste na base $2$ enquanto não é \omterm{def:g12:arith:prime}{melhor}. Moral: Fermat
congruência é necessário, não suficiente --- teste de primalidade
precisa de ferramentas mais precisas (e as consegue, na universidade
volumes).

\textbf{13.} $3d \equiv 1 \pmod{20}$: $ d = 7$
($21 = 20 + 1$).

\textbf{14.} $ c = 4^3 = 64 \equiv 31 \pmod{33}$.

\textbf{15.} $31 \equiv -2$: $(-2)^7 = -128$, e
$-128 + 4 \times 33 = 4$: o texto cifrado é descriptografado para
$ m = 4$. A fechadura gira.

\textbf{16.} \omterm{def:g12:complex:modulus}{Módulo} $3$: se $3 \nmid m $,
$ m^2 \equiv 1$ (Fermat), então
$ m^{21} = m \cdot \left(m^2\right)^{10} \equiv m $; se
$3 \mid m $, ambos os lados são $\equiv 0$. \omterm{def:g12:complex:modulus}{Módulo} $11$:
$ m^{10} \equiv 1$ ou $11 \mid m $, e
$ m^{21} = m \cdot \left(m^{10}\right)^2 \equiv m $. Ambos $3$
e $11$ dividir $ m^{21} - m $, e sendo \omterm{def:g12:arith:gcd}{coprime} seu produto
$33$ também faz (Gauss): $ m^{21} \equiv m \pmod{33}$. O
expoente $ ed = 21 = 1 + 20k $ foi construído de modo que tanto Fermat
expoentes ($2$ e $10$, dividindo $20$) desaparecer.

\textbf{17.} Multiplicando dois $300$-dígito primos leva um
microssegundo; recuperando-os das derrotas de seus produtos a cada
algoritmo conhecido e todos os computadores do mundo --- o bloqueio é
uma rua de mão única. (Nosso $ n = 33$ é a rua em escala de brinquedo,
percorrível em ambas as direções.)

\textbf{18.} Testando resíduos (ou construindo com Bézout):
$ n \equiv 8 \pmod{15}$: as contagens $8, 23, 38, 53, \dots $
\omterm{def:g12:complex:modulus}{Módulo} de exclusividade $15$: duas soluções diferem por um múltiplo de
$3$ e de $5$, portanto de $15$ ($3$ e $5$ \omterm{def:g12:arith:gcd}{coprime}, Gauss).
O geral com $1000$ soldados anuncia ``$8$'' por três
escalações rápidas --- o antigo truque do número de funcionários.

\textbf{19.} $10 \equiv 1 \pmod 9$ dá $10^k \equiv 1$, então
$\sum d_k 10^k \equiv \sum d_k $: um número e sua soma de dígitos
são \omterm{def:g12:complex:modulus}{módulo} congruente $9$ (e \omterm{def:g12:complex:modulus}{módulo} $3$). E
$10 \equiv -1 \pmod{11}$ dá
$\sum d_k 10^k \equiv \sum (-1)^k d_k $: a regra alternada.
Duas regras de infância, uma linha cada.

\textbf{20.} Congruências transformou restos em \omterm{def:g12:seq:arith-geom}{aritmética}
(Parte I); Bézout cunhou os inversos que resolvem linearmente
congruências e RSA $ d$ (questões 4, 13); O pequeno Fermat
teorema abriu e fechou a fechadura (questões 15-16); colando
\omterm{def:g12:arith:gcd}{coprime} moduli contou soldados e terminou a prova
(questões 16, 18). Veredicto sobre Hardy: o teorema mais puro que ele
sabia agora que protege cada compra - a pureza, com o tempo, é o
coisa mais aplicável que existe.
\end{solution}
