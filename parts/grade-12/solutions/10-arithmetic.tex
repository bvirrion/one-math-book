\section*{Chapter \ref{ch:g12:arith} --- Arithmetic}

\begin{solution}{exo:g12:arith:1}
$17 \times 119 = 2023$, so $2026 = 17 \times 119 + 3$: quotient $119$,
remainder $3$. For $-2026$: $-2026 = 17\times(-120) + 14$ (indeed
$17 \times 120 = 2040$ and $2040 - 2026 = 14$): quotient $-120$, remainder
$14$ (the remainder must lie in $\intco{0}{17}$, so it is \emph{not} $-3$).
\end{solution}

\begin{solution}{exo:g12:arith:2}
Modulo $10$: $7^2 = 49 \equiv 9 \equiv -1$. Hence
$7^{100} = \left(7^2\right)^{50} \equiv (-1)^{50} = 1 \pmod{10}$: the last
digit of $7^{100}$ is $1$.
\end{solution}

\begin{solution}{exo:g12:arith:3}
Euclid: $1071 = 2\times462 + 147$; $462 = 3\times147 + 21$;
$147 = 7\times21 + 0$. So $\gcd = 21$.

Back-substitution: $21 = 462 - 3\times147 = 462 - 3(1071 - 2\times462)
= 7\times462 - 3\times1071$. Thus $u = -3$, $v = 7$:
$1071\times(-3) + 462\times7 = 21$.
\end{solution}

\begin{solution}{exo:g12:arith:4}
Every \omterm{def:g10:numbers:sets}{integer} is $\equiv 0, 1, 2$ or $3 \pmod 4$, and squaring:
$0^2 \equiv 0$, $1^2 \equiv 1$, $2^2 = 4 \equiv 0$, $3^2 = 9 \equiv 1$. So
$n^2 \equiv 0$ or $1 \pmod 4$. A sum of two squares is then congruent to
$0 + 0$, $0 + 1$ or $1 + 1$, \emph{i.e.} to $0$, $1$ or $2 \pmod 4$ ---
never to $3$.
\end{solution}

\begin{solution}{exo:g12:arith:5}
\omterm{def:g12:arith:divides}{Divisibility} by $2$: among $n$ and $n + 1$, one is even. \omterm{def:g12:arith:divides}{Divisibility} by
$3$: if $n \equiv 0$, then $3 \mid n$; if $n \equiv 1 \pmod 3$, then
$2n + 1 \equiv 3 \equiv 0$; if $n \equiv 2$, then $n + 1 \equiv 0$. In all
cases $3$ \omterm{def:g12:arith:divides}{divides} the product. Since $\gcd(2,3) = 1$,
\cref{cor:g12:arith:coprimeprod} gives $6 \mid n(n+1)(2n+1)$.
(This also re-proves that $\frac{n(n+1)(2n+1)}{6}$, the sum of squares of
\cref{exo:g12:seq:1}, is an \omterm{def:g10:numbers:sets}{integer}.)
\end{solution}

\begin{solution}{exo:g12:arith:6}
We look for the inverse of $5$ modulo $11$: testing (or Bézout),
$5 \times 9 = 45 = 44 + 1 \equiv 1 \pmod{11}$. Multiplying the \omterm{def:g12:arith:congruence}{congruence} by
$9$:
\[
x \equiv 9 \times 3 = 27 \equiv 5 \pmod{11}.
\]
The solutions are the \omterm{def:g10:numbers:sets}{integers} $x = 5 + 11k$, $k \in \Z$. (Check:
$5\times5 = 25 \equiv 3 \pmod{11}$.)
\end{solution}

\begin{solution}{exo:g12:arith:7}
$\gcd(17, 40) = 1$, so solutions exist. Euclid:
$40 = 2\times17 + 6$; $17 = 2\times6 + 5$; $6 = 5 + 1$. Back-substituting:
$1 = 6 - 5 = 6 - (17 - 2\times6) = 3\times6 - 17
= 3(40 - 2\times17) - 17 = 3\times40 - 7\times17$. Hence
$17\times(-7) - 40\times(-3) = 1$: the particular solution
$(x_0, y_0) = (-7, -3)$.

General solution of $17x - 40y = 1$: subtracting the particular relation,
$17(x + 7) = 40(y + 3)$; since $\gcd(17, 40) = 1$, Gauss gives
$40 \mid x + 7$, so $x = -7 + 40k$ and then $y = -3 + 17k$, $k \in \Z$
(all of which check).

For $17x - 40y = 6$, multiply the particular solution by $6$:
$(x_1, y_1) = (-42, -18)$, and the same reasoning gives
\[
x = -42 + 40k, \qquad y = -18 + 17k, \qquad k \in \Z .
\]
(E.g.\ $k = 2$: $x = 38$, $y = 16$; indeed $17\times38 - 40\times16
= 646 - 640 = 6$.)
\end{solution}

\begin{solution}{exo:g12:arith:8}
Suppose $\sqrt2 = \frac ab$ with $a, b \in \N^*$; then $a^2 = 2b^2$. In the
\omterm{def:g12:arith:prime}{prime} factorization of a square, every exponent is even; so the exponent of
$2$ in $a^2$ is even, while in $2b^2$ it is odd (one more than an even
number). Two factorizations of the same \omterm{def:g10:numbers:sets}{integer} with different exponents of
$2$ contradict the uniqueness in \cref{thm:g12:arith:fta}. Hence no such
fraction exists: $\sqrt2 \notin \Q$.
\end{solution}

\begin{solution}{exo:g12:arith:9}
\emph{1.} From $k\binom pk = p \binom{p-1}{k-1}$, $p$ \omterm{def:g12:arith:divides}{divides}
$k\binom pk$. For $1 \leq k \leq p-1$, $p \nmid k$ and $p$ \omterm{def:g12:arith:prime}{prime} give
$\gcd(p, k) = 1$, so Gauss's lemma yields $p \mid \binom pk$.

\emph{2.} Induction on $a$. For $a = 0$: $0^p \equiv 0$. Assume
$a^p \equiv a \pmod p$. By the binomial theorem,
\[
(a+1)^p = \sum_{k=0}^{p} \binom pk a^k
\equiv a^p + 1 \pmod p,
\]
all middle terms vanishing modulo $p$ by point 1. By the induction
hypothesis, $(a+1)^p \equiv a + 1 \pmod p$. This proves
$a^p \equiv a$ for all $a \in \N$, and the case $a < 0$ follows by writing
$a \equiv a + kp$ for a suitable positive representative.
\end{solution}

\begin{solution}{exo:g12:arith:10}
$n \equiv 2 \pmod 3$ and $n \equiv 3 \pmod 5$: write $n = 2 + 3s$; then
$2 + 3s \equiv 3 \pmod 5$, \emph{i.e.} $3s \equiv 1 \pmod 5$. The inverse of
$3$ modulo $5$ is $2$ ($3\times2 = 6 \equiv 1$), so $s \equiv 2 \pmod 5$,
say $s = 2 + 5t$, and $n = 8 + 15t$: the first two conditions \omterm{def:g11:stat:mean}{mean}
$n \equiv 8 \pmod{15}$.

Adding $n \equiv 2 \pmod 7$: $8 + 15t \equiv 2 \pmod 7$, and
$15 \equiv 1 \pmod 7$, so $t \equiv -6 \equiv 1 \pmod 7$, say
$t = 1 + 7u$. Hence $n = 23 + 105u$:
\[
n \equiv 23 \pmod{105}.
\]
(Check: $23 = 3\times7 + 2 = 5\times4 + 3 = 7\times3 + 2$.)
\end{solution}

\begin{solution}{pb:g12:arith:1}
\textbf{1.} $2026 = 289 \times 7 + 3$: $2026 \equiv 3
\pmod 7$. Powers of $7$ mod $10$: $7, 9, 3, 1$, cycle of
length $4$; $100 \equiv 0 \pmod 4$: last digit of
$7^{100}$ is $1$.

\textbf{2.} $5^2 = 25 \equiv -1 \pmod{13}$, so
$5^{116} = \left(5^2\right)^{58} \equiv (-1)^{58} = 1$ and
$5^{117} \equiv 5 \pmod{13}$.

\textbf{3.} The inverse of $3$ modulo $7$ is $5$
($15 \equiv 1$): $x \equiv 5 \times 5 = 25 \equiv 4 \pmod 7$.

\textbf{4.} $97 = 2 \times 35 + 27$; $35 = 27 + 8$;
$27 = 3 \times 8 + 3$; $8 = 2 \times 3 + 2$; $3 = 2 + 1$.
Back-substituting: $1 = 97 \times 13 + 35 \times (-36)$. So
$35 \times (-36) \equiv 1 \pmod{97}$: the inverse of $35$ is
$-36 \equiv 61 \pmod{97}$.

\textbf{5.} $a$ is invertible modulo $n$ exactly when
$\gcd(a, n) = 1$: B\'ezout provides $au + nv = 1$, i.e.\
$au \equiv 1$; conversely an inverse forces the \omterm{def:g12:arith:gcd}{gcd} to divide
$1$.

\textbf{6.} $0{\cdot}10 + 3{\cdot}9 + 0{\cdot}8 + 6{\cdot}7 +
4{\cdot}6 + 0{\cdot}5 + 6{\cdot}4 + 1{\cdot}3 + 5{\cdot}2 +
2{\cdot}1 = 132 = 12 \times 11 \equiv 0 \pmod{11}$: valid.

\textbf{7.} The sum changes by $wd$ with
$1 \leq w \leq 10$ and $1 \leq \abs d \leq 9$: since $11$ is
\omterm{def:g12:arith:prime}{prime} and \omterm{def:g12:arith:divides}{divides} neither factor, it cannot divide the product
(\cref{thm:g12:arith:gauss} / \cref{prop:g12:arith:primedivides}):
the changed sum is never again $\equiv 0$: every single-digit
error trips the alarm.

\textbf{8.} Swapping adjacent digits $a, b$ (weights
$w + 1, w$) changes the sum by
$(w+1)b + wa - (w+1)a - wb = b - a \not\equiv 0$ for
$a \neq b$: detected. The secret is the \emph{primality} of
$11$: modulo $10$, products like $5 \times 2$ vanish with
neither factor zero, so a weight-$5$ error of $\pm 2$ (or an
unlucky transposition) could slip through.

\textbf{9.} Weighted sum of the twelve digits: $119$; the
check digit must complete it to a multiple of $10$: $1$ (full
code $978\,2940199\,051$). EAN misses adjacent transpositions
with $2(a - b) \equiv 0 \pmod{10}$, i.e.\
$\abs{a - b} = 5$: swapping a $2$ and a $7$, say, passes
unseen --- the price of the friendly \omterm{def:g12:complex:modulus}{modulus} $10$.

\textbf{10.} Doubling every second digit from the right and
folding ($16 \to 7$, etc.), the sum comes to $80 \equiv 0
\pmod{10}$: the test card validates.

\textbf{11.} With a \omterm{def:g12:arith:prime}{prime} \omterm{def:g12:complex:modulus}{modulus} every weight is invertible,
so \emph{all} single errors and \emph{all} adjacent
transpositions are caught --- ISBN's luxury; mod-$10$ schemes
keep human-friendly digits and accept a short blind spot.

\textbf{12.} $2^{10} = 1024 = 3 \times 341 + 1 \equiv 1
\pmod{341}$, hence $2^{340} = \left(2^{10}\right)^{34} \equiv
1$. Yet $341 = 11 \times 31$ is composite: it passes Fermat's
test at base $2$ while being no \omterm{def:g12:arith:prime}{prime}. Moral: Fermat's
\omterm{def:g12:arith:congruence}{congruence} is necessary, not sufficient --- primality testing
needs sharper tools (and gets them, in the university
volumes).

\textbf{13.} $3d \equiv 1 \pmod{20}$: $d = 7$
($21 = 20 + 1$).

\textbf{14.} $c = 4^3 = 64 \equiv 31 \pmod{33}$.

\textbf{15.} $31 \equiv -2$: $(-2)^7 = -128$, and
$-128 + 4 \times 33 = 4$: the ciphertext decrypts to
$m = 4$. The lock turns.

\textbf{16.} Modulo $3$: if $3 \nmid m$,
$m^2 \equiv 1$ (Fermat), so
$m^{21} = m \cdot \left(m^2\right)^{10} \equiv m$; if
$3 \mid m$, both sides are $\equiv 0$. Modulo $11$:
$m^{10} \equiv 1$ or $11 \mid m$, and
$m^{21} = m \cdot \left(m^{10}\right)^2 \equiv m$. Both $3$
and $11$ divide $m^{21} - m$, and being \omterm{def:g12:arith:gcd}{coprime} their product
$33$ does too (Gauss): $m^{21} \equiv m \pmod{33}$. The
exponent $ed = 21 = 1 + 20k$ was built so that both Fermat
exponents ($2$ and $10$, dividing $20$) disappear.

\textbf{17.} Multiplying two $300$-digit \omterm{def:g12:arith:prime}{primes} takes a
microsecond; recovering them from their product defeats every
known algorithm and all the world's computers --- the lock is
a one-way street. (Our $n = 33$ is the street at toy scale,
walkable in both directions.)

\textbf{18.} Testing residues (or building with B\'ezout):
$n \equiv 8 \pmod{15}$: the counts $8, 23, 38, 53, \dots$
Uniqueness modulo $15$: two solutions differ by a multiple of
$3$ and of $5$, hence of $15$ ($3$ and $5$ \omterm{def:g12:arith:gcd}{coprime}, Gauss).
The general with $1000$ soldiers announces ``$8$'' by three
quick line-ups --- the ancient headcount trick.

\textbf{19.} $10 \equiv 1 \pmod 9$ gives $10^k \equiv 1$, so
$\sum d_k 10^k \equiv \sum d_k$: a number and its digit sum
are congruent modulo $9$ (and modulo $3$). And
$10 \equiv -1 \pmod{11}$ gives
$\sum d_k 10^k \equiv \sum (-1)^k d_k$: the alternating rule.
Two childhood rules, one line each.

\textbf{20.} \omterm{def:g12:arith:congruence}{Congruences} turned remainders into an arithmetic
(Part I); B\'ezout minted the inverses that solve linear
\omterm{def:g12:arith:congruence}{congruences} and RSA's $d$ (questions 4, 13); Fermat's little
theorem opened and closed the lock (questions 15--16); gluing
\omterm{def:g12:arith:gcd}{coprime} moduli counted soldiers and finished the proof
(questions 16, 18). Verdict on Hardy: the purest theorem he
knew now guards every purchase --- purity, given time, is the
most applicable thing there is.
\end{solution}
