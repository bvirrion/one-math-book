\begin{solution}{exo:g12:trigo:1}
$\cos\frac{2\pi}{3} = \cos\left(\pi - \frac\pi3\right) = -\cos\frac\pi3
= -\frac12$.

$\sin\frac{5\pi}{6} = \sin\left(\pi - \frac\pi6\right) = \sin\frac\pi6
= \frac12$.

$\cos\frac{7\pi}{4} = \cos\left(2\pi - \frac{\pi}{4}\right)
= \cos\left(-\frac\pi4\right) = \frac{\sqrt2}{2}$.

$\sin\left(-\frac\pi3\right) = -\sin\frac\pi3 = -\frac{\sqrt3}{2}$.
\end{solution}

\begin{solution}{exo:g12:trigo:2}
$f'(x) = 2\sin x\cos x = \sin 2x$ (chain rule).

$g'(x) = -3\sin(3x) + \sin x + x\cos x$ (chain and product rules).

Quotient rule:
\[
h'(x) = \frac{\cos x\,(2 + \cos x) - \sin x\,(-\sin x)}{(2+\cos x)^2}
= \frac{2\cos x + \cos^2 x + \sin^2 x}{(2+\cos x)^2}
= \frac{2\cos x + 1}{(2+\cos x)^2}.
\]
\end{solution}

\begin{solution}{exo:g12:trigo:3}
\emph{(a)} A particular solution of $\sin x = \frac{\sqrt3}{2}$ is
$\frac\pi3$. General solutions: $x = \frac\pi3 + 2k\pi$ or
$x = \pi - \frac\pi3 + 2k\pi = \frac{2\pi}{3} + 2k\pi$. In
$\intoc{-\pi}{\pi}$: $x \in \left\{\frac\pi3, \frac{2\pi}{3}\right\}$.

\emph{(b)} $\cos 2x = \frac{\sqrt2}{2} = \cos\frac\pi4$ gives
$2x = \pm\frac\pi4 + 2k\pi$, so $x = \pm\frac\pi8 + k\pi$. In
$\intoc{-\pi}{\pi}$:
$x \in \left\{-\frac{7\pi}{8}, -\frac\pi8, \frac\pi8, \frac{7\pi}{8}\right\}$.
\end{solution}

\begin{solution}{exo:g12:trigo:4}
With $a = \frac\pi3$, $b = \frac\pi4$:
\[
\cos\frac{\pi}{12} = \cos a\cos b + \sin a \sin b
= \frac12\cdot\frac{\sqrt2}{2} + \frac{\sqrt3}{2}\cdot\frac{\sqrt2}{2}
= \frac{\sqrt2 + \sqrt6}{4},
\]
\[
\sin\frac{\pi}{12} = \sin a\cos b - \cos a \sin b
= \frac{\sqrt3}{2}\cdot\frac{\sqrt2}{2} - \frac12\cdot\frac{\sqrt2}{2}
= \frac{\sqrt6 - \sqrt2}{4}.
\]
\end{solution}

\begin{solution}{exo:g12:trigo:5}
Set $s = \sin x$: $2s^2 - s - 1 = (2s + 1)(s - 1) \geq 0$ iff
$s \leq -\frac12$ or $s = 1$.

$\sin x = 1$ gives $x = \frac\pi2$. $\sin x \leq -\frac12$: on the unit
circle, the arc below the horizontal line $y = -\frac12$, which in
$\intcc{0}{2\pi}$ is $x \in \intcc{\frac{7\pi}{6}}{\frac{11\pi}{6}}$.
Hence
\[
S = \left\{\frac{\pi}{2}\right\} \cup
\intcc{\frac{7\pi}{6}}{\frac{11\pi}{6}} .
\]
\end{solution}

\begin{solution}{exo:g12:trigo:6}
$\dfrac{\sin 3x}{x} = 3\,\dfrac{\sin 3x}{3x} \to 3 \cdot 1 = 3$
(composition with $u = 3x \to 0$).

Using $1 - \cos x = \dfrac{\sin^2 x}{1 + \cos x}$:
\[
\frac{1 - \cos x}{x^2}
= \left(\frac{\sin x}{x}\right)^{\!2} \frac{1}{1 + \cos x}
\xrightarrow[x\to0]{} 1 \cdot \frac12 = \frac12 .
\]

Substituting $u = \frac1x \to 0^+$:
$x \sin\frac1x = \frac{\sin u}{u} \to 1$.
\end{solution}

\begin{solution}{exo:g12:trigo:7}
$f'(x) = 1 - \sin x \geq 0$ for all $x$, with equality exactly when
$\sin x = 1$, \emph{i.e.} $x = \frac\pi2 + 2k\pi$. On $\intcc{0}{2\pi}$, $f$
is increasing from $f(0) = 1$ to $f(2\pi) = 2\pi + 1$ (the zero of $f'$ at
$\frac{\pi}{2}$ is isolated, so the increase is even strict); minimum $1$ at
$0$, maximum $2\pi + 1$ at $2\pi$. On $\R$, $f' \geq 0$ everywhere with only
isolated zeros, so $f$ is (strictly) increasing by
\cref{thm:g12:deriv:variations}, although $f'$ vanishes at every
$\frac\pi2 + 2k\pi$.
\end{solution}

\begin{solution}{exo:g12:trigo:8}
\emph{1.} Let $R = \sqrt{A^2 + B^2} > 0$. The point
$\left(\frac AR, \frac BR\right)$ lies on the unit circle, so there exists
$\varphi$ with $\cos\varphi = \frac AR$ and $\sin\varphi = \frac BR$. Then
\[
R\cos(x - \varphi) = R\bigl(\cos x\cos\varphi + \sin x \sin\varphi\bigr)
= A\cos x + B\sin x = f(x).
\]

\emph{2.} Since $\cos$ has range $\intcc{-1}{1}$, $f$ has maximum $R$ and
minimum $-R$. For $\cos x + \sin x$: $A = B = 1$, $R = \sqrt2$,
$\varphi = \frac\pi4$, so the equation becomes
$\sqrt2\cos\left(x - \frac\pi4\right) = 1$, \emph{i.e.}
$\cos\left(x - \frac\pi4\right) = \frac{\sqrt2}{2}$, giving
$x - \frac\pi4 = \pm\frac\pi4 + 2k\pi$: in $\intoc{-\pi}{\pi}$,
$x \in \left\{0, \frac\pi2\right\}$.
\end{solution}

\begin{solution}{exo:g12:trigo:9}
$(\sin)'' = (\cos)' = -\sin$ and $(\cos)'' = (-\sin)' = -\cos$: both solve
$y'' = -y$.

Let $g = y - a\cos - b\sin$. Then $g'' = -g$ (linearity of differentiation),
$g(0) = y(0) - a = 0$ and $g'(0) = y'(0) - b = 0$. Set
$E = g^2 + (g')^2$. Then
\[
E' = 2g g' + 2g' g'' = 2g g' - 2g' g = 0,
\]
so $E$ is constant equal to $E(0) = 0$. A sum of two squares vanishing
identically forces $g = 0$, hence $y = a\cos + b\sin$.
\end{solution}
