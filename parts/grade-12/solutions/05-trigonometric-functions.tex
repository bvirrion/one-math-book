\section*{Chapter \ref{ch:g12:trigo} --- Trigonometric Functions}

\begin{solution}{exo:g12:trigo:1}
$\cos\frac{2\pi}{3} = \cos\left(\pi - \frac\pi3\right) = -\cos\frac\pi3
= -\frac12$.

$\sin\frac{5\pi}{6} = \sin\left(\pi - \frac\pi6\right) = \sin\frac\pi6
= \frac12$.

$\cos\frac{7\pi}{4} = \cos\left(2\pi - \frac{\pi}{4}\right)
= \cos\left(-\frac\pi4\right) = \frac{\sqrt2}{2}$.

$\sin\left(-\frac\pi3\right) = -\sin\frac\pi3 = -\frac{\sqrt3}{2}$.
\end{solution}

\begin{solution}{exo:g12:trigo:2}
$f'(x) = 2\sin x\cos x = \sin 2x$ (chain rule).

$g'(x) = -3\sin(3x) + \sin x + x\cos x$ (chain and product rules).

Quotient rule:
\[
h'(x) = \frac{\cos x\,(2 + \cos x) - \sin x\,(-\sin x)}{(2+\cos x)^2}
= \frac{2\cos x + \cos^2 x + \sin^2 x}{(2+\cos x)^2}
= \frac{2\cos x + 1}{(2+\cos x)^2}.
\]
\end{solution}

\begin{solution}{exo:g12:trigo:3}
\emph{(a)} A particular solution of $\sin x = \frac{\sqrt3}{2}$ is
$\frac\pi3$. General solutions: $x = \frac\pi3 + 2k\pi$ or
$x = \pi - \frac\pi3 + 2k\pi = \frac{2\pi}{3} + 2k\pi$. In
$\intoc{-\pi}{\pi}$: $x \in \left\{\frac\pi3, \frac{2\pi}{3}\right\}$.

\emph{(b)} $\cos 2x = \frac{\sqrt2}{2} = \cos\frac\pi4$ gives
$2x = \pm\frac\pi4 + 2k\pi$, so $x = \pm\frac\pi8 + k\pi$. In
$\intoc{-\pi}{\pi}$:
$x \in \left\{-\frac{7\pi}{8}, -\frac\pi8, \frac\pi8, \frac{7\pi}{8}\right\}$.
\end{solution}

\begin{solution}{exo:g12:trigo:4}
With $a = \frac\pi3$, $b = \frac\pi4$:
\[
\cos\frac{\pi}{12} = \cos a\cos b + \sin a \sin b
= \frac12\cdot\frac{\sqrt2}{2} + \frac{\sqrt3}{2}\cdot\frac{\sqrt2}{2}
= \frac{\sqrt2 + \sqrt6}{4},
\]
\[
\sin\frac{\pi}{12} = \sin a\cos b - \cos a \sin b
= \frac{\sqrt3}{2}\cdot\frac{\sqrt2}{2} - \frac12\cdot\frac{\sqrt2}{2}
= \frac{\sqrt6 - \sqrt2}{4}.
\]
\end{solution}

\begin{solution}{exo:g12:trigo:5}
Set $s = \sin x$: $2s^2 - s - 1 = (2s + 1)(s - 1) \geq 0$ iff
$s \leq -\frac12$ or $s = 1$.

$\sin x = 1$ gives $x = \frac\pi2$. $\sin x \leq -\frac12$: on the unit
circle, the arc below the horizontal line $y = -\frac12$, which in
$\intcc{0}{2\pi}$ is $x \in \intcc{\frac{7\pi}{6}}{\frac{11\pi}{6}}$.
Hence
\[
S = \left\{\frac{\pi}{2}\right\} \cup
\intcc{\frac{7\pi}{6}}{\frac{11\pi}{6}} .
\]
\end{solution}

\begin{solution}{exo:g12:trigo:6}
$\dfrac{\sin 3x}{x} = 3\,\dfrac{\sin 3x}{3x} \to 3 \cdot 1 = 3$
(composition with $u = 3x \to 0$).

Using $1 - \cos x = \dfrac{\sin^2 x}{1 + \cos x}$:
\[
\frac{1 - \cos x}{x^2}
= \left(\frac{\sin x}{x}\right)^{\!2} \frac{1}{1 + \cos x}
\xrightarrow[x\to0]{} 1 \cdot \frac12 = \frac12 .
\]

Substituting $u = \frac1x \to 0^+$:
$x \sin\frac1x = \frac{\sin u}{u} \to 1$.
\end{solution}

\begin{solution}{exo:g12:trigo:7}
$f'(x) = 1 - \sin x \geq 0$ for all $x$, with equality exactly when
$\sin x = 1$, \emph{i.e.} $x = \frac\pi2 + 2k\pi$. On $\intcc{0}{2\pi}$, $f$
is \omterm{def:g12:seq:monotonic}{increasing} from $f(0) = 1$ to $f(2\pi) = 2\pi + 1$ (the zero of $f'$ at
$\frac{\pi}{2}$ is isolated, so the increase is even strict); \omterm{def:g10:functions:extrema}{minimum} $1$ at
$0$, \omterm{def:g10:functions:extrema}{maximum} $2\pi + 1$ at $2\pi$. On $\R$, $f' \geq 0$ everywhere with only
isolated zeros, so $f$ is (strictly) \omterm{def:g12:seq:monotonic}{increasing} by
\cref{thm:g12:deriv:variations}, although $f'$ vanishes at every
$\frac\pi2 + 2k\pi$.
\end{solution}

\begin{solution}{exo:g12:trigo:8}
\emph{1.} Let $R = \sqrt{A^2 + B^2} > 0$. The point
$\left(\frac AR, \frac BR\right)$ lies on the unit circle, so there exists
$\varphi$ with $\cos\varphi = \frac AR$ and $\sin\varphi = \frac BR$. Then
\[
R\cos(x - \varphi) = R\bigl(\cos x\cos\varphi + \sin x \sin\varphi\bigr)
= A\cos x + B\sin x = f(x).
\]

\emph{2.} Since $\cos$ has range $\intcc{-1}{1}$, $f$ has \omterm{def:g10:functions:extrema}{maximum} $R$ and
\omterm{def:g10:functions:extrema}{minimum} $-R$. For $\cos x + \sin x$: $A = B = 1$, $R = \sqrt2$,
$\varphi = \frac\pi4$, so the \omterm{def:g10:algebra:equation}{equation} becomes
$\sqrt2\cos\left(x - \frac\pi4\right) = 1$, \emph{i.e.}
$\cos\left(x - \frac\pi4\right) = \frac{\sqrt2}{2}$, giving
$x - \frac\pi4 = \pm\frac\pi4 + 2k\pi$: in $\intoc{-\pi}{\pi}$,
$x \in \left\{0, \frac\pi2\right\}$.
\end{solution}

\begin{solution}{exo:g12:trigo:9}
$(\sin)'' = (\cos)' = -\sin$ and $(\cos)'' = (-\sin)' = -\cos$: both solve
$y'' = -y$.

Let $g = y - a\cos - b\sin$. Then $g'' = -g$ (linearity of differentiation),
$g(0) = y(0) - a = 0$ and $g'(0) = y'(0) - b = 0$. Set
$E = g^2 + (g')^2$. Then
\[
E' = 2g g' + 2g' g'' = 2g g' - 2g' g = 0,
\]
so $E$ is constant equal to $E(0) = 0$. A sum of two squares vanishing
identically forces $g = 0$, hence $y = a\cos + b\sin$.
\end{solution}

\begin{solution}{pb:g12:trigo:1}
\textbf{1.} $0.99833\ldots$ and $0.99998\ldots$: creeping to
$1$.

\textbf{2.} Triangle $OAM$: base $1$, height $\sin x$: area
$\frac{\sin x}{2}$. Sector $OAM$: fraction
$\frac{x}{2\pi}$ of the unit disk: area $\frac x2$. Triangle
$OAT$: base $1$, height $\tan x$: area $\frac{\tan x}{2}$.

\textbf{3.} The triangle sits inside the sector inside the big
triangle: $\frac{\sin x}{2} < \frac x2 < \frac{\tan x}{2}$,
i.e.\ $\sin x < x < \tan x$. From $\sin x < x$:
$\frac{\sin x}{x} < 1$; and from
$x < \tan x = \frac{\sin x}{\cos x}$, multiplying by
$\frac{\cos x}{x} > 0$: $\cos x < \frac{\sin x}{x}$. As
$x \to 0^+$, $\cos x \to 1$: squeezed,
$\frac{\sin x}{x} \to 1$; and $\frac{\sin x}{x}$ is even, so
the two-sided limit is $1$.

\textbf{4.} $\frac{1 - \cos x}{x^2} =
\frac{1 - \cos^2 x}{x^2 (1 + \cos x)} =
\left(\frac{\sin x}{x}\right)^2 \frac{1}{1 + \cos x}
\to 1 \times \frac12$.

\textbf{5.} $\sin'(0) = \lim_{h \to 0} \frac{\sin h - 0}{h} =
1$: the whole differentiation of \omterm{def:g12:trigo:cossin}{sine} and \omterm{def:g12:trigo:cossin}{cosine} flows from
this one limit. Measured in degrees, the limit would be
$\frac{\pi}{180}$, and every \omterm{def:g12:deriv:derivative}{derivative} would drag that
constant: \omterm{def:g11:trigo:radian}{radians} are exactly the unit that makes the calculus
of oscillation clean.

\textbf{6.} $y' = -R\omega\sin(\omega t - \varphi)$,
$y'' = -R\omega^2\cos(\omega t - \varphi) = -\omega^2 y$.

\textbf{7.} $\omega = 2$. $y(0) = A = 3$;
$y'(0) = 2B = 8$: $B = 4$: $y = 3\cos 2t + 4\sin 2t$.

\textbf{8.} $R = \sqrt{9 + 16} = 5$; period
$\frac{2\pi}{2} = \pi$; phase
$\varphi = \arctan\frac43 \approx 0.927$:
$y = 5\cos(2t - 0.927)$.

\textbf{9.} $5\cos(2t - \varphi) = 0$ first when
$2t - \varphi = \frac\pi2$:
$t = \frac{\pi/2 + 0.927}{2} \approx 1.25$ s.

\textbf{10.} With $\sin\theta \approx \theta$:
$\theta'' = -\frac gL \theta$: harmonic with
$\omega = \sqrt{\frac gL}$, period
$T = \frac{2\pi}{\omega} = 2\pi\sqrt{\frac Lg}$. For
$L = 1$: $T \approx 2.006$ s. The period does not depend on
the amplitude (for small swings) --- Galileo's isochronism,
the fact that makes pendulum clocks possible.

\textbf{11.} $L = g\left(\frac{T}{2\pi}\right)^2 =
\frac{9.81}{\pi^2} \approx 0.994$ m: the seconds pendulum is
about $6$ mm short of the meridian meter. The Academy chose
the meridian (gravity varies from place to place, betraying
the pendulum); had $g$ been a whisker larger, our meter would
tick.

\textbf{12.} $\cos\left(x + \frac\pi2\right) =
\cos x \cos\frac\pi2 - \sin x \sin\frac\pi2 = -\sin x$;
$\sin\left(x + \frac\pi2\right) = \cos x$. So differentiating
\omterm{def:g12:trigo:cossin}{sine} gives \omterm{def:g12:trigo:cossin}{sine} shifted a quarter turn ($\cos$), and again
($-\sin$), and again, and home in four steps: differentiation
rotates the circle.

\textbf{13.} Adding the two expansions of $\cos(a \mp b)$:
$\cos(a-b) + \cos(a+b) = 2\cos a\cos b$. With $a = b = x$:
$1 + \cos 2x = 2\cos^2 x$, i.e.\
$\cos^2 x = \frac{1 + \cos 2x}{2}$.

\textbf{14.} With $p = \frac{p+q}{2} + \frac{p-q}{2}$ and
$q$ its mirror, \omterm{def:g10:algebra:expand}{expanding} and adding:
$\sin p + \sin q = 2\sin\frac{p+q}{2}\cos\frac{p-q}{2}$. For
$440$ and $444$ Hz: a tone at $442$ Hz whose loudness is
modulated by $\cos(2\pi \cdot 2t)$ --- the loudness peaks $4$
times per second (the envelope's \omterm{def:g10:numbers:abs}{absolute value}): four beats
a second, vanishing as the forks agree.

\textbf{15.} $\cos 3x = \cos 2x\cos x - \sin 2x \sin x =
(2\cos^2 x - 1)\cos x - 2\sin^2 x\cos x = 4\cos^3 x -
3\cos x$. At $x = 20^\circ$: $\cos 60^\circ = \frac12$, so
$c = \cos 20^\circ$ satisfies $8c^3 - 6c - 1 = 0$ --- a cubic
with no \omterm{def:g11:quad:discriminant}{square-root} solution: constructing $20^\circ$, i.e.\
trisecting $60^\circ$, is beyond ruler and compass.

\textbf{16.} $R = \sqrt{9 + 27} = 6$;
$\tan\varphi = \frac{3\sqrt3}{3} = \sqrt3$:
$\varphi = \frac\pi3$: $I(t) = 6\cos\left(100\pi t -
\frac\pi3\right)$: peak $6$ amperes, phase lag $60^\circ$.

\textbf{17.} $\sin^2 t = \frac{1 - \cos 2t}{2}$: period
$\pi$. $\sin 2t$ has period $\pi$, $\sin 3t$ has
$\frac{2\pi}{3}$: the sum repeats after the smallest common
multiple, $2\pi$.

\textbf{18.} An oscillation of period $1$ inside the shrinking
envelope $\pm\eu^{-t/10}$: each swing about $10\,\%$ lower. At
$t = 10$ the envelope is $\eu^{-1} \approx 0.368$ --- the
constant of \cref{pb:g12:exp:1} yet again. \omterm{def:g10:algebra:equation}{Equations} with such
damped solutions ($y'' + a y' + b y = 0$) are solved in the
differential \omterm{def:g10:algebra:equation}{equations} chapter.

\textbf{19.} A force pulling back proportionally to the
displacement gives $y'' = -\omega^2 y$; and
\cref{exo:g12:trigo:9} shows the solutions of that \omterm{def:g10:algebra:equation}{equation}
are \emph{exactly} the combinations of $\cos$ and $\sin$ ---
existence by exhibition, uniqueness by the exercise: vibration
has no choice but to be sinusoidal.

\textbf{20.} One limit ($\frac{\sin x}{x} \to 1$, by
sandwiched areas), two \omterm{def:g12:deriv:derivative}{derivatives} ($\sin' = \cos$,
$\cos' = -\sin$), one \omterm{def:g10:algebra:equation}{equation} ($y'' = -\omega^2 y$), a world
of clocks, strings, currents and tides. Asterisk: for wide
swings, $T = 4\sqrt{\frac Lg} \cdot
\frac{\pi/2}{M(1, \cos(\theta_0/2))}$ --- the
arithmetic--geometric \omterm{def:g11:stat:mean}{mean} of \cref{pb:g12:seq:1} computes the
elliptic integral in a handful of iterations: Gauss's diary
entry and Galileo's pendulum, one formula apart.
\end{solution}
