\section*{Chapter \ref{ch:g12:seq} --- Sequences}

\begin{solution}{exo:g12:seq:1}
Let $P(n)$ be the statement $\sum_{k=1}^{n} k^2 = \frac{n(n+1)(2n+1)}{6}$.
\emph{Base case:} for $n = 0$ both sides are $0$ (empty sum).
\emph{Inductive step:} assume $P(n)$. Then
\[
\sum_{k=1}^{n+1} k^2 = \frac{n(n+1)(2n+1)}{6} + (n+1)^2
= \frac{(n+1)\bigl(n(2n+1) + 6(n+1)\bigr)}{6}
= \frac{(n+1)(2n^2 + 7n + 6)}{6}.
\]
Since $2n^2 + 7n + 6 = (n+2)(2n+3)$, this is
$\frac{(n+1)(n+2)(2(n+1)+1)}{6}$, which is $P(n+1)$. By induction, $P(n)$
holds for all $n$.
\end{solution}

\begin{solution}{exo:g12:seq:2}
$a_{n+1} - a_n = \frac{n+2}{n+1} - \frac{n+1}{n}
= \frac{n(n+2) - (n+1)^2}{n(n+1)} = \frac{-1}{n(n+1)} < 0$: $(a_n)$ is
strictly \omterm{def:g10:functions:variations}{decreasing}.

$(b_n)$ has positive terms and
$\frac{b_{n+1}}{b_n} = \frac{2^{n+1}}{n+1}\cdot\frac{n}{2^n} =
\frac{2n}{n+1} \geq 1 \iff 2n \geq n+1 \iff n \geq 1$: $(b_n)$ is \omterm{def:g12:seq:monotonic}{increasing}
(strictly for $n \geq 2$).

$c_{n+1} - c_n = (n+1)^2 - 10(n+1) - n^2 + 10n = 2n - 9$, which is negative
for $n \leq 4$ and positive for $n \geq 5$: $(c_n)$ decreases up to
$c_5 = -25$, its \omterm{def:g10:functions:extrema}{minimum}, then increases. It is not \omterm{def:g12:seq:monotonic}{monotonic}.
\end{solution}

\begin{solution}{exo:g12:seq:3}
Factor dominant terms:
\[
u_n = \frac{n^2(3 - 1/n + 1/n^2)}{n^2(2 + 5/n^2)} \xrightarrow[n\to+\infty]{} \frac{3}{2}.
\]
Multiply by the conjugate:
\[
v_n = \frac{(n+1) - n}{\sqrt{n+1} + \sqrt{n}} = \frac{1}{\sqrt{n+1}+\sqrt{n}}
\xrightarrow[n\to+\infty]{} 0.
\]
Divide numerator and denominator by $3^n$:
\[
w_n = \frac{(2/3)^n - 1}{1 + (1/3)^n} \xrightarrow[n\to+\infty]{} \frac{0-1}{1+0} = -1,
\]
using $\lim q^n = 0$ for $\abs{q} < 1$.
\end{solution}

\begin{solution}{exo:g12:seq:4}
$u_n = 5 + 3n \to +\infty$ and $v_n = 8 \cdot (1/2)^n = 2^{3-n} \to 0$.
The sums are
\[
\sum_{k=0}^{n} u_k = (n+1)\,\frac{5 + (5+3n)}{2} = \frac{(n+1)(10+3n)}{2}
\xrightarrow[n\to+\infty]{} +\infty,
\]
\[
\sum_{k=0}^{n} v_k = 8\,\frac{1 - (1/2)^{n+1}}{1 - 1/2}
= 16\left(1 - \left(\tfrac{1}{2}\right)^{n+1}\right)
\xrightarrow[n\to+\infty]{} 16 .
\]
\end{solution}

\begin{solution}{exo:g12:seq:5}
Since $-1 \leq \cos n \leq 1$,
\[
\frac{n-1}{n+1} \leq \frac{n + \cos n}{n+1} \leq 1,
\]
and $\frac{n-1}{n+1} \to 1$, so the limit is $1$ by the squeeze theorem.

For the second limit, write
\[
0 \leq \frac{n!}{n^n}
= \frac{1}{n}\cdot\frac{2}{n}\cdots\frac{n}{n}
\leq \frac{1}{n},
\]
because every factor $\frac{k}{n}$ with $2 \leq k \leq n$ is at most $1$.
Since $\frac1n \to 0$, the squeeze theorem gives $\frac{n!}{n^n} \to 0$. (The
suggested \omterm{def:g12:seq:arith-geom}{geometric} bound also works: each factor with $k \leq n/2$ is at most
$\frac12$, giving the stronger bound $(1/2)^{\floor{n/2}}$.)
\end{solution}

\begin{solution}{exo:g12:seq:6}
\emph{1.} $u_0 = 0 \in \intcc{0}{2}$. If $0 \leq u_n \leq 2$, then
$2 \leq u_n + 2 \leq 4$, so $\sqrt{2} \leq u_{n+1} \leq 2$; in particular
$0 \leq u_{n+1} \leq 2$. By induction the property holds for all $n$.

\emph{2.} $u_{n+1} - u_n = \sqrt{u_n + 2} - u_n$. For $x \in \intcc{0}{2}$,
$\sqrt{x+2} \geq x \iff x + 2 \geq x^2 \iff (2-x)(x+1) \geq 0$, which is true.
Hence $(u_n)$ is \omterm{def:g12:seq:monotonic}{increasing}.

\emph{3.} \omterm{def:g12:seq:monotonic}{Increasing} and \omterm{def:g12:seq:bounded}{bounded above} by $2$, $(u_n)$ \omterm{def:g12:seq:limit}{converges} to some
$\ell \in \intcc{0}{2}$. Passing to the limit in $u_{n+1} = \sqrt{u_n + 2}$
(the map $x \mapsto \sqrt{x+2}$ is continuous) gives $\ell = \sqrt{\ell + 2}$,
so $\ell^2 - \ell - 2 = 0$, i.e.\ $\ell \in \{-1, 2\}$. Since $\ell \geq 0$,
$\lim u_n = 2$.
\end{solution}

\begin{solution}{exo:g12:seq:7}
\emph{1.} Between two doses, $40\%$ of the drug is eliminated, so the
quantity $u_n$ becomes $0.6\,u_n$; the next dose adds $1$ unit:
$u_{n+1} = 0.6\,u_n + 1$.

\emph{2.} $v_{n+1} = u_{n+1} - 2.5 = 0.6\,u_n + 1 - 2.5 = 0.6(u_n - 2.5)
= 0.6\,v_n$: $(v_n)$ is \omterm{def:g12:seq:arith-geom}{geometric} with ratio $0.6$ and first term
$v_0 = 1 - 2.5 = -1.5$. Hence $v_n = -1.5 \times 0.6^n$ and
\[
u_n = 2.5 - 1.5 \times 0.6^n .
\]

\emph{3.} Since $0.6^n \to 0$, $u_n \to 2.5$: the quantity of drug stabilizes
at $2.5$ units.
\end{solution}

\begin{solution}{exo:g12:seq:8}
\emph{1.} $u_0 = 3 > 1$. If $u_n > 1$, then $u_n + 2 > 0$ and
\[
u_{n+1} - 1 = \frac{4u_n - 1 - u_n - 2}{u_n + 2} = \frac{3(u_n - 1)}{u_n + 2} > 0 .
\]
By induction, $u_n > 1$ for all $n$ (and in particular $u_n + 2 \neq 0$, so
the \omterm{def:g12:seq:sequence}{sequence} is well defined).

\emph{2.} Using the identity above,
\[
v_{n+1} = \frac{1}{u_{n+1} - 1} = \frac{u_n + 2}{3(u_n - 1)}
= \frac{(u_n - 1) + 3}{3(u_n - 1)} = \frac{1}{3} + v_n .
\]
So $(v_n)$ is \omterm{def:g12:seq:arith-geom}{arithmetic} with \omterm{def:g11:seq:arithmetic}{common difference} $\frac13$ and
$v_0 = \frac{1}{u_0 - 1} = \frac12$.

\emph{3.} $v_n = \frac12 + \frac{n}{3}$, hence
$u_n = 1 + \frac{1}{v_n} = 1 + \frac{6}{3 + 2n}$. Since $v_n \to +\infty$,
$u_n \to 1$.
\end{solution}

\begin{solution}{exo:g12:seq:9}
\emph{1.} $H_{2n} - H_n = \sum_{k=n+1}^{2n} \frac{1}{k}$ is a sum of $n$
terms, each at least $\frac{1}{2n}$; hence
$H_{2n} - H_n \geq n \cdot \frac{1}{2n} = \frac12$.

\emph{2.} By induction on $k$: $H_{2^0} = H_1 = 1 \geq 1$. If
$H_{2^k} \geq 1 + \frac{k}{2}$, then applying point 1 with $n = 2^k$,
\[
H_{2^{k+1}} \geq H_{2^k} + \frac12 \geq 1 + \frac{k+1}{2}.
\]
The \omterm{def:g12:seq:sequence}{sequence} $(H_n)$ is \omterm{def:g12:seq:monotonic}{increasing} (each step adds $\frac{1}{n+1} > 0$) and
the subsequence $H_{2^k}$ is unbounded, so $(H_n)$ is not \omterm{def:g12:seq:bounded}{bounded above}.
\omterm{def:g12:seq:monotonic}{Increasing} and unbounded, it tends to $+\infty$
(\cref{thm:g12:seq:monotone}).
\end{solution}

\begin{solution}{exo:g12:seq:10}
\emph{1.} The \omterm{def:g12:seq:sequence}{sequence} $d_n = b_n - a_n$ satisfies
$d_{n+1} - d_n = (b_{n+1} - b_n) - (a_{n+1} - a_n) \leq 0$, so $(d_n)$ is
\omterm{def:g10:functions:variations}{decreasing}; since $d_n \to 0$, we get $d_n \geq 0$ for all $n$ (a \omterm{def:g10:functions:variations}{decreasing}
\omterm{def:g12:seq:sequence}{sequence} with a negative term would stay below it forever, preventing the
limit $0$). Hence $a_n \leq b_n$.

\emph{2.} From $a_n \leq b_n \leq b_0$, the \omterm{def:g12:seq:monotonic}{increasing} \omterm{def:g12:seq:sequence}{sequence} $(a_n)$ is
\omterm{def:g12:seq:bounded}{bounded above}, so it \omterm{def:g12:seq:limit}{converges} to some $\ell$. Similarly $(b_n)$, \omterm{def:g10:functions:variations}{decreasing}
and \omterm{def:g12:seq:bounded}{bounded below} by $a_0$, \omterm{def:g12:seq:limit}{converges} to some $\ell'$. Then
$\ell' - \ell = \lim (b_n - a_n) = 0$, so $\ell = \ell'$.

\emph{3.} $(a_n)$ is (strictly) \omterm{def:g12:seq:monotonic}{increasing} since
$a_{n+1} - a_n = \frac{1}{(n+1)!} > 0$. For $(b_n)$,
\[
b_{n+1} - b_n = \frac{1}{(n+1)!} + \frac{1}{(n+1)(n+1)!} - \frac{1}{n\,n!}
= \frac{n(n+1) + n - (n+1)^2}{n(n+1)(n+1)!}
= \frac{-1}{n(n+1)(n+1)!} < 0 ,
\]
so $(b_n)$ is \omterm{def:g10:functions:variations}{decreasing}. Finally $b_n - a_n = \frac{1}{n\,n!} \to 0$. The
two \omterm{def:g12:seq:sequence}{sequences} are adjacent, hence converge to a common limit.
\end{solution}

\begin{solution}{pb:g12:seq:1}
\textbf{1.} True for $n = 1$ ($1 = 1^2$). If
$1 + 3 + \dots + (2n - 1) = n^2$, then adding the next odd
number: $n^2 + (2n + 1) = (n + 1)^2$: heredity. By induction,
true for all $n \geq 1$.

\textbf{2.} $2^0 = 1 > 0$. If $2^n > n$, then
$2^{n+1} = 2 \cdot 2^n > 2n \geq n + 1$ for $n \geq 1$ (and
$n = 0$ checks directly): heredity, done.

\textbf{3.} $n = 0$: $1 \geq 1$. If
$(1 + x)^n \geq 1 + nx$, multiply by $1 + x \geq 1 > 0$:
$(1 + x)^{n+1} \geq (1 + nx)(1 + x) = 1 + (n + 1)x + nx^2
\geq 1 + (n + 1)x$.

\textbf{4.} The step from $n = 1$ to $n = 2$: among two
marbles, ``the first $n$'' and ``the last $n$'' are two
\emph{disjoint} single marbles --- no common marble bridges the
two groups, so nothing forces their colors to agree. The
heredity argument silently requires the two groups to overlap,
true only from $n \geq 2$; with the base case $n = 1$ the chain
never starts.

\textbf{5.} $4^0 - 1 = 0 = 3 \times 0$. If $4^n - 1 = 3k$,
then $4^{n+1} - 1 = 4(4^n - 1) + 3 = 3(4k + 1)$: heredity.

\textbf{6.} $x_1 = \frac32$, $x_2 = \frac{17}{12}$,
$x_3 = \frac{577}{408}$.

\textbf{7.} $x_{n+1}^2 - 2 = \frac{(x_n^2 + 2)^2 - 8x_n^2}
{4x_n^2} = \frac{(x_n^2 - 2)^2}{4 x_n^2}$: a square over a
positive, hence $\geq 0$, and $> 0$ whenever $x_n^2 \neq 2$.
Induction: $x_0 = 2 > 0$ with $x_0^2 = 4 > 2$; if $x_n > 0$
and $x_n^2 > 2$, then $x_{n+1}$ (an average of positives) is
positive and $x_{n+1}^2 - 2 > 0$.

\textbf{8.} $x_{n+1} - x_n = \frac{2 - x_n^2}{2x_n} < 0$ by
question 7: strictly \omterm{def:g10:functions:variations}{decreasing}.

\textbf{9.} \omterm{def:g10:functions:variations}{Decreasing} and \omterm{def:g12:seq:bounded}{bounded below} (by $1$, since
$x_n^2 > 2 > 1$ and $x_n > 0$): by the monotone convergence
theorem, $(x_n)$ \omterm{def:g12:seq:limit}{converges} to some $L \geq 1$.

\textbf{10.} Limits respect the algebra: from
$x_{n+1} = \frac12\left(x_n + \frac{2}{x_n}\right)$ and
$x_n \to L \geq 1 > 0$:
$L = \frac12\left(L + \frac2L\right)$, so $L^2 = 2$ and, $L$
being positive, $L = \sqrt2$. Verdict: convergence proved,
limit identified --- Heron acquitted with honors.

\textbf{11.} $x_{n+1} - \sqrt2 = \frac{x_n^2 - 2\sqrt2\,x_n +
2}{2x_n} = \frac{(x_n - \sqrt2)^2}{2x_n}$: exactly
$e_{n+1} = \frac{e_n^2}{2x_n}$, and $x_n > \sqrt2$ gives
$e_{n+1} \leq \frac{e_n^2}{2\sqrt2}$. Squared error: each step
doubles the number of correct decimals, as observed since
grade 9.

\textbf{12.} $e_0 \approx 0.5858$, $e_1 \approx 0.0858$,
$e_2 \approx 0.00245$, $e_3 \approx 2.1 \times 10^{-6}$.
Ratios $\frac{e_1}{e_0^2} \approx 0.25 = \frac{1}{2x_0}$;
$\frac{e_2}{e_1^2} \approx 0.333 = \frac{1}{2x_1}$;
$\frac{e_3}{e_2^2} \approx 0.353 \approx \frac{1}{2x_2}$: the
theorem in action.

\textbf{13.} $H_{2^{18}} \geq 1 + 9 = 10$: about
$260\,000$ terms ($2^{18} = 262\,144$) merely to pass $10$ ---
divergence at a crawl (and $H_n > 100$ would need more terms
than atoms in any library).

\textbf{14.} $S_n = 2 - \frac{1}{2^n}$ (\omterm{def:g12:seq:arith-geom}{geometric} sum), and
$\frac{1}{2^n} \to 0$ (\cref{thm:g12:seq:geometric}), so
$S_n \to 2$: the endlessly bitten chocolate bar tends to,
without ever reaching, the whole --- now in the official
language of limits.

\textbf{15.} For $k \geq 2$:
$\frac{1}{k^2} \leq \frac{1}{k(k-1)} = \frac{1}{k-1} -
\frac1k$, so
$S_n \leq 1 + \left(1 - \frac1n\right) < 2$: \omterm{def:g12:seq:monotonic}{increasing} and
\omterm{def:g12:seq:bounded}{bounded above}, hence convergent (monotone convergence). Euler
later named the limit: $\frac{\pi^2}{6}$.

\textbf{16.} Terms tending to $0$ is \emph{necessary} for the
sums to settle but decides nothing: the harmonic terms
$\frac1n \to 0$ yet the sums blow up; the terms
$\frac{1}{n^2} \to 0$ and the sums converge. How \emph{fast}
the terms die is the whole question --- the theory of series,
built in the university volumes.

\textbf{17.} $a_1 = \sqrt2 \approx 1.41421$, $b_1 = 1.5$;
$a_2 \approx 1.45648$, $b_2 \approx 1.45711$: two iterations
already agree to three decimals --- staggering speed.

\textbf{18.} $b_{n+1} - a_{n+1} = \frac{a_n + b_n}{2} -
\sqrt{a_n b_n} = \frac{(\sqrt{b_n} - \sqrt{a_n})^2}{2} \geq
0$: means stay ordered. $(a_n)$ increases:
$a_{n+1} = \sqrt{a_n b_n} \geq \sqrt{a_n \cdot a_n} = a_n$;
$(b_n)$ decreases symmetrically.

\textbf{19.} $\frac{b_{n+1} - a_{n+1}}{b_n - a_n} =
\frac{(\sqrt{b_n} - \sqrt{a_n})^2}
{2(\sqrt{b_n} - \sqrt{a_n})(\sqrt{b_n} + \sqrt{a_n})}
= \frac{\sqrt{b_n} - \sqrt{a_n}}{2(\sqrt{b_n} + \sqrt{a_n})}
\leq \frac12$: the gap at least halves, so $b_n - a_n \to 0$;
with question 18, the \omterm{def:g12:seq:sequence}{sequences} are adjacent and share a limit
$M(1, 2)$.

\textbf{20.} Iteration three gives
$a_3 \approx b_3 \approx 1.456791$: $M(1, 2) \approx
1.456791$ in three turns of the crank (the gap roughly
\emph{squares}, like Heron's). Speed ranking of the problem's
\omterm{def:g12:seq:sequence}{sequences}, slowest to fastest: the harmonic sums (glacial
divergence), the \omterm{def:g12:seq:arith-geom}{geometric} sums (error halved each step),
Heron and the AGM (error squared each step) --- and it was the
AGM's unearthly speed that told Gauss he had struck a new vein
of analysis.
\end{solution}
