\section*{Chapter \ref{ch:g12:condprob} --- Conditional Probability and Independence}

\begin{solution}{exo:g12:condprob:1}
There are $12$ face cards, of which $4$ are kings:
$\pcond{\text{face}}{\text{king}} = \frac{4}{12} = \frac13$.

\omterm{def:g12:condprob:indep}{Independence}: $\P(\text{king} \cap \text{heart}) = \frac{1}{52}$ (the king
of hearts), and
$\P(\text{king})\,\P(\text{heart}) = \frac{4}{52}\times\frac{13}{52}
= \frac{1}{52}$. Equal: the \omterm{def:g11:prob:model}{events} are \omterm{def:g12:condprob:indep}{independent}.
\end{solution}

\begin{solution}{exo:g12:condprob:2}
\emph{1.} First branch: red $\frac58$, blue $\frac38$; second branches
(without replacement): after red, red $\frac47$ / blue $\frac37$; after
blue, red $\frac57$ / blue $\frac27$.

\emph{2.} $\P(RR) = \frac58 \times \frac47 = \frac{5}{14}$. By total
\omterm{def:g10:proba:distribution}{probability},
\[
\P(\text{2nd red}) = \frac58\cdot\frac47 + \frac38\cdot\frac57
= \frac{20 + 15}{56} = \frac58 .
\]
(The same as the first draw --- by symmetry, the second ball is a uniformly
random ball of the urn.)
\end{solution}

\begin{solution}{exo:g12:condprob:3}
$\P(A) = \frac{6}{36} = \frac16$ (six pairs sum to $7$),
$\P(B) = \frac16$, and $A \cap B = \{(3,4)\}$ has \omterm{def:g10:proba:distribution}{probability}
$\frac{1}{36} = \P(A)\P(B)$: $A$ and $B$ are \omterm{def:g12:condprob:indep}{independent}. (The sum $7$ is
special: whatever the first die, exactly one value of the second gives it.)

$\P(C) = \frac{5}{36}$ and $B \cap C = \{(3,3)\}$:
$\P(B \cap C) = \frac{1}{36} \neq \frac16 \times \frac{5}{36}
= \frac{5}{216}$. Not \omterm{def:g12:condprob:indep}{independent}.
\end{solution}

\begin{solution}{exo:g12:condprob:4}
\emph{1.} Total \omterm{def:g10:proba:distribution}{probability} with the partition by machine:
\[
\P(D) = 0.5\times0.01 + 0.3\times0.02 + 0.2\times0.04
= 0.005 + 0.006 + 0.008 = 0.019 = 1.9\% .
\]

\emph{2.} Bayes:
$\pcond{D}{M_3} = \dfrac{0.008}{0.019} = \dfrac{8}{19} \approx 0.42$. The
machine producing only a fifth of the output accounts for over $40\%$ of
the defects.
\end{solution}

\begin{solution}{exo:g12:condprob:5}
The condition is
\[
\frac{0.99\,p}{0.99\,p + 0.05\,(1-p)} \geq 0.9 .
\]
The denominator is positive, so this reads
$0.99p \geq 0.891p + 0.045 - 0.045p$, \emph{i.e.} $0.144\,p \geq 0.045$,
\emph{i.e.}
\[
p \geq \frac{0.045}{0.144} = 0.3125 .
\]
The test alone reaches $90\%$ certainty only if the disease already affects
more than $31\%$ of the tested population --- which is why mass screening
of rare diseases requires confirmation tests.
\end{solution}

\begin{solution}{exo:g12:condprob:6}
\emph{1.} Three \omterm{def:g12:seq:sequence}{sequences} realize $E$ (HHT, HTH, THH), each of \omterm{def:g10:proba:distribution}{probability}
$p^2(1-p)$ by \omterm{def:g12:condprob:indep}{independence}: $\P(E) = 3p^2(1 - p)$.

\emph{2.} $f(p) = 3p^2 - 3p^3$ has $f'(p) = 6p - 9p^2 = 3p(2 - 3p)$,
positive then negative on $\intoo{0}{1}$: \omterm{def:g10:functions:extrema}{maximum} at $p = \frac23$, where
$\P(E) = 3\cdot\frac49\cdot\frac13 = \frac49$.
\end{solution}

\begin{solution}{exo:g12:condprob:7}
Say you picked door 1 and the host opened door 3 (\omterm{def:g11:prob:model}{event} $H_3$). With
$B_i$ = ``prize behind door $i$'', $\P(B_i) = \frac13$ and
\[
\pcond{B_1}{H_3} = \frac12, \qquad
\pcond{B_2}{H_3} = 1, \qquad
\pcond{B_3}{H_3} = 0
\]
(if the prize is behind your door, the host chooses between doors 2 and 3
at random; if it is behind door 2, he is forced to open 3). Bayes:
\[
\pcond{H_3}{B_1}
= \frac{\frac13\cdot\frac12}{\frac13\cdot\frac12 + \frac13\cdot1 + 0}
= \frac{1/6}{1/2} = \frac13,
\qquad
\pcond{H_3}{B_2} = \frac{\frac13}{\frac12} = \frac23 .
\]
Staying wins with \omterm{def:g10:proba:distribution}{probability} $\frac13$, switching with \omterm{def:g10:proba:distribution}{probability}
$\frac23$.
\end{solution}

\begin{solution}{exo:g12:condprob:8}
\emph{1.} The majority is wrong when at least two of the three copies are
flipped:
\[
3\varepsilon^2(1-\varepsilon) + \varepsilon^3
= 3(0.01)(0.9) + 0.001 = 0.028,
\]
much smaller than $\varepsilon = 0.1$: repetition coding works.

\emph{2.} If $1$ was sent (as $111$), receiving $101$ requires exactly one
flip: \omterm{def:g10:proba:distribution}{probability} $\varepsilon(1-\varepsilon)^2 = 0.081$. If $0$ was sent
($000$), two flips are needed: $\varepsilon^2(1-\varepsilon) = 0.009$.
Bayes with equal priors:
\[
\pcond{101}{\,\text{sent }1} = \frac{0.081}{0.081 + 0.009} = 0.9 .
\]
The majority decoding ($101 \mapsto 1$) is indeed the more probable guess.
\end{solution}

\begin{solution}{pb:g12:condprob:1}
\textbf{1.} $\P(\text{red}) = \frac12 \cdot \frac23 + \frac12
\cdot \frac14 = \frac13 + \frac18 = \frac{11}{24}$.

\textbf{2.} $\P(\text{I} \mid \text{red}) =
\dfrac{\frac12 \cdot \frac23}{\frac{11}{24}} =
\dfrac{1/3}{11/24} = \dfrac{8}{11}$.

\textbf{3.} $\P(\text{even} \cap \leq 4) = \P(\{2, 4\}) =
\frac13$ and $\P(\text{even})\P(\leq 4) = \frac12 \cdot
\frac23 = \frac13$: \omterm{def:g12:condprob:indep}{independent} (unexpectedly!). For
$\leq 3$: $\P(\{2\}) = \frac16$ against
$\frac12 \cdot \frac12 = \frac14$: dependent.

\textbf{4.} The \omterm{def:g10:proba:operations}{complements} of
\omterm{def:g12:condprob:indep}{independent}
\omterm{def:g11:prob:model}{events} are themselves
\omterm{def:g12:condprob:indep}{independent}, so
\[
\P(\text{none}) = 0.1 \times 0.1 = 0.01,
\qquad
\P(\text{at least one}) = 0.99 .
\]

\textbf{5.} $\frac{4}{52} \times \frac{3}{51} =
\frac{1}{221}$.

\textbf{6.} Sick: $100$; healthy: $99\,900$. Positives:
$99$ true (99\,\% of $100$) and $4\,995$ false (5\,\% of
$99\,900$): $5\,094$ positives in all.

\textbf{7.} $\P(\text{sick} \mid +) = \frac{99}{5\,094}
\approx 1.9\,\%$: of the people the test flags, fewer than
one in fifty is actually sick.

\textbf{8.} $\dfrac{0.001 \times 0.99}{0.001 \times 0.99 +
0.999 \times 0.05} = \dfrac{0.00099}{0.05094} \approx 0.019$:
the counts and the formula agree.

\textbf{9.} Confused: $\P(+ \mid \text{sick}) = 99\,\%$
(the test's advertised quality) with
$\P(\text{sick} \mid +)$ (the patient's actual question). With
a base rate of $\frac{1}{1000}$, the sick are so rare that
even a $5\,\%$ leak from the huge healthy majority swamps the
true positives fifty to one.

\textbf{10.} $\dfrac{0.019 \times 0.99}{0.019 \times 0.99 +
0.981 \times 0.05} \approx 28\,\%$. Each \omterm{def:g12:condprob:indep}{independent} positive
multiplies the odds by the same factor: evidence accumulates,
and a third positive would push past $90\,\%$ --- which is
why confirmatory tests exist.

\textbf{11.} $\dfrac{0.05 \times 0.99}{0.05 \times 0.99 +
0.95 \times 0.05} \approx 51\,\%$: on a high-risk group the
same test is informative. Moral: screen where the base rate
is substantial; mass-screening a rare condition manufactures
false alarms by the thousand.

\textbf{12.} Expected innocent matches:
$10^7 \times 10^{-6} = 10$. Among the roughly $11$ matching
people ($10$ innocent, $1$ guilty --- if the culprit is in
the city), the suspect found \emph{by the match alone} is
innocent with \omterm{def:g10:proba:distribution}{probability} about $\frac{10}{11} \approx
91\,\%$.

\textbf{13.} The prosecutor quoted
$\P(\text{match} \mid \text{innocent}) = 10^{-6}$; the court
needed $\P(\text{innocent} \mid \text{match})$ --- question
9's confusion under oath. (And in the real case that
popularized these errors, a second blunder compounded it:
see question 14.)

\textbf{14.} Naive squaring: $\left(\frac{1}{8500}\right)^2
\approx \frac{1}{72 \text{ million}}$. With the second
tragedy five times likelier given the first:
$\frac{1}{8500} \times \frac{5}{8500} \approx
\frac{1}{14.5 \text{ million}}$ --- five times larger, and
still ignoring that the \emph{alternative} hypothesis must be
weighed by Bayes too. Lesson: multiplying probabilities
requires \omterm{def:g12:condprob:indep}{independence}, and shared causes (genetics,
environment) destroy it.

\textbf{15.} Bayes multiplies the likelihood ratios of
\emph{all} the evidence into one posterior: the match, the
alibi, the fibers, each shifting the odds. The juror's
sentence: \emph{ask how probable the evidence is under both
hypotheses, and never mistake the one conditional for its
reverse}.

\textbf{16.} $\dfrac{0.4 \times 0.15}{0.4 \times 0.15 +
0.6 \times 0.005} = \dfrac{0.060}{0.063} \approx 95\,\%$
spam: one word nearly settles it.

\textbf{17.} Five copies fail when $3$, $4$ or $5$ flip:
$\binom53 \varepsilon^3(1-\varepsilon)^2 + \binom54
\varepsilon^4 (1 - \varepsilon) + \varepsilon^5 \approx
0.0086$ --- against $0.028$ for three copies: repetition
buys reliability at the price of bandwidth.

\textbf{18.} Even flips: $\binom40 0.9^4 + \binom42 0.9^2
0.1^2 + \binom44 0.1^4 = 0.6561 + 0.0486 + 0.0001 = 0.7048$;
and $\frac{1 + 0.8^4}{2} = \frac{1 + 0.4096}{2} = 0.7048$:
the trick formula tracks the bias $(1 - 2\varepsilon)$
through each relay.

\textbf{19.} Spammy words travel in packs --- a message with
``lottery'' very likely contains ``win'', so their joint
evidence is weaker than the product pretends. The filter
still works because the exaggeration usually pushes in the
\emph{right direction}: classification needs the correct side
of $50\,\%$, not the correct posterior.

\textbf{20.} Clinic: a rare condition turns a strong test
into a weak verdict --- update from the base rate, then
accumulate. Courtroom: a rare match in a large city indicts
the city, not the man --- likelihoods are not verdicts.
Inbox: many weak clues, multiplied, sort the mail --- even
naively. The three traps: forget the prior and be fooled;
fake \omterm{def:g12:condprob:indep}{independence} and be very fooled; read evidence piecemeal
and never know what you know.
\end{solution}
