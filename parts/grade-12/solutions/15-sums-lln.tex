\section*{Chapter \ref{ch:g12:sums} --- Sums of Random Variables and the Law of Large Numbers}

\begin{solution}{exo:g12:sums:1}
Each die has \omterm{def:g12:randvar:exp}{expectation}
$\frac{1 + 2 + \dots + 6}{6} = \frac{7}{2}$, so by linearity
$\E(S) = \frac72 + \frac72 = 7$. The dice are \omterm{def:g12:sums:indep}{independent}, so
$\V(S) = \frac{35}{12} + \frac{35}{12} = \frac{35}{6} \approx 5.83$.
\end{solution}

\begin{solution}{exo:g12:sums:2}
$\E(X) = 50$ and $\V(X) = 100 \times 0.25 = 25$.

Markov ($X \geq 0$): $\P(X \geq 75) \leq \frac{50}{75} = \frac23$.

Bienaymé--Chebyshev:
$\P(\abs{X - 50} \geq 25) \leq \frac{25}{25^2} = 0.04$ --- and this even
bounds the \emph{two-sided} \omterm{def:g11:prob:model}{event}, of which $X \geq 75$ is half. Chebyshev
is far sharper here because it exploits the \omterm{def:g12:randvar:exp}{variance}, not just the \omterm{def:g11:stat:mean}{mean}.
(The true value of $\P(X \geq 75)$ is smaller than $10^{-6}$: both bounds
are crude.)
\end{solution}

\begin{solution}{exo:g12:sums:3}
$F_n$ is the \omterm{prop:g12:sums:mean}{sample mean} of $n$ Bernoulli$\left(\frac12\right)$ variables,
of \omterm{def:g12:randvar:exp}{variance} $\sigma^2 = \frac14$. The bound
\[
\P\left(\abs{F_n - 0.5} \geq 0.05\right)
\leq \frac{1/4}{n \times 0.05^2} = \frac{100}{n}
\]
is $\leq 0.05$ as soon as $n \geq 2000$.
\end{solution}

\begin{solution}{exo:g12:sums:4}
By \cref{prop:g12:sums:mean}, $\E(M_n) = 5\%$ (diversification does not
change the expected return) and
$\sigma(M_n) = \frac{20\%}{\sqrt n}$. Requiring
$\frac{20}{\sqrt n} < 4$ gives $\sqrt n > 5$, \emph{i.e.} $n \geq 26$.
This is the principle of \emph{diversification}: spreading capital over
\omterm{def:g12:sums:indep}{independent} risks \omterm{def:g12:arith:divides}{divides} the risk (\omterm{def:g11:stat:variance}{standard deviation}) by $\sqrt n$
without reducing the expected return.
\end{solution}

\begin{solution}{exo:g12:sums:5}
\emph{1.} Counting the $9$ equally likely pairs: $S$ takes the values
$2, 3, 4, 5, 6$ with probabilities
$\frac19, \frac29, \frac39, \frac29, \frac19$. Hence
$\E(S) = \frac{2 + 6 + 12 + 10 + 6}{9} = 4$ and
$\E(S^2) = \frac{4 + 18 + 48 + 50 + 36}{9} = \frac{156}{9} = \frac{52}{3}$,
so $\V(S) = \frac{52}{3} - 16 = \frac43$.

\emph{2.} One variable: $\E(X) = 2$,
$\E(X^2) = \frac{1 + 4 + 9}{3} = \frac{14}{3}$,
$\V(X) = \frac{14}{3} - 4 = \frac23$. Then $\E(S) = 2 + 2 = 4$ and, by
\omterm{def:g12:condprob:indep}{independence}, $\V(S) = \frac23 + \frac23 = \frac43$. Same values.
\end{solution}

\begin{solution}{exo:g12:sums:6}
With $Y = X$: $\V(X + Y) = \V(2X) = 4\V(X)$, while
$\V(X) + \V(Y) = 2\V(X)$. The two agree only when $\V(X) = 0$, \emph{i.e.}
when $X$ is constant --- so additivity genuinely requires \omterm{def:g12:condprob:indep}{independence}
(here $X$ is maximally dependent on itself).
\end{solution}

\begin{solution}{exo:g12:sums:7}
Under fairness, $F_n$ is the \omterm{def:g11:stat:mean}{mean} of $n = 1200$ Bernoulli variables with
$p = \frac16$, $\sigma^2 = \frac16\cdot\frac56 = \frac{5}{36}$:
\[
\P\left(\abs{F_n - \tfrac16} \geq 0.05\right)
\leq \frac{5/36}{1200 \times 0.0025} = \frac{5}{108} \approx 0.046 .
\]
The observed deviation is exactly $0.05$: an \omterm{def:g11:prob:model}{event} that a fair die produces
with \omterm{def:g10:proba:distribution}{probability} at most $4.6\%$ --- and since Chebyshev is very
conservative, the true \omterm{def:g10:proba:distribution}{probability} is far smaller. The fairness hypothesis
is not plausible; the die is very likely loaded.
\end{solution}

\begin{solution}{exo:g12:sums:8}
\emph{1.} $p(1-p) = \frac14 - \left(p - \frac12\right)^2 \leq \frac14$.

\emph{2.} $F_n$ has \omterm{def:g12:randvar:exp}{expectation} $p$ and \omterm{def:g12:randvar:exp}{variance}
$\frac{p(1-p)}{n} \leq \frac{1}{4n}$; Bienaymé--Chebyshev gives
\[
\P\bigl(\abs{F_n - p} \geq \varepsilon\bigr)
\leq \frac{p(1-p)}{n\varepsilon^2} \leq \frac{1}{4n\varepsilon^2},
\]
valid whatever the unknown $p$.

\emph{3.} With $\varepsilon = 0.03$ and level $0.05$:
\[
\frac{1}{4n(0.03)^2} \leq 0.05
\iff n \geq \frac{1}{4 \times 0.0009 \times 0.05} \approx 5556 .
\]
About $5600$ people suffice by this crude bound (the classical
\omterm{def:g10:numbers:approx}{normal-approximation} answer is nearer $1100$, see
\cref{ch:g12:contdist}).
\end{solution}

\begin{solution}{pb:g12:sums:1}
\textbf{1.} $V(X + Y) = V(X) + V(Y) = \frac{35}{12} +
\frac{35}{12} = \frac{35}{6}$ (\omterm{def:g12:condprob:indep}{independence}).

\textbf{2.} $\E(\bar X) = 3.5$;
$\sigma(\bar X) = \frac{1.71}{\sqrt{100}} \approx 0.17$: the
hundred-dice average hugs $3.5$ within a fifth of a point.

\textbf{3.} $\P(X \geq 10) \leq \frac{2}{10} = 0.2$.

\textbf{4.} $\P \leq \frac{V(\bar X)}{0.5^2} =
\frac{35/1200}{0.25} \approx 0.117$: at most about
$12\,\%$.

\textbf{5.} With $Y = -X$: $V(X + Y) = V(0) = 0$, while
$V(X) + V(Y) = 2V(X) > 0$: perfectly anti-correlated variables
cancel --- \omterm{def:g12:randvar:exp}{variance} addition is an \omterm{def:g12:condprob:indep}{independence} privilege.

\textbf{6.} $\E = \frac{18}{37} - \frac{19}{37} =
-\frac{1}{37} \approx -0.027$;
$\sigma = \sqrt{1 - \left(\frac{1}{37}\right)^2} \approx
1.00$.

\textbf{7.} $\E(G) = -2.70$, $\sigma(G) = 10 \times 1.00 =
10$: the drift is only $0.27\sigma$. The night's noise dwarfs
the edge --- a large minority of gamblers (roughly $40\,\%$,
says the bell) walk out ahead, which is precisely what keeps
them coming back.

\textbf{8.} Casino's take: \omterm{def:g12:randvar:exp}{expectation} $+27\,027$ euros,
\omterm{def:g11:stat:variance}{standard deviation} $\approx 1\,000$: the profit sits $27$
\omterm{def:g11:stat:variance}{standard deviations} above zero. At $27\sigma$, ``the casino
might lose this year'' is not a risk, it is a rounding error.

\textbf{9.} $\P(\text{loss}) = \P(G \geq 27\,027$ for the
gamblers$)$ $\leq \frac{10^6 \times 1}{27\,027^2} \approx
0.0014$: even the bluntest inequality in the book guarantees
the house at $99.86\,\%$ --- the truth is astronomically
stronger.

\textbf{10.} Each euro staked, whenever and however chosen,
has expected return $-\frac{1}{37}$ of itself; by linearity
the expected total gain is $-\frac{1}{37} \times
(\text{total staked})$, negative for every strategy that
stakes anything. A winning system would violate the linearity
of \omterm{def:g12:randvar:exp}{expectation} --- no clever sequencing of bad bets makes a
good one. (Doubling systems merely trade many small wins for
rare catastrophic losses.)

\textbf{11.} The theorem: the \emph{\omterm{def:g10:stats:series}{frequency}} of red over
$n$ spins \omterm{def:g12:seq:limit}{converges} (in \omterm{def:g10:proba:distribution}{probability}) to $\frac{18}{37}$. It
says nothing about spin $n + 1$: the wheel has no memory, and
after ten reds the chance of red is still $\frac{18}{37}$.
Believing otherwise is the \emph{gambler's fallacy}, and
question 12 shows what actually happens to streaks.

\textbf{12.} The surplus of $100$ reds is not repaid ---
future spins are fair copies, expected surplus stays $100$.
But $\frac{100}{10^6} = 0.0001$: one hundredth of a point.
The fallacy's error in one sentence: \emph{the law of large
numbers dilutes past accidents in an ocean of new trials; it
never sends the wheel to collect debts.}

\textbf{13.} $n \geq \frac{1}{4 \times 0.05 \times 0.03^2}
\approx 5\,556$ people.

\textbf{14.} $\frac{1.96^2 \times 0.25}{0.03^2} \approx
1\,067$: five times fewer. Chebyshev holds for \emph{every}
\omterm{def:g12:randvar:rv}{distribution} and pays for its universality with slack; the
pollsters' constant $1.96$ comes from the actual bell shape
of the fluctuations, which concentrates far harder.

\textbf{15.} Margin $\propto \frac{1}{\sqrt n}$: halving it
quadruples the \omterm{def:g10:proba:sample}{sample} --- about $4\,400$ people for
$\pm 1.5$ points. Precision is bought at quadratic prices.

\textbf{16.} The model is $n$ \omterm{def:g12:sums:indep}{independent} draws with success
\omterm{def:g10:proba:distribution}{probability} $p$ --- the population size $N$ never appears
(sampling a tiny fraction of a large, well-mixed population
is what \omterm{def:g12:condprob:indep}{independence} encodes). The pollsters' analogy: to
taste the soup, one well-stirred spoonful suffices ---
whether the pot serves ten or ten thousand. The catch is the
stirring --- a biased sampling frame, as in the Middle School
volume's poll that fooled a country --- not the pot.

\textbf{17.} $315\sqrt n < 30n \iff \sqrt n > 10.5 \iff
n > 110$: beyond a hundred policies the drift outruns the
noise, and each further policy widens the gap by the
$\sqrt n$ law --- pooling \emph{is} the business model.

\textbf{18.} \omterm{def:g12:sums:indep}{Independent} equal shares: $\sigma_{\text{pf}} =
\frac{20\,\%}{\sqrt{25}} = 4\,\%$: same expected return,
one-fifth the fluctuation. The lunch: risk reduction at zero
cost in \omterm{def:g12:randvar:exp}{expectation} --- averaging \omterm{def:g12:sums:indep}{independent} randomness.

\textbf{19.} Perfectly correlated: the portfolio is one stock
in $25$ costumes: $\sigma = 20\,\%$, no reduction at all.
Diversification rents \emph{\omterm{def:g12:condprob:indep}{independence}}; when a crisis
correlates everything (2008: all housing bets were one bet),
the $\sqrt n$ protection evaporates exactly when it is
needed.

\textbf{20.} Sums drift like $n$ and fluctuate like
$\sqrt n$, so means settle like $\frac{1}{\sqrt n}$: the
casino banks the drift over a million spins; the pollster
buys $\frac{1}{\sqrt{1100}}$ of noise for a phone budget;
the insurer outgrows its own claims at $n > 110$; the
investor \omterm{def:g12:arith:divides}{divides} risk by $\sqrt{25}$. Abusers: the gambler
who believes in debts (there is only dilution), and the
financier who believes in \omterm{def:g12:condprob:indep}{independence} (there is sometimes
only one bet). The fluctuations' universal shape --- the
bell --- is the \omterm{def:g12:limcont:continuity}{continuous} chapter's star, and its theorem,
the central limit theorem, is the summit of the university
volumes' \omterm{def:g10:proba:distribution}{probability}.
\end{solution}
