\section*{Chapitre \ref{ch:g12:sums} --- Sommes de variables aléatoires et loi des grands nombres}

\begin{solution}{exo:g12:sums:1}
Chaque dé a pour espérance
$\frac{1 + 2 + \dots + 6}{6} = \frac{7}{2}$, donc par linéarité
$\E(S) = \frac72 + \frac72 = 7$. Les dés sont indépendants, donc
$\V(S) = \frac{35}{12} + \frac{35}{12} = \frac{35}{6} \approx 5.83$.
\end{solution}

\begin{solution}{exo:g12:sums:2}
$\E(X) = 50$ et $\V(X) = 100 \times 0.25 = 25$.

Markov ($X \geq 0$)~: $\P(X \geq 75) \leq \frac{50}{75} = \frac23$.

Bienaymé--Tchebychev~:
$\P(\abs{X - 50} \geq 25) \leq \frac{25}{25^2} = 0.04$ --- et cela borne
même l'événement \emph{bilatéral}, dont $X \geq 75$ est la moitié.
Tchebychev est bien plus fin ici car il exploite la variance, pas seulement
la moyenne. (La vraie valeur de $\P(X \geq 75)$ est inférieure à
$10^{-6}$~: les deux bornes sont grossières.)
\end{solution}

\begin{solution}{exo:g12:sums:3}
$F_n$ est la moyenne empirique de $n$ variables de Bernoulli
$\left(\frac12\right)$, de variance $\sigma^2 = \frac14$. La borne
\[
\P\left(\abs{F_n - 0.5} \geq 0.05\right)
\leq \frac{1/4}{n \times 0.05^2} = \frac{100}{n}
\]
est $\leq 0.05$ dès que $n \geq 2000$.
\end{solution}

\begin{solution}{exo:g12:sums:4}
Par la \cref{prop:g12:sums:mean}, $\E(M_n) = 5\%$ (la diversification ne
change pas le rendement espéré) et
$\sigma(M_n) = \frac{20\%}{\sqrt n}$. Exiger
$\frac{20}{\sqrt n} < 4$ donne $\sqrt n > 5$, \emph{c.-à-d.} $n \geq 26$.
C'est le principe de \emph{diversification}~: répartir le capital sur des
risques indépendants divise le risque (écart type) par $\sqrt n$ sans
réduire le rendement espéré.
\end{solution}

\begin{solution}{exo:g12:sums:5}
\emph{1.} En comptant les $9$ couples équiprobables~: $S$ prend les
valeurs $2, 3, 4, 5, 6$ avec probabilités
$\frac19, \frac29, \frac39, \frac29, \frac19$. D'où
$\E(S) = \frac{2 + 6 + 12 + 10 + 6}{9} = 4$ et
$\E(S^2) = \frac{4 + 18 + 48 + 50 + 36}{9} = \frac{156}{9} = \frac{52}{3}$,
donc $\V(S) = \frac{52}{3} - 16 = \frac43$.

\emph{2.} Une variable~: $\E(X) = 2$,
$\E(X^2) = \frac{1 + 4 + 9}{3} = \frac{14}{3}$,
$\V(X) = \frac{14}{3} - 4 = \frac23$. Puis $\E(S) = 2 + 2 = 4$ et, par
indépendance, $\V(S) = \frac23 + \frac23 = \frac43$. Mêmes valeurs.
\end{solution}

\begin{solution}{exo:g12:sums:6}
Avec $Y = X$~: $\V(X + Y) = \V(2X) = 4\V(X)$, tandis que
$\V(X) + \V(Y) = 2\V(X)$. Les deux coïncident seulement lorsque
$\V(X) = 0$, \emph{c.-à-d.} lorsque $X$ est constante --- donc
l'additivité exige vraiment l'indépendance (ici $X$ est maximalement
dépendante d'elle-même).
\end{solution}

\begin{solution}{exo:g12:sums:7}
Sous l'hypothèse d'équilibre, $F_n$ est la moyenne de $n = 1200$ variables
de Bernoulli avec $p = \frac16$,
$\sigma^2 = \frac16\cdot\frac56 = \frac{5}{36}$~:
\[
\P\left(\abs{F_n - \tfrac16} \geq 0.05\right)
\leq \frac{5/36}{1200 \times 0.0025} = \frac{5}{108} \approx 0.046 .
\]
L'écart observé est exactement $0.05$~: un événement qu'un dé équilibré
produit avec probabilité au plus $4.6\%$ --- et comme Tchebychev est très
conservateur, la vraie probabilité est bien plus petite. L'hypothèse
d'équilibre n'est pas plausible~; le dé est très probablement pipé.
\end{solution}

\begin{solution}{exo:g12:sums:8}
\emph{1.} $p(1-p) = \frac14 - \left(p - \frac12\right)^2 \leq \frac14$.

\emph{2.} $F_n$ a pour espérance $p$ et pour variance
$\frac{p(1-p)}{n} \leq \frac{1}{4n}$~; Bienaymé--Tchebychev donne
\[
\P\bigl(\abs{F_n - p} \geq \varepsilon\bigr)
\leq \frac{p(1-p)}{n\varepsilon^2} \leq \frac{1}{4n\varepsilon^2},
\]
valable quel que soit le $p$ inconnu.

\emph{3.} Avec $\varepsilon = 0.03$ et niveau $0.05$~:
\[
\frac{1}{4n(0.03)^2} \leq 0.05
\iff n \geq \frac{1}{4 \times 0.0009 \times 0.05} \approx 5556 .
\]
Environ $5600$ personnes suffisent par cette borne grossière (la réponse
classique par approximation normale est plus proche de $1100$, voir le
\cref{ch:g12:contdist}).
\end{solution}
