\section*{Chapitre \ref{ch:g12:limcont} --- Limites et continuité}

\begin{solution}{exo:g12:limcont:1}
En divisant par $x^3$~:
$\dfrac{2x^3 - x + 1}{5x^3 + x^2} = \dfrac{2 - 1/x^2 + 1/x^3}{5 + 1/x}
\to \dfrac{2}{5}$.

$x^2 + 3x - 1 = x^2(1 + 3/x - 1/x^2) \to +\infty$ lorsque
$x \to -\infty$, car $x^2 \to +\infty$ et le crochet tend vers $1$.

Lorsque $x \to 2^+$, le numérateur tend vers $3 > 0$ et le dénominateur
vers $0^+$, donc $\dfrac{x+1}{x-2} \to +\infty$.

Pour $x > 0$, $-\dfrac1x \leq \dfrac{\cos x}{x} \leq \dfrac1x$, donc la
limite est $0$ par le théorème des gendarmes.
\end{solution}

\begin{solution}{exo:g12:limcont:2}
\emph{1.} $2x^2 - 3x + 1 = (x-1)(2x-1)$ (la racine $x=1$ est apparente
d'après $2 - 3 + 1 = 0$). Pour $x \neq 1$, simplifier $x - 1$~:
$f(x) = 2x - 1$.

\emph{2.} D'où $\lim_{x\to1} f(x) = 1$. Poser $f(1) = 1$ prolonge $f$ en
la fonction polynomiale $x \mapsto 2x-1$ sur $\R$, continue en $1$.
\end{solution}

\begin{solution}{exo:g12:limcont:3}
Écrire $f(x) = \dfrac{3x - 1}{x + 2} = \dfrac{3(x+2) - 7}{x+2}
= 3 - \dfrac{7}{x+2}$. Lorsque $x \to \pm\infty$, $f(x) \to 3$~: la
droite $y = 3$ est une asymptote horizontale aux deux infinis. Lorsque
$x \to (-2)^\pm$, $f(x) \to \mp\infty$~: la droite $x = -2$ est une
asymptote verticale. La courbe est une hyperbole, sous $y=3$ pour
$x > -2$ et au-dessus pour $x < -2$.
\end{solution}

\begin{solution}{exo:g12:limcont:4}
Multiplier par le conjugué~:
\[
\sqrt{x^2 + x} - x = \frac{x^2 + x - x^2}{\sqrt{x^2+x} + x}
= \frac{x}{x\left(\sqrt{1 + 1/x} + 1\right)}
= \frac{1}{\sqrt{1 + 1/x} + 1}
\xrightarrow[x\to+\infty]{} \frac{1}{2}.
\]
De même,
\[
\frac{\sqrt{1+x} - 1}{x} = \frac{(1+x) - 1}{x\left(\sqrt{1+x} + 1\right)}
= \frac{1}{\sqrt{1+x} + 1} \xrightarrow[x\to0]{} \frac{1}{2}.
\]
\end{solution}

\begin{solution}{exo:g12:limcont:5}
\emph{1.} Comme $\sin x \geq -1$, $f(x) \geq x - 1 \to +\infty$, et on
conclut par comparaison. Pourtant $f'(x) = 1 + \cos x$ s'annule en tout
multiple impair de $\pi$, et sur tout intervalle $\intco{A}{+\infty}$ la
fonction alterne entre des intervalles de croissance stricte~; elle est
croissante mais non strictement, et surtout sa croissance n'est pas
nécessaire~: la comparaison suffit.

\emph{2.} Pour $x > 1$, diviser par $x$~:
\[
g(x) = \frac{1 + \frac{\sin x}{x}}{1 - \frac{\sin x}{x}},
\]
et $\frac{\sin x}{x} \to 0$ par le théorème des gendarmes
($\abs{\sin x} \leq 1$). D'où $g(x) \to \frac{1+0}{1-0} = 1$.
\end{solution}

\begin{solution}{exo:g12:limcont:6}
$f(x) = x^3 - 3x + 1$ est continue, avec $f'(x) = 3x^2 - 3 = 3(x-1)(x+1)$~:
$f$ croît sur $\intoc{-\infty}{-1}$, décroît sur $\intcc{-1}{1}$, croît
sur $\intco{1}{+\infty}$, avec maximum local $f(-1) = 3 > 0$ et minimum
local $f(1) = -1 < 0$, $\lim_{-\infty} f = -\infty$,
$\lim_{+\infty} f = +\infty$.

Sur chacun des trois intervalles de monotonie, $f$ est continue,
strictement monotone, et $0$ est strictement entre les valeurs/limites
aux extrémités, donc le théorème de la bijection donne exactement un zéro
par intervalle~: trois zéros au total. Contrôles de signe~:
$f(-2) = -1 < 0 < f(-1) = 3$, donc un zéro dans $\intoo{-2}{-1}$~;
$f(0) = 1 > 0 > f(1) = -1$, un zéro dans $\intoo{0}{1}$~;
$f(1) = -1 < 0 < f(2) = 3$, un zéro dans $\intoo{1}{2}$.
\end{solution}

\begin{solution}{exo:g12:limcont:7}
Soit $g(x) = f(x) - x$, continue sur $\intcc{0}{1}$. Comme
$f(0) \in \intcc{0}{1}$, $g(0) = f(0) \geq 0$~; comme
$f(1) \in \intcc{0}{1}$, $g(1) = f(1) - 1 \leq 0$. Par le théorème des
valeurs intermédiaires, $g$ s'annule en un certain $c \in \intcc{0}{1}$,
c'est-à-dire $f(c) = c$.
\end{solution}

\begin{solution}{exo:g12:limcont:8}
Soient $u(t)$ et $d(t)$ les positions (mesurées comme distance depuis le
bas du sentier) à l'instant $t \in \intcc{8}{20}$ à la montée et à la
descente. Toutes deux sont continues, $u(8) = 0$, $u(20) = L$ (la
longueur du sentier), $d(8) = L$, $d(20) = 0$. La fonction $g = u - d$
est continue avec $g(8) = -L \leq 0$ et $g(20) = L \geq 0$, donc
$g(c) = 0$ pour un certain instant $c$ par le théorème des valeurs
intermédiaires~: à l'instant $c$, les positions des deux jours
coïncident.
\end{solution}

\begin{solution}{exo:g12:limcont:9}
\emph{1.} $f'(x) = 5x^4 + 3x^2 + 1 > 0$, donc $f$ est strictement
croissante sur $\R$~; avec $\lim_{\pm\infty} f = \pm\infty$ et la
continuité, le théorème de la bijection donne un unique zéro réel
$\alpha$. Comme $f(0) = -1 < 0$ et $f(1) = 2 > 0$,
$\alpha \in \intoo{0}{1}$.

\emph{2.} Maintenir un intervalle $\intcc{a_n}{b_n}$ avec
$f(a_n) < 0 < f(b_n)$, en partant de $\intcc{0}{1}$~; à chaque étape
évaluer $f$ au milieu $m$ et garder le demi-intervalle où le changement
de signe persiste. Après $n$ étapes l'intervalle a pour longueur
$2^{-n}$, donc le milieu approxime $\alpha$ à $2^{-(n+1)}$ près. Il faut
$2^{-(n+1)} \leq 10^{-6}$, c'est-à-dire
$n + 1 \geq \frac{6}{\log_{10} 2} \approx 19.93$~: $n = 20$ itérations
suffisent.
\end{solution}
