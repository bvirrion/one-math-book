\section*{Chapitre \ref{ch:g12:condprob} --- Probabilité conditionnelle et indépendance}

\begin{solution}{exo:g12:condprob:1}
Il y a $12$ figures, dont $4$ rois~:
$\pcond{\text{figure}}{\text{roi}} = \frac{4}{12} = \frac13$.

\omterm{def:g12:condprob:indep}{Indépendance}~: $\P(\text{roi} \cap \text{cœur}) = \frac{1}{52}$ (le roi
de cœur), et
$\P(\text{roi})\,\P(\text{cœur}) = \frac{4}{52}\times\frac{13}{52}
= \frac{1}{52}$. Égaux~: les \omterm{def:g11:prob:model}{événements} sont \omterm{def:g12:condprob:indep}{indépendants}.
\end{solution}

\begin{solution}{exo:g12:condprob:2}
\emph{1.} Première branche~: rouge $\frac58$, bleue $\frac38$~; secondes
branches (sans remise)~: après rouge, rouge $\frac47$ / bleue $\frac37$~;
après bleue, rouge $\frac57$ / bleue $\frac27$.

\emph{2.} $\P(RR) = \frac58 \times \frac47 = \frac{5}{14}$. Par les
\omterm{def:g10:proba:distribution}{probabilités} totales,
\[
\P(\text{2e rouge}) = \frac58\cdot\frac47 + \frac38\cdot\frac57
= \frac{20 + 15}{56} = \frac58 .
\]
(La même que le premier tirage --- par symétrie, la seconde boule est une
boule uniformément aléatoire de l'urne.)
\end{solution}

\begin{solution}{exo:g12:condprob:3}
$\P(A) = \frac{6}{36} = \frac16$ (six couples de somme $7$),
$\P(B) = \frac16$, et $A \cap B = \{(3,4)\}$ a pour \omterm{def:g10:proba:distribution}{probabilité}
$\frac{1}{36} = \P(A)\P(B)$~: $A$ et $B$ sont \omterm{def:g12:condprob:indep}{indépendants}. (La somme $7$
est spéciale~: quel que soit le premier dé, exactement une valeur du
second la donne.)

$\P(C) = \frac{5}{36}$ et $B \cap C = \{(3,3)\}$~:
$\P(B \cap C) = \frac{1}{36} \neq \frac16 \times \frac{5}{36}
= \frac{5}{216}$. Non \omterm{def:g12:condprob:indep}{indépendants}.
\end{solution}

\begin{solution}{exo:g12:condprob:4}
\emph{1.} \omterm{def:g10:proba:distribution}{Probabilités} totales avec la partition par machine~:
\[
\P(D) = 0.5\times0.01 + 0.3\times0.02 + 0.2\times0.04
= 0.005 + 0.006 + 0.008 = 0.019 = 1.9\% .
\]

\emph{2.} Bayes~:
$\pcond{D}{M_3} = \dfrac{0.008}{0.019} = \dfrac{8}{19} \approx 0.42$. La
machine produisant seulement un cinquième de la production représente plus
de $40\%$ des défauts.
\end{solution}

\begin{solution}{exo:g12:condprob:5}
La condition est
\[
\frac{0.99\,p}{0.99\,p + 0.05\,(1-p)} \geq 0.9 .
\]
Le dénominateur est positif, donc cela s'écrit
$0.99p \geq 0.891p + 0.045 - 0.045p$, \emph{c.-à-d.} $0.144\,p \geq 0.045$,
\emph{c.-à-d.}
\[
p \geq \frac{0.045}{0.144} = 0.3125 .
\]
Le test seul atteint $90\%$ de certitude seulement si la maladie touche
déjà plus de $31\%$ de la population testée --- d'où la nécessité de tests
de confirmation pour le dépistage de masse des maladies rares.
\end{solution}

\begin{solution}{exo:g12:condprob:6}
\emph{1.} Trois \omterm{def:g12:seq:sequence}{suites} réalisent $E$ (PPF, PFP, FPP), chacune de
\omterm{def:g10:proba:distribution}{probabilité} $p^2(1-p)$ par \omterm{def:g12:condprob:indep}{indépendance}~: $\P(E) = 3p^2(1 - p)$.

\emph{2.} $f(p) = 3p^2 - 3p^3$ a $f'(p) = 6p - 9p^2 = 3p(2 - 3p)$,
positive puis négative sur $\intoo{0}{1}$~: \omterm{def:g10:functions:extrema}{maximum} en $p = \frac23$, où
$\P(E) = 3\cdot\frac49\cdot\frac13 = \frac49$.
\end{solution}

\begin{solution}{exo:g12:condprob:7}
Disons que vous avez choisi la porte~1 et que l'animateur a ouvert la
porte~3 (\omterm{def:g11:prob:model}{événement} $H_3$). Avec $B_i$ = «~prix derrière la porte $i$~»,
$\P(B_i) = \frac13$ et
\[
\pcond{B_1}{H_3} = \frac12, \qquad
\pcond{B_2}{H_3} = 1, \qquad
\pcond{B_3}{H_3} = 0
\]
(si le prix est derrière votre porte, l'animateur choisit au hasard entre
les portes~2 et~3~; s'il est derrière la porte~2, il est forcé d'ouvrir la
$3$). Bayes~:
\[
\pcond{H_3}{B_1}
= \frac{\frac13\cdot\frac12}{\frac13\cdot\frac12 + \frac13\cdot1 + 0}
= \frac{1/6}{1/2} = \frac13,
\qquad
\pcond{H_3}{B_2} = \frac{\frac13}{\frac12} = \frac23 .
\]
Rester gagne avec \omterm{def:g10:proba:distribution}{probabilité} $\frac13$, basculer avec \omterm{def:g10:proba:distribution}{probabilité}
$\frac23$.
\end{solution}

\begin{solution}{exo:g12:condprob:8}
\emph{1.} La majorité est fausse lorsque au moins deux des trois copies
sont inversées~:
\[
3\varepsilon^2(1-\varepsilon) + \varepsilon^3
= 3(0.01)(0.9) + 0.001 = 0.028,
\]
bien plus petit que $\varepsilon = 0.1$~: le codage par répétition
fonctionne.

\emph{2.} Si $1$ a été envoyé (comme $111$), recevoir $101$ exige
exactement une inversion~: \omterm{def:g10:proba:distribution}{probabilité}
$\varepsilon(1-\varepsilon)^2 = 0.081$. Si $0$ a été envoyé ($000$),
deux inversions sont nécessaires~: $\varepsilon^2(1-\varepsilon) = 0.009$.
Bayes avec prioris égaux~:
\[
\pcond{101}{\,\text{envoyé }1} = \frac{0.081}{0.081 + 0.009} = 0.9 .
\]
Le décodage majoritaire ($101 \mapsto 1$) est bien le guess le plus
probable.
\end{solution}
