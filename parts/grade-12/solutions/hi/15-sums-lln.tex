\section*{अध्याय \ref{ch:g12:sums} --- यादृच्छिक चरों का योग और बड़ी संख्याओं का नियम}

\begin{solution}{exo:g12:sums:1}
प्रत्येक पासे में होता है \omterm{def:g12:randvar:exp}{अपेक्षा}
$\frac{1 + 2 + \dots + 6}{6} = \frac{7}{2}$, तो रैखिकता से
$\E(S) = \frac72 + \frac72 = 7$. पासे हैं \omterm{def:g12:sums:indep}{स्वतंत्र}, इसलिए
$\V(S) = \frac{35}{12} + \frac{35}{12} = \frac{35}{6} \approx 5.83$.
\end{solution}

\begin{solution}{exo:g12:sums:2}
$\E(X) = 50$ और $\V(X) = 100 \times 0.25 = 25$.

मार्कोव ($ X \geq 0$): $\P(X \geq 75) \leq \frac{50}{75} = \frac23$.

बिएनेमे--चेबीशेव:
$\P(\abs{X - 50} \geq 25) \leq \frac{25}{25^2} = 0.04$ --- और यह भी
की सीमा तय करता है \emph{दोस्र्खा} \omterm{def:g11:prob:model}{आयोजन}, जिसका कि $ X \geq 75$ आधा है. चेबीशेव
यहाँ कहीं अधिक तीव्र है क्योंकि यह इसका शोषण करता है \omterm{def:g12:randvar:exp}{झगड़ा}, सिर्फ नहीं \omterm{def:g11:stat:mean}{अर्थ}.
(का सही मूल्य $\P(X \geq 75)$ की तुलना में छोटा है $10^{-6}$: दोनों सीमाएँ
कच्चे हैं.)
\end{solution}

\begin{solution}{exo:g12:sums:3}
$ F_n $ है \omterm{prop:g12:sums:mean}{नमूना माध्य} का $ n $ Bernoulli$\left(\frac12\right)$ चर,
का \omterm{def:g12:randvar:exp}{झगड़ा} $\sigma^2 = \frac14$. बाध्य
\[
\P\left(\abs{F_n - 0.5} \geq 0.05\right)
\leq \frac{1/4}{n \times 0.05^2} = \frac{100}{n}
\]
है $\leq 0.05$ जैसे ही $ n \geq 2000$.
\end{solution}

\begin{solution}{exo:g12:sums:4}
द्वारा \cref{prop:g12:sums:mean}, $\E(M_n) = 5\%$ (विविधीकरण नहीं होता
अपेक्षित रिटर्न बदलें) और
$\sigma(M_n) = \frac{20\%}{\sqrt n}$. की आवश्यकता होती है
$\frac{20}{\sqrt n} < 4$ देता है $\sqrt n > 5$, \emph{यानी} $ n \geq 26$.
का \omterm{def:g11:vect:det}{सिद्ध}ांत यही है \emph{विविधता}: पूंजी फैलाना
\omterm{def:g12:sums:indep}{स्वतंत्र} जोखिम विभाजित जोखिम (\omterm{def:g11:stat:variance}{मानक विचलन}) द्वारा $\sqrt n $
अपेक्षित रिटर्न को कम किए बिना।
\end{solution}

\begin{solution}{exo:g12:sums:5}
\emph{1.} की गिनती $9$ समान रूप से संभावित \omterm{def:g10:vectors:sum}{जोड़}े: $ S $ मान लेता है
$2, 3, 4, 5, 6$ संभावनाओं के साथ
$\frac19, \frac29, \frac39, \frac29, \frac19$. इस तरह
$\E(S) = \frac{2 + 6 + 12 + 10 + 6}{9} = 4$ और
$\E(S^2) = \frac{4 + 18 + 48 + 50 + 36}{9} = \frac{156}{9} = \frac{52}{3}$,
इसलिए $\V(S) = \frac{52}{3} - 16 = \frac43$.

\emph{2.} एक चर: $\E(X) = 2$,
$\E(X^2) = \frac{1 + 4 + 9}{3} = \frac{14}{3}$,
$\V(X) = \frac{14}{3} - 4 = \frac23$. तब $\E(S) = 2 + 2 = 4$ और तक
स्वतंत्रता, $\V(S) = \frac23 + \frac23 = \frac43$. समान मूल्य.
\end{solution}

\begin{solution}{exo:g12:sums:6}
साथ $ Y = X $: $\V(X + Y) = \V(2X) = 4\V(X)$, जबकि
$\V(X) + \V(Y) = 2\V(X)$. दोनों तभी सहमत होते हैं जब $\V(X) = 0$, \emph{यानी}
कब $ X $ स्थिर है --- इसलिए वास्तव में संवेदनशीलता की आवश्यकता है स्वतंत्रता
(यहाँ $ X $ \omterm{def:g10:functions:extrema}{अधिकतम} रूप से स्वयं पर निर्भर है)।
\end{solution}

\begin{solution}{exo:g12:sums:7}
निष्पक्षता के तहत, $ F_n $ है \omterm{def:g11:stat:mean}{अर्थ} का $ n = 1200$ बर्नौली चर के साथ
$ p = \frac16$, $\sigma^2 = \frac16\cdot\frac56 = \frac{5}{36}$:
\[
\P\left(\abs{F_n - \tfrac16} \geq 0.05\right)
\leq \frac{5/36}{1200 \times 0.0025} = \frac{5}{108} \approx 0.046 .
\]
देखा गया विचलन बिल्कुल सही है $0.05$: एक \omterm{def:g11:prob:model}{आयोजन} जो एक निष्पक्ष पासा पैदा करता है
के साथ \omterm{def:g10:proba:distribution}{संभावना} अधिक से अधिक $4.6\%$ --- और चूँकि चेबीशेव बहुत है
रूढ़िवादी, सच्चा \omterm{def:g10:proba:distribution}{संभावना} बहुत छोटा है. निष्पक्षता परिकल्पना
प्रशंसनीय नहीं है; बहुत संभव है कि पासा भरा हुआ हो।
\end{solution}

\begin{solution}{exo:g12:sums:8}
\emph{1.} $ p(1-p) = \frac14 - \left(p - \frac12\right)^2 \leq \frac14$.

\emph{2.} $ F_n $ है \omterm{def:g12:randvar:exp}{अपेक्षा} $ p $ और \omterm{def:g12:randvar:exp}{झगड़ा}
$\frac{p(1-p)}{n} \leq \frac{1}{4n}$; बिएनेमे--चेबीशेव देता है
\[
\P\bigl(\abs{F_n - p} \geq \varepsilon\bigr)
\leq \frac{p(1-p)}{n\varepsilon^2} \leq \frac{1}{4n\varepsilon^2},
\]
अज्ञात जो भी हो वैध $ p $.

\emph{3.} साथ $\varepsilon = 0.03$ और स्तर $0.05$:
\[
\frac{1}{4n(0.03)^2} \leq 0.05
\iff n \geq \frac{1}{4 \times 0.0009 \times 0.05} \approx 5556 .
\]
के बारे में $5600$ लोग इस अपरिष्कृत बंधन (शास्त्रीय) से संतुष्ट हैं
सामान्य-अनुमान उत्तर निकट है $1100$, देखना
\cref{ch:g12:contdist}).
\end{solution}

\begin{solution}{pb:g12:sums:1}
\textbf{1.} $ V(X + Y) = V(X) + V(Y) = \frac{35}{12} +
\frac{35}{12} = \frac{35}{6}$ (स्वतंत्रता).

\textbf{2.} $\E(\bar X) = 3.5$;
$\sigma(\bar X) = \frac{1.71}{\sqrt{100}} \approx 0.17$: द
सौ-पासा औसत आलिंगन $3.5$ एक बिंदु के पांचवें हिस्से के भीतर।

\textbf{3.} $\P(X \geq 10) \leq \frac{2}{10} = 0.2$.

\textbf{4.} $\P \leq \frac{V(\bar X)}{0.5^2} =
\frac{35/1200}{0.25} \approx 0.117$: अधिक से अधिक के बारे में
$12\,\%$.

\textbf{5.} साथ $ Y = -X $: $ V(X + Y) = V(0) = 0$, जबकि
$ V(X) + V(Y) = 2V(X) > 0$: पूरी तरह से विरोधी सहसंबद्ध चर
रद्द करें --- \omterm{def:g12:randvar:exp}{झगड़ा} \omterm{def:g10:vectors:sum}{जोड़} एक है स्वतंत्रता विशेषाधिकार।

\textbf{6.} $\E = \frac{18}{37} - \frac{19}{37} =
-\frac{1}{37} \approx -0.027$;
$\sigma = \sqrt{1 - \left(\frac{1}{37}\right)^2} \approx
1.00$.

\textbf{7.} $\E(G) = -2.70$, $\sigma(G) = 10 \times 1.00 =
10$: बहाव ही है $0.27\sigma $. रात का शोर बौना हो जाता है
किनारा --- जुआरियों का एक बड़ा अल्पसंख्यक वर्ग (मोटे तौर पर)। $40\,\%$,
घंटी कहती है) आगे बढ़ो, जो वास्तव में वही रखता है
वे वापस आ रहे हैं.

\textbf{8.} कैसीनो की राय: \omterm{def:g12:randvar:exp}{अपेक्षा} $+27\,027$ यूरो,
\omterm{def:g11:stat:variance}{मानक विचलन} $\approx 1\,000$: लाभ बैठता है $27$
\omterm{def:g11:stat:variance}{मानक विचलन} शून्य से ऊपर. पर $27\sigma $, ``कैसीनो
इस वर्ष हार सकता है'' यह कोई जोखिम नहीं है, यह एक पूर्णांक त्रुटि है।

\textbf{9.} $\P(\text{loss}) = \P(G \geq 27\,027$ के लिए
जुआरी $)$ $\leq \frac{10^6 \times 1}{27\,027^2} \approx
0.0014$: यहां तक कि पुस्तक में सबसे स्पष्ट असमानता भी गारंटी देती है
घर पर $99.86\,\%$ --- सत्य खगोलीय है
मजबूत.

\textbf{10.} प्रत्येक यूरो दांव पर लगाया गया, जब भी और जैसे भी चुना गया,
वापसी की उम्मीद है $-\frac{1}{37}$ स्वयं का; रैखिकता द्वारा
अपेक्षित कुल लाभ है $-\frac{1}{37} \times
(\text{total staked})$, हर रणनीति के लिए नकारात्मक
कुछ भी दांव पर लगा देता है. एक विजेता प्रणाली रैखिकता का उल्लंघन करेगी
का \omterm{def:g12:randvar:exp}{अपेक्षा} --- खराब दांवों का कोई चतुर \omterm{def:g12:seq:sequence}{अनुक्रम} नहीं बनता
अच्छा है. (दोहरीकरण प्रणाली केवल कई छोटी जीतों का व्यापार करती है
दुर्लभ विनाशकारी नुकसान।)

\textbf{11.} प्रमेय: \emph{\omterm{def:g10:stats:series}{आवृत्ति}} लाल रंग के ऊपर
$ n $ स्पिन \omterm{def:g12:seq:limit}{अभिसरण} (में \omterm{def:g10:proba:distribution}{संभावना}) को $\frac{18}{37}$. यह
स्पिन के बारे में कुछ नहीं कहता $ n + 1$: पहिए में कोई मेमोरी नहीं है, और
दस रेड के बाद भी रेड की \omterm{def:g10:proba:distribution}{संभावना} बनी रहती है $\frac{18}{37}$.
अन्यथा विश्वास करना ही है \emph{जुआरी का भ्रम}, और
प्रश्न 12 दिखाता है कि वास्तव में धारियों का क्या होता है।

\textbf{12.} का अधिशेष $100$ रेड्स का भुगतान नहीं किया गया ---
भविष्य के स्पिन निष्पक्ष प्रतियां हैं, अपेक्षित अधिशेष रहता है $100$.
लेकिन $\frac{100}{10^6} = 0.0001$: एक बिंदु का सौवाँ भाग।
एक वाक्य में भ्रांति की त्रुटि: \emph{बड़े का कानून
संख्याएँ नए परीक्षणों के सागर में पिछली दुर्घटनाओं को कम कर देती हैं; यह
कभी भी कर्ज वसूलने के लिए पहिया नहीं भेजता।}

\textbf{13.} $ n \geq \frac{1}{4 \times 0.05 \times 0.03^2}
\approx 5\,556$ लोग।

\textbf{14.} $\frac{1.96^2 \times 0.25}{0.03^2} \approx
1\,067$: पांच गुना कम. चेबीशेव का मानना है \emph{प्रत्येक}
\omterm{def:g12:randvar:rv}{वितरण} और इसकी सार्वभौमिकता की कीमत सुस्ती से चुकानी पड़ती है; द
प्रदूषकों का स्थिरांक $1.96$ वास्तविक घंटी के आकार से आता है
उतार-चढ़ाव का, जो बहुत अधिक ध्यान \omterm{pb:g11:quad:1}{केंद्र}ित करता है।

\textbf{15.} अंतर $\propto \frac{1}{\sqrt n}$: इसे आधा करना
को चौगुना कर देता है \omterm{def:g10:proba:sample}{नमूना} --- के बारे में $4\,400$ लोगों के लिए
$\pm 1.5$ अंक. परिशुद्धता को द्विघात कीमतों पर खरीदा जाता है।

\textbf{16.} मॉडल है $ n $ \omterm{def:g12:sums:indep}{स्वतंत्र} सफलता के साथ खींचता है
\omterm{def:g10:proba:distribution}{संभावना} $ p $ ---जनसंख्या का आकार $ N $ कभी प्रकट नहीं होता
(एक बड़ी, अच्छी तरह से मिश्रित आबादी के एक छोटे से हिस्से का \omterm{def:g10:proba:sample}{नमूना} लेना
क्या है स्वतंत्रता एन्कोड)। प्रदूषकों की सादृश्यता: से
सूप का स्वाद चखें, एक अच्छी तरह से हिलाया हुआ चम्मच पर्याप्त है ---
चाहे बर्तन दस परोसे या दस हज़ार। पकड़ है
सरगर्मी --- एक पक्षपातपूर्ण \omterm{def:g10:proba:sample}{नमूना} फ्रेम, जैसा कि मिडिल स्कूल में होता है
वॉल्यूम का सर्वेक्षण जिसने एक देश को मूर्ख बनाया --- पॉट नहीं।

\textbf{17.} $315\sqrt n < 30n \iff \sqrt n > 10.5 \iff
n > 110$: सौ नीतियों से परे बहाव आगे निकल जाता है
शोर, और प्रत्येक आगे की नीति इस अंतर को बढ़ाती है
$\sqrt n $ कानून --- पूलिंग \emph{है} बिजनेस मॉडल.

\textbf{18.} \omterm{def:g12:sums:indep}{स्वतंत्र} समान शेयर: $\sigma_{\text{pf}} =
\frac{20\,\%}{\sqrt{25}} = 4\,\%$: वही अपेक्षित रिटर्न,
एक-पाँचवाँ उतार-चढ़ाव। दोपहर का भोजन: शून्य पर जोखिम में कमी
में लागत \omterm{def:g12:randvar:exp}{अपेक्षा} --- औसत \omterm{def:g12:sums:indep}{स्वतंत्र} यादृच्छिकता.

\textbf{19.} बिल्कुल सहसंबद्ध: पोर्टफोलियो एक स्टॉक है
में $25$ वेशभूषा: $\sigma = 20\,\%$, बिल्कुल कोई कमी नहीं।
विविधीकरण किराया \emph{स्वतंत्रता}; जब कोई संकट हो
सब कुछ सहसंबंधित करता है (2008: सभी हाउसिंग दांव एक ही दांव थे),
द $\sqrt n $ सुरक्षा ठीक उसी समय वाष्पित हो जाती है जब वह होती है
जरुरत.

\textbf{20.} रकम जैसे बह जाती है $ n $ और जैसे उतार-चढ़ाव
$\sqrt n $, तो इसका मतलब है जैसे व्यवस्थित करें $\frac{1}{\sqrt n}$: द
कैसीनो बैंक एक लाख स्पिन से अधिक का बहाव; सर्वेक्षणकर्ता
खरीदता है $\frac{1}{\sqrt{1100}}$ फ़ोन बजट के लिए शोर का;
बीमाकर्ता अपने स्वयं के दावों से आगे निकल जाता है $ n > 110$; द
निवेशक विभाजित द्वारा जोखिम $\sqrt{25}$. दुर्व्यवहार करने वाले: जुआरी
जो ऋण में विश्वास करता है (केवल कमजोर पड़ने वाला है), और
फाइनेंसर जो विश्वास करता है स्वतंत्रता (कभी-कभी होता है
केवल एक शर्त)। उतार-चढ़ाव का सार्वभौमिक आकार---
घंटी --- है निरंतर अध्याय का तारा, और उसका प्रमेय,
\omterm{pb:g11:quad:1}{केंद्र}ीय सीमा प्रमेय, विश्वविद्यालय का शिखर है
खंड' \omterm{def:g10:proba:distribution}{संभावना}.
\end{solution}
