\begin{solution}{exo:g12:deriv:1}
$f'(x) = 3x^2 - 5$.

भागफल नियम:
$ g'(x) = \dfrac{(x^2+1) - x\cdot 2x}{(x^2+1)^2} = \dfrac{1 - x^2}{(x^2+1)^2}$.

चेन नियम के साथ $ u = 2x+1$: $ h'(x) = 5 \cdot 2 \cdot (2x+1)^4 = 10(2x+1)^4$.

चेन नियम के साथ $ u = x^2 + 1$:
$ k'(x) = \dfrac{2x}{2\sqrt{x^2+1}} = \dfrac{x}{\sqrt{x^2+1}}$.
\end{solution}

\begin{solution}{exo:g12:deriv:2}
$ f'(x) = 2x - 3$, इसलिए $ f(2) = -1$ और $ f'(2) = 1$: द \omterm{def:g12:deriv:tangent}{स्पर्शरेखा} है
$ y = -1 + (x - 2) = x - 3$. अंतर है
\[
f(x) - (x - 3) = x^2 - 4x + 4 = (x-2)^2 \geq 0 ,
\]
अतः वक्र ऊपर स्थित है \omterm{def:g12:deriv:tangent}{स्पर्शरेखा} हर जगह, केवल इसे छूना $ x = 2$
(आशा के अनुसार: $ f $ है \omterm{def:g12:deriv:convex}{उत्तल}).
\end{solution}

\begin{solution}{exo:g12:deriv:3}
दोनों भिन्न भागफल हैं। साथ $ f(x) = (1+x)^{100}$ पर $0$:
$ f'(x) = 100(1+x)^{99}$, तो सीमा है $ f'(0) = 100$.
साथ $ g(x) = \sqrt{x}$ पर $4$: $ g'(x) = \frac{1}{2\sqrt{x}}$, तो सीमा है
$ g'(4) = \frac14$.
\end{solution}

\begin{solution}{exo:g12:deriv:4}
\omterm{def:g11:func:function}{कार्यक्षेत्र} $\R \setminus \{1\}$. सीमाएँ: पर $\pm\infty $, $ f(x) \sim x \to
\pm\infty $; पर $1^\pm $, अंश $\to 4 > 0$ और हर $\to 0^\pm $,
इसलिए $ f \to \pm\infty $; रेखा $ x = 1$ एक है \omterm{def:g12:limcont:asymptote}{ऊर्ध्वाधर अनंतस्पर्शी}.

\[
f'(x) = \frac{2x(x-1) - (x^2+3)}{(x-1)^2} = \frac{x^2 - 2x - 3}{(x-1)^2}
= \frac{(x-3)(x+1)}{(x-1)^2}.
\]
इस तरह $ f' > 0$ पर $\intoo{-\infty}{-1}$ और $\intoo{3}{+\infty}$, $ f' < 0$ पर
$\intoo{-1}{1}$ और $\intoo{1}{3}$. \omterm{def:g11:func:function}{समारोह} एक तक बढ़ जाता है \omterm{def:g12:deriv:extremum}{स्थानीय अधिकतम}
$ f(-1) = -2$, कम हो जाता है $-\infty $, कूदता है $+\infty $ के बाद अनंतस्पर्शी,
स्थानीय स्तर पर घट जाती है \omterm{def:g10:functions:extrema}{न्यूनतम} $ f(3) = 6$, फिर बढ़ जाता है।
\end{solution}

\begin{solution}{exo:g12:deriv:5}
के लिए $ x \in \intoo{0}{15}$, बॉक्स का आधार भुजा वर्गाकार है $30 - 2x $ और
ऊंचाई $ x $, इसलिए
\[
V(x) = x(30 - 2x)^2 .
\]
$ V'(x) = (30-2x)^2 + x \cdot 2(30-2x)(-2) = (30-2x)(30 - 2x - 4x)
= (30-2x)(30-6x)$. पर $\intoo{0}{15}$, $30 - 2x > 0$, इसलिए $ V'$ का चिन्ह है
$30 - 6x $: पहले सकारात्मक $ x = 5$, बाद में नकारात्मक। के लिए मात्रा \omterm{def:g10:functions:extrema}{अधिकतम} है
$ x = 5$ सेमी, दे रहा हूँ $ V(5) = 5 \times 20^2 = 2000\ \text{cm}^3$.
\end{solution}

\begin{solution}{exo:g12:deriv:6}
\emph{1.} $ f'(x) = 4x^3 - 12x^2$ और $ f''(x) = 12x^2 - 24x = 12x(x-2)$.
इस प्रकार $ f'' > 0$ पर $\intoo{-\infty}{0}$ और पर $\intoo{2}{+\infty}$ (\omterm{def:g12:deriv:convex}{उत्तल}),
$ f'' < 0$ पर $\intoo{0}{2}$ (\omterm{def:g12:deriv:convex}{नतोदर}). $ f''$ पर परिवर्तन चिह्न $0$ और कम से $2$:
दोनों हैं \omterm{def:g12:deriv:inflection}{विभक्ति बिंदु}.

\emph{2.} पर $ a = 2$: $ f(2) = 16 - 32 + 10 = -6$, $ f'(2) = 32 - 48 = -16$,
\omterm{def:g12:deriv:tangent}{स्पर्शरेखा} $ y = -6 - 16(x-2) = -16x + 26$. अंतर है
\[
g(x) = f(x) + 16x - 26 = x^4 - 4x^3 + 16x - 16 .
\]
तब से $ g(2) = g'(2) = g''(2) = 0$, $(x-2)^3$ विभाजित $ g $; विभाजन देता है
$ g(x) = (x-2)^3(x+2)$. पास में $ x = 2$, $ x + 2 > 0$, इसलिए $ g $ का चिन्ह है
$(x-2)^3$: पहले नकारात्मक $2$, बाद में सकारात्मक। वक्र को पार करता है
\omterm{def:g12:deriv:tangent}{स्पर्शरेखा}: $2$ वास्तव में एक है \omterm{def:g12:deriv:inflection}{विभक्ति बिंदु}.
\end{solution}

\begin{solution}{exo:g12:deriv:7}
 \omterm{def:g11:func:function}{समारोह} $ f(x) = \sqrt{x}$ संतुष्ट
$ f''(x) = -\frac{1}{4}x^{-3/2} < 0$ पर $\intoo{0}{+\infty}$: यह है \omterm{def:g12:deriv:convex}{नतोदर},
तो यह है \omterm{def:g10:functions:graph}{ग्राफ} झूठ \emph{नीचे} प्रत्येक \omterm{def:g12:deriv:tangent}{स्पर्शरेखा}. \omterm{def:g12:deriv:tangent}{स्पर्शरेखा} पर $ a = 1$ है
\[
y = f(1) + f'(1)(x - 1) = 1 + \tfrac12 (x-1) = \frac{x+1}{2},
\]
जहां से $\sqrt{x} \leq \frac{x+1}{2}$ सभी के लिए $ x > 0$. समानता का मतलब है
वक्र स्पर्श करता है \omterm{def:g12:deriv:tangent}{स्पर्शरेखा}, जो केवल \omterm{def:g12:deriv:tangent}{स्पर्शरेखा} बिंदु पर होता है
$ x = 1$. (समान रूप से: $\left(\sqrt x - 1\right)^2 \geq 0$.)
\end{solution}

\begin{solution}{exo:g12:deriv:8}
द्वारा \cref{thm:g12:deriv:convexity}, द \omterm{def:g10:functions:graph}{ग्राफ} एक का \omterm{def:g12:deriv:derivative}{विभेदक} \omterm{def:g12:deriv:convex}{उत्तल}
\omterm{def:g11:func:function}{समारोह} इसके प्रत्येक के ऊपर स्थित है \omterm{def:g12:deriv:tangent}{स्पर्शरेखा}. \omterm{def:g12:deriv:tangent}{स्पर्शरेखा} पर $ a $ क्षैतिज है
($ f'(a) = 0$), का \omterm{def:g10:algebra:equation}{समीकरण} $ y = f(a)$. इस तरह $ f(x) \geq f(a)$ सभी के लिए
$ x \in \R $: $ f(a)$ वैश्विक है \omterm{def:g10:functions:extrema}{न्यूनतम}.
\end{solution}

\begin{solution}{exo:g12:deriv:9}
\emph{निश्चित परिधि.} भुजाओं वाला एक आयत $ x $ और $\frac{p}{2} - x $
($0 < x < \frac p2$) का क्षेत्रफल है $ S(x) = x\left(\frac p2 - x\right)$. तब
$ S'(x) = \frac p2 - 2x $ पर गायब हो जाता है $ x = \frac p4$, पहले सकारात्मक और
नकारात्मक के बाद: \omterm{def:g10:functions:extrema}{अधिकतम} पर है $ x = \frac{p}{4}$, जहां दोनों पक्ष बराबर हैं
$\frac p4$ --- एक वर्ग.

\emph{निश्चित क्षेत्र.} पक्षों $ x $ और $\frac Ax $ परिधि दीजिए
$ P(x) = 2\left(x + \frac Ax\right)$, साथ
$ P'(x) = 2\left(1 - \frac{A}{x^2}\right)$, के लिए नकारात्मक $ x < \sqrt A $ और
सकारात्मक बाद: \omterm{def:g10:functions:extrema}{न्यूनतम} पर $ x = \sqrt{A}$, फिर से एक वर्ग।

\emph{रिश्ता।} दोनों कथन दोहरे हैं। मान लीजिए कि एक गैर-वर्गाकार आयत है
निश्चित क्षेत्र पर परिधि को \omterm{def:g10:functions:extrema}{न्यूनतम} कर दिया $ A $; समान परिधि का वर्ग
क्षेत्रफल होगा $> A $ पहले कथन द्वारा, और इसे समरूप रूप से छोटा करना
क्षेत्र के लिए $ A $ सख्ती से इसकी परिधि को कम कर देगा --- \omterm{thm:g12:limcont:ivt}{विरोधाभास}ी
न्यूनता. इस प्रकार प्रत्येक कथन दूसरे को दर्शाता है।
\end{solution}

\begin{solution}{pb:g12:deriv:1}
\textbf{1.} $30x\left(3x^2 + 1\right)^4$; \quad
$\dfrac{x}{\sqrt{x^2 + 9}}$; \quad
$-\dfrac{2x}{\left(x^2 + 1\right)^2}$.

\textbf{2.} $ y(4) = 5$, \omterm{def:g10:reffunc:affine}{ढलान} $\frac45$: \omterm{def:g12:deriv:tangent}{स्पर्शरेखा}
$ y = 5 + \frac45(x - 4)$, यानी\ $ y = \frac45 x + \frac95$.

\textbf{3.} $ f'(x) = \frac{(x^2 + 1) - x \cdot 2x}{(x^2+1)^2}
= \frac{1 - x^2}{(x^2 + 1)^2}$: नकारात्मक, सकारात्मक, नकारात्मक
पार $-1$ और $1$: \omterm{def:g10:functions:extrema}{न्यूनतम} $ f(-1) = -\frac12$, \omterm{def:g10:functions:extrema}{अधिकतम}
$ f(1) = \frac12$, साथ \omterm{def:g12:limcont:asymptote}{समस्तरीय अनंतस्पर्शी रेखा} $0$ दोनों तरफ.

\textbf{4.} $ g'(x) = 3x^2 - 6x = 3x(x - 2)$: \omterm{def:g12:seq:monotonic}{की बढ़ती},
\omterm{def:g10:functions:variations}{घटते}, \omterm{def:g12:seq:monotonic}{की बढ़ती} आर-पार $0$ और $2$; \omterm{def:g12:deriv:extremum}{स्थानीय अधिकतम}
$ g(0) = 4$, स्थानीय मि $ g(2) = 0$. $ g''(x) = 6x - 6$: \omterm{def:g12:deriv:convex}{नतोदर}
पहले $1$, \omterm{def:g12:deriv:convex}{उत्तल} पश्चात: पर विभक्ति $(1, 2)$.

\textbf{5.} \omterm{def:g12:deriv:tangent}{स्पर्शरेखा} पर $1$: \omterm{def:g10:reffunc:affine}{ढलान} $ g'(1) = -3$:
$ y = 2 - 3(x - 1) = -3x + 5$. अंतर:
$ x^3 - 3x^2 + 4 - (-3x + 5) = x^3 - 3x^2 + 3x - 1 =
(x - 1)^3$, जो साइन बदलता है $1$: द \omterm{def:g12:deriv:tangent}{स्पर्शरेखा}
\emph{क्रॉस} वक्र---ए पर हस्ताक्षर व्यवहार
\omterm{def:g12:deriv:inflection}{विभक्ति बिंदु}, जहां वक्र पक्ष बदलता है।

\textbf{6.} $ T(x) = \dfrac{\sqrt{x^2 + 900}}{5} +
\dfrac{\sqrt{(40 - x)^2 + 400}}{2}$. पहले प्रवेश करना $0$ या
परे $40$ लंबा \emph{दोनों} पैर: व्यर्थ.

\textbf{7.} $ T'(x) = \dfrac{x}{5\sqrt{x^2 + 900}} -
\dfrac{40 - x}{2\sqrt{(40 - x)^2 + 400}}$.

\textbf{8.} चल रहे त्रिकोण में, किनारे के साथ पक्ष
है $ x $ और कर्ण $\sqrt{x^2 + 900}$: द \omterm{def:g11:trigo:cossin}{ज्या} का
लम्ब के साथ कोण वास्तव में उनका भागफल है
(कर्ण के विपरीत); वैसे ही
$\sin\theta_2 = \frac{40 - x}{\sqrt{(40-x)^2 + 400}}$. तब
$ T'(x) = 0$ पढ़ता
$\frac{\sin\theta_1}{5} = \frac{\sin\theta_2}{2}$ --- स्नेल का
गति के साथ कानून $5$ और $2$.

\textbf{9.} $ T(24) \approx 20.5$ एस; $ T(40) = 10 + 10 = 20$ एस;
$ T(0) = 6 + \sqrt{2000}/2 \approx 28.4$ एस। सीधी रेखा
``पूरे रास्ते दौड़ने, फिर सीधे तैरने'' तक भी हार जाता है --- और
दोनों अपवर्तित पथ से भटक जाते हैं।

\textbf{10.} \omterm{thm:g12:limcont:ivt}{विरोधाभास} पर $ T'$ (नकारात्मक पर $24$, सकारात्मक पर
$40$) \omterm{def:g12:seq:limit}{अभिसरण} को $ x^* \approx 34$ मी, साथ
$ T(x^*) \approx 19.5$ s: से लगभग एक सेकंड तेज
सीधी रेखा--- बचाव और बचाव के बीच का अंतर
त्रासदी.

\textbf{11.} प्रत्येक लेग-टाइम एक है \omterm{def:g12:deriv:convex}{उत्तल} \omterm{def:g11:func:function}{समारोह} का $ x $
(ए $\sqrt{\text{quadratic}}$ शाखा), तो $ T $ है \omterm{def:g12:deriv:convex}{उत्तल}, और ए
ए का महत्वपूर्ण बिंदु \omterm{def:g12:deriv:convex}{उत्तल} \omterm{def:g12:deriv:derivative}{विभेदक} \omterm{def:g11:func:function}{समारोह} इसका है
वैश्विक \omterm{def:g10:functions:extrema}{न्यूनतम} (\cref{exo:g12:deriv:8}): $ x^*$ मात्र नहीं है
स्थिर, यह अपराजेय है.

\textbf{12.} पानी में प्रकाश धीमा होता है; फ़र्मेट के \omterm{def:g11:vect:det}{सिद्ध}ांत द्वारा
सबसे तेज़ हवा से पानी का रास्ता सतह पर बिल्कुल मुड़ता है
स्नेल का नियम --- अत: जलमग्न पेंसिल से किरणें आँख तक पहुँचती हैं
मुड़ा हुआ, और मस्तिष्क, सीधी रेखाओं को फैलाकर, देखता है
वॉटरलाइन पर पेंसिल टूट गई.

\textbf{13.} $(x^2)'' = 2 > 0$: \omterm{def:g12:deriv:convex}{उत्तल}. उत्तलता पर
\omterm{prop:g10:coordgeom:midpoint}{मध्य}: $ f\left(\frac{a+b}{2}\right) \leq
\frac{f(a) + f(b)}{2}$, जो दावा है. हाथ से:
$\frac{a^2 + b^2}{2} - \left(\frac{a+b}{2}\right)^2 =
\frac{(a - b)^2}{4} \geq 0$.

\textbf{14.} वर्गमूल लेना (\omterm{def:g12:seq:monotonic}{की बढ़ती}) प्रश्न 13 में:
$\frac{a + b}{2} \leq \sqrt{\frac{a^2 + b^2}{2}}$. पूरी शृंखला
पर $2, 8$: हार्मोनिक $\frac{2 \cdot 16}{10} = 3.2$; \omterm{def:g12:seq:arith-geom}{ज्यामितिक}
$\sqrt{16} = 4$; \omterm{def:g12:seq:arith-geom}{अंकगणित} $5$; द्विघात
$\sqrt{34} \approx 5.83$: प्रत्येक \omterm{def:g11:stat:mean}{अर्थ} अगले को प्रणाम करता है.

\textbf{15.} \omterm{def:g12:deriv:tangent}{स्पर्शरेखा} को $\frac1x $ पर $1$: कीमत $1$, \omterm{def:g10:reffunc:affine}{ढलान}
$-1$: $ y = 2 - x $. \omterm{def:g12:deriv:convex}{उत्तल} वक्र उनके ऊपर बैठते हैं \omterm{def:g12:deriv:tangent}{स्पर्शरेखा}
(\cref{thm:g12:deriv:convexity}): $\frac1x \geq 2 - x $ सभी के लिए
$ x > 0$, बिल्कुल संपर्क बिंदु पर समानता के साथ $ x = 1$.

\textbf{16.} विभक्ति पर, \emph{दैनिक} मामले की गिनती
(द यौगिक) शिखर: विकास की गति रुक जाती है और शुरू हो जाती है
धीमा करने के लिए --- पहला गणितीय संकेत है कि लहर
गिनती में गिरावट आने से बहुत पहले, घूम रहा है। इससे पहले,
$ f'' > 0$ (प्रत्येक दिन पिछले से भी बदतर); इसके बाद,
$ f'' < 0$ (अभी भी बढ़ रहा है, लेकिन अधिक धीरे-धीरे)।

\textbf{17.} $ S'(r) = 4\pi r - \frac{2V}{r^2} = 0$ देता है
$ r^3 = \frac{V}{2\pi} = \frac{330}{2\pi}$:
$ r \approx 3.74$ सेमी, और

\textbf{18.} $ h = \frac{V}{\pi r^2} = \frac{2\pi r^3}{\pi r^2}

आपने जो लिखा, वह नहीं जो आपका मतलब था।

\textbf{20.} लाइफगार्ड: मॉडल $ T(x)$, अंतर करना, स्नेल,

\end{solution}
