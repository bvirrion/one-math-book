\section*{अध्याय \ref{ch:g12:trigo} --- त्रिकोणमितीय कार्य}

\begin{solution}{exo:g12:trigo:1}
$\cos\frac{2\pi}{3} = \cos\left(\pi - \frac\pi3\right) = -\cos\frac\pi3
= -\frac12$.

$\sin\frac{5\pi}{6} = \sin\left(\pi - \frac\pi6\right) = \sin\frac\pi6
= \frac12$.

$\cos\frac{7\pi}{4} = \cos\left(2\pi - \frac{\pi}{4}\right)
= \cos\left(-\frac\pi4\right) = \frac{\sqrt2}{2}$.

$\sin\left(-\frac\pi3\right) = -\sin\frac\pi3 = -\frac{\sqrt3}{2}$.
\end{solution}

\begin{solution}{exo:g12:trigo:2}
$ f'(x) = 2\sin x\cos x = \sin 2x $ (श्रृंखला नियम).

$ g'(x) = -3\sin(3x) + \sin x + x\cos x $ (श्रृंखला और उत्पाद नियम)।

भागफल नियम:
\[
h'(x) = \frac{\cos x\,(2 + \cos x) - \sin x\,(-\sin x)}{(2+\cos x)^2}
= \frac{2\cos x + \cos^2 x + \sin^2 x}{(2+\cos x)^2}
= \frac{2\cos x + 1}{(2+\cos x)^2}.
\]
\end{solution}

\begin{solution}{exo:g12:trigo:3}
\emph{(ए)} का एक विशेष समाधान $\sin x = \frac{\sqrt3}{2}$ है
$\frac\pi3$. सामान्य समाधान: $ x = \frac\pi3 + 2k\pi $ या
$ x = \pi - \frac\pi3 + 2k\pi = \frac{2\pi}{3} + 2k\pi $. में
$\intoc{-\pi}{\pi}$: $ x \in \left\{\frac\pi3, \frac{2\pi}{3}\right\}$.

\emph{(बी)} $\cos 2x = \frac{\sqrt2}{2} = \cos\frac\pi4$ देता है
$2x = \pm\frac\pi4 + 2k\pi $, इसलिए $ x = \pm\frac\pi8 + k\pi $. में
$\intoc{-\pi}{\pi}$:
$ x \in \left\{-\frac{7\pi}{8}, -\frac\pi8, \frac\pi8, \frac{7\pi}{8}\right\}$.
\end{solution}

\begin{solution}{exo:g12:trigo:4}
साथ $ a = \frac\pi3$, $ b = \frac\pi4$:
\[
\cos\frac{\pi}{12} = \cos a\cos b + \sin a \sin b
= \frac12\cdot\frac{\sqrt2}{2} + \frac{\sqrt3}{2}\cdot\frac{\sqrt2}{2}
= \frac{\sqrt2 + \sqrt6}{4},
\]
\[
\sin\frac{\pi}{12} = \sin a\cos b - \cos a \sin b
= \frac{\sqrt3}{2}\cdot\frac{\sqrt2}{2} - \frac12\cdot\frac{\sqrt2}{2}
= \frac{\sqrt6 - \sqrt2}{4}.
\]
\end{solution}

\begin{solution}{exo:g12:trigo:5}
तय करना $ s = \sin x $: $2s^2 - s - 1 = (2s + 1)(s - 1) \geq 0$ आईएफएफ
$ s \leq -\frac12$ या $ s = 1$.

$\sin x = 1$ देता है $ x = \frac\pi2$. $\sin x \leq -\frac12$: इकाई पर
वृत्त, क्षैतिज रेखा के नीचे का चाप $ y = -\frac12$, जो में
$\intcc{0}{2\pi}$ है $ x \in \intcc{\frac{7\pi}{6}}{\frac{11\pi}{6}}$.
इस तरह
\[
S = \left\{\frac{\pi}{2}\right\} \cup
\intcc{\frac{7\pi}{6}}{\frac{11\pi}{6}} .
\]
\end{solution}

\begin{solution}{exo:g12:trigo:6}
$\dfrac{\sin 3x}{x} = 3\,\dfrac{\sin 3x}{3x} \to 3 \cdot 1 = 3$
(साथ रचना $ u = 3x \to 0$).

का उपयोग करते हुए $1 - \cos x = \dfrac{\sin^2 x}{1 + \cos x}$:
\[
\frac{1 - \cos x}{x^2}
= \left(\frac{\sin x}{x}\right)^{\!2} \frac{1}{1 + \cos x}
\xrightarrow[x\to0]{} 1 \cdot \frac12 = \frac12 .
\]

स्थानापन्न $ u = \frac1x \to 0^+$:
$ x \sin\frac1x = \frac{\sin u}{u} \to 1$.
\end{solution}

\begin{solution}{exo:g12:trigo:7}
$ f'(x) = 1 - \sin x \geq 0$ सभी के लिए $ x $, बिल्कुल समानता के साथ जब
$\sin x = 1$, \emph{यानी} $ x = \frac\pi2 + 2k\pi $. पर $\intcc{0}{2\pi}$, $ f $
है \omterm{def:g12:seq:monotonic}{की बढ़ती} से $ f(0) = 1$ को $ f(2\pi) = 2\pi + 1$ (का शून्य $ f'$ पर
$\frac{\pi}{2}$ पृथक है, इसलिए वृद्धि और भी सख्त है); \omterm{def:g10:functions:extrema}{न्यूनतम} $1$ पर
$0$, \omterm{def:g10:functions:extrema}{अधिकतम} $2\pi + 1$ पर $2\pi $. पर $\R $, $ f' \geq 0$ हर जगह सिर्फ साथ
पृथक शून्य, इसलिए $ f $ है (कड़ाई से) \omterm{def:g12:seq:monotonic}{की बढ़ती} द्वारा
\cref{thm:g12:deriv:variations}, हालांकि $ f'$ हर समय गायब हो जाता है
$\frac\pi2 + 2k\pi $.
\end{solution}

\begin{solution}{exo:g12:trigo:8}
\emph{1.} होने देना $ R = \sqrt{A^2 + B^2} > 0$. बात
$\left(\frac AR, \frac BR\right)$ इकाई वृत्त पर स्थित है, इसलिए वहाँ मौजूद है
$\varphi $ साथ $\cos\varphi = \frac AR $ और $\sin\varphi = \frac BR $. तब
\[
R\cos(x - \varphi) = R\bigl(\cos x\cos\varphi + \sin x \sin\varphi\bigr)
= A\cos x + B\sin x = f(x).
\]

\emph{2.} तब से $\cos $ रेंज है $\intcc{-1}{1}$, $ f $ है \omterm{def:g10:functions:extrema}{अधिकतम} $ R $ और
\omterm{def:g10:functions:extrema}{न्यूनतम} $-R $. के लिए $\cos x + \sin x $: $ A = B = 1$, $ R = \sqrt2$,
$\varphi = \frac\pi4$, इतना \omterm{def:g10:algebra:equation}{समीकरण} बन जाता है
$\sqrt2\cos\left(x - \frac\pi4\right) = 1$, \emph{यानी}
$\cos\left(x - \frac\pi4\right) = \frac{\sqrt2}{2}$, देना
$ x - \frac\pi4 = \pm\frac\pi4 + 2k\pi $: में $\intoc{-\pi}{\pi}$,
$ x \in \left\{0, \frac\pi2\right\}$.
\end{solution}

\begin{solution}{exo:g12:trigo:9}
$(\sin)'' = (\cos)' = -\sin $ और $(\cos)'' = (-\sin)' = -\cos $: दोनों हल करते हैं
$ y'' = -y $.

होने देना $ g = y - a\cos - b\sin $. तब $ g'' = -g $ (विभेदीकरण की रैखिकता),
$ g(0) = y(0) - a = 0$ और $ g'(0) = y'(0) - b = 0$. तय करना
$ E = g^2 + (g')^2$. तब
\[
E' = 2g g' + 2g' g'' = 2g g' - 2g' g = 0,
\]
इसलिए $ E $ के बराबर स्थिरांक है $ E(0) = 0$. दो वर्गों का योग लुप्त हो रहा है
समान रूप से बल $ g = 0$, इस तरह $ y = a\cos + b\sin $.
\end{solution}

\begin{solution}{pb:g12:trigo:1}
\textbf{1.} $0.99833\ldots $ और $0.99998\ldots $: रेंगना
$1$.

\textbf{2.} त्रिकोण $ OAM $: आधार $1$, ऊंचाई $\sin x $: क्षेत्र
$\frac{\sin x}{2}$. सेक्टर $ OAM $: अंश
$\frac{x}{2\pi}$ यूनिट डिस्क का: क्षेत्र $\frac x2$. त्रिकोण
$ OAT $: आधार $1$, ऊंचाई $\tan x $: क्षेत्र $\frac{\tan x}{2}$.

\textbf{3.} त्रिभुज बड़े के अंदर सेक्टर के अंदर बैठता है
त्रिकोण: $\frac{\sin x}{2} < \frac x2 < \frac{\tan x}{2}$,
यानी\ $\sin x < x < \tan x $. से $\sin x < x $:
$\frac{\sin x}{x} < 1$; और से
$ x < \tan x = \frac{\sin x}{\cos x}$, से गुणा करना
$\frac{\cos x}{x} > 0$: $\cos x < \frac{\sin x}{x}$. जैसा
$ x \to 0^+$, $\cos x \to 1$: निचोड़ा हुआ,
$\frac{\sin x}{x} \to 1$; और $\frac{\sin x}{x}$ सम है, इसलिए
दो तरफा सीमा है $1$.

\textbf{4.} $\frac{1 - \cos x}{x^2} =
\frac{1 - \cos^2 x}{x^2 (1 + \cos x)} =
\left(\frac{\sin x}{x}\right)^2 \frac{1}{1 + \cos x}
\to 1 \times \frac12$.

\textbf{5.} $\sin'(0) = \lim_{h \to 0} \frac{\sin h - 0}{h} =
1$: का संपूर्ण विभेदन \omterm{def:g12:trigo:cossin}{ज्या} और \omterm{def:g11:trigo:cossin}{कोज्या} से बहती है
यह एक सीमा है. डिग्री में मापी गई सीमा होगी
$\frac{\pi}{180}$, और हर यौगिक उसे खींच लेंगे
स्थिरांक: रेडियंस बिल्कुल वही इकाई है जो कैलकुलस बनाती है
दोलन का साफ.

\textbf{6.} $ y' = -R\omega\sin(\omega t - \varphi)$,
$ y'' = -R\omega^2\cos(\omega t - \varphi) = -\omega^2 y $.

\textbf{7.} $\omega = 2$. $ y(0) = A = 3$;
$ y'(0) = 2B = 8$: $ B = 4$: $ y = 3\cos 2t + 4\sin 2t $.

\textbf{8.} $ R = \sqrt{9 + 16} = 5$; अवधि
$\frac{2\pi}{2} = \pi $; चरण
$\varphi = \arctan\frac43 \approx 0.927$:
$ y = 5\cos(2t - 0.927)$.

\textbf{9.} $5\cos(2t - \varphi) = 0$ पहले कब
$2t - \varphi = \frac\pi2$:
$ t = \frac{\pi/2 + 0.927}{2} \approx 1.25$ एस।

\textbf{10.} साथ $\sin\theta \approx \theta $:
$\theta'' = -\frac gL \theta $: हार्मोनिक के साथ
$\omega = \sqrt{\frac gL}$, अवधि
$ T = \frac{2\pi}{\omega} = 2\pi\sqrt{\frac Lg}$. के लिए
$ L = 1$: $ T \approx 2.006$ एस। अवधि निर्भर नहीं करती
आयाम (छोटे झूलों के लिए) --- गैलीलियो की समकालिकता,
तथ्य यह है कि पेंडुलम घड़ियों को संभव बनाता है।

\textbf{11.} $ L = g\left(\frac{T}{2\pi}\right)^2 =
\frac{9.81}{\pi^2} \approx 0.994$ मी: सेकंड पेंडुलम है
के बारे में $6$ मेरिडियन मीटर से मिमी कम। अकादमी ने चुना
मेरिडियन (गुरुत्वाकर्षण स्थान-स्थान पर भिन्न-भिन्न होता है, विश्वासघात करता है
पेंडुलम); था $ g $ यदि मूंछें थोड़ी बड़ी होतीं, तो हमारा मीटर बड़ा होता
टिक.

\textbf{12.} $\cos\left(x + \frac\pi2\right) =
\cos x \cos\frac\pi2 - \sin x \sin\frac\pi2 = -\sin x $;
$\sin\left(x + \frac\pi2\right) = \cos x $. इतना अंतर करना
\omterm{def:g12:trigo:cossin}{ज्या} देता है \omterm{def:g12:trigo:cossin}{ज्या} एक चौथाई मोड़ स्थानांतरित कर दिया ($\cos $), और फिर
($-\sin $), और फिर, और चार चरणों में घर: भेदभाव
वृत्त घुमाता है.

\textbf{13.} के दो \omterm{def:g10:algebra:expand}{विस्तार}ों को जोड़ना $\cos(a \mp b)$:
$\cos(a-b) + \cos(a+b) = 2\cos a\cos b $. साथ $ a = b = x $:
$1 + \cos 2x = 2\cos^2 x $, \omterm{def:g11:stat:mean}{अर्थ}ात\
$\cos^2 x = \frac{1 + \cos 2x}{2}$.

\textbf{14.} साथ $ p = \frac{p+q}{2} + \frac{p-q}{2}$ और
$ q $ इसका दर्पण, का \omterm{def:g10:algebra:expand}{विस्तार} और जोड़ना:
$\sin p + \sin q = 2\sin\frac{p+q}{2}\cos\frac{p-q}{2}$. के लिए
$440$ और $444$ हर्ट्ज: एक टोन पर $442$ Hz जिसकी प्रबलता है
द्वारा संशोधित $\cos(2\pi \cdot 2t)$ --- तीव्रता चरम पर है $4$
प्रति सेकंड बार (लिफाफे का)। \omterm{def:g10:numbers:abs}{निरपेक्ष मूल्य}): चार बीट
एक सेकंड, कांटे सहमत होते ही गायब हो जाता है।

\textbf{15.} $\cos 3x = \cos 2x\cos x - \sin 2x \sin x =
(2\cos^2 x - 1)\cos x - 2\sin^2 x\cos x = 4\cos^3 x -
3\cos x $. पर $ x = 20^\circ $: $\cos 60^\circ = \frac12$, इसलिए
$ c = \cos 20^\circ $ संतुष्ट $8c^3 - 6c - 1 = 0$ --- एक घन
नहीं के साथ वर्गमूल समाधान: निर्माण $20^\circ $, \omterm{def:g11:stat:mean}{अर्थ}ात\
त्रिविभाजक $60^\circ $, शासक और दिशा सूचक यंत्र से परे है।

\textbf{16.} $ R = \sqrt{9 + 27} = 6$;
$\tan\varphi = \frac{3\sqrt3}{3} = \sqrt3$:
$\varphi = \frac\pi3$: $ I(t) = 6\cos\left(100\pi t -
\frac\pi3\right)$: चोटी $6$ एम्पीयर, चरण \omterm{def:g10:numbers:interval}{अंतराल} $60^\circ $.

\textbf{17.} $\sin^2 t = \frac{1 - \cos 2t}{2}$: अवधि
$\pi $. $\sin 2t $ अवधि है $\pi $, $\sin 3t $ है
$\frac{2\pi}{3}$: योग सबसे छोटे सामान्य के बाद दोहराया जाता है
एकाधिक, $2\pi $.

\textbf{18.} काल का एक दोलन $1$ अंदर सिकुड़ रहा है
लिफाफा $\pm\eu^{-t/10}$: प्रत्येक झूले के बारे में $10\,\%$ निचला। पर
$ t = 10$ लिफाफा है $\eu^{-1} \approx 0.368$ --- द
का स्थिरांक \cref{pb:g12:exp:1} फिर भी. \omterm{def:g10:algebra:equation}{समीकरण} ऐसे के साथ
नम समाधान ($ y'' + a y' + b y = 0$) में हल किये जाते हैं
अंतर \omterm{def:g10:algebra:equation}{समीकरण} अध्याय.

\textbf{19.} आनुपातिक रूप से पीछे खींचने वाला बल
विस्थापन देता है $ y'' = -\omega^2 y $; और
\cref{exo:g12:trigo:9} उसका समाधान दिखाता है \omterm{def:g10:algebra:equation}{समीकरण}
हैं \emph{बिल्कुल} के संयोजन $\cos $ और $\sin $ ---
प्रदर्शन से अस्तित्व, अभ्यास से विशिष्टता: कंपन
साइनसॉइडल होने के अलावा कोई विकल्प नहीं है।

\textbf{20.} एक सीमा ($\frac{\sin x}{x} \to 1$, द्वारा
सैंडविच क्षेत्र), दो डेरिवेटिव ($\sin' = \cos $,
$\cos' = -\sin $), एक \omterm{def:g10:algebra:equation}{समीकरण} ($ y'' = -\omega^2 y $), एक दुनिया
घड़ियों, तारों, धाराओं और ज्वार की। तारांकन: विस्तृत के लिए
झूले, $ T = 4\sqrt{\frac Lg} \cdot
\frac{\pi/2}{M(1, \cos(\theta_0/2))}$ --- द
अंकगणित--ज्यामितीय \omterm{def:g11:stat:mean}{अर्थ} का \cref{pb:g12:seq:1} की गणना करता है
मुट्ठी भर पुनरावृत्तियों में अण्डाकार अभिन्न अंग: गॉस की डायरी
प्रवेश और गैलीलियो का पेंडुलम, एक सूत्र से अलग।
\end{solution}
