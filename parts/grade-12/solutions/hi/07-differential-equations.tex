\section*{अध्याय \ref{ch:g12:diffeq} --- विभेदक समीकरण}

\begin{solution}{exo:g12:diffeq:1}
\emph{(ए)} $y = C\eu^{3x}$; $ y(0) = 2$ देता है $ y = 2\eu^{3x}$.

\emph{(बी)} $ y' = -\frac12 y $, इसलिए $ y = C\eu^{-x/2}$; $ y(0) = -1$ देता है
$ y = -\eu^{-x/2}$.

\emph{(सी)} संतुलन $ y_p = 5$; $ y = C\eu^{-x} + 5$; $ y(0) = 0$ देता है
$ C = -5$: $ y = 5\left(1 - \eu^{-x}\right)$.
\end{solution}

\begin{solution}{exo:g12:diffeq:2}
$ N' = kN $, इसलिए $ N(t) = N_0 \eu^{kt}$. दोगुना हो जाना $3$ घंटे का मतलब है
$\eu^{3k} = 2$, \emph{यानी}
\[
k = \frac{\ln 2}{3} \approx 0.231\ \text{h}^{-1}.
\]
\end{solution}

\begin{solution}{exo:g12:diffeq:3}
$ y_p'(x) = \eu^x + x\eu^x = y_p(x) + \eu^x $: $ y_p $ एक विशेष समाधान है
(यह इसका प्रतिध्वनित मामला है \cref{met:g12:diffeq:particular}). द्वारा
\cref{prop:g12:diffeq:general}, समाधान हैं
$ y = C\eu^{x} + x\eu^{x} = (C + x)\,\eu^x $, $ C \in \R $.
\end{solution}

\begin{solution}{exo:g12:diffeq:4}
कोशिश $ y_p = \beta\eu^x $: $\beta\eu^x = -2\beta\eu^x + \eu^x $ देता है
$3\beta = 1$, इसलिए $ y_p = \frac13\eu^x $. सामान्य समाधान
$ y = C\eu^{-2x} + \frac13\eu^{x}$; स्थिति $ y(0) = 1$ देता है
$ C = \frac23$:
\[
y(x) = \frac{2}{3}\,\eu^{-2x} + \frac{1}{3}\,\eu^{x}.
\]
\end{solution}

\begin{solution}{exo:g12:diffeq:5}
$ N(t) = N_0\,\eu^{-\lambda t}$ साथ $\lambda = \frac{\ln 2}{5730}$
(\cref{ex:g12:exp:halflife}). हमने सलुझाया $\eu^{-\lambda t} = 0.6$:
\[
t = \frac{\ln(1/0.6)}{\lambda} = 5730\,\frac{\ln(5/3)}{\ln 2}
\approx 5730 \times \frac{0.5108}{0.6931} \approx 4220 \text{ years}.
\]
\end{solution}

\begin{solution}{exo:g12:diffeq:6}
\emph{1.} नमक बहता है $0.2 \times 5 = 1$ किग्रा/मिनट. बहिर्वाह वहन करता है
एकाग्रता $\frac{m}{100}$ किग्रा/लीटर पर $5$ एल/मिनट, \emph{यानी} $\frac{m}{20}$
किग्रा/मिनट. इस तरह $ m' = 1 - \frac{m}{20}$.

\emph{2.} संतुलन $ m_p = 20$; $ m(t) = 20 + C\eu^{-t/20}$, और
$ m(0) = 0$ देता है $ C = -20$:
\[
m(t) = 20\left(1 - \eu^{-t/20}\right) \xrightarrow[t\to+\infty]{} 20 \text{ kg}.
\]
लंबे समय में टैंक की सांद्रता आने वाले नमकीन पानी के बराबर होती है:
$0.2$ किग्रा/ली $\times $ $100$ एल $= 20$ किग्रा.
\end{solution}

\begin{solution}{exo:g12:diffeq:7}
\emph{1.} $ v' = g - \frac{k}{m} v $ साथ $\frac km = \frac{16}{80} = 0.2$.
संतुलन $ v_\infty = \frac{mg}{k} = \frac{9.8}{0.2} = 49$ एमएस;
$ v(t) = 49 + C\eu^{-0.2t}$, और $ v(0) = 0$ देता है
\[
v(t) = 49\left(1 - \eu^{-0.2 t}\right).
\]

\emph{2.} टर्मिनल वेग $49$ एमएस ($\approx 176$ किमी/घंटा). हम चाहते हैं
$1 - \eu^{-0.2t} = 0.95$, \emph{यानी} $\eu^{-0.2t} = 0.05$:
\[
t = \frac{\ln 20}{0.2} \approx \frac{3.00}{0.2} \approx 15 \text{ s}.
\]
\end{solution}

\begin{solution}{exo:g12:diffeq:8}
\emph{1.} $ z = \frac1y $ देता है $ z' = -\frac{y'}{y^2}
= -\frac{y(1-y)}{y^2} = -\frac{1-y}{y} = -\frac1y + 1 = -z + 1$.

\emph{2.} संतुलन $ z_p = 1$, इसलिए $ z(t) = 1 + C\eu^{-t}$. से
$ y(0) = \frac{1}{10}$, $ z(0) = 10$, इसलिए $ C = 9$ और
\[
y(t) = \frac{1}{1 + 9\,\eu^{-t}} .
\]

\emph{3.} जैसा $ t \to +\infty $, $9\eu^{-t} \to 0$ और $ y(t) \to 1$. वक्र
क्लासिक एस-आकार है (\emph{अवग्रह}) लॉजिस्टिक वक्र: धीमी वृद्धि
प्रथम ($ y $ छोटा, $ y' \approx y $), सबसे तेज वृद्धि कब $ y = \frac12$
(कहाँ $ y' = y(1-y)$ \omterm{def:g10:functions:extrema}{अधिकतम} है), फिर ले जाने की दिशा में संतृप्ति
क्षमता $1$.
\end{solution}

\begin{solution}{exo:g12:diffeq:9}
द्वारा \cref{thm:g12:diffeq:affine}, $ y(x) = C\eu^{ax} - \frac ba $. इस तरह
\[
u_{n+1} = C\eu^{a(n+1)} - \frac ba
= \eu^{a}\left(C\eu^{an} - \frac ba\right) + \frac ba\left(\eu^a - 1\right)
= q\,u_n + r
\]
साथ $ q = \eu^{a}$ और $ r = \frac{b}{a}\left(\eu^{a} - 1\right)$.
स्थिति $\abs q < 1$ मतलब $\eu^a < 1$, \emph{यानी} $ a < 0$: बिलकुल
वह स्थिति जिसके तहत समाधान \omterm{def:g12:diffeq:ode}{विभेदक समीकरण}
संतुलन में एकत्रित हों $-\frac ba $ --- और वास्तव में \omterm{pb:g10:functions:1}{नियत बिन्दु} का
पुनरावृत्ति है $\frac{r}{1 - q} = -\frac{b}{a}$.
\end{solution}

\begin{solution}{pb:g12:diffeq:1}
\textbf{1.} $ y = 2\eu^{3t}$. दूसरे के लिए: संतुलन $3$,
इसलिए $ y = 3 + C\eu^{-2t}$ साथ $ y(0) = 0$: $ C = -3$:
$ y = 3\left(1 - \eu^{-2t}\right)$.

\textbf{2.} $\left(y\eu^{-at}\right)' = y'\eu^{-at} -
ay\eu^{-at} = (y' - ay)\eu^{-at} = 0$: उत्पाद एक है
स्थिर $ C $, इसलिए $ y = C\eu^{at}$ --- कोई समाधान नहीं बचता.

\textbf{3.} संतुलन: $ y \equiv 3$. हर दूसरा समाधान है
$3 + C\eu^{-2t}$ साथ $ C \neq 0$: घातांक मर जाता है और
समाधान पर ग्लाइड होता है $3$ --- एक स्थिर \omterm{pb:g10:functions:1}{नियत बिन्दु}, द
निरंतर पुनरावृत्ति का जुड़वां निश्चित बिंदु का
\cref{pb:g11:seq:1}.

\textbf{4.} $ y = y_0\eu^{-t/\tau}$ आधा जब
$\eu^{-t/\tau} = \frac12$: $ t = \tau\ln 2$.

\textbf{5.} सभी समाधान क्षय के ऊर्ध्वाधर अनुवाद हैं
की ओर $3$: जो ऊपर से शुरू करते हैं वे नीचे आते हैं $3$, जो नीचे हैं
की ओर बढ़ना $3$, और निरंतर समाधान $3$ अभी भी बैठता है --- ए
संतुलन के चारों ओर फ़नल।

\textbf{6.} $\lambda = \frac{\ln 2}{5730} \approx 1.21 \times
10^{-4}$ प्रति वर्ष.

\textbf{7.} $\eu^{-\lambda t} = 0.2$:
$ t = \frac{\ln 5}{\lambda} \approx 13\,300$ साल।

\textbf{8.} $ t = \frac{\ln(1/0.15)}{\lambda} \approx 15\,700$
वर्ष: लास्काक्स के बैल हिमयुग के अंत में हैं।

\textbf{9.} $ t = \frac{\ln(1/0.53)}{\lambda} \approx 5\,250$
वर्ष: पुरातत्वविदों के जीवनकाल के भीतर ---
घड़ी काम करती है.

\textbf{10.} दस आधे जीवन छूट जाते हैं $2^{-10} \approx 0.1\,\%$:
माप के किनारे पर. बाद $65$ मिलियन वर्ष
अंश है $2^{-65\,000\,000/5\,730} \approx 2^{-11\,344}$:
किसी भी जीवाश्म में मूल स्टॉक का कोई भी परमाणु नहीं रहता ---
डायनासोरों का काल धीमी घड़ियों (पोटेशियम--आर्गन और) द्वारा निर्धारित किया गया है
परिजन)।

\textbf{11.} $52\,\%$ देता है $\approx 5\,405$ साल,
$54\,\%$ देता है $\approx 5\,095$: द $\pm 1\,\%$ रसायन विज्ञान का
मोटे तौर पर हो जाता है $\pm 155$ इतिहास के वर्षों --- परिशुद्धता में
प्रयोगशाला संग्रहालय लेबल में सटीक है।

\textbf{12.} $ z = T - T_a $ संतुष्ट $ z' = -kz $:
$ z = (T_0 - T_a)\eu^{-kt}$, और $ T = T_a + z $.

\textbf{13.} प्रारंभिक \omterm{def:g10:numbers:interval}{अंतराल} है $ T_0 - T_a = 70$; पांच के बाद
मिनट का अंतर है $70 - 20 = 50$, इसलिए $50 = 70\eu^{-5k}$:
$ k = \frac{\ln(70/50)}{5} \approx 0.067$ प्रति मिनट. पीने योग्य:
$55 - 20 = 35 = 70\eu^{-kt}$:
$ t = \frac{\ln 2}{k} \approx 10.3$ मिनट।

\textbf{14.} हानि दर अंतर के समानुपाती होती है
$ T - T_a $: गरम-गरम कॉफी सबसे तेजी से गर्मी खत्म करती है। \omterm{def:g10:vectors:sum}{जोड़} रहा हूँ
दूध \emph{अब} अंतर को तुरंत कम कर देता है, इसलिए कम गर्मी होती है
दस मिनट से अधिक खो गया; अंत में इसे जोड़ने से
कॉफ़ी को पहले पूरी गति से ठंडा करें। सबसे गर्म कप के लिए: दूध
प्रथम. (अधीर मेज़बानों की ऊष्मागतिकी पीछे की ओर होती है।)

\textbf{15.} परिवेश से \omterm{def:g10:numbers:interval}{अंतराल}: आधी रात को $10$, एक घंटा
बाद में $8$: $\eu^{-k} = 0.8$:
$ k = \ln\frac{10}{8} \approx 0.223$ घंटे से। मृत्यु पर \omterm{def:g10:numbers:interval}{अंतराल}
था $17$; यह सड़ गया $10$ आधी रात से:
$ s = \frac{\ln(17/10)}{k} \approx 2.4$ घंटे। मृत्यु का समय:
के बारे में $21{:}40$.

\textbf{16.} एक गर्म या सूखा कमरा ($ T_a $ स्थिर नहीं)
घड़ी को किसी भी ओर मोड़ देता है; कपड़े या शरीर के द्रव्यमान में परिवर्तन $ k $
(कोरोनर की टेबल इसके लिए समायोजित होती हैं) --- गलत $ k $ तराजू
संपूर्ण \omterm{def:g10:numbers:interval}{अंतराल}; और एक निकाय कहीं और से स्थानांतरित होकर रीसेट हो जाता है
$ T_a $ मध्य-क्षय, पहले या बाद में मृत्यु का दिखावा करना।
\omterm{def:g10:algebra:equation}{समीकरण} ईमानदार है; धारणाएँ जोखिम उठाती हैं।

\textbf{17.} $ z = \frac1y $ का अनुसरण करता है $ z' = 1 - z $:
$ z = 1 + 9\eu^{-t}$ (से $ z(0) = 10$), इसलिए
$ y = \frac{1}{1 + 9\eu^{-t}}$. पर विभक्ति $ y = \frac12$:
$9\eu^{-t} = 1$: $ t = \ln 9 \approx 2.2$ ---महामारी का
परिवर्तन की तिथि, सीधे सूत्र से।

\textbf{18.} टर्मिनल वेग: $ v' = 0$ पर $ v = 20$ एमएस।
समाधान: $ v(t) = 20\left(1 - \eu^{-t/2}\right)$. तब
$0.95$: $\eu^{-t/2} = 0.05$: $ t = 2\ln 20 \approx 6$ एस ---
बारिश की बूंदें कुछ ही सेकंड में अपनी गति पकड़ लेती हैं, जो कि है
बारिश क्यों नहीं मारती.

\textbf{19.} स्थिर अवस्था: $ c = 8$. इसका आधा हिस्सा:
$4 = 8\left(1 - \eu^{-t/2}\right)$: $\eu^{-t/2} = \frac12$:
$ t = 2\ln 2 \approx 1.4$ घंटे। ड्रिप है निरंतर
ऋण का जुड़वां: निरंतर प्रवाह, आनुपातिक बहिर्वाह ---
\cref{exo:g12:diffeq:9} शब्दकोश को सटीक बनाता है।

\textbf{20.} क्षय ($ a < 0$, $ b = 0$), ठंडा करना ($ y = T - T_a $),
गिरना ($ a < 0$, $ b > 0$: से संतुलन के लिए दृष्टिकोण
नीचे), खुराक (समान, एक नस में): एक \omterm{def:g10:algebra:equation}{समीकरण}, चार लोक।
लूप: परिवर्तन का नियम बताएं; लिखना \omterm{def:g10:algebra:equation}{समीकरण}; हल करना
(\cref{met:g12:diffeq:solve}); स्थिरांकों को अंशांकित करें
माप; भविष्यवाणी करना; फिर मान्यताओं का ऑडिट करें (प्रश्न)।
16). दोलन की आवश्यकता है $ y'' = -\omega^2 y$ --- मिले, सुलझे और
झूलते हुए सेट करें \cref{pb:g12:trigo:1}.
\end{solution}
