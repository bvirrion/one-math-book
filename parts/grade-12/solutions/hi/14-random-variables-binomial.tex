\section*{अध्याय \ref{ch:g12:randvar} --- यादृच्छिक चर और द्विपद वितरण}

\begin{solution}{exo:g12:randvar:1}
$\E(X) = -0.3 + 0 + 0.8 + 0.5 = 1$.
$\E(X^2) = 0.3 \times 1 + 0 + 0.4\times4 + 0.1\times25 = 4.4$, तो द्वारा
कोनिग--ह्यूजेन्स $\V(X) = 4.4 - 1 = 3.4$ और
$\sigma(X) = \sqrt{3.4} \approx 1.84$.
\end{solution}

\begin{solution}{exo:g12:randvar:2}
\emph{1.}  $10$ प्रश्न हैं \omterm{def:g12:condprob:indep}{स्वतंत्र} \omterm{def:g12:randvar:bernoulli}{बर्नौली परीक्षण} साथ
$ p = \frac14$: $ X \sim \mathcal B\left(10, \frac14\right)$,
$\E(X) = 2.5$,
$\sigma(X) = \sqrt{10 \times \frac14 \times \frac34} = \sqrt{1.875}
\approx 1.37$.

\emph{2.} $\P(X = 0) = \left(\frac34\right)^{10} \approx 0.056$;
\[
\P(X = 5) = \binom{10}{5}\left(\frac14\right)^5\left(\frac34\right)^5
\approx 0.058;
\quad
\P(X \geq 1) = 1 - \left(\tfrac34\right)^{10} \approx 0.944 .
\]
\end{solution}

\begin{solution}{exo:g12:randvar:3}
\omterm{def:g12:condprob:indep}{स्वतंत्र} समान परीक्षण: $ X \sim \mathcal B(400,\ 0.05)$, इसलिए
$\E(X) = 20$ दावे और
\[
\sigma(X) = \sqrt{400 \times 0.05 \times 0.95} = \sqrt{19} \approx 4.4 .
\]
\end{solution}

\begin{solution}{exo:g12:randvar:4}
भुगतान $ P $ संतुष्टि प्राप्त हुई
$\E(P) = \frac{5 + 6}{6} = \frac{11}{6}$, तो उचित मूल्य है
$ c = \frac{11}{6} \approx 1.83$ यूरो. उचित लाभ $ G = P - \frac{11}6$
मान लेता है $-\frac{11}{6}, \frac{19}{6}, \frac{25}{6}$ संभावनाओं के साथ
$\frac46, \frac16, \frac16$:
\[
\V(G) = \E(G^2)
= \frac{4 \times 121 + 361 + 625}{6 \times 36} = \frac{1470}{216}
= \frac{245}{36} \approx 6.8,
\qquad \sigma(G) \approx 2.6 .
\]
एक निष्पक्ष खेल में शून्य होता है \emph{औसत} लाभ लेकिन इसके परिणाम अभी भी उतार-चढ़ाव वाले हैं:
निष्पक्ष जोखिम रहित नहीं है.
\end{solution}

\begin{solution}{exo:g12:randvar:5}
$ X \sim \mathcal B(8,\ 0.7)$.
\[
\P(X = 6) = \binom86 (0.7)^6(0.3)^2 \approx 0.296 ;
\]
\[
\P(X \geq 6) = \P(6) + \P(7) + \P(8)
\approx 0.296 + 8(0.7)^7(0.3) + (0.7)^8
\approx 0.296 + 0.198 + 0.058 = 0.552 ;
\]
$\P(X \geq 1) = 1 - (0.3)^8 \approx 0.99993$. तरीका:
$(n+1)p = 6.3$, इसलिए सबसे संभावित मान है $ k^* = 6$
(\cref{exo:g12:randvar:7}).
\end{solution}

\begin{solution}{exo:g12:randvar:6}
यात्रियों की संख्या दिख रही है $ X \sim \mathcal B(104,\ 0.9)$; द
जब फ्लाइट ओवरबुक हो जाती है $ X \geq 101$:
\[
\P(X \geq 101) = \sum_{k=101}^{104}\binom{104}{k}(0.9)^k(0.1)^{104-k}
\approx 0.006 .
\]
बेचना $4\%$ सीटों से ज्यादा टिकटें ही किसी घटना का कारण बनती हैं
$0.6\%$ उड़ानों की संख्या --- ओवरबुकिंग के पीछे का अर्थशास्त्र।
\end{solution}

\begin{solution}{exo:g12:randvar:7}
\[
\frac{\P(X = k+1)}{\P(X = k)}
= \frac{\binom{n}{k+1}}{\binom nk}\cdot\frac{p}{1-p}
= \frac{n - k}{k + 1}\cdot\frac{p}{1-p},
\]
का उपयोग करते हुए $\binom{n}{k+1} = \binom nk \frac{n-k}{k+1}$. यह अनुपात है $\geq 1$
आईएफएफ $(n-k)p \geq (k+1)(1-p)$ आईएफएफ $ np - k p \geq k - kp + 1 - p $ आईएफएफ
$ k \leq (n+1)p - 1$. इसलिए संभावनाएं सख्ती से बढ़ जाती हैं
$ k + 1 \leq (n+1)p $ और बाद में कमी: \omterm{def:g10:functions:extrema}{अधिकतम} पर प्राप्त होता है
$ k^* = \floor{(n+1)p}$ (के साथ साझा किया गया) $ k^* - 1$ कब $(n+1)p $ एक
पूर्णांक).
\end{solution}

\begin{solution}{exo:g12:randvar:8}
\emph{1.} टॉस में सबसे पहले हेड $ k $ मतलब $ k-1$ पूँछ फिर सिर:
\omterm{def:g10:proba:distribution}{संभावना} $\left(\frac12\right)^{k-1}\cdot\frac12 = 2^{-k}$, के लिए
$1 \leq k \leq 10$. कुल जीत \omterm{def:g10:proba:distribution}{संभावना}
$\sum_{k=1}^{10} 2^{-k} = 1 - 2^{-10} = \frac{1023}{1024}$ (खेल है
केवल दस लगातार टेल्स पर हारे)।

\emph{2.} अपेक्षित भुगतान:
\[
\sum_{k=1}^{10} 2^k \cdot 2^{-k} = \sum_{k=1}^{10} 1 = 10 \text{ euros}.
\]
बिना सीमा के, योग $\sum_{k\geq1} 1$ विचलन: अपेक्षित भुगतान है
अनंत, हालांकि खेल लगभग हमेशा एक छोटी राशि का भुगतान करता है ---
प्रसिद्ध \emph{सेंट पीटर्सबर्ग \omterm{thm:g12:limcont:ivt}{विरोधाभास}}, वह दिखा रहा है \omterm{def:g12:randvar:exp}{अपेक्षा} अकेला करता है
किसी खेल के मूल्य को मापें नहीं।
\end{solution}

\begin{solution}{pb:g12:randvar:1}
\textbf{1.} $ X \sim \mathcal B\!\left(3, \frac16\right)$:
$\P(X = 0) = \frac{125}{216}$,
$\P(X = 1) = \frac{75}{216}$,
$\P(X = 2) = \frac{15}{216}$,
$\P(X = 3) = \frac{1}{216}$.

\textbf{2.} $\E(X) = 3 \times \frac16 = \frac12$;
$ V(X) = 3 \times \frac16 \times \frac56 = \frac{5}{12}$.

\textbf{3.} अपेक्षित भुगतान $2\,\E(X) = 1$ यूरो बनाम ए
$1$-यूरो हिस्सेदारी: लाभ $0$ --- बिल्कुल उचित (एक दुर्लभ वस्तु)।

\textbf{4.} प्रतिस्थापन के बिना ड्रॉ निर्भर हैं (द
दूसरे कार्ड की \omterm{def:g10:proba:distribution}{संभावना} पहले पर निर्भर करती है): द्विपद नहीं ---
चेकलिस्ट की स्वतंत्रता बॉक्स विफल. प्रतिस्थापन के साथ:
निश्चित $ n = 5$, स्थिर $ p = \frac14$, \omterm{def:g12:condprob:indep}{स्वतंत्र} खींचता है:
द्विपद.

\textbf{5.} $\E(X) = 6$, $\sigma(X) = \sqrt{4.2} \approx
2.05$. $\E(S) = 5 \times 6 - 10 = 20$;
$\sigma(S) = 5\sigma(X) \approx 10.25$.

\textbf{6.} तय $ n = 105$ परीक्षण (टिकट), प्रत्येक ए
उसी के साथ दिखाएँ/न दिखाएँ $ p = 0.9$, मान लिया गया \omterm{def:g12:condprob:indep}{स्वतंत्र}:
द्विपद. स्वीकारोक्ति: परिवारों को उड़ानें याद आती हैं
\emph{एक साथ} और तूफ़ान ने पूरे विमानों को खाली कर दिया --- स्वतंत्रता
यह मॉडल के विश्वास की छलांग है, और सहसंबद्ध नो-शो मेक है
पूँछें द्विपद वादों से भी मोटी हैं।

\textbf{7.} $\E(X) = 94.5$;
$\sigma(X) = \sqrt{105 \times 0.9 \times 0.1} =
\sqrt{9.45} \approx 3.07$.

\textbf{8.} $\P(X \geq 101) = \sum_{k=101}^{105}
\binom{105}{k} 0.9^k\, 0.1^{105 - k}$.

\textbf{9.} $\approx 0.0167$: लगभग साठ में एक उड़ान
किसी को भी टक्कर मार देता है.

\textbf{10.} $\E(\text{bumped}) \approx 0.023$ यात्री प्रति
उड़ान: अपेक्षित मुआवजा $\approx 19$ यूरो --- विरुद्ध
$1\,000$ अतिरिक्त राजस्व के यूरो. पांच तक ओवरबुकिंग है
अत्यधिक लाभदायक; इसलिए हर एयरलाइन ऐसा करती है।

\textbf{11.} पूंछ संभावनाओं की तालिका से:
$ n = 105$: $1.7\,\%$; $ n = 106$: $4.0\,\%$; $ n = 107$:
$8.1\,\%$. सबसे बड़ी अनुपालन बिक्री है $ n = 106$ टिकट.

\textbf{12.} $ z = \frac{100.5 - 94.5}{3.07} \approx 1.95$:
दहलीज दो बैठती है \omterm{def:g11:stat:variance}{मानक विचलन} के ऊपर \omterm{def:g11:stat:mean}{अर्थ},
और अंगूठे का घंटी-वक्र नियम (``ऊपरी $2\sigma $ पूँछ
$\approx 2.5\,\%$'') सटीक के करीब भूमि $1.7\,\%$ ---
द्विपद को छाया देने वाला चिकना वक्र आ रहा है
अध्याय का नायक.

\textbf{13.} $0.98^{20} + 20 \times 0.02 \times 0.98^{19}
\approx 0.94$: ईमानदार व्यक्ति गुजर जाता है $94\,\%$ समय का।

\textbf{14.} $0.9^{20} + 20 \times 0.1 \times 0.9^{19}
\approx 0.39$: बुरी स्थिति अभी भी छुपकर आती है $39\,\%$ समय का।

\textbf{15.} निर्माता का जोखिम: एक अच्छा लॉट अस्वीकृत
($\approx 6\,\%$); उपभोक्ता का जोखिम: एक ख़राब लॉट स्वीकार किया गया
($\approx 39\,\%$). अपेक्षाकृत व्यापक \omterm{def:g10:proba:sample}{नमूना} (आनुपातिक के साथ
स्वीकृति सीमा) दोनों को कम करती है --- अधिक की कीमत पर
निरीक्षण: गुणवत्ता की एक बजट रेखा होती है।

\textbf{16.} $ V\!\left(\frac Xn\right) = \frac{V(X)}{n^2} =
\frac{np(1-p)}{n^2} = \frac{p(1-p)}{n}$: \omterm{def:g11:stat:variance}{मानक विचलन}
$\sqrt{\frac{p(1-p)}{n}}$. चौगुना \omterm{def:g10:proba:sample}{नमूना}, आधा कर दो
शोर: $\frac{1}{\sqrt n}$ ग्रेड 10 के उतार-चढ़ाव का नियम
\omterm{def:g10:numbers:interval}{अंतराल}, अब व्युत्पन्न।

\textbf{17.} यह सीमा भुगतान को सीमित करती है $2^{10} = 1024$
यूरो, इसलिए दस राउंड में से प्रत्येक बिल्कुल सटीक योगदान देता है $1$ यूरो
का \omterm{def:g12:randvar:exp}{अपेक्षा}: $\E = 10$. अनकैप्ड गेम अनंत है
\omterm{def:g12:randvar:exp}{अपेक्षा} पूरी तरह से खगोलीय रूप से दुर्लभ से आया,
खगोलीय रूप से बड़े भुगतान; कैपिंग ने स्वीकार किया कि कोई बैंक नहीं
भुगतान करता है $2^{50}$ यूरो. कैप्ड गेम के लिए उचित टिकट मूल्य:
$10$ यूरो --- और अचानक कोई भी आश्चर्यचकित नहीं होता।

\textbf{18.} प्रति टिकट अपेक्षित पुरस्कार राशि:
$\frac{100 + 2 \times 50}{200} = 1$ यूरो बनाम ए $2$-यूरो
टिकट: मार्जिन $50\,\%$ --- अठारह गुना रूलेट
$2.7\,\%$. रैफ़ल जीवित रहता है क्योंकि उसके खिलाड़ी जीवित रहते हैं
जानबूझकर दान करना: ``नुकसान'' ही \omterm{def:g12:arith:prime}{मुख्य} बात है।

\textbf{19.} अपेक्षित दावे: $1\,000 \times 0.01 \times
10\,000 = 100\,000$; प्रीमियम $130\,000$: अपेक्षित लाभ
$30\,000$ यूरो. \omterm{def:g11:stat:variance}{मानक विचलन} कुल दावों का:
$10\,000 \times \sqrt{1\,000 \times 0.01 \times 0.99}
\approx 31\,500$ यूरो --- एक साधारण बुरा वर्ष नष्ट हो जाता है
संपूर्ण अपेक्षित लाभ. इसलिए पूंजी भंडार, पुनर्बीमा,
और पोर्टफोलियो एक हजार पॉलिसियों से कहीं बड़ा है: बीमाकर्ता
बड़ी संख्या के कानून से दूर रहें और इसके खिलाफ रिजर्व रखें
इसकी सुस्ती.

\textbf{20.} पहले चेकलिस्ट --- तय की गई $ n $, वही $ p $,
स्वतंत्रता \emph{दावा किया} --- या कोई द्विपद ही नहीं. फिर
द्वारा नेविगेट करें $\E \pm \sigma $: का \omterm{def:g11:stat:mean}{अर्थ} है पता लगाना, विचलन चेतावनी देना।
फिर पूंछों की कीमत लगाएं: धमाकेदार, खराब लॉट, सभी विनाशकारी वर्ष
परे जियो $2\sigma$. चेतावनियाँ: स्वतंत्रता मॉडलर का है
वादा करो, दुनिया का नहीं; और \omterm{def:g12:randvar:exp}{अपेक्षा} बिल्कुल अनदेखा करता है
जो तुम्हें नष्ट कर देता है. नियम पुस्तिका के गायब पन्ने --- कितनी तेजी से
आवृत्तियाँ स्थिर हो जाती हैं, और उतार-चढ़ाव क्या आकार लेते हैं ---
अगले दो अध्याय हैं.
\end{solution}
