\begin{solution}{exo:g12:complex:3}
$\abs{z_1} = \sqrt2$, \omterm{def:g12:complex:argument}{तर्क} $\frac{3\pi}{4}$ (दूसरा चतुर्थांश):
$ z_1 = \sqrt2\,\eu^{3\iu\pi/4}$.
$\abs{z_2} = 2$, \omterm{def:g12:complex:argument}{तर्क} $-\frac\pi6$:
$ z_2 = 2\,\eu^{-\iu\pi/6}$. इस तरह
\[
\frac{z_1}{z_2} = \frac{\sqrt2}{2}\,\eu^{\iu\left(\frac{3\pi}{4} + \frac{\pi}{6}\right)}
= \frac{\sqrt2}{2}\,\eu^{11\iu\pi/12}.
\]
\omterm{def:g12:complex:def}{बीजगणितीय रूप} से,
\[
\frac{z_1}{z_2} = \frac{-1+\iu}{\sqrt3 - \iu}
= \frac{(-1+\iu)(\sqrt3+\iu)}{4}
= \frac{-\sqrt3 - 1 + \iu(\sqrt3 - 1)}{4}.
\]
पहचान करना वास्तविक भाग:
$\frac{\sqrt2}{2}\cos\frac{11\pi}{12} = \frac{-\sqrt3-1}{4}$, इसलिए
\[
\cos\frac{11\pi}{12} = -\frac{\sqrt6 + \sqrt2}{4}.
\]
\end{solution}

\begin{solution}{exo:g12:complex:4}
\emph{(ए)} $\abs{z - 2} = \abs{z - (-\iu)}$ मतलब $ M $ से समान दूरी पर है
$ A(2, 0)$ और $ B(0, -1)$: द \emph{लंब समद्विभाजक} का $[AB]$.

\emph{(बी)} दूरी $3$ बिन्दु से $(1,1)$: द \emph{घेरा} \omterm{pb:g11:quad:1}{केंद्र} का
$1 + \iu $ और त्रि\omterm{def:g12:trigo:cossin}{ज्या} $3$.

\emph{(सी)} द्वारा \cref{prop:g12:complex:geometry}(4), $\frac{z-1}{z+1}$ विशुद्ध रूप से
काल्पनिक का मतलब है वैक्टर से $ A(1,0)$ को $ M $ और से $ B(-1,0)$ को $ M $
हैं \omterm{def:g11:scal:orthogonal}{ओर्थोगोनल}: $ M $ खंड देखता है $[AB]$ एक समकोण के नीचे. सेट है
द \emph{व्यास का चक्र $[AB]$} (इकाई चक्र) \emph{शून्य से
अंक $ A $ और $ B $} खुद।
\end{solution}

\begin{solution}{exo:g12:complex:5}
डी मोइवरे: $\cos3\theta + \iu\sin3\theta = (\cos\theta + \iu\sin\theta)^3$.
\omterm{def:g10:algebra:expand}{विस्तार} द्विपद प्रमेय द्वारा $ c = \cos\theta $, $ s = \sin\theta $:
\[
(c + \iu s)^3 = c^3 + 3\iu c^2 s - 3 c s^2 - \iu s^3
= (c^3 - 3cs^2) + \iu(3c^2 s - s^3).
\]
भागों की पहचान करना और उपयोग करना $ s^2 = 1 - c^2$, $ c^2 = 1 - s^2$:
\[
\cos3\theta = 4\cos^3\theta - 3\cos\theta,
\qquad
\sin3\theta = 3\sin\theta - 4\sin^3\theta .
\]
\end{solution}

\begin{solution}{exo:g12:complex:6}
\[
\cos^3\theta = \left(\frac{\eu^{\iu\theta} + \eu^{-\iu\theta}}{2}\right)^{\!3}
= \frac{\eu^{3\iu\theta} + 3\eu^{\iu\theta} + 3\eu^{-\iu\theta} + \eu^{-3\iu\theta}}{8}
= \frac{\cos3\theta + 3\cos\theta}{4}.
\]
इस तरह
\[
\int_0^{\pi/2}\cos^3\theta\,\dd\theta
= \frac14\left[\frac{\sin3\theta}{3} + 3\sin\theta\right]_0^{\pi/2}
= \frac14\left(-\frac13 + 3\right) = \frac{2}{3}.
\]
(के साथ संगत $ W_3 = \frac23$ में \cref{exo:g12:integ:9}, तब से
$\int_0^{\pi/2}\cos^3 = \int_0^{\pi/2}\sin^3$ समरूपता द्वारा
$\theta \mapsto \frac\pi2 - \theta $.)
\end{solution}

\begin{solution}{exo:g12:complex:7}
$ c = a + \eu^{\pm\iu\pi/3}(b - a)$ साथ $ b - a = 2 + \iu $ और
$\eu^{\pm\iu\pi/3} = \frac12 \pm \iu\frac{\sqrt3}{2}$:
\[
\eu^{\iu\pi/3}(2+\iu) = \left(\tfrac12 + \iu\tfrac{\sqrt3}{2}\right)(2+\iu)
= \left(1 - \tfrac{\sqrt3}{2}\right) + \iu\left(\tfrac12 + \sqrt3\right),
\]
इसलिए $ c_1 = \left(2 - \frac{\sqrt3}{2}\right) +
\iu\left(\frac32 + \sqrt3\right)$, और इसी प्रकार कोण के साथ भी
$-\frac\pi3$:
$ c_2 = \left(2 + \frac{\sqrt3}{2}\right) + \iu\left(\frac32 - \sqrt3\right)$.
\end{solution}

\begin{solution}{exo:g12:complex:8}
$-4 = 4\,\eu^{\iu\pi}$, तो के समाधान $ z^4 = -4$ हैं
\[
z_k = \sqrt2\,\eu^{\iu\left(\frac\pi4 + \frac{k\pi}{2}\right)},
\quad k = 0, 1, 2, 3,
\]
\emph{यानी} $1 + \iu $, $-1 + \iu $, $-1 - \iu $, $1 - \iu $: a के शीर्ष
परित्रि\omterm{def:g12:trigo:cossin}{ज्या} का वर्ग $\sqrt 2$. समूहन \omterm{def:g12:complex:conjugate}{संयुग्मित} \omterm{def:g11:quad:discriminant}{जड़}ों,
\[
z^4 + 4 = \bigl(z - (1{+}\iu)\bigr)\bigl(z - (1{-}\iu)\bigr)
\bigl(z + (1{-}\iu)\bigr)\bigl(z + (1{+}\iu)\bigr)
= (z^2 - 2z + 2)(z^2 + 2z + 2).
\]
\end{solution}

\begin{solution}{exo:g12:complex:9}
साथ $ q = \eu^{\iu\theta} \neq 1$:
\[
\sum_{k=0}^{n} \eu^{\iu k\theta} = \frac{\eu^{\iu(n+1)\theta} - 1}{\eu^{\iu\theta} - 1}
= \frac{\eu^{\iu(n+1)\theta/2}}{\eu^{\iu\theta/2}}
\cdot\frac{\eu^{\iu(n+1)\theta/2} - \eu^{-\iu(n+1)\theta/2}}{\eu^{\iu\theta/2} - \eu^{-\iu\theta/2}}
= \eu^{\iu n\theta/2}\,
\frac{\sin\frac{(n+1)\theta}{2}}{\sin\frac{\theta}{2}},
\]
का उपयोग करते हुए $\eu^{\iu\alpha} - \eu^{-\iu\alpha} = 2\iu\sin\alpha $ (द
\emph{अर्ध-कोण \omterm{def:g10:algebra:expand}{फैक्टरिंग}}). लेना वास्तविक भाग:
\[
S = \frac{\sin\frac{(n+1)\theta}{2}}{\sin\frac{\theta}{2}}\,
\cos\frac{n\theta}{2}.
\]
\end{solution}

\begin{solution}{exo:g12:complex:10}
\emph{1.} $\omega $ एक है $5$-वां \omterm{def:g11:quad:discriminant}{जड़} एकता से भिन्न $1$, तो योग
सभी पांचों में से \omterm{def:g11:quad:discriminant}{जड़}ों गायब हो जाता है (\cref{thm:g12:complex:rootsofunity}), जो पढ़ता है
$1 + \omega + \omega^2 + \omega^3 + \omega^4 = 0$.

\emph{2.} $ u + v = \omega + \omega^2 + \omega^3 + \omega^4 = -1$ बिंदु 1 के अनुसार.
उत्पाद के लिए, उपयोग कर रहे हैं $\omega^5 = 1$:
\[
uv = (\omega + \omega^4)(\omega^2 + \omega^3)
= \omega^3 + \omega^4 + \omega^6 + \omega^7
= \omega^3 + \omega^4 + \omega + \omega^2 = -1 .
\]
इसलिए $ u, v $ हैं \omterm{def:g11:quad:discriminant}{जड़}ों का $ X^2 + X - 1 = 0$:
$\left\{\frac{-1+\sqrt5}{2}, \frac{-1-\sqrt5}{2}\right\}$. अब
$ u = \omega + \conj\omega = 2\cos\frac{2\pi}{5} > 0$ (तब से
$\frac{2\pi}{5} < \frac{\pi}{2}$), इसलिए $ u = \frac{-1 + \sqrt5}{2}$.

\emph{3.} $\cos\frac{2\pi}{5} = \frac u2 = \frac{\sqrt5 - 1}{4}$.
\end{solution}

\begin{solution}{pb:g12:complex:1}
\textbf{1.} $(2 + \iu)(3 - \iu) = 6 - 2\iu + 3\iu + 1 =
7 + \iu $. \quad $\frac{1 + \iu}{1 - \iu} =
\frac{(1 + \iu)^2}{2} = \frac{2\iu}{2} = \iu $. \quad
$\abs{3 + 4\iu} = 5$.

\textbf{2.} $1 + \iu = \sqrt2\,\eu^{\iu\pi/4}$;
$-2 = 2\eu^{\iu\pi}$;
$1 - \iu\sqrt3 = 2\eu^{-\iu\pi/3}$.

\textbf{3.} $(1 + \iu)^8 = \left(\sqrt2\right)^8
\eu^{8\iu\pi/4} = 16\,\eu^{2\iu\pi} = 16$.

\textbf{4.} $ z^2 = \eu^{\iu\pi/2}$:
$ z = \pm\eu^{\iu\pi/4} = \pm\frac{\sqrt2}{2}(1 + \iu)$. और
$\Delta = 4 - 20 = -16$: $ z = 1 \pm 2\iu $.

\textbf{5.} $ z \mapsto \iu z $: एक चौथाई मोड़ से घूमना
उत्पत्ति के बारे में. $\abs{z - (1 + \iu)} = 2$: का घेरा
\omterm{pb:g11:quad:1}{केंद्र} $1 + \iu $ और त्रि\omterm{def:g12:trigo:cossin}{ज्या} $2$.

\textbf{6.} $\left(\omega^k\right)^5 = \eu^{2\iu\pi k} = 1$,
और पांच अंक $\eu^{2\iu k\pi/5}$ अलग हैं: सभी पाँच
के समाधान $ z^5 = 1$. वे इकाई वृत्त पर बराबर पर बैठते हैं
के कोण $72^\circ $: एक शीर्ष पर एक नियमित पंचभुज
$1$.

\textbf{7.} $ S = 1 + \omega + \dots + \omega^4$ संतुष्ट
$(1 - \omega)S = 1 - \omega^5 = 0$, और $\omega \neq 1$:
$ S = 0$. (पंचकोण का संतुलन बिंदु इसका \omterm{pb:g11:quad:1}{केंद्र} है।)

\textbf{8.} $\omega^4 = \overline\omega $ और
$\omega^3 = \overline{\omega^2}$: \omterm{def:g10:vectors:sum}{जोड़}ी \omterm{def:g12:complex:conjugate}{संयुग्मित} करता है, द असली
भाग योग का पढ़ता है
$1 + 2\cos\frac{2\pi}{5} + 2\cos\frac{4\pi}{5} = 0$.

\textbf{9.} $\cos\frac{4\pi}{5} = 2c^2 - 1$, इसलिए
$1 + 2c + 4c^2 - 2 = 0$, यानी\ $4c^2 + 2c - 1 = 0$:
$ c = \frac{-1 + \sqrt5}{4}$ (सकारात्मक \omterm{def:g11:quad:discriminant}{जड़}, तब से
$72^\circ $ तीव्र है):
$\cos 72^\circ = \frac{\sqrt5 - 1}{4}$.

\textbf{10.} संख्यानुसार $ c \approx 0.30902$: से मेल खाता है
कैलकुलेटर. और $\frac{1}{2\varphi} = \frac{1}{1 + \sqrt5} =
\frac{\sqrt5 - 1}{4}$ (तर्कसंगत): $ c = \frac{1}{2\varphi}$.
पंचकोण पूरी तरह से सुनहरा है: इसके विकर्ण कट
अनुपात में इसका पक्ष $\varphi $ --- पेंटाग्राम अंतहीन है
स्व-समानता.

\textbf{11.} वर्गमूल कम्पास-निर्माण योग्य हैं (द
मिडिल स्कूल वॉल्यूम की ज्यामितीय अर्ध-वृत्त मशीन
\omterm{def:g11:stat:mean}{अर्थ}), इसलिए केवल वर्गमूलों द्वारा परिमेय से निर्मित एक संख्या
--- जैसे $\cos 72^\circ $ --- रचनात्मक है; घनमूल हैं
नहीं, और $\cos 20^\circ $ एक की जरूरत है: पेंटागन उपज,
ट्राइसेक्शन प्रतिरोध करता है। गॉस की कसौटी हर नियमित रूप से तय होती है
बहुभुज --- उन्नीस साल की उम्र में उसके 17-गॉन ने उसे चुनने पर मजबूर कर दिया
गणित.

\textbf{12.} $-4 = 4\eu^{\iu\pi}$: चौथा \omterm{def:g11:quad:discriminant}{जड़}ों पास होना
\omterm{def:g12:complex:modulus}{मापांक} $\sqrt2$ और बहस
$\frac{\pi}{4} + k\frac{\pi}{2}$: चार संख्याएँ
$\pm 1 \pm \iu $. बाँधना \omterm{def:g12:complex:conjugate}{संयुग्मित} करता है:
$(z - 1 - \iu)(z - 1 + \iu) = z^2 - 2z + 2$ और
$(z + 1 - \iu)(z + 1 + \iu) = z^2 + 2z + 2$: द
का गुणनखंडन \cref{exo:g12:complex:8}.

\textbf{13.} $(\cos\theta + \iu\sin\theta)^3 = \cos^3\theta +
3\iu\cos^2\theta\sin\theta - 3\cos\theta\sin^2\theta -
\iu\sin^3\theta $; डी मोइवरे द्वारा यह बराबर है
$\cos 3\theta + \iu \sin 3\theta $. असली हिस्से:
$\cos 3\theta = \cos^3\theta - 3\cos\theta(1 - \cos^2\theta)
= 4\cos^3\theta - 3\cos\theta $ --- और यह \omterm{def:g12:complex:def}{काल्पनिक भाग}
सौंप दो $\sin 3\theta $ मुक्त करने के लिए।

\textbf{14.} यह है \omterm{def:g12:complex:def}{असली हिस्सा} का
$1 + \omega + \dots + \omega^4 = 0$: योग बराबर है $0$ ---
बिल्कुल प्रश्न 8, देखा गया
\cref{exo:g12:complex:9}की गिरी.

\textbf{15.} $8\iu = 8\eu^{\iu\pi/2}$: \omterm{def:g11:quad:discriminant}{जड़}ों
$2\eu^{\iu\pi/6} = \sqrt3 + \iu $;
$2\eu^{5\iu\pi/6} = -\sqrt3 + \iu $;
$2\eu^{3\iu\pi/2} = -2\iu $.

\textbf{16.} $1 + \iu = \sqrt2\,\eu^{\iu\pi/4}$: नक्शा
द्वारा विमान को घुमाता है $45^\circ $ और इसे मापता है $\sqrt2$. द
इकाई वर्ग भुजा का झुका हुआ वर्ग बन जाता है $\sqrt2$ ---
क्षेत्रफल में दोगुना हो गया, एक मोड़ का आठवां हिस्सा घूम गया।

\textbf{17.} $\abs{z_1 z_2}^2 = z_1 z_2 \overline{z_1 z_2} =
z_1\overline{z_1}\, z_2\overline{z_2} =
\abs{z_1}^2\abs{z_2}^2$: वर्गमूल लें. की शक्तियां
$1 + \iu $: $1 + \iu $, $2\iu $, $-2 + 2\iu $, $-4$: प्रत्येक ए
$45^\circ $ आगे मुड़ें और $\sqrt2$ कई गुना अधिक --- एक
मॉड्यूलि के माध्यम से बाहरी सर्पिल
$\sqrt2, 2, 2\sqrt2, 4, \dots $

\textbf{18.} $\mathrm{j}$ एक तिहाई है \omterm{def:g11:quad:discriminant}{जड़} एकता का
से भिन्न $1$, तो (प्रश्न 7) \omterm{def:g12:complex:argument}{तर्क} साथ $ n = 3$)
$1 + \mathrm{j} + \mathrm{j}^2 = 0$. मानक पर कसौटी
त्रिकोण: $ a + \mathrm{j}b + \mathrm{j}^2 c = 1 +
\mathrm{j}\cdot\mathrm{j} + \mathrm{j}^2 \cdot \mathrm{j}^2 =
1 + \mathrm{j}^2 + \mathrm{j}^4 = 1 + \mathrm{j}^2 +
\mathrm{j} = 0$: संतुष्ट, समबाहु त्रिभुज के रूप में
$\left(1, \mathrm{j}, \mathrm{j}^2\right)$ मांग.

\textbf{19.} ऊपर $\C $, एक वास्तविक बहुपद रैखिक में विभाजित होता है
कारक; यह अवास्तविक है \omterm{def:g11:quad:discriminant}{जड़}ों अंदर आएं \omterm{def:g12:complex:conjugate}{संयुग्मित} \omterm{def:g10:vectors:sum}{जोड़}े (\omterm{def:g12:complex:conjugate}{संयुग्मित}
 \omterm{def:g10:algebra:equation}{समीकरण}), और प्रत्येक \omterm{def:g10:vectors:sum}{जोड़}ी एक वास्तविक द्विघात में गुणा हो जाती है
$ z^2 - 2\operatorname{Re}(z_0)\,z + \abs{z_0}^2$: इसलिए
डिग्री में वास्तविक गुणनखंड-$1$ और डिग्री-$2$ टुकड़े ---
प्रश्न 12 ने इस पर प्रदर्शन किया $ z^4 + 4$.

\textbf{20.} अंक; रोटेशन-और-स्केलिंग; नियमित बहुभुज;
\omterm{def:g10:algebra:expand}{विस्तार} द्वारा त्रिकोणमितीय पहचान; और एक बंद दुनिया
