\section*{अध्याय \ref{ch:g12:contdist} --- सतत यादृच्छिक चर}

\begin{solution}{exo:g12:contdist:1}
$\P(X \leq 5) = \frac{5}{15} = \frac13$;
$\P(4 \leq X \leq 10) = \frac{6}{15} = \frac25$;
$\E(X) = \frac{15}{2} = 7.5$ मिनट;
$\sigma(X) = \frac{15}{\sqrt{12}} = \frac{5\sqrt3}{2} \approx 4.3$ मि.
\end{solution}

\begin{solution}{exo:g12:contdist:2}
$ f \geq 0$ और
$\int_0^2 \frac38 t^2\,\dd t = \frac38\left[\frac{t^3}{3}\right]_0^2
= \frac38 \times \frac83 = 1$: ए घनत्व.
\[
\E(X) = \int_0^2 \frac38 t^3\,\dd t = \frac38 \times \frac{16}{4} = \frac32,
\qquad
\P(X \geq 1) = \int_1^2 \frac38 t^2\,\dd t = \frac{8 - 1}{8} = \frac78 .
\]
\end{solution}

\begin{solution}{exo:g12:contdist:3}
\emph{1.} $\E(X) = \frac{1}{0.2} = 5$ साल;
$\P(X > 5) = \eu^{-0.2\times5} = \eu^{-1} \approx 0.37$.

\emph{2.} स्मृतिहीनता से (\cref{thm:g12:contdist:memoryless}),
$\pcond{X > 3}{X > 8} = \P(X > 5) = \eu^{-1} \approx 0.37$: तीन वर्ष
सेवा से कुछ नहीं बदला.
\end{solution}

\begin{solution}{exo:g12:contdist:4}
$\P(X > m) = \eu^{-\lambda m} = \frac12$ देता है
$ m = \frac{\ln 2}{\lambda} \approx \frac{0.69}{\lambda}$, तुलना में छोटा
$\E(X) = \frac1\lambda $. घातांकीय घनत्व एक लंबी दाहिनी पूँछ है: a
कुछ सामान्य रूप से लंबे जीवनकाल खींचते हैं \emph{\omterm{def:g11:stat:mean}{अर्थ}} के ऊपर
\emph{\omterm{def:g11:stat:median}{MEDIAN}}, वह मान जो आधी आबादी से अधिक है। (यह है
का आधा जीवन \cref{ex:g12:exp:halflife}: आधे परमाणु भी इससे बचे रहते हैं
हालाँकि औसत जीवनकाल लंबा है।)
\end{solution}

\begin{solution}{exo:g12:contdist:5}
$168 = \mu - \sigma $ और $182 = \mu + \sigma $: के बारे में $68\%$.
$189 = \mu + 2\sigma $: इसके ऊपर शेष का आधा भाग है
$100 - 95.4 = 4.6\%$, तो के बारे में $2.3\%$.
$154 = \mu - 3\sigma $: के बारे में $\frac{100 - 99.7}{2} = 0.15\%$.
\end{solution}

\begin{solution}{exo:g12:contdist:6}
$\P(X < 500) \leq 2.5\%$ मतलब $500$ कम से कम दो अवश्य बैठें \omterm{def:g11:stat:variance}{मानक विचलन} के नीचे \omterm{def:g11:stat:mean}{अर्थ} (सामान्य पत्तियां $2.3\% \approx 2.5\%$ नीचे
$\mu - 2\sigma $; पारंपरिक के साथ $1.96$, \omterm{def:g12:complex:argument}{तर्क} समान है):
\[
\mu - 2\sigma \geq 500 \iff \mu \geq 500 + 8 = 508 \text{ g}.
\]
मशीन को सेट करना होगा $508$ जी औसतन --- की कीमत
गारंटी है $8$ प्रति बैग ``मुफ़्त'' उत्पाद का ग्राम।
\end{solution}

\begin{solution}{exo:g12:contdist:7}
\emph{1.} साथ $ p = \frac16$, $ n = 1200$:
$\sqrt{\frac{p(1-p)}{n}} = \sqrt{\frac{5/36}{1200}} \approx 0.0108$, इतना
$95\%$ अस्थिरता \omterm{def:g10:numbers:interval}{अंतराल} है
\[
\frac16 \pm 1.96 \times 0.0108 \approx \intcc{0.146}{0.188}.
\]

\emph{2.} अवलोकन किया गया \omterm{def:g10:stats:series}{आवृत्ति} $\frac{260}{1200} \approx 0.217$ बहुत दूर है
बाहर: निष्पक्षता परिकल्पना को खारिज कर दिया गया है $5\%$ स्तर। द
विचलन के बारे में है $4.6$ \omterm{def:g11:stat:variance}{मानक विचलन} --- सामान्य के लिए
\omterm{def:g10:numbers:approx}{सन्निकटन}, ए \omterm{def:g10:proba:distribution}{संभावना} आदेश की $10^{-6}$, से कहीं अधिक निर्णायक
द $\leq 4.6\%$ बिएनायमे--चेबीशेव से बंधा हुआ
(\cref{exo:g12:sums:7}).
\end{solution}

\begin{solution}{exo:g12:contdist:8}
\emph{1.} अंतर $\frac{1}{\sqrt n} \leq 0.02$ आवश्यक है
$ n \geq 2500$ लोग।

\emph{2.} स्कोरों में अंतर करना $1$ बिंदु अलग, मार्जिन अच्छा होना चाहिए
के अंतर्गत $0.5$ बिंदु, आवश्यकता $ n \geq \left(\frac{1}{0.005}\right)^2 =
40\,000$ सरलीकृत फार्मूले द्वारा --- अधिकांश चुनावों के लिए पहले से ही अव्यावहारिक है।
इससे भी बुरी बात यह है कि केवल सांख्यिकीय मार्जिन ही जिम्मेदार है \emph{\omterm{def:g10:proba:sample}{नमूना}} गलती;
व्यवस्थित पूर्वाग्रह (अप्रतिनिधि) नमूने, गैर-प्रतिक्रिया, अंतिम क्षण
झूले) के रूप में सिकुड़ते नहीं हैं $ n $ बढ़ता है और आम तौर पर एक बिंदु से अधिक हो जाता है। ए
$1$-प्वाइंट रेस वास्तव में कॉल के बहुत करीब है।
\end{solution}

\begin{solution}{exo:g12:contdist:9}
\emph{1.} तब से $\alpha > 0$, $ c \leq \alpha X + \beta \leq d
\iff \frac{c - \beta}{\alpha} \leq X \leq \frac{d - \beta}{\alpha}$, इसलिए
\[
\P(c \leq Y \leq d)
= \int_{(c-\beta)/\alpha}^{(d-\beta)/\alpha} f(t)\,\dd t .
\]
प्रतिस्थापन $ y = \alpha t + \beta $ (\omterm{def:g11:stat:mean}{अर्थ}ात्\ में क्षेत्र को पढ़ना $ y $
चर, $ t = \frac{y - \beta}{\alpha}$, $\dd t = \frac{\dd y}{\alpha}$)
इसे में बदल देता है
$\int_c^d \frac{1}{\alpha} f\left(\frac{y-\beta}{\alpha}\right)\dd y $: द
\omterm{def:g11:func:function}{समारोह} $ g(y) = \frac1\alpha f\left(\frac{y-\beta}{\alpha}\right)$ एक है
घनत्व का $ Y $.

\emph{2.} साथ $ f = \varphi $, $\alpha = \sigma $, $\beta = \mu $:
\[
g(y) = \frac{1}{\sigma}\,\varphi\!\left(\frac{y - \mu}{\sigma}\right)
= \frac{1}{\sigma\sqrt{2\pi}}\,
\exp\left(-\frac{(y-\mu)^2}{2\sigma^2}\right),
\]
जो इसलिए है घनत्व का
$\mathcal N(\mu, \sigma^2) = \mu + \sigma\,\mathcal N(0,1)$.
\end{solution}

\begin{solution}{pb:g12:contdist:1}
\textbf{1.} $\P(\text{wait} \leq 10) = \frac{10}{30} =
\frac13$; $\E = 15$ मिनट;
$\sigma = \frac{30}{\sqrt{12}} \approx 8.7$ मि.

\textbf{2.} $\int_0^1 2x\,\dd x = 1$: ए घनत्व.
$\E(X) = \int_0^1 2x^2\,\dd x = \frac23$;
$\P\left(X > \frac12\right) = \int_{1/2}^1 2x\,\dd x =
1 - \frac14 = \frac34$.

\textbf{3.} $\P(T > 15) = \eu^{-15/10} \approx 0.22$. मंझला:
$\eu^{-m/10} = \frac12$: $ m = 10\ln 2 \approx 6.9$ मिनट ---
से कम \omterm{def:g11:stat:mean}{अर्थ} $10$ क्योंकि घातांक लंबा दाएँ है
पूँछ (दुर्लभ विशाल प्रतीक्षा) खींचती है \omterm{def:g11:stat:mean}{अर्थ} छोड़ते समय ऊपर की ओर
\omterm{def:g11:stat:median}{MEDIAN} ``सामान्य'' प्रतीक्षा में।

\textbf{4.} स्मृतिहीनता से,
$\P(T > 35 \mid T > 20) = \P(T > 15) \approx 0.22$: द
बीस मिनट पहले ही परोसे जा चुके हैं, कुछ भी नहीं खरीदें --- स्ट्रीम करता है
उम्र नहीं.

\textbf{5.} (ए) घातीय (रेडियोधर्मी क्षय मॉडल है
स्मृतिहीनता का आविष्कार किया गया था); (बी) वर्दी पर
$\intcc{0}{8}$ (यादृच्छिक ऑफसेट वाला शेड्यूल); (सी)
घातीय; (डी) न तो: एक घिसा हुआ बल्ब है \emph{अधिक} \omterm{def:g10:proba:distribution}{संभावना} है
एक नए की तुलना में अगले घंटे में असफल हो जाओ, इसलिए
$\P(T > s + t \mid T > s) < \P(T > t)$: उम्र बढ़ना इसके विपरीत है
स्मृतिहीनता प्रमेय.

\textbf{6.} $68\,\%$, $95\,\%$, $99.7\,\%$: तीन
मील का पत्थर \omterm{def:g10:numbers:interval}{अंतराल}.

\textbf{7.} $ z = \frac{190 - 172}{8} = 2.25$: ऊपरी पूँछ
$\approx 1.2\,\%$.

\textbf{8.} $ z = 2$: के बारे में $2.3\,\%$ --- मोटे तौर पर एक व्यक्ति
में $44$.

\textbf{9.} विशिष्टता पर बैठता है $\pm 2\sigma $: के बारे में
$95.4\,\%$ उत्तीर्ण, $4.6\,\%$ अस्वीकृत --- बड़े पैमाने पर विनाशकारी। पर
$\pm 6\sigma $ विफलता दर घटकर लगभग दो भाग प्रति हो जाती है
\emph{अरब}: सिक्स-सिग्मा बड़े पैमाने पर उत्पादन का अधिकार खरीदता है
बिना सामूहिक निरीक्षण के.

\textbf{10.} $\sigma = 5$; साथ निरंतरता सुधार,
$\P \approx \P\left(\abs Z \leq \frac{5.5}{5}\right) =
\P(\abs Z \leq 1.1) \approx 0.73$. प्रमेय सुचारू करता है
गैल्टन बोर्ड ऑफ \cref{pb:g11:binom:1}: खाँचों की सीढ़ियाँ
बन जाता है निरंतर घंटी.

\textbf{11.} $ n \approx \frac{1.96^2 \times 0.25}{0.005^2}
\approx 38\,400$ लोग --- और उस आकार पर
\emph{गैर}-सैंपलिंग त्रुटियां (फ्रेम, इनकार, झूठ) बौनी हो जाती हैं
मार्जिन: एक-बिंदु की दौड़ ईमानदार मतदान से परे है, जो कि है
फिर गंभीर संस्थान क्यों कहते हैं ``कॉल करने के बहुत करीब''।

\textbf{12.} स्मृतिहीनता: जिस क्षण से आप पहुँचते हैं, उसी क्षण से
शेष प्रतीक्षा घातीय है \omterm{def:g11:stat:mean}{अर्थ} $10$ --- पूर्ण
$10$ मिनट, नहीं $5$. समय सारिणी का ``\omterm{def:g11:stat:mean}{अर्थ} अंतर $10$''
गुमराह करता है: तुम्हारा इंतज़ार भी वैसा ही है \omterm{def:g12:randvar:rv}{वितरण} संपूर्ण \omterm{def:g10:numbers:interval}{अंतराल} के रूप में।

\textbf{13.} यादृच्छिक आगमन \omterm{def:g10:numbers:interval}{अंतराल} के अनुपात में भूमि
लंबाई: $\P(\text{long gap}) = \frac{15}{20} = \frac34$.
अपेक्षित प्रतीक्षा: $\frac34 \times \frac{15}{2} + \frac14 \times
\frac52 = 5.625 + 0.625 = 6.25$ मिनट --- भोले से भी अधिक
$5$, हालांकि \omterm{def:g11:stat:mean}{अर्थ} गैप है $10$. अपराधी: \emph{लंबाई-पक्षपाती
नमूनाकरण} --- लंबा \omterm{def:g10:numbers:interval}{अंतराल} अधिक यात्रियों को पकड़ता है।

\textbf{14.} पीछे और आगे की प्रतीक्षा प्रत्येक घातांकीय होती है
के साथ \omterm{def:g11:stat:mean}{अर्थ} $10$ (स्मृतिहीनता दोनों दिशाओं में चलती है
आपका आगमन): आपके औसत वाला अंतर $20$ मिनट
--- \omterm{def:g11:seq:arithmetic}{सामान्य अंतर} से दोगुना. एक वाक्य में निरीक्षण \omterm{thm:g12:limcont:ivt}{विरोधाभास}:
\emph{ \omterm{def:g10:numbers:interval}{अंतराल} आपका निरीक्षण करना कोई सामान्य बात नहीं है
\omterm{def:g10:numbers:interval}{अंतराल}, क्योंकि आपके किसी बड़ी मुसीबत में फंसने की अधिक \omterm{def:g10:proba:distribution}{संभावना} थी
एक.}

\textbf{15.} औसत वर्ग: $\frac{9 \times 10 + 110}{10} =
20$ छात्र. विद्यार्थी-अनुभवी औसत:
$\frac{90 \times 10 + 110 \times 110}{200} = \frac{13\,000}
{200} = 65$ छात्र. ब्रोशर छपता है $20$; पांच में तीन
की कक्षा में छात्र बैठते हैं $110$ और जियो $65$.

\textbf{16.} लोकप्रिय लोग कई मित्र-सूचियों में दिखाई देते हैं, इसलिए
``एक मित्र'' का \omterm{def:g10:proba:sample}{नमूना} लेना मिलनसार व्यक्ति के प्रति पक्षपातपूर्ण है
--- आपके मित्र सामाजिक रूप से लंबे \omterm{def:g10:numbers:interval}{अंतराल} से आते हैं
समय सारिणी.

\textbf{17.} के लिए $ t \geq 0$:
$\P(T > t) = \P(-10\ln U > t) = \P\left(U <
\eu^{-t/10}\right) = \eu^{-t/10}$: बिल्कुल घातांक का
अस्तित्व \omterm{def:g11:func:function}{समारोह} --- एक लोगारित्म कंप्यूटर को परिवर्तित करता है
किसी भी प्रतीक्षा समय में एक समान शोर।

\textbf{18.} प्रत्येक वर्दी में है \omterm{def:g11:stat:mean}{अर्थ} $\frac12$ और \omterm{def:g12:randvar:exp}{झगड़ा}
$\frac{1}{12}$: बारह का योग है \omterm{def:g11:stat:mean}{अर्थ} $6$ और \omterm{def:g12:randvar:exp}{झगड़ा}
$1$ --- तो नुस्खा आउटपुट देता है \omterm{def:g11:stat:mean}{अर्थ} $0$, \omterm{def:g12:randvar:exp}{झगड़ा} $1$. बारह
\omterm{def:g12:sums:indep}{स्वतंत्र} छोटे-छोटे धक्का का सारांश: गैल्टन तंत्र, और द्वारा
डी मोइवर--लाप्लास का आशीर्वाद हिस्टोग्राम पहले से ही है
आँख के लिए घंटी के आकार का.

\textbf{19.} मॉडल के बारे में: बाज़ार बहुतों का योग नहीं हैं
\emph{\omterm{def:g12:sums:indep}{स्वतंत्र}} छोटे-छोटे कारण--घबराहट सहसंबद्ध है
सब कुछ (पिछली समस्या का 2008 का पाठ) और
मोटी पूँछ पैदा करना कोई सामान्य बात नहीं है घनत्व मालिक है. जब डेटा
फुसफुसाते हुए $10^{-137}$, सांख्यिकीविद् की जिरह उत्तर देती है:
\emph{मॉडल को खारिज किया गया है, दिन को नहीं}.

\textbf{20.} वर्दी: सभी पद समान, ईमानदार ``मैं
केवल सीमाएँ जानो''। घातीय: वह ख़तरा जो कभी नहीं
उम्र --- क्षय, आगमन, क्लिक। सामान्य: लोकतांत्रिक
कई छोटी की घंटी \omterm{def:g12:sums:indep}{स्वतंत्र} कारण --- ऊँचाई, त्रुटियाँ,
औसत. स्टिंग: आप क्या \omterm{def:g10:proba:sample}{नमूना} आप कैसे टकराए उससे पक्षपातपूर्ण है
इसमें --- बसें, कक्षाएं, दोस्त। और चाप: बच्चा
जिसने ग्रेड 6 में जूस के डिब्बों की गिनती की, उसके पास दस खंड हैं
समस्याओं के बाद, एक वक्र के साथ भीड़ को मापा; एप्सिलॉन,
द \omterm{def:g12:integ:area}{अभिन्न} और \omterm{pb:g11:quad:1}{केंद्र}ीय सीमा प्रमेय में प्रतीक्षा कर रहे हैं
समझाने के लिए विश्वविद्यालय के संस्करण \emph{क्यों} के लिए घंटी बजती है
सब कुछ.
\end{solution}
