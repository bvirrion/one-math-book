\section*{अध्याय \ref{ch:g12:limcont} --- सीमाएँ और निरंतरता}

\begin{solution}{exo:g12:limcont:1}
द्वारा विभाजित करना $x^3$:
$\dfrac{2x^3 - x + 1}{5x^3 + x^2} = \dfrac{2 - 1/x^2 + 1/x^3}{5 + 1/x}
\to \dfrac{2}{5}$.

$ x^2 + 3x - 1 = x^2(1 + 3/x - 1/x^2) \to +\infty $ जैसा $ x \to -\infty $, तब से
$ x^2 \to +\infty $ और ब्रैकेट की प्रवृत्ति होती है $1$.

जैसा $ x \to 2^+$, अंश की प्रवृत्ति होती है $3 > 0$ और हर को $0^+$,
इसलिए $\dfrac{x+1}{x-2} \to +\infty $.

के लिए $ x > 0$, $-\dfrac1x \leq \dfrac{\cos x}{x} \leq \dfrac1x $, तो सीमा
है $0$ निचोड़ प्रमेय द्वारा.
\end{solution}

\begin{solution}{exo:g12:limcont:2}
\emph{1.} $2x^2 - 3x + 1 = (x-1)(2x-1)$ (द \omterm{def:g11:quad:discriminant}{जड़} $ x=1$ से स्पष्ट है
$2 - 3 + 1 = 0$). के लिए $ x \neq 1$, रद्द करना $ x - 1$: $ f(x) = 2x - 1$.

\emph{2.} इस तरह $\lim_{x\to1} f(x) = 1$. सेटिंग $ f(1) = 1$ का \omterm{def:g10:algebra:expand}{विस्तार} $ f $ को
बहुपद \omterm{def:g11:func:function}{समारोह} $ x \mapsto 2x-1$ पर $\R $, जो है निरंतर पर $1$.
\end{solution}

\begin{solution}{exo:g12:limcont:3}
लिखना $ f(x) = \dfrac{3x - 1}{x + 2} = \dfrac{3(x+2) - 7}{x+2}
= 3 - \dfrac{7}{x+2}$. जैसा $ x \to \pm\infty $, $ f(x) \to 3$: पंक्ति $ y = 3$
एक है \omterm{def:g12:limcont:asymptote}{समस्तरीय अनंतस्पर्शी रेखा} दोनों अनंत पर. जैसा $ x \to (-2)^\pm $,
$ f(x) \to \mp\infty $: पंक्ति $ x = -2$ एक है \omterm{def:g12:limcont:asymptote}{ऊर्ध्वाधर अनंतस्पर्शी}. वक्र है
ए \omterm{def:g11:func:reference}{अतिशयोक्ति}, नीचे $ y=3$ के लिए $ x > -2$ और इसके ऊपर के लिए $ x < -2$.
\end{solution}

\begin{solution}{exo:g12:limcont:4}
संयुग्म द्वारा गुणा करें:
\[
\sqrt{x^2 + x} - x = \frac{x^2 + x - x^2}{\sqrt{x^2+x} + x}
= \frac{x}{x\left(\sqrt{1 + 1/x} + 1\right)}
= \frac{1}{\sqrt{1 + 1/x} + 1}
\xrightarrow[x\to+\infty]{} \frac{1}{2}.
\]
इसी प्रकार,
\[
\frac{\sqrt{1+x} - 1}{x} = \frac{(1+x) - 1}{x\left(\sqrt{1+x} + 1\right)}
= \frac{1}{\sqrt{1+x} + 1} \xrightarrow[x\to0]{} \frac{1}{2}.
\]
\end{solution}

\begin{solution}{exo:g12:limcont:5}
\emph{1.} तब से $\sin x \geq -1$, $ f(x) \geq x - 1 \to +\infty $, और हम
तुलना करके निष्कर्ष निकालें। अभी तक $ f'(x) = 1 + \cos x $ हर \omterm{def:g11:func:parity}{विषम} परिस्थिति में गायब हो जाता है
का एकाधिक $\pi $, और हर पर \omterm{def:g10:numbers:interval}{अंतराल} $\intco{A}{+\infty}$  \omterm{def:g11:func:function}{समारोह}
के बीच वैकल्पिक \omterm{def:g10:numbers:interval}{अंतराल} सख्त वृद्धि का; यह है \omterm{def:g12:seq:monotonic}{की बढ़ती} लेकिन नहीं
सख्ती से, और अधिक महत्वपूर्ण बात यह है कि इसकी वृद्धि की आवश्यकता नहीं है: तुलना ही पर्याप्त है।

\emph{2.} के लिए $ x > 1$, से विभाजित करें $ x $:
\[
g(x) = \frac{1 + \frac{\sin x}{x}}{1 - \frac{\sin x}{x}},
\]
और $\frac{\sin x}{x} \to 0$ निचोड़ प्रमेय द्वारा
($\abs{\sin x} \leq 1$). इस तरह $ g(x) \to \frac{1+0}{1-0} = 1$.
\end{solution}

\begin{solution}{exo:g12:limcont:6}
$ f(x) = x^3 - 3x + 1$ है निरंतर, साथ $ f'(x) = 3x^2 - 3 = 3(x-1)(x+1)$:
$ f $ पर बढ़ता है $\intoc{-\infty}{-1}$, पर घटता है $\intcc{-1}{1}$,
पर बढ़ता है $\intco{1}{+\infty}$, स्थानीय के साथ \omterm{def:g10:functions:extrema}{अधिकतम} $ f(-1) = 3 > 0$ और
स्थानीय \omterm{def:g10:functions:extrema}{न्यूनतम} $ f(1) = -1 < 0$, $\lim_{-\infty} f = -\infty $,
$\lim_{+\infty} f = +\infty $.

तीनों में से प्रत्येक पर एकरसता है \omterm{def:g10:numbers:interval}{अंतराल}, $ f $ है निरंतर, कठोरता से
एकरस, और $0$ समापन बिंदु मानों/सीमाओं के बीच सख्ती से स्थित है, इसलिए
आक्षेप प्रमेय बिल्कुल एक शून्य प्रति देता है \omterm{def:g10:numbers:interval}{अंतराल}: कुल तीन शून्य.
साइन चेक: $ f(-2) = -1 < 0 < f(-1) = 3$, तो एक शून्य में
$\intoo{-2}{-1}$; $ f(0) = 1 > 0 > f(1) = -1$, एक शून्य में $\intoo{0}{1}$;
$ f(1) = -1 < 0 < f(2) = 3$, एक शून्य में $\intoo{1}{2}$.
\end{solution}

\end{solution}

\begin{solution}{exo:g12:limcont:9}
\emph{1.} $ f'(x) = 5x^4 + 3x^2 + 1 > 0$, इसलिए $ f $ सख्ती से है \omterm{def:g12:seq:monotonic}{की बढ़ती} पर
$\R $; साथ $\lim_{\pm\infty} f = \pm\infty $ और निरंतरता, आक्षेप
प्रमेय एक अद्वितीय वास्तविक शून्य देता है $\alpha $. तब से $ f(0) = -1 < 0$ और
$ f(1) = 2 > 0$, $\alpha \in \intoo{0}{1}$.

\emph{2.} एक बनाए रखें \omterm{def:g10:numbers:interval}{अंतराल} $\intcc{a_n}{b_n}$ साथ
$ f(a_n) < 0 < f(b_n)$, से शुरू $\intcc{0}{1}$; प्रत्येक चरण पर मूल्यांकन करें
$ f $ पर \omterm{prop:g10:coordgeom:midpoint}{मध्य} $ m $ और रखें अर्\omterm{def:g10:numbers:interval}{ध-अंतराल} जिस पर चिन्ह बदल जाता है
कायम है. बाद $ n $ कदम \omterm{def:g10:numbers:interval}{अंतराल} लंबाई है $2^{-n}$, इतना \omterm{prop:g10:coordgeom:midpoint}{मध्य}
अनुमानित $\alpha $ भीतर तक $2^{-(n+1)}$. ज़रुरत है
$2^{-(n+1)} \leq 10^{-6}$, यानी\ $ n + 1 \geq \frac{6}{\log_{10} 2} \approx 19.93$:
$ n = 20$ पुनरावृत्तियाँ पर्याप्त हैं।
\end{solution}

\begin{solution}{pb:g12:limcont:1}
\textbf{1.} प्रमुख शब्द: $\frac{3x^2}{x^2} \to 3$;
$\frac{2x}{x^2} = \frac2x \to 0$;
$ x^3 - x^2 = x^3\left(1 - \frac1x\right) \to +\infty $.

\textbf{2.} $\lim_{x \to \pm\infty} f = 3$: \omterm{def:g12:limcont:asymptote}{समस्तरीय अनंतस्पर्शी रेखा} $ y = 3$; $\lim_{x \to (-2)^\pm} f = \mp\infty $:
\omterm{def:g12:limcont:asymptote}{ऊर्ध्वाधर अनंतस्पर्शी} $ x = -2$.

\textbf{3.} $\frac{(x-3)(x+3)}{x - 3} = x + 3 \to 6$. और
$\frac{\sqrt{x+1} - 1}{x} = \frac{x}{x(\sqrt{x+1} + 1)} =
\frac{1}{\sqrt{x+1} + 1} \to \frac12$.

\textbf{4.} $\abs{x \sin\frac1x} \leq \abs x \to 0$, तो
सीमा है $0$. और
$\abs{\frac{\sin x}{x}} \leq \frac1x \to 0$: सीमा $0$.

\textbf{5.} $\frac{\sqrt{4x^2 + x}}{x} =
\sqrt{4 + \frac1x} \to \sqrt4 = 2$ (निरंतरता चौक का
\omterm{def:g11:quad:discriminant}{जड़}, रचना)।

\textbf{6.} $ f(x) = x^3 + x $ है निरंतर साथ
$ f(1) = 2 < 5 < 10 = f(2)$: आईवीटी द्वारा, $ f(c) = 5$ कुछ के लिए
$ c \in \intoo{1}{2}$.

\textbf{7.} $ f'(x) = 3x^2 + 1 > 0$: कठोरता से \omterm{def:g12:seq:monotonic}{की बढ़ती} पर
$\R $, निरंतर, सीमा के साथ $\mp\infty $:आक्षेप से
प्रमेय, प्रत्येक मान --- विशेष रूप से $5$ ---प्राप्त है
बिल्कुल एक बार.

\textbf{8.} $ p(x) = x^3\left(1 + \frac ax + \frac{b}{x^2} +
\frac{c}{x^3}\right)$: जैसा $ x \to +\infty $, $ p \to +\infty $; जैसा
$ x \to -\infty $, $ p \to -\infty $. तो कुछ विशाल पर \omterm{def:g10:numbers:interval}{अंतराल}
$\intcc{-M}{M}$, $ p(-M) < 0 < p(M)$: आईवीटी डिलीवर करता है \omterm{def:g11:quad:discriminant}{जड़}.
वही प्रमुख-शब्द तर्क किसी भी विषमता के लिए शब्दशः काम करता है
डिग्री---विषम शक्तियां दोनों पर विपरीत चिन्ह रखती हैं
अनंत

\textbf{9.} संयोग $ n $ टाइम्स एक छोड़ देता है \omterm{def:g10:numbers:interval}{अंतराल} लम्बाई का
$2^{-n}$; $2^{-20} \approx 9.5 \times 10^{-7} < 10^{-6}$:
बीस कदम. न्यूटन प्रत्येक चरण में सही अंकों को दोगुना करता है: से
एक सही अंक, के बारे में $3$--$4$ चरण छह तक पहुँचते हैं ---
कछुआ और जड़-खोजने वाला ग्रेहाउंड।

\textbf{10.} $ g $ क्या नहीं है निरंतर पर $\intcc{-1}{1}$: यह
पर विस्फोट होता है $0$, के जो \omterm{def:g10:numbers:interval}{अंतराल} रोकना। आईवीटी
निष्कर्ष अपनी परिकल्पना के साथ लुप्त हो जाता है --- हमेशा जाँच करें
निरंतरता \emph{कुल मिलाकर \omterm{def:g10:numbers:interval}{अंतराल}}, सिर्फ पर नहीं
समापनबिंदु.

\textbf{11.} $ g(x) = f(x) - x $ है निरंतर;
$ g(0) = f(0) \geq 0$ (मूल्य निहित हैं $\intcc{0}{1}$) और
$ g(1) = f(1) - 1 \leq 0$. आईवीटी द्वारा, $ g(c) = 0$ कुछ के लिए $ c $:
$ f(c) = c $.

\textbf{12.} देश को एक खंड के रूप में \omterm{def:g11:scal:dot}{आदर्श} बनाना
$\intcc{0}{1}$: $ f $ प्रत्येक स्थान को दिखाए गए स्थान पर भेजता है
टूटे-फूटे नक्शे पर सीधे इसके नीचे; क्रम्प्लिंग रखता है
देश भर में नक्शा, तो $ f $ में उतरता है $\intcc{0}{1}$, और यह
है निरंतर (कागज नहीं फटता). \omterm{pb:g10:functions:1}{नियत बिन्दु} $ c $ है
वह स्थान अपने ठीक ऊपर स्थित है \omterm{def:g10:functions:function}{छवि} --- ब्रौवर का
प्रमेय ईमानदार द्वि-आयामी मानचित्र को संभालता है।
