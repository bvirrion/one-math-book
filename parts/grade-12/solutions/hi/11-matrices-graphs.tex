\section*{अध्याय \ref{ch:g12:matrix} --- मैट्रिक्स और ग्राफ़}

\begin{solution}{exo:g12:matrix:1}
\[
A + B = \begin{pmatrix} 3 & 2\\ 1 & 2\end{pmatrix}, \quad
AB = \begin{pmatrix} 4 & 2\\ 1 & 1\end{pmatrix}, \quad
BA = \begin{pmatrix} 2 & 4\\ 1 & 3\end{pmatrix}, \quad
A^2 = \begin{pmatrix} 1 & 4\\ 0 & 1\end{pmatrix}.
\]
टिप्पणी $AB \neq BA $.
\end{solution}

\begin{solution}{exo:g12:matrix:2}
$\det A = 6 - 5 = 1 \neq 0$:
$ A^{-1} = \begin{pmatrix} 3 & -5\\ -1 & 2 \end{pmatrix}$.
$\det B = 12 - 12 = 0$: $ B $ क्या नहीं है \omterm{def:g12:matrix:inverse}{उलटी}.
\end{solution}

\begin{solution}{exo:g12:matrix:3}
सिस्टम है $ AX = Y $ साथ $ A $ के रूप में \cref{exo:g12:matrix:2} और
$ Y = \begin{pmatrix} 1\\ 2\end{pmatrix}$:
\[
X = A^{-1}Y = \begin{pmatrix} 3 & -5\\ -1 & 2\end{pmatrix}
\begin{pmatrix} 1\\ 2\end{pmatrix}
= \begin{pmatrix} -7\\ 3\end{pmatrix}:
\qquad x = -7,\ y = 3 .
\]
\end{solution}

\begin{solution}{exo:g12:matrix:4}
\emph{1.} $ N^2 = \begin{pmatrix} 0&1\\0&0\end{pmatrix}
\begin{pmatrix} 0&1\\0&0\end{pmatrix} = 0$, और $ I $ प्रत्येक के साथ आवागमन करता है
मैट्रिक्स.

\emph{2.} चूँकि दो पद लघुकरण करते हैं, द्विपद प्रमेय लागू होता है और सभी
युक्त शब्द $ N^2$ गायब होना:
\[
A^k = (2I + N)^k = 2^k I + k\,2^{k-1} N
= \begin{pmatrix} 2^k & k\,2^{k-1}\\ 0 & 2^k \end{pmatrix}.
\]
की जाँच करें $ k = 2$:
$ A^2 = \begin{pmatrix} 2&1\\0&2\end{pmatrix}^2
= \begin{pmatrix} 4&4\\0&4\end{pmatrix}$, और सूत्र देता है
$2^2 = 4$, $2 \times 2 = 4$. \checkmark
\end{solution}

\begin{solution}{exo:g12:matrix:5}
\emph{प्रेरण।} के लिए $ n = 1$:
$ A^1 = \begin{pmatrix} 0&1\\1&1\end{pmatrix}
= \begin{pmatrix} F_0 & F_1\\ F_1 & F_2\end{pmatrix}$. सूत्र मान लीजिए
के लिए $ n $; तब
\[
A^{n+1} = A^n A
= \begin{pmatrix} F_{n-1} & F_n\\ F_n & F_{n+1}\end{pmatrix}
\begin{pmatrix} 0 & 1\\ 1 & 1\end{pmatrix}
= \begin{pmatrix} F_n & F_{n-1} + F_n\\ F_{n+1} & F_n + F_{n+1}\end{pmatrix}
= \begin{pmatrix} F_n & F_{n+1}\\ F_{n+1} & F_{n+2}\end{pmatrix}.
\]
\emph{पहचान।} के लिए $2\times2$ मैट्रिक्स, का \omterm{def:g10:algebra:expand}{विस्तार} दिखाता है
$\det(MN) = \det M \det N $; इस तरह
$\det(A^n) = (\det A)^n = (-1)^n $, और
$\det A^n = F_{n-1}F_{n+1} - F_n^2$. (यह है \emph{कैसिनी की पहचान}.)
\end{solution}

\begin{solution}{exo:g12:matrix:6}
\emph{1.} शीर्षों का आदेश देना $1, 2, 3$:
\[
M = \begin{pmatrix} 0&1&1\\ 0&0&1\\ 1&0&0\end{pmatrix}, \quad
M^2 = \begin{pmatrix} 1&0&1\\ 1&0&0\\ 0&1&1\end{pmatrix}, \quad
M^3 = \begin{pmatrix} 1&1&1\\ 1&0&1\\ 1&0&1 \end{pmatrix}.
\]

\emph{2.} $\bigl(M^3\bigr)_{11} = 1$: लंबाई की बिल्कुल एक बंद चाल $3$
शिखर पर $1$, \omterm{def:g11:stat:mean}{अर्थ}ात् $1 \to 2 \to 3 \to 1$. (टहलना
$1 \to 3 \to 1$ लंबाई है $2$ केवल, और $1 \to 3$ तब $3\to1$ तब
$1\to3$ पर समाप्त होता है $3$.)
\end{solution}

\begin{solution}{exo:g12:matrix:7}
\emph{1.} $ a_{n+1} = 0.8a_n + 0.3b_n $, $ b_{n+1} = 0.2a_n + 0.7b_n $:
$ M = \begin{pmatrix} 0.8 & 0.3\\ 0.2 & 0.7\end{pmatrix}$.

\emph{2.} $ MX = X $ देता है $0.8a + 0.3b = a $, \emph{यानी} $0.3b = 0.2a $, इसलिए
$ b = \frac23 a $; साथ $ a + b = 1$: $ a = 0.6$, $ b = 0.4$.

\emph{3.} का उपयोग करते हुए $ b_n = 1 - a_n $:
$ a_{n+1} = 0.8a_n + 0.3(1 - a_n) = 0.5a_n + 0.3$, इसलिए
\[
c_{n+1} = a_{n+1} - 0.6 = 0.5a_n + 0.3 - 0.6 = 0.5(a_n - 0.6) = 0.5\,c_n .
\]
इस तरह $ c_n = 0.5^n c_0 \to 0$: $ a_n \to 0.6$ और $ b_n \to 0.4$, जो भी हो
आरंभिक \omterm{def:g11:prob:rv}{वितरण}.
\end{solution}

\begin{solution}{exo:g12:matrix:8}
\emph{1.} $\det P = -2$, इसलिए
$ P^{-1} = -\frac12\begin{pmatrix} -1 & -1\\ -1 & 1\end{pmatrix}
= \frac12\begin{pmatrix} 1 & 1\\ 1 & -1\end{pmatrix}$. तब
\[
AP = \begin{pmatrix} 4 & 2\\ 4 & -2 \end{pmatrix}, \qquad
D = P^{-1}AP = \frac12\begin{pmatrix} 1&1\\1&-1\end{pmatrix}
\begin{pmatrix} 4&2\\4&-2\end{pmatrix}
= \begin{pmatrix} 4 & 0\\ 0 & 2\end{pmatrix}.
\]

\emph{2.} से $ A = PDP^{-1}$, एक तत्काल प्रेरण देता है
$ A^n = PD^nP^{-1}$ साथ
$ D^n = \begin{pmatrix} 4^n & 0\\ 0 & 2^n\end{pmatrix}$, इसलिए
\[
A^n = P D^n P^{-1}
= \begin{pmatrix} 4^n & 2^n\\ 4^n & -2^n\end{pmatrix}\cdot
\frac12\begin{pmatrix} 1&1\\1&-1\end{pmatrix}
= \frac12\begin{pmatrix} 4^n + 2^n & 4^n - 2^n\\
4^n - 2^n & 4^n + 2^n\end{pmatrix}.
\]
को लागू करने $ A^n $ को $ X_0 = \begin{pmatrix} u_0\\v_0\end{pmatrix}$ पुनरुत्पादन करता है
बिल्कुल के सूत्र \cref{ex:g12:matrix:coupled}.
\end{solution}

\begin{solution}{pb:g12:matrix:1}
\textbf{1.} $ AB = \begin{pmatrix} 2 & 1\\ 4 & 3\end{pmatrix}$
और $ BA = \begin{pmatrix} 3 & 4\\ 1 & 2\end{pmatrix}$: मैट्रिक्स
गुणन क्रमविनिमेय नहीं है --- $ B $ पर कॉलम स्वैप करता है
दाईं ओर, बाईं ओर पंक्तियाँ।

\textbf{2.} \omterm{def:g11:vect:det}{सिद्ध} $6 - 5 = 1$: श्लोक में
$\begin{pmatrix} 3 & -1\\ -5 & 2\end{pmatrix}$. इसे लागू करना
$\binom{4}{7}$: $ x = 12 - 7 = 5$, $ y = -20 + 14 = -6$.

\textbf{3.} $ N^2 = 0$. तब $(I + N)^n = I + nN $ प्रेरण द्वारा:
$(I + nN)(I + N) = I + (n+1)N + nN^2 = I + (n+1)N $.

\textbf{4.} $ A = \begin{pmatrix} 0&1&1\\ 1&0&1\\ 1&1&0
\end{pmatrix}$, और $ A^3$ विकर्ण प्रविष्टियाँ हैं $2$: प्रत्येक से
शीर्ष, लंबाई के ठीक दो बंद रास्ते $3$ (त्रिकोण
दक्षिणावर्त या वामावर्त चला) --- गिनती प्रमेय
कार्रवाई में.

\textbf{5.} $ D^n = \begin{pmatrix} 2^n & 0\\ 0 & 2^{-n}
\end{pmatrix}$: एक दिशा विस्फोट करती है, दूसरी मर जाती है ---
विकर्ण भाग्य स्वतंत्र होते हैं ज्यामितीय \omterm{def:g12:seq:sequence}{अनुक्रम}.

\textbf{6.} $ F^2 = \begin{pmatrix} 2 & 1\\ 1 & 1\end{pmatrix}$,
$ F^3 = \begin{pmatrix} 3 & 2\\ 2 & 1\end{pmatrix}$,
$ F^4 = \begin{pmatrix} 5 & 3\\ 3 & 2\end{pmatrix}$: फाइबोनैचि
हर जगह; अनुमान जैसा कहा गया है।

\textbf{7.} अगर $ F^n = \begin{pmatrix} F_{n+1} & F_n\\ F_n &
F_{n-1}\end{pmatrix}$, तब
\[
F^{n+1} = F^n F =
\begin{pmatrix} F_{n+1} + F_n & F_{n+1}\\
F_n + F_{n-1} & F_n \end{pmatrix}
= \begin{pmatrix} F_{n+2} & F_{n+1}\\ F_{n+1} & F_n
\end{pmatrix} :
\]
आनुवंशिकता; आधार मामला $ n = 1$ है $ F $ स्वयं, का उपयोग करते हुए
सम्मलेन $ F_0 = 0$ (जो पुनरावृत्ति को पीछे की ओर बढ़ाता है)।

\textbf{8.} $\det F = -1$, इसलिए
$\det(F^n) = (\det F)^n = (-1)^n $; और सीधे
$\det(F^n) = F_{n+1}F_{n-1} - F_n^2$: कैसिनी, एक पंक्ति। (द
के लिए उत्पाद नियम $2 \times 2$ निर्धारकों एक सुखद है
पांच मिनट का \omterm{def:g10:algebra:expand}{विस्तार}।)

\textbf{9.} के शीर्ष-दाईं ओर $ F^m F^n $:
$ F_{m+1}F_n + F_m F_{n-1}$; के शीर्ष-दाईं ओर $ F^{m+n}$:
$ F_{m+n}$. के लिए $ m = n = 3$:
$ F_4 F_3 + F_3 F_2 = 3 \times 2 + 2 \times 1 = 8 = F_6$.

\textbf{10.} के लिए $ k = 1$: मामूली। अगर $ F_n \mid F_{kn}$, द
इसके अलावा सूत्र के साथ $ m = kn $:
$ F_{(k+1)n} = F_{kn+1}F_n + F_{kn}F_{n-1}$: दोनों पद हैं
के गुणज $ F_n $. इसलिए $ F_n \mid F_{kn}$ सभी के लिए $ k $: जाँच करना
$ F_3 = 2$ विभाजित $ F_6 = 8$ और $ F_9 = 34$.

\textbf{11.} $ F^{100} = F^{64} F^{32} F^4$: सात वर्ग
($ F^2, F^4, \dots, F^{64}$) प्लस दो संयोजन --- नौ
निन्यानवे के बजाय गुणा। यह है
मिस्र के शास्त्रियों की दोहरी-तालिका युक्ति, से उठाई गई
आव्यूहों को संख्याएँ: लिखें $100$ बाइनरी में, गुणा करें
दोहरीकरण की आपको आवश्यकता है।

\textbf{12.} $0.8 + 0.2 = 1$ और $0.4 + 0.6 = 1$: कल
होना ही चाहिए \emph{कुछ} मौसम --- प्रत्येक कॉलम पूर्ण है
\omterm{def:g10:proba:distribution}{संभावना} \omterm{def:g11:prob:rv}{वितरण}, इसलिए संभावनाएँ संरक्षित रहती हैं।

\textbf{13.} कल: $\binom{0.8}{0.2}$. परसों:
$ M\binom{0.8}{0.2} = \binom{0.72}{0.28}$.

\textbf{14.} $ Mv = v $ साथ $ v = \binom{s}{r}$, $ s + r = 1$:
$0.8s + 0.4r = s $ देता है $0.4r = 0.2s $, $ s = 2r $:
$ v = \binom{2/3}{1/3}$. लंबे समय में, तीन में दो दिन होते हैं
सनी --- आज जो भी दिखता है।

\textbf{15.} से $\binom01$: धूप वाले घटक $0.4$, $0.56$,
$0.624$, $0.6496$; के लिए \omterm{def:g10:numbers:interval}{अंतराल} $\frac23$: $0.267$, $0.107$,
$0.043$, $0.017$ --- प्रत्येक चरण अंतर को बिल्कुल गुणा कर देता है
$0.4$ (मशीन का दूसरा eigenvalue): ज्यामितीय
स्थिर अवस्था में \omterm{def:g12:seq:limit}{अभिसरण}.

\textbf{16.} कॉलम (से $ A $, $ B $, $ C $):
$ P = \begin{pmatrix} 0 & 0 & 1\\ \frac12 & 0 & 0\\
\frac12 & 1 & 0\end{pmatrix}$. स्थिर अवस्था: $ v_A = v_C $,
$ v_B = \frac{v_A}{2}$, $ v_C = \frac{v_A}{2} + v_B $; संक्षेप में
$1$: $ v = \left(\frac25, \frac15, \frac25\right)$. रैंकिंग:
$ A $ और $ C $ पहले के लिए टाई, $ B $ अंतिम।

\textbf{17.} $ C $ प्राप्त करता है \emph{सभी} का $ B $का ट्रैफिक और आधा
का $ A $है, और यह हर चीज़ को वापस फ़नल कर देता है $ A $: स्थिर
उपाय बताएं कि सर्फ़र कहां है \emph{समय बिताता है}, कैसे नहीं
कई लिंक इंगित करते हैं --- एक लोकप्रिय पेज का एक लिंक भारी पड़ता है
वीरान लोगों में से कई। वह पुनरावर्ती भार है
बिल्कुल Google का संस्थापक विचार; भिगोना मकड़ी के जाल को संभालता है
और मृत अंत.

\textbf{18.} $ A^n = P D^n P^{-1}$ देता है
$ u_n = \frac{4^n + 2^n}{2}$ (और
$ v_n = \frac{4^n - 2^n}{2}$). जाँच करना: $ u_1 = 3$, $ u_2 = 10$,
$ u_3 = 36$; सीधे: $(1,0) \to (3,1) \to (10,6) \to
(36, 28)$: मेल मिलाना।

\textbf{19.} विकर्णीकरण में \omterm{def:g12:comb:permutation}{परिवर्तन} होता है \omterm{def:g10:coordgeom:system}{COORDINATES} जिसमें
युग्मित प्रणाली स्वतंत्र रूप से टूट जाती है ज्यामितीय
क्रम --- प्रत्येक eigenvalue अपनी दौड़ चलाता है। द मार्कोव
स्थिर अवस्था eigenvalue का eigenvector है $1$, और
प्रश्न 15 की \omterm{def:g12:seq:limit}{अभिसरण} दर अगला eigenvalue है:
मौसम मशीन हमेशा से एक \omterm{def:g11:scal:dot}{आदर्श} कहानी थी।

\textbf{20.} बहीखाता पद्धति: एक प्रणाली एक है मैट्रिक्स \omterm{def:g10:algebra:equation}{समीकरण},
एक व्युत्क्रम द्वारा हल किया गया। कॉम्बिनेटरिक्स: की शक्तियां \omterm{def:g12:matrix:graph}{सहखंडज मैट्रिक्स} वॉक, लिंक, कनेक्शन गिनें। विकास: की शक्तियाँ
मशीन राज्यों को उनकी नियति तक ले जाती है, और
ईजेन-दिशाएँ (फाइबोनैचि की सुनहरी दिशा, मौसम की
स्थिर स्थिति, वेब की रैंकिंग \omterm{def:g11:vect:vector}{वेक्टर}) नियति हैं.
विश्वविद्यालय के संस्करणों में रैखिक बीजगणित, का विज्ञान है
बिलकुल यही.
\end{solution}
