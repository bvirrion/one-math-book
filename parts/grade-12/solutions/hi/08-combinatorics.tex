\begin{solution}{exo:g12:comb:7}
\emph{बीजगणितीय रूप से:}
\[
k\binom nk = \frac{k\,n!}{k!(n-k)!} = \frac{n!}{(k-1)!\,(n-k)!}
= n\,\frac{(n-1)!}{(k-1)!\bigl((n-1)-(k-1)\bigr)!} = n\binom{n-1}{k-1}.
\]
\emph{दोहरी गिनती से:} \omterm{def:g10:vectors:sum}{जोड़}े गिनें (की समिति) $ k $, इसमें अध्यक्ष)।
या तो समिति चुनें ($\binom nk $) तो इसके अध्यक्ष ($ k $):
$ k\binom nk $ \omterm{def:g10:vectors:sum}{जोड़}े. या पहले राष्ट्रपति चुनें ($ n $ विकल्प) फिर
अन्य $ k-1$ शेष के बीच सदस्य $ n-1$: $ n\binom{n-1}{k-1}$ \omterm{def:g10:vectors:sum}{जोड़}े.
\end{solution}

\begin{solution}{exo:g12:comb:8}
एक पथ बिल्कुल से मिलकर बनता है $ m + n $ चरण, जिनमें से $ m $ पूर्व और हैं $ n $ हैं
उत्तर; यह पूरी तरह से उदाहरणों के सेट (इनमें से) द्वारा निर्धारित होता है $ m+n $) पर
जो पूर्व की ओर कदम रखता है. वहाँ हैं $\binom{m+n}{m}$ ऐसे विकल्प.
\end{solution}

\begin{solution}{exo:g12:comb:9}
का एक सेट विभाजित करें $ m + n $ लोगों को एक समूह में $ A $ का $ m $ और एक समूह $ B $ का
$ n $. ए $ k $-तत्व उपसमुच्चय में कुछ संख्या होती है $ j $ के सदस्यों का $ A $
($0 \leq j \leq k $) और $ k - j $ के सदस्य $ B $; निश्चित के लिए $ j $ वहाँ हैं
$\binom mj \binom{n}{k-j}$ ऐसे उपसमुच्चय, और अतिरिक्त \omterm{def:g11:vect:det}{सिद्ध}ांत $ j $
वेंडरमोंडे की पहचान देता है।

साथ $ m = n = k $:
\[
\binom{2n}{n} = \sum_{j=0}^n \binom nj \binom{n}{n-j}
= \sum_{j=0}^n \binom nj^{2},
\]
समरूपता का उपयोग करना $\binom{n}{n-j} = \binom nj $.
\end{solution}

\begin{solution}{exo:g12:comb:10}
\emph{के जरिए \cref{exo:g12:comb:7}:}
\[
\sum_{k=1}^n k\binom nk = \sum_{k=1}^n n\binom{n-1}{k-1}
= n\sum_{j=0}^{n-1}\binom{n-1}{j} = n\,2^{n-1},
\]
द्वारा \cref{cor:g12:comb:sums}. \emph{विभेदीकरण के माध्यम से:} फर्क
$(1+x)^n = \sum_k \binom nk x^k $ देता है
$ n(1+x)^{n-1} = \sum_k k \binom nk x^{k-1}$; पर मूल्यांकन करें $ x = 1$.
\end{solution}

\begin{solution}{pb:g12:comb:1}
\textbf{1.} $26^2 \times 10^3 = 676\,000$ प्लेटें. केला: $6$
A वाले अक्षर तिगुने और N वाले अक्षर दुगुने हो गए:
$\frac{6!}{3!\,2!} = 60$ विपर्यय.

\textbf{2.} $\binom{32}{5} = 201\,376$ हाथ;
$\binom42 \binom{28}{3} = 6 \times 3\,276 = 19\,656$ के साथ
बिल्कुल दो इक्के.

\textbf{3.} पथ एक शब्द है $4$ आर और $3$ यू: चुनें
यू स्थिति: $\binom73 = 35$.

\textbf{4.} $(1+x)^4 = 1 + 4x + 6x^2 + 4x^3 + x^4$. पर
$ x = 1$: $\sum_k \binom nk = 2^n $; पर $ x = -1$:
$\sum_k (-1)^k \binom nk = 0$ --- पंक्ति योग और वैकल्पिक पंक्ति
का योग \omterm{prop:g11:binom:pascal}{पास्कल का त्रिकोण}.

\textbf{5.} की समितियाँ $ k $ कुर्सी वाले लोग, से $ n $:
समिति चुनें फिर उसका अध्यक्ष ($\binom nk \times k $), या
अध्यक्ष फिर अन्य सदस्य
($ n \times \binom{n-1}{k-1}$): बराबर। संक्षेप में कहना $ k $: द
दाईं ओर का योग है $ n \sum_j \binom{n-1}{j} = n\,2^{n-1}$.

\textbf{6.} की एक पंक्ति $10$ सितारे और $3$ बार्स ऑर्डर को एनकोड करता है
(पहली बार से पहले स्वाद के स्कूप 1, आदि); पंक्ति है
$13$ प्रतीकों और बार पदों द्वारा निर्धारित किया जाता है:
$\binom{13}{3} = 286$ आदेश.

\textbf{7.} $12$ तारे, $2$ बार: $\binom{14}{2} = 91$.

\textbf{8.} साथ $ x', y', z' \geq 0$ और
$ x' + y' + z' = 9$: $\binom{11}{2} = 55$.

\textbf{9.} एकपदी $ a^i b^j c^k $ साथ $ i + j + k = 5$:
$\binom72 = 21$.

\textbf{10.} सूत्र: $3$ तारे, $1$ छड़: $\binom41 = 4$;
सूची: $(3,0)$, $(2,1)$, $(1,2)$, $(0,3)$: समझौता।

\textbf{11.} आदेश घोषित होने पर ``समान'' दर्ज किया गया
के अलावा और कुछ नहीं होना \emph{गिनता} प्रति स्वाद --- तारे
कोई नाम न रखें. यदि स्कूप क्रम से खाए जाएं, तो प्रत्येक
$10$ अलग-अलग स्थितियाँ स्वतंत्र रूप से स्वाद चुनती हैं:
$4^{10} = 1\,048\,576$ दृश्यों --- एक अलग मॉडल और एक
अलग दुनिया (\cref{met:g12:comb:model}: हमेशा पूछिये
\emph{आदेश दिया? विशिष्ट? पुनरावृत्ति की अनुमति है?}).

\textbf{12.} $ D_1 = 0$; $ D_2 = 1$ (बदलना); $ D_3 = 2$ (दो
$3$-चक्र); $ D_4 = 9$.

\textbf{13.} अतिथि 1 को टोपी मिलती है $ k \neq 1$: $ n - 1$ विकल्प. यदि
अतिथि $ k $ टोपी 1 मिलती है, शेष $ n - 2$ मेहमान विक्षिप्त
उनकी अपनी टोपियाँ: $ D_{n-2}$ तौर तरीकों। यदि अतिथि $ k $ करता है \emph{नहीं}
हैट 1 प्राप्त करें, हैट 1 को अतिथि के रूप में पुनः लेबल करें $ k $की वर्जित टोपी:
$ n - 1$ शेष मेहमान विक्षिप्त: $ D_{n-1}$ तौर तरीकों। इस तरह
$ D_n = (n-1)(D_{n-1} + D_{n-2})$. जाँच करना:
$ D_4 = 3(2 + 1) = 9$; और $ D_5 = 4(9 + 2) = 44$.

\textbf{14.} की $3! = 6$ असाइनमेंट, उन्हें घटाएं
कम से कम एक टोपी ठीक करना: किसी दी गई टोपी को तीन ठीक करना ($2!$ प्रत्येक,
$3 \times 2 = 6$), \omterm{def:g10:vectors:sum}{जोड़}ियों की अधिक गिनती ($3$ \omterm{def:g10:vectors:sum}{जोड़}े, $1!$
प्रत्येक) जिसे वापस लौटना होगा, और पहचान को फिर से घटाना होगा
($1$): $ D_3 = 6 - 6 + 3 - 1 = 2$, \omterm{def:g11:stat:mean}{अर्थ}ात\
$3!\left(1 - 1 + \frac12 - \frac16\right) = 2$. सामान्य तौर पर
$ D_n = n!\sum_{k=0}^{n} \frac{(-1)^k}{k!}$.

\textbf{18.} $ n = 1$: $1 = 1$. अगर
$1^3 + \dots + n^3 = \left(\frac{n(n+1)}{2}\right)^2$, जोड़ना
$(n+1)^3$:
\[
\frac{n^2(n+1)^2}{4} + (n+1)^3
= \frac{(n+1)^2\left(n^2 + 4n + 4\right)}{4}
= \left(\frac{(n+1)(n+2)}{2}\right)^{\!2} :
\]
आनुवंशिकता के लिए $ n = 3$: $1 + 8 + 27 = 36 = 6^2$.

\textbf{19.} के लिए एक रास्ता $(n, n)$ बनाता है $2n $ कदम और पार
विरोधी विकर्ण $ x + y = n $ बिल्कुल एक जाली बिंदु पर
$(j, n - j)$; पहला भाग एक पथ है $ j $ आर के बीच में है $ n $
चरण ($\binom nj $ विकल्प), दूसरा भाग, पीछे की ओर पढ़ें,
इसी तरह ($\binom nj $ फिर से, समरूपता द्वारा)। संक्षेप में कहें तो
क्रॉसिंग पॉइंट: $\binom{2n}{n} = \sum_j \binom nj^2$ ---
वेंडरमोंडे की पहचान, खींची गई।

मैट्रिक्स दो अध्याय आगे।
