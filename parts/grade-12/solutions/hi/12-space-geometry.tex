\end{solution}

\begin{solution}{exo:g12:space:4}
के अंक $\mathcal D $: $(1 + t,\, 2 - t,\, 2t)$. में प्रतिस्थापित करना
\omterm{def:g10:algebra:equation}{समीकरण} का $\mathcal P $:
\[
2(1+t) + (2 - t) - 2t + 1 = 5 - t = 0 \iff t = 5 .
\]
अद्वितीय \omterm{def:g10:numbers:interunion}{चौराहा} बिंदु: $(6, -3, 10)$.
\end{solution}

\begin{solution}{exo:g12:space:5}
 \omterm{def:g12:space:normal}{सामान्य वेक्टर} $(1, -1, 2)$ और $(2, 1, 1)$ नहीं हैं \omterm{def:g11:vect:collinear}{समरेख}, तो
विमान एक रेखा में प्रतिच्छेद करते हैं। दोनों को जोड़ना \omterm{def:g10:algebra:equation}{समीकरण}: $3x + 3z = 5$, इसलिए
$ x = \frac53 - z $; फिर पहला \omterm{def:g10:algebra:equation}{समीकरण} देता है
$ y = x + 2z - 1 = \frac23 + z $. साथ $ z = t $:
\[
(x, y, z) = \left(\frac53 - t,\; \frac23 + t,\; t\right), \qquad t \in \R,
\]
के साथ एक पंक्ति \omterm{def:g11:vect:direction}{दिशा सदिश} $(-1, 1, 1)$. (दोनों \omterm{def:g10:algebra:equation}{समीकरण} जांचें
समान रूप से।)
\end{solution}

\begin{solution}{exo:g12:space:6}
दिशा-निर्देश $\vec u_1(1, 2, -1)$ और $\vec u_2(1, 1, 2)$ नहीं हैं \omterm{def:g11:vect:collinear}{समरेख},
अतः रेखाएँ समानांतर नहीं हैं। अन्तर्विभाजक की आवश्यकता होगी
\[
1 + t = 2 + s, \qquad 2t = 1 + s, \qquad 3 - t = 1 + 2s .
\]
पहले दो देते हैं $ t = 1 + s $ और $2(1+s) = 1 + s $, इसलिए $ s = -1$, $ t = 0$;
लेकिन फिर तीसरा पढ़ता है $3 = -1$, बेतुका। कोई सामान्य बात नहीं और नहीं
समानांतर: रेखाएँ तिरछी हैं।
\end{solution}

\begin{solution}{exo:g12:space:7}
\emph{1.} $\operatorname{dist} = \dfrac{\abs{2\cdot3 - 1 + 2\cdot1 - 3}}
{\sqrt{4 + 1 + 4}} = \dfrac{4}{3}$.

\emph{2.} $ H = M_0 + t\,\vec n $ साथ $\vec n(2, -1, 2)$, चुनना $ t $ ताकि $ H \in \mathcal P $:
\[
2(3 + 2t) - (1 - t) + 2(1 + 2t) - 3 = 4 + 9t = 0 \iff t = -\frac49 .
\]
इस तरह $ H\left(\frac{19}{9}, \frac{13}{9}, \frac19\right)$, और
\[
M_0H = \norm{t\,\vec n} = \frac49 \times 3 = \frac43,
\]
दूरी सूत्र का मिलान।
\end{solution}

\begin{solution}{exo:g12:space:8}
\omterm{def:g10:coordgeom:system}{COORDINATES}: $ G(1,1,1)$, और विमान $(BDE)$ से गुजरता है $ B(1,0,0)$,
$ D(0,1,0)$, $ E(0,0,1)$.

\emph{1.} विमान $(BDE)$ है \omterm{def:g10:algebra:equation}{समीकरण} $ x + y + z = 1$ (इससे संतुष्ट हूं
तीन बिंदु, और वे संरेखित नहीं हैं), इसलिए $\vec n(1,1,1)$ के लिए सामान्य है
यह. तब से $\vect{AG} = (1,1,1) = \vec n $, विकर्ण $(AG)$ है \omterm{def:g11:scal:orthogonal}{ओर्थोगोनल}

\emph{2.} से दूरी $ A(0,0,0)$:
$\frac{\abs{0 + 0 + 0 - 1}}{\sqrt3} = \frac{1}{\sqrt3} = \frac{\sqrt3}{3}$.
रेखा $(AG)$ है $(t, t, t)$; यह विमान से मिलता है जब $3t = 1$:पर

\end{solution}

\begin{solution}{exo:g12:space:9}
\emph{1.} $\vect{AB}(1, 0, -1)$, $\vect{AC}(-1, 1, 0)$,
$\vect{AD}(1, 1, 1)$. अगर $\vect{AD} = a\vect{AB} + b\vect{AC}$, द
\omterm{def:g10:coordgeom:system}{COORDINATES} देना $ a - b = 1$, $ b = 1$, $-a = 1$: पहले दो बल
$ a = 2$, \omterm{thm:g12:limcont:ivt}{विरोधाभास}ी $ a = -1$. नहीं \omterm{def:g12:space:collinear}{समतलीय}.

\emph{2.} $\vect{BC}(-2, 1, 1)$, $\vect{BD}(0, 1, 2)$. हल
$\vec n \cdot \vect{BC} = \vec n \cdot \vect{BD} = 0$: से
$ b + 2c = 0$, $ b = -2c $; तब $-2a - 2c + c = 0$ देता है $ c = -2a $. साथ
$ a = 1$: $\vec n(1, 4, -2)$. के माध्यम से विमान $ B(2,1,0)$:
\[
x + 4y - 2z - 6 = 0 .
\]

\emph{3.} से दूरी $ A(1,1,1)$:
$\dfrac{\abs{1 + 4 - 2 - 6}}{\sqrt{21}} = \dfrac{3}{\sqrt{21}}$.
का क्षेत्रफल $ BCD $: $\norm{\vect{BC}}^2 = 6$, $\norm{\vect{BD}}^2 = 5$,
$\vect{BC}\cdot\vect{BD} = 3$, इसलिए
\[
\text{Area} = \frac12\sqrt{6 \times 5 - 9} = \frac{\sqrt{21}}{2}.
\]

\emph{4.}
\[
V = \frac13 \times \frac{\sqrt{21}}{2} \times \frac{3}{\sqrt{21}}
= \frac12 .
\]
\end{solution}

\begin{solution}{pb:g12:space:1}
\textbf{1.} विमान: $ x + y + z = 1$. रेखा: $(t, t, t)$.
\omterm{def:g10:numbers:interunion}{चौराहा}: $3t = 1$:
$\left(\frac13, \frac13, \frac13\right)$.

\textbf{2.} $\dfrac{\abs{0 + 0 + 0 - 1}}{\sqrt3} =
\dfrac{1}{\sqrt3} = \dfrac{\sqrt3}{3}$.

\textbf{3.} $1 - 1 + 0 = 0$: \omterm{def:g11:scal:orthogonal}{ओर्थोगोनल}.

\textbf{4.} वही सामान्य $(1,1,1)$, लेकिन $\frac52 \neq 1$:
कड़ाई से समानांतर विमान। रेखा: दिशा
$(1,1,0) \cdot (1,1,1) = 2 \neq 0$: समतल के समानांतर नहीं,
तो यह इसे एक बिंदु पर छेदता है।

\textbf{5.} \omterm{prop:g10:coordgeom:midpoint}{मध्य} बिन्दुओं: $\left(1, \frac12, 0\right)$,
$\left(\frac12, 1, 0\right)$, $\left(0, 1, \frac12\right)$,
$\left(0, \frac12, 1\right)$, $\left(\frac12, 0, 1\right)$,
$\left(1, 0, \frac12\right)$: प्रत्येक का समन्वित योग है
$\frac32$.

\textbf{6.} $ P $यह सामान्य है $(1,1,1)$, इसकी दिशा
$(AG)$: लंबवत. \omterm{pb:g11:quad:1}{केंद्र}
$\left(\frac12,\frac12,\frac12\right)$ समन्वित योग है
$\frac32$: पर $ P $.

\textbf{7.} लगातार \omterm{prop:g10:coordgeom:midpoint}{मध्य} बिन्दुओं से भिन्न वैक्टर पसंद
$\left(-\frac12, \frac12, 0\right)$: लंबाई
$\frac{\sqrt2}{2}$ हर बार --- छह बराबर भुजाएँ।

\textbf{8.} पर $\left(\frac12, 1, 0\right)$: धा\omterm{def:g11:vect:vector}{र-वेक्टर}
$\left(\frac12, -\frac12, 0\right)$ और
$\left(-\frac12, 0, \frac12\right)$: डॉट उत्पाद $-\frac14$,
मानदंड $\frac{\sqrt2}{2}$: $\cos = -\frac12$: कोण
$120^\circ $. समान भुजाएँ, समान $120^\circ $ कोण: एक नियमित
षट्भुज

\textbf{9.} परिधि $6 \times \frac{\sqrt2}{2} = 3\sqrt2
\approx 4.24$. क्षेत्र $6 \times \frac{\sqrt3}{4} \times
\frac12 = \frac{3\sqrt3}{4} \approx 1.30$: चेहरे से बड़ा
(क्षेत्र $1$) --- बॉक्स के किसी भी तरफ से बड़ा टुकड़ा; केवल
विकर्ण आयत ($\sqrt2 \approx 1.41$) इसे हरा देता है।

\textbf{10.} $(t,t,t)$ की बैठक $ P $ पर $ t = \frac12$: \omterm{pb:g11:quad:1}{केंद्र}
--- जो है \omterm{prop:g10:coordgeom:midpoint}{मध्य} का $[AG]$ ($ A $ पर $ t = 0$, $ G $ पर
$ t = 1$).

\textbf{11.} के लिए $ c = \frac12$: विमान तीन किनारों को काटता है
पर $ A $के कोने पर, पर $\left(\frac12,0,0\right)$,
$\left(0,\frac12,0\right)$, $\left(0,0,\frac12\right)$: एक
भुजा का समबाहु त्रिभुज $\frac{\sqrt2}{2}$. जैसा $ c $ बढ़ता है,
त्रिभुज बढ़ता है, इसके कोने एक षट्भुज में कट जाते हैं
(बिल्कुल नियमित रूप से $ c = \frac32$), तब चित्र सिकुड़ जाता है
सममित रूप से वापस एक त्रिभुज पर $ G $का कोना: द
क्रिस्टल-कट मॉर्फिंग।

\textbf{12.} अनुभाग का प्रत्येक पक्ष है \omterm{def:g10:numbers:interunion}{चौराहा} का
एक के साथ काटने वाला विमान \emph{चेहरा} घन का, और ए
विमान \omterm{def:g10:functions:extrema}{अधिकतम} एक खंड में एक फलक (एक सपाट बहुभुज) से मिलता है:
अधिक से अधिक $6$ पक्ष. सात असंभव है; तीन से छह तक सभी
घटित होता है (प्रश्न 8 और 11 उनमें से दो दिखाते हैं)।

\textbf{13.} आधार $ ABD $: क्षेत्रफल का समकोण त्रिभुज $\frac12$;
ऊंचाई $ AE = 1$: आयतन
$\frac13 \times \frac12 \times 1 = \frac16$.

\textbf{14.} प्रत्येक \omterm{def:g10:vectors:sum}{जोड़}ी के बीच $ B, D, E, G $ बिल्कुल अलग है
दो \omterm{def:g10:coordgeom:system}{COORDINATES} द्वारा $1$: सभी छह दूरियाँ बराबर $\sqrt2$ ---
नियमित। घन विभाजित हो जाता है $ BDEG $ प्लस चार कोने
टेट्राहेड्रा के अनुरूप $ ABDE $: आयतन
$1 - 4 \times \frac16 = \frac13$.

\textbf{15.} विमान $(BDE)$: $ x + y + z = 1$; \omterm{pb:g11:quad:1}{केंद्र}
$\left(\frac12,\frac12,\frac12\right)$: दूरी
$\frac{\abs{\frac32 - 1}}{\sqrt3} = \frac{1}{2\sqrt3} =
\frac{\sqrt3}{6}$.

\textbf{16.} $\vect{AG} = (1,1,1)$,
$\vect{BH} = (-1,1,1)$: $\cos\theta = \frac{-1 + 1 + 1}{3} =
\frac13$: $\theta \approx 70.5^\circ $, \omterm{def:g10:proba:operations}{पूरक}
$\approx 109.5^\circ $. चार इलेक्ट्रॉन \omterm{def:g10:vectors:sum}{जोड़}े एक दूसरे को प्रतिकर्षित करते हैं
और कार्बन के चारों ओर जहां तक संभव हो दूर-दूर तक फैलें: द
इष्टतम नियमित टेट्राहेड्रोन के कोने हैं, जिनका
\omterm{pb:g11:quad:1}{केंद्र} से शीर्ष तक की दिशाएँ बिल्कुल इसी कोण पर मिलती हैं ---
प्रश्न 14 हीरा क्रिस्टलीकृत है।

\textbf{17.} $(1, t, t)$ पर $ P $: $1 + 2t = \frac32$:
$ t = \frac14$: मुद्दा $\left(1, \frac14, \frac14\right)$.

\textbf{18.} $(AB)$: अंक $(t, 0, 0)$; $(EG)$: अंक
$(s, s, 1)$. मीटिंग की जरूरत पड़ेगी $0 = 1$ तीसरे में

\textbf{19.} $ B + t(1,1,1) = (1 + t, t, t)$ पर $ P $:
$1 + 3t = \frac32$: $ t = \frac16$: प्रक्षेपण

\textbf{20.} पैरामीट्रिक रेखाएँ छेदती हैं (प्रश्न 1, 10, 17);
\omterm{def:g12:space:normal}{सामान्य वेक्टर} कोण और दूरियाँ मापें (प्रश्न 2, 8,
