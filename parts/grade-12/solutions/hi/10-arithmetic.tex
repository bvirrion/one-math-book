\section*{अध्याय \ref{ch:g12:arith} --- अंकगणित}

\begin{solution}{exo:g12:arith:1}
$17 \times 119 = 2023$, इसलिए $2026 = 17 \times 119 + 3$: भागफल $119$,
शेष $3$. के लिए $-2026$: $-2026 = 17\times(-120) + 14$ (वास्तव में
$17 \times 120 = 2040$ और $2040 - 2026 = 14$): भागफल $-120$, शेष
$14$ (शेष को अंदर रहना चाहिए $\intco{0}{17}$, सो है \emph{नहीं} $-3$).
\end{solution}

\begin{solution}{exo:g12:arith:2}
सापेक्ष $10$: $7^2 = 49 \equiv 9 \equiv -1$. इस तरह
$7^{100} = \left(7^2\right)^{50} \equiv (-1)^{50} = 1 \pmod{10}$: आखिरी
का अंक $7^{100}$ है $1$.
\end{solution}

\begin{solution}{exo:g12:arith:3}
यूक्लिड: $1071 = 2\times462 + 147$; $462 = 3\times147 + 21$;
$147 = 7\times21 + 0$. इसलिए $\gcd = 21$.

पश्च-प्रतिस्थापन: $21 = 462 - 3\times147 = 462 - 3(1071 - 2\times462)
= 7\times462 - 3\times1071$. इस प्रकार $ u = -3$, $ v = 7$:
$1071\times(-3) + 462\times7 = 21$.
\end{solution}

\begin{solution}{exo:g12:arith:4}
प्रत्येक पूर्णांक है $\equiv 0, 1, 2$ या $3 \pmod 4$, और वर्ग करना:
$0^2 \equiv 0$, $1^2 \equiv 1$, $2^2 = 4 \equiv 0$, $3^2 = 9 \equiv 1$. इसलिए
$ n^2 \equiv 0$ या $1 \pmod 4$. तब दो वर्गों का योग सर्वांगसम होता है
$0 + 0$, $0 + 1$ या $1 + 1$, \emph{यानी} को $0$, $1$ या $2 \pmod 4$ ---
कभी नहीं $3$.
\end{solution}

\begin{solution}{exo:g12:arith:5}
\omterm{def:g12:arith:divides}{भाजकत्व} द्वारा $2$: के बीच $ n $ और $ n + 1$, एक सम है. \omterm{def:g12:arith:divides}{भाजकत्व} द्वारा
$3$: अगर $ n \equiv 0$, तब $3 \mid n $; अगर $ n \equiv 1 \pmod 3$, तब
$2n + 1 \equiv 3 \equiv 0$; अगर $ n \equiv 2$, तब $ n + 1 \equiv 0$. कुल मिलाकर
मामले $3$ विभाजित उत्पाद. तब से $\gcd(2,3) = 1$,
\cref{cor:g12:arith:coprimeprod} देता है $6 \mid n(n+1)(2n+1)$.
(इससे यह भी पुनः \omterm{def:g11:vect:det}{सिद्ध} होता है $\frac{n(n+1)(2n+1)}{6}$, के वर्गों का योग
\cref{exo:g12:seq:1}, एक पूर्णांक.)
\end{solution}

\begin{solution}{exo:g12:arith:6}
हम इसके विपरीत की तलाश करते हैं $5$ सापेक्ष $11$: परीक्षण (या बेज़आउट),
$5 \times 9 = 45 = 44 + 1 \equiv 1 \pmod{11}$. का गुणा करना अनुरूपता द्वारा
$9$:
\[
x \equiv 9 \times 3 = 27 \equiv 5 \pmod{11}.
\]
समाधान हैं \omterm{def:g10:numbers:sets}{पूर्णांकों} $ x = 5 + 11k $, $ k \in \Z $. (जाँच करना:
$5\times5 = 25 \equiv 3 \pmod{11}$.)
\end{solution}

\begin{solution}{exo:g12:arith:7}
$\gcd(17, 40) = 1$, इसलिए समाधान मौजूद हैं। यूक्लिड:
$40 = 2\times17 + 6$; $17 = 2\times6 + 5$; $6 = 5 + 1$. पश्च-प्रतिस्थापन:
$1 = 6 - 5 = 6 - (17 - 2\times6) = 3\times6 - 17
= 3(40 - 2\times17) - 17 = 3\times40 - 7\times17$. इस तरह
$17\times(-7) - 40\times(-3) = 1$: विशेष समाधान
$(x_0, y_0) = (-7, -3)$.

का सामान्य समाधान $17x - 40y = 1$: विशेष संबंध को घटाना,
$17(x + 7) = 40(y + 3)$; तब से $\gcd(17, 40) = 1$, गॉस देता है
$40 \mid x + 7$, इसलिए $ x = -7 + 40k $ और तब $ y = -3 + 17k $, $ k \in \Z $
(ये सभी जाँचें)।

के लिए $17x - 40y = 6$, विशेष समाधान को इससे गुणा करें $6$:
$(x_1, y_1) = (-42, -18)$, और वही \omterm{def:g12:complex:argument}{तर्क} देता है
\[
x = -42 + 40k, \qquad y = -18 + 17k, \qquad k \in \Z .
\]
(उदा.\ $ k = 2$: $ x = 38$, $ y = 16$; वास्तव में $17\times38 - 40\times16
= 646 - 640 = 6$.)
\end{solution}

\begin{solution}{exo:g12:arith:8}
कल्पना करना $\sqrt2 = \frac ab $ साथ $ a, b \in \N^*$; तब $ a^2 = 2b^2$. में
\omterm{def:g12:arith:prime}{मुख्य} एक वर्ग का गुणनखंडन, प्रत्येक घातांक सम है; तो के प्रतिपादक
$2$ में $ a^2$ सम है, जबकि अंदर $2b^2$ यह \omterm{def:g11:func:parity}{विषम} है (सम से एक अधिक)।
संख्या). एक ही के दो गुणनखंड पूर्णांक के विभिन्न प्रतिपादकों के साथ
$2$ में विशिष्टता का खंडन करें \cref{thm:g12:arith:fta}. अत: ऐसा कुछ नहीं
अंश मौजूद है: $\sqrt2 \notin \Q $.
\end{solution}

\begin{solution}{exo:g12:arith:9}
\emph{1.} से $ k\binom pk = p \binom{p-1}{k-1}$, $ p $ विभाजित
$ k\binom pk $. के लिए $1 \leq k \leq p-1$, $ p \nmid k $ और $ p $ \omterm{def:g12:arith:prime}{मुख्य} देना
$\gcd(p, k) = 1$, तो \omterm{thm:g12:arith:gauss}{गॉस की लेम्मा} उपज देती है $ p \mid \binom pk $.

\emph{2.} प्रेरण चालू $ a $. के लिए $ a = 0$: $0^p \equiv 0$. मान लीजिए
$ a^p \equiv a \pmod p $. द्विपद प्रमेय द्वारा,
\[
(a+1)^p = \sum_{k=0}^{p} \binom pk a^k
\equiv a^p + 1 \pmod p,
\]
सभी \omterm{prop:g10:coordgeom:midpoint}{मध्य} पद लुप्त हो रहे हैं $ p $ बिंदु 1 द्वारा. प्रेरण द्वारा
परिकल्पना, $(a+1)^p \equiv a + 1 \pmod p $. ये साबित होता है
$ a^p \equiv a $ सभी के लिए $ a \in \N $, और मामला $ a < 0$ लिखकर अनुसरण करता है
$ a \equiv a + kp $ एक उपयुक्त सकारात्मक प्रतिनिधि के लिए.
\end{solution}

\begin{solution}{exo:g12:arith:10}
$ n \equiv 2 \pmod 3$ और $ n \equiv 3 \pmod 5$: लिखना $ n = 2 + 3s $; तब
$2 + 3s \equiv 3 \pmod 5$, \emph{यानी} $3s \equiv 1 \pmod 5$. का उलटा
$3$ सापेक्ष $5$ है $2$ ($3\times2 = 6 \equiv 1$), इसलिए $ s \equiv 2 \pmod 5$,
कहना $ s = 2 + 5t $, और $ n = 8 + 15t $: पहली दो शर्तें \omterm{def:g11:stat:mean}{अर्थ}
$ n \equiv 8 \pmod{15}$.

\omterm{def:g10:vectors:sum}{जोड़}ा जा रहा है $ n \equiv 2 \pmod 7$: $8 + 15t \equiv 2 \pmod 7$, और
$15 \equiv 1 \pmod 7$, इसलिए $ t \equiv -6 \equiv 1 \pmod 7$, कहना
$ t = 1 + 7u $. इस तरह $ n = 23 + 105u $:
\[
n \equiv 23 \pmod{105}.
\]
(जाँच करना: $23 = 3\times7 + 2 = 5\times4 + 3 = 7\times3 + 2$.)
\end{solution}

\begin{solution}{pb:g12:arith:1}
\textbf{1.} $2026 = 289 \times 7 + 3$: $2026 \equiv 3
\pmod 7$. की शक्तियां $7$ आधुनिक $10$: $7, 9, 3, 1$, का चक्र
लंबाई $4$; $100 \equiv 0 \pmod 4$: का अंतिम अंक
$7^{100}$ है $1$.

\textbf{2.} $5^2 = 25 \equiv -1 \pmod{13}$, इसलिए
$5^{116} = \left(5^2\right)^{58} \equiv (-1)^{58} = 1$ और
$5^{117} \equiv 5 \pmod{13}$.

\textbf{3.} का उलटा $3$ सापेक्ष $7$ है $5$
($15 \equiv 1$): $ x \equiv 5 \times 5 = 25 \equiv 4 \pmod 7$.

\textbf{4.} $97 = 2 \times 35 + 27$; $35 = 27 + 8$;
$27 = 3 \times 8 + 3$; $8 = 2 \times 3 + 2$; $3 = 2 + 1$.
पश्च-प्रतिस्थापन: $1 = 97 \times 13 + 35 \times (-36)$. इसलिए
$35 \times (-36) \equiv 1 \pmod{97}$: का उलटा $35$ है
$-36 \equiv 61 \pmod{97}$.

\textbf{5.} $ a $ व्युत्क्रमणीय मॉड्यूलो है $ n $ बिल्कुल कब
$\gcd(a, n) = 1$: बीézout प्रदान करता है $ au + nv = 1$, \omterm{def:g11:stat:mean}{अर्थ}ात\
$ au \equiv 1$; इसके विपरीत एक व्युत्क्रम बल देता है जी.सी.डी अलग करना
$1$.

\textbf{6.} $0{\cdot}10 + 3{\cdot}9 + 0{\cdot}8 + 6{\cdot}7 +
4{\cdot}6 + 0{\cdot}5 + 6{\cdot}4 + 1{\cdot}3 + 5{\cdot}2 +
2{\cdot}1 = 132 = 12 \times 11 \equiv 0 \pmod{11}$: वैध।

\textbf{7.} योग में \omterm{def:g12:comb:permutation}{परिवर्तन} होता है $ wd $ साथ
$1 \leq w \leq 10$ और $1 \leq \abs d \leq 9$: तब से $11$ है
\omterm{def:g12:arith:prime}{मुख्य} और विभाजित न तो कारक, यह उत्पाद को विभाजित नहीं कर सकता
(\cref{thm:g12:arith:gauss} / \cref{prop:g12:arith:primedivides}):
बदली हुई राशि दोबारा कभी नहीं होगी $\equiv 0$: हर एक अंक
त्रुटि अलार्म को ट्रिप कर देती है।

\textbf{8.} आसन्न अंकों की अदला-बदली $ a, b $ (वजन
$ w + 1, w $) द्वारा योग बदलता है
$(w+1)b + wa - (w+1)a - wb = b - a \not\equiv 0$ के लिए
$ a \neq b $: पता चला. रहस्य यह है \emph{आदिमता} का
$11$: मॉड्यूलो $10$, जैसे उत्पाद $5 \times 2$ के साथ गायब हो जाओ
न तो कारक शून्य, इसलिए एक भार-$5$ की त्रुटि $\pm 2$ (या एक
अशुभ स्थानान्तरण) फिसल सकता है।

\textbf{9.} बारह अंकों का भारित योग: $119$; द
चेक अंक को इसके गुणज में पूरा करना होगा $10$: $1$ (पूर्ण
कोड $978\,2940199\,051$). ईएएन आसन्न स्थानान्तरण से चूक जाता है
के साथ $2(a - b) \equiv 0 \pmod{10}$, \omterm{def:g11:stat:mean}{अर्थ}ात\
$\abs{a - b} = 5$: अदला-बदली ए $2$ और ए $7$, कहते हैं, गुजरता है
अनदेखी --- मित्रवत की कीमत \omterm{def:g12:complex:modulus}{मापांक} $10$.

\textbf{10.} दाईं ओर से प्रत्येक दूसरे अंक को दोगुना करना और
तह ($16 \to 7$, आदि), योग आता है $80 \equiv 0
\pmod{10}$: परीक्षण कार्ड मान्य करता है।

\textbf{11.} के साथ \omterm{def:g12:arith:prime}{मुख्य} \omterm{def:g12:complex:modulus}{मापांक} प्रत्येक भार उलटा है,
तो \emph{सभी} एकल त्रुटियाँ और \emph{सभी} आसन्न
ट्रांसपोज़िशन पकड़े गए हैं --- आईएसबीएन की विलासिता; आधुनिकीकरण $10$ योजनाएं
मानव-अनुकूल अंक रखें और एक छोटा ब्लाइंड स्पॉट स्वीकार करें।

\textbf{12.} $2^{10} = 1024 = 3 \times 341 + 1 \equiv 1
\pmod{341}$, इस तरह $2^{340} = \left(2^{10}\right)^{34} \equiv
1$. अभी तक $341 = 11 \times 31$ मिश्रित है: यह फ़र्मेट से गुजरता है
आधार पर परीक्षण करें $2$ जबकि नहीं \omterm{def:g12:arith:prime}{मुख्य}. नैतिक: फ़र्मेट का
अनुरूपता आवश्यक है, पर्याप्त नहीं --- प्रारंभिक परीक्षण
तेज़ उपकरणों की आवश्यकता है (और उन्हें विश्वविद्यालय में प्राप्त किया जाता है)।
वॉल्यूम)।

\textbf{13.} $3d \equiv 1 \pmod{20}$: $ d = 7$
($21 = 20 + 1$).

\textbf{14.} $ c = 4^3 = 64 \equiv 31 \pmod{33}$.

\textbf{15.} $31 \equiv -2$: $(-2)^7 = -128$, और
$-128 + 4 \times 33 = 4$: सिफरटेक्स्ट डिक्रिप्ट करता है
$ m = 4$. ताला घूम जाता है.

\textbf{16.} सापेक्ष $3$: अगर $3 \nmid m $,
$ m^2 \equiv 1$ (फर्मेट), तो
$ m^{21} = m \cdot \left(m^2\right)^{10} \equiv m $; अगर
$3 \mid m $, दोनों पक्ष हैं $\equiv 0$. सापेक्ष $11$:
$ m^{10} \equiv 1$ या $11 \mid m $, और
$ m^{21} = m \cdot \left(m^{10}\right)^2 \equiv m $. दोनों $3$
और $11$ विभाजित करना $ m^{21} - m $, और होना \omterm{def:g12:arith:gcd}{सह अभाज्य} उनका उत्पाद
$33$ भी करता है (गॉस): $ m^{21} \equiv m \pmod{33}$. द
प्रतिपादक $ ed = 21 = 1 + 20k $ इसलिए बनाया गया था कि दोनों फर्मेट
प्रतिपादक ($2$ और $10$, विभाजन $20$) गायब।

\textbf{17.} दो का गुणा करना $300$-अंक अभाज्य एक लेता है
माइक्रोसेकंड; उन्हें अपने उत्पाद से पुनर्प्राप्त करना हर किसी को हरा देता है
ज्ञात एल्गोरिदम और दुनिया के सभी कंप्यूटर --- लॉक है
एक तरफ़ा सड़क. (हमारा $ n = 33$ खिलौना पैमाने पर सड़क है,
दोनों दिशाओं में चलने योग्य।)

\textbf{18.} परीक्षण अवशेष (या बी के साथ निर्माण)\'बाहर):
$ n \equiv 8 \pmod{15}$: मायने रखता है $8, 23, 38, 53, \dots $
विशिष्टता मॉड्यूलो $15$: दो समाधानों में गुणज का अंतर होता है
$3$ और का $5$, इसलिए का $15$ ($3$ और $5$ \omterm{def:g12:arith:gcd}{सह अभाज्य}, गॉस)।
जनरल के साथ $1000$ सैनिकों ने घोषणा की ``$8$''तीन से
त्वरित लाइन-अप --- \omterm{def:g12:integ:primitive}{प्राचीन} गणना युक्ति।

\textbf{19.} $10 \equiv 1 \pmod 9$ देता है $10^k \equiv 1$, इसलिए
$\sum d_k 10^k \equiv \sum d_k $: एक संख्या और उसके अंकों का योग
सर्वांगसम मॉड्यूलो हैं $9$ (और मॉड्यूलो $3$). और
$10 \equiv -1 \pmod{11}$ देता है
$\sum d_k 10^k \equiv \sum (-1)^k d_k $: वैकल्पिक नियम.
बचपन के दो नियम, एक-एक पंक्ति।

\textbf{20.} सर्वांगसमताएँ शेषफलों को \omterm{def:g12:seq:arith-geom}{अंकगणित} में बदल दिया
(भाग I); बीézout ने ऐसे व्युत्क्रम गढ़े जो रैखिक हल करते हैं
सर्वांगसमताएँ और आरएसए $ d$ (प्रश्न 4,13); फ़र्मेट छोटा है
प्रमेय ने ताला खोला और बंद किया (प्रश्न 15-16); चिपकाने
\omterm{def:g12:arith:gcd}{सह अभाज्य} मोडुली ने सैनिकों की गिनती की और सबूत पूरा किया
(प्रश्न 16,18). हार्डी पर निर्णय: वह सबसे शुद्धतम प्रमेय है
जानता था कि अब हर खरीद पर पहरा है --- शुद्धता, दिया गया समय, है
वहां सबसे अधिक लागू होने वाली बात है.
\end{solution}
