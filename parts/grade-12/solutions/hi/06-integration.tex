\begin{solution}{exo:g12:integ:1}
\[
\int_1^2 \left(3x^2 - \frac{1}{x^2}\right)\dd x
= \left[x^3 + \frac1x\right]_1^2 = \left(8 + \tfrac12\right) - 2 = \frac{13}{2}.
\]
\[
\int_0^{\pi/2} \cos t\,\dd t = \bigl[\sin t\bigr]_0^{\pi/2} = 1 .
\]
साथ $u = 2t+1$, $\frac{1}{2t+1} = \frac12\,\frac{u'}{u}$:
\[
\int_0^1 \frac{\dd t}{2t+1} = \frac12\bigl[\ln(2t+1)\bigr]_0^1
= \frac{\ln 3}{2}.
\]
\end{solution}

\begin{solution}{exo:g12:integ:2}
$ x\,\eu^{x^2} = \frac12\,(x^2)'\,\eu^{x^2}$ है पुरातन
$\frac12\eu^{x^2} + c $. स्थिति $ F(0) = 1$ देता है $\frac12 + c = 1$,
इसलिए
\[
F(x) = \frac{\eu^{x^2} + 1}{2}.
\]
\end{solution}

\begin{solution}{exo:g12:integ:3}
\[
\mu = \frac{1}{\pi}\int_0^\pi \sin t\,\dd t
= \frac{1}{\pi}\bigl[-\cos t\bigr]_0^\pi = \frac{2}{\pi} \approx 0.64 .
\]
\omterm{def:g10:functions:graph}{ग्राफ}़िक रूप से, आधार का आयत $\intcc{0}{\pi}$ और ऊंचाई
$\frac{2}{\pi}$ के मेहराब के समान क्षेत्रफल है \omterm{def:g12:trigo:cossin}{ज्या} वक्र.
\end{solution}

\begin{solution}{exo:g12:integ:4}
लेना $ u' = 1$, $ v = \ln t $, इसलिए $ u = t $, $ v' = \frac1t $:
\[
\int_1^{\eu} \ln t\,\dd t = \bigl[t\ln t\bigr]_1^{\eu} - \int_1^{\eu} 1\,\dd t
= \eu - (\eu - 1) = 1 .
\]
लेना $ u' = \sin t $, $ v = t $, इसलिए $ u = -\cos t $, $ v' = 1$:
\[
\int_0^\pi t\sin t\,\dd t = \bigl[-t\cos t\bigr]_0^\pi + \int_0^\pi \cos t\,\dd t
= \pi + 0 = \pi .
\]
\end{solution}

\begin{solution}{exo:g12:integ:5}
वक्र कहाँ प्रतिच्छेद करते हैं $ x^2 = x + 2$, \emph{यानी} $ x \in \{-1, 2\}$, और पर $\intcc{-1}{2}$ लाइन ऊपर है \omterm{def:g11:quad:parabola}{परवलय}. क्षेत्र है
\[
\int_{-1}^{2} \bigl(x + 2 - x^2\bigr)\dd x
= \left[\frac{x^2}{2} + 2x - \frac{x^3}{3}\right]_{-1}^{2}
= \frac{10}{3} - \left(-\frac{7}{6}\right) = \frac{9}{2}.
\]
\end{solution}

\begin{solution}{exo:g12:integ:6}
\emph{1.} $ I = \bigl[\ln(1+t)\bigr]_0^1 = \ln 2$.

\emph{2.}
\[
I_n + I_{n+1} = \int_0^1 \frac{t^n(1 + t)}{1+t}\,\dd t
= \int_0^1 t^n \,\dd t = \frac{1}{n+1}.
\]
पर $\intcc{0}{1}$, $0 \leq \dfrac{t^n}{1+t} \leq t^n $, तो तुलना से
$0 \leq I_n \leq \frac{1}{n+1}$.

\emph{3.} पुनरावृत्ति से, टेलीस्कोपिंग प्रेरण द्वारा,
\[
I_n = (-1)^n\left(I_0 - \left(1 - \frac12 + \dots + \frac{(-1)^{n-1}}{n}\right)\right),
\]
\emph{यानी}
$\displaystyle\sum_{k=1}^{n} \frac{(-1)^{k-1}}{k} = \ln 2 - (-1)^n I_n $
(याद करना $ I_0 = \ln 2$). तब से $\abs{I_n} \leq \frac{1}{n+1} \to 0$, द
प्रत्यावर्ती हार्मोनिक योग की प्रवृत्ति होती है $\ln 2$.
\end{solution}

\begin{solution}{exo:g12:integ:7}
दूरी:
\[
d = \int_0^{100} 30\left(1 - \eu^{-t/50}\right)\dd t
= 30\left[t + 50\,\eu^{-t/50}\right]_0^{100}
= 30\left(100 + 50\eu^{-2} - 50\right)
= 1500 + 1500\,\eu^{-2}.
\]
संख्यानुसार $ d \approx 1500 + 203 = 1703$ एम। औसत गति:
$\frac{d}{100} = 15\bigl(1 + \eu^{-2}\bigr) \approx 17.0$ एमएस।
\end{solution}

\begin{solution}{exo:g12:integ:8}
\emph{1.} $ S_n $ का कुल क्षेत्रफल है $ n $ चौड़ाई के आयत $\frac1n $ और
ऊंचाई $ f\left(\frac kn\right)$, $ k = 1, \dots, n $, कहाँ
$ f(t) = \frac{1}{1+t}$: यह ``सही समापन बिंदु'' आयत है \omterm{def:g10:numbers:approx}{सन्निकटन}
के अंतर्गत क्षेत्र का $ f $ पर $\intcc{0}{1}$.

\emph{2.} $ f $ है \omterm{def:g10:functions:variations}{घटते}, तो प्रत्येक पर \omterm{def:g10:numbers:interval}{अंतराल}
$\intcc{\frac{k-1}{n}}{\frac kn}$,
$ f\left(\frac kn\right) \leq f(t) \leq f\left(\frac{k-1}{n}\right)$;
एकीकृत करना और सारांशित करना $ k $:
\[
S_n \leq \int_0^1 f(t)\,\dd t \leq S_n + \frac1n\bigl(f(0) - f(1)\bigr)
= S_n + \frac{1}{2n}.
\]
(अभ्यास के प्रदर्शन का \omterm{prop:g10:coordgeom:midpoint}{मध्य} पद, साथ में
$ f(0) - f(1) = 1 - \frac12$, क्या वास्तव में यह बाध्य है।) इसलिए
$0 \leq \ln 2 - S_n \leq \frac{1}{2n} \to 0$, इसलिए $ S_n \to \ln 2$.
\end{solution}

\begin{solution}{exo:g12:integ:9}
\emph{1.} $ W_0 = \int_0^{\pi/2} \dd t = \frac{\pi}{2}$ और
$ W_1 = \bigl[-\cos t\bigr]_0^{\pi/2} = 1$.

\emph{2.} के साथ भागों द्वारा एकीकृत करें $ u' = \sin t $, $ v = \sin^{n+1} t $, इसलिए
$ u = -\cos t $, $ v' = (n+1)\sin^n t \cos t $:
\[
W_{n+2} = \bigl[-\cos t \sin^{n+1} t\bigr]_0^{\pi/2}
+ (n+1)\int_0^{\pi/2} \cos^2 t\,\sin^n t\,\dd t
= (n+1)\int_0^{\pi/2} (1 - \sin^2 t)\sin^n t\,\dd t,
\]
ब्रैकेट दोनों सिरों पर गायब हो रहा है। इस तरह
$ W_{n+2} = (n+1)(W_n - W_{n+2})$, \emph{यानी}
$ W_{n+2} = \frac{n+1}{n+2} W_n $.

\emph{3.} $ W_2 = \frac12 W_0 = \frac{\pi}{4}$,
$ W_3 = \frac23 W_1 = \frac23$,
$ W_4 = \frac34 W_2 = \frac{3\pi}{16}$.
पर $\intoo{0}{\frac\pi2}$, $0 < \sin t < 1$, इसलिए
$\sin^{n+1} t \leq \sin^n t $ इंटीरियर पर सख्त असमानता के साथ;
एकीकृत करना, $0 < W_{n+1} \leq W_n $ (वास्तव में $<$): $(W_n)$ है \omterm{def:g10:functions:variations}{घटते} और
सकारात्मक.
\end{solution}

\begin{solution}{pb:g12:integ:1}
\textbf{1.} $\left[x^3 - x^2 + x\right]_0^1 = 1$; \quad
$\left[\ln x\right]_1^{\eu} = 1$; \quad
$\left[\sin x\right]_0^{\pi/2} = 1$.

\textbf{2.} $\left[\frac12 \eu^{x^2}\right]_0^1 =
\frac{\eu - 1}{2}$; \quad
$\left[\ln(x^2 + 1)\right]_0^1 = \ln 2$.

\textbf{3.} $\int_0^1 x\eu^x = \left[x\eu^x\right]_0^1 -
\int_0^1 \eu^x = \eu - (\eu - 1) = 1$. और
$\int_1^{\eu} \ln x = \left[x\ln x\right]_1^{\eu} -
\int_1^{\eu} 1 = \eu - (\eu - 1) = 1$.

\textbf{4.} $\frac{1}{\pi}\int_0^\pi \sin x\,\dd x =
\frac{2}{\pi} \approx 0.64$: की औसत ऊंचाई \omterm{def:g12:trigo:cossin}{ज्या}
आर्क है $\frac2\pi $, उल्लेखनीय रूप से आधे से अधिक ---
इसके शीर्ष के पास मेहराब मोटा है।

\textbf{5.} $\int_0^1 (x - x^2)\,\dd x = \frac12 - \frac13 =
\frac16$.

\textbf{6.} $ L_n = \frac1n \sum_{k=0}^{n-1}
\left(\frac kn\right)^2$ और
$ U_n = \frac1n \sum_{k=1}^{n} \left(\frac kn\right)^2$:
चौड़ाई के आयत $\frac1n $ बायीं ओर पढ़ी गई ऊँचाइयों के साथ
(वक्र के नीचे) या दाएँ (ऊपर) समापन बिंदु।

\textbf{7.} के लिए सच है $ n = 1$ ($1 = \frac{1 \cdot 2 \cdot
3}{6}$). यदि यह कायम है $ n $, फिर जोड़ना $(n+1)^2$:
\[
\frac{n(n+1)(2n+1)}{6} + (n+1)^2
= \frac{(n+1)\left(2n^2 + n + 6n + 6\right)}{6}
= \frac{(n+1)(n+2)(2n+3)}{6},
\]
सूत्र पर $ n + 1$: आनुवंशिकता, हो गया।

\textbf{8.} $ U_n = \frac{1}{n^3} \cdot
\frac{n(n+1)(2n+1)}{6} = \frac{(1 + \frac1n)(2 + \frac1n)}{6}
\to \frac26 = \frac13$, और $ L_n = U_n - \frac1n \to \frac13$:
इसकी सीढ़ियों के बीच का क्षेत्र बिल्कुल वैसा ही है
$\frac13$.

\textbf{9.} $\int_0^1 x^2\,\dd x =
\left[\frac{x^3}{3}\right]_0^1 = \frac13$: एक लकीर। द
मौलिक प्रमेय एक अनंत निचोड़ को एकल में परिवर्तित करता है
antiderivative मूल्यांकन---इसलिए कहा जाता है
मौलिक.

\textbf{10.} खंड: $\int_{-1}^{1}(1 - x^2)\,\dd x =
2 - \frac23 = \frac43$. आधार के साथ त्रिकोण, जीवा (लंबाई)
$2$, ऊंचाई पर $1$) और शीर्ष $(0,0)$: क्षेत्र
$\frac12 \times 2 \times 1 = 1$. अनुपात: $\frac43$ --- बिलकुल
आर्किमिडीज़ का प्रमेय, उस मशीन द्वारा सत्यापित किया गया जिसके पास उसकी कमी थी।

\textbf{11.} $ T\left(1 + \frac14 + \frac{1}{16} + \dots\right)
= T \cdot \frac{1}{1 - \frac14} = \frac43 T $ (ज्यामितीय श्रृंखला
अनुपात का $\frac14$): साथ $ T = 1$, खंड का क्षेत्रफल
$\frac43$ दोबारा। आर्किमिडीज़ ने श्रृंखला का सारांश विशुद्ध रूप से दिया
ज्यामितीय तर्क---चॉकलेट-बार के चौथाई हिस्से
दयालु, दो सहस्राब्दी पहले सीमाएं शब्द थीं।

\textbf{12.} थकावट के लिए एक चतुर ज्यामिति की आवश्यकता होती है
आत्म-समानता और प्रत्येक मामले में सटीक क्षेत्र देता है; रीमैन
केवल एकरसता की आवश्यकता है और एक सार्वभौमिक परिभाषा देता है
रकम की गणना की कीमत; मौलिक प्रमेय की आवश्यकता है
की पूरी इमारत डेरिवेटिव और क्षेत्रफल बनाकर इसका भुगतान करता है
एक-पंक्ति गणना.

\textbf{13.} $ A(1) = 0$ और $ A'(x) = \frac1x $ (मौलिक
प्रमेय): $ A $ है \omterm{def:g12:integ:primitive}{प्राचीन} का $\frac1x $ पर
$\intoo{0}{+\infty}$ पर लुप्त हो रहा है $1$ --- जो बिल्कुल सही है
द \omterm{def:g12:exp:ln}{प्राकृतिक}.

\textbf{14.} $ g'(x) = \frac{a}{ax} - \frac1x = 0$: $ g $ है
निरंतर; पर $ x = 1$: $ g(1) = A(a) - A(1) = A(a)$. इस तरह
$ A(ax) = A(a) + A(x)$ सभी के लिए $ x $, और साथ में $ x = b $:
$\ln(ab) = \ln a + \ln b $. अतिशयोक्तिके क्षेत्र की परवाह नहीं है
क्षैतिज पैमाने के बारे में: खींचना $ t $ द्वारा $ a $ स्क्वैश
$\frac1t $ द्वारा $ a $, और क्षेत्र बचता है --- इसके अतिरिक्त
लघुगणक एक स्केलिंग समरूपता है.

\textbf{15.} पुनरावृत्त: $\ln(a^n) = n\ln a $ (प्रेरण पर
$ n $). और $0 = \ln 1 = \ln\left(a \cdot \frac1a\right) =
\ln a + \ln\frac1a $: $\ln\frac1a = -\ln a $.

\textbf{16.} $\frac{1}{11} + \frac{1}{12} + \dots +
\frac{1}{20} \approx 0.669$, ख़िलाफ़ $\ln 2 \approx 0.693$:
नीचे दस आयतें $\frac{1}{1+t}$, पहले से ही बंद है। द
हार्मोनिक श्रृंखला अलग हो जाती है, लेकिन यह \emph{स्लाइस} से $ n $ को
$2n $ पर बसा है $\ln 2$: क्रॉल पर विचलन, एक द्वारा मापा जाता है
क्षेत्र.

\textbf{17.} दूरी: $\int_0^{10} 3t^2\,\dd t = 1000$ एम।
\omterm{def:g11:stat:mean}{अर्थ} रफ़्तार: $100$ एमएस। के साथ तुरंत $ v = 100$:
$3t^2 = 100$: $ t = \sqrt{\frac{100}{3}} \approx 5.77$ एस।

\textbf{18.} द्रव्यमान: $\int_0^4 (2 + x)\,\dd x = 8 + 8 = 16$ किग्रा.
पल: $\int_0^4 (2x + x^2)\,\dd x = 16 + \frac{64}{3} =
\frac{112}{3}$. सेंटर ऑफ मास:
$\frac{112/3}{16} = \frac73 \approx 2.33$ मी--- परे धकेल दिया गया
जैसा कि अंतर्ज्ञान की मांग है, भारी दाहिने सिरे से \omterm{prop:g10:coordgeom:midpoint}{मध्य}।

\textbf{19.} $ W_0 = \frac\pi2$; $ W_1 = \left[-\cos
t\right]_0^{\pi/2} = 1$; और
$ W_2 = \int_0^{\pi/2} \frac{1 - \cos 2t}{2}\,\dd t =
\frac\pi4$.

\textbf{20.} क्षेत्र: परिभाषा, सीढ़ियों द्वारा निचोड़ा हुआ।
संचय: गति से दूरी, घनत्व से द्रव्यमान ---
द \omterm{def:g12:integ:area}{अभिन्न} एक चालू कुल के रूप में। antiderivative: द
मौलिक प्रमेय, न्यूटन और लीबनिज़ का शॉर्टकट
आर्किमिडीज़ का धैर्य. मोती: द $\ln $ बटन क्षेत्र है
के अंतर्गत $\frac1t $, और उस क्षेत्र की स्केलिंग समरूपता \emph{है}
पहचान $\ln(ab) = \ln a + \ln b$.
\end{solution}
