\section*{अध्याय \ref{ch:g12:seq} --- दृश्यों}

\begin{solution}{exo:g12:seq:1}
होने देना $P(n)$ कथन हो $\sum_{k=1}^{n} k^2 = \frac{n(n+1)(2n+1)}{6}$.
\emph{बेस केस:} के लिए $ n = 0$ दोनों पक्ष हैं $0$ (खाली राशि).
\emph{आगमनात्मक कदम:} मान लीजिए $ P(n)$. तब
\[
\sum_{k=1}^{n+1} k^2 = \frac{n(n+1)(2n+1)}{6} + (n+1)^2
= \frac{(n+1)\bigl(n(2n+1) + 6(n+1)\bigr)}{6}
= \frac{(n+1)(2n^2 + 7n + 6)}{6}.
\]
तब से $2n^2 + 7n + 6 = (n+2)(2n+3)$, यह है
$\frac{(n+1)(n+2)(2(n+1)+1)}{6}$, जो है $ P(n+1)$. प्रेरण द्वारा, $ P(n)$
सभी के लिए धारण करता है $ n $.
\end{solution}

\begin{solution}{exo:g12:seq:2}
$ a_{n+1} - a_n = \frac{n+2}{n+1} - \frac{n+1}{n}
= \frac{n(n+2) - (n+1)^2}{n(n+1)} = \frac{-1}{n(n+1)} < 0$: $(a_n)$ है
सख्ती से \omterm{def:g10:functions:variations}{घटते}.

$(b_n)$ सकारात्मक शर्तें हैं और
$\frac{b_{n+1}}{b_n} = \frac{2^{n+1}}{n+1}\cdot\frac{n}{2^n} =
\frac{2n}{n+1} \geq 1 \iff 2n \geq n+1 \iff n \geq 1$: $(b_n)$ है \omterm{def:g12:seq:monotonic}{की बढ़ती}
(सख्ती से के लिए $ n \geq 2$).

$ c_{n+1} - c_n = (n+1)^2 - 10(n+1) - n^2 + 10n = 2n - 9$, जो नकारात्मक है
के लिए $ n \leq 4$ और के लिए सकारात्मक $ n \geq 5$: $(c_n)$ तक घट जाती है
$ c_5 = -25$, इसका \omterm{def:g10:functions:extrema}{न्यूनतम}, फिर बढ़ जाता है। यह नहीं है एकरस.
\end{solution}

\begin{solution}{exo:g12:seq:3}
कारक प्रधान पद:
\[
u_n = \frac{n^2(3 - 1/n + 1/n^2)}{n^2(2 + 5/n^2)} \xrightarrow[n\to+\infty]{} \frac{3}{2}.
\]
संयुग्म द्वारा गुणा करें:
\[
v_n = \frac{(n+1) - n}{\sqrt{n+1} + \sqrt{n}} = \frac{1}{\sqrt{n+1}+\sqrt{n}}
\xrightarrow[n\to+\infty]{} 0.
\]
अंश और हर को इससे विभाजित करें $3^n $:
\[
w_n = \frac{(2/3)^n - 1}{1 + (1/3)^n} \xrightarrow[n\to+\infty]{} \frac{0-1}{1+0} = -1,
\]
का उपयोग करते हुए $\lim q^n = 0$ के लिए $\abs{q} < 1$.
\end{solution}

\begin{solution}{exo:g12:seq:4}
$ u_n = 5 + 3n \to +\infty $ और $ v_n = 8 \cdot (1/2)^n = 2^{3-n} \to 0$.
योग हैं
\[
\sum_{k=0}^{n} u_k = (n+1)\,\frac{5 + (5+3n)}{2} = \frac{(n+1)(10+3n)}{2}
\xrightarrow[n\to+\infty]{} +\infty,
\]
\[
\sum_{k=0}^{n} v_k = 8\,\frac{1 - (1/2)^{n+1}}{1 - 1/2}
= 16\left(1 - \left(\tfrac{1}{2}\right)^{n+1}\right)
\xrightarrow[n\to+\infty]{} 16 .
\]
\end{solution}

\begin{solution}{exo:g12:seq:5}
तब से $-1 \leq \cos n \leq 1$,
\[
\frac{n-1}{n+1} \leq \frac{n + \cos n}{n+1} \leq 1,
\]
और $\frac{n-1}{n+1} \to 1$, तो सीमा है $1$ निचोड़ प्रमेय द्वारा.

दूसरी सीमा के लिए लिखें
\[
0 \leq \frac{n!}{n^n}
= \frac{1}{n}\cdot\frac{2}{n}\cdots\frac{n}{n}
\leq \frac{1}{n},
\]
क्योंकि हर कारक $\frac{k}{n}$ साथ $2 \leq k \leq n $ \omterm{def:g10:functions:extrema}{अधिकतम} है $1$.
तब से $\frac1n \to 0$, निचोड़ प्रमेय देता है $\frac{n!}{n^n} \to 0$. (द
सुझाव दिया \omterm{def:g12:seq:arith-geom}{ज्यामितिक} बाउंड भी काम करता है: प्रत्येक कारक के साथ $ k \leq n/2$ \omterm{def:g10:functions:extrema}{अधिकतम} है
$\frac12$, मजबूत बंधन दे रहा है $(1/2)^{\floor{n/2}}$.)
\end{solution}

\begin{solution}{exo:g12:seq:6}
\emph{1.} $ u_0 = 0 \in \intcc{0}{2}$. अगर $0 \leq u_n \leq 2$, तब
$2 \leq u_n + 2 \leq 4$, इसलिए $\sqrt{2} \leq u_{n+1} \leq 2$; विशेष रूप से
$0 \leq u_{n+1} \leq 2$. प्रेरण द्वारा संपत्ति सभी के लिए धारण करती है $ n $.

\emph{2.} $ u_{n+1} - u_n = \sqrt{u_n + 2} - u_n $. के लिए $ x \in \intcc{0}{2}$,
$\sqrt{x+2} \geq x \iff x + 2 \geq x^2 \iff (2-x)(x+1) \geq 0$, कौन सा सही है।
इस तरह $(u_n)$ है \omterm{def:g12:seq:monotonic}{की बढ़ती}.

\emph{3.} \omterm{def:g12:seq:monotonic}{की बढ़ती} और \omterm{def:g12:seq:bounded}{ऊपर से घिरा हुआ} द्वारा $2$, $(u_n)$ \omterm{def:g12:seq:limit}{अभिसरण} कुछ करने के लिए
$\ell \in \intcc{0}{2}$. में सीमा तक गुजरना $ u_{n+1} = \sqrt{u_n + 2}$
(वो नक्शा $ x \mapsto \sqrt{x+2}$ सतत है) देता है $\ell = \sqrt{\ell + 2}$,
इसलिए $\ell^2 - \ell - 2 = 0$, यानी\ $\ell \in \{-1, 2\}$. तब से $\ell \geq 0$,
$\lim u_n = 2$.
\end{solution}

\begin{solution}{exo:g12:seq:7}
\emph{1.} दो खुराकों के बीच, $40\%$ दवा का सफाया हो जाता है, इसलिए
मात्रा $ u_n $ बन जाता है $0.6\,u_n $; अगली खुराक जोड़ता है $1$ इकाई:
$ u_{n+1} = 0.6\,u_n + 1$.

\emph{2.} $ v_{n+1} = u_{n+1} - 2.5 = 0.6\,u_n + 1 - 2.5 = 0.6(u_n - 2.5)
= 0.6\,v_n $: $(v_n)$ है \omterm{def:g12:seq:arith-geom}{ज्यामितिक} अनुपात के साथ $0.6$ और पहला कार्यकाल
$ v_0 = 1 - 2.5 = -1.5$. इस तरह $ v_n = -1.5 \times 0.6^n $ और
\[
u_n = 2.5 - 1.5 \times 0.6^n .
\]

\emph{3.} तब से $0.6^n \to 0$, $ u_n \to 2.5$: दवा की मात्रा स्थिर हो जाती है
पर $2.5$ इकाइयाँ।
\end{solution}

\begin{solution}{exo:g12:seq:8}
\emph{1.} $ u_0 = 3 > 1$. अगर $ u_n > 1$, तब $ u_n + 2 > 0$ और
\[
u_{n+1} - 1 = \frac{4u_n - 1 - u_n - 2}{u_n + 2} = \frac{3(u_n - 1)}{u_n + 2} > 0 .
\]
प्रेरण द्वारा, $ u_n > 1$ सभी के लिए $ n $ (खास तरीके से $ u_n + 2 \neq 0$, इतना \omterm{def:g12:seq:sequence}{अनुक्रम} अच्छी तरह से परिभाषित है)।

\emph{2.} उपरोक्त पहचान का उपयोग करते हुए,
\[
v_{n+1} = \frac{1}{u_{n+1} - 1} = \frac{u_n + 2}{3(u_n - 1)}
= \frac{(u_n - 1) + 3}{3(u_n - 1)} = \frac{1}{3} + v_n .
\]
इसलिए $(v_n)$ है \omterm{def:g12:seq:arith-geom}{अंकगणित} साथ \omterm{def:g11:seq:arithmetic}{सामान्य अंतर} $\frac13$ और
$ v_0 = \frac{1}{u_0 - 1} = \frac12$.

\emph{3.} $ v_n = \frac12 + \frac{n}{3}$, इस तरह
$ u_n = 1 + \frac{1}{v_n} = 1 + \frac{6}{3 + 2n}$. तब से $ v_n \to +\infty $,
$ u_n \to 1$.
\end{solution}

\begin{solution}{exo:g12:seq:9}
\emph{1.} $ H_{2n} - H_n = \sum_{k=n+1}^{2n} \frac{1}{k}$ का योग है $ n $
शर्तें, प्रत्येक कम से कम $\frac{1}{2n}$; इस तरह
$ H_{2n} - H_n \geq n \cdot \frac{1}{2n} = \frac12$.

\emph{2.} पर प्रेरण द्वारा $ k $: $ H_{2^0} = H_1 = 1 \geq 1$. अगर
$ H_{2^k} \geq 1 + \frac{k}{2}$, फिर बिंदु 1 को साथ में लागू करें $ n = 2^k $,
\[
H_{2^{k+1}} \geq H_{2^k} + \frac12 \geq 1 + \frac{k+1}{2}.
\]
 \omterm{def:g12:seq:sequence}{अनुक्रम} $(H_n)$ है \omterm{def:g12:seq:monotonic}{की बढ़ती} (प्रत्येक चरण जोड़ता है $\frac{1}{n+1} > 0$) और
परिणाम $ H_{2^k}$ असीमित है, इसलिए $(H_n)$ क्या नहीं है \omterm{def:g12:seq:bounded}{ऊपर से घिरा हुआ}.
\omterm{def:g12:seq:monotonic}{की बढ़ती} और असीमित, इसकी प्रवृत्ति होती है $+\infty $
(\cref{thm:g12:seq:monotone}).
\end{solution}

\begin{solution}{exo:g12:seq:10}
\emph{1.}  \omterm{def:g12:seq:sequence}{अनुक्रम} $ d_n = b_n - a_n $ संतुष्ट
$ d_{n+1} - d_n = (b_{n+1} - b_n) - (a_{n+1} - a_n) \leq 0$, इसलिए $(d_n)$ है
\omterm{def:g10:functions:variations}{घटते}; तब से $ d_n \to 0$, हम पाते हैं $ d_n \geq 0$ सभी के लिए $ n $ (ए \omterm{def:g10:functions:variations}{घटते}
\omterm{def:g12:seq:sequence}{अनुक्रम} एक नकारात्मक पद हमेशा के लिए इसके नीचे रहेगा, जिससे इसे रोका जा सकेगा
सीमा $0$). इस तरह $ a_n \leq b_n $.

\emph{2.} से $ a_n \leq b_n \leq b_0$, द \omterm{def:g12:seq:monotonic}{की बढ़ती} \omterm{def:g12:seq:sequence}{अनुक्रम} $(a_n)$ है
\omterm{def:g12:seq:bounded}{ऊपर से घिरा हुआ}, इसलिए यह \omterm{def:g12:seq:limit}{अभिसरण} कुछ करने के लिए $\ell $. उसी प्रकार $(b_n)$, \omterm{def:g10:functions:variations}{घटते}
और नीचे बंधा हुआ द्वारा $ a_0$, \omterm{def:g12:seq:limit}{अभिसरण} कुछ करने के लिए $\ell'$. तब
$\ell' - \ell = \lim (b_n - a_n) = 0$, इसलिए $\ell = \ell'$.

\emph{3.} $(a_n)$ है (कड़ाई से) \omterm{def:g12:seq:monotonic}{की बढ़ती} तब से
$ a_{n+1} - a_n = \frac{1}{(n+1)!} > 0$. के लिए $(b_n)$,
\[
b_{n+1} - b_n = \frac{1}{(n+1)!} + \frac{1}{(n+1)(n+1)!} - \frac{1}{n\,n!}
= \frac{n(n+1) + n - (n+1)^2}{n(n+1)(n+1)!}
= \frac{-1}{n(n+1)(n+1)!} < 0 ,
\]
इसलिए $(b_n)$ है \omterm{def:g10:functions:variations}{घटते}. अंत में $ b_n - a_n = \frac{1}{n\,n!} \to 0$. दो दृश्यों आसन्न हैं, इसलिए एक सामान्य सीमा पर एकत्रित होते हैं।
\end{solution}

\begin{solution}{pb:g12:seq:1}
\textbf{1.} के लिए सच है $ n = 1$ ($1 = 1^2$). अगर
$1 + 3 + \dots + (2n - 1) = n^2$, फिर अगला \omterm{def:g11:func:parity}{विषम} \omterm{def:g10:vectors:sum}{जोड़}ें
संख्या: $ n^2 + (2n + 1) = (n + 1)^2$: आनुवंशिकता. प्रेरण द्वारा,
सभी के लिए सच है $ n \geq 1$.

\textbf{2.} $2^0 = 1 > 0$. अगर $2^n > n $, तब
$2^{n+1} = 2 \cdot 2^n > 2n \geq n + 1$ के लिए $ n \geq 1$ (और
$ n = 0$ सीधे जांचें): आनुवंशिकता, किया गया।

\textbf{3.} $ n = 0$: $1 \geq 1$. अगर
$(1 + x)^n \geq 1 + nx $, गुणा करके $1 + x \geq 1 > 0$:
$(1 + x)^{n+1} \geq (1 + nx)(1 + x) = 1 + (n + 1)x + nx^2
\geq 1 + (n + 1)x $.

\textbf{4.} से कदम $ n = 1$ को $ n = 2$: दो के बीच
मार्बल्स, ``पहला $ n $'' और ``आखिरी $ n $''दो हैं
\emph{विभिन्न करना} सिंगल मार्बल्स --- कोई सामान्य मार्बल ब्रिज नहीं
दो समूह, इसलिए कोई भी चीज़ उनके रंगों को सहमत होने के लिए बाध्य नहीं करती। द
आनुवंशिकता तर्क के लिए चुपचाप दो समूहों को ओवरलैप करने की आवश्यकता होती है,
केवल से सच है $ n \geq 2$; बेस केस के साथ $ n = 1$ श्रृंखला
कभी शुरू नहीं होता.

\textbf{5.} $4^0 - 1 = 0 = 3 \times 0$. अगर $4^n - 1 = 3k $,
तब $4^{n+1} - 1 = 4(4^n - 1) + 3 = 3(4k + 1)$: आनुवंशिकता.

\textbf{6.} $ x_1 = \frac32$, $ x_2 = \frac{17}{12}$,
$ x_3 = \frac{577}{408}$.

\textbf{7.} $ x_{n+1}^2 - 2 = \frac{(x_n^2 + 2)^2 - 8x_n^2}
{4x_n^2} = \frac{(x_n^2 - 2)^2}{4 x_n^2}$: a के ऊपर एक वर्ग
सकारात्मक, इसलिए $\geq 0$, और $> 0$ जब कभी भी $ x_n^2 \neq 2$.
प्रेरण: $ x_0 = 2 > 0$ साथ $ x_0^2 = 4 > 2$; अगर $ x_n > 0$
और $ x_n^2 > 2$, तब $ x_{n+1}$ (सकारात्मकता का औसत) है
सकारात्मक और $ x_{n+1}^2 - 2 > 0$.

\textbf{8.} $ x_{n+1} - x_n = \frac{2 - x_n^2}{2x_n} < 0$ द्वारा
प्रश्न 7: सख्ती से \omterm{def:g10:functions:variations}{घटते}.

\textbf{9.} घटाना और नीचे बंधा हुआ (द्वारा $1$, तब से
$ x_n^2 > 2 > 1$ और $ x_n > 0$): मोनोटोन \omterm{def:g12:seq:limit}{अभिसरण} द्वारा
प्रमेय, $(x_n)$ \omterm{def:g12:seq:limit}{अभिसरण} कुछ करने के लिए $ L \geq 1$.

\textbf{10.} सीमाएँ बीजगणित का सम्मान करती हैं: से
$ x_{n+1} = \frac12\left(x_n + \frac{2}{x_n}\right)$ और
$ x_n \to L \geq 1 > 0$:
$ L = \frac12\left(L + \frac2L\right)$, इसलिए $ L^2 = 2$ और, $ L $
सकारात्मक होना, $ L = \sqrt2$. फैसला: \omterm{def:g12:seq:limit}{अभिसरण} साबित हुआ,
सीमा की पहचान की गई --- बगुला बाइज्जत बरी हो गया।

\textbf{11.} $ x_{n+1} - \sqrt2 = \frac{x_n^2 - 2\sqrt2\,x_n +
2}{2x_n} = \frac{(x_n - \sqrt2)^2}{2x_n}$: बिल्कुल
$ e_{n+1} = \frac{e_n^2}{2x_n}$, और $ x_n > \sqrt2$ देता है
$ e_{n+1} \leq \frac{e_n^2}{2\sqrt2}$. चुकता त्रुटि: प्रत्येक चरण
जैसा कि देखा गया है, सही दशमलवों की संख्या को दोगुना कर देता है
ग्रेड 9.

\textbf{12.} $ e_0 \approx 0.5858$, $ e_1 \approx 0.0858$,
$ e_2 \approx 0.00245$, $ e_3 \approx 2.1 \times 10^{-6}$.
अनुपात $\frac{e_1}{e_0^2} \approx 0.25 = \frac{1}{2x_0}$;
$\frac{e_2}{e_1^2} \approx 0.333 = \frac{1}{2x_1}$;
$\frac{e_3}{e_2^2} \approx 0.353 \approx \frac{1}{2x_2}$: द
कार्रवाई में प्रमेय.

\textbf{13.} $ H_{2^{18}} \geq 1 + 9 = 10$: के बारे में
$260\,000$ शर्तें ($2^{18} = 262\,144$) केवल पास होने के लिए $10$ ---
एक क्रॉल पर विचलन (और) $ H_n > 100$ अधिक शर्तों की आवश्यकता होगी
किसी भी पुस्तकालय में परमाणुओं की तुलना में)।

\textbf{14.} $ S_n = 2 - \frac{1}{2^n}$ (\omterm{def:g12:seq:arith-geom}{ज्यामितिक} योग), और
$\frac{1}{2^n} \to 0$ (\cref{thm:g12:seq:geometric}), इसलिए
$ S_n \to 2$: अंतहीन रूप से काटी जाने वाली चॉकलेट बार,
बिना कभी पहुंचे, पूरा --- अब आधिकारिक में
मर्यादा की भाषा.

\textbf{15.} के लिए $ k \geq 2$:
$\frac{1}{k^2} \leq \frac{1}{k(k-1)} = \frac{1}{k-1} -
\frac1k $, इसलिए
$ S_n \leq 1 + \left(1 - \frac1n\right) < 2$: \omterm{def:g12:seq:monotonic}{की बढ़ती} और
\omterm{def:g12:seq:bounded}{ऊपर से घिरा हुआ}, इसलिए \omterm{def:g12:seq:limit}{अभिसरण} (मोनोटोन \omterm{def:g12:seq:limit}{अभिसरण})। यूलर
बाद में सीमा का नाम दिया गया: $\frac{\pi^2}{6}$.

\textbf{16.} शर्तें प्रवृत्त होती हैं $0$ है \emph{ज़रूरी} के लिए
व्यवस्थित करने के लिए योग लेकिन कुछ भी तय नहीं करता: हार्मोनिक शर्तें
$\frac1n \to 0$ फिर भी रकम उड़ जाती है; शर्तें
$\frac{1}{n^2} \to 0$ और रकम एकत्रित हो जाती है। कैसे \emph{तेज़}
शर्तें मरना ही संपूर्ण प्रश्न है --- श्रृंखला का \omterm{def:g11:vect:det}{सिद्ध}ांत,
विश्वविद्यालय खंडों में निर्मित।

\textbf{17.} $ a_1 = \sqrt2 \approx 1.41421$, $ b_1 = 1.5$;
$ a_2 \approx 1.45648$, $ b_2 \approx 1.45711$: दो पुनरावृत्तियाँ
पहले से ही तीन दशमलव पर सहमत हैं --- चौंका देने वाली गति।

\textbf{18.} $ b_{n+1} - a_{n+1} = \frac{a_n + b_n}{2} -
\sqrt{a_n b_n} = \frac{(\sqrt{b_n} - \sqrt{a_n})^2}{2} \geq
0$: का \omterm{def:g11:stat:mean}{अर्थ} है आदेशित रहना। $(a_n)$ बढ़ता है:
$ a_{n+1} = \sqrt{a_n b_n} \geq \sqrt{a_n \cdot a_n} = a_n $;
$(b_n)$ सममित रूप से घटता है।

\textbf{19.} $\frac{b_{n+1} - a_{n+1}}{b_n - a_n} =
\frac{(\sqrt{b_n} - \sqrt{a_n})^2}
{2(\sqrt{b_n} - \sqrt{a_n})(\sqrt{b_n} + \sqrt{a_n})}
= \frac{\sqrt{b_n} - \sqrt{a_n}}{2(\sqrt{b_n} + \sqrt{a_n})}
\leq \frac12$: अंतर कम से कम आधा हो, तो $ b_n - a_n \to 0$;
प्रश्न 18 के साथ, द दृश्यों आसन्न हैं और एक सीमा साझा करते हैं
$ M(1, 2)$.

\textbf{20.} पुनरावृत्ति तीन देता है
$ a_3 \approx b_3 \approx 1.456791$: $ M(1, 2) \approx
1.456791$ क्रैंक के तीन घुमावों में (अंतर मोटे तौर पर
\emph{चौकों}, बगुला की तरह)। समस्या की स्पीड रैंकिंग
दृश्यों, सबसे धीमे से सबसे तेज़: हार्मोनिक योग (हिमनदी
विचलन), द \omterm{def:g12:seq:arith-geom}{ज्यामितिक} योग (प्रत्येक चरण में त्रुटि आधी हो गई),
हेरोन और एजीएम (प्रत्येक चरण में त्रुटि वर्ग) --- और यह था
एजीएम की अलौकिक गति ने गॉस को बताया कि उसने एक नई नस पर प्रहार किया है
विश्लेषण का.
\end{solution}
