\section*{अध्याय \ref{ch:g12:exp} --- घातांक और लघुगणक}

\begin{solution}{exo:g12:exp:1}
$\dfrac{\eu^{3x}\,\eu^{-x+1}}{\eu^{x}} = \eu^{3x - x + 1 - x} = \eu^{x+1}$.

$\ln\!\left(\dfrac{\eu^2\sqrt{\eu}}{\eu^{-3}}\right)
= 2 + \tfrac12 + 3 = \tfrac{11}{2}$.
\end{solution}

\begin{solution}{exo:g12:exp:2}
\emph{(ए)} तय करना $ X = \eu^x > 0$: $ X^2 - 3X + 2 = 0$ देता है $ X = 1$ या $ X = 2$,
दोनों सकारात्मक, इसलिए $ x = 0$ या $ x = \ln 2$.

\emph{(बी)} \omterm{def:g11:func:function}{कार्यक्षेत्र}: $ x - 1 > 0$ और $ x + 2 > 0$, इसलिए $ x > 1$. \omterm{def:g10:algebra:equation}{समीकरण}
बन जाता है $\ln\bigl((x-1)(x+2)\bigr) = \ln 4$, इस तरह $(x-1)(x+2) = 4$,
\emph{यानी} $ x^2 + x - 6 = 0$, इसलिए $ x = 2$ या $ x = -3$. केवल $ x = 2$ में है
\omterm{def:g11:func:function}{कार्यक्षेत्र}: $ S = \{2\}$.
\end{solution}

\begin{solution}{exo:g12:exp:3}
द्वारा विभाजित करें $\eu^x $:
$\dfrac{1 - x^2\eu^{-x}}{1 + \eu^{-x}} \to \dfrac{1-0}{1+0} = 1$, का उपयोग करना
$ x^2 \eu^{-x} \to 0$ (\cref{thm:g12:exp:growth}).

$\dfrac{\ln x}{\sqrt x} \to 0$ द्वारा \cref{prop:g12:exp:lnanalytic} (मामला
$ n = 2$).

$ x^2 \ln x = x \cdot (x \ln x) \to 0 \times 0 = 0$.

$ x - \ln x = x\left(1 - \frac{\ln x}{x}\right) \to +\infty $ तब से
$\frac{\ln x}{x} \to 0$.
\end{solution}

\begin{solution}{exo:g12:exp:4}
$ f'(x) = \eu^{-x} - x\eu^{-x} = (1 - x)\eu^{-x}$, के चिन्ह का $1 - x $:
$ f $ पर बढ़ता है $\intoc{-\infty}{1}$, पर घटता है $\intco{1}{+\infty}$,
एक वैश्विक के साथ \omterm{def:g10:functions:extrema}{अधिकतम} $ f(1) = \eu^{-1}$.

सीमाएँ: जैसे $ x \to +\infty $, $ f(x) = \frac{x}{\eu^x} \to 0$
(\cref{thm:g12:exp:growth}); जैसा $ x \to -\infty $, $ x \to -\infty $ और
$\eu^{-x} \to +\infty $, इसलिए $ f(x) \to -\infty $.

$ f''(x) = -\eu^{-x} - (1-x)\eu^{-x} = (x - 2)\eu^{-x}$, जो साइन बदलता है
$ x = 2$: \omterm{def:g12:deriv:inflection}{विभक्ति बिंदु} पर $\bigl(2,\, 2\eu^{-2}\bigr)$.
\end{solution}

\begin{solution}{exo:g12:exp:5}
होने देना $ g(x) = x - \ln(1+x)$ पर $\intoo{-1}{+\infty}$. तब
$ g'(x) = 1 - \frac{1}{1+x} = \frac{x}{1+x}$, नकारात्मक पर $\intoo{-1}{0}$ और
पर सकारात्मक $\intoo{0}{+\infty}$: $ g $ उसे प्राप्त कर लेता है \omterm{def:g10:functions:extrema}{न्यूनतम} $ g(0) = 0$, इसलिए
$ g \geq 0$ केवल समानता के साथ $0$. इस तरह $\ln(1+x) \leq x $.

इसके साथ इसे लागू करें $ x = \frac1n $:
$ n\ln\left(1 + \frac1n\right) \leq 1$, इसलिए
$\left(1+\frac1n\right)^n = \eu^{\,n\ln(1+1/n)} \leq \eu $.
इसके साथ लगाएं $ x = -\frac{1}{n+1} > -1$:
$\ln\left(1 - \frac{1}{n+1}\right) \leq -\frac{1}{n+1}$, इसलिए
$-(n+1)\ln\left(1 - \frac{1}{n+1}\right) \geq 1$ और
$\left(1 - \frac{1}{n+1}\right)^{-(n+1)} \geq \eu $.
\end{solution}

\begin{solution}{exo:g12:exp:6}
\emph{1.} प्रत्येक वर्ष पूंजी को कई गुना बढ़ा देता है $1.03$: $ C_n = C_0 (1.03)^n $.

\emph{2.} $ C_n \geq 2C_0 \iff (1.03)^n \geq 2 \iff n \ln 1.03 \geq \ln 2
\iff n \geq \frac{\ln 2}{\ln 1.03} \approx 23.45$: के बाद पूंजी दोगुनी हो जाती है
$24$ साल।

\emph{3.} $\frac{70}{3} \approx 23.3$, सटीक के करीब
$\frac{\ln 2}{\ln 1.03}$. नियम इसलिए काम करता है
$\ln(1 + r) \approx r $ छोटे के लिए $ r $, इसलिए
$\frac{\ln 2}{\ln(1+r)} \approx \frac{0.693}{r} \approx \frac{70}{100r}$.
\end{solution}

\begin{solution}{exo:g12:exp:7}
$\eu^{2x} > \eu^{x+1} \iff 2x > x + 1 \iff x > 1$ (घातांक है
सख्ती से \omterm{def:g12:seq:monotonic}{की बढ़ती}): $ S = \intoo{1}{+\infty}$.

\omterm{def:g11:func:function}{कार्यक्षेत्र} दूसरी असमानता का: $ x^2 - 1 > 0$ और $ x + 5 > 0$, \emph{यानी}
$ x \in \intoo{-5}{-1} \cup \intoo{1}{+\infty}$. इस पर  \omterm{def:g11:func:function}{कार्यक्षेत्र}, $\ln $ किया जा रहा है
सख्ती से \omterm{def:g12:seq:monotonic}{की बढ़ती},
\[
\ln(x^2-1) \leq \ln(x+5) \iff x^2 - 1 \leq x + 5 \iff x^2 - x - 6 \leq 0
\iff x \in \intcc{-2}{3}.
\]
के साथ अन्तर्विभाजक \omterm{def:g11:func:function}{कार्यक्षेत्र}: $ S = \intco{-2}{-1} \cup \intoc{1}{3}$.
\end{solution}

\begin{solution}{exo:g12:exp:8}
\emph{1.} $ f'(x) = \dfrac{\eu^x x - \eu^x}{x^2} = \dfrac{(x-1)\eu^x}{x^2}$,
पर नकारात्मक $\intoo{0}{1}$, सकारात्मक पर $\intoo{1}{+\infty}$: \omterm{def:g10:functions:extrema}{न्यूनतम}
$ f(1) = \eu $. सीमाएँ: $ f \to +\infty $ पर $0^+$ (अंशांक $\to 1$,
हर $\to 0^+$) और कम से $+\infty $ (\cref{thm:g12:exp:growth}).

\emph{2.} के लिए $ x > 0$, $\eu^x = kx \iff f(x) = k $. से \omterm{def:g10:functions:table}{विविधता तालिका}
($+\infty \searrow \eu \nearrow +\infty $, निरंतर और सख्ती से एकरस
प्रत्येक पक्ष पर): कोई समाधान नहीं $ k < \eu $; बिल्कुल एक ($ x = 1$) के लिए
$ k = \eu $; बिल्कुल दो के लिए $ k > \eu $ (एक इंच $\intoo{0}{1}$, एक में
$\intoo{1}{+\infty}$, प्रत्येक पर आक्षेप प्रमेय द्वारा \omterm{def:g10:numbers:interval}{अंतराल}).
\end{solution}

\begin{solution}{exo:g12:exp:9}
\emph{1.} की पहली असमानता से प्रत्यक्ष \cref{exo:g12:exp:5}:
$ u_n = \eu^{\,n\ln(1+1/n)} \leq \eu^1 = \eu $.

\emph{2.} लिखना
\[
\ln u_n = n \ln\!\left(1 + \frac1n\right)
= \frac{\ln\!\left(1 + \frac1n\right) - \ln 1}{\frac1n},
\]
का एक अंतर भागफल $\ln $ बिंदु पर $1$ वृद्धि के साथ
$ h = \frac1n \to 0$. तब से $\ln $ है \omterm{def:g12:deriv:derivative}{विभेदक} पर $1$ साथ यौगिक
$1$, $\ln u_n \to 1$. द्वारा निरंतरता का $\exp $
(\cref{prop:g12:limcont:seqcont}), $ u_n = \eu^{\ln u_n} \to \eu^1 = \eu $.
\end{solution}

\begin{solution}{pb:g12:exp:1}
\textbf{1.} $ x = \frac{\ln 5}{2}$. \quad
$3x - 1 = \eu^2$: $ x = \frac{\eu^2 + 1}{3}$. \quad साथ
$ u = \eu^x $: $ u^2 - 3u + 2 = (u - 1)(u - 2) = 0$: $ x = 0$ या
$ x = \ln 2$.

\textbf{2.} $\frac{3\ln 2}{\ln 2} = 3$; \quad
$3 + \frac12 = \frac72$; \quad $\frac52$.

\textbf{3.} $ f'(x) = \eu^x - 1$: पहले नकारात्मक $0$, सकारात्मक
बाद: \omterm{def:g10:functions:extrema}{न्यूनतम} $ f(0) = 0$, इसलिए $ f \geq 0$ हर जगह:
$\eu^x \geq 1 + x $. स्थानापन्न $ x = \ln(1 + u)$:
$1 + u \geq 1 + \ln(1 + u)$, यानी\ $\ln(1 + u) \leq u $.

\textbf{4.} $ x^2\eu^{-x} = \frac{x^2}{\eu^x} \to 0$ और
$\frac{\ln x}{x} \to 0$: घातांक शक्तियों, शक्तियों को कुचलते हैं
कुचलना लघुगणक. और $\frac{\eu^x - 1}{x} =
\frac{\eu^x - \eu^0}{x - 0} \to \exp'(0) = 1$.

\textbf{5.} $ f'(x) = \ln x + 1$: शून्य पर $\eu^{-1}$, नकारात्मक
पहले, सकारात्मक बाद में: \omterm{def:g10:functions:extrema}{न्यूनतम}
$ f\!\left(\frac1\eu\right) = -\frac1\eu $; और
$ x \ln x \to 0$ पर $0^+$: वक्र मूल को छोड़ देता है, नीचे गिर जाता है
$-\frac1\eu $, और चढ़ जाता है।

\textbf{6.} $ n = \frac{\ln 2}{\ln 1.03} \approx 23.4$ साल।

\textbf{7.} एकदम सही: $23.4$, $11.9$, $8.0$ साल; 72 का नियम:
$24$, $12$, $8$. तब से $\ln(1+r) \approx r $,
$\frac{\ln 2}{\ln(1+r)} \approx \frac{0.693}{r}$, \omterm{def:g11:stat:mean}{अर्थ}ात\
$\frac{69.3}{\text{rate in }\%}$; बैंकर्स तक चक्कर लगाते हैं $72$
क्योंकि यह खूबसूरती से विभाजित होता है $2, 3, 4, 6, 8, 9, 12$ ---
मानसिक \omterm{def:g12:seq:arith-geom}{अंकगणित} सटीकता के दशमलव को मात देता है।

\textbf{8.} $2^{t/20} = \eu^{t \ln 2 / 20}$:
$\lambda = \frac{\ln 2}{20}$ प्रति मिनट. बाद
$24 \times 60 = 1440$ मिनट: $2^{72} \approx 4.7 \times
10^{21}$ बैक्टीरिया---पृथ्वी पर रेत के कणों से भी अधिक,
एक दिन में एक सेल से. मॉडल केवल भोजन के समय ही ईमानदार है
और अंतिम स्थान: वास्तविक विकास लॉजिस्टिक एस-वक्र में झुकता है
(महामारीविज्ञानी का वक्र \cref{pb:g12:deriv:1}).

\textbf{9.} 23:00 बजे ($8$ घंटे):
$100 \times 2^{-8/5} \approx 33$ एमजी. नीचे $10$ एमजी:
$2^{-t/5} < 0.1$ देता है $ t > 5\,\frac{\ln 10}{\ln 2} \approx
16.6$ ज: अगली सुबह लगभग 7:40 बजे --- एस्प्रेसो
तीन की एक लंबी पूँछ है।

\textbf{10.} वार्षिक: $2$. महीने के:
$\left(1 + \frac{1}{12}\right)^{12} \approx 2.613$. दैनिक:
$\approx 2.715$. छत: $\eu = 2.71828\ldots $
(\cref{exo:g12:exp:9}) --- समझौता लगातार, लालच
\omterm{def:g12:seq:limit}{अभिसरण}.

\textbf{11.} यदि मामले तेजी से बढ़ते हैं, $ c(t) = c_0
\eu^{\lambda t}$, तब $\ln c(t) = \ln c_0 + \lambda t $: ए
की सीधी रेखा \omterm{def:g10:reffunc:affine}{ढलान} $\lambda $ --- विकास दर. ए
लॉग प्लॉट पर सीधा खिंचाव \emph{है} घातीय
चरण, और इसकी तीव्रता ही महामारी की गति है।

\textbf{12.} $120 - 60 = 60$ डीबी का मतलब है
$10\log_{10}(I_2/I_1) = 60$: संगीत कार्यक्रम है
$10^6$ ---लाख गुना ---से भी अधिक तीव्र
बातचीत.

\textbf{13.} आयाम: $10^{7-5} = 100$. ऊर्जा:
$100^{3/2} = 1000$: दो परिमाण बिंदु तीन आदेशों को छिपाते हैं
ऊर्जा में परिमाण.

\textbf{14.} $10^{6.5 - 2} = 10^{4.5} \approx 32\,000$: नींबू
जूस दूध से तीस हजार गुना अधिक अम्लीय होता है--पीएच
रसायन शास्त्र की खाइयों को पॉकेट स्केल में संपीड़ित करता है।

\textbf{15.} $\log_2\!\frac{4186}{27.5} =
\frac{\ln(4186/27.5)}{\ln 2} \approx 7.25$: एक पियानो बस फैलता है
सात सप्तक से अधिक. इन्द्रियाँ प्रतिक्रिया करती हैं \emph{अनुपात} उत्तेजनाओं का
--- तीव्रता को दोगुना करना एक कदम जैसा लगता है, चाहे कुछ भी हो
प्रारंभिक स्तर --- इसलिए धारणा लघुगणक पर बनी है
पैमाना, और इसके लिए हमने जिन इकाइयों का आविष्कार किया है।

\textbf{16.} के लिए छड़ी रखना $ a $ के साथ अंत-से-अंत तक
के लिए छड़ी की स्थिति $ b $ लंबाई जोड़ता है $\ln a + \ln b $, और
स्नातक उस कुल लंबाई पर बैठता है पढ़ता है
$\eu^{\ln a + \ln b} = ab $: स्लाइड नियम उत्पादों की गणना करता है
जोड़कर लघुगणक --- $\ln(ab) = \ln a + \ln b $
(\cref{prop:g12:exp:lnalg}) बॉक्सवुड में नक्काशी की गई।

\textbf{17.}  \omterm{def:g10:functions:extrema}{न्यूनतम} का $\frac{\eu^x}{x}$ पर
$\intoo{0}{+\infty}$ है $\eu $, पर $ x = 1$: इसका कोई समाधान नहीं
$ k < \eu $, बिल्कुल एक के लिए $ k = \eu $, दो के लिए $ k > \eu $.
\omterm{def:g12:deriv:tangent}{स्पर्शरेखा} जांचें: पर $ a = 1$  \omterm{def:g12:deriv:tangent}{स्पर्शरेखा} को $\eu^x $ है
$ y = \eu + \eu(x - 1) = \eu x $: यह मूल से होकर गुजरता है
--- दहलीज रेखा, वक्र को छूती हुई $(1, \eu)$.

\textbf{18.} $ n \ln\!\left(1 - \frac1n\right) =
\frac{\ln(1 - 1/n)}{-1/n} \times (-1) \to -1$, इसलिए
$\left(1 - \frac1n\right)^n \to \eu^{-1} \approx 0.368$: द
$0.999^{1000} \approx 0.368$ का \cref{pb:g11:binom:1},
समझाया --- लॉटरी का निकट-चूक स्थिरांक है
$\frac1\eu $.

\textbf{19.} दोनों ``अनेक स्वतंत्र'' की सीमा आकृति हैं
घटनाएँ, प्रत्येक व्यक्तिगत रूप से असंभावित'': \omterm{def:g10:proba:distribution}{संभावना} है कि
\emph{कोई नहीं} का $ n $ आकार की \omterm{def:g10:proba:distribution}{संभावना} $\frac1n $ आग लग जाती है
$\eu^{-1}$, और सचिव नियम इसकी अस्वीकृति विंडो को ट्यून करता है
ताकि सफलता उसी स्थिरांक पर \omterm{pb:g11:quad:1}{केंद्र}ित रहे।
$\frac1\eu \approx 0.368$.

\textbf{20.} कंपाउंडिंग की छत; अद्वितीय आधार जिसका
\omterm{def:g12:deriv:tangent}{स्पर्शरेखा} पर $0$ है \omterm{def:g10:reffunc:affine}{ढलान} $1$ (यही कारण है कि कैलकुलस इसे पसंद करता है);
वह विकास जो हर शक्ति से आगे निकल जाता है; स्थिरांक कहाँ
निकट-चूक और इष्टतम रोक निपटान; और यह लोगारित्म ---
गुणन \omterm{def:g10:vectors:sum}{जोड़} बन गया, दुनिया की सबसे जंगली श्रेणियाँ
एक रूलर पर मुड़ा हुआ।
\end{solution}
