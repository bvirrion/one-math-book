\section*{अध्याय \ref{ch:g12:condprob} --- सशर्त संभाव्यता और स्वतंत्रता}

\begin{solution}{exo:g12:condprob:1}
वहाँ हैं $12$ फेस कार्ड, जिनमें से $4$ राजा हैं:
$\pcond{\text{face}}{\text{king}} = \frac{4}{12} = \frac13$.

आज़ादी: $\P(\text{king} \cap \text{heart}) = \frac{1}{52}$ (राजा
दिलों का), और
$\P(\text{king})\,\P(\text{heart}) = \frac{4}{52}\times\frac{13}{52}
= \frac{1}{52}$. बराबर: द घटनाएँ हैं \omterm{def:g12:condprob:indep}{स्वतंत्र}.
\end{solution}

\begin{solution}{exo:g12:condprob:2}
\emph{1.} पहली शाखा: लाल $\frac58$, नीला $\frac38$; दूसरी शाखाएँ
(प्रतिस्थापन के बिना): लाल के बाद, लाल $\frac47$ / नीला $\frac37$; के बाद
नीला, लाल $\frac57$ / नीला $\frac27$.

\emph{2.} $\P(RR) = \frac58 \times \frac47 = \frac{5}{14}$. कुल मिलाकर
\omterm{def:g10:proba:distribution}{संभावना},
\[
\P(\text{2nd red}) = \frac58\cdot\frac47 + \frac38\cdot\frac57
= \frac{20 + 15}{56} = \frac58 .
\]
(पहले ड्रा के समान --- समरूपता से, दूसरी गेंद एक समान रूप से होती है
कलश की यादृच्छिक गेंद।)
\end{solution}

\begin{solution}{exo:g12:condprob:3}
$\P(A) = \frac{6}{36} = \frac16$ (छह \omterm{def:g10:vectors:sum}{जोड़}े का योग है $7$),
$\P(B) = \frac16$, और $ A \cap B = \{(3,4)\}$ है \omterm{def:g10:proba:distribution}{संभावना}
$\frac{1}{36} = \P(A)\P(B)$: $ A $ और $ B $ हैं \omterm{def:g12:condprob:indep}{स्वतंत्र}. (योग $7$ है
विशेष: पहला पासा जो भी हो, दूसरे का ठीक एक मान उसे देता है।)

$\P(C) = \frac{5}{36}$ और $ B \cap C = \{(3,3)\}$:
$\P(B \cap C) = \frac{1}{36} \neq \frac16 \times \frac{5}{36}
= \frac{5}{216}$. नहीं \omterm{def:g12:condprob:indep}{स्वतंत्र}.
\end{solution}

\begin{solution}{exo:g12:condprob:4}
\emph{1.} कुल \omterm{def:g10:proba:distribution}{संभावना} मशीन द्वारा विभाजन के साथ:
\[
\P(D) = 0.5\times0.01 + 0.3\times0.02 + 0.2\times0.04
= 0.005 + 0.006 + 0.008 = 0.019 = 1.9\% .
\]

\emph{2.} बेयस:
$\pcond{D}{M_3} = \dfrac{0.008}{0.019} = \dfrac{8}{19} \approx 0.42$. द
उत्पादन का केवल पाँचवाँ भाग उत्पादित करने वाली मशीनें अधिक जिम्मेदार होती हैं $40\%$ का
दोष.
\end{solution}

\begin{solution}{exo:g12:condprob:5}
शर्त यह है
\[
\frac{0.99\,p}{0.99\,p + 0.05\,(1-p)} \geq 0.9 .
\]
हर सकारात्मक है, इसलिए यह पढ़ता है
$0.99p \geq 0.891p + 0.045 - 0.045p $, \emph{यानी} $0.144\,p \geq 0.045$,
\emph{यानी}
\[
p \geq \frac{0.045}{0.144} = 0.3125 .
\]
परीक्षा ही पहुंचती है $90\%$ निश्चितता केवल तभी जब रोग पहले से ही प्रभावित हो
से भी अधिक $31\%$ परीक्षण की गई जनसंख्या का --- यही कारण है कि बड़े पैमाने पर स्क्रीनिंग की जाती है
दुर्लभ बीमारियों के लिए पुष्टिकरण परीक्षणों की आवश्यकता होती है।
\end{solution}

\begin{solution}{exo:g12:condprob:6}
\emph{1.} तीन दृश्यों समझना $ E $ (एचएचटी, एचटीएच, टीएचएच), प्रत्येक \omterm{def:g10:proba:distribution}{संभावना}
$ p^2(1-p)$ द्वारा स्वतंत्रता: $\P(E) = 3p^2(1 - p)$.

\emph{2.} $ f(p) = 3p^2 - 3p^3$ है $ f'(p) = 6p - 9p^2 = 3p(2 - 3p)$,
सकारात्मक फिर नकारात्मक $\intoo{0}{1}$: \omterm{def:g10:functions:extrema}{अधिकतम} पर $ p = \frac23$, कहाँ
$\P(E) = 3\cdot\frac49\cdot\frac13 = \frac49$.
\end{solution}

\begin{solution}{exo:g12:condprob:7}
मान लें कि आपने दरवाज़ा 1 चुना और मेज़बान ने दरवाज़ा 3 खोला (\omterm{def:g11:prob:model}{आयोजन} $ H_3$). साथ
$ B_i $ = ``दरवाजे के पीछे पुरस्कार $ i $'', $\P(B_i) = \frac13$ और
\[
\pcond{B_1}{H_3} = \frac12, \qquad
\pcond{B_2}{H_3} = 1, \qquad
\pcond{B_3}{H_3} = 0
\]
(यदि पुरस्कार आपके दरवाजे के पीछे है, तो मेजबान दरवाजे 2 और 3 के बीच चयन करता है
बेतरतीब ढंग से; यदि यह दरवाजे 2 के पीछे है, तो उसे 3 खोलने के लिए मजबूर किया जाता है)। बेयस:
\[
\pcond{H_3}{B_1}
= \frac{\frac13\cdot\frac12}{\frac13\cdot\frac12 + \frac13\cdot1 + 0}
= \frac{1/6}{1/2} = \frac13,
\qquad
\pcond{H_3}{B_2} = \frac{\frac13}{\frac12} = \frac23 .
\]
साथ रहने से जीत होती है \omterm{def:g10:proba:distribution}{संभावना} $\frac13$, के साथ स्विच करना \omterm{def:g10:proba:distribution}{संभावना}
$\frac23$.
\end{solution}

\begin{solution}{exo:g12:condprob:8}
\emph{1.} बहुमत गलत है जब तीन में से कम से कम दो प्रतियां हों
फ़्लिप:
\[
3\varepsilon^2(1-\varepsilon) + \varepsilon^3
= 3(0.01)(0.9) + 0.001 = 0.028,
\]
से बहुत छोटा $\varepsilon = 0.1$: पुनरावृत्ति कोडिंग कार्य।

\emph{2.} अगर $1$ (के रूप में) भेजा गया था $111$), प्राप्त करना $101$ बिल्कुल एक की आवश्यकता है
पलटें: \omterm{def:g10:proba:distribution}{संभावना} $\varepsilon(1-\varepsilon)^2 = 0.081$. अगर $0$ भेजा था
($000$), दो फ़्लिप की आवश्यकता है: $\varepsilon^2(1-\varepsilon) = 0.009$.
समान पुजारियों के साथ बेयस:
\[
\pcond{101}{\,\text{sent }1} = \frac{0.081}{0.081 + 0.009} = 0.9 .
\]
बहुमत डिकोडिंग ($101 \mapsto 1$) वास्तव में अधिक संभावित अनुमान है।
\end{solution}

\begin{solution}{pb:g12:condprob:1}
\textbf{1.} $\P(\text{red}) = \frac12 \cdot \frac23 + \frac12
\cdot \frac14 = \frac13 + \frac18 = \frac{11}{24}$.

\textbf{2.} $\P(\text{I} \mid \text{red}) =
\dfrac{\frac12 \cdot \frac23}{\frac{11}{24}} =
\dfrac{1/3}{11/24} = \dfrac{8}{11}$.

\textbf{3.} $\P(\text{even} \cap \leq 4) = \P(\{2, 4\}) =
\frac13$ और $\P(\text{even})\P(\leq 4) = \frac12 \cdot
\frac23 = \frac13$: \omterm{def:g12:condprob:indep}{स्वतंत्र} (अप्रत्याशित रूप से!) के लिए
$\leq 3$: $\P(\{2\}) = \frac16$ ख़िलाफ़
$\frac12 \cdot \frac12 = \frac14$: आश्रित.

\textbf{4.}  \omterm{def:g10:proba:operations}{पूरक} का
\omterm{def:g12:condprob:indep}{स्वतंत्र}
घटनाएँ स्वयं हैं
\omterm{def:g12:condprob:indep}{स्वतंत्र}, इसलिए
\[
\P(\text{none}) = 0.1 \times 0.1 = 0.01,
\qquad
\P(\text{at least one}) = 0.99 .
\]

\textbf{5.} $\frac{4}{52} \times \frac{3}{51} =
\frac{1}{221}$.

\textbf{6.} बीमार: $100$; स्वस्थ: $99\,900$. सकारात्मक बातें:
$99$ सच (99\,\% का $100$) और $4\,995$ मिथ्या (5\,\% का
$99\,900$): $5\,094$ सभी में सकारात्मकता.

\textbf{7.} $\P(\text{sick} \mid +) = \frac{99}{5\,094}
\approx 1.9\,\%$: परीक्षण झंडे वाले लोगों की तुलना में कम
पचास में से एक वास्तव में बीमार है।

\textbf{8.} $\dfrac{0.001 \times 0.99}{0.001 \times 0.99 +
0.999 \times 0.05} = \dfrac{0.00099}{0.05094} \approx 0.019$:
गिनती और सूत्र सहमत हैं।

\textbf{9.} अस्पष्ट: $\P(+ \mid \text{sick}) = 99\,\%$
(परीक्षण की विज्ञापित गुणवत्ता) के साथ
$\P(\text{sick} \mid +)$ (रोगी का वास्तविक प्रश्न)। साथ
की एक आधार दर $\frac{1}{1000}$, बीमार इतने दुर्लभ हैं कि
यहां तक कि ए $5\,\%$ विशाल स्वस्थ बहुमत से रिसाव से दलदल हो जाता है
सच्चे सकारात्मक पचास से एक।

\textbf{10.} $\dfrac{0.019 \times 0.99}{0.019 \times 0.99 +
0.981 \times 0.05} \approx 28\,\%$. प्रत्येक \omterm{def:g12:condprob:indep}{स्वतंत्र} सकारात्मक
बाधाओं को उसी कारक से गुणा करता है: साक्ष्य जमा होते हैं,
और तीसरा सकारात्मक अतीत को धक्का देगा $90\,\%$ --- जो है
पुष्टिकारक परीक्षण क्यों मौजूद हैं.

\textbf{11.} $\dfrac{0.05 \times 0.99}{0.05 \times 0.99 +
0.95 \times 0.05} \approx 51\,\%$: उच्च जोखिम वाले समूह पर
वही परीक्षण जानकारीपूर्ण है. नैतिक: स्क्रीन जहां आधार दर
पर्याप्त है; बड़े पैमाने पर स्क्रीनिंग एक दुर्लभ स्थिति का निर्माण करती है
हजारों की संख्या में झूठे अलार्म।

\textbf{12.} अपेक्षित निर्दोष मेल:
$10^7 \times 10^{-6} = 10$. मोटे तौर पर बीच में $11$ मिलान
लोग ($10$ मासूम, $1$ दोषी --- यदि अपराधी अंदर है
शहर), संदिग्ध मिला \emph{अकेले मैच से} है
निर्दोष के साथ \omterm{def:g10:proba:distribution}{संभावना} के बारे में $\frac{10}{11} \approx
91\,\%$.

\textbf{13.} अभियोजक ने उद्धृत किया
$\P(\text{match} \mid \text{innocent}) = 10^{-6}$; अदालत
जरुरत $\P(\text{innocent} \mid \text{match})$ --- प्रश्न
शपथ के तहत 9 की उलझन. (और वास्तविक मामले में वह
इन त्रुटियों को लोकप्रिय बनाने के बाद एक दूसरी गलती ने इसे और जटिल बना दिया:
प्रश्न 14 देखें)

\textbf{14.} अनुभवहीन वर्ग: $\left(\frac{1}{8500}\right)^2
\approx \frac{1}{72 \text{ million}}$. दूसरे के साथ
पहले को देखते हुए त्रासदी पाँच गुना अधिक संभावित:
$\frac{1}{8500} \times \frac{5}{8500} \approx
\frac{1}{14.5 \text{ million}}$ --- पांच गुना बड़ा, और
अभी भी इसे अनदेखा कर रहा हूँ \emph{विकल्प} परिकल्पना होनी चाहिए
बेयस द्वारा भी तौला गया। पाठ: संभावनाओं को गुणा करना
आवश्यकता है स्वतंत्रता, और साझा कारण (आनुवांशिकी,
पर्यावरण) इसे नष्ट कर दो।

\textbf{15.} बेयस \omterm{def:g10:proba:distribution}{संभावना} अनुपात को कई गुना बढ़ा देता है
\emph{सभी} एक पश्च भाग में साक्ष्य: मैच, द
अन्यत्र, फ़ाइबर, प्रत्येक बाधाओं को बदल रहा है। जूरी सदस्य का
वाक्य: \emph{पूछें कि दोनों के अंतर्गत साक्ष्य कितने संभावित हैं
परिकल्पनाएं, और किसी सशर्त को कभी भी उसकी परिकल्पना न समझें
उलटा}.

\textbf{16.} $\dfrac{0.4 \times 0.15}{0.4 \times 0.15 +
0.6 \times 0.005} = \dfrac{0.060}{0.063} \approx 95\,\%$
स्पैम: एक शब्द इसे लगभग सुलझा देता है।

\textbf{17.} जब पांच प्रतियां फेल हो गईं $3$, $4$ या $5$ पलटना:
$\binom53 \varepsilon^3(1-\varepsilon)^2 + \binom54
\varepsilon^4 (1 - \varepsilon) + \varepsilon^5 \approx
0.0086$ --- ख़िलाफ़ $0.028$ तीन प्रतियों के लिए: पुनरावृत्ति
बैंडविड्थ की कीमत पर विश्वसनीयता खरीदता है।

\textbf{18.} यहां तक कि फ़्लिप: $\binom40 0.9^4 + \binom42 0.9^2
0.1^2 + \binom44 0.1^4 = 0.6561 + 0.0486 + 0.0001 = 0.7048$;
और $\frac{1 + 0.8^4}{2} = \frac{1 + 0.4096}{2} = 0.7048$:
चाल सूत्र पूर्वाग्रह को ट्रैक करता है $(1 - 2\varepsilon)$
प्रत्येक रिले के माध्यम से.

\textbf{19.} स्पैमयुक्त शब्द पैक्स में यात्रा करते हैं --- एक संदेश के साथ
``लॉटरी'' में बहुत \omterm{def:g10:proba:distribution}{संभावना} है कि ``जीत'' शामिल हो, इसलिए उनका \omterm{def:g10:vectors:sum}{जोड़}
उत्पाद के दिखावे की तुलना में साक्ष्य कमज़ोर हैं। फ़िल्टर
अभी भी काम करता है क्योंकि \omterm{def:g11:func:reference}{अतिशयोक्ति} आम तौर पर इसमें धक्का देती है
\emph{सही दिशा}: वर्गीकरण के लिए सही पक्ष की आवश्यकता है
का $50\,\%$, सही पश्च भाग नहीं।

\textbf{20.} क्लिनिक: एक दुर्लभ स्थिति एक मजबूत परीक्षा बन जाती है
एक कमजोर निर्णय में --- आधार दर से अद्यतन, फिर
जमा करो. कोर्ट रूम: एक बड़े शहर में अभियोगों का एक दुर्लभ मैच
शहर, आदमी नहीं --- संभावनाएँ निर्णय नहीं हैं।
इनबॉक्स: कई कमजोर सुराग, गुणा, मेल को सॉर्ट करें --- यहां तक कि
भोलेपन से. तीन जाल: पूर्व को भूल जाओ और मूर्ख बनो;
नकली स्वतंत्रता और बहुत मूर्ख बनो; साक्ष्य को टुकड़ों में पढ़ें
और कभी नहीं जान पाओगे कि तुम क्या जानते हो।
\end{solution}
