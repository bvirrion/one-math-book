\section*{Chapter \ref{ch:g12:integ} --- Integration}

\begin{solution}{exo:g12:integ:1}
\[
\int_1^2 \left(3x^2 - \frac{1}{x^2}\right)\dd x
= \left[x^3 + \frac1x\right]_1^2 = \left(8 + \tfrac12\right) - 2 = \frac{13}{2}.
\]
\[
\int_0^{\pi/2} \cos t\,\dd t = \bigl[\sin t\bigr]_0^{\pi/2} = 1 .
\]
With $u = 2t+1$, $\frac{1}{2t+1} = \frac12\,\frac{u'}{u}$:
\[
\int_0^1 \frac{\dd t}{2t+1} = \frac12\bigl[\ln(2t+1)\bigr]_0^1
= \frac{\ln 3}{2}.
\]
\end{solution}

\begin{solution}{exo:g12:integ:2}
$x\,\eu^{x^2} = \frac12\,(x^2)'\,\eu^{x^2}$ has \omterm{def:g12:integ:primitive}{primitives}
$\frac12\eu^{x^2} + c$. The condition $F(0) = 1$ gives $\frac12 + c = 1$,
so
\[
F(x) = \frac{\eu^{x^2} + 1}{2}.
\]
\end{solution}

\begin{solution}{exo:g12:integ:3}
\[
\mu = \frac{1}{\pi}\int_0^\pi \sin t\,\dd t
= \frac{1}{\pi}\bigl[-\cos t\bigr]_0^\pi = \frac{2}{\pi} \approx 0.64 .
\]
Graphically, the rectangle of base $\intcc{0}{\pi}$ and height
$\frac{2}{\pi}$ has the same area as the arch of the \omterm{def:g12:trigo:cossin}{sine} curve.
\end{solution}

\begin{solution}{exo:g12:integ:4}
Take $u' = 1$, $v = \ln t$, so $u = t$, $v' = \frac1t$:
\[
\int_1^{\eu} \ln t\,\dd t = \bigl[t\ln t\bigr]_1^{\eu} - \int_1^{\eu} 1\,\dd t
= \eu - (\eu - 1) = 1 .
\]
Take $u' = \sin t$, $v = t$, so $u = -\cos t$, $v' = 1$:
\[
\int_0^\pi t\sin t\,\dd t = \bigl[-t\cos t\bigr]_0^\pi + \int_0^\pi \cos t\,\dd t
= \pi + 0 = \pi .
\]
\end{solution}

\begin{solution}{exo:g12:integ:5}
The curves intersect where $x^2 = x + 2$, \emph{i.e.} $x \in \{-1, 2\}$, and
on $\intcc{-1}{2}$ the line is above the \omterm{def:g11:quad:parabola}{parabola}. The area is
\[
\int_{-1}^{2} \bigl(x + 2 - x^2\bigr)\dd x
= \left[\frac{x^2}{2} + 2x - \frac{x^3}{3}\right]_{-1}^{2}
= \frac{10}{3} - \left(-\frac{7}{6}\right) = \frac{9}{2}.
\]
\end{solution}

\begin{solution}{exo:g12:integ:6}
\emph{1.} $I = \bigl[\ln(1+t)\bigr]_0^1 = \ln 2$.

\emph{2.}
\[
I_n + I_{n+1} = \int_0^1 \frac{t^n(1 + t)}{1+t}\,\dd t
= \int_0^1 t^n \,\dd t = \frac{1}{n+1}.
\]
On $\intcc{0}{1}$, $0 \leq \dfrac{t^n}{1+t} \leq t^n$, so by comparison
$0 \leq I_n \leq \frac{1}{n+1}$.

\emph{3.} From the recurrence, by a telescoping induction,
\[
I_n = (-1)^n\left(I_0 - \left(1 - \frac12 + \dots + \frac{(-1)^{n-1}}{n}\right)\right),
\]
\emph{i.e.}
$\displaystyle\sum_{k=1}^{n} \frac{(-1)^{k-1}}{k} = \ln 2 - (-1)^n I_n$
(recall $I_0 = \ln 2$). Since $\abs{I_n} \leq \frac{1}{n+1} \to 0$, the
alternating harmonic sum tends to $\ln 2$.
\end{solution}

\begin{solution}{exo:g12:integ:7}
Distance:
\[
d = \int_0^{100} 30\left(1 - \eu^{-t/50}\right)\dd t
= 30\left[t + 50\,\eu^{-t/50}\right]_0^{100}
= 30\left(100 + 50\eu^{-2} - 50\right)
= 1500 + 1500\,\eu^{-2}.
\]
Numerically $d \approx 1500 + 203 = 1703$ m. Average speed:
$\frac{d}{100} = 15\bigl(1 + \eu^{-2}\bigr) \approx 17.0$ m/s.
\end{solution}

\begin{solution}{exo:g12:integ:8}
\emph{1.} $S_n$ is the total area of $n$ rectangles of width $\frac1n$ and
heights $f\left(\frac kn\right)$, $k = 1, \dots, n$, where
$f(t) = \frac{1}{1+t}$: it is the ``right endpoint'' rectangle \omterm{def:g10:numbers:approx}{approximation}
of the area under $f$ on $\intcc{0}{1}$.

\emph{2.} $f$ is \omterm{def:g10:functions:variations}{decreasing}, so on each \omterm{def:g10:numbers:interval}{interval}
$\intcc{\frac{k-1}{n}}{\frac kn}$,
$f\left(\frac kn\right) \leq f(t) \leq f\left(\frac{k-1}{n}\right)$;
integrating and summing over $k$:
\[
S_n \leq \int_0^1 f(t)\,\dd t \leq S_n + \frac1n\bigl(f(0) - f(1)\bigr)
= S_n + \frac{1}{2n}.
\]
(The middle term of the exercise's display, with
$f(0) - f(1) = 1 - \frac12$, is exactly this bound.) Hence
$0 \leq \ln 2 - S_n \leq \frac{1}{2n} \to 0$, so $S_n \to \ln 2$.
\end{solution}

\begin{solution}{exo:g12:integ:9}
\emph{1.} $W_0 = \int_0^{\pi/2} \dd t = \frac{\pi}{2}$ and
$W_1 = \bigl[-\cos t\bigr]_0^{\pi/2} = 1$.

\emph{2.} Integrate by parts with $u' = \sin t$, $v = \sin^{n+1} t$, so
$u = -\cos t$, $v' = (n+1)\sin^n t \cos t$:
\[
W_{n+2} = \bigl[-\cos t \sin^{n+1} t\bigr]_0^{\pi/2}
+ (n+1)\int_0^{\pi/2} \cos^2 t\,\sin^n t\,\dd t
= (n+1)\int_0^{\pi/2} (1 - \sin^2 t)\sin^n t\,\dd t,
\]
the bracket vanishing at both ends. Hence
$W_{n+2} = (n+1)(W_n - W_{n+2})$, \emph{i.e.}
$W_{n+2} = \frac{n+1}{n+2} W_n$.

\emph{3.} $W_2 = \frac12 W_0 = \frac{\pi}{4}$,
$W_3 = \frac23 W_1 = \frac23$,
$W_4 = \frac34 W_2 = \frac{3\pi}{16}$.
On $\intoo{0}{\frac\pi2}$, $0 < \sin t < 1$, so
$\sin^{n+1} t \leq \sin^n t$ with strict inequality on the interior;
integrating, $0 < W_{n+1} \leq W_n$ (in fact $<$): $(W_n)$ is \omterm{def:g10:functions:variations}{decreasing} and
positive.
\end{solution}

\begin{solution}{pb:g12:integ:1}
\textbf{1.} $\left[x^3 - x^2 + x\right]_0^1 = 1$; \quad
$\left[\ln x\right]_1^{\eu} = 1$; \quad
$\left[\sin x\right]_0^{\pi/2} = 1$.

\textbf{2.} $\left[\frac12 \eu^{x^2}\right]_0^1 =
\frac{\eu - 1}{2}$; \quad
$\left[\ln(x^2 + 1)\right]_0^1 = \ln 2$.

\textbf{3.} $\int_0^1 x\eu^x = \left[x\eu^x\right]_0^1 -
\int_0^1 \eu^x = \eu - (\eu - 1) = 1$. And
$\int_1^{\eu} \ln x = \left[x\ln x\right]_1^{\eu} -
\int_1^{\eu} 1 = \eu - (\eu - 1) = 1$.

\textbf{4.} $\frac{1}{\pi}\int_0^\pi \sin x\,\dd x =
\frac{2}{\pi} \approx 0.64$: the average height of the \omterm{def:g12:trigo:cossin}{sine}
arch is $\frac2\pi$, noticeably more than one half --- the
arch is fat near its top.

\textbf{5.} $\int_0^1 (x - x^2)\,\dd x = \frac12 - \frac13 =
\frac16$.

\textbf{6.} $L_n = \frac1n \sum_{k=0}^{n-1}
\left(\frac kn\right)^2$ and
$U_n = \frac1n \sum_{k=1}^{n} \left(\frac kn\right)^2$:
rectangles of width $\frac1n$ with heights read at the left
(below the curve) or right (above) endpoints.

\textbf{7.} True for $n = 1$ ($1 = \frac{1 \cdot 2 \cdot
3}{6}$). If it holds for $n$, then adding $(n+1)^2$:
\[
\frac{n(n+1)(2n+1)}{6} + (n+1)^2
= \frac{(n+1)\left(2n^2 + n + 6n + 6\right)}{6}
= \frac{(n+1)(n+2)(2n+3)}{6},
\]
the formula at $n + 1$: heredity, done.

\textbf{8.} $U_n = \frac{1}{n^3} \cdot
\frac{n(n+1)(2n+1)}{6} = \frac{(1 + \frac1n)(2 + \frac1n)}{6}
\to \frac26 = \frac13$, and $L_n = U_n - \frac1n \to \frac13$:
squeezed between its staircases, the area is exactly
$\frac13$.

\textbf{9.} $\int_0^1 x^2\,\dd x =
\left[\frac{x^3}{3}\right]_0^1 = \frac13$: one line. The
fundamental theorem converts an infinite squeeze into a single
\omterm{def:g12:deriv:derivative}{anti-derivative} evaluation --- that is why it is called
fundamental.

\textbf{10.} Segment: $\int_{-1}^{1}(1 - x^2)\,\dd x =
2 - \frac23 = \frac43$. Triangle with base the chord (length
$2$, at height $1$) and apex $(0,0)$: area
$\frac12 \times 2 \times 1 = 1$. Ratio: $\frac43$ --- exactly
Archimedes' theorem, verified by the machine he lacked.

\textbf{11.} $T\left(1 + \frac14 + \frac{1}{16} + \dots\right)
= T \cdot \frac{1}{1 - \frac14} = \frac43 T$ (geometric series
of ratio $\frac14$): with $T = 1$, the segment's area
$\frac43$ again. Archimedes summed the series by a purely
geometric argument --- the quarter-bites of the chocolate-bar
kind, two millennia before limits were words.

\textbf{12.} Exhaustion needs a clever geometric
self-similarity and gives exact areas case by case; Riemann
needs only monotonicity and gives a universal definition at
the price of computing sums; the fundamental theorem needs the
whole edifice of \omterm{def:g12:deriv:derivative}{derivatives} and repays it by making areas
one-line computations.

\textbf{13.} $A(1) = 0$ and $A'(x) = \frac1x$ (fundamental
theorem): $A$ is the \omterm{def:g12:integ:primitive}{primitive} of $\frac1x$ on
$\intoo{0}{+\infty}$ vanishing at $1$ --- which is precisely
the \omterm{def:g12:exp:ln}{natural logarithm}.

\textbf{14.} $g'(x) = \frac{a}{ax} - \frac1x = 0$: $g$ is
constant; at $x = 1$: $g(1) = A(a) - A(1) = A(a)$. Hence
$A(ax) = A(a) + A(x)$ for all $x$, and with $x = b$:
$\ln(ab) = \ln a + \ln b$. The \omterm{def:g11:func:reference}{hyperbola}'s area does not care
about horizontal scale: stretching $t$ by $a$ squashes
$\frac1t$ by $a$, and the area survives --- addition of
\omterm{def:g12:exp:ln}{logarithms} is a scaling symmetry.

\textbf{15.} Iterating: $\ln(a^n) = n\ln a$ (induction on
$n$). And $0 = \ln 1 = \ln\left(a \cdot \frac1a\right) =
\ln a + \ln\frac1a$: $\ln\frac1a = -\ln a$.

\textbf{16.} $\frac{1}{11} + \frac{1}{12} + \dots +
\frac{1}{20} \approx 0.669$, against $\ln 2 \approx 0.693$:
ten rectangles under $\frac{1}{1+t}$, already close. The
harmonic series diverges, but its \emph{slices} from $n$ to
$2n$ settle on $\ln 2$: divergence at a crawl, measured by an
area.

\textbf{17.} Distance: $\int_0^{10} 3t^2\,\dd t = 1000$ m.
\omterm{def:g11:stat:mean}{Mean} speed: $100$ m/s. Instant with $v = 100$:
$3t^2 = 100$: $t = \sqrt{\frac{100}{3}} \approx 5.77$ s.

\textbf{18.} Mass: $\int_0^4 (2 + x)\,\dd x = 8 + 8 = 16$ kg.
Moment: $\int_0^4 (2x + x^2)\,\dd x = 16 + \frac{64}{3} =
\frac{112}{3}$. Center of mass:
$\frac{112/3}{16} = \frac73 \approx 2.33$ m --- pushed beyond
the middle by the heavier right end, as intuition demands.

\textbf{19.} $W_0 = \frac\pi2$; $W_1 = \left[-\cos
t\right]_0^{\pi/2} = 1$; and
$W_2 = \int_0^{\pi/2} \frac{1 - \cos 2t}{2}\,\dd t =
\frac\pi4$.

\textbf{20.} Area: the definition, squeezed by staircases.
Accumulation: distances from speeds, masses from densities ---
the \omterm{def:g12:integ:area}{integral} as a running total. \omterm{def:g12:deriv:derivative}{Anti-derivative}: the
fundamental theorem, Newton's and Leibniz's shortcut through
Archimedes' patience. The pearl: the $\ln$ button is the area
under $\frac1t$, and that area's scaling symmetry \emph{is}
the identity $\ln(ab) = \ln a + \ln b$.
\end{solution}
