\chapter{Sums of Random Variables and the Law of Large Numbers}\label{ch:g12:sums}

Why do casinos always win in the end, and why do polls work? Because
averages of many independent random quantities fluctuate less and less. This
chapter proves it: linearity of expectation, additivity of variance for
independent variables, the Bienaymé--Chebyshev inequality, and the law of
large numbers.

\section{Sums of random variables}

Given two random variables $X, Y$ on the same finite sample space
$\Omega$, the sum $X + Y$ is the random variable
$\omega \mapsto X(\omega) + Y(\omega)$.

\begin{theorem}[Linearity of expectation]\label{thm:g12:sums:linearity}
\index{expectation!linearity}
For all random variables $X, Y$ on $\Omega$ and $a, b \in \R$:
\[
\E(X + Y) = \E(X) + \E(Y), \qquad \E(aX + b) = a\E(X) + b .
\]
\emph{No independence assumption is needed.}
\end{theorem}

\begin{proof}
Write the expectation as a sum over outcomes: since
$\P(X = x) = \sum_{\omega : X(\omega) = x} \P(\{\omega\})$, grouping terms
gives $\E(X) = \sum_{\omega \in \Omega} \P(\{\omega\})\,X(\omega)$. Then
\[
\E(X + Y) = \sum_{\omega} \P(\{\omega\})\bigl(X(\omega) + Y(\omega)\bigr)
= \sum_{\omega} \P(\{\omega\})X(\omega)
+ \sum_{\omega} \P(\{\omega\})Y(\omega) = \E(X) + \E(Y). \qedhere
\]
\end{proof}

\begin{definition}[Independent random variables]\label{def:g12:sums:indep}
$X$ and $Y$ are \emph{independent}\index{random variable!independent} if
for all values $x, y$:
\[
\P(X = x \text{ and } Y = y) = \P(X = x)\,\P(Y = y).
\]
Several variables $X_1, \dots, X_n$ are independent if this product rule
holds for every choice of values of every subfamily.
\end{definition}

\begin{proposition}\label{prop:g12:sums:prodexp}
If $X$ and $Y$ are independent, $\E(XY) = \E(X)\,\E(Y)$.
\end{proposition}

\begin{proof}
\[
\begin{aligned}
\E(XY) &= \sum_{x, y} xy\;\P(X = x \text{ and } Y = y)
= \sum_{x, y} xy\,\P(X=x)\P(Y=y) \\
&= \Bigl(\sum_x x\P(X=x)\Bigr)\Bigl(\sum_y y\P(Y=y)\Bigr). \qedhere
\end{aligned}
\]
\end{proof}

\begin{theorem}[Variance of a sum]\label{thm:g12:sums:variance}
\index{variance!of a sum}
If $X$ and $Y$ are \emph{independent}, then
\[
\V(X + Y) = \V(X) + \V(Y).
\]
More generally, for independent $X_1, \dots, X_n$:
$\V(X_1 + \dots + X_n) = \V(X_1) + \dots + \V(X_n)$.
\end{theorem}

\begin{proof}
Using König--Huygens (\cref{prop:g12:randvar:konig}) and linearity:
\begin{align*}
\V(X + Y) &= \E\bigl((X+Y)^2\bigr) - \bigl(\E X + \E Y\bigr)^2\\
&= \E(X^2) + 2\E(XY) + \E(Y^2)
- \E(X)^2 - 2\E(X)\E(Y) - \E(Y)^2\\
&= \V(X) + \V(Y) + 2\bigl(\E(XY) - \E(X)\E(Y)\bigr),
\end{align*}
and the last bracket vanishes for independent variables
(\cref{prop:g12:sums:prodexp}). The general case follows by induction.
\end{proof}

\begin{example}[Binomial revisited]\label{ex:g12:sums:binomial}
A binomial variable $X \sim \mathcal B(n, p)$ is a sum
$X = X_1 + \dots + X_n$ of $n$ independent Bernoulli variables. Hence,
structurally:
\[
\E(X) = np, \qquad \V(X) = np(1-p),
\]
recovering \cref{thm:g12:randvar:binomial} without any computation.
\end{example}

\begin{proposition}[Sample mean]\label{prop:g12:sums:mean}
Let $X_1, \dots, X_n$ be independent with the same distribution as $X$
(expectation $\mu$, variance $\sigma^2$), and let
$M_n = \frac{X_1 + \dots + X_n}{n}$ be their \emph{sample
mean}\index{sample mean}. Then
\[
\E(M_n) = \mu, \qquad \V(M_n) = \frac{\sigma^2}{n}, \qquad
\sigma(M_n) = \frac{\sigma}{\sqrt n}.
\]
\end{proposition}

\begin{proof}
Linearity gives $\E(M_n) = \frac{n\mu}{n} = \mu$; independence gives
$\V(X_1 + \dots + X_n) = n\sigma^2$, and dividing by $n$ scales the
variance by $\frac{1}{n^2}$ (\cref{prop:g12:randvar:affine}).
\end{proof}

The $\frac{\sigma}{\sqrt n}$ is the fundamental \emph{square root law}: to
halve the fluctuations of an average, quadruple the sample size.

\section{Concentration inequalities}

\begin{theorem}[Markov's inequality]\label{thm:g12:sums:markov}
\index{Markov's inequality}
If $X \geq 0$ and $a > 0$:
\[
\P(X \geq a) \leq \frac{\E(X)}{a}.
\]
\end{theorem}

\begin{proof}
In $\E(X) = \sum_i p_i x_i$ (all terms nonnegative), keep only the terms
with $x_i \geq a$: each is at least $a\,p_i$, so
$\E(X) \geq a \sum_{x_i \geq a} p_i = a\,\P(X \geq a)$.
\end{proof}

\begin{theorem}[Bienaymé--Chebyshev inequality]\label{thm:g12:sums:chebyshev}
\index{Bienaymé--Chebyshev inequality}
For any random variable $X$ and any $\varepsilon > 0$:
\[
\P\bigl(\abs{X - \E(X)} \geq \varepsilon\bigr)
\leq \frac{\V(X)}{\varepsilon^2}.
\]
\end{theorem}

\begin{proof}
Apply Markov's inequality to the nonnegative variable
$Y = (X - \E(X))^2$ with $a = \varepsilon^2$:
\[
\P\bigl(\abs{X - \E(X)} \geq \varepsilon\bigr)
= \P(Y \geq \varepsilon^2)
\leq \frac{\E(Y)}{\varepsilon^2} = \frac{\V(X)}{\varepsilon^2}. \qedhere
\]
\end{proof}

\begin{theorem}[Law of large numbers]\label{thm:g12:sums:lln}
\index{law of large numbers}
Let $X_1, \dots, X_n$ be independent with the same distribution
(expectation $\mu$, variance $\sigma^2$) and $M_n$ their sample mean. For
every $\varepsilon > 0$:
\[
\P\bigl(\abs{M_n - \mu} \geq \varepsilon\bigr)
\leq \frac{\sigma^2}{n\,\varepsilon^2}
\xrightarrow[n \to +\infty]{} 0 .
\]
The sample mean concentrates around the expectation.
\end{theorem}

\begin{proof}
Bienaymé--Chebyshev applied to $M_n$, whose expectation is $\mu$ and whose
variance is $\frac{\sigma^2}{n}$ (\cref{prop:g12:sums:mean}).
\end{proof}

\begin{remark}
This theorem is the bridge between probability theory and statistics: the
\emph{frequency} of an event over many independent repetitions approaches
its \emph{probability} (take $X_i$ the indicator of the event, so that
$\mu = p$). It justifies estimating a probability by simulation
(\emph{Monte Carlo method}) and a population proportion by a poll ---
quantitatively: see \cref{ch:g12:contdist}.
\end{remark}

\begin{method}[Using Bienaymé--Chebyshev]\label{met:g12:sums:chebyshev}
To guarantee $\P(\abs{M_n - \mu} \geq \varepsilon) \leq \alpha$, it
suffices that $n \geq \frac{\sigma^2}{\alpha\,\varepsilon^2}$. The bound is
crude (real fluctuations are usually much smaller) but perfectly general:
it requires nothing about the distribution beyond a finite variance.
\end{method}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:g12:sums:1}
Two fair dice are rolled; let $S$ be the sum. Using linearity (not the
distribution of $S$!), compute $\E(S)$; then compute $\V(S)$ using
independence, given that a single fair die has variance $\frac{35}{12}$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:sums:2}
Let $X \sim \mathcal B(100,\ 0.5)$. Bound
$\P(X \geq 75)$ using Markov's inequality, then
$\P(\abs{X - 50} \geq 25)$ using Bienaymé--Chebyshev. Compare.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:sums:3}
A fair coin is tossed $n$ times and $F_n$ denotes the frequency of heads.
How large must $n$ be so that, by the Bienaymé--Chebyshev bound,
$\P\bigl(\abs{F_n - 0.5} \geq 0.05\bigr) \leq 0.05$?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:sums:4}
An investor splits her capital equally between $n$ independent assets,
each with expected return $\mu = 5\%$ and standard deviation
$\sigma = 20\%$. Compute the expectation and standard deviation of the
portfolio return $M_n$, and the number of assets needed to bring the
standard deviation below $4\%$. What financial principle does this
illustrate?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:sums:5}
Let $X$ and $Y$ be independent, both uniform on $\{1, 2, 3\}$.
\begin{enumerate}
  \item Give the distribution of $S = X + Y$ and compute $\E(S)$, $\V(S)$
        directly from it.
  \item Recover both values by linearity and additivity of variance.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:sums:6}
Show that the additivity of variance can fail without independence:
compute $\V(X + Y)$ for $Y = X$, and compare with $\V(X) + \V(Y)$. For
which variables $X$ is $\V(2X) = 2\V(X)$?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:sums:7}
A die is suspected of being loaded. It is rolled $1200$ times and shows a
six $260$ times ($f = 0.2167$ instead of $\frac16 \approx 0.1667$). Under
the hypothesis that the die is fair, bound
$\P\bigl(\abs{F_n - \frac16} \geq 0.05\bigr)$ by Bienaymé--Chebyshev, and
discuss whether the fairness hypothesis is plausible.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:sums:8}
\emph{(A better inequality for the coin.)} Let
$X \sim \mathcal B(n,\ p)$ and $F_n = \frac Xn$.
\begin{enumerate}
  \item Show that $p(1-p) \leq \frac14$ for $p \in \intcc{0}{1}$.
  \item Deduce the distribution-free bound
        $\P\bigl(\abs{F_n - p} \geq \varepsilon\bigr) \leq
        \frac{1}{4n\varepsilon^2}$.
  \item How many people must be polled so that the observed frequency is
        within $3$ points of the true proportion with probability at least
        $95\%$, using this bound? (Real polls use sharper estimates, but
        the order of magnitude is right.)
\end{enumerate}
\end{exercise}
