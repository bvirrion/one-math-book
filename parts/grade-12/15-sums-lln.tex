\chapter{Sums of Random Variables and the Law of Large Numbers}\label{ch:g12:sums}

Why do casinos always win in the end, and why do polls work? Because
averages of many \omterm{def:g12:sums:indep}{independent} random quantities fluctuate less and less. This
chapter proves it: linearity of \omterm{def:g12:randvar:exp}{expectation}, additivity of \omterm{def:g12:randvar:exp}{variance} for
\omterm{def:g12:sums:indep}{independent} variables, the Bienaymé--Chebyshev inequality, and the law of
large numbers.

\section{Sums of random variables}

Given two \omterm{def:g12:randvar:rv}{random variables} $X, Y$ on the same finite \omterm{def:g11:prob:model}{sample space}
$\Omega$, the sum $X + Y$ is the \omterm{def:g12:randvar:rv}{random variable}
$\omega \mapsto X(\omega) + Y(\omega)$.

\begin{theorem}[Linearity of expectation]\label{thm:g12:sums:linearity}
\index{expectation!linearity}
For all \omterm{def:g12:randvar:rv}{random variables} $X, Y$ on $\Omega$ and $a, b \in \R$:
\[
\E(X + Y) = \E(X) + \E(Y), \qquad \E(aX + b) = a\E(X) + b .
\]
\emph{No \omterm{def:g12:condprob:indep}{independence} assumption is needed.}
\end{theorem}

\begin{proof}
Write the \omterm{def:g12:randvar:exp}{expectation} as a sum over outcomes: since
$\P(X = x) = \sum_{\omega : X(\omega) = x} \P(\{\omega\})$, grouping terms
gives $\E(X) = \sum_{\omega \in \Omega} \P(\{\omega\})\,X(\omega)$. Then
\[
\E(X + Y) = \sum_{\omega} \P(\{\omega\})\bigl(X(\omega) + Y(\omega)\bigr)
= \sum_{\omega} \P(\{\omega\})X(\omega)
+ \sum_{\omega} \P(\{\omega\})Y(\omega) = \E(X) + \E(Y). \qedhere
\]
\end{proof}

\begin{definition}[Independent random variables]\label{def:g12:sums:indep}
$X$ and $Y$ are \emph{independent}\index{random variable!independent} if
for all values $x, y$:
\[
\P(X = x \text{ and } Y = y) = \P(X = x)\,\P(Y = y).
\]
Several variables $X_1, \dots, X_n$ are independent if this product rule
holds for every choice of values of every subfamily.
\end{definition}

\begin{proposition}\label{prop:g12:sums:prodexp}
If $X$ and $Y$ are \omterm{def:g12:sums:indep}{independent}, $\E(XY) = \E(X)\,\E(Y)$.
\end{proposition}

\begin{proof}
\[
\begin{aligned}
\E(XY) &= \sum_{x, y} xy\;\P(X = x \text{ and } Y = y)
= \sum_{x, y} xy\,\P(X=x)\P(Y=y) \\
&= \Bigl(\sum_x x\P(X=x)\Bigr)\Bigl(\sum_y y\P(Y=y)\Bigr). \qedhere
\end{aligned}
\]
\end{proof}

\begin{theorem}[Variance of a sum]\label{thm:g12:sums:variance}
\index{variance!of a sum}
If $X$ and $Y$ are \emph{\omterm{def:g12:sums:indep}{independent}}, then
\[
\V(X + Y) = \V(X) + \V(Y).
\]
More generally, for \omterm{def:g12:sums:indep}{independent} $X_1, \dots, X_n$:
$\V(X_1 + \dots + X_n) = \V(X_1) + \dots + \V(X_n)$.
\end{theorem}

\begin{proof}
Using König--Huygens (\cref{prop:g12:randvar:konig}) and linearity:
\begin{align*}
\V(X + Y) &= \E\bigl((X+Y)^2\bigr) - \bigl(\E X + \E Y\bigr)^2\\
&= \E(X^2) + 2\E(XY) + \E(Y^2)
- \E(X)^2 - 2\E(X)\E(Y) - \E(Y)^2\\
&= \V(X) + \V(Y) + 2\bigl(\E(XY) - \E(X)\E(Y)\bigr),
\end{align*}
and the last bracket vanishes for \omterm{def:g12:sums:indep}{independent} variables
(\cref{prop:g12:sums:prodexp}). The general case follows by induction.
\end{proof}

\begin{example}[Binomial revisited]\label{ex:g12:sums:binomial}
A binomial variable $X \sim \mathcal B(n, p)$ is a sum
$X = X_1 + \dots + X_n$ of $n$ \omterm{def:g12:sums:indep}{independent} Bernoulli variables. Hence,
structurally:
\[
\E(X) = np, \qquad \V(X) = np(1-p),
\]
recovering \cref{thm:g12:randvar:binomial} without any computation.
\end{example}

\begin{proposition}[Sample mean]\label{prop:g12:sums:mean}
Let $X_1, \dots, X_n$ be \omterm{def:g12:sums:indep}{independent} with the same \omterm{def:g12:randvar:rv}{distribution} as $X$
(\omterm{def:g12:randvar:exp}{expectation} $\mu$, \omterm{def:g12:randvar:exp}{variance} $\sigma^2$), and let
$M_n = \frac{X_1 + \dots + X_n}{n}$ be their \emph{sample
mean}\index{sample mean}. Then
\[
\E(M_n) = \mu, \qquad \V(M_n) = \frac{\sigma^2}{n}, \qquad
\sigma(M_n) = \frac{\sigma}{\sqrt n}.
\]
\end{proposition}

\begin{proof}
Linearity gives $\E(M_n) = \frac{n\mu}{n} = \mu$; \omterm{def:g12:condprob:indep}{independence} gives
$\V(X_1 + \dots + X_n) = n\sigma^2$, and dividing by $n$ scales the
\omterm{def:g12:randvar:exp}{variance} by $\frac{1}{n^2}$ (\cref{prop:g12:randvar:affine}).
\end{proof}

The $\frac{\sigma}{\sqrt n}$ is the fundamental \emph{square root law}: to
halve the fluctuations of an average, quadruple the \omterm{def:g10:proba:sample}{sample} size.

\section{Concentration inequalities}

\begin{theorem}[Markov's inequality]\label{thm:g12:sums:markov}
\index{Markov's inequality}
If $X \geq 0$ and $a > 0$:
\[
\P(X \geq a) \leq \frac{\E(X)}{a}.
\]
\end{theorem}

\begin{proof}
In $\E(X) = \sum_i p_i x_i$ (all terms nonnegative), keep only the terms
with $x_i \geq a$: each is at least $a\,p_i$, so
$\E(X) \geq a \sum_{x_i \geq a} p_i = a\,\P(X \geq a)$.
\end{proof}

\begin{theorem}[Bienaymé--Chebyshev inequality]\label{thm:g12:sums:chebyshev}
\index{Bienaymé--Chebyshev inequality}
For any \omterm{def:g12:randvar:rv}{random variable} $X$ and any $\varepsilon > 0$:
\[
\P\bigl(\abs{X - \E(X)} \geq \varepsilon\bigr)
\leq \frac{\V(X)}{\varepsilon^2}.
\]
\end{theorem}

\begin{proof}
Apply Markov's inequality to the nonnegative variable
$Y = (X - \E(X))^2$ with $a = \varepsilon^2$:
\[
\P\bigl(\abs{X - \E(X)} \geq \varepsilon\bigr)
= \P(Y \geq \varepsilon^2)
\leq \frac{\E(Y)}{\varepsilon^2} = \frac{\V(X)}{\varepsilon^2}. \qedhere
\]
\end{proof}

\begin{theorem}[Law of large numbers]\label{thm:g12:sums:lln}
\index{law of large numbers}
Let $X_1, \dots, X_n$ be \omterm{def:g12:sums:indep}{independent} with the same \omterm{def:g12:randvar:rv}{distribution}
(\omterm{def:g12:randvar:exp}{expectation} $\mu$, \omterm{def:g12:randvar:exp}{variance} $\sigma^2$) and $M_n$ their \omterm{prop:g12:sums:mean}{sample mean}. For
every $\varepsilon > 0$:
\[
\P\bigl(\abs{M_n - \mu} \geq \varepsilon\bigr)
\leq \frac{\sigma^2}{n\,\varepsilon^2}
\xrightarrow[n \to +\infty]{} 0 .
\]
The \omterm{prop:g12:sums:mean}{sample mean} concentrates around the \omterm{def:g12:randvar:exp}{expectation}.
\end{theorem}

\begin{proof}
Bienaymé--Chebyshev applied to $M_n$, whose \omterm{def:g12:randvar:exp}{expectation} is $\mu$ and whose
\omterm{def:g12:randvar:exp}{variance} is $\frac{\sigma^2}{n}$ (\cref{prop:g12:sums:mean}).
\end{proof}

\begin{remark}
This theorem is the bridge between \omterm{def:g10:proba:distribution}{probability} theory and statistics: the
\emph{\omterm{def:g10:stats:series}{frequency}} of an \omterm{def:g11:prob:model}{event} over many \omterm{def:g12:sums:indep}{independent} repetitions approaches
its \emph{\omterm{def:g10:proba:distribution}{probability}} (take $X_i$ the indicator of the \omterm{def:g11:prob:model}{event}, so that
$\mu = p$). It justifies estimating a \omterm{def:g10:proba:distribution}{probability} by simulation
(\emph{Monte Carlo method}) and a population proportion by a poll ---
quantitatively: see \cref{ch:g12:contdist}.
\end{remark}

\begin{method}[Using Bienaymé--Chebyshev]\label{met:g12:sums:chebyshev}
To guarantee $\P(\abs{M_n - \mu} \geq \varepsilon) \leq \alpha$, it
suffices that $n \geq \frac{\sigma^2}{\alpha\,\varepsilon^2}$. The bound is
crude (real fluctuations are usually much smaller) but perfectly general:
it requires nothing about the \omterm{def:g12:randvar:rv}{distribution} beyond a finite \omterm{def:g12:randvar:exp}{variance}.
\end{method}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:g12:sums:1}
Two fair dice are rolled; let $S$ be the sum. Using linearity (not the
\omterm{def:g12:randvar:rv}{distribution} of $S$!), compute $\E(S)$; then compute $\V(S)$ using
\omterm{def:g12:condprob:indep}{independence}, given that a single fair die has \omterm{def:g12:randvar:exp}{variance} $\frac{35}{12}$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:sums:2}
Let $X \sim \mathcal B(100,\ 0.5)$. Bound
$\P(X \geq 75)$ using Markov's inequality, then
$\P(\abs{X - 50} \geq 25)$ using Bienaymé--Chebyshev. Compare.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:sums:3}
A fair coin is tossed $n$ times and $F_n$ denotes the \omterm{def:g10:stats:series}{frequency} of heads.
How large must $n$ be so that, by the Bienaymé--Chebyshev bound,
$\P\bigl(\abs{F_n - 0.5} \geq 0.05\bigr) \leq 0.05$?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:sums:4}
An investor splits her capital equally between $n$ \omterm{def:g12:sums:indep}{independent} assets,
each with expected return $\mu = 5\%$ and \omterm{def:g11:stat:variance}{standard deviation}
$\sigma = 20\%$. Compute the \omterm{def:g12:randvar:exp}{expectation} and \omterm{def:g11:stat:variance}{standard deviation} of the
portfolio return $M_n$, and the number of assets needed to bring the
\omterm{def:g11:stat:variance}{standard deviation} below $4\%$. What financial principle does this
illustrate?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:sums:5}
Let $X$ and $Y$ be \omterm{def:g12:sums:indep}{independent}, both uniform on $\{1, 2, 3\}$.
\begin{enumerate}
  \item Give the \omterm{def:g12:randvar:rv}{distribution} of $S = X + Y$ and compute $\E(S)$, $\V(S)$
        directly from it.
  \item Recover both values by linearity and additivity of \omterm{def:g12:randvar:exp}{variance}.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:sums:6}
Show that the additivity of \omterm{def:g12:randvar:exp}{variance} can fail without \omterm{def:g12:condprob:indep}{independence}:
compute $\V(X + Y)$ for $Y = X$, and compare with $\V(X) + \V(Y)$. For
which variables $X$ is $\V(2X) = 2\V(X)$?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:sums:7}
A die is suspected of being loaded. It is rolled $1200$ times and shows a
six $260$ times ($f = 0.2167$ instead of $\frac16 \approx 0.1667$). Under
the hypothesis that the die is fair, bound
$\P\bigl(\abs{F_n - \frac16} \geq 0.05\bigr)$ by Bienaymé--Chebyshev, and
discuss whether the fairness hypothesis is plausible.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:sums:8}
\emph{(A better inequality for the coin.)} Let
$X \sim \mathcal B(n,\ p)$ and $F_n = \frac Xn$.
\begin{enumerate}
  \item Show that $p(1-p) \leq \frac14$ for $p \in \intcc{0}{1}$.
  \item Deduce the distribution-free bound
        $\P\bigl(\abs{F_n - p} \geq \varepsilon\bigr) \leq
        \frac{1}{4n\varepsilon^2}$.
  \item How many people must be polled so that the observed \omterm{def:g10:stats:series}{frequency} is
        within $3$ points of the true proportion with \omterm{def:g10:proba:distribution}{probability} at least
        $95\%$, using this bound? (Real polls use sharper estimates, but
        the order of magnitude is right.)
\end{enumerate}
\end{exercise}

\section{Problem: The house always wins}

\begin{problem}[{Weekend problem --- the law of large numbers
explains casinos, polls and insurance, and demolishes the
gambler's fallacy on the way}]
\label{pb:g12:sums:1}
A roulette player after $100$ spins is, almost as often as not,
\emph{ahead}. The casino running a million spins is ahead with a
certainty no court would question. Same game, same tiny edge ---
the difference is $\sqrt n$, and it is the subject of this
chapter (\cref{prop:g12:sums:mean},
\cref{thm:g12:sums:chebyshev}, \cref{thm:g12:sums:lln}). This
problem runs the casino's books, sizes an election poll, prices
diversification --- and dismantles the most expensive fallacy in
the history of gambling.

\medskip
\noindent\textbf{Part I --- Fluency.}
\begin{enumerate}
  \item Two \omterm{def:g12:sums:indep}{independent} dice: compute $V(X + Y)$
        (\cref{thm:g12:sums:variance}).
  \item Roll $100$ \omterm{def:g12:sums:indep}{independent} dice and average the results:
        give the \omterm{def:g12:randvar:exp}{expectation} and the \omterm{def:g11:stat:variance}{standard deviation} of the
        \omterm{prop:g12:sums:mean}{sample mean} (\cref{prop:g12:sums:mean}; for one die,
        $\sigma = \sqrt{35/12} \approx 1.71$).
  \item A nonnegative \omterm{def:g12:randvar:rv}{random variable} has \omterm{def:g12:randvar:exp}{expectation} $2$.
        What does Markov's inequality
        (\cref{thm:g12:sums:markov}) say about
        $\P(X \geq 10)$?
  \item Bound $\P\left(\abs{\bar X_{100} - 3.5} \geq
        0.5\right)$ for the dice average of question 2 with
        Bienaym\'e--Chebyshev.
  \item \omterm{def:g12:randvar:exp}{Expectations} add always, \omterm{def:g12:randvar:exp}{variances} only under
        \omterm{def:g12:condprob:indep}{independence}: exhibit dependent $X, Y$ (hint:
        $Y = -X$) for which
        $V(X + Y) \neq V(X) + V(Y)$.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The house's books.}
European roulette, bet $1$ euro on red: win $+1$ with
\omterm{def:g10:proba:distribution}{probability} $\frac{18}{37}$, lose $-1$ with \omterm{def:g10:proba:distribution}{probability}
$\frac{19}{37}$.
\begin{enumerate}[resume]
  \item Compute the \omterm{def:g12:randvar:exp}{expectation} and \omterm{def:g11:stat:variance}{standard deviation} of one
        bet's gain.
  \item A gambler makes $100$ \omterm{def:g12:sums:indep}{independent} bets; let $G$ be the
        total gain. Compute $\E(G)$ and $\sigma(G)$, then the
        ratio $\frac{\abs{\E(G)}}{\sigma(G)}$. What does a
        drift of a quarter of a \omterm{def:g11:stat:variance}{standard deviation} \omterm{def:g11:stat:mean}{mean} for the
        gambler's chances of being ahead tonight?
  \item The casino sees $1\,000\,000$ bets. Compute the
        \omterm{def:g12:randvar:exp}{expectation} and \omterm{def:g11:stat:variance}{standard deviation} of its total take,
        and the same ratio. Interpret the number $27$.
  \item Certify with Chebyshev: bound the \omterm{def:g10:proba:distribution}{probability} that the
        casino \emph{loses} money over the million bets.
  \item Betting systems: doubling after losses, quitting when
        ahead\,\dots\ Using the linearity of \omterm{def:g12:randvar:exp}{expectation}
        (\cref{thm:g12:sums:linearity}) over the (possibly
        random) \omterm{def:g12:seq:sequence}{sequence} of stakes, explain why \emph{every}
        strategy in a negative-edge game has negative expected
        gain --- what would a winning system violate?
  \item State what the law of large numbers
        (\cref{thm:g12:sums:lln}) promises about the \omterm{def:g10:stats:series}{frequency}
        of red --- and what it does \emph{not} promise about
        the next spin after ten reds in a row. Name the
        fallacy.
  \item The subtlest point: the law works by \emph{dilution},
        not compensation. If reds run $100$ ahead of
        \omterm{def:g12:randvar:exp}{expectation} after some evening, the surplus is never
        ``paid back'' --- compute what happens instead to the
        \emph{fraction} $\frac{100}{n}$ by $n = 10^6$, and
        rewrite the gambler's fallacy's error in one sentence.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Polls.}
\begin{enumerate}[resume]
  \item From \cref{exo:g12:sums:8}: how many people must be
        polled for the observed \omterm{def:g10:stats:series}{frequency} to fall within $3$
        points of the truth with \omterm{def:g10:proba:distribution}{probability} at least
        $95\,\%$, by the distribution-free Chebyshev bound?
  \item Real polling institutes use about $1\,100$ people for
        a $\pm 3$-point margin at $95\,\%$: their formula is
        $n \approx \frac{1.96^2 \times p(1-p)}
        {\varepsilon^2}$ with $p(1-p) \leq \frac14$. Evaluate
        it, and explain the gap with question 13 (what
        does Chebyshev not know about the shape of the
        fluctuations?).
  \item To halve the margin of error, how must the \omterm{def:g10:proba:sample}{sample}
        grow? From $\pm 3$ points at $1\,100$, what \omterm{def:g10:proba:sample}{sample}
        gives $\pm 1.5$ points?
  \item The counterintuitive classic: the required \omterm{def:g10:proba:sample}{sample} size
        never used the population's size --- $1\,100$ people
        suffice for a city or a continent. Point to the place
        in the model where the population size is absent, and
        give the kitchen analogy that pollsters use.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Diversification.}
\begin{enumerate}[resume]
  \item The insurer of \cref{pb:g12:randvar:1} earns
        $30n$ euros in \omterm{def:g12:randvar:exp}{expectation} on $n$ policies with a
        claims \omterm{def:g11:stat:variance}{standard deviation} of about $315\sqrt n$. For
        which $n$ does the expected profit finally exceed one
        \omterm{def:g11:stat:variance}{standard deviation} of the claims? What is the
        $\sqrt n$ doing for the insurer?
  \item One stock's yearly return fluctuates with
        $\sigma = 20\,\%$. Split the money equally over $25$
        \emph{\omterm{def:g12:sums:indep}{independent}} such stocks: compute the
        portfolio's $\sigma$. Finance calls diversification
        the only free lunch --- what is the lunch, exactly?
  \item Now let the $25$ stocks be perfectly correlated (they
        all move together): what is the portfolio's $\sigma$?
        Compare the two extremes and state which assumption
        every diversification \omterm{def:g12:complex:argument}{argument} secretly rents ---
        and what happened when it failed system-wide in 2008.
  \item Finale --- the $\sqrt n$ symphony: sums fluctuate like
        $\sqrt n$ while their means settle like
        $\frac{1}{\sqrt n}$; play the theme through the four
        industries of this problem (casino, polling, insurance,
        portfolios), name the two standing abusers (the
        gambler's fallacy and fake \omterm{def:g12:condprob:indep}{independence}), and give the
        forward pointer: the \emph{shape} of the fluctuations
        --- the bell --- is the next chapter's \omterm{def:g12:limcont:continuity}{continuous}
        star, and its full theorem crowns the university
        volumes.
\end{enumerate}
\end{problem}
