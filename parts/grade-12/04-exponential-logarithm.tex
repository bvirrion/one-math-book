\chapter{Exponential and Logarithm}\label{ch:g12:exp}

The \omterm{thm:g12:exp:existence}{exponential function} is the unique \omterm{def:g11:func:function}{function} equal to its own \omterm{def:g12:deriv:derivative}{derivative}
and taking the value $1$ at $0$. It converts sums into products; its inverse,
the \omterm{def:g12:exp:ln}{natural logarithm}, converts products into sums. Together they describe
every phenomenon whose rate of change is proportional to its size:
radioactive decay, population growth, compound interest.

\section{The exponential function}

\begin{theorem}[Existence and uniqueness]\label{thm:g12:exp:existence}
There exists a unique \omterm{def:g12:deriv:derivative}{differentiable} \omterm{def:g11:func:function}{function} $f \colon \R \to \R$ such that
\[
f' = f \qquad\text{and}\qquad f(0) = 1 .
\]
It is called the \emph{exponential function}\index{exponential} and written
$\exp$, or $x \mapsto \eu^x$.
\end{theorem}

\begin{proof}[Proof of uniqueness]
First, such a \omterm{def:g11:func:function}{function} never vanishes. Indeed, let
$g(x) = f(x)f(-x)$; then
\[
g'(x) = f'(x)f(-x) - f(x)f'(-x) = f(x)f(-x) - f(x)f(-x) = 0,
\]
so $g$ is constant equal to $g(0) = 1$: for every $x$,
$f(x)f(-x) = 1$, and in particular $f(x) \neq 0$.

Now let $f_1, f_2$ be two solutions and $h = \dfrac{f_1}{f_2}$ (legitimate
since $f_2$ never vanishes). Then
\[
h' = \frac{f_1' f_2 - f_1 f_2'}{f_2^2} = \frac{f_1 f_2 - f_1 f_2}{f_2^2} = 0,
\]
so $h$ is constant equal to $h(0) = 1$, \emph{i.e.} $f_1 = f_2$.

\emph{Existence} is admitted at this level (it can be obtained via the
\omterm{def:g12:seq:sequence}{sequence} of \cref{exo:g12:seq:10}, or as the inverse of the \omterm{def:g12:exp:ln}{logarithm}
constructed by integration in \cref{ch:g12:integ}).
\end{proof}

\begin{proposition}[Functional equation]\label{prop:g12:exp:funceq}
For all $x, y \in \R$ and $n \in \Z$:
\[
\eu^{x+y} = \eu^x\,\eu^y, \qquad
\eu^{-x} = \frac{1}{\eu^x}, \qquad
\eu^{x-y} = \frac{\eu^x}{\eu^y}, \qquad
\bigl(\eu^{x}\bigr)^n = \eu^{nx}.
\]
Moreover $\eu^x > 0$ for all $x$.
\end{proposition}

\begin{proof}
Fix $y$ and consider $\varphi(x) = \dfrac{\exp(x+y)}{\exp(x)\exp(y)}$. Its
numerator and denominator, as \omterm{def:g11:func:function}{functions} of $x$, are both solutions of
$f' = f$ up to the constant $\exp(y)$; differentiating $\varphi$ directly
(quotient rule) gives $\varphi' = 0$, so
$\varphi \equiv \varphi(0) = 1$, proving the first identity. Taking
$y = -x$ gives the second (with the value $\eu^0 = 1$), and the third
follows. The fourth is an induction from the first for $n \geq 0$, extended
to $n < 0$ by the second.

Positivity: $\eu^x = \left(\eu^{x/2}\right)^2 \geq 0$ and $\eu^x \neq 0$
(shown in \cref{thm:g12:exp:existence}), so $\eu^x > 0$.
\end{proof}

\begin{proposition}[Variations and limits]\label{prop:g12:exp:variations}
The exponential is strictly \omterm{def:g12:seq:monotonic}{increasing}, \omterm{def:g12:deriv:convex}{convex}, and
\[
\lim_{x\to-\infty} \eu^x = 0, \qquad
\lim_{x\to+\infty} \eu^x = +\infty .
\]
\end{proposition}

\begin{proof}
$(\exp)' = \exp > 0$ gives strict increase; $(\exp)'' = \exp > 0$ gives
\omterm{def:g12:deriv:convex}{convexity}. By \omterm{def:g12:deriv:convex}{convexity}, $\eu^x \geq 1 + x$ (\omterm{def:g12:deriv:tangent}{tangent} at $0$, see
\cref{ex:g12:deriv:ineq}), so $\eu^x \to +\infty$ as $x \to +\infty$ by
comparison. Then $\eu^{x} = 1/\eu^{-x} \to 0$ as $x \to -\infty$.
\end{proof}

\begin{theorem}[Growth comparison]\label{thm:g12:exp:growth}
\index{growth comparison}
For every \omterm{def:g10:numbers:sets}{integer} $n \geq 1$,
\[
\lim_{x\to+\infty} \frac{\eu^x}{x^n} = +\infty,
\qquad
\lim_{x\to-\infty} x^n \eu^x = 0 .
\]
In words: the exponential beats every power of $x$.
\end{theorem}

\begin{proof}
For $n = 1$: applying $\eu^t \geq 1 + t \geq t$ at $t = x/2$,
\[
\frac{\eu^x}{x} = \frac{\left(\eu^{x/2}\right)^2}{x}
\geq \frac{(x/2)^2}{x} = \frac{x}{4} \xrightarrow[x\to+\infty]{} +\infty .
\]
For general $n$, write
\[
\frac{\eu^x}{x^n}
= \frac{1}{n^n}\left(\frac{\eu^{x/n}}{x/n}\right)^{n}
\xrightarrow[x\to+\infty]{} +\infty
\]
by the case $n=1$ and composition. The limit at $-\infty$ follows by the
substitution $x \mapsto -x$:
$\abs{x^n \eu^x} = \frac{\abs{x}^n}{\eu^{\abs x}} \to 0$.
\end{proof}

\section{The natural logarithm}

\begin{definition}[Natural logarithm]\label{def:g12:exp:ln}
The exponential is \omterm{def:g12:limcont:continuity}{continuous} and strictly \omterm{def:g12:seq:monotonic}{increasing} from $\R$ onto
$\intoo{0}{+\infty}$; by the bijection theorem
(\cref{thm:g12:limcont:bijection}), for every $y > 0$ the \omterm{def:g10:algebra:equation}{equation}
$\eu^x = y$ has a unique solution. This solution is the \emph{natural
logarithm}\index{logarithm} of $y$, written $\ln y$. Thus
\[
\text{for } x \in \R,\ y > 0: \qquad y = \eu^x \iff x = \ln y .
\]
In particular $\ln 1 = 0$, $\ln \eu = 1$, and
$\eu^{\ln y} = y$, $\ln(\eu^x) = x$.
\end{definition}

\begin{proposition}[Algebraic properties]\label{prop:g12:exp:lnalg}
For all $a, b > 0$ and $n \in \Z$:
\[
\ln(ab) = \ln a + \ln b, \quad
\ln\frac{1}{a} = -\ln a, \quad
\ln\frac{a}{b} = \ln a - \ln b, \quad
\ln(a^n) = n \ln a, \quad
\ln\sqrt{a} = \tfrac12 \ln a .
\]
\end{proposition}

\begin{proof}
$\eu^{\ln a + \ln b} = \eu^{\ln a}\eu^{\ln b} = ab$, and taking $\ln$ of both
sides gives the first identity. The others follow by the same mechanism from
the corresponding identities of \cref{prop:g12:exp:funceq}.
\end{proof}

\begin{proposition}[Analytic properties]\label{prop:g12:exp:lnanalytic}
The \omterm{def:g11:func:function}{function} $\ln$ is \omterm{def:g12:deriv:derivative}{differentiable} on $\intoo{0}{+\infty}$ with
\[
(\ln x)' = \frac{1}{x},
\]
strictly \omterm{def:g12:seq:monotonic}{increasing}, \omterm{def:g12:deriv:convex}{concave}, and
$\lim\limits_{x\to0^+} \ln x = -\infty$,
$\lim\limits_{x\to+\infty} \ln x = +\infty$. Moreover, for every \omterm{def:g10:numbers:sets}{integer}
$n \geq 1$,
\[
\lim_{x\to+\infty} \frac{\ln x}{x^{1/n}} = 0, \qquad
\lim_{x\to0^+} x \ln x = 0 .
\]
\end{proposition}

\begin{proof}
Differentiability of the inverse \omterm{def:g11:func:function}{function} is admitted at this level; granting
it, differentiate the identity $\eu^{\ln x} = x$ by the chain rule:
$(\ln x)'\,\eu^{\ln x} = 1$, so $(\ln x)' = \frac{1}{x} > 0$, whence strict
increase, and $(\ln x)'' = -\frac1{x^2} < 0$, whence concavity. The limits at
$0^+$ and $+\infty$ mirror those of $\exp$ through the bijection. For the
\omterm{thm:g12:exp:growth}{growth comparison}, substitute $x = \eu^{t}$:
$\frac{\ln x}{x} = \frac{t}{\eu^t} \to 0$ as $t \to +\infty$ by
\cref{thm:g12:exp:growth}, and similarly for the other limits with
$x = \eu^{-t}$, giving $x\ln x = -t\eu^{-t} \to 0$.
\end{proof}

\begin{method}[Solving equations with $\exp$ and $\ln$]\label{met:g12:exp:equations}
Both \omterm{def:g11:func:function}{functions} are strictly \omterm{def:g12:seq:monotonic}{increasing}, so they can be applied to (or removed
from) both sides of an \omterm{def:g10:algebra:equation}{equation} or inequality without changing its direction:
\[
\eu^{u} = \eu^{v} \iff u = v, \qquad
\eu^{u} \leq \eu^{v} \iff u \leq v,
\]
and likewise for $\ln$ on positive quantities. \emph{Always check \omterm{def:g11:func:function}{domains}
first} ($\ln$ requires positive arguments). For \omterm{def:g10:algebra:equation}{equations} like
$a^x = b$ with $a>0$, $a \neq 1$, rewrite $a^x = \eu^{x\ln a}$ and solve
$x = \frac{\ln b}{\ln a}$.
\end{method}

\begin{example}\label{ex:g12:exp:halflife}
A radioactive \omterm{def:g10:proba:sample}{sample} decays following $N(t) = N_0\,\eu^{-\lambda t}$. Its
\emph{half-life} $T$ satisfies $N(T) = N_0/2$, \emph{i.e.}
$\eu^{-\lambda T} = \frac12$, so $T = \frac{\ln 2}{\lambda}$: the half-life
does not depend on the initial quantity.
\end{example}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:g12:exp:1}
Simplify
$\dfrac{\eu^{3x}\,\eu^{-x+1}}{\eu^{x}}$ and
$\ln\!\left(\dfrac{\eu^2\sqrt{\eu}}{\eu^{-3}}\right)$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:exp:2}
Solve in $\R$:
\[
\text{(a) } \eu^{2x} - 3\eu^{x} + 2 = 0;
\qquad
\text{(b) } \ln(x - 1) + \ln(x + 2) = \ln 4 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:exp:3}
Compute the limits:
\[
\lim_{x\to+\infty} \frac{\eu^x - x^2}{\eu^x + 1}, \qquad
\lim_{x\to+\infty} \frac{\ln x}{\sqrt x}, \qquad
\lim_{x\to0^+} x^2 \ln x, \qquad
\lim_{x\to+\infty} \bigl(x - \ln x\bigr).
\]
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:exp:4}
Study the \omterm{def:g11:func:function}{function} $f(x) = x\,\eu^{-x}$ on $\R$: variations, limits,
\omterm{def:g12:deriv:extremum}{extremum}; show that its curve has an \omterm{def:g12:deriv:inflection}{inflection point} and give its
\omterm{def:g10:coordgeom:system}{coordinates}.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:exp:5}
Show that for all $x > -1$, $\ln(1 + x) \leq x$, with equality only at
$x = 0$. Deduce that for all $n \geq 1$,
\[
\left(1 + \frac{1}{n}\right)^n \leq \eu \leq \left(1 - \frac{1}{n+1}\right)^{-(n+1)} .
\]
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:exp:6}
A capital $C_0$ is invested at an annual rate of $3\%$, interest compounded
each year.
\begin{enumerate}
  \item Express the capital $C_n$ after $n$ years.
  \item After how many years does the capital double? Give the exact answer
        using $\ln$, then a numerical value.
  \item Compare with the \omterm{def:g10:numbers:approx}{approximation} ``$70$ divided by the rate in
        percent'' used by bankers.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:exp:7}
Solve the inequality $\eu^{2x} - \eu^{x+1} > 0$, then the inequality
$\ln(x^2 - 1) \leq \ln(x + 5)$.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:exp:8}
Let $f(x) = \dfrac{\eu^x}{x}$ for $x > 0$.
\begin{enumerate}
  \item Study the variations of $f$ on $\intoo{0}{+\infty}$ and give its
        \omterm{def:g10:functions:extrema}{minimum}.
  \item For which values of $k$ does the \omterm{def:g10:algebra:equation}{equation} $\eu^x = kx$ have $0$, $1$
        or $2$ solutions in $\intoo{0}{+\infty}$?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:exp:9}
For $n \geq 1$, let $u_n = \left(1 + \frac1n\right)^n$.
\begin{enumerate}
  \item Using \cref{exo:g12:exp:5}, show that $(u_n)$ is \omterm{def:g12:seq:bounded}{bounded above}
        by~$\eu$.
  \item Show that $\ln u_n = n \ln\left(1 + \frac1n\right) \to 1$, and
        deduce that $u_n \to \eu$.
        (Hint: recognize a difference quotient of $\ln$ at $1$.)
\end{enumerate}
\end{exercise}

\section{Problem: The logarithm tames the world}

\begin{problem}[{Weekend problem --- doubling times and the rule
of 72, the scales of earthquakes, acids and pianos, and the
constant $\eu$ leaving fingerprints everywhere}]
\label{pb:g12:exp:1}
Whatever grows by a fixed \emph{percentage} grows
exponentially --- savings, bacteria, epidemics --- and whatever
spans too many powers of ten to grasp --- earthquake energies,
acidities, sound intensities --- is tamed by a \omterm{def:g12:exp:ln}{logarithm}. This
problem computes doubling times and unmasks the bankers' rule of
72, reads the world's logarithmic scales, and collects the
fingerprints that the number $\eu$ leaves at every crime scene
(\cref{prop:g12:exp:funceq}, \cref{thm:g12:exp:growth},
\cref{met:g12:exp:equations}).

\medskip
\noindent\textbf{Part I --- Fluency.}
\begin{enumerate}
  \item Solve: $\eu^{2x} = 5$; \quad $\ln(3x - 1) = 2$; \quad
        $\eu^{2x} - 3\eu^x + 2 = 0$ (a quadratic in disguise).
  \item Simplify: $\dfrac{\ln 8}{\ln 2}$; \quad
        $\ln\!\left(\eu^3 \sqrt{\eu}\right)$; \quad
        $\eu^{\ln 5 - \ln 2}$.
  \item Prove the mother inequality: $\eu^x \geq 1 + x$ for all
        real $x$ (study $f(x) = \eu^x - 1 - x$), and deduce
        $\ln(1 + u) \leq u$ for $u > -1$.
  \item Compute
        $\lim_{x \to +\infty} x^2 \eu^{-x}$,
        $\lim_{x \to +\infty} \frac{\ln x}{x}$
        (\cref{thm:g12:exp:growth}), and
        $\lim_{x \to 0} \frac{\eu^x - 1}{x}$ (recognize a
        \omterm{def:g12:deriv:derivative}{derivative}).
  \item Study $f(x) = x \ln x$ on $\intoo{0}{+\infty}$:
        variations, \omterm{def:g10:functions:extrema}{minimum}, and the limit at $0^+$ (admitted:
        $x \ln x \to 0$). This little \omterm{def:g11:func:function}{function} measures
        information and entropy across the university volumes.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Doubling times and the rule of
72.}
\begin{enumerate}[resume]
  \item Savings grow at $3\,\%$ per year. Solve
        $1.03^n = 2$: how long to double the capital?
  \item Bankers estimate doubling time as
        $\frac{72}{\text{rate in }\%}$. Test the rule at
        $3\,\%$, $6\,\%$ and $9\,\%$ against the exact
        $\frac{\ln 2}{\ln(1 + r)}$, then explain it: for small
        $r$, $\ln(1 + r) \approx r$ (question 3's inequality is
        half of the story), so the exact constant is
        $100 \ln 2 \approx 69.3$ --- why do bankers prefer
        $72$?
  \item A bacterium divides every $20$ minutes:
        $p(t) = 2^{t/20}$ ($t$ in minutes). Rewrite it as
        $\eu^{\lambda t}$, then compute $p$ after $24$ hours.
        The answer (more than $10^{21}$) proves what about the
        model --- and what stops real colonies?
  \item Caffeine leaves the body with a half-life of about $5$
        hours. After a $100$ mg coffee at 15:00, how much
        remains at 23:00? At what time does it drop below
        $10$ mg? (Insomnia has a \omterm{def:g12:exp:ln}{logarithm}.)
  \item One euro at $100\,\%$ annual interest: compute the
        year-end capital under yearly, monthly and daily
        compounding, and give the ceiling that
        \cref{exo:g12:exp:9} proved unbreakable. Which famous
        constant is the limit of pure greed?
  \item Why do scientists plot $\ln(\text{cases})$ against
        time during an epidemic's early phase? What does a
        straight line on that plot reveal, and what does its
        \omterm{def:g10:reffunc:affine}{slope} measure?
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The world's logarithmic scales.}
(Write $\log_{10} x = \frac{\ln x}{\ln 10}$.)
\begin{enumerate}[resume]
  \item Sound level in decibels: $L = 10 \log_{10}(I/I_0)$. A
        conversation measures $60$ dB, a rock concert $120$ dB:
        by what factor do the sound intensities differ?
  \item Earthquake magnitudes rise by $1$ when the seismic
        amplitude is multiplied by $10$, and the released
        energy scales like amplitude$^{3/2}$. Compare
        magnitude-$5$ and magnitude-$7$ quakes: amplitude
        ratio, then energy ratio.
  \item Chemistry: $\text{pH} = -\log_{10}[\mathrm{H^+}]$.
        Lemon juice has pH $2$, milk pH $6.5$: what is the
        ratio of their acid concentrations?
  \item Music: each octave doubles the \omterm{def:g10:stats:series}{frequency}. From the
        piano's lowest A ($27.5$ Hz) to its highest C
        ($4\,186$ Hz), how many octaves does the keyboard span?
        And why do our senses --- hearing, sight, quake-feeling
        --- prefer logarithmic scales? (One sentence.)
  \item The slide rule, the engineers' calculator for $350$
        years: two sticks graduated so that the \emph{length}
        to the mark $x$ is proportional to $\ln x$. Explain how
        sliding one stick along the other multiplies numbers,
        and name the identity of
        \cref{prop:g12:exp:lnalg} doing the work.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- $\eu$'s fingerprints.}
\begin{enumerate}[resume]
  \item From \cref{exo:g12:exp:8}: the \omterm{def:g10:algebra:equation}{equation}
        $\eu^x = kx$ ($x > 0$) has $0$, $1$ or $2$ solutions
        according to the position of $k$ relative to a
        threshold. Restate the result --- and verify the
        geometric fact behind it: the line $y = \eu x$ is
        exactly the \omterm{def:g12:deriv:tangent}{tangent} to the exponential through the
        origin.
  \item The near-miss constant: with $n$ lottery tickets of
        winning \omterm{def:g10:proba:distribution}{probability} $\frac1n$ each,
        $\P(\text{no win}) = \left(1 - \frac1n\right)^n$.
        Compute its limit via
        $n \ln\!\left(1 - \frac1n\right)$ (use the standard
        limit $\frac{\ln(1 + u)}{u} \to 1$), and reconcile with
        the mysterious $0.368$ of \cref{pb:g11:binom:1}.
  \item The $37\,\%$ rule: to choose the best of $n$
        candidates interviewed in random order (no going back),
        the optimal strategy rejects the first
        $\frac{n}{\eu} \approx 37\,\%$ and then takes the first
        candidate better than all so far --- succeeding with
        \omterm{def:g10:proba:distribution}{probability} about $\frac1\eu$. Where do the two
        $\frac1\eu$'s of questions 18 and 19 come from ---
        state the common mechanism (many independent
        small-chance \omterm{def:g11:prob:model}{events}), and compute $\frac1\eu$ to three
        decimals.
  \item Finale --- $\eu$'s portrait: the ceiling of compounding
        (question 10); the base whose \omterm{def:g12:deriv:tangent}{tangent} at $0$ has \omterm{def:g10:reffunc:affine}{slope}
        exactly $1$ (questions 3 and 4); the growth no
        polynomial catches (question 4); the constant of
        near-misses and of optimal stopping (questions 18--19);
        and its inverse $\ln$, which turns products into sums
        (question 16) and decades into inches (Part III). One
        sentence each.
\end{enumerate}
\end{problem}
