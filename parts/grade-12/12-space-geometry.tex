\chapter{Vectors, Lines and Planes in Space}\label{ch:g12:space}

Three-dimensional geometry becomes computational once \omterm{def:g11:vect:vector}{vectors} and
\omterm{def:g10:coordgeom:system}{coordinates} are available: lines get parametric representations, planes get
Cartesian \omterm{def:g10:algebra:equation}{equations}, and the \omterm{def:g12:space:dot}{scalar product} measures angles and distances.
This chapter builds this toolbox and uses it to solve the classical
\omterm{def:g10:numbers:interunion}{intersection} and distance problems.

\section{Vectors in space}

\omterm{def:g11:vect:vector}{Vectors} in space obey the same rules as in the plane: they add, they scale,
and $\vect{AB} = \vect{CD}$ exactly when $ABDC$ is a parallelogram. In a
\omterm{def:g10:coordgeom:system}{coordinate system} $(O; \vec\imath, \vec\jmath, \vec k)$, every \omterm{def:g11:vect:vector}{vector} has
\omterm{def:g10:coordgeom:system}{coordinates} $(x, y, z)$.

\begin{definition}[Collinearity, coplanarity]\label{def:g12:space:collinear}
Two \omterm{def:g11:vect:vector}{vectors} are \emph{collinear} if one is a multiple of the other. Three
\omterm{def:g11:vect:vector}{vectors} $\vec u, \vec v, \vec w$ are \emph{coplanar}\index{coplanar} if one
of them can be written as a combination of the other two, say
$\vec w = a\vec u + b\vec v$.
\end{definition}

\begin{proposition}\label{prop:g12:space:basis}
If $\vec u, \vec v, \vec w$ are not \omterm{def:g12:space:collinear}{coplanar}, every \omterm{def:g11:vect:vector}{vector} of space
decomposes uniquely as $x\vec u + y\vec v + z\vec w$.
\end{proposition}

\begin{proof}[Idea of proof]
Through the tip of the \omterm{def:g11:vect:vector}{vector} to decompose, draw the line parallel to
$\vec w$; it meets the plane of $\vec u, \vec v$ in a single point, which
splits the \omterm{def:g11:vect:vector}{vector} into a component in that plane (uniquely $x\vec u + y\vec
v$, plane geometry) and a component along $\vec w$. Uniqueness: a difference
of two decompositions would express one of the \omterm{def:g11:vect:vector}{vectors} in terms of the other
two, contradicting non-coplanarity.
\end{proof}

\section{Lines and planes}

\begin{definition}[Parametric representation of a line]\label{def:g12:space:line}
The line through $A(x_A, y_A, z_A)$ with \omterm{def:g11:vect:direction}{direction vector}
$\vec u (a, b, c) \neq \vec 0$ is the set of points
\[
M(x_A + ta,\; y_A + tb,\; z_A + tc), \qquad t \in \R .
\]
\end{definition}

\begin{definition}[Plane]\label{def:g12:space:plane}
The plane through $A$ directed by two \omterm{def:g12:space:collinear}{non-collinear} \omterm{def:g11:vect:vector}{vectors}
$\vec u, \vec v$ is the set of points $M$ with
$\vect{AM} = s\vec u + t\vec v$, $(s, t) \in \R^2$.
\end{definition}

\begin{proposition}[Relative positions]\label{prop:g12:space:positions}
Two distinct planes are either parallel or intersect in a line. A line and a
plane are either parallel (possibly contained) or intersect in a single
point. Two lines in space may be intersecting, strictly parallel, identical,
or \emph{skew} (\omterm{def:g12:space:collinear}{non-coplanar}).
\end{proposition}

\begin{proof}[Sketch]
These are incidence statements; each case analysis reduces, in \omterm{def:g10:coordgeom:system}{coordinates},
to solving a \omterm{def:g10:lines:system}{linear system} and counting its solutions --- see
\cref{met:g12:space:intersections}.
\end{proof}

\section{Scalar product}

\begin{definition}[Scalar product]\label{def:g12:space:dot}
The \emph{scalar product}\index{scalar product} of $\vec u$ and $\vec v$ is
\[
\vec u \cdot \vec v = \tfrac12\left(\norm{\vec u + \vec v}^2
- \norm{\vec u}^2 - \norm{\vec v}^2\right),
\]
where $\norm{\vec u}$ is the length of $\vec u$. If both \omterm{def:g11:vect:vector}{vectors} are
nonzero, $\vec u \cdot \vec v = \norm{\vec u}\,\norm{\vec v}\cos\theta$
where $\theta$ is the angle between them; and in an \emph{\omterm{def:g10:coordgeom:system}{orthonormal}}
\omterm{def:g10:coordgeom:system}{coordinate system},
\[
\vec u \cdot \vec v = xx' + yy' + zz' .
\]
Two \omterm{def:g11:vect:vector}{vectors} are \emph{\omterm{def:g11:scal:orthogonal}{orthogonal}} if their scalar product is $0$.
\end{definition}

\begin{proposition}[Properties]\label{prop:g12:space:dotprops}
The \omterm{def:g12:space:dot}{scalar product} is symmetric ($\vec u\cdot\vec v = \vec v\cdot\vec u$),
bilinear ($(a\vec u + b\vec v)\cdot \vec w = a\,\vec u\cdot\vec w +
b\,\vec v\cdot \vec w$), and
$\vec u \cdot \vec u = \norm{\vec u}^2 \geq 0$.
\end{proposition}

\begin{proof}
In an \omterm{def:g10:coordgeom:system}{orthonormal system}, all three properties are immediate on the formula
$xx' + yy' + zz'$, which itself follows from the defining formula and the
Pythagorean computation of lengths:
$\norm{\vec u}^2 = x^2 + y^2 + z^2$.
\end{proof}

\begin{definition}[Normal vector]\label{def:g12:space:normal}
A \emph{normal vector}\index{normal vector} of a plane $\mathcal P$ is a
nonzero \omterm{def:g11:vect:vector}{vector} \omterm{def:g11:scal:orthogonal}{orthogonal} to every \omterm{def:g11:vect:vector}{vector} lying in $\mathcal P$ (it
suffices that it be \omterm{def:g11:scal:orthogonal}{orthogonal} to two \omterm{def:g12:space:collinear}{non-collinear} directions of
$\mathcal P$).
\end{definition}

\begin{theorem}[Cartesian equation of a plane]\label{thm:g12:space:planeeq}
In an \omterm{def:g10:coordgeom:system}{orthonormal} \omterm{def:g10:coordgeom:system}{coordinate system}, the plane through
$A$ with \omterm{def:g12:space:normal}{normal vector} $\vec n(a, b, c)$ has \omterm{def:g10:algebra:equation}{equation}
\[
a x + b y + c z + d = 0,
\]
where $d = -(ax_A + by_A + cz_A)$. Conversely, every such \omterm{def:g10:algebra:equation}{equation} with
$(a,b,c) \neq (0,0,0)$ defines a plane with \omterm{def:g12:space:normal}{normal vector} $(a, b, c)$.
\end{theorem}

\begin{proof}
$M \in \mathcal P \iff \vect{AM} \perp \vec n \iff
a(x - x_A) + b(y - y_A) + c(z - z_A) = 0$, which expands to the \omterm{def:g10:algebra:equation}{equation}.
Conversely, pick any solution point $A$ of the \omterm{def:g10:algebra:equation}{equation}; the same
computation backwards shows the solution set is
$\{M : \vect{AM}\cdot\vec n = 0\}$, the plane through $A$ normal to
$\vec n$.
\end{proof}

\begin{method}[Intersections in practice]\label{met:g12:space:intersections}
\begin{itemize}
  \item \emph{Line $\cap$ plane}: substitute the parametric \omterm{def:g10:coordgeom:system}{coordinates} of
        the line into the plane \omterm{def:g10:algebra:equation}{equation}; solve for $t$ (one solution: one
        point; $0 = $ nonzero: parallel; $0 = 0$: line contained).
  \item \emph{Plane $\cap$ plane}: solve the system of the two \omterm{def:g10:algebra:equation}{equations},
        parametrizing by one free coordinate; the solution is a line (or the
        planes are parallel when the \omterm{def:g12:space:normal}{normal vectors} are \omterm{def:g12:space:collinear}{collinear}).
  \item \emph{Orthogonality checks}: line $\perp$ plane iff its direction
        is \omterm{def:g12:space:collinear}{collinear} with the normal; two planes are perpendicular iff their
        normals are \omterm{def:g11:scal:orthogonal}{orthogonal}.
\end{itemize}
\end{method}

\begin{proposition}[Distance from a point to a plane]\label{prop:g12:space:distance}
\index{distance!point to plane}
In an \omterm{def:g10:coordgeom:system}{orthonormal system}, the distance from $M_0(x_0, y_0, z_0)$ to the
plane $\mathcal P : ax + by + cz + d = 0$ is
\[
\operatorname{dist}(M_0, \mathcal P)
= \frac{\abs{ax_0 + by_0 + cz_0 + d}}{\sqrt{a^2 + b^2 + c^2}}.
\]
\end{proposition}

\begin{proof}
Let $H$ be the \omterm{def:g11:scal:orthogonal}{orthogonal} projection of $M_0$ on $\mathcal P$: the foot of
the perpendicular, so $\vect{HM_0}$ is \omterm{def:g12:space:collinear}{collinear} with the unit normal
$\frac{\vec n}{\norm{\vec n}}$, and the distance is
$\abs{\vect{HM_0}\cdot \frac{\vec n}{\norm{\vec n}}}$. For any
$H \in \mathcal P$,
$\vect{HM_0}\cdot\vec n = ax_0 + by_0 + cz_0 - (ax_H + by_H + cz_H)
= ax_0 + by_0 + cz_0 + d$ (using the \omterm{def:g10:algebra:equation}{equation} at $H$), whence the formula
after dividing by $\norm{\vec n} = \sqrt{a^2+b^2+c^2}$.
\end{proof}

\begin{example}\label{ex:g12:space:distance}
Distance from the origin to the plane $x + 2y + 2z - 6 = 0$:
$\frac{\abs{-6}}{\sqrt{1 + 4 + 4}} = \frac63 = 2$.
\end{example}

\section{Exercises}

In all exercises the \omterm{def:g10:coordgeom:system}{coordinate system} is \omterm{def:g10:coordgeom:system}{orthonormal}.

\begin{exercise}[$\star$]\label{exo:g12:space:1}
Let $A(1, 0, 2)$, $B(3, 1, 1)$ and $C(2, -1, 3)$. Are the points $A$, $B$,
$C$ aligned? Give a parametric representation of the line $(AB)$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:space:2}
Determine a Cartesian \omterm{def:g10:algebra:equation}{equation} of the plane through $A(1, 1, 0)$ with
\omterm{def:g12:space:normal}{normal vector} $\vec n(2, -1, 3)$. Does the point $B(0, 2, 1)$ belong to it?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:space:3}
Compute the angle at $A$ (to the nearest degree) of the triangle with
vertices $A(0,0,0)$, $B(1, 1, 0)$ and $C(1, 0, 1)$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:space:4}
Consider the line $\mathcal D$ through $A(1, 2, 0)$ with direction
$\vec u(1, -1, 2)$, and the plane $\mathcal P : 2x + y - z + 1 = 0$.
Determine $\mathcal D \cap \mathcal P$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:space:5}
Let $\mathcal P : x - y + 2z = 1$ and $\mathcal Q : 2x + y + z = 4$.
Show that $\mathcal P$ and $\mathcal Q$ intersect in a line and give a
parametric representation of it.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:space:6}
Show that the lines
\[
\mathcal D_1 : (x, y, z) = (1 + t,\; 2t,\; 3 - t), \qquad
\mathcal D_2 : (x, y, z) = (2 + s,\; 1 + s,\; 1 + 2s)
\]
are skew (not \omterm{def:g12:space:collinear}{coplanar}).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:space:7}
Let $\mathcal P : 2x - y + 2z - 3 = 0$.
\begin{enumerate}
  \item Compute the distance from $M_0(3, 1, 1)$ to $\mathcal P$.
  \item Determine the \omterm{def:g10:coordgeom:system}{coordinates} of the \omterm{def:g11:scal:orthogonal}{orthogonal} projection $H$ of
        $M_0$ on $\mathcal P$, and check the distance $M_0H$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:space:8}
The cube $ABCDEFGH$ has side $1$; place \omterm{def:g10:coordgeom:system}{coordinates} so that
$A(0,0,0)$, $B(1,0,0)$, $D(0,1,0)$, $E(0,0,1)$ (with
$C = B + D$, $F = B + E$, $G = B + D + E$, $H = D + E$ as \omterm{def:g11:vect:vector}{vectors} from
$A$).
\begin{enumerate}
  \item Show that the diagonal $(AG)$ is \omterm{def:g11:scal:orthogonal}{orthogonal} to the plane $(BDE)$.
  \item Compute the distance from $A$ to the plane $(BDE)$, and the point
        where $(AG)$ pierces it.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:space:9}
\emph{(Volume of a tetrahedron.)} Let $A(1,1,1)$, $B(2,1,0)$, $C(0,2,1)$
and $D(2,2,2)$.
\begin{enumerate}
  \item Verify that $\vect{AB}$, $\vect{AC}$, $\vect{AD}$ are not \omterm{def:g12:space:collinear}{coplanar}
        (the points form a genuine tetrahedron).
  \item Find a \omterm{def:g11:vect:vector}{vector} $\vec n$ \omterm{def:g11:scal:orthogonal}{orthogonal} to both $\vect{BC}$ and
        $\vect{BD}$, and deduce a Cartesian \omterm{def:g10:algebra:equation}{equation} of the plane $(BCD)$.
  \item Compute the distance from $A$ to the plane $(BCD)$ and the area of
        the triangle $BCD$, using the formula
        $\text{Area} = \frac12\sqrt{\norm{\vec u}^2\norm{\vec v}^2 -
        (\vec u \cdot \vec v)^2}$ for the triangle spanned by
        $\vec u, \vec v$.
  \item Deduce the volume of the tetrahedron
        ($V = \frac13 \times \text{base area} \times \text{height}$).
\end{enumerate}
\end{exercise}

\section{Problem: Slicing the cube}

\begin{problem}[{Weekend problem --- a plane cuts a cube into a
perfect hexagon, a regular tetrahedron hides in the corners, and
the diagonal's angle shapes every diamond}]
\label{pb:g12:space:1}
Slice a cube and you expect squares and rectangles --- yet one
famous cut produces a \emph{regular hexagon}, and four clever
corner cuts leave behind a \emph{regular tetrahedron}. Both
surprises, and the $109.5^\circ$ angle that carbon atoms worship,
fall to this chapter's toolkit: parametric lines, plane \omterm{def:g10:algebra:equation}{equations}
and the \omterm{def:g12:space:dot}{scalar product} (\cref{thm:g12:space:planeeq},
\cref{prop:g12:space:distance}). Use throughout the cube of
\cref{exo:g12:space:8}: side $1$, $A(0,0,0)$, $B(1,0,0)$,
$D(0,1,0)$, $E(0,0,1)$, $C(1,1,0)$, $F(1,0,1)$, $H(0,1,1)$,
$G(1,1,1)$.

\medskip
\noindent\textbf{Part I --- Fluency.}
\begin{enumerate}
  \item Write the \omterm{def:g10:algebra:equation}{equation} of the plane through $B(1,0,0)$ with
        \omterm{def:g12:space:normal}{normal vector} $(1,1,1)$, a parametric representation of
        the line through $A$ directed by $(1,1,1)$, and their
        \omterm{def:g10:numbers:interunion}{intersection} point.
  \item Compute the distance from $A$ to that plane
        (\cref{prop:g12:space:distance}).
  \item Are the \omterm{def:g11:vect:vector}{vectors} $(1,-1,0)$ and $(1,1,-2)$ \omterm{def:g11:scal:orthogonal}{orthogonal}?
  \item Position of the planes $x + y + z = 1$ and
        $2x + 2y + 2z = 5$? Of the line through $A$ directed by
        $(1,1,0)$ relative to the first plane?
  \item List the \omterm{prop:g10:coordgeom:midpoint}{midpoints} of the six edges
        $[BC], [CD], [DH], [HE], [EF], [FB]$ of the cube, and
        verify that all six satisfy the \omterm{def:g10:algebra:equation}{equation}
        $x + y + z = \frac32$.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The hexagonal section.}
Let $P$ be the plane $x + y + z = \frac32$.
\begin{enumerate}[resume]
  \item Show that $P$ is perpendicular to the great diagonal
        $(AG)$ and passes through the cube's center.
  \item Compute the distances between consecutive \omterm{prop:g10:coordgeom:midpoint}{midpoints} of
        question 5 around the section: what do you find?
  \item Compute the angle of the section's polygon at one
        vertex (two consecutive \omterm{def:g11:vect:vector}{edge-vectors} and a dot
        product). Conclude: the section is a \emph{regular
        hexagon}.
  \item Compute the hexagon's perimeter and area, and compare
        the area with that of a face of the cube. (For the
        record: the largest planar section of the unit cube is
        the $1 \times \sqrt2$ diagonal rectangle, of area
        $\sqrt2$ --- the regular hexagon takes silver.)
  \item Where does the diagonal $(AG)$ pierce $P$? Verify that
        the piercing point is the \omterm{prop:g10:coordgeom:midpoint}{midpoint} of $[AG]$.
  \item Slide the cut: describe the sections
        $x + y + z = c$ as $c$ grows from $0$ to $3$ ---
        compute the section for $c = \frac12$ (which polygon,
        which size?), and narrate the morphing that passes
        through your hexagon at $c = \frac32$.
  \item Why can no plane cut a cube in a polygon with
        \emph{seven} sides? (Where must each side of the
        section lie?)
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The hidden tetrahedron.}
\begin{enumerate}[resume]
  \item Compute the volume of the corner tetrahedron $ABDE$
        (base, height, and the one-third formula the Middle
        School volume admitted --- \cref{exo:g12:space:9}
        computes such volumes in general).
  \item The four vertices $B$, $D$, $E$, $G$: compute all six
        pairwise distances and conclude that $BDEG$ is a
        \emph{regular} tetrahedron. Then obtain its volume by
        subtracting corner tetrahedra from the cube.
  \item Compute the distance from the cube's center to the
        plane $(BDE)$.
  \item The diamond angle: compute the angle between the
        diagonals $\vect{AG}$ and $\vect{BH}$ of the cube. Its
        supplement, about $109.5^\circ$, is the angle between
        bonds in diamond and methane --- why do four
        electron pairs around a carbon atom choose the corners
        of question 14's tetrahedron?
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Genuinely three-dimensional.}
\begin{enumerate}[resume]
  \item Find where the line through $B(1,0,0)$ directed by
        $(0,1,1)$ pierces the hexagon's plane $P$.
  \item Show that the lines $(AB)$ and $(EG)$ neither meet nor
        are parallel: \emph{skew} lines, the phenomenon that
        cannot happen in a plane.
  \item Compute the \omterm{def:g11:scal:orthogonal}{orthogonal} projection of $B$ onto $P$
        (walk from $B$ along the normal until the plane), and
        check your point satisfies $P$'s \omterm{def:g10:algebra:equation}{equation}.
  \item Finale --- the space surveyor's kit: parametric lines
        for piercing, \omterm{def:g12:space:normal}{normal vectors} for angles and distances,
        plane \omterm{def:g10:algebra:equation}{equations} for sections; and the cube as the
        playground where they produced a hexagon, a regular
        tetrahedron, the diamond angle and a pair of skew
        lines. One sentence each.
\end{enumerate}
\end{problem}
