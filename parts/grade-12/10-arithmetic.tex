\chapter{Arithmetic}\label{ch:g12:arith}

Arithmetic studies the integers: divisibility, prime numbers, remainders.
Long considered the purest of pure mathematics, it now protects every online
payment: the RSA cryptosystem rests on the theorems of Bézout, Gauss and
Fermat proved in this chapter.

\section{Divisibility and Euclidean division}

\begin{definition}[Divisibility]\label{def:g12:arith:divides}
Let $a, b \in \Z$. We say $b$ \emph{divides} $a$\index{divisibility}, written
$b \mid a$, if there exists $k \in \Z$ with $a = kb$. We also say $a$ is a
\emph{multiple} of $b$.
\end{definition}

\begin{proposition}\label{prop:g12:arith:divprops}
If $c \mid a$ and $c \mid b$, then $c$ divides every integer combination
$au + bv$ ($u, v \in \Z$). If $a \mid b$ and $b \mid a$ with
$a,b \in \N$, then $a = b$. If $a \mid b$ and $b \neq 0$, then $\abs a \leq \abs b$.
\end{proposition}

\begin{proof}
Write $a = kc$, $b = lc$: then $au + bv = (ku + lv)c$. The other points
follow from $\abs{a} = \abs{k}\,\abs{b}$ with $\abs k \geq 1$ when
$b = ka \neq 0$.
\end{proof}

\begin{theorem}[Euclidean division]\label{thm:g12:arith:euclid}
\index{Euclidean division}
Let $a \in \Z$ and $b \in \N^*$. There exists a unique pair
$(q, r) \in \Z \times \N$ such that
\[
a = bq + r \qquad\text{and}\qquad 0 \leq r < b .
\]
$q$ is the \emph{quotient} and $r$ the \emph{remainder}.
\end{theorem}

\begin{proof}
\emph{Existence.} The set of multiples of $b$ not exceeding $a$ has a
largest element $bq$ (it is nonempty and bounded above); set $r = a - bq$.
By maximality, $b(q+1) > a$, so $0 \leq r < b$.
\emph{Uniqueness.} If $bq + r = bq' + r'$ with $0 \leq r, r' < b$, then
$b(q - q') = r' - r$ and $\abs{r' - r} < b$: a multiple of $b$ of absolute
value less than $b$ must be $0$, so $r = r'$ and $q = q'$.
\end{proof}

\section{Congruences}

\begin{definition}[Congruence]\label{def:g12:arith:congruence}
Let $n \in \N^*$. Two integers $a, b$ are \emph{congruent modulo
$n$}\index{congruence}, written $a \equiv b \pmod n$, if $n \mid (a - b)$
--- equivalently, if $a$ and $b$ have the same remainder in the Euclidean
division by $n$.
\end{definition}

\begin{proposition}[Compatibility with operations]\label{prop:g12:arith:congops}
If $a \equiv b \pmod n$ and $c \equiv d \pmod n$, then
\[
a + c \equiv b + d, \qquad
ac \equiv bd, \qquad
a^k \equiv b^k \ (k \in \N) \pmod n .
\]
\end{proposition}

\begin{proof}
$n$ divides $(a-b) + (c-d) = (a+c) - (b+d)$, and
$ac - bd = a(c - d) + d(a - b)$ is also a multiple of $n$. The power rule
follows by induction from the product rule.
\end{proof}

\begin{method}[Computing powers modulo $n$]\label{met:g12:arith:powers}
To compute $a^k \bmod n$, reduce the base modulo $n$, then look for a small
power of $a$ congruent to $\pm1$, and use it to collapse the exponent. For
instance $2^{100} \bmod 7$: since $2^3 = 8 \equiv 1 \pmod 7$ and
$100 = 3\times33 + 1$,
\[
2^{100} = \left(2^{3}\right)^{33} \times 2 \equiv 1^{33}\times 2 = 2 \pmod 7 .
\]
\end{method}

\section{GCD, Bézout and Gauss}

\begin{definition}[GCD]\label{def:g12:arith:gcd}
Let $a, b$ be integers, not both zero. The \emph{greatest common
divisor}\index{gcd} $\gcd(a, b)$ is the largest integer dividing both $a$
and $b$. When $\gcd(a,b) = 1$, $a$ and $b$ are said to be
\emph{coprime}\index{coprime}.
\end{definition}

\begin{proposition}[Euclid's algorithm]\label{prop:g12:arith:euclidalgo}
\index{Euclid's algorithm}
If $a = bq + r$ ($b \neq 0$), then $\gcd(a, b) = \gcd(b, r)$. Iterating the
Euclidean division therefore computes $\gcd(a,b)$: the gcd is the last
nonzero remainder.
\end{proposition}

\begin{proof}
Any common divisor of $a$ and $b$ divides $r = a - bq$
(\cref{prop:g12:arith:divprops}), hence is a common divisor of $b$ and $r$;
and conversely, since $a = bq + r$. The two pairs have the same common
divisors, so the same gcd. The algorithm terminates because the remainders
form a strictly decreasing sequence of nonnegative integers.
\end{proof}

\begin{example}\label{ex:g12:arith:euclidalgo}
$\gcd(252, 198)$: $252 = 198 + 54$; $198 = 3\times54 + 36$;
$54 = 36 + 18$; $36 = 2 \times 18 + 0$. Hence $\gcd(252,198) = 18$.
\end{example}

\begin{theorem}[Bézout's identity]\label{thm:g12:arith:bezout}
\index{Bézout's identity}
Let $a, b$ be integers, not both zero, and $d = \gcd(a,b)$. There exist
$u, v \in \Z$ such that
\[
au + bv = d .
\]
In particular, $a$ and $b$ are coprime if and only if $au + bv = 1$ for some
integers $u, v$.
\end{theorem}

\begin{proof}
Run Euclid's algorithm backwards: each remainder is an integer combination
of the previous two, and the initial data $a, b$ are combinations of
themselves; by descending substitution, the last nonzero remainder $d$ is an
integer combination of $a$ and $b$. (In \cref{ex:g12:arith:euclidalgo}:
$18 = 54 - 36 = 54 - (198 - 3\times54) = 4\times54 - 198 =
4(252 - 198) - 198 = 4\times252 - 5\times198$.)

For the equivalence: if $\gcd(a,b) = 1$, Bézout provides $u, v$; conversely,
any common divisor of $a$ and $b$ divides $au + bv = 1$, forcing
$\gcd(a,b) = 1$.
\end{proof}

\begin{theorem}[Gauss's lemma]\label{thm:g12:arith:gauss}
\index{Gauss's lemma}
Let $a, b, c \in \Z$. If $a \mid bc$ and $\gcd(a, b) = 1$, then $a \mid c$.
\end{theorem}

\begin{proof}
Bézout gives $au + bv = 1$; multiply by $c$: $acu + bcv = c$. Both terms of
the left side are multiples of $a$ (the second because $a \mid bc$), hence
so is $c$.
\end{proof}

\begin{corollary}\label{cor:g12:arith:coprimeprod}
If $a \mid c$, $b \mid c$ and $\gcd(a,b) = 1$, then $ab \mid c$.
\end{corollary}

\begin{proof}
Write $c = ak$. From $b \mid ak$ and $\gcd(a,b)=1$, Gauss gives
$b \mid k$, say $k = bl$; then $c = abl$.
\end{proof}

\section{Prime numbers}

\begin{definition}[Prime]\label{def:g12:arith:prime}
An integer $p \geq 2$ is \emph{prime}\index{prime number} if its only
positive divisors are $1$ and $p$.
\end{definition}

\begin{proposition}\label{prop:g12:arith:primedivides}
Every integer $n \geq 2$ has a prime divisor; if $n$ is not prime, it has a
prime divisor $\leq \sqrt n$. If a prime $p$ divides a product $ab$, then
$p \mid a$ or $p \mid b$ (\emph{Euclid's lemma}).
\end{proposition}

\begin{proof}
The smallest divisor $d \geq 2$ of $n$ is prime (any proper divisor of $d$
would be a smaller divisor of $n$). If $n = de$ is composite with
$2 \leq d \leq e$, then $d^2 \leq de = n$, so $d \leq \sqrt n$. For Euclid's
lemma: if $p \nmid a$, then $\gcd(p, a) = 1$ (the only divisors of $p$ are
$1$ and $p$), and Gauss's lemma gives $p \mid b$.
\end{proof}

\begin{theorem}[Euclid]\label{thm:g12:arith:infiniteprimes}
There are infinitely many primes.
\end{theorem}

\begin{proof}
Given any finite list $p_1, \dots, p_k$ of primes, consider
$N = p_1 p_2 \cdots p_k + 1$. Some prime $p$ divides $N$; but no $p_i$
divides $N$ (the remainder is $1$), so $p$ is a prime not in the list. No
finite list exhausts the primes.
\end{proof}

\begin{theorem}[Fundamental theorem of arithmetic]\label{thm:g12:arith:fta}
\index{fundamental theorem of arithmetic}
Every integer $n \geq 2$ is a product of primes, and this factorization is
unique up to the order of the factors:
\[
n = p_1^{\alpha_1} p_2^{\alpha_2} \cdots p_r^{\alpha_r},
\qquad p_1 < p_2 < \dots < p_r \text{ primes},\ \alpha_i \geq 1 .
\]
\end{theorem}

\begin{proof}
\emph{Existence}, by strong induction: $n$ prime is its own factorization;
otherwise $n = de$ with $2 \leq d, e < n$, and both factor by the induction
hypothesis. \emph{Uniqueness}: suppose
$p_1\cdots p_s = q_1 \cdots q_t$ (primes, with repetitions allowed). By
Euclid's lemma, $p_1$ divides some $q_j$, and being prime, $p_1 = q_j$;
cancel and repeat. The two factorizations match term by term.
\end{proof}

\begin{theorem}[Fermat's little theorem]\label{thm:g12:arith:fermat}
\index{Fermat's little theorem}
Let $p$ be prime and $a \in \Z$ with $p \nmid a$. Then
\[
a^{p-1} \equiv 1 \pmod p .
\]
For every $a \in \Z$ (no coprimality assumed), $a^p \equiv a \pmod p$.
\end{theorem}

\begin{proof}
Consider the $p - 1$ integers $a, 2a, 3a, \dots, (p-1)a$ modulo $p$. None
is $\equiv 0$ (if $p \mid ka$ with $1 \leq k \leq p-1$, Euclid's lemma forces
$p \mid k$, impossible), and they are pairwise distinct modulo $p$ (if
$ka \equiv la$, then $p \mid (k - l)a$, so $p \mid k - l$, so $k = l$).
Hence, modulo $p$, they are the numbers $1, 2, \dots, p-1$ in some order.
Multiplying all congruences:
\[
a^{p-1}\,(p-1)! \equiv (p-1)! \pmod p .
\]
Since $p$ divides none of $1, \dots, p-1$, repeated use of Euclid's lemma
allows cancelling $(p-1)!$, leaving $a^{p-1} \equiv 1$. The second form
follows by multiplying by $a$ (and is trivial when $p \mid a$).
\end{proof}

\begin{example}[Application to cryptography]\label{ex:g12:arith:rsa}
Fermat's theorem makes exponentiation modulo $n$ reversible when the
exponents are chosen suitably --- the heart of the \emph{RSA} cryptosystem.
With $p, q$ large primes and $n = pq$, one publishes $n$ and an exponent
$e$; encryption is $x \mapsto x^e \bmod n$. Decrypting requires an exponent
$d$ with $ed \equiv 1 \pmod{(p-1)(q-1)}$, which only someone knowing $p$
and $q$ can compute --- and recovering $p, q$ from $n$ means factoring a
number hundreds of digits long, which no known algorithm does in reasonable
time.
\end{example}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:g12:arith:1}
Compute the quotient and remainder of the Euclidean division of $2026$ by
$17$, and of $-2026$ by $17$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:arith:2}
What is the remainder of $7^{100}$ modulo $10$? (What is the last digit of
$7^{100}$?)
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:arith:3}
Using Euclid's algorithm, compute $\gcd(1071, 462)$, and find integers
$u, v$ with $1071u + 462v = \gcd(1071, 462)$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:arith:4}
Show that for every $n \in \Z$, $n^2$ is congruent to $0$ or $1$ modulo
$4$. Deduce that an integer $\equiv 3 \pmod 4$ is never a sum of two
squares.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:arith:5}
Show that for all $n \in \N$, $n(n+1)(2n+1)$ is divisible by $6$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:arith:6}
Solve in $\Z$ the congruence $5x \equiv 3 \pmod{11}$.
(Hint: find the inverse of $5$ modulo $11$.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:arith:7}
Solve in $\Z \times \Z$ the Diophantine equation
\[
17x - 40y = 1,
\]
then describe all solutions of $17x - 40y = 6$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:arith:8}
Show that $\sqrt2$ is irrational, using the uniqueness of prime
factorization (compare the exponent of $2$ on both sides of $a^2 = 2b^2$).
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:arith:9}
Let $p$ be a prime.
\begin{enumerate}
  \item Show that for $1 \leq k \leq p - 1$, $p$ divides $\dbinom{p}{k}$.
        (Hint: use $k\binom pk = p\binom{p-1}{k-1}$, \cref{exo:g12:comb:7},
        and Gauss's lemma.)
  \item Deduce, by induction on $a \geq 0$, another proof of Fermat's little
        theorem in the form $a^p \equiv a \pmod p$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:arith:10}
\emph{(Chinese remainder problem.)} Find all integers $n$ such that
\[
n \equiv 2 \pmod 3, \qquad n \equiv 3 \pmod 5, \qquad n \equiv 2 \pmod 7 .
\]
(Hint: solve the first two conditions, then incorporate the third; Bézout
coefficients help.)
\end{exercise}
