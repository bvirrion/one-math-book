\chapter{Équations différentielles}\label{ch:g12:diffeq}

Une \omterm{def:g12:diffeq:ode}{équation différentielle} relie une \omterm{def:g11:func:function}{fonction} à ses \omterm{def:g12:deriv:derivative}{dérivées}. La physique,
la chimie, la biologie et l'économie expriment leurs \omterm{def:g11:prob:rv}{lois} sous cette forme~:
la vitesse de refroidissement d'un corps, la désintégration d'un noyau
radioactif, la croissance d'une population sont toutes des énoncés sur
$y'$. Ce chapitre résout complètement les \omterm{def:g10:algebra:equation}{équations} linéaires du premier
ordre à coefficients constants.

\section{L'équation \texorpdfstring{$y' = ay$}{y' = ay}}

\begin{definition}[Équation différentielle]\label{def:g12:diffeq:ode}
Une \emph{équation différentielle}\index{équation différentielle} est une
\omterm{def:g10:algebra:equation}{équation} dont l'inconnue est une \omterm{def:g11:func:function}{fonction} $y$, et qui fait intervenir $y$ et
ses \omterm{def:g12:deriv:derivative}{dérivées}. Une \emph{solution} sur un \omterm{def:g10:numbers:interval}{intervalle} $I$ est une \omterm{def:g11:func:function}{fonction}
\omterm{def:g12:deriv:derivative}{dérivable} vérifiant l'\omterm{def:g10:algebra:equation}{équation} en tout point de $I$.
\end{definition}

\begin{theorem}[Solutions de $y' = ay$]\label{thm:g12:diffeq:homogeneous}
Soit $a \in \R$. Les solutions sur $\R$ de l'\omterm{def:g10:algebra:equation}{équation} $y' = ay$ sont
exactement les \omterm{def:g11:func:function}{fonctions}
\[
y(x) = C\,\eu^{ax}, \qquad C \in \R .
\]
Pour tout couple $(x_0, y_0)$ il existe une unique solution avec
$y(x_0) = y_0$.
\end{theorem}

\begin{proof}
Chaque $y = C\eu^{ax}$ vérifie $y' = Ca\,\eu^{ax} = ay$. Réciproquement, soit
$y$ une solution quelconque et posons $z(x) = y(x)\,\eu^{-ax}$. Alors
\[
z'(x) = y'(x)\,\eu^{-ax} - a\,y(x)\,\eu^{-ax}
= \bigl(y'(x) - a y(x)\bigr)\eu^{-ax} = 0,
\]
donc $z$ est constante, disons $C$, et $y(x) = C\eu^{ax}$. La condition
initiale $y(x_0) = y_0$ impose $C = y_0 \eu^{-a x_0}$, de façon unique.
\end{proof}

\begin{remark}
L'exponentielle est ainsi \emph{caractérisée} par la plus simple de toutes
les \omterm{def:g12:diffeq:ode}{équations différentielles}~: une croissance proportionnelle à la taille.
C'est pourquoi elle apparaît partout dans la nature.
\end{remark}

\begin{example}[Désintégration radioactive]\label{ex:g12:diffeq:decay}
Une quantité radioactive vérifie $N' = -\lambda N$ avec $\lambda > 0$, donc
$N(t) = N_0\,\eu^{-\lambda t}$~; voir \cref{ex:g12:exp:halflife} pour la
demi-vie.
\end{example}

\section{L'équation \texorpdfstring{$y' = ay + b$}{y' = ay + b}}

\begin{theorem}[Solutions de $y' = ay + b$]\label{thm:g12:diffeq:affine}
Soient $a \neq 0$ et $b \in \R$. Les solutions sur $\R$ de $y' = ay + b$
sont exactement les \omterm{def:g11:func:function}{fonctions}
\[
y(x) = C\,\eu^{ax} - \frac{b}{a}, \qquad C \in \R .
\]
La \omterm{def:g11:func:function}{fonction} constante $y_p = -\frac ba$ est la \emph{solution
d'équilibre}. Pour tout $(x_0, y_0)$ il existe une unique solution avec
$y(x_0) = y_0$~; si $a < 0$, toute solution tend vers l'équilibre
$-\frac ba$ lorsque $x \to +\infty$.
\end{theorem}

\begin{proof}
La constante $y_p = -\frac{b}{a}$ vérifie $y_p' = 0 = a y_p + b$. Or $y$ est
solution si et seulement si
\[
(y - y_p)' = y' = ay + b = a(y - y_p) + \underbrace{a y_p + b}_{=\,0}
= a (y - y_p),
\]
c'est-à-dire si et seulement si $z = y - y_p$ résout $z' = az$. Par le
\cref{thm:g12:diffeq:homogeneous}, $z = C\eu^{ax}$, d'où la formule,
l'existence et l'unicité. Si $a < 0$, $\eu^{ax} \to 0$ lorsque
$x \to +\infty$, donc $y(x) \to -\frac ba$.
\end{proof}

\begin{method}[Résoudre $y' = ay + b$ avec condition initiale]\label{met:g12:diffeq:solve}
\begin{enumerate}
  \item Trouver la solution d'équilibre $y_p = -\frac{b}{a}$ (résoudre
        $y' = 0$).
  \item Écrire la solution générale $y = C\eu^{ax} + y_p$.
  \item Déterminer $C$ grâce à la condition initiale.
  \item Vérifier le comportement à long terme face à l'intuition physique
        (la solution converge-t-elle vers l'équilibre~?).
\end{enumerate}
La même stratégie --- \emph{solution particulière + solution générale de
l'\omterm{def:g10:algebra:equation}{équation} homogène} --- s'étend à $y' = ay + f(x)$~: voir la
\cref{prop:g12:diffeq:general}.
\end{method}

\begin{example}[Loi de refroidissement de Newton]\label{ex:g12:diffeq:cooling}
Une tasse de café à $80\,^\circ$C se trouve dans une pièce à $20\,^\circ$C. La \omterm{def:g11:prob:rv}{loi}
de Newton affirme que la température $T$ vérifie $T' = -k(T - 20)$ pour un
certain $k > 0$, \emph{c.-à-d.} $T' = -kT + 20k$. L'équilibre est $20$, et
$T(t) = 20 + 60\,\eu^{-kt}$~: le café refroidit exponentiellement vers la
température ambiante.
\end{example}

\section{L'équation \texorpdfstring{$y' = ay + f(x)$}{y' = ay + f(x)}}

\begin{proposition}[Structure de l'ensemble des solutions]\label{prop:g12:diffeq:general}
Soit $f$ \omterm{def:g12:limcont:continuity}{continue} sur un \omterm{def:g10:numbers:interval}{intervalle} $I$ et soit $y_p$ une solution
particulière de
\[
y' = ay + f(x)
\]
sur $I$. Alors les solutions sur $I$ sont exactement les \omterm{def:g11:func:function}{fonctions}
$y = C\eu^{ax} + y_p$, $C \in \R$.
\end{proposition}

\begin{proof}
Comme dans le \cref{thm:g12:diffeq:affine}~: $y$ est solution si et seulement
si $z = y - y_p$ vérifie
$z' = (ay + f) - (ay_p + f) = az$, si et seulement si $z = C\eu^{ax}$.
\end{proof}

\begin{method}[Deviner une solution particulière]\label{met:g12:diffeq:particular}
Chercher une solution particulière \emph{de même forme} que $f$~:
\begin{itemize}
  \item $f$ polynôme de degré $n$~: essayer un polynôme de degré $n$~;
  \item $f(x) = \alpha\,\eu^{kx}$ avec $k \neq a$~: essayer
        $y_p = \beta\,\eu^{kx}$~;
  \item $f(x) = \alpha\,\eu^{ax}$ (cas résonant)~: essayer
        $y_p = \beta x\,\eu^{ax}$.
\end{itemize}
Substituer dans l'\omterm{def:g10:algebra:equation}{équation} et identifier les coefficients.
\end{method}

\begin{example}\label{ex:g12:diffeq:particular}
Résoudre $y' = 2y + 4x$. Essayer $y_p = \alpha x + \beta$~:
$\alpha = 2(\alpha x + \beta) + 4x$ pour tout $x$ impose
$2\alpha + 4 = 0$ et $\alpha = 2\beta$, donc $\alpha = -2$, $\beta = -1$.
Solution générale~: $y = C\eu^{2x} - 2x - 1$.
\end{example}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:g12:diffeq:1}
Résoudre sur $\R$~:
(a) $y' = 3y$ avec $y(0) = 2$~;\quad
(b) $2y' + y = 0$ avec $y(0) = -1$~;\quad
(c) $y' = -y + 5$ avec $y(0) = 0$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:diffeq:2}
Une population bactérienne croît à un taux proportionnel à sa taille, en
doublant toutes les $3$~heures. Écrire l'\omterm{def:g12:diffeq:ode}{équation différentielle} satisfaite
par la population $N(t)$ et déterminer la constante de proportionnalité.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:diffeq:3}
Vérifier que $y_p(x) = x\,\eu^{x}$ est solution de $y' = y + \eu^x$, et
donner toutes les solutions sur $\R$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:diffeq:4}
Résoudre $y' = -2y + \eu^{x}$ avec $y(0) = 1$.
(Indication~: chercher une solution particulière de la forme
$\beta\,\eu^{x}$.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:diffeq:5}
Le carbone~14 se désintègre avec une demi-vie de $5730$~ans. Un \omterm{def:g10:proba:sample}{échantillon}
archéologique contient $60\%$ du carbone~14 d'un organisme vivant. Estimer
son âge.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:diffeq:6}
Un réservoir contient $100$~L d'eau pure. De la saumure contenant $0.2$~kg de
sel par litre entre à $5$~L/min, le mélange (maintenu homogène) sort au même
débit. Soit $m(t)$ la masse de sel dans le réservoir à l'instant $t$ (en
minutes).
\begin{enumerate}
  \item Justifier que $m' = 1 - \dfrac{m}{20}$.
  \item Résoudre, et déterminer la limite de $m(t)$ lorsque
        $t \to +\infty$. Interpréter.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:diffeq:7}
Un parachutiste de masse $80$~kg tombe sous l'effet de la gravité
($g = 9.8\ \text{m/s}^2$) et d'une résistance de l'air proportionnelle à la
vitesse, de sorte que sa vitesse vérifie $v' = g - \frac{k}{m}v$ avec
$k = 16$~kg/s.
\begin{enumerate}
  \item Résoudre l'\omterm{def:g10:algebra:equation}{équation} avec $v(0) = 0$.
  \item Calculer la \emph{vitesse limite} $\lim_{t\to+\infty} v(t)$, et le
        temps nécessaire pour atteindre $95\%$ de celle-ci.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:diffeq:8}
\emph{(\omterm{def:g10:algebra:equation}{Équation} logistique.)} Une population $y(t) \in \intoo{0}{1}$ (comme
fraction de la population maximale) vérifie
\[
y' = y(1 - y).
\]
\begin{enumerate}
  \item Soit $z = \dfrac{1}{y}$. Montrer que $z$ satisfait l'\omterm{def:g10:algebra:equation}{équation}
        linéaire $z' = -z + 1$.
  \item Résoudre pour $z$, puis pour $y$, avec $y(0) = \frac{1}{10}$.
  \item Montrer que $y(t) \to 1$ lorsque $t \to +\infty$ et esquisser
        l'allure de la courbe solution.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:diffeq:9}
Soit $y$ une solution de $y' = ay + b$ et soit $u_n = y(n)$ pour
$n \in \N$. Montrer que $(u_n)$ satisfait une récurrence
arithmético-géométrique $u_{n+1} = q u_n + r$, et exprimer $q$ et $r$ en
\omterm{def:g11:func:function}{fonction} de $a$ et $b$. À quoi correspond la condition $\abs{q} < 1$ pour
l'\omterm{def:g12:diffeq:ode}{équation différentielle}~?
\end{exercise}
