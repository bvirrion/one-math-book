\chapter{Combinatoire et dénombrement}\label{ch:g12:comb}

La combinatoire est l'art de compter sans énumérer. Ses deux principes
élémentaires --- ajouter les cardinaux d'alternatives disjointes, multiplier
les nombres de choix indépendants --- suffisent à compter les \omterm{def:g12:comb:tuples}{arrangements},
les \omterm{def:g12:comb:permutation}{permutations} et les parties d'un ensemble fini, et culminent dans la
formule du binôme.

\section{Les deux principes de dénombrement}

On note $\abs{E}$ le nombre d'éléments (le \emph{cardinal}) d'un ensemble
fini $E$.

\begin{proposition}[Principe d'addition]\label{prop:g12:comb:addition}
Si un ensemble fini $E$ est partitionné en sous-ensembles
$A_1, \dots, A_k$ (deux à deux disjoints, d'union $E$), alors
\[
\abs{E} = \abs{A_1} + \abs{A_2} + \dots + \abs{A_k}.
\]
\end{proposition}

\begin{proposition}[Principe de multiplication]\label{prop:g12:comb:multiplication}
Si un objet est construit par une succession de $k$ choix, avec $n_1$
options pour le premier choix et, \emph{quels que soient les choix
précédents}, $n_i$ options pour le $i$-ième, alors le nombre d'objets
construits est $n_1 \times n_2 \times \dots \times n_k$.
\end{proposition}

\begin{proof}
Les deux énoncés se prouvent par récurrence sur $k$~; le cas $k = 2$ du
second revient à compter un tableau rectangulaire par lignes.
\end{proof}

\begin{example}
Un restaurant propose 4 entrées, 6 plats, 3 desserts~: $4 \times 6 \times 3
= 72$ menus à trois plats différents.
\end{example}

\section{Listes, permutations, factorielles}

\begin{definition}[$k$-uplets]\label{def:g12:comb:tuples}
Un \emph{$k$-uplet} d'un ensemble $E$ est une liste ordonnée
$(x_1, \dots, x_k)$ d'éléments de $E$, répétitions autorisées. Un
$k$-uplet d'éléments \emph{distincts} est un
\emph{arrangement}\index{arrangement} de $k$ éléments de $E$.
\end{definition}

\begin{proposition}\label{prop:g12:comb:tuples}
Soit $\abs E = n$. Le nombre de $k$-uplets de $E$ est $n^k$. Le nombre
d'\omterm{def:g12:comb:tuples}{arrangements} de $k$ éléments de $E$ ($0 \leq k \leq n$) est
\[
n(n-1)(n-2)\cdots(n-k+1) = \frac{n!}{(n-k)!},
\]
où $n! = 1 \times 2 \times \dots \times n$ (et $0! = 1$) est la
\emph{factorielle}\index{factorielle} de $n$.
\end{proposition}

\begin{proof}
Principe de multiplication~: pour un $k$-uplet il y a $n$ options à chacune
des $k$ étapes~; pour un \omterm{def:g12:comb:tuples}{arrangement}, $n$ options pour $x_1$, puis $n - 1$
pour $x_2$ (un élément est déjà pris), \ldots, $n - k + 1$ pour $x_k$.
\end{proof}

\begin{definition}[Permutation]\label{def:g12:comb:permutation}
Une \emph{permutation}\index{permutation} de $E$ est un \omterm{def:g12:comb:tuples}{arrangement} des $n$
éléments de $E$~: un ordre total de $E$. Par la
\cref{prop:g12:comb:tuples} (cas $k = n$), le nombre de permutations d'un
ensemble à $n$ éléments est $n!$.
\end{definition}

\begin{example}
Cinq coureurs peuvent finir une course dans $5! = 120$ ordres différents.
Le nombre de podiums possibles (trois premières places) est
$5 \times 4 \times 3 = 60$.
\end{example}

\section{Combinaisons et coefficients binomiaux}

\begin{definition}[Combinaisons]\label{def:g12:comb:combination}
Une \emph{combinaison}\index{combinaison} de $k$ éléments de $E$ est une
partie de $E$ à $k$ éléments (sans ordre, sans répétition). Leur nombre
s'écrit $\dbinom{n}{k}$\index{coefficient binomial}, lu «~$n$ parmi $k$~»
ou «~$k$ parmi $n$~».
\end{definition}

\begin{theorem}\label{thm:g12:comb:binomformula}
Pour $0 \leq k \leq n$~:
\[
\binom{n}{k} = \frac{n!}{k!\,(n-k)!} .
\]
\end{theorem}

\begin{proof}
Compter les \omterm{def:g12:comb:tuples}{arrangements} de $k$ éléments de $E$ de deux manières.
Directement~: $\frac{n!}{(n-k)!}$. Alternativement, choisir d'abord le
sous-ensemble sous-jacent ($\binom nk$ façons), puis l'ordonner ($k!$
façons)~; le principe de multiplication donne $\binom{n}{k}\,k!$. En
égalant, $\binom nk = \frac{n!}{k!(n-k)!}$.
\end{proof}

\begin{proposition}[Identités de base]\label{prop:g12:comb:identities}
Pour $0 \leq k \leq n$~:
\[
\binom{n}{0} = \binom{n}{n} = 1, \qquad
\binom{n}{1} = n, \qquad
\binom{n}{k} = \binom{n}{n-k},
\]
et la \emph{règle de Pascal}\index{triangle de Pascal}~: pour
$1 \leq k \leq n-1$,
\[
\binom{n}{k} = \binom{n-1}{k-1} + \binom{n-1}{k}.
\]
\end{proposition}

\begin{proof}
La symétrie $\binom nk = \binom{n}{n-k}$ vaut car le passage au
\omterm{def:g10:proba:operations}{complémentaire} met en bijection les parties à $k$ éléments avec celles à
$(n-k)$ éléments. Pour la \omterm{prop:g12:comb:identities}{règle de Pascal}, fixer un élément $a \in E$ et
trier les parties à $k$ éléments en celles contenant $a$ --- obtenues en
adjoignant $a$ à une partie à $(k-1)$ éléments de $E \setminus \{a\}$, au
nombre de $\binom{n-1}{k-1}$ --- et celles évitant $a$, qui sont les
parties à $k$ éléments de $E \setminus \{a\}$, au nombre de
$\binom{n-1}{k}$. Conclure par le principe d'addition.
\end{proof}

La \omterm{prop:g12:comb:identities}{règle de Pascal} engendre les coefficients ligne par ligne --- le
\emph{\omterm{prop:g11:binom:pascal}{triangle de Pascal}}~: chaque entrée est la somme des deux au-dessus
d'elle.

\begin{theorem}[Formule du binôme]\label{thm:g12:comb:binomial}
\index{formule du binôme}
Pour tous $a, b \in \R$ (ou $\C$) et $n \in \N$~:
\[
(a+b)^n = \sum_{k=0}^{n} \binom{n}{k}\, a^{k}\, b^{\,n-k} .
\]
\end{theorem}

\begin{proof}
Développer le produit $(a+b)(a+b)\cdots(a+b)$ ($n$ facteurs)~: chaque terme
du développement choisit $a$ ou $b$ dans chaque facteur, produisant
$a^k b^{n-k}$ où $k$ est le nombre de facteurs contribuant $a$. Le nombre
de façons de choisir ces $k$ facteurs parmi $n$ est $\binom nk$, qui est
donc le coefficient de $a^k b^{n-k}$.
\end{proof}

\begin{corollary}\label{cor:g12:comb:sums}
$\displaystyle\sum_{k=0}^{n} \binom{n}{k} = 2^n$
\quad et \quad
$\displaystyle\sum_{k=0}^{n} (-1)^k\binom{n}{k} = 0$ \ ($n \geq 1$).
\end{corollary}

\begin{proof}
Prendre $a = b = 1$, puis $a = -1$, $b = 1$ dans la formule du binôme. La
première identité a aussi un sens direct~: un ensemble à $n$ éléments a
$2^n$ parties (chaque élément est dedans ou dehors~: principe de
multiplication), triées par taille.
\end{proof}

\begin{method}[Choisir le bon modèle]\label{met:g12:comb:model}
Avant de compter, répondre à deux questions~: \emph{l'ordre compte-t-il~?}
et \emph{les répétitions sont-elles autorisées~?}
\begin{center}
\begin{tabular}{l|cc}
 & l'ordre compte & l'ordre n'importe pas\\
\hline
répétitions autorisées & $n^k$ (listes) & (université)\\
sans répétition & $\frac{n!}{(n-k)!}$ (\omterm{def:g12:comb:tuples}{arrangements}) & $\binom nk$ (parties)
\end{tabular}
\end{center}
Tirer des boules d'une urne~: \emph{avec remise, dans l'ordre} $\to$
listes~; \emph{sans remise, dans l'ordre} $\to$ \omterm{def:g12:comb:tuples}{arrangements}~; \emph{une
poignée d'un coup} $\to$ parties.
\end{method}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:g12:comb:1}
Une plaque d'immatriculation se compose de 2 lettres (A--Z), puis 3
chiffres, puis 2 lettres. Combien de plaques sont possibles~? Combien n'ont
aucun caractère répété~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:comb:2}
Calculer $\dbinom{8}{3}$, $\dbinom{10}{8}$, et simplifier
$\dfrac{\binom{n}{2}}{\binom{n+1}{2}}$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:comb:3}
Dans une classe de 30 élèves, il faut élire un comité de 4 élèves, puis un
président et un trésorier au sein du comité (une même personne ne peut
occuper les deux postes). Combien d'issues sont possibles~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:comb:4}
Développer $(x + 2)^5$ et $(1 - x)^6$ à l'aide de la formule du binôme. Quel
est le coefficient de $x^3$ dans $(2x + 3)^7$~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:comb:5}
Une main de poker standard se compose de 5 cartes tirées d'un jeu de 52
cartes.
\begin{enumerate}
  \item Combien y a-t-il de mains~?
  \item Combien de mains contiennent exactement un as~? Au moins un as~?
  \item Combien de mains sont des «~fulls~» (trois cartes d'une valeur, deux
        d'une autre)~?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:comb:6}
Combien d'anagrammes (réarrangements de lettres, sensés ou non) le mot MATH
possède-t-il~? Le mot BANANA~?
(Indication pour BANANA~: placer d'abord les trois A.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:comb:7}
Prouver l'identité $k\dbinom{n}{k} = n\dbinom{n-1}{k-1}$
($1 \leq k \leq n$) de deux façons~: par la formule \omterm{prop:g12:comb:tuples}{factorielle}, et en
comptant de deux manières les couples (comité de $k$ personnes, son
président) choisis parmi $n$ personnes.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:comb:8}
Un chemin dans le plan va de $(0,0)$ à $(m, n)$ par des pas unitaires Est ou
Nord. Montrer que le nombre de tels chemins est $\dbinom{m+n}{m}$.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:comb:9}
Prouver l'\emph{identité de Vandermonde}~: pour $0 \leq k \leq m + n$,
\[
\binom{m+n}{k} = \sum_{j=0}^{k} \binom{m}{j}\binom{n}{k-j},
\]
en comptant les parties à $k$ éléments d'un ensemble scindé en un groupe de
$m$ et un groupe de $n$. En déduire que
$\displaystyle\sum_{j=0}^{n}\binom{n}{j}^{\!2} = \binom{2n}{n}$.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:comb:10}
À l'aide de la formule du binôme, montrer que pour tout $n \geq 1$,
\[
\sum_{k=1}^{n} k \binom{n}{k} = n\,2^{n-1}.
\]
(Indication~: soit dériver $(1+x)^n$, soit utiliser
l'\cref{exo:g12:comb:7}.)
\end{exercise}
