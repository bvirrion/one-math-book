\chapter{Variables aléatoires et loi binomiale}\label{ch:g12:randvar}

Une \omterm{def:g12:randvar:rv}{variable aléatoire} attache un nombre à chaque issue d'une expérience~:
un gain, un compte, une durée. Son \omterm{def:g12:randvar:exp}{espérance} est la \omterm{def:g11:stat:mean}{moyenne} à long terme
des valeurs qu'elle produit, sa \omterm{def:g12:randvar:exp}{variance} mesure leur dispersion. La star de
ce chapitre est la \omterm{def:g12:randvar:binomial}{loi binomiale}, qui compte les succès dans des épreuves
indépendantes répétées. Les \omterm{def:g12:randvar:rv}{variables aléatoires} et la \omterm{def:g12:randvar:binomial}{loi binomiale} ont
été rencontrées dans les
\cref{ch:g11:prob,ch:g11:binom}~; ce chapitre les revoit et les approfondit,
avec les outils combinatoires du \cref{ch:g12:comb} désormais disponibles.

\section{Variables aléatoires discrètes}

\begin{definition}[Variable aléatoire, loi]\label{def:g12:randvar:rv}
Une \emph{variable aléatoire}\index{variable aléatoire} sur un \omterm{def:g11:prob:model}{univers} fini
$\Omega$ est une \omterm{def:g11:func:function}{fonction} $X \colon \Omega \to \R$. Sa
\emph{loi}\index{loi} (ou distribution) est la donnée de ses valeurs
possibles $x_1, \dots, x_k$ et des \omterm{def:g10:proba:distribution}{probabilités}
\[
p_i = \P(X = x_i), \qquad \sum_{i=1}^{k} p_i = 1 .
\]
\end{definition}

\begin{definition}[Espérance, variance, écart-type]\label{def:g12:randvar:exp}
L'\emph{espérance}\index{espérance} de $X$ est
\[
\E(X) = \sum_{i=1}^{k} p_i\, x_i ,
\]
sa \emph{variance}\index{variance} et son \emph{écart-type} sont
\[
\V(X) = \E\bigl((X - \E(X))^2\bigr) = \sum_{i=1}^k p_i\bigl(x_i - \E(X)\bigr)^2,
\qquad
\sigma(X) = \sqrt{\V(X)} .
\]
\end{definition}

\begin{proposition}[Formule de König--Huygens]\label{prop:g12:randvar:konig}
\index{formule de König--Huygens}
$\V(X) = \E(X^2) - \E(X)^2$.
\end{proposition}

\begin{proof}
Écrire $m = \E(X)$ et \omterm{def:g10:algebra:expand}{développer}~:
\[
\V(X) = \sum_i p_i (x_i^2 - 2m x_i + m^2)
= \E(X^2) - 2m\sum_i p_i x_i + m^2\sum_i p_i
= \E(X^2) - 2m^2 + m^2 . \qedhere
\]
\end{proof}

\begin{proposition}[Transformation affine]\label{prop:g12:randvar:affine}
Pour $a, b \in \R$~:
\[
\E(aX + b) = a\,\E(X) + b, \qquad
\V(aX + b) = a^2\,\V(X) .
\]
\end{proposition}

\begin{proof}
La première est un réarrangement de la somme définissante. Pour la seconde,
$aX + b$ s'écarte de sa \omterm{def:g11:stat:mean}{moyenne} de $a(X - \E(X))$, et le carré multiplie par
$a^2$.
\end{proof}

\begin{example}[Jeux équitables]\label{ex:g12:randvar:game}
Un jeu coûte $2$ euros~; on lance un dé, et le joueur reçoit la valeur
affichée si elle est au moins $5$, rien sinon. Le gain $G$ prend les valeurs
$-2$ (\omterm{def:g10:proba:distribution}{probabilité} $\frac46$), $3$ ($\frac16$), $4$ ($\frac16$)~:
\[
\E(G) = \frac{-8 + 3 + 4}{6} = -\frac{1}{6} < 0 .
\]
En \omterm{def:g11:stat:mean}{moyenne}, le joueur perd $17$ centimes par partie~: le jeu est défavorable
(comme la plupart des jeux réels).
\end{example}

\section{Épreuves de Bernoulli et loi binomiale}

\begin{definition}[Loi de Bernoulli]\label{def:g12:randvar:bernoulli}
Une \emph{épreuve de Bernoulli}\index{épreuve de Bernoulli} est une
expérience à deux issues, \emph{succès} (\omterm{def:g10:proba:distribution}{probabilité} $p$) et
\emph{échec} ($q = 1 - p$). L'indicatrice $X$ de succès ($X = 1$ en cas de
succès, $0$ en cas d'échec) suit la \emph{\omterm{def:g12:randvar:rv}{loi} de Bernoulli}
$\mathcal B(p)$~:
\[
\E(X) = p, \qquad \V(X) = p(1-p) .
\]
(En effet $\E(X) = p$, $\E(X^2) = p$, et König--Huygens donne
$\V(X) = p - p^2$.)
\end{definition}

\begin{definition}[Loi binomiale]\label{def:g12:randvar:binomial}
Répéter une \omterm{def:g12:randvar:bernoulli}{épreuve de Bernoulli} $n$ fois indépendamment, et soit $X$ le
nombre total de succès. La \omterm{def:g12:randvar:rv}{loi} de $X$ est la \emph{loi
binomiale}\index{loi binomiale} $\mathcal B(n, p)$.
\end{definition}

\begin{theorem}\label{thm:g12:randvar:binomial}
Si $X \sim \mathcal B(n, p)$, alors pour $0 \leq k \leq n$~:
\[
\P(X = k) = \binom{n}{k} p^k (1-p)^{n-k},
\]
et
\[
\E(X) = np, \qquad \V(X) = np(1-p) .
\]
\end{theorem}

\begin{proof}
Une \omterm{def:g12:seq:sequence}{suite} spécifiée d'issues avec $k$ succès et $n-k$ échecs a pour
\omterm{def:g10:proba:distribution}{probabilité} $p^k(1-p)^{n-k}$ par \omterm{def:g12:condprob:indep}{indépendance}~; le nombre de telles \omterm{def:g12:seq:sequence}{suites}
est le nombre de façons de placer les $k$ succès parmi les $n$ épreuves,
soit $\binom nk$ (\cref{ch:g12:comb}). En sommant sur les \omterm{def:g12:seq:sequence}{suites} on obtient
la formule --- et la formule du binôme confirme
$\sum_k \P(X = k) = (p + q)^n = 1$.

Pour l'\omterm{def:g12:randvar:exp}{espérance}, en utilisant $k\binom nk = n\binom{n-1}{k-1}$
(\cref{exo:g12:comb:7})~:
\[
\E(X) = \sum_{k=1}^{n} k\binom nk p^k q^{n-k}
= np\sum_{k=1}^{n} \binom{n-1}{k-1} p^{k-1} q^{(n-1)-(k-1)}
= np\,(p + q)^{n-1} = np .
\]
La formule de la \omterm{def:g12:randvar:exp}{variance} se prouve de façon analogue avec l'identité
$k(k-1)\binom nk = n(n-1)\binom{n-2}{k-2}$, donnant
$\E(X(X-1)) = n(n-1)p^2$, d'où
$\V(X) = n(n-1)p^2 + np - (np)^2 = np(1-p)$. (Une preuve structurelle ---
la \omterm{def:g12:randvar:exp}{variance} d'une somme de variables indépendantes --- arrive avec
\cref{thm:g12:sums:variance}.)
\end{proof}

\begin{method}[Reconnaître une situation binomiale]\label{met:g12:randvar:recognize}
Vérifier les trois ingrédients avant d'écrire $X \sim \mathcal B(n,p)$~: un
\emph{nombre fixé} $n$ d'épreuves~; \emph{deux issues} par épreuve avec la
\emph{même} \omterm{def:g10:proba:distribution}{probabilité} de succès $p$~; \emph{\omterm{def:g12:condprob:indep}{indépendance}} des épreuves
(tirage \emph{avec} remise, ou dans une grande population). Puis utiliser
\[
\P(X \geq 1) = 1 - (1-p)^n
\]
pour «~au moins un succès~», et une calculatrice ou des tables cumulatives
pour $\P(X \leq k)$ en général.
\end{method}

\begin{example}\label{ex:g12:randvar:atleastone}
Combien de fois faut-il lancer un dé pour avoir au moins $99\%$ de chance
d'obtenir un six~? Avec $X \sim \mathcal B\left(n, \frac16\right)$~:
$\P(X \geq 1) = 1 - \left(\frac56\right)^n \geq 0.99$ signifie
$\left(\frac56\right)^n \leq 0.01$, \emph{c'est-à-dire}
$n \geq \frac{\ln 0.01}{\ln(5/6)} \approx 25.3$~: à partir de $n = 26$
lancers.
\end{example}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:g12:randvar:1}
Une \omterm{def:g12:randvar:rv}{variable aléatoire} $X$ prend les valeurs $-1, 0, 2, 5$ avec les
\omterm{def:g10:proba:distribution}{probabilités} $0.3, 0.2, 0.4, 0.1$. Calculer $\E(X)$, $\V(X)$ et
$\sigma(X)$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:randvar:2}
Un QCM a $10$ questions, chacune à $4$ choix dont un seul est correct. Un
élève répond uniformément au hasard, indépendamment. Soit $X$ le nombre de
réponses correctes.
\begin{enumerate}
  \item Donner la \omterm{def:g12:randvar:rv}{loi} de $X$, $\E(X)$ et $\sigma(X)$.
  \item Calculer $\P(X = 0)$, $\P(X = 5)$ et $\P(X \geq 1)$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:randvar:3}
Une compagnie d'assurance assure $n = 400$ clients~; chacun déclare un
sinistre dans l'année avec \omterm{def:g10:proba:distribution}{probabilité} $p = 0.05$, indépendamment. Soit $X$
le nombre de sinistres. Identifier la \omterm{def:g12:randvar:rv}{loi} de $X$ et calculer son \omterm{def:g12:randvar:exp}{espérance}
et son écart-type.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:randvar:4}
Dans le jeu de l'\cref{ex:g12:randvar:game}, l'organisateur veut un jeu
équitable ($\E(G) = 0$) en changeant le prix d'entrée $c$. Trouver $c$.
Calculer la \omterm{def:g12:randvar:exp}{variance} du gain pour cette version équitable~; «~équitable~»
est-il la même chose que «~sans risque~»~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:randvar:5}
Un joueur de basket marque ses lancers francs avec \omterm{def:g10:proba:distribution}{probabilité} $0.7$. Elle
tire $8$ fois (tirs \omterm{def:g12:condprob:indep}{indépendants}). Calculer la \omterm{def:g10:proba:distribution}{probabilité} qu'elle marque~:
exactement $6$~; au moins $6$~; au moins une fois. Quel est le nombre le
plus probable de paniers~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:randvar:6}
Une compagnie aérienne sait que chaque passager ayant réservé se présente
avec \omterm{def:g10:proba:distribution}{probabilité} $0.9$, indépendamment. Un vol a $100$ places et la
compagnie vend $104$ billets. Exprimer, à l'aide d'une \omterm{def:g12:randvar:binomial}{loi binomiale}, la
\omterm{def:g10:proba:distribution}{probabilité} que plus de passagers se présentent qu'il n'y a de places, et
la borner numériquement avec une calculatrice (donner l'expression exacte).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:randvar:7}
Soit $X \sim \mathcal B(n, p)$. Montrer que
\[
\frac{\P(X = k+1)}{\P(X = k)} = \frac{n-k}{k+1}\cdot\frac{p}{1-p},
\]
et en déduire que la \omterm{def:g12:randvar:rv}{loi} croît jusqu'à
$k^* = \floor{(n+1)p}$ et décroît ensuite (le \emph{mode} de la
binomiale).
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:randvar:8}
\emph{(Saint-Pétersbourg, apprivoisé.)} On lance une pièce équilibrée
jusqu'à l'apparition de pile, mais au plus $10$ fois. Soit $N$ le nombre de
lancers utilisés, et le joueur reçoit $2^N$ euros si pile est apparu, $0$
sinon.
\begin{enumerate}
  \item Donner la \omterm{def:g12:randvar:rv}{loi} de $N$ restreinte aux issues gagnantes~:
        $\P(\text{pile d'abord au lancer } k) = 2^{-k}$ pour
        $1 \leq k \leq 10$, et vérifier la \omterm{def:g10:proba:distribution}{probabilité} totale de gagner.
  \item Calculer le gain espéré. Que deviendrait-il sans le plafond de
        $10$ lancers~?
\end{enumerate}
\end{exercise}
