\chapter{Sommes de variables aléatoires et loi des grands nombres}\label{ch:g12:sums}

Pourquoi les casinos gagnent-ils toujours à la fin, et pourquoi les sondages
fonctionnent-ils~? Parce que les moyennes de nombreuses quantités aléatoires
indépendantes fluctuent de moins en moins. Ce chapitre le prouve~: linéarité
de l'espérance, additivité de la variance pour des variables indépendantes,
inégalité de Bienaymé--Tchebychev, et loi des grands nombres.

\section{Sommes de variables aléatoires}

Étant données deux variables aléatoires $X, Y$ sur le même univers fini
$\Omega$, la somme $X + Y$ est la variable aléatoire
$\omega \mapsto X(\omega) + Y(\omega)$.

\begin{theorem}[Linéarité de l'espérance]\label{thm:g12:sums:linearity}
\index{espérance!linéarité}
Pour toutes variables aléatoires $X, Y$ sur $\Omega$ et $a, b \in \R$~:
\[
\E(X + Y) = \E(X) + \E(Y), \qquad \E(aX + b) = a\E(X) + b .
\]
\emph{Aucune hypothèse d'indépendance n'est nécessaire.}
\end{theorem}

\begin{proof}
Écrire l'espérance comme somme sur les issues~: comme
$\P(X = x) = \sum_{\omega : X(\omega) = x} \P(\{\omega\})$, en regroupant
les termes on obtient $\E(X) = \sum_{\omega \in \Omega} \P(\{\omega\})\,X(\omega)$.
Alors
\[
\E(X + Y) = \sum_{\omega} \P(\{\omega\})\bigl(X(\omega) + Y(\omega)\bigr)
= \sum_{\omega} \P(\{\omega\})X(\omega)
+ \sum_{\omega} \P(\{\omega\})Y(\omega) = \E(X) + \E(Y). \qedhere
\]
\end{proof}

\begin{definition}[Variables aléatoires indépendantes]\label{def:g12:sums:indep}
$X$ et $Y$ sont \emph{indépendantes}\index{variable aléatoire!indépendante}
si pour toutes valeurs $x, y$~:
\[
\P(X = x \text{ et } Y = y) = \P(X = x)\,\P(Y = y).
\]
Plusieurs variables $X_1, \dots, X_n$ sont indépendantes si cette règle de
produit vaut pour tout choix de valeurs de toute sous-famille.
\end{definition}

\begin{proposition}\label{prop:g12:sums:prodexp}
Si $X$ et $Y$ sont indépendantes, $\E(XY) = \E(X)\,\E(Y)$.
\end{proposition}

\begin{proof}
\[
\begin{aligned}
\E(XY) &= \sum_{x, y} xy\;\P(X = x \text{ et } Y = y)
= \sum_{x, y} xy\,\P(X=x)\P(Y=y) \\
&= \Bigl(\sum_x x\P(X=x)\Bigr)\Bigl(\sum_y y\P(Y=y)\Bigr). \qedhere
\end{aligned}
\]
\end{proof}

\begin{theorem}[Variance d'une somme]\label{thm:g12:sums:variance}
\index{variance!d'une somme}
Si $X$ et $Y$ sont \emph{indépendantes}, alors
\[
\V(X + Y) = \V(X) + \V(Y).
\]
Plus généralement, pour $X_1, \dots, X_n$ indépendantes~:
$\V(X_1 + \dots + X_n) = \V(X_1) + \dots + \V(X_n)$.
\end{theorem}

\begin{proof}
En utilisant König--Huygens (\cref{prop:g12:randvar:konig}) et la linéarité~:
\begin{align*}
\V(X + Y) &= \E\bigl((X+Y)^2\bigr) - \bigl(\E X + \E Y\bigr)^2\\
&= \E(X^2) + 2\E(XY) + \E(Y^2)
- \E(X)^2 - 2\E(X)\E(Y) - \E(Y)^2\\
&= \V(X) + \V(Y) + 2\bigl(\E(XY) - \E(X)\E(Y)\bigr),
\end{align*}
et le dernier crochet s'annule pour des variables indépendantes
(\cref{prop:g12:sums:prodexp}). Le cas général suit par récurrence.
\end{proof}

\begin{example}[Binomiale revisitée]\label{ex:g12:sums:binomial}
Une variable binomiale $X \sim \mathcal B(n, p)$ est une somme
$X = X_1 + \dots + X_n$ de $n$ variables de Bernoulli indépendantes. D'où,
structurellement~:
\[
\E(X) = np, \qquad \V(X) = np(1-p),
\]
retrouvant \cref{thm:g12:randvar:binomial} sans aucun calcul.
\end{example}

\begin{proposition}[Moyenne empirique]\label{prop:g12:sums:mean}
Soient $X_1, \dots, X_n$ indépendantes de même loi que $X$ (espérance
$\mu$, variance $\sigma^2$), et soit
$M_n = \frac{X_1 + \dots + X_n}{n}$ leur \emph{moyenne
empirique}\index{moyenne empirique}. Alors
\[
\E(M_n) = \mu, \qquad \V(M_n) = \frac{\sigma^2}{n}, \qquad
\sigma(M_n) = \frac{\sigma}{\sqrt n}.
\]
\end{proposition}

\begin{proof}
La linéarité donne $\E(M_n) = \frac{n\mu}{n} = \mu$~; l'indépendance donne
$\V(X_1 + \dots + X_n) = n\sigma^2$, et diviser par $n$ multiplie la
variance par $\frac{1}{n^2}$ (\cref{prop:g12:randvar:affine}).
\end{proof}

La quantité $\frac{\sigma}{\sqrt n}$ est la \emph{loi de la racine carrée}
fondamentale~: pour diviser par deux les fluctuations d'une moyenne, il
faut quadrupler la taille de l'échantillon.

\section{Inégalités de concentration}

\begin{theorem}[Inégalité de Markov]\label{thm:g12:sums:markov}
\index{inégalité de Markov}
Si $X \geq 0$ et $a > 0$~:
\[
\P(X \geq a) \leq \frac{\E(X)}{a}.
\]
\end{theorem}

\begin{proof}
Dans $\E(X) = \sum_i p_i x_i$ (tous les termes positifs), ne garder que les
termes avec $x_i \geq a$~: chacun vaut au moins $a\,p_i$, donc
$\E(X) \geq a \sum_{x_i \geq a} p_i = a\,\P(X \geq a)$.
\end{proof}

\begin{theorem}[Inégalité de Bienaymé--Tchebychev]\label{thm:g12:sums:chebyshev}
\index{inégalité de Bienaymé--Tchebychev}
Pour toute variable aléatoire $X$ et tout $\varepsilon > 0$~:
\[
\P\bigl(\abs{X - \E(X)} \geq \varepsilon\bigr)
\leq \frac{\V(X)}{\varepsilon^2}.
\]
\end{theorem}

\begin{proof}
Appliquer l'inégalité de Markov à la variable positive
$Y = (X - \E(X))^2$ avec $a = \varepsilon^2$~:
\[
\P\bigl(\abs{X - \E(X)} \geq \varepsilon\bigr)
= \P(Y \geq \varepsilon^2)
\leq \frac{\E(Y)}{\varepsilon^2} = \frac{\V(X)}{\varepsilon^2}. \qedhere
\]
\end{proof}

\begin{theorem}[Loi des grands nombres]\label{thm:g12:sums:lln}
\index{loi des grands nombres}
Soient $X_1, \dots, X_n$ indépendantes de même loi (espérance $\mu$,
variance $\sigma^2$) et $M_n$ leur moyenne empirique. Pour tout
$\varepsilon > 0$~:
\[
\P\bigl(\abs{M_n - \mu} \geq \varepsilon\bigr)
\leq \frac{\sigma^2}{n\,\varepsilon^2}
\xrightarrow[n \to +\infty]{} 0 .
\]
La moyenne empirique se concentre autour de l'espérance.
\end{theorem}

\begin{proof}
Bienaymé--Tchebychev appliqué à $M_n$, d'espérance $\mu$ et de variance
$\frac{\sigma^2}{n}$ (\cref{prop:g12:sums:mean}).
\end{proof}

\begin{remark}
Ce théorème est le pont entre théorie des probabilités et statistique~: la
\emph{fréquence} d'un événement sur de nombreuses répétitions indépendantes
approche sa \emph{probabilité} (prendre $X_i$ l'indicatrice de l'événement,
de sorte que $\mu = p$). Il justifie l'estimation d'une probabilité par
simulation (\emph{méthode de Monte-Carlo}) et d'une proportion dans une
population par un sondage --- quantitativement~: voir
\cref{ch:g12:contdist}.
\end{remark}

\begin{method}[Utiliser Bienaymé--Tchebychev]\label{met:g12:sums:chebyshev}
Pour garantir $\P(\abs{M_n - \mu} \geq \varepsilon) \leq \alpha$, il
suffit que $n \geq \frac{\sigma^2}{\alpha\,\varepsilon^2}$. La borne est
grossière (les fluctuations réelles sont en général bien plus petites) mais
parfaitement générale~: elle n'exige rien sur la loi au-delà d'une variance
finie.
\end{method}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:g12:sums:1}
On lance deux dés équilibrés~; soit $S$ la somme. En utilisant la linéarité
(pas la loi de $S$~!), calculer $\E(S)$~; puis calculer $\V(S)$ en
utilisant l'indépendance, sachant qu'un dé équilibré a pour variance
$\frac{35}{12}$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:sums:2}
Soit $X \sim \mathcal B(100,\ 0.5)$. Borner
$\P(X \geq 75)$ par l'inégalité de Markov, puis
$\P(\abs{X - 50} \geq 25)$ par Bienaymé--Tchebychev. Comparer.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:sums:3}
On lance une pièce équilibrée $n$ fois et $F_n$ désigne la fréquence de
pile. Quelle taille $n$ doit-on prendre pour que, par la borne de
Bienaymé--Tchebychev,
$\P\bigl(\abs{F_n - 0.5} \geq 0.05\bigr) \leq 0.05$~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:sums:4}
Une investisseuse répartit son capital également entre $n$ actifs
indépendants, chacun de rendement espéré $\mu = 5\%$ et d'écart-type
$\sigma = 20\%$. Calculer l'espérance et l'écart-type du rendement du
portefeuille $M_n$, et le nombre d'actifs nécessaires pour ramener
l'écart-type sous $4\%$. Quel principe financier cela illustre-t-il~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:sums:5}
Soient $X$ et $Y$ indépendantes, toutes deux uniformes sur $\{1, 2, 3\}$.
\begin{enumerate}
  \item Donner la loi de $S = X + Y$ et calculer $\E(S)$, $\V(S)$
        directement à partir de celle-ci.
  \item Retrouver les deux valeurs par linéarité et additivité de la
        variance.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:sums:6}
Montrer que l'additivité de la variance peut échouer sans indépendance~:
calculer $\V(X + Y)$ pour $Y = X$, et comparer avec $\V(X) + \V(Y)$. Pour
quelles variables $X$ a-t-on $\V(2X) = 2\V(X)$~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:sums:7}
Un dé est soupçonné d'être pipé. On le lance $1200$ fois et on obtient un
six $260$ fois ($f = 0.2167$ au lieu de $\frac16 \approx 0.1667$). Sous
l'hypothèse que le dé est équilibré, borner
$\P\bigl(\abs{F_n - \frac16} \geq 0.05\bigr)$ par Bienaymé--Tchebychev, et
discuter si l'hypothèse d'équité est plausible.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:sums:8}
\emph{(Une meilleure inégalité pour la pièce.)} Soient
$X \sim \mathcal B(n,\ p)$ et $F_n = \frac Xn$.
\begin{enumerate}
  \item Montrer que $p(1-p) \leq \frac14$ pour $p \in \intcc{0}{1}$.
  \item En déduire la borne indépendante de la loi
        $\P\bigl(\abs{F_n - p} \geq \varepsilon\bigr) \leq
        \frac{1}{4n\varepsilon^2}$.
  \item Combien de personnes faut-il interroger pour que la fréquence
        observée soit à moins de $3$ points de la vraie proportion avec
        probabilité au moins $95\%$, en utilisant cette borne~? (Les
        vrais sondages utilisent des estimations plus fines, mais l'ordre
        de grandeur est correct.)
\end{enumerate}
\end{exercise}
