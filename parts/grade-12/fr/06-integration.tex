\begin{proposition}[PropriÃĐtÃĐs de l'intÃĐgrale]\label{prop:g12:integ:properties}

\begin{enumerate}
  \item \emph{LinÃĐaritÃĐ~:}
        $\displaystyle\int_a^b (f + \lambda g) = \int_a^b f + \lambda \int_a^b g$.
  \item \emph{Relation de Chasles~:}

  \item \emph{PositivitÃĐ~:} si $a \leq b$ et $f \geq 0$ sur $\intcc{a}{b}$,

\end{enumerate}
\end{proposition}

\begin{proof}

\begin{definition}[Primitive]\label{def:g12:integ:primitive}
Une \emph{primitive}\index{primitive} (ou antÃĐdÃĐrivÃĐe) de $f$ sur un

\begin{proposition}\label{prop:g12:integ:primunique}

\end{proposition}

\begin{proof}
Si $G' = F' = f$, alors $(G - F)' = 0$ sur l'intervalle $I$, donc
$G - F$ est constante.\footnote{Qu'une fonction de dÃĐrivÃĐe nulle sur un
intervalle soit constante suit du \cref{thm:g12:deriv:variations}~: elle
est Ã  la fois croissante et dÃĐcroissante.} RÃĐciproquement toute $F + c$
est une primitive. La condition $F(x_0) = y_0$ fixe la constante.
\end{proof}

\begin{theorem}[ThÃĐorÃĻme fondamental du calcul]\label{thm:g12:integ:ftc}
\index{thÃĐorÃĻme fondamental du calcul}
Soit $f$ continue sur un intervalle $I$ et $a \in I$. La fonction
\[

\]
est la primitive de $f$ sur $I$ s'annulant en $a$. Par consÃĐquent, pour
toute primitive $G$ de $f$ et $a, b \in I$~:
\[
\int_a^b f(t)\,\dd t = \bigl[G(t)\bigr]_a^b = G(b) - G(a).
\]
\end{theorem}

\begin{proof}[DÃĐmonstration lorsque $f$ est croissante]
Fixer $x \in I$ et $h > 0$ avec $x + h \in I$. Par Chasles,
\[
F(x+h) - F(x) = \int_x^{x+h} f(t)\,\dd t .
\]
Comme $f$ est croissante, $f(x) \leq f(t) \leq f(x+h)$ pour
$t \in \intcc{x}{x+h}$, et par comparaison d'intÃĐgrales (l'intÃĐgrale
d'une constante $c$ sur un intervalle de longueur $h$ est $ch$)~:
\[

\]
donc
\[
f(x) \leq \frac{F(x+h) - F(x)}{h} \leq f(x+h).
\]
Lorsque $h \to 0^+$, $f(x+h) \to f(x)$ par continuitÃĐ, et le thÃĐorÃĻme des
gendarmes donne que le taux d'accroissement tend vers $f(x)$~; le cas

(non monotone) continu est prouvÃĐ Ã  l'universitÃĐ.

\end{proof}

\begin{theorem}[IntÃĐgration par parties]\label{thm:g12:integ:ibp}
\index{intÃĐgration par parties}

\[

\]
\end{theorem}

\begin{proof}
La rÃĻgle du produit donne $(uv)' = u'v + uv'$~; en intÃĐgrant les deux
membres sur $\intcc{a}{b}$ et en utilisant le thÃĐorÃĻme fondamental pour le

\end{proof}

\begin{example}\label{ex:g12:integ:ibp}
$\displaystyle\int_0^1 t\,\eu^{t}\,\dd t$~: prendre $u' = \eu^t$, $v = t$,
donc $u = \eu^t$, $v' = 1$~:
\[
\int_0^1 t\,\eu^t\,\dd t = \bigl[t\,\eu^t\bigr]_0^1 - \int_0^1 \eu^t\,\dd t
= \eu - (\eu - 1) = 1 .
\]
\end{example}

\section{Applications}

\begin{definition}[Valeur moyenne]\label{def:g12:integ:mean}
La \emph{valeur moyenne}\index{valeur moyenne} d'une fonction continue $f$
sur $\intcc{a}{b}$ ($a<b$) est
\[

\]
\end{definition}

\begin{proposition}\label{prop:g12:integ:meanbounds}
Si $m \leq f \leq M$ sur $\intcc{a}{b}$, alors $m \leq \mu \leq M$.
\end{proposition}

\begin{proof}
IntÃĐgrer les inÃĐgalitÃĐs $m \leq f(t) \leq M$ sur $\intcc{a}{b}$ et

\end{proof}

\begin{method}[Aire entre deux courbes]\label{met:g12:integ:areabetween}
Si $f \geq g$ sur $\intcc{a}{b}$, l'aire entre les deux courbes est
$\int_a^b \bigl(f(x) - g(x)\bigr)\dd x$. Si les courbes se croisent,
dÃĐcouper l'intervalle aux points de croisement et intÃĐgrer $\abs{f - g}$
morceau par morceau.
\end{method}
