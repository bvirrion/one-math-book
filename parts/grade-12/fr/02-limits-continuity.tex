\chapter{Limites et continuité}\label{ch:g12:limcont}

Ce chapitre étend la notion de limite des \omterm{def:g12:seq:sequence}{suites} aux \omterm{def:g11:func:function}{fonctions} d'une
variable réelle, et introduit la \omterm{def:g12:limcont:continuity}{continuité}. Le résultat central est le
théorème des valeurs intermédiaires, qui transforme une information
qualitative («~$f$ est \omterm{def:g12:limcont:continuity}{continue} et change de signe~») en l'existence de
solutions d'\omterm{def:g10:algebra:equation}{équations}.

\section{Limites de fonctions}

Tout au long, $f$ est une \omterm{def:g11:func:function}{fonction} définie sur un \omterm{def:g10:numbers:interval}{intervalle} ou une \omterm{def:g10:numbers:interunion}{réunion}
d'\omterm{def:g10:numbers:interval}{intervalles} $D \subseteq \R$.

\begin{definition}[Limite à l'infini]\label{def:g12:limcont:liminf}
Soit $f$ définie sur un \omterm{def:g10:numbers:interval}{intervalle} $\intco{a}{+\infty}$ et $\ell \in \R$.
On dit que $f$ \emph{tend vers $\ell$ en $+\infty$}\index{limite!d'une
fonction}, noté $\lim\limits_{x\to+\infty} f(x) = \ell$, si tout
\omterm{def:g10:numbers:interval}{intervalle} ouvert contenant $\ell$ contient toutes les valeurs $f(x)$ pour
$x$ assez grand~: pour tout $\varepsilon > 0$ il existe $A$ tel que pour
tout $x \geq A$, $\abs{f(x) - \ell} \leq \varepsilon$.

On dit que $f$ tend vers $+\infty$ en $+\infty$ si pour tout $M \in \R$ il
existe $A$ tel que $f(x) \geq M$ pour tout $x \geq A$. Les limites en
$-\infty$ et les limites égales à $-\infty$ se définissent de façon
analogue.
\end{definition}

\begin{definition}[Limite en un point]\label{def:g12:limcont:limpoint}
Soit $a \in \R$ et soit $f$ définie au voisinage de $a$ (sauf éventuellement
en $a$). On dit que $f$ tend vers $\ell \in \R$ en $a$, noté
$\lim\limits_{x \to a} f(x) = \ell$, si pour tout $\varepsilon > 0$ il
existe $\delta > 0$ tel que pour tout $x \in D$ avec
$\abs{x - a} \leq \delta$, $\abs{f(x) - \ell} \leq \varepsilon$.

On dit que $f$ tend vers $+\infty$ en $a$ si pour tout $M$ il existe
$\delta > 0$ tel que $f(x) \geq M$ dès que $x \in D$ et
$\abs{x - a} \leq \delta$. Les limites à droite et à gauche
($x \to a^+$, $x \to a^-$) restreignent $x$ à $x > a$ ou $x < a$.
\end{definition}

\begin{example}
$\lim\limits_{x\to+\infty} \frac{1}{x} = 0$,
$\lim\limits_{x\to 0^+} \frac{1}{x} = +\infty$,
$\lim\limits_{x\to 0^-} \frac{1}{x} = -\infty$. La \omterm{def:g11:func:function}{fonction}
$x \mapsto \frac1x$ n'a pas de limite (bilatérale) en $0$.
\end{example}

\begin{definition}[Asymptotes]\label{def:g12:limcont:asymptote}
La droite $y = \ell$ est une \emph{asymptote horizontale}\index{asymptote}
de la courbe de $f$ en $+\infty$ (resp.\ $-\infty$) si
$\lim_{x\to+\infty} f(x) = \ell$ (resp.\ en $-\infty$). La droite $x = a$
est une \emph{asymptote verticale} si $f$ a une limite infinie unilatérale
en $a$.
\end{definition}

\begin{proposition}[Opérations sur les limites]\label{prop:g12:limcont:operations}
Les règles de la \cref{prop:g12:seq:operations} (sommes, produits,
quotients) valent verbatim pour les limites de \omterm{def:g11:func:function}{fonctions}, en un point ou à
l'infini, avec les mêmes quatre formes indéterminées
$(+\infty)+(-\infty)$, $0\times\infty$, $\frac{\infty}{\infty}$,
$\frac{0}{0}$.
\end{proposition}

\begin{proof}
Les démonstrations sont identiques au cas des \omterm{def:g12:seq:sequence}{suites}, en remplaçant
«~pour $n \geq N$~» par «~pour $x \geq A$~» ou «~pour
$\abs{x-a} \leq \delta$~».
\end{proof}

\begin{theorem}[Limite d'une composition]\label{thm:g12:limcont:composition}
Soient $f, g$ des \omterm{def:g11:func:function}{fonctions} et $a, b, c$ des \omterm{def:g10:numbers:sets}{nombres réels} ou $\pm\infty$.
Si $\lim\limits_{x \to a} f(x) = b$ et $\lim\limits_{y \to b} g(y) = c$,
alors
\[
\lim_{x \to a} g\bigl(f(x)\bigr) = c .
\]
Le même énoncé vaut avec une \omterm{def:g12:seq:sequence}{suite} à la place de $f$~: si $u_n \to b$ et
$\lim_{y\to b} g(y) = c$, alors $g(u_n) \to c$.
\end{theorem}

\begin{proof}
Prendre $b, c$ finis (les autres cas sont analogues). Soit
$\varepsilon > 0$. Il existe $\delta > 0$ tel que $\abs{y - b} \leq \delta$
implique $\abs{g(y) - c} \leq \varepsilon$~; et il existe $\delta' > 0$ tel
que $\abs{x - a} \leq \delta'$ implique $\abs{f(x) - b} \leq \delta$. En
enchaînant les deux, $\abs{g(f(x)) - c} \leq \varepsilon$ pour
$\abs{x - a} \leq \delta'$.
\end{proof}

\begin{theorem}[Comparaison et gendarmes]\label{thm:g12:limcont:squeeze}
Soient $f, g, h$ des \omterm{def:g11:func:function}{fonctions} et $a \in \R \cup \{\pm\infty\}$.
\begin{enumerate}
  \item Si $f \leq g$ au voisinage de $a$ et $f(x) \to +\infty$ lorsque
        $x \to a$, alors $g(x) \to +\infty$.
  \item Si $f \leq g \leq h$ au voisinage de $a$ et $f, h$ tendent vers la
        même limite finie $\ell$ en $a$, alors $g(x) \to \ell$ lorsque
        $x \to a$.
\end{enumerate}
\end{theorem}

\begin{proof}
Même argument que pour les \omterm{def:g12:seq:sequence}{suites} (\cref{thm:g12:seq:squeeze}).
\end{proof}

\begin{method}[Calculer des limites en pratique]\label{met:g12:limcont:limits}
\begin{itemize}
  \item Polynômes et \omterm{def:g11:func:function}{fonctions} rationnelles en $\pm\infty$~: \omterm{def:g10:algebra:expand}{factoriser} la
        plus haute puissance de $x$~; la limite est celle du rapport des
        termes dominants.
  \item $\frac{0}{0}$ en un point $a$~: \omterm{def:g10:algebra:expand}{factoriser} $(x-a)$ au numérateur et
        au dénominateur, ou reconnaître une \omterm{def:g11:deriv:derivative}{dérivée}
        $\lim_{x\to a}\frac{f(x)-f(a)}{x-a} = f'(a)$
        (\cref{ch:g12:deriv}).
  \item Racines carrées~: multiplier par l'expression conjuguée.
  \item Facteurs oscillants ($\cos$, $\sin$)~: utiliser le théorème des
        gendarmes.
\end{itemize}
\end{method}

\section{Continuité}

\begin{definition}[Continuité]\label{def:g12:limcont:continuity}
Soit $f$ définie sur un \omterm{def:g10:numbers:interval}{intervalle} $I$ et $a \in I$. La \omterm{def:g11:func:function}{fonction} $f$ est
\emph{continue en $a$}\index{continuité} si
$\lim\limits_{x \to a} f(x) = f(a)$. Elle est \emph{continue sur $I$}\index{continue} si
elle est continue en tout point de $I$.
\end{definition}

Intuitivement, le \omterm{def:g10:functions:graph}{graphique} d'une \omterm{def:g11:func:function}{fonction} \omterm{def:g12:limcont:continuity}{continue} sur un \omterm{def:g10:numbers:interval}{intervalle} peut
être tracé sans lever le stylo.

\begin{proposition}[Continuité des fonctions usuelles]\label{prop:g12:limcont:usual}
Les polynômes, les \omterm{def:g11:func:function}{fonctions} rationnelles, $x \mapsto \sqrt{x}$,
$x \mapsto \abs{x}$, $\cos$, $\sin$, $\exp$ et $\ln$ sont \omterm{def:g12:limcont:continuity}{continues} sur
tout \omterm{def:g10:numbers:interval}{intervalle} contenu dans leur \omterm{def:g11:func:function}{ensemble de définition}. Les sommes,
produits, quotients (là où ils sont définis) et compositions de \omterm{def:g11:func:function}{fonctions}
\omterm{def:g12:limcont:continuity}{continues} sont \omterm{def:g12:limcont:continuity}{continues}.
\end{proposition}

\begin{proof}[Démonstration partielle]
La stabilité sous sommes, produits, quotients et composition suit des
théorèmes correspondants sur les limites. La \omterm{def:g12:limcont:continuity}{continuité} de
$x \mapsto x$ et des constantes est immédiate d'après la définition, d'où
les polynômes et les \omterm{def:g11:func:function}{fonctions} rationnelles sont \omterm{def:g12:limcont:continuity}{continues}. La \omterm{def:g12:limcont:continuity}{continuité}
des autres \omterm{def:g11:func:function}{fonctions} usuelles est prouvée dans les chapitres où elles sont
construites, ou admise.
\end{proof}

\begin{example}
La \emph{\omterm{def:g11:func:function}{fonction} partie entière} $x \mapsto \floor{x}$ (le plus grand
entier $\leq x$) n'est \omterm{def:g12:limcont:continuity}{continue} en aucun entier $n$~: sa limite à gauche y
est $n - 1$ tandis que $\floor{n} = n$.
\end{example}

\begin{proposition}[Suites et continuité]\label{prop:g12:limcont:seqcont}
Si $u_n \to \ell$ et $f$ est \omterm{def:g12:limcont:continuity}{continue} en $\ell$, alors
$f(u_n) \to f(\ell)$. En particulier, si une \omterm{def:g12:seq:sequence}{suite} définie par
$u_{n+1} = f(u_n)$ \omterm{def:g12:seq:limit}{converge} vers $\ell$ et $f$ est \omterm{def:g12:limcont:continuity}{continue} en $\ell$,
alors $f(\ell) = \ell$.
\end{proposition}

\begin{proof}
La première assertion est le \cref{thm:g12:limcont:composition} appliqué
avec la \omterm{def:g12:seq:sequence}{suite} $(u_n)$. Pour la seconde, passer à la limite des deux côtés
de $u_{n+1} = f(u_n)$~: le membre de gauche tend vers $\ell$, celui de
droite vers $f(\ell)$, et les limites sont uniques.
\end{proof}

\section{Le théorème des valeurs intermédiaires}

\begin{theorem}[Théorème des valeurs intermédiaires]\label{thm:g12:limcont:ivt}
\index{théorème des valeurs intermédiaires}
Soit $f$ \omterm{def:g12:limcont:continuity}{continue} sur $\intcc{a}{b}$ et soit $k$ un nombre quelconque entre
$f(a)$ et $f(b)$. Alors il existe $c \in \intcc{a}{b}$ tel que
$f(c) = k$.
\end{theorem}

{\small Le théorème des valeurs intermédiaires~: une courbe \omterm{def:g12:limcont:continuity}{continue}
joignant $(a, f(a))$ à $(b, f(b))$ doit croiser toute droite horizontale
$y = k$ située entre $f(a)$ et $f(b)$.}
\end{omfigure}

\begin{proof}
En remplaçant $f$ par $f - k$ (et par $k - f$ si besoin), on peut supposer
$f(a) \leq 0 \leq f(b)$ et chercher un zéro de $f$. On construit deux
\omterm{def:g12:seq:sequence}{suites} par \emph{dichotomie}\index{dichotomie}~: poser $a_0 = a$,
$b_0 = b$~; étant donnés $a_n \leq b_n$ avec $f(a_n) \leq 0 \leq f(b_n)$,
soit $m = \frac{a_n + b_n}{2}$ et définir
\[
(a_{n+1}, b_{n+1}) =
\begin{cases}
(a_n, m) & \text{si } f(m) \geq 0,\\
(m, b_n) & \text{si } f(m) < 0.
\end{cases}
\]
Dans les deux cas $f(a_{n+1}) \leq 0 \leq f(b_{n+1})$, la \omterm{def:g12:seq:sequence}{suite} $(a_n)$ est
\omterm{def:g12:seq:monotonic}{croissante}, $(b_n)$ est \omterm{def:g10:functions:variations}{décroissante}, et
$b_n - a_n = \frac{b-a}{2^n} \to 0$~: les \omterm{def:g12:seq:sequence}{suites} sont adjacentes (voir
l'\cref{exo:g12:seq:10}) et convergent vers une limite commune
$c \in \intcc{a}{b}$. Par \omterm{def:g12:limcont:continuity}{continuité}, $f(a_n) \to f(c)$ et
$f(b_n) \to f(c)$~; en passant à la limite dans $f(a_n) \leq 0$ et
$f(b_n) \geq 0$ on obtient $f(c) \leq 0 \leq f(c)$, donc $f(c) = 0$.
\end{proof}

\begin{remark}
La démonstration par \omterm{thm:g12:limcont:ivt}{dichotomie} est un algorithme~: elle produit des
\omterm{def:g10:numbers:approx}{approximations} arbitrairement précises d'une solution, en divisant l'erreur
par deux à chaque étape. C'est ainsi qu'un ordinateur peut résoudre
$f(x) = 0$ en ne sachant qu'évaluer $f$.
\end{remark}

\begin{theorem}[Théorème de la bijection]\label{thm:g12:limcont:bijection}
Soit $f$ \omterm{def:g12:limcont:continuity}{continue} et \emph{strictement \omterm{def:g11:func:monotone}{monotone}} sur un \omterm{def:g10:numbers:interval}{intervalle} $I$.
Alors pour tout $k$ strictement entre les limites de $f$ aux extrémités de
$I$, l'\omterm{def:g10:algebra:equation}{équation} $f(x) = k$ a une solution \emph{unique} dans $I$.
\end{theorem}

\begin{proof}
Existence~: choisir $a, b \in I$ avec $k$ entre $f(a)$ et $f(b)$ (possible
puisque les valeurs de $f$ approchent ses limites aux extrémités), et
appliquer le théorème des valeurs intermédiaires sur $\intcc{a}{b}$.
Unicité~: si $x < y$, la monotonie stricte donne $f(x) \neq f(y)$, donc
deux points distincts ne peuvent tous deux être solutions.
\end{proof}

\begin{method}[Résoudre $f(x) = k$ avec le théorème de la bijection]\label{met:g12:limcont:bijection}
Pour prouver que $f(x) = k$ a exactement une solution dans un \omterm{def:g10:numbers:interval}{intervalle}~:
\begin{enumerate}
  \item établir un \omterm{def:g10:functions:table}{tableau de variations} de $f$ (signe de $f'$, voir le
        \cref{ch:g12:deriv})~;
  \item sur chaque \omterm{def:g10:numbers:interval}{intervalle} où $f$ est \omterm{def:g12:limcont:continuity}{continue} et strictement \omterm{def:g11:func:monotone}{monotone},
        comparer $k$ aux valeurs ou limites aux extrémités~;
  \item appliquer le théorème de la bijection sur chaque tel \omterm{def:g10:numbers:interval}{intervalle} et
        compter les solutions.
\end{enumerate}
Une calculatrice ou la \omterm{thm:g12:limcont:ivt}{dichotomie} localise ensuite chaque solution aussi
finement que voulu.
\end{method}

\begin{example}\label{ex:g12:limcont:cubic}
L'\omterm{def:g10:algebra:equation}{équation} $x^3 + x = 1$ a une unique solution réelle. En effet
$f(x) = x^3 + x$ est \omterm{def:g12:limcont:continuity}{continue} et strictement \omterm{def:g12:seq:monotonic}{croissante} sur $\R$ (somme de
\omterm{def:g11:func:function}{fonctions} strictement \omterm{def:g12:seq:monotonic}{croissantes}), $\lim_{-\infty} f = -\infty$ et
$\lim_{+\infty} f = +\infty$, donc le théorème de la bijection s'applique
avec $k = 1$. Comme $f(0.6) = 0.816 < 1$ et $f(0.7) = 1.043 > 1$, la
solution est dans $\intoo{0.6}{0.7}$.
\end{example}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:g12:limcont:1}
Calculer les limites suivantes~:
\[
\lim_{x\to+\infty} \frac{2x^3 - x + 1}{5x^3 + x^2}, \qquad
\lim_{x\to-\infty} \bigl(x^2 + 3x - 1\bigr), \qquad
\lim_{x\to 2^+} \frac{x+1}{x-2}, \qquad
\lim_{x\to+\infty} \frac{\cos x}{x}.
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:limcont:2}
Soit $f(x) = \dfrac{2x^2 - 3x + 1}{x - 1}$, définie pour $x \neq 1$.
\begin{enumerate}
  \item Montrer que $f(x) = 2x - 1$ pour tout $x \neq 1$.
  \item En déduire $\lim\limits_{x \to 1} f(x)$. Peut-on prolonger $f$ en
        une \omterm{def:g11:func:function}{fonction} \omterm{def:g12:limcont:continuity}{continue} en $1$~?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:limcont:3}
Déterminer les \omterm{def:g12:limcont:asymptote}{asymptotes} de la courbe de
$f(x) = \dfrac{3x - 1}{x + 2}$ et croquer la courbe.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:limcont:4}
Calculer
\[
\lim_{x\to+\infty} \bigl(\sqrt{x^2 + x} - x\bigr)
\qquad\text{et}\qquad
\lim_{x\to 0} \frac{\sqrt{1+x} - 1}{x}.
\]
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:limcont:5}
Soit $f(x) = x + \sin x$.
\begin{enumerate}
  \item Montrer que $f(x) \to +\infty$ lorsque $x \to +\infty$, bien que
        $f$ ne soit \omterm{def:g11:func:monotone}{monotone} sur aucun \omterm{def:g10:numbers:interval}{intervalle} $\intco{A}{+\infty}$.
  \item Montrer que $g(x) = \dfrac{x + \sin x}{x - \sin x}$ tend vers $1$
        en $+\infty$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:limcont:6}
Montrer que l'\omterm{def:g10:algebra:equation}{équation} $x^3 - 3x + 1 = 0$ a exactement trois solutions
réelles, et donner pour chacune un \omterm{def:g10:numbers:interval}{intervalle} de longueur $1$ la
contenant. (Indication~: étudier les variations de
$f(x) = x^3 - 3x + 1$~; sa \omterm{def:g11:deriv:derivative}{dérivée} est $3x^2 - 3$.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:limcont:7}
Soit $f \colon \intcc{0}{1} \to \intcc{0}{1}$ \omterm{def:g12:limcont:continuity}{continue}. Montrer que $f$ a
un \emph{point fixe}~: il existe $c \in \intcc{0}{1}$ avec $f(c) = c$.
(Indication~: appliquer le théorème des valeurs intermédiaires à
$g(x) = f(x) - x$.)
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:limcont:8}
Une randonneuse gravit un sentier de montagne, partant à 8~h et arrivant à
20~h. Le lendemain elle redescend le même sentier, partant à nouveau à 8~h
et arrivant à 20~h. Montrer qu'il existe une heure de la journée à laquelle
elle est exactement au même point du sentier les deux jours. (Indication~:
introduire la différence des deux \omterm{def:g11:func:function}{fonctions} de position.)
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:limcont:9}
Soit $f(x) = x^5 + x^3 + x - 1$.
\begin{enumerate}
  \item Montrer que $f$ a un unique zéro réel $\alpha$, et que
        $\alpha \in \intoo{0}{1}$.
  \item Décrire une procédure de \omterm{thm:g12:limcont:ivt}{dichotomie} calculant $\alpha$ à
        $10^{-6}$ près, et donner le nombre d'itérations nécessaires en
