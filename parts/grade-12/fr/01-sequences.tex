\chapter{Suites}\label{ch:g12:seq}

Une \omterm{def:g12:seq:sequence}{suite} est une liste de \omterm{def:g10:numbers:sets}{nombres réels} indexés par les entiers naturels.
Les \omterm{def:g12:seq:sequence}{suites} modélisent des évolutions discrètes --- populations comptées
année par année, soldes d'un compte bancaire, \omterm{def:g10:numbers:approx}{approximations} successives
d'un nombre --- et leurs limites sont la première rencontre sérieuse avec
l'infini. Ce chapitre met en place le vocabulaire, le principe de
récurrence, et les théorèmes fondamentaux de convergence.

\section{Raisonnement par récurrence}

\begin{theorem}[Principe de récurrence]\label{thm:g12:seq:induction}
Soit $P(n)$ un énoncé dépendant d'un entier $n$, et soit $n_0 \in \N$. Si
\begin{enumerate}
  \item \emph{(initialisation)} $P(n_0)$ est vrai, et
  \item \emph{(hérédité)} pour tout $n \geq n_0$, $P(n)$ implique $P(n+1)$,
\end{enumerate}
alors $P(n)$ est vrai pour tout $n \geq n_0$.
\end{theorem}

\begin{proof}
Supposons, par l'absurde, que l'ensemble $A$ des entiers $n \geq n_0$ pour
lesquels $P(n)$ est faux est non vide. Alors $A$ a un plus petit élément
$m$.\footnote{Toute partie non vide de $\N$ a un plus petit élément~; cette
propriété de $\N$ est prise comme axiome.} Comme $P(n_0)$ est vrai,
$m > n_0$, donc $m - 1 \geq n_0$ et $m-1 \notin A$, c'est-à-dire $P(m-1)$
est vrai. L'hérédité appliquée à $n = m-1$ montre alors que $P(m)$ est
vrai, contredisant $m \in A$.
\end{proof}

\begin{example}\label{ex:g12:seq:bernoulli}
Prouvons l'\emph{inégalité de Bernoulli}~: pour tout réel $a > 0$ et tout
$n \in \N$,
\[
(1+a)^n \geq 1 + na.
\]
\emph{Initialisation.} Pour $n = 0$, les deux membres valent $1$.
\emph{Hérédité.} Supposons $(1+a)^n \geq 1+na$ pour un certain $n \in \N$.
Comme $1 + a > 0$, multiplier les deux membres par $1+a$ préserve
l'inégalité~:
\[
(1+a)^{n+1} \geq (1+na)(1+a) = 1 + (n+1)a + na^2 \geq 1 + (n+1)a .
\]
Par récurrence, l'inégalité vaut pour tout $n \in \N$.
\end{example}

\begin{method}[Rédiger une démonstration par récurrence]\label{met:g12:seq:induction}
Toujours rendre l'énoncé $P(n)$ explicite avant de commencer. Une
démonstration complète a trois parties visibles~: l'initialisation,
l'hérédité («~supposons $P(n)$~; prouvons $P(n+1)$~»), et la conclusion
invoquant le principe de récurrence. L'erreur la plus courante est de
prouver l'hérédité sans jamais utiliser l'hypothèse $P(n)$~: si cela
arrive, soit la preuve est fausse, soit la récurrence n'était pas
nécessaire.
\end{method}

\section{Vocabulaire des suites}

\begin{definition}[Suite]\label{def:g12:seq:sequence}
Une \emph{suite}\index{suite} est une \omterm{def:g11:func:function}{fonction} $u \colon \N \to \R$ (ou
de $\{n \in \N : n \geq n_0\}$ vers $\R$). L'\omterm{def:g10:functions:function}{image} de $n$ s'écrit $u_n$,
et la suite elle-même $(u_n)_{n\in\N}$ ou simplement $(u_n)$.
\end{definition}

Une \omterm{def:g12:seq:sequence}{suite} peut être définie \emph{explicitement}, par une formule
$u_n = f(n)$, ou par \emph{récurrence}, par son premier terme et une
relation $u_{n+1} = f(u_n)$.

\begin{definition}[Monotonie]\label{def:g12:seq:monotonic}
Une \omterm{def:g12:seq:sequence}{suite} $(u_n)$ est \emph{croissante}\index{suite!croissante} si
$u_{n+1} \geq u_n$ pour tout $n$, \emph{\omterm{def:g10:functions:variations}{décroissante}} si
$u_{n+1} \leq u_n$ pour tout $n$, et \emph{\omterm{def:g11:func:monotone}{monotone}} si elle est
croissante ou \omterm{def:g10:functions:variations}{décroissante}. Elle est \emph{strictement} croissante
(resp.\ \omterm{def:g10:functions:variations}{décroissante}) lorsque les inégalités sont strictes.
\end{definition}

\begin{method}[Étudier la monotonie d'une suite]\label{met:g12:seq:monotonicity}
Trois techniques standard~:
\begin{enumerate}
  \item étudier le signe de $u_{n+1} - u_n$~;
  \item si tous les termes sont positifs, comparer $\dfrac{u_{n+1}}{u_n}$
        à $1$~;
  \item si $u_n = f(n)$ avec $f$ définie sur $\intco{0}{+\infty}$, utiliser
        les variations de $f$.
\end{enumerate}
\end{method}

\begin{definition}[Suite bornée]\label{def:g12:seq:bounded}
Une \omterm{def:g12:seq:sequence}{suite} $(u_n)$ est \emph{majorée}\index{suite!bornée} s'il existe
$M \in \R$ avec $u_n \leq M$ pour tout $n$~; \emph{minorée} s'il existe
$m \in \R$ avec $u_n \geq m$ pour tout $n$~; et \emph{bornée} si les deux
valent.
\end{definition}

\subsection{Suites arithmétiques et géométriques}

\begin{definition}[Suites arithmétiques et géométriques]\label{def:g12:seq:arith-geom}
Une \omterm{def:g12:seq:sequence}{suite} $(u_n)$ est \emph{arithmétique}\index{suite!arithmétique} de
raison $r$ si $u_{n+1} = u_n + r$ pour tout $n$, et
\emph{géométrique}\index{suite!géométrique} de raison $q$ si
$u_{n+1} = q\,u_n$ pour tout $n$.
\end{definition}

\begin{proposition}[Forme explicite et sommes]\label{prop:g12:seq:arith-geom}
Soit $n \in \N$.
\begin{enumerate}
  \item Si $(u_n)$ est \omterm{def:g12:seq:arith-geom}{arithmétique} de raison $r$, alors
        $u_n = u_0 + nr$ et
        \[
        u_0 + u_1 + \dots + u_n = (n+1)\,\frac{u_0 + u_n}{2}.
        \]
  \item Si $(u_n)$ est \omterm{def:g12:seq:arith-geom}{géométrique} de raison $q \neq 1$, alors
        $u_n = u_0\, q^n$ et
        \[
        u_0 + u_1 + \dots + u_n = u_0\,\frac{1 - q^{n+1}}{1 - q}.
        \]
\end{enumerate}
\end{proposition}

\begin{proof}
Les formes explicites suivent par des récurrences immédiates. Pour la
somme \omterm{def:g12:seq:arith-geom}{arithmétique}, écrire $S = u_0 + \dots + u_n$ et ajouter la même
somme écrite à l'envers~: chacune des $n+1$ sommes de colonnes égale
$u_0 + u_n$, donc $2S = (n+1)(u_0+u_n)$. Pour la somme \omterm{def:g12:seq:arith-geom}{géométrique},
calculer $S - qS$~: tous les termes s'annulent par paires sauf le premier
et le dernier, donc $(1-q)S = u_0(1 - q^{n+1})$.
\end{proof}

\section{Limite d'une suite}

\begin{definition}[Suite convergente]\label{def:g12:seq:limit}
Une \omterm{def:g12:seq:sequence}{suite} $(u_n)$ \emph{converge}\index{limite!d'une suite} vers le \omterm{def:g10:numbers:sets}{nombre
réel} $\ell$ si tout \omterm{def:g10:numbers:interval}{intervalle} ouvert contenant $\ell$ contient tous les
termes $u_n$ à partir d'un certain rang. On écrit alors
$\lim\limits_{n\to+\infty} u_n = \ell$.

De façon équivalente~: pour tout $\varepsilon > 0$, il existe $N \in \N$
tel que pour tout $n \geq N$, $\abs{u_n - \ell} \leq \varepsilon$.
\end{definition}

\begin{definition}[Divergence vers l'infini]\label{def:g12:seq:infinity}
La \omterm{def:g12:seq:sequence}{suite} $(u_n)$ \emph{tend vers $+\infty$} si pour tout $A \in \R$, il
existe $N \in \N$ tel que $u_n \geq A$ pour tout $n \geq N$. On écrit
$\lim\limits_{n\to+\infty} u_n = +\infty$~; la définition de
$\lim u_n = -\infty$ est analogue. Une \omterm{def:g12:seq:sequence}{suite} qui ne \omterm{def:g12:seq:limit}{converge} pas est dite
\emph{divergente}.
\end{definition}

\begin{remark}
Une \omterm{def:g12:seq:sequence}{suite} peut diverger sans tendre vers $\pm\infty$~: la \omterm{def:g12:seq:sequence}{suite}
$u_n = (-1)^n$ ne prend que les valeurs $1$ et $-1$ et n'a pas de limite.
\end{remark}

\begin{proposition}[Unicité de la limite]\label{prop:g12:seq:uniqueness}
Si $(u_n)$ \omterm{def:g12:seq:limit}{converge}, sa limite est unique.
\end{proposition}

\begin{proof}
Supposons $u_n \to \ell$ et $u_n \to \ell'$ avec $\ell \neq \ell'$, disons
$\ell < \ell'$. Poser $\varepsilon = \frac{\ell' - \ell}{3} > 0$. À partir
d'un certain rang, $\abs{u_n - \ell} \leq \varepsilon$ et
$\abs{u_n - \ell'} \leq \varepsilon$, d'où
\[
\ell' - \ell \leq \abs{\ell' - u_n} + \abs{u_n - \ell}
\leq 2\varepsilon = \tfrac{2}{3}(\ell' - \ell) < \ell' - \ell,
\]
une contradiction.
\end{proof}

\begin{proposition}[Opérations sur les limites]\label{prop:g12:seq:operations}
Soient $(u_n)$ et $(v_n)$ des \omterm{def:g12:seq:sequence}{suites} de limites $\ell$ et $\ell'$ (finies
ou infinies). Alors, chaque fois que le second membre n'est pas une forme
indéterminée,
\[
\lim (u_n + v_n) = \ell + \ell', \qquad
\lim (u_n v_n) = \ell\,\ell', \qquad
\lim \frac{u_n}{v_n} = \frac{\ell}{\ell'}.
\]
Les formes indéterminées sont
$(+\infty) + (-\infty)$, $0 \times \infty$, $\frac{\infty}{\infty}$ et
$\frac{0}{0}$.
\end{proposition}

\begin{proof}
On prouve la règle de la somme pour des limites finies~; les autres cas
sont analogues et laissés en exercice. Soit $\varepsilon > 0$. Il existe
$N_1, N_2$ tels que $\abs{u_n - \ell} \leq \varepsilon/2$ pour
$n \geq N_1$ et $\abs{v_n - \ell'} \leq \varepsilon/2$ pour
$n \geq N_2$. Pour $n \geq \max(N_1, N_2)$, l'inégalité triangulaire donne
\[
\abs{(u_n + v_n) - (\ell + \ell')}
\leq \abs{u_n - \ell} + \abs{v_n - \ell'} \leq \varepsilon. \qedhere
\]
\end{proof}

\begin{method}[Lever une forme indéterminée]\label{met:g12:seq:indeterminate}
Face à une forme indéterminée, \omterm{def:g10:algebra:expand}{factoriser} le terme dominant. Par exemple
\[
n^2 - n = n^2\left(1 - \tfrac{1}{n}\right) \xrightarrow[n\to+\infty]{} +\infty,
\qquad
\frac{2n^2+1}{n^2 - n} = \frac{2 + 1/n^2}{1 - 1/n}
\xrightarrow[n\to+\infty]{} 2 .
\]
\end{method}

\section{Théorèmes de convergence}

\begin{theorem}[Comparaison et théorème des gendarmes]\label{thm:g12:seq:squeeze}
Soient $(u_n)$, $(v_n)$, $(w_n)$ des \omterm{def:g12:seq:sequence}{suites}.
\begin{enumerate}
  \item Si $u_n \leq v_n$ à partir d'un certain rang et $u_n \to +\infty$,
        alors $v_n \to +\infty$.
  \item \emph{(Théorème des gendarmes)}\index{théorème des gendarmes} Si
        $u_n \leq v_n \leq w_n$ à partir d'un certain rang et $(u_n)$ et
        $(w_n)$ convergent toutes deux vers la même limite $\ell$, alors
        $(v_n)$ \omterm{def:g12:seq:limit}{converge} vers~$\ell$.
\end{enumerate}
\end{theorem}

\begin{proof}
\emph{1.} Soit $A \in \R$. Comme $u_n \to +\infty$, il existe $N$ avec
$u_n \geq A$ pour $n \geq N$~; en élargissant $N$ si besoin,
$v_n \geq u_n \geq A$ pour $n \geq N$.

\emph{2.} Soit $\varepsilon > 0$. À partir d'un certain rang, à la fois
$\ell - \varepsilon \leq u_n$ et $w_n \leq \ell + \varepsilon$ valent,
d'où $\ell - \varepsilon \leq u_n \leq v_n \leq w_n \leq \ell + \varepsilon$,
c'est-à-dire $\abs{v_n - \ell} \leq \varepsilon$.
\end{proof}

\begin{example}
Pour tout $n \geq 1$, $-\frac{1}{n} \leq \frac{(-1)^n}{n} \leq \frac{1}{n}$,
et les deux bornes tendent vers $0$~; d'où $\frac{(-1)^n}{n} \to 0$.
\end{example}

\begin{theorem}[Théorème de la convergence monotone]\label{thm:g12:seq:monotone}
Une \omterm{def:g12:seq:sequence}{suite} \omterm{def:g12:seq:monotonic}{croissante} \omterm{def:g12:seq:bounded}{majorée} \omterm{def:g12:seq:limit}{converge}. Une \omterm{def:g12:seq:sequence}{suite} \omterm{def:g10:functions:variations}{décroissante} minorée
\omterm{def:g12:seq:limit}{converge}. Une \omterm{def:g12:seq:sequence}{suite} \omterm{def:g12:seq:monotonic}{croissante} non \omterm{def:g12:seq:bounded}{majorée} tend vers $+\infty$.
\end{theorem}

\begin{proof}[Démonstration partielle]
On prouve le troisième énoncé. Soit $(u_n)$ \omterm{def:g12:seq:monotonic}{croissante} et non \omterm{def:g12:seq:bounded}{majorée}, et
soit $A \in \R$. Comme $A$ n'est pas un majorant, il existe $N$ avec
$u_N \geq A$~; par monotonie, $u_n \geq u_N \geq A$ pour tout
$n \geq N$. D'où $u_n \to +\infty$.

Les deux énoncés de convergence s'appuient sur la propriété de la borne
supérieure de $\R$~; ils sont \emph{admis à ce niveau} (et prouvés en
première année d'université).
\end{proof}

\begin{remark}
Le théorème garantit l'\emph{existence} de la limite mais n'en donne pas
la valeur. Une \omterm{def:g12:seq:sequence}{suite} \omterm{def:g12:seq:monotonic}{croissante} \omterm{def:g12:seq:bounded}{majorée} par $M$ \omterm{def:g12:seq:limit}{converge} vers un certain
$\ell \leq M$, pas nécessairement vers $M$.
\end{remark}

\begin{theorem}[Limite des suites géométriques]\label{thm:g12:seq:geometric}
Soit $q \in \R$.
\begin{enumerate}
  \item Si $q > 1$, alors $q^n \to +\infty$.
  \item Si $q = 1$, alors $q^n \to 1$.
  \item Si $\abs{q} < 1$, alors $q^n \to 0$.
  \item Si $q \leq -1$, alors $(q^n)$ diverge et n'a pas de limite.
\end{enumerate}
\end{theorem}

\begin{proof}
\emph{1.} Écrire $q = 1 + a$ avec $a > 0$. L'inégalité de Bernoulli
(\cref{ex:g12:seq:bernoulli}) donne $q^n \geq 1 + na \to +\infty$, et on
conclut par comparaison (\cref{thm:g12:seq:squeeze}).

\emph{2.} Immédiat.

\emph{3.} Si $q = 0$ l'assertion est claire. Sinon $\abs{q} < 1$ donne
$1/\abs{q} > 1$, donc $(1/\abs{q})^n \to +\infty$ par le point~1, d'où
$\abs{q}^n \to 0$, et $-\abs{q}^n \leq q^n \leq \abs{q}^n$ permet de
conclure par le théorème des gendarmes.

\emph{4.} Pour $q \leq -1$, $(q^{2n})$ prend des valeurs $\geq 1$ tandis
que $(q^{2n+1})$ prend des valeurs $\leq -1$~: aucune limite unique ne
peut attirer les deux \omterm{def:g12:seq:sequence}{sous-suites}.
\end{proof}

\begin{method}[Suites récurrentes $u_{n+1} = f(u_n)$]\label{met:g12:seq:recurrent}
Pour étudier une \omterm{def:g12:seq:sequence}{suite} définie par $u_{n+1} = f(u_n)$~:
\begin{enumerate}
  \item prouver par récurrence que $(u_n)$ reste dans un \omterm{def:g10:numbers:interval}{intervalle} $I$
        sur lequel $f$ se comporte bien (et, souvent, que $(u_n)$ est
        \omterm{def:g11:func:monotone}{monotone})~;
  \item déduire la convergence du théorème de la convergence \omterm{def:g11:func:monotone}{monotone}~;
  \item passer à la limite dans la relation $u_{n+1} = f(u_n)$~: si $f$
        est continue et $u_n \to \ell \in I$, alors $\ell$ satisfait
        $f(\ell) = \ell$ (voir le \cref{ch:g12:limcont})~; résoudre cette
        \omterm{def:g10:algebra:equation}{équation} et sélectionner la bonne \omterm{def:g11:quad:discriminant}{racine}.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g12:seq:sqrt2}
Soit $u_0 = 2$ et $u_{n+1} = \frac{1}{2}\left(u_n + \frac{2}{u_n}\right)$.
On vérifie par récurrence que $u_n \geq \sqrt{2}$ pour tout $n$
(l'inégalité $\frac{1}{2}(x + 2/x) \geq \sqrt{2}$ pour $x>0$ équivaut à
$(x - \sqrt2)^2 \geq 0$), puis que $(u_n)$ est \omterm{def:g10:functions:variations}{décroissante}, car
\[
u_{n+1}-u_n=\frac{2-u_n^2}{2u_n}\leq 0 .
\]
\omterm{def:g10:functions:variations}{Décroissante} et minorée, $(u_n)$ \omterm{def:g12:seq:limit}{converge} vers un certain
$\ell \geq \sqrt{2}$, qui doit satisfaire
$\ell = \frac{1}{2}(\ell + 2/\ell)$, c'est-à-dire $\ell^2 = 2$. D'où
$u_n \to \sqrt{2}$. C'est l'algorithme d'Héron, déjà utilisé par les
Babyloniens~; sa convergence est extrêmement rapide ($u_3$ donne déjà
$\sqrt 2$ à huit décimales).
\end{example}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:g12:seq:1}
Prouver par récurrence que pour tout $n \in \N$,
\[
1^2 + 2^2 + \dots + n^2 = \frac{n(n+1)(2n+1)}{6}.
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:seq:2}
Étudier la monotonie des \omterm{def:g12:seq:sequence}{suites} définies pour $n \geq 1$ par
\[
a_n = \frac{n+1}{n}, \qquad
b_n = \frac{2^n}{n}, \qquad
c_n = n^2 - 10n .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:seq:3}
Calculer les limites des \omterm{def:g12:seq:sequence}{suites} de termes généraux
\[
u_n = \frac{3n^2 - n + 1}{2n^2 + 5}, \qquad
v_n = \sqrt{n+1} - \sqrt{n}, \qquad
w_n = \frac{2^n - 3^n}{3^n + 1}.
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:seq:4}
Soit $(u_n)$ la \omterm{def:g11:seq:arithmetic}{suite arithmétique} avec $u_0 = 5$ et raison $r = 3$, et
$(v_n)$ la \omterm{def:g11:seq:geometric}{suite géométrique} avec $v_0 = 8$ et raison $q = \frac{1}{2}$.
Calculer $u_n$, $v_n$, $\sum_{k=0}^{n} u_k$ et $\sum_{k=0}^{n} v_k$, et
les limites des quatre expressions lorsque $n \to +\infty$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:seq:5}
En utilisant le théorème des gendarmes, calculer
\[
\lim_{n\to+\infty} \frac{n + \cos n}{n + 1}
\qquad\text{et}\qquad
\lim_{n\to+\infty} \frac{n!}{n^n},
\]
où $n! = 1 \times 2 \times \dots \times n$. Pour la seconde limite, borner
$\frac{n!}{n^n}$ par un terme d'une \omterm{def:g11:seq:geometric}{suite géométrique}.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:seq:6}
Soit $u_0 = 0$ et $u_{n+1} = \sqrt{u_n + 2}$ pour tout $n \in \N$.
\begin{enumerate}
  \item Prouver par récurrence que $0 \leq u_n \leq 2$ pour tout $n$.
  \item Montrer que $(u_n)$ est \omterm{def:g12:seq:monotonic}{croissante}.
  \item En déduire que $(u_n)$ \omterm{def:g12:seq:limit}{converge} et déterminer sa limite.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:seq:7}
Un patient prend une dose de $1$ unité d'un médicament chaque matin.
Pendant chaque période de $24$ heures, le corps élimine $40\,\%$ du
médicament présent. Soit $u_n$ la quantité de médicament dans le corps
juste après la dose du jour $n$, de sorte que $u_0 = 1$.
\begin{enumerate}
  \item Justifier que $u_{n+1} = 0.6\,u_n + 1$.
  \item Soit $v_n = u_n - 2.5$. Montrer que $(v_n)$ est \omterm{def:g12:seq:arith-geom}{géométrique} et en
        déduire une formule explicite pour $u_n$.
  \item Déterminer la quantité de médicament à long terme dans le corps.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:seq:8}
Soit $(u_n)$ définie par $u_0 = 3$ et $u_{n+1} = \frac{4u_n - 1}{u_n + 2}$.
\begin{enumerate}
  \item Montrer par récurrence que $u_n > 1$ pour tout $n \in \N$.
  \item Montrer que $v_n = \dfrac{1}{u_n - 1}$ définit une \omterm{def:g11:seq:arithmetic}{suite
        arithmétique}.
  \item En déduire des formules explicites pour $v_n$ et $u_n$, et la
        limite de $(u_n)$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:seq:9}
Pour $n \geq 1$, soit
$H_n = 1 + \frac{1}{2} + \frac{1}{3} + \dots + \frac{1}{n}$.
\begin{enumerate}
  \item Montrer que pour tout $n \geq 1$, $H_{2n} - H_n \geq \frac{1}{2}$.
  \item En déduire que $H_{2^k} \geq 1 + \frac{k}{2}$ pour tout
        $k \in \N$, et conclure que $H_n \to +\infty$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:seq:10}
\emph{(\omterm{def:g12:seq:sequence}{Suites} adjacentes.)} Deux \omterm{def:g12:seq:sequence}{suites} $(a_n)$ et $(b_n)$ sont
\emph{adjacentes} si $(a_n)$ est \omterm{def:g12:seq:monotonic}{croissante}, $(b_n)$ est \omterm{def:g10:functions:variations}{décroissante}, et
$b_n - a_n \to 0$.
\begin{enumerate}
  \item Montrer que pour tout $n$, $a_n \leq b_n$. (Indication~: étudier
        la monotonie de $(b_n - a_n)$.)
  \item Montrer que des \omterm{def:g12:seq:sequence}{suites} adjacentes convergent toutes deux vers la
        même limite.
  \item Application~: montrer que les \omterm{def:g12:seq:sequence}{suites}
        $a_n = \sum_{k=0}^{n} \frac{1}{k!}$ et
        $b_n = a_n + \frac{1}{n \cdot n!}$ ($n \geq 1$) sont adjacentes.
        (Leur limite commune est le nombre $\eu$, étudié dans le
        \cref{ch:g12:exp}.)
\end{enumerate}
\end{exercise}
