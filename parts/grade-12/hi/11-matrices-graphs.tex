\chapter{मैट्रिक्स और ग्राफ़}\label{ch:g12:matrix}

ए मैट्रिक्स यह संख्याओं की एक आयताकार सरणी है, जिसे नियमों के अनुसार \omterm{def:g10:vectors:sum}{जोड़}ा और गुणा किया जाता है
ऐसा डिज़ाइन किया गया है मैट्रिक्स बीजगणित रैखिक की संरचना का प्रतिनिधित्व करता है
\omterm{def:g12:comb:permutation}{परिवर्तन}. मैट्रिक्स हल करते हैं रैखिक प्रणाली, ड्राइव युग्मित आवर्तक
दृश्यों, और नेटवर्क में चलने की गिनती --- खोज के पीछे का गणित
इंजन और सबसे छोटा पथ एल्गोरिदम।

\section{मैट्रिक्स बीजगणित}

\begin{definition}[मैट्रिक्स]\label{def:g12:matrix:matrix}
एक \emph{$m \times n $ मैट्रिक्स}\index{$ m \times n $ मैट्रिक्स} की एक तालिका है \omterm{def:g10:numbers:sets}{वास्तविक संख्या} साथ
$ m $ पंक्तियाँ और $ n $ कॉलम:
$ A = (a_{ij})$, कहाँ $ a_{ij}$ पंक्ति में प्रविष्टि है $ i $, स्तंभ $ j $.
एक ही आकार के दो आव्यूहों को प्रविष्टि दर प्रविष्टि \omterm{def:g10:vectors:sum}{जोड़}ा जाता है, और
$\lambda A = (\lambda a_{ij})$.
\end{definition}

\begin{definition}[मैट्रिक्स उत्पाद]\label{def:g12:matrix:product}
होने देना $ A $ होना $ m \times n $ और $ B $ होना $ n \times p $. उत्पाद $ AB $ है
$ m \times p $ मैट्रिक्स किसका $(i,j)$ प्रवेश है
\[
(AB)_{ij} = \sum_{k=1}^{n} a_{ik} b_{kj}
\]
(``पंक्ति $ i $ का $ A $ समय स्तंभ $ j $ का $ B $'' नियम)।
\end{definition}

\begin{example}
$\begin{pmatrix} 1 & 2\\ 3 & 4\end{pmatrix}
\begin{pmatrix} 0 & 1\\ 1 & 1\end{pmatrix}
= \begin{pmatrix} 2 & 3\\ 4 & 7\end{pmatrix}$,
\quad जबकि \quad
$\begin{pmatrix} 0 & 1\\ 1 & 1\end{pmatrix}
\begin{pmatrix} 1 & 2\\ 3 & 4\end{pmatrix}
= \begin{pmatrix} 3 & 4\\ 4 & 6\end{pmatrix}$:
\emph{मैट्रिक्स गुणन क्रमविनिमेय नहीं है}.
\end{example}

\begin{proposition}[मैट्रिक्स बीजगणित के नियम]\label{prop:g12:matrix:rules}
जब भी आकार उत्पादों को सार्थक बनाते हैं:
\[
(AB)C = A(BC), \qquad A(B + C) = AB + AC, \qquad (A+B)C = AC + BC,
\]
और यह \emph{पहचान मैट्रिक्स} $ I_n $ (विकर्ण पर एक, अन्यत्र शून्य)
संतुष्ट करता है $ I_m A = A I_n = A $ के लिए $ A $ आकार का $ m \times n $.
\end{proposition}

\begin{proof}
सभी प्रवेश-दर-प्रवेश सत्यापन हैं
\cref{def:g12:matrix:product}; साहचर्य, एकमात्र गैर-तुच्छ,
यह दो सीमित राशियों के आदान-प्रदान के बराबर है:
$\bigl((AB)C\bigr)_{ij} = \sum_l \left(\sum_k a_{ik}b_{kl}\right) c_{lj}
= \sum_k a_{ik} \left(\sum_l b_{kl} c_{lj}\right)
= \bigl(A(BC)\bigr)_{ij}$.
\end{proof}

\begin{definition}[श्लोक में]\label{def:g12:matrix:inverse}
एक वर्ग मैट्रिक्स $ A $ आकार का $ n $ है \emph{उलटी}\index{उलटी}
अगर कोई है मैट्रिक्स $ B $ साथ $ AB = BA = I_n $; $ B $ फिर अद्वितीय है, लिखा गया है
$ A^{-1}$.
\end{definition}

\begin{proposition}[ए का उलटा $2\times2$ मैट्रिक्स]\label{prop:g12:matrix:inverse2x2}
होने देना $ A = \begin{pmatrix} a & b\\ c & d\end{pmatrix}$ और
$\det A = ad - bc $ (द \emph{सिद्ध}\index{सिद्ध}). तब $ A $ है
\omterm{def:g12:matrix:inverse}{उलटी} अगर और केवल अगर $\det A \neq 0$, किस स्थिति में
\[
A^{-1} = \frac{1}{ad - bc}\begin{pmatrix} d & -b\\ -c & a\end{pmatrix}.
\]
\end{proposition}

\begin{proof}
एक गणना देती है
$ A \begin{pmatrix} d & -b\\ -c & a\end{pmatrix}
= \begin{pmatrix} d & -b\\ -c & a\end{pmatrix} A = (ad - bc) I_2$; अगर
$ ad - bc \neq 0$, विभाजित करना। इसके विपरीत, यदि $ ad - bc = 0$, के कॉलम $ A $
आनुपातिक हैं, और इसी प्रकार के स्तंभ भी हैं $ AB $ किसी के लिए $ B $; लेकिन
के कॉलम $ I_2$ आनुपातिक नहीं हैं, इसलिए नहीं $ B $ संतुष्ट कर सकते हैं $ AB = I_2$.
\end{proof}

\begin{method}[रैखिक प्रणाली]\label{met:g12:matrix:systems}
सिस्टम
$\begin{cases} ax + by = e\\ cx + dy = f \end{cases}$
है मैट्रिक्स \omterm{def:g10:algebra:equation}{समीकरण} $ AX = Y $ साथ
$ X = \begin{pmatrix} x \\ y\end{pmatrix}$,
$ Y = \begin{pmatrix} e \\ f\end{pmatrix}$. अगर $\det A \neq 0$, यह अद्वितीय है
समाधान है $ X = A^{-1}Y $. वही औपचारिकता संभालती है $ n $ \omterm{def:g10:algebra:equation}{समीकरण} में $ n $
अज्ञात.
\end{method}

\section{मैट्रिक्स शक्तियाँ और आवर्ती अनुक्रम}

\begin{definition}\label{def:g12:matrix:power}
एक वर्ग के लिए मैट्रिक्स $ A $ और $ k \in \N $, $ A^k = A \times \dots \times A $
($ k $ कारक), साथ $ A^0 = I $.
\end{definition}

\begin{method}[विकर्ण-प्लस-निलपोटेंट और विकर्णीय मामले]\label{met:g12:matrix:powers}
गणना करने के दो मानक तरीके $ A^k $:
\begin{itemize}
  \item अगर $ A = \lambda I + N $ कहाँ $ N^2 = 0$, द्विपद प्रमेय
        (तब से यहां मान्य है $ I $ और $ N $ आवागमन) दो शब्दों में संक्षिप्त हो जाता है:
        $ A^k = \lambda^k I + k \lambda^{k-1} N $.
  \item अगर किसी को कोई मिल जाए \omterm{def:g12:matrix:inverse}{उलटी} $ P $ साथ $ A = PDP^{-1}$ और $ D $ विकर्ण,
        फिर $ A^k = P D^k P^{-1}$, और $ D^k $ प्रविष्टि द्वारा प्रविष्टि की गणना की जाती है।
        (ऐसा ढूँढना $ P $ व्यवस्थित रूप से का \omterm{def:g11:vect:det}{सिद्ध}ांत है
        \emph{विकर्णन}, विश्वविद्यालय में विकसित; इस स्तर पर $ P $ दिया हुआ है।)
\end{itemize}
\end{method}

\begin{example}[युग्मित अनुक्रम]\label{ex:g12:matrix:coupled}
होने देना $ u_{n+1} = 3u_n + v_n $ और $ v_{n+1} = u_n + 3v_n $. सेटिंग
$ X_n = \begin{pmatrix} u_n\\ v_n \end{pmatrix}$ और
$ A = \begin{pmatrix} 3 & 1\\ 1 & 3\end{pmatrix}$, हम पाते हैं
$ X_{n+1} = AX_n $, इसलिए $ X_n = A^n X_0$. सहायक दृश्यों
$ s_n = u_n + v_n $ और $ d_n = u_n - v_n $ संतुष्ट $ s_{n+1} = 4s_n $ और
$ d_{n+1} = 2d_n $, इसलिए $ s_n = 4^n s_0$, $ d_n = 2^n d_0$ और
\[
u_n = \frac{4^n(u_0+v_0) + 2^n(u_0-v_0)}{2}, \qquad
v_n = \frac{4^n(u_0+v_0) - 2^n(u_0-v_0)}{2}.
\]
(पर्दे के पीछे: $(1,1)$ और $(1,-1)$ की eigenvector दिशाएँ हैं $ A $.)
\end{example}

\section{रेखांकन और चलता है}

\begin{definition}[ग्राफ़, आसन्नता मैट्रिक्स]\label{def:g12:matrix:graph}
ए \emph{ग्राफ}\index{ग्राफ} शीर्षों से मिलकर बना है $1, 2, \dots, n $ और किनारे
शीर्षों के कुछ युग्मों को जोड़ना (ए के लिए क्रमित युग्म)। \emph{निर्देशित}
ग्राफ़)। इसका \emph{सहखंडज मैट्रिक्स}\index{सहखंडज मैट्रिक्स} है
$ n \times n $ मैट्रिक्स $ M $ साथ $ m_{ij} = 1$ अगर कोई किनारा है $ i $ को
$ j $, और $0$ अन्यथा। ए \emph{टहलना} लम्बाई का $ k $ से $ i $ को $ j $ एक है
\omterm{def:g12:seq:sequence}{अनुक्रम} का $ k $ लगातार किनारे से आगे बढ़ रहे हैं $ i $ को $ j $.
\end{definition}

\begin{tikzpicture}[baseline=(current bounding box.center)]
  \node[circle, draw, inner sep=2.5pt] (v1) at (0,0) {$1$};
  \node[circle, draw, inner sep=2.5pt] (v2) at (2.6,1.2) {$2$};
  \node[circle, draw, inner sep=2.5pt] (v3) at (2.6,-1.2) {$3$};
  \draw[-Stealth, omDef, thick] (v1) -- (v2);
  \draw[-Stealth, omDef, thick] (v2) -- (v3);
  \draw[-Stealth, omDef, thick] (v1) to[bend right=18] (v3);
  \draw[-Stealth, omDef, thick] (v3) to[bend right=18] (v1);
\end{tikzpicture}
\qquad\qquad
$ M = \begin{pmatrix} 0 & 1 & 1\\ 0 & 0 & 1\\ 1 & 0 & 0 \end{pmatrix}$

{\small ए निर्देशित \omterm{def:g12:matrix:graph}{ग्राफ} और इसके \omterm{def:g12:matrix:graph}{सहखंडज मैट्रिक्स}
(\cref{exo:g12:matrix:6}): $ m_{ij} = 1$ बिल्कुल जब कोई किनारा हो
से $ i $ को $ j $.}
\end{omfigure}

\begin{theorem}[गिनती चलती है]\label{thm:g12:matrix:walks}
लंबाई में चलने की संख्या $ k $ शिखर से $ i $ शिखर तक $ j $ है
$(i,j)$ का प्रवेश $ M^k $.
\end{theorem}

\begin{proof}
प्रेरण चालू $ k $. के लिए $ k = 1$ यह की परिभाषा है $ M $. मान लीजिये
के लिए दावा करें $ k $. लम्बी सैर $ k+1$ से $ i $ को $ j $ लम्बी सैर है
$ k $ से $ i $ किसी शिखर तक $ l $, इसके बाद बढ़त हुई $ l $ को $ j $; द्वारा
\omterm{def:g10:vectors:sum}{जोड़} और गुणन \omterm{def:g11:vect:det}{सिद्ध}ांत, उनकी संख्या है
\[
\sum_{l=1}^{n} \bigl(M^k\bigr)_{il}\, m_{lj} = \bigl(M^{k+1}\bigr)_{ij}.
\qedhere
\]
\end{proof}

\begin{example}
त्रिकोण के लिए \omterm{def:g12:matrix:graph}{ग्राफ} ($3$ शीर्ष, सभी \omterm{def:g10:vectors:sum}{जोड़}े जुड़े हुए),
$ M = \begin{pmatrix} 0&1&1\\ 1&0&1\\ 1&1&0\end{pmatrix}$ और
$ M^2 = \begin{pmatrix} 2&1&1\\ 1&2&1\\ 1&1&2\end{pmatrix}$: प्रत्येक शीर्ष से
वहाँ हैं $2$ लंबाई की सैर $2$ अपने आप में वापस (किसी भी पड़ोसी के माध्यम से) और
$1$ एक दूसरे शीर्ष पर.
\end{example}

\section{अभ्यास}

\begin{exercise}[$\star $]\label{exo:g12:matrix:1}
होने देना $ A = \begin{pmatrix} 1 & 2\\ 0 & 1 \end{pmatrix}$ और
$ B = \begin{pmatrix} 2 & 0\\ 1 & 1 \end{pmatrix}$. गणना करना $ A + B $, $ AB $,
$ BA $ और $ A^2$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g12:matrix:2}
निर्धारित करें कि क्या निम्नलिखित आव्यूह हैं \omterm{def:g12:matrix:inverse}{उलटी}, और गणना करें
जब वे मौजूद हों तो व्युत्क्रम:
\[
A = \begin{pmatrix} 2 & 5\\ 1 & 3\end{pmatrix}, \qquad
B = \begin{pmatrix} 3 & 6\\ 2 & 4\end{pmatrix}.
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g12:matrix:3}
द्वारा हल करें मैट्रिक्स सिस्टम को उलट देना
$\begin{cases} 2x + 5y = 1\\ x + 3y = 2 . \end{cases}$
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:matrix:4}
होने देना $ A = \begin{pmatrix} 2 & 1\\ 0 & 2\end{pmatrix} = 2I + N $ साथ
$ N = \begin{pmatrix} 0 & 1\\ 0 & 0 \end{pmatrix}$.
\begin{enumerate}
  \item जाँच करें कि $ N^2 = 0$ ओर वो $ I $ और $ N $ आना-जाना।
  \item परिणाम निकालना $ A^k $ सभी के लिए $ k \in \N $ और इसके लिए सूत्र को सत्यापित करें $ k=2$ द्वारा
        प्रत्यक्ष गणना.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:matrix:5}
होने देना $ A = \begin{pmatrix} 0 & 1\\ 1 & 1\end{pmatrix}$ और $ F_n $ फाइबोनैचि
\omterm{def:g12:seq:sequence}{अनुक्रम} ($ F_0 = 0$, $ F_1 = 1$, $ F_{n+2} = F_{n+1} + F_n $). द्वारा दिखाएँ
उस के लिए प्रेरण $ n \geq 1$,
\[
A^n = \begin{pmatrix} F_{n-1} & F_n\\ F_n & F_{n+1}\end{pmatrix},
\]
और पहचान निकालें $ F_{n+1}F_{n-1} - F_n^2 = (-1)^n $.
(संकेत देना: निर्धारकों गुणा करें: $\det(MN) = \det M \det N $, जो आप कर सकते हैं
के लिए जाँच करें $2\times2$ मैट्रिक्स।)
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:matrix:6}
ए निर्देशित \omterm{def:g12:matrix:graph}{ग्राफ} शीर्षों पर $\{1, 2, 3\}$ किनारे हैं
$1\to2$, $2\to3$, $3\to1$ और $1\to3$.
\begin{enumerate}
  \item लिखना \omterm{def:g12:matrix:graph}{सहखंडज मैट्रिक्स} $ M $ और गणना करें $ M^2$ और $ M^3$.
  \item लंबाई कितनी है $3$ यहां से जाओ $1$ को $1$? उनकी सूची बनाओ।
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:matrix:7}
एक कार-शेयरिंग कंपनी दो शहरों के बीच वाहनों को ले जाती है $ A $ और $ B $. प्रत्येक सप्ताह, $80\%$ अंदर की कारों में से $ A $ अंदर ही रहना $ A $ और $20\%$ करने के लिए कदम $ B $; $30\%$
अंदर की कारों में से $ B $ करने के लिए कदम $ A $ और $70\%$ रहना। होने देना $ a_n, b_n $ हो
प्रत्येक शहर में बेड़े का अनुपात.
\begin{omfigure}
\begin{tikzpicture}
  \node[circle, draw, inner sep=3pt] (A) at (0,0) {$ A $};
  \node[circle, draw, inner sep=3pt] (B) at (3.2,0) {$ B $};
  \draw[-Stealth, omDef, thick] (A) to[bend left=25]
    node[above, font=\small] {$0.2$} (B);
  \draw[-Stealth, omDef, thick] (B) to[bend left=25]

\end{tikzpicture}
\end{omfigure}
\begin{enumerate}
  \item लिखना $ X_{n+1} = MX_n $ साथ
        $ X_n = \begin{pmatrix} a_n\\ b_n\end{pmatrix}$ और पहचानो $ M $.
  \item संतुलन अनुपात ज्ञात करें (हल करें)। $ MX = X $ साथ
        $ a + b = 1$).
  \item बताते हैं कि $ c_n = a_n - 0.6$ संतुष्ट $ c_{n+1} = 0.5\,c_n $, और
        निष्कर्ष निकालें कि बेड़ा \omterm{def:g11:prob:rv}{वितरण} \omterm{def:g12:seq:limit}{अभिसरण} संतुलन के लिए.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g12:matrix:8}
होने देना $ A = \begin{pmatrix} 3 & 1\\ 1 & 3 \end{pmatrix}$,
$ P = \begin{pmatrix} 1 & 1\\ 1 & -1 \end{pmatrix}$.
\begin{enumerate}
  \item गणना करना $ P^{-1}$, तब $ D = P^{-1}AP $, और उसकी जांच करें $ D $ है
        विकर्ण.
  \item के लिए एक बंद सूत्र निकालें $ A^n $ और तुलना करें
        \cref{ex:g12:matrix:coupled}.
\end{enumerate}
\end{exercise}

\section{समस्या: मैट्रिक्स जो फाइबोनैचि (और) को जानता है
मौसम)}

\begin{problem}[{सप्ताहान्त समस्या---एक $2 \times 2$ मैट्रिक्स सभी फाइबोनैचि को वहन करता है, एक मार्कोव मैट्रिक्स लंबे समय तक पूर्वानुमान लगाता है मौसम, और एक आइजनवेक्टर की कीमत एक अरब डॉलर है}]
\label{pb:g12:matrix:1}
ए मैट्रिक्स एक मशीन है जो एक राज्य को खाती है और अगले को वापस कर देती है
--- और इसके \emph{पॉवर्स} इसलिए संपूर्ण वायदा संभाल कर रखें। यह
समस्या आश्चर्यजनक रूप से खुलती है मैट्रिक्स जिनकी शक्तियाँ सूचीबद्ध हैं
फाइबोनैचि संख्याएँ (और प्रत्येक पंक्ति में उनकी पहचान \omterm{def:g11:vect:det}{सिद्ध} करें),
फिर मौसम को मार्कोव श्रृंखला के रूप में उसकी स्थिर अवस्था में चलाता है, और
ईजेनवेक्टर के साथ बंद हो जाता है जिस पर एक खोज इंजन बनाया गया था
(\cref{thm:g12:matrix:walks},
\cref{met:g12:matrix:powers}).

\medskip
\noindent\textbf{Part I --- Fluency.}
\begin{enumerate}
  \item साथ $ A = \begin{pmatrix} 1 & 2\\ 3 & 4\end{pmatrix}$
        और $ B = \begin{pmatrix} 0 & 1\\ 1 & 0\end{pmatrix}$:
        गणना करें $ AB $ और $ BA $. कम्यूटेटिविटी पर फैसला?
  \item औंधाना $\begin{pmatrix} 2 & 1\\ 5 & 3\end{pmatrix}$
        (\cref{prop:g12:matrix:inverse2x2}) और व्युत्क्रम का प्रयोग करें
        हल करना $2x + y = 4$, $5x + 3y = 7$.
  \item होने देना $ N = \begin{pmatrix} 0 & 1\\ 0 & 0\end{pmatrix}$:
        गणना करें $ N^2$, और निष्कर्ष निकालें
        $(I + N)^n = I + nN $ हरएक के लिए $ n $.
  \item त्रिकोण \omterm{def:g12:matrix:graph}{ग्राफ} (तीन शीर्ष, सभी \omterm{def:g10:vectors:sum}{जोड़}े जुड़े हुए):
        इसे लिखें \omterm{def:g12:matrix:graph}{सहखंडज मैट्रिक्स} $ A $, गणना करें $ A^3$, और
        विकर्ण प्रविष्टियों की व्याख्या करें
        (\cref{thm:g12:matrix:walks}).
  \item के लिए $ D = \begin{pmatrix} 2 & 0\\ 0 & \frac12
        \end{pmatrix}$: देना $ D^n $ और इसका व्यवहार इस प्रकार है
        $ n \to \infty $.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The Fibonacci मैट्रिक्स.}
होने देना $ F = \begin{pmatrix} 1 & 1\\ 1 & 0\end{pmatrix}$ और चलो
$ F_1 = F_2 = 1, F_3 = 2, \dots $ की फाइबोनैचि संख्याएँ हों
\cref{pb:g11:seq:1}.
\begin{enumerate}[resume]
  \item गणना करना $ F^2$, $ F^3$, $ F^4$ और सामान्य अनुमान लगाएं
        का रूप $ F^n $ फाइबोनैचि संख्या के संदर्भ में.
  \item अनुमान को \omterm{def:g11:vect:det}{सिद्ध} करें
        $ F^n = \begin{pmatrix} F_{n+1} & F_n\\ F_n & F_{n-1}
        \end{pmatrix}$ प्रेरण द्वारा.
  \item लेना निर्धारकों दोनों पक्षों का (.) \omterm{def:g11:vect:det}{सिद्ध} एक का
        उत्पाद का उत्पाद है निर्धारकों --- इसे जांचें
        पर $2 \times 2$ यदि आपने इसे कभी नहीं देखा है तो मैट्रिक्स):
        निष्कर्ष निकालना \emph{कैसिनी की पहचान}
        $ F_{n+1}F_{n-1} - F_n^2 = (-1)^n $ --- द
        लुप्त-स्क्वायर इंजन, एक पंक्ति में साबित हुआ।
  \item से $ F^{m+n} = F^m F^n $, ऊपर दाईं ओर पढ़ें
        प्रविष्टियाँ करें और प्राप्त करें \emph{अतिरिक्त सूत्र}
        \[
        F_{m+n} = F_{m+1} F_n + F_m F_{n-1} .
        \]
        इसकी जांच करें $ m = n = 3$.
  \item \omterm{def:g10:vectors:sum}{जोड़} सूत्र से निष्कर्ष निकालें (प्रेरण पर $ k $)
        वह $ F_n $ विभाजित $ F_{kn}$, और सत्यापित करें
        $ F_3 \mid F_6$ और $ F_3 \mid F_9$.
  \item गणना करना $ F_{100}$, किसी को गुणा करने की आवश्यकता नहीं है $100$
        मैट्रिक्स: बार-बार वर्ग ($ F^2, F^4, F^8, \dots $)
        और गठबंधन करें. कितने मैट्रिक्स गुणन पर्याप्त है,
        और \omterm{prop:g10:coordgeom:midpoint}{मध्य} से कौन सी \omterm{def:g12:integ:primitive}{प्राचीन} गुणन युक्ति
        स्कूल की मात्रा यह है, मैट्रिक्स में पदोन्नत?
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The weather machine.}
एक निश्चित शहर में: एक धूप वाले दिन के बाद, अगला दिन धूप वाला होता है
\omterm{def:g10:proba:distribution}{संभावना} $0.8$; बरसात के दिन के बाद, धूप खिली \omterm{def:g10:proba:distribution}{संभावना}
$0.4$. दिन का एनकोड करें \omterm{def:g11:prob:rv}{वितरण} एक स्तंभ के रूप में
$\binom{p_{\text{sun}}}{p_{\text{rain}}}$ और विकास द्वारा
\[
M = \begin{pmatrix} 0.8 & 0.4\\ 0.2 & 0.6 \end{pmatrix}.
\]
\begin{enumerate}[resume]
  \item जांचें कि प्रत्येक कॉलम $ M $ का योग है $1$, और बताओ क्यों
        किसी भी मौसम मशीन में यह गुण अवश्य होना चाहिए।
  \item आज धूप है. कल और के लिए पूर्वानुमान की गणना करें
        अगले दिन के लिए.
  \item खोजें \emph{स्थिर अवस्था}: द \omterm{def:g11:prob:rv}{वितरण} $ v $ साथ
        $ Mv = v $ (और प्रविष्टियों का योग $1$). का कितना अंश
        क्या दिन लंबे समय तक धूप वाले रहते हैं?
  \item बरसात के दिन से शुरू करें, $\binom01$, और आवेदन करें $ M $ चार
        समय, दूरी को स्थिर अवस्था में ट्रैक करना
        हर कदम. प्रति चरण \omterm{def:g10:numbers:interval}{अंतराल} किस कारक से कम होता है?
        --- और यह किस प्रकार का \omterm{def:g12:seq:limit}{अभिसरण} है?
  \item लघु रूप में पेजरैंक: तीन पेज, लिंक के साथ
        $ A \to B $, $ A \to C $, $ B \to C $, $ C \to A $. एक यादृच्छिक
        सर्फ़र यादृच्छिक रूप से एक आउटगोइंग लिंक का समान रूप से अनुसरण करता है।
        संक्रमण लिखें मैट्रिक्स, स्थिर अवस्था ज्ञात करें, और
        पेजों को रैंक करें.
  \item रैंकिंग की व्याख्या करें: क्यों करता है $ C $ जितना उच्च स्कोर
        $ A $ कम पृष्ठों से लिंक प्राप्त होने के बावजूद --- क्या
        क्या स्थिर अवस्था वास्तव में मापती है? (वास्तविक पेजरैंक
        मृत सिरों और छलांगों के लिए एक अवमंदन कारक जोड़ता है; द
        ईजेनवेक्टर विचार बिल्कुल यही है।)
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Diagonal dividends.}
\begin{enumerate}[resume]
  \item दो युग्मित मात्राएँ आज्ञापालन करती हैं
        $ u_{n+1} = 3u_n + v_n $, $ v_{n+1} = u_n + 3v_n $, \omterm{def:g11:stat:mean}{अर्थ}ात\
        द मैट्रिक्स $ A $ का \cref{exo:g12:matrix:8}. उसका उपयोग करना
        व्यायाम का विकर्णीकरण
        ($ D = \operatorname{diag}(4, 2)$), बंद दे दो
        के लिए सूत्र $ u_n $ कब $ u_0 = 1$, $ v_0 = 0$, और जांचें
        यह प्रत्यक्ष गणना के विरुद्ध है $ n = 1, 2, 3$.
  \item एक या दो वाक्यों में: विकर्णीकरण क्या करता है
        \emph{करना} एक युग्मित प्रणाली के लिए --- और किस \omterm{def:g11:stat:mean}{अर्थ} में है
        प्रश्न 14 की मार्कोव स्थिर अवस्था एक आइजेनवेक्टर
        कहानी भी?
  \item समापन--- द मैट्रिक्सइस सप्ताहांत के तीन चेहरे:
        बहीखाता (सिस्टम और व्युत्क्रम), कॉम्बिनेटरिक्स
        (चलता है और लिंक शक्तियों द्वारा गिना जाता है), और विकास
        (फाइबोनैचि, मौसम, वेब --- वायदा पढ़ें
        eigen-दिशाएँ)। प्रत्येक एक वाक्य, प्लस
        फॉरवर्ड पॉइंटर: यूनिवर्सिटी वॉल्यूम्स का रैखिक बीजगणित
        इनमें से प्रत्येक चेहरे को एक \omterm{def:g11:vect:det}{सिद्ध}ांत बनाता है।
\end{enumerate}
\end{problem}
