\chapter{यादृच्छिक चरों का योग और बड़ी संख्याओं का नियम}\label{ch:g12:sums}

कैसीनो हमेशा अंत में क्यों जीतते हैं, और मतदान क्यों काम करते हैं? क्योंकि
कई का औसत \omterm{def:g12:sums:indep}{स्वतंत्र} यादृच्छिक मात्राएँ कम और कम उतार-चढ़ाव करती हैं। यह
अध्याय इसे \omterm{def:g11:vect:det}{सिद्ध} करता है: की रैखिकता \omterm{def:g12:randvar:exp}{अपेक्षा}, की संवेदनशीलता \omterm{def:g12:randvar:exp}{झगड़ा} के लिए
\omterm{def:g12:sums:indep}{स्वतंत्र} चर, Bienaymé--चेबीशेव असमानता, और का कानून
बड़ी संख्या.

\section{यादृच्छिक चरों का योग}

दो दिए गए यादृच्छिक चर $X, Y $ उसी परिमित पर \omterm{def:g11:prob:model}{नमूना अंतरिक्ष}
$\Omega $, योग $ X + Y $ है \omterm{def:g12:randvar:rv}{अनियमित परिवर्तनशील वस्तु}
$\omega \mapsto X(\omega) + Y(\omega)$.

\begin{theorem}[अपेक्षा की रैखिकता]\label{thm:g12:sums:linearity}
\index{अपेक्षा!रैखिकता}
सभी के लिए यादृच्छिक चर $ X, Y $ पर $\Omega $ और $ a, b \in \R $:
\[
\E(X + Y) = \E(X) + \E(Y), \qquad \E(aX + b) = a\E(X) + b .
\]
\emph{नहीं स्वतंत्रता धारणा की आवश्यकता है.}
\end{theorem}

\begin{proof}
लिखना \omterm{def:g12:randvar:exp}{अपेक्षा} परिणामों के योग के रूप में: चूँकि
$\P(X = x) = \sum_{\omega : X(\omega) = x} \P(\{\omega\})$, समूहीकरण शर्तें
देता है $\E(X) = \sum_{\omega \in \Omega} \P(\{\omega\})\,X(\omega)$. तब
\[
\E(X + Y) = \sum_{\omega} \P(\{\omega\})\bigl(X(\omega) + Y(\omega)\bigr)
= \sum_{\omega} \P(\{\omega\})X(\omega)
+ \sum_{\omega} \P(\{\omega\})Y(\omega) = \E(X) + \E(Y). \qedhere
\]
\end{proof}

\begin{definition}[स्वतंत्र यादृच्छिक चर]\label{def:g12:sums:indep}
$ X $ और $ Y $ हैं \emph{स्वतंत्र}\index{स्वतंत्र} यदि
सभी मूल्यों के लिए $ x, y $:
\[
\P(X = x \text{ and } Y = y) = \P(X = x)\,\P(Y = y).
\]
अनेक चर $ X_1, \dots, X_n $ यदि यह उत्पाद नियम स्वतंत्र है
प्रत्येक उपपरिवार के मूल्यों की प्रत्येक पसंद को ध्यान में रखता है।
\end{definition}

\begin{proposition}\label{prop:g12:sums:prodexp}
अगर $ X $ और $ Y $ हैं \omterm{def:g12:sums:indep}{स्वतंत्र}, $\E(XY) = \E(X)\,\E(Y)$.
\end{proposition}

\begin{proof}
\[
\begin{aligned}
\E(XY) &= \sum_{x, y} xy\;\P(X = x \text{ and } Y = y)
= \sum_{x, y} xy\,\P(X=x)\P(Y=y) \\
&= \Bigl(\sum_x x\P(X=x)\Bigr)\Bigl(\sum_y y\P(Y=y)\Bigr). \qedhere
\end{aligned}
\]
\end{proof}

\begin{theorem}[एक राशि का विचरण]\label{thm:g12:sums:variance}
\index{किसी राशि का विचरण!}
अगर $ X $ और $ Y $ हैं \emph{\omterm{def:g12:sums:indep}{स्वतंत्र}}, तब
\[
\V(X + Y) = \V(X) + \V(Y).
\]
अधिक सामान्यतः, के लिए \omterm{def:g12:sums:indep}{स्वतंत्र} $ X_1, \dots, X_n $:
$\V(X_1 + \dots + X_n) = \V(X_1) + \dots + \V(X_n)$.
\end{theorem}

\begin{proof}
कोनिग-ह्यूजेन्स का उपयोग करना (\cref{prop:g12:randvar:konig}) और रैखिकता:
\begin{align*}
\V(X + Y) &= \E\bigl((X+Y)^2\bigr) - \bigl(\E X + \E Y\bigr)^2\\
&= \E(X^2) + 2\E(XY) + \E(Y^2)
- \E(X)^2 - 2\E(X)\E(Y) - \E(Y)^2\\
&= \V(X) + \V(Y) + 2\bigl(\E(XY) - \E(X)\E(Y)\bigr),
\end{align*}
और अंतिम कोष्ठक गायब हो जाता है \omterm{def:g12:sums:indep}{स्वतंत्र} चर
(\cref{prop:g12:sums:prodexp}). सामान्य मामला प्रेरण द्वारा अनुसरण करता है।
\end{proof}

\begin{example}[द्विपद पर दोबारा गौर किया गया]\label{ex:g12:sums:binomial}
एक द्विपद चर $ X \sim \mathcal B(n, p)$ एक राशि है
$ X = X_1 + \dots + X_n $ का $ n $ \omterm{def:g12:sums:indep}{स्वतंत्र} बर्नौली चर. इसलिए,
संरचनात्मक रूप से:
\[
\E(X) = np, \qquad \V(X) = np(1-p),
\]
ठीक हो \cref{thm:g12:randvar:binomial} बिना किसी गणना के.
\end{example}

\begin{proposition}[नमूना माध्य]\label{prop:g12:sums:mean}
होने देना $ X_1, \dots, X_n $ होना \omterm{def:g12:sums:indep}{स्वतंत्र} उसी के साथ \omterm{def:g12:randvar:rv}{वितरण} जैसा $ X $
(\omterm{def:g12:randvar:exp}{अपेक्षा} $\mu $, \omterm{def:g12:randvar:exp}{झगड़ा} $\sigma^2$), और चलो
$ M_n = \frac{X_1 + \dots + X_n}{n}$ उनका हो \emph{नमूना माध्य}\index{नमूना माध्य}. तब
\[
\E(M_n) = \mu, \qquad \V(M_n) = \frac{\sigma^2}{n}, \qquad
\sigma(M_n) = \frac{\sigma}{\sqrt n}.
\]
\end{proposition}

\begin{proof}
रैखिकता देता है $\E(M_n) = \frac{n\mu}{n} = \mu $; स्वतंत्रता देता है
$\V(X_1 + \dots + X_n) = n\sigma^2$, और से विभाजित करना $ n $ तराजू
\omterm{def:g12:randvar:exp}{झगड़ा} द्वारा $\frac{1}{n^2}$ (\cref{prop:g12:randvar:affine}).
\end{proof}

 $\frac{\sigma}{\sqrt n}$ मौलिक है \emph{वर्गमूल नियम}: को
औसत के उतार-चढ़ाव को आधा करें, चौगुना करें \omterm{def:g10:proba:sample}{नमूना} आकार।

\section{एकाग्रता असमानताएँ}

\begin{theorem}[मार्कोव की असमानता]\label{thm:g12:sums:markov}
\index{मार्कोव की असमानता}
अगर $ X \geq 0$ और $ a > 0$:
\[
\P(X \geq a) \leq \frac{\E(X)}{a}.
\]
\end{theorem}

\begin{proof}
में $\E(X) = \sum_i p_i x_i $ (सभी पद अऋणात्मक), केवल पद ही रखें
के साथ $ x_i \geq a $: प्रत्येक कम से कम है $ a\,p_i $, इसलिए
$\E(X) \geq a \sum_{x_i \geq a} p_i = a\,\P(X \geq a)$.
\end{proof}

\begin{theorem}[बिएनेमे--चेबीशेव असमानता]\label{thm:g12:sums:chebyshev}
\index{बिएनेमे--चेबीशेव असमानता}
किसी के लिए \omterm{def:g12:randvar:rv}{अनियमित परिवर्तनशील वस्तु} $ X $ और कोई भी $\varepsilon > 0$:
\[
\P\bigl(\abs{X - \E(X)} \geq \varepsilon\bigr)
\leq \frac{\V(X)}{\varepsilon^2}.
\]
\end{theorem}

\begin{proof}
मार्कोव की असमानता को गैर-नकारात्मक चर पर लागू करें
$ Y = (X - \E(X))^2$ साथ $ a = \varepsilon^2$:
\[
\P\bigl(\abs{X - \E(X)} \geq \varepsilon\bigr)
= \P(Y \geq \varepsilon^2)
\leq \frac{\E(Y)}{\varepsilon^2} = \frac{\V(X)}{\varepsilon^2}. \qedhere
\]
\end{proof}

\begin{theorem}[बड़ी संख्या का नियम]\label{thm:g12:sums:lln}
\index{बड़ी संख्या का नियम}
होने देना $ X_1, \dots, X_n $ होना \omterm{def:g12:sums:indep}{स्वतंत्र} उसी के साथ \omterm{def:g12:randvar:rv}{वितरण}
(\omterm{def:g12:randvar:exp}{अपेक्षा} $\mu $, \omterm{def:g12:randvar:exp}{झगड़ा} $\sigma^2$) और $ M_n $ उनका \omterm{prop:g12:sums:mean}{नमूना माध्य}. हरएक के लिए $\varepsilon > 0$:
\[
\P\bigl(\abs{M_n - \mu} \geq \varepsilon\bigr)
\leq \frac{\sigma^2}{n\,\varepsilon^2}
\xrightarrow[n \to +\infty]{} 0 .
\]
 \omterm{prop:g12:sums:mean}{नमूना माध्य} के आसपास केन्द्रित होता है \omterm{def:g12:randvar:exp}{अपेक्षा}.
\end{theorem}

\begin{proof}
बिएनेमे--चेबीशेव ने आवेदन किया $ M_n $, किसका \omterm{def:g12:randvar:exp}{अपेक्षा} है $\mu $ और किसका
\omterm{def:g12:randvar:exp}{झगड़ा} है $\frac{\sigma^2}{n}$ (\cref{prop:g12:sums:mean}).
\end{proof}

\begin{remark}
यह प्रमेय बीच का सेतु है \omterm{def:g10:proba:distribution}{संभावना} \omterm{def:g11:vect:det}{सिद्ध}ांत और सांख्यिकी:
\emph{\omterm{def:g10:stats:series}{आवृत्ति}} की एक \omterm{def:g11:prob:model}{आयोजन} बहुतों के ऊपर \omterm{def:g12:sums:indep}{स्वतंत्र} दोहराव दृष्टिकोण
इसका \emph{\omterm{def:g10:proba:distribution}{संभावना}} (लेना $ X_i $ का सूचक \omterm{def:g11:prob:model}{आयोजन}, ताकि
$\mu = p $). यह अनुमान लगाने को उचित ठहराता है \omterm{def:g10:proba:distribution}{संभावना} अनुकरण द्वारा
(\emph{मोंटे कार्लो विधि}) और एक सर्वेक्षण द्वारा जनसंख्या अनुपात ---
मात्रात्मक रूप से: देखें \cref{ch:g12:contdist}.
\end{remark}

\begin{method}[Bienaymé--चेबीशेव का उपयोग करना]\label{met:g12:sums:chebyshev}
गारंटी करने के लिए $\P(\abs{M_n - \mu} \geq \varepsilon) \leq \alpha $, यह
इतना ही काफी है $ n \geq \frac{\sigma^2}{\alpha\,\varepsilon^2}$. बाध्य है
कच्चा (वास्तविक उतार-चढ़ाव आमतौर पर बहुत छोटा होता है) लेकिन बिल्कुल सामान्य:
इसके बारे में कुछ भी आवश्यक नहीं है \omterm{def:g12:randvar:rv}{वितरण} एक परिमित से परे \omterm{def:g12:randvar:exp}{झगड़ा}.
\end{method}

\section{अभ्यास}

\begin{exercise}[$\star $]\label{exo:g12:sums:1}
दो निष्पक्ष पासे फेंके गए; होने देना $ S $ योग हो. रैखिकता का उपयोग करना (नहीं
\omterm{def:g12:randvar:rv}{वितरण} का $ S $!), गणना करें $\E(S)$; फिर गणना करें $\V(S)$ का उपयोग करते हुए
स्वतंत्रता, यह देखते हुए कि एक भी निष्पक्ष पासा है \omterm{def:g12:randvar:exp}{झगड़ा} $\frac{35}{12}$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g12:sums:2}
होने देना $ X \sim \mathcal B(100,\ 0.5)$. अवश्यंभावी
$\P(X \geq 75)$ मार्कोव की असमानता का उपयोग करते हुए, फिर
$\P(\abs{X - 50} \geq 25)$ Bienaymé--चेबीशेव का उपयोग करना। तुलना करना।
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g12:sums:3}
एक निष्पक्ष सिक्का उछाला जाता है $ n $ समय और $ F_n $ को दर्शाता है \omterm{def:g10:stats:series}{आवृत्ति} सिरों का.
कितना बड़ा होना चाहिए $ n $ ऐसा हो कि, बिएनायमे--चेबीशेव द्वारा बाध्य,
$\P\bigl(\abs{F_n - 0.5} \geq 0.05\bigr) \leq 0.05$?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:sums:4}
एक निवेशक अपनी पूंजी को समान रूप से विभाजित करता है $ n $ \omterm{def:g12:sums:indep}{स्वतंत्र} संपत्ति,
प्रत्येक अपेक्षित रिटर्न के साथ $\mu = 5\%$ और \omterm{def:g11:stat:variance}{मानक विचलन}
$\sigma = 20\%$. की गणना करें \omterm{def:g12:randvar:exp}{अपेक्षा} और \omterm{def:g11:stat:variance}{मानक विचलन} का
पोर्टफोलियो वापसी $ M_n $, और लाने के लिए आवश्यक संपत्तियों की संख्या
\omterm{def:g11:stat:variance}{मानक विचलन} नीचे $4\%$. यह कौन सा वित्तीय \omterm{def:g11:vect:det}{सिद्ध}ांत करता है
वर्णन करें?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:sums:5}
होने देना $ X $ और $ Y $ होना \omterm{def:g12:sums:indep}{स्वतंत्र}, दोनों एक समान हैं $\{1, 2, 3\}$.
\begin{enumerate}
  \item दे दो \omterm{def:g12:randvar:rv}{वितरण} का $ S = X + Y $ और गणना करें $\E(S)$, $\V(S)$
        सीधे उससे.
  \item रैखिकता और योगात्मकता द्वारा दोनों मान पुनर्प्राप्त करें \omterm{def:g12:randvar:exp}{झगड़ा}.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:sums:6}
दिखाएँ कि की additiveity \omterm{def:g12:randvar:exp}{झगड़ा} बिना विफल हो सकता है स्वतंत्रता:
गणना करें $\V(X + Y)$ के लिए $ Y = X $, और तुलना करें $\V(X) + \V(Y)$. के लिए
कौन से चर $ X $ है $\V(2X) = 2\V(X)$?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:sums:7}
एक डाई लादे जाने की आशंका है। यह लुढ़का हुआ है $1200$ समय और शो ए
छह $260$ समय ($ f = 0.2167$ के बजाय $\frac16 \approx 0.1667$). के अंतर्गत
यह परिकल्पना कि मृत्यु उचित है, बाध्य है
$\P\bigl(\abs{F_n - \frac16} \geq 0.05\bigr)$ बायनेमे द्वारा--चेबीशेव, और
चर्चा करें कि क्या निष्पक्षता परिकल्पना प्रशंसनीय है।
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g12:sums:8}
\emph{(सिक्के के लिए एक बेहतर असमानता।)} होने देना
$ X \sim \mathcal B(n,\ p)$ और $ F_n = \frac Xn $.
\begin{enumerate}
  \item बताते हैं कि $ p(1-p) \leq \frac14$ के लिए $ p \in \intcc{0}{1}$.
  \item वितरण-मुक्त सीमा का पता लगाएं
        $\P\bigl(\abs{F_n - p} \geq \varepsilon\bigr) \leq
        \frac{1}{4n\varepsilon^2}$.
  \item कितने लोगों से मतदान कराया जाना चाहिए ताकि देखा जा सके \omterm{def:g10:stats:series}{आवृत्ति} भीतर है $3$ के साथ सही अनुपात के अंक \omterm{def:g10:proba:distribution}{संभावना} कम से कम
        $95\%$, इस बाउंड का उपयोग कर रहे हैं? (असली सर्वेक्षण तेज़ अनुमानों का उपयोग करते हैं, लेकिन
        परिमाण का क्रम सही है।)
\end{enumerate}
\end{exercise}

\section{समस्या: घर की हमेशा जीत होती है}

\begin{problem}[{सप्ताहान्त समस्या---बड़ी संख्या का नियम कैसीनो, चुनाव और बीमा की व्याख्या करता है, और ध्वस्त कर देता है रास्ते में जुआरी का भ्रम}]
\label{pb:g12:sums:1}
एक रूलेट खिलाड़ी के बाद $100$ घूमता है, लगभग उतनी ही बार, जितनी बार नहीं,
\emph{आगे}. एक मिलियन स्पिन चलाने वाला कैसीनो आगे है
निश्चितता कोई भी अदालत सवाल नहीं करेगी। वही खेल, वही छोटी धार ---
अंतर यह है $\sqrt n $, और यह इसका विषय है
अध्याय (\cref{prop:g12:sums:mean},
\cref{thm:g12:sums:chebyshev}, \cref{thm:g12:sums:lln}). यह
समस्या कैसीनो की किताबें चलाती है, चुनाव सर्वेक्षण का आकार, कीमतें
विविधीकरण --- और सबसे महंगी भ्रांति को ख़त्म करता है
जुए का इतिहास.

\medskip
\noindent\textbf{Part I --- Fluency.}
\begin{enumerate}
  \item दो \omterm{def:g12:sums:indep}{स्वतंत्र} पासा: गणना करें $ V(X + Y)$
        (\cref{thm:g12:sums:variance}).
  \item रोल $100$ \omterm{def:g12:sums:indep}{स्वतंत्र} पासा और औसत परिणाम:
        दे दो \omterm{def:g12:randvar:exp}{अपेक्षा} और यह \omterm{def:g11:stat:variance}{मानक विचलन} की
        \omterm{prop:g12:sums:mean}{नमूना माध्य} (\cref{prop:g12:sums:mean}; एक मरने के लिए,
        $\sigma = \sqrt{35/12} \approx 1.71$).
  \item एक गैरनकारात्मक \omterm{def:g12:randvar:rv}{अनियमित परिवर्तनशील वस्तु} है \omterm{def:g12:randvar:exp}{अपेक्षा} $2$.
        मार्कोव की असमानता क्या दर्शाती है?
        (\cref{thm:g12:sums:markov}) के बारे में कहो
        $\P(X \geq 10)$?
  \item अवश्यंभावी $\P\left(\abs{\bar X_{100} - 3.5} \geq
        0.5\right)$ प्रश्न 2 के पासे के औसत के लिए
        Bienaym\'ई--चेबीशेव।
  \item अपेक्षाएं हमेशा \omterm{def:g10:vectors:sum}{जोड़}ें, प्रसरण केवल के अंतर्गत
        स्वतंत्रता: आश्रित प्रदर्शित करें $ X, Y $ (संकेत देना:
        $ Y = -X $) जिसके लिए
        $ V(X + Y) \neq V(X) + V(Y)$.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The house's books.}
यूरोपीय रूलेट, शर्त $1$ यूरो लाल पर: जीत $+1$ साथ
\omterm{def:g10:proba:distribution}{संभावना} $\frac{18}{37}$, खोना $-1$ साथ \omterm{def:g10:proba:distribution}{संभावना}
$\frac{19}{37}$.
\begin{enumerate}[resume]
  \item की गणना करें \omterm{def:g12:randvar:exp}{अपेक्षा} और \omterm{def:g11:stat:variance}{मानक विचलन} एक का
        दांव का लाभ.
  \item एक जुआरी बनाता है $100$ \omterm{def:g12:sums:indep}{स्वतंत्र} दांव; होने देना $ G $ हो
        कुल लाभ. गणना करना $\E(G)$ और $\sigma(G)$, फिर
        अनुपात $\frac{\abs{\E(G)}}{\sigma(G)}$. ए क्या करता है?
        एक चौथाई का बहाव \omterm{def:g11:stat:variance}{मानक विचलन} \omterm{def:g11:stat:mean}{अर्थ} के लिए
        आज रात जुआरी के आगे रहने की \omterm{def:g10:proba:distribution}{संभावना}?
  \item कैसीनो देखता है $1\,000\,000$ दांव. की गणना करें
        \omterm{def:g12:randvar:exp}{अपेक्षा} और \omterm{def:g11:stat:variance}{मानक विचलन} इसके कुल लाभ का,
        और वही अनुपात. संख्या की व्याख्या करें $27$.
  \item चेबीशेव के साथ प्रमाणित करें: बाध्य करें \omterm{def:g10:proba:distribution}{संभावना} वह
        कैसीनो \emph{खो देता है} दस लाख से अधिक दांव पर पैसा।
  \item सट्टेबाजी प्रणाली: नुकसान के बाद दोगुना, कब छोड़ना
        आगे\,\dots\ की रैखिकता का उपयोग करना \omterm{def:g12:randvar:exp}{अपेक्षा}
        (\cref{thm:g12:sums:linearity}) के ऊपर (संभवतः)।
        यादृच्छिक) \omterm{def:g12:seq:sequence}{अनुक्रम} दांव का, कारण स्पष्ट करें \emph{प्रत्येक}
        नकारात्मक बढ़त वाले खेल में रणनीति से नकारात्मक \omterm{def:g12:randvar:exp}{अपेक्षा} की जाती है
        लाभ --- एक विजयी प्रणाली किसका उल्लंघन करेगी?
  \item बताएं कि बड़ी संख्या का नियम क्या है?
        (\cref{thm:g12:sums:lln}) के बारे में वादे \omterm{def:g10:stats:series}{आवृत्ति}
        लाल रंग का --- और यह क्या करता है \emph{नहीं} के बारे में वादा करें
        लगातार दस रेड के बाद अगला स्पिन। नाम बताएं
        भ्रांति.
  \item सबसे सूक्ष्म बिंदु: कानून काम करता है \emph{पतला करने की क्रिया},
        मुआवजा नहीं. अगर लाल भागते हैं $100$ से आगे
        \omterm{def:g12:randvar:exp}{अपेक्षा} कुछ शाम के बाद, अधिशेष कभी नहीं होता
        ``भुगतान किया गया'' --- गणना करें कि इसके बजाय क्या होता है
        \emph{अंश} $\frac{100}{n}$ द्वारा $ n = 10^6$, और
        जुआरी की भ्रांति की त्रुटि को एक वाक्य में पुनः लिखें।
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Polls.}
\begin{enumerate}[resume]
  \item से \cref{exo:g12:sums:8}: कितने लोग होंगे
        अवलोकन के लिए मतदान किया गया \omterm{def:g10:stats:series}{आवृत्ति} भीतर गिरना $3$
        सत्य के बिंदु साथ में \omterm{def:g10:proba:distribution}{संभावना} कम से कम
        $95\,\%$, वितरण-मुक्त चेबीशेव द्वारा बाध्य?
  \item वास्तविक मतदान संस्थान इसके बारे में उपयोग करते हैं $1\,100$ लोगों के लिए
        ए $\pm 3$-बिंदु मार्जिन पर $95\,\%$: उनका सूत्र है
        $ n \approx \frac{1.96^2 \times p(1-p)}
        {\varepsilon^2}$ साथ $ p(1-p) \leq \frac14$. मूल्यांकन करें
        यह, और प्रश्न 13 (क्या) के साथ अंतर स्पष्ट करें
        क्या चेबीशेव को इसके आकार के बारे में पता नहीं है
        उतार-चढ़ाव?)
  \item त्रुटि की \omterm{def:g10:proba:distribution}{संभावना} को आधा करने के लिए, कैसे करना होगा? \omterm{def:g10:proba:sample}{नमूना}
        बढ़ना? से $\pm 3$ पर इंगित करता है $1\,100$, क्या \omterm{def:g10:proba:sample}{नमूना}
        देता है $\pm 1.5$ अंक?
  \item प्रति-सहज ज्ञान युक्त क्लासिक: आवश्यक \omterm{def:g10:proba:sample}{नमूना} आकार
        जनसंख्या के आकार का कभी उपयोग नहीं किया गया --- $1\,100$ लोग
        किसी शहर या महाद्वीप के लिए पर्याप्त। जगह की ओर इशारा करें
        उस मॉडल में जहां जनसंख्या का आकार अनुपस्थित है, और
        सर्वेक्षणकर्ताओं द्वारा उपयोग की जाने वाली रसोई की उपमा दीजिए।
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Diversification.}
\begin{enumerate}[resume]
  \item का बीमाकर्ता \cref{pb:g12:randvar:1} कमाता
        $30n $ यूरो में \omterm{def:g12:randvar:exp}{अपेक्षा} पर $ n $ ए के साथ नीतियां
        दावा \omterm{def:g11:stat:variance}{मानक विचलन} के विषय का $315\sqrt n $. के लिए
        जो $ n $ क्या अपेक्षित लाभ अंततः एक से अधिक हो जाता है?
        \omterm{def:g11:stat:variance}{मानक विचलन} दावों का? क्या है
        $\sqrt n $ बीमाकर्ता के लिए कर रहे हैं?
  \item एक स्टॉक के वार्षिक रिटर्न में उतार-चढ़ाव होता रहता है
        $\sigma = 20\,\%$. पैसे को बराबर-बराबर बाँट लें $25$
        \emph{\omterm{def:g12:sums:indep}{स्वतंत्र}} ऐसे स्टॉक: की गणना करें
        पोर्टफ़ोलियो का $\sigma $. वित्त विविधीकरण को कहता है
        एकमात्र निःशुल्क दोपहर का भोजन --- दोपहर का भोजन वास्तव में क्या है?
  \item अब चलो $25$ स्टॉक पूरी तरह से सहसंबंधित हों (वे
        सभी एक साथ चलते हैं): पोर्टफोलियो क्या है $\sigma $?
        दो चरम सीमाओं की तुलना करें और बताएं कि कौन सी धारणा है
        हर विविधीकरण \omterm{def:g12:complex:argument}{तर्क} गुप्त रूप से किराए पर ---
        और क्या हुआ जब यह 2008 में पूरे सिस्टम में विफल हो गया।
  \item समापन--- द $\sqrt n $ सिम्फनी: रकम में उतार-चढ़ाव होता है
        $\sqrt n $ जबकि उनके साधन जैसे व्यवस्थित होते हैं
        $\frac{1}{\sqrt n}$; चारों के माध्यम से विषय को चलायें
        इस समस्या के उद्योग (कैसीनो, मतदान, बीमा,
        पोर्टफ़ोलियो), दो स्थायी दुर्व्यवहारियों के नाम बताएं (द
        जुआरी की भ्रांति और जालसाजी स्वतंत्रता), और दे
        आगे सूचक: \emph{आकार} उतार-चढ़ाव का
        --- घंटी --- अगले अध्याय की है निरंतर
        तारा, और इसका पूरा प्रमेय विश्वविद्यालय को ताज पहनाता है
        वॉल्यूम.
\end{enumerate}
\end{problem}
