\chapter{यादृच्छिक चर और द्विपद वितरण}\label{ch:g12:randvar}

ए \omterm{def:g12:randvar:rv}{अनियमित परिवर्तनशील वस्तु} प्रयोग के प्रत्येक परिणाम के साथ एक संख्या जोड़ता है: a
लाभ, एक गिनती, एक अवधि। इसका \omterm{def:g12:randvar:exp}{अपेक्षा} का दीर्घकालीन औसत है
यह जो मूल्य पैदा करता है, वह है \omterm{def:g12:randvar:exp}{झगड़ा} उनके प्रसार को मापता है। इसका सितारा
अध्याय है \omterm{def:g12:randvar:binomial}{द्विपद वितरण}, जो बार-बार सफलताओं को गिनता है
\omterm{def:g12:condprob:indep}{स्वतंत्र} परीक्षण. यादृच्छिक चर और यह \omterm{def:g12:randvar:binomial}{द्विपद वितरण} थे
में पहली बार मिले \cref{ch:g11:prob,ch:g11:binom}; यह अध्याय समीक्षा करता है और
के संयोजक उपकरणों के साथ, उन्हें गहरा करता है \cref{ch:g12:comb} अब उपलब्ध है।

\section{असतत यादृच्छिक चर}

\begin{definition}[यादृच्छिक चर, वितरण]\label{def:g12:randvar:rv}
ए \emph{अनियमित परिवर्तनशील वस्तु}\index{अनियमित परिवर्तनशील वस्तु} एक परिमित पर \omterm{def:g11:prob:model}{नमूना अंतरिक्ष}
$\Omega $ एक है \omterm{def:g11:func:function}{समारोह} $ X \colon \Omega \to \R $. इसका
\emph{वितरण}\index{वितरण} (या क़ानून) इसकी सम्भावना का आँकड़ा है
मूल्य $ x_1, \dots, x_k $ और संभावनाओं का
\[
p_i = \P(X = x_i), \qquad \sum_{i=1}^{k} p_i = 1 .
\]
\end{definition}

\begin{definition}[अपेक्षा, विचरण, मानक विचलन]\label{def:g12:randvar:exp}
 \emph{अपेक्षा}\index{अपेक्षा} का $ X $ है
\[
\E(X) = \sum_{i=1}^{k} p_i\, x_i ,
\]
इसका \emph{झगड़ा}\index{झगड़ा} और \emph{\omterm{def:g11:stat:variance}{मानक विचलन}} हैं
\[
\V(X) = \E\bigl((X - \E(X))^2\bigr) = \sum_{i=1}^k p_i\bigl(x_i - \E(X)\bigr)^2,
\qquad
\sigma(X) = \sqrt{\V(X)} .
\]
\end{definition}

\begin{proposition}[कोनिग - ह्यूजेन्स सूत्र]\label{prop:g12:randvar:konig}
\index{कोनिग - ह्यूजेन्स सूत्र}
$\V(X) = \E(X^2) - \E(X)^2$.
\end{proposition}

\begin{proof}
लिखना $ m = \E(X)$ और \omterm{def:g10:algebra:expand}{विस्तार} करें:
\[
\V(X) = \sum_i p_i (x_i^2 - 2m x_i + m^2)
= \E(X^2) - 2m\sum_i p_i x_i + m^2\sum_i p_i
= \E(X^2) - 2m^2 + m^2 . \qedhere
\]
\end{proof}

\begin{proposition}[एफ़िन परिवर्तन]\label{prop:g12:randvar:affine}
के लिए $ a, b \in \R $:
\[
\E(aX + b) = a\,\E(X) + b, \qquad
\V(aX + b) = a^2\,\V(X) .
\]
\end{proposition}

\begin{proof}
पहला परिभाषित योग की पुनर्\omterm{def:g12:comb:tuples}{व्यवस्था} है। दूसरे के लिए, $ aX + b $
अपने से भटक जाता है \omterm{def:g11:stat:mean}{अर्थ} द्वारा $ a(X - \E(X))$, और वर्ग को गुणा किया जाता है
$ a^2$.
\end{proof}

\begin{example}[निष्पक्ष खेल]\label{ex:g12:randvar:game}
एक खेल की लागत $2$ यूरो; एक पासा फेंका जाता है और खिलाड़ी को मूल्य प्राप्त होता है
यदि यह कम से कम है तो दिखाया गया है $5$, अन्यथा कुछ नहीं. लाभ $ G $ मान लेता है
$-2$ (\omterm{def:g10:proba:distribution}{संभावना} $\frac46$), $3$ ($\frac16$), $4$ ($\frac16$):
\[
\E(G) = \frac{-8 + 3 + 4}{6} = -\frac{1}{6} < 0 .
\]
औसतन, खिलाड़ी हार जाता है $17$ प्रति खेल सेंट: खेल प्रतिकूल है
(जैसा कि अधिकांश वास्तविक खेल होते हैं)।
\end{example}

\section{बर्नौली परीक्षण और द्विपद वितरण}

\begin{definition}[बर्नौली वितरण]\label{def:g12:randvar:bernoulli}
ए \emph{बर्नौली परीक्षण}\index{बर्नौली परीक्षण} दो के साथ एक प्रयोग है
परिणाम, \emph{सफलता} (\omterm{def:g10:proba:distribution}{संभावना} $ p $) और \emph{असफलता}
($ q = 1 - p $). सूचक $ X $ सफलता की ($ X = 1$ सफलता पर, $0$ पर
विफलता) का अनुसरण करता है \emph{Bernoulli \omterm{def:g12:randvar:rv}{वितरण}} $\mathcal B(p)$:
\[
\E(X) = p, \qquad \V(X) = p(1-p) .
\]
(वास्तव में $\E(X) = p $, $\E(X^2) = p $, और कोनिग--ह्यूजेंस देता है
$\V(X) = p - p^2$.)
\end{definition}

\begin{definition}[द्विपद वितरण]\label{def:g12:randvar:binomial}
दोहराएँ ए \omterm{def:g12:randvar:bernoulli}{बर्नौली परीक्षण} $ n $ समय \omterm{def:g12:condprob:indep}{स्वतंत्र} रूप से, और जाने दो $ X $ कुल हो
सफलताओं की संख्या. \omterm{def:g12:randvar:rv}{वितरण} का $ X $ है \emph{द्विपद
वितरण}\index{द्विपद
वितरण} $\mathcal B(n, p)$.
\end{definition}

\begin{theorem}\label{thm:g12:randvar:binomial}
अगर $ X \sim \mathcal B(n, p)$, फिर के लिए $0 \leq k \leq n $:
\[
\P(X = k) = \binom{n}{k} p^k (1-p)^{n-k},
\]
और
\[
\E(X) = np, \qquad \V(X) = np(1-p) .
\]
\end{theorem}

\begin{proof}
एक निर्दिष्ट \omterm{def:g12:seq:sequence}{अनुक्रम} परिणामों के साथ $ k $ सफलताएँ और $ n-k $ असफलताएं हैं
\omterm{def:g10:proba:distribution}{संभावना} $ p^k(1-p)^{n-k}$ द्वारा स्वतंत्रता; ऐसे की संख्या दृश्यों
रखने के तरीकों की संख्या है $ k $ के बीच सफलताएँ $ n $ परीक्षण,
\omterm{def:g11:stat:mean}{अर्थ}ात् $\binom nk $ (\cref{ch:g12:comb}). संक्षेप में कहें तो दृश्यों देता है
सूत्र --- और द्विपद प्रमेय पुष्टि करता है
$\sum_k \P(X = k) = (p + q)^n = 1$.

के लिए \omterm{def:g12:randvar:exp}{अपेक्षा}, का उपयोग करना $ k\binom nk = n\binom{n-1}{k-1}$
(\cref{exo:g12:comb:7}):
\[
\E(X) = \sum_{k=1}^{n} k\binom nk p^k q^{n-k}
= np\sum_{k=1}^{n} \binom{n-1}{k-1} p^{k-1} q^{(n-1)-(k-1)}
= np\,(p + q)^{n-1} = np .
\]
 \omterm{def:g12:randvar:exp}{झगड़ा} सूत्र इसी प्रकार सर्वसमिका से \omterm{def:g11:vect:det}{सिद्ध} होता है
$ k(k-1)\binom nk = n(n-1)\binom{n-2}{k-2}$, देना
$\E(X(X-1)) = n(n-1)p^2$, कहाँ से
$\V(X) = n(n-1)p^2 + np - (np)^2 = np(1-p)$. (एक संरचनात्मक प्रमाण---
\omterm{def:g12:randvar:exp}{झगड़ा} की राशि का \omterm{def:g12:condprob:indep}{स्वतंत्र} वेरिएबल --- के साथ आता है
\cref{thm:g12:sums:variance}.)
\end{proof}

\begin{method}[द्विपद स्थिति को पहचानना]\label{met:g12:randvar:recognize}
लिखने से पहले तीन सामग्रियों की जाँच करें $ X \sim \mathcal B(n,p)$: ए
\emph{निर्धारित अंक} $ n $ परीक्षणों का; \emph{दो परिणाम} के साथ प्रति परीक्षण
\emph{वही} सफलता \omterm{def:g10:proba:distribution}{संभावना} $ p $; \emph{स्वतंत्रता} परीक्षणों का
(\omterm{def:g10:proba:sample}{नमूना} \emph{साथ} प्रतिस्थापन, या एक बड़ी आबादी से)। फिर प्रयोग करें
\[
\P(X \geq 1) = 1 - (1-p)^n
\]
``कम से कम एक सफलता'' के लिए, और एक कैलकुलेटर या संचयी तालिकाएँ
सामान्य $\P(X \leq k)$.
\end{method}

\begin{example}\label{ex:g12:randvar:atleastone}
कम से कम एक पाने के लिए एक पासे को कितनी बार घुमाना होगा? $99\%$ का मौका
छक्का लगाना? साथ $ X \sim \mathcal B\left(n, \frac16\right)$:
$\P(X \geq 1) = 1 - \left(\frac56\right)^n \geq 0.99$ मतलब
$\left(\frac56\right)^n \leq 0.01$, \emph{यानी}
$ n \geq \frac{\ln 0.01}{\ln(5/6)} \approx 25.3$: से $ n = 26$ चलता रहता है.
\end{example}

\section{अभ्यास}

\begin{exercise}[$\star $]\label{exo:g12:randvar:1}
ए \omterm{def:g12:randvar:rv}{अनियमित परिवर्तनशील वस्तु} $ X $ मान लेता है $-1, 0, 2, 5$ संभावनाओं के साथ
$0.3, 0.2, 0.4, 0.1$. गणना करना $\E(X)$, $\V(X)$ और $\sigma(X)$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g12:randvar:2}
एक बहुविकल्पीय परीक्षा है $10$ प्रश्न, प्रत्येक के साथ $4$ विकल्प, में से एक
जो सही है. एक विद्यार्थी यादृच्छिक रूप से, \omterm{def:g12:condprob:indep}{स्वतंत्र} रूप से समान रूप से उत्तर देता है।
होने देना $ X $ सही उत्तरों की संख्या हो.
\begin{enumerate}
  \item दे दो \omterm{def:g12:randvar:rv}{वितरण} का $ X $, $\E(X)$ और $\sigma(X)$.
  \item गणना करना $\P(X = 0)$, $\P(X = 5)$ और $\P(X \geq 1)$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g12:randvar:3}
एक बीमा कंपनी बीमा करती है $ n = 400$ ग्राहक; प्रत्येक इस दौरान दावा दायर करता है
वर्ष के साथ \omterm{def:g10:proba:distribution}{संभावना} $ p = 0.05$, \omterm{def:g12:condprob:indep}{स्वतंत्र} रूप से। होने देना $ X $ संख्या हो
दावों का. की पहचान करें \omterm{def:g12:randvar:rv}{वितरण} का $ X $ और इसकी गणना करें \omterm{def:g12:randvar:exp}{अपेक्षा} और
\omterm{def:g11:stat:variance}{मानक विचलन}.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:randvar:4}
के खेल में \cref{ex:g12:randvar:game}, आयोजक निष्पक्ष खेल चाहता है
($\E(G) = 0$) प्रवेश मूल्य को बदलकर $ c $. खोजो $ c $. की गणना करें
\omterm{def:g12:randvar:exp}{झगड़ा} इस उचित संस्करण के लाभ का; ``उचित'' के समान ही है
``जोखिम रहित''?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:randvar:5}
एक बास्केटबॉल खिलाड़ी फ्री थ्रो पर स्कोर करता है \omterm{def:g10:proba:distribution}{संभावना} $0.7$. वह गोली मारती है
$8$ समय (\omterm{def:g12:condprob:indep}{स्वतंत्र} शॉट्स). की गणना करें \omterm{def:g10:proba:distribution}{संभावना} वह स्कोर करती है:
बिलकुल $6$; कम से कम $6$; कम से कम एक बार. सबसे संभावित संख्या कौन सी है
अंकों का?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:randvar:6}
एक एयरलाइन को पता है कि बुक किया गया प्रत्येक यात्री उसके साथ आता है \omterm{def:g10:proba:distribution}{संभावना}
$0.9$, \omterm{def:g12:condprob:indep}{स्वतंत्र} रूप से। एक उड़ान है $100$ सीटें और एयरलाइन बेचती है $104$
टिकट. एक्सप्रेस, का उपयोग कर \omterm{def:g12:randvar:binomial}{द्विपद वितरण}, द \omterm{def:g10:proba:distribution}{संभावना} वह और अधिक
यात्रियों को सीटों की तुलना में दिखाई देता है, और इसे संख्यात्मक रूप से एक का उपयोग करके बाध्य किया जाता है
कैलकुलेटर (सटीक अभिव्यक्ति दें)।
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:randvar:7}
होने देना $ X \sim \mathcal B(n, p)$. बताते हैं कि
\[
\frac{\P(X = k+1)}{\P(X = k)} = \frac{n-k}{k+1}\cdot\frac{p}{1-p},
\]
और यह निष्कर्ष निकालें कि \omterm{def:g12:randvar:rv}{वितरण} तक बढ़ जाता है
$ k^* = \floor{(n+1)p}$ और बाद में घट जाती है (द \emph{तरीका} का
द्विपद)।
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g12:randvar:8}
\emph{(सेंट पीटर्सबर्ग, नामांकित।)} एक निष्पक्ष सिक्के को चित तक उछाला जाता है
प्रकट होता है, लेकिन अधिक से अधिक $10$ बार. होने देना $ N $ उपयोग किए गए टॉस की संख्या हो, और
खिलाड़ी को प्राप्त होता है $2^N $ यूरो अगर सिर दिखाई दिए, $0$ अन्यथा।
\begin{enumerate}
  \item दे दो \omterm{def:g12:randvar:rv}{वितरण} का $ N $ जीतने वाले परिणामों तक ही सीमित:
        $\P(\text{heads first at toss } k) = 2^{-k}$ के लिए
        $1 \leq k \leq 10$, और कुल की जाँच करें \omterm{def:g10:proba:distribution}{संभावना} जीतने का.
  \item अपेक्षित भुगतान की गणना करें। बिना टोपी के क्या बनेगा
        का $10$ उछालता है?
\end{enumerate}
\end{exercise}

\section{समस्या: अत्यधिक बुक की गई उड़ान}

\begin{problem}[{सप्ताहान्त समस्या---एयरलाइंस अधिक सीटें बेचती हैं उनके पास है, फ़ैक्टरियाँ ऐसे लॉट स्वीकार करती हैं जिनका उन्होंने बमुश्किल निरीक्षण किया होता है, और द्विपद वितरण रेफरी दोनों}]
\label{pb:g12:randvar:1}
के साथ एक एयरलाइन $100$ सीटें ख़ुशी से बिकती हैं $105$ टिकट: के बारे में
$10\,\%$ बहुत सारे यात्री कभी नहीं आते, और खाली सीटें कमाते हैं
कुछ नहीं. टकराए हुए यात्रियों के सामने एयरलाइन कितनी दूर तक धक्का दे सकती है
मुनाफा खाओ? \omterm{def:g12:randvar:binomial}{द्विपद वितरण}
(\cref{thm:g12:randvar:binomial}) दशमलव के उत्तर --- और
वही मशीनरी फ़ैक्टरी लॉट, कीमतों की लॉटरी आदि का निरीक्षण करती है
बीमाकर्ताओं को विलायक रखता है। पुनरावृत्ति के अंतर्गत निर्णय: यह है
द्विपद का दिन का काम।

\medskip
\noindent\textbf{Part I --- Fluency.}
\begin{enumerate}
  \item पूरा दो \omterm{def:g12:randvar:rv}{वितरण} संख्या की तालिका $ X $ का
        एक पासे के तीन रोल में छक्के।
  \item गणना करना $\E(X)$ और $ V(X)$
        (\cref{def:g12:randvar:exp}).
  \item एक स्टॉल शुल्क लेता है $1$ तीन रोल और भुगतान के लिए यूरो $2$
        प्रति छह यूरो प्राप्त हुए। अपेक्षित लाभ की गणना करें: उचित
        खेल?
  \item चेकलिस्ट अभ्यास (\cref{met:g12:randvar:recognize}):
        में दिलों की संख्या है $5$ कार्ड निकाले गए
        \emph{बिना} प्रतिस्थापन द्विपद? प्रतिस्थापन के साथ?
        औचित्य।
  \item के लिए $ X \sim \mathcal B(20,\ 0.3)$: देना $\E(X)$ और
        $\sigma(X)$, तब $\E(S)$ और $\sigma(S)$ स्कोर के लिए
        $ S = 5X - 10$ (\cref{prop:g12:randvar:affine}).
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The overbooked flight.}
सीटें: $100$. टिकट बिके: $105$. प्रत्येक टिकट धारक दिखाई देता है
के साथ \omterm{def:g10:proba:distribution}{संभावना} $0.9$, \omterm{def:g12:condprob:indep}{स्वतंत्र} रूप से; होने देना $ X $ वह संख्या हो जो
दिखाओ.
\begin{enumerate}[resume]
  \item \omterm{def:g11:scal:dot}{आदर्श}: औचित्य \omterm{def:g11:vect:det}{सिद्ध} करें $ X \sim \mathcal B(105,\ 0.9)$ के साथ
        चेकलिस्ट --- और मॉडल के सबसे कमजोर बिंदु को स्वीकार करें
        (वास्तव में कोई शो नहीं हैं \omterm{def:g12:condprob:indep}{स्वतंत्र}? सोचो समूह और
        तूफ़ान)।
  \item गणना करना $\E(X)$ और $\sigma(X)$.
  \item टक्कर तब होती है जब $ X \geq 101$: लिखना
        $\P(X \geq 101)$ पाँच द्विपदों के स्पष्ट योग के रूप में
        शर्तें.
  \item योग का मूल्यांकन करें (कैलकुलेटर): उड़ानों का कौन सा अंश
        कम से कम एक टकराया हुआ यात्री देखता है?
  \item पैसा: पाँच अतिरिक्त टिकट लाते हैं
        $5 \times 200 = 1\,000$ यूरो; प्रत्येक टकराया हुआ यात्री
        लागत $800$ मुआवज़े में यूरो. अपेक्षित की गणना करें
        टकराए यात्रियों की संख्या,
        $\sum_{k=101}^{105} (k - 100)\,\P(X = k)$, अपेक्षित
        मुआवजा, और पॉलिसी पर फैसला।
  \item डिज़ाइन: नियामक सहन करता है
        $\P(\text{bumping}) < 5\,\%$. परीक्षण करके
        $ n = 105, 106, 107, \dots $ टिकट, सबसे बड़ा खोजें
        अनुमति $ n $.
  \item घंटी शॉर्टकट: के लिए $ n = 105$, गणना करें \omterm{pb:g11:stat:1}{z के स्कोर}
        उछलती दहलीज़ से,
        $ z = \frac{100.5 - \E(X)}{\sigma(X)}$, और तुलना करें
        क्रूड ``के बारे में $2\sigma $, तो मोटे तौर पर $2$--$3\,\%$ में
        आपके सटीक उत्तर के साथ ऊपरी पूँछ'' अनुमान। (द
        इस शॉर्टकट के पीछे का वक्र अगला अध्याय है'
        सितारा.)
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The factory gate.}
बहुत से भाग स्वीकार किये जाते हैं यदि a \omterm{def:g10:proba:sample}{नमूना} का $20$ \omterm{def:g10:functions:extrema}{अधिकतम} शामिल है
एक दोषपूर्ण भाग.
\begin{enumerate}[resume]
  \item यदि सही दोष दर है $2\,\%$ (एक ईमानदार समूह),
        की गणना करें \omterm{def:g10:proba:distribution}{संभावना} स्वीकृति का.
  \item यदि दर है $10\,\%$ (एक ख़राब लॉट), इसकी दोबारा गणना करें।
  \item प्रक्रिया के दो जोखिमों के नाम बताइए (ईमानदार लॉट)।
        अस्वीकृत; खराब लोगों को स्वीकार किया गया), उनके मूल्यों को पढ़ें
        प्रश्न 13-14, और बताएं कि एक \omterm{def:g12:comb:permutation}{परिवर्तन} से क्या सुधार होता है
        दोनों एक साथ --- किस कीमत पर।
  \item सुधार के पीछे: दिखाएँ कि देखा गया दोष
        \emph{\omterm{def:g10:stats:series}{आवृत्ति}} $\frac Xn $ में एक \omterm{def:g10:proba:sample}{नमूना} आकार का $ n $ है
        \omterm{def:g11:stat:variance}{मानक विचलन} $\sqrt{\frac{p(1-p)}{n}}$, और
        निष्कर्ष निकालें कि परिशुद्धता कैसे मापी जाती है \omterm{def:g10:proba:sample}{नमूना} आकार (एक पुराना
        दोस्त: द $\frac{1}{\sqrt n}$ ग्रेड 10 का)।
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Games, raffles, reserves.}
\begin{enumerate}[resume]
  \item छाया हुआ सेंट पीटर्सबर्ग खेल
        \cref{exo:g12:randvar:8} भुगतान की उम्मीद है $10$
        यूरो. एक पैरा\omterm{def:g12:matrix:graph}{ग्राफ} में अनकैप्ड के साथ तुलना करें
        \omterm{thm:g12:limcont:ivt}{विरोधाभास} सामने आया \cref{pb:g11:prob:1}: वास्तव में क्या करता है
        कैप वश में, और अब कौन सी कीमत उचित होगी?
  \item एक चैरिटी रैफ़ल बिकता है $200$ पर टिकट $2$ यूरो;
        पुरस्कार: एक $100$-यूरो और दो $50$-यूरो टोकरियाँ।
        टिकट के अपेक्षित लाभ और रैफ़ल की गणना करें
        मार्जिन; यूरोपीय रूलेट के हाउस मार्जिन से तुलना करें
        का $\frac{1}{37} \approx 2.7\,\%$. रैफल क्यों करता है
        इससे दूर हो जाओ?
  \item एक बीमाकर्ता रखता है $1\,000$ \omterm{def:g12:condprob:indep}{स्वतंत्र} नीतियां: दावा
        \omterm{def:g10:proba:distribution}{संभावना} $0.01$ प्रत्येक, दावे का आकार $10\,000$ यूरो,
        प्रीमियम $130$ यूरो. अपेक्षित वार्षिक गणना करें
        मुनाफ़ा \emph{और}  \omterm{def:g11:stat:variance}{मानक विचलन} कुल का
        दावा. दो संख्याओं की तुलना करें: क्या करता है
        तुलना वास्तविक बीमाकर्ताओं को धारण करने के लिए बाध्य करती है?
  \item समापन --- निर्णय रेफरी के रूप में द्विपद:
        मान्यता चेकलिस्ट; द कम्पास $\E \pm \sigma$;
        जोखिम की कीमत के रूप में पूंछ संभावनाएँ; और दो
        स्थायी चेतावनी (स्वतंत्रता एक मॉडलिंग दावा है, और
        दुर्लभ पूंछ, मतलब नहीं, बर्बादी का कारण)। प्रत्येक एक वाक्य,
        सूचक के साथ: बड़ी संख्या का नियम और घंटी
        वक्र, अगले अध्याय, रेफरी की नियम पुस्तिका को पूरा करें।
\end{enumerate}
\end{problem}
