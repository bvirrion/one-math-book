\chapter{दृश्यों}\label{ch:g12:seq}

ए \omterm{def:g12:seq:sequence}{अनुक्रम} की एक सूची है \omterm{def:g10:numbers:sets}{वास्तविक संख्या} द्वारा \omterm{def:g12:seq:sequence}{अनुक्रम}ित \omterm{def:g10:numbers:sets}{प्राकृतिक संख्या}. दृश्यों
मॉडल असतत विकास --- वर्ष दर वर्ष गणना की गई जनसंख्या, का संतुलन
बैंक खाता, क्रमिक अनुमान एक संख्या की --- और उनकी सीमाएँ हैं
अनंत के साथ पहली गंभीर मुठभेड़. यह अध्याय स्थापित करता है
शब्दावली, प्रेरण का \omterm{def:g11:vect:det}{सिद्ध}ांत और मौलिक \omterm{def:g12:seq:limit}{अभिसरण}
प्रमेय

\section{प्रेरण द्वारा तर्क}

\begin{theorem}[प्रेरण का सिद्धांत]\label{thm:g12:seq:induction}
होने देना $P(n)$ एक के आधार पर एक बयान हो पूर्णांक $ n $, और चलो $ n_0 \in \N $.
अगर
\begin{enumerate}
  \item \emph{(बेस केस)} $ P(n_0)$ सच है, और
  \item \emph{(आगमनात्मक कदम)} हरएक के लिए $ n \geq n_0$, $ P(n)$ तात्पर्य $ P(n+1)$,
\end{enumerate}
तब $ P(n)$ हर किसी के लिए सच है $ n \geq n_0$.
\end{theorem}

\begin{proof}
मान लीजिए, विरोधाभास के लिए, सेट $ A $ का \omterm{def:g10:numbers:sets}{पूर्णांकों} $ n \geq n_0$ के लिए
जो $ P(n)$ असत्य है शून्य है. तब $ A $ सबसे छोटा तत्व है
$ m $.\footnote{Every nonempty subset of $\N $ has a smallest element; this
property of $\N $ is taken as an axiom.} तब से $ P(n_0)$ क्या सच है, $ m > n_0$, इसलिए
$ m - 1 \geq n_0$ और $ m-1 \notin A $, यानी\ $ P(m-1)$ क्या सच है। आगमनात्मक
चरण लागू किया गया $ n = m-1$ फिर उसे दिखाता है $ P(m)$ सत्य है, विरोधाभासी है
$ m \in A $.
\end{proof}

\begin{example}\label{ex:g12:seq:bernoulli}
आइए साबित करें \emph{बर्नौली की असमानता}: हर वास्तविक के लिए $ a > 0$ और हर
$ n \in \N $,
\[
(1+a)^n \geq 1 + na.
\]
\emph{बेस केस।} के लिए $ n = 0$, दोनों पक्ष बराबर $1$.
\emph{आगमनात्मक कदम.} कल्पना करना $(1+a)^n \geq 1+na $ कुछ के लिए $ n \in \N $. तब से
$1 + a > 0$, दोनों पक्षों को गुणा करने पर $1+a $ असमानता को बरकरार रखता है:
\[
(1+a)^{n+1} \geq (1+na)(1+a) = 1 + (n+1)a + na^2 \geq 1 + (n+1)a .
\]
प्रेरण द्वारा, असमानता सभी के लिए बनी रहती है $ n \in \N $.
\end{example}

\begin{method}[एक प्रेरण प्रमाण लिखना]\label{met:g12:seq:induction}
हमेशा बयान दें $ P(n)$ शुरू करने से पहले स्पष्ट. एक पूर्ण प्रमाण
इसके तीन दृश्यमान भाग हैं: आधार मामला, आगमनात्मक चरण (``मान लें $ P(n)$;
हम साबित करते हैं $ P(n+1)$''), और निष्कर्ष प्रेरण के \omterm{def:g11:vect:det}{सिद्ध}ांत का आह्वान करता है।
सबसे आम त्रुटि है बिना इसका उपयोग किए आगमनात्मक चरण को \omterm{def:g11:vect:det}{सिद्ध} करना
परिकल्पना $ P(n)$: यदि ऐसा होता है, तो या तो प्रमाण ग़लत है या प्रेरण था
जरूरत नहीं.
\end{method}

\section{अनुक्रमों की शब्दावली}

\begin{definition}[अनुक्रम]\label{def:g12:seq:sequence}
ए \emph{अनुक्रम}\index{अनुक्रम} एक है \omterm{def:g11:func:function}{समारोह} $ u \colon \N \to \R $ (या से
$\{n \in \N : n \geq n_0\}$ को $\R $). \omterm{def:g10:functions:function}{छवि} का $ n $ लिखा है $ u_n $, और
अनुक्रम ही $(u_n)_{n\in\N}$ या केवल $(u_n)$.
\end{definition}

ए \omterm{def:g12:seq:sequence}{अनुक्रम} परिभाषित किया जा सकता है \emph{स्पष्ट रूप से}, एक सूत्र द्वारा $ u_n = f(n)$, या द्वारा
\emph{पुनरावृत्ति}, इसके पहले पद और एक संबंध से $ u_{n+1} = f(u_n)$.

\begin{definition}[दिष्टता]\label{def:g12:seq:monotonic}
ए \omterm{def:g12:seq:sequence}{अनुक्रम} $(u_n)$ है \emph{की बढ़ती}\index{की बढ़ती} अगर
$ u_{n+1} \geq u_n $ सभी के लिए $ n $, \emph{\omterm{def:g10:functions:variations}{घटते}} अगर $ u_{n+1} \leq u_n $ सभी के लिए $ n $, और \emph{एकरस} यदि यह बढ़ रहा है या \omterm{def:g10:functions:variations}{घटते}. यह है
\emph{कठोरता से} बढ़ रहा है (सम्मान)\ \omterm{def:g10:functions:variations}{घटते}) जब असमानताएं हों
सख्त.
\end{definition}

\begin{method}[किसी अनुक्रम की एकरसता का अध्ययन करना]\label{met:g12:seq:monotonicity}
तीन मानक तकनीकें:
\begin{enumerate}
  \item के संकेत का अध्ययन करें $ u_{n+1} - u_n $;
  \item यदि सभी पद सकारात्मक हैं, तो तुलना करें $\dfrac{u_{n+1}}{u_n}$ को $1$;
  \item अगर $ u_n = f(n)$ साथ $ f $ पर परिभाषित $\intco{0}{+\infty}$, का उपयोग करें
        की विविधताएँ $ f $.
\end{enumerate}
\end{method}

\begin{definition}[बंधा हुआ क्रम]\label{def:g12:seq:bounded}
ए \omterm{def:g12:seq:sequence}{अनुक्रम} $(u_n)$ है \emph{ऊपर से घिरा हुआ}\index{ऊपर से घिरा हुआ} अगर वहाँ
मौजूद है $ M \in \R $ साथ $ u_n \leq M $ सभी के लिए $ n $; \emph{नीचे बंधा हुआ} अगर वहाँ
मौजूद है $ m \in \R $ साथ $ u_n \geq m $ सभी के लिए $ n $; और \emph{घिरे} यदि दोनों
पकड़ो.
\end{definition}

\subsection{अंकगणित और ज्यामितीय अनुक्रम}

\begin{definition}[अंकगणित और ज्यामितीय अनुक्रम]\label{def:g12:seq:arith-geom}
ए \omterm{def:g12:seq:sequence}{अनुक्रम} $(u_n)$ है \emph{अंकगणित}\index{अंकगणित} साथ \omterm{def:g11:seq:arithmetic}{सामान्य
अंतर} $ r $ अगर $ u_{n+1} = u_n + r $ सभी के लिए $ n $, और
\emph{ज्यामितिक}\index{ज्यामितिक} साथ \omterm{def:g11:seq:geometric}{सामान्य अनुपात} $ q $ अगर
$ u_{n+1} = q\,u_n $ सभी के लिए $ n $.
\end{definition}

\begin{proposition}[स्पष्ट रूप और योग]\label{prop:g12:seq:arith-geom}
होने देना $ n \in \N $.
\begin{enumerate}
  \item अगर $(u_n)$ है \omterm{def:g12:seq:arith-geom}{अंकगणित} साथ \omterm{def:g11:seq:arithmetic}{सामान्य अंतर} $ r $, तब
        $ u_n = u_0 + nr $ और
        \[
        u_0 + u_1 + \dots + u_n = (n+1)\,\frac{u_0 + u_n}{2}.
        \]
  \item अगर $(u_n)$ है \omterm{def:g12:seq:arith-geom}{ज्यामितिक} साथ \omterm{def:g11:seq:geometric}{सामान्य अनुपात} $ q \neq 1$, तब
        $ u_n = u_0\, q^n $ और
        \[
        u_0 + u_1 + \dots + u_n = u_0\,\frac{1 - q^{n+1}}{1 - q}.
        \]
\end{enumerate}
\end{proposition}

\begin{proof}
स्पष्ट रूपों का अनुसरण तत्काल प्रेरणों द्वारा किया जाता है। के लिए \omterm{def:g12:seq:arith-geom}{अंकगणित} योग,
लिखो $ S = u_0 + \dots + u_n $ और लिखे गए समान योग को उल्टे क्रम में \omterm{def:g10:vectors:sum}{जोड़}ें:
प्रत्येक $ n+1$ कॉलम का योग बराबर है $ u_0 + u_n $, इसलिए $2S = (n+1)(u_0+u_n)$.
के लिए \omterm{def:g12:seq:arith-geom}{ज्यामितिक} योग, गणना $ S - qS $: को छोड़कर सभी पद \omterm{def:g10:vectors:sum}{जोड़}े में रद्द हो जाते हैं
पहला और आखिरी, इसलिए $(1-q)S = u_0(1 - q^{n+1})$.
\end{proof}

\section{अनुक्रम की सीमा}

\begin{definition}[अभिसारी क्रम]\label{def:g12:seq:limit}
ए \omterm{def:g12:seq:sequence}{अनुक्रम} $(u_n)$ \emph{अभिसरण}\index{अभिसरण} तक असली
संख्या $\ell $ अगर हर खुला \omterm{def:g10:numbers:interval}{अंतराल} युक्त $\ell $ इसमें सभी शर्तें शामिल हैं
$ u_n $ कुछ सूचकांक से. फिर हम लिखते हैं $\lim\limits_{n\to+\infty} u_n = \ell $.

समान रूप से: प्रत्येक के लिए $\varepsilon > 0$, वहाँ मौजूद है $ N \in \N $ ऐसा कि
सबके लिए $ n \geq N $, $\abs{u_n - \ell} \leq \varepsilon $.
\end{definition}

\begin{definition}[अनंत की ओर विचलन]\label{def:g12:seq:infinity}
 \omterm{def:g12:seq:sequence}{अनुक्रम} $(u_n)$ \emph{आदत है $+\infty $} यदि प्रत्येक के लिए $ A \in \R $, वहाँ
मौजूद है $ N \in \N $ ऐसा है कि $ u_n \geq A $ सभी के लिए $ n \geq N $. हम लिखते हैं
$\lim\limits_{n\to+\infty} u_n = +\infty $; की परिभाषा
$\lim u_n = -\infty $ अनुरूप है. ए \omterm{def:g12:seq:sequence}{अनुक्रम} ऐसा कहा जाता है कि \omterm{def:g12:seq:limit}{अभिसरण} नहीं होता है
को \emph{हट जाना}.
\end{definition}

\begin{remark}
ए \omterm{def:g12:seq:sequence}{अनुक्रम} बिना ध्यान दिए अलग हो सकते हैं $\pm\infty $: द \omterm{def:g12:seq:sequence}{अनुक्रम}
$ u_n = (-1)^n $ केवल मान लेता है $1$ और $-1$ और इसकी कोई सीमा नहीं है.
\end{remark}

\begin{proposition}[सीमा की विशिष्टता]\label{prop:g12:seq:uniqueness}
अगर $(u_n)$ \omterm{def:g12:seq:limit}{अभिसरण}, इसकी सीमा अद्वितीय है.
\end{proposition}

\begin{proof}
कल्पना करना $ u_n \to \ell $ और $ u_n \to \ell'$ साथ $\ell \neq \ell'$, कहना
$\ell < \ell'$. तय करना $\varepsilon = \frac{\ell' - \ell}{3} > 0$. कुछ से
सूचकांक चालू, $\abs{u_n - \ell} \leq \varepsilon $ और
$\abs{u_n - \ell'} \leq \varepsilon $, इस तरह
\[
\ell' - \ell \leq \abs{\ell' - u_n} + \abs{u_n - \ell}
\leq 2\varepsilon = \tfrac{2}{3}(\ell' - \ell) < \ell' - \ell,
\]
एक विरोधाभास.
\end{proof}

\begin{proposition}[सीमा पर संचालन]\label{prop:g12:seq:operations}
होने देना $(u_n)$ और $(v_n)$ होना दृश्यों सीमा के साथ $\ell $ और $\ell'$ (परिमित या
अनंत)। फिर, जब भी दाहिनी ओर एक अनिश्चित रूप नहीं है,
\[
\lim (u_n + v_n) = \ell + \ell', \qquad
\lim (u_n v_n) = \ell\,\ell', \qquad
\lim \frac{u_n}{v_n} = \frac{\ell}{\ell'}.
\]
अनिश्चित रूप हैं
$(+\infty) + (-\infty)$, $0 \times \infty $, $\frac{\infty}{\infty}$ और
$\frac{0}{0}$.
\end{proposition}

\begin{proof}
हम परिमित सीमाओं के लिए योग नियम \omterm{def:g11:vect:det}{सिद्ध} करते हैं; अन्य मामले भी ऐसे ही हैं और छोड़ दिए गए हैं
व्यायाम के रूप में. होने देना $\varepsilon > 0$. वहां है $ N_1, N_2$ ऐसा है कि
$\abs{u_n - \ell} \leq \varepsilon/2$ के लिए $ n \geq N_1$ और
$\abs{v_n - \ell'} \leq \varepsilon/2$ के लिए $ n \geq N_2$. के लिए
$ n \geq \max(N_1, N_2)$, त्रिभुज असमानता देता है
\[
\abs{(u_n + v_n) - (\ell + \ell')}
\leq \abs{u_n - \ell} + \abs{v_n - \ell'} \leq \varepsilon. \qedhere
\]
\end{proof}

\begin{method}[अनिश्चित रूप को उठाना]\label{met:g12:seq:indeterminate}
एक अनिश्चित रूप का सामना करते हुए, प्रमुख शब्द का गुणनखंड करें। उदाहरण के लिए
\[
n^2 - n = n^2\left(1 - \tfrac{1}{n}\right) \xrightarrow[n\to+\infty]{} +\infty,
\qquad
\frac{2n^2+1}{n^2 - n} = \frac{2 + 1/n^2}{1 - 1/n}
\xrightarrow[n\to+\infty]{} 2 .
\]
\end{method}

\section{अभिसरण प्रमेय}

\begin{theorem}[तुलना और निचोड़ प्रमेय]\label{thm:g12:seq:squeeze}
होने देना $(u_n)$, $(v_n)$, $(w_n)$ होना दृश्यों.
\begin{enumerate}
  \item अगर $ u_n \leq v_n $ कुछ सूचकांक से और $ u_n \to +\infty $, तब
        $ v_n \to +\infty $.
  \item \emph{(निचोड़ प्रमेय)}\index{(निचोड़ प्रमेय)} अगर
        $ u_n \leq v_n \leq w_n $ कुछ सूचकांक से और $(u_n)$ और $(w_n)$
        दोनों एक ही सीमा पर एकत्रित होते हैं $\ell $, तब $(v_n)$ \omterm{def:g12:seq:limit}{अभिसरण}
        को~$\ell $.
\end{enumerate}
\end{theorem}

\begin{proof}
\emph{1.} होने देना $ A \in \R $. तब से $ u_n \to +\infty $, वहाँ है $ N $ साथ
$ u_n \geq A $ के लिए $ n \geq N $; एनलार्जिंग $ N $ यदि ज़रूरत हो तो, $ v_n \geq u_n \geq A $
के लिए $ n \geq N $.

\emph{2.} होने देना $\varepsilon > 0$. कुछ सूचकांक से, दोनों
$\ell - \varepsilon \leq u_n $ और $ w_n \leq \ell + \varepsilon $ इसलिए, पकड़ो
$\ell - \varepsilon \leq u_n \leq v_n \leq w_n \leq \ell + \varepsilon $, वह है $\abs{v_n - \ell} \leq \varepsilon $.
\end{proof}

\begin{example}
सभी के लिए $ n \geq 1$, $-\frac{1}{n} \leq \frac{(-1)^n}{n} \leq \frac{1}{n}$, और
दोनों सीमाएँ प्रवृत्त होती हैं $0$; इस तरह $\frac{(-1)^n}{n} \to 0$.
\end{example}

\begin{theorem}[मोनोटोन अभिसरण प्रमेय]\label{thm:g12:seq:monotone}
एक \omterm{def:g12:seq:monotonic}{की बढ़ती} \omterm{def:g12:seq:sequence}{अनुक्रम} वह है \omterm{def:g12:seq:bounded}{ऊपर से घिरा हुआ} \omterm{def:g12:seq:limit}{अभिसरण}. ए \omterm{def:g10:functions:variations}{घटते} \omterm{def:g12:seq:sequence}{अनुक्रम}
वह है नीचे बंधा हुआ \omterm{def:g12:seq:limit}{अभिसरण}. एक \omterm{def:g12:seq:monotonic}{की बढ़ती} \omterm{def:g12:seq:sequence}{अनुक्रम} जो कि नहीं घिरा हुआ
ऊपर आदत है $+\infty $.
\end{theorem}

\begin{proof}[आंशिक प्रमाण]
हम तीसरा कथन \omterm{def:g11:vect:det}{सिद्ध} करते हैं। होने देना $(u_n)$ होना \omterm{def:g12:seq:monotonic}{की बढ़ती} और नहीं घिरा हुआ
ऊपर, और चलो $ A \in \R $. तब से $ A $ कोई ऊपरी सीमा नहीं है, अस्तित्व है $ N $
साथ $ u_N \geq A $; एकरसता से, $ u_n \geq u_N \geq A $ सभी के लिए $ n \geq N $.
इस तरह $ u_n \to +\infty $.

दो \omterm{def:g12:seq:limit}{अभिसरण} कथन \omterm{def:g10:functions:extrema}{न्यूनतम} ऊपरी सीमा वाली संपत्ति पर निर्भर करते हैं
$\R $; वे हैं \emph{इस स्तर पर स्वीकार किया गया} (और के पहले वर्ष में साबित हुआ
विश्वविद्यालय).
\end{proof}

\begin{remark}
प्रमेय इसकी गारंटी देता है \emph{अस्तित्व} की सीमा लेकिन नहीं देता
इसका मूल्य. एक \omterm{def:g12:seq:monotonic}{की बढ़ती} \omterm{def:g12:seq:sequence}{अनुक्रम} \omterm{def:g12:seq:bounded}{ऊपर से घिरा हुआ} द्वारा $ M $ \omterm{def:g12:seq:limit}{अभिसरण} कुछ करने के लिए
$\ell \leq M $, जरूरी नहीं है $ M $.
\end{remark}

\begin{theorem}[ज्यामितीय अनुक्रमों की सीमा]\label{thm:g12:seq:geometric}
होने देना $ q \in \R $.
\begin{enumerate}
  \item अगर $ q > 1$, तब $ q^n \to +\infty $.
  \item अगर $ q = 1$, तब $ q^n \to 1$.
  \item अगर $\abs{q} < 1$, तब $ q^n \to 0$.
  \item अगर $ q \leq -1$, तब $(q^n)$ विचलन करता है और इसकी कोई सीमा नहीं है।
\end{enumerate}
\end{theorem}

\begin{proof}
\emph{1.} लिखना $ q = 1 + a $ साथ $ a > 0$. बर्नौली की असमानता
(\cref{ex:g12:seq:bernoulli}) देता है $ q^n \geq 1 + na \to +\infty $, और हम
तुलना करके निष्कर्ष निकालें (\cref{thm:g12:seq:squeeze}).

\emph{2.} तुरंत।

\emph{3.} अगर $ q = 0$ दावा स्पष्ट है. अन्यथा $\abs{q} < 1$ देता है
$1/\abs{q} > 1$, इसलिए $(1/\abs{q})^n \to +\infty $ बिंदु 1 से, इसलिए
$\abs{q}^n \to 0$, और $-\abs{q}^n \leq q^n \leq \abs{q}^n $ हमें इसकी अनुमति देता है
निचोड़ प्रमेय द्वारा निष्कर्ष निकालें।

\emph{4.} के लिए $ q \leq -1$, $(q^{2n})$ मान लेता है $\geq 1$ जबकि
$(q^{2n+1})$ मान लेता है $\leq -1$: कोई भी एक सीमा दोनों को आकर्षित नहीं कर सकती
परिणाम.
\end{proof}

\begin{method}[आवर्ती क्रम $ u_{n+1} = f(u_n)$]\label{met:g12:seq:recurrent}
अध्ययन करने के लिए ए \omterm{def:g12:seq:sequence}{अनुक्रम} द्वारा परिभाषित $ u_{n+1} = f(u_n)$:
\begin{enumerate}
  \item प्रेरण द्वारा \omterm{def:g11:vect:det}{सिद्ध} करें कि $(u_n)$ एक में रहता है \omterm{def:g10:numbers:interval}{अंतराल} $ I $ जिस पर $ f $
        अच्छा व्यवहार किया जाता है (और, अक्सर, वह $(u_n)$ है एकरस);
  \item मोनोटोन \omterm{def:g12:seq:limit}{अभिसरण} प्रमेय से \omterm{def:g12:seq:limit}{अभिसरण} निकालना;
  \item रिश्ते में हद से गुजर जाओ $ u_{n+1} = f(u_n)$: अगर $ f $ है
        निरंतर और $ u_n \to \ell \in I $, तब $\ell $ संतुष्ट
        $ f(\ell) = \ell $ (देखना \cref{ch:g12:limcont}); इसका समाधान \omterm{def:g10:algebra:equation}{समीकरण} और
        सही का चयन करें \omterm{def:g11:quad:discriminant}{जड़}.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g12:seq:sqrt2}
होने देना $ u_0 = 2$ और $ u_{n+1} = \frac{1}{2}\left(u_n + \frac{2}{u_n}\right)$.
एक प्रेरण द्वारा जाँच करता है कि $ u_n \geq \sqrt{2}$ सभी के लिए $ n $ (असमानता
$\frac{1}{2}(x + 2/x) \geq \sqrt{2}$ के लिए $ x>0$ के बराबर है
$(x - \sqrt2)^2 \geq 0$), फिर उस $(u_n)$ है \omterm{def:g10:functions:variations}{घटते}, तब से
\[
u_{n+1}-u_n=\frac{2-u_n^2}{2u_n}\leq 0 .
\]
घटाना और नीचे बंधा हुआ, $(u_n)$
\omterm{def:g12:seq:limit}{अभिसरण} कुछ करने के लिए $\ell \geq \sqrt{2}$, जिसे संतुष्ट करना होगा
$\ell = \frac{1}{2}(\ell + 2/\ell)$, यानी\ $\ell^2 = 2$. इस तरह
$ u_n \to \sqrt{2}$. यह हेरॉन का एल्गोरिदम है, जिसका उपयोग पहले से ही किया जा रहा है
बेबीलोनियाई; इसका \omterm{def:g12:seq:limit}{अभिसरण} अत्यंत तीव्र है ($ u_3$ पहले से ही देता है $\sqrt 2$
आठ दशमलव स्थानों तक)।
\end{example}

\begin{exercise}[$\star $]\label{exo:g12:seq:1}

\begin{exercise}[$\star $]\label{exo:g12:seq:2}
की एकरसता का अध्ययन करें दृश्यों के लिए परिभाषित $ n \geq 1$ द्वारा

a_n = \frac{n+1}{n}, \qquad
b_n = \frac{2^n}{n}, \qquad

\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g12:seq:3}
की सीमा की गणना करें दृश्यों सामान्य शर्तों के साथ
\[
u_n = \frac{3n^2 - n + 1}{2n^2 + 5}, \qquad
v_n = \sqrt{n+1} - \sqrt{n}, \qquad
w_n = \frac{2^n - 3^n}{3^n + 1}.
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g12:seq:4}
होने देना $(u_n)$ हो \omterm{def:g12:seq:arith-geom}{अंकगणित} \omterm{def:g12:seq:sequence}{अनुक्रम} साथ $ u_0 = 5$ और \omterm{def:g11:seq:arithmetic}{सामान्य अंतर}
$ r = 3$, और $(v_n)$  ज्यामितीय क्रम साथ $ v_0 = 8$ और \omterm{def:g11:seq:geometric}{सामान्य अनुपात}
$ q = \frac{1}{2}$. गणना करना $ u_n $, $ v_n $, $\sum_{k=0}^{n} u_k $ और
$\sum_{k=0}^{n} v_k $, और सभी चार अभिव्यक्तियों की सीमाएँ
$ n \to +\infty $.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:seq:5}
निचोड़ प्रमेय का उपयोग करके गणना करें
\[
\lim_{n\to+\infty} \frac{n + \cos n}{n + 1}
\qquad\text{and}\qquad
\lim_{n\to+\infty} \frac{n!}{n^n},
\]
कहाँ $ n! = 1 \times 2 \times \dots \times n $. दूसरी सीमा के लिए, बाध्य
$\frac{n!}{n^n}$ एक शब्द के द्वारा ज्यामितीय क्रम.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:seq:6}
होने देना $ u_0 = 0$ और $ u_{n+1} = \sqrt{u_n + 2}$ सभी के लिए $ n \in \N $.
\begin{enumerate}
  \item प्रेरण द्वारा \omterm{def:g11:vect:det}{सिद्ध} करें कि $0 \leq u_n \leq 2$ सभी के लिए $ n $.
  \item बताते हैं कि $(u_n)$ है \omterm{def:g12:seq:monotonic}{की बढ़ती}.
  \item उसका निष्कर्ष निकालो $(u_n)$ \omterm{def:g12:seq:limit}{अभिसरण} और इसकी सीमा निर्धारित करें.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:seq:7}
एक मरीज़ इसकी एक खुराक लेता है $1$ हर सुबह एक दवा की इकाई. प्रत्येक के दौरान
24 घंटे की अवधि में शरीर समाप्त कर देता है $40\%$ मौजूद दवा का. होने देना $ u_n $ हो
दिन में खुराक के तुरंत बाद शरीर में दवा की मात्रा $ n $, ताकि
$ u_0 = 1$.
\begin{enumerate}
  \item उसे उचित ठहराओ $ u_{n+1} = 0.6\,u_n + 1$.
  \item होने देना $ v_n = u_n - 2.5$. बताते हैं कि $(v_n)$ है \omterm{def:g12:seq:arith-geom}{ज्यामितिक} और एक निष्कर्ष निकालें
        के लिए स्पष्ट सूत्र $ u_n $.
  \item शरीर में दवा की दीर्घकालिक मात्रा निर्धारित करें।
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:seq:8}
होने देना $(u_n)$ द्वारा परिभाषित किया जाए $ u_0 = 3$ और $ u_{n+1} = \frac{4u_n - 1}{u_n + 2}$.
\begin{enumerate}
  \item प्रेरण द्वारा दिखाएँ कि $ u_n > 1$ सभी के लिए $ n \in \N $.
  \item बताते हैं कि $ v_n = \dfrac{1}{u_n - 1}$ एक को परिभाषित करता है \omterm{def:g12:seq:arith-geom}{अंकगणित} \omterm{def:g12:seq:sequence}{अनुक्रम}.
  \item के लिए स्पष्ट सूत्र निकालें $ v_n $ और $ u_n $, और की सीमा
        $(u_n)$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g12:seq:9}
के लिए $ n \geq 1$, होने देना
$ H_n = 1 + \frac{1}{2} + \frac{1}{3} + \dots + \frac{1}{n}$.
\begin{enumerate}
  \item वह सभी को दिखाओ $ n \geq 1$, $ H_{2n} - H_n \geq \frac{1}{2}$.
  \item उसका निष्कर्ष निकालो $ H_{2^k} \geq 1 + \frac{k}{2}$ सभी के लिए $ k \in \N $, और
        यह निष्कर्ष निकालें $ H_n \to +\infty $.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g12:seq:10}
\emph{(नज़दीक दृश्यों.)} दो दृश्यों $(a_n)$ और $(b_n)$ हैं
\emph{नज़दीक} अगर $(a_n)$ है \omterm{def:g12:seq:monotonic}{की बढ़ती}, $(b_n)$ है \omterm{def:g10:functions:variations}{घटते}, और
$ b_n - a_n \to 0$.
\begin{enumerate}
  \item वह सभी को दिखाओ $ n $, $ a_n \leq b_n $. (संकेत: एकरसता का अध्ययन करें
        का $(b_n - a_n)$.)
  \item वह आसन्न दिखाओ दृश्यों दोनों मिलते हैं, एक ही सीमा तक।
  \item आवेदन: दिखाएँ कि दृश्यों
        $ a_n = \sum_{k=0}^{n} \frac{1}{k!}$ और
        $ b_n = a_n + \frac{1}{n \cdot n!}$ ($ n \geq 1$) आसन्न हैं। (उनके
        सामान्य सीमा संख्या है $\eu $, में अध्ययन किया \cref{ch:g12:exp}.)
\end{enumerate}
\end{exercise}

\section{समस्या: बगुले का अनुक्रम, अंततः निर्णय लिया गया}

\begin{problem}[{सप्ताहांत समस्या --- इंडक्शन प्रमाणित, मोनोटोन अभिसरण वाक्य, और दो हजार साल पुराना नुस्खा $\sqrt2$ अंततः इसका प्रमाण मिल गया (गॉस के अद्भुत माध्य के साथ)। मिठाई)}]
\label{pb:g12:seq:1}
तीन बार यह शृंखला हेरॉन की रेसिपी से मिल चुकी है --- औसत
के साथ अनुमान लगाएं $2/\text{guess}$ --- और तीन बार ही ऐसा हो सका
\emph{निरीक्षण} कि नुस्खा काम करता है. यह अध्याय आख़िरकार स्वामी है
निर्णय के उपकरण: प्रेरण
(\cref{thm:g12:seq:induction}), मोनोटोन \omterm{def:g12:seq:limit}{अभिसरण} प्रमेय
(\cref{thm:g12:seq:monotone}), और पुनरावृत्ति की सीमाएं। द
निर्णय और प्रमाणित गति ही इसके \omterm{pb:g11:quad:1}{केंद्र} में हैं
समस्या; इसके चारों ओर, इंडक्शन के क्लासिक जाल, सबसे धीमे
गणित में विचलन, और अब तक का सबसे तेज़ \omterm{def:g12:seq:limit}{अभिसरण} गॉस
मिला.

\medskip
\noindent\textbf{Part I --- Induction warm-ups.}
\begin{enumerate}
  \item प्रेरण द्वारा \omterm{def:g11:vect:det}{सिद्ध} करें:
        $1 + 3 + 5 + \dots + (2n - 1) = n^2$ (सीढ़ी
        अब मिडिल स्कूल खंड में निकाली गई \omterm{def:g11:func:parity}{विषम} संख्याएँ
        प्रमाणित)।
  \item प्रेरण द्वारा \omterm{def:g11:vect:det}{सिद्ध} करें कि $2^n > n $ हरएक के लिए
        $ n \in \N $.
  \item बर्नौली की असमानता को प्रेरण द्वारा \omterm{def:g11:vect:det}{सिद्ध} करें: के लिए
        $ x \geq 0$ और $ n \in \N $,
        $(1 + x)^n \geq 1 + nx $.
  \item क्लासिक जाल: ``सभी मार्बल्स का रंग एक जैसा होता है ---
        एक संगमरमर के लिए सच है; और यदि कोई हो $ n $ मार्बल्स हमेशा होते हैं
        मोनोक्रोम, फिर बीच में $ n + 1$ पहले मार्बल्स $ n $
        एक रंग साझा करें, आखिरी $ n $ एक रंग साझा करें, तो सब
        $ n + 1$ करो।'' हर बच्चा जानता है कि निष्कर्ष क्या है
        बेतुका: सटीक चरण ढूंढें जहां इंडक्शन टूट जाता है।
  \item प्रेरण द्वारा \omterm{def:g11:vect:det}{सिद्ध} करें कि $4^n - 1$ से विभाज्य है $3$
        हरएक के लिए $ n \in \N $.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The trial of Heron.}
होने देना $ x_0 = 2$ और $ x_{n+1} = \dfrac12\left(x_n +
\dfrac{2}{x_n}\right)$.
\begin{enumerate}[resume]
  \item गणना करना $ x_1$, $ x_2$, $ x_3$ सटीक भिन्नों के रूप में (पुराना)।
        दोस्तो)।
  \item मुख्य पहचान \omterm{def:g11:vect:det}{सिद्ध} करें
        \[
        x_{n+1}^2 - 2 = \left(\frac{x_n^2 - 2}{2x_n}\right)^{\!2}
        \geq 0,
        \]
        और प्रेरण द्वारा निष्कर्ष निकालें $ x_n > 0$ और
        $ x_n^2 > 2$ हरएक के लिए $ n $.
  \item बताते हैं कि $(x_n)$ सख्ती से है \omterm{def:g10:functions:variations}{घटते} (गणना करें
        $ x_{n+1} - x_n $ और प्रश्न 7 का उपयोग करें)।
  \item मोनोटोन \omterm{def:g12:seq:limit}{अभिसरण} प्रमेय का आह्वान करें: क्यों करता है
        $(x_n)$ किसी सीमा तक एकत्रित होना $ L \geq 1$?
  \item सीमा को पहचानें: पुनरावृत्ति को सीमा तक पास करें
        (\cref{prop:g12:seq:operations}) और निष्कर्ष निकालें
        $ L = \sqrt2$. ऐतिहासिक फैसला बताएं: दो के बाद
        हजारों वर्ष की वफ़ादार सेवा, बगुले का नुस्खा है
        \emph{साबित} जुटना.
  \item प्रमाणित गति: के साथ $ e_n = x_n - \sqrt2$, \omterm{def:g11:vect:det}{सिद्ध} करना
        \[
        e_{n+1} = \frac{e_n^2}{2 x_n} ,
        \]
        और निष्कर्ष निकालें $ e_{n+1} \leq \frac{e_n^2}{2\sqrt2}$: द
        प्रत्येक चरण पर त्रुटि का वर्ग किया जाता है--अंक-दोगुना
        मिडिल स्कूल वॉल्यूम के बाद से देखा गया, अब एक प्रमेय।
  \item संख्यात्मक रूप से पुष्टि करें: गणना करें $ e_0, e_1, e_2, e_3$ (से
        प्रश्न 6) और प्रत्येक की जाँच करें $\frac{e_{n+1}}{e_n^2}$
        के करीब है $\frac{1}{2x_n}$.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The slowest divergence.}
\begin{enumerate}[resume]
  \item \Cref{exo:g12:seq:9} साबित
        $ H_{2^k} \geq 1 + \frac k2$ हार्मोनिक रकम के लिए. कैसे
        कई शर्तें गारंटी देती हैं $ H_n > 10$? (दो इच्छाशक्ति की शक्ति
        करो; इसके आकार पर आश्चर्य करें।)
  \item इसके विपरीत, \omterm{def:g12:seq:arith-geom}{ज्यामितिक} रकम
        $1 + \frac12 + \frac14 + \dots + \frac{1}{2^n} =
        2 - \frac{1}{2^n}$ में जुटना $2$
        (\cref{thm:g12:seq:geometric}): चॉकलेट-बार
        \omterm{prop:g10:coordgeom:midpoint}{मध्य} विद्यालय की मात्रा का अंतर्ज्ञान, अंततः एक सीमा
        बयान. दो पंक्ति का प्रमाण लिखिए।
  \item दोनों के बीच: दिखाएँ कि रकम
        $ S_n = 1 + \frac{1}{4} + \frac{1}{9} + \dots +
        \frac{1}{n^2}$ एकाग्र होकर, सीमाबद्ध होकर
        $\frac{1}{k^2} \leq \frac{1}{k(k-1)} =
        \frac{1}{k-1} - \frac{1}{k}$ (के लिए $ k \geq 2$),
        टेलीस्कोपिंग, और मोनोटोन \omterm{def:g12:seq:limit}{अभिसरण} लागू करना। (द
        सीमा, $\frac{\pi^2}{6}$, यूलर के चमत्कारों में से एक है,
        विश्वविद्यालय के संस्करणों में साबित हुआ।)
  \item प्रश्न 13-15 का नैतिक दो वाक्यों में बताएं:
        ``शब्द किस ओर प्रवृत्त होते हैं $0$'' के बारे में निर्णय लें
        रकमों का \omterm{def:g12:seq:limit}{अभिसरण} --- और यह क्या नहीं करता है?
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Gauss's arithmetic--geometric
\omterm{def:g11:stat:mean}{अर्थ}.}
होने देना $ a_0 = 1$, $ b_0 = 2$, और
\[
a_{n+1} = \sqrt{a_n b_n},
\qquad
b_{n+1} = \frac{a_n + b_n}{2} .
\]
\begin{enumerate}[resume]
  \item गणना करना $ a_1, b_1, a_2, b_2$ (पांच दशमलव). क्या करें?
        क्या आप गति के बारे में निरीक्षण करते हैं?
  \item बताते हैं कि $ a_n \leq b_n $ हरएक के लिए $ n $ (द
        अंकगणित--ज्यामितीय असमानता, इस दौरान पूरी हुई
        शृंखला), वह $(a_n)$ बढ़ता है और $(b_n)$ घट जाती है.
  \item बताते हैं कि
        $ b_{n+1} - a_{n+1} \leq \frac{b_n - a_n}{2}$
        (कारक $ b_{n+1} - a_{n+1} =
        \frac{(\sqrt{b_n} - \sqrt{a_n})^2}{2}$ और तुलना करें),
        और के साथ समाप्त करें \cref{exo:g12:seq:10} वह दो
        दृश्यों आसन्न हैं: वे एक सामान्य सीमा साझा करते हैं
        $ M(1, 2)$, द \emph{अंकगणित--ज्यामितीय \omterm{def:g11:stat:mean}{अर्थ}}.
  \item गणना करना $ M(1, 2)$ छह दशमलव तक (कितने पुनरावृत्तियाँ
        क्या आपको इसकी आवश्यकता थी?) 30 मई 1799 को गॉस ने गणना की
        $ M(1, \sqrt2)$ ग्यारह दशमलव तक, मान्यता प्राप्त
        $\frac{\pi}{M(1,\sqrt2)}$ एक ज्ञात अभिन्न अंग के रूप में, और
        लिखा कि ``विश्लेषण का एक नया क्षेत्र'' खुल गया है ---
        इसमें: अण्डाकार अभिन्न अंग थे, जो विश्वविद्यालय में बताया गया था
        वॉल्यूम. की देखी गई \omterm{def:g12:seq:limit}{अभिसरण} गति के साथ बंद करें
        यह समस्या, सबसे धीमी से सबसे तेज़।
\end{enumerate}
\end{problem}
