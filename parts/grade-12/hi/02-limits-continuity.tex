\chapter{सीमाएँ और निरंतरता}\label{ch:g12:limcont}

यह अध्याय सीमा की धारणा का \omterm{def:g10:algebra:expand}{विस्तार} करता है दृश्यों को कार्य एक का
वास्तविक चर, और परिचय देता है निरंतरता. केन्द्रीय परिणाम है
मध्यवर्ती मूल्य प्रमेय, जो गुणात्मक जानकारी (``) को बदल देता है $f$ है
निरंतर और चिह्न'') को समाधान के अस्तित्व में बदल देता है \omterm{def:g10:algebra:equation}{समीकरण}.

\section{कार्यों की सीमाएँ}

लगातार, $ f $ एक है \omterm{def:g11:func:function}{समारोह} एक पर परिभाषित \omterm{def:g10:numbers:interval}{अंतराल} या ए \omterm{def:g10:numbers:interunion}{मिलन} का \omterm{def:g10:numbers:interval}{अंतराल}
$ D \subseteq \R $.

\begin{definition}[अनंत पर सीमा]\label{def:g12:limcont:liminf}
होने देना $ f $ एक पर परिभाषित किया जाए \omterm{def:g10:numbers:interval}{अंतराल} $\intco{a}{+\infty}$ और $\ell \in \R $. हम
ऐसा कहो $ f $ \emph{आदत है $\ell $ पर $+\infty $}\index{आदत है $\ell $ पर $+\infty $},
लिखा हुआ $\lim\limits_{x\to+\infty} f(x) = \ell $, यदि हर खुला \omterm{def:g10:numbers:interval}{अंतराल}
युक्त $\ell $ सभी मान समाहित हैं $ f(x)$ के लिए $ x $ काफी बड़ा:
प्रत्येक के लिए $\varepsilon > 0$ वहाँ मौजूद है $ A $ ऐसा कि सभी के लिए $ x \geq A $,
$\abs{f(x) - \ell} \leq \varepsilon $.

हम कहते हैं $ f $ आदत है $+\infty $ पर $+\infty $ यदि प्रत्येक के लिए $ M \in \R $ वहाँ
मौजूद है $ A $ ऐसा है कि $ f(x) \geq M $ सभी के लिए $ x \geq A $. पर सीमा $-\infty $
और सीमाएँ बराबर हैं $-\infty $ अनुरूप रूप से परिभाषित किए गए हैं।
\end{definition}

\begin{definition}[एक बिंदु पर सीमित करें]\label{def:g12:limcont:limpoint}
होने देना $ a \in \R $ और चलो $ f $ निकट परिभाषित किया जाए $ a $ (संभवतः को छोड़कर $ a $). हम
ऐसा कहो $ f $ आदत है $\ell \in \R $ पर $ a $, लिखा हुआ
$\lim\limits_{x \to a} f(x) = \ell $, यदि प्रत्येक के लिए $\varepsilon > 0$ वहाँ
मौजूद है $\delta > 0$ ऐसा कि सभी के लिए $ x \in D $ साथ
$\abs{x - a} \leq \delta $, $\abs{f(x) - \ell} \leq \varepsilon $.

हम कहते हैं $ f $ आदत है $+\infty $ पर $ a $ यदि प्रत्येक के लिए $ M $ वहाँ है $\delta > 0$
ऐसा है कि $ f(x) \geq M $ जब कभी भी $ x \in D $ और $\abs{x - a} \leq \delta $.
एकतरफ़ा सीमा ($ x \to a^+$, $ x \to a^-$) प्रतिबंध लगाना $ x $ को $ x > a $ या
$ x < a $.
\end{definition}

\begin{example}
$\lim\limits_{x\to+\infty} \frac{1}{x} = 0$,
$\lim\limits_{x\to 0^+} \frac{1}{x} = +\infty $,
$\lim\limits_{x\to 0^-} \frac{1}{x} = -\infty $. \omterm{def:g11:func:function}{समारोह}
$ x \mapsto \frac1x $ इसकी कोई (दोतरफा) सीमा नहीं है $0$.
\end{example}

\begin{definition}[स्पर्शोन्मुख]\label{def:g12:limcont:asymptote}
रेखा $ y = \ell $ एक है \emph{समस्तरीय अनंतस्पर्शी रेखा}\index{समस्तरीय अनंतस्पर्शी रेखा} का
का वक्र $ f $ पर $+\infty $ (सम्मान.\ $-\infty $) अगर
$\lim_{x\to+\infty} f(x) = \ell $ (सम्मान.\ पर $-\infty $). रेखा $ x = a $ एक है
\emph{ऊर्ध्वाधर अनंतस्पर्शी}\index{ऊर्ध्वाधर अनंतस्पर्शी} अगर $ f $ पर अनंत एकतरफ़ा सीमा है $ a $.
\end{definition}

\begin{proposition}[सीमा पर संचालन]\label{prop:g12:limcont:operations}
के नियम \cref{prop:g12:seq:operations} (योग, गुणनफल, भागफल) पकड़ें
की सीमा के लिए शब्दशः कार्य, एक बिंदु पर या अनंत पर, उसी के साथ
चार अनिश्चित रूप $(+\infty)+(-\infty)$, $0\times\infty $,
$\frac{\infty}{\infty}$, $\frac{0}{0}$.
\end{proposition}

\begin{proof}
सबूत के समान हैं \omterm{def:g12:seq:sequence}{अनुक्रम} मामला, ``के लिए'' की जगह $ n \geq N $''
``के लिए $ x \geq A $'' या ``के लिए $\abs{x-a} \leq \delta $''.
\end{proof}

\begin{theorem}[किसी रचना की सीमा]\label{thm:g12:limcont:composition}
होने देना $ f, g $ होना कार्य और $ a, b, c $ होना \omterm{def:g10:numbers:sets}{वास्तविक संख्या} या $\pm\infty $. अगर
$\lim\limits_{x \to a} f(x) = b $ और $\lim\limits_{y \to b} g(y) = c $, तब
\[
\lim_{x \to a} g\bigl(f(x)\bigr) = c .
\]
यही कथन a के साथ भी लागू होता है \omterm{def:g12:seq:sequence}{अनुक्रम} की जगह $ f $: अगर
$ u_n \to b $ और $\lim_{y\to b} g(y) = c $, तब $ g(u_n) \to c $.
\end{theorem}

\begin{proof}
लेना $ b, c $ परिमित (अन्य मामले समान हैं)। होने देना $\varepsilon > 0$.
वहाँ है $\delta > 0$ ऐसा है कि $\abs{y - b} \leq \delta $ तात्पर्य
$\abs{g(y) - c} \leq \varepsilon $; और वहाँ है $\delta' > 0$ ऐसा है कि
$\abs{x - a} \leq \delta'$ तात्पर्य $\abs{f(x) - b} \leq \delta $. जंजीर से बांधना
दो देता है $\abs{g(f(x)) - c} \leq \varepsilon $ के लिए
$\abs{x - a} \leq \delta'$.
\end{proof}

\begin{theorem}[तुलना और निचोड़]\label{thm:g12:limcont:squeeze}
होने देना $ f, g, h $ होना कार्य और $ a \in \R \cup \{\pm\infty\}$.
\begin{enumerate}
  \item अगर $ f \leq g $ पास में $ a $ और $ f(x) \to +\infty $ जैसा $ x \to a $, तब
        $ g(x) \to +\infty $.
  \item अगर $ f \leq g \leq h $ पास में $ a $ और $ f, h $ समान परिमित की ओर प्रवृत्त होते हैं
        सीमा $\ell $ पर $ a $, तब $ g(x) \to \ell $ जैसा $ x \to a $.
\end{enumerate}
\end{theorem}

\begin{proof}
के लिए भी वही तर्क दृश्यों (\cref{thm:g12:seq:squeeze}).
\end{proof}

\begin{method}[व्यवहार में कंप्यूटिंग सीमाएँ]\label{met:g12:limcont:limits}
\begin{itemize}
  \item बहुपद और तर्कसंगत कार्य पर $\pm\infty $: गुणनखंड निकालो
        की उच्चतम शक्ति $ x $; सीमा अग्रणी के अनुपात की है
        शर्तें.
  \item $\frac{0}{0}$ एक बिंदु पर $ a $: कारक $(x-a)$ अंश से बाहर और
        हर, या पहचानें ए यौगिक
        $\lim_{x\to a}\frac{f(x)-f(a)}{x-a} = f'(a)$ (\cref{ch:g12:deriv}).
  \item वर्गमूल: संयुग्मी व्यंजक से गुणा करें।
  \item दोलन कारक ($\cos $, $\sin $): निचोड़ प्रमेय का उपयोग करें।
\end{itemize}
\end{method}

\section{निरंतरता}

\begin{definition}[निरंतरता]\label{def:g12:limcont:continuity}
होने देना $ f $ एक पर परिभाषित किया जाए \omterm{def:g10:numbers:interval}{अंतराल} $ I $ और $ a \in I $. \omterm{def:g11:func:function}{समारोह} $ f $ है
\emph{पर निरंतर $ a $}\index{पर लगातार $ a $} अगर
$\lim\limits_{x \to a} f(x) = f(a)$. यह है \emph{निरंतर चालू $ I $}\index{निरंतर $ I $} यदि यह है
के हर बिंदु पर निरंतर $ I $.
\end{definition}

सहज रूप से, \omterm{def:g10:functions:graph}{ग्राफ} एक का निरंतर \omterm{def:g11:func:function}{समारोह} एक पर \omterm{def:g10:numbers:interval}{अंतराल} खींचा जा सकता है
बिना कलम उठाए.

\begin{proposition}[सामान्य कार्यों की निरंतरता]\label{prop:g12:limcont:usual}
बहुपद, तर्कसंगत कार्य, $ x \mapsto \sqrt{x}$, $ x \mapsto \abs{x}$,
$\cos $, $\sin $, $\exp $ और $\ln $ हैं निरंतर सभी पर \omterm{def:g10:numbers:interval}{अंतराल} निहित
उनके में \omterm{def:g11:func:function}{कार्यक्षेत्र}. योग, उत्पाद, भागफल (जहाँ परिभाषित) और रचनाएँ
का निरंतर कार्य हैं निरंतर.
\end{proposition}

\begin{proof}[आंशिक प्रमाण]
योगों, उत्पादों, भागफलों और संरचना के अंतर्गत स्थिरता इसी से आती है
सीमाओं पर संगत प्रमेय। निरंतरता का $ x \mapsto x $ और स्थिरांक का
कार्य परिभाषा से निकटतम है, इसलिए बहुपद और तर्कसंगत है
कार्य हैं निरंतर. निरंतरता शेष सामान्य का कार्य है
उन अध्यायों में \omterm{def:g11:vect:det}{सिद्ध} किया गया है जहाँ उनका निर्माण किया गया है, या उन्हें स्वीकार किया गया है।
\end{proof}

\begin{example}
 \emph{ज़मीन \omterm{def:g11:func:function}{समारोह}} $ x \mapsto \floor{x}$ (महानतम पूर्णांक
$\leq x $) क्या नहीं है निरंतर किसी पर पूर्णांक $ n $: इसकी बाईं सीमा है
$ n - 1$ जबकि $\floor{n} = n $.
\end{example}

\begin{proposition}[अनुक्रम और निरंतरता]\label{prop:g12:limcont:seqcont}
अगर $ u_n \to \ell $ और $ f $ है निरंतर पर $\ell $, तब $ f(u_n) \to f(\ell)$.
विशेष रूप से, यदि ए \omterm{def:g12:seq:sequence}{अनुक्रम} द्वारा परिभाषित $ u_{n+1} = f(u_n)$ \omterm{def:g12:seq:limit}{अभिसरण} को
$\ell $ और $ f $ है निरंतर पर $\ell $, तब $ f(\ell) = \ell $.
\end{proposition}

\begin{proof}
पहला दावा है \cref{thm:g12:limcont:composition} के साथ लागू किया गया
\omterm{def:g12:seq:sequence}{अनुक्रम} $(u_n)$. दूसरे के लिए, दोनों तरफ की सीमा से गुजरें
$ u_{n+1} = f(u_n)$: बाईं ओर की ओर झुकाव होता है $\ell $, दाहिनी ओर
$ f(\ell)$, और सीमाएँ अद्वितीय हैं।
\end{proof}

\section{मध्यवर्ती मूल्य प्रमेय}

\begin{theorem}[मध्यवर्ती मूल्य प्रमेय]\label{thm:g12:limcont:ivt}
\index{मध्यवर्ती मूल्य प्रमेय}
होने देना $ f $ होना निरंतर पर $\intcc{a}{b}$ और चलो $ k $ के बीच कोई भी संख्या हो
$ f(a)$ और $ f(b)$. तभी अस्तित्व है $ c \in \intcc{a}{b}$ ऐसा है कि
$ f(c) = k $.
\end{theorem}

{\small मध्यवर्ती मूल्य प्रमेय: ए निरंतर वक्र \omterm{def:g10:vectors:sum}{जोड़}
$(a, f(a))$ को $(b, f(b))$ प्रत्येक क्षैतिज रेखा को पार करना होगा
$ y = k $ बीच में पड़ा हुआ $ f(a)$ और $ f(b)$.}
\end{omfigure}

\begin{proof}
की जगह $ f $ द्वारा $ f - k $ (और तक $ k - f $ यदि आवश्यक हो), तो हम मान सकते हैं
$ f(a) \leq 0 \leq f(b)$ और शून्य की तलाश करें $ f $. हम दो का निर्माण करते हैं
दृश्यों द्वारा \emph{विरोधाभास}\index{विरोधाभास}: तय करना $ a_0 = a $, $ b_0 = b $;
दिया गया $ a_n \leq b_n $ साथ $ f(a_n) \leq 0 \leq f(b_n)$, होने देना
$ m = \frac{a_n + b_n}{2}$ और परिभाषित करें
\[
(a_{n+1}, b_{n+1}) =
\begin{cases}
(a_n, m) & \text{if } f(m) \geq 0,\\
(m, b_n) & \text{if } f(m) < 0.
\end{cases}
\]
दोनों ही मामलों में $ f(a_{n+1}) \leq 0 \leq f(b_{n+1})$, द \omterm{def:g12:seq:sequence}{अनुक्रम} $(a_n)$ है
\omterm{def:g12:seq:monotonic}{की बढ़ती}, $(b_n)$ है \omterm{def:g10:functions:variations}{घटते}, और
$ b_n - a_n = \frac{b-a}{2^n} \to 0$: द दृश्यों आसन्न हैं (देखें)
\cref{exo:g12:seq:10}) और एक सामान्य सीमा पर एकत्रित हो जाते हैं
$ c \in \intcc{a}{b}$. द्वारा निरंतरता, $ f(a_n) \to f(c)$ और
$ f(b_n) \to f(c)$; में सीमा तक जाना $ f(a_n) \leq 0$ और
$ f(b_n) \geq 0$ देता है $ f(c) \leq 0 \leq f(c)$, इसलिए $ f(c) = 0$.
\end{proof}

\begin{remark}
 \omterm{thm:g12:limcont:ivt}{विरोधाभास} प्रमाण एक एल्गोरिदम है: यह मनमाने ढंग से सटीक परिणाम देता है
अनुमान एक समाधान, प्रत्येक चरण पर त्रुटि को आधा करना। इस प्रकार ए
कंप्यूटर हल कर सकता है $ f(x) = 0$ केवल यह जानना कि मूल्यांकन कैसे करना है $ f $.
\end{remark}

\begin{theorem}[विक्षेप प्रमेय]\label{thm:g12:limcont:bijection}
होने देना $ f $ होना निरंतर और \emph{कठोरता से एकरस} एक पर \omterm{def:g10:numbers:interval}{अंतराल} $ I $. फिर
प्रत्येक के लिए $ k $ की सीमाओं के बीच सख्ती से $ f $ के अंतिम बिंदु पर $ I $, द
\omterm{def:g10:algebra:equation}{समीकरण} $ f(x) = k $ एक \emph{अद्वितीय} समाधान में $ I $.
\end{theorem}

\begin{proof}
अस्तित्व: चुनें $ a, b \in I $ साथ $ k $ बीच में $ f(a)$ और $ f(b)$ (संभव
के मूल्यों के बाद से $ f $ इसकी अंतिम बिंदु सीमा तक पहुंचें), और लागू करें
मध्यवर्ती मूल्य प्रमेय पर $\intcc{a}{b}$. विशिष्टता: यदि $ x < y $, सख्त
एकरसता देता है $ f(x) \neq f(y)$, इसलिए दो अलग-अलग बिंदु दोनों नहीं हो सकते
समाधान.
\end{proof}

\begin{method}[हल $ f(x) = k $ आक्षेप प्रमेय के साथ]\label{met:g12:limcont:bijection}
यह साबित करने के लिए $ f(x) = k $ में बिल्कुल एक ही समाधान है \omterm{def:g10:numbers:interval}{अंतराल}:
\begin{enumerate}
  \item एक स्थापित करें \omterm{def:g10:functions:table}{विविधता तालिका} का $ f $ (हस्ताक्षर करना $ f'$, देखना
        \cref{ch:g12:deriv});
  \item सभी के ऊपर \omterm{def:g10:numbers:interval}{अंतराल} कहाँ $ f $ है निरंतर और सख्ती से एकरस,
        तुलना करना $ k $ समापन बिंदु मानों या सीमाओं के साथ;
  \item ऐसे प्रत्येक पर आक्षेप प्रमेय लागू करें \omterm{def:g10:numbers:interval}{अंतराल} और गिनें
        समाधान.
\end{enumerate}
एक कैलकुलेटर या \omterm{thm:g12:limcont:ivt}{विरोधाभास} फिर प्रत्येक समाधान को इच्छानुसार सूक्ष्मता से स्थानीयकृत करता है।
\end{method}

\begin{example}\label{ex:g12:limcont:cubic}
 \omterm{def:g10:algebra:equation}{समीकरण} $ x^3 + x = 1$ एक अनोखा वास्तविक समाधान है. वास्तव में
$ f(x) = x^3 + x $ है निरंतर और सख्ती से \omterm{def:g12:seq:monotonic}{की बढ़ती} पर $\R $ (कुल योग के रूप में)
सख्ती से कार्यों में वृद्धि), $\lim_{-\infty} f = -\infty $ और
$\lim_{+\infty} f = +\infty $, इसलिए आक्षेप प्रमेय लागू होता है $ k = 1$.
तब से $ f(0.6) = 0.816 < 1$ और $ f(0.7) = 1.043 > 1$, समाधान निहित है
$\intoo{0.6}{0.7}$.
\end{example}

\section{अभ्यास}

\begin{exercise}[$\star $]\label{exo:g12:limcont:1}
निम्नलिखित सीमाओं की गणना करें:
\[
\lim_{x\to+\infty} \frac{2x^3 - x + 1}{5x^3 + x^2}, \qquad
\lim_{x\to-\infty} \bigl(x^2 + 3x - 1\bigr), \qquad
\lim_{x\to 2^+} \frac{x+1}{x-2}, \qquad
\lim_{x\to+\infty} \frac{\cos x}{x}.
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g12:limcont:2}
होने देना $ f(x) = \dfrac{2x^2 - 3x + 1}{x - 1}$, के लिए परिभाषित $ x \neq 1$.
\begin{enumerate}
  \item बताते हैं कि $ f(x) = 2x - 1$ सभी के लिए $ x \neq 1$.
  \item परिणाम निकालना $\lim\limits_{x \to 1} f(x)$. कर सकना $ f $ ए तक बढ़ाया जाए
        \omterm{def:g11:func:function}{समारोह} निरंतर पर $1$?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g12:limcont:3}
निर्धारित करें स्पर्शोन्मुख के वक्र का
$ f(x) = \dfrac{3x - 1}{x + 2}$ और वक्र का रेखाचित्र बनाएं।
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:limcont:4}
गणना करना
\[
\lim_{x\to+\infty} \bigl(\sqrt{x^2 + x} - x\bigr)
\qquad\text{and}\qquad
\lim_{x\to 0} \frac{\sqrt{1+x} - 1}{x}.
\]
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:limcont:5}
होने देना $ f(x) = x + \sin x $.
\begin{enumerate}
  \item बताते हैं कि $ f(x) \to +\infty $ जैसा $ x \to +\infty $, हालांकि $ f $ क्या नहीं है
        एकरस किसी पर \omterm{def:g10:numbers:interval}{अंतराल} $\intco{A}{+\infty}$.
  \item बताते हैं कि $ g(x) = \dfrac{x + \sin x}{x - \sin x}$ आदत है $1$ पर
        $+\infty $.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:limcont:6}
दिखाओ कि \omterm{def:g10:algebra:equation}{समीकरण} $ x^3 - 3x + 1 = 0$ इसके बिल्कुल तीन वास्तविक समाधान हैं,
और प्रत्येक के लिए दे दो \omterm{def:g10:numbers:interval}{अंतराल} लम्बाई का $1$ इसमें शामिल है.
(संकेत: की विविधताओं का अध्ययन करें $ f(x) = x^3 - 3x + 1$; इसका यौगिक है
$3x^2 - 3$.)
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:limcont:7}
होने देना $ f \colon \intcc{0}{1} \to \intcc{0}{1}$ होना निरंतर. बताते हैं कि $ f $ एक \emph{\omterm{pb:g10:functions:1}{नियत बिन्दु}}:वहाँ मौजूद है $ c \in \intcc{0}{1}$ साथ $ f(c) = c $.
(संकेत: मध्यवर्ती मूल्य प्रमेय को लागू करें $ g(x) = f(x) - x $.)
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g12:limcont:8}
एक यात्री पहाड़ी रास्ते पर चढ़ता है, जो 8:00 बजे शुरू होता है और 20:00 बजे पहुँचता है। द
अगले दिन वह उसी रास्ते से नीचे उतरती है, फिर से 8:00 बजे शुरू होती है और पहुंचती है
20:00 बजे। दिखाएँ कि दिन का एक समय होता है जब वह बिल्कुल वैसी ही होती है
दोनों दिन निशान का बिंदु. (संकेत: दोनों के अंतर का परिचय दें
स्थिति कार्य.)
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g12:limcont:9}
होने देना $ f(x) = x^5 + x^3 + x - 1$.
\begin{enumerate}
  \item बताते हैं कि $ f $ एक अद्वितीय वास्तविक शून्य है $\alpha $, ओर वो
        $\alpha \in \intoo{0}{1}$.
  \item वर्णन करें ए \omterm{thm:g12:limcont:ivt}{विरोधाभास} प्रक्रिया कंप्यूटिंग $\alpha $ भीतर तक
        $10^{-6}$, और आरंभ करने के लिए आवश्यक पुनरावृत्तियों की संख्या दें
        $\intcc{0}{1}$.
\end{enumerate}
\end{exercise}

\section{समस्या: आप भीगे बिना नदी पार नहीं कर सकते}

\begin{problem}[{सप्ताहांत समस्या---मध्यवर्ती मूल्य प्रमेय निश्चित बिंदुओं, समान-तापमान एंटीपोड और को बल देता है स्थिर तालिकाएँ: एक विचार से असंभव लगने वाले तीन प्रमेय}]
\label{pb:g12:limcont:1}
फ़्रांस के मानचित्र को तोड़-मरोड़ कर फ़्रांस की धरती पर कहीं भी गिरा दें: कुछ
मानचित्र का बिंदु निहित है \emph{बिल्कुल} इसे जगह के ऊपर
प्रतिनिधित्व करता है. हर पल, दो बिल्कुल विपरीत बिंदु
भू\omterm{prop:g10:coordgeom:midpoint}{मध्य} रेखा का है \emph{बिल्कुल} वही तापमान. और ए
ऊबड़-खाबड़ जमीन पर डगमगाती चौकोर मेज को हमेशा स्थिर रखा जा सकता है
\emph{घूर्णन} यह। तीन पक्ष युक्तियाँ, एक प्रमेय: ए निरंतर
मात्रा लुप्त हुए बिना नकारात्मक से सकारात्मक की ओर नहीं जा सकती
(\cref{thm:g12:limcont:ivt}). यह समस्या तब सीमा को प्रशिक्षित करती है
तीनों आश्चर्यों का निर्माण करता है --- और जानता है कि वास्तव में प्रमेय क्या है
वादा नहीं करता.

\medskip
\noindent\textbf{Part I --- Limit fluency.}
\begin{enumerate}
  \item गणना करना $\lim_{x \to +\infty} \dfrac{3x^2 - x}{x^2 + 5}$,
        \ $\lim_{x \to +\infty} \dfrac{2x + 1}{x^2 + 1}$, \ और
        $\lim_{x \to +\infty} (x^3 - x^2)$
        (\cref{met:g12:limcont:limits}).
  \item दे दो स्पर्शोन्मुख का
        $ f(x) = \dfrac{3x - 1}{x + 2}$
        (\cref{def:g12:limcont:asymptote}), की गणना
        प्रासंगिक सीमाएँ.
  \item गणना करना $\lim_{x \to 3} \dfrac{x^2 - 9}{x - 3}$
        (कारक) और
        $\lim_{x \to 0} \dfrac{\sqrt{x + 1} - 1}{x}$
        (संयुग्मित)।
  \item निचोड़ प्रमेय द्वारा
        (\cref{thm:g12:limcont:squeeze}): गणना करें
        $\lim_{x \to 0} x \sin\frac1x $ और
        $\lim_{x \to +\infty} \frac{\sin x}{x}$.
  \item रचना और प्रमुख शब्दों के अनुसार:
        $\lim_{x \to +\infty} \dfrac{\sqrt{4x^2 + x}}{x}$.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- \omterm{def:g11:quad:discriminant}{जड़}ों, certified.}
\begin{enumerate}[resume]
  \item दिखाओ कि \omterm{def:g10:algebra:equation}{समीकरण} $ x^3 + x = 5$ कम से कम एक है
        समाधान में $\intcc{1}{2}$.
  \item दिखाएँ कि समाधान अद्वितीय है $\R $
        (\cref{thm:g12:limcont:bijection}).
  \item क्लासिक साबित करें: \emph{\omterm{def:g11:func:parity}{विषम} डिग्री का प्रत्येक बहुपद
        कम से कम एक असली है \omterm{def:g11:quad:discriminant}{जड़}}. (तर्क लिखिए
        के लिए $ p(x) = x^3 + a x^2 + b x + c $: दोनों की गणना करें
        प्रमुख पद द्वारा अनंत सीमाएँ, फिर IVT लागू करें
        एक बड़ा \omterm{def:g10:numbers:interval}{अंतराल}.)
  \item का पता लगा रहा है \omterm{def:g11:quad:discriminant}{जड़} द्वारा प्रश्न 6 का \omterm{thm:g12:limcont:ivt}{विरोधाभास}: कितने
        का आधा भाग $\intcc{1}{2}$ नीचे एक त्रुटि की गारंटी दें
        $10^{-6}$ (\cref{exo:g12:limcont:9})? से तुलना करें
        न्यूटन की विधि का अंक-दोहरीकरण
        (\cref{pb:g11:deriv:1}): मोटे तौर पर कितने कदम होंगे
        इसकी आवश्यकता है?
  \item प्रतिउदाहरण जो प्रमेय की रक्षा करता है:
        $ g(x) = \frac1x $ संतुष्ट $ g(-1) = -1 < 0$ और
        $ g(1) = 1 > 0$, फिर भी कभी गायब नहीं होता। की कौन सी परिकल्पना
        आईवीटी विफल हो जाता है --- और उदाहरण किस बारे में सिखाता है
        परिकल्पनाओं की जाँच कर रहे हैं?
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Three impossible-sounding
theorems.}
\begin{enumerate}[resume]
  \item खंड का निश्चित-बिंदु प्रमेय: यदि
        $ f : \intcc{0}{1} \to \intcc{0}{1}$ है निरंतर, साबित करो
        वह कुछ $ c $ संतुष्ट $ f(c) = c $. (अध्ययन
        $ g(x) = f(x) - x $ दोनों सिरों पर.)
  \item प्रश्न 11 का अनुवाद टूटे हुए मानचित्र में करें: क्या खेलता है
        की भूमिका $ f $, यह क्यों उतरता है
        $\intcc{0}{1}$ (लाक्षणिक रूप से: फ्रांस के भीतर), और क्या
        क्या बिंदु अपने स्थान के ठीक ऊपर स्थित है? (एक
        आयाम को चुपचाप स्वीकार किया जाता है --- ईमानदार
        द्वि-आयामी कथन ब्रौवर का प्रमेय है,
        विश्वविद्यालय खंड।)
  \item भू\omterm{prop:g10:coordgeom:midpoint}{मध्य} रेखा: चलो $ T $ एक हो निरंतर ए पर तापमान
        वृत्त, के रूप में देखा जाता है $2\pi $-आवधिक \omterm{def:g11:func:function}{समारोह} साथ
        $ T(0) = T(2\pi)$. तय करना $ g(t) = T(t) - T(t + \pi)$.
        गणना करना $ g(0) + g(\pi)$, वह निष्कर्ष निकालो $ g $ गायब हो जाता है
        कहीं पर $\intcc{0}{\pi}$, और प्रमेय बताएं
        एंटीपोडल बिंदुओं के बारे में।
  \item डगमगाती मेज: एक बिल्कुल कठोर वर्गाकार मेज खड़ी है
        पर निरंतर (कोई कदम नहीं!) ऊबड़-खाबड़ ज़मीन; तीन पैर
        हमेशा स्पर्श करें, कोई ऊंचाई से मँडराता है $ h(\theta)$ कि निर्भर करता है लगातार मेज के घूर्णन कोण पर
        $\theta $. बताएं कि एक चौथाई मोड़ बलों द्वारा क्यों घूम रहा है
        संकेत बदलने के लिए मँडरा अंतर --- और निष्कर्ष निकालो
        कुछ कोण $\theta^*$ चारों पैरों को स्थिर करता है। (यह
        तर्क को 2005 में एक वास्तविक प्रमेय के रूप में प्रकाशित किया गया था;
        कैफ़े\'ई मालिकों को चाल पहले पता थी।)
  \item का पदयात्री \cref{exo:g12:limcont:8} वही प्रयोग किया
        योजना. प्रश्न 11, 13 द्वारा साझा किया गया टेम्पलेट बताएं
        14 और यात्री --- जो निर्माण करें \omterm{def:g11:func:function}{समारोह}, जांचें क्या
        अंत में तीन पंक्तियों में क्या --- का आह्वान करें।
  \item आईवीटी क्या करता है \emph{नहीं} देना: अस्तित्व मुक्त हो गया,
        लेकिन यह किन दो प्रश्नों को खुला छोड़ देता है, और कौन से
        इस समस्या के उपकरण उनका उत्तर देते हैं (प्रश्न 7 और 9)?
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Portraits with स्पर्शोन्मुख.}
\begin{enumerate}[resume]
  \item पुनर्लेखन $ f(x) = \dfrac{3x - 1}{x + 2}$ जैसा
        $3 - \dfrac{7}{x + 2}$ और वक्र का वर्णन करने के लिए इसका उपयोग करें
        पूरी तरह से: स्पर्शोन्मुख, प्रत्येक पक्ष पर भिन्नताएं,
        स्केच. (ए स्थानांतरित हो गया \omterm{def:g11:func:reference}{अतिशयोक्ति} --- औसत लागत वक्र
        का \cref{pb:g10:reffunc:1} भी एक था.)
  \item  \omterm{def:g11:func:function}{समारोह} $ f(x) = \dfrac{x^2 - 1}{x - 1}$ है
        पर अपरिभाषित $1$. कौन सा मान असाइन किया गया है $ f(1)$ उसे बनाता है निरंतर वहाँ --- और विश्लेषक ऐसा क्यों कहते हैं
        बिंदु ए \emph{हटाने योग्य} असंततता?
  \item मंज़िल \omterm{def:g11:func:function}{समारोह} $ x \mapsto \lfloor x \rfloor $
        (महान पूर्णांक $\leq x $): कहाँ है निरंतर,
        यह कहां छलांग लगाता है, और आईवीटी का निष्कर्ष क्यों निकलता है
        इसके लिए असफल (खोजें $ a < b $ साथ
        $\lfloor a \rfloor < \frac12 < \lfloor b \rfloor $ अभी तक
        का कोई समाधान नहीं $\lfloor x \rfloor = \frac12$)?
  \item समापन --- निरंतरताके दो चेहरे: स्थानीय वादा
        (मूल्य $=$ सीमा: कोई टेलीपोर्टिंग नहीं) और वैश्विक शक्ति
        (आईवीटी: प्रत्येक मध्यवर्ती मूल्य का दौरा किया गया)। चार फ़ाइल करें
        दाहिने चेहरे के नीचे भाग III के चमत्कार, और दे दो
        नदी पार करने का नारा इसका सटीक गणितीय नाम है।
\end{enumerate}
\end{problem}
