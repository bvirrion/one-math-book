\chapter{सशर्त संभाव्यता और स्वतंत्रता}\label{ch:g12:condprob}

\omterm{def:g10:proba:distribution}{संभावना} अनिश्चितता की मात्रा निर्धारित करता है; सशर्त संभाव्यता मात्रा निर्धारित करता है कि कैसे
सूचना इसे बदल देती है. यह सीखना कि ए \omterm{def:g11:prob:model}{आयोजन} $B $ हुआ पुनः आकार देता है
अन्य सभी की संभावनाएँ घटनाएँ --- बेयस और द्वारा औपचारिक रूप दिया गया एक तंत्र
अदालतों और अखबारों में रोजाना इसका दुरुपयोग होता है। यह अध्याय नियम स्थापित करता है
कंडीशनिंग का और इसका सटीक \omterm{def:g11:stat:mean}{अर्थ} स्वतंत्रता.

\section{संभाव्यता स्थान (अनुस्मारक)}

परिमित रूप से अनेक परिणामों वाला एक प्रयोग किसके द्वारा प्रतिरूपित किया जाता है? \emph{\omterm{def:g11:prob:model}{नमूना अंतरिक्ष}} $\Omega $ (परिणामों का सेट) और ए \omterm{def:g10:proba:distribution}{संभावना} $\P $ को सौंपना
प्रत्येक \omterm{def:g11:prob:model}{आयोजन} $ A \subseteq \Omega $ एक संख्या $\P(A) \in \intcc{0}{1}$, योगात्मक
विच्छेदन पर यूनियन और साथ में $\P(\Omega) = 1$. बुनियादी नियमों को याद करें:
\[
\P(\bar A) = 1 - \P(A),
\qquad
\P(A \cup B) = \P(A) + \P(B) - \P(A \cap B).
\]
जब सभी परिणाम समान संभाव्य हों,
$\P(A) = \frac{\abs A}{\abs\Omega}$ --- और कंप्यूटिंग संभावनाएं कम हो जाती हैं
गिनती की तकनीक के लिए \cref{ch:g12:comb}.

\section{सशर्त संभाव्यता}

\begin{definition}[सशर्त संभाव्यता]\label{def:g12:condprob:cond}
होने देना $ B $ एक हो \omterm{def:g11:prob:model}{आयोजन} साथ $\P(B) > 0$. \emph{की संभावना $ A $ दिया गया
$ B $}\index{की संभावना $ A $ दिया गया
$ बी $} है
\[
\pcond{B}{A} = \frac{\P(A \cap B)}{\P(B)} .
\]
\end{definition}

वो नक्शा $ A \mapsto \pcond BA $ स्वयं एक है \omterm{def:g10:proba:distribution}{संभावना} (सभी नियम लागू); यह
यह उस व्यक्ति के ज्ञान की नई स्थिति का प्रतिनिधित्व करता है जिसने इसे सीखा है $ B $
घटित हुआ।

\begin{proposition}[गुणन नियम]\label{prop:g12:condprob:product}
के लिए घटनाएँ शून्येतर संभावनाओं के साथ:
\[
\P(A \cap B) = \P(B)\,\pcond{B}{A} = \P(A)\,\pcond{A}{B},
\]
और अधिक सामान्यतः
$\P(A_1 \cap A_2 \cap A_3) = \P(A_1)\,\pcond{A_1}{A_2}\,
\pcond{A_1 \cap A_2}{A_3}$, वगैरह।
\end{proposition}

\begin{proof}
परिभाषा को पुनर्व्यवस्थित करें; श्रृंखला सूत्र पुनरावृत्त करके अनुसरण करता है।
\end{proof}

\begin{theorem}[कुल संभाव्यता का नियम]\label{thm:g12:condprob:total}
\index{कुल संभाव्यता का नियम}
होने देना $ B_1, \dots, B_n $ का एक विभाजन हो $\Omega $ में घटनाएँ शून्येतर का
\omterm{def:g10:proba:distribution}{संभावना}. हरएक के लिए \omterm{def:g11:prob:model}{आयोजन} $ A $:
\[
\P(A) = \sum_{i=1}^{n} \P(B_i)\,\pcond{B_i}{A}.
\]
\end{theorem}

\begin{proof}
सेट $ A \cap B_i $ \omterm{def:g10:vectors:sum}{जोड़}ीवार असंयुक्त हैं \omterm{def:g10:numbers:interunion}{मिलन} $ A $, इसलिए
$\P(A) = \sum_i \P(A \cap B_i) = \sum_i \P(B_i)\pcond{B_i}{A}$ द्वारा
गुणन नियम.
\end{proof}

\begin{method}[संभाव्यता वृक्ष]\label{met:g12:condprob:tree}
ए वृक्ष आरेख सशर्त संभावनाओं को व्यवस्थित करता है: प्रत्येक शाखा वहन करती है
\omterm{def:g10:proba:distribution}{संभावना} अगले का \omterm{def:g11:prob:model}{आयोजन} \emph{दिया गया} अब तक का रास्ता.
\begin{itemize}
  \item  \omterm{def:g10:proba:distribution}{संभावना} एक पत्ते का (एक पूर्ण पथ) है \emph{उत्पाद} का
        इसकी शाखाओं के साथ संभावनाएँ (गुणा नियम)।
  \item  \omterm{def:g10:proba:distribution}{संभावना} की एक \omterm{def:g11:prob:model}{आयोजन} है \emph{\omterm{def:g10:vectors:sum}{जोड़}} संभावनाओं का
        पत्तियों का इसे साकार करना (कुल)। \omterm{def:g10:proba:distribution}{संभावना}).
  \item एक नोड छोड़ने वाली शाखाओं पर संभावनाएँ जुड़ जाती हैं $1$.
\end{itemize}
\end{method}

\begin{theorem}[बेयस का सूत्र]\label{thm:g12:condprob:bayes}
\index{बेयस का सूत्र}
होने देना $ B_1, \dots, B_n $ का एक विभाजन हो $\Omega $ जैसा कि ऊपर और $ A $ एक \omterm{def:g11:prob:model}{आयोजन}
साथ $\P(A) > 0$. तब
\[
\pcond{A}{B_j} =
\frac{\P(B_j)\,\pcond{B_j}{A}}{\sum_{i=1}^{n} \P(B_i)\,\pcond{B_i}{A}} .
\]
\end{theorem}

\begin{proof}
$\pcond{A}{B_j} = \frac{\P(A \cap B_j)}{\P(A)}$; द्वारा अंश का \omterm{def:g10:algebra:expand}{विस्तार} करें
गुणन नियम और हर को कुल से \omterm{def:g10:proba:distribution}{संभावना}.
\end{proof}

\begin{example}[स्क्रीनिंग टेस्ट]\label{ex:g12:condprob:test}
एक बीमारी असर करती है $1\%$ एक आबादी का. एक परीक्षण से इसका पता चलता है \omterm{def:g10:proba:distribution}{संभावना}
$0.99$ (संवेदनशीलता) और एक झूठी सकारात्मकता देता है \omterm{def:g10:proba:distribution}{संभावना} $0.05$.
एक सकारात्मक परीक्षण को देखते हुए, \omterm{def:g10:proba:distribution}{संभावना} असल में यह बीमारी होने का मतलब है
\[
\frac{0.01 \times 0.99}{0.01\times0.99 + 0.99\times0.05}
= \frac{0.0099}{0.0099 + 0.0495} = \frac{1}{6} \approx 0.17 .
\]
सटीक परीक्षण के बावजूद, छह में से पांच सकारात्मक गलत हैं --- क्योंकि
रोग दुर्लभ है. भ्रामक $\pcond{A}{B}$ साथ $\pcond{B}{A}$ है
\emph{अभियोजक की भ्रांति}.
\end{example}

\section{आज़ादी}

\begin{definition}[स्वतंत्र घटनाएँ]\label{def:g12:condprob:indep}
दो घटनाएँ $ A $ और $ B $ हैं \emph{स्वतंत्र}\index{स्वतंत्र} अगर
\[
\P(A \cap B) = \P(A)\,\P(B) .
\]
कब $\P(B) > 0$, यह इसके बराबर है $\pcond{B}{A} = \P(A)$: जानना $ B $
को नहीं बदलता \omterm{def:g10:proba:distribution}{संभावना} का $ A $.
\end{definition}

\begin{proposition}\label{prop:g12:condprob:indepcompl}
अगर $ A $ और $ B $ हैं \omterm{def:g12:condprob:indep}{स्वतंत्र}, तो हैं $ A $ और $\bar B $ (और $\bar A $ और
$\bar B $).
\end{proposition}

\begin{proof}
$\P(A \cap \bar B) = \P(A) - \P(A \cap B) = \P(A) - \P(A)\P(B)
= \P(A)\bigl(1 - \P(B)\bigr) = \P(A)\,\P(\bar B)$.
\end{proof}

\begin{remark}
भ्रमित न करें \emph{\omterm{def:g12:condprob:indep}{स्वतंत्र}} ($\P(A\cap B) = \P(A)\P(B)$) साथ
\emph{\omterm{def:g10:proba:operations}{असंगत}} ($ A \cap B = \varnothing $). दो \omterm{def:g10:proba:operations}{असंगत} घटनाएँ का
शून्येतर \omterm{def:g10:proba:distribution}{संभावना} कभी नहीं हैं \omterm{def:g12:condprob:indep}{स्वतंत्र}: एक को जानना गारंटी देता है
दूसरे ने नहीं किया.
\end{remark}

\begin{definition}[स्वतंत्र दोहराव]\label{def:g12:condprob:repet}
जब कोई प्रयोग दोहराया जाता है $ n $ प्रत्येक परीक्षण के परिणाम को कई बार
दूसरों को प्रभावित नहीं करता, \omterm{def:g10:proba:distribution}{संभावना} एक निर्दिष्ट का \omterm{def:g12:seq:sequence}{अनुक्रम} का
परिणाम व्यक्तिगत संभावनाओं का उत्पाद है। यह मॉडल है
द्विपद के अंतर्गत \omterm{def:g11:prob:rv}{वितरण} (\cref{ch:g12:randvar}).
\end{definition}

\section{अभ्यास}

\begin{exercise}[$\star $]\label{exo:g12:condprob:1}
मानक 52-कार्ड डेक से एक कार्ड निकाला जाता है। की गणना करें \omterm{def:g10:proba:distribution}{संभावना} वह
यह एक राजा है, यह देखते हुए कि यह एक फेस कार्ड (जैक, रानी या राजा) है। हैं
घटनाएँ ``राजा'' और ``दिल'' \omterm{def:g12:condprob:indep}{स्वतंत्र}?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g12:condprob:2}
एक कलश में 5 लाल और 3 नीली गेंदें हैं। लगातार दो गेंदें निकाली जाती हैं
\emph{बिना} प्रतिस्थापन।
\begin{enumerate}
  \item खींचना \omterm{def:g10:proba:distribution}{संभावना} \omterm{met:g10:proba:tree}{पेड़}.
  \item की गणना करें \omterm{def:g10:proba:distribution}{संभावना} कि दोनों लाल हैं, और \omterm{def:g10:proba:distribution}{संभावना} वह
        दूसरा लाल है.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g12:condprob:3}
दो निष्पक्ष पासे फेंके गए। होने देना $ A $ = ``योग है $7$'', $ B $ = ``पहला
डाई शो $3$'', $ C $ = ``योग है $6$''. निर्धारित करें कि क्या $ A $ और $ B $
हैं \omterm{def:g12:condprob:indep}{स्वतंत्र}, फिर चाहे $ B $ और $ C $ हैं।
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:condprob:4}
एक कारखाने में क्रमशः उत्पादन करने वाली तीन मशीनें हैं $50\%$, $30\%$ और
$20\%$ कुल उत्पादन का, दोष दरों के साथ $1\%$, $2\%$ और $4\%$.
\begin{enumerate}
  \item उत्पादन का कितना भाग दोषपूर्ण है?
  \item यादृच्छिक रूप से चुनी गई वस्तु दोषपूर्ण है। क्या है \omterm{def:g10:proba:distribution}{संभावना} यह
        तीसरी मशीन से आया?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:condprob:5}
में \cref{ex:g12:condprob:test}, किस बीमारी का प्रचलन है $ p $ (के बजाय
$1\%$) एक सकारात्मक परीक्षण होगा \omterm{def:g11:stat:mean}{अर्थ} कम से कम ए $90\%$ बीमारी की \omterm{def:g10:proba:distribution}{संभावना}? हल करो
असमानता और टिप्पणी.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:condprob:6}
एक पक्षपाती सिक्का शीर्ष के साथ उतरता है \omterm{def:g10:proba:distribution}{संभावना} $ p \in \intoo{0}{1}$. यह है
तीन बार उछाला गया, उछाला गया \omterm{def:g12:condprob:indep}{स्वतंत्र}.
\begin{enumerate}
  \item की गणना करें \omterm{def:g10:proba:distribution}{संभावना} की \omterm{def:g11:prob:model}{आयोजन} $ E $ = ``बिल्कुल दो सिर''।
  \item जिसके लिए $ p $ है $\P(E)$ \omterm{def:g10:functions:extrema}{अधिकतम}?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g12:condprob:7}
\emph{(मोंटी हॉल।)} एक पुरस्कार तीन दरवाजों में से एक के पीछे छिपा है। आप एक चुनें
दरवाज़ा; मेज़बान, जो जानता है कि पुरस्कार कहाँ है, दोनों में से एक खोलता है
शेष दरवाजे, हमेशा एक खाली दिखाते हैं (यदि दोनों हों तो यादृच्छिक रूप से चुनें)।
खाली हैं), और आपको दूसरे बंद दरवाजे पर स्विच करने की पेशकश करता है। उपयोग करना
बेयस का सूत्र, गणना करें \omterm{def:g10:proba:distribution}{संभावना} यदि आप स्विच करते हैं तो जीतने की, और यदि
आप ऐसा नहीं करते.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g12:condprob:8}
एक सूचना चैनल बिट्स प्रसारित करता है। प्रत्येक बिट को शोर के साथ फ़्लिप किया जाता है
\omterm{def:g10:proba:distribution}{संभावना} $\varepsilon = 0.1$, \omterm{def:g12:condprob:indep}{स्वतंत्र} रूप से। थोड़ी सुरक्षा के लिए भेजा जाता है
तीन बार और बहुमत से डिकोड किया गया।
\begin{enumerate}
  \item की गणना करें \omterm{def:g10:proba:distribution}{संभावना} कि डिकोड किया गया बिट गलत है।
  \item प्राप्त शब्द है $101$. क्या है \omterm{def:g10:proba:distribution}{संभावना} वह भेजा गया
        बिट था $1$? (मान लीजिए $0$ और $1$ बराबर के साथ भेजा जाता है \omterm{def:g10:proba:distribution}{संभावना}.)
\end{enumerate}
\end{exercise}

\section{समस्या: डॉक्टर, जज और स्पैम फ़िल्टर}

\begin{problem}[{सप्ताहांत समस्या --- बेयस का फार्मूला विरुद्ध बेस-रेट भ्रांति: क्यों ए $99\,\%$ परीक्षण का मतलब हो सकता है $2\,\%$, क्यों ए दस लाख में से एक मैच किसी को दोषी नहीं ठहराता, और आपका मेलबॉक्स कैसा है सोचता है}]
\label{pb:g12:condprob:1}
एक परीक्षण जो पकड़ लेता है $99\,\%$ कितने मामले सकारात्मक आते हैं ---
और सच \omterm{def:g10:proba:distribution}{संभावना} रोग का विवरण नीचे है $2\,\%$. एक डीएनए
दस लाख में एक व्यक्ति से मिलती जुलती प्रोफ़ाइल मिलती है --- और
संदिग्ध शायद निर्दोष है. दोनों कथन सही हैं, दोनों
डॉक्टरों और अदालतों द्वारा गलत निर्णय लिया गया है, और दोनों का निपटारा हो गया है
इस अध्याय की एक पंक्ति से: बेयस का सूत्र
(\cref{thm:g12:condprob:bayes}). यह समस्या इसे कलशों पर प्रशिक्षित करती है,
फिर क्लिनिक, कोर्ट रूम और इनबॉक्स में।

\medskip
\noindent\textbf{Part I --- Fluency.}
\begin{enumerate}
  \item एक निष्पक्ष सिक्का कलश I को चुनता है ($2$ लाल, $1$ नीला) या कलश II
        ($1$ लाल, $3$ नीला), फिर एक गेंद निकाली जाती है। गणना करना
        $\P(\text{red})$ कुल के नियम के साथ \omterm{def:g10:proba:distribution}{संभावना}
        (\cref{thm:g12:condprob:total}).
  \item गेंद लाल है: गणना करें $\P(\text{urn I} \mid
        \text{red})$.
  \item वन डाई रोल: हैं घटनाएँ ``सम'' और
        ``$\leq 4$'' \omterm{def:g12:condprob:indep}{स्वतंत्र}
        (\cref{def:g12:condprob:indep})? और ``सम'' और
        ``$\leq 3$''?
  \item दो \omterm{def:g12:condprob:indep}{स्वतंत्र} प्रत्येक ट्रिगर के साथ धूम्रपान अलार्म
        \omterm{def:g10:proba:distribution}{संभावना} $0.9$ आग में. का उपयोग करते हुए
        \cref{prop:g12:condprob:indepcompl}, गणना करें
        \omterm{def:g10:proba:distribution}{संभावना} वह कम से कम एक ट्रिगर है।
  \item प्रतिस्थापन के बिना दो कार्ड: गणना करें
        $\P(\text{two kings})$ गुणन नियम के साथ
        (\cref{prop:g12:condprob:product}).
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The doctor.}
एक बीमारी असर करती है $1$ व्यक्ति में $1\,000$. परीक्षण से पता चलता है
$99\,\%$ बीमार (संवेदनशीलता) की लेकिन आग भी लगती है $5\,\%$ का
स्वस्थ (झूठी सकारात्मक)।
\begin{enumerate}[resume]
  \item \omterm{def:g12:exp:ln}{प्राकृतिक} आवृत्तियाँ: कल्पना कीजिए $100\,000$ लोग। कैसे
        कितने बीमार, कितने स्वस्थ? कितनी सकारात्मकताएँ
        क्या प्रत्येक समूह उत्पादन करता है, और उसमें कितनी सकारात्मकताएँ हैं
        सब?
  \item गिनती से, गणना करें
        $\P(\text{sick} \mid \text{positive})$. को अवशोषित करें
        सदमा.
  \item बेयस सूत्र के साथ इसकी पुनः गणना करें और जाँच करें
        समझौता.
  \item सर्वेक्षणों में, अधिकांश चिकित्सक ``के बारे में'' उत्तर देते हैं
        $99\,\%$''. दो सशर्त संभावनाओं के नाम बताइए
        भ्रमित होना, और समझाना कि छोटी आधार दर क्यों है
        सच्चा उत्तर देता है।
  \item एक दूसरा, \omterm{def:g12:condprob:indep}{स्वतंत्र} परीक्षण भी सकारात्मक आता है।
        अद्यतन: ले रहा हूँ $1.9\,\%$ नए पूर्व के रूप में, गणना करें
        $\P(\text{sick} \mid \text{two positives})$. क्या है?
        साक्ष्य \emph{संचय} कर रहा है?
  \item गणना करना $\P(\text{sick} \mid \text{positive})$ यदि
        एक ही परीक्षण का उपयोग केवल उच्च जोखिम वाले समूह पर किया जाता है
        व्यापकता $5\,\%$. सार्वजनिक-स्वास्थ्य नैतिकता का निष्कर्ष निकालें
        दुर्लभ स्थितियों की सामूहिक जांच के बारे में।
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The judge.}
\begin{enumerate}[resume]
  \item अपराध-स्थल का डीएनए प्रोफ़ाइल एक व्यक्ति से मेल खाता है
        मिलियन. डेटाबेस में खोज करने पर संदिग्ध का पता चला
        शहर का $10$ मिलियन निवासी, माचिस। अपेक्षित
        शहर में निर्दोष मैचों की संख्या? केवल दिया गया
        मिलान, क्या है $\P(\text{innocent} \mid
        \text{match})$, मोटे तौर पर?
  \item उस भ्रांति का नाम बताइए --- जो \omterm{def:g10:proba:distribution}{संभावना} किया
        अभियोजक बोली, अदालत को किसकी ज़रूरत थी? ---
        और यह प्रश्न 9 के बिल्कुल समानांतर है।
  \item  स्वतंत्रता जाल: एक परिवार में दो दुर्लभ त्रासदियाँ
        एक बार ``एक इंच'' में वर्गित किया गया था $73$ मिलियन'' द्वारा
        उनके साथ ऐसा व्यवहार करना \omterm{def:g12:condprob:indep}{स्वतंत्र}, एक निर्दोष को दोषी ठहराना
        माँ (सैली क्लार्क मामला)। यदि घटनाएँ साझा करें
        पारिवारिक जोखिम कारक ताकि दूसरा, दिया जा सके
        पहला, है $5$ आधार दर से कई गुना अधिक \omterm{def:g10:proba:distribution}{संभावना},
        परिमाण के क्रम की पुनः गणना करें --- और बताएं
        संभावनाओं को गुणा करने के बारे में पाठ।
  \item बचाव पक्ष भी धोखा दे सकता है: ``दस निर्दोष मेल खाते हैं, इसलिए
        मैच कुछ भी साबित नहीं करता'' बाकी सब चीजों को नजरअंदाज कर देता है।
        बेयस का सूत्र किससे संबंधित है? \emph{अनेक} टुकड़े
        साक्ष्य का, और प्रत्येक वाक्य को क्या होना चाहिए
        जूरी सदस्य को पता है?
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The spam filter.}
\begin{enumerate}[resume]
  \item आपका मेल है $40\,\%$ अवांछित ईमेल। शब्द ``लॉटरी''
        में प्रकट होता है $15\,\%$ स्पैम का और $0.5\,\%$ ईमानदार का
        मेल. एक संदेश में यह शामिल है: गणना करें
        $\P(\text{spam} \mid \text{lottery})$.
  \item त्रुटि सुधार पुनरावृत्ति
        (\cref{exo:g12:condprob:8} जारी): फ्लिप के साथ
        \omterm{def:g10:proba:distribution}{संभावना} $\varepsilon = 0.1$, का बहुमत डिकोडिंग
        \emph{पाँच} प्रतिलिपियाँ किस चीज़ से विफल हो जाती हैं \omterm{def:g10:proba:distribution}{संभावना}?
        तीन-कॉपी कोड से तुलना करें।
  \item थोड़ा पार $4$ \omterm{def:g12:condprob:indep}{स्वतंत्र} शोर रिले, प्रत्येक
        इसके साथ फ़्लिप करना \omterm{def:g10:proba:distribution}{संभावना} $0.1$: यह सही आता है
        बिल्कुल तब जब फ़्लिप की संख्या सम हो। उसका हिसाब लगाओ
        \omterm{def:g10:proba:distribution}{संभावना} दो बार: द्विपद पदों का योग करके, और
        सुरुचिपूर्ण सूत्र द्वारा
        $\frac{1 + (1 - 2\varepsilon)^4}{2}$.
  \item वास्तविक स्पैम फ़िल्टर कई शब्दों के प्रमाण को कई गुना बढ़ा देते हैं
        मानो \omterm{def:g12:condprob:indep}{स्वतंत्र} (``बेवकूफ'' बेयस)। ऐसा क्यों है?
        ``जीत'' और जैसे शब्दों के लिए धारणा गलत है
        ``लॉटरी'' --- और फ़िल्टर अच्छा काम क्यों करता है
        वैसे भी? (एक या दो वाक्य।)
  \item समापन --- विश्वास के \omterm{def:g12:seq:arith-geom}{अंकगणित} के रूप में बेयस: पूर्व
        $\times$ \omterm{def:g10:proba:distribution}{संभावना}, पुनर्सामान्यीकरण, पश्च। पुनः चलाएँ
        एक में तीन फैसले (क्लिनिक, कोर्ट रूम, इनबॉक्स)।
        प्रत्येक को वाक्य दें, फिर इस समस्या के तीन जाल सूचीबद्ध करें
        उजागर: भूली गई आधार दरें, नकली स्वतंत्रता, और
        साक्ष्य एक समय में एक टुकड़ा पढ़ें।
\end{enumerate}
\end{problem}
