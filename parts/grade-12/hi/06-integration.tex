\chapter{एकीकरण}\label{ch:g12:integ}

एकीकरण एक साथ दो प्रश्नों का उत्तर देता है: वक्र के नीचे का क्षेत्रफल क्या है,
और कोई इसे कैसे पुनर्प्राप्त कर सकता है? \omterm{def:g11:func:function}{समारोह} इसके परिवर्तन की दर से? मौलिक
कैलकुलस के प्रमेय में कहा गया है कि ये वही प्रश्न हैं --- सबसे गहरे
और सत्रहवीं सदी के गणित की सबसे उपयोगी खोज।

\section{एक सतत कार्य का अभिन्न अंग}

\begin{definition}[एक सकारात्मक कार्य का अभिन्न अंग]\label{def:g12:integ:area}
होने देना $f $ होना निरंतर और सकारात्मक पर $\intcc{a}{b}$.
\emph{अभिन्न}\index{अभिन्न}
\[
\int_a^b f(x)\,\dd x
\]
के वक्र से घिरे क्षेत्र का क्षेत्रफल, क्षेत्रफल की इकाइयों में है $ f $,
द $ x $-अक्ष, और ऊर्ध्वाधर रेखाएँ $ x = a $ और $ x = b $.
\end{definition}

\begin{proposition}[अभिन्न के गुण]\label{prop:g12:integ:properties}
होने देना $ f, g $ होना निरंतर एक पर \omterm{def:g10:numbers:interval}{अंतराल} युक्त $ a, b, c $, और
$\lambda \in \R $.
\begin{enumerate}
  \item \emph{रैखिकता:}
        $\displaystyle\int_a^b (f + \lambda g) = \int_a^b f + \lambda \int_a^b g $.
  \item \emph{चैसल्स संबंध:}
        $\displaystyle\int_a^c f = \int_a^b f + \int_b^c f $.
  \item \emph{सकारात्मकता:} अगर $ a \leq b $ और $ f \geq 0$ पर $\intcc{a}{b}$,
        तब $\displaystyle\int_a^b f \geq 0$; अगर $ f \leq g $, तब
        $\displaystyle\int_a^b f \leq \int_a^b g $.
\end{enumerate}
\end{proposition}

\begin{proof}
क्षेत्र की व्याख्या (क्षेत्र \omterm{def:g10:vectors:sum}{जोड़}ें) से चैसल्स और सकारात्मकता तत्काल होती है
जब क्षेत्रों को एक साथ रखा जाता है; धनात्मक ऊंचाई वाले क्षेत्र का मान धनात्मक होता है
क्षेत्र) तुलना सकारात्मकता लागू करने के बाद होती है $ g - f $. रैखिकता है
सकारात्मक के योग के लिए सहज रूप से स्पष्ट कार्य (क्षेत्रों को ढेर करें) और है
विश्वविद्यालय में कठोरता से \omterm{def:g11:vect:det}{सिद्ध} किया गया; \emph{इसे यहां स्वीकार किया गया है}.
\end{proof}

\section{कलन का मौलिक प्रमेय}

\begin{definition}[प्राचीन]\label{def:g12:integ:primitive}
ए \emph{प्राचीन}\index{प्राचीन} (या प्रतिव्युत्पन्न) का $ f $ एक पर \omterm{def:g10:numbers:interval}{अंतराल}
$ I $ एक है \omterm{def:g12:deriv:derivative}{विभेदक} \omterm{def:g11:func:function}{समारोह} $ F $ पर $ I $ ऐसा है कि $ F' = f $.
\end{definition}

\begin{proposition}\label{prop:g12:integ:primunique}
अगर $ F $ एक है \omterm{def:g12:integ:primitive}{प्राचीन} का $ f $ एक पर \omterm{def:g10:numbers:interval}{अंतराल} $ I $, द पुरातन का $ f $ पर
$ I $ बिल्कुल वही हैं कार्य $ F + c $, $ c \in \R $. दिया गया $ x_0 \in I $ और
$ y_0 \in \R $, एक अनोखा है \omterm{def:g12:integ:primitive}{प्राचीन} साथ $ F(x_0) = y_0$.
\end{proposition}

\begin{proof}
अगर $ G' = F' = f $, तब $(G - F)' = 0$ पर \omterm{def:g10:numbers:interval}{अंतराल} $ I $, इसलिए $ G - F $ है
स्थिर.\footnote{That a \omterm{def:g11:func:function}{समारोह} with zero यौगिक on an \omterm{def:g10:numbers:interval}{अंतराल} is
constant follows from \cref{thm:g12:deriv:variations}: it is both \omterm{def:g12:seq:monotonic}{की बढ़ती}
and \omterm{def:g10:functions:variations}{घटते}.} इसके विपरीत प्रत्येक $ F + c $ एक है \omterm{def:g12:integ:primitive}{प्राचीन}. स्थिति
$ F(x_0) = y_0$ स्थिरांक को पिन करता है।
\end{proof}

\begin{theorem}[कलन का मौलिक प्रमेय]\label{thm:g12:integ:ftc}
\index{कैलकुलस का मौलिक प्रमेय}
होने देना $ f $ होना निरंतर एक पर \omterm{def:g10:numbers:interval}{अंतराल} $ I $ और $ a \in I $. \omterm{def:g11:func:function}{समारोह}
\[
F \colon x \longmapsto \int_a^x f(t)\,\dd t
\]
है \omterm{def:g12:integ:primitive}{प्राचीन} का $ f $ पर $ I $ पर लुप्त हो रहा है $ a $. नतीजतन, किसी के लिए
\omterm{def:g12:integ:primitive}{प्राचीन} $ G $ का $ f $ और $ a, b \in I $:
\[
\int_a^b f(t)\,\dd t = \bigl[G(t)\bigr]_a^b = G(b) - G(a).
\]
\end{theorem}

{\small प्रमाण का विचार: वेतन वृद्धि $ F(x+h) - F(x)$ क्षेत्र है
ऊँचाई के आयतों के बीच दबी हुई संकीर्ण पट्टी (नारंगी) की

\begin{proof}[प्रमाण कब $ f $ बढ़ रहा है]
ठीक करो $ x \in I $ और $ h > 0$ साथ $ x + h \in I $. चैसल्स द्वारा,
\[
F(x+h) - F(x) = \int_x^{x+h} f(t)\,\dd t .
\]
तब से $ f $ है \omterm{def:g12:seq:monotonic}{की बढ़ती}, $ f(x) \leq f(t) \leq f(x+h)$ के लिए
$ t \in \intcc{x}{x+h}$, और की तुलना से \omterm{def:g12:integ:area}{अभिन्न} (द \omterm{def:g12:integ:area}{अभिन्न} एक का
स्थिर $ c $ एक से अधिक \omterm{def:g10:numbers:interval}{अंतराल} लम्बाई का $ h $ है $ ch $):
\[
h\,f(x) \leq F(x+h) - F(x) \leq h\,f(x+h),
\]
इसलिए
\[
f(x) \leq \frac{F(x+h) - F(x)}{h} \leq f(x+h).
\]
जैसा $ h \to 0^+$, $ f(x+h) \to f(x)$ द्वारा निरंतरता, और निचोड़ प्रमेय
देता है कि अंतर भागफल की ओर प्रवृत्त होता है $ f(x)$; मामला $ h < 0$ है
सममित. इस तरह $ F' = f $, और $ F(a) = 0$. सामान्य (गैर monotonic)
निरंतर मामला विश्वविद्यालय में साबित हुआ है।

अंततः, यदि $ G $ कोई है \omterm{def:g12:integ:primitive}{प्राचीन}, $ G = F + c $
(\cref{prop:g12:integ:primunique}), इसलिए
$ G(b) - G(a) = F(b) - F(a) = \int_a^b f $.
\end{proof}

\begin{example}
$\displaystyle\int_0^1 x^2\,\dd x = \left[\frac{x^3}{3}\right]_0^1 =
\frac13$: के अंतर्गत क्षेत्र \omterm{def:g11:quad:parabola}{परवलय} इकाई वर्ग का एक तिहाई है, जैसे
आर्किमिडीज़ जानता था.
\end{example}

($ u $ एक को दर्शाता है \omterm{def:g12:deriv:derivative}{विभेदक} \omterm{def:g11:func:function}{समारोह}, $ c $ एक मनमाना स्थिरांक):

\begin{method}[स्वरूप को पहचानना $ u' \times (\text{something in } u)$]\label{met:g12:integ:uprime}
किसी उत्पाद को एकीकृत करने के लिए, एक ऐसे कारक की तलाश करें जो है यौगिक एक आंतरिक का
\omterm{def:g11:func:function}{समारोह} $ u $, गुणात्मक स्थिरांक तक। उदाहरण के लिए में
$\int_0^1 x\,\eu^{x^2}\dd x $, कारक $ x $ है $\frac12 (x^2)'$:
\[
\int_0^1 x\,\eu^{x^2}\dd x = \frac12\left[\eu^{x^2}\right]_0^1
= \frac{\eu - 1}{2}.
\]
\end{method}

\begin{theorem}[भागों द्वारा एकीकरण]\label{thm:g12:integ:ibp}
\index{भागों द्वारा एकीकरण}
होने देना $ u, v $ होना \omterm{def:g12:deriv:derivative}{विभेदक} पर $\intcc{a}{b}$ साथ निरंतर डेरिवेटिव.
तब
\[
\int_a^b u'(t)\,v(t)\,\dd t
= \bigl[u(t)\,v(t)\bigr]_a^b - \int_a^b u(t)\,v'(t)\,\dd t .
\]
\end{theorem}

\begin{proof}
उत्पाद नियम देता है $(uv)' = u'v + uv'$; दोनों पक्षों को एकीकृत करना
$\intcc{a}{b}$ और बाईं ओर के लिए मौलिक प्रमेय का उपयोग करना
पैदावार $\bigl[uv\bigr]_a^b = \int_a^b u'v + \int_a^b uv'$.
\end{proof}

\begin{example}\label{ex:g12:integ:ibp}
$\displaystyle\int_0^1 t\,\eu^{t}\,\dd t $: लेना $ u' = \eu^t $, $ v = t $, इसलिए
$ u = \eu^t $, $ v' = 1$:
\[
\int_0^1 t\,\eu^t\,\dd t = \bigl[t\,\eu^t\bigr]_0^1 - \int_0^1 \eu^t\,\dd t
= \eu - (\eu - 1) = 1 .
\]
\end{example}

\section{अनुप्रयोग}

\begin{definition}[औसत मूल्य]\label{def:g12:integ:mean}
 \emph{औसत मूल्य}\index{औसत मूल्य} एक का निरंतर \omterm{def:g11:func:function}{समारोह} $ f $ पर
$\intcc{a}{b}$ ($ a<b $) है
\[
\mu = \frac{1}{b-a}\int_a^b f(t)\,\dd t .
\]
\end{definition}

\begin{proposition}\label{prop:g12:integ:meanbounds}
अगर $ m \leq f \leq M $ पर $\intcc{a}{b}$, तब $ m \leq \mu \leq M $.
\end{proposition}

\begin{proof}
असमानताओं को एकीकृत करें $ m \leq f(t) \leq M $ ऊपर $\intcc{a}{b}$ और
द्वारा विभाजित करें $ b - a > 0$.
\end{proof}

\begin{method}[दो वक्रों के बीच का क्षेत्र]\label{met:g12:integ:areabetween}
अगर $ f \geq g $ पर $\intcc{a}{b}$, दोनों वक्रों के बीच का क्षेत्रफल है
$\int_a^b \bigl(f(x) - g(x)\bigr)\dd x $. यदि वक्र क्रॉस करते हैं, तो विभाजित करें
\omterm{def:g10:numbers:interval}{अंतराल} क्रॉसिंग बिंदुओं पर और एकीकृत करें $\abs{f - g}$ टुकड़ा दर टुकड़ा.
\end{method}

\begin{exercise}[$\star\star $]\label{exo:g12:integ:6}

\begin{exercise}[$\star\star $]\label{exo:g12:integ:7}
पहले के दौरान ट्रेन की गति (मीटर/सेकेंड में)। $100$ प्रस्थान के कुछ सेकंड बाद
द्वारा प्रतिरूपित किया गया है $ v(t) = 30\bigl(1 - \eu^{-t/50}\bigr)$. दूरी की गणना करें
इस दौरान यात्रा की $100$ सेकंड, और ट्रेन की औसत गति से अधिक
\omterm{def:g10:numbers:interval}{अंतराल}.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g12:integ:8}
के लिए $ n \geq 1$, होने देना $ S_n = \dfrac1n \displaystyle\sum_{k=1}^{n} \frac{1}{1 + k/n}$.
\begin{enumerate}
  \item व्याख्या $ S_n $ एक क्षेत्र का अनुमान लगाने वाले आयतों के एक क्षेत्र के रूप में
        के वक्र के नीचे $ t \mapsto \frac{1}{1+t}$ पर $\intcc{0}{1}$.
  \item की एकरसता का उपयोग करना $ t \mapsto \frac{1}{1+t}$, बताते हैं कि
        \[
        S_n \leq \int_0^1 \frac{\dd t}{1+t} \leq S_n + \frac1n\left(1 - \frac12\right),
        \]
        और निष्कर्ष निकालें $\lim\limits_{n\to+\infty} S_n = \ln 2$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g12:integ:9}
\emph{(वालिस \omterm{def:g12:integ:area}{अभिन्न}.)} के लिए $ n \in \N $, होने देना
$ W_n = \displaystyle\int_0^{\pi/2} \sin^n t\,\dd t $.
\begin{enumerate}
  \item गणना करना $ W_0$ और $ W_1$.
  \item लिखना $\sin^{n+2}t = \sin t \cdot \sin^{n+1} t $ और द्वारा एकीकृत किया जा रहा है
        भाग, वह दिखाओ $ W_{n+2} = \dfrac{n+1}{n+2}\,W_n $.
  \item परिणाम निकालना $ W_2$, $ W_3$, $ W_4$ और दिखाओ $(W_n)$ है \omterm{def:g10:functions:variations}{घटते} और
        सकारात्मक.
\end{enumerate}
\end{exercise}

\section{समस्या: आर्किमिडीज़ मशीन के विरुद्ध}

\begin{problem}[{सप्ताहांत समस्या---परवलय के अंतर्गत क्षेत्र, बाईस शताब्दियों में तीन तरीकों से गणना की गई, और एक क्षेत्र के रूप में लघुगणक की गुप्त पहचान}]
\label{pb:g12:integ:1}
लगभग 240 ईसा पूर्व, आर्किमिडीज़ ने \omterm{def:g11:quad:parabola}{परवलय}िक के सटीक क्षेत्र की गणना की
खंड --- बिना \omterm{def:g10:coordgeom:system}{COORDINATES}, बिना सीमा के, बिना बीजगणित के।
उन्नीस शताब्दियों के बाद, रीमैन की आयतों ने इसे क्रूरता से फिर से बनाया
निचोड़ना; और कैलकुलस का मौलिक प्रमेय
(\cref{thm:g12:integ:ftc}) अब इसे एक पंक्ति में करता है। यह समस्या
सभी तीन मैच खेलता है, फिर खुलासा करने के लिए उसी मशीन का उपयोग करता है
क्या लोगारित्म वास्तव में है: स्केलिंग समरूपता वाला एक क्षेत्र।

\medskip
\noindent\textbf{Part I --- Fluency.}
\begin{enumerate}
  \item गणना करना $\displaystyle\int_0^1 (3x^2 - 2x + 1)\,\dd x $,
        \ $\displaystyle\int_1^{\eu} \frac{\dd x}{x}$, \ और
        $\displaystyle\int_0^{\pi/2} \cos x\,\dd x $.
  \item स्पॉट करें $ u'u $-रूप
        (\cref{met:g12:integ:uprime}):
        $\displaystyle\int_0^1 x\,\eu^{x^2}\dd x $ और
        $\displaystyle\int_0^1 \frac{2x}{x^2 + 1}\,\dd x $.
  \item भागों द्वारा (\cref{thm:g12:integ:ibp}):
        $\displaystyle\int_0^1 x\,\eu^x \dd x $ और
        $\displaystyle\int_1^{\eu} \ln x\,\dd x $.
  \item की गणना करें \omterm{def:g12:integ:mean}{औसत मूल्य}
        (\cref{def:g12:integ:mean}) का $\sin $ ऊपर
        $\intcc{0}{\pi}$ --- और ध्यान दें कि यह है \emph{नहीं}
        $\frac12$.
  \item रेखा के बीच के क्षेत्रफल की गणना करें $ y = x $ और यह
        \omterm{def:g11:quad:parabola}{परवलय} $ y = x^2$ ऊपर $\intcc{0}{1}$
        (\cref{met:g12:integ:areabetween}).
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The \omterm{def:g11:quad:parabola}{परवलय}, three ways.}
\begin{enumerate}[resume]
  \item क्षेत्र के लिए रीमैन की स्थापना $ y = x^2$ पर
        $\intcc{0}{1}$: कम योग लिखें $ L_n $ और ऊपरी
        योग $ U_n $ ऊपर $ n $ समान आयत (जैसा कि)
        \cref{exo:g12:integ:8}).
  \item प्रेरण द्वारा \omterm{def:g11:vect:det}{सिद्ध} करें (\cref{pb:g12:seq:1}की मशीन)
        वर्गों का योग सूत्र
        \[
        1^2 + 2^2 + \dots + n^2 = \frac{n(n+1)(2n+1)}{6} .
        \]
  \item के लिए बंद प्रपत्र निकालें $ U_n $ और $ L_n $, उनकी गणना करें
        सामान्य सीमा, और निष्कर्ष निकालें: क्षेत्र है $\frac13$.
  \item अब मशीन: गणना करें
        $\int_0^1 x^2\,\dd x $ मौलिक प्रमेय द्वारा, में
        एक पंक्ति. प्रयासों की तुलना करें.
  \item आर्किमिडीज़ ने इसे अन्यथा कहा: \emph{एक \omterm{def:g11:quad:parabola}{परवलय}िक
        खंड है $\frac43$ इसके उत्कीर्ण त्रिभुज का}. के लिए
        खंड से काटा गया $ y = x^2$ तार जोड़ने से
        $(-1, 1)$ को $(1, 1)$: खंड के क्षेत्रफल की गणना करें
        एक \omterm{def:g12:integ:area}{अभिन्न}, उत्कीर्ण त्रिभुज का क्षेत्रफल (शीर्ष पर)
        शिखर $(0, 0)$), और मास्टर अनुपात को सत्यापित करें।
  \item आर्किमिडीज़ की अपनी विधि: खंड को बड़े से भरें
        त्रिकोण $ T $, फिर कुल मिलाकर दो त्रिभुज
        $\frac T4$, फिर कुल चार $\frac{T}{16}$, और इसी तरह। ज्यामितीय श्रृंखला का योग करें और पुनर्प्राप्त करें
        $\frac43 T $ --- अंतहीन रूप से काटी गई चॉकलेट बार
        मिडिल स्कूल वॉल्यूम, एक ग्रीक जियोमीटर द्वारा खाया गया।
  \item प्रत्येक एक वाक्य: थकावट (आर्किमिडीज़), निचोड़
        (रीमैन), भेदभाव-विरोधी (न्यूटन--लीबनिज) ---
        प्रत्येक को क्या चाहिए, और प्रत्येक क्या देता है?
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The लोगारित्म is an area.}
के लिए $ x > 0$, तय करना
$ A(x) = \displaystyle\int_1^x \frac{\dd t}{t}$.
\begin{enumerate}[resume]
  \item देना $ A(1)$ और $ A'(x)$
        (\cref{thm:g12:integ:ftc}), और निष्कर्ष निकालो $ A $ है
        बिलकुल \omterm{def:g12:exp:ln}{प्राकृतिक} का
        \cref{def:g12:exp:ln}.
  \item स्केलिंग चमत्कार: ठीक करें $ a > 0$ और अध्ययन
        $ g(x) = A(ax) - A(x)$. गणना करना $ g'$ (श्रृंखला नियम),
        उसका निष्कर्ष निकालो $ g $ स्थिर है, स्थिरांक का मूल्यांकन करें ---
        और कार्यात्मकता समाप्त करें \omterm{def:g10:algebra:equation}{समीकरण}
        \[
        \ln(ab) = \ln a + \ln b ,
        \]
        शुद्ध कलन द्वारा \omterm{def:g11:vect:det}{सिद्ध}: से क्षेत्र $1$ को $ ab $
        स्केल की गई प्रतियों में विभाजित हो जाता है।
  \item प्रश्न 14 से निष्कर्ष निकालें: $\ln(a^n) = n\ln a $ और
        $\ln\frac1a = -\ln a $.
  \item \Cref{exo:g12:integ:8} पढ़ता
        $\frac{1}{n+1} + \frac{1}{n+2} + \dots + \frac{1}{2n}$
        नीचे आयतों के रूप में $\frac{1}{1 + t}$: इस राशि की गणना करें
        के लिए $ n = 10$ (तीन दशमलव) और तुलना करें
        $\ln 2$. जो हार्मोनिक-स्वाद वाले योग, भिन्न शब्द
        शब्द के अनुसार, यहाँ एक क्षेत्र में \omterm{def:g12:seq:limit}{अभिसरण} होता है?
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Accumulation.}
\begin{enumerate}[resume]
  \item एक कार गति से तेज होती है $ v(t) = 3t^2$ मैसर्स के लिए
        $ t \in \intcc{0}{10}$. तय की गई दूरी की गणना करें,
        द \omterm{def:g11:stat:mean}{अर्थ} गति, और वह क्षण जिस पर
        तात्कालिक गति के बराबर होती है \omterm{def:g11:stat:mean}{अर्थ} रफ़्तार।
  \item ए $4$-मीटर रॉड में रैखिक घनत्व होता है
        $\rho(x) = 2 + x $ किग्रा/मी. इसके कुल द्रव्यमान और इसकी गणना करें
        द्रव्यमान का \omterm{pb:g11:quad:1}{केंद्र}
        $\dfrac{\int_0^4 x\,\rho(x)\,\dd x}{\int_0^4
        \rho(x)\,\dd x}$ --- \omterm{prop:g10:coordgeom:midpoint}{मध्य} का संतुलन बिंदु
        स्कूल वॉल्यूम का कार्डबोर्ड त्रिकोण, अंततः गणना की गई
        वज़न के साथ जो भिन्न होता है।
  \item वालिस का \omterm{def:g12:integ:area}{अभिन्न} (\cref{exo:g12:integ:9}): गणना करें
        $ W_0$, $ W_1$, और
        $ W_2 = \int_0^{\pi/2} \sin^2 t\,\dd t $ का उपयोग कर रहा हूँ
        का रैखिककरण \cref{pb:g12:trigo:1}. (उनकी अनंत
        विश्वविद्यालय के खंडों में, किसी उत्पाद पर सीढ़ी चढ़ना
        के लिए सूत्र $\pi $ और घंटी के सामान्यीकरण के लिए
        वक्र.)
  \item समापन --- एकीकरण के तीन पहलू: एक \emph{क्षेत्र}
        परिभाषा के अनुसार, एक \emph{संचय} दुनिया में
        (दूरी, द्रव्यमान), ए \emph{antiderivative} द्वारा
        मौलिक प्रमेय; और तीन नामों में इसका इतिहास.
        अध्याय के मोती के साथ बंद करें: कौन सा रोजमर्रा का बटन
        कैलकुलेटर पर गुप्त रूप से नीचे का क्षेत्र है
        $\frac1t$ ---और उस क्षेत्र ने कौन सी पहचान \omterm{def:g11:vect:det}{सिद्ध} की?
\end{enumerate}
\end{problem}
