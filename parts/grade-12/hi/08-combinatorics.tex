\chapter{कॉम्बिनेटरिक्स और काउंटिंग}\label{ch:g12:comb}

कॉम्बिनेटरिक्स बिना सूचीबद्ध किए गिनने की कला है। यह दो प्राथमिक हैं
\omterm{def:g11:vect:det}{सिद्ध}ांत --- असंयुक्त विकल्पों के आकार \omterm{def:g10:vectors:sum}{जोड़}ें, संख्याओं को गुणा करें
स्वतंत्र विकल्पों में से ---गिनने के लिए पर्याप्त है \omterm{def:g12:comb:tuples}{व्यवस्था}, क्रमपरिवर्तन
और एक परिमित समुच्चय के उपसमुच्चय, और द्विपद प्रमेय में परिणत होते हैं।

\section{गिनती के दो सिद्धांत}

हम लिखते हैं $\abs{E}$ तत्वों की संख्या के लिए ( \emph{प्रमुखता}) का ए
परिमित समुच्चय $ E $.

\begin{proposition}[जोड़ सिद्धांत]\label{prop:g12:comb:addition}
यदि एक परिमित समुच्चय $ E $ उपसमूहों में विभाजित किया गया है $ A_1, \dots, A_k $ (\omterm{def:g10:vectors:sum}{जोड़}ी में
विच्छेद, के साथ \omterm{def:g10:numbers:interunion}{मिलन} $ E $), तब
\[
\abs{E} = \abs{A_1} + \abs{A_2} + \dots + \abs{A_k}.
\]
\end{proposition}

\begin{proposition}[गुणन सिद्धांत]\label{prop:g12:comb:multiplication}
यदि किसी वस्तु का निर्माण क्रमबद्ध तरीके से किया गया है $ k $ विकल्प, के साथ $ n_1$ के लिए विकल्प
पहली पसंद और, \emph{पिछले विकल्प जो भी हों}, $ n_i $ के लिए विकल्प
द $ i $-वें, तो निर्मित वस्तुओं की संख्या है
$ n_1 \times n_2 \times \dots \times n_k $.
\end{proposition}

\begin{proof}
दोनों कथन प्रेरण द्वारा \omterm{def:g11:vect:det}{सिद्ध} होते हैं $ k $; मामला $ k = 2$ का
दूसरा, एक आयताकार सरणी को पंक्तियों द्वारा गिनने के बराबर है।
\end{proof}

\begin{example}
एक रेस्तरां 4 स्टार्टर, 6 मुख्य, 3 मिठाइयाँ प्रदान करता है: $4 \times 6 \times 3 =
72$ अलग-अलग तीन-कोर्स भोजन।
\end{example}

\section{टुपल्स, क्रमपरिवर्तन, फैक्टोरियल}

\begin{definition}[$ k $-ट्यूपल्स]\label{def:g12:comb:tuples}
ए \emph{$ k $-टुपल} एक सेट का $ E $ एक आदेशित सूची है $(x_1, \dots, x_k)$ का
के तत्व $ E $, दोहराव की अनुमति है। ए $ k $-टुपल का \emph{विशिष्ट}
तत्व एक है \emph{व्यवस्था}\index{व्यवस्था} का $ k $ घटक $ E $.
\end{definition}

\begin{proposition}\label{prop:g12:comb:tuples}
होने देना $\abs E = n $. की संख्या $ k $-के टुपल्स $ E $ है $ n^k $. की संख्या
\omterm{def:g12:comb:tuples}{व्यवस्था} का $ k $ घटक $ E $ ($0 \leq k \leq n $) है
\[
n(n-1)(n-2)\cdots(n-k+1) = \frac{n!}{(n-k)!},
\]
कहाँ $ n! = 1 \times 2 \times \dots \times n $ (और $0! = 1$) है
\emph{कारख़ाने का}\index{कारख़ाने का} का $ n $.
\end{proposition}

\begin{proof}
गुणन \omterm{def:g11:vect:det}{सिद्ध}ांत: a के लिए $ k $-टुपल वहाँ हैं $ n $ प्रत्येक पर विकल्प
द $ k $ कदम; एक के लिए \omterm{def:g12:comb:tuples}{व्यवस्था}, $ n $ के लिए विकल्प $ x_1$, तब $ n - 1$ के लिए
$ x_2$ (एक तत्व का उपयोग किया जाता है), \dots, $ n - k + 1$ के लिए $ x_k $.
\end{proof}

\begin{definition}[परिवर्तन]\label{def:g12:comb:permutation}
ए \emph{परिवर्तन}\index{परिवर्तन} का $ E $ एक \omterm{def:g12:comb:tuples}{व्यवस्था} के सभी $ n $
घटक $ E $: का एक आदेश $ E $. द्वारा
\cref{prop:g12:comb:tuples} (मामला $ k = n $), के क्रमपरिवर्तन की संख्या
$ n $-तत्व सेट है $ n!$.
\end{definition}

\begin{example}
पाँच धावक एक दौड़ पूरी कर सकते हैं $5! = 120$ अलग-अलग आदेश. की संख्या
संभावित पोडियम (पहले तीन स्थान) है $5 \times 4 \times 3 = 60$.
\end{example}

\section{संयोजन और द्विपद गुणांक}

\begin{definition}[युग्म]\label{def:g12:comb:combination}
ए \emph{संयोजन}\index{संयोजन} का $ k $ घटक $ E $ एक उपसमुच्चय है
का $ E $ साथ $ k $ तत्व (कोई क्रम नहीं, कोई पुनरावृत्ति नहीं)। उनका नंबर लिखा हुआ है
$\dbinom{n}{k}$\index{द्विपद गुणांक}, पढ़ें ``$ n $ चुनना $ k $''.
\end{definition}

\begin{theorem}\label{thm:g12:comb:binomformula}
के लिए $0 \leq k \leq n $:
\[
\binom{n}{k} = \frac{n!}{k!\,(n-k)!} .
\]
\end{theorem}

\begin{proof}
गिनें \omterm{def:g12:comb:tuples}{व्यवस्था} का $ k $ घटक $ E $ दो तरह से. सीधे:
$\frac{n!}{(n-k)!}$. वैकल्पिक रूप से, पहले अंतर्निहित उपसमुच्चय चुनें
($\binom nk $ तरीके), फिर इसे ऑर्डर करें ($ k!$ तौर तरीकों); गुणन \omterm{def:g11:vect:det}{सिद्ध}ांत
देता है $\binom{n}{k}\,k!$. बराबरी करना,
$\binom nk = \frac{n!}{k!(n-k)!}$.
\end{proof}

\begin{proposition}[बुनियादी पहचान]\label{prop:g12:comb:identities}
के लिए $0 \leq k \leq n $:
\[
\binom{n}{0} = \binom{n}{n} = 1, \qquad
\binom{n}{1} = n, \qquad
\binom{n}{k} = \binom{n}{n-k},
\]
और \emph{पास्कल का नियम}\index{पास्कल का नियम}: के लिए $1 \leq k \leq n-1$,
\[
\binom{n}{k} = \binom{n-1}{k-1} + \binom{n-1}{k}.
\]
\end{proposition}

\begin{proof}
समरूपता $\binom nk = \binom{n}{n-k}$ लेने के कारण धारण करता है \omterm{def:g10:proba:operations}{पूरक}
माचिस $ k $-तत्व उपसमुच्चय के साथ $(n-k)$-तत्व उपसमुच्चय, एक के लिए एक। के लिए
\omterm{prop:g12:comb:identities}{पास्कल का नियम}, एक तत्व को ठीक करें $ a \in E $ और क्रमबद्ध करें $ k $-तत्व उपसमुच्चय में
जिनमें शामिल हैं $ a $ --- निकटवर्ती द्वारा प्राप्त किया गया $ a $ एक को $(k-1)$-तत्व
का उपसमुच्चय $ E \setminus \{a\}$, जिनमें से हैं $\binom{n-1}{k-1}$ --- और
टालने वाले $ a $, जो हैं $ k $-तत्व उपसमुच्चय
$ E \setminus \{a\}$, क्रमांकन $\binom{n-1}{k}$. \omterm{def:g10:vectors:sum}{जोड़} द्वारा निष्कर्ष निकालें
\omterm{def:g11:vect:det}{सिद्ध}ांत.
\end{proof}

\omterm{prop:g12:comb:identities}{पास्कल का नियम} पंक्ति दर पंक्ति गुणांक उत्पन्न करता है --- \emph{\omterm{prop:g11:binom:pascal}{पास्कल का
त्रिकोण}}: प्रत्येक प्रविष्टि उसके ऊपर की दो प्रविष्टियों का योग है।

\begin{theorem}[द्विपद प्रमेय]\label{thm:g12:comb:binomial}
\index{द्विपद प्रमेय}
सभी के लिए $ a, b \in \R $ (या $\C $) और $ n \in \N $:
\[
(a+b)^n = \sum_{k=0}^{n} \binom{n}{k}\, a^{k}\, b^{\,n-k} .
\]
\end{theorem}

\begin{proof}
उत्पाद का \omterm{def:g10:algebra:expand}{विस्तार} करें $(a+b)(a+b)\cdots(a+b)$ ($ n $ कारक): का प्रत्येक पद
\omterm{def:g10:algebra:expand}{विस्तार} चुनता है $ a $ या $ b $ हर कारक में, उत्पादन $ a^k b^{n-k}$ कहाँ
$ k $ योगदान देने वाले कारकों की संख्या है $ a $. चुनने के तरीकों की संख्या
ये $ k $ बीच में कारक $ n $ है $\binom nk $, जो इसलिए है
का गुणांक $ a^k b^{n-k}$.
\end{proof}

\begin{corollary}\label{cor:g12:comb:sums}
$\displaystyle\sum_{k=0}^{n} \binom{n}{k} = 2^n $
\quad और \quad
$\displaystyle\sum_{k=0}^{n} (-1)^k\binom{n}{k} = 0$ \ ($ n \geq 1$).
\end{corollary}

\begin{proof}
लेना $ a = b = 1$, तब $ a = -1$, $ b = 1$ द्विपद प्रमेय में. पहला
पहचान का सीधा \omterm{def:g11:stat:mean}{अर्थ} यह भी है: ए $ n $-तत्व सेट है $2^n $ उपसमुच्चय
(प्रत्येक तत्व अंदर या बाहर है: गुणन \omterm{def:g11:vect:det}{सिद्ध}ांत), आकार के अनुसार क्रमबद्ध।
\end{proof}

\begin{method}[सही मॉडल चुनना]\label{met:g12:comb:model}
गिनती से पहले, दो प्रश्नों के उत्तर दें: \emph{क्या आदेश मायने रखता है?} और
\emph{क्या दोहराव की अनुमति है?}
\begin{center}
\begin{tabular}{l|cc}
 & order matters & order irrelevant\\
\hline
repetitions allowed & $ n^k $ (tuples) & (university)\\
no repetitions & $\frac{n!}{(n-k)!}$ (\omterm{def:g12:comb:tuples}{व्यवस्था}) & $\binom nk $ (subsets)
\end{tabular}
\end{center}
कलश से गेंदें निकालना: \emph{प्रतिस्थापन के साथ, क्रम में} $\to $ टुपल्स;
\emph{प्रतिस्थापन के बिना, क्रम में} $\to $ \omterm{def:g12:comb:tuples}{व्यवस्था}; \emph{एक मुट्ठी सब
एक बार में} $\to $ उपसमुच्चय
\end{method}

\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g12:comb:9}
\omterm{def:g11:vect:det}{सिद्ध} करना \emph{वेंडरमोंडे की पहचान}: के लिए $0 \leq k \leq m + n $,
\[
\binom{m+n}{k} = \sum_{j=0}^{k} \binom{m}{j}\binom{n}{k-j},
\]
गिनती करके $ k $-एक सेट के तत्व उपसमुच्चय को एक समूह में विभाजित किया जाता है $ m $ और ए
का समूह $ n $. उसका निष्कर्ष निकालो
$\displaystyle\sum_{j=0}^{n}\binom{n}{j}^{\!2} = \binom{2n}{n}$.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g12:comb:10}

\[
\sum_{k=1}^{n} k \binom{n}{k} = n\,2^{n-1}.
\]
(संकेत: या तो अंतर करें $(1+x)^n $, या उपयोग करें \cref{exo:g12:comb:7}.)
\end{exercise}

\section{समस्या: दो बार गिनने की कला}

\begin{problem}[{सप्ताहांत समस्या---सितारे और बार, विक्षिप्त टोपियाँ, और एक चीज़ को दो तरह से गिनने से पहचान साबित होती है}]
\label{pb:g12:comb:1}
कॉम्बिनेटरिक्स में सबसे गहरी चाल बेहद सरल है: गिनती
एक ही संग्रह को दो बार, दो अलग-अलग तरीकों से, और बराबर करें
उत्तर. यह समस्या मॉडलों का अभ्यास करती है
\cref{met:g12:comb:model}, अध्याय के पाठ्यक्रम में एक तकनीक जोड़ता है
जरूरत नहीं थी --- की \emph{सितारे और बार} आइसक्रीम की गिनती
--- फिर प्रसिद्ध को गिना जाता है \emph{विक्षिप्त टोपी} बिल्कुल, और
नंबर ढूंढता है $\frac1\eu $ टोपी के नीचे इंतज़ार कर रहा हूँ
ढेर, इस पुस्तक में इसकी तीसरी उपस्थिति है।

\medskip
\noindent\textbf{Part I --- Choosing the model.}
\begin{enumerate}
  \item बनी हुई लाइसेंस प्लेटों की गिनती करें $2$ पत्रों के बाद
        $3$ अंक; फिर केले का विपर्यय।
  \item ए से $32$-कार्ड डेक, गिनें $5$-कार्ड हाथ; फिर
        बिल्कुल युक्त हाथ $2$ की $4$ इक्के.
  \item एक रोबोट चलता है $(0,0)$ को $(4,3)$ केवल इकाई का उपयोग करना
        दाएँ या ऊपर कदम: कितने रास्ते? (एक पथ को एक के रूप में एन्कोड करें
        आर और यू में शब्द)
  \item बढ़ाना $(1 + x)^4$ द्विपद प्रमेय द्वारा
        (\cref{thm:g12:comb:binomial}); फिर मूल्यांकन करें
        $ x = 1$ और $ x = -1$: किन दो पहचानों के बारे में
        नंबर $\binom nk $ ड्रॉप आउट?
  \item उसे दो बार गिनकर \omterm{def:g11:vect:det}{सिद्ध} करें
        $ k\binom nk = n\binom{n-1}{k-1}$ (गिनो
        समिति-सह-अध्यक्ष दो तरीके), और निष्कर्ष निकालें
        $\sum_{k=0}^{n} k\binom nk = n\,2^{n-1}$.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Stars and bars.}
\begin{enumerate}[resume]
  \item एक आइसक्रीम की दुकान बेचती है $4$ स्वाद; आप ऑर्डर करें $10$
        स्कूप्स (स्वाद दोहराया जा सकता है, कप में ऑर्डर है
        अप्रासंगिक)। किसी ऑर्डर को एक पंक्ति के रूप में एन्कोड करें $10$ सितारे
        (स्कूप्स) द्वारा अलग किया गया $3$ बार (स्वाद \omterm{def:g12:comb:permutation}{परिवर्तन}), और
        ऑर्डर गिनें.
  \item अऋणात्मक के त्रिगुण गिनें \omterm{def:g10:numbers:sets}{पूर्णांकों} साथ
        $ x + y + z = 12$.
  \item के त्रिगुण गिनें \emph{सकारात्मक} \omterm{def:g10:numbers:sets}{पूर्णांकों} साथ
        $ x + y + z = 12$ (विकल्प $ x = 1 + x'$, वगैरह।)।
  \item के \omterm{def:g10:algebra:expand}{विस्तार} में कितने भिन्न एकपदी दिखाई देते हैं?
        $(a + b + c)^5$?
  \item विवेक विधि की जाँच करें: के आदेशों की गिनती करें $3$
        से स्कूप्स $2$ सूत्र के साथ स्वाद, फिर सूची
        उन सभी और तुलना करें।
  \item ठीक-ठीक कहें कि ``स्कूप समान हैं'' कहाँ दर्ज किया गया है
        एन्कोडिंग --- और गिनें कि इसके बजाय क्या होता है यदि
        स्कूप को क्रम में (अलग-अलग स्थिति में) खाया जाता है
        \cref{met:g12:comb:model}की चेकलिस्ट.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The deranged hats.}
ए \emph{गड़बड़ी} का पुनर्\omterm{def:g11:prob:rv}{वितरण} है $ n $ उनके लिए टोपियाँ
$ n $ जिनमें मालिक हैं \emph{कोई नहीं} अपनी टोपी प्राप्त करता है; होने देना
$ D_n $ उन्हें गिनें. (\Cref{pb:g11:prob:1} उस एक अतिथि को दिखाया
औसतन अपनी टोपी पुनः प्राप्त करता है --- अब हम पूरी गिनती करते हैं
बिल्कुल बदकिस्मत पार्टियाँ।)
\begin{enumerate}[resume]
  \item गणना करना $ D_1$, $ D_2$, $ D_3$ सूचीबद्ध करके, और $ D_4$
        धैर्यपूर्वक (या चतुराई से)।
  \item पुनरावृत्ति का औचित्य \omterm{def:g11:vect:det}{सिद्ध} करें
        $ D_n = (n - 1)\left(D_{n-1} + D_{n-2}\right)$: अतिथि 1
        कुछ टोपी मिलती है $ k \neq 1$ ($ n - 1$ विकल्प); विभाजन
        चाहे अतिथि के अनुसार $ k $ टोपी 1 प्राप्त करता है या नहीं.
        जांचें कि यह पुनरुत्पादित करता है $ D_4$, और गणना करें $ D_5$.
  \item के लिए $ n = 3$, समावेशन--बहिष्करण (घटाना) द्वारा \omterm{def:g11:vect:det}{सिद्ध} करें
        कम से कम एक टोपी को ठीक करने वाले असाइनमेंट को वापस \omterm{def:g10:vectors:sum}{जोड़}ें
        अधिक गिनती) वह
        $ D_3 = 3!\left(1 - \frac{1}{1!} + \frac{1}{2!} -
        \frac{1}{3!}\right)$, और सामान्य सूत्र बताएं।
  \item गणना करना $\frac{D_5}{5!}$ और तुलना करें
        $\frac1\eu \approx 0.3679$: द \omterm{def:g10:proba:distribution}{संभावना} वह ए
        बड़ी फेरबदल वाली पार्टी पूरी तरह से विक्षिप्त है
        $\frac1\eu $ --- इस स्थिरांक का तीसरा कैमियो, इसके बाद
        लॉटरी और सचिव \cref{pb:g12:exp:1}.
        (क्यों: प्रश्न 14 का सूत्र a की शुरुआत है
        के लिए प्रसिद्ध श्रृंखला $\eu^{-1}$, विश्वविद्यालय में बताया गया
        वॉल्यूम.)
  \item बीच में गुप्त सांता $10$ मित्र: नाम निकाले गए हैं
        समान रूप से यादृच्छिक रूप से। क्या है \omterm{def:g10:proba:distribution}{संभावना} ड्रा
        वैध है (कोई भी स्वयं नहीं खींचता), और कितने
        क्या समूह को पुनर्निर्धारण की \omterm{def:g11:prob:expectation}{अपेक्षा} करनी चाहिए?
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Counting twice, winning twice.}
\begin{enumerate}[resume]
  \item हाथ मिलाने की प्रमेयिका: किसी भी पार्टी में, मेहमानों के बारे में संक्षेप में बताना
        प्रत्येक हाथ मिलाने पर हाथ मिलाने की संख्या मायने रखती है
        बिल्कुल दो बार. उसका निष्कर्ष निकालो \emph{अतिथियों की संख्या
        जिसने \omterm{def:g11:func:parity}{विषम} संख्या में हाथ मिलाया वह हमेशा सम होता है} ---
        और तीन-मेहमानों की पार्टी में यह जांचें कि दावा सही है या नहीं।
  \item रत्न \omterm{def:g11:vect:det}{सिद्ध} करो
        $1^3 + 2^3 + \dots + n^3 = (1 + 2 + \dots + n)^2$ द्वारा
        प्रेरण, और इसके लिए सत्यापित करें $ n = 3$. (लिटिल गॉस)
        योग, वर्ग, घन गिनता है।)
  \item वेंडरमोंडे की पहचान (\cref{exo:g12:comb:9}) के माध्यम से
        पथ: व्याख्या करें $\binom{2n}{n}$ के जाली पथ के रूप में
        प्रश्न 3 का प्रकार से $(0,0)$ को $(n,n)$, प्रत्येक को काटें
        इसके प्रति-विकर्ण को पार करने का पथ, और समझाइए
        कैसे $\sum_j \binom nj^2$ प्रकट होता है।
  \item समापन --- काउंटर की चार चालें, प्रत्येक में एक पंक्ति
        इस समस्या से एक उदाहरण: चरणों को गुणा करें और \omterm{def:g10:vectors:sum}{जोड़}ें
        मामले; चतुराई से एनकोड करें (सितारे और बार, पथ शब्द);
        एक ही चीज़ को दो बार गिनें (समिति-सह-अध्यक्ष,
        हाथ मिलाना); अवांछित को घटाएं और सही करें
        अत्यधिक गिनती (विक्षोभ)। और ध्यान दें कि गिनती कहां हो रही है
        आगे काम पर जाता है: \omterm{def:g10:proba:distribution}{संभावना}, और के रास्ते
        मैट्रिक्स-और-ग्राफ़ अध्याय।
\end{enumerate}
\end{problem}
