\chapter{त्रिकोणमितीय कार्य}\label{ch:g12:trigo}

 कार्य $\cos $ और $\sin $ वृत्त की ज्यामिति को बदलिए
विश्लेषण. यूनिट सर्कल के चारों ओर वास्तविक रेखा को घुमाकर परिभाषित किया गया है, वे हैं
आवधिक का मूलरूप कार्य, और उनके डेरिवेटिव उन्हें बनाओ
के समाधान \omterm{def:g10:algebra:equation}{समीकरण} दोलनों का $ y'' = -y $.

\section{इकाई चक्र}

\begin{definition}[कोज्या और ज्या]\label{def:g12:trigo:cossin}
होने देना $\mathcal{C}$ त्रिज्या 1 का वृत्त a के मूल पर केन्द्रित हो
\omterm{def:g10:coordgeom:system}{ऑर्थोनॉर्मल} \omterm{def:g10:coordgeom:system}{निर्देशांक तरीका}. प्रत्येक वास्तविक को $ t $, बिंदु को संबद्ध करें $ M(t)$
लंबाई घुमाकर प्राप्त किया जाता है $\abs{t}$ साथ में $\mathcal{C}$ बिन्दु से
$ I(1,0)$, यदि वामावर्त $ t \geq 0$ और अन्यथा दक्षिणावर्त। तब
\emph{$\cos t $ और $\sin t $ के निर्देशांक हैं
$ M(t)$}\index{$\cos t $ और $\sin t $ के निर्देशांक हैं
$ M(t)$}\index{ज्या}:
\[
M(t) = (\cos t, \sin t).
\]
\end{definition}

चूँकि वृत्त की परिधि होती है $2\pi $, बात $ M(t)$ जब अपरिवर्तित है
$ t $ से बढ़ जाता है $2\pi $; और तबसे $ M(t)$ इकाई वृत्त पर स्थित है:

\begin{proposition}[मौलिक पहचान]\label{prop:g12:trigo:identities}
सभी के लिए $ t \in \R $ और $ k \in \Z $:
\[
\cos^2 t + \sin^2 t = 1, \qquad
\cos(t + 2k\pi) = \cos t, \qquad
\sin(t + 2k\pi) = \sin t,
\]
\[
\cos(-t) = \cos t, \qquad \sin(-t) = -\sin t,
\]
\[
\cos(\pi - t) = -\cos t, \quad \sin(\pi - t) = \sin t, \quad
\cos\left(\tfrac{\pi}{2} - t\right) = \sin t, \quad
\sin\left(\tfrac{\pi}{2} - t\right) = \cos t .
\]
\end{proposition}

\begin{proof}
पहली पहचान है \omterm{def:g10:algebra:equation}{समीकरण} यूनिट सर्कल का; दूसरे व्यक्त करते हैं
वृत्त की समरूपताएँ: $ M(-t)$ का प्रतिबिम्ब है $ M(t)$ में
$ x $-अक्ष, $ M(\pi - t)$ में $ y $-अक्ष, और $ M(\frac\pi2 - t)$ पंक्ति में
$ y = x $.
\end{proof}

हृदय से जानने योग्य मूल्य:
\begin{center}
\begin{tabular}{c|ccccc}
$ t $ & $0$ & $\dfrac{\pi}{6}$ & $\dfrac{\pi}{4}$ & $\dfrac{\pi}{3}$ & $\dfrac{\pi}{2}$\\[6pt]
\hline\rule{0pt}{14pt}%
$\cos t $ & $1$ & $\dfrac{\sqrt3}{2}$ & $\dfrac{\sqrt2}{2}$ & $\dfrac12$ & $0$\\[6pt]
$\sin t $ & $0$ & $\dfrac12$ & $\dfrac{\sqrt2}{2}$ & $\dfrac{\sqrt3}{2}$ & $1$
\end{tabular}
\end{center}

\begin{proposition}[अतिरिक्त सूत्र]\label{prop:g12:trigo:addition}
सभी के लिए $ a, b \in \R $:
\begin{align*}
\cos(a+b) &= \cos a \cos b - \sin a \sin b, &
\sin(a+b) &= \sin a \cos b + \cos a \sin b,\\
\cos(a-b) &= \cos a \cos b + \sin a \sin b, &
\sin(a-b) &= \sin a \cos b - \cos a \sin b.
\end{align*}
विशेष रूप से $\cos 2a = 2\cos^2 a - 1 = 1 - 2\sin^2 a $ और
$\sin 2a = 2 \sin a \cos a $.
\end{proposition}

\begin{proof}
 वैक्टर $\vect{u} = (\cos a, \sin a)$ और $\vect{v} = (\cos b, \sin b)$
इकाई हैं वैक्टर एक कोण बनाना $ a - b $; उनका \omterm{def:g11:scal:dot}{अदिश उत्पाद} दोनों के बराबर है
$\cos a\cos b + \sin a \sin b $ (\omterm{def:g10:coordgeom:system}{COORDINATES}) और
$\norm{\vect u}\,\norm{\vect v}\cos(a-b) = \cos(a - b)$ (ज्यामिति), जो है
तीसरा सूत्र. की जगह $ b $ द्वारा $-b $ पहला देता है; \omterm{def:g12:trigo:cossin}{ज्या} सूत्र
का उपयोग करके पालन करें $\sin x = \cos(\frac\pi2 - x)$. दोहराव सूत्र हैं
मामला $ b = a $ के साथ संयुक्त $\cos^2 + \sin^2 = 1$.
\end{proof}

\section{विश्लेषणात्मक अध्ययन}

\begin{lemma}[मौलिक सीमा]\label{lem:g12:trigo:sinxx}
\[
\lim_{t \to 0} \frac{\sin t}{t} = 1
\qquad\text{and}\qquad
\lim_{t \to 0} \frac{\cos t - 1}{t} = 0 .
\]
\end{lemma}

\begin{proof}
के लिए $0 < t < \frac{\pi}{2}$, यूनिट सर्कल में तीन क्षेत्रों की तुलना करें: द
त्रिकोण $ OIM(t)$, वृत्ताकार क्षेत्र $ OIM(t)$, और समकोण त्रिभुज के साथ
आधार $ OI $ और ऊंचाई $\tan t $:
\[
\frac{\sin t}{2} \leq \frac{t}{2} \leq \frac{\tan t}{2}.
\]

$\frac{\sin t}{2}$), वृत्ताकार क्षेत्र (लाल चाप से घिरा, क्षेत्र
$\frac{t}{2}$), और ऊंचाई का समकोण त्रिभुज $\tan t $ (नारंगी, क्षेत्र

पहली असमानता देती है $\frac{\sin t}{t} \leq 1$; दूसरा देता है
$\cos t \leq \frac{\sin t}{t}$. जैसा $ t \to 0^+$, $\cos t \to 1$ (निरंतरता),
और निचोड़ प्रमेय उपज देता है $\frac{\sin t}{t} \to 1$; समता इसका \omterm{def:g10:algebra:expand}{विस्तार} करती है
को $ t \to 0$. दूसरी सीमा के लिए,
\[
\frac{\cos t - 1}{t} = \frac{\cos^2 t - 1}{t(\cos t + 1)}
= -\frac{\sin t}{t}\cdot\frac{\sin t}{\cos t + 1}
\xrightarrow[t\to0]{} -1 \cdot \frac{0}{2} = 0. \qedhere
\]
\end{proof}

\begin{theorem}[साइन और कोसाइन के व्युत्पन्न]\label{thm:g12:trigo:derivatives}
 कार्य $\sin $ और $\cos $ हैं \omterm{def:g12:deriv:derivative}{विभेदक} पर $\R $, साथ
\[
(\sin)' = \cos, \qquad (\cos)' = -\sin .
\]
आम तौर पर अधिक, $(\sin(ax+b))' = a\cos(ax+b)$ और
$(\cos(ax+b))' = -a\sin(ax+b)$.
\end{theorem}

\begin{proof}
योग सूत्र द्वारा,
\[
\frac{\sin(x+h) - \sin x}{h}
= \sin x\,\frac{\cos h - 1}{h} + \cos x\,\frac{\sin h}{h}
\xrightarrow[h\to0]{} \sin x \cdot 0 + \cos x \cdot 1 = \cos x,
\]
का उपयोग करते हुए \cref{lem:g12:trigo:sinxx}. के लिए गणना $\cos $ समान है,
और सामान्य रूप श्रृंखला नियम से अनुसरण करता है।
\end{proof}

\begin{proposition}[बदलाव]\label{prop:g12:trigo:variations}
पर $\intcc{0}{\pi}$, $\cos $ से घट जाती है $1$ को $-1$. पर
$\intcc{-\frac\pi2}{\frac\pi2}$, $\sin $ से बढ़ता है $-1$ को $1$. दोनों
कार्य हैं $2\pi $-आवधिक और सीमाएं हैं $\intcc{-1}{1}$; $\cos $ है
यहां तक कि, $\sin $ अजीब है.
\end{proposition}

\begin{proof}
पर $\intoo{0}{\pi}$, $(\cos)' = -\sin < 0$ बिंदु के बाद से $ M(t)$ है
सकारात्मक \omterm{def:g10:coordgeom:system}{तालमेल} वहाँ; इसी तरह $(\sin)' = \cos > 0$ पर
$\intoo{-\frac\pi2}{\frac\pi2}$. आवधिकता और समता आती है
\cref{prop:g12:trigo:identities}.
\end{proof}

\begin{method}[हल $\cos x = a $ और $\sin x = a $]\label{met:g12:trigo:equations}
के लिए $ a \in \intcc{-1}{1}$, एक विशेष समाधान खोजें $\alpha $ (से
मूल्यों की तालिका या कैलकुलेटर)। फिर सभी समाधान हैं:
\[
\cos x = \cos\alpha \iff x = \alpha + 2k\pi \text{ or } x = -\alpha + 2k\pi,
\quad k \in \Z;
\]
\[
\sin x = \sin\alpha \iff x = \alpha + 2k\pi \text{ or } x = \pi - \alpha + 2k\pi,
\quad k \in \Z.
\]
असमानताओं के लिए, इकाई वृत्त पर समाधान चाप का पता लगाएं और पढ़ें
द \omterm{def:g10:numbers:interval}{अंतराल} एक अवधि के भीतर.
\end{method}

\begin{example}\label{ex:g12:trigo:equation}
हल करना $\cos x = \frac12$ में $\intoc{-\pi}{\pi}$: एक विशेष समाधान है
$\frac\pi3$, तो समाधान हैं $ x = \frac\pi3$ और $ x = -\frac\pi3$.
सभी में $\R $: $ x = \pm\frac\pi3 + 2k\pi $, $ k \in \Z $.
\end{example}

\begin{exercise}[$\star\star\star $]\label{exo:g12:trigo:8}
होने देना $ f(x) = A\cos x + B \sin x $ साथ $(A,B) \neq (0,0)$.
\begin{enumerate}
  \item बताते हैं कि $ f $ लिखा जा सकता है $ f(x) = R\cos(x - \varphi)$ साथ
        $ R = \sqrt{A^2 + B^2}$ और एक उपयुक्त $\varphi $.
  \item निष्कर्ष निकालें \omterm{def:g10:functions:extrema}{अधिकतम} और \omterm{def:g10:functions:extrema}{न्यूनतम} का $ f $, और हल करें
        $\cos x + \sin x = 1$ में $\intoc{-\pi}{\pi}$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g12:trigo:9}
बताते हैं कि $\sin $ और $\cos $ दोनों अंतर को संतुष्ट करते हैं \omterm{def:g10:algebra:equation}{समीकरण}
$ y'' = -y $. इसके विपरीत, चलो $ y $ का समाधान हो $ y'' = -y $ साथ
$ y(0) = a $ और $ y'(0) = b $; बताते हैं कि $ y = a\cos + b\sin $.
(संकेत: विचार करें $ g = (y - a\cos - b\sin)$ और ``ऊर्जा''
$ E = g^2 + (g')^2$.)
\end{exercise}

\section{समस्या: पेंडुलम का रहस्य}

\begin{problem}[{सप्ताहांत समस्या---एक ज्यामितीय सीमा शक्तियाँ साइन और कोसाइन के व्युत्पन्न, और उन सभी से दोलन: स्प्रिंग्स, पेंडुलम, बीट्स और मीटर जो लगभग था}]
\label{pb:g12:trigo:1}
हर घड़ी जो कभी टिक-टिक करती थी, हर तार जो कभी बजती थी,
उसी गणित का पालन करता है: आनुपातिक पुनर्स्थापना बल
विस्थापन, इसलिए \omterm{def:g10:algebra:equation}{समीकरण} $ y'' = -\omega^2 y $, इस तरह
कोसाइन. इन सबके आधार पर एक मासूम सी दिखने वाली सीमा बैठी है,
$\frac{\sin x}{x} \to 1$ --- ग्रेड 9 में वादा किया गया, द्वारा उपयोग किया गया
\cref{thm:g12:trigo:derivatives}, और यहां एक तस्वीर के साथ साबित हुआ।
यह समस्या उस सीमा से एक मीटर के झूले तक चढ़ जाती है
पेंडुलम, और बताता है कि मीटर, एक मूंछ से, क्यों नहीं है
इसके द्वारा परिभाषित.

\medskip
\noindent\textbf{Part I --- The fundamental limit.}
\begin{enumerate}
  \item संख्यात्मक रूप से वार्मअप करें: गणना करें $\frac{\sin x}{x}$ के लिए
        $ x = 0.1$ और $ x = 0.01$ (रेडियंस!).
  \item सैंडविच, यूनिट सर्कल पर साथ
        $0 < x < \frac\pi2$: प्रारंभिक ज्यामिति द्वारा व्यक्त करें
        (i) शीर्षों वाले त्रिभुज का क्षेत्रफल $ O $, $ A(1,0)$
        और वृत्त बिंदु $ M(\cos x, \sin x)$; (ii) द
        वृत्ताकार क्षेत्र $ OAM $; (iii) समकोण त्रिभुज $ OAT $
        कहाँ $ T(1, \tan x)$.
  \item क्षेत्र से सैंडविच निकालें
        $\sin x < x < \tan x $, तब
        $\cos x < \frac{\sin x}{x} < 1$, और निष्कर्ष निकालो
        निचोड़ प्रमेय कि
        $\lim_{x \to 0} \frac{\sin x}{x} = 1$ (समता संभालती है
        $ x < 0$).
  \item परिणाम निकालना
        $\lim_{x \to 0} \frac{1 - \cos x}{x^2} = \frac12$
        (संयुग्म द्वारा गुणा करें)।
  \item बताएं कि यह सीमा का इंजन क्यों है?
        \cref{thm:g12:trigo:derivatives} (क्या है
        $\sin'(0)$?), और यह पुराने ऋण का निपटान क्यों करता है: में
        क्या भाव हैं \emph{रेडियंस} के लिए एकमात्र कोण इकाई
        जो $\sin' = \cos $ बिना किसी स्थिरांक के धारण करता है?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Harmonic motion.}
\begin{enumerate}[resume]
  \item श्रृंखला नियम द्वारा सत्यापित करें
        $ y(t) = R\cos(\omega t - \varphi)$ संतुष्ट
        $ y'' = -\omega^2 y $.
  \item एक झरने पर एक द्रव्यमान आज्ञा का पालन करता है $ y'' = -4y $, साथ
        $ y(0) = 3$ और $ y'(0) = 8$. स्वीकार करना (से
        \cref{exo:g12:trigo:9}) कि सभी समाधान हैं
        $ A\cos 2t + B\sin 2t $, ठानना $ A $ और $ B $.
  \item समाधान को इस रूप में पुनः लिखें $ R\cos(2t - \varphi)$
        (\cref{exo:g12:trigo:8}): आयाम दीजिये $ R $, द
        अवधि, और चरण $\varphi $ (तीन दशमलव).
  \item द्रव्यमान सबसे पहले किस समय गुजरता है?
        संतुलन की स्थिति $ y = 0$?
  \item पेंडुलम: लंबाई की एक डोरी पर एक बॉब $ L $ का अनुसरण करता है
        \emph{बिल्कुल} $\theta'' = -\frac gL \sin\theta $ ---
        प्राथमिक द्वारा अघुलनशील कार्य. छोटे-छोटे झूलों के लिए,
        बदलें $\sin\theta $ द्वारा $\theta $ (भाग I इसे उचित ठहराता है!)
        और अवधि घटाएं
        $ T = 2\pi\sqrt{\frac Lg}$. गणना करना $ T $ के लिए
        $ L = 1$ एम ($ g = 9.81$): गैलीलियो की प्रसिद्ध खोज ---
        फार्मूला क्या करता है \emph{नहीं} पर निर्भर?
  \item सेकंड पेंडुलम: लंबाई क्या देती है
        $ T = 2$ बिल्कुल सही है? 1791 की फ्रांसीसी अकादमी झिझकी
        इस लंबाई और इसके दस लाखवें हिस्से के बीच
        परिभाषित करने के लिए क्वार्टर-मेरिडियन \emph{मीटर} --- द्वारा
        दोनों उम्मीदवारों में कितने मिलीमीटर का अंतर है?
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Addition formulas at work.}
\begin{enumerate}[resume]
  \item योग सूत्र सहित दिखाएँ
        (\cref{prop:g12:trigo:addition}) वह
        $\cos\left(x + \frac\pi2\right) = -\sin x $ और
        $\sin\left(x + \frac\pi2\right) = \cos x $: गोलाकार
        विभेदन एक चौथाई-मोड़ है --- स्थिरता की जाँच करें
        के साथ \cref{thm:g12:trigo:derivatives}.
  \item उत्पाद-से-योग पहचान प्राप्त करें
        $\cos a \cos b = \frac12\left(\cos(a - b) +
        \cos(a + b)\right)$ और रैखिकरण
        $\cos^2 x = \frac{1 + \cos 2x}{2}$ (एकीकरण
        अध्याय आपको धन्यवाद देगा)।
  \item धड़कता है: प्राप्त करें
        $\sin p + \sin q = 2 \sin\frac{p + q}{2}
        \cos\frac{p - q}{2}$, और इसे दो ट्यूनिंग फ़ोर्क्स पर लागू करें
        पर $440$ और $444$ हर्ट्ज: कान कौन सा स्वर सुनता है, और
        प्रति सेकंड कितनी धड़कनें? (इस तरह ऑर्केस्ट्रा होता है
        धुन.)
  \item त्रिकोण पहचान प्राप्त करें
        $\cos 3x = 4\cos^3 x - 3\cos x $
        (लिखना $3x = 2x + x $). सेटिंग $ x = 20^\circ $, क्या
        घन \omterm{def:g10:algebra:equation}{समीकरण} अवश्य $\cos 20^\circ $ संतुष्ट? (वह यह
        घन को केवल वर्गमूल से हल नहीं किया जा सकता
        सटीक रूप से कोण का त्रिखंड क्यों पराजित हुआ
        दो सहस्राब्दियों तक शासक और दिशा सूचक यंत्र --- पूरी कहानी
        विश्वविद्यालय के संस्करणों में बताया गया है।)
  \item एक प्रत्यावर्ती धारा पढ़ती है
        $ I(t) = 3\cos(100\pi t) + 3\sqrt3 \sin(100\pi t)$.
        इसे इस रूप में पुनः लिखें $ R\cos(100\pi t - \varphi)$: शिखर
        वर्तमान और चरण?
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The oscillating world.}
\begin{enumerate}[resume]
  \item अवधि: (सबसे छोटी) अवधि क्या है?
        $\sin^2 t $ (प्रश्न 13 का प्रयोग करें)? का
        $\sin(2t) + \sin(3t)$?
  \item एक वास्तविक पेंडुलम नम:
        $ y(t) = \eu^{-t/10}\cos(2\pi t)$. गति का वर्णन करें,
        लिफाफे के मूल्य की गणना करें $ t = 10$ (एक पुराना मित्र
        प्रकट होता है, \cref{pb:g12:exp:1}), और अध्याय को नाम दें
        कहाँ \omterm{def:g10:algebra:equation}{समीकरण} ऐसे समाधानों से समाधान हो जाता है।
  \item क्यों करते हो जीवाओं और कोसाइन हर कंपन पर शासन करें? दे दो
        दो वाक्यों में भौतिकी-से-गणित शब्दकोश
        (विस्थापन के अनुपात में बल बहाल करना
        $\rightarrow $ कौन \omterm{def:g10:algebra:equation}{समीकरण} $\rightarrow$ जो
        समाधान), और क्या कहें \cref{exo:g12:trigo:9}
        दावे में योगदान देता है.
  \item समापन: अध्याय का चक्र चार चरणों में --- एक
        ज्यामितीय सीमा, दो डेरिवेटिव, एक अंतर
        \omterm{def:g10:algebra:equation}{समीकरण}, सभी दोलन। और ईमानदार तारांकन:
        बड़े उतार-चढ़ाव के लिए पेंडुलम की वास्तविक अवधि में एक शामिल होता है
        अण्डाकार अभिन्न, बिजली की गति से गणना की गई\,\dots
        गॉस का अंकगणित--ज्यामितीय \omterm{def:g11:stat:mean}{अर्थ}, की मिठाई
        \cref{pb:g12:seq:1}. सीरीज के धागे अपने आप जुड़ते हैं.
\end{enumerate}
\end{problem}
