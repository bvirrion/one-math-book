\chapter{घातांक और लघुगणक}\label{ch:g12:exp}

 \omterm{thm:g12:exp:existence}{घातांक प्रकार्य} अद्वितीय है \omterm{def:g11:func:function}{समारोह} अपने बराबर यौगिक
और मूल्य ले रहा हूँ $1$ पर $0$. यह रकम को उत्पादों में परिवर्तित करता है; इसका उलटा,
द \omterm{def:g12:exp:ln}{प्राकृतिक}, उत्पादों को रकम में परिवर्तित करता है। वे मिलकर वर्णन करते हैं
प्रत्येक घटना जिसके परिवर्तन की दर उसके आकार के समानुपाती होती है:
रेडियोधर्मी क्षय, जनसंख्या वृद्धि, चक्रवृद्धि ब्याज।

\section{घातांकीय फलन}

\begin{theorem}[अस्तित्व और विशिष्टता]\label{thm:g12:exp:existence}
वहाँ एक अनोखा अस्तित्व है \omterm{def:g12:deriv:derivative}{विभेदक} \omterm{def:g11:func:function}{समारोह} $ f \colon \R \to \R $ ऐसा है कि
\[
f' = f \qquad\text{and}\qquad f(0) = 1 .
\]
इसे कहा जाता है \emph{घातांक प्रकार्य}\index{घातांक प्रकार्य} और लिखा
$\exp $, या $ x \mapsto \eu^x $.
\end{theorem}

\begin{proof}[अद्वितीयता का प्रमाण]
सबसे पहले, ऐसे \omterm{def:g11:func:function}{समारोह} कभी गायब नहीं होता. वास्तव में, चलो
$ g(x) = f(x)f(-x)$; तब
\[
g'(x) = f'(x)f(-x) - f(x)f'(-x) = f(x)f(-x) - f(x)f(-x) = 0,
\]
इसलिए $ g $ के बराबर स्थिरांक है $ g(0) = 1$: हरएक के लिए $ x $,
$ f(x)f(-x) = 1$, खास तरीके से $ f(x) \neq 0$.

अब चलो $ f_1, f_2$ दो समाधान हो और $ h = \dfrac{f_1}{f_2}$ (वैध
तब से $ f_2$ कभी गायब नहीं होता) तब
\[
h' = \frac{f_1' f_2 - f_1 f_2'}{f_2^2} = \frac{f_1 f_2 - f_1 f_2}{f_2^2} = 0,
\]
इसलिए $ h $ के बराबर स्थिरांक है $ h(0) = 1$, \emph{यानी} $ f_1 = f_2$.

\emph{अस्तित्व} इस स्तर पर प्रवेश दिया जाता है (इसे इसके माध्यम से प्राप्त किया जा सकता है)।
\omterm{def:g12:seq:sequence}{अनुक्रम} का \cref{exo:g12:seq:10}, या इसके विपरीत के रूप में लोगारित्म
में एकीकरण द्वारा निर्मित \cref{ch:g12:integ}).
\end{proof}

\begin{proposition}[कार्यात्मक समीकरण]\label{prop:g12:exp:funceq}
सभी के लिए $ x, y \in \R $ और $ n \in \Z $:
\[
\eu^{x+y} = \eu^x\,\eu^y, \qquad
\eu^{-x} = \frac{1}{\eu^x}, \qquad
\eu^{x-y} = \frac{\eu^x}{\eu^y}, \qquad
\bigl(\eu^{x}\bigr)^n = \eu^{nx}.
\]
इसके अतिरिक्त $\eu^x > 0$ सभी के लिए $ x $.
\end{proposition}

\begin{proof}
हल करना $ y $ और विचार करें $\varphi(x) = \dfrac{\exp(x+y)}{\exp(x)\exp(y)}$. यह
अंश और हर, जैसे कार्य का $ x $, दोनों समाधान हैं
$ f' = f $ स्थिरांक तक $\exp(y)$; फर्क $\varphi $ सीधे
(भागफल नियम) देता है $\varphi' = 0$, इसलिए
$\varphi \equiv \varphi(0) = 1$, पहली पहचान साबित करना। लेना
$ y = -x $ दूसरा (मूल्य के साथ) देता है $\eu^0 = 1$), और तीसरा
अनुसरण करता है। चौथा पहले से एक प्रेरण है $ n \geq 0$, \omterm{def:g10:algebra:expand}{विस्तार}ित
को $ n < 0$ दूसरे द्वारा.

सकारात्मकता: $\eu^x = \left(\eu^{x/2}\right)^2 \geq 0$ और $\eu^x \neq 0$
(में दिखाया गया है \cref{thm:g12:exp:existence}), इसलिए $\eu^x > 0$.
\end{proof}

\begin{proposition}[विविधताएँ और सीमाएँ]\label{prop:g12:exp:variations}
घातांक सख्ती से है \omterm{def:g12:seq:monotonic}{की बढ़ती}, \omterm{def:g12:deriv:convex}{उत्तल}, और
\[
\lim_{x\to-\infty} \eu^x = 0, \qquad
\lim_{x\to+\infty} \eu^x = +\infty .
\]
\end{proposition}

\begin{proof}
$(\exp)' = \exp > 0$ सख्त वृद्धि देता है; $(\exp)'' = \exp > 0$ देता है
उत्तलता. द्वारा उत्तलता, $\eu^x \geq 1 + x $ (\omterm{def:g12:deriv:tangent}{स्पर्शरेखा} पर $0$, देखना
\cref{ex:g12:deriv:ineq}), इसलिए $\eu^x \to +\infty $ जैसा $ x \to +\infty $ द्वारा
तुलना. तब $\eu^{x} = 1/\eu^{-x} \to 0$ जैसा $ x \to -\infty $.
\end{proof}

\begin{theorem}[विकास तुलना]\label{thm:g12:exp:growth}
\index{विकास तुलना}
हरएक के लिए पूर्णांक $ n \geq 1$,
\[
\lim_{x\to+\infty} \frac{\eu^x}{x^n} = +\infty,
\qquad
\lim_{x\to-\infty} x^n \eu^x = 0 .
\]
शब्दों में: घातांक हर शक्ति को मात देता है $ x $.
\end{theorem}

\begin{proof}
के लिए $ n = 1$: आवेदन करना $\eu^t \geq 1 + t \geq t $ पर $ t = x/2$,
\[
\frac{\eu^x}{x} = \frac{\left(\eu^{x/2}\right)^2}{x}
\geq \frac{(x/2)^2}{x} = \frac{x}{4} \xrightarrow[x\to+\infty]{} +\infty .
\]
सामान्य के लिए $ n $, लिखना
\[
\frac{\eu^x}{x^n}
= \frac{1}{n^n}\left(\frac{\eu^{x/n}}{x/n}\right)^{n}
\xrightarrow[x\to+\infty]{} +\infty
\]
मामले से $ n=1$ और रचना. पर सीमा $-\infty $ का अनुसरण करता है
प्रतिस्थापन $ x \mapsto -x $:
$\abs{x^n \eu^x} = \frac{\abs{x}^n}{\eu^{\abs x}} \to 0$.
\end{proof}

\section{प्राकृतिक लघुगणक}

\begin{definition}[प्राकृतिक]\label{def:g12:exp:ln}
घातांक है निरंतर और सख्ती से \omterm{def:g12:seq:monotonic}{की बढ़ती} से $\R $ पर
$\intoo{0}{+\infty}$; आक्षेप प्रमेय द्वारा
(\cref{thm:g12:limcont:bijection}), हरएक के लिए $ y > 0$  \omterm{def:g10:algebra:equation}{समीकरण}
$\eu^x = y $ एक अनोखा समाधान है. यह समाधान है \emph{प्राकृतिक}\index{प्राकृतिक} का $ y $, लिखा हुआ $\ln y $. इस प्रकार
\[
\text{for } x \in \R,\ y > 0: \qquad y = \eu^x \iff x = \ln y .
\]
विशेष रूप से $\ln 1 = 0$, $\ln \eu = 1$, और
$\eu^{\ln y} = y $, $\ln(\eu^x) = x $.
\end{definition}

\begin{proposition}[बीजगणितीय गुण]\label{prop:g12:exp:lnalg}
सभी के लिए $ a, b > 0$ और $ n \in \Z $:
\[
\ln(ab) = \ln a + \ln b, \quad
\ln\frac{1}{a} = -\ln a, \quad
\ln\frac{a}{b} = \ln a - \ln b, \quad
\ln(a^n) = n \ln a, \quad
\ln\sqrt{a} = \tfrac12 \ln a .
\]
\end{proposition}

\begin{proof}
$\eu^{\ln a + \ln b} = \eu^{\ln a}\eu^{\ln b} = ab $, और ले रहा हूँ $\ln $ दोनों का
पक्ष पहली पहचान देता है. अन्य लोग उसी तंत्र का अनुसरण करते हैं
की संगत पहचान \cref{prop:g12:exp:funceq}.
\end{proof}

\begin{proposition}[विश्लेषणात्मक गुण]\label{prop:g12:exp:lnanalytic}
 \omterm{def:g11:func:function}{समारोह} $\ln $ है \omterm{def:g12:deriv:derivative}{विभेदक} पर $\intoo{0}{+\infty}$ साथ
\[
(\ln x)' = \frac{1}{x},
\]
कठोरता से \omterm{def:g12:seq:monotonic}{की बढ़ती}, \omterm{def:g12:deriv:convex}{नतोदर}, और
$\lim\limits_{x\to0^+} \ln x = -\infty $,
$\lim\limits_{x\to+\infty} \ln x = +\infty $. इसके अलावा, प्रत्येक के लिए पूर्णांक
$ n \geq 1$,
\[
\lim_{x\to+\infty} \frac{\ln x}{x^{1/n}} = 0, \qquad
\lim_{x\to0^+} x \ln x = 0 .
\]
\end{proposition}

\begin{proof}
व्युत्क्रम की भिन्नता \omterm{def:g11:func:function}{समारोह} इस स्तर पर स्वीकार किया जाता है; अनुदान
यह, पहचान को अलग करता है $\eu^{\ln x} = x $ शृंखला नियम द्वारा:
$(\ln x)'\,\eu^{\ln x} = 1$, इसलिए $(\ln x)' = \frac{1}{x} > 0$, जहां से सख्त
वृद्धि, और $(\ln x)'' = -\frac1{x^2} < 0$, जहां से अवतलता. सीमा पर
$0^+$ और $+\infty $ उन का दर्पण $\exp $ आक्षेप के माध्यम से. के लिए
\omterm{thm:g12:exp:growth}{विकास तुलना}, विकल्प $ x = \eu^{t}$:
$\frac{\ln x}{x} = \frac{t}{\eu^t} \to 0$ जैसा $ t \to +\infty $ द्वारा
\cref{thm:g12:exp:growth}, और इसी तरह अन्य सीमाओं के लिए भी
$ x = \eu^{-t}$, देना $ x\ln x = -t\eu^{-t} \to 0$.
\end{proof}

\begin{method}[के साथ समीकरणों को हल करना $\exp $ और $\ln $]\label{met:g12:exp:equations}
दोनों कार्य सख्ती से हैं \omterm{def:g12:seq:monotonic}{की बढ़ती}, इसलिए उन्हें लागू किया जा सकता है (या हटाया जा सकता है
से) ए के दोनों तरफ \omterm{def:g10:algebra:equation}{समीकरण} या अपनी दिशा बदले बिना असमानता:
\[
\eu^{u} = \eu^{v} \iff u = v, \qquad
\eu^{u} \leq \eu^{v} \iff u \leq v,
\]
और इसी तरह के लिए $\ln $ सकारात्मक मात्रा पर. \emph{हमेशा जांचें डोमेन
पहला} ($\ln $ सकारात्मक तर्क की आवश्यकता है)। के लिए \omterm{def:g10:algebra:equation}{समीकरण} पसंद
$ a^x = b $ साथ $ a>0$, $ a \neq 1$, पुनः लिखें $ a^x = \eu^{x\ln a}$ और हल करें
$ x = \frac{\ln b}{\ln a}$.
\end{method}

\begin{example}\label{ex:g12:exp:halflife}
एक रेडियोधर्मी \omterm{def:g10:proba:sample}{नमूना} निम्नलिखित क्षय होता है $ N(t) = N_0\,\eu^{-\lambda t}$. इसका
\emph{हाफ लाइफ} $ T $ संतुष्ट $ N(T) = N_0/2$, \emph{यानी}
$\eu^{-\lambda T} = \frac12$, इसलिए $ T = \frac{\ln 2}{\lambda}$: आधा जीवन
प्रारंभिक मात्रा पर निर्भर नहीं करता.
\end{example}

\section{अभ्यास}

\begin{exercise}[$\star $]\label{exo:g12:exp:1}
सरल
$\dfrac{\eu^{3x}\,\eu^{-x+1}}{\eu^{x}}$ और
$\ln\!\left(\dfrac{\eu^2\sqrt{\eu}}{\eu^{-3}}\right)$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g12:exp:2}
में हल करें $\R $:
\[
\text{(a) } \eu^{2x} - 3\eu^{x} + 2 = 0;
\qquad
\text{(b) } \ln(x - 1) + \ln(x + 2) = \ln 4 .
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g12:exp:3}
सीमा की गणना करें:
\[
\lim_{x\to+\infty} \frac{\eu^x - x^2}{\eu^x + 1}, \qquad
\lim_{x\to+\infty} \frac{\ln x}{\sqrt x}, \qquad
\lim_{x\to0^+} x^2 \ln x, \qquad
\lim_{x\to+\infty} \bigl(x - \ln x\bigr).
\]
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:exp:4}
का अध्ययन करें \omterm{def:g11:func:function}{समारोह} $ f(x) = x\,\eu^{-x}$ पर $\R $: विविधताएँ, सीमाएँ,
चरम; दिखाएँ कि इसके वक्र में एक है \omterm{def:g12:deriv:inflection}{विभक्ति बिंदु} और उसे दे दो
\omterm{def:g10:coordgeom:system}{COORDINATES}.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:exp:5}
वह सभी को दिखाओ $ x > -1$, $\ln(1 + x) \leq x $, केवल समानता के साथ
$ x = 0$. सभी के लिए यह निष्कर्ष निकालें $ n \geq 1$,
\[
\left(1 + \frac{1}{n}\right)^n \leq \eu \leq \left(1 - \frac{1}{n+1}\right)^{-(n+1)} .
\]
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:exp:6}
राजधानी $ C_0$ की वार्षिक दर पर निवेश किया जाता है $3\%$, ब्याज चक्रवृद्धि
प्रत्येक वर्ष.
\begin{enumerate}
  \item राजधानी व्यक्त करें $ C_n $ बाद $ n $ साल।
  \item कितने वर्ष बाद पूंजी दोगुनी हो जाती है? सटीक उत्तर दीजिए
        का उपयोग कर रहे हैं $\ln $, फिर एक संख्यात्मक मान।
  \item से तुलना करें \omterm{def:g10:numbers:approx}{सन्निकटन} ``$70$ में दर से विभाजित किया गया
        प्रतिशत'' बैंकरों द्वारा उपयोग किया जाता है।
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:exp:7}
असमानता का समाधान करें $\eu^{2x} - \eu^{x+1} > 0$, फिर असमानता
$\ln(x^2 - 1) \leq \ln(x + 5)$.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g12:exp:8}
होने देना $ f(x) = \dfrac{\eu^x}{x}$ के लिए $ x > 0$.
\begin{enumerate}
  \item की विविधताओं का अध्ययन करें $ f $ पर $\intoo{0}{+\infty}$ और उसे दे दो
        \omterm{def:g10:functions:extrema}{न्यूनतम}.
  \item किस मान के लिए $ k $ करता है \omterm{def:g10:algebra:equation}{समीकरण} $\eu^x = kx $ पास होना $0$, $1$
        या $2$ में समाधान $\intoo{0}{+\infty}$?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g12:exp:9}
के लिए $ n \geq 1$, होने देना $ u_n = \left(1 + \frac1n\right)^n $.
\begin{enumerate}
  \item का उपयोग करते हुए \cref{exo:g12:exp:5}, बताते हैं कि $(u_n)$ है \omterm{def:g12:seq:bounded}{ऊपर से घिरा हुआ}
        द्वारा~$\eu $.
  \item बताते हैं कि $\ln u_n = n \ln\left(1 + \frac1n\right) \to 1$, और
        उसका निष्कर्ष निकालो $ u_n \to \eu $.
        (संकेत: अंतर भागफल को पहचानें $\ln $ पर $1$.)
\end{enumerate}
\end{exercise}

\section{समस्या: लघुगणक दुनिया को वश में करता है}

\begin{problem}[{सप्ताहांत समस्या---दोगुना समय और नियम 72 में से, भूकंप, एसिड और पियानो के पैमाने, और स्थिर $\eu $ हर जगह उंगलियों के निशान छोड़ना}]
\label{pb:g12:exp:1}
जो कुछ भी निश्चित रूप से बढ़ता है \emph{को PERCENTAGE} बढ़ता है
तेजी से --- बचत, बैक्टीरिया, महामारी --- और जो भी
---भूकंप ऊर्जाओं को समझने के लिए दस की बहुत अधिक शक्तियाँ फैली हुई हैं,
अम्लता, ध्वनि तीव्रता --- द्वारा नियंत्रित किया जाता है लोगारित्म. यह
समस्या दोगुनी समय की गणना करती है और बैंकरों के नियम को उजागर करती है
72, दुनिया के लघुगणकीय पैमानों को पढ़ता है, और एकत्र करता है
उस नंबर पर उंगलियों के निशान $\eu $ हर अपराध स्थल पर निकल जाता है
(\cref{prop:g12:exp:funceq}, \cref{thm:g12:exp:growth},
\cref{met:g12:exp:equations}).

\medskip
\noindent\textbf{Part I --- Fluency.}
\begin{enumerate}
  \item हल करना: $\eu^{2x} = 5$; \quad $\ln(3x - 1) = 2$; \quad
        $\eu^{2x} - 3\eu^x + 2 = 0$ (भेष में एक द्विघात)।
  \item सरल बनाएं: $\dfrac{\ln 8}{\ln 2}$; \quad
        $\ln\!\left(\eu^3 \sqrt{\eu}\right)$; \quad
        $\eu^{\ln 5 - \ln 2}$.
  \item मातृ असमानता \omterm{def:g11:vect:det}{सिद्ध} करें: $\eu^x \geq 1 + x $ सबके लिए
        असली $ x $ (अध्ययन $ f(x) = \eu^x - 1 - x $), और निष्कर्ष निकालें
        $\ln(1 + u) \leq u $ के लिए $ u > -1$.
  \item गणना करना
        $\lim_{x \to +\infty} x^2 \eu^{-x}$,
        $\lim_{x \to +\infty} \frac{\ln x}{x}$
        (\cref{thm:g12:exp:growth}), और
        $\lim_{x \to 0} \frac{\eu^x - 1}{x}$ (पहचानें ए
        यौगिक).
  \item अध्ययन $ f(x) = x \ln x $ पर $\intoo{0}{+\infty}$:
        विविधताएं, \omterm{def:g10:functions:extrema}{न्यूनतम}, और सीमा पर $0^+$ (स्वीकार किया गया:
        $ x \ln x \to 0$). यह छोटा सा \omterm{def:g11:func:function}{समारोह} उपाय
        विश्वविद्यालय खंडों में सूचना और एन्ट्रापी।
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Doubling times and the rule of
72.}
\begin{enumerate}[resume]
  \item बचत बढ़ती है $3\,\%$ प्रति वर्ष. हल करना
        $1.03^n = 2$: पूंजी कब तक दोगुनी करनी है?
  \item बैंकर्स का अनुमान दोगुना होने का समय है
        $\frac{72}{\text{rate in }\%}$. नियम का परीक्षण करें
        $3\,\%$, $6\,\%$ और $9\,\%$ सटीक के विरुद्ध
        $\frac{\ln 2}{\ln(1 + r)}$, फिर इसे समझाएं: छोटे के लिए
        $ r $, $\ln(1 + r) \approx r $ (प्रश्न 3 की असमानता है
        कहानी का आधा हिस्सा), इसलिए सटीक स्थिरांक है
        $100 \ln 2 \approx 69.3$ ---बैंकर्स क्यों पसंद करते हैं?
        $72$?
  \item एक जीवाणु प्रत्येक को विभाजित करता है $20$ मिनट:
        $ p(t) = 2^{t/20}$ ($ t $ मिनटों में)। इसे इस रूप में पुनः लिखें
        $\eu^{\lambda t}$, फिर गणना करें $ p $ बाद $24$ घंटे।
        उत्तर (से अधिक) $10^{21}$) के बारे में क्या साबित करता है
        मॉडल --- और वास्तविक कालोनियों को क्या रोकता है?
  \item कैफीन लगभग आधे जीवन के साथ शरीर छोड़ देता है $5$
        घंटे। बाद एक $100$ 15:00 बजे एमजी कॉफी, कितनी
        23:00 बजे रहता है? यह कितने बजे नीचे गिरता है
        $10$ एमजी? (अनिद्रा है लोगारित्म.)
  \item एक यूरो पर $100\,\%$ वार्षिक ब्याज: गणना करें
        वार्षिक, मासिक और दैनिक के अंतर्गत वर्षांत पूंजी
        कंपाउंडिंग, और उसे \omterm{def:g10:functions:extrema}{अधिकतम} सीमा दें
        \cref{exo:g12:exp:9} अटूट साबित हुआ. जो प्रसिद्ध है
        शुद्ध लालच की सीमा स्थिर है?
  \item वैज्ञानिक साजिश क्यों रचते हैं? $\ln(\text{cases})$ विरुद्ध
        महामारी के प्रारंभिक चरण के दौरान का समय? ए क्या करता है?
        उस कथानक पर सीधी रेखा से पता चलता है, और उसका क्या मतलब है
        \omterm{def:g10:reffunc:affine}{ढलान} उपाय?
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The world's logarithmic scales.}
(लिखना $\log_{10} x = \frac{\ln x}{\ln 10}$.)
\begin{enumerate}[resume]
  \item डेसिबल में ध्वनि स्तर: $ L = 10 \log_{10}(I/I_0)$. ए
        बातचीत के उपाय $60$ डीबी, एक रॉक कॉन्सर्ट $120$ डीबी:
        ध्वनि की तीव्रता किस कारक से भिन्न होती है?
  \item भूकंप की तीव्रता बढ़ जाती है $1$ जब भूकंपीय
        आयाम को गुणा किया जाता है $10$, और जारी किया गया
        आयाम जैसे ऊर्जा पैमाने $^{3/2}$. तुलना करें
        परिमाण-$5$ और परिमाण-$7$ भूकंप: आयाम
        अनुपात, फिर ऊर्जा अनुपात।
  \item रसायन विज्ञान: $\text{pH} = -\log_{10}[\mathrm{H^+}]$.
        नींबू के रस का पीएच होता है $2$, दूध पीएच $6.5$: क्या है
        उनके एसिड सांद्रता का अनुपात?
  \item संगीत: प्रत्येक सप्तक दोगुना हो जाता है \omterm{def:g10:stats:series}{आवृत्ति}. से
        पियानो का निम्नतम A ($27.5$ हर्ट्ज) अपने उच्चतम सी तक
        ($4\,186$ हर्ट्ज), कीबोर्ड कितने सप्तक तक फैला है?
        और हमारी इंद्रियाँ ---सुनना, देखना, भूकंप-महसूस क्यों करती हैं
        --- लघुगणकीय पैमाने पसंद करते हैं? (एक वाक्य.)
  \item स्लाइड नियम, इंजीनियरों का कैलकुलेटर $350$
        वर्ष: दो छड़ियाँ स्नातक की उपाधि प्राप्त की ताकि \emph{लंबाई}
        निशान तक $ x $ के लिए आनुपातिक है $\ln x $. बताएं कैसे
        एक छड़ी को दूसरी छड़ी के साथ सरकाने से संख्याएँ कई गुना बढ़ जाती हैं,
        और की पहचान बताएं
        \cref{prop:g12:exp:lnalg} काम कर रहा हूँ.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- $\eu $'s fingerprints.}
\begin{enumerate}[resume]
  \item से \cref{exo:g12:exp:8}: द \omterm{def:g10:algebra:equation}{समीकरण}
        $\eu^x = kx $ ($ x > 0$) है $0$, $1$ या $2$ समाधान
        की स्थिति के अनुसार $ k $ ए के सापेक्ष
        दहलीज. परिणाम को दोबारा दोहराएं --- और सत्यापित करें
        इसके पीछे ज्यामितीय तथ्य: रेखा $ y = \eu x $ है
        बिलकुल \omterm{def:g12:deriv:tangent}{स्पर्शरेखा} के माध्यम से घातांक तक
        उत्पत्ति.
  \item निकट-चूक स्थिरांक: साथ $ n $ के लॉटरी टिकट
        जीतना \omterm{def:g10:proba:distribution}{संभावना} $\frac1n $ प्रत्येक,
        $\P(\text{no win}) = \left(1 - \frac1n\right)^n $.
        के माध्यम से इसकी सीमा की गणना करें
        $ n \ln\!\left(1 - \frac1n\right)$ (मानक का उपयोग करें
        सीमा $\frac{\ln(1 + u)}{u} \to 1$), और के साथ सामंजस्य स्थापित करें
        रहस्यमय $0.368$ का \cref{pb:g11:binom:1}.
  \item  $37\,\%$ नियम: सर्वोत्तम का चयन करना $ n $
        उम्मीदवारों का यादृच्छिक क्रम में साक्षात्कार लिया गया (वापस नहीं जाना है),
        इष्टतम रणनीति पहले को अस्वीकार करती है
        $\frac{n}{\eu} \approx 37\,\%$ और फिर पहला लेता है
        अब तक के सभी उम्मीदवारों से बेहतर उम्मीदवार---साथ सफल हो रहे हैं
        \omterm{def:g10:proba:distribution}{संभावना} के बारे में $\frac1\eu $. दोनों कहाँ हैं
        $\frac1\eu $ प्रश्न 18 और 19 से आते हैं ---
        सामान्य तंत्र बताएं (कई स्वतंत्र
        छोटा मौका घटनाएँ), और गणना करें $\frac1\eu $ तीन तक
        दशमलव
  \item समापन --- $\eu $ का चित्र: कंपाउंडिंग की छत
        (प्रश्न 10); आधार जिसका \omterm{def:g12:deriv:tangent}{स्पर्शरेखा} पर $0$ है \omterm{def:g10:reffunc:affine}{ढलान}
        बिल्कुल $1$ (प्रश्न 3 और 4); विकास नं
        बहुपद कैच (प्रश्न 4); का स्थिरांक
        निकट-चूक और इष्टतम रोक (प्रश्न 18-19);
        और इसका उलटा $\ln$, जो उत्पादों को रकम में बदल देता है
        (प्रश्न 16) और दशकों को इंच में (भाग III)। एक
        प्रत्येक वाक्य.
\end{enumerate}
\end{problem}
