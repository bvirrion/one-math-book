\chapter{अंकगणित}\label{ch:g12:arith}

\omterm{def:g12:seq:arith-geom}{अंकगणित} का अध्ययन करता है \omterm{def:g10:numbers:sets}{पूर्णांकों}: \omterm{def:g12:arith:divides}{भाजकत्व}, प्रमुख संख्या, अवशेष.
लंबे समय से इसे सबसे शुद्ध गणित माना जाता है, अब यह हर ऑनलाइन की सुरक्षा करता है
भुगतान: आरएसए क्रिप्टोसिस्टम बेज़आउट, गॉस और के प्रमेय पर आधारित है
इस अध्याय में फ़र्मेट \omterm{def:g11:vect:det}{सिद्ध} हुआ।

\section{विभाज्यता और यूक्लिडियन विभाजन}

\begin{definition}[भाजकत्व]\label{def:g12:arith:divides}
होने देना $a, b \in \Z $. हम कहते हैं $ b $ \emph{विभाजित} $ a $\index{भाजकत्व}, लिखा हुआ
$ b \mid a $, यदि मौजूद है $ k \in \Z $ साथ $ a = kb $. हम भी कहते हैं $ a $ एक है
\emph{एकाधिक} का $ b $.
\end{definition}

\begin{proposition}\label{prop:g12:arith:divprops}
अगर $ c \mid a $ और $ c \mid b $, तब $ c $ विभाजित प्रत्येक पूर्णांक संयोजन
$ au + bv $ ($ u, v \in \Z $). अगर $ a \mid b $ और $ b \mid a $ साथ
$ a,b \in \N $, तब $ a = b $. अगर $ a \mid b $ और $ b \neq 0$, तब $\abs a \leq \abs b $.
\end{proposition}

\begin{proof}
लिखना $ a = kc $, $ b = lc $: तब $ au + bv = (ku + lv)c $. अन्य बिंदु
से अनुसरण करें $\abs{a} = \abs{k}\,\abs{b}$ साथ $\abs k \geq 1$ कब
$ b = ka \neq 0$.
\end{proof}

\begin{theorem}[यूक्लिडियन प्रभाग]\label{thm:g12:arith:euclid}
\index{यूक्लिडियन प्रभाग}
होने देना $ a \in \Z $ और $ b \in \N^*$. वहाँ एक अनोखी \omterm{def:g10:vectors:sum}{जोड़}ी मौजूद है
$(q, r) \in \Z \times \N $ ऐसा है कि
\[
a = bq + r \qquad\text{and}\qquad 0 \leq r < b .
\]
$ q $ है \emph{भागफल} और $ r $  \emph{शेष}.
\end{theorem}

\begin{proof}
\emph{अस्तित्व।} के गुणजों का समुच्चय $ b $ जो निम्न से अधिक नहीं है $ a $ एक है
सबसे बड़ा तत्व $ bq $ (यह गैर-रिक्त है और \omterm{def:g12:seq:bounded}{ऊपर से घिरा हुआ}); तय करना $ r = a - bq $.
अधिकतमता से, $ b(q+1) > a $, इसलिए $0 \leq r < b $.
\emph{विशिष्टता.} अगर $ bq + r = bq' + r'$ साथ $0 \leq r, r' < b $, तब
$ b(q - q') = r' - r $ और $\abs{r' - r} < b $: का एक गुणज $ b $ का \omterm{def:g10:numbers:abs}{निरपेक्ष
मूल्य} से कम $ b $ होना ही चाहिए $0$, इसलिए $ r = r'$ और $ q = q'$.
\end{proof}

\section{सर्वांगसमताएँ}

\begin{definition}[अनुरूपता]\label{def:g12:arith:congruence}
होने देना $ n \in \N^*$. दो \omterm{def:g10:numbers:sets}{पूर्णांकों} $ a, b $ हैं \emph{सर्वांगसम मॉड्यूलो
$ n $}\index{सर्वांग समन्वितो
$ n $}, लिखा हुआ $ a \equiv b \pmod n $, अगर $ n \mid (a - b)$
--- समान रूप से, यदि $ a $ और $ b $ में समान शेषफल है \omterm{thm:g12:arith:euclid}{यूक्लिडियन
प्रभाग} द्वारा $ n $.
\end{definition}

\begin{proposition}[संचालन के साथ अनुकूलता]\label{prop:g12:arith:congops}
अगर $ a \equiv b \pmod n $ और $ c \equiv d \pmod n $, तब
\[
a + c \equiv b + d, \qquad
ac \equiv bd, \qquad
a^k \equiv b^k \ (k \in \N) \pmod n .
\]
\end{proposition}

\begin{proof}
$ n $ विभाजित $(a-b) + (c-d) = (a+c) - (b+d)$, और
$ ac - bd = a(c - d) + d(a - b)$ का गुणज भी है $ n $. शक्ति नियम
उत्पाद नियम से प्रेरण द्वारा अनुसरण किया जाता है।
\end{proof}

\begin{method}[कंप्यूटिंग शक्तियाँ मॉड्यूलो $ n $]\label{met:g12:arith:powers}
गणना करना $ a^k \bmod n $, बेस मॉड्यूलो को कम करें $ n $, फिर एक छोटे की तलाश करें
की शक्ति $ a $ के अनुरूप $\pm1$, और इसका उपयोग घातांक को संक्षिप्त करने के लिए करें। उदाहरण के लिए $2^{100} \bmod 7$: तब से $2^3 = 8 \equiv 1 \pmod 7$ और
$100 = 3\times33 + 1$,
\[
2^{100} = \left(2^{3}\right)^{33} \times 2 \equiv 1^{33}\times 2 = 2 \pmod 7 .
\]
\end{method}

\section{जीसीडी, बेज़आउट और गॉस}

\begin{definition}[जीसीडी]\label{def:g12:arith:gcd}
होने देना $ a, b $ होना \omterm{def:g10:numbers:sets}{पूर्णांकों}, दोनों शून्य नहीं. \emph{महत्तम सामान्य भाजक}\index{महत्तम सामान्य भाजक} $\gcd(a, b)$ सबसे बड़ा है पूर्णांक दोनों को बांटना $ a $
और $ b $. कब $\gcd(a,b) = 1$, $ a $ और $ b $ कहा जाता है
\emph{सह अभाज्य}\index{सह अभाज्य}.
\end{definition}

\begin{proposition}[यूक्लिड का एल्गोरिदम]\label{prop:g12:arith:euclidalgo}
\index{यूक्लिड का एल्गोरिदम}
अगर $ a = bq + r $ ($ b \neq 0$), तब $\gcd(a, b) = \gcd(b, r)$. पुनरावृत्त करना
\omterm{thm:g12:arith:euclid}{यूक्लिडियन प्रभाग} इसलिए गणना करता है $\gcd(a,b)$: द जी.सी.डी आखिरी है
शून्येतर शेष.
\end{proposition}

\begin{proof}
का कोई भी सामान्य भाजक $ a $ और $ b $ विभाजित $ r = a - bq $
(\cref{prop:g12:arith:divprops}), इसलिए यह एक सामान्य भाजक है $ b $ और $ r $;
और इसके विपरीत, तब से $ a = bq + r $. दोनों \omterm{def:g10:vectors:sum}{जोड़}ियों में समान समानता है
भाजक, तो वही जी.सी.डी. एल्गोरिथ्म शेष के कारण समाप्त हो जाता है
एक सख्ती से फार्म \omterm{def:g10:functions:variations}{घटते} \omterm{def:g12:seq:sequence}{अनुक्रम} गैरनकारात्मक का \omterm{def:g10:numbers:sets}{पूर्णांकों}.
\end{proof}

\begin{example}\label{ex:g12:arith:euclidalgo}
$\gcd(252, 198)$: $252 = 198 + 54$; $198 = 3\times54 + 36$;
$54 = 36 + 18$; $36 = 2 \times 18 + 0$. इस तरह $\gcd(252,198) = 18$.
\end{example}

\begin{theorem}[बेज़ौट की पहचान]\label{thm:g12:arith:bezout}
\index{बेज़ौट की पहचान}
होने देना $ a, b $ होना \omterm{def:g10:numbers:sets}{पूर्णांकों}, दोनों शून्य नहीं, और $ d = \gcd(a,b)$. वहां है
$ u, v \in \Z $ ऐसा है कि
\[
au + bv = d .
\]
विशेष रूप से, $ a $ और $ b $ हैं \omterm{def:g12:arith:gcd}{सह अभाज्य} अगर और केवल अगर $ au + bv = 1$ कुछ के लिए
\omterm{def:g10:numbers:sets}{पूर्णांकों} $ u, v $.
\end{theorem}

\begin{proof}
दौड़ना \omterm{prop:g12:arith:euclidalgo}{यूक्लिड का एल्गोरिदम} पीछे की ओर: प्रत्येक शेषफल एक है पूर्णांक संयोजन
पिछले दो में से, और प्रारंभिक डेटा $ a, b $ के संयोजन हैं
स्वयं; अवरोही प्रतिस्थापन द्वारा, अंतिम शून्येतर शेषफल $ d $ एक
पूर्णांक की संयोजन $ a $ और $ b $. (में \cref{ex:g12:arith:euclidalgo}:
$18 = 54 - 36 = 54 - (198 - 3\times54) = 4\times54 - 198 =
4(252 - 198) - 198 = 4\times252 - 5\times198$.)

समतुल्यता के लिए: यदि $\gcd(a,b) = 1$, बेज़आउट प्रदान करता है $ u, v $; इसके विपरीत,
का कोई भी सामान्य भाजक $ a $ और $ b $ विभाजित $ au + bv = 1$, मजबूर करना
$\gcd(a,b) = 1$.
\end{proof}

\begin{theorem}[गॉस की लेम्मा]\label{thm:g12:arith:gauss}
\index{गॉस की लेम्मा}
होने देना $ a, b, c \in \Z $. अगर $ a \mid bc $ और $\gcd(a, b) = 1$, तब $ a \mid c $.
\end{theorem}

\begin{proof}
बेज़आउट देता है $ au + bv = 1$; गुणा करके $ c $: $ acu + bcv = c $. की दोनों शर्तें
बायीं ओर के गुणज हैं $ a $ (दूसरा क्योंकि $ a \mid bc $), इसलिए
ऐसा ही है $ c $.
\end{proof}

\begin{corollary}\label{cor:g12:arith:coprimeprod}
अगर $ a \mid c $, $ b \mid c $ और $\gcd(a,b) = 1$, तब $ ab \mid c $.
\end{corollary}

\begin{proof}
लिखना $ c = ak $. से $ b \mid ak $ और $\gcd(a,b)=1$, गॉस देता है
$ b \mid k $, कहना $ k = bl $; तब $ c = abl $.
\end{proof}

\section{प्रमुख संख्या}

\begin{definition}[मुख्य]\label{def:g12:arith:prime}
एक पूर्णांक $ p \geq 2$ है \emph{मुख्य}\index{मुख्य} यदि यह केवल है
सकारात्मक विभाजक हैं $1$ और $ p $.
\end{definition}

\begin{proposition}\label{prop:g12:arith:primedivides}
प्रत्येक पूर्णांक $ n \geq 2$ एक \omterm{def:g12:arith:prime}{मुख्य} भाजक; अगर $ n $ क्या नहीं है \omterm{def:g12:arith:prime}{मुख्य}, यह है एक
\omterm{def:g12:arith:prime}{मुख्य} भाजक $\leq \sqrt n $. यदि एक \omterm{def:g12:arith:prime}{मुख्य} $ p $ विभाजित एक उत्पाद $ ab $, तब
$ p \mid a $ या $ p \mid b $ (\emph{यूक्लिड की प्रमेयिका}).
\end{proposition}

\begin{proof}
सबसे छोटा भाजक $ d \geq 2$ का $ n $ है \omterm{def:g12:arith:prime}{मुख्य} (कोई भी उचित विभाजक $ d $
का एक छोटा विभाजक होगा $ n $). अगर $ n = de $ के साथ मिश्रित है
$2 \leq d \leq e $, तब $ d^2 \leq de = n $, इसलिए $ d \leq \sqrt n $. यूक्लिड के लिए
लेम्मा: यदि $ p \nmid a $, तब $\gcd(p, a) = 1$ (के एकमात्र विभाजक $ p $ हैं
$1$ और $ p $), और \omterm{thm:g12:arith:gauss}{गॉस की लेम्मा} देती है $ p \mid b $.
\end{proof}

\begin{theorem}[यूक्लिड]\label{thm:g12:arith:infiniteprimes}
अनन्त रूप से अनेक हैं अभाज्य.
\end{theorem}

\begin{proof}
किसी भी सीमित सूची को देखते हुए $ p_1, \dots, p_k $ का अभाज्य, विचार करना
$ N = p_1 p_2 \cdots p_k + 1$. कुछ \omterm{def:g12:arith:prime}{मुख्य} $ p $ विभाजित $ N $; लेकिन कोई नहीं $ p_i $
विभाजित $ N $ (शेष है $1$), इसलिए $ p $ एक है \omterm{def:g12:arith:prime}{मुख्य} सूची में नहीं. नहीं
परिमित सूची समाप्त हो जाती है अभाज्य.
\end{proof}

\begin{theorem}[अंकगणित का मौलिक प्रमेय]\label{thm:g12:arith:fta}
\index{अंकगणित का मौलिक प्रमेय}
प्रत्येक पूर्णांक $ n \geq 2$ का एक उत्पाद है अभाज्य, और यह गुणनखंडन है
कारकों के क्रम तक अद्वितीय:
\[
n = p_1^{\alpha_1} p_2^{\alpha_2} \cdots p_r^{\alpha_r},
\qquad p_1 < p_2 < \dots < p_r \text{ primes},\ \alpha_i \geq 1 .
\]
\end{theorem}

\begin{proof}
\emph{अस्तित्व}, मजबूत प्रेरण द्वारा: $ n $ \omterm{def:g12:arith:prime}{मुख्य} इसका अपना कारकीकरण है;
अन्यथा $ n = de $ साथ $2 \leq d, e < n $, और प्रेरण द्वारा दोनों कारक
परिकल्पना. \emph{विशिष्टता}: कल्पना करना
$ p_1\cdots p_s = q_1 \cdots q_t $ (अभाज्य, दोहराव की अनुमति के साथ)। द्वारा
यूक्लिड की लेम्मा, $ p_1$ विभाजित कुछ $ q_j $, और होना \omterm{def:g12:arith:prime}{मुख्य}, $ p_1 = q_j $;
रद्द करें और दोहराएँ. दोनों गुणनखंड पद दर पद मेल खाते हैं।
\end{proof}

\begin{theorem}[फ़र्मेट का छोटा प्रमेय]\label{thm:g12:arith:fermat}
\index{फ़र्मेट का छोटा प्रमेय}
होने देना $ p $ होना \omterm{def:g12:arith:prime}{मुख्य} और $ a \in \Z $ साथ $ p \nmid a $. तब
\[
a^{p-1} \equiv 1 \pmod p .
\]
हरएक के लिए $ a \in \Z $ (कोई सहप्रधानता नहीं मानी गई), $ a^p \equiv a \pmod p $.
\end{theorem}

\begin{proof}
पर विचार करें $ p - 1$ \omterm{def:g10:numbers:sets}{पूर्णांकों} $ a, 2a, 3a, \dots, (p-1)a $ सापेक्ष $ p $. कोई नहीं
है $\equiv 0$ (अगर $ p \mid ka $ साथ $1 \leq k \leq p-1$, यूक्लिड की लेम्मा बल
$ p \mid k $, असंभव), और वे \omterm{def:g10:vectors:sum}{जोड़}ीवार अलग मॉड्यूलो हैं $ p $ (अगर
$ ka \equiv la $, तब $ p \mid (k - l)a $, इसलिए $ p \mid k - l $, इसलिए $ k = l $).
इसलिए, मॉड्यूलो $ p $, वे संख्याएँ हैं $1, 2, \dots, p-1$ किसी क्रम में.
सभी को गुणा करना सर्वांगसमताएँ:
\[
a^{p-1}\,(p-1)! \equiv (p-1)! \pmod p .
\]
तब से $ p $ विभाजित कुछ भी नहीं $1, \dots, p-1$, यूक्लिड के लेम्मा का बार-बार उपयोग
रद्द करने की अनुमति देता है $(p-1)!$, जा रहा हूँ $ a^{p-1} \equiv 1$. दूसरा रूप
से गुणा करके अनुसरण करता है $ a $ (और जब यह तुच्छ है $ p \mid a $).
\end{proof}

\begin{example}[क्रिप्टोग्राफी के लिए आवेदन]\label{ex:g12:arith:rsa}
फ़र्मेट का प्रमेय घातांक को मॉड्यूलो बनाता है $ n $ प्रतिवर्ती जब
प्रतिपादकों को उपयुक्त रूप से चुना जाता है --- का हृदय \emph{आरएसए} क्रिप्टोसिस्टम।
साथ $ p, q $ बड़ा अभाज्य और $ n = pq $, एक प्रकाशित करता है $ n $ और एक प्रतिपादक
$ e $; एन्क्रिप्शन है $ x \mapsto x^e \bmod n $. डिक्रिप्टिंग के लिए एक प्रतिपादक की आवश्यकता होती है
$ d $ साथ $ ed \equiv 1 \pmod{(p-1)(q-1)}$, जो केवल कोई ही जानता है $ p $
और $ q $ --- और पुनर्प्राप्ति की गणना कर सकते हैं $ p, q $ से $ n $ मतलब \omterm{def:g10:algebra:expand}{फैक्टरिंग} ए
सैकड़ों अंकों की संख्या, जो कोई भी ज्ञात एल्गोरिदम उचित रूप से नहीं करता है
समय.
\end{example}

\begin{exercise}[$\star $]\label{exo:g12:arith:3}
का उपयोग करते हुए \omterm{prop:g12:arith:euclidalgo}{यूक्लिड का एल्गोरिदम}, गणना करें $\gcd(1071, 462)$, और खोजें \omterm{def:g10:numbers:sets}{पूर्णांकों}
$ u, v $ साथ $1071u + 462v = \gcd(1071, 462)$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g12:arith:4}
इसे हर एक के लिए दिखाएँ $ n \in \Z $, $ n^2$ के अनुरूप है $0$ या $1$ सापेक्ष
$4$. यह निष्कर्ष निकालें कि ए पूर्णांक $\equiv 3 \pmod 4$ कभी भी दो का योग नहीं होता
वर्ग।
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:arith:5}
वह सभी को दिखाओ $ n \in \N $, $ n(n+1)(2n+1)$ से विभाज्य है $6$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:arith:6}
में हल करें $\Z $  अनुरूपता $5x \equiv 3 \pmod{11}$.
(संकेत: इसका व्युत्क्रम ज्ञात कीजिए $5$ सापेक्ष $11$.)
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:arith:7}
में हल करें $\Z \times \Z $ डायोफैंटाइन \omterm{def:g10:algebra:equation}{समीकरण}
\[
17x - 40y = 1,
\]
फिर सभी समाधानों का वर्णन करें $17x - 40y = 6$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:arith:8}
बताते हैं कि $\sqrt2$ है \omterm{ex:g10:numbers:classify}{तर्कहीन}, की विशिष्टता का उपयोग करते हुए \omterm{def:g12:arith:prime}{मुख्य}
गुणनखंडन (के घातांक की तुलना करें $2$ के दोनों तरफ $ a^2 = 2b^2$).
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g12:arith:9}
होने देना $ p $ एक हो \omterm{def:g12:arith:prime}{मुख्य}.
\begin{enumerate}
  \item उसके लिए दिखाओ $1 \leq k \leq p - 1$, $ p $ विभाजित $\dbinom{p}{k}$.
        (संकेत: उपयोग करें $ k\binom pk = p\binom{p-1}{k-1}$, \cref{exo:g12:comb:7},
        और \omterm{thm:g12:arith:gauss}{गॉस की लेम्मा}।)
  \item प्रेरण द्वारा, निष्कर्ष निकालें $ a \geq 0$, फ़र्मेट के छोटे होने का एक और प्रमाण
        प्रपत्र में प्रमेय $ a^p \equiv a \pmod p $.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g12:arith:10}
\emph{(चीनी शेष समस्या.)} सभी खोजें \omterm{def:g10:numbers:sets}{पूर्णांकों} $ n $ ऐसा है कि
\[
n \equiv 2 \pmod 3, \qquad n \equiv 3 \pmod 5, \qquad n \equiv 2 \pmod 7 .
\]
(संकेत: पहले दो शर्तों को हल करें, फिर तीसरी को शामिल करें; बेज़आउट
गुणांक मदद करते हैं।)
\end{exercise}

\section{समस्या: गुप्त कोड और जाँच अंक}

\begin{problem}[{सप्ताहांत समस्या --- सर्वांगसमताएं हर की रक्षा करती हैं बारकोड और क्रेडिट कार्ड, और फ़र्मेट का छोटा प्रमेय चलता है दुनिया के रहस्यों पर ताला}]
\label{pb:g12:arith:1}
जी।\,एच.~हार्डी ने 1940 में दावा किया था कि संख्या \omterm{def:g11:vect:det}{सिद्ध}ांत ``निर्दोष'' था
अनुप्रयोगों द्वारा. अस्सी साल बाद, हर बारकोड बीप, हर
क्रेडिट-कार्ड से भुगतान और प्रत्येक एन्क्रिप्टेड संदेश उसका खंडन करता है
--- बिल्कुल इस अध्याय के टूल के साथ: सर्वांगसमताएँ
(\cref{prop:g12:arith:congops}), बीézout व्युत्क्रम
(\cref{thm:g12:arith:bezout}) और फ़र्मेट की छोटी प्रमेय
(\cref{exo:g12:arith:9}). यह समस्या कोड की जाँच करती है, a को तोड़ देती है
ताले का खिलौना संस्करण, और सीखता है कि असली ताला क्यों लगा रहता है।

\medskip
\noindent\textbf{Part I --- अनुरूपता fluency.}
\begin{enumerate}
  \item गणना करना $2026 \bmod 7$; फिर अंतिम अंक
        $7^{100}$ (की शक्तियों का चक्र ज्ञात कीजिए $7$ सापेक्ष
        $10$).
  \item तेज़ घातांक (\cref{met:g12:arith:powers}):
        गणना करें $5^{117} \bmod 13$ (से शुरु करें
        $5^2 \equiv -1$).
  \item हल करना $3x \equiv 5 \pmod 7$.
  \item दौड़ना \omterm{prop:g12:arith:euclidalgo}{यूक्लिड का एल्गोरिदम} पर $(97, 35)$, बैक-स्थानापन्न
        खोजने के लिए \omterm{def:g10:numbers:sets}{पूर्णांकों} $ u, v $ साथ $97u + 35v = 1$, और
        का व्युत्क्रम निकालें $35$ सापेक्ष $97$.
  \item सटीक रूप से बताएं कि कब $ a $ व्युत्क्रमणीय मॉड्यूलो है $ n $, और
        कौन सा प्रमेय व्युत्क्रम प्रदान करता है।
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Check digits.}
\begin{enumerate}[resume]
  \item आईएसबीएन-10: दस अंक $ d_1 \dots d_{10}$ एक किताब का
        कोड को संतुष्ट करना होगा
        $10d_1 + 9d_2 + \dots + 2d_9 + 1d_{10} \equiv 0
        \pmod{11}$. वास्तविक आईएसबीएन सत्यापित करें
        $0\,306\,40615\,2$.
  \item साबित करें कि आईएसबीएन योजना पता लगाती है \emph{प्रत्येक}
        एकल-अंकीय त्रुटि: यदि एक अंक बदलता है
        $ d \not\equiv 0$, भारित योग बदल जाता है $ w d $
        साथ $1 \leq w \leq 10$ --- ऐसा कभी क्यों नहीं हो सकता
        $\equiv 0 \pmod{11}$
        (\cref{thm:g12:arith:gauss})?
  \item साबित करें कि यह दो के प्रत्येक स्थानान्तरण का भी पता लगाता है
        आसन्न (अलग) अंक. फिर डिज़ाइन समझाएं
        रहस्य: किसकी संपत्ति $11$ दोनों प्रमाणों को कार्यान्वित किया,
        और क्या गलत हो सकता है \omterm{def:g12:complex:modulus}{मापांक} $10$?
  \item EAN-13 बारकोड अंकों पर भार डालते हैं $1, 3, 1, 3, \dots $
        सापेक्ष $10$. चेक अंक पूरा होने की गणना करें
        $978\,2940199\,05$. कौन सा आसन्न स्थानान्तरण करता है
        ईएएन \emph{असफल} भांप लेना? (कब है
        $2(a - b) \equiv 0 \pmod{10}$?)
  \item क्रेडिट कार्ड लुहान की योजना का उपयोग करते हैं: दाईं ओर से, दोगुना
        हर दूसरा अंक (घटाना $9$ जब डबल
        से अधिक है $9$), हर चीज़ का योग करें, और उसके गुणज की आवश्यकता है
        $10$. परीक्षण संख्या सत्यापित करें
        $4539\,1488\,0343\,6467$.
  \item एक वाक्य में: क्या किया \omterm{def:g12:arith:prime}{मुख्य} \omterm{def:g12:complex:modulus}{मापांक} आईएसबीएन खरीदें
        वह ईएएन और लुहान, जंजीर से बंधे हुए हैं $10$, क्या नहीं हो सकता?
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Fermat's lock.}
\begin{enumerate}[resume]
  \item खजाने से पहले एक जाल: गणना करें
        $2^{10} \bmod 341$, निष्कर्ष निकालना $2^{340} \bmod 341$ --- और
        फिर कारक $341$. यह उदाहरण क्या दर्शाता है (a
        \emph{फ़र्मेट स्यूडोप्राइम}) फ़र्मेट के उपयोग के बारे में कहें
        प्रारंभिक परीक्षण के रूप में छोटा प्रमेय?
  \item लघु रूप में आरएसए: ले लो $ p = 3$, $ q = 11$, इसलिए
        $ n = 33$ और $(p-1)(q-1) = 20$; सार्वजनिक प्रतिपादक है
        $ e = 3$. निजी प्रतिपादक खोजें $ d $ साथ
        $3d \equiv 1 \pmod{20}$ (प्रश्न 4 की विधि).
  \item संदेश एन्क्रिप्ट करें $ m = 4$: गणना करें
        $ c = m^3 \bmod 33$.
  \item डिक्रिप्ट: गणना करें $ c^d \bmod 33$ (उपयोग
        $ c \equiv -2 \pmod{33}$) और संदेश पुनर्प्राप्त करें।
  \item डिक्रिप्शन हमेशा काम क्यों करता है: यह दिखाएँ
        $ m^{21} \equiv m $ दोनों मॉड्यूलो $3$ और मॉड्यूलो $11$
        (प्रत्येक विश्व में फ़र्मेट का छोटा प्रमेय), और निष्कर्ष निकालें
        modulo $33$ (\cref{thm:g12:arith:gauss} दोनों को चिपका देता है
        सर्वांगसमताएँ). जहाँ किया $21 = ed $ का विशेष रूप
        $1 + 20k $ प्रवेश करना?
  \item ताले की सुरक्षा: हर कोई जानता है $ n $ और $ e $;
        ठीक हो $ d $ आवश्यक है $(p-1)(q-1)$, इसलिए कारक
        का $ n $. हमारा $33$ नजर में कारक --- ऐसा ही क्यों होता है
        योजना, के साथ $ n $ छह सौ अंकों की, रक्षा करें
        विश्व के बैंक? (बीच की विषमता पर एक वाक्य
        गुणा करना और \omterm{def:g10:algebra:expand}{फैक्टरिंग}.)
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Classics.}
\begin{enumerate}[resume]
  \item पुराने चीनी सैनिकों की गिनती (तुलना करें)
        \cref{exo:g12:arith:10}): कई सैनिक चले जाते हैं
        शेष $2$ जब रैंक किया गया $3$ और शेष $3$ कब
        द्वारा रैंक किया गया $5$. सभी संभावित गणनाएँ ढूँढ़ें और समझाएँ
        उत्तर अद्वितीय मॉड्यूलो क्यों है? $15$.
  \item अंत में एक-पंक्ति प्रमाण: से $10 \equiv 1 \pmod 9$,
        \omterm{def:g11:vect:det}{सिद्ध} करें कि प्रत्येक संख्या उसके अंकों के योग के सर्वांगसम होती है
        modulo $9$; से $10 \equiv -1 \pmod{11}$, प्राप्त करें
        के लिए प्रत्यावर्ती-योग नियम $11$. (मिडिल स्कूल
        वॉल्यूम ने इन्हें स्पष्ट बीजगणित के साथ \omterm{def:g11:vect:det}{सिद्ध} किया --- प्रशंसा करें
        संपीड़न.)
  \item समापन --- बारकोड के विरुद्ध हार्डी: पुनर्कथन
        अध्याय का टूलकिट (अनुरूपता \omterm{def:g12:seq:arith-geom}{अंकगणित}, बीézout
        व्युत्क्रम, फ़र्मेट का छोटा प्रमेय, ग्लूइंग \omterm{def:g12:arith:gcd}{सह अभाज्य}
        मॉड्यूली) और इसमें प्रत्येक ने अपनी जगह कहां क्लिक की
        समस्या; फिर आधुनिक फैसला सुनाओ
        ``बेदाग''।
\end{enumerate}
\end{problem}
