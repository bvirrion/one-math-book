\chapter{अंतरिक्ष में सदिश, रेखाएँ और तल}\label{ch:g12:space}

त्रि-आयामी ज्यामिति एक बार कम्प्यूटेशनल बन जाती है वैक्टर और
\omterm{def:g10:coordgeom:system}{COORDINATES} उपलब्ध हैं: रेखाओं को पैरामीट्रिक प्रतिनिधित्व मिलता है, विमानों को मिलता है
कार्तीय \omterm{def:g10:algebra:equation}{समीकरण}, और यह \omterm{def:g12:space:dot}{अदिश उत्पाद} कोणों और दूरियों को मापता है।
यह अध्याय इस टूलबॉक्स का निर्माण करता है और शास्त्रीय समाधान के लिए इसका उपयोग करता है
\omterm{def:g10:numbers:interunion}{चौराहा} और दूरी की समस्या.

\section{अंतरिक्ष में सदिश}

वैक्टर अंतरिक्ष में विमान के समान नियमों का पालन करें: वे जोड़ते हैं, वे मापते हैं,
और $\vect{AB} = \vect{CD}$ बिल्कुल कब $ ABDC $ एक समांतर चतुर्भुज है. में एक
\omterm{def:g10:coordgeom:system}{निर्देशांक तरीका} $(O; \vec\imath, \vec\jmath, \vec k)$, प्रत्येक \omterm{def:g11:vect:vector}{वेक्टर} है
\omterm{def:g10:coordgeom:system}{COORDINATES} $(x, y, z)$.

\begin{definition}[संरेखता, समतलीयता]\label{def:g12:space:collinear}
दो वैक्टर हैं \emph{\omterm{def:g11:vect:collinear}{समरेख}} यदि एक दूसरे का गुणज है। तीन
वैक्टर $\vec u, \vec v, \vec w $ हैं \emph{समतलीय}\index{समतलीय} यदि एक
उनमें से अन्य दो के संयोजन के रूप में लिखा जा सकता है, मान लीजिए
$\vec w = a\vec u + b\vec v $.
\end{definition}

\begin{proposition}\label{prop:g12:space:basis}
अगर $\vec u, \vec v, \vec w $ नहीं हैं \omterm{def:g12:space:collinear}{समतलीय}, प्रत्येक \omterm{def:g11:vect:vector}{वेक्टर} अंतरिक्ष का
के रूप में विशिष्ट रूप से विघटित होता है $ x\vec u + y\vec v + z\vec w $.
\end{proposition}

\begin{proof}[प्रमाण का विचार]
की नोक के माध्यम से \omterm{def:g11:vect:vector}{वेक्टर} विघटित करने के लिए समान्तर रेखा खींचिए
$\vec w $; यह के विमान से मिलता है $\vec u, \vec v $ एक ही बिंदु में, जो
को विभाजित करता है \omterm{def:g11:vect:vector}{वेक्टर} उस विमान में एक घटक में (विशिष्ट रूप से)। $ x\vec u + y\vec
v $, समतल ज्यामिति) और साथ में एक घटक $\vec w $. विशिष्टता: एक अंतर
दो विघटनों में से एक को व्यक्त किया जाएगा वैक्टर दूसरे के संदर्भ में
दो, गैर-सहतलीयता का खंडन।
\end{proof}

\section{रेखाएँ और तल}

\begin{definition}[एक रेखा का पैरामीट्रिक प्रतिनिधित्व]\label{def:g12:space:line}
के माध्यम से लाइन $ A(x_A, y_A, z_A)$ साथ \omterm{def:g11:vect:direction}{दिशा सदिश}
$\vec u (a, b, c) \neq \vec 0$ बिंदुओं का समूह है
\[
M(x_A + ta,\; y_A + tb,\; z_A + tc), \qquad t \in \R .
\]
\end{definition}

\begin{definition}[विमान]\label{def:g12:space:plane}
विमान के माध्यम से $ A $ दो द्वारा निर्देशित गैर \omterm{def:g11:vect:collinear}{समरेख} वैक्टर
$\vec u, \vec v $ बिंदुओं का समूह है $ M $ साथ
$\vect{AM} = s\vec u + t\vec v $, $(s, t) \in \R^2$.
\end{definition}

\begin{proposition}[सापेक्ष स्थिति]\label{prop:g12:space:positions}
दो अलग-अलग तल या तो समानांतर हैं या एक रेखा में प्रतिच्छेद करते हैं। एक पंक्ति और एक
समतल या तो समानांतर होते हैं (संभवतः समाहित होते हैं) या एक में प्रतिच्छेद करते हैं
बिंदु. अंतरिक्ष में दो रेखाएं प्रतिच्छेद कर सकती हैं, बिल्कुल समानांतर, समान,
या \emph{तिरछा} (गैर \omterm{def:g12:space:collinear}{समतलीय}).
\end{proposition}

\begin{proof}[स्केच]
ये घटना विवरण हैं; प्रत्येक मामले का विश्लेषण कम हो जाता है \omterm{def:g10:coordgeom:system}{COORDINATES},
हल करने के लिए ए रैखिक प्रणाली और उसके समाधान गिनना---देखें
\cref{met:g12:space:intersections}.
\end{proof}

\section{अदिश गुणनफल}

\begin{definition}[अदिश गुणनफल]\label{def:g12:space:dot}
 \emph{अदिश उत्पाद}\index{अदिश उत्पाद} का $\vec u $ और $\vec v $ है
\[
\vec u \cdot \vec v = \tfrac12\left(\norm{\vec u + \vec v}^2
- \norm{\vec u}^2 - \norm{\vec v}^2\right),
\]
कहाँ $\norm{\vec u}$ की लम्बाई है $\vec u $. अगर दोनों वैक्टर हैं
शून्येतर, $\vec u \cdot \vec v = \norm{\vec u}\,\norm{\vec v}\cos\theta $
कहाँ $\theta $ उनके बीच का कोण है; और एक में \emph{\omterm{def:g10:coordgeom:system}{ऑर्थोनॉर्मल}}
\omterm{def:g10:coordgeom:system}{निर्देशांक तरीका},
\[
\vec u \cdot \vec v = xx' + yy' + zz' .
\]
दो वैक्टर हैं \emph{\omterm{def:g11:scal:orthogonal}{ओर्थोगोनल}} यदि उनका अदिश गुणनफल है $0$.
\end{definition}

\begin{proposition}[गुण]\label{prop:g12:space:dotprops}
 \omterm{def:g12:space:dot}{अदिश उत्पाद} सममित है ($\vec u\cdot\vec v = \vec v\cdot\vec u $),
द्विरेखीय ($(a\vec u + b\vec v)\cdot \vec w = a\,\vec u\cdot\vec w +
b\,\vec v\cdot \vec w $), और
$\vec u \cdot \vec u = \norm{\vec u}^2 \geq 0$.
\end{proposition}

\begin{proof}
एक में \omterm{def:g10:coordgeom:system}{ऑर्थोनॉर्मल} प्रणाली, सभी तीन गुण सूत्र पर तत्काल हैं
$ xx' + yy' + zz'$, जो स्वयं परिभाषित सूत्र से अनुसरण करता है और
लंबाई की पायथागॉरियन गणना:
$\norm{\vec u}^2 = x^2 + y^2 + z^2$.
\end{proof}

\begin{definition}[सामान्य वेक्टर]\label{def:g12:space:normal}
ए \emph{सामान्य वेक्टर}\index{सामान्य वेक्टर} एक विमान का $\mathcal P $ एक है
शून्येतर \omterm{def:g11:vect:vector}{वेक्टर} \omterm{def:g11:scal:orthogonal}{ओर्थोगोनल} प्रत्येक को \omterm{def:g11:vect:vector}{वेक्टर} में लेटा हुआ $\mathcal P $ (यह
इतना ही काफी है \omterm{def:g11:scal:orthogonal}{ओर्थोगोनल} दो को गैर \omterm{def:g11:vect:collinear}{समरेख} के निर्देश
$\mathcal P $).
\end{definition}

\begin{theorem}[एक समतल का कार्तीय समीकरण]\label{thm:g12:space:planeeq}
एक में \omterm{def:g10:coordgeom:system}{ऑर्थोनॉर्मल} \omterm{def:g10:coordgeom:system}{निर्देशांक तरीका}, विमान के माध्यम से
$ A $ साथ \omterm{def:g12:space:normal}{सामान्य वेक्टर} $\vec n(a, b, c)$ है \omterm{def:g10:algebra:equation}{समीकरण}
\[
a x + b y + c z + d = 0,
\]
कहाँ $ d = -(ax_A + by_A + cz_A)$. इसके विपरीत, प्रत्येक ऐसे \omterm{def:g10:algebra:equation}{समीकरण} साथ
$(a,b,c) \neq (0,0,0)$ एक विमान को परिभाषित करता है \omterm{def:g12:space:normal}{सामान्य वेक्टर} $(a, b, c)$.
\end{theorem}

\begin{proof}
$ M \in \mathcal P \iff \vect{AM} \perp \vec n \iff
a(x - x_A) + b(y - y_A) + c(z - z_A) = 0$, जिसका \omterm{def:g10:algebra:expand}{विस्तार} होता है \omterm{def:g10:algebra:equation}{समीकरण}.
इसके विपरीत, कोई भी समाधान बिंदु चुनें $ A $ की \omterm{def:g10:algebra:equation}{समीकरण}; वही
पीछे की ओर गणना से पता चलता है कि समाधान सेट है
$\{M : \vect{AM}\cdot\vec n = 0\}$, विमान के माध्यम से $ A $ सामान्य
$\vec n $.
\end{proof}

\begin{method}[व्यवहार में अंतर्विरोध]\label{met:g12:space:intersections}
\begin{itemize}
  \item \emph{रेखा $\cap $ विमान}: पैरामीट्रिक को प्रतिस्थापित करें \omterm{def:g10:coordgeom:system}{COORDINATES} का
        विमान में लाइन \omterm{def:g10:algebra:equation}{समीकरण}; के लिए हल करें $ t $ (एक समाधान: एक
        बिंदु; $0 = $ शून्येतर: समानांतर; $0 = 0$: पंक्ति समाहित)।
  \item \emph{विमान $\cap $ विमान}: दोनों की प्रणाली को हल करें \omterm{def:g10:algebra:equation}{समीकरण},
        एक मुक्त समन्वय द्वारा पैरामीट्रिज़िंग; समाधान एक रेखा है (या
        समतल तब समानांतर होते हैं जब \omterm{def:g12:space:normal}{सामान्य वेक्टर} हैं \omterm{def:g11:vect:collinear}{समरेख}).
  \item \emph{रूढ़िवादिता की जाँच}: रेखा $\perp $ विमान यदि इसकी दिशा
        है \omterm{def:g11:vect:collinear}{समरेख} with the normal; दो तल लंबवत हैं यदि उनके
        normals are \omterm{def:g11:scal:orthogonal}{ओर्थोगोनल}.
\end{itemize}
\end{method}

\begin{proposition}[एक बिंदु से एक समतल तक की दूरी]\label{prop:g12:space:distance}
\index{दूरी!विमान की ओर इंगित करें}
एक में \omterm{def:g10:coordgeom:system}{ऑर्थोनॉर्मल} प्रणाली, से दूरी $ M_0(x_0, y_0, z_0)$ को
विमान $\mathcal P : ax + by + cz + d = 0$ है
\[
\operatorname{dist}(M_0, \mathcal P)
= \frac{\abs{ax_0 + by_0 + cz_0 + d}}{\sqrt{a^2 + b^2 + c^2}}.
\]
\end{proposition}

\begin{proof}
होने देना $ H $ हो \omterm{def:g11:scal:orthogonal}{ओर्थोगोनल} का प्रक्षेपण $ M_0$ पर $\mathcal P $: का पैर
लंबवत, तो $\vect{HM_0}$ है \omterm{def:g11:vect:collinear}{समरेख} यूनिट सामान्य के साथ
$\frac{\vec n}{\norm{\vec n}}$, और दूरी है
$\abs{\vect{HM_0}\cdot \frac{\vec n}{\norm{\vec n}}}$. किसी के लिए
$ H \in \mathcal P $,
$\vect{HM_0}\cdot\vec n = ax_0 + by_0 + cz_0 - (ax_H + by_H + cz_H)
= ax_0 + by_0 + cz_0 + d $ (का उपयोग \omterm{def:g10:algebra:equation}{समीकरण} पर $ H $), सूत्र कहाँ से
से विभाजित करने के बाद $\norm{\vec n} = \sqrt{a^2+b^2+c^2}$.
\end{proof}

\begin{example}\label{ex:g12:space:distance}
उद्गम से तल तक की दूरी $ x + 2y + 2z - 6 = 0$:
$\frac{\abs{-6}}{\sqrt{1 + 4 + 4}} = \frac63 = 2$.
\end{example}

\section{अभ्यास}

सभी अभ्यासों में \omterm{def:g10:coordgeom:system}{निर्देशांक तरीका} है \omterm{def:g10:coordgeom:system}{ऑर्थोनॉर्मल}.

\begin{exercise}[$\star $]\label{exo:g12:space:1}
होने देना $ A(1, 0, 2)$, $ B(3, 1, 1)$ और $ C(2, -1, 3)$. बिंदु हैं $ A $, $ B $,
$ C $ संरेखित? रेखा का पैरामीट्रिक निरूपण दीजिए $(AB)$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g12:space:2}
एक कार्टेशियन निर्धारित करें \omterm{def:g10:algebra:equation}{समीकरण} विमान के माध्यम से $ A(1, 1, 0)$ साथ
\omterm{def:g12:space:normal}{सामान्य वेक्टर} $\vec n(2, -1, 3)$. क्या बात है $ B(0, 2, 1)$ इसके हैं?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g12:space:3}
पर कोण की गणना करें $ A $ (निकटतम डिग्री तक) त्रिभुज के साथ
शिखर $ A(0,0,0)$, $ B(1, 1, 0)$ और $ C(1, 0, 1)$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:space:4}
रेखा पर विचार करें $\mathcal D $ के माध्यम से $ A(1, 2, 0)$ दिशा के साथ
$\vec u(1, -1, 2)$, और विमान $\mathcal P : 2x + y - z + 1 = 0$.
ठानना $\mathcal D \cap \mathcal P $.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:space:5}
होने देना $\mathcal P : x - y + 2z = 1$ और $\mathcal Q : 2x + y + z = 4$.
बताते हैं कि $\mathcal P $ और $\mathcal Q $ एक पंक्ति में काटें और एक दें
इसका पैरामीट्रिक प्रतिनिधित्व।
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:space:6}
दिखाएँ कि पंक्तियाँ
\[
\mathcal D_1 : (x, y, z) = (1 + t,\; 2t,\; 3 - t), \qquad
\mathcal D_2 : (x, y, z) = (2 + s,\; 1 + s,\; 1 + 2s)
\]
तिरछे हैं (नहीं \omterm{def:g12:space:collinear}{समतलीय}).
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:space:7}
होने देना $\mathcal P : 2x - y + 2z - 3 = 0$.
\begin{enumerate}
  \item से दूरी की गणना करें $ M_0(3, 1, 1)$ को $\mathcal P $.
  \item निर्धारित करें \omterm{def:g10:coordgeom:system}{COORDINATES} की \omterm{def:g11:scal:orthogonal}{ओर्थोगोनल} अनुमान $ H $ का
        $ M_0$ पर $\mathcal P $, और दूरी की जाँच करें $ M_0H $.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g12:space:8}
घन $ ABCDEFGH $ पक्ष है $1$; जगह \omterm{def:g10:coordgeom:system}{COORDINATES} ताकि
$ A(0,0,0)$, $ B(1,0,0)$, $ D(0,1,0)$, $ E(0,0,1)$ (साथ
$ C = B + D $, $ F = B + E $, $ G = B + D + E $, $ H = D + E $ जैसा वैक्टर से
$ A $).
\begin{enumerate}
  \item दिखाओ कि विकर्ण $(AG)$ है \omterm{def:g11:scal:orthogonal}{ओर्थोगोनल} विमान के लिए $(BDE)$.
  \item से दूरी की गणना करें $ A $ विमान के लिए $(BDE)$, and the point
        कहाँ $(AG)$ इसे छेदता है.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g12:space:9}
\emph{(चतुष्फलक का आयतन।)} होने देना $ A(1,1,1)$, $ B(2,1,0)$, $ C(0,2,1)$
और $ D(2,2,2)$.
\begin{enumerate}
  \item सत्यापित करें कि $\vect{AB}$, $\vect{AC}$, $\vect{AD}$ नहीं हैं \omterm{def:g12:space:collinear}{समतलीय}
        (बिंदु एक वास्तविक चतुष्फलक बनाते हैं)।
  \item लगता है \omterm{def:g11:vect:vector}{वेक्टर} $\vec n $ \omterm{def:g11:scal:orthogonal}{ओर्थोगोनल} दोनों के लिए $\vect{BC}$ और
        $\vect{BD}$, और एक कार्टेशियन व्युत्पन्न करें \omterm{def:g10:algebra:equation}{समीकरण} विमान का $(BCD)$.
  \item से दूरी की गणना करें $ A $ विमान के लिए $(BCD)$ और का क्षेत्रफल
        त्रिकोण $ BCD $, सूत्र का उपयोग कर
        $\text{Area} = \frac12\sqrt{\norm{\vec u}^2\norm{\vec v}^2 -
        (\vec u \cdot \vec v)^2}$ द्वारा फैलाए गए त्रिभुज के लिए
        $\vec u, \vec v $.
  \item चतुष्फलक का आयतन निकालें
        ($ V = \frac13 \times \text{base area} \times \text{height}$).
\end{enumerate}
\end{exercise}

\section{समस्या: घन को काटना}

\begin{problem}[{सप्ताहांत समस्या--- एक विमान एक घन को काटता है उत्तम षट्कोण, एक नियमित चतुष्फलक कोनों में छिपा होता है, और विकर्ण का कोण प्रत्येक हीरे को आकार देता है}]
\label{pb:g12:space:1}
एक घन को काटें और आप वर्गों और आयतों की \omterm{def:g11:prob:expectation}{अपेक्षा} करते हैं --- फिर भी एक
प्रसिद्ध कट का उत्पादन होता है \emph{नियमित षट्कोण}, और चार चतुर
कोने में कटौती एक को पीछे छोड़ देती है \emph{नियमित चतुष्फलक}. दोनों
आश्चर्य, और $109.5^\circ $ वह कोण जिसकी कार्बन परमाणु पूजा करते हैं,
इस अध्याय के टूलकिट पर जाएँ: पैरामीट्रिक रेखाएँ, समतल \omterm{def:g10:algebra:equation}{समीकरण}
और यह \omterm{def:g12:space:dot}{अदिश उत्पाद} (\cref{thm:g12:space:planeeq},
\cref{prop:g12:space:distance}). के पूरे घन में उपयोग करें
\cref{exo:g12:space:8}: ओर $1$, $ A(0,0,0)$, $ B(1,0,0)$,
$ D(0,1,0)$, $ E(0,0,1)$, $ C(1,1,0)$, $ F(1,0,1)$, $ H(0,1,1)$,
$ G(1,1,1)$.

\medskip
\noindent\textbf{Part I --- Fluency.}
\begin{enumerate}
  \item लिखना \omterm{def:g10:algebra:equation}{समीकरण} विमान के माध्यम से $ B(1,0,0)$ साथ
        \omterm{def:g12:space:normal}{सामान्य वेक्टर} $(1,1,1)$, का एक पैरामीट्रिक प्रतिनिधित्व
        के माध्यम से लाइन $ A $ निर्देशक $(1,1,1)$, और उनके
        \omterm{def:g10:numbers:interunion}{चौराहा} बिंदु।
  \item से दूरी की गणना करें $ A $ उस विमान को
        (\cref{prop:g12:space:distance}).
  \item हैं वैक्टर $(1,-1,0)$ और $(1,1,-2)$ \omterm{def:g11:scal:orthogonal}{ओर्थोगोनल}?
  \item विमानों की स्थिति $ x + y + z = 1$ और
        $2x + 2y + 2z = 5$? लाइन के माध्यम से $ A $ निर्देशक
        $(1,1,0)$ पहले विमान के सापेक्ष?
  \item लिस्ट \omterm{prop:g10:coordgeom:midpoint}{मध्य} बिन्दुओं छह किनारों में से
        $[BC], [CD], [DH], [HE], [EF], [FB]$ घन का, और
        सत्यापित करें कि सभी छह संतुष्ट हैं \omterm{def:g10:algebra:equation}{समीकरण}
        $ x + y + z = \frac32$.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The hexagonal section.}
होने देना $ P $ विमान हो $ x + y + z = \frac32$.
\begin{enumerate}[resume]
  \item बताते हैं कि $ P $ बड़े विकर्ण पर लंबवत है
        $(AG)$ और घन के \omterm{pb:g11:quad:1}{केंद्र} से होकर गुजरता है।
  \item लगातार के बीच की दूरी की गणना करें \omterm{prop:g10:coordgeom:midpoint}{मध्य} बिन्दुओं का
        अनुभाग के चारों ओर प्रश्न 5: आप क्या पाते हैं?
  \item एक पर अनुभाग के बहुभुज के कोण की गणना करें
        शीर्ष (दो लगातार) धा\omterm{def:g11:vect:vector}{र-वेक्टर} और एक बिंदु
        उत्पाद). निष्कर्ष: अनुभाग एक है \emph{नियमित
        षट्भुज}.
  \item षट्भुज की परिधि और क्षेत्रफल की गणना करें और तुलना करें
        घन के मुख वाला क्षेत्र। (के लिए
        रिकार्ड: इकाई घन का सबसे बड़ा \omterm{def:g12:space:collinear}{समतलीय} खंड है
        द $1 \times \sqrt2$ क्षेत्रफल का विकर्ण आयत
        $\sqrt2$ --- नियमित षट्कोण चांदी लेता है।)
  \item विकर्ण कहाँ है $(AG)$ प्रवेश करना $ P $? उसे सत्यापित करें
        भेदन बिंदु है \omterm{prop:g10:coordgeom:midpoint}{मध्य} का $[AG]$.
  \item कट को स्लाइड करें: अनुभागों का वर्णन करें
        $ x + y + z = c $ जैसा $ c $ से बढ़ता है $0$ को $3$ ---
        के लिए अनुभाग की गणना करें $ c = \frac12$ (कौन सा बहुभुज,
        कौन सा आकार?), और जो मॉर्फिंग गुजरती है उसका वर्णन करें
        आपके षट्भुज के माध्यम से $ c = \frac32$.
  \item बहुभुज में कोई समतल एक घन को क्यों नहीं काट सकता?
        \emph{सात} पक्ष? (प्रत्येक पक्ष को कहाँ होना चाहिए
        अनुभाग झूठ?)
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The hidden tetrahedron.}
\begin{enumerate}[resume]
  \item कोने के चतुष्फलक के आयतन की गणना करें $ ABDE $
        (आधार, ऊंचाई और एक तिहाई सूत्र \omterm{prop:g10:coordgeom:midpoint}{मध्य}
        स्कूल की मात्रा स्वीकृत --- \cref{exo:g12:space:9}
        सामान्य तौर पर ऐसे वॉल्यूम की गणना करता है)।
  \item चार शिखर $ B $, $ D $, $ E $, $ G $: सभी छह की गणना करें
        \omterm{def:g10:vectors:sum}{जोड़}ीवार दूरियाँ और निष्कर्ष निकालो $ BDEG $ एक है
        \emph{नियमित} चतुष्फलक. फिर इसका आयतन प्राप्त करें
        घन से कोने के टेट्राहेड्रा को घटाना।
  \item घन के \omterm{pb:g11:quad:1}{केंद्र} से दूरी की गणना करें
        विमान $(BDE)$.
  \item हीरा कोण: के बीच के कोण की गणना करें
        विकर्ण $\vect{AG}$ और $\vect{BH}$ घन का. यह
        \omterm{def:g10:proba:operations}{पूरक}, के बारे में $109.5^\circ $, के बीच का कोण है
        हीरे और मीथेन में बंधन --- चार क्यों करते हैं
        कार्बन परमाणु के चारों ओर इलेक्ट्रॉन \omterm{def:g10:vectors:sum}{जोड़}े कोनों का चयन करते हैं
        प्रश्न 14 के चतुष्फलक का?
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Genuinely three-dimensional.}
\begin{enumerate}[resume]
  \item पता लगाएं कि रेखा कहां से होकर गुजरती है $ B(1,0,0)$ निर्देशक
        $(0,1,1)$ षट्कोण के तल को छेदता है $ P $.
  \item दिखाएँ कि पंक्तियाँ $(AB)$ और $(EG)$ न मिलते हैं न
        समानांतर हैं: \emph{तिरछा} पंक्तियाँ, वह घटना
        हवाई जहाज़ में ऐसा नहीं हो सकता.
  \item की गणना करें \omterm{def:g11:scal:orthogonal}{ओर्थोगोनल} का प्रक्षेपण $ B $ पर $ P $
        (से चलना $ B $ विमान तक सामान्य के साथ), और
        जांचें कि आपकी बात संतुष्ट है $ P$'एस \omterm{def:g10:algebra:equation}{समीकरण}.
  \item समापन --- अंतरिक्ष सर्वेक्षक की किट: पैरामीट्रिक रेखाएँ
        छेदने के लिए, \omterm{def:g12:space:normal}{सामान्य वेक्टर} कोणों और दूरियों के लिए,
        विमान \omterm{def:g10:algebra:equation}{समीकरण} अनुभागों के लिए; और घन के रूप में
        खेल का मैदान जहां उन्होंने एक नियमित षट्भुज का निर्माण किया
        चतुष्फलक, हीरा कोण और तिरछा \omterm{def:g10:vectors:sum}{जोड़}ा
        पंक्तियाँ. एक-एक वाक्य.
\end{enumerate}
\end{problem}
