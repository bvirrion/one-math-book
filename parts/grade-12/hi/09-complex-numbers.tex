\chapter{जटिल संख्याएँ}\label{ch:g12:complex}

 \omterm{def:g10:algebra:equation}{समीकरण} $x^2 = -1$ कोई वास्तविक समाधान नहीं है. एनलार्जिंग $\R $ एक के साथ
नया नंबर $\iu $ जिसका वर्ग है $-1$ क्षेत्र का उत्पादन करता है $\C $ का जटिल
नंबर, जिसमें \emph{प्रत्येक} बहुपद \omterm{def:g10:algebra:equation}{समीकरण} समाधान हैं --- और
जिसका \omterm{def:g12:seq:arith-geom}{अंकगणित} विमान की ज्यामिति को कूटबद्ध करता है: अनुवाद, घूर्णन
और स्केलिंग \omterm{def:g10:vectors:sum}{जोड़} और गुणा बन जाती है।

\section{बीजगणितीय रूप}

\begin{definition}[जटिल संख्याएँ]\label{def:g12:complex:def}
के समुच्चय \emph{सम्मिश्र संख्याएँ}\index{सम्मिश्र संख्याएँ} है
\[
\C = \{a + \iu b : a, b \in \R\},
\]
कहाँ $\iu $ एकल नियम के अधीन एक प्रतीक है $\iu^2 = -1$; इसके अलावा और
गुणन वास्तविक बहुपदों की तरह ही किया जाता है $\iu $, कम करना
$\iu^2$ को $-1$. असली $ a $ और $ b $ हैं \emph{असली हिस्सा}\index{असली हिस्सा}
$\Rea(z)$ और \emph{काल्पनिक भाग}\index{काल्पनिक भाग} $\Ima(z)$ का $ z = a + \iu b $; यह
अभिव्यक्ति है \emph{बीजगणितीय रूप}\index{बीजगणितीय रूप} का $ z $, और यह अद्वितीय है:
$ a + \iu b = a' + \iu b'$ अगर और केवल अगर $ a = a'$ और $ b = b'$.
\end{definition}

\begin{example}
$(2 + 3\iu)(1 - \iu) = 2 - 2\iu + 3\iu - 3\iu^2 = 5 + \iu $.
\end{example}

\begin{definition}[संयुग्मित]\label{def:g12:complex:conjugate}
 \emph{संयुग्मित}\index{संयुग्मित} का $ z = a + \iu b $ है
$\conj{z} = a - \iu b $.
\end{definition}

\begin{proposition}[संयुग्मन के गुण]\label{prop:g12:complex:conjugate}
सभी के लिए $ z, w \in \C $:
\[
\conj{z + w} = \conj z + \conj w, \quad
\conj{zw} = \conj z\,\conj w, \quad
\conj{\conj z} = z, \quad
z + \conj z = 2\Rea(z), \quad
z - \conj z = 2\iu\,\Ima(z),
\]
और $ z\conj z = a^2 + b^2 \geq 0$ के लिए $ z = a + \iu b $. इसके अतिरिक्त $ z \in \R $
अगर और केवल अगर $ z = \conj z $.
\end{proposition}

\begin{proof}
सभी की सीधी गणनाएँ हैं \omterm{def:g12:complex:def}{बीजगणितीय रूप}; उदाहरण के लिए
$ z\conj z = (a + \iu b)(a - \iu b) = a^2 - (\iu b)^2 = a^2 + b^2$.
\end{proof}

\begin{method}[सम्मिश्र संख्याओं का विभाजन]\label{met:g12:complex:division}
में भागफल लिखना \omterm{def:g12:complex:def}{बीजगणितीय रूप}, अंश और हर को इससे गुणा करें
द \omterm{def:g12:complex:conjugate}{संयुग्मित} हर का:
\[
\frac{1}{2 + \iu} = \frac{2 - \iu}{(2+\iu)(2-\iu)} = \frac{2 - \iu}{5}
= \frac25 - \frac15\iu .
\]
प्रत्येक अशून्य सम्मिश्र संख्या इसलिए इसका उलटा है: $\C $ एक है
\emph{मैदान}.
\end{method}

\section{जटिल तल, मापांक और तर्क}

\begin{definition}[प्रत्यय, मापांक]\label{def:g12:complex:modulus}
के साथ विमान में \omterm{def:g10:coordgeom:system}{ऑर्थोनॉर्मल} \omterm{def:g10:coordgeom:system}{COORDINATES}, बात $ M(a, b)$ है
\emph{\omterm{def:g10:functions:function}{छवि}} का $ z = a + \iu b $, और $ z $ है \emph{प्रत्यय}\index{प्रत्यय} का $ M $.
\emph{मापांक}\index{मापांक} का $ z $ की दूरी है $ M $ मूल की ओर:
\[
\abs{z} = \sqrt{a^2 + b^2} = \sqrt{z\conj z}.
\]
\end{definition}

\begin{proposition}[मापांक के गुण]\label{prop:g12:complex:modulus}
के लिए $ z, w \in \C $:
\[
\abs{zw} = \abs z\,\abs w, \qquad
\abs{\tfrac zw} = \tfrac{\abs z}{\abs w}\ (w \neq 0), \qquad
\abs{\conj z} = \abs z, \qquad
\abs{z + w} \leq \abs{z} + \abs{w} \ \text{(triangle inequality)}.
\]
इसके अतिरिक्त $\abs{z_B - z_A}$ के बिंदुओं के बीच की दूरी है affixes
$ z_A $ और $ z_B $.
\end{proposition}

\begin{proof}
$\abs{zw}^2 = zw\,\conj{zw} = z\conj z\, w \conj w = \abs z^2 \abs w^2$
पहला देता है; दूसरा और तीसरा समान हैं. त्रिकोण के लिए
असमानता:
\[
\abs{z+w}^2 = (z+w)(\conj z + \conj w)
= \abs z^2 + \abs w^2 + 2\Rea(z \conj w)
\leq \abs z^2 + \abs w^2 + 2\abs{z\conj w}
= \bigl(\abs z + \abs w\bigr)^2,
\]
का उपयोग करते हुए $\Rea(u) \leq \abs u $. दूरी कथन पायथागॉरियन है
के लिए फार्मूला लागू किया गया \omterm{def:g10:coordgeom:system}{COORDINATES} का $ z_B - z_A $.
\end{proof}

\begin{definition}[तर्क, त्रिकोणमितीय और घातांकीय रूप]\label{def:g12:complex:argument}
होने देना $ z \neq 0$. एक \emph{तर्क}\index{तर्क} का $ z $, लिखा हुआ
$\Arg(z)$, कोई माप है $\theta $ सकारात्मक वास्तविक अक्ष से कोण का
किरण को $ OM $; तक परिभाषित किया गया है $2k\pi $. लिखना $ r = \abs z $,
\[
z = r(\cos\theta + \iu\sin\theta)
\qquad\text{(trigonometric form)}.
\]
संकेतन का परिचय\index{घातीय रूप}
\[
\eu^{\iu\theta} = \cos\theta + \iu\sin\theta,
\]
यह हो जाता है \emph{घातीय रूप} $ z = r\,\eu^{\iu\theta}$.
\end{definition}

\begin{theorem}\label{thm:g12:complex:expform}
सभी के लिए $\theta, \theta' \in \R $:
\[
\eu^{\iu\theta}\,\eu^{\iu\theta'} = \eu^{\iu(\theta + \theta')},
\qquad
\conj{\eu^{\iu\theta}} = \eu^{-\iu\theta},
\qquad
\abs{\eu^{\iu\theta}} = 1 .
\]
नतीजतन, के लिए $ z = r\eu^{\iu\theta}$ और $ z' = r'\eu^{\iu\theta'}$
शून्येतर:
\[
zz' = rr'\,\eu^{\iu(\theta+\theta')},
\qquad
\frac{z}{z'} = \frac{r}{r'}\,\eu^{\iu(\theta - \theta')} :
\]
\emph{मॉड्यूलि गुणा, बहस जोड़ना।}
\end{theorem}

\begin{proof}
पहली पहचान \emph{है} \omterm{def:g10:vectors:sum}{जोड़} सूत्रों की \omterm{def:g10:vectors:sum}{जोड़}ी
(\cref{prop:g12:trigo:addition}):
\[
(\cos\theta + \iu\sin\theta)(\cos\theta' + \iu\sin\theta')
= (\cos\theta\cos\theta' - \sin\theta\sin\theta')
+ \iu(\sin\theta\cos\theta' + \cos\theta\sin\theta').
\]
शेष प्रत्यक्ष गणना द्वारा अनुसरण करता है।
\end{proof}

\begin{corollary}[डी मोइवरे का सूत्र]\label{cor:g12:complex:demoivre}
\index{डी मोइवरे का सूत्र}
के लिए $\theta \in \R $ और $ n \in \Z $:
$\bigl(\cos\theta + \iu\sin\theta\bigr)^n = \cos n\theta + \iu \sin n\theta $.
\end{corollary}

\begin{proof}
प्रेरण चालू $ n \geq 0$ का उपयोग करते हुए \cref{thm:g12:complex:expform}; तक \omterm{def:g10:algebra:expand}{विस्तार}
$ n < 0$ व्युत्क्रम लेने से.
\end{proof}

\begin{method}[प्रपत्रों के बीच स्विच करना]\label{met:g12:complex:forms}
बीजीय से घातांक तक: गणना करें $ r = \abs z $, फिर खोजें $\theta $ साथ
$\cos\theta = \frac ar $, $\sin\theta = \frac br $ (सही चतुर्थांश का पता लगाएं
व्युत्क्रम लागू करने से पहले \omterm{def:g11:trigo:cossin}{कोज्या}). घातांक से बीजीय तक: \omterm{def:g10:algebra:expand}{विस्तार} करें
$ r\cos\theta + \iu\, r\sin\theta $. उपयोग \omterm{def:g12:complex:def}{बीजगणितीय रूप} रकम के लिए,
\omterm{def:g12:complex:argument}{घातीय रूप} उत्पादों, शक्तियों और भागफल के लिए।
\end{method}

\begin{example}
$ z = 1 + \iu $: $\abs z = \sqrt2$ और $\cos\theta = \sin\theta =
\frac{1}{\sqrt2}$, इसलिए $\theta = \frac\pi4$ और $ z = \sqrt2\,\eu^{\iu\pi/4}$.
इस तरह $ z^{20} = 2^{10}\,\eu^{5\iu\pi} = -1024$.
\end{example}

\section{द्वितीय-डिग्री समीकरण और एकता की जड़ें}

\begin{theorem}[वास्तविक गुणांकों के साथ द्विघात समीकरण]\label{thm:g12:complex:quadratic}
होने देना $ a, b, c \in \R $, $ a \neq 0$, और $\Delta = b^2 - 4ac $. \omterm{def:g10:algebra:equation}{समीकरण}
$ az^2 + bz + c = 0$ में समाधान हैं $\C $:
\[
\begin{aligned}
&\Delta > 0:\ z = \frac{-b \pm \sqrt{\Delta}}{2a} \ (\text{two real}); \qquad
\Delta = 0:\ z = \frac{-b}{2a} \ (\text{double}); \\
&\Delta < 0:\ z = \frac{-b \pm \iu\sqrt{-\Delta}}{2a} \ (\text{two conjugate}).
\end{aligned}
\]
\end{theorem}

\begin{proof}
वर्ग पूरा करें:
$ az^2 + bz + c = a\left(\left(z + \frac{b}{2a}\right)^2 -
\frac{\Delta}{4a^2}\right)$. अगर $\Delta < 0$, तब
$\frac{\Delta}{4a^2} = \left(\iu\frac{\sqrt{-\Delta}}{2a}\right)^2$, और
वर्ग कारकों का अंतर हमेशा की तरह।
\end{proof}

\begin{definition}[एकता की जड़ें]\label{def:g12:complex:rootsofunity}
के लिए $ n \geq 1$, द \emph{$ n $- एकता की जड़ें}\index{$ n $- एकता की जड़ें} हैं
के समाधान $ z^n = 1$.
\end{definition}

\begin{theorem}\label{thm:g12:complex:rootsofunity}
 \omterm{def:g10:algebra:equation}{समीकरण} $ z^n = 1$ बिलकुल है $ n $ समाधान:
\[
\omega_k = \eu^{2\iu k\pi/n}, \qquad k = 0, 1, \dots, n-1 .
\]
उनका इमेजिस एक नियमित के शीर्ष हैं $ n $-गोन इकाई में अंकित है
वृत्त, और के लिए $ n \geq 2$ उनका योग है $0$.
\end{theorem}

\begin{proof}
प्रत्येक $\omega_k $ संतुष्ट $\omega_k^n = \eu^{2\iu k\pi} = 1$, और वे हैं
उनके बाद से \omterm{def:g10:vectors:sum}{जोड़}ीवार अलग बहस रिहायश $\intco{0}{2\pi}$.
इसके विपरीत, यदि $ z^n = 1$ तब $\abs z^n = 1$, इसलिए $\abs z = 1$ (सकारात्मक
असली) और $ z = \eu^{\iu\theta}$ साथ $ n\theta \equiv 0 \pmod{2\pi}$,
\emph{यानी} $\theta = \frac{2k\pi}{n}$; कमी $ k $ सापेक्ष $ n $ बीच में भूमि
द $\omega_k $. कोने कोण द्वारा समान दूरी पर हैं $\frac{2\pi}{n}$:
एक नियमित $ n $-गोन. अंत में, साथ $\omega = \omega_1$,
$\omega \neq 1$ और ज्यामितीय योग देता है
\[
\sum_{k=0}^{n-1}\omega_k = \sum_{k=0}^{n-1}\omega^k
= \frac{\omega^n - 1}{\omega - 1} = 0 . \qedhere
\]
\end{proof}

$1 + j + j^2 = 0$.
\end{example}

\section{सम्मिश्र संख्याएँ और समतल ज्यामिति}

\begin{proposition}[ज्यामितीय शब्दकोश]\label{prop:g12:complex:geometry}
होने देना $ A, B, C, D $ के अंक हो affixes $ a, b, c, d $ ($ a \neq b $,
$ c \neq d $).
\begin{enumerate}
  \item द्वारा अनुवाद \omterm{def:g11:vect:vector}{वेक्टर} का \omterm{def:g12:complex:modulus}{प्रत्यय} $ t $ है $ z \mapsto z + t $.
  \item केन्द्र का घूर्णन $\omega $ (\omterm{def:g12:complex:modulus}{प्रत्यय}) और कोण $\theta $ है
        $ z \mapsto \omega + \eu^{\iu\theta}(z - \omega)$.
  \item \omterm{pb:g11:quad:1}{केंद्र} की स्केलिंग (समरूपता)। $\omega $ और अनुपात
        $ k \in \R^*$ है $ z \mapsto \omega + k(z - \omega)$.
  \item $\dfrac{d - c}{b - a}$ है \omterm{def:g12:complex:modulus}{मापांक} $\dfrac{CD}{AB}$ और \omterm{def:g12:complex:argument}{तर्क} द
        के बीच का कोण वैक्टर $\vect{AB}$ और $\vect{CD}$. विशेष रूप से, $ AB \perp CD $ अगर और केवल अगर $\frac{d-c}{b-a}$ विशुद्ध रूप से है
        काल्पनिक, और $ A, B, C $ हैं \omterm{def:g11:vect:collinear}{समरेख} ($ C \ne A $, $ C\ne B $) अगर और केवल अगर $\frac{c-a}{b-a} \in \R $.
\end{enumerate}
\end{proposition}

\begin{proof}
(1) और (3) निर्देशांक सूत्रों को पुनः स्थापित करें। (2) के लिए, मानचित्र
$ w \mapsto \eu^{\iu\theta} w $ मॉड्यूलि को गुणा करता है $1$ और जोड़ता है $\theta $ को
बहस: यह कोण का घूर्णन है $\theta $ उत्पत्ति के बारे में; संयुग्मन
अनुवाद भेजकर $\omega $ को $0$ सामान्य \omterm{pb:g11:quad:1}{केंद्र} देता है. के लिए
(4), मॉड्यूलि और बहस भागफल का घटाना
(\cref{thm:g12:complex:expform}).
\end{proof}

\begin{exercise}[$\star $]\label{exo:g12:complex:2}
में हल करें $\C $: (ए) $ z^2 - 4z + 13 = 0$; (बी) $ z^2 + z + 1 = 0$. प्रत्येक में जाँच करें
मामला यह है कि दो समाधान हैं \omterm{def:g12:complex:conjugate}{संयुग्मित} करता है और उनके उत्पाद की गणना करें।
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g12:complex:3}
अंदर डालो \omterm{def:g12:complex:argument}{घातीय रूप}: $ z_1 = -1 + \iu $, $ z_2 = \sqrt3 - \iu $, और गणना करें
$\dfrac{z_1}{z_2}$ फिर घातीय में \omterm{def:g12:complex:def}{बीजगणितीय रूप}. सटीक निष्कर्ष निकालें
का मूल्य $\cos\dfrac{11\pi}{12}$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:complex:4}
बिन्दुओं के समुच्चय का ज्यामितीय वर्णन कीजिए $ M $ का \omterm{def:g12:complex:modulus}{प्रत्यय} $ z $ ऐसा कि:
(ए) $\abs{z - 2} = \abs{z + \iu}$;\quad
(बी) $\abs{z - 1 - \iu} = 3$;\quad
(सी) $\dfrac{z - 1}{z + 1}$ पूर्णतः काल्पनिक है.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:complex:5}
डी मोइवर के सूत्र और द्विपद प्रमेय का उपयोग करके व्यक्त करें $\cos 3\theta $
में एक बहुपद के रूप में $\cos\theta $, और $\sin 3\theta $ के अनुसार
$\sin\theta $.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:complex:6}
\emph{(रैखिकीकरण।)} का उपयोग करते हुए
$\cos\theta = \frac{\eu^{\iu\theta} + \eu^{-\iu\theta}}{2}$, रैखिक बनाना
$\cos^3\theta $ (इसे इसके संयोजन के रूप में लिखें $\cos k\theta $), और निष्कर्ष निकालें
$\displaystyle\int_0^{\pi/2}\cos^3\theta\,\dd\theta $.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:complex:7}
होने देना $ A $ और $ B $ पास होना affixes $ a = 1 + \iu $ और $ b = 3 + 2\iu $. निर्धारित करें
दो अंक $ C $ त्रिकोण बनाना $ ABC $ समबाहु, जैसे इमेजिस का $ B $
\omterm{pb:g11:quad:1}{केंद्र} के घूर्णन के अंतर्गत $ A $ और कोण $\pm\frac{\pi}{3}$.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g12:complex:8}
हल करना $ z^4 = -4$ में $\C $ और समाधान प्लॉट करें. कारक
$ z^4 + 4$ वास्तविक गुणांक वाले दो द्विघात बहुपदों में।
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g12:complex:9}
के लिए $\theta \in \intoo{0}{2\pi}$ और $ n \in \N $, गणना करें
\[
S = 1 + \cos\theta + \cos 2\theta + \dots + \cos n\theta
\]
ज्यामितीय प्रगति का योग करके
$\sum_{k=0}^n \eu^{\iu k\theta}$ और ले रहा हूँ वास्तविक भाग. (उत्तर:
$ S = \dfrac{\sin\frac{(n+1)\theta}{2}}{\sin\frac{\theta}{2}}
\cos\frac{n\theta}{2}$.)
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g12:complex:10}
होने देना $\omega = \eu^{2\iu\pi/5}$.
\begin{enumerate}
  \item उसे उचित ठहराओ $1 + \omega + \omega^2 + \omega^3 + \omega^4 = 0$.
  \item होने देना $ u = \omega + \omega^4$ और $ v = \omega^2 + \omega^3$. बताते हैं कि
        $ u + v = -1$ और $ uv = -1$, और उसका निष्कर्ष निकालो $ u = \frac{-1+\sqrt5}{2}$.
  \item समाप्त करें कि $\cos\dfrac{2\pi}{5} = \dfrac{\sqrt5 - 1}{4}$.
\end{enumerate}
\end{exercise}

\section{समस्या: एकता के दर्पण में पंचकोण}

\begin{problem}[{सप्ताहांत समस्या---एकता की पांचवी जड़ें गणना करें $\cos 72^\circ $ बिल्कुल, स्वर्णिम अनुपात का संकेत है परिणाम, और गुणन घूर्णन बन जाता है}]
\label{pb:g12:complex:1}
यूक्लिड नियमित पंचकोण का निर्माण कर सकता था लेकिन कारण
\emph{क्यों} यह शासक और कम्पास को उपज देता है --- जबकि विनम्र
$20^\circ $ कोण नहीं --- दो हजार तक छिपा रहा
साल. यह इस अध्याय में छिपा है: पाँच पाँचवाँ एकता की \omterm{def:g11:quad:discriminant}{जड़}ें
(\cref{thm:g12:complex:rootsofunity}) शून्य का योग, और वह एक
बीजगणित की रेखा सिकुड़ती है $\cos 72^\circ $ एक द्विघात से बाहर
\omterm{def:g10:algebra:equation}{समीकरण} --- केवल वर्गमूल, इसलिए रचनात्मक, और हस्ताक्षरित,
बेशक, सुनहरे अनुपात से। इस रत्न के चारों ओर: प्रवाह, दे
मोइवरे का त्रिकोणमिति का शॉर्टकट, और गुणन के रूप में उजागर हुआ
ज्यामिति.

\medskip
\noindent\textbf{Part I --- Fluency.}
\begin{enumerate}
  \item गणना करना $(2 + \iu)(3 - \iu)$, \
        $\dfrac{1 + \iu}{1 - \iu}$
        (\cref{met:g12:complex:division}), और
        $\abs{3 + 4\iu}$.
  \item अंदर डालो \omterm{def:g12:complex:argument}{घातीय रूप}: $1 + \iu $; \ $-2$; \
        $1 - \iu\sqrt3$.
  \item गणना करना $(1 + \iu)^8$ का उपयोग \omterm{def:g12:complex:argument}{घातीय रूप}.
  \item हल करना $ z^2 = \iu $, तब $ z^2 - 2z + 5 = 0$
        (\cref{thm:g12:complex:quadratic}).
  \item ज्यामितीय शब्दकोश
        (\cref{prop:g12:complex:geometry}): मानचित्र का वर्णन करें
        $ z \mapsto \iu z $, और सेट
        $\abs{z - (1 + \iu)} = 2$.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Five \omterm{def:g11:quad:discriminant}{जड़}ों, one pentagon.}
होने देना $\omega = \eu^{2\iu\pi/5}$ और विचार करें
$1, \omega, \omega^2, \omega^3, \omega^4$.
\begin{enumerate}[resume]
  \item जांचें कि ये वास्तव में समाधान हैं $ z^5 = 1$,
        और उनके द्वारा समतल में खींची गई आकृति का वर्णन करें।
  \item \omterm{def:g11:vect:det}{सिद्ध} कीजिए कि उनका योग शून्य है। (ज्यामितीय प्रगति
        --- या योग को इससे गुणा करें $1 - \omega $.)
  \item लेना वास्तविक भाग और निष्कर्ष निकालें
        \[
        1 + 2\cos\frac{2\pi}{5} + 2\cos\frac{4\pi}{5} = 0 .
        \]
  \item तय करना $ c = \cos\frac{2\pi}{5}$ और डबल-एंगल का उपयोग करें
        के लिए सूत्र $\cos\frac{4\pi}{5}$ प्रश्न 8 को परिवर्तित करने के लिए
        एक द्विघात में \omterm{def:g10:algebra:equation}{समीकरण} के लिए $ c $; इसे हल करें और
        निष्कर्ष
        \[
        \cos 72^\circ = \frac{\sqrt5 - 1}{4} .
        \]
  \item संख्यात्मक रूप से सत्यापित करें, फिर सुनहरे अनुपात पर हस्ताक्षर करें
        काम: दिखाओ $ c = \frac{1}{2\varphi}$ कहाँ
        $\varphi = \frac{1 + \sqrt5}{2}$
        (\cref{pb:g10:algebra:1}) --- और प्रसिद्ध बताएं
        ज्यामितीय प्रतिध्वनि: एक नियमित पंचकोण में, विकर्ण होता है
        $\varphi $ कई बार पक्ष.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- De Moivre's shortcuts.}
\begin{enumerate}[resume]
  \item निर्माण का फैसला: $\cos 72^\circ $ केवल जरूरत है
        वर्गमूल (प्रश्न 9), अतः पंचकोण है
        शासक-और-दिशासूचक यंत्र निर्माण योग्य; $\cos 20^\circ $
        एक अटूट घन को संतुष्ट करता है
        (\cref{pb:g12:trigo:1}), इतना $60^\circ $ कोण नहीं कर सकता
        त्रिविभाजित होना. उभरता हुआ मानदंड बताएं (वर्ग)।
        \omterm{def:g11:quad:discriminant}{जड़}ें निर्माण योग्य, घन \omterm{def:g11:quad:discriminant}{जड़}ें नहीं), जिसका पूरा \omterm{def:g11:vect:det}{सिद्ध}ांत ---
        गॉस, उम्र 19, और 17-गॉन --- में रहता है
        विश्वविद्यालय की मात्रा.
  \item हल करना $ z^4 = -4$ द्वारा \omterm{def:g12:complex:argument}{घातीय रूप} और ठीक हो जाओ
        \cref{exo:g12:complex:8}का गुणनखंडन
        $ z^4 + 4 = (z^2 - 2z + 2)(z^2 + 2z + 2)$ \omterm{def:g10:vectors:sum}{जोड़}ी बनाकर
        \omterm{def:g12:complex:conjugate}{संयुग्मित} \omterm{def:g11:quad:discriminant}{जड़}ों.
  \item बढ़ाना $(\cos\theta + \iu\sin\theta)^3$ के साथ
        द्विपद प्रमेय और डी मोइवर, और पुनः व्युत्पन्न
        $\cos 3\theta = 4\cos^3\theta - 3\cos\theta $ तीन में
        पंक्तियाँ --- के जोड़-सूत्र मार्ग से तुलना करें
        \cref{pb:g12:trigo:1}.
  \item के साथ क्रॉस-चेक करें \cref{exo:g12:complex:9}: के लिए
        $\theta = \frac{2\pi}{5}$ और $ n = 4$, क्या करता है
        योग $1 + \cos\theta + \dots + \cos 4\theta $ बराबर, और
        यह प्रश्न 8 के अनुरूप क्यों है?
  \item के तीन घनमूलों की गणना करें $8\iu $ में बीजीय
        प्रपत्र.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Multiplication is geometry.}
\begin{enumerate}[resume]
  \item मानचित्र का पूर्ण वर्णन करें $ z \mapsto (1 + \iu)z $:
        यह किस कोण से घूमता है, किस कारक से घूमता है
        स्केल, और यह इकाई वर्ग पर क्या करता है?
  \item \omterm{def:g11:vect:det}{सिद्ध} करना $\abs{z_1 z_2} = \abs{z_1}\,\abs{z_2}$ साथ
        \omterm{def:g12:complex:conjugate}{संयुग्मित} करता है (\cref{prop:g12:complex:modulus}), फिर
        गणना करें $(1+\iu)^n $ के लिए $ n = 1, 2, 3, 4$ और वर्णन करें
        सर्पिल शक्तियों का पता लगाता है।
  \item जो नंबर $\mathrm{j} = \eu^{2\iu\pi/3}$: दिखाओ
        $1 + \mathrm{j} + \mathrm{j}^2 = 0$, और त्रिभुज पर सत्यापित करें
        $\left(1, \mathrm{j}, \mathrm{j}^2\right)$ शास्त्रीय मानदंड:
        $ a + \mathrm{j} b + \mathrm{j}^2 c = 0$ (एक ओरिएंटेशन) के लिए धारण करता है
        का) एक समबाहु त्रिभुज $ abc $.
  \item बीजगणित का मौलिक प्रमेय (स्वीकृत:
        डी'अलेम्बर्ट--गॉस): प्रत्येक बहुपद गुणनखंड पूरी तरह से
        खत्म $\C $. वास्तविक दुनिया का परिणाम निकालें --- प्रत्येक
        वास्तविक बहुपद गुणनखंडों को वास्तविक रैखिक और द्विघात में
        टुकड़े --- और उदाहरण के तौर पर प्रश्न 12 की ओर संकेत करें।
  \item समापन --- $\C$ का चित्र, प्रत्येक वाक्य: संख्याएँ
        बिंदु बनें; गुणा हो जाता है
        रोटेशन-और-स्केलिंग; एकता की \omterm{def:g11:quad:discriminant}{जड़}ें नियमित हो जाओ
        बहुभुज; डी मोइवर घातों को त्रिकोणमितीय में बदल देता है
        पहचान; और बीजगणितीय समापन इसकी गारंटी देता है कि नहीं
        \omterm{def:g10:algebra:equation}{समीकरण} कभी भी एक बड़ी दुनिया की जरूरत है. कोडा: कौन से दो
        प्रश्न 10 में मनाए गए नंबर मिले?
\end{enumerate}
\end{problem}
