\chapter{विभेदन और उत्तलता}\label{ch:g12:deriv}

 यौगिक के परिवर्तन की तात्कालिक दर को मापता है \omterm{def:g11:func:function}{समारोह}; यह है
द \omterm{def:g10:reffunc:affine}{ढलान} की \omterm{def:g12:deriv:tangent}{स्पर्शरेखा} इसके लिए लाइन \omterm{def:g10:functions:graph}{ग्राफ}. यह अध्याय समीक्षा और \omterm{def:g10:algebra:expand}{विस्तार} करता है
विभेदन नियम, लिंक का चिह्न $f'$ की विविधताओं के लिए $ f $,
और परिचय देता है उत्तलता, जो दूसरे द्वारा शासित होता है यौगिक.

\section{व्युत्पन्न}

\begin{definition}[एक बिंदु पर व्युत्पन्न]\label{def:g12:deriv:derivative}
होने देना $ f $ एक पर परिभाषित किया जाए \omterm{def:g10:numbers:interval}{अंतराल} $ I $ और $ a \in I $. \omterm{def:g11:func:function}{समारोह} $ f $ है
\emph{पर भिन्न $ a $}\index{पर भिन्न $ a $}\index{विभेदक} यदि अंतर भागफल
\[
\frac{f(a+h) - f(a)}{h}
\]
की एक सीमित सीमा होती है $ h \to 0$. इस सीमा को कहा जाता है \emph{का व्युत्पन्न
$ f $ पर $ a $} और लिखा $ f'(a)$. अगर $ f $ के हर बिंदु पर भिन्न है
$ I $, द \omterm{def:g11:func:function}{समारोह} $ f' \colon x \mapsto f'(x)$ है \emph{यौगिक} का
$ f $.
\end{definition}

\begin{definition}[स्पर्श रेखा]\label{def:g12:deriv:tangent}
अगर $ f $ है \omterm{def:g12:deriv:derivative}{विभेदक} पर $ a $, द \emph{स्पर्शरेखा}\index{स्पर्शरेखा} को
का वक्र $ f $ बिंदु पर $(a, f(a))$ की पंक्ति है \omterm{def:g10:algebra:equation}{समीकरण}
\[
y = f(a) + f'(a)(x - a).
\]
\end{definition}

\begin{proposition}\label{prop:g12:deriv:diffcont}
ए \omterm{def:g11:func:function}{समारोह} \omterm{def:g12:deriv:derivative}{विभेदक} पर $ a $ है निरंतर पर $ a $. उलटा झूठ है.
\end{proposition}

\begin{proof}
के लिए $ h \neq 0$ छोटा, $ f(a+h) - f(a) = h \cdot \frac{f(a+h)-f(a)}{h}$. जैसा
$ h \to 0$, दाहिनी ओर की प्रवृत्ति होती है $0 \times f'(a) = 0$, इसलिए
$ f(a+h) \to f(a)$. बातचीत के लिए, $ x \mapsto \abs{x}$ है निरंतर पर
$0$, लेकिन इसका अंतर भागफल पर है $0$ के बराबर होती है $\frac{\abs h}{h} = \pm 1$
के संकेत पर निर्भर करता है $ h $, और इसकी कोई सीमा नहीं है.
\end{proof}

\subsection{विभेदन नियम}

\begin{proposition}[संचालन]\label{prop:g12:deriv:operations}
होने देना $ u, v $ होना \omterm{def:g12:deriv:derivative}{विभेदक} पर $ I $ और $\lambda \in \R $. तब $ u + v $,
$\lambda u $, $ uv $ हैं \omterm{def:g12:deriv:derivative}{विभेदक} पर $ I $, जैसा है $\frac{u}{v}$ कहाँ
$ v \neq 0$, और
\[
(u+v)' = u' + v', \quad (\lambda u)' = \lambda u', \quad
(uv)' = u'v + uv', \quad
\left(\frac{u}{v}\right)' = \frac{u'v - uv'}{v^2}.
\]
\end{proposition}

\begin{proof}[उत्पाद नियम का प्रमाण]
का अंतर भागफल लिखिए $ uv $ पर $ a $ जैसा
\[
\frac{u(a+h)v(a+h) - u(a)v(a)}{h}
= \frac{u(a+h) - u(a)}{h}\,v(a+h) + u(a)\,\frac{v(a+h) - v(a)}{h}.
\]
जैसा $ h \to 0$, $ v(a+h) \to v(a)$ द्वारा निरंतरता
(\cref{prop:g12:deriv:diffcont}), इसलिए दाहिनी ओर की प्रवृत्ति होती है
$ u'(a)v(a) + u(a)v'(a)$. अन्य नियम भी इसी प्रकार \omterm{def:g11:vect:det}{सिद्ध} होते हैं।
\end{proof}

\begin{theorem}[शृंखला नियम]\label{thm:g12:deriv:chain}
\index{शृंखला नियम}
होने देना $ u $ होना \omterm{def:g12:deriv:derivative}{विभेदक} पर $ I $ मूल्यों के साथ $ J $, और $ g $ \omterm{def:g12:deriv:derivative}{विभेदक}
पर $ J $. तब $ g \circ u $ है \omterm{def:g12:deriv:derivative}{विभेदक} पर $ I $ और
\[
(g \circ u)' = (g' \circ u) \cdot u' .
\]
विशेष रूप से, के लिए $ u $ \omterm{def:g12:deriv:derivative}{विभेदक}:
\[
(u^n)' = n\,u'\,u^{n-1} \ (n \in \Z), \qquad
\left(\sqrt{u}\right)' = \frac{u'}{2\sqrt{u}} \ (u > 0), \qquad
\left(\eu^{u}\right)' = u'\,\eu^{u}.
\]
\end{theorem}

\begin{proof}[प्रमाण रेखाचित्र]
कब $ u(a+h) \neq u(a)$ के लिए $ h $ छोटा, लिखो
\[
\frac{g(u(a+h)) - g(u(a))}{h}
= \frac{g(u(a+h)) - g(u(a))}{u(a+h) - u(a)} \cdot \frac{u(a+h) - u(a)}{h}.
\]
जैसा $ h \to 0$, $ u(a+h) \to u(a)$, इसलिए पहला कारक प्रवृत्त होता है $ g'(u(a))$ और
दूसरे को $ u'(a)$. (एक पूर्ण प्रमाण को उस मामले को संभालना चाहिए जहां
$ u(a+h) = u(a)$ के लिए $ h $ मनमाने ढंग से छोटा; इस तकनीकी बिंदु पर विचार किया गया है
विश्वविद्यालय.)
\end{proof}

\begin{example}
सामान्य डेरिवेटिव, प्राकृतिक पर मान्य डोमेन:
\[
(x^n)' = nx^{n-1}, \quad
\left(\frac{1}{x}\right)' = -\frac{1}{x^2}, \quad
(\sqrt{x})' = \frac{1}{2\sqrt{x}}, \quad
(\cos x)' = -\sin x, \quad
(\sin x)' = \cos x .
\]
पहला उत्पाद नियम से प्रेरण द्वारा \omterm{def:g11:vect:det}{सिद्ध} होता है, और अंतिम दो में
\cref{ch:g12:trigo}.
\end{example}

\section{विविधताएँ और चरम}

\begin{theorem}[व्युत्पन्न और विविधता का संकेत]\label{thm:g12:deriv:variations}
होने देना $ f $ होना \omterm{def:g12:deriv:derivative}{विभेदक} एक पर \omterm{def:g10:numbers:interval}{अंतराल} $ I $.
\begin{enumerate}
  \item $ f $ है \omterm{def:g12:seq:monotonic}{की बढ़ती} पर $ I $ अगर और केवल अगर $ f' \geq 0$ पर $ I $.
  \item $ f $ है \omterm{def:g10:functions:variations}{घटते} पर $ I $ अगर और केवल अगर $ f' \leq 0$ पर $ I $.
  \item अगर $ f' > 0$ पर $ I $ सिवाय इसके कि बहुत से बिंदुओं पर यह है
        फिर गायब हो जाता है $ f $ सख्ती से है \omterm{def:g12:seq:monotonic}{की बढ़ती} पर $ I $.
\end{enumerate}
\end{theorem}

\begin{proof}[आंशिक प्रमाण]
यदि $ f $ है \omterm{def:g12:seq:monotonic}{की बढ़ती}, हर अंतर भागफल
$\frac{f(a+h)-f(a)}{h}$ है $\geq 0$, और सीमा तक जाने से मिलता है
$ f'(a) \geq 0$. इसके विपरीत निहितार्थ इस पर निर्भर हैं \omterm{def:g11:stat:mean}{अर्थ} मूल्य प्रमेय,
विश्वविद्यालय में साबित हुआ; वे हैं \emph{इस स्तर पर स्वीकार किया गया}.
\end{proof}

\begin{definition}[स्थानीय चरम]\label{def:g12:deriv:extremum}
$ f $ एक \emph{स्थानीय अधिकतम}\index{स्थानीय अधिकतम} पर $ a $ अगर $ f(x) \leq f(a)$ सभी के लिए $ x $ पास में $ a $; स्थानीय मिनिमा को सममित रूप से परिभाषित किया गया है।
\end{definition}

\begin{proposition}[प्रथम क्रम की स्थिति]\label{prop:g12:deriv:critical}
अगर $ f $ है \omterm{def:g12:deriv:derivative}{विभेदक} एक पर \emph{खुला} \omterm{def:g10:numbers:interval}{अंतराल} $ I $ और एक स्थानीय है
चरम पर $ a \in I $, तब $ f'(a) = 0$. उलटा झूठ है
($ f(x) = x^3$ पर $ a = 0$).
\end{proposition}

\begin{proof}
कहना $ a $ एक है \omterm{def:g12:deriv:extremum}{स्थानीय अधिकतम}. के लिए $ h > 0$ छोटा,
$\frac{f(a+h)-f(a)}{h} \leq 0$, तो दे रहे हैं $ h \to 0^+$ देता है $ f'(a) \leq 0$;
के लिए $ h < 0$ भागफल है $\geq 0$, देना $ f'(a) \geq 0$. इस तरह
$ f'(a) = 0$.
\end{proof}

\begin{method}[विविधता तालिका]\label{met:g12:deriv:table}
अध्ययन करने के लिए ए \omterm{def:g11:func:function}{समारोह} $ f $: इसका निर्धारण करें \omterm{def:g11:func:function}{कार्यक्षेत्र} और सीमा पर सीमाएं;
गणना $ f'$ और इसे कारक बनाएं; का चिह्न निर्धारित करें $ f'$ प्रत्येक उपअंतराल पर;
सब कुछ एक में रिकॉर्ड करें \omterm{def:g10:functions:table}{विविधता तालिका}, एक्स्ट्रेमा को चिह्नित करना; की संख्या निकालें
के समाधान $ f(x) = k $ आक्षेप प्रमेय का उपयोग करना
(\cref{thm:g12:limcont:bijection}).
\end{method}

\section{उत्तलता}

\begin{definition}[उत्तल कार्य]\label{def:g12:deriv:convex}
ए \omterm{def:g11:func:function}{समारोह} $ f $ एक पर परिभाषित \omterm{def:g10:numbers:interval}{अंतराल} $ I $ है \emph{उत्तल}\index{उत्तल}
पर $ I $ यदि इसका प्रत्येक राग \omterm{def:g10:functions:graph}{ग्राफ} के ऊपर स्थित है \omterm{def:g10:functions:graph}{ग्राफ}: सभी के लिए
$ a, b \in I $ और $ t \in \intcc{0}{1}$,
\[
f\bigl((1-t)a + t b\bigr) \leq (1-t) f(a) + t f(b).
\]
यह है \emph{नतोदर}\index{नतोदर} यदि विपरीत असमानता कायम रहती है ($-f $ उत्तल)।
\end{definition}

\begin{theorem}[उत्तलता और व्युत्पन्न]\label{thm:g12:deriv:convexity}
होने देना $ f $ होना \omterm{def:g12:deriv:derivative}{विभेदक} एक पर \omterm{def:g10:numbers:interval}{अंतराल} $ I $. निम्नलिखित समतुल्य हैं:
\begin{enumerate}
  \item $ f $ है \omterm{def:g12:deriv:convex}{उत्तल} पर $ I $;
  \item $ f'$ है \omterm{def:g12:seq:monotonic}{की बढ़ती} पर $ I $;
  \item  \omterm{def:g10:functions:graph}{ग्राफ} का $ f $ इसके प्रत्येक के ऊपर स्थित है \omterm{def:g12:deriv:tangent}{स्पर्शरेखा}.
\end{enumerate}
अगर $ f $ दो बार है \omterm{def:g12:deriv:derivative}{विभेदक}, ये भी समकक्ष हैं $ f'' \geq 0$ पर
$ I $.
\end{theorem}

\begin{proof}[का प्रमाण (2) $\Rightarrow $ (3)]
ठीक करो $ a \in I $ और सेट करें
$ g(x) = f(x) - f(a) - f'(a)(x-a)$, वक्र और के बीच का अंतर \omterm{def:g12:deriv:tangent}{स्पर्शरेखा}
पर $ a $. तब $ g $ है \omterm{def:g12:deriv:derivative}{विभेदक} और $ g'(x) = f'(x) - f'(a)$. अगर $ f'$ है
\omterm{def:g12:seq:monotonic}{की बढ़ती}, $ g' \leq 0$ पर $ I \cap \intoc{-\infty}{a}$ और $ g' \geq 0$ पर
$ I \cap \intco{a}{+\infty}$: $ g $ पहले घट जाती है $ a $ और उसके बाद बढ़ता है,
तो $ g $ उसे प्राप्त कर लेता है \omterm{def:g10:functions:extrema}{न्यूनतम} $ g(a) = 0$, इस तरह $ g \geq 0$.
शेष निहितार्थों को इस स्तर पर स्वीकार किया जाता है; के साथ समानता
$ f'' \geq 0$ से अनुसरण करता है \cref{thm:g12:deriv:variations} के लिए आवेदन किया $ f'$.
\end{proof}

\begin{definition}[विभक्ति बिंदु]\label{def:g12:deriv:inflection}
एक बिंदु जहां का वक्र $ f $ इसे पार करता है \omterm{def:g12:deriv:tangent}{स्पर्शरेखा} एक \emph{विभक्ति
बिंदु}\index{विभक्ति
बिंदु}. के लिए $ f $ दो बार \omterm{def:g12:deriv:derivative}{विभेदक}, विभक्ति
बिंदु वे बिंदु हैं जहां $ f''$ संकेत बदलता है.
\end{definition}

\begin{example}\label{ex:g12:deriv:cube}
$ f(x) = x^3$ है $ f''(x) = 6x $: $ f $ है \omterm{def:g12:deriv:convex}{नतोदर} पर $\intoc{-\infty}{0}$,
\omterm{def:g12:deriv:convex}{उत्तल} पर $\intco{0}{+\infty}$, एक साथ \omterm{def:g12:deriv:inflection}{विभक्ति बिंदु} मूल में,
जहां वक्र इसे पार करता है \omterm{def:g12:deriv:tangent}{स्पर्शरेखा} (द $ x $-अक्ष).
\end{example}

\begin{example}[उत्तलता असमानताएँ]\label{ex:g12:deriv:ineq}
घातांक है \omterm{def:g12:deriv:convex}{उत्तल} पर $\R $ (यह दूसरा है यौगिक स्वयं सकारात्मक है)।
इसका \omterm{def:g12:deriv:tangent}{स्पर्शरेखा} पर $0$ है $ y = 1 + x $, इसलिए
\[
\eu^x \geq 1 + x \quad \text{for all } x \in \R .
\]
ऐसा \emph{\omterm{def:g12:deriv:tangent}{स्पर्शरेखा} रेखा असमानताएँ} के एक मानक उत्पाद हैं उत्तलता.
\end{example}

\begin{method}[उत्तलता का उपयोग करना]\label{met:g12:deriv:convexity}
\begin{itemize}
  \item असमानता साबित करने के लिए $ f(x) \geq ax + b $: रेखा को a के रूप में पहचानें
        \omterm{def:g12:deriv:tangent}{स्पर्शरेखा} एक का \omterm{def:g12:deriv:convex}{उत्तल} $ f $ और आह्वान करें
        \cref{thm:g12:deriv:convexity}(3).
  \item स्थित करना \omterm{def:g12:deriv:inflection}{विभक्ति बिंदु}: हल करना $ f''(x) = 0$ \emph{और} जाँच करें कि $ f''$ वहां परिवर्तन का चिह्न है.
  \item में एक \omterm{def:g10:functions:table}{विविधता तालिका}, उत्तलता वक्र के रेखाचित्र को परिष्कृत करता है
        (यह किस ओर झुकता है)।
\end{itemize}
\end{method}

\begin{exercise}[$\star $]\label{exo:g12:deriv:2}

\begin{exercise}[$\star $]\label{exo:g12:deriv:3}

\qquad\text{and}\qquad

यौगिक, \omterm{def:g10:functions:table}{विविधता तालिका}, स्थानीय एक्स्ट्रेमा।
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:deriv:5}
बिना ढक्कन वाला एक बक्सा कार्डबोर्ड की एक चौकोर शीट से बनाया गया है
$30\,$ भुजा के बराबर वर्ग काटकर सेमी $x $ कोनों से और मोड़कर
पक्ष. ठानना $ x $ बॉक्स का आयतन \omterm{def:g10:functions:extrema}{अधिकतम} करना।
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:deriv:6}
होने देना $ f(x) = x^4 - 4x^3 + 10$.
\begin{enumerate}
  \item गणना करना $ f''$ और निर्धारित करें उत्तलता \omterm{def:g10:numbers:interval}{अंतराल} और विभक्ति
        अंक का $ f $.
  \item दिखाओ कि \omterm{def:g12:deriv:tangent}{स्पर्शरेखा} पर \omterm{def:g12:deriv:inflection}{विभक्ति बिंदु} का \omterm{def:g10:coordgeom:system}{सूच्याकार आकृति का भुज} $2$
        वहाँ वक्र को पार करता है।
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:deriv:7}
ए का उपयोग करना \omterm{def:g12:deriv:tangent}{स्पर्शरेखा} रेखा असमानता, इसे सभी के लिए दिखाएं $ x > 0$,
\[
\sqrt{x} \leq \frac{x + 1}{2},
\]
और पहचानें कि समानता कब कायम रहती है। (संकेत: वर्गमूल है \omterm{def:g12:deriv:convex}{नतोदर}.)
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g12:deriv:8}
होने देना $ f $ होना \omterm{def:g12:deriv:convex}{उत्तल} और \omterm{def:g12:deriv:derivative}{विभेदक} पर $\R $, और मान लीजिये $ f'$ गायब हो जाता है
किसी बिंदु पर $ a $. बताते हैं कि $ f(a)$ है \emph{वैश्विक} \omterm{def:g10:functions:extrema}{न्यूनतम} का $ f $
पर~$\R $.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g12:deriv:9}
इसे निश्चित परिधि वाले सभी आयतों के बीच दिखाएँ $ p $, वर्ग में है
सबसे बड़ा क्षेत्र. फिर उसे निश्चित क्षेत्रफल के सभी आयतों के बीच दिखाएँ $ A $, द
वर्ग का परिमाप सबसे छोटा है और समझाइए कि दोनों कथन कैसे हैं
संबंधित.
\end{exercise}

\section{समस्या: लाइफगार्ड की गणना}

\begin{problem}[{सप्ताहांत समस्या--- सबसे तेज़ बचाव पथ का पालन करता है स्नेल का प्रकाश का नियम, उत्तलता प्रत्येक इष्टतम को प्रमाणित करता है, और \omterm{def:g11:scal:dot}{आदर्श} सोडा कैन बिल्कुल उतना ही लंबा होता है जितना चौड़ा होता है}]
\label{pb:g12:deriv:1}
एक लाइफगार्ड एक तैराक को मुसीबत में देखता है --- तिरछे नीचे
समुद्रतट, बाहर पानी में। दौड़ना तैराकी से भी तेज़ है:
सीधी रेखा है \emph{नहीं} सबसे तेज़ मार्ग. वह मार्ग
तटरेखा पर समय के मोड़ को कम करता है, और इसका झुकने का नियम है
ठीक वह नियम जिसके द्वारा प्रकाश पानी में प्रवेश करके अपवर्तित होता है: प्रकृति,
भी, अंतर करता है. यह समस्या श्रृंखला नियम को प्रशिक्षित करती है
(\cref{thm:g12:deriv:chain}), बचाव करता है, कटाई करता है
की असमानताएँ उत्तलता (\cref{thm:g12:deriv:convexity}), और
एक टिन कैन डिज़ाइन करता है।

\medskip
\noindent\textbf{Part I --- Chain-rule fluency.}
\begin{enumerate}
  \item अंतर करें: $\left(3x^2 + 1\right)^5$; \
        $\sqrt{x^2 + 9}$; \ $\dfrac{1}{x^2 + 1}$.
  \item दे दो \omterm{def:g12:deriv:tangent}{स्पर्शरेखा} को $ y = \sqrt{x^2 + 9}$ पर $ x = 4$.
  \item का निर्माण करें \omterm{def:g10:functions:table}{विविधता तालिका} का
        $ f(x) = \dfrac{x}{x^2 + 1}$ पर $\R $ (एक्स्ट्रीमा शामिल)।
  \item के लिए $ g(x) = x^3 - 3x^2 + 4$: \omterm{def:g10:functions:table}{विविधता तालिका} \emph{और}
        उत्तलता मेज़ ($ g''$), साथ \omterm{def:g12:deriv:inflection}{विभक्ति बिंदु}
        (\cref{def:g12:deriv:inflection}).
  \item की गणना करें \omterm{def:g12:deriv:tangent}{स्पर्शरेखा} को $ g $ पर इसके \omterm{def:g12:deriv:inflection}{विभक्ति बिंदु},
        और दिखाओ $ g(x) - (\text{tangent}) = (x - 1)^3$:
        कैसे के बारे में संकेत परिवर्तन क्या कहता है \omterm{def:g12:deriv:tangent}{स्पर्शरेखा}
        एक विभक्ति पर वक्र से मिलता है?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The rescue.}
तटरेखा के किनारे $ x $-अक्ष; लाइफगार्ड रेत पर खड़ा है
$ A(0, 30)$, तैराक इंतज़ार करता है $ B(40, -20)$ (मीटर)। द
गार्ड चलता है $5$ रेत पर मी/से, तैरता है $2$ एम/एस, और में प्रवेश करता है
एक बिंदु पर पानी $(x, 0)$ उसकी पसंद का.
\begin{enumerate}[resume]
  \item मॉडल: कुल बचाव समय व्यक्त करें $ T(x)$, और बताओ क्यों
        केवल $ x \in \intcc{0}{40}$ विचार करने योग्य है.
  \item अंतर $ T $ (काम पर श्रृंखला नियम --- दो बार)।
  \item होने देना $\theta_1$ के साथ चलने वाले पैर का कोण बनें
        \emph{सीधा} किनारे करने के लिए, $\theta_2$ का
        तैराकी पैर. बताते हैं कि
        $\sin\theta_1 = \frac{x}{\sqrt{x^2 + 900}}$ और वह
        शर्त $ T'(x) = 0$ पढ़ता
        \[
        \frac{\sin\theta_1}{5} = \frac{\sin\theta_2}{2}
        \]
        --- \emph{स्नेल का नियम}, के स्थान पर दो गतियों के साथ
        हवा और पानी में प्रकाश की गति.
  \item सीधा खंड $[AB]$ पर तट पार करता है
        $ x = 24$. गणना करना $ T(24)$, $ T(40)$ (साथ-साथ दौड़ें
        समुद्र तट, फिर सीधे तैरना) और $ T(0)$: है
        ज्यामितीय सीधी रेखा सबसे तेज़? या तो अति है
        रणनीति?
  \item द्वारा इष्टतम प्रवेश बिंदु का पता लगाएं \omterm{thm:g12:limcont:ivt}{विरोधाभास} पर
        $ T'$ (यह है $\approx 33.7$ एम): देना $ x^*$ मीटर को
        और रिकॉर्ड समय $ T(x^*)$ एक सेकंड के दसवें हिस्से तक.
        इष्टतम मोड़ सीधे की तुलना में कितनी बचत करता है?
        लाइन?
  \item महत्वपूर्ण बिंदु क्यों है? \emph{वैश्विक} \omterm{def:g10:functions:extrema}{न्यूनतम}?
        (प्रत्येक पद $ T $ है \omterm{def:g12:deriv:convex}{उत्तल} --- स्वीकार करें कि का योग
        \omterm{def:g12:deriv:convex}{उत्तल} कार्य है \omterm{def:g12:deriv:convex}{उत्तल} --- और आवेदन करें
        \cref{exo:g12:deriv:8}.)
  \item फ़र्मेट का \omterm{def:g11:vect:det}{सिद्ध}ांत कहता है कि प्रकाश हमेशा यात्रा करता है
        कम से कम का रास्ता \emph{समय}. पता लगाएँ कि प्रकाश किरण क्यों मुड़ती है
        पानी में प्रवेश करते समय बिल्कुल सतह पर
        (जहाँ प्रकाश धीमा है), और प्रतिदिन का नाम बताएं
        अवलोकन यह एक गिलास में एक पेंसिल के बारे में बताता है
        पानी.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- उत्तलता's harvest.}
\begin{enumerate}[resume]
  \item बताते हैं कि $ x \mapsto x^2$ है \omterm{def:g12:deriv:convex}{उत्तल}, और \omterm{def:g11:vect:det}{सिद्ध} करें
        \omterm{prop:g10:coordgeom:midpoint}{मध्य} असमानता
        $\left(\frac{a + b}{2}\right)^2 \leq
        \frac{a^2 + b^2}{2}$ दो बार: एक बार से उत्तलता, एक बार
        द्वारा का \omterm{def:g10:algebra:expand}{विस्तार} $(a - b)^2 \geq 0$.
  \item यह निष्कर्ष निकालें कि \emph{द्विघात \omterm{def:g11:stat:mean}{अर्थ}}
        $\sqrt{\frac{a^2 + b^2}{2}}$ \omterm{def:g12:seq:arith-geom}{अंकगणित} पर हावी है
        \omterm{def:g11:stat:mean}{अर्थ}, और श्रृंखला की पूरी श्रृंखला को इकट्ठा करें ---
        हार्मोनिक $\leq $ \omterm{def:g12:seq:arith-geom}{ज्यामितिक} $\leq $ \omterm{def:g12:seq:arith-geom}{अंकगणित} $\leq $
        द्विघात --- चारों को सत्यापित करना $ a = 2$, $ b = 8$.
  \item स्पर्शरेखा-वक्र के नीचे: $ x \mapsto \frac1x $ है \omterm{def:g12:deriv:convex}{उत्तल}
        पर $\intoo{0}{+\infty}$; इसे लिखें \omterm{def:g12:deriv:tangent}{स्पर्शरेखा} पर $1$ और
        असमानता कम करें
        $\frac1x \geq 2 - x $ सभी के लिए $ x > 0$. यह कहाँ है?
        समानता?
  \item जंगली में विभक्ति: एक महामारी के दौरान,
        संचयी मामले की गिनती एस-आकार के वक्र का अनुसरण करती है। क्या
        अपने पर होता है \omterm{def:g12:deriv:inflection}{विभक्ति बिंदु}, और वह तारीख क्यों है
        महामारीविज्ञानी किस पर नज़र रखते हैं? संकेत से संबंधित
        दूसरे का यौगिक हर तरफ.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The ideal can.}
एक बेलनाकार कैन अवश्य धारण करना चाहिए $ V = 330$ सेमी $^3$ सबसे कम के साथ
धातु: सतह $ S = 2\pi r^2 + 2\pi r h $, बाधा
$\pi r^2 h = V $.
\begin{enumerate}[resume]
  \item अभिव्यक्त करना $ S(r) = 2\pi r^2 + \dfrac{2V}{r}$,
        अंतर करें, और इष्टतम त्रि\omterm{def:g11:trigo:cossin}{ज्या} की गणना करें
        के लिए ऊंचाई $ V = 330$ (मिलीमीटर परिशुद्धता).
  \item सुरुचिपूर्ण सामान्य नियम \omterm{def:g11:vect:det}{सिद्ध} करें: इष्टतम पर,
        $ h = 2r $ --- \omterm{def:g11:scal:dot}{आदर्श} कैन बिल्कुल उतना ही लंबा है जितना वह है
        विस्तृत.
  \item जाँच करना $ S''(r) > 0$ और निष्कर्ष (के माध्यम से)
        \cref{exo:g12:deriv:8} पुनः) कि इष्टतम है
        वैश्विक. असली सोडा के डिब्बे चौड़े की तुलना में स्पष्ट रूप से लम्बे होते हैं:
        एक गैर-गणितीय कारण का नाम बताएं।
  \item समापन --- अनुकूलन पाइपलाइन: मॉडल, पूछें
        यौगिक, महत्वपूर्ण बिंदुओं को हल करें, प्रमाणित करें
        उत्तलता या एक तालिका, व्याख्या करें। सूची को ऊपर चलाएँ
        लाइफगार्ड, कैन, और ग्रेड 11 की सबसे मजबूत किरण
        (\cref{pb:g11:deriv:1}) --- और बताएं कि फ़र्मेट क्या है
        \omterm{def:g11:vect:det}{सिद्ध}ांत कहता है कि इस पाइपलाइन को और कौन चलाता है।
\end{enumerate}
\end{problem}
