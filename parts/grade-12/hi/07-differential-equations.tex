\chapter{विभेदक समीकरण}\label{ch:g12:diffeq}

ए \omterm{def:g12:diffeq:ode}{विभेदक समीकरण} ए से संबंधित है \omterm{def:g11:func:function}{समारोह} इसके लिए डेरिवेटिव. भौतिकी,
रसायन विज्ञान, जीव विज्ञान और अर्थशास्त्र अपने कानूनों को इस रूप में व्यक्त करते हैं: की दर
शरीर का ठंडा होना, रेडियोधर्मी नाभिक का क्षय, की वृद्धि
जनसंख्या के बारे में सभी कथन हैं $y'$. यह अध्याय पूर्णतः हल करता है
रैखिक प्रथम क्रम \omterm{def:g10:algebra:equation}{समीकरण} निरंतर गुणांक के साथ.

\section{समीकरण \texorpdfstring{$ y' = ay $}{y' = ay}}

\begin{definition}[विभेदक समीकरण]\label{def:g12:diffeq:ode}
ए \emph{विभेदक समीकरण}\index{विभेदक समीकरण} एक \omterm{def:g10:algebra:equation}{समीकरण}
जिसका अज्ञात है \omterm{def:g11:func:function}{समारोह} $ y $, शामिल है $ y $ और इसके डेरिवेटिव. ए
\emph{समाधान} एक पर \omterm{def:g10:numbers:interval}{अंतराल} $ I $ एक है \omterm{def:g12:deriv:derivative}{विभेदक} \omterm{def:g11:func:function}{समारोह} संतोषजनक
द \omterm{def:g10:algebra:equation}{समीकरण} के हर बिंदु पर $ I $.
\end{definition}

\begin{theorem}[के समाधान $ y' = ay $]\label{thm:g12:diffeq:homogeneous}
होने देना $ a \in \R $. समाधान चालू $\R $ की \omterm{def:g10:algebra:equation}{समीकरण} $ y' = ay $ बिलकुल हैं
द कार्य
\[
y(x) = C\,\eu^{ax}, \qquad C \in \R .
\]
हर \omterm{def:g10:vectors:sum}{जोड़}ी के लिए $(x_0, y_0)$ के साथ एक अनोखा समाधान है
$ y(x_0) = y_0$.
\end{theorem}

\begin{proof}
प्रत्येक $ y = C\eu^{ax}$ संतुष्ट $ y' = Ca\,\eu^{ax} = ay $. इसके विपरीत, चलो $ y $
कोई भी समाधान और सेट हो $ z(x) = y(x)\,\eu^{-ax}$. तब
\[
z'(x) = y'(x)\,\eu^{-ax} - a\,y(x)\,\eu^{-ax}
= \bigl(y'(x) - a y(x)\bigr)\eu^{-ax} = 0,
\]
इसलिए $ z $ स्थिर है, मान लीजिए $ C $, और $ y(x) = C\eu^{ax}$. प्रारंभिक स्थिति
$ y(x_0) = y_0$ बल $ C = y_0 \eu^{-a x_0}$, विशिष्ट रूप से।
\end{proof}

\begin{remark}
घातांक इस प्रकार है \emph{विशेषता} सबसे सरलतम तरीके से
\omterm{def:g12:diffeq:ode}{विभेदक समीकरण}: आकार के अनुपात में वृद्धि। यही कारण है कि ऐसा प्रतीत होता है
प्रकृति में हर जगह.
\end{remark}

\begin{example}[रेडियोधर्मी क्षय]\label{ex:g12:diffeq:decay}
एक रेडियोधर्मी मात्रा संतुष्ट करती है $ N' = -\lambda N $ साथ $\lambda > 0$, इसलिए
$ N(t) = N_0\,\eu^{-\lambda t}$; देखना \cref{ex:g12:exp:halflife} के लिए
आधा जीवन.
\end{example}

\section{समीकरण \texorpdfstring{$ y' = ay + b $}{y' = ay + b}}

\begin{theorem}[के समाधान $ y' = ay + b $]\label{thm:g12:diffeq:affine}
होने देना $ a \neq 0$ और $ b \in \R $. समाधान चालू $\R $ का $ y' = ay + b $ हैं
बिलकुल कार्य
\[
y(x) = C\,\eu^{ax} - \frac{b}{a}, \qquad C \in \R .
\]
स्थिरांक \omterm{def:g11:func:function}{समारोह} $ y_p = -\frac ba $ है \emph{संतुलन समाधान}.
हरएक के लिए $(x_0, y_0)$ के साथ एक अनोखा समाधान है $ y(x_0) = y_0$; अगर
$ a < 0$, प्रत्येक समाधान संतुलन की ओर प्रवृत्त होता है $-\frac ba $ जैसा
$ x \to +\infty $.
\end{theorem}

\begin{proof}
स्थिरांक $ y_p = -\frac{b}{a}$ संतुष्ट $ y_p' = 0 = a y_p + b $. अब $ y $
एक समाधान है यदि और केवल यदि
\[
(y - y_p)' = y' = ay + b = a(y - y_p) + \underbrace{a y_p + b}_{=\,0}
= a (y - y_p),
\]
\omterm{def:g11:stat:mean}{अर्थ}ात्, यदि और केवल यदि $ z = y - y_p $ हल करती है $ z' = az $. द्वारा
\cref{thm:g12:diffeq:homogeneous}, $ z = C\eu^{ax}$, सूत्र कहां से,
अस्तित्व और विशिष्टता. अगर $ a < 0$, $\eu^{ax} \to 0$ जैसा
$ x \to +\infty $, इसलिए $ y(x) \to -\frac ba $.
\end{proof}

\begin{method}[हल $ y' = ay + b $ प्रारंभिक शर्त के साथ]\label{met:g12:diffeq:solve}
\begin{enumerate}
  \item संतुलन समाधान खोजें $ y_p = -\frac{b}{a}$ (हल करना $ y' = 0$).
  \item सामान्य समाधान लिखें $ y = C\eu^{ax} + y_p $.
  \item ठानना $ C $ प्रारंभिक स्थिति से.
  \item शारीरिक अंतर्ज्ञान के विरुद्ध दीर्घकालिक व्यवहार की जाँच करें (करता है)।
        समाधान संतुलन में परिवर्तित होता है?)
\end{enumerate}
वही रणनीति --- \emph{विशेष समाधान + सामान्य समाधान
सजातीय \omterm{def:g10:algebra:equation}{समीकरण}} ---तक \omterm{def:g10:algebra:expand}{विस्तार}ित है $ y' = ay + f(x)$: देखना
\cref{prop:g12:diffeq:general}.
\end{method}

\begin{example}[न्यूटन का शीतलन नियम]\label{ex:g12:diffeq:cooling}
पर एक कप कॉफ़ी $80\,^\circ $ C एक कमरे में बैठता है $20\,^\circ $ C. न्यूटन का नियम बताता है
वह तापमान $T $ संतुष्ट $ T' = -k(T - 20)$ कुछ के लिए $ k > 0$,
\emph{यानी} $ T' = -kT + 20k $. संतुलन है $20$, और
$ T(t) = 20 + 60\,\eu^{-kt}$: कॉफ़ी कमरे में तेजी से ठंडी हो जाती है
तापमान.
\end{example}

\section{समीकरण \texorpdfstring{$ y' = ay + f(x)$}{y' = ay + f(x)}}

\begin{proposition}[समाधान सेट की संरचना]\label{prop:g12:diffeq:general}
होने देना $ f $ होना निरंतर एक पर \omterm{def:g10:numbers:interval}{अंतराल} $ I $ और चलो $ y_p $ एक विशेष हो
का समाधान
\[
y' = ay + f(x)
\]
पर $ I $. फिर समाधान चालू $ I $ बिल्कुल वही हैं कार्य
$ y = C\eu^{ax} + y_p $, $ C \in \R $.
\end{proposition}

\begin{proof}
के रूप में \cref{thm:g12:diffeq:affine}: $ y $ एक समाधान है यदि और केवल यदि
$ z = y - y_p $ संतुष्ट
$ z' = (ay + f) - (ay_p + f) = az $, अगर और केवल अगर $ z = C\eu^{ax}$.
\end{proof}

\begin{method}[किसी विशेष समाधान का अनुमान लगाना]\label{met:g12:diffeq:particular}
किसी विशेष समाधान की तलाश करें \emph{एक ही आकार का} जैसा $ f $:
\begin{itemize}
  \item $ f $ डिग्री का बहुपद $ n $: घात के बहुपद का प्रयास करें $ n $;
  \item $ f(x) = \alpha\,\eu^{kx}$ साथ $ k \neq a $: कोशिश $ y_p = \beta\,\eu^{kx}$;
  \item $ f(x) = \alpha\,\eu^{ax}$ (गुंजयमान मामला): प्रयास करें
        $ y_p = \beta x\,\eu^{ax}$.
\end{itemize}
में प्रतिस्थापित करें \omterm{def:g10:algebra:equation}{समीकरण} और गुणांकों की पहचान करें।
\end{method}

\begin{example}\label{ex:g12:diffeq:particular}
हल करना $ y' = 2y + 4x $. कोशिश $ y_p = \alpha x + \beta $:
$\alpha = 2(\alpha x + \beta) + 4x $ सभी के लिए $ x $ बल
$2\alpha + 4 = 0$ और $\alpha = 2\beta $, इसलिए $\alpha = -2$, $\beta = -1$.
सामान्य समाधान: $ y = C\eu^{2x} - 2x - 1$.
\end{example}

\section{अभ्यास}

\begin{exercise}[$\star $]\label{exo:g12:diffeq:1}
पर समाधान करें $\R $:
(ए) $ y' = 3y $ साथ $ y(0) = 2$;\quad
(बी) $2y' + y = 0$ साथ $ y(0) = -1$;\quad
(सी) $ y' = -y + 5$ साथ $ y(0) = 0$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g12:diffeq:2}
एक जीवाणु आबादी अपने आकार के अनुपात में दोगुनी दर से बढ़ती है
प्रत्येक $3$ घंटे। लिखना \omterm{def:g12:diffeq:ode}{विभेदक समीकरण} जनसंख्या से संतुष्ट
$ N(t)$ और आनुपातिकता स्थिरांक निर्धारित करें।
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g12:diffeq:3}
सत्यापित करें कि $ y_p(x) = x\,\eu^{x}$ का एक समाधान है $ y' = y + \eu^x $, और
पर सभी समाधान दें $\R $.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:diffeq:4}
हल करना $ y' = -2y + \eu^{x}$ साथ $ y(0) = 1$.
(संकेत: फॉर्म के किसी विशेष समाधान की तलाश करें $\beta\,\eu^{x}$.)
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:diffeq:5}
कार्बन-14 का आधे जीवन के साथ क्षय होता है $5730$ साल। एक पुरातात्विक \omterm{def:g10:proba:sample}{नमूना}
रोकना $60\%$ किसी जीवित जीव के कार्बन-14 का। इसकी उम्र का अनुमान लगाएं.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:diffeq:6}
एक टैंक में शामिल है $100$ शुद्ध जल का एल. नमकीन पानी युक्त $0.2$ प्रति किलो नमक
लीटर में प्रवाहित होता है $5$ एल/मिनट, मिश्रण (एक समान रखा गया) उसी गति से बहता है
दर. होने देना $ m(t)$ समय पर टैंक में नमक का द्रव्यमान हो $ t $ (मिनटों में).
\begin{enumerate}
  \item उसे उचित ठहराओ $ m' = 1 - \dfrac{m}{20}$.
  \item हल करें, और की सीमा निर्धारित करें $ m(t)$ जैसा $ t \to +\infty $.
        व्याख्या करना।
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:diffeq:7}
द्रव्यमान का एक स्काइडाइवर $80$ किलोग्राम गुरुत्वाकर्षण के अधीन गिरता है
($ g = 9.8\ \text{m/s}^2$) और वायु प्रतिरोध गति के समानुपाती होता है, ताकि
उसकी गति संतुष्ट करती है $ v' = g - \frac{k}{m}v $ साथ $ k = 16$ किग्रा/से.
\begin{enumerate}
  \item हल करें \omterm{def:g10:algebra:equation}{समीकरण} साथ $ v(0) = 0$.
  \item की गणना करें \emph{टर्मिनल वेग} $\lim_{t\to+\infty} v(t)$, और
        पहुँचने के लिए आवश्यक समय $95\%$ इसका.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g12:diffeq:8}
\emph{(लॉजिस्टिक \omterm{def:g10:algebra:equation}{समीकरण}.)} एक आबादी $ y(t) \in \intoo{0}{1}$ (एक के रूप में)
\omterm{def:g10:functions:extrema}{अधिकतम} जनसंख्या का अंश) संतुष्ट करता है
\[
y' = y(1 - y).
\]
\begin{enumerate}
  \item होने देना $ z = \dfrac{1}{y}$. बताते हैं कि $ z $ रैखिक को संतुष्ट करता है \omterm{def:g10:algebra:equation}{समीकरण}
        $ z' = -z + 1$.
  \item के लिए हल करें $ z $, फिर के लिए $ y $, साथ $ y(0) = \frac{1}{10}$.
  \item बताते हैं कि $ y(t) \to 1$ जैसा $ t \to +\infty $ और इसका आकार स्केच करें
        समाधान वक्र.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g12:diffeq:9}
होने देना $ y $ का समाधान हो $ y' = ay + b $ और चलो $ u_n = y(n)$ के लिए $ n \in \N $.
बताते हैं कि $(u_n)$ अंकगणित-ज्यामितीय पुनरावृत्ति को संतुष्ट करता है
$ u_{n+1} = q u_n + r $, और व्यक्त करें $ q $ और $ r $ के अनुसार $ a $ और $ b $. क्या
शर्त करता है $\abs{q} < 1$ के लिए अनुरूप अंतर
\omterm{def:g10:algebra:equation}{समीकरण}?
\end{exercise}

\section{समस्या: चीज़ों के अंदर की घड़ी}

\begin{problem}[{सप्ताहांत समस्या --- कार्बन-14 गुफाओं की तिथियां, न्यूटन का शीतलन नियम एक अपराध का समय है, और एक छोटा सा समीकरण ख़राब हो जाता है चार पोशाकें}]
\label{pb:g12:diffeq:1}
ए \omterm{def:g12:diffeq:ode}{विभेदक समीकरण} परिवर्तन का नियम है; इसे हल करने से यह बदल जाता है
एक में कानून \emph{घड़ी}. वही छोटा सा \omterm{def:g10:algebra:equation}{समीकरण}
$ y' = ay + b $ (\cref{thm:g12:diffeq:affine}) अंदर टिक करता है
प्रागैतिहासिक लकड़ी का कोयला, ठंडी कॉफी, गिरती बारिश की बूंदें और
अस्पताल में टपकना --- और उन घड़ियों को पढ़ना इस समस्या का है
व्यवसाय: यह लास्कॉक्स पेंटिंग्स की तारीख तय करता है, मृत्यु का समय तय करता है,
और एक अच्छे वैज्ञानिक की तरह अपनी मान्यताओं का ऑडिट करता है।

\medskip
\noindent\textbf{Part I --- Fluency.}
\begin{enumerate}
  \item हल करना $ y' = 3y $, $ y(0) = 2$; तब
        $ y' = -2y + 6$, $ y(0) = 0$
        (\cref{met:g12:diffeq:solve}).
  \item पीछे की विशिष्टता युक्ति को पुनः प्राप्त करें
        \cref{thm:g12:diffeq:homogeneous}: अगर $ y' = ay $, गणना करें
        द यौगिक का $ y(t)\,\eu^{-at}$ और निष्कर्ष निकालो
        \emph{प्रत्येक} समाधान है $ C\eu^{at}$.
  \item के लिए $ y' = -2y + 6$: संतुलन समाधान खोजें, और
        प्रत्येक अन्य समाधान के भाग्य का वर्णन इस प्रकार करें
        $ t \to +\infty $. (द निश्चित बिंदु का
        \cref{pb:g11:seq:1}की पुनरावृत्ति, चला गया निरंतर.)
  \item क्षय कानून के लिए $ y' = -\frac{y}{\tau}$:दिखाओ कि
        आधा जीवन है $ t_{1/2} = \tau \ln 2$.
  \item के समाधानों के परिवार का रेखाचित्र बनाएं (या वर्णन करें)।
        $ y' = -2y + 6$: ऊपर से शुरू होने वाले समाधान क्या हैं $3$
        करना? नीचे $3$? पर $3$?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Carbon-14.}
जीवित ऊतक रेडियोधर्मी का एक स्थिर अनुपात बनाए रखता है
कार्बन-14; मृत्यु पर सेवन बंद हो जाता है और स्टॉक ख़त्म हो जाता है:
$ N' = -\lambda N $, आधे जीवन के साथ $5\,730$ साल।
\begin{enumerate}[resume]
  \item गणना करना $\lambda $ (प्रति वर्ष).
  \item एक हड्डी बरकरार रहती है $20\,\%$ जीवित अनुपात का: कितना पुराना है
        यह?
  \item लास्काक्स की चित्रित गुफाओं से लकड़ी का कोयला बचा हुआ है
        के बारे में $15\,\%$: चित्रों की तिथि निर्धारित करें।
  \item हिम पुरुष \"अल्पाइन ग्लेशियर में पाए जाने वाले ओट्ज़ी को मापा गया
        के बारे में $53\,\%$: उसे दिनांकित करें (पुरातत्ववेत्ता इसके बारे में कहते हैं
        $5\,300$ वर्ष ---आपने कैसे किया?)।
  \item कार्बन-14 डायनासोर की तिथि क्यों नहीं बता सकता? की गणना करें
        दस अर्ध-जीवन के बाद शेष अंश बताएं
        अंश के बाद $65$ की शक्ति के रूप में लाखों वर्ष $2$,
        और निष्कर्ष निकालो.
  \item त्रुटि पट्टियाँ: यदि \"ओत्ज़ी का $53\,\%$ ही जाना जाता है
        $\pm 1\,\%$, आयु सीमा की गणना करें। क्या प्रयोगशाला परिशुद्धता
        क्या खरीदता है डेटिंग परिशुद्धता?
\end{enumerate}

\noindent\textbf{Part III --- Newton's cooling, and a crime.}
तापमान पर एक शरीर $ T $ लगातार परिवेश में $ T_a $
द्वारा ठंडा होता है $ T' = -k\,(T - T_a)$.
\begin{enumerate}[resume]
  \item हल करें \omterm{def:g10:algebra:equation}{समीकरण} (विकल्प $ z = T - T_a $):
        \[
        T(t) = T_a + (T_0 - T_a)\,\eu^{-kt} .
        \]
  \item कॉफ़ी डाली गई $90\,^\circ $ सी में ए $20\,^\circ $ सी कमरा
        पढ़ता है $70\,^\circ $ सी के बाद $5$ मिनट। ठानना $ k $,
        फिर पीने योग्य होने तक प्रतीक्षा का समय
        $55\,^\circ $ सी।
  \item दूध का प्रश्न: यथासंभव गर्म कॉफी पीना
        दस मिनट में, ठंडा दूध अभी जाना चाहिए या इतने बजे
        अंतिम क्षण? के साथ उत्तर दें \omterm{def:g10:algebra:equation}{समीकरण} (क्या करता है
        गैप में दूध डालकर मिला दीजिये $T - T_a $, और कैसे करता है
        गैप से नुकसान होता है?)
  \item फोरेंसिक: आधी रात को एक शव मिला
        $30\,^\circ $ सी में ए $20\,^\circ $ सी कमरा; एक घंटे बाद
        यह पढ़ता है $28\,^\circ $ सी. मान लिया जाए $37\,^\circ $ सी पर
        मृत्यु का समय, ठीक हो जाओ $k $ दो मापों से,
        फिर मृत्यु के समय की गणना करें।
  \item कोरोनर की घड़ी का ऑडिट करें: वास्तविक दुनिया के तीन नाम बताएं
        मॉडल की मान्यताओं और दिशा का उल्लंघन
        जिसमें प्रत्येक व्यक्ति मृत्यु के अनुमानित समय में पक्षपात करेगा।
\end{enumerate}

\noindent\textbf{Part IV --- Four costumes and an S-curve.}
\begin{enumerate}[resume]
  \item रसद \omterm{def:g10:algebra:equation}{समीकरण} का \cref{exo:g12:diffeq:8},
        $ y' = y(1 - y)$ साथ $ y(0) = 0.1$: व्यायाम का उपयोग करना
        प्रतिस्थापन, व्युत्पन्न
        $ y(t) = \dfrac{1}{1 + 9\eu^{-t}}$, और का समय ज्ञात करें
        द \omterm{def:g12:deriv:inflection}{विभक्ति बिंदु} $ y = \frac12$ --- द
        महामारी विज्ञानी की तिथि \cref{pb:g12:deriv:1}, अभी
        गणना योग्य.
  \item एक बारिश की बूंद आज्ञा का पालन करती है $ v' = 10 - \frac v2$ (गुरुत्वाकर्षण शून्य
        खींचें), $ v(0) = 0$. टर्मिनल वेग ज्ञात करें, हल करें
        के लिए $ v(t)$, और गणना करें कि बूंद कब पहुँचती है
        $95\,\%$ टर्मिनल गति का.
  \item एक अस्पताल ड्रिप स्थिर दर पर एक दवा वितरित करता है
        शरीर इसे आनुपातिक रूप से समाप्त कर देता है:
        $ c' = 4 - 0.5c $, $ c(0) = 0$. स्थिर अवस्था ज्ञात कीजिए
        एकाग्रता और उसके आधे तक पहुँचने का समय। राज्य
        एक वाक्य में ऋण पुनरावृत्ति के साथ समानांतर
        का \cref{exo:g12:diffeq:9}.
  \item समापन---एक \omterm{def:g10:algebra:equation}{समीकरण}, चार वेशभूषा: क्षय, शीतलता,
        गिरना, डूबना, सब कुछ $ y' = ay + b $ विभिन्न चिन्हों के साथ
        और नाम. इस समस्या से जुड़े मॉडलिंग लूप को याद करें
        चार बार (कानून $\to $ \omterm{def:g10:algebra:equation}{समीकरण} $\to $ समाधान $\to $
        अंशांकन $\to $ भविष्यवाणी $\to $ ऑडिट), और नाम
        वह घटना जिसकी आवश्यकता है $ y''$ ---और कौन सी समस्या
        यह पहले ही मिल चुका है।
\end{enumerate}
\end{problem}
