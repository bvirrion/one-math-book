\chapter{Exponencial e Logaritmo}\label{ch:g12:exp}

O \omterm{thm:g12:exp:existence}{função exponencial} é o único \omterm{def:g11:func:function}{função} igual ao seu próprio derivado
e pegando o valor $1$ no $0$. Converte somas em produtos; seu inverso,
o logaritmo natural, converte produtos em somas. Juntos eles descrevem
todo fenômeno cuja taxa de mudança é proporcional ao seu tamanho:
decaimento radioativo, crescimento populacional, juros compostos.

\section{A função exponencial}

\begin{theorem}[Existência e singularidade]\label{thm:g12:exp:existence}
Existe um único \omterm{def:g12:deriv:derivative}{diferenciável} \omterm{def:g11:func:function}{função} $ f \colon \R \to \R $ tal que
\[
f' = f \qquad\text{and}\qquad f(0) = 1 .
\]
É chamado de \emph{função exponencial}\index{função exponencial} e escrito
$\exp $, ou $ x \mapsto \eu^x $.
\end{theorem}

\begin{proof}[Prova de exclusividade]
Primeiro, tal \omterm{def:g11:func:function}{função} nunca desaparece. Na verdade, deixe
$ g(x) = f(x)f(-x)$; então
\[
g'(x) = f'(x)f(-x) - f(x)f'(-x) = f(x)f(-x) - f(x)f(-x) = 0,
\]
então $ g $ é constante igual a $ g(0) = 1$: para cada $ x $,
$ f(x)f(-x) = 1$, e em particular $ f(x) \neq 0$.

Agora deixe $ f_1, f_2$ ser duas soluções e $ h = \dfrac{f_1}{f_2}$ (legítimo
desde $ f_2$ nunca desaparece). Então
\[
h' = \frac{f_1' f_2 - f_1 f_2'}{f_2^2} = \frac{f_1 f_2 - f_1 f_2}{f_2^2} = 0,
\]
então $ h $ é constante igual a $ h(0) = 1$, \emph{ou seja} $ f_1 = f_2$.

\emph{Existência} é admitido neste nível (pode ser obtido através do
\omterm{def:g12:seq:sequence}{sequência} de \cref{exo:g12:seq:10}, ou como o inverso do logaritmo
construído pela integração em \cref{ch:g12:integ}).
\end{proof}

\begin{proposition}[Equação funcional]\label{prop:g12:exp:funceq}
Para todos $ x, y \in \R $ e $ n \in \Z $:
\[
\eu^{x+y} = \eu^x\,\eu^y, \qquad
\eu^{-x} = \frac{1}{\eu^x}, \qquad
\eu^{x-y} = \frac{\eu^x}{\eu^y}, \qquad
\bigl(\eu^{x}\bigr)^n = \eu^{nx}.
\]
Além disso $\eu^x > 0$ para todos $ x $.
\end{proposition}

\begin{proof}
Consertar $ y $ e considere $\varphi(x) = \dfrac{\exp(x+y)}{\exp(x)\exp(y)}$. É
numerador e denominador, como funções de $ x $, são ambas soluções de
$ f' = f $ até a constante $\exp(y)$; diferenciando $\varphi $ diretamente
(regra do quociente) dá $\varphi' = 0$, então
$\varphi \equiv \varphi(0) = 1$, provando a primeira identidade. Tirando
$ y = -x $ dá o segundo (com o valor $\eu^0 = 1$) e o terceiro
segue. A quarta é uma indução da primeira para $ n \geq 0$, estendido
para $ n < 0$ pelo segundo.

Positividade: $\eu^x = \left(\eu^{x/2}\right)^2 \geq 0$ e $\eu^x \neq 0$
(mostrado em \cref{thm:g12:exp:existence}), então $\eu^x > 0$.
\end{proof}

\begin{proposition}[Variações e limites]\label{prop:g12:exp:variations}
A exponencial é estritamente \omterm{def:g12:seq:monotonic}{aumentando}, \omterm{def:g12:deriv:convex}{convexo}, e
\[
\lim_{x\to-\infty} \eu^x = 0, \qquad
\lim_{x\to+\infty} \eu^x = +\infty .
\]
\end{proposition}

\begin{proof}
$(\exp)' = \exp > 0$ dá aumento estrito; $(\exp)'' = \exp > 0$ dá
convexidade. Por convexidade, $\eu^x \geq 1 + x $ (\omterm{def:g12:deriv:tangent}{tangente} no $0$, ver
\cref{ex:g12:deriv:ineq}), então $\eu^x \to +\infty $ como $ x \to +\infty $ por
comparação. Então $\eu^{x} = 1/\eu^{-x} \to 0$ como $ x \to -\infty $.
\end{proof}

\begin{theorem}[Comparação de crescimento]\label{thm:g12:exp:growth}
\index{comparação de crescimento}
Para cada inteiro $ n \geq 1$,
\[
\lim_{x\to+\infty} \frac{\eu^x}{x^n} = +\infty,
\qquad
\lim_{x\to-\infty} x^n \eu^x = 0 .
\]
Em palavras: o exponencial vence qualquer potência de $ x $.
\end{theorem}

\begin{proof}
Para $ n = 1$: aplicando $\eu^t \geq 1 + t \geq t $ no $ t = x/2$,
\[
\frac{\eu^x}{x} = \frac{\left(\eu^{x/2}\right)^2}{x}
\geq \frac{(x/2)^2}{x} = \frac{x}{4} \xrightarrow[x\to+\infty]{} +\infty .
\]
Para geral $ n $, escrever
\[
\frac{\eu^x}{x^n}
= \frac{1}{n^n}\left(\frac{\eu^{x/n}}{x/n}\right)^{n}
\xrightarrow[x\to+\infty]{} +\infty
\]
pelo caso $ n=1$ e composição. O limite em $-\infty $ segue pelo
substituição $ x \mapsto -x $:
$\abs{x^n \eu^x} = \frac{\abs{x}^n}{\eu^{\abs x}} \to 0$.
\end{proof}

\section{O logaritmo natural}

\begin{definition}[Logaritmo natural]\label{def:g12:exp:ln}
O exponencial é contínuo e estritamente \omterm{def:g12:seq:monotonic}{aumentando} de $\R $ para
$\intoo{0}{+\infty}$; pelo teorema da bijeção
(\cref{thm:g12:limcont:bijection}), para cada $ y > 0$ o equação
$\eu^x = y $ tem uma solução única. Esta solução é a \emph{naturais
logaritmo}\index{naturais
logaritmo} de $ y $, escrito $\ln y $. Por isso
\[
\text{for } x \in \R,\ y > 0: \qquad y = \eu^x \iff x = \ln y .
\]
Em particular $\ln 1 = 0$, $\ln \eu = 1$, e
$\eu^{\ln y} = y $, $\ln(\eu^x) = x $.
\end{definition}

\begin{proposition}[Propriedades algébricas]\label{prop:g12:exp:lnalg}
Para todos $ a, b > 0$ e $ n \in \Z $:
\[
\ln(ab) = \ln a + \ln b, \quad
\ln\frac{1}{a} = -\ln a, \quad
\ln\frac{a}{b} = \ln a - \ln b, \quad
\ln(a^n) = n \ln a, \quad
\ln\sqrt{a} = \tfrac12 \ln a .
\]
\end{proposition}

\begin{proof}
$\eu^{\ln a + \ln b} = \eu^{\ln a}\eu^{\ln b} = ab $, e tomando $\ln $ de ambos
lados dá a primeira identidade. Os demais seguem pelo mesmo mecanismo de
as identidades correspondentes de \cref{prop:g12:exp:funceq}.
\end{proof}

\begin{proposition}[Propriedades analíticas]\label{prop:g12:exp:lnanalytic}
O \omterm{def:g11:func:function}{função} $\ln $ é \omterm{def:g12:deriv:derivative}{diferenciável} sobre $\intoo{0}{+\infty}$ com
\[
(\ln x)' = \frac{1}{x},
\]
estritamente \omterm{def:g12:seq:monotonic}{aumentando}, \omterm{def:g12:deriv:convex}{côncavo}, e
$\lim\limits_{x\to0^+} \ln x = -\infty $,
$\lim\limits_{x\to+\infty} \ln x = +\infty $. Além disso, para cada inteiro
$ n \geq 1$,
\[
\lim_{x\to+\infty} \frac{\ln x}{x^{1/n}} = 0, \qquad
\lim_{x\to0^+} x \ln x = 0 .
\]
\end{proposition}

\begin{proof}
Diferenciabilidade do inverso \omterm{def:g11:func:function}{função} é admitido neste nível; concessão
isso, diferencie a identidade $\eu^{\ln x} = x $ pela regra da cadeia:
$(\ln x)'\,\eu^{\ln x} = 1$, então $(\ln x)' = \frac{1}{x} > 0$, de onde estrito
aumentar, e $(\ln x)'' = -\frac1{x^2} < 0$, de onde a concavidade. Os limites em
$0^+$ e $+\infty $ espelhar os de $\exp $ através da bijeção. Para o
\omterm{thm:g12:exp:growth}{comparação de crescimento}, substitua $ x = \eu^{t}$:
$\frac{\ln x}{x} = \frac{t}{\eu^t} \to 0$ como $ t \to +\infty $ por
\cref{thm:g12:exp:growth}, e da mesma forma para os outros limites com
$ x = \eu^{-t}$, dando $ x\ln x = -t\eu^{-t} \to 0$.
\end{proof}

\begin{method}[Resolvendo equações com $\exp $ e $\ln $]\label{met:g12:exp:equations}
Ambos funções são estritamente \omterm{def:g12:seq:monotonic}{aumentando}, para que possam ser aplicados (ou removidos)
de) ambos os lados de um equação ou desigualdade sem mudar sua direção:
\[
\eu^{u} = \eu^{v} \iff u = v, \qquad
\eu^{u} \leq \eu^{v} \iff u \leq v,
\]
e da mesma forma para $\ln $ em quantidades positivas. \emph{Sempre verifique domínios
primeiro} ($\ln $ requer argumentos positivos). Para equações como
$ a^x = b $ com $ a>0$, $ a \neq 1$, reescrever $ a^x = \eu^{x\ln a}$ e resolver
$ x = \frac{\ln b}{\ln a}$.
\end{method}

\begin{example}\label{ex:g12:exp:halflife}
Um radioativo \omterm{def:g10:proba:sample}{amostra} decai seguindo $ N(t) = N_0\,\eu^{-\lambda t}$. Isso é
\emph{meia-vida} $ T $ satisfaz $ N(T) = N_0/2$, \emph{ou seja}
$\eu^{-\lambda T} = \frac12$, então $ T = \frac{\ln 2}{\lambda}$: a meia-vida
não depende da quantidade inicial.
\end{example}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g12:exp:1}
Simplificar
$\dfrac{\eu^{3x}\,\eu^{-x+1}}{\eu^{x}}$ e
$\ln\!\left(\dfrac{\eu^2\sqrt{\eu}}{\eu^{-3}}\right)$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g12:exp:2}
Resolva em $\R $:
\[
\text{(a) } \eu^{2x} - 3\eu^{x} + 2 = 0;
\qquad
\text{(b) } \ln(x - 1) + \ln(x + 2) = \ln 4 .
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g12:exp:3}
Calcule os limites:
\[
\lim_{x\to+\infty} \frac{\eu^x - x^2}{\eu^x + 1}, \qquad
\lim_{x\to+\infty} \frac{\ln x}{\sqrt x}, \qquad
\lim_{x\to0^+} x^2 \ln x, \qquad
\lim_{x\to+\infty} \bigl(x - \ln x\bigr).
\]
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:exp:4}
Estude o \omterm{def:g11:func:function}{função} $ f(x) = x\,\eu^{-x}$ sobre $\R $: variações, limites,
extremo; Mostre que sua curva tem uma ponto de inflexão e dê o seu
\omterm{def:g10:coordgeom:system}{coordenadas}.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:exp:5}
Mostre isso para todos $ x > -1$, $\ln(1 + x) \leq x $, com igualdade apenas em
$ x = 0$. Deduza isso para todos $ n \geq 1$,
\[
\left(1 + \frac{1}{n}\right)^n \leq \eu \leq \left(1 - \frac{1}{n+1}\right)^{-(n+1)} .
\]
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:exp:6}
Uma capital $ C_0$ é investido a uma taxa anual de $3\%$, juros compostos
cada ano.
\begin{enumerate}
  \item Expresse a capital $ C_n $ depois $ n $ anos.
  \item Depois de quantos anos o capital dobra? Dê a resposta exata
        usando $\ln $, em seguida, um valor numérico.
  \item Compare com o \omterm{def:g10:numbers:approx}{aproximação} ``$70$ dividido pela taxa em
        por cento'' usado pelos banqueiros.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:exp:7}
Resolva a desigualdade $\eu^{2x} - \eu^{x+1} > 0$, então a desigualdade
$\ln(x^2 - 1) \leq \ln(x + 5)$.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g12:exp:8}
Deixar $ f(x) = \dfrac{\eu^x}{x}$ para $ x > 0$.
\begin{enumerate}
  \item Estude as variações de $ f $ sobre $\intoo{0}{+\infty}$ e dê o seu
        \omterm{def:g10:functions:extrema}{mínimo}.
  \item Para quais valores de $ k $ faz o equação $\eu^x = kx $ ter $0$, $1$
        ou $2$ soluções em $\intoo{0}{+\infty}$?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g12:exp:9}
Para $ n \geq 1$, deixar $ u_n = \left(1 + \frac1n\right)^n $.
\begin{enumerate}
  \item Usando \cref{exo:g12:exp:5}, mostre que $(u_n)$ é \omterm{def:g12:seq:bounded}{limitado acima}
        por ~$\eu $.
  \item Mostre isso $\ln u_n = n \ln\left(1 + \frac1n\right) \to 1$ e
        deduzir isso $ u_n \to \eu $.
        (Dica: reconheça um quociente de diferença de $\ln $ no $1$.)
\end{enumerate}
\end{exercise}

\section{Problema: O logaritmo doma o mundo}

\begin{problem}[{Problema de fim de semana --- tempos de duplicação e a regra
de 72, as escalas de terremotos, ácidos e pianos, e o
constante $\eu $ deixando impressões digitais por toda parte}]
\label{pb:g12:exp:1}
Tudo o que cresce por um período fixo \emph{percentagem} cresce
exponencialmente --- poupanças, bactérias, epidemias --- e o que quer que seja
abrange muitas potências de dez para compreender --- energias de terremotos,
acidez, intensidade sonora --- é domado por um logaritmo. Isto
problema calcula tempos de duplicação e desmascara a regra de
72, lê as escalas logarítmicas do mundo e coleta os
impressões digitais que o número $\eu $ sai em cada cena do crime
(\cref{prop:g12:exp:funceq}, \cref{thm:g12:exp:growth},
\cref{met:g12:exp:equations}).

\medskip
\noindent\textbf{Part I --- Fluency.}
\begin{enumerate}
  \item Resolver: $\eu^{2x} = 5$; \quad $\ln(3x - 1) = 2$; \quad
        $\eu^{2x} - 3\eu^x + 2 = 0$ (uma quadrática disfarçada).
  \item Simplificar: $\dfrac{\ln 8}{\ln 2}$; \quad
        $\ln\!\left(\eu^3 \sqrt{\eu}\right)$; \quad
        $\eu^{\ln 5 - \ln 2}$.
  \item Prove a desigualdade materna: $\eu^x \geq 1 + x $ para todos
        verdadeiro $ x $ (estudar $ f(x) = \eu^x - 1 - x $) e deduza
        $\ln(1 + u) \leq u $ para $ u > -1$.
  \item Calcular
        $\lim_{x \to +\infty} x^2 \eu^{-x}$,
        $\lim_{x \to +\infty} \frac{\ln x}{x}$
        (\cref{thm:g12:exp:growth}), e
        $\lim_{x \to 0} \frac{\eu^x - 1}{x}$ (reconhecer um
        derivado).
  \item Estudar $ f(x) = x \ln x $ sobre $\intoo{0}{+\infty}$:
        variações, \omterm{def:g10:functions:extrema}{mínimo}, e o limite em $0^+$ (admitido:
        $ x \ln x \to 0$). Este pequeno \omterm{def:g11:func:function}{função} medidas
        informação e entropia nos volumes universitários.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Doubling times and the rule of
72.}
\begin{enumerate}[resume]
  \item A poupança cresce em $3\,\%$ por ano. Resolver
        $1.03^n = 2$: quanto tempo para dobrar o capital?
  \item Os banqueiros estimam o tempo de duplicação como
        $\frac{72}{\text{rate in }\%}$. Teste a regra em
        $3\,\%$, $6\,\%$ e $9\,\%$ contra o exato
        $\frac{\ln 2}{\ln(1 + r)}$, então explique: para pequenos
        $ r $, $\ln(1 + r) \approx r $ (a desigualdade da questão 3 é
        metade da história), então a constante exata é
        $100 \ln 2 \approx 69.3$ --- por que os banqueiros preferem
        $72$?
  \item Uma bactéria se divide a cada $20$ minutos:
        $ p(t) = 2^{t/20}$ ($ t $ em minutos). Reescreva-o como
        $\eu^{\lambda t}$, então calcule $ p $ depois $24$ horas.
        A resposta (mais de $10^{21}$) prova o que acontece com o
        modelo --- e o que impede as colônias reais?
  \item A cafeína deixa o corpo com meia-vida de cerca de $5$
        horas. Depois de um $100$ mg de café às 15h, quanto
        permanece às 23:00? A que horas cai abaixo
        $10$ mg? (A insônia tem um logaritmo.)
  \item Um euro em $100\,\%$ juros anuais: calcule o
        capital de final de ano em termos anuais, mensais e diários
        composição, e dar o teto que
        \cref{exo:g12:exp:9} provou ser inquebrável. Qual famoso
        constante é o limite da pura ganância?
  \item Por que os cientistas conspiram $\ln(\text{cases})$ contra
        tempo durante a fase inicial de uma epidemia? O que um
        linha reta nesse \omterm{def:g10:functions:graph}{gráfico} revela, e o que seu
        declive medir?
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The world's logarithmic scales.}
(Escrever $\log_{10} x = \frac{\ln x}{\ln 10}$.)
\begin{enumerate}[resume]
  \item Nível de som em decibéis: $ L = 10 \log_{10}(I/I_0)$. Um
        medidas de conversação $60$ dB, um show de rock $120$ dB:
        por que fator as intensidades sonoras diferem?
  \item A magnitude do terremoto aumenta em $1$ quando a sísmica
        amplitude é multiplicada por $10$, e o liberado
        escalas de energia como amplitude $^{3/2}$. Comparar
        magnitude-$5$ e magnitude-$7$ terremotos: amplitude
        proporção e, em seguida, proporção de energia.
  \item Química: $\text{pH} = -\log_{10}[\mathrm{H^+}]$.
        Suco de limão tem pH $2$, pH do leite $6.5$: qual é o
        proporção de suas concentrações de ácido?
  \item Música: cada oitava duplica o freqüência. Do
        o lá mais grave do piano ($27.5$ Hz) até seu dó mais alto
        ($4\,186$ Hz), quantas oitavas o teclado abrange?
        E por que nossos sentidos --- audição, visão, sensação de terremoto
        --- prefere escalas logarítmicas? (Uma frase.)
  \item A régua de cálculo, a calculadora dos engenheiros para $350$
        anos: duas varetas se formaram para que o \emph{comprimento}
        para a marca $ x $ é proporcional a $\ln x $. Explique como
        deslizar uma vareta ao longo da outra multiplica os números,
        e nomeie a identidade de
        \cref{prop:g12:exp:lnalg} fazendo o trabalho.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- $\eu $'s fingerprints.}
\begin{enumerate}[resume]
  \item De \cref{exo:g12:exp:8}: o equação
        $\eu^x = kx $ ($ x > 0$) tem $0$, $1$ ou $2$ soluções
        de acordo com a posição $ k $ relativo a um
        limite. Reafirme o resultado --- e verifique o
        fato \omterm{def:g12:seq:arith-geom}{geométrico} por trás disso: a linha $ y = \eu x $ é
        exatamente o \omterm{def:g12:deriv:tangent}{tangente} para o exponencial através do
        origem.
  \item A constante de quase acidente: com $ n $ bilhetes de loteria de
        vencendo \omterm{def:g10:proba:distribution}{probabilidade} $\frac1n $ cada,
        $\P(\text{no win}) = \left(1 - \frac1n\right)^n $.
        Calcule seu limite via
        $ n \ln\!\left(1 - \frac1n\right)$ (use o padrão
        limite $\frac{\ln(1 + u)}{u} \to 1$) e reconciliar-se com
        o misterioso $0.368$ de \cref{pb:g11:binom:1}.
  \item O $37\,\%$ regra: escolher o melhor $ n $
        candidatos entrevistados em ordem aleatória (sem retorno),
        a estratégia ótima rejeita a primeira
        $\frac{n}{\eu} \approx 37\,\%$ e então pega o primeiro
        candidato melhor do que todos até agora --- tendo sucesso com
        \omterm{def:g10:proba:distribution}{probabilidade} sobre $\frac1\eu $. Onde é que os dois
        $\frac1\eu $ As perguntas 18 e 19 vêm de ---
        indicar o mecanismo comum (muitos mecanismos independentes
