\chapter{Equações Diferenciais}\label{ch:g12:diffeq}

UM equação diferencial relaciona um \omterm{def:g11:func:function}{função} para o seu derivados. Física,
química, biologia e economia expressam suas leis desta forma: a taxa de
resfriamento de um corpo, o decaimento de um núcleo radioativo, o crescimento de um
população são todas declarações sobre $y'$. Este capítulo resolve completamente o
linear de primeira ordem equações com coeficientes constantes.

\section{A equação \texorpdfstring{$ y' = ay $}{y' = ay}}

\begin{definition}[Equação diferencial]\label{def:g12:diffeq:ode}
UM \emph{solução diferencial}\index{solução diferencial} é um equação
cujo desconhecido é um \omterm{def:g11:func:function}{função} $ y $, envolvendo $ y $ e seu derivados. UM
\emph{solução} em um \omterm{def:g10:numbers:interval}{intervalo} $ I $ é um \omterm{def:g12:deriv:derivative}{diferenciável} \omterm{def:g11:func:function}{função} satisfatório
o equação em cada ponto de $ I $.
\end{definition}

\begin{theorem}[Soluções de $ y' = ay $]\label{thm:g12:diffeq:homogeneous}
Deixar $ a \in \R $. As soluções em $\R $ do equação $ y' = ay $ são exatamente
o funções
\[
y(x) = C\,\eu^{ax}, \qquad C \in \R .
\]
Para cada par $(x_0, y_0)$ existe uma solução única com
$ y(x_0) = y_0$.
\end{theorem}

\begin{proof}
Cada $ y = C\eu^{ax}$ satisfaz $ y' = Ca\,\eu^{ax} = ay $. Por outro lado, deixe $ y $
seja qualquer solução e defina $ z(x) = y(x)\,\eu^{-ax}$. Então
\[
z'(x) = y'(x)\,\eu^{-ax} - a\,y(x)\,\eu^{-ax}
= \bigl(y'(x) - a y(x)\bigr)\eu^{-ax} = 0,
\]
então $ z $ é constante, digamos $ C $, e $ y(x) = C\eu^{ax}$. A condição inicial
$ y(x_0) = y_0$ forças $ C = y_0 \eu^{-a x_0}$, exclusivamente.
\end{proof}

\begin{remark}
A exponencial é assim \emph{caracterizado} pelo mais simples de todos
equações diferenciais: crescimento proporcional ao tamanho. É por isso que parece
em todos os lugares da natureza.
\end{remark}

\begin{example}[Decaimento radioativo]\label{ex:g12:diffeq:decay}
Uma quantidade radioativa satisfaz $ N' = -\lambda N $ com $\lambda > 0$, então
$ N(t) = N_0\,\eu^{-\lambda t}$; ver \cref{ex:g12:exp:halflife} para o
meia-vida.
\end{example}

\section{A equação \texorpdfstring{$ y' = ay + b $}{y' = ay + b}}

\begin{theorem}[Soluções de $ y' = ay + b $]\label{thm:g12:diffeq:affine}
Deixar $ a \neq 0$ e $ b \in \R $. As soluções em $\R $ de $ y' = ay + b $ são
exatamente o funções
\[
y(x) = C\,\eu^{ax} - \frac{b}{a}, \qquad C \in \R .
\]
A constante \omterm{def:g11:func:function}{função} $ y_p = -\frac ba $ é o \emph{solução de equilíbrio}.
Para cada $(x_0, y_0)$ existe uma solução única com $ y(x_0) = y_0$; se
$ a < 0$, toda solução tende ao equilíbrio $-\frac ba $ como
$ x \to +\infty $.
\end{theorem}

\begin{proof}
A constante $ y_p = -\frac{b}{a}$ satisfaz $ y_p' = 0 = a y_p + b $. Agora $ y $
é uma solução se e somente se
\[
(y - y_p)' = y' = ay + b = a(y - y_p) + \underbrace{a y_p + b}_{=\,0}
= a (y - y_p),
\]
isto é, se e somente se $ z = y - y_p $ resolve $ z' = az $. Por
\cref{thm:g12:diffeq:homogeneous}, $ z = C\eu^{ax}$, de onde a fórmula, o
existência e a singularidade. Se $ a < 0$, $\eu^{ax} \to 0$ como
$ x \to +\infty $, então $ y(x) \to -\frac ba $.
\end{proof}

\begin{method}[Resolvendo $ y' = ay + b $ com uma condição inicial]\label{met:g12:diffeq:solve}
\begin{enumerate}
  \item Encontre a solução de equilíbrio $ y_p = -\frac{b}{a}$ (resolver $ y' = 0$).
  \item Escreva a solução geral $ y = C\eu^{ax} + y_p $.
  \item Determinar $ C $ da condição inicial.
  \item Verifique o comportamento a longo prazo contra a intuição física (o
        solução converge para o equilíbrio?).
\end{enumerate}
A mesma estratégia --- \emph{solução particular + solução geral do
homogêneo equação} --- estende-se a $ y' = ay + f(x)$: ver
\cref{prop:g12:diffeq:general}.
\end{method}

\begin{example}[Lei do resfriamento de Newton]\label{ex:g12:diffeq:cooling}
Uma xícara de café em $80\,^\circ $ C está sentado em uma sala às $20\,^\circ $ C. A lei de Newton afirma
que a temperatura $ T $ satisfaz $ T' = -k(T - 20)$ para alguns $ k > 0$,
\emph{ou seja} $ T' = -kT + 20k $. O equilíbrio é $20$, e
$ T(t) = 20 + 60\,\eu^{-kt}$: o café esfria exponencialmente rápido para o ambiente
temperatura.
\end{example}

\section{A equação \texorpdfstring{$ y' = ay + f(x)$}{y' = ay + f(x)}}

\begin{proposition}[Estrutura do conjunto de soluções]\label{prop:g12:diffeq:general}
Deixar $ f $ ser contínuo em um \omterm{def:g10:numbers:interval}{intervalo} $ I $ e deixe $ y_p $ seja um em particular
solução de
\[
y' = ay + f(x)
\]
sobre $ I $. Então as soluções em $ I $ são exatamente os funções
$ y = C\eu^{ax} + y_p $, $ C \in \R $.
\end{proposition}

\begin{proof}
Como em \cref{thm:g12:diffeq:affine}: $ y $ é uma solução se e somente se
$ z = y - y_p $ satisfaz
$ z' = (ay + f) - (ay_p + f) = az $, se e somente se $ z = C\eu^{ax}$.
\end{proof}

\begin{method}[Adivinhando uma solução específica]\label{met:g12:diffeq:particular}
Procure uma solução específica \emph{da mesma forma} como $ f $:
\begin{itemize}
  \item $ f $ polinômio de grau $ n $: tente um polinômio de grau $ n $;
  \item $ f(x) = \alpha\,\eu^{kx}$ com $ k \neq a $: tentar $ y_p = \beta\,\eu^{kx}$;
  \item $ f(x) = \alpha\,\eu^{ax}$ (caso ressonante): tente
        $ y_p = \beta x\,\eu^{ax}$.
\end{itemize}
Substitua no equação e identifique os coeficientes.
\end{method}

\begin{example}\label{ex:g12:diffeq:particular}
Resolver $ y' = 2y + 4x $. Tentar $ y_p = \alpha x + \beta $:
$\alpha = 2(\alpha x + \beta) + 4x $ para todos $ x $ forças
$2\alpha + 4 = 0$ e $\alpha = 2\beta $, então $\alpha = -2$, $\beta = -1$.
Solução geral: $ y = C\eu^{2x} - 2x - 1$.
\end{example}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g12:diffeq:1}
Resolver $\R $:
(a) $ y' = 3y $ com $ y(0) = 2$;\quad
(b) $2y' + y = 0$ com $ y(0) = -1$;\quad
(c) $ y' = -y + 5$ com $ y(0) = 0$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g12:diffeq:2}
Uma população bacteriana cresce a uma taxa proporcional ao seu tamanho, duplicando
cada $3$ horas. Escreva o equação diferencial satisfeito pela população
$ N(t)$ e determine a constante de proporcionalidade.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g12:diffeq:3}
Verifique isso $ y_p(x) = x\,\eu^{x}$ é uma solução de $ y' = y + \eu^x $ e
dê todas as soluções em $\R $.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:diffeq:4}
Resolver $ y' = -2y + \eu^{x}$ com $ y(0) = 1$.
(Dica: procure uma solução particular da forma $\beta\,\eu^{x}$.)
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:diffeq:5}
O carbono-14 decai com meia-vida de $5730$ anos. Um arqueológico \omterm{def:g10:proba:sample}{amostra}
contém $60\%$ do carbono-14 de um organismo vivo. Estime sua idade.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:diffeq:6}
Um tanque contém $100$ L de água pura. Salmoura contendo $0.2$ kg de sal por
litro entra em $5$ L/min, a mistura (mantida uniforme) flui na mesma
taxa. Deixar $ m(t)$ seja a massa de sal no tanque no momento $ t $ (em minutos).
\begin{enumerate}
  \item Justifique isso $ m' = 1 - \dfrac{m}{20}$.
  \item Resolva e determine o limite de $ m(t)$ como $ t \to +\infty $.
        Interpretar.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:diffeq:7}
Um paraquedista de massa $80$ kg cai sujeito à gravidade
($ g = 9.8\ \text{m/s}^2$) e a resistência do ar proporcional à velocidade, de modo que
sua velocidade satisfaz $ v' = g - \frac{k}{m}v $ com $ k = 16$ kg/s.
\begin{enumerate}
  \item Resolva o equação com $ v(0) = 0$.
  \item Calcule o \emph{velocidade terminal} $\lim_{t\to+\infty} v(t)$ e
        o tempo necessário para chegar $95\%$ disso.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g12:diffeq:8}
\emph{(Logística equação.)} Uma população $ y(t) \in \intoo{0}{1}$ (como um
fração da população máxima) satisfaz
\[
y' = y(1 - y).
\]
\begin{enumerate}
  \item Deixar $ z = \dfrac{1}{y}$. Mostre isso $ z $ satisfaz o linear equação
        $ z' = -z + 1$.
  \item Resolva para $ z $, então para $ y $, com $ y(0) = \frac{1}{10}$.
  \item Mostre isso $ y(t) \to 1$ como $ t \to +\infty $ e esboce a forma de
        a curva da solução.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g12:diffeq:9}
Deixar $ y $ ser uma solução de $ y' = ay + b $ e deixe $ u_n = y(n)$ para $ n \in \N $.
Mostre isso $(u_n)$ satisfaz uma recorrência aritmético-geométrica
$ u_{n+1} = q u_n + r $ e expressar $ q $ e $ r $ em termos de $ a $ e $ b $. O que
a condição $\abs{q} < 1$ corresponde para o diferencial
equação?
\end{exercise}

\section{Problema: o relógio dentro das coisas}

\begin{problem}[{Problema de fim de semana --- carbono-14 data as cavernas,
A lei de resfriamento de Newton vezes um crime, e uma pequena equação se desgasta
quatro fantasias}]
\label{pb:g12:diffeq:1}
UM equação diferencial é uma lei de mudança; resolvê-lo transforma o
lei em um \emph{relógio}. O mesmo minúsculo equação
$ y' = ay + b $ (\cref{thm:g12:diffeq:affine}) carrapatos dentro
carvão pré-histórico, café refrescante, gotas de chuva caindo e
gotejamentos hospitalares --- e ler esses relógios é o problema
negócio: data as pinturas de Lascaux, fixa a hora da morte,
e audita suas próprias suposições como um bom cientista.

\medskip
\noindent\textbf{Part I --- Fluency.}
\begin{enumerate}
  \item Resolver $ y' = 3y $, $ y(0) = 2$; então
        $ y' = -2y + 6$, $ y(0) = 0$
        (\cref{met:g12:diffeq:solve}).
  \item Derive novamente o truque da exclusividade por trás
        \cref{thm:g12:diffeq:homogeneous}: se $ y' = ay $, calcular
        o derivado de $ y(t)\,\eu^{-at}$ e concluir que
        \emph{todo} solução é $ C\eu^{at}$.
  \item Para $ y' = -2y + 6$: encontrar a solução de equilíbrio, e
        descrever o destino de todas as outras soluções como
        $ t \to +\infty $. (O pontos fixos de
        \cref{pb:g11:seq:1}as recorrências, desapareceram contínuo.)
  \item Para a lei de decadência $ y' = -\frac{y}{\tau}$: mostre que o
        meia-vida é $ t_{1/2} = \tau \ln 2$.
  \item Esboce (ou descreva) a família de soluções de
        $ y' = -2y + 6$: o que as soluções começam acima $3$
        fazer? Abaixo $3$? Sobre $3$?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Carbon-14.}
O tecido vivo mantém uma proporção constante de radioatividade
carbono-14; na morte, a ingestão para e o estoque decai:
$ N' = -\lambda N $, com meia-vida $5\,730$ anos.
\begin{enumerate}[resume]
  \item Calcular $\lambda $ (por ano).
  \item Um osso retém $20\,\%$ da proporção de vida: quantos anos tem
        isso?
  \item O carvão das cavernas pintadas de Lascaux conserva
        sobre $15\,\%$: datar as pinturas.
  \item O Homem de Gelo Ötzi, encontrado em uma geleira alpina, medido
        sobre $53\,\%$: namorar com ele (os arqueólogos dizem sobre
        $5\,300$ anos --- como você se saiu?).
  \item Por que o carbono-14 não pode datar os dinossauros? Calcule o
        fração restante após dez meias-vidas, indique a
        fração depois $65$ milhões de anos como uma potência de $2$,
        e concluir.
  \item Barras de erro: se Ötzi's $53\,\%$ só é conhecido por
        $\pm 1\,\%$, calcule a \omterm{def:g10:stats:quartiles}{faixa} etária. Que precisão de laboratório
        compra que precisão de namoro?
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Newton's cooling, and a crime.}
Um corpo à temperatura $ T $ no ambiente em constante $ T_a $
esfria por $ T' = -k\,(T - T_a)$.
\begin{enumerate}[resume]
  \item Resolva o equação (substituir $ z = T - T_a $):
        \[
        T(t) = T_a + (T_0 - T_a)\,\eu^{-kt} .
        \]
  \item Café servido em $90\,^\circ $ C em um $20\,^\circ $ Sala C
        lê $70\,^\circ $ C depois $5$ minutos. Determinar $ k $,
        então o tempo de espera até o potável
        $55\,^\circ $ C.
  \item A questão do leite: beber o café mais quente possível
        em dez minutos, o leite frio deve entrar agora ou às
        o último momento? Responda com o equação (o que faz
        adicionando leite na lacuna $ T - T_a $, e como é que
        lacuna leva à perda?).
  \item Perícia: corpo é encontrado à meia-noite
        $30\,^\circ $ C em um $20\,^\circ $ Sala C; uma hora depois
        lê $28\,^\circ $ C. Supondo $37\,^\circ $ C no
        hora da morte, recuperar $ k $ das duas medições,
        então calcule a hora da morte.
  \item Audite o relógio do legista: cite três do mundo real
        violações das suposições do modelo e da direção
        em que cada um distorceria a hora estimada da morte.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Four costumes and an S-curve.}
\begin{enumerate}[resume]
  \item A logística equação de \cref{exo:g12:diffeq:8},
        $ y' = y(1 - y)$ com $ y(0) = 0.1$: usando o exercício
        substituição, derivar
        $ y(t) = \dfrac{1}{1 + 9\eu^{-t}}$, e encontre o horário de
        o ponto de inflexão $ y = \frac12$ --- o
        data do epidemiologista de \cref{pb:g12:deriv:1}, agora
        computável.
  \item Uma gota de chuva obedece $ v' = 10 - \frac v2$ (gravidade menos
        arrastar), $ v(0) = 0$. Encontre a velocidade terminal, resolva
        para $ v(t)$, e calcule quando a queda atinge
        $95\,\%$ da velocidade terminal.
  \item Um gotejamento hospitalar administra um medicamento a uma taxa constante enquanto
        o corpo o elimina proporcionalmente:
        $ c' = 4 - 0.5c $, $ c(0) = 0$. Encontre o estado estacionário
        concentração e o tempo para chegar à metade dela. Estado
        em uma frase o paralelo com as recorrências do empréstimo
        de \cref{exo:g12:diffeq:9}.
  \item Final --- um equação, quatro fantasias: decadência, resfriamento,
        caindo, dosando, tudo $ y' = ay + b $ com sinais diferentes
        e nomes. Recite o loop de modelagem que este problema executou
        quatro vezes (lei $\to $ equação $\to $ solução $\to $
        calibração $\to $ previsão $\to $ auditoria) e nome
        o fenômeno que precisa $ y''$ --- e qual problema
        já conheci.
\end{enumerate}
\end{problem}
