\chapter{Matrizes e Gráficos}\label{ch:g12:matrix}

UM matriz é uma matriz retangular de números, somados e multiplicados por regras
projetado para que matriz álgebra representa a composição de linear
transformações. Matrizes resolvem sistemas lineares, drive acoplado recorrente
sequênciase contar caminhadas nas redes --- a matemática por trás da \omterm{def:g10:algebra:equation}{pesquisa}
motores e algoritmos de caminho mais curto.

\section{Álgebra matricial}

\begin{definition}[Matriz]\label{def:g12:matrix:matrix}
Um \emph{$m \times n $ matriz}\index{$ m \times n $ matriz} é uma tabela de \omterm{def:g10:numbers:sets}{números reais} com
$ m $ linhas e $ n $ colunas:
$ A = (a_{ij})$, onde $ a_{ij}$ é a entrada na linha $ i $, coluna $ j $.
Duas matrizes do mesmo tamanho são somadas entrada por entrada, e
$\lambda A = (\lambda a_{ij})$.
\end{definition}

\begin{definition}[Produto matricial]\label{def:g12:matrix:product}
Deixar $ A $ ser $ m \times n $ e $ B $ ser $ n \times p $. O produto $ AB $ é o
$ m \times p $ matriz cujo $(i,j)$ a entrada é
\[
(AB)_{ij} = \sum_{k=1}^{n} a_{ik} b_{kj}
\]
(a ``linha $ i $ de $ A $ coluna vezes $ j $ de $ B $'' regra).
\end{definition}

\begin{example}
$\begin{pmatrix} 1 & 2\\ 3 & 4\end{pmatrix}
\begin{pmatrix} 0 & 1\\ 1 & 1\end{pmatrix}
= \begin{pmatrix} 2 & 3\\ 4 & 7\end{pmatrix}$,
\quad enquanto \quad
$\begin{pmatrix} 0 & 1\\ 1 & 1\end{pmatrix}
\begin{pmatrix} 1 & 2\\ 3 & 4\end{pmatrix}
= \begin{pmatrix} 3 & 4\\ 4 & 6\end{pmatrix}$:
\emph{matriz multiplicação não é comutativa}.
\end{example}

\begin{proposition}[Regras da álgebra matricial]\label{prop:g12:matrix:rules}
Sempre que os tamanhos tornam os produtos \omterm{def:g11:stat:mean}{significativos}:
\[
(AB)C = A(BC), \qquad A(B + C) = AB + AC, \qquad (A+B)C = AC + BC,
\]
e o \emph{identidade matriz} $ I_n $ (uns na diagonal, zeros em outros lugares)
satisfaz $ I_m A = A I_n = A $ para $ A $ de tamanho $ m \times n $.
\end{proposition}

\begin{proof}
Todas são verificações entrada por entrada de
\cref{def:g12:matrix:product}; associatividade, a única não trivial,
equivale a trocar duas somas finitas:
$\bigl((AB)C\bigr)_{ij} = \sum_l \left(\sum_k a_{ik}b_{kl}\right) c_{lj}
= \sum_k a_{ik} \left(\sum_l b_{kl} c_{lj}\right)
= \bigl(A(BC)\bigr)_{ij}$.
\end{proof}

\begin{definition}[Inverso]\label{def:g12:matrix:inverse}
Um quadrado matriz $ A $ de tamanho $ n $ é \emph{invertível}\index{invertível}
se houver um matriz $ B $ com $ AB = BA = I_n $; $ B $ é então único, escrito
$ A^{-1}$.
\end{definition}

\begin{proposition}[Inverso de um $2\times2$ matriz]\label{prop:g12:matrix:inverse2x2}
Deixar $ A = \begin{pmatrix} a & b\\ c & d\end{pmatrix}$ e
$\det A = ad - bc $ (o \emph{determinante}\index{determinante}). Então $ A $ é
\omterm{def:g12:matrix:inverse}{invertível} se e somente se $\det A \neq 0$, nesse caso
\[
A^{-1} = \frac{1}{ad - bc}\begin{pmatrix} d & -b\\ -c & a\end{pmatrix}.
\]
\end{proposition}

\begin{proof}
Um cálculo dá
$ A \begin{pmatrix} d & -b\\ -c & a\end{pmatrix}
= \begin{pmatrix} d & -b\\ -c & a\end{pmatrix} A = (ad - bc) I_2$; se
$ ad - bc \neq 0$, dividir. Por outro lado, se $ ad - bc = 0$, as colunas de $ A $
são proporcionais, assim como as colunas de $ AB $ para qualquer $ B $; mas o
colunas de $ I_2$ não são proporcionais, então não $ B $ pode satisfazer $ AB = I_2$.
\end{proof}

\begin{method}[Sistemas lineares]\label{met:g12:matrix:systems}
O sistema
$\begin{cases} ax + by = e\\ cx + dy = f \end{cases}$
é o matriz equação $ AX = Y $ com
$ X = \begin{pmatrix} x \\ y\end{pmatrix}$,
$ Y = \begin{pmatrix} e \\ f\end{pmatrix}$. Se $\det A \neq 0$, é único
solução é $ X = A^{-1}Y $. O mesmo formalismo trata $ n $ equações em $ n $
desconhecidos.
\end{method}

\section{Potências matriciais e sequências recorrentes}

\begin{definition}\label{def:g12:matrix:power}
Para um quadrado matriz $ A $ e $ k \in \N $, $ A^k = A \times \dots \times A $
($ k $ fatores), com $ A^0 = I $.
\end{definition}

\begin{method}[Casos diagonal mais nilpotentes e diagonalizáveis]\label{met:g12:matrix:powers}
Duas maneiras padrão de calcular $ A^k $:
\begin{itemize}
  \item Se $ A = \lambda I + N $ onde $ N^2 = 0$, o teorema binomial
        (válido aqui desde $ I $ e $ N $ deslocamento diário) se resume a dois termos:
        $ A^k = \lambda^k I + k \lambda^{k-1} N $.
  \item Se alguém encontrar um \omterm{def:g12:matrix:inverse}{invertível} $ P $ com $ A = PDP^{-1}$ e $ D $ diagonal,
        então $ A^k = P D^k P^{-1}$, e $ D^k $ é computado entrada por entrada.
        (Encontrar tal $ P $ sistematicamente é a teoria da
        \emph{diagonalização}, desenvolvido na universidade; neste nível $ P $ é
        dado.)
\end{itemize}
\end{method}

\begin{example}[Sequências acopladas]\label{ex:g12:matrix:coupled}
Deixar $ u_{n+1} = 3u_n + v_n $ e $ v_{n+1} = u_n + 3v_n $. Contexto
$ X_n = \begin{pmatrix} u_n\\ v_n \end{pmatrix}$ e
$ A = \begin{pmatrix} 3 & 1\\ 1 & 3\end{pmatrix}$, obtemos
$ X_{n+1} = AX_n $, então $ X_n = A^n X_0$. O auxiliar \omterm{def:g12:seq:sequence}{sequências}
$ s_n = u_n + v_n $ e $ d_n = u_n - v_n $ satisfazer $ s_{n+1} = 4s_n $ e
$ d_{n+1} = 2d_n $, então $ s_n = 4^n s_0$, $ d_n = 2^n d_0$ e
\[
u_n = \frac{4^n(u_0+v_0) + 2^n(u_0-v_0)}{2}, \qquad
v_n = \frac{4^n(u_0+v_0) - 2^n(u_0-v_0)}{2}.
\]
(Nos bastidores: $(1,1)$ e $(1,-1)$ são direções autovetoriais de $ A $.)
\end{example}

\section{Gráficos e caminhadas}

\begin{definition}[Gráfico, matriz de adjacência]\label{def:g12:matrix:graph}
UM \emph{gráfico}\index{gráfico} consiste em vértices $1, 2, \dots, n $ e bordas
juntando certos pares de vértices (pares ordenados para um \emph{dirigido}
gráfico). Isso é \emph{matriz de adjacência}\index{matriz de adjacência} é o
$ n \times n $ matriz $ M $ com $ m_{ij} = 1$ se houver uma borda de $ i $ para
$ j $, e $0$ de outra forma. UM \emph{andar} de comprimento $ k $ de $ i $ para $ j $ é um
\omterm{def:g12:seq:sequence}{sequência} de $ k $ arestas consecutivas partindo de $ i $ para $ j $.
\end{definition}

  \node[circle, draw, inner sep=2.5pt] (v2) at (2.6,1.2) {$2$};
  \node[circle, draw, inner sep=2.5pt] (v3) at (2.6,-1.2) {$3$};
  \draw[-Stealth, omDef, thick] (v1) -- (v2);

  \draw[-Stealth, omDef, thick] (v3) to[bend right=18] (v1);
\end{tikzpicture}
\qquad\qquad
$ M = \begin{pmatrix} 0 & 1 & 1\\ 0 & 0 & 1\\ 1 & 0 & 0 \end{pmatrix}$

{\small Um dirigido \omterm{def:g12:matrix:graph}{gráfico} e seu \omterm{def:g12:matrix:graph}{matriz de adjacência}
(\cref{exo:g12:matrix:6}): $ m_{ij} = 1$ exatamente quando há uma vantagem
de $ i $ para $ j $.}
\end{omfigure}

\begin{theorem}[Contando caminhadas]\label{thm:g12:matrix:walks}
O número de caminhadas de comprimento $ k $ do vértice $ i $ para o vértice $ j $ é o
$(i,j)$ entrada de $ M^k $.
\end{theorem}

\begin{proof}
Indução ativada $ k $. Para $ k = 1$ esta é a definição de $ M $. Suponha que
reivindicação por $ k $. Uma caminhada de comprimento $ k+1$ de $ i $ para $ j $ é uma caminhada de comprimento
$ k $ de $ i $ para algum vértice $ l $, seguido por uma aresta de $ l $ para $ j $; pelo
princípios de adição e multiplicação, seu número é
\[
\sum_{l=1}^{n} \bigl(M^k\bigr)_{il}\, m_{lj} = \bigl(M^{k+1}\bigr)_{ij}.
\qedhere
\]
\end{proof}

\begin{example}
Para o triângulo \omterm{def:g12:matrix:graph}{gráfico} ($3$ vértices, todos os pares unidos),
$ M = \begin{pmatrix} 0&1&1\\ 1&0&1\\ 1&1&0\end{pmatrix}$ e
$ M^2 = \begin{pmatrix} 2&1&1\\ 1&2&1\\ 1&1&2\end{pmatrix}$: de cada vértice
existem $2$ caminhadas de comprimento $2$ de volta para si mesmo (através de qualquer vizinho) e
$1$ um ao outro vértice.
\end{example}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g12:matrix:1}
Deixar $ A = \begin{pmatrix} 1 & 2\\ 0 & 1 \end{pmatrix}$ e
$ B = \begin{pmatrix} 2 & 0\\ 1 & 1 \end{pmatrix}$. Calcular $ A + B $, $ AB $,
$ BA $ e $ A^2$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g12:matrix:2}
Determine se as seguintes matrizes são \omterm{def:g12:matrix:inverse}{invertível}, e calcule o
inversos quando existem:
\[
A = \begin{pmatrix} 2 & 5\\ 1 & 3\end{pmatrix}, \qquad
B = \begin{pmatrix} 3 & 6\\ 2 & 4\end{pmatrix}.
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g12:matrix:3}
Resolver por matriz inverter o sistema
$\begin{cases} 2x + 5y = 1\\ x + 3y = 2 . \end{cases}$
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:matrix:4}
Deixar $ A = \begin{pmatrix} 2 & 1\\ 0 & 2\end{pmatrix} = 2I + N $ com
$ N = \begin{pmatrix} 0 & 1\\ 0 & 0 \end{pmatrix}$.
\begin{enumerate}
  \item Verifique isso $ N^2 = 0$ e isso $ I $ e $ N $ trajeto.
  \item Deduzir $ A^k $ para todos $ k \in \N $ e verifique a fórmula para $ k=2$ por
        cálculo direto.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:matrix:5}
Deixar $ A = \begin{pmatrix} 0 & 1\\ 1 & 1\end{pmatrix}$ e $ F_n $ o Fibonacci
\omterm{def:g12:seq:sequence}{sequência} ($ F_0 = 0$, $ F_1 = 1$, $ F_{n+2} = F_{n+1} + F_n $). Mostrar por
indução que para $ n \geq 1$,
\[
A^n = \begin{pmatrix} F_{n-1} & F_n\\ F_n & F_{n+1}\end{pmatrix},
\]
e deduza a identidade $ F_{n+1}F_{n-1} - F_n^2 = (-1)^n $.
(Dica: \omterm{def:g11:vect:det}{determinantes} multiplicar: $\det(MN) = \det M \det N $, que você pode
verifique $2\times2$ matrizes.)
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:matrix:6}
Um dirigido \omterm{def:g12:matrix:graph}{gráfico} em vértices $\{1, 2, 3\}$ tem bordas
$1\to2$, $2\to3$, $3\to1$ e $1\to3$.
\begin{enumerate}
  \item Escreva o \omterm{def:g12:matrix:graph}{matriz de adjacência} $ M $ e calcular $ M^2$ e $ M^3$.
  \item Quantos passeios de comprimento $3$ ir de $1$ para $1$? Liste-os.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:matrix:7}
Uma empresa de compartilhamento de carros transporta veículos entre duas cidades $ A $ e $ B $. Cada
semana, $80\%$ dos carros em $ A $ fique em $ A $ e $20\%$ mudar para $ B $; $30\%$
dos carros em $ B $ mudar para $ A $ e $70\%$ ficar. Deixar $ a_n, b_n $ seja o
proporções da frota em cada cidade.
\begin{omfigure}

  \node[circle, draw, inner sep=3pt] (B) at (3.2,0) {$ B $};

\begin{enumerate}
  \item Escrever $ X_{n+1} = MX_n $ com
        $ X_n = \begin{pmatrix} a_n\\ b_n\end{pmatrix}$ e identificar $ M $.
  \item Encontre as proporções de equilíbrio (resolva $ MX = X $ com
        $ a + b = 1$).
  \item Mostre isso $ c_n = a_n - 0.6$ satisfaz $ c_{n+1} = 0.5\,c_n $ e
        concluir que a frota \omterm{def:g11:prob:rv}{distribuição} converge ao equilíbrio.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g12:matrix:8}
Deixar $ A = \begin{pmatrix} 3 & 1\\ 1 & 3 \end{pmatrix}$,
$ P = \begin{pmatrix} 1 & 1\\ 1 & -1 \end{pmatrix}$.
\begin{enumerate}
  \item Calcular $ P^{-1}$, então $ D = P^{-1}AP $ e verifique isso $ D $ é
        diagonal.
  \item Deduza uma fórmula fechada para $ A^n $ e compare com
        \cref{ex:g12:matrix:coupled}.
\end{enumerate}
\end{exercise}

\section{Problema: A matriz que conhece Fibonacci (e o
clima)}

\begin{problem}[{Problema de fim de semana --- um $2 \times 2$ matriz
carrega todo Fibonacci, uma matriz de Markov prevê o longo prazo
clima, e um vetor próprio vale um bilhão de dólares}]
\label{pb:g12:matrix:1}
UM matriz é uma máquina que come um estado e retorna o próximo
--- e seu \emph{poderes} portanto, detêm futuros \omterm{def:g10:numbers:sets}{inteiros}. Isto
problema abre com o surpreendente matriz cujos poderes listam os
Números de Fibonacci (e provar suas identidades em uma linha cada),
em seguida, executa o clima como uma cadeia de Markov até seu estado estacionário, e
fecha com o autovetor no qual um mecanismo de busca foi construído
(\cref{thm:g12:matrix:walks},
\cref{met:g12:matrix:powers}).

\medskip
\noindent\textbf{Part I --- Fluency.}
\begin{enumerate}
  \item Com $ A = \begin{pmatrix} 1 & 2\\ 3 & 4\end{pmatrix}$
        e $ B = \begin{pmatrix} 0 & 1\\ 1 & 0\end{pmatrix}$:
        calcular $ AB $ e $ BA $. Veredicto sobre comutatividade?
  \item Invertido $\begin{pmatrix} 2 & 1\\ 5 & 3\end{pmatrix}$
        (\cref{prop:g12:matrix:inverse2x2}) e use o inverso
        resolver $2x + y = 4$, $5x + 3y = 7$.
  \item Deixar $ N = \begin{pmatrix} 0 & 1\\ 0 & 0\end{pmatrix}$:
        calcular $ N^2$, e deduza
        $(I + N)^n = I + nN $ para cada $ n $.
  \item O triângulo \omterm{def:g12:matrix:graph}{gráfico} (três vértices, todos os pares unidos):
        escreva seu \omterm{def:g12:matrix:graph}{matriz de adjacência} $ A $, calcular $ A^3$ e
        interpretar as entradas diagonais
        (\cref{thm:g12:matrix:walks}).
  \item Para $ D = \begin{pmatrix} 2 & 0\\ 0 & \frac12
        \end{pmatrix}$: dar $ D^n $ e seu comportamento como
        $ n \to \infty $.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The Fibonacci matriz.}
Deixar $ F = \begin{pmatrix} 1 & 1\\ 1 & 0\end{pmatrix}$ e deixe
$ F_1 = F_2 = 1, F_3 = 2, \dots $ sejam os números de Fibonacci de
\cref{pb:g11:seq:1}.
\begin{enumerate}[resume]
  \item Calcular $ F^2$, $ F^3$, $ F^4$ e conjecturar o geral
        forma de $ F^n $ em termos de números de Fibonacci.
  \item Prove a conjectura
        $ F^n = \begin{pmatrix} F_{n+1} & F_n\\ F_n & F_{n-1}
        \end{pmatrix}$ por indução.
  \item Pegar \omterm{def:g11:vect:det}{determinantes} de ambos os lados (o \omterm{def:g11:vect:det}{determinante} de um
        produto é o produto do \omterm{def:g11:vect:det}{determinantes} --- confira
        ligado $2 \times 2$ matrizes se você nunca viu):
        deduzir \emph{Identidade da Cassini}
        $ F_{n+1}F_{n-1} - F_n^2 = (-1)^n $ --- o
        motor quadrado evanescente, comprovado em uma linha.
  \item De $ F^{m+n} = F^m F^n $, leia o canto superior direito
        entradas e derivar o \emph{fórmula de adição}
        \[
        F_{m+n} = F_{m+1} F_n + F_m F_{n-1} .
        \]
        Verifique se $ m = n = 3$.
  \item Deduza da fórmula de adição (indução em $ k $)
        que $ F_n $ \omterm{def:g12:arith:divides}{divide} $ F_{kn}$ e verifique em
        $ F_3 \mid F_6$ e $ F_3 \mid F_9$.
  \item Para calcular $ F_{100}$, não é preciso multiplicar $100$
        matrizes: quadrado repetidamente ($ F^2, F^4, F^8, \dots $)
        e combinar. Quantos matriz multiplicações são suficientes,
        e qual antigo truque de multiplicação do Médio
        Volume escolar é esse, promovido para matrizes?
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The weather machine.}
Numa determinada cidade: depois de um dia ensolarado, o próximo está ensolarado com
\omterm{def:g10:proba:distribution}{probabilidade} $0.8$; depois de um dia chuvoso, ensolarado com \omterm{def:g10:proba:distribution}{probabilidade}
$0.4$. Codifique o dia \omterm{def:g11:prob:rv}{distribuição} como uma coluna
$\binom{p_{\text{sun}}}{p_{\text{rain}}}$ e a evolução por
\[
M = \begin{pmatrix} 0.8 & 0.4\\ 0.2 & 0.6 \end{pmatrix}.
\]
\begin{enumerate}[resume]
  \item Verifique se cada coluna de $ M $ somas para $1$ e diga por que
        qualquer máquina meteorológica deve ter esta propriedade.
  \item Hoje está ensolarado. Calcule a previsão para amanhã e
        para o dia seguinte.
  \item Encontre o \emph{curso estável}: o \omterm{def:g11:prob:rv}{distribuição} $ v $ com
        $ Mv = v $ (e entradas somando $1$). Que fração
        dias são ensolarados a longo prazo?
  \item Comece em um dia chuvoso, $\binom01$ e aplique $ M $ quatro
        vezes, rastreando a distância até o estado estacionário em
        cada passo. Por que fator a lacuna diminui por etapa
        --- e que tipo de convergência é essa?
  \item PageRank em miniatura: três páginas, com links
        $ A \to B $, $ A \to C $, $ B \to C $, $ C \to A $. Um aleatório
        surfista segue um link de saída uniformemente e aleatoriamente.
        Escreva a transição matriz, encontre o estado estacionário e
        classificar as páginas.
  \item Interprete a classificação: por que $ C $ pontuação tão alta quanto
        $ A $ apesar de receber links de menos páginas --- o que
        o estado estacionário realmente mede? (PageRank real
        adiciona um fator de amortecimento para becos sem saída e saltos; o
        A ideia do autovetor é exatamente esta.)
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Diagonal dividends.}
\begin{enumerate}[resume]
  \item Duas quantidades acopladas obedecem
        $ u_{n+1} = 3u_n + v_n $, $ v_{n+1} = u_n + 3v_n $, ou seja\
        o matriz $ A $ de \cref{exo:g12:matrix:8}. Usando isso
        diagonalização do exercício
        ($ D = \operatorname{diag}(4, 2)$), dê o fechado
        fórmula para $ u_n $ quando $ u_0 = 1$, $ v_0 = 0$ e verifique
        contra a computação direta para $ n = 1, 2, 3$.
  \item Em uma ou duas frases: o que significa diagonalização
        \emph{fazer} para um sistema acoplado --- e em que sentido é
        o estado estacionário de Markov da questão 14 é um autovetor
        história também?
  \item Final --- o matriztrês rostos neste fim de semana:
        escrituração contábil (sistemas e inversos), combinatória
        (caminhadas e links contados por potências) e evolução
        (Fibonacci, clima, web --- leitura de futuros
        direções próprias). Uma frase para cada, mais a
        ponteiro direto: a álgebra linear dos volumes universitários
        faz de cada uma dessas faces uma teoria.
\end{enumerate}
\end{problem}
