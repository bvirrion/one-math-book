\chapter{Funções trigonométricas}\label{ch:g12:trigo}

O funções $\cos $ e $\sin $ transforme a geometria do círculo em
análise. Definidos enrolando a linha real ao redor do círculo unitário, eles são
o arquétipo do periódico funções, e seus derivados faça-os
soluções do equação de oscilações $ y'' = -y $.

\section{O círculo unitário}

\begin{definition}[Cosseno e seno]\label{def:g12:trigo:cossin}
Deixar $\mathcal{C}$ ser o círculo de raio 1 centrado na origem de um
\omterm{def:g10:coordgeom:system}{ortonormal} sistema de \omterm{def:g10:coordgeom:system}{coordenadas}. Para cada real $ t $, associe o ponto $ M(t)$
obtido enrolando um comprimento $\abs{t}$ junto $\mathcal{C}$ do ponto
$ I(1,0)$, no sentido anti-horário se $ t \geq 0$ e no sentido horário caso contrário. Então
\emph{$\cos t $ e $\sin t $ são as coordenadas de
$ M(t)$}\index{$\cos t $ e $\sin t $ são as coordenadas de
$ M(t)$}\index{seno}:
\[
M(t) = (\cos t, \sin t).
\]
\end{definition}

Como o círculo tem circunferência $2\pi $, o ponto $ M(t)$ permanece inalterado quando
$ t $ aumenta em $2\pi $; e desde então $ M(t)$ encontra-se no círculo unitário:

\begin{proposition}[Identidades fundamentais]\label{prop:g12:trigo:identities}
Para todos $ t \in \R $ e $ k \in \Z $:
\[
\cos^2 t + \sin^2 t = 1, \qquad
\cos(t + 2k\pi) = \cos t, \qquad
\sin(t + 2k\pi) = \sin t,
\]
\[
\cos(-t) = \cos t, \qquad \sin(-t) = -\sin t,
\]
\[
\cos(\pi - t) = -\cos t, \quad \sin(\pi - t) = \sin t, \quad
\cos\left(\tfrac{\pi}{2} - t\right) = \sin t, \quad
\sin\left(\tfrac{\pi}{2} - t\right) = \cos t .
\]
\end{proposition}

\begin{proof}
A primeira identidade é a equação do círculo unitário; os outros expressam o
simetrias do círculo: $ M(-t)$ é o reflexo de $ M(t)$ no
$ x $-eixo, $ M(\pi - t)$ no $ y $-eixo, e $ M(\frac\pi2 - t)$ na linha
$ y = x $.
\end{proof}

Os valores para saber de cor:
\begin{center}
\begin{tabular}{c|ccccc}
$ t $ & $0$ & $\dfrac{\pi}{6}$ & $\dfrac{\pi}{4}$ & $\dfrac{\pi}{3}$ & $\dfrac{\pi}{2}$\\[6pt]
\hline\rule{0pt}{14pt}%
$\cos t $ & $1$ & $\dfrac{\sqrt3}{2}$ & $\dfrac{\sqrt2}{2}$ & $\dfrac12$ & $0$\\[6pt]
$\sin t $ & $0$ & $\dfrac12$ & $\dfrac{\sqrt2}{2}$ & $\dfrac{\sqrt3}{2}$ & $1$
\end{tabular}
\end{center}

\begin{proposition}[Fórmulas de adição]\label{prop:g12:trigo:addition}
Para todos $ a, b \in \R $:
\begin{align*}
\cos(a+b) &= \cos a \cos b - \sin a \sin b, &
\sin(a+b) &= \sin a \cos b + \cos a \sin b,\\
\cos(a-b) &= \cos a \cos b + \sin a \sin b, &
\sin(a-b) &= \sin a \cos b - \cos a \sin b.
\end{align*}
Em particular $\cos 2a = 2\cos^2 a - 1 = 1 - 2\sin^2 a $ e
$\sin 2a = 2 \sin a \cos a $.
\end{proposition}

\begin{proof}
O \omterm{def:g11:vect:vector}{vetores} $\vect{u} = (\cos a, \sin a)$ e $\vect{v} = (\cos b, \sin b)$
são unidades \omterm{def:g11:vect:vector}{vetores} fazendo um ângulo $ a - b $; deles \omterm{def:g11:scal:dot}{produto escalar} é igual a ambos
$\cos a\cos b + \sin a \sin b $ (\omterm{def:g10:coordgeom:system}{coordenadas}) e
$\norm{\vect u}\,\norm{\vect v}\cos(a-b) = \cos(a - b)$ (geometria), que é
a terceira fórmula. Substituindo $ b $ por $-b $ dá o primeiro; o \omterm{def:g12:trigo:cossin}{seno} fórmulas
siga usando $\sin x = \cos(\frac\pi2 - x)$. As fórmulas de duplicação são as
caso $ b = a $ combinado com $\cos^2 + \sin^2 = 1$.
\end{proof}

\section{Estudo analítico}

\begin{lemma}[Limite fundamental]\label{lem:g12:trigo:sinxx}
\[
\lim_{t \to 0} \frac{\sin t}{t} = 1
\qquad\text{and}\qquad
\lim_{t \to 0} \frac{\cos t - 1}{t} = 0 .
\]
\end{lemma}

\begin{proof}
Para $0 < t < \frac{\pi}{2}$, compare três áreas no círculo unitário: o
triângulo $ OIM(t)$, o setor circular $ OIM(t)$, e o triângulo retângulo com
básico $ OI $ e altura $\tan t $:
\[
\frac{\sin t}{2} \leq \frac{t}{2} \leq \frac{\tan t}{2}.
\]

  \draw[->] (0,-0.25) -- (0,1.45) node[above] {$ y $};

  \node[left, font=\small] at (0.68,0.35) {$\sin t $};

\end{tikzpicture}

{\small As três áreas aninhadas: o triângulo $ OIM $ (azul, área
$\frac{\sin t}{2}$), o setor circular (delimitado pelo arco vermelho, área
$\frac{t}{2}$) e o triângulo retângulo de altura $\tan t $ (laranja, área
$\frac{\tan t}{2}$).}
\end{omfigure}

A primeira desigualdade dá $\frac{\sin t}{t} \leq 1$; o segundo dá
$\cos t \leq \frac{\sin t}{t}$. Como $ t \to 0^+$, $\cos t \to 1$ (continuidade),
e o teorema da compressão produz $\frac{\sin t}{t} \to 1$; paridade estende isso
para $ t \to 0$. Para o segundo limite,
\[
\frac{\cos t - 1}{t} = \frac{\cos^2 t - 1}{t(\cos t + 1)}
= -\frac{\sin t}{t}\cdot\frac{\sin t}{\cos t + 1}
\xrightarrow[t\to0]{} -1 \cdot \frac{0}{2} = 0. \qedhere
\]
\end{proof}

\begin{theorem}[Derivadas de seno e cosseno]\label{thm:g12:trigo:derivatives}
O funções $\sin $ e $\cos $ são \omterm{def:g12:deriv:derivative}{diferenciável} sobre $\R $, com
\[
(\sin)' = \cos, \qquad (\cos)' = -\sin .
\]
Mais geralmente, $(\sin(ax+b))' = a\cos(ax+b)$ e
$(\cos(ax+b))' = -a\sin(ax+b)$.
\end{theorem}

\begin{proof}
Pelas fórmulas de adição,
\[
\frac{\sin(x+h) - \sin x}{h}
= \sin x\,\frac{\cos h - 1}{h} + \cos x\,\frac{\sin h}{h}
\xrightarrow[h\to0]{} \sin x \cdot 0 + \cos x \cdot 1 = \cos x,
\]
usando \cref{lem:g12:trigo:sinxx}. O cálculo para $\cos $ é idêntico,
e a forma geral segue da regra da cadeia.
\end{proof}

\begin{proposition}[Variações]\label{prop:g12:trigo:variations}
Sobre $\intcc{0}{\pi}$, $\cos $ diminui de $1$ para $-1$. Sobre
$\intcc{-\frac\pi2}{\frac\pi2}$, $\sin $ aumenta de $-1$ para $1$. Ambos
funções são $2\pi $-periódico e tem \omterm{def:g10:numbers:interval}{intervalos} $\intcc{-1}{1}$; $\cos $ é
mesmo, $\sin $ é estranho.
\end{proposition}

\begin{proof}
Sobre $\intoo{0}{\pi}$, $(\cos)' = -\sin < 0$ desde o ponto $ M(t)$ tem
positivo \omterm{def:g10:coordgeom:system}{ordenar} lá; de forma similar $(\sin)' = \cos > 0$ sobre
$\intoo{-\frac\pi2}{\frac\pi2}$. Periodicidade e paridade vêm de
\cref{prop:g12:trigo:identities}.
\end{proof}

\begin{method}[Resolvendo $\cos x = a $ e $\sin x = a $]\label{met:g12:trigo:equations}
Para $ a \in \intcc{-1}{1}$, encontre uma solução específica $\alpha $ (do
tabela de valores ou uma calculadora). Então todas as soluções são:
\[
\cos x = \cos\alpha \iff x = \alpha + 2k\pi \text{ or } x = -\alpha + 2k\pi,
\quad k \in \Z;
\]
\[
\sin x = \sin\alpha \iff x = \alpha + 2k\pi \text{ or } x = \pi - \alpha + 2k\pi,
\quad k \in \Z.
\]
Para desigualdades, localize os arcos de solução no círculo unitário e leia
o \omterm{def:g10:numbers:interval}{intervalos} dentro de um período.
\end{method}

\begin{example}\label{ex:g12:trigo:equation}
Resolver $\cos x = \frac12$ em $\intoc{-\pi}{\pi}$: uma solução particular é
$\frac\pi3$, então as soluções são $ x = \frac\pi3$ e $ x = -\frac\pi3$.
Em todos $\R $: $ x = \pm\frac\pi3 + 2k\pi $, $ k \in \Z $.
\end{example}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g12:trigo:1}
Calcular $\cos\frac{2\pi}{3}$, $\sin\frac{5\pi}{6}$, $\cos\frac{7\pi}{4}$ e
$\sin\left(-\frac{\pi}{3}\right)$ usando as identidades de
\cref{prop:g12:trigo:identities}.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g12:trigo:2}
Diferenciar $ f(x) = \sin^2 x $, $ g(x) = \cos(3x) + x\sin x $ e
$ h(x) = \dfrac{\sin x}{2 + \cos x}$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g12:trigo:3}
Resolva em $\intoc{-\pi}{\pi}$, então em $\R $:
\[
\text{(a) } \sin x = \frac{\sqrt3}{2};
\qquad
\text{(b) } \cos\left(2x\right) = \frac{\sqrt2}{2} .
\]
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:trigo:4}
Usando as fórmulas de adição, calcule o valor exato de
$\cos\frac{\pi}{12}$ e $\sin\frac{\pi}{12}$.
(Dica: $\frac{\pi}{12} = \frac{\pi}{3} - \frac{\pi}{4}$.)
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:trigo:5}
Resolva em $\intcc{0}{2\pi}$ a desigualdade $2\sin^2 x - \sin x - 1 \geq 0$.
(Dica: fatore o quadrático em $\sin x $.)
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:trigo:6}
Calcule os limites
\[
\lim_{x\to0} \frac{\sin 3x}{x}, \qquad
\lim_{x\to0} \frac{1 - \cos x}{x^2}, \qquad
\lim_{x\to+\infty} x \sin\frac{1}{x}.
\]
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:trigo:7}
Estude o \omterm{def:g11:func:function}{função} $ f(x) = x + \cos x $ sobre $\intcc{0}{2\pi}$: variações e
extremos. Mostre isso $ f $ é \omterm{def:g12:seq:monotonic}{aumentando} sobre $\R $ embora $ f'$ desaparece em
infinitos pontos.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g12:trigo:8}
Deixar $ f(x) = A\cos x + B \sin x $ com $(A,B) \neq (0,0)$.
\begin{enumerate}
  \item Mostre isso $ f $ pode ser escrito $ f(x) = R\cos(x - \varphi)$ com
        $ R = \sqrt{A^2 + B^2}$ e um adequado $\varphi $.
  \item Deduza o \omterm{def:g10:functions:extrema}{máximo} e \omterm{def:g10:functions:extrema}{mínimo} de $ f $ e resolva
        $\cos x + \sin x = 1$ em $\intoc{-\pi}{\pi}$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g12:trigo:9}
Mostre isso $\sin $ e $\cos $ ambos satisfazem o diferencial equação
$ y'' = -y $. Por outro lado, deixe $ y $ ser uma solução de $ y'' = -y $ com
$ y(0) = a $ e $ y'(0) = b $; mostre isso $ y = a\cos + b\sin $.
(Dica: considere $ g = (y - a\cos - b\sin)$ e a ``energia''
$ E = g^2 + (g')^2$.)
\end{exercise}

\section{Problema: o segredo do pêndulo}

\begin{problem}[{Problema de fim de semana --- um limite geométrico de potências
as derivadas de seno e cosseno, e delas todas
oscilação: molas, pêndulos, batidas e o medidor que quase
era}]
\label{pb:g12:trigo:1}
Cada relógio que já bateu, cada corda que já soou,
obedece à mesma matemática: uma força restauradora proporcional a
deslocamento, daí o equação $ y'' = -\omega^2 y $, por isso
\omterm{def:g11:trigo:cossin}{cossenos}. Na base de tudo está um limite de aparência inocente,
$\frac{\sin x}{x} \to 1$ --- prometido na 9ª série, usado por
\cref{thm:g12:trigo:derivatives}, e provou aqui com uma foto.
Este problema sobe desse limite até o balanço de um metro
pêndulo, e explica por que o medidor não é, nem por um fio,
definido por ele.

\medskip
\noindent\textbf{Part I --- The fundamental limit.}
\begin{enumerate}
  \item Aquecer numericamente: calcular $\frac{\sin x}{x}$ para
        $ x = 0.1$ e $ x = 0.01$ (\omterm{def:g11:trigo:radian}{radianos}!).
  \item O sanduíche, no círculo unitário com
        $0 < x < \frac\pi2$: expressar pela geometria elementar o
        áreas de (i) o triângulo com vértices $ O $, $ A(1,0)$
        e o ponto do círculo $ M(\cos x, \sin x)$; (ii) o
        setor circular $ OAM $; (iii) o triângulo retângulo $ OAT $
        onde $ T(1, \tan x)$.
  \item Da área sanduíche deduzir
        $\sin x < x < \tan x $, então
        $\cos x < \frac{\sin x}{x} < 1$, e concluir pelo
        teorema do aperto que
        $\lim_{x \to 0} \frac{\sin x}{x} = 1$ (alçadores de paridade
        $ x < 0$).
  \item Deduzir
        $\lim_{x \to 0} \frac{1 - \cos x}{x^2} = \frac12$
        (multiplique pelo conjugado).
  \item Explique por que esse limite é o motor da
        \cref{thm:g12:trigo:derivatives} (o que é
        $\sin'(0)$?), e por que também liquida uma dívida antiga: em
        que sentido são \emph{\omterm{def:g11:trigo:radian}{radianos}} a única unidade angular para
        qual $\sin' = \cos $ mantém sem uma constante?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Harmonic motion.}
\begin{enumerate}[resume]
  \item Verifique pela regra da cadeia que
        $ y(t) = R\cos(\omega t - \varphi)$ satisfaz
        $ y'' = -\omega^2 y $.
  \item Uma massa sobre uma mola obedece $ y'' = -4y $, com
        $ y(0) = 3$ e $ y'(0) = 8$. Admitindo (de
        \cref{exo:g12:trigo:9}) que todas as soluções são
        $ A\cos 2t + B\sin 2t $, determinar $ A $ e $ B $.
  \item Reescreva a solução como $ R\cos(2t - \varphi)$
        (\cref{exo:g12:trigo:8}): dê a amplitude $ R $, o
        período e a fase $\varphi $ (três decimais).
  \item Em que instante a massa passa pela primeira vez
        posição de equilíbrio $ y = 0$?
  \item O pêndulo: um peso em uma corda de comprimento $ L $ obedece
        \emph{exatamente} $\theta'' = -\frac gL \sin\theta $ ---
        insolúvel pelo elementar funções. Para pequenos balanços,
        substituir $\sin\theta $ por $\theta $ (A Parte I justifica!)
        e deduza o período
        $ T = 2\pi\sqrt{\frac Lg}$. Calcular $ T $ para
        $ L = 1$ m ($ g = 9.81$): A famosa descoberta de Galileu ---
        o que a fórmula \emph{não} depende?
  \item O pêndulo dos segundos: que comprimento dá
        $ T = 2$ é exatamente? A Academia Francesa de 1791 hesitou
        entre este comprimento e o décimo milionésimo do
        quarto de meridiano para definir o \emph{medidor} --- por
        quantos milímetros os dois candidatos diferem?
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Addition formulas at work.}
\begin{enumerate}[resume]
  \item Mostrar com as fórmulas de adição
        (\cref{prop:g12:trigo:addition}) que
        $\cos\left(x + \frac\pi2\right) = -\sin x $ e
        $\sin\left(x + \frac\pi2\right) = \cos x $: circular
        a diferenciação é um quarto de volta --- verifique a consistência
        com \cref{thm:g12:trigo:derivatives}.
  \item Derive a identidade do produto à soma
        $\cos a \cos b = \frac12\left(\cos(a - b) +
        \cos(a + b)\right)$ e a linearização
        $\cos^2 x = \frac{1 + \cos 2x}{2}$ (a integração
        capítulo vai agradecer).
  \item Batidas: derivar
        $\sin p + \sin q = 2 \sin\frac{p + q}{2}
        \cos\frac{p - q}{2}$, e aplique-o a dois diapasões
        em $440$ e $444$ Hz: que tom o ouvido ouve e
        quantas pulsações por segundo? (É assim que as orquestras
        melodia.)
  \item Derive a identidade de ângulo triplo
        $\cos 3x = 4\cos^3 x - 3\cos x $
        (escrever $3x = 2x + x $). Contexto $ x = 20^\circ $, o que
        cúbico equação deve $\cos 20^\circ $ satisfazer? (Que isso
        cúbica não pode ser resolvida apenas com raízes quadradas é
        precisamente por que a trissecção do ângulo derrotou
        régua e compasso por dois milênios --- a história completa
        é contado nos volumes da universidade.)
  \item Uma corrente alternada lê
        $ I(t) = 3\cos(100\pi t) + 3\sqrt3 \sin(100\pi t)$.
        Reescreva-o como $ R\cos(100\pi t - \varphi)$: pico
        corrente e fase?
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The oscillating world.}
\begin{enumerate}[resume]
  \item Períodos: qual é o (menor) período de
        $\sin^2 t $ (use a pergunta 13)? De
        $\sin(2t) + \sin(3t)$?
  \item Um pêndulo real amortece:
        $ y(t) = \eu^{-t/10}\cos(2\pi t)$. Descreva o movimento,
        calcule o valor do envelope em $ t = 10$ (um velho amigo
        aparece, \cref{pb:g12:exp:1}) e nomeie o capítulo
        onde equações com tais soluções são resolvidos.
  \item Por que fazer \omterm{def:g12:trigo:cossin}{senos} e \omterm{def:g11:trigo:cossin}{cossenos} governar todas as vibrações? Dê o
        dicionário de física para matemática em duas frases
        (força restauradora proporcional ao deslocamento
        $\rightarrow $ qual equação $\rightarrow$ qual
        soluções) e diga o que \cref{exo:g12:trigo:9}
        contribui para a reivindicação.
  \item Finale: o arco do capítulo em quatro etapas --- uma
        limite \omterm{def:g12:seq:arith-geom}{geométrico}, dois derivados, um diferencial
        equação, tudo de oscilação. E o asterisco honesto:
        para grandes oscilações, o período verdadeiro do pêndulo envolve um
        integral elíptica, calculada na velocidade da luz por\,\dots
        \omterm{def:g12:seq:arith-geom}{Aritmética} de Gauss - geométrica significar, a sobremesa de
        \cref{pb:g12:seq:1}. Os fios da série se amarram.
\end{enumerate}
\end{problem}
