\chapter{Sequências}\label{ch:g12:seq}

UM \omterm{def:g12:seq:sequence}{sequência} é uma lista de \omterm{def:g10:numbers:sets}{números reais} indexado pelo \omterm{def:g10:numbers:sets}{números naturais}. \omterm{def:g12:seq:sequence}{Sequências}
modelar evoluções discretas --- populações contadas ano a ano, saldos de um
conta bancária, sucessiva aproximações de um número --- e seus limites são
o primeiro encontro sério com o infinito. Este capítulo estabelece
vocabulário, o princípio da indução e a convergência fundamental
teoremas.

\section{Raciocínio por indução}

\begin{theorem}[Princípio da indução]\label{thm:g12:seq:induction}
Deixar $P(n)$ ser uma declaração dependendo de um inteiro $ n $, e deixe $ n_0 \in \N $.
Se
\begin{enumerate}
  \item \emph{(caso base)} $ P(n_0)$ é verdade, e
  \item \emph{(etapa indutiva)} para cada $ n \geq n_0$, $ P(n)$ implica $ P(n+1)$,
\end{enumerate}
então $ P(n)$ é verdade para todos $ n \geq n_0$.
\end{theorem}

\begin{proof}
Suponha, por contradição, que o conjunto $ A $ de \omterm{def:g10:numbers:sets}{inteiros} $ n \geq n_0$ para
qual $ P(n)$ é falso não é vazio. Então $ A $ tem um menor elemento
$ m $.\footnote{Every nonempty subset of $\N $ has a smallest element; this
property of $\N $ is taken as an axiom.} Desde $ P(n_0)$ é verdade, $ m > n_0$, então
$ m - 1 \geq n_0$ e $ m-1 \notin A $, ou seja\ $ P(m-1)$ é verdade. O indutivo
etapa aplicada a $ n = m-1$ então mostra que $ P(m)$ é verdade, contradizendo
$ m \in A $.
\end{proof}

\begin{example}\label{ex:g12:seq:bernoulli}
Vamos provar \emph{A desigualdade de Bernoulli}: para cada real $ a > 0$ e cada
$ n \in \N $,
\[
(1+a)^n \geq 1 + na.
\]
\emph{Caso básico.} Para $ n = 0$, ambos os lados iguais $1$.
\emph{Etapa indutiva.} Suponha $(1+a)^n \geq 1+na $ para alguns $ n \in \N $. Desde
$1 + a > 0$, multiplicando ambos os lados por $1+a $ preserva a desigualdade:
\[
(1+a)^{n+1} \geq (1+na)(1+a) = 1 + (n+1)a + na^2 \geq 1 + (n+1)a .
\]
Por indução, a desigualdade vale para todos $ n \in \N $.
\end{example}

\begin{method}[Escrevendo uma prova de indução]\label{met:g12:seq:induction}
Sempre faça a declaração $ P(n)$ explícito antes de começar. Uma prova completa
tem três partes visíveis: o caso base, o passo indutivo (``assumir $ P(n)$;
nós provamos $ P(n+1)$''), e a conclusão invocando o princípio da indução.
O erro mais comum é provar o passo indutivo sem nunca usar o
hipótese $ P(n)$: se isso acontecer, ou a prova está errada ou a indução foi
não é necessário.
\end{method}

\section{Vocabulário de sequências}

\begin{definition}[Sequência]\label{def:g12:seq:sequence}
UM \emph{sequência}\index{sequência} é um \omterm{def:g11:func:function}{função} $ u \colon \N \to \R $ (ou de
$\{n \in \N : n \geq n_0\}$ para $\R $). O \omterm{def:g10:functions:function}{imagem} de $ n $ está escrito $ u_n $ e
a sequência em si $(u_n)_{n\in\N}$ ou simplesmente $(u_n)$.
\end{definition}

UM \omterm{def:g12:seq:sequence}{sequência} pode ser definido \emph{explicitamente}, por uma fórmula $ u_n = f(n)$, ou por
\emph{recorrência}, por seu primeiro termo e uma relação $ u_{n+1} = f(u_n)$.

\begin{definition}[Monotonicidade]\label{def:g12:seq:monotonic}
UM \omterm{def:g12:seq:sequence}{sequência} $(u_n)$ é \emph{aumentando}\index{aumentando} se
$ u_{n+1} \geq u_n $ para todos $ n $, \emph{diminuindo} se $ u_{n+1} \leq u_n $ para
tudo $ n $, e \emph{monótono} se estiver aumentando ou diminuindo. Isso é
\emph{estritamente} aumentando (resp.\ diminuindo) quando as desigualdades são
rigoroso.
\end{definition}

\begin{method}[Estudando a monotonicidade de uma sequência]\label{met:g12:seq:monotonicity}
Três técnicas padrão:
\begin{enumerate}
  \item estude o sinal de $ u_{n+1} - u_n $;
  \item se todos os termos forem positivos, compare $\dfrac{u_{n+1}}{u_n}$ para $1$;
  \item se $ u_n = f(n)$ com $ f $ definido em $\intco{0}{+\infty}$, use o
        variações de $ f $.
\end{enumerate}
\end{method}

\begin{definition}[Sequência limitada]\label{def:g12:seq:bounded}
UM \omterm{def:g12:seq:sequence}{sequência} $(u_n)$ é \emph{limitado acima}\index{limitado acima} se houver
existe $ M \in \R $ com $ u_n \leq M $ para todos $ n $; \emph{limitado abaixo} se houver
existe $ m \in \R $ com $ u_n \geq m $ para todos $ n $; e \emph{limitado} se ambos
segure.
\end{definition}

\subsection{Sequências aritméticas e geométricas}

\begin{definition}[Sequências aritméticas e geométricas]\label{def:g12:seq:arith-geom}
UM \omterm{def:g12:seq:sequence}{sequência} $(u_n)$ é \emph{aritmética}\index{aritmética} com comum
diferença $ r $ se $ u_{n+1} = u_n + r $ para todos $ n $, e
\emph{geométrico}\index{geométrico} com proporção comum $ q $ se
$ u_{n+1} = q\,u_n $ para todos $ n $.
\end{definition}

\begin{proposition}[Forma explícita e somas]\label{prop:g12:seq:arith-geom}
Deixar $ n \in \N $.
\begin{enumerate}
  \item Se $(u_n)$ é \omterm{def:g12:seq:arith-geom}{aritmética} com \omterm{def:g11:seq:arithmetic}{diferença comum} $ r $, então
        $ u_n = u_0 + nr $ e
        \[
        u_0 + u_1 + \dots + u_n = (n+1)\,\frac{u_0 + u_n}{2}.
        \]
  \item Se $(u_n)$ é \omterm{def:g12:seq:arith-geom}{geométrico} com proporção comum $ q \neq 1$, então
        $ u_n = u_0\, q^n $ e
        \[
        u_0 + u_1 + \dots + u_n = u_0\,\frac{1 - q^{n+1}}{1 - q}.
        \]
\end{enumerate}
\end{proposition}

\begin{proof}
As formas explícitas seguem induções imediatas. Para o \omterm{def:g12:seq:arith-geom}{aritmética} soma,
escrever $ S = u_0 + \dots + u_n $ e adicione a mesma soma escrita na ordem inversa:
cada um dos $ n+1$ somas de colunas são iguais $ u_0 + u_n $, então $2S = (n+1)(u_0+u_n)$.
Para o \omterm{def:g12:seq:arith-geom}{geométrico} soma, calcular $ S - qS $: todos os termos são cancelados em pares, exceto o
primeiro e o último, então $(1-q)S = u_0(1 - q^{n+1})$.
\end{proof}

\section{Limite de uma sequência}

\begin{definition}[Sequência convergente]\label{def:g12:seq:limit}
UM \omterm{def:g12:seq:sequence}{sequência} $(u_n)$ \emph{convergir}\index{convergir} para o verdadeiro
número $\ell $ se cada aberto \omterm{def:g10:numbers:interval}{intervalo} contendo $\ell $ contém todos os termos
$ u_n $ de algum índice em diante. Escrevemos então $\lim\limits_{n\to+\infty} u_n = \ell $.

Equivalentemente: para cada $\varepsilon > 0$, existe $ N \in \N $ tal que
para todos $ n \geq N $, $\abs{u_n - \ell} \leq \varepsilon $.
\end{definition}

\begin{definition}[Divergência ao infinito]\label{def:g12:seq:infinity}
O \omterm{def:g12:seq:sequence}{sequência} $(u_n)$ \emph{tende a $+\infty $} se para cada $ A \in \R $, lá
existe $ N \in \N $ tal que $ u_n \geq A $ para todos $ n \geq N $. Nós escrevemos
$\lim\limits_{n\to+\infty} u_n = +\infty $; a definição de
$\lim u_n = -\infty $ é análogo. UM \omterm{def:g12:seq:sequence}{sequência} que não converge é dito
para \emph{divergir}.
\end{definition}

\begin{remark}
UM \omterm{def:g12:seq:sequence}{sequência} pode divergir sem tender a $\pm\infty $: o \omterm{def:g12:seq:sequence}{sequência}
$ u_n = (-1)^n $ leva apenas os valores $1$ e $-1$ e não tem limite.
\end{remark}

\begin{proposition}[Singularidade do limite]\label{prop:g12:seq:uniqueness}
Se $(u_n)$ converge, seu limite é único.
\end{proposition}

\begin{proof}
Suponha $ u_n \to \ell $ e $ u_n \to \ell'$ com $\ell \neq \ell'$, dizer
$\ell < \ell'$. Definir $\varepsilon = \frac{\ell' - \ell}{3} > 0$. De alguns
indexar ativado, $\abs{u_n - \ell} \leq \varepsilon $ e
$\abs{u_n - \ell'} \leq \varepsilon $, por isso
\[
\ell' - \ell \leq \abs{\ell' - u_n} + \abs{u_n - \ell}
\leq 2\varepsilon = \tfrac{2}{3}(\ell' - \ell) < \ell' - \ell,
\]
uma contradição.
\end{proof}

\begin{proposition}[Operações sobre limites]\label{prop:g12:seq:operations}
Deixar $(u_n)$ e $(v_n)$ ser \omterm{def:g12:seq:sequence}{sequências} com limites $\ell $ e $\ell'$ (finito ou
infinito). Então, sempre que o lado direito não for uma forma indeterminada,
\[
\lim (u_n + v_n) = \ell + \ell', \qquad
\lim (u_n v_n) = \ell\,\ell', \qquad
\lim \frac{u_n}{v_n} = \frac{\ell}{\ell'}.
\]
As formas indeterminadas são
$(+\infty) + (-\infty)$, $0 \times \infty $, $\frac{\infty}{\infty}$ e
$\frac{0}{0}$.
\end{proposition}

\begin{proof}
Provamos a regra da soma para limites finitos; os outros casos são semelhantes e deixaram
como exercícios. Deixar $\varepsilon > 0$. Existem $ N_1, N_2$ tal que
$\abs{u_n - \ell} \leq \varepsilon/2$ para $ n \geq N_1$ e
$\abs{v_n - \ell'} \leq \varepsilon/2$ para $ n \geq N_2$. Para
$ n \geq \max(N_1, N_2)$, a desigualdade triangular dá
\[
\abs{(u_n + v_n) - (\ell + \ell')}
\leq \abs{u_n - \ell} + \abs{v_n - \ell'} \leq \varepsilon. \qedhere
\]
\end{proof}

\begin{method}[Levantando uma forma indeterminada]\label{met:g12:seq:indeterminate}
Diante de uma forma indeterminada, fatore o termo dominante. Por exemplo
\[
n^2 - n = n^2\left(1 - \tfrac{1}{n}\right) \xrightarrow[n\to+\infty]{} +\infty,
\qquad
\frac{2n^2+1}{n^2 - n} = \frac{2 + 1/n^2}{1 - 1/n}
\xrightarrow[n\to+\infty]{} 2 .
\]
\end{method}

\section{Teoremas de convergência}

\begin{theorem}[Teoremas de comparação e compressão]\label{thm:g12:seq:squeeze}
Deixar $(u_n)$, $(v_n)$, $(w_n)$ ser \omterm{def:g12:seq:sequence}{sequências}.
\begin{enumerate}
  \item Se $ u_n \leq v_n $ de algum índice em diante e $ u_n \to +\infty $, então
        $ v_n \to +\infty $.
  \item \emph{(Teorema da abertura)}\index{(Teorema da abertura)} Se
        $ u_n \leq v_n \leq w_n $ de algum índice em diante e $(u_n)$ e $(w_n)$
        ambos convergem para o mesmo limite $\ell $, então $(v_n)$ converge
        para ~$\ell $.
\end{enumerate}
\end{theorem}

\begin{proof}
\emph{1.} Deixar $ A \in \R $. Desde $ u_n \to +\infty $, há $ N $ com
$ u_n \geq A $ para $ n \geq N $; ampliando $ N $ se necessário, $ v_n \geq u_n \geq A $
para $ n \geq N $.

\emph{2.} Deixar $\varepsilon > 0$. De algum índice em diante, ambos
$\ell - \varepsilon \leq u_n $ e $ w_n \leq \ell + \varepsilon $ segure, portanto
$\ell - \varepsilon \leq u_n \leq v_n \leq w_n \leq \ell + \varepsilon $, aquilo é $\abs{v_n - \ell} \leq \varepsilon $.
\end{proof}

\begin{example}
Para todos $ n \geq 1$, $-\frac{1}{n} \leq \frac{(-1)^n}{n} \leq \frac{1}{n}$ e
ambos os limites tendem a $0$; por isso $\frac{(-1)^n}{n} \to 0$.
\end{example}

\begin{theorem}[Teorema da convergência monótona]\label{thm:g12:seq:monotone}
Um \omterm{def:g12:seq:monotonic}{aumentando} \omterm{def:g12:seq:sequence}{sequência} aquilo é \omterm{def:g12:seq:bounded}{limitado acima} converge. UM diminuindo \omterm{def:g12:seq:sequence}{sequência}
aquilo é limitado abaixo converge. Um \omterm{def:g12:seq:monotonic}{aumentando} \omterm{def:g12:seq:sequence}{sequência} isso não é \omterm{def:g12:seq:bounded}{limitado
acima} tende a $+\infty $.
\end{theorem}

\begin{proof}[Prova parcial]
Provamos a terceira afirmação. Deixar $(u_n)$ ser \omterm{def:g12:seq:monotonic}{aumentando} e não \omterm{def:g12:seq:bounded}{limitado
acima}, e deixe $ A \in \R $. Desde $ A $ não é um limite superior, existe $ N $
com $ u_N \geq A $; pela monotonicidade, $ u_n \geq u_N \geq A $ para todos $ n \geq N $.
Por isso $ u_n \to +\infty $.

As duas declarações de convergência baseiam-se na propriedade do limite superior \omterm{def:g10:functions:extrema}{mínimo} de
$\R $; eles são \emph{admitido neste nível} (e comprovado no primeiro ano de
universidade).
\end{proof}

\begin{remark}
O teorema garante a \emph{existência} do limite, mas não dá
seu valor. Um \omterm{def:g12:seq:monotonic}{aumentando} \omterm{def:g12:seq:sequence}{sequência} \omterm{def:g12:seq:bounded}{limitado acima} por $ M $ converge para alguns
$\ell \leq M $, não necessariamente para $ M $.
\end{remark}

\begin{theorem}[Limite de sequências geométricas]\label{thm:g12:seq:geometric}
Deixar $ q \in \R $.
\begin{enumerate}
  \item Se $ q > 1$, então $ q^n \to +\infty $.
  \item Se $ q = 1$, então $ q^n \to 1$.
  \item Se $\abs{q} < 1$, então $ q^n \to 0$.
  \item Se $ q \leq -1$, então $(q^n)$ diverge e não tem limite.
\end{enumerate}
\end{theorem}

\begin{proof}
\emph{1.} Escrever $ q = 1 + a $ com $ a > 0$. A desigualdade de Bernoulli
(\cref{ex:g12:seq:bernoulli}) dá $ q^n \geq 1 + na \to +\infty $, e nós
concluir por comparação (\cref{thm:g12:seq:squeeze}).

\emph{2.} Imediato.

\emph{3.} Se $ q = 0$ a afirmação é clara. De outra forma $\abs{q} < 1$ dá
$1/\abs{q} > 1$, então $(1/\abs{q})^n \to +\infty $ pelo ponto 1, portanto
$\abs{q}^n \to 0$, e $-\abs{q}^n \leq q^n \leq \abs{q}^n $ nos permite
conclua pelo teorema do aperto.

\emph{4.} Para $ q \leq -1$, $(q^{2n})$ leva valores $\geq 1$ enquanto
$(q^{2n+1})$ leva valores $\leq -1$: nenhum limite único pode atrair ambos
subsequências.
\end{proof}

\begin{method}[Sequências recorrentes $ u_{n+1} = f(u_n)$]\label{met:g12:seq:recurrent}
Para estudar um \omterm{def:g12:seq:sequence}{sequência} definido por $ u_{n+1} = f(u_n)$:
\begin{enumerate}
  \item prove por indução que $(u_n)$ fica em um \omterm{def:g10:numbers:interval}{intervalo} $ I $ em que $ f $
        é bem comportado (e, muitas vezes, que $(u_n)$ é monótono);
  \item deduzir a convergência do teorema da convergência monótona;
  \item passe ao limite na relação $ u_{n+1} = f(u_n)$: se $ f $ é
        contínuo e $ u_n \to \ell \in I $, então $\ell $ satisfaz
        $ f(\ell) = \ell $ (ver \cref{ch:g12:limcont}); resolva isso equação e
        selecione o certo \omterm{def:g11:quad:discriminant}{raiz}.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g12:seq:sqrt2}
Deixar $ u_0 = 2$ e $ u_{n+1} = \frac{1}{2}\left(u_n + \frac{2}{u_n}\right)$.
Verifica-se por indução que $ u_n \geq \sqrt{2}$ para todos $ n $ (a desigualdade
$\frac{1}{2}(x + 2/x) \geq \sqrt{2}$ para $ x>0$ é equivalente a
$(x - \sqrt2)^2 \geq 0$), então isso $(u_n)$ é diminuindo, desde
\[
u_{n+1}-u_n=\frac{2-u_n^2}{2u_n}\leq 0 .
\]
Diminuindo e limitado abaixo, $(u_n)$
converge para alguns $\ell \geq \sqrt{2}$, que deve satisfazer
$\ell = \frac{1}{2}(\ell + 2/\ell)$, ou seja\ $\ell^2 = 2$. Por isso
$ u_n \to \sqrt{2}$. Este é o algoritmo de Heron, já utilizado pelo
Babilônios; sua convergência é extremamente rápida ($ u_3$ já dá $\sqrt 2$
até oito casas decimais).
\end{example}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g12:seq:1}
Prove por indução que para todo $ n \in \N $,
\[
1^2 + 2^2 + \dots + n^2 = \frac{n(n+1)(2n+1)}{6}.
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g12:seq:2}
Estude a monotonicidade do \omterm{def:g12:seq:sequence}{sequências} definido para $ n \geq 1$ por
\[
a_n = \frac{n+1}{n}, \qquad
b_n = \frac{2^n}{n}, \qquad
c_n = n^2 - 10n .
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g12:seq:3}
Calcule os limites do \omterm{def:g12:seq:sequence}{sequências} com termos gerais
\[
u_n = \frac{3n^2 - n + 1}{2n^2 + 5}, \qquad
v_n = \sqrt{n+1} - \sqrt{n}, \qquad
w_n = \frac{2^n - 3^n}{3^n + 1}.
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g12:seq:4}
Deixar $(u_n)$ seja o \omterm{def:g12:seq:sequence}{sequência} \omterm{def:g12:seq:arith-geom}{aritmética} com $ u_0 = 5$ e \omterm{def:g11:seq:arithmetic}{diferença comum}
$ r = 3$, e $(v_n)$ o \omterm{def:g12:seq:sequence}{sequência} geométrica com $ v_0 = 8$ e proporção comum
$ q = \frac{1}{2}$. Calcular $ u_n $, $ v_n $, $\sum_{k=0}^{n} u_k $ e
$\sum_{k=0}^{n} v_k $, e os limites de todas as quatro expressões como
$ n \to +\infty $.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:seq:5}
Usando o teorema da compressão, calcule
\[
\lim_{n\to+\infty} \frac{n + \cos n}{n + 1}
\qquad\text{and}\qquad
\lim_{n\to+\infty} \frac{n!}{n^n},
\]
onde $ n! = 1 \times 2 \times \dots \times n $. Para o segundo limite, limite
$\frac{n!}{n^n}$ por um termo de \omterm{def:g12:seq:sequence}{sequência} geométrica.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:seq:6}
Deixar $ u_0 = 0$ e $ u_{n+1} = \sqrt{u_n + 2}$ para todos $ n \in \N $.
\begin{enumerate}
  \item Prove por indução que $0 \leq u_n \leq 2$ para todos $ n $.
  \item Mostre isso $(u_n)$ é \omterm{def:g12:seq:monotonic}{aumentando}.
  \item Deduza isso $(u_n)$ converge e determine seu limite.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:seq:7}
Um paciente toma uma dose de $1$ unidade de um medicamento todas as manhãs. Durante cada
Período de 24 horas, o corpo elimina $40\%$ da droga presente. Deixar $ u_n $ ser
a quantidade de medicamento no corpo logo após a dose no dia $ n $, para que
$ u_0 = 1$.
\begin{enumerate}
  \item Justifique isso $ u_{n+1} = 0.6\,u_n + 1$.
  \item Deixar $ v_n = u_n - 2.5$. Mostre isso $(v_n)$ é \omterm{def:g12:seq:arith-geom}{geométrico} e deduzir um
        fórmula explícita para $ u_n $.
  \item Determine a quantidade de droga no corpo a longo prazo.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:seq:8}
Deixar $(u_n)$ ser definido por $ u_0 = 3$ e $ u_{n+1} = \frac{4u_n - 1}{u_n + 2}$.
\begin{enumerate}
  \item Mostre por indução que $ u_n > 1$ para todos $ n \in \N $.
  \item Mostre isso $ v_n = \dfrac{1}{u_n - 1}$ define um \omterm{def:g12:seq:sequence}{sequência} \omterm{def:g12:seq:arith-geom}{aritmética}.
  \item Deduzir fórmulas explícitas para $ v_n $ e $ u_n $, e o limite de
        $(u_n)$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g12:seq:9}
Para $ n \geq 1$, deixar
$ H_n = 1 + \frac{1}{2} + \frac{1}{3} + \dots + \frac{1}{n}$.
\begin{enumerate}
  \item Mostre isso para todos $ n \geq 1$, $ H_{2n} - H_n \geq \frac{1}{2}$.
  \item Deduza isso $ H_{2^k} \geq 1 + \frac{k}{2}$ para todos $ k \in \N $ e
        concluir que $ H_n \to +\infty $.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g12:seq:10}
\emph{(Adjacente \omterm{def:g12:seq:sequence}{sequências}.)} Dois \omterm{def:g12:seq:sequence}{sequências} $(a_n)$ e $(b_n)$ são
\emph{adjacente} se $(a_n)$ é \omterm{def:g12:seq:monotonic}{aumentando}, $(b_n)$ é diminuindo, e
$ b_n - a_n \to 0$.
\begin{enumerate}
  \item Mostre isso para todos $ n $, $ a_n \leq b_n $. (Dica: estude a monotonicidade
        de $(b_n - a_n)$.)
  \item Mostre que adjacente \omterm{def:g12:seq:sequence}{sequências} ambos convergem, para o mesmo limite.
  \item Aplicação: mostre que \omterm{def:g12:seq:sequence}{sequências}
        $ a_n = \sum_{k=0}^{n} \frac{1}{k!}$ e
        $ b_n = a_n + \frac{1}{n \cdot n!}$ ($ n \geq 1$) são adjacentes. (Seu
        limite comum é o número $\eu $, estudou em \cref{ch:g12:exp}.)
\end{enumerate}
\end{exercise}

\section{Problema: sequência de Heron, finalmente julgada}

\begin{problem}[{Problema de fim de semana --- certificação de indução, monótono
sentenças de convergência e a receita de dois mil anos para
$\sqrt2$ finalmente obtém sua prova (com a maravilhosa média de Gauss para
sobremesa)}]
\label{pb:g12:seq:1}
Três vezes esta série atendeu à receita de Heron --- média
adivinhe com $2/\text{guess}$ --- e três vezes só poderia
\emph{observar} que a receita funciona. Este capítulo finalmente possui
os instrumentos de julgamento: indução
(\cref{thm:g12:seq:induction}), o teorema da convergência monótona
(\cref{thm:g12:seq:monotone}) e limites de recorrências. O
o veredicto e a velocidade certificada ocupam o cerne deste
problema; em torno dele, as armadilhas clássicas da indução, as mais lentas
divergência em matemática, e a convergência mais rápida de Gauss já
encontrado.

\medskip
\noindent\textbf{Part I --- Induction warm-ups.}
\begin{enumerate}
  \item Prove por indução:
        $1 + 3 + 5 + \dots + (2n - 1) = n^2$ (a escada de
        números ímpares, sorteados no volume do Ensino Médio, agora
        certificado).
  \item Prove por indução que $2^n > n $ para cada
        $ n \in \N $.
  \item Prove a desigualdade de Bernoulli por indução: para
        $ x \geq 0$ e $ n \in \N $,
        $(1 + x)^n \geq 1 + nx $.
  \item A armadilha clássica: ``todas as bolas de gude têm a mesma cor ---
        verdadeiro para uma bola de gude; e se houver $ n $ bolinhas de gude são sempre
        monocromático, então entre $ n + 1$ mármores o primeiro $ n $
        compartilhe uma cor, a última $ n $ compartilham uma cor, então todos
        $ n + 1$ fazer.'' Toda criança sabe que a conclusão é
        absurdo: encontre o passo exato onde a indução é interrompida.
  \item Prove por indução que $4^n - 1$ é divisível por $3$
        para cada $ n \in \N $.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The trial of Heron.}
Deixar $ x_0 = 2$ e $ x_{n+1} = \dfrac12\left(x_n +
\dfrac{2}{x_n}\right)$.
\begin{enumerate}[resume]
  \item Calcular $ x_1$, $ x_2$, $ x_3$ como frações exatas (antigo
        amigos).
  \item Prove a identidade da chave
        \[
        x_{n+1}^2 - 2 = \left(\frac{x_n^2 - 2}{2x_n}\right)^{\!2}
        \geq 0,
        \]
        e deduza por indução que $ x_n > 0$ e
        $ x_n^2 > 2$ para cada $ n $.
  \item Mostre isso $(x_n)$ é estritamente diminuindo (calcular
        $ x_{n+1} - x_n $ e use a pergunta 7).
  \item Invoque o teorema da convergência monótona: por que
        $(x_n)$ convergem para algum limite $ L \geq 1$?
  \item Identifique o limite: passe a recorrência até o limite
        (\cref{prop:g12:seq:operations}) e concluir
        $ L = \sqrt2$. Declare o veredicto histórico: depois de dois
        mil anos de serviço fiel, a receita de Heron é
        \emph{comprovado} \omterm{def:g12:seq:limit}{convergir}.
  \item A velocidade certificada: com $ e_n = x_n - \sqrt2$, provar
        \[
        e_{n+1} = \frac{e_n^2}{2 x_n} ,
        \]
        e deduzir $ e_{n+1} \leq \frac{e_n^2}{2\sqrt2}$: o
        o erro é elevado ao quadrado em cada etapa --- a duplicação de dígitos
        observado desde o volume do Ensino Médio, agora um teorema.
  \item Confirme numericamente: calcule $ e_0, e_1, e_2, e_3$ (de
        questão 6) e verifique se cada $\frac{e_{n+1}}{e_n^2}$
        está perto de $\frac{1}{2x_n}$.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The slowest divergence.}
\begin{enumerate}[resume]
  \item \Cref{exo:g12:seq:9} comprovado
        $ H_{2^k} \geq 1 + \frac k2$ para as somas harmônicas. Como
        muitos termos garantem $ H_n > 10$? (Uma potência de dois irá
        fazer; maravilhe-se com seu tamanho.)
  \item Em contrapartida, o \omterm{def:g12:seq:arith-geom}{geométrico} somas
        $1 + \frac12 + \frac14 + \dots + \frac{1}{2^n} =
        2 - \frac{1}{2^n}$ \omterm{def:g12:seq:limit}{convergir} para $2$
        (\cref{thm:g12:seq:geometric}): a barra de chocolate
        intuição do volume do ensino médio, finalmente um limite
        declaração. Escreva a prova de duas linhas.
  \item Entre os dois: mostre que as somas
        $ S_n = 1 + \frac{1}{4} + \frac{1}{9} + \dots +
        \frac{1}{n^2}$ \omterm{def:g12:seq:limit}{convergir}, delimitando
        $\frac{1}{k^2} \leq \frac{1}{k(k-1)} =
        \frac{1}{k-1} - \frac{1}{k}$ (para $ k \geq 2$),
        telescópica e aplicando convergência monótona. (O
        limite, $\frac{\pi^2}{6}$, é um dos milagres de Euler,
        comprovado nos volumes da universidade.)
  \item Declare a moral das questões 13 a 15 em duas frases:
        o que significa "os termos tendem a $0$'' decidir sobre o
        convergência das somas --- e o que não acontece?
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Gauss's arithmetic--geometric
significar.}
Deixar $ a_0 = 1$, $ b_0 = 2$, e
\[
a_{n+1} = \sqrt{a_n b_n},
\qquad
b_{n+1} = \frac{a_n + b_n}{2} .
\]
\begin{enumerate}[resume]
  \item Calcular $ a_1, b_1, a_2, b_2$ (cinco casas decimais). O que fazer
        você observa sobre a velocidade?
  \item Mostre isso $ a_n \leq b_n $ para cada $ n $ (o
        desigualdade aritmética-geométrica, encontrada ao longo deste
        série), que $(a_n)$ aumenta e $(b_n)$ diminui.
  \item Mostre isso
        $ b_{n+1} - a_{n+1} \leq \frac{b_n - a_n}{2}$
        (fator $ b_{n+1} - a_{n+1} =
        \frac{(\sqrt{b_n} - \sqrt{a_n})^2}{2}$ e comparar),
        e concluo com \cref{exo:g12:seq:10} que os dois
        \omterm{def:g12:seq:sequence}{sequências} são adjacentes: eles compartilham um limite comum
        $ M(1, 2)$, o \emph{\omterm{def:g12:seq:arith-geom}{aritmética} - geométrica significar}.
  \item Calcular $ M(1, 2)$ com seis casas decimais (quantas iterações
        você precisava?). Em 30 de maio de 1799, Gauss calculou
        $ M(1, \sqrt2)$ com onze casas decimais, reconhecido
        $\frac{\pi}{M(1,\sqrt2)}$ como uma integral conhecida, e
        escreveu que um “novo campo de análise” havia se aberto ---
        tinha: integrais elípticas, contadas na universidade
        volumes. Fechar com as velocidades de convergência observadas de
        este problema, do mais lento para o mais rápido.
\end{enumerate}
\end{problem}
