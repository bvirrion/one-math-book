\chapter{Números Complexos}\label{ch:g12:complex}

O equação $x^2 = -1$ não tem solução real. Ampliando $\R $ com um único
novo número $\iu $ cujo quadrado é $-1$ produz o campo $\C $ de complexo
números, em que \emph{todo} polinomial equação tem soluções --- e
cuja \omterm{def:g12:seq:arith-geom}{aritmética} codifica a geometria do plano: translações, rotações
e as escalas tornam-se adições e multiplicações.

\section{Forma algébrica}

\begin{definition}[Números complexos]\label{def:g12:complex:def}
O conjunto de \emph{números complexos}\index{números complexos} é
\[
\C = \{a + \iu b : a, b \in \R\},
\]
onde $\iu $ é um símbolo sujeito à regra única $\iu^2 = -1$; adição e
multiplicação são realizadas como para polinômios reais em $\iu $, reduzindo
$\iu^2$ para $-1$. Os reais $ a $ e $ b $ são os \emph{parte real}\index{parte real}
$\Rea(z)$ e \emph{parte imaginária}\index{parte imaginária} $\Ima(z)$ de $ z = a + \iu b $; isso
expressão é a \emph{forma algébrica}\index{forma algébrica} de $ z $, e é único:
$ a + \iu b = a' + \iu b'$ se e somente se $ a = a'$ e $ b = b'$.
\end{definition}

\begin{example}
$(2 + 3\iu)(1 - \iu) = 2 - 2\iu + 3\iu - 3\iu^2 = 5 + \iu $.
\end{example}

\begin{definition}[Conjugado]\label{def:g12:complex:conjugate}
O \emph{conjugado}\index{conjugado} de $ z = a + \iu b $ é
$\conj{z} = a - \iu b $.
\end{definition}

\begin{proposition}[Propriedades de conjugação]\label{prop:g12:complex:conjugate}
Para todos $ z, w \in \C $:
\[
\conj{z + w} = \conj z + \conj w, \quad
\conj{zw} = \conj z\,\conj w, \quad
\conj{\conj z} = z, \quad
z + \conj z = 2\Rea(z), \quad
z - \conj z = 2\iu\,\Ima(z),
\]
e $ z\conj z = a^2 + b^2 \geq 0$ para $ z = a + \iu b $. Além disso $ z \in \R $
se e somente se $ z = \conj z $.
\end{proposition}

\begin{proof}
Todos são cálculos diretos do formas algébricas; por exemplo
$ z\conj z = (a + \iu b)(a - \iu b) = a^2 - (\iu b)^2 = a^2 + b^2$.
\end{proof}

\begin{method}[Divisão de números complexos]\label{met:g12:complex:division}
Para escrever um quociente em \omterm{def:g12:complex:def}{forma algébrica}, multiplique o numerador e o denominador por
o \omterm{def:g12:complex:conjugate}{conjugado} do denominador:
\[
\frac{1}{2 + \iu} = \frac{2 - \iu}{(2+\iu)(2-\iu)} = \frac{2 - \iu}{5}
= \frac25 - \frac15\iu .
\]
Todo diferente de zero número complexo portanto tem um inverso: $\C $ é um
\emph{campo}.
\end{method}

\section{O plano complexo, módulo e argumento}

\begin{definition}[Afixo, módulo]\label{def:g12:complex:modulus}
No avião com \omterm{def:g10:coordgeom:system}{ortonormal} \omterm{def:g10:coordgeom:system}{coordenadas}, o ponto $ M(a, b)$ é o
\emph{\omterm{def:g10:functions:function}{imagem}} de $ z = a + \iu b $, e $ z $ é o \emph{afixo}\index{afixo} de $ M $. O
\emph{módulo}\index{módulo} de $ z $ é a distância de $ M $ para a origem:
\[
\abs{z} = \sqrt{a^2 + b^2} = \sqrt{z\conj z}.
\]
\end{definition}

\begin{proposition}[Propriedades do módulo]\label{prop:g12:complex:modulus}
Para $ z, w \in \C $:
\[
\abs{zw} = \abs z\,\abs w, \qquad
\abs{\tfrac zw} = \tfrac{\abs z}{\abs w}\ (w \neq 0), \qquad
\abs{\conj z} = \abs z, \qquad
\abs{z + w} \leq \abs{z} + \abs{w} \ \text{(triangle inequality)}.
\]
Além disso $\abs{z_B - z_A}$ é a distância entre os pontos de \omterm{def:g12:complex:modulus}{afixos}
$ z_A $ e $ z_B $.
\end{proposition}

\begin{proof}
$\abs{zw}^2 = zw\,\conj{zw} = z\conj z\, w \conj w = \abs z^2 \abs w^2$
dá o primeiro; o segundo e o terceiro são semelhantes. Para o triângulo
desigualdade:
\[
\abs{z+w}^2 = (z+w)(\conj z + \conj w)
= \abs z^2 + \abs w^2 + 2\Rea(z \conj w)
\leq \abs z^2 + \abs w^2 + 2\abs{z\conj w}
= \bigl(\abs z + \abs w\bigr)^2,
\]
usando $\Rea(u) \leq \abs u $. A declaração de distância é a pitagórica
fórmula aplicada ao \omterm{def:g10:coordgeom:system}{coordenadas} de $ z_B - z_A $.
\end{proof}

\begin{definition}[Argumento, formas trigonométricas e exponenciais]\label{def:g12:complex:argument}
Deixar $ z \neq 0$. Um \emph{argumento}\index{argumento} de $ z $, escrito
$\Arg(z)$, é qualquer medida $\theta $ do ângulo do eixo real positivo
para o raio $ OM $; está definido até $2k\pi $. Escrita $ r = \abs z $,
\[
z = r(\cos\theta + \iu\sin\theta)
\qquad\text{(trigonometric form)}.
\]
Apresentando a notação\index{forma exponencial}
\[
\eu^{\iu\theta} = \cos\theta + \iu\sin\theta,
\]
isso se torna o \emph{forma exponencial} $ z = r\,\eu^{\iu\theta}$.
\end{definition}

\begin{theorem}\label{thm:g12:complex:expform}
Para todos $\theta, \theta' \in \R $:
\[
\eu^{\iu\theta}\,\eu^{\iu\theta'} = \eu^{\iu(\theta + \theta')},
\qquad
\conj{\eu^{\iu\theta}} = \eu^{-\iu\theta},
\qquad
\abs{\eu^{\iu\theta}} = 1 .
\]
Consequentemente, para $ z = r\eu^{\iu\theta}$ e $ z' = r'\eu^{\iu\theta'}$
diferente de zero:
\[
zz' = rr'\,\eu^{\iu(\theta+\theta')},
\qquad
\frac{z}{z'} = \frac{r}{r'}\,\eu^{\iu(\theta - \theta')} :
\]
\emph{\omterm{def:g12:complex:modulus}{módulos} se multiplicam, \omterm{def:g12:complex:argument}{argumentos} adicionar.}
\end{theorem}

\begin{proof}
A primeira identidade \emph{é} o par de fórmulas de adição
(\cref{prop:g12:trigo:addition}):
\[
(\cos\theta + \iu\sin\theta)(\cos\theta' + \iu\sin\theta')
= (\cos\theta\cos\theta' - \sin\theta\sin\theta')
+ \iu(\sin\theta\cos\theta' + \cos\theta\sin\theta').
\]
O resto segue por cálculo direto.
\end{proof}

\begin{corollary}[Fórmula de De Moivre]\label{cor:g12:complex:demoivre}
\index{Fórmula de De Moivre}
Para $\theta \in \R $ e $ n \in \Z $:
$\bigl(\cos\theta + \iu\sin\theta\bigr)^n = \cos n\theta + \iu \sin n\theta $.
\end{corollary}

\begin{proof}
Indução ativada $ n \geq 0$ usando \cref{thm:g12:complex:expform}; estender para
$ n < 0$ tomando inversos.
\end{proof}

\begin{method}[Alternando entre formulários]\label{met:g12:complex:forms}
Do algébrico ao exponencial: calcular $ r = \abs z $, então encontre $\theta $ com
$\cos\theta = \frac ar $, $\sin\theta = \frac br $ (localize o quadrante correto
antes de invocar um inverso \omterm{def:g11:trigo:cossin}{cosseno}). Do exponencial ao algébrico: expandir
$ r\cos\theta + \iu\, r\sin\theta $. Use o \omterm{def:g12:complex:def}{forma algébrica} para somas, o
\omterm{def:g12:complex:argument}{forma exponencial} para produtos, potências e quocientes.
\end{method}

\begin{example}
$ z = 1 + \iu $: $\abs z = \sqrt2$ e $\cos\theta = \sin\theta =
\frac{1}{\sqrt2}$, então $\theta = \frac\pi4$ e $ z = \sqrt2\,\eu^{\iu\pi/4}$.
Por isso $ z^{20} = 2^{10}\,\eu^{5\iu\pi} = -1024$.
\end{example}

\section{Equações de segundo grau e raízes da unidade}

\begin{theorem}[Equações quadráticas com coeficientes reais]\label{thm:g12:complex:quadratic}
Deixar $ a, b, c \in \R $, $ a \neq 0$, e $\Delta = b^2 - 4ac $. O equação
$ az^2 + bz + c = 0$ tem soluções em $\C $:
\[
\begin{aligned}
&\Delta > 0:\ z = \frac{-b \pm \sqrt{\Delta}}{2a} \ (\text{two real}); \qquad
\Delta = 0:\ z = \frac{-b}{2a} \ (\text{double}); \\
&\Delta < 0:\ z = \frac{-b \pm \iu\sqrt{-\Delta}}{2a} \ (\text{two conjugate}).
\end{aligned}
\]
\end{theorem}

\begin{proof}
Complete o quadrado:
$ az^2 + bz + c = a\left(\left(z + \frac{b}{2a}\right)^2 -
\frac{\Delta}{4a^2}\right)$. Se $\Delta < 0$, então
$\frac{\Delta}{4a^2} = \left(\iu\frac{\sqrt{-\Delta}}{2a}\right)^2$, e o
diferença de fatores quadrados como de costume.
\end{proof}

\begin{definition}[Raízes da unidade]\label{def:g12:complex:rootsofunity}
Para $ n \geq 1$, o \emph{$ n $-as raízes da unidade}\index{$ n $-as raízes da unidade} são
as soluções de $ z^n = 1$.
\end{definition}

\begin{theorem}\label{thm:g12:complex:rootsofunity}
O equação $ z^n = 1$ tem exatamente $ n $ soluções:
\[
\omega_k = \eu^{2\iu k\pi/n}, \qquad k = 0, 1, \dots, n-1 .
\]
Deles imagens são os vértices de um regular $ n $-gon inscrito na unidade
círculo, e para $ n \geq 2$ a soma deles é $0$.
\end{theorem}

\begin{proof}
Cada $\omega_k $ satisfaz $\omega_k^n = \eu^{2\iu k\pi} = 1$, e eles são
pares distintos, uma vez que seus \omterm{def:g12:complex:argument}{argumentos} deitar em $\intco{0}{2\pi}$.
Por outro lado, se $ z^n = 1$ então $\abs z^n = 1$, então $\abs z = 1$ (positivo
reais) e $ z = \eu^{\iu\theta}$ com $ n\theta \equiv 0 \pmod{2\pi}$,
\emph{ou seja} $\theta = \frac{2k\pi}{n}$; reduzindo $ k $ \omterm{def:g12:complex:modulus}{módulo} $ n $ terras entre
o $\omega_k $. Os vértices são igualmente espaçados por ângulo $\frac{2\pi}{n}$:
um normal $ n $-Vai. Finalmente, com $\omega = \omega_1$,
$\omega \neq 1$ e a soma geométrica dá
\[
\sum_{k=0}^{n-1}\omega_k = \sum_{k=0}^{n-1}\omega^k
= \frac{\omega^n - 1}{\omega - 1} = 0 . \qedhere
\]
\end{proof}

\end{example}

\section{Números complexos e geometria plana}

\begin{proposition}[Dicionário geométrico]\label{prop:g12:complex:geometry}
Deixar $ A, B, C, D $ ser pontos de \omterm{def:g12:complex:modulus}{afixos} $ a, b, c, d $ ($ a \neq b $,
$ c \neq d $).
\begin{enumerate}
  \item A tradução feita pelo vetor de \omterm{def:g12:complex:modulus}{afixo} $ t $ é $ z \mapsto z + t $.
  \item A rotação do centro $\omega $ (\omterm{def:g12:complex:modulus}{afixo}) e ângulo $\theta $ é
        $ z \mapsto \omega + \eu^{\iu\theta}(z - \omega)$.
  \item A escala (homotetia) do centro $\omega $ e proporção
        $ k \in \R^*$ é $ z \mapsto \omega + k(z - \omega)$.
  \item $\dfrac{d - c}{b - a}$ tem \omterm{def:g12:complex:modulus}{módulo} $\dfrac{CD}{AB}$ e \omterm{def:g12:complex:argument}{argumento} o
        ângulo entre o \omterm{def:g11:vect:vector}{vetores} $\vect{AB}$ e $\vect{CD}$. Em particular, $ AB \perp CD $ se e somente se $\frac{d-c}{b-a}$ é puramente
        imaginário, e $ A, B, C $ são \omterm{def:g11:vect:collinear}{colinear} ($ C \ne A $, $ C\ne B $) se e
        somente se $\frac{c-a}{b-a} \in \R $.
\end{enumerate}
\end{proposition}

\begin{proof}
(1) e (3) reformulam as fórmulas de \omterm{def:g10:coordgeom:system}{coordenadas}. Para (2), o mapa
$ w \mapsto \eu^{\iu\theta} w $ multiplica \omterm{def:g12:complex:modulus}{módulos} por $1$ e adiciona $\theta $ para
\omterm{def:g12:complex:argument}{argumentos}: é a rotação do ângulo $\theta $ sobre a origem; conjugando
pelo envio da tradução $\omega $ para $0$ dá o centro geral. Para
(4), \omterm{def:g12:complex:modulus}{módulos} e \omterm{def:g12:complex:argument}{argumentos} de quocientes subtrair
(\cref{thm:g12:complex:expform}).
\end{proof}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g12:complex:1}
Escreva em \omterm{def:g12:complex:def}{forma algébrica}:
$(3 - 2\iu)(1 + 4\iu)$, \quad $\dfrac{2 + \iu}{1 - \iu}$, \quad
$\iu^{2027}$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g12:complex:2}
Resolva em $\C $: (a) $ z^2 - 4z + 13 = 0$; (b) $ z^2 + z + 1 = 0$. Confira em cada
caso as duas soluções sejam \omterm{def:g12:complex:conjugate}{conjugados} e calcule seu produto.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g12:complex:3}
Colocar \omterm{def:g12:complex:argument}{forma exponencial}: $ z_1 = -1 + \iu $, $ z_2 = \sqrt3 - \iu $ e calcular
$\dfrac{z_1}{z_2}$ em exponencial então \omterm{def:g12:complex:def}{forma algébrica}. Deduza o exato
valor de $\cos\dfrac{11\pi}{12}$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:complex:4}
Descreva geometricamente o conjunto de pontos $ M $ de \omterm{def:g12:complex:modulus}{afixo} $ z $ tal que:
(a) $\abs{z - 2} = \abs{z + \iu}$;\quad
(b) $\abs{z - 1 - \iu} = 3$;\quad
(c) $\dfrac{z - 1}{z + 1}$ é puramente imaginário.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:complex:5}
Usando a fórmula de De Moivre e o teorema binomial, expresse $\cos 3\theta $
como um polinômio em $\cos\theta $, e $\sin 3\theta $ em termos de
$\sin\theta $.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:complex:6}
\emph{(Linearização.)} Usando
$\cos\theta = \frac{\eu^{\iu\theta} + \eu^{-\iu\theta}}{2}$, linearizar
$\cos^3\theta $ (escreva-o como uma combinação de $\cos k\theta $) e deduza
$\displaystyle\int_0^{\pi/2}\cos^3\theta\,\dd\theta $.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:complex:7}
Deixar $ A $ e $ B $ ter \omterm{def:g12:complex:modulus}{afixos} $ a = 1 + \iu $ e $ b = 3 + 2\iu $. Determine o
dois pontos $ C $ fazendo o triângulo $ ABC $ equilátero, como o imagens de $ B $
sob as rotações do centro $ A $ e ângulos $\pm\frac{\pi}{3}$.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g12:complex:8}
Resolver $ z^4 = -4$ em $\C $ e traçar as soluções. Fator
$ z^4 + 4$ em dois polinômios quadráticos com coeficientes reais.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g12:complex:9}
Para $\theta \in \intoo{0}{2\pi}$ e $ n \in \N $, calcular
\[
S = 1 + \cos\theta + \cos 2\theta + \dots + \cos n\theta
\]
somando a progressão geométrica
$\sum_{k=0}^n \eu^{\iu k\theta}$ e tomando peças reais. (Responder:
$ S = \dfrac{\sin\frac{(n+1)\theta}{2}}{\sin\frac{\theta}{2}}
\cos\frac{n\theta}{2}$.)
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g12:complex:10}
Deixar $\omega = \eu^{2\iu\pi/5}$.
\begin{enumerate}
  \item Justifique isso $1 + \omega + \omega^2 + \omega^3 + \omega^4 = 0$.
  \item Deixar $ u = \omega + \omega^4$ e $ v = \omega^2 + \omega^3$. Mostre isso
        $ u + v = -1$ e $ uv = -1$, e deduza que $ u = \frac{-1+\sqrt5}{2}$.
  \item Conclua isso $\cos\dfrac{2\pi}{5} = \dfrac{\sqrt5 - 1}{4}$.
\end{enumerate}
\end{exercise}

\section{Problema: O pentágono no espelho da unidade}

\begin{problem}[{Problema de fim de semana --- as quintas raízes da unidade
calcular $\cos 72^\circ $ exatamente, a proporção áurea assina o
resultado, e a multiplicação acaba sendo rotação}]
\label{pb:g12:complex:1}
Euclides poderia construir o pentágono regular, mas a razão
\emph{por que} cede ao governante e ao compasso --- enquanto o humilde
$20^\circ $ ângulo não --- ficou escondido por dois mil
anos. Esconde-se neste capítulo: o quinto quinto raízes da unidade
(\cref{thm:g12:complex:rootsofunity}) soma zero, e aquele
linha de compressão de álgebra $\cos 72^\circ $ fora de uma quadrática
equação --- apenas raízes quadradas, portanto construtíveis e assinadas,
claro, pela proporção áurea. Em torno desta joia: fluência, de
O atalho de Moivre para a trigonometria e a multiplicação desmascarada como
geometria.

\medskip
\noindent\textbf{Part I --- Fluency.}
\begin{enumerate}
  \item Calcular $(2 + \iu)(3 - \iu)$, \
        $\dfrac{1 + \iu}{1 - \iu}$
        (\cref{met:g12:complex:division}), e
        $\abs{3 + 4\iu}$.
  \item Colocar \omterm{def:g12:complex:argument}{forma exponencial}: $1 + \iu $; \ $-2$; \
        $1 - \iu\sqrt3$.
  \item Calcular $(1 + \iu)^8$ usando o \omterm{def:g12:complex:argument}{forma exponencial}.
  \item Resolver $ z^2 = \iu $, então $ z^2 - 2z + 5 = 0$
        (\cref{thm:g12:complex:quadratic}).
  \item Dicionário \omterm{def:g12:seq:arith-geom}{geométrico}
        (\cref{prop:g12:complex:geometry}): descreva o mapa
        $ z \mapsto \iu z $, e o conjunto
        $\abs{z - (1 + \iu)} = 2$.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Five raízes, one pentagon.}
Deixar $\omega = \eu^{2\iu\pi/5}$ e considere
$1, \omega, \omega^2, \omega^3, \omega^4$.
\begin{enumerate}[resume]
  \item Verifique se essas são exatamente as soluções de $ z^5 = 1$,
        e descreva a figura que eles desenham no plano.
  \item Prove que a soma deles é zero. (Progressão geométrica
        --- ou multiplique a soma por $1 - \omega $.)
  \item Pegar peças reais e deduzir
        \[
        1 + 2\cos\frac{2\pi}{5} + 2\cos\frac{4\pi}{5} = 0 .
        \]
  \item Definir $ c = \cos\frac{2\pi}{5}$ e use o ângulo duplo
        fórmula para $\cos\frac{4\pi}{5}$ para converter a pergunta 8
        em um quadrático equação para $ c $; resolva isso e
        concluir
        \[
        \cos 72^\circ = \frac{\sqrt5 - 1}{4} .
        \]
  \item Verifique numericamente e deixe a proporção áurea assinar seu
        trabalho: mostrar $ c = \frac{1}{2\varphi}$ onde
        $\varphi = \frac{1 + \sqrt5}{2}$
        (\cref{pb:g10:algebra:1}) --- e indique o famoso
        eco \omterm{def:g12:seq:arith-geom}{geométrico}: em um pentágono regular, a diagonal é
        $\varphi $ vezes o lado.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- De Moivre's shortcuts.}
\begin{enumerate}[resume]
  \item O veredicto da construção: $\cos 72^\circ $ precisa apenas
        raízes quadradas (questão 9), então o pentágono é
        construtível com régua e compasso; $\cos 20^\circ $
        satisfaz uma cúbica inquebrável
        (\cref{pb:g12:trigo:1}), então o $60^\circ $ ângulo não pode
        ser trissectado. Indique o critério emergente (quadrado
        raízes edificáveis, raízes cúbicas não), cuja teoria completa ---
        Gauss, 19 anos e 17 anos --- mora no
        volumes universitários.
  \item Resolver $ z^4 = -4$ por \omterm{def:g12:complex:argument}{forma exponencial} e recuperar
        \cref{exo:g12:complex:8}fatoração de
        $ z^4 + 4 = (z^2 - 2z + 2)(z^2 + 2z + 2)$ por emparelhamento
        \omterm{def:g12:complex:conjugate}{conjugado} raízes.
  \item Expandir $(\cos\theta + \iu\sin\theta)^3$ com o
        teorema binomial e de Moivre, e derivar novamente
        $\cos 3\theta = 4\cos^3\theta - 3\cos\theta $ em três
        linhas --- compare com a rota da fórmula de adição de
        \cref{pb:g12:trigo:1}.
  \item Verifique com \cref{exo:g12:complex:9}: para
        $\theta = \frac{2\pi}{5}$ e $ n = 4$, o que o
        soma $1 + \cos\theta + \dots + \cos 4\theta $ igual, e
        por que isso é consistente com a pergunta 8?
  \item Calcule as três raízes cúbicas de $8\iu $ em algébrico
        formulário.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Multiplication is geometry.}
\begin{enumerate}[resume]
  \item Descreva completamente o mapa $ z \mapsto (1 + \iu)z $:
        em que ângulo ele gira, em que fator ele gira
        escala, e o que isso faz com o quadrado unitário?
  \item Provar $\abs{z_1 z_2} = \abs{z_1}\,\abs{z_2}$ com
        \omterm{def:g12:complex:conjugate}{conjugados} (\cref{prop:g12:complex:modulus}), então
        calcular $(1+\iu)^n $ para $ n = 1, 2, 3, 4$ e descrever
        a espiral que os poderes traçam.
  \item O número $\mathrm{j} = \eu^{2\iu\pi/3}$: mostrar
        $1 + \mathrm{j} + \mathrm{j}^2 = 0$, e verifique no triângulo
        $\left(1, \mathrm{j}, \mathrm{j}^2\right)$ o critério clássico:
        $ a + \mathrm{j} b + \mathrm{j}^2 c = 0$ vale para (uma orientação
        de) um triângulo equilátero $ abc $.
  \item O teorema fundamental da álgebra (admitido:
        d'Alembert--Gauss): todos os fatores polinomiais completamente
        acabou $\C $. Deduza o corolário do mundo real --- cada
        fatores polinomiais reais em lineares e quadráticos reais
        peças --- e aponte para a questão 12 como um exemplo.
  \item Final --- $\C$ retrato de, uma frase cada: números
        tornam-se pontos; multiplicação torna-se
        rotação e escala; o raízes da unidade tornar-se regular
        polígonos; de Moivre transforma potências em trigonométricas
        identidades; e o fechamento algébrico garante que não
        equação sempre precisa de um mundo maior. Coda: quais dois
        números célebres encontrados na questão 10?
\end{enumerate}
\end{problem}
