\chapter{Somas de variáveis aleatórias e a lei dos grandes números}\label{ch:g12:sums}

Porque é que os casinos ganham sempre no final e porque é que as sondagens funcionam? Porque
médias de muitos \omterm{def:g12:sums:indep}{independente} quantidades aleatórias flutuam cada vez menos. Isto
capítulo prova isso: linearidade de \omterm{def:g12:randvar:exp}{expectativa}, aditividade de \omterm{def:g12:randvar:exp}{variação} para
\omterm{def:g12:sums:indep}{independente} variáveis, a desigualdade de Bienaymé-Chebyshev e a lei de
grandes números.

\section{Somas de variáveis aleatórias}

Dados dois variáveis aleatórias $X, Y $ no mesmo finito \omterm{def:g10:proba:events}{espaço amostral}
$\Omega $, a soma $ X + Y $ é o variável aleatória
$\omega \mapsto X(\omega) + Y(\omega)$.

\begin{theorem}[Linearidade de expectativa]\label{thm:g12:sums:linearity}
\index{expectativa!linearidade}
Para todos variáveis aleatórias $ X, Y $ sobre $\Omega $ e $ a, b \in \R $:
\[
\E(X + Y) = \E(X) + \E(Y), \qquad \E(aX + b) = a\E(X) + b .
\]
\emph{Não independência é necessária uma suposição.}
\end{theorem}

\begin{proof}
Escreva o \omterm{def:g12:randvar:exp}{expectativa} como uma soma dos resultados: uma vez que
$\P(X = x) = \sum_{\omega : X(\omega) = x} \P(\{\omega\})$, agrupando termos
dá $\E(X) = \sum_{\omega \in \Omega} \P(\{\omega\})\,X(\omega)$. Então
\[
\E(X + Y) = \sum_{\omega} \P(\{\omega\})\bigl(X(\omega) + Y(\omega)\bigr)
= \sum_{\omega} \P(\{\omega\})X(\omega)
+ \sum_{\omega} \P(\{\omega\})Y(\omega) = \E(X) + \E(Y). \qedhere
\]
\end{proof}

\begin{definition}[Variáveis aleatórias independentes]\label{def:g12:sums:indep}
$ X $ e $ Y $ são \emph{independente}\index{independente} se
para todos os valores $ x, y $:
\[
\P(X = x \text{ and } Y = y) = \P(X = x)\,\P(Y = y).
\]
Várias variáveis $ X_1, \dots, X_n $ são independentes se esta regra do produto
vale para todas as escolhas de valores de cada subfamília.
\end{definition}

\begin{proposition}\label{prop:g12:sums:prodexp}
Se $ X $ e $ Y $ são \omterm{def:g12:sums:indep}{independente}, $\E(XY) = \E(X)\,\E(Y)$.
\end{proposition}

\begin{proof}
\[
\begin{aligned}
\E(XY) &= \sum_{x, y} xy\;\P(X = x \text{ and } Y = y)
= \sum_{x, y} xy\,\P(X=x)\P(Y=y) \\
&= \Bigl(\sum_x x\P(X=x)\Bigr)\Bigl(\sum_y y\P(Y=y)\Bigr). \qedhere
\end{aligned}
\]
\end{proof}

\begin{theorem}[Variância de uma soma]\label{thm:g12:sums:variance}
\index{variação!de uma soma}
Se $ X $ e $ Y $ são \emph{\omterm{def:g12:sums:indep}{independente}}, então
\[
\V(X + Y) = \V(X) + \V(Y).
\]
Mais genericamente, para \omterm{def:g12:sums:indep}{independente} $ X_1, \dots, X_n $:
$\V(X_1 + \dots + X_n) = \V(X_1) + \dots + \V(X_n)$.
\end{theorem}

\begin{proof}
Usando König--Huygens (\cref{prop:g12:randvar:konig}) e linearidade:
\begin{align*}
\V(X + Y) &= \E\bigl((X+Y)^2\bigr) - \bigl(\E X + \E Y\bigr)^2\\
&= \E(X^2) + 2\E(XY) + \E(Y^2)
- \E(X)^2 - 2\E(X)\E(Y) - \E(Y)^2\\
&= \V(X) + \V(Y) + 2\bigl(\E(XY) - \E(X)\E(Y)\bigr),
\end{align*}
e o último colchete desaparece para \omterm{def:g12:sums:indep}{independente} variáveis
(\cref{prop:g12:sums:prodexp}). O caso geral segue por indução.
\end{proof}

\begin{example}[Binômio revisitado]\label{ex:g12:sums:binomial}
Uma variável binomial $ X \sim \mathcal B(n, p)$ é uma soma
$ X = X_1 + \dots + X_n $ de $ n $ \omterm{def:g12:sums:indep}{independente} Variáveis de Bernoulli. Portanto,
estruturalmente:
\[
\E(X) = np, \qquad \V(X) = np(1-p),
\]
recuperando \cref{thm:g12:randvar:binomial} sem nenhum cálculo.
\end{example}

\begin{proposition}[Média amostral]\label{prop:g12:sums:mean}
Deixar $ X_1, \dots, X_n $ ser \omterm{def:g12:sums:indep}{independente} com o mesmo \omterm{def:g12:randvar:rv}{distribuição} como $ X $
(\omterm{def:g12:randvar:exp}{expectativa} $\mu $, \omterm{def:g12:randvar:exp}{variação} $\sigma^2$) e deixe
$ M_n = \frac{X_1 + \dots + X_n}{n}$ seja seu \emph{amostra
significa}\index{amostra
significa}. Então
\[
\E(M_n) = \mu, \qquad \V(M_n) = \frac{\sigma^2}{n}, \qquad
\sigma(M_n) = \frac{\sigma}{\sqrt n}.
\]
\end{proposition}

\begin{proof}
Linearidade dá $\E(M_n) = \frac{n\mu}{n} = \mu $; independência dá
$\V(X_1 + \dots + X_n) = n\sigma^2$, e dividindo por $ n $ dimensiona o
\omterm{def:g12:randvar:exp}{variação} por $\frac{1}{n^2}$ (\cref{prop:g12:randvar:affine}).
\end{proof}

O $\frac{\sigma}{\sqrt n}$ é o fundamental \emph{lei da \omterm{def:g11:quad:discriminant}{raiz} quadrada}: para
reduzir pela metade as flutuações de uma média, quadruplicar o \omterm{def:g10:proba:sample}{amostra} tamanho.

\section{Desigualdades de concentração}

\begin{theorem}[Desigualdade de Markov]\label{thm:g12:sums:markov}
\index{Desigualdade de Markov}
Se $ X \geq 0$ e $ a > 0$:
\[
\P(X \geq a) \leq \frac{\E(X)}{a}.
\]
\end{theorem}

\begin{proof}
Em $\E(X) = \sum_i p_i x_i $ (todos os termos não negativos), mantenha apenas os termos
com $ x_i \geq a $: cada um é pelo menos $ a\,p_i $, então
$\E(X) \geq a \sum_{x_i \geq a} p_i = a\,\P(X \geq a)$.
\end{proof}

\begin{theorem}[Desigualdade de Bienaymé-Chebyshev]\label{thm:g12:sums:chebyshev}
\index{Desigualdade de Bienaymé-Chebyshev}
Para qualquer variável aleatória $ X $ e qualquer $\varepsilon > 0$:
\[
\P\bigl(\abs{X - \E(X)} \geq \varepsilon\bigr)
\leq \frac{\V(X)}{\varepsilon^2}.
\]
\end{theorem}

\begin{proof}
Aplique a desigualdade de Markov à variável não negativa
$ Y = (X - \E(X))^2$ com $ a = \varepsilon^2$:
\[
\P\bigl(\abs{X - \E(X)} \geq \varepsilon\bigr)
= \P(Y \geq \varepsilon^2)
\leq \frac{\E(Y)}{\varepsilon^2} = \frac{\V(X)}{\varepsilon^2}. \qedhere
\]
\end{proof}

\begin{theorem}[Lei dos grandes números]\label{thm:g12:sums:lln}
\index{lei dos grandes números}
Deixar $ X_1, \dots, X_n $ ser \omterm{def:g12:sums:indep}{independente} com o mesmo \omterm{def:g12:randvar:rv}{distribuição}
(\omterm{def:g12:randvar:exp}{expectativa} $\mu $, \omterm{def:g12:randvar:exp}{variação} $\sigma^2$) e $ M_n $ deles média amostral. Para
cada $\varepsilon > 0$:
\[
\P\bigl(\abs{M_n - \mu} \geq \varepsilon\bigr)
\leq \frac{\sigma^2}{n\,\varepsilon^2}
\xrightarrow[n \to +\infty]{} 0 .
\]
O média amostral concentra-se em torno do \omterm{def:g12:randvar:exp}{expectativa}.
\end{theorem}

\begin{proof}
Bienaymé--Chebyshev solicitou $ M_n $, cujo \omterm{def:g12:randvar:exp}{expectativa} é $\mu $ e cujo
\omterm{def:g12:randvar:exp}{variação} é $\frac{\sigma^2}{n}$ (\cref{prop:g12:sums:mean}).
\end{proof}

\begin{remark}
Este teorema é a ponte entre \omterm{def:g10:proba:distribution}{probabilidade} teoria e estatística: o
\emph{freqüência} de um \omterm{def:g11:prob:model}{evento} ao longo de muitos \omterm{def:g12:sums:indep}{independente} abordagens de repetições
é \emph{\omterm{def:g10:proba:distribution}{probabilidade}} (pegar $ X_i $ o indicador do \omterm{def:g11:prob:model}{evento}, para que
$\mu = p $). Isso justifica estimar um \omterm{def:g10:proba:distribution}{probabilidade} por simulação
(\emph{Método Monte Carlo}) e uma proporção da população por uma \omterm{def:g10:algebra:equation}{pesquisa} ---
quantitativamente: veja \cref{ch:g12:contdist}.
\end{remark}

\begin{method}[Usando Bienaymé-Chebyshev]\label{met:g12:sums:chebyshev}
Para garantir $\P(\abs{M_n - \mu} \geq \varepsilon) \leq \alpha $, isso
basta que $ n \geq \frac{\sigma^2}{\alpha\,\varepsilon^2}$. O limite é
bruto (as flutuações reais são geralmente muito menores), mas perfeitamente geral:
não requer nada sobre o \omterm{def:g12:randvar:rv}{distribuição} além de um finito \omterm{def:g12:randvar:exp}{variação}.
\end{method}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g12:sums:1}
Dois dados justos são lançados; deixar $ S $ seja a soma. Usando linearidade (não o
\omterm{def:g12:randvar:rv}{distribuição} de $ S $!), calcular $\E(S)$; então calcule $\V(S)$ usando
independência, dado que um único dado justo tem \omterm{def:g12:randvar:exp}{variação} $\frac{35}{12}$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g12:sums:2}
Deixar $ X \sim \mathcal B(100,\ 0.5)$. Vinculado
$\P(X \geq 75)$ usando a desigualdade de Markov, então
$\P(\abs{X - 50} \geq 25)$ usando Bienaymé-Chebyshev. Comparar.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g12:sums:3}
Uma moeda justa é lançada $ n $ vezes e $ F_n $ denota o freqüência de cabeças.
Qual deve ser o tamanho $ n $ seja para que, pela ligação Bienaymé-Chebyshev,
$\P\bigl(\abs{F_n - 0.5} \geq 0.05\bigr) \leq 0.05$?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:sums:4}
Um investidor \omterm{def:g12:arith:divides}{divide} seu capital igualmente entre $ n $ \omterm{def:g12:sums:indep}{independente} ativos,
cada um com retorno esperado $\mu = 5\%$ e \omterm{def:g11:stat:variance}{desvio padrão}
$\sigma = 20\%$. Calcule o \omterm{def:g12:randvar:exp}{expectativa} e \omterm{def:g11:stat:variance}{desvio padrão} do
retorno do portfólio $ M_n $, e o número de ativos necessários para trazer o
\omterm{def:g11:stat:variance}{desvio padrão} abaixo $4\%$. Que princípio financeiro faz isso
ilustrar?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:sums:5}
Deixar $ X $ e $ Y $ ser \omterm{def:g12:sums:indep}{independente}, ambos uniformes em $\{1, 2, 3\}$.
\begin{enumerate}
  \item Dê o \omterm{def:g12:randvar:rv}{distribuição} de $ S = X + Y $ e calcular $\E(S)$, $\V(S)$
        diretamente dele.
  \item Recupere ambos os valores por linearidade e aditividade de \omterm{def:g12:randvar:exp}{variação}.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:sums:6}
Mostre que a aditividade de \omterm{def:g12:randvar:exp}{variação} pode falhar sem independência:
calcular $\V(X + Y)$ para $ Y = X $, e compare com $\V(X) + \V(Y)$. Para
quais variáveis $ X $ é $\V(2X) = 2\V(X)$?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:sums:7}
Suspeita-se que um dado esteja carregado. Está enrolado $1200$ vezes e mostra um
seis $260$ vezes ($ f = 0.2167$ em vez de $\frac16 \approx 0.1667$). Abaixo
a hipótese de que o dado é justo, vinculado
$\P\bigl(\abs{F_n - \frac16} \geq 0.05\bigr)$ por Bienaymé--Chebyshev, e
discuta se a hipótese de justiça é plausível.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g12:sums:8}
\emph{(Uma desigualdade \omterm{def:g12:arith:prime}{melhor} para a moeda.)} Deixar
$ X \sim \mathcal B(n,\ p)$ e $ F_n = \frac Xn $.
\begin{enumerate}
  \item Mostre isso $ p(1-p) \leq \frac14$ para $ p \in \intcc{0}{1}$.
  \item Deduza o limite livre de \omterm{def:g12:randvar:rv}{distribuição}
        $\P\bigl(\abs{F_n - p} \geq \varepsilon\bigr) \leq
        \frac{1}{4n\varepsilon^2}$.
  \item Quantas pessoas devem ser entrevistadas para que o observado freqüência é
        dentro $3$ pontos da verdadeira proporção com \omterm{def:g10:proba:distribution}{probabilidade} pelo menos
        $95\%$, usando este limite? (As \omterm{def:g10:algebra:equation}{pesquisas} reais usam estimativas mais precisas, mas
        a ordem de grandeza está correta.)
\end{enumerate}
\end{exercise}

\section{Problema: A casa sempre ganha}

\begin{problem}[{Problema de fim de semana --- a lei dos grandes números
explica cassinos, pesquisas e seguros, e destrói o
falácia do jogador a caminho}]
\label{pb:g12:sums:1}
Um jogador de roleta depois $100$ gira é, quase sempre,
\emph{à frente}. O cassino com um milhão de rodadas está à frente com um
certeza que nenhum tribunal questionaria. O mesmo jogo, a mesma pequena vantagem ---
a diferença é $\sqrt n $, e é o assunto deste
capítulo (\cref{prop:g12:sums:mean},
\cref{thm:g12:sums:chebyshev}, \cref{thm:g12:sums:lln}). Isto
problema controla os livros do cassino, avalia uma \omterm{def:g10:algebra:equation}{pesquisa} eleitoral, preços
diversificação --- e desmantela a falácia mais cara do
a história do jogo.

\medskip
\noindent\textbf{Part I --- Fluency.}
\begin{enumerate}
  \item Dois \omterm{def:g12:sums:indep}{independente} dados: calcular $ V(X + Y)$
        (\cref{thm:g12:sums:variance}).
  \item Rolar $100$ \omterm{def:g12:sums:indep}{independente} dados e calcule a média dos resultados:
        dê o \omterm{def:g12:randvar:exp}{expectativa} e o \omterm{def:g11:stat:variance}{desvio padrão} do
        média amostral (\cref{prop:g12:sums:mean}; por um dado,
        $\sigma = \sqrt{35/12} \approx 1.71$).
  \item Um não negativo variável aleatória tem \omterm{def:g12:randvar:exp}{expectativa} $2$.
        O que significa a desigualdade de Markov
        (\cref{thm:g12:sums:markov}) dizer sobre
        $\P(X \geq 10)$?
  \item Vinculado $\P\left(\abs{\bar X_{100} - 3.5} \geq
        0.5\right)$ para a média dos dados da questão 2 com
        Bienaymé-Chebyshev.
  \item \omterm{def:g12:randvar:exp}{Expectativas} adicione sempre, variações apenas sob
        independência: exibição dependente $ X, Y $ (dica:
        $ Y = -X $) para o qual
        $ V(X + Y) \neq V(X) + V(Y)$.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The house's books.}
Roleta Europeia, aposta $1$ euro no vermelho: vitória $+1$ com
\omterm{def:g10:proba:distribution}{probabilidade} $\frac{18}{37}$, perder $-1$ com \omterm{def:g10:proba:distribution}{probabilidade}
$\frac{19}{37}$.
\begin{enumerate}[resume]
  \item Calcule o \omterm{def:g12:randvar:exp}{expectativa} e \omterm{def:g11:stat:variance}{desvio padrão} de um
        ganho da aposta.
  \item Um jogador faz $100$ \omterm{def:g12:sums:indep}{independente} apostas; deixar $ G $ seja o
        ganho total. Calcular $\E(G)$ e $\sigma(G)$, então o
        proporção $\frac{\abs{\E(G)}}{\sigma(G)}$. O que um
        desvio de um quarto de \omterm{def:g11:stat:variance}{desvio padrão} significar para o
        as \omterm{def:g11:func:parity}{chances} do jogador de estar à frente esta noite?
  \item O cassino vê $1\,000\,000$ apostas. Calcule o
        \omterm{def:g12:randvar:exp}{expectativa} e \omterm{def:g11:stat:variance}{desvio padrão} da sua tomada total,
        e a mesma proporção. Interprete o número $27$.
  \item Certificar com Chebyshev: vincular o \omterm{def:g10:proba:distribution}{probabilidade} que o
        cassino \emph{perde} dinheiro acima do milhão de apostas.
  \item Sistemas de apostas: dobrar após perdas, desistir quando
        à frente\,\dots\ Usando a linearidade de \omterm{def:g12:randvar:exp}{expectativa}
        (\cref{thm:g12:sums:linearity}) sobre o (possivelmente
        aleatório) \omterm{def:g12:seq:sequence}{sequência} de apostas, explique por que \emph{todo}
        estratégia em um jogo com vantagem negativa tem \omterm{def:g12:randvar:exp}{expectativa} negativa
        ganho --- o que um sistema vencedor violaria?
  \item Explique o que é a lei dos grandes números
        (\cref{thm:g12:sums:lln}) promessas sobre o freqüência
        de vermelho --- e o que ele faz \emph{não} prometer sobre
        a próxima rodada depois de dez vermelhos seguidos. Nomeie o
        falácia.
  \item O ponto mais sutil: a lei funciona por \emph{diluição},
        não compensação. Se os vermelhos correrem $100$ à frente de
        \omterm{def:g12:randvar:exp}{expectativa} depois de alguma noite, o excedente nunca é
        ``reembolsado'' --- calcule o que acontece com o
        \emph{fração} $\frac{100}{n}$ por $ n = 10^6$ e
        reescreva o erro da falácia do jogador em uma frase.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Polls.}
\begin{enumerate}[resume]
  \item De \cref{exo:g12:sums:8}: quantas pessoas devem ser
        pesquisado para o observado freqüência cair dentro $3$
        pontos da verdade com \omterm{def:g10:proba:distribution}{probabilidade} pelo menos
        $95\,\%$, pelo limite de Chebyshev sem \omterm{def:g12:randvar:rv}{distribuição}?
  \item Os verdadeiros institutos de votação usam cerca de $1\,100$ pessoas para
        um $\pm 3$ margem de ponto em $95\,\%$: sua fórmula é
        $ n \approx \frac{1.96^2 \times p(1-p)}
        {\varepsilon^2}$ com $ p(1-p) \leq \frac14$. Avaliar
        e explique a lacuna com a questão 13 (o que
        Chebyshev não sabe sobre a forma do
        flutuações?).
  \item Para reduzir pela metade a margem de erro, como deve o \omterm{def:g10:proba:sample}{amostra}
        crescer? De $\pm 3$ pontos em $1\,100$, o que \omterm{def:g10:proba:sample}{amostra}
        dá $\pm 1.5$ pontos?
  \item O clássico contra-intuitivo: o necessário \omterm{def:g10:proba:sample}{amostra} tamanho
        nunca usei o tamanho da população --- $1\,100$ pessoas
        basta para uma cidade ou um continente. Aponte para o lugar
        no modelo onde o tamanho da população está ausente, e
        faça a analogia da cozinha que os pesquisadores usam.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Diversification.}
\begin{enumerate}[resume]
  \item A seguradora de \cref{pb:g12:randvar:1} ganha
        $30n $ euros em \omterm{def:g12:randvar:exp}{expectativa} sobre $ n $ políticas com um
        reivindicações \omterm{def:g11:stat:variance}{desvio padrão} de cerca de $315\sqrt n $. Para
        qual $ n $ o lucro esperado finalmente excede um
        \omterm{def:g11:stat:variance}{desvio padrão} das reivindicações? Qual é o
        $\sqrt n $ fazendo para a seguradora?
  \item O retorno anual de uma ação flutua com
        $\sigma = 20\,\%$. Divida o dinheiro igualmente $25$
        \emph{\omterm{def:g12:sums:indep}{independente}} tais ações: calcule o
        portfólio $\sigma $. Finanças chama diversificação
        o único almoço grátis --- qual é o almoço, exatamente?
  \item Agora deixe o $25$ ações sejam perfeitamente correlacionadas (eles
        todos se movem juntos): qual é o valor do portfólio $\sigma $?
        Compare os dois extremos e indique qual suposição
        every diversification \omterm{def:g12:complex:argument}{argumento} aluga secretamente ---
        e o que aconteceu quando falhou em todo o sistema em 2008.
  \item Final --- o $\sqrt n $ sinfonia: as somas flutuam como
        $\sqrt n $ enquanto seus meios se estabelecem como
        $\frac{1}{\sqrt n}$; toque o tema através dos quatro
        indústrias deste problema (casino, \omterm{def:g10:algebra:equation}{pesquisas}, seguros,
        carteiras), nomeie os dois abusadores permanentes (o
        falácia e falácia do jogador independência) e dê o
        ponteiro para frente: o \emph{forma} das flutuações
        --- o sino --- é o próximo capítulo contínuo
        estrela, e seu teorema completo coroa a universidade
        volumes.
\end{enumerate}
\end{problem}
