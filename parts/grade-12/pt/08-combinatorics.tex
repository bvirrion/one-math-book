\chapter{Combinatória e Contagem}\label{ch:g12:comb}

Combinatória é a arte de contar sem listar. Seus dois elementares
princípios --- some os tamanhos das alternativas disjuntas, multiplique os números
de escolhas independentes --- basta contar o \omterm{def:g12:comb:tuples}{arranjos}, permutações
e subconjuntos de um conjunto finito e culminam no teorema binomial.

\section{Os dois princípios de contagem}

Nós escrevemos $\abs{E}$ para o número de elementos (o \emph{cardinalidade}) de um
conjunto finito $ E $.

\begin{proposition}[Princípio de adição]\label{prop:g12:comb:addition}
Se um conjunto finito $ E $ é particionado em subconjuntos $ A_1, \dots, A_k $ (em pares
disjunto, com \omterm{def:g10:numbers:interunion}{união} $ E $), então
\[
\abs{E} = \abs{A_1} + \abs{A_2} + \dots + \abs{A_k}.
\]
\end{proposition}

\begin{proposition}[Princípio de multiplicação]\label{prop:g12:comb:multiplication}
Se um objeto é construído por uma sucessão de $ k $ escolhas, com $ n_1$ opções para
a primeira escolha e, \emph{quaisquer que sejam as escolhas anteriores}, $ n_i $ opções para
o $ i $-th, então o número de objetos construídos é
$ n_1 \times n_2 \times \dots \times n_k $.
\end{proposition}

\begin{proof}
Ambas as afirmações são provadas por indução em $ k $; o caso $ k = 2$ do
o segundo equivale a contar uma matriz retangular por linhas.
\end{proof}

\begin{example}
Um restaurante oferece 4 entradas, 6 pratos principais e 3 sobremesas: $4 \times 6 \times 3 =
72$ diferentes refeições de três pratos.
\end{example}

\section{Tuplas, permutações, fatoriais}

\begin{definition}[$ k $-tuplas]\label{def:g12:comb:tuples}
UM \emph{$ k $-tupla} de um conjunto $ E $ é uma lista ordenada $(x_1, \dots, x_k)$ de
elementos de $ E $, repetições permitidas. UM $ k $-tupla de \emph{distinto}
elementos é um \emph{arranjo}\index{arranjo} de $ k $ elementos de $ E $.
\end{definition}

\begin{proposition}\label{prop:g12:comb:tuples}
Deixar $\abs E = n $. O número de $ k $-tuplas de $ E $ é $ n^k $. O número de
\omterm{def:g12:comb:tuples}{arranjos} de $ k $ elementos de $ E $ ($0 \leq k \leq n $) é
\[
n(n-1)(n-2)\cdots(n-k+1) = \frac{n!}{(n-k)!},
\]
onde $ n! = 1 \times 2 \times \dots \times n $ (e $0! = 1$) é o
\emph{fatorial}\index{fatorial} de $ n $.
\end{proposition}

\begin{proof}
Princípio da multiplicação: para um $ k $-tupla existem $ n $ opções em cada um
o $ k $ passos; por um \omterm{def:g12:comb:tuples}{arranjo}, $ n $ opções para $ x_1$, então $ n - 1$ para
$ x_2$ (um elemento é usado), \dots, $ n - k + 1$ para $ x_k $.
\end{proof}

\begin{definition}[Permutação]\label{def:g12:comb:permutation}
UM \emph{permutação}\index{permutação} de $ E $ é um \omterm{def:g12:comb:tuples}{arranjo} de tudo $ n $
elementos de $ E $: uma ordem de $ E $. Por
\cref{prop:g12:comb:tuples} (caso $ k = n $), o número de permutações de um
$ n $-conjunto de elementos é $ n!$.
\end{definition}

\begin{example}
Cinco corredores podem terminar uma corrida em $5! = 120$ ordens diferentes. O número de
possíveis pódios (três primeiros lugares) é $5 \times 4 \times 3 = 60$.
\end{example}

\section{Combinações e coeficientes binomiais}

\begin{definition}[Combinações]\label{def:g12:comb:combination}
UM \emph{combinação}\index{combinação} de $ k $ elementos de $ E $ é um subconjunto
de $ E $ com $ k $ elementos (sem ordem, sem repetição). O número deles está escrito
$\dbinom{n}{k}$\index{coeficiente binomial}, leia ``$ n $ escolher $ k $''.
\end{definition}

\begin{theorem}\label{thm:g12:comb:binomformula}
Para $0 \leq k \leq n $:
\[
\binom{n}{k} = \frac{n!}{k!\,(n-k)!} .
\]
\end{theorem}

\begin{proof}
Conte o \omterm{def:g12:comb:tuples}{arranjos} de $ k $ elementos de $ E $ de duas maneiras. Diretamente:
$\frac{n!}{(n-k)!}$. Alternativamente, escolha primeiro o subconjunto subjacente
($\binom nk $ maneiras), então faça o pedido ($ k!$ caminhos); o princípio da multiplicação
dá $\binom{n}{k}\,k!$. Igualando,
$\binom nk = \frac{n!}{k!(n-k)!}$.
\end{proof}

\begin{proposition}[Identidades básicas]\label{prop:g12:comb:identities}
Para $0 \leq k \leq n $:
\[
\binom{n}{0} = \binom{n}{n} = 1, \qquad
\binom{n}{1} = n, \qquad
\binom{n}{k} = \binom{n}{n-k},
\]
e \emph{Régra de Pascal}\index{Régra de Pascal}: para $1 \leq k \leq n-1$,
\[
\binom{n}{k} = \binom{n-1}{k-1} + \binom{n-1}{k}.
\]
\end{proposition}

\begin{proof}
A simetria $\binom nk = \binom{n}{n-k}$ segura porque tomando \omterm{def:g10:proba:operations}{complementos}
partidas $ k $-subconjuntos de elementos com $(n-k)$ subconjuntos de elementos, um por um. Para
Regra de Pascal, corrija um elemento $ a \in E $ e classificar o $ k $-subconjuntos de elementos em
aqueles que contêm $ a $ --- obtido por contíguo $ a $ para um $(k-1)$-elemento
subconjunto de $ E \setminus \{a\}$, dos quais existem $\binom{n-1}{k-1}$ --- e
aqueles que evitam $ a $, quais são os $ k $-subconjuntos de elementos de
$ E \setminus \{a\}$, numeração $\binom{n-1}{k}$. Conclua pela adição
princípio.
\end{proof}

Regra de Pascal gera os coeficientes linha por linha --- \emph{Pascal
triângulo}: cada entrada é a soma das duas acima dela.

\begin{theorem}[Teorema binomial]\label{thm:g12:comb:binomial}
\index{teorema binomial}
Para todos $ a, b \in \R $ (ou $\C $) e $ n \in \N $:
\[
(a+b)^n = \sum_{k=0}^{n} \binom{n}{k}\, a^{k}\, b^{\,n-k} .
\]
\end{theorem}

\begin{proof}
Expanda o produto $(a+b)(a+b)\cdots(a+b)$ ($ n $ fatores): cada termo do
escolhas de expansão $ a $ ou $ b $ em todos os fatores, produzindo $ a^k b^{n-k}$ onde
$ k $ é o número de fatores que contribuem $ a $. O número de maneiras de escolher
estes $ k $ fatores entre $ n $ é $\binom nk $, que é portanto o
coeficiente de $ a^k b^{n-k}$.
\end{proof}

\begin{corollary}\label{cor:g12:comb:sums}
$\displaystyle\sum_{k=0}^{n} \binom{n}{k} = 2^n $
\quad e \quad
$\displaystyle\sum_{k=0}^{n} (-1)^k\binom{n}{k} = 0$ \ ($ n \geq 1$).
\end{corollary}

\begin{proof}
Pegar $ a = b = 1$, então $ a = -1$, $ b = 1$ no teorema binomial. O primeiro
identidade também tem um significado direto: uma $ n $-conjunto de elementos tem $2^n $ subconjuntos
(cada elemento está dentro ou fora: princípio da multiplicação), classificado por tamanho.
\end{proof}

\begin{method}[Escolhendo o modelo certo]\label{met:g12:comb:model}
Antes de contar, responda a duas perguntas: \emph{a ordem importa?} e
\emph{as repetições são permitidas?}
\begin{center}
\begin{tabular}{l|cc}
 & order matters & order irrelevant\\
\hline
repetitions allowed & $ n^k $ (tuples) & (university)\\
no repetitions & $\frac{n!}{(n-k)!}$ (\omterm{def:g12:comb:tuples}{arranjos}) & $\binom nk $ (subsets)
\end{tabular}
\end{center}
Tirar bolas de uma urna: \emph{com reposição, a fim} $\to $ tuplas;
\emph{sem reposição, a fim} $\to $ \omterm{def:g12:comb:tuples}{arranjos}; \emph{um punhado de tudo
de uma vez} $\to $ subconjuntos.
\end{method}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g12:comb:1}
Uma placa consiste em 2 letras (A-Z), depois 3 dígitos e depois 2 letras.
Quantas placas são possíveis? Quantos não têm caracteres repetidos?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g12:comb:2}
Calcular $\dbinom{8}{3}$, $\dbinom{10}{8}$ e simplificar
$\dfrac{\binom{n}{2}}{\binom{n+1}{2}}$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g12:comb:3}
Numa turma de 30 alunos, deve-se eleger uma comissão de 4 alunos, depois um
presidente e um tesoureiro dentro do comitê (uma pessoa não pode ocupar ambos
escritórios). Quantos resultados são possíveis?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g12:comb:4}
Expandir $(x + 2)^5$ e $(1 - x)^6$ usando o teorema binomial. Qual é o
coeficiente de $ x^3$ em $(2x + 3)^7$?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:comb:5}
Uma mão de pôquer padrão consiste em 5 cartas de um baralho de 52 cartas.
\begin{enumerate}
  \item Quantas mãos existem?
  \item Quantas mãos contêm exatamente um ás? Pelo menos um ás?
  \item Quantas mãos são ``full house'' (três cartas do mesmo valor, duas do
        outro)?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:comb:6}
Quantos anagramas (rearranjos de letras, \omterm{def:g11:stat:mean}{significativos} ou não) o
palavra MATEMÁTICA tem? A palavra BANANA?
(Dica para BANANA: primeiro coloque os três As.)
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:comb:7}
Prove a identidade $ k\dbinom{n}{k} = n\dbinom{n-1}{k-1}$ ($1 \leq k \leq n $)
de duas maneiras: pelo \omterm{prop:g12:comb:tuples}{fatorial} fórmula, e contando de duas maneiras os pares
(comitê de $ k $ povo, seu presidente) escolhido entre $ n $ pessoas.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:comb:8}
Um caminho no avião vai de $(0,0)$ para $(m, n)$ por passos unitários Leste ou
Norte. Mostre que o número desses caminhos é $\dbinom{m+n}{m}$.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g12:comb:9}
Provar \emph{Identidade de Vandermonde}: para $0 \leq k \leq m + n $,
\[
\binom{m+n}{k} = \sum_{j=0}^{k} \binom{m}{j}\binom{n}{k-j},
\]
contando o $ k $- subconjuntos de elementos de um conjunto divididos em um grupo de $ m $ e um
grupo de $ n $. Deduza isso
$\displaystyle\sum_{j=0}^{n}\binom{n}{j}^{\!2} = \binom{2n}{n}$.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g12:comb:10}
Usando o teorema binomial, mostre que para todos $ n \geq 1$,
\[
\sum_{k=1}^{n} k \binom{n}{k} = n\,2^{n-1}.
\]
(Dica: diferencie $(1+x)^n $, ou usar \cref{exo:g12:comb:7}.)
\end{exercise}

\section{Problema: A arte de contar duas vezes}

\begin{problem}[{Problema de fim de semana --- estrelas e bares, perturbados
chapéus e identidades provadas pela contagem de uma coisa de duas maneiras}]
\label{pb:g12:comb:1}
O truque mais profundo da combinatória é surpreendentemente simples: contar
mesma coleção duas vezes, por dois métodos diferentes, e igualar
as respostas. Este problema pratica os modelos de
\cref{met:g12:comb:model}, adiciona uma técnica ao curso do capítulo
não precisava --- o \emph{estrelas e bares} de contagem de sorvete
--- depois conta os famosos \emph{chapéus malucos} exatamente, e
encontra o número $\frac1\eu $ esperando na parte inferior do chapéu
pilha, sua terceira aparição neste livro.

\medskip
\noindent\textbf{Part I --- Choosing the model.}
\begin{enumerate}
  \item Conte as placas feitas de $2$ letras seguidas de
        $3$ dígitos; depois os anagramas de BANANA.
  \item De um $32$- baralho de cartas, conte o $5$-mãos de cartas; então
        as mãos contendo exatamente $2$ do $4$ ases.
  \item Um robô caminha de $(0,0)$ para $(4,3)$ usando apenas unidade
        passos para a direita ou para cima: quantos caminhos? (Codifique um caminho como um
        palavra em R e U.)
  \item Expandir $(1 + x)^4$ pelo teorema binomial
        (\cref{thm:g12:comb:binomial}); então avalie em
        $ x = 1$ e $ x = -1$: quais duas identidades sobre o
        números $\binom nk $ cair fora?
  \item Prove por contagem dupla que
        $ k\binom nk = n\binom{n-1}{k-1}$ (contar
        comitês com presidente de duas maneiras) e deduzir
        $\sum_{k=0}^{n} k\binom nk = n\,2^{n-1}$.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Stars and bars.}
\begin{enumerate}[resume]
  \item Uma sorveteria vende $4$ sabores; você pede $10$
        colheres (os sabores podem se repetir, a ordem na xícara é
        irrelevante). Codifique um pedido como uma linha de $10$ estrelas
        (colheres) separadas por $3$ barras (mudanças de sabor) e
        conte os pedidos.
  \item Conte os triplos de não negativos \omterm{def:g10:numbers:sets}{inteiros} com
        $ x + y + z = 12$.
  \item Conte os triplos de \emph{positivo} \omterm{def:g10:numbers:sets}{inteiros} com
        $ x + y + z = 12$ (substituir $ x = 1 + x'$, etc.).
  \item Quantos monômios distintos aparecem na expansão de
        $(a + b + c)^5$?
  \item Verificação de sanidade do método: contar os pedidos de $3$
        colheres de $2$ sabores com a fórmula e, em seguida, liste
        todos eles e compare.
  \item Diga exatamente onde “as colheres são idênticas” entraram no
        codificação --- e conte o que acontece se o
        as colheres são comidas em ordem (posições distintas), com
        \cref{met:g12:comb:model}lista de verificação.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The deranged hats.}
UM \emph{desarranjo} é uma redistribuição de $ n $ chapéus para eles
$ n $ proprietários em que \emph{ninguém} recebe seu próprio chapéu; deixar
$ D_n $ conte-os. (\Cref{pb:g11:prob:1} mostrou que um convidado
em média recupera o seu próprio chapéu --- agora contamos o total
exatamente festas azaradas.)
\begin{enumerate}[resume]
  \item Calcular $ D_1$, $ D_2$, $ D_3$ por listagem, e $ D_4$
        pacientemente (ou habilmente).
  \item Justifique a recorrência
        $ D_n = (n - 1)\left(D_{n-1} + D_{n-2}\right)$: convidado 1
        recebe um chapéu $ k \neq 1$ ($ n - 1$ escolhas); dividir
        de acordo com se o convidado $ k $ recebe chapéu 1 ou não.
        Verifique se ele reproduz $ D_4$ e calcular $ D_5$.
  \item Para $ n = 3$, prove por inclusão-exclusão (subtraia
        as tarefas fixando pelo menos um chapéu, adicione de volta o
        contagens excessivas) que
        $ D_3 = 3!\left(1 - \frac{1}{1!} + \frac{1}{2!} -
        \frac{1}{3!}\right)$ e indique a fórmula geral.
  \item Calcular $\frac{D_5}{5!}$ e compare com
        $\frac1\eu \approx 0.3679$: o \omterm{def:g10:proba:distribution}{probabilidade} que um
        grande festa embaralhada perturba completamente é
        $\frac1\eu $ --- a terceira participação especial desta constante, depois
        a loteria e o secretário de \cref{pb:g12:exp:1}.
        (Porquê: a fórmula da questão 14 é o início de uma
        série famosa para $\eu^{-1}$, contado na universidade
        volumes.)
  \item Papai Noel Secreto entre $10$ amigos: nomes são sorteados
        uniformemente ao acaso. Qual é o \omterm{def:g10:proba:distribution}{probabilidade} o sorteio
        é válido (ninguém desenha sozinho), e quantos
        redesenhos o grupo deve esperar?
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Counting twice, winning twice.}
\begin{enumerate}[resume]
  \item O lema do aperto de mão: em qualquer festa, resumindo os convidados
        o número de mãos que cada um apertou conta cada aperto de mão
        exatamente duas vezes. Deduza isso \emph{o número de convidados
        quem apertou um número ímpar de mãos é sempre par} ---
        e verifique se a afirmação faz sentido em uma festa com três convidados.
  \item Prove a joia
        $1^3 + 2^3 + \dots + n^3 = (1 + 2 + \dots + n)^2$ por
        indução e verifique se há $ n = 3$. (Pequeno Gauss
        soma, ao quadrado, conta cubos.)
  \item Identidade de Vandermonde (\cref{exo:g12:comb:9}) através
        caminhos: interpretar $\binom{2n}{n}$ como caminhos de rede de
        a pergunta 3 é do tipo de $(0,0)$ para $(n,n)$, corte cada
        caminho no cruzamento da anti-diagonal, e explicar
        como $\sum_j \binom nj^2$ aparece.
  \item Finale --- os quatro movimentos do contador, uma linha cada com
        um exemplo deste problema: multiplique estágios e adicione
        casos; codificar de forma inteligente (estrelas e barras, palavras de caminho);
        contar a mesma coisa duas vezes (comitê com presidente,
        apertos de mão); subtraia o indesejado e corrija o
        contagens excessivas (desarranjos). E observe onde a contagem
        vai trabalhar a seguir: probabilidadee os caminhos do
        capítulo sobre matrizes e \omterm{def:g10:functions:graph}{gráficos}.
\end{enumerate}
\end{problem}
