\chapter{Variáveis Aleatórias e a Distribuição Binomial}\label{ch:g12:randvar}

UM variável aleatória atribui um número a cada resultado de um experimento: um
ganho, uma contagem, uma duração. Isso é \omterm{def:g12:randvar:exp}{expectativa} é a média de longo prazo do
valores que produz, seu \omterm{def:g12:randvar:exp}{variação} mede sua propagação. A estrela disso
capítulo é o \omterm{def:g12:randvar:rv}{distribuição} binomial, que conta sucessos em repetidos
\omterm{def:g12:condprob:indep}{independente} ensaios. Variáveis aleatórias e o \omterm{def:g12:randvar:rv}{distribuição} binomial eram
conheci pela primeira vez em \cref{ch:g11:prob,ch:g11:binom}; este capítulo analisa e
aprofunda-os, com as ferramentas combinatórias de \cref{ch:g12:comb} agora
disponível.

\section{Variáveis aleatórias discretas}

\begin{definition}[Variável aleatória, distribuição]\label{def:g12:randvar:rv}
UM \emph{variável calculada}\index{variável calculada} em um finito \omterm{def:g10:proba:events}{espaço amostral}
$\Omega $ é um \omterm{def:g11:func:function}{função} $ X \colon \Omega \to \R $. Isso é
\emph{distribuição}\index{distribuição} (ou lei) são os dados de sua possível
valores $ x_1, \dots, x_k $ e das \omterm{def:g10:proba:distribution}{probabilidades}
\[
p_i = \P(X = x_i), \qquad \sum_{i=1}^{k} p_i = 1 .
\]
\end{definition}

\begin{definition}[Expectativa, variância, desvio padrão]\label{def:g12:randvar:exp}
O \emph{expectativa}\index{expectativa} de $ X $ é
\[
\E(X) = \sum_{i=1}^{k} p_i\, x_i ,
\]
isso é \emph{variação}\index{variação} e \emph{\omterm{def:g11:stat:variance}{desvio padrão}} são
\[
\V(X) = \E\bigl((X - \E(X))^2\bigr) = \sum_{i=1}^k p_i\bigl(x_i - \E(X)\bigr)^2,
\qquad
\sigma(X) = \sqrt{\V(X)} .
\]
\end{definition}

\begin{proposition}[Fórmula de König-Huygens]\label{prop:g12:randvar:konig}
\index{Fórmula de König-Huygens}
$\V(X) = \E(X^2) - \E(X)^2$.
\end{proposition}

\begin{proof}
Escrever $ m = \E(X)$ e expandir:
\[
\V(X) = \sum_i p_i (x_i^2 - 2m x_i + m^2)
= \E(X^2) - 2m\sum_i p_i x_i + m^2\sum_i p_i
= \E(X^2) - 2m^2 + m^2 . \qedhere
\]
\end{proof}

\begin{proposition}[Transformação afim]\label{prop:g12:randvar:affine}
Para $ a, b \in \R $:
\[
\E(aX + b) = a\,\E(X) + b, \qquad
\V(aX + b) = a^2\,\V(X) .
\]
\end{proposition}

\begin{proof}
O primeiro é um rearranjo da soma definidora. Para o segundo, $ aX + b $
se desvia do seu significar por $ a(X - \E(X))$, e o quadrado multiplica por
$ a^2$.
\end{proof}

\begin{example}[Jogos justos]\label{ex:g12:randvar:game}
Um jogo custa $2$ euros; um dado é lançado e o jogador recebe o valor
mostrado se for pelo menos $5$, nada de outra forma. O ganho $ G $ leva valores
$-2$ (\omterm{def:g10:proba:distribution}{probabilidade} $\frac46$), $3$ ($\frac16$), $4$ ($\frac16$):
\[
\E(G) = \frac{-8 + 3 + 4}{6} = -\frac{1}{6} < 0 .
\]
Em média, o jogador perde $17$ centavos por jogo: o jogo é desfavorável
(como acontece com a maioria dos jogos reais).
\end{example}

\section{Ensaios de Bernoulli e a distribuição binomial}

\begin{definition}[Distribuição Bernoulli]\label{def:g12:randvar:bernoulli}
UM \emph{Julgamento de Bernoulli}\index{Julgamento de Bernoulli} é um experimento com dois
resultados, \emph{sucesso} (\omterm{def:g10:proba:distribution}{probabilidade} $ p $) e \emph{falha}
($ q = 1 - p $). O indicador $ X $ de sucesso ($ X = 1$ no sucesso, $0$ ligado
falha) segue o \emph{Bernoulli \omterm{def:g12:randvar:rv}{distribuição}} $\mathcal B(p)$:
\[
\E(X) = p, \qquad \V(X) = p(1-p) .
\]
(De fato $\E(X) = p $, $\E(X^2) = p $, e König--Huygens dá
$\V(X) = p - p^2$.)
\end{definition}

\begin{definition}[Distribuição binomial]\label{def:g12:randvar:binomial}
Repita um \omterm{def:g12:randvar:bernoulli}{Julgamento de Bernoulli} $ n $ vezes de forma \omterm{def:g12:condprob:indep}{independente}, e deixe $ X $ seja o total
número de sucessos. O \omterm{def:g12:randvar:rv}{distribuição} de $ X $ é o \emph{binômio
distribuição}\index{binômio
distribuição} $\mathcal B(n, p)$.
\end{definition}

\begin{theorem}\label{thm:g12:randvar:binomial}
Se $ X \sim \mathcal B(n, p)$, então para $0 \leq k \leq n $:
\[
\P(X = k) = \binom{n}{k} p^k (1-p)^{n-k},
\]
e
\[
\E(X) = np, \qquad \V(X) = np(1-p) .
\]
\end{theorem}

\begin{proof}
Um especificado \omterm{def:g12:seq:sequence}{sequência} de resultados com $ k $ sucessos e $ n-k $ falhas tem
\omterm{def:g10:proba:distribution}{probabilidade} $ p^k(1-p)^{n-k}$ por independência; o número de tais \omterm{def:g12:seq:sequence}{sequências}
é o número de maneiras de colocar o $ k $ sucessos entre os $ n $ testes,
a saber $\binom nk $ (\cref{ch:g12:comb}). Resumindo o \omterm{def:g12:seq:sequence}{sequências} dá
a fórmula --- e o teorema binomial confirma
$\sum_k \P(X = k) = (p + q)^n = 1$.

Para o \omterm{def:g12:randvar:exp}{expectativa}, usando $ k\binom nk = n\binom{n-1}{k-1}$
(\cref{exo:g12:comb:7}):
\[
\E(X) = \sum_{k=1}^{n} k\binom nk p^k q^{n-k}
= np\sum_{k=1}^{n} \binom{n-1}{k-1} p^{k-1} q^{(n-1)-(k-1)}
= np\,(p + q)^{n-1} = np .
\]
O \omterm{def:g12:randvar:exp}{variação} a fórmula é provada de forma semelhante com a identidade
$ k(k-1)\binom nk = n(n-1)\binom{n-2}{k-2}$, dando
$\E(X(X-1)) = n(n-1)p^2$, de onde
$\V(X) = n(n-1)p^2 + np - (np)^2 = np(1-p)$. (Uma prova estrutural --- o
\omterm{def:g12:randvar:exp}{variação} de uma soma de \omterm{def:g12:condprob:indep}{independente} variáveis --- vem com
\cref{thm:g12:sums:variance}.)
\end{proof}

\begin{method}[Reconhecendo uma situação binomial]\label{met:g12:randvar:recognize}
Verifique os três ingredientes antes de escrever $ X \sim \mathcal B(n,p)$: um
\emph{número fixo} $ n $ de provações; \emph{dois resultados} por teste com o
\emph{mesmo} sucesso \omterm{def:g10:proba:distribution}{probabilidade} $ p $; \emph{independência} de testes
(amostragem \emph{com} substituição, ou de uma grande população). Então use
\[
\P(X \geq 1) = 1 - (1-p)^n
\]
para ``pelo menos um sucesso'', e uma calculadora ou tabelas cumulativas para
geral $\P(X \leq k)$.
\end{method}

\begin{example}\label{ex:g12:randvar:atleastone}
Quantas vezes é necessário lançar um dado para obter pelo menos $99\%$ \omterm{def:g11:func:parity}{chance} de
rolando um seis? Com $ X \sim \mathcal B\left(n, \frac16\right)$:
$\P(X \geq 1) = 1 - \left(\frac56\right)^n \geq 0.99$ significa
$\left(\frac56\right)^n \leq 0.01$, \emph{ou seja}
$ n \geq \frac{\ln 0.01}{\ln(5/6)} \approx 25.3$: de $ n = 26$ continua.
\end{example}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g12:randvar:1}
UM variável aleatória $ X $ leva os valores $-1, 0, 2, 5$ com \omterm{def:g10:proba:distribution}{probabilidades}
$0.3, 0.2, 0.4, 0.1$. Calcular $\E(X)$, $\V(X)$ e $\sigma(X)$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g12:randvar:2}
Um teste de múltipla escolha tem $10$ perguntas, cada uma com $4$ escolhas, uma delas
o que está correto. Um aluno responde de maneira uniforme e aleatória, de forma \omterm{def:g12:condprob:indep}{independente}.
Deixar $ X $ seja o número de respostas corretas.
\begin{enumerate}
  \item Dê o \omterm{def:g12:randvar:rv}{distribuição} de $ X $, $\E(X)$ e $\sigma(X)$.
  \item Calcular $\P(X = 0)$, $\P(X = 5)$ e $\P(X \geq 1)$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g12:randvar:3}
Uma companhia de seguros assegura $ n = 400$ clientes; cada um registra uma reclamação durante
o ano com \omterm{def:g10:proba:distribution}{probabilidade} $ p = 0.05$, de forma \omterm{def:g12:condprob:indep}{independente}. Deixar $ X $ seja o número
de reivindicações. Identifique o \omterm{def:g12:randvar:rv}{distribuição} de $ X $ e calcule seu \omterm{def:g12:randvar:exp}{expectativa} e
\omterm{def:g11:stat:variance}{desvio padrão}.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:randvar:4}
No jogo de \cref{ex:g12:randvar:game}, o organizador quer um jogo justo
($\E(G) = 0$) alterando o preço de entrada $ c $. Encontrar $ c $. Calcule o
\omterm{def:g12:randvar:exp}{variação} do ganho para esta versão justa; é ``justo'' o mesmo que
``sem risco''?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:randvar:5}
Um jogador de basquete marca lances livres com \omterm{def:g10:proba:distribution}{probabilidade} $0.7$. Ela atira
$8$ vezes (\omterm{def:g12:condprob:indep}{independente} tiros). Calcule o \omterm{def:g10:proba:distribution}{probabilidade} que ela pontua:
exatamente $6$; pelo menos $6$; pelo menos uma vez. Qual é o número mais provável
de pontuações?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:randvar:6}
Uma companhia aérea sabe que cada passageiro reservado chega com \omterm{def:g10:proba:distribution}{probabilidade}
$0.9$, de forma \omterm{def:g12:condprob:indep}{independente}. Um voo tem $100$ assentos e a companhia aérea vende $104$
ingressos. Expresse, usando um \omterm{def:g12:randvar:rv}{distribuição} binomial, o \omterm{def:g10:proba:distribution}{probabilidade} isso mais
passageiros aparecem do que há assentos, e amarrou-o numericamente usando um
calculadora (dê a expressão exata).
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:randvar:7}
Deixar $ X \sim \mathcal B(n, p)$. Mostre isso
\[
\frac{\P(X = k+1)}{\P(X = k)} = \frac{n-k}{k+1}\cdot\frac{p}{1-p},
\]
e deduza que \omterm{def:g12:randvar:rv}{distribuição} aumenta até
$ k^* = \floor{(n+1)p}$ e diminui depois (o \emph{modo} do
binômio).
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g12:randvar:8}
\emph{(São Petersburgo, domesticado.)} Uma moeda justa é lançada até dar cara
aparece, mas no \omterm{def:g10:functions:extrema}{máximo} $10$ vezes. Deixar $ N $ ser o número de lançamentos usados, e
o jogador recebe $2^N $ euros se aparecessem cabeças, $0$ de outra forma.
\begin{enumerate}
  \item Dê o \omterm{def:g12:randvar:rv}{distribuição} de $ N $ restrito aos resultados vencedores:
        $\P(\text{heads first at toss } k) = 2^{-k}$ para
        $1 \leq k \leq 10$ e verifique o total \omterm{def:g10:proba:distribution}{probabilidade} de vencer.
  \item Calcule o retorno esperado. O que seria sem a tampa
        de $10$ joga?
\end{enumerate}
\end{exercise}

\section{Problema: o voo com overbook}

\begin{problem}[{Problema de fim de semana --- as companhias aéreas vendem mais assentos do que
eles têm, as fábricas aceitam lotes que mal inspecionaram, e o
distribuição binomial arbitra ambos}]
\label{pb:g12:randvar:1}
Uma companhia aérea com $100$ assentos vende felizmente $105$ ingressos: sobre
$10\,\%$ dos passageiros nunca aparecem e os assentos vazios ganham
nada. Até onde a companhia aérea pode avançar antes de atropelar passageiros
comer o lucro? O \omterm{def:g12:randvar:rv}{distribuição} binomial
(\cref{thm:g12:randvar:binomial}) respostas para o decimal --- e
o mesmo maquinário inspeciona lotes de fábricas, rifas de preços e
mantém as seguradoras solventes. Decisões sob repetição: esta é a
trabalho diário do binômio.

\medskip
\noindent\textbf{Part I --- Fluency.}
\begin{enumerate}
  \item Dê o completo \omterm{def:g12:randvar:rv}{distribuição} tabela do número $ X $ de
        seis em três lançamentos de um dado.
  \item Calcular $\E(X)$ e $ V(X)$
        (\cref{def:g12:randvar:exp}).
  \item Uma barraca cobra $1$ euro por três rolos e paga $2$
        euros por seis obtidos. Calcule o ganho esperado: justo
        jogo?
  \item Prática de lista de verificação (\cref{met:g12:randvar:recognize}):
        é o número de corações em $5$ cartas desenhadas
        \emph{sem} binômio de substituição? Com reposição?
        Justificar.
  \item Para $ X \sim \mathcal B(20,\ 0.3)$: dar $\E(X)$ e
        $\sigma(X)$, então $\E(S)$ e $\sigma(S)$ para a pontuação
        $ S = 5X - 10$ (\cref{prop:g12:randvar:affine}).
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The overbooked flight.}
Assentos: $100$. Ingressos vendidos: $105$. Cada portador de ingresso aparece
com \omterm{def:g10:proba:distribution}{probabilidade} $0.9$, de forma \omterm{def:g12:condprob:indep}{independente}; deixar $ X $ seja o número que
mostrar.
\begin{enumerate}[resume]
  \item Modelo: justificar $ X \sim \mathcal B(105,\ 0.9)$ com o
        checklist --- e confessar o ponto mais fraco do modelo
        (as não comparências são realmente \omterm{def:g12:condprob:indep}{independente}? pense em grupos e
        tempestades).
  \item Calcular $\E(X)$ e $\sigma(X)$.
  \item O choque ocorre quando $ X \geq 101$: escrever
        $\P(X \geq 101)$ como uma soma explícita de cinco binômios
        termos.
  \item Avalie a soma (calculadora): qual fração de voos
        vê pelo menos um passageiro atropelado?
  \item Dinheiro: os cinco ingressos extras trazem
        $5 \times 200 = 1\,000$ euros; cada passageiro atropelado
        custos $800$ euros em compensação. Calcule o esperado
        número de passageiros atropelados,
        $\sum_{k=101}^{105} (k - 100)\,\P(X = k)$, o esperado
        compensação e o veredicto sobre a apólice.
  \item Design: o regulador tolera
        $\P(\text{bumping}) < 5\,\%$. Testando
        $ n = 105, 106, 107, \dots $ ingressos, encontre o maior
        permitido $ n $.
  \item O atalho do sino: para $ n = 105$, calcule o \omterm{pb:g11:stat:1}{pontuação z}
        do limite de colisão,
        $ z = \frac{100.5 - \E(X)}{\sigma(X)}$, e compare o
        bruto ``sobre $2\sigma $, tão aproximadamente $2$--$3\,\%$ no
        estimativa da cauda superior'' com sua resposta exata. (O
        curva por trás deste atalho são os próximos capítulos
        estrela.)
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The factory gate.}
Muitas peças são aceitas se um \omterm{def:g10:proba:sample}{amostra} de $20$ contém no \omterm{def:g10:functions:extrema}{máximo}
uma peça com defeito.
\begin{enumerate}[resume]
  \item Se a verdadeira taxa de defeitos for $2\,\%$ (muito honesto),
        calcular o \omterm{def:g10:proba:distribution}{probabilidade} de aceitação.
  \item Se a taxa for $10\,\%$ (muito ruim), calcule novamente.
  \item Cite os dois riscos do procedimento (o lote honesto
        rejeitado; o lote ruim foi aceito), leia seus valores em
        questões 13 a 14 e diga qual mudança única melhora
        ambos ao mesmo tempo --- a que custo.
  \item Por trás da melhoria: mostre que o defeito observado
        \emph{freqüência} $\frac Xn $ em um \omterm{def:g10:proba:sample}{amostra} de tamanho $ n $ tem
        \omterm{def:g11:stat:variance}{desvio padrão} $\sqrt{\frac{p(1-p)}{n}}$ e
        concluir como a precisão é dimensionada com \omterm{def:g10:proba:sample}{amostra} tamanho (um velho
        amigo: o $\frac{1}{\sqrt n}$ do 10º ano).
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Games, raffles, reserves.}
\begin{enumerate}[resume]
  \item O jogo internacional de São Petersburgo
        \cref{exo:g12:randvar:8} esperava recompensa $10$
        euros. Compare em um parágrafo com o sem limite
        paradoxo encontrado em \cref{pb:g11:prob:1}: o que exatamente faz
        o limite é domesticado, e que preço seria agora justo?
  \item Uma rifa de caridade vende $200$ ingressos em $2$ euros;
        prêmios: um $100$-euro e dois $50$-cestas de euros.
        Calcule o ganho esperado de um bilhete e o ganho do sorteio
        margem; compare com a margem da casa da roleta europeia
        de $\frac{1}{37} \approx 2.7\,\%$. Por que o sorteio
        escapar impune?
  \item Uma seguradora detém $1\,000$ \omterm{def:g12:condprob:indep}{independente} políticas: reivindicação
        \omterm{def:g10:proba:distribution}{probabilidade} $0.01$ cada, tamanho da reivindicação $10\,000$ euros,
        prêmio $130$ euros. Calcule o anual esperado
        lucro \emph{e} o \omterm{def:g11:stat:variance}{desvio padrão} do total
        reivindicações. Compare os dois números: o que significa o
        comparação força as seguradoras reais a segurar?
  \item Finale --- o binômio como árbitro de decisão: o
        lista de verificação de reconhecimento; a bússola $\E \pm \sigma$;
        \omterm{def:g10:proba:distribution}{probabilidades} de cauda como o preço do risco; e os dois
        advertências permanentes (independência é uma afirmação de modelagem, e
        caudas raras, não significam, causam ruína). Uma frase cada,
        com o ponteiro: a lei dos grandes números e o sino
        curva, próximos capítulos, completam o livro de regras do árbitro.
\end{enumerate}
\end{problem}
