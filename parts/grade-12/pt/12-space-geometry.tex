\chapter{Vetores, linhas e planos no espaço}\label{ch:g12:space}

A geometria tridimensional torna-se computacional uma vez \omterm{def:g11:vect:vector}{vetores} e
\omterm{def:g10:coordgeom:system}{coordenadas} estão disponíveis: as linhas obtêm representações paramétricas, os planos obtêm
Cartesiano equações, e o \omterm{def:g12:space:dot}{produto escalar} mede ângulos e distâncias.
Este capítulo constrói esta caixa de ferramentas e a utiliza para resolver o problema clássico
\omterm{def:g10:numbers:interunion}{interseção} e problemas de distância.

\section{Vetores no espaço}

\omterm{def:g11:vect:vector}{Vetores} no espaço obedecem às mesmas regras que no plano: somam, dimensionam,
e $\vect{AB} = \vect{CD}$ exatamente quando $ ABDC $ é um paralelogramo. Em um
sistema de \omterm{def:g10:coordgeom:system}{coordenadas} $(O; \vec\imath, \vec\jmath, \vec k)$, todo vetor tem
\omterm{def:g10:coordgeom:system}{coordenadas} $(x, y, z)$.

\begin{definition}[Colinearidade, coplanaridade]\label{def:g12:space:collinear}
Dois \omterm{def:g11:vect:vector}{vetores} são \emph{\omterm{def:g11:vect:collinear}{colinear}} se um for múltiplo do outro. Três
\omterm{def:g11:vect:vector}{vetores} $\vec u, \vec v, \vec w $ são \emph{coplanar}\index{coplanar} se um
deles pode ser escrito como uma combinação dos outros dois, digamos
$\vec w = a\vec u + b\vec v $.
\end{definition}

\begin{proposition}\label{prop:g12:space:basis}
Se $\vec u, \vec v, \vec w $ não são \omterm{def:g12:space:collinear}{coplanar}, todo vetor do espaço
se decompõe exclusivamente como $ x\vec u + y\vec v + z\vec w $.
\end{proposition}

\begin{proof}[Ideia de prova]
Pela ponta do vetor para decompor, desenhe a linha paralela a
$\vec w $; encontra o plano de $\vec u, \vec v $ em um único ponto, que
\omterm{def:g12:arith:divides}{divide} o vetor em um componente nesse plano (exclusivamente $ x\vec u + y\vec
v $, geometria plana) e um componente ao longo $\vec w $. Singularidade: uma diferença
de duas decomposições expressaria uma das \omterm{def:g11:vect:vector}{vetores} em termos do outro
dois, contradizendo a não coplanaridade.
\end{proof}

\section{Linhas e planos}

\begin{definition}[Representação paramétrica de uma linha]\label{def:g12:space:line}
A linha através $ A(x_A, y_A, z_A)$ com \omterm{def:g11:vect:direction}{vetor de direção}
$\vec u (a, b, c) \neq \vec 0$ é o conjunto de pontos
\[
M(x_A + ta,\; y_A + tb,\; z_A + tc), \qquad t \in \R .
\]
\end{definition}

\begin{definition}[Avião]\label{def:g12:space:plane}
O avião passou $ A $ dirigido por dois não \omterm{def:g11:vect:collinear}{colinear} \omterm{def:g11:vect:vector}{vetores}
$\vec u, \vec v $ é o conjunto de pontos $ M $ com
$\vect{AM} = s\vec u + t\vec v $, $(s, t) \in \R^2$.
\end{definition}

\begin{proposition}[Posições relativas]\label{prop:g12:space:positions}
Dois planos distintos são paralelos ou se cruzam em uma linha. Uma linha e um
plano são paralelos (possivelmente contidos) ou se cruzam em um único
ponto. Duas linhas no espaço podem estar se cruzando, estritamente paralelas, idênticas,
ou \emph{inclinar} (não \omterm{def:g12:space:collinear}{coplanar}).
\end{proposition}

\begin{proof}[Esboço]
Estas são declarações de incidência; cada análise de caso reduz, em \omterm{def:g10:coordgeom:system}{coordenadas},
para resolver um sistema linear e contando suas soluções --- veja
\cref{met:g12:space:intersections}.
\end{proof}

\section{Produto escalar}

\begin{definition}[Produto escalar]\label{def:g12:space:dot}
O \emph{produto escalar}\index{produto escalar} de $\vec u $ e $\vec v $ é
\[
\vec u \cdot \vec v = \tfrac12\left(\norm{\vec u + \vec v}^2
- \norm{\vec u}^2 - \norm{\vec v}^2\right),
\]
onde $\norm{\vec u}$ é o comprimento de $\vec u $. Se ambos \omterm{def:g11:vect:vector}{vetores} são
diferente de zero, $\vec u \cdot \vec v = \norm{\vec u}\,\norm{\vec v}\cos\theta $
onde $\theta $ é o ângulo entre eles; e em um \emph{\omterm{def:g10:coordgeom:system}{ortonormal}}
sistema de \omterm{def:g10:coordgeom:system}{coordenadas},
\[
\vec u \cdot \vec v = xx' + yy' + zz' .
\]
Dois \omterm{def:g11:vect:vector}{vetores} são \emph{\omterm{def:g11:scal:orthogonal}{ortogonal}} se seu produto escalar for $0$.
\end{definition}

\begin{proposition}[Propriedades]\label{prop:g12:space:dotprops}
O \omterm{def:g12:space:dot}{produto escalar} é simétrico ($\vec u\cdot\vec v = \vec v\cdot\vec u $),
bilinear ($(a\vec u + b\vec v)\cdot \vec w = a\,\vec u\cdot\vec w +
b\,\vec v\cdot \vec w $), e
$\vec u \cdot \vec u = \norm{\vec u}^2 \geq 0$.
\end{proposition}

\begin{proof}
Em um sistema \omterm{def:g10:coordgeom:system}{ortonormal}, todas as três propriedades são imediatas na fórmula
$ xx' + yy' + zz'$, que por sua vez decorre da fórmula definidora e do
Cálculo pitagórico de comprimentos:
$\norm{\vec u}^2 = x^2 + y^2 + z^2$.
\end{proof}

\begin{definition}[Vetor normal]\label{def:g12:space:normal}
UM \emph{vetor normal}\index{vetor normal} de um avião $\mathcal P $ é um
diferente de zero vetor \omterm{def:g11:scal:orthogonal}{ortogonal} para cada vetor deitado $\mathcal P $ (isso
basta que seja \omterm{def:g11:scal:orthogonal}{ortogonal} para dois não \omterm{def:g11:vect:collinear}{colinear} direções de
$\mathcal P $).
\end{definition}

\begin{theorem}[Equação cartesiana de um plano]\label{thm:g12:space:planeeq}
Em um \omterm{def:g10:coordgeom:system}{ortonormal} sistema de \omterm{def:g10:coordgeom:system}{coordenadas}, o avião passou
$ A $ com \omterm{def:g12:space:normal}{vetor normal} $\vec n(a, b, c)$ tem equação
\[
a x + b y + c z + d = 0,
\]
onde $ d = -(ax_A + by_A + cz_A)$. Por outro lado, cada um desses equação com
$(a,b,c) \neq (0,0,0)$ define um plano com \omterm{def:g12:space:normal}{vetor normal} $(a, b, c)$.
\end{theorem}

\begin{proof}
$ M \in \mathcal P \iff \vect{AM} \perp \vec n \iff
a(x - x_A) + b(y - y_A) + c(z - z_A) = 0$, que se expande para equação.
Por outro lado, escolha qualquer ponto de solução $ A $ do equação; o mesmo
a computação de trás para frente mostra que o conjunto solução é
$\{M : \vect{AM}\cdot\vec n = 0\}$, o avião passou $ A $ normal para
$\vec n $.
\end{proof}

\begin{method}[Intersecções na prática]\label{met:g12:space:intersections}
\begin{itemize}
  \item \emph{Linha $\cap $ avião}: substitua o paramétrico \omterm{def:g10:coordgeom:system}{coordenadas} de
        a linha no avião equação; resolver para $ t $ (uma solução: uma
        ponto; $0 = $ diferente de zero: paralelo; $0 = 0$: linha contida).
  \item \emph{Avião $\cap $ avião}: resolva o sistema dos dois equações,
        parametrização por uma coordenada livre; a solução é uma linha (ou o
        planos são paralelos quando \omterm{def:g11:vect:vector}{vetores} normais são \omterm{def:g11:vect:collinear}{colinear}).
  \item \emph{Verificações de ortogonalidade}: linha $\perp $ avião se sua direção
        é \omterm{def:g11:vect:collinear}{colinear} com o normal; dois planos são perpendiculares se seus
        normais são \omterm{def:g11:scal:orthogonal}{ortogonal}.
\end{itemize}
\end{method}

\begin{proposition}[Distância de um ponto a um plano]\label{prop:g12:space:distance}
\index{distância!aponte para o avião}
Em um sistema \omterm{def:g10:coordgeom:system}{ortonormal}, a distância de $ M_0(x_0, y_0, z_0)$ para o
avião $\mathcal P : ax + by + cz + d = 0$ é
\[
\operatorname{dist}(M_0, \mathcal P)
= \frac{\abs{ax_0 + by_0 + cz_0 + d}}{\sqrt{a^2 + b^2 + c^2}}.
\]
\end{proposition}

\begin{proof}
Deixar $ H $ seja o \omterm{def:g11:scal:orthogonal}{ortogonal} projeção de $ M_0$ sobre $\mathcal P $: o pé de
a perpendicular, então $\vect{HM_0}$ é \omterm{def:g11:vect:collinear}{colinear} com a unidade normal
$\frac{\vec n}{\norm{\vec n}}$, e a distância é
$\abs{\vect{HM_0}\cdot \frac{\vec n}{\norm{\vec n}}}$. Para qualquer
$ H \in \mathcal P $,
$\vect{HM_0}\cdot\vec n = ax_0 + by_0 + cz_0 - (ax_H + by_H + cz_H)
= ax_0 + by_0 + cz_0 + d $ (usando o equação no $ H $), daí a fórmula
depois de dividir por $\norm{\vec n} = \sqrt{a^2+b^2+c^2}$.
\end{proof}

\begin{example}\label{ex:g12:space:distance}
Distância da origem ao plano $ x + 2y + 2z - 6 = 0$:
$\frac{\abs{-6}}{\sqrt{1 + 4 + 4}} = \frac63 = 2$.
\end{example}

\section{Exercícios}

Em todos os exercícios o sistema de \omterm{def:g10:coordgeom:system}{coordenadas} é \omterm{def:g10:coordgeom:system}{ortonormal}.

\begin{exercise}[$\star $]\label{exo:g12:space:1}
Deixar $ A(1, 0, 2)$, $ B(3, 1, 1)$ e $ C(2, -1, 3)$. São os pontos $ A $, $ B $,
$ C $ alinhado? Dê uma representação paramétrica da linha $(AB)$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g12:space:2}
Determine um cartesiano equação do avião através $ A(1, 1, 0)$ com
\omterm{def:g12:space:normal}{vetor normal} $\vec n(2, -1, 3)$. Será que o ponto $ B(0, 2, 1)$ pertence a isso?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g12:space:3}
Calcule o ângulo em $ A $ (para o grau mais próximo) do triângulo com
vértices $ A(0,0,0)$, $ B(1, 1, 0)$ e $ C(1, 0, 1)$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:space:4}
Considere a linha $\mathcal D $ através $ A(1, 2, 0)$ com direção
$\vec u(1, -1, 2)$, e o avião $\mathcal P : 2x + y - z + 1 = 0$.
Determinar $\mathcal D \cap \mathcal P $.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:space:5}
Deixar $\mathcal P : x - y + 2z = 1$ e $\mathcal Q : 2x + y + z = 4$.
Mostre isso $\mathcal P $ e $\mathcal Q $ cruzam em uma linha e dão um
representação paramétrica dele.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:space:6}
Mostre que as linhas
\[
\mathcal D_1 : (x, y, z) = (1 + t,\; 2t,\; 3 - t), \qquad
\mathcal D_2 : (x, y, z) = (2 + s,\; 1 + s,\; 1 + 2s)
\]
estão distorcidos (não \omterm{def:g12:space:collinear}{coplanar}).
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:space:7}
Deixar $\mathcal P : 2x - y + 2z - 3 = 0$.
\begin{enumerate}
  \item Calcule a distância de $ M_0(3, 1, 1)$ para $\mathcal P $.
  \item Determine o \omterm{def:g10:coordgeom:system}{coordenadas} do \omterm{def:g11:scal:orthogonal}{ortogonal} projeção $ H $ de
        $ M_0$ sobre $\mathcal P $ e verifique a distância $ M_0H $.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g12:space:8}
O cubo $ ABCDEFGH $ tem lado $1$; lugar \omterm{def:g10:coordgeom:system}{coordenadas} para que
$ A(0,0,0)$, $ B(1,0,0)$, $ D(0,1,0)$, $ E(0,0,1)$ (com
$ C = B + D $, $ F = B + E $, $ G = B + D + E $, $ H = D + E $ como \omterm{def:g11:vect:vector}{vetores} de
$ A $).
\begin{enumerate}
  \item Mostre que a diagonal $(AG)$ é \omterm{def:g11:scal:orthogonal}{ortogonal} para o avião $(BDE)$.
  \item Calcule a distância de $ A $ para o avião $(BDE)$, e o ponto
        onde $(AG)$ perfura isso.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g12:space:9}
\emph{(Volume de um tetraedro.)} Deixar $ A(1,1,1)$, $ B(2,1,0)$, $ C(0,2,1)$
e $ D(2,2,2)$.
\begin{enumerate}
  \item Verifique isso $\vect{AB}$, $\vect{AC}$, $\vect{AD}$ não são \omterm{def:g12:space:collinear}{coplanar}
        (os pontos formam um tetraedro genuíno).
  \item Encontre um vetor $\vec n $ \omterm{def:g11:scal:orthogonal}{ortogonal} para ambos $\vect{BC}$ e
        $\vect{BD}$, e deduza um cartesiano equação do avião $(BCD)$.
  \item Calcule a distância de $ A $ para o avião $(BCD)$ e a área de
        o triângulo $ BCD $, usando a fórmula
        $\text{Area} = \frac12\sqrt{\norm{\vec u}^2\norm{\vec v}^2 -
        (\vec u \cdot \vec v)^2}$ para o triângulo medido por
        $\vec u, \vec v $.
  \item Deduza o volume do tetraedro
        ($ V = \frac13 \times \text{base area} \times \text{height}$).
\end{enumerate}
\end{exercise}

\section{Problema: Fatiar o cubo}

\begin{problem}[{Problema de fim de semana --- um avião corta um cubo em um
hexágono perfeito, um tetraedro regular se esconde nos cantos e
o ângulo da diagonal molda cada diamante}]
\label{pb:g12:space:1}
Fatie um cubo e você verá quadrados e retângulos --- ainda um
famoso corte produz um \emph{hexágono regular}e quatro inteligentes
cortes de canto deixam para trás um \emph{tetraedro regular}. Ambos
surpresas e o $109.5^\circ $ ângulo que os átomos de carbono adoram,
caem no kit de ferramentas deste capítulo: linhas paramétricas, plano equações
e o \omterm{def:g12:space:dot}{produto escalar} (\cref{thm:g12:space:planeeq},
\cref{prop:g12:space:distance}). Use em todo o cubo de
\cref{exo:g12:space:8}: lado $1$, $ A(0,0,0)$, $ B(1,0,0)$,
$ D(0,1,0)$, $ E(0,0,1)$, $ C(1,1,0)$, $ F(1,0,1)$, $ H(0,1,1)$,
$ G(1,1,1)$.

\medskip
\noindent\textbf{Part I --- Fluency.}
\begin{enumerate}
  \item Escreva o equação do avião através $ B(1,0,0)$ com
        \omterm{def:g12:space:normal}{vetor normal} $(1,1,1)$, uma representação paramétrica de
        a linha através $ A $ dirigido por $(1,1,1)$, e seus
        \omterm{def:g10:numbers:interunion}{interseção} apontar.
  \item Calcule a distância de $ A $ para aquele avião
        (\cref{prop:g12:space:distance}).
  \item São os \omterm{def:g11:vect:vector}{vetores} $(1,-1,0)$ e $(1,1,-2)$ \omterm{def:g11:scal:orthogonal}{ortogonal}?
  \item Posição dos aviões $ x + y + z = 1$ e
        $2x + 2y + 2z = 5$? Da linha através $ A $ dirigido por
        $(1,1,0)$ em relação ao primeiro plano?
  \item Liste o pontos médios das seis arestas
        $[BC], [CD], [DH], [HE], [EF], [FB]$ do cubo e
        verifique se todos os seis satisfazem o equação
        $ x + y + z = \frac32$.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The hexagonal section.}
Deixar $ P $ seja o avião $ x + y + z = \frac32$.
\begin{enumerate}[resume]
  \item Mostre isso $ P $ é perpendicular à grande diagonal
        $(AG)$ e passa pelo centro do cubo.
  \item Calcule as distâncias entre pontos médios de
        pergunta 5 da seção: o que você acha?
  \item Calcule o ângulo do polígono da seção em um
        vértice (dois consecutivos \omterm{def:g11:vect:vector}{vetores} de borda e um ponto
        produto). Conclua: a seção é uma \emph{normal
        hexágono}.
  \item Calcule o perímetro e a área do hexágono e compare
        a área com a de uma face do cubo. (Para o
        registro: a maior seção plana do cubo unitário é
        o $1 \times \sqrt2$ retângulo diagonal, de área
        $\sqrt2$ --- o hexágono regular leva prata.)
  \item Onde fica a diagonal $(AG)$ perfurar $ P $? Verifique isso
        o ponto perfurante é o \omterm{prop:g10:coordgeom:midpoint}{ponto médio} de $[AG]$.
  \item Deslize o corte: descreva as seções
        $ x + y + z = c $ como $ c $ cresce de $0$ para $3$ ---
        calcular a seção para $ c = \frac12$ (qual polígono,
        qual tamanho?) e narrar a transformação que passa
        através do seu hexágono em $ c = \frac32$.
  \item Por que nenhum plano pode cortar um cubo em um polígono com
        \emph{Sete} lados? (Onde deve cada lado do
        seção mentira?)
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The hidden tetrahedron.}
\begin{enumerate}[resume]
  \item Calcule o volume do tetraedro de canto $ ABDE $
        (base, altura e a fórmula de um terço do meio
        Volume escolar admitido --- \cref{exo:g12:space:9}
        calcula esses volumes em geral).
  \item Os quatro vértices $ B $, $ D $, $ E $, $ G $: calcule todos os seis
        distâncias aos pares e concluir que $ BDEG $ é um
        \emph{regular} tetraedro. Então obtenha seu volume por
        subtraindo tetraedros de canto do cubo.
  \item Calcule a distância do centro do cubo até o
        avião $(BDE)$.
  \item O ângulo do diamante: calcule o ângulo entre o
        diagonais $\vect{AG}$ e $\vect{BH}$ do cubo. É
        suplemento, cerca $109.5^\circ $, é o ângulo entre
        ligações em diamante e metano --- por que quatro
        pares de elétrons em torno de um átomo de carbono escolhem os vértices
        do tetraedro da questão 14?
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Genuinely three-dimensional.}
\begin{enumerate}[resume]
  \item Descubra por onde passa a linha $ B(1,0,0)$ dirigido por
        $(0,1,1)$ perfura o plano do hexágono $ P $.
  \item Mostre que as linhas $(AB)$ e $(EG)$ nem se encontram nem
        são paralelos: \emph{inclinar} linhas, o fenômeno que
        não pode acontecer em um avião.
  \item Calcule o \omterm{def:g11:scal:orthogonal}{ortogonal} projeção de $ B $ para $ P $
        (caminhar de $ B $ ao longo da normal até o plano), e
        verifique se seu ponto satisfaz $ P$ de equação.
  \item Finale --- o kit do topógrafo espacial: linhas paramétricas
        para piercing, \omterm{def:g11:vect:vector}{vetores} normais para ângulos e distâncias,
        avião equações para seções; e o cubo como
        playground onde produziram um hexágono, um
        tetraedro, o ângulo do diamante e um par de inclinação
        linhas. Uma frase cada.
\end{enumerate}
\end{problem}
