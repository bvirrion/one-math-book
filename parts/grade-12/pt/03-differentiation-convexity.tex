\chapter{Diferenciação e Convexidade}\label{ch:g12:deriv}

O derivado mede a taxa instantânea de \omterm{def:g11:prob:variance}{variação} de um \omterm{def:g11:func:function}{função}; é
o declive do \omterm{def:g12:deriv:tangent}{tangente} linha para o seu \omterm{def:g10:functions:graph}{gráfico}. Este capítulo revisa e amplia
as regras de diferenciação, liga o sinal de $f'$ às variações de $ f $,
e apresenta convexidade, que é regido pelo segundo derivado.

\section{A derivada}

\begin{definition}[Derivada em um ponto]\label{def:g12:deriv:derivative}
Deixar $ f $ ser definido em um \omterm{def:g10:numbers:interval}{intervalo} $ I $ e $ a \in I $. O \omterm{def:g11:func:function}{função} $ f $ é
\emph{diferenciável em $ a $}\index{diferenciável em $ a $}\index{diferenciável} se o quociente de diferença
\[
\frac{f(a+h) - f(a)}{h}
\]
tem um limite finito como $ h \to 0$. Este limite é denominado \emph{derivado de
$ f $ no $ a $} e escrito $ f'(a)$. Se $ f $ é diferenciável em todos os pontos
$ I $, o \omterm{def:g11:func:function}{função} $ f' \colon x \mapsto f'(x)$ é o \emph{derivado} de
$ f $.
\end{definition}

\begin{definition}[Linha tangente]\label{def:g12:deriv:tangent}
Se $ f $ é \omterm{def:g12:deriv:derivative}{diferenciável} no $ a $, o \emph{tangente}\index{tangente} para o
curva de $ f $ no ponto $(a, f(a))$ é a linha de equação
\[
y = f(a) + f'(a)(x - a).
\]
\end{definition}

\begin{proposition}\label{prop:g12:deriv:diffcont}
UM \omterm{def:g11:func:function}{função} \omterm{def:g12:deriv:derivative}{diferenciável} no $ a $ é contínuo no $ a $. O inverso é falso.
\end{proposition}

\begin{proof}
Para $ h \neq 0$ pequeno, $ f(a+h) - f(a) = h \cdot \frac{f(a+h)-f(a)}{h}$. Como
$ h \to 0$, o lado direito tende a $0 \times f'(a) = 0$, então
$ f(a+h) \to f(a)$. Pelo contrário, $ x \mapsto \abs{x}$ é contínuo no
$0$, mas seu quociente de diferença em $0$ é igual $\frac{\abs h}{h} = \pm 1$
dependendo do sinal de $ h $, e não tem limite.
\end{proof}

\subsection{Regras de diferenciação}

\begin{proposition}[Operações]\label{prop:g12:deriv:operations}
Deixar $ u, v $ ser \omterm{def:g12:deriv:derivative}{diferenciável} sobre $ I $ e $\lambda \in \R $. Então $ u + v $,
$\lambda u $, $ uv $ são \omterm{def:g12:deriv:derivative}{diferenciável} sobre $ I $, como é $\frac{u}{v}$ onde
$ v \neq 0$, e
\[
(u+v)' = u' + v', \quad (\lambda u)' = \lambda u', \quad
(uv)' = u'v + uv', \quad
\left(\frac{u}{v}\right)' = \frac{u'v - uv'}{v^2}.
\]
\end{proposition}

\begin{proof}[Prova da regra do produto]
Escreva o quociente de diferença de $ uv $ no $ a $ como
\[
\frac{u(a+h)v(a+h) - u(a)v(a)}{h}
= \frac{u(a+h) - u(a)}{h}\,v(a+h) + u(a)\,\frac{v(a+h) - v(a)}{h}.
\]
Como $ h \to 0$, $ v(a+h) \to v(a)$ por continuidade
(\cref{prop:g12:deriv:diffcont}), então o lado direito tende a
$ u'(a)v(a) + u(a)v'(a)$. As outras regras são provadas de forma semelhante.
\end{proof}

\begin{theorem}[Regra da cadeia]\label{thm:g12:deriv:chain}
\index{regra da cadeia}
Deixar $ u $ ser \omterm{def:g12:deriv:derivative}{diferenciável} sobre $ I $ com valores em $ J $, e $ g $ \omterm{def:g12:deriv:derivative}{diferenciável}
sobre $ J $. Então $ g \circ u $ é \omterm{def:g12:deriv:derivative}{diferenciável} sobre $ I $ e
\[
(g \circ u)' = (g' \circ u) \cdot u' .
\]
Em particular, para $ u $ \omterm{def:g12:deriv:derivative}{diferenciável}:
\[
(u^n)' = n\,u'\,u^{n-1} \ (n \in \Z), \qquad
\left(\sqrt{u}\right)' = \frac{u'}{2\sqrt{u}} \ (u > 0), \qquad
\left(\eu^{u}\right)' = u'\,\eu^{u}.
\]
\end{theorem}

\begin{proof}[Esboço de prova]
Quando $ u(a+h) \neq u(a)$ para $ h $ pequeno, escreva
\[
\frac{g(u(a+h)) - g(u(a))}{h}
= \frac{g(u(a+h)) - g(u(a))}{u(a+h) - u(a)} \cdot \frac{u(a+h) - u(a)}{h}.
\]
Como $ h \to 0$, $ u(a+h) \to u(a)$, então o primeiro fator tende a $ g'(u(a))$ e
o segundo a $ u'(a)$. (Uma prova completa deve tratar do caso em que
$ u(a+h) = u(a)$ para $ h $ arbitrariamente pequeno; este ponto técnico é tratado em
universidade.)
\end{proof}

\begin{example}
O habitual derivados, válido no natural domínios:
\[
(x^n)' = nx^{n-1}, \quad
\left(\frac{1}{x}\right)' = -\frac{1}{x^2}, \quad
(\sqrt{x})' = \frac{1}{2\sqrt{x}}, \quad
(\cos x)' = -\sin x, \quad
(\sin x)' = \cos x .
\]
A primeira é provada por indução a partir da regra do produto, e as duas últimas em
\cref{ch:g12:trigo}.
\end{example}

\section{Variações e extremos}

\begin{theorem}[Sinal da derivada e variações]\label{thm:g12:deriv:variations}
Deixar $ f $ ser \omterm{def:g12:deriv:derivative}{diferenciável} em um \omterm{def:g10:numbers:interval}{intervalo} $ I $.
\begin{enumerate}
  \item $ f $ é \omterm{def:g12:seq:monotonic}{aumentando} sobre $ I $ se e somente se $ f' \geq 0$ sobre $ I $.
  \item $ f $ é diminuindo sobre $ I $ se e somente se $ f' \leq 0$ sobre $ I $.
  \item Se $ f' > 0$ sobre $ I $ exceto em um número finito de pontos onde
        desaparece, então $ f $ é estritamente \omterm{def:g12:seq:monotonic}{aumentando} sobre $ I $.
\end{enumerate}
\end{theorem}

\begin{proof}[Prova parcial]
Se $ f $ é \omterm{def:g12:seq:monotonic}{aumentando}, cada quociente de diferença
$\frac{f(a+h)-f(a)}{h}$ é $\geq 0$, e passando ao limite dá
$ f'(a) \geq 0$. As implicações inversas repousam no significar teorema do valor,
comprovado na universidade; eles são \emph{admitido neste nível}.
\end{proof}

\begin{definition}[Extremo local]\label{def:g12:deriv:extremum}
$ f $ tem um \emph{máximo local}\index{máximo local} no $ a $ se $ f(x) \leq f(a)$ para
tudo $ x $ aproximar $ a $; os \omterm{def:g10:functions:extrema}{mínimos} locais são definidos simetricamente.
\end{definition}

\begin{proposition}[Condição de primeira ordem]\label{prop:g12:deriv:critical}
Se $ f $ é \omterm{def:g12:deriv:derivative}{diferenciável} em um \emph{abrir} \omterm{def:g10:numbers:interval}{intervalo} $ I $ e tem um local
extremo no $ a \in I $, então $ f'(a) = 0$. A recíproca é falsa
($ f(x) = x^3$ no $ a = 0$).
\end{proposition}

\begin{proof}
Dizer $ a $ é um \omterm{def:g12:deriv:extremum}{máximo local}. Para $ h > 0$ pequeno,
$\frac{f(a+h)-f(a)}{h} \leq 0$, então deixando $ h \to 0^+$ dá $ f'(a) \leq 0$;
para $ h < 0$ o quociente é $\geq 0$, dando $ f'(a) \geq 0$. Por isso
$ f'(a) = 0$.
\end{proof}

\begin{method}[Tabela de variação]\label{met:g12:deriv:table}
Para estudar um \omterm{def:g11:func:function}{função} $ f $: determinar seu domínio e limites na fronteira;
calcular $ f'$ e fatorá-lo; determinar o sinal de $ f'$ em cada subintervalo;
registre tudo em um \omterm{def:g10:functions:table}{tabela de variação}, marcando extremos; deduzir o número de
soluções de $ f(x) = k $ usando o teorema da bijeção
(\cref{thm:g12:limcont:bijection}).
\end{method}

\section{Convexidade}

\begin{definition}[Função convexa]\label{def:g12:deriv:convex}
UM \omterm{def:g11:func:function}{função} $ f $ definido em um \omterm{def:g10:numbers:interval}{intervalo} $ I $ é \emph{convexo}\index{convexo}
sobre $ I $ se cada acorde do seu \omterm{def:g10:functions:graph}{gráfico} fica acima do \omterm{def:g10:functions:graph}{gráfico}: para todos
$ a, b \in I $ e $ t \in \intcc{0}{1}$,
\[
f\bigl((1-t)a + t b\bigr) \leq (1-t) f(a) + t f(b).
\]
Isso é \emph{côncavo}\index{côncavo} se a desigualdade inversa for válida ($-f $ convexo).
\end{definition}

\begin{theorem}[Convexidade e derivadas]\label{thm:g12:deriv:convexity}
Deixar $ f $ ser \omterm{def:g12:deriv:derivative}{diferenciável} em um \omterm{def:g10:numbers:interval}{intervalo} $ I $. Os seguintes são equivalentes:
\begin{enumerate}
  \item $ f $ é \omterm{def:g12:deriv:convex}{convexo} sobre $ I $;
  \item $ f'$ é \omterm{def:g12:seq:monotonic}{aumentando} sobre $ I $;
  \item o \omterm{def:g10:functions:graph}{gráfico} de $ f $ está acima de cada um de seus \omterm{def:g12:deriv:tangent}{tangentes}.
\end{enumerate}
Se $ f $ é duas vezes \omterm{def:g12:deriv:derivative}{diferenciável}, estes também são equivalentes a $ f'' \geq 0$ sobre
$ I $.
\end{theorem}

\begin{proof}[Prova de (2) $\Rightarrow $ (3)]
Correção $ a \in I $ e definir
$ g(x) = f(x) - f(a) - f'(a)(x-a)$, a lacuna entre a curva e o \omterm{def:g12:deriv:tangent}{tangente}
no $ a $. Então $ g $ é \omterm{def:g12:deriv:derivative}{diferenciável} e $ g'(x) = f'(x) - f'(a)$. Se $ f'$ é
\omterm{def:g12:seq:monotonic}{aumentando}, $ g' \leq 0$ sobre $ I \cap \intoc{-\infty}{a}$ e $ g' \geq 0$ sobre
$ I \cap \intco{a}{+\infty}$: $ g $ diminui antes $ a $ e aumenta depois,
então $ g $ atinge o seu \omterm{def:g10:functions:extrema}{mínimo} $ g(a) = 0$, por isso $ g \geq 0$.
As restantes implicações são admitidas a este nível; a equivalência com
$ f'' \geq 0$ segue de \cref{thm:g12:deriv:variations} aplicado a $ f'$.
\end{proof}

\begin{definition}[Ponto de inflexão]\label{def:g12:deriv:inflection}
Um ponto onde a curva de $ f $ atravessa seu \omterm{def:g12:deriv:tangent}{tangente} é um \emph{inflexão
ponto}\index{inflexão
ponto}. Para $ f $ duas vezes \omterm{def:g12:deriv:derivative}{diferenciável}, inflexão
pontos são os pontos onde $ f''$ muda de sinal.
\end{definition}

\begin{example}\label{ex:g12:deriv:cube}
$ f(x) = x^3$ tem $ f''(x) = 6x $: $ f $ é \omterm{def:g12:deriv:convex}{côncavo} sobre $\intoc{-\infty}{0}$,
\omterm{def:g12:deriv:convex}{convexo} sobre $\intco{0}{+\infty}$, com um ponto de inflexão na origem,
onde a curva cruza seu \omterm{def:g12:deriv:tangent}{tangente} (o $ x $-eixo).
\end{example}

\begin{example}[Desigualdades de convexidade]\label{ex:g12:deriv:ineq}
O exponencial é \omterm{def:g12:deriv:convex}{convexo} sobre $\R $ (é o segundo derivado é em si, positivo).
Isso é \omterm{def:g12:deriv:tangent}{tangente} no $0$ é $ y = 1 + x $, então
\[
\eu^x \geq 1 + x \quad \text{for all } x \in \R .
\]
Tal \emph{\omterm{def:g12:deriv:tangent}{tangente} desigualdades de linha} são um produto padrão de convexidade.
\end{example}

\begin{method}[Usando convexidade]\label{met:g12:deriv:convexity}
\begin{itemize}
  \item Para provar uma desigualdade $ f(x) \geq ax + b $: identificar a linha como um
        \omterm{def:g12:deriv:tangent}{tangente} de um \omterm{def:g12:deriv:convex}{convexo} $ f $ e invocar
        \cref{thm:g12:deriv:convexity}(3).
  \item Para localizar pontos de inflexão: resolver $ f''(x) = 0$ \emph{e} verificar
        isso $ f''$ muda o sinal ali.
  \item Em um \omterm{def:g10:functions:table}{tabela de variação}, convexidade refina o esboço da curva
        (para que lado ele dobra).
\end{itemize}
\end{method}

\section{Problema: O cálculo do salva-vidas}

\begin{problem}[{Problema de fim de semana --- o caminho de resgate mais rápido obedece
A lei da luz de Snell, a convexidade certifica todos os ótimos, e o
a lata de refrigerante ideal é exatamente tão alta quanto larga}]
\label{pb:g12:deriv:1}
Um salva-vidas vê um nadador em apuros --- na diagonal
praia, na água. Correr é mais rápido que nadar: o
linha reta é \emph{não} o caminho mais rápido. A rota que
minimiza curvas de tempo na costa, e sua lei de curvatura é
precisamente a lei pela qual a luz refrata a entrada na água: natureza,
também, diferencia. Este problema treina a regra da cadeia
(\cref{thm:g12:deriv:chain}), realiza o resgate, colhe o
desigualdades de convexidade (\cref{thm:g12:deriv:convexity}) e
projeta uma lata.

\medskip
\noindent\textbf{Part I --- Chain-rule fluency.}
\begin{enumerate}
  \item Diferenciar: $\left(3x^2 + 1\right)^5$; \
        $\sqrt{x^2 + 9}$; \ $\dfrac{1}{x^2 + 1}$.
  \item Dê o \omterm{def:g12:deriv:tangent}{tangente} para $ y = \sqrt{x^2 + 9}$ no $ x = 4$.
  \item Construa o \omterm{def:g10:functions:table}{tabela de variação} de
        $ f(x) = \dfrac{x}{x^2 + 1}$ sobre $\R $ (extremos incluídos).
  \item Para $ g(x) = x^3 - 3x^2 + 4$: \omterm{def:g10:functions:table}{tabela de variação} \emph{e}
        convexidade mesa ($ g''$), com o ponto de inflexão
        (\cref{def:g12:deriv:inflection}).
  \item Calcule o \omterm{def:g12:deriv:tangent}{tangente} para $ g $ em seu ponto de inflexão,
        e mostre isso $ g(x) - (\text{tangent}) = (x - 1)^3$:
        o que a mudança de sinal diz sobre como o \omterm{def:g12:deriv:tangent}{tangente}
        encontra a curva em uma inflexão?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The rescue.}
Litoral ao longo do $ x $-eixo; o salva-vidas fica na areia em
$ A(0, 30)$, o nadador espera em $ B(40, -20)$ (metros). O
guarda corre em $5$ m/s na areia, nada em $2$ m/s, e entra no
água em um ponto $(x, 0)$ de sua escolha.
\begin{enumerate}[resume]
  \item Modelo: expressar o tempo total de resgate $ T(x)$ e diga por que
        apenas $ x \in \intcc{0}{40}$ merece consideração.
  \item Diferenciar $ T $ (regra da cadeia no trabalho --- duas vezes).
  \item Deixar $\theta_1$ ser o ângulo da perna que corre com o
        \emph{perpendicular} para a costa, $\theta_2$ o de
        a perna nadadora. Mostre isso
        $\sin\theta_1 = \frac{x}{\sqrt{x^2 + 900}}$ e isso
        a condição $ T'(x) = 0$ lê
        \[
        \frac{\sin\theta_1}{5} = \frac{\sin\theta_2}{2}
        \]
        --- \emph{Lei de Snell}, com as duas velocidades no lugar de
        as velocidades da luz no ar e na água.
  \item O segmento reto $[AB]$ atravessa a costa em
        $ x = 24$. Calcular $ T(24)$, $ T(40)$ (correr ao longo do
        praia, depois nade direto) e $ T(0)$: é o
        linha reta geométrica é a mais rápida? É extremo
        estratégia?
  \item Localize o ponto de entrada ideal \omterm{thm:g12:limcont:ivt}{dicotomia} sobre
        $ T'$ (isso é $\approx 33.7$ m): dar $ x^*$ para o medidor
        e o tempo recorde $ T(x^*)$ ao décimo de segundo.
        Quanto a curva ideal economiza em relação à reta
        linha?
  \item Por que o ponto crítico é um \emph{global} \omterm{def:g10:functions:extrema}{mínimo}?
        (Cada termo de $ T $ é \omterm{def:g12:deriv:convex}{convexo} --- admitir que uma soma de
        \omterm{def:g12:deriv:convex}{convexo} funções é \omterm{def:g12:deriv:convex}{convexo} --- e aplique
        \cref{exo:g12:deriv:8}.)
  \item O princípio de Fermat afirma que a luz sempre viaja
        caminho de menos \emph{tempo}. Deduza por que um raio de luz se curva
        exatamente na superfície ao entrar na água
        (onde a luz é mais lenta) e nomeie o cotidiano
        observação isso explica sobre um lápis em um copo de
        água.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Convexidade's harvest.}
\begin{enumerate}[resume]
  \item Mostre isso $ x \mapsto x^2$ é \omterm{def:g12:deriv:convex}{convexo}, e provar
        \omterm{prop:g10:coordgeom:midpoint}{ponto médio} desigualdade
        $\left(\frac{a + b}{2}\right)^2 \leq
        \frac{a^2 + b^2}{2}$ duas vezes: uma vez de convexidade, uma vez
        por expandindo $(a - b)^2 \geq 0$.
  \item Deduza que o \emph{quadrático significar}
        $\sqrt{\frac{a^2 + b^2}{2}}$ domina a \omterm{def:g12:seq:arith-geom}{aritmética}
        significare monte a cadeia completa da série ---
        harmônico $\leq $ \omterm{def:g12:seq:arith-geom}{geométrico} $\leq $ \omterm{def:g12:seq:arith-geom}{aritmética} $\leq $
        quadrático --- verificando todos os quatro em $ a = 2$, $ b = 8$.
  \item \omterm{def:g12:deriv:tangent}{Tangente} abaixo da curva: $ x \mapsto \frac1x $ é \omterm{def:g12:deriv:convex}{convexo}
        sobre $\intoo{0}{+\infty}$; escreva seu \omterm{def:g12:deriv:tangent}{tangente} no $1$ e
        deduzir a desigualdade
        $\frac1x \geq 2 - x $ para todos $ x > 0$. Onde é que
        igualdade?
  \item Inflexão na natureza: durante uma epidemia, o
        a contagem cumulativa de casos segue uma curva em forma de S. O que
        acontece no seu ponto de inflexão, e por que essa data é
        aquele que os epidemiologistas procuram? Relacione-se com o signo
        do segundo derivado de cada lado.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The ideal can.}
Uma lata cilíndrica deve conter $ V = 330$ cm $^3$ com o \omterm{def:g10:functions:extrema}{mínimo}
metal: superfície $ S = 2\pi r^2 + 2\pi r h $, restrição
$\pi r^2 h = V $.
\begin{enumerate}[resume]
  \item Expressar $ S(r) = 2\pi r^2 + \dfrac{2V}{r}$,
        diferencie e calcule o raio ideal e
        altura para $ V = 330$ (precisão milimétrica).
  \item Prove a elegante lei geral: no ótimo,
        $ h = 2r $ --- a lata ideal é exatamente tão alta quanto é
        amplo.
  \item Verificar $ S''(r) > 0$ e concluir (via
        \cref{exo:g12:deriv:8} novamente) que o ótimo é
        globais. Latas de refrigerante reais são visivelmente mais altas do que largas:
        cite uma razão não matemática.
  \item Finale --- o pipeline de otimização: modelo, pergunte ao
        derivado, resolver pontos críticos, certificar com
        convexidade ou uma tabela, interprete. Execute a lista ao longo
        salva-vidas, a lata e o feixe mais forte do 11º ano
        (\cref{pb:g11:deriv:1}) --- e indique o que Fermat
        princípio diz sobre quem mais administra esse pipeline.
\end{enumerate}
\end{problem}
