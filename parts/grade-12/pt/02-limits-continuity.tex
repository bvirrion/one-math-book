\chapter{Limites e Continuidade}\label{ch:g12:limcont}

Este capítulo estende a noção de limite de \omterm{def:g12:seq:sequence}{sequências} para funções de um
variável real e introduz continuidade. O resultado central é o
teorema do valor intermediário, que transforma informações qualitativas (``$f $ é
contínuo e muda de sinal'') para a existência de soluções de equações.

\section{Limites de funções}

Por todo, $ f $ é um \omterm{def:g11:func:function}{função} definido em um \omterm{def:g10:numbers:interval}{intervalo} ou um \omterm{def:g10:numbers:interunion}{união} de \omterm{def:g10:numbers:interval}{intervalos}
$ D \subseteq \R $.

\begin{definition}[Limite no infinito]\label{def:g12:limcont:liminf}
Deixar $ f $ ser definido em um \omterm{def:g10:numbers:interval}{intervalo} $\intco{a}{+\infty}$ e $\ell \in \R $. Nós
diga isso $ f $ \emph{cuidar de um $\ell $ não $+\infty $}\index{tende a $\ell $ no $+\infty $},
escrito $\lim\limits_{x\to+\infty} f(x) = \ell $, se cada aberto \omterm{def:g10:numbers:interval}{intervalo}
contendo $\ell $ contém todos os valores $ f(x)$ para $ x $ grande o suficiente:
para cada $\varepsilon > 0$ existe $ A $ tal que para todos $ x \geq A $,
$\abs{f(x) - \ell} \leq \varepsilon $.

Nós dizemos $ f $ tende a $+\infty $ no $+\infty $ se para cada $ M \in \R $ lá
existe $ A $ tal que $ f(x) \geq M $ para todos $ x \geq A $. Limites em $-\infty $
e limites iguais a $-\infty $ são definidos de forma análoga.
\end{definition}

\begin{definition}[Limite em um ponto]\label{def:g12:limcont:limpoint}
Deixar $ a \in \R $ e deixe $ f $ ser definido próximo $ a $ (exceto possivelmente em $ a $). Nós
diga isso $ f $ tende a $\ell \in \R $ no $ a $, escrito
$\lim\limits_{x \to a} f(x) = \ell $, se para cada $\varepsilon > 0$ lá
existe $\delta > 0$ tal que para todos $ x \in D $ com
$\abs{x - a} \leq \delta $, $\abs{f(x) - \ell} \leq \varepsilon $.

Nós dizemos $ f $ tende a $+\infty $ no $ a $ se para cada $ M $ há $\delta > 0$
tal que $ f(x) \geq M $ em qualquer momento $ x \in D $ e $\abs{x - a} \leq \delta $.
Limites unilaterais ($ x \to a^+$, $ x \to a^-$) restringir $ x $ para $ x > a $ ou
$ x < a $.
\end{definition}

\begin{example}
$\lim\limits_{x\to+\infty} \frac{1}{x} = 0$,
$\lim\limits_{x\to 0^+} \frac{1}{x} = +\infty $,
$\lim\limits_{x\to 0^-} \frac{1}{x} = -\infty $. O \omterm{def:g11:func:function}{função}
$ x \mapsto \frac1x $ não tem limite (bilateral) em $0$.
\end{example}

\begin{definition}[Assíntotas]\label{def:g12:limcont:asymptote}
A linha $ y = \ell $ é um \emph{assíntota horizontal}\index{assíntota horizontal} do
curva de $ f $ no $+\infty $ (resp.\ $-\infty $) se
$\lim_{x\to+\infty} f(x) = \ell $ (resp.\ em $-\infty $). A linha $ x = a $ é um
\emph{assíntota vertical}\index{assíntota vertical} se $ f $ tem um limite unilateral infinito em $ a $.
\end{definition}

\begin{proposition}[Operações sobre limites]\label{prop:g12:limcont:operations}
As regras de \cref{prop:g12:seq:operations} (somas, produtos, quocientes) mantêm
literalmente para limites de funções, em um ponto ou no infinito, com o mesmo
quatro formas indeterminadas $(+\infty)+(-\infty)$, $0\times\infty $,
$\frac{\infty}{\infty}$, $\frac{0}{0}$.
\end{proposition}

\begin{proof}
As provas são idênticas às \omterm{def:g12:seq:sequence}{sequência} caso, substituindo ``for $ n \geq N $''
por ``para $ x \geq A $'' ou ``para $\abs{x-a} \leq \delta $''.
\end{proof}

\begin{theorem}[Limite de uma composição]\label{thm:g12:limcont:composition}
Deixar $ f, g $ ser funções e $ a, b, c $ ser \omterm{def:g10:numbers:sets}{números reais} ou $\pm\infty $. Se
$\lim\limits_{x \to a} f(x) = b $ e $\lim\limits_{y \to b} g(y) = c $, então
\[
\lim_{x \to a} g\bigl(f(x)\bigr) = c .
\]
A mesma afirmação vale para um \omterm{def:g12:seq:sequence}{sequência} no lugar de $ f $: se
$ u_n \to b $ e $\lim_{y\to b} g(y) = c $, então $ g(u_n) \to c $.
\end{theorem}

\begin{proof}
Pegar $ b, c $ finito (os outros casos são análogos). Deixar $\varepsilon > 0$.
Há $\delta > 0$ tal que $\abs{y - b} \leq \delta $ implica
$\abs{g(y) - c} \leq \varepsilon $; e há $\delta' > 0$ tal que
$\abs{x - a} \leq \delta'$ implica $\abs{f(x) - b} \leq \delta $. Encadeando o
dois dá $\abs{g(f(x)) - c} \leq \varepsilon $ para
$\abs{x - a} \leq \delta'$.
\end{proof}

\begin{theorem}[Comparação e aperto]\label{thm:g12:limcont:squeeze}
Deixar $ f, g, h $ ser funções e $ a \in \R \cup \{\pm\infty\}$.
\begin{enumerate}
  \item Se $ f \leq g $ aproximar $ a $ e $ f(x) \to +\infty $ como $ x \to a $, então
        $ g(x) \to +\infty $.
  \item Se $ f \leq g \leq h $ aproximar $ a $ e $ f, h $ tendem ao mesmo finito
        limite $\ell $ no $ a $, então $ g(x) \to \ell $ como $ x \to a $.
\end{enumerate}
\end{theorem}

\begin{proof}
O mesmo argumento que para \omterm{def:g12:seq:sequence}{sequências} (\cref{thm:g12:seq:squeeze}).
\end{proof}

\begin{method}[Calculando limites na prática]\label{met:g12:limcont:limits}
\begin{itemize}
  \item Polinômios e racionais funções no $\pm\infty $: fatorar o
        maior poder de $ x $; o limite é o da proporção de líderes
        termos.
  \item $\frac{0}{0}$ em um ponto $ a $: fator $(x-a)$ fora do numerador e
        denominador, ou reconhecer um derivado
        $\lim_{x\to a}\frac{f(x)-f(a)}{x-a} = f'(a)$ (\cref{ch:g12:deriv}).
  \item Raízes quadradas: multiplique pela expressão conjugada.
  \item Fatores oscilantes ($\cos $, $\sin $): use o teorema do aperto.
\end{itemize}
\end{method}

\section{Continuidade}

\begin{definition}[Continuidade]\label{def:g12:limcont:continuity}
Deixar $ f $ ser definido em um \omterm{def:g10:numbers:interval}{intervalo} $ I $ e $ a \in I $. O \omterm{def:g11:func:function}{função} $ f $ é
\emph{contínuo em $ a $}\index{contínua em $ a $} se
$\lim\limits_{x \to a} f(x) = f(a)$. Isso é \emph{contínuo ligado $ I $}\index{contínuo ligado $ I $} se for
contínua em todos os pontos $ I $.
\end{definition}

Intuitivamente, o \omterm{def:g10:functions:graph}{gráfico} de um contínuo \omterm{def:g11:func:function}{função} em um \omterm{def:g10:numbers:interval}{intervalo} pode ser desenhado
sem levantar a caneta.

\begin{proposition}[Continuidade das funções habituais]\label{prop:g12:limcont:usual}
Polinômios racionais funções, $ x \mapsto \sqrt{x}$, $ x \mapsto \abs{x}$,
$\cos $, $\sin $, $\exp $ e $\ln $ são contínuo em cada \omterm{def:g10:numbers:interval}{intervalo} contido
em seu domínio. Somas, produtos, quocientes (quando definidos) e composições
de contínuo funções são contínuo.
\end{proposition}

\begin{proof}[Prova parcial]
A estabilidade em somas, produtos, quocientes e composição decorre da
teoremas correspondentes sobre limites. Continuidade de $ x \mapsto x $ e de constante
funções é imediato da definição, portanto polinômios e racionais
funções são contínuo. Continuidade do restante habitual funções é
comprovados nos capítulos onde são construídos ou admitidos.
\end{proof}

\begin{example}
O \emph{chão \omterm{def:g11:func:function}{função}} $ x \mapsto \floor{x}$ (o maior inteiro
$\leq x $) não é contínuo em qualquer inteiro $ n $: seu limite esquerdo existe
$ n - 1$ enquanto $\floor{n} = n $.
\end{example}

\begin{proposition}[Sequências e continuidade]\label{prop:g12:limcont:seqcont}
Se $ u_n \to \ell $ e $ f $ é contínuo no $\ell $, então $ f(u_n) \to f(\ell)$.
Em particular, se um \omterm{def:g12:seq:sequence}{sequência} definido por $ u_{n+1} = f(u_n)$ converge para
$\ell $ e $ f $ é contínuo no $\ell $, então $ f(\ell) = \ell $.
\end{proposition}

\begin{proof}
A primeira alegação é \cref{thm:g12:limcont:composition} aplicado com o
\omterm{def:g12:seq:sequence}{sequência} $(u_n)$. Para o segundo, passe até o limite em ambos os lados
$ u_{n+1} = f(u_n)$: o lado esquerdo tende a $\ell $, o lado direito para
$ f(\ell)$, e os limites são únicos.
\end{proof}

\section{O teorema do valor intermediário}

\begin{theorem}[Teorema do valor intermediário]\label{thm:g12:limcont:ivt}
\index{teorema do valor intermediário}
Deixar $ f $ ser contínuo sobre $\intcc{a}{b}$ e deixe $ k $ seja qualquer número entre
$ f(a)$ e $ f(b)$. Então existe $ c \in \intcc{a}{b}$ tal que
$ f(c) = k $.
\end{theorem}

\end{axis}
\end{tikzpicture}

{\small O teorema do valor intermediário: a contínuo junção de curva
$(a, f(a))$ para $(b, f(b))$ deve cruzar todas as linhas horizontais
$ y = k $ deitado entre $ f(a)$ e $ f(b)$.}
\end{omfigure}

\begin{proof}
Substituindo $ f $ por $ f - k $ (e por $ k - f $ se necessário), podemos assumir
$ f(a) \leq 0 \leq f(b)$ e procure por um zero de $ f $. Construímos dois
\omterm{def:g12:seq:sequence}{sequências} por \emph{dicotomia}\index{dicotomia}: definir $ a_0 = a $, $ b_0 = b $;
dado $ a_n \leq b_n $ com $ f(a_n) \leq 0 \leq f(b_n)$, deixar
$ m = \frac{a_n + b_n}{2}$ e definir
\[
(a_{n+1}, b_{n+1}) =
\begin{cases}
(a_n, m) & \text{if } f(m) \geq 0,\\
(m, b_n) & \text{if } f(m) < 0.
\end{cases}
\]
Em ambos os casos $ f(a_{n+1}) \leq 0 \leq f(b_{n+1})$, o \omterm{def:g12:seq:sequence}{sequência} $(a_n)$ é
\omterm{def:g12:seq:monotonic}{aumentando}, $(b_n)$ é diminuindo, e
$ b_n - a_n = \frac{b-a}{2^n} \to 0$: o \omterm{def:g12:seq:sequence}{sequências} são adjacentes (ver
\cref{exo:g12:seq:10}) e convergem para um limite comum
$ c \in \intcc{a}{b}$. Por continuidade, $ f(a_n) \to f(c)$ e
$ f(b_n) \to f(c)$; passando ao limite em $ f(a_n) \leq 0$ e
$ f(b_n) \geq 0$ dá $ f(c) \leq 0 \leq f(c)$, então $ f(c) = 0$.
\end{proof}

\begin{remark}
O \omterm{thm:g12:limcont:ivt}{dicotomia} a prova é um algoritmo: ela produz resultados arbitrariamente precisos
aproximações de uma solução, reduzindo pela metade o erro em cada etapa. É assim que um
computador pode resolver $ f(x) = 0$ sabendo apenas avaliar $ f $.
\end{remark}

\begin{theorem}[Teorema da bijeção]\label{thm:g12:limcont:bijection}
Deixar $ f $ ser contínuo e \emph{estritamente monótono} em um \omterm{def:g10:numbers:interval}{intervalo} $ I $. Então
para cada $ k $ estritamente entre os limites de $ f $ nos extremos de $ I $, o
equação $ f(x) = k $ tem um \emph{exclusivo} solução em $ I $.
\end{theorem}

\begin{proof}
Existência: escolha $ a, b \in I $ com $ k $ entre $ f(a)$ e $ f(b)$ (possível
já que os valores de $ f $ aproximar-se de seus limites finais) e aplicar o
teorema do valor intermediário em $\intcc{a}{b}$. Singularidade: se $ x < y $, rigoroso
monotonicidade dá $ f(x) \neq f(y)$, então dois pontos distintos não podem ser ambos
soluções.
\end{proof}

\begin{method}[Resolvendo $ f(x) = k $ com o teorema da bijeção]\label{met:g12:limcont:bijection}
Para provar isso $ f(x) = k $ tem exatamente uma solução em um \omterm{def:g10:numbers:interval}{intervalo}:
\begin{enumerate}
  \item estabelecer um \omterm{def:g10:functions:table}{tabela de variação} de $ f $ (sinal de $ f'$, ver
        \cref{ch:g12:deriv});
  \item em cada \omterm{def:g10:numbers:interval}{intervalo} onde $ f $ é contínuo e estritamente monótono,
        comparar $ k $ com os valores ou limites do ponto final;
  \item aplique o teorema da bijeção em cada um desses \omterm{def:g10:numbers:interval}{intervalo} e conte o
        soluções.
\end{enumerate}
Uma calculadora ou \omterm{thm:g12:limcont:ivt}{dicotomia} em seguida, localiza cada solução com a precisão desejada.
\end{method}

\begin{example}\label{ex:g12:limcont:cubic}
O equação $ x^3 + x = 1$ tem uma solução real única. De fato
$ f(x) = x^3 + x $ é contínuo e estritamente \omterm{def:g12:seq:monotonic}{aumentando} sobre $\R $ (como uma soma de
estritamente funções crescentes), $\lim_{-\infty} f = -\infty $ e
$\lim_{+\infty} f = +\infty $, então o teorema da bijeção se aplica com $ k = 1$.
Desde $ f(0.6) = 0.816 < 1$ e $ f(0.7) = 1.043 > 1$, a solução está em
$\intoo{0.6}{0.7}$.
\end{example}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g12:limcont:1}
Calcule os seguintes limites:
\[
\lim_{x\to+\infty} \frac{2x^3 - x + 1}{5x^3 + x^2}, \qquad
\lim_{x\to-\infty} \bigl(x^2 + 3x - 1\bigr), \qquad
\lim_{x\to 2^+} \frac{x+1}{x-2}, \qquad
\lim_{x\to+\infty} \frac{\cos x}{x}.
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g12:limcont:2}
Deixar $ f(x) = \dfrac{2x^2 - 3x + 1}{x - 1}$, definido para $ x \neq 1$.
\begin{enumerate}
  \item Mostre isso $ f(x) = 2x - 1$ para todos $ x \neq 1$.
  \item Deduzir $\lim\limits_{x \to 1} f(x)$. Pode $ f $ ser estendido a um
        \omterm{def:g11:func:function}{função} contínuo no $1$?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star $]\label{exo:g12:limcont:3}
Determine o assíntotas da curva de
$ f(x) = \dfrac{3x - 1}{x + 2}$ e esboce a curva.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:limcont:4}
Calcular
\[
\lim_{x\to+\infty} \bigl(\sqrt{x^2 + x} - x\bigr)
\qquad\text{and}\qquad
\lim_{x\to 0} \frac{\sqrt{1+x} - 1}{x}.
\]
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:limcont:5}
Deixar $ f(x) = x + \sin x $.
\begin{enumerate}
  \item Mostre isso $ f(x) \to +\infty $ como $ x \to +\infty $, embora $ f $ não é
        monótono em qualquer \omterm{def:g10:numbers:interval}{intervalo} $\intco{A}{+\infty}$.
  \item Mostre isso $ g(x) = \dfrac{x + \sin x}{x - \sin x}$ tende a $1$ no
        $+\infty $.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:limcont:6}
Mostre que o equação $ x^3 - 3x + 1 = 0$ tem exatamente três soluções reais,
e dar para cada um um \omterm{def:g10:numbers:interval}{intervalo} de comprimento $1$ contendo-o.
(Dica: estude as variações de $ f(x) = x^3 - 3x + 1$; isso é derivado é
$3x^2 - 3$.)
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:limcont:7}
Deixar $ f \colon \intcc{0}{1} \to \intcc{0}{1}$ ser contínuo. Mostre isso $ f $ tem
um \emph{\omterm{pb:g10:functions:1}{ponto fixo}}: existe $ c \in \intcc{0}{1}$ com $ f(c) = c $.
(Dica: aplique o teorema do valor intermediário para $ g(x) = f(x) - x $.)
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g12:limcont:8}
Um caminhante sobe uma trilha de montanha, começando às 8h e chegando às 20h. O
no dia seguinte ela desce a mesma trilha, novamente começando às 8h e chegando às
20:00. Mostre que há uma hora do dia em que ela está exatamente na mesma hora
ponto da trilha nos dois dias. (Dica: apresente a diferença entre os dois
posição funções.)
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g12:limcont:9}
Deixar $ f(x) = x^5 + x^3 + x - 1$.
\begin{enumerate}
  \item Mostre isso $ f $ tem um único zero real $\alpha $, e isso
        $\alpha \in \intoo{0}{1}$.
  \item Descreva um \omterm{thm:g12:limcont:ivt}{dicotomia} computação de procedimento $\alpha $ para dentro
        $10^{-6}$, e forneça o número de iterações necessárias a partir de
        $\intcc{0}{1}$.
\end{enumerate}
\end{exercise}

\section{Problema: você não pode atravessar um rio sem se molhar}

\begin{problem}[{Problema de fim de semana --- o valor intermediário
teorema força pontos fixos, antípodas de igual temperatura e
tabelas constantes: três teoremas que parecem impossíveis de uma ideia}]
\label{pb:g12:limcont:1}
Amasse um mapa da França e jogue-o em qualquer lugar do solo francês: alguns
ponto do mapa está \emph{exatamente} acima do lugar
representa. A cada instante, dois pontos diametralmente opostos
do equador tem \emph{exatamente} a mesma temperatura. E um
mesa quadrada bamba em terreno acidentado sempre pode ser firmada por
\emph{girando} isto. Três truques de festa, um teorema: a contínuo
quantidade não pode passar de negativo para positivo sem desaparecer
(\cref{thm:g12:limcont:ivt}). Este problema treina limites, então
constrói todas as três maravilhas --- e sabe exatamente qual é o teorema
não promete.

\medskip
\noindent\textbf{Part I --- Limit fluency.}
\begin{enumerate}
  \item Calcular $\lim_{x \to +\infty} \dfrac{3x^2 - x}{x^2 + 5}$,
        \ $\lim_{x \to +\infty} \dfrac{2x + 1}{x^2 + 1}$, \ e
        $\lim_{x \to +\infty} (x^3 - x^2)$
        (\cref{met:g12:limcont:limits}).
  \item Dê o assíntotas de
        $ f(x) = \dfrac{3x - 1}{x + 2}$
        (\cref{def:g12:limcont:asymptote}), calculando o
        limites relevantes.
  \item Calcular $\lim_{x \to 3} \dfrac{x^2 - 9}{x - 3}$
        (fator) e
        $\lim_{x \to 0} \dfrac{\sqrt{x + 1} - 1}{x}$
        (conjugado).
  \item Pelo teorema do aperto
        (\cref{thm:g12:limcont:squeeze}): calcular
        $\lim_{x \to 0} x \sin\frac1x $ e
        $\lim_{x \to +\infty} \frac{\sin x}{x}$.
  \item Por composição e termos dominantes:
        $\lim_{x \to +\infty} \dfrac{\sqrt{4x^2 + x}}{x}$.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Raízes, certified.}
\begin{enumerate}[resume]
  \item Mostre que o equação $ x^3 + x = 5$ tem pelo menos um
        solução em $\intcc{1}{2}$.
  \item Mostre que a solução é única em $\R $
        (\cref{thm:g12:limcont:bijection}).
  \item Prove o clássico: \emph{todo polinômio de grau ímpar
        tem pelo menos um real \omterm{def:g11:quad:discriminant}{raiz}}. (Escreva o argumento
        para $ p(x) = x^3 + a x^2 + b x + c $: calcule os dois
        limites infinitos por termo dominante, então aplique o IVT em
        um grande \omterm{def:g10:numbers:interval}{intervalo}.)
  \item Localizando o \omterm{def:g11:quad:discriminant}{raiz} da questão 6 por \omterm{thm:g12:limcont:ivt}{dicotomia}: quantos
        metades de $\intcc{1}{2}$ garanta um erro abaixo
        $10^{-6}$ (\cref{exo:g12:limcont:9})? Compare com o
        duplicação de dígitos do método de Newton
        (\cref{pb:g11:deriv:1}): aproximadamente quantas etapas seriam
        isso precisa?
  \item O contra-exemplo que protege o teorema:
        $ g(x) = \frac1x $ satisfaz $ g(-1) = -1 < 0$ e
        $ g(1) = 1 > 0$, mas nunca desaparece. Qual hipótese de
        o IVT falha --- e o que o exemplo ensina sobre
        verificando hipóteses?
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Three impossible-sounding
theorems.}
\begin{enumerate}[resume]
  \item O teorema do \omterm{pb:g10:functions:1}{ponto fixo} do segmento: se
        $ f : \intcc{0}{1} \to \intcc{0}{1}$ é contínuo, provar
        que alguns $ c $ satisfaz $ f(c) = c $. (Estudar
        $ g(x) = f(x) - x $ nas duas extremidades.)
  \item Traduza a questão 11 no mapa amassado: o que acontece
        o papel de $ f $, por que ele pousa em
        $\intcc{0}{1}$ (figurativamente: dentro da França), e o que
        o ponto está exatamente acima de sua própria localização? (Um
        dimensão é admitida silenciosamente --- o honesto
        afirmação bidimensional é o teorema de Brouwer,
        volumes universitários.)
  \item O equador: deixe $ T $ seja um contínuo temperatura em um
        círculo, visto como um $2\pi $-periódico \omterm{def:g11:func:function}{função} com
        $ T(0) = T(2\pi)$. Definir $ g(t) = T(t) - T(t + \pi)$.
        Calcular $ g(0) + g(\pi)$, deduza que $ g $ desaparece
        em algum lugar $\intcc{0}{\pi}$, e declare o teorema
        sobre pontos antípodas.
  \item A mesa instável: uma mesa quadrada perfeitamente rígida
        ligado contínuo (sem degraus!) terreno acidentado; três pernas
        sempre toco, paira por uma altura $ h(\theta)$ isso
        depende continuamente no ângulo de rotação da mesa
        $\theta $. Explique por que girar um quarto de volta força
        a lacuna flutuante para mudar de sinal --- e concluir que
        algum ângulo $\theta^*$ estabiliza todas as quatro pernas. (Este
        o argumento foi publicado como um teorema genuíno em 2005;
        caféÓs proprietários conheceram o truque primeiro.)
  \item O caminhante de \cref{exo:g12:limcont:8} usei o mesmo
        esquema. Indique o modelo compartilhado pelas questões 11, 13,
        14 e o caminhante --- construir qual \omterm{def:g11:func:function}{função}, verifique o que
        no final, invoque what --- em três linhas.
  \item O que o IVT faz \emph{não} dar: a existência veio de graça,
        mas quais são as duas questões que deixa em aberto e quais
        ferramentas deste problema as respondem (questões 7 e 9)?
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Portraits with assíntotas.}
\begin{enumerate}[resume]
  \item Reescrever $ f(x) = \dfrac{3x - 1}{x + 2}$ como
        $3 - \dfrac{7}{x + 2}$ e use-o para descrever a curva
        completamente: assíntotas, variações de cada lado,
        esboço. (Uma mudança \omterm{def:g11:func:reference}{hipérbole} --- a curva de custo médio
        de \cref{pb:g10:reffunc:1} também era um.)
  \item O \omterm{def:g11:func:function}{função} $ f(x) = \dfrac{x^2 - 1}{x - 1}$ é
        indefinido em $1$. Qual valor atribuído a $ f(1)$ faz
        isso contínuo lá --- e por que os analistas chamam tal
        apontar um \emph{removível} descontinuidade?
  \item O chão \omterm{def:g11:func:function}{função} $ x \mapsto \lfloor x \rfloor $
        (o melhor inteiro $\leq x $): Cadê contínuo,
        para onde ele salta e por que a conclusão do IVT
        falhar por isso (encontrar $ a < b $ com
        $\lfloor a \rfloor < \frac12 < \lfloor b \rfloor $ ainda
        nenhuma solução de $\lfloor x \rfloor = \frac12$)?
  \item Final --- continuidadeas duas faces: a promessa local
        (valor $=$ limite: sem teletransporte) e o poder global
        (IVT: todo valor intermediário visitado). Arquive os quatro
        maravilhas da Parte III sob a face direita, e dar a
        slogan da travessia do rio seu nome matemático preciso.
\end{enumerate}
\end{problem}
