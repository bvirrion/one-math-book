\chapter{Aritmética}\label{ch:g12:arith}

A \omterm{def:g12:seq:arith-geom}{aritmética} estuda o \omterm{def:g10:numbers:sets}{inteiros}: \omterm{def:g12:arith:divides}{divisibilidade}, números primos, restos.
Há muito considerado o mais puro da matemática pura, agora protege todos os
pagamento: o criptossistema RSA baseia-se nos teoremas de Bézout, Gauss e
Fermat provou neste capítulo.

\section{Divisibilidade e divisão euclidiana}

\begin{definition}[Divisibilidade]\label{def:g12:arith:divides}
Deixar $a, b \in \Z $. Nós dizemos $ b $ \emph{divide} $ a $\index{divisibilidade}, escrito
$ b \mid a $, se existir $ k \in \Z $ com $ a = kb $. Nós também dizemos $ a $ é um
\emph{múltiplo} de $ b $.
\end{definition}

\begin{proposition}\label{prop:g12:arith:divprops}
Se $ c \mid a $ e $ c \mid b $, então $ c $ \omterm{def:g12:arith:divides}{divide} todo inteiro combinação
$ au + bv $ ($ u, v \in \Z $). Se $ a \mid b $ e $ b \mid a $ com
$ a,b \in \N $, então $ a = b $. Se $ a \mid b $ e $ b \neq 0$, então $\abs a \leq \abs b $.
\end{proposition}

\begin{proof}
Escrever $ a = kc $, $ b = lc $: então $ au + bv = (ku + lv)c $. Os outros pontos
seguir de $\abs{a} = \abs{k}\,\abs{b}$ com $\abs k \geq 1$ quando
$ b = ka \neq 0$.
\end{proof}

\begin{theorem}[Divisão euclidiana]\label{thm:g12:arith:euclid}
\index{Divisão euclidiana}
Deixar $ a \in \Z $ e $ b \in \N^*$. Existe um par único
$(q, r) \in \Z \times \N $ tal que
\[
a = bq + r \qquad\text{and}\qquad 0 \leq r < b .
\]
$ q $ é o \emph{quociente} e $ r $ o \emph{restante}.
\end{theorem}

\begin{proof}
\emph{Existência.} O conjunto de múltiplos de $ b $ não excedendo $ a $ tem um
maior elemento $ bq $ (não está vazio e \omterm{def:g12:seq:bounded}{limitado acima}); definir $ r = a - bq $.
Por maximidade, $ b(q+1) > a $, então $0 \leq r < b $.
\emph{Singularidade.} Se $ bq + r = bq' + r'$ com $0 \leq r, r' < b $, então
$ b(q - q') = r' - r $ e $\abs{r' - r} < b $: um múltiplo de $ b $ de absoluto
valor menor que $ b $ deve ser $0$, então $ r = r'$ e $ q = q'$.
\end{proof}

\section{Congruências}

\begin{definition}[Congruência]\label{def:g12:arith:congruence}
Deixar $ n \in \N^*$. Dois \omterm{def:g10:numbers:sets}{inteiros} $ a, b $ são \emph{módulo congruente
$ n $}\index{módulo congruente
$ n $}, escrito $ a \equiv b \pmod n $, se $ n \mid (a - b)$
--- equivalentemente, se $ a $ e $ b $ tem o mesmo resto no Euclidiano
divisão por $ n $.
\end{definition}

\begin{proposition}[Compatibilidade com operações]\label{prop:g12:arith:congops}
Se $ a \equiv b \pmod n $ e $ c \equiv d \pmod n $, então
\[
a + c \equiv b + d, \qquad
ac \equiv bd, \qquad
a^k \equiv b^k \ (k \in \N) \pmod n .
\]
\end{proposition}

\begin{proof}
$ n $ \omterm{def:g12:arith:divides}{divide} $(a-b) + (c-d) = (a+c) - (b+d)$, e
$ ac - bd = a(c - d) + d(a - b)$ também é um múltiplo de $ n $. A regra do poder
segue por indução da regra do produto.
\end{proof}

\begin{method}[Módulo de poderes de computação $ n $]\label{met:g12:arith:powers}
Para calcular $ a^k \bmod n $, reduza o \omterm{def:g12:complex:modulus}{módulo} base $ n $, então procure por um pequeno
poder de $ a $ congruente com $\pm1$ e use-o para recolher o expoente. Por exemplo $2^{100} \bmod 7$: desde $2^3 = 8 \equiv 1 \pmod 7$ e
$100 = 3\times33 + 1$,
\[
2^{100} = \left(2^{3}\right)^{33} \times 2 \equiv 1^{33}\times 2 = 2 \pmod 7 .
\]
\end{method}

\section{GCD, Bézout e Gauss}

\begin{definition}[GCD]\label{def:g12:arith:gcd}
Deixar $ a, b $ ser \omterm{def:g10:numbers:sets}{inteiros}, não ambos zero. O \emph{maior comum
divisor}\index{maior comum
divisor} $\gcd(a, b)$ é o maior inteiro dividindo ambos $ a $
e $ b $. Quando $\gcd(a,b) = 1$, $ a $ e $ b $ dizem ser
\emph{coprime}\index{coprime}.
\end{definition}

\begin{proposition}[Algoritmo de Euclides]\label{prop:g12:arith:euclidalgo}
\index{Algoritmo de Euclides}
Se $ a = bq + r $ ($ b \neq 0$), então $\gcd(a, b) = \gcd(b, r)$. Iterando o
\omterm{thm:g12:arith:euclid}{Divisão euclidiana} portanto calcula $\gcd(a,b)$: o mdc é o último
resto diferente de zero.
\end{proposition}

\begin{proof}
Qualquer divisor comum de $ a $ e $ b $ \omterm{def:g12:arith:divides}{divide} $ r = a - bq $
(\cref{prop:g12:arith:divprops}), portanto, é um divisor comum de $ b $ e $ r $;
e inversamente, uma vez que $ a = bq + r $. Os dois pares têm o mesmo comum
divisores, então o mesmo mdc. O algoritmo termina porque os restos
formar um estritamente diminuindo \omterm{def:g12:seq:sequence}{sequência} de não negativo \omterm{def:g10:numbers:sets}{inteiros}.
\end{proof}

\begin{example}\label{ex:g12:arith:euclidalgo}
$\gcd(252, 198)$: $252 = 198 + 54$; $198 = 3\times54 + 36$;
$54 = 36 + 18$; $36 = 2 \times 18 + 0$. Por isso $\gcd(252,198) = 18$.
\end{example}

\begin{theorem}[Identidade de Bézout]\label{thm:g12:arith:bezout}
\index{Identidade de Bézout}
Deixar $ a, b $ ser \omterm{def:g10:numbers:sets}{inteiros}, não ambos zero, e $ d = \gcd(a,b)$. Existem
$ u, v \in \Z $ tal que
\[
au + bv = d .
\]
Em particular, $ a $ e $ b $ são \omterm{def:g12:arith:gcd}{coprime} se e somente se $ au + bv = 1$ para alguns
\omterm{def:g10:numbers:sets}{inteiros} $ u, v $.
\end{theorem}

\begin{proof}
Correr \omterm{prop:g12:arith:euclidalgo}{Algoritmo de Euclides} de trás para frente: cada resto é um inteiro combinação
dos dois anteriores, e os dados iniciais $ a, b $ são combinações de
eles mesmos; por substituição descendente, o último resto diferente de zero $ d $ é um
inteiro combinação de $ a $ e $ b $. (Em \cref{ex:g12:arith:euclidalgo}:
$18 = 54 - 36 = 54 - (198 - 3\times54) = 4\times54 - 198 =
4(252 - 198) - 198 = 4\times252 - 5\times198$.)

Para a equivalência: se $\gcd(a,b) = 1$, Bézout fornece $ u, v $; inversamente,
qualquer divisor comum de $ a $ e $ b $ \omterm{def:g12:arith:divides}{divide} $ au + bv = 1$, forçando
$\gcd(a,b) = 1$.
\end{proof}

\begin{theorem}[Lema de Gauss]\label{thm:g12:arith:gauss}
\index{Lema de Gauss}
Deixar $ a, b, c \in \Z $. Se $ a \mid bc $ e $\gcd(a, b) = 1$, então $ a \mid c $.
\end{theorem}

\begin{proof}
Bézout dá $ au + bv = 1$; multiplique por $ c $: $ acu + bcv = c $. Ambos os termos
o lado esquerdo são múltiplos de $ a $ (o segundo porque $ a \mid bc $), portanto
assim é $ c $.
\end{proof}

\begin{corollary}\label{cor:g12:arith:coprimeprod}
Se $ a \mid c $, $ b \mid c $ e $\gcd(a,b) = 1$, então $ ab \mid c $.
\end{corollary}

\begin{proof}
Escrever $ c = ak $. De $ b \mid ak $ e $\gcd(a,b)=1$, Gauss dá
$ b \mid k $, dizer $ k = bl $; então $ c = abl $.
\end{proof}

\section{Números primos}

\begin{definition}[Melhor]\label{def:g12:arith:prime}
Um inteiro $ p \geq 2$ é \emph{melhor}\index{melhor} se for apenas
divisores positivos são $1$ e $ p $.
\end{definition}

\begin{proposition}\label{prop:g12:arith:primedivides}
Todo inteiro $ n \geq 2$ tem um \omterm{def:g12:arith:prime}{melhor} divisor; se $ n $ não é \omterm{def:g12:arith:prime}{melhor}, tem um
\omterm{def:g12:arith:prime}{melhor} divisor $\leq \sqrt n $. Se um \omterm{def:g12:arith:prime}{melhor} $ p $ \omterm{def:g12:arith:divides}{divide} um produto $ ab $, então
$ p \mid a $ ou $ p \mid b $ (\emph{Lema de Euclides}).
\end{proposition}

\begin{proof}
O menor divisor $ d \geq 2$ de $ n $ é \omterm{def:g12:arith:prime}{melhor} (qualquer divisor próprio de $ d $
seria um divisor menor de $ n $). Se $ n = de $ é composto com
$2 \leq d \leq e $, então $ d^2 \leq de = n $, então $ d \leq \sqrt n $. Para Euclides
lema: se $ p \nmid a $, então $\gcd(p, a) = 1$ (os únicos divisores de $ p $ são
$1$ e $ p $), e o lema de Gauss dá $ p \mid b $.
\end{proof}

\begin{theorem}[Euclides]\label{thm:g12:arith:infiniteprimes}
Existem infinitamente muitos primos.
\end{theorem}

\begin{proof}
Dada qualquer lista finita $ p_1, \dots, p_k $ de primos, considere
$ N = p_1 p_2 \cdots p_k + 1$. Alguns \omterm{def:g12:arith:prime}{melhor} $ p $ \omterm{def:g12:arith:divides}{divide} $ N $; mas não $ p_i $
\omterm{def:g12:arith:divides}{divide} $ N $ (o restante é $1$), então $ p $ é um \omterm{def:g12:arith:prime}{melhor} não na lista. Não
lista finita esgota o primos.
\end{proof}

\begin{theorem}[Teorema fundamental da aritmética]\label{thm:g12:arith:fta}
\index{teorema fundamental da aritmética}
Todo inteiro $ n \geq 2$ é um produto de primos, e esta fatoração é
único até a ordem dos fatores:
\[
n = p_1^{\alpha_1} p_2^{\alpha_2} \cdots p_r^{\alpha_r},
\qquad p_1 < p_2 < \dots < p_r \text{ primes},\ \alpha_i \geq 1 .
\]
\end{theorem}

\begin{proof}
\emph{Existência}, por indução forte: $ n $ \omterm{def:g12:arith:prime}{melhor} é a sua própria fatoração;
de outra forma $ n = de $ com $2 \leq d, e < n $, e ambos fatoram pela indução
hipótese. \emph{Singularidade}: suponha
$ p_1\cdots p_s = q_1 \cdots q_t $ (primos, com repetições permitidas). Por
Lema de Euclides, $ p_1$ \omterm{def:g12:arith:divides}{divide} alguns $ q_j $, e sendo \omterm{def:g12:arith:prime}{melhor}, $ p_1 = q_j $;
cancelar e repetir. As duas fatorações correspondem termo a termo.
\end{proof}

\begin{theorem}[O pequeno teorema de Fermat]\label{thm:g12:arith:fermat}
\index{O pequeno teorema de Fermat}
Deixar $ p $ ser \omterm{def:g12:arith:prime}{melhor} e $ a \in \Z $ com $ p \nmid a $. Então
\[
a^{p-1} \equiv 1 \pmod p .
\]
Para cada $ a \in \Z $ (sem coprimalidade assumida), $ a^p \equiv a \pmod p $.
\end{theorem}

\begin{proof}
Considere o $ p - 1$ \omterm{def:g10:numbers:sets}{inteiros} $ a, 2a, 3a, \dots, (p-1)a $ \omterm{def:g12:complex:modulus}{módulo} $ p $. Nenhum
é $\equiv 0$ (se $ p \mid ka $ com $1 \leq k \leq p-1$, As forças do lema de Euclides
$ p \mid k $, impossível), e eles são \omterm{def:g12:complex:modulus}{módulo} distinto entre pares $ p $ (se
$ ka \equiv la $, então $ p \mid (k - l)a $, então $ p \mid k - l $, então $ k = l $).
Portanto, \omterm{def:g12:complex:modulus}{módulo} $ p $, eles são os números $1, 2, \dots, p-1$ em alguma ordem.
Multiplicando tudo congruências:
\[
a^{p-1}\,(p-1)! \equiv (p-1)! \pmod p .
\]
Desde $ p $ \omterm{def:g12:arith:divides}{divide} nenhum dos $1, \dots, p-1$, uso repetido do lema de Euclides
permite cancelar $(p-1)!$, saindo $ a^{p-1} \equiv 1$. A segunda forma
segue multiplicando por $ a $ (e é trivial quando $ p \mid a $).
\end{proof}

\begin{example}[Aplicação à criptografia]\label{ex:g12:arith:rsa}
O teorema de Fermat torna o \omterm{def:g12:complex:modulus}{módulo} de exponenciação $ n $ reversível quando o
expoentes são escolhidos adequadamente --- o coração do \emph{RSA} criptossistema.
Com $ p, q $ grande primos e $ n = pq $, um publica $ n $ e um expoente
$ e $; criptografia é $ x \mapsto x^e \bmod n $. Descriptografar requer um expoente
$ d $ com $ ed \equiv 1 \pmod{(p-1)(q-1)}$, que só quem conhece $ p $
e $ q $ pode calcular --- e recuperar $ p, q $ de $ n $ significa \omterm{def:g10:algebra:expand}{factoring} um
numerar centenas de dígitos, o que nenhum algoritmo conhecido faz em razoável
tempo.
\end{example}

\section{Exercícios}

\begin{exercise}[$\star $]\label{exo:g12:arith:1}
Calcule o quociente e o resto do \omterm{thm:g12:arith:euclid}{Divisão euclidiana} de $2026$ por

\end{exercise}

\begin{exercise}[$\star $]\label{exo:g12:arith:2}
Qual é o restante $7^{100}$ \omterm{def:g12:complex:modulus}{módulo} $10$? (Qual é o último dígito de

\end{exercise}

\begin{exercise}[$\star $]\label{exo:g12:arith:3}
Usando \omterm{prop:g12:arith:euclidalgo}{Algoritmo de Euclides}, calcular $\gcd(1071, 462)$ e encontre \omterm{def:g10:numbers:sets}{inteiros}

\end{exercise}

\begin{exercise}[$\star $]\label{exo:g12:arith:4}
Mostre isso para cada $ n \in \Z $, $ n^2$ é congruente com $0$ ou $1$ \omterm{def:g12:complex:modulus}{módulo}
$4$. Deduza que um inteiro $\equiv 3 \pmod 4$ nunca é uma soma de dois
quadrados.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:arith:5}

\end{exercise}

(Dica: encontre o inverso de $5$ \omterm{def:g12:complex:modulus}{módulo} $11$.)
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:arith:7}
Resolva em $\Z \times \Z $ o Diofantino equação

17x - 40y = 1,
\]
então descreva todas as soluções de $17x - 40y = 6$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:g12:arith:8}

\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g12:arith:9}
Deixar $ p $ seja um \omterm{def:g12:arith:prime}{melhor}.
\begin{enumerate}
  \item Mostre isso para $1 \leq k \leq p - 1$, $ p $ \omterm{def:g12:arith:divides}{divide} $\dbinom{p}{k}$.
        (Dica: use $ k\binom pk = p\binom{p-1}{k-1}$, \cref{exo:g12:comb:7},
        e o lema de Gauss.)
  \item Deduza, por indução em $ a \geq 0$, mais uma prova do pouco
        teorema na forma $ a^p \equiv a \pmod p $.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:g12:arith:10}

\]
(Dica: resolva as duas primeiras condições e depois incorpore a terceira; Bézout

\end{exercise}

\section{Problema: códigos secretos e dígitos de verificação}

\begin{problem}[{Problema de fim de semana --- congruências protegem todos
código de barras e cartão de crédito, e o pequeno teorema de Fermat executa o
bloqueie os segredos do mundo}]
\label{pb:g12:arith:1}
G.\,H.~Hardy vangloriou-se em 1940 de que a teoria dos números era “imaculada”.
por aplicativos. Oitenta anos depois, cada sinal sonoro de código de barras, cada
pagamento com cartão de crédito e cada mensagem criptografada o contradiz
--- exatamente com as ferramentas deste capítulo: congruências
(\cref{prop:g12:arith:congops}),Bézout inverso
(\cref{thm:g12:arith:bezout}) e o pequeno teorema de Fermat
(\cref{exo:g12:arith:9}). Este problema verifica os códigos, quebra um
versão de brinquedo da fechadura e aprende por que a fechadura real funciona.

\medskip
\noindent\textbf{Part I --- Congruência fluency.}
\begin{enumerate}
  \item Calcular $2026 \bmod 7$; então o último dígito de
        $7^{100}$ (encontre o ciclo de potências de $7$ \omterm{def:g12:complex:modulus}{módulo}
        $10$).
  \item Exponenciação rápida (\cref{met:g12:arith:powers}):
        calcular $5^{117} \bmod 13$ (começa de
        $5^2 \equiv -1$).
  \item Resolver $3x \equiv 5 \pmod 7$.
  \item Correr \omterm{prop:g12:arith:euclidalgo}{Algoritmo de Euclides} sobre $(97, 35)$, substituto traseiro
        encontrar \omterm{def:g10:numbers:sets}{inteiros} $ u, v $ com $97u + 35v = 1$ e
        deduzir o inverso de $35$ \omterm{def:g12:complex:modulus}{módulo} $97$.
  \item Indique precisamente quando $ a $ é \omterm{def:g12:complex:modulus}{módulo} invertível $ n $ e
        cujo teorema fornece o inverso.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Check digits.}
\begin{enumerate}[resume]
  \item ISBN-10: os dez dígitos $ d_1 \dots d_{10}$ de um livro
        o código deve satisfazer
        $10d_1 + 9d_2 + \dots + 2d_9 + 1d_{10} \equiv 0
        \pmod{11}$. Verifique o verdadeiro ISBN
        $0\,306\,40615\,2$.
  \item Prove que o esquema ISBN detecta \emph{todo}
        erro de um dígito: se um dígito mudar em
        $ d \not\equiv 0$, a soma ponderada muda em $ w d $
        com $1 \leq w \leq 10$ --- por que isso nunca pode ser
        $\equiv 0 \pmod{11}$
        (\cref{thm:g12:arith:gauss})?
  \item Prove que também detecta toda transposição de dois
        dígitos adjacentes (distintos). Depois explique o design
        segredo: qual propriedade de $11$ fez ambas as provas funcionarem,
        e o que poderia dar errado com \omterm{def:g12:complex:modulus}{módulo} $10$?
  \item Os códigos de barras EAN-13 pesam os dígitos $1, 3, 1, 3, \dots $
        \omterm{def:g12:complex:modulus}{módulo} $10$. Calcule o dígito de verificação completando
        $978\,2940199\,05$. Quais transposições adjacentes fazem
        EAN \emph{falhar} detectar? (Quando é
        $2(a - b) \equiv 0 \pmod{10}$?)
  \item Os cartões de crédito usam o esquema de Luhn: da direita, duplo
        cada segundo dígito (subtraindo $9$ quando o duplo
        excede $9$), soma tudo e exige um múltiplo de
        $10$. Verifique o número do teste
        $4539\,1488\,0343\,6467$.
  \item Em uma frase: o que o \omterm{def:g12:arith:prime}{melhor} \omterm{def:g12:complex:modulus}{módulo} comprar ISBN
        that EAN and Luhn, chained to $10$, não pode ter?
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Fermat's lock.}
\begin{enumerate}[resume]
  \item Uma armadilha antes do tesouro: calcule
        $2^{10} \bmod 341$, deduzir $2^{340} \bmod 341$ --- e
        então fatore $341$. O que este exemplo (um
        \emph{Fermat pseudoprimo}) diz sobre o uso do Fermat
        pequeno teorema como um teste de primalidade?
  \item RSA em miniatura: pegue $ p = 3$, $ q = 11$, então
        $ n = 33$ e $(p-1)(q-1) = 20$; o expoente público é
        $ e = 3$. Encontre o expoente privado $ d $ com
        $3d \equiv 1 \pmod{20}$ (método da questão 4).
  \item Criptografar a mensagem $ m = 4$: calcular
        $ c = m^3 \bmod 33$.
  \item Descriptografar: calcular $ c^d \bmod 33$ (usar
        $ c \equiv -2 \pmod{33}$) e recupere a mensagem.
  \item Por que a descriptografia sempre funciona: mostre isso
        $ m^{21} \equiv m $ ambos \omterm{def:g12:complex:modulus}{módulo} $3$ e \omterm{def:g12:complex:modulus}{módulo} $11$
        (pequeno teorema de Fermat em cada mundo), e concluir
        \omterm{def:g12:complex:modulus}{módulo} $33$ (\cref{thm:g12:arith:gauss} cola os dois
        congruências). Onde foi $21 = ed $ formulário especial
        $1 + 20k $ digitar?
  \item A segurança da fechadura: todo mundo sabe $ n $ e $ e $;
        recuperando $ d $ requer $(p-1)(q-1)$, daí os fatores
        de $ n $. Nosso $33$ fatores à vista --- por que o mesmo
        esquema, com $ n $ de seiscentos dígitos, proteja o
        bancos do mundo? (Uma frase sobre a assimetria entre
        multiplicando e \omterm{def:g10:algebra:expand}{factoring}.)
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Classics.}
\begin{enumerate}[resume]
  \item A antiga contagem de soldados chineses (compare
        \cref{exo:g12:arith:10}): vários soldados saem
        restante $2$ quando classificado por $3$ e restante $3$ quando
        classificado por $5$. Encontre todas as contagens possíveis e explique
        porque a resposta é \omterm{def:g12:complex:modulus}{módulo} único $15$.
  \item Finalmente provas de uma linha: de $10 \equiv 1 \pmod 9$,
        provar que todo número é congruente com a soma de seus algarismos
        \omterm{def:g12:complex:modulus}{módulo} $9$; de $10 \equiv -1 \pmod{11}$, derivar o
        regra de soma alternada para $11$. (O Ensino Médio
        o volume provou isso com álgebra explícita --- admiro
        a compressão.)
  \item Finale --- Hardy contra o código de barras: recapitule o
        kit de ferramentas do capítulo (congruência \omterm{def:g12:seq:arith-geom}{aritmética}, Bézout
        inversos, pequeno teorema de Fermat, colando \omterm{def:g12:arith:gcd}{coprime}
        \omterm{def:g12:complex:modulus}{módulos}) e onde cada um se encaixou neste
        problema; então dê o veredicto moderno sobre
        ``Imaculado''.
\end{enumerate}
