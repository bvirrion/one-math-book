\chapter{Límites y continuidad}\label{ch:g12:limcont}

Este capítulo amplía la noción de límite de \omterm{def:g12:seq:sequence}{secuencias} a \omterm{def:g11:func:function}{funciones} de un
variable real e introduce \omterm{def:g12:limcont:continuity}{continuidad}. El resultado central es el
teorema del valor intermedio, que convierte la información cualitativa (``$f$ es
\omterm{def:g12:limcont:continuity}{continuo} y cambia de signo'') a la existencia de soluciones de \omterm{def:g10:algebra:equation}{ecuaciones}.

\section{Límites de funciones}

En todo momento, $f$ es un \omterm{def:g11:func:function}{función} definido en un \omterm{def:g10:numbers:interval}{intervalo} o un \omterm{def:g10:numbers:interunion}{unión} de \omterm{def:g10:numbers:interval}{intervalos}.
$D \subseteq \R$.

\begin{definition}[Límite al infinito]\label{def:g12:limcont:liminf}
Dejemos que $f$ se defina en \omterm{def:g10:numbers:interval}{intervalo}, $\intco{a}{+\infty}$ y $\ell \in \R$. Nosotros
decir que $f$ \emph{tends to $\ell$ at $+\infty$}\index{límite!de una función},
escrito $\lim\limits_{x\to+\infty} f(x) = \ell$, si cada \omterm{def:g10:numbers:interval}{intervalo} abierto
que contiene $\ell$ contiene todos los valores $f(x)$ para $x$ lo suficientemente grandes:
para cada $\varepsilon > 0$ existe $A$ tal que para todos $x \geq A$,
$\abs{f(x) - \ell} \leq \varepsilon$.

Decimos que $f$ tiende a $+\infty$ en $+\infty$ si por cada $M \in \R$ hay
existe $A$ tal que $f(x) \geq M$ para todos los $x \geq A$. Límites en $-\infty$
y los límites iguales a $-\infty$ se definen de manera análoga.
\end{definition}

\begin{definition}[Límite en un punto]\label{def:g12:limcont:limpoint}
Dejemos que $a \in \R$ y $f$ se definan cerca de $a$ (excepto posiblemente en $a$). Nosotros
digamos que $f$ tiende a $\ell \in \R$ en $a$, escrito
$\lim\limits_{x \to a} f(x) = \ell$, si por cada $\varepsilon > 0$ hay
existe $\delta > 0$ tal que para todos los $x \in D$ con
$\abs{x - a} \leq \delta$, $\abs{f(x) - \ell} \leq \varepsilon$.

Decimos que $f$ tiende a $+\infty$ en $a$ si por cada $M$ hay $\delta > 0$
tal que $f(x) \geq M$ siempre que $x \in D$ y $\abs{x - a} \leq \delta$.
Los límites unilaterales ($x \to a^+$, $x \to a^-$) restringen $x$ a $x > a$ o
$x < a$.
\end{definition}

\begin{example}
$\lim\limits_{x\to+\infty} \frac{1}{x} = 0$,
$\lim\limits_{x\to 0^+} \frac{1}{x} = +\infty$,
$\lim\limits_{x\to 0^-} \frac{1}{x} = -\infty$. El \omterm{def:g11:func:function}{función}
$x \mapsto \frac1x$ no tiene límite (de dos caras) en $0$.
\end{example}

\begin{definition}[Asíntotas]\label{def:g12:limcont:asymptote}
La línea $y = \ell$ es una \emph{asíntota horizontal}\index{asíntota} de la
curva de $f$ en $+\infty$ (resp.\ $-\infty$) si
$\lim_{x\to+\infty} f(x) = \ell$ (resp.\ en $-\infty$). La línea $x = a$ es una
\emph{asíntota vertical}\index{asíntota} si $f$ tiene un límite unilateral infinito en $a$.
\end{definition}

\begin{proposition}[Operaciones en límites]\label{prop:g12:limcont:operations}
Las reglas de \cref{prop:g12:seq:operations} (sumas, productos, cocientes) se cumplen
textualmente para límites de \omterm{def:g11:func:function}{funciones}, en un punto o en el infinito, con el mismo
cuatro formas indeterminadas $(+\infty)+(-\infty)$, $0\times\infty$,
$\frac{\infty}{\infty}$, $\frac{0}{0}$.
\end{proposition}

\begin{proof}
Las pruebas son idénticas al caso \omterm{def:g12:seq:sequence}{secuencia}, reemplazando ``for $n \geq N$''
por ``para $x \geq A$'' o ``para $\abs{x-a} \leq \delta$''.
\end{proof}

\begin{theorem}[Límite de una composición]\label{thm:g12:limcont:composition}
Sea $f, g$ \omterm{def:g11:func:function}{funciones} y $a, b, c$ sea \omterm{def:g10:numbers:sets}{números reales} o $\pm\infty$. si
$\lim\limits_{x \to a} f(x) = b$ y $\lim\limits_{y \to b} g(y) = c$, entonces
\[
\lim_{x \to a} g\bigl(f(x)\bigr) = c .
\]
La misma afirmación se aplica con \omterm{def:g12:seq:sequence}{secuencia} en lugar de $f$: si
$u_n \to b$ y $\lim_{y\to b} g(y) = c$, luego $g(u_n) \to c$.
\end{theorem}

\begin{proof}
Tome $b, c$ finito (los demás casos son análogos). Deje $\varepsilon > 0$.
Existe $\delta > 0$ tal que $\abs{y - b} \leq \delta$ implica
$\abs{g(y) - c} \leq \varepsilon$; y hay $\delta' > 0$ tal que
$\abs{x - a} \leq \delta'$ implica $\abs{f(x) - b} \leq \delta$. Encadenando el
dos da $\abs{g(f(x)) - c} \leq \varepsilon$ para
$\abs{x - a} \leq \delta'$.
\end{proof}

\begin{theorem}[Comparación y compresión]\label{thm:g12:limcont:squeeze}
Sea $f, g, h$ \omterm{def:g11:func:function}{funciones} y $a \in \R \cup \{\pm\infty\}$.
\begin{enumerate}
  \item Si $f \leq g$ está cerca de $a$ y $f(x) \to +\infty$ como $x \to a$, entonces
        $g(x) \to +\infty$.
  \item Si $f \leq g \leq h$ cerca de $a$ y $f, h$ tienden al mismo finito
        limite $\ell$ a $a$, luego $g(x) \to \ell$ a $x \to a$.
\end{enumerate}
\end{theorem}

\begin{proof}
Mismo argumento que para \omterm{def:g12:seq:sequence}{secuencias} (\cref{thm:g12:seq:squeeze}).
\end{proof}

\begin{method}[Límites de cálculo en la práctica]\label{met:g12:limcont:limits}
\begin{itemize}
  \item Polinomios y racionales \omterm{def:g11:func:function}{funciones} en $\pm\infty$: factorizar el
        potencia más alta de $x$; el límite es el de la relación de avance
        términos.
  \item $\frac{0}{0}$ en un punto $a$: factorizar $(x-a)$ del numerador y
        denominador, o reconocer un \omterm{def:g11:deriv:derivative}{derivado}
        $\lim_{x\to a}\frac{f(x)-f(a)}{x-a} = f'(a)$ (\cref{ch:g12:deriv}).
  \item Raíces cuadradas: multiplicar por la expresión conjugada.
  \item Factores oscilantes ($\cos$, $\sin$): utilice el teorema de compresión.
\end{itemize}
\end{method}

\section{Continuidad}

\begin{definition}[continuidad]\label{def:g12:limcont:continuity}
Dejemos que $f$ se defina en \omterm{def:g10:numbers:interval}{intervalo}, $I$ y $a \in I$. El \omterm{def:g11:func:function}{función} $f$ es
\emph{continuo en $a$}\index{continuidad} si
$\lim\limits_{x \to a} f(x) = f(a)$. Es \emph{continuo en $I$}\index{continuo} si es
continuo en cada punto de $I$.
\end{definition}

Intuitivamente se puede dibujar el \omterm{def:g10:functions:graph}{gráfico} de un \omterm{def:g12:limcont:continuity}{continuo} \omterm{def:g11:func:function}{función} en un \omterm{def:g10:numbers:interval}{intervalo}
sin levantar el bolígrafo.

\begin{proposition}[Continuidad de funciones habituales]\label{prop:g12:limcont:usual}
Polinomios racionales \omterm{def:g11:func:function}{funciones}, $x \mapsto \sqrt{x}$, $x \mapsto \abs{x}$,
$\cos$, $\sin$, $\exp$ y $\ln$ son \omterm{def:g12:limcont:continuity}{continuo} en cada \omterm{def:g10:numbers:interval}{intervalo} contenido
en su \omterm{def:g11:func:function}{dominio}. Sumas, productos, cocientes (donde se definan) y composiciones
de \omterm{def:g12:limcont:continuity}{continuo} \omterm{def:g11:func:function}{funciones} son \omterm{def:g12:limcont:continuity}{continuo}.
\end{proposition}

\begin{proof}[prueba parcial]
La estabilidad bajo sumas, productos, cocientes y composición se deriva de la
teoremas correspondientes sobre límites. \omterm{def:g12:limcont:continuity}{Continuidad} de $x \mapsto x$ y de constante
\omterm{def:g11:func:function}{funciones} es inmediato de la definición, por lo tanto polinomios y racionales
\omterm{def:g11:func:function}{funciones} son \omterm{def:g12:limcont:continuity}{continuo}. \omterm{def:g12:limcont:continuity}{Continuidad} del \omterm{def:g11:func:function}{funciones} habitual restante es
probados en los capítulos donde se construyen o admiten.
\end{proof}

\begin{example}
El \emph{floor \omterm{def:g11:func:function}{función}} $x \mapsto \floor{x}$ (el mayor \omterm{def:g10:numbers:sets}{entero}
$\leq x$) no es \omterm{def:g12:limcont:continuity}{continuo} en ningún \omterm{def:g10:numbers:sets}{entero} $n$: su límite izquierdo existe
$n - 1$ mientras que $\floor{n} = n$.
\end{example}

\begin{proposition}[Secuencias y continuidad]\label{prop:g12:limcont:seqcont}
Si $u_n \to \ell$ y $f$ es \omterm{def:g12:limcont:continuity}{continuo} en $\ell$, entonces $f(u_n) \to f(\ell)$.
En particular, si un \omterm{def:g12:seq:sequence}{secuencia} definido por $u_{n+1} = f(u_n)$ \omterm{def:g12:seq:limit}{converge} para
$\ell$ y $f$ es \omterm{def:g12:limcont:continuity}{continuo} en $\ell$, luego $f(\ell) = \ell$.
\end{proposition}

\begin{proof}
El primer reclamo es \cref{thm:g12:limcont:composition} aplicado con el
\omterm{def:g12:seq:sequence}{secuencia} $(u_n)$. Para el segundo, pase al límite a ambos lados de
$u_{n+1} = f(u_n)$: el lado izquierdo tiende a $\ell$, el lado derecho a
$f(\ell)$ y los límites son únicos.
\end{proof}

\section{El teorema del valor intermedio}

\begin{theorem}[Teorema del valor intermedio]\label{thm:g12:limcont:ivt}
\index{teorema del valor intermedio}
Sea $f$ \omterm{def:g12:limcont:continuity}{continuo} en $\intcc{a}{b}$ y $k$ sea cualquier número entre
$f(a)$ y $f(b)$. Entonces existe $c \in \intcc{a}{b}$ tal que
$f(c) = k$.
\end{theorem}

{\small El teorema del valor intermedio: una curva \omterm{def:g12:limcont:continuity}{continuo} que se une
$(a, f(a))$ a $(b, f(b))$ deben cruzar cada línea horizontal
$y = k$ se encuentra entre $f(a)$ y $f(b)$.}
\end{omfigure}

\begin{proof}
Reemplazando $f$ por $f - k$ (y por $k - f$ si es necesario), podemos asumir
$f(a) \leq 0 \leq f(b)$ y busque un cero de $f$. Construimos dos
\omterm{def:g12:seq:sequence}{secuencias} por \emph{dicotomía}\index{dicotomía}: establecer $a_0 = a$, $b_0 = b$;
dado $a_n \leq b_n$ con $f(a_n) \leq 0 \leq f(b_n)$, dejemos
$m = \frac{a_n + b_n}{2}$ y definir
\[
(a_{n+1}, b_{n+1}) =
\begin{cases}
(a_n, m) & \text{if } f(m) \geq 0,\\
(m, b_n) & \text{if } f(m) < 0.
\end{cases}
\]
En ambos casos $f(a_{n+1}) \leq 0 \leq f(b_{n+1})$, el \omterm{def:g12:seq:sequence}{secuencia} $(a_n)$ es
\omterm{def:g12:seq:monotonic}{creciente}, $(b_n)$ es \omterm{def:g10:functions:variations}{decreciente}, y
$b_n - a_n = \frac{b-a}{2^n} \to 0$: los \omterm{def:g12:seq:sequence}{secuencias} son adyacentes (ver
\cref{exo:g12:seq:10}) y convergen a un límite común
$c \in \intcc{a}{b}$. Por \omterm{def:g12:limcont:continuity}{continuidad}, $f(a_n) \to f(c)$ y
$f(b_n) \to f(c)$; pasando al límite en $f(a_n) \leq 0$ y
$f(b_n) \geq 0$ da $f(c) \leq 0 \leq f(c)$, por lo que $f(c) = 0$.
\end{proof}

\begin{remark}
La prueba \omterm{thm:g12:limcont:ivt}{dicotomía} es un algoritmo: produce resultados arbitrariamente precisos.
\omterm{def:g10:numbers:approx}{aproximaciones} de una solución, reduciendo a la mitad el error en cada paso. Así es como un
La computadora puede resolver $f(x) = 0$ sabiendo solo cómo evaluar $f$.
\end{remark}

\begin{theorem}[Teorema de biyección]\label{thm:g12:limcont:bijection}
Sea $f$ \omterm{def:g12:limcont:continuity}{continuo} y \emph{strictly \omterm{def:g12:seq:monotonic}{monótono}} en un \omterm{def:g10:numbers:interval}{intervalo} $I$. entonces
para cada $k$ estrictamente entre los límites de $f$ en los puntos finales de $I$, el
\omterm{def:g10:algebra:equation}{ecuación} $f(x) = k$ tiene una solución \emph{único} en $I$.
\end{theorem}

\begin{proof}
Existencia: elegir $a, b \in I$ con $k$ entre $f(a)$ y $f(b)$ (posible
ya que los valores de $f$ se acercan a sus límites de punto final), y aplique el
Teorema del valor intermedio en $\intcc{a}{b}$. Unicidad: si $x < y$, estricto
La monotonicidad da $f(x) \neq f(y)$, por lo que dos puntos distintos no pueden ser ambos.
soluciones.
\end{proof}

\begin{method}[Resolviendo $f(x) = k$ con el teorema de biyección]\label{met:g12:limcont:bijection}
Para demostrar que $f(x) = k$ tiene exactamente una solución en un \omterm{def:g10:numbers:interval}{intervalo}:
\begin{enumerate}
  \item establece un \omterm{def:g10:functions:table}{tabla de variaciones} de $f$ (signo de $f'$, ver
        \cref{ch:g12:deriv});
  \item en cada \omterm{def:g10:numbers:interval}{intervalo} donde $f$ es \omterm{def:g12:limcont:continuity}{continuo} y estrictamente \omterm{def:g12:seq:monotonic}{monótono},
        comparar $k$ con los valores o límites del punto final;
  \item aplica el teorema de biyección en cada uno de esos \omterm{def:g10:numbers:interval}{intervalo} y cuenta el
        soluciones.
\end{enumerate}
Luego, una calculadora o \omterm{thm:g12:limcont:ivt}{dicotomía} localiza cada solución con la precisión deseada.
\end{method}

\begin{example}\label{ex:g12:limcont:cubic}
El \omterm{def:g10:algebra:equation}{ecuación} $x^3 + x = 1$ tiene una solución real única. De hecho
$f(x) = x^3 + x$ es \omterm{def:g12:limcont:continuity}{continuo} y estrictamente \omterm{def:g12:seq:monotonic}{creciente} en $\R$ (como suma de
estrictamente \omterm{def:g11:func:monotone}{funciones crecientes}), $\lim_{-\infty} f = -\infty$ y
$\lim_{+\infty} f = +\infty$, por lo que el teorema de biyección se aplica con $k = 1$.
Desde $f(0.6) = 0.816 < 1$ y $f(0.7) = 1.043 > 1$, la solución está en
$\intoo{0.6}{0.7}$.
\end{example}

\section{Ceremonias}

\begin{exercise}[$\star$]\label{exo:g12:limcont:1}
Calcule los siguientes límites:
\[
\lim_{x\to+\infty} \frac{2x^3 - x + 1}{5x^3 + x^2}, \qquad
\lim_{x\to-\infty} \bigl(x^2 + 3x - 1\bigr), \qquad
\lim_{x\to 2^+} \frac{x+1}{x-2}, \qquad
\lim_{x\to+\infty} \frac{\cos x}{x}.
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:limcont:2}
Sea $f(x) = \dfrac{2x^2 - 3x + 1}{x - 1}$, definido para $x \neq 1$.
\begin{enumerate}
  \item Muestre que $f(x) = 2x - 1$ para todos los $x \neq 1$.
  \item Deducir $\lim\limits_{x \to 1} f(x)$. ¿Se puede extender $f$ a un
        \omterm{def:g11:func:function}{función} \omterm{def:g12:limcont:continuity}{continuo} en $1$?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:limcont:3}

\begin{exercise}[$\star\star$]\label{exo:g12:limcont:5}

\section{Problema: No se puede cruzar un río sin mojarse}

\begin{problem}[{Problema de fin de semana --- el valor intermedio
teorema fuerza puntos fijos, antípodas de igual temperatura y
tablas estables: tres teoremas que parecen imposibles a partir de una idea}]
\label{pb:g12:limcont:1}
Arrugar un mapa de Francia y dejarlo caer en cualquier lugar del suelo francés: algunos
punto del mapa se encuentra \emph{exactamente} encima del lugar donde
representa. En cada instante, dos puntos diametralmente opuestos
del ecuador tienen \emph{exactamente} la misma temperatura. y un
Una mesa cuadrada que se tambalea sobre un terreno lleno de baches siempre se puede estabilizar con
\emph{giratorio}. Tres trucos, un teorema: a \omterm{def:g12:limcont:continuity}{continuo}
La cantidad no puede pasar de negativo a positivo sin desaparecer.
(\cref{thm:g12:limcont:ivt}). Este problema entrena límites, entonces
construye las tres maravillas --- y sabe exactamente cuál es el teorema
no promete.

\medskip
\noindent\textbf{Parte I --- Limit fluency.}
\begin{enumerate}
  \item Calcular $\lim_{x \to +\infty} \dfrac{3x^2 - x}{x^2 + 5}$,
        $\lim_{x \to +\infty} \dfrac{2x + 1}{x^2 + 1}$,\ y
        $\lim_{x \to +\infty} (x^3 - x^2)$
        (\cref{met:g12:limcont:limits}).
  \item Dar el \omterm{def:g12:limcont:asymptote}{asíntotas} de
        $f(x) = \dfrac{3x - 1}{x + 2}$
        (\cref{def:g12:limcont:asymptote}), calculando el
        límites pertinentes.
  \item Calcular $\lim_{x \to 3} \dfrac{x^2 - 9}{x - 3}$
        (factor) y
        $\lim_{x \to 0} \dfrac{\sqrt{x + 1} - 1}{x}$
        (conjugado).
  \item Por el teorema del apretón
        (\cref{thm:g12:limcont:squeeze}): calcular
        $\lim_{x \to 0} x \sin\frac1x$ y
        $\lim_{x \to +\infty} \frac{\sin x}{x}$.
  \item Por composición y términos dominantes:
        $\lim_{x \to +\infty} \dfrac{\sqrt{4x^2 + x}}{x}$.
\end{enumerate}

\medskip
\noindent\textbf{Parte II --- \omterm{def:g11:quad:discriminant}{Raíces}, certified.}
\begin{enumerate}[resume]
  \item Demuestre que el \omterm{def:g10:algebra:equation}{ecuación} $x^3 + x = 5$ tiene al menos un
        solución en $\intcc{1}{2}$.
  \item Mostrar que la solución es única en $\R$
        (\cref{thm:g12:limcont:bijection}).
  \item Prueba el clásico: \emph{every polynomial of odd degree
        has at least one real \omterm{def:g11:quad:discriminant}{raíz}}. (Escribe el argumento
        para $p(x) = x^3 + a x^2 + b x + c$: calcular los dos
        límites infinitos por término dominante, luego aplicar el IVT en
        un \omterm{def:g10:numbers:interval}{intervalo} grande.)
  \item Localizando el \omterm{def:g11:quad:discriminant}{raíz} de la pregunta 6 por \omterm{thm:g12:limcont:ivt}{dicotomía}: cuantos
        las reducciones a la mitad de $\intcc{1}{2}$ garantizan un error a continuación
        $10^{-6}$ (\cref{exo:g12:limcont:9})? Comparar con el
        duplicación de dígitos del método de Newton
        (\cref{pb:g11:deriv:1}): aproximadamente ¿cuántos pasos se necesitarían?
        ¿Necesita?
  \item El contraejemplo que guarda el teorema:
        $g(x) = \frac1x$ satisface $g(-1) = -1 < 0$ y
        $g(1) = 1 > 0$, pero nunca desaparece. ¿Qué hipótesis de
        el IVT falla --- y ¿qué enseña el ejemplo sobre
        comprobar hipótesis?
\end{enumerate}

\medskip
\noindent\textbf{Parte III --- Three impossible-sounding
theorems.}
\begin{enumerate}[resume]
  \item El teorema del punto fijo del segmento: si
        $f : \intcc{0}{1} \to \intcc{0}{1}$ es \omterm{def:g12:limcont:continuity}{continuo}, prueba
        que algún $c$ satisface $f(c) = c$. (Estudiar
        $g(x) = f(x) - x$ en los dos extremos.)
  \item Traduce la pregunta 11 al mapa arrugado: qué juega
        el papel de $f$, ¿por qué aterriza en
        $\intcc{0}{1}$ (en sentido figurado: dentro de Francia), y qué
        ¿Está el punto exactamente encima de su propia ubicación? (Uno
        dimensión se admite en silencio --- el honesto
        El enunciado bidimensional es el teorema de Brouwer.
        volúmenes universitarios.)
  \item El ecuador: sea $T$ una temperatura \omterm{def:g12:limcont:continuity}{continuo} en un
        círculo, visto como un $2\pi$-\omterm{def:g11:func:function}{función} periódico con
        $T(0) = T(2\pi)$. Establezca $g(t) = T(t) - T(t + \pi)$.
        Calcule $g(0) + g(\pi)$, deduzca que $g$ desaparece
        en algún lugar de $\intcc{0}{\pi}$ y establece el teorema
        sobre los puntos antípodas.
  \item La mesa tambaleante: una mesa cuadrada perfectamente rígida
        en \omterm{def:g12:limcont:continuity}{continuo} (¡sin escalones!) terreno lleno de baches; tres piernas
        siempre se toca, uno flota por una altura $h(\theta)$ que

  \item El excursionista de \cref{exo:g12:limcont:8} usó el mismo
        esquema. Indique la plantilla compartida por las preguntas 11, 13,
        14 y el excursionista --- construye cuál \omterm{def:g11:func:function}{función}, verifica qué
        al final, invoca what --- en tres líneas.
  \item Lo que da el IVT \emph{no}: la existencia salió gratis,
        pero ¿qué dos cuestiones deja abiertas y cuáles
        ¿Las herramientas de este problema las responden (preguntas 7 y 9)?
\end{enumerate}

\medskip
\noindent\textbf{Parte IV --- Portraits with \omterm{def:g12:limcont:asymptote}{asíntotas}.}
\begin{enumerate}[resume]
  \item Reescribe $f(x) = \dfrac{3x - 1}{x + 2}$ como
        $3 - \dfrac{7}{x + 2}$ y úsalo para describir la curva.
        completamente: \omterm{def:g12:limcont:asymptote}{asíntotas}, variaciones en cada lado,
        boceto. (Un \omterm{def:g11:func:reference}{hipérbola} desplazado --- la curva de costo promedio
        de \cref{pb:g10:reffunc:1} también lo era).
  \item El \omterm{def:g11:func:function}{función} $f(x) = \dfrac{x^2 - 1}{x - 1}$ es
        indefinido en $1$. ¿Qué valor asignado a $f(1)$ hace
        es \omterm{def:g12:limcont:continuity}{continuo} allí --- y ¿por qué los analistas llaman a tal
        ¿Señala una discontinuidad \emph{desmontable}?
  \item El piso \omterm{def:g11:func:function}{función} $x \mapsto \lfloor x \rfloor$
        (mayor \omterm{def:g10:numbers:sets}{entero} $\leq x$): dónde está \omterm{def:g12:limcont:continuity}{continuo},
        ¿Dónde salta y por qué la conclusión del IVT?
        falla (busca $a < b$ con
        $\lfloor a \rfloor < \frac12 < \lfloor b \rfloor$ todavía
        ¿No hay solución de $\lfloor x \rfloor = \frac12$)?
  \item Finale --- Las dos caras de \omterm{def:g12:limcont:continuity}{continuidad}: la promesa local
        (valor $=$ límite: sin teletransporte) y el poder global
        (IVT: cada valor intermedio visitado). Archivar los cuatro
        maravillas de la Parte III bajo la cara derecha, y dar la
        lema del cruce de ríos su nombre matemático preciso.
\end{enumerate}
\end{problem}
