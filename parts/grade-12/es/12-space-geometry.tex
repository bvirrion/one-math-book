Este capítulo construye esta caja de herramientas y la utiliza para resolver los problemas clásicos.
\omterm{def:g10:numbers:interunion}{intersección} y problemas de distancia.

\section{Vectores en el espacio}

\omterm{def:g11:vect:vector}{Vectores} en el espacio obedecen a las mismas reglas que en el avión: suman, escalan,
y $\vect{AB} = \vect{CD}$ exactamente cuando $ABDC$ es un paralelogramo. en un
\omterm{def:g10:coordgeom:system}{sistema de coordenadas} $(O; \vec\imath, \vec\jmath, \vec k)$, cada \omterm{def:g11:vect:vector}{vector} tiene
\omterm{def:g10:coordgeom:system}{coordenadas} $(x, y, z)$.

\begin{definition}[Colinealidad, coplanaridad]\label{def:g12:space:collinear}
Dos \omterm{def:g11:vect:vector}{vectores} son \emph{colineal} si uno es múltiplo del otro. tres
\omterm{def:g11:vect:vector}{vectores} $\vec u, \vec v, \vec w$ son \emph{coplanar}\index{coplanar} si uno
de ellos se puede escribir como una combinación de los otros dos, digamos
$\vec w = a\vec u + b\vec v$.
\end{definition}

\begin{proposition}\label{prop:g12:space:basis}
Si $\vec u, \vec v, \vec w$ no es \omterm{def:g12:space:collinear}{coplanar}, cada \omterm{def:g11:vect:vector}{vector} de espacio
se descompone únicamente como $x\vec u + y\vec v + z\vec w$.
\end{proposition}

\begin{proof}[Idea de prueba]
A través de la punta del \omterm{def:g11:vect:vector}{vector} para descomponer, trazar la línea paralela a
$\vec w$; se encuentra con el plano de $\vec u, \vec v$ en un solo punto, lo que
divide el \omterm{def:g11:vect:vector}{vector} en un componente en ese plano (exclusivamente $x\vec u + y\vec
v$, geometría del plano) y un componente a lo largo de $\vec w$. Singularidad: una diferencia
de dos descomposiciones expresaría uno de los \omterm{def:g11:vect:vector}{vectores} en términos del otro
dos, contradiciendo la no coplanaridad.
\end{proof}

\section{lineas y planos}

\begin{definition}[Representación paramétrica de una línea]\label{def:g12:space:line}
La línea que pasa por $A(x_A, y_A, z_A)$ con \omterm{def:g11:vect:direction}{vector de direccion}
$\vec u (a, b, c) \neq \vec 0$ es el conjunto de puntos
\[
M(x_A + ta,\; y_A + tb,\; z_A + tc), \qquad t \in \R .
\]
\end{definition}

\begin{definition}[Avión]\label{def:g12:space:plane}
El avión que pasa por $A$ dirigido por dos \omterm{def:g12:space:collinear}{no colineal} \omterm{def:g11:vect:vector}{vectores}
$\vec u, \vec v$ es el conjunto de puntos $M$ con
$\vect{AM} = s\vec u + t\vec v$, $(s, t) \in \R^2$.
\end{definition}

\begin{proposition}[Posiciones relativas]\label{prop:g12:space:positions}
Dos planos distintos son paralelos o se cruzan en una línea. Una línea y un
plano son paralelos (posiblemente contenidos) o se cruzan en una sola
punto. Dos líneas en el espacio pueden ser intersectadas, estrictamente paralelas, idénticas,
o \emph{sesgar} (\omterm{def:g12:space:collinear}{no coplanar}).
\end{proposition}

\begin{proof}[boceto]
Éstas son declaraciones de incidencia; El análisis de cada caso reduce, en \omterm{def:g10:coordgeom:system}{coordenadas},
para resolver un \omterm{def:g10:lines:system}{sistema lineal} y contar sus soluciones --- ver
\cref{met:g12:space:intersections}.
\end{proof}

\section{Producto escalar}

\begin{definition}[Escalar producto]\label{def:g12:space:dot}
El \emph{producto escalar}\index{producto escalar} de $\vec u$ y $\vec v$ es
\[
\vec u \cdot \vec v = \tfrac12\left(\norm{\vec u + \vec v}^2
- \norm{\vec u}^2 - \norm{\vec v}^2\right),
\]
donde $\norm{\vec u}$ es la longitud de $\vec u$. Si ambos \omterm{def:g11:vect:vector}{vectores} son
distinto de cero, $\vec u \cdot \vec v = \norm{\vec u}\,\norm{\vec v}\cos\theta$
donde $\theta$ es el ángulo entre ellos; y en un \emph{\omterm{def:g10:coordgeom:system}{ortonormal}}
\omterm{def:g10:coordgeom:system}{sistema de coordenadas},
\[
\vec u \cdot \vec v = xx' + yy' + zz' .
\]
Dos \omterm{def:g11:vect:vector}{vectores} son \emph{\omterm{def:g11:scal:orthogonal}{ortogonal}} si su producto escalar es $0$.
\end{definition}

\begin{proposition}[Propiedades]\label{prop:g12:space:dotprops}
El \omterm{def:g12:space:dot}{producto escalar} es simétrico ($\vec u\cdot\vec v = \vec v\cdot\vec u$),
bilineal ($(a\vec u + b\vec v)\cdot \vec w = a\,\vec u\cdot\vec w +
b\,\vec v\cdot \vec w$), y
$\vec u \cdot \vec u = \norm{\vec u}^2 \geq 0$.
\end{proposition}

\begin{proof}
En un \omterm{def:g10:coordgeom:system}{sistema ortonormal}, las tres propiedades son inmediatas en la fórmula
$xx' + yy' + zz'$, que a su vez se deriva de la fórmula definitoria y de la
Cálculo pitagórico de longitudes:
$\norm{\vec u}^2 = x^2 + y^2 + z^2$.
\end{proof}

\begin{definition}[vectores normales]\label{def:g12:space:normal}
Un \emph{vectores normales}\index{vectores normales} de un avión $\mathcal P$ es un
distinto de cero \omterm{def:g11:vect:vector}{vector} \omterm{def:g11:scal:orthogonal}{ortogonal} a cada \omterm{def:g11:vect:vector}{vector} que se encuentra en $\mathcal P$ (es
basta que sea \omterm{def:g11:scal:orthogonal}{ortogonal} a dos direcciones \omterm{def:g12:space:collinear}{no colineal} de
$\mathcal P$).
\end{definition}

\begin{theorem}[Ecuación cartesiana de un plano]\label{thm:g12:space:planeeq}
En un \omterm{def:g10:coordgeom:system}{ortonormal} \omterm{def:g10:coordgeom:system}{sistema de coordenadas}, el avión que pasa
$A$ con \omterm{def:g12:space:normal}{vectores normales} $\vec n(a, b, c)$ tiene \omterm{def:g10:algebra:equation}{ecuación}
\[
a x + b y + c z + d = 0,
\]
donde $d = -(ax_A + by_A + cz_A)$. Por el contrario, cada uno de esos \omterm{def:g10:algebra:equation}{ecuación} con
$(a,b,c) \neq (0,0,0)$ define un plano con \omterm{def:g12:space:normal}{vectores normales} $(a, b, c)$.
\end{theorem}

\begin{proof}
$M \in \mathcal P \iff \vect{AM} \perp \vec n \iff
a(x - x_A) + b(y - y_A) + c(z - z_A) = 0$, que se expande a \omterm{def:g10:algebra:equation}{ecuación}.
Por el contrario, elija cualquier punto de solución $A$ de \omterm{def:g10:algebra:equation}{ecuación}; lo mismo
El cálculo al revés muestra que el conjunto solución es
$\{M : \vect{AM}\cdot\vec n = 0\}$, el avión que pasa por $A$ normal a
$\vec n$.
\end{proof}

\begin{method}[Intersecciones en la práctica]\label{met:g12:space:intersections}
\begin{itemize}
  \item \emph{Line $\cap$ plane}: sustituir el paramétrico \omterm{def:g10:coordgeom:system}{coordenadas} de
        la línea hacia el plano \omterm{def:g10:algebra:equation}{ecuación}; resolver para $t$ (una solución: una
        punto; $0 = $ distinto de cero: paralelo; $0 = 0$: línea contenida).
  \item \emph{Plane $\cap$ plane}: resuelve el sistema de los dos \omterm{def:g10:algebra:equation}{ecuaciones},
        parametrizar por una coordenada libre; la solución es una línea (o la
        los planos son paralelos cuando los \omterm{def:g12:space:normal}{vectores normales} son \omterm{def:g12:space:collinear}{colineal}).
  \item \emph{Comprobaciones de ortogonalidad}: línea $\perp$ plano si su dirección
        es \omterm{def:g12:space:collinear}{colineal} con lo normal; dos planos son perpendiculares si y sólo si
        los normales son \omterm{def:g11:scal:orthogonal}{ortogonal}.
\end{itemize}
\end{method}

\begin{proposition}[Distancia de un punto a un plano]\label{prop:g12:space:distance}
\index{distancia!apuntar al avión}
En un \omterm{def:g10:coordgeom:system}{sistema ortonormal}, la distancia desde $M_0(x_0, y_0, z_0)$ hasta el
el avión $\mathcal P : ax + by + cz + d = 0$ es
\[
\operatorname{dist}(M_0, \mathcal P)
= \frac{\abs{ax_0 + by_0 + cz_0 + d}}{\sqrt{a^2 + b^2 + c^2}}.
\]
\end{proposition}

\begin{proof}
Sea $H$ la proyección \omterm{def:g11:scal:orthogonal}{ortogonal} de $M_0$ sobre $\mathcal P$: el pie de
la perpendicular, por lo que $\vect{HM_0}$ es \omterm{def:g12:space:collinear}{colineal} con la unidad normal
$\frac{\vec n}{\norm{\vec n}}$, y la distancia es
$\abs{\vect{HM_0}\cdot \frac{\vec n}{\norm{\vec n}}}$. Para cualquier
$H \in \mathcal P$,
$\vect{HM_0}\cdot\vec n = ax_0 + by_0 + cz_0 - (ax_H + by_H + cz_H)
= ax_0 + by_0 + cz_0 + d$ (usando \omterm{def:g10:algebra:equation}{ecuación} en $H$), de donde la fórmula
después de dividir por $\norm{\vec n} = \sqrt{a^2+b^2+c^2}$.
\end{proof}

\begin{example}\label{ex:g12:space:distance}
Distancia del origen al avión $x + 2y + 2z - 6 = 0$:
$\frac{\abs{-6}}{\sqrt{1 + 4 + 4}} = \frac63 = 2$.
\end{example}

\section{Ceremonias}

En todos los ejercicios el \omterm{def:g10:coordgeom:system}{sistema de coordenadas} es \omterm{def:g10:coordgeom:system}{ortonormal}.

\begin{exercise}[$\star$]\label{exo:g12:space:1}
Sean $A(1, 0, 2)$, $B(3, 1, 1)$ y $C(2, -1, 3)$. ¿Están los puntos $A$, $B$,
$C$ alineado? Dé una representación paramétrica de la línea $(AB)$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:space:2}
Determine un cartesiano \omterm{def:g10:algebra:equation}{ecuación} del plano que pasa por $A(1, 1, 0)$ con
\omterm{def:g12:space:normal}{vectores normales} $\vec n(2, -1, 3)$. ¿Le pertenece el punto $B(0, 2, 1)$?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:space:3}
Calcula el ángulo en $A$ (al grado más cercano) del triángulo con
vértices $A(0,0,0)$, $B(1, 1, 0)$ y $C(1, 0, 1)$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:space:4}
Considere la línea $\mathcal D$ hasta $A(1, 2, 0)$ con dirección
$\vec u(1, -1, 2)$ y el avión $\mathcal P : 2x + y - z + 1 = 0$.
Determinar $\mathcal D \cap \mathcal P$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:space:5}
Sean $\mathcal P : x - y + 2z = 1$ y $\mathcal Q : 2x + y + z = 4$.
Demuestre que $\mathcal P$ y $\mathcal Q$ se cruzan en una línea y dé un
representación paramétrica del mismo.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:space:6}
Demuestre que las líneas
\[
\mathcal D_1 : (x, y, z) = (1 + t,\; 2t,\; 3 - t), \qquad
\mathcal D_2 : (x, y, z) = (2 + s,\; 1 + s,\; 1 + 2s)
\]
están sesgados (no \omterm{def:g12:space:collinear}{coplanar}).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:space:7}
Vamos $\mathcal P : 2x - y + 2z - 3 = 0$.
\begin{enumerate}
  \item Calcule la distancia desde $M_0(3, 1, 1)$ a $\mathcal P$.
  \item Determinar el \omterm{def:g10:coordgeom:system}{coordenadas} de la proyección \omterm{def:g11:scal:orthogonal}{ortogonal} $H$ de
        $M_0$ en $\mathcal P$ y verifique la distancia $M_0H$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:space:8}
El cubo $ABCDEFGH$ tiene el lado $1$; coloque \omterm{def:g10:coordgeom:system}{coordenadas} para que
$A(0,0,0)$, $B(1,0,0)$, $D(0,1,0)$, $E(0,0,1)$ (con
$C = B + D$, $F = B + E$, $G = B + D + E$, $H = D + E$ como \omterm{def:g11:vect:vector}{vectores} de
$A$).
\begin{enumerate}
  \item Demuestre que la diagonal $(AG)$ es \omterm{def:g11:scal:orthogonal}{ortogonal} con respecto al plano $(BDE)$.
  \item Calcula la distancia desde $A$ al plano $(BDE)$ y el punto
        donde $(AG)$ lo perfora.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:space:9}
\emph{(Volumen de un tetraedro).} Vamos $A(1,1,1)$, $B(2,1,0)$, $C(0,2,1)$
y $D(2,2,2)$.
\begin{enumerate}
  \item Verifique que $\vect{AB}$, $\vect{AC}$, $\vect{AD}$ no sean \omterm{def:g12:space:collinear}{coplanar}
        (los puntos forman un tetraedro genuino).
  \item Encuentre un \omterm{def:g11:vect:vector}{vector} $\vec n$ \omterm{def:g11:scal:orthogonal}{ortogonal} para $\vect{BC}$ y
        $\vect{BD}$, y deducir un cartesiano \omterm{def:g10:algebra:equation}{ecuación} del plano $(BCD)$.
  \item Calcula la distancia desde $A$ al plano $(BCD)$ y el área de
        el triángulo $BCD$, usando la fórmula
        $\text{Area} = \frac12\sqrt{\norm{\vec u}^2\norm{\vec v}^2 -
        (\vec u \cdot \vec v)^2}$ para el triángulo abarcado por
        $\vec u, \vec v$.
  \item Deducir el volumen del tetraedro
        ($V = \frac13 \times \text{base area} \times \text{height}$).
\end{enumerate}
\end{exercise}

\begin{problem}[{Problema de fin de semana --- un avión corta un cubo en un

Corta un cubo y esperas cuadrados y rectángulos, pero uno

\medskip
\noindent\textbf{Parte I --- Fluency.}
\begin{enumerate}

        son paralelas: líneas \emph{sesgar}, el fenómeno que
        No puede suceder en un avión.
  \item Calcule la proyección \omterm{def:g11:scal:orthogonal}{ortogonal} de $B$ sobre $P$
        (caminar desde $B$ por la normal hasta el avión), y
        compruebe que su punto satisface el \omterm{def:g10:algebra:equation}{ecuación} de $P$.
  \item Finale --- el kit del topógrafo espacial: líneas paramétricas
        para perforar, \omterm{def:g12:space:normal}{vectores normales} para ángulos y distancias,
        plano \omterm{def:g10:algebra:equation}{ecuaciones} para secciones; y el cubo como
        patio de juegos donde produjeron un hexágono, un
        tetraedro, el ángulo del diamante y un par de sesgos
        líneas. Una frase cada uno.
\end{enumerate}
\end{problem}
