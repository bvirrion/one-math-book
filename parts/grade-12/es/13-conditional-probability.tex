\section{Espacios de probabilidad (recordatorio)}

Un experimento con un número finito de resultados se modela mediante un \emph{\omterm{def:g11:prob:model}{muestra
espacio}} $\Omega$ (el conjunto de resultados) y un \omterm{def:g10:proba:distribution}{probabilidad} $\P$ asignando a
cada \omterm{def:g11:prob:model}{evento} $A \subseteq \Omega$ un número $\P(A) \in \intcc{0}{1}$, aditivo
sobre \omterm{def:g10:numbers:interunion}{sindicatos} disjunto y con $\P(\Omega) = 1$. Recuerde las reglas básicas:
\[
\P(\bar A) = 1 - \P(A),
\qquad
\P(A \cup B) = \P(A) + \P(B) - \P(A \cap B).
\]
Cuando todos los resultados son igualmente probables,
$\P(A) = \frac{\abs A}{\abs\Omega}$ --- y las probabilidades de cálculo reducen
a las técnicas de conteo de \cref{ch:g12:comb}.

\section{Probabilidad condicional}

\begin{definition}[Probabilidad condicional]\label{def:g12:condprob:cond}
Sea $B$ un \omterm{def:g11:prob:model}{evento} con $\P(B) > 0$. El \emph{probabilidad de $A$ dada
$B$}\index{probabilidad condicional} es
\[
\pcond{B}{A} = \frac{\P(A \cap B)}{\P(B)} .
\]
\end{definition}

El mapa $A \mapsto \pcond BA$ es en sí mismo un \omterm{def:g10:proba:distribution}{probabilidad} (se aplican todas las reglas); eso
representa el nuevo estado de conocimiento de alguien que ha aprendido que $B$
ocurrió.

\begin{proposition}[regla de multiplicacion]\label{prop:g12:condprob:product}
Para \omterm{def:g11:prob:model}{eventos} con probabilidades distintas de cero:
\[
\P(A \cap B) = \P(B)\,\pcond{B}{A} = \P(A)\,\pcond{A}{B},
\]
y más en general
$\P(A_1 \cap A_2 \cap A_3) = \P(A_1)\,\pcond{A_1}{A_2}\,
\pcond{A_1 \cap A_2}{A_3}$, etc.
\end{proposition}

\begin{proof}
Reorganizar la definición; la fórmula de la cadena sigue iterando.
\end{proof}

\begin{theorem}[Ley de probabilidad total]\label{thm:g12:condprob:total}
\index{ley de probabilidad total}
Sea $B_1, \dots, B_n$ una partición de $\Omega$ en \omterm{def:g11:prob:model}{eventos} de distinto de cero
\omterm{def:g10:proba:distribution}{probabilidad}. Por cada \omterm{def:g11:prob:model}{evento} $A$:
\[
\P(A) = \sum_{i=1}^{n} \P(B_i)\,\pcond{B_i}{A}.
\]
\end{theorem}

\begin{proof}
Los conjuntos $A \cap B_i$ son separados por pares de \omterm{def:g10:numbers:interunion}{unión} $A$, por lo que
$\P(A) = \sum_i \P(A \cap B_i) = \sum_i \P(B_i)\pcond{B_i}{A}$ por el
regla de multiplicación.
\end{proof}

\begin{method}[Árboles de probabilidad]\label{met:g12:condprob:tree}
A \omterm{met:g10:proba:tree}{diagrama de arbol} organiza probabilidades condicionales: cada rama lleva el
\omterm{def:g10:proba:distribution}{probabilidad} del siguiente \omterm{def:g11:prob:model}{evento} \emph{dado} la ruta hasta el momento.
\begin{itemize}
  \item El \omterm{def:g10:proba:distribution}{probabilidad} de una hoja (una ruta completa) es el \emph{producto} de
        las probabilidades a lo largo de sus ramas (regla de multiplicación).
  \item El \omterm{def:g10:proba:distribution}{probabilidad} de un \omterm{def:g11:prob:model}{evento} es el \emph{suma} de las probabilidades
        de las hojas al darse cuenta (total \omterm{def:g10:proba:distribution}{probabilidad}).
  \item Las probabilidades en las ramas que salen de un nodo suman $1$.
\end{itemize}
\end{method}

\begin{theorem}[fórmula de bayes]\label{thm:g12:condprob:bayes}
\index{fórmula de bayes}
Sea $B_1, \dots, B_n$ una partición de $\Omega$ como arriba y $A$ una \omterm{def:g11:prob:model}{evento}
con $\P(A) > 0$. entonces
\[
\pcond{A}{B_j} =
\frac{\P(B_j)\,\pcond{B_j}{A}}{\sum_{i=1}^{n} \P(B_i)\,\pcond{B_i}{A}} .
\]
\end{theorem}

\begin{proof}
$\pcond{A}{B_j} = \frac{\P(A \cap B_j)}{\P(A)}$; ampliar el numerador en
la regla de multiplicación y el denominador por el total \omterm{def:g10:proba:distribution}{probabilidad}.
\end{proof}

\begin{example}[Prueba de detección]\label{ex:g12:condprob:test}
Una enfermedad afecta a $1\%$ de una población. Un test lo detecta con \omterm{def:g10:proba:distribution}{probabilidad}
$0.99$ (sensibilidad) y da un falso positivo con \omterm{def:g10:proba:distribution}{probabilidad} $0.05$.
Dada una prueba positiva, el \omterm{def:g10:proba:distribution}{probabilidad} de tener realmente la enfermedad es
\[
\frac{0.01 \times 0.99}{0.01\times0.99 + 0.99\times0.05}
= \frac{0.0099}{0.0099 + 0.0495} = \frac{1}{6} \approx 0.17 .
\]
A pesar de la precisión de la prueba, cinco de seis positivos son falsos --- porque
la enfermedad es rara. Confundir $\pcond{A}{B}$ con $\pcond{B}{A}$ es el
\emph{falacia del fiscal}.
\end{example}

\section{Independencia}

\begin{definition}[Eventos independientes]\label{def:g12:condprob:indep}
Dos \omterm{def:g11:prob:model}{eventos} $A$ y $B$ son \emph{independiente}\index{independencia} si
\[
\P(A \cap B) = \P(A)\,\P(B) .
\]
Cuando $\P(B) > 0$, esto equivale a $\pcond{B}{A} = \P(A)$: conociendo $B$
no cambia el \omterm{def:g10:proba:distribution}{probabilidad} de $A$.
\end{definition}

\begin{proposition}\label{prop:g12:condprob:indepcompl}
Si $A$ y $B$ son \omterm{def:g12:condprob:indep}{independiente}, también lo son $A$ y $\bar B$ (y $\bar A$ y
$\bar B$).
\end{proposition}

\begin{proof}
$\P(A \cap \bar B) = \P(A) - \P(A \cap B) = \P(A) - \P(A)\P(B)
= \P(A)\bigl(1 - \P(B)\bigr) = \P(A)\,\P(\bar B)$.
\end{proof}

\begin{remark}
No confunda \emph{\omterm{def:g12:condprob:indep}{independiente}} ($\P(A\cap B) = \P(A)\P(B)$) con
\emph{\omterm{def:g10:proba:operations}{incompatible}} ($A \cap B = \varnothing$). Dos \omterm{def:g10:proba:operations}{incompatible} \omterm{def:g11:prob:model}{eventos} de
\omterm{def:g10:proba:distribution}{probabilidad} distintos de cero nunca son \omterm{def:g12:condprob:indep}{independiente}: saber que ocurrió uno garantiza
el otro no.
\end{remark}

\begin{definition}[repeticiones independientes]\label{def:g12:condprob:repet}
Cuando un experimento se repite $n$ veces para que el resultado de cada prueba
no influye en los demás, el \omterm{def:g10:proba:distribution}{probabilidad} de un \omterm{def:g12:seq:sequence}{secuencia} especificado de
Los resultados son el producto de las probabilidades individuales. este es el modelo
subyacente al binomio \omterm{def:g11:prob:rv}{distribución} (\cref{ch:g12:randvar}).
\end{definition}
