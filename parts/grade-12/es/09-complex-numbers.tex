\chapter{Números complejos}\label{ch:g12:complex}

El \omterm{def:g10:algebra:equation}{ecuación} $x^2 = -1$ no tiene una solución real. Ampliando $\R$ con un solo
el nuevo número $\iu$ cuyo cuadrado es $-1$ produce el campo $\C$ de \omterm{def:g12:complex:def}{complejo
numeros}, en el cual el polinomio \emph{cada} \omterm{def:g10:algebra:equation}{ecuación} tiene soluciones --- y
cuya aritmética codifica la geometría del plano: traslaciones, rotaciones
y las escalas se convierten en sumas y multiplicaciones.

\section{forma algebraica}

\begin{definition}[Números complejos]\label{def:g12:complex:def}
El conjunto de \emph{números complejos}\index{numero complejo} es
\[
\C = \{a + \iu b : a, b \in \R\},
\]
donde $\iu$ es un símbolo sujeto a la regla única $\iu^2 = -1$; adición y
las multiplicaciones se realizan como para polinomios reales en $\iu$, reduciendo
$\iu^2$ a $-1$. Los reales $a$ y $b$ son los \emph{parte real}\index{parte real}.
$\Rea(z)$ y \emph{parte imaginaria}\index{parte imaginaria} $\Ima(z)$ de $z = a + \iu b$; esto
La expresión es \emph{forma algebraica}\index{forma algebraica} de $z$ y es única:
$a + \iu b = a' + \iu b'$ si y sólo si $a = a'$ y $b = b'$.
\end{definition}

\begin{example}
$(2 + 3\iu)(1 - \iu) = 2 - 2\iu + 3\iu - 3\iu^2 = 5 + \iu$.
\end{example}

\begin{definition}[conjugado]\label{def:g12:complex:conjugate}
El \emph{conjugado}\index{conjugado} de $z = a + \iu b$ es
$\conj{z} = a - \iu b$.
\end{definition}

\begin{proposition}[Propiedades de conjugación]\label{prop:g12:complex:conjugate}
Para todos los $z, w \in \C$:
\[
\conj{z + w} = \conj z + \conj w, \quad
\conj{zw} = \conj z\,\conj w, \quad
\conj{\conj z} = z, \quad
z + \conj z = 2\Rea(z), \quad
z - \conj z = 2\iu\,\Ima(z),
\]
y $z\conj z = a^2 + b^2 \geq 0$ para $z = a + \iu b$. Además $z \in \R$
si y sólo si $z = \conj z$.
\end{proposition}

\begin{proof}
Todos son cálculos directos del \omterm{def:g12:complex:def}{formas algebraicas}; por ejemplo
$z\conj z = (a + \iu b)(a - \iu b) = a^2 - (\iu b)^2 = a^2 + b^2$.
\end{proof}

\begin{method}[División de números complejos]\label{met:g12:complex:division}
Para escribir un cociente en \omterm{def:g12:complex:def}{forma algebraica}, multiplica el numerador y el denominador por
el \omterm{def:g12:complex:conjugate}{conjugado} del denominador:
\[
\frac{1}{2 + \iu} = \frac{2 - \iu}{(2+\iu)(2-\iu)} = \frac{2 - \iu}{5}
= \frac25 - \frac15\iu .
\]
Por lo tanto, todo \omterm{def:g12:complex:def}{numero complejo} distinto de cero tiene una inversa: $\C$ es un
\emph{campo}.
\end{method}

\section{El plano complejo, módulo y argumento.}

\begin{definition}[Afijo, módulo]\label{def:g12:complex:modulus}
En el plano con \omterm{def:g10:coordgeom:system}{ortonormal} \omterm{def:g10:coordgeom:system}{coordenadas}, el punto $M(a, b)$ es el
\emph{\omterm{def:g10:functions:function}{imagen}} de $z = a + \iu b$, y $z$ es el \emph{afijo}\index{afijo} de $M$. el
\emph{módulo}\index{módulo} de $z$ es la distancia desde $M$ al origen:
\[
\abs{z} = \sqrt{a^2 + b^2} = \sqrt{z\conj z}.
\]
\end{definition}

\begin{proposition}[Propiedades del módulo]\label{prop:g12:complex:modulus}
Para $z, w \in \C$:
\[
\abs{zw} = \abs z\,\abs w, \qquad
\abs{\tfrac zw} = \tfrac{\abs z}{\abs w}\ (w \neq 0), \qquad
\abs{\conj z} = \abs z, \qquad
\abs{z + w} \leq \abs{z} + \abs{w} \ \text{(triangle inequality)}.
\]
Además $\abs{z_B - z_A}$ es la distancia entre los puntos de \omterm{def:g12:complex:modulus}{afijos}.
$z_A$ y $z_B$.
\end{proposition}

\begin{proof}
$\abs{zw}^2 = zw\,\conj{zw} = z\conj z\, w \conj w = \abs z^2 \abs w^2$
da el primero; el segundo y el tercero son similares. para el triangulo
desigualdad:
\[
\abs{z+w}^2 = (z+w)(\conj z + \conj w)
= \abs z^2 + \abs w^2 + 2\Rea(z \conj w)
\leq \abs z^2 + \abs w^2 + 2\abs{z\conj w}
= \bigl(\abs z + \abs w\bigr)^2,
\]
utilizando $\Rea(u) \leq \abs u$. El enunciado de la distancia es el pitagórico.
Fórmula aplicada al \omterm{def:g10:coordgeom:system}{coordenadas} de $z_B - z_A$.
\end{proof}

\begin{definition}[Argumento, formas trigonométricas y exponenciales]\label{def:g12:complex:argument}
Deje $z \neq 0$. Un \emph{argumento}\index{argumento} de $z$, escrito
$\Arg(z)$, es cualquier medida $\theta$ del ángulo desde el eje real positivo
al rayo $OM$; está definido hasta $2k\pi$. Escribiendo $r = \abs z$,
\[
z = r(\cos\theta + \iu\sin\theta)
\qquad\text{(trigonometric form)}.
\]
Introduciendo la notación\index{forma exponencial}
\[
\eu^{\iu\theta} = \cos\theta + \iu\sin\theta,
\]
esto se convierte en \emph{forma exponencial} $z = r\,\eu^{\iu\theta}$.
\end{definition}

\begin{theorem}\label{thm:g12:complex:expform}
Para todos los $\theta, \theta' \in \R$:
\[
\eu^{\iu\theta}\,\eu^{\iu\theta'} = \eu^{\iu(\theta + \theta')},
\qquad
\conj{\eu^{\iu\theta}} = \eu^{-\iu\theta},
\qquad
\abs{\eu^{\iu\theta}} = 1 .
\]
En consecuencia, para $z = r\eu^{\iu\theta}$ y $z' = r'\eu^{\iu\theta'}$
distinto de cero:
\[
zz' = rr'\,\eu^{\iu(\theta+\theta')},
\qquad
\frac{z}{z'} = \frac{r}{r'}\,\eu^{\iu(\theta - \theta')} :
\]
\emph{moduli multiply, \omterm{def:g12:complex:argument}{argumentos} add.}
\end{theorem}

\begin{proof}
La primera identidad \emph{is} el par de fórmulas de suma.
(\cref{prop:g12:trigo:addition}):
\[
(\cos\theta + \iu\sin\theta)(\cos\theta' + \iu\sin\theta')
= (\cos\theta\cos\theta' - \sin\theta\sin\theta')
+ \iu(\sin\theta\cos\theta' + \cos\theta\sin\theta').
\]
El resto se realiza mediante cálculo directo.
\end{proof}

\begin{corollary}[Fórmula de Moivre]\label{cor:g12:complex:demoivre}
\index{La fórmula de De Moivre}
Para $\theta \in \R$ y $n \in \Z$:
$\bigl(\cos\theta + \iu\sin\theta\bigr)^n = \cos n\theta + \iu \sin n\theta$.
\end{corollary}

\begin{proof}
Inducción en $n \geq 0$ usando \cref{thm:g12:complex:expform}; extender a
$n < 0$ tomando inversas.
\end{proof}

\begin{method}[Cambio entre formularios]\label{met:g12:complex:forms}
De algebraico a exponencial: calcule $r = \abs z$, luego encuentre $\theta$ con
$\cos\theta = \frac ar$, $\sin\theta = \frac br$ (ubique el cuadrante correcto
antes de invocar un \omterm{def:g12:trigo:cossin}{coseno} inverso). De exponencial a algebraico: expandir
$r\cos\theta + \iu\, r\sin\theta$. Utilice el \omterm{def:g12:complex:def}{forma algebraica} para sumas, el
\omterm{def:g12:complex:argument}{forma exponencial} para productos, potencias y cocientes.
\end{method}

\begin{example}
$z = 1 + \iu$: $\abs z = \sqrt2$ y $\cos\theta = \sin\theta =
\frac{1}{\sqrt2}$, entonces $\theta = \frac\pi4$ y $z = \sqrt2\,\eu^{\iu\pi/4}$.
Por lo tanto $z^{20} = 2^{10}\,\eu^{5\iu\pi} = -1024$.
\end{example}

\section{Ecuaciones de segundo grado y raíces de la unidad.}

\begin{theorem}[Ecuaciones cuadráticas con coeficientes reales]\label{thm:g12:complex:quadratic}
Sean $a, b, c \in \R$, $a \neq 0$ y $\Delta = b^2 - 4ac$. El \omterm{def:g10:algebra:equation}{ecuación}
$az^2 + bz + c = 0$ tiene soluciones en $\C$:
\[
\begin{aligned}
&\Delta > 0:\ z = \frac{-b \pm \sqrt{\Delta}}{2a} \ (\text{two real}); \qquad
\Delta = 0:\ z = \frac{-b}{2a} \ (\text{double}); \\
&\Delta < 0:\ z = \frac{-b \pm \iu\sqrt{-\Delta}}{2a} \ (\text{two conjugate}).
\end{aligned}
\]
\end{theorem}

\begin{proof}
Completa el cuadrado:
$az^2 + bz + c = a\left(\left(z + \frac{b}{2a}\right)^2 -
\frac{\Delta}{4a^2}\right)$. Si $\Delta < 0$, entonces
$\frac{\Delta}{4a^2} = \left(\iu\frac{\sqrt{-\Delta}}{2a}\right)^2$, y el
diferencia de factores de cuadrados como de costumbre.
\end{proof}

\begin{definition}[Raíces de la unidad]\label{def:g12:complex:rootsofunity}
Para $n \geq 1$, \emph{$n$-ésimas raíces de la unidad}\index{raíces de la unidad} son
las soluciones de $z^n = 1$.
\end{definition}

\begin{theorem}\label{thm:g12:complex:rootsofunity}
El \omterm{def:g10:algebra:equation}{ecuación} $z^n = 1$ tiene exactamente soluciones $n$:
\[
\omega_k = \eu^{2\iu k\pi/n}, \qquad k = 0, 1, \dots, n-1 .
\]
Sus \omterm{def:g10:functions:function}{imágenes} son los vértices de un gon $n$ regular inscrito en la unidad.
círculo, y para $n \geq 2$ su suma es $0$.
\end{theorem}

\begin{proof}
Cada $\omega_k$ satisface $\omega_k^n = \eu^{2\iu k\pi} = 1$ y son
distintos por pares ya que su \omterm{def:g12:complex:argument}{argumentos} se encuentra en $\intco{0}{2\pi}$.
Por el contrario, si $z^n = 1$ entonces $\abs z^n = 1$, entonces $\abs z = 1$ (positivo
real) y $z = \eu^{\iu\theta}$ con $n\theta \equiv 0 \pmod{2\pi}$,
\emph{i.e.} $\theta = \frac{2k\pi}{n}$; reduciendo $k$ módulo $n$ aterriza entre
el $\omega_k$. Los vértices están igualmente espaciados por el ángulo $\frac{2\pi}{n}$:
un $n$-gon normal. Finalmente, con $\omega = \omega_1$,
$\omega \neq 1$ y la suma geométrica da
\[
\sum_{k=0}^{n-1}\omega_k = \sum_{k=0}^{n-1}\omega^k
= \frac{\omega^n - 1}{\omega - 1} = 0 . \qedhere
\]
\end{proof}

\end{example}

\section{Números complejos y geometría plana.}

\begin{proposition}[diccionario geométrico]\label{prop:g12:complex:geometry}
Sean $A, B, C, D$ puntos de \omterm{def:g12:complex:modulus}{afijos} $a, b, c, d$ ($a \neq b$,
$c \neq d$).
\begin{enumerate}
  \item La traducción realizada por \omterm{def:g11:vect:vector}{vector} de \omterm{def:g12:complex:modulus}{afijo} $t$ es $z \mapsto z + t$.
  \item La rotación del centro $\omega$ (\omterm{def:g12:complex:modulus}{afijo}) y el ángulo $\theta$ es
        $z \mapsto \omega + \eu^{\iu\theta}(z - \omega)$.
  \item La escala (homotetia) del centro $\omega$ y la relación
        $k \in \R^*$ es $z \mapsto \omega + k(z - \omega)$.
  \item $\dfrac{d - c}{b - a}$ tiene \omterm{def:g12:complex:modulus}{módulo} $\dfrac{CD}{AB}$ y \omterm{def:g12:complex:argument}{argumento} el
        ángulo entre \omterm{def:g11:vect:vector}{vectores} $\vect{AB}$ y $\vect{CD}$. en
        particular, $AB \perp CD$ si y sólo si $\frac{d-c}{b-a}$ es puramente
        imaginario, y $A, B, C$ son \omterm{def:g11:vect:collinear}{colineal} ($C \ne A$, $C\ne B$) si y
        sólo si $\frac{c-a}{b-a} \in \R$.
\end{enumerate}
\end{proposition}

\begin{proof}
(1) y (3) reformulan las fórmulas de coordenadas. Para (2), el mapa
$w \mapsto \eu^{\iu\theta} w$ multiplica los módulos por $1$ y suma $\theta$ a
\omterm{def:g12:complex:argument}{argumentos}: es la rotación del ángulo $\theta$ con respecto al origen; conjugando
por la traducción enviando $\omega$ a $0$ da el centro general. Para
(4), módulos y \omterm{def:g12:complex:argument}{argumentos} de cocientes restan
(\cref{thm:g12:complex:expform}).
\end{proof}

\section{Problema: el pentágono en el espejo de la unidad}

\begin{problem}[{Problema del fin de semana --- las quintas raíces de la unidad
calcular $\cos 72^\circ$ exactamente, la proporción áurea firma el
resultado, y la multiplicación resulta ser rotación}]
\label{pb:g12:complex:1}
Euclides pudo construir el pentágono regular pero la razón
\emph{por qué} se somete a regla y compás --- mientras el humilde
$20^\circ$ el ángulo no --- permaneció oculto durante dos mil
años. Se esconde en este capítulo: el quinto quinto \omterm{def:g12:complex:rootsofunity}{raíces de la unidad}
(\cref{thm:g12:complex:rootsofunity}) suma cero, y ese
línea de álgebra exprime $\cos 72^\circ$ de una cuadrática
\omterm{def:g10:algebra:equation}{ecuación} --- solo raíces cuadradas, por lo tanto construibles y con signo,
por supuesto, por la proporción áurea. Alrededor de esta joya: fluidez, de
El atajo de Moivre hacia la trigonometría y la multiplicación desenmascarada como
geometría.

\medskip
\noindent\textbf{Parte I --- Fluency.}
\begin{enumerate}
  \item Calcular $(2 + \iu)(3 - \iu)$, \
        $\dfrac{1 + \iu}{1 - \iu}$
        (\cref{met:g12:complex:division}), y
        $\abs{3 + 4\iu}$.
  \item Poner \omterm{def:g12:complex:argument}{forma exponencial}: $1 + \iu$; $-2$; \
        $1 - \iu\sqrt3$.
  \item Calcule $(1 + \iu)^8$ utilizando \omterm{def:g12:complex:argument}{forma exponencial}.
  \item Resuelve $z^2 = \iu$, luego $z^2 - 2z + 5 = 0$
        (\cref{thm:g12:complex:quadratic}).
  \item Diccionario geométrico
        (\cref{prop:g12:complex:geometry}): describe el mapa
        $z \mapsto \iu z$, y el conjunto
        $\abs{z - (1 + \iu)} = 2$.
\end{enumerate}

\medskip
\noindent\textbf{Parte II --- Five \omterm{def:g11:quad:discriminant}{raíces}, one pentagon.}
Dejemos $\omega = \eu^{2\iu\pi/5}$ y consideremos
$1, \omega, \omega^2, \omega^3, \omega^4$.
\begin{enumerate}[resume]
  \item Compruebe que estas son exactamente las soluciones de $z^5 = 1$,
        y describir la figura que dibujan en el avión.
  \item Demuestre que su suma es cero. (Progresión geométrica
        --- o multiplicar la suma por $1 - \omega$.)
  \item Tome \omterm{def:g12:complex:def}{piezas reales} y deduzca
        \[
        1 + 2\cos\frac{2\pi}{5} + 2\cos\frac{4\pi}{5} = 0 .
        \]
  \item Configure $c = \cos\frac{2\pi}{5}$ y use el ángulo doble.
        fórmula para $\cos\frac{4\pi}{5}$ para convertir la pregunta 8
        en una \omterm{def:g10:algebra:equation}{ecuación} cuadrática para $c$; resolverlo y
        concluir
        \[
        \cos 72^\circ = \frac{\sqrt5 - 1}{4} .
