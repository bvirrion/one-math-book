\chapter{Ecuaciones diferenciales}\label{ch:g12:diffeq}

Un \omterm{def:g12:diffeq:ode}{ecuación diferencial} relaciona un \omterm{def:g11:func:function}{función} con su \omterm{def:g12:deriv:derivative}{derivados}. física,
La química, la biología y la economía expresan sus leyes de esta forma: la tasa de
enfriamiento de un cuerpo, la desintegración de un núcleo radiactivo, el crecimiento de un
La población son todas declaraciones sobre $y'$. Este capítulo resuelve completamente el
\omterm{def:g10:algebra:equation}{ecuaciones} lineal de primer orden con coeficientes constantes.

\section{La ecuación \texorpdfstring{$y' = ay$}{y' = ay}}

\begin{definition}[Ecuación diferencial]\label{def:g12:diffeq:ode}
Un \emph{ecuación diferencial}\index{ecuación diferencial} es un \omterm{def:g10:algebra:equation}{ecuación}
cuya incógnita es un \omterm{def:g11:func:function}{función} $y$, involucrando a $y$ y su \omterm{def:g12:deriv:derivative}{derivados}. un
\emph{solución} en un \omterm{def:g10:numbers:interval}{intervalo} $I$ es un \omterm{def:g12:deriv:derivative}{diferenciable} \omterm{def:g11:func:function}{función} satisfactorio
el \omterm{def:g10:algebra:equation}{ecuación} en cada punto de $I$.
\end{definition}

\begin{theorem}[Soluciones de $y' = ay$]\label{thm:g12:diffeq:homogeneous}
Deje $a \in \R$. Las soluciones en $\R$ de \omterm{def:g10:algebra:equation}{ecuación} $y' = ay$ son exactamente
el \omterm{def:g11:func:function}{funciones}
\[
y(x) = C\,\eu^{ax}, \qquad C \in \R .
\]
Para cada par $(x_0, y_0)$ existe una solución única con
$y(x_0) = y_0$.
\end{theorem}

\begin{proof}
Cada $y = C\eu^{ax}$ satisface $y' = Ca\,\eu^{ax} = ay$. Por el contrario, dejemos que $y$
Sea cualquier solución y establezca $z(x) = y(x)\,\eu^{-ax}$. entonces
\[
z'(x) = y'(x)\,\eu^{-ax} - a\,y(x)\,\eu^{-ax}
= \bigl(y'(x) - a y(x)\bigr)\eu^{-ax} = 0,
\]
entonces $z$ es constante, digamos $C$ y $y(x) = C\eu^{ax}$. La condición inicial
$y(x_0) = y_0$ fuerza a $C = y_0 \eu^{-a x_0}$, de forma única.
\end{proof}

\begin{remark}
El exponencial es entonces \emph{caracterizado} por el más simple de todos.
\omterm{def:g12:diffeq:ode}{ecuaciones diferenciales}: crecimiento proporcional al tamaño. Por eso aparece
en todas partes de la naturaleza.
\end{remark}

\begin{example}[Desintegración radiactiva]\label{ex:g12:diffeq:decay}
Una cantidad radiactiva satisface $N' = -\lambda N$ con $\lambda > 0$, por lo que
$N(t) = N_0\,\eu^{-\lambda t}$; ver \cref{ex:g12:exp:halflife} para el
vida media.
\end{example}

\section{La ecuación \texorpdfstring{$y' = ay + b$}{y' = ay + b}}

\begin{theorem}[Soluciones de $y' = ay + b$]\label{thm:g12:diffeq:affine}
Sean $a \neq 0$ y $b \in \R$. Las soluciones en $\R$ de $y' = ay + b$ son
exactamente el \omterm{def:g11:func:function}{funciones}
\[
y(x) = C\,\eu^{ax} - \frac{b}{a}, \qquad C \in \R .
\]
La constante \omterm{def:g11:func:function}{función} $y_p = -\frac ba$ es la \emph{solución de equilibrio}.
Para cada $(x_0, y_0)$ existe una solución única con $y(x_0) = y_0$; si
$a < 0$, toda solución tiende al equilibrio $-\frac ba$ como
$x \to +\infty$.
\end{theorem}

\begin{proof}
La constante $y_p = -\frac{b}{a}$ satisface $y_p' = 0 = a y_p + b$. Ahora $y$
es una solución si y sólo si
\[
(y - y_p)' = y' = ay + b = a(y - y_p) + \underbrace{a y_p + b}_{=\,0}
= a (y - y_p),
\]
es decir, si y sólo si $z = y - y_p$ resuelve $z' = az$. Por
\cref{thm:g12:diffeq:homogeneous}, $z = C\eu^{ax}$, de donde la fórmula, la
existencia y la unicidad. Si $a < 0$, $\eu^{ax} \to 0$ como
$x \to +\infty$, entonces $y(x) \to -\frac ba$.
\end{proof}

\begin{method}[Resolviendo $y' = ay + b$ con una condición inicial]\label{met:g12:diffeq:solve}
\begin{enumerate}
  \item Encuentre la solución de equilibrio $y_p = -\frac{b}{a}$ (resuelva $y' = 0$).
  \item Escriba la solución general $y = C\eu^{ax} + y_p$.
  \item Determine $C$ a partir de la condición inicial.
  \item Verifique el comportamiento a largo plazo contra la intuición física (¿el
        solución converge al equilibrio?).
\end{enumerate}
La misma estrategia --- \emph{particular solution + general solution of the
homogeneous \omterm{def:g10:algebra:equation}{ecuación}} --- se extiende a $y' = ay + f(x)$: ver
\cref{prop:g12:diffeq:general}.
\end{method}

\begin{example}[Ley de enfriamiento de Newton]\label{ex:g12:diffeq:cooling}
Una taza de café en $80\,^\circ$C se encuentra en una habitación en $20\,^\circ$C. La ley de Newton establece
que la temperatura $T$ satisface $T' = -k(T - 20)$ para algunos $k > 0$,
\emph{i.e.} $T' = -kT + 20k$. El equilibrio es $20$, y
$T(t) = 20 + 60\,\eu^{-kt}$: el café se enfría exponencialmente rápido a la habitación
temperatura.
\end{example}

\section{La ecuación \texorpdfstring{$y' = ay + f(x)$}{y' = ay + f(x)}}

\begin{proposition}[Estructura del conjunto de soluciones]\label{prop:g12:diffeq:general}
Sea $f$ \omterm{def:g12:limcont:continuity}{continuo} en un \omterm{def:g10:numbers:interval}{intervalo} $I$ y sea $y_p$ un
solución de
\[
y' = ay + f(x)
\]
en $I$. Entonces las soluciones en $I$ son exactamente las \omterm{def:g11:func:function}{funciones}.
$y = C\eu^{ax} + y_p$, $C \in \R$.
\end{proposition}

\begin{proof}
Como en \cref{thm:g12:diffeq:affine}: $y$ es una solución si y sólo si
$z = y - y_p$ satisface
$z' = (ay + f) - (ay_p + f) = az$, si y sólo si $z = C\eu^{ax}$.
\end{proof}

\begin{method}[Adivinando una solución particular]\label{met:g12:diffeq:particular}
Busque una solución particular \emph{de la misma forma} como $f$:
\begin{itemize}
  \item $f$ polinomio de grado $n$: pruebe con un polinomio de grado $n$;
  \item $f(x) = \alpha\,\eu^{kx}$ con $k \neq a$: prueba $y_p = \beta\,\eu^{kx}$;
  \item $f(x) = \alpha\,\eu^{ax}$ (caso resonante): intente
        $y_p = \beta x\,\eu^{ax}$.
\end{itemize}
Sustituya en \omterm{def:g10:algebra:equation}{ecuación} e identifique los coeficientes.
\end{method}

\begin{example}\label{ex:g12:diffeq:particular}
Resuelve $y' = 2y + 4x$. Pruebe $y_p = \alpha x + \beta$:
$\alpha = 2(\alpha x + \beta) + 4x$ para todas las fuerzas $x$
$2\alpha + 4 = 0$ y $\alpha = 2\beta$, entonces $\alpha = -2$, $\beta = -1$.
Solución general: $y = C\eu^{2x} - 2x - 1$.
\end{example}

\begin{exercise}[$\star\star$]\label{exo:g12:diffeq:5}

\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:diffeq:6}
Un tanque contiene $100$ L de agua pura. Salmuera que contiene $0.2$ kg de sal por
litro entra a $5$ L/min, la mezcla (mantenida uniforme) sale al mismo
tasa. Sea $m(t)$ la masa de sal en el tanque en el momento $t$ (en minutos).
\begin{enumerate}
  \item Justifique que $m' = 1 - \dfrac{m}{20}$.
  \item Resuelva y determine el límite de $m(t)$ como $t \to +\infty$.

\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:diffeq:7}
Un paracaidista de masa $80$ kg cae sujeto a la gravedad
($g = 9.8\ \text{m/s}^2$) y resistencia del aire proporcional a la velocidad, de modo que
su velocidad satisface $v' = g - \frac{k}{m}v$ con $k = 16$ kg/s.
\begin{enumerate}
  \item Resuelve el \omterm{def:g10:algebra:equation}{ecuación} con $v(0) = 0$.
  \item Calcular el \emph{velocidad terminal} $\lim_{t\to+\infty} v(t)$, y
        el tiempo necesario para llegar a $95\%$ del mismo.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:diffeq:8}
\emph{(Logistic \omterm{def:g10:algebra:equation}{ecuación}.)} Una población $y(t) \in \intoo{0}{1}$ (como
fracción de la población máxima) satisface
\[
y' = y(1 - y).
\]
\begin{enumerate}
  \item Vamos $z = \dfrac{1}{y}$. Demuestre que $z$ satisface el lineal \omterm{def:g10:algebra:equation}{ecuación}
        $z' = -z + 1$.
  \item Resuelva para $z$, luego para $y$, con $y(0) = \frac{1}{10}$.
  \item Muestre que $y(t) \to 1$ es $t \to +\infty$ y dibuje la forma de
        la curva de solución.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:diffeq:9}

¿Corresponde la condición $\abs{q} < 1$ para el \omterm{def:g12:diffeq:ode}{diferencial
ecuación}?
\end{exercise}

\section{Problema: el reloj dentro de las cosas}

\begin{problem}[{Problema de fin de semana --- el carbono-14 data las cuevas,
La ley de enfriamiento de Newton multiplica un delito, y una pequeña ecuación desgasta
cuatro disfraces}]
\label{pb:g12:diffeq:1}
A \omterm{def:g12:diffeq:ode}{ecuación diferencial} es una ley de cambio; resolverlo convierte el
ley en un \emph{reloj}. La misma pequeñita \omterm{def:g10:algebra:equation}{ecuación}
$y' = ay + b$ (\cref{thm:g12:diffeq:affine}) garrapatas en el interior
carbón prehistórico, café refrescante, gotas de lluvia que caen y
goteos hospitalarios --- y leer esos relojes es el problema de este problema.
negocio: fecha las pinturas de Lascaux, fija la hora de la muerte,
y audita sus propios supuestos como un buen científico.

\medskip
\noindent\textbf{Parte I --- Fluency.}
\begin{enumerate}
  \item Resuelve $y' = 3y$, $y(0) = 2$; entonces
        $y' = -2y + 6$, $y(0) = 0$
        (\cref{met:g12:diffeq:solve}).
  \item Vuelva a derivar el truco de unicidad detrás
        \cref{thm:g12:diffeq:homogeneous}: si $y' = ay$, calcular
        el \omterm{def:g12:deriv:derivative}{derivado} de $y(t)\,\eu^{-at}$ y concluir que
        La solución \emph{cada} es $C\eu^{at}$.
  \item Para $y' = -2y + 6$: encuentre la solución de equilibrio y
        describir el destino de cualquier otra solución como
        $t \to +\infty$. (El \omterm{pb:g10:functions:1}{puntos fijos} de
        Las recurrencias de \cref{pb:g11:seq:1}, desaparecieron \omterm{def:g12:limcont:continuity}{continuo}.)
  \item Para la ley de decaimiento $y' = -\frac{y}{\tau}$: demuestre que el
        la vida media es $t_{1/2} = \tau \ln 2$.
  \item Dibuje (o describa) la familia de soluciones de
        $y' = -2y + 6$: ¿Qué significan las soluciones que comienzan arriba de $3$?
        hacer? ¿Debajo de $3$? ¿En $3$?
\end{enumerate}
