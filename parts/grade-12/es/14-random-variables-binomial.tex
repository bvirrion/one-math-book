\chapter{Variables aleatorias y distribución binomial}\label{ch:g12:randvar}

A \omterm{def:g12:randvar:rv}{variable aleatoria} asigna un número a cada resultado de un experimento: a
ganancia, una cuenta, una duración. Su \omterm{def:g12:randvar:exp}{expectativa} es el promedio de largo plazo del
valores que produce, su \omterm{def:g12:randvar:exp}{diferencia} mide su propagación. La estrella de este
El capítulo es el \omterm{def:g12:randvar:binomial}{distribución binomial}, que cuenta los éxitos en repetidas
\omterm{def:g12:condprob:indep}{independiente} ensayos. \omterm{def:g12:randvar:rv}{variables aleatorias} y \omterm{def:g12:randvar:binomial}{distribución binomial} fueron
se conoció por primera vez en \cref{ch:g11:prob,ch:g11:binom}; este capítulo revisa y
los profundiza, con las herramientas combinatorias de \cref{ch:g12:comb} ahora
disponible.

\section{Variables aleatorias discretas}

\begin{definition}[Variable aleatoria, distribución.]\label{def:g12:randvar:rv}
A \emph{variable aleatoria}\index{variable aleatoria} en un \omterm{def:g11:prob:model}{espacio muestral} finito
$\Omega$ es un \omterm{def:g11:func:function}{función} $X \colon \Omega \to \R$. su
\emph{distribución}\index{distribución} (o ley) son los datos de su posible
valores $x_1, \dots, x_k$ y de las probabilidades
\[
p_i = \P(X = x_i), \qquad \sum_{i=1}^{k} p_i = 1 .
\]
\end{definition}

\begin{definition}[Expectativa, variación, desviación estándar]\label{def:g12:randvar:exp}
El \emph{expectativa}\index{expectativa} de $X$ es
\[
\E(X) = \sum_{i=1}^{k} p_i\, x_i ,
\]
sus \emph{diferencia}\index{diferencia} y \emph{\omterm{def:g11:stat:variance}{desviación estándar}} son
\[
\V(X) = \E\bigl((X - \E(X))^2\bigr) = \sum_{i=1}^k p_i\bigl(x_i - \E(X)\bigr)^2,
\qquad
\sigma(X) = \sqrt{\V(X)} .
\]
\end{definition}

\begin{proposition}[fórmula de König--Huygens]\label{prop:g12:randvar:konig}
\index{Fórmula de König-Huygens}
$\V(X) = \E(X^2) - \E(X)^2$.
\end{proposition}

\begin{proof}
Escribe $m = \E(X)$ y expande:
\[
\V(X) = \sum_i p_i (x_i^2 - 2m x_i + m^2)
= \E(X^2) - 2m\sum_i p_i x_i + m^2\sum_i p_i
= \E(X^2) - 2m^2 + m^2 . \qedhere
\]
\end{proof}

\begin{proposition}[Transformación afín]\label{prop:g12:randvar:affine}
Para $a, b \in \R$:
\[
\E(aX + b) = a\,\E(X) + b, \qquad
\V(aX + b) = a^2\,\V(X) .
\]
\end{proposition}

\begin{proof}
La primera es una reordenación de la suma definitoria. Para el segundo, $aX + b$
se desvía de su \omterm{def:g11:stat:mean}{significar} en $a(X - \E(X))$, y al elevar al cuadrado se multiplica por
$a^2$.
\end{proof}

\begin{example}[juegos de feria]\label{ex:g12:randvar:game}
Un juego cuesta $2$ euros; Se lanza un dado y el jugador recibe el valor.
Se muestra si es al menos $5$, nada más. La ganancia $G$ toma valores
$-2$ (\omterm{def:g10:proba:distribution}{probabilidad} $\frac46$), $3$ ($\frac16$), $4$ ($\frac16$):
\[
\E(G) = \frac{-8 + 3 + 4}{6} = -\frac{1}{6} < 0 .
\]
En promedio, el jugador pierde $17$ centavos por juego: el juego es desfavorable
(como lo son la mayoría de los juegos reales).
\end{example}

\section{Ensayos de Bernoulli y distribución binomial.}

\begin{definition}[distribución Bernoulli]\label{def:g12:randvar:bernoulli}
Un \emph{juicio de Bernoulli}\index{juicio de Bernoulli} es un experimento con dos
resultados, \emph{éxito} (\omterm{def:g10:proba:distribution}{probabilidad} $p$) y \emph{falla}
($q = 1 - p$). El indicador $X$ de éxito ($X = 1$ de éxito, $0$ de
falla) sigue el \emph{Bernoulli \omterm{def:g12:randvar:rv}{distribución}} $\mathcal B(p)$:
\[
\E(X) = p, \qquad \V(X) = p(1-p) .
\]
(De hecho, $\E(X) = p$, $\E(X^2) = p$ y König--Huygens dan
$\V(X) = p - p^2$.)
\end{definition}

\begin{definition}[Distribución binomial]\label{def:g12:randvar:binomial}
Repita \omterm{def:g12:randvar:bernoulli}{juicio de Bernoulli} $n$ veces de forma independiente y deje que $X$ sea el total
número de éxitos. El \omterm{def:g12:randvar:rv}{distribución} de $X$ es el \emph{binomio
distribución}\index{distribución binomial} $\mathcal B(n, p)$.
\end{definition}

\begin{theorem}\label{thm:g12:randvar:binomial}
Si $X \sim \mathcal B(n, p)$, entonces para $0 \leq k \leq n$:
\[
\P(X = k) = \binom{n}{k} p^k (1-p)^{n-k},
\]
y
\[
\E(X) = np, \qquad \V(X) = np(1-p) .
\]
\end{theorem}

\begin{proof}
Se ha especificado un \omterm{def:g12:seq:sequence}{secuencia} de resultados con $k$ éxitos y $n-k$ fracasos.
\omterm{def:g10:proba:distribution}{probabilidad} $p^k(1-p)^{n-k}$ por \omterm{def:g12:condprob:indep}{independencia}; el número de tales \omterm{def:g12:seq:sequence}{secuencias}
es el número de formas de colocar los éxitos $k$ entre las pruebas $n$,
a saber, $\binom nk$ (\cref{ch:g12:comb}). Resumiendo el \omterm{def:g12:seq:sequence}{secuencias} da
la fórmula --- y el teorema del binomio lo confirma
$\sum_k \P(X = k) = (p + q)^n = 1$.

Para \omterm{def:g12:randvar:exp}{expectativa}, usando $k\binom nk = n\binom{n-1}{k-1}$
(\cref{exo:g12:comb:7}):
\[
\E(X) = \sum_{k=1}^{n} k\binom nk p^k q^{n-k}
= np\sum_{k=1}^{n} \binom{n-1}{k-1} p^{k-1} q^{(n-1)-(k-1)}
= np\,(p + q)^{n-1} = np .
\]
La fórmula \omterm{def:g12:randvar:exp}{diferencia} se demuestra de manera similar con la identidad
$k(k-1)\binom nk = n(n-1)\binom{n-2}{k-2}$, dando
$\E(X(X-1)) = n(n-1)p^2$, de donde
$\V(X) = n(n-1)p^2 + np - (np)^2 = np(1-p)$. (Una prueba estructural --- la
\omterm{def:g12:randvar:exp}{diferencia} de una suma de variables \omterm{def:g12:condprob:indep}{independiente} --- viene con
\cref{thm:g12:sums:variance}.)
\end{proof}

\begin{method}[Reconociendo una situación binomial]\label{met:g12:randvar:recognize}
Comprueba los tres ingredientes antes de escribir $X \sim \mathcal B(n,p)$: a
\emph{numero fijo} $n$ de ensayos; \emph{dos resultados} por prueba con el
\emph{mismo} éxito \omterm{def:g10:proba:distribution}{probabilidad} $p$; \emph{\omterm{def:g12:condprob:indep}{independencia}} de ensayos
(muestra de reemplazo \emph{con}, o de una población grande). Entonces usa
\[
\P(X \geq 1) = 1 - (1-p)^n
\]
para "al menos un éxito", y una calculadora o tablas acumulativas para
generales $\P(X \leq k)$.
\end{method}

\begin{example}\label{ex:g12:randvar:atleastone}
¿Cuántas veces se debe tirar un dado para tener al menos una probabilidad $99\%$ de
sacar un seis? Con $X \sim \mathcal B\left(n, \frac16\right)$:
$\P(X \geq 1) = 1 - \left(\frac56\right)^n \geq 0.99$ significa
$\left(\frac56\right)^n \leq 0.01$, \emph{i.e.}
$n \geq \frac{\ln 0.01}{\ln(5/6)} \approx 25.3$: de $n = 26$ continúa.
\end{example}

\section{Problema: el vuelo con overbooking}

\begin{problem}[{Problema de fin de semana --- las aerolíneas venden más asientos que
tienen, las fábricas aceptan lotes que apenas inspeccionaron y el
árbitros de distribución binomial ambos}]
\label{pb:g12:randvar:1}
Una aerolínea con asientos $100$ vende felizmente boletos $105$: aproximadamente
$10\,\%$ de los pasajeros nunca se presentan y los asientos vacíos ganan
nada. ¿Hasta dónde puede llegar la aerolínea antes de que los pasajeros sean golpeados?
comerse las ganancias? El \omterm{def:g12:randvar:binomial}{distribución binomial}
(\cref{thm:g12:randvar:binomial}) respuestas al decimal --- y
la misma maquinaria inspecciona lotes de fábrica, cotiza sorteos y
mantiene solventes a las aseguradoras. Decisiones bajo repetición: esta es la
El trabajo diario del binomio.
