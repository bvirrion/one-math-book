\chapter{Combinatoria y Conteo}\label{ch:g12:comb}

La combinatoria es el arte de contar sin enumerar. Sus dos elementales
principios --- sumar los tamaños de alternativas disjuntas, multiplicar los números
de opciones independientes --- basta con contar \omterm{def:g12:comb:tuples}{preparativos}, \omterm{def:g12:comb:permutation}{permutaciones}
y subconjuntos de un conjunto finito, y culminan en el teorema del binomio.

\section{Los dos principios de conteo}

Escribimos $\abs{E}$ para el número de elementos (el \emph{cardinalidad}) de un
conjunto finito $E$.

\begin{proposition}[Principio de suma]\label{prop:g12:comb:addition}
Si un conjunto finito $E$ se divide en subconjuntos $A_1, \dots, A_k$ (por pares
disjunto, con \omterm{def:g10:numbers:interunion}{unión} $E$), entonces
\[
\abs{E} = \abs{A_1} + \abs{A_2} + \dots + \abs{A_k}.
\]
\end{proposition}

\begin{proposition}[Principio de multiplicación]\label{prop:g12:comb:multiplication}
Si un objeto se construye mediante una sucesión de opciones $k$, con opciones $n_1$ para
la primera opción y, \emph{cualesquiera que sean las elecciones anteriores}, $n_i$ opciones para
el $i$-ésimo, entonces el número de objetos construidos es
$n_1 \times n_2 \times \dots \times n_k$.
\end{proposition}

\begin{proof}
Ambas afirmaciones se prueban por inducción sobre $k$; el caso $k = 2$ de la
El segundo equivale a contar una matriz rectangular por filas.
\end{proof}

\begin{example}
Un restaurante ofrece 4 entrantes, 6 segundos, 3 postres: $4 \times 6 \times 3 =
72$ comidas diferentes de tres platos.
\end{example}

\section{Tuplas, permutaciones, factoriales.}

\begin{definition}[$k$-tuplas]\label{def:g12:comb:tuples}
Un \emph{$k$-tupla} de un conjunto $E$ es una lista ordenada $(x_1, \dots, x_k)$ de
elementos de $E$, se permiten repeticiones. Una tupla $k$ de \emph{distinto}
elementos es un \emph{acuerdo}\index{acuerdo} de $k$ elementos de $E$.
\end{definition}

\begin{proposition}\label{prop:g12:comb:tuples}
Deje $\abs E = n$. El número de tuplas $k$ de $E$ es $n^k$. el numero de
\omterm{def:g12:comb:tuples}{preparativos} de $k$ elementos de $E$ ($0 \leq k \leq n$) es
\[
n(n-1)(n-2)\cdots(n-k+1) = \frac{n!}{(n-k)!},
\]
donde $n! = 1 \times 2 \times \dots \times n$ (y $0! = 1$) es el
\emph{factorial}\index{factorial} de $n$.
\end{proposition}

\begin{proof}
Principio de multiplicación: para una tupla $k$ hay opciones $n$ en cada una de
los pasos $k$; para \omterm{def:g12:comb:tuples}{acuerdo}, opciones $n$ para $x_1$, luego $n - 1$ para
$x_2$ (se utiliza un elemento), \dots, $n - k + 1$ para $x_k$.
\end{proof}

\begin{definition}[Permutación]\label{def:g12:comb:permutation}
Un \emph{permutación}\index{permutación} de $E$ es un \omterm{def:g12:comb:tuples}{acuerdo} de todos los $n$
elementos de $E$: un ordenamiento de $E$. Por
\cref{prop:g12:comb:tuples} (caso $k = n$), el número de permutaciones de un
El conjunto de elementos $n$ es $n!$.
\end{definition}

\begin{example}
Cinco corredores pueden terminar una carrera en $5! = 120$ órdenes diferentes. el numero de
posibles podios (tres primeros lugares) es $5 \times 4 \times 3 = 60$.
\end{example}

\section{Combinaciones y coeficientes binomiales.}

\begin{definition}[Combinaciones]\label{def:g12:comb:combination}
Un \emph{combinación}\index{combinación} de elementos $k$ de $E$ es un subconjunto
de $E$ con elementos $k$ (sin orden, sin repetición). Su numero esta escrito
$\dbinom{n}{k}$\index{coeficiente binomial}, lea ``$n$ elija $k$''.
\end{definition}

\begin{theorem}\label{thm:g12:comb:binomformula}
Para $0 \leq k \leq n$:
\[
\binom{n}{k} = \frac{n!}{k!\,(n-k)!} .
\]
\end{theorem}

\begin{proof}
Cuente los elementos \omterm{def:g12:comb:tuples}{preparativos} de $k$ de $E$ de dos maneras. Directamente:
$\frac{n!}{(n-k)!}$. Alternativamente, elija primero el subconjunto subyacente
($\binom nk$ formas), luego pídalo ($k!$ formas); el principio de multiplicación
da $\binom{n}{k}\,k!$. equiparando,
$\binom nk = \frac{n!}{k!(n-k)!}$.
\end{proof}

\begin{proposition}[Identidades básicas]\label{prop:g12:comb:identities}
Para $0 \leq k \leq n$:
\[
\binom{n}{0} = \binom{n}{n} = 1, \qquad
\binom{n}{1} = n, \qquad
\binom{n}{k} = \binom{n}{n-k},
\]
y \emph{regla de pascal}\index{el triangulo de pascal}: para $1 \leq k \leq n-1$,
\[
\binom{n}{k} = \binom{n-1}{k-1} + \binom{n-1}{k}.
\]
\end{proposition}

\begin{proof}
La simetría $\binom nk = \binom{n}{n-k}$ se cumple porque tomando \omterm{def:g10:proba:operations}{complementos}
coincide con los subconjuntos de elementos $k$ con los subconjuntos de elementos $(n-k)$, uno por uno. Para
\omterm{prop:g12:comb:identities}{regla de pascal}, arregla un elemento $a \in E$ y ordena los subconjuntos de elementos $k$ en
aquellos que contienen $a$ --- obtenidos uniendo $a$ a un elemento $(k-1)$
subconjunto de $E \setminus \{a\}$, de los cuales hay $\binom{n-1}{k-1}$ --- y
aquellos que evitan $a$, que son los subconjuntos de elementos $k$ de
$E \setminus \{a\}$, numeración $\binom{n-1}{k}$. Concluir con la adición
principio.
\end{proof}

\omterm{prop:g12:comb:identities}{regla de pascal} genera los coeficientes fila por fila --- \emph{\omterm{prop:g11:binom:pascal}{Pascal
triangulo}}: cada entrada es la suma de las dos anteriores.

\begin{theorem}[Teorema del binomio]\label{thm:g12:comb:binomial}
\index{teorema del binomio}
Para todos los $a, b \in \R$ (o $\C$) y $n \in \N$:
\[
(a+b)^n = \sum_{k=0}^{n} \binom{n}{k}\, a^{k}\, b^{\,n-k} .
\]
\end{theorem}

\begin{proof}
Ampliar el producto $(a+b)(a+b)\cdots(a+b)$ (factores $n$): cada término de la
La expansión selecciona $a$ o $b$ en cada factor, produciendo $a^k b^{n-k}$ donde
$k$ es el número de factores que contribuyen a $a$. La cantidad de formas de elegir.
estos factores $k$ entre $n$ es $\binom nk$, que por lo tanto es el
coeficiente de $a^k b^{n-k}$.
\end{proof}

\begin{corollary}\label{cor:g12:comb:sums}
$\displaystyle\sum_{k=0}^{n} \binom{n}{k} = 2^n$
\quad y \quad
$\displaystyle\sum_{k=0}^{n} (-1)^k\binom{n}{k} = 0$ \ ($n \geq 1$).
\end{corollary}

\begin{proof}
Tome $a = b = 1$, luego $a = -1$, $b = 1$ en el teorema del binomio. el primero
La identidad también tiene un significado directo: un conjunto de elementos $n$ tiene subconjuntos $2^n$
(cada elemento está dentro o fuera: principio de multiplicación), ordenados por tamaño.
\end{proof}

\begin{method}[Elegir el modelo correcto]\label{met:g12:comb:model}
Antes de contar, responda dos preguntas: \emph{¿Importa el orden?} y
\emph{¿Se permiten repeticiones?}
\begin{center}
\begin{tabular}{l|cc}
 & order matters & order irrelevant\\
\hline
repetitions allowed & $n^k$ (tuples) & (university)\\
no repetitions & $\frac{n!}{(n-k)!}$ (\omterm{def:g12:comb:tuples}{preparativos}) & $\binom nk$ (subsets)
\end{tabular}
\end{center}
Sacar bolas de una urna: \emph{con reemplazo, para} $\to$ tuplas;
\emph{sin reemplazo, con el fin} $\to$ \omterm{def:g12:comb:tuples}{preparativos}; \emph{un puñado de todos
de una vez} $\to$ subconjuntos.
\end{method}

\section{Ceremonias}

\begin{exercise}[$\star$]\label{exo:g12:comb:1}
Una matrícula consta de 2 letras (A--Z), luego 3 dígitos y luego 2 letras.
¿Cuántas placas son posibles? ¿Cuántos no tienen carácter repetido?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:comb:2}
Calcule $\dbinom{8}{3}$, $\dbinom{10}{8}$ y simplifique
$\dfrac{\binom{n}{2}}{\binom{n+1}{2}}$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:comb:3}
En una clase de 30 estudiantes, se debe elegir un comité de 4 estudiantes, luego un
presidente y un tesorero dentro del comité (una persona no puede ocupar ambos
oficinas). ¿Cuántos resultados son posibles?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:comb:4}
Expanda $(x + 2)^5$ y $(1 - x)^6$ usando el teorema del binomio. cual es el
coeficiente de $x^3$ en $(2x + 3)^7$?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:comb:5}
Una mano de póquer estándar consta de 5 cartas de una baraja de 52 cartas.
\begin{enumerate}
  \item ¿Cuántas manos hay?
  \item ¿Cuántas manos contienen exactamente un as? ¿Al menos un as?
  \item ¿Cuántas manos hay ``full'' (tres cartas de un valor, dos de
        otro)?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:comb:6}
¿Cuántos anagramas (reordenamientos de letras, significativos o no) tiene el
palabra MATEMÁTICAS? ¿La palabra PLÁTANO?
(Pista para BANANA: primero coloque las tres A.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:comb:7}
Acreditar la identidad $k\dbinom{n}{k} = n\dbinom{n-1}{k-1}$ ($1 \leq k \leq n$)
de dos maneras: mediante la fórmula \omterm{prop:g12:comb:tuples}{factorial}, y contando de dos maneras los pares
(comité de $k$ personas, su presidente) elegido entre $n$ personas.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:comb:8}
Un camino en el plano va de $(0,0)$ a $(m, n)$ en pasos unitarios Este o
Norte. Demuestre que el número de dichas rutas es $\dbinom{m+n}{m}$.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:comb:9}
Demuestre \emph{La identidad de Vandermonde}: para $0 \leq k \leq m + n$,
\[
\binom{m+n}{k} = \sum_{j=0}^{k} \binom{m}{j}\binom{n}{k-j},
\]
contando los subconjuntos de elementos $k$ de un conjunto dividido en un grupo de $m$ y un
grupo de $n$. deducir eso
$\displaystyle\sum_{j=0}^{n}\binom{n}{j}^{\!2} = \binom{2n}{n}$.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:comb:10}
Utilizando el teorema del binomio, demuestre que para todo $n \geq 1$,
\[
\sum_{k=1}^{n} k \binom{n}{k} = n\,2^{n-1}.
\]
(Sugerencia: diferencie $(1+x)^n$ o utilice \cref{exo:g12:comb:7}).
\end{exercise}

\section{Problema: el arte de contar dos veces}

\begin{problem}[{Problema de fin de semana --- estrellas y barras, trastornadas
sombreros e identidades demostradas contando una cosa de dos maneras}]
\label{pb:g12:comb:1}
El truco más profundo de la combinatoria es sorprendentemente simple: contar
la misma colección dos veces, por dos métodos diferentes, y equiparar
las respuestas. Este problema practica los modelos de
\cref{met:g12:comb:model}, agrega una técnica al curso del capítulo.
no necesitaba --- el \emph{estrellas y barras} de conteo de helados
--- luego cuenta exactamente el famoso \emph{sombreros trastornados}, y
encuentra el número $\frac1\eu$ esperando en la parte inferior del sombrero
pila, su tercera aparición en este libro.

\medskip
\noindent\textbf{Parte I --- Choosing the model.}
\begin{enumerate}
  \item Cuente las placas formadas por las letras $2$ seguidas de
        $3$ dígitos; luego los anagramas de BANANA.
  \item De una baraja de cartas $32$, cuente las manos de cartas $5$; entonces
        las manos que contienen exactamente $2$ de los ases $4$.
  \item Un robot camina desde $(0,0)$ a $(4,3)$ usando solo la unidad
        Pasos hacia la derecha o hacia arriba: ¿cuántos caminos? (Codifique una ruta como
        palabra en R y U.)
  \item Expande $(1 + x)^4$ por el teorema del binomio
        (\cref{thm:g12:comb:binomial}); luego evaluar en
        $x = 1$ y $x = -1$: ¿cuáles dos identidades sobre el
        los números $\binom nk$ abandonan?
  \item Demuestre por conteo doble que
        $k\binom nk = n\binom{n-1}{k-1}$ (cuenta
        comités-con-presidente de dos maneras), y deducir
        $\sum_{k=0}^{n} k\binom nk = n\,2^{n-1}$.
\end{enumerate}

\medskip
\noindent\textbf{Parte II --- Stars and bars.}
\begin{enumerate}[resume]
  \item Una heladería vende sabores $4$; usted ordena $10$
        cucharadas (los sabores pueden repetirse, el orden en la taza es
        irrelevante). Codificar un pedido como una fila de estrellas $10$
        (cucharadas) separadas por barras $3$ (cambios de sabor), y
        contar los pedidos.
  \item Cuente los triples de \omterm{def:g10:numbers:sets}{números enteros} no negativos con
        $x + y + z = 12$.
  \item Cuente los triples de \emph{positivo} \omterm{def:g10:numbers:sets}{números enteros} con
        $x + y + z = 12$ (sustituya $x = 1 + x'$, etc.).
  \item ¿Cuántos monomios distintos aparecen en la expansión de
        $(a + b + c)^5$?
  \item Método de verificación de cordura: contar los pedidos de $3$
        cucharadas de sabores $2$ con la fórmula, luego enumere
        Todos ellos y comparar.
  \item Diga exactamente dónde ingresó "las bolas son idénticas"
        codificación --- y contar lo que sucede en su lugar si el
        las cucharadas se comen en orden (posiciones distintas), con
        Lista de verificación de \cref{met:g12:comb:model}.
\end{enumerate}

\medskip
\noindent\textbf{Parte III --- The deranged hats.}
Un \emph{trastorno mental} es una redistribución de sombreros $n$ a sus
propietarios de $n$ en los que \emph{nadie} recibe su propio sombrero; dejar
$D_n$ cuéntalos. (\Cref{pb:g11:prob:1} mostró que un invitado
en promedio recupera su propio sombrero --- ahora contamos el total
fiestas desafortunadas exactamente.)
\begin{enumerate}[resume]
  \item Calcular $D_1$, $D_2$, $D_3$ por listado y $D_4$.
        pacientemente (o inteligentemente).
  \item Justificar la recurrencia
        $D_n = (n - 1)\left(D_{n-1} + D_{n-2}\right)$: invitado 1
        recibe un sombrero $k \neq 1$ (opciones $n - 1$); dividir
        según si el invitado $k$ recibe el sombrero 1 o no.
        Verifique que reproduzca $D_4$ y calcule $D_5$.
  \item Para $n = 3$, probar por inclusión--exclusión (resta
        las tareas de arreglar al menos un sombrero, vuelva a agregar el
        recuentos excesivos) que
        $D_3 = 3!\left(1 - \frac{1}{1!} + \frac{1}{2!} -
        \frac{1}{3!}\right)$ e indique la fórmula general.
  \item Calcular $\frac{D_5}{5!}$ y comparar con
        $\frac1\eu \approx 0.3679$: el \omterm{def:g10:proba:distribution}{probabilidad} que un
        gran grupo barajado desarregla completamente es
        $\frac1\eu$ --- el tercer cameo de esta constante, después
        la lotería y la secretaria de \cref{pb:g12:exp:1}.
        (Por qué: la fórmula de la pregunta 14 es el comienzo de una
        serie famosa para $\eu^{-1}$, contada en la universidad
        volúmenes.)
  \item Papá Noel secreto entre amigos de $10$: se sortean nombres
        uniformemente al azar. ¿Qué es el \omterm{def:g10:proba:distribution}{probabilidad} del sorteo?
        es válido (nadie se dibuja a sí mismo), y cuántos
        ¿Los nuevos sorteos que debería esperar el grupo?
\end{enumerate}

\medskip
\noindent\textbf{Parte IV --- Counting twice, winning twice.}
\begin{enumerate}[resume]
  \item El lema del apretón de manos: en cualquier fiesta, sumando a los invitados
        el número de manos que cada apretón cuenta cada apretón de manos
        exactamente dos veces. Deduce que \emph{el número de invitados
        quien estrechó un número impar de manos siempre es par} ---
        y compruebe que el reclamo tiene sentido en una fiesta de tres invitados.
  \item Prueba la joya
        $1^3 + 2^3 + \dots + n^3 = (1 + 2 + \dots + n)^2$ por
        inducción y verifíquelo para $n = 3$. (El pequeño Gauss
        suma, al cuadrado, cuenta cubos.)
  \item Identidad de Vandermonde (\cref{exo:g12:comb:9}) vía
        caminos: interpretar $\binom{2n}{n}$ como caminos de celosía de
        tipo de pregunta 3 de $(0,0)$ a $(n,n)$, corte cada una
        camino en su cruce de la antidiagonal, y explicar
        cómo aparece $\sum_j \binom nj^2$.
  \item Finale --- los cuatro movimientos del contador, una línea cada uno con
        un ejemplo de este problema: multiplica las etapas y suma
        casos; codificar inteligentemente (estrellas y barras, palabras de ruta);
        contar lo mismo dos veces (comité con presidente,
        apretones de manos); restar lo no deseado y corregir lo
        recuentos excesivos (trastornos). Y observe dónde está el conteo
        va a trabajar a continuación: \omterm{def:g10:proba:distribution}{probabilidad}, y las rutas del
        Capítulo de matrices y gráficas.
\end{enumerate}
\end{problem}
