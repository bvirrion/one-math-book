\chapter{Secuencias}\label{ch:g12:seq}

Un \omterm{def:g12:seq:sequence}{secuencia} es una lista de \omterm{def:g10:numbers:sets}{números reales} indexados por \omterm{def:g10:numbers:sets}{números naturales}. \omterm{def:g12:seq:sequence}{Secuencias}
modelo de evoluciones discretas --- poblaciones contadas año tras año, saldos de un
cuenta bancaria, \omterm{def:g10:numbers:approx}{aproximaciones} sucesivos de un número --- y sus límites son
el primer encuentro serio con el infinito. Este capítulo establece el
vocabulario, el principio de inducción y la convergencia fundamental
teoremas

\section{Razonamiento por inducción}

\begin{theorem}[Principio de inducción]\label{thm:g12:seq:induction}
Sea $P(n)$ una declaración que dependa de \omterm{def:g10:numbers:sets}{entero} $n$ y sea $n_0 \in \N$.
si
\begin{enumerate}
  \item \emph{(caso base)} $P(n_0)$ es cierto, y
  \item \emph{(paso inductivo)} por cada $n \geq n_0$, $P(n)$ implica $P(n+1)$,
\end{enumerate}
entonces $P(n)$ es verdadero para cada $n \geq n_0$.
\end{theorem}

\begin{proof}
Supongamos, por contradicción, que el conjunto $A$ de \omterm{def:g10:numbers:sets}{números enteros} $n \geq n_0$ para
cuyo $P(n)$ es falso no está vacío. Entonces $A$ tiene un elemento más pequeño.
$m$.\footnote{Every nonempty subset of $\N$ has a smallest element; this
property of $\N$ is taken as an axiom.} Dado que $P(n_0)$ es verdadero, $m > n_0$, entonces
$m - 1 \geq n_0$ y $m-1 \notin A$, es decir,\ $P(m-1)$ es verdadero. el inductivo
El paso aplicado a $n = m-1$ luego muestra que $P(m)$ es verdadero, lo que contradice
$m \in A$.
\end{proof}

\begin{example}\label{ex:g12:seq:bernoulli}
Probemos \emph{La desigualdad de Bernoulli}: para cada $a > 0$ real y cada
$n \in \N$,
\[
(1+a)^n \geq 1 + na.
\]
\emph{Caso base.} Para $n = 0$, ambos lados son iguales a $1$.
\emph{Paso inductivo.} Supongamos $(1+a)^n \geq 1+na$ para algunos $n \in \N$. desde
$1 + a > 0$, multiplicar ambos lados por $1+a$ preserva la desigualdad:
\[
(1+a)^{n+1} \geq (1+na)(1+a) = 1 + (n+1)a + na^2 \geq 1 + (n+1)a .
\]
Por inducción, la desigualdad es válida para todo $n \in \N$.
\end{example}

\begin{method}[Escribiendo una prueba de inducción]\label{met:g12:seq:induction}
Haga siempre explícita la declaración $P(n)$ antes de comenzar. Una prueba completa
tiene tres partes visibles: el caso base, el paso inductivo (``supongamos $P(n)$;
probamos $P(n+1)$''), y la conclusión invocando el principio de inducción.
El error más común es probar el paso inductivo sin utilizar nunca el
Hipótesis $P(n)$: si eso sucede, o la prueba es incorrecta o la inducción fue incorrecta.
no es necesario.
\end{method}

\section{Vocabulario de secuencias.}

\begin{definition}[Secuencia]\label{def:g12:seq:sequence}
Un \emph{secuencia}\index{secuencia} es un \omterm{def:g11:func:function}{función} $u \colon \N \to \R$ (o de
$\{n \in \N : n \geq n_0\}$ a $\R$). El \omterm{def:g10:functions:function}{imagen} de $n$ se escribe $u_n$, y
la secuencia misma $(u_n)_{n\in\N}$ o simplemente $(u_n)$.
\end{definition}

Un \omterm{def:g12:seq:sequence}{secuencia} puede definirse como \emph{explícitamente}, mediante una fórmula $u_n = f(n)$ o mediante
\emph{reaparición}, por su primer término y una relación $u_{n+1} = f(u_n)$.

\begin{definition}[Monotonicidad]\label{def:g12:seq:monotonic}
A \omterm{def:g12:seq:sequence}{secuencia} $(u_n)$ es \emph{creciente}\index{secuencia!creciente} si
$u_{n+1} \geq u_n$ para todos $n$, \emph{\omterm{def:g10:functions:variations}{decreciente}} si $u_{n+1} \leq u_n$ para
todo $n$ y \emph{monótono} si es creciente o \omterm{def:g10:functions:variations}{decreciente}. es
\emph{estrictamente} aumenta (resp.\ \omterm{def:g10:functions:variations}{decreciente}) cuando las desigualdades son
estricto.
\end{definition}

\begin{method}[Estudiando la monotonicidad de una secuencia]\label{met:g12:seq:monotonicity}
Tres técnicas estándar:
\begin{enumerate}
  \item estudia el signo de $u_{n+1} - u_n$;
  \item si todos los términos son positivos, compare $\dfrac{u_{n+1}}{u_n}$ con $1$;
  \item si $u_n = f(n)$ con $f$ definido en $\intco{0}{+\infty}$, use el
        variaciones de $f$.
\end{enumerate}
\end{method}

\begin{definition}[Secuencia acotada]\label{def:g12:seq:bounded}
A \omterm{def:g12:seq:sequence}{secuencia} $(u_n)$ es \emph{delimitado arriba}\index{secuencia!encerrado} si hay
existe $M \in \R$ con $u_n \leq M$ para todos los $n$; \emph{delimitado por debajo} si hay
existe $m \in \R$ con $u_n \geq m$ para todos los $n$; y \emph{encerrado} si ambos
espera.
\end{definition}

\subsection{Arithmetic and geometric sequences}

\begin{definition}[Secuencias aritméticas y geométricas.]\label{def:g12:seq:arith-geom}
A \omterm{def:g12:seq:sequence}{secuencia} $(u_n)$ es \emph{aritmética}\index{secuencia!aritmética} con \omterm{def:g11:seq:arithmetic}{común
diferencia} $r$ si $u_{n+1} = u_n + r$ para todos $n$, y
\emph{geométrico}\index{secuencia!geométrico} con \omterm{def:g11:seq:geometric}{proporción común} $q$ si
$u_{n+1} = q\,u_n$ para todos los $n$.
\end{definition}

\begin{proposition}[forma y sumas]\label{prop:g12:seq:arith-geom}
Deje $n \in \N$.
\begin{enumerate}
  \item Si $(u_n)$ es \omterm{def:g12:seq:arith-geom}{aritmética} con \omterm{def:g11:seq:arithmetic}{diferencia común} $r$, entonces
        $u_n = u_0 + nr$ y
        \[
        u_0 + u_1 + \dots + u_n = (n+1)\,\frac{u_0 + u_n}{2}.
        \]
  \item Si $(u_n)$ es \omterm{def:g12:seq:arith-geom}{geométrico} con \omterm{def:g11:seq:geometric}{proporción común} $q \neq 1$, entonces
        $u_n = u_0\, q^n$ y
        \[
        u_0 + u_1 + \dots + u_n = u_0\,\frac{1 - q^{n+1}}{1 - q}.
        \]
\end{enumerate}
\end{proposition}

\begin{proof}
Las formas explícitas siguen por inducciones inmediatas. Para la suma \omterm{def:g12:seq:arith-geom}{aritmética},
escribe $S = u_0 + \dots + u_n$ y suma la misma suma escrita en orden inverso:
cada una de las sumas de las columnas $n+1$ es igual a $u_0 + u_n$, por lo que $2S = (n+1)(u_0+u_n)$.
Para la suma \omterm{def:g12:seq:arith-geom}{geométrico}, calcule $S - qS$: todos los términos se cancelan en pares excepto el
primero y último, entonces $(1-q)S = u_0(1 - q^{n+1})$.
\end{proof}

\section{Límite de una secuencia}

\begin{definition}[secuencia convergente]\label{def:g12:seq:limit}
A \omterm{def:g12:seq:sequence}{secuencia} $(u_n)$ \emph{converge}\index{límite!de una secuencia} al \omterm{def:g10:numbers:sets}{real
numero} $\ell$ si cada \omterm{def:g10:numbers:interval}{intervalo} abierto que contiene $\ell$ contiene todos los términos
$u_n$ a partir de algún índice. Luego escribimos $\lim\limits_{n\to+\infty} u_n = \ell$.

De manera equivalente: por cada $\varepsilon > 0$, existe $N \in \N$ tal que
para todos $n \geq N$, $\abs{u_n - \ell} \leq \varepsilon$.
\end{definition}

\begin{definition}[Divergencia al infinito]\label{def:g12:seq:infinity}
El \omterm{def:g12:seq:sequence}{secuencia} $(u_n)$ \emph{tends to $+\infty$} si por cada $A \in \R$, hay
existe $N \in \N$ tal que $u_n \geq A$ para todos los $n \geq N$. escribimos
$\lim\limits_{n\to+\infty} u_n = +\infty$; la definición de
$\lim u_n = -\infty$ es análogo. Se dice un \omterm{def:g12:seq:sequence}{secuencia} que no converge
a \emph{divergir}.
\end{definition}

\begin{remark}
Un \omterm{def:g12:seq:sequence}{secuencia} puede divergir sin atender a $\pm\infty$: el \omterm{def:g12:seq:sequence}{secuencia}
$u_n = (-1)^n$ toma sólo los valores $1$ y $-1$ y no tiene límite.
\end{remark}

\begin{proposition}[Singularidad del límite]\label{prop:g12:seq:uniqueness}
Si $(u_n)$ \omterm{def:g12:seq:limit}{converge}, su límite es único.
\end{proposition}

\begin{proof}
Supongamos $u_n \to \ell$ y $u_n \to \ell'$ con $\ell \neq \ell'$, digamos
$\ell < \ell'$. Establezca $\varepsilon = \frac{\ell' - \ell}{3} > 0$. De algunos
índice en, $\abs{u_n - \ell} \leq \varepsilon$ y
$\abs{u_n - \ell'} \leq \varepsilon$, por lo tanto
\[
\ell' - \ell \leq \abs{\ell' - u_n} + \abs{u_n - \ell}
\leq 2\varepsilon = \tfrac{2}{3}(\ell' - \ell) < \ell' - \ell,
\]
una contradicción.
\end{proof}

\begin{proposition}[Operaciones en límites]\label{prop:g12:seq:operations}
Sean $(u_n)$ y $(v_n)$ \omterm{def:g12:seq:sequence}{secuencias} con límites $\ell$ y $\ell'$ (finitos o
infinito). Entonces, siempre que el lado derecho no sea una forma indeterminada,
\[
\lim (u_n + v_n) = \ell + \ell', \qquad
\lim (u_n v_n) = \ell\,\ell', \qquad
\lim \frac{u_n}{v_n} = \frac{\ell}{\ell'}.
\]
Las formas indeterminadas son
$(+\infty) + (-\infty)$, $0 \times \infty$, $\frac{\infty}{\infty}$ y
$\frac{0}{0}$.
\end{proposition}

\begin{proof}
Probamos la regla de la suma para límites finitos; los otros casos son similares y quedan
como ejercicios. Sea $\varepsilon > 0$. Existe $N_1, N_2$ tal que
$\abs{u_n - \ell} \leq \varepsilon/2$ para $n \geq N_1$ y
$\abs{v_n - \ell'} \leq \varepsilon/2$ para $n \geq N_2$. Para
$n \geq \max(N_1, N_2)$, la desigualdad del triángulo da
\[
\abs{(u_n + v_n) - (\ell + \ell')}
\leq \abs{u_n - \ell} + \abs{v_n - \ell'} \leq \varepsilon. \qedhere
\]
\end{proof}

\begin{method}[Levantando una forma indeterminada]\label{met:g12:seq:indeterminate}
Ante una forma indeterminada, factorice el término dominante. por ejemplo
\[
n^2 - n = n^2\left(1 - \tfrac{1}{n}\right) \xrightarrow[n\to+\infty]{} +\infty,
\qquad
\frac{2n^2+1}{n^2 - n} = \frac{2 + 1/n^2}{1 - 1/n}
\xrightarrow[n\to+\infty]{} 2 .
\]
\end{method}

\section{Teoremas de convergencia}

\begin{theorem}[Teoremas de comparación y compresión]\label{thm:g12:seq:squeeze}
Sea $(u_n)$, $(v_n)$, $(w_n)$ \omterm{def:g12:seq:sequence}{secuencias}.
\begin{enumerate}
  \item Si $u_n \leq v_n$ de algún índice en adelante y $u_n \to +\infty$, entonces
        $v_n \to +\infty$.
  \item \emph{(Teorema de compresión)}\index{teorema de compresión} Si
        $u_n \leq v_n \leq w_n$ de algún índice en adelante y $(u_n)$ y $(w_n)$
        ambos convergen al mismo límite $\ell$, luego $(v_n)$ \omterm{def:g12:seq:limit}{converge}
        a ~$\ell$.
\end{enumerate}
\end{theorem}

\begin{proof}
\emph{1.} Vamos $A \in \R$. Desde $u_n \to +\infty$, existe $N$ con
$u_n \geq A$ para $n \geq N$; ampliando $N$ si es necesario, $v_n \geq u_n \geq A$
para $n \geq N$.

\emph{2.} Vamos $\varepsilon > 0$. A partir de algún índice, ambos
$\ell - \varepsilon \leq u_n$ y $w_n \leq \ell + \varepsilon$ se mantienen, por lo tanto
$\ell - \varepsilon \leq u_n \leq v_n \leq w_n \leq \ell + \varepsilon$, que
es $\abs{v_n - \ell} \leq \varepsilon$.
\end{proof}

\begin{example}
Para todos los $n \geq 1$, $-\frac{1}{n} \leq \frac{(-1)^n}{n} \leq \frac{1}{n}$ y
ambos límites tienden a $0$; por lo tanto $\frac{(-1)^n}{n} \to 0$.
\end{example}

\begin{theorem}[Teorema de convergencia monótona]\label{thm:g12:seq:monotone}
Un \omterm{def:g12:seq:monotonic}{creciente} \omterm{def:g12:seq:sequence}{secuencia} que es \omterm{def:g12:seq:bounded}{delimitado arriba} \omterm{def:g12:seq:limit}{converge}. \omterm{def:g10:functions:variations}{decreciente} \omterm{def:g12:seq:sequence}{secuencia}
es decir \omterm{def:g12:seq:bounded}{delimitado por debajo} \omterm{def:g12:seq:limit}{converge}. Un \omterm{def:g12:seq:monotonic}{creciente} \omterm{def:g12:seq:sequence}{secuencia} que no es \omterm{def:g12:seq:bounded}{delimitado
arriba} tiende a $+\infty$.
\end{theorem}

\begin{proof}[prueba parcial]
Probamos la tercera afirmación. Sea $(u_n)$ \omterm{def:g12:seq:monotonic}{creciente} y no \omterm{def:g12:seq:bounded}{delimitado
arriba}, y sea $A \in \R$. Dado que $A$ no es un límite superior, existe $N$
con $u_N \geq A$; por monotonicidad, $u_n \geq u_N \geq A$ para todos los $n \geq N$.
Por lo tanto $u_n \to +\infty$.

Las dos declaraciones de convergencia se basan en la propiedad del límite superior mínimo de
$\R$; son \emph{admitido a este nivel} (y probados en el primer año de
universidad).
\end{proof}

\begin{remark}
El teorema garantiza el \emph{existencia} del límite pero no da
su valor. Un \omterm{def:g12:seq:monotonic}{creciente} \omterm{def:g12:seq:sequence}{secuencia} \omterm{def:g12:seq:bounded}{delimitado arriba} por $M$ \omterm{def:g12:seq:limit}{converge} para algunos
$\ell \leq M$, no necesariamente a $M$.
\end{remark}

\begin{theorem}[Límite de secuencias geométricas]\label{thm:g12:seq:geometric}
Deje $q \in \R$.
\begin{enumerate}
  \item Si $q > 1$, entonces $q^n \to +\infty$.
  \item Si $q = 1$, entonces $q^n \to 1$.
  \item Si $\abs{q} < 1$, entonces $q^n \to 0$.
  \item Si $q \leq -1$, entonces $(q^n)$ diverge y no tiene límite.
\end{enumerate}
\end{theorem}

\begin{proof}
\emph{1.} Escriba $q = 1 + a$ con $a > 0$. La desigualdad de Bernoulli
(\cref{ex:g12:seq:bernoulli}) da $q^n \geq 1 + na \to +\infty$, y nosotros
concluir por comparación (\cref{thm:g12:seq:squeeze}).

\emph{2.} Inmediato.

\emph{3.} Si $q = 0$ el reclamo es claro. De lo contrario $\abs{q} < 1$ da
$1/\abs{q} > 1$, entonces $(1/\abs{q})^n \to +\infty$ por el punto 1, por lo tanto
$\abs{q}^n \to 0$ y $-\abs{q}^n \leq q^n \leq \abs{q}^n$ nos permiten
concluir por el teorema de compresión.

\emph{4.} Para $q \leq -1$, $(q^{2n})$ toma valores $\geq 1$ mientras
$(q^{2n+1})$ toma valores $\leq -1$: ningún límite puede atraer a ambos
subsecuencias.
\end{proof}

\begin{method}[Secuencias recurrentes $u_{n+1} = f(u_n)$]\label{met:g12:seq:recurrent}
Para estudiar un \omterm{def:g12:seq:sequence}{secuencia} definido por $u_{n+1} = f(u_n)$:
\begin{enumerate}
  \item demostrar por inducción que $(u_n)$ permanece en un \omterm{def:g10:numbers:interval}{intervalo} $I$ en el que $f$
        se porta bien (y, a menudo, que $(u_n)$ es \omterm{def:g12:seq:monotonic}{monótono});
  \item deduce la convergencia a partir del teorema de convergencia monótona;
  \item pasa al límite en la relación $u_{n+1} = f(u_n)$: si $f$ es
        continuo y $u_n \to \ell \in I$, entonces $\ell$ satisface
        $f(\ell) = \ell$ (ver \cref{ch:g12:limcont}); resuelve este \omterm{def:g10:algebra:equation}{ecuación} y
        seleccione el \omterm{def:g11:quad:discriminant}{raíz} correcto.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g12:seq:sqrt2}
Sean $u_0 = 2$ y $u_{n+1} = \frac{1}{2}\left(u_n + \frac{2}{u_n}\right)$.
Se comprueba por inducción que $u_n \geq \sqrt{2}$ para todo $n$ (la desigualdad
$\frac{1}{2}(x + 2/x) \geq \sqrt{2}$ para $x>0$ es equivalente a
$(x - \sqrt2)^2 \geq 0$), entonces ese $(u_n)$ es \omterm{def:g10:functions:variations}{decreciente}, ya que
\[
u_{n+1}-u_n=\frac{2-u_n^2}{2u_n}\leq 0 .
\]
\omterm{def:g10:functions:variations}{Decreciente} y \omterm{def:g12:seq:bounded}{delimitado por debajo}, $(u_n)$
\omterm{def:g12:seq:limit}{converge} a algún $\ell \geq \sqrt{2}$, que debe satisfacer
$\ell = \frac{1}{2}(\ell + 2/\ell)$, es decir$\ell^2 = 2$. Por lo tanto
$u_n \to \sqrt{2}$. Este es el algoritmo de Heron, ya utilizado por el
babilonios; su convergencia es extremadamente rápida ($u_3$ ya da $\sqrt 2$
hasta ocho decimales).
\end{example}

\section{Ceremonias}

\begin{exercise}[$\star$]\label{exo:g12:seq:1}
Demuestre por inducción que para todo $n \in \N$,
\[
1^2 + 2^2 + \dots + n^2 = \frac{n(n+1)(2n+1)}{6}.
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:seq:2}
Estudiar la monotonicidad del \omterm{def:g12:seq:sequence}{secuencias} definido para $n \geq 1$ por
\[
a_n = \frac{n+1}{n}, \qquad
b_n = \frac{2^n}{n}, \qquad
c_n = n^2 - 10n .

\section{Problema: la secuencia de Heron, finalmente juzgada}

\begin{problem}[{Problema de fin de semana --- inducción certifica, monotono
sentencias de convergencia y la receta de hace dos mil años para
$\sqrt2$ finalmente obtiene su prueba (con la maravillosa media de Gauss para
postre)}]
\label{pb:g12:seq:1}
Tres veces esta serie ha cumplido con la receta de Heron --- promedia el
adivina con $2/\text{guess}$ --- y tres veces solo pudo
\emph{observar} que la receta funciona. Este capítulo por fin posee
los instrumentos de juicio: la inducción
(\cref{thm:g12:seq:induction}), el teorema de convergencia monótona
(\cref{thm:g12:seq:monotone}), y límites de recurrencias. el
veredicto y la velocidad certificada ocupan el centro de este
problema; A su alrededor, las clásicas trampas de la inducción, las más lentas.
divergencia en matemáticas, y la convergencia más rápida de Gauss jamás
encontrado.

\medskip
\noindent\textbf{Parte I --- Induction warm-ups.}
\begin{enumerate}
  \item Demostrar por inducción:
        $1 + 3 + 5 + \dots + (2n - 1) = n^2$ (la escalera de
        números impares, dibujados en el volumen de la Escuela Secundaria, ahora
        certificado).
  \item Demuestre por inducción que $2^n > n$ para cada
        $n \in \N$.
  \item Demuestre la desigualdad de Bernoulli por inducción: para
        $x \geq 0$ y $n \in \N$,
        $(1 + x)^n \geq 1 + nx$.
  \item La clásica trampa: ``todas las canicas tienen el mismo color ---
        cierto para una canica; y si alguna canica $n$ siempre está
        monocromo, luego entre las canicas $n + 1$ la primera $n$
        comparten un color, el último $n$ comparte un color, por lo que todos
        $n + 1$ hacer". Cada niño sabe que la conclusión es
        Absurdo: encontrar el paso exacto donde se rompe la inducción.
  \item Demostrar por inducción que $4^n - 1$ es divisible por $3$
        por cada $n \in \N$.
\end{enumerate}

\medskip
\noindent\textbf{Parte II --- The trial of Heron.}
Sean $x_0 = 2$ y $x_{n+1} = \dfrac12\left(x_n +
\dfrac{2}{x_n}\right)$.
\begin{enumerate}[resume]
  \item Calcular $x_1$, $x_2$, $x_3$ como fracciones exactas (antiguo
        amigos).
  \item Demostrar la identidad de la clave
        \[
        x_{n+1}^2 - 2 = \left(\frac{x_n^2 - 2}{2x_n}\right)^{\!2}
        \geq 0,
        \]
        y deducir por inducción que $x_n > 0$ y
        $x_n^2 > 2$ por cada $n$.
  \item Demuestre que $(x_n)$ es estrictamente \omterm{def:g10:functions:variations}{decreciente} (calcule
        $x_{n+1} - x_n$ y utilice la pregunta 7).
  \item Invocar el teorema de convergencia monótona: ¿por qué
        ¿$(x_n)$ converge a algún límite $L \geq 1$?
  \item Identificar el límite: pasar la recurrencia al límite
        (\cref{prop:g12:seq:operations}) y concluir
        $L = \sqrt2$. Enuncia el veredicto histórico: después de dos
        mil años de fiel servicio, la receta de Heron es
        \emph{probado} para converger.
  \item La velocidad certificada: con $e_n = x_n - \sqrt2$, demostrar
        \[
        e_{n+1} = \frac{e_n^2}{2 x_n} ,
        \]
        y deducir $e_{n+1} \leq \frac{e_n^2}{2\sqrt2}$: el
        el error se eleva al cuadrado en cada paso --- la duplicación de dígitos
        observado desde el volumen de la escuela secundaria, ahora un teorema.
  \item Confirmar numéricamente: calcular $e_0, e_1, e_2, e_3$ (de
        pregunta 6) y verifique que cada $\frac{e_{n+1}}{e_n^2}$
        está cerca de $\frac{1}{2x_n}$.
\end{enumerate}

\medskip
\noindent\textbf{Parte III --- The slowest divergence.}
\begin{enumerate}[resume]
  \item \Cref{exo:g12:seq:9} probado
        $H_{2^k} \geq 1 + \frac k2$ para las sumas armónicas. como
        ¿Cuántos términos garantizan $H_n > 10$? (Una potencia de dos voluntades
        hacer; Maravíllate con su tamaño.)
  \item Por el contrario, el \omterm{def:g12:seq:arith-geom}{geométrico} suma
        $1 + \frac12 + \frac14 + \dots + \frac{1}{2^n} =
        2 - \frac{1}{2^n}$ converge a $2$
        (\cref{thm:g12:seq:geometric}): la barra de chocolate
        Intuición del volumen de la Escuela Secundaria, finalmente un límite.
        declaración. Escribe la prueba de dos líneas.
  \item Entre los dos: muestra que las sumas
        $S_n = 1 + \frac{1}{4} + \frac{1}{9} + \dots +
        \frac{1}{n^2}$ converge, por delimitación
        $\frac{1}{k^2} \leq \frac{1}{k(k-1)} =
        \frac{1}{k-1} - \frac{1}{k}$ (para $k \geq 2$),
        telescópico y aplicación de convergencia monótona. (El
        límite, $\frac{\pi^2}{6}$, es uno de los milagros de Euler,
        demostrado en los volúmenes de la universidad.)
  \item Indique la moraleja de las preguntas 13 a 15 en dos oraciones:
        ¿Qué deciden "los términos tienden a $0$" sobre el
        convergencia de las sumas --- ¿y qué no?
\end{enumerate}

\medskip
\noindent\textbf{Parte IV --- Gauss's arithmetic--geometric
\omterm{def:g11:stat:mean}{significar}.}
Dejemos que $a_0 = 1$, $b_0 = 2$ y
\[
a_{n+1} = \sqrt{a_n b_n},
\qquad
b_{n+1} = \frac{a_n + b_n}{2} .
\]
\begin{enumerate}[resume]
  \item Calcula $a_1, b_1, a_2, b_2$ (cinco decimales). ¿Qué hacer?
        observas sobre la velocidad?
  \item Muestre que $a_n \leq b_n$ por cada $n$ (el
        aritmética--desigualdad geométrica, encontrada a lo largo de este
        serie), que $(a_n)$ aumenta y $(b_n)$ disminuye.
  \item Demuestra que
        $b_{n+1} - a_{n+1} \leq \frac{b_n - a_n}{2}$
        (factor $b_{n+1} - a_{n+1} =
        \frac{(\sqrt{b_n} - \sqrt{a_n})^2}{2}$ y comparar),
        y concluir con \cref{exo:g12:seq:10} que los dos
        \omterm{def:g12:seq:sequence}{secuencias} son adyacentes: comparten un límite común
        $M(1, 2)$, el \emph{arithmetic--geometric \omterm{def:g11:stat:mean}{significar}}.
  \item Calcule $M(1, 2)$ con seis decimales (cuántas iteraciones
        necesitabas?). El 30 de mayo de 1799, Gauss calculó
        $M(1, \sqrt2)$ hasta once decimales, reconocido
        $\frac{\pi}{M(1,\sqrt2)}$ como integral conocida, y
        escribió que se había abierto un "nuevo campo de análisis" ---
        tenía: integrales elípticas, contadas en la universidad
        volúmenes. Cierre con las velocidades de convergencia observadas de
        este problema, del más lento al más rápido.
\end{enumerate}
\end{problem}
