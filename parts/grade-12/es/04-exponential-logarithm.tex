\chapter{Exponencial y Logaritmo}\label{ch:g12:exp}

The \omterm{thm:g12:exp:existence}{función exponencial} is the unique \omterm{def:g11:func:function}{función} equal to its own \omterm{def:g12:deriv:derivative}{derivado}
y tomando el valor $1$ en $0$. Convierte sumas en productos; es inverso,
el \omterm{def:g12:exp:ln}{logaritmo natural}, convierte productos en sumas. Juntos describen
every phenomenon whose rate of change is proportional to its size:
radioactive decay, population growth, compound interest.

\section{La función exponencial}

\begin{theorem}[Existencia y unidad]\label{thm:g12:exp:existence}
Existe un \omterm{def:g12:deriv:derivative}{diferenciable} \omterm{def:g11:func:function}{función} $f \colon \R \to \R$ único tal que
\[
f' = f \qquad\text{and}\qquad f(0) = 1 .
\]
Se llama \emph{función exponencial}\index{exponencial} y está escrito
$\exp$ o $x \mapsto \eu^x$.
\end{theorem}

\begin{proof}[prueba de unicidad]
Primero, tal \omterm{def:g11:func:function}{función} nunca desaparece. De hecho, deja
$g(x) = f(x)f(-x)$; entonces
\[
g'(x) = f'(x)f(-x) - f(x)f'(-x) = f(x)f(-x) - f(x)f(-x) = 0,
\]
entonces $g$ es una constante igual a $g(0) = 1$: por cada $x$,
$f(x)f(-x) = 1$, y en particular $f(x) \neq 0$.

Ahora dejemos que $f_1, f_2$ sean dos soluciones y $h = \dfrac{f_1}{f_2}$ (legítimo
ya que $f_2$ nunca desaparece). entonces
\[
h' = \frac{f_1' f_2 - f_1 f_2'}{f_2^2} = \frac{f_1 f_2 - f_1 f_2}{f_2^2} = 0,
\]
entonces $h$ es una constante igual a $h(0) = 1$, \emph{i.e.} $f_1 = f_2$.

En este nivel se admite \emph{Existencia} (se puede obtener a través del
\omterm{def:g12:seq:sequence}{secuencia} de \cref{exo:g12:seq:10}, o como el inverso del \omterm{def:g12:exp:ln}{logaritmo}
construido por integración en \cref{ch:g12:integ}).
\end{proof}

\begin{proposition}[Ecuación funcional]\label{prop:g12:exp:funceq}
Para todos los $x, y \in \R$ y $n \in \Z$:
\[
\eu^{x+y} = \eu^x\,\eu^y, \qquad
\eu^{-x} = \frac{1}{\eu^x}, \qquad
\eu^{x-y} = \frac{\eu^x}{\eu^y}, \qquad
\bigl(\eu^{x}\bigr)^n = \eu^{nx}.
\]
Además $\eu^x > 0$ para todos los $x$.
\end{proposition}

\begin{proof}
Arregle $y$ y considere $\varphi(x) = \dfrac{\exp(x+y)}{\exp(x)\exp(y)}$. su
numerador y denominador, como \omterm{def:g11:func:function}{funciones} de $x$, son soluciones de
$f' = f$ hasta la constante $\exp(y)$; diferenciando $\varphi$ directamente
(regla del cociente) da $\varphi' = 0$, entonces
$\varphi \equiv \varphi(0) = 1$, acreditando la primera identidad. tomando
$y = -x$ da el segundo (con el valor $\eu^0 = 1$), y el tercero
sigue. La cuarta es una inducción de la primera para $n \geq 0$, extendida
a $n < 0$ por segundo.

Positividad: $\eu^x = \left(\eu^{x/2}\right)^2 \geq 0$ y $\eu^x \neq 0$
(mostrado en \cref{thm:g12:exp:existence}), entonces $\eu^x > 0$.
\end{proof}

\begin{proposition}[Variaciones y límites]\label{prop:g12:exp:variations}
El exponencial es estrictamente \omterm{def:g12:seq:monotonic}{creciente}, \omterm{def:g12:deriv:convex}{convexo} y
\[
\lim_{x\to-\infty} \eu^x = 0, \qquad
\lim_{x\to+\infty} \eu^x = +\infty .
\]
\end{proposition}

\begin{proof}
$(\exp)' = \exp > 0$ da aumento estricto; $(\exp)'' = \exp > 0$ da
\omterm{def:g12:deriv:convex}{convexidad}. Por \omterm{def:g12:deriv:convex}{convexidad}, $\eu^x \geq 1 + x$ (\omterm{def:g12:deriv:tangent}{tangente} en $0$, ver
\cref{ex:g12:deriv:ineq}), entonces $\eu^x \to +\infty$ como $x \to +\infty$ por
comparación. Luego $\eu^{x} = 1/\eu^{-x} \to 0$ como $x \to -\infty$.
\end{proof}

\begin{theorem}[Comparación de crecimiento]\label{thm:g12:exp:growth}
\index{comparación de crecimiento}
Por cada \omterm{def:g10:numbers:sets}{entero} $n \geq 1$,
\[
\lim_{x\to+\infty} \frac{\eu^x}{x^n} = +\infty,
\qquad
\lim_{x\to-\infty} x^n \eu^x = 0 .
\]
En palabras: el exponencial supera todas las potencias de $x$.
\end{theorem}

\begin{proof}
Para $n = 1$: aplicando $\eu^t \geq 1 + t \geq t$ en $t = x/2$,
\[
\frac{\eu^x}{x} = \frac{\left(\eu^{x/2}\right)^2}{x}
\geq \frac{(x/2)^2}{x} = \frac{x}{4} \xrightarrow[x\to+\infty]{} +\infty .
\]
Para $n$ general, escriba
\[
\frac{\eu^x}{x^n}
= \frac{1}{n^n}\left(\frac{\eu^{x/n}}{x/n}\right)^{n}
\xrightarrow[x\to+\infty]{} +\infty
\]
por el caso $n=1$ y composición. El límite en $-\infty$ sigue el
sustitución $x \mapsto -x$:
$\abs{x^n \eu^x} = \frac{\abs{x}^n}{\eu^{\abs x}} \to 0$.
\end{proof}

\section{El logaritmo natural}

\begin{definition}[logaritmo natural]\label{def:g12:exp:ln}
El exponencial es \omterm{def:g12:limcont:continuity}{continuo} y estrictamente \omterm{def:g12:seq:monotonic}{creciente} desde $\R$ hasta
$\intoo{0}{+\infty}$; por el teorema de biyección
(\cref{thm:g12:limcont:bijection}), por cada $y > 0$ el \omterm{def:g10:algebra:equation}{ecuación}
$\eu^x = y$ tiene una solución única. Esta solución es \emph{naturales
logaritmo}\index{logaritmo} de $y$, escrito $\ln y$. Así
\[
\text{for } x \in \R,\ y > 0: \qquad y = \eu^x \iff x = \ln y .
\]
En particular $\ln 1 = 0$, $\ln \eu = 1$ y
$\eu^{\ln y} = y$, $\ln(\eu^x) = x$.
\end{definition}

\begin{proposition}[Propiedades algebraicas]\label{prop:g12:exp:lnalg}
Para todos los $a, b > 0$ y $n \in \Z$:
\[
\ln(ab) = \ln a + \ln b, \quad
\ln\frac{1}{a} = -\ln a, \quad
\ln\frac{a}{b} = \ln a - \ln b, \quad
\ln(a^n) = n \ln a, \quad
\ln\sqrt{a} = \tfrac12 \ln a .
\]
\end{proposition}

\begin{proof}
$\eu^{\ln a + \ln b} = \eu^{\ln a}\eu^{\ln b} = ab$, y tomando $\ln$ de ambos
lados da la primera identidad. Los demás siguen por el mismo mecanismo desde
las identidades correspondientes de \cref{prop:g12:exp:funceq}.
\end{proof}

\begin{proposition}[Propiedades analíticas]\label{prop:g12:exp:lnanalytic}
El \omterm{def:g11:func:function}{función} $\ln$ es \omterm{def:g12:deriv:derivative}{diferenciable} en $\intoo{0}{+\infty}$ con
\[
(\ln x)' = \frac{1}{x},
\]
estrictamente \omterm{def:g12:seq:monotonic}{creciente}, \omterm{def:g12:deriv:convex}{cóncavo} y
$\lim\limits_{x\to0^+} \ln x = -\infty$,
$\lim\limits_{x\to+\infty} \ln x = +\infty$. Además, por cada \omterm{def:g10:numbers:sets}{entero}
$n \geq 1$,
\[
\lim_{x\to+\infty} \frac{\ln x}{x^{1/n}} = 0, \qquad
\lim_{x\to0^+} x \ln x = 0 .
\]
\end{proposition}

\begin{proof}
En este nivel se admite la diferenciabilidad de la inversa \omterm{def:g11:func:function}{función}; otorgando
diferenciar la identidad $\eu^{\ln x} = x$ por la regla de la cadena:
$(\ln x)'\,\eu^{\ln x} = 1$, entonces $(\ln x)' = \frac{1}{x} > 0$, de donde es estricto
aumento, y $(\ln x)'' = -\frac1{x^2} < 0$, de donde la concavidad. Los límites en
$0^+$ y $+\infty$ reflejan los de $\exp$ a través de la biyección. Para el
\omterm{thm:g12:exp:growth}{comparación de crecimiento}, sustituir $x = \eu^{t}$:
$\frac{\ln x}{x} = \frac{t}{\eu^t} \to 0$ como $t \to +\infty$ por
\cref{thm:g12:exp:growth}, y de manera similar para los demás límites con
$x = \eu^{-t}$, dando $x\ln x = -t\eu^{-t} \to 0$.
\end{proof}

\begin{method}[Resolución de ecuaciones con $\exp$ y $\ln$]\label{met:g12:exp:equations}
Ambos \omterm{def:g11:func:function}{funciones} son estrictamente \omterm{def:g12:seq:monotonic}{creciente}, por lo que se pueden aplicar (o eliminar)
de) ambos lados de una \omterm{def:g10:algebra:equation}{ecuación} o desigualdad sin cambiar su dirección:
\[
\eu^{u} = \eu^{v} \iff u = v, \qquad
\eu^{u} \leq \eu^{v} \iff u \leq v,
\]
y lo mismo para $\ln$ en cantidades positivas. \emph{Always check \omterm{def:g11:func:function}{dominios}
first} ($\ln$ requiere argumentos positivos). Para \omterm{def:g10:algebra:equation}{ecuaciones} me gusta
$a^x = b$ con $a>0$, $a \neq 1$, reescribe $a^x = \eu^{x\ln a}$ y resuelve
$x = \frac{\ln b}{\ln a}$.
\end{method}

\begin{example}\label{ex:g12:exp:halflife}
Un \omterm{def:g10:proba:sample}{muestra} radiactivo se desintegra después de $N(t) = N_0\,\eu^{-\lambda t}$. su
\emph{vida media} $T$ satisface $N(T) = N_0/2$, \emph{i.e.}
$\eu^{-\lambda T} = \frac12$, entonces $T = \frac{\ln 2}{\lambda}$: la vida media
No depende de la cantidad inicial.
\end{example}

\lim_{x\to+\infty} \frac{\eu^x - x^2}{\eu^x + 1}, \qquad
\lim_{x\to+\infty} \frac{\ln x}{\sqrt x}, \qquad

\begin{exercise}[$\star\star$]\label{exo:g12:exp:4}

\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:exp:5}
Muestre que para todos $x > -1$, $\ln(1 + x) \leq x$, con igualdad solo en
$x = 0$. Deduce eso para todo $n \geq 1$,
\[
\left(1 + \frac{1}{n}\right)^n \leq \eu \leq \left(1 - \frac{1}{n+1}\right)^{-(n+1)} .
\]
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:exp:6}
Se invierte un capital $C_0$ a una tasa anual de $3\%$, interés compuesto
cada año.
\begin{enumerate}
  \item Expresar el capital $C_n$ después de $n$ años.
  \item ¿Después de cuántos años se duplica el capital? da la respuesta exacta
        usando $\ln$, luego un valor numérico.
  \item Compare con \omterm{def:g10:numbers:approx}{aproximación} ``$70$ dividido por la tasa en
        porcentaje'' utilizado por los banqueros.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:exp:7}
Resuelve la desigualdad $\eu^{2x} - \eu^{x+1} > 0$, luego la desigualdad

\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:exp:8}
Deje $f(x) = \dfrac{\eu^x}{x}$ por $x > 0$.
\begin{enumerate}
  \item Estudie las variaciones de $f$ en $\intoo{0}{+\infty}$ y dé su
        \omterm{def:g10:functions:extrema}{mínimo}.
  \item ¿Para qué valores de $k$ el \omterm{def:g10:algebra:equation}{ecuación} $\eu^x = kx$ tiene $0$, $1$?
        o soluciones $2$ en $\intoo{0}{+\infty}$?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:exp:9}
Para $n \geq 1$, sea $u_n = \left(1 + \frac1n\right)^n$.
\begin{enumerate}
  \item Usando \cref{exo:g12:exp:5}, muestre que $(u_n)$ es \omterm{def:g12:seq:bounded}{delimitado arriba}
        por~$\eu$.
  \item Demuestre que $\ln u_n = n \ln\left(1 + \frac1n\right) \to 1$, y
        deducir que $u_n \to \eu$.
        (Pista: reconozca un cociente de diferencias de $\ln$ en $1$).
\end{enumerate}
\end{exercise}

\section{Problema: el logaritmo domina al mundo}

\begin{problem}[{Problema del fin de semana --- tiempos de duplicación y la regla
de 72, las escalas de terremotos, ácidos y pianos, y la
constante $\eu$ dejando huellas dactilares por todos lados}]
\label{pb:g12:exp:1}
Todo lo que crece con un \emph{porcentaje} fijo crece
exponencialmente --- ahorros, bacterias, epidemias --- y lo que sea
abarca demasiadas potencias de diez para comprenderlas: energías sísmicas,
acidez, intensidades de sonido --- es domesticado por un \omterm{def:g12:exp:ln}{logaritmo}. esto
El problema calcula los tiempos de duplicación y desenmascara la regla de los banqueros.
72, lee las escalas logarítmicas del mundo y recopila las
Huellas dactilares que deja el número $\eu$ en cada escena del crimen
(\cref{prop:g12:exp:funceq}, \cref{thm:g12:exp:growth},
\cref{met:g12:exp:equations}).

\medskip
\noindent\textbf{Parte I --- Fluency.}
\begin{enumerate}
  \item Resolver: $\eu^{2x} = 5$; \quad$\ln(3x - 1) = 2$; \quad
        $\eu^{2x} - 3\eu^x + 2 = 0$ (una cuadrática disfrazada).
  \item Simplificar: $\dfrac{\ln 8}{\ln 2}$; \quad
        $\ln\!\left(\eu^3 \sqrt{\eu}\right)$; \quad
        $\eu^{\ln 5 - \ln 2}$.
  \item Demostrar la desigualdad madre: $\eu^x \geq 1 + x$ para todos
        real $x$ (estudio $f(x) = \eu^x - 1 - x$), y deducir
        $\ln(1 + u) \leq u$ para $u > -1$.
  \item Computar
        $\lim_{x \to +\infty} x^2 \eu^{-x}$,
        $\lim_{x \to +\infty} \frac{\ln x}{x}$
        (\cref{thm:g12:exp:growth}), y
        $\lim_{x \to 0} \frac{\eu^x - 1}{x}$ (reconoce un
        \omterm{def:g12:deriv:derivative}{derivado}).
  \item Estudio $f(x) = x \ln x$ sobre $\intoo{0}{+\infty}$:
        variaciones, \omterm{def:g10:functions:extrema}{mínimo}, y el límite en $0^+$ (admitido:
        $x \ln x \to 0$). Este pequeño \omterm{def:g11:func:function}{función} mide
        información y entropía en los volúmenes universitarios.
\end{enumerate}

\medskip
\noindent\textbf{Parte II --- Doubling times and the rule of
72.}
\begin{enumerate}[resume]
  \item Los ahorros crecen a $3\,\%$ por año. resolver
        $1.03^n = 2$: ¿cuánto tiempo para duplicar el capital?
  \item Los banqueros estiman que el tiempo de duplicación es
        $\frac{72}{\text{rate in }\%}$. Pruebe la regla en
        $3\,\%$, $6\,\%$ y $9\,\%$ contra el exacto
        $\frac{\ln 2}{\ln(1 + r)}$, luego explícalo: para pequeños
        $r$, $\ln(1 + r) \approx r$ (la desigualdad de la pregunta 3 es
        la mitad de la historia), por lo que la constante exacta es
        $100 \ln 2 \approx 69.3$ --- ¿por qué los banqueros prefieren
        $72$?
  \item Una bacteria se divide cada $20$ minutos:
        $p(t) = 2^{t/20}$ ($t$ en minutos). Reescribirlo como
        $\eu^{\lambda t}$, luego calcule $p$ después de $24$ horas.
        La respuesta (más de $10^{21}$) demuestra ¿qué pasa con el
        modelo --- ¿y qué detiene a las verdaderas colonias?
  \item La cafeína sale del cuerpo con una vida media de aproximadamente $5$
        horas. Después de un café $100$ mg a las 15:00, ¿cuánto
        queda a las 23:00? ¿A qué hora baja?
        $10$mg? (El insomnio tiene un \omterm{def:g12:exp:ln}{logaritmo}).
  \item Un euro a $100\,\%$ interés anual: calcular el
        Capital de fin de año anual, mensual y diario.
        composición, y dar el techo que
        \cref{exo:g12:exp:9} resultó irrompible. cual famoso
        ¿Constante es el límite de la pura codicia?
  \item ¿Por qué los científicos conspiran contra $\ln(\text{cases})$?
        ¿Cuál es el momento durante la fase inicial de una epidemia? ¿Qué significa un
        línea recta en esa trama revela, y ¿qué significa su
        \omterm{def:g10:reffunc:affine}{pendiente} medida?
\end{enumerate}

\medskip
\noindent\textbf{Parte III --- The world's logarithmic scales.}
(Escribe $\log_{10} x = \frac{\ln x}{\ln 10}$.)
\begin{enumerate}[resume]
  \item Nivel sonoro en decibelios: $L = 10 \log_{10}(I/I_0)$. un
        una conversación mide $60$ dB, un concierto de rock $120$ dB:
        ¿En qué factor difieren las intensidades del sonido?
  \item Las magnitudes del terremoto aumentan en $1$ cuando la actividad sísmica
        La amplitud se multiplica por $10$ y se libera.
        escalas de energía como amplitud$^{3/2}$. Comparar
        Terremotos de magnitud-$5$ y magnitud-$7$: amplitud
        relación, luego relación de energía.
  \item Química: $\text{pH} = -\log_{10}[\mathrm{H^+}]$.
        El jugo de limón tiene pH $2$, la leche pH $6.5$: ¿cuál es el
        relación de sus concentraciones de ácido?
  \item Música: cada octava duplica la \omterm{def:g10:stats:series}{frecuencia}. Desde el

        ¿Y por qué nuestros sentidos (oído, vista, sensación de temblor)

  \item La regla de cálculo, la calculadora de ingenieros para $350$
        años: dos palos graduados para que el \emph{longitud}
        a la marca $x$ es proporcional a $\ln x$. Explica como

\medskip
\noindent\textbf{Parte IV --- $\eu$'s fingerprints.}
\begin{enumerate}[resume]
  \item De \cref{exo:g12:exp:8}: el \omterm{def:g10:algebra:equation}{ecuación}
        $\eu^x = kx$ ($x > 0$) tiene soluciones $0$, $1$ o $2$
        según la posición de $k$ relativa a un
        umbral. Vuelva a expresar el resultado --- y verifique el
        hecho geométrico detrás de esto: la línea $y = \eu x$ es
        exactamente el \omterm{def:g12:deriv:tangent}{tangente} a la exponencial a través del

  \item La constante casi fallida: con $n$ billetes de lotería de
        ganando \omterm{def:g10:proba:distribution}{probabilidad} $\frac1n$ cada uno,
        $\P(\text{no win}) = \left(1 - \frac1n\right)^n$.
        Calcule su límite mediante
        $n \ln\!\left(1 - \frac1n\right)$ (use el estándar
        límite $\frac{\ln(1 + u)}{u} \to 1$), y conciliar con
        el misterioso $0.368$ de \cref{pb:g11:binom:1}.
  \item La regla $37\,\%$: elegir lo mejor de $n$
        candidatos entrevistados en orden aleatorio (sin vuelta atrás),
        la estrategia óptima rechaza la primera
        $\frac{n}{\eu} \approx 37\,\%$ y luego toma la primera
        candidato mejor que todos hasta ahora --- teniendo éxito con
        \omterm{def:g10:proba:distribution}{probabilidad} sobre $\frac1\eu$. ¿Dónde están los dos?
        $\frac1\eu$ de las preguntas 18 y 19 provienen de ---
        indicar el mecanismo común (muchos independientes
        de pequeña probabilidad \omterm{def:g11:prob:model}{eventos}), y calcular $\frac1\eu$ a tres
        decimales.
  \item Finale --- Retrato de $\eu$: el techo de la capitalización
        (pregunta 10); la base cuyo \omterm{def:g12:deriv:tangent}{tangente} en $0$ tiene \omterm{def:g10:reffunc:affine}{pendiente}
        exactamente $1$ (preguntas 3 y 4); el crecimiento no
        capturas polinómicas (pregunta 4); la constante de
        cuasi accidentes y de parada óptima (preguntas 18-19);
        y su inverso $\ln$, que convierte productos en sumas
        (pregunta 16) y décadas en pulgadas (Parte III). uno
        frase cada uno.
\end{enumerate}
