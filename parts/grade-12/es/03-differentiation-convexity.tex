\chapter{Diferenciación y convexidad}\label{ch:g12:deriv}

El \omterm{def:g12:deriv:derivative}{derivado} mide la tasa de cambio instantánea de un \omterm{def:g11:func:function}{función}; es
el \omterm{def:g10:reffunc:affine}{pendiente} de la línea \omterm{def:g12:deriv:tangent}{tangente} a su \omterm{def:g10:functions:graph}{gráfico}. Este capítulo revisa y amplía
las reglas de diferenciación, vincula el signo de $f'$ a las variaciones de $f$,
e introduce \omterm{def:g12:deriv:convex}{convexidad}, que se rige por el segundo \omterm{def:g12:deriv:derivative}{derivado}.

\section{la derivada}

\begin{definition}[Derivada en un punto]\label{def:g12:deriv:derivative}
Dejemos que $f$ se defina en \omterm{def:g10:numbers:interval}{intervalo}, $I$ y $a \in I$. El \omterm{def:g11:func:function}{función} $f$ es
\emph{diferenciable en $a$}\index{derivado}\index{diferenciable} si el cociente de diferencias
\[
\frac{f(a+h) - f(a)}{h}
\]
tiene un límite finito como $h \to 0$. Este límite se llama \emph{derivada de
$f$ en $a$} y se escribe $f'(a)$. Si $f$ es diferenciable en cada punto de
$I$, el \omterm{def:g11:func:function}{función} $f' \colon x \mapsto f'(x)$ es el \emph{derivado} de
$f$.
\end{definition}

\begin{definition}[recta tangente]\label{def:g12:deriv:tangent}
Si $f$ es \omterm{def:g12:deriv:derivative}{diferenciable} en $a$, el \emph{tangente}\index{tangente} al
curva de $f$ en el punto $(a, f(a))$ es la recta de \omterm{def:g10:algebra:equation}{ecuación}
\[
y = f(a) + f'(a)(x - a).
\]
\end{definition}

\begin{proposition}\label{prop:g12:deriv:diffcont}
Un \omterm{def:g11:func:function}{función} \omterm{def:g12:deriv:derivative}{diferenciable} en $a$ es \omterm{def:g12:limcont:continuity}{continuo} en $a$. Lo contrario es falso.
\end{proposition}

\begin{proof}
Para $h \neq 0$ pequeño, $f(a+h) - f(a) = h \cdot \frac{f(a+h)-f(a)}{h}$. como
$h \to 0$, el lado derecho tiende a $0 \times f'(a) = 0$, por lo que
$f(a+h) \to f(a)$. Por el contrario, $x \mapsto \abs{x}$ es \omterm{def:g12:limcont:continuity}{continuo} en
$0$, pero su cociente de diferencias en $0$ es igual a $\frac{\abs h}{h} = \pm 1$
dependiendo del signo de $h$, y no tiene límite.
\end{proof}

\subsection{Differentiation rules}

\begin{proposition}[Operaciones]\label{prop:g12:deriv:operations}
Sea $u, v$ \omterm{def:g12:deriv:derivative}{diferenciable} en $I$ y $\lambda \in \R$. Entonces $u + v$,
$\lambda u$, $uv$ son \omterm{def:g12:deriv:derivative}{diferenciable} en $I$, al igual que $\frac{u}{v}$ donde
$v \neq 0$, y
\[
(u+v)' = u' + v', \quad (\lambda u)' = \lambda u', \quad
(uv)' = u'v + uv', \quad
\left(\frac{u}{v}\right)' = \frac{u'v - uv'}{v^2}.
\]
\end{proposition}

\begin{proof}[Prueba de la regla del producto]
Escribe el cociente de diferencias de $uv$ en $a$ como
\[
\frac{u(a+h)v(a+h) - u(a)v(a)}{h}
= \frac{u(a+h) - u(a)}{h}\,v(a+h) + u(a)\,\frac{v(a+h) - v(a)}{h}.
\]
Como $h \to 0$, $v(a+h) \to v(a)$ por \omterm{def:g12:limcont:continuity}{continuidad}
(\cref{prop:g12:deriv:diffcont}), por lo que el lado derecho tiende a
$u'(a)v(a) + u(a)v'(a)$. Las demás reglas se demuestran de manera similar.
\end{proof}

\begin{theorem}[regla de cadena]\label{thm:g12:deriv:chain}
\index{regla de la cadena}
Sea $u$ \omterm{def:g12:deriv:derivative}{diferenciable} en $I$ con valores en $J$ y $g$ \omterm{def:g12:deriv:derivative}{diferenciable}.
en $J$. Entonces $g \circ u$ es \omterm{def:g12:deriv:derivative}{diferenciable} en $I$ y
\[
(g \circ u)' = (g' \circ u) \cdot u' .
\]
En particular, para $u$ \omterm{def:g12:deriv:derivative}{diferenciable}:
\[
(u^n)' = n\,u'\,u^{n-1} \ (n \in \Z), \qquad
\left(\sqrt{u}\right)' = \frac{u'}{2\sqrt{u}} \ (u > 0), \qquad
\left(\eu^{u}\right)' = u'\,\eu^{u}.
\]
\end{theorem}

\begin{proof}[Boceto de prueba]
Cuando $u(a+h) \neq u(a)$ para $h$ sea pequeño, escriba
\[
\frac{g(u(a+h)) - g(u(a))}{h}
= \frac{g(u(a+h)) - g(u(a))}{u(a+h) - u(a)} \cdot \frac{u(a+h) - u(a)}{h}.
\]
Como $h \to 0$, $u(a+h) \to u(a)$, el primer factor tiende a $g'(u(a))$ y
el segundo a $u'(a)$. (Una prueba completa debe manejar el caso donde
$u(a+h) = u(a)$ para $h$ arbitrariamente pequeño; Este punto técnico es tratado en
universidad.)
\end{proof}

\begin{example}
El \omterm{def:g12:deriv:derivative}{derivados} habitual, válido sobre el \omterm{def:g11:func:function}{dominios} natural:
\[
(x^n)' = nx^{n-1}, \quad
\left(\frac{1}{x}\right)' = -\frac{1}{x^2}, \quad
(\sqrt{x})' = \frac{1}{2\sqrt{x}}, \quad
(\cos x)' = -\sin x, \quad
(\sin x)' = \cos x .
\]
El primero se prueba por inducción a partir de la regla del producto, y los dos últimos en
\cref{ch:g12:trigo}.
\end{example}

\section{Variaciones y extremos}

\begin{theorem}[Signo de la derivada y variaciones]\label{thm:g12:deriv:variations}
Sea $f$ \omterm{def:g12:deriv:derivative}{diferenciable} en un \omterm{def:g10:numbers:interval}{intervalo} $I$.
\begin{enumerate}
  \item $f$ es \omterm{def:g12:seq:monotonic}{creciente} en $I$ si y sólo si $f' \geq 0$ en $I$.
  \item $f$ es \omterm{def:g10:functions:variations}{decreciente} en $I$ si y solo si $f' \leq 0$ en $I$.
  \item Si $f' > 0$ en $I$ excepto en un número finito de puntos donde
        desaparece, entonces $f$ es estrictamente \omterm{def:g12:seq:monotonic}{creciente} en $I$.
\end{enumerate}
\end{theorem}

\begin{proof}[prueba parcial]
Si $f$ es \omterm{def:g12:seq:monotonic}{creciente}, cada cociente de diferencias
$\frac{f(a+h)-f(a)}{h}$ es $\geq 0$, y pasar al límite da
$f'(a) \geq 0$. Las implicaciones inversas se basan en el teorema del valor \omterm{def:g11:stat:mean}{significar},
probado en la universidad; son \emph{admitido a este nivel}.
\end{proof}

\begin{definition}[extremo local]\label{def:g12:deriv:extremum}
$f$ tiene un \emph{máximo local}\index{extremo} en $a$ si $f(x) \leq f(a)$ para
todo $x$ cerca de $a$; los mínimos locales se definen simétricamente.
\end{definition}

\begin{proposition}[Condición de primer orden]\label{prop:g12:deriv:critical}
Si $f$ es \omterm{def:g12:deriv:derivative}{diferenciable} en un \emph{abierto} \omterm{def:g10:numbers:interval}{intervalo} $I$ y tiene un local
\omterm{def:g12:deriv:extremum}{extremo} en $a \in I$, luego $f'(a) = 0$. Lo contrario es falso
($f(x) = x^3$ en $a = 0$).
\end{proposition}

\begin{proof}
Digamos que $a$ es un \omterm{def:g12:deriv:extremum}{máximo local}. Para $h > 0$ pequeño,
$\frac{f(a+h)-f(a)}{h} \leq 0$, por lo que dejar $h \to 0^+$ da $f'(a) \leq 0$;
para $h < 0$ el cociente es $\geq 0$, lo que da $f'(a) \geq 0$. Por lo tanto
$f'(a) = 0$.
\end{proof}

\begin{method}[Tabla de variaciones]\label{met:g12:deriv:table}
Para estudiar un \omterm{def:g11:func:function}{función} $f$: determine su \omterm{def:g11:func:function}{dominio} y sus límites en el límite;
calcular $f'$ y factorizarlo; determinar el signo de $f'$ en cada subintervalo;
registrar todo en un \omterm{def:g10:functions:table}{tabla de variaciones}, marcando extremos; deducir el número de
soluciones de $f(x) = k$ usando el teorema de biyección
(\cref{thm:g12:limcont:bijection}).
\end{method}

\section{Convexidad}

\begin{definition}[Función convexa]\label{def:g12:deriv:convex}
Un \omterm{def:g11:func:function}{función} $f$ definido en un \omterm{def:g10:numbers:interval}{intervalo} $I$ es \emph{convexo}\index{convexidad}
en $I$ si cada acorde de su \omterm{def:g10:functions:graph}{gráfico} se encuentra por encima del \omterm{def:g10:functions:graph}{gráfico}: para todos
$a, b \in I$ y $t \in \intcc{0}{1}$,
\[
f\bigl((1-t)a + t b\bigr) \leq (1-t) f(a) + t f(b).
\]
Es \emph{cóncavo}\index{cóncavo} si se cumple la desigualdad inversa ($-f$ convexa).
\end{definition}

\begin{theorem}[Convexidad y derivadas]\label{thm:g12:deriv:convexity}
Sea $f$ \omterm{def:g12:deriv:derivative}{diferenciable} en un \omterm{def:g10:numbers:interval}{intervalo} $I$. Los siguientes son equivalentes:
\begin{enumerate}
  \item $f$ es \omterm{def:g12:deriv:convex}{convexo} en $I$;
  \item $f'$ es \omterm{def:g12:seq:monotonic}{creciente} en $I$;
  \item el \omterm{def:g10:functions:graph}{gráfico} de $f$ se encuentran encima de cada uno de sus \omterm{def:g12:deriv:tangent}{tangentes}.
\end{enumerate}
Si $f$ es el doble de \omterm{def:g12:deriv:derivative}{diferenciable}, estos también equivalen a $f'' \geq 0$ en
$I$.
\end{theorem}

\begin{proof}[Prueba de (2) $\Rightarrow$ (3)]
Reparar $a \in I$ y configurar
$g(x) = f(x) - f(a) - f'(a)(x-a)$, la brecha entre la curva y el \omterm{def:g12:deriv:tangent}{tangente}
en $a$. Entonces $g$ es \omterm{def:g12:deriv:derivative}{diferenciable} y $g'(x) = f'(x) - f'(a)$. Si $f'$ es
\omterm{def:g12:seq:monotonic}{creciente}, $g' \leq 0$ en $I \cap \intoc{-\infty}{a}$ y $g' \geq 0$ en
$I \cap \intco{a}{+\infty}$: $g$ disminuye antes de $a$ y aumenta después,
entonces $g$ alcanza su \omterm{def:g10:functions:extrema}{mínimo} $g(a) = 0$, por lo tanto $g \geq 0$.
Las implicaciones restantes se admiten a este nivel; la equivalencia con
$f'' \geq 0$ se deriva de \cref{thm:g12:deriv:variations} aplicado a $f'$.
\end{proof}

\begin{definition}[Punto de inflexión]\label{def:g12:deriv:inflection}
Un punto donde la curva de $f$ cruza su \omterm{def:g12:deriv:tangent}{tangente} es un \emph{inflexión
punto}\index{punto de inflexión}. Para $f$ dos veces \omterm{def:g12:deriv:derivative}{diferenciable}, inflexión
Los puntos son los puntos donde $f''$ cambia de signo.
\end{definition}

\begin{example}\label{ex:g12:deriv:cube}
$f(x) = x^3$ tiene $f''(x) = 6x$: $f$ es \omterm{def:g12:deriv:convex}{cóncavo} en $\intoc{-\infty}{0}$,
\omterm{def:g12:deriv:convex}{convexo} en $\intco{0}{+\infty}$, con un \omterm{def:g12:deriv:inflection}{punto de inflexión} en el origen,
donde la curva cruza su \omterm{def:g12:deriv:tangent}{tangente} (el eje $x$).
\end{example}

\begin{example}[Desigualdades de convexidad]\label{ex:g12:deriv:ineq}
El exponencial es \omterm{def:g12:deriv:convex}{convexo} en $\R$ (su segundo \omterm{def:g12:deriv:derivative}{derivado} es en sí mismo, positivo).
Su \omterm{def:g12:deriv:tangent}{tangente} en $0$ es $y = 1 + x$, por lo que
\[
\eu^x \geq 1 + x \quad \text{for all } x \in \R .
\]
Estos \emph{\omterm{def:g12:deriv:tangent}{tangente} line inequalities} son un producto estándar de \omterm{def:g12:deriv:convex}{convexidad}.
\end{example}

\begin{method}[convexidad]\label{met:g12:deriv:convexity}
\begin{itemize}
  \item Para demostrar una desigualdad $f(x) \geq ax + b$: identifica la recta como
        \omterm{def:g12:deriv:tangent}{tangente} de un \omterm{def:g12:deriv:convex}{convexo} $f$ e invocar
        \cref{thm:g12:deriv:convexity}(3).
  \item Para localizar \omterm{def:g12:deriv:inflection}{puntos de inflexión}: resolver $f''(x) = 0$ \emph{y} comprobar
        que $f''$ cambia de signo allí.
  \item En un \omterm{def:g10:functions:table}{tabla de variaciones}, \omterm{def:g12:deriv:convex}{convexidad} refina el boceto de la curva.
        (en qué dirección se dobla).
\end{itemize}
\end{method}
