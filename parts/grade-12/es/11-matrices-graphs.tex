\chapter{Matrices y gráficas}\label{ch:g12:matrix}

Un \omterm{def:g12:matrix:matrix}{matriz} es una matriz rectangular de números, sumados y multiplicados por reglas
diseñado para que el álgebra \omterm{def:g12:matrix:matrix}{matriz} represente la composición de
transformaciones. Matrices resuelven \omterm{def:g10:lines:system}{sistemas lineales}, unidad acoplada recurrente
\omterm{def:g12:seq:sequence}{secuencias} y contar paseos en redes: las matemáticas detrás de la búsqueda
motores y algoritmos de camino más corto.

\section{Álgebra matricial}

\begin{definition}[Matriz]\label{def:g12:matrix:matrix}
Un \emph{$m \times n$ matrix}\index{matriz} es una tabla de \omterm{def:g10:numbers:sets}{números reales} con
$m$ filas y $n$ columnas:
$A = (a_{ij})$, donde $a_{ij}$ es la entrada en la fila $i$, columna $j$.
Se suman dos matrices del mismo tamaño entrada por entrada, y
$\lambda A = (\lambda a_{ij})$.
\end{definition}

\begin{definition}[Producto matriz]\label{def:g12:matrix:product}
Sea $A$ $m \times n$ y $B$ sea $n \times p$. El producto $AB$ es el
$m \times p$ \omterm{def:g12:matrix:matrix}{matriz} cuya entrada $(i,j)$ es
\[
(AB)_{ij} = \sum_{k=1}^{n} a_{ik} b_{kj}
\]
(la regla ``fila $i$ de $A$ multiplicada por la columna $j$ de $B$'').
\end{definition}

\begin{example}
$\begin{pmatrix} 1 & 2\\ 3 & 4\end{pmatrix}
\begin{pmatrix} 0 & 1\\ 1 & 1\end{pmatrix}
= \begin{pmatrix} 2 & 3\\ 4 & 7\end{pmatrix}$,
\quad mientras \quad
$\begin{pmatrix} 0 & 1\\ 1 & 1\end{pmatrix}
\begin{pmatrix} 1 & 2\\ 3 & 4\end{pmatrix}
= \begin{pmatrix} 3 & 4\\ 4 & 6\end{pmatrix}$:
\emph{\omterm{def:g12:matrix:matrix}{matriz} multiplication is not commutative}.
\end{example}

\begin{proposition}[Reglas del álgebra matricial]\label{prop:g12:matrix:rules}
Siempre que los tamaños hagan que los productos tengan significado:
\[
(AB)C = A(BC), \qquad A(B + C) = AB + AC, \qquad (A+B)C = AC + BC,
\]
y el \emph{identity \omterm{def:g12:matrix:matrix}{matriz}} $I_n$ (unos en la diagonal, ceros en el resto)
satisface $I_m A = A I_n = A$ para $A$ de tamaño $m \times n$.
\end{proposition}

\begin{proof}
Todas son verificaciones entrada por entrada de
\cref{def:g12:matrix:product}; asociatividad, la única no trivial,
equivale a intercambiar dos sumas finitas:
$\bigl((AB)C\bigr)_{ij} = \sum_l \left(\sum_k a_{ik}b_{kl}\right) c_{lj}
= \sum_k a_{ik} \left(\sum_l b_{kl} c_{lj}\right)
= \bigl(A(BC)\bigr)_{ij}$.
\end{proof}

\begin{definition}[inversa]\label{def:g12:matrix:inverse}
Un cuadrado \omterm{def:g12:matrix:matrix}{matriz} $A$ de tamaño $n$ es \emph{reversible}\index{matriz!inverso}
si hay un \omterm{def:g12:matrix:matrix}{matriz} $B$ con $AB = BA = I_n$; $B$ es entonces único, escrito
$A^{-1}$.
\end{definition}

\begin{proposition}[Inversa de una matriz $2\times2$]\label{prop:g12:matrix:inverse2x2}
Deje $A = \begin{pmatrix} a & b\\ c & d\end{pmatrix}$ y
$\det A = ad - bc$ (el \emph{determinante}\index{determinante}). Entonces $A$ es
\omterm{def:g12:matrix:inverse}{reversible} si y sólo si $\det A \neq 0$, en cuyo caso
\[
A^{-1} = \frac{1}{ad - bc}\begin{pmatrix} d & -b\\ -c & a\end{pmatrix}.
\]
\end{proposition}

\begin{proof}
Un cálculo da
$A \begin{pmatrix} d & -b\\ -c & a\end{pmatrix}
= \begin{pmatrix} d & -b\\ -c & a\end{pmatrix} A = (ad - bc) I_2$; si
$ad - bc \neq 0$, dividir. Por el contrario, si $ad - bc = 0$, las columnas de $A$
son proporcionales, al igual que las columnas de $AB$ para cualquier $B$; pero el
Las columnas de $I_2$ no son proporcionales, por lo que ninguna $B$ puede satisfacer $AB = I_2$.
\end{proof}

\begin{method}[sistemas lineales]\label{met:g12:matrix:systems}
el sistema
$\begin{cases} ax + by = e\\ cx + dy = f \end{cases}$
es el \omterm{def:g12:matrix:matrix}{matriz} \omterm{def:g10:algebra:equation}{ecuación} $AX = Y$ con
$X = \begin{pmatrix} x \\ y\end{pmatrix}$,
$Y = \begin{pmatrix} e \\ f\end{pmatrix}$. Si $\det A \neq 0$, es único
la solución es $X = A^{-1}Y$. El mismo formalismo maneja $n$ \omterm{def:g10:algebra:equation}{ecuaciones} en $n$
incógnitas.
\end{method}

\section{Poderes matriciales y secuencias recurrentes.}

\begin{definition}\label{def:g12:matrix:power}
Para un cuadrado \omterm{def:g12:matrix:matrix}{matriz} $A$ y $k \in \N$, $A^k = A \times \dots \times A$
(factores $k$), con $A^0 = I$.
\end{definition}

\begin{method}[Casos diagonal más nilpotente y diagonalizable]\label{met:g12:matrix:powers}
Dos formas estándar de calcular $A^k$:
\begin{itemize}
  \item Si $A = \lambda I + N$ donde $N^2 = 0$, el teorema del binomio
        (válido aquí desde que $I$ y $N$ conmutan) se colapsa en dos términos:
        $A^k = \lambda^k I + k \lambda^{k-1} N$.
  \item Si se encuentra un \omterm{def:g12:matrix:inverse}{reversible} $P$ con $A = PDP^{-1}$ y $D$ en diagonal,
        luego $A^k = P D^k P^{-1}$ y $D^k$ se calcula entrada por entrada.
        (Encontrar tal $P$ sistemáticamente es la teoría de
        \emph{diagonalización}, desarrollado en la universidad; en este nivel $P$ es
        dado.)
\end{itemize}
\end{method}

\begin{example}[secuencias acopladas]\label{ex:g12:matrix:coupled}
Sean $u_{n+1} = 3u_n + v_n$ y $v_{n+1} = u_n + 3v_n$. Configuración
$X_n = \begin{pmatrix} u_n\\ v_n \end{pmatrix}$ y
$A = \begin{pmatrix} 3 & 1\\ 1 & 3\end{pmatrix}$, obtenemos
$X_{n+1} = AX_n$, entonces $X_n = A^n X_0$. El auxiliar \omterm{def:g12:seq:sequence}{secuencias}
$s_n = u_n + v_n$ y $d_n = u_n - v_n$ satisfacen $s_{n+1} = 4s_n$ y
$d_{n+1} = 2d_n$, entonces $s_n = 4^n s_0$, $d_n = 2^n d_0$ y
\[
u_n = \frac{4^n(u_0+v_0) + 2^n(u_0-v_0)}{2}, \qquad
v_n = \frac{4^n(u_0+v_0) - 2^n(u_0-v_0)}{2}.
\]
(Detrás de escena: $(1,1)$ y $(1,-1)$ son direcciones de vectores propios de $A$).
\end{example}

\section{Graficas y paseos}

\begin{definition}[Gráfico, matriz de adyacencia]\label{def:g12:matrix:graph}
Un \emph{gráfico}\index{gráfico} consta de vértices $1, 2, \dots, n$ y aristas
unir ciertos pares de vértices (pares ordenados para un \emph{dirigido}
gráfico). Su \emph{matriz de adyacencia}\index{matriz de adyacencia} es el
$n \times n$ \omterm{def:g12:matrix:matrix}{matriz} $M$ con $m_{ij} = 1$ si hay un borde de $i$ a
$j$ y $0$ en caso contrario. Un \emph{caminar} de longitud $k$ de $i$ a $j$ es un
\omterm{def:g12:seq:sequence}{secuencia} de $k$ bordes consecutivos que van desde $i$ a $j$.
\end{definition}

\begin{theorem}[Contando paseos]\label{thm:g12:matrix:walks}
El número de recorridos de longitud $k$ desde el vértice $i$ hasta el vértice $j$ es el
$(i,j)$ entrada de $M^k$.
\end{theorem}

\begin{proof}
Inducción en $k$. Para $k = 1$ esta es la definición de $M$. asumir el
reclamación para $k$. Un paseo de longitud $k+1$ desde $i$ hasta $j$ es un paseo de longitud
$k$ desde $i$ hasta algún vértice $l$, seguido de una arista desde $l$ hasta $j$; por el
principios de suma y multiplicación, su número es
\[
\sum_{l=1}^{n} \bigl(M^k\bigr)_{il}\, m_{lj} = \bigl(M^{k+1}\bigr)_{ij}.
\qedhere
\]
\end{proof}

\begin{example}
Para el triángulo \omterm{def:g12:matrix:graph}{gráfico} (vértices $3$, todos los pares unidos),
$M = \begin{pmatrix} 0&1&1\\ 1&0&1\\ 1&1&0\end{pmatrix}$ y
$M^2 = \begin{pmatrix} 2&1&1\\ 1&2&1\\ 1&1&2\end{pmatrix}$: de cada vértice
hay $2$ recorridos de longitud $2$ de regreso a sí mismo (a través de cualquier vecino) y
$1$ entre sí.
\end{example}

\section{Problema: La matriz que conoce a Fibonacci (y la
clima)}

\begin{problem}[{Problema de fin de semana --- una matriz $2 \times 2$
lleva todo Fibonacci, una matriz de Markov pronóstica a largo plazo
clima, y un vector propio vale mil millones de dólares}]
\label{pb:g12:matrix:1}
Un \omterm{def:g12:matrix:matrix}{matriz} es una máquina que come un estado y devuelve el siguiente
--- y su \emph{potestades} poseen, por tanto, futuros completos. esto
El problema comienza con el asombroso \omterm{def:g12:matrix:matrix}{matriz} cuyos poderes enumeran los
Números de Fibonacci (y probar sus identidades en una línea cada uno),
luego ejecuta el clima como una cadena de Markov hasta su estado estacionario, y
cierra con el vector propio sobre el que se construyó un motor de búsqueda
(\cref{thm:g12:matrix:walks},
\cref{met:g12:matrix:powers}).

\medskip
\noindent\textbf{Parte I --- Fluency.}
\begin{enumerate}
  \item Con $A = \begin{pmatrix} 1 & 2\\ 3 & 4\end{pmatrix}$
        y $B = \begin{pmatrix} 0 & 1\\ 1 & 0\end{pmatrix}$:
        calcular $AB$ y $BA$. ¿Veredicto sobre la conmutatividad?
  \item Invertir $\begin{pmatrix} 2 & 1\\ 5 & 3\end{pmatrix}$
        (\cref{prop:g12:matrix:inverse2x2}) y utilizar el inverso
        para resolver $2x + y = 4$, $5x + 3y = 7$.
  \item Vamos a $N = \begin{pmatrix} 0 & 1\\ 0 & 0\end{pmatrix}$:
        calcular $N^2$ y deducir
        $(I + N)^n = I + nN$ por cada $n$.
  \item El triángulo \omterm{def:g12:matrix:graph}{gráfico} (tres vértices, todos los pares unidos):
        escriba su \omterm{def:g12:matrix:graph}{matriz de adyacencia} $A$, calcule $A^3$ y
        interpretar las entradas diagonales
        (\cref{thm:g12:matrix:walks}).
  \item Para $D = \begin{pmatrix} 2 & 0\\ 0 & \frac12
        \end{pmatrix}$: proporcione $D^n$ y su comportamiento como
        $n \to \infty$.
\end{enumerate}

\medskip
\noindent\textbf{Parte II --- The Fibonacci \omterm{def:g12:matrix:matrix}{matriz}.}
Deja $F = \begin{pmatrix} 1 & 1\\ 1 & 0\end{pmatrix}$ y deja
$F_1 = F_2 = 1, F_3 = 2, \dots$ sean los números de Fibonacci de
\cref{pb:g11:seq:1}.
\begin{enumerate}[resume]
  \item Calcula $F^2$, $F^3$, $F^4$ y conjetura el general
        forma de $F^n$ en términos de números de Fibonacci.
  \item Demuestra la conjetura
        $F^n = \begin{pmatrix} F_{n+1} & F_n\\ F_n & F_{n-1}
        \end{pmatrix}$ por inducción.
  \item Tome \omterm{def:g11:vect:det}{determinantes} de ambos lados (el \omterm{def:g11:vect:det}{determinante} de un
        El producto es el producto del \omterm{def:g11:vect:det}{determinantes} --- compruébalo.
        en matrices $2 \times 2$ si nunca lo has visto):
        deducir \emph{La identidad de Cassini}
        $F_{n+1}F_{n-1} - F_n^2 = (-1)^n$ --- el
        Motor de cuadrado fugaz, probado en una línea.
  \item De $F^{m+n} = F^m F^n$, lea la parte superior derecha
        entradas y derivar el \emph{fórmula de suma}
        \[
        F_{m+n} = F_{m+1} F_n + F_m F_{n-1} .
        \]
        Compruébelo para $m = n = 3$.
  \item Deducir de la fórmula de la suma (inducción en $k$)
        que $F_n$ \omterm{def:g12:arith:divides}{divide} $F_{kn}$, y verifique en
        $F_3 \mid F_6$ y $F_3 \mid F_9$.
  \item Para calcular $F_{100}$, no es necesario multiplicar $100$
        matrices: cuadrado repetidamente ($F^2, F^4, F^8, \dots$)
        y combinar. ¿Cuántas multiplicaciones \omterm{def:g12:matrix:matrix}{matriz} son suficientes?
        y qué antiguo truco de multiplicación del Medio
        ¿El volumen escolar es este, promovido a matrices?
\end{enumerate}

$\binom{p_{\text{sun}}}{p_{\text{rain}}}$ y la evolución por
\[

\]

  \item Hoy está soleado. Calcule el pronóstico para mañana y

  \item PageRank en miniatura: tres páginas, con enlaces

  \item Interprete el ranking: ¿por qué $C$ obtiene una puntuación tan alta como

\medskip
\noindent\textbf{Parte IV --- Diagonal dividends.}
\begin{enumerate}[resume]
  \item Dos cantidades acopladas obedecen
        $u_{n+1} = 3u_n + v_n$, $v_{n+1} = u_n + 3v_n$, es decir\

  \item Finale --- las tres caras del \omterm{def:g12:matrix:matrix}{matriz} este fin de semana:
