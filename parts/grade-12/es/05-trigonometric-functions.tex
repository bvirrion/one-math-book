\chapter{Funciones trigonométricas}\label{ch:g12:trigo}

El \omterm{def:g11:func:function}{funciones} $\cos$ y $\sin$ convierten la geometría del círculo en
análisis. Definidos enrollando la línea real alrededor del círculo unitario, son
el arquetipo del periódico \omterm{def:g11:func:function}{funciones}, y su \omterm{def:g12:deriv:derivative}{derivados} los hacen
Soluciones del \omterm{def:g10:algebra:equation}{ecuación} de oscilaciones $y'' = -y$.

\section{El círculo unitario}

\begin{definition}[Coseño y seno]\label{def:g12:trigo:cossin}
Sea $\mathcal{C}$ el círculo de radio 1 con centro en el origen de un
\omterm{def:g10:coordgeom:system}{ortonormal}\omterm{def:g10:coordgeom:system}{sistema de coordenadas}. A cada $t$ real asociar el punto $M(t)$
obtenido enrollando una longitud $\abs{t}$ a lo largo de $\mathcal{C}$ desde el punto
$I(1,0)$, en sentido antihorario si es $t \geq 0$ y en sentido horario en caso contrario. entonces
\emph{$\cos t$ and $\sin t$ are the coordinates of
$M(t)$}\index{coseno}\index{seno}:
\[
M(t) = (\cos t, \sin t).
\]
\end{definition}

Dado que el círculo tiene circunferencia $2\pi$, el punto $M(t)$ no cambia cuando
$t$ aumenta en $2\pi$; y dado que $M(t)$ se encuentra en el círculo unitario:

\begin{proposition}[Identidades fundamentales]\label{prop:g12:trigo:identities}
Para todos los $t \in \R$ y $k \in \Z$:
\[
\cos^2 t + \sin^2 t = 1, \qquad
\cos(t + 2k\pi) = \cos t, \qquad
\sin(t + 2k\pi) = \sin t,
\]
\[
\cos(-t) = \cos t, \qquad \sin(-t) = -\sin t,
\]
\[
\cos(\pi - t) = -\cos t, \quad \sin(\pi - t) = \sin t, \quad
\cos\left(\tfrac{\pi}{2} - t\right) = \sin t, \quad
\sin\left(\tfrac{\pi}{2} - t\right) = \cos t .
\]
\end{proposition}

\begin{proof}
La primera identidad es \omterm{def:g10:algebra:equation}{ecuación} del círculo unitario; los demás expresan el
simetrías del círculo: $M(-t)$ es el reflejo de $M(t)$ en el
$x$-eje, $M(\pi - t)$ en el eje $y$ y $M(\frac\pi2 - t)$ en la línea
$y = x$.
\end{proof}

Los valores a saber de memoria:
\begin{center}
\begin{tabular}{c|ccccc}
$t$ & $0$ & $\dfrac{\pi}{6}$ & $\dfrac{\pi}{4}$ & $\dfrac{\pi}{3}$ & $\dfrac{\pi}{2}$\\[6pt]
\hline\rule{0pt}{14pt}%
$\cos t$ & $1$ & $\dfrac{\sqrt3}{2}$ & $\dfrac{\sqrt2}{2}$ & $\dfrac12$ & $0$\\[6pt]
$\sin t$ & $0$ & $\dfrac12$ & $\dfrac{\sqrt2}{2}$ & $\dfrac{\sqrt3}{2}$ & $1$
\end{tabular}
\end{center}

\begin{proposition}[Fórmulas de suma]\label{prop:g12:trigo:addition}
Para todos los $a, b \in \R$:
\begin{align*}
\cos(a+b) &= \cos a \cos b - \sin a \sin b, &
\sin(a+b) &= \sin a \cos b + \cos a \sin b,\\
\cos(a-b) &= \cos a \cos b + \sin a \sin b, &
\sin(a-b) &= \sin a \cos b - \cos a \sin b.
\end{align*}
En particular $\cos 2a = 2\cos^2 a - 1 = 1 - 2\sin^2 a$ y
$\sin 2a = 2 \sin a \cos a$.
\end{proposition}

\begin{proof}
El \omterm{def:g11:vect:vector}{vectores} $\vect{u} = (\cos a, \sin a)$ y $\vect{v} = (\cos b, \sin b)$
son la unidad \omterm{def:g11:vect:vector}{vectores} formando un ángulo $a - b$; su \omterm{def:g11:scal:dot}{producto escalar} es igual a ambos
$\cos a\cos b + \sin a \sin b$ (\omterm{def:g10:coordgeom:system}{coordenadas}) y
$\norm{\vect u}\,\norm{\vect v}\cos(a-b) = \cos(a - b)$ (geometría), que es
la tercera fórmula. Reemplazar $b$ por $-b$ da el primero; las fórmulas \omterm{def:g12:trigo:cossin}{seno}
siga usando $\sin x = \cos(\frac\pi2 - x)$. Las fórmulas de duplicación son las
caso $b = a$ combinado con $\cos^2 + \sin^2 = 1$.
\end{proof}

\section{estudio analítico}

\begin{lemma}[límite fundamental]\label{lem:g12:trigo:sinxx}
\[
\lim_{t \to 0} \frac{\sin t}{t} = 1
\qquad\text{and}\qquad
\lim_{t \to 0} \frac{\cos t - 1}{t} = 0 .
\]
\end{lemma}

\begin{proof}
Para $0 < t < \frac{\pi}{2}$, compare tres áreas en el círculo unitario: la
el triángulo $OIM(t)$, el sector circular $OIM(t)$ y el triángulo rectángulo con
base $OI$ y altura $\tan t$:
\[
\frac{\sin t}{2} \leq \frac{t}{2} \leq \frac{\tan t}{2}.
\]

{\small Las tres áreas anidadas: el triángulo $OIM$ (azul, área
$\frac{\sin t}{2}$), el sector circular (delimitado por el arco rojo, área

La primera desigualdad da $\frac{\sin t}{t} \leq 1$; el segundo da
$\cos t \leq \frac{\sin t}{t}$. Como $t \to 0^+$, $\cos t \to 1$ (\omterm{def:g12:limcont:continuity}{continuidad}),
y el teorema de compresión produce $\frac{\sin t}{t} \to 1$; la paridad extiende esto
a $t \to 0$. Para el segundo límite,
\[
\frac{\cos t - 1}{t} = \frac{\cos^2 t - 1}{t(\cos t + 1)}
= -\frac{\sin t}{t}\cdot\frac{\sin t}{\cos t + 1}
\xrightarrow[t\to0]{} -1 \cdot \frac{0}{2} = 0. \qedhere
\]
\end{proof}

\begin{theorem}[Derivadas de seno y coseno]\label{thm:g12:trigo:derivatives}
Los \omterm{def:g11:func:function}{funciones} $\sin$ y $\cos$ son \omterm{def:g12:deriv:derivative}{diferenciable} en $\R$, con
\[
(\sin)' = \cos, \qquad (\cos)' = -\sin .
\]
De manera más general, $(\sin(ax+b))' = a\cos(ax+b)$ y
$(\cos(ax+b))' = -a\sin(ax+b)$.
\end{theorem}

\begin{proof}
Por las fórmulas de suma,
\[
\frac{\sin(x+h) - \sin x}{h}
= \sin x\,\frac{\cos h - 1}{h} + \cos x\,\frac{\sin h}{h}
\xrightarrow[h\to0]{} \sin x \cdot 0 + \cos x \cdot 1 = \cos x,
\]
utilizando \cref{lem:g12:trigo:sinxx}. El cálculo para $\cos$ es idéntico,
y la forma general se deriva de la regla de la cadena.
\end{proof}

\begin{proposition}[variaciones]\label{prop:g12:trigo:variations}
En $\intcc{0}{\pi}$, $\cos$ disminuye de $1$ a $-1$. encendido
$\intcc{-\frac\pi2}{\frac\pi2}$, $\sin$ aumenta de $-1$ a $1$. ambos
\omterm{def:g11:func:function}{funciones} son periódicos $2\pi$ y tienen rangos $\intcc{-1}{1}$; $\cos$ es
par, $\sin$ es impar.
\end{proposition}

\begin{proof}
En $\intoo{0}{\pi}$, $(\cos)' = -\sin < 0$ desde el punto $M(t)$ tiene
positivo \omterm{def:g10:coordgeom:system}{ordenada} allí; de manera similar $(\sin)' = \cos > 0$ en
$\intoo{-\frac\pi2}{\frac\pi2}$. La periodicidad y la paridad provienen de
\cref{prop:g12:trigo:identities}.
\end{proof}

\begin{method}[Resolviendo $\cos x = a$ y $\sin x = a$]\label{met:g12:trigo:equations}
Para $a \in \intcc{-1}{1}$, busque una solución particular $\alpha$ (de la
tabla de valores o una calculadora). Entonces todas las soluciones son:
\[
\cos x = \cos\alpha \iff x = \alpha + 2k\pi \text{ or } x = -\alpha + 2k\pi,
\quad k \in \Z;
\]
\[
\sin x = \sin\alpha \iff x = \alpha + 2k\pi \text{ or } x = \pi - \alpha + 2k\pi,
\quad k \in \Z.
\]
Para desigualdades, ubique los arcos solución en el círculo unitario y lea
el \omterm{def:g10:numbers:interval}{intervalos} dentro de un período.
\end{method}

\begin{example}\label{ex:g12:trigo:equation}
Resolver $\cos x = \frac12$ en $\intoc{-\pi}{\pi}$: una solución particular es
$\frac\pi3$, por lo que las soluciones son $x = \frac\pi3$ y $x = -\frac\pi3$.
En todo $\R$: $x = \pm\frac\pi3 + 2k\pi$, $k \in \Z$.
\end{example}

\section{Ceremonias}

\begin{exercise}[$\star$]\label{exo:g12:trigo:1}
Calcule $\cos\frac{2\pi}{3}$, $\sin\frac{5\pi}{6}$, $\cos\frac{7\pi}{4}$ y
$\sin\left(-\frac{\pi}{3}\right)$ usando las identidades de
\cref{prop:g12:trigo:identities}.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:trigo:2}
Diferenciar $f(x) = \sin^2 x$, $g(x) = \cos(3x) + x\sin x$ y
$h(x) = \dfrac{\sin x}{2 + \cos x}$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:trigo:3}
Resuelva en $\intoc{-\pi}{\pi}$, luego en $\R$:
\[
\text{(a) } \sin x = \frac{\sqrt3}{2};
\qquad
\text{(b) } \cos\left(2x\right) = \frac{\sqrt2}{2} .
\]
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:trigo:4}
Usando las fórmulas de suma, calcule el valor exacto de
$\cos\frac{\pi}{12}$ y $\sin\frac{\pi}{12}$.
(Pista: $\frac{\pi}{12} = \frac{\pi}{3} - \frac{\pi}{4}$.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:trigo:5}
Resuelva en $\intcc{0}{2\pi}$ la desigualdad $2\sin^2 x - \sin x - 1 \geq 0$.
(Pista: factoriza la cuadrática en $\sin x$).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:trigo:6}
Calcular los límites
\[
\lim_{x\to0} \frac{\sin 3x}{x}, \qquad
\lim_{x\to0} \frac{1 - \cos x}{x^2}, \qquad
\lim_{x\to+\infty} x \sin\frac{1}{x}.
\]
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:trigo:7}
Estudie el \omterm{def:g11:func:function}{función} $f(x) = x + \cos x$ sobre $\intcc{0}{2\pi}$: variaciones y
extremos. Muestre que $f$ es \omterm{def:g12:seq:monotonic}{creciente} en $\R$ aunque $f'$ desaparece en
infinitos puntos.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:trigo:8}
Deje $f(x) = A\cos x + B \sin x$ con $(A,B) \neq (0,0)$.
\begin{enumerate}
  \item Demuestre que $f$ se puede escribir $f(x) = R\cos(x - \varphi)$ con
        $R = \sqrt{A^2 + B^2}$ y un $\varphi$ adecuado.
  \item Deduce el \omterm{def:g10:functions:extrema}{máximo} y el \omterm{def:g10:functions:extrema}{mínimo} de $f$ y resuelve
        $\cos x + \sin x = 1$ en $\intoc{-\pi}{\pi}$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:trigo:9}
Demuestre que $\sin$ y $\cos$ satisfacen el diferencial \omterm{def:g10:algebra:equation}{ecuación}.
$y'' = -y$. Por el contrario, sea $y$ una solución de $y'' = -y$ con
$y(0) = a$ y $y'(0) = b$; muestra que $y = a\cos + b\sin$.
(Pista: considere $g = (y - a\cos - b\sin)$ y la ``energía''
$E = g^2 + (g')^2$.)
\end{exercise}

\section{Problema: el secreto del péndulo}

\begin{problem}[{Problema de fin de semana --- un límite geométrico de potencias
las derivadas del seno y el coseno, y de ellas todas
oscilación: resorts, péndulos, latidos y el metro que casi
época}]
\label{pb:g12:trigo:1}
Cada reloj que alguna vez hizo tictac, cada cuerda que alguna vez sonó,
obedece a las mismas matemáticas: una fuerza restauradora proporcional a
desplazamiento, de ahí el \omterm{def:g10:algebra:equation}{ecuación} $y'' = -\omega^2 y$, de ahí el
\omterm{def:g12:trigo:cossin}{cosenos}. En la base de todo esto se encuentra un límite que parece inocente,
$\frac{\sin x}{x} \to 1$ --- prometido en grado 9, usado por
\cref{thm:g12:trigo:derivatives}, y demostrado aquí con una foto.
Este problema sube desde ese límite hasta el balanceo de un metro
péndulo, y explica por qué el metro es, por un pelo, no
definido por él.

\medskip
\noindent\textbf{Parte I --- The fundamental limit.}
\begin{enumerate}
  \item Calentar numéricamente: calcular $\frac{\sin x}{x}$ para
        $x = 0.1$ y $x = 0.01$ (¡\omterm{def:g11:trigo:radian}{radianes}!).
  \item El sándwich, en el círculo unitario con
        $0 < x < \frac\pi2$: expresar por geometría elemental el
        áreas de (i) el triángulo con vértices $O$, $A(1,0)$
        y el punto del círculo $M(\cos x, \sin x)$; (ii) el
        sector circular $OAM$; (iii) el triángulo rectángulo $OAT$
        donde $T(1, \tan x)$.
  \item De la zona sándwich deducir
        $\sin x < x < \tan x$, entonces
        $\cos x < \frac{\sin x}{x} < 1$, y concluir con el
        teorema de compresión que
        $\lim_{x \to 0} \frac{\sin x}{x} = 1$ (controles de paridad
        $x < 0$).
  \item Deducir
        $\lim_{x \to 0} \frac{1 - \cos x}{x^2} = \frac12$
        (multiplicar por el conjugado).
  \item Explique por qué este límite es el motor de
        \cref{thm:g12:trigo:derivatives} (¿qué es?
        $\sin'(0)$?), y por qué también salda una deuda antigua: en
        ¿Qué sentido tiene \emph{\omterm{def:g11:trigo:radian}{radianes}} la única unidad de ángulo para
        ¿Qué $\sin' = \cos$ se cumple sin una constante?
\end{enumerate}

\medskip
\noindent\textbf{Parte II --- Harmonic motion.}
\begin{enumerate}[resume]
  \item Verifique mediante la regla de la cadena que
        $y(t) = R\cos(\omega t - \varphi)$ satisface
        $y'' = -\omega^2 y$.
  \item Una masa sobre un resorte obedece a $y'' = -4y$, con
        $y(0) = 3$ y $y'(0) = 8$. Admitir (de
        \cref{exo:g12:trigo:9}) que todas las soluciones son
        $A\cos 2t + B\sin 2t$, determine $A$ y $B$.
  \item Reescribe la solución como $R\cos(2t - \varphi)$
        (\cref{exo:g12:trigo:8}): dé la amplitud $R$, la
        periodo, y la fase $\varphi$ (tres decimales).
  \item ¿En qué momento la masa pasa por primera vez por el
        posición de equilibrio $y = 0$?
  \item El péndulo: una pesa sobre un hilo de longitud $L$ obedece
        \emph{exactamente} $\theta'' = -\frac gL \sin\theta$ ---
        irresoluble por elemental \omterm{def:g11:func:function}{funciones}. Para columpios pequeños,
        reemplace $\sin\theta$ por $\theta$ (¡La Parte I lo justifica!)
        y deducir el periodo
        $T = 2\pi\sqrt{\frac Lg}$. Calcular $T$ para
        $L = 1$ m ($g = 9.81$): El famoso descubrimiento de Galileo ---
        ¿De qué depende la fórmula \emph{no}?
  \item El péndulo de segundos: qué longitud da
        $T = 2$ es exactamente? La Academia Francesa de 1791 vaciló
        entre esta longitud y la diezmillonésima parte del
        cuarto de meridiano para definir el \emph{metro} --- por
        ¿En cuántos milímetros se diferencian los dos candidatos?
\end{enumerate}

\medskip
\noindent\textbf{Parte III --- Addition formulas at work.}
\begin{enumerate}[resume]
  \item Mostrar con las fórmulas de suma.
        (\cref{prop:g12:trigo:addition}) que
        $\cos\left(x + \frac\pi2\right) = -\sin x$ y
        $\sin\left(x + \frac\pi2\right) = \cos x$: circular
        la diferenciación es un cuarto de vuelta --- comprobar la coherencia
        con \cref{thm:g12:trigo:derivatives}.
  \item Derivar la identidad del producto a la suma
        $\cos a \cos b = \frac12\left(\cos(a - b) +
        \cos(a + b)\right)$ y la linealización
        $\cos^2 x = \frac{1 + \cos 2x}{2}$ (la integración
        El capítulo te lo agradecerá).
  \item Latidos: derivar
        $\sin p + \sin q = 2 \sin\frac{p + q}{2}
        \cos\frac{p - q}{2}$ y aplíquelo a dos diapasones.
        en $440$ y $444$ Hz: qué tono escucha el oído, y
        cuantas pulsaciones por segundo? (Así es como las orquestas
        melodía.)
  \item Deducir la identidad del triple ángulo
        $\cos 3x = 4\cos^3 x - 3\cos x$
        (escribir $3x = 2x + x$). Configurando $x = 20^\circ$, ¿qué
        \omterm{def:g10:algebra:equation}{ecuación} cúbico debe satisfacer $\cos 20^\circ$? (Que esto
        cúbico no se puede resolver sólo con raíces cuadradas es
        precisamente por qué la trisección del ángulo derrotó
        regla y compás durante dos milenios --- la historia completa
        se cuenta en los volúmenes de la universidad.)
  \item Una corriente alterna lee
        $I(t) = 3\cos(100\pi t) + 3\sqrt3 \sin(100\pi t)$.
        Reescríbalo como $R\cos(100\pi t - \varphi)$: pico
        corriente y fase?
\end{enumerate}

\medskip
\noindent\textbf{Parte IV --- The oscillating world.}
\begin{enumerate}[resume]
  \item Períodos: ¿cuál es el período (más pequeño) de
        $\sin^2 t$ (use la pregunta 13)? de
        $\sin(2t) + \sin(3t)$?
  \item Un auténtico péndulo amortigua:
        $y(t) = \eu^{-t/10}\cos(2\pi t)$. Describe el movimiento,
        calcular el valor del sobre en $t = 10$ (un viejo amigo
        aparece, \cref{pb:g12:exp:1}), y nombre el capítulo
        donde se resuelven \omterm{def:g10:algebra:equation}{ecuaciones} con tales soluciones.
  \item ¿Por qué \omterm{def:g12:trigo:cossin}{senos} y \omterm{def:g12:trigo:cossin}{cosenos} gobiernan cada vibración? Dale el
        Diccionario de física a matemáticas en dos oraciones.
        (fuerza restauradora proporcional al desplazamiento
        $\rightarrow$ cual \omterm{def:g10:algebra:equation}{ecuación} $\rightarrow$ cual
        soluciones), y decir lo que \cref{exo:g12:trigo:9}
        contribuye al reclamo.
  \item Final: el arco del capítulo en cuatro pasos --- uno
        límite geométrico, dos \omterm{def:g12:deriv:derivative}{derivados}, uno diferencial
        \omterm{def:g10:algebra:equation}{ecuación}, todo de oscilación. Y el asterisco honesto:
        Para grandes oscilaciones, el verdadero período del péndulo implica una
        Integral elíptica, calculada a la velocidad del rayo por\,\dots
        La aritmética de Gauss: geométrica \omterm{def:g11:stat:mean}{significar}, el postre de
        \cref{pb:g12:seq:1}. Los hilos de la serie se atan solos.
\end{enumerate}
\end{problem}
