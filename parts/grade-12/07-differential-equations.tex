\chapter{Differential Equations}\label{ch:g12:diffeq}

A differential equation relates a function to its derivatives. Physics,
chemistry, biology and economics express their laws in this form: the rate of
cooling of a body, the decay of a radioactive nucleus, the growth of a
population are all statements about $y'$. This chapter solves completely the
linear first-order equations with constant coefficients.

\section{The equation \texorpdfstring{$y' = ay$}{y' = ay}}

\begin{definition}[Differential equation]\label{def:g12:diffeq:ode}
A \emph{differential equation}\index{differential equation} is an equation
whose unknown is a function $y$, involving $y$ and its derivatives. A
\emph{solution} on an interval $I$ is a differentiable function satisfying
the equation at every point of $I$.
\end{definition}

\begin{theorem}[Solutions of $y' = ay$]\label{thm:g12:diffeq:homogeneous}
Let $a \in \R$. The solutions on $\R$ of the equation $y' = ay$ are exactly
the functions
\[
y(x) = C\,\eu^{ax}, \qquad C \in \R .
\]
For every pair $(x_0, y_0)$ there is a unique solution with
$y(x_0) = y_0$.
\end{theorem}

\begin{proof}
Each $y = C\eu^{ax}$ satisfies $y' = Ca\,\eu^{ax} = ay$. Conversely, let $y$
be any solution and set $z(x) = y(x)\,\eu^{-ax}$. Then
\[
z'(x) = y'(x)\,\eu^{-ax} - a\,y(x)\,\eu^{-ax}
= \bigl(y'(x) - a y(x)\bigr)\eu^{-ax} = 0,
\]
so $z$ is constant, say $C$, and $y(x) = C\eu^{ax}$. The initial condition
$y(x_0) = y_0$ forces $C = y_0 \eu^{-a x_0}$, uniquely.
\end{proof}

\begin{remark}
The exponential is thus \emph{characterized} by the simplest of all
differential equations: growth proportional to size. This is why it appears
everywhere in nature.
\end{remark}

\begin{example}[Radioactive decay]\label{ex:g12:diffeq:decay}
A radioactive quantity satisfies $N' = -\lambda N$ with $\lambda > 0$, so
$N(t) = N_0\,\eu^{-\lambda t}$; see \cref{ex:g12:exp:halflife} for the
half-life.
\end{example}

\section{The equation \texorpdfstring{$y' = ay + b$}{y' = ay + b}}

\begin{theorem}[Solutions of $y' = ay + b$]\label{thm:g12:diffeq:affine}
Let $a \neq 0$ and $b \in \R$. The solutions on $\R$ of $y' = ay + b$ are
exactly the functions
\[
y(x) = C\,\eu^{ax} - \frac{b}{a}, \qquad C \in \R .
\]
The constant function $y_p = -\frac ba$ is the \emph{equilibrium solution}.
For every $(x_0, y_0)$ there is a unique solution with $y(x_0) = y_0$; if
$a < 0$, every solution tends to the equilibrium $-\frac ba$ as
$x \to +\infty$.
\end{theorem}

\begin{proof}
The constant $y_p = -\frac{b}{a}$ satisfies $y_p' = 0 = a y_p + b$. Now $y$
is a solution if and only if
\[
(y - y_p)' = y' = ay + b = a(y - y_p) + \underbrace{a y_p + b}_{=\,0}
= a (y - y_p),
\]
that is, if and only if $z = y - y_p$ solves $z' = az$. By
\cref{thm:g12:diffeq:homogeneous}, $z = C\eu^{ax}$, whence the formula, the
existence and the uniqueness. If $a < 0$, $\eu^{ax} \to 0$ as
$x \to +\infty$, so $y(x) \to -\frac ba$.
\end{proof}

\begin{method}[Solving $y' = ay + b$ with an initial condition]\label{met:g12:diffeq:solve}
\begin{enumerate}
  \item Find the equilibrium solution $y_p = -\frac{b}{a}$ (solve $y' = 0$).
  \item Write the general solution $y = C\eu^{ax} + y_p$.
  \item Determine $C$ from the initial condition.
  \item Check the long-term behavior against physical intuition (does the
        solution converge to equilibrium?).
\end{enumerate}
The same strategy --- \emph{particular solution + general solution of the
homogeneous equation} --- extends to $y' = ay + f(x)$: see
\cref{prop:g12:diffeq:general}.
\end{method}

\begin{example}[Newton's law of cooling]\label{ex:g12:diffeq:cooling}
A cup of coffee at $80\,^\circ$C sits in a room at $20\,^\circ$C. Newton's law states
that the temperature $T$ satisfies $T' = -k(T - 20)$ for some $k > 0$,
\emph{i.e.} $T' = -kT + 20k$. The equilibrium is $20$, and
$T(t) = 20 + 60\,\eu^{-kt}$: the coffee cools exponentially fast to room
temperature.
\end{example}

\section{The equation \texorpdfstring{$y' = ay + f(x)$}{y' = ay + f(x)}}

\begin{proposition}[Structure of the solution set]\label{prop:g12:diffeq:general}
Let $f$ be continuous on an interval $I$ and let $y_p$ be one particular
solution of
\[
y' = ay + f(x)
\]
on $I$. Then the solutions on $I$ are exactly the functions
$y = C\eu^{ax} + y_p$, $C \in \R$.
\end{proposition}

\begin{proof}
As in \cref{thm:g12:diffeq:affine}: $y$ is a solution if and only if
$z = y - y_p$ satisfies
$z' = (ay + f) - (ay_p + f) = az$, if and only if $z = C\eu^{ax}$.
\end{proof}

\begin{method}[Guessing a particular solution]\label{met:g12:diffeq:particular}
Look for a particular solution \emph{of the same shape} as $f$:
\begin{itemize}
  \item $f$ polynomial of degree $n$: try a polynomial of degree $n$;
  \item $f(x) = \alpha\,\eu^{kx}$ with $k \neq a$: try $y_p = \beta\,\eu^{kx}$;
  \item $f(x) = \alpha\,\eu^{ax}$ (resonant case): try
        $y_p = \beta x\,\eu^{ax}$.
\end{itemize}
Substitute into the equation and identify the coefficients.
\end{method}

\begin{example}\label{ex:g12:diffeq:particular}
Solve $y' = 2y + 4x$. Try $y_p = \alpha x + \beta$:
$\alpha = 2(\alpha x + \beta) + 4x$ for all $x$ forces
$2\alpha + 4 = 0$ and $\alpha = 2\beta$, so $\alpha = -2$, $\beta = -1$.
General solution: $y = C\eu^{2x} - 2x - 1$.
\end{example}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:g12:diffeq:1}
Solve on $\R$:
(a) $y' = 3y$ with $y(0) = 2$;\quad
(b) $2y' + y = 0$ with $y(0) = -1$;\quad
(c) $y' = -y + 5$ with $y(0) = 0$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:diffeq:2}
A bacterial population grows at a rate proportional to its size, doubling
every $3$ hours. Write the differential equation satisfied by the population
$N(t)$ and determine the proportionality constant.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:diffeq:3}
Verify that $y_p(x) = x\,\eu^{x}$ is a solution of $y' = y + \eu^x$, and
give all solutions on $\R$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:diffeq:4}
Solve $y' = -2y + \eu^{x}$ with $y(0) = 1$.
(Hint: look for a particular solution of the form $\beta\,\eu^{x}$.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:diffeq:5}
Carbon-14 decays with a half-life of $5730$ years. An archaeological sample
contains $60\%$ of the carbon-14 of a living organism. Estimate its age.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:diffeq:6}
A tank contains $100$ L of pure water. Brine containing $0.2$ kg of salt per
liter flows in at $5$ L/min, the mixture (kept uniform) flows out at the same
rate. Let $m(t)$ be the mass of salt in the tank at time $t$ (in minutes).
\begin{enumerate}
  \item Justify that $m' = 1 - \dfrac{m}{20}$.
  \item Solve, and determine the limit of $m(t)$ as $t \to +\infty$.
        Interpret.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:diffeq:7}
A skydiver of mass $80$ kg falls subject to gravity
($g = 9.8\ \text{m/s}^2$) and air resistance proportional to speed, so that
her speed satisfies $v' = g - \frac{k}{m}v$ with $k = 16$ kg/s.
\begin{enumerate}
  \item Solve the equation with $v(0) = 0$.
  \item Compute the \emph{terminal velocity} $\lim_{t\to+\infty} v(t)$, and
        the time needed to reach $95\%$ of it.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:diffeq:8}
\emph{(Logistic equation.)} A population $y(t) \in \intoo{0}{1}$ (as a
fraction of the maximal population) satisfies
\[
y' = y(1 - y).
\]
\begin{enumerate}
  \item Let $z = \dfrac{1}{y}$. Show that $z$ satisfies the linear equation
        $z' = -z + 1$.
  \item Solve for $z$, then for $y$, with $y(0) = \frac{1}{10}$.
  \item Show that $y(t) \to 1$ as $t \to +\infty$ and sketch the shape of
        the solution curve.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:diffeq:9}
Let $y$ be a solution of $y' = ay + b$ and let $u_n = y(n)$ for $n \in \N$.
Show that $(u_n)$ satisfies an arithmetico-geometric recurrence
$u_{n+1} = q u_n + r$, and express $q$ and $r$ in terms of $a$ and $b$. What
does the condition $\abs{q} < 1$ correspond to for the differential
equation?
\end{exercise}
