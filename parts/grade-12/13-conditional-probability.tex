\chapter{Conditional Probability and Independence}\label{ch:g12:condprob}

\omterm{def:g10:proba:distribution}{Probability} quantifies uncertainty; \omterm{def:g12:condprob:cond}{conditional probability} quantifies how
information changes it. Learning that an \omterm{def:g11:prob:model}{event} $B$ occurred reshapes the
probabilities of all other \omterm{def:g11:prob:model}{events} --- a mechanism formalized by Bayes and
misused daily in courtrooms and newspapers. This chapter sets up the rules
of conditioning and the exact meaning of \omterm{def:g12:condprob:indep}{independence}.

\section{Probability spaces (reminder)}

An experiment with finitely many outcomes is modeled by a \emph{\omterm{def:g11:prob:model}{sample
space}} $\Omega$ (the set of outcomes) and a \omterm{def:g10:proba:distribution}{probability} $\P$ assigning to
each \omterm{def:g11:prob:model}{event} $A \subseteq \Omega$ a number $\P(A) \in \intcc{0}{1}$, additive
over disjoint \omterm{def:g10:numbers:interunion}{unions} and with $\P(\Omega) = 1$. Recall the basic rules:
\[
\P(\bar A) = 1 - \P(A),
\qquad
\P(A \cup B) = \P(A) + \P(B) - \P(A \cap B).
\]
When all outcomes are equally likely,
$\P(A) = \frac{\abs A}{\abs\Omega}$ --- and computing probabilities reduces
to the counting techniques of \cref{ch:g12:comb}.

\section{Conditional probability}

\begin{definition}[Conditional probability]\label{def:g12:condprob:cond}
Let $B$ be an \omterm{def:g11:prob:model}{event} with $\P(B) > 0$. The \emph{probability of $A$ given
$B$}\index{conditional probability} is
\[
\pcond{B}{A} = \frac{\P(A \cap B)}{\P(B)} .
\]
\end{definition}

The map $A \mapsto \pcond BA$ is itself a \omterm{def:g10:proba:distribution}{probability} (all rules apply); it
represents the new state of knowledge of someone who has learned that $B$
occurred.

\begin{proposition}[Multiplication rule]\label{prop:g12:condprob:product}
For \omterm{def:g11:prob:model}{events} with nonzero probabilities:
\[
\P(A \cap B) = \P(B)\,\pcond{B}{A} = \P(A)\,\pcond{A}{B},
\]
and more generally
$\P(A_1 \cap A_2 \cap A_3) = \P(A_1)\,\pcond{A_1}{A_2}\,
\pcond{A_1 \cap A_2}{A_3}$, etc.
\end{proposition}

\begin{proof}
Rearrange the definition; the chain formula follows by iterating.
\end{proof}

\begin{theorem}[Law of total probability]\label{thm:g12:condprob:total}
\index{law of total probability}
Let $B_1, \dots, B_n$ be a partition of $\Omega$ into \omterm{def:g11:prob:model}{events} of nonzero
\omterm{def:g10:proba:distribution}{probability}. For every \omterm{def:g11:prob:model}{event} $A$:
\[
\P(A) = \sum_{i=1}^{n} \P(B_i)\,\pcond{B_i}{A}.
\]
\end{theorem}

\begin{proof}
The sets $A \cap B_i$ are pairwise disjoint with \omterm{def:g10:numbers:interunion}{union} $A$, so
$\P(A) = \sum_i \P(A \cap B_i) = \sum_i \P(B_i)\pcond{B_i}{A}$ by the
multiplication rule.
\end{proof}

\begin{method}[Probability trees]\label{met:g12:condprob:tree}
A \omterm{met:g10:proba:tree}{tree diagram} organizes conditional probabilities: each branch carries the
\omterm{def:g10:proba:distribution}{probability} of the next \omterm{def:g11:prob:model}{event} \emph{given} the path so far.
\begin{itemize}
  \item The \omterm{def:g10:proba:distribution}{probability} of a leaf (a complete path) is the \emph{product} of
        the probabilities along its branches (multiplication rule).
  \item The \omterm{def:g10:proba:distribution}{probability} of an \omterm{def:g11:prob:model}{event} is the \emph{sum} of the probabilities
        of the leaves realizing it (total \omterm{def:g10:proba:distribution}{probability}).
  \item Probabilities on branches leaving one node add up to $1$.
\end{itemize}
\end{method}

\begin{theorem}[Bayes' formula]\label{thm:g12:condprob:bayes}
\index{Bayes' formula}
Let $B_1, \dots, B_n$ be a partition of $\Omega$ as above and $A$ an \omterm{def:g11:prob:model}{event}
with $\P(A) > 0$. Then
\[
\pcond{A}{B_j} =
\frac{\P(B_j)\,\pcond{B_j}{A}}{\sum_{i=1}^{n} \P(B_i)\,\pcond{B_i}{A}} .
\]
\end{theorem}

\begin{proof}
$\pcond{A}{B_j} = \frac{\P(A \cap B_j)}{\P(A)}$; expand the numerator by
the multiplication rule and the denominator by total \omterm{def:g10:proba:distribution}{probability}.
\end{proof}

\begin{example}[Screening test]\label{ex:g12:condprob:test}
A disease affects $1\%$ of a population. A test detects it with \omterm{def:g10:proba:distribution}{probability}
$0.99$ (sensitivity) and gives a false positive with \omterm{def:g10:proba:distribution}{probability} $0.05$.
Given a positive test, the \omterm{def:g10:proba:distribution}{probability} of actually having the disease is
\[
\frac{0.01 \times 0.99}{0.01\times0.99 + 0.99\times0.05}
= \frac{0.0099}{0.0099 + 0.0495} = \frac{1}{6} \approx 0.17 .
\]
Despite the accurate test, five positives out of six are false --- because
the disease is rare. Confusing $\pcond{A}{B}$ with $\pcond{B}{A}$ is the
\emph{prosecutor's fallacy}.
\end{example}

\section{Independence}

\begin{definition}[Independent events]\label{def:g12:condprob:indep}
Two \omterm{def:g11:prob:model}{events} $A$ and $B$ are \emph{independent}\index{independence} if
\[
\P(A \cap B) = \P(A)\,\P(B) .
\]
When $\P(B) > 0$, this is equivalent to $\pcond{B}{A} = \P(A)$: knowing $B$
does not change the \omterm{def:g10:proba:distribution}{probability} of $A$.
\end{definition}

\begin{proposition}\label{prop:g12:condprob:indepcompl}
If $A$ and $B$ are \omterm{def:g12:condprob:indep}{independent}, so are $A$ and $\bar B$ (and $\bar A$ and
$\bar B$).
\end{proposition}

\begin{proof}
$\P(A \cap \bar B) = \P(A) - \P(A \cap B) = \P(A) - \P(A)\P(B)
= \P(A)\bigl(1 - \P(B)\bigr) = \P(A)\,\P(\bar B)$.
\end{proof}

\begin{remark}
Do not confuse \emph{\omterm{def:g12:condprob:indep}{independent}} ($\P(A\cap B) = \P(A)\P(B)$) with
\emph{\omterm{def:g10:proba:operations}{incompatible}} ($A \cap B = \varnothing$). Two \omterm{def:g10:proba:operations}{incompatible} \omterm{def:g11:prob:model}{events} of
nonzero \omterm{def:g10:proba:distribution}{probability} are never \omterm{def:g12:condprob:indep}{independent}: knowing one occurred guarantees
the other did not.
\end{remark}

\begin{definition}[Independent repetitions]\label{def:g12:condprob:repet}
When an experiment is repeated $n$ times so that the outcome of each trial
does not influence the others, the \omterm{def:g10:proba:distribution}{probability} of a specified \omterm{def:g12:seq:sequence}{sequence} of
outcomes is the product of the individual probabilities. This is the model
underlying the binomial \omterm{def:g11:prob:rv}{distribution} (\cref{ch:g12:randvar}).
\end{definition}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:g12:condprob:1}
A card is drawn from a standard 52-card deck. Compute the \omterm{def:g10:proba:distribution}{probability} that
it is a king, given that it is a face card (jack, queen or king). Are the
\omterm{def:g11:prob:model}{events} ``king'' and ``heart'' \omterm{def:g12:condprob:indep}{independent}?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:condprob:2}
An urn contains 5 red and 3 blue balls. Two balls are drawn in succession
\emph{without} replacement.
\begin{enumerate}
  \item Draw the \omterm{def:g10:proba:distribution}{probability} \omterm{met:g10:proba:tree}{tree}.
  \item Compute the \omterm{def:g10:proba:distribution}{probability} that both are red, and the \omterm{def:g10:proba:distribution}{probability} that
        the second is red.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:condprob:3}
Two fair dice are rolled. Let $A$ = ``the sum is $7$'', $B$ = ``the first
die shows $3$'', $C$ = ``the sum is $6$''. Determine whether $A$ and $B$
are \omterm{def:g12:condprob:indep}{independent}, then whether $B$ and $C$ are.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:condprob:4}
A factory has three machines producing respectively $50\%$, $30\%$ and
$20\%$ of total output, with defect rates $1\%$, $2\%$ and $4\%$.
\begin{enumerate}
  \item What proportion of the production is defective?
  \item A randomly chosen item is defective. What is the \omterm{def:g10:proba:distribution}{probability} it
        came from the third machine?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:condprob:5}
In \cref{ex:g12:condprob:test}, at what disease prevalence $p$ (instead of
$1\%$) would a positive test \omterm{def:g11:stat:mean}{mean} at least a $90\%$ chance of disease? Solve
the inequality and comment.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:condprob:6}
A biased coin lands heads with \omterm{def:g10:proba:distribution}{probability} $p \in \intoo{0}{1}$. It is
tossed three times, tosses being \omterm{def:g12:condprob:indep}{independent}.
\begin{enumerate}
  \item Compute the \omterm{def:g10:proba:distribution}{probability} of the \omterm{def:g11:prob:model}{event} $E$ = ``exactly two heads''.
  \item For which $p$ is $\P(E)$ maximal?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:condprob:7}
\emph{(Monty Hall.)} A prize hides behind one of three doors. You pick a
door; the host, who knows where the prize is, opens one of the two
remaining doors, always revealing an empty one (choosing at random if both
are empty), and offers you to switch to the other closed door. Using
Bayes' formula, compute the \omterm{def:g10:proba:distribution}{probability} of winning if you switch, and if
you do not.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:condprob:8}
An information channel transmits bits. Each bit is flipped by noise with
\omterm{def:g10:proba:distribution}{probability} $\varepsilon = 0.1$, independently. To protect a bit, it is sent
three times and decoded by majority.
\begin{enumerate}
  \item Compute the \omterm{def:g10:proba:distribution}{probability} that the decoded bit is wrong.
  \item The received word is $101$. What is the \omterm{def:g10:proba:distribution}{probability} that the sent
        bit was $1$? (Assume $0$ and $1$ are sent with equal \omterm{def:g10:proba:distribution}{probability}.)
\end{enumerate}
\end{exercise}

\section{Problem: The doctor, the judge and the spam filter}

\begin{problem}[{Weekend problem --- Bayes' formula against the
base-rate fallacy: why a $99\,\%$ test can mean $2\,\%$, why a
one-in-a-million match convicts nobody, and how your mailbox
thinks}]
\label{pb:g12:condprob:1}
A test that catches $99\,\%$ of cases comes back positive ---
and the true \omterm{def:g10:proba:distribution}{probability} of disease is below $2\,\%$. A DNA
profile matching one person in a million is found --- and the
suspect is probably innocent. Both statements are correct, both
have been misjudged by doctors and courts, and both are settled
by one line of this chapter: Bayes' formula
(\cref{thm:g12:condprob:bayes}). This problem trains it on urns,
then in the clinic, the courtroom and the inbox.

\medskip
\noindent\textbf{Part I --- Fluency.}
\begin{enumerate}
  \item A fair coin chooses urn I ($2$ red, $1$ blue) or urn II
        ($1$ red, $3$ blue), then one ball is drawn. Compute
        $\P(\text{red})$ with the law of total \omterm{def:g10:proba:distribution}{probability}
        (\cref{thm:g12:condprob:total}).
  \item The ball is red: compute $\P(\text{urn I} \mid
        \text{red})$.
  \item One die roll: are the \omterm{def:g11:prob:model}{events} ``even'' and
        ``$\leq 4$'' \omterm{def:g12:condprob:indep}{independent}
        (\cref{def:g12:condprob:indep})? And ``even'' and
        ``$\leq 3$''?
  \item Two \omterm{def:g12:condprob:indep}{independent} smoke alarms each trigger with
        \omterm{def:g10:proba:distribution}{probability} $0.9$ in a fire. Using
        \cref{prop:g12:condprob:indepcompl}, compute the
        \omterm{def:g10:proba:distribution}{probability} that at least one triggers.
  \item Two cards without replacement: compute
        $\P(\text{two kings})$ with the multiplication rule
        (\cref{prop:g12:condprob:product}).
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The doctor.}
A disease affects $1$ person in $1\,000$. The test detects
$99\,\%$ of the sick (sensitivity) but also fires on $5\,\%$ of
the healthy (false positives).
\begin{enumerate}[resume]
  \item Natural frequencies: imagine $100\,000$ people. How
        many are sick, how many healthy? How many positives
        does each group produce, and how many positives in
        all?
  \item From the counts, compute
        $\P(\text{sick} \mid \text{positive})$. Absorb the
        shock.
  \item Recompute it with Bayes' formula and check the
        agreement.
  \item In surveys, most physicians answer ``about
        $99\,\%$''. Name the two conditional probabilities
        being confused, and explain why the tiny base rate
        drives the true answer.
  \item A second, \omterm{def:g12:condprob:indep}{independent} test also comes back positive.
        Update: taking $1.9\,\%$ as the new prior, compute
        $\P(\text{sick} \mid \text{two positives})$. What is
        evidence \emph{accumulation} doing?
  \item Compute $\P(\text{sick} \mid \text{positive})$ if the
        same test is used only on a high-risk group with
        prevalence $5\,\%$. Deduce the public-health moral
        about mass screening of rare conditions.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The judge.}
\begin{enumerate}[resume]
  \item A crime-scene DNA profile matches one person in a
        million. The suspect, found by searching a database of
        the city's $10$ million inhabitants, matches. Expected
        number of innocent matches in the city? Given only the
        match, what is $\P(\text{innocent} \mid
        \text{match})$, roughly?
  \item Name the fallacy --- which \omterm{def:g10:proba:distribution}{probability} did the
        prosecutor quote, which one did the court need? ---
        and its exact parallel with question 9.
  \item The \omterm{def:g12:condprob:indep}{independence} trap: two rare tragedies in one family
        were once squared into ``one in $73$ million'' by
        treating them as \omterm{def:g12:condprob:indep}{independent}, convicting an innocent
        mother (the Sally Clark case). If the \omterm{def:g11:prob:model}{events} share
        family risk factors so that the second, given the
        first, is $5$ times likelier than the base rate,
        recompute the order of magnitude --- and state the
        lesson about multiplying probabilities.
  \item The defense can cheat too: ``ten innocents match, so
        the match proves nothing'' ignores everything else.
        What does Bayes' formula do with \emph{several} pieces
        of evidence, and what single sentence should every
        juror know?
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The spam filter.}
\begin{enumerate}[resume]
  \item Your mail is $40\,\%$ spam. The word ``lottery''
        appears in $15\,\%$ of spam and $0.5\,\%$ of honest
        mail. A message contains it: compute
        $\P(\text{spam} \mid \text{lottery})$.
  \item Error-correcting repetition
        (\cref{exo:g12:condprob:8} continued): with flip
        \omterm{def:g10:proba:distribution}{probability} $\varepsilon = 0.1$, majority decoding of
        \emph{five} copies fails with what \omterm{def:g10:proba:distribution}{probability}?
        Compare with the three-copy code.
  \item A bit crosses $4$ \omterm{def:g12:condprob:indep}{independent} noisy relays, each
        flipping it with \omterm{def:g10:proba:distribution}{probability} $0.1$: it arrives correct
        exactly when the number of flips is even. Compute that
        \omterm{def:g10:proba:distribution}{probability} twice: by summing the binomial terms, and
        by the elegant formula
        $\frac{1 + (1 - 2\varepsilon)^4}{2}$.
  \item Real spam filters multiply the evidence of many words
        as if \omterm{def:g12:condprob:indep}{independent} (``naive'' Bayes). Why is that
        assumption false for words like ``win'' and
        ``lottery'' --- and why does the filter work well
        anyway? (One or two sentences.)
  \item Finale --- Bayes as the arithmetic of belief: prior
        $\times$ likelihood, renormalize, posterior. Replay the
        three verdicts (clinic, courtroom, inbox) in one
        sentence each, then list the three traps this problem
        exposed: forgotten base rates, fake \omterm{def:g12:condprob:indep}{independence}, and
        evidence read one piece at a time.
\end{enumerate}
\end{problem}
