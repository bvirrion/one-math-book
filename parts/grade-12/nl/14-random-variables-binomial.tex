\chapter{Stochastische variabelen en de binomiale verdeling}\label{ch:g12:randvar}

Een \omterm{def:g12:randvar:rv}{stochastische variabele} koppelt een getal aan elke uitkomst van een
experiment: een winst, een telling, een duur. Haar \omterm{def:g12:randvar:exp}{verwachting} is het
lange-termijngemiddelde van de waarden die zij produceert, haar \omterm{def:g12:randvar:exp}{variantie}
meet hun spreiding. De ster van dit hoofdstuk is de \omterm{def:g12:randvar:binomial}{binomiale verdeling},
die successen telt in herhaalde onafhankelijke proeven. \omterm{def:g12:randvar:rv}{Stochastische
variabelen} en de \omterm{def:g12:randvar:binomial}{binomiale verdeling} werden eerst ontmoet in
\cref{ch:g11:prob,ch:g11:binom}; dit hoofdstuk herneemt en verdiept ze,
met de combinatorische gereedschappen van \cref{ch:g12:comb} nu
beschikbaar.

\section{Discrete stochastische variabelen}

\begin{definition}[Stochastische variabele, verdeling]\label{def:g12:randvar:rv}
Een \emph{stochastische variabele}\index{stochastische variabele} op een
eindige \omterm{def:g11:prob:model}{uitkomstenruimte} $\Omega$ is een \omterm{def:g11:func:function}{functie} $X \colon \Omega \to \R$.
Haar \emph{verdeling}\index{verdeling} (of wet) is de data van haar
mogelijke waarden $x_1, \dots, x_k$ en van de kansen
\[
p_i = \P(X = x_i), \qquad \sum_{i=1}^{k} p_i = 1 .
\]
\end{definition}

\begin{definition}[Verwachting, variantie, standaardafwijking]\label{def:g12:randvar:exp}
De \emph{verwachting}\index{verwachting} van $X$ is
\[
\E(X) = \sum_{i=1}^{k} p_i\, x_i ,
\]
haar \emph{variantie}\index{variantie} en \emph{\omterm{def:g11:stat:variance}{standaardafwijking}} zijn
\[
\V(X) = \E\bigl((X - \E(X))^2\bigr) = \sum_{i=1}^k p_i\bigl(x_i - \E(X)\bigr)^2,
\qquad
\sigma(X) = \sqrt{\V(X)} .
\]
\end{definition}

\begin{proposition}[Formule van König--Huygens]\label{prop:g12:randvar:konig}
\index{formule van König--Huygens}
$\V(X) = \E(X^2) - \E(X)^2$.
\end{proposition}

\begin{proof}
Schrijf $m = \E(X)$ en ontwikkel:
\[
\V(X) = \sum_i p_i (x_i^2 - 2m x_i + m^2)
= \E(X^2) - 2m\sum_i p_i x_i + m^2\sum_i p_i
= \E(X^2) - 2m^2 + m^2 . \qedhere
\]
\end{proof}

\begin{proposition}[Affine transformatie]\label{prop:g12:randvar:affine}
Voor $a, b \in \R$:
\[
\E(aX + b) = a\,\E(X) + b, \qquad
\V(aX + b) = a^2\,\V(X) .
\]
\end{proposition}

\begin{proof}
De eerste is een herschikking van de definiërende som. Voor de tweede
wijkt $aX + b$ van haar \omterm{def:g11:stat:mean}{gemiddelde} af met $a(X - \E(X))$, en kwadrateren
vermenigvuldigt met $a^2$.
\end{proof}

\begin{example}[Eerlijke spellen]\label{ex:g12:randvar:game}
Een spel kost $2$ euro; een dobbelsteen wordt geworpen, en de speler
ontvangt de getoonde waarde als die minstens $5$ is, niets anders. De
winst $G$ neemt waarden $-2$ (\omterm{def:g11:prob:model}{kans} $\frac46$), $3$ ($\frac16$), $4$
($\frac16$):
\[
\E(G) = \frac{-8 + 3 + 4}{6} = -\frac{1}{6} < 0 .
\]
Gemiddeld verliest de speler $17$ cent per spel: het spel is ongunstig
(zoals de meeste echte spellen).
\end{example}

\section{Bernoulli-proeven en de binomiale verdeling}

\begin{definition}[Bernoulli-verdeling]\label{def:g12:randvar:bernoulli}
Een \emph{Bernoulli-proef}\index{Bernoulli-proef} is een experiment met
twee uitkomsten, \emph{succes} (\omterm{def:g11:prob:model}{kans} $p$) en \emph{mislukking}
($q = 1 - p$). De indicator $X$ van succes ($X = 1$ bij succes, $0$ bij
mislukking) volgt de \emph{\omterm{def:g12:randvar:rv}{Bernoulli-verdeling}} $\mathcal B(p)$:
\[
\E(X) = p, \qquad \V(X) = p(1-p) .
\]
(Inderdaad $\E(X) = p$, $\E(X^2) = p$, en König--Huygens geeft
$\V(X) = p - p^2$.)
\end{definition}

\begin{definition}[Binomiale verdeling]\label{def:g12:randvar:binomial}
Herhaal een \omterm{def:g12:randvar:bernoulli}{Bernoulli-proef} $n$ keer \omterm{def:g12:condprob:indep}{onafhankelijk}, en laat $X$ het totale
aantal successen zijn. De \omterm{def:g12:randvar:rv}{verdeling} van $X$ is de \emph{binomiale
verdeling}\index{binomiale verdeling} $\mathcal B(n, p)$.
\end{definition}

\begin{theorem}\label{thm:g12:randvar:binomial}
Als $X \sim \mathcal B(n, p)$, dan voor $0 \leq k \leq n$:
\[
\P(X = k) = \binom{n}{k} p^k (1-p)^{n-k},
\]
en
\[
\E(X) = np, \qquad \V(X) = np(1-p) .
\]
\end{theorem}

\begin{proof}
Een gespecificeerde reeks uitkomsten met $k$ successen en $n-k$
mislukkingen heeft \omterm{def:g11:prob:model}{kans} $p^k(1-p)^{n-k}$ door \omterm{def:g12:condprob:indep}{onafhankelijkheid}; het aantal
zulke reeksen is het aantal manieren om de $k$ successen onder de $n$
proeven te plaatsen, namelijk $\binom nk$ (\cref{ch:g12:comb}). Sommeren
over de reeksen geeft de formule --- en de binomiaalstelling bevestigt
$\sum_k \P(X = k) = (p + q)^n = 1$.

Voor de \omterm{def:g12:randvar:exp}{verwachting}, met $k\binom nk = n\binom{n-1}{k-1}$
(\cref{exo:g12:comb:7}):
\[
\E(X) = \sum_{k=1}^{n} k\binom nk p^k q^{n-k}
= np\sum_{k=1}^{n} \binom{n-1}{k-1} p^{k-1} q^{(n-1)-(k-1)}
= np\,(p + q)^{n-1} = np .
\]
De variantieformule wordt analoog bewezen met de identiteit
$k(k-1)\binom nk = n(n-1)\binom{n-2}{k-2}$, wat
$\E(X(X-1)) = n(n-1)p^2$ geeft, waaruit
$\V(X) = n(n-1)p^2 + np - (np)^2 = np(1-p)$. (Een structureel bewijs --- de
\omterm{def:g12:randvar:exp}{variantie} van een som van onafhankelijke variabelen --- komt met
\cref{thm:g12:sums:variance}.)
\end{proof}

\begin{method}[Een binomiale situatie herkennen]\label{met:g12:randvar:recognize}
Controleer de drie ingrediënten vóór $X \sim \mathcal B(n,p)$ te
schrijven: een \emph{vast aantal} $n$ proeven; \emph{twee uitkomsten} per
proef met dezelfde succeskans $p$; \emph{\omterm{def:g12:condprob:indep}{onafhankelijkheid}} van proeven
(bemonstering \emph{met} teruglegging, of uit een grote populatie). Gebruik
daarna
\[
\P(X \geq 1) = 1 - (1-p)^n
\]
voor ``minstens één succes'', en een rekenmachine of cumulatieve tabellen
voor algemene $\P(X \leq k)$.
\end{method}

\begin{example}\label{ex:g12:randvar:atleastone}
Hoe vaak moet men een dobbelsteen werpen om minstens $99\%$ \omterm{def:g11:prob:model}{kans} te hebben
op een zes? Met $X \sim \mathcal B\left(n, \frac16\right)$:
$\P(X \geq 1) = 1 - \left(\frac56\right)^n \geq 0.99$ betekent
$\left(\frac56\right)^n \leq 0.01$, \emph{d.w.z.}\
$n \geq \frac{\ln 0.01}{\ln(5/6)} \approx 25.3$: vanaf $n = 26$ worpen.
\end{example}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g12:randvar:1}
Een \omterm{def:g12:randvar:rv}{stochastische variabele} $X$ neemt de waarden $-1, 0, 2, 5$ aan met
kansen $0.3, 0.2, 0.4, 0.1$. Bereken $\E(X)$, $\V(X)$ en $\sigma(X)$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:randvar:2}
Een meerkeuzetoets heeft $10$ vragen, elk met $4$ keuzes, waarvan één
correct. Een student antwoordt uniform willekeurig, \omterm{def:g12:condprob:indep}{onafhankelijk}. Laat
$X$ het aantal correcte antwoorden zijn.
\begin{enumerate}
  \item Geef de \omterm{def:g12:randvar:rv}{verdeling} van $X$, $\E(X)$ en $\sigma(X)$.
  \item Bereken $\P(X = 0)$, $\P(X = 5)$ en $\P(X \geq 1)$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:randvar:3}
Een verzekeringsmaatschappij verzekert $n = 400$ cliënten; elk dient in de
loop van het jaar een claim in met \omterm{def:g11:prob:model}{kans} $p = 0.05$, \omterm{def:g12:condprob:indep}{onafhankelijk}. Laat $X$
het aantal claims zijn. Identificeer de \omterm{def:g12:randvar:rv}{verdeling} van $X$ en bereken haar
\omterm{def:g12:randvar:exp}{verwachting} en \omterm{def:g11:stat:variance}{standaardafwijking}.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:randvar:4}
In het spel van \cref{ex:g12:randvar:game} wil de organisator een eerlijk
spel ($\E(G) = 0$) door de inzetprijs $c$ te veranderen. Vind $c$. Bereken
de \omterm{def:g12:randvar:exp}{variantie} van de winst voor deze eerlijke versie; is ``eerlijk''
hetzelfde als ``risicoloos''?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:randvar:5}
Een basketballer scoort vrije worpen met \omterm{def:g11:prob:model}{kans} $0.7$. Zij schiet $8$ keer
(onafhankelijke schoten). Bereken de \omterm{def:g11:prob:model}{kans} dat zij scoort: precies $6$;
minstens $6$; minstens één keer. Wat is het meest waarschijnlijke aantal
scores?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:randvar:6}
Een luchtvaartmaatschappij weet dat elke geboekte passagier verschijnt met
\omterm{def:g11:prob:model}{kans} $0.9$, \omterm{def:g12:condprob:indep}{onafhankelijk}. Een vlucht heeft $100$ stoelen en de
maatschappij verkoopt $104$ tickets. Druk, met een \omterm{def:g12:randvar:binomial}{binomiale verdeling}, de
\omterm{def:g11:prob:model}{kans} uit dat meer passagiers verschijnen dan er stoelen zijn, en begrens
haar numeriek met een rekenmachine (geef de exacte uitdrukking).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:randvar:7}
Laat $X \sim \mathcal B(n, p)$. Toon dat
\[
\frac{\P(X = k+1)}{\P(X = k)} = \frac{n-k}{k+1}\cdot\frac{p}{1-p},
\]
en leid af dat de \omterm{def:g12:randvar:rv}{verdeling} toeneemt tot
$k^* = \floor{(n+1)p}$ en daarna afneemt (de \emph{modus} van de
binomiaal).
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:randvar:8}
\emph{(Sint-Petersburg, getemd.)} Een eerlijke munt wordt geworpen tot
kop verschijnt, maar hoogstens $10$ keer. Laat $N$ het aantal gebruikte
worpen zijn, en de speler ontvangt $2^N$ euro als kop verscheen, $0$
anders.
\begin{enumerate}
  \item Geef de \omterm{def:g12:randvar:rv}{verdeling} van $N$ beperkt tot de winnende uitkomsten:
        $\P(\text{kop eerst bij worp } k) = 2^{-k}$ voor
        $1 \leq k \leq 10$, en controleer de totale winstkans.
  \item Bereken de verwachte uitbetaling. Wat zou zij worden zonder de
        cap van $10$ worpen?
\end{enumerate}
\end{exercise}
