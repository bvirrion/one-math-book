\chapter{Sommen van stochastische variabelen en de wet van de grote aantallen}\label{ch:g12:sums}

Waarom winnen casino's altijd op den duur, en waarom werken peilingen?
Omdat gemiddelden van veel onafhankelijke stochastische grootheden steeds
minder fluctueren. Dit hoofdstuk bewijst het: lineariteit van de
verwachting, additiviteit van de variantie voor onafhankelijke variabelen,
de ongelijkheid van Bienaymé--Tsjebysjev, en de wet van de grote
aantallen.

\section{Sommen van stochastische variabelen}

Gegeven twee stochastische variabelen $X, Y$ op dezelfde eindige
uitkomstenruimte $\Omega$, is de som $X + Y$ de stochastische variabele
$\omega \mapsto X(\omega) + Y(\omega)$.

\begin{theorem}[Lineariteit van de verwachting]\label{thm:g12:sums:linearity}
\index{verwachting!lineariteit}
Voor alle stochastische variabelen $X, Y$ op $\Omega$ en $a, b \in \R$:
\[
\E(X + Y) = \E(X) + \E(Y), \qquad \E(aX + b) = a\E(X) + b .
\]
\emph{Geen onafhankelijkheidshypothese is nodig.}
\end{theorem}

\begin{proof}
Schrijf de verwachting als som over uitkomsten: omdat
$\P(X = x) = \sum_{\omega : X(\omega) = x} \P(\{\omega\})$, groeperen van
termen geeft $\E(X) = \sum_{\omega \in \Omega} \P(\{\omega\})\,X(\omega)$.
Dan
\[
\E(X + Y) = \sum_{\omega} \P(\{\omega\})\bigl(X(\omega) + Y(\omega)\bigr)
= \sum_{\omega} \P(\{\omega\})X(\omega)
+ \sum_{\omega} \P(\{\omega\})Y(\omega) = \E(X) + \E(Y). \qedhere
\]
\end{proof}

\begin{definition}[Onafhankelijke stochastische variabelen]\label{def:g12:sums:indep}
$X$ en $Y$ zijn \emph{onafhankelijk}\index{stochastische
variabele!onafhankelijk} als voor alle waarden $x, y$:
\[
\P(X = x \text{ en } Y = y) = \P(X = x)\,\P(Y = y).
\]
Meerdere variabelen $X_1, \dots, X_n$ zijn onafhankelijk als deze
productregel geldt voor elke keuze van waarden van elke subfamilie.
\end{definition}

\begin{proposition}\label{prop:g12:sums:prodexp}
Als $X$ en $Y$ onafhankelijk zijn, $\E(XY) = \E(X)\,\E(Y)$.
\end{proposition}

\begin{proof}
\[
\begin{aligned}
\E(XY) &= \sum_{x, y} xy\;\P(X = x \text{ en } Y = y)
= \sum_{x, y} xy\,\P(X=x)\P(Y=y) \\
&= \Bigl(\sum_x x\P(X=x)\Bigr)\Bigl(\sum_y y\P(Y=y)\Bigr). \qedhere
\end{aligned}
\]
\end{proof}

\begin{theorem}[Variantie van een som]\label{thm:g12:sums:variance}
\index{variantie!van een som}
Als $X$ en $Y$ \emph{onafhankelijk} zijn, dan
\[
\V(X + Y) = \V(X) + \V(Y).
\]
Algemener, voor onafhankelijke $X_1, \dots, X_n$:
$\V(X_1 + \dots + X_n) = \V(X_1) + \dots + \V(X_n)$.
\end{theorem}

\begin{proof}
Met König--Huygens (\cref{prop:g12:randvar:konig}) en lineariteit:
\begin{align*}
\V(X + Y) &= \E\bigl((X+Y)^2\bigr) - \bigl(\E X + \E Y\bigr)^2\\
&= \E(X^2) + 2\E(XY) + \E(Y^2)
- \E(X)^2 - 2\E(X)\E(Y) - \E(Y)^2\\
&= \V(X) + \V(Y) + 2\bigl(\E(XY) - \E(X)\E(Y)\bigr),
\end{align*}
en de laatste haak verdwijnt voor onafhankelijke variabelen
(\cref{prop:g12:sums:prodexp}). Het algemene geval volgt door inductie.
\end{proof}

\begin{example}[Binomiaal hernomen]\label{ex:g12:sums:binomial}
Een binomiale variabele $X \sim \mathcal B(n, p)$ is een som
$X = X_1 + \dots + X_n$ van $n$ onafhankelijke Bernoulli-variabelen.
Dus, structureel:
\[
\E(X) = np, \qquad \V(X) = np(1-p),
\]
wat \cref{thm:g12:randvar:binomial} terugvindt zonder enige berekening.
\end{example}

\begin{proposition}[Steekproefgemiddelde]\label{prop:g12:sums:mean}
Laat $X_1, \dots, X_n$ onafhankelijk zijn met dezelfde verdeling als $X$
(verwachting $\mu$, variantie $\sigma^2$), en laat
$M_n = \frac{X_1 + \dots + X_n}{n}$ hun \emph{steekproefgemiddelde}
\index{steekproefgemiddelde} zijn. Dan
\[
\E(M_n) = \mu, \qquad \V(M_n) = \frac{\sigma^2}{n}, \qquad
\sigma(M_n) = \frac{\sigma}{\sqrt n}.
\]
\end{proposition}

\begin{proof}
Lineariteit geeft $\E(M_n) = \frac{n\mu}{n} = \mu$; onafhankelijkheid geeft
$\V(X_1 + \dots + X_n) = n\sigma^2$, en delen door $n$ schaalt de
variantie met $\frac{1}{n^2}$ (\cref{prop:g12:randvar:affine}).
\end{proof}

De $\frac{\sigma}{\sqrt n}$ is de fundamentele \emph{wortelwet}: om de
fluctuaties van een gemiddelde te halveren, verviervoudig de
steekproefgrootte.

\section{Concentratieongelijkheden}

\begin{theorem}[Ongelijkheid van Markov]\label{thm:g12:sums:markov}
\index{ongelijkheid van Markov}
Als $X \geq 0$ en $a > 0$:
\[
\P(X \geq a) \leq \frac{\E(X)}{a}.
\]
\end{theorem}

\begin{proof}
In $\E(X) = \sum_i p_i x_i$ (alle termen niet-negatief), behoud alleen de
termen met $x_i \geq a$: elk is minstens $a\,p_i$, dus
$\E(X) \geq a \sum_{x_i \geq a} p_i = a\,\P(X \geq a)$.
\end{proof}

\begin{theorem}[Ongelijkheid van Bienaymé--Tsjebysjev]\label{thm:g12:sums:chebyshev}
\index{ongelijkheid van Bienaymé--Tsjebysjev}
Voor elke stochastische variabele $X$ en elke $\varepsilon > 0$:
\[
\P\bigl(\abs{X - \E(X)} \geq \varepsilon\bigr)
\leq \frac{\V(X)}{\varepsilon^2}.
\]
\end{theorem}

\begin{proof}
Pas de ongelijkheid van Markov toe op de niet-negatieve variabele
$Y = (X - \E(X))^2$ met $a = \varepsilon^2$:
\[
\P\bigl(\abs{X - \E(X)} \geq \varepsilon\bigr)
= \P(Y \geq \varepsilon^2)
\leq \frac{\E(Y)}{\varepsilon^2} = \frac{\V(X)}{\varepsilon^2}. \qedhere
\]
\end{proof}

\begin{theorem}[Wet van de grote aantallen]\label{thm:g12:sums:lln}
\index{wet van de grote aantallen}
Laat $X_1, \dots, X_n$ onafhankelijk zijn met dezelfde verdeling
(verwachting $\mu$, variantie $\sigma^2$) en $M_n$ hun
steekproefgemiddelde. Voor elke $\varepsilon > 0$:
\[
\P\bigl(\abs{M_n - \mu} \geq \varepsilon\bigr)
\leq \frac{\sigma^2}{n\,\varepsilon^2}
\xrightarrow[n \to +\infty]{} 0 .
\]
Het steekproefgemiddelde concentreert rond de verwachting.
\end{theorem}

\begin{proof}
Bienaymé--Tsjebysjev toegepast op $M_n$, waarvan de verwachting $\mu$ is
en de variantie $\frac{\sigma^2}{n}$ (\cref{prop:g12:sums:mean}).
\end{proof}

\begin{remark}
Deze stelling is de brug tussen kansrekening en statistiek: de
\emph{frequentie} van een gebeurtenis over veel onafhankelijke herhalingen
nadert haar \emph{kans} (neem $X_i$ de indicator van de gebeurtenis, zodat
$\mu = p$). Zij rechtvaardigt het schatten van een kans door simulatie
(\emph{Monte Carlo-methode}) en een populatieproportie door een peiling ---
kwantitatief: zie \cref{ch:g12:contdist}.
\end{remark}

\begin{method}[Bienaymé--Tsjebysjev gebruiken]\label{met:g12:sums:chebyshev}
Om $\P(\abs{M_n - \mu} \geq \varepsilon) \leq \alpha$ te garanderen, volstaat
$n \geq \frac{\sigma^2}{\alpha\,\varepsilon^2}$. De grens is grof (echte
fluctuaties zijn meestal veel kleiner) maar perfect algemeen: zij eist
niets over de verdeling buiten een eindige variantie.
\end{method}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g12:sums:1}
Twee eerlijke dobbelstenen worden geworpen; laat $S$ de som zijn. Met
lineariteit (niet de verdeling van $S$!), bereken $\E(S)$; bereken daarna
$\V(S)$ met onafhankelijkheid, gegeven dat één eerlijke steen variantie
$\frac{35}{12}$ heeft.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:sums:2}
Laat $X \sim \mathcal B(100,\ 0.5)$. Begrens
$\P(X \geq 75)$ met de ongelijkheid van Markov, daarna
$\P(\abs{X - 50} \geq 25)$ met Bienaymé--Tsjebysjev. Vergelijk.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:sums:3}
Een eerlijke munt wordt $n$ keer geworpen en $F_n$ duidt de frequentie van
kop aan. Hoe groot moet $n$ zijn zodat, volgens de grens van
Bienaymé--Tsjebysjev,
$\P\bigl(\abs{F_n - 0.5} \geq 0.05\bigr) \leq 0.05$?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:sums:4}
Een investeerder splitst haar kapitaal gelijk over $n$ onafhankelijke
activa, elk met verwachte opbrengst $\mu = 5\%$ en standaardafwijking
$\sigma = 20\%$. Bereken de verwachting en standaardafwijking van de
portefeuilleopbrengst $M_n$, en het aantal activa nodig om de
standaardafwijking onder $4\%$ te brengen. Welk financieel principe
illustreert dit?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:sums:5}
Laat $X$ en $Y$ onafhankelijk zijn, beide uniform op $\{1, 2, 3\}$.
\begin{enumerate}
  \item Geef de verdeling van $S = X + Y$ en bereken $\E(S)$, $\V(S)$
        direct daaruit.
  \item Herwin beide waarden door lineariteit en additiviteit van
        variantie.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:sums:6}
Toon dat de additiviteit van variantie kan falen zonder onafhankelijkheid:
bereken $\V(X + Y)$ voor $Y = X$, en vergelijk met $\V(X) + \V(Y)$. Voor
welke variabelen $X$ is $\V(2X) = 2\V(X)$?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:sums:7}
Een dobbelsteen wordt verdacht oneerlijk te zijn. Zij wordt $1200$ keer
geworpen en toont een zes $260$ keer ($f = 0.2167$ in plaats van
$\frac16 \approx 0.1667$). Onder de hypothese dat de steen eerlijk is,
begrens $\P\bigl(\abs{F_n - \frac16} \geq 0.05\bigr)$ door
Bienaymé--Tsjebysjev, en bespreek of de eerlijkheidshypothese aannemelijk
is.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:sums:8}
\emph{(Een betere ongelijkheid voor de munt.)} Laat
$X \sim \mathcal B(n,\ p)$ en $F_n = \frac Xn$.
\begin{enumerate}
  \item Toon dat $p(1-p) \leq \frac14$ voor $p \in \intcc{0}{1}$.
  \item Leid de verdelingsvrije grens af
        $\P\bigl(\abs{F_n - p} \geq \varepsilon\bigr) \leq
        \frac{1}{4n\varepsilon^2}$.
  \item Hoeveel mensen moeten gepolld worden zodat de waargenomen
        frequentie binnen $3$ punten van de ware proportie ligt met kans
        minstens $95\%$, met deze grens? (Echte peilingen gebruiken
        scherpere schattingen, maar de ordegrootte klopt.)
\end{enumerate}
\end{exercise}
