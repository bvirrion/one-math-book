\chapter{Exponentiële functie en logaritme}\label{ch:g12:exp}

De exponentiële functie is de unieke functie die gelijk is aan haar eigen
afgeleide en de waarde $1$ in $0$ aanneemt. Ze zet sommen om in producten;
haar inverse, de natuurlijke logaritme, zet producten om in sommen.
Samen beschrijven ze elk verschijnsel waarvan de veranderingsverhouding
evenredig is met de grootte: radioactief verval, populatiegroei, samengestelde
rente.

\section{De exponentiële functie}

\begin{theorem}[Bestaan en uniciteit]\label{thm:g12:exp:existence}
Er bestaat een unieke afleidbare functie $f \colon \R \to \R$ zodat
\[
f' = f \qquad\text{en}\qquad f(0) = 1 .
\]
Ze heet de \emph{exponentiële functie}\index{exponentiële functie} en wordt
genoteerd $\exp$, of $x \mapsto \eu^x$.
\end{theorem}

\begin{proof}[Bewijs van uniciteit]
Eerst verdwijnt zo'n functie nooit. Inderdaad, laat
$g(x) = f(x)f(-x)$; dan
\[
g'(x) = f'(x)f(-x) - f(x)f'(-x) = f(x)f(-x) - f(x)f(-x) = 0,
\]
dus $g$ is constant gelijk aan $g(0) = 1$: voor elke $x$,
$f(x)f(-x) = 1$, en in het bijzonder $f(x) \neq 0$.

Laat nu $f_1, f_2$ twee oplossingen zijn en $h = \dfrac{f_1}{f_2}$
(legitiem omdat $f_2$ nooit verdwijnt). Dan
\[
h' = \frac{f_1' f_2 - f_1 f_2'}{f_2^2} = \frac{f_1 f_2 - f_1 f_2}{f_2^2} = 0,
\]
dus $h$ is constant gelijk aan $h(0) = 1$, d.w.z.\ $f_1 = f_2$.

\emph{Bestaan} is toegegeven op dit niveau (men kan het verkrijgen via de
rij van \cref{exo:g12:seq:10}, of als inverse van de logaritme
geconstrueerd door integratie in \cref{ch:g12:integ}).
\end{proof}

\begin{proposition}[Functionaalvergelijking]\label{prop:g12:exp:funceq}
Voor alle $x, y \in \R$ en $n \in \Z$:
\[
\eu^{x+y} = \eu^x\,\eu^y, \qquad
\eu^{-x} = \frac{1}{\eu^x}, \qquad
\eu^{x-y} = \frac{\eu^x}{\eu^y}, \qquad
\bigl(\eu^{x}\bigr)^n = \eu^{nx}.
\]
Bovendien is $\eu^x > 0$ voor alle $x$.
\end{proposition}

\begin{proof}
Fixeer $y$ en beschouw $\varphi(x) = \dfrac{\exp(x+y)}{\exp(x)\exp(y)}$. Haar
teller en noemer, als functies van $x$, zijn beide oplossingen van
$f' = f$ op de constante $\exp(y)$ na; differentiëren van $\varphi$
direct (quotiëntrregel) geeft $\varphi' = 0$, dus
$\varphi \equiv \varphi(0) = 1$, wat de eerste identiteit bewijst. Nemen
$y = -x$ geeft de tweede (met de waarde $\eu^0 = 1$), en de derde volgt.
De vierde is een inductie uit de eerste voor $n \geq 0$, uitgebreid tot
$n < 0$ door de tweede.

Positiviteit: $\eu^x = \left(\eu^{x/2}\right)^2 \geq 0$ en $\eu^x \neq 0$
(getoond in \cref{thm:g12:exp:existence}), dus $\eu^x > 0$.
\end{proof}

\begin{proposition}[Variaties en limieten]\label{prop:g12:exp:variations}
De exponentiële is strikt stijgend, convex, en
\[
\lim_{x\to-\infty} \eu^x = 0, \qquad
\lim_{x\to+\infty} \eu^x = +\infty .
\]
\end{proposition}

\begin{proof}
$(\exp)' = \exp > 0$ geeft strikte stijging; $(\exp)'' = \exp > 0$ geeft
convexiteit. Door convexiteit is $\eu^x \geq 1 + x$ (raaklijn in $0$, zie
\cref{ex:g12:deriv:ineq}), dus $\eu^x \to +\infty$ als $x \to +\infty$ door
vergelijking. Dan $\eu^{x} = 1/\eu^{-x} \to 0$ als $x \to -\infty$.
\end{proof}

\begin{theorem}[Groeivergelijking]\label{thm:g12:exp:growth}
\index{groeivergelijking}
Voor elk geheel getal $n \geq 1$,
\[
\lim_{x\to+\infty} \frac{\eu^x}{x^n} = +\infty,
\qquad
\lim_{x\to-\infty} x^n \eu^x = 0 .
\]
In woorden: de exponentiële verslaat elke macht van $x$.
\end{theorem}

\begin{proof}
Voor $n = 1$: toepassen van $\eu^t \geq 1 + t \geq t$ in $t = x/2$,
\[
\frac{\eu^x}{x} = \frac{\left(\eu^{x/2}\right)^2}{x}
\geq \frac{(x/2)^2}{x} = \frac{x}{4} \xrightarrow[x\to+\infty]{} +\infty .
\]
Voor algemene $n$, schrijf
\[
\frac{\eu^x}{x^n}
= \frac{1}{n^n}\left(\frac{\eu^{x/n}}{x/n}\right)^{n}
\xrightarrow[x\to+\infty]{} +\infty
\]
door het geval $n=1$ en samenstelling. De limiet in $-\infty$ volgt door
de substitutie $x \mapsto -x$:
$\abs{x^n \eu^x} = \frac{\abs{x}^n}{\eu^{\abs x}} \to 0$.
\end{proof}

\section{De natuurlijke logaritme}

\begin{definition}[Natuurlijke logaritme]\label{def:g12:exp:ln}
De exponentiële is continu en strikt stijgend van $\R$ op
$\intoo{0}{+\infty}$; volgens de bijectiestelling
(\cref{thm:g12:limcont:bijection}) heeft voor elke $y > 0$ de vergelijking
$\eu^x = y$ een unieke oplossing. Deze oplossing is de \emph{natuurlijke
logaritme}\index{logaritme} van $y$, genoteerd $\ln y$. Dus
\[
\text{voor } x \in \R,\ y > 0: \qquad y = \eu^x \iff x = \ln y .
\]
In het bijzonder $\ln 1 = 0$, $\ln \eu = 1$, en
$\eu^{\ln y} = y$, $\ln(\eu^x) = x$.
\end{definition}

\begin{proposition}[Algebraïsche eigenschappen]\label{prop:g12:exp:lnalg}
Voor alle $a, b > 0$ en $n \in \Z$:
\[
\ln(ab) = \ln a + \ln b, \quad
\ln\frac{1}{a} = -\ln a, \quad
\ln\frac{a}{b} = \ln a - \ln b, \quad
\ln(a^n) = n \ln a, \quad
\ln\sqrt{a} = \tfrac12 \ln a .
\]
\end{proposition}

\begin{proof}
$\eu^{\ln a + \ln b} = \eu^{\ln a}\eu^{\ln b} = ab$, en $\ln$ nemen van
beide kanten geeft de eerste identiteit. De andere volgen met hetzelfde
mechanisme uit de bijbehorende identiteiten van
\cref{prop:g12:exp:funceq}.
\end{proof}

\begin{proposition}[Analytische eigenschappen]\label{prop:g12:exp:lnanalytic}
De functie $\ln$ is afleidbaar op $\intoo{0}{+\infty}$ met
\[
(\ln x)' = \frac{1}{x},
\]
strikt stijgend, concaaf, en
$\lim\limits_{x\to0^+} \ln x = -\infty$,
$\lim\limits_{x\to+\infty} \ln x = +\infty$. Bovendien, voor elk geheel
getal $n \geq 1$,
\[
\lim_{x\to+\infty} \frac{\ln x}{x^{1/n}} = 0, \qquad
\lim_{x\to0^+} x \ln x = 0 .
\]
\end{proposition}

\begin{proof}
Afleidbaarheid van de inverse functie is toegegeven op dit niveau; dat
aannemend, differentieer de identiteit $\eu^{\ln x} = x$ met de
kettingregel: $(\ln x)'\,\eu^{\ln x} = 1$, dus $(\ln x)' = \frac{1}{x} > 0$,
vandaar strikte stijging, en $(\ln x)'' = -\frac1{x^2} < 0$, vandaar
concaviteit. De limieten in $0^+$ en $+\infty$ weerspiegelen die van
$\exp$ via de bijectie. Voor de groeivergelijking, substitueer
$x = \eu^{t}$: $\frac{\ln x}{x} = \frac{t}{\eu^t} \to 0$ als
$t \to +\infty$ volgens \cref{thm:g12:exp:growth}, en evenzo voor de
andere limieten met $x = \eu^{-t}$, wat $x\ln x = -t\eu^{-t} \to 0$
geeft.
\end{proof}

\begin{method}[Vergelijkingen met $\exp$ en $\ln$ oplossen]\label{met:g12:exp:equations}
Beide functies zijn strikt stijgend, dus kunnen ze op (of van) beide
kanten van een vergelijking of ongelijkheid worden toegepast zonder de
richting te veranderen:
\[
\eu^{u} = \eu^{v} \iff u = v, \qquad
\eu^{u} \leq \eu^{v} \iff u \leq v,
\]
en evenzo voor $\ln$ op positieve grootheden. \emph{Controleer altijd
eerst de domeinen} ($\ln$ eist positieve argumenten). Voor vergelijkingen
zoals $a^x = b$ met $a>0$, $a \neq 1$, herschrijf $a^x = \eu^{x\ln a}$ en
los op $x = \frac{\ln b}{\ln a}$.
\end{method}

\begin{example}\label{ex:g12:exp:halflife}
Een radioactief monster vervalt volgens $N(t) = N_0\,\eu^{-\lambda t}$.
Haar \emph{halveringstijd} $T$ voldoet aan $N(T) = N_0/2$, d.w.z.\
$\eu^{-\lambda T} = \frac12$, dus $T = \frac{\ln 2}{\lambda}$: de
halveringstijd hangt niet af van de beginkwantiteit.
\end{example}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g12:exp:1}
Vereenvoudig
$\dfrac{\eu^{3x}\,\eu^{-x+1}}{\eu^{x}}$ en
$\ln\!\left(\dfrac{\eu^2\sqrt{\eu}}{\eu^{-3}}\right)$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:exp:2}
Los op in $\R$:
\[
\text{(a) } \eu^{2x} - 3\eu^{x} + 2 = 0;
\qquad
\text{(b) } \ln(x - 1) + \ln(x + 2) = \ln 4 .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:exp:3}
Bereken de limieten:
\[
\lim_{x\to+\infty} \frac{\eu^x - x^2}{\eu^x + 1}, \qquad
\lim_{x\to+\infty} \frac{\ln x}{\sqrt x}, \qquad
\lim_{x\to0^+} x^2 \ln x, \qquad
\lim_{x\to+\infty} \bigl(x - \ln x\bigr).
\]
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:exp:4}
Bestudeer de functie $f(x) = x\,\eu^{-x}$ op $\R$: variaties, limieten,
extremum; toon aan dat haar kromme een buigpunt heeft en geef de
coördinaten.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:exp:5}
Toon aan dat voor alle $x > -1$, $\ln(1 + x) \leq x$, met gelijkheid
alleen in $x = 0$. Leid af dat voor alle $n \geq 1$,
\[
\left(1 + \frac{1}{n}\right)^n \leq \eu \leq \left(1 - \frac{1}{n+1}\right)^{-(n+1)} .
\]
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:exp:6}
Een kapitaal $C_0$ wordt belegd tegen een jaarlijks tarief van $3\%$,
rente jaarlijks samengesteld.
\begin{enumerate}
  \item Druk het kapitaal $C_n$ na $n$ jaren uit.
  \item Na hoeveel jaren verdubbelt het kapitaal? Geef het exacte
        antwoord met $\ln$, daarna een numerieke waarde.
  \item Vergelijk met de bankiersbenadering ``$70$ gedeeld door het tarief
        in procent''.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:exp:7}
Los de ongelijkheid $\eu^{2x} - \eu^{x+1} > 0$ op, daarna de ongelijkheid
$\ln(x^2 - 1) \leq \ln(x + 5)$.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:exp:8}
Laat $f(x) = \dfrac{\eu^x}{x}$ voor $x > 0$.
\begin{enumerate}
  \item Bestudeer de variaties van $f$ op $\intoo{0}{+\infty}$ en geef
        haar minimum.
  \item Voor welke waarden van $k$ heeft de vergelijking $\eu^x = kx$
        $0$, $1$ of $2$ oplossingen in $\intoo{0}{+\infty}$?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:exp:9}
Voor $n \geq 1$, laat $u_n = \left(1 + \frac1n\right)^n$.
\begin{enumerate}
  \item Toon met \cref{exo:g12:exp:5} aan dat $(u_n)$ van boven begrensd
        is door~$\eu$.
  \item Toon aan dat $\ln u_n = n \ln\left(1 + \frac1n\right) \to 1$, en
        leid af dat $u_n \to \eu$.
        (Hint: herken een verschilquotiënt van $\ln$ in $1$.)
\end{enumerate}
\end{exercise}
