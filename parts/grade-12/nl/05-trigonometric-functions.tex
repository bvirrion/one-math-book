\chapter{Trigonometrische functies}\label{ch:g12:trigo}

De functies $\cos$ en $\sin$ zetten de meetkunde van de cirkel om in
analyse. Gedefinieerd door de reële lijn om de eenheidscirkel te
winden, zijn ze het archetypie van periodieke functies, en hun afgeleiden
maken er oplossingen van van de oscillatievergelijking $y'' = -y$.

\section{De eenheidscirkel}

\begin{definition}[Cosinus en sinus]\label{def:g12:trigo:cossin}
Laat $\mathcal{C}$ de cirkel van straal $1$ zijn, gecentreerd in de
oorsprong van een orthonormaal stelsel. Aan elk reëel getal $t$ koppelt
men het punt $M(t)$ verkregen door een lengte $\abs{t}$ langs
$\mathcal{C}$ te winden vanaf het punt $I(1,0)$, in de
tegenwijzerzin als $t \geq 0$ en met de klok mee anders. Dan zijn
\emph{$\cos t$ en $\sin t$ de coördinaten van
$M(t)$}\index{cosinus}\index{sinus}:
\[
M(t) = (\cos t, \sin t).
\]
\end{definition}

Omdat de cirkel omtrek $2\pi$ heeft, blijft het punt $M(t)$ onveranderd
wanneer $t$ met $2\pi$ toeneemt; en omdat $M(t)$ op de eenheidscirkel
ligt:

\begin{proposition}[Fundamentele identiteiten]\label{prop:g12:trigo:identities}
Voor alle $t \in \R$ en $k \in \Z$:
\[
\cos^2 t + \sin^2 t = 1, \qquad
\cos(t + 2k\pi) = \cos t, \qquad
\sin(t + 2k\pi) = \sin t,
\]
\[
\cos(-t) = \cos t, \qquad \sin(-t) = -\sin t,
\]
\[
\cos(\pi - t) = -\cos t, \quad \sin(\pi - t) = \sin t, \quad
\cos\left(\tfrac{\pi}{2} - t\right) = \sin t, \quad
\sin\left(\tfrac{\pi}{2} - t\right) = \cos t .
\]
\end{proposition}

\begin{proof}
De eerste identiteit is de vergelijking van de eenheidscirkel; de andere
drukken de symmetrieën van de cirkel uit: $M(-t)$ is het spiegelbeeld van
$M(t)$ t.o.v.\ de $x$-as, $M(\pi - t)$ t.o.v.\ de $y$-as, en
$M(\frac\pi2 - t)$ t.o.v.\ de lijn $y = x$.
\end{proof}

De waarden om uit het hoofd te kennen:
\begin{center}
\begin{tabular}{c|ccccc}
$t$ & $0$ & $\dfrac{\pi}{6}$ & $\dfrac{\pi}{4}$ & $\dfrac{\pi}{3}$ & $\dfrac{\pi}{2}$\\[6pt]
\hline\rule{0pt}{14pt}%
$\cos t$ & $1$ & $\dfrac{\sqrt3}{2}$ & $\dfrac{\sqrt2}{2}$ & $\dfrac12$ & $0$\\[6pt]
$\sin t$ & $0$ & $\dfrac12$ & $\dfrac{\sqrt2}{2}$ & $\dfrac{\sqrt3}{2}$ & $1$
\end{tabular}
\end{center}

\begin{proposition}[Optelformules]\label{prop:g12:trigo:addition}
Voor alle $a, b \in \R$:
\begin{align*}
\cos(a+b) &= \cos a \cos b - \sin a \sin b, &
\sin(a+b) &= \sin a \cos b + \cos a \sin b,\\
\cos(a-b) &= \cos a \cos b + \sin a \sin b, &
\sin(a-b) &= \sin a \cos b - \cos a \sin b.
\end{align*}
In het bijzonder $\cos 2a = 2\cos^2 a - 1 = 1 - 2\sin^2 a$ en
$\sin 2a = 2 \sin a \cos a$.
\end{proposition}

\begin{proof}
De vectoren $\vect{u} = (\cos a, \sin a)$ en $\vect{v} = (\cos b, \sin b)$
zijn eenheidsvectoren die een hoek $a - b$ maken; hun inwendig product
is tegelijk $\cos a\cos b + \sin a \sin b$ (coördinaten) en
$\norm{\vect u}\,\norm{\vect v}\cos(a-b) = \cos(a - b)$ (meetkunde), wat
de derde formule is. Vervangen van $b$ door $-b$ geeft de eerste; de
sinusformules volgen met $\sin x = \cos(\frac\pi2 - x)$. De
verdubbelingsformules zijn het geval $b = a$ gecombineerd met
$\cos^2 + \sin^2 = 1$.
\end{proof}

\section{Analytische studie}

\begin{lemma}[Fundamentele limiet]\label{lem:g12:trigo:sinxx}
\[
\lim_{t \to 0} \frac{\sin t}{t} = 1
\qquad\text{en}\qquad
\lim_{t \to 0} \frac{\cos t - 1}{t} = 0 .
\]
\end{lemma}

\begin{proof}
Voor $0 < t < \frac{\pi}{2}$, vergelijk drie oppervlaktes in de
eenheidscirkel: de driehoek $OIM(t)$, de cirkelsector $OIM(t)$, en de
rechthoekige driehoek met basis $OI$ en hoogte $\tan t$:
\[
\frac{\sin t}{2} \leq \frac{t}{2} \leq \frac{\tan t}{2}.
\]

{\small De drie geneste oppervlaktes: de driehoek $OIM$ (blauw, oppervlakte
$\frac{\sin t}{2}$), de cirkelsector (begrensd door de rode boog, oppervlakte
$\frac{t}{2}$), en de rechthoekige driehoek van hoogte $\tan t$ (oranje,
oppervlakte $\frac{\tan t}{2}$).}
\end{omfigure}

De eerste ongelijkheid geeft $\frac{\sin t}{t} \leq 1$; de tweede geeft
$\cos t \leq \frac{\sin t}{t}$. Als $t \to 0^+$, $\cos t \to 1$
(continuïteit), en de insluitstelling geeft $\frac{\sin t}{t} \to 1$;
evenheid breidt dit uit naar $t \to 0$. Voor de tweede limiet,
\[
\frac{\cos t - 1}{t} = \frac{\cos^2 t - 1}{t(\cos t + 1)}
= -\frac{\sin t}{t}\cdot\frac{\sin t}{\cos t + 1}
\xrightarrow[t\to0]{} -1 \cdot \frac{0}{2} = 0. \qedhere
\]
\end{proof}

\begin{theorem}[Afgeleiden van sinus en cosinus]\label{thm:g12:trigo:derivatives}
De functies $\sin$ en $\cos$ zijn afleidbaar op $\R$, met
\[
(\sin)' = \cos, \qquad (\cos)' = -\sin .
\]
Algemener, $(\sin(ax+b))' = a\cos(ax+b)$ en
$(\cos(ax+b))' = -a\sin(ax+b)$.
\end{theorem}

\begin{proof}
Met de optelformules,
\[
\frac{\sin(x+h) - \sin x}{h}
= \sin x\,\frac{\cos h - 1}{h} + \cos x\,\frac{\sin h}{h}
\xrightarrow[h\to0]{} \sin x \cdot 0 + \cos x \cdot 1 = \cos x,
\]
met \cref{lem:g12:trigo:sinxx}. De berekening voor $\cos$ is identiek, en
de algemene vorm volgt uit de kettingregel.
\end{proof}

\begin{proposition}[Variaties]\label{prop:g12:trigo:variations}
Op $\intcc{0}{\pi}$ daalt $\cos$ van $1$ tot $-1$. Op
$\intcc{-\frac\pi2}{\frac\pi2}$ stijgt $\sin$ van $-1$ tot $1$. Beide
functies zijn $2\pi$-periodiek en hebben beeld $\intcc{-1}{1}$;
$\cos$ is even, $\sin$ is oneven.
\end{proposition}

\begin{proof}
Op $\intoo{0}{\pi}$ is $(\cos)' = -\sin < 0$ omdat het punt $M(t)$ daar
een positieve ordinaat heeft; evenzo $(\sin)' = \cos > 0$ op
$\intoo{-\frac\pi2}{\frac\pi2}$. Periodiciteit en pariteit komen uit
\cref{prop:g12:trigo:identities}.
\end{proof}

\begin{method}[$\cos x = a$ en $\sin x = a$ oplossen]\label{met:g12:trigo:equations}
Voor $a \in \intcc{-1}{1}$, vind een bijzondere oplossing $\alpha$
(uit de waardentabel of een rekenmachine). Dan zijn alle oplossingen:
\[
\cos x = \cos\alpha \iff x = \alpha + 2k\pi \text{ of } x = -\alpha + 2k\pi,
\quad k \in \Z;
\]
\[
\sin x = \sin\alpha \iff x = \alpha + 2k\pi \text{ of } x = \pi - \alpha + 2k\pi,
\quad k \in \Z.
\]
Voor ongelijkheden, lokaliseer de oplossingsbogen op de eenheidscirkel en
lees de intervallen af op één periode.
\end{method}

\begin{example}\label{ex:g12:trigo:equation}
Los $\cos x = \frac12$ op in $\intoc{-\pi}{\pi}$: een bijzondere
oplossing is $\frac\pi3$, dus de oplossingen zijn $x = \frac\pi3$ en
$x = -\frac\pi3$. In heel $\R$: $x = \pm\frac\pi3 + 2k\pi$,
$k \in \Z$.
\end{example}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g12:trigo:1}
Bereken $\cos\frac{2\pi}{3}$, $\sin\frac{5\pi}{6}$, $\cos\frac{7\pi}{4}$
en $\sin\left(-\frac{\pi}{3}\right)$ met de identiteiten van
\cref{prop:g12:trigo:identities}.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:trigo:2}
Differentieer $f(x) = \sin^2 x$, $g(x) = \cos(3x) + x\sin x$ en
$h(x) = \dfrac{\sin x}{2 + \cos x}$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:trigo:3}
Los op in $\intoc{-\pi}{\pi}$, daarna in $\R$:
\[
\text{(a) } \sin x = \frac{\sqrt3}{2};
\qquad
\text{(b) } \cos\left(2x\right) = \frac{\sqrt2}{2} .
\]
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:trigo:4}
Bereken met de optelformules de exacte waarde van
$\cos\frac{\pi}{12}$ en $\sin\frac{\pi}{12}$.
(Hint: $\frac{\pi}{12} = \frac{\pi}{3} - \frac{\pi}{4}$.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:trigo:5}
Los op in $\intcc{0}{2\pi}$ de ongelijkheid
$2\sin^2 x - \sin x - 1 \geq 0$.
(Hint: ontbind de trinomiaal in $\sin x$.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:trigo:6}
Bereken de limieten
\[
\lim_{x\to0} \frac{\sin 3x}{x}, \qquad
\lim_{x\to0} \frac{1 - \cos x}{x^2}, \qquad
\lim_{x\to+\infty} x \sin\frac{1}{x}.
\]
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:trigo:7}
Bestudeer de functie $f(x) = x + \cos x$ op $\intcc{0}{2\pi}$: variaties
en extrema. Toon aan dat $f$ stijgend is op $\R$ hoewel $f'$ in oneindig
veel punten verdwijnt.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:trigo:8}
Laat $f(x) = A\cos x + B \sin x$ met $(A,B) \neq (0,0)$.
\begin{enumerate}
  \item Toon aan dat $f$ geschreven kan worden als
        $f(x) = R\cos(x - \varphi)$ met $R = \sqrt{A^2 + B^2}$ en een
        geschikte $\varphi$.
  \item Leid het maximum en minimum van $f$ af, en los
        $\cos x + \sin x = 1$ op in $\intoc{-\pi}{\pi}$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:trigo:9}
Toon aan dat $\sin$ en $\cos$ beide de differentiaalvergelijking
$y'' = -y$ voldoen. Omgekeerd, laat $y$ een oplossing van $y'' = -y$ zijn
met $y(0) = a$ en $y'(0) = b$; toon aan dat $y = a\cos + b\sin$.
(Hint: beschouw $g = (y - a\cos - b\sin)$ en de ``energie''
$E = g^2 + (g')^2$.)
\end{exercise}
