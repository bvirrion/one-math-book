\chapter{Combinatoriek en tellen}\label{ch:g12:comb}

Combinatoriek is de kunst van tellen zonder opsommen. Haar twee
elementaire principes --- tel de groottes van disjuncte alternatieven op,
vermenigvuldig de aantallen onafhankelijke keuzes --- volstaan om de
rijen, permutaties en deelverzamelingen van een eindige verzameling te
tellen, en culmineren in de binomiaalstelling.

\section{De twee telprincipes}

We schrijven $\abs{E}$ voor het aantal elementen (de \emph{kardinaliteit})
van een eindige verzameling $E$.

\begin{proposition}[Optelprincipe]\label{prop:g12:comb:addition}
Als een eindige verzameling $E$ is opgedeeld in deelverzamelingen
$A_1, \dots, A_k$ (paarsgewijs disjunct, met vereniging $E$), dan
\[
\abs{E} = \abs{A_1} + \abs{A_2} + \dots + \abs{A_k}.
\]
\end{proposition}

\begin{proposition}[Vermenigvuldigingsprincipe]\label{prop:g12:comb:multiplication}
Als een object wordt opgebouwd door een opeenvolging van $k$ keuzes, met
$n_1$ opties voor de eerste keuze en, \emph{ongeacht de vorige keuzes},
$n_i$ opties voor de $i$-de, dan is het aantal opgebouwde objecten
$n_1 \times n_2 \times \dots \times n_k$.
\end{proposition}

\begin{proof}
Beide uitspraken worden bewezen door inductie op $k$; het geval $k = 2$
van de tweede komt neer op het tellen van een rechthoekig rooster per rij.
\end{proof}

\begin{example}
Een restaurant biedt $4$ voorgerechten, $6$ hoofdgerechten, $3$ desserts:
$4 \times 6 \times 3 = 72$ verschillende driemenu's.
\end{example}

\section{Tuples, permutaties, factorialen}

\begin{definition}[$k$-tupels]\label{def:g12:comb:tuples}
Een \emph{$k$-tupel} van een verzameling $E$ is een geordende lijst
$(x_1, \dots, x_k)$ van elementen van $E$, herhalingen toegestaan. Een
$k$-tupel van \emph{verschillende} elementen is een
\emph{arrangement}\index{arrangement} van $k$ elementen van $E$.
\end{definition}

\begin{proposition}\label{prop:g12:comb:tuples}
Laat $\abs E = n$. Het aantal $k$-tupels van $E$ is $n^k$. Het aantal
arrangementen van $k$ elementen van $E$ ($0 \leq k \leq n$) is
\[
n(n-1)(n-2)\cdots(n-k+1) = \frac{n!}{(n-k)!},
\]
waar $n! = 1 \times 2 \times \dots \times n$ (en $0! = 1$) de
\emph{faculteit}\index{faculteit} van $n$ is.
\end{proposition}

\begin{proof}
Vermenigvuldigingsprincipe: voor een $k$-tupel zijn er $n$ opties bij elk
van de $k$ stappen; voor een arrangement, $n$ opties voor $x_1$, daarna
$n - 1$ voor $x_2$ (één element is gebruikt), \ldots, $n - k + 1$ voor
$x_k$.
\end{proof}

\begin{definition}[Permutatie]\label{def:g12:comb:permutation}
Een \emph{permutatie}\index{permutatie} van $E$ is een arrangement van
alle $n$ elementen van $E$: een ordening van $E$. Volgens
\cref{prop:g12:comb:tuples} (geval $k = n$) is het aantal permutaties van
een $n$-elementenverzameling $n!$.
\end{definition}

\begin{example}
Vijf hardlopers kunnen een race in $5! = 120$ verschillende volgordes
finishen. Het aantal mogelijke podia (eerste drie plaatsen) is
$5 \times 4 \times 3 = 60$.
\end{example}

\section{Combinaties en binomiaalcoëfficiënten}

\begin{definition}[Combinaties]\label{def:g12:comb:combination}
Een \emph{combinatie}\index{combinatie} van $k$ elementen van $E$ is een
deelverzameling van $E$ met $k$ elementen (geen volgorde, geen herhaling).
Hun aantal wordt geschreven $\dbinom{n}{k}$\index{binomiaalcoëfficiënt},
gelezen ``$n$ over $k$'' of ``$n$ kies $k$''.
\end{definition}

\begin{theorem}\label{thm:g12:comb:binomformula}
Voor $0 \leq k \leq n$:
\[
\binom{n}{k} = \frac{n!}{k!\,(n-k)!} .
\]
\end{theorem}

\begin{proof}
Tel de arrangementen van $k$ elementen van $E$ op twee manieren. Direct:
$\frac{n!}{(n-k)!}$. Alternatief: kies eerst de onderliggende
deelverzameling ($\binom nk$ manieren), orden haar daarna ($k!$ manieren);
het vermenigvuldigingsprincipe geeft $\binom{n}{k}\,k!$. Gelijkstellen:
$\binom nk = \frac{n!}{k!(n-k)!}$.
\end{proof}

\begin{proposition}[Basisidentiteiten]\label{prop:g12:comb:identities}
Voor $0 \leq k \leq n$:
\[
\binom{n}{0} = \binom{n}{n} = 1, \qquad
\binom{n}{1} = n, \qquad
\binom{n}{k} = \binom{n}{n-k},
\]
en de \emph{regel van Pascal}\index{driehoek van Pascal}: voor
$1 \leq k \leq n-1$,
\[
\binom{n}{k} = \binom{n-1}{k-1} + \binom{n-1}{k}.
\]
\end{proposition}

\begin{proof}
De symmetrie $\binom nk = \binom{n}{n-k}$ geldt omdat complementen
$k$-elementen-deelverzamelingen één-op-één matchen met
$(n-k)$-elementen-deelverzamelingen. Voor de regel van Pascal: fixeer een
element $a \in E$ en sorteer de $k$-elementen-deelverzamelingen in die
die $a$ bevatten --- verkregen door $a$ toe te voegen aan een
$(k-1)$-elementen-deelverzameling van $E \setminus \{a\}$, waarvan er
$\binom{n-1}{k-1}$ zijn --- en die die $a$ vermijden, die de
$k$-elementen-deelverzamelingen van $E \setminus \{a\}$ zijn, in aantal
$\binom{n-1}{k}$. Besluit met het optelprincipe.
\end{proof}

De regel van Pascal genereert de coëfficiënten rij voor rij --- de
\emph{driehoek van Pascal}: elke ingang is de som van de twee erboven.

\begin{theorem}[Binomiaalstelling]\label{thm:g12:comb:binomial}
\index{binomiaalstelling}
Voor alle $a, b \in \R$ (of $\C$) en $n \in \N$:
\[
(a+b)^n = \sum_{k=0}^{n} \binom{n}{k}\, a^{k}\, b^{\,n-k} .
\]
\end{theorem}

\begin{proof}
Ontwikkel het product $(a+b)(a+b)\cdots(a+b)$ ($n$ factoren): elke term
van de ontwikkeling kiest $a$ of $b$ in elke factor, en produceert
$a^k b^{n-k}$ waar $k$ het aantal factoren is die $a$ leveren. Het aantal
manieren om deze $k$ factoren onder $n$ te kiezen is $\binom nk$, dat
daarom de coëfficiënt van $a^k b^{n-k}$ is.
\end{proof}

\begin{corollary}\label{cor:g12:comb:sums}
$\displaystyle\sum_{k=0}^{n} \binom{n}{k} = 2^n$
\quad en \quad
$\displaystyle\sum_{k=0}^{n} (-1)^k\binom{n}{k} = 0$ \ ($n \geq 1$).
\end{corollary}

\begin{proof}
Neem $a = b = 1$, daarna $a = -1$, $b = 1$ in de binomiaalstelling. De
eerste identiteit heeft ook een directe betekenis: een
$n$-elementenverzameling heeft $2^n$ deelverzamelingen (elk element is in
of uit: vermenigvuldigingsprincipe), gesorteerd naar grootte.
\end{proof}

\begin{method}[Het juiste model kiezen]\label{met:g12:comb:model}
Voor het tellen: beantwoord twee vragen: \emph{speelt volgorde een rol?}
en \emph{zijn herhalingen toegestaan?}
\begin{center}
\begin{tabular}{l|cc}
 & volgorde telt & volgorde irrelevant\\
\hline
herhalingen toegestaan & $n^k$ (tupels) & (universiteit)\\
geen herhalingen & $\frac{n!}{(n-k)!}$ (arrangementen) & $\binom nk$ (deelverz.)
\end{tabular}
\end{center}
Ballen trekken uit een urn: \emph{met teruglegging, in volgorde} $\to$
tupels; \emph{zonder teruglegging, in volgorde} $\to$ arrangementen;
\emph{een handvol in één keer} $\to$ deelverzamelingen.
\end{method}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g12:comb:1}
Een nummerplaat bestaat uit $2$ letters (A--Z), dan $3$ cijfers, dan
$2$ letters. Hoeveel platen zijn mogelijk? Hoeveel hebben geen herhaald
teken?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:comb:2}
Bereken $\dbinom{8}{3}$, $\dbinom{10}{8}$, en vereenvoudig
$\dfrac{\binom{n}{2}}{\binom{n+1}{2}}$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:comb:3}
In een klas van $30$ leerlingen moet men een commissie van $4$ leerlingen
kiezen, daarna een voorzitter en een penningmeester binnen de commissie
(één persoon kan niet beide functies bekleden). Hoeveel uitkomsten zijn
mogelijk?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:comb:4}
Ontwikkel $(x + 2)^5$ en $(1 - x)^6$ met de binomiaalstelling. Wat is de
coëfficiënt van $x^3$ in $(2x + 3)^7$?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:comb:5}
Een standaard pokerhand bestaat uit $5$ kaarten uit een speelkaartendeck
van $52$.
\begin{enumerate}
  \item Hoeveel handen zijn er?
  \item Hoeveel handen bevatten precies één aas? Minstens één aas?
  \item Hoeveel handen zijn ``full houses'' (drie kaarten van één rang,
        twee van een andere)?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:comb:6}
Hoeveel anagrammen (herschikkingen van letters, zinvol of niet) heeft het
woord MATH? Het woord BANANA?
(Hint voor BANANA: plaats eerst de drie A's.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:comb:7}
Bewijs de identiteit $k\dbinom{n}{k} = n\dbinom{n-1}{k-1}$
($1 \leq k \leq n$) op twee manieren: met de factoriale formule, en door
op twee manieren de paren (commissie van $k$ personen, haar voorzitter)
te tellen gekozen uit $n$ personen.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:comb:8}
Een pad in het vlak gaat van $(0,0)$ naar $(m, n)$ met eenheidstappen Oost
of Noord. Toon dat het aantal zulke paden $\dbinom{m+n}{m}$ is.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:comb:9}
Bewijs de \emph{identiteit van Vandermonde}: voor $0 \leq k \leq m + n$,
\[
\binom{m+n}{k} = \sum_{j=0}^{k} \binom{m}{j}\binom{n}{k-j},
\]
door de $k$-elementen-deelverzamelingen te tellen van een verzameling
opgedeeld in een groep van $m$ en een groep van $n$. Leid af dat
$\displaystyle\sum_{j=0}^{n}\binom{n}{j}^{\!2} = \binom{2n}{n}$.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:comb:10}
Met de binomiaalstelling, toon dat voor alle $n \geq 1$,
\[
\sum_{k=1}^{n} k \binom{n}{k} = n\,2^{n-1}.
\]
(Hint: differentieer ofwel $(1+x)^n$, of gebruik \cref{exo:g12:comb:7}.)
\end{exercise}
