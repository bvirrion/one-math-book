\chapter{Limieten en continuĂŻteit}\label{ch:g12:limcont}

Dit hoofdstuk breidt het limietbegrip uit van rijen naar functies van een
reĂ«le variabele, en introduceert continuĂŻteit. Het centrale resultaat is
de tussenwaardestelling, die kwalitatieve informatie (``$f$ is continu en
wisselt van teken'') omzet in het bestaan van oplossingen van
vergelijkingen.

\section{Limieten van functies}

Doorheen dit hoofdstuk is $f$ een functie gedefinieerd op een interval of
een vereniging van intervallen $D \subseteq \R$.

\begin{definition}[Limiet in het oneindige]\label{def:g12:limcont:liminf}
Laat $f$ gedefinieerd zijn op een interval $\intco{a}{+\infty}$ en
$\ell \in \R$. We zeggen dat $f$ \emph{neigt naar $\ell$ in
$+\infty$}\index{limiet!van een functie}, genoteerd
$\lim\limits_{x\to+\infty} f(x) = \ell$, als elk open interval dat
$\ell$ bevat alle waarden $f(x)$ bevat voor $x$ groot genoeg: voor elke
$\varepsilon > 0$ bestaat er $A$ zodat voor alle $x \geq A$,
$\abs{f(x) - \ell} \leq \varepsilon$.

We zeggen dat $f$ neigt naar $+\infty$ in $+\infty$ als voor elke
$M \in \R$ er een $A$ bestaat zodat $f(x) \geq M$ voor alle $x \geq A$.
Limieten in $-\infty$ en limieten gelijk aan $-\infty$ worden analoog
gedefinieerd.
\end{definition}

\begin{definition}[Limiet in een punt]\label{def:g12:limcont:limpoint}
Laat $a \in \R$ en laat $f$ gedefinieerd zijn nabij $a$ (behalve
eventueel in $a$). We zeggen dat $f$ neigt naar $\ell \in \R$ in $a$,
genoteerd $\lim\limits_{x \to a} f(x) = \ell$, als voor elke
$\varepsilon > 0$ er een $\delta > 0$ bestaat zodat voor alle $x \in D$
met $\abs{x - a} \leq \delta$, $\abs{f(x) - \ell} \leq \varepsilon$.

We zeggen dat $f$ neigt naar $+\infty$ in $a$ als voor elke $M$ er een
$\delta > 0$ is zodat $f(x) \geq M$ wanneer $x \in D$ en
$\abs{x - a} \leq \delta$. Eenzijdige limieten ($x \to a^+$, $x \to a^-$)
beperken $x$ tot $x > a$ of $x < a$.
\end{definition}

\begin{example}
$\lim\limits_{x\to+\infty} \frac{1}{x} = 0$,
$\lim\limits_{x\to 0^+} \frac{1}{x} = +\infty$,
$\lim\limits_{x\to 0^-} \frac{1}{x} = -\infty$. De functie
$x \mapsto \frac1x$ heeft geen (tweezijdige) limiet in $0$.
\end{example}

\begin{definition}[Asymptoten]\label{def:g12:limcont:asymptote}
De lijn $y = \ell$ is een \emph{horizontale asymptoot}\index{asymptoot}
van de kromme van $f$ in $+\infty$ (resp.\ $-\infty$) als
$\lim_{x\to+\infty} f(x) = \ell$ (resp.\ in $-\infty$). De lijn $x = a$
is een \emph{verticale asymptoot} als $f$ een oneindige eenzijdige limiet
heeft in $a$.
\end{definition}

\begin{proposition}[Bewerkingen op limieten]\label{prop:g12:limcont:operations}
De regels van \cref{prop:g12:seq:operations} (sommen, producten,
quotiĂ«nten) gelden letterlijk voor limieten van functies, in een punt of
in het oneindige, met dezelfde vier onbepaalde vormen
$(+\infty)+(-\infty)$, $0\times\infty$, $\frac{\infty}{\infty}$,
$\frac{0}{0}$.
\end{proposition}

\begin{proof}
De bewijzen zijn identiek aan het rij-geval, waarbij ``voor $n \geq N$''
vervangen wordt door ``voor $x \geq A$'' of ``voor
$\abs{x-a} \leq \delta$''.
\end{proof}

\begin{theorem}[Limiet van een samenstelling]\label{thm:g12:limcont:composition}
Laat $f, g$ functies zijn en $a, b, c$ reĂ«le getallen of $\pm\infty$. Als
$\lim\limits_{x \to a} f(x) = b$ en $\lim\limits_{y \to b} g(y) = c$, dan
\[
\lim_{x \to a} g\bigl(f(x)\bigr) = c .
\]
Dezelfde uitspraak geldt met een rij in plaats van $f$: als
$u_n \to b$ en $\lim_{y\to b} g(y) = c$, dan $g(u_n) \to c$.
\end{theorem}

\begin{proof}
Neem $b, c$ eindig (de andere gevallen zijn analoog). Laat
$\varepsilon > 0$. Er is $\delta > 0$ zodat $\abs{y - b} \leq \delta$
impliceert $\abs{g(y) - c} \leq \varepsilon$; en er is $\delta' > 0$
zodat $\abs{x - a} \leq \delta'$ impliceert $\abs{f(x) - b} \leq \delta$.
Aaneenschakelen geeft $\abs{g(f(x)) - c} \leq \varepsilon$ voor
$\abs{x - a} \leq \delta'$.
\end{proof}

\begin{theorem}[Vergelijking en insluiting]\label{thm:g12:limcont:squeeze}
Laat $f, g, h$ functies zijn en $a \in \R \cup \{\pm\infty\}$.
\begin{enumerate}
  \item Als $f \leq g$ nabij $a$ en $f(x) \to +\infty$ als $x \to a$, dan
        $g(x) \to +\infty$.
  \item Als $f \leq g \leq h$ nabij $a$ en $f, h$ neigen naar dezelfde
        eindige limiet $\ell$ in $a$, dan $g(x) \to \ell$ als $x \to a$.
\end{enumerate}
\end{theorem}

\begin{proof}
Zelfde argument als voor rijen (\cref{thm:g12:seq:squeeze}).
\end{proof}

\begin{method}[Limieten in de praktijk berekenen]\label{met:g12:limcont:limits}
\begin{itemize}
  \item Polynomen en rationale functies in $\pm\infty$: factoriseer de
        hoogste macht van $x$ uit; de limiet is die van de verhouding
        van de leidende termen.
  \item $\frac{0}{0}$ in een punt $a$: factoriseer $(x-a)$ uit teller en
        noemer, of herken een afgeleide
        $\lim_{x\to a}\frac{f(x)-f(a)}{x-a} = f'(a)$
        (\cref{ch:g12:deriv}).
  \item Vierkantswortels: vermenigvuldig met de toegevoegde uitdrukking.
  \item Oscillerende factoren ($\cos$, $\sin$): gebruik de
        insluitstelling.
\end{itemize}
\end{method}

\section{ContinuĂŻteit}

\begin{definition}[ContinuĂŻteit]\label{def:g12:limcont:continuity}
Laat $f$ gedefinieerd zijn op een interval $I$ en $a \in I$. De functie
$f$ is \emph{continu in $a$}\index{continuĂŻteit} als
$\lim\limits_{x \to a} f(x) = f(a)$. Ze is \emph{continu op $I$} als ze
continu is in elk punt van $I$.
\end{definition}

IntuĂŻtief kan de grafiek van een continue functie op een interval
getekend worden zonder de pen op te tillen.

\begin{proposition}[ContinuĂŻteit van gebruikelijke functies]\label{prop:g12:limcont:usual}
Polynomen, rationale functies, $x \mapsto \sqrt{x}$, $x \mapsto \abs{x}$,
$\cos$, $\sin$, $\exp$ en $\ln$ zijn continu op elk interval bevat in hun
domein. Sommen, producten, quotiĂ«nten (waar gedefinieerd) en
samenstellingen van continue functies zijn continu.
\end{proposition}

\begin{proof}[Gedeeltelijk bewijs]
Stabiliteit onder sommen, producten, quotiĂ«nten en samenstelling volgt
uit de bijbehorende limietstellingen. ContinuĂŻteit van $x \mapsto x$ en
van constante functies is onmiddellijk uit de definitie, dus polynomen en
rationale functies zijn continu. ContinuĂŻteit van de overige
gebruikelijke functies wordt bewezen in de hoofdstukken waar ze
geconstrueerd worden, of toegeven.
\end{proof}

\begin{example}
De \emph{entierfunctie} $x \mapsto \floor{x}$ (het grootste gehele
getal $\leq x$) is niet continu in enig geheel getal $n$: haar
linkerlimiet daar is $n - 1$ terwijl $\floor{n} = n$.
\end{example}

\begin{proposition}[Rijen en continuĂŻteit]\label{prop:g12:limcont:seqcont}
Als $u_n \to \ell$ en $f$ continu is in $\ell$, dan $f(u_n) \to f(\ell)$.
In het bijzonder, als een rij gedefinieerd door $u_{n+1} = f(u_n)$
convergeert naar $\ell$ en $f$ continu is in $\ell$, dan
$f(\ell) = \ell$.
\end{proposition}

\begin{proof}
De eerste claim is \cref{thm:g12:limcont:composition} toegepast met de
rij $(u_n)$. Voor de tweede, ga naar de limiet aan beide kanten van
$u_{n+1} = f(u_n)$: de linkerkant neigt naar $\ell$, de rechterkant naar
$f(\ell)$, en limieten zijn uniek.
\end{proof}

\section{De tussenwaardestelling}

\begin{theorem}[Tussenwaardestelling]\label{thm:g12:limcont:ivt}
\index{tussenwaardestelling}
Laat $f$ continu zijn op $\intcc{a}{b}$ en laat $k$ een willekeurig
getal zijn tussen $f(a)$ en $f(b)$. Dan bestaat er $c \in \intcc{a}{b}$
zodat $f(c) = k$.
\end{theorem}

{\small De tussenwaardestelling: een continue kromme die $(a, f(a))$ met
$(b, f(b))$ verbindt, moet elke horizontale lijn $y = k$ tussen $f(a)$
en $f(b)$ kruisen.}
\end{omfigure}

\begin{proof}
Door $f$ te vervangen door $f - k$ (en door $k - f$ indien nodig), mogen
we aannemen dat $f(a) \leq 0 \leq f(b)$ en zoeken naar een nulpunt van
$f$. We construeren twee rijen door \emph{dichotomie}\index{dichotomie}:
stel $a_0 = a$, $b_0 = b$; gegeven $a_n \leq b_n$ met
$f(a_n) \leq 0 \leq f(b_n)$, laat $m = \frac{a_n + b_n}{2}$ en definieer
\[
(a_{n+1}, b_{n+1}) =
\begin{cases}
(a_n, m) & \text{als } f(m) \geq 0,\\
(m, b_n) & \text{als } f(m) < 0.
\end{cases}
\]
In beide gevallen $f(a_{n+1}) \leq 0 \leq f(b_{n+1})$, de rij $(a_n)$ is
stijgend, $(b_n)$ is dalend, en
$b_n - a_n = \frac{b-a}{2^n} \to 0$: de rijen zijn aangrenzend (zie
\cref{exo:g12:seq:10}) en convergeren naar een gemeenschappelijke limiet
$c \in \intcc{a}{b}$. Door continuĂŻteit, $f(a_n) \to f(c)$ en
$f(b_n) \to f(c)$; naar de limiet gaan in $f(a_n) \leq 0$ en
$f(b_n) \geq 0$ geeft $f(c) \leq 0 \leq f(c)$, dus $f(c) = 0$.
\end{proof}

\begin{remark}
Het dichotomiebewijs is een algoritme: het levert willekeurig precieze
benaderingen van een oplossing, met de fout bij elke stap gehalveerd.
Zo kan een computer $f(x) = 0$ oplossen met alleen de mogelijkheid om
$f$ te evalueren.
\end{remark}

\begin{theorem}[Bijectiestelling]\label{thm:g12:limcont:bijection}
Laat $f$ continu en \emph{strikt monotoon} zijn op een interval $I$. Dan
heeft voor elke $k$ strikt tussen de limieten van $f$ in de eindpunten
van $I$ de vergelijking $f(x) = k$ een \emph{unieke} oplossing in $I$.
\end{theorem}

\begin{proof}
Bestaan: kies $a, b \in I$ met $k$ tussen $f(a)$ en $f(b)$ (mogelijk
omdat de waarden van $f$ haar eindpuntlimieten naderen), en pas de
tussenwaardestelling toe op $\intcc{a}{b}$. Uniciteit: als $x < y$,
geeft strikte monotonie $f(x) \neq f(y)$, dus twee verschillende punten
kunnen niet beide oplossingen zijn.
\end{proof}

\begin{method}[$f(x) = k$ oplossen met de bijectiestelling]\label{met:g12:limcont:bijection}
Om te bewijzen dat $f(x) = k$ precies Ă©Ă©n oplossing heeft in een
interval:
\begin{enumerate}
  \item stel een veranderingstabel van $f$ op (teken van $f'$, zie
        \cref{ch:g12:deriv});
  \item op elk interval waar $f$ continu en strikt monotoon is, vergelijk
        $k$ met de eindpuntwaarden of limieten;
  \item pas de bijectiestelling toe op elk zulk interval en tel de
        oplossingen.
\end{enumerate}
Een rekenmachine of dichotomie lokaliseert daarna elke oplossing zo fijn
als gewenst.
\end{method}

\begin{example}\label{ex:g12:limcont:cubic}
De vergelijking $x^3 + x = 1$ heeft een unieke reĂ«le oplossing.
Inderdaad is $f(x) = x^3 + x$ continu en strikt stijgend op $\R$ (als
som van strikt stijgende functies), $\lim_{-\infty} f = -\infty$ en
$\lim_{+\infty} f = +\infty$, dus past de bijectiestelling met $k = 1$.
Omdat $f(0.6) = 0.816 < 1$ en $f(0.7) = 1.043 > 1$, ligt de oplossing in
$\intoo{0.6}{0.7}$.
\end{example}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g12:limcont:1}
Bereken de volgende limieten:
\[
\lim_{x\to+\infty} \frac{2x^3 - x + 1}{5x^3 + x^2}, \qquad
\lim_{x\to-\infty} \bigl(x^2 + 3x - 1\bigr), \qquad
\lim_{x\to 2^+} \frac{x+1}{x-2}, \qquad
\lim_{x\to+\infty} \frac{\cos x}{x}.
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:limcont:2}
Laat $f(x) = \dfrac{2x^2 - 3x + 1}{x - 1}$, gedefinieerd voor $x \neq 1$.
\begin{enumerate}
  \item Toon aan dat $f(x) = 2x - 1$ voor alle $x \neq 1$.
  \item Leid $\lim\limits_{x \to 1} f(x)$ af. Kan $f$ uitgebreid worden
        tot een functie continu in $1$?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:limcont:3}
Bepaal de asymptoten van de kromme van
$f(x) = \dfrac{3x - 1}{x + 2}$ en schets de kromme.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:limcont:4}
Bereken
\[
\lim_{x\to+\infty} \bigl(\sqrt{x^2 + x} - x\bigr)
\qquad\text{en}\qquad
\lim_{x\to 0} \frac{\sqrt{1+x} - 1}{x}.
\]
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:limcont:5}
Laat $f(x) = x + \sin x$.
\begin{enumerate}
  \item Toon aan dat $f(x) \to +\infty$ als $x \to +\infty$, hoewel $f$
        niet monotoon is op enig interval $\intco{A}{+\infty}$.
  \item Toon aan dat $g(x) = \dfrac{x + \sin x}{x - \sin x}$ neigt naar
        $1$ in $+\infty$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:limcont:6}
Toon aan dat de vergelijking $x^3 - 3x + 1 = 0$ precies drie reĂ«le
oplossingen heeft, en geef voor elk een interval van lengte $1$ dat haar
bevat. (Hint: bestudeer de variaties van $f(x) = x^3 - 3x + 1$; haar
afgeleide is $3x^2 - 3$.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:limcont:7}
Laat $f \colon \intcc{0}{1} \to \intcc{0}{1}$ continu zijn. Toon aan dat
$f$ een \emph{vastpunt} heeft: er bestaat $c \in \intcc{0}{1}$ met
$f(c) = c$. (Hint: pas de tussenwaardestelling toe op
$g(x) = f(x) - x$.)
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:limcont:8}
Een wandelaar beklimt een bergpad, start om 8:00 en arriveert om 20:00.
De volgende dag daalt ze hetzelfde pad, opnieuw startend om 8:00 en
arriverend om 20:00. Toon aan dat er een tijdstip van de dag is waarop
ze op precies hetzelfde punt van het pad is op beide dagen. (Hint:
introduceer het verschil van de twee positiefuncties.)
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:limcont:9}
Laat $f(x) = x^5 + x^3 + x - 1$.
\begin{enumerate}
  \item Toon aan dat $f$ een uniek reĂ«el nulpunt $\alpha$ heeft, en dat
        $\alpha \in \intoo{0}{1}$.
  \item Beschrijf een dichotomieprocedure die $\alpha$ tot op $10^{-6}$
        berekent, en geef het aantal iteraties nodig startend van
        $\intcc{0}{1}$.
\end{enumerate}
\end{exercise}
