\chapter{Differentiaalrekening en convexiteit}\label{ch:g12:deriv}

De \omterm{def:g12:deriv:derivative}{afgeleide} meet de ogenblikkelijke veranderingsverhouding van een
\omterm{def:g11:func:function}{functie}; het is de \omterm{def:g10:reffunc:affine}{richtingscoëfficiënt} van de \omterm{def:g12:deriv:tangent}{raaklijn} aan haar \omterm{def:g10:functions:graph}{grafiek}.
Dit hoofdstuk herhaalt en breidt de differentiatieregels uit, koppelt het
teken van $f'$ aan de variaties van $f$, en introduceert \omterm{def:g12:deriv:convex}{convexiteit},
die door de tweede \omterm{def:g12:deriv:derivative}{afgeleide} wordt gestuurd.

\section{De afgeleide}

\begin{definition}[Afgeleide in een punt]\label{def:g12:deriv:derivative}
Laat $f$ gedefinieerd zijn op een \omterm{def:g10:numbers:interval}{interval} $I$ en $a \in I$. De \omterm{def:g11:func:function}{functie}
$f$ is \emph{afleidbaar in $a$}\index{afgeleide}\index{afleidbaar} als het
verschilquotiënt
\[
\frac{f(a+h) - f(a)}{h}
\]
een eindige limiet heeft als $h \to 0$. Deze limiet heet de
\emph{afgeleide van $f$ in $a$} en wordt genoteerd $f'(a)$. Als $f$
afleidbaar is in elk punt van $I$, is de \omterm{def:g11:func:function}{functie}
$f' \colon x \mapsto f'(x)$ de \emph{afgeleide} van $f$.
\end{definition}

\begin{definition}[Raaklijn]\label{def:g12:deriv:tangent}
Als $f$ \omterm{def:g12:deriv:derivative}{afleidbaar} is in $a$, is de \emph{raaklijn}\index{raaklijn} aan
de kromme van $f$ in het punt $(a, f(a))$ de lijn met \omterm{def:g10:algebra:equation}{vergelijking}
\[
y = f(a) + f'(a)(x - a).
\]
\end{definition}

\begin{proposition}\label{prop:g12:deriv:diffcont}
Een \omterm{def:g11:func:function}{functie} \omterm{def:g12:deriv:derivative}{afleidbaar} in $a$ is \omterm{def:g12:limcont:continuity}{continu} in $a$. Het omgekeerde is
onwaar.
\end{proposition}

\begin{proof}
Voor $h \neq 0$ klein is
$f(a+h) - f(a) = h \cdot \frac{f(a+h)-f(a)}{h}$. Als $h \to 0$, neigt
het rechterlid naar $0 \times f'(a) = 0$, dus $f(a+h) \to f(a)$. Voor
het omgekeerde is $x \mapsto \abs{x}$ \omterm{def:g12:limcont:continuity}{continu} in $0$, maar haar
verschilquotiënt in $0$ is gelijk aan $\frac{\abs h}{h} = \pm 1$
afhankelijk van het teken van $h$, en heeft geen limiet.
\end{proof}

\subsection{Differentiatieregels}

\begin{proposition}[Bewerkingen]\label{prop:g12:deriv:operations}
Laat $u, v$ \omterm{def:g12:deriv:derivative}{afleidbaar} zijn op $I$ en $\lambda \in \R$. Dan zijn
$u + v$, $\lambda u$, $uv$ \omterm{def:g12:deriv:derivative}{afleidbaar} op $I$, evenals $\frac{u}{v}$ waar
$v \neq 0$, en
\[
(u+v)' = u' + v', \quad (\lambda u)' = \lambda u', \quad
(uv)' = u'v + uv', \quad
\left(\frac{u}{v}\right)' = \frac{u'v - uv'}{v^2}.
\]
\end{proposition}

\begin{proof}[Bewijs van de productregel]
Schrijf het verschilquotiënt van $uv$ in $a$ als
\[
\frac{u(a+h)v(a+h) - u(a)v(a)}{h}
= \frac{u(a+h) - u(a)}{h}\,v(a+h) + u(a)\,\frac{v(a+h) - v(a)}{h}.
\]
Als $h \to 0$, $v(a+h) \to v(a)$ door \omterm{def:g12:limcont:continuity}{continuïteit}
(\cref{prop:g12:deriv:diffcont}), dus neigt het rechterlid naar
$u'(a)v(a) + u(a)v'(a)$. De andere regels worden evenzo bewezen.
\end{proof}

\begin{theorem}[Kettingregel]\label{thm:g12:deriv:chain}
\index{kettingregel}
Laat $u$ \omterm{def:g12:deriv:derivative}{afleidbaar} zijn op $I$ met waarden in $J$, en $g$ \omterm{def:g12:deriv:derivative}{afleidbaar} op
$J$. Dan is $g \circ u$ \omterm{def:g12:deriv:derivative}{afleidbaar} op $I$ en
\[
(g \circ u)' = (g' \circ u) \cdot u' .
\]
In het bijzonder, voor $u$ \omterm{def:g12:deriv:derivative}{afleidbaar}:
\[
(u^n)' = n\,u'\,u^{n-1} \ (n \in \Z), \qquad
\left(\sqrt{u}\right)' = \frac{u'}{2\sqrt{u}} \ (u > 0), \qquad
\left(\eu^{u}\right)' = u'\,\eu^{u}.
\]
\end{theorem}

\begin{proof}[Bewijsschets]
Wanneer $u(a+h) \neq u(a)$ voor $h$ klein, schrijf
\[
\frac{g(u(a+h)) - g(u(a))}{h}
= \frac{g(u(a+h)) - g(u(a))}{u(a+h) - u(a)} \cdot \frac{u(a+h) - u(a)}{h}.
\]
Als $h \to 0$, $u(a+h) \to u(a)$, dus neigt de eerste factor naar
$g'(u(a))$ en de tweede naar $u'(a)$. (Een volledig bewijs moet het geval
behandelen waarin $u(a+h) = u(a)$ voor willekeurig kleine $h$; dit
technische punt wordt op de universiteit behandeld.)
\end{proof}

\begin{example}
De gebruikelijke \omterm{def:g12:deriv:derivative}{afgeleiden}, geldig op de natuurlijke \omterm{def:g11:func:function}{domeinen}:
\[
(x^n)' = nx^{n-1}, \quad
\left(\frac{1}{x}\right)' = -\frac{1}{x^2}, \quad
(\sqrt{x})' = \frac{1}{2\sqrt{x}}, \quad
(\cos x)' = -\sin x, \quad
(\sin x)' = \cos x .
\]
De eerste wordt door inductie bewezen uit de productregel, en de laatste
twee in \cref{ch:g12:trigo}.
\end{example}

\section{Variaties en extrema}

\begin{theorem}[Teken van de afgeleide en variaties]\label{thm:g12:deriv:variations}
Laat $f$ \omterm{def:g12:deriv:derivative}{afleidbaar} zijn op een \omterm{def:g10:numbers:interval}{interval} $I$.
\begin{enumerate}
  \item $f$ is \omterm{def:g12:seq:monotonic}{stijgend} op $I$ dan en slechts dan als $f' \geq 0$ op $I$.
  \item $f$ is \omterm{def:g10:functions:variations}{dalend} op $I$ dan en slechts dan als $f' \leq 0$ op $I$.
  \item Als $f' > 0$ op $I$ behalve in eindig veel punten waar ze
        verdwijnt, dan is $f$ strikt \omterm{def:g12:seq:monotonic}{stijgend} op $I$.
\end{enumerate}
\end{theorem}

\begin{proof}[Gedeeltelijk bewijs]
Als $f$ \omterm{def:g12:seq:monotonic}{stijgend} is, is elk verschilquotiënt
$\frac{f(a+h)-f(a)}{h}$ $\geq 0$, en naar de limiet gaan geeft
$f'(a) \geq 0$. De omgekeerde implicaties steunen op de
middelwaardestelling, bewezen op de universiteit; ze zijn
\emph{toegegeven op dit niveau}.
\end{proof}

\begin{definition}[Lokaal extremum]\label{def:g12:deriv:extremum}
$f$ heeft een \emph{lokaal maximum}\index{extremum} in $a$ als
$f(x) \leq f(a)$ voor alle $x$ nabij $a$; lokale minima worden
symmetrisch gedefinieerd.
\end{definition}

\begin{proposition}[Eerste-orde-voorwaarde]\label{prop:g12:deriv:critical}
Als $f$ \omterm{def:g12:deriv:derivative}{afleidbaar} is op een \emph{open} \omterm{def:g10:numbers:interval}{interval} $I$ en een lokaal
\omterm{def:g12:deriv:extremum}{extremum} heeft in $a \in I$, dan $f'(a) = 0$. Het omgekeerde is onwaar
($f(x) = x^3$ in $a = 0$).
\end{proposition}

\begin{proof}
Zeg $a$ is een \omterm{def:g12:deriv:extremum}{lokaal maximum}. Voor $h > 0$ klein,
$\frac{f(a+h)-f(a)}{h} \leq 0$, dus $h \to 0^+$ geeft $f'(a) \leq 0$;
voor $h < 0$ is het quotiënt $\geq 0$, wat $f'(a) \geq 0$ geeft. Dus
$f'(a) = 0$.
\end{proof}

\begin{method}[Veranderingstabel]\label{met:g12:deriv:table}
Om een \omterm{def:g11:func:function}{functie} $f$ te bestuderen: bepaal haar \omterm{def:g11:func:function}{domein} en limieten aan de
rand; bereken $f'$ en ontbind haar; bepaal het teken van $f'$ op elk
deelinterval; noteer alles in een \omterm{def:g10:functions:table}{veranderingstabel}, markeer extrema;
leid het aantal oplossingen van $f(x) = k$ af met de bijectiestelling
(\cref{thm:g12:limcont:bijection}).
\end{method}

\section{Convexiteit}

\begin{definition}[Convexe functie]\label{def:g12:deriv:convex}
Een \omterm{def:g11:func:function}{functie} $f$ gedefinieerd op een \omterm{def:g10:numbers:interval}{interval} $I$ is
\emph{convex}\index{convexiteit} op $I$ als elke koorde van haar \omterm{def:g10:functions:graph}{grafiek}
boven de \omterm{def:g10:functions:graph}{grafiek} ligt: voor alle $a, b \in I$ en $t \in \intcc{0}{1}$,
\[
f\bigl((1-t)a + t b\bigr) \leq (1-t) f(a) + t f(b).
\]
Ze is \emph{concaaf}\index{concaaf} als de omgekeerde ongelijkheid geldt ($-f$ convex).
\end{definition}

\begin{theorem}[Convexiteit en afgeleiden]\label{thm:g12:deriv:convexity}
Laat $f$ \omterm{def:g12:deriv:derivative}{afleidbaar} zijn op een \omterm{def:g10:numbers:interval}{interval} $I$. De volgende zijn
equivalent:
\begin{enumerate}
  \item $f$ is \omterm{def:g12:deriv:convex}{convex} op $I$;
  \item $f'$ is \omterm{def:g12:seq:monotonic}{stijgend} op $I$;
  \item de \omterm{def:g10:functions:graph}{grafiek} van $f$ ligt boven elk van haar \omterm{def:g12:deriv:tangent}{raaklijnen}.
\end{enumerate}
Als $f$ tweemaal \omterm{def:g12:deriv:derivative}{afleidbaar} is, zijn deze ook equivalent met
$f'' \geq 0$ op $I$.
\end{theorem}

\begin{proof}[Bewijs van (2) $\Rightarrow$ (3)]
Fixeer $a \in I$ en stel
$g(x) = f(x) - f(a) - f'(a)(x-a)$, de kloof tussen de kromme en de
\omterm{def:g12:deriv:tangent}{raaklijn} in $a$. Dan is $g$ \omterm{def:g12:deriv:derivative}{afleidbaar} en $g'(x) = f'(x) - f'(a)$. Als
$f'$ \omterm{def:g12:seq:monotonic}{stijgend} is, is $g' \leq 0$ op $I \cap \intoc{-\infty}{a}$ en
$g' \geq 0$ op $I \cap \intco{a}{+\infty}$: $g$ daalt vóór $a$ en stijgt
erna, dus bereikt $g$ haar \omterm{def:g10:functions:extrema}{minimum} $g(a) = 0$, vandaar $g \geq 0$.
De overige implicaties zijn toegegeven op dit niveau; de equivalentie
met $f'' \geq 0$ volgt uit \cref{thm:g12:deriv:variations} toegepast op
$f'$.
\end{proof}

\begin{definition}[Buigpunt]\label{def:g12:deriv:inflection}
Een punt waar de kromme van $f$ haar \omterm{def:g12:deriv:tangent}{raaklijn} kruist is een
\emph{buigpunt}\index{buigpunt}. Voor $f$ tweemaal \omterm{def:g12:deriv:derivative}{afleidbaar} zijn
buigpunten de punten waar $f''$ van teken wisselt.
\end{definition}

\begin{example}\label{ex:g12:deriv:cube}
$f(x) = x^3$ heeft $f''(x) = 6x$: $f$ is \omterm{def:g12:deriv:convex}{concaaf} op
$\intoc{-\infty}{0}$, \omterm{def:g12:deriv:convex}{convex} op $\intco{0}{+\infty}$, met een \omterm{def:g12:deriv:inflection}{buigpunt}
in de oorsprong, waar de kromme haar \omterm{def:g12:deriv:tangent}{raaklijn} (de $x$-as) kruist.
\end{example}

\begin{example}[Convexiteitsongelijkheden]\label{ex:g12:deriv:ineq}
De exponentiële is \omterm{def:g12:deriv:convex}{convex} op $\R$ (haar tweede \omterm{def:g12:deriv:derivative}{afgeleide} is zijzelf,
positief). Haar \omterm{def:g12:deriv:tangent}{raaklijn} in $0$ is $y = 1 + x$, dus
\[
\eu^x \geq 1 + x \quad \text{voor alle } x \in \R .
\]
Dergelijke \emph{raaklijnongelijkheden} zijn een standaardproduct van
\omterm{def:g12:deriv:convex}{convexiteit}.
\end{example}

\begin{method}[Convexiteit gebruiken]\label{met:g12:deriv:convexity}
\begin{itemize}
  \item Om een ongelijkheid $f(x) \geq ax + b$ te bewijzen: herken de
        lijn als een \omterm{def:g12:deriv:tangent}{raaklijn} van een convexe $f$ en roep
        \cref{thm:g12:deriv:convexity}(3) in.
  \item Om \omterm{def:g12:deriv:inflection}{buigpunten} te lokaliseren: los $f''(x) = 0$ op \emph{en}
        controleer dat $f''$ daar van teken wisselt.
  \item In een \omterm{def:g10:functions:table}{veranderingstabel} verfijnt \omterm{def:g12:deriv:convex}{convexiteit} de schets van de
        kromme (welke kant ze buigt).
\end{itemize}
\end{method}
