\chapter{Vectoren, lijnen en vlakken in de ruimte}\label{ch:g12:space}

Driedimensionale meetkunde wordt computationeel zodra \omterm{def:g11:vect:vector}{vectoren} en
\omterm{def:g10:coordgeom:system}{coördinaten} beschikbaar zijn: lijnen krijgen parametrische voorstellingen,
vlakken krijgen Cartesiaanse \omterm{def:g10:algebra:equation}{vergelijkingen}, en het \omterm{def:g12:space:dot}{inwendig product} meet
hoeken en afstanden. Dit hoofdstuk bouwt deze gereedschapskist en gebruikt
haar om de klassieke snij- en afstandsproblemen op te lossen.

\section{Vectoren in de ruimte}

\omterm{def:g11:vect:vector}{Vectoren} in de ruimte gehoorzamen dezelfde regels als in het vlak: ze
tellen op, ze schalen, en $\vect{AB} = \vect{CD}$ precies wanneer $ABDC$
een parallellogram is. In een \omterm{def:g10:coordgeom:system}{coördinatenstelsel}
$(O; \vec\imath, \vec\jmath, \vec k)$ heeft elke \omterm{def:g11:vect:vector}{vector} \omterm{def:g10:coordgeom:system}{coördinaten}
$(x, y, z)$.

\begin{definition}[Collineariteit, coplanariteit]\label{def:g12:space:collinear}
Twee \omterm{def:g11:vect:vector}{vectoren} zijn \emph{\omterm{def:g11:vect:collinear}{collineair}} als de ene een veelvoud van de andere
is. Drie \omterm{def:g11:vect:vector}{vectoren} $\vec u, \vec v, \vec w$ zijn
\emph{coplanair}\index{coplanair} als een van hen als combinatie van de
andere twee te schrijven is, zeg $\vec w = a\vec u + b\vec v$.
\end{definition}

\begin{proposition}\label{prop:g12:space:basis}
Als $\vec u, \vec v, \vec w$ niet \omterm{def:g12:space:collinear}{coplanair} zijn, ontbindt elke \omterm{def:g11:vect:vector}{vector} van
de ruimte uniek als $x\vec u + y\vec v + z\vec w$.
\end{proposition}

\begin{proof}[Bewijs-idee]
Door de top van de te \omterm{def:g10:algebra:expand}{ontbinden} \omterm{def:g11:vect:vector}{vector}, teken de lijn evenwijdig aan
$\vec w$; zij snijdt het vlak van $\vec u, \vec v$ in één punt, wat de
\omterm{def:g11:vect:vector}{vector} splitst in een component in dat vlak (uniek $x\vec u + y\vec v$,
vlakke meetkunde) en een component langs $\vec w$. Uniciteit: een verschil
van twee ontbindingen zou een van de \omterm{def:g11:vect:vector}{vectoren} in termen van de andere twee
uitdrukken, in strijd met non-coplanariteit.
\end{proof}

\section{Lijnen en vlakken}

\begin{definition}[Parametrische voorstelling van een lijn]\label{def:g12:space:line}
De lijn door $A(x_A, y_A, z_A)$ met \omterm{def:g11:vect:direction}{richtingsvector}
$\vec u (a, b, c) \neq \vec 0$ is de verzameling van punten
\[
M(x_A + ta,\; y_A + tb,\; z_A + tc), \qquad t \in \R .
\]
\end{definition}

\begin{definition}[Vlak]\label{def:g12:space:plane}
Het vlak door $A$ gericht door twee niet-collineaire \omterm{def:g11:vect:vector}{vectoren}
$\vec u, \vec v$ is de verzameling van punten $M$ met
$\vect{AM} = s\vec u + t\vec v$, $(s, t) \in \R^2$.
\end{definition}

\begin{proposition}[Relatieve posities]\label{prop:g12:space:positions}
Twee distincte vlakken zijn ofwel evenwijdig of snijden in een lijn. Een
lijn en een vlak zijn ofwel evenwijdig (eventueel bevat) of snijden in één
punt. Twee lijnen in de ruimte kunnen snijdend, strikt evenwijdig,
identiek, of \emph{kruisend} (niet \omterm{def:g12:space:collinear}{coplanair}) zijn.
\end{proposition}

\begin{proof}[Schets]
Dit zijn incidentie-uitspraken; elke gevalsanalyse reduceert, in
\omterm{def:g10:coordgeom:system}{coördinaten}, tot het oplossen van een \omterm{def:g10:lines:system}{lineair stelsel} en het tellen van
zijn oplossingen --- zie \cref{met:g12:space:intersections}.
\end{proof}

\section{Inwendig product}

\begin{definition}[Inwendig product]\label{def:g12:space:dot}
Het \emph{inwendig product}\index{inwendig product}\index{scalair product}
van $\vec u$ en $\vec v$ is
\[
\vec u \cdot \vec v = \tfrac12\left(\norm{\vec u + \vec v}^2
- \norm{\vec u}^2 - \norm{\vec v}^2\right),
\]
waar $\norm{\vec u}$ de lengte van $\vec u$ is. Als beide \omterm{def:g11:vect:vector}{vectoren}
niet-nul zijn, $\vec u \cdot \vec v = \norm{\vec u}\,\norm{\vec v}\cos\theta$
waar $\theta$ de hoek ertussen is; en in een \emph{\omterm{def:g10:coordgeom:system}{orthonormaal}}
\omterm{def:g10:coordgeom:system}{coördinatenstelsel},
\[
\vec u \cdot \vec v = xx' + yy' + zz' .
\]
Twee \omterm{def:g11:vect:vector}{vectoren} zijn \emph{loodrecht} (\omterm{def:g11:scal:orthogonal}{orthogonaal}) als hun inwendig product
$0$ is.
\end{definition}

\begin{proposition}[Eigenschappen]\label{prop:g12:space:dotprops}
Het \omterm{def:g12:space:dot}{inwendig product} is symmetrisch ($\vec u\cdot\vec v = \vec v\cdot\vec u$),
bilineair ($(a\vec u + b\vec v)\cdot \vec w = a\,\vec u\cdot\vec w +
b\,\vec v\cdot \vec w$), en
$\vec u \cdot \vec u = \norm{\vec u}^2 \geq 0$.
\end{proposition}

\begin{proof}
In een \omterm{def:g10:coordgeom:system}{orthonormaal stelsel} zijn alle drie de eigenschappen onmiddellijk op
de formule $xx' + yy' + zz'$, die zelf volgt uit de definiërende formule en
de Pythagoras-berekening van lengtes:
$\norm{\vec u}^2 = x^2 + y^2 + z^2$.
\end{proof}

\begin{definition}[Normaalvector]\label{def:g12:space:normal}
Een \emph{normaalvector}\index{normaalvector} van een vlak $\mathcal P$ is
een niet-nul \omterm{def:g11:vect:vector}{vector} loodrecht op elke \omterm{def:g11:vect:vector}{vector} die in $\mathcal P$ ligt (het
volstaat dat hij loodrecht is op twee niet-collineaire richtingen van
$\mathcal P$).
\end{definition}

\begin{theorem}[Cartesiaanse vergelijking van een vlak]\label{thm:g12:space:planeeq}
In een \omterm{def:g10:coordgeom:system}{orthonormaal} \omterm{def:g10:coordgeom:system}{coördinatenstelsel} heeft het vlak door $A$ met
\omterm{def:g12:space:normal}{normaalvector} $\vec n(a, b, c)$ de \omterm{def:g10:algebra:equation}{vergelijking}
\[
a x + b y + c z + d = 0,
\]
waar $d = -(ax_A + by_A + cz_A)$. Omgekeerd definieert elke zulke
\omterm{def:g10:algebra:equation}{vergelijking} met $(a,b,c) \neq (0,0,0)$ een vlak met \omterm{def:g12:space:normal}{normaalvector}
$(a, b, c)$.
\end{theorem}

\begin{proof}
$M \in \mathcal P \iff \vect{AM} \perp \vec n \iff
a(x - x_A) + b(y - y_A) + c(z - z_A) = 0$, wat tot de \omterm{def:g10:algebra:equation}{vergelijking}
ontwikkelt. Omgekeerd, kies enig oplossingspunt $A$ van de \omterm{def:g10:algebra:equation}{vergelijking};
dezelfde berekening achterwaarts toont dat de oplossingsverzameling
$\{M : \vect{AM}\cdot\vec n = 0\}$ is, het vlak door $A$ loodrecht op
$\vec n$.
\end{proof}

\begin{method}[Snijdingen in de praktijk]\label{met:g12:space:intersections}
\begin{itemize}
  \item \emph{Lijn $\cap$ vlak}: substitueer de parametrische \omterm{def:g10:coordgeom:system}{coördinaten}
        van de lijn in de vlakvergelijking; los op naar $t$ (één oplossing:
        één punt; $0 = $ niet-nul: evenwijdig; $0 = 0$: lijn bevat).
  \item \emph{Vlak $\cap$ vlak}: los het stelsel van de twee \omterm{def:g10:algebra:equation}{vergelijkingen}
        op, parametriseer met één vrije coördinaat; de oplossing is een
        lijn (of de vlakken zijn evenwijdig wanneer de \omterm{def:g12:space:normal}{normaalvectoren}
        \omterm{def:g11:vect:collinear}{collineair} zijn).
  \item \emph{Orthogonaliteitscontroles}: lijn $\perp$ vlak iff haar
        richting \omterm{def:g11:vect:collinear}{collineair} is met de normaal; twee vlakken zijn loodrecht
        iff hun normalen loodrecht zijn.
\end{itemize}
\end{method}

\begin{proposition}[Afstand van een punt tot een vlak]\label{prop:g12:space:distance}
\index{afstand!punt tot vlak}
In een \omterm{def:g10:coordgeom:system}{orthonormaal stelsel} is de afstand van $M_0(x_0, y_0, z_0)$ tot het
vlak $\mathcal P : ax + by + cz + d = 0$
\[
\operatorname{dist}(M_0, \mathcal P)
= \frac{\abs{ax_0 + by_0 + cz_0 + d}}{\sqrt{a^2 + b^2 + c^2}}.
\]
\end{proposition}

\begin{proof}
Laat $H$ de \omterm{def:g11:scal:orthogonal}{orthogonale} projectie van $M_0$ op $\mathcal P$ zijn: de voet
van de loodlijn, dus $\vect{HM_0}$ is \omterm{def:g11:vect:collinear}{collineair} met de eenheidsnormaal
$\frac{\vec n}{\norm{\vec n}}$, en de afstand is
$\abs{\vect{HM_0}\cdot \frac{\vec n}{\norm{\vec n}}}$. Voor elke
$H \in \mathcal P$,
$\vect{HM_0}\cdot\vec n = ax_0 + by_0 + cz_0 - (ax_H + by_H + cz_H)
= ax_0 + by_0 + cz_0 + d$ (met de \omterm{def:g10:algebra:equation}{vergelijking} in $H$), waaruit de formule
na deling door $\norm{\vec n} = \sqrt{a^2+b^2+c^2}$.
\end{proof}

\begin{example}\label{ex:g12:space:distance}
Afstand van de oorsprong tot het vlak $x + 2y + 2z - 6 = 0$:
$\frac{\abs{-6}}{\sqrt{1 + 4 + 4}} = \frac63 = 2$.
\end{example}

\section{Oefeningen}

In alle oefeningen is het \omterm{def:g10:coordgeom:system}{coördinatenstelsel} \omterm{def:g10:coordgeom:system}{orthonormaal}.

\begin{exercise}[$\star$]\label{exo:g12:space:1}
Laat $A(1, 0, 2)$, $B(3, 1, 1)$ en $C(2, -1, 3)$. Zijn de punten $A$, $B$,
$C$ \omterm{def:g11:vect:collinear}{collineair}? Geef een parametrische voorstelling van de lijn $(AB)$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:space:2}
Bepaal een Cartesiaanse \omterm{def:g10:algebra:equation}{vergelijking} van het vlak door $A(1, 1, 0)$ met
\omterm{def:g12:space:normal}{normaalvector} $\vec n(2, -1, 3)$. Behoort het punt $B(0, 2, 1)$ ertoe?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:space:3}
Bereken de hoek in $A$ (tot op de graad) van de driehoek met hoekpunten
$A(0,0,0)$, $B(1, 1, 0)$ en $C(1, 0, 1)$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:space:4}
Beschouw de lijn $\mathcal D$ door $A(1, 2, 0)$ met richting
$\vec u(1, -1, 2)$, en het vlak $\mathcal P : 2x + y - z + 1 = 0$.
Bepaal $\mathcal D \cap \mathcal P$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:space:5}
Laat $\mathcal P : x - y + 2z = 1$ en $\mathcal Q : 2x + y + z = 4$.
Toon dat $\mathcal P$ en $\mathcal Q$ snijden in een lijn en geef een
parametrische voorstelling ervan.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:space:6}
Toon dat de lijnen
\[
\mathcal D_1 : (x, y, z) = (1 + t,\; 2t,\; 3 - t), \qquad
\mathcal D_2 : (x, y, z) = (2 + s,\; 1 + s,\; 1 + 2s)
\]
kruisend zijn (niet \omterm{def:g12:space:collinear}{coplanair}).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:space:7}
Laat $\mathcal P : 2x - y + 2z - 3 = 0$.
\begin{enumerate}
  \item Bereken de afstand van $M_0(3, 1, 1)$ tot $\mathcal P$.
  \item Bepaal de \omterm{def:g10:coordgeom:system}{coördinaten} van de \omterm{def:g11:scal:orthogonal}{orthogonale} projectie $H$ van $M_0$
        op $\mathcal P$, en controleer de afstand $M_0H$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:space:8}
De kubus $ABCDEFGH$ heeft zijde $1$; plaats \omterm{def:g10:coordgeom:system}{coördinaten} zodat
$A(0,0,0)$, $B(1,0,0)$, $D(0,1,0)$, $E(0,0,1)$ (met
$C = B + D$, $F = B + E$, $G = B + D + E$, $H = D + E$ als \omterm{def:g11:vect:vector}{vectoren} uit
$A$).
\begin{enumerate}
  \item Toon dat de diagonaal $(AG)$ loodrecht is op het vlak $(BDE)$.
  \item Bereken de afstand van $A$ tot het vlak $(BDE)$, en het punt waar
        $(AG)$ het doorboort.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:space:9}
\emph{(Volume van een tetraëder.)} Laat $A(1,1,1)$, $B(2,1,0)$, $C(0,2,1)$
en $D(2,2,2)$.
\begin{enumerate}
  \item Verifieer dat $\vect{AB}$, $\vect{AC}$, $\vect{AD}$ niet \omterm{def:g12:space:collinear}{coplanair}
        zijn (de punten vormen een echte tetraëder).
  \item Vind een \omterm{def:g11:vect:vector}{vector} $\vec n$ loodrecht op zowel $\vect{BC}$ als
        $\vect{BD}$, en leid een Cartesiaanse \omterm{def:g10:algebra:equation}{vergelijking} van het vlak
        $(BCD)$ af.
  \item Bereken de afstand van $A$ tot het vlak $(BCD)$ en de oppervlakte
        van de driehoek $BCD$, met de formule
        $\text{Opp.} = \frac12\sqrt{\norm{\vec u}^2\norm{\vec v}^2 -
        (\vec u \cdot \vec v)^2}$ voor de driehoek opgespannen door
        $\vec u, \vec v$.
  \item Leid het volume van de tetraëder af
        ($V = \frac13 \times \text{grondvlak} \times \text{hoogte}$).
\end{enumerate}
\end{exercise}
