\chapter{Rekenkunde}\label{ch:g12:arith}

Rekenkunde bestudeert de gehele getallen: deelbaarheid, priemgetallen,
resten. Lang beschouwd als de puurste der pure wiskunde, beschermt zij nu
elke online betaling: het RSA-cryptosysteem rust op de stellingen van
Bézout, Gauss en Fermat bewezen in dit hoofdstuk.

\section{Deelbaarheid en euclidische deling}

\begin{definition}[Deelbaarheid]\label{def:g12:arith:divides}
Laat $a, b \in \Z$. We zeggen dat $b$ \emph{deelt}\index{deelbaarheid}
$a$, geschreven $b \mid a$, als er een $k \in \Z$ bestaat met $a = kb$. We
zeggen ook dat $a$ een \emph{veelvoud} van $b$ is.
\end{definition}

\begin{proposition}\label{prop:g12:arith:divprops}
Als $c \mid a$ en $c \mid b$, dan deelt $c$ elke gehele combinatie
$au + bv$ ($u, v \in \Z$). Als $a \mid b$ en $b \mid a$ met
$a,b \in \N$, dan $a = b$. Als $a \mid b$ en $b \neq 0$, dan
$\abs a \leq \abs b$.
\end{proposition}

\begin{proof}
Schrijf $a = kc$, $b = lc$: dan $au + bv = (ku + lv)c$. De andere punten
volgen uit $\abs{a} = \abs{k}\,\abs{b}$ met $\abs k \geq 1$ wanneer
$b = ka \neq 0$.
\end{proof}

\begin{theorem}[Euclidische deling]\label{thm:g12:arith:euclid}
\index{euclidische deling}
Laat $a \in \Z$ en $b \in \N^*$. Er bestaat een uniek paar
$(q, r) \in \Z \times \N$ zodat
\[
a = bq + r \qquad\text{en}\qquad 0 \leq r < b .
\]
$q$ is het \emph{quotiënt} en $r$ de \emph{rest}.
\end{theorem}

\begin{proof}
\emph{Bestaan.} De verzameling van veelvouden van $b$ die $a$ niet
overschrijden heeft een grootste element $bq$ (niet-leeg en naar boven
begrensd); stel $r = a - bq$. Door maximaliteit is $b(q+1) > a$, dus
$0 \leq r < b$.
\emph{Uniciteit.} Als $bq + r = bq' + r'$ met $0 \leq r, r' < b$, dan
$b(q - q') = r' - r$ en $\abs{r' - r} < b$: een veelvoud van $b$ met
absolute waarde kleiner dan $b$ moet $0$ zijn, dus $r = r'$ en $q = q'$.
\end{proof}

\section{Congruenties}

\begin{definition}[Congruentie]\label{def:g12:arith:congruence}
Laat $n \in \N^*$. Twee gehele getallen $a, b$ zijn \emph{congruent modulo
$n$}\index{congruentie}, geschreven $a \equiv b \pmod n$, als
$n \mid (a - b)$ --- equivalent, als $a$ en $b$ dezelfde rest hebben bij
euclidische deling door $n$.
\end{definition}

\begin{proposition}[Verenigbaarheid met bewerkingen]\label{prop:g12:arith:congops}
Als $a \equiv b \pmod n$ en $c \equiv d \pmod n$, dan
\[
a + c \equiv b + d, \qquad
ac \equiv bd, \qquad
a^k \equiv b^k \ (k \in \N) \pmod n .
\]
\end{proposition}

\begin{proof}
$n$ deelt $(a-b) + (c-d) = (a+c) - (b+d)$, en
$ac - bd = a(c - d) + d(a - b)$ is eveneens een veelvoud van $n$. De
machtenregel volgt door inductie uit de productregel.
\end{proof}

\begin{method}[Machten modulo $n$ berekenen]\label{met:g12:arith:powers}
Om $a^k \bmod n$ te berekenen, reduceer de basis modulo $n$, zoek daarna
een kleine macht van $a$ congruent met $\pm1$, en gebruik die om de
exponent in te klappen. Bijvoorbeeld $2^{100} \bmod 7$: omdat
$2^3 = 8 \equiv 1 \pmod 7$ en $100 = 3\times33 + 1$,
\[
2^{100} = \left(2^{3}\right)^{33} \times 2 \equiv 1^{33}\times 2 = 2 \pmod 7 .
\]
\end{method}

\section{GGD, Bézout en Gauss}

\begin{definition}[GGD]\label{def:g12:arith:gcd}
Laat $a, b$ gehele getallen zijn, niet beide nul. De \emph{grootste
gemene deler}\index{ggd} $\gcd(a, b)$ is het grootste gehele getal dat
zowel $a$ als $b$ deelt. Wanneer $\gcd(a,b) = 1$, heten $a$ en $b$
\emph{relatief priem}\index{relatief priem}.
\end{definition}

\begin{proposition}[Algoritme van Euclides]\label{prop:g12:arith:euclidalgo}
\index{algoritme van Euclides}
Als $a = bq + r$ ($b \neq 0$), dan $\gcd(a, b) = \gcd(b, r)$. Itereren van
de euclidische deling berekent dus $\gcd(a,b)$: de ggd is de laatste
niet-nul rest.
\end{proposition}

\begin{proof}
Elke gemeenschappelijke deler van $a$ en $b$ deelt $r = a - bq$
(\cref{prop:g12:arith:divprops}), dus is een gemeenschappelijke deler van
$b$ en $r$; en omgekeerd, omdat $a = bq + r$. De twee paren hebben dezelfde
gemeenschappelijke delers, dus dezelfde ggd. Het algoritme eindigt omdat
de resten een strikt dalende rij van niet-negatieve gehele getallen vormen.
\end{proof}

\begin{example}\label{ex:g12:arith:euclidalgo}
$\gcd(252, 198)$: $252 = 198 + 54$; $198 = 3\times54 + 36$;
$54 = 36 + 18$; $36 = 2 \times 18 + 0$. Dus $\gcd(252,198) = 18$.
\end{example}

\begin{theorem}[Identiteit van Bézout]\label{thm:g12:arith:bezout}
\index{identiteit van Bézout}
Laat $a, b$ gehele getallen zijn, niet beide nul, en $d = \gcd(a,b)$. Er
bestaan $u, v \in \Z$ zodat
\[
au + bv = d .
\]
In het bijzonder zijn $a$ en $b$ relatief priem dan en slechts dan als
$au + bv = 1$ voor zekere gehele $u, v$.
\end{theorem}

\begin{proof}
Voer het algoritme van Euclides achterwaarts uit: elke rest is een gehele
combinatie van de vorige twee, en de beginwaarden $a, b$ zijn combinaties
van zichzelf; door dalende substitutie is de laatste niet-nul rest $d$ een
gehele combinatie van $a$ en $b$. (In \cref{ex:g12:arith:euclidalgo}:
$18 = 54 - 36 = 54 - (198 - 3\times54) = 4\times54 - 198 =
4(252 - 198) - 198 = 4\times252 - 5\times198$.)

Voor de equivalentie: als $\gcd(a,b) = 1$, levert Bézout $u, v$; omgekeerd
deelt elke gemeenschappelijke deler van $a$ en $b$ $au + bv = 1$, wat
$\gcd(a,b) = 1$ dwingt.
\end{proof}

\begin{theorem}[Lemma van Gauss]\label{thm:g12:arith:gauss}
\index{lemma van Gauss}
Laat $a, b, c \in \Z$. Als $a \mid bc$ en $\gcd(a, b) = 1$, dan $a \mid c$.
\end{theorem}

\begin{proof}
Bézout geeft $au + bv = 1$; vermenigvuldig met $c$: $acu + bcv = c$. Beide
termen van de linkerzijde zijn veelvouden van $a$ (de tweede omdat
$a \mid bc$), dus ook $c$.
\end{proof}

\begin{corollary}\label{cor:g12:arith:coprimeprod}
Als $a \mid c$, $b \mid c$ en $\gcd(a,b) = 1$, dan $ab \mid c$.
\end{corollary}

\begin{proof}
Schrijf $c = ak$. Uit $b \mid ak$ en $\gcd(a,b)=1$ geeft Gauss
$b \mid k$, zeg $k = bl$; dan $c = abl$.
\end{proof}

\section{Priemgetallen}

\begin{definition}[Priem]\label{def:g12:arith:prime}
Een geheel getal $p \geq 2$ is \emph{priem}\index{priemgetal} als zijn
enige positieve delers $1$ en $p$ zijn.
\end{definition}

\begin{proposition}\label{prop:g12:arith:primedivides}
Elk geheel getal $n \geq 2$ heeft een priemdeler; als $n$ niet priem is,
heeft het een priemdeler $\leq \sqrt n$. Als een priemgetal $p$ een
product $ab$ deelt, dan $p \mid a$ of $p \mid b$ (\emph{lemma van
Euclides}).
\end{proposition}

\begin{proof}
De kleinste deler $d \geq 2$ van $n$ is priem (elke echte deler van $d$
zou een kleinere deler van $n$ zijn). Als $n = de$ samengesteld is met
$2 \leq d \leq e$, dan $d^2 \leq de = n$, dus $d \leq \sqrt n$. Voor het
lemma van Euclides: als $p \nmid a$, dan $\gcd(p, a) = 1$ (de enige delers
van $p$ zijn $1$ en $p$), en het lemma van Gauss geeft $p \mid b$.
\end{proof}

\begin{theorem}[Euclides]\label{thm:g12:arith:infiniteprimes}
Er zijn oneindig veel priemgetallen.
\end{theorem}

\begin{proof}
Gegeven een willekeurige eindige lijst $p_1, \dots, p_k$ van priemgetallen,
beschouw $N = p_1 p_2 \cdots p_k + 1$. Enig priemgetal $p$ deelt $N$; maar
geen $p_i$ deelt $N$ (de rest is $1$), dus $p$ is een priemgetal niet in de
lijst. Geen eindige lijst put de priemgetallen uit.
\end{proof}

\begin{theorem}[Hoofdstelling van de rekenkunde]\label{thm:g12:arith:fta}
\index{hoofdstelling van de rekenkunde}
Elk geheel getal $n \geq 2$ is een product van priemgetallen, en deze
factorisatie is uniek op de volgorde van de factoren na:
\[
n = p_1^{\alpha_1} p_2^{\alpha_2} \cdots p_r^{\alpha_r},
\qquad p_1 < p_2 < \dots < p_r \text{ priem},\ \alpha_i \geq 1 .
\]
\end{theorem}

\begin{proof}
\emph{Bestaan}, door sterke inductie: $n$ priem is zijn eigen factorisatie;
anders $n = de$ met $2 \leq d, e < n$, en beide factoren volgens de
inductiehypothese. \emph{Uniciteit}: stel
$p_1\cdots p_s = q_1 \cdots q_t$ (priemgetallen, herhalingen toegestaan).
Volgens het lemma van Euclides deelt $p_1$ enig $q_j$, en als priem is
$p_1 = q_j$; schrap en herhaal. De twee factorisaties komen term voor term
overeen.
\end{proof}

\begin{theorem}[Kleine stelling van Fermat]\label{thm:g12:arith:fermat}
\index{kleine stelling van Fermat}
Laat $p$ priem zijn en $a \in \Z$ met $p \nmid a$. Dan
\[
a^{p-1} \equiv 1 \pmod p .
\]
Voor elk $a \in \Z$ (zonder relatieve-priemheid), $a^p \equiv a \pmod p$.
\end{theorem}

\begin{proof}
Beschouw de $p - 1$ gehele getallen $a, 2a, 3a, \dots, (p-1)a$ modulo $p$.
Geen is $\equiv 0$ (als $p \mid ka$ met $1 \leq k \leq p-1$, dwingt het
lemma van Euclides $p \mid k$, onmogelijk), en ze zijn paarsgewijs
verschillend modulo $p$ (als $ka \equiv la$, dan $p \mid (k - l)a$, dus
$p \mid k - l$, dus $k = l$). Dus, modulo $p$, zijn ze de getallen
$1, 2, \dots, p-1$ in zekere volgorde. Alle congruenties vermenigvuldigen:
\[
a^{p-1}\,(p-1)! \equiv (p-1)! \pmod p .
\]
Omdat $p$ geen van $1, \dots, p-1$ deelt, staat herhaald gebruik van het
lemma van Euclides toe $(p-1)!$ te schrappen, wat $a^{p-1} \equiv 1$
achterlaat. De tweede vorm volgt door met $a$ te vermenigvuldigen (en is
triviaal wanneer $p \mid a$).
\end{proof}

\begin{example}[Toepassing op cryptografie]\label{ex:g12:arith:rsa}
De stelling van Fermat maakt machtsverheffen modulo $n$ omkeerbaar wanneer
de exponenten geschikt gekozen zijn --- het hart van het
\emph{RSA}-cryptosysteem. Met $p, q$ grote priemgetallen en $n = pq$
publiceert men $n$ en een exponent $e$; encryptie is
$x \mapsto x^e \bmod n$. Ontcijferen vereist een exponent $d$ met
$ed \equiv 1 \pmod{(p-1)(q-1)}$, die alleen iemand die $p$ en $q$ kent kan
berekenen --- en $p, q$ uit $n$ herstellen betekent een getal van
honderden cijfers factoriseren, wat geen bekend algoritme in redelijke
tijd doet.
\end{example}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g12:arith:1}
Bereken het quotiënt en de rest van de euclidische deling van $2026$ door
$17$, en van $-2026$ door $17$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:arith:2}
Wat is de rest van $7^{100}$ modulo $10$? (Wat is het laatste cijfer van
$7^{100}$?)
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:arith:3}
Met het algoritme van Euclides, bereken $\gcd(1071, 462)$, en vind gehele
$u, v$ met $1071u + 462v = \gcd(1071, 462)$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:arith:4}
Toon dat voor elk $n \in \Z$, $n^2$ congruent is met $0$ of $1$ modulo
$4$. Leid af dat een geheel getal $\equiv 3 \pmod 4$ nooit een som van
twee kwadraten is.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:arith:5}
Toon dat voor alle $n \in \N$, $n(n+1)(2n+1)$ deelbaar is door $6$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:arith:6}
Los in $\Z$ de congruentie $5x \equiv 3 \pmod{11}$ op.
(Hint: vind de inverse van $5$ modulo $11$.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:arith:7}
Los in $\Z \times \Z$ de Diophantische vergelijking
\[
17x - 40y = 1,
\]
op, beschrijf daarna alle oplossingen van $17x - 40y = 6$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:arith:8}
Toon dat $\sqrt2$ irrationaal is, met de uniciteit van de
priemfactorisatie (vergelijk de exponent van $2$ aan beide zijden van
$a^2 = 2b^2$).
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:arith:9}
Laat $p$ een priemgetal zijn.
\begin{enumerate}
  \item Toon dat voor $1 \leq k \leq p - 1$, $p$ deelt $\dbinom{p}{k}$.
        (Hint: gebruik $k\binom pk = p\binom{p-1}{k-1}$,
        \cref{exo:g12:comb:7}, en het lemma van Gauss.)
  \item Leid af, door inductie op $a \geq 0$, een ander bewijs van de
        kleine stelling van Fermat in de vorm $a^p \equiv a \pmod p$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:arith:10}
\emph{(Chinees-restenprobleem.)} Vind alle gehele $n$ zodat
\[
n \equiv 2 \pmod 3, \qquad n \equiv 3 \pmod 5, \qquad n \equiv 2 \pmod 7 .
\]
(Hint: los de eerste twee voorwaarden op, neem daarna de derde mee;
Bézout-coëfficiënten helpen.)
\end{exercise}
