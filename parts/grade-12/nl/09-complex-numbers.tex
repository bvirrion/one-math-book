\chapter{Complexe getallen}\label{ch:g12:complex}

De \omterm{def:g10:algebra:equation}{vergelijking} $x^2 = -1$ heeft geen reĆ«le oplossing. $\R$ uitbreiden
met Ć©Ć©n nieuw getal $\iu$ waarvan het kwadraat $-1$ is, produceert het
lichaam $\C$ van de \omterm{def:g12:complex:def}{complexe getallen}, waarin \emph{elke}
polynoomvergelijking oplossingen heeft --- en waarvan de rekenkunde de
meetkunde van het vlak codeert: translaties, rotaties en schalingen worden
optellingen en vermenigvuldigingen.

\section{AlgebraĆÆsche vorm}

\begin{definition}[Complexe getallen]\label{def:g12:complex:def}
De verzameling van \emph{complexe getallen}\index{complex getal} is
\[
\C = \{a + \iu b : a, b \in \R\},
\]
waar $\iu$ een symbool is onderworpen aan de enige regel $\iu^2 = -1$;
optelling en vermenigvuldiging gebeuren als voor reĆ«le polynomen in $\iu$,
waarbij $\iu^2$ tot $-1$ gereduceerd wordt. De \omterm{def:g10:numbers:sets}{reĆ«le getallen} $a$ en $b$
zijn het \emph{reĆ«le deel}\index{reĆ«le deel} $\Rea(z)$ en het \emph{imaginaire deel}\index{imaginaire deel}
$\Ima(z)$ van $z = a + \iu b$; deze uitdrukking is de
\emph{algebraĆÆsche vorm}\index{algebraĆÆsche vorm} van $z$, en zij is uniek:
$a + \iu b = a' + \iu b'$ dan en slechts dan als $a = a'$ en $b = b'$.
\end{definition}

\begin{example}
$(2 + 3\iu)(1 - \iu) = 2 - 2\iu + 3\iu - 3\iu^2 = 5 + \iu$.
\end{example}

\begin{definition}[Geconjugeerde]\label{def:g12:complex:conjugate}
De \emph{geconjugeerde}\index{geconjugeerde} van $z = a + \iu b$ is
$\conj{z} = a - \iu b$.
\end{definition}

\begin{proposition}[Eigenschappen van conjugatie]\label{prop:g12:complex:conjugate}
Voor alle $z, w \in \C$:
\[
\conj{z + w} = \conj z + \conj w, \quad
\conj{zw} = \conj z\,\conj w, \quad
\conj{\conj z} = z, \quad
z + \conj z = 2\Rea(z), \quad
z - \conj z = 2\iu\,\Ima(z),
\]
en $z\conj z = a^2 + b^2 \geq 0$ voor $z = a + \iu b$. Bovendien $z \in \R$
dan en slechts dan als $z = \conj z$.
\end{proposition}

\begin{proof}
Alles volgt door directe berekening uit de \omterm{def:g12:complex:def}{algebraĆÆsche vormen}; bijvoorbeeld
$z\conj z = (a + \iu b)(a - \iu b) = a^2 - (\iu b)^2 = a^2 + b^2$.
\end{proof}

\begin{method}[Complexe getallen delen]\label{met:g12:complex:division}
Om een quotiĆ«nt in \omterm{def:g12:complex:def}{algebraĆÆsche vorm} te schrijven, vermenigvuldig teller en
noemer met de \omterm{def:g12:complex:conjugate}{geconjugeerde} van de noemer:
\[
\frac{1}{2 + \iu} = \frac{2 - \iu}{(2+\iu)(2-\iu)} = \frac{2 - \iu}{5}
= \frac25 - \frac15\iu .
\]
Elk niet-nul \omterm{def:g12:complex:def}{complex getal} heeft dus een inverse: $\C$ is een
\emph{lichaam}.
\end{method}

\section{Het complexe vlak, modulus en argument}

\begin{definition}[Affix, modulus]\label{def:g12:complex:modulus}
In het vlak met \omterm{def:g10:coordgeom:system}{orthonormale} \omterm{def:g10:coordgeom:system}{coĆ¶rdinaten} is het punt $M(a, b)$ het
\emph{\omterm{def:g10:functions:function}{beeld}} van $z = a + \iu b$, en $z$ is de \emph{affix}\index{affix} van $M$. De
\emph{modulus}\index{modulus} van $z$ is de afstand van $M$ tot de
oorsprong:
\[
\abs{z} = \sqrt{a^2 + b^2} = \sqrt{z\conj z}.
\]
\end{definition}

\begin{proposition}[Eigenschappen van de modulus]\label{prop:g12:complex:modulus}
Voor $z, w \in \C$:
\[
\abs{zw} = \abs z\,\abs w, \qquad
\abs{\tfrac zw} = \tfrac{\abs z}{\abs w}\ (w \neq 0), \qquad
\abs{\conj z} = \abs z, \qquad
\abs{z + w} \leq \abs{z} + \abs{w} \ \text{(driehoeksongelijkheid)}.
\]
Bovendien is $\abs{z_B - z_A}$ de afstand tussen de punten met \omterm{def:g12:complex:modulus}{affixen}
$z_A$ en $z_B$.
\end{proposition}

\begin{proof}
$\abs{zw}^2 = zw\,\conj{zw} = z\conj z\, w \conj w = \abs z^2 \abs w^2$
geeft de eerste; de tweede en derde zijn analoog. Voor de
driehoeksongelijkheid:
\[
\abs{z+w}^2 = (z+w)(\conj z + \conj w)
= \abs z^2 + \abs w^2 + 2\Rea(z \conj w)
\leq \abs z^2 + \abs w^2 + 2\abs{z\conj w}
= \bigl(\abs z + \abs w\bigr)^2,
\]
met $\Rea(u) \leq \abs u$. De afstandsuitspraak is de stelling van
Pythagoras toegepast op de \omterm{def:g10:coordgeom:system}{coĆ¶rdinaten} van $z_B - z_A$.
\end{proof}

\begin{definition}[Argument, goniometrische en exponentiĆ«le vormen]\label{def:g12:complex:argument}
Laat $z \neq 0$. Een \emph{argument}\index{argument} van $z$, geschreven
$\Arg(z)$, is elke maat $\theta$ van de hoek van de positieve reĆ«le as
naar de straal $OM$; zij is gedefinieerd op $2k\pi$ na. Met $r = \abs z$,
\[
z = r(\cos\theta + \iu\sin\theta)
\qquad\text{(goniometrische vorm)}.
\]
Met de notatie\index{exponentiĆ«le vorm}
\[
\eu^{\iu\theta} = \cos\theta + \iu\sin\theta,
\]
wordt dit de \emph{exponentiĆ«le vorm} $z = r\,\eu^{\iu\theta}$.
\end{definition}

\begin{theorem}\label{thm:g12:complex:expform}
Voor alle $\theta, \theta' \in \R$:
\[
\eu^{\iu\theta}\,\eu^{\iu\theta'} = \eu^{\iu(\theta + \theta')},
\qquad
\conj{\eu^{\iu\theta}} = \eu^{-\iu\theta},
\qquad
\abs{\eu^{\iu\theta}} = 1 .
\]
Bijgevolg, voor $z = r\eu^{\iu\theta}$ en $z' = r'\eu^{\iu\theta'}$
niet-nul:
\[
zz' = rr'\,\eu^{\iu(\theta+\theta')},
\qquad
\frac{z}{z'} = \frac{r}{r'}\,\eu^{\iu(\theta - \theta')} :
\]
\emph{moduli vermenigvuldigen, \omterm{def:g12:complex:argument}{argumenten} tellen op.}
\end{theorem}

\begin{proof}
De eerste identiteit \emph{is} het paar optelformules
(\cref{prop:g12:trigo:addition}):
\[
(\cos\theta + \iu\sin\theta)(\cos\theta' + \iu\sin\theta')
= (\cos\theta\cos\theta' - \sin\theta\sin\theta')
+ \iu(\sin\theta\cos\theta' + \cos\theta\sin\theta').
\]
De rest volgt door directe berekening.
\end{proof}

\begin{corollary}[Formule van de Moivre]\label{cor:g12:complex:demoivre}
\index{formule van de Moivre}
Voor $\theta \in \R$ en $n \in \Z$:
$\bigl(\cos\theta + \iu\sin\theta\bigr)^n = \cos n\theta + \iu \sin n\theta$.
\end{corollary}

\begin{proof}
Inductie op $n \geq 0$ met \cref{thm:g12:complex:expform}; uitbreiden naar
$n < 0$ door inverses te nemen.
\end{proof}

\begin{method}[Tussen vormen wisselen]\label{met:g12:complex:forms}
Van algebraĆÆsch naar exponentieel: bereken $r = \abs z$, vind daarna
$\theta$ met $\cos\theta = \frac ar$, $\sin\theta = \frac br$ (bepaal het
juiste kwadrant vĆ³Ć³r een inverse \omterm{def:g12:trigo:cossin}{cosinus}). Van exponentieel naar
algebraĆÆsch: ontwikkel $r\cos\theta + \iu\, r\sin\theta$. Gebruik de
\omterm{def:g12:complex:def}{algebraĆÆsche vorm} voor sommen, de exponentiĆ«le voor producten, machten en
quotiĆ«nten.
\end{method}

\begin{example}
$z = 1 + \iu$: $\abs z = \sqrt2$ en $\cos\theta = \sin\theta =
\frac{1}{\sqrt2}$, dus $\theta = \frac\pi4$ en $z = \sqrt2\,\eu^{\iu\pi/4}$.
Dus $z^{20} = 2^{10}\,\eu^{5\iu\pi} = -1024$.
\end{example}

\section{Tweedegraadsvergelijkingen en eenheidswortels}

\begin{theorem}[Kwadratische vergelijkingen met reĆ«le coĆ«fficiĆ«nten]\label{thm:g12:complex:quadratic}
Laat $a, b, c \in \R$, $a \neq 0$, en $\Delta = b^2 - 4ac$. De \omterm{def:g10:algebra:equation}{vergelijking}
$az^2 + bz + c = 0$ heeft oplossingen in $\C$:
\[
\begin{aligned}
&\Delta > 0:\ z = \frac{-b \pm \sqrt{\Delta}}{2a} \ (\text{twee reĆ«le}); \qquad
\Delta = 0:\ z = \frac{-b}{2a} \ (\text{dubbel}); \\
&\Delta < 0:\ z = \frac{-b \pm \iu\sqrt{-\Delta}}{2a} \ (\text{twee geconjugeerde}).
\end{aligned}
\]
\end{theorem}

\begin{proof}
Maak het kwadraat af:
$az^2 + bz + c = a\left(\left(z + \frac{b}{2a}\right)^2 -
\frac{\Delta}{4a^2}\right)$. Als $\Delta < 0$, dan
$\frac{\Delta}{4a^2} = \left(\iu\frac{\sqrt{-\Delta}}{2a}\right)^2$, en het
verschil van kwadraten factort zoals gewoonlijk.
\end{proof}

\begin{definition}[Eenheidswortels]\label{def:g12:complex:rootsofunity}
Voor $n \geq 1$ zijn de \emph{$n$-de eenheidswortels}\index{eenheidswortels}
de oplossingen van $z^n = 1$.
\end{definition}

\begin{theorem}\label{thm:g12:complex:rootsofunity}
De \omterm{def:g10:algebra:equation}{vergelijking} $z^n = 1$ heeft precies $n$ oplossingen:
\[
\omega_k = \eu^{2\iu k\pi/n}, \qquad k = 0, 1, \dots, n-1 .
\]
Hun \omterm{def:g10:functions:function}{beelden} zijn de hoekpunten van een regelmatige $n$-hoek ingeschreven in
de eenheidscirkel, en voor $n \geq 2$ is hun som $0$.
\end{theorem}

\begin{proof}
Elke $\omega_k$ voldoet aan $\omega_k^n = \eu^{2\iu k\pi} = 1$, en ze zijn
paarsgewijs verschillend omdat hun \omterm{def:g12:complex:argument}{argumenten} in $\intco{0}{2\pi}$ liggen.
Omgekeerd, als $z^n = 1$ dan $\abs z^n = 1$, dus $\abs z = 1$ (positief
reĆ«el) en $z = \eu^{\iu\theta}$ met $n\theta \equiv 0 \pmod{2\pi}$,
\emph{d.w.z.}\ $\theta = \frac{2k\pi}{n}$; $k$ modulo $n$ reduceren landt
onder de $\omega_k$. De hoekpunten liggen op gelijke hoeken $\frac{2\pi}{n}$:
een regelmatige $n$-hoek. Ten slotte, met $\omega = \omega_1$,
$\omega \neq 1$ en de geometrische som:
\[
\sum_{k=0}^{n-1}\omega_k = \sum_{k=0}^{n-1}\omega^k
= \frac{\omega^n - 1}{\omega - 1} = 0 . \qedhere
\]
\end{proof}

\section{Complexe getallen en vlakke meetkunde}

\begin{proposition}[Meetkundig woordenboek]\label{prop:g12:complex:geometry}
Laat $A, B, C, D$ punten zijn met \omterm{def:g12:complex:modulus}{affixen} $a, b, c, d$ ($a \neq b$,
$c \neq d$).
\begin{enumerate}
  \item De translatie over de \omterm{def:g11:vect:vector}{vector} van \omterm{def:g12:complex:modulus}{affix} $t$ is $z \mapsto z + t$.
  \item De rotatie met middelpunt $\omega$ (\omterm{def:g12:complex:modulus}{affix}) en hoek $\theta$ is
        $z \mapsto \omega + \eu^{\iu\theta}(z - \omega)$.
  \item De homothetie met middelpunt $\omega$ en verhouding
        $k \in \R^*$ is $z \mapsto \omega + k(z - \omega)$.
  \item $\dfrac{d - c}{b - a}$ heeft \omterm{def:g12:complex:modulus}{modulus} $\dfrac{CD}{AB}$ en \omterm{def:g12:complex:argument}{argument} de
        hoek tussen de \omterm{def:g11:vect:vector}{vectoren} $\vect{AB}$ en $\vect{CD}$. In het
        bijzonder: $AB \perp CD$ dan en slechts dan als $\frac{d-c}{b-a}$
        zuiver imaginair is, en $A, B, C$ zijn \omterm{def:g11:vect:collinear}{collineair} ($C \ne A$,
        $C\ne B$) dan en slechts dan als $\frac{c-a}{b-a} \in \R$.
\end{enumerate}
\end{proposition}

\begin{proof}
(1) en (3) herformuleren de coĆ¶rdinaatformules. Voor (2): de afbeelding
$w \mapsto \eu^{\iu\theta} w$ vermenigvuldigt moduli met $1$ en telt
$\theta$ bij \omterm{def:g12:complex:argument}{argumenten} op: het is de rotatie over hoek $\theta$ om de
oorsprong; conjugeren met de translatie die $\omega$ naar $0$ stuurt geeft
het algemene middelpunt. Voor (4): moduli en \omterm{def:g12:complex:argument}{argumenten} van quotiĆ«nten
trekken af (\cref{thm:g12:complex:expform}).
\end{proof}

\begin{exercise}[$\star\star\star$]\label{exo:g12:complex:10}

\end{exercise}
