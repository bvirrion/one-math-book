\chapter{Differentiaalvergelijkingen}\label{ch:g12:diffeq}

Een \omterm{def:g12:diffeq:ode}{differentiaalvergelijking} koppelt een \omterm{def:g11:func:function}{functie} aan haar \omterm{def:g12:deriv:derivative}{afgeleiden}.
Natuurkunde, scheikunde, biologie en economie drukken hun wetten in deze
vorm uit: de afkoelingssnelheid van een lichaam, het verval van een
radioactieve kern, de groei van een populatie zijn allemaal uitspraken over
$y'$. Dit hoofdstuk lost volledig de lineaire eerste-ordevergelijkingen
met constante coëfficiënten op.

\section{De vergelijking \texorpdfstring{$y' = ay$}{y' = ay}}

\begin{definition}[Differentiaalvergelijking]\label{def:g12:diffeq:ode}
Een \emph{differentiaalvergelijking}\index{differentiaalvergelijking} is
een \omterm{def:g10:algebra:equation}{vergelijking} waarvan de onbekende een \omterm{def:g11:func:function}{functie} $y$ is, en die $y$ en
haar \omterm{def:g12:deriv:derivative}{afgeleiden} bevat. Een \emph{oplossing} op een \omterm{def:g10:numbers:interval}{interval} $I$ is een
differentieerbare \omterm{def:g11:func:function}{functie} die de \omterm{def:g10:algebra:equation}{vergelijking} in elk punt van $I$
voldoet.
\end{definition}

\begin{theorem}[Oplossingen van $y' = ay$]\label{thm:g12:diffeq:homogeneous}
Laat $a \in \R$. De oplossingen op $\R$ van de \omterm{def:g10:algebra:equation}{vergelijking} $y' = ay$ zijn
precies de \omterm{def:g11:func:function}{functies}
\[
y(x) = C\,\eu^{ax}, \qquad C \in \R .
\]
Voor elk paar $(x_0, y_0)$ bestaat er precies één oplossing met
$y(x_0) = y_0$.
\end{theorem}

\begin{proof}
Elke $y = C\eu^{ax}$ voldoet aan $y' = Ca\,\eu^{ax} = ay$. Omgekeerd, laat
$y$ een willekeurige oplossing zijn en stel $z(x) = y(x)\,\eu^{-ax}$. Dan
\[
z'(x) = y'(x)\,\eu^{-ax} - a\,y(x)\,\eu^{-ax}
= \bigl(y'(x) - a y(x)\bigr)\eu^{-ax} = 0,
\]
dus $z$ is constant, zeg $C$, en $y(x) = C\eu^{ax}$. De beginvoorwaarde
$y(x_0) = y_0$ dwingt $C = y_0 \eu^{-a x_0}$, uniek.
\end{proof}

\begin{remark}
De \omterm{thm:g12:exp:existence}{exponentiële functie} wordt zo \emph{gekarakteriseerd} door de
eenvoudigste van alle \omterm{def:g12:diffeq:ode}{differentiaalvergelijkingen}: groei evenredig met de
grootte. Daarom verschijnt zij overal in de natuur.
\end{remark}

\begin{example}[Radioactief verval]\label{ex:g12:diffeq:decay}
Een radioactieve hoeveelheid voldoet aan $N' = -\lambda N$ met
$\lambda > 0$, dus $N(t) = N_0\,\eu^{-\lambda t}$; zie
\cref{ex:g12:exp:halflife} voor de halfwaardetijd.
\end{example}

\section{De vergelijking \texorpdfstring{$y' = ay + b$}{y' = ay + b}}

\begin{theorem}[Oplossingen van $y' = ay + b$]\label{thm:g12:diffeq:affine}
Laat $a \neq 0$ en $b \in \R$. De oplossingen op $\R$ van $y' = ay + b$
zijn precies de \omterm{def:g11:func:function}{functies}
\[
y(x) = C\,\eu^{ax} - \frac{b}{a}, \qquad C \in \R .
\]
De constante \omterm{def:g11:func:function}{functie} $y_p = -\frac ba$ is de \emph{evenwichtsoplossing}.
Voor elk $(x_0, y_0)$ bestaat er precies één oplossing met
$y(x_0) = y_0$; als $a < 0$, dan nadert elke oplossing het evenwicht
$-\frac ba$ als $x \to +\infty$.
\end{theorem}

\begin{proof}
De constante $y_p = -\frac{b}{a}$ voldoet aan $y_p' = 0 = a y_p + b$. Nu is
$y$ een oplossing dan en slechts dan als
\[
(y - y_p)' = y' = ay + b = a(y - y_p) + \underbrace{a y_p + b}_{=\,0}
= a (y - y_p),
\]
dat wil zeggen dan en slechts dan als $z = y - y_p$ de \omterm{def:g10:algebra:equation}{vergelijking}
$z' = az$ oplost. Volgens \cref{thm:g12:diffeq:homogeneous} is
$z = C\eu^{ax}$, waaruit de formule, het bestaan en de uniciteit volgen.
Als $a < 0$, dan $\eu^{ax} \to 0$ als $x \to +\infty$, dus
$y(x) \to -\frac ba$.
\end{proof}

\begin{method}[Oplossen van $y' = ay + b$ met beginvoorwaarde]\label{met:g12:diffeq:solve}
\begin{enumerate}
  \item Vind de evenwichtsoplossing $y_p = -\frac{b}{a}$ (los $y' = 0$ op).
  \item Schrijf de algemene oplossing $y = C\eu^{ax} + y_p$.
  \item Bepaal $C$ uit de beginvoorwaarde.
  \item Controleer het lange-termijngedrag tegen fysische intuïtie (\omterm{def:g12:seq:limit}{convergeert}
        de oplossing naar het evenwicht?).
\end{enumerate}
Dezelfde strategie --- \emph{bijzondere oplossing + algemene oplossing van
de homogene \omterm{def:g10:algebra:equation}{vergelijking}} --- strekt zich uit tot $y' = ay + f(x)$: zie
\cref{prop:g12:diffeq:general}.
\end{method}

\begin{example}[Newtons afkoelingswet]\label{ex:g12:diffeq:cooling}
Een kop koffie van $80\,^\circ$C staat in een kamer van $20\,^\circ$C. Newtons
wet zegt dat de temperatuur $T$ voldoet aan $T' = -k(T - 20)$ voor zekere
$k > 0$, \emph{d.w.z.}\ $T' = -kT + 20k$. Het evenwicht is $20$, en
$T(t) = 20 + 60\,\eu^{-kt}$: de koffie koelt exponentieel af naar
kamertemperatuur.
\end{example}

\section{De vergelijking \texorpdfstring{$y' = ay + f(x)$}{y' = ay + f(x)}}

\begin{proposition}[Structuur van de oplossingsverzameling]\label{prop:g12:diffeq:general}
Laat $f$ \omterm{def:g12:limcont:continuity}{continu} zijn op een \omterm{def:g10:numbers:interval}{interval} $I$ en laat $y_p$ één bijzondere
oplossing zijn van
\[
y' = ay + f(x)
\]
op $I$. Dan zijn de oplossingen op $I$ precies de \omterm{def:g11:func:function}{functies}
$y = C\eu^{ax} + y_p$, $C \in \R$.
\end{proposition}

\begin{proof}
Zoals in \cref{thm:g12:diffeq:affine}: $y$ is een oplossing dan en slechts
dan als $z = y - y_p$ voldoet aan
$z' = (ay + f) - (ay_p + f) = az$, dan en slechts dan als
$z = C\eu^{ax}$.
\end{proof}

\begin{method}[Een bijzondere oplossing raden]\label{met:g12:diffeq:particular}
Zoek een bijzondere oplossing \emph{van dezelfde vorm} als $f$:
\begin{itemize}
  \item $f$ polynoom van graad $n$: probeer een polynoom van graad $n$;
  \item $f(x) = \alpha\,\eu^{kx}$ met $k \neq a$: probeer
        $y_p = \beta\,\eu^{kx}$;
  \item $f(x) = \alpha\,\eu^{ax}$ (resonant geval): probeer
        $y_p = \beta x\,\eu^{ax}$.
\end{itemize}
Substitueer in de \omterm{def:g10:algebra:equation}{vergelijking} en identificeer de coëfficiënten.
\end{method}

\begin{example}\label{ex:g12:diffeq:particular}
Los $y' = 2y + 4x$ op. Probeer $y_p = \alpha x + \beta$:
$\alpha = 2(\alpha x + \beta) + 4x$ voor alle $x$ dwingt
$2\alpha + 4 = 0$ en $\alpha = 2\beta$, dus $\alpha = -2$, $\beta = -1$.
Algemene oplossing: $y = C\eu^{2x} - 2x - 1$.
\end{example}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g12:diffeq:1}
Los op $\R$ op:
(a) $y' = 3y$ met $y(0) = 2$;\quad
(b) $2y' + y = 0$ met $y(0) = -1$;\quad
(c) $y' = -y + 5$ met $y(0) = 0$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:diffeq:2}
Een bacteriële populatie groeit met een snelheid evenredig met haar
grootte, en verdubbelt elke $3$ uur. Schrijf de \omterm{def:g12:diffeq:ode}{differentiaalvergelijking}
waaraan de populatie $N(t)$ voldoet en bepaal de evenredigheidsconstante.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:diffeq:3}
Verifieer dat $y_p(x) = x\,\eu^{x}$ een oplossing is van
$y' = y + \eu^x$, en geef alle oplossingen op $\R$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:diffeq:4}
Los $y' = -2y + \eu^{x}$ op met $y(0) = 1$.
(Hint: zoek een bijzondere oplossing van de vorm $\beta\,\eu^{x}$.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:diffeq:5}
Koolstof-14 vervalt met een halfwaardetijd van $5730$ jaar. Een
archeologisch monster bevat $60\%$ van de koolstof-14 van een levend
organisme. Schat de ouderdom.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:diffeq:6}
Een tank bevat $100$ L zuiver water. Pekel met $0.2$ kg zout per liter
stroomt in met $5$ L/min; het mengsel (homogeen gehouden) stroomt uit met
dezelfde snelheid. Laat $m(t)$ de massa zout in de tank op tijdstip $t$
zijn (in minuten).
\begin{enumerate}
  \item Rechtvaardig dat $m' = 1 - \dfrac{m}{20}$.
  \item Los op, en bepaal de limiet van $m(t)$ als $t \to +\infty$.
        Interpreteer.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:diffeq:7}
Een skydiver van massa $80$ kg valt onder invloed van de zwaartekracht
($g = 9.8\ \text{m/s}^2$) en luchtweerstand evenredig met de snelheid, zodat
haar snelheid voldoet aan $v' = g - \frac{k}{m}v$ met $k = 16$ kg/s.
\begin{enumerate}
  \item Los de \omterm{def:g10:algebra:equation}{vergelijking} op met $v(0) = 0$.
  \item Bereken de \emph{eindsnelheid} $\lim_{t\to+\infty} v(t)$, en de
        tijd nodig om $95\%$ ervan te bereiken.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:diffeq:8}
\emph{(Logistische \omterm{def:g10:algebra:equation}{vergelijking}.)} Een populatie $y(t) \in \intoo{0}{1}$
(als fractie van de maximale populatie) voldoet aan
\[
y' = y(1 - y).
\]
\begin{enumerate}
  \item Laat $z = \dfrac{1}{y}$. Toon dat $z$ de lineaire \omterm{def:g10:algebra:equation}{vergelijking}
        $z' = -z + 1$ voldoet.
  \item Los op voor $z$, daarna voor $y$, met $y(0) = \frac{1}{10}$.
  \item Toon dat $y(t) \to 1$ als $t \to +\infty$ en schets de vorm van
        de oplossingskromme.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:diffeq:9}
Laat $y$ een oplossing zijn van $y' = ay + b$ en laat $u_n = y(n)$ voor
$n \in \N$. Toon dat $(u_n)$ een arithmetico-geometrische recurrentie
$u_{n+1} = q u_n + r$ voldoet, en druk $q$ en $r$ uit in $a$ en $b$. Waaraan
beantwoordt de voorwaarde $\abs{q} < 1$ voor de
\omterm{def:g12:diffeq:ode}{differentiaalvergelijking}?
\end{exercise}
