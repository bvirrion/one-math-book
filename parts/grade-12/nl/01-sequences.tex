\chapter{Rijen}\label{ch:g12:seq}

Een \omterm{def:g12:seq:sequence}{rij} is een lijst \omterm{def:g10:numbers:sets}{reële getallen} geïndexeerd door de \omterm{def:g10:numbers:sets}{natuurlijke getallen}.
\omterm{def:g12:seq:sequence}{Rijen} modelleren discrete evoluties --- populaties jaar na jaar geteld,
saldi van een bankrekening, opeenvolgende \omterm{def:g10:numbers:approx}{benaderingen} van een getal ---
en hun limieten zijn de eerste serieuze ontmoeting met het oneindige. Dit
hoofdstuk zet de woordenschat, het inductieprincipe en de fundamentele
convergentie-stellingen op.

\section{Redeneren door inductie}

\begin{theorem}[Inductieprincipe]\label{thm:g12:seq:induction}
Laat $P(n)$ een uitspraak zijn die van een \omterm{def:g10:numbers:sets}{geheel getal} $n$ afhangt, en
laat $n_0 \in \N$. Als
\begin{enumerate}
  \item \emph{(basisstap)} $P(n_0)$ waar is, en
  \item \emph{(inductiestap)} voor elke $n \geq n_0$, $P(n)$ impliceert
        $P(n+1)$,
\end{enumerate}
dan is $P(n)$ waar voor elke $n \geq n_0$.
\end{theorem}

\begin{proof}
Stel, uit het ongerijmde, dat de verzameling $A$ van \omterm{def:g10:numbers:sets}{gehele getallen}
$n \geq n_0$ waarvoor $P(n)$ onwaar is, niet-leeg is. Dan heeft $A$ een
kleinste element $m$.\footnote{Elke niet-lege deelverzameling van $\N$
heeft een kleinste element; deze eigenschap van $\N$ wordt als axioma
genomen.} Omdat $P(n_0)$ waar is, is $m > n_0$, dus $m - 1 \geq n_0$ en
$m-1 \notin A$, d.w.z.\ $P(m-1)$ is waar. De inductiestap toegepast op
$n = m-1$ toont dan dat $P(m)$ waar is, in strijd met $m \in A$.
\end{proof}

\begin{example}\label{ex:g12:seq:bernoulli}
Laten we de \emph{ongelijkheid van Bernoulli} bewijzen: voor elke reële
$a > 0$ en elke $n \in \N$,
\[
(1+a)^n \geq 1 + na.
\]
\emph{Basisstap.} Voor $n = 0$ zijn beide kanten $1$.
\emph{Inductiestap.} Stel $(1+a)^n \geq 1+na$ voor zekere $n \in \N$.
Omdat $1 + a > 0$, behoudt vermenigvuldigen van beide kanten met $1+a$ de
ongelijkheid:
\[
(1+a)^{n+1} \geq (1+na)(1+a) = 1 + (n+1)a + na^2 \geq 1 + (n+1)a .
\]
Door inductie geldt de ongelijkheid voor alle $n \in \N$.
\end{example}

\begin{method}[Een inductiebewijs schrijven]\label{met:g12:seq:induction}
Maak de uitspraak $P(n)$ altijd expliciet vóór je begint. Een volledig
bewijs heeft drie zichtbare delen: de basisstap, de inductiestap
(``neem $P(n)$ aan; we bewijzen $P(n+1)$''), en de conclusie die het
inductieprincipe inroept. De meest voorkomende fout is de inductiestap
te bewijzen zonder ooit de hypothese $P(n)$ te gebruiken: als dat gebeurt,
is het bewijs ofwel fout of inductie was niet nodig.
\end{method}

\section{Woordenschat van rijen}

\begin{definition}[Rij]\label{def:g12:seq:sequence}
Een \emph{rij}\index{rij} is een \omterm{def:g11:func:function}{functie} $u \colon \N \to \R$ (of van
$\{n \in \N : n \geq n_0\}$ naar $\R$). Het \omterm{def:g10:functions:function}{beeld} van $n$ wordt
genoteerd $u_n$, en de rij zelf $(u_n)_{n\in\N}$ of eenvoudig $(u_n)$.
\end{definition}

Een \omterm{def:g12:seq:sequence}{rij} kan \emph{expliciet} gedefinieerd worden, door een formule
$u_n = f(n)$, of door \emph{recurrentie}, door haar eerste term en een
relatie $u_{n+1} = f(u_n)$.

\begin{definition}[Monotonie]\label{def:g12:seq:monotonic}
Een \omterm{def:g12:seq:sequence}{rij} $(u_n)$ is \emph{stijgend}\index{rij!stijgend} als
$u_{n+1} \geq u_n$ voor alle $n$, \emph{\omterm{def:g10:functions:variations}{dalend}} als $u_{n+1} \leq u_n$
voor alle $n$, en \emph{monotoon} als ze stijgend of \omterm{def:g10:functions:variations}{dalend} is. Ze is
\emph{strikt} stijgend (resp.\ \omterm{def:g10:functions:variations}{dalend}) wanneer de ongelijkheden strikt
zijn.
\end{definition}

\begin{method}[De monotonie van een rij bestuderen]\label{met:g12:seq:monotonicity}
Drie standaardtechnieken:
\begin{enumerate}
  \item bestudeer het teken van $u_{n+1} - u_n$;
  \item als alle termen positief zijn, vergelijk $\dfrac{u_{n+1}}{u_n}$
        met $1$;
  \item als $u_n = f(n)$ met $f$ gedefinieerd op $\intco{0}{+\infty}$,
        gebruik de variaties van $f$.
\end{enumerate}
\end{method}

\begin{definition}[Begrensde rij]\label{def:g12:seq:bounded}
Een \omterm{def:g12:seq:sequence}{rij} $(u_n)$ is \emph{van boven begrensd}\index{rij!begrensd} als er
een $M \in \R$ bestaat met $u_n \leq M$ voor alle $n$; \emph{van onder
begrensd} als er een $m \in \R$ bestaat met $u_n \geq m$ voor alle $n$;
en \emph{begrensd} als beide gelden.
\end{definition}

\subsection{Rekenkundige en meetkundige rijen}

\begin{definition}[Rekenkundige en meetkundige rijen]\label{def:g12:seq:arith-geom}
Een \omterm{def:g12:seq:sequence}{rij} $(u_n)$ is \emph{rekenkundig}\index{rij!rekenkundig} met
\omterm{def:g11:seq:arithmetic}{gemeenschappelijk verschil} $r$ als $u_{n+1} = u_n + r$ voor alle $n$, en
\emph{meetkundig}\index{rij!meetkundig} met \omterm{def:g11:seq:geometric}{gemeenschappelijke verhouding}
$q$ als $u_{n+1} = q\,u_n$ voor alle $n$.
\end{definition}

\begin{proposition}[Expliciete vorm en sommen]\label{prop:g12:seq:arith-geom}
Laat $n \in \N$.
\begin{enumerate}
  \item Als $(u_n)$ \omterm{def:g12:seq:arith-geom}{rekenkundig} is met \omterm{def:g11:seq:arithmetic}{gemeenschappelijk verschil} $r$,
        dan $u_n = u_0 + nr$ en
        \[
        u_0 + u_1 + \dots + u_n = (n+1)\,\frac{u_0 + u_n}{2}.
        \]
  \item Als $(u_n)$ \omterm{def:g12:seq:arith-geom}{meetkundig} is met \omterm{def:g11:seq:geometric}{gemeenschappelijke verhouding}
        $q \neq 1$, dan $u_n = u_0\, q^n$ en
        \[
        u_0 + u_1 + \dots + u_n = u_0\,\frac{1 - q^{n+1}}{1 - q}.
        \]
\end{enumerate}
\end{proposition}

\begin{proof}
De expliciete vormen volgen door onmiddellijke inducties. Voor de
rekenkundige som, schrijf $S = u_0 + \dots + u_n$ en tel dezelfde som
in omgekeerde volgorde op: elk van de $n+1$ kolomsommen is gelijk aan
$u_0 + u_n$, dus $2S = (n+1)(u_0+u_n)$. Voor de meetkundige som, bereken
$S - qS$: alle termen vallen paarsgewijs weg behalve de eerste en de
laatste, dus $(1-q)S = u_0(1 - q^{n+1})$.
\end{proof}

\section{Limiet van een rij}

\begin{definition}[Convergente rij]\label{def:g12:seq:limit}
Een \omterm{def:g12:seq:sequence}{rij} $(u_n)$ \emph{convergeert}\index{limiet!van een rij} naar het
reële getal $\ell$ als elk open \omterm{def:g10:numbers:interval}{interval} dat $\ell$ bevat, alle termen
$u_n$ vanaf zekere index bevat. We schrijven dan
$\lim\limits_{n\to+\infty} u_n = \ell$.

Equivalent: voor elke $\varepsilon > 0$ bestaat er $N \in \N$ zodat voor
alle $n \geq N$, $\abs{u_n - \ell} \leq \varepsilon$.
\end{definition}

\begin{definition}[Divergentie naar oneindig]\label{def:g12:seq:infinity}
De \omterm{def:g12:seq:sequence}{rij} $(u_n)$ \emph{neigt naar $+\infty$} als voor elke $A \in \R$ er
een $N \in \N$ bestaat zodat $u_n \geq A$ voor alle $n \geq N$. We
schrijven $\lim\limits_{n\to+\infty} u_n = +\infty$; de definitie van
$\lim u_n = -\infty$ is analoog. Een \omterm{def:g12:seq:sequence}{rij} die niet \omterm{def:g12:seq:limit}{convergeert} noemt men
\emph{divergent}.
\end{definition}

\begin{remark}
Een \omterm{def:g12:seq:sequence}{rij} kan divergeren zonder naar $\pm\infty$ te neigen: de \omterm{def:g12:seq:sequence}{rij}
$u_n = (-1)^n$ neemt alleen de waarden $1$ en $-1$ aan en heeft geen
limiet.
\end{remark}

\begin{proposition}[Uniciteit van de limiet]\label{prop:g12:seq:uniqueness}
Als $(u_n)$ \omterm{def:g12:seq:limit}{convergeert}, is haar limiet uniek.
\end{proposition}

\begin{proof}
Stel $u_n \to \ell$ en $u_n \to \ell'$ met $\ell \neq \ell'$, zeg
$\ell < \ell'$. Stel $\varepsilon = \frac{\ell' - \ell}{3} > 0$. Vanaf
zekere index is $\abs{u_n - \ell} \leq \varepsilon$ en
$\abs{u_n - \ell'} \leq \varepsilon$, dus
\[
\ell' - \ell \leq \abs{\ell' - u_n} + \abs{u_n - \ell}
\leq 2\varepsilon = \tfrac{2}{3}(\ell' - \ell) < \ell' - \ell,
\]
een tegenspraak.
\end{proof}

\begin{proposition}[Bewerkingen op limieten]\label{prop:g12:seq:operations}
Laat $(u_n)$ en $(v_n)$ \omterm{def:g12:seq:sequence}{rijen} zijn met limieten $\ell$ en $\ell'$ (eindig
of oneindig). Dan, wanneer het rechterlid geen onbepaalde vorm is,
\[
\lim (u_n + v_n) = \ell + \ell', \qquad
\lim (u_n v_n) = \ell\,\ell', \qquad
\lim \frac{u_n}{v_n} = \frac{\ell}{\ell'}.
\]
De onbepaalde vormen zijn
$(+\infty) + (-\infty)$, $0 \times \infty$, $\frac{\infty}{\infty}$ en
$\frac{0}{0}$.
\end{proposition}

\begin{proof}
We bewijzen de somregel voor eindige limieten; de andere gevallen zijn
analoog en als oefening gelaten. Laat $\varepsilon > 0$. Er bestaan
$N_1, N_2$ zodat $\abs{u_n - \ell} \leq \varepsilon/2$ voor $n \geq N_1$
en $\abs{v_n - \ell'} \leq \varepsilon/2$ voor $n \geq N_2$. Voor
$n \geq \max(N_1, N_2)$ geeft de driehoeksongelijkheid
\[
\abs{(u_n + v_n) - (\ell + \ell')}
\leq \abs{u_n - \ell} + \abs{v_n - \ell'} \leq \varepsilon. \qedhere
\]
\end{proof}

\begin{method}[Een onbepaalde vorm opheffen]\label{met:g12:seq:indeterminate}
Bij een onbepaalde vorm, factoriseer de dominante term. Bijvoorbeeld
\[
n^2 - n = n^2\left(1 - \tfrac{1}{n}\right) \xrightarrow[n\to+\infty]{} +\infty,
\qquad
\frac{2n^2+1}{n^2 - n} = \frac{2 + 1/n^2}{1 - 1/n}
\xrightarrow[n\to+\infty]{} 2 .
\]
\end{method}

\section{Convergentie-stellingen}

\begin{theorem}[Vergelijkings- en insluitstelling]\label{thm:g12:seq:squeeze}
Laat $(u_n)$, $(v_n)$, $(w_n)$ \omterm{def:g12:seq:sequence}{rijen} zijn.
\begin{enumerate}
  \item Als $u_n \leq v_n$ vanaf zekere index en $u_n \to +\infty$, dan
        $v_n \to +\infty$.
  \item \emph{(Insluitstelling)}\index{insluitstelling} Als
        $u_n \leq v_n \leq w_n$ vanaf zekere index en $(u_n)$ en $(w_n)$
        beide naar dezelfde limiet $\ell$ convergeren, dan \omterm{def:g12:seq:limit}{convergeert}
        $(v_n)$ naar~$\ell$.
\end{enumerate}
\end{theorem}

\begin{proof}
\emph{1.} Laat $A \in \R$. Omdat $u_n \to +\infty$, is er $N$ met
$u_n \geq A$ voor $n \geq N$; $N$ zo nodig vergrotend, $v_n \geq u_n \geq A$
voor $n \geq N$.

\emph{2.} Laat $\varepsilon > 0$. Vanaf zekere index gelden zowel
$\ell - \varepsilon \leq u_n$ als $w_n \leq \ell + \varepsilon$, dus
$\ell - \varepsilon \leq u_n \leq v_n \leq w_n \leq \ell + \varepsilon$,
d.w.z.\ $\abs{v_n - \ell} \leq \varepsilon$.
\end{proof}

\begin{example}
Voor alle $n \geq 1$, $-\frac{1}{n} \leq \frac{(-1)^n}{n} \leq \frac{1}{n}$,
en beide grenzen neigen naar $0$; dus $\frac{(-1)^n}{n} \to 0$.
\end{example}

\begin{theorem}[Monotone-convergentie-stelling]\label{thm:g12:seq:monotone}
Een stijgende \omterm{def:g12:seq:sequence}{rij} die \omterm{def:g12:seq:bounded}{van boven begrensd} is, \omterm{def:g12:seq:limit}{convergeert}. Een dalende \omterm{def:g12:seq:sequence}{rij}
die van onder begrensd is, \omterm{def:g12:seq:limit}{convergeert}. Een stijgende \omterm{def:g12:seq:sequence}{rij} die niet \omterm{def:g12:seq:bounded}{van
boven begrensd} is, neigt naar $+\infty$.
\end{theorem}

\begin{proof}[Gedeeltelijk bewijs]
We bewijzen de derde uitspraak. Laat $(u_n)$ \omterm{def:g12:seq:monotonic}{stijgend} en niet \omterm{def:g12:seq:bounded}{van boven
begrensd} zijn, en laat $A \in \R$. Omdat $A$ geen bovengrens is, bestaat
er $N$ met $u_N \geq A$; door monotonie is $u_n \geq u_N \geq A$ voor
alle $n \geq N$. Dus $u_n \to +\infty$.

De twee convergentie-uitspraken steunen op de kleinste-bovengrens-eigenschap
van $\R$; ze zijn \emph{toegegeven op dit niveau} (en bewezen in het
eerste universiteitsjaar).
\end{proof}

\begin{remark}
De stelling garandeert het \emph{bestaan} van de limiet maar geeft haar
waarde niet. Een stijgende \omterm{def:g12:seq:sequence}{rij} \omterm{def:g12:seq:bounded}{van boven begrensd} door $M$ \omterm{def:g12:seq:limit}{convergeert}
naar zekere $\ell \leq M$, niet noodzakelijk naar $M$.
\end{remark}

\begin{theorem}[Limiet van meetkundige rijen]\label{thm:g12:seq:geometric}
Laat $q \in \R$.
\begin{enumerate}
  \item Als $q > 1$, dan $q^n \to +\infty$.
  \item Als $q = 1$, dan $q^n \to 1$.
  \item Als $\abs{q} < 1$, dan $q^n \to 0$.
  \item Als $q \leq -1$, dan divergeert $(q^n)$ en heeft geen limiet.
\end{enumerate}
\end{theorem}

\begin{proof}
\emph{1.} Schrijf $q = 1 + a$ met $a > 0$. De ongelijkheid van Bernoulli
(\cref{ex:g12:seq:bernoulli}) geeft $q^n \geq 1 + na \to +\infty$, en we
concluderen door \omterm{def:g10:algebra:equation}{vergelijking} (\cref{thm:g12:seq:squeeze}).

\emph{2.} Onmiddellijk.

\emph{3.} Als $q = 0$ is de claim duidelijk. Anders geeft $\abs{q} < 1$
dat $1/\abs{q} > 1$, dus $(1/\abs{q})^n \to +\infty$ volgens punt 1, dus
$\abs{q}^n \to 0$, en $-\abs{q}^n \leq q^n \leq \abs{q}^n$ laat ons
concluderen via de insluitstelling.

\emph{4.} Voor $q \leq -1$ neemt $(q^{2n})$ waarden $\geq 1$ aan terwijl
$(q^{2n+1})$ waarden $\leq -1$ aanneemt: geen enkele limiet kan beide
deelrijen aantrekken.
\end{proof}

\begin{method}[Recurrente rijen $u_{n+1} = f(u_n)$]\label{met:g12:seq:recurrent}
Om een \omterm{def:g12:seq:sequence}{rij} gedefinieerd door $u_{n+1} = f(u_n)$ te bestuderen:
\begin{enumerate}
  \item bewijs door inductie dat $(u_n)$ in een \omterm{def:g10:numbers:interval}{interval} $I$ blijft
        waarop $f$ zich goed gedraagt (en vaak dat $(u_n)$ monotoon is);
  \item leid convergentie af uit de monotone-convergentie-stelling;
  \item ga naar de limiet in de relatie $u_{n+1} = f(u_n)$: als $f$
        continu is en $u_n \to \ell \in I$, dan voldoet $\ell$ aan
        $f(\ell) = \ell$ (zie \cref{ch:g12:limcont}); los deze
        \omterm{def:g10:algebra:equation}{vergelijking} op en selecteer de juiste \omterm{def:g11:quad:discriminant}{wortel}.
\end{enumerate}
\end{method}

\begin{example}\label{ex:g12:seq:sqrt2}
Laat $u_0 = 2$ en $u_{n+1} = \frac{1}{2}\left(u_n + \frac{2}{u_n}\right)$.
Men controleert door inductie dat $u_n \geq \sqrt{2}$ voor alle $n$ (de
ongelijkheid $\frac{1}{2}(x + 2/x) \geq \sqrt{2}$ voor $x>0$ is equivalent
met $(x - \sqrt2)^2 \geq 0$), daarna dat $(u_n)$ \omterm{def:g10:functions:variations}{dalend} is, omdat
\[
u_{n+1}-u_n=\frac{2-u_n^2}{2u_n}\leq 0 .
\]
\omterm{def:g10:functions:variations}{Dalend} en van onder begrensd, \omterm{def:g12:seq:limit}{convergeert} $(u_n)$ naar zekere
$\ell \geq \sqrt{2}$, die moet voldoen aan
$\ell = \frac{1}{2}(\ell + 2/\ell)$, d.w.z.\ $\ell^2 = 2$. Dus
$u_n \to \sqrt{2}$. Dit is het algoritme van Heron, al door de
Babyloniërs gebruikt; de convergentie is extreem snel ($u_3$ geeft al
$\sqrt 2$ tot acht decimalen).
\end{example}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g12:seq:1}
Bewijs door inductie dat voor alle $n \in \N$,
\[
1^2 + 2^2 + \dots + n^2 = \frac{n(n+1)(2n+1)}{6}.
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:seq:2}
Bestudeer de monotonie van de \omterm{def:g12:seq:sequence}{rijen} gedefinieerd voor $n \geq 1$ door
\[
a_n = \frac{n+1}{n}, \qquad
b_n = \frac{2^n}{n}, \qquad
c_n = n^2 - 10n .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:seq:3}
Bereken de limieten van de \omterm{def:g12:seq:sequence}{rijen} met algemene termen

u_n = \frac{3n^2 - n + 1}{2n^2 + 5}, \qquad
v_n = \sqrt{n+1} - \sqrt{n}, \qquad

\begin{exercise}[$\star$]\label{exo:g12:seq:4}
Laat $(u_n)$ de \omterm{def:g11:seq:arithmetic}{rekenkundige rij} zijn met $u_0 = 5$ en \omterm{def:g11:seq:arithmetic}{gemeenschappelijk
verschil} $r = 3$, en $(v_n)$ de \omterm{def:g11:seq:geometric}{meetkundige rij} met $v_0 = 8$ en
\omterm{def:g11:seq:geometric}{gemeenschappelijke verhouding} $q = \frac{1}{2}$. Bereken $u_n$, $v_n$,
$\sum_{k=0}^{n} u_k$ en $\sum_{k=0}^{n} v_k$, en de limieten van alle
vier uitdrukkingen als $n \to +\infty$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:seq:5}
Bereken met de insluitstelling
\[
\lim_{n\to+\infty} \frac{n + \cos n}{n + 1}
\qquad\text{en}\qquad
\lim_{n\to+\infty} \frac{n!}{n^n},
\]
waar $n! = 1 \times 2 \times \dots \times n$. Voor de tweede limiet,
begrens $\frac{n!}{n^n}$ door een term van een \omterm{def:g11:seq:geometric}{meetkundige rij}.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:seq:6}
Laat $u_0 = 0$ en $u_{n+1} = \sqrt{u_n + 2}$ voor alle $n \in \N$.
\begin{enumerate}
  \item Bewijs door inductie dat $0 \leq u_n \leq 2$ voor alle $n$.
  \item Toon aan dat $(u_n)$ \omterm{def:g12:seq:monotonic}{stijgend} is.
  \item Leid af dat $(u_n)$ \omterm{def:g12:seq:limit}{convergeert} en bepaal haar limiet.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:seq:7}
Een patiënt neemt elke ochtend een dosis van $1$ eenheid van een
medicijn. In elke periode van $24$ uur elimineert het lichaam $40\%$ van
het aanwezige medicijn. Laat $u_n$ de hoeveelheid medicijn in het lichaam
zijn vlak na de dosis op dag $n$, zodat $u_0 = 1$.
\begin{enumerate}
  \item Rechtvaardig dat $u_{n+1} = 0.6\,u_n + 1$.
  \item Laat $v_n = u_n - 2.5$. Toon aan dat $(v_n)$ \omterm{def:g12:seq:arith-geom}{meetkundig} is en
        leid een expliciete formule voor $u_n$ af.
  \item Bepaal de lange-termijn-hoeveelheid medicijn in het lichaam.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:seq:8}
Laat $(u_n)$ gedefinieerd zijn door $u_0 = 3$ en
$u_{n+1} = \frac{4u_n - 1}{u_n + 2}$.
\begin{enumerate}
  \item Toon door inductie aan dat $u_n > 1$ voor alle $n \in \N$.
  \item Toon aan dat $v_n = \dfrac{1}{u_n - 1}$ een \omterm{def:g11:seq:arithmetic}{rekenkundige rij}
        definieert.
  \item Leid expliciete formules af voor $v_n$ en $u_n$, en de limiet van
        $(u_n)$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:seq:9}
Voor $n \geq 1$, laat
$H_n = 1 + \frac{1}{2} + \frac{1}{3} + \dots + \frac{1}{n}$.
\begin{enumerate}
  \item Toon aan dat voor alle $n \geq 1$, $H_{2n} - H_n \geq \frac{1}{2}$.
  \item Leid af dat $H_{2^k} \geq 1 + \frac{k}{2}$ voor alle $k \in \N$,
        en concludeer dat $H_n \to +\infty$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:seq:10}
\emph{(Aangrenzende \omterm{def:g12:seq:sequence}{rijen}.)} Twee \omterm{def:g12:seq:sequence}{rijen} $(a_n)$ en $(b_n)$ zijn
\emph{aangrenzend} als $(a_n)$ \omterm{def:g12:seq:monotonic}{stijgend} is, $(b_n)$ \omterm{def:g10:functions:variations}{dalend} is, en
$b_n - a_n \to 0$.
\begin{enumerate}
  \item Toon aan dat voor alle $n$, $a_n \leq b_n$. (Hint: bestudeer de
        monotonie van $(b_n - a_n)$.)
  \item Toon aan dat aangrenzende \omterm{def:g12:seq:sequence}{rijen} beide convergeren, naar dezelfde
        limiet.
  \item Toepassing: toon aan dat de \omterm{def:g12:seq:sequence}{rijen}
        $a_n = \sum_{k=0}^{n} \frac{1}{k!}$ en
        $b_n = a_n + \frac{1}{n \cdot n!}$ ($n \geq 1$) aangrenzend zijn.
        (Hun gemeenschappelijke limiet is het getal $\eu$, bestudeerd in
        \cref{ch:g12:exp}.)
\end{enumerate}
\end{exercise}
