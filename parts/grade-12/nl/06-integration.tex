\chapter{Integratie}\label{ch:g12:integ}

Integratie beantwoordt twee vragen tegelijk: wat is de oppervlakte onder
een kromme, en hoe herstelt men een \omterm{def:g11:func:function}{functie} uit haar
veranderingsverhouding? De fundamentele stelling van de analyse zegt dat
dit dezelfde vraag is --- de diepste en nuttigste ontdekking van de
wiskunde van de zeventiende eeuw.

\section{De integraal van een continue functie}

\begin{definition}[Integraal van een positieve functie]\label{def:g12:integ:area}
Laat $f$ \omterm{def:g12:limcont:continuity}{continu} en positief zijn op $\intcc{a}{b}$. De
\emph{integraal}\index{integraal}
\[
\int_a^b f(x)\,\dd x
\]
is de oppervlakte, in oppervlakte-eenheden, van het gebied begrensd door
de kromme van $f$, de $x$-as, en de verticale lijnen $x = a$ en $x = b$.
\end{definition}

\begin{proposition}[Eigenschappen van de integraal]\label{prop:g12:integ:properties}
Laat $f, g$ \omterm{def:g12:limcont:continuity}{continu} zijn op een \omterm{def:g10:numbers:interval}{interval} dat $a, b, c$ bevat, en
$\lambda \in \R$.
\begin{enumerate}
  \item \emph{Lineariteit:}
        $\displaystyle\int_a^b (f + \lambda g) = \int_a^b f + \lambda \int_a^b g$.
  \item \emph{Relatie van Chasles:}
        $\displaystyle\int_a^c f = \int_a^b f + \int_b^c f$.
  \item \emph{Positiviteit:} als $a \leq b$ en $f \geq 0$ op $\intcc{a}{b}$,
        dan $\displaystyle\int_a^b f \geq 0$; als $f \leq g$, dan
        $\displaystyle\int_a^b f \leq \int_a^b g$.
\end{enumerate}
\end{proposition}

\begin{proof}
Chasles en positiviteit zijn onmiddellijk uit de oppervlakte-interpretatie
(oppervlaktes tellen op wanneer gebieden naast elkaar liggen; een gebied
van positieve hoogte heeft positieve oppervlakte). Vergelijking volgt door
positiviteit toe te passen op $g - f$. Lineariteit is intuïtief voor de
som van positieve \omterm{def:g11:func:function}{functies} (oppervlaktes stapelen) en wordt rigoriseerd
op de universiteit; \emph{ze is hier toegegeven}.
\end{proof}

\section{De fundamentele stelling van de analyse}

\begin{definition}[Primitieve]\label{def:g12:integ:primitive}
Een \emph{primitieve}\index{primitieve} (of antiafgeleide) van $f$ op een
\omterm{def:g10:numbers:interval}{interval} $I$ is een afleidbare \omterm{def:g11:func:function}{functie} $F$ op $I$ zodat $F' = f$.
\end{definition}

\begin{proposition}\label{prop:g12:integ:primunique}
Als $F$ een \omterm{def:g12:integ:primitive}{primitieve} is van $f$ op een \omterm{def:g10:numbers:interval}{interval} $I$, zijn de \omterm{def:g12:integ:primitive}{primitieven}
van $f$ op $I$ precies de \omterm{def:g11:func:function}{functies} $F + c$, $c \in \R$. Gegeven
$x_0 \in I$ en $y_0 \in \R$, bestaat er een unieke \omterm{def:g12:integ:primitive}{primitieve} met
$F(x_0) = y_0$.
\end{proposition}

\begin{proof}
Als $G' = F' = f$, dan $(G - F)' = 0$ op het \omterm{def:g10:numbers:interval}{interval} $I$, dus is
$G - F$ constant.\footnote{Dat een \omterm{def:g11:func:function}{functie} met \omterm{def:g12:deriv:derivative}{afgeleide} nul op een
\omterm{def:g10:numbers:interval}{interval} constant is, volgt uit \cref{thm:g12:deriv:variations}: ze is
zowel \omterm{def:g12:seq:monotonic}{stijgend} als \omterm{def:g10:functions:variations}{dalend}.} Omgekeerd is elke $F + c$ een \omterm{def:g12:integ:primitive}{primitieve}. De
voorwaarde $F(x_0) = y_0$ legt de constante vast.
\end{proof}

\begin{theorem}[Fundamentele stelling van de analyse]\label{thm:g12:integ:ftc}
\index{fundamentele stelling van de analyse}
Laat $f$ \omterm{def:g12:limcont:continuity}{continu} zijn op een \omterm{def:g10:numbers:interval}{interval} $I$ en $a \in I$. De \omterm{def:g11:func:function}{functie}
\[
F \colon x \longmapsto \int_a^x f(t)\,\dd t
\]
is de \omterm{def:g12:integ:primitive}{primitieve} van $f$ op $I$ die in $a$ verdwijnt. Bijgevolg, voor
elke \omterm{def:g12:integ:primitive}{primitieve} $G$ van $f$ en $a, b \in I$:
\[
\int_a^b f(t)\,\dd t = \bigl[G(t)\bigr]_a^b = G(b) - G(a).
\]
\end{theorem}

{\small Het idee van het bewijs: de toename $F(x+h) - F(x)$ is de
oppervlakte van de smalle band (oranje), ingekadererd tussen de rechthoeken
van hoogtes $f(x)$ en $f(x+h)$ op een basis van lengte $h$.}
\end{omfigure}

\begin{proof}[Bewijs wanneer $f$ stijgend is]
Fixeer $x \in I$ en $h > 0$ met $x + h \in I$. Volgens Chasles,
\[
F(x+h) - F(x) = \int_x^{x+h} f(t)\,\dd t .
\]
Omdat $f$ \omterm{def:g12:seq:monotonic}{stijgend} is, $f(x) \leq f(t) \leq f(x+h)$ voor
$t \in \intcc{x}{x+h}$, en door vergelijking van integralen (de \omterm{def:g12:integ:area}{integraal}
van een constante $c$ over een \omterm{def:g10:numbers:interval}{interval} van lengte $h$ is $ch$):
\[
h\,f(x) \leq F(x+h) - F(x) \leq h\,f(x+h),
\]
dus
\[
f(x) \leq \frac{F(x+h) - F(x)}{h} \leq f(x+h).
\]
Als $h \to 0^+$, $f(x+h) \to f(x)$ door \omterm{def:g12:limcont:continuity}{continuïteit}, en de
insluitstelling geeft dat het verschilquotiënt naar $f(x)$ neigt; het
geval $h < 0$ is symmetrisch. Dus $F' = f$, en $F(a) = 0$. Het algemene
geval (niet monotoon) \omterm{def:g12:limcont:continuity}{continu} wordt op de universiteit bewezen.

Ten slotte, als $G$ een willekeurige \omterm{def:g12:integ:primitive}{primitieve} is, $G = F + c$
(\cref{prop:g12:integ:primunique}), dus
$G(b) - G(a) = F(b) - F(a) = \int_a^b f$.
\end{proof}

\begin{example}
$\displaystyle\int_0^1 x^2\,\dd x = \left[\frac{x^3}{3}\right]_0^1 =
\frac13$: de oppervlakte onder de \omterm{def:g11:quad:parabola}{parabool} is een derde van het
eenheidsvierkant, zoals Archimedes wist.
\end{example}

De tabel van \omterm{def:g12:integ:primitive}{primitieven} leest men van de tabel van \omterm{def:g12:deriv:derivative}{afgeleiden}
($u$ is een afleidbare \omterm{def:g11:func:function}{functie}, $c$ een willekeurige constante):
\begin{center}
\renewcommand{\arraystretch}{1.7}
\begin{tabular}{c|c||c|c}
$f(x)$ & \omterm{def:g12:integ:primitive}{primitieve} & $f$ & \omterm{def:g12:integ:primitive}{primitieve} \\
\hline
$x^n \ (n \neq -1)$ & $\dfrac{x^{n+1}}{n+1} + c$ &
$u'u^n \ (n \neq -1)$ & $\dfrac{u^{n+1}}{n+1} + c$\\
$\dfrac1x \ (x > 0)$ & $\ln x + c$ &
$\dfrac{u'}{u} \ (u > 0)$ & $\ln u + c$\\
$\eu^x$ & $\eu^x + c$ &
$u'\eu^u$ & $\eu^u + c$\\
$\cos x$ & $\sin x + c$ &
$\dfrac{u'}{\sqrt u} \ (u>0)$ & $2\sqrt u + c$\\
$\sin x$ & $-\cos x + c$ & &
\end{tabular}
\end{center}

\begin{method}[De vorm $u' \times (\text{iets in } u)$ herkennen]\label{met:g12:integ:uprime}
Om een product te integreren, zoek een factor die de \omterm{def:g12:deriv:derivative}{afgeleide} is van een
binnelfunctie $u$, op een multiplicatieve constante na. Bijvoorbeeld in
$\int_0^1 x\,\eu^{x^2}\dd x$ is de factor $x$ gelijk aan $\frac12 (x^2)'$:
\[
\int_0^1 x\,\eu^{x^2}\dd x = \frac12\left[\eu^{x^2}\right]_0^1
= \frac{\eu - 1}{2}.
\]
\end{method}

\begin{theorem}[Partieel integreren]\label{thm:g12:integ:ibp}
\index{partieel integreren}
Laat $u, v$ \omterm{def:g12:deriv:derivative}{afleidbaar} zijn op $\intcc{a}{b}$ met continue \omterm{def:g12:deriv:derivative}{afgeleiden}. Dan
\[
\int_a^b u'(t)\,v(t)\,\dd t
= \bigl[u(t)\,v(t)\bigr]_a^b - \int_a^b u(t)\,v'(t)\,\dd t .
\]
\end{theorem}

\begin{proof}
De productregel geeft $(uv)' = u'v + uv'$; integreren van beide kanten
over $\intcc{a}{b}$ en de fundamentele stelling toepassen op het
linkerlid geeft $\bigl[uv\bigr]_a^b = \int_a^b u'v + \int_a^b uv'$.
\end{proof}

\begin{example}\label{ex:g12:integ:ibp}
$\displaystyle\int_0^1 t\,\eu^{t}\,\dd t$: neem $u' = \eu^t$, $v = t$,
dus $u = \eu^t$, $v' = 1$:
\[
\int_0^1 t\,\eu^t\,\dd t = \bigl[t\,\eu^t\bigr]_0^1 - \int_0^1 \eu^t\,\dd t
= \eu - (\eu - 1) = 1 .
\]
\end{example}

\section{Toepassingen}

\begin{definition}[Gemiddelde waarde]\label{def:g12:integ:mean}
De \emph{gemiddelde waarde}\index{gemiddelde waarde} van een continue
\omterm{def:g11:func:function}{functie} $f$ op $\intcc{a}{b}$ ($a<b$) is
\[
\mu = \frac{1}{b-a}\int_a^b f(t)\,\dd t .
\]
\end{definition}

\begin{proposition}\label{prop:g12:integ:meanbounds}
Als $m \leq f \leq M$ op $\intcc{a}{b}$, dan $m \leq \mu \leq M$.
\end{proposition}

\begin{proof}
Integreer de ongelijkheden $m \leq f(t) \leq M$ over $\intcc{a}{b}$ en
deel door $b - a > 0$.
\end{proof}

\begin{method}[Oppervlakte tussen twee krommen]\label{met:g12:integ:areabetween}
Als $f \geq g$ op $\intcc{a}{b}$, is de oppervlakte tussen de twee
krommen $\int_a^b \bigl(f(x) - g(x)\bigr)\dd x$. Als de krommen snijden,
knip het \omterm{def:g10:numbers:interval}{interval} bij de snijpunten en integreer $\abs{f - g}$ stuk voor
stuk.
\end{method}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g12:integ:1}
Bereken
\[
\int_1^2 \left(3x^2 - \frac{1}{x^2}\right)\dd x, \qquad
\int_0^{\pi/2} \cos t \,\dd t, \qquad
\int_0^{1} \frac{\dd t}{2t+1} .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:integ:2}
Vind de \omterm{def:g12:integ:primitive}{primitieve} $F$ van $f(x) = x\eu^{x^2}$ op $\R$ zodat
$F(0) = 1$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:integ:3}
Bereken de \omterm{def:g12:integ:mean}{gemiddelde waarde} van $f(t) = \sin t$ op $\intcc{0}{\pi}$, en
interpreteer het resultaat op een \omterm{def:g10:functions:graph}{grafiek}.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:integ:4}
Bereken met \omterm{thm:g12:integ:ibp}{partieel integreren}
\[
\int_1^{\eu} \ln t\,\dd t
\qquad\text{en}\qquad
\int_0^{\pi} t \sin t\,\dd t .
\]
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:integ:5}
Bereken de oppervlakte van het gebied tussen de \omterm{def:g11:quad:parabola}{parabool} $y = x^2$ en de
lijn $y = x + 2$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:integ:6}
Laat $I = \displaystyle\int_0^1 \frac{\dd t}{1 + t}$.
\begin{enumerate}
  \item Bereken $I$.
  \item Voor $n \in \N$, laat
        $I_n = \displaystyle\int_0^1 \frac{t^n}{1+t}\dd t$. Toon aan dat
        $I_n + I_{n+1} = \dfrac{1}{n+1}$, en dat
        $0 \leq I_n \leq \dfrac{1}{n+1}$.
  \item Leid af dat
        $1 - \frac12 + \frac13 - \dots + \frac{(-1)^{n-1}}{n}
        \xrightarrow[n \to +\infty]{} \ln 2$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:integ:7}
De snelheid van een trein (in m/s) tijdens de eerste $100$ seconden na
vertrek wordt gemodelleerd door $v(t) = 30\bigl(1 - \eu^{-t/50}\bigr)$.
Bereken de afgelegde afstand tijdens deze $100$ seconden, en de
\omterm{def:g11:stat:mean}{gemiddelde} snelheid van de trein over het \omterm{def:g10:numbers:interval}{interval}.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:integ:8}
Voor $n \geq 1$, laat $S_n = \dfrac1n \displaystyle\sum_{k=1}^{n} \frac{1}{1 + k/n}$.
\begin{enumerate}
  \item Interpreteer $S_n$ als een oppervlakte van rechthoeken die een
        gebied onder de kromme van $t \mapsto \frac{1}{1+t}$ op
        $\intcc{0}{1}$ benaderen.
  \item Toon met de monotonie van $t \mapsto \frac{1}{1+t}$ aan dat
        \[
        S_n \leq \int_0^1 \frac{\dd t}{1+t} \leq S_n + \frac1n\left(1 - \frac12\right),
        \]
        en leid af $\lim\limits_{n\to+\infty} S_n = \ln 2$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:integ:9}
\emph{(Integralen van Wallis.)} Voor $n \in \N$, laat
$W_n = \displaystyle\int_0^{\pi/2} \sin^n t\,\dd t$.
\begin{enumerate}
  \item Bereken $W_0$ en $W_1$.
  \item Toon door $\sin^{n+2}t = \sin t \cdot \sin^{n+1} t$ te schrijven
        en partieel te integreren aan dat
        $W_{n+2} = \dfrac{n+1}{n+2}\,W_n$.
  \item Leid $W_2$, $W_3$, $W_4$ af en toon aan dat $(W_n)$ \omterm{def:g10:functions:variations}{dalend} en
        positief is.
\end{enumerate}
\end{exercise}
