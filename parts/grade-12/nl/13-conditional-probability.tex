\chapter{Voorwaardelijke kans en onafhankelijkheid}\label{ch:g12:condprob}

Kans rekent onzekerheid; voorwaardelijke kans rekent hoe informatie die
verandert. Leren dat een gebeurtenis $B$ optrad, herschikt de kansen van
alle andere gebeurtenissen --- een mechanisme geformaliseerd door Bayes en
dagelijks misbruikt in rechtbanken en kranten. Dit hoofdstuk zet de regels
van conditionering op en de exacte betekenis van onafhankelijkheid.

\section{Kansruimten (herhaling)}

Een experiment met eindig veel uitkomsten wordt gemodelleerd door een
\emph{uitkomstenruimte} $\Omega$ (de verzameling uitkomsten) en een kans
$\P$ die aan elke gebeurtenis $A \subseteq \Omega$ een getal
$\P(A) \in \intcc{0}{1}$ toekent, additief over disjuncte unies en met
$\P(\Omega) = 1$. Herinner de basisregels:
\[
\P(\bar A) = 1 - \P(A),
\qquad
\P(A \cup B) = \P(A) + \P(B) - \P(A \cap B).
\]
Wanneer alle uitkomsten even waarschijnlijk zijn,
$\P(A) = \frac{\abs A}{\abs\Omega}$ --- en kansen berekenen reduceert tot
de teltechnieken van \cref{ch:g12:comb}.

\section{Voorwaardelijke kans}

\begin{definition}[Voorwaardelijke kans]\label{def:g12:condprob:cond}
Laat $B$ een gebeurtenis zijn met $\P(B) > 0$. De \emph{kans van $A$
gegeven $B$}\index{voorwaardelijke kans} is
\[
\pcond{B}{A} = \frac{\P(A \cap B)}{\P(B)} .
\]
\end{definition}

De afbeelding $A \mapsto \pcond BA$ is zelf een kans (alle regels gelden);
zij stelt de nieuwe kennisstaat voor van iemand die geleerd heeft dat $B$
optrad.

\begin{proposition}[Vermenigvuldigingsregel]\label{prop:g12:condprob:product}
Voor gebeurtenissen met niet-nul kansen:
\[
\P(A \cap B) = \P(B)\,\pcond{B}{A} = \P(A)\,\pcond{A}{B},
\]
en algemener
$\P(A_1 \cap A_2 \cap A_3) = \P(A_1)\,\pcond{A_1}{A_2}\,
\pcond{A_1 \cap A_2}{A_3}$, enz.
\end{proposition}

\begin{proof}
Herschik de definitie; de kettingformule volgt door itereren.
\end{proof}

\begin{theorem}[Wet van de totale kans]\label{thm:g12:condprob:total}
\index{wet van de totale kans}
Laat $B_1, \dots, B_n$ een partitie van $\Omega$ zijn in gebeurtenissen
van niet-nul kans. Voor elke gebeurtenis $A$:
\[
\P(A) = \sum_{i=1}^{n} \P(B_i)\,\pcond{B_i}{A}.
\]
\end{theorem}

\begin{proof}
De verzamelingen $A \cap B_i$ zijn paarsgewijs disjunct met unie $A$, dus
$\P(A) = \sum_i \P(A \cap B_i) = \sum_i \P(B_i)\pcond{B_i}{A}$ volgens de
vermenigvuldigingsregel.
\end{proof}

\begin{method}[Kansbomen]\label{met:g12:condprob:tree}
Een boomdiagram organiseert voorwaardelijke kansen: elke tak draagt de
kans van de volgende gebeurtenis \emph{gegeven} het pad tot nu toe.
\begin{itemize}
  \item De kans van een blad (een volledig pad) is het \emph{product} van
        de kansen langs zijn takken (vermenigvuldigingsregel).
  \item De kans van een gebeurtenis is de \emph{som} van de kansen van de
        blader die haar realiseren (totale kans).
  \item Kansen op takken die één knoop verlaten sommeren tot $1$.
\end{itemize}
\end{method}

\begin{theorem}[Formule van Bayes]\label{thm:g12:condprob:bayes}
\index{formule van Bayes}
Laat $B_1, \dots, B_n$ een partitie van $\Omega$ zijn als hierboven en $A$
een gebeurtenis met $\P(A) > 0$. Dan
\[
\pcond{A}{B_j} =
\frac{\P(B_j)\,\pcond{B_j}{A}}{\sum_{i=1}^{n} \P(B_i)\,\pcond{B_i}{A}} .
\]
\end{theorem}

\begin{proof}
$\pcond{A}{B_j} = \frac{\P(A \cap B_j)}{\P(A)}$; ontwikkel de teller met de
vermenigvuldigingsregel en de noemer met de totale kans.
\end{proof}

\begin{example}[Screeningtest]\label{ex:g12:condprob:test}
Een ziekte treft $1\%$ van een populatie. Een test detecteert haar met kans
$0.99$ (sensitiviteit) en geeft een vals positief met kans $0.05$. Gegeven
een positieve test is de kans om daadwerkelijk de ziekte te hebben
\[
\frac{0.01 \times 0.99}{0.01\times0.99 + 0.99\times0.05}
= \frac{0.0099}{0.0099 + 0.0495} = \frac{1}{6} \approx 0.17 .
\]
Ondanks de accurate test zijn vijf positieven op zes vals --- omdat de
ziekte zeldzaam is. $\pcond{A}{B}$ verwarren met $\pcond{B}{A}$ is de
\emph{aanklagersdrogreden}.
\end{example}

\section{Onafhankelijkheid}

\begin{definition}[Onafhankelijke gebeurtenissen]\label{def:g12:condprob:indep}
Twee gebeurtenissen $A$ en $B$ zijn
\emph{onafhankelijk}\index{onafhankelijkheid} als
\[
\P(A \cap B) = \P(A)\,\P(B) .
\]
Wanneer $\P(B) > 0$, is dit equivalent met $\pcond{B}{A} = \P(A)$: $B$
kennen verandert de kans van $A$ niet.
\end{definition}

\begin{proposition}\label{prop:g12:condprob:indepcompl}
Als $A$ en $B$ onafhankelijk zijn, dan ook $A$ en $\bar B$ (en $\bar A$ en
$\bar B$).
\end{proposition}

\begin{proof}
$\P(A \cap \bar B) = \P(A) - \P(A \cap B) = \P(A) - \P(A)\P(B)
= \P(A)\bigl(1 - \P(B)\bigr) = \P(A)\,\P(\bar B)$.
\end{proof}

\begin{remark}
Verwar niet \emph{onafhankelijk} ($\P(A\cap B) = \P(A)\P(B)$) met
\emph{incompatibel} ($A \cap B = \varnothing$). Twee incompatibele
gebeurtenissen van niet-nul kans zijn nooit onafhankelijk: weten dat de
ene optrad garandeert dat de andere niet optrad.
\end{remark}

\begin{definition}[Onafhankelijke herhalingen]\label{def:g12:condprob:repet}
Wanneer een experiment $n$ keer herhaald wordt zodat de uitkomst van elke
proef de andere niet beïnvloedt, is de kans van een gespecificeerde reeks
uitkomsten het product van de individuele kansen. Dit is het model achter
de binomiale verdeling (\cref{ch:g12:randvar}).
\end{definition}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:g12:condprob:1}
Een kaart wordt getrokken uit een standaard speelkaartendeck van $52$.
Bereken de kans dat het een heer is, gegeven dat het een plaatje is (boer,
vrouw of heer). Zijn de gebeurtenissen ``heer'' en ``harten''
onafhankelijk?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:condprob:2}
Een urn bevat $5$ rode en $3$ blauwe ballen. Twee ballen worden
opeenvolgend getrokken \emph{zonder} teruglegging.
\begin{enumerate}
  \item Teken de kansboom.
  \item Bereken de kans dat beide rood zijn, en de kans dat de tweede rood
        is.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:condprob:3}
Twee eerlijke dobbelstenen worden geworpen. Laat $A$ = ``de som is $7$'',
$B$ = ``de eerste steen toont $3$'', $C$ = ``de som is $6$''. Bepaal of
$A$ en $B$ onafhankelijk zijn, daarna of $B$ en $C$ dat zijn.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:condprob:4}
Een fabriek heeft drie machines die respectievelijk $50\%$, $30\%$ en
$20\%$ van de totale productie leveren, met defectpercentages $1\%$, $2\%$
en $4\%$.
\begin{enumerate}
  \item Welk aandeel van de productie is defect?
  \item Een willekeurig gekozen item is defect. Wat is de kans dat het van
        de derde machine komt?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:condprob:5}
In \cref{ex:g12:condprob:test}, bij welke ziekteprevalentie $p$ (in plaats
van $1\%$) zou een positieve test minstens $90\%$ kans op ziekte betekenen?
Los de ongelijkheid op en becommentarieer.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:condprob:6}
Een oneerlijke munt landt op kop met kans $p \in \intoo{0}{1}$. Zij wordt
drie keer geworpen, worpen onafhankelijk.
\begin{enumerate}
  \item Bereken de kans van de gebeurtenis $E$ = ``precies twee keer kop''.
  \item Voor welke $p$ is $\P(E)$ maximaal?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:condprob:7}
\emph{(Monty Hall.)} Een prijs verbergt zich achter één van drie deuren. U
kiest een deur; de gastheer, die weet waar de prijs is, opent één van de
twee overgebleven deuren, altijd een lege onthullend (kiest willekeurig als
beide leeg zijn), en biedt u aan te wisselen naar de andere gesloten deur.
Met de formule van Bayes, bereken de kans om te winnen als u wisselt, en
als u niet wisselt.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:condprob:8}
Een informatiekanaal zendt bits. Elk bit wordt door ruis omgeklapt met kans
$\varepsilon = 0.1$, onafhankelijk. Om een bit te beschermen, wordt het
drie keer verzonden en gedecodeerd bij meerderheid.
\begin{enumerate}
  \item Bereken de kans dat het gedecodeerde bit fout is.
  \item Het ontvangen woord is $101$. Wat is de kans dat het verzonden bit
        $1$ was? (Neem aan dat $0$ en $1$ met gelijke kans worden
        verzonden.)
\end{enumerate}
\end{exercise}
