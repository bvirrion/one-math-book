\chapter{Combinatorics and Counting}\label{ch:g12:comb}

Combinatorics is the art of counting without listing. Its two elementary
principles --- add the sizes of disjoint alternatives, multiply the numbers
of independent choices --- suffice to count the \omterm{def:g12:comb:tuples}{arrangements}, \omterm{def:g12:comb:permutation}{permutations}
and subsets of a finite set, and culminate in the binomial theorem.

\section{The two counting principles}

We write $\abs{E}$ for the number of elements (the \emph{cardinality}) of a
finite set $E$.

\begin{proposition}[Addition principle]\label{prop:g12:comb:addition}
If a finite set $E$ is partitioned into subsets $A_1, \dots, A_k$ (pairwise
disjoint, with \omterm{def:g10:numbers:interunion}{union} $E$), then
\[
\abs{E} = \abs{A_1} + \abs{A_2} + \dots + \abs{A_k}.
\]
\end{proposition}

\begin{proposition}[Multiplication principle]\label{prop:g12:comb:multiplication}
If an object is built by a succession of $k$ choices, with $n_1$ options for
the first choice and, \emph{whatever the previous choices}, $n_i$ options for
the $i$-th, then the number of objects built is
$n_1 \times n_2 \times \dots \times n_k$.
\end{proposition}

\begin{proof}
Both statements are proved by induction on $k$; the case $k = 2$ of the
second amounts to counting a rectangular array by rows.
\end{proof}

\begin{example}
A restaurant offers 4 starters, 6 mains, 3 desserts: $4 \times 6 \times 3 =
72$ different three-course meals.
\end{example}

\section{Tuples, permutations, factorials}

\begin{definition}[$k$-tuples]\label{def:g12:comb:tuples}
A \emph{$k$-tuple} of a set $E$ is an ordered list $(x_1, \dots, x_k)$ of
elements of $E$, repetitions allowed. A $k$-tuple of \emph{distinct}
elements is an \emph{arrangement}\index{arrangement} of $k$ elements of $E$.
\end{definition}

\begin{proposition}\label{prop:g12:comb:tuples}
Let $\abs E = n$. The number of $k$-tuples of $E$ is $n^k$. The number of
\omterm{def:g12:comb:tuples}{arrangements} of $k$ elements of $E$ ($0 \leq k \leq n$) is
\[
n(n-1)(n-2)\cdots(n-k+1) = \frac{n!}{(n-k)!},
\]
where $n! = 1 \times 2 \times \dots \times n$ (and $0! = 1$) is the
\emph{factorial}\index{factorial} of $n$.
\end{proposition}

\begin{proof}
Multiplication principle: for a $k$-tuple there are $n$ options at each of
the $k$ steps; for an \omterm{def:g12:comb:tuples}{arrangement}, $n$ options for $x_1$, then $n - 1$ for
$x_2$ (one element is used), \dots, $n - k + 1$ for $x_k$.
\end{proof}

\begin{definition}[Permutation]\label{def:g12:comb:permutation}
A \emph{permutation}\index{permutation} of $E$ is an \omterm{def:g12:comb:tuples}{arrangement} of all $n$
elements of $E$: an ordering of $E$. By
\cref{prop:g12:comb:tuples} (case $k = n$), the number of permutations of an
$n$-element set is $n!$.
\end{definition}

\begin{example}
Five runners can finish a race in $5! = 120$ different orders. The number of
possible podiums (first three places) is $5 \times 4 \times 3 = 60$.
\end{example}

\section{Combinations and binomial coefficients}

\begin{definition}[Combinations]\label{def:g12:comb:combination}
A \emph{combination}\index{combination} of $k$ elements of $E$ is a subset
of $E$ with $k$ elements (no order, no repetition). Their number is written
$\dbinom{n}{k}$\index{binomial coefficient}, read ``$n$ choose $k$''.
\end{definition}

\begin{theorem}\label{thm:g12:comb:binomformula}
For $0 \leq k \leq n$:
\[
\binom{n}{k} = \frac{n!}{k!\,(n-k)!} .
\]
\end{theorem}

\begin{proof}
Count the \omterm{def:g12:comb:tuples}{arrangements} of $k$ elements of $E$ in two ways. Directly:
$\frac{n!}{(n-k)!}$. Alternatively, choose first the underlying subset
($\binom nk$ ways), then order it ($k!$ ways); the multiplication principle
gives $\binom{n}{k}\,k!$. Equating,
$\binom nk = \frac{n!}{k!(n-k)!}$.
\end{proof}

\begin{proposition}[Basic identities]\label{prop:g12:comb:identities}
For $0 \leq k \leq n$:
\[
\binom{n}{0} = \binom{n}{n} = 1, \qquad
\binom{n}{1} = n, \qquad
\binom{n}{k} = \binom{n}{n-k},
\]
and \emph{Pascal's rule}\index{Pascal's triangle}: for $1 \leq k \leq n-1$,
\[
\binom{n}{k} = \binom{n-1}{k-1} + \binom{n-1}{k}.
\]
\end{proposition}

\begin{proof}
The symmetry $\binom nk = \binom{n}{n-k}$ holds because taking \omterm{def:g10:proba:operations}{complements}
matches $k$-element subsets with $(n-k)$-element subsets, one for one. For
\omterm{prop:g12:comb:identities}{Pascal's rule}, fix an element $a \in E$ and sort the $k$-element subsets into
those containing $a$ --- obtained by adjoining $a$ to a $(k-1)$-element
subset of $E \setminus \{a\}$, of which there are $\binom{n-1}{k-1}$ --- and
those avoiding $a$, which are the $k$-element subsets of
$E \setminus \{a\}$, numbering $\binom{n-1}{k}$. Conclude by the addition
principle.
\end{proof}

\omterm{prop:g12:comb:identities}{Pascal's rule} generates the coefficients row by row --- \emph{\omterm{prop:g11:binom:pascal}{Pascal's
triangle}}: each entry is the sum of the two above it.

\begin{theorem}[Binomial theorem]\label{thm:g12:comb:binomial}
\index{binomial theorem}
For all $a, b \in \R$ (or $\C$) and $n \in \N$:
\[
(a+b)^n = \sum_{k=0}^{n} \binom{n}{k}\, a^{k}\, b^{\,n-k} .
\]
\end{theorem}

\begin{proof}
Expand the product $(a+b)(a+b)\cdots(a+b)$ ($n$ factors): each term of the
expansion picks $a$ or $b$ in every factor, producing $a^k b^{n-k}$ where
$k$ is the number of factors contributing $a$. The number of ways to choose
these $k$ factors among $n$ is $\binom nk$, which is therefore the
coefficient of $a^k b^{n-k}$.
\end{proof}

\begin{corollary}\label{cor:g12:comb:sums}
$\displaystyle\sum_{k=0}^{n} \binom{n}{k} = 2^n$
\quad and \quad
$\displaystyle\sum_{k=0}^{n} (-1)^k\binom{n}{k} = 0$ \ ($n \geq 1$).
\end{corollary}

\begin{proof}
Take $a = b = 1$, then $a = -1$, $b = 1$ in the binomial theorem. The first
identity also has a direct meaning: an $n$-element set has $2^n$ subsets
(each element is in or out: multiplication principle), sorted by size.
\end{proof}

\begin{method}[Choosing the right model]\label{met:g12:comb:model}
Before counting, answer two questions: \emph{does order matter?} and
\emph{are repetitions allowed?}
\begin{center}
\begin{tabular}{l|cc}
 & order matters & order irrelevant\\
\hline
repetitions allowed & $n^k$ (tuples) & (university)\\
no repetitions & $\frac{n!}{(n-k)!}$ (\omterm{def:g12:comb:tuples}{arrangements}) & $\binom nk$ (subsets)
\end{tabular}
\end{center}
Drawing balls from an urn: \emph{with replacement, in order} $\to$ tuples;
\emph{without replacement, in order} $\to$ \omterm{def:g12:comb:tuples}{arrangements}; \emph{a handful all
at once} $\to$ subsets.
\end{method}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:g12:comb:1}
A license plate consists of 2 letters (A--Z), then 3 digits, then 2 letters.
How many plates are possible? How many have no repeated character?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:comb:2}
Compute $\dbinom{8}{3}$, $\dbinom{10}{8}$, and simplify
$\dfrac{\binom{n}{2}}{\binom{n+1}{2}}$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:comb:3}
In a class of 30 students, one must elect a committee of 4 students, then a
president and a treasurer within the committee (one person cannot hold both
offices). How many outcomes are possible?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:g12:comb:4}
Expand $(x + 2)^5$ and $(1 - x)^6$ using the binomial theorem. What is the
coefficient of $x^3$ in $(2x + 3)^7$?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:comb:5}
A standard poker hand consists of 5 cards from a 52-card deck.
\begin{enumerate}
  \item How many hands are there?
  \item How many hands contain exactly one ace? At least one ace?
  \item How many hands are ``full houses'' (three cards of one rank, two of
        another)?
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:comb:6}
How many anagrams (rearrangements of letters, meaningful or not) does the
word MATH have? The word BANANA?
(Hint for BANANA: first place the three A's.)
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:comb:7}
Prove the identity $k\dbinom{n}{k} = n\dbinom{n-1}{k-1}$ ($1 \leq k \leq n$)
in two ways: by the \omterm{prop:g12:comb:tuples}{factorial} formula, and by counting in two ways the pairs
(committee of $k$ people, its president) chosen from $n$ people.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:comb:8}
A path in the plane goes from $(0,0)$ to $(m, n)$ by unit steps East or
North. Show that the number of such paths is $\dbinom{m+n}{m}$.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:comb:9}
Prove \emph{Vandermonde's identity}: for $0 \leq k \leq m + n$,
\[
\binom{m+n}{k} = \sum_{j=0}^{k} \binom{m}{j}\binom{n}{k-j},
\]
by counting the $k$-element subsets of a set split into a group of $m$ and a
group of $n$. Deduce that
$\displaystyle\sum_{j=0}^{n}\binom{n}{j}^{\!2} = \binom{2n}{n}$.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:comb:10}
Using the binomial theorem, show that for all $n \geq 1$,
\[
\sum_{k=1}^{n} k \binom{n}{k} = n\,2^{n-1}.
\]
(Hint: either differentiate $(1+x)^n$, or use \cref{exo:g12:comb:7}.)
\end{exercise}

\section{Problem: The art of counting twice}

\begin{problem}[{Weekend problem --- stars and bars, deranged
hats, and identities proved by counting one thing two ways}]
\label{pb:g12:comb:1}
The deepest trick in combinatorics is disarmingly simple: count
the same collection twice, by two different methods, and equate
the answers. This problem practices the models of
\cref{met:g12:comb:model}, adds a technique the chapter's course
did not need --- the \emph{stars and bars} of ice-cream counting
--- then counts the famous \emph{deranged hats} exactly, and
finds the number $\frac1\eu$ waiting at the bottom of the hat
pile, its third appearance in this book.

\medskip
\noindent\textbf{Part I --- Choosing the model.}
\begin{enumerate}
  \item Count the license plates made of $2$ letters followed by
        $3$ digits; then the anagrams of BANANA.
  \item From a $32$-card deck, count the $5$-card hands; then
        the hands containing exactly $2$ of the $4$ aces.
  \item A robot walks from $(0,0)$ to $(4,3)$ using only unit
        steps right or up: how many paths? (Encode a path as a
        word in R and U.)
  \item Expand $(1 + x)^4$ by the binomial theorem
        (\cref{thm:g12:comb:binomial}); then evaluate at
        $x = 1$ and $x = -1$: which two identities about the
        numbers $\binom nk$ drop out?
  \item Prove by double counting that
        $k\binom nk = n\binom{n-1}{k-1}$ (count
        committees-with-chair two ways), and deduce
        $\sum_{k=0}^{n} k\binom nk = n\,2^{n-1}$.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Stars and bars.}
\begin{enumerate}[resume]
  \item An ice-cream shop sells $4$ flavors; you order $10$
        scoops (flavors may repeat, order in the cup is
        irrelevant). Encode an order as a row of $10$ stars
        (scoops) separated by $3$ bars (flavor changes), and
        count the orders.
  \item Count the triples of nonnegative \omterm{def:g10:numbers:sets}{integers} with
        $x + y + z = 12$.
  \item Count the triples of \emph{positive} \omterm{def:g10:numbers:sets}{integers} with
        $x + y + z = 12$ (substitute $x = 1 + x'$, etc.).
  \item How many distinct monomials appear in the expansion of
        $(a + b + c)^5$?
  \item Sanity check the method: count the orders of $3$
        scoops from $2$ flavors with the formula, then list
        them all and compare.
  \item Say exactly where ``scoops are identical'' entered the
        encoding --- and count what happens instead if the
        scoops are eaten in order (distinct positions), with
        \cref{met:g12:comb:model}'s checklist.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The deranged hats.}
A \emph{derangement} is a redistribution of $n$ hats to their
$n$ owners in which \emph{nobody} receives their own hat; let
$D_n$ count them. (\Cref{pb:g11:prob:1} showed that one guest
on average recovers their own hat --- now we count the fully
unlucky parties exactly.)
\begin{enumerate}[resume]
  \item Compute $D_1$, $D_2$, $D_3$ by listing, and $D_4$
        patiently (or cleverly).
  \item Justify the recurrence
        $D_n = (n - 1)\left(D_{n-1} + D_{n-2}\right)$: guest 1
        receives some hat $k \neq 1$ ($n - 1$ choices); split
        according to whether guest $k$ receives hat 1 or not.
        Check it reproduces $D_4$, and compute $D_5$.
  \item For $n = 3$, prove by inclusion--exclusion (subtract
        the assignments fixing at least one hat, add back the
        overcounts) that
        $D_3 = 3!\left(1 - \frac{1}{1!} + \frac{1}{2!} -
        \frac{1}{3!}\right)$, and state the general formula.
  \item Compute $\frac{D_5}{5!}$ and compare with
        $\frac1\eu \approx 0.3679$: the \omterm{def:g10:proba:distribution}{probability} that a
        large shuffled party deranges completely is
        $\frac1\eu$ --- the third cameo of this constant, after
        the lottery and the secretary of \cref{pb:g12:exp:1}.
        (Why: the formula of question 14 is the beginning of a
        famous series for $\eu^{-1}$, told in the university
        volumes.)
  \item Secret Santa among $10$ friends: names are drawn
        uniformly at random. What is the \omterm{def:g10:proba:distribution}{probability} the draw
        is valid (nobody draws themselves), and how many
        redraws should the group expect?
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Counting twice, winning twice.}
\begin{enumerate}[resume]
  \item The handshake lemma: at any party, summing over guests
        the number of hands each shook counts every handshake
        exactly twice. Deduce that \emph{the number of guests
        who shook an odd number of hands is always even} ---
        and check the claim makes sense at a three-guest party.
  \item Prove the jewel
        $1^3 + 2^3 + \dots + n^3 = (1 + 2 + \dots + n)^2$ by
        induction, and verify it for $n = 3$. (Little Gauss's
        sum, squared, counts cubes.)
  \item Vandermonde's identity (\cref{exo:g12:comb:9}) via
        paths: interpret $\binom{2n}{n}$ as lattice paths of
        question 3's kind from $(0,0)$ to $(n,n)$, cut each
        path at its crossing of the anti-diagonal, and explain
        how $\sum_j \binom nj^2$ appears.
  \item Finale --- the counter's four moves, one line each with
        an example from this problem: multiply stages and add
        cases; encode cleverly (stars and bars, path words);
        count the same thing twice (committee-with-chair,
        handshakes); subtract the unwanted and correct the
        overcounts (derangements). And note where the counting
        goes to work next: \omterm{def:g10:proba:distribution}{probability}, and the paths of the
        matrices-and-graphs chapter.
\end{enumerate}
\end{problem}
