\chapter{Integration}\label{ch:g12:integ}

Integration answers two questions at once: what is the area under a curve,
and how can one recover a \omterm{def:g11:func:function}{function} from its rate of change? The fundamental
theorem of calculus states that these are the same question --- the deepest
and most useful discovery of seventeenth-century mathematics.

\section{The integral of a continuous function}

\begin{definition}[Integral of a positive function]\label{def:g12:integ:area}
Let $f$ be \omterm{def:g12:limcont:continuity}{continuous} and positive on $\intcc{a}{b}$. The
\emph{integral}\index{integral}
\[
\int_a^b f(x)\,\dd x
\]
is the area, in units of area, of the region bounded by the curve of $f$,
the $x$-axis, and the vertical lines $x = a$ and $x = b$.
\end{definition}

\begin{proposition}[Properties of the integral]\label{prop:g12:integ:properties}
Let $f, g$ be \omterm{def:g12:limcont:continuity}{continuous} on an \omterm{def:g10:numbers:interval}{interval} containing $a, b, c$, and
$\lambda \in \R$.
\begin{enumerate}
  \item \emph{Linearity:}
        $\displaystyle\int_a^b (f + \lambda g) = \int_a^b f + \lambda \int_a^b g$.
  \item \emph{Chasles relation:}
        $\displaystyle\int_a^c f = \int_a^b f + \int_b^c f$.
  \item \emph{Positivity:} if $a \leq b$ and $f \geq 0$ on $\intcc{a}{b}$,
        then $\displaystyle\int_a^b f \geq 0$; if $f \leq g$, then
        $\displaystyle\int_a^b f \leq \int_a^b g$.
\end{enumerate}
\end{proposition}

\begin{proof}
Chasles and positivity are immediate from the area interpretation (areas add
up when regions are juxtaposed; a region of positive height has positive
area). The comparison follows by applying positivity to $g - f$. Linearity is
intuitively clear for the sum of positive \omterm{def:g11:func:function}{functions} (stack the areas) and is
proved rigorously at university; \emph{it is admitted here}.
\end{proof}

\section{The fundamental theorem of calculus}

\begin{definition}[Primitive]\label{def:g12:integ:primitive}
A \emph{primitive}\index{primitive} (or antiderivative) of $f$ on an \omterm{def:g10:numbers:interval}{interval}
$I$ is a \omterm{def:g12:deriv:derivative}{differentiable} \omterm{def:g11:func:function}{function} $F$ on $I$ such that $F' = f$.
\end{definition}

\begin{proposition}\label{prop:g12:integ:primunique}
If $F$ is a \omterm{def:g12:integ:primitive}{primitive} of $f$ on an \omterm{def:g10:numbers:interval}{interval} $I$, the \omterm{def:g12:integ:primitive}{primitives} of $f$ on
$I$ are exactly the \omterm{def:g11:func:function}{functions} $F + c$, $c \in \R$. Given $x_0 \in I$ and
$y_0 \in \R$, there is a unique \omterm{def:g12:integ:primitive}{primitive} with $F(x_0) = y_0$.
\end{proposition}

\begin{proof}
If $G' = F' = f$, then $(G - F)' = 0$ on the \omterm{def:g10:numbers:interval}{interval} $I$, so $G - F$ is
constant.\footnote{That a \omterm{def:g11:func:function}{function} with zero \omterm{def:g12:deriv:derivative}{derivative} on an \omterm{def:g10:numbers:interval}{interval} is
constant follows from \cref{thm:g12:deriv:variations}: it is both \omterm{def:g12:seq:monotonic}{increasing}
and \omterm{def:g10:functions:variations}{decreasing}.} Conversely every $F + c$ is a \omterm{def:g12:integ:primitive}{primitive}. The condition
$F(x_0) = y_0$ pins down the constant.
\end{proof}

\begin{theorem}[Fundamental theorem of calculus]\label{thm:g12:integ:ftc}
\index{fundamental theorem of calculus}
Let $f$ be \omterm{def:g12:limcont:continuity}{continuous} on an \omterm{def:g10:numbers:interval}{interval} $I$ and $a \in I$. The \omterm{def:g11:func:function}{function}
\[
F \colon x \longmapsto \int_a^x f(t)\,\dd t
\]
is the \omterm{def:g12:integ:primitive}{primitive} of $f$ on $I$ vanishing at $a$. Consequently, for any
\omterm{def:g12:integ:primitive}{primitive} $G$ of $f$ and $a, b \in I$:
\[
\int_a^b f(t)\,\dd t = \bigl[G(t)\bigr]_a^b = G(b) - G(a).
\]
\end{theorem}

\begin{proof}[Proof when $f$ is increasing]
Fix $x \in I$ and $h > 0$ with $x + h \in I$. By Chasles,
\[
F(x+h) - F(x) = \int_x^{x+h} f(t)\,\dd t .
\]
Since $f$ is \omterm{def:g12:seq:monotonic}{increasing}, $f(x) \leq f(t) \leq f(x+h)$ for
$t \in \intcc{x}{x+h}$, and by comparison of \omterm{def:g12:integ:area}{integrals} (the \omterm{def:g12:integ:area}{integral} of a
constant $c$ over an \omterm{def:g10:numbers:interval}{interval} of length $h$ is $ch$):
\[
h\,f(x) \leq F(x+h) - F(x) \leq h\,f(x+h),
\]
so
\[
f(x) \leq \frac{F(x+h) - F(x)}{h} \leq f(x+h).
\]
As $h \to 0^+$, $f(x+h) \to f(x)$ by \omterm{def:g12:limcont:continuity}{continuity}, and the squeeze theorem
gives that the difference quotient tends to $f(x)$; the case $h < 0$ is
symmetric. Hence $F' = f$, and $F(a) = 0$. The general (\omterm{def:g12:seq:monotonic}{non-monotonic})
\omterm{def:g12:limcont:continuity}{continuous} case is proved at university.

Finally, if $G$ is any \omterm{def:g12:integ:primitive}{primitive}, $G = F + c$
(\cref{prop:g12:integ:primunique}), so
$G(b) - G(a) = F(b) - F(a) = \int_a^b f$.
\end{proof}

\begin{example}
$\displaystyle\int_0^1 x^2\,\dd x = \left[\frac{x^3}{3}\right]_0^1 =
\frac13$: the area under the \omterm{def:g11:quad:parabola}{parabola} is one third of the unit square, as
Archimedes knew.
\end{example}

The table of \omterm{def:g12:integ:primitive}{primitives} is read from the table of \omterm{def:g12:deriv:derivative}{derivatives}
($u$ denotes a \omterm{def:g12:deriv:derivative}{differentiable} \omterm{def:g11:func:function}{function}, $c$ an arbitrary constant):

$f(x)$ & \omterm{def:g12:integ:primitive}{primitive} & $f$ & \omterm{def:g12:integ:primitive}{primitive} \\
\hline
$x^n \ (n \neq -1)$ & $\dfrac{x^{n+1}}{n+1} + c$ &

\begin{method}[Recognizing the form $u' \times (\text{something in } u)$]\label{met:g12:integ:uprime}
To integrate a product, look for a factor which is the \omterm{def:g12:deriv:derivative}{derivative} of an inner
\omterm{def:g11:func:function}{function} $u$, up to a multiplicative constant. For instance in
$\int_0^1 x\,\eu^{x^2}\dd x$, the factor $x$ is $\frac12 (x^2)'$:
\[
\int_0^1 x\,\eu^{x^2}\dd x = \frac12\left[\eu^{x^2}\right]_0^1
= \frac{\eu - 1}{2}.
\]
\end{method}

\begin{theorem}[Integration by parts]\label{thm:g12:integ:ibp}
\index{integration by parts}
Let $u, v$ be \omterm{def:g12:deriv:derivative}{differentiable} on $\intcc{a}{b}$ with \omterm{def:g12:limcont:continuity}{continuous} \omterm{def:g12:deriv:derivative}{derivatives}.
Then
\[
\int_a^b u'(t)\,v(t)\,\dd t
= \bigl[u(t)\,v(t)\bigr]_a^b - \int_a^b u(t)\,v'(t)\,\dd t .
\]
\end{theorem}

\begin{proof}
The product rule gives $(uv)' = u'v + uv'$; integrating both sides over
$\intcc{a}{b}$ and using the fundamental theorem for the left-hand side
yields $\bigl[uv\bigr]_a^b = \int_a^b u'v + \int_a^b uv'$.
\end{proof}

\begin{example}\label{ex:g12:integ:ibp}
$\displaystyle\int_0^1 t\,\eu^{t}\,\dd t$: take $u' = \eu^t$, $v = t$, so
$u = \eu^t$, $v' = 1$:
\[
\int_0^1 t\,\eu^t\,\dd t = \bigl[t\,\eu^t\bigr]_0^1 - \int_0^1 \eu^t\,\dd t
= \eu - (\eu - 1) = 1 .
\]
\end{example}

\section{Applications}

\begin{definition}[Mean value]\label{def:g12:integ:mean}
The \emph{mean value}\index{mean value} of a \omterm{def:g12:limcont:continuity}{continuous} \omterm{def:g11:func:function}{function} $f$ on
$\intcc{a}{b}$ ($a<b$) is
\[
\mu = \frac{1}{b-a}\int_a^b f(t)\,\dd t .
\]
\end{definition}

\begin{proposition}\label{prop:g12:integ:meanbounds}
If $m \leq f \leq M$ on $\intcc{a}{b}$, then $m \leq \mu \leq M$.
\end{proposition}

\begin{proof}
Integrate the inequalities $m \leq f(t) \leq M$ over $\intcc{a}{b}$ and
divide by $b - a > 0$.
\end{proof}

\begin{method}[Area between two curves]\label{met:g12:integ:areabetween}
If $f \geq g$ on $\intcc{a}{b}$, the area between the two curves is
$\int_a^b \bigl(f(x) - g(x)\bigr)\dd x$. If the curves cross, split the
\omterm{def:g10:numbers:interval}{interval} at the crossing points and integrate $\abs{f - g}$ piece by piece.
\end{method}

\qquad\text{and}\qquad

\]
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:integ:5}
Compute the area of the region between the \omterm{def:g11:quad:parabola}{parabola} $y = x^2$ and the line
$y = x + 2$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:integ:6}
Let $I = \displaystyle\int_0^1 \frac{\dd t}{1 + t}$.
\begin{enumerate}
  \item Compute $I$.
  \item For $n \in \N$, let $I_n = \displaystyle\int_0^1 \frac{t^n}{1+t}\dd t$.
        Show that $I_n + I_{n+1} = \dfrac{1}{n+1}$, and that
        $0 \leq I_n \leq \dfrac{1}{n+1}$.
  \item Deduce that
        $1 - \frac12 + \frac13 - \dots + \frac{(-1)^{n-1}}{n}
        \xrightarrow[n \to +\infty]{} \ln 2$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:g12:integ:7}
The speed of a train (in m/s) during the first $100$ seconds after departure
is modeled by $v(t) = 30\bigl(1 - \eu^{-t/50}\bigr)$. Compute the distance
traveled during these $100$ seconds, and the train's average speed over the
\omterm{def:g10:numbers:interval}{interval}.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:integ:8}
For $n \geq 1$, let $S_n = \dfrac1n \displaystyle\sum_{k=1}^{n} \frac{1}{1 + k/n}$.
\begin{enumerate}
  \item Interpret $S_n$ as an area of rectangles approximating a region
        under the curve of $t \mapsto \frac{1}{1+t}$ on $\intcc{0}{1}$.
  \item Using the monotonicity of $t \mapsto \frac{1}{1+t}$, show that
        \[
        S_n \leq \int_0^1 \frac{\dd t}{1+t} \leq S_n + \frac1n\left(1 - \frac12\right),
        \]
        and deduce $\lim\limits_{n\to+\infty} S_n = \ln 2$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:g12:integ:9}
\emph{(Wallis \omterm{def:g12:integ:area}{integrals}.)} For $n \in \N$, let
$W_n = \displaystyle\int_0^{\pi/2} \sin^n t\,\dd t$.
\begin{enumerate}
  \item Compute $W_0$ and $W_1$.
  \item Writing $\sin^{n+2}t = \sin t \cdot \sin^{n+1} t$ and integrating by
        parts, show that $W_{n+2} = \dfrac{n+1}{n+2}\,W_n$.
  \item Deduce $W_2$, $W_3$, $W_4$ and show that $(W_n)$ is \omterm{def:g10:functions:variations}{decreasing} and
        positive.
\end{enumerate}
\end{exercise}

\section{Problem: Archimedes against the machine}

\begin{problem}[{Weekend problem --- the area under the parabola,
computed three ways across twenty-two centuries, and the
logarithm's secret identity as an area}]
\label{pb:g12:integ:1}
Around 240 BC, Archimedes computed the exact area of a parabolic
segment --- without \omterm{def:g10:coordgeom:system}{coordinates}, without limits, without algebra.
Nineteen centuries later, Riemann's rectangles redid it by brute
squeeze; and the fundamental theorem of calculus
(\cref{thm:g12:integ:ftc}) now does it in one line. This problem
plays all three matches, then uses the same machine to reveal
what the \omterm{def:g12:exp:ln}{logarithm} really is: an area with a scaling symmetry.

\medskip
\noindent\textbf{Part I --- Fluency.}
\begin{enumerate}
  \item Compute $\displaystyle\int_0^1 (3x^2 - 2x + 1)\,\dd x$,
        \ $\displaystyle\int_1^{\eu} \frac{\dd x}{x}$, \ and
        $\displaystyle\int_0^{\pi/2} \cos x\,\dd x$.
  \item Spot the $u'u$-forms
        (\cref{met:g12:integ:uprime}):
        $\displaystyle\int_0^1 x\,\eu^{x^2}\dd x$ and
        $\displaystyle\int_0^1 \frac{2x}{x^2 + 1}\,\dd x$.
  \item By parts (\cref{thm:g12:integ:ibp}):
        $\displaystyle\int_0^1 x\,\eu^x \dd x$ and
        $\displaystyle\int_1^{\eu} \ln x\,\dd x$.
  \item Compute the \omterm{def:g12:integ:mean}{mean value}
        (\cref{def:g12:integ:mean}) of $\sin$ over
        $\intcc{0}{\pi}$ --- and note it is \emph{not}
        $\frac12$.
  \item Compute the area between the line $y = x$ and the
        \omterm{def:g11:quad:parabola}{parabola} $y = x^2$ over $\intcc{0}{1}$
        (\cref{met:g12:integ:areabetween}).
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The \omterm{def:g11:quad:parabola}{parabola}, three ways.}
\begin{enumerate}[resume]
  \item Riemann's setup for the area under $y = x^2$ on
        $\intcc{0}{1}$: write the lower sum $L_n$ and the upper
        sum $U_n$ over $n$ equal rectangles (as in
        \cref{exo:g12:integ:8}).
  \item Prove by induction (\cref{pb:g12:seq:1}'s machine) the
        sum-of-squares formula
        \[
        1^2 + 2^2 + \dots + n^2 = \frac{n(n+1)(2n+1)}{6} .
        \]
  \item Deduce closed forms for $U_n$ and $L_n$, compute their
        common limit, and conclude: the area is $\frac13$.
  \item Now the machine: compute
        $\int_0^1 x^2\,\dd x$ by the fundamental theorem, in
        one line. Compare the efforts.
  \item Archimedes stated it otherwise: \emph{a parabolic
        segment is $\frac43$ of its inscribed triangle}. For
        the segment cut from $y = x^2$ by the chord joining
        $(-1, 1)$ to $(1, 1)$: compute the segment's area with
        an \omterm{def:g12:integ:area}{integral}, the inscribed triangle's area (apex at the
        vertex $(0, 0)$), and verify the master's ratio.
  \item Archimedes' own method: fill the segment with the big
        triangle $T$, then two triangles totalling
        $\frac T4$, then four totalling $\frac{T}{16}$, and so
        on. Sum the geometric series and recover
        $\frac43 T$ --- the endlessly bitten chocolate bar of
        the Middle School volume, eaten by a Greek geometer.
  \item One sentence each: exhaustion (Archimedes), squeeze
        (Riemann), anti-differentiation (Newton--Leibniz) ---
        what does each need, and what does each give?
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The \omterm{def:g12:exp:ln}{logarithm} is an area.}
For $x > 0$, set
$A(x) = \displaystyle\int_1^x \frac{\dd t}{t}$.
\begin{enumerate}[resume]
  \item Give $A(1)$ and $A'(x)$
        (\cref{thm:g12:integ:ftc}), and conclude that $A$ is
        exactly the \omterm{def:g12:exp:ln}{natural logarithm} of
        \cref{def:g12:exp:ln}.
  \item The scaling miracle: fix $a > 0$ and study
        $g(x) = A(ax) - A(x)$. Compute $g'$ (chain rule),
        deduce that $g$ is constant, evaluate the constant ---
        and conclude the functional \omterm{def:g10:algebra:equation}{equation}
        \[
        \ln(ab) = \ln a + \ln b ,
        \]
        proved by pure calculus: the area from $1$ to $ab$
        splits into scaled copies.
  \item Deduce from question 14: $\ln(a^n) = n\ln a$ and
        $\ln\frac1a = -\ln a$.
  \item \Cref{exo:g12:integ:8} reads
        $\frac{1}{n+1} + \frac{1}{n+2} + \dots + \frac{1}{2n}$
        as rectangles under $\frac{1}{1 + t}$: compute this sum
        for $n = 10$ (three decimals) and compare with
        $\ln 2$. Which harmonic-flavored sums, divergent term
        by term, here converge to an area?
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Accumulation.}
\begin{enumerate}[resume]
  \item A car accelerates with speed $v(t) = 3t^2$ m/s for
        $t \in \intcc{0}{10}$. Compute the distance covered,
        the \omterm{def:g11:stat:mean}{mean} speed, and the instant at which the
        instantaneous speed equals the \omterm{def:g11:stat:mean}{mean} speed.
  \item A $4$-meter rod has linear density
        $\rho(x) = 2 + x$ kg/m. Compute its total mass and its
        center of mass
        $\dfrac{\int_0^4 x\,\rho(x)\,\dd x}{\int_0^4
        \rho(x)\,\dd x}$ --- the balance point of the Middle
        School volume's cardboard triangle, finally computed
        with weights that vary.
  \item Wallis's \omterm{def:g12:integ:area}{integrals} (\cref{exo:g12:integ:9}): compute
        $W_0$, $W_1$, and
        $W_2 = \int_0^{\pi/2} \sin^2 t\,\dd t$ using the
        linearization of \cref{pb:g12:trigo:1}. (Their infinite
        ladder climbs, in the university volumes, to a product
        formula for $\pi$ and to the normalization of the bell
        curve.)
  \item Finale --- integration's three faces: an \emph{area}
        by definition, an \emph{accumulation} in the world
        (distance, mass), an \emph{\omterm{def:g12:deriv:derivative}{anti-derivative}} by the
        fundamental theorem; and its history in three names.
        Close with the chapter's pearl: which everyday button
        on the calculator is secretly the area under
        $\frac1t$ --- and which identity did that area prove?
\end{enumerate}
\end{problem}
