\chapter{Differentiation and Convexity}\label{ch:g12:deriv}

The \omterm{def:g12:deriv:derivative}{derivative} measures the instantaneous rate of change of a \omterm{def:g11:func:function}{function}; it is
the \omterm{def:g10:reffunc:affine}{slope} of the \omterm{def:g12:deriv:tangent}{tangent} line to its \omterm{def:g10:functions:graph}{graph}. This chapter reviews and extends
the differentiation rules, links the sign of $f'$ to the variations of $f$,
and introduces \omterm{def:g12:deriv:convex}{convexity}, which is governed by the second \omterm{def:g12:deriv:derivative}{derivative}.

\section{The derivative}

\begin{definition}[Derivative at a point]\label{def:g12:deriv:derivative}
Let $f$ be defined on an \omterm{def:g10:numbers:interval}{interval} $I$ and $a \in I$. The \omterm{def:g11:func:function}{function} $f$ is
\emph{differentiable at $a$}\index{derivative}\index{differentiable} if the difference quotient
\[
\frac{f(a+h) - f(a)}{h}
\]
has a finite limit as $h \to 0$. This limit is called the \emph{derivative of
$f$ at $a$} and written $f'(a)$. If $f$ is differentiable at every point of
$I$, the \omterm{def:g11:func:function}{function} $f' \colon x \mapsto f'(x)$ is the \emph{derivative} of
$f$.
\end{definition}

\begin{definition}[Tangent line]\label{def:g12:deriv:tangent}
If $f$ is \omterm{def:g12:deriv:derivative}{differentiable} at $a$, the \emph{tangent}\index{tangent} to the
curve of $f$ at the point $(a, f(a))$ is the line of \omterm{def:g10:algebra:equation}{equation}
\[
y = f(a) + f'(a)(x - a).
\]
\end{definition}

\begin{proposition}\label{prop:g12:deriv:diffcont}
A \omterm{def:g11:func:function}{function} \omterm{def:g12:deriv:derivative}{differentiable} at $a$ is \omterm{def:g12:limcont:continuity}{continuous} at $a$. The converse is false.
\end{proposition}

\begin{proof}
For $h \neq 0$ small, $f(a+h) - f(a) = h \cdot \frac{f(a+h)-f(a)}{h}$. As
$h \to 0$, the right side tends to $0 \times f'(a) = 0$, so
$f(a+h) \to f(a)$. For the converse, $x \mapsto \abs{x}$ is \omterm{def:g12:limcont:continuity}{continuous} at
$0$, but its difference quotient at $0$ equals $\frac{\abs h}{h} = \pm 1$
depending on the sign of $h$, and has no limit.
\end{proof}

\subsection{Differentiation rules}

\begin{proposition}[Operations]\label{prop:g12:deriv:operations}
Let $u, v$ be \omterm{def:g12:deriv:derivative}{differentiable} on $I$ and $\lambda \in \R$. Then $u + v$,
$\lambda u$, $uv$ are \omterm{def:g12:deriv:derivative}{differentiable} on $I$, as is $\frac{u}{v}$ where
$v \neq 0$, and
\[
(u+v)' = u' + v', \quad (\lambda u)' = \lambda u', \quad
(uv)' = u'v + uv', \quad
\left(\frac{u}{v}\right)' = \frac{u'v - uv'}{v^2}.
\]
\end{proposition}

\begin{proof}[Proof of the product rule]
Write the difference quotient of $uv$ at $a$ as
\[
\frac{u(a+h)v(a+h) - u(a)v(a)}{h}
= \frac{u(a+h) - u(a)}{h}\,v(a+h) + u(a)\,\frac{v(a+h) - v(a)}{h}.
\]
As $h \to 0$, $v(a+h) \to v(a)$ by \omterm{def:g12:limcont:continuity}{continuity}
(\cref{prop:g12:deriv:diffcont}), so the right side tends to
$u'(a)v(a) + u(a)v'(a)$. The other rules are proved similarly.
\end{proof}

\begin{theorem}[Chain rule]\label{thm:g12:deriv:chain}
\index{chain rule}
Let $u$ be \omterm{def:g12:deriv:derivative}{differentiable} on $I$ with values in $J$, and $g$ \omterm{def:g12:deriv:derivative}{differentiable}
on $J$. Then $g \circ u$ is \omterm{def:g12:deriv:derivative}{differentiable} on $I$ and
\[
(g \circ u)' = (g' \circ u) \cdot u' .
\]
In particular, for $u$ \omterm{def:g12:deriv:derivative}{differentiable}:
\[
(u^n)' = n\,u'\,u^{n-1} \ (n \in \Z), \qquad
\left(\sqrt{u}\right)' = \frac{u'}{2\sqrt{u}} \ (u > 0), \qquad
\left(\eu^{u}\right)' = u'\,\eu^{u}.
\]
\end{theorem}

\begin{proof}[Proof sketch]
When $u(a+h) \neq u(a)$ for $h$ small, write
\[
\frac{g(u(a+h)) - g(u(a))}{h}
= \frac{g(u(a+h)) - g(u(a))}{u(a+h) - u(a)} \cdot \frac{u(a+h) - u(a)}{h}.
\]
As $h \to 0$, $u(a+h) \to u(a)$, so the first factor tends to $g'(u(a))$ and
the second to $u'(a)$. (A complete proof must handle the case where
$u(a+h) = u(a)$ for $h$ arbitrarily small; this technical point is treated at
university.)
\end{proof}

\begin{example}
The usual \omterm{def:g12:deriv:derivative}{derivatives}, valid on the natural \omterm{def:g11:func:function}{domains}:
\[
(x^n)' = nx^{n-1}, \quad
\left(\frac{1}{x}\right)' = -\frac{1}{x^2}, \quad
(\sqrt{x})' = \frac{1}{2\sqrt{x}}, \quad
(\cos x)' = -\sin x, \quad
(\sin x)' = \cos x .
\]
The first is proved by induction from the product rule, and the last two in
\cref{ch:g12:trigo}.
\end{example}

\section{Variations and extrema}

\begin{theorem}[Sign of the derivative and variations]\label{thm:g12:deriv:variations}
Let $f$ be \omterm{def:g12:deriv:derivative}{differentiable} on an \omterm{def:g10:numbers:interval}{interval} $I$.
\begin{enumerate}
  \item $f$ is \omterm{def:g12:seq:monotonic}{increasing} on $I$ if and only if $f' \geq 0$ on $I$.
  \item $f$ is \omterm{def:g10:functions:variations}{decreasing} on $I$ if and only if $f' \leq 0$ on $I$.
  \item If $f' > 0$ on $I$ except at finitely many points where it
        vanishes, then $f$ is strictly \omterm{def:g12:seq:monotonic}{increasing} on $I$.
\end{enumerate}
\end{theorem}

\begin{proof}[Partial proof]
If $f$ is \omterm{def:g12:seq:monotonic}{increasing}, every difference quotient
$\frac{f(a+h)-f(a)}{h}$ is $\geq 0$, and passing to the limit gives
$f'(a) \geq 0$. The converse implications rest on the \omterm{def:g11:stat:mean}{mean} value theorem,
proved at university; they are \emph{admitted at this level}.
\end{proof}

\begin{definition}[Local extremum]\label{def:g12:deriv:extremum}
$f$ has a \emph{local maximum}\index{extremum} at $a$ if $f(x) \leq f(a)$ for
all $x$ near $a$; local minima are defined symmetrically.
\end{definition}

\begin{proposition}[First-order condition]\label{prop:g12:deriv:critical}
If $f$ is \omterm{def:g12:deriv:derivative}{differentiable} on an \emph{open} \omterm{def:g10:numbers:interval}{interval} $I$ and has a local
\omterm{def:g12:deriv:extremum}{extremum} at $a \in I$, then $f'(a) = 0$. The converse is false
($f(x) = x^3$ at $a = 0$).
\end{proposition}

\begin{proof}
Say $a$ is a \omterm{def:g12:deriv:extremum}{local maximum}. For $h > 0$ small,
$\frac{f(a+h)-f(a)}{h} \leq 0$, so letting $h \to 0^+$ gives $f'(a) \leq 0$;
for $h < 0$ the quotient is $\geq 0$, giving $f'(a) \geq 0$. Hence
$f'(a) = 0$.
\end{proof}

\begin{method}[Variation table]\label{met:g12:deriv:table}
To study a \omterm{def:g11:func:function}{function} $f$: determine its \omterm{def:g11:func:function}{domain} and limits at the boundary;
compute $f'$ and factor it; determine the sign of $f'$ on each subinterval;
record everything in a \omterm{def:g10:functions:table}{variation table}, marking extrema; deduce the number of
solutions of $f(x) = k$ using the bijection theorem
(\cref{thm:g12:limcont:bijection}).
\end{method}

\section{Convexity}

\begin{definition}[Convex function]\label{def:g12:deriv:convex}
A \omterm{def:g11:func:function}{function} $f$ defined on an \omterm{def:g10:numbers:interval}{interval} $I$ is \emph{convex}\index{convexity}
on $I$ if every chord of its \omterm{def:g10:functions:graph}{graph} lies above the \omterm{def:g10:functions:graph}{graph}: for all
$a, b \in I$ and $t \in \intcc{0}{1}$,
\[
f\bigl((1-t)a + t b\bigr) \leq (1-t) f(a) + t f(b).
\]
It is \emph{concave}\index{concave} if the reverse inequality holds ($-f$ convex).
\end{definition}

\begin{theorem}[Convexity and derivatives]\label{thm:g12:deriv:convexity}
Let $f$ be \omterm{def:g12:deriv:derivative}{differentiable} on an \omterm{def:g10:numbers:interval}{interval} $I$. The following are equivalent:
\begin{enumerate}
  \item $f$ is \omterm{def:g12:deriv:convex}{convex} on $I$;
  \item $f'$ is \omterm{def:g12:seq:monotonic}{increasing} on $I$;
  \item the \omterm{def:g10:functions:graph}{graph} of $f$ lies above each of its \omterm{def:g12:deriv:tangent}{tangents}.
\end{enumerate}
If $f$ is twice \omterm{def:g12:deriv:derivative}{differentiable}, these are also equivalent to $f'' \geq 0$ on
$I$.
\end{theorem}

\begin{proof}[Proof of (2) $\Rightarrow$ (3)]
Fix $a \in I$ and set
$g(x) = f(x) - f(a) - f'(a)(x-a)$, the gap between the curve and the \omterm{def:g12:deriv:tangent}{tangent}
at $a$. Then $g$ is \omterm{def:g12:deriv:derivative}{differentiable} and $g'(x) = f'(x) - f'(a)$. If $f'$ is
\omterm{def:g12:seq:monotonic}{increasing}, $g' \leq 0$ on $I \cap \intoc{-\infty}{a}$ and $g' \geq 0$ on
$I \cap \intco{a}{+\infty}$: $g$ decreases before $a$ and increases after,
so $g$ attains its \omterm{def:g10:functions:extrema}{minimum} $g(a) = 0$, hence $g \geq 0$.
The remaining implications are admitted at this level; the equivalence with
$f'' \geq 0$ follows from \cref{thm:g12:deriv:variations} applied to $f'$.
\end{proof}

\begin{definition}[Inflection point]\label{def:g12:deriv:inflection}
A point where the curve of $f$ crosses its \omterm{def:g12:deriv:tangent}{tangent} is an \emph{inflection
point}\index{inflection point}. For $f$ twice \omterm{def:g12:deriv:derivative}{differentiable}, inflection
points are the points where $f''$ changes sign.
\end{definition}

\begin{example}\label{ex:g12:deriv:cube}
$f(x) = x^3$ has $f''(x) = 6x$: $f$ is \omterm{def:g12:deriv:convex}{concave} on $\intoc{-\infty}{0}$,
\omterm{def:g12:deriv:convex}{convex} on $\intco{0}{+\infty}$, with an \omterm{def:g12:deriv:inflection}{inflection point} at the origin,
where the curve crosses its \omterm{def:g12:deriv:tangent}{tangent} (the $x$-axis).
\end{example}

\begin{example}[Convexity inequalities]\label{ex:g12:deriv:ineq}
The exponential is \omterm{def:g12:deriv:convex}{convex} on $\R$ (its second \omterm{def:g12:deriv:derivative}{derivative} is itself, positive).
Its \omterm{def:g12:deriv:tangent}{tangent} at $0$ is $y = 1 + x$, so
\[
\eu^x \geq 1 + x \quad \text{for all } x \in \R .
\]
Such \emph{\omterm{def:g12:deriv:tangent}{tangent} line inequalities} are a standard product of \omterm{def:g12:deriv:convex}{convexity}.
\end{example}

\begin{method}[Using convexity]\label{met:g12:deriv:convexity}
\begin{itemize}
  \item To prove an inequality $f(x) \geq ax + b$: identify the line as a
        \omterm{def:g12:deriv:tangent}{tangent} of a \omterm{def:g12:deriv:convex}{convex} $f$ and invoke
        \cref{thm:g12:deriv:convexity}(3).
  \item To locate \omterm{def:g12:deriv:inflection}{inflection points}: solve $f''(x) = 0$ \emph{and} check
        that $f''$ changes sign there.
  \item In a \omterm{def:g10:functions:table}{variation table}, \omterm{def:g12:deriv:convex}{convexity} refines the sketch of the curve
        (which way it bends).
\end{itemize}
\end{method}

\section{Problem: The lifeguard's calculation}

\begin{problem}[{Weekend problem --- the fastest rescue path obeys
Snell's law of light, convexity certifies every optimum, and the
ideal soda can is exactly as tall as it is wide}]
\label{pb:g12:deriv:1}
A lifeguard sees a swimmer in trouble --- diagonally down the
beach, out in the water. Running is faster than swimming: the
straight line is \emph{not} the fastest route. The route that
minimizes time bends at the shoreline, and its bending law is
precisely the law by which light refracts entering water: nature,
too, differentiates. This problem trains the chain rule
(\cref{thm:g12:deriv:chain}), performs the rescue, harvests the
inequalities of \omterm{def:g12:deriv:convex}{convexity} (\cref{thm:g12:deriv:convexity}), and
designs a tin can.

\medskip
\noindent\textbf{Part I --- Chain-rule fluency.}
\begin{enumerate}
  \item Differentiate: $\left(3x^2 + 1\right)^5$; \
        $\sqrt{x^2 + 9}$; \ $\dfrac{1}{x^2 + 1}$.
  \item Give the \omterm{def:g12:deriv:tangent}{tangent} to $y = \sqrt{x^2 + 9}$ at $x = 4$.
  \item Build the \omterm{def:g10:functions:table}{variation table} of
        $f(x) = \dfrac{x}{x^2 + 1}$ on $\R$ (extrema included).
  \item For $g(x) = x^3 - 3x^2 + 4$: \omterm{def:g10:functions:table}{variation table} \emph{and}
        \omterm{def:g12:deriv:convex}{convexity} table ($g''$), with the \omterm{def:g12:deriv:inflection}{inflection point}
        (\cref{def:g12:deriv:inflection}).
  \item Compute the \omterm{def:g12:deriv:tangent}{tangent} to $g$ at its \omterm{def:g12:deriv:inflection}{inflection point},
        and show that $g(x) - (\text{tangent}) = (x - 1)^3$:
        what does the sign change say about how the \omterm{def:g12:deriv:tangent}{tangent}
        meets the curve at an inflection?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The rescue.}
Shoreline along the $x$-axis; the lifeguard stands on sand at
$A(0, 30)$, the swimmer waits at $B(40, -20)$ (meters). The
guard runs at $5$ m/s on sand, swims at $2$ m/s, and enters the
water at a point $(x, 0)$ of her choice.
\begin{enumerate}[resume]
  \item Model: express the total rescue time $T(x)$, and say why
        only $x \in \intcc{0}{40}$ deserves consideration.
  \item Differentiate $T$ (chain rule at work --- twice).
  \item Let $\theta_1$ be the angle of the running leg with the
        \emph{perpendicular} to the shore, $\theta_2$ that of
        the swimming leg. Show that
        $\sin\theta_1 = \frac{x}{\sqrt{x^2 + 900}}$ and that
        the condition $T'(x) = 0$ reads
        \[
        \frac{\sin\theta_1}{5} = \frac{\sin\theta_2}{2}
        \]
        --- \emph{Snell's law}, with the two speeds in place of
        the speeds of light in air and water.
  \item The straight segment $[AB]$ crosses the shore at
        $x = 24$. Compute $T(24)$, $T(40)$ (run along the
        beach, then swim straight out) and $T(0)$: is the
        geometric straight line the fastest? Is either extreme
        strategy?
  \item Locate the optimal entry point by \omterm{thm:g12:limcont:ivt}{dichotomy} on
        $T'$ (it is $\approx 33.7$ m): give $x^*$ to the meter
        and the record time $T(x^*)$ to the tenth of a second.
        How much does the optimal bend save over the straight
        line?
  \item Why is the critical point a \emph{global} \omterm{def:g10:functions:extrema}{minimum}?
        (Each term of $T$ is \omterm{def:g12:deriv:convex}{convex} --- admit that a sum of
        \omterm{def:g12:deriv:convex}{convex} \omterm{def:g11:func:function}{functions} is \omterm{def:g12:deriv:convex}{convex} --- and apply
        \cref{exo:g12:deriv:8}.)
  \item Fermat's principle states that light always travels the
        path of least \emph{time}. Deduce why a light ray bends
        exactly at the surface when entering water
        (where light is slower), and name the everyday
        observation this explains about a pencil in a glass of
        water.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- \omterm{def:g12:deriv:convex}{Convexity}'s harvest.}
\begin{enumerate}[resume]
  \item Show that $x \mapsto x^2$ is \omterm{def:g12:deriv:convex}{convex}, and prove the
        \omterm{prop:g10:coordgeom:midpoint}{midpoint} inequality
        $\left(\frac{a + b}{2}\right)^2 \leq
        \frac{a^2 + b^2}{2}$ twice: once from \omterm{def:g12:deriv:convex}{convexity}, once
        by \omterm{def:g10:algebra:expand}{expanding} $(a - b)^2 \geq 0$.
  \item Deduce that the \emph{quadratic \omterm{def:g11:stat:mean}{mean}}
        $\sqrt{\frac{a^2 + b^2}{2}}$ dominates the arithmetic
        \omterm{def:g11:stat:mean}{mean}, and assemble the complete chain of the series ---
        harmonic $\leq$ geometric $\leq$ arithmetic $\leq$
        quadratic --- verifying all four on $a = 2$, $b = 8$.
  \item Tangent-below-the-curve: $x \mapsto \frac1x$ is \omterm{def:g12:deriv:convex}{convex}
        on $\intoo{0}{+\infty}$; write its \omterm{def:g12:deriv:tangent}{tangent} at $1$ and
        deduce the inequality
        $\frac1x \geq 2 - x$ for all $x > 0$. Where is it an
        equality?
  \item Inflection in the wild: during an epidemic, the
        cumulative case count follows an S-shaped curve. What
        happens at its \omterm{def:g12:deriv:inflection}{inflection point}, and why is that date
        the one epidemiologists watch for? Relate to the sign
        of the second \omterm{def:g12:deriv:derivative}{derivative} on each side.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The ideal can.}
A cylindrical can must hold $V = 330$ cm$^3$ with the least
metal: surface $S = 2\pi r^2 + 2\pi r h$, constraint
$\pi r^2 h = V$.
\begin{enumerate}[resume]
  \item Express $S(r) = 2\pi r^2 + \dfrac{2V}{r}$,
        differentiate, and compute the optimal radius and
        height for $V = 330$ (millimeter precision).
  \item Prove the elegant general law: at the optimum,
        $h = 2r$ --- the ideal can is exactly as tall as it is
        wide.
  \item Check $S''(r) > 0$ and conclude (via
        \cref{exo:g12:deriv:8} again) that the optimum is
        global. Real soda cans are visibly taller than wide:
        name a non-mathematical reason.
  \item Finale --- the optimization pipeline: model, ask the
        \omterm{def:g12:deriv:derivative}{derivative}, solve for critical points, certify with
        \omterm{def:g12:deriv:convex}{convexity} or a table, interpret. Run the list over the
        lifeguard, the can, and grade 11's strongest beam
        (\cref{pb:g11:deriv:1}) --- and state what Fermat's
        principle says about who else runs this pipeline.
\end{enumerate}
\end{problem}
