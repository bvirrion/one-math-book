\chapter{Functions of Two Variables}\label{ch:b1:multivar}

The year ends with a first walk into higher dimension: functions
$f(x, y)$ of two real variables. Everything generalizes --- limits,
\omterm{def:b1:continuity:continuous}{continuity}, \omterm{def:b1:derivative:def}{derivatives}, extrema --- but each notion gains a twist:
limits can be approached along every direction at once, \omterm{def:b1:derivative:def}{derivatives}
split into partials, and the \omterm{def:b1:multivar:partial}{gradient} points the way uphill. The
full theory (differentials, general $\R^n$, submanifolds) belongs to
the second year; here we set the vocabulary and the first honest
theorems.

\section{The plane $\R^2$; continuity}

\begin{definition}\label{def:b1:multivar:topology}
On $\R^2$, use the Euclidean norm $\norm{(x,y)} = \sqrt{x^2 + y^2}$
(\cref{ch:b1:euclid}). \omterm{def:b1:topology:open}{Open} balls, \omterm{def:b1:topology:open}{neighborhoods}, \emph{\omterm{def:b1:topology:open}{open}
subsets} of $\R^2$ are defined exactly as in
\cref{ch:b1:topology}, with balls in place of \omterm{prop:b1:reals:intervals}{intervals}. A function
$f \colon U \to \R$ ($U \subseteq \R^2$ \omterm{def:b1:topology:open}{open}) is \emph{\omterm{def:b1:continuity:continuous}{continuous}
at} $a \in U$ when
\[
\forall\varepsilon > 0,\ \exists\delta > 0, \quad
\norm{X - a} \leq \delta \implies \abs{f(X) - f(a)} \leq
\varepsilon,
\]
with the same sequential characterization as in one variable.
Sums, products, quotients and compositions with \omterm{def:b1:continuity:continuous}{continuous}
one-variable functions preserve \omterm{def:b1:continuity:continuous}{continuity}; the coordinate \omterm{def:b1:logic:map}{maps} are
\omterm{def:b1:continuity:continuous}{continuous}, hence so are \omterm{def:b1:poly:def}{polynomials} in $(x,y)$.
\end{definition}

\begin{example}[The polar bound, the clean way to prove a limit]
\label{ex:b1:multivar:polarbound}
Show that $f(x, y) = \dfrac{x^2y^2}{x^2 + y^2}$ (with $f(0,0) =
0$) is \omterm{def:b1:continuity:continuous}{continuous} at the origin. In polar \omterm{prop:b1:vspaces:coordinates}{coordinates} $x =
\rho\cos\theta$, $y = \rho\sin\theta$:
\[
\abs{f} = \frac{\rho^4\cos^2\theta\sin^2\theta}{\rho^2}
= \rho^2\,(\cos\theta\sin\theta)^2 \leq \frac{\rho^2}{4}
\xrightarrow[\rho \to 0]{} 0 ,
\]
a bound \emph{independent of $\theta$}: whatever the direction
of approach, the values are squeezed to $0$. That uniformity in
$\theta$ is the whole point --- a bound like $\abs g =
\abs{\cos\theta\sin\theta}$ (no $\rho$ left) proves nothing,
and indeed that $g$ is the discontinuous radial trap of the
next example.
\end{example}

\begin{example}[The radial trap]\label{ex:b1:multivar:trap}
Let $f(x, y) = \dfrac{xy}{x^2 + y^2}$ for $(x,y) \neq (0,0)$, $f(0,
0) = 0$. Along each axis, $f = 0 \to 0$; but along the diagonal $y
= x$, $f(x, x) = \frac12 \not\to 0$. \emph{No limit at the origin}:
approaching along every line, and even finding the same limit along
each, is not enough (here the line limits disagree; worse examples
agree along all lines yet fail along a parabola,
\cref{exo:b1:multivar:3}). \omterm{def:b1:continuity:continuous}{Continuity} in each variable separately
does not imply \omterm{def:b1:continuity:continuous}{continuity}.
\end{example}

\section{Partial derivatives}

\begin{definition}\label{def:b1:multivar:partial}
The \emph{partial derivatives}\index{partial derivative} of $f$ at
$(a, b)$ are the one-variable \omterm{def:b1:derivative:def}{derivatives} along the axes:
\[
\frac{\partial f}{\partial x}(a,b)
= \lim_{h \to 0} \frac{f(a + h, b) - f(a,b)}{h},
\qquad
\frac{\partial f}{\partial y}(a,b)
= \lim_{k \to 0} \frac{f(a, b + k) - f(a,b)}{k}.
\]
$f$ is of class $C^1$ on $U$ when both exist and are \omterm{def:b1:continuity:continuous}{continuous} on
$U$. The \emph{gradient}\index{gradient} is $\nabla f(a,b) =
\bigl(\frac{\partial f}{\partial x},\, \frac{\partial f}{\partial
y}\bigr)(a,b)$.
\end{definition}

\begin{theorem}[$C^1$ implies a tangent plane]\label{thm:b1:multivar:c1}
Let $f$ be $C^1$ on $U$ and $(a,b) \in U$. Then, as $(h, k) \to
(0,0)$:
\[
f(a + h, b + k) = f(a, b)
+ h\,\frac{\partial f}{\partial x}(a,b)
+ k\,\frac{\partial f}{\partial y}(a,b)
+ o\bigl(\norm{(h,k)}\bigr) .
\]
In particular $f$ is \omterm{def:b1:continuity:continuous}{continuous}, and the graph $z = f(x,y)$ has at
each point the tangent plane read off the formula.
\end{theorem}

\begin{example}[Linear approximation in action]
\label{ex:b1:multivar:linapprox}
Estimate $f(1.02,\ 0.99)$ for $f(x, y) = x^3y^2$. At $(1, 1)$:
$f = 1$, $\frac{\partial f}{\partial x} = 3x^2y^2 = 3$,
$\frac{\partial f}{\partial y} = 2x^3y = 2$, so
\cref{thm:b1:multivar:c1} gives
\[
f(1.02,\ 0.99) \approx 1 + 3\,(0.02) + 2\,(-0.01) = 1.04 ,
\]
against the true value $1.02^3 \times 0.99^2 = 1.04006\dots$ ---
the error is of second order in the increments, as the
$o(\norm{(h,k)})$ promises. The tangent plane to the graph at
$(1, 1, 1)$ is $z = 1 + 3(x - 1) + 2(y - 1)$, the equation
implicit in the estimate.
\end{example}

\begin{proof}
Move one coordinate at a time:
\[
f(a+h, b+k) - f(a,b)
= \bigl[f(a+h, b+k) - f(a, b+k)\bigr] + \bigl[f(a, b+k) -
f(a,b)\bigr].
\]
By the one-variable mean value theorem
(\cref{thm:b1:derivative:mvt}), the first bracket is $h\,
\frac{\partial f}{\partial x}(a + \theta h,\, b + k)$ for some
$\theta \in \intoo{0}{1}$, and the second is $k\,\frac{\partial
f}{\partial y}(a,\, b + \theta' k)$. \omterm{def:b1:continuity:continuous}{Continuity} of the partials at
$(a,b)$ lets us write each as (value at $(a,b)$) $+$
(error $\to 0$); the total error is $h\,\varepsilon_1 +
k\,\varepsilon_2 = o(\norm{(h,k)})$ since $\abs h, \abs k \leq
\norm{(h,k)}$.
\end{proof}

\begin{theorem}[Chain rule]\label{thm:b1:multivar:chain}
Let $f$ be $C^1$ on $U$ and $t \mapsto (x(t), y(t))$ be $C^1$ from
an \omterm{prop:b1:reals:intervals}{interval} into $U$. Then $g(t) = f\bigl(x(t), y(t)\bigr)$ is
$C^1$, with\index{chain rule!two variables}
\[
g'(t) = x'(t)\,\frac{\partial f}{\partial x}\bigl(x(t),y(t)\bigr)
+ y'(t)\,\frac{\partial f}{\partial y}\bigl(x(t),y(t)\bigr)
= \bigl\langle \nabla f,\ (x', y')\bigr\rangle .
\]
\end{theorem}

\begin{proof}
Apply \cref{thm:b1:multivar:c1} at $(x(t), y(t))$ with $(h, k) =
(x(t+s) - x(t),\, y(t+s) - y(t))$: as $s \to 0$, one-variable
differentiability gives $h = s\,x'(t) + o(s)$ and $k = s\,y'(t)
+ o(s)$, so $\norm{(h, k)} = O(s)$ and
\[
g(t+s) - g(t)
= h\,\frac{\partial f}{\partial x} + k\,\frac{\partial
f}{\partial y} + o\bigl(\norm{(h,k)}\bigr)
= s\,\Bigl(x'\,\frac{\partial f}{\partial x} +
y'\,\frac{\partial f}{\partial y}\Bigr) + o(s) ,
\]
the final error absorbing both the $o(s)$'s of $h$ and $k$
(multiplied by the fixed values of the partials) and the
$o(O(s))$ of the tangent-plane estimate. Divide by $s$ and let
$s \to 0$. \omterm{def:b1:continuity:continuous}{Continuity} of $g'$ follows from that of all
ingredients.
\end{proof}

\begin{example}[The chain rule, verified both ways]
\label{ex:b1:multivar:chainrun}
Let $f(x, y) = x^2 y$ and $g(t) = f(t, t^2)$. Directly: $g(t) =
t^2\cdot t^2 = t^4$, so $g'(t) = 4t^3$. By the chain rule:
$\frac{\partial f}{\partial x} = 2xy$ and $\frac{\partial
f}{\partial y} = x^2$, evaluated along the curve $(t, t^2)$:
\[
g'(t) = 1\cdot 2t\cdot t^2 + 2t\cdot t^2 = 2t^3 + 2t^3 = 4t^3 .
\]
The two computations agree, and the split is meaningful: $2t^3$
of the growth comes from moving \emph{rightward} through the
$x$-slope, $2t^3$ from moving \emph{upward} through the
$y$-slope. On curves where no closed form for $g$ exists, only
the second computation survives --- that is the theorem's point.
\end{example}

\begin{remark}[Reading the gradient]
Along a unit direction $u$, the chain rule on $t \mapsto f(a + tu)$
gives the \emph{directional \omterm{def:b1:derivative:def}{derivative}} $\langle \nabla f(a),
u\rangle$: maximal when $u$ points along $\nabla f(a)$ (by
Cauchy--Schwarz, \cref{thm:b1:euclid:cs}). The \omterm{def:b1:multivar:partial}{gradient} is the
direction of steepest ascent, and it is \omterm{def:b1:euclid:orthogonal}{orthogonal} to the level
curves $\{f = c\}$ (differentiate $f$ along a curve drawn in a
level \omterm{def:b1:logic:sets}{set}: the chain rule gives $\langle\nabla f,\,
\text{tangent}\rangle = 0$).
\end{remark}

\begin{example}[Level curves and gradients, on one function]
\label{ex:b1:multivar:levelcurves}
Take $f(x, y) = x^2 - y^2$. Its level \omterm{def:b1:logic:sets}{sets}: $\{f = c\}$ is a
hyperbola opening left-right for $c > 0$, up-down for $c < 0$,
and the crossed pair of lines $y = \pm x$ for $c = 0$ --- the
contour \omterm{def:b1:logic:map}{map} of a mountain pass, with the \omterm{met:b1:multivar:monge}{saddle point} at the
origin where the two zero-level lines cross. \omterm{def:b1:multivar:partial}{Gradient}: $\nabla
f = (2x, -2y)$. At the point $(2, 1)$ (on the level $c = 3$):
$\nabla f = (4, -2)$, while the tangent vector of the level
curve, parametrized near that point by $\bigl(t,
\sqrt{t^2 - 3}\bigr)$, is $\bigl(1, \frac{t}{\sqrt{t^2 -
3}}\bigr) = (1, 2)$ at $t = 2$ --- and indeed
\[
\langle (4, -2),\ (1, 2)\rangle = 4 - 4 = 0 :
\]
\omterm{def:b1:multivar:partial}{gradient} perpendicular to the contour, pointing toward higher
values of $f$ (here: away from the $y$-axis). Two more
readings: the \omterm{def:b1:multivar:partial}{gradient} vanishes exactly at the saddle, where
the contour \omterm{def:b1:logic:map}{map} pinches; and the tangent \emph{line} to the
level curve at $(2,1)$ is $4(x - 2) - 2(y - 1) = 0$, i.e.\ $2x -
y = 3$ --- the equation ``$\langle \nabla f,\ M - M_0\rangle =
0$'' that generalizes the ellipse tangent of
\cref{exo:b1:curves:11}.
\end{example}

\begin{theorem}[Schwarz]\label{thm:b1:multivar:schwarz}
If $f$ is of class $C^2$ (partials of partials exist and are
\omterm{def:b1:continuity:continuous}{continuous}), then\index{Schwarz's theorem}
\[
\frac{\partial^2 f}{\partial x\,\partial y}
= \frac{\partial^2 f}{\partial y\,\partial x} .
\]
\end{theorem}
\admitted

\section{Local extrema}

\begin{method}[Extremum studies, organized]
\label{met:b1:multivar:extremamethod}
\begin{enumerate}
  \item \emph{Solve $\nabla f = 0$ completely.} Factor each
        partial whenever possible (products of \omterm{def:b1:linmaps:def}{linear} factors
        split the system into transparent cases, as in
        \cref{ex:b1:multivar:fourpoints} below); a forgotten
        case is a forgotten \omterm{thm:b1:multivar:critical}{critical point}.
  \item \emph{Classify each point} with the Monge data $r, s,
        t$ --- recomputed \emph{at each point}, never once for
        all.
  \item \emph{If $rt - s^2 = 0$}, examine $f$ directly along
        well-chosen curves through the point (lines first, then
        parabolas), hunting either for two signs (no extremum)
        or for a locked sign with an argument covering all
        directions.
  \item \emph{Step back for the global picture}: check the
        behavior at infinity (a local minimum may be no global
        one), and if the domain is not \omterm{def:b1:topology:open}{open}, treat its \omterm{def:b1:topology:closure}{boundary}
        separately (\cref{exo:b1:multivar:12}) --- the \omterm{thm:b1:multivar:critical}{critical
        point} theorem sees \omterm{def:b1:topology:closure}{interior} points only.
\end{enumerate}
\end{method}

\begin{theorem}[Critical points]\label{thm:b1:multivar:critical}
If $f$ ($C^1$ on the \omterm{def:b1:topology:open}{open set} $U$) has a local extremum at $(a,b)
\in U$, then $\nabla f(a,b) = (0,0)$: the point is
\emph{critical}\index{critical point}.
\end{theorem}

\begin{proof}
The one-variable functions $x \mapsto f(x, b)$ and $y \mapsto f(a,
y)$ have \omterm{def:b1:topology:closure}{interior} local extrema at $a$, resp.\ $b$:
\cref{prop:b1:derivative:fermat} kills both partials.
\end{proof}

\begin{method}[Second-order test (Monge notation)]\label{met:b1:multivar:monge}
At a \omterm{thm:b1:multivar:critical}{critical point} of a $C^2$ function, set
\[
r = \frac{\partial^2 f}{\partial x^2},
\qquad
s = \frac{\partial^2 f}{\partial x \partial y},
\qquad
t = \frac{\partial^2 f}{\partial y^2}
\qquad (\text{values at the point}).
\]
\begin{itemize}
  \item If $rt - s^2 > 0$: local extremum --- minimum for $r > 0$,
        maximum for $r < 0$;
  \item if $rt - s^2 < 0$: no extremum (a \emph{saddle
        point}\index{saddle point});
  \item if $rt - s^2 = 0$: the test is silent; examine directly.
\end{itemize}
(Justification --- a Taylor--Young expansion at order $2$ and the
sign study of the quadratic form $r h^2 + 2shk + tk^2$ --- is
carried out in the second year; the test is used here as a working
tool.)
\end{method}

\begin{example}\label{ex:b1:multivar:extrema}
$f(x,y) = x^3 + y^3 - 3xy$. \omterm{thm:b1:multivar:critical}{Critical points}: $\nabla f = (3x^2 -
3y,\; 3y^2 - 3x) = 0$ gives $y = x^2$ and $x = y^2$, so $x = x^4$:
$x \in \{0, 1\}$: points $(0,0)$ and $(1,1)$.

Second \omterm{def:b1:derivative:def}{derivatives}: $r = 6x$, $s = -3$, $t = 6y$. At $(0,0)$: $rt -
s^2 = -9 < 0$: saddle. At $(1,1)$: $rt - s^2 = 36 - 9 > 0$, $r = 6
> 0$: local minimum, $f(1,1) = -1$. (Not global: $f(x, 0) = x^3 \to
-\infty$.)
\end{example}

\begin{example}[A four-point study, in full]
\label{ex:b1:multivar:fourpoints}
$f(x, y) = xy\,(3 - x - y) = 3xy - x^2y - xy^2$. \omterm{def:b1:multivar:partial}{Gradient}:
\[
\frac{\partial f}{\partial x} = y\,(3 - 2x - y),
\qquad
\frac{\partial f}{\partial y} = x\,(3 - x - 2y).
\]
\omterm{thm:b1:multivar:critical}{Critical points}: if $y = 0$, the second equation gives $x \in
\{0, 3\}$; if $x = 0$, the first gives $y \in \{0, 3\}$; if $xy
\neq 0$, solve $2x + y = 3$, $x + 2y = 3$: $x = y = 1$. Four
points: $(0,0)$, $(3,0)$, $(0,3)$, $(1,1)$. Second \omterm{def:b1:derivative:def}{derivatives}:
$r = -2y$, $s = 3 - 2x - 2y$, $t = -2x$.
\begin{itemize}
  \item $(1,1)$: $r = -2$, $s = -1$, $t = -2$: $rt - s^2 = 3 >
        0$, $r < 0$: local maximum, $f(1,1) = 1$.
  \item $(0,0)$: $r = t = 0$, $s = 3$: $rt - s^2 = -9 < 0$:
        saddle; likewise $(3, 0)$ ($s = -3$) and $(0, 3)$: three
        saddles.
\end{itemize}
The maximum is only local: $f(-T, -T) = T^2(3 + 2T) \to
+\infty$. Symmetry check: $f(x, y) = f(y, x)$, and indeed the
critical \omterm{def:b1:logic:sets}{set} and the classification are symmetric in $x
\leftrightarrow y$. Interpretation: among rectangles-with-slack
$x, y \geq 0$, $x + y \leq 3$, the product $xy(3 - x - y)$ of
the three ``parts'' of $3$ is largest when the parts are equal
--- a two-variable shadow of the AM--GM principle.
\end{example}

\begin{remark}[Common pitfalls]
\label{rem:b1:multivar:pitfalls}
\emph{\omterm{def:b1:multivar:partial}{Partial derivatives} may exist at a point of
discontinuity}: the radial trap $g(x,y) = \frac{xy}{x^2+y^2}$ of
\cref{ex:b1:multivar:trap} has $\frac{\partial g}{\partial
x}(0,0) = \frac{\partial g}{\partial y}(0,0) = 0$ (both axis
restrictions vanish identically), yet $g$ has no limit at the
origin --- partials probe two directions only, \omterm{def:b1:continuity:continuous}{continuity} needs
all of them; only the $C^1$ hypothesis restores order
(\cref{thm:b1:multivar:c1}).
\emph{Line limits never suffice}: \cref{exo:b1:multivar:3}'s
function has limit $0$ along every line and still no limit ---
always try parabolas (or polar bounds valid uniformly in
$\theta$).
\emph{Critical is necessary, not sufficient}: saddles abound
(three of four points in \cref{ex:b1:multivar:fourpoints}); and
the theorem holds on \emph{\omterm{def:b1:topology:open}{open}} \omterm{def:b1:logic:sets}{sets} only --- on a \omterm{def:b1:topology:closed}{closed} disc,
extrema may sit on the \omterm{def:b1:topology:closure}{boundary} with nonzero \omterm{def:b1:multivar:partial}{gradient}
(\cref{exo:b1:multivar:12}).
\emph{The silent case $rt - s^2 = 0$ is genuinely silent}: $x^4
+ y^4$ (minimum) and $x^3 + y^3$ (neither) both have $r = s = t
= 0$ at the origin; only a direct sign study decides
(\cref{exo:b1:multivar:6}, function $h$).
\emph{The \omterm{def:b1:multivar:partial}{gradient} is \omterm{def:b1:euclid:orthogonal}{orthogonal} to level curves, not along
them}: to follow a contour line, move perpendicular to $\nabla
f$; to climb fastest, move along it --- mixing the two reverses
the geometry of every contour \omterm{def:b1:logic:map}{map}.
\end{remark}

\begin{remark}[Where two variables lead]
\label{rem:b1:multivar:whereused}
This chapter is a doorway. The \omterm{def:b1:multivar:partial}{gradient} and the chain rule extend
verbatim to $n$ variables in the Year~2 volume, where the
$o(\norm{(h,k)})$ of \cref{thm:b1:multivar:c1} becomes the
\emph{differential} and the Monge test is proved in full via the
order-two Taylor formula and quadratic forms. The special case
that can be settled \emph{this} year --- quadratic functions,
for which the second-order expansion is exact --- is the subject
of the weekend problem, and it happens to be the case that runs
the world's data fitting: \omterm{pb:b1:multivar:1}{least squares} regression. Constrained
extrema (\cref{exo:b1:multivar:5} was a preview) become Lagrange
multipliers in Year~2; harmonic functions
(\cref{exo:b1:multivar:7}) return in the complex analysis of the
Year~3 volume.
\end{remark}

\label{rem:b1:multivar:perspectives}
This chapter is where the volume's two halves shake hands. The
analysis half supplied its tools one \omterm{def:b1:derivative:def}{derivative} at a time: the
mean value theorem drives \cref{thm:b1:multivar:c1}, Taylor

reconstruct why each of these five hand-offs works has, in
effect, revised the entire year --- which is the real purpose
of this final chapter.
\end{remark}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:b1:multivar:1}
Compute the \omterm{def:b1:multivar:partial}{partial derivatives}:
$f(x,y) = x^2 y + \eu^{xy}$; $\;g(x,y) = \ln(x^2 + y^2)$ (on
$\R^2\setminus\{0\}$); $\;h(x,y) = \arctan\frac yx$ (on $x > 0$).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:multivar:2}
Study the \omterm{def:b1:continuity:continuous}{continuity} at $(0,0)$ (with value $0$ there) of:
\[
f(x,y) = \frac{x^2 y}{x^2 + y^2},
\qquad
g(x,y) = \frac{xy}{x^2 + y^2},
\qquad
h(x,y) = \frac{x^3 + y^3}{x^2 + y^2}.
\]
\emph{(Polar \omterm{prop:b1:vspaces:coordinates}{coordinates} $x = \rho\cos\theta$, $y = \rho\sin\theta$
help: bound by a function of $\rho$ alone when possible.)}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:multivar:3}
Let $f(x,y) = \dfrac{x^2 y}{x^4 + y^2}$ ($f(0,0) = 0$). Prove that
$f$ has limit $0$ at the origin \emph{along every straight line},
but that $f\bigl(x, x^2\bigr) = \frac12$: $f$ is not \omterm{def:b1:continuity:continuous}{continuous} at
$(0,0)$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:multivar:4}
Verify Schwarz's theorem by hand on $f(x, y) = x^3 y^2 + \sin(xy)$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:multivar:5}
Let $f$ be $C^1$ on $\R^2$ and $g(t) = f(\cos t, \sin t)$. Express
$g'(t)$ via the chain rule. Deduce that $f$ restricted to the unit
circle attains extrema at points where $\nabla f$ is parallel to
the radius vector.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:multivar:6}
Find and classify the \omterm{thm:b1:multivar:critical}{critical points} of:
\[
f(x, y) = x^2 + xy + y^2 - 3x,
\qquad
g(x, y) = x^2 - y^2 + 4y,
\qquad
h(x, y) = x^4 + y^4 - 2(x - y)^2 .
\]
\emph{(For $h$, the determinant test is silent at the origin:
examine $h(x, x)$ and $h(x, -x)$.)}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:multivar:7}
A function $f$ is \emph{harmonic} when $\frac{\partial^2 f}{\partial
x^2} + \frac{\partial^2 f}{\partial y^2} = 0$. Check that $x^2 -
y^2$, $xy$, $\eu^x\cos y$ and $\ln(x^2 + y^2)$ (off the origin) are
harmonic.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:multivar:8}
Find the point of the plane $\{x + 2y - z = 4\} \subseteq \R^3$
closest to the origin, two ways: by \omterm{thm:b1:euclid:projection}{orthogonal projection}
(\cref{ch:b1:euclid}), and by minimizing the two-variable function
$f(x, y) = x^2 + y^2 + (x + 2y - 4)^2$ obtained by eliminating $z$.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:multivar:9}
(Laplacian in polar \omterm{prop:b1:vspaces:coordinates}{coordinates}, first contact) Let $f$ be $C^2$ on
$\R^2 \setminus \{0\}$ and \emph{radial}: $f(x, y) =
\varphi\bigl(\sqrt{x^2+y^2}\bigr)$ with $\varphi$ of class $C^2$ on
$\intoo{0}{+\infty}$. Prove that
\[
\frac{\partial^2 f}{\partial x^2} + \frac{\partial^2 f}{\partial
y^2} = \varphi''(\rho) + \frac{\varphi'(\rho)}{\rho},
\qquad \rho = \sqrt{x^2 + y^2},
\]
and find all radial harmonic functions on the punctured plane.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:multivar:10}
Give the equation of the tangent plane to the graph $z = xy$ at
the point $(1, 1, 1)$. Then find all points of the graph of
$f(x,y) = x^3 + y^3 - 3xy$ where the tangent plane is
\emph{horizontal}, and relate the answer to
\cref{ex:b1:multivar:extrema}.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:multivar:11}
Let $f$ be $C^1$ on $\R^2$.
\begin{enumerate}
  \item If $\frac{\partial f}{\partial x} = 0$ everywhere, prove
        that $f(x, y)$ depends only on $y$.
  \item Find all $C^1$ solutions of the equation
        $\frac{\partial f}{\partial x} = \frac{\partial
        f}{\partial y}$ on $\R^2$. \emph{(Set $g(u, v) =
        f\bigl(\frac{u+v}2, \frac{u-v}2\bigr)$ and compute
        $\frac{\partial g}{\partial v}$ by the chain rule.)}
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:multivar:12}
Find the global maximum and minimum of $f(x, y) = xy$ on the
\omterm{def:b1:topology:closed}{closed} disc $x^2 + y^2 \leq 1$. \emph{(Admit the two-variable
extreme value theorem: a \omterm{def:b1:continuity:continuous}{continuous} function on the \omterm{def:b1:topology:closed}{closed} disc
attains its bounds --- proved in the Year~2 volume. Treat the
\omterm{def:b1:topology:open}{open} disc by \omterm{thm:b1:multivar:critical}{critical points} and the \omterm{def:b1:topology:closure}{boundary} circle by the
parametrization of \cref{exo:b1:multivar:5}.)}
\end{exercise}

\section{Problem: least squares and the regression line}

\begin{problem}\label{pb:b1:multivar:1}
Given $n$ data points, which straight line passes ``closest'' to
all of them? Legendre and Gauss answered: the line minimizing the
sum of the \emph{squared} vertical errors --- because that
minimization is exactly solvable by linear algebra. This problem
first proves the Monge test of \cref{met:b1:multivar:monge}
honestly for quadratic functions (the one case where the
second-order expansion is exact), then builds the \omterm{pb:b1:multivar:1}{normal
equations}, the \omterm{pb:b1:multivar:1}{regression line}, and the correlation coefficient
on top of the Euclidean geometry of
\cref{ch:b1:euclid}.\index{least squares}\index{regression line}
\index{normal equations}\index{correlation}

\medskip
\noindent\textbf{Part I --- Quadratic functions: the Monge test,
proved.} Fix reals $r, s, t$ and let $q(h, k) = r h^2 + 2s hk +
t k^2$.
\begin{enumerate}
  \item Suppose $r \neq 0$. Establish the completed-square form
        \[
        q(h, k) = r\Bigl(h + \frac{s}{r}k\Bigr)^{\!2}
        + \frac{rt - s^2}{r}\,k^2 ,
        \]
        and deduce: if $rt - s^2 > 0$, $q$ has the strict sign
        of $r$ off the origin; if $rt - s^2 < 0$, $q$ takes both
        signs.
  \item Settle the remaining cases: $r = 0$, $t \neq 0$
        (symmetric); and $r = t = 0$ ($q = 2shk$). Conclude:
        $q$ takes both signs iff $rt - s^2 < 0$, and $q$
        vanishes only at the origin iff $rt - s^2 > 0$.
  \item Let now $f(x, y) = \frac12\bigl(r x^2 + 2s xy + t
        y^2\bigr) + \beta x + \gamma y + \delta$ be a quadratic
        function with a \omterm{thm:b1:multivar:critical}{critical point} $X_0 = (a, b)$. Prove the
        \emph{exact} expansion
        \[
        f(X_0 + (h,k)) = f(X_0) + \tfrac12\,q(h, k)
        \qquad (\text{no remainder}),
        \]
        and deduce the Monge classification for quadratic
        functions: strict global minimum if $rt - s^2 > 0$, $r >
        0$; strict global maximum if $rt - s^2 > 0$, $r < 0$;
        saddle if $rt - s^2 < 0$.
  \item Suppose $rt - s^2 > 0$ and $r > 0$ (so also $t > 0$).
        Prove the explicit coercivity bound
        \[
        q(h, k) \geq c\,(h^2 + k^2),
        \qquad
        c = \frac{rt - s^2}{r + t} > 0 .
        \]
        \emph{(Show that the quadratic form $q - c(h^2 + k^2)$,
        with coefficients $r - c$, $s$, $t - c$, still has
        nonnegative discriminant-test data.)}
  \item Deduce that a quadratic function with positive definite
        quadratic part tends to $+\infty$ as $\norm{(x,y)} \to
        \infty$, and therefore has a \emph{unique global}
        minimizer: its \omterm{thm:b1:multivar:critical}{critical point}. Contrast with
        \cref{ex:b1:multivar:extrema}, where a local minimum of
        a nonquadratic function was not global.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The \omterm{pb:b1:multivar:1}{normal equations}.} Let $C_1,
C_2, b \in \R^n$ (canonical \omterm{def:b1:euclid:def}{inner product}), and
\[
f(u, v) = \norm{u\,C_1 + v\,C_2 - b}^2 .
\]
\begin{enumerate}[resume]
  \item Expand $f$ and show it is a quadratic function of $(u,
        v)$ with quadratic-part coefficients $r = 2\norm{C_1}^2$,
        $s = 2\langle C_1, C_2\rangle$, $t = 2\norm{C_2}^2$.
  \item Show that $rt - s^2 > 0$ if and only if $(C_1, C_2)$ is
        \omterm{def:b1:vspaces:free}{free} \emph{(the equality case of Cauchy--Schwarz,
        \cref{thm:b1:euclid:cs})}; assume this from now on.
  \item Show that the critical equations $\nabla f = 0$ are the
        \emph{\omterm{pb:b1:multivar:1}{normal equations}}
        \[
        \begin{pmatrix}
        \norm{C_1}^2 & \langle C_1, C_2\rangle\\
        \langle C_1, C_2\rangle & \norm{C_2}^2
        \end{pmatrix}
        \begin{pmatrix} u\\ v\end{pmatrix}
        =
        \begin{pmatrix} \langle C_1, b\rangle\\ \langle C_2,
        b\rangle\end{pmatrix}
        \]
        --- the Gram matrix of \cref{exo:b1:euclid:11} on the
        left --- and that they say exactly: $b - (uC_1 + vC_2)
        \perp C_1, C_2$. Conclude with Part I and
        \cref{thm:b1:euclid:projection}: the unique minimizer
        gives $p = uC_1 + vC_2 = $ the \omterm{thm:b1:euclid:projection}{orthogonal projection} of
        $b$ onto $\operatorname{Vect}(C_1, C_2)$.
  \item Show that the minimal value is $\norm{b}^2 - \langle p,
        b\rangle = \norm{b - p}^2$, and draw the Pythagorean
        picture.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The \omterm{pb:b1:multivar:1}{regression line}.} Data points
$(x_1, y_1), \dots, (x_n, y_n)$, not all $x_i$ equal. Minimize
\[
E(\alpha, \beta) = \sum_{i=1}^{n}\bigl(y_i - \alpha - \beta
x_i\bigr)^2 .
\]
Write $\bar x = \frac1n\sum x_i$, $\bar y = \frac1n\sum y_i$,
$v_x = \frac1n\sum x_i^2 - \bar x^2$, $v_y = \frac1n\sum y_i^2 -
\bar y^2$, $c_{xy} = \frac1n\sum x_i y_i - \bar x\bar y$.
\begin{enumerate}[resume]
  \item Recognize Part II with $C_1 = (1, \dots, 1)$, $C_2 =
        (x_1, \dots, x_n)$, $b = (y_1, \dots, y_n)$, and write
        the \omterm{pb:b1:multivar:1}{normal equations}
        \[
        n\alpha + \Bigl(\sum x_i\Bigr)\beta = \sum y_i,
        \qquad
        \Bigl(\sum x_i\Bigr)\alpha + \Bigl(\sum
        x_i^2\Bigr)\beta = \sum x_i y_i .
        \]
  \item Solve them:
        \[
        \beta = \frac{c_{xy}}{v_x},
        \qquad
        \alpha = \bar y - \beta\,\bar x ,
        \]
        and observe that the \emph{\omterm{pb:b1:multivar:1}{regression line}} $y = \alpha
        + \beta x$ passes through the mean point $(\bar x, \bar
        y)$.
  \item Check $v_x > 0$ exactly when the $x_i$ are not all
        equal, and match this with question 7.
  \item Prove that the minimal error is
        \[
        E_{\min} = n\Bigl(v_y - \frac{c_{xy}^2}{v_x}\Bigr)
        = n\,v_y\,(1 - \rho^2),
        \qquad
        \rho = \frac{c_{xy}}{\sqrt{v_x v_y}}
        \quad (v_y \neq 0).
        \]
        Deduce $\abs\rho \leq 1$, with $\abs\rho = 1$ if and
        only if the data are perfectly aligned.
  \item Worked example: for the data $(0,0)$, $(1,1)$, $(2,3)$,
        $(3,4)$, compute $\bar x, \bar y, v_x, c_{xy}$, the
        \omterm{pb:b1:multivar:1}{regression line}, the four residuals and $E_{\min}$.
  \item Prove that the residuals $\varepsilon_i = y_i - \alpha -
        \beta x_i$ of the optimal line satisfy $\sum_i
        \varepsilon_i = 0$ and $\sum_i x_i\varepsilon_i = 0$,
        and interpret both via orthogonality.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Variations and applications.}
\begin{enumerate}[resume]
  \item (Best constant) Show that the constant $\alpha$
        minimizing $\sum_i (y_i - \alpha)^2$ is the mean $\bar
        y$, and that the minimal value is $n\,v_y$: the variance
        measures the failure of the data to be constant.
  \item (Line through the origin) Show that the slope minimizing
        $\sum_i (y_i - \beta x_i)^2$ is $\beta_0 = \frac{\sum
        x_iy_i}{\sum x_i^2}$, and give a condition on the data
        for $\beta_0$ to coincide with the slope
        $c_{xy}/v_x$ of question 11.
  \item (Distance to a line, again) For a point $M$ and the line
        $D$ through $P_0$ directed by the unit vector $w$,
        minimize $g(\tau) = \norm{P_0 + \tau w - M}^2$ and
        deduce $d(M, D)^2 = \norm{M - P_0}^2 - \langle M - P_0,
        w\rangle^2$; recover the formula $d = \frac{\abs{a x_0 +
        b y_0 + c}}{\sqrt{a^2 + b^2}}$ for the line $ax + by + c
        = 0$ in the plane.
  \item (Analysis of variance) Prove the decomposition $v_y =
        \beta^2 v_x + \frac{E_{\min}}n$: the variance of the
        $y_i$ splits into the part explained by the line plus
        the residual variance.
  \item (Parabolic fit) To fit $y = a + bx + cx^2$, show that
        the \omterm{pb:b1:multivar:1}{normal equations} are the $3 \times 3$ system with
        the moment matrix
        \[
        \begin{pmatrix}
        n & \sum x_i & \sum x_i^2\\
        \sum x_i & \sum x_i^2 & \sum x_i^3\\
        \sum x_i^2 & \sum x_i^3 & \sum x_i^4
        \end{pmatrix},
        \]
        a Gram matrix which is invertible as soon as three of
        the $x_i$ are distinct \emph{(freeness of $(1, X, X^2)$
        sampled at the data, \cref{exo:b1:euclid:11}; compare
        the moment matrices of the weekend problem of
        \cref{ch:b1:det})}.
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Robustness, and synthesis.}
\begin{enumerate}[resume]
  \item Classify the \omterm{thm:b1:multivar:critical}{critical points} of the three quadratic
        functions
        \[
        x^2 + xy + y^2, \qquad x^2 + 3xy + y^2, \qquad
        x^2 + 2xy + y^2 ,
        \]
        by Part I, treating the degenerate third case by hand
        (where is the minimum attained?).
  \item (Outlier experiment) Append the point $(10, 0)$ to the
        data of question 14 and recompute the slope $\beta$.
        What happened, and why is the \emph{squared} error so
        sensitive to one distant point?
  \item (Weighted \omterm{pb:b1:multivar:1}{least squares}) Given weights $w_i > 0$,
        minimize $\sum_i w_i(y_i - \alpha - \beta x_i)^2$: show
        the solution is given by the same formulas as question
        11 with weighted means, variance and covariance
        (define them).
  \item Show that for the optimal line, $E_{\min} = 0$ forces
        all points to lie exactly on it, and connect with the
        equality case $\abs\rho = 1$ of question 13; test on the
        aligned data $(1,1), (2,3), (3,5)$.
  \item Synthesis, in four sentences: why quadratic functions
        are the one class where this chapter's second-order
        test needs no admitted theorem; how the \omterm{pb:b1:multivar:1}{normal equations}
        identify the analytic minimization with the \omterm{thm:b1:euclid:projection}{orthogonal
        projection} of \cref{ch:b1:euclid}; what the correlation
        coefficient measures and which inequality bounds it;
        and which pieces of Chapters 18--23 (bases, Gram and
        moment matrices, \omterm{def:b1:linmaps:projection}{projections}) reappeared. Name the
        method and the equations studied in Parts II--III.
\end{enumerate}
\end{problem}
