\chapter{Integer Arithmetic}\label{ch:b1:arith}

Arithmetic --- the study of \omterm{def:b1:arith:divides}{divisibility} in $\Z$ --- was begun in
the High School volume. This chapter rebuilds it completely from the
Euclidean division, with full proofs: \omterm{thm:b1:arith:gcd}{greatest common divisor} and the
\omterm{met:b1:arith:euclid}{Euclidean algorithm}, Bézout's identity and Gauss's lemma, \omterm{thm:b1:arith:fta}{prime
factorization}, and the calculus of \omterm{def:b1:arith:congruence}{congruences} up to Fermat's little
theorem. Beyond its own charm, this material is the model that
\cref{ch:b1:poly} imitates for polynomials.

\section{Divisibility and Euclidean division}

\begin{definition}[Divisibility]\label{def:b1:arith:divides}
For $a, b \in \Z$, $b$ \emph{divides} $a$ (written $b \mid
a$)\index{divisibility} when $a = bq$ for some $q \in \Z$. Basic
consequences: if $b \mid a$ and $b \mid a'$ then $b \mid (ua + va')$
for all $u, v \in \Z$; if $b \mid a$ and $a \neq 0$ then $\abs b \leq
\abs a$; and $a \mid b$ together with $b \mid a$ force $b = \pm a$.
\end{definition}

\begin{theorem}[Euclidean division]\label{thm:b1:arith:division}
For all $a \in \Z$ and $b \in \N^*$, there is exactly one pair $(q, r)
\in \Z \times \N$ with
\[
a = bq + r, \qquad 0 \leq r < b .
\]
\end{theorem}

\begin{proof}
\emph{Existence.} The \omterm{def:b1:logic:sets}{set} $A = \{a - bk : k \in \Z\} \cap \N$ is a
nonempty subset of $\N$ (take $k = -\abs a$: $a + b\abs a \geq a +
\abs a \geq 0$). Let $r = a - bq$ be its least element. If $r \geq b$,
then $r - b = a - b(q+1)$ would be a smaller element of $A$:
contradiction. So $0 \leq r < b$.

\emph{Uniqueness.} If $bq + r = bq' + r'$ with $0 \leq r, r' < b$,
then $b(q - q') = r' - r$ and $\abs{r' - r} < b$: the multiple of $b$
in the left-hand side must be $0$, so $q = q'$ and $r = r'$.
\end{proof}

\begin{example}[Positional numeration by repeated division]\label{ex:b1:arith:base7}
Write $2026$ in base $7$. Divide repeatedly by $7$, keeping the
remainders:
\[
2026 = 7 \times 289 + 3, \quad
289 = 7 \times 41 + 2, \quad
41 = 7 \times 5 + 6, \quad
5 = 7 \times 0 + 5 .
\]
Reading the remainders from last to first: $2026 =
(5\,6\,2\,3)_7$. Check: $5 \times 343 + 6 \times 49 + 2 \times 7
+ 3 = 1715 + 294 + 14 + 3 = 2026$. The uniqueness of Euclidean
division is exactly what makes each digit \emph{forced}: at every
step the remainder is the only integer in $\intint06$ congruent
to the current value mod $7$, so base-$7$ writing is unique ---
the fact silently used whenever the weekend problem manipulates
``the digits of $n$ in base $p$''.
\end{example}

\section{Greatest common divisor}

\begin{theorem}[Subgroups of $\Z$; existence of the gcd]\label{thm:b1:arith:gcd}
\begin{enumerate}
  \item Every subgroup of $(\Z, +)$ is of the form $n\Z = \{nk : k
        \in \Z\}$ for a unique $n \in \N$.
  \item For $a, b \in \Z$ not both zero, the \omterm{def:b1:logic:sets}{set} $a\Z + b\Z = \{au +
        bv : u, v \in \Z\}$ is a subgroup of $\Z$, hence equals
        $d\,\Z$ for a unique $d \in \N^*$. This $d$ is the
        \emph{greatest common divisor}\index{greatest common divisor}
        $\gcd(a, b)$: it \omterm{def:b1:arith:divides}{divides} $a$ and $b$, and every common
        divisor of $a$ and $b$ \omterm{def:b1:arith:divides}{divides} $d$.
\end{enumerate}
\end{theorem}

\begin{proof}
(1) Let $H \subseteq \Z$ be a subgroup (nonempty, stable under
subtraction; the formal definition is in \cref{ch:b1:structures},
and only these two properties are used). If $H = \{0\}$, take $n = 0$.
Otherwise $H$ contains a nonzero element and its opposite, hence a
smallest strictly positive element $n$. Then $n\Z \subseteq H$. For $x
\in H$, write $x = nq + r$ with $0 \leq r < n$
(\cref{thm:b1:arith:division}); $r = x - nq \in H$, and minimality of
$n$ forces $r = 0$: $x \in n\Z$. Uniqueness: $n$ is the least positive
element of $n\Z$.

(2) $a\Z + b\Z$ contains $0$ and is stable under subtraction, so it is
$d\Z$ with $d \geq 1$ (it contains $a$ or $b$ nonzero). Since $a, b
\in d\Z$, $d$ \omterm{def:b1:arith:divides}{divides} both. And if $c$ \omterm{def:b1:arith:divides}{divides} $a$ and $b$, then $c$
\omterm{def:b1:arith:divides}{divides} every $au + bv$ --- in particular $c \mid d$, since $d \in a\Z
+ b\Z$. This is the property announced (and it implies $\abs c \leq
d$, so $d$ deserves the name \emph{greatest} common divisor).
\end{proof}

\begin{corollary}[Bézout's identity]\label{cor:b1:arith:bezout}
For $a, b$ not both zero, there exist $u, v \in \Z$ with
\[
au + bv = \gcd(a, b) .
\]
In particular ($\gcd(a,b) = 1$, the \emph{coprime}\index{coprime}
case): $a$ and $b$ are \omterm{cor:b1:arith:bezout}{coprime} if and only if $au + bv = 1$ has a
solution.
\end{corollary}

\begin{proof}
$\gcd(a,b) = d \in d\Z = a\Z + b\Z$. For the equivalence: if
$\gcd(a,b) = 1$, Bézout provides the solution; conversely $au + bv =
1$ forces every common divisor of $a, b$ to divide $1$.
\end{proof}

\begin{method}[Euclidean algorithm, extended]\label{met:b1:arith:euclid}
To compute $\gcd(a, b)$ ($a > b > 0$): divide $a = bq + r$; then
$\gcd(a, b) = \gcd(b, r)$ (common divisors of $(a,b)$ and of $(b,r)$
coincide, since $r = a - bq$); iterate until the remainder is $0$; the
last nonzero remainder is the gcd\index{Euclidean algorithm}. Running
the divisions backwards (or maintaining the coefficients on the way
down) produces a Bézout pair $(u, v)$.
\end{method}

\begin{example}\label{ex:b1:arith:euclid}
$\gcd(120, 23)$: $120 = 5 \times 23 + 5$; $23 = 4 \times 5 + 3$; $5 =
1\times 3 + 2$; $3 = 1 \times 2 + 1$; $2 = 2 \times 1 + 0$. So
$\gcd = 1$. Backwards:
\begin{align*}
1 &= 3 - 2 = 3 - (5 - 3) = 2\times 3 - 5 = 2(23 - 4\times 5) - 5 \\
&= 2 \times 23 - 9 \times 5 = 2\times 23 - 9(120 - 5\times 23)
= 47 \times 23 - 9 \times 120 .
\end{align*}
Check: $47 \times 23 = 1081$, $9 \times 120 = 1080$.
\end{example}

\begin{theorem}[Gauss's lemma and consequences]\label{thm:b1:arith:gauss}
Let $a, b, c \in \Z$.
\begin{enumerate}
  \item (Gauss's lemma\index{Gauss's lemma}) If $a \mid bc$ and
        $\gcd(a, b) = 1$, then $a \mid c$.
  \item If $a \mid c$, $b \mid c$ and $\gcd(a,b) = 1$, then $ab \mid
        c$.
  \item If $\gcd(a, b) = \gcd(a, c) = 1$, then $\gcd(a, bc) = 1$.
\end{enumerate}
\end{theorem}

\begin{proof}
(1) Bézout: $au + bv = 1$. Multiply by $c$: $acu + bcv = c$. Both
terms are divisible by $a$ (the second because $a \mid bc$), so $a
\mid c$.

(2) Write $c = aq$; from $b \mid aq$ and $\gcd(a, b) = 1$, point (1)
gives $b \mid q$, so $ab \mid aq = c$.

(3) $au + bv = 1$ and $au' + cv' = 1$. Multiply the two
relations:
\[
1 = (au + bv)(au' + cv')
= a\,\bigl(auu' + ucv' + u'bv\bigr) + bc\,(vv') ,
\]
a Bézout relation between $a$ and $bc$: by
\cref{cor:b1:arith:bezout}, $\gcd(a, bc) = 1$.
\end{proof}

\begin{example}[Solving a linear Diophantine equation]\label{ex:b1:arith:diophantine}
Find all $(x, y) \in \Z^2$ with $6x + 10y = 4$. First, the
\emph{existence test}: $\gcd(6, 10) = 2$ \omterm{def:b1:arith:divides}{divides} $4$, so solutions
exist (if the gcd did not divide the right-hand side, the left side
would always be a multiple of it and there would be none). Divide
through: $3x + 5y = 2$. A particular solution is visible: $(x_0,
y_0) = (-1, 1)$. For the general one, subtract: $3(x + 1) = -5(y -
1)$, so $3 \mid 5(y-1)$, and Gauss's lemma ($\gcd(3,5) = 1$) gives
$3 \mid y - 1$: $y = 1 - 3k$, then $x = -1 + 5k$. Conversely every
such pair works:
\[
(x, y) = (-1 + 5k,\ 1 - 3k), \qquad k \in \Z .
\]
The pattern is general: one particular solution plus the integer
multiples of $\bigl(\frac b{\gcd}, -\frac a{\gcd}\bigr)$ --- the
same ``particular plus homogeneous'' structure as in
\cref{ch:b1:diffeq}, with Gauss's lemma playing the uniqueness
role.
\end{example}

\begin{definition}[Least common multiple]\label{def:b1:arith:lcm}
$\operatorname{lcm}(a, b)$\index{least common multiple} is the
generator in $\N$ of the subgroup $a\Z \cap b\Z$: it is a common
multiple of $a$ and $b$ that \omterm{def:b1:arith:divides}{divides} every common multiple, and for
$a, b \in \N^*$,
\[
\gcd(a,b) \times \operatorname{lcm}(a,b) = ab
\qquad (\text{proof in } \cref{exo:b1:arith:5}).
\]
\end{definition}

\begin{example}[Alignment problems are lcm problems]\label{ex:b1:arith:gears}
Two meshed gears have $84$ and $36$ teeth. After how many teeth of
common motion do they return to their initial position together?
The configuration repeats when the number of elapsed teeth is a
common multiple of $84$ and $36$; the first time is
\[
\operatorname{lcm}(84, 36) = \frac{84 \times 36}{\gcd(84, 36)}
= \frac{3024}{12} = 252
\]
teeth --- that is, $3$ turns of the large gear and $7$ of the
small one ($252/84$ and $252/36$). Note the practical route:
\emph{compute the gcd first} (Euclid: $84 = 2\times36 + 12$, $36
= 3\times12$), then divide --- never build the lcm by listing
multiples. Every periodic-coincidence question (gears, planetary
alignments, repeating decimals meeting) reduces to this one
computation.
\end{example}

\section{Prime numbers}

\begin{definition}\label{def:b1:arith:prime}
An integer $p \geq 2$ is \emph{prime}\index{prime number} when its
only positive divisors are $1$ and $p$. For $p$ prime and $a \in \Z$:
either $p \mid a$, or $\gcd(p, a) = 1$. Consequently
(\cref{thm:b1:arith:gauss}), \emph{Euclid's lemma} holds: if $p \mid
ab$ then $p \mid a$ or $p \mid b$.
\end{definition}

\begin{remark}[Testing primality by trial division]\label{rem:b1:arith:trialdivision}
If $n = ab$ with $2 \leq a \leq b$, then $a^2 \leq ab = n$, so $a
\leq \sqrt n$: a composite $n$ always has a \omterm{def:b1:arith:prime}{prime} divisor $\leq
\sqrt n$. Hence to test whether $n$ is \omterm{def:b1:arith:prime}{prime} it suffices to try
the \omterm{def:b1:arith:prime}{primes} up to $\sqrt n$. For $n = 271$: $\sqrt{271} < 17$, and
$271$ is divisible by none of $2, 3, 5, 7, 11, 13$ (odd, digit
sum $10$, does not end in $0$ or $5$, $271 = 7\cdot38 + 5 =
11\cdot24 + 7 = 13\cdot20 + 11$): \omterm{def:b1:arith:prime}{prime}, after six divisions
instead of two hundred. The $\sqrt n$ barrier is a genuine
threshold: crossing it efficiently for hundred-digit numbers
requires the modern primality tests grown out of
\cref{thm:b1:arith:fermat}.
\end{remark}

\begin{theorem}[Euclid]\label{thm:b1:arith:euclidprimes}
There are infinitely many \omterm{def:b1:arith:prime}{primes}.
\end{theorem}

\begin{proof}
Every integer $n \geq 2$ has a \omterm{def:b1:arith:prime}{prime} divisor: its smallest divisor
$\geq 2$ is \omterm{def:b1:arith:prime}{prime} (a proper factorization of it would produce a
smaller divisor of $n$). Now suppose $p_1, \dots, p_k$ were all the
\omterm{def:b1:arith:prime}{primes}, and let $N = p_1 p_2 \cdots p_k + 1 \geq 2$. Some \omterm{def:b1:arith:prime}{prime} $p_i$
\omterm{def:b1:arith:divides}{divides} $N$; but $p_i$ also \omterm{def:b1:arith:divides}{divides} $N - 1 = p_1\cdots p_k$, so $p_i
\mid 1$ --- absurd.
\end{proof}

\begin{theorem}[Fundamental theorem of arithmetic]\label{thm:b1:arith:fta}
Every integer $n \geq 2$ is a product of \omterm{def:b1:arith:prime}{primes}, and the factorization
\[
n = p_1^{\alpha_1} p_2^{\alpha_2} \cdots p_k^{\alpha_k}
\qquad (p_1 < p_2 < \dots < p_k \text{ primes},\ \alpha_i \in \N^*)
\]
is unique.\index{prime factorization}
\end{theorem}

\begin{proof}
\emph{Existence} by strong induction (\cref{thm:b1:logic:induction}):
$n = 2$ is \omterm{def:b1:arith:prime}{prime}; for $n > 2$, either $n$ is \omterm{def:b1:arith:prime}{prime}, or $n = ab$ with
$2 \leq a, b < n$, and the induction hypothesis factors $a$ and $b$.

\emph{Uniqueness.} Suppose $p_1 \cdots p_r = q_1 \cdots q_s$
(\omterm{def:b1:arith:prime}{primes} listed with repetition, $r \leq s$ say), and induct on
$r$. If $r = 0$ the left side is $1$, forcing $s = 0$ (a nonempty
product of \omterm{def:b1:arith:prime}{primes} exceeds $1$). For $r \geq 1$: the \omterm{def:b1:arith:prime}{prime} $p_1$
\omterm{def:b1:arith:divides}{divides} $q_1(q_2\cdots q_s)$, so by Euclid's lemma either $p_1
\mid q_1$ or $p_1 \mid q_2\cdots q_s$; iterating, $p_1$ \omterm{def:b1:arith:divides}{divides}
some $q_j$. But $q_j$ is \omterm{def:b1:arith:prime}{prime} and $p_1 \geq 2$: necessarily $p_1
= q_j$. Cancel this common factor (legitimate: $\Z$ is an
integral domain) to get
\[
p_2 \cdots p_r = q_1 \cdots \widehat{q_j} \cdots q_s
\]
(the hat marking omission), an equality of shorter products; the
induction hypothesis says the two lists $p_2, \dots, p_r$ and
$q_1, \dots, \widehat{q_j}, \dots, q_s$ coincide up to order,
hence so did the original ones. The exponent form groups equal
\omterm{def:b1:arith:prime}{primes}.
\end{proof}

\begin{proposition}[Valuations]\label{prop:b1:arith:valuation}
For $p$ \omterm{def:b1:arith:prime}{prime} and $n \in \N^*$, write $v_p(n)$\index{p-adic
valuation@$p$-adic valuation} for the exponent of $p$ in the
factorization of $n$ (with $v_p(n) = 0$ if $p \nmid n$). Then
\[
v_p(mn) = v_p(m) + v_p(n),
\qquad
m \mid n \iff \forall p,\ v_p(m) \leq v_p(n),
\]
\[
v_p\bigl(\gcd(m,n)\bigr) = \min\bigl(v_p(m), v_p(n)\bigr),
\qquad
v_p\bigl(\operatorname{lcm}(m,n)\bigr) = \max\bigl(v_p(m),
v_p(n)\bigr).
\]
\end{proposition}

\begin{proof}
The first identity holds because the factorizations multiply and the
factorization of $mn$ is unique. If $m \mid n$, write $n = mq$ and
apply it. Conversely, if all $v_p(m) \leq v_p(n)$, the integer $q =
\prod_p p^{\,v_p(n) - v_p(m)}$ satisfies $mq = n$. The gcd formula:
the integer $d = \prod p^{\min}$ \omterm{def:b1:arith:divides}{divides} both by the criterion, and
every common divisor $c$ has $v_p(c) \leq \min$ for all $p$, so $c
\mid d$; same reasoning for the lcm with $\max$.
\end{proof}

\begin{example}[Squares and cubes through valuations]\label{ex:b1:arith:squarecube}
An integer $n \geq 1$ is a perfect square if and only if every
$v_p(n)$ is even (if $n = m^2$, then $v_p(n) = 2v_p(m)$;
conversely halve every exponent). Likewise for cubes with
multiples of $3$. Thus $21168 = 2^4 \times 3^3 \times 7^2$ is not
a square ($v_3 = 3$ is odd) and not a cube ($v_2 = 4$); the
smallest positive integer $m$ such that $21168\,m$ \emph{is} a
cube is found by topping up each exponent to the next multiple of
$3$:
\[
m = 2^{6-4} \times 3^{3-3} \times 7^{3-2} = 2^2 \times 7 = 28,
\qquad
21168 \times 28 = 2^6\,3^3\,7^3 = (2^2 \times 3 \times 7)^3
= 84^3 .
\]
The insight: multiplicative questions (squares, cubes, divisors,
gcd, lcm) become \emph{coordinatewise} questions on the exponent
vectors $(v_2, v_3, v_5, \dots)$ --- unique factorization is the
\omterm{def:b1:logic:statement}{statement} that these coordinates exist and are well defined.
\end{example}

\section{Congruences}

\begin{definition}\label{def:b1:arith:congruence}
For $n \in \N^*$: $a \equiv b \pmod n$\index{congruence} when $n \mid
a - b$. This is an \omterm{def:b1:logic:equiv}{equivalence relation} compatible with addition and
multiplication: if $a \equiv b$ and $a' \equiv b'$ (mod $n$), then
$a + a' \equiv b + b'$, $aa' \equiv bb'$, and $a^k \equiv b^k$ for
$k \in \N$.
\end{definition}

\begin{example}[Casting out nines]\label{ex:b1:arith:castingnines}
Compatibility with $+$ and $\times$ is a checking device as old
as commerce. Since $10 \equiv 1 \pmod 9$, every integer is
congruent mod $9$ to its digit sum (proved as
\cref{exo:b1:arith:2}). To check the claim $1234 \times 567 =
699\,678$: digit sums give $1234 \equiv 1$ and $567 \equiv 18
\equiv 0 \pmod 9$, so the product must be $\equiv 1 \times 0 =
0$; and indeed $6 + 9 + 9 + 6 + 7 + 8 = 45 \equiv 0$. The check
passes (and the product is in fact correct). Had someone reported
$699\,478$, the digit sum $43 \equiv 7 \not\equiv 0$ would
convict them instantly. The test is one-sided --- it catches an
error unless the error is itself a multiple of $9$ --- which is
exactly the pseudoprime lesson of
\cref{ex:b1:arith:pseudoprime} in miniature: \omterm{def:b1:arith:congruence}{congruence} checks
refute, they do not certify.
\end{example}

\begin{proposition}[Invertibility mod $n$]\label{prop:b1:arith:invmod}
$a$ is \emph{invertible mod $n$} (i.e.\ $ab \equiv 1 \pmod n$ for some
$b$) if and only if $\gcd(a, n) = 1$. The inverse is then unique mod
$n$ and computed by the extended \omterm{met:b1:arith:euclid}{Euclidean algorithm}.
\end{proposition}

\begin{proof}
$ab \equiv 1 \pmod n$ means $ab + nk = 1$ for some $k$: a Bézout
relation, which exists iff $\gcd(a,n) = 1$
(\cref{cor:b1:arith:bezout}). Uniqueness: if $ab \equiv ab' \equiv 1$,
then $b \equiv b(ab') = (ab)b' \equiv b' \pmod n$.
\end{proof}

\begin{example}[Inverting $7$ modulo $26$]\label{ex:b1:arith:inv7mod26}
Since $\gcd(7, 26) = 1$, the class of $7$ is invertible mod $26$.
Extended Euclid:
\[
26 = 3 \times 7 + 5, \qquad
7 = 1 \times 5 + 2, \qquad
5 = 2 \times 2 + 1 ,
\]
then backwards:
\[
1 = 5 - 2 \times 2 = 5 - 2(7 - 5) = 3 \times 5 - 2 \times 7
= 3(26 - 3 \times 7) - 2 \times 7 = 3 \times 26 - 11 \times 7 .
\]
Hence $7 \times (-11) \equiv 1 \pmod{26}$, i.e.\ $7^{-1} \equiv
-11 \equiv 15 \pmod{26}$; check: $7 \times 15 = 105 = 4 \times 26
+ 1$. With the inverse in hand, any \omterm{def:b1:arith:congruence}{congruence} $7x \equiv c
\pmod{26}$ is solved in one multiplication: $x \equiv 15c$. This
mechanical inversion is the workhorse of modular arithmetic ---
and of the public-key protocols mentioned in
\cref{rem:b1:arith:whereused}, where the moduli have hundreds of
digits but the algorithm is exactly this one.
\end{example}

\begin{example}[When the coefficient is not invertible]\label{ex:b1:arith:noninvertible}
Solve $12x \equiv 8 \pmod{20}$. Here $\gcd(12, 20) = 4$, so $12$
is not invertible mod $20$ --- but the equation is still
tractable. The \omterm{def:b1:arith:congruence}{congruence} says $20 \mid 12x - 8$; dividing the
whole relation by $4$ (divisor of all three ingredients), it is
equivalent to $5 \mid 3x - 2$, i.e.\
\[
3x \equiv 2 \pmod 5 .
\]
Now $\gcd(3, 5) = 1$ and $3^{-1} \equiv 2 \pmod 5$ ($3 \times 2 =
6 \equiv 1$), so $x \equiv 4 \pmod 5$: the solutions are $x
\equiv 4, 9, 14, 19 \pmod{20}$ --- \emph{four} classes mod $20$,
matching the gcd. (Had the right side not been divisible by $4$,
say $12x \equiv 6 \pmod{20}$, there would be no solution at all:
the left side is always $\equiv 0 \pmod 4$.) General shape: $ax
\equiv b \pmod n$ is solvable iff $\gcd(a, n) \mid b$, and then
has exactly $\gcd(a, n)$ solution classes --- divide everything
by the gcd and invert.
\end{example}

\begin{theorem}[Fermat's little theorem]\label{thm:b1:arith:fermat}
Let $p$ be \omterm{def:b1:arith:prime}{prime}. For every $a \in \Z$:
\[
a^p \equiv a \pmod p,
\]
and if $p \nmid a$, then $a^{p-1} \equiv 1 \pmod
p$.\index{Fermat's little theorem}
\end{theorem}

\begin{proof}
First, for $1 \leq k \leq p - 1$, the \omterm{def:b1:counting:objects}{binomial coefficient} $\binom pk
= \frac{p!}{k!(p-k)!}$ is divisible by $p$: indeed $k!\,(p-k)!\,
\binom pk = p!$ and $p$ \omterm{def:b1:arith:divides}{divides} $p!$ but is \omterm{cor:b1:arith:bezout}{coprime} to $k!(p-k)!$
(all factors are $< p$), so Gauss's lemma gives $p \mid \binom pk$.

Now prove $a^p \equiv a$ for $a \in \N$ by induction. True for $a =
0$. If $a^p \equiv a$, then by the binomial theorem
\[
(a+1)^p = \sum_{k=0}^{p} \binom pk a^k
\equiv a^p + 1 \equiv a + 1 \pmod p,
\]
all middle terms vanishing mod $p$. For $a < 0$, apply the result to
$-a$ and separate $p = 2$ (where $x \equiv -x$) from odd $p$ (where
$(-a)^p = -a^p$). Finally, if $p \nmid a$, multiply $a^p \equiv a$ by
an inverse of $a$ mod $p$ (\cref{prop:b1:arith:invmod}).
\end{proof}

\begin{example}[Fermat's converse fails: $341$]\label{ex:b1:arith:pseudoprime}
Fermat's little theorem gives a cheap \emph{compositeness} test:
if $a^{n-1} \not\equiv 1 \pmod n$ for some $a$ \omterm{cor:b1:arith:bezout}{coprime} to $n$,
then $n$ is not \omterm{def:b1:arith:prime}{prime}. Could the test also certify primality? No:
take $n = 341 = 11 \times 31$, composite, and $a = 2$. Since
$2^{10} = 1024 = 3 \times 341 + 1$,
\[
2^{10} \equiv 1 \pmod{341}
\qquad\Longrightarrow\qquad
2^{340} = \bigl(2^{10}\bigr)^{34} \equiv 1 \pmod{341} :
\]
the composite $341$ passes Fermat's test for the base $2$ (it is
the smallest such \emph{pseudoprime}). The base $3$ unmasks it
($3^{340} \not\equiv 1$), and practical primality testing
therefore runs the test on several bases, plus refinements ---
the industrial versions of this idea are what certify the large
\omterm{def:b1:arith:prime}{primes} of \cref{rem:b1:arith:whereused}. Moral: an implication
and its converse live separate lives
(\cref{rem:b1:logic:pitfalls}), even for theorems.
\end{example}

\begin{example}[Practical congruence computations]\label{ex:b1:arith:congruences}
What is the remainder of $7^{2026}$ mod $11$? By Fermat, $7^{10}
\equiv 1 \pmod{11}$. Since $2026 = 10 \times 202 + 6$:
\[
7^{2026} \equiv 7^6 = (7^2)^3 = 49^3 \equiv 5^3 = 125 \equiv 4
\pmod{11}.
\]
The remainder is $4$. The strategy: reduce the exponent modulo the
order provided by Fermat, then reduce intermediate powers at every
step.
\end{example}

\begin{remark}[Common pitfalls in arithmetic]\label{rem:b1:arith:pitfalls}
\begin{enumerate}
  \item \emph{Dividing a \omterm{def:b1:arith:congruence}{congruence}.} From $ac \equiv bc \pmod n$
        one may \emph{not} conclude $a \equiv b$ unless $\gcd(c,
        n) = 1$: $6 \equiv 2 \pmod 4$ but $3 \not\equiv 1 \pmod
        4$. The correct general rule \omterm{def:b1:arith:divides}{divides} the \omterm{def:b1:complex:field}{modulus} too: $ac
        \equiv bc \pmod n \iff a \equiv b \pmod{n/\gcd(c,n)}$.
  \item \emph{Misusing Euclid's lemma.} $a \mid bc$ implies $a
        \mid b$ or $a \mid c$ only for $a$ \emph{\omterm{def:b1:arith:prime}{prime}} (or
        \omterm{cor:b1:arith:bezout}{coprime} to one factor): $6 \mid 4 \times 9$ yet $6$
        \omterm{def:b1:arith:divides}{divides} neither factor.
  \item \emph{\omterm{cor:b1:arith:bezout}{Coprime} is a relation, not a property.} ``$8$ and
        $9$ are \omterm{cor:b1:arith:bezout}{coprime}'' is true though neither is \omterm{def:b1:arith:prime}{prime};
        ``pairwise \omterm{cor:b1:arith:bezout}{coprime}'' is stronger than ``globally
        \omterm{cor:b1:arith:bezout}{coprime}'' ($\gcd(6, 10, 15) = 1$ but no pair is
        \omterm{cor:b1:arith:bezout}{coprime}).
  \item \emph{Exponents do not live mod $n$.} In $a^k \bmod n$,
        the exponent may only be reduced modulo the \emph{order}
        of $a$ (for instance $p - 1$ when Fermat applies), never
        modulo $n$: $2^{10} \bmod 11$ is $1$, not $2^{10 \bmod
        11} = 2^{10}$ --- the reduction that works is the one
        \cref{ex:b1:arith:congruences} performs.
\end{enumerate}
\end{remark}

\begin{remark}[Where this chapter is used]\label{rem:b1:arith:whereused}
This chapter is a template as much as a toolbox. The whole chain
--- Euclidean division, gcd, Bézout, Gauss, unique factorization
--- is replayed verbatim for polynomials in \cref{ch:b1:poly},
where ``degree'' plays the role of absolute value; comparing the
two chapters side by side is the best way to understand both. The
\omterm{def:b1:arith:congruence}{congruence} calculus becomes the ring $\Z/n\Z$ in
\cref{ch:b1:structures}, whose invertible elements
(\cref{prop:b1:arith:invmod}) form the first nontrivial example of
a group of units. Valuations return in the weekend problem below
(Legendre's formula) and power the irrationality proofs of
\cref{ch:b1:reals}. Beyond this volume, Bézout inversion mod $n$
is the engine of public-key cryptography, and Fermat's little
theorem is the grandfather of the primality tests that certify the
large \omterm{def:b1:arith:prime}{primes} used there.
\end{remark}

\begin{remark}[Interlude: $\Z$ as a template]\label{rem:b1:arith:interlude}
Step back from the individual theorems and observe the
architecture of the chapter: one tool (Euclidean division)
produced a classification (subgroups $n\Z$), which produced an
existence theorem (gcd, Bézout), which produced a \omterm{def:b1:arith:divides}{divisibility}
calculus (Gauss), which produced unique factorization --- each
floor resting only on the one below. The same building will be
erected twice more in this volume with different ground floors:
in \cref{ch:b1:poly}, where division by degree replaces division
by size and everything above repeats \emph{verbatim}; and, in
miniature, inside every $\Z/n\Z$ of \cref{ch:b1:structures},
where invertibility questions (this chapter's
\cref{prop:b1:arith:invmod}) become structural \omterm{def:b1:logic:statement}{statements} about
rings and fields. Recognizing an argument as ``the $\Z$-argument,
transplanted'' is the fastest way to learn those chapters --- and
the first taste of algebra's core habit, proving theorems about
\emph{axioms} rather than about objects.
\end{remark}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:b1:arith:1}
Compute $\gcd(1\,001, 777)$ by the \omterm{met:b1:arith:euclid}{Euclidean algorithm}, and a Bézout
pair for it.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:arith:2}
Prove the \omterm{def:b1:arith:divides}{divisibility} rules in base $10$: an integer is congruent mod
$9$ to the sum of its digits, and mod $11$ to the alternating sum of
its digits. What is $123\,456\,789$ mod $9$ and mod $11$?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:arith:3}
Solve in $\Z$: $91x \equiv 1 \pmod{237}$ \emph{(extended Euclid)}.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:arith:4}
Find all pairs $(x, y) \in \Z^2$ with $17x + 39y = 1$; then all pairs
with $17 x + 39 y = 5$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:arith:5}
Prove that for $a, b \in \N^*$: $\gcd(a,b) \times
\operatorname{lcm}(a,b) = ab$. \emph{(Use the valuation formulas of
\cref{prop:b1:arith:valuation} and $\min(\alpha,\beta) +
\max(\alpha,\beta) = \alpha + \beta$.)}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:arith:6}
Let $a = 2^{10} \times 3^4 \times 5^2$ and $b = 2^6 \times 3^7 \times
7$. Compute $\gcd(a, b)$, $\operatorname{lcm}(a,b)$, and the number of
positive divisors of $a$. \emph{(Prove the divisor-count formula
$\prod_i (\alpha_i + 1)$.)}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:arith:7}
Prove that $\sqrt p$ is irrational for every \omterm{def:b1:arith:prime}{prime} $p$, using
valuations: compare $v_p$ of both sides of $p q^2 = r^2$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:arith:8}
(Chinese remainder problem) Find all integers $x$ with
\[
x \equiv 2 \pmod 7, \qquad x \equiv 5 \pmod{11}.
\]
Prove along the way that for \omterm{cor:b1:arith:bezout}{coprime} $m, n$, the pair of \omterm{def:b1:arith:congruence}{congruences}
$x \equiv a \ (m)$, $x \equiv b\ (n)$ always has a solution, unique
mod $mn$.\index{Chinese remainder theorem}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:arith:9}
Compute $3^{1000}$ mod $7$, and the last two decimal digits of
$7^{100}$ \emph{(mod $100 = 4 \times 25$: use
\cref{exo:b1:arith:8})}.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:arith:10}
For $m, n \in \N^*$, prove that $\gcd(2^m - 1,\, 2^n - 1) =
2^{\gcd(m,n)} - 1$. \emph{Hint: show first that the remainder of $2^m
- 1$ mod $2^n - 1$ is $2^r - 1$ where $r$ is the remainder of $m$ mod
$n$; then follow the \omterm{met:b1:arith:euclid}{Euclidean algorithm}.}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:arith:11}
(Wilson's theorem) Let $p$ be a \omterm{def:b1:arith:prime}{prime}. Prove that
\[
(p-1)! \equiv -1 \pmod p ,
\]
by pairing each factor of $(p-1)!$ with its inverse mod $p$ and
identifying the self-paired factors (solve $x^2 \equiv 1 \pmod p$
first). Check the converse: if $n \geq 2$ is not \omterm{def:b1:arith:prime}{prime}, then $(n-1)!
\not\equiv -1 \pmod n$.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:arith:12}
(Fermat numbers) For $n \in \N$, let $F_n = 2^{2^n} + 1$.
\begin{enumerate}
  \item Prove that $F_0 F_1 \cdots F_{n-1} = F_n - 2$ for $n \geq
        1$ (induction).
  \item Deduce that the Fermat numbers are pairwise \omterm{cor:b1:arith:bezout}{coprime}.
  \item Deduce a second proof, independent of
        \cref{thm:b1:arith:euclidprimes}, that there are
        infinitely many \omterm{def:b1:arith:prime}{primes}.
\end{enumerate}
\end{exercise}

\section{Problem: Legendre's formula and Kummer's carries}

\begin{problem}\label{pb:b1:arith:1}
How many zeros end the decimal writing of $1000!$ --- and, deeper,
what is the exact power of a \omterm{def:b1:arith:prime}{prime} $p$ dividing $n!$, or dividing a
\omterm{def:b1:counting:objects}{binomial coefficient}? The complete answers are two gems of
elementary arithmetic: \emph{Legendre's formula}
\index{Legendre's formula} $v_p(n!) = \sum_{k\geq1}
\lfloor n/p^k \rfloor$, with its digital avatar $v_p(n!) = \frac{n
- s_p(n)}{p-1}$, and \emph{Kummer's theorem}\index{Kummer's
theorem}: $v_p\binom{m+n}m$ counts the \emph{carries} when adding
$m$ and $n$ in base $p$. This problem proves both, checks them
against each other numerically, and harvests the classical
consequences --- trailing zeros, the parity of Pascal's triangle,
and a first bound in the direction of the \omterm{def:b1:arith:prime}{prime number} theorem.
Throughout, $p$ is a \omterm{def:b1:arith:prime}{prime}, $\floor{x}$ is the integer part, and
$s_p(n)$ denotes the sum of the digits of $n$ written in base $p$.

\medskip
\noindent\textbf{Part I --- Floors, valuations, and Legendre's
formula.}
\begin{enumerate}
  \item Warm-up: compute $10!$ and read off its number of trailing
        zeros; compute $v_2(10!)$ and $v_5(10!)$ directly from the
        factorization of each factor $1, 2, \dots, 10$.
  \item Prove that for $x \in \R$ and $n \in \N^*$,
        $\bigl\lfloor \lfloor x \rfloor / n \bigr\rfloor =
        \lfloor x/n \rfloor$.
  \item Prove that $v_p(a + b) \geq \min\bigl(v_p(a),
        v_p(b)\bigr)$ for all $a, b \in \N^*$, with equality
        whenever $v_p(a) \neq v_p(b)$.
  \item Show that the number of multiples of $m$ in $\intint1n$ is
        $\lfloor n/m \rfloor$.
  \item Prove \emph{Legendre's formula}: for every $n \in \N^*$,
        \[
        v_p(n!) = \sum_{k=1}^{\infty}
        \Bigl\lfloor \frac{n}{p^k} \Bigr\rfloor
        \]
        (a finite sum: the terms vanish once $p^k > n$).
        \emph{Count, for each $k$, the factors of $\intint1n$
        divisible by $p^k$: each contributes exactly one unit per
        level it reaches.}
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The digital form and trailing
zeros.}
\begin{enumerate}[resume]
  \item Compute $v_5(1000!)$ and $v_2(1000!)$, and conclude: how
        many zeros end $1000!$?
  \item Prove the digital form of Legendre's formula: writing $n =
        \sum_i a_i p^i$ in base $p$,
        \[
        v_p(n!) = \frac{n - s_p(n)}{p - 1} .
        \]
  \item Two consequences for $p = 2$: show that $2^n$ never
        \omterm{def:b1:arith:divides}{divides} $n!$, and that $2^{n-1}$ \omterm{def:b1:arith:divides}{divides} $n!$ exactly when
        $n$ is a power of $2$.
  \item Bound the defect: show $\frac n{p-1} - \log_p(n) - 1 \leq
        v_p(n!) < \frac n{p-1}$, so that $\frac{v_p(n!)}{n} \to
        \frac1{p-1}$: in the long run, a proportion
        $\frac1{p-1}$ of one factor $p$ accrues per unit.
  \item Let $Z(n) = v_5(n!)$ be the number of trailing zeros of
        $n!$. Show $Z(n) - Z(n-1) = v_5(n)$, deduce that $Z$ skips
        the value $5$ entirely (compute $Z(24)$ and $Z(25)$), and
        prove that no factorial ends in exactly five zeros.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Kummer's theorem.}
\begin{enumerate}[resume]
  \item Prove that $\lfloor x + y \rfloor - \lfloor x \rfloor -
        \lfloor y \rfloor \in \{0, 1\}$ for all $x, y \in \R$, and
        deduce from Legendre's formula that
        \[
        v_p\binom{m+n}m
        = \sum_{k\geq1}\Bigl(
        \Bigl\lfloor\frac{m+n}{p^k}\Bigr\rfloor
        - \Bigl\lfloor\frac{m}{p^k}\Bigr\rfloor
        - \Bigl\lfloor\frac{n}{p^k}\Bigr\rfloor\Bigr),
        \]
        a sum of terms each equal to $0$ or $1$.
  \item Prove \emph{Kummer's theorem}: the $k$-th term of that sum
        equals $1$ exactly when the addition of $m$ and $n$ in
        base $p$ produces a carry into position $k$; hence
        $v_p\binom{m+n}m$ is the total number of carries.
        \emph{(Write $m = p^km_1 + m_0$ and $n = p^kn_1 + n_0$
        with $0 \leq m_0, n_0 < p^k$ and inspect $\lfloor (m_0 +
        n_0)/p^k \rfloor$.)}
  \item Deduce that for $0 < j < p^k$:
        \[
        v_p\binom{p^k}{j} = k - v_p(j) ,
        \]
        by counting the carries in the addition $j + (p^k - j)$.
        (In particular $p \mid \binom p j$ for $0 < j < p$: the
        key step of \cref{thm:b1:arith:fermat}, recovered.)
  \item Prove that $v_2\binom{2n}n = s_2(n)$. Deduce that the
        central \omterm{def:b1:counting:objects}{binomial coefficient} is always even, and that
        $\binom{2n}n \equiv 2 \pmod 4$ exactly when $n$ is a power
        of $2$.
  \item Show, using Vandermonde's identity
        (\cref{exo:b1:counting:7}) and question 13, that
        $\binom{2p}p \equiv 2 \pmod p$ for every \omterm{def:b1:arith:prime}{prime} $p$.
  \item Compute $v_3\binom{1000}{500}$ twice: once by Kummer
        (write $500$ in base $3$ and count the carries in $500 +
        500$), once by Legendre's digital form (compute $s_3(500)$
        and $s_3(1000)$); check that both give the same value.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The parity of Pascal's triangle, and
a prime-density bound.}
\begin{enumerate}[resume]
  \item Prove the digit criterion: $\binom nk$ is \emph{odd} if
        and only if every binary digit of $k$ is at most the
        corresponding digit of $n$. State and prove the analogous
        criterion for $p \nmid \binom nk$ in base $p$.
  \item Deduce that row $n$ of Pascal's triangle contains exactly
        $2^{s_2(n)}$ odd entries; verify on rows $4$ and $5$.
  \item Deduce that all interior entries $\binom nk$ ($0 < k < n$)
        are even if and only if $n$ is a power of $2$.
  \item Prove that every \omterm{def:b1:arith:prime}{prime} power dividing $\binom{m+n}m$ is at
        most $m + n$: if $p^a \mid \binom{m+n}m$ then $p^a \leq
        m + n$. \emph{(How many nonzero terms can the sum of
        question 11 have?)}
  \item Deduce that $\binom{2n}n$ \omterm{def:b1:arith:divides}{divides}
        $\operatorname{lcm}(1, 2, \dots, 2n)$, and combine with
        the lower bound $\binom{2n}n \geq \frac{4^n}{2n+1}$
        (which you will prove: the central entry is the largest
        of the $2n + 1$ entries of row $2n$) to obtain
        \[
        \operatorname{lcm}(1, \dots, 2n) \geq \frac{4^n}{2n+1} :
        \]
        the common multiples of the first integers grow
        \emph{exponentially} --- a first quantitative glimpse of
        the abundance of \omterm{def:b1:arith:prime}{primes}.
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Synthesis.}
\begin{enumerate}[resume]
  \item Find the smallest $n$ such that $n!$ ends in at least
        $2026$ zeros. \emph{(Estimate $Z(n) \approx n/4$, then
        adjust using the exact formula.)}
  \item One last cross-check: show that $7$ does \emph{not} divide
        $\binom{100}{50}$, first by writing $50$ in base $7$ and
        checking that the addition $50 + 50$ is carry-free, then
        by computing $v_7(100!)$ and $v_7(50!)$ with Legendre's
        formula.
  \item Where exactly did the problem use: (i) unique
        factorization; (ii) the Euclidean-division decomposition
        $n = p^k n_1 + n_0$; (iii) a counting argument from
        \cref{ch:b1:counting}? One sentence each.
  \item Synthesis, in a short paragraph: Legendre's formula turns
        a \omterm{def:b1:arith:divides}{divisibility} question into digit arithmetic, and
        Kummer's theorem reads the answer off the carries of one
        addition --- comment on this translation, on the checks of
        question 16, and on what the bound of question 21 suggests
        about \omterm{def:b1:arith:prime}{primes} (the full \omterm{def:b1:logic:statement}{statement}, the \omterm{def:b1:arith:prime}{prime number}
        theorem, is far beyond this volume; the polynomial
        analogue of this chapter's toolkit is
        \cref{ch:b1:poly}).
\end{enumerate}
\end{problem}
