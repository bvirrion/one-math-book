\chapter{Linear Differential Equations}\label{ch:b1:diffeq}

First met in the High School volume, differential equations are treated
here with full proofs and in greater generality: first-order linear
equations with variable coefficients (solved completely by the
variation-of-constants method) and second-order linear equations with
constant coefficients, the model for oscillations. Both cases display
the same structure: \emph{general solution $=$ one particular solution
$+$ general solution of the homogeneous equation}.

\section{First-order linear equations}

\begin{definition}\label{def:b1:diffeq:linear1}
Let $I$ be an interval and $a, b \colon I \to \R$ (or $\C$) be
continuous. The equation
\[
(E)\colon\quad y' + a(x)\,y = b(x),
\]
in the unknown differentiable function $y \colon I \to \R$ (or $\C$),
is a \emph{first-order linear differential
equation}\index{differential equation!first-order linear}. The
equation $(H)\colon y' + a(x) y = 0$ is its \emph{homogeneous}
equation.
\end{definition}

\begin{theorem}[Solving the homogeneous equation]\label{thm:b1:diffeq:homogeneous1}
Let $A$ be a primitive of $a$ on $I$ (it exists:
\cref{ch:b1:integration}). The solutions of $(H)$ on $I$ are exactly
the functions
\[
y(x) = \lambda\, \eu^{-A(x)}, \qquad \lambda \in \R \text{ (or } \C).
\]
\end{theorem}

\begin{proof}
These functions are solutions: $y' = -\lambda A' \eu^{-A} = -a y$.
Conversely, let $y$ solve $(H)$ and set $z(x) = y(x)\, \eu^{A(x)}$.
Then
\[
z' = y' \eu^{A} + y\, a\, \eu^{A} = (y' + ay)\, \eu^{A} = 0 ,
\]
so $z$ is constant on the interval $I$, say $z = \lambda$: $y =
\lambda \eu^{-A}$. (Note the logic: no solution is lost, because
\emph{every} solution has been written in the announced form.)
\end{proof}

\begin{theorem}[Variation of constants; Cauchy problem]\label{thm:b1:diffeq:voc}
With the notation above:
\begin{enumerate}
  \item The solutions of $(E)$ on $I$ are exactly
        \[
        y(x) = \Bigl(\lambda + \int_{x_0}^{x} b(t)\, \eu^{A(t)} \dd t
        \Bigr)\, \eu^{-A(x)},
        \qquad \lambda \in \R,
        \]
        where $x_0 \in I$ is fixed. Equivalently: general solution of
        $(H)$ plus one particular solution of $(E)$.
  \item For every $x_0 \in I$ and $y_0$, the \emph{Cauchy
        problem}\index{Cauchy problem} ``$(E)$ and $y(x_0) = y_0$''
        has exactly one solution on $I$.
\end{enumerate}
\end{theorem}

\begin{proof}
(1) Following the method named \emph{variation of
constants}\index{variation of constants}, look for solutions of the
form $y = \mu(x)\, \eu^{-A(x)}$ with $\mu$ differentiable --- no
generality is lost, since every function on $I$ can be so written
($\mu = y\,\eu^{A}$). Substituting,
\[
y' + ay = \mu' \eu^{-A} - \mu a \eu^{-A} + a\mu\eu^{-A}
= \mu'\, \eu^{-A},
\]
so $y$ solves $(E)$ if and only if $\mu'(x) = b(x)\,\eu^{A(x)}$, if
and only if $\mu(x) = \lambda + \int_{x_0}^x b(t)\eu^{A(t)}\dd t$ for
some constant $\lambda$ (two primitives of the same continuous
function on an interval differ by a constant).

(2) In the formula, $y(x_0) = \lambda\,\eu^{-A(x_0)}$: the condition
$y(x_0) = y_0$ determines $\lambda = y_0 \eu^{A(x_0)}$ uniquely.
\end{proof}

\begin{example}\label{ex:b1:diffeq:order1}
Solve $y' + \dfrac{y}{x} = x^2$ on $I = \intoo{0}{+\infty}$. Here
$a(x) = \frac 1x$, $A(x) = \ln x$, $\eu^{-A(x)} = \frac 1x$.
Homogeneous solutions: $\frac{\lambda}{x}$. Variation of constants:
$\mu'(x) = x^2 \cdot x = x^3$, so $\mu = \frac{x^4}{4} + \lambda$, and
\[
y(x) = \frac{x^3}{4} + \frac{\lambda}{x}, \qquad \lambda \in \R .
\]
With the initial condition $y(1) = 0$: $\lambda = -\frac14$.
\emph{Check:} $y' + \frac yx = \frac{3x^2}{4} - \frac{\lambda}{x^2} +
\frac{x^2}{4} + \frac{\lambda}{x^2} = x^2$.
\end{example}

\begin{remark}[Intervals matter]
The theorem lives on an \emph{interval} where $a$ and $b$ are
continuous. For $y' + \frac yx = 0$ on $\R^*$, the solutions are
$\frac{\lambda}{x}$ on $\intoo{0}{+\infty}$ and
$\frac{\mu}{x}$ on $\intoo{-\infty}{0}$ with \emph{independent}
constants: there is no reason for a single formula to glue across the
singularity at $0$.
\end{remark}

\section{Second-order linear equations with constant coefficients}

\begin{definition}\label{def:b1:diffeq:linear2}
Let $a, b \in \R$ and $f \colon I \to \R$ continuous. The equation
\[
(E)\colon\quad y'' + a\,y' + b\,y = f(x)
\]
is a \emph{second-order linear equation with constant
coefficients}\index{differential equation!second-order linear};
$(H)\colon y'' + ay' + by = 0$ is its homogeneous equation, and
$\chi(r) = r^2 + ar + b$ its \emph{characteristic
polynomial}\index{characteristic polynomial}.
\end{definition}

\begin{theorem}[Homogeneous solutions]\label{thm:b1:diffeq:homogeneous2}
Let $\Delta = a^2 - 4b$ be the discriminant of $\chi$. The real
solutions of $(H)$ on $\R$ are:
\begin{enumerate}
  \item if $\Delta > 0$, with $r_1 \neq r_2$ the two real roots:
        $\;y = \lambda\, \eu^{r_1 x} + \mu\, \eu^{r_2 x}$;
  \item if $\Delta = 0$, with $r_0$ the double root:
        $\;y = (\lambda + \mu x)\, \eu^{r_0 x}$;
  \item if $\Delta < 0$, with roots $\alpha \pm \iu\omega$
        ($\omega > 0$):
        $\;y = \eu^{\alpha x} (\lambda \cos\omega x + \mu
        \sin\omega x)$;
\end{enumerate}
in each case with $(\lambda, \mu)$ running over $\R^2$.
\end{theorem}

\begin{proof}
First observe that for $r \in \C$, $x \mapsto \eu^{rx}$ solves $(H)$
if and only if $\chi(r) = 0$ (substitute: $(r^2 + ar + b)\eu^{rx} =
0$).

The key step is a change of unknown that reduces the order. Let $r$ be
a (possibly complex) root of $\chi$ and write $y = z\, \eu^{rx}$,
which loses no generality. Then
\[
y'' + ay' + by
= \bigl(z'' + (2r + a) z' + \chi(r) z\bigr)\eu^{rx}
= \bigl(z'' + (2r + a) z'\bigr)\eu^{rx},
\]
so $(H)$ becomes the \emph{first-order} equation $u' + (2r + a) u = 0$
for $u = z'$.

\emph{Case $\Delta \neq 0$:} choose $r = r_1$; then $2r_1 + a = r_1 -
r_2$ (since $r_1 + r_2 = -a$). By
\cref{thm:b1:diffeq:homogeneous1}, $z' = c\,\eu^{(r_2 - r_1)x}$ for
some constant $c$; integrating on $\R$, $z = \mu\, \eu^{(r_2 - r_1)x}
+ \lambda$ with $\mu = \frac{c}{r_2 - r_1}$, and therefore $y = z\,
\eu^{r_1 x} = \lambda\,\eu^{r_1x} + \mu\,\eu^{r_2x}$.
When $\Delta < 0$, the roots are $\alpha \pm \iu\omega$ and the
complex solutions $\eu^{(\alpha\pm\iu\omega)x}$ combine into the real
ones: taking real and imaginary parts of $\eu^{(\alpha +
\iu\omega)x} = \eu^{\alpha x}(\cos\omega x + \iu \sin\omega x)$, the
real solution space is as announced (a real solution is a complex
solution equal to its conjugate, which forces conjugate coefficients).

\emph{Case $\Delta = 0$:} $r = r_0$, $2r_0 + a = 0$, so $z'' = 0$:
$z = \lambda + \mu x$ and $y = (\lambda + \mu x)\eu^{r_0 x}$.
\end{proof}

\begin{theorem}[Structure and Cauchy problem]\label{thm:b1:diffeq:structure2}
\begin{enumerate}
  \item If $y_p$ is one particular solution of $(E)$, the solutions of
        $(E)$ are exactly $y_p + y_h$, $y_h$ running over the
        solutions of $(H)$.
  \item (Superposition) If $y_1$ solves $y'' + ay' + by = f_1$ and
        $y_2$ solves $y'' + ay' + by = f_2$, then $y_1 + y_2$ solves
        the equation with right-hand side $f_1 + f_2$.
  \item For all $x_0 \in I$ and $(y_0, y_0')$, the Cauchy problem
        ``$(E)$, $y(x_0) = y_0$, $y'(x_0) = y_0'$'' has exactly one
        solution on $I$. \emph{(Existence granted a particular
        solution; uniqueness in full.)}
\end{enumerate}
\end{theorem}

\begin{proof}
(1) $y$ solves $(E)$ iff $y - y_p$ solves $(H)$, by linearity of $y
\mapsto y'' + ay' + by$. (2) is the same linearity.

(3) By (1) it suffices to prove that the homogeneous Cauchy problem
$y(x_0) = y'(x_0) = 0$ has only the zero solution, and that the
constants $(\lambda, \mu)$ can always be adjusted to the data. In each
case of \cref{thm:b1:diffeq:homogeneous2}, the map $(\lambda, \mu)
\mapsto (y(x_0), y'(x_0))$ is linear with determinant (computed at
$x_0 = 0$ after translating, which changes nothing):
$r_2 - r_1 \neq 0$ in case (1) [matrix $\bigl(\begin{smallmatrix}1 &
1\\ r_1 & r_2\end{smallmatrix}\bigr)$], $1$ in case (2), $\omega \neq
0$ in case (3). A linear bijection: each data pair $(y_0, y_0')$ is
reached by exactly one $(\lambda, \mu)$.
\end{proof}

\begin{method}[Particular solution for $f(x) = P(x)\,\eu^{\gamma x}$]\label{met:b1:diffeq:particular}
When the right-hand side is $P(x)\,\eu^{\gamma x}$ with $P$ a
polynomial and $\gamma \in \R$ (this covers polynomials, exponentials,
and via complex $\gamma$ or superposition, $\cos$ and $\sin$): look
for a particular solution of the form
\[
y_p(x) = x^{m}\, Q(x)\, \eu^{\gamma x},
\qquad
m = \text{multiplicity of } \gamma \text{ as a root of } \chi
\ (m = 0, 1 \text{ or } 2),
\]
with $Q$ a polynomial of the same degree as $P$, whose coefficients
are found by substitution and identification. For $f = K\cos\omega x$
(or $\sin$), solve with right-hand side $K\eu^{\iu\omega x}$ and take
the real (resp.\ imaginary) part.
\end{method}

\begin{example}[A forced oscillation]\label{ex:b1:diffeq:oscillation}
Solve $y'' + y = \cos x$, $y(0) = 0$, $y'(0) = 0$.

\emph{Homogeneous:} $\chi(r) = r^2 + 1$, roots $\pm\iu$: $y_h =
\lambda\cos x + \mu \sin x$.

\emph{Particular:} right-hand side $\Re(\eu^{\iu x})$ with $\gamma =
\iu$ a simple root of $\chi$: try $z_p = c\, x\, \eu^{\iu x}$
($c \in \C$). Then $z_p'' + z_p = c\,(2\iu)\eu^{\iu x}$, which equals
$\eu^{\iu x}$ for $c = \frac{1}{2\iu} = -\frac\iu2$. So $z_p =
-\frac{\iu}{2} x (\cos x + \iu \sin x)$ and $y_p = \Re(z_p) =
\frac{x \sin x}{2}$.

\emph{General solution:} $y = \frac{x\sin x}{2} + \lambda\cos x +
\mu\sin x$. Conditions: $y(0) = \lambda = 0$; $y' = \frac{\sin x +
x\cos x}{2} + \mu\cos x$, so $y'(0) = \mu = 0$. Answer: $y =
\frac{x\sin x}{2}$ --- an oscillation whose amplitude grows linearly:
the \emph{resonance}\index{resonance} phenomenon, caused by forcing
the system at its natural frequency.
\end{example}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:b1:diffeq:1}
Solve on $\R$: $\;y' + 2y = \eu^{3x}$; then the Cauchy problem $y(0) =
1$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:diffeq:2}
Solve on $\intoo{0}{+\infty}$: $\;x y' - 2y = x^3$ \emph{(put the
equation in the normalized form first)}.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:diffeq:3}
Solve on $\R$, giving the real general solution:
$\;y'' - 3y' + 2y = 0$; $\;y'' + 4y' + 4y = 0$; $\;y'' - 2y' + 5y =
0$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:diffeq:4}
Solve $y'' - y = x^2$ on $\R$, then the Cauchy problem $y(0) = 0$,
$y'(0) = 1$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:diffeq:5}
Solve on $\intoo{-\frac\pi2}{\frac\pi2}$:
$\;y' + y\tan x = \sin 2x$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:diffeq:6}
Solve $y'' - 4y' + 3y = (2x + 1)\,\eu^{x}$ on $\R$. \emph{(Mind the
multiplicity: is $1$ a root of the characteristic polynomial?)}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:diffeq:7}
Solve $y'' + 4y = \sin 2x + x$ on $\R$ \emph{(superposition; treat
each right-hand side separately)}.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:diffeq:8}
A cup of coffee at temperature $T_0 = 80\,^\circ$C sits in a room at
$20\,^\circ$C. Newton's law of cooling states $T' = -k\,(T - 20)$ with
$k > 0$. Solve for $T(t)$, and given that the coffee is at
$50\,^\circ$C after $10$ minutes, find when it reaches
$25\,^\circ$C.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:diffeq:9}
(Damped oscillator) For $\varepsilon \geq 0$, consider
$y'' + 2\varepsilon y' + y = 0$.
\begin{enumerate}
  \item Solve for $\varepsilon \in \intco{0}{1}$, $\varepsilon = 1$,
        and $\varepsilon > 1$.
  \item Show that for $\varepsilon > 0$ every solution tends to $0$
        at $+\infty$, and that for $\varepsilon = 0$ the nonzero
        solutions do not.
  \item For $\varepsilon \in \intoo{0}{1}$, show that the zeros of a
        nonzero solution are regularly spaced, with gap
        $\frac{\pi}{\sqrt{1 - \varepsilon^2}}$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:diffeq:10}
Find all functions $f \colon \R \to \R$, twice differentiable, such
that
\[
\forall x, y \in \R, \qquad f(x + y) + f(x - y) = 2 f(x) f(y),
\]
with $f(0) \ne 0$ and $f$ not constant.
\emph{Hint: fix $y$, differentiate twice with respect to $x$ at $0$;
show $f(0) = 1$ and $f'' = c f$ for some constant $c$; then solve
according to the sign of $c$ and check which solutions satisfy the
functional equation.}
\end{exercise}
