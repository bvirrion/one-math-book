\chapter{Linear Differential Equations}\label{ch:b1:diffeq}

First met in the High School volume, differential equations are treated
here with full proofs and in greater generality: first-order linear
equations with variable coefficients (solved completely by the
variation-of-constants method) and second-order linear equations with
constant coefficients, the model for oscillations. Both cases display
the same structure: \emph{general solution $=$ one particular solution
$+$ general solution of the homogeneous equation}.

\section{First-order linear equations}

\begin{definition}\label{def:b1:diffeq:linear1}
Let $I$ be an interval and $a, b \colon I \to \R$ (or $\C$) be
continuous. The equation
\[
(E)\colon\quad y' + a(x)\,y = b(x),
\]
in the unknown differentiable function $y \colon I \to \R$ (or $\C$),
is a \emph{first-order linear differential
equation}\index{differential equation!first-order linear}. The
equation $(H)\colon y' + a(x) y = 0$ is its \emph{homogeneous}
equation.
\end{definition}

\begin{theorem}[Solving the homogeneous equation]\label{thm:b1:diffeq:homogeneous1}
Let $A$ be a primitive of $a$ on $I$ (it exists:
\cref{ch:b1:integration}). The solutions of $(H)$ on $I$ are exactly
the functions
\[
y(x) = \lambda\, \eu^{-A(x)}, \qquad \lambda \in \R \text{ (or } \C).
\]
\end{theorem}

\begin{proof}
These functions are solutions: $y' = -\lambda A' \eu^{-A} = -a y$.
Conversely, let $y$ solve $(H)$ and set $z(x) = y(x)\, \eu^{A(x)}$.
Then
\[
z' = y' \eu^{A} + y\, a\, \eu^{A} = (y' + ay)\, \eu^{A} = 0 ,
\]
so $z$ is constant on the interval $I$, say $z = \lambda$: $y =
\lambda \eu^{-A}$. (Note the logic: no solution is lost, because
\emph{every} solution has been written in the announced form.)
\end{proof}

\begin{example}[A homogeneous equation with variable coefficient]\label{ex:b1:diffeq:varcoeff}
Solve $y' + (\cos x)\,y = 0$ on $\R$. A primitive of $a(x) =
\cos x$ is $A(x) = \sin x$, so the solutions are
\[
y(x) = \lambda\,\eu^{-\sin x}, \qquad \lambda \in \R .
\]
Two readings. Every solution is periodic (period $2\pi$) and never
vanishes unless $\lambda = 0$ --- the sign of $\lambda$ is the
sign of $y$ forever, since an exponential cannot cross zero. And
the solution through $y(0) = y_0$ is $y_0\eu^{-\sin x}$: exactly
one curve of the family through each initial point, the
one-dimensional picture of \cref{thm:b1:diffeq:voc}~(2).
\end{example}

\begin{theorem}[Variation of constants; Cauchy problem]\label{thm:b1:diffeq:voc}
With the notation above:
\begin{enumerate}
  \item The solutions of $(E)$ on $I$ are exactly
        \[
        y(x) = \Bigl(\lambda + \int_{x_0}^{x} b(t)\, \eu^{A(t)} \dd t
        \Bigr)\, \eu^{-A(x)},
        \qquad \lambda \in \R,
        \]
        where $x_0 \in I$ is fixed. Equivalently: general solution of
        $(H)$ plus one particular solution of $(E)$.
  \item For every $x_0 \in I$ and $y_0$, the \emph{Cauchy
        problem}\index{Cauchy problem} ``$(E)$ and $y(x_0) = y_0$''
        has exactly one solution on $I$.
\end{enumerate}
\end{theorem}

\begin{proof}
(1) Following the method named \emph{variation of
constants}\index{variation of constants}, look for solutions of the
form $y = \mu(x)\, \eu^{-A(x)}$ with $\mu$ differentiable --- no
generality is lost, since every function on $I$ can be so written
($\mu = y\,\eu^{A}$). Substituting,
\[
y' + ay = \mu' \eu^{-A} - \mu a \eu^{-A} + a\mu\eu^{-A}
= \mu'\, \eu^{-A},
\]
so $y$ solves $(E)$ if and only if $\mu'(x) = b(x)\,\eu^{A(x)}$, if
and only if $\mu(x) = \lambda + \int_{x_0}^x b(t)\eu^{A(t)}\dd t$ for
some constant $\lambda$ (two primitives of the same continuous
function on an interval differ by a constant).

(2) In the formula, $y(x_0) = \lambda\,\eu^{-A(x_0)}$: the condition
$y(x_0) = y_0$ determines $\lambda = y_0 \eu^{A(x_0)}$ uniquely.
\end{proof}

\begin{example}\label{ex:b1:diffeq:order1}
Solve $y' + \dfrac{y}{x} = x^2$ on $I = \intoo{0}{+\infty}$. Here
$a(x) = \frac 1x$, $A(x) = \ln x$, $\eu^{-A(x)} = \frac 1x$.
Homogeneous solutions: $\frac{\lambda}{x}$. Variation of constants:
$\mu'(x) = x^2 \cdot x = x^3$, so $\mu = \frac{x^4}{4} + \lambda$, and
\[
y(x) = \frac{x^3}{4} + \frac{\lambda}{x}, \qquad \lambda \in \R .
\]
With the initial condition $y(1) = 0$: $\lambda = -\frac14$.
\emph{Check:} $y' + \frac yx = \frac{3x^2}{4} - \frac{\lambda}{x^2} +
\frac{x^2}{4} + \frac{\lambda}{x^2} = x^2$.
\end{example}

\begin{example}[Guessing beats integrating]\label{ex:b1:diffeq:guess}
Solve $y' + 2x\,y = x$ on $\R$. Variation of constants works ($A =
x^2$, $\mu' = x\,\eu^{x^2}$, $\mu = \frac12\eu^{x^2} + \lambda$),
but observing that the \emph{constant} $y_p = \frac12$ solves the
equation ($0 + 2x\cdot\frac12 = x$) is faster. With the
homogeneous solutions $\lambda\,\eu^{-x^2}$:
\[
y(x) = \frac12 + \lambda\,\eu^{-x^2}, \qquad \lambda \in \R .
\]
Every solution converges to $\frac12$ extremely fast as $x \to
\pm\infty$: the constant particular solution is an
\emph{equilibrium} that all solutions join. The insight: before
launching the general method, spend ten seconds looking for an
obvious particular solution (constant, monomial, multiple of the
right-hand side); the structure theorem then finishes the job.
\end{example}

\begin{remark}[Intervals matter]
The theorem lives on an \emph{interval} where $a$ and $b$ are
continuous. For $y' + \frac yx = 0$ on $\R^*$, the solutions are
$\frac{\lambda}{x}$ on $\intoo{0}{+\infty}$ and
$\frac{\mu}{x}$ on $\intoo{-\infty}{0}$ with \emph{independent}
constants: there is no reason for a single formula to glue across the
singularity at $0$.
\end{remark}

\begin{example}[A complex right-hand side, two real answers]\label{ex:b1:diffeq:complexrhs}
Solve $y' - y = \cos x$ and $y' - y = \sin x$ in one stroke. Work
in $\C$ with the right-hand side $\eu^{\iu x}$: trying $y_p =
c\,\eu^{\iu x}$ gives $c(\iu - 1)\eu^{\iu x} = \eu^{\iu x}$, so
\[
c = \frac1{\iu - 1} = \frac{-1 - \iu}2,
\qquad
y_p = -\frac{(1 + \iu)(\cos x + \iu\sin x)}2
= \frac{\sin x - \cos x}2 + \iu\,\frac{-\sin x - \cos x}2 .
\]
Since the equation has real coefficients, real and imaginary
parts split: $\frac{\sin x - \cos x}2$ solves $y' - y = \cos x$,
and $-\frac{\sin x + \cos x}2$ solves $y' - y = \sin x$ (check
the first: derivative $\frac{\cos x + \sin x}2$, minus the
function, gives $\cos x$). One complex line replaced two runs of
variation of constants --- the same economy that
\cref{met:b1:diffeq:particular} systematizes for second order,
and a recurring dividend of \cref{ch:b1:complex}.
\end{example}

\section{Second-order linear equations with constant coefficients}

\begin{definition}\label{def:b1:diffeq:linear2}
Let $a, b \in \R$ and $f \colon I \to \R$ continuous. The equation
\[
(E)\colon\quad y'' + a\,y' + b\,y = f(x)
\]
is a \emph{second-order linear equation with constant
coefficients}\index{differential equation!second-order linear};
$(H)\colon y'' + ay' + by = 0$ is its homogeneous equation, and
$\chi(r) = r^2 + ar + b$ its \emph{characteristic
polynomial}\index{characteristic polynomial}.
\end{definition}

\begin{theorem}[Homogeneous solutions]\label{thm:b1:diffeq:homogeneous2}
Let $\Delta = a^2 - 4b$ be the discriminant of $\chi$. The real
solutions of $(H)$ on $\R$ are:
\begin{enumerate}
  \item if $\Delta > 0$, with $r_1 \neq r_2$ the two real roots:
        $\;y = \lambda\, \eu^{r_1 x} + \mu\, \eu^{r_2 x}$;
  \item if $\Delta = 0$, with $r_0$ the double root:
        $\;y = (\lambda + \mu x)\, \eu^{r_0 x}$;
  \item if $\Delta < 0$, with roots $\alpha \pm \iu\omega$
        ($\omega > 0$):
        $\;y = \eu^{\alpha x} (\lambda \cos\omega x + \mu
        \sin\omega x)$;
\end{enumerate}
in each case with $(\lambda, \mu)$ running over $\R^2$.
\end{theorem}

\begin{proof}
First observe that for $r \in \C$, $x \mapsto \eu^{rx}$ solves $(H)$
if and only if $\chi(r) = 0$ (substitute: $(r^2 + ar + b)\eu^{rx} =
0$). This is why exponentials are the natural first guess:
differentiation acts on $\eu^{rx}$ as multiplication by the number
$r$, so the differential equation becomes the numerical equation
$\chi(r) = 0$ --- the whole analytic problem is compressed into
finding the roots of one quadratic.

The key step is a change of unknown that reduces the order. Let $r$ be
a (possibly complex) root of $\chi$ and write $y = z\, \eu^{rx}$,
which loses no generality. Then
\[
y'' + ay' + by
= \bigl(z'' + (2r + a) z' + \chi(r) z\bigr)\eu^{rx}
= \bigl(z'' + (2r + a) z'\bigr)\eu^{rx},
\]
so $(H)$ becomes the \emph{first-order} equation $u' + (2r + a) u = 0$
for $u = z'$.

\emph{Case $\Delta \neq 0$:} choose $r = r_1$; then $2r_1 + a = r_1 -
r_2$ (since $r_1 + r_2 = -a$). By
\cref{thm:b1:diffeq:homogeneous1}, $z' = c\,\eu^{(r_2 - r_1)x}$ for
some constant $c$; integrating on $\R$, $z = \mu\, \eu^{(r_2 - r_1)x}
+ \lambda$ with $\mu = \frac{c}{r_2 - r_1}$, and therefore $y = z\,
\eu^{r_1 x} = \lambda\,\eu^{r_1x} + \mu\,\eu^{r_2x}$.
When $\Delta < 0$, the roots are $\alpha \pm \iu\omega$ and the
complex solutions are $y = c_1\eu^{(\alpha+\iu\omega)x} +
c_2\eu^{(\alpha-\iu\omega)x}$ with $c_1, c_2 \in \C$. Which of
these are real-valued? Since $\conj{\eu^{(\alpha+\iu\omega)x}} =
\eu^{(\alpha-\iu\omega)x}$, the conjugate of $y$ is $\conj{c_2}\,
\eu^{(\alpha+\iu\omega)x} + \conj{c_1}\,\eu^{(\alpha-\iu\omega)x}$,
and $y = \conj y$ for all $x$ forces $c_2 = \conj{c_1}$ (the two
exponentials are linearly independent: evaluate at two points, or
compare at $x=0$ after dividing by $\eu^{\alpha x}$). Writing $c_1
= \frac{\lambda - \iu\mu}2$ with $\lambda, \mu$ real:
\[
y = 2\,\Re\Bigl(c_1\,\eu^{\alpha x}(\cos\omega x +
\iu\sin\omega x)\Bigr)
= \eu^{\alpha x}\bigl(\lambda\cos\omega x + \mu\sin\omega
x\bigr),
\]
and conversely every such function is a solution (real part of a
complex solution of a real equation): the real solution space is
as announced.

\emph{Case $\Delta = 0$:} $r = r_0$, $2r_0 + a = 0$, so $z'' = 0$:
$z = \lambda + \mu x$ and $y = (\lambda + \mu x)\eu^{r_0 x}$.
\end{proof}

\begin{example}[A Cauchy problem, start to finish]\label{ex:b1:diffeq:cauchy2}
Solve $y'' - 3y' + 2y = 0$ with $y(0) = 0$, $y'(0) = 1$. The
characteristic polynomial $r^2 - 3r + 2 = (r - 1)(r - 2)$ has the
real roots $1$ and $2$: general solution $y = \lambda\eu^{x} +
\mu\eu^{2x}$. The two conditions give the linear system
\[
\lambda + \mu = 0, \qquad \lambda + 2\mu = 1 ,
\]
so $\mu = 1$, $\lambda = -1$:
\[
y(x) = \eu^{2x} - \eu^{x} .
\]
Check: $y(0) = 0$; $y' = 2\eu^{2x} - \eu^x$ has $y'(0) = 1$; and
$y'' - 3y' + 2y = (4 - 6 + 2)\eu^{2x} + (-1 + 3 - 2)\eu^x = 0$.
Note the shape of the answer: near $-\infty$ the slow mode
$-\eu^x$ dominates; near $+\infty$ the fast mode $\eu^{2x}$ does.
Reading solutions as superpositions of modes with different decay
or growth rates is the profitable habit --- it is how the
transient/steady-state split of the weekend problem is organized.
\end{example}

\begin{theorem}[Structure and Cauchy problem]\label{thm:b1:diffeq:structure2}
\begin{enumerate}
  \item If $y_p$ is one particular solution of $(E)$, the solutions of
        $(E)$ are exactly $y_p + y_h$, $y_h$ running over the
        solutions of $(H)$.
  \item (Superposition) If $y_1$ solves $y'' + ay' + by = f_1$ and
        $y_2$ solves $y'' + ay' + by = f_2$, then $y_1 + y_2$ solves
        the equation with right-hand side $f_1 + f_2$.
  \item For all $x_0 \in I$ and $(y_0, y_0')$, the Cauchy problem
        ``$(E)$, $y(x_0) = y_0$, $y'(x_0) = y_0'$'' has exactly one
        solution on $I$. \emph{(Existence granted a particular
        solution; uniqueness in full.)}
\end{enumerate}
\end{theorem}

\begin{proof}
(1) $y$ solves $(E)$ iff $y - y_p$ solves $(H)$, by linearity of $y
\mapsto y'' + ay' + by$. (2) is the same linearity.

(3) By (1) it suffices to prove that the constants $(\lambda,
\mu)$ can always be adjusted, uniquely, to any data $(y_0, y_0')$.
Translating the variable, assume $x_0 = 0$. In case (1) of
\cref{thm:b1:diffeq:homogeneous2}, $y = \lambda\eu^{r_1x} +
\mu\eu^{r_2x}$ gives
\[
y(0) = \lambda + \mu, \qquad y'(0) = r_1\lambda + r_2\mu :
\]
a linear system in $(\lambda, \mu)$ whose determinant is $r_2 -
r_1 \neq 0$; solving it explicitly, $\mu = \frac{y_0' -
r_1y_0}{r_2 - r_1}$ and $\lambda = y_0 - \mu$: exactly one
solution. In case (2), $y(0) = \lambda$ and $y'(0) = r_0\lambda +
\mu$: the system is triangular with determinant $1$, solved by
$\lambda = y_0$, $\mu = y_0' - r_0y_0$. In case (3), $y(0) =
\lambda$ and $y'(0) = \alpha\lambda + \omega\mu$: determinant
$\omega \neq 0$, solved by $\lambda = y_0$, $\mu = \frac{y_0' -
\alpha y_0}\omega$. In each case the map $(\lambda, \mu) \mapsto
(y(x_0), y'(x_0))$ is a linear bijection --- the language of
\cref{ch:b1:linmaps} will compress this case check into one
sentence.
\end{proof}

\begin{method}[Particular solution for $f(x) = P(x)\,\eu^{\gamma x}$]\label{met:b1:diffeq:particular}
When the right-hand side is $P(x)\,\eu^{\gamma x}$ with $P$ a
polynomial and $\gamma \in \R$ (this covers polynomials, exponentials,
and via complex $\gamma$ or superposition, $\cos$ and $\sin$): look
for a particular solution of the form
\[
y_p(x) = x^{m}\, Q(x)\, \eu^{\gamma x},
\qquad
m = \text{multiplicity of } \gamma \text{ as a root of } \chi
\ (m = 0, 1 \text{ or } 2),
\]
with $Q$ a polynomial of the same degree as $P$, whose coefficients
are found by substitution and identification. For $f = K\cos\omega x$
(or $\sin$), solve with right-hand side $K\eu^{\iu\omega x}$ and take
the real (resp.\ imaginary) part.
\end{method}

\begin{example}[Superposition in action]\label{ex:b1:diffeq:superposition}
Solve $y'' - y = \eu^{x} + 4$ on $\R$. Homogeneous: $\chi(r) = r^2
- 1$, roots $\pm1$, so $y_h = \lambda\eu^x + \mu\eu^{-x}$. Split
the right-hand side and treat each piece by the method box.
\emph{Piece $\eu^x$:} here $\gamma = 1$ is a simple root of
$\chi$, so try $y_1 = c\,x\,\eu^x$: then $y_1'' - y_1 = c(x +
2)\eu^x - cx\eu^x = 2c\,\eu^x$, giving $c = \frac12$.
\emph{Piece $4$:} $\gamma = 0$ is not a root; the constant $y_2 =
-4$ works. By superposition
(\cref{thm:b1:diffeq:structure2}~(2)):
\[
y = \frac{x\,\eu^x}2 - 4 + \lambda\,\eu^x + \mu\,\eu^{-x},
\qquad (\lambda, \mu) \in \R^2 .
\]
Note how the two pieces demanded \emph{different} shapes ($m = 1$
versus $m = 0$): the multiplicity test is applied to each
exponent separately, which is the whole point of splitting the
right-hand side before guessing.
\end{example}

\begin{example}[The multiplicity rule at work]\label{ex:b1:diffeq:multiplicity}
Solve $y'' + y' = x$ on $\R$. The right-hand side is $P(x)\eu^{0
\cdot x}$ with $P(x) = x$, and $\gamma = 0$ is a \emph{simple} root
of $\chi(r) = r^2 + r = r(r + 1)$: so $m = 1$, and the correct
guess is $y_p = x\,(\alpha x + \beta) = \alpha x^2 + \beta x$, one
degree higher than $P$. Substituting:
\[
y_p'' + y_p' = 2\alpha + (2\alpha x + \beta)
= 2\alpha x + (2\alpha + \beta) ,
\]
and identification with $x$ gives $\alpha = \frac12$, $\beta = -1$:
$y_p = \frac{x^2}2 - x$. General solution: $y = \frac{x^2}2 - x +
\lambda + \mu\,\eu^{-x}$. Had we guessed $y_p = \alpha x + \beta$
(ignoring the multiplicity), substitution would give $y_p'' + y_p'
= \alpha$, a constant --- no choice of $\alpha, \beta$ can match
$x$, and the failure is structural: constants already solve the
homogeneous equation, so they are invisible to the left-hand side.
The factor $x^m$ exists precisely to climb out of the homogeneous
solution space.
\end{example}

\begin{remark}[The thirty-second insurance policy]\label{rem:b1:diffeq:check}
Every solved equation in this chapter ends with a substitution
check, and this is not decorative. A differential-equation
computation chains many small steps (a primitive, a product rule,
two constants), and a single sign error propagates invisibly;
substituting the final formula back into the equation catches
essentially all of them at the cost of one differentiation.
Cultivate the reflex in three layers: check the \emph{particular
solution} alone (the homogeneous part cancels anyway), check the
\emph{initial conditions} on the full solution, and when a
parameter is present, check a \emph{degenerate value} (does the
formula for general $\Omega$ reproduce the known answer at
$\Omega = 0$?). The habit costs half a minute; it converts
``probably right'' into ``verified''.
\end{remark}

\begin{remark}[Common pitfalls]\label{rem:b1:diffeq:pitfalls}
\begin{enumerate}
  \item \emph{Normalize first.} The formulas assume the equation
        reads $y' + a(x)y = b(x)$ --- coefficient $1$ on $y'$.
        For $xy' - 2y = x^3$, divide by $x$ (on an interval
        avoiding $0$) before identifying $a$ and $b$, as in
        \cref{exo:b1:diffeq:2}.
  \item \emph{One constant per dimension, fixed at the end.} The
        general first-order solution carries one constant, the
        second-order one two; initial conditions are imposed on
        the \emph{complete} solution $y_p + y_h$, never on $y_h$
        alone --- imposing them before adding $y_p$ is the most
        frequent structural error.
  \item \emph{Mind the multiplicity.} A particular-solution guess
        that solves the homogeneous equation is invisible to the
        left-hand side; the factor $x^m$ of
        \cref{met:b1:diffeq:particular} is not optional
        (\cref{ex:b1:diffeq:multiplicity}).
  \item \emph{Intervals are part of the answer.} Solutions live
        on intervals where the coefficients are continuous;
        gluing across a singularity can create spurious constants
        (\cref{exo:b1:diffeq:12}) or destroy uniqueness. ``Solve
        on $\R^*$'' means two independent problems.
\end{enumerate}
\end{remark}

\begin{example}[Off-resonance forcing]\label{ex:b1:diffeq:offresonance}
Solve $y'' + 4y = \sin x$ on $\R$. Natural frequency $2$, forcing
frequency $1$: since $\iu$ is \emph{not} a root of $\chi(r) = r^2
+ 4$, the multiplicity is $m = 0$ and a plain sinusoid suffices.
Trying $y_p = \alpha\sin x$ (no cosine needed: the equation has
no $y'$ term, and $\sin$ regenerates $\sin$):
\[
y_p'' + 4y_p = -\alpha\sin x + 4\alpha\sin x = 3\alpha\sin x ,
\]
so $\alpha = \frac13$ and the general solution is
\[
y = \frac{\sin x}3 + \lambda\cos 2x + \mu\sin 2x .
\]
Every solution stays bounded: a superposition of two oscillations
at the frequencies $1$ (forced) and $2$ (natural). Compare with
the next example, where forcing \emph{at} the natural frequency
changes the answer's very shape.
\end{example}

\begin{example}[A forced oscillation]\label{ex:b1:diffeq:oscillation}
Solve $y'' + y = \cos x$, $y(0) = 0$, $y'(0) = 0$.

\emph{Homogeneous:} $\chi(r) = r^2 + 1$, roots $\pm\iu$: $y_h =
\lambda\cos x + \mu \sin x$.

\emph{Particular:} right-hand side $\Re(\eu^{\iu x})$ with $\gamma =
\iu$ a simple root of $\chi$: try $z_p = c\, x\, \eu^{\iu x}$
($c \in \C$). Then $z_p'' + z_p = c\,(2\iu)\eu^{\iu x}$, which equals
$\eu^{\iu x}$ for $c = \frac{1}{2\iu} = -\frac\iu2$. So $z_p =
-\frac{\iu}{2} x (\cos x + \iu \sin x)$ and $y_p = \Re(z_p) =
\frac{x \sin x}{2}$.

\emph{General solution:} $y = \frac{x\sin x}{2} + \lambda\cos x +
\mu\sin x$. Conditions: $y(0) = \lambda = 0$; $y' = \frac{\sin x +
x\cos x}{2} + \mu\cos x$, so $y'(0) = \mu = 0$. Answer: $y =
\frac{x\sin x}{2}$ --- an oscillation whose amplitude grows linearly:
the \emph{resonance}\index{resonance} phenomenon, caused by forcing
the system at its natural frequency.
\end{example}

\begin{remark}[Interlude: linearity is a geometry]\label{rem:b1:diffeq:interlude}
Look back at the shape of every solution set in this chapter: a
special solution plus a space of homogeneous solutions with one
free constant (first order) or two (second order). The chapters on
linear algebra (\cref{ch:b1:vspaces,ch:b1:findim,ch:b1:linmaps})
will supply the exact
vocabulary: the map $L(y) = y'' + ay' + by$ is \emph{linear}, its
homogeneous solutions form the \emph{kernel} of $L$, a vector
space whose \emph{dimension} equals the order of the equation ---
that is the honest content of ``one constant per order'' --- and
the solution set of $L(y) = f$ is an \emph{affine subspace}, a
translate of the kernel. Even the Cauchy map $(\lambda, \mu)
\mapsto (y(x_0), y'(x_0))$ of \cref{thm:b1:diffeq:structure2} is a
linear bijection between two planes, i.e.\ an invertible $2
\times 2$ system (\cref{ch:b1:matrices}). Nothing in this chapter
will need to be redone --- it will only need to be renamed, and
the renaming is the best possible warm-up for linear algebra:
every abstract definition there has already earned its living
here.
\end{remark}

\begin{remark}[Where this chapter is used]\label{rem:b1:diffeq:whereused}
The structure theorem --- solutions of $(E)$ form ``a particular
solution plus the solutions of $(H)$'' --- is the first appearance
of a pattern that \cref{ch:b1:vspaces,ch:b1:linmaps} will name: the
solution set of $(H)$ is the \emph{kernel} of the linear map $y
\mapsto y'' + ay' + by$, and the solution set of $(E)$ is an
affine translate of it. The characteristic polynomial reappears as
the characteristic polynomial of a matrix in \cref{ch:b1:matrices}:
a second-order equation is a $2 \times 2$ first-order system in
disguise, a viewpoint the Year 2 volume systematizes. The
integrals demanded by variation of constants are supplied by
\cref{ch:b1:integration}, and the weekend problem below --- the
driven damped oscillator --- is the model case for every
oscillation question in the sciences, from circuits to suspension
bridges.
\end{remark}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:b1:diffeq:1}
Solve on $\R$: $\;y' + 2y = \eu^{3x}$; then the Cauchy problem $y(0) =
1$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:diffeq:2}
Solve on $\intoo{0}{+\infty}$: $\;x y' - 2y = x^3$ \emph{(put the
equation in the normalized form first)}.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:diffeq:3}
Solve on $\R$, giving the real general solution:
$\;y'' - 3y' + 2y = 0$; $\;y'' + 4y' + 4y = 0$; $\;y'' - 2y' + 5y =
0$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:diffeq:4}
Solve $y'' - y = x^2$ on $\R$, then the Cauchy problem $y(0) = 0$,
$y'(0) = 1$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:diffeq:5}
Solve on $\intoo{-\frac\pi2}{\frac\pi2}$:
$\;y' + y\tan x = \sin 2x$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:diffeq:6}
Solve $y'' - 4y' + 3y = (2x + 1)\,\eu^{x}$ on $\R$. \emph{(Mind the
multiplicity: is $1$ a root of the characteristic polynomial?)}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:diffeq:7}
Solve $y'' + 4y = \sin 2x + x$ on $\R$ \emph{(superposition; treat
each right-hand side separately)}.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:diffeq:8}
A cup of coffee at temperature $T_0 = 80\,^\circ$C sits in a room at
$20\,^\circ$C. Newton's law of cooling states $T' = -k\,(T - 20)$ with
$k > 0$. Solve for $T(t)$, and given that the coffee is at
$50\,^\circ$C after $10$ minutes, find when it reaches
$25\,^\circ$C.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:diffeq:9}
(Damped oscillator) For $\varepsilon \geq 0$, consider
$y'' + 2\varepsilon y' + y = 0$.
\begin{enumerate}
  \item Solve for $\varepsilon \in \intco{0}{1}$, $\varepsilon = 1$,
        and $\varepsilon > 1$.
  \item Show that for $\varepsilon > 0$ every solution tends to $0$
        at $+\infty$, and that for $\varepsilon = 0$ the nonzero
        solutions do not.
  \item For $\varepsilon \in \intoo{0}{1}$, show that the zeros of a
        nonzero solution are regularly spaced, with gap
        $\frac{\pi}{\sqrt{1 - \varepsilon^2}}$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:diffeq:10}
Find all functions $f \colon \R \to \R$, twice differentiable, such
that
\[
\forall x, y \in \R, \qquad f(x + y) + f(x - y) = 2 f(x) f(y),
\]
with $f(0) \ne 0$ and $f$ not constant.
\emph{Hint: fix $y$, differentiate twice with respect to $x$ at $0$;
show $f(0) = 1$ and $f'' = c f$ for some constant $c$; then solve
according to the sign of $c$ and check which solutions satisfy the
functional equation.}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:diffeq:11}
(Euler equation) Solve $x^2 y'' - x y' + y = 0$ on
$\intoo{0}{+\infty}$. \emph{Hint: set $z(t) = y(\eu^t)$, i.e.\
substitute $x = \eu^t$, and show that $z$ satisfies a linear
equation with constant coefficients.}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:diffeq:12}
Consider the equation $x\,y' = 2y$ on the whole real line, in the
unknown differentiable function $y \colon \R \to \R$.
\begin{enumerate}
  \item Solve on $\intoo{0}{+\infty}$ and on $\intoo{-\infty}{0}$.
  \item Show that for \emph{any} constants $a, b \in \R$, the
        function equal to $ax^2$ for $x \geq 0$ and to $bx^2$ for
        $x < 0$ is differentiable on $\R$ and solves the equation
        everywhere.
  \item Conclude that the solution set on $\R$ is a two-parameter
        family, and explain why this does not contradict the
        uniqueness in \cref{thm:b1:diffeq:voc}.
\end{enumerate}
\end{exercise}

\section{Problem: The driven damped oscillator}

\begin{problem}\label{pb:b1:diffeq:1}
One equation governs a mass on a spring in a viscous medium, the
charge in an RLC circuit, and a building swaying in the wind:
\[
(E_\Omega)\colon\quad
x'' + 2\lambda x' + \omega_0^2\,x = A\cos(\Omega t),
\]
with $\lambda \geq 0$ the damping, $\omega_0 > 0$ the natural
frequency, and $A > 0$, $\Omega > 0$ the amplitude and frequency of
the forcing. This problem extracts its complete behavior: the decay
of transients, the unique periodic steady state, the resonance
curve and its sharpness (the \emph{quality factor}), the beats of
the undamped case, and the energy balance that sustains the
oscillation.\index{resonance}\index{oscillator}
Unless stated otherwise, $0 < \lambda < \omega_0$ (underdamped
regime) and we write $\omega_d = \sqrt{\omega_0^2 - \lambda^2}$.

\medskip
\noindent\textbf{Part I --- The free oscillator.} Here $A = 0$.
\begin{enumerate}
  \item Solve the homogeneous equation $(H)$ for $0 < \lambda <
        \omega_0$, and for $\lambda = 0$. (The regimes $\lambda
        \geq \omega_0$ were treated in \cref{exo:b1:diffeq:9};
        quote them.)
  \item Show that for every $\lambda > 0$, all solutions of $(H)$
        tend to $0$ at $+\infty$ --- in all three regimes.
  \item Define the energy $\mathcal E(t) = \frac12 x'(t)^2 +
        \frac12\omega_0^2\,x(t)^2$ along a solution of $(H)$. Show
        $\mathcal E'(t) = -2\lambda\,x'(t)^2 \leq 0$, and deduce
        (without solving anything) that the Cauchy problem
        ``$(H)$, $x(t_0) = x'(t_0) = 0$'' has only the zero
        solution, for every $\lambda \geq 0$.
  \item For $0 < \lambda < \omega_0$, write the nonzero solution as
        $x(t) = R\,\eu^{-\lambda t}\cos(\omega_d t - \varphi)$ and
        let $T_d = \frac{2\pi}{\omega_d}$ be the pseudo-period.
        Show that $x(t + T_d) = \eu^{-\lambda T_d}\,x(t)$: each
        swing is the previous one shrunk by the constant factor
        $\eu^{-\delta}$, $\delta = \frac{2\pi\lambda}{\omega_d}$
        (the \emph{logarithmic decrement}). Compute $\delta$ for
        $\omega_0 = 1$, $\lambda = 0.1$.
  \item Define the \emph{quality factor} $Q =
        \dfrac{\omega_0}{2\lambda}$\index{quality factor}. Show
        that after the time $\frac1\lambda$ (one amplitude
        $\eu$-folding), the oscillator has completed
        $\frac{\omega_d}{2\pi\lambda}$ pseudo-periods, which for
        weak damping ($\lambda \ll \omega_0$) is approximately
        $\frac Q\pi$: the quality factor counts, up to $\pi$, the
        oscillations survived before the amplitude decays by
        $\eu$.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The steady state.} Now $A > 0$ and
$\lambda > 0$.
\begin{enumerate}[resume]
  \item Look for a particular solution as the real part of
        $z\,\eu^{\iu\Omega t}$ with $z \in \C$
        (\cref{met:b1:diffeq:particular}). Show that this works
        with
        \[
        z = \frac{A}{\omega_0^2 - \Omega^2 + 2\iu\lambda\Omega} .
        \]
  \item Deduce the steady state in amplitude--phase form:
        $x_p(t) = R(\Omega)\cos\bigl(\Omega t -
        \varphi(\Omega)\bigr)$ with
        \[
        R(\Omega) = \frac{A}{\sqrt{(\omega_0^2 - \Omega^2)^2 +
        4\lambda^2\Omega^2}},
        \qquad
        \tan\varphi = \frac{2\lambda\Omega}{\omega_0^2 -
        \Omega^2},
        \quad \varphi \in \intoo0\pi .
        \]
  \item Interpret the two extreme regimes: compute the limits of
        $R$ and $\varphi$ as $\Omega \to 0^+$ (quasi-static
        response $A/\omega_0^2$, phase $0$) and as $\Omega \to
        +\infty$ ($R \sim A/\Omega^2 \to 0$, phase $\to \pi$: the
        mass moves opposite to a too-fast forcing).
  \item Show that \emph{every} solution of $(E_\Omega)$ is $x_p$
        plus a solution of $(H)$, hence converges to the steady
        state $x_p$ as $t \to +\infty$, whatever the initial
        conditions: after the transient dies, the oscillator has
        no memory of how it started.
  \item Show that $x_p$ is the \emph{only} periodic solution of
        $(E_\Omega)$.
  \item Work one Cauchy problem to the end: for $x'' + 2x' + 2x =
        \cos t$ with $x(0) = x'(0) = 0$, show that the solution is
        \[
        x(t) = \frac{\cos t + 2\sin t}5
        - \eu^{-t}\,\frac{\cos t + 3\sin t}5 ,
        \]
        and identify transient and steady parts.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The resonance curve.} Study of
$\Omega \mapsto R(\Omega)$ on $\intoo0{+\infty}$.
\begin{enumerate}[resume]
  \item Setting $u = \Omega^2$ and $g(u) = (\omega_0^2 - u)^2 +
        4\lambda^2 u$, show: if $2\lambda^2 < \omega_0^2$, then $R$
        attains a strict maximum at the \emph{resonance frequency}
        $\Omega_r = \sqrt{\omega_0^2 - 2\lambda^2}$, with
        \[
        R_{\max} = R(\Omega_r)
        = \frac{A}{2\lambda\sqrt{\omega_0^2 - \lambda^2}} .
        \]
  \item Show that $\dfrac{R_{\max}}{R(0)} = Q\,\bigl(1 -
        \tfrac{\lambda^2}{\omega_0^2}\bigr)^{-1/2} \geq Q$: at
        resonance, the forcing is amplified by (essentially) the
        quality factor.
  \item Prove that the \emph{velocity} amplitude $V(\Omega) =
        \Omega\,R(\Omega)$ is maximal exactly at $\Omega =
        \omega_0$ (not at $\Omega_r$), and that the phase there is
        $\varphi(\omega_0) = \frac\pi2$: at $\Omega = \omega_0$
        the velocity is exactly in phase with the force.
  \item (Bandwidth) Solve $g(u) = 2\,g(u_r)$ exactly, where $u_r =
        \omega_0^2 - 2\lambda^2$, and deduce that the two
        frequencies $\Omega_\pm$ where $R = R_{\max}/\sqrt2$
        satisfy $\Omega_+^2 - \Omega_-^2 =
        4\lambda\sqrt{\omega_0^2 - \lambda^2}$; conclude that for
        weak damping the bandwidth is $\Omega_+ - \Omega_-
        \approx 2\lambda$, i.e.\ $Q \approx
        \frac{\omega_0}{\Omega_+ - \Omega_-}$: sharp resonance
        peaks are high-$Q$ systems.
  \item Numerical portrait for $\omega_0 = 1$, $\lambda = 0.05$
        ($Q = 10$), $A = 1$: compute $\Omega_r$, $R_{\max}$, the
        static response $R(0)$, and the approximate bandwidth.
  \item Show that if $2\lambda^2 \geq \omega_0^2$, then $R$ is
        strictly decreasing on $\intoo0{+\infty}$: heavily damped
        systems have no resonance peak at all.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- No damping: beats and resonance.}
Here $\lambda = 0$.
\begin{enumerate}[resume]
  \item For $\Omega \neq \omega_0$, find the general solution of
        $x'' + \omega_0^2 x = A\cos(\Omega t)$.
  \item Solve the Cauchy problem $x(0) = x'(0) = 0$ and transform
        the answer into the product form
        \[
        x(t) = \frac{2A}{\omega_0^2 - \Omega^2}\,
        \sin\Bigl(\frac{(\omega_0 - \Omega)t}2\Bigr)
        \sin\Bigl(\frac{(\omega_0 + \Omega)t}2\Bigr) .
        \]
  \item For $\Omega$ close to $\omega_0$, read the product as a
        fast oscillation at frequency $\frac{\omega_0 + \Omega}2$
        modulated by a slow envelope at frequency
        $\frac{\abs{\omega_0 - \Omega}}2$: the \emph{beats}
        \index{beats}. Give the period of the envelope and the
        maximal amplitude, and note how both blow up as $\Omega
        \to \omega_0$.
  \item Fix $t$ and let $\Omega \to \omega_0$ in question 19's
        formula: show the limit is
        \[
        x_\infty(t) = \frac{A\,t\,\sin(\omega_0 t)}{2\omega_0} ,
        \]
        and check directly that $x_\infty$ solves the resonant
        equation $x'' + \omega_0^2 x = A\cos(\omega_0 t)$ with
        $x(0) = x'(0) = 0$ (compare
        \cref{ex:b1:diffeq:oscillation}): resonance is the limit
        of ever-slower, ever-larger beats.
  \item Contrast the two fates of resonance: linear growth
        $\frac{At}{2\omega_0}$ without damping, versus saturation
        at $R_{\max} \approx Q\,\frac{A}{\omega_0^2}$ with weak
        damping. In one sentence: what physical mechanism converts
        the first into the second?
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Energy balance and synthesis.}
\begin{enumerate}[resume]
  \item In the steady state of Part II, compute the average over
        one period $\frac{2\pi}\Omega$ of (a) the power injected
        by the forcing, $P_{\mathrm{in}}(t) = A\cos(\Omega t)
        \cdot x_p'(t)$, and (b) the power dissipated by the
        damping, $P_{\mathrm{diss}}(t) = 2\lambda\,x_p'(t)^2$.
        Show both averages equal $\lambda\,R^2\Omega^2$: the
        forcing feeds in exactly what the damping burns --- this
        is why the steady state is steady.
  \item Where exactly did the problem use: (i) the structure
        theorem \cref{thm:b1:diffeq:structure2}; (ii) the complex
        exponential method; (iii) a real-variable function study
        in the style of \cref{ch:b1:functions}? One sentence
        each.
  \item Synthesis: describe the full behavior map of $(E_\Omega)$
        --- free versus forced, damped versus undamped, the role
        of $Q$ as the single dimensionless dial tuning peak
        height, bandwidth and transient lifetime --- and mention
        where the story continues: first-order $2 \times 2$
        systems (\cref{ch:b1:matrices} and the Year 2 volume) and
        the decomposition of a general periodic forcing into
        sinusoids (Fourier series, in the Year 3 volume), for
        which this problem's sinusoidal case is the fundamental
        building block.
\end{enumerate}
\end{problem}
