\chapter{तर्कसंगत भिन्न}\label{ch:b1:fractions}

ए \omterm{def:b1:fractions:field}{तर्कसंगत अंश} का भागफल है बहुआयामी पद. केंद्रीय प्रमेय
इस छोटे अध्याय का --- आंशिक अंश
विघटन\index{आंशिक अंश अपघटन} --- कोई भी तोड़ देता है
प्राथमिक ईंटों के योग में ऐसा भागफल $\frac{c}{(X - a)^k}$.
अपनी बीजीय रुचि से परे, यह मानक मशीन है
तर्कसंगत कार्यों को एकीकृत करना (\cref{ch:b1:integration}) और के लिए
निश्चित श्रृंखला का योग (\cref{ch:b1:series}).

\section{फील्ड $ K(X)$}

\begin{definition}\label{def:b1:fractions:field}
ए \emph{तर्कसंगत अंश}\index{तर्कसंगत अंश} ऊपर $ K $ ($= \R $
या $\C $) एक भागफल है $ F = \frac{A}{B}$ साथ $ A, B \in K[X]$, $ B \neq
0$; दो भागफल $\frac AB $ और $\frac{A'}{B'}$ कब पहचाने जाते हैं
$ AB' = A'B $. प्रत्येक अंश में एक होता है \emph{छोटा रूप} साथ
$\gcd(A, B) = 1$, स्थिरांक तक अद्वितीय। प्राकृतिक संचालन के साथ,
द \omterm{def:b1:logic:sets}{तय करना} $ K(X)$ परिमेय भिन्नों का एक है \omterm{def:b1:structures:field}{मैदान}.

 \emph{डंडे}\index{डंडे} का $ F $ (संक्षिप्त रूप में) की जड़ें हैं
$ B $; \emph{आदेश} एक पोल का है इसका \omterm{def:b1:poly:derivative}{बहुलता} की जड़ के रूप में $ B $.
 \emph{डिग्री} का $ F $ है $\deg F = \deg A - \deg B \in \Z \cup
\{-\infty\}$.
\end{definition}

\begin{example}[डंडे, आदेश और डिग्री पढ़ना]\label{ex:b1:fractions:vocab}
होने देना $ F = \dfrac{X^3 - X}{X^4 - 2X^3 + X^2}$. दोनों परतों का गुणनखंड करें:
अंश $ X(X-1)(X+1)$, हर $ X^2(X-1)^2$; रद्द करें
सामान्य कारक $ X(X-1)$:
\[
F = \frac{X + 1}{X(X - 1)} \quad\text{(reduced form).}
\]
\omterm{def:b1:fractions:field}{डंडे}: $0$ और $1$, दोनों \emph{सरल} ---आदेश पढ़े जाते हैं
घटा हुआ हर, इसलिए स्पष्ट दोहरी जड़ें
मूल अप्रासंगिक हैं. डिग्री: $\deg F = 1 - 2 = -1$,
स्पर्शोन्मुख रूप से दृश्यमान ($ xF(x) \to 1$ जैसा $ x \to \infty $). द
डिग्री की तरह व्यवहार करता है \omterm{def:b1:poly:def}{बहुपद} डिग्री ($\deg FG = \deg F +
\deg G $, $\deg(F + G) \leq \max $), एक बहीखाता नियम का उपयोग किया गया
नीचे दिए गए गुणांक में लगातार शिकार होता है: प्रत्येक ``की सीमा
$ xF(x)$''तर्क प्रच्छन्न रूप से एक डिग्री गणना है।
\end{example}

\begin{proposition}[पूर्णांक भाग]\label{prop:b1:fractions:integerpart}
प्रत्येक $ F = \frac AB \in K(X)$ अद्वितीय है $ F = E + \frac RB $ साथ $ E
\in K[X]$ (द \emph{पूर्णांक भाग}, या \omterm{def:b1:poly:def}{बहुपद} का हिस्सा $ F $) और
$\deg R < \deg B $. एक के पास है $ E \neq 0$ आईएफएफ $\deg F \geq 0$.
\end{proposition}

\begin{proof}
यूक्लिडियन प्रभाग $ A = BE + R $ (\cref{thm:b1:poly:division}), द्वारा विभाजित $ B $. विशिष्टता: यदि $ E + \frac RB = E' + \frac{R'}{B}$, तब $(E -
E')B = R' - R $ साथ $\deg(R' - R) < \deg B $, मजबूर करना $ E = E'$ तब $ R
= R'$.
\end{proof}

\begin{example}[पहले घटाओ, फिर बांटो]\label{ex:b1:fractions:integerpart}
का पूर्णांक भाग ज्ञात कीजिये $ F = \dfrac{X^3 + 1}{X^2 - 1}$. बाँटना
आँख मूँद कर: $ X^3 + 1 = (X^2 - 1)X + (X + 1)$, इसलिए $ F = X +
\frac{X + 1}{X^2 - 1}$. लेकिन अंश कम नहीं हुआ: $ X^3 + 1
= (X + 1)(X^2 - X + 1)$ और $ X^2 - 1 = (X+1)(X-1)$ कारक साझा करें
$ X + 1$, और
\[
F = \frac{X^2 - X + 1}{X - 1} = X + \frac{1}{X - 1} :
\]
वही पूर्णांक भाग $ X $, लेकिन भिन्नात्मक भाग ढहकर a हो जाता है
एकल ईंट, और ``खंभा'' पर $-1$ कभी नहीं था खंभा बिल्कुल भी।
शिकार से पहले हमेशा न्यूनतम शर्तों को कम करें \omterm{def:b1:fractions:field}{डंडे}: द \omterm{def:b1:fractions:field}{डंडे} का
$ F $ की जड़ें हैं \emph{कम किया हुआ} हर (पूर्णांक
भाग सरलीकरण के प्रति असंवेदनशील है, क्योंकि इसमें विशिष्टता है
\cref{prop:b1:fractions:integerpart} गारंटी.)
\end{example}

\section{आंशिक अंश अपघटन खत्म $\C $}

\begin{theorem}[विघटन ख़त्म $\C $]\label{thm:b1:fractions:complex}
होने देना $ F = \frac AB \in \C(X)$ कम रूप में हो, साथ $ B = c\,(X -
a_1)^{m_1} \cdots (X - a_r)^{m_r}$. तब $ F $ एक अनोखे तरीके से,
\[
F = E + \sum_{i=1}^{r} \sum_{k=1}^{m_i}
\frac{c_{i,k}}{(X - a_i)^k},
\qquad E \in \C[X],\ c_{i,k} \in \C ,
\]
$ E $ का पूर्णांक भाग है $ F $.
\end{theorem}

\begin{proof}
द्वारा \cref{prop:b1:fractions:integerpart} हम मान सकते हैं $\deg A < \deg
B $ और योग अपघटन को सिद्ध करें $ E = 0$.

\emph{बँटवारा करना \omterm{def:b1:fractions:field}{डंडे}.} लिखना $ B = (X - a_1)^{m_1} B_1$ साथ
$ B_1(a_1) \neq 0$. बहुआयामी पद $(X-a_1)^{m_1}$ और $ B_1$ हैं
\omterm{cor:b1:arith:bezout}{सह अभाज्य} (कोई सामान्य जड़ नहीं), इसलिए बेज़आउट द्वारा $\C[X]$ (टिप्पणी देखें
\cref{ch:b1:poly}) वहाँ हैं $ U_0, V_0$ साथ $ U_0 (X-a_1)^{m_1} +
V_0 B_1 = 1$; से गुणा करना $ A $ और सेटिंग $ U = AU_0$, $ V =
AV_0$, फिर से विभाजित करें $ B $:
\[
\frac AB = \frac{V}{(X - a_1)^{m_1}} + \frac{U}{B_1} .
\]
डिग्री शर्तों को लागू किया जा सकता है: विभाजित करें $ V $ द्वारा
$(X-a_1)^{m_1}$, कहना $ V = (X-a_1)^{m_1}Q + V_1$ साथ $\deg V_1 <
m_1$, और भागफल को दूसरे पद में अवशोषित करें ($ U_1 = U +
QB_1$):
\[
\frac AB = \frac{V_1}{(X - a_1)^{m_1}} + \frac{U_1}{B_1},
\qquad \deg V_1 < m_1 ;
\]
डिग्री की तुलना ($\deg A < \deg B $ और $\deg V_1 < m_1$) बल
$\deg U_1 < \deg B_1$ भी। पुनरावृत्त करना $\frac{U_1}{B_1}$,
खंभा बाद खंभा, सब कुछ कम कर देता है एक ध्रुव नीचे मामला.

\emph{एक खंभा.} के लिए $\frac{A}{(X-a)^m}$ साथ $\deg A < m $: बढ़ाना $ A $
की शक्तियों में $(X - a)$, $ A = \sum_{j=0}^{m-1} \alpha_j (X - a)^j $
(टेलर का विस्तार ए \omterm{def:b1:poly:def}{बहुपद}, जैसा कि प्रमाण में है
\cref{prop:b1:poly:multiplicity}); विभाजित करने से बिल्कुल ईंटें मिलती हैं
$\frac{\alpha_j}{(X-a)^{m-j}}$. ठोस रूप से, के लिए $\frac{X^2 +
1}{(X+2)^3}$: प्रतिस्थापित करना $ X = Y - 2$,
\[
X^2 + 1 = (Y - 2)^2 + 1 = Y^2 - 4Y + 5 ,
\qquad\text{so}\qquad
\frac{X^2+1}{(X+2)^3} = \frac1{Y} - \frac4{Y^2} + \frac5{Y^3}
\]
साथ $ Y = X + 2$: के विभाजन मात्र से तीन ईंटें प्रकट हो जाती हैं
स्थानांतरित विस्तार --- सबसे तेज़ मार्ग जब भी कोई हो
उच्च कोटि का खंभा शामिल है, और इसके लिए अनुशंसित है
\cref{exo:b1:fractions:3}.

\emph{विशिष्टता.} मान लीजिए कि दो विघटन मेल खाते हैं; उनके
अंतर एक पहचान है $0 = \sum_{i,k} \frac{d_{i,k}}{(X -
a_i)^k}$. से गुणा करें $(X - a_1)^{m_1}$: प्रत्येक पद
पर लुप्त होने वाला एक कारक प्राप्त करता है $ a_1$ \emph{के अलावा} इसके साथ वाला
$ i = 1$, $ k = m_1$, जिसका गुणांक बन जाता है $ d_{1,m_1}$ प्लस
कम से कम एक कारक रखने वाले शब्द $(X - a_1)$. पर मूल्यांकन कर रहे हैं
$ a_1$ (वैध: गुणा के बाद, नहीं खंभा पर रहता है
$ a_1$) देता है $ d_{1,m_1} = 0$. शीर्ष गुणांक के चले जाने से,
के साथ दोहराएँ $(X - a_1)^{m_1 - 1}$, और इसी तरह नीचे की ओर $ k = 1$;
फिर अगले पर जाएँ खंभा. सभी $ d_{i,k}$ लुप्त:
अपघटन अद्वितीय है.
\end{proof}

\begin{method}[गुणांकों की गणना]\label{met:b1:fractions:compute}
व्यवहार में, बेज़आउट से बचें; मिलाना:
\begin{enumerate}
  \item \emph{उच्चतम शक्ति के लिए कवर-अप}: का गुणांक
        $\frac{1}{(X-a)^m}$ (साथ $ m $ का क्रम खंभा $ a $) है
        \[
        c_{a,m} = \Bigl[\,(X - a)^m F\,\Bigr]_{X = a};
        \]
        एक के लिए \emph{सरल} खंभा का $ F = \frac AB $, यह है
        $\frac{A(a)}{B'(a)}$;
  \item \emph{मूल्यांकन} सुविधाजनक बिंदुओं पर और \emph{सीमा} का
        $ xF(x)$ जैसा $ x \to \infty $ के लिए रैखिक संबंध एकत्र करना
        शेष गुणांक;
  \item \emph{समता} या संयुग्मन समरूपता, जब मौजूद हो, काटना
        काम आधा-अधूरा
\end{enumerate}
\end{method}

\begin{remark}[आंशिक भिन्नों के साथ सामान्य हानियाँ]\label{rem:b1:fractions:pitfalls}
\begin{enumerate}
  \item \emph{पूर्णांक भाग को छोड़ना.} ईंट का विघटन
        डिग्री के अंशों पर लागू होता है $< 0$; कब $\deg A \geq
        \deg B $, पहले विभाजित करें
        (\cref{prop:b1:fractions:integerpart}), या
        गुणांक खोज विरोधाभास उत्पन्न करेगी।
  \item \emph{अपने दायरे से बाहर छिपाना।} से गुणा करना
        $(X - a)^k $ और मूल्यांकन कर रहे हैं $ a $ गुणांक उत्पन्न करता है
        केवल के लिए $ k = m $, द \emph{भरा हुआ} का आदेश खंभा; द
        निचले क्रम के गुणांकों को अन्य संबंधों (सीमाओं) की आवश्यकता होती है
        मूल्यांकन) ---देखें \cref{ex:b1:fractions:multiple}.
  \item \emph{कम करना भूल रहे हैं.} \omterm{def:b1:fractions:field}{डंडे} पर पढ़े जाते हैं
        \emph{कम किया हुआ} रूप; अंश के बीच एक सामान्य कारक
        और हर प्रेत बनाता है \omterm{def:b1:fractions:field}{डंडे}
        (\cref{ex:b1:fractions:integerpart}).
  \item \emph{गलत ईंट के आकार खत्म $\R $.} एक अघुलनशील से ऊपर
        अंश द्विघात हैं \emph{एफ़िन} ($\alpha X +
        \beta $), स्थिर नहीं; लिखना $\frac{c}{X^2 + 1}$ अकेले
        समाधान खो देता है --- सही आकार निर्धारित होते हैं
        \cref{thm:b1:fractions:real}, कभी भी सुधार नहीं किया गया।
\end{enumerate}
\end{remark}

\begin{proof}[सरल-ध्रुव सूत्र का प्रमाण]
एक साधारण के पास खंभा $ a $: $ B = (X - a) Q $ साथ $ Q(a) \neq 0$, और $ B' =
Q + (X - a) Q'$, इसलिए $ B'(a) = Q(a)$. कवर-अप मूल्य है
$\frac{A(a)}{Q(a)} = \frac{A(a)}{B'(a)}$.
\end{proof}

\begin{example}\label{ex:b1:fractions:simple}
विघटित $ F = \dfrac{1}{X(X-1)(X-2)}$. तीन सरल \omterm{def:b1:fractions:field}{डंडे}; छुपाना
प्रत्येक पर:
\[
c_0 = \frac{1}{(0-1)(0-2)} = \frac12,
\quad
c_1 = \frac{1}{1 \times (1 - 2)} = -1,
\quad
c_2 = \frac{1}{2 \times 1} = \frac12,
\]
इसलिए $ F = \dfrac{1/2}{X} - \dfrac{1}{X-1} + \dfrac{1/2}{X-2}$.
\emph{पर जांचें $ X = 3$:} सीधे, $ F(3) = \frac{1}{3\cdot2\cdot1} =
\frac16$; विघटन से, $\frac{1/2}{3} - \frac{1}{2} +
\frac{1/2}{1} = \frac16 - \frac12 + \frac12 = \frac16$.
\end{example}

\begin{example}[एकाधिक ध्रुव]\label{ex:b1:fractions:multiple}
विघटित $ F = \dfrac{X}{(X-1)^2 (X+1)}$. आकार:
$\frac{a}{(X-1)^2} + \frac{b}{X-1} + \frac{c}{X+1}$.
\begin{itemize}
  \item डबल पर कवर-अप खंभा: $ a = \bigl[\frac{X}{X+1}\bigr]_{X=1}
        = \frac12$.
  \item पर पर्दा डालना $-1$: $ c = \bigl[\frac{X}{(X-1)^2}\bigr]_{X=-1} =
        -\frac14$.
  \item की सीमा $ xF(x)$ पर $\infty $: $0 = b + c $, इसलिए $ b = \frac14$.
\end{itemize}
\[
F = \frac{1/2}{(X-1)^2} + \frac{1/4}{X-1} - \frac{1/4}{X+1} .
\]
\emph{पर जांचें $ X = 0$:} $ F(0) = 0$ और $\frac12 - \frac14 - \frac14
= 0$.
\end{example}

\section{विघटन ख़त्म $\R $}

\begin{theorem}[विघटन ख़त्म $\R $]\label{thm:b1:fractions:real}
होने देना $ F \in \R(X)$ हर के साथ लघु रूप में हो
\[
B = c \prod_i (X - a_i)^{m_i} \prod_j (X^2 + p_j X + q_j)^{n_j}
\qquad (p_j^2 - 4q_j < 0).
\]
तब $ F $ अपने पूर्णांक भाग और पदों के रूप में विशिष्ट रूप से विघटित होता है
\[
\frac{c_{i,k}}{(X - a_i)^k} \quad (1 \leq k \leq m_i),
\qquad
\frac{\alpha_{j,l}\, X + \beta_{j,l}}{(X^2 + p_j X + q_j)^{l}}
\quad (1 \leq l \leq n_j),
\]
वास्तविक गुणांकों के साथ.
\end{theorem}

\begin{proof}
खत्म विघटित करें $\C $ (\cref{thm:b1:fractions:complex}). तब से $ F $ है
वास्तविक, पर गुणांक खंभा $\conj a $ (प्रत्येक आदेश पर) है
गुणांक का संयुग्मन $ a $ (संयुग्मन लागू करें
अपघटन और विशिष्टता का आह्वान)। प्रत्येक संयुग्मी युग्म को \omterm{def:b1:structures:group}{समूह}ित करें:
\[
\frac{c}{(X - z)^l} + \frac{\conj c}{(X - \conj z)^l}
= \frac{c\,(X - \conj z)^l + \conj c\,(X - z)^l}{\bigl(X^2 - 2\Re(z)X
+ \abs z^2\bigr)^{l}},
\]
जिसका अंश घात का अपना संयुग्मक है, इसलिए वास्तविक है $\leq l $;
वास्तविक द्विघात के गुणजों को विभाजित करने से यह डिग्री तक कम हो जाता है
$\leq 1$ प्रत्येक स्तर पर $ l $ (एक छोटा सा नीचे की ओर प्रेरण)। असली \omterm{def:b1:fractions:field}{डंडे}
उनके वास्तविक गुणांक रखें (संयुग्मन उन्हें ठीक करता है)। विशिष्टता
विशिष्टता से ऊपर का अनुसरण करता है $\C $.
\end{proof}

\begin{example}[संयुग्मी युग्म को देखना]\label{ex:b1:fractions:pairing}
सबसे छोटे मामले पर सबूत का तंत्र: खत्म $\C $, द
\omterm{def:b1:fractions:field}{डंडे} का $\frac1{X^2+1}$ हैं $\pm\iu $, कवर-अप गुणांक के साथ
$\frac1{2\iu}$ पर $\iu $ और $\frac1{-2\iu}$ पर $-\iu $ ---
एक दूसरे के संयुग्म, जैसा कि प्रमेय भविष्यवाणी करता है:
\[
\frac{1}{X^2 + 1}
= \frac{1/(2\iu)}{X - \iu} - \frac{1/(2\iu)}{X + \iu} .
\]
सामान्य विभाजक पर पुनर्संयोजन:
\[
\frac{1}{2\iu}\cdot\frac{(X + \iu) - (X - \iu)}{X^2 + 1}
= \frac{1}{2\iu}\cdot\frac{2\iu}{X^2 + 1} = \frac{1}{X^2+1} :
\]
काल्पनिक हिस्से रद्द हो जाते हैं और असली ईंट फिर से बरकरार दिखाई देती है।
वास्तविक एकीकृत लोगों के लिए आमतौर पर कोई कभी नहीं छोड़ता है $\R $ --- लेकिन कब
मूल्यांकन \emph{रकम} जटिल बिंदुओं पर (सप्ताहांत समस्या के रूप में)।
के साथ करता है एकता की जड़ें), जटिल ईंटें प्राकृतिक हैं
मुद्रा, और यह जोड़ी दोनों के बीच विनिमय दर है
विघटन.
\end{example}

\begin{example}\label{ex:b1:fractions:real}
विघटित $ F = \dfrac{4}{(X^2+1)(X-1)^2}$ ऊपर $\R $. आकार:
$\frac{aX + b}{X^2 + 1} + \frac{c}{(X-1)^2} + \frac{d}{X - 1}$.
डबल पर कवर-अप खंभा: $ c = \bigl[\frac{4}{X^2+1}\bigr]_{X=1} =
2$. परिसर में लीपापोती खंभा $\iu $ (संख्याकार खत्म $ X^2 + 1$
जटिल अपघटन के माध्यम से या सीधे मूल्यांकन किया गया): से गुणा करें
$ X^2 + 1$ और सेट करें $ X = \iu $:
\[
a\iu + b = \frac{4}{(\iu - 1)^2} = \frac{4}{-2\iu} = 2\iu ,
\]
इसलिए $ a = 2$, $ b = 0$. की सीमा $ xF(x)$ अनंत पर: $0 = a + d $, इसलिए $ d
= -2$. इस तरह
\[
F = \frac{2X}{X^2+1} + \frac{2}{(X-1)^2} - \frac{2}{X-1} .
\]
\emph{पर जांचें $ X = 0$:} $ F(0) = 4$ और $0 + 2 + 2 = 4$.
\end{example}

\begin{example}[एक घन हर, प्रारंभ से समाप्त तक]\label{ex:b1:fractions:cubic}
विघटित $ F = \dfrac{1}{X^3 + 1}$ ऊपर $\R $. पहला कारक: $ X^3 +
1 = (X + 1)(X^2 - X + 1)$, द्विघात विभेदक है $-3 <
0$. आकार: $\frac{a}{X+1} + \frac{bX + c}{X^2 - X + 1}$. छिपाना
सरल पर खंभा $-1$: $ a = \bigl[\frac1{X^2 - X +
1}\bigr]_{X=-1} = \frac13$. की सीमा $ xF(x)$ अनंत पर: $0 = a +
b $, इसलिए $ b = -\frac13$. पर मूल्यांकन $ X = 0$: $1 = a + c $, इसलिए $ c =
\frac23$. इस तरह
\[
\frac{1}{X^3+1} = \frac13\Bigl(\frac{1}{X+1}
+ \frac{-X + 2}{X^2 - X + 1}\Bigr) ,
\]
पर पुष्टि की गई $ X = 1$: बाईं तरफ $\frac12$, दाहिनी ओर
$\frac13\bigl(\frac12 + 1\bigr) = \frac12$. अर्थव्यवस्था पर ध्यान दें:
तीन अज्ञात, तीन सस्ते रेखीय तथ्य (एक छुपाने वाला, एक
सीमा, एक मूल्यांकन), किसी भी चीज़ का कोई विस्तार नहीं --- वर्कफ़्लो
का \cref{met:b1:fractions:compute} अपने शुद्ध रूप में.
\end{example}

\begin{remark}[यह किस लिए है]
एक बार विघटित हो जाने पर, एक तर्कसंगत फ़ंक्शन शब्द दर शब्द एकीकृत होता है: ईंटें
$\frac{1}{(x-a)^k}$ प्राथमिक आदिम और ईंटें हैं
$\frac{\alpha x + \beta}{(x^2 + px + q)^l}$ को कम करें $\ln $ और
$\arctan $ (\cref{ch:b1:integration}). टेलीस्कोपिंग योग अन्य हैं
मानक अनुप्रयोग (\cref{exo:b1:fractions:8}).
\end{remark}

\begin{remark}[जहाँ इस अध्याय का प्रयोग किया जाता है]\label{rem:b1:fractions:whereused}
आंशिक भिन्न सबसे ऊपर हैं a \emph{पूर्वप्रसंस्करण} कदम:
एकीकरण अध्याय (\cref{ch:b1:integration}) हर तर्कसंगत को खिलाता है
के माध्यम से एकीकृत करें \cref{thm:b1:fractions:real} एकीकृत करने से पहले,
और श्रृंखला अध्याय (\cref{ch:b1:series}) दूरबीन तर्कसंगत
बिल्कुल वैसे ही शब्द \cref{exo:b1:fractions:8} और सप्ताहांत में
नीचे दी गई समस्या --- जो तकनीक को हर तरह से आगे बढ़ाती है
$\sum 1/k^2 = \pi^2/6$. लघुगणकीय व्युत्पन्न $ P'/P = \sum
m_i/(X - a_i)$ (\cref{exo:b1:poly:11}) जब भी रूट होता है तो पुनः प्रकट होता है
स्थानों का अध्ययन किया जाता है। इस आयतन से परे, का अपघटन
$1/\chi(X)$ एक विशेषता के लिए \omterm{def:b1:poly:def}{बहुपद} $\chi $ को रेखांकित करता है
वर्ष में मैट्रिक्स शक्तियों और लाप्लास \omterm{def:b1:counting:objects}{परिवर्तन}ों की गणना
2 खंड: ईंटें $\frac{c}{(X - a)^k}$ \omterm{pb:b1:logic:1}{बीजगणितीय} छाया हैं
समाधानों का $ t^{k-1}\eu^{at}$ में मिले \cref{ch:b1:diffeq}.
\end{remark}

\section{अभ्यास}

\begin{exercise}[$\star $]\label{exo:b1:fractions:1}
खत्म विघटित करें $\R $: $\dfrac{1}{X^2 - 1}$;
$\;\dfrac{X}{X^2 - 3X + 2}$; $\;\dfrac{X^2 + 1}{X(X-1)}$ \emph{(मन
पूर्णांक भाग)}.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:fractions:2}
खत्म विघटित करें $\R $: $\dfrac{1}{X(X^2 + 1)}$ और
$\dfrac{X^3}{X^2 + X + 1}$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:fractions:3}
विघटित $\dfrac{1}{X^2(X - 1)}$ और $\dfrac{X + 1}{(X - 1)^3}$
\emph{(दूसरे के लिए, स्थानापन्न $ Y = X - 1$)}.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:fractions:4}
खत्म विघटित करें $\C $, फिर खत्म $\R $: $\dfrac{1}{X^4 - 1}$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:fractions:5}
खत्म विघटित करें $\R $: $\dfrac{X^2}{(X^2 + 1)^2}$, और एक निष्कर्ष निकालें
का आदिम $ x \mapsto \dfrac{x^2}{(x^2+1)^2}$ दिया गया
$\int \frac{\dd x}{(x^2+1)^2} = \frac12\bigl(\arctan x +
\frac{x}{x^2+1}\bigr) + C $.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:fractions:6}
के लिए $ n \in \N^*$, विघटित करना $ F_n = \dfrac{n!}{X(X+1)\cdots(X+n)}$
\emph{(सरल \omterm{def:b1:fractions:field}{डंडे} पर $0, -1, \dots, -n $; कवर-अप फ़ॉर्मूले का उपयोग करें
और पहचानो \omterm{def:b1:counting:objects}{द्विपद गुणांक})}.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:fractions:7}
की पहचान का उपयोग करना \cref{exo:b1:poly:11} के लिए $ P = X^n - 1$,
साबित करें कि
\[
\sum_{k=0}^{n-1} \frac{1}{X - \omega^k} = \frac{n X^{n-1}}{X^n - 1},
\qquad \omega = \eu^{2\iu\pi/n},
\]
और दोनों पक्षों का मूल्यांकन करें $ X = 2$ के लिए $ n = 4$ चेक के रूप में.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:fractions:8}
विघटित $\dfrac{1}{k(k+1)(k+2)}$ और गणना करें
\[
S_n = \sum_{k=1}^{n} \frac{1}{k(k+1)(k+2)},
\qquad\text{then}\qquad
\lim_{n \to \infty} S_n .
\]
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b1:fractions:9}
होने देना $ P \in \R[X]$ होना \omterm{def:b1:poly:def}{राक्षसी} डिग्री का $ n $ साथ $ n $ विशिष्ट वास्तविक जड़ें
$ x_1 < \dots < x_n $. साबित करें कि
\[
\sum_{i=1}^{n} \frac{1}{P'(x_i)} = 0 \quad (n \geq 2),
\qquad
\sum_{i=1}^{n} \frac{x_i^{\,n-1}}{P'(x_i)} = 1 .
\]
\emph{संकेत: विघटित करें $\frac{X^m}{P}$ के लिए $ m \leq n - 1$ और देखो
अनंत पर गुणांक क्षय --- या उपयोग करें \omterm{thm:b1:poly:lagrange}{लैग्रेंज इंटरपोलेशन}
(\cref{thm:b1:poly:lagrange}) एकपदी का $ X^m $ नोड्स पर
$ x_i $.}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:fractions:10}
विघटित $\dfrac{1}{X(X+1)^2}$ ऊपर $\R $. मूल्य स्वीकार करना
$\sum_{k \geq 1} \frac1{k^2} = \frac{\pi^2}6$ (इसमें साबित हुआ
अध्याय की सप्ताहांत समस्या), निष्कर्ष निकालें
\[
\sum_{n=1}^{\infty} \frac{1}{n(n+1)^2} = 2 - \frac{\pi^2}{6} .
\]
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:fractions:11}
खत्म विघटित करें $\R $: $\dfrac{1}{(X^2+1)(X^2+4)}$, तब
$\dfrac{X^2}{(X^2+1)(X^2+4)}$. \emph{संकेत: दोनों हर हैं
बहुआयामी पद में $ X^2$: विघटित करें $\frac1{(Y+1)(Y+4)}$ पहला।}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b1:fractions:12}
(धर्मनिरपेक्ष समीकरण) चलिए $ F = \sum_{i=1}^{r} \frac{c_i}{X - p_i}$
साथ $ p_1 < p_2 < \dots < p_r $ असली और सब कुछ $ c_i > 0$.
\begin{enumerate}
  \item बताते हैं कि $ F $ के प्रत्येक अंतराल पर सख्ती से घट रही है
        इसका डोमेन, और इसकी सीमाएं बताएं $\pm\infty $ और दोनों पर
        प्रत्येक के पक्ष खंभा.
  \item प्रत्येक के लिए यह निष्कर्ष निकालें $\lambda > 0$, समीकरण $ F(x) =
        \lambda $ बिलकुल है $ r $ वास्तविक समाधान, प्रत्येक में एक
        अंतराल $\intoo{p_i}{p_{i+1}}$ और एक परे $ p_r $.
        (ऐसे समीकरण eigenvalue गड़बड़ी को नियंत्रित करते हैं;
        मध्यवर्ती मूल्य प्रमेय का उपयोग हाई स्कूल स्तर पर किया जाता है
        यहाँ और साबित हुआ \cref{ch:b1:continuity}.)
\end{enumerate}
\end{exercise}

\section{समस्या: एक इंजन के रूप में आंशिक अंश}

\begin{problem}\label{pb:b1:fractions:1}
आंशिक अंश अपघटन बहीखाता पद्धति जैसा दिखता है; यह समस्या
दिखाता है कि यह एक इंजन है। अंश से खिलाया
$\frac1{X(X+1)\cdots(X+k)}$, यह रकम के पूरे परिवारों को दूरबीन से देखता है
बंद रूप में; के साथ खिलाया $\frac1{X^n - 1}$, यह पैदा करता है
त्रिकोणमितीय पहचान जैसे
\[
\sum_{k=1}^{n-1}\frac{1}{\sin^2\frac{k\pi}{n}} = \frac{n^2-1}{3} ;
\]
और, एक कदम आगे बढ़ने पर, वह पहचान इनमें से एक को निचोड़ देती है
गणित में सबसे प्रसिद्ध सूत्र, यूलर का
\[
\sum_{k=1}^{\infty}\frac1{k^2} = \frac{\pi^2}{6}
\]
--- यहाँ इस अध्याय के बीजगणित से परे कुछ भी प्राप्त नहीं हुआ है
हाई-स्कूल त्रिकोणमिति.\index{अंतःसर्पण}\index{बेसल समस्या}
लगातार, $\omega = \eu^{2\iu\pi/n}$; अनुक्रमों की सीमाएँ हैं
हाई स्कूल स्तर पर उपयोग किया जाता है (\cref{ch:b1:seq} उन्हें औपचारिक बनाता है)।

\medskip
\noindent\textbf{Part I --- The telescope.}
\begin{enumerate}
  \item विघटित $\frac1{X(X+1)}$ और गणना करें $\sum_{n=1}^{N}
        \frac1{n(n+1)}$ बिल्कुल; निष्कर्ष निकालें कि योग की प्रवृत्ति होती है
        $1$.
  \item के लिए भी वैसा ही $\frac1{X(X+2)}$: दिखाओ $\sum_{n=1}^{N}
        \frac1{n(n+2)} = \frac12\bigl(\frac32 - \frac1{N+1} -
        \frac1{N+2}\bigr) \to \frac34$. (अंतराल के साथ, \emph{दो}
        सीमा शर्तें प्रत्येक छोर पर जीवित रहती हैं।)
  \item तंत्र को औपचारिक बनाएं: यदि $ F(X) = G(X) - G(X+1)$ के लिए
        कुछ तर्कसंगत $ G $ बिना \omterm{def:b1:fractions:field}{डंडे} में $\intco1{+\infty}$,
        तब $\sum_{n=1}^N F(n) = G(1) - G(N+1)$. पुनर्प्राप्त करें
        मूल्य $\frac14$ का \cref{exo:b1:fractions:8} प्रदर्शन करके
        गवाह $ G $ के लिए $ F = \frac1{X(X+1)(X+2)}$.
  \item सामान्य फैक्टोरियल टेलीस्कोप सिद्ध करें: के लिए $ k \geq 1$,
        \[
        \frac{1}{X(X+1)\cdots(X+k)}
        = \frac1k\biggl(\frac{1}{X(X+1)\cdots(X+k-1)}
        - \frac{1}{(X+1)\cdots(X+k)}\biggr),
        \]
        और निष्कर्ष निकालें
        \[
        \sum_{n=1}^{\infty}\frac{1}{n(n+1)\cdots(n+k)}
        = \frac{1}{k \cdot k!} .
        \]
        मामले की जाँच करें $ k = 2$ प्रश्न 3 के विरुद्ध.
  \item के अपघटन का मूल्यांकन करें \cref{exo:b1:fractions:6}
        साबित करने के लिए अच्छी तरह से चुने गए बिंदुओं पर
        \[
        \sum_{j=0}^{n}(-1)^j\binom nj\,\frac1{j+1}
        = \frac1{n+1},
        \qquad
        \sum_{j=0}^{n}(-1)^j\binom nj\,\frac1{j+2}
        = \frac1{(n+1)(n+2)} .
        \]
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The fraction $1/(X^n - 1)$.}
\begin{enumerate}[resume]
  \item छुपाने के फार्मूले से दिखाओ कि
        \[
        \frac{1}{X^n - 1}
        = \frac1n\sum_{k=0}^{n-1}\frac{\omega^k}{X - \omega^k} .
        \]
  \item दो विवेक जांच: सूत्र को सीधे सत्यापित करें $ n =
        2$, और दिखाएँ कि का योग $ n $ गुणांक
        के लिए गायब हो जाता है $ n \geq 2$ --- समझाएं कि ऐसा क्यों करना चाहिए (विचार करें)।
        $ xF(x)$ जैसा $ x \to \infty $).
  \item की पहचान छिपाकर पुनः प्राप्त करना
        \cref{exo:b1:fractions:7}:
        $\frac{nX^{n-1}}{X^n-1} = \sum_k \frac1{X - \omega^k}$.
  \item \omterm{def:b1:structures:group}{समूह} संयुग्म \omterm{def:b1:fractions:field}{डंडे} वास्तविक अपघटन सिद्ध करने के लिए:
        के साथ $\theta_k = \frac{2k\pi}n $,
        \[
        \frac{\omega^k}{X - \omega^k} +
        \frac{\omega^{n-k}}{X - \omega^{n-k}}
        = \frac{2\cos\theta_k\,X - 2}
        {X^2 - 2\cos\theta_k\,X + 1} ,
        \]
        और इसका पूर्ण वास्तविक अपघटन लिखिए
        $\frac1{X^n-1}$ (अंतर करना $ n $ अजीब और $ n $ यहां तक की)।
  \item करने में विशेषज्ञता $ n = 4$ और इसके विरुद्ध जाँच करें
        \cref{exo:b1:fractions:4}.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Trigonometric sums, and Euler's
$\pi^2/6$.} होने देना $ P = 1 + X + \dots + X^{n-1}$, जिनकी जड़ें हैं
$\omega, \omega^2, \dots, \omega^{n-1}$ (सभी सरल).
\begin{enumerate}[resume]
  \item का उपयोग करते हुए $\frac{P'}P = \sum_{k=1}^{n-1}\frac1{X - \omega^k}$
        (\cref{exo:b1:poly:11}), सिद्ध करना
        \[
        \sum_{k=1}^{n-1}\frac{1}{1 - \omega^k} = \frac{n-1}2 .
        \]
  \item सिद्ध करना $\dfrac1{1 - \eu^{\iu\theta}} = \dfrac12 +
        \dfrac\iu2\,\frac{\cos(\theta/2)}{\sin(\theta/2)}$ के लिए
        $\theta \notin 2\pi\Z $ (अर्ध-कोण गुणनखंडन,
        \cref{met:b1:complex:trig}), और प्रश्न 11 से निष्कर्ष निकालें
        वह $\sum_{k=1}^{n-1}
        \frac{\cos(k\pi/n)}{\sin(k\pi/n)} = 0$ ---दिखाई भी देता है
        समरूपता द्वारा $ k \leftrightarrow n - k $.
  \item प्रश्न 11 की पहचान में अंतर करना (अर्थात\ का उपयोग करते हुए
        $\bigl(\frac{P'}P\bigr)' = \frac{P''}P -
        \bigl(\frac{P'}P\bigr)^2$ पर मूल्यांकन किया गया $ X = 1$), सिद्ध करना
        \[
        \sum_{k=1}^{n-1}\frac{1}{(1 - \omega^k)^2}
        = \frac{(n-1)(5-n)}{12} .
        \]
  \item लिखना $\cot t = \frac{\cos t}{\sin t}$, से निष्कर्ष निकालें
        प्रश्न 12-13 दो बंद प्रपत्र
        \[
        \sum_{k=1}^{n-1}\cot^2\frac{k\pi}n = \frac{(n-1)(n-2)}3,
        \qquad
        \sum_{k=1}^{n-1}\frac1{\sin^2\frac{k\pi}n}
        = \frac{n^2-1}3 .
        \]
  \item दोनों फ़ॉर्मूलों को हाथ से सत्यापित करें $ n = 3$ और $ n = 4$.
  \item असमानताओं को सिद्ध करें $\cot t < \frac1t < \frac1{\sin t}$
        के लिए $ t \in \intoo0{\frac\pi2}$ (से $\sin t < t < \tan
        t $), और निष्कर्ष निकालें, के लिए $ n = 2m + 1$ और $1 \leq k \leq m $:
        \[
        \cot^2\frac{k\pi}n \;<\; \frac{n^2}{k^2\pi^2}
        \;<\; \frac1{\sin^2\frac{k\pi}n} .
        \]
  \item इन असमानताओं का योग कीजिए $ k = 1, \dots, m $ (का उपयोग करते हुए
        समरूपता $ k \leftrightarrow n - k $ सूत्रों को आधा करने के लिए
        प्रश्न 14 का) और निचोड़ें:
        \[
        \sum_{k=1}^{\infty}\frac1{k^2} = \frac{\pi^2}6 .
        \]
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Higher bricks.}
\begin{enumerate}[resume]
  \item के अपघटन का वर्ग करके $\frac1{X^2-1}$ और
        क्रॉस टर्म को फिर से विघटित करना, साबित करना
        \[
        \frac1{(X^2-1)^2}
        = \frac14\Bigl(\frac1{(X-1)^2} + \frac1{(X+1)^2}\Bigr)
        - \frac14\Bigl(\frac1{X-1} - \frac1{X+1}\Bigr),
        \]
        और इसकी जांच करें $ X = 0$.
  \item प्रश्न 18, दूरबीन और यूलर के मूल्य को मिलाएं
        (प्रश्न 17) सिद्ध करना
        \[
        \sum_{n=2}^{\infty}\frac1{(n^2-1)^2}
        = \frac{\pi^2}{12} - \frac{11}{16} ,
        \]
        और तीन दशमलव तक संख्यात्मक रूप से मान की पुष्टि करें।
  \item a प्राप्त करने के लिए प्रश्न 8 की पहचान में अंतर करें
        के लिए बंद प्रपत्र $\sum_{k=0}^{n-1}\frac1{(X -
        \omega^k)^2}$, और इसकी जांच करें $ X = 2$, $ n = 2$.
  \item इसके लिए सिद्ध करें $ k \geq 2$,
        \[
        \sum_{n=k}^{\infty}\frac{1}{\binom nk} = \frac{k}{k-1} .
        \]
        \emph{(प्रश्न 4 को लिखकर छोटा करें $1/\binom nk $ के साथ
        भाज्य.)}
  \item के लिए $ k = 3$, सटीक आंशिक योग दें
        $\sum_{n=3}^{N}\frac1{\binom n3}$ और इसकी सीमा.
\end{enumerate}

\noindent\textbf{Part V --- Synthesis.}
\begin{enumerate}[resume]
  \item कैपस्टोन गणना के रूप में, पूर्ण वास्तविक लिखें
        का विघटन $\dfrac1{X^6 - 1}$ और इसकी जांच करें $ X =
        0$.
  \item वास्तव में समस्या का उपयोग कहाँ हुआ: (i) की विशिष्टता
        विघटन; (ii) द एकता की जड़ें से
        \cref{ch:b1:complex}; (iii) लघुगणकीय व्युत्पन्न
        से \cref{exo:b1:poly:11}? एक-एक वाक्य.
  \item संश्लेषण, एक संक्षिप्त अनुच्छेद में: एक \omterm{pb:b1:logic:1}{बीजगणितीय} पहचान
        --- एक अंश को ईंटों में तोड़ना --- सटीक उत्पन्न होता है
        योग, त्रिकोणमितीय पहचान, और $\pi^2/6$. पर टिप्पणी करें
        बीजगणित के बीच श्रम का विभाजन (सटीक)।
        विघटन, हर जगह मान्य) और विश्लेषण (सीमाएँ,
        निचोड़ना), और इंगित करें कि प्रत्येक धागा कहाँ है
        औद्योगीकृत: दूरबीन और तुलना
        \cref{ch:b1:series}, ईंटों का एकीकरण
        \cref{ch:b1:integration}.
\end{enumerate}
\end{problem}
