\chapter{जटिल संख्याएँ}\label{ch:b1:complex}

कम्प्यूटेशनल के रूप में हाई स्कूल वॉल्यूम में जटिल संख्याएँ पूरी की गईं
द्विघात समीकरणों के लिए उपकरण. यह अध्याय उन्हें केन्द्रीय मानता है
वस्तु: घातीय रूप और उसके परिणाम (डी मोइवरे, $n $-वें
जड़ें, एकता की जड़ें), के बीच व्यवस्थित अनुवाद $\C $ और
समतल ज्यामिति, और इसका उपयोग $\eu^{\iu\theta}$ के लिए एक मशीन के रूप में
त्रिकोणमितीय सर्वसमिकाओं को सिद्ध करना।

\section{फील्ड $\C $, मापांक और संयुग्म}

\begin{definition}[जटिल क्षेत्र]\label{def:b1:complex:field}
$\C = \{a + \iu b : a, b \in \R\}$ सामान्य जोड़ के साथ और
गुणन द्वारा निर्धारित किया जाता है $\iu^2 = -1$. प्रत्येक अशून्य $ z = a + \iu
b $ इसका उलटा है: $ z^{-1} = \frac{a - \iu b}{a^2 + b^2}$ --- में
की भाषा \cref{ch:b1:structures}, $\C $ एक मैदान है. एक लिखता है $ a =
\Re(z)$, $ b = \Im(z)$, $\conj{z} = a - \iu b $ (द
\emph{संयुग्मित}\index{संयुग्मित}) और $\abs{z} = \sqrt{a^2 + b^2}$
(द \emph{मापांक}\index{मापांक}).
\end{definition}

\begin{proposition}[संयुग्म और मापांक के नियम]\label{prop:b1:complex:rules}
के लिए $ z, w \in \C $:
\begin{enumerate}
  \item $\conj{z + w} = \conj z + \conj w $, $\conj{zw} = \conj z\,
        \conj w $, $\conj{\conj z} = z $;
  \item $ z \conj z = \abs z^2$; $\;\Re(z) = \frac{z + \conj z}{2}$,
        $\Im(z) = \frac{z - \conj z}{2\iu}$;
  \item $\abs{zw} = \abs z\, \abs w $, और $\abs{z^{-1}} =
        \abs{z}^{-1}$ के लिए $ z \neq 0$;
  \item (त्रिकोण असमानता\index{त्रिकोण असमानता}) $\abs{z + w}
        \leq \abs z + \abs w $, समानता के साथ यदि और केवल यदि $ z $ और
        $ w $ से एक सामान्य किरण पर लेटें $0$ ($ w = \lambda z $ या
        $ z = \lambda w $ साथ $\lambda \geq 0$);
  \item (विपरीत त्रिभुज असमानता) $\bigl|\abs z - \abs w\bigr|
        \leq \abs{z - w}$.
\end{enumerate}
\end{proposition}

\begin{proof}
(1) और (2) वास्तविक और काल्पनिक भागों पर सीधी गणना हैं। (3):
$\abs{zw}^2 = zw\,\conj{zw} = z\conj z\, w \conj w = \abs z^2 \abs
w^2$, फिर वर्गमूल लें; पर लागू $ z \cdot z^{-1} = 1$ के लिए
उलटा.

(4) दोनों पक्ष अऋणात्मक हैं, इसलिए वर्गों की तुलना करें:
\[
\abs{z+w}^2 = (z+w)(\conj z + \conj w)
= \abs z^2 + \abs w^2 + 2\,\Re(z \conj w),
\]
और $\Re(z\conj w) \leq \abs{z \conj w} = \abs z \abs w $ देता है
$\abs{z+w}^2 \leq (\abs z + \abs w)^2$. समानता बल $\Re(z \conj
w) = \abs{z \conj w}$, यानी\ $ z \conj w \in \R_+$; अगर $ w \neq 0$ यह
देता है $ z = \frac{z\conj w}{\abs w^2}\, w = \lambda w $ साथ $\lambda
\geq 0$ (और मामला $ w = 0$ तुच्छ है)।

(5) $\abs z = \abs{(z - w) + w} \leq \abs{z-w} + \abs w $ देता है $\abs z
- \abs w \leq \abs{z - w}$; बदलना $ z $ और $ w $ दूसरे चिन्ह के लिए.
\end{proof}

\begin{example}[एक संपूर्ण मापांक-और-तर्क कसरत]\label{ex:b1:complex:workout}
रखना $ w = \dfrac{3 + 4\iu}{1 - 2\iu}$ \omterm{pb:b1:logic:1}{बीजगणितीय} रूप में और गणना करें
इसका \omterm{def:b1:complex:field}{मापांक} दो बार। से गुणा करना \omterm{def:b1:complex:field}{संयुग्मित} का
हर:
\[
w = \frac{(3 + 4\iu)(1 + 2\iu)}{(1 - 2\iu)(1 + 2\iu)}
= \frac{3 + 6\iu + 4\iu - 8}{1 + 4}
= \frac{-5 + 10\iu}{5} = -1 + 2\iu .
\]
सीधे: $\abs w = \sqrt{1 + 4} = \sqrt5$. भागफल नियम के माध्यम से
(\cref{prop:b1:complex:rules}~(3)): $\abs w = \frac{\abs{3 +
4\iu}}{\abs{1 - 2\iu}} = \frac5{\sqrt5} = \sqrt5$ --- वही उत्तर,
किसी \omterm{pb:b1:logic:1}{बीजगणितीय} रूप की आवश्यकता नहीं है। पाठ सामान्यीकरण करता है: मॉड्यूलि और
तर्क उत्पादों और भागफलों, वास्तविक और के माध्यम से अच्छी तरह से यात्रा करते हैं
काल्पनिक भाग रकम के माध्यम से अच्छी तरह से यात्रा करते हैं। चुनें
प्रतिनिधित्व जो हाथ में संचालन से मेल खाता है, और परिवर्तित होता है
केवल जब मजबूर किया जाए.
\end{example}

\begin{example}[संयुग्म से जुड़े समीकरण]\label{ex:b1:complex:conjeq}
में हल करें $\C $: $\;z + 2\conj z = 6 + 2\iu $. एक समीकरण मिश्रण $ z $
और $\conj z $ है \emph{नहीं} बहुपद में $ z $; विश्वसनीय कदम
वास्तविक निर्देशांक में विभाजित करना है। साथ $ z = x + \iu y $:
\[
z + 2\conj z = 3x - \iu y ,
\]
तो समीकरण पढ़ता है $3x = 6$ और $-y = 2$: अनोखा समाधान
है $ z = 2 - 2\iu $. (जाँच करना: $(2 - 2\iu) + 2(2 + 2\iu) = 6 +
2\iu $.) वैकल्पिक रूप से, \omterm{def:b1:complex:field}{संयुग्मित} पाने के लिए पूरा समीकरण
$\conj z + 2z = 6 - 2\iu $ और रैखिक प्रणाली को हल करें
अज्ञात $ z, \conj z $ --- वही उत्तर, और एक उपयोगी युक्ति जब
गुणांक जटिल हैं. में समीकरण $ z $ और $\conj z $ हैं
वास्तव में दो वास्तविक समीकरणों की प्रणाली; ``डिग्री'' की उम्मीद $1$, एक
समाधान'' यहां सुरक्षित है, लेकिन $ z\conj z = -1$ (कोई समाधान नहीं) दिखाता है
उत्पाद प्रकट होते ही बहुपद अंतर्ज्ञान विफल हो जाता है।
\end{example}

\section{घातांकीय रूप}

\begin{definition}[एक काल्पनिक तर्क का जटिल घातांक]\label{def:b1:complex:expi}
के लिए $\theta \in \R $ एक परिभाषित करता है
\[
\eu^{\iu\theta} = \cos\theta + \iu \sin\theta .
\]
प्रत्येक $ z \neq 0$ लिखा जा सकता है $ z = r\,\eu^{\iu\theta}$ साथ $ r =
\abs z > 0$; $\theta $ एक \emph{तर्क}\index{तर्क} का $ z $,
का गुणज जोड़ने तक का निर्णय लिया गया $2\pi $. में मूल्य
$\intoc{-\pi}{\pi}$ है \emph{प्रमुख तर्क}, लिखा हुआ
$\arg z $.
\end{definition}

\begin{example}[पहले मूल्य, और एक प्रसिद्ध पहचान]\label{ex:b1:complex:eulerid}
कार्डिनल कोणों पर परिभाषा पढ़ना:
\[
\eu^{\iu\pi/2} = \iu, \qquad
\eu^{\iu\pi} = -1, \qquad
\eu^{2\iu\pi} = 1, \qquad
\eu^{\iu\pi/4} = \frac{\sqrt2}2\,(1 + \iu) .
\]
दूसरा, इस प्रकार पुनर्व्यवस्थित किया गया $\eu^{\iu\pi} + 1 = 0$, यूलर का है
पहचान जोड़ने का जश्न मनाया $\eu $, $\iu $, $\pi $, $1$ और $0$; पर
पुस्तक के इस चरण में यह एक के बजाय खुलने वाली एक परिभाषा है
प्रमेय, और इसकी वास्तविक सामग्री --- विश्लेषणात्मक घातांक क्यों
का कार्य \cref{ch:b1:functions}, काल्पनिक तक विस्तारित
तर्क, उसी नाम के पात्र हैं --- शक्ति द्वारा तय किया जाता है
की श्रृंखला \cref{ch:b1:series}. इस बीच ऊपर डिस्प्ले है
रूपांतरण तालिका के रूप में याद रखने योग्य: इसका उपयोग हर बार चुपचाप किया जाता है
समय एक तर्क को एक तस्वीर से पढ़ा जाता है।
\end{example}

\begin{theorem}[कार्यात्मक समीकरण]\label{thm:b1:complex:funceq}
सभी के लिए $\theta, \varphi \in \R $:
\[
\eu^{\iu\theta}\, \eu^{\iu\varphi} = \eu^{\iu(\theta + \varphi)},
\qquad
\abs{\eu^{\iu\theta}} = 1,
\qquad
\conj{\eu^{\iu\theta}} = \eu^{-\iu\theta} = (\eu^{\iu\theta})^{-1}.
\]
फलस्वरूप $\abs{zw} = \abs z \abs w $ तर्क भी रखता है:
$\arg(zw) \equiv \arg z + \arg w \pmod{2\pi}$.
\end{theorem}

\begin{proof}
उत्पाद का विस्तार करें और अतिरिक्त सूत्रों का उपयोग करें:
\[
(\cos\theta + \iu\sin\theta)(\cos\varphi + \iu\sin\varphi)
= (\cos\theta\cos\varphi - \sin\theta\sin\varphi)
+ \iu\,(\sin\theta\cos\varphi + \cos\theta\sin\varphi),
\]
जो है $\cos(\theta+\varphi) + \iu\sin(\theta+\varphi)$. \omterm{def:b1:complex:field}{मापांक}
है $\sqrt{\cos^2\theta + \sin^2\theta} = 1$, और यह \omterm{def:b1:complex:field}{संयुग्मित} सूत्र
कोसाइन और साइन की समता है; यह उलटा है $\eu^{\iu\theta}$ तब से
$\eu^{\iu\theta}\eu^{-\iu\theta} = \eu^0 = 1$.
\end{proof}

\begin{corollary}[डी मोइवरे का सूत्र]\label{cor:b1:complex:demoivre}
के लिए $\theta \in \R $ और $ n \in \Z $:
$\;(\cos\theta + \iu\sin\theta)^n = \cos(n\theta) +
\iu\sin(n\theta)$.\index{डी मोइवरे का सूत्र}
\end{corollary}

\begin{proof}
के लिए $ n \geq 0$, प्रेरण. मामला $ n = 0$ पढ़ता $1 = 1$. मान लेना
के लिए सूत्र $ n $, कार्यात्मक समीकरण
(\cref{thm:b1:complex:funceq}) देता है
\[
\bigl(\eu^{\iu\theta}\bigr)^{n+1}
= \bigl(\eu^{\iu\theta}\bigr)^{n}\,\eu^{\iu\theta}
= \eu^{\iu n\theta}\,\eu^{\iu\theta}
= \eu^{\iu(n+1)\theta} ,
\]
जो रैंक पर सूत्र है $ n + 1$. के लिए $ n < 0$, लिखना $ n = -m $
साथ $ m > 0$: तब से $\eu^{\iu m\theta}$ उलटा है
$\eu^{-\iu m\theta}$ (\cref{thm:b1:complex:funceq} दोबारा),
\[
\bigl(\eu^{\iu\theta}\bigr)^{-m}
= \bigl(\eu^{\iu m\theta}\bigr)^{-1}
= \eu^{\iu(-m)\theta} . \qedhere
\]
\end{proof}

\begin{example}[विस्तार $\cos 3\theta $ डी मोइवरे द्वारा]\label{ex:b1:complex:cos3}
लिखना $ c = \cos\theta $, $ s = \sin\theta $. डी मोइवरे और द्विपद
प्रमेय दे
\[
\cos 3\theta + \iu \sin 3\theta = (c + \iu s)^3
= c^3 - 3cs^2 + \iu\,(3c^2 s - s^3),
\]
और वास्तविक भागों की पहचान करना, फिर प्रतिस्थापित करना $ s^2 = 1 - c^2$:
\[
\cos 3\theta = c^3 - 3c(1 - c^2) = 4\cos^3\theta - 3\cos\theta .
\]
काल्पनिक भाग उपज देता है $\sin 3\theta = 3\sin\theta -
4\sin^3\theta $ निःशुल्क: एक जटिल पहचान हमेशा साथ लेकर चलती है
\emph{दो} असली में से एक। पीछे की ओर पढ़ें, बॉक्स वाली पहचान है
शास्त्रीय ट्राइसेक्शन समीकरण की कुंजी: निर्माण
$\cos(\theta/3)$ से $\cos\theta $ मतलब घन को हल करना $4x^3 -
3x = \cos\theta $, जो कि बीजगणित है
\cref{ch:b1:poly} अधिग्रहण।
\end{example}

\begin{proposition}[यूलर के सूत्र]\label{prop:b1:complex:euler}
\[
\cos\theta = \frac{\eu^{\iu\theta} + \eu^{-\iu\theta}}{2},
\qquad
\sin\theta = \frac{\eu^{\iu\theta} - \eu^{-\iu\theta}}{2\iu}.
\]
\end{proposition}

\begin{proof}
क्रमशः जोड़ें, घटाएँ, $\eu^{\iu\theta} = \cos\theta +
\iu\sin\theta $ और $\eu^{-\iu\theta} = \cos\theta - \iu\sin\theta $.
\end{proof}

\begin{method}[घातांक के माध्यम से त्रिकोणमिति]\label{met:b1:complex:trig}
\begin{enumerate}
  \item \emph{रैखिक बनाना} $\cos^p\theta \sin^q\theta $ (शक्तियों को चालू करें
        का योग $\cos k\theta $, $\sin k\theta $): यूलर का स्थानापन्न करें
        सूत्र, द्विपद प्रमेय के साथ विस्तार करें
        (\cref{thm:b1:counting:binomial}), पुनः समूहित करें \omterm{def:b1:complex:field}{संयुग्मित} शर्तें।
  \item \emph{बढ़ाना} $\cos n\theta $ में एक बहुपद के रूप में $\cos\theta $:
        लिखना $\cos n\theta = \Re\bigl((\cos\theta +
        \iu\sin\theta)^n\bigr)$, विस्तार करें और यहां तक कि शक्तियों को परिवर्तित करें
        $\sin $ के जरिए $\sin^2 = 1 - \cos^2$.
  \item \emph{योग त्रिकोणमितीय श्रृंखला} पसंद $\sum_k \cos k\theta $:
        एक ज्यामितीय योग के वास्तविक भाग को पहचानें $\sum_k
        (\eu^{\iu\theta})^k $.
  \item  \emph{अर्ध-कोण गुणनखंडन}: सभी के लिए $ p, q $,
        \[
        \eu^{\iu p} + \eu^{\iu q}
        = 2 \cos\tfrac{p - q}{2}\; \eu^{\iu \frac{p+q}{2}},
        \qquad
        \eu^{\iu p} - \eu^{\iu q}
        = 2\iu \sin\tfrac{p - q}{2}\; \eu^{\iu \frac{p+q}{2}} .
        \]
\end{enumerate}
\end{method}

\begin{example}[linearization]\label{ex:b1:complex:linearize}
\[
\cos^3\theta
= \Bigl(\frac{\eu^{\iu\theta} + \eu^{-\iu\theta}}{2}\Bigr)^{\!3}
= \frac{\eu^{3\iu\theta} + 3\eu^{\iu\theta} + 3\eu^{-\iu\theta} +
\eu^{-3\iu\theta}}{8}
= \frac{\cos 3\theta + 3\cos\theta}{4}.
\]
यह फॉर्म तुरंत एकीकृत होता है --- बहुत ही कारण रैखिकरण
में मायने रखता है \cref{ch:b1:integration}. एक मिश्रित उत्पाद काम करता है
उसी तरह, केवल दोनों यूलर फ़ार्मुलों के साथ एक साथ:
\[
\sin^2\theta\cos^2\theta
= \Bigl(\frac{\sin2\theta}2\Bigr)^{\!2}
= \frac{1}{4}\cdot
\Bigl(\frac{\eu^{2\iu\theta} - \eu^{-2\iu\theta}}{2\iu}
\Bigr)^{\!2}
= \frac{2 - \eu^{4\iu\theta} - \eu^{-4\iu\theta}}{16}
= \frac{1 - \cos4\theta}{8} ,
\]
जहां पहले चरण में डबल-एंगल शॉर्टकट ने एक को बचाया
द्विपद विस्तार --- हमेशा पहले से स्कैन करने लायक
मशीनीकरण.
\end{example}

\begin{example}[एक द्विपद त्रिकोणमितीय योग]\label{ex:b1:complex:binomsum}
के लिए $ n \in \N $ और $\theta \in \R $, मूल्यांकन करना $ S =
\sum_{k=0}^{n}\binom nk \cos k\theta $. के असली हिस्से को पहचानें
एक द्विपद विस्तार:
\[
S = \Re\sum_{k=0}^n \binom nk \bigl(\eu^{\iu\theta}\bigr)^k
= \Re\bigl(1 + \eu^{\iu\theta}\bigr)^n ,
\]
फिर आधे कोण का गुणनखंड करें (\cref{met:b1:complex:trig}~(4)): $1 +
\eu^{\iu\theta} = 2\cos\frac\theta2\,\eu^{\iu\theta/2}$, इसलिए
\[
S = \Re\Bigl(2^n\cos^n\frac\theta2\;\eu^{\iu n\theta/2}\Bigr)
= 2^n \cos^n\frac\theta2\,\cos\frac{n\theta}2 .
\]
काल्पनिक भाग उपज देता है $\sum_k\binom nk\sin k\theta =
2^n\cos^n\frac\theta2\sin\frac{n\theta}2$ मुक्त करने के लिए। स्वच्छता जांच:
$\theta = 0$ ठीक $\sum\binom nk = 2^n $, और $\theta = \pi $
देता है $ S = 0$ के लिए $ n \geq 1$ (प्रत्येक कारक $\cos\frac\pi2$
गायब हो जाता है), यानी\ का वैकल्पिक पंक्ति योग
\cref{ex:b1:counting:binomial}. विधि --- ``कोसाइन देखें
एक जटिल शक्ति की छाया के रूप में योग करें, फिर अर्ध-कोणों का गुणनखंड करें''
--- बिलकुल वैसा ही है \cref{exo:b1:complex:6}, द्विपद के साथ
ज्यामितीय श्रृंखला की जगह लेने वाला प्रमेय।
\end{example}

\section{सम्मिश्र संख्याओं की जड़ें}

\begin{theorem}[$ n $-वें जड़ें]\label{thm:b1:complex:roots}
होने देना $ a = r\,\eu^{\iu\alpha} \neq 0$ और $ n \in \N^*$. समीकरण
$ z^n = a $ बिलकुल है $ n $ समाधान:
\[
z_k = r^{1/n}\, \eu^{\iu\left(\frac{\alpha}{n} +
\frac{2k\pi}{n}\right)},
\qquad k = 0, 1, \dots, n - 1 .
\]
\end{theorem}

\begin{proof}
लिखना $ z = \rho\,\eu^{\iu\theta}$ ($\rho > 0$). तब $ z^n =
\rho^n \eu^{\iu n\theta} = r \eu^{\iu\alpha}$ अगर और केवल अगर $\rho^n
= r $ (मोडुलि) और $ n\theta \equiv \alpha \pmod{2\pi}$ (तर्क),
यानी\ $\rho = r^{1/n}$ और $\theta = \frac{\alpha}{n} +
\frac{2k\pi}{n}$ कुछ के लिए $ k \in \Z $. देखना यह है कि कब दो
पूर्णांक $ k, k'$ समान संख्या प्राप्त करें: यह बिल्कुल तब होता है जब
कोणों के गुणज में अंतर होता है $2\pi $,
\[
\frac{2k\pi}n - \frac{2k'\pi}n \in 2\pi\Z
\iff \frac{k - k'}n \in \Z
\iff n \mid k - k' .
\]
यूक्लिडियन डिवीजन द्वारा, हर $ k \in \Z $ सर्वांगसम मॉड है $ n $ को
बिल्कुल एक तत्व $\{0, 1, \dots, n-1\}$, इसलिए यह श्रेणी सूचीबद्ध है
प्रत्येक समाधान एक बार और गिनती बिल्कुल सही है $ n $. (वही
तर्क करो, अंदर भागो $\mathbb U_n $, दिखाता है कि जड़ें नियमित रूप से बनती हैं
$ n $-गॉन: के लगातार मान $ k $ निश्चित कोण से घुमाएँ
$\frac{2\pi}n $.)
\end{proof}

\begin{example}[की घन जड़ें $-27$]\label{ex:b1:complex:cuberoots}
हल करना $ z^3 = -27$. दाहिनी ओर का घातीय रूप: $-27 =
27\,\eu^{\iu\pi}$, तो तीन जड़ें हैं
\[
z_k = 3\,\eu^{\iu(\frac\pi3 + \frac{2k\pi}3)}, \quad k = 0, 1, 2 :
\qquad
z_0 = 3\eu^{\iu\pi/3} = \frac32 + \frac{3\sqrt3}2\,\iu,
\quad
z_1 = -3,
\quad
z_2 = \conj{z_0} .
\]
दो चेक. सबसे पहले, असली जड़ $-3$ स्पष्ट एक है, और
अन्य दो इसके घूर्णन हैं $\pm\frac{2\pi}3$ --- समान रूप से
$-3j $ और $-3j^2$. दूसरा, बीजगणित पुष्टि करता है: $ z^3 + 27 = (z +
3)(z^2 - 3z + 9)$, और द्विघात में विभेदक है $9 - 36 =
-27 < 0$ जड़ों के साथ $\frac{3 \pm 3\iu\sqrt3}2 = z_0, \conj{z_0}$.
अंतर्दृष्टि: वास्तविक दाहिनी ओर के लिए, हमेशा गैर-वास्तविक जड़ें
अंदर आओ \omterm{def:b1:complex:field}{संयुग्मित} जोड़े, तो समाधान की एक तस्वीर \omterm{def:b1:logic:sets}{तय करना} है
वास्तविक अक्ष के बारे में सममित --- वास्तविक का पूर्वावलोकन
का गुणनखंडन प्रमेय \cref{ch:b1:poly}.
\end{example}

\begin{example}[एक अवास्तविक दाहिना हाथ]\label{ex:b1:complex:fourthroots}
हल करना $ z^4 = -8 + 8\iu\sqrt3$. दाहिनी ओर का घातीय रूप:
\omterm{def:b1:complex:field}{मापांक} $\sqrt{64 + 192} = 16$, तर्क $\theta $ साथ
$\cos\theta = -\frac12$, $\sin\theta = \frac{\sqrt3}2$, अर्थात\
$\theta = \frac{2\pi}3$. चार जड़ें हैं
\[
z_k = 2\,\eu^{\iu(\frac\pi6 + \frac{k\pi}2)}, \quad k = 0, 1, 2,
3 :
\qquad
z_0 = \sqrt3 + \iu,\quad z_1 = \iu z_0 = -1 + \iu\sqrt3,
\]
\[
z_2 = -z_0 = -\sqrt3 - \iu,\qquad z_3 = -\iu z_0 = 1 - \iu\sqrt3 .
\]
(जाँच करना: $ z_0^2 = 2 + 2\iu\sqrt3$, इसलिए $ z_0^4 = (2 +
2\iu\sqrt3)^2 = 4 - 12 + 8\iu\sqrt3 = -8 + 8\iu\sqrt3$।) एक बार
\emph{एक} जड़ मिल जाती है, अन्य तीन मुक्त हो जाते हैं: वे इसके हैं
द्वारा क्रमिक घूर्णन $\frac\pi2$, यानी\ इसके उत्पादों के साथ
चौथा एकता की जड़ें --- पीछे सामान्य संरचना
\cref{thm:b1:complex:roots}, पुनर्गणना से पहले दोहन के लायक
खरोंच से प्रत्येक जड़. नहीं \omterm{def:b1:complex:field}{संयुग्मित} इस बार समरूपता:
दायीं ओर का भाग वास्तविक नहीं है।
\end{example}

\begin{definition}[एकता की जड़ें]\label{def:b1:complex:unity}
 \emph{$ n $- एकता की जड़ें}\index{$ n $- एकता की जड़ें} हैं
के समाधान $ z^n = 1$:
\[
\mathbb{U}_n = \bigl\{\, \omega^k : k = 0, \dots, n-1 \,\bigr\},
\qquad \omega = \eu^{2\iu\pi/n}.
\]
वे गुणन के अंतर्गत एक समूह बनाते हैं (\cref{ch:b1:structures}) और
एक नियमित शीर्ष पर बैठें $ n $-गॉन यूनिट सर्कल में अंकित है।
\end{definition}

\begin{proposition}[एकता की जड़ों का योग]\label{prop:b1:complex:sumroots}
के लिए $ n \geq 2$, द $ n $-वां एकता की जड़ें शून्य का योग:
$\sum_{k=0}^{n-1} \omega^k = 0$.
\end{proposition}

\begin{proof}
अनुपात के साथ ज्यामितीय योग $\omega \neq 1$:
$\sum_{k=0}^{n-1} \omega^k = \frac{\omega^n - 1}{\omega - 1} = 0$
तब से $\omega^n = 1$.
\end{proof}

\begin{example}[वास्तविक और काल्पनिक भागों को पढ़ना]\label{ex:b1:complex:sumrootsparts}
विभाजन $\sum_{k=0}^{n-1}\omega^k = 0$ वास्तविक और काल्पनिक में
भागों से दो त्रिकोणमितीय पहचान निःशुल्क प्राप्त होती हैं:
\[
\sum_{k=0}^{n-1}\cos\frac{2k\pi}n = 0,
\qquad
\sum_{k=0}^{n-1}\sin\frac{2k\pi}n = 0
\qquad (n \geq 2).
\]
ज्यामितीय रूप से: एक नियमित का केन्द्रक $ n $-गॉन में अंकित है
इकाई वृत्त इसका केंद्र है --- शीर्ष बिल्कुल संतुलित होते हैं।
के लिए $ n = 5$ पहली पहचान देती है $1 + 2\cos\frac{2\pi}5 +
2\cos\frac{4\pi}5 = 0$ (जोड़ना $ k $ साथ $ n - k $), आरंभ
का बिंदु \cref{exo:b1:complex:8}की गणना
$\cos\frac{2\pi}5$.
\end{example}

\begin{example}[बीजगणितीय रूप में वर्गमूल]\label{ex:b1:complex:sqrt}
समाधान करना $ z^2 = 3 + 4\iu $ त्रिकोणमिति के बिना, सेट $ z = x + \iu y $:
\[
x^2 - y^2 = 3, \qquad 2xy = 4, \qquad x^2 + y^2 = \abs{3 + 4\iu} = 5 .
\]
पहला और आखिरी जोड़ना: $ x^2 = 4$, इसलिए $ x = \pm 2$, तब $ y = 2/x =
\pm 1$ साथ \emph{वही} साइन पेयरिंग ($ xy = 2 > 0$): $ z = \pm(2 +
\iu)$. सामान्य सूत्र के साथ मिलकर, यह प्रत्येक द्विघात को हल करता है
जटिल गुणांकों वाला समीकरण
(\cref{exo:b1:complex:7}).
\end{example}

\section{सम्मिश्र संख्याएँ और समतल ज्यामिति}

\begin{proposition}[ज्यामितीय शब्दकोश]\label{prop:b1:complex:geometry}
बिंदु को पहचानें $ M(x, y)$ इसके साथ विमान का \emph{प्रत्यय} $ z = x
+ \iu y $\index{प्रत्यय}. विशिष्ट बिंदुओं के लिए $ A, B, C $ प्रत्ययों का $ a, b,
c $:
\begin{enumerate}
  \item $\abs{b - a}$ दूरी है $ AB $;
  \item $\arg\dfrac{c - a}{b - a}$ सदिशों के बीच का कोण है
        $\vect{AB}$ और $\vect{AC}$ (मॉड $2\pi $);
  \item $ A, B, C $ यदि और केवल यदि संरेखित हैं $\dfrac{c - a}{b - a} \in
        \R $; पंक्तियाँ $ AB $ और $ AC $ लंबवत हैं यदि और केवल यदि
        $\dfrac{c - a}{b - a} \in \iu\R $.
\end{enumerate}
\end{proposition}

\begin{proof}
(1) की परिभाषा है \omterm{def:b1:complex:field}{मापांक} के लिए आवेदन किया $ b - a $, का प्रत्यय
$\vect{AB}$. (2):लिखो $ b - a = r\eu^{\iu\theta}$, $ c - a =
s\eu^{\iu\varphi}$; तब $\frac{c-a}{b-a} = \frac sr
\eu^{\iu(\varphi - \theta)}$ तर्क है $\varphi - \theta $, द
से कोण $\vect{AB}$ को $\vect{AC}$. (3): संरेखण का अर्थ है कोण $0$
या $\pi $, यानी\ में तर्क $\pi\Z $, यानी\ भागफल वास्तविक है;
लम्बवत् का अर्थ है कोण $\pm\frac\pi2$, यानी\ भागफल है
पूर्णतः काल्पनिक. (तब से भागफल शून्येतर है $ C \neq A $.)
\end{proof}

\begin{remark}[अंतराल: $\C $ गुणन वाला तल है]\label{rem:b1:complex:interlude}
यह इस बात पर रुकने लायक है कि इस अध्याय को आखिर क्या संभव बनाता है:
विमान $\R^2$ हर दिशा में परिवर्धन करता है, लेकिन नहीं
ईश्वर प्रदत्त गुणन --- और $\C $ \emph{है} विमान
एक से सुसज्जित, जिसमें एक निश्चित संख्या से गुणा किया जाता है
घूमता है और स्केल करता है। इस एक ढाँचे को तीन और ढाँचे से दुह लिया जाएगा
इस खंड में कई बार. में \cref{ch:b1:matrices}, से गुणा
$ a + \iu b $ के रूप में पुनः प्रकट होता है $2 \times 2$ पंक्तियों के साथ मैट्रिक्स $(a,
-b)$ और $(b, a)$: जटिल अंकगणित पहली बार है, पूरी तरह से
मैट्रिक्स उत्पादों का ठोस परिवार। में \cref{ch:b1:euclid}, द
सूत्र $\Re(\conj z\,w)$ डॉट उत्पाद बन जाता है, और
$\abs z $ यूक्लिडियन मानदंड: त्रिकोण असमानता यहाँ सिद्ध हुई
वहां कॉची-श्वार्ज़ कहानी का मॉडल है। और में
\cref{ch:b1:curves}, एक गतिशील बिंदु सबसे अच्छा लिखा जाता है $ t \mapsto
z(t)$, ताकि वेग और त्वरण जटिल-मूल्यवान हो जाएं
व्युत्पन्न --- वृत्ताकार गति, एक के लिए, बस है $ z(t) =
R\,\eu^{\iu\omega t}$. एक अच्छा गुणन, चार अध्याय
लाभांश
\end{remark}

\begin{proposition}[एमएपीएस $ z \mapsto az + b $]\label{prop:b1:complex:similitude}
होने देना $ a \in \C^*$, $ b \in \C $. रूपान्तरण $ f(z) = az + b $ का
विमान:
\begin{itemize}
  \item यदि एक अनुवाद है $ a = 1$;
  \item अन्यथा एक अद्वितीय निश्चित बिंदु है $\zeta = \frac{b}{1-a}$,
        और $ f(z) - \zeta = a\,(z - \zeta)$: $ f $ का घूर्णन है
        केंद्र $\zeta $ और कोण $\arg a $, स्केलिंग के साथ बना
        केंद्र का $\zeta $ और अनुपात $\abs a $.
\end{itemize}
विशेष रूप से $ z \mapsto \eu^{\iu\theta} z $ कोण का घूर्णन है
$\theta $ उत्पत्ति के बारे में, और $\conj z $ यथार्थ में प्रतिबिम्ब है
अक्ष.
\end{proposition}

\begin{proof}
अगर $ a = 1$, $ f(z) = z + b $ प्रत्यय के वेक्टर द्वारा अनुवाद करता है $ b $. अगर
$ a \neq 1$, निश्चित-बिंदु समीकरण $ z = az + b $ अद्वितीय है
समाधान $\zeta = \frac{b}{1-a}$, और तब $ f(z) - \zeta = az + b -
(a\zeta + b) = a(z - \zeta)$. लिखना $ a = \abs a\, \eu^{\iu\arg a}$,
से गुणा $ a $ दूरियों को मापता है $\zeta $ द्वारा $\abs a $ और जोड़ता है
$\arg a $ कोणों पर $\zeta $, जो कि घोषित रचना है।
\end{proof}

\begin{example}[मानचित्र का वर्गीकरण $ z \mapsto az+b $]\label{ex:b1:complex:classify}
लेना $ f(z) = \iu z + 1$. यहाँ $ a = \iu \neq 1$: निश्चित बिंदु है
\[
\zeta = \frac{b}{1 - a} = \frac1{1 - \iu}
= \frac{1 + \iu}{2},
\]
और तबसे $\abs a = 1$, बिल्कुल कोई स्केलिंग नहीं है: $ f $ शुद्ध है
केंद्र का घूर्णन $\frac{1+\iu}2$ और कोण $\arg \iu = \frac\pi2$.
जाँच करना: $ f(\zeta) = \iu\,\frac{1+\iu}2 + 1 = \frac{\iu - 1}2 + 1 =
\frac{1 + \iu}2 = \zeta $, और $ f(0) = 1$, $ f(1) = 1 + \iu $, $ f(1 +
\iu) = \iu\,(1 + \iu) + 1 = \iu $: चार बिंदु $0, 1, 1+\iu, \iu $
उनके केंद्र के चारों ओर इकाई वर्ग चक्र का $\zeta $, एक चौथाई
एक समय में घूमें---बिल्कुल एक रोटेशन द्वारा $\frac\pi2$ के बारे में
वर्ग का केंद्र अवश्य करना चाहिए.
\end{example}

\begin{remark}[मॉड्यूलि और तर्क के साथ सामान्य नुकसान]\label{rem:b1:complex:pitfalls}
\begin{enumerate}
  \item \emph{में कोई असमानता नहीं $\C $.} लिखना $ z \leq w $ के लिए
        अवास्तविक संख्याएँ अर्थहीन हैं; केवल मॉड्यूली, वास्तविक भाग
        और काल्पनिक भागों की तुलना की जा सकती है।
  \item \emph{$\sqrt{\phantom z}$ के लिए आरक्षित है $\R_+$.} हर
        शून्येतर सम्मिश्र संख्या है \emph{दो} वर्गमूल और
        न ही विशेषाधिकार प्राप्त है: ``चलो'' लिखें $\delta $ एक वर्ग हो
        की जड़ $\Delta $'' (के रूप में गणना की गई
        \cref{ex:b1:complex:sqrt}), कभी नहीं $\sqrt\Delta $ ---
        नियम $\sqrt{ab} = \sqrt a\sqrt b $ पहले से ही विफल है
        $ a = b = -1$.
  \item \emph{तर्क मॉड्यूलो रहते हैं $2\pi $.} से
        $\eu^{\iu\alpha} = \eu^{\iu\beta}$ निष्कर्ष $\alpha
        \equiv \beta \pmod{2\pi}$, नहीं $\alpha = \beta $;
        को भूल जाना $2k\pi $ समाधान कैसा है सेट का $ z^n = a $
        खोना $ n - 1$ उनके $ n $ तत्व.
  \item \emph{$\abs{z + w}$ क्या नहीं है $\abs z + \abs w $.} समानता
        त्रिभुज में असमानता असाधारण रूप से संरेखित है
        मामला (\cref{prop:b1:complex:rules}~(4)); सामान्य तौर पर
        \omterm{def:b1:complex:field}{मापांक} किसी राशि का अनुमान लगाया जाना चाहिए, गणना नहीं।
\end{enumerate}
\end{remark}

\begin{example}\label{ex:b1:complex:equilateral}
$ A, B, C $ (लड़ता है $ a, b, c $, जोड़ीवार अलग-अलग) एक समबाहु बनाते हैं
सीधे (वामावर्त) क्रम में शीर्षों वाला त्रिभुज यदि और
केवल यदि $\frac{c - a}{b - a} = \eu^{\iu\pi/3}$: का घूर्णन
केंद्र $ A $ और कोण $\frac\pi3$ भेजता है $ B $ को $ C $. दोनों ओरिएंटेशन
सममित समीकरण द्वारा एक साथ कैप्चर किया जाता है $ a^2 + b^2 + c^2 = ab
+ bc + ca $ (\cref{exo:b1:complex:10}).
\end{example}

\begin{remark}[जहाँ इस अध्याय का प्रयोग किया जाता है]\label{rem:b1:complex:whereused}
घातीय रूप सबसे अधिक उपयोग किया जाने वाला कम्प्यूटेशनल उपकरण है
मात्रा का. एकता की जड़ें का मानक उदाहरण बनें
चक्रीय समूह में \cref{ch:b1:structures} और गुणनखंडन को संचालित करें
का $ X^n - 1$ में \cref{ch:b1:poly}; समूहन \omterm{def:b1:complex:field}{संयुग्मित} जड़ें वहाँ
आंशिक भिन्नों द्वारा प्रयुक्त वास्तविक गुणनखंडन उत्पन्न करता है
\cref{ch:b1:fractions}. रैखिककरण (\cref{met:b1:complex:trig})
त्रिकोणमितीय शक्तियों को एकीकृत करने के लिए मानक तैयारी है
\cref{ch:b1:integration}, और के विशेषता समीकरण
\cref{ch:b1:diffeq} जटिल जड़ें हैं जिनकी वास्तविक और काल्पनिक
भाग दोलनशील विलयन उत्पन्न करते हैं $\eu^{\lambda t}\cos\omega
t $. ज्यामितीय शब्दकोश मैट्रिक्स कपड़ों में लौटता है: घूर्णन
और समानताएं ऑर्थोगोनल मैट्रिक्स बन जाती हैं
\cref{ch:b1:matrices,ch:b1:euclid}, और पैरामीट्रिज्ड वक्र
\cref{ch:b1:curves} अक्सर सर्वोत्तम रूप में लिखे जाते हैं एमएपीएस $ t \mapsto z(t)
\in \C $.
\end{remark}

\section{अभ्यास}

\begin{exercise}[$\star $]\label{exo:b1:complex:1}
घातीय रूप में रखें: $1 + \iu $; $\;\sqrt 3 - \iu $; $\;-5$;
$\;\dfrac{1 + \iu}{\sqrt 3 - \iu}$. परिणाम निकालना
$\cos\frac{5\pi}{12}$ और $\sin\frac{5\pi}{12}$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:complex:2}
गणना करना $(1 + \iu)^{20}$. जिसके लिए $ n \in \N $ है $(1 + \iu)^n $ ए
वास्तविक संख्या?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:complex:3}
ज्यामितीय रूप से वर्णन करें \omterm{def:b1:logic:sets}{तय करना} का $ z \in \C $ ऐसा है कि:
$\abs{z - 2} = \abs{z + \iu}$; $\;\abs{z - 1} = 2$;
$\;\dfrac{z - 1}{z + 1} \in \iu\R $ (के लिए $ z \neq -1$).
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:complex:4}
रैखिक बनाना $\sin^4\theta $, और विस्तार करें $\cos 4\theta $ में एक बहुपद के रूप में
$\cos\theta $.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:complex:5}
हल करना $ z^3 = 8\iu $ और समाधानों को एक चित्र पर रखें। हल करना
$ z^4 = -4$ और कारक $ X^4 + 4$ दो वास्तविक द्विघात बहुपदों में।
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:complex:6}
के लिए $\theta \in \R $ साथ $\eu^{\iu\theta} \neq 1$ और $ n \in \N $,
गणना करें
\[
C_n = \sum_{k=0}^{n} \cos k\theta
\qquad\text{and}\qquad
S_n = \sum_{k=0}^{n} \sin k\theta
\]
ज्यामितीय श्रृंखला का योग करके $\sum_k \eu^{\iu k\theta}$ और उपयोग कर रहे हैं
अर्ध-कोण गुणनखंडन।
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:complex:7}
में हल करें $\C $: $ z^2 - (3 + 4\iu) z + (-1 + 5\iu) = 0$. \emph{(गणना करें
विवेचक, इसके वर्गमूल को इस प्रकार निकालें
\cref{ex:b1:complex:sqrt}.)}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:complex:8}
होने देना $\omega = \eu^{2\iu\pi/5}$.
\begin{enumerate}
  \item औचित्य $1 + \omega + \omega^2 + \omega^3 + \omega^4 = 0$.
  \item \omterm{def:b1:logic:sets}{तय करना} $ u = \omega + \omega^4$ और $ v = \omega^2 + \omega^3$.
        गणना करना $ u + v $ और $ uv $, और उसका निष्कर्ष निकालो $ u $ और $ v $ हैं
        की जड़ें $ X^2 + X - 1$.
  \item समाप्त करें कि $\cos\frac{2\pi}{5} = \frac{\sqrt 5 - 1}{4}$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:complex:9}
इसे सभी के लिए साबित करें $ z, w \in \C $ (समानांतर चतुर्भुज पहचान):
\[
\abs{z + w}^2 + \abs{z - w}^2 = 2\abs z^2 + 2\abs w^2 ,
\]
और शीर्षों के साथ समांतर चतुर्भुज में इसकी ज्यामितीय व्याख्या करें $0,
z, w, z + w $.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b1:complex:10}
सिद्ध कीजिए कि प्रत्यय के तीन जोड़ीवार भिन्न बिंदु हैं $ a, b, c $ फॉर्म ए
समबाहु त्रिभुज (कोई भी अभिविन्यास) यदि और केवल यदि
\[
a^2 + b^2 + c^2 = ab + bc + ca .
\]
\emph{संकेत: सीधा मामला है $\frac{c-a}{b-a} = -j^2$ और
अप्रत्यक्ष मामला $\frac{c-a}{b-a} = -j $, कहाँ $ j = \eu^{2\iu\pi/3}$
संतुष्ट $ j^2 + j + 1 = 0$; या कारक $ a + jb + j^2c $ और $ a + j^2 b
+ jc $.}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b1:complex:11}
के लिए $ n \in \N^*$, मूल्यांकन करना $ P = \prod_{k=1}^{n-1}
\bigl(1 - \omega^k\bigr)$ कहाँ $\omega = \eu^{2\iu\pi/n}$.
\emph{संकेत देना: $ X^n - 1 = \prod_{k=0}^{n-1} (X - \omega^k)$; द्वारा विभाजित करें
$ X - 1$ और मूल्यांकन करें $ X = 1$.} परिणाम निकालना
$\prod_{k=1}^{n-1} \sin\frac{k\pi}{n} = \dfrac{n}{2^{n-1}}$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:complex:12}
होने देना $ n \geq 2$. प्रश्न हल करें $(z + 1)^n = (z - 1)^n $ में $\C $:
दिखाओ कि यह बिल्कुल वैसा ही है $ n - 1$ समाधान, सभी पूर्णतः काल्पनिक,
अर्थात्
\[
z_k = -\iu\,\frac{\cos\frac{k\pi}{n}}{\sin\frac{k\pi}{n}},
\qquad k = 1, \dots, n - 1 .
\]
\emph{संकेत देना: $ z = 1$ यह कोई समाधान नहीं है, इसलिए विभाजित करें और उपयोग करें की जड़ें
एकता; फिर अर्ध-कोण गुणनखंडन लागू करें
\cref{met:b1:complex:trig}.}
\end{exercise}

\section{समस्या: नेपोलियन की प्रमेय}

\begin{problem}\label{pb:b1:complex:1}
एक के प्रत्येक तरफ बाहर की ओर एक समबाहु त्रिभुज खड़ा करें
\emph{मनमाना} त्रिभुज: उन त्रिभुजों के तीन केंद्र
हमेशा एक समबाहु त्रिभुज बनाएं। यह नेपोलियन का प्रमेय है
\index{नेपोलियन का प्रमेय} --- ए \omterm{def:b1:logic:statement}{कथन} बिना किसी स्पष्ट कारण के
सत्य हो, जो बीजगणित है $ j = \eu^{2\iu\pi/3}$ तीन में सिद्ध होता है
गणना की पंक्तियाँ. यह समस्या संपूर्ण टूलकिट बनाती है
(रोटेशन, प्रत्यक्ष समानताएं,) $ j $-की कसौटी
\cref{exo:b1:complex:10}), नेपोलियन के प्रमेय को आंतरिक और सिद्ध करता है
बाहरी, ख़राब केस का पता लगाता है, और दूसरे रत्न से बंद कर देता है
उसी स्कूल का: समबाहु त्रिभुजों पर वैन शूटेन का प्रमेय
और \omterm{pb:b1:complex:1}{टॉलेमी की असमानता}। कुल मिलाकर, समतल के बिंदु हैं
उनके प्रत्ययों से पहचाना जाता है।

\medskip
\noindent\textbf{Part I --- The number $ j $ and rotations.}
\begin{enumerate}
  \item के \omterm{pb:b1:logic:1}{बीजगणितीय} रूप की गणना करें $ j $, तब $ j^2$, $ j^3$, $1 + j
        + j^2$, $\conj j $ और $ j^{-1}$. जगह $1$, $ j $, $ j^2$ एक पर
        यूनिट सर्कल का स्केच।
  \item दिखाएँ कि केंद्र का घूर्णन $ a $ और कोण $\theta $ है
        $ r(z) = a + \eu^{\iu\theta}(z - a)$, और वह
        कोणों की, दो घुमावों की संरचना $\theta $ और
        $\theta'$, कोण का घूर्णन है $\theta + \theta'$ अगर
        $\theta + \theta' \notin 2\pi\Z $, और एक अनुवाद
        अन्यथा. (उपयोग \cref{prop:b1:complex:similitude}.)
  \item होने देना $ b \neq c $. दिखाएँ कि बिल्कुल दो बिंदु हैं $ p $
        निर्माण $(b, c, p)$ समबाहु, अर्थात् $ p = b +
        \eu^{\pm\iu\pi/3}(c - b)$. एक के लिए \emph{प्रत्यक्ष} त्रिकोण
        $(a, b, c)$, उदाहरण पर जाँच करें $ a = 0$, $ b = 1$, $ c =
        \iu $ वह विकल्प $\eu^{-\iu\pi/3}$ वही लेटा हुआ है
        रेखा का सबसे दूर वाला भाग $ BC $ से $ a $ --- द
        \emph{जावक} शीर्ष.
  \item के समाधान से \cref{exo:b1:complex:10}, $(a, b, c)$
        प्रत्यक्ष समबाहु iff है $ a + jb + j^2c = 0$. दोनों को सिद्ध करो
        घूर्णन पहचान
        \[
        b + jc + j^2a = j^2\,(a + jb + j^2c),
        \qquad
        c + ja + j^2b = j\,(a + jb + j^2c),
        \]
        और यह निष्कर्ष निकालें कि मानदंड चक्रीय के अंतर्गत अपरिवर्तनीय है
        \omterm{def:b1:counting:objects}{परिवर्तन} का $(a, b, c)$.
  \item दिखाओ अगर $ a + jb + j^2c = 0$ और तीन में से दो अंक
        संपाती, तो तीनों संपाती। (तो कसौटी बिल्कुल
        विशेषताएँ: सीधा समबाहु त्रिभुज, या एकल
        बिंदु.)
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Direct similarities.}
\begin{enumerate}[resume]
  \item दिखाओ कि एमएपीएस $ f(z) = az + b $ साथ $ a \in \C^*$
        (\emph{प्रत्यक्ष समानताएँ}) रचना के अंतर्गत स्थिर हैं
        और उलटा: उनमें से दो की संरचना, और
        उनमें से किसी का व्युत्क्रम पुनः इसी रूप का होता है। (में
        की भाषा \cref{ch:b1:structures}: वे एक समूह बनाते हैं।)
  \item दिया गया $ z_1 \neq z_2$ और $ w_1 \neq w_2$, दिखाओ वहाँ एक है
        \emph{अद्वितीय} प्रत्यक्ष समानता $ f $ साथ $ f(z_1) = w_1$ और
        $ f(z_2) = w_2$, और दें $ a $ और $ b $ स्पष्ट रूप से.
  \item वह प्रत्यक्ष समानता दिखाओ $ f(z) = az + b $ गुणा करता है
        द्वारा सभी दूरियाँ $\abs a $ और सुरक्षित रखता है \emph{आकार}
        $\frac{c - a'}{b' - a'}$ किसी भी त्रिभुज का $(a', b', c')$
        \emph{(अंश और हर दोनों को गुणा किया जाता है
        $ a $)}. निष्कर्ष निकालें कि दो त्रिभुज बिल्कुल समरूप हैं
        बिल्कुल तब जब उनकी आकृतियाँ बराबर हों।
  \item पूरी तरह से प्रत्यक्ष समानता निर्धारित करें $ f(0) = 1$
        और $ f(1) = \iu $: देना $ a $, $ b $, निश्चित बिंदु, अनुपात
        और कोण.
  \item होने देना $\alpha + \beta = 1$. दिखाएँ कि ``भारित बिंदु''
        $ g(a', b') = \alpha a' + \beta b'$ प्रत्येक के साथ आवागमन करता है
        प्रत्यक्ष समानता: $ f(\alpha a' + \beta b') = \alpha f(a')
        + \beta f(b')$. यह निष्कर्ष निकालें कि केन्द्रक, मध्यबिंदु, और
        नीचे दिए गए नेपोलियन केंद्र सभी समानताओं द्वारा संचालित हैं
        --- हर ``बिना किसी नुकसान के'' के पीछे \omterm{pb:b1:logic:1}{बीजगणितीय} लाइसेंस
        व्यापकता, परिवृत्त को इकाई वृत्त पर रखें''
        तर्क.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Napoleon's theorem.} होने देना $(a, b, c)$
एक सीधा त्रिकोण हो. प्रत्येक तरफ बाहरी समबाहु को खड़ा करें
त्रिकोण (प्रश्न 3) और चलो $ n_a $, $ n_b $, $ n_c $ केंद्र बनें
(केन्द्रक) पर बने त्रिभुजों का $[b, c]$, $[c, a]$,
$[a, b]$ क्रमश।
\begin{enumerate}[resume]
  \item बताते हैं कि
        \[
        n_a = \alpha b + \beta c, \qquad
        n_b = \alpha c + \beta a, \qquad
        n_c = \alpha a + \beta b,
        \qquad\text{with}\quad
        \alpha = \frac{3 + \iu\sqrt3}{6},\ \beta = \conj\alpha .
        \]
  \item जाँच करना $\alpha + \beta = 1$, और यह निष्कर्ष निकालें कि त्रिभुज
        $(n_a, n_b, n_c)$ के पास है \emph{वही केन्द्रक} जैसा
        $(a, b, c)$.
  \item प्रश्न 4 की सर्वसमिकाओं का प्रयोग करते हुए दिखाइए
        \[
        n_a + j\,n_b + j^2 n_c
        = (\alpha j^2 + \beta j)\,(a + jb + j^2c) .
        \]
  \item गणना करना $\alpha j + \beta $ और निष्कर्ष निकालें: $\alpha j^2 +
        \beta j = j(\alpha j + \beta) = 0$, इसलिए $ n_a + j n_b + j^2
        n_c = 0$ के लिए \emph{प्रत्येक} त्रिकोण: बाहरी नेपोलियन
        त्रिभुज सीधा समबाहु (या एक बिंदु) है। नेपोलियन का
        प्रमेय सिद्ध है.
  \item समबाहु त्रिभुजों को सीधा करें \emph{आंतरिक} इसके बजाय
        (पसंद $\eu^{+\iu\pi/3}$ प्रश्न 3 में) और चलो $ m_a, m_b,
        m_c $ उनके केंद्र बनें. दिखाओ $ m_a = \beta b + \alpha c $ (और
        चक्रीय रूप से), फिर सिद्ध करें $ m_a + j^2 m_b + j\,m_c = 0$: द
        आंतरिक नेपोलियन त्रिभुज भी समबाहु है
        विपरीत दिशा.
  \item पतन का पता लगाएँ: उसे दिखाएँ $ n_a = n_b = n_c $ होता है
        बिल्कुल कब $ a + j^2 b + jc = 0$, यानी\ कब $(a, b, c)$
        एक \emph{अप्रत्यक्ष} समान भुजाओं वाला त्रिकोण। \emph{(दो
        बीच में अंक $ n_a, n_b, n_c $ संयोग करें यदि तीनों ऐसा करते हैं, द्वारा
        प्रश्न 5; फिर उपयोग करें $ n_a + j^2 n_b + j n_c = j(\alpha +
        \beta j)(a + j^2b + jc)$ और जांचें $\alpha + \beta j \neq
        0$.)}
  \item पूरी गणना त्रिभुज पर करें $ a = 0$,
        $ b = 1$, $ c = \iu $: देना $ n_a, n_b, n_c $ बिल्कुल, और
        तीन वर्ग भुजाओं की सीधी गणना द्वारा सत्यापित करें
        लंबाई यह है कि त्रिभुज समबाहु है, भुजा वर्गाकार है
        $\frac{2 + \sqrt3}{3}$.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Ptolemy and van Schooten.}
\begin{enumerate}[resume]
  \item पहचान साबित करें, सभी कॉम्प्लेक्स के लिए मान्य $ a, b, c, d $:
        \[
        (a - b)(c - d) + (a - d)(b - c) = (a - c)(b - d) .
        \]
  \item परिणाम निकालना \emph{टॉलेमी की असमानता}\index{टॉलेमी की असमानता}: किन्हीं चार बिंदुओं के लिए $ A, B, C, D $,
        \[
        AC \cdot BD \;\leq\; AB \cdot CD + AD \cdot BC ,
        \]
        समानता के साथ यदि और केवल यदि $(a-b)(c-d)$ और
        $(a-d)(b-c)$ से एक सामान्य किरण पर लेटें $0$.
  \item उसके लिए दिखाओ $\theta, \varphi \in \R $,
        $\abs{\eu^{\iu\theta} - \eu^{\iu\varphi}} =
        2\,\abs{\sin\frac{\theta - \varphi}2}$.
  \item (वैन शूटेन का प्रमेय\index{वैन शूटेन का प्रमेय}) होने देना
        $(a, b, c) = (1, j, j^2)$ --- भाग II से यह संख्या खो देता है
        सीधे समबाहु त्रिभुजों के बीच व्यापकता --- और चलो
        $ p = \eu^{\iu\theta}$ साथ $\theta \in
        \intoo{2\pi/3}{4\pi/3}$, पर परिवृत्त का एक बिंदु
        चाप $ BC $ युक्त नहीं $ A $. सिद्ध करना
        \[
        PA = PB + PC .
        \]
        \emph{(तीन दूरियों को प्रश्न 20 से व्यक्त करें और प्रयोग करें
        एक योग-से-उत्पाद सूत्र।)}
  \item पर सत्यापित करें $ p = -1$ यह बिल्कुल समानता का मामला है
        चक्रीय क्रम के लिए \omterm{pb:b1:complex:1}{टॉलेमी की असमानता} $ A, B, P, C $:
        गणना करें $(a - b)(p - c)$ और $(a - c)(b - p)$ और जांचें
        कि उनका अनुपात एक सकारात्मक वास्तविक है।
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Synthesis.}
\begin{enumerate}[resume]
  \item समबाहु के बाहरी नेपोलियन त्रिभुज की गणना करें
        त्रिकोण $(1, j, j^2)$ स्वयं, और परिणाम का वर्णन करें
        ज्यामितीय रूप से.
  \item वास्तव में समस्या का उपयोग कहां किया गया: (i) गुणन के रूप में
        घूर्णन; (ii) समानताओं की समूह संरचना और
        आकार अपरिवर्तनीय; (iii) का अर्ध-कोण गुणनखंडन
        \cref{met:b1:complex:trig}? एक-एक वाक्य.
  \item एक संक्षिप्त पैराग्राफ में, समस्या का नैतिक अर्थ स्पष्ट करें:
        समतल ज्यामिति और के बीच का शब्दकोश क्या करता है?
        का बीजगणित $\C $ खरीदें, इसकी कीमत क्या है और इनमें से कौन सा
        दो सबूत कदम --- पहचान $\alpha j + \beta = 0$ या
        शास्त्रीय चित्र --- \emph{बताते हैं} नेपोलियन का
        आपके विचार में प्रमेय बेहतर है? एक जगह का जिक्र करें जहां
        शब्दकोश बाद में मैट्रिक्स रूप में फिर से दिखाई देगा
        आयतन.
\end{enumerate}
\end{problem}
