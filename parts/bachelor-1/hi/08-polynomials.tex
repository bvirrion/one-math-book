\chapter{बहुपदों}\label{ch:b1:poly}

\omterm{def:b1:poly:def}{बहुपद}ों बीजगणितज्ञ के पसंदीदा कार्य हैं --- सिवाय इसके
उन्हें यहां कार्यों के रूप में नहीं, बल्कि औपचारिक अभिव्यक्ति के रूप में माना जाता है
एक अनिश्चित $X $, a के नियम से जोड़ा और गुणा किया जाता है
क्रमविनिमेय \omterm{def:b1:structures:ring}{अँगूठी}. सिद्धांत आश्चर्यजनक रूप से समानांतर चलता है
\cref{ch:b1:arith}: एक यूक्लिडियन डिवीजन, एक जीसीडी और बेज़आउट संबंध,
अघुलनशील तत्व और एक अद्वितीय गुणनखंडन। लगातार, $ K $
अर्थ है $\Q $, $\R $ या $\C $.

\section{अंगूठी $ K[X]$}

\begin{definition}[बहुपद, डिग्री]\label{def:b1:poly:def}
ए \emph{बहुपद}\index{बहुपद} में गुणांक के साथ $ K $ एक है
औपचारिक योग
\[
P = a_0 + a_1 X + a_2 X^2 + \dots + a_n X^n
= \sum_{k} a_k X^k,
\]
साथ $ a_k \in K $ कुछ सूचकांक से सभी शून्य। प्राकृतिक के साथ
अतिरिक्त और उत्पाद
\[
\Bigl(\sum_i a_i X^i\Bigr)\Bigl(\sum_j b_j X^j\Bigr)
= \sum_k \Bigl(\sum_{i+j=k} a_i b_j\Bigr) X^k,
\]
 \omterm{def:b1:logic:sets}{तय करना} $ K[X]$ एक क्रमविनिमेय है \omterm{def:b1:structures:ring}{अँगूठी}. \emph{डिग्री} $\deg P $ का
$ P \neq 0$ सबसे बड़ा है $ n $ साथ $ a_n \neq 0$; $ a_n $ है
\emph{नेतृत्व गुणांक} ($ P $ है \emph{राक्षसी}\index{राक्षसी} कब $ a_n = 1$), और सम्मेलन द्वारा $\deg 0 = -\infty $.
प्रत्येक बहुपद एक फलन को परिभाषित करता है $ x \mapsto P(x)$ पर $ K $ द्वारा
प्रतिस्थापन.
\end{definition}

\begin{proposition}[डिग्री नियम; अभिन्न डोमेन]\label{prop:b1:poly:degree}
के लिए $ P, Q \in K[X]$:
\[
\deg(P + Q) \leq \max(\deg P, \deg Q),
\qquad
\deg(PQ) = \deg P + \deg Q .
\]
फलस्वरूप $ K[X]$ एक \omterm{def:b1:structures:field}{अभिन्न डोमेन}, और इसकी इकाइयाँ हैं
शून्येतर स्थिरांक.
\end{proposition}

\begin{proof}
योग नियम स्पष्ट है (अधिकतम से परे गुणांक गायब हो जाते हैं)। के लिए
उत्पाद, चलो $ a_m $ और $ b_n $ अग्रणी गुणांक बनें:
का गुणांक $ X^{m+n}$ में $ PQ $ है $ a_m b_n \neq 0$ ($ K $ एक है
\omterm{def:b1:structures:field}{मैदान}, इसलिए एक \omterm{def:b1:structures:field}{अभिन्न डोमेन}), और सभी उच्च गुणांक गायब हो जाते हैं।
अगर $ P, Q \neq 0$ तब $\deg PQ = \deg P + \deg Q \geq 0$, इसलिए $ PQ \neq
0$: \omterm{def:b1:structures:field}{अभिन्न डोमेन}. अगर $ PQ = 1$ तब $\deg P + \deg Q = 0$ बल
$\deg P = \deg Q = 0$: व्युत्क्रमणीय तत्व व्युत्क्रमणीय होते हैं
स्थिरांक, यानी\ के सभी $ K^*$.
\end{proof}

\begin{theorem}[यूक्लिडियन प्रभाग]\label{thm:b1:poly:division}
होने देना $ A, B \in K[X]$ साथ $ B \neq 0$. बिल्कुल एक जोड़ी है $(Q,
R)$ का बहुआयामी पद साथ
\[
A = BQ + R, \qquad \deg R < \deg B .
\]
\end{theorem}

\begin{proof}
\emph{अस्तित्व}, मजबूत प्रेरण द्वारा $\deg A $. अगर $\deg A < \deg
B $, लेना $(Q, R) = (0, A)$. अन्यथा लिखो $ A = a X^m + \dots $, $ B =
b X^n + \dots $ साथ $ m \geq n $; \omterm{def:b1:poly:def}{बहुपद} $ A_1 = A - \frac ab
X^{m-n} B $ की डिग्री है $< m $ (प्रमुख शब्द रद्द करें), इसलिए द्वारा
प्रेरण $ A_1 = BQ_1 + R $ साथ $\deg R < \deg B $, और $ A = B(Q_1 +
\frac ab X^{m-n}) + R $.

\emph{विशिष्टता}: अगर $ BQ + R = BQ' + R'$, तब $ B(Q - Q') = R' - R $
साथ $\deg(R' - R) < \deg B $; डिग्री नियम द्वारा यह बल देता है $ Q - Q'
= 0$, तब $ R = R'$.
\end{proof}

\begin{example}\label{ex:b1:poly:division}
विभाजित करना $ A = X^4 + X^3 - 2X + 1$ द्वारा $ B = X^2 + 1$:
\[
X^4 + X^3 - 2X + 1 = (X^2 + 1)(X^2 + X - 1) + (-3X + 2).
\]
(गणना करें: घटाएं $ X^2 B $, तब $ X B $, तब $-B $; शेष
$-3X + 2$ की डिग्री है $1 < 2$.)
\end{example}

\begin{method}[हॉर्नर की योजना]\label{met:b1:poly:horner}
मूल्यांकन करना $ P = a_nX^n + \dots + a_0$ पर $ x $, या विभाजित करना $ P $ द्वारा
$ X - x $, कंप्यूटिंग शक्तियों से बचें: बचे हुए गुणांकों को पढ़ें
सही है, और पुनरावृत्त करें \emph{गुणा करके $ x $, अगला जोड़ें
गुणांक}:
\[
b_n = a_n, \qquad b_{k} = a_{k} + x\,b_{k+1}
\quad (k = n-1, \dots, 0) .
\]
तब $ b_0 = P(x)$, और पहले वाला $ b_k $ भागफल के हैं
गुणांक: $ P = (X - x)(b_nX^{n-1} + \dots + b_1) + b_0$
(विस्तार करें और तुलना करें)। उदाहरण: $ P = X^4 - 5X^3 + 6X^2 + 4X - 8$
पर $ x = 2$: द $ b $के हैं $1, -3, 0, 4, 0$, इसलिए $ P(2) = 0$ और $ P
= (X-2)(X^3 - 3X^2 + 4)$ --- लंबी के बजाय एक पंक्ति
विभाजन, और $ n $ के बजाय गुणन $\approx n^2/2$
अनुभवहीन मूल्यांकन का. योजना को उसी बिंदु पर पुनरावृत्त करना
गुणनफल निकालता है (तुलना करें)।
\cref{ex:b1:poly:multexample}).
\end{method}

\begin{remark}[{$ K[X का अंकगणित]$}]
यूक्लिडियन विभाजन के साथ, संपूर्ण अंकगणित
\cref{ch:b1:arith} को स्थानांतरित करता है $ K[X]$, समान प्रमाणों के साथ, डिग्री
निरपेक्ष मान की भूमिका निभाना: gcd (सामान्यीकृत होना)। \omterm{def:b1:poly:def}{राक्षसी}),
विस्तारित \omterm{met:b1:arith:euclid}{यूक्लिडियन एल्गोरिथ्म}, बेज़ाउट की पहचान, \omterm{thm:b1:arith:gauss}{गॉस की लेम्मा},
अपरिवर्तनीय बहुआयामी पद और अद्वितीय गुणनखंडन। हम इनका स्वतंत्र रूप से उपयोग करते हैं
स्थानांतरित परिणाम, और \cref{exo:b1:poly:6} उनमें से एक का अभ्यास करता है।
\end{remark}

\section{जड़ों}

\begin{theorem}[कारक प्रमेय]\label{thm:b1:poly:factor}
होने देना $ P \in K[X]$ और $ a \in K $. का शेष $ P $ द्वारा विभाजित करने पर
$ X - a $ स्थिरांक है $ P(a)$. विशेष रूप से
\[
P(a) = 0 \iff (X - a) \mid P .
\]
अधिक सामान्यतः, विशिष्ट जड़ें $ a_1, \dots, a_r $ का $ P $ दे दो
गुणनखंडीकरण $ P = (X - a_1)\cdots(X - a_r)\, Q $.
\end{theorem}

\begin{proof}
विभाजित करना: $ P = (X - a) Q + R $ साथ $\deg R < 1$, इसलिए $ R $ एक स्थिरांक है
$ c $; प्रतिस्थापन $ X = a $ (प्रतिस्थापन रकम और उत्पादों का सम्मान करता है)
देता है $ P(a) = c $. समतुल्यता इस प्रकार है। अनेक जड़ों के लिए,
पर शामिल करें $ r $: मामला $ r = 1$ क्या समतुल्यता अभी सिद्ध हुई है?
मान लीजिये \omterm{def:b1:logic:statement}{कथन} के लिए $ r - 1$ जड़ें और चलो $ a_1, \dots, a_r $
की अलग जड़ें हों $ P $. लिखना $ P = (X - a_1)Q_1$; प्रत्येक के लिए $ i
\geq 2$, प्रतिस्थापित करना $ a_i $:
\[
0 = P(a_i) = (a_i - a_1)\,Q_1(a_i),
\qquad a_i - a_1 \neq 0 ,
\]
और तबसे $ K $ कोई शून्य भाजक नहीं है, $ Q_1(a_i) = 0$: द $ r - 1$
विशिष्ट बिंदु $ a_2, \dots, a_r $ की जड़ें हैं $ Q_1$. द
प्रेरण परिकल्पना कारक $ Q_1 = (X - a_2)\cdots(X - a_r)\,Q $,
और वापस प्रतिस्थापित करने पर दावा मिलता है।
\end{proof}

\begin{corollary}[डिग्री का एक बहुपद $ n $ अधिकतम है $ n $ जड़ों]\label{cor:b1:poly:nroots}
एक शून्येतर $ P \in K[X]$ डिग्री का $ n $ अधिकतम है $ n $ विशिष्ट जड़ें
में $ K $. नतीजतन, ए \omterm{def:b1:poly:def}{बहुपद} (डिग्री का $\leq n $) लुप्त हो रहा है
$ n + 1$ विशिष्ट बिंदु शून्य है \omterm{def:b1:poly:def}{बहुपद}, और दो बहुआयामी पद
डिग्री का $\leq n $ पर सहमति $ n+1$ अंक बराबर हैं.
\end{corollary}

\begin{proof}
अगर $ a_1, \dots, a_r $ विशिष्ट जड़ें हैं, \cref{thm:b1:poly:factor}
देता है $ P = (X-a_1)\cdots(X-a_r) Q $, इसलिए $ n = \deg P \geq r $. दो
परिणाम अंतर्विरोध और अंतर से उत्पन्न होते हैं।
\end{proof}

\begin{example}[सहायक-बहुपद चाल]\label{ex:b1:poly:auxiliary}
होने देना $ P $ हो \omterm{def:b1:poly:def}{बहुपद} डिग्री का $\leq n $ साथ
\[
P(k) = \frac{k}{k+1} \qquad (k = 0, 1, \dots, n) ;
\]
यह अस्तित्व में है और अद्वितीय है \omterm{thm:b1:poly:lagrange}{लैग्रेंज इंटरपोलेशन} नीचे। क्या है
$ P(n+1)$? स्पष्ट भाजक: द \omterm{def:b1:poly:def}{बहुपद} $ Q = (X+1)P - X $ है
डिग्री $\leq n + 1$ और पर गायब हो जाता है $ n + 1$ अंक $0, 1,
\dots, n $, तो द्वारा \cref{thm:b1:poly:factor}
\[
Q = c\,X(X-1)(X-2)\cdots(X-n)
\]
कुछ स्थिरांक के लिए $ c $. कहां मूल्यांकन करें $ Q $ स्वतंत्र रूप से जाना जाता है:
पर $ X = -1$, $ Q(-1) = 0 \cdot P(-1) + 1 = 1$, जबकि उत्पाद
बराबर $(-1)(-2)\cdots(-1-n) = (-1)^{n+1}(n+1)!$; इस तरह $ c =
\frac{(-1)^{n+1}}{(n+1)!}$. अब मूल्यांकन करें $ X = n + 1$:
\[
(n+2)\,P(n+1) - (n+1) = Q(n+1) = c\,(n+1)! = (-1)^{n+1} ,
\]
इसलिए $ P(n+1) = \dfrac{(n+1) + (-1)^{n+1}}{n+2}$: के बराबर $1$ के लिए
अजीब $ n $, और को $\frac{n}{n+2}$ एक जैसे के लिए $ n $ --- द
प्रक्षेप \omterm{def:b1:poly:def}{बहुपद} करता है \emph{नहीं} पैटर्न जारी रखें
$\frac{n+1}{n+2}$. याद रखने की युक्ति: डेटा को एन्कोड करें
एक सहायक की जड़ें \omterm{def:b1:poly:def}{बहुपद}, अज्ञात स्थिरांक की पहचान करें
डेटा के बाहर एक बिंदु पर, और फ़सल।
\end{example}

\begin{definition}[व्युत्पन्न, बहुलता]\label{def:b1:poly:derivative}
 \emph{औपचारिक व्युत्पन्न} का $ P = \sum a_k X^k $ है $ P' = \sum_{k
\geq 1} k\,a_k X^{k-1}$; यह सामान्य नियमों को पूरा करता है $(P+Q)' = P' +
Q'$, $(PQ)' = P'Q + PQ'$ (एकपदी पर जाँच की गई और द्वारा बढ़ाया गया
रैखिकता)। जड़ $ a $ का $ P $ है
\emph{बहुलता}\index{बहुलता} $ m \geq 1$ कब
$(X-a)^m \mid P $ लेकिन $(X-a)^{m+1} \nmid P $; जड़ है \emph{सरल}
अगर $ m = 1$, \emph{एकाधिक} अगर $ m \geq 2$.
\end{definition}

\begin{proposition}[डेरिवेटिव के माध्यम से बहुलता]\label{prop:b1:poly:multiplicity}
$ a $ की जड़ है $ P $ का \omterm{def:b1:poly:derivative}{बहुलता} $\geq m $ अगर और केवल अगर
\[
P(a) = P'(a) = \dots = P^{(m-1)}(a) = 0 .
\]
विशेष रूप से, $ a $ का एक बहुमूल है $ P $ अगर और केवल अगर $ P(a) =
P'(a) = 0$.
\end{proposition}

\begin{proof}
लिखना $ P = (X - a)^m Q + R $ कहाँ $ R $ विभाजन का शेष भाग है
द्वारा $(X-a)^m $, $\deg R < m $. फर्क $ k \leq m - 1$ समय और
पर मूल्यांकन कर रहे हैं $ a $: पहला पद योगदान देता है $0$ (प्रत्येक व्युत्पन्न
एक कारक बरकरार रखता है $(X-a)$), इसलिए $ P^{(k)}(a) = R^{(k)}(a)$.

अब ए \omterm{def:b1:poly:def}{बहुपद} $ R $ डिग्री का $< m $ इसके द्वारा निर्धारित किया जाता है $ R(a), R'(a),
\dots, R^{(m-1)}(a)$: लिखना $ R = \sum_{k < m} c_k (X - a)^k $
(संभव: की शक्तियों का विस्तार करें $ X = (X - a) + a $), कोई पाता है
$ R^{(k)}(a) = k!\, c_k $. अत: सभी $ P^{(k)}(a) = 0$ के लिए $ k < m $
$\iff $ सभी $ c_k = 0$ $\iff $ $ R = 0$ $\iff $ $(X-a)^m \mid P $.
\end{proof}

\begin{example}[बहुलता की गणना]\label{ex:b1:poly:multexample}
क्या है \omterm{def:b1:poly:derivative}{बहुलता} जड़ का $2$ में $ P = X^4 - 5X^3 + 6X^2
+ 4X - 8$? क्रमिक व्युत्पन्नों का मूल्यांकन करें $2$:
\[
P(2) = 16 - 40 + 24 + 8 - 8 = 0, \qquad
P'(2) = 32 - 60 + 24 + 4 = 0,
\]
\[
P''(2) = 48 - 60 + 12 = 0, \qquad
P'''(2) = 48 - 30 = 18 \neq 0
\]
(साथ $ P' = 4X^3 - 15X^2 + 12X + 4$, $ P'' = 12X^2 - 30X + 12$,
$ P''' = 24X - 30$). तीन लुप्त मान फिर एक शून्येतर मान:
\omterm{def:b1:poly:derivative}{बहुलता} बिल्कुल $3$. बाँटना, $ P = (X - 2)^3(X + 1)$ ---
जिसे विस्तार करके जांचा जाता है $(X-2)^3 = X^3 - 6X^2 + 12X - 8$ और
से गुणा करना $ X + 1$. अंतर्दृष्टि: बहुलताओं को पढ़ा जाता है
\emph{मूल्यांकन}, किसी गुणनखंडन की आवश्यकता नहीं है --- जो बिल्कुल सही है
जब गुणनखंड पहुंच से बाहर हो तो कोई उनका पता कैसे लगाता है।
\end{example}

\begin{example}[एक जीसीडी के साथ एकाधिक जड़ों का पता लगाना]\label{ex:b1:poly:gcdmultiple}
जब कोई जड़ ज्ञात न हो, \cref{prop:b1:poly:multiplicity} अभी भी
एक देता है \emph{वैश्विक} मल्टीपल-रूट डिटेक्टर: $ a $ एक बहु है
की जड़ $ P $ यदि यह एक सामान्य जड़ है $ P $ और $ P'$, इसलिए $ P $ एक है
एकाधिक जड़ (में) $\C $) यदि $\gcd(P, P') \neq 1$ --- द्वारा गणना योग्य
द \omterm{met:b1:arith:euclid}{यूक्लिडियन एल्गोरिथ्म} बिना कुछ हल किये. नमूना: $ P =
X^3 - 3X + 2$, $ P' = 3X^2 - 3 = 3(X - 1)(X + 1)$. का परीक्षण
जड़ें $\pm1$ का $ P'$ अंदर $ P $: $ P(1) = 0$ लेकिन $ P(-1) = 4$, इसलिए
\[
\gcd(P, P') = X - 1 :
\]
जड़ $1$ एकाधिक है; दो बार विभाजित करना, $ P = (X - 1)^2(X + 2)$.
जीसीडी पूरी रिपोर्ट भी देती है \omterm{def:b1:logic:sets}{तय करना} अनेक जड़ों की, प्रत्येक के साथ
\omterm{def:b1:poly:derivative}{बहुलता} एक से कम --- तथ्य हर कंप्यूटर-बीजगणित
सिस्टम किसी से भी पहले ``वर्ग-मुक्त गुणनखंड'' का उपयोग करता है
जड़-शिकार, और \omterm{def:b1:poly:def}{बहुपद} का जुड़वां
\cref{exo:b1:poly:9}कोई-एकाधिक-रूट तर्क नहीं है।
\end{example}

\begin{theorem}[बीजगणित का मौलिक प्रमेय]\label{thm:b1:poly:fta}
प्रत्येक अचर \omterm{def:b1:poly:def}{बहुपद} का $\C[X]$ में एक जड़ है
$\C $.\index{बीजगणित का मौलिक प्रमेय}
\end{theorem}
\admitted

\begin{remark}
अपने नाम के बावजूद, प्रमेय एक है \omterm{def:b1:logic:statement}{कथन} का \emph{विश्लेषण}:
प्रत्येक ज्ञात प्रमाण की पूर्णता का उपयोग करता है $\R $ किसी न किसी रूप में, और
कोई भी पूर्णतः \omterm{pb:b1:logic:1}{बीजगणितीय} नहीं है --- ईमानदार प्रमाण इसमें दिया गया है
वर्ष 3 खंड, एक बार या तो जटिल एकीकरण या सघनता
तर्क उपलब्ध हैं. यह अध्याय वास्तव में क्या साबित करता है
द \emph{कमी}: प्रत्येक अचर के लिए एक मूल प्रदान किया गया
\omterm{def:b1:poly:def}{बहुपद}, पूर्ण गुणनखंडन समाप्त $\C $ और $\R $ नीचे
शुद्ध बीजगणित का अनुसरण करें।
\end{remark}

\begin{corollary}[गुणनखंडन ख़त्म $\C $ और खत्म होता है $\R $]\label{cor:b1:poly:factorization}
\begin{enumerate}
  \item प्रत्येक अशून्य $ P \in \C[X]$ जैसे कारक
        \[
        P = c\, (X - a_1)^{m_1} \cdots (X - a_r)^{m_r},
        \]
        साथ $ c $ अग्रणी गुणांक, $ a_i $ विशिष्ट परिसर
        जड़ें, $\sum m_i = \deg P $: \emph{के साथ गिना जाता है \omterm{def:b1:poly:derivative}{बहुलता},
        ए \omterm{def:b1:poly:def}{बहुपद} डिग्री का $ n $ बिलकुल है $ n $ जटिल जड़ें}.
  \item प्रत्येक अशून्य $ P \in \R[X]$ कारक खत्म $\R $ जैसा
        \[
        P = c \prod_i (X - a_i)^{m_i} \prod_j (X^2 + p_j X +
        q_j)^{n_j},
        \]
        द्विघात कारकों से भिन्न होना $ p_j^2 - 4q_j < 0$
        (कोई वास्तविक जड़ें नहीं)।
\end{enumerate}
\end{corollary}

\begin{proof}
(1) डिग्री पर प्रेरण, एक समय में एक जड़ को विभाजित करना
\cref{thm:b1:poly:factor}; प्रत्येक चरण पर डिग्रियों की गिनती मेल खाती है।

(2) चलो $ P $ वास्तविक गुणांक हैं। अगर $ z $ की एक जटिल जड़ है
\omterm{def:b1:poly:derivative}{बहुलता} $ m $, ऐसा ही है $\conj z $: संयुग्मन $ P(z) = 0$ देता है
$ P(\conj z) = \conj{P(z)} = 0$ (गुणांक उनके अपने हैं
संयुग्म), और यही बात व्युत्पन्नों पर भी लागू होती है
(\cref{prop:b1:poly:multiplicity}). अवास्तविक जड़ों को \omterm{def:b1:structures:group}{समूह} में रखें
संयुग्मी जोड़े: प्रत्येक जोड़ी योगदान देती है
\[
(X - z)(X - \conj z) = X^2 - 2\Re(z)\, X + \abs z^2 ,
\]
नकारात्मक विभेदक के साथ एक वास्तविक द्विघात। वास्तविक जड़ें योगदान देती हैं
रैखिक कारक.
\end{proof}

\begin{example}\label{ex:b1:poly:factorexample}
$ X^4 + 4$ पर विचार किया गया $\R $ में \cref{exo:b1:complex:5} जोड़ी बनाकर
चार जटिल जड़ें $\pm 1 \pm \iu $:
$ X^4 + 4 = (X^2 - 2X + 2)(X^2 + 2X + 2)$. न ही द्विघात विभाजन
खत्म $\R $ (विभेदक $-4$). ध्यान दें: ए \emph{अलघुकरणीय} असली
\omterm{def:b1:poly:def}{बहुपद} की डिग्री है $1$ या $2$ ---यह बिल्कुल यही है
गुणनखंडन प्रमेय कहता है। वही संयुग्म-युग्म चलता रहता है
$ X^4 + 1$, जिनकी जड़ें हैं $\eu^{\pm\iu\pi/4}$ और
$\eu^{\pm3\iu\pi/4}$: प्रत्येक जोड़ी योगदान देती है $ X^2 -
2\cos\theta\,X + 1$, इसलिए
\[
X^4 + 1 = \bigl(X^2 - \sqrt2\,X + 1\bigr)
\bigl(X^2 + \sqrt2\,X + 1\bigr) ,
\]
एक पहचान जो भोले-भाले फैक्टरिंग प्रयासों के लिए अदृश्य है $\Q $ ---
वास्तविक पर जोर देने की कीमत (यहाँ भी तर्कहीन)
गुणांक, और एकीकृत करने के लिए एक मानक इनपुट $\frac1{x^4 +
1}$ में \cref{ch:b1:integration}.
\end{example}

\begin{remark}[बहुपदों के साथ सामान्य हानियाँ]\label{rem:b1:poly:pitfalls}
\begin{enumerate}
  \item \emph{\omterm{def:b1:poly:def}{बहुपद} बनाम फ़ंक्शन।} ऊपर $ K = \Q, \R, \C $
        दोनों धारणाएँ मेल खाती हैं (समान कार्य समान होते हैं
        गुणांक, द्वारा \cref{cor:b1:poly:nroots} और
        की अनंतता $ K $), लेकिन वैचारिक रूप से ए \omterm{def:b1:poly:def}{बहुपद} है
        इसकी गुणांक सूची: दो-तत्व से अधिक \omterm{def:b1:structures:field}{मैदान}
        $\Z/2\Z $ का \cref{ch:b1:structures}, $ X^2 + X $ गायब हो जाता है
        दोनों बिंदुओं पर, फिर भी शून्य नहीं है \omterm{def:b1:poly:def}{बहुपद}.
  \item \emph{जोड़ के अंतर्गत डिग्रियाँ।} $\deg(P + Q)$ गिरा सकता है
        नीचे $\max(\deg P, \deg Q)$ जब प्रमुख शर्तें रद्द हो जाती हैं;
        ``लिखना $\deg(P + Q) = \max(\dots)$'' के लिए ही सुरक्षित है
        विशिष्ट डिग्री.
  \item \emph{जड़ें सही ढंग से गिनी गईं.} ``$ n $ जड़ें'' में
        \cref{cor:b1:poly:factorization} मतलब \emph{साथ
        \omterm{def:b1:poly:derivative}{बहुलता}, में $\C $}: $ X^2 + 1$ इसकी कोई वास्तविक जड़ें नहीं हैं, और
        $(X-1)^2$ इसकी एक अलग जड़ है लेकिन साथ में दो हैं
        \omterm{def:b1:poly:derivative}{बहुलता}. बयान तीनों को मिलाकर गिनती होती है
        झूठे सबूतों का सबसे सामान्य स्रोत।
  \item \emph{अपरिवर्तनीयता किस पर निर्भर करती है? \omterm{def:b1:structures:field}{मैदान}.} $ X^2 - 2$ है
        अपरिवर्तनीय खत्म $\Q $, विभाजित हो जाता है $\R $; $ X^2 + 1$ है
        अपरिवर्तनीय खत्म $\R $, विभाजित हो जाता है $\C $. नंगा शब्द
        ``इरेड्यूसिबल'' का गुणांक तक कोई मतलब नहीं है
        \omterm{def:b1:structures:field}{मैदान} नाम रखा गया है।
\end{enumerate}
\end{remark}

\section{गुणांक और जड़ें}

\begin{theorem}[विएटा के सूत्र]\label{thm:b1:poly:vieta}
होने देना $ P = X^n + c_{n-1} X^{n-1} + \dots + c_0$ होना \omterm{def:b1:poly:def}{राक्षसी} जड़ों के साथ
$ a_1, \dots, a_n \in \C $ (साथ \omterm{def:b1:poly:derivative}{बहुलता}). तब\index{विएटा का
सूत्र}
\[
\sum_i a_i = -c_{n-1},
\qquad
\sum_{i < j} a_i a_j = c_{n-2},
\qquad \dots, \qquad
a_1 a_2 \cdots a_n = (-1)^n c_0 ,
\]
 $ k $- जड़ों का सममित कार्य $(-1)^k c_{n-k}$.
\end{theorem}

\begin{proof}
द्वारा \cref{cor:b1:poly:factorization}, $ P = (X - a_1)\cdots(X -
a_n)$ (\omterm{def:b1:poly:def}{राक्षसी}, सभी जड़ें सूचीबद्ध हैं)। उत्पाद का विस्तार करना
वितरणात्मक रूप से प्रत्येक में चुनने के तरीके से एक शब्द उत्पन्न होता है
कारक, या तो $ X $ या मूल पद $-a_i $: जड़ों को चुनना
द्वारा अनुक्रमित कारक $ i_1 < \dots < i_k $ और $ X $ में $ n - k $
अन्य लोग योगदान देते हैं $(-a_{i_1})\cdots(-a_{i_k})\,X^{n-k}$. \omterm{def:b1:structures:group}{समूह}ीकरण
की शक्ति से $ X $:
\[
P = \sum_{k=0}^{n} (-1)^k
\Bigl(\sum_{i_1 < \dots < i_k} a_{i_1}\cdots a_{i_k}\Bigr)
X^{n-k} ,
\]
और पहचानना $ P = \sum_k c_{n-k}X^{n-k}$ (गुणांक
अद्वितीय हैं, \cref{def:b1:poly:def}) देता है $ c_{n-k} = (-1)^k
\sigma_k $, यानी\ $\sigma_k = (-1)^kc_{n-k}$, कहाँ $\sigma_k $
को दर्शाता है $ k $-वां सममित फ़ंक्शन ऊपर प्रदर्शित है। तीन
प्रदर्शित मामले हैं $ k = 1$, $ k = 2$ और $ k = n $.
\end{proof}

\begin{example}\label{ex:b1:poly:vieta}
द्विघात के लिए $ X^2 - sX + p $: जड़ों का योग $ s $, उत्पाद $ p $ ---
पहले से ही बार-बार उपयोग किया गया (\cref{exo:b1:complex:8}). एक के लिए \omterm{def:b1:poly:def}{राक्षसी} घन
$ X^3 + aX^2 + bX + c $ जड़ों के साथ $\alpha, \beta, \gamma $:
\[
\alpha + \beta + \gamma = -a,
\quad
\alpha\beta + \beta\gamma + \gamma\alpha = b,
\quad
\alpha\beta\gamma = -c ,
\]
जो सममित मात्राओं की गणना करने की अनुमति देता है $\alpha^2 + \beta^2
+ \gamma^2 = a^2 - 2b $ बिना हल किये.
\end{example}

\begin{example}[जड़ों को बिना खोजे रूपांतरित करना]\label{ex:b1:poly:transformroots}
होने देना $\alpha, \beta $ की जड़ें बनें $ X^2 - 3X + 1$. कौन \omterm{def:b1:poly:def}{राक्षसी}
द्विघात की जड़ें हैं $\alpha^2, \beta^2$? विएटा द्वारा, $\alpha +
\beta = 3$ और $\alpha\beta = 1$, इसलिए
\[
\alpha^2 + \beta^2 = (\alpha+\beta)^2 - 2\alpha\beta = 7,
\qquad
\alpha^2\beta^2 = (\alpha\beta)^2 = 1 :
\]
जवाब है $ X^2 - 7X + 1$ --- बिना गणना के प्राप्त किया गया
$\alpha = \frac{3 + \sqrt5}2$. (जाँच करना: $\alpha^2 = \frac{7 +
3\sqrt5}2$ सचमुच $\alpha^2 + \beta^2 = 7$.) वही
रणनीति पारस्परिक को संभालती है ($ X^2 - \frac ba X + \frac
ca $-प्रकार \omterm{def:b1:counting:objects}{परिवर्तन}), बदलाव, और कोई सममित डेटा: विएटा
के बारे में प्रश्नों को परिवर्तित करता है \emph{अज्ञात} बीजगणित में जड़ें
\emph{ज्ञात} गुणांक. यह लगातार सेवा करेगा जब
जड़ें eigenvalues हैं (\cref{ch:b1:det}).
\end{example}

\begin{example}[पलिंड्रोमिक समीकरण]\label{ex:b1:poly:palindrome}
हल करना $ X^4 + X^3 - 4X^2 + X + 1 = 0$. गुणांक पढ़ते हैं
दोनों दिशाओं में समान, इसलिए $0$ एक जड़ नहीं है और से विभाजित करना
$ X^2$ कोई समाधान नहीं खोता:
\[
X^2 + X - 4 + \frac1X + \frac1{X^2} = 0 .
\]
\omterm{def:b1:logic:sets}{तय करना} $ y = X + \frac1X $: तब $ X^2 + \frac1{X^2} = y^2 - 2$, और
समीकरण ध्वस्त हो जाता है
\[
y^2 + y - 6 = 0 \iff (y + 3)(y - 2) = 0 .
\]
प्रत्येक मान को इसके माध्यम से प्रकट करें $ X^2 - yX + 1 = 0$: के लिए $ y = 2$, $ X^2 -
2X + 1 = (X - 1)^2$ दोहरा मूल देता है $1$; के लिए $ y = -3$,
$ X^2 + 3X + 1 = 0$ देता है $ X = \frac{-3 \pm \sqrt5}2$. चार जड़ें
के साथ \omterm{def:b1:poly:derivative}{बहुलता} एक चतुर्थांश के लिए, जैसे
\cref{cor:b1:poly:factorization} आवश्यकता है --- हल करके प्राप्त की गई
दो द्विघात. यह युक्ति प्रत्येक को कवर करती है \emph{मुरजबंध संबंधी}
\omterm{def:b1:poly:def}{बहुपद}: इनकी जड़ें पारस्परिक जोड़े में आती हैं $\{x, 1/x\}$
(प्रतिस्थापित करें $ X $ द्वारा $1/X $ और स्पष्ट हर), और $ y = X +
\frac1X $ ऐसे युग्मों पर निश्चित रूप से मात्रा स्थिर होती है,
डिग्री आधी करना.
\end{example}

\begin{theorem}[लैग्रेंज इंटरपोलेशन]\label{thm:b1:poly:lagrange}
होने देना $ x_0, \dots, x_n $ के अलग-अलग बिंदु हों $ K $ और $ y_0, \dots, y_n
\in K $. बिलकुल एक है $ P \in K[X]$ डिग्री का $\leq n $ साथ
$ P(x_i) = y_i $ सभी के लिए $ i $, अर्थात्\index{लैग्रेंज इंटरपोलेशन}
\[
P = \sum_{i=0}^{n} y_i\, L_i,
\qquad
L_i = \prod_{j \neq i} \frac{X - x_j}{x_i - x_j} .
\]
\end{theorem}

\begin{proof}
प्रत्येक $ L_i $ की डिग्री है $ n $ और संतुष्ट करता है $ L_i(x_i) = 1$, $ L_i(x_j) =
0$ के लिए $ j \neq i $ (प्रत्येक कारक संगत पर लुप्त हो जाता है $ x_j $).
तो प्रदर्शित किया गया $ P $ की डिग्री है $\leq n $ और अंतर्विष्ट करता है. विशिष्टता:
दो प्रक्षेप बहुआयामी पद डिग्री का $\leq n $ पर सहमत हूँ $ n+1$
अंक $ x_i $, इसलिए बराबर हैं (\cref{cor:b1:poly:nroots}).
\end{proof}

\begin{remark}[अंतराल: बहुपद भी सदिश होते हैं]\label{rem:b1:poly:interlude}
दृष्टिकोण में बदलाव कि \cref{ch:b1:vspaces} बनाऊंगा
अधिकारी: बहुआयामी पद डिग्री का $\leq n $ में एक जगह बनाएं
कौन सा जोड़ और स्केलिंग बिल्कुल निर्देशांक की तरह व्यवहार करते हैं ---
ए \omterm{def:b1:poly:def}{बहुपद} \emph{है} इसकी सूची $ n + 1$ गुणांक. तीन
\omterm{def:b1:logic:statement}{कथन} इस अध्याय के गुप्त रूप से रैखिक बीजगणित हैं।
\omterm{thm:b1:poly:lagrange}{लैग्रेंज इंटरपोलेशन} (\cref{thm:b1:poly:lagrange}) ऐसा कहता है
मूल्यांकन डेटा $(P(x_0), \dots, P(x_n))$ निर्धारित करता है $ P $
विशिष्ट रूप से: मूल्यांकन पर $ n + 1$ बिंदु एक रेखीय आक्षेप है,
और $ L_i $ इसके लिए अनुकूलित आधार हैं। विस्तार $ R =
\sum c_k(X - a)^k $ के प्रमाण में
\cref{prop:b1:poly:multiplicity} का कहना है कि की शक्तियां
$(X - a)$ एक और समन्वय प्रणाली बनाएं, साथ में $ c_k =
R^{(k)}(a)/k!$ निर्देशांक के रूप में। और
\cref{cor:b1:poly:nroots} --- डिग्री बलों की तुलना में अधिक जड़ें
शून्य \omterm{def:b1:poly:def}{बहुपद} --- सभी विशिष्टता का इंजन है: यह होगा
``एक रैखिक'' बनें \omterm{def:b1:logic:map}{नक्शा} \omterm{def:b1:logic:inj}{इंजेक्शन} आयाम के एक स्थान पर $ n + 1$''
में \cref{ch:b1:findim}. जब वे अध्याय आते हैं, तो स्थान
$ K_n[X]$ उनका पसंदीदा उदाहरण होगा; यह पहुंचने लायक है
वहाँ पहले से ही इसमें धाराप्रवाह है।
\end{remark}

\begin{remark}[जहाँ इस अध्याय का प्रयोग किया जाता है]\label{rem:b1:poly:whereused}
गुणनखंडन ख़त्म $\R $ और $\C $
(\cref{cor:b1:poly:factorization}) आंशिक का इंजन है
अंशों में \cref{ch:b1:fractions}, इसलिए एक बड़े वर्ग का
में अभिन्न \cref{ch:b1:integration}. ए का विस्तार
\omterm{def:b1:poly:def}{बहुपद} की शक्तियों में $(X - a)$, के प्रमाण में मिले
\cref{prop:b1:poly:multiplicity}, की \omterm{pb:b1:logic:1}{बीजगणितीय} छाया है
टेलर के सूत्र \cref{ch:b1:taylor}. विशेषता बहुआयामी पद
विभेदक समीकरणों के लिए पहले ही प्रकट हो चुका है (\cref{ch:b1:diffeq})
और मैट्रिक्स के लिए वापस लौटें \cref{ch:b1:det}; लैग्रेंज
अंतर्वेशन का पहला अस्तित्व-और-विशिष्टता प्रमेय है
संख्यात्मक विश्लेषण, और \omterm{pb:b1:poly:1}{चेबीशेव बहुपद} का
\cref{exo:b1:poly:10} --- जिसकी इष्टतमता सप्ताहांत समस्या है
नीचे स्थापित किया गया है --- बताएं कि अनुशासन \emph{कहाँ} को
अंतर्वेशन आख़िरकार पूरा गणित $ K[X]$, से कॉपी किया गया
\cref{ch:b1:arith}, का अध्ययन खिलाता है $ K[X]$-आदर्श और भागफल
के छल्ले वर्ष 2 खंड में.
\end{remark}

\section{अभ्यास}

\begin{exercise}[$\star $]\label{exo:b1:poly:1}
यूक्लिडियन डिवीजनों को पूरा करें: $ X^5 - 1$ द्वारा $ X^2 + X + 1$; तब
$2X^4 + X^3 - X + 3$ द्वारा $ X^2 - 2$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:poly:2}
जिसके लिए $ n \in \N $ करता है $ X^2 + X + 1$ विभाजित करना $ X^{2n} + X^n + 1$?
\emph{संकेत: की जड़ें $ X^2 + X + 1$ हैं $ j $ और $ j^2$ साथ $ j =
\eu^{2\iu\pi/3}$; चर्चा करना $ n $ आधुनिक $3$.}
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:poly:3}
वास्तविक का निर्धारण करें $ a, b $ ताकि $(X-1)^2$ विभाजित $ P = X^4 + aX^3 +
bX^2 + 1$, फिर कारक $ P $ ऊपर $\R $ इन मूल्यों के लिए.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:poly:4}
फैक्टर ख़त्म $\C $ और खत्म होता है $\R $: $ X^3 - 1$; $\;X^4 + X^2 + 1$;
$\;X^6 - 1$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:poly:5}
होने देना $ P = X^3 - 6X^2 + 11X - 6$.
\begin{enumerate}
  \item तर्कसंगत जड़ें खोजें \emph{(एक तर्कसंगत जड़ $ p/q $ में
        ए की सबसे कम शर्तें \omterm{def:b1:poly:def}{राक्षसी} पूर्णांक \omterm{def:b1:poly:def}{बहुपद} एक पूर्णांक है
        अचर पद को विभाजित करना --- इसे सिद्ध करें)}, और कारक $ P $.
  \item हल किए बिना, वर्गों का योग और योग की गणना करें
        विएटा के माध्यम से जड़ों के व्युत्क्रमों की जांच करें
        गुणनखंडीकरण.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:poly:6}
गणना करना $\gcd(X^4 - 1,\; X^3 - X^2 + X - 1)$ से यूक्लिडियन
एल्गोरिदम, और इसे एक संयोजन के रूप में लिखें $ AU + BV $ दोनों में से
बहुआयामी पद.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:poly:7}
होने देना $ P \in \R[X]$ साथ $ P(x) \geq 0$ सभी के लिए $ x \in \R $. साबित करें कि
$ P $ वास्तविक के दो वर्गों का योग है बहुआयामी पद: $ P = A^2 + B^2$.
\emph{संकेत: वास्तविक गुणनखंडन में, वास्तविक जड़ें सम होती हैं
\omterm{def:b1:poly:derivative}{बहुलता}; द्विघात गुणनखंडों को इस प्रकार लिखें $(X - z)(X - \conj z)$
और उपयोग करें $\abs{\,\cdot\,}^2 = (\Re)^2 + (\Im)^2$ के उत्पाद पर
$(X - z)$'एस।}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:poly:8}
खोजें \omterm{def:b1:poly:def}{बहुपद} $ P $ डिग्री का $\leq 2$ साथ $ P(0) = 1$, $ P(1) =
3$, $ P(2) = 2$, पहले लैग्रेंज के सूत्र द्वारा, फिर हल करके
गुणांकों पर रैखिक प्रणाली. सत्यापित करें कि दोनों उत्तर सहमत हैं।
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:poly:9}
साबित करें कि $ P = X^{2n+1} - 1$ बिल्कुल एक वास्तविक जड़ है, और वह इसके लिए है
प्रत्येक $ n \geq 1$  \omterm{def:b1:poly:def}{बहुपद} $1 + X + \frac{X^2}{2!} + \dots +
\frac{X^n}{n!}$ कोई एकाधिक मूल नहीं है \emph{(तुलना करना $ P $ और
$ P'$)}.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b1:poly:10}
(\omterm{pb:b1:poly:1}{चेबीशेव बहुपद}) परिभाषित करना $ T_0 = 1$, $ T_1 = X $ और $ T_{n+1} =
2X\,T_n - T_{n-1}$.
\begin{enumerate}
  \item प्रेरण द्वारा सिद्ध करें कि $ T_n(\cos\theta) = \cos n\theta $ सभी के लिए $\theta $.
  \item निष्कर्ष निकालें $ n $ की जड़ें $ T_n $ और इसका अग्रणी गुणांक।
  \item साबित करें कि $\sup_{x \in \intcc{-1}{1}} \abs{T_n(x)} = 1$,
        पर प्राप्त किया $ n + 1$ के अंक $\intcc{-1}{1}$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b1:poly:11}
होने देना $ P \in \C[X]$ अलग-अलग जड़ों के साथ अचर रहें $ a_1, \dots,
a_r $ (\omterm{def:b1:poly:derivative}{बहुलता} $ m_1, \dots, m_r $). की पहचान सिद्ध करें
तर्कसंगत कार्य
\[
\frac{P'(X)}{P(X)} = \sum_{i=1}^{r} \frac{m_i}{X - a_i},
\]
और गॉस--लुकास प्रमेय की गणना करें: प्रत्येक मूल $ P'$ में निहित है
की जड़ों का उत्तल आवरण $ P $ \emph{(ए पर पहचान का मूल्यांकन करें
जड़ $ w $ का $ P'$ जिसका मूल नहीं है $ P $, संयुग्म लें, और
परिणाम को ऐसे पढ़ें $ w $ का भारित औसत होना $ a_i $)}.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:poly:12}
(रूट्स-ऑफ़-यूनिटी फ़िल्टर) आइए $ n \in \N^*$ और $ j = \eu^{2\iu\pi/3}$.
का मूल्यांकन $(1 + X)^n $ पर $1$, $ j $ और $ j^2$, साबित करें कि
\[
\sum_{k \geq 0} \binom{n}{3k}
= \frac{2^n + 2\cos\frac{n\pi}{3}}{3} ,
\]
और इसके लिए सूत्र की जाँच करें $ n = 3$ और $ n = 6$. \emph{संकेत देना: $1 +
j^m + j^{2m}$ के बराबर होती है $3$ अगर $3 \mid m $ और $0$ अन्यथा; और $1 +
j = \eu^{\iu\pi/3}$.}
\end{exercise}

\section{समस्या: चेबीशेव बहुपद और सबसे सपाट बहुपद}

\begin{problem}\label{pb:b1:poly:1}
इन सब में \emph{\omterm{def:b1:poly:def}{राक्षसी}} बहुआयामी पद डिग्री का $ n $, कौन सा रहता है
शून्य के सबसे करीब $\intcc{-1}1$? उत्तर --- चेबीशेव का
प्रमेय, सन्निकटन सिद्धांत का जन्म प्रमाण पत्र --- है
$2^{1-n}T_n $, कहाँ $ T_n $ चेबीशेव है \omterm{def:b1:poly:def}{बहुपद} का
\cref{exo:b1:poly:10}, और नहीं \omterm{def:b1:poly:def}{राक्षसी} प्रतिस्पर्धी इसे हरा सकता है
विचलन $2^{1-n}$.\index{चेबीशेव बहुपद} यह समस्या
परिवार का बीजगणित विकसित करता है $(T_n)$ (रचना कानून,
स्पष्ट गुणांक, दूसरे प्रकार का परिवार $ U_n $, ए
विभेदक समीकरण), चरमता प्रमेय को इसके साथ सिद्ध करता है
समानता का मामला, और अनुप्रयोग एकत्र करता है: इष्टतम प्रक्षेप
नोड्स, का सटीक मान $\cos 36^\circ $, और ए \omterm{def:b1:arith:congruence}{अनुरूपता}
$ T_p \equiv X^p \pmod p $. लगातार, $ T_0 = 1$, $ T_1 = X $,
$ T_{n+1} = 2X\,T_n - T_{n-1}$, और हम स्वतंत्र रूप से उपयोग करते हैं
$ T_n(\cos\theta) = \cos n\theta $ से \cref{exo:b1:poly:10}.

\medskip
\noindent\textbf{Part I --- The family $(T_n)$.}
\begin{enumerate}
  \item गणना करना $ T_2, T_3, T_4, T_5$ पुनरावृत्ति से. (तुलना करना
        $ T_3$ पहचान के साथ $\cos3\theta = 4\cos^3\theta -
        3\cos\theta $ का \cref{ex:b1:complex:cos3}.)
  \item प्रेरण द्वारा सिद्ध करें: $\deg T_n = n $ अग्रणी के साथ
        गुणांक $2^{n-1}$ के लिए $ n \geq 1$, और $ T_n $ के पास है
        की समता $ n $ (केवल सम या केवल विषम शक्तियाँ ही प्रकट होती हैं)।
  \item विशिष्टता सिद्धांत सिद्ध करें: $ T_n $ है \emph{केवल}
        \omterm{def:b1:poly:def}{बहुपद} संतुष्टि देने वाला $ P(\cos\theta) = \cos n\theta $ सभी के लिए $\theta $. (दो बहुआयामी पद पर सहमति $\intcc{-1}1$
        हर जगह सहमत: \cref{cor:b1:poly:nroots}.)
  \item संरचना और उत्पाद कानून निकालें:
        \[
        T_m \circ T_n = T_{mn},
        \qquad
        2\,T_m T_n = T_{m+n} + T_{\abs{m-n}} .
        \]
  \item से स्मरण करें \cref{exo:b1:poly:10} जड़ें $ x_k =
        \cos\frac{(2k+1)\pi}{2n}$ और समद्विबाहु बिंदु
        $ y_k = \cos\frac{k\pi}n $ साथ $ T_n(y_k) = (-1)^k $. लिखो
        का पूर्ण गुणनखंडन नीचे $ T_n $ ऊपर $\R $, और
        उचित ठहराइए कि $ y_k $ जिल्द बनाना: $ y_n < x_{n-1} < y_{n-1}
        < \dots < x_0 < y_0$.
  \item साबित करें कि $ T_n(\cosh t) = \cosh(nt)$ सभी के लिए $ t \in \R $
        (समान प्रेरण, का उपयोग कर
        \cref{prop:b1:functions:hyprules}), और इसके लिए निष्कर्ष निकालें $ x
        \geq 1$ बंद प्रपत्र
        \[
        T_n(x) = \frac{\bigl(x + \sqrt{x^2 - 1}\bigr)^n +
        \bigl(x - \sqrt{x^2 - 1}\bigr)^n}{2} ,
        \]
        इसलिए $ T_n(x) > 1$ के लिए $ x > 1$: बाहर $\intcc{-1}1$ 
        \omterm{def:b1:poly:def}{बहुपद} तुरंत भाग जाता है.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Coefficients, the family $ U_n $, a
differential equation.}
\begin{enumerate}[resume]
  \item डी मोइवरे के सूत्र से
        (\cref{cor:b1:complex:demoivre}), स्पष्ट सिद्ध करें
        अभिव्यक्ति
        \[
        T_n(x) = \sum_{0 \leq 2j \leq n} \binom{n}{2j}\,
        x^{\,n-2j}\,(x^2 - 1)^j ,
        \]
        और इसके लिए इसे सत्यापित करें $ n = 3$.
  \item गणना करना $ T_n(1)$, $ T_n(-1)$ और $ T_n(0)$ सभी के लिए $ n $.
  \item परिभाषित करना $ U_n $ (\emph{दूसरे प्रकार का}) द्वारा $ U_0 = 1$, $ U_1 =
        2X $, $ U_{n+1} = 2X\,U_n - U_{n-1}$. साबित करें कि
        $ U_n(\cos\theta) = \frac{\sin(n+1)\theta}{\sin\theta}$
        के लिए $\theta \notin \pi\Z $, ओर वो $ T_n' = n\,U_{n-1}$
        के लिए $ n \geq 1$.
  \item साबित करें कि $\abs{\sin n\theta} \leq n\,\abs{\sin\theta}$
        सभी के लिए $\theta $ (प्रेरण), और मार्कोव-प्रकार का अनुमान लगाएं
        बाध्य
        \[
        \abs{T_n'(x)} \leq n^2
        \quad\text{on } \intcc{-1}1,
        \qquad\text{with } T_n'(\pm1) = (\pm1)^{n-1}\,n^2 .
        \]
  \item बताते हैं कि $ y = T_n $ विभेदक समीकरण को संतुष्ट करता है
        \[
        (1 - x^2)\,y'' - x\,y' + n^2\,y = 0 ,
        \]
        पहचान को अलग करके $\sin\theta\,
        T_n'(\cos\theta) = n\sin n\theta $ इसके संबंध में
        $\theta $; के लिए सीधे सत्यापित करें $ T_2$.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Chebyshev's extremality theorem.}
होने देना $\widetilde T_n = 2^{1-n}\,T_n $ (\omterm{def:b1:poly:def}{राक्षसी} प्रश्न 2) और द्वारा
लिखो $\norm{P}_\infty = \sup_{x \in \intcc{-1}1}\abs{P(x)}$.
\begin{enumerate}[resume]
  \item औचित्य $\norm{\widetilde T_n}_\infty = 2^{1-n}$, प्राप्त किया
        पर वैकल्पिक संकेतों के साथ $ n + 1$ अंक $ y_n < \dots
        < y_0$.
  \item मान लीजिए कुछ \omterm{def:b1:poly:def}{राक्षसी} $ P $ डिग्री का $ n $ था
        $\norm P_\infty < 2^{1-n}$, और सेट करें $ D = \widetilde T_n -
        P $. दिखाओ $\deg D \leq n - 1$, ओर वो $ D(y_k)$ के पास है
        का सख्त संकेत $(-1)^k $ प्रत्येक के लिए $ k = 0, \dots, n $.
  \item उसका निष्कर्ष निकालो $ D $ कम से कम है $ n $ विशिष्ट वास्तविक जड़ें (एक
        प्रत्येक अंतराल में, मध्यवर्ती मूल्य संपत्ति द्वारा, उपयोग किया जाता है
        यहां हाई स्कूल स्तर पर और साबित हुआ
        \cref{ch:b1:continuity}), और निष्कर्ष निकालें \emph{चेबीशेव का
        प्रमेय}: प्रत्येक \omterm{def:b1:poly:def}{राक्षसी} $ P $ डिग्री का $ n $ संतुष्ट
        \[
        \norm{P}_\infty \geq 2^{1-n} .
        \]
  \item (समानता का मामला, पहला कदम) अब मान लीजिए $\norm P_\infty =
        2^{1-n}$ बिल्कुल, $ P $ \omterm{def:b1:poly:def}{राक्षसी} डिग्री का $ n $, और चलो $ D =
        \widetilde T_n - P $. बताते हैं कि $(-1)^kD(y_k) \geq 0$ सभी के लिए $ k $, और वह यदि $ D(y_k) = 0$ एक के लिए \emph{आंतरिक भाग}
        बिंदु $ y_k $ ($0 < k < n $), तब $ D'(y_k) = 0$ भी।
        \emph{(एक इंटीरियर में $ y_k $, दोनों $\widetilde T_n $ और
        $ P $ निरपेक्ष मूल्य की चरम सीमा प्राप्त करें $\norm{\cdot}
        _\infty $; एक अवकलनीय फलन का व्युत्पन्न शून्य होता है
        आंतरिक चरम पर --- हाई स्कूल स्तर पर उपयोग किया जाता है,
        में साबित हुआ \cref{ch:b1:derivative}.)}
  \item (समानता का मामला, निष्कर्ष) की जड़ें गिनें $ D $ साथ
        \omterm{def:b1:poly:derivative}{बहुलता} जाहिर करना $ D = 0$: मिनिमाइज़र है
        \emph{अद्वितीय}, $ P = \widetilde T_n $.
  \item एक मनमाने खंड में परिवहन $\intcc ab $: वह दिखाओ
        ए का न्यूनतम समर्थन-मानदंड \omterm{def:b1:poly:def}{राक्षसी} डिग्री-$ n $ \omterm{def:b1:poly:def}{बहुपद} पर
        $\intcc ab $ है $2\bigl(\frac{b-a}4\bigr)^n $, एक द्वारा प्राप्त किया गया
        चेबीशेव का पुनर्मूल्यांकन किया गया \omterm{def:b1:poly:def}{बहुपद}. \emph{(विकल्प $ x =
        \frac{a+b}2 + \frac{b-a}2\,t $ और अग्रणी को ट्रैक करें
        गुणांक.)}
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Applications.}
\begin{enumerate}[resume]
  \item मामले का निपटारा करें $ n = 3$ हाथ से: के चरम का पता लगाएं
        $\widetilde T_3 = X^3 - \frac34X $ पर $\intcc{-1}1$,
        मूल्य के साथ चार गुना समद्विबाहुता को सत्यापित करें
        $\frac14$, और निष्कर्ष निकालें कि नहीं \omterm{def:b1:poly:def}{राक्षसी} क्यूबिक बेहतर करता है.
  \item (इष्टतम इंटरपोलेशन नोड्स) के लिए $ n + 1$ नोड्स $ x_0,
        \dots, x_n \in \intcc{-1}1$, अंतर्वेशन त्रुटि है
        द्वारा शासित $\omega(X) = \prod_i (X - x_i)$ (जैसा
        \cref{ch:b1:taylor} मात्रा निर्धारित करेगा) साबित करें कि विकल्प
        न्यूनतम करना $\norm\omega_\infty $ है \omterm{def:b1:logic:sets}{तय करना} का $ n + 1$
        की जड़ें $ T_{n+1}$, साथ $\norm\omega_\infty = 2^{-n}$:
        चेबीशेव नोड्स प्रक्षेप करने के लिए सही स्थान हैं।
  \item का उपयोग करते हुए $ T_5$, साबित करें कि $ c = \cos 36^\circ $ संतुष्ट
        $16c^5 - 20c^3 + 5c + 1 = 0$, इसे कारक बनाएं \omterm{def:b1:poly:def}{बहुपद} जैसा
        $(x + 1)(4x^2 - 2x - 1)^2$, और निष्कर्ष निकालें
        \[
        \cos 36^\circ = \frac{1 + \sqrt5}4 .
        \]
        के साथ एकरूपता की जाँच करें $\cos 72^\circ =
        \frac{\sqrt5 - 1}4$ से \cref{exo:b1:complex:8}.
  \item अनुमान लगाना $ T_{10}(1.1)$ प्रश्न 6 के बंद प्रपत्र के साथ
        (दो महत्वपूर्ण अंक पर्याप्त हैं), और व्याख्या करें: a
        \omterm{def:b1:poly:def}{बहुपद} से घिरा $1$ पर $\intcc{-1}1$ पहले से ही कर सकते हैं
        से अधिक $40$ पर $ x = 1.1$. (वह $ T_n $ बढ़ता है
        \emph{सबसे तेज़} ऐसे में बहुआयामी पद दूसरा है
        परिवार की अत्यधिक संपत्ति, इस समस्या से परे।)
  \item सिद्ध करो \omterm{def:b1:arith:congruence}{अनुरूपता}: प्रत्येक विषम के लिए \omterm{def:b1:arith:prime}{मुख्य} $ p $, सब
        के गुणांक $ T_p - X^p $ से विभाज्य हैं $ p $.
        \emph{(प्रश्न 7 और का प्रयोग करें $ p \mid \binom p{2j}$ के लिए $0 <
        2j < p $, के प्रमाण से
        \cref{thm:b1:arith:fermat}.)} पर सत्यापित करें $ T_3$ और $ T_5$.
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Synthesis.}
\begin{enumerate}[resume]
  \item स्पष्ट रूप से गणना करें \omterm{def:b1:poly:def}{राक्षसी} न्यूनतम का द्विघात
        सुपर-मानदंड पर $\intcc01$ और उसका विचलन. (प्रश्न 17
        के साथ $ n = 2$.)
  \item वास्तव में समस्या का उपयोग कहाँ हुआ: (i) की कठोरता
        बहुआयामी पद (\cref{cor:b1:poly:nroots}); (ii) द
        की त्रिकोणमिति \cref{ch:b1:complex} और
        \cref{ch:b1:functions}; (iii) का अंकगणित \omterm{def:b1:counting:objects}{द्विपद
        गुणांक} से \cref{ch:b1:arith}? एक-एक वाक्य.
  \item संश्लेषण, एक संक्षिप्त अनुच्छेद में: प्रमेय यही कहता है
        सबसे सपाट \omterm{def:b1:poly:def}{राक्षसी} \omterm{def:b1:poly:def}{बहुपद} वही है
        \emph{इक्विओसिलेट्स}, और प्रमाण इष्टतमता को परिवर्तित करता है
        जड़-गणना में. इस तंत्र पर टिप्पणी करें
        प्रतिस्थापन की भूमिका $ x = \cos\theta$ एक पुल के रूप में
        बीजगणित और त्रिकोणमिति के बीच अंतर बताइए और दोनों के नाम बताइए
        वे स्थान जहां समस्या के लिए तथ्यों के विश्लेषण की आवश्यकता थी (आईवीटी,
        आंतरिक चरम) जिसे बाद के अध्याय साबित करते हैं।
\end{enumerate}
\end{problem}
