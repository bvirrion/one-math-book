\chapter{यूक्लिडियन स्पेस}\label{ch:b1:euclid}

एक जोड़ना \omterm{def:b1:euclid:def}{अंदरूनी प्रोडक्ट} एक असली के लिए सदिश स्थल ज्यामितीय खरीदता है
धारणाएँ --- लंबाई, कोण, लम्बवतता, दूरियाँ --- और एक
प्रमेय जो अध्याय से ऊपर है: प्रत्येक \omterm{def:b1:vspaces:subspace}{उपस्पेस} एक स्वीकार करता है
ऑर्थोगोनल प्रक्षेपण, ग्राम-श्मिट द्वारा गणना योग्य, साकार
सबसे कम दूरी. समतल सममिति अध्याय को \omterm{def:b1:topology:closure}{बंद} कर देती है और
वर्ष की ज्यामिति.

लगातार, $E $ एक है \emph{असली} सदिश स्थल.

\section{आंतरिक उत्पाद}

\begin{definition}\label{def:b1:euclid:def}
एक \emph{अंदरूनी प्रोडक्ट}\index{अंदरूनी प्रोडक्ट} पर $ E $ एक है \omterm{def:b1:logic:map}{नक्शा}
$\langle\cdot,\cdot\rangle \colon E \times E \to \R $ वह है
द्वि\omterm{def:b1:linmaps:def}{रेखीय}, सममित और सकारात्मक निश्चित ($\langle x, x\rangle >
0$ के लिए $ x \neq 0$). ए \omterm{def:b1:findim:def}{परिमित-आयामी} इतनी सुसज्जित जगह है एक
\emph{यूक्लिडियन स्थान}\index{यूक्लिडियन स्थान}. \emph{आदर्श} का
$ x $ है $\norm{x} = \sqrt{\langle x, x\rangle}$, और $ d(x, y) =
\norm{x - y}$.
\end{definition}

\begin{example}\label{ex:b1:euclid:examples}
पर $\R^n $: विहित उत्पाद $\langle x, y\rangle = \sum x_i
y_i $. पर $ C(\intcc{a}{b})$: $\langle f, g \rangle = \int_a^b fg $
(सकारात्मक निश्चितता है \cref{thm:b1:integration:props}~(4)). पर
$\R_n[X]$: $\langle P, Q\rangle = \int_0^1 PQ $, या $\sum_{i}
P(x_i)Q(x_i)$ ऊपर $ n+1$ विशिष्ट बिंदु.
\end{example}

\begin{example}[दो बहुपदों के बीच का कोण]
\label{ex:b1:euclid:polyangle}
एक बार एक \omterm{def:b1:euclid:def}{अंदरूनी प्रोडक्ट} चुना जाता है, \emph{कोई} दो अशून्य सदिश
एक कोण है, के माध्यम से $\cos\theta = \frac{\langle x,
y\rangle}{\norm x\,\norm y}$ (द्वारा एक वैध कोसाइन
कॉची-श्वार्ज़)। के लिए $ X $ और $ X^2$ में $\int_0^1$:
\[
\langle X, X^2\rangle = \frac14, \qquad
\norm X = \frac1{\sqrt3}, \qquad \norm{X^2} = \frac1{\sqrt5},
\qquad
\cos\theta = \frac{1/4}{1/\sqrt{15}} = \frac{\sqrt{15}}{4}
\approx 0.968 :
\]
के बारे में एक कोण $14.5$ डिग्री ---पर $\intcc{0}{1}$, द
के रेखांकन $ x $ और $ x^2$ में ``लगभग \omterm{thm:b1:matrices:conjugation}{समान}ांतर'' हैं
द्विघात-माध्य भाव, यही कारण है कि साझा को हटा रहा है
दिशा (ग्राम--श्मिट, नीचे) केवल छोटा छोड़ता है
सुधार $ X^2 - X + \frac16$.
\end{example}

\begin{theorem}[कॉची-श्वार्ज़; आदर्श गुण]\label{thm:b1:euclid:cs}
सभी के लिए $ x, y \in E $:\index{कॉची-श्वार्ज़ असमानता}
\[
\abs{\langle x, y\rangle} \leq \norm x\, \norm y ,
\]
समानता के साथ यदि $ x, y $ आनुपातिक हैं. फलस्वरूप $\norm\cdot $
त्रिभुज असमानता को संतुष्ट करता है $\norm{x + y} \leq \norm x + \norm
y $ (और $\norm{\lambda x} = \abs\lambda \norm x $, $\norm x = 0 \iff
x = 0$). इसके अतिरिक्त:
\[
\norm{x+y}^2 = \norm x^2 + 2\langle x, y\rangle + \norm y^2,
\qquad
\langle x, y \rangle = \tfrac14\bigl(\norm{x+y}^2 -
\norm{x-y}^2\bigr) .
\]
\end{theorem}

\begin{proof}
अगर $ y = 0$, सब कुछ तुच्छ है. अन्यथा द्विघात $ t
\mapsto \norm{x + ty}^2 = \norm x^2 + 2t\langle x, y\rangle + t^2
\norm y^2$ है $\geq 0$ सभी के लिए $ t $: इसका विभेदक है $\leq 0$,
जो कॉची है--श्वार्ज़; समानता का अर्थ है दोहरा मूल $ t_0$, अर्थात\
$ x + t_0 y = 0$ (निश्चितता): आनुपातिकता. त्रिकोण
असमानता: विस्तार,
\[
\norm{x + y}^2 = \norm x^2 + 2\langle x, y\rangle + \norm y^2
\leq \norm x^2 + 2\norm x\,\norm y + \norm y^2
= \bigl(\norm x + \norm y\bigr)^2 ,
\]
मध्य चरण कॉची-श्वार्ज़ है; समानता बल $\langle
x, y\rangle = \norm x\norm y $, द \emph{सकारात्मक} समानता
मामला, यानी\ गैरऋणात्मक अनुपात के साथ आनुपातिकता ---
ज्यामितीय रूप से, त्रिभुज तभी विकृत होता है जब दोनों
वेक्टर उसी तरह इंगित करते हैं। आखिरी दो
पहचान \omterm{def:b1:vspaces:sum}{प्रत्यक्ष} विस्तार हैं (दूसरा,)।
\emph{ध्रुवीकरण पहचान}, उत्पाद को मानक से पुनर्प्राप्त करता है)।
\end{proof}

\section{ओर्थोगोनालिटी}

\begin{definition}\label{def:b1:euclid:orthogonal}
$ x \perp y $ कब $\langle x, y \rangle = 0$. एक परिवार है
\emph{ओर्थोगोनल} जब इसके सदिश जोड़ीवार ओर्थोगोनल हों,
\emph{ऑर्थोनॉर्मल}\index{ऑर्थोनॉर्मल} जब इसके अलावा प्रत्येक के पास है
आदर्श $1$. \emph{ऑर्थोगोनल \omterm{def:b1:vspaces:sum}{पूरक}} एक का \omterm{def:b1:vspaces:subspace}{उपस्पेस} $ F $ है
\[
F^{\perp} = \{x \in E : \forall y \in F,\ \langle x, y\rangle =
0\},
\]
ए \omterm{def:b1:vspaces:subspace}{उपस्पेस} का $ E $.\index{ऑर्थोगोनल पूरक}
\end{definition}

\begin{proposition}\label{prop:b1:euclid:orthfree}
(पाइथागोरस\index{पाइथागोरस प्रमेय}) अगर $ x \perp y $ तब
$\norm{x+y}^2 = \norm x^2 + \norm y^2$. एक ओर्थोगोनल का परिवार
अशून्य सदिश है \omterm{def:b1:vspaces:free}{मुक्त}. एक में \omterm{def:b1:euclid:orthogonal}{ऑर्थोनॉर्मल} \omterm{def:b1:vspaces:free}{आधार} $(e_1, \dots,
e_n)$, \omterm{prop:b1:vspaces:coordinates}{COORDINATES} और उत्पाद हैं
\[
x = \sum_{i} \langle x, e_i\rangle\, e_i,
\qquad
\langle x, y \rangle = \sum_i \langle x, e_i\rangle \langle y,
e_i\rangle,
\qquad
\norm x^2 = \sum_i \langle x, e_i\rangle^2 .
\]
\end{proposition}

\begin{proof}
पाइथागोरस: विस्तार करें. उन्मुक्तता: ले लो $\langle\,\cdot\,, x_j\rangle $
एक शून्य संयोजन का: $\lambda_j \norm{x_j}^2 = 0$. \omterm{prop:b1:vspaces:coordinates}{COORDINATES}:
लिखना $ x = \sum \lambda_i e_i $ और उत्पाद को अपने साथ ले जाएं $ e_j $:
$\lambda_j = \langle x, e_j\rangle $; दो सूत्र अनुसरण करते हैं
द्विरेखीयता
\end{proof}

\begin{example}[पार्सेवल जांच के साथ ऑर्थोनॉर्मल निर्देशांक]
\label{ex:b1:euclid:parsevalcheck}
बढ़ाना $ x = (1, 2, 3)$ में \omterm{def:b1:euclid:orthogonal}{ऑर्थोनॉर्मल} \omterm{def:b1:vspaces:free}{आधार} का
\cref{exo:b1:euclid:3},
\[
e_1 = \tfrac{1}{\sqrt2}(1,1,0), \quad
e_2 = \tfrac{1}{\sqrt6}(1,-1,2), \quad
e_3 = \tfrac{1}{\sqrt3}(-1,1,1).
\]
समाधान की कोई व्यवस्था नहीं---तीन आंतरिक उत्पाद:
\[
\langle x, e_1\rangle = \frac{3}{\sqrt2},
\qquad
\langle x, e_2\rangle = \frac{1 - 2 + 6}{\sqrt6} =
\frac{5}{\sqrt6},
\qquad
\langle x, e_3\rangle = \frac{-1 + 2 + 3}{\sqrt3} =
\frac{4}{\sqrt3}.
\]
प्रस्ताव के मानक सूत्र द्वारा प्रमाणीकरण:
\[
\frac{9}{2} + \frac{25}{6} + \frac{16}{3}
= \frac{27 + 25 + 32}{6} = 14 = \norm{x}^2 = 1 + 4 + 9 .
\]
यह वर्ग-निर्देशांक का योग (एक परिमित पार्सेवल) है
पहचान) सेकंड खर्च करता है और संकेत और सामान्यीकरण पकड़ता है
लगभग निश्चितता के साथ त्रुटियाँ --- इसे एक आदत बना लें
\omterm{def:b1:euclid:orthogonal}{ऑर्थोनॉर्मल} विस्तार की गणना की जाती है; यह अनंत-आयामी है
संस्करण, फूरियर गुणांक के लिए
\cref{ex:b1:euclid:trigortho}, वर्ष का एक प्रमेय है~3
आयतन.
\end{example}

\begin{theorem}[ग्राम--श्मिट]\label{thm:b1:euclid:gramschmidt}
प्रत्येक \omterm{def:b1:euclid:def}{यूक्लिडियन स्थान} है \omterm{def:b1:euclid:orthogonal}{ऑर्थोनॉर्मल} \omterm{def:b1:vspaces:free}{आधार}. स्पष्ट रूप से, किसी से भी
\omterm{def:b1:vspaces:free}{आधार} $(v_1, \dots, v_n)$, रेसिपी\index{ग्राम--श्मिट प्रक्रिया}
\[
w_k = v_k - \sum_{i=1}^{k-1} \langle v_k, e_i\rangle\, e_i ,
\qquad
e_k = \frac{w_k}{\norm{w_k}}
\]
एक पैदा करता है \omterm{def:b1:euclid:orthogonal}{ऑर्थोनॉर्मल} \omterm{def:b1:vspaces:free}{आधार} $(e_1, \dots, e_n)$ साथ
$\operatorname{Vect}(e_1, \dots, e_k) = \operatorname{Vect}(v_1,
\dots, v_k)$ हरएक के लिए $ k $.
\end{theorem}

\begin{proof}
प्रेरण चालू $ k $. यह मानते हुए $(e_1, \dots, e_{k-1})$ \omterm{def:b1:euclid:orthogonal}{ऑर्थोनॉर्मल}
फैले $\operatorname{Vect}(v_1, \dots, v_{k-1})$: वेक्टर
$ w_k $ है ओर्थोगोनल प्रत्येक के लिए $ e_j $ ($ j < k $) निर्माण द्वारा
($\langle w_k, e_j \rangle = \langle v_k, e_j\rangle - \langle v_k,
e_j\rangle $), और $ w_k \neq 0$ तब से $ v_k \notin
\operatorname{Vect}(v_1, \dots, v_{k-1})$. सामान्यीकरण रहता है
रूढ़िवादिता; \omterm{def:b1:vspaces:span}{अवधि} \omterm{def:b1:logic:statement}{कथन} तब से कायम है $ e_k $ एक है
का संयोजन $ v_k $ और पहले $ e_i $'s, उलटा।
\end{proof}

\begin{example}[ग्राम--श्मिट बहुपद पर, पूर्णतः]
\label{ex:b1:euclid:legendre}
\omterm{def:b1:euclid:orthogonal}{ऑर्थोनॉर्मल}ाइज़ करें $(1, X, X^2)$ में $\R_2[X]$ साथ $\langle P,
Q\rangle = \int_0^1 PQ $. \emph{स्टेप 1}: $\norm{1}^2 = 1$, इसलिए
$ e_1 = 1$. \emph{चरण दो}: $ w_2 = X - \langle X, 1\rangle\,1 = X -
\frac12$, और $\norm{w_2}^2 = \int_0^1\bigl(x - \frac12\bigr)^2
\dd x = \frac1{12}$: $ e_2 = \sqrt{12}\,\bigl(X - \frac12\bigr)$.
\emph{चरण 3}: $\langle X^2, e_1\rangle = \frac13$ और
\[
\langle X^2, e_2\rangle
= \sqrt{12}\int_0^1 x^2\Bigl(x - \frac12\Bigr)\dd x
= \frac{\sqrt{12}}{12},
\qquad\text{so}\qquad
w_3 = X^2 - \frac13 - \Bigl(X - \frac12\Bigr)
= X^2 - X + \frac16 .
\]
इसके मानक की गणना की गई \cref{exo:b1:euclid:9}: $\norm{w_3}^2 =
\frac1{180}$, कहाँ से $ e_3 = \sqrt{180}\,\bigl(X^2 - X +
\frac16\bigr)$. बहुआयामी पद $1$, $ X - \frac12$, $ X^2 - X +
\frac16$ पैमाने पर, पहले हैं \emph{लेगेंद्रे बहुआयामी पद}
की \omterm{prop:b1:reals:intervals}{अंतराल} $\intcc{0}{1}$; निर्माण एक जारी है
एक समय में डिग्री, प्रत्येक नई \omterm{def:b1:poly:def}{बहुपद} ओर्थोगोनल इसके सभी को
पूर्ववर्ती। ध्यान दें कि एल्गोरिदम पहले के काम को कैसे रीसायकल करता है:
अनुमान चरण 3 पर घटाया गया बिल्कुल सर्वोत्तम एफ़िन है
का सन्निकटन $ X^2$ में पाया \cref{ex:b1:euclid:bestapprox}
---ग्राम--श्मिट \emph{है} आवर्ती ऑर्थोगोनल प्रक्षेपण.
\end{example}

\begin{theorem}[ऑर्थोगोनल प्रक्षेपण]\label{thm:b1:euclid:projection}
होने देना $ F $ एक हो \omterm{def:b1:vspaces:subspace}{उपस्पेस} की \omterm{def:b1:euclid:def}{यूक्लिडियन स्थान} $ E $. तब
\[
E = F \oplus F^{\perp},
\]
और सम्बंधित अनुमान $ p_F $ पर $ F $ (द \emph{ओर्थोगोनल
प्रक्षेपण}\index{ओर्थोगोनल
प्रक्षेपण}) किसी भी रूप में दिया गया है
\omterm{def:b1:euclid:orthogonal}{ऑर्थोनॉर्मल} \omterm{def:b1:vspaces:free}{आधार} $(e_1, \dots, e_k)$ का $ F $, द्वारा $ p_F(x) = \sum_i
\langle x, e_i\rangle e_i $. इससे दूरी का एहसास होता है $ F $:
सभी के लिए $ y \in F $,
\[
\norm{x - p_F(x)} \leq \norm{x - y},
\]
केवल समानता के साथ $ y = p_F(x)$; एक लिखता है $ d(x, F) = \norm{x -
p_F(x)}$.
\end{theorem}

\begin{proof}
एक ले लो \omterm{def:b1:euclid:orthogonal}{ऑर्थोनॉर्मल} \omterm{def:b1:vspaces:free}{आधार} $(e_i)_{i \leq k}$ का $ F $
(\cref{thm:b1:euclid:gramschmidt} अंदर $ F $) और सेट करें $\pi(x) =
\sum \langle x, e_i\rangle e_i \in F $. तब $ x - \pi(x) \perp e_j $
प्रत्येक के लिए $ j $ (ऊपर जैसा ही रद्दीकरण), इसलिए $ x - \pi(x) \in
F^\perp $: $ E = F + F^\perp $. और $ F \cap F^\perp = \{0\}$: ऐसा
वेक्टर संतुष्ट करता है $\langle x, x\rangle = 0$. तो योग सीधा है और
$\pi = p_F $.

दूरी: के लिए $ y \in F $, विघटित करना $ x - y = (x - p_F(x)) + (p_F(x) -
y)$, ओर्थोगोनल टुकड़े ($ F^\perp $ और $ F $); पाइथागोरस:
\[
\norm{x - y}^2 = \norm{x - p_F(x)}^2 + \norm{p_F(x) - y}^2
\geq \norm{x - p_F(x)}^2,
\]
समानता यदि $ y = p_F(x)$.
\end{proof}

\begin{example}[प्रक्षेपण कभी लम्बे नहीं होते]
\label{ex:b1:euclid:bessel}
विभाजन पर पाइथागोरस लगाना $ x = p_F(x) + (x - p_F(x))$:
\[
\norm{p_F(x)}^2 = \norm x^2 - \norm{x - p_F(x)}^2 \leq
\norm x^2 ,
\]
समानता के साथ यदि $ x \in F $. एक में \omterm{def:b1:euclid:orthogonal}{ऑर्थोनॉर्मल} \omterm{def:b1:vspaces:free}{आधार} $(e_1,
\dots, e_k)$ का $ F $ यह पढ़ता है $\sum_{i \leq k}\langle x,
e_i\rangle^2 \leq \norm x^2$ (ए \emph{बेसेल असमानता}):
हालाँकि बहुत सारे \omterm{def:b1:euclid:orthogonal}{ऑर्थोनॉर्मल} दिशाएँ एक माप, वर्ग
\omterm{prop:b1:vspaces:coordinates}{COORDINATES} कभी भी वर्ग लंबाई से अधिक न हो --- तुलना करें
की सटीक समानता \cref{ex:b1:euclid:parsevalcheck} जब
परिवार भरापूरा है \omterm{def:b1:vspaces:free}{आधार}. यह एक-पंक्ति असमानता ही बनाती है
वर्ष~3 खंड में सारांशित फूरियर गुणांक; यह यहाँ है
पहले से ही बताता है कि और अधिक क्यों जोड़ा जा रहा है \omterm{def:b1:vspaces:free}{आधार} कम से कम कार्य करता है
वर्ग फिट केवल अवशिष्ट को कम कर सकता है।
\end{example}

\begin{example}[सर्वोत्तम द्विघात सन्निकटन]\label{ex:b1:euclid:bestapprox}
में $ C(\intcc{0}{1})$ साथ $\langle f, g\rangle = \int_0^1 fg $, द
\omterm{def:b1:poly:def}{बहुपद} डिग्री का $\leq 1$ से नजदीकी $ f(x) = x^2$ में
संबद्ध (द्विघात माध्य) दूरी है $ p_F(f)$ कहाँ $ F = \R_1[X]$.
ग्राम--श्मिट ऑन $(1, X)$: $ e_1 = 1$, $ w_2 = X - \frac12$,
$\norm{w_2}^2 = \int_0^1 (x - \frac12)^2 = \frac{1}{12}$, $ e_2 =
\sqrt{12}\,(X - \tfrac12)$. तब
\[
p_F(f) = \langle f, e_1\rangle e_1 + \langle f, e_2\rangle e_2
= \frac13 + 12\Bigl(\int_0^1 x^2\bigl(x - \tfrac12\bigr)\dd
x\Bigr)\bigl(X - \tfrac12\bigr)
= X - \frac{1}{6},
\]
का उपयोग करते हुए $\int_0^1 x^2(x - \frac12)\dd x = \frac14 - \frac16 =
\frac{1}{12}$. की एक पंक्ति में ``न्यूनतम वर्ग'' विचार \omterm{def:b1:linmaps:def}{रेखीय}
बीजगणित
\end{example}

\begin{method}[की दूरी तक तीन मार्ग $ d(x, F)$]
\label{met:b1:euclid:distance}
\begin{enumerate}
  \item \emph{\omterm{def:b1:euclid:orthogonal}{ऑर्थोनॉर्मल} \omterm{def:b1:vspaces:free}{आधार} का $ F $}: तब $ p_F(x) = \sum_i
        \langle x, e_i\rangle e_i $ और, पाइथागोरस द्वारा,
        \[
        d(x, F)^2 = \norm{x}^2 - \norm{p_F(x)}^2
        = \norm x^2 - \sum_i \langle x, e_i\rangle^2 ,
        \]
        अक्सर कंप्यूटिंग से सस्ता $ x - p_F(x)$ स्वयं.
  \item \emph{सामान्य समीकरण}: किसी भी उत्पादक परिवार के साथ
        $ F $, हल करना $\langle x - p, v_j\rangle = 0$ के लिए
        के गुणांक $ p $ (\cref{exo:b1:euclid:5}) --- नहीं
        \omterm{def:b1:euclid:orthogonal}{ऑर्थोनॉर्मल}ाइजेशन की जरूरत है.
  \item \emph{\omterm{def:b1:vspaces:sum}{पूरक} के माध्यम से}: अगर $ F^\perp $ की तुलना में छोटा है $ F $ (उदा.\ $ F $ ए \omterm{def:b1:linmaps:forms}{हाइपरप्लेन}, $ F^\perp $ एक पंक्ति
        $\operatorname{Vect}(n)$), पर प्रोजेक्ट करें $ F^\perp $
        बजाय:
        \[
        d(x, F) = \norm{p_{F^\perp}(x)} =
        \frac{\abs{\langle x, n\rangle}}{\norm n} ,
        \]
        जो एक समतल से दूरी का शास्त्रीय सूत्र है
        (\cref{exo:b1:multivar:8} इसका उपयोग करता है)।
\end{enumerate}
रूट 3 सामान्य प्रतिवर्त का एक विशेष मामला है: हमेशा प्रोजेक्ट करें
इनमें से किसी एक पर $ F $, $ F^\perp $ छोटा आयाम है.
\end{method}

\begin{example}[त्रिकोणमितीय ऑर्थोगोनैलिटी: एक फूरियर पूर्वावलोकन]
\label{ex:b1:euclid:trigortho}
पर $ C(\intcc{0}{2\pi})$ साथ $\langle f, g\rangle =
\frac1\pi\int_0^{2\pi} fg $, परिवार
\[
\Bigl(\frac{1}{\sqrt2},\ \cos x,\ \sin x,\ \cos 2x,\ \sin 2x,\
\dots\Bigr)
\]
है \omterm{def:b1:euclid:orthogonal}{ऑर्थोनॉर्मल}: उदाहरण के लिए $\langle \cos px, \cos qx\rangle =
\frac1\pi\int_0^{2\pi}\cos px\cos qx\,\dd x = 0$ के लिए $ p \neq q $
(उत्पाद को रैखिक बनाएं $\frac12[\cos(p{-}q)x +
\cos(p{+}q)x]$ और पूर्ण \omterm{def:b1:vspaces:span}{अवधि} में एकीकृत करें), जबकि
$\frac1\pi\int_0^{2\pi}\cos^2 px\,\dd x = 1$. \omterm{thm:b1:euclid:projection}{ओर्थोगोनल
प्रक्षेपण} पर \omterm{def:b1:vspaces:span}{अवधि} पहले का $2N + 1$ इन कार्यों का
इसलिए है \omterm{prop:b1:vspaces:coordinates}{COORDINATES} $\langle f, e_i\rangle $ --- \omterm{thm:b1:integration:def}{अभिन्न}
कोसाइन और साइन के विरुद्ध। ये हैं \emph{फूरियर
गुणांक} का $ f $, और यह अनुमान यह सर्वोत्तम है
द्विघात-माध्य त्रिकोणमितीय सन्निकटन; वर्ष~3 खंड
उनके अभिसरण का अध्ययन करता है। ऑर्थोगोनैलिटी कर रही है सारा काम:
गुणांकों के सूत्र हैं
\cref{thm:b1:euclid:projection} शब्दशः.
\end{example}

\section{विमान की आइसोमेट्री}

\begin{definition}\label{def:b1:euclid:isometry}
एक एंडोमोर्फिज्म $ u $ एक का \omterm{def:b1:euclid:def}{यूक्लिडियन स्थान} एक
\emph{आइसोमेट्री}\index{आइसोमेट्री} (या ओर्थोगोनल \omterm{def:b1:logic:map}{नक्शा}) जब यह
आदर्श को बरकरार रखता है: $\norm{u(x)} = \norm x $ सभी के लिए $ x $ ---
समतुल्य रूप से (ध्रुवीकरण) यह संरक्षित करता है \omterm{def:b1:euclid:def}{अंदरूनी प्रोडक्ट};
समकक्ष रूप से इसका मैट्रिक्स $ A $ एक में \omterm{def:b1:euclid:orthogonal}{ऑर्थोनॉर्मल} \omterm{def:b1:vspaces:free}{आधार} संतुष्ट
$ A^{\mathsf T} A = I $. आइसोमेट्रीज़ फॉर्म ए \omterm{def:b1:structures:group}{समूह}, द ओर्थोगोनल \omterm{def:b1:structures:group}{समूह}
$ O(E)$.
\end{definition}

\begin{example}[दृष्टि में आइसोमेट्री को पहचानना]
\label{ex:b1:euclid:recognize}
है $ A = \dfrac15\begin{pmatrix} 3 & -4\\ 4 & 3\end{pmatrix}$
ओर्थोगोनल? कॉलम: मानदंड $\frac15\sqrt{9 + 16} = 1$ और
$\frac15\sqrt{16 + 9} = 1$; उत्पाद $\frac1{25}(3\cdot(-4) +
4\cdot3) = 0$. हां और $\det A = \frac{9 + 16}{25} = 1$, तो
यह घूर्णन है $ R_\theta $ साथ $\cos\theta = \frac35$,
$\sin\theta = \frac45$ (''$3$-$4$-$5$ रोटेशन'', किसका
कोण का कोई उल्लेखनीय अंश नहीं है $\pi $). इसके विपरीत $ B =
\frac{1}{\sqrt2}\begin{pmatrix} 1 & 1\\ 0 & 1\end{pmatrix}$ है
इकाई निर्धारक-स्केल्ड लुक लेकिन गैर-इकाई पहला कॉलम
($\frac1{\sqrt2}$): नहीं ओर्थोगोनल --- निर्धारक $\pm1$ अकेले
कुछ भी प्रमाणित नहीं करता, कॉलमों की जाँच होनी चाहिए।
\end{example}

\begin{theorem}[समतल सममिति]\label{thm:b1:euclid:plane}
एक में \omterm{def:b1:euclid:orthogonal}{ऑर्थोनॉर्मल} \omterm{def:b1:vspaces:free}{आधार} एक यूक्लिडियन विमान की, मैट्रिक्स की
\omterm{def:b1:euclid:isometry}{आइसोमेट्री} बिल्कुल सही हैं
\[
R_\theta = \begin{pmatrix}
\cos\theta & -\sin\theta\\ \sin\theta & \cos\theta
\end{pmatrix}
\quad (\text{rotation of angle } \theta,\ \det = 1),
\]
\[
S_\theta = \begin{pmatrix}
\cos\theta & \sin\theta\\ \sin\theta & -\cos\theta
\end{pmatrix}
\quad (\det = -1),
\]
उत्तरार्द्ध कोण बनाने वाली रेखा में प्रतिबिंब है
$\frac\theta2$ पहले के साथ \omterm{def:b1:vspaces:free}{आधार} वेक्टर.
\end{theorem}

\begin{proof}
होने देना $ A = \begin{pmatrix} a & c\\ b & d\end{pmatrix}$ साथ
$ A^{\mathsf T}A = I $: कॉलम इकाई हैं और ओर्थोगोनल. पहला
कॉलम है $(\cos\theta, \sin\theta)$ कुछ के लिए $\theta $; दूसरा,
इकाई और ओर्थोगोनल इसे करने के लिए, है $\pm(-\sin\theta, \cos\theta)$. संकेत $+$ देता है $ R_\theta $; संकेत $-$ देता है $ S_\theta $. एक जाँच करता है
$ S_\theta^2 = I $ और वह वेक्टर $(\cos\frac\theta2,
\sin\frac\theta2)$ तय हो गया है जबकि यह ओर्थोगोनल उलटा है: ए
प्रतिबिंब. (और $ R_\alpha R_\beta = R_{\alpha+\beta}$: द
घूर्णन \omterm{def:b1:structures:group}{समूह} कोण है \omterm{def:b1:structures:group}{समूह} --- तुलना करना
\cref{thm:b1:complex:funceq}.)
\end{proof}

\begin{remark}[सामान्य ख़तरे]
\label{rem:b1:euclid:pitfalls}
\emph{ अनुमान सूत्र की आवश्यकता है \omterm{def:b1:euclid:orthogonal}{ऑर्थोनॉर्मल} \omterm{def:b1:vspaces:free}{आधार}}: एक के लिए
केवल परिवार पैदा करना $(v_i)$ का $ F $, योग $\sum_i\langle
x, v_i\rangle v_i $ है \emph{नहीं} $ p_F(x)$ (परीक्षा $ F = \R^2$, $ v_1
= e_1$, $ v_2 = e_1 + e_2$); के साथ गैर-असामान्य परिवार, समाधान करो
इसके बजाय सामान्य समीकरण
(\cref{met:b1:euclid:distance}~(2)).
\emph{ओर्थोगोनल परिवारों को बचना चाहिए $0$ होना \omterm{def:b1:vspaces:free}{मुक्त}}: शून्य
वेक्टर है ओर्थोगोनल हर चीज़ के लिए, स्वयं सहित ---
में उन्मुक्तता \cref{prop:b1:euclid:orthfree} शून्येतर की आवश्यकता है
वेक्टर
\emph{$ F^\perp $ पर निर्भर होता है \omterm{def:b1:euclid:def}{अंदरूनी प्रोडक्ट}}: में $\R_1[X]$, द
का \omterm{def:b1:vspaces:sum}{पूरक} $\operatorname{Vect}(X)$ के लिए $\int_0^1 PQ $ नहीं है
स्थिरांक लेकिन $\operatorname{Vect}(1 - \frac32 X)$ ---
गणना करें $\int_0^1 x(1 - \frac32 x) = \frac12 - \frac12 = 0$;
``लंबवत'' तब तक अर्थहीन है जब तक उत्पाद का नाम न दिया जाए।
\emph{विस्तार मत करो $\norm{x + y}$ रैखिक}: सही
पहचान है $\norm{x+y}^2 = \norm x^2 + 2\langle x, y\rangle +
\norm y^2$; क्रॉस टर्म केवल ऑर्थोगोनैलिटी के तहत गायब हो जाता है
(पाइथागोरस), और त्रिभुज असमानता एक है
\emph{असमानता}.
\emph{यूनिट वैक्टर को यूनिट वैक्टर में भेजना पर्याप्त नहीं है}: $ u(x,
y) = (x + y,\ 0)$ एमएपीएस दोनों विहित \omterm{def:b1:vspaces:free}{आधार} इकाई के लिए सदिश
वेक्टर $(1, 0)$, अभी तक $\norm{u(1,1)} = 2 \neq \sqrt2$: नहीं
\omterm{def:b1:euclid:isometry}{आइसोमेट्री}. परिभाषा की मांग है $\norm{u(x)} = \norm x $ के लिए
\emph{सभी} $ x $; मैट्रिक्स शब्दों में $ A^{\mathsf T}A = I $, अर्थात\
कॉलम जो इकाई हैं \emph{और जोड़ीदार ओर्थोगोनल} --- दोनों
स्थितियाँ, एक साथ जाँच की गईं।
\end{remark}

\begin{remark}[आंतरिक उत्पाद कहां जाता है]
\label{rem:b1:euclid:whereused}
ऑर्थोगोनल प्रक्षेपण अध्याय का सबसे अधिक प्रयोग किया जाने वाला प्रमेय है:
यह न्यूनतम वर्ग (सप्ताहांत समस्या) को रेखांकित करता है
\cref{ch:b1:multivar} उस पर प्रतिगमन रेखाएँ बनाता है), फूरियर
गुणांक (\cref{ex:b1:euclid:trigortho}), और सामान्य
के समीकरण \cref{exo:b1:euclid:5}, जो संख्यात्मक विश्लेषण
पैमाने पर हल करता है. वर्गीकरण $ R_\theta / S_\theta $ है
नीचे पूरा किया गया: सप्ताहांत समस्या वर्गीकृत होती है \emph{सभी}
विमान के दूरी-संरक्षण \omterm{def:b1:counting:objects}{परिवर्तन}, \omterm{def:b1:linmaps:def}{रेखीय} या नहीं,
और उनकी परिमितता \omterm{def:b1:structures:group}{समूह} --- रोसेट्स का गणित और
नियमित बहुभुज. वर्ष~2 खंड में \omterm{def:b1:euclid:def}{अंदरूनी प्रोडक्ट} मिलते हैं
eigenvalue सिद्धांत (सममित मैट्रिक्स, द्विघात रूप); में
वर्ष~3, अनंत-आयामी यूक्लिडियन ज्यामिति हिल्बर्ट बन जाती है
अंतरिक्ष सिद्धांत.
\end{remark}

\begin{remark}[पुस्तक 3 के अंदर परिप्रेक्ष्य]
\label{rem:b1:euclid:perspectives}
दो पुल इस अध्याय को छोड़ते हैं। \emph{पिछड़ा}, तक \omterm{def:b1:linmaps:def}{रेखीय}
बीजगणित: ग्राम मैट्रिक्स \cref{exo:b1:euclid:11} संकुल
आंतरिक उत्पाद की निर्धारक मशीनरी में
\cref{ch:b1:det}, और ऑर्थोगोनल प्रक्षेपण विशेष है
का प्रोजेक्टर \cref{ch:b1:linmaps} जिसका कर्नेल है
$ F^\perp $ --- इसके सभी बीजगणित ($ p^2 = p $, $ s = 2p -
\mathrm{id}$) शब्दशः लागू होता है, अब बोनस के साथ $\norm{x
- p(x)}$ एक है \emph{दूरी}. \emph{आगे}, विश्लेषण करने के लिए:
\cref{ch:b1:curves} इस अध्याय के साथ चाप की लंबाई मापें
किसी भी चीज़ को उसके \omterm{def:b1:euclid:isometry}{आइसोमेट्री} के बिना मानक और वर्गीकृत नहीं करता है;
\cref{ch:b1:multivar} ग्रेडिएंट को पढ़ता है
कॉची--श्वार्ज़ (सबसे तीव्र चढ़ाई) और वॉल्यूम को \omterm{def:b1:topology:closure}{बंद} कर देता है
न्यूनतम वर्ग, जो है \cref{thm:b1:euclid:projection} लागू किया गया
एक डेटा वेक्टर में $\R^n $. \omterm{def:b1:euclid:def}{अंदरूनी प्रोडक्ट} मुद्दा यह है
जहां पुस्तक का बीजगणित और उसका विश्लेषण अंततः मिलते हैं।
\end{remark}

\section{समस्या: समतल सममिति और लियोनार्डो का प्रमेय}

\begin{problem}\label{pb:b1:euclid:1}
एक \emph{\omterm{def:b1:euclid:isometry}{आइसोमेट्री} विमान का} कोई है \omterm{def:b1:logic:map}{नक्शा} $ f \colon \R^2 \to
\R^2$ दूरियाँ बनाए रखना: $\norm{f(x) - f(y)} = \norm{x - y}$
सभी के लिए $ x, y $ --- कोई रैखिकता नहीं मानी गई। यह समस्या सिद्ध होती है
वह ऐसा एमएपीएस बिल्कुल अनुवाद, घुमाव हैं,
प्रतिबिंब और \omterm{pb:b1:euclid:1}{सरकना प्रतिबिंब} (विमान का वर्गीकरण
आइसोमेट्रीज़), उनकी रचनाओं की गणना करता है, और सभी को निर्धारित करता है
उनके \emph{परिमित} \omterm{def:b1:structures:group}{समूह}: लियोनार्डो का प्रमेय, गणित
रोसेट पैटर्न के पीछे. भाग II से हम पहचानते हैं $\R^2$ साथ
$\C $ (\cref{ch:b1:complex}): विहित \omterm{def:b1:euclid:def}{अंदरूनी प्रोडक्ट} है
$\langle z, w\rangle = \operatorname{Re}(z\conj w)$ और आदर्श
है \omterm{def:b1:complex:field}{मापांक}.\index{आइसोमेट्री}\index{सरकना प्रतिबिंब}
\index{लियोनार्डो का प्रमेय}\index{डायहेड्रल समूह}

\medskip
\noindent\textbf{Part I --- Every \omterm{def:b1:euclid:isometry}{आइसोमेट्री} is affine.}
\begin{enumerate}
  \item उस अनुवाद की जाँच करें $ t_a(x) = x + a $, \omterm{def:b1:linmaps:def}{रेखीय}
        आइसोमेट्रीज़, और उनकी सभी रचनाएँ आइसोमेट्रीज़ हैं,
        और यह कि सममिति एक बनाती है \omterm{def:b1:structures:group}{समूह} रचना के अंतर्गत.
  \item होने देना $ f $ एक हो \omterm{def:b1:euclid:isometry}{आइसोमेट्री} साथ $ f(0) = 0$. बताते हैं कि $ f $
        मानदंडों को संरक्षित करता है, फिर --- ध्रुवीकरण द्वारा,
        \cref{thm:b1:euclid:cs} --- वह $\langle f(x),
        f(y)\rangle = \langle x, y\rangle $ सभी के लिए $ x, y $.
  \item अभी भी साथ है $ f(0) = 0$: बढ़ाना $\norm{f(x + y) - f(x) -
        f(y)}^2$ और $\norm{f(\lambda x) - \lambda f(x)}^2$
        प्रश्न 2 का उपयोग करके निष्कर्ष निकालिए $ f $ है
        \emph{\omterm{def:b1:linmaps:def}{रेखीय}}: $ f \in O(\R^2)$.
  \item यह निष्कर्ष निकालें कि प्रत्येक \omterm{def:b1:euclid:isometry}{आइसोमेट्री} $ f $ लिखते हैं \emph{विशिष्ट} जैसा
        $ f = t_a \circ g $ साथ $ a = f(0)$ और $ g $ ए \omterm{def:b1:linmaps:def}{रेखीय}
        \omterm{def:b1:euclid:isometry}{आइसोमेट्री} (द \emph{\omterm{def:b1:linmaps:def}{रेखीय} भाग} का $ f $).
  \item पुकारना $ f $ \emph{\omterm{def:b1:vspaces:sum}{प्रत्यक्ष}} अगर $\det g = 1$, \emph{अप्रत्यक्ष}
        अगर $\det g = -1$. दिखाओ कि \omterm{def:b1:linmaps:def}{रेखीय} ए का हिस्सा
        रचना की रचना है \omterm{def:b1:linmaps:def}{रेखीय} भागों, और
        परिणामी चिह्न नियम बताएं
        (\omterm{def:b1:vspaces:sum}{प्रत्यक्ष}/अप्रत्यक्ष जैसे लिखें $+1/-1$).
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The four types.} के जरिए
\cref{thm:b1:euclid:plane}, द \omterm{def:b1:linmaps:def}{रेखीय} की सममिति $\C $ हैं $ z
\mapsto az $ और $ z \mapsto a\conj z $ साथ $\abs a = 1$; तो हर
विमान \omterm{def:b1:euclid:isometry}{आइसोमेट्री} है
\[
f(z) = a z + b \quad (\text{direct}) \qquad\text{or}\qquad
f(z) = a\conj z + b \quad (\text{indirect}), \qquad \abs a = 1 .
\]
\begin{enumerate}[resume]
  \item शब्दकोश सत्यापित करें: $ R_\theta $ है $ z \mapsto
        \eu^{\iu\theta}z $ और $ S_\theta $ है $ z \mapsto
        \eu^{\iu\theta}\conj z $ (दोनों को जांचें $1$ और $\iu $).
  \item (\omterm{def:b1:vspaces:sum}{प्रत्यक्ष} मामला) चलो $ f(z) = az + b $, $\abs a = 1$. दिखाएँ:
        यदि $ a = 1$, $ f $ एक अनुवाद है; अगर $ a \neq 1$, $ f $ है
        अद्वितीय निश्चित बिंदु $ z_0 = b/(1 - a)$ और $ f(z) - z_0
        = a(z - z_0)$: ए \emph{ROTATION} केंद्र का $ z_0$ और
        कोण $\arg a $.
  \item (अप्रत्यक्ष मामला) चलो $ f(z) = a\conj z + b $, और $ v =
        a\conj b + b $. दिखाओ $ f \circ f = t_v $ और $ f \circ t_v =
        t_v \circ f $. अगर $ v = 0$:दिखाएँ कि इसका मध्यबिंदु $ z $
        और $ f(z)$ एक निश्चित बिंदु है, और वह $ f $ एक है
        \emph{प्रतिबिंब} एक पंक्ति में. अगर $ v \neq 0$: बताते हैं कि $ r
        = t_{-v/2}\circ f $ एक प्रतिबिम्ब है जिसका अक्ष है
        के \omterm{thm:b1:matrices:conjugation}{समान}ांतर $ v $, ताकि $ f = t_{v/2} \circ r $ एक है
        \emph{\omterm{pb:b1:euclid:1}{सरकना प्रतिबिंब}}. निष्कर्ष: प्रत्येक विमान \omterm{def:b1:euclid:isometry}{आइसोमेट्री}
        एक अनुवाद, एक घूर्णन, एक प्रतिबिंब या एक है \omterm{pb:b1:euclid:1}{सरकना
        प्रतिबिंब} (प्लेन \omterm{def:b1:euclid:isometry}{आइसोमेट्री} का वर्गीकरण)।
  \item (रचनाएँ) दिखाएँ: के घूर्णन की संरचना
        कोण $\alpha $ और $\beta $ कोण का घूर्णन है
        $\alpha + \beta $ (एक अनुवाद यदि $\alpha + \beta \in
        2\pi\Z $); दो प्रतिबिंबों की संरचना है a
        अक्षों के बीच के कोण का दोगुना कोण का घूमना (a
        यदि अक्ष \omterm{thm:b1:matrices:conjugation}{समान}ांतर हैं तो अनुवाद)।
  \item यह निष्कर्ष निकालें कि हर विमान \omterm{def:b1:euclid:isometry}{आइसोमेट्री} पर की एक रचना है
        सर्वाधिक तीन प्रतिबिम्ब.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Three identifications.}
\begin{enumerate}[resume]
  \item पूर्णतः वर्गीकृत करें $ f(z) = \iu\conj z + 1 + \iu $:
        प्रकार, अक्ष, ग्लाइड वेक्टर।
  \item होने देना $ f $ कोण का घूर्णन हो $\frac\pi2$ के बारे में $0$
        और $ g $ कोण का घूमना $\frac\pi2$ के बारे में $1$.
        गणना करना $ g \circ f $ प्रपत्र में $ z \mapsto az + b $ और
        इसे पहचानें (प्रकार, केंद्र, कोण)।
  \item होने देना $ r_1(z) = \conj z $ (वास्तविक अक्ष में प्रतिबिंब) और
        $ r_2(z) = \iu\conj z $ (पंक्ति में प्रतिबिंब $ y = x $).
        गणना करना $ r_2 \circ r_1$ और इस पर प्रश्न 9 की जाँच करें
        उदाहरण.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Finite \omterm{def:b1:structures:group}{समूह}: Leonardo's theorem.}
होने देना $ G $ एक हो \emph{परिमित} \omterm{def:b1:structures:group}{समूह} समतल सममिति का.
\begin{enumerate}[resume]
  \item दिखाएँ कि \omterm{def:b1:euclid:isometry}{आइसोमेट्री} बैरीसेंटर को संरक्षित करती है: यदि
        $\lambda_1 + \dots + \lambda_m = 1$ और $ f = t_a \circ
        g $ ($ g $ \omterm{def:b1:linmaps:def}{रेखीय}), तब $ f\bigl(\sum_i \lambda_i
        x_i\bigr) = \sum_i \lambda_i f(x_i)$.
  \item \omterm{def:b1:logic:sets}{तय करना} $ c = \frac{1}{\abs G}\sum_{g \in G} g(x_0)$ किसी के लिए
        चुना गया $ x_0$. वह हर दिखाओ $ h \in G $ ठीक करता है $ c $: ए
        परिमित \omterm{def:b1:structures:group}{समूह} \omterm{def:b1:euclid:isometry}{आइसोमेट्री} का एक सामान्य निश्चित बिंदु होता है।
  \item द्वारा \omterm{def:b1:complex:field}{संयुग्मित} करने के बाद, उसका निष्कर्ष निकालें $ t_{-c}$, कोई भी हो सकता है
        मान लीजिए $ G \subseteq O(\R^2)$. होने देना $ G^{+} = \{g \in G :
        \det g = 1\}$; वह भी दिखाओ $ G = G^{+}$ या $ G^{+}$
        बिल्कुल सूचकांक है $2$ में $ G $ (आक्षेप प्रदर्शित करें $ G^+
        \to G \setminus G^+$).
  \item वह परिमित दिखाओ \omterm{def:b1:structures:group}{समूह} का \emph{रोटेशन} के बारे में $ c $
        चक्रीय है: इसके तत्वों में से घूर्णन को चुनें
        $ R_{\theta_0}$ सबसे छोटे कोण का $\theta_0 \in
        \intoo{0}{2\pi}$, और कोणों के यूक्लिडियन विभाजन का उपयोग करें
        यह साबित करने के लिए कि यह उत्पन्न करता है; निष्कर्ष $\theta_0 =
        \frac{2\pi}n $ और $ G^{+} = \{R_{\theta_0}^{\,k}\} \cong
        C_n $.
  \item कल्पना करना $ G \neq G^{+}$ और एक प्रतिबिंब चुनें $ s \in G $.
        दिखाओ $ G = G^{+} \cup sG^{+}$, वह $ s r s = r^{-1}$ के लिए
        हर घुमाव $ r \in G^{+}$, और वह सब $ n $ घटक $ sG^{+}$ प्रतिबिंब हैं: $ G $ है \emph{\omterm{pb:b1:euclid:1}{डायहेड्रल
        समूह}} $ D_n $ आदेश की $2n $.
  \item निष्कर्ष (\emph{लियोनार्डो का प्रमेय}): हर परिमित \omterm{def:b1:structures:group}{समूह}
        समतल की सममिति चक्रीय है $ C_n $ या डायहेड्रल
        $ D_n $.
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Dividends, and synthesis.}
\begin{enumerate}[resume]
  \item सीधे दिखाओ कि एक परिमित \omterm{def:b1:structures:group}{समूह} \omterm{def:b1:euclid:isometry}{आइसोमेट्री} का कर सकते हैं
        इसमें कोई अनुवाद नहीं है और नहीं \omterm{pb:b1:euclid:1}{सरकना प्रतिबिंब} अन्य
        पहचान की तुलना में (ऐसे की शक्तियों पर विचार करें
        तत्त्व).
  \item होने देना $ P_n $ नियमित रहें $ n $-गॉन शीर्षों के साथ
        $ n $-वां एकता की जड़ें ($ n \geq 3$). दिखाओ कि यह है
        \omterm{def:b1:linmaps:projection}{समरूपता} \omterm{def:b1:structures:group}{समूह} बिलकुल है $ D_n $: द $ n $ रोटेशन $ z
        \mapsto \omega^k z $ और यह $ n $ कुछ विचार $ z \mapsto
        \omega^k \conj z $, $\omega = \eu^{2\iu\pi/n}$, और नहीं
        अन्य.
  \item समतल आकृतियाँ प्रदर्शित करें जिनकी \omterm{def:b1:linmaps:projection}{समरूपता} हो \omterm{def:b1:structures:group}{समूह} हैं $ C_1$,
        $ D_1$, $ D_2$ और $ C_3$ क्रमश।
  \item \omterm{def:b1:linmaps:projection}{समरूपता} के आठ तत्वों की सूची बनाएं \omterm{def:b1:structures:group}{समूह} का
        शीर्षों के साथ वर्गाकार $\pm1, \pm\iu $ जैसा एमएपीएस $ z \mapsto
        \omega^k z $ या $ z \mapsto \omega^k\conj z $, और दे
        चार प्रतिबिंबों में से प्रत्येक की धुरी।
  \item होने देना $ f, g $ एक ही कोण के घूर्णन हों $\theta
        \notin 2\pi\Z $ के बारे में \emph{विशिष्ट} केन्द्रों $ c_1 \neq
        c_2$. गणना करना $ f\circ g - g\circ f $ बिंदुवार दिखाओ
        $ f\circ g \neq g\circ f $; इसके अलावा दिखाओ $(f \circ
        g)\circ(g\circ f)^{-1}$ यह एक गैर-तुच्छ अनुवाद है, इसलिए
        वह कोई भी \omterm{def:b1:structures:group}{समूह} युक्त $ f $ और $ g $ अनंत है ---
        लियोनार्डो के एकल केंद्र की दूसरी व्याख्या
        प्रमेय.
  \item संश्लेषण, चार वाक्यों में: कौन से दो संरचनात्मक
        परिणाम मनमाना \omterm{def:b1:euclid:isometry}{आइसोमेट्री} को रैखिक बीजगणित में कम कर देते हैं
        (प्रश्न 3-4) और मनमाना परिमित \omterm{def:b1:structures:group}{समूह} को
        \omterm{def:b1:structures:subgroup}{उपसमूह}ों का $ O(2)$ (प्रश्न 15); क्या पूरा
        समतल \omterm{def:b1:euclid:isometry}{आइसोमेट्री} की सूची और कौन से अपरिवर्तनीय हैं
        (\omterm{def:b1:vspaces:sum}{प्रत्यक्ष}/अप्रत्यक्ष, निश्चित बिंदु) चार प्रकारों को अलग करें;
        प्रश्न 9 के रचना नियम प्रतिबिंब क्यों बनाते हैं?
        हर चीज़ के जनरेटर; और लियोनार्डो का क्या है
        प्रमेय परिमित पैमाने पर जोड़ता है। दो प्रमेयों के नाम बताइये
        भाग II और IV में सिद्ध किया गया।
\end{enumerate}
\end{problem}
