\chapter{संख्यात्मक श्रृंखला}\label{ch:b1:series}

अपरिमित रूप से अनेक संख्याओं का योग करने का अर्थ है आंशिक की \omterm{def:b1:topology:closure}{सीमा} लेना
योग --- न अधिक, न कम। यह अध्याय सेट ऊपर
प्रथम वर्ष में प्रयोग की जाने वाली परिभाषाएँ और अभिसरण परीक्षण:
सकारात्मक शब्दों के लिए तुलना और समकक्ष, \omterm{thm:b1:series:ratio}{अनुपात परीक्षण}, द
\omterm{thm:b1:integration:def}{अभिन्न} तुलना दे रही है \omterm{thm:b1:series:riemann}{रीमैन श्रृंखला}, पूर्ण अभिसरण,
और यह \omterm{thm:b1:series:alternating}{वैकल्पिक श्रृंखला} प्रमेय. बेहतर सिद्धांत (के उत्पाद)
\omterm{def:b1:series:def}{शृंखला}, पैकेट द्वारा योग, \omterm{def:b1:series:def}{शृंखला} कार्यों का) से संबंधित है
दूसरा वर्ष.

\section{सामान्यिकी}

\begin{definition}\label{def:b1:series:def}
एक क्रम दिया गया है $(u_n)$, द \emph{शृंखला}\index{शृंखला} $\sum u_n $
का क्रम है \emph{आंशिक रकम} $ S_N = \sum_{n=0}^{N} u_n $.
शृंखला \emph{अभिसरण} कब $(S_N)$ अभिसरण; \omterm{def:b1:topology:closure}{सीमा} है
\emph{जोड़} $\sum_{n=0}^{\infty} u_n $, और $ R_N = \sum_{n > N} u_n =
S - S_N $ है \emph{शेष}, जिसकी प्रवृत्ति होती है $0$.
\end{definition}

\begin{example}[जियोमीट्रिक श्रंखला]\label{ex:b1:series:geometric}
के लिए $ q \in \C $: $\;S_N = \sum_{n=0}^{N} q^n = \frac{1 -
q^{N+1}}{1-q}$ ($ q \neq 1$). \omterm{def:b1:series:def}{शृंखला} यदि अभिसरण करता है $\abs q < 1$
(\cref{exo:b1:seq:3}), साथ\index{जियोमीट्रिक श्रंखला}
\[
\sum_{n=0}^{\infty} q^n = \frac{1}{1 - q} .
\]
\end{example}

\begin{example}[आवर्त दशमलव ज्यामितीय श्रृंखला हैं]\label{ex:b1:series:periodicdecimal}
कौन सा नंबर है $0.363636\dots $? इसका लेखन ही एक है \omterm{def:b1:series:def}{शृंखला}:
\[
0.\overline{36} = \sum_{k=1}^{\infty} \frac{36}{100^k}
= 36\cdot\frac{1/100}{1 - 1/100} = \frac{36}{99} =
\frac{4}{11} ,
\]
ज्यामितीय योग द्वारा $ q = \frac{1}{100}$. सामान्य तौर पर ए
ब्लॉक $ B $ का $ p $ अंकों को हमेशा के लिए दोहराना सार्थक है
$\frac{B}{10^p - 1}$ --- आवधिकता के पीछे का तंत्र
की कसौटी \cref{pb:b1:reals:1}, जो इस अध्याय की भाषा है
अंत में एक पंक्ति में कहा गया है: a \omterm{pb:b1:reals:1}{दशमलव विस्तार} एक अभिसरण है
\omterm{def:b1:series:def}{शृंखला}, अंततः बिल्कुल आवधिक जब इसका योग तर्कसंगत होता है।
अध्याय 10 की डिजिट मशीनरी, नंगे के साथ वहां बनाई गई
सुप्रीमा, था \omterm{def:b1:series:def}{शृंखला} सिद्धांत गुप्त यात्रा.
\end{example}

\begin{proposition}[पहला तथ्य]\label{prop:b1:series:first}
\begin{enumerate}
  \item अगर $\sum u_n $ फिर एकत्रित हो जाता है $ u_n \to 0$. (विपरीत यह है
        \emph{असत्य}: हार्मोनिक \omterm{def:b1:series:def}{शृंखला}.)
  \item रैखिकता: अभिसरण \omterm{def:b1:series:def}{शृंखला} अपेक्षित के साथ जोड़ें और स्केल करें
        रकम.
  \item (टेलीस्कोपिंग\index{दूरबीन श्रृंखला}) $\sum (v_{n+1} -
        v_n)$ यदि अभिसरण करता है $(v_n)$ योग के साथ अभिसरण होता है $\lim v_n -
        v_0$.
  \item सीमित रूप से कई पदों को बदलने से अभिसरण पर कोई प्रभाव नहीं पड़ता है
        (केवल योग).
\end{enumerate}
\end{proposition}

\begin{proof}
(1) $ u_N = S_N - S_{N-1} \to S - S = 0$. हार्मोनिक \omterm{def:b1:series:def}{शृंखला} है
$ u_n = \frac1n \to 0$ फिर भी अलग हो जाता है (\cref{exo:b1:seq:5}). (2)
\omterm{def:b1:topology:closure}{सीमा} पर संचालन. (3) $ S_N = v_{N+1} - v_0$. (4) आंशिक
रकम अंततः स्थिर राशि से बदल जाती है।
\end{proof}

\begin{example}[ज्यामितीय शेषफल के साथ अंकों की योजना बनाना]\label{ex:b1:series:georemainder}
के लिए $\abs q < 1$ का शेष \omterm{ex:b1:series:geometric}{जियोमीट्रिक श्रंखला} है
स्पष्ट:
\[
R_N = \sum_{n = N+1}^{\infty} q^n = \frac{q^{N+1}}{1 - q} .
\]
यह सटीकता लक्ष्यों को किसी से पहले टर्म काउंट में परिवर्तित करता है
गणना. मूल्यांकन करना $\sum_{n\geq0} \bigl(\frac13\bigr)^n =
\frac32$ अंदर $10^{-10}$: ज़रूरत $\frac{(1/3)^{N+1}}{2/3} \leq
10^{-10}$, यानी\ $3^{N} \geq \frac{3}{2}\cdot 10^{10}$, यानी\ $ N
\geq 22$ (जैसा $3^{22} \approx 3.1\cdot10^{10}$): तेईस
शर्तें, पहले से ज्ञात। का प्रत्येक ज्यामितीय-दर अनुमान
सप्ताहांत की समस्याएँ ( $\frac13$-श्रृंखला के लिए $\ln 2$, मशीन का
आर्कटिक स्पर्शरेखाएँ में \cref{pb:b1:taylor:1}) यह दो लाइन का बजट है
पेशेवर पोशाक में.
\end{example}

\begin{example}[एक लम्बी दूरबीन]\label{ex:b1:series:telescope3}
गणना करना $\sum_{n\geq1} \frac{1}{n(n+1)(n+2)}$. आंशिक
अंश (\cref{ch:b1:fractions}):
\[
\frac{1}{n(n+1)(n+2)}
= \frac{1/2}{n} - \frac{1}{n+1} + \frac{1/2}{n+2}
= \frac12\Bigl(\frac{1}{n(n+1)} - \frac{1}{(n+1)(n+2)}\Bigr),
\]
जहां दूसरा रूप --- के लगातार मूल्यों का अंतर है
$ w_n = \frac{1}{n(n+1)}$ --- दूरबीन वाला है. इस तरह
\[
\sum_{n=1}^{N} \frac{1}{n(n+1)(n+2)}
= \frac12\Bigl(w_1 - w_{N+1}\Bigr)
= \frac12\Bigl(\frac12 - \frac{1}{(N+1)(N+2)}\Bigr)
\longrightarrow \frac14 .
\]
समापन अंतर्दृष्टि: तीन-अवधि आंशिक अंश शायद ही कभी
लिखित रूप में दूरबीन; उन्हें एक अंतर में पुनः एकत्रित करें $ w_n -
w_{n+1}$ पहला --- इनाम सिर्फ अभिसरण नहीं है बल्कि
सटीक योग, जो कोई तुलना परीक्षण कभी नहीं देता।
\end{example}

\section{गैर-नकारात्मक शब्दों वाली श्रृंखला}

\begin{theorem}[बंधी हुई आंशिक रकम]\label{thm:b1:series:positive}
अगर $ u_n \geq 0$ सभी के लिए $ n $, आंशिक रकम बढ़ जाती है, इसलिए: $\sum
u_n $ अभिसरण $\iff $ इसका आंशिक योग ऊपर परिबद्ध है। इसलिए
\emph{तुलना परीक्षण}: अगर $0 \leq u_n \leq v_n $ सभी के लिए (बड़े)
$ n $,
\[
\sum v_n \text{ converges} \implies \sum u_n \text{ converges},
\qquad
\sum u_n \text{ diverges} \implies \sum v_n \text{ diverges}.
\]
और यह \emph{समकक्ष परीक्षण}: अगर $ u_n \sim v_n $ साथ $ v_n \geq
0$, दो \omterm{def:b1:series:def}{शृंखला} एक ही स्वभाव है.
\end{theorem}

\begin{proof}
मोनोटोन \omterm{def:b1:topology:closure}{सीमा} प्रमेय (\cref{thm:b1:seq:monotone}) पहले के लिए
बिंदु; दूसरे के लिए आंशिक योगों की तुलना। समकक्ष: के लिए
बड़ा $ n $, $\frac12 v_n \leq u_n \leq 2 v_n $ (की परिभाषा $\sim $
साथ $\varepsilon = \frac12$), और तुलना दोनों तरीकों से लागू होती है।
\end{proof}

\begin{example}[एक समतुल्य जो विचलन को सिद्ध करता है]\label{ex:b1:series:equivdiverge}
की प्रकृति $\sum_{n\geq1} n\sin\dfrac{1}{n^2}$? तब से
$\frac{1}{n^2} \to 0$ और $\sin h \sim h $ पर $0$:
\[
n\sin\frac{1}{n^2} \;\sim\; n\cdot\frac{1}{n^2} = \frac1n ,
\]
और समकक्ष परीक्षण के विचलन को स्थानांतरित करता है
हार्मोनिक \omterm{def:b1:series:def}{शृंखला}: भिन्न--- भले ही शर्तों की प्रवृत्ति हो
$0$. एक विस्तार, एक पैमाना, एक फैसला; वही दो चरण
पैटर्न (समकक्ष, फिर रीमैन या ज्यामितीय लुकअप) तय करता है
सभी चार \omterm{def:b1:series:def}{शृंखला} का \cref{exo:b1:series:3}.
\end{example}

\begin{example}[समतुल्य परीक्षण एक पंक्ति में]\label{ex:b1:series:equivtest}
की प्रकृति $\sum_{n \geq 1} \frac{\sqrt{n+1} - \sqrt n}{n}$?
अंश को \omterm{def:b1:complex:field}{संयुग्मित} करें:
\[
\frac{\sqrt{n+1} - \sqrt n}{n}
= \frac{1}{n\,(\sqrt{n+1} + \sqrt n)}
\sim \frac{1}{2\,n^{3/2}} ,
\]
एक अभिसारी रीमैन स्केल ($\alpha = \frac32 > 1$): द \omterm{def:b1:series:def}{शृंखला}
जुटता है. पूरे निर्णय में एक समकक्ष और एक लुकअप लिया गया
--- बशर्ते शर्तें गैर-नकारात्मक हों, जो वे हैं। द
समापन अंतर्दृष्टि: सकारात्मक के लिए \omterm{def:b1:series:def}{शृंखला}, संपूर्ण अभिसरण
सिद्धांत एक है \emph{तराजू का शब्दकोश} ($ n^{-\alpha}$, $ q^n $,
$\frac{1}{n(\ln n)^\alpha}$) साथ ही किसी शब्द को बदलने का लाइसेंस
समकक्ष द्वारा; विश्लेषणात्मक कार्य स्पर्शोन्मुखता में है
(\cref{ch:b1:taylor}), संक्षेप में कभी नहीं.
\end{example}

\begin{theorem}[अभिन्न तुलना; रीमैन श्रृंखला]\label{thm:b1:series:riemann}
होने देना $ f $ होना निरंतर, गैर-नकारात्मक और \emph{घटते} पर
$\intco{1}{+\infty}$. तब
\[
\int_1^{N+1} f(t)\,\dd t \;\leq\; \sum_{n=1}^{N} f(n) \;\leq\; f(1)
+ \int_1^{N} f(t)\,\dd t ,
\]
इसलिए $\sum f(n)$ यदि अभिसरण करता है $\bigl(\int_1^x f\bigr)$ घिरा है। में
विशेष रूप से, के लिए $\alpha \in \R $:\index{रीमैन श्रृंखला}
\[
\sum_{n \geq 1} \frac{1}{n^\alpha} \text{ converges}
\iff \alpha > 1,
\]
और $\sum_{n=1}^{N} \frac1n = \ln N + O(1)$.
\end{theorem}

\begin{proof}
के लिए $ n \leq t \leq n+1$, एकरसता देता है $ f(n+1) \leq f(t)
\leq f(n)$; पर एकीकरण $\intcc{n}{n+1}$ (का एक खंड
लंबाई $1$):
\[
f(n+1) \;\leq\; \int_n^{n+1} f(t)\,\dd t \;\leq\; f(n) .
\]
के लिए सही असमानताओं का योग करना $ n = 1, \dots, N-1$ देता है
$\int_1^{N} f \leq \sum_{n=1}^{N-1} f(n)$, इसलिए ऊपरी
जोड़ने के बाद फ्रेमिंग $ f(N) \leq f(1)$; बाएँ को संक्षेप में प्रस्तुत करें
$ n = 1, \dots, N $ देता है $\sum_{n=2}^{N+1} f(n) \leq
\int_1^{N+1} f $, जो रीइंडेक्सिंग के बाद निचली फ़्रेमिंग है।
अभिसरण: आंशिक रकम और \omterm{thm:b1:integration:def}{अभिन्न} $\int_1^x f $
स्थिरांक के भीतर एक दूसरे को बांधें $ f(1)$, और दोनों हैं
गैर-घटता नहीं, इसलिए यदि दूसरा है तो एक परिबद्ध है
(\cref{thm:b1:series:positive}). के लिए $ f(t)
= t^{-\alpha}$ ($\alpha \neq 1$): $\int_1^x t^{-\alpha}\dd t =
\frac{x^{1-\alpha} - 1}{1 - \alpha}$, परिबद्ध iff $\alpha > 1$; के लिए
$\alpha = 1$  \omterm{thm:b1:integration:def}{अभिन्न} है $\ln x \to \infty $, और फ़्रेमिंग
देता है $\ln(N+1) \leq H_N \leq 1 + \ln N $. के लिए $\alpha \leq 0$ द
शर्तें प्रवृत्त नहीं होतीं $0$.
\end{proof}

\begin{example}[हार्मोनिक स्टैक]\label{ex:b1:series:harmonicstack}
हार्मोनिक में कितने पद होने चाहिए \omterm{def:b1:series:def}{शृंखला} पारित करने के लिए जमा करें $20$?
फ़्रेमिंग $\ln(N+1) \leq H_N \leq 1 + \ln N $ नहीं में उत्तर देता है
संक्षेप में: $ H_N \geq 20$ आवश्यक है $1 + \ln N \geq 20$,
यानी\ $ N \geq \eu^{19} \approx 1.8\cdot10^{8}$, और इसकी गारंटी है
एक बार $\ln(N + 1) \geq 20$, यानी\ $ N \approx \eu^{20} \approx
4.9\cdot10^{8}$. (सप्ताहान्त की समस्या इसे और तीव्र कर देती है $ N \approx
\eu^{20 - \gamma} \approx 2.7\cdot10^{8}$ के जरिए \omterm{ex:b1:series:oddtelescope}{यूलर स्थिरांक}.)
समापन अंतर्दृष्टि: \omterm{thm:b1:integration:def}{अभिन्न} तुलना मात्र नहीं है
अभिसरण तय करें --- यह \emph{रेखांकित} आंशिक रकम के साथ
लघुगणक परिशुद्धता, एक निराशाजनक गणना बदल रही है
(सैकड़ों लाखों शब्द) दो-पंक्ति अनुमान में।
\end{example}

\begin{theorem}[अनुपात परीक्षण (डी'एलेम्बर्ट)]\label{thm:b1:series:ratio}
होने देना $ u_n > 0$ साथ $\frac{u_{n+1}}{u_n} \to \ell $.\index{अनुपात परीक्षण}
\begin{itemize}
  \item अगर $\ell < 1$: $\sum u_n $ अभिसरण;
  \item अगर $\ell > 1$: $ u_n \to +\infty $, विचलन;
  \item अगर $\ell = 1$: कोई निष्कर्ष नहीं ($\sum \frac1n $ विचलन, $\sum
        \frac{1}{n^2}$ अभिसरण)।
\end{itemize}
\end{theorem}

\begin{proof}
अगर $\ell < 1$, हल करना $ q \in \intoo{\ell}{1}$: कुछ से परे $ N $,
$ u_{n+1} \leq q\,u_n $, इसलिए $ u_n \leq u_N q^{\,n-N}$ प्रेरण द्वारा:
ए के साथ तुलना \omterm{ex:b1:series:geometric}{जियोमीट्रिक श्रंखला}. अगर $\ell > 1$: कुछ से परे $ N $
क्रम $(u_n)$ बढ़ रहा है, इसलिए इसकी प्रवृत्ति नहीं हो सकती $0$
(इसकी \omterm{def:b1:topology:closure}{सीमा}, यदि कोई हो, है $\geq u_N > 0$); द्वारा
\cref{prop:b1:series:first}~(1), विचलन --- और वास्तव में $ u_n
\geq u_N q^{n-N}$ साथ $ q > 1$ देता है $ u_n \to \infty $.
\end{proof}

\begin{example}\label{ex:b1:series:ratio}
$\sum \frac{x^n}{n!}$ प्रत्येक के लिए अभिसरण होता है $ x > 0$: अनुपात
$\frac{x}{n+1} \to 0$. इसका योग है $\eu^x $: टेलर--लैग्रेंज द्वारा
(\cref{thm:b1:taylor:lagrange}) पर $\intcc{0}{x}$,
\[
\Bigl| \eu^x - \sum_{k=0}^{n} \frac{x^k}{k!} \Bigr|
\leq \eu^{x}\, \frac{x^{n+1}}{(n+1)!} \xrightarrow[n\to\infty]{} 0 ,
\]
बाध्य करने के लिए प्रवृत्त $0$ क्योंकि फैक्टोरियल हावी है
(\cref{exo:b1:integration:9}~(1) उसी तथ्य का प्रयोग किया गया)। वैसा ही
तर्क का सार है $\sin $, $\cos $, $\sinh $, $\cosh $ \omterm{def:b1:series:def}{शृंखला} सब पर
$\R $.
\end{example}

\begin{example}[अनुपात परीक्षण पर्याप्त है, आवश्यक नहीं]\label{ex:b1:series:rationotnecessary}
होने देना $ u_n = 2^{-n}$ एक जैसे के लिए $ n $ और $ u_n = 2^{-n-2}$ विषम के लिए
$ n $. लगातार अनुपात के बीच उतार-चढ़ाव होता है $\frac{1}{8}$ और
$\frac12\cdot4 = 2$, इसलिए $\frac{u_{n+1}}{u_n}$ इसकी कोई \omterm{def:b1:topology:closure}{सीमा} नहीं है और
डी'एलेम्बर्ट मूक है --- अभी तक $ u_n \leq 2^{-n}$ और तुलना
परीक्षण अभिसरण को तुरंत सुलझाता है। परीक्षण की परिकल्पना (
अनुपात \emph{अभिसरण}) एक वास्तविक प्रतिबंध है: यह शर्तों के अनुरूप है
एक प्रमुख गुणात्मक संरचना (फैक्टोरियल, शक्तियां) के साथ,
और सांस लेने वाली किसी भी चीज़ पर विफल रहता है। जब अनुपात ख़राब होते हैं,
एक ज्यामितीय लिफ़ाफ़े के विरुद्ध तुलना करने के लिए पीछे हटें --- जो
सब कुछ है \omterm{thm:b1:series:ratio}{अनुपात परीक्षण} जैसा कि इसके प्रमाण से पता चलता है, कभी था।
\end{example}

\begin{example}[तथ्यात्मक लड़ाइयों पर अनुपात परीक्षण]\label{ex:b1:series:ratiobinom}
की प्रकृति $\sum_{n\geq0} \dfrac{(n!)^2}{(2n)!}$ (के पारस्परिक
केंद्रीय \omterm{def:b1:counting:objects}{द्विपद गुणांक}, कारक तक $ n + 1$)?
अनुपात भाज्य को ध्वस्त कर देता है:
\[
\frac{u_{n+1}}{u_n}
= \frac{((n+1)!)^2}{(n!)^2}\cdot\frac{(2n)!}{(2n+2)!}
= \frac{(n+1)^2}{(2n+1)(2n+2)}
\longrightarrow \frac14 < 1 :
\]
अभिसरण, अतिरिक्त गुंजाइश के साथ --- शब्द अनिवार्य रूप से क्षय हो जाते हैं
जैसे $4^{-n}$, के साथ संगत $\binom{2n}{n} \geq
\frac{4^n}{2n+1}$ से \cref{pb:b1:integration:1}. समापन
अंतर्दृष्टि: फैक्टोरियल के भागफल बिल्कुल वही होते हैं \omterm{thm:b1:series:ratio}{अनुपात परीक्षण}
डाइजेस्ट --- प्रत्येक फैक्टोरियल एक तर्कसंगत कार्य में रद्द हो जाता है
$ n $, जिसकी \omterm{def:b1:topology:closure}{सीमा} प्रमुख पदों से पढ़ी जाती है।
\end{example}

\section{पूर्ण अभिसरण; वैकल्पिक श्रृंखला}

\begin{theorem}[पूर्ण अभिसरण]\label{thm:b1:series:absolute}
अगर $\sum \abs{u_n}$ अभिसरण (\emph{निरपेक्ष
अभिसरण}\index{निरपेक्ष
अभिसरण}), तब $\sum u_n $ जुटता है,
और $\bigl|\sum u_n\bigr| \leq \sum \abs{u_n}$. यह सच है
या जटिल शब्द.
\end{theorem}

\begin{proof}
आंशिक रकम संतुष्ट करती है, के लिए $ M > N $ (कौची मानदंड,
\cref{thm:b1:seq:complete}):
\[
\abs{S_M - S_N} = \Bigl| \sum_{n=N+1}^{M} u_n \Bigr|
\leq \sum_{n=N+1}^{M} \abs{u_n},
\]
जो बड़े के लिए छोटा है $ N $ के आंशिक योग के बाद से $\sum\abs{u_n}$
फॉर्म ए \omterm{def:b1:seq:cauchy}{कॉची अनुक्रम}. इसलिए $(S_N)$ कॉची है, इसलिए अभिसरण है। द
असमानता परिमित त्रिभुज असमानता से \omterm{def:b1:topology:closure}{सीमा} तक गुजरती है।
\end{proof}

\begin{example}[पूर्ण अभिसरण, वास्तविक और जटिल]\label{ex:b1:series:absexamples}
$\sum_{n\geq1} \frac{\sin n}{n^2}$: शर्तें संकेत बदलती हैं
अनियमित रूप से (वास्तव में) $(\sin n)$ है \omterm{def:b1:topology:dense}{घना} में $\intcc{-1}{1}$,
\cref{exo:b1:seq:12}), और कोई वैकल्पिक संरचना नहीं है
दृष्टि. पूर्ण अभिसरण एक ही बार में सब कुछ बचा लेता है:
$\bigl|\frac{\sin n}{n^2}\bigr| \leq \frac{1}{n^2}$, ए
अभिसरण पैमाने, तो \omterm{def:b1:series:def}{शृंखला} जुटता है. वही ढाल काम करती है
खत्म $\C $: $\sum_{n\geq1}\frac{\eu^{\iu n}}{n^2}$ जुटता है
क्योंकि $\bigl|\frac{\eu^{\iu n}}{n^2}\bigr| = \frac{1}{n^2}$
--- संकेत पैटर्न, यहां तक कि द्वि-आयामी भी, अप्रासंगिक हैं
एक बार मॉड्यूल सारांशित हो जाएं। समापन अंतर्दृष्टि: \omterm{thm:b1:series:absolute}{निरपेक्ष
अभिसरण} इस अध्याय का एकमात्र उपकरण है जो कभी नहीं पूछता कि कैसे
संकेत व्यवस्थित हैं; पहले इसे आज़माएं
(\cref{met:b1:series:decide}), और नाजुक परीक्षण सुरक्षित रखें
के लिए \omterm{def:b1:series:def}{शृंखला} वह इसे विफल कर देता है।
\end{example}

\begin{theorem}[वैकल्पिक श्रृंखला परीक्षण]\label{thm:b1:series:alternating}
होने देना $(a_n)$ के साथ घट रहा है $ a_n \to 0$. फिर बारी-बारी से
\omterm{def:b1:series:def}{शृंखला} $\sum (-1)^n a_n $ अभिसरण; इसका योग किन्हीं दो के बीच होता है
लगातार आंशिक रकम, और\index{वैकल्पिक श्रृंखला}
\[
\abs{R_N} = \Bigl| \sum_{n > N} (-1)^n a_n \Bigr| \leq a_{N+1} .
\]
\end{theorem}

\begin{proof}
सम और विषम आंशिक योग आसन्न हैं: $ S_{2p+2} - S_{2p} =
a_{2p+2} - a_{2p+1} \leq 0$ (घटता हुआ), $ S_{2p+1} - S_{2p-1} =
a_{2p} - a_{2p+1} \geq 0$ (बढ़ रहा है), और $ S_{2p} - S_{2p+1} =
a_{2p+1} \to 0$. द्वारा \cref{thm:b1:seq:adjacent} वे एक \omterm{def:b1:topology:closure}{सीमा} साझा करते हैं
$ S $, जो दो परिणाम कसौटी
(\cref{prop:b1:seq:subsequences}) की \omterm{def:b1:topology:closure}{सीमा} बनाता है $(S_N)$;
इसके अतिरिक्त $ S $ लगातार आंशिक रकम के बीच फंसा हुआ है, और
$\abs{S - S_N}$ अगले वाले के लिए अधिकतम अंतर है, $ a_{N+1}$.
\end{proof}

\begin{example}[वैकल्पिक हार्मोनिक श्रृंखला]\label{ex:b1:series:ln2}
$\sum_{n \geq 1} \frac{(-1)^{n-1}}{n}$ अभिसरण (वैकल्पिक परीक्षण)
लेकिन बिल्कुल नहीं (हार्मोनिक)। \omterm{def:b1:series:def}{शृंखला}). इसका योग है $\ln 2$: से
परिमित ज्यामितीय पहचान $\frac{1}{1+t} = \sum_{k=0}^{n-1} (-t)^k +
\frac{(-t)^n}{1+t}$, एकीकृत करें $\intcc{0}{1}$:
\[
\ln 2 = \sum_{k=1}^{n} \frac{(-1)^{k-1}}{k}
+ (-1)^n \int_0^1 \frac{t^n}{1+t}\,\dd t ,
\qquad
0 \leq \int_0^1 \frac{t^n}{1+t}\,\dd t \leq \frac{1}{n+1} \to 0 .
\]
अभिसरण अत्यंत धीमी गति से होता है ($ R_N \approx \frac{1}{N}$) ---
\omterm{thm:b1:series:alternating}{वैकल्पिक श्रृंखला} रद्दीकरण से जुटें, लघुता से नहीं।
\end{example}

\begin{remark}[श्रृंखला के साथ आम नुकसान]
(मैं) \emph{समकक्ष परीक्षण के लिए एक चिन्ह की आवश्यकता होती है}: होने देना $ v_n =
\frac{(-1)^n}{\sqrt n}$ और $ u_n = v_n + \frac1n $. तब
$\frac{u_n}{v_n} = 1 + \frac{(-1)^n}{\sqrt n} \to 1$, इसलिए $ u_n
\sim v_n $; अभी तक $\sum v_n $ जबकि अभिसरण (वैकल्पिक परीक्षण)।
$\sum u_n = \sum v_n + \sum \frac1n $ विचलन समतुल्यता
को नियंत्रित करता है \emph{आकार} शर्तों की, और हस्ताक्षरित के लिए \omterm{def:b1:series:def}{शृंखला} आकार
नियति नहीं है --- परीक्षण कहा गया है, और सच है, के लिए
(अंततः) केवल गैर-नकारात्मक शब्द। (ii) \emph{$ u_n \to 0$
कुछ भी साबित नहीं होता}: हार्मोनिक \omterm{def:b1:series:def}{शृंखला} शाश्वत है
प्रतिउदाहरण; विपरीत दिशा
(\cref{prop:b1:series:first}~(1)) केवल एक त्वरित विचलन है
परीक्षण. (iii) \emph{अनुपात \omterm{def:b1:topology:closure}{सीमा} $1$ मौन है, नहीं
अभिसरण}: दोनों $\sum\frac1n $ और $\sum\frac{1}{n^2}$ है
अनुपात $\to 1$; रीमैन स्केल पर स्विच करें या \omterm{thm:b1:integration:def}{अभिन्न} तुलना।
(iv) \emph{वैकल्पिक आवश्यकताएँ कम हो रही हैं}: $\sum
\frac{(-1)^n}{n + (-1)^n}$ परिवर्तनशील दिखता है और संभाला जाता है
केवल विस्तार से (\cref{exo:b1:series:5}); सप्ताहांत की समस्या
का \cref{ch:b1:taylor} (वहां प्रश्न 23) परीक्षण कैन दिखाता है
एकरसता के बिना पूरी तरह असफल हो जाना। (v) \emph{\omterm{def:b1:structures:group}{समूह}ीकरण और
पुन: ऑर्डर करना निःशुल्क नहीं है}: कोष्ठक सम्मिलित करना हानिरहित है
अभिसारी \omterm{def:b1:series:def}{शृंखला} लेकिन विचलन से अभिसरण पैदा कर सकता है
($1 - 1 + 1 - \cdots $ जोड़े में \omterm{def:b1:structures:group}{समूह}ीकृत), और पुन: व्यवस्थित किया जा सकता है
इस अध्याय में मंचित नाटक का योग ही बदल दें
सप्ताहांत समस्या (\cref{pb:b1:series:1}).
\end{remark}

\begin{method}[किसी शृंखला की प्रकृति का निर्णय करना]\label{met:b1:series:decide}
\begin{enumerate}
  \item करता है $ u_n \to 0$? यदि नहीं, तो विचलन, रुकें।
  \item गैर-नकारात्मक शब्द: के समकक्ष खोजें $ u_n $ (विस्तार,
        \cref{ch:b1:taylor}!), रीमैन या जियोमेट्रिक से तुलना करें
        तराजू; भाज्य और शक्तियों की आवश्यकता होती है \omterm{thm:b1:series:ratio}{अनुपात परीक्षण};
        घटते $ f(n)$ के लिए बुलाता है \omterm{thm:b1:integration:def}{अभिन्न} तुलना।
  \item संकेत अलग-अलग होते हैं: प्रयास करें पूर्ण अभिसरण पहला; यदि यह विफल हो जाता है, तो
        वैकल्पिक परीक्षण (चेक \emph{घटते} सावधानी से); परे
        वह, दूसरे वर्ष के उपकरण।
\end{enumerate}
\end{method}

\begin{example}[विषम भाजक, आधा दूरबीन]\label{ex:b1:series:oddtelescope}
गणना करना $\sum_{n \geq 1} \dfrac{1}{4n^2 - 1}$. आंशिक
भिन्न: $\frac{1}{(2n-1)(2n+1)} =
\frac12\bigl(\frac{1}{2n-1} - \frac{1}{2n+1}\bigr)$, इसलिए
\[
\sum_{n=1}^{N} \frac{1}{4n^2 - 1}
= \frac12\Bigl(1 - \frac{1}{2N+1}\Bigr)
\longrightarrow \frac12 .
\]
साथ तुलना करें $\sum \frac{1}{n(n+1)} = 1$
(\cref{exo:b1:series:1}): वही टेलीस्कोपिंग कंकाल, लेकिन
यहां लगातार पद विषम संख्याओं में दो अलग-अलग हैं, और
कारक $\frac12$ चरण रिकॉर्ड करता है. समापन अंतर्दृष्टि:
टेलीस्कोपिंग दृष्टिकोण का \omterm{def:b1:counting:objects}{परिवर्तन} है, कोई चाल नहीं --- जब भी
सामान्य शब्द एक अंतर है $ w_n - w_{n+1}$ एक क्रम का
एक \omterm{def:b1:topology:closure}{सीमा} के साथ, योग है $ w_1 - \lim w $, बिल्कुल
\cref{prop:b1:series:first}~(3).
\end{example}

\begin{remark}[विश्लेषण पाइपलाइन, पूर्वव्यापी में]
यह अध्याय वह है जहां वॉल्यूम का विश्लेषण एकत्रित होता है, और प्रत्येक
परीक्षण अपने पूर्वज का नाम बताता है। परिबद्ध आंशिक रकम एकस्वर है
\omterm{def:b1:topology:closure}{सीमा} प्रमेय (\cref{ch:b1:seq}), स्वयं पूर्णता स्वयंसिद्ध है
का \cref{ch:b1:reals}; पूर्ण अभिसरण कॉची है
कसौटी; \omterm{thm:b1:integration:def}{अभिन्न} परीक्षण है \cref{ch:b1:integration}का
क्षेत्रों का निर्धारण; सामान्य शब्दों के समकक्ष हैं
\cref{ch:b1:taylor}का विस्तार; और प्रत्यावर्ती प्रमेय है
इसके रविवार के कपड़ों में आसन्न-अनुक्रम लेम्मा। पढ़ें
पीछे की ओर, पाइपलाइन बताती है कि प्रत्येक अध्याय क्या था
\emph{के लिए} --- और सप्ताहांत की समस्याएं इसमें उलझ गईं
($ b $-एडिक अंक, सेस\`एआरओ--स्टोल्ज़, अतार्किकता मशीनें,
\omterm{ex:b1:series:oddtelescope}{यूलर स्थिरांक}) वही कुछ विचार उच्च स्तर पर मिलते हैं और
अधिक ऊंचाई. रैखिक बीजगणित जो \omterm{def:b1:counting:objects}{परिवर्तन}ों का अनुसरण करता है
विषय, मानक नहीं: सटीक की आदत \omterm{def:b1:logic:statement}{कथन} के साथ
प्रमाणित त्रुटि \omterm{def:b1:topology:closure}{सीमा} से आयाम की ओर बढ़ने से बची रहती है।
\end{remark}

\begin{remark}[श्रृंखला आगे कहां जाएगी]
यह अध्याय वॉल्यूम का विश्लेषण \omterm{def:b1:topology:closure}{बंद} कर देता है खुलता है तीन
दरवाजे. वर्ष 2 खंड में, \omterm{def:b1:series:def}{शृंखला} एक चर प्राप्त करें ($\sum
a_n x^n $: शक्ति \omterm{def:b1:series:def}{शृंखला}, उनके अभिसरण की त्रिज्या के साथ) और
फिर एक फ़ंक्शन-मूल्यवान सिद्धांत (फूरियर)। \omterm{def:b1:series:def}{शृंखला}); द्विभाजन
पूर्ण-बनाम-सशर्त अभिसरण, नाटकीय रूप से
नीचे सप्ताहांत की समस्या, दोनों की आधारशिला बन जाती है। में
संभाव्यता (वर्ष 3 खंड), असतत यादृच्छिक की अपेक्षाएँ
चर \emph{हैं} \omterm{def:b1:series:def}{शृंखला}, और पूर्ण अभिसरण क्या है
उन्हें अच्छी तरह से परिभाषित करता है। और यह \omterm{thm:b1:series:riemann}{रीमैन श्रृंखला} $\sum n^{-s}$,
जटिल की ओर धकेला गया $ s $, जीटा फ़ंक्शन बन जाता है --- एकल
सबसे अधिक अध्ययन किया गया \omterm{def:b1:series:def}{शृंखला} गणित में।
\end{remark}

\section{अभ्यास}

\begin{exercise}[$\star $]\label{exo:b1:series:1}
प्रकृति (और संक्षेप में, जब दूरबीन से):
\[
\sum_{n\geq1} \frac{1}{n(n+1)},
\qquad
\sum_{n\geq2} \ln\Bigl(1 - \frac{1}{n^2}\Bigr),
\qquad
\sum_{n\geq0} \frac{3^n + 4^n}{5^n} .
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:series:2}
की प्रकृति: $\;\sum \dfrac{n^2}{2^n}$; $\;\sum \dfrac{n!}{n^n}$;
$\;\sum \dfrac{2^n\,n!}{n^n}$; $\;\sum \dfrac{3^n\,n!}{n^n}$.
\emph{(\omterm{thm:b1:series:ratio}{अनुपात परीक्षण}; याद करना $\bigl(1 + \frac1n\bigr)^n \to \eu $.)}
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:series:3}
की प्रकृति: $\;\sum \sin\dfrac{1}{n^2}$; $\;\sum
\Bigl(1 - \cos\dfrac1n\Bigr)$; $\;\sum \dfrac{1}{\sqrt{n(n+1)}}$;
$\;\sum \dfrac{\ln n}{n^2}$ \emph{(साथ तुलना करें $ n^{-3/2}$)}.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:series:4}
साबित करें कि $\sum_{n\geq1} \frac{1}{n^2}$ योग के साथ अभिसरण होता है $\leq 2$,
का उपयोग कर रहे हैं $\frac{1}{n^2} \leq \frac{1}{n(n-1)}$ के लिए $ n \geq 2$ और ए
दूरबीन बाध्य.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:series:5}
की प्रकृति $\;\sum \dfrac{(-1)^n}{\sqrt n}$, का $\;\sum
\dfrac{(-1)^n}{n + (-1)^n}$ \emph{(विस्तृत करें: वैकल्पिक परीक्षण करता है
सीधे लागू नहीं --- क्यों?)}, और का $\;\sum
\sin\bigl(\pi\sqrt{n^2+1}\,\bigr)$ \emph{(मॉड्यूलो कम करें $\pi $:
$\sqrt{n^2+1} = n + \frac{1}{2n} + O(n^{-3})$)}.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:series:6}
जिसके लिए $\alpha > 0$ करता है $\sum_{n \geq 2}
\dfrac{1}{n (\ln n)^{\alpha}}$ जुटना? \emph{(\omterm{thm:b1:integration:def}{अभिन्न} तुलना;
विकल्प $ u = \ln t $.)}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:series:7}
होने देना $ u_n = \dfrac{1}{n} - \ln\Bigl(1 + \dfrac1n\Bigr)$. साबित करें कि
$0 \leq u_n \leq \dfrac{1}{2n^2}$, वह $\sum u_n $ अभिसरण, और
के अस्तित्व का अनुमान लगाएं \omterm{ex:b1:series:oddtelescope}{यूलर स्थिरांक}\index{यूलर स्थिरांक}:
\[
\gamma = \lim_{N \to \infty} \Bigl( \sum_{n=1}^{N} \frac 1n - \ln N
\Bigr) .
\]
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:series:8}
रकम की गणना करें
\[
\sum_{n=1}^{\infty} \frac{1}{n(n+2)}
\qquad\text{and}\qquad
\sum_{n=0}^{\infty} \frac{n}{2^n} .
\]
\emph{(पहले के लिए: आंशिक भिन्न। दूसरे के लिए: गणना करें
$\sum_{n=1}^{N} n x^{n-1}$ \omterm{def:b1:topology:closure}{बंद} रूप में और चलो $ N \to \infty $ पर
$ x = \frac12$.)}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b1:series:9}
(कॉची संक्षेपण) चलो $(u_n)$ अऋणात्मक और घटता हुआ हो।
साबित करें कि
\[
\sum_{n \geq 1} u_n \text{ converges}
\iff
\sum_{k \geq 0} 2^k\, u_{2^k} \text{ converges},
\]
की लगातार शक्तियों के बीच शब्दों के पैकेट की तुलना करके $2$.
इससे रीमैन मानदंड पुनर्प्राप्त करें और
\cref{exo:b1:series:6}.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b1:series:10}
का उपयोग \omterm{thm:b1:integration:def}{अभिन्न} की पहचान \cref{ex:b1:series:ln2} के लिए अनुकूलित
$\frac{1}{1+t^2}$, लीबनिज़ के सूत्र को सिद्ध करें
\[
\frac{\pi}{4} = \sum_{n=0}^{\infty} \frac{(-1)^n}{2n+1}
= 1 - \frac13 + \frac15 - \frac17 + \cdots
\]
त्रुटि बाध्य के साथ $\abs{R_N} \leq \frac{1}{2N+3}$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:series:11}
की प्रकृति $\displaystyle\sum_{n \geq 1} \frac{1}{n^{1 + 1/n}}$.
\emph{(की \omterm{def:b1:topology:closure}{सीमा} की गणना करें $ n^{1/n}$ और इसके समकक्ष खोजें
सामान्य शब्द: रीमैन परीक्षण की आवश्यकता है \emph{fixed}
प्रतिपादक.)}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b1:series:12}
होने देना $(u_n)$ गैरनकारात्मक हो और \emph{घटते} साथ $\sum u_n $
अभिसारी। साबित करें कि $ n\,u_n \to 0$ \emph{(अवश्यंभावी $ n\,u_{2n}$
एक टुकड़े से $\sum_{k=n+1}^{2n} u_k $ और कॉची का उपयोग करें
कसौटी)}. दिखाएँ कि बातचीत विफल हो जाती है, और वह
एकरसता परिकल्पना को हटाया नहीं जा सकता।
\end{exercise}

\section{समस्या: यूलर स्थिरांक और श्रृंखला जो बदलती है
इसका योग}

\begin{problem}[{सप्ताहांत समस्या --- $ H_n = \ln n + \gamma + \frac{1}{2n} + O(n^{-2})$, और पुनर्व्यवस्थित करना $1 - \frac12 + \frac13 - \dots $ को $\frac{\ln 2}{2}$}]
\label{pb:b1:series:1}
दो कहानियाँ हार्मोनिक साझा करती हैं \omterm{def:b1:series:def}{शृंखला}. सबसे पहले, सटीक
इसके विचलन का बहीखाता: $ H_n - \ln n $ में एकत्रित हो जाता है \omterm{pb:b1:series:1}{यूलर का
स्थिर} $\gamma $ (\cref{exo:b1:series:7}), और यह समस्या
को तेज़ करता है \omterm{def:b1:logic:statement}{कथन} दोतरफा कानून में $\frac{1}{2(n+1)}
\leq H_n - \ln n - \gamma \leq \frac{1}{2n}$, प्रमाणित करना $\gamma
= 0.5772\dots $ हाथ से. दूसरा, सशर्त का घोटाला
अभिसरण: प्रत्यावर्ती हार्मोनिक \omterm{def:b1:series:def}{शृंखला} का योग है $\ln 2$
(\cref{ex:b1:series:ln2}), अभी तक \emph{समान शर्तें, ए में
अलग क्रम}, योग करने के लिए $\frac{\ln 2}{2}$ --- या करने के लिए $\ln 2 +
\frac12\ln\frac pq $ किसी के लिए $ p, q $, या किसी भी वास्तविक चीज़ के लिए
(रीमैन)। दोनों कहानियाँ एक हैं: पुनर्व्यवस्थित योग हैं
गणना की गई \emph{साथ}  $\gamma $-कानून।\index{यूलर का
स्थिर}\index{श्रृंखला की पुनर्व्यवस्था}

\medskip
\noindent\textbf{Part I --- $\gamma $, bracketed.} \omterm{def:b1:logic:sets}{तय करना} $ a_n = H_n
- \ln n $ और $ b_n = H_n - \ln(n+1)$.
\begin{enumerate}
  \item का उपयोग करते हुए $\frac{t}{1+t} \leq \ln(1 + t) \leq t $, बताते हैं कि
        $(a_n)$ घट जाती है, $(b_n)$ बढ़ता है, और वे हैं
        निकटवर्ती; उनकी सामान्य \omterm{def:b1:topology:closure}{सीमा} है $\gamma $, साथ $ b_n \leq
        \gamma \leq a_n $ हरएक के लिए $ n $.
  \item संख्यात्मक पहला शॉट: से $ H_{10} = 2.928968\dots $,
        ब्रैकेट $\gamma $ बीच में $ b_{10} = 0.5311$ और $ a_{10} =
        0.6264$. कितना बड़ा $ n $ क्या इस कच्चे ब्रैकेट की आवश्यकता होगी
        चार दशमलव के लिए?
  \item पूँछ का सटीक प्रतिनिधित्व दिखाएँ $ a_n - \gamma =
        \sum_{k \geq n} w_k $ (आंशिक रकम की \omterm{def:b1:topology:closure}{सीमा}), जहां
        \[
        w_k = a_k - a_{k+1}
        = \ln\Bigl(1 + \frac1k\Bigr) - \frac{1}{k+1}
        = \int_k^{k+1} \frac{(k + 1 - t)}{t\,(k+1)}\,\dd t ,
        \]
        और से निष्कर्ष निकालें \omterm{thm:b1:integration:def}{अभिन्न} दो तरफा बंधन बनाओ
        $\dfrac{1}{2(k+1)^2} \leq w_k \leq \dfrac{1}{2k(k+1)}$.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The $\frac{1}{2n}$ law.}
\begin{enumerate}[resume]
  \item प्रश्न 3 की सीमाओं का योग करें (दोनों तरफ दूरबीन या
        दूरबीनों से तुलना करें) और नियम का निष्कर्ष निकालें:
        \[
        \frac{1}{2(n+1)} \;\leq\; H_n - \ln n - \gamma \;\leq\;
        \frac{1}{2n} \qquad (n \geq 1).
        \]
  \item परिणाम निकालना $ H_n = \ln n + \gamma + \frac{1}{2n} +
        O\bigl(\frac{1}{n^2}\bigr)$; ठीक-ठीक, वह दिखाओ
        $\gamma_n = H_n - \ln n - \frac{1}{2n}$ संतुष्ट
        $-\frac{1}{2n(n+1)} \leq \gamma_n - \gamma \leq 0$.
  \item चार दशमलव के साथ प्रमाणित करें $ n = 100$: दिया गया $ H_{100} =
        5.1873775\dots $, गणना करें $\gamma_{100} = 0.577207\dots $
        और निष्कर्ष निकालो $\gamma = 0.5772 \pm 5\cdot10^{-5}$ (सच
        मूल्य $0.5772156\dots $).
  \item कानून के दो लाभांश, दोनों की बाद में आवश्यकता थी: जैसे $ m \to
        \infty $,
        \[
        H_{2m} - H_m = \ln 2 - \frac{1}{4m} +
        O\Bigl(\frac{1}{m^2}\Bigr),
        \qquad
        \sum_{j=1}^{m} \frac{1}{2j-1} = \frac{\ln m}{2} + \ln 2
        + \frac\gamma2 + o(1) ,
        \]
        के माध्यम से दूसरा $\sum_{j \leq m} \frac{1}{2j-1} = H_{2m}
        - \frac12 H_m $, और इसी तरह $\sum_{j=1}^{m}
        \frac{1}{2j} = \frac{\ln m}{2} + \frac\gamma2 + o(1)$.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The alternating harmonic \omterm{def:b1:series:def}{शृंखला},
to second order.}
\begin{enumerate}[resume]
  \item पहचान दिखाएँ (प्रेरण, या \omterm{def:b1:structures:group}{समूह}ीकरण)।
        $\sum_{k=1}^{2m} \frac{(-1)^{k-1}}{k} = H_{2m} - H_m $,
        और दोनों का योग घटाएं $\ln 2$ (फिर से) और सटीक
        गति:
        \[
        \sum_{k=1}^{2m} \frac{(-1)^{k-1}}{k}
        = \ln 2 - \frac{1}{4m} + O\Bigl(\frac{1}{m^2}\Bigr) .
        \]
  \item प्रत्यावर्ती हार्मोनिक की स्पर्शोन्मुख त्रुटि का पता लगाएं
        \omterm{def:b1:series:def}{शृंखला} पर \emph{कोई} अनुक्रमणिका: $ S - S_N \sim
        \frac{(-1)^N}{2N}$ --- सबसे खराब स्थिति से दोगुना छोटा
        बाध्य $ a_{N+1} \approx \frac1N $ का
        \cref{thm:b1:series:alternating}.
  \item (मुक्त करने के लिए त्वरण) दिखाएँ कि औसत रकम
        $\tilde S_N = \frac{S_N + S_{N+1}}{2}$ संतुष्ट $\tilde
        S_N = \ln 2 + O\bigl(\frac{1}{N^2}\bigr)$. जाँच करना: $ S_{10}
        = 0.64563$, $ S_{11} = 0.73654$, $\tilde S_{10} =
        0.69109$, ख़िलाफ़ $\ln 2 = 0.69315$: एक औसत खरीदारी
        दो दशमलव स्थान.
  \item दो वाक्यों में बताएं कि ऐसी कोई भी युक्ति मदद क्यों नहीं कर सकती?
        \emph{सकारात्मक} प्रश्न जैसी अपसारी-पूंछ घटना
        2 का ब्रैकेट: वैकल्पिक त्रुटि दोलन करती है (चिह्न)।
        $(-1)^N $), इसलिए औसत इसके प्रमुख पद को रद्द कर देता है, जबकि
        द $\gamma $-ब्रैकेट त्रुटि $\frac{1}{2n}$ स्थिरांक है
        संकेत. (औसत $ a_n $ और $ b_n $ \emph{करता है} सहायता:
        संबंधित $\frac{a_n + b_n}{2}$ मध्यबिंदु अनुमान तक
        $ H_n - \ln\bigl(n + \frac12\bigr)$ और दिखाओ इसकी त्रुटि है
        $ O\bigl(\frac{1}{n^2}\bigr)$.)
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Rigidity and its failure.}
\begin{enumerate}[resume]
  \item वह सकारात्मक भाग दिखाएँ $\sum \frac{1}{2j-1}$ और
        नकारात्मक भाग $\sum \frac{1}{2j}$ प्रत्यावर्तन का
        हार्मोनिक \omterm{def:b1:series:def}{शृंखला} दोनों अलग-अलग हैं --- के हस्ताक्षर
        \emph{सशर्त} अभिसरण.
  \item सामान्य सिद्ध करो \omterm{def:b1:logic:statement}{कथन} प्रश्न 12 के पीछे: यदि
        $\sum u_n $ अभिसरण लेकिन $\sum \abs{u_n}$ फिर, अलग हो जाता है
        द \omterm{def:b1:series:def}{शृंखला} सकारात्मक भागों का $\sum u_n^+$ और का
        नकारात्मक भाग $\sum u_n^-$ दोनों अलग हो जाते हैं \emph{(से
        $ u_n^\pm = \frac{\abs{u_n} \pm u_n}{2}$: यदि एक
        अभिसरण, इसलिए दूसरा, इसलिए $\sum\abs{u_n}$)}.
        सकारात्मक और नकारात्मक का यह अक्षय भंडार
        द्रव्यमान वही है जो रीमैन का नुस्खा खर्च करेगा।
  \item (कठोरता) सिद्ध करें: यदि $\sum u_n $ अभिसरण
        \emph{बिल्कुल} और $\sigma \colon \N \to \N $ एक है
        आपत्ति, तो $\sum u_{\sigma(n)}$ में परिवर्तित हो जाता है
        वही राशि \emph{(के लिए $ N $ सबसे पहले बड़ा $ M $ पुनर्व्यवस्थित
        शर्तों में शामिल हैं $ u_0, \dots, u_N $; आंशिक योगों की तुलना करें
        पूंछ के माध्यम से $\sum_{n > N}\abs{u_n}$)}.
  \item (रीमैन की रेसिपी) चलो $ t \in \R $. लालची का वर्णन करें
        प्रत्यावर्ती हार्मोनिक की पुनर्व्यवस्था \omterm{def:b1:series:def}{शृंखला}: ले लो
        सकारात्मक शर्तें $1, \frac13, \frac15, \dots $ जब तक
        आंशिक योग पहले से अधिक है $ t $, फिर नकारात्मक शब्द
        $-\frac12, -\frac14, \dots $ जब तक यह पहले नीचे न गिर जाए
        $ t $, और दोहराएँ. दिखाएँ कि प्रत्येक शब्द का सटीक प्रयोग किया गया है
        एक बार, वह पहली बार आंशिक रकम को पार करने के बाद
        के अंतिम प्रयुक्त शब्द के भीतर रहें $ t $, और निष्कर्ष निकालो
        पुनर्व्यवस्थित \omterm{def:b1:series:def}{शृंखला} में एकत्रित हो जाता है $ t $: \emph{कोई}
        निर्धारित राशि प्राप्य है.
\end{enumerate}

\medskip
\noindent\textbf{Part V --- The $(p, q)$ formula.} पूर्णांकों को ठीक करें
$ p, q \geq 1$. प्रत्यावर्ती हार्मोनिक को पुनर्व्यवस्थित करें \omterm{def:b1:series:def}{शृंखला} में
ब्लॉक: $ p $ फिर सकारात्मक पद (अगले विषम व्युत्क्रम)। $ q $
नकारात्मक पद (अगले सम व्युत्क्रम), और दोहराएँ।
\begin{enumerate}[resume]
  \item जांचें कि यह एक वास्तविक पुनर्व्यवस्था है (प्रत्येक पद)।
        बिल्कुल एक बार), और उसके लिए $(p, q) = (1, 2)$ यह पढ़ता है
        \[
        1 - \frac12 - \frac14 + \frac13 - \frac16 - \frac18 +
        \frac15 - \frac1{10} - \frac1{12} + \dots
        \]
  \item (सटीक पड़ाव) के लिए $(p, q) = (1, 2)$, सिद्ध करो
        ब्लॉक पहचान
        \[
        \frac{1}{2k-1} - \frac{1}{4k-2} - \frac{1}{4k}
        = \frac12\Bigl(\frac{1}{2k-1} - \frac{1}{2k}\Bigr),
        \]
        और निष्कर्ष निकालें \emph{एकदम सही} रिश्ता $ T_{3K} = \frac12
        S_{2K}$ पुनर्व्यवस्थित आंशिक योगों और के बीच
        मूल वाले: राशि का आधा होना दिखाई दे रहा है
        प्रत्येक परिमित अवस्था, केवल \omterm{def:b1:topology:closure}{सीमा} में नहीं।
  \item दिखाएँ कि आंशिक योग बाद में है $ K $ पूर्ण ब्लॉक
        बराबर $\sum_{j=1}^{pK} \frac{1}{2j-1} -
        \sum_{j=1}^{qK} \frac{1}{2j}$, और इसकी \omterm{def:b1:topology:closure}{सीमा} की गणना करें
        प्रश्न 7 के साथ:
        \[
        \ln 2 + \frac12 \ln\frac pq .
        \]
  \item आंशिक रकम पर नियंत्रण रखें \emph{अंदर} एक ब्लॉक (
        शर्तें प्रवृत्त होती हैं $0$) और निष्कर्ष निकालें कि $(p,
        q)$-पुनर्व्यवस्थित \omterm{def:b1:series:def}{शृंखला} में एकत्रित हो जाता है $\ln 2 + \frac12\ln
        \frac pq $. विशेष रूप से $(1, 2)$ देता है $\frac{\ln
        2}{2}$: पहले नौ पदों से सत्यापित करें, $ T_9 =
        0.3083$, की ओर रेंगना $0.3466$.
  \item विवेक जांच और \omterm{def:b1:topology:closure}{सीमा}: $(1,1)$ ठीक $\ln 2$;
        $(2,1)$ देता है $\frac32\ln 2$; कौन सी रकम पहुंच योग्य है
        द्वारा $(p, q)$-ब्लॉक, और यह गणनीय मेनू कैसे होता है
        रीमैन के पूर्ण कार्टे (प्रश्न 14) से तुलना करें?
\end{enumerate}

\medskip
\noindent\textbf{Part VI --- Epilogue: $\gamma $ at work, and
synthesis.}
\begin{enumerate}[resume]
  \item अभिसरण का योग पहचानें \omterm{def:b1:series:def}{शृंखला}
        $\sum_{k\geq1} \bigl(\frac1k - \ln\frac{k+1}{k}\bigr)$
        (\cref{exo:b1:series:7}): इसे बराबर दिखाएं $\gamma $.
  \item लक्ष्य के लिए रीमैन की रेसिपी (प्रश्न 14) चलाएँ $ t =
        1$ और निर्मित पहले बारह शब्दों की सूची बनाएं ($1, \frac13,
        -\frac12, \frac15, -\frac14, \frac17, \frac19, -\frac16,
        \frac1{11}, \frac1{13}, -\frac18, \frac1{15}$),
        आंशिक योग की गणना ($\approx 0.980$) ---देखो
        एल्गोरिदम अपने लक्ष्य के चारों ओर सांस लेता है।
  \item दिखाएँ कि प्रत्यावर्ती हार्मोनिक की कुछ पुनर्व्यवस्था
        \omterm{def:b1:series:def}{शृंखला} की ओर विमुख हो जाता है $+\infty $ \emph{(सकारात्मक के ब्लॉक
        हासिल करने के लिए शर्तें काफी लंबी हैं $1$ हर बार, प्रश्न का उपयोग करते हुए
        12, एकल नकारात्मक शब्दों से अलग)}.
  \item पैना \cref{ex:b1:series:harmonicstack} साथ
        $\gamma $-कानून: दिखाएँ कि पहला सूचकांक $ H_N \geq
        20$ संतुष्ट $ N = \eu^{\,20 - \gamma}\,(1 + o(1))
        \approx 2.7\cdot10^{8}$ --- \omterm{ex:b1:series:oddtelescope}{यूलर स्थिरांक} बिलकुल है
        सुधार में क्रूड फ़्रेमिंग गायब थी।
  \item संश्लेषण, प्रत्येक एक वाक्य: (i) द $\gamma $-कानून और
        इसके तीन टुकड़ों में से प्रत्येक क्या है ($\ln n $, $\gamma $,
        $\frac{1}{2n}$) योगदान देता है; (ii) सशर्त क्यों
        अभिसरण योग को क्रम-निर्भर बनाता है \omterm{thm:b1:series:absolute}{निरपेक्ष
        अभिसरण} इसे मना करता है; (iii) कैसे $(p,q)$ सूत्र
        एक था \emph{गणना} साथ $\gamma$-कानून बल्कि
        एक अमूर्त अस्तित्व के दावे की तुलना में; (iv) ये कहाँ हैं
        धागे जारी हैं --- शक्ति \omterm{def:b1:series:def}{शृंखला} और के उत्पाद \omterm{def:b1:series:def}{शृंखला}
        वर्ष 2 खंड में, और वर्ष 3 खंड का सप्ताहांत
        स्टर्लिंग के फार्मूले पर समस्या, जहां वही है
        योग-बना\omterm{thm:b1:integration:def}{म-अभिन्न} बहीखाता पद्धति पूरी शक्ति से चलती है।
\end{enumerate}
\end{problem}
