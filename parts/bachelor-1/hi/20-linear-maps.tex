\chapter{रेखीय मानचित्र}\label{ch:b1:linmaps}

 एमएपीएस के बीच अध्ययन करने लायक है वेक्टर रिक्त स्थान क्या वे संगत हैं
संरचना के साथ: \emph{\omterm{def:b1:linmaps:def}{रेखीय} मानचित्र}. उनके दो मौलिक
उपस्थान --- \omterm{def:b1:linmaps:kerim}{कर्नेल} और \omterm{def:b1:linmaps:kerim}{छवि} --- इंजेक्टिविटी को मापें और
प्रक्षेपात्मकता, और में परिमित आयाम रैंक--शून्यता प्रमेय
उनके आकार को एक संरक्षण कानून में बांधता है। अनुमान और
\omterm{def:b1:linmaps:projection}{समरूपता}, फिर \omterm{def:b1:linmaps:forms}{रैखिक रूप} और \omterm{def:b1:linmaps:forms}{हाइपरप्लेन}, अध्याय \omterm{def:b1:topology:closure}{बंद} करें.

\section{परिभाषाएँ और प्रथम गुण}

\begin{definition}\label{def:b1:linmaps:def}
होने देना $E, F $ होना $ K $-वेक्टर रिक्त स्थान. ए \omterm{def:b1:logic:map}{नक्शा} $ u \colon E \to F $ है
\emph{रेखीय}\index{रेखीय} कब
\[
\forall x, y \in E,\ \forall \lambda \in K, \qquad
u(x + \lambda y) = u(x) + \lambda u(y).
\]
तब $ u(0) = 0$ और $ u(\sum \lambda_i x_i) = \sum \lambda_i u(x_i)$.
 \omterm{def:b1:logic:sets}{तय करना} $\mathcal{L}(E, F)$ रैखिक का एमएपीएस स्वयं एक है सदिश स्थल;
रैखिक की रचना एमएपीएस जोड़ी में रैखिक और द्विरेखीय है। एक
\emph{एंडोमोर्फिज्म} एक रैखिक है $ E \to E $; एक \emph{समाकृतिकता} एक है \omterm{def:b1:logic:inj}{द्विभाजित} रेखीय \omterm{def:b1:logic:map}{नक्शा} (तब इसका व्युत्क्रम स्वचालित रूप से रैखिक होता है);
$ u^{-1}$ एक \omterm{def:b1:linmaps:projection}{समरूपता} की, \omterm{def:b1:linmaps:projection}{समरूपता} की रचनाएँ हैं
समरूपताएँ।
\end{definition}

\begin{proof}[प्रमाण है कि व्युत्क्रम रैखिक है]
चलो $ u $ होना \omterm{def:b1:linmaps:def}{रेखीय} \omterm{def:b1:logic:inj}{द्विभाजित}, $ y, y' \in F $ और $\lambda \in K $.
रखना $ x = u^{-1}(y)$ और $ x' = u^{-1}(y')$. तब
\[
u\bigl(x + \lambda x'\bigr) = u(x) + \lambda u(x')
= y + \lambda y' ,
\]
और आवेदन कर रहे हैं $ u^{-1}$ दोनों सिरों तक: $ u^{-1}(y + \lambda y') = x
+ \lambda x' = u^{-1}(y) + \lambda\,u^{-1}(y')$. के बारे में कुछ नहीं
$ u^{-1}$ गणना की गई: रैखिकता परिवहन के माध्यम से
की संपत्ति को परिभाषित करना $ u $ अकेले --- याद रखने लायक एक पैटर्न,
चूँकि संरचना अक्सर मुफ़्त में आपत्तियों के साथ चलती है।
\end{proof}

\begin{proposition}[रेखीय मानचित्र को एक आधार पर जाना जाता है]\label{prop:b1:linmaps:basis}
होने देना $(e_1, \dots, e_n)$ एक हो \omterm{def:b1:vspaces:free}{आधार} का $ E $ और $(v_1, \dots, v_n)$
के मनमाना वेक्टर $ F $. बिलकुल एक है \omterm{def:b1:linmaps:def}{रेखीय} मानचित्र $ u \colon
E \to F $ साथ $ u(e_i) = v_i $ सभी के लिए $ i $. इसके अतिरिक्त:
\[
u \text{ injective} \iff (v_i) \text{ free};
\qquad
u \text{ surjective} \iff (v_i) \text{ generates } F .
\]
\end{proposition}

\begin{proof}
अस्तित्व/अद्वितीयता: प्रत्येक $ x $ अद्वितीय है \omterm{prop:b1:vspaces:coordinates}{COORDINATES} $ x = \sum
\lambda_i e_i $ (\cref{prop:b1:vspaces:coordinates}); रैखिकता
बल $ u(x) = \sum \lambda_i v_i $, और यह सूत्र परिभाषित करता है a
\omterm{def:b1:linmaps:def}{रेखीय} मानचित्र.

इंजेक्टिविटी: द्वारा \cref{prop:b1:linmaps:kernel} नीचे, $ u $ है
\omterm{def:b1:logic:inj}{इंजेक्शन} यदि यह है \omterm{def:b1:linmaps:kerim}{कर्नेल} तुच्छ है. अब $ u\bigl(\sum\lambda_i
e_i\bigr) = 0$ बिलकुल मतलब $\sum\lambda_i v_i = 0$. अगर $(v_i)$
है \omterm{def:b1:vspaces:free}{मुक्त}, यह हर किसी को मजबूर करता है $\lambda_i = 0$, यानी\  \omterm{def:b1:linmaps:kerim}{कर्नेल}
तक कम कर देता है $0$: \omterm{def:b1:logic:inj}{इंजेक्शन}. अगर $(v_i)$ जुड़ा हुआ है, एक गैरतुच्छ
संबंध $\sum\lambda_i v_i = 0$ अशून्य वेक्टर उत्पन्न करता है
$\sum\lambda_i e_i $ में \omterm{def:b1:linmaps:kerim}{कर्नेल} ($(e_i)$ है \omterm{def:b1:vspaces:free}{मुक्त}): नहीं
\omterm{def:b1:logic:inj}{इंजेक्शन}. दोनों स्थितियाँ पद दर पद मेल खाती हैं।

\omterm{def:b1:logic:inj}{विशेषण}ात्मकता: द \omterm{def:b1:linmaps:kerim}{छवि} का $ u $ है \omterm{def:b1:logic:sets}{तय करना} के सभी $\sum\lambda_i
v_i $, यानी\ बिल्कुल $\operatorname{Vect}(v_1, \dots, v_n)$, जो
बराबर $ F $ यदि परिवार उत्पन्न करता है।
\end{proof}

\begin{definition}[कर्नेल और छवि]\label{def:b1:linmaps:kerim}
के लिए $ u \in \mathcal{L}(E, F)$:
\[
\ker u = \{x \in E : u(x) = 0\} \subseteq E,
\qquad
\operatorname{im} u = u(E) \subseteq F ,
\]
दोनों उपस्थान (के साथ सीधा सत्यापन
कसौटी).\index{कर्नेल}\index{छवि}
\end{definition}

\begin{method}[व्यवहार में कर्नेल और छवि]
\label{met:b1:linmaps:kerim}
\emph{गुठली}: लिखना $ u(x) = 0$ पर एक प्रणाली के रूप में \omterm{prop:b1:vspaces:coordinates}{COORDINATES}
(या गुणांक) का $ x $, हल करें, और पैरामीटराइज़ करें --- \omterm{def:b1:linmaps:kerim}{कर्नेल}
एक के साथ बाहर आता है \omterm{def:b1:vspaces:free}{आधार} जुड़ा हुआ
(\cref{met:b1:findim:computedim}). \emph{\omterm{def:b1:linmaps:kerim}{छवि}}: यह है \omterm{def:b1:vspaces:span}{अवधि}
की इमेजिस का \emph{कोई} का सृजनशील परिवार $ E $ ---
आमतौर पर ए \omterm{def:b1:vspaces:free}{आधार}, इसलिए $\operatorname{im} u =
\operatorname{Vect}\bigl(u(e_1), \dots, u(e_n)\bigr)$; फिर
अनावश्यक को हटा दें इमेजिस एक निकालने के लिए \omterm{def:b1:vspaces:free}{आधार}. \emph{शॉर्टकट}:
दोनों में से जो भी आसान हो उसकी गणना करें और दूसरे का पता लगाएं
\omterm{def:b1:linmaps:rank}{रैंक} द्वारा निःशुल्क आयाम--शून्यता
(\cref{thm:b1:linmaps:ranknullity}); जब एक विश्वसनीय उम्मीदवार
\omterm{def:b1:vspaces:subspace}{उपस्पेस} के लिए \omterm{def:b1:linmaps:kerim}{छवि} आयाम उन्नयन की तुलना करने पर ज्ञात होता है
समानता में आसान समावेश
(\cref{thm:b1:findim:subspaces}). दोनों शॉर्टकट का उपयोग किया जाता है
\cref{ex:b1:linmaps:delta} नीचे।
\end{method}

\begin{proposition}\label{prop:b1:linmaps:kernel}
$ u $ है \omterm{def:b1:logic:inj}{इंजेक्शन} $\iff $ $\ker u = \{0\}$; $ u $ है \omterm{def:b1:logic:inj}{विशेषण} $\iff $
$\operatorname{im} u = F $.
\end{proposition}

\begin{proof}
से संबंधित \omterm{def:b1:structures:group}{समूह} (\cref{prop:b1:structures:kernel}): $ u(x) = u(y) \iff
u(x - y) = 0 \iff x - y \in \ker u $. दूसरा बिंदु है
परिभाषा.
\end{proof}

\section{रैंक--शून्यता प्रमेय}

\begin{definition}\label{def:b1:linmaps:rank}
 \emph{रैंक}\index{रैंक} का $ u \in \mathcal{L}(E,
F)$ (साथ $ E $ \omterm{def:b1:findim:def}{परिमित-आयामी}) है $\operatorname{rk} u = \dim
\operatorname{im} u $ --- परिवार का पद भी
$\bigl(u(e_1), \dots, u(e_n)\bigr)$ किसी के लिए \omterm{def:b1:vspaces:free}{आधार} $(e_i)$ का $ E $.
\end{definition}

\begin{theorem}[पद--शून्यता]\label{thm:b1:linmaps:ranknullity}
होने देना $ E $ होना \omterm{def:b1:findim:def}{परिमित-आयामी} और $ u \in \mathcal{L}(E, F)$.
तब\index{रैंक--शून्यता प्रमेय}
\[
\dim E = \dim \ker u + \operatorname{rk} u .
\]
अधिक सटीक रूप से, यदि $ S $ कोई है \omterm{def:b1:vspaces:sum}{पूरक} \omterm{def:b1:vspaces:subspace}{उपस्पेस} का $\ker u $ में
$ E $, तब $ u $ एक तक सीमित है \emph{समाकृतिकता} से $ S $ पर
$\operatorname{im} u $.
\end{theorem}

\begin{proof}
होने देना $ S $ संतुष्ट $ E = \ker u \oplus S $
(\cref{thm:b1:findim:subspaces}), और चलो $ v \colon S \to
\operatorname{im} u $ का प्रतिबंध हो $ u $.

$ v $ है \omterm{def:b1:logic:inj}{इंजेक्शन}: $\ker v = S \cap \ker u = \{0\}$.

$ v $ है \omterm{def:b1:logic:inj}{विशेषण}: कोई $ u(x)$ साथ $ x = k + s $ ($ k \in \ker u $, $ s
\in S $) बराबर है $ u(s) = v(s)$.

इसलिए $ v $ एक \omterm{def:b1:linmaps:projection}{समरूपता} है; एक \omterm{def:b1:linmaps:projection}{समरूपता} एक भेजता है \omterm{def:b1:vspaces:free}{आधार} एक को \omterm{def:b1:vspaces:free}{आधार}
(\cref{prop:b1:linmaps:basis}), इसलिए $\dim S = \dim\operatorname{im}
u $, और $\dim E = \dim\ker u + \dim S $ निष्कर्ष.
\end{proof}

\begin{example}[विशिष्टता के अनुसार मानचित्र बनाना]
\label{ex:b1:linmaps:tospec}
निर्माण $ u \in \mathcal{L}(\R^3)$ साथ $\ker u =
\operatorname{Vect}(1,1,1)$ और $\operatorname{im} u = \{z =
0\}$. विवेक पहले: रैंक--शून्यता की मांग $1 + 2 = 3$ ---
सुसंगत, इसलिए समाधान मौजूद हो सकता है। एक विकल्प चुनें \omterm{def:b1:vspaces:free}{आधार} के लिए अनुकूलित
द \omterm{def:b1:linmaps:kerim}{कर्नेल}, कहना $\bigl((1,1,1),\ e_1,\ e_2\bigr)$
(\cref{ex:b1:findim:completion}), और निर्धारित करें इमेजिस
(\cref{prop:b1:linmaps:basis}):
\[
u(1,1,1) = 0, \qquad u(e_1) = e_1, \qquad u(e_2) = e_2 .
\]
तब $\ker u \supseteq \operatorname{Vect}(1,1,1)$ और
$\operatorname{im} u = \operatorname{Vect}(e_1, e_2) = \{z =
0\}$; रैंक--शून्यता ताकतें $\dim\ker u = 1$, इतना \omterm{def:b1:linmaps:kerim}{कर्नेल} है
बिल्कुल निर्धारित लाइन. स्पष्ट रूप से, विघटनकारी $(x, y, z)
= z(1,1,1) + (x - z)e_1 + (y - z)e_2$:
\[
u(x, y, z) = (x - z,\ y - z,\ 0).
\]
नुस्खा सामान्यीकरण करता है: a \omterm{def:b1:linmaps:def}{रेखीय} मानचित्र निर्धारित के साथ \omterm{def:b1:linmaps:kerim}{कर्नेल} $ N $
और \omterm{def:b1:linmaps:kerim}{छवि} $ I $ ठीक उसी समय मौजूद है $\dim N + \dim I = \dim E $
---आवश्यकता पद है--शून्यता, पर्याप्तता यह है
निर्माण.
\end{example}

\begin{corollary}\label{cor:b1:linmaps:samedim}
अगर $\dim E = \dim F $ (परिमित), फिर के लिए $ u \in \mathcal{L}(E, F)$:
\[
u \text{ injective} \iff u \text{ surjective} \iff u \text{
bijective}.
\]
विशेष रूप से यह एंडोमोर्फिज्म के लिए लागू होता है परिमित आयाम. (यह
अनंत आयाम में विफल: पर $ K[X]$, द यौगिक है \omterm{def:b1:logic:inj}{विशेषण}
लेकिन नहीं \omterm{def:b1:logic:inj}{इंजेक्शन}, और $ P \mapsto XP $ है \omterm{def:b1:logic:inj}{इंजेक्शन} लेकिन नहीं
\omterm{def:b1:logic:inj}{विशेषण}.)
\end{corollary}

\begin{proof}
\omterm{def:b1:logic:inj}{इंजेक्शन} $\iff \dim\ker u = 0 \iff \operatorname{rk} u = \dim E =
\dim F \iff \operatorname{im} u = F $ (ए \omterm{def:b1:vspaces:subspace}{उपस्पेस} पूर्ण आयाम का
सब कुछ है, \cref{thm:b1:findim:subspaces}) $\iff $ \omterm{def:b1:logic:inj}{विशेषण}.
\end{proof}

\begin{example}[अंतर्वेशन, संरचनात्मक रूप से]\label{ex:b1:linmaps:interpolation}
अलग ठीक करें $ x_0, \dots, x_n $ और चलो $ u \colon \R_n[X] \to
\R^{n+1}$, $ P \mapsto (P(x_0), \dots, P(x_n))$: \omterm{def:b1:linmaps:def}{रेखीय}. इसका \omterm{def:b1:linmaps:kerim}{कर्नेल}
है $\{P : \deg P \leq n,\ n+1 \text{ roots}\} = \{0\}$
(\cref{cor:b1:poly:nroots}). समान आयाम $ n + 1$: $ u $ एक है
\omterm{def:b1:linmaps:projection}{समरूपता} --- अस्तित्व \emph{और} लैग्रेंज की विशिष्टता
इंटरपोलेंट (\cref{thm:b1:poly:lagrange}) एक पंक्ति में.

वही एक-पंक्ति पैटर्न डेटा मिश्रण मानों को संभालता है और
डेरिवेटिव:
\[
  v \colon \R_3[X] \to \R^4,\quad
  P \mapsto \bigl(P(0), P'(0), P(1), P'(1)\bigr)
\]
$v$ है \omterm{def:b1:linmaps:def}{रेखीय}, और इसके \omterm{def:b1:linmaps:kerim}{कर्नेल} में
डिग्री-$\leq 3$ बहुपद दोहरी जड़ों के साथ $0$ \emph{और}
$1$ हैं, यानी $X^2(X-1)^2$ (डिग्री $4$) से विभाज्य: केवल $P =
0$. फिर से समान आयाम: डेटा का प्रत्येक चौगुना $(P(0),
P'(0), P(1), P'(1))$ ठीक एक घन से एहसास होता है ---
हर्मिट इंटरपोलेशन, एक \omterm{def:b1:linmaps:kerim}{कर्नेल} गणना द्वारा पहले
किसी भी सूत्र के लिखे जाने से (सप्ताहांत समस्या \cref{ch:b1:det}
इसके निर्धारक से मिलती है)।
\end{example}

\begin{example}[रैंक--काम पर शून्यता: अंतर ऑपरेटर]
\label{ex:b1:linmaps:delta}
होने देना $\Delta \colon \R_n[X] \to \R_n[X]$, $ P \mapsto P(X+1) -
P(X)$: \omterm{def:b1:linmaps:def}{रेखीय}. गुठली: अगर $\Delta P = 0$, तब $ P(0) = P(1) =
P(2) = \dots $, इसलिए $ P - P(0)$ अनंत रूप से कई जड़ें हैं और
गायब हो जाता है (\cref{cor:b1:poly:nroots}): $\ker\Delta $ की पंक्ति है
स्थिरांक रैंक--शून्यता: $\operatorname{rk}\Delta = (n + 1) - 1
= n $. तब से $\deg \Delta P < \deg P $ अस्थिर के लिए $ P $ (शीर्ष
शर्तें रद्द करें), $\operatorname{im}\Delta \subseteq \R_{n-1}[X]$,
जिसका सटीक आयाम है $ n $: समावेशन एक समानता है.
बिना किसी गणना के निष्कर्ष पूर्वछवियाँ: \emph{प्रत्येक}
\omterm{def:b1:poly:def}{बहुपद} $ Q $ डिग्री का $\leq n - 1$ एक अंतर है $ Q = P(X+1)
- P(X)$ --- असतत प्रतिअवकलन मौजूद है। (तुलना करें
सप्ताहांत की समस्या \cref{ch:b1:vspaces}, कहाँ $\Delta $ था
द्विपद में स्पष्ट रूप से उलटा \omterm{def:b1:vspaces:free}{आधार}.)
\end{example}

\begin{example}[किसी रचना के अनुसार बहीखाता पद्धति को क्रमबद्ध करें]
\label{ex:b1:linmaps:rankbookkeeping}
पर $\R_2[X]$, रचना करें यौगिक $ D(P) = P'$ (\omterm{def:b1:linmaps:rank}{रैंक} $2$:
\omterm{def:b1:linmaps:kerim}{छवि} $\R_1[X]$, \omterm{def:b1:linmaps:kerim}{कर्नेल} स्थिरांक) स्वयं के साथ। तब $ D
\circ D = D^2$ एमएपीएस $ P \mapsto P''$, साथ \omterm{def:b1:linmaps:kerim}{छवि} $\R_0[X]$: \omterm{def:b1:linmaps:rank}{रैंक}
$1$. सामान्य \omterm{def:b1:topology:closure}{सीमा} से तुलना करें: अपरिष्कृत \omterm{def:b1:topology:closure}{सीमा} देता है
$\operatorname{rk} D^2 \leq \min(2, 2) = 2$; सटीक सूत्र
का \cref{exo:b1:linmaps:12} हानि का सटीक हिसाब लगाता है,
\[
\operatorname{rk} D^2 = \operatorname{rk} D -
\dim\bigl(\ker D \cap \operatorname{im} D\bigr) = 2 - 1 = 1 ,
\]
स्थिरांक के बाद से (\omterm{def:b1:linmaps:kerim}{कर्नेल} बाहरी का $ D $) अंदर बैठो
$\R_1[X]$ (\omterm{def:b1:linmaps:kerim}{छवि} भीतर का $ D $) आयाम के साथ $1$. पद है
बिल्कुल वहीं खो गया जहां बाहरी है \omterm{def:b1:linmaps:kerim}{कर्नेल} भीतर घात करता है \omterm{def:b1:linmaps:kerim}{छवि}
--- रचना करते समय याद रखने योग्य वाक्य \omterm{def:b1:linmaps:rank}{रैंक} दुर्व्यवहार करना.
\end{example}

\begin{example}[यूलर ऑपरेटर की कर्नेल और छवि]
\label{ex:b1:linmaps:euler}
पर $\R_n[X]$, होने देना $ u(P) = X\,P'$ (\omterm{def:b1:linmaps:def}{रेखीय}: भेदभाव और
से गुणा $ X $ हैं)। \emph{गुठली}: $ XP' = 0$ बल $ P' =
0$ (का एक उत्पाद बहुआयामी पद केवल एक कारक गायब होने पर गायब हो जाता है),
तो $\ker u $ स्थिरांक की रेखा है. \emph{\omterm{def:b1:linmaps:kerim}{छवि}}: पर
एकपदी \omterm{def:b1:vspaces:free}{आधार},
\[
u(X^k) = k\,X^{k} \qquad (k = 0, 1, \dots, n),
\]
इसलिए $\operatorname{im} u = \operatorname{Vect}(X, 2X^2, \dots,
nX^n) = \operatorname{Vect}(X, X^2, \dots, X^n)$: द
बहुआयामी पद शून्य स्थिर पद के साथ. रैंक--शून्यता के विरुद्ध जाँच करें:
$\operatorname{rk} u = (n + 1) - 1 = n $, जो वास्तव में है
आयाम मिला. दो टिप्पणियाँ ध्यान देने योग्य हैं। सबसे पहले, यहाँ
$\operatorname{im} u \oplus \ker u = \R_n[X]$ --- लेकिन वह एक है
\emph{सुखद दुर्घटना} इस ऑपरेटर का, प्रमेय नहीं: के लिए
बदलाव की तरह $ v(P) = P'$ पर $\R_1[X]$, $\ker v = \operatorname{im}
v = \R_0[X]$ और योग \omterm{def:b1:vspaces:sum}{प्रत्यक्ष} नहीं है. दूसरा, संबंध
$ u(X^k) = kX^k $ कहता है कि प्रत्येक एकपदी का केवल पुनर्मूल्यांकन किया जाता है $ u $ ---
ए \omterm{def:b1:vspaces:free}{आधार} के लिए अनुकूलित \omterm{def:b1:logic:map}{नक्शा}, eigenvalue विचार का रोगाणु
वर्ष~2 खंड में विकसित किया गया।
\end{example}

\section{प्रक्षेपण और समरूपता}

\begin{definition}\label{def:b1:linmaps:projection}
होने देना $ E = F \oplus G $. \emph{पर प्रक्षेपण $ F $ साथ में
$ G $}\index{पर $ F $ साथ में
$ जी $} एमएपीएस $ x = f + g $ (अद्वितीय अपघटन) को
$ p(x) = f $; जुड़े \emph{समरूपता}\index{समरूपता} है $ s(x)
= f - g $. दोनों हैं \omterm{def:b1:linmaps:def}{रेखीय}, और $ s = 2p - \mathrm{id}$.
\end{definition}

\begin{theorem}[बीजगणितीय लक्षण वर्णन]\label{thm:b1:linmaps:projchar}
\begin{enumerate}
  \item एक एंडोमोर्फिज्म $ p $ एक है अनुमान (कुछ पर $ F $ साथ में
        कुछ $ G $) अगर और केवल अगर $ p \circ p = p $; तब $ F =
        \operatorname{im} p = \ker(p - \mathrm{id})$ और $ G = \ker
        p $.
  \item एक एंडोमोर्फिज्म $ s $ एक \omterm{def:b1:linmaps:projection}{समरूपता} है यदि और केवल यदि $ s \circ s
        = \mathrm{id}$; तब $ E = \ker(s - \mathrm{id}) \oplus
        \ker(s + \mathrm{id})$.
\end{enumerate}
\end{theorem}

\begin{proof}
(1) ए अनुमान संतुष्ट $ p(f + g) = f $ और $ p(f) = f $: $ p^2 = p $.
इसके विपरीत, चलो $ p^2 = p $; \omterm{def:b1:logic:sets}{तय करना} $ F = \operatorname{im} p $, $ G = \ker
p $. प्रत्येक $ x $ लिखते हैं $ x = p(x) + (x - p(x))$ साथ $ p(x) \in F $ और
$ p\bigl(x - p(x)\bigr) = p(x) - p^2(x) = 0$: $ E = F + G $. अगर $ y \in
F \cap G $: $ y = p(z)$ और $ p(y) = 0$, इसलिए $ y = p(z) = p^2(z) = p(y) =
0$: \omterm{def:b1:vspaces:sum}{प्रत्यक्ष} योग, और $ p $ है अनुमान पर $ F $ साथ में $ G $.
अंततः चालू $ F $: $ y = p(z)$ देता है $ p(y) = y $, इसलिए $ F \subseteq
\ker(p - \mathrm{id})$, और इसके विपरीत $ p(y) = y $ डालता है $ y $ में
\omterm{def:b1:linmaps:kerim}{छवि}.

(2) पत्र व्यवहार $ s = 2p - \mathrm{id}$, $ p = \frac{s +
\mathrm{id}}2$ एंडोमोर्फिज्म और इसके अंतर्गत एक आक्षेप है
\[
s^2 = 4p^2 - 4p + \mathrm{id} = \mathrm{id}
\iff 4p^2 = 4p \iff p^2 = p :
\]
\omterm{def:b1:linmaps:projection}{समरूपता} बिल्कुल मेल खाती है अनुमान. का अनुवाद कर रहा हूँ
उपस्थान: $ s(x) = x \iff p(x) = x $, इसलिए $\ker(s - \mathrm{id}) =
\operatorname{im} p = F $; और $ s(x) = -x \iff 2p(x) = 0 \iff x
\in \ker p = G $, इसलिए $\ker(s + \mathrm{id}) = G $. \omterm{def:b1:vspaces:sum}{प्रत्यक्ष} योग
$ E = F \oplus G $ बिंदु (1) का घोषित अपघटन बन जाता है
निश्चित सदिशों और उलटे सदिशों में $ s $.
\end{proof}

\begin{example}[एक प्रक्षेपण और इसकी समरूपता, स्पष्ट रूप से]
\label{ex:b1:linmaps:projexplicit}
में $\R^2$, पर प्रोजेक्ट करें $ F = \operatorname{Vect}(1,1)$ साथ में $ G =
\operatorname{Vect}(0,1)$. विघटित $(x, y) = a(1,1) + b(0,1)$:
पहला निर्देशांक देता है $ a = x $, दूसरा $ b = y - x $.
इस तरह
\[
p(x, y) = (x, x),
\qquad
s(x, y) = 2p(x,y) - (x,y) = (x,\ 2x - y).
\]
बीजगणित की जाँच करें: $ p(p(x,y)) = p(x,x) = (x,x)$, और $ s(s(x,y)) =
s(x, 2x - y) = (x, 2x - (2x - y)) = (x, y)$. ज्यामितीय रूप से, $ s $
रेखा के पार ``तिरछा प्रतिबिंब'' है $ y = x $ में
ऊर्ध्वाधर दिशा: यह ठीक करता है $ F $ बिंदुवार और उलटा $ G $. था
हमने उसी पर प्रक्षेपित किया $ F $ साथ में $ G' =
\operatorname{Vect}(1,-1)$ इसके बजाय, सूत्र बदल जाएगा
$ p'(x,y) = \bigl(\frac{x+y}2, \frac{x+y}2\bigr)$: ए अनुमान है
इसके द्वारा निर्धारित किया जाता है \omterm{def:b1:linmaps:kerim}{छवि} \emph{और} इसका \omterm{def:b1:linmaps:kerim}{कर्नेल}, द्वारा कभी नहीं \omterm{def:b1:linmaps:kerim}{छवि}
अकेला।
\end{example}

\section{रैखिक रूप और हाइपरप्लेन}

\begin{definition}\label{def:b1:linmaps:forms}
ए \emph{रैखिक रूप}\index{रैखिक रूप} पर $ E $ एक है \omterm{def:b1:linmaps:def}{रेखीय} मानचित्र
$\varphi \colon E \to K $. ए \emph{हाइपरप्लेन}\index{हाइपरप्लेन} का
$ E $ ($\dim E = n $) एक है \omterm{def:b1:vspaces:subspace}{उपस्पेस} आयाम का $ n - 1$.
\end{definition}

\begin{example}[एक मूल्यांकन प्रपत्र और उसका हाइपरप्लेन]
\label{ex:b1:linmaps:evaluationform}
पर $\R_2[X]$, मूल्यांकन $\varphi(P) = P(2)$ एक है रैखिक
प्रपत्र, अशून्य ($\varphi(1) = 1$). इसका \omterm{def:b1:linmaps:kerim}{कर्नेल} है \omterm{def:b1:linmaps:forms}{हाइपरप्लेन}
का बहुआयामी पद पर लुप्त हो रहा है $2$, यानी\ (कारक प्रमेय,
\cref{thm:b1:poly:factor}) के गुणज $ X - 2$ अंदर
$\R_2[X]$:
\[
\ker\varphi = \operatorname{Vect}\bigl(X - 2,\ X(X - 2)\bigr),
\qquad \dim = 2 .
\]
में \omterm{prop:b1:vspaces:coordinates}{COORDINATES} पर $(1, X, X^2)$, $\varphi(a + bX + cX^2) = a +
2b + 4c $: प्रत्येक \omterm{def:b1:linmaps:forms}{रैखिक रूप} एक पर \omterm{def:b1:findim:def}{परिमित-आयामी} अंतरिक्ष है,
एक बार ए \omterm{def:b1:vspaces:free}{आधार} निश्चित है, में एक निश्चित रैखिक अभिव्यक्ति
\omterm{prop:b1:vspaces:coordinates}{COORDINATES} --- फॉर्म ``पंक्ति वेक्टर'' हैं, जैसे
\cref{ch:b1:matrices} शाब्दिक बना देगा, और गुणांक
यहाँ पंक्ति, $(1, 2, 4)$, एक वेंडरमोंडे पंक्ति है: मूल्यांकन प्रपत्र
का प्रक्षेप सिद्धांत कैसा है \cref{ch:b1:det}का सप्ताहांत
समस्या रैखिक बीजगणित में प्रवेश करती है।
\end{example}

\begin{theorem}\label{thm:b1:linmaps:hyperplanes}
 \omterm{def:b1:linmaps:forms}{हाइपरप्लेन} का $ E $ बिल्कुल वही हैं \omterm{def:b1:linmaps:kerim}{कर्नेल} शून्येतर का रैखिक
प्रपत्र. दो अशून्य रूप समान हैं \omterm{def:b1:linmaps:kerim}{कर्नेल} यदि और केवल यदि वे
आनुपातिक हैं.
\end{theorem}

\begin{proof}
अगर $\varphi \neq 0$: $\operatorname{rk}\varphi = 1$ (\omterm{def:b1:linmaps:kerim}{छवि} एक शून्येतर
\omterm{def:b1:vspaces:subspace}{उपस्पेस} का $ K $), इसलिए $\dim\ker\varphi = n - 1$: ए \omterm{def:b1:linmaps:forms}{हाइपरप्लेन}.
इसके विपरीत, चलो $ H $ एक हो \omterm{def:b1:linmaps:forms}{हाइपरप्लेन}, $(e_1, \dots, e_{n-1})$ ए \omterm{def:b1:vspaces:free}{आधार}
का $ H $ द्वारा पूर्ण की गयी $ e_n $: प्रपत्र ``अंतिम निर्देशांक'' है \omterm{def:b1:linmaps:kerim}{कर्नेल}
$ H $.

आनुपातिक रूप उनका साझा करते हैं \omterm{def:b1:linmaps:kerim}{कर्नेल}. इसके विपरीत, मान लीजिए
$\ker\varphi = \ker\psi = H $ और चुनें $ a \notin H $: प्रत्येक $ x $ लिखते हैं
$ x = h + \lambda a $ (जैसा $ E = H \oplus Ka $), और
\[
\varphi(x) = \lambda \varphi(a), \qquad \psi(x) = \lambda\psi(a):
\]
इसलिए $\varphi = \frac{\varphi(a)}{\psi(a)}\,\psi $.
\end{proof}

\begin{example}\label{ex:b1:linmaps:hyperplane}
में $ K^n $, ए \omterm{def:b1:linmaps:forms}{हाइपरप्लेन} एक समाधान है \omterm{def:b1:logic:sets}{तय करना} $\{a_1 x_1 + \dots + a_n
x_n = 0\}$ सबके साथ नहीं $ a_i $ शून्य --- ए का परिचित समीकरण
मूल के माध्यम से विमान $\R^3$. फ़ंक्शन स्थानों में, मूल्यांकन
प्रपत्र $ P \mapsto P(1)$ या $ f \mapsto \int_0^1 f $ परिभाषित करना \omterm{def:b1:linmaps:forms}{हाइपरप्लेन}
का $\R_n[X]$, का $ C(\intcc{0}{1})$ (सीएफ.\ \cref{exo:b1:findim:6}).
\end{example}

\begin{example}[एक हाइपरप्लेन, तीन तरह से काम करता था]
\label{ex:b1:linmaps:hyperplanerun}
लेना $\varphi(x, y, z) = x - 2y + 3z $ पर $\R^3$ और $ H =
\ker\varphi $. \emph{\omterm{def:b1:vspaces:free}{आधार}}: हल करना $ x = 2y - 3z $:
\[
(2y - 3z,\ y,\ z) = y\,(2, 1, 0) + z\,(-3, 0, 1),
\]
दो \omterm{def:b1:vspaces:free}{मुक्त} वेक्टर: $\dim H = 2$, ए \omterm{def:b1:linmaps:forms}{हाइपरप्लेन}, जैसा
\cref{thm:b1:linmaps:hyperplanes} से भविष्यवाणी करता है $\varphi \neq
0$. \emph{\omterm{def:b1:vspaces:sum}{पूरक} रेखा}: कोई भी वेक्टर बाहर $ H $ तक फैला एक,
जैसे\ $ a = (1, 0, 0)$ ($\varphi(a) = 1 \neq 0$); द
एक मनमाना का अपघटन $ v $ स्पष्ट है:
\[
v = \underbrace{\bigl(v - \varphi(v)\,a\bigr)}_{\in\,H}
+ \underbrace{\varphi(v)\,a}_{\in\,\operatorname{Vect}(a)},
\]
तब से $\varphi\bigl(v - \varphi(v)a\bigr) = \varphi(v) -
\varphi(v)\varphi(a) = 0$. \emph{समानता}: अगर $\psi(x,y,z)
= -2x + 4y - 6z $, तब $\psi = -2\varphi $ और दोनों के पास है \omterm{def:b1:linmaps:kerim}{कर्नेल}
$ H $; इसके विपरीत, कोई भी रूप लुप्त हो रहा है $ H $ का गुणज है
$\varphi $ (\cref{exo:b1:linmaps:8}) --- ए का समीकरण
\omterm{def:b1:linmaps:forms}{हाइपरप्लेन} पैमाने की दृष्टि से अद्वितीय है, एक ऐसा तथ्य जिसके लिए लगातार उपयोग किया जाता है
ज्यामिति में विमान.
\end{example}

\begin{remark}[सामान्य ख़तरे]
\label{rem:b1:linmaps:pitfalls}
\emph{गुठली और \omterm{def:b1:linmaps:kerim}{छवि} विभिन्न स्थानों में रहते हैं}: $\ker u
\subseteq E $, $\operatorname{im} u \subseteq F $; योग $\ker u
+ \operatorname{im} u $ केवल एंडोमोर्फिज्म के लिए समझ में आता है, और
फिर भी इसका \omterm{def:b1:vspaces:sum}{प्रत्यक्ष} होना आवश्यक नहीं है ($ u(x, y) = (y, 0)$ है $\ker u
= \operatorname{im} u $; \cref{exo:b1:linmaps:7} विशेषताएँ
जब प्रत्यक्षता कायम रहती है)।
\emph{$ u^2 = 0$ यह अर्थ नहीं $ u = 0$}: जो उसी $ u(x,y) =
(y, 0)$ वर्गों को गायब किए बिना शून्य तक --- क्या $ u^2 = 0$
वास्तव में कहता है $\operatorname{im} u \subseteq \ker u $
(\cref{exo:b1:linmaps:5}).
\emph{\omterm{def:b1:logic:inj}{इंजेक्शन} $\iff $ \omterm{def:b1:logic:inj}{विशेषण} बराबर की जरूरत है परिमित
आयाम}: पर $ K[X]$, द यौगिक है \omterm{def:b1:logic:inj}{विशेषण} और नहीं
\omterm{def:b1:logic:inj}{इंजेक्शन}, $ P \mapsto XP $ \omterm{def:b1:logic:inj}{इंजेक्शन} और नहीं \omterm{def:b1:logic:inj}{विशेषण}
(\cref{cor:b1:linmaps:samedim}); और के रिक्त स्थान के बीच
\emph{अलग} आयाम, एक निहितार्थ सरल है
असंभव ($\operatorname{rk} u \leq \min(\dim E, \dim F)$).
\emph{निर्धारण इमेजिस ए पर काम करता है \omterm{def:b1:vspaces:free}{आधार}, किसी परिवार पर नहीं}:
बहुत अपेक्षाएँ रखने वाला $ u(1, 0) = a $, $ u(0, 1) = b $, $ u(1, 1) = c $
अति निर्धारित करता है $ u $ जब तक $ c = a + b $; ए \omterm{def:b1:linmaps:def}{रेखीय} मानचित्र है \omterm{def:b1:vspaces:free}{मुक्त} एक पर
\omterm{def:b1:vspaces:free}{आधार}, हर जगह गुलाम बना लिया गया।
\emph{पद रचना द्वारा संरक्षित नहीं है}: यह केवल गिर सकता है,
$\operatorname{rk}(vu) \leq \min(\operatorname{rk} u,
\operatorname{rk} v)$ (\cref{exo:b1:linmaps:4}), सटीक हानि के साथ
में मापा गया \cref{exo:b1:linmaps:12}.
\end{remark}

\begin{remark}[ये नक्शे कहां जाते हैं]
\label{rem:b1:linmaps:whereused}
\omterm{def:b1:linmaps:def}{रेखीय} मानचित्र बनने वाले हैं \emph{मैट्रिक्स}: एक बार \omterm{def:b1:vspaces:free}{आधार} हैं
निश्चित, \cref{ch:b1:matrices} प्रत्येक को एन्कोड करता है $ u \in \mathcal{L}(E,
F)$ एक आयताकार सरणी द्वारा, और संरचना मैट्रिक्स बन जाती है
उत्पाद--- रैंक--शून्यता तब के सिद्धांत को संचालित करती है \omterm{def:b1:linmaps:def}{रेखीय}
सिस्टम में \cref{ch:b1:det}. अनुमान में लौटें
\cref{ch:b1:euclid} उनके सबसे उपयोगी विशेष मामले में,
\emph{ओर्थोगोनल} अनुमान, जहां \omterm{def:b1:linmaps:kerim}{कर्नेल} चुना गया है
के लंबवत \omterm{def:b1:linmaps:kerim}{छवि}. नीचे दी गई सप्ताहांत समस्या आगे बढ़ाती है
जहाँ तक प्रथम वर्ष के उपकरणों की पहुँच है, प्रोजेक्टर बीजगणित तक
\omterm{pb:b1:linmaps:1}{फिटिंग की लेम्मा}; वर्ष~2 की मात्रा ट्रेस के साथ आगे बढ़ती है
और eigenvalue सिद्धांत के साथ, जिसके लिए स्थिर पर प्रोजेक्टर
उपस्थान बुनियादी निर्माण खंड हैं.
\end{remark}

\begin{remark}[पुस्तक 3 के अंदर परिप्रेक्ष्य: रैंक--शून्यता तीन बार और]
\label{rem:b1:linmaps:perspectives}
संरक्षण कानून $\dim E = \dim\ker u + \operatorname{rk} u $
खंड समाप्त होने से पहले तीन बार पुनः पढ़ा जाएगा। में
\cref{ch:b1:det} यह विलयन का आकार बन जाता है सेट: ए
के साथ संगत प्रणाली $ p $ अज्ञात और \omterm{def:b1:linmaps:rank}{रैंक} $ r $ एक समाधान है
\omterm{def:b1:logic:sets}{तय करना} आयाम का $ p - r $ --- \omterm{def:b1:linmaps:kerim}{कर्नेल} भेष में आयाम. में
\cref{ch:b1:euclid} यह ओर्थोगोनल रूप से विभाजित होता है, $\dim F + \dim
F^\perp = \dim E $, और हर दूरी की गणना को शक्ति प्रदान करता है। में
की सप्ताहांत समस्या \cref{ch:b1:multivar}, यह है
न्यूनतम वर्गों का लेखाकार: $ n $ अवलोकन, $2$ फिट
पैरामीटर, $ n - 2$ अवशिष्ट आयाम, और पायथागॉरियन
पहचान $\norm b^2 = \norm p^2 + \norm{b - p}^2$ है
\omterm{def:b1:linmaps:rank}{रैंक} - अशक्तता की यूक्लिडियन छाया। एक प्रमेय, चार वेशभूषा.
\end{remark}

\section{अभ्यास}

\begin{exercise}[$\star $]\label{exo:b1:linmaps:1}
कौन एमएपीएस हैं \omterm{def:b1:linmaps:def}{रेखीय}?
\begin{enumerate}
  \item $\R^2 \to \R^2$, $(x, y) \mapsto (x + y, x - 2y)$;
  \item $\R^2 \to \R $, $(x, y) \mapsto xy $;
  \item $\R[X] \to \R[X]$, $ P \mapsto P' + XP $;
  \item $\mathcal{F}(\R,\R) \to \R $, $ f \mapsto f(3)$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:linmaps:2}
होने देना $ u \colon \R^3 \to \R^3$, $(x,y,z) \mapsto (x + y - z,\; 2x + y
+ z,\; 3x + 2y)$. ठानना $\ker u $ (\omterm{def:b1:vspaces:free}{आधार}, आयाम),
$\operatorname{rk} u $, और ए \omterm{def:b1:vspaces:free}{आधार} का $\operatorname{im} u $. है $ u $
\omterm{def:b1:logic:inj}{इंजेक्शन}? \omterm{def:b1:logic:inj}{विशेषण}?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:linmaps:3}
होने देना $ u \colon \R_n[X] \to \R_n[X]$, $ P \mapsto P - P'$. साबित करें कि
$ u $ एक \omterm{def:b1:linmaps:projection}{समरूपता} है: एक बार के माध्यम से $\ker u $, एक बार प्रदर्शन करके
उलटा \emph{(विचार करना $ P + P' + P'' + \dots $)}.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:linmaps:4}
होने देना $ u \in \mathcal{L}(E, F)$ और $ v \in \mathcal{L}(F, G)$, रिक्त स्थान
\omterm{def:b1:findim:def}{परिमित-आयामी}. सिद्ध करना:
\[
\operatorname{rk}(v \circ u) \leq
\min\bigl(\operatorname{rk} u,\ \operatorname{rk} v\bigr).
\]
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:linmaps:5}
होने देना $ u $ का एक एंडोमोर्फिज्म हो $ E $ (\omterm{def:b1:findim:def}{परिमित-आयामी}) साथ $ u^2 =
0$. साबित करें कि $\operatorname{im} u \subseteq \ker u $, इस तरह
$\operatorname{rk} u \leq \frac{\dim E}{2}$. के लिए $ E = \R^2$, एक दे दो
समानता के साथ उदाहरण.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:linmaps:6}
होने देना $ p, q $ होना अनुमान का $ E $ साथ $ p \circ q = q \circ p $.
साबित करें कि $ p \circ q $ एक है अनुमान, साथ
\[
\operatorname{im}(pq) = \operatorname{im} p \cap \operatorname{im}
q ,
\qquad
\ker (pq) = \ker p + \ker q .
\]
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:linmaps:7}
होने देना $ u \in \mathcal{L}(E)$, $ E $ \omterm{def:b1:findim:def}{परिमित-आयामी}. सिद्ध करो
की समतुल्यता:
\begin{enumerate}
  \item $ E = \ker u \oplus \operatorname{im} u $;
  \item $\ker u = \ker u^2$;
  \item $\operatorname{im} u = \operatorname{im} u^2$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:linmaps:8}
होने देना $\varphi, \psi $ होना \omterm{def:b1:linmaps:forms}{रैखिक रूप} पर $ E $ साथ $\ker\varphi
\subseteq \ker\psi $. सिद्ध करना $\psi = \lambda\varphi $ कुछ के लिए
$\lambda \in K $ (पतित मामलों सहित)।
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b1:linmaps:9}
होने देना $ u \in \mathcal{L}(E)$ साथ $\dim E = n $, और मान लीजिए $ u^n = 0$
लेकिन $ u^{n-1} \neq 0$ (ए \emph{अधिकतम शून्यशक्तिशाली} एंडोमोर्फिज्म)।
चुनना $ x $ साथ $ u^{n-1}(x) \neq 0$; साबित करें कि $\bigl(x, u(x), \dots,
u^{n-1}(x)\bigr)$ एक है \omterm{def:b1:vspaces:free}{आधार} का $ E $. \emph{(की शक्तियाँ लागू करें $ u $ एक को
शून्य संयोजन, से आरंभ $ u^{n-1}$.)}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b1:linmaps:10}
होने देना $ f \in \mathcal{L}(\R^n)$ साथ $ f \circ f = -\mathrm{id}$.
\begin{enumerate}
  \item साबित करें कि $ f $ एक \omterm{def:b1:linmaps:projection}{समरूपता} है और वह नहीं $ x \neq 0$
        संतुष्ट $ f(x) = \lambda x $ साथ $\lambda \in \R $.
  \item साबित करें कि $ n $ सम है. \emph{संकेत: चुनें $ x_1 \neq 0$; दिखाओ
        $\operatorname{Vect}(x_1, f(x_1))$ एक विमान के नीचे स्थिर है
        $ f $; चुनना $ x_2$ इसके बाहर और इसे सिद्ध करते हुए पुनरावृत्त करें
        $\bigl(x_1, f(x_1), x_2, f(x_2), \dots\bigr)$ रहता है \omterm{def:b1:vspaces:free}{मुक्त}.}
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:linmaps:11}
होने देना $ u, v \in \mathcal{L}(E, F)$, रिक्त स्थान \omterm{def:b1:findim:def}{परिमित-आयामी}.
उभयपक्षीय बंधन सिद्ध करें
\[
\abs{\operatorname{rk} u - \operatorname{rk} v}
\;\leq\; \operatorname{rk}(u + v)
\;\leq\; \operatorname{rk} u + \operatorname{rk} v .
\]
\emph{(के लिए \omterm{def:b1:reals:bounds}{ऊपरी सीमा}, तुलना करना $\operatorname{im}(u+v)$ साथ
$\operatorname{im} u + \operatorname{im} v $; निचले वाले के लिए,
लागू करें \omterm{def:b1:reals:bounds}{ऊपरी सीमा} चतुराई से।)}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b1:linmaps:12}
(फ्रोबेनियस की असमानता) आइए $ u \in \mathcal{L}(E, F)$, $ w \in
\mathcal{L}(F, G)$ और $ v \in \mathcal{L}(G, H)$, सभी स्थान
\omterm{def:b1:findim:def}{परिमित-आयामी}. सटीक सूत्र सिद्ध करें
\[
\operatorname{rk}(v \circ w) = \operatorname{rk} w -
\dim\bigl(\ker v \cap \operatorname{im} w\bigr),
\]
और फ्रोबेनियस की असमानता को कम करें
\[
\operatorname{rk}(v \circ w) + \operatorname{rk}(w \circ u)
\;\leq\; \operatorname{rk} w + \operatorname{rk}(v \circ w \circ
u) .
\]
उस मामले की जाँच करें $ w = \mathrm{id}_F $ सिलवेस्टर का है
असमानता, मैट्रिक्स रूप में सिद्ध \cref{exo:b1:matrices:10}.
\end{exercise}

\section{समस्या: प्रोजेक्टर कैलकुलस और फिटिंग की लेम्मा}

\begin{problem}\label{pb:b1:linmaps:1}
अनुमान एंडोमोर्फिज्म हैं जो कि \omterm{def:b1:vspaces:sum}{प्रत्यक्ष} रकम उत्पादन, और
इसके विपरीत: हर पहचान $ E = F_1 \oplus \dots \oplus F_k $ है
गुप्त रूप से पहचान के लिए योग प्रोजेक्टर का एक परिवार। यह
समस्या यह विकसित करती है कि शब्दकोश --- एक का बीजगणित
प्रोजेक्टर, दो का, का $ k $ --- फिर वही लागू होता है
एक मनमाना एंडोमोर्फिज्म के लिए स्थिरीकरण विचार और साबित होता है
\emph{\omterm{pb:b1:linmaps:1}{फिटिंग की लेम्मा}}: ए की प्रत्येक एंडोमोर्फिज्म
\omterm{def:b1:findim:def}{परिमित-आयामी} अंतरिक्ष एक निलपोटेंट भाग और एक में विभाजित हो जाता है
उलटा भाग. लगातार, $ E $ एक है $ K $-वेक्टर स्थान का
आयाम $ n $, और \emph{प्रक्षेपक} मतलब $ p \in \mathcal{L}(E)$
साथ $ p^2 = p $ (\cref{thm:b1:linmaps:projchar}).
\index{अनुमान}\index{फिटिंग की लेम्मा}\index{निलपोटेंट}

\medskip
\noindent\textbf{Part I --- The algebra around one projector.}
होने देना $ p $ प्रोजेक्टर बनो, $ p \neq 0$, $ p \neq \mathrm{id}$.
\begin{enumerate}
  \item बताते हैं कि $\mathrm{id} - p $ एक प्रोजेक्टर और पहचान है
        $\operatorname{im}(\mathrm{id} - p)$ और
        $\ker(\mathrm{id} - p)$.
  \item गणना करना $(\lambda\,\mathrm{id} + \mu\,p)^2$ और निर्धारित करें
        सभी जोड़े $(\lambda, \mu) \in K^2$ जिसके लिए
        $\lambda\,\mathrm{id} + \mu\,p $ एक प्रोजेक्टर है.
  \item वह विमान दिखाओ $\operatorname{Vect}(\mathrm{id}, p)$
        का $\mathcal{L}(E)$ रचना के अंतर्गत स्थिर है, और वह
        प्रत्येक के लिए \omterm{def:b1:poly:def}{बहुपद} $ Q \in K[X]$,
        \[
        Q(p) = Q(0)\,\mathrm{id} + \bigl(Q(1) -
        Q(0)\bigr)\,p .
        \]
  \item किसके लिए निर्धारित करें $(\lambda, \mu)$  \omterm{def:b1:logic:map}{नक्शा}
        $\lambda\,\mathrm{id} + \mu\,p $ उलटा है, और दे
        इसका उलटा रूप है $\alpha\,\mathrm{id} + \beta\,p $.
        की क्रिया के माध्यम से उत्तर की व्याख्या करें
        $\lambda\,\mathrm{id} + \mu\,p $ पर $\operatorname{im} p $
        और पर $\ker p $.
  \item होने देना $ p'$ के साथ एक और प्रोजेक्टर बनें \emph{वही \omterm{def:b1:linmaps:kerim}{छवि}}
        $\operatorname{im} p' = \operatorname{im} p $. बताते हैं कि
        $ p\,p' = p'$ और $ p'\,p = p $. ये पहचान क्या करती हैं
        रचना के बारे में कहें अनुमान उसी पर \omterm{def:b1:vspaces:subspace}{उपस्पेस}
        अलग-अलग \omterm{def:b1:linmaps:kerim}{कर्नेल}?
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Two projectors.} होने देना $ p, q $ हो
के प्रोजेक्टर $ E $; मान लें कि विशेषता नहीं है $2$ (सच
के लिए $ K = \R, \C $).
\begin{enumerate}[resume]
  \item कल्पना करना $ p + q $ एक प्रोजेक्टर है. विस्तार $(p + q)^2$,
        दिखाओ $ pq + qp = 0$; के साथ रचना करना $ p $ बाईं ओर, फिर
        दाईं ओर, निष्कर्ष निकालें $ pq = qp $, और निष्कर्ष निकालें $ pq = qp =
        0$.
  \item इसके विपरीत, मान लीजिए $ pq = qp = 0$. बताते हैं कि $ p + q $ एक है
        प्रोजेक्टर, के साथ
        \[
        \operatorname{im}(p + q) = \operatorname{im} p \oplus
        \operatorname{im} q,
        \qquad
        \ker(p + q) = \ker p \cap \ker q .
        \]
  \item बताते हैं कि $ p - q $ एक प्रोजेक्टर है यदि और केवल यदि $ pq = qp
        = q $. \emph{(प्रश्न 6-7 को लागू करें $\mathrm{id} - p $
        और $ q $.)}
  \item का ज्यामितीय अर्थ दिखाएँ $ pq = qp = q $: यदि यह धारण करता है
        और केवल यदि $\operatorname{im} q \subseteq
        \operatorname{im} p $ और $\ker p \subseteq \ker q $. (एक
        फिर लिखता है $ q \leq p $: ``$ q $ कम पर परियोजनाएं, साथ में
        अधिक''.)
  \item अब चलो $ p, q $ आना-जाना। से स्मरण करें
        \cref{exo:b1:linmaps:6} वह $ pq $ प्रोजेक्टर चालू है
        $\operatorname{im} p \cap \operatorname{im} q $ साथ में
        $\ker p + \ker q $. बताते हैं कि $ r = p + q - pq $ एक है
        प्रोजेक्टर के साथ
        \[
        \operatorname{im} r = \operatorname{im} p +
        \operatorname{im} q,
        \qquad
        \ker r = \ker p \cap \ker q .
        \]
        \emph{(विचार करना $\mathrm{id} - r = (\mathrm{id} -
        p)(\mathrm{id} - q)$.)}
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Decompositions of the identity.}
\begin{enumerate}[resume]
  \item होने देना $ E = F_1 \oplus \dots \oplus F_k $ और के लिए $ x = x_1 +
        \dots + x_k $ (अद्वितीय अपघटन, $ x_i \in F_i $), \omterm{def:b1:logic:sets}{तय करना}
        $ p_i(x) = x_i $. प्रत्येक को दिखाओ $ p_i $ एक प्रोजेक्टर है, वह
        $ p_i p_j = 0$ के लिए $ i \neq j $, ओर वो $ p_1 + \dots + p_k
        = \mathrm{id}$; पहचान करना $\operatorname{im} p_i $ और
        $\ker p_i $.
  \item इसके विपरीत, चलो $ p_1, \dots, p_k \in \mathcal{L}(E)$
        संतुष्ट $ p_1 + \dots + p_k = \mathrm{id}$ और $ p_i p_j =
        0$ सभी के लिए $ i \neq j $. प्रत्येक को दिखाओ $ p_i $ एक है
        प्रोजेक्टर और वह $ E = \operatorname{im} p_1 \oplus
        \dots \oplus \operatorname{im} p_k $.
  \item दो प्रोजेक्टर के साथ $ p + q = \mathrm{id}$: बताते हैं कि $ pq
        = qp = 0$ स्वचालित रूप से धारण करता है.
  \item तीन प्रोजेक्टर के साथ $ p + q + r = \mathrm{id}$: बताते हैं कि $ p + q $ एक प्रोजेक्टर है, और प्रश्न 6 से निष्कर्ष निकालें
        वह \emph{सभी} जोड़ीवार उत्पाद गायब हो जाते हैं --- इसलिए $ E =
        \operatorname{im} p \oplus \operatorname{im} q \oplus
        \operatorname{im} r $, उत्पादों पर कोई परिकल्पना नहीं।
  \item के लिए $ k $ प्रोजेक्टर के साथ $ p_1 + \dots + p_k =
        \mathrm{id}$: पहले दिखाओ कि किसी के लिए उपस्थान, $\dim
        (F_1 + \dots + F_k) \leq \dim F_1 + \dots + \dim F_k $,
        समानता के साथ यदि और केवल यदि योग \omterm{def:b1:vspaces:sum}{प्रत्यक्ष} है; तो दिखाओ $ E = \operatorname{im} p_1 + \dots +
        \operatorname{im} p_k $, और इसे साबित करें \emph{अगर}
        इसके अतिरिक्त $\sum_i \operatorname{rk} p_i \leq n $, योग
        \omterm{def:b1:vspaces:sum}{प्रत्यक्ष} है और $ p_i p_j = 0$ के लिए $ i \neq j $.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Iterated \omterm{def:b1:linmaps:kerim}{कर्नेल}: Fitting's lemma.}
होने देना $ u \in \mathcal{L}(E)$, $\dim E = n $.
\begin{enumerate}[resume]
  \item प्रत्येक के लिए मान्य, दो शृंखलाएँ दिखाएँ $ k \geq 0$:
        \[
        \ker u^k \subseteq \ker u^{k+1},
        \qquad
        \operatorname{im} u^{k+1} \subseteq \operatorname{im}
        u^k .
        \]
  \item दिखाओ अगर $\ker u^{r} = \ker u^{r+1}$ कुछ के लिए $ r $,
        तब $\ker u^{k} = \ker u^{r}$ सभी के लिए $ k \geq r $; राज्य
        और इसके लिए अनुरूप स्थिरीकरण सिद्ध करें इमेजिस.
  \item निष्कर्ष निकालें कि एक सबसे छोटा पूर्णांक है $ r $ साथ $\ker
        u^{r} = \ker u^{r+1}$, वह $ r \leq n $, ओर वो इमेजिस
        उसी पर स्थिर हो जाओ $ r $.
  \item (\omterm{pb:b1:linmaps:1}{फिटिंग की लेम्मा}) इसे साबित करें
        \[
        E \;=\; \ker u^{r} \,\oplus\, \operatorname{im} u^{r} .
        \]
  \item वो दोनों दिखाओ उपस्थान के अंतर्गत स्थिर हैं $ u $, कि
        का प्रतिबंध $ u $ को $\ker u^{r}$ शून्यशक्तिशाली है, और वह
        का प्रतिबंध $ u $ को $\operatorname{im} u^{r}$ एक है
        की \omterm{def:b1:linmaps:projection}{समरूपता} $\operatorname{im} u^{r}$: हर
        एंडोमोर्फिज्म एक विहित पर है \omterm{def:b1:vspaces:sum}{प्रत्यक्ष} योग, `` निलपोटेंट
        प्लस उलटा''।
  \item होने देना $\pi $ प्रोजेक्टर पर हो $\ker u^{r}$ साथ में
        $\operatorname{im} u^{r}$. बताते हैं कि $\pi \circ u = u
        \circ \pi $.
\end{enumerate}

\medskip
\noindent\textbf{Part V --- A worked case, and synthesis.}
\begin{enumerate}[resume]
  \item होने देना $ u(x, y, z) = (y, 0, z)$ पर $\R^3$. गणना करना $ u^2$ और
        $ u^3$, स्थिरीकरण सूचकांक निर्धारित करें $ r $, द
        उपस्थान $\ker u^{r}$ और $\operatorname{im} u^{r}$, द
        फिटिंग प्रोजेक्टर $\pi $, और सूत्रों पर सत्यापित करें कि
        $\pi u = u\pi $ ओर वो $ u $ एक कारक पर शून्य है,
        \omterm{def:b1:logic:inj}{द्विभाजित} दूसरे पर।
  \item समतुल्यताएँ दिखाएँ: $ u $ निलपोटेंट $\iff $ $\ker u^{r} =
        E $ $\iff $ $\pi = \mathrm{id}$; और यह निष्कर्ष निकालें कि ए
        एन की निलपोटेंट एंडोमोर्फिज्म $ n $-आयामी स्थान
        सदैव संतुष्ट करता है $ u^{n} = 0$ (निलपोटेंसी इंडेक्स कभी नहीं
        आयाम से अधिक है)।
  \item (विशिष्टता) मान लीजिए $ E = A \oplus B $ साथ $ A, B $ स्थिर
        के अंतर्गत $ u $, प्रतिबंध $ u|_A $ निलपोटेंट और $ u|_B $
        \omterm{def:b1:logic:inj}{द्विभाजित}. सिद्ध करना $ A = \ker u^{r}$ और $ B =
        \operatorname{im} u^{r}$: फिटिंग अपघटन है
        अद्वितीय.
  \item संश्लेषण, चार वाक्यों में: कौन सा शब्दकोश भाग III
        के बीच स्थापित करता है \omterm{def:b1:vspaces:sum}{प्रत्यक्ष} रकम और के परिवार
        प्रोजेक्टर; प्रश्न 14 को किसी उत्पाद की आवश्यकता क्यों नहीं है?
        परिकल्पना जबकि प्रश्न 15 की आवश्यकता है \omterm{def:b1:linmaps:rank}{रैंक} परिकल्पना
        (और कौन सा वर्ष~2 उपकरण, ट्रेस, इसे हटा देता है); किस में
        सेंस फिटिंग का लेम्मा का स्थिर संस्करण है
        \cref{exo:b1:linmaps:7}; और दो कारक क्या हैं?
        वर्ष~2 के eigenvalue सिद्धांत में फिटिंग बनें
        आयतन. भाग IV में सिद्ध प्रमेय का नाम बताइए।
\end{enumerate}
\end{problem}
