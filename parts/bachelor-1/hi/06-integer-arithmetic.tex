\chapter{पूर्णांक अंकगणित}\label{ch:b1:arith}

अंकगणित --- का अध्ययन \omterm{def:b1:arith:divides}{भाजकत्व} में $\Z $ --- में शुरू किया गया था
हाई स्कूल वॉल्यूम. यह अध्याय इसे पूरी तरह से पुनर्निर्मित करता है
यूक्लिडियन विभाजन, पूर्ण प्रमाणों के साथ: \omterm{thm:b1:arith:gcd}{महत्तम सामान्य भाजक} और यह
\omterm{met:b1:arith:euclid}{यूक्लिडियन एल्गोरिथ्म}, बेज़ाउट की पहचान और \omterm{thm:b1:arith:gauss}{गॉस की लेम्मा}, \omterm{thm:b1:arith:fta}{मुख्य गुणनखंड प्रक्रिया}, और की गणना सर्वांगसमताएँ फ़र्मेट के छोटे तक
प्रमेय. अपने आकर्षण से परे, यह सामग्री वह मॉडल है
\cref{ch:b1:poly} बहुपदों के लिए अनुकरण करता है।

\section{विभाज्यता और यूक्लिडियन विभाजन}

\begin{definition}[भाजकत्व]\label{def:b1:arith:divides}
के लिए $ a, b \in \Z $, $ b $ \emph{विभाजित} $ a $ (लिखा हुआ $ b \mid
a $)\index{भाजकत्व} कब $ a = bq $ कुछ के लिए $ q \in \Z $. बुनियादी
परिणाम: यदि $ b \mid a $ और $ b \mid a'$ तब $ b \mid (ua + va')$
सभी के लिए $ u, v \in \Z $; अगर $ b \mid a $ और $ a \neq 0$ तब $\abs b \leq
\abs a $; और $ a \mid b $ के साथ साथ $ b \mid a $ बल $ b = \pm a $.
\end{definition}

\begin{theorem}[यूक्लिडियन प्रभाग]\label{thm:b1:arith:division}
सभी के लिए $ a \in \Z $ और $ b \in \N^*$, बिल्कुल एक जोड़ी है $(q, r)
\in \Z \times \N $ साथ
\[
a = bq + r, \qquad 0 \leq r < b .
\]
\end{theorem}

\begin{proof}
\emph{अस्तित्व।}  \omterm{def:b1:logic:sets}{तय करना} $ A = \{a - bk : k \in \Z\} \cap \N $ एक है
का गैररिक्त उपसमुच्चय $\N $ (लेना $ k = -\abs a $: $ a + b\abs a \geq a +
\abs a \geq 0$). होने देना $ r = a - bq $ इसका सबसे छोटा तत्व हो. अगर $ r \geq b $,
तब $ r - b = a - b(q+1)$ का एक छोटा तत्व होगा $ A $:
विरोधाभास। इसलिए $0 \leq r < b $.

\emph{विशिष्टता.} अगर $ bq + r = bq' + r'$ साथ $0 \leq r, r' < b $,
तब $ b(q - q') = r' - r $ और $\abs{r' - r} < b $: का गुणज $ b $
बाईं ओर होना चाहिए $0$, इसलिए $ q = q'$ और $ r = r'$.
\end{proof}

\begin{example}[बार-बार विभाजन द्वारा स्थितीय अंकन]\label{ex:b1:arith:base7}
लिखना $2026$ आधार में $7$. द्वारा बार-बार विभाजित करें $7$, रखते हुए
शेष:
\[
2026 = 7 \times 289 + 3, \quad
289 = 7 \times 41 + 2, \quad
41 = 7 \times 5 + 6, \quad
5 = 7 \times 0 + 5 .
\]
अंतिम से प्रथम तक के शेषफलों को पढ़ना: $2026 =
(5\,6\,2\,3)_7$. जाँच करना: $5 \times 343 + 6 \times 49 + 2 \times 7
+ 3 = 1715 + 294 + 14 + 3 = 2026$. यूक्लिडियन की विशिष्टता
विभाजन ही वास्तव में प्रत्येक अंक बनाता है \emph{मजबूर}: प्रत्येक पर
चरण में शेषफल ही एकमात्र पूर्णांक है $\intint06$ सर्वांगसम
वर्तमान मूल्य मॉड के लिए $7$, तो आधार-$7$ लेखन अद्वितीय है ---
जब भी सप्ताहांत की समस्या में हेरफेर होता है तो तथ्य चुपचाप उपयोग किया जाता है
``के अंक $ n $ आधार में $ p $''.
\end{example}

\section{महत्तम सामान्य भाजक}

\begin{theorem}[के उपसमूह $\Z $; जीसीडी का अस्तित्व]\label{thm:b1:arith:gcd}
\begin{enumerate}
  \item का प्रत्येक उपसमूह $(\Z, +)$ स्वरूप का है $ n\Z = \{nk : k
        \in \Z\}$ एक अद्वितीय के लिए $ n \in \N $.
  \item के लिए $ a, b \in \Z $ दोनों शून्य नहीं, द \omterm{def:b1:logic:sets}{तय करना} $ a\Z + b\Z = \{au +
        bv : u, v \in \Z\}$ का एक उपसमूह है $\Z $, इसलिए बराबर है
        $ d\,\Z $ एक अद्वितीय के लिए $ d \in \N^*$. यह $ d $ है
        \emph{महत्तम सामान्य भाजक}\index{महत्तम सामान्य भाजक}
        $\gcd(a, b)$: यह विभाजित $ a $ और $ b $, और हर आम
        का भाजक $ a $ और $ b $ विभाजित $ d $.
\end{enumerate}
\end{theorem}

\begin{proof}
(1) चलो $ H \subseteq \Z $ एक उपसमूह बनें (गैर-रिक्त, स्थिर)।
घटाव; औपचारिक परिभाषा में है \cref{ch:b1:structures},
और केवल इन दो गुणों का उपयोग किया जाता है)। अगर $ H = \{0\}$, लेना $ n = 0$.
अन्यथा $ H $ इसमें एक शून्येतर तत्व और उसका विपरीत तत्व शामिल है, इसलिए a
सबसे छोटा सख्ती से सकारात्मक तत्व $ n $. तब $ n\Z \subseteq H $. के लिए $ x
\in H $, लिखना $ x = nq + r $ साथ $0 \leq r < n $
(\cref{thm:b1:arith:division}); $ r = x - nq \in H $, और की न्यूनतमता
$ n $ बल $ r = 0$: $ x \in n\Z $. विशिष्टता: $ n $ सबसे कम सकारात्मक है
का तत्व $ n\Z $.

(2) $ a\Z + b\Z $ रोकना $0$ और घटाव के तहत स्थिर है, इसलिए यह है
$ d\Z $ साथ $ d \geq 1$ (इसमें है $ a $ या $ b $ शून्येतर)। तब से $ a, b
\in d\Z $, $ d $ विभाजित दोनों। और यदि $ c $ विभाजित $ a $ और $ b $, तब $ c $
विभाजित प्रत्येक $ au + bv $ --- विशेष रूप से $ c \mid d $, तब से $ d \in a\Z
+ b\Z $. यह संपत्ति की घोषणा की गई है (और इसका तात्पर्य है $\abs c \leq
d $, इसलिए $ d $ नाम के योग्य है \emph{महानतम} सामान्य भाजक)।
\end{proof}

\begin{corollary}[बेज़ौट की पहचान]\label{cor:b1:arith:bezout}
के लिए $ a, b $ दोनों शून्य नहीं, अस्तित्व में हैं $ u, v \in \Z $ साथ
\[
au + bv = \gcd(a, b) .
\]
विशेष रूप से ($\gcd(a,b) = 1$, द \emph{सह अभाज्य}\index{सह अभाज्य}
मामला): $ a $ और $ b $ हैं \omterm{cor:b1:arith:bezout}{सह अभाज्य} अगर और केवल अगर $ au + bv = 1$ एक है
समाधान.
\end{corollary}

\begin{proof}
$\gcd(a,b) = d \in d\Z = a\Z + b\Z $. समतुल्यता के लिए: यदि
$\gcd(a,b) = 1$, बेज़ाउट समाधान प्रदान करता है; इसके विपरीत $ au + bv =
1$ के प्रत्येक सामान्य विभाजक को बाध्य करता है $ a, b $ अलग करना $1$.
\end{proof}

\begin{method}[यूक्लिडियन एल्गोरिथ्म, विस्तारित]\label{met:b1:arith:euclid}
गणना करना $\gcd(a, b)$ ($ a > b > 0$): विभाजित करना $ a = bq + r $; तब
$\gcd(a, b) = \gcd(b, r)$ (के सामान्य भाजक $(a,b)$ और का $(b,r)$
संयोग, तब से $ r = a - bq $); शेष रहने तक दोहराएँ $0$; द
अंतिम शून्येतर शेषफल gcd है\index{यूक्लिडियन एल्गोरिथ्म}. चल रहा है
विभाजन पीछे की ओर (या रास्ते में गुणांक बनाए रखना)।
डाउन) एक बेज़आउट जोड़ी का निर्माण करता है $(u, v)$.
\end{method}

\begin{example}\label{ex:b1:arith:euclid}
$\gcd(120, 23)$: $120 = 5 \times 23 + 5$; $23 = 4 \times 5 + 3$; $5 =
1\times 3 + 2$; $3 = 1 \times 2 + 1$; $2 = 2 \times 1 + 0$. इसलिए
$\gcd = 1$. पीछे की ओर:
\begin{align*}
1 &= 3 - 2 = 3 - (5 - 3) = 2\times 3 - 5 = 2(23 - 4\times 5) - 5 \\
&= 2 \times 23 - 9 \times 5 = 2\times 23 - 9(120 - 5\times 23)
= 47 \times 23 - 9 \times 120 .
\end{align*}
जाँच करना: $47 \times 23 = 1081$, $9 \times 120 = 1080$.
\end{example}

\begin{theorem}[गॉस की प्रमेयिका और परिणाम]\label{thm:b1:arith:gauss}
होने देना $ a, b, c \in \Z $.
\begin{enumerate}
  \item (\omterm{thm:b1:arith:gauss}{गॉस की लेम्मा}\index{गॉस की लेम्मा}) अगर $ a \mid bc $ और
        $\gcd(a, b) = 1$, तब $ a \mid c $.
  \item अगर $ a \mid c $, $ b \mid c $ और $\gcd(a,b) = 1$, तब $ ab \mid
        c $.
  \item अगर $\gcd(a, b) = \gcd(a, c) = 1$, तब $\gcd(a, bc) = 1$.
\end{enumerate}
\end{theorem}

\begin{proof}
(1) बेज़आउट: $ au + bv = 1$. गुणा करके $ c $: $ acu + bcv = c $. दोनों
पद विभाज्य हैं $ a $ (दूसरा क्योंकि $ a \mid bc $), इसलिए $ a
\mid c $.

(2)लिखो $ c = aq $; से $ b \mid aq $ और $\gcd(a, b) = 1$, बिंदु (1)
देता है $ b \mid q $, इसलिए $ ab \mid aq = c $.

(3) $ au + bv = 1$ और $ au' + cv' = 1$. दोनों को गुणा करें
संबंध:
\[
1 = (au + bv)(au' + cv')
= a\,\bigl(auu' + ucv' + u'bv\bigr) + bc\,(vv') ,
\]
के बीच एक बेज़आउट संबंध $ a $ और $ bc $: द्वारा
\cref{cor:b1:arith:bezout}, $\gcd(a, bc) = 1$.
\end{proof}

\begin{example}[एक रैखिक डायोफैंटाइन समीकरण को हल करना]\label{ex:b1:arith:diophantine}
सभी खोजें $(x, y) \in \Z^2$ साथ $6x + 10y = 4$. सबसे पहले,
\emph{अस्तित्व परीक्षण}: $\gcd(6, 10) = 2$ विभाजित $4$, तो समाधान
अस्तित्व में है (यदि जीसीडी ने दाहिनी ओर, बाईं ओर को विभाजित नहीं किया है
हमेशा इसका गुणज होगा और कोई नहीं होगा)। फूट डालो
के माध्यम से: $3x + 5y = 2$. एक विशेष समाधान दिखाई दे रहा है: $(x_0,
y_0) = (-1, 1)$. सामान्य के लिए, घटाएँ: $3(x + 1) = -5(y -
1)$, इसलिए $3 \mid 5(y-1)$, और \omterm{thm:b1:arith:gauss}{गॉस की लेम्मा} ($\gcd(3,5) = 1$) देता है
$3 \mid y - 1$: $ y = 1 - 3k $, तब $ x = -1 + 5k $. इसके विपरीत प्रत्येक
ऐसी जोड़ी काम करती है:
\[
(x, y) = (-1 + 5k,\ 1 - 3k), \qquad k \in \Z .
\]
पैटर्न सामान्य है: एक विशेष समाधान और पूर्णांक
के गुणज $\bigl(\frac b{\gcd}, -\frac a{\gcd}\bigr)$ --- द
जैसी ही ``विशेष प्लस सजातीय'' संरचना
\cref{ch:b1:diffeq}, \omterm{thm:b1:arith:gauss}{गॉस की लेम्मा} विशिष्टता की भूमिका निभा रही है
भूमिका.
\end{example}

\begin{definition}[न्यूनतम समापवर्तक]\label{def:b1:arith:lcm}
$\operatorname{lcm}(a, b)$\index{न्यूनतम समापवर्तक} है
जनरेटर में $\N $ उपसमूह का $ a\Z \cap b\Z $: यह एक आम बात है
का एकाधिक $ a $ और $ b $ वह विभाजित प्रत्येक सामान्य गुणक, और के लिए
$ a, b \in \N^*$,
\[
\gcd(a,b) \times \operatorname{lcm}(a,b) = ab
\qquad (\text{proof in } \cref{exo:b1:arith:5}).
\]
\end{definition}

\begin{example}[संरेखण समस्याएँ एलसीएम समस्याएँ हैं]\label{ex:b1:arith:gears}
दो जालीदार गियर हैं $84$ और $36$ दाँत। कितने दांत के बाद
सामान्य गति से क्या वे एक साथ अपनी प्रारंभिक स्थिति में लौट आते हैं?
जब ख़त्म हुए दांतों की संख्या a हो तो कॉन्फ़िगरेशन दोहराता है
का सामान्य गुणज $84$ और $36$; पहली बार है
\[
\operatorname{lcm}(84, 36) = \frac{84 \times 36}{\gcd(84, 36)}
= \frac{3024}{12} = 252
\]
दांत --- अर्थात, $3$ बड़े गियर के घुमाव और $7$ का
छोटा सा ($252/84$ और $252/36$). व्यावहारिक मार्ग पर ध्यान दें:
\emph{पहले जीसीडी की गणना करें} (यूक्लिड: $84 = 2\times36 + 12$, $36
= 3\times12$), फिर विभाजित करें --- कभी भी सूचीबद्ध करके एलसीएम न बनाएं
एकाधिक. प्रत्येक आवधिक-संयोग प्रश्न (गियर्स, ग्रहीय
संरेखण, दशमलव को दोहराते हुए पूरा करना) इसे कम कर देता है
गणना.
\end{example}

\section{प्रमुख संख्या}

\begin{definition}\label{def:b1:arith:prime}
पूर्णांक $ p \geq 2$ है \emph{मुख्य}\index{मुख्य} जब यह
केवल सकारात्मक विभाजक हैं $1$ और $ p $. के लिए $ p $ प्रधान और $ a \in \Z $:
दोनों में से एक $ p \mid a $, या $\gcd(p, a) = 1$. फलस्वरूप
(\cref{thm:b1:arith:gauss}), \emph{यूक्लिड की प्रमेयिका} धारण: यदि $ p \mid
ab $ तब $ p \mid a $ या $ p \mid b $.
\end{definition}

\begin{remark}[परीक्षण प्रभाग द्वारा आदिमता का परीक्षण]\label{rem:b1:arith:trialdivision}
अगर $ n = ab $ साथ $2 \leq a \leq b $, तब $ a^2 \leq ab = n $, इसलिए $ a
\leq \sqrt n $: एक समग्र $ n $ हमेशा एक होता है \omterm{def:b1:arith:prime}{मुख्य} भाजक $\leq
\sqrt n $. अतः यह परीक्षण करना है कि क्या $ n $ है \omterm{def:b1:arith:prime}{मुख्य} प्रयास करना ही काफी है
द अभाज्य तक $\sqrt n $. के लिए $ n = 271$: $\sqrt{271} < 17$, और
$271$ किसी से भी विभाज्य नहीं है $2, 3, 5, 7, 11, 13$ (विषम, अंक
योग $10$, में ख़त्म नहीं होता $0$ या $5$, $271 = 7\cdot38 + 5 =
11\cdot24 + 7 = 13\cdot20 + 11$): \omterm{def:b1:arith:prime}{मुख्य}, छह डिवीजनों के बाद
दो सौ की जगह. $\sqrt n $ बाधा वास्तविक है
दहलीज: सौ-अंकीय संख्याओं के लिए इसे कुशलतापूर्वक पार करना
इसके लिए आधुनिक प्रारंभिक परीक्षणों की आवश्यकता होती है
\cref{thm:b1:arith:fermat}.
\end{remark}

\begin{theorem}[यूक्लिड]\label{thm:b1:arith:euclidprimes}
अनन्त रूप से अनेक हैं अभाज्य.
\end{theorem}

\begin{proof}
प्रत्येक पूर्णांक $ n \geq 2$ एक \omterm{def:b1:arith:prime}{मुख्य} भाजक: इसका सबसे छोटा भाजक
$\geq 2$ है \omterm{def:b1:arith:prime}{मुख्य} (इसका उचित गुणनखंडन एक उत्पादन करेगा
का छोटा भाजक $ n $). अब मान लीजिए $ p_1, \dots, p_k $ सभी थे
अभाज्य, और चलो $ N = p_1 p_2 \cdots p_k + 1 \geq 2$. कुछ \omterm{def:b1:arith:prime}{मुख्य} $ p_i $
विभाजित $ N $; लेकिन $ p_i $ भी विभाजित $ N - 1 = p_1\cdots p_k $, इसलिए $ p_i
\mid 1$ --- बेतुका।
\end{proof}

\begin{theorem}[अंकगणित का मौलिक प्रमेय]\label{thm:b1:arith:fta}
प्रत्येक पूर्णांक $ n \geq 2$ का एक उत्पाद है अभाज्य, और गुणनखंडन
\[
n = p_1^{\alpha_1} p_2^{\alpha_2} \cdots p_k^{\alpha_k}
\qquad (p_1 < p_2 < \dots < p_k \text{ primes},\ \alpha_i \in \N^*)
\]
निराला है।\index{मुख्य गुणनखंड प्रक्रिया}
\end{theorem}

\begin{proof}
\emph{अस्तित्व} मजबूत प्रेरण द्वारा (\cref{thm:b1:logic:induction}):
$ n = 2$ है \omterm{def:b1:arith:prime}{मुख्य}; के लिए $ n > 2$, दोनों में से एक $ n $ है \omterm{def:b1:arith:prime}{मुख्य}, या $ n = ab $ साथ
$2 \leq a, b < n $, और प्रेरण परिकल्पना कारक $ a $ और $ b $.

\emph{विशिष्टता.} कल्पना करना $ p_1 \cdots p_r = q_1 \cdots q_s $
(अभाज्य पुनरावृत्ति के साथ सूचीबद्ध, $ r \leq s $ कहें), और शामिल करें
$ r $. अगर $ r = 0$ बाईं ओर है $1$, मजबूर करना $ s = 0$ (एक गैररिक्त
का उत्पाद अभाज्य से अधिक है $1$). के लिए $ r \geq 1$: द \omterm{def:b1:arith:prime}{मुख्य} $ p_1$
विभाजित $ q_1(q_2\cdots q_s)$, तो यूक्लिड की प्रमेयिका द्वारा भी $ p_1
\mid q_1$ या $ p_1 \mid q_2\cdots q_s $; पुनरावर्ती, $ p_1$ विभाजित
कुछ $ q_j $. लेकिन $ q_j $ है \omterm{def:b1:arith:prime}{मुख्य} और $ p_1 \geq 2$: अनिवार्य रूप से $ p_1
= q_j $. इस सामान्य कारक को रद्द करें (वैध: $\Z $ एक है
इंटीग्रल डोमेन) प्राप्त करना
\[
p_2 \cdots p_r = q_1 \cdots \widehat{q_j} \cdots q_s
\]
(हैट मार्किंग चूक), छोटे उत्पादों की समानता; द
प्रेरण परिकल्पना दो सूचियाँ कहती है $ p_2, \dots, p_r $ और
$ q_1, \dots, \widehat{q_j}, \dots, q_s $ ऑर्डर के अनुरूप,
इसलिए मूल लोगों ने भी ऐसा ही किया। घातांक समान समूह बनाते हैं
अभाज्य.
\end{proof}

\begin{proposition}[वैल्यूएशन]\label{prop:b1:arith:valuation}
के लिए $ p $ \omterm{def:b1:arith:prime}{मुख्य} और $ n \in \N^*$, लिखना $ v_p(n)$\index{पी-एडिक
वैल्यूएशन@$ p $-एडिक वैल्यूएशन} के प्रतिपादक के लिए $ p $ में
का गुणनखंडन $ n $ (साथ $ v_p(n) = 0$ अगर $ p \nmid n $). तब
\[
v_p(mn) = v_p(m) + v_p(n),
\qquad
m \mid n \iff \forall p,\ v_p(m) \leq v_p(n),
\]
\[
v_p\bigl(\gcd(m,n)\bigr) = \min\bigl(v_p(m), v_p(n)\bigr),
\qquad
v_p\bigl(\operatorname{lcm}(m,n)\bigr) = \max\bigl(v_p(m),
v_p(n)\bigr).
\]
\end{proposition}

\begin{proof}
पहली पहचान इसलिए कायम है क्योंकि गुणनखंड कई गुना बढ़ जाते हैं और
का गुणनखंडन $ mn $ निराला है। अगर $ m \mid n $, लिखना $ n = mq $ और
इसे लागू करें. इसके विपरीत, यदि सभी $ v_p(m) \leq v_p(n)$, पूर्णांक $ q =
\prod_p p^{\,v_p(n) - v_p(m)}$ संतुष्ट $ mq = n $. जीसीडी सूत्र:
पूर्णांक $ d = \prod p^{\min}$ विभाजित दोनों कसौटी से, और
प्रत्येक सामान्य विभाजक $ c $ है $ v_p(c) \leq \min $ सभी के लिए $ p $, इसलिए $ c
\mid d $; एलसीएम के लिए भी यही तर्क $\max $.
\end{proof}

\begin{example}[मूल्यांकन के माध्यम से वर्ग और घन]\label{ex:b1:arith:squarecube}
पूर्णांक $ n \geq 1$ एक पूर्ण वर्ग है यदि और केवल यदि प्रत्येक
$ v_p(n)$ सम है (यदि $ n = m^2$, तब $ v_p(n) = 2v_p(m)$;
इसके विपरीत प्रत्येक घातांक को आधा कर दें)। इसी प्रकार घनों के लिए भी
के गुणज $3$. इस प्रकार $21168 = 2^4 \times 3^3 \times 7^2$ नहीं है
एक वर्ग ($ v_3 = 3$ विषम है) और घन नहीं है ($ v_2 = 4$); द
सबसे छोटा धनात्मक पूर्णांक $ m $ ऐसा है कि $21168\,m $ \emph{है} ए
प्रत्येक घातांक को अगले गुणज में जोड़कर घन पाया जाता है
$3$:
\[
m = 2^{6-4} \times 3^{3-3} \times 7^{3-2} = 2^2 \times 7 = 28,
\qquad
21168 \times 28 = 2^6\,3^3\,7^3 = (2^2 \times 3 \times 7)^3
= 84^3 .
\]
अंतर्दृष्टि: गुणनात्मक प्रश्न (वर्ग, घन, भाजक,
जीसीडी, एलसीएम) बन गया \emph{समन्वय के अनुसार} घातांक पर प्रश्न
वेक्टर $(v_2, v_3, v_5, \dots)$ --- अद्वितीय गुणनखंडन है
\omterm{def:b1:logic:statement}{कथन} ये निर्देशांक मौजूद हैं और अच्छी तरह से परिभाषित हैं।
\end{example}

\section{सर्वांगसमताएँ}

\begin{definition}\label{def:b1:arith:congruence}
के लिए $ n \in \N^*$: $ a \equiv b \pmod n $\index{अनुरूपता} कब $ n \mid
a - b $. यह एक है \omterm{def:b1:logic:equiv}{समतुल्य संबंध} जोड़ के साथ संगत और
गुणन: यदि $ a \equiv b $ और $ a' \equiv b'$ (मॉड $ n $), तब
$ a + a' \equiv b + b'$, $ aa' \equiv bb'$, और $ a^k \equiv b^k $ के लिए
$ k \in \N $.
\end{definition}

\begin{example}[नौ को बाहर निकालना]\label{ex:b1:arith:castingnines}
के साथ अनुकूलता $+$ और $\times $ एक जांच उपकरण पुराना है
वाणिज्य के रूप में. तब से $10 \equiv 1 \pmod 9$, प्रत्येक पूर्णांक है
सर्वांगसम मॉड $9$ इसके अंकों के योग तक (जैसा सिद्ध हुआ)
\cref{exo:b1:arith:2}). दावे की जांच करने के लिए $1234 \times 567 =
699\,678$: अंकों का योग दें $1234 \equiv 1$ और $567 \equiv 18
\equiv 0 \pmod 9$, तो उत्पाद होना चाहिए $\equiv 1 \times 0 =
0$; सचमुच $6 + 9 + 9 + 6 + 7 + 8 = 45 \equiv 0$. चेक
पास हो जाता है (और उत्पाद वास्तव में सही है)। क्या किसी ने रिपोर्ट की थी
$699\,478$, अंक योग $43 \equiv 7 \not\equiv 0$ होगा
उन्हें तुरंत दोषी ठहराओ. परीक्षण एकतरफ़ा है --- यह एक पकड़ लेता है
त्रुटि जब तक कि त्रुटि स्वयं का गुणज न हो $9$ --- जो है
बिलकुल छद्मप्रधान पाठ
\cref{ex:b1:arith:pseudoprime} लघु रूप में: \omterm{def:b1:arith:congruence}{अनुरूपता} जाँच करता है
खंडन करते हैं, प्रमाणित नहीं करते।
\end{example}

\begin{proposition}[इन्वर्टिबिलिटी मॉड $ n $]\label{prop:b1:arith:invmod}
$ a $ है \emph{उलटा मॉड $ n $} (अर्थात्\ $ ab \equiv 1 \pmod n $ कुछ के लिए
$ b $) अगर और केवल अगर $\gcd(a, n) = 1$. इसके बाद व्युत्क्रम अद्वितीय मॉड है
$ n $ और विस्तारित द्वारा गणना की गई \omterm{met:b1:arith:euclid}{यूक्लिडियन एल्गोरिथ्म}.
\end{proposition}

\begin{proof}
$ ab \equiv 1 \pmod n $ मतलब $ ab + nk = 1$ कुछ के लिए $ k $: एक बेज़आउट
संबंध, जो मौजूद है यदि $\gcd(a,n) = 1$
(\cref{cor:b1:arith:bezout}). विशिष्टता: यदि $ ab \equiv ab' \equiv 1$,
तब $ b \equiv b(ab') = (ab)b' \equiv b' \pmod n $.
\end{proof}

\begin{example}[उलटना $7$ सापेक्ष $26$]\label{ex:b1:arith:inv7mod26}
तब से $\gcd(7, 26) = 1$, का वर्ग $7$ उलटा मॉड है $26$.
विस्तारित यूक्लिड:
\[
26 = 3 \times 7 + 5, \qquad
7 = 1 \times 5 + 2, \qquad
5 = 2 \times 2 + 1 ,
\]
फिर पीछे की ओर:
\[
1 = 5 - 2 \times 2 = 5 - 2(7 - 5) = 3 \times 5 - 2 \times 7
= 3(26 - 3 \times 7) - 2 \times 7 = 3 \times 26 - 11 \times 7 .
\]
इस तरह $7 \times (-11) \equiv 1 \pmod{26}$, यानी\ $7^{-1} \equiv
-11 \equiv 15 \pmod{26}$; जाँच करना: $7 \times 15 = 105 = 4 \times 26
+ 1$. हाथ में उलटा लेकर, कोई भी \omterm{def:b1:arith:congruence}{अनुरूपता} $7x \equiv c
\pmod{26}$ एक गुणन में हल किया जाता है: $ x \equiv 15c $. यह
यांत्रिक व्युत्क्रम मॉड्यूलर अंकगणित का कार्यक्षेत्र है ---
और सार्वजनिक-कुंजी प्रोटोकॉल में उल्लिखित है
\cref{rem:b1:arith:whereused}, जहां मॉड्यूलि में सैकड़ों हैं
अंक लेकिन एल्गोरिदम बिल्कुल यही है।
\end{example}

\begin{example}[जब गुणांक व्युत्क्रमणीय न हो]\label{ex:b1:arith:noninvertible}
हल करना $12x \equiv 8 \pmod{20}$. यहाँ $\gcd(12, 20) = 4$, इसलिए $12$
उलटा मॉड नहीं है $20$ --- लेकिन समीकरण अभी भी है
ट्रैक्टेबल \omterm{def:b1:arith:congruence}{अनुरूपता} कहते हैं $20 \mid 12x - 8$; विभाजित करना
द्वारा संपूर्ण संबंध $4$ (तीनों अवयवों का विभाजक), यह है
के बराबर $5 \mid 3x - 2$, अर्थात\
\[
3x \equiv 2 \pmod 5 .
\]
अब $\gcd(3, 5) = 1$ और $3^{-1} \equiv 2 \pmod 5$ ($3 \times 2 =
6 \equiv 1$), इसलिए $ x \equiv 4 \pmod 5$: समाधान हैं $ x
\equiv 4, 9, 14, 19 \pmod{20}$ --- \emph{चार} कक्षाएं मॉड $20$,
जीसीडी से मिलान। (यदि दाहिना भाग विभाज्य न होता $4$,
कहना $12x \equiv 6 \pmod{20}$, इसका कोई समाधान नहीं होगा:
बाईं ओर हमेशा है $\equiv 0 \pmod 4$.) सामान्य आकार: $ ax
\equiv b \pmod n $ यदि हल करने योग्य है $\gcd(a, n) \mid b $, और फिर
बिलकुल है $\gcd(a, n)$ समाधान वर्ग --- सब कुछ विभाजित करें
जीसीडी और उलटा द्वारा।
\end{example}

\begin{theorem}[फ़र्मेट का छोटा प्रमेय]\label{thm:b1:arith:fermat}
होने देना $ p $ होना \omterm{def:b1:arith:prime}{मुख्य}. हरएक के लिए $ a \in \Z $:
\[
a^p \equiv a \pmod p,
\]
और यदि $ p \nmid a $, तब $ a^{p-1} \equiv 1 \pmod
p $.\index{फ़र्मेट का छोटा प्रमेय}
\end{theorem}

\begin{proof}
सबसे पहले, के लिए $1 \leq k \leq p - 1$, द \omterm{def:b1:counting:objects}{द्विपद गुणांक} $\binom pk
= \frac{p!}{k!(p-k)!}$ से विभाज्य है $ p $: वास्तव में $ k!\,(p-k)!\,
\binom pk = p!$ और $ p $ विभाजित $ p!$ लेकिन है \omterm{cor:b1:arith:bezout}{सह अभाज्य} को $ k!(p-k)!$
(सभी कारक हैं $< p $), तो गॉस का लेम्मा देता है $ p \mid \binom pk $.

अब साबित करो $ a^p \equiv a $ के लिए $ a \in \N $ प्रेरण द्वारा. के लिए सच है $ a =
0$. अगर $ a^p \equiv a $, फिर द्विपद प्रमेय द्वारा
\[
(a+1)^p = \sum_{k=0}^{p} \binom pk a^k
\equiv a^p + 1 \equiv a + 1 \pmod p,
\]
सभी मध्य पद लुप्त हो रहे हैं $ p $. के लिए $ a < 0$, परिणाम को लागू करें
$-a $ और अलग $ p = 2$ (कहाँ $ x \equiv -x $) विषम से $ p $ (कहाँ
$(-a)^p = -a^p $). अंततः, यदि $ p \nmid a $, गुणा करें $ a^p \equiv a $ द्वारा
का उलटा $ a $ आधुनिक $ p $ (\cref{prop:b1:arith:invmod}).
\end{proof}

\begin{example}[फ़र्मेट का वार्तालाप विफल रहता है: $341$]\label{ex:b1:arith:pseudoprime}
फ़र्मेट का छोटा प्रमेय सस्ता देता है \emph{समग्रता} परीक्षण:
यदि $ a^{n-1} \not\equiv 1 \pmod n $ कुछ के लिए $ a $ \omterm{cor:b1:arith:bezout}{सह अभाज्य} को $ n $,
तब $ n $ क्या नहीं है \omterm{def:b1:arith:prime}{मुख्य}. क्या परीक्षण आदिमता को भी प्रमाणित कर सकता है? नहीं:
ले लो $ n = 341 = 11 \times 31$, समग्र, और $ a = 2$. तब से
$2^{10} = 1024 = 3 \times 341 + 1$,
\[
2^{10} \equiv 1 \pmod{341}
\qquad\Longrightarrow\qquad
2^{340} = \bigl(2^{10}\bigr)^{34} \equiv 1 \pmod{341} :
\]
समग्र $341$ आधार के लिए फ़र्मेट का परीक्षण पास कर लेता है $2$ (यह है
सबसे छोटा ऐसा \emph{स्यूडोप्राइम}). आधार $3$ इसे बेनकाब करता है
($3^{340} \not\equiv 1$), और व्यावहारिक मौलिकता परीक्षण
इसलिए परीक्षण कई आधारों पर चलता है, साथ ही परिशोधन ---
इस विचार के औद्योगिक संस्करण ही बड़े पैमाने पर प्रमाणित होते हैं
अभाज्य का \cref{rem:b1:arith:whereused}. नैतिक: एक निहितार्थ
और इसके विपरीत अलग-अलग जीवन जीते हैं
(\cref{rem:b1:logic:pitfalls}), यहां तक कि प्रमेयों के लिए भी।
\end{example}

\begin{example}[व्यावहारिक सर्वांगसमता गणना]\label{ex:b1:arith:congruences}
का शेष क्या है $7^{2026}$ आधुनिक $11$? फ़र्मेट द्वारा, $7^{10}
\equiv 1 \pmod{11}$. तब से $2026 = 10 \times 202 + 6$:
\[
7^{2026} \equiv 7^6 = (7^2)^3 = 49^3 \equiv 5^3 = 125 \equiv 4
\pmod{11}.
\]
शेष है $4$. रणनीति: घातांक मॉड्यूलो को कम करें
फ़र्मेट द्वारा प्रदान किया गया आदेश, फिर प्रत्येक पर मध्यवर्ती शक्तियों को कम करें
कदम.
\end{example}

\begin{remark}[अंकगणित में सामान्य गड़बड़ियाँ]\label{rem:b1:arith:pitfalls}
\begin{enumerate}
  \item \emph{ए को विभाजित करना \omterm{def:b1:arith:congruence}{अनुरूपता}.} से $ ac \equiv bc \pmod n $
        कोई भी हो सकता है \emph{नहीं} निष्कर्ष $ a \equiv b $ जब तक $\gcd(c,
        n) = 1$: $6 \equiv 2 \pmod 4$ लेकिन $3 \not\equiv 1 \pmod
        4$. सही सामान्य नियम विभाजित  \omterm{def:b1:complex:field}{मापांक} बहुत: $ ac
        \equiv bc \pmod n \iff a \equiv b \pmod{n/\gcd(c,n)}$.
  \item \emph{यूक्लिड की प्रमेयिका का दुरुपयोग करना।} $ a \mid bc $ तात्पर्य $ a
        \mid b $ या $ a \mid c $ केवल $ a $ \emph{\omterm{def:b1:arith:prime}{मुख्य}} (या
        \omterm{cor:b1:arith:bezout}{सह अभाज्य} एक कारक के लिए): $6 \mid 4 \times 9$ अभी तक $6$
        विभाजित कोई भी कारक नहीं.
  \item \emph{\omterm{cor:b1:arith:bezout}{सह अभाज्य} एक रिश्ता है, संपत्ति नहीं.} ``$8$ और
        $9$ हैं \omterm{cor:b1:arith:bezout}{सह अभाज्य}'' सच है हालांकि दोनों में से कोई भी सच नहीं है \omterm{def:b1:arith:prime}{मुख्य};
        ``जोड़ी में \omterm{cor:b1:arith:bezout}{सह अभाज्य}'' विश्व स्तर पर '' से अधिक मजबूत है
        \omterm{cor:b1:arith:bezout}{सह अभाज्य}'' ($\gcd(6, 10, 15) = 1$ लेकिन कोई जोड़ी नहीं है
        \omterm{cor:b1:arith:bezout}{सह अभाज्य}).
  \item \emph{घातांक मॉड नहीं रहते $ n $.} में $ a^k \bmod n $,
        घातांक को केवल मॉड्यूलो से कम किया जा सकता है \emph{आदेश}
        का $ a $ (उदाहरण के लिए $ p - 1$ जब फ़र्मेट लागू होता है), कभी नहीं
        modulo $ n $: $2^{10} \bmod 11$ है $1$, नहीं $2^{10 \bmod
        11} = 2^{10}$ --- कमी जो काम करती है वही है
        \cref{ex:b1:arith:congruences} प्रदर्शन करता है.
\end{enumerate}
\end{remark}

\begin{remark}[जहाँ इस अध्याय का प्रयोग किया जाता है]\label{rem:b1:arith:whereused}
यह अध्याय एक टेम्पलेट के साथ-साथ एक टूलबॉक्स भी है। पूरी शृंखला
--- यूक्लिडियन डिवीजन, जीसीडी, बेज़ाउट, गॉस, अद्वितीय गुणनखंडन
--- बहुपदों के लिए शब्दशः पुनः चलाया जाता है \cref{ch:b1:poly},
जहां ``डिग्री'' पूर्ण मूल्य की भूमिका निभाती है; तुलना कर रहे हैं
दो अध्यायों को साथ-साथ पढ़ना दोनों को समझने का सबसे अच्छा तरीका है।
\omterm{def:b1:arith:congruence}{अनुरूपता} कैलकुलस वलय बन जाता है $\Z/n\Z $ में
\cref{ch:b1:structures}, जिसके व्युत्क्रमणीय तत्व
(\cref{prop:b1:arith:invmod}) का पहला गैर-तुच्छ उदाहरण बनाएं
इकाइयों का एक समूह. नीचे सप्ताहांत समस्या में मूल्यांकन वापस आते हैं
(लीजेन्ड्रे का सूत्र) और शक्ति की अतार्किकता का प्रमाण
\cref{ch:b1:reals}. इस वॉल्यूम से परे, बेज़ाउट इनवर्जन मॉड $ n $
सार्वजनिक-कुंजी क्रिप्टोग्राफी का इंजन है, और फ़र्मेट का छोटा
प्रमेय उन आदिम परीक्षणों का पितामह है जो प्रमाणित करते हैं
बड़ा अभाज्य वहां उपयोग किया जाता है.
\end{remark}

\begin{remark}[अंतराल: $\Z $ एक टेम्पलेट के रूप में]\label{rem:b1:arith:interlude}
व्यक्तिगत प्रमेयों से पीछे हटें और निरीक्षण करें
अध्याय की वास्तुकला: एक उपकरण (यूक्लिडियन डिवीजन)
एक वर्गीकरण तैयार किया (उपसमूह)। $ n\Z $), जिसने एक का उत्पादन किया
अस्तित्व प्रमेय (जीसीडी, बेज़ाउट), जिसने एक का निर्माण किया \omterm{def:b1:arith:divides}{भाजकत्व}
कैलकुलस (गॉस), जिसने अद्वितीय गुणनखंडन उत्पन्न किया --- प्रत्येक
फर्श केवल नीचे वाले पर टिका हुआ है। वही बिल्डिंग होगी
इस खंड में अलग-अलग भूतल के साथ दो बार और निर्माण किया गया:
में \cref{ch:b1:poly}, जहां डिग्री द्वारा विभाजन विभाजन का स्थान ले लेता है
आकार के अनुसार और उपरोक्त सभी चीजें दोहराई जाती हैं \emph{प्रतिशब्द}; और, में
लघु, प्रत्येक के अंदर $\Z/n\Z $ का \cref{ch:b1:structures},
जहां उलटापन प्रश्न (इस अध्याय का)
\cref{prop:b1:arith:invmod}) संरचनात्मक बनें \omterm{def:b1:logic:statement}{कथन} के बारे में
छल्ले और क्षेत्र. किसी तर्क को ``द'' के रूप में पहचानना $\Z $-तर्क,
ट्रांसप्लांट'' उन अध्यायों को सीखने का सबसे तेज़ तरीका है --- और
बीजगणित की मूल आदत का पहला स्वाद, प्रमेयों को सिद्ध करना
\emph{अभिगृहीत} वस्तुओं के बजाय.
\end{remark}

\section{अभ्यास}

\begin{exercise}[$\star $]\label{exo:b1:arith:1}
गणना करना $\gcd(1\,001, 777)$ से \omterm{met:b1:arith:euclid}{यूक्लिडियन एल्गोरिथ्म}, और एक बेज़आउट
इसके लिए जोड़ी.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:arith:2}
सिद्ध करो \omterm{def:b1:arith:divides}{भाजकत्व} आधार में नियम $10$: एक पूर्णांक सर्वांगसम मॉड है
$9$ इसके अंकों के योग तक, और मॉड $11$ के वैकल्पिक योग के लिए
इसके अंक. क्या है $123\,456\,789$ आधुनिक $9$ और मॉड $11$?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:arith:3}
में हल करें $\Z $: $91x \equiv 1 \pmod{237}$ \emph{(विस्तारित यूक्लिड)}.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:arith:4}
सभी जोड़े खोजें $(x, y) \in \Z^2$ साथ $17x + 39y = 1$; फिर सभी जोड़े
के साथ $17 x + 39 y = 5$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:arith:5}
इसके लिए सिद्ध करें $ a, b \in \N^*$: $\gcd(a,b) \times
\operatorname{lcm}(a,b) = ab $. \emph{(मूल्यांकन सूत्रों का उपयोग करें
\cref{prop:b1:arith:valuation} और $\min(\alpha,\beta) +
\max(\alpha,\beta) = \alpha + \beta $.)}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:arith:6}
होने देना $ a = 2^{10} \times 3^4 \times 5^2$ और $ b = 2^6 \times 3^7 \times
7$. गणना करना $\gcd(a, b)$, $\operatorname{lcm}(a,b)$, और की संख्या
के सकारात्मक विभाजक $ a $. \emph{(भाजक-गणना सूत्र सिद्ध करें
$\prod_i (\alpha_i + 1)$.)}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:arith:7}
साबित करें कि $\sqrt p $ प्रत्येक के लिए तर्कहीन है \omterm{def:b1:arith:prime}{मुख्य} $ p $, का उपयोग करना
मूल्यांकन: तुलना करें $ v_p $ के दोनों तरफ $ p q^2 = r^2$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:arith:8}
(चीनी शेष समस्या) सभी पूर्णांक ज्ञात करें $ x $ साथ
\[
x \equiv 2 \pmod 7, \qquad x \equiv 5 \pmod{11}.
\]
इसके लिए रास्ते में साबित करें \omterm{cor:b1:arith:bezout}{सह अभाज्य} $ m, n $, की जोड़ी सर्वांगसमताएँ
$ x \equiv a \ (m)$, $ x \equiv b\ (n)$ हमेशा एक समाधान होता है, अनोखा
मॉड $ mn $.\index{चीनी शेषफल प्रमेय}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:arith:9}
गणना करना $3^{1000}$ आधुनिक $7$, और अंतिम दो दशमलव अंक
$7^{100}$ \emph{(मॉड $100 = 4 \times 25$: उपयोग
\cref{exo:b1:arith:8})}.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b1:arith:10}
के लिए $ m, n \in \N^*$, साबित करें कि $\gcd(2^m - 1,\, 2^n - 1) =
2^{\gcd(m,n)} - 1$. \emph{संकेत: पहले दिखाएँ कि शेष $2^m
- 1$ आधुनिक $2^n - 1$ है $2^r - 1$ कहाँ $ r $ का शेष है $ m $ आधुनिक
$ n $; फिर का पालन करें \omterm{met:b1:arith:euclid}{यूक्लिडियन एल्गोरिथ्म}.}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b1:arith:11}
(विल्सन का प्रमेय) चलो $ p $ एक हो \omterm{def:b1:arith:prime}{मुख्य}. साबित करें कि
\[
(p-1)! \equiv -1 \pmod p ,
\]
के प्रत्येक कारक को जोड़कर $(p-1)!$ इसके व्युत्क्रम मॉड के साथ $ p $ और
स्व-युग्मित कारकों की पहचान करना (समाधान)। $ x^2 \equiv 1 \pmod p $
पहला)। व्युत्क्रम की जाँच करें: यदि $ n \geq 2$ क्या नहीं है \omterm{def:b1:arith:prime}{मुख्य}, तब $(n-1)!
\not\equiv -1 \pmod n $.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b1:arith:12}
(फर्मेट नंबर) के लिए $ n \in \N $, होने देना $ F_n = 2^{2^n} + 1$.
\begin{enumerate}
  \item साबित करें कि $ F_0 F_1 \cdots F_{n-1} = F_n - 2$ के लिए $ n \geq
        1$ (प्रेरण)।
  \item निष्कर्ष निकालें कि फ़र्मेट संख्याएँ जोड़ीवार हैं \omterm{cor:b1:arith:bezout}{सह अभाज्य}.
  \item इससे स्वतंत्र दूसरा प्रमाण निकालें
        \cref{thm:b1:arith:euclidprimes}, कि वहाँ हैं
        असीम रूप से अनेक अभाज्य.
\end{enumerate}
\end{exercise}

\section{समस्या: लीजेंड्रे का फॉर्मूला और कुमेर का कैरीज़}

\begin{problem}\label{pb:b1:arith:1}
दशमलव लेखन में कितने शून्य का अंत होता है? $1000!$ --- और, गहरा,
ए की सटीक शक्ति क्या है? \omterm{def:b1:arith:prime}{मुख्य} $ p $ डिवाइडिंग $ n!$, या ए को विभाजित करना
\omterm{def:b1:counting:objects}{द्विपद गुणांक}? संपूर्ण उत्तर दो रत्न हैं
प्रारंभिक अंकगणित: \emph{लीजेंड्रे का सूत्र}\index{लीजेंड्रे का सूत्र} $ v_p(n!) = \sum_{k\geq1}
\lfloor n/p^k \rfloor $, अपने डिजिटल अवतार के साथ $ v_p(n!) = \frac{n
- s_p(n)}{p-1}$, और \emph{कुमेर का प्रमेय}\index{कुमेर का प्रमेय}: $ v_p\binom{m+n}m $ गिनती करता है \emph{किया जाता है} जोड़ते समय
$ m $ और $ n $ आधार में $ p $. यह समस्या दोनों को सिद्ध करती है, जाँचती है
संख्यात्मक रूप से एक दूसरे के विरुद्ध, और शास्त्रीयता का लाभ उठाता है
परिणाम --- अनुगामी शून्य, पास्कल त्रिभुज की समता,
और की दिशा में पहला बाउंड प्रधान संख्या प्रमेय.
लगातार, $ p $ एक है \omterm{def:b1:arith:prime}{मुख्य}, $\floor{x}$ पूर्णांक भाग है, और
$ s_p(n)$ के अंकों के योग को दर्शाता है $ n $ आधार में लिखा है $ p $.

\medskip
\noindent\textbf{Part I --- Floors, valuations, and Legendre's
formula.}
\begin{enumerate}
  \item वार्म-अप: गणना करें $10!$ और इसकी अनुगामी संख्या को पढ़ें
        शून्य; गणना $ v_2(10!)$ और $ v_5(10!)$ सीधे से
        प्रत्येक कारक का गुणनखंडन $1, 2, \dots, 10$.
  \item इसके लिए सिद्ध करें $ x \in \R $ और $ n \in \N^*$,
        $\bigl\lfloor \lfloor x \rfloor / n \bigr\rfloor =
        \lfloor x/n \rfloor $.
  \item साबित करें कि $ v_p(a + b) \geq \min\bigl(v_p(a),
        v_p(b)\bigr)$ सभी के लिए $ a, b \in \N^*$, समानता के साथ
        जब भी $ v_p(a) \neq v_p(b)$.
  \item दिखाएँ कि के गुणजों की संख्या $ m $ में $\intint1n $ है
        $\lfloor n/m \rfloor $.
  \item सिद्ध करना \emph{\omterm{pb:b1:arith:1}{लीजेंड्रे का सूत्र}}: हरएक के लिए $ n \in \N^*$,
        \[
        v_p(n!) = \sum_{k=1}^{\infty}
        \Bigl\lfloor \frac{n}{p^k} \Bigr\rfloor
        \]
        (एक सीमित योग: शब्द एक बार गायब हो जाते हैं $ p^k > n $).
        \emph{प्रत्येक के लिए गिनती करें $ k $, के कारक $\intint1n $
        द्वारा विभाज्य $ p^k $: प्रत्येक बिल्कुल एक इकाई का योगदान देता है
        जिस स्तर पर यह पहुंचता है.}
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The digital form and trailing
zeros.}
\begin{enumerate}[resume]
  \item गणना करना $ v_5(1000!)$ और $ v_2(1000!)$, और निष्कर्ष निकालें: कैसे
        कई शून्य ख़त्म $1000!$?
  \item लीजेंड्रे के सूत्र का डिजिटल रूप सिद्ध करें: लेखन $ n =
        \sum_i a_i p^i $ आधार में $ p $,
        \[
        v_p(n!) = \frac{n - s_p(n)}{p - 1} .
        \]
  \item के लिए दो परिणाम $ p = 2$: बताते हैं कि $2^n $ कभी नहीं
        विभाजित $ n!$, ओर वो $2^{n-1}$ विभाजित $ n!$ बिल्कुल कब
        $ n $ की एक शक्ति है $2$.
  \item दोष बाँधें:दिखाएँ $\frac n{p-1} - \log_p(n) - 1 \leq
        v_p(n!) < \frac n{p-1}$, ताकि $\frac{v_p(n!)}{n} \to
        \frac1{p-1}$: लंबे समय में, एक अनुपात
        $\frac1{p-1}$ एक कारक का $ p $ प्रति यूनिट जमा होता है.
  \item होने देना $ Z(n) = v_5(n!)$ के अनुगामी शून्यों की संख्या हो
        $ n!$. दिखाओ $ Z(n) - Z(n-1) = v_5(n)$, वह निष्कर्ष निकालो $ Z $ छोड़ देता है
        मूल्य $5$ पूरी तरह से (गणना करें $ Z(24)$ और $ Z(25)$), और
        साबित करें कि कोई भी फैक्टोरियल बिल्कुल पांच शून्य पर समाप्त नहीं होता है।
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Kummer's theorem.}
\begin{enumerate}[resume]
  \item साबित करें कि $\lfloor x + y \rfloor - \lfloor x \rfloor -
        \lfloor y \rfloor \in \{0, 1\}$ सभी के लिए $ x, y \in \R $, और
        लिजेंड्रे के सूत्र से निष्कर्ष निकालें
        \[
        v_p\binom{m+n}m
        = \sum_{k\geq1}\Bigl(
        \Bigl\lfloor\frac{m+n}{p^k}\Bigr\rfloor
        - \Bigl\lfloor\frac{m}{p^k}\Bigr\rfloor
        - \Bigl\lfloor\frac{n}{p^k}\Bigr\rfloor\Bigr),
        \]
        प्रत्येक के बराबर पदों का योग $0$ या $1$.
  \item सिद्ध करना \emph{\omterm{pb:b1:arith:1}{कुमेर का प्रमेय}}: द $ k $-उस राशि का -वाँ पद
        बराबर $1$ बिल्कुल जब का जोड़ $ m $ और $ n $ में
        आधार $ p $ स्थिति में ले जाना उत्पन्न करता है $ k $; इस तरह
        $ v_p\binom{m+n}m $ वहन की कुल संख्या है.
        \emph{(लिखना $ m = p^km_1 + m_0$ और $ n = p^kn_1 + n_0$
        साथ $0 \leq m_0, n_0 < p^k $ और निरीक्षण करें $\lfloor (m_0 +
        n_0)/p^k \rfloor $.)}
  \item उसके लिए निष्कर्ष निकालें $0 < j < p^k $:
        \[
        v_p\binom{p^k}{j} = k - v_p(j) ,
        \]
        जोड़ में कैर्री की गिनती करके $ j + (p^k - j)$.
        (विशेष रूप से $ p \mid \binom p j $ के लिए $0 < j < p $: द
        का \omterm{def:b1:arith:prime}{मुख्य} चरण \cref{thm:b1:arith:fermat}, पुनर्प्राप्त।)
  \item साबित करें कि $ v_2\binom{2n}n = s_2(n)$. यह निष्कर्ष निकालें कि
        केंद्रीय \omterm{def:b1:counting:objects}{द्विपद गुणांक} हमेशा सम होता है, और वह
        $\binom{2n}n \equiv 2 \pmod 4$ बिल्कुल कब $ n $ एक शक्ति है
        का $2$.
  \item वेंडरमोंडे की पहचान का उपयोग करके दिखाएँ
        (\cref{exo:b1:counting:7}) और प्रश्न 13, वह
        $\binom{2p}p \equiv 2 \pmod p $ हरएक के लिए \omterm{def:b1:arith:prime}{मुख्य} $ p $.
  \item गणना करना $ v_3\binom{1000}{500}$ दो बार: एक बार कुमेर द्वारा
        (लिखें $500$ आधार में $3$ और अंदर आने वाले सामान को गिनें $500 +
        500$), एक बार लीजेंड्रे के डिजिटल फॉर्म (गणना) द्वारा $ s_3(500)$
        और $ s_3(1000)$); जांचें कि दोनों समान मान देते हैं।
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The parity of Pascal's triangle, and
a prime-density bound.}
\begin{enumerate}[resume]
  \item अंक मानदंड सिद्ध करें: $\binom nk $ है \emph{विषम} यदि
        और केवल यदि प्रत्येक बाइनरी अंक $ k $ अधिक से अधिक है
        का संगत अंक $ n $. अनुरूप बताएं और सिद्ध करें
        के लिए मानदंड $ p \nmid \binom nk $ आधार में $ p $.
  \item वह पंक्ति घटाएँ $ n $ पास्कल के त्रिभुज में बिल्कुल शामिल है
        $2^{s_2(n)}$ विषम प्रविष्टियाँ; पंक्तियों पर सत्यापित करें $4$ और $5$.
  \item सभी आंतरिक प्रविष्टियाँ निकालें $\binom nk $ ($0 < k < n $)
        यदि और केवल यदि हैं तो भी हैं $ n $ की एक शक्ति है $2$.
  \item प्रत्येक को सिद्ध करें \omterm{def:b1:arith:prime}{मुख्य} शक्ति विभाजन $\binom{m+n}m $ पर है
        सबसे $ m + n $: अगर $ p^a \mid \binom{m+n}m $ तब $ p^a \leq
        m + n $. \emph{(कितने अशून्य पदों का योग हो सकता है?
        प्रश्न 11 है?)}
  \item उसका निष्कर्ष निकालो $\binom{2n}n $ विभाजित
        $\operatorname{lcm}(1, 2, \dots, 2n)$, और साथ संयोजित करें
        निचली सीमा $\binom{2n}n \geq \frac{4^n}{2n+1}$
        (जो आप साबित करेंगे: केंद्रीय प्रविष्टि सबसे बड़ी है
        का $2n + 1$ पंक्ति की प्रविष्टियाँ $2n $) प्राप्त करने के लिए
        \[
        \operatorname{lcm}(1, \dots, 2n) \geq \frac{4^n}{2n+1} :
        \]
        पहले पूर्णांकों के सामान्य गुणज बढ़ते हैं
        \emph{तेजी से} --- की पहली मात्रात्मक झलक
        की प्रचुरता अभाज्य.
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Synthesis.}
\begin{enumerate}[resume]
  \item सबसे छोटा खोजें $ n $ ऐसा है कि $ n!$ कम से कम में समाप्त होता है
        $2026$ शून्य. \emph{(अनुमान लगाना $ Z(n) \approx n/4$, फिर
        सटीक सूत्र का उपयोग करके समायोजित करें।)}
  \item एक आखिरी क्रॉस-चेक: वह दिखाओ $7$ करता है \emph{नहीं} विभाजित करना
        $\binom{100}{50}$, सबसे पहले लिखकर $50$ आधार में $7$ और
        जाँच कर रहा है कि जोड़ $50 + 50$ तो फिर, कैरी-फ़्री है
        गणना करके $ v_7(100!)$ और $ v_7(50!)$ लीजेंड्रे के साथ
        सूत्र.
  \item वास्तव में समस्या का उपयोग कहाँ हुआ: (i) अद्वितीय
        गुणनखंडीकरण; (ii) यूक्लिडियन-डिवीजन अपघटन
        $ n = p^k n_1 + n_0$; (iii) गिनती का तर्क
        \cref{ch:b1:counting}? एक-एक वाक्य.
  \item संश्लेषण, एक संक्षिप्त पैराग्राफ में: \omterm{pb:b1:arith:1}{लीजेंड्रे का सूत्र} बदल जाता है
        ए \omterm{def:b1:arith:divides}{भाजकत्व} अंक अंकगणित में प्रश्न, और
        कुमेर की प्रमेय एक व्यक्ति के उत्तर को पढ़ती है
        इसके अलावा --- इस अनुवाद पर टिप्पणी करें, के चेक पर
        प्रश्न 16, और प्रश्न 21 की सीमा क्या बताती है
        के बारे में अभाज्य (पूर्ण \omterm{def:b1:logic:statement}{कथन}, द प्रधान संख्या
        प्रमेय, इस आयतन से कहीं आगे है; बहुपद
        इस चैप्टर के टूलकिट का एनालॉग है
        \cref{ch:b1:poly}).
\end{enumerate}
\end{problem}
