\chapter{दृश्यों}\label{ch:b1:seq}

हाई स्कूल वॉल्यूम में सीमा के साथ अनुक्रमों में हेरफेर किया गया था
अवधारणा को विश्वास पर आधा ले लिया गया। यहां सिद्धांत का पुनर्निर्माण किया गया है
की पूर्णता $\R $ (\cref{ch:b1:reals}): प्रत्येक शास्त्रीय प्रमेय
--- एकस्वर अभिसरण, \omterm{thm:b1:seq:adjacent}{आसन्न अनुक्रम}, बोलजानो--वीयरस्ट्रैस,
कॉची मानदंड --- उस एकल सिद्धांत का एक चेहरा है। अध्याय
द्वारा परिभाषित अनुक्रमों के व्यावहारिक अध्ययन के साथ समाप्त होता है $ u_{n+1} =
f(u_n)$.

\section{अभिसरण}

\begin{definition}[अनुक्रम की सीमा]\label{def:b1:seq:limit}
एक क्रम $(u_n)$ वास्तविक का \emph{अभिसरण} को $\ell \in \R $
कब\index{एक क्रम की सीमा!}
\[
\forall \varepsilon > 0,\ \exists N \in \N,\ \forall n \geq N,
\qquad \abs{u_n - \ell} \leq \varepsilon .
\]
एक लिखता है $ u_n \to \ell $ या $\lim u_n = \ell $. एक क्रम जो करता है
अभिसरण नहीं (किसी भी वास्तविक में) \emph{विचलन}. विचलन \emph{को
$+\infty $}: $\forall M,\ \exists N,\ \forall n \geq N,\ u_n \geq M $
(इसी तरह $-\infty $).
\end{definition}

\begin{example}[एक $\varepsilon $--$ N $ प्रमाण, एक बार लिखा हुआ]\label{ex:b1:seq:epsilonN}
दावा करना: $ u_n = \dfrac{n^2 + 1}{2n^2 - 3} \to \dfrac12$. पहला
त्रुटि को अलग करें:
\[
\Bigl| u_n - \frac12 \Bigr|
= \Bigl| \frac{2(n^2 + 1) - (2n^2 - 3)}{2(2n^2 - 3)} \Bigr|
= \frac{5}{2\,\abs{2n^2 - 3}}
= \frac{5}{2\,(2n^2 - 3)} \quad (n \geq 2).
\]
फिर इसे किसी सरल चीज़ से नियंत्रित करें: के लिए $ n \geq 2$, $2n^2 - 3
\geq n^2$, तो त्रुटि है $\leq \frac{5}{2n^2} \leq \frac 5{2n}$.
दिया गया $\varepsilon > 0$, द \omterm{thm:b1:reals:archimedes}{आर्किमिडीज़ संपत्ति} प्रदान $ N \geq
\max\bigl(2, \frac{5}{2\varepsilon}\bigr)$; के लिए $ n \geq N $ द
त्रुटि है $\leq \varepsilon $. हो गया। समापन अंतर्दृष्टि: ए
$\varepsilon $--$ N $ प्रमाण में ठीक तीन चालें हैं --- गणना करें
त्रुटि, इसे घटती प्राथमिक अभिव्यक्ति से बांधें, हल करें
दहलीज --- और इस अध्याय के प्रमेयों (संचालन,
निचोड़) कोई भी ऐसा प्रमाण दोबारा कभी नहीं लिखता: प्रमेय
तीनों चालों को एक बार और सभी के लिए पैकेज करें।
\end{example}

\begin{example}[अनंत से विचलन, प्रमाणित]\label{ex:b1:seq:toinfinity}
दावा करना: $ u_n = n^2 - 100n \to +\infty $. प्रमुख शब्द का कारक:
$ u_n = n^2\bigl(1 - \frac{100}{n}\bigr) \geq \frac{n^2}{2}$ के लिए
$ n \geq 200$. दिया गया $ M $, लेना $ N = \max\bigl(200,
\lceil\sqrt{2M}\rceil\bigr)$: के लिए $ n \geq N $, $ u_n \geq
\frac{n^2}{2} \geq M $. दो आदतें प्रदर्शन पर हैं: प्रमुख-शब्द
फैक्टरिंग एक प्रतिस्पर्धा को परिवर्तित करती है ($ n^2$ ख़िलाफ़ $-100n $) में ए
एकल पैमाने बार एक कारक की ओर रुझान $1$; और दहलीज
बहुत बड़ा हो सकता है ($ u_{100} = 0$, अनुक्रम पहले भी नकारात्मक है
$ n = 100$) --- से विचलन $+\infty $ एक है \omterm{def:b1:logic:statement}{कथन} के बारे में
पूंछ, किसी भी सीमित मात्रा में दुर्व्यवहार के प्रति उदासीन।
\end{example}

\begin{proposition}[प्रथम गुण]\label{prop:b1:seq:first}
\begin{enumerate}
  \item सीमा, यदि मौजूद है, अद्वितीय है।
  \item एक अभिसारी अनुक्रम परिबद्ध है।
  \item अगर $ u_n \to \ell $, परिमित रूप से अनेक पदों का प्रत्येक संशोधन
        अभिसरण और सीमा को अपरिवर्तित छोड़ देता है।
\end{enumerate}
\end{proposition}

\begin{proof}
(1) यदि $ u_n \to \ell $ और $ u_n \to \ell'$ साथ $\ell \neq \ell'$,
लेना $\varepsilon = \frac{\abs{\ell - \ell'}}{3}$: दोनों से परे
दहलीज़, $\abs{\ell - \ell'} \leq \abs{\ell - u_n} + \abs{u_n -
\ell'} \leq 2\varepsilon = \frac23 \abs{\ell - \ell'}$, बेतुका।

(2)साथ $\varepsilon = 1$: आगे $ N $, $\abs{u_n} \leq \abs\ell +
1$; पहले के बहुत से निश्चित शब्द भी परिबद्ध हैं, इसलिए $\abs{u_n}
\leq \max(\abs{u_0}, \dots, \abs{u_{N-1}}, \abs\ell + 1)$.

(3) विस्तार से: मान लीजिये $ v_n = u_n $ के लिए $ n \geq n_0$ और $ u_n \to
\ell $. दिया गया $\varepsilon > 0$, दहलीज ले लो $ N $ के लिए
$(u_n)$: के लिए $ n \geq \max(N, n_0)$, $\abs{v_n - \ell} = \abs{u_n
- \ell} \leq \varepsilon $. इसलिए $ v_n \to \ell $: परिभाषा
केवल अधिक परिमाणित करता है $ n \geq N $, और कोई भी परिमित उपसर्ग हो सकता है
सीमा को बढ़ाने की कीमत पर अधिलेखित किया गया। (यही कारण है
परिकल्पनाएँ ``सभी के लिए बड़ी $ n $'' इसमें सर्वत्र पर्याप्त है
अध्याय.)
\end{proof}

\begin{theorem}[सीमा पर संचालन]\label{thm:b1:seq:operations}
अगर $ u_n \to \ell $ और $ v_n \to m $, तब
\[
u_n + v_n \to \ell + m, \qquad u_n v_n \to \ell m, \qquad
\frac{u_n}{v_n} \to \frac{\ell}{m} \ (\text{if } m \neq 0),
\qquad \abs{u_n} \to \abs\ell .
\]
\end{theorem}

\begin{proof}
\emph{जोड़:} $\abs{(u_n + v_n) - (\ell + m)} \leq \abs{u_n - \ell} +
\abs{v_n - m} \leq 2\varepsilon $ बड़ी सीमा से परे.
\emph{उत्पाद:} लिखना
\[
u_n v_n - \ell m = (u_n - \ell)\,v_n + \ell\,(v_n - m);
\]
$(v_n)$ कुछ से घिरा हुआ है $ B $ (\cref{prop:b1:seq:first}), तो
दाईं ओर है $\leq B\abs{u_n - \ell} + \abs{\ell}\,\abs{v_n - m}$,
मनमाने ढंग से छोटा. \emph{भागफल:} यह इलाज के लिए पर्याप्त है $\frac
1{v_n}$. साथ $\varepsilon = \frac{\abs m}{2}$: कुछ से परे $ N_0$,
$\abs{v_n} \geq \frac{\abs m}{2}$, इसलिए
\[
\Bigl| \frac{1}{v_n} - \frac 1m \Bigr|
= \frac{\abs{m - v_n}}{\abs{v_n m}}
\leq \frac{2}{m^2}\,\abs{v_n - m} \longrightarrow 0 .
\]
\emph{निरपेक्ष मूल्य:} $\bigl|\abs{u_n} - \abs\ell\bigr| \leq
\abs{u_n - \ell}$ (विपरीत त्रिभुज असमानता,
\cref{prop:b1:complex:rules}).
\end{proof}

\begin{example}[संचालन और एक बीजगणितीय युक्ति]\label{ex:b1:seq:conjugatetrick}
गणना करना $\lim\,\bigl(\sqrt{n^2 + n} - n\bigr)$. दो टुकड़े
अलग से प्रवृत्त होते हैं $+\infty $: संचालन प्रमेय कुछ नहीं कहता
उनके अंतर के बारे में (ए \emph{अनिश्चित रूप}). गुणा करें
संयुग्म द्वारा:
\[
\sqrt{n^2 + n} - n
= \frac{(n^2 + n) - n^2}{\sqrt{n^2 + n} + n}
= \frac{n}{\sqrt{n^2+n} + n}
= \frac{1}{\sqrt{1 + \frac1n} + 1} .
\]
अब सब कुछ एक हो गया है: $\sqrt{1 + \frac1n} \to 1$, क्योंकि
$0 \leq \sqrt{1 + h} - 1 = \frac{h}{\sqrt{1+h} + 1} \leq h $
(फिर से \omterm{def:b1:complex:field}{संयुग्मित} करें, फिर निचोड़ें $ h = \frac1n $); फिर
संचालन प्रमेय सीमा देता है $\frac{1}{1 + 1} =
\frac12$. समापन अंतर्दृष्टि: संचालन प्रमेय एक नहीं है
सभी सीमाओं के लिए कैलकुलेटर --- अनिश्चित रूप ($\infty -
\infty $, $\frac00$, $0 \times \infty $, $1^\infty $) पहले होना चाहिए
\emph{तब्दील} बीजगणित द्वारा (\omterm{def:b1:complex:field}{संयुग्मित}, प्रमुख गुणनखंडन)।
अवधि) जब तक कि प्रत्येक टुकड़ा अभिसरण न हो जाए; के लिए व्यवस्थित मशीन
प्रतिरोधी मामलों का स्पर्शोन्मुख विस्तार है
\cref{ch:b1:taylor}.
\end{example}

\begin{theorem}[सीमाएँ और व्यवस्था]\label{thm:b1:seq:order}
\begin{enumerate}
  \item अगर $ u_n \leq v_n $ सभी बड़े के लिए $ n $, और फिर दोनों एक हो जाते हैं
        $\lim u_n \leq \lim v_n $. (सख्त असमानताएं होती हैं \emph{नहीं}
        सीमा से गुजरें: $\frac 1n > 0$ लेकिन $\lim = 0$.)
  \item (निचोड़ प्रमेय\index{निचोड़ प्रमेय}) अगर $ u_n \leq w_n
        \leq v_n $ सभी बड़े के लिए $ n $ और $ u_n, v_n \to \ell $, तब
        $ w_n \to \ell $.
  \item अगर $ u_n \to \ell > 0$, तब $ u_n > \frac\ell2 > 0$ सबके लिए
        बड़ा $ n $.
\end{enumerate}
\end{theorem}

\begin{proof}
(1) मान लीजिये $\ell = \lim u_n > m = \lim v_n $; साथ $\varepsilon =
\frac{\ell - m}{3}$, बड़े शब्द संतुष्ट करते हैं $ v_n \leq m + \varepsilon <
\ell - \varepsilon \leq u_n $, विरोधाभासी $ u_n \leq v_n $.

(2) दहलीज से परे: $\ell - \varepsilon \leq u_n \leq w_n \leq
v_n \leq \ell + \varepsilon $.

(3) है \cref{def:b1:seq:limit} साथ $\varepsilon = \frac\ell2$.
\end{proof}

\begin{example}[दो निचोड़]\label{ex:b1:seq:squeezes}
(मैं) $\dfrac{\sin n}{n} \to 0$: से $-\frac1n \leq \frac{\sin
n}{n} \leq \frac1n $, दोनों दीवारें ढह गईं $0$ --- कोई जरूरत नहीं
अनियमित अंश को बिल्कुल समझें। (ii) $(2^n +
3^n)^{1/n} \to 3$: अंदर का फ्रेम,
\[
3^n \leq 2^n + 3^n \leq 2\cdot3^n
\quad\Longrightarrow\quad
3 \leq (2^n + 3^n)^{1/n} \leq 3\cdot 2^{1/n} ,
\]
और $2^{1/n} = \eu^{\frac{\ln 2}{n}} \to 1$ (से संबंधित $5^{1/n}$ में
\cref{exo:b1:seq:2}): निचोड़ बचाता है $3$. समापन
अंतर्दृष्टि: प्रतिस्पर्धी घातांकों का योग उसके जैसा व्यवहार करता है
\emph{सबसे बड़ा} अवधि --- छोटे वाले एक द्वारा अवशोषित होते हैं
हानिरहित स्थिर कारक, जो $ n $-th रूट फिर मिट जाता है।
\end{example}

\section{मोनोटोन अनुक्रम}

\begin{theorem}[मोनोटोन सीमा प्रमेय]\label{thm:b1:seq:monotone}
ऊपर से घिरा हुआ एक बढ़ता हुआ क्रम अभिसरित होता है $\sup\{u_n : n \in
\N\}$; एक बढ़ता हुआ क्रम ऊपर से घिरा नहीं है, जिससे विचलन होता है
$+\infty $. (आईना \omterm{def:b1:logic:statement}{कथन} घटने के लिए
अनुक्रम।)\index{मोनोटोन सीमा प्रमेय}
\end{theorem}

\begin{proof}
होने देना $ s = \sup\{u_n\}$ (\cref{thm:b1:reals:sup}). दिया गया $\varepsilon >
0$, द $\varepsilon $-विशेषता
(\cref{prop:b1:reals:epsilon}) पैदावार $ N $ साथ $ u_N > s -
\varepsilon $; एकरसता से, $ s - \varepsilon < u_N \leq u_n \leq
s $ सभी के लिए $ n \geq N $: अभिसरण $ s $. यदि \omterm{def:b1:reals:bounds}{असीम}ित: प्रत्येक के लिए
$ M $ कुछ $ u_N > M $, और एकरसता बाद के सभी शब्दों को ऊपर रखती है $ M $.
\end{proof}

\begin{example}[एक अस्तित्व मशीन के रूप में मोनोटोन प्रमेय]\label{ex:b1:seq:existencemachine}
होने देना $ u_n = \prod_{k=1}^{n} \bigl(1 + \frac{1}{2^k}\bigr)$. प्रत्येक
कारक से अधिक है $1$, इसलिए $(u_n)$ यह बढ़ रहा है। ऊपर बाउंड? लो
लघुगणक और उपयोग $\ln(1 + x) \leq x $
(\cref{ex:b1:derivative:convexineq} इसका अनुमान लगाता है; या
कच्चा $1 + x \leq \eu^x $ हाई स्कूल वॉल्यूम से):
\[
\ln u_n = \sum_{k=1}^{n} \ln\Bigl(1 + \frac{1}{2^k}\Bigr)
\leq \sum_{k=1}^{n} \frac{1}{2^k} < 1 ,
\]
इसलिए $ u_n < \eu $. बढ़ती और सीमाबद्ध: $(u_n)$ कुछ में परिवर्तित हो जाता है
$\ell \in \intoc{u_1}{\eu}$ --- एक पूरी तरह से अच्छी तरह से परिभाषित वास्तविकता
कोई बंद प्रपत्र न दिखने वाला नंबर ($\ell = 2.384\dots $). द
समापन अंतर्दृष्टि: मोनोटोन सीमा प्रमेय सबसे सस्ता है
विश्लेषण में अस्तित्व मशीन; इसका नाम रखा गया $\eu $ स्वयं
(\cref{ex:b1:seq:e} नीचे), और अंदर \cref{ch:b1:series} यह होगा
प्रत्येक सकारात्मक शृंखला का अभिसरण केवल मात्र से तय करें
सीमाबद्धता
\end{example}

\begin{theorem}[आसन्न अनुक्रम]\label{thm:b1:seq:adjacent}
होने देना $(a_n)$ बढ़ रहा हो, $(b_n)$ घट रहा है, साथ $ b_n - a_n \to
0$. फिर दोनों एक हो जाते हैं, एक में \emph{सामान्य} आप LIMIT $\ell $, और $ a_n
\leq \ell \leq b_n $ सभी के लिए $ n $.\index{आसन्न अनुक्रम}
\end{theorem}

\begin{proof}
पहला, $ a_n \leq b_n $ सभी के लिए $ n $: क्रम $(b_n - a_n)$ है
घट रहा है और प्रवृत्त होता है $0$, सो है $\geq 0$ (एक नकारात्मक शब्द होगा
इसे नीचे फ्रीज करें $0$). तब $(a_n)$ बढ़ रहा है और ऊपर से घिरा हुआ है
$ b_0$: यह कुछ में परिवर्तित हो जाता है $\ell $ (\cref{thm:b1:seq:monotone});
वैसे ही $(b_n) \to \ell'$; और $\ell' - \ell = \lim (b_n - a_n) =
0$. असमानताएँ $ a_n \leq \ell \leq b_n $ से अनुसरण करें
एकरसता ($\ell = \sup a_k \geq a_n $, वगैरह।)।
\end{proof}

और उनके बीच का अंतर कम हो जाता है $0$. प्रत्येक \omterm{prop:b1:reals:intervals}{अंतराल}
$\intcc{a_n}{b_n}$ इसमें सभी बाद वाले और सामान्य सीमा शामिल है
$\ell $ प्रत्येक में अद्वितीय बिंदु बचा हुआ है \omterm{prop:b1:reals:intervals}{अंतराल} --- द
बोल्ज़ानो-वीयरस्ट्रैस के द्वंद्वात्मक प्रमाणों के पीछे की तस्वीर
नीचे और मध्यवर्ती मूल्य प्रमेय के
\cref{ch:b1:continuity}.}
\end{omfigure}

\begin{example}[जो नंबर $\eu $]\label{ex:b1:seq:e}
\omterm{def:b1:logic:sets}{तय करना} $ a_n = \sum_{k=0}^{n} \frac{1}{k!}$ और $ b_n = a_n + \frac{1}{n
\cdot n!}$ ($ n \geq 1$). तब $(a_n)$ बढ़ता है; और
\[
b_{n+1} - b_n = \frac{1}{(n+1)!} + \frac{1}{(n+1)(n+1)!} -
\frac{1}{n\,n!}
= \frac{n(n+1) + n - (n+1)^2}{n(n+1)(n+1)!}
= \frac{-1}{n(n+1)(n+1)!} < 0 ,
\]
इसलिए $(b_n)$ घट जाती है, और $ b_n - a_n \to 0$: नज़दीक। उनका आम
सीमा (यहाँ परिभाषा के अनुसार) संख्या है $\eu \approx 2.71828$; द
असमानताएँ $ a_n < \eu < b_n $ साबित करने के लिए काफी तेज़ हैं $\eu \notin
\Q $ (\cref{exo:b1:seq:9}).
\end{example}

\section{अनुवर्ती और बोल्ज़ानो--वीयरस्ट्रैस}

\begin{definition}[परिणाम को]\label{def:b1:seq:subsequence}
ए \emph{परिणाम को}\index{परिणाम को} का $(u_n)$ एक क्रम है
$(u_{\varphi(n)})$ कहाँ $\varphi \colon \N \to \N $ सख्ती से है
बढ़ रहा है (ध्यान दें) $\varphi(n) \geq n $, प्रेरण द्वारा)।
\end{definition}

\begin{proposition}\label{prop:b1:seq:subsequences}
अगर $ u_n \to \ell $ ($\ell \in \R $ या $\pm\infty $), प्रत्येक \omterm{def:b1:seq:subsequence}{परिणाम को}
आदत है $\ell $. नतीजतन, दो के साथ एक अनुक्रम परिणाम का
अलग-अलग सीमाएँ अलग-अलग होती हैं। इसके विपरीत, यदि $(u_{2n})$ और
$(u_{2n+1})$ दोनों एक हो जाते हैं \emph{वही} $\ell $, तब $ u_n \to
\ell $.
\end{proposition}

\begin{proof}
दहलीज से परे $ N $ के लिए $(u_n)$, सभी सूचकांक $\varphi(n) \geq n
\geq N $ अर्हता प्राप्त करें (असमानता $\varphi(n) \geq n $ है
प्रेरण में उल्लेख किया गया \cref{def:b1:seq:subsequence}: $\varphi(0)
\geq 0$, और $\varphi(n+1) > \varphi(n) \geq n $ बल
$\varphi(n+1) \geq n + 1$). व्युत्क्रम के लिए: दिया गया
$\varepsilon $, दो दहलीज ले लो $ N_0$ (सम) और $ N_1$
(कठिनाइयाँ); एक मनमाना सूचकांक $ n \geq \max(2N_0, 2N_1 + 1)$ है
या तो सम, $ n = 2k $ साथ $ k \geq N_0$, या अजीब, $ n = 2k+1$ साथ
$ k \geq N_1$ --- दोनों ही मामलों में $\abs{u_n - \ell} \leq
\varepsilon $: प्रत्येक सूचकांक दो में से एक द्वारा कवर किया गया है
परिणाम, और यही संपूर्ण मुद्दा है।
\end{proof}

\begin{example}[बाद की सीमाएँ]\label{ex:b1:seq:sublimits}
के लिए $ u_n = (-1)^n \frac{n}{n+1}$: सम \omterm{def:b1:seq:subsequence}{परिणाम को} आदत है
$1$, अजीब बात है $-1$, इसलिए अनुक्रम अलग हो जाता है --- लेकिन ऐसा होता है
इसलिए एक संगठित तरीके से, दो मूल्यों के आसपास क्लस्टरिंग $\pm 1$.
के लिए $ u_n = \cos\frac{2\pi n}{3}$: तीन परिणाम का
सूचकांक $3k $, $3k + 1$, $3k + 2$ स्थिर हैं, बराबर हैं $1$,
$-\frac12$, $-\frac12$; \omterm{def:b1:logic:sets}{तय करना} अनुवर्ती सीमाओं का है $\{1,
-\frac12\}$. समापन अंतर्दृष्टि: ए \emph{घिरे} क्रम
ठीक उसी समय अभिसरण होता है जब इसकी एकल अनुवर्ती सीमा होती है
(\cref{exo:b1:seq:8}); किसी बंधे हुए अनुक्रम का सदैव विचलन
मतलब कम से कम दो क्लस्टर, और नीचे बोलजानो--वीयरस्ट्रैस
गारंटी देता है कि कम से कम एक तो है।
\end{example}

\begin{theorem}[बोल्ज़ानो--वीयरस्ट्रैस]\label{thm:b1:seq:bw}
वास्तविकताओं के प्रत्येक बंधे हुए अनुक्रम में एक अभिसरण होता है
\omterm{def:b1:seq:subsequence}{परिणाम को}.\index{बोल्ज़ानो--वीयरस्ट्रैस प्रमेय}
\end{theorem}

\begin{proof}
होने देना $ u_n \in \intcc{a}{b}$ सभी के लिए $ n $. द्वारा नेस्टेड खंड बनाएं
द्विभाजन: सेट $\intcc{a_0}{b_0} = \intcc{a}{b}$; दिया गया
$\intcc{a_k}{b_k}$ युक्त $ u_n $ अनंत अनेकों के लिए $ n $, में से एक
इसके दो हिस्से अभी भी मौजूद हैं $ u_n $ अनंत अनेकों के लिए $ n $ --- इसे कहते हैं $\intcc{a_{k+1}}{b_{k+1}}$. क्रम $(a_k)$, $(b_k)$ हैं
आसन्न ($ b_k - a_k = \frac{b-a}{2^k} \to 0$), सामान्य सीमा के साथ
$\ell $ (\cref{thm:b1:seq:adjacent}).

निकालें: चुनें $\varphi(0)$ साथ $ u_{\varphi(0)} \in
\intcc{a_0}{b_0}$, फिर, आगमनात्मक रूप से, $\varphi(k+1) > \varphi(k)$
साथ $ u_{\varphi(k+1)} \in \intcc{a_{k+1}}{b_{k+1}}$ ---संभव
चूँकि उस खंड में अपरिमित रूप से अनेक पद हैं। तब $ a_k \leq
u_{\varphi(k)} \leq b_k $, और निचोड़ प्रमेय देता है
$ u_{\varphi(k)} \to \ell $.
\end{proof}

\begin{remark}[बोलजानो-वीयरस्ट्रैस क्या कहते हैं और क्या नहीं]
यह \emph{करता है} कहते हैं: केवल सीमा से, कुछ \omterm{def:b1:seq:subsequence}{परिणाम को}
अभिसरण --- बिना किसी सूत्र के अस्तित्व, द्विभाजन प्रमाण के रूप में
स्पष्ट करता है (कुछ भी हमें नहीं बताता \emph{कौन} सूचकांक जीवित रहते हैं)। ऐसा होता है \emph{नहीं} कहें कि सीमा अद्वितीय है: $((-1)^n)$ है
परिणाम में एकत्रित होना $1$ और को $-1$, और यह \omterm{def:b1:logic:sets}{तय करना} का
अनुवर्ती सीमाएँ अनंत भी हो सकती हैं
(\cref{ex:b1:seq:sublimits}, और सभी कैंटर \omterm{def:b1:logic:sets}{तय करना} में
\cref{pb:b1:topology:1}). यह \omterm{def:b1:reals:bounds}{असीम}ितता से जीवित नहीं रहता:
$(n)$ कोई अभिसरण नहीं है \omterm{def:b1:seq:subsequence}{परिणाम को} बिल्कुल --- हालांकि कोई कर सकता है
हमेशा एक निकालें \omterm{def:b1:seq:subsequence}{परिणाम को} की ओर रुझान $+\infty $ या $-\infty $
किसी भी \omterm{def:b1:reals:bounds}{असीम}ित अनुक्रम से (चुनें $\varphi(k)$ साथ
$ u_{\varphi(k)} \geq k $, कहना)। सही ढंग से प्रयुक्त, प्रमेय एक है
\emph{अस्तित्व पंप}: यह कॉची के मूल बिंदु पर प्रकट होता है
नीचे मानदंड, हेन के प्रमेय का, और चरम मूल्य का
प्रमेय --- हमेशा एक बिंदु उत्पन्न करने के लिए जो कोई स्पष्ट नहीं है
निर्माण प्रस्ताव.
\end{remark}

\section{कॉची अनुक्रम और पूर्णता}

\begin{definition}\label{def:b1:seq:cauchy}
एक क्रम $(u_n)$ एक है \emph{कॉची अनुक्रम}\index{कॉची अनुक्रम}
जब इसकी शर्तें मनमाने ढंग से करीब हो जाती हैं \emph{एक दूसरे से}:
\[
\forall \varepsilon > 0,\ \exists N,\ \forall p, q \geq N,
\qquad \abs{u_p - u_q} \leq \varepsilon .
\]
\end{definition}

\begin{example}[कॉची संपत्ति को हाथ से सत्यापित करना]\label{ex:b1:seq:cauchyhand}
होने देना $ u_n = \sum_{k=0}^{n} \frac{\cos k}{2^k}$ --- कोई एकरसता नहीं,
कोई अनुमान लगाने योग्य सीमा नहीं. के लिए $ p > q $:
\[
\abs{u_p - u_q}
= \Bigl| \sum_{k=q+1}^{p} \frac{\cos k}{2^k} \Bigr|
\leq \sum_{k=q+1}^{p} \frac{1}{2^k}
< \frac{1}{2^{q}} ,
\]
त्रिभुज असमानता से, $\abs{\cos k} \leq 1$ और एक परिमित
ज्यामितीय योग. दिया गया $\varepsilon > 0$, चुनना $ N $ साथ $2^{-N}
\leq \varepsilon $: सभी \omterm{prop:b1:reals:intervals}{अंतराल}ों से परे $ N $ हैं $\leq \varepsilon $,
अनुक्रम कॉची है, इसलिए किसी सीमा तक अभिसरण नहीं करता है
बंद रूप में नाम दे सकते हैं, जो बिल्कुल सही बात है। समापन
अंतर्दृष्टि: वेतन वृद्धि का ज्यामितीय वर्चस्व मानक है
कॉची संपत्ति अर्जित करने का तरीका, और \cref{ch:b1:series} होगा
तर्क को ``पूर्ण अभिसरण का तात्पर्य'' के रूप में बोतलबंद करें
अभिसरण''।
\end{example}

\begin{theorem}[की पूर्णता $\R $]\label{thm:b1:seq:complete}
वास्तविकताओं का एक क्रम अभिसरित होता है यदि और केवल यदि यह एक है कॉची
क्रम.\index{completeness of R@completeness of $\R $}
\end{theorem}

\begin{proof}
($\Rightarrow $) अगर $ u_n \to \ell $: के लिए दहलीज से परे
$\frac\varepsilon2$, $\abs{u_p - u_q} \leq \abs{u_p - \ell} +
\abs{\ell - u_q} \leq \varepsilon $.

($\Leftarrow $) होने देना $(u_n)$ कॉची हो. \emph{यह बंधा हुआ है}: साथ
$\varepsilon = 1$, आगे $ N $ सभी शर्तें भीतर निहित हैं $1$ का $ u_N $, और
सिर सीमित है. \emph{निकालना}: द्वारा
\cref{thm:b1:seq:bw}, कुछ \omterm{def:b1:seq:subsequence}{परिणाम को} $ u_{\varphi(n)} \to \ell $.
\emph{निष्कर्ष निकालना}: दिया गया $\varepsilon > 0$, लेना $ N $ (कौची, के लिए
$\frac\varepsilon2$) और $ n \geq N $ साथ $\abs{u_{\varphi(n)} - \ell}
\leq \frac\varepsilon2$ और $\varphi(n) \geq N $; फिर हर एक के लिए $ p
\geq N $:
\[
\abs{u_p - \ell} \leq \abs{u_p - u_{\varphi(n)}} +
\abs{u_{\varphi(n)} - \ell} \leq \varepsilon . \qedhere
\]
\end{proof}

\begin{remark}
मानदंड का मूल्य: यह अभिसरण को प्रमाणित करता है \emph{बिना
सीमा का नामकरण}. यह खत्म हो जाता है $\Q $ (दशमलव काट-छाँट)
$\sqrt 2$ फॉर्म ए \omterm{def:b1:seq:cauchy}{कॉची अनुक्रम} बिना किसी तर्कसंगतता के तर्कसंगतता का
सीमा): पूर्णता का गुण है $\R $, के बराबर \omterm{def:b1:reals:bounds}{ऊपरी सीमा} स्वयंसिद्ध. यह के अभिसरण के पीछे का कार्यकर्ता भी है
शृंखला (\cref{ch:b1:series}).
\end{remark}

\begin{example}[एक अदृश्य सीमा के साथ एक कॉची अनुक्रम]\label{ex:b1:seq:basel}
होने देना $ S_n = \sum_{k=1}^{n} \frac{1}{k^2}$. के लिए $ p > q \geq 1$:
\[
S_p - S_q = \sum_{k=q+1}^{p} \frac{1}{k^2}
\leq \sum_{k=q+1}^{p} \frac{1}{k(k-1)}
= \sum_{k=q+1}^{p} \Bigl(\frac{1}{k-1} - \frac 1k\Bigr)
= \frac 1q - \frac 1p < \frac 1q ,
\]
इतना परे $ N > \frac1\varepsilon $ सभी \omterm{prop:b1:reals:intervals}{अंतराल} हैं $\leq \varepsilon $:
$(S_n)$ कॉची है, इसलिए अभिसरण होता है। ध्यान दें कि अभी क्या हुआ: हम
सिद्ध किया कि एक विशिष्ट वास्तविक संख्या बिना किसी नाम के अस्तित्व में है
इसके लिए. (यह है $\frac{\pi^2}{6}$ --- की एक प्रतिष्ठित पहचान
यूलर, वर्ष 2 खंड में सिद्ध; इस अध्याय में कुछ भी नहीं हो सकता
हमें वह बताएं।) श्रम का यह विभाजन --- अब अस्तित्व,
पहचान बाद में, यदि कभी हो --- कॉची मानदंड की संपूर्णता है
बिंदु, और श्रृंखला के सिद्धांत का इंजन
\cref{ch:b1:series}.
\end{example}

\section{आवर्ती क्रम}

\begin{method}[पढ़ना $ u_{n+1} = f(u_n)$]\label{met:b1:seq:recurrent}
दिया गया $ f $ और एक प्रारंभिक बिंदु $ u_0$:\index{आवर्ती क्रम}
\begin{enumerate}
  \item \emph{स्थिर \omterm{prop:b1:reals:intervals}{अंतराल}:} एक खोजें \omterm{prop:b1:reals:intervals}{अंतराल} $ I $ साथ $ f(I)
        \subseteq I $ युक्त $ u_0$: तो सब $ u_n \in I $ (द्वारा
        प्रेरण)।
  \item \emph{उम्मीदवार की सीमाएँ:} अगर $ u_n \to \ell \in I $ और $ f $ है
        पर निरंतर $\ell $ (\cref{ch:b1:continuity}), तब $\ell $
        एक है \emph{नियत बिन्दु}: $ f(\ell) = \ell $. हल करना $ f(x) = x $.
  \item \emph{एकरसता:} अगर $ f $ बढ़ता जा रहा है $ I $, तब
        $(u_n)$ मोनोटोनिक है (यदि बढ़ रहा है)। $ u_1 \geq u_0$,
        अन्यथा घट रहा है); सीमाबद्धता के साथ संयुक्त,
        \cref{thm:b1:seq:monotone} निष्कर्ष. अगर $ f $ घट रहा है,
        दोनों का अध्ययन करें परिणाम $(u_{2n})$ और $(u_{2n+1})$,
        जो कि एकरस हैं $ f \circ f $.
  \item \emph{त्रुटि नियंत्रण:} एक असमानता $\abs{f(x) - \ell} \leq
        k\abs{x - \ell}$ साथ $ k < 1$ देता है $\abs{u_n - \ell} \leq
        k^n \abs{u_0 - \ell} \to 0$ सीधे.
\end{enumerate}
\end{method}

\begin{example}[बगुले की विधि]\label{ex:b1:seq:heron}
होने देना $ u_0 = 2$ और $ u_{n+1} = \dfrac12\Bigl(u_n +
\dfrac{2}{u_n}\Bigr)$: के लिए प्राचीन एल्गोरिदम
$\sqrt 2$.\index{बगुले की विधि}
\begin{itemize}
  \item \emph{स्थिरता:} के लिए $ x > 0$, अंकगणित--ज्यामितीय माध्य
        असमानता देती है $\frac12(x + \frac2x) \geq \sqrt{x \cdot
        \frac 2x} = \sqrt 2$; इसलिए $ I = \intco{\sqrt 2}{+\infty}$ है
        स्थिर और समाहित है $ u_1$ (वास्तव में $ u_1 = \frac32 \geq
        \sqrt 2$).
  \item \emph{एकरसता:} के लिए $ x \geq \sqrt 2$, $\;x - f(x) =
        \frac{x^2 - 2}{2x} \geq 0$: से क्रम घटता जाता है
        $ u_1$ पर, और नीचे से घिरा हुआ है $\sqrt 2$: यह अभिसरण करता है।
  \item \emph{सीमा:} निश्चित बिंदु हल करते हैं $ x = \frac12(x +
        \frac2x)$, यानी\ $ x^2 = 2$: पर $ I $, $\ell = \sqrt 2$.
  \item \emph{रफ़्तार:} $ u_{n+1} - \sqrt 2 = \frac{(u_n -
        \sqrt2)^2}{2u_n} \leq \frac{(u_n - \sqrt 2)^2}{2\sqrt 2}$:
        मोटे तौर पर सही अंकों की संख्या \emph{दोगुना हो जाता है} प्रत्येक पर
        चरण (द्विघात अभिसरण)।
\end{itemize}
\end{example}

\begin{remark}[सीमाओं के साथ सामान्य ख़तरे]
चार क्लासिक्स. (मैं) \emph{छोटे कदमों का अर्थ अभिसरण नहीं है}:
$ u_{n+1} - u_n \to 0$ कॉची संपत्ति की तुलना में बहुत कमजोर है ---
हार्मोनिक रकम $ H_n $ कदम हैं $\frac{1}{n+1} \to 0$ अभी तक
की ओर विचलन $+\infty $ (\cref{exo:b1:seq:5}); कॉची स्थिति
नियंत्रण $\abs{u_p - u_q}$ के लिए \emph{सभी} बड़े जोड़े, नहीं
लगातार वाले. (ii) \emph{कठोर असमानताएँ मर जाती हैं
सीमा}: से $ u_n < v_n $ सभी के लिए $ n $ एक ही मिलता है $\lim u_n
\leq \lim v_n $ (\cref{thm:b1:seq:order}); $\frac1n > 0$ और अभी तक
$\lim = 0$. (iii) \emph{परिबद्ध अभिसरण नहीं है}: $((-1)^n)$
घिरा हुआ है और अलग हो गया है; सीमाबद्धता प्लस \emph{दिष्टता}
अभिसरण, केवल सीमाबद्धता ही अभिसरण की गारंटी देती है
\omterm{def:b1:seq:subsequence}{परिणाम को} (\cref{thm:b1:seq:bw}). (iv) \emph{निश्चित-बिंदु
समीकरण पहले नहीं, दूसरे स्थान पर आता है}: के लिए $ u_{n+1} = f(u_n)$,
सुलझाना $ f(\ell) = \ell $ सीमा की पहचान करता है \emph{केवल बाद}
अभिसरण सिद्ध होता है। पुनरावृत्ति $ u_{n+1} = 2u_n $ के पास है
अद्वितीय निश्चित बिंदु $\ell = 0$, फिर भी से $ u_0 = 1$ क्रम
की ओर दौड़ता है $+\infty $: समीकरण $\ell = 2\ell $ कभी नहीं था
एक सीमा का हकदार. संचालन का क्रम, हमेशा: अस्तित्व
प्रथम (\cref{met:b1:seq:recurrent}, चरण 1--3), पहचान
दूसरा.
\end{remark}

\begin{example}[एक घटती हुई $ f $: स्वर्णिम पुनरावृत्ति]\label{ex:b1:seq:golden}
होने देना $ u_0 = 1$ और $ u_{n+1} = \dfrac{1}{1 + u_n}$. यहाँ $ f(x) =
\frac{1}{1+x}$ है \emph{घटते}, इसलिए अनुक्रम नहीं है
मोनोटोनिक (यह अपनी सीमा के आसपास बदलता रहता है); संकुचन चरण
का \cref{met:b1:seq:recurrent} सही उपकरण है. स्थिरता: यदि
$ x \in \intcc{\frac12}{1}$ तब $1 + x \in
\intcc{\frac32}{2}$, इसलिए $ f(x) \in \intcc{\frac12}{\frac23}
\subseteq \intcc{\frac12}{1}$, और $ u_1 = \frac12$ डालता है
वहाँ पूरा क्रम. नियत बिन्दु: $\ell = \frac{1}{1+\ell}$
साथ $\ell > 0$ देता है $\ell^2 + \ell - 1 = 0$, यानी
\[
\ell = \frac{\sqrt5 - 1}{2} = 0.6180\dots
\]
(स्वर्णिम अनुपात का व्युत्क्रम)। संकुचन: के लिए $ x, y \in
\intcc{\frac12}{1}$,
\[
\abs{f(x) - f(y)} = \frac{\abs{x - y}}{(1+x)(1+y)}
\leq \frac{\abs{x-y}}{(3/2)^2} = \frac49\,\abs{x - y} ,
\]
इसलिए $\abs{u_n - \ell} \leq \bigl(\frac49\bigr)^{n-1}\abs{u_1 -
\ell} \to 0$: अभिसरण, ज्यामितीय गति के साथ, नहीं
एकरसता की आवश्यकता. समापन अंतर्दृष्टि: मोनोटोन तरीके और
संकुचन विधियाँ आवर्ती दुनिया को उनके बीच विभाजित करती हैं ---
बढ़ रहा है $ f $ घटते हुए मोनोटोन कक्षाएँ देता है $ f $ देता है
वैकल्पिक कक्षाएँ लिप्सचिट्ज़ स्थिरांक द्वारा नियंत्रित की जाती हैं $< 1$ (द
व्यवस्थित सिद्धांत है \cref{exo:b1:derivative:11}).
\end{example}

\begin{remark}[इस खंड के अंदर परिप्रेक्ष्य]
अनुक्रम शेष आयतन को मापने वाला उपकरण है
प्रत्येक वस्तु को धारण करता है। में \cref{ch:b1:topology} वे
\emph{चिह्नित करना} बंदता और सघनता; में
\cref{ch:b1:continuity} वे कार्यों की सीमा का परिवहन करते हैं; में
\cref{ch:b1:integration} रीमैन योग अभिसरण अनुक्रम हैं
अभिन्न को; \cref{ch:b1:series} \emph{है} एक का सिद्धांत
अनुक्रमों का विशेष वर्ग, आंशिक योग। यहां तक कि बीजगणित भी
अध्याय उनका उपभोग करते हैं: एक मैट्रिक्स की पुनरावृत्ति
\cref{ch:b1:matrices} अनुक्रम बनाओ जिसका व्यवहार
(का अभिसरण $ A^n $) इसके साथ एक रैखिक-बीजगणित प्रश्न है
अध्याय की शब्दावली. हर जगह ले जाने योग्य दो प्रमेय:
मोनोटोन सीमा (आदेश से अस्तित्व) और बोलजानो--वीयरस्ट्रैस
(सीमा से अस्तित्व) --- उनके बीच, लगभग हर
इस किताब में सीमा का जन्म हुआ है.
\end{remark}

\begin{remark}[जटिल अनुक्रम]
एक क्रम $(z_n)$ सम्मिश्र संख्याओं का अभिसरण होता है $\ell $ कब
$\abs{z_n - \ell} \to 0$; समान रूप से, जब $\Re(z_n) \to \Re(\ell)$
और $\Im(z_n) \to \Im(\ell)$ (तुलना करना $\abs{z}$ साथ $\abs{\Re z} +
\abs{\Im z}$). जिन प्रमेयों में क्रम--संचालन शामिल नहीं है,
बोलजानो--वीयरस्ट्रैस (दो बार निकालें), कॉची मानदंड--- आगे ले जाएं
शब्दशः.
\end{remark}

\section{समस्या: सेस\`एआरओ, स्टोल्ज़, और साइन की धीमी गिरावट}

\begin{problem}[{सप्ताहांत समस्या --- सेस\`एआरओ--स्टोल्ज़ प्रमेय और स्पर्शोन्मुखता $ u_n \sim \sqrt{3/n}$ के लिए $ u_{n+1} = \sin u_n $}]
\label{pb:b1:seq:1}
सेस\`एआरओ--स्टोल्ज़ प्रमेय असतत एल'हॉस्पिटल नियम है: को
भागफल की सीमा ज्ञात कीजिए $ a_n/b_n $, यह खोजने के लिए पर्याप्त है
के भागफल की सीमा \emph{वेतन वृद्धि} $(a_{n+1} -
a_n)/(b_{n+1} - b_n)$. यह समस्या प्रमेय को सिद्ध करती है, फ़सल
इसके साथ शास्त्रीय सीमाएँ, और फिर इसे एक प्रसिद्ध लक्ष्य पर लक्षित करें:
क्रम $ u_{n+1} = \sin u_n $, जो रेंगता है $0$ पर
बिल्कुल गणना योग्य गति $ u_n \sim \sqrt{3/n}$.\index{Cesaro--Stolz
theorem@Cesàro--Stolz theorem} हाई स्कूल से दो तथ्य
वॉल्यूम यहां दिए गए हैं और बाद में इसे ईमानदारी से फिर से साबित किया गया है
आयतन: स्पर्शरेखा-रेखा असमानता
\[
\tag{G1} \eu^{u} \geq 1 + u \quad (u \in \R),
\]
में उत्तलता द्वारा पुनः सिद्ध किया गया \cref{ch:b1:derivative}, और साइन
ब्रैकेटिंग
\[
\tag{G2} x - \frac{x^3}{6} \;\leq\; \sin x \;\leq\; x -
\frac{x^3}{6} + \frac{x^5}{120} \quad (0 \leq x \leq 1), \qquad
\abs{\sin x} \leq \abs{x} \quad (x \in \R),
\]
में टेलर के सूत्र द्वारा पुनः सिद्ध किया गया \cref{ch:b1:taylor}.

\medskip
\noindent\textbf{Part I --- Sums without closed formulas.}
\begin{enumerate}
  \item का उपयोग करते हुए $1 + 2 + \dots + n = \frac{n(n+1)}{2}$ और $1^2 +
        \dots + n^2 = \frac{n(n+1)(2n+1)}{6}$, गणना करें
        $\lim \frac{1 + 2 + \dots + n}{n^2}$ और $\lim \frac{1^2 +
        \dots + n^2}{n^3}$.
  \item होने देना $ T_n = \sum_{k=1}^n \sqrt k $, जिसके लिए कोई बंद नहीं है
        सूत्र मौजूद है. ब्रैकेटिंग सिद्ध करें
        \[
        \frac{1}{2\sqrt 2}\,n^{3/2} \;\leq\; T_n \;\leq\;
        n^{3/2}
        \]
        \emph{(केवल शर्तें रखें $ k > \frac n2$ निचले के लिए
        बाध्य)}. इसलिए $ T_n $ आदेश है $ n^{3/2}$ --- लेकिन कौन सा
        स्थिर? प्रश्न को प्रश्न 8 तक रोके रखें।
  \item (टेलीस्कोपिंग लेम्मा) चलो $(b_n)$ सख्ती से बढ़ो
        और मान लीजिए कि यह सबके लिए है $ k \geq N $,
        \[
        m \;\leq\; \frac{a_{k+1} - a_k}{b_{k+1} - b_k} \;\leq\;
        M .
        \]
        साबित करें कि $ m \leq \dfrac{a_n - a_N}{b_n - b_N} \leq M $
        हरएक के लिए $ n > N $.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The \omterm{pb:b1:seq:1}{Cesàro--Stolz theorem}.} होने देना
$(b_n)$ के साथ सख्ती से बढ़ोतरी की जाए $ b_n \to +\infty $, और
मान लीजिये $\dfrac{a_{n+1} - a_n}{b_{n+1} - b_n} \to \ell \in \R $.
\begin{enumerate}[resume]
  \item हल करना $\varepsilon > 0$. दिखाओ वहाँ है $ N $ ऐसा है कि
        $\ell - \varepsilon \leq \dfrac{a_n - a_N}{b_n - b_N}
        \leq \ell + \varepsilon $ सभी के लिए $ n > N $.
  \item स्थापित करें, के लिए $ n > N $, पहचान
        \[
        \frac{a_n}{b_n} - \ell
        = \frac{a_N - \ell\,b_N}{b_n}
        + \Bigl(1 - \frac{b_N}{b_n}\Bigr)
        \Bigl(\frac{a_n - a_N}{b_n - b_N} - \ell\Bigr),
        \]
        और प्रमेय समाप्त करें: $\dfrac{a_n}{b_n} \to \ell $.
  \item सिद्ध करो $+\infty $ वैरिएंट: यदि $\dfrac{a_{n+1} -
        a_n}{b_{n+1} - b_n} \to +\infty $ (पर भी वही परिकल्पनाएँ
        $(b_n)$), तब $\dfrac{a_n}{b_n} \to +\infty $.
  \item लेना $ b_n = n $: सेस पुनर्प्राप्त करें\`एआरओ माध्य प्रमेय
        \cref{exo:b1:seq:10}. फिर इसका विपरीतार्थक दिखाएँ
        सेस\`एआरओ--स्टोल्ज़ विफल: के लिए $ a_n = (-1)^n $, $ b_n = n $, द
        भागफल $ a_n/b_n $ के भागफल के दौरान अभिसरण होता है
        वेतन वृद्धि नहीं होती. स्टोल्ज़ एक तरफ़ा सड़क है।
\end{enumerate}

\medskip
\noindent\textbf{Part III --- First dividends.}
\begin{enumerate}[resume]
  \item सिद्ध करना $(1+h)^{3/2} - 1 = \dfrac{3h + 3h^2 +
        h^3}{(1+h)^{3/2} + 1}$ संयुग्मन द्वारा, निष्कर्ष निकालना
        $ n\bigl((1 + \tfrac1n)^{3/2} - 1\bigr) \to \tfrac32$,
        और सेस के साथ समाप्त करें\`एआरओ--स्टोल्ज़:
        \[
        T_n = \sum_{k=1}^{n} \sqrt k \;\sim\; \tfrac23\,
        n^{3/2} ,
        \]
        प्रश्न 2 की उलझन को हल करना।
  \item अकेले (G1) से, लघुगणक ब्रैकेटिंग प्राप्त करें
        \[
        \frac{t}{1 + t} \;\leq\; \ln(1 + t) \;\leq\; t
        \qquad (t > -1)
        \]
        \emph{((जी1) पर लागू करें $ u = \ln(1+t)$ और कम से $ u =
        -t/(1+t)$)}.
  \item बताते हैं कि $ b_n = \ln n $ के साथ सख्ती से बढ़ रहा है
        $\ln n \to +\infty $, और सेस के साथ साबित करें\`एआरओ--स्टोल्ज़ और
        प्रश्न 9 वह
        \[
        H_n = \sum_{k=1}^{n} \frac 1k \;\sim\; \ln n .
        \]
        (बेहतर संरचना $ H_n = \ln n + \gamma + o(1)$ है
        सप्ताहांत की समस्या \cref{ch:b1:series}.)
  \item (अनुपात से मूल तक) चलो $ u_n > 0 $ साथ
        $\frac{u_{n+1}}{u_n} \to L > 0$. प्रश्न 9 का प्रयोग करके दिखाएँ
        $\ln\frac{u_{n+1}}{u_n} \to \ln L $; सेस लागू करें\`एआरओ को
        निष्कर्ष $\frac{\ln u_n}{n} \to \ln L $, फिर, (G1) के साथ,
        वह $ u_n^{1/n} \to L $. अनुप्रयोग: गणना करें
        $\lim\,\binom{2n}{n}^{1/n}$.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The slow fall of the sine.} होने देना
$ u_0 \in \R $ और $ u_{n+1} = \sin u_n $.
\begin{enumerate}[resume]
  \item (जी2) से, दिखाओ $0 < \sin x < x $ के लिए $0 < x \leq 1$.
        निष्कर्ष: $ u_1 \in \intcc{-1}{1}$; अगर $ u_1 = 0$ द
        क्रम रैंक से शून्य है $1$; और यदि $ u_1 > 0$ (मामला $ u_1 < 0$ सममित होना, $\sin $ अजीब), तो
        $(u_n)_{n \geq 1}$ सख्ती से घट रहा है, सकारात्मक, और
        में एकत्रित हो जाता है $0$ \emph{(के माध्यम से सीमा की पहचान करें $\ell =
        \sin \ell $, का उपयोग करना $\abs{\sin a - \sin b} \leq \abs{a -
        b}$, स्वयं (G2) और उत्पाद-से-योग का परिणाम है
        सूत्र)}.
  \item अभी से मान लीजिये $ u_1 \in \intoc{0}{1}$. द्वारा दिखाएँ
        निचोड़ना, (G2) का उपयोग करके:
        \[
        \frac{\sin u_n}{u_n} \to 1
        \qquad\text{and}\qquad
        \frac{u_n - \sin u_n}{u_n^{3}} \to \frac16 .
        \]
  \item गुणनखंडन सिद्ध करें
        \[
        w_n := \frac{1}{u_{n+1}^{2}} - \frac{1}{u_n^{2}}
        = \frac{u_n - \sin u_n}{u_n^{3}} \cdot
        \frac{u_n + \sin u_n}{u_n} \cdot
        \Bigl(\frac{u_n}{\sin u_n}\Bigr)^{2},
        \]
        और निष्कर्ष निकालें $ w_n \to \frac13$.
  \item के साथ समापन करें \cref{exo:b1:seq:10} (वृद्धि संस्करण)
        वह $\frac{1}{n\,u_n^{2}} \to \frac13$, फिर, ए द्वारा
        वर्गमूल के लिए संयुग्मन तर्क, शीर्षक:
        \[
        \sqrt n\;u_n \longrightarrow \sqrt 3 ,
        \qquad\text{i.e.}\qquad
        u_n \sim \sqrt{\frac 3n} .
        \]
  \item धीमेपन को मापें: अंततः उसे दिखाएं $\sqrt{2/n}
        \leq u_n \leq 2/\sqrt n $, ताकि पहुंच सके $ u_n \leq
        10^{-2}$ से अधिक की आवश्यकता है $20\,000$ पुनरावृत्तियों (के बारे में)
        $30\,000$, एसिम्प्टोटिक्स द्वारा)। के साथ तुलना करें बगुला का
        विधि (\cref{ex:b1:seq:heron}) और समझाएं
        संरचनात्मक कारण: निश्चित बिंदु पर $0$, का ढलान
        $\sin $ है $1$ (ए \emph{तटस्थ} निश्चित बिंदु), जबकि
        त्रुटि-आधे पुनरावृत्तियों के लिए ढलान की आवश्यकता होती है \omterm{def:b1:complex:field}{मापांक} $< 1$.
  \item उसके लिए दिखाओ \emph{प्रत्येक} प्रस्थान बिंदू $ u_0 \in \R $,
        दोनों में से एक $ u_n = 0$ रैंक से $1$ पर, या $\abs{u_n} \sim
        \sqrt{3/n}$ --- पतन सर्वव्यापी है, केवल संकेत है
        याद है $ u_0$.
\end{enumerate}

\medskip
\noindent\textbf{Part V --- The general principle.} साइन है
एक मशीन का एक उदाहरण.
\begin{enumerate}[resume]
  \item होने देना $ u_n > 0$, $ u_n \to 0$, और $\dfrac{u_n -
        u_{n+1}}{u_n^{2}} \to a > 0$. क्रमिक रूप से सिद्ध करें:
        $\frac{u_{n+1}}{u_n} \to 1$; तब $\frac{1}{u_{n+1}} -
        \frac{1}{u_n} \to a $; तब $ n\,u_n \to \frac1a $.
  \item (सटीक मॉडल) के लिए $ u_{n+1} = \dfrac{u_n}{1 + u_n}$, $ u_0
        > 0$: दिखाओ $\frac{1}{u_n}$ अंकगणित है, बिल्कुल हल करें,
        और प्रश्न 18 के निष्कर्ष को सटीक से जांचें
        सूत्र.
  \item के लिए $ u_{n+1} = u_n \eu^{-u_n}$, $ u_0 > 0$: दिखाओ $ u_n \to
        0$, निचोड़ने के लिए (G1) का उपयोग करें $\frac{1 - \eu^{-t}}{t}$ बीच में
        $\frac{1}{1+t}$ और $1$ के लिए $ t > 0$, और निष्कर्ष निकालें $ u_n
        \sim \frac 1n $.
  \item (घन संपर्क, वर्गाकार दूरबीन) चलो $ u_n > 0$, $ u_n
        \to 0$, $\dfrac{u_n - u_{n+1}}{u_n^{3}} \to a > 0$.
        दिखाने के लिए प्रश्न 14 का गुणनखंड अपनाएँ
        $\frac{1}{u_{n+1}^2} - \frac{1}{u_n^2} \to 2a $, और
        निष्कर्ष $ n\,u_n^{2} \to \frac{1}{2a}$. जाँच करें कि $ a =
        \frac16$ भाग IV पुनर्प्राप्त करता है।
\end{enumerate}

\medskip
\noindent\textbf{Part VI --- Limits of the method, and morals.}
\begin{enumerate}[resume]
  \item वह परिकल्पना दिखाओ $ b_n \to +\infty $ नहीं हो सकता
        गिराया: के लिए $ a_n = 2 - 2^{-n}$ और $ b_n = 1 - 2^{-n}$,
        वृद्धि भागफल की ओर प्रवृत्त होता है $1$ जबकि $\frac{a_n}
        {b_n} \to 2$. प्रश्न 5 की सटीक पंक्ति को इंगित करें
        सबूत जो टूट जाता है.
  \item (स्टोल्ज़ दो बार) सिद्ध करें $\sum_{k=1}^{n} H_k \sim n \ln n $
        \emph{(सेस का एक आवेदन\`एआरओ--स्टोल्ज़, फिर प्रश्न
        10; अवश्यंभावी $(n+1)\ln(n+1) - n\ln n $ प्रश्न 9 का उपयोग करते हुए)}.
  \item (ज्यामितीय साधन) यदि $ u_n > 0$ और $ u_n \to \ell > 0$,
        दिखाओ $(u_1 u_2 \cdots u_n)^{1/n} \to \ell $; अगर $ u_n \to
        +\infty $, दिखाओ $(u_1 \cdots u_n)^{1/n} \to +\infty $.
        परिणाम निकालना $(n!)^{1/n} \to +\infty $.
  \item संश्लेषण, प्रत्येक वाक्य: (i) वास्तव में कहाँ किया
        पूर्णता इस समस्या में प्रवेश करती है; (ii) किस अर्थ में है
        सेस\`एआरओ--स्टोलज़ एक असतत एल'अस्पताल नियम (इसका
        डिफरेंशियल ट्विन के माध्य मान प्रमेय पर आधारित है
        \cref{ch:b1:derivative}); (iii) अनुमान बताएं
        के संपर्क आदेश को लिंक करना $ f $ एक तटस्थ निश्चित पर
        के क्षय घातांक को इंगित करें $ u_{n+1} = f(u_n)$;
        (iv) स्थिरांक का पता लगाएं $3$ का $\sqrt{3/n}$ के माध्यम से वापस
        पाइपलाइन $\frac16 \to \frac13 \to 3$.
\end{enumerate}
\end{problem}
