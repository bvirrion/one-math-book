\chapter{तर्क, सेट और मानचित्र}\label{ch:b1:logic}

अब तक, प्रमाण एक अनौपचारिक लेकिन ईमानदार विचार के साथ किए गए हैं
``साबित करना'' का क्या मतलब है। स्नातक का यह पहला अध्याय
गणित खेल के नियमों को स्पष्ट करता है: क्या गणित है
\omterm{def:b1:logic:statement}{कथन} यह है कि संयोजक और परिमाणक कैसे संयोजित होते हैं \omterm{def:b1:logic:statement}{कथन},
कौन सी चालें प्रमाण में वैध हैं --- और फिर इस आधार पर निर्माण करती हैं,
गणित की दो सार्वभौमिक भाषाएँ: सेट और एमएपीएस.

\section{कथन और संयोजक}

\begin{definition}[कथन, संयोजक]\label{def:b1:logic:statement}
ए \emph{कथन}\index{कथन} (या \emph{प्रस्ताव}) एक है
वाक्य जो या तो सत्य है (टी) या गलत है (एफ) --- बिल्कुल इनमें से एक
दो. बयानों से $P $ और $ Q $ एक रूप:
\begin{itemize}
  \item  \emph{नकार} $\lnot P $ (``नहीं $ P $''), सच बिल्कुल कब
        $ P $ मिथ्या है;
  \item  \emph{संयोजक} $ P \land Q $ (``$ P $ और $ Q $''), सत्य
        ठीक तब जब दोनों सत्य हों;
  \item  \emph{अलगाव} $ P \lor Q $ (``$ P $ या $ Q $''), सत्य
        बिल्कुल तब जब कम से कम एक सत्य हो (यह ``या'' सम्मिलित है);
  \item  \emph{असरः} $ P \implies Q $, बिल्कुल गलत जब $ P $
        सत्य है और $ Q $ मिथ्या है;
  \item  \emph{समानक} $ P \iff Q $, सच बिल्कुल कब $ P $ और
        $ Q $ सत्य मूल्य समान है।
\end{itemize}
\end{definition}

\begin{remark}
की सत्य तालिका $ P \implies Q $ विराम का पात्र है: कब $ P $ झूठ है,
$ P \implies Q $ है \emph{सत्य}, जो कुछ भी $ Q $ है। ``अगर $2 < 1$ तब
$0 = 5$'' एक सच्चा निहितार्थ है. एक निहितार्थ के बारे में कुछ भी दावा नहीं करता है
क्या होता है जब इसकी परिकल्पना विफल हो जाती है.
\end{remark}

\begin{proposition}[कथनों पर गणना के नियम]\label{prop:b1:logic:rules}
सभी के लिए \omterm{def:b1:logic:statement}{कथन} $ P $, $ Q $, $ R $:
\begin{enumerate}
  \item $\lnot(\lnot P) \iff P $;
  \item डी मॉर्गन के नियम\index{डी मॉर्गन के नियम}:
        $\lnot(P \land Q) \iff (\lnot P) \lor (\lnot Q)$ और
        $\lnot(P \lor Q) \iff (\lnot P) \land (\lnot Q)$;
  \item $(P \implies Q) \iff \bigl((\lnot P) \lor Q\bigr)$, इस तरह
        $\lnot(P \implies Q) \iff P \land (\lnot Q)$;
  \item कोंटरापज़िशन\index{कोंटरापज़िशन}:
        $(P \implies Q) \iff \bigl((\lnot Q) \implies (\lnot P)\bigr)$;
  \item $(P \iff Q) \iff \bigl((P \implies Q) \land (Q \implies
        P)\bigr)$;
  \item वितरणशीलता: $ P \land (Q \lor R) \iff (P \land Q) \lor
        (P \land R)$ और $ P \lor (Q \land R) \iff (P \lor Q) \land
        (P \lor R)$.
\end{enumerate}
\end{proposition}

\begin{proof}
प्रत्येक तुल्यता की जाँच सत्य तालिकाओं की तुलना करके की जाती है: दो यौगिक
\omterm{def:b1:logic:statement}{कथन} से निर्मित $ P $, $ Q $, $ R $ बिल्कुल समतुल्य हैं जब वे
प्रत्येक (चार या आठ) मामलों में समान सत्य मान लें। आइये
पहले डी मॉर्गन कानून के लिए पूरी एक तालिका प्रदर्शित करें:
\begin{center}
\begin{tabular}{cc|c|c|cc|c}
$ P $ & $ Q $ & $ P \land Q $ & $\lnot(P \land Q)$ & $\lnot P $ & $\lnot Q $ &
$(\lnot P) \lor (\lnot Q)$ \\
\hline
T & T & T & F & F & F & F \\
T & F & F & T & F & T & T \\
F & T & F & T & T & F & T \\
F & F & F & T & T & T & T \\
\end{tabular}
\end{center}
कॉलम $4$ और $7$ संयोग, जो नियम को सिद्ध करता है। विरोधाभास के लिए
मौखिक शॉर्टकट तेज़ है: $ P \implies Q $ बिल्कुल गलत है
मामला ($ P $ सत्य, $ Q $ झूठा), और $(\lnot Q) \implies (\lnot P)$ झूठ है
बिल्कुल मामले में ($\lnot Q $ सत्य, $\lnot P $ झूठा), यानी\ ($ Q $
असत्य, $ P $ सत्य) --- एक ही मामला, इसलिए दो निहितार्थ हैं
समान तालिकाएँ. शेष नियमों की जाँच इसी प्रकार की जाती है; नोट
वह (3) प्रत्येक निहितार्थ को विच्छेद में बदल देता है, ताकि (2)
यंत्रवत् निषेध नियम उत्पन्न करता है $\lnot(P \implies Q) \iff P
\land (\lnot Q)$: किसी निहितार्थ का खंडन करने के लिए किसी मामले को प्रदर्शित करना होगा
जहां परिकल्पना कायम रहती है और निष्कर्ष विफल हो जाता है।
\end{proof}

\section{परिमाणकों}

\begin{definition}[परिमाणकों]\label{def:b1:logic:quantifiers}
होने देना $ P(x)$ किसी तत्व का गुण होना $ x $ एक का \omterm{def:b1:logic:sets}{तय करना} $ E $.
\begin{itemize}
  \item $\forall x \in E,\ P(x)$ (``सभी के लिए $ x $ में $ E $, $ P(x)$'') है
        सत्य है जब प्रत्येक तत्व $ E $ संतुष्ट $ P $;
  \item $\exists x \in E,\ P(x)$ (``वहाँ मौजूद है $ x $ में $ E $ ऐसा है कि
        $ P(x)$'') सत्य है जब कम से कम एक तत्व $ E $ संतुष्ट
        $ P $.
\end{itemize}
एक लिखता है $\exists!$ क्योंकि ``वहाँ एक अद्वितीय अस्तित्व है''।
\end{definition}

\begin{proposition}[परिमाणकों का निषेध]\label{prop:b1:logic:negquant}
\[
\lnot\bigl(\forall x \in E,\ P(x)\bigr) \iff
\exists x \in E,\ \lnot P(x),
\qquad
\lnot\bigl(\exists x \in E,\ P(x)\bigr) \iff
\forall x \in E,\ \lnot P(x).
\]
\end{proposition}

\begin{proof}
आइए हम दोनों दिशाओं में पहली तुल्यता पर बहस करें; दूसरा
सममित है. अगर $\forall x \in E,\ P(x)$ झूठ है तो हर नहीं
तत्व संतुष्ट करता है $ P $: द \omterm{def:b1:logic:sets}{तय करना} $ A = \{x \in E : \lnot P(x)\}$
खाली नहीं हो सकता, और इसका कोई भी तत्व गवाह नहीं है $\exists x \in
E,\ \lnot P(x)$. इसके विपरीत, यदि कुछ $ x_0 \in E $ संतुष्ट $\lnot
P(x_0)$, तब $ x_0$ प्रतिउदाहरण एवं सार्वभौमिक है \omterm{def:b1:logic:statement}{कथन}
विफल रहता है. दूसरे नियम के लिए: ``नहीं $ x $ संतुष्ट $ P $'' का मतलब है \omterm{def:b1:logic:sets}{तय करना}
$\{x : P(x)\}$ खाली है, यानी\ प्रत्येक $ x $ इसके पूरक में निहित है
$ A $. क्वांटिफायर के नेस्टेड उपसर्ग में कैस्केड में लागू, दो
नियम इसकी यांत्रिक प्रक्रिया देते हैं \cref{ex:b1:logic:limit}:
निषेध बाएँ से दाएँ चलता है, प्रत्येक को पलटता हुआ $\forall $ में
$\exists $ और हर $\exists $ में $\forall $, और अंततः नकार देता है
अंतरतम विधेय.
\end{proof}

\begin{example}[रोजमर्रा के गणितीय वाक्यों को नकारना]\label{ex:b1:logic:negpractice}
होने देना $ f \colon \R \to \R $. वाक्य ``$ f $ बढ़ रहा है'' पढ़ता है
\[
\forall x \in \R,\ \forall y \in \R,\quad
x \leq y \implies f(x) \leq f(y) ,
\]
और इसका निषेध, द्वारा \cref{prop:b1:logic:negquant} प्लस नियम
$\lnot(P \implies Q) \iff P \land \lnot Q $:
\[
\exists x \in \R,\ \exists y \in \R,\quad
x \leq y \ \text{ and }\ f(x) > f(y) :
\]
एक साक्षी जोड़ा ही काफी है। इसी तरह ``$ f $ परिबद्ध है'' है
$\exists M \in \R,\ \forall x \in \R,\ \abs{f(x)} \leq M $, के साथ
निषेध
\[
\forall M \in \R,\ \exists x \in \R,\quad \abs{f(x)} > M :
\]
\emph{जो कुछ भी} बाउंड प्रस्तावित है, कुछ बिंदु इसे मात देते हैं। द
अंतर्दृष्टि: एक सही निषेध में कभी भी ``नहीं'' लागू नहीं होता है
परिमाणित ब्लॉक --- यह एक नया सकारात्मक है \omterm{def:b1:logic:statement}{कथन}, जिसमें
भूमिकाओं का आदान-प्रदान होता है: एक अब गवाह पैदा करता है जो पहले था
प्राप्त.
\end{example}

\begin{example}[परिमाणकों का क्रम]\label{ex:b1:logic:order}
 \omterm{def:b1:logic:order}{आदेश} विभिन्न परिमाणकों का महत्व:
\[
\forall x \in \R,\ \exists y \in \R,\ y > x
\quad\text{is true (take } y = x+1\text{),}
\]
\[
\exists y \in \R,\ \forall x \in \R,\ y > x
\quad\text{is false (no real number exceeds all reals).}
\]
पहले में \omterm{def:b1:logic:statement}{कथन} $ y $ पर निर्भर हो सकता है $ x $; दूसरे में, एक सिंगल
$ y $ सभी के लिए काम करना चाहिए $ x $. दूसरी ओर, दो समान परिमाणक
हाथ, हमेशा आवागमन.
\end{example}

\begin{example}[तीन परिमाणकों वाली परिभाषा पढ़ना]\label{ex:b1:logic:limit}
वाक्य ``अनुक्रम $(u_n)$ में एकत्रित हो जाता है $\ell $''होगा
में लिखा है \cref{ch:b1:seq} जैसा
\[
\forall \varepsilon > 0,\ \exists N \in \N,\ \forall n \geq N,\quad
\abs{u_n - \ell} \leq \varepsilon .
\]
इसका निषेध, द्वारा \cref{prop:b1:logic:negquant} तीन बार आवेदन किया है, है
\[
\exists \varepsilon > 0,\ \forall N \in \N,\ \exists n \geq N,\quad
\abs{u_n - \ell} > \varepsilon .
\]
ऐसे वाक्यों को यंत्रवत्, बिना सोचे-समझे नकारने में सक्षम होना
उनका क्या मतलब है, यह एक वास्तविक कौशल है: यह तार्किक कार्य को अलग करता है
गणितीय कार्य से.
\end{example}

\section{प्रमाण तकनीक}

\begin{method}[मानक प्रमाण पैटर्न]\label{met:b1:logic:proofs}
साबित करना\dots
\begin{enumerate}
  \item \emph{एक निहितार्थ $ P \implies Q $ सीधे}: मान लीजिए $ P $,
        निष्कर्ष निकालना $ Q $;
  \item \emph{विरोधाभास द्वारा}: मान लीजिए $\lnot Q $, निष्कर्ष निकालना $\lnot P $
        --- द्वारा मान्य \cref{prop:b1:logic:rules}~(4);
  \item \emph{विरोधाभास से}: मान लीजिए \omterm{def:b1:logic:statement}{कथन} मिथ्या है, व्युत्पन्न करें
        एक विरोधाभास;
  \item \emph{एक तुल्यता}: दोनों निहितार्थों को अलग-अलग सिद्ध करें (या
        श्रृंखला ज्ञात समकक्ष);
  \item \emph{एक ``सभी के लिए'' \omterm{def:b1:logic:statement}{कथन}}: एक चुनें \emph{मनमाना} $ x $
        में $ E $ (``चलो $ x \in E $'') और साबित करें $ P(x)$;
  \item \emph{एक ``वहाँ मौजूद है'' \omterm{def:b1:logic:statement}{कथन}}: एक गवाह प्रदर्शित करें, या
        अप्रत्यक्ष रूप से अस्तित्व सिद्ध करें;
  \item \emph{प्रेरण द्वारा}: देखना \cref{thm:b1:logic:induction}.
\end{enumerate}
A सिद्ध करते समय \omterm{def:b1:logic:statement}{कथन} एक अच्छी तरह से चुने गए लेकिन मनमाने तत्व के बारे में,
तत्व को कभी भी अतिरिक्त गुण न दें: ``चलो $ x \in \R $'' इसके बाद
``से $ x > 0$\dots''नकारात्मक के बारे में कुछ भी साबित नहीं होता $ x $.
\end{method}

\begin{remark}[प्रमाणों में सामान्य गड़बड़ियाँ]\label{rem:b1:logic:pitfalls}
चार शास्त्रीय जाल, सभी का एक बार नाम लेना उचित है।
\begin{enumerate}
  \item \emph{गर्भनिरोधक के बजाय बातचीत करें।} $ Q \implies P $
        है \emph{नहीं} के बराबर $ P \implies Q $; केवल $\lnot Q
        \implies \lnot P $ है। ``अगर बारिश होती है, तो सड़क गीली होती है''
        किसी को गीली सड़क से बारिश का निष्कर्ष निकालने का अधिकार नहीं है।
  \item \emph{एक निहितार्थ द्वारा तुल्यता सिद्ध करना।} एक
        ``iff'' दावा दो प्रमेय है; घोषणा करें कि कौन सी दिशा है
        सिद्ध किया जा रहा है, और दोनों सिद्ध करें। की जंजीरें $\iff $ कानूनी हैं
        केवल यदि \emph{प्रत्येक} लिंक वास्तव में प्रतिवर्ती है ---
        उदाहरण के लिए, किसी समीकरण का वर्ग करना नहीं है।
  \item \emph{पीछे के प्रमाण.} वांछित से शुरू
        निष्कर्ष निकालना और सत्य निकालना \omterm{def:b1:logic:statement}{कथन} कुछ भी साबित नहीं होता
        (से $-1 = 1$ कोई सत्य को प्राप्त कर लेता है $1 = 1$ वर्ग करके)।
        एक गणना हो सकती है \emph{की खोज की} पीछे की ओर, लेकिन यह
        होना ही चाहिए \emph{लिखा हुआ} आगे, या स्पष्ट के साथ
        समकक्षताएँ
  \item \emph{स्थिर साक्षी बनाम मनमाना तत्व।} साबित करना
        $\exists x,\ P(x)$, कोई व्यक्ति किसी एक को चतुराई से प्रदर्शित कर सकता है
        चुना गया $ x $; साबित करना $\forall x,\ P(x)$, चुनिंदा $ x $
        मनमाना रहना चाहिए. दोनों को मिलाना--- जाँचना
        एक उदाहरण पर सार्वभौमिक दावा --- सबसे आम है
        शुरुआती प्रतियों में त्रुटि.
\end{enumerate}
\end{remark}

\begin{example}[कार्यस्थल पर अंतर्विरोध और विरोधाभास]\label{ex:b1:logic:sqrt2}
\emph{के लिए $ n \in \N $: अगर $ n^2$ तब भी है $ n $ सम है.} द्वारा
विरोधाभास: यदि $ n $ अजीब है, $ n = 2k+1$, तब
$ n^2 = 4k^2 + 4k + 1$ अजीब है.

\emph{$\sqrt 2$ तर्कहीन है.} विरोधाभास से: मान लीजिए
$\sqrt 2 = p/q $ साथ $ p, q \in \N^*$ और भिन्न न्यूनतम पदों में.
तब $ p^2 = 2q^2$ सम है, इसलिए $ p $ सम है (पिछला बिंदु), $ p = 2r $;
तब $ q^2 = 2r^2$ सम है, इसलिए $ q $ सम है---निम्नतम का खंडन
शर्तें.
\end{example}

\begin{theorem}[प्रेरण]\label{thm:b1:logic:induction}
होने देना $ P(n)$ पूर्णांक का गुण हो $ n $. अगर
\begin{enumerate}
  \item $ P(0)$ सच है, और
  \item सभी के लिए $ n \in \N $, $ P(n) \implies P(n+1)$,
\end{enumerate}
तब $ P(n)$ सभी के लिए सच है $ n \in \N $.

\emph{मजबूत प्रेरण:} यदि (2) को प्रतिस्थापित कर दिया जाए तो निष्कर्ष अपरिवर्तित रहेगा:
सबके लिए $ n $, $\bigl(P(0) \land \dots \land P(n)\bigr) \implies
P(n+1)$.
\end{theorem}

\begin{proof}
यह की एक संपत्ति है $\N $ स्वयं, इसके बराबर: \emph{हर गैररिक्त
का उपसमुच्चय $\N $ सबसे कम तत्व है} (जिसे हम ज्ञात मानते हैं)। वास्तव में,
मान लीजिए (1) और (2) पकड़ें और छोड़ें $ A = \{n \in \N : P(n)
\text{ false}\}$. अगर $ A \neq \emptyset $, इसमें सबसे कम तत्व है $ m $;
$ m \neq 0$ द्वारा (1); तब $ m - 1 \notin A $, इसलिए $ P(m-1)$ धारण करता है, और (2)
देता है $ P(m)$ --- विरोधाभास। इसलिए $ A = \emptyset $. मजबूत के लिए
प्रेरण, वही तर्क लागू करें: $ P(0), \dots, P(m-1)$ सब पकड़ो
तब से $ m $ सबसे कम में है $ A $.
\end{proof}

\begin{example}[अद्वितीय अस्तित्व को सिद्ध करना |]\label{ex:b1:logic:uniqueexistence}
ए \omterm{def:b1:logic:statement}{कथन} $\exists!\,x,\ P(x)$ है \emph{दो} \omterm{def:b1:logic:statement}{कथन}, सिद्ध हुआ
अलग से: अस्तित्व (कुछ प्रदर्शित करें या निर्माण करें $ x_0$ साथ
$ P(x_0)$) और विशिष्टता (मान लीजिए $ P(x)$ और $ P(x')$, निष्कर्ष निकालना $ x =
x'$). नमूना: \emph{एक अनोखा यथार्थ है $ x $ साथ $ x^3 + x =
2$.} अस्तित्व: $ x_0 = 1$ तब से काम करता है $1 + 1 = 2$. विशिष्टता:
यदि $ x^3 + x = x'^3 + x'$, तब
\[
0 = (x^3 - x'^3) + (x - x')
= (x - x')\,\bigl(x^2 + xx' + x'^2 + 1\bigr),
\]
और दूसरा कारक सकारात्मक है (यह बराबर है)। $\bigl(x +
\tfrac{x'}2\bigr)^2 + \tfrac34 x'^2 + 1 \geq 1$), इसलिए $ x = x'$.
श्रम के विभाजन पर ध्यान दें: अस्तित्व ने एक भाग्यशाली अनुमान का इस्तेमाल किया,
विशिष्टता के लिए मान्य बीजगणित का उपयोग किया जाता है \emph{मनमाना} समाधान ---
कोई भी तर्क दूसरे का काम नहीं करता, और दूसरे को भूल जाना
एक बार समाधान मिल जाने पर आधा स्थायी प्रलोभन है।
\end{example}

\begin{example}\label{ex:b1:logic:induction}
सभी के लिए $ n \in \N^*$: $\;\sum_{k=1}^n k = \frac{n(n+1)}{2}$. बेस केस
$ n = 1$: दोनों पक्ष बराबर $1$. चरण: सूत्र मानकर $ n $,
\[
\sum_{k=1}^{n+1} k = \frac{n(n+1)}{2} + (n+1)
= (n+1)\Bigl(\frac n2 + 1\Bigr) = \frac{(n+1)(n+2)}{2}. \qedhere
\]
\end{example}

\begin{example}[काम पर मजबूत प्रेरण]\label{ex:b1:logic:strongind}
\emph{प्रत्येक पूर्णांक $ n \geq 2$ अभाज्य संख्याओं का गुणनफल है} (एक प्रधान
एक पूर्णांक होना $\geq 2$ जिनके केवल भाजक हैं $\geq 1$ हैं $1$ और
स्वयं; अभाज्य संख्याओं का अध्ययन उनके स्वयं के लिए किया जाता है \cref{ch:b1:arith}).
साधारण प्रेरण यहाँ असहाय है: यह जानना $95 = 5 \times 19$
कारक इसके बारे में कुछ नहीं कहते $96$. मजबूत प्रेरण बिल्कुल फिट बैठता है. बेस केस: $2$ अभाज्य है, इसलिए अभाज्यों का (एक-कारक) उत्पाद है। चरण: चलो
$ n \geq 2$ और प्रत्येक पूर्णांक मान लें $ m $ साथ $2 \leq m \leq n $ एक है
अभाज्यों का उत्पाद. अगर $ n + 1$ प्रधान है, हो गया. अन्यथा $ n + 1 = ab $
साथ $2 \leq a, b \leq n $; दोनों की मजबूत परिकल्पना द्वारा $ a $ और $ b $
अभाज्यों के उत्पाद हैं, इसलिए ऐसा है $ n + 1$. अंतर्दृष्टि: मजबूत
जब भी ``कारण'' हो, प्रेरण सही उपकरण है $ P(n+1)$ रहता है
किसी अप्रत्याशित पूर्व रैंक पर, रैंक पर नहीं $ n $.
\end{example}

\section{सेट}

\begin{definition}[संचालन सेट करें]\label{def:b1:logic:sets}
हम की धारणा लेते हैं \emph{तय करना}\index{तय करना} और सदस्यता संबंध
$ x \in E $ आदिम के रूप में. सेट के लिए $ A, B $ एक परिवेश सेट के अंदर $ E $:
\begin{itemize}
  \item \emph{समावेश}: $ A \subseteq B $ कब $\forall x,\ x \in A
        \implies x \in B $; समानता $ A = B $ कब $ A \subseteq B $ और
        $ B \subseteq A $;
  \item \emph{मिलन} $ A \cup B $, \emph{चौराहा} $ A \cap B $,
        \emph{अंतर} $ A \setminus B = \{x \in A : x \notin B\}$,
        \emph{पूरक} $\overline{A} = E \setminus A $;
  \item  \emph{खाली सेट} $\emptyset $, हर सेट में निहित;
  \item  \emph{सत्ता स्थापित}\index{सत्ता स्थापित} $\mathcal{P}(E)$: सेट
        के सभी उपसमूहों में से $ E $;
  \item  \emph{उत्पाद} $ E \times F $: क्रमित युग्मों का समुच्चय
        $(x, y)$ साथ $ x \in E $, $ y \in F $.
\end{itemize}
\end{definition}

\begin{example}[पावर सेट का आदी होना]\label{ex:b1:logic:powerset}
के लिए $ E = \{a, b\}$:
\[
\mathcal P(E) = \bigl\{\, \emptyset,\ \{a\},\ \{b\},\ \{a, b\}
\,\bigr\},
\]
चार तत्व --- और अनुशासन के प्रकार पर ध्यान दें: $ a \in E $ लेकिन
$\{a\} \in \mathcal P(E)$; \omterm{def:b1:logic:statement}{कथन} $ a \in \mathcal P(E)$
और $\{a\} \subseteq \mathcal P(E)$ जैसा कि लिखा गया है दोनों झूठे हैं
(दूसरे की आवश्यकता होगी $ a $ बनने के लिए \emph{उपसमुच्चय} का $ E $).
शून्य से पुनरावृत्ति: $\mathcal P(\emptyset) = \{\emptyset\}$
एक तत्व है, $\mathcal P(\mathcal P(\emptyset)) = \{\emptyset,
\{\emptyset\}\}$ दो हैं, अगले में चार हैं --- सेट का सेट हैं
साधारण सेट, और \cref{ch:b1:counting} दोहरीकरण की पुष्टि करेगा
पैटर्न: $\abs{\mathcal P(E)} = 2^{\abs E}$. स्तरों को बनाए रखना
($ x $, $\{x\}$, $\{\{x\}\}$) सीधी आधी लड़ाई है
जैसे व्यायाम \cref{exo:b1:logic:11,exo:b1:logic:12}.
\end{example}

\begin{proposition}[बीजगणित सेट करें]\label{prop:b1:logic:setalgebra}
उपसमुच्चय के लिए $ A, B, C $ का $ E $:
\begin{enumerate}
  \item $ A \cap (B \cup C) = (A \cap B) \cup (A \cap C)$ और
        $ A \cup (B \cap C) = (A \cup B) \cap (A \cup C)$;
  \item डी मॉर्गन: $\overline{A \cup B} = \overline{A} \cap
        \overline{B}$ और $\overline{A \cap B} = \overline{A} \cup
        \overline{B}$;
  \item $ A \subseteq B \iff \overline{B} \subseteq \overline{A}$.
\end{enumerate}
\end{proposition}

\begin{proof}
प्रत्येक पहचान एक नियम का अनुवाद करती है \cref{prop:b1:logic:rules} के माध्यम से
शब्दकोश ($\in A $ या नहीं) $\leftrightarrow $ (\omterm{def:b1:logic:statement}{कथन} सच है या
असत्य): उदा.\ $ x \in \overline{A \cup B} \iff \lnot(x \in A \lor x \in
B) \iff (x \notin A) \land (x \notin B) \iff x \in \overline{A} \cap
\overline{B}$. बिंदु (3) विरोधाभास है। दूसरे नमूने के रूप में,
पूर्ण रूप से पहला वितरण कानून:
\[
x \in A \cap (B \cup C)
\iff (x \in A) \land \bigl(x \in B \lor x \in C\bigr)
\iff \bigl(x \in A \land x \in B\bigr) \lor
\bigl(x \in A \land x \in C\bigr),
\]
के वितरण द्वारा \cref{prop:b1:logic:rules}~(6), और
अंतिम \omterm{def:b1:logic:statement}{कथन} पढ़ता $ x \in (A \cap B) \cup (A \cap C)$. प्रत्येक \omterm{def:b1:logic:sets}{तय करना}
इस प्रकार की पहचान इसी एक यांत्रिकी द्वारा सिद्ध होती है
अनुवाद --- यही कारण है कि उनमें से किसी को भी याद रखने की आवश्यकता नहीं है।
\end{proof}

\begin{method}[समानताएं स्थापित करना]\label{met:b1:logic:doubleinc}
साबित करना $ A = B $, दो समावेशन सिद्ध करें: चलो $ x \in A $, दिखाओ $ x \in
B $; तो करने दें $ x \in B $, दिखाओ $ x \in A $. वैकल्पिक रूप से, श्रृंखला
समकक्षताएँ $ x \in A \iff \dots \iff x \in B $ जब प्रत्येक चरण वास्तव में होता है
एक तुल्यता.
\end{method}

\section{एमएपीएस}

\begin{definition}[मानचित्र, छवि, पूर्व छवि]\label{def:b1:logic:map}
ए \emph{नक्शा}\index{नक्शा} (या \emph{समारोह}) $ f \colon E \to F $
प्रत्येक तत्व को असाइन करता है $ x $ की \omterm{def:b1:logic:sets}{तय करना} $ E $ (द \emph{कार्यक्षेत्र}) बिलकुल
एक तत्व $ f(x)$ की \omterm{def:b1:logic:sets}{तय करना} $ F $ (द \emph{कोडोमेन}). के लिए $ A
\subseteq E $ और $ B \subseteq F $:
\[
f(A) = \{f(x) : x \in A\} \subseteq F,
\qquad
f^{-1}(B) = \{x \in E : f(x) \in B\} \subseteq E
\]
हैं \emph{प्रत्यक्ष छवि} का $ A $ और यह \emph{पूर्वछवि}\index{पूर्वछवि} का $ B $. \emph{संघटन} का $ f \colon E \to F $
और $ g \colon F \to G $ है $ g \circ f \colon E \to G $, $ x \mapsto
g(f(x))$.
\end{definition}

\begin{remark}
संकेतन $ f^{-1}(B)$ करता है \emph{नहीं} एक उलटा अनुमान लगाएं \omterm{def:b1:logic:map}{नक्शा}:
$ f^{-1}(B)$ प्रत्येक के लिए परिभाषित किया गया है $ f $. पूर्वछवियाँ से बेहतर व्यवहार करें
छवियाँ: $ f^{-1}$ संघों, अंतर्संबंधों और पूरकों को संरक्षित करता है,
जबकि $ f(A \cap A') \subseteq f(A) \cap f(A')$ सख्त हो सकते हैं
(\cref{exo:b1:logic:8}).
\end{remark}

\begin{example}[छवियों और पूर्वछवियों की गणना करना]\label{ex:b1:logic:images}
होने देना $ f \colon \R \to \R $, $ x \mapsto x^2$. तब:
\[
f\bigl(\intcc{-1}{2}\bigr) = \intcc04, \qquad
f^{-1}\bigl(\intcc14\bigr) = \intcc{-2}{-1} \cup \intcc12, \qquad
f^{-1}(\{-1\}) = \emptyset .
\]
पहले के लिए: हर $ x \in \intcc{-1}2$ है $ x^2 \in \intcc04$, और
प्रत्येक $ y \in \intcc04$ के रूप में प्राप्त किया जाता है $ y = (\sqrt y)^2$ साथ $\sqrt y
\in \intcc02 \subseteq \intcc{-1}2$ ---ध्यान दें कि छवि है
\emph{नहीं} $\intcc14 = \{(-1)^2, 2^2\}$: अंतराल की छवियां नहीं हैं
अकेले समापन बिंदुओं से गणना की गई। दूसरे के लिए: $1 \leq x^2 \leq 4 \iff
1 \leq \abs x \leq 2$, जो दो टुकड़ों में बंट जाता है। तीसरा
दर्शाता है कि ए \omterm{def:b1:logic:map}{पूर्वछवि} खाली हो सकता है --- $ f^{-1}(B)$ हमेशा बनाता है
भाव, चाहे कितना ही छोटा चौराहा क्यों न हो $ B $ छवि के साथ. आख़िरकार
इस उदाहरण पर उपरोक्त टिप्पणी की सख्ती की घटना का निरीक्षण करें:
के साथ $ A = \intcc{-1}0$ और $ A' = \intcc01$, एक के पास है $ f(A \cap A') =
f(\{0\}) = \{0\}$, जबकि $ f(A) \cap f(A') = \intcc01$.
\end{example}

\begin{definition}[विशेषण, विशेषण, विशेषण]\label{def:b1:logic:inj}
ए \omterm{def:b1:logic:map}{नक्शा} $ f \colon E \to F $ है:
\begin{itemize}
  \item \emph{इंजेक्शन}\index{इंजेक्शन} जब अलग-अलग तत्व हों
        विशिष्ट छवियाँ: $\forall x, x' \in E,\ f(x) = f(x') \implies
        x = x'$;
  \item \emph{विशेषण}\index{विशेषण} जब प्रत्येक तत्व $ F $
        प्राप्त होता है: $\forall y \in F,\ \exists x \in E,\ f(x) = y $;
  \item \emph{द्विभाजित}\index{द्विभाजित} जब यह दोनों है, यानी\ प्रत्येक
        $ y \in F $ बिल्कुल एक है \omterm{def:b1:logic:map}{पूर्वछवि}.
\end{itemize}
\end{definition}

\begin{theorem}[उलटा नक्शा]\label{thm:b1:logic:inverse}
ए \omterm{def:b1:logic:map}{नक्शा} $ f \colon E \to F $ है \omterm{def:b1:logic:inj}{द्विभाजित} यदि और केवल यदि कोई है \omterm{def:b1:logic:map}{नक्शा}
$ g \colon F \to E $ साथ $ g \circ f = \mathrm{id}_E $ और $ f \circ g =
\mathrm{id}_F $. उस मामले में $ g $ निराला है; यह लिखा है $ f^{-1}$ और
को बुलाया गया \emph{श्लोक में} का $ f $, और $ f^{-1}$ स्वयं है \omterm{def:b1:logic:inj}{द्विभाजित} साथ
$(f^{-1})^{-1} = f $.
\end{theorem}

\begin{proof}
($\Rightarrow $) अगर $ f $ है \omterm{def:b1:logic:inj}{द्विभाजित}, प्रत्येक $ y \in F $ एक अनोखा है
\omterm{def:b1:logic:map}{पूर्वछवि}; परिभाषित करना $ g(y)$ वैसा होना \omterm{def:b1:logic:map}{पूर्वछवि}. तब $ f(g(y)) = y $ द्वारा
निर्माण, और $ g(f(x)) = x $ क्योंकि $ x $ है \emph{} \omterm{def:b1:logic:map}{पूर्वछवि} का
$ f(x)$.

($\Leftarrow $) ऐसा मान लीजिए $ g $ मौजूद है. अगर $ f(x) = f(x')$, आवेदन करना
$ g $ देता है $ x = x'$: $ f $ है \omterm{def:b1:logic:inj}{इंजेक्शन}. के लिए $ y \in F $, $ x = g(y)$
संतुष्ट $ f(x) = y $: $ f $ है \omterm{def:b1:logic:inj}{विशेषण}.

विशिष्टता: यदि $ g $ और $ h $ तो दोनों काम करते हैं $ g = g \circ \mathrm{id}_F
= g \circ (f \circ h) = (g \circ f) \circ h = h $. अंत में की जोड़ी
पहचान सममित है $ f $ और $ g $, इसलिए $ g = f^{-1}$ है \omterm{def:b1:logic:inj}{द्विभाजित}
व्युत्क्रम के साथ $ f $.
\end{proof}

\begin{example}[व्यवहार में व्युत्क्रम की गणना करना]\label{ex:b1:logic:inversecomp}
होने देना $ f \colon \R \to \intoo0{+\infty}$, $ f(x) = \eu^{2x+1}$. को
उलटना, हल करना $ y = f(x)$ किसी प्रदत्त के लिए $ y > 0$:
\[
y = \eu^{2x+1} \iff \ln y = 2x + 1 \iff x = \frac{\ln y - 1}2 ,
\]
घोषित डोमेन पर प्रत्येक चरण प्रतिवर्ती है। द
गणना एक ही बार में सब कुछ प्रदान करती है: प्रत्येक के लिए $ y $ में
कोडोमेन का बिल्कुल एक ही समाधान है $ x $, इसलिए $ f $ है \omterm{def:b1:logic:inj}{द्विभाजित},
और
\[
f^{-1} \colon \intoo0{+\infty} \to \R,
\qquad
f^{-1}(y) = \frac{\ln y - 1}2 .
\]
दोनों रचनाओं की त्वरित जाँच ($ f^{-1}(f(x)) = \frac{(2x+1) -
1}2 = x $ और $ f(f^{-1}(y)) = \eu^{\ln y} = y $) पुष्टि करता है
\cref{thm:b1:logic:inverse}की कसौटी. अंतर्दृष्टि: ``के लिए हल करें
$ x $ और समतुल्यता को देखना'' एक साथ अस्तित्व है
प्रमाण, विशिष्टता प्रमाण, और सूत्र --- लेकिन यह केवल काम करता है
यदि कोडोमेन सही ढंग से घोषित किया गया था ($ f $ है \emph{नहीं}
\omterm{def:b1:logic:inj}{विशेषण} पर $\R $).
\end{example}

\begin{proposition}[रचना और तीन गुण]\label{prop:b1:logic:comp}
होने देना $ f \colon E \to F $ और $ g \colon F \to G $.
\begin{enumerate}
  \item अगर $ f $ और $ g $ हैं \omterm{def:b1:logic:inj}{इंजेक्शन} (सम्मान.\ \omterm{def:b1:logic:inj}{विशेषण}, \omterm{def:b1:logic:inj}{द्विभाजित}), तो
        है $ g \circ f $; और तब $(g \circ f)^{-1} = f^{-1} \circ
        g^{-1}$ में \omterm{def:b1:logic:inj}{द्विभाजित} मामला।
  \item अगर $ g \circ f $ है \omterm{def:b1:logic:inj}{इंजेक्शन}, तब $ f $ है \omterm{def:b1:logic:inj}{इंजेक्शन}. अगर
        $ g \circ f $ है \omterm{def:b1:logic:inj}{विशेषण}, तब $ g $ है \omterm{def:b1:logic:inj}{विशेषण}.
\end{enumerate}
\end{proposition}

\begin{proof}
(1) यदि $ g(f(x)) = g(f(x'))$, की injectivity $ g $ देता है $ f(x) = f(x')$,
फिर का \omterm{def:b1:logic:inj}{इंजेक्शन} $ f $ देता है $ x = x'$. अगर $ z \in G $, की \omterm{def:b1:logic:inj}{विशेषण}ात्मकता
$ g $ देता है $ y $ साथ $ g(y) = z $, फिर की \omterm{def:b1:logic:inj}{विशेषण}ात्मकता $ f $ देता है $ x $ साथ
$ f(x) = y $, इसलिए $ g(f(x)) = z $. में \omterm{def:b1:logic:inj}{द्विभाजित} केस एक सीधे जाँच करता है
वह $ f^{-1} \circ g^{-1}$ का दोतरफा उलटा है $ g \circ f $, और
में विशिष्टता \cref{thm:b1:logic:inverse} निष्कर्ष.

(2) यदि $ f(x) = f(x')$ तब $ g(f(x)) = g(f(x'))$, और \omterm{def:b1:logic:inj}{इंजेक्शन} की क्षमता
$ g \circ f $ देता है $ x = x'$. अगर $ z \in G $, की \omterm{def:b1:logic:inj}{विशेषण}ात्मकता $ g \circ f $
देता है $ x $ साथ $ g(f(x)) = z $: तब $ y = f(x)$ संतुष्ट $ g(y) = z $.
\end{proof}

\begin{example}[बिंदु (2) तीव्र है]\label{ex:b1:logic:compsharp}
में \cref{prop:b1:logic:comp}~(2), कोई अपग्रेड नहीं कर सकता
निष्कर्ष: $ g \circ f $ \omterm{def:b1:logic:inj}{द्विभाजित} करता है \emph{नहीं} बल $ f $
\omterm{def:b1:logic:inj}{विशेषण} या $ g $ \omterm{def:b1:logic:inj}{इंजेक्शन}. लेना $ E = G = \{1\}$, $ F = \{1,
2\}$, साथ $ f(1) = 1$ और $ g(1) = g(2) = 1$: तब $ g \circ f =
\mathrm{id}_E $ है \omterm{def:b1:logic:inj}{द्विभाजित}, अभी तक $ f $ तत्व चूक जाता है $2$ और
$ g $ दोनों तत्वों को नष्ट कर देता है। नैतिक एक सटीक बहीखाता है
नियम: रचना संबंधी जानकारी प्रवाहित होती है \emph{आंतरिक} \omterm{def:b1:logic:map}{नक्शा} के लिए
\omterm{def:b1:logic:inj}{इंजेक्शन} और करने के लिए \emph{आउटर} \omterm{def:b1:logic:map}{नक्शा} आक्षेपात्मकता के लिए, कभी नहीं
दूसरे तरीके से. (\cref{exo:b1:logic:9} एक ही घटना का निर्माण करता है
अनंत के साथ सेट, जहां यह एक तरफा के पीछे का इंजन है
व्युत्क्रम।)
\end{example}

\begin{example}\label{ex:b1:logic:injexamples}
$ f \colon \R \to \R $, $ x \mapsto x^2$ नहीं है \omterm{def:b1:logic:inj}{इंजेक्शन} ($ f(-1) =
f(1)$) और न \omterm{def:b1:logic:inj}{विशेषण} ($-1$ है कोई \omterm{def:b1:logic:map}{पूर्वछवि}). डोमेन को प्रतिबंधित करना
और कोडोमेन, $ f \colon \R_+ \to \R_+$, $ x \mapsto x^2$ है \omterm{def:b1:logic:inj}{द्विभाजित},
व्युत्क्रम के साथ $ y \mapsto \sqrt y $. ए नक्शाकी injectivity या प्रक्षेपात्मकता
घोषित डोमेन और कोडोमेन पर निर्भर करता है, न कि केवल सूत्र पर।
\end{example}

\section{संबंध}

\begin{definition}[तुल्यता संबंध]\label{def:b1:logic:equiv}
ए \emph{द्विआधारी संबंध} $\mathcal{R}$ एक पर \omterm{def:b1:logic:sets}{तय करना} $ E $ एक
\emph{समतुल्य संबंध}\index{समतुल्य संबंध} जब यह है:
\emph{कर्मकर्त्ता} ($ x \mathbin{\mathcal{R}} x $ सभी के लिए $ x $),
\emph{सममित} ($ x \mathbin{\mathcal{R}} y \implies y
\mathbin{\mathcal{R}} x $) और \emph{सकर्मक} ($ x
\mathbin{\mathcal{R}} y $ और $ y \mathbin{\mathcal{R}} z $ संकेत करना $ x
\mathbin{\mathcal{R}} z $). \emph{तुल्यता वर्ग} का $ x $ है
$\mathrm{cl}(x) = \{y \in E : x \mathbin{\mathcal{R}} y\}$.
\end{definition}

\begin{example}[तीन स्वयंसिद्धों की जाँच करना]\label{ex:b1:logic:fracpart}
पर $\R $, घोषित करें $ x \mathbin{\mathcal{R}} y $ कब $ x - y \in \Z $.
\emph{प्रतिवर्ती:} $ x - x = 0 \in \Z $. \emph{सममित:} अगर $ x - y
\in \Z $ तब $ y - x = -(x - y) \in \Z $. \emph{सकर्मक:} अगर $ x -
y \in \Z $ और $ y - z \in \Z $, तब $ x - z = (x - y) + (y - z) \in
\Z $ (पूर्णांकों का योग)। इसलिए $\mathcal R $ एक समतुल्यता
संबंध, और $\mathrm{cl}(x) = x + \Z = \{x + k : k \in \Z\}$:
प्रत्येक वर्ग में बिल्कुल एक प्रतिनिधि होता है $\intco01$, इसका
\emph{आंशिक हिस्सा}. इसके विपरीत, संबंध ``$\abs{x - y}
\leq 1$'' पर $\R $ प्रतिवर्ती और सममित है लेकिन \emph{नहीं}
सकर्मक ($0 \mathbin{\mathcal R} 1$ और $1 \mathbin{\mathcal R}
2$, अभी तक $\abs{0 - 2} > 1$): निकटता का प्रचार नहीं होता, और नहीं
\omterm{thm:b1:logic:partition}{PARTITION} कक्षाओं में मौजूद है --- रखने के लिए एक उपयोगी प्रति-उदाहरण
स्वयंसिद्धों की पुष्टि करते समय मन में नियमितता महसूस होती है।
\end{example}

\begin{theorem}[कक्षाएँ एक विभाजन बनाती हैं]\label{thm:b1:logic:partition}
होने देना $\mathcal{R}$ एक हो \omterm{def:b1:logic:equiv}{समतुल्य संबंध} पर $ E $. फिर
तुल्यता वर्ग गैर-रिक्त, जोड़ीदार असंयुक्त या समान हैं, और उनके
संघ है $ E $: वे एक बनाते हैं \emph{PARTITION}\index{PARTITION} का $ E $.
इसके विपरीत, प्रत्येक \omterm{thm:b1:logic:partition}{PARTITION} का $ E $ बिल्कुल एक से इस प्रकार उत्पन्न होता है
\omterm{def:b1:logic:equiv}{समतुल्य संबंध} (``एक ही टुकड़े में होना'')।
\end{theorem}

\begin{proof}
$ x \in \mathrm{cl}(x)$ रिफ्लेक्सिविटी द्वारा, इसलिए कक्षाएं गैर-रिक्त हैं
संघ $ E $. कल्पना करना $\mathrm{cl}(x) \cap \mathrm{cl}(y) \neq \emptyset $,
कहना $ z $ दोनों में निहित है. तब $ x \mathbin{\mathcal{R}} z $ और $ y
\mathbin{\mathcal{R}} z $, तो समरूपता और परिवर्तनशीलता द्वारा $ x
\mathbin{\mathcal{R}} y $. अब किसी के लिए $ t \in \mathrm{cl}(y)$,
परिवर्तनशीलता देता है $ t \in \mathrm{cl}(x)$, और सममित रूप से: दो
वर्ग समान हैं. बातचीत के लिए, आइए $(E_i)_{i \in I}$ एक हो
\omterm{thm:b1:logic:partition}{PARTITION} का $ E $ और परिभाषित करें $ x \mathbin{\mathcal S} y $ मतलब
``कुछ टुकड़े में दोनों शामिल हैं $ x $ और $ y $''. \emph{प्रतिवर्ती:} $ x $
किसी टुकड़े में निहित है, जो तब शामिल है $ x $ दो बार से अधिक.
\emph{सममित:} परिभाषित करने की स्थिति सममित है $ x $ और
$ y $. \emph{सकर्मक:} अगर $ x, y \in E_i $ और $ y, z \in E_j $, तब
$ y \in E_i \cap E_j $, इसलिए $ E_i = E_j $ (विशिष्ट टुकड़े हैं
असंयुक्त) और $ x, z $ एक टुकड़ा साझा करें. $\mathcal S $-की श्रेणी $ x $
बिल्कुल वही टुकड़ा है जिसमें शामिल है $ x $, इसलिए कक्षाएं दी गई हैं
टुकड़े. अंततः संबंध इसके वर्गों द्वारा निर्धारित होता है: दो
\omterm{def:b1:logic:equiv}{समतुल्य संबंध} समान वर्गों के साथ समान जोड़े संबंधित हैं,
चूँकि प्रत्येक संबंधित है $ x $ और $ y $ बिल्कुल कब $ y $ का है
की कक्षा $ x $ --- विशिष्टता का दावा कहां से हुआ।
\end{proof}

\begin{example}\label{ex:b1:logic:congruence}
पर $\Z $, सर्वांगसमता मॉड्यूलो $ n $ ($ x \equiv y \pmod n $ कब $ n $ विभाजित
$ x - y $) एक \omterm{def:b1:logic:equiv}{समतुल्य संबंध}; इसकी कक्षाएं हैं $ n $ सेट का
से भाग देने पर दिए गए शेषफल वाले पूर्णांक $ n $. यह उदाहरण
अंगूठी बन जाती है $\Z/n\Z $ में \cref{ch:b1:structures}.
\end{example}

\begin{definition}[आदेश संबंध]\label{def:b1:logic:order}
एक रिश्ता $\preceq $ पर $ E $ एक \emph{आदेश}\index{आदेश}
जब यह प्रतिवर्ती हो, \emph{antisymmetric} ($ x \preceq y $ और $ y
\preceq x $ संकेत करना $ x = y $) और सकर्मक। आदेश है \emph{कुल}
जब कोई दो तत्व तुलनीय हों, \emph{आंशिक} अन्यथा। तत्व $ M \in A \subseteq E $ एक है \emph{सबसे बड़ा तत्व} का $ A $ कब
$ a \preceq M $ सभी के लिए $ a \in A $; सबसे बड़े (और सबसे छोटे) तत्व हैं
जब वे मौजूद होते हैं तो अद्वितीय होते हैं।
\end{definition}

\begin{example}\label{ex:b1:logic:orders}
$(\R, \leq)$ पूरी तरह से ऑर्डर किया गया है. $(\mathcal{P}(E), \subseteq)$ है
जैसे ही आंशिक रूप से ऑर्डर किया गया $ E $ दो तत्व हैं: $\{a\}$ और
$\{b\}$ तुलनीय नहीं हैं. उपसमुच्चय $ A = \{\{a\}, \{b\}\}$ का
$\mathcal{P}(\{a,b\})$ इसका कोई महानतम तत्व नहीं है, फिर भी इसकी ऊपरी सीमा है
$\{a, b\}$: महानतम तत्वों और ऊपरी सीमाओं के बीच अंतर
रिटर्न, के लिए $\R $, में \cref{ch:b1:reals}.
\end{example}

\begin{example}[ग्रिड पर दो ऑर्डर $\N^2$]\label{ex:b1:logic:lexorder}
प्राकृतिक जोड़े पर, घटकवार तुलना करें: $(a, b) \preceq (a',
b')$ कब $ a \leq a'$ \emph{और} $ b \leq b'$ (द \emph{उत्पाद
\omterm{def:b1:logic:order}{आदेश}}). यह एक है \omterm{def:b1:logic:order}{आदेश} --- प्रत्येक अभिगृहीत को समन्वय विरासत में मिला है
समन्वय द्वारा --- लेकिन एक आंशिक: $(1, 3)$ और $(2, 0)$ हैं
अतुलनीय. अब शब्दकोश की तरह तुलना करें: $(a, b)
\preceq_{\mathrm{lex}} (a', b')$ कब $ a < a'$, या $ a = a'$ और $ b
\leq b'$ (द \emph{कोषगत \omterm{def:b1:logic:order}{आदेश}}). परिवर्तनशीलता की आवश्यकता है
दो-मामले की जांच लेकिन कायम है, और कोई भी दो जोड़े अब तुलनीय हैं:
द \omterm{def:b1:logic:order}{आदेश} कुल है. दो \omterm{def:b1:logic:order}{आदेश} समान रैंक \omterm{def:b1:logic:sets}{तय करना} अलग ढंग से
--- $(0, 100) \preceq_{\mathrm{lex}} (1, 0)$ हालांकि उत्पाद
\omterm{def:b1:logic:order}{आदेश} कुछ नहीं कहता --- एक अनुस्मारक कि \omterm{def:b1:logic:order}{आदेश} एक संरचना है
एक \emph{चुनता}, की संपत्ति नहीं है \omterm{def:b1:logic:sets}{तय करना}. शब्दकोष संबंधी
तुलना कई सॉर्टिंग को बदलने के लिए मानक चाल भी है
एक में मानदंड.
\end{example}

\begin{remark}[अंतराल: आक्षेप के रूप में आकार]\label{rem:b1:logic:interlude}
इस अध्याय का एक शांत विषय सुर्खियों में आने लायक है: आक्षेप हैं
गणितज्ञ की ``समान आकार'' की धारणा। परिमित के लिए सेट यह
की गिनती गणना बन जाती है \cref{ch:b1:counting}, कहाँ
प्रत्येक सूत्र गुप्त रूप से एक आक्षेप है; अनंत के लिए सेट यह
नीचे सप्ताहांत समस्या बन जाती है, कहाँ $\N $, $\Q $ और $\R $ बारी
वास्तव में विभिन्न आकारों के लिए। वही शब्दकोष
इस खंड में परिष्कृत रूपों में दो बार पुनः प्रकट होता है: अनुक्रम
(\cref{ch:b1:seq}) के अलावा और कुछ नहीं हैं एमएपीएस $\N \to \R $, इसलिए \omterm{def:b1:logic:statement}{कथन}
अनुक्रमों के बारे में हैं \omterm{def:b1:logic:statement}{कथन} इसके बारे में \omterm{def:b1:logic:sets}{तय करना} का एमएपीएस; और रैखिक
बीजगणित सदिश स्थानों को आक्षेपों द्वारा नहीं बल्कि आक्षेपों द्वारा मापेगा
\emph{रेखीय} आक्षेप, जिनका अस्तित्व एक ही द्वारा नियंत्रित होता है
संख्या, आयाम (\cref{ch:b1:findim}). जब भी कोई नया
``समानता'' प्रकट होती है --- समविभवता, समूहों की समरूपता
(\cref{ch:b1:structures}), रैखिक समरूपता --- का पैटर्न
\cref{thm:b1:logic:inverse} दोहराता है: समानता एक उलटा है,
संरचना का सम्मान करने वाला \omterm{def:b1:logic:map}{नक्शा}.
\end{remark}

\begin{remark}[जहाँ इस अध्याय का प्रयोग किया जाता है]\label{rem:b1:logic:whereused}
हर जगह --- लेकिन कुछ स्थान ध्वजारोहण के लायक हैं। तीन-मात्रक
का जिम्नास्टिक \cref{ex:b1:logic:limit} की दैनिक रोटी है
\cref{ch:b1:seq,ch:b1:continuity}: प्रत्येक सीमा प्रमाण एक खेला जाने वाला खेल है
एक मनमानी के खिलाफ $\varepsilon $. तुल्यता वर्ग पुनः प्रकट होते हैं
की सर्वांगसमता कक्षाएं $\Z/n\Z $ में \cref{ch:b1:structures}, जहां
\omterm{thm:b1:logic:partition}{PARTITION} का \cref{thm:b1:logic:partition} बीजगणित प्राप्त करता है
अपनी खुद की संरचना. संबंधों को क्रमबद्ध करें, ऊपरी सीमा और सबसे कम ऊपरी
सीमाएँ स्वयंसिद्ध हृदय बन जाती हैं $\R $ में \cref{ch:b1:reals}.
\omterm{def:b1:logic:inj}{इंजेक्शन}, अनुमान और आक्षेप रैखिक के रूप में लौटते हैं एमएपीएस का
\cref{ch:b1:linmaps}, जहां एकल पर इंजेक्टिविटी का परीक्षण किया जा सकता है
वेक्टर (कर्नेल); और नीचे सप्ताहांत की समस्या उजागर हो जाती है
के एक सिद्धांत में आपत्ति की धारणा \emph{अनंत के आकार
सेट}, जिसका निष्कर्ष (गणनीयता) $\Q $, की बेशुमारता
$\R $) पुनः सतह पर आना \cref{ch:b1:reals,ch:b1:topology}.
\end{remark}

\section{अभ्यास}

\begin{exercise}[$\star $]\label{exo:b1:logic:1}
प्रत्येक का निषेध लिखिए \omterm{def:b1:logic:statement}{कथन}, ``नहीं'' शब्द का उपयोग किए बिना:
\begin{enumerate}
  \item $\forall x \in \R,\ \exists y \in \R,\ x + y > 0$;
  \item $\exists x \in \R,\ \forall y \in \R,\ xy = 0$;
  \item $\forall \varepsilon > 0,\ \exists \delta > 0,\ \forall x \in
        \R,\ \abs{x} \leq \delta \implies \abs{f(x)} \leq \varepsilon $
        (एक निश्चित के लिए \omterm{def:b1:logic:map}{नक्शा} $ f \colon \R \to \R $).
\end{enumerate}
फिर तय करें कि क्या \omterm{def:b1:logic:statement}{कथन} (1) और (2) सत्य हैं।
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:logic:2}
होने देना $ P, Q $ होना \omterm{def:b1:logic:statement}{कथन}. सत्य तालिकाओं का प्रयोग करते हुए इसे सिद्ध करें
$\lnot(P \implies Q) \iff P \land (\lnot Q)$, और निषेधन निकालें
का: ``यदि कोई फ़ंक्शन अवकलनीय है तो यह सतत है''।
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:logic:3}
विरोधाभास द्वारा सिद्ध करें: के लिए $ x \in \R $, अगर $ x^3 + x \geq 2$ तब
$ x \geq 1$. फिर विरोधाभास से साबित करें: सख्ती से कोई छोटा नहीं है
सकारात्मक वास्तविक संख्या.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:logic:4}
प्रेरण द्वारा सिद्ध करें कि सभी के लिए $ n \in \N $:
\begin{enumerate}
  \item $\sum_{k=0}^{n} 2^k = 2^{n+1} - 1$;
  \item $4^n + 5$ से विभाज्य है $3$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:logic:5}
निम्नलिखित ``प्रमाण'' में दोष खोजें कि सभी पेंसिलों में एक समान गुण होते हैं
रंग. \emph{होने देना $ P(n)$: ``प्रत्येक में \omterm{def:b1:logic:sets}{तय करना} का $ n $ पेंसिलें, सभी पेंसिलें
एक ही रंग है'' $ P(1)$ स्पष्ट है। मान लीजिए $ P(n)$ और ले लो $ n+1$
पेंसिल; आखिरी को हटाकर पहले को $ n $ उनका रंग साझा करें;
पहले को हटाना, आखिरी को $ n $ उनका रंग साझा करें; इसलिए सभी
$ n+1$ उनका रंग साझा करें.}
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:logic:6}
होने देना $ A, B, C $ के उपसमुच्चय हों $ E $. सिद्ध करना:
\begin{enumerate}
  \item $ A \setminus B = A \cap \overline{B}$;
  \item $(A \cup B) \setminus C = (A \setminus C) \cup (B \setminus
        C)$;
  \item $ A \subseteq B \iff A \cup B = B \iff A \cap B = A $.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:logic:7}
प्रत्येक के लिए \omterm{def:b1:logic:map}{नक्शा}, निर्णय करें (प्रमाण के साथ) कि क्या यह है \omterm{def:b1:logic:inj}{इंजेक्शन}, \omterm{def:b1:logic:inj}{विशेषण},
\omterm{def:b1:logic:inj}{द्विभाजित}:
\begin{enumerate}
  \item $ f \colon \N \to \N $, $ n \mapsto n + 1$;
  \item $ g \colon \Z \to \Z $, $ n \mapsto n + 1$;
  \item $ h \colon \R \setminus \{1\} \to \R $, $ x \mapsto
        \frac{x+1}{x-1}$.
\end{enumerate}
के लिए $ h $, इसे बनाने के लिए कोडोमेन को समायोजित करें \omterm{def:b1:logic:inj}{द्विभाजित} और गणना करें
उलटा.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:logic:8}
होने देना $ f \colon E \to F $, होने देना $ A, A' \subseteq E $ और $ B, B' \subseteq
F $.
\begin{enumerate}
  \item सिद्ध करना $ f^{-1}(B \cap B') = f^{-1}(B) \cap f^{-1}(B')$ और
        $ f(A \cup A') = f(A) \cup f(A')$.
  \item सिद्ध करना $ f(A \cap A') \subseteq f(A) \cap f(A')$ और एक दे दो
        उदाहरण जहां समावेशन सख्त है.
  \item सिद्ध करना: $ f $ है \omterm{def:b1:logic:inj}{इंजेक्शन} अगर और केवल अगर $ f(A \cap A') = f(A)
        \cap f(A')$ सभी के लिए $ A, A'$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:logic:9}
होने देना $ f \colon E \to F $ और $ g \colon F \to E $ संतुष्ट $ g \circ f =
\mathrm{id}_E $. साबित करें कि $ f $ है \omterm{def:b1:logic:inj}{इंजेक्शन} और $ g $ है \omterm{def:b1:logic:inj}{विशेषण}.
एक उदाहरण दीजिए जहां न तो $ f $ और न $ g $ है \omterm{def:b1:logic:inj}{द्विभाजित}.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:logic:10}
पर $\R $, परिभाषित करना $ x \mathbin{\mathcal{R}} y \iff x^2 - y^2 = x - y $.
साबित करें कि $\mathcal{R}$ एक \omterm{def:b1:logic:equiv}{समतुल्य संबंध} और वर्णन करें
तुल्यता वर्ग प्रत्येक वास्तविक का $ x $. कौन सी कक्षाओं में बिल्कुल एक है
तत्व?
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b1:logic:11}
(कैंटर) चलो $ E $ एक हो \omterm{def:b1:logic:sets}{तय करना}. साबित करें कि कोई आपत्ति नहीं है $ E $
पर $\mathcal{P}(E)$. \emph{संकेत: दिया गया $ f \colon E \to
\mathcal{P}(E)$, विचार करना $ D = \{x \in E : x \notin f(x)\}$.}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b1:logic:12}
होने देना $ f \colon E \to F $ एक हो \omterm{def:b1:logic:map}{नक्शा}. परिभाषित करना $\Phi \colon \mathcal{P}(F)
\to \mathcal{P}(E)$ द्वारा $\Phi(B) = f^{-1}(B)$.
\begin{enumerate}
  \item साबित करें कि $ f $ है \omterm{def:b1:logic:inj}{विशेषण} अगर और केवल अगर $\Phi $ है
        \omterm{def:b1:logic:inj}{इंजेक्शन}.
  \item साबित करें कि $ f $ है \omterm{def:b1:logic:inj}{इंजेक्शन} अगर और केवल अगर $\Phi $ है
        \omterm{def:b1:logic:inj}{विशेषण}.
\end{enumerate}
\end{exercise}

\section{समस्या: अनन्तताओं की तुलना करना}

\begin{problem}\label{pb:b1:logic:1}
दो कब करें सेट ``तत्वों की समान संख्या'' है? कैंटर का
उत्तर---जब उनके बीच विरोध होता है---हो जाता है
अनंत के लिए भी प्रयोग करने योग्य सेट, और यह अनंत को वास्तव में विभाजित करता है
विभिन्न आकार. यह समस्या संपूर्ण टूलबॉक्स का निर्माण करती है
इस अध्याय की परिभाषाएँ: कैंटोर--श्र\"ओडर--बर्नस्टीन प्रमेय
(दो \omterm{def:b1:logic:inj}{इंजेक्शन} एक आक्षेप का निर्माण करते हैं), की गणनीयता $\Q $,
की बेशुमारता $\R $ विकर्ण तर्क और कैंटर द्वारा
आश्चर्यजनक 1874 निष्कर्ष: \emph{पारलौकिक संख्याएँ अस्तित्व में है, और
बड़े पैमाने पर तो}, एक भी प्रदर्शन किए बिना।
\index{गणनीय समुच्चय}\index{कैंटर--श्रोडर--बर्नस्टीन प्रमेय}
भर में, के लिए सेट $ E $ और $ F $, लिखना $ E \preceq F $ जब वहाँ
से एक \omterm{def:b1:logic:inj}{इंजेक्शन} मौजूद है $ E $ में $ F $, और $ E \approx F $ (``$ E $ और
$ F $ हैं \emph{सुसज्जित}'') जब कोई आपत्ति मौजूद हो $ E $
पर $ F $.

\medskip
\noindent\textbf{Part I --- The vocabulary of comparison.}
\begin{enumerate}
  \item बताते हैं कि $\approx $ की तरह व्यवहार करता है \omterm{def:b1:logic:equiv}{समतुल्य संबंध}:
        $ E \approx E $; अगर $ E \approx F $ तब $ F \approx E $; अगर
        $ E \approx F $ और $ F \approx G $ तब $ E \approx G $. (उद्धरण
        बिल्कुल \cref{thm:b1:logic:inverse} और
        \cref{prop:b1:logic:comp}.)
  \item बताते हैं कि $\preceq $ सकर्मक है, और वह एक \omterm{def:b1:logic:inj}{इंजेक्शन} है
        $ f \colon E \to F $ हमेशा प्रेरित करता है $ E \approx f(E)$.
  \item होने देना $ E \neq \emptyset $. बताते हैं कि $ E \preceq F $ यदि और केवल
        यदि कोई आपत्ति मौजूद है $ F $ पर $ E $.
  \item सत्यापित करें कि $ n \mapsto n + 1$ से एक आक्षेप है $\N $ पर
        $\N^* = \N \setminus \{0\}$, ओर वो
        \[
        \sigma(n) = \frac n2 \ \ (n \text{ even}), \qquad
        \sigma(n) = -\frac{n+1}2 \ \ (n \text{ odd})
        \]
        से एक आक्षेप है $\N $ पर $\Z $. अतः एक बिंदु हटाकर,
        या नकारात्मक में दोगुना होने से, का आकार नहीं बदलता है
        $\N $.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The Cantor--Schröder--Bernstein
theorem.} होने देना $ f \colon E \to F $ और $ g \colon F \to E $ दो हो
\omterm{def:b1:logic:inj}{इंजेक्शन}. परिभाषित करना
\[
C_0 = E \setminus g(F), \qquad C_{n+1} = g\bigl(f(C_n)\bigr)
\ \ (n \in \N), \qquad C = \bigcup_{n \in \N} C_n,
\]
और चलो $ h \colon E \to F $ भेजना $ x \in C $ को $ f(x)$, और $ x \notin C $
अद्वितीय के लिए $ y \in F $ साथ $ g(y) = x $.
\begin{enumerate}[resume]
  \item जाँच करें कि $ h $ अच्छी तरह से परिभाषित है: यदि $ x \notin C $ तब $ x \in
        g(F)$, और तत्व $ y $ साथ $ g(y) = x $ निराला है।
  \item बताते हैं कि $ g\bigl(f(C)\bigr) = \bigcup_{n \geq 1} C_n
        \subseteq C $. (प्रत्यक्ष छवियां यूनियनों के साथ चलती हैं:
        \cref{exo:b1:logic:8}.)
  \item बताते हैं कि $ h $ है \omterm{def:b1:logic:inj}{इंजेक्शन}. (तीन मामले; मिश्रित मामले में
        $ x \in C $, $ x' \notin C $, बताते हैं कि $ h(x) = h(x')$ होगा
        बल $ x' \in g(f(C)) \subseteq C $.)
  \item बताते हैं कि $ h $ है \omterm{def:b1:logic:inj}{विशेषण}: दिया गया $ y \in F $, भेद करना
        मामले $ g(y) \notin C $ और $ g(y) \in C_n $ कुछ के लिए $ n
        \geq 1$ (क्यों $ g(y) \in C_0$ असंभव?), और प्रदर्शन a
        \omterm{def:b1:logic:map}{पूर्वछवि} का $ y $ हर मामले में।
  \item के साथ समापन करें \emph{कैंटर--श्र\"ओडर--बर्नस्टीन
        प्रमेय}: अगर $ E \preceq F $ और $ F \preceq E $, तब $ E
        \approx F $. इसका क्या कारण है इस पर एक वाक्य में टिप्पणी करें
        \omterm{def:b1:logic:statement}{कथन} गैरतुच्छ.
  \item दो आवेदन. (एक शो $\intcc01 \approx \intoo01$.
        (बी) वह दिखाओ $\varphi(p, q) = 2^p(2q + 1) - 1$ ए को परिभाषित करता है
        से आपत्ति $\N \times \N $ पर $\N $ --- ए द्वारा \omterm{def:b1:logic:inj}{इंजेक्शन}
        समता तर्क, मजबूत प्रेरण द्वारा \omterm{def:b1:logic:inj}{विशेषण}
        (\cref{thm:b1:logic:induction}). इस तरह $\N \times \N \approx
        \N $: पूर्णांक बिंदुओं का तल इससे बड़ा नहीं है
        लाइन.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- \omterm{pb:b1:logic:1}{गणनीय समुच्चय}.} ए को कॉल करें \omterm{def:b1:logic:sets}{तय करना} $ E $
\emph{अधिक से अधिक गणनीय} कब $ E \preceq \N $, और \emph{गणनीय}
कब $ E \approx \N $.
\begin{enumerate}[resume]
  \item दिखाएँ कि प्रत्येक अनंत उपसमुच्चय $ A \subseteq \N $ है
        गणनीय. (परिभाषित करना $\varphi(n)$ पुनरावर्ती रूप से न्यूनतम के रूप में
        का तत्व $ A \setminus \{\varphi(0), \dots,
        \varphi(n-1)\}$; दिखाओ $\varphi $ सख्ती से बढ़ रहा है,
        संतुष्ट करता है $\varphi(n) \geq n $, और प्रत्येक तत्व को प्राप्त करता है
        $ A $.)
  \item निष्कर्ष निकालें कि ए \omterm{def:b1:logic:sets}{तय करना} अधिकतम गणनीय है यदि और केवल यदि यह है
        परिमित या गणनीय, और ध्यान दें कि प्रश्न 9 देता है
        शॉर्टकट: यदि $ E \preceq \N $ और $\N \preceq E $, तब $ E $ है
        गणनीय.
  \item दिखाओ अगर $ E $ और $ F $ अधिक से अधिक गणनीय हैं, इसलिए हैं $ E
        \times F $. उसका निष्कर्ष निकालो $\Z \times \N^*$ गणनीय है.
  \item बताते हैं कि $\Q $ गणनीय है. (\omterm{def:b1:logic:inj}{इंजेक्शन}) $\Q $ में $\Z \times
        \N^*$ प्रत्येक तर्कसंगत को सकारात्मक के साथ न्यूनतम शब्दों में लिखकर
        हर --- उस निरूपण की विशिष्टता है
        में साबित हुआ \cref{ch:b1:arith}; फिर प्रश्न 12 लागू करें।)
  \item दिखाएँ कि अधिक से अधिक एक गणनीय संघ है \omterm{pb:b1:logic:1}{गणनीय समुच्चय} पर है
        सर्वाधिक गणनीय: यदि प्रत्येक $ E_n $ ($ n \in \N $) अधिकतम है
        गणनीय, तो है $\bigcup_{n \in \N} E_n $. (भेजना $ x $ को
        जोड़ी $(n, f_n(x))$ कहाँ $ n $ है \emph{कम से कम} के साथ सूचकांक
        $ x \in E_n $.)
  \item दिखाओ कि \omterm{def:b1:logic:sets}{तय करना} का \emph{परिमित} के उपसमुच्चय $\N $ है
        गणनीय. (एक सीमित उपसमुच्चय का मानचित्र बनाएं $ F $ को $\sum_{i \in F} 2^i $;
        जहां दो सबसे बड़े तत्व की तुलना करके इंजेक्टिविटी साबित करें
        परिमित सेट भिन्न, उपयोग करना $\sum_{k=0}^{m-1} 2^k = 2^m - 1$
        से \cref{exo:b1:logic:4}.)
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Diagonalization.} होने देना $\{0,1\}^{\N}$
निरूपित करें \omterm{def:b1:logic:sets}{तय करना} के सभी एमएपीएस $ u \colon \N \to \{0, 1\}$, यानी\  \omterm{def:b1:logic:sets}{तय करना}
द्विआधारी अनुक्रमों का.
\begin{enumerate}[resume]
  \item के बीच एक आक्षेप का निर्माण करें $\mathcal{P}(\N)$ और
        $\{0,1\}^{\N}$ (सूचक कार्य)।
  \item (विकर्ण तर्क) चलो $\Phi \colon \N \to
        \{0,1\}^{\N}$ कोई भी हो \omterm{def:b1:logic:map}{नक्शा}. क्रम पर विचार करें $ d $ परिभाषित
        द्वारा $ d(n) = 1 - \Phi(n)(n)$. बताते हैं कि $ d $ में नहीं है
        की छवि $\Phi $, और निष्कर्ष निकालो $\{0,1\}^{\N}$ है
        \emph{नहीं} अधिक से अधिक गणनीय. एक वाक्य में बताएं क्यों,
        प्रश्न 17 के माध्यम से, यह बिल्कुल कैंटर का प्रमेय है
        (\cref{exo:b1:logic:11}) के लिए $ E = \N $.
  \item स्वीकार करें --- जैसा कि स्कूल से परिचित है, और स्थापित है
        सख्ती से अंदर \cref{ch:b1:reals} --- वह हर $ x \in
        \intco01$ एक अनोखा है \emph{उचित} दशमलव विस्तार $ x =
        0.d_1 d_2 d_3\dots $ (एक अनंत स्ट्रिंग में समाप्त नहीं हो रहा है
        $9$ एस)। किसी भी क्रम को देखते हुए $(x_n)_{n \geq 1}$ के तत्वों का
        $\intco01$, निर्माण $ x \in \intco01$ साथ $ x \neq x_n $ सभी के लिए $ n $: इसका चयन करें $ n $-वाँ अंक होना $5$ यदि $ n $-वें
        का अंक $ x_n $ से मतभेद होना $5$, और $6$ अन्यथा। औचित्य सिद्ध करें
        ध्यान से वह $ x $ उचित है और हर चीज़ से बचता है $ x_n $, और
        यह निष्कर्ष निकालें $\intco01$ अधिकतम गणनीय नहीं है.
  \item उसका निष्कर्ष निकालो $\R $ बेशुमार है, और वह \omterm{def:b1:logic:sets}{तय करना} $\R
        \setminus \Q $ अपरिमेय संख्याओं की भी गणना नहीं की जा सकती।
        किस सटीक अर्थ में ``अधिकांश'' वास्तविक संख्याएँ अपरिमेय हैं?
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Cantor's 1874 theorem: पारमार्थिक
नंबर exist.} एक वास्तविक संख्या $ x $ है \emph{बीजगणितीय}\index{बीजगणितीय} कब $ P(x) = 0$ कुछ अशून्य बहुपद के लिए
$ P $ पूर्णांक गुणांक के साथ, और \emph{ट्रान्सेंडैंटल}\index{ट्रान्सेंडैंटल} अन्यथा। इस भाग के लिए स्वीकार करें --- यह
में सिद्ध है \cref{ch:b1:poly} --- वह एक अशून्य बहुपद है
डिग्री $ n $ अधिकतम है $ n $ असली जड़ें.
\begin{enumerate}[resume]
  \item दिखाएँ कि प्रत्येक परिमेय संख्या \omterm{pb:b1:logic:1}{बीजगणितीय} है, और ज्ञात कीजिए
        पूर्णांक गुणांक वाले स्पष्ट बहुपद नष्ट हो रहे हैं
        $\sqrt 2$ और $\sqrt 2 + \sqrt 3$.
  \item तय के लिए $ n \in \N $, दिखाओ कि \omterm{def:b1:logic:sets}{तय करना} के बहुपदों का
        अधिकतम डिग्री $ n $ पूर्णांक गुणांक के साथ गणनीय है।
        (इसे इंजेक्ट करें $\Z^{n+1}$ और शामिल करें $ n $ के साथ
        प्रश्न 13.)
  \item यह निष्कर्ष निकालें कि \omterm{def:b1:logic:sets}{तय करना} का \emph{सभी} पूर्णांक वाले बहुपद
        गुणांक गणनीय है.
  \item सिद्ध करना \emph{कैंटर का प्रमेय पर \omterm{pb:b1:logic:1}{बीजगणितीय} संख्याएँ}: द \omterm{def:b1:logic:sets}{तय करना}
        $\mathcal{A}$ \omterm{pb:b1:logic:1}{बीजगणितीय} वास्तविक संख्याएँ गणनीय हैं।
  \item निष्कर्ष निकालें: पारलौकिक वास्तविक संख्याएँ मौजूद हैं, और \omterm{def:b1:logic:sets}{तय करना} का
        पारलौकिक संख्याएँ बेशुमार है. फिर जायजा लीजिए
        पूरी समस्या कुछ वाक्यों में: श्रृंखला $\N \approx
        \Z \approx \Q \approx \mathcal{A}$, सख्त छलांग $\R
        \approx $ (अनिवार्य रूप से) $\mathcal{P}(\N)$, जहां प्रत्येक उपकरण
        (कैंटर--श्र\"ओडर--बर्नस्टीन, गणनीय यूनियनें, द
        विकर्ण) निर्णायक था --- और दार्शनिक पंच
        यह साबित करना बेशुमार है पारलौकिक संख्याएँ मौजूद
        एक का भी नाम लिए बिना. (साबित करना ए \emph{विशिष्ट} संख्या
        जैसे $\pi$ पारलौकिक इससे कहीं अधिक कठिन और परे है
        आयतन.)
\end{enumerate}
\end{problem}
