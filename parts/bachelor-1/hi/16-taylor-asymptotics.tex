\chapter{टेलर सूत्र और स्पर्शोन्मुख विस्तार}\label{ch:b1:taylor}

एक बिंदु के निकट, एक सुचारु कार्य उतना ही अच्छा होता है \omterm{def:b1:poly:def}{बहुपद} --- एक के साथ
नियंत्रणीय त्रुटि. टेलर सूत्र इसे तीन में सटीक बनाते हैं
स्वाद (\omterm{thm:b1:integration:def}{अभिन्न} शेष, लैग्रेंज शेष, युवा शेष),
और परिणामी \emph{स्पर्शोन्मुख विस्तार}, हेरफेर किया गया
\omterm{pb:b1:logic:1}{बीजगणितीय} रूप से, प्रारंभिक विश्लेषण का सबसे तेज़ उपकरण बनें:
सीमाएँ, समकक्ष, स्थानीय व्यवहार, स्पर्शोन्मुख।

\section{तुलना संकेतन}

\begin{definition}[लैंडौ संकेतन]\label{def:b1:taylor:landau}
होने देना $f, g $ निकट परिभाषित किया जाए $ x_0$ ($ x_0 \in \R $ या $\pm\infty $). एक
लिखते हैं, जैसे $ x \to x_0$:\index{लैंडौ संकेतन}
\begin{itemize}
  \item $ f = o(g)$ (``लिटिल-ओ'') कब $ f = \varepsilon g $ साथ
        $\varepsilon(x) \to 0$;
  \item $ f = O(g)$ (``बिग-ओ'') कब $ f = u g $ साथ $ u $ के पास घिरा हुआ
        $ x_0$;
  \item $ f \sim g $ (``समकक्ष''\index{समतुल्य कार्य}) कब
        $ f = (1 + \varepsilon) g $ साथ $\varepsilon \to 0$ ---
        समकक्ष $ f - g = o(g)$.
\end{itemize}
वही संकेतन अनुक्रमों पर लागू होता है ($ n \to \infty $).
\end{definition}

\begin{proposition}[नियम]\label{prop:b1:taylor:landaurules}
जैसा $ x \to x_0$:
\begin{enumerate}
  \item $\sim $ एक \omterm{def:b1:logic:equiv}{समतुल्य संबंध}; $ f \sim g $ तात्पर्य यह है
        $ f $ और $ g $ शेयर \omterm{def:b1:topology:closure}{सीमा}, संकेत (निकट) $ x_0$), और शून्य'
        अनुपस्थिति;
  \item समकक्ष \emph{गुणा और भाग करें}: $ f_1 \sim g_1$, $ f_2
        \sim g_2$ संकेत करना $ f_1 f_2 \sim g_1 g_2$ और $\frac{f_1}{f_2}
        \sim \frac{g_1}{g_2}$;
  \item समकक्ष करते हैं \emph{नहीं} जोड़ना: $ x + 1 \sim x $ और $-x \sim
        -x + 2$ पर $+\infty $, फिर भी रकम $1$ और $2$ नहीं हैं
        समकक्ष. जोड़ने के लिए, स्पष्ट के साथ विस्तार पर वापस लौटें
        $ o(\cdot)$ शर्तें;
  \item $ o(g) + o(g) = o(g)$, $\;u \cdot o(g) = o(ug)$, $\;o(o(g)) =
        o(g)$, और $ f \sim g \iff f = g + o(g)$.
\end{enumerate}
\end{proposition}

\begin{proof}
प्रत्येक परिभाषाओं का एक संक्षिप्त हेरफेर है; उदाहरण के लिए
$ f_1 f_2 = (1+\varepsilon_1)(1+\varepsilon_2) g_1 g_2$ और $(1 +
\varepsilon_1)(1+\varepsilon_2) \to 1$. (4) की दो वस्तुएँ योग्य हैं
उनकी पंक्ति. $ u\cdot o(g) = o(ug)$: अगर $ f = \varepsilon g $ साथ
$\varepsilon \to 0$, तब $ uf = \varepsilon\,(ug)$ उसी के साथ
$\varepsilon $. $ o(o(g)) = o(g)$: अगर $ f = \varepsilon_1 h $ और
$ h = \varepsilon_2 g $ दोनों के साथ $\varepsilon_i \to 0$, तब $ f =
(\varepsilon_1\varepsilon_2) g $ और दोनों का उत्पाद
इनफिनिटिमल्स एक है। समतुल्यता $ f \sim g \iff f = g +
o(g)$ क्या परिभाषा दो बार पढ़ी गई है: $ f - g = \varepsilon g $.
(3) में प्रतिउदाहरण (3) का प्रमाण है।
\end{proof}

\begin{example}[तुलना का पैमाना]\label{ex:b1:taylor:scale}
जैसा $ x \to +\infty $, मानक पैमाना बढ़ता हुआ पढ़ता है
शक्ति का क्रम:
\[
1 \;=\; o(\ln x), \quad
\ln x = o(x^{0.01}), \quad
x^{0.01} = o(\sqrt x), \quad
\sqrt x = o(x^{10}), \quad
x^{10} = o(\eu^{x}), \quad
\eu^x = o(\eu^{2x}) ,
\]
हर कदम इसका एक उदाहरण है \omterm{prop:b1:functions:powerrules}{विकास तुलना} का
\cref{prop:b1:functions:powerrules} (शक्तियाँ लघुगणक को हराती हैं,
घातांक शक्तियों को हराते हैं, और एक परिवार के भीतर घातांक
निर्णय लेता है)। दो आदतें बनाने लायक: पहला, एक $ O(\cdot)$ उतरना
एक छोटी कक्षा में चुपचाप उन्नयन ($ O(\ln x)$ ई आल्सो
$ o(x^{0.01})$); दूसरा, पर $ x \to 0^+$ पूरी सीढ़ी
प्रतिस्थापन के माध्यम से उलट जाता है $ x \mapsto \frac1x $ ---
$\ln x = o(x^{-0.01})$ वहाँ, तो ``$ x^\alpha \ln x \to 0$''
प्रत्येक के लिए धारण करता है $\alpha > 0$. तराजू को सीधा रखना है
प्रत्येक स्पर्शोन्मुख तर्क का आधा
\cref{ch:b1:series}.
\end{example}

\begin{example}[विस्तार की विशिष्टता, और समता लाभांश]\label{ex:b1:taylor:uniqueness}
यदि कोई फ़ंक्शन दो विस्तार स्वीकार करता है $0$ उसी क्रम में,
\[
a_0 + a_1 x + \dots + a_n x^n + o(x^n)
= b_0 + b_1 x + \dots + b_n x^n + o(x^n),
\]
तब $ a_k = b_k $ हरएक के लिए $ k $: घटाना और सेट करना $ c_k =
a_k - b_k $, पहचान का मूल्यांकन करें $ c_0 + c_1 x + \dots + c_n x^n
= o(x^n)$ पर $ x \to 0$ पाने के $ c_0 = 0$; द्वारा विभाजित करें $ x $ और
दोहराएँ --- प्रत्येक विभाजन शेष के रूप में वैध है
अभिव्यक्ति फिर से है $ o(x^{n-k})$. गुणांक इसलिए हैं
आंतरिक, और कोई उनकी गणना इसके द्वारा कर सकता है \emph{कोई} मार्ग (टेलर
डेरिवेटिव, ज्ञात विस्तार पर बीजगणित, एकीकरण): सभी
मार्गों को सहमत होना चाहिए. लाभांश: ए \emph{यहां तक की} फ़ंक्शन केवल है
यहाँ तक कि शक्तियाँ भी अपने विस्तार में --- प्रतिस्थापित करती हैं $ x $ द्वारा $-x $ और आह्वान करें
विशिष्टता; इसी प्रकार विषम कार्यों में भी विषम शक्तियाँ होती हैं। इसलिए
$\cos $ किया जाता है $ o(x^{2p+1})$ इसके बजाय $ o(x^{2p})$ में
नीचे दी गई तालिका: अनुपस्थित विषम पद निःशुल्क जानकारी, एक क्रम है
बिना कुछ लिए परिशुद्धता की।
\end{example}

\section{तीन टेलर सूत्र}

\begin{theorem}[अभिन्न शेषफल के साथ टेलर]\label{thm:b1:taylor:integral}
होने देना $ f $ क्लास का हो $ C^{n+1}$ एक पर \omterm{prop:b1:reals:intervals}{अंतराल} युक्त $ a $ और $ x $.
तब\index{टेलर फार्मूला}
\[
f(x) = \sum_{k=0}^{n} \frac{f^{(k)}(a)}{k!}\,(x - a)^k
+ \int_a^x \frac{(x - t)^n}{n!}\, f^{(n+1)}(t)\, \dd t .
\]
\end{theorem}

\begin{proof}
प्रेरण चालू $ n $. के लिए $ n = 0$: $ f(x) = f(a) + \int_a^x f'(t)\dd t $ है
मौलिक प्रमेय (\cref{thm:b1:integration:ftc}). चरण:
शेष को भागों द्वारा एकीकृत करें,
\[
\int_a^x \frac{(x-t)^n}{n!} f^{(n+1)}(t)\,\dd t
= \Bigl[-\frac{(x-t)^{n+1}}{(n+1)!} f^{(n+1)}(t)\Bigr]_a^x
+ \int_a^x \frac{(x-t)^{n+1}}{(n+1)!} f^{(n+2)}(t)\,\dd t ,
\]
शब्द का योगदान करने वाला कोष्ठक $\frac{f^{(n+1)}(a)}{(n+1)!}(x -
a)^{n+1}$.
\end{proof}

\begin{example}[इसके शेषफल के साथ एक सटीक विस्तार]\label{ex:b1:taylor:lnexact}
के लिए $\ln(1 + x)$  \omterm{thm:b1:integration:def}{अभिन्न} शेष को पूर्णतः बनाया जा सकता है
बिना किसी भेदभाव के स्पष्ट $ n $ समय: एकीकृत करें
परिमित ज्यामितीय पहचान $\frac{1}{1+t} =
\sum_{k=0}^{n-1}(-t)^k + \frac{(-t)^n}{1+t}$ से $0$ को $ x $:
\[
\ln(1 + x) = \sum_{k=1}^{n} \frac{(-1)^{k-1}x^k}{k}
+ (-1)^n \int_0^x \frac{t^n}{1 + t}\,\dd t ,
\]
और के लिए $0 \leq x \leq 1$ शेषफल से परिबद्ध है $\int_0^x
t^n\,\dd t = \frac{x^{n+1}}{n+1}$. ये इससे भी ज्यादा मजबूत है
टेलर--दो तरह से युवा: यह एक है \emph{पहचान} ए के लिए मान्य
निश्चित $ x $ (न केवल $ x \to 0$), और त्रुटि \omterm{def:b1:topology:closure}{सीमा} संख्यात्मक है।
सप्ताहांत की समस्या (\cref{pb:b1:taylor:1}) ऐसे सटीक पर रहता है
रूप; टेलर--यंग नीचे \omterm{def:b1:topology:closure}{सीमा} के लिए हल्का उपकरण है, जहां
केवल त्रुटि का आकार मायने रखता है।
\end{example}

\begin{theorem}[टेलर - लैग्रेंज असमानता]\label{thm:b1:taylor:lagrange}
होने देना $ f $ होना $ C^{n+1}$ साथ $\abs{f^{(n+1)}} \leq M $ बीच में $ a $ और
$ x $. तब
\[
\Bigl| f(x) - \sum_{k=0}^{n} \frac{f^{(k)}(a)}{k!}(x-a)^k \Bigr|
\leq M\, \frac{\abs{x - a}^{n+1}}{(n+1)!} .
\]
\end{theorem}

\begin{proof}
बांध दिया \omterm{thm:b1:integration:def}{अभिन्न} शेष: $\bigl|\int_a^x \frac{(x-t)^n}{n!}
f^{(n+1)}(t)\,\dd t\bigr| \leq M \bigl|\int_a^x
\frac{\abs{x-t}^n}{n!}\dd t\bigr| = M\frac{\abs{x-a}^{n+1}}{(n+1)!}$.
\end{proof}

\begin{example}[प्रमाणित अंक]\label{ex:b1:taylor:certified}
क्या है $\sqrt{1.02}$? टेलर--लैग्रेंज को लागू करें $ f(t) =
\sqrt{1 + t}$ पर $ a = 0$, आदेश $2$, $ x = 0.02$:
\[
\sqrt{1.02} \approx 1 + \frac{0.02}{2} - \frac{0.02^2}{8}
= 1.00995 ,
\qquad
\abs{f'''(t)} = \frac{3}{8}(1+t)^{-5/2} \leq \frac 38 ,
\]
तो त्रुटि अधिकतम है $\frac38 \cdot \frac{0.02^3}{6} =
5\cdot10^{-7}$: $\sqrt{1.02} = 1.00995$ छह प्रमाणित के साथ
दशमलव (सही मान $1.0099504938\dots $ --- \omterm{def:b1:topology:closure}{सीमा} लगभग है
तेज़)। समापन अंतर्दृष्टि: टेलर--यंग केवल कहते हैं \emph{कैसे
तेज़} त्रुटि गायब हो जाती है; टेलर--लैग्रेंज उसी को रूपांतरित करता है
\omterm{def:b1:poly:def}{बहुपद} एक में \emph{प्रमाणपत्र}, एक संख्या प्लस एक सिद्ध
त्रुटि पट्टी. जब भी इस पुस्तक में दशमलव का दावा किया जाता है, a
इसके पीछे लैग्रेंज-टाइप बाउंड खड़ा है; सप्ताहांत की समस्या
(\cref{pb:b1:taylor:1}) विचार का औद्योगीकरण करता है।
\end{example}

\begin{theorem}[टेलर--युवा]\label{thm:b1:taylor:young}
होने देना $ f $ होना $ n $ टाइम्स विभेदक पर $ a $. फिर ऐसे $ x \to a $:
\[
f(x) = \sum_{k=0}^{n} \frac{f^{(k)}(a)}{k!}\,(x-a)^k
+ o\bigl((x-a)^n\bigr) .
\]
\end{theorem}

\begin{proof}
प्रेरण चालू $ n $. के लिए $ n = 1$ यह की परिभाषा है
यौगिक (\cref{def:b1:derivative:def}). मान लीजिये \omterm{def:b1:logic:statement}{कथन} पर
आदेश $ n - 1$, और चलो $ f $ होना $ n $ टाइम्स विभेदक पर $ a $. आवेदन करें
प्रेरण परिकल्पना $ f'$ (जो है $ n-1$ टाइम्स विभेदक
पर $ a $):
\[
f'(t) = \sum_{k=0}^{n-1} \frac{f^{(k+1)}(a)}{k!}(t-a)^k + r(t),
\qquad r(t) = o\bigl((t-a)^{n-1}\bigr).
\]
होने देना $ g(x) = f(x) - \sum_{k=0}^{n} \frac{f^{(k)}(a)}{k!}(x-a)^k $; तब
$ g' = r $ और $ g(a) = 0$. दिया गया $\varepsilon > 0$, चुनना $\delta $
साथ $\abs{r(t)} \leq \varepsilon\abs{t - a}^{n-1}$ के लिए $\abs{t-a}
\leq \delta $; औसत मूल्य असमानता
(\cref{thm:b1:derivative:mvt}) से खंड पर लागू किया गया $ a $ को
$ x $ (कहाँ $\abs{g'} \leq \varepsilon\abs{x-a}^{n-1}$) पैदावार
$\abs{g(x)} \leq \varepsilon\abs{x - a}^n $: बिल्कुल $ g(x) =
o((x-a)^n)$.
\end{proof}

\begin{remark}[तीन सूत्र, तीन कीमतें, तीन उत्पाद]
परिकल्पनाएं बिल्कुल निष्कर्षों के अनुरूप श्रेणीबद्ध होती हैं।
टेलर--यंग सबसे कम पूछता है ($ n $ डेरिवेटिव \emph{पर
बिंदु} केवल) और न्यूनतम उपज देता है: एक गुणात्मक $ o((x-a)^n)$,
\omterm{def:b1:topology:closure}{सीमा} के लिए बिल्कुल सही, प्रमाणित अंकों के लिए बेकार। लैग्रेंज
असमानता मांगती है $ C^{n+1}$ \emph{पर \omterm{prop:b1:reals:intervals}{अंतराल}} और एक बाध्य
$ M $ वहां, और एक संख्यात्मक त्रुटि पट्टी लौटाता है। \omterm{thm:b1:integration:def}{अभिन्न} प्रपत्र
समान नियमितता पूछता है और सबसे अधिक लौटाता है: त्रुटि के रूप में
स्पष्ट वस्तु को कोई रूपांतरित कर सकता है (भागों द्वारा एकीकृत, बाध्य)।
टुकड़े-टुकड़े करके, चर बदलें) --- यह वह रूप है जिसने इसे संचालित किया है
की अतार्किकता मशीन \cref{pb:b1:integration:1}. चुनना
लक्ष्य का समर्थन करने वाला सबसे कमजोर सूत्र पांडित्य नहीं है:
का फ्लैट फ़ंक्शन \cref{pb:b1:taylor:1} टेलर--यंग को संतुष्ट करता है
हर आदेश पर जबकि इसके बारे में हर मजबूत निष्कर्ष है
झूठ से दूर $0$.
\end{remark}

\begin{proposition}[मानक विस्तार पर $0$]
जैसा $ x \to 0$, प्रत्येक निश्चित ऑर्डर के लिए $ n $:
\index{टेलर विस्तार! मानक}
\begin{align*}
\eu^x &= 1 + x + \frac{x^2}{2!} + \dots + \frac{x^n}{n!} + o(x^n),\\
\cos x &= 1 - \frac{x^2}{2!} + \frac{x^4}{4!} - \dots
+ \frac{(-1)^p x^{2p}}{(2p)!} + o(x^{2p+1}),\\
\sin x &= x - \frac{x^3}{3!} + \dots + \frac{(-1)^p
x^{2p+1}}{(2p+1)!} + o(x^{2p+2}),\\
\frac{1}{1 - x} &= 1 + x + x^2 + \dots + x^n + o(x^n),\\
\ln(1 + x) &= x - \frac{x^2}{2} + \frac{x^3}{3} - \dots +
\frac{(-1)^{n-1} x^n}{n} + o(x^n),\\
(1 + x)^\alpha &= 1 + \alpha x + \frac{\alpha(\alpha-1)}{2!}x^2 +
\dots + \binom{\alpha}{n} x^n + o(x^n),
\end{align*}
कहाँ $\binom{\alpha}{n} = \frac{\alpha(\alpha - 1)\cdots(\alpha - n
+ 1)}{n!}$ वास्तव में $\alpha $. ($\cosh $ और $\sinh $: के समान $\cos $,
$\sin $ वैकल्पिक संकेतों के बिना।)
\label{prop:b1:taylor:standard}
\end{proposition}

\begin{proof}
प्रत्येक फ़ंक्शन के पास सुचारू है $0$ साथ डेरिवेटिव मूल्यांकन करना आसान:
$(\eu^x)^{(k)} = \eu^x $; डेरिवेटिव का $\sin $ और $\cos $ चक्र
अवधि के साथ $4$; $\bigl((1+x)^\alpha\bigr)^{(k)} = \alpha(\alpha -
1)\cdots(\alpha - k + 1)(1+x)^{\alpha - k}$;
$\bigl(\ln(1+x)\bigr)^{(k)} = \frac{(-1)^{k-1}(k-1)!}{(1+x)^k}$.
टेलर लागू करें - युवा पर $ a = 0$. (ज्यामितीय सटीक है:
$\frac{1}{1-x} - \sum_0^n x^k = \frac{x^{n+1}}{1 - x} = o(x^n)$.)
\end{proof}

\begin{method}[विस्तार के साथ कंप्यूटिंग]\label{met:b1:taylor:compute}
\begin{enumerate}
  \item \emph{लक्ष्य क्रम निश्चित करें} $ n $ सबसे पहले, और प्रत्येक को छोटा करें
        वहाँ मध्यवर्ती परिणाम---उच्च पदों को ले जाना है
        काम बर्बाद हुआ, निचले हिस्से को गिराना एक गलती है।
  \item \emph{रकम, उत्पाद}: क्रमानुसार प्रत्येक कारक का विस्तार करें $ n $ और
        गुणा करना, परे छोड़ना $ x^n $.
  \item \emph{संघटन} $ f(u(x))$ साथ $ u(x) \to 0$: स्थानापन्न
        का विस्तार $ u $ उसमें $ f $, क्रम दर क्रम।
  \item \emph{भागफल}: लिखना $\frac{1}{1 + v}$ साथ $ v \to 0$ और
        ज्यामितीय विस्तार का उपयोग करें.
  \item \emph{एकीकृत करें} पद दर पद विस्तार (विभेदीकरण)।
        अधिक देखभाल की आवश्यकता है --- औचित्य: \omterm{thm:b1:integration:def}{अभिन्न} का $ o(t^n)$
        से $0$ को $ x $ है $ o(x^{n+1})$, प्रत्यक्ष बाउंडिंग द्वारा)।
\end{enumerate}
\end{method}

\begin{example}[रचना, बहीखाता के साथ दिखाया गया है]\label{ex:b1:taylor:esin}
बढ़ाना $\eu^{\sin x}$ ऑर्डर करने के लिए $3$. आंतरिक विस्तार: $ u = \sin x
= x - \frac{x^3}{6} + o(x^3)$, जो वास्तव में होता है $0$. बाहरी:
$\eu^u = 1 + u + \frac{u^2}{2} + \frac{u^3}{6} + o(u^3)$, और
$ o(u^3) = o(x^3)$ तब से $ u \sim x $. की शक्तियां $ u $, पर काट दिया गया
$ x^3$:
\[
u^2 = x^2 + o(x^3), \qquad u^3 = x^3 + o(x^3)
\]
(क्रॉस टर्म $2x\cdot(-\frac{x^3}{6})$ पहले से $ x^4$).
इकट्ठा करें:
\[
\eu^{\sin x} = 1 + \Bigl(x - \frac{x^3}{6}\Bigr) + \frac{x^2}{2}
+ \frac{x^3}{6} + o(x^3)
= 1 + x + \frac{x^2}{2} + o(x^3) :
\]
दो $ x^3$ योगदान बिल्कुल रद्द करें. समापन अंतर्दृष्टि:
$\eu^{\sin x}$ और $\eu^x $ आदेश देने के लिए सहमत हैं $3$ --- नहीं क्योंकि $\sin x \approx x $ अपरिष्कृत रूप से, लेकिन क्योंकि पहला
प्रतिपादकों की असहमति ($-\frac{x^3}{6}$) प्रवेश करता है
से गुणा किया गया $\eu^0 = 1$ और फिर के घन पद से मिलता है
बाहरी घातांक; ऑर्डर-दर-ऑर्डर बहीखाता पद्धति इसका पता लगाती है
षडयंत्र, आँख मूँदना कभी नहीं चलता। (अगला पद है
$-\frac{x^4}{8}$: युद्धविराम क्रम पर समाप्त होता है $4$.)
\end{example}

\begin{example}\label{ex:b1:taylor:tan}
का विस्तार $\tan $ क्रम में $5$. लिखना $\tan x =
\sin x \cdot \frac{1}{\cos x}$:
\[
\frac{1}{\cos x} = \frac{1}{1 - \bigl(\frac{x^2}{2} - \frac{x^4}{24}
+ o(x^5)\bigr)}
= 1 + \Bigl(\frac{x^2}{2} - \frac{x^4}{24}\Bigr)
+ \Bigl(\frac{x^2}{2}\Bigr)^{\!2} + o(x^5)
= 1 + \frac{x^2}{2} + \frac{5x^4}{24} + o(x^5),
\]
तब
\[
\tan x = \Bigl(x - \frac{x^3}{6} + \frac{x^5}{120}\Bigr)
\Bigl(1 + \frac{x^2}{2} + \frac{5x^4}{24}\Bigr) + o(x^5)
= x + \frac{x^3}{3} + \frac{2x^5}{15} + o(x^5) .
\]
\end{example}

\begin{example}[सीमाएं]\label{ex:b1:taylor:limits}
\[
\lim_{x \to 0} \frac{x - \sin x}{x^3}:
\qquad
x - \sin x = \frac{x^3}{6} + o(x^3) \sim \frac{x^3}{6},
\qquad\text{so the limit is } \frac16
\]
--- में उठाए गए प्रश्न का निपटारा करना \cref{exo:b1:functions:9}.
वैसे ही $\displaystyle\lim_{x\to0}\Bigl(\frac{\sin
x}{x}\Bigr)^{1/x^2}$: लघुगणक है
\[
\frac{1}{x^2}\ln\Bigl(1 - \frac{x^2}{6} + o(x^2)\Bigr)
= \frac{1}{x^2}\Bigl(-\frac{x^2}{6} + o(x^2)\Bigr)
\longrightarrow -\frac16,
\qquad\text{limit } \eu^{-1/6}.
\]
\end{example}

\begin{remark}[विस्तार के साथ सामान्य ख़तरे]
(मैं) \emph{समकक्षों को कभी भी जोड़ें या घटाएँ नहीं}: से $\tan x \sim
x $ और $\sin x \sim x $ कोई भी हो सकता है \emph{नहीं} निष्कर्ष $\tan x -
\sin x \sim 0$ (अर्थहीन) --- ईमानदार मार्ग है
विस्तार:
\[
\tan x - \sin x
= \Bigl(x + \frac{x^3}{3}\Bigr) - \Bigl(x - \frac{x^3}{6}\Bigr)
+ o(x^3) = \frac{x^3}{2} + o(x^3) \sim \frac{x^3}{2} .
\]
(ii) \emph{नरसंहार से आगे बढ़ें}: उसी गणना में,
आदेश $1$ केवल देखता है $ x - x = 0$; जब भी प्रमुख पद रद्द हो जाते हैं,
जब तक एक गैर-शून्य गुणांक जीवित रहता है, तब तक ऑर्डर बढ़ाएं, और केवल
फिर वापस समकक्ष में परिवर्तित करें। (iii) \emph{समकक्ष करते हैं
घातांक से न गुजरें}: $ n^2 + n \sim n^2$, अभी तक
$\eu^{n^2+n} = \eu^{n}\,\eu^{n^2}$ है \emph{नहीं} के बराबर
$\eu^{n^2}$ --- केवल घातांक लगाएं \emph{का विस्तार
प्रतिपादक} जिसकी गलती हो जाती है $0$, कभी भी इसके समकक्ष नहीं
प्रतिपादक. (लघुगणक अधिक सुरक्षित हैं: यदि $ u_n \sim v_n \to \ell \neq
1$, $\ell > 0$, तब $\ln u_n \sim \ln v_n $.) (iv) \emph{
$ o(\cdot)$ कैलकुलस एक-दिशात्मक है}: $ o(x^2) + o(x^2) =
o(x^2)$, $5\,o(x^2) = o(x^2)$, $ x\cdot o(x^2) = o(x^3)$ --- लेकिन
एक $ o(x^2)$ यह कोई विशिष्ट फ़ंक्शन नहीं है, इसलिए इनमें से दो को कभी भी रद्द न करें
वे एक दूसरे के विरुद्ध: $ o(x^2) - o(x^2)$ है $ o(x^2)$, नहीं
$0$.
\end{remark}

\begin{proposition}[स्थानीय व्यवहार]\label{prop:b1:taylor:local}
कल्पना करना $ f(x) = f(a) + c\,(x - a)^p + o\bigl((x-a)^p\bigr)$ साथ $ c
\neq 0$ (स्थिरांक के बाद पहला अशून्य पद; $ p \geq 2$ एक पर
महत्वपूर्ण बिंदु).
\begin{itemize}
  \item अगर $ p $ सम है: $ f $ पर स्थानीय न्यूनतम है $ a $ अगर $ c > 0$, ए
        स्थानीय अधिकतम यदि $ c < 0$.
  \item अगर $ p $ अजीब है: कोई चरम \omterm{def:b1:topology:closure}{सीमा} नहीं ($ f - f(a)$ \omterm{def:b1:counting:objects}{परिवर्तन} चिह्न);
        इसके अलावा, विस्तार एक रैखिक पद के बाद शुरू होता है $ f'(a)(x
        - a)$, ग्राफ़ अपनी स्पर्श रेखा को पार करता है: a \emph{मोड़}.
\end{itemize}
\end{proposition}

\begin{proof}
पास में $ a $, $ f(x) - f(a) = (x-a)^p\bigl(c + o(1)\bigr)$ का चिन्ह है
का $ c\,(x-a)^p $: के लिए निरंतर संकेत $ p $ यहाँ तक कि, के लिए बदल रहा है $ p $ विषम।
\end{proof}

\begin{example}[घातांक का विस्तार किया जाना चाहिए $ o(1)$]\label{ex:b1:taylor:expdiscipline}
के समकक्ष खोजें $ u_n = \bigl(1 + \frac1n\bigr)^{n^2}$.
इसका विस्तार करें \emph{प्रतिपादक} जब तक उसकी त्रुटि ठीक न हो जाए $0$:
\[
n^2 \ln\Bigl(1 + \frac1n\Bigr)
= n^2\Bigl(\frac1n - \frac{1}{2n^2} +
O\Bigl(\frac{1}{n^3}\Bigr)\Bigr)
= n - \frac12 + O\Bigl(\frac1n\Bigr),
\]
इसलिए $ u_n = \eu^{\,n - 1/2}\,\eu^{O(1/n)}$ साथ $\eu^{O(1/n)} \to
1$:
\[
u_n \;\sim\; \eu^{\,n - \frac12} .
\]
ध्यान दें कि कम देखभाल से क्या गलत हो सकता था: रोकना
पर प्रतिपादक $ n^2\cdot\frac1n = n + O(1)$ एक छोड़ देता है
$\eu^{O(1)}$ कारक --- घिरा हुआ है लेकिन प्रवृत्त नहीं है $1$ --- और
किसी समकक्षता का दावा नहीं किया जा सकता. का नियम
उपरोक्त नुकसान, सकारात्मक रूप में: के बराबर
$\eu^{a_n}$ के विस्तार की आवश्यकता है $ a_n $ \emph{एक अवधि तक
शून्य की ओर रुझान}, उससे पहले के प्रत्येक गुणांक को रखा जा रहा है
बिलकुल.
\end{example}

\begin{example}[एक सपाट महत्वपूर्ण बिंदु को वर्गीकृत करना]\label{ex:b1:taylor:flatmin}
अध्ययन $ f(x) = \cos x + \frac{x^2}{2}$ पास में $0$. दोनों $ f'(0) =
0$ और $ f''(0) = -\cos 0 + 1 = 0$: द द्वितीय-व्युत्पन्न परीक्षण है
मूक. इसके बजाय विस्तार करें:
\[
f(x) = \Bigl(1 - \frac{x^2}{2} + \frac{x^4}{24} +
o(x^4)\Bigr) + \frac{x^2}{2}
= 1 + \frac{x^4}{24} + o(x^4) :
\]
पहला शून्येतर पद $ c\,x^p $ साथ $ p = 4$ सम और $ c =
\frac{1}{24} > 0$: एक स्थानीय न्यूनतम, असामान्य सपाटता का (द
ग्राफ़ अपना न्यूनतम मान छोड़ता है जैसे $ x^4$, नहीं $ x^2$). द
समापन अंतर्दृष्टि: विस्तार एक पंक्ति में देखता है कि क्या दोहराया गया है
विभेदन अस्पष्ट करता है --- और \cref{prop:b1:taylor:local} है
``प्रथम जीवित शब्द'' से व्यवस्थित शब्दकोश
``स्थानीय आकार''।
\end{example}

\begin{example}[अनंत पर विस्तार]\label{ex:b1:taylor:atinfinity}
दो संगणनाएँ जहाँ चर चलता है $+\infty $ और
प्रतिस्थापन $ h = \frac1x \to 0^+$ संपूर्ण टूलबॉक्स आयात करता है.
पहला,
\[
x - x^2\ln\Bigl(1 + \frac1x\Bigr)
= x - x^2\Bigl(\frac1x - \frac{1}{2x^2} +
O\Bigl(\frac{1}{x^3}\Bigr)\Bigr)
= \frac12 + O\Bigl(\frac1x\Bigr) \longrightarrow \frac12 .
\]
दूसरा, \omterm{def:b1:functions:arc}{आर्कटिक स्पर्शरेखा} अनंत पर: से $\arctan x +
\arctan\frac1x = \frac\pi2$ के लिए $ x > 0$
(\cref{prop:b1:functions:arcidentities}) और का विस्तार
$\arctan $ पर $0$ (\cref{exo:b1:taylor:3}),
\[
\arctan x = \frac\pi2 - \arctan\frac1x
= \frac\pi2 - \frac1x + \frac{1}{3x^3} +
o\Bigl(\frac{1}{x^3}\Bigr) :
\]
ग्राफ़ अपने अनंतस्पर्शी तक पहुंचता है $ y = \frac\pi2$ नीचे से
गति $\frac1x $. समापन अंतर्दृष्टि: कोई अलग नहीं है
अनंत पर विस्तार का सिद्धांत --- एक पारस्परिक प्रतिस्थापन
उन्हें विस्तार में कम कर देता है $0$, हर मध्यवर्ती प्रदान किया गया
$ O $ और $ o $ ईमानदारी से साथ निभाया जाता है.
\end{example}

\begin{example}[विस्तार द्वारा स्पर्शोन्मुख]\label{ex:b1:taylor:asymptote}
जैसा $ x \to +\infty $,
\[
\sqrt{x^2 + x} = x\sqrt{1 + \tfrac1x}
= x\Bigl(1 + \frac{1}{2x} - \frac{1}{8x^2} + o\bigl(\tfrac{1}{x^2}\bigr)\Bigr)
= x + \frac12 - \frac{1}{8x} + o\bigl(\tfrac 1x\bigr):
\]
रेखा $ y = x + \frac12$ एक स्पर्शोन्मुख है, संपर्क किया गया \emph{नीचे की ओर से} (अगला कार्यकाल $-\frac{1}{8x}$ नकारात्मक है)।
\end{example}

\begin{remark}[जहां विस्तार आगे काम करता है]
स्पर्शोन्मुख विस्तार शेष की स्थायी भाषा है
किताब: में \cref{ch:b1:series} वे अभिसरण का निर्णय लेते हैं
(समकक्ष तुलना परीक्षण और अध्ययन को फ़ीड करते हैं $\sum
\frac{1}{n^\alpha}$ भेस में एक विस्तार है); वर्ष 2 में
वे आयतन बन जाते हैं \emph{बिजली की श्रृंखला}, जहां टेलर
\omterm{def:b1:poly:def}{बहुपद} के अपरिमित रूप से अनेक पद और त्रिज्या प्राप्त करता है
अभिसरण; और भौतिकी में प्रत्येक रैखिककरण --- पेंडुलम,
प्रथम-क्रम गड़बड़ी --- एक टेलर--यंग है \omterm{def:b1:logic:statement}{कथन} साथ $ o(\cdot)$ चुपचाप गिरा दिया. उत्कीर्णन के लायक एक चेतावनी:
एक विस्तार केवल एक फ़ंक्शन का वर्णन करता है \emph{एक बिंदु के निकट} ---
सप्ताहांत समस्या का सपाट कार्य देखें, जिसका विस्तार पर है
$0$ फ़ंक्शन के ऐसा न होने पर यह समान रूप से शून्य है।
\end{remark}

\begin{remark}[इस खंड के अंदर परिप्रेक्ष्य]
विस्तार शेष विश्लेषण की कामकाजी भाषा है
और आने वाली ज्यामिति का। \Cref{ch:b1:series} उन्हें परिवर्तित करता है
अभिसरण निर्णयों में: सामान्य शब्द के बराबर है
एक विस्तार को इसके पहले कार्यकाल में छोटा कर दिया गया, और बेहतर परीक्षण
(त्रुटि नियंत्रण के साथ वैकल्पिक) दूसरे पद का भी उपभोग करें।
\Cref{ch:b1:curves} के विस्तार से स्थानीय ज्यामिति को पढ़ता है
दो समन्वय कार्य: क्या एक पैरामीट्रिज्ड वक्र पार करता है,
किसी बिंदु पर चुंबन, या चुंबन किसकी शक्तियों द्वारा तय किया जाता है $ t $
में जीवित रहना $ x(t)$ और $ y(t)$ --- का विमान संस्करण
\cref{prop:b1:taylor:local}. और \cref{ch:b1:multivar} पर रुकता है
उद्देश्य पर एक आदेश दें: स्पर्शरेखा तल एक दो-चर है
टेलर--युवा \omterm{def:b1:logic:statement}{कथन}, पूर्ण द्वितीय-क्रम सिद्धांत के साथ
(हेस्सियन्स, सैडल पॉइंट्स) को वर्ष 2 खंड के लिए स्थगित कर दिया गया। द
सामान्य सूत्र: इस पुस्तक में प्रत्येक ``स्थानीय'' प्रश्न का उत्तर दिया गया है
किसी विस्तार के पहले जीवित पद को लिखकर।
\end{remark}

\begin{exercise}[$\star $]\label{exo:b1:taylor:1}
पर विस्तार दीजिए $0$: $\eu^{2x}$ ऑर्डर करने के लिए $3$;
$\;\ln(1 - x)$ ऑर्डर करने के लिए $4$; $\;\sqrt{1 + x}$ ऑर्डर करने के लिए $3$;
$\;\dfrac{1}{1 + x^2}$ ऑर्डर करने के लिए $6$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:taylor:2}
\omterm{def:b1:topology:closure}{सीमा} की गणना करें:
\[
\lim_{x\to 0} \frac{\eu^x - 1 - x}{x^2},
\qquad
\lim_{x\to 0} \frac{\cos x - \sqrt{1 - x^2}}{x^4},
\qquad
\lim_{x\to 0} \frac{\ln(1+x) - \sin x}{x^2}.
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:taylor:3}
बढ़ाना $\arctan x $ पर $0$ ऑर्डर करने के लिए $5$ विस्तार को एकीकृत करके
का $\frac{1}{1 + x^2}$, और $\arcsin x $ ऑर्डर करने के लिए $5$ एकीकृत करके
का $(1 - x^2)^{-1/2}$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:taylor:4}
टेलर--लैग्रेंज का उपयोग करना $\exp $ पर $\intcc{0}{1}$, साबित करें कि
\[
\Bigl| \eu - \sum_{k=0}^{n} \frac{1}{k!} \Bigr| \leq
\frac{3}{(n+1)!},
\]
और एक निर्धारित करें $ n $ की गारंटी $6$ का सटीक दशमलव $\eu $.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:taylor:5}
ऑर्डर करने के लिए विस्तार करें $2$ में $\frac1n $ और \omterm{def:b1:topology:closure}{सीमा} घटाएं और
अभिसरण की गति:
\[
\Bigl(1 + \frac 1n\Bigr)^{\!n}
= \eu\Bigl(1 - \frac{1}{2n} + \frac{11}{24n^2} +
o\Bigl(\frac{1}{n^2}\Bigr)\Bigr).
\]
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:taylor:6}
स्थानीय व्यवहार का अध्ययन करें $0$ का $ f(x) = x^2 - x^4$ और का $ g(x) =
x^3 + x^5$; और ग्राफ की स्थिति ज्ञात कीजिए $ h(x) = \eu^x $
इसके स्पर्शरेखा के सापेक्ष $ a = 1$, स्थानीय स्तर पर फिर वैश्विक स्तर पर।
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:taylor:7}
पर अनंतस्पर्शी निर्धारित करें $\pm\infty $ का $ f(x) =
\sqrt[3]{x^3 + x^2}$ और वक्र की स्थिति के सापेक्ष
उन्हें.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:taylor:8}
समतुल्य ज्ञात कीजिए, जैसे $ n \to \infty $, का
\[
u_n = \sqrt{n+1} - \sqrt n, \qquad
v_n = \ln(n+1) - \ln n, \qquad
w_n = \sin\frac{1}{n} - \tan\frac{1}{n},
\]
प्रत्येक की एक शक्ति के रूप में $ n $ कई बार एक स्थिरांक.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b1:taylor:9}
होने देना $ f $ होना $ C^2$ पर $\R $. इसे हर एक के लिए साबित करें $ x $ और $ h > 0$:
\[
\abs{f'(x)} \leq \frac{\abs{f(x+h) - f(x-h)}}{2h} +
\frac{h}{2}\sup_{\intcc{x-h}{x+h}}\abs{f''} ,
\]
और लैंडौ-कोलमोगोरोव-प्रकार की असमानता को कम करें: यदि $\abs f \leq
M_0$ और $\abs{f''} \leq M_2$ पर $\R $, तब $\abs{f'} \leq
\sqrt{2 M_0 M_2}$ हर जगह. \emph{(ऑप्टिमाइज़ करें $ h $.)}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b1:taylor:10}
क्रम $ u_0 \in \intoo{0}{\pi}$, $ u_{n+1} = \sin u_n $
तक कम हो जाता है $0$ (संक्षेप में औचित्य सिद्ध करें)। इसकी गति ज्ञात करने के लिए विचार करें $ v_n
= \frac{1}{u_n^2}$:
\begin{enumerate}
  \item के विस्तार का उपयोग करना $\sin $, सिद्ध करना $ v_{n+1} - v_n \to
        \frac13$;
  \item सेस के साथ\`एआरओ (\cref{exo:b1:seq:10}), निष्कर्ष निकालना $\frac{v_n}{n}
        \to \frac13$, फिर समतुल्य $ u_n \sim
        \sqrt{\dfrac{3}{n}}$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:taylor:11}
(अनंत के अंतर) गणना करें
\[
\lim_{x \to 0} \Bigl(\frac{1}{x^2} - \frac{1}{\sin^2 x}\Bigr)
\qquad\text{and}\qquad
\lim_{x \to 0^+} \Bigl(\frac 1x - \frac{1}{\ln(1 + x)}\Bigr)
\]
एक सामान्य हर को घटाकर और अंश का विस्तार करके और
हर अलग से.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b1:taylor:12}
(अंतर्निहित जड़ों के स्पर्शोन्मुख) प्रत्येक के लिए दिखाएँ $ k \in
\N^*$ समीकरण $\tan x = x $ बिल्कुल एक ही समाधान है $ x_k $
में $\intoo{k\pi - \frac\pi2}{k\pi + \frac\pi2}$, वह $ x_k = k\pi
+ \frac\pi2 - \varepsilon_k $ साथ $\varepsilon_k =
\arctan\frac{1}{x_k}$, और विस्तार का अनुमान लगाएं
\[
x_k = k\pi + \frac\pi2 - \frac{1}{k\pi} + o\Bigl(\frac
1k\Bigr) \qquad (k \to \infty).
\]
\end{exercise}

\section{समस्या: वैकल्पिक योग, प्रमाणित अंक, और
की अतार्किकता $\cos 1$}

\begin{problem}[{सप्ताहांत समस्या --- वैकल्पिक अनुमान $\abs{S - S_n} \leq a_{n+1}$: $\ln 2$ और $\pi $ सिद्ध के साथ दशमलव, मशीन का सूत्र, और $\cos 1 \notin \Q $}]
\label{pb:b1:taylor:1}
घटते पदों के साथ एक वैकल्पिक योग सबसे अनुकूल है
संख्यात्मक विश्लेषण में वस्तु: इसकी त्रुटि पहले से सीमित है
छोड़ा गया शब्द, \emph{ज्ञात चिन्ह के साथ}. यह समस्या यह सिद्ध करती है
आसन्न-अनुक्रम प्रमेय के साथ सिद्धांत, फिर इसे खर्च करता है
तीन तरीके: प्रमाणित दशमलव के लिए $\ln 2$ (तीन प्रतिस्पर्धा कर रहे हैं
मार्ग) और के लिए $\pi $ (लीबनिज, तब माचिन का 1706 फॉर्मूला, अभी भी
सदियों से रिकॉर्ड गणनाओं के पीछे का विचार),
की अतार्किकता $\cos 1$, $\sin 1$ और $\cosh 1$, और, एक के रूप में
काउंटरवेट, टेलर--लैग्रेंज \emph{समानता} और फ्लैट
फ़ंक्शन जिसका टेलर विस्तार निहित है।\index{वैकल्पिक श्रृंखला
अनुमान}\index{मशीन का सूत्र}\index{irrationality!of cos 1@of
$\cos 1$} कुल मिलाकर, ``श्रृंखला'' की भाषा अनौपचारिक है: प्रत्येक योग
यहाँ एक है \emph{आंशिक रकम का क्रम}, के रूप में
\cref{ex:b1:seq:e}; सिद्धांत उचित खुलता है में
\cref{ch:b1:series}.

\medskip
\noindent\textbf{Part I --- The alternating estimate.} होने देना
$(a_k)_{k \geq 0}$ को कम करें $0$ और $ S_n = \sum_{k=0}^{n}
(-1)^k a_k $.
\begin{enumerate}
  \item बताते हैं कि $(S_{2n+1})$ घट नहीं रहा है, $(S_{2n})$
        गैर-बढ़ते हुए, और यह कि वे आसन्न हैं
        (\cref{thm:b1:seq:adjacent}): दोनों एक समान में परिवर्तित हो जाते हैं
        $ S $ के साथ, हर किसी के लिए $ n $,
        \[
        S_{2n+1} \leq S \leq S_{2n},
        \qquad
        \abs{S - S_n} \leq a_{n+1} ,
        \]
        पहले छोड़े गए पद के चिह्न वाली त्रुटि।
        इसके अलावा दिखाएँ कि यदि कमी है \emph{कठोर}, सब
        ये असमानताएँ सख्त हैं।
  \item पहला लाभांश: के लिए $ x = 1$ घातीय श्रृंखला में,
        तुलना करें \cref{thm:b1:taylor:lagrange} पर $ a = 0$:
        बताते हैं कि $ T_n = \sum_{k=0}^{n} \frac{(-1)^k}{k!}$
        में एकत्रित हो जाता है $\eu^{-1}$ साथ $\abs{\eu^{-1} - T_n} <
        \frac{1}{(n+1)!}$.
  \item (लीबनिज़, 1674) से \emph{एकदम सही} परिमित पहचान
        \[
        \frac{1}{1 + t^2} = \sum_{k=0}^{n} (-1)^k t^{2k}
        + \frac{(-1)^{n+1} t^{2n+2}}{1 + t^2} ,
        \]
        पर एकीकृत $\intcc{0}{1}$, सिद्ध करना
        \[
        \frac\pi4 = \sum_{k=0}^{n} \frac{(-1)^k}{2k+1} + \rho_n,
        \qquad \abs{\rho_n} \leq \frac{1}{2n+3} .
        \]
  \item धीमापन: लीबनिज़ की कितनी शर्तें छह सटीक की गारंटी देती हैं
        का दशमलव $\pi $? (लगभग दो मिलियन।) मूल्यांकन करें $4 S_4
        = 4\bigl(1 - \frac13 + \frac15 - \frac17 +
        \frac19\bigr)$ और इसकी दूरी $\pi $, महसूस करने के लिए
        दर्द.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- $\ln 2$ three ways.}
\begin{enumerate}[resume]
  \item (तरीका 1: वैकल्पिक हार्मोनिक) से $\frac{1}{1+t} =
        \sum_{k=0}^{n-1}(-1)^k t^k + \frac{(-1)^n t^n}{1+t}$
        पर एकीकृत $\intcc{0}{1}$:
        \[
        \ln 2 = \sum_{k=1}^{n} \frac{(-1)^{k-1}}{k} + (-1)^n
        R_n, \qquad \frac{1}{2(n+1)} \leq R_n \leq
        \frac{1}{n+1} :
        \]
        त्रुटि सटीक क्रम की है $\frac 1n $ --- एक लाख
        छह दशमलव के लिए शर्तें.
  \item (तरीका 2: तेज़ श्रृंखला) एकीकृत करें $\frac{1}{1 - t^2} =
        \sum_{k=0}^{n} t^{2k} + \frac{t^{2n+2}}{1-t^2}$ से $0$
        को $ x \in \intoo{0}{1}$ और मूल्यांकन करें $ x = \frac13$
        (टिप्पणी $\frac{1 + 1/3}{1 - 1/3} = 2$):
        \[
        \ln 2 = 2\sum_{k=0}^{n} \frac{(1/3)^{2k+1}}{2k+1} +
        \tilde\rho_n,
        \qquad
        0 < \tilde\rho_n \leq \frac{9}{4}\cdot
        \frac{(1/3)^{2n+3}}{2n+3} :
        \]
        ज्यामितीय अभिसरण, प्रति पद लगभग एक अंक।
  \item (रास्ता 3: \omterm{thm:b1:integration:riemann}{रीमैन रकम} और एक छिपी हुई पहचान) द्वारा सिद्ध करें
        पहचान प्रेरण
        \[
        \sum_{k=1}^{2n} \frac{(-1)^{k-1}}{k} = H_{2n} - H_n =
        \sum_{k=1}^{n} \frac{1}{n + k} ,
        \]
        और ठीक हो जाओ $\ln 2$ रीमैन-योग \omterm{def:b1:topology:closure}{सीमा} के रूप में
        \cref{ex:b1:integration:riemannexample}: तरीके 1 और 3
        क्या गुप्त रूप से एक ही नंबर दो बार देखा जाता है?
  \item छह शब्दों पर गोलीबारी: तुलना करें $\sum_{k=1}^{6}
        \frac{(-1)^{k-1}}{k} = 0.6167$ रास्ता 2 के साथ $ n = 5$,
        जो पहले से ही देता है $\ln 2 = 0.693147$ त्रुटि के साथ $\leq
        1.1\cdot10^{-7}$. संरचनात्मक कारण स्पष्ट कीजिए
        (मूल्यांकन बिंदु गहराई से अंदर \omterm{prop:b1:reals:intervals}{अंतराल} का
        अभिसरण बनाम इसके पर \omterm{def:b1:topology:closure}{सीमा}).
  \item मार्ग 2 की कितनी शर्तें प्रमाणित करती हैं \emph{दस} का दशमलव
        $\ln 2$? दिखाओ $ n = 10$ पर्याप्त.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Machin's formula.}
\begin{enumerate}[resume]
  \item गणना करना $(5 + \iu)^4$ और जटिल पहचान सत्यापित करें
        \[
        (5 + \iu)^4 = 2\,(1 + \iu)\,(239 + \iu) .
        \]
        तर्क लेना (\cref{ch:b1:complex} कन्वेंशन),
        मशीन का सूत्र निकालें
        \[
        \frac\pi4 = 4\arctan\frac15 - \arctan\frac{1}{239} .
        \]
        (जाँचें कि कोई तर्क न छूटे
        $\intoo{-\frac\pi2}{\frac\pi2}$.)
  \item जैसा कि प्रश्न 3 में है, के लिए स्थापित करें $0 < x < 1$:
        \[
        \arctan x = \sum_{k=0}^{n} \frac{(-1)^k
        x^{2k+1}}{2k+1} + r_n(x), \qquad \abs{r_n(x)} \leq
        \frac{x^{2n+3}}{2n+3} .
        \]
  \item प्रमाणित $\pi $ छह पदों के साथ सात दशमलव तक: बाध्य
        की कुल त्रुटि
        \[
        \pi \approx 16\sum_{k=0}^{4}
        \frac{(-1)^k (1/5)^{2k+1}}{2k+1}
        - 4\Bigl(\frac{1}{239} - \frac{1}{3\cdot239^3}\Bigr)
        \]
        द्वारा $16\,\frac{(1/5)^{11}}{11} +
        4\,\frac{(1/239)^5}{5} < 5\cdot10^{-8}$, और दे दो
        परिणामी मूल्य $3.1415926\dots $
  \item तीन मार्गों की तुलना करें $\pi $ अब उपलब्ध ---
        लीबनिज़ (प्रश्न 4), द डेलज़ेल \omterm{thm:b1:integration:def}{अभिन्न} का
        \cref{pb:b1:integration:1} (गलती $4^{1-5m}$), मशीन
        (त्रुटि $\approx 16\cdot 5^{-(2n+3)}$)---अंकों में प्रति
        शब्द, और बताएं कि मूल्यांकन बिंदु क्यों सिकुड़ रहा है
        सब कुछ हरा देता है.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The integer trap, alternating
edition.}
\begin{enumerate}[resume]
  \item कल्पना करना $\cos 1 = \frac pq $. सख्त को गुणा करें
        की वैकल्पिक ब्रैकेटिंग $\sum_{k}
        \frac{(-1)^k}{(2k)!}$ (प्रश्न 1) द्वारा $(2n)!$ साथ $2n
        \geq q $, और एक विरोधाभास प्राप्त करें: $\cos 1$ है
        तर्कहीन.
  \item के अनुकूल होना $\sin 1 = \sum_k \frac{(-1)^k}{(2k+1)!}$
        (गुणा करके $(2n+1)!$): $\sin 1 \notin \Q $. दोनों
        तर्कहीन, फिर भी $\cos^2 1 + \sin^2 1 = 1$: अतार्किकता
        बीजगणित के अंतर्गत स्थिर नहीं है।
  \item गैर-वैकल्पिक चचेरा भाई: $\cosh 1 = \sum_k
        \frac{1}{(2k)!}$ (आंशिक-योग अर्थ में, के साथ
        दो तरफा पूँछ बँधी हुई $0 < \cosh 1 - \sum_{k \leq n}
        \frac{1}{(2k)!} < \frac{2}{(2n+2)!}$, सिद्ध किया जाना है)।
        निष्कर्ष निकालना $\cosh 1 \notin \Q $ उसी जाल से.
  \item की ओर धकेलें $\cos\frac 1m $ प्रत्येक पूर्णांक के लिए $ m \geq 1$:
        गुणा करके $ m^{2n}(2n)!$ और निष्कर्ष निकालो $\cos\frac1m
        \notin \Q $. वही कोशिश कहां टूटती है
        $\cos\frac ab $ साथ $ b > 1$ सामान्य? (पहचानें
        हर जो अब साफ़ नहीं होता।)
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Sharper and darker: the equality
form, and a function that fools Taylor.}
\begin{enumerate}[resume]
  \item (टेलर - लैग्रेंज, समानता रूप) चलो $ f $ होना $ n + 1$
        टाइम्स विभेदक बीच में $ a $ और $ x $. परिभाषित करना $ g(t) =
        f(x) - \sum_{k=0}^{n} \frac{f^{(k)}(t)}{k!}(x - t)^k -
        A\,\frac{(x-t)^{n+1}}{(n+1)!}$ स्थिरांक के साथ $ A $
        इसलिए चुना गया $ g(a) = 0$. गणना करना $ g'$ (योग
        दूरबीन), रोले को लागू करें $\intcc{a}{x}$, और निष्कर्ष निकालें
        वह मौजूद है $ c $ सख्ती से बीच में $ a $ और $ x $ साथ
        \[
        f(x) = \sum_{k=0}^{n} \frac{f^{(k)}(a)}{k!}(x-a)^k
        + \frac{f^{(n+1)}(c)}{(n+1)!}\,(x-a)^{n+1} .
        \]
  \item समानता का लाभांश: के लिए $ x > 0$ दिखाओ
        \[
        \eu^x > 1 + x + \frac{x^2}{2!} + \dots +
        \frac{x^n}{n!}
        \]
        (कड़ाई से, प्रत्येक के लिए $ n $), और पता लगाएं कि कहां
        असमानता उलट जाती है $ x < 0$ समता के अनुसार
        का $ n $.
  \item शेषों को बेंचमार्क करें $\sin(0.5)$ क्रम में $3$:
        युवा ही देता है $ o(x^3)$ (कोई संख्या नहीं); लैग्रेंज देता है
        $\abs{\sin 0.5 - (0.5 - \frac{0.5^3}{6})} \leq
        \frac{0.5^5}{120} = 2.61\cdot10^{-4}$; प्रत्यावर्तन
        अनुमान समान \omterm{def:b1:topology:closure}{सीमा} देता है \emph{प्लस} संकेत
        जानकारी $\sin 0.5 > 0.5 - \frac{0.5^3}{6}$. तुलना करें
        सच्ची त्रुटि के साथ $2.59\cdot10^{-4}$: \omterm{def:b1:topology:closure}{सीमा} है
        लगभग प्राप्त कर लिया। आप किस उपकरण तक पहुंचेंगे, और
        कब?
  \item (फ्लैट फ़ंक्शन) चलो $ f(x) = \eu^{-1/x^2}$ के लिए $ x
        \neq 0$, $ f(0) = 0$. दिखाओ $ f $ है निरंतर पर $0$, वह
        $ f'(0) = 0$, और अधिक सामान्यतः --- इसे साबित करके
        प्रत्येक यौगिक रूप है $ f^{(k)}(x) =
        P_k\bigl(\frac1x\bigr)\eu^{-1/x^2}$ एक के लिए \omterm{def:b1:poly:def}{बहुपद}
        $ P_k $ (प्रेरण)---वह $ f^{(k)}(0) = 0$ सभी के लिए $ k $
        \emph{(\omterm{prop:b1:functions:powerrules}{विकास तुलना}
        \cref{prop:b1:functions:powerrules})}. निष्कर्ष: सभी
        टेलर बहुआयामी पद का $ f $ पर $0$ गायब हो जाओ, फिर भी $ f(x) > 0$
        के लिए $ x \neq 0$: टेलर--यंग हर आदेश पर पकड़ रखता है, और
        के बारे में कुछ नहीं कहता $ f $ से दूर $0$. विस्तार
        वर्णन करें \emph{कीटाणुओं}, कार्य नहीं.
\end{enumerate}

\noindent\textbf{Part VI --- Synthesis.}
\begin{enumerate}[resume]
  \item जाल को एक बार फिर चलाओ, चालू करो $\eu^{-1} = \sum_k
        \frac{(-1)^k}{k!}$: सख्त विकल्प को गुणा करें
        द्वारा ब्रैकेटिंग $ n!$ और निष्कर्ष निकालो $\eu^{-1} \notin \Q $,
        इस तरह $\eu \notin \Q $ ---इस तथ्य का तीसरा प्रमाण
        मात्रा में. तीन की सूची बनाएं (\omterm{thm:b1:seq:adjacent}{आसन्न अनुक्रम},
        \cref{exo:b1:seq:9}; \omterm{thm:b1:integration:def}{अभिन्न},
        \cref{pb:b1:integration:1}; वैकल्पिक योग, यहां) और
        प्रत्येक को क्या चाहिए।
  \item बढ़िया प्रिंट: एकरसता सजावटी नहीं है। होने देना
        $ b_k = \frac1k $ विषम के लिए $ k $ और $ b_k = \frac{1}{k^2}$
        एक जैसे के लिए $ k $: द $ b_k $ सकारात्मक हैं और प्रवृत्ति रखते हैं $0$,
        अभी तक का आंशिक योग $\sum (-1)^k b_k $ की ओर विचलन
        $-\infty $. इसे साबित करो \emph{(आंशिक योग को विभाजित करें
        सम भाग, से घिरा हुआ
        \cref{ex:b1:seq:basel}, और विषम भाग, जो
        आधे हार्मोनिक श्रृंखला पर हावी है,
        \cref{exo:b1:seq:5})}, और बताएं कि वास्तव में कौन सा चरण है
        प्रश्न 1 में एकरसता का प्रयोग किया गया।
  \item यूलर की सरल पहचान सत्यापित करें $\arctan\frac12 +
        \arctan\frac13 = \frac\pi4$ के जरिए $(2 + \iu)(3 + \iu) =
        5(1 + \iu)$, छह दशमलव के लिए आवश्यक पदों का अनुमान लगाएं
        का $\pi $ इस मार्ग से ($ n = 10$ पर्याप्त), और इसे रखें
        प्रश्न की रैंकिंग में लीबनिज़ और माचिन के बीच
        13.
  \item संश्लेषण, प्रत्येक एक वाक्य: (i) प्रत्यावर्तन बताएं
        अनुमान और उसके दो आउटपुट (बाध्य और चिह्न); (ii) क्यों
        स्पष्ट अवशेषों के साथ सटीक परिमित पहचानें हराती हैं
        \omterm{def:b1:topology:closure}{सीमा} \omterm{def:b1:logic:statement}{कथन} प्रमाणित अंकों के लिए; (iii) सूची
        समस्या का ($\pi $ को $10^{-7}$ हाथ से, $\ln 2$ को
        दस दशमलव, चार अतार्किकता प्रमाण, एक समानता
        प्रमेय, एक चेतावनी उदाहरण); (iv) इनमें से कौन सा
        धागे \cref{ch:b1:series} (वैकल्पिक रूप से) उठाएगा
        श्रृंखला परीक्षण, निरपेक्ष बनाम सशर्त अभिसरण,
        और इसकी सप्ताहांत समस्या का पुनर्व्यवस्था नाटक)।
\end{enumerate}
\end{problem}
