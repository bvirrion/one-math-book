\chapter{रैखिक विभेदक समीकरण}\label{ch:b1:diffeq}

पहली बार हाई स्कूल वॉल्यूम में मिले, अंतर समीकरणों का इलाज किया जाता है
यहां पूर्ण प्रमाणों के साथ और अधिक व्यापकता में: प्रथम-क्रम रैखिक
चर गुणांक वाले समीकरण (द्वारा पूरी तरह से हल किए गए
भिन्नता-की-स्थिरांक विधि) और दूसरे क्रम के रैखिक समीकरण
निरंतर गुणांक, दोलनों के लिए मॉडल। दोनों मामले प्रदर्शित होते हैं
वही संरचना: \emph{सामान्य समाधान $=$ एक विशेष समाधान
$+$ सजातीय समीकरण का सामान्य समाधान}.

\section{प्रथम-क्रम रैखिक समीकरण}

\begin{definition}\label{def:b1:diffeq:linear1}
होने देना $ I $ एक अंतराल हो और $ a, b \colon I \to \R $ (या $\C $) हो
निरंतर. समीकरण
\[
(E)\colon\quad y' + a(x)\,y = b(x),
\]
अज्ञात अवकलनीय फलन में $ y \colon I \to \R $ (या $\C $),
एक है \emph{प्रथम-क्रम रैखिक अंतर
समीकरण}\index{प्रथम-क्रम रैखिक अंतर
समीकरण}. द
समीकरण $(H)\colon y' + a(x) y = 0$ इसका है \emph{सजातीय}
समीकरण.
\end{definition}

\begin{theorem}[सजातीय समीकरण को हल करना]\label{thm:b1:diffeq:homogeneous1}
होने देना $ A $ का आदिम हो $ a $ पर $ I $ (यह मौजूद है:
\cref{ch:b1:integration}). के समाधान $(H)$ पर $ I $ बिलकुल हैं
कार्य
\[
y(x) = \lambda\, \eu^{-A(x)}, \qquad \lambda \in \R \text{ (or } \C).
\]
\end{theorem}

\begin{proof}
ये कार्य समाधान हैं: $ y' = -\lambda A' \eu^{-A} = -a y $.
इसके विपरीत, चलो $ y $ हल करना $(H)$ और सेट करें $ z(x) = y(x)\, \eu^{A(x)}$.
तब
\[
z' = y' \eu^{A} + y\, a\, \eu^{A} = (y' + ay)\, \eu^{A} = 0 ,
\]
इसलिए $ z $ अंतराल पर स्थिर है $ I $, कहना $ z = \lambda $: $ y =
\lambda \eu^{-A}$. (तर्क पर ध्यान दें: कोई समाधान नहीं खोता है, क्योंकि
\emph{प्रत्येक} समाधान घोषित रूप में लिखा गया है।)
\end{proof}

\begin{example}[चर गुणांक के साथ एक सजातीय समीकरण]\label{ex:b1:diffeq:varcoeff}
हल करना $ y' + (\cos x)\,y = 0$ पर $\R $. का एक आदिम $ a(x) =
\cos x $ है $ A(x) = \sin x $, तो समाधान हैं
\[
y(x) = \lambda\,\eu^{-\sin x}, \qquad \lambda \in \R .
\]
दो वाचन. प्रत्येक समाधान आवधिक (अवधि) होता है $2\pi $) और कभी नहीं
जब तक गायब नहीं हो जाता $\lambda = 0$ --- का संकेत $\lambda $ है
का संकेत $ y $ हमेशा के लिए, चूँकि एक घातांक शून्य को पार नहीं कर सकता। और
समाधान के माध्यम से $ y(0) = y_0$ है $ y_0\eu^{-\sin x}$: बिलकुल
प्रत्येक प्रारंभिक बिंदु के माध्यम से परिवार का एक वक्र
का एक आयामी चित्र \cref{thm:b1:diffeq:voc}~(2).
\end{example}

\begin{theorem}[स्थिरांक की भिन्नता; कॉची समस्या]\label{thm:b1:diffeq:voc}
उपरोक्त संकेतन के साथ:
\begin{enumerate}
  \item के समाधान $(E)$ पर $ I $ बिलकुल हैं
        \[
        y(x) = \Bigl(\lambda + \int_{x_0}^{x} b(t)\, \eu^{A(t)} \dd t
        \Bigr)\, \eu^{-A(x)},
        \qquad \lambda \in \R,
        \]
        कहाँ $ x_0 \in I $ तय है. समतुल्य: का सामान्य समाधान
        $(H)$ साथ ही इसका एक विशेष समाधान $(E)$.
  \item हरएक के लिए $ x_0 \in I $ और $ y_0$, द \emph{कॉची
        समस्या}\index{कॉची
        समस्या} ``$(E)$ और $ y(x_0) = y_0$''
        बिल्कुल एक समाधान चालू है $ I $.
\end{enumerate}
\end{theorem}

\begin{proof}
(1) नामित विधि का पालन करना \emph{की भिन्नता
स्थिरांक}\index{की भिन्नता
स्थिरांक}, के समाधान खोजें
प्रपत्र $ y = \mu(x)\, \eu^{-A(x)}$ साथ $\mu $ विभेदनीय --- नहीं
सामान्यता खो गई है, क्योंकि हर कार्य चालू है $ I $ ऐसा लिखा जा सकता है
($\mu = y\,\eu^{A}$). प्रतिस्थापित करना,
\[
y' + ay = \mu' \eu^{-A} - \mu a \eu^{-A} + a\mu\eu^{-A}
= \mu'\, \eu^{-A},
\]
इसलिए $ y $ हल करती है $(E)$ अगर और केवल अगर $\mu'(x) = b(x)\,\eu^{A(x)}$, अगर और केवल अगर $\mu(x) = \lambda + \int_{x_0}^x b(t)\eu^{A(t)}\dd t $ के लिए
कुछ स्थिर $\lambda $ (एक ही निरंतर के दो आदिम
किसी अंतराल पर फ़ंक्शन एक स्थिरांक से भिन्न होता है)।

(2) सूत्र में, $ y(x_0) = \lambda\,\eu^{-A(x_0)}$: स्थिति
$ y(x_0) = y_0$ निर्धारित करता है $\lambda = y_0 \eu^{A(x_0)}$ विशिष्ट रूप से.
\end{proof}

\begin{example}\label{ex:b1:diffeq:order1}
हल करना $ y' + \dfrac{y}{x} = x^2$ पर $ I = \intoo{0}{+\infty}$. यहाँ
$ a(x) = \frac 1x $, $ A(x) = \ln x $, $\eu^{-A(x)} = \frac 1x $.
सजातीय समाधान: $\frac{\lambda}{x}$. स्थिरांकों का \omterm{def:b1:counting:objects}{परिवर्तन}:
$\mu'(x) = x^2 \cdot x = x^3$, इसलिए $\mu = \frac{x^4}{4} + \lambda $, और
\[
y(x) = \frac{x^3}{4} + \frac{\lambda}{x}, \qquad \lambda \in \R .
\]
प्रारंभिक स्थिति के साथ $ y(1) = 0$: $\lambda = -\frac14$.
\emph{जाँच करना:} $ y' + \frac yx = \frac{3x^2}{4} - \frac{\lambda}{x^2} +
\frac{x^2}{4} + \frac{\lambda}{x^2} = x^2$.
\end{example}

\begin{example}[अनुमान लगाने वाली धड़कनें एकीकृत हो रही हैं]\label{ex:b1:diffeq:guess}
हल करना $ y' + 2x\,y = x $ पर $\R $. स्थिरांकों का \omterm{def:b1:counting:objects}{परिवर्तन} काम करता है ($ A =
x^2$, $\mu' = x\,\eu^{x^2}$, $\mu = \frac12\eu^{x^2} + \lambda $),
लेकिन यह देखते हुए कि \emph{स्थिर} $ y_p = \frac12$ हल करता है
समीकरण ($0 + 2x\cdot\frac12 = x $) तेज है. के साथ
सजातीय समाधान $\lambda\,\eu^{-x^2}$:
\[
y(x) = \frac12 + \lambda\,\eu^{-x^2}, \qquad \lambda \in \R .
\]
हर समाधान जुटता है $\frac12$ के रूप में बेहद तेज $ x \to
\pm\infty $: स्थिर विशेष समाधान एक है
\emph{संतुलन} कि सभी समाधान जुड़ें। अंतर्दृष्टि: पहले
सामान्य विधि शुरू करने के बाद, इसकी तलाश में दस सेकंड बिताएं
स्पष्ट विशेष समाधान (स्थिर, एकपदी, का गुणज)।
दाहिनी ओर); संरचना प्रमेय तब कार्य समाप्त करता है।
\end{example}

\begin{remark}[अंतराल मायने रखता है]
प्रमेय एक पर रहता है \emph{अंतराल} कहाँ $ a $ और $ b $ हैं
निरंतर. के लिए $ y' + \frac yx = 0$ पर $\R^*$, समाधान हैं
$\frac{\lambda}{x}$ पर $\intoo{0}{+\infty}$ और
$\frac{\mu}{x}$ पर $\intoo{-\infty}{0}$ साथ \emph{स्वतंत्र}
स्थिरांक: किसी एक सूत्र के एक-दूसरे से चिपके रहने का कोई कारण नहीं है
विलक्षणता पर $0$.
\end{remark}

\begin{example}[एक जटिल दाहिनी ओर, दो वास्तविक उत्तर]\label{ex:b1:diffeq:complexrhs}
हल करना $ y' - y = \cos x $ और $ y' - y = \sin x $ एक झटके में. में काम $\C $ दाहिनी ओर के साथ $\eu^{\iu x}$: कोशिश कर रहा हूँ $ y_p =
c\,\eu^{\iu x}$ देता है $ c(\iu - 1)\eu^{\iu x} = \eu^{\iu x}$, इसलिए
\[
c = \frac1{\iu - 1} = \frac{-1 - \iu}2,
\qquad
y_p = -\frac{(1 + \iu)(\cos x + \iu\sin x)}2
= \frac{\sin x - \cos x}2 + \iu\,\frac{-\sin x - \cos x}2 .
\]
चूँकि समीकरण में वास्तविक गुणांक, वास्तविक और काल्पनिक हैं
भागों का विभाजन: $\frac{\sin x - \cos x}2$ हल करती है $ y' - y = \cos x $,
और $-\frac{\sin x + \cos x}2$ हल करती है $ y' - y = \sin x $ (चेक करें
पहला: व्युत्पन्न $\frac{\cos x + \sin x}2$, माइनस द
कार्य, देता है $\cos x $). एक जटिल रेखा ने दो रनों का स्थान ले लिया
स्थिरांक की भिन्नता --- वही अर्थव्यवस्था जो
\cref{met:b1:diffeq:particular} दूसरे क्रम के लिए व्यवस्थित करता है,
और का आवर्ती लाभांश \cref{ch:b1:complex}.
\end{example}

\section{स्थिर गुणांक वाले दूसरे क्रम के रैखिक समीकरण}

\begin{definition}\label{def:b1:diffeq:linear2}
होने देना $ a, b \in \R $ और $ f \colon I \to \R $ निरंतर। समीकरण
\[
(E)\colon\quad y'' + a\,y' + b\,y = f(x)
\]
एक है \emph{स्थिरांक के साथ दूसरे क्रम का रैखिक समीकरण
गुणांक}\index{स्थिरांक के साथ दूसरे क्रम का रैखिक समीकरण
गुणांक};
$(H)\colon y'' + ay' + by = 0$ इसका सजातीय समीकरण है, और
$\chi(r) = r^2 + ar + b $ इसका \emph{विशेषता
बहुपद}\index{विशेषता
बहुपद}.
\end{definition}

\begin{theorem}[सजातीय समाधान]\label{thm:b1:diffeq:homogeneous2}
होने देना $\Delta = a^2 - 4b $ का विभेदक हो $\chi $. असली
के समाधान $(H)$ पर $\R $ हैं:
\begin{enumerate}
  \item अगर $\Delta > 0$, साथ $ r_1 \neq r_2$ दो वास्तविक जड़ें:
        $\;y = \lambda\, \eu^{r_1 x} + \mu\, \eu^{r_2 x}$;
  \item अगर $\Delta = 0$, साथ $ r_0$ दोहरी जड़:
        $\;y = (\lambda + \mu x)\, \eu^{r_0 x}$;
  \item अगर $\Delta < 0$, जड़ों के साथ $\alpha \pm \iu\omega $
        ($\omega > 0$):
        $\;y = \eu^{\alpha x} (\lambda \cos\omega x + \mu
        \sin\omega x)$;
\end{enumerate}
प्रत्येक मामले में $(\lambda, \mu)$ भागना $\R^2$.
\end{theorem}

\begin{proof}
सबसे पहले उस पर गौर करें $ r \in \C $, $ x \mapsto \eu^{rx}$ हल करती है $(H)$
अगर और केवल अगर $\chi(r) = 0$ (विकल्प: $(r^2 + ar + b)\eu^{rx} =
0$). यही कारण है कि घातांक स्वाभाविक पहला अनुमान है:
विभेदीकरण कार्य करता है $\eu^{rx}$ संख्या से गुणा के रूप में
$ r $, इसलिए अवकल समीकरण संख्यात्मक समीकरण बन जाता है
$\chi(r) = 0$ --- संपूर्ण विश्लेषणात्मक समस्या को संक्षेप में प्रस्तुत किया गया है
एक द्विघात की जड़ें ढूँढना।

मुख्य कदम अज्ञात का \omterm{def:b1:counting:objects}{परिवर्तन} है जो ऑर्डर को कम करता है। होने देना $ r $ हो
की एक (संभवतः जटिल) जड़ $\chi $ और लिखा $ y = z\, \eu^{rx}$,
जो कोई व्यापकता नहीं खोता। तब
\[
y'' + ay' + by
= \bigl(z'' + (2r + a) z' + \chi(r) z\bigr)\eu^{rx}
= \bigl(z'' + (2r + a) z'\bigr)\eu^{rx},
\]
इसलिए $(H)$ बन जाता है \emph{पहले के आदेश} समीकरण $ u' + (2r + a) u = 0$
के लिए $ u = z'$.

\emph{मामला $\Delta \neq 0$:} चुनना $ r = r_1$; तब $2r_1 + a = r_1 -
r_2$ (तब से $ r_1 + r_2 = -a $). द्वारा
\cref{thm:b1:diffeq:homogeneous1}, $ z' = c\,\eu^{(r_2 - r_1)x}$ के लिए
कुछ स्थिर $ c $; पर एकीकरण $\R $, $ z = \mu\, \eu^{(r_2 - r_1)x}
+ \lambda $ साथ $\mu = \frac{c}{r_2 - r_1}$, और इसलिए $ y = z\,
\eu^{r_1 x} = \lambda\,\eu^{r_1x} + \mu\,\eu^{r_2x}$.
कब $\Delta < 0$, जड़ें हैं $\alpha \pm \iu\omega $ और
जटिल समाधान हैं $ y = c_1\eu^{(\alpha+\iu\omega)x} +
c_2\eu^{(\alpha-\iu\omega)x}$ साथ $ c_1, c_2 \in \C $. कौन सा
ये वास्तविक मूल्यवान हैं? तब से $\conj{\eu^{(\alpha+\iu\omega)x}} =
\eu^{(\alpha-\iu\omega)x}$, का संयुग्म $ y $ है $\conj{c_2}\,
\eu^{(\alpha+\iu\omega)x} + \conj{c_1}\,\eu^{(\alpha-\iu\omega)x}$,
और $ y = \conj y $ सभी के लिए $ x $ बल $ c_2 = \conj{c_1}$ (दो
घातांक रैखिक रूप से स्वतंत्र हैं: दो बिंदुओं पर मूल्यांकन करें, या
पर तुलना करें $ x=0$ से विभाजित करने के बाद $\eu^{\alpha x}$). लिखना $ c_1
= \frac{\lambda - \iu\mu}2$ साथ $\lambda, \mu $ असली:
\[
y = 2\,\Re\Bigl(c_1\,\eu^{\alpha x}(\cos\omega x +
\iu\sin\omega x)\Bigr)
= \eu^{\alpha x}\bigl(\lambda\cos\omega x + \mu\sin\omega
x\bigr),
\]
और इसके विपरीत प्रत्येक ऐसा फ़ंक्शन एक समाधान (ए का वास्तविक भाग) है
वास्तविक समीकरण का जटिल समाधान): वास्तविक समाधान स्थान है
जैसा कि घोषणा की गई है।

\emph{मामला $\Delta = 0$:} $ r = r_0$, $2r_0 + a = 0$, इसलिए $ z'' = 0$:
$ z = \lambda + \mu x $ और $ y = (\lambda + \mu x)\eu^{r_0 x}$.
\end{proof}

\begin{example}[एक कॉची समस्या, आरंभ से अंत तक]\label{ex:b1:diffeq:cauchy2}
हल करना $ y'' - 3y' + 2y = 0$ साथ $ y(0) = 0$, $ y'(0) = 1$.
\omterm{def:b1:diffeq:linear2}{विशेषता बहुपद} $ r^2 - 3r + 2 = (r - 1)(r - 2)$ के पास है
असली जड़ें $1$ और $2$: सामान्य समाधान $ y = \lambda\eu^{x} +
\mu\eu^{2x}$. दो स्थितियाँ रैखिक प्रणाली देती हैं
\[
\lambda + \mu = 0, \qquad \lambda + 2\mu = 1 ,
\]
इसलिए $\mu = 1$, $\lambda = -1$:
\[
y(x) = \eu^{2x} - \eu^{x} .
\]
जाँच करना: $ y(0) = 0$; $ y' = 2\eu^{2x} - \eu^x $ है $ y'(0) = 1$; और
$ y'' - 3y' + 2y = (4 - 6 + 2)\eu^{2x} + (-1 + 3 - 2)\eu^x = 0$.
उत्तर के आकार पर ध्यान दें: निकट $-\infty $ धीमा मोड
$-\eu^x $ हावी है; पास में $+\infty $ तेज़ मोड $\eu^{2x}$ करता है।
विभिन्न क्षय के साथ मोड के सुपरपोजिशन के रूप में समाधान पढ़ना
या विकास दर लाभदायक आदत है --- यह कैसे है
सप्ताहांत समस्या का क्षणिक/स्थिर-अवस्था विभाजन व्यवस्थित किया जाता है।
\end{example}

\begin{theorem}[संरचना और कॉची समस्या]\label{thm:b1:diffeq:structure2}
\begin{enumerate}
  \item अगर $ y_p $ का एक विशेष समाधान है $(E)$, के समाधान
        $(E)$ बिलकुल हैं $ y_p + y_h $, $ y_h $ के ऊपर चल रहा है
        के समाधान $(H)$.
  \item (सुपरपोज़िशन) यदि $ y_1$ हल करती है $ y'' + ay' + by = f_1$ और
        $ y_2$ हल करती है $ y'' + ay' + by = f_2$, तब $ y_1 + y_2$ हल करता है
        दाईं ओर वाला समीकरण $ f_1 + f_2$.
  \item सभी के लिए $ x_0 \in I $ और $(y_0, y_0')$, द \omterm{thm:b1:diffeq:voc}{कॉची समस्या}
        ``$(E)$, $ y(x_0) = y_0$, $ y'(x_0) = y_0'$'' बिल्कुल एक है
        समाधान चालू $ I $. \emph{(अस्तित्व ने एक विशेष प्रदान किया
        समाधान; पूर्ण विशिष्टता।)}
\end{enumerate}
\end{theorem}

\begin{proof}
(1) $ y $ हल करती है $(E)$ आईएफएफ $ y - y_p $ हल करती है $(H)$, की रैखिकता द्वारा $ y
\mapsto y'' + ay' + by $. (2) वही रैखिकता है।

(3) (1) द्वारा यह सिद्ध करना पर्याप्त है कि स्थिरांक $(\lambda,
\mu)$ किसी भी डेटा को हमेशा विशिष्ट रूप से समायोजित किया जा सकता है $(y_0, y_0')$.
चर का अनुवाद करते हुए, मान लीजिए $ x_0 = 0$. के मामले (1) में
\cref{thm:b1:diffeq:homogeneous2}, $ y = \lambda\eu^{r_1x} +
\mu\eu^{r_2x}$ देता है
\[
y(0) = \lambda + \mu, \qquad y'(0) = r_1\lambda + r_2\mu :
\]
में एक रेखीय प्रणाली $(\lambda, \mu)$ जिसका निर्धारक है $ r_2 -
r_1 \neq 0$; इसे स्पष्ट रूप से हल करना, $\mu = \frac{y_0' -
r_1y_0}{r_2 - r_1}$ और $\lambda = y_0 - \mu $: बिल्कुल एक
समाधान. मामले में (2), $ y(0) = \lambda $ और $ y'(0) = r_0\lambda +
\mu $: प्रणाली निर्धारक के साथ त्रिकोणीय है $1$, द्वारा हल किया गया
$\lambda = y_0$, $\mu = y_0' - r_0y_0$. मामले में (3), $ y(0) =
\lambda $ और $ y'(0) = \alpha\lambda + \omega\mu $: निर्धारक
$\omega \neq 0$, द्वारा हल किया गया $\lambda = y_0$, $\mu = \frac{y_0' -
\alpha y_0}\omega $. प्रत्येक मामले में \omterm{def:b1:logic:map}{नक्शा} $(\lambda, \mu) \mapsto
(y(x_0), y'(x_0))$ एक रेखीय आक्षेप है --- की भाषा
\cref{ch:b1:linmaps} इस केस चेक को एक में संपीड़ित कर देगा
वाक्य.
\end{proof}

\begin{method}[के लिए विशेष समाधान $ f(x) = P(x)\,\eu^{\gamma x}$]\label{met:b1:diffeq:particular}
जब दाहिनी ओर है $ P(x)\,\eu^{\gamma x}$ साथ $ P $ ए
बहुपद और $\gamma \in \R $ (इसमें बहुपद, घातांक शामिल हैं,
और जटिल के माध्यम से $\gamma $ या सुपरपोज़िशन, $\cos $ और $\sin $): देखो
प्रपत्र के किसी विशेष समाधान के लिए
\[
y_p(x) = x^{m}\, Q(x)\, \eu^{\gamma x},
\qquad
m = \text{multiplicity of } \gamma \text{ as a root of } \chi
\ (m = 0, 1 \text{ or } 2),
\]
साथ $ Q $ के समान डिग्री का एक बहुपद $ P $, जिसका गुणांक
प्रतिस्थापन और पहचान द्वारा पाए जाते हैं। के लिए $ f = K\cos\omega x $
(या $\sin $), दाईं ओर से हल करें $ K\eu^{\iu\omega x}$ और ले लो
वास्तविक (सम्मान)\ काल्पनिक) भाग.
\end{method}

\begin{example}[क्रिया में सुपरपोजिशन]\label{ex:b1:diffeq:superposition}
हल करना $ y'' - y = \eu^{x} + 4$ पर $\R $. सजातीय: $\chi(r) = r^2
- 1$, जड़ें $\pm1$, इसलिए $ y_h = \lambda\eu^x + \mu\eu^{-x}$. विभाजन
दाहिनी ओर और विधि बॉक्स द्वारा प्रत्येक टुकड़े का उपचार करें।
\emph{टुकड़ा $\eu^x $:} यहाँ $\gamma = 1$ की एक सरल जड़ है
$\chi $, तो प्रयास करें $ y_1 = c\,x\,\eu^x $: तब $ y_1'' - y_1 = c(x +
2)\eu^x - cx\eu^x = 2c\,\eu^x $, देना $ c = \frac12$.
\emph{टुकड़ा $4$:} $\gamma = 0$ जड़ नहीं है; स्थिरांक $ y_2 =
-4$ काम करता है. सुपरपोजिशन द्वारा
(\cref{thm:b1:diffeq:structure2}~(2)):
\[
y = \frac{x\,\eu^x}2 - 4 + \lambda\,\eu^x + \mu\,\eu^{-x},
\qquad (\lambda, \mu) \in \R^2 .
\]
ध्यान दें कि दोनों टुकड़ों की मांग कैसी थी \emph{अलग} आकृतियाँ ($ m = 1$
बनाम $ m = 0$): बहुलता परीक्षण प्रत्येक पर लागू होता है
घातांक अलग से, जो विभाजित करने का संपूर्ण बिंदु है
अनुमान लगाने से पहले दाईं ओर।
\end{example}

\begin{example}[काम पर बहुलता नियम]\label{ex:b1:diffeq:multiplicity}
हल करना $ y'' + y' = x $ पर $\R $. दायीं ओर है $ P(x)\eu^{0
\cdot x}$ साथ $ P(x) = x $, और $\gamma = 0$ एक है \emph{सरल} जड़
का $\chi(r) = r^2 + r = r(r + 1)$: इसलिए $ m = 1$, और सही
अनुमान है $ y_p = x\,(\alpha x + \beta) = \alpha x^2 + \beta x $, एक
से अधिक डिग्री $ P $. प्रतिस्थापन:
\[
y_p'' + y_p' = 2\alpha + (2\alpha x + \beta)
= 2\alpha x + (2\alpha + \beta) ,
\]
और पहचान के साथ $ x $ देता है $\alpha = \frac12$, $\beta = -1$:
$ y_p = \frac{x^2}2 - x $. सामान्य समाधान: $ y = \frac{x^2}2 - x +
\lambda + \mu\,\eu^{-x}$. क्या हमने अनुमान लगाया था $ y_p = \alpha x + \beta $
(बहुलता को नजरअंदाज करते हुए), प्रतिस्थापन देगा $ y_p'' + y_p'
= \alpha $, एक स्थिरांक --- कोई विकल्प नहीं $\alpha, \beta $ मेल कर सकते हैं
$ x $, और विफलता संरचनात्मक है: स्थिरांक पहले से ही हल करते हैं
सजातीय समीकरण, इसलिए वे बाईं ओर अदृश्य हैं।
कारक $ x^m $ सजातीय से बाहर निकलने के लिए सटीक रूप से मौजूद है
समाधान स्थान.
\end{example}

\begin{remark}[बत्तीसवीं बीमा पॉलिसी]\label{rem:b1:diffeq:check}
इस अध्याय में प्रत्येक हल किया गया समीकरण एक प्रतिस्थापन के साथ समाप्त होता है
जांचें, और यह सजावटी नहीं है। एक अंतर-समीकरण
संगणना शृंखला में कई छोटे चरण होते हैं (एक आदिम, एक उत्पाद नियम,
दो स्थिरांक), और एक एकल संकेत त्रुटि अदृश्य रूप से फैलती है;
अंतिम सूत्र को वापस समीकरण कैच में प्रतिस्थापित करना
अनिवार्य रूप से उनमें से सभी एक भेदभाव की कीमत पर।
तीन परतों में प्रतिबिम्ब विकसित करें: जाँच करें \emph{विशेष
समाधान} अकेले (सजातीय भाग वैसे भी रद्द हो जाता है), जाँच करें
\emph{प्रारंभिक शर्तें} पूर्ण समाधान पर, और जब ए
पैरामीटर मौजूद है, जांचें a \emph{पतित मूल्य} (करता है
सामान्य के लिए सूत्र $\Omega $ ज्ञात उत्तर को पुन: प्रस्तुत करें
$\Omega = 0$?). आदत में आधा मिनट लगता है; यह परिवर्तित हो जाता है
``शायद सही'' को ``सत्यापित'' में बदलें।
\end{remark}

\begin{remark}[सामान्य ख़तरे]\label{rem:b1:diffeq:pitfalls}
\begin{enumerate}
  \item \emph{पहले सामान्यीकरण करें.} सूत्र समीकरण मानते हैं
        पढ़ता है $ y' + a(x)y = b(x)$ --- गुणांक $1$ पर $ y'$.
        के लिए $ xy' - 2y = x^3$, से विभाजित करें $ x $ (अंतराल पर
        टालना $0$) पहचानने से पहले $ a $ और $ b $, के रूप में
        \cref{exo:b1:diffeq:2}.
  \item \emph{प्रति आयाम एक स्थिरांक, अंत में निश्चित।} द
        सामान्य प्रथम-क्रम समाधान में एक स्थिरांक होता है
        दूसरे क्रम का एक दो; प्रारंभिक शर्तें लगाई गई हैं
        द \emph{पूरा} समाधान $ y_p + y_h $, कभी चालू नहीं $ y_h $
        अकेले --- जोड़ने से पहले उन्हें थोपना $ y_p $ सबसे ज्यादा है
        बार-बार संरचनात्मक त्रुटि.
  \item \emph{बहुलता का ध्यान रखें.} एक विशेष-समाधान अनुमान
        जो सजातीय समीकरण को हल करता है वह अदृश्य है
        बाएँ हाथ की ओर; कारक $ x^m $ का
        \cref{met:b1:diffeq:particular} वैकल्पिक नहीं है
        (\cref{ex:b1:diffeq:multiplicity}).
  \item \emph{अंतराल उत्तर का हिस्सा हैं.} समाधान रहते हैं
        अंतराल पर जहां गुणांक निरंतर हैं;
        एक विलक्षणता को जोड़ने से नकली स्थिरांक बन सकते हैं
        (\cref{exo:b1:diffeq:12}) या विशिष्टता को नष्ट कर दें। ``समाधान
        पर $\R^*$'' का अर्थ है दो स्वतंत्र समस्याएं।
\end{enumerate}
\end{remark}

\begin{example}[ऑफ-रेजोनेंस फोर्सिंग]\label{ex:b1:diffeq:offresonance}
हल करना $ y'' + 4y = \sin x $ पर $\R $. प्राकृतिक आवृत्ति $2$, मजबूर करना
आवृत्ति $1$: तब से $\iu $ है \emph{नहीं} की एक जड़ $\chi(r) = r^2
+ 4$, बहुलता है $ m = 0$ और एक सादा साइनसॉइड पर्याप्त है।
कोशिश कर रहा हूँ $ y_p = \alpha\sin x $ (कोई कोज्या आवश्यक नहीं: समीकरण है
नहीं $ y'$ अवधि, और $\sin $ पुन: बनाता है $\sin $):
\[
y_p'' + 4y_p = -\alpha\sin x + 4\alpha\sin x = 3\alpha\sin x ,
\]
इसलिए $\alpha = \frac13$ और सामान्य समाधान है
\[
y = \frac{\sin x}3 + \lambda\cos 2x + \mu\sin 2x .
\]
प्रत्येक समाधान परिबद्ध रहता है: दो दोलनों का एक सुपरपोजिशन
आवृत्तियों पर $1$ (मजबूर) और $2$ (प्राकृतिक)। से तुलना करें
अगला उदाहरण, जहां जबरदस्ती \emph{पर} प्राकृतिक आवृत्ति
उत्तर का स्वरूप ही बदल देता है।
\end{example}

\begin{example}[एक मजबूर दोलन]\label{ex:b1:diffeq:oscillation}
हल करना $ y'' + y = \cos x $, $ y(0) = 0$, $ y'(0) = 0$.

\emph{सजातीय:} $\chi(r) = r^2 + 1$, जड़ें $\pm\iu $: $ y_h =
\lambda\cos x + \mu \sin x $.

\emph{विशिष्ट:} दाहिनी ओर $\Re(\eu^{\iu x})$ साथ $\gamma =
\iu $ की एक सरल जड़ $\chi $: कोशिश $ z_p = c\, x\, \eu^{\iu x}$
($ c \in \C $). तब $ z_p'' + z_p = c\,(2\iu)\eu^{\iu x}$, जो बराबर है
$\eu^{\iu x}$ के लिए $ c = \frac{1}{2\iu} = -\frac\iu2$. इसलिए $ z_p =
-\frac{\iu}{2} x (\cos x + \iu \sin x)$ और $ y_p = \Re(z_p) =
\frac{x \sin x}{2}$.

\emph{सामान्य समाधान:} $ y = \frac{x\sin x}{2} + \lambda\cos x +
\mu\sin x $. स्थितियाँ: $ y(0) = \lambda = 0$; $ y' = \frac{\sin x +
x\cos x}{2} + \mu\cos x $, इसलिए $ y'(0) = \mu = 0$. उत्तर: $ y =
\frac{x\sin x}{2}$ --- एक दोलन जिसका आयाम रैखिक रूप से बढ़ता है:
द \emph{गूंज}\index{गूंज} जबरदस्ती के कारण होने वाली घटना
सिस्टम अपनी प्राकृतिक आवृत्ति पर।
\end{example}

\begin{remark}[अंतराल: रैखिकता एक ज्यामिति है]\label{rem:b1:diffeq:interlude}
प्रत्येक समाधान के आकार को पीछे मुड़कर देखें \omterm{def:b1:logic:sets}{तय करना} इस अध्याय में: ए
विशेष समाधान और एक के साथ सजातीय समाधान का स्थान
मुक्त स्थिरांक (प्रथम क्रम) या दो (द्वितीय क्रम)। पर अध्याय
रैखिक बीजगणित (\cref{ch:b1:vspaces,ch:b1:findim,ch:b1:linmaps})
सटीक आपूर्ति करेगा
शब्दावली: द \omterm{def:b1:logic:map}{नक्शा} $ L(y) = y'' + ay' + by $ है \emph{रेखीय}, इसका
सजातीय समाधान बनाते हैं \emph{कर्नेल} का $ L $, एक वेक्टर
अंतरिक्ष जिसका \emph{आयाम} समीकरण के क्रम के बराबर है ---
वह ``प्रति ऑर्डर एक स्थिरांक'' की ईमानदार सामग्री है --- और
समाधान \omterm{def:b1:logic:sets}{तय करना} का $ L(y) = f $ एक \emph{एफ़िन सबस्पेस}, ए
कर्नेल का अनुवाद. यहां तक कि कॉची भी \omterm{def:b1:logic:map}{नक्शा} $(\lambda, \mu)
\mapsto (y(x_0), y'(x_0))$ का \cref{thm:b1:diffeq:structure2} एक है
दो तलों के बीच रैखिक आक्षेप, अर्थात्\ एक उलटा $2
\times 2$ प्रणाली (\cref{ch:b1:matrices}). इस अध्याय में कुछ भी नहीं
इसे फिर से करने की आवश्यकता होगी --- इसे केवल नाम बदलने की आवश्यकता होगी, और
रैखिक बीजगणित के लिए नाम बदलना सर्वोत्तम संभव वार्म-अप है:
वहां की प्रत्येक अमूर्त परिभाषा पहले से ही अपना अस्तित्व अर्जित कर चुकी है
यहाँ.
\end{remark}

\begin{remark}[जहाँ इस अध्याय का प्रयोग किया जाता है]\label{rem:b1:diffeq:whereused}
संरचना प्रमेय --- का समाधान $(E)$ फॉर्म ``एक विशेष
समाधान प्लस समाधान $(H)$'' --- पहली उपस्थिति है
एक पैटर्न का कि \cref{ch:b1:vspaces,ch:b1:linmaps} नाम होगा: द
समाधान \omterm{def:b1:logic:sets}{तय करना} का $(H)$ है \emph{कर्नेल} रैखिक का \omterm{def:b1:logic:map}{नक्शा} $ y
\mapsto y'' + ay' + by $, और समाधान \omterm{def:b1:logic:sets}{तय करना} का $(E)$ एक है
इसका अनुवाद करें। \omterm{def:b1:diffeq:linear2}{विशेषता बहुपद} के रूप में पुनः प्रकट होता है
द \omterm{def:b1:diffeq:linear2}{विशेषता बहुपद} में एक मैट्रिक्स का \cref{ch:b1:matrices}:
दूसरे क्रम का समीकरण एक है $2 \times 2$ प्रथम-क्रम प्रणाली में
छद्मवेश, एक दृष्टिकोण जो वर्ष 2 खंड को व्यवस्थित करता है। द
इंटीग्रल द्वारा मांग की गई स्थिरांक की भिन्नता द्वारा आपूर्ति की जाती है
\cref{ch:b1:integration}, और नीचे सप्ताहांत समस्या ---
संचालित डैम्प्ड ऑसिलेटर --- प्रत्येक के लिए मॉडल केस है
विज्ञान में सर्किट से निलंबन तक दोलन प्रश्न
पुल.
\end{remark}

\begin{exercise}[$\star $]\label{exo:b1:diffeq:1}
पर समाधान करें $\R $: $\;y' + 2y = \eu^{3x}$; फिर \omterm{thm:b1:diffeq:voc}{कॉची समस्या} $ y(0) =
1$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:diffeq:2}

पहले सामान्यीकृत रूप में समीकरण)}.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:diffeq:3}
पर समाधान करें $\R $, वास्तविक सामान्य समाधान दे रहा है:
$\;y'' - 3y' + 2y = 0$; $\;y'' + 4y' + 4y = 0$; $\;y'' - 2y' + 5y =
0$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:diffeq:4}
हल करना $ y'' - y = x^2$ पर $\R $, फिर \omterm{thm:b1:diffeq:voc}{कॉची समस्या} $ y(0) = 0$,
$ y'(0) = 1$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:diffeq:5}
पर समाधान करें $\intoo{-\frac\pi2}{\frac\pi2}$:
$\;y' + y\tan x = \sin 2x $.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:diffeq:6}
हल करना $ y'' - 4y' + 3y = (2x + 1)\,\eu^{x}$ पर $\R $. \emph{(ध्यान दें
बहुलता: है $1$ की एक जड़ \omterm{def:b1:diffeq:linear2}{विशेषता बहुपद}?)}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:diffeq:7}
हल करना $ y'' + 4y = \sin 2x + x $ पर $\R $ \emph{(सुपरपोज़िशन; इलाज

\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:diffeq:8}
तापमान पर एक कप कॉफ़ी $ T_0 = 80\,^\circ $C एक कमरे में बैठता है
$20\,^\circ $ C. न्यूटन का शीतलन नियम बताता है $T' = -k\,(T - 20)$ साथ
$ k > 0$. के लिए हल करें $ T(t)$, और यह देखते हुए कि कॉफ़ी चालू है
$50\,^\circ $ सी के बाद $10$ मिनट, पता लगाएं कि यह कब पहुंचता है
$25\,^\circ $ सी।
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b1:diffeq:9}
(डम्प्ड ऑसिलेटर) के लिए $\varepsilon \geq 0$, विचार करना
$ y'' + 2\varepsilon y' + y = 0$.
\begin{enumerate}
  \item के लिए हल करें $\varepsilon \in \intco{0}{1}$, $\varepsilon = 1$,
        और $\varepsilon > 1$.
  \item उसके लिए दिखाओ $\varepsilon > 0$ हर समाधान की प्रवृत्ति होती है $0$
        पर $+\infty $, और वह के लिए $\varepsilon = 0$ शून्येतर
        समाधान नहीं है.
  \item के लिए $\varepsilon \in \intoo{0}{1}$, दिखाएँ कि a के शून्य
        शून्येतर विलयन को अंतराल के साथ नियमित रूप से स्थान दिया जाता है
        $\frac{\pi}{\sqrt{1 - \varepsilon^2}}$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b1:diffeq:10}
सभी फ़ंक्शन ढूंढें $ f \colon \R \to \R $, दो बार भिन्न, जैसे
वह
\[
\forall x, y \in \R, \qquad f(x + y) + f(x - y) = 2 f(x) f(y),
\]
साथ $ f(0) \ne 0$ और $ f $ स्थिर नहीं.
\emph{संकेत: ठीक करें $ y $, के संबंध में दो बार अंतर करें $ x $ पर $0$;
दिखाओ $ f(0) = 1$ और $ f'' = c f $ कुछ स्थिरांक के लिए $ c $; फिर हल करें
के संकेत के अनुसार $ c $ और जांचें कि कौन से समाधान संतुष्ट करते हैं
कार्यात्मक समीकरण.}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:diffeq:11}
(यूलर समीकरण) हल करें $ x^2 y'' - x y' + y = 0$ पर
$\intoo{0}{+\infty}$. \emph{संकेत: सेट $ z(t) = y(\eu^t)$, अर्थात\
स्थानापन्न $ x = \eu^t $, और दिखाओ $ z $ एक रैखिक को संतुष्ट करता है
निरंतर गुणांक के साथ समीकरण.}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b1:diffeq:12}
समीकरण पर विचार करें $ x\,y' = 2y $ पूरी वास्तविक लाइन पर, में
अज्ञात अवकलनीय फलन $ y \colon \R \to \R $.
\begin{enumerate}
  \item पर समाधान करें $\intoo{0}{+\infty}$ और पर $\intoo{-\infty}{0}$.
  \item उसके लिए दिखाओ \emph{कोई} स्थिरांक $ a, b \in \R $, द
        के बराबर कार्य करें $ ax^2$ के लिए $ x \geq 0$ और को $ bx^2$ के लिए
        $ x < 0$ पर भिन्न है $\R $ और समीकरण को हल करता है
        हर जगह.
  \item निष्कर्ष निकालें कि समाधान \omterm{def:b1:logic:sets}{तय करना} पर $\R $ एक दो-पैरामीटर है
        परिवार, और बताएं कि यह इसका खंडन क्यों नहीं करता है
        में विशिष्टता \cref{thm:b1:diffeq:voc}.
\end{enumerate}
\end{exercise}

\section{समस्या: चालित नम थरथरानवाला}

\begin{problem}\label{pb:b1:diffeq:1}
एक समीकरण एक चिपचिपे माध्यम में एक स्प्रिंग पर द्रव्यमान को नियंत्रित करता है
आरएलसी सर्किट में चार्ज, और हवा में हिलती हुई इमारत:
\[
(E_\Omega)\colon\quad
x'' + 2\lambda x' + \omega_0^2\,x = A\cos(\Omega t),
\]
साथ $\lambda \geq 0$ भीगना, $\omega_0 > 0$ प्राकृतिक
आवृत्ति, और $ A > 0$, $\Omega > 0$ का आयाम और आवृत्ति
जबरदस्ती. यह समस्या अपना संपूर्ण व्यवहार निकालती है: क्षय
क्षणभंगुर की, अद्वितीय आवधिक स्थिर स्थिति, \omterm{ex:b1:diffeq:oscillation}{गूंज}
वक्र और इसकी तीक्ष्णता ( \emph{\omterm{pb:b1:diffeq:1}{गुणवत्ता कारक}}), की धड़कन
अविभाजित केस, और ऊर्जा संतुलन जो इसे कायम रखता है
दोलन.\index{गूंज}\index{थरथरानवाला}
जब तक अन्यथा कहा न जाए, $0 < \lambda < \omega_0$ (अंडरडैम्प्ड
शासन) और हम लिखते हैं $\omega_d = \sqrt{\omega_0^2 - \lambda^2}$.

\medskip
\noindent\textbf{Part I --- The free oscillator.} यहाँ $ A = 0$.
\begin{enumerate}
  \item सजातीय समीकरण को हल करें $(H)$ के लिए $0 < \lambda <
        \omega_0$, और के लिए $\lambda = 0$. (शासन $\lambda
        \geq \omega_0$ में इलाज किया गया \cref{exo:b1:diffeq:9};
        उन्हें उद्धृत करें.)
  \item इसे हर एक के लिए दिखाएँ $\lambda > 0$, के सभी समाधान $(H)$
        प्रवृत्त $0$ पर $+\infty $ --- तीनों शासनों में।
  \item ऊर्जा को परिभाषित करें $\mathcal E(t) = \frac12 x'(t)^2 +
        \frac12\omega_0^2\,x(t)^2$ के समाधान के साथ $(H)$. दिखाओ
        $\mathcal E'(t) = -2\lambda\,x'(t)^2 \leq 0$, और निष्कर्ष निकालें
        (बिना कुछ सुलझाये) कि \omterm{thm:b1:diffeq:voc}{कॉची समस्या}
        ``$(H)$, $ x(t_0) = x'(t_0) = 0$'' में केवल शून्य है
        समाधान, हर एक के लिए $\lambda \geq 0$.
  \item के लिए $0 < \lambda < \omega_0$, अशून्य समाधान को इस प्रकार लिखें
        $ x(t) = R\,\eu^{-\lambda t}\cos(\omega_d t - \varphi)$ और
        चलो $ T_d = \frac{2\pi}{\omega_d}$ छद्म काल हो.
        बताते हैं कि $ x(t + T_d) = \eu^{-\lambda T_d}\,x(t)$: प्रत्येक
        स्विंग स्थिर कारक द्वारा सिकुड़ा हुआ पिछला वाला है
        $\eu^{-\delta}$, $\delta = \frac{2\pi\lambda}{\omega_d}$
        (द \emph{लघुगणकीय कमी}). गणना करना $\delta $ के लिए
        $\omega_0 = 1$, $\lambda = 0.1$.
  \item को परिभाषित करो \emph{\omterm{pb:b1:diffeq:1}{गुणवत्ता कारक}} $ Q =
        \dfrac{\omega_0}{2\lambda}$\index{गुणवत्ता कारक}. दिखाओ
        वह समय के बाद $\frac1\lambda $ (एक आयाम
        $\eu $-फोल्डिंग), ऑसिलेटर पूरा हो गया है
        $\frac{\omega_d}{2\pi\lambda}$ छद्म काल, जिसके लिए
        कमजोर भिगोना ($\lambda \ll \omega_0$) लगभग है
        $\frac Q\pi $: द \omterm{pb:b1:diffeq:1}{गुणवत्ता कारक} तक गिना जाता है $\pi $, द
        आयाम के क्षय होने से पहले दोलन जीवित रहे
        $\eu $.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The steady state.} अब $ A > 0$ और
$\lambda > 0$.
\begin{enumerate}[resume]
  \item वास्तविक भाग के रूप में किसी विशेष समाधान की तलाश करें
        $ z\,\eu^{\iu\Omega t}$ साथ $ z \in \C $
        (\cref{met:b1:diffeq:particular}). दिखाएँ कि यह काम करता है
        के साथ
        \[
        z = \frac{A}{\omega_0^2 - \Omega^2 + 2\iu\lambda\Omega} .
        \]
  \item आयाम-चरण रूप में स्थिर अवस्था का पता लगाएं:
        $ x_p(t) = R(\Omega)\cos\bigl(\Omega t -
        \varphi(\Omega)\bigr)$ साथ
        \[
        R(\Omega) = \frac{A}{\sqrt{(\omega_0^2 - \Omega^2)^2 +
        4\lambda^2\Omega^2}},
        \qquad
        \tan\varphi = \frac{2\lambda\Omega}{\omega_0^2 -
        \Omega^2},
        \quad \varphi \in \intoo0\pi .
        \]
  \item दो चरम व्यवस्थाओं की व्याख्या करें: की सीमाओं की गणना करें
        $ R $ और $\varphi $ जैसा $\Omega \to 0^+$ (अर्ध-स्थैतिक
        प्रतिक्रिया $ A/\omega_0^2$, चरण $0$) और के रूप में $\Omega \to
        +\infty $ ($ R \sim A/\Omega^2 \to 0$, चरण $\to \pi $: द
        द्रव्यमान बहुत तेज़ बल के विपरीत चलता है)।
  \item बताते हैं कि \emph{प्रत्येक} का समाधान $(E_\Omega)$ है $ x_p $
        प्लस का एक समाधान $(H)$, इसलिए स्थिर में परिवर्तित हो जाता है
        राज्य $ x_p $ जैसा $ t \to +\infty $, प्रारंभिक जो भी हो
        स्थितियाँ: क्षणिक मृत्यु के बाद, थरथरानवाला के पास है
        इसकी शुरुआत कैसे हुई इसकी कोई याद नहीं.
  \item बताते हैं कि $ x_p $ है \emph{केवल} का आवधिक समाधान
        $(E_\Omega)$.
  \item एक काम करो \omterm{thm:b1:diffeq:voc}{कॉची समस्या} अंत तक: के लिए $ x'' + 2x' + 2x =
        \cos t $ साथ $ x(0) = x'(0) = 0$, दिखाओ कि समाधान है
        \[
        x(t) = \frac{\cos t + 2\sin t}5
        - \eu^{-t}\,\frac{\cos t + 3\sin t}5 ,
        \]
        और क्षणिक और स्थिर भागों की पहचान करें।
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The \omterm{ex:b1:diffeq:oscillation}{गूंज} curve.} का अध्ययन
$\Omega \mapsto R(\Omega)$ पर $\intoo0{+\infty}$.
\begin{enumerate}[resume]
  \item सेटिंग $ u = \Omega^2$ और $ g(u) = (\omega_0^2 - u)^2 +
        4\lambda^2 u $, दिखाएँ: यदि $2\lambda^2 < \omega_0^2$, तब $ R $
        पर एक सख्त अधिकतम प्राप्त करता है \emph{\omterm{ex:b1:diffeq:oscillation}{गूंज} आवृत्ति}
        $\Omega_r = \sqrt{\omega_0^2 - 2\lambda^2}$, साथ
        \[
        R_{\max} = R(\Omega_r)
        = \frac{A}{2\lambda\sqrt{\omega_0^2 - \lambda^2}} .
        \]
  \item बताते हैं कि $\dfrac{R_{\max}}{R(0)} = Q\,\bigl(1 -
        \tfrac{\lambda^2}{\omega_0^2}\bigr)^{-1/2} \geq Q $: पर
        \omterm{ex:b1:diffeq:oscillation}{गूंज}, बल को (अनिवार्य रूप से) द्वारा बढ़ाया जाता है
        \omterm{pb:b1:diffeq:1}{गुणवत्ता कारक}.
  \item सिद्ध करें कि \emph{वेग} आयाम $ V(\Omega) =
        \Omega\,R(\Omega)$ बिल्कुल अधिकतम पर है $\Omega =
        \omega_0$ (नहीं $\Omega_r $), और वह चरण वहां है
        $\varphi(\omega_0) = \frac\pi2$: पर $\Omega = \omega_0$
        वेग बिल्कुल बल के चरण में है।
  \item (बैंडविड्थ) हल करें $ g(u) = 2\,g(u_r)$ बिलकुल, कहाँ $ u_r =
        \omega_0^2 - 2\lambda^2$, और यह निष्कर्ष निकालें कि दोनों
        आवृत्तियाँ $\Omega_\pm $ कहाँ $ R = R_{\max}/\sqrt2$
        संतुष्ट $\Omega_+^2 - \Omega_-^2 =
        4\lambda\sqrt{\omega_0^2 - \lambda^2}$; इसके लिए निष्कर्ष निकालें
        बैंडविड्थ कमजोर भिगोना है $\Omega_+ - \Omega_-
        \approx 2\lambda $, यानी\ $ Q \approx
        \frac{\omega_0}{\Omega_+ - \Omega_-}$: तीखा \omterm{ex:b1:diffeq:oscillation}{गूंज}
        शिखर ऊंचे हैं-$ Q $ सिस्टम.
  \item के लिए संख्यात्मक चित्र $\omega_0 = 1$, $\lambda = 0.05$
        ($ Q = 10$), $ A = 1$: गणना करें $\Omega_r $, $ R_{\max}$, द
        स्थैतिक प्रतिक्रिया $ R(0)$, और अनुमानित बैंडविड्थ।
  \item दिखाओ अगर $2\lambda^2 \geq \omega_0^2$, तब $ R $ है
        सख्ती से कम हो रहा है $\intoo0{+\infty}$: अत्यधिक नमीयुक्त
        सिस्टम में कोई नहीं है \omterm{ex:b1:diffeq:oscillation}{गूंज} बिल्कुल चरम पर.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- No damping: beats and \omterm{ex:b1:diffeq:oscillation}{गूंज}.}
यहाँ $\lambda = 0$.
\begin{enumerate}[resume]
  \item के लिए $\Omega \neq \omega_0$, का सामान्य समाधान खोजें
        $ x'' + \omega_0^2 x = A\cos(\Omega t)$.
  \item हल करें \omterm{thm:b1:diffeq:voc}{कॉची समस्या} $ x(0) = x'(0) = 0$ और \omterm{def:b1:counting:objects}{परिवर्तन}
        उत्पाद प्रपत्र में उत्तर
        \[
        x(t) = \frac{2A}{\omega_0^2 - \Omega^2}\,
        \sin\Bigl(\frac{(\omega_0 - \Omega)t}2\Bigr)
        \sin\Bigl(\frac{(\omega_0 + \Omega)t}2\Bigr) .
        \]
  \item के लिए $\Omega $ के करीब $\omega_0$, उत्पाद को इस प्रकार पढ़ें
        आवृत्ति पर तीव्र दोलन $\frac{\omega_0 + \Omega}2$
        आवृत्ति पर धीमे लिफाफे द्वारा संशोधित
        $\frac{\abs{\omega_0 - \Omega}}2$: द \emph{धड़कता है}\index{धड़कता है}. लिफाफे की अवधि बतायें तथा
        अधिकतम आयाम, और ध्यान दें कि दोनों कैसे उड़ते हैं $\Omega
        \to \omega_0$.
  \item हल करना $ t $ और चलो $\Omega \to \omega_0$ प्रश्न 19 में
        सूत्र: दिखाओ सीमा है
        \[
        x_\infty(t) = \frac{A\,t\,\sin(\omega_0 t)}{2\omega_0} ,
        \]
        और उसे सीधे जांचें $ x_\infty $ गुंजायमान को हल करता है
        समीकरण $ x'' + \omega_0^2 x = A\cos(\omega_0 t)$ साथ
        $ x(0) = x'(0) = 0$ (तुलना करना
        \cref{ex:b1:diffeq:oscillation}): \omterm{ex:b1:diffeq:oscillation}{गूंज} हद है
        कभी धीमी, कभी बड़ी धड़कनों की।
  \item के दो भाग्यों की तुलना करें \omterm{ex:b1:diffeq:oscillation}{गूंज}:रेखीय वृद्धि
        $\frac{At}{2\omega_0}$ बिना नमी के, बनाम संतृप्ति
        पर $ R_{\max} \approx Q\,\frac{A}{\omega_0^2}$ कमजोर के साथ
        भीगना। एक वाक्य में: भौतिक तंत्र क्या परिवर्तित करता है
        पहले से दूसरे में?
\end{enumerate}
