\chapter{दो चरों के कार्य}\label{ch:b1:multivar}

वर्ष का अंत उच्च आयाम में पहली बार चलने के साथ होता है: कार्य
$f(x, y)$ दो वास्तविक चरों में से। हर चीज़ का सामान्यीकरण होता है --- सीमाएँ,
निरंतरता, डेरिवेटिव, चरम --- लेकिन प्रत्येक धारणा एक मोड़ प्राप्त करती है:
हर दिशा में एक साथ \omterm{def:b1:topology:closure}{सीमा} तक पहुंचा जा सकता है, डेरिवेटिव
आंशिक भागों में विभाजित, और \omterm{def:b1:multivar:partial}{ग्रेडियेंट} ऊपर की ओर रास्ता बताता है. द
पूर्ण सिद्धांत (अंतर, सामान्य $\R^n $, सबमैनिफोल्ड्स) का है
दूसरा वर्ष; यहां हमने शब्दावली और पहला ईमानदार सेट किया है
प्रमेय

\section{विमान $\R^2$; निरंतरता}

\begin{definition}\label{def:b1:multivar:topology}
पर $\R^2$, यूक्लिडियन मानदंड का उपयोग करें $\norm{(x,y)} = \sqrt{x^2 + y^2}$
(\cref{ch:b1:euclid}). \omterm{def:b1:topology:open}{खुला} गेंदें, पड़ोस, \emph{\omterm{def:b1:topology:open}{खुला}
सबसेट} का $\R^2$ बिल्कुल वैसे ही परिभाषित हैं जैसे कि
\cref{ch:b1:topology}, के स्थान पर गेंदों के साथ \omterm{prop:b1:reals:intervals}{अंतराल}. एक समारोह
$ f \colon U \to \R $ ($ U \subseteq \R^2$ \omterm{def:b1:topology:open}{खुला}) है \emph{निरंतर
पर} $ a \in U $ कब
\[
\forall\varepsilon > 0,\ \exists\delta > 0, \quad
\norm{X - a} \leq \delta \implies \abs{f(X) - f(a)} \leq
\varepsilon,
\]
एक चर के \omterm{thm:b1:matrices:conjugation}{समान} अनुक्रमिक लक्षण वर्णन के साथ।
योग, उत्पाद, भागफल और रचनाएँ निरंतर
\omterm{def:b1:counting:objects}{एक-परिवर्तन}ीय फ़ंक्शन संरक्षित रहते हैं निरंतरता; समन्वय एमएपीएस हैं
निरंतर, इसलिए ऐसे हैं बहुआयामी पद में $(x,y)$.
\end{definition}

\begin{example}[ध्रुवीय सीमा, सीमा साबित करने का साफ़ तरीका]
\label{ex:b1:multivar:polarbound}
बताते हैं कि $ f(x, y) = \dfrac{x^2y^2}{x^2 + y^2}$ (साथ $ f(0,0) =
0$) है निरंतर मूल पर. ध्रुवीय में \omterm{prop:b1:vspaces:coordinates}{COORDINATES} $ x =
\rho\cos\theta $, $ y = \rho\sin\theta $:
\[
\abs{f} = \frac{\rho^4\cos^2\theta\sin^2\theta}{\rho^2}
= \rho^2\,(\cos\theta\sin\theta)^2 \leq \frac{\rho^2}{4}
\xrightarrow[\rho \to 0]{} 0 ,
\]
एक बाध्य \emph{से स्वतंत्र $\theta $}: दिशा कोई भी हो
दृष्टिकोण के अनुसार, मूल्यों को निचोड़ा जाता है $0$. वह एकरूपता
$\theta $ संपूर्ण बिंदु है --- एक बंधा हुआ जैसा $\abs g =
\abs{\cos\theta\sin\theta}$ (नहीं $\rho $ बायां) कुछ भी साबित नहीं करता,
और वास्तव में वह $ g $ का असंतुलित रेडियल जाल है
अगला उदाहरण.
\end{example}

\begin{example}[रेडियल जाल]\label{ex:b1:multivar:trap}
होने देना $ f(x, y) = \dfrac{xy}{x^2 + y^2}$ के लिए $(x,y) \neq (0,0)$, $ f(0,
0) = 0$. प्रत्येक अक्ष के साथ, $ f = 0 \to 0$; लेकिन विकर्ण के अनुदिश $ y
= x $, $ f(x, x) = \frac12 \not\to 0$. \emph{मूल पर कोई \omterm{def:b1:topology:closure}{सीमा} नहीं}:
प्रत्येक पंक्ति के साथ-साथ आ रहा है, और यहाँ तक कि \omterm{thm:b1:matrices:conjugation}{समान} \omterm{def:b1:topology:closure}{सीमा} भी पा रहा है
प्रत्येक, पर्याप्त नहीं है (यहाँ पंक्ति सीमाएँ असहमत हैं; बदतर उदाहरण हैं
सभी पंक्तियों पर सहमत होना फिर भी एक परवलय पर विफल होना,
\cref{exo:b1:multivar:3}). निरंतरता प्रत्येक चर में अलग से
मतलब नहीं है निरंतरता.
\end{example}

\section{आंशिक व्युत्पन्न}

\begin{definition}\label{def:b1:multivar:partial}
 \emph{आंशिक व्युत्पन्न}\index{आंशिक व्युत्पन्न} का $ f $ पर
$(a, b)$ एक-चर हैं डेरिवेटिव अक्षों के अनुदिश:
\[
\frac{\partial f}{\partial x}(a,b)
= \lim_{h \to 0} \frac{f(a + h, b) - f(a,b)}{h},
\qquad
\frac{\partial f}{\partial y}(a,b)
= \lim_{k \to 0} \frac{f(a, b + k) - f(a,b)}{k}.
\]
$ f $ वर्ग का है $ C^1$ पर $ U $ जब दोनों मौजूद हैं और हैं निरंतर पर
$ U $. \emph{ग्रेडियेंट}\index{ग्रेडियेंट} है $\nabla f(a,b) =
\bigl(\frac{\partial f}{\partial x},\, \frac{\partial f}{\partial
y}\bigr)(a,b)$.
\end{definition}

\begin{theorem}[$ C^1$ एक स्पर्शरेखा तल का तात्पर्य है]\label{thm:b1:multivar:c1}
होने देना $ f $ होना $ C^1$ पर $ U $ और $(a,b) \in U $. फिर ऐसे $(h, k) \to
(0,0)$:
\[
f(a + h, b + k) = f(a, b)
+ h\,\frac{\partial f}{\partial x}(a,b)
+ k\,\frac{\partial f}{\partial y}(a,b)
+ o\bigl(\norm{(h,k)}\bigr) .
\]
विशेष रूप से $ f $ है निरंतर, और ग्राफ़ $ z = f(x,y)$ पर है
प्रत्येक बिंदु पर स्पर्शरेखा तल सूत्र को पढ़ता है।
\end{theorem}

\begin{example}[क्रिया में रैखिक सन्निकटन]
\label{ex:b1:multivar:linapprox}
अनुमान लगाना $ f(1.02,\ 0.99)$ के लिए $ f(x, y) = x^3y^2$. पर $(1, 1)$:
$ f = 1$, $\frac{\partial f}{\partial x} = 3x^2y^2 = 3$,
$\frac{\partial f}{\partial y} = 2x^3y = 2$, इसलिए
\cref{thm:b1:multivar:c1} देता है
\[
f(1.02,\ 0.99) \approx 1 + 3\,(0.02) + 2\,(-0.01) = 1.04 ,
\]
वास्तविक मूल्य के विरुद्ध $1.02^3 \times 0.99^2 = 1.04006\dots $ ---
त्रुटि वृद्धि में दूसरे क्रम की है, जैसा कि
$ o(\norm{(h,k)})$ वादे. ग्राफ़ का स्पर्शरेखा तल
$(1, 1, 1)$ है $ z = 1 + 3(x - 1) + 2(y - 1)$, समीकरण
अनुमान में निहित है.
\end{example}

\begin{proof}
एक समय में एक निर्देशांक को स्थानांतरित करें:
\[
f(a+h, b+k) - f(a,b)
= \bigl[f(a+h, b+k) - f(a, b+k)\bigr] + \bigl[f(a, b+k) -
f(a,b)\bigr].
\]
एक-चर माध्य मान प्रमेय द्वारा
(\cref{thm:b1:derivative:mvt}), पहला ब्रैकेट है $ h\,
\frac{\partial f}{\partial x}(a + \theta h,\, b + k)$ कुछ के लिए
$\theta \in \intoo{0}{1}$, और दूसरा है $ k\,\frac{\partial
f}{\partial y}(a,\, b + \theta' k)$. निरंतरता आंशिक भाग में
$(a,b)$ आइए हम प्रत्येक को (मान पर) के रूप में लिखें $(a,b)$) $+$
(गलती $\to 0$); कुल त्रुटि है $ h\,\varepsilon_1 +
k\,\varepsilon_2 = o(\norm{(h,k)})$ तब से $\abs h, \abs k \leq
\norm{(h,k)}$.
\end{proof}

\begin{theorem}[शृंखला नियम]\label{thm:b1:multivar:chain}
होने देना $ f $ होना $ C^1$ पर $ U $ और $ t \mapsto (x(t), y(t))$ होना $ C^1$ एक से \omterm{prop:b1:reals:intervals}{अंतराल} में $ U $. तब $ g(t) = f\bigl(x(t), y(t)\bigr)$ है
$ C^1$, साथ\index{श्रृंखला नियम! दो चर}
\[
g'(t) = x'(t)\,\frac{\partial f}{\partial x}\bigl(x(t),y(t)\bigr)
+ y'(t)\,\frac{\partial f}{\partial y}\bigl(x(t),y(t)\bigr)
= \bigl\langle \nabla f,\ (x', y')\bigr\rangle .
\]
\end{theorem}

\begin{proof}
आवेदन करना \cref{thm:b1:multivar:c1} पर $(x(t), y(t))$ साथ $(h, k) =
(x(t+s) - x(t),\, y(t+s) - y(t))$: जैसा $ s \to 0$, एक-चर
भिन्नता प्रदान करती है $ h = s\,x'(t) + o(s)$ और $ k = s\,y'(t)
+ o(s)$, इसलिए $\norm{(h, k)} = O(s)$ और
\[
g(t+s) - g(t)
= h\,\frac{\partial f}{\partial x} + k\,\frac{\partial
f}{\partial y} + o\bigl(\norm{(h,k)}\bigr)
= s\,\Bigl(x'\,\frac{\partial f}{\partial x} +
y'\,\frac{\partial f}{\partial y}\Bigr) + o(s) ,
\]
दोनों को अवशोषित करने वाली अंतिम त्रुटि $ o(s)$का $ h $ और $ k $
(आंशिकों के निश्चित मानों से गुणा किया गया) और
$ o(O(s))$ स्पर्शरेखा-तल अनुमान का. द्वारा विभाजित करें $ s $ और चलो
$ s \to 0$. निरंतरता का $ g'$ सभी का अनुसरण करता है
सामग्री.
\end{proof}

\begin{example}[श्रृंखला नियम, दोनों तरीकों से सत्यापित]
\label{ex:b1:multivar:chainrun}
होने देना $ f(x, y) = x^2 y $ और $ g(t) = f(t, t^2)$. सीधे: $ g(t) =
t^2\cdot t^2 = t^4$, इसलिए $ g'(t) = 4t^3$. \omterm{def:b1:series:def}{शृंखला} नियम द्वारा:
$\frac{\partial f}{\partial x} = 2xy $ और $\frac{\partial
f}{\partial y} = x^2$, वक्र के साथ मूल्यांकन किया गया $(t, t^2)$:
\[
g'(t) = 1\cdot 2t\cdot t^2 + 2t\cdot t^2 = 2t^3 + 2t^3 = 4t^3 .
\]
दोनों गणनाएँ सहमत हैं, और विभाजन सार्थक है: $2t^3$
विकास चलने से आता है \emph{दाहिनी ओर} के माध्यम से
$ x $-ढलान, $2t^3$ हिलने से \emph{ऊपर की ओर} के माध्यम से
$ y $-ढलान। उन वक्रों पर जहां के लिए कोई \omterm{def:b1:topology:closure}{बंद} रूप नहीं है $ g $ मौजूद है, केवल
दूसरी गणना जीवित रहती है --- यही प्रमेय का बिंदु है।
\end{example}

\begin{remark}[ग्रेडिएंट पढ़ना]
एक इकाई दिशा के साथ $ u $, श्रृंखला नियम चालू $ t \mapsto f(a + tu)$
देता है \emph{दिशात्मक यौगिक} $\langle \nabla f(a),
u\rangle $: अधिकतम कब $ u $ साथ में अंक $\nabla f(a)$ (द्वारा
कॉची-श्वार्ज़, \cref{thm:b1:euclid:cs}). \omterm{def:b1:multivar:partial}{ग्रेडियेंट} है
तीव्रतम चढ़ाई की दिशा, और यह है ओर्थोगोनल स्तर तक
वक्र $\{f = c\}$ (अंतर करें $ f $ ए में खींचे गए वक्र के अनुदिश
स्तर \omterm{def:b1:logic:sets}{तय करना}: श्रृंखला नियम देता है $\langle\nabla f,\,
\text{tangent}\rangle = 0$).
\end{remark}

\begin{example}[लेवल कर्व्स और ग्रेडिएंट्स, एक फ़ंक्शन पर]
\label{ex:b1:multivar:levelcurves}
लेना $ f(x, y) = x^2 - y^2$. इसका स्तर सेट: $\{f = c\}$ एक है
हाइपरबोला बाएँ-दाएँ खुलता है $ c > 0$, ऊपर-नीचे के लिए $ c < 0$,
और रेखाओं का क्रॉस किया हुआ जोड़ा $ y = \pm x $ के लिए $ c = 0$ --- द
समोच्च \omterm{def:b1:logic:map}{नक्शा} एक पहाड़ी दर्रे के साथ \omterm{met:b1:multivar:monge}{लादने की सीमा} पर
उद्गम जहां दो शून्य-स्तरीय रेखाएं प्रतिच्छेद करती हैं। ढाल: $\nabla
f = (2x, -2y)$. बिंदु पर $(2, 1)$ (स्तर पर $ c = 3$):
$\nabla f = (4, -2)$, जबकि स्तर का स्पर्शरेखा वेक्टर
वक्र, उस बिंदु के निकट पैरामीट्रिज्ड $\bigl(t,
\sqrt{t^2 - 3}\bigr)$, है $\bigl(1, \frac{t}{\sqrt{t^2 -
3}}\bigr) = (1, 2)$ पर $ t = 2$ --- सचमुच
\[
\langle (4, -2),\ (1, 2)\rangle = 4 - 4 = 0 :
\]
\omterm{def:b1:multivar:partial}{ग्रेडियेंट} समोच्च के लंबवत, ऊपर की ओर इशारा करते हुए
के मान $ f $ (यहाँ: से दूर $ y $-अक्ष). दो और
रीडिंग: द \omterm{def:b1:multivar:partial}{ग्रेडियेंट} बिल्कुल काठी पर गायब हो जाता है, जहां
समोच्च \omterm{def:b1:logic:map}{नक्शा} चुटकी; और स्पर्शरेखा \emph{रेखा} को
स्तर वक्र पर $(2,1)$ है $4(x - 2) - 2(y - 1) = 0$, यानी\ $2x -
y = 3$ --- समीकरण ``$\langle \nabla f,\ M - M_0\rangle =
0$'' जो दीर्घवृत्त स्पर्शरेखा को सामान्यीकृत करता है
\cref{exo:b1:curves:11}.
\end{example}

\begin{theorem}[श्वार्ज]\label{thm:b1:multivar:schwarz}
अगर $ f $ वर्ग का है $ C^2$ (आंशिक के आंशिक भाग मौजूद हैं और हैं
निरंतर), तब\index{श्वार्ज का प्रमेय}
\[
\frac{\partial^2 f}{\partial x\,\partial y}
= \frac{\partial^2 f}{\partial y\,\partial x} .
\]
\end{theorem}
\admitted

\section{स्थानीय चरम}

\begin{method}[चरम अध्ययन, संगठित]
\label{met:b1:multivar:extremamethod}
\begin{enumerate}
  \item \emph{हल करना $\nabla f = 0$ पूरी तरह।} प्रत्येक का गुणनखंड करें
        जब भी संभव हो आंशिक (के उत्पाद) \omterm{def:b1:linmaps:def}{रेखीय} कारक
        सिस्टम को पारदर्शी मामलों में विभाजित करें, जैसे कि
        \cref{ex:b1:multivar:fourpoints} नीचे); एक भूला हुआ
        मामला भुला दिया गया है महत्वपूर्ण बिन्दू.
  \item \emph{प्रत्येक बिंदु को वर्गीकृत करें} स्पंज डेटा के साथ $ r, s,
        t $ --- पुनर्गणना की गई \emph{प्रत्येक बिंदु पर}, एक बार भी नहीं
        सब.
  \item \emph{अगर $ rt - s^2 = 0$}, परीक्षण करना $ f $ सीधे साथ
        बिंदु के माध्यम से अच्छी तरह से चुने गए वक्र (पहले रेखाएँ, फिर
        परवलय), या तो दो संकेतों के लिए शिकार (कोई चरम \omterm{def:b1:topology:closure}{सीमा} नहीं)
        या सभी को कवर करने वाले तर्क के साथ एक \omterm{def:b1:topology:closure}{बंद} चिह्न के लिए
        दिशानिर्देश.
  \item \emph{वैश्विक तस्वीर के लिए पीछे हटें}: जांचें
        अनंत पर व्यवहार (एक स्थानीय न्यूनतम कोई वैश्विक नहीं हो सकता है
        एक), और यदि डोमेन नहीं है \omterm{def:b1:topology:open}{खुला}, इसका इलाज करें \omterm{def:b1:topology:closure}{सीमा}
        अलग से (\cref{exo:b1:multivar:12}) --- द महत्वपूर्ण बिन्दू प्रमेय देखता है \omterm{def:b1:topology:closure}{आंतरिक भाग} केवल अंक.
\end{enumerate}
\end{method}

\begin{theorem}[महत्वपूर्ण बिंदु]\label{thm:b1:multivar:critical}
अगर $ f $ ($ C^1$ पर \omterm{def:b1:topology:open}{खुला} सेट $ U $) पर एक स्थानीय चरम \omterm{def:b1:topology:closure}{सीमा} है $(a,b)
\in U $, तब $\nabla f(a,b) = (0,0)$: बात यह है
\emph{गंभीर}\index{गंभीर}.
\end{theorem}

\begin{proof}
एक-चर फ़ंक्शन $ x \mapsto f(x, b)$ और $ y \mapsto f(a,
y)$ पास होना \omterm{def:b1:topology:closure}{आंतरिक भाग} स्थानीय चरम पर $ a $, सम्मान।\ $ b $:
\cref{prop:b1:derivative:fermat} दोनों आंशिक को मारता है।
\end{proof}

\begin{method}[दूसरे क्रम का परीक्षण (मोन्ज नोटेशन)]\label{met:b1:multivar:monge}
ए पर महत्वपूर्ण बिन्दू एक का $ C^2$ फ़ंक्शन, सेट
\[
r = \frac{\partial^2 f}{\partial x^2},
\qquad
s = \frac{\partial^2 f}{\partial x \partial y},
\qquad
t = \frac{\partial^2 f}{\partial y^2}
\qquad (\text{values at the point}).
\]
\begin{itemize}
  \item अगर $ rt - s^2 > 0$: स्थानीय चरम --- के लिए न्यूनतम $ r > 0$,
        के लिए अधिकतम $ r < 0$;
  \item अगर $ rt - s^2 < 0$: कोई चरम नहीं (ए \emph{लादने की सीमा}\index{लादने की सीमा});
  \item अगर $ rt - s^2 = 0$: परीक्षण मौन है; सीधे जांच करें.
\end{itemize}
(औचित्य---एक टेलर--क्रम में युवा विस्तार $2$ और
द्विघात रूप का संकेत अध्ययन $ r h^2 + 2shk + tk^2$ --- है
दूसरे वर्ष में किया गया; परीक्षण का उपयोग यहां एक कार्यप्रणाली के रूप में किया जाता है
उपकरण.)
\end{method}

\begin{example}\label{ex:b1:multivar:extrema}
$ f(x,y) = x^3 + y^3 - 3xy $. महत्वपूर्ण बिंदु: $\nabla f = (3x^2 -
3y,\; 3y^2 - 3x) = 0$ देता है $ y = x^2$ और $ x = y^2$, इसलिए $ x = x^4$:
$ x \in \{0, 1\}$: अंक $(0,0)$ और $(1,1)$.

दूसरा डेरिवेटिव: $ r = 6x $, $ s = -3$, $ t = 6y $. पर $(0,0)$: $ rt -
s^2 = -9 < 0$: काठी. पर $(1,1)$: $ rt - s^2 = 36 - 9 > 0$, $ r = 6
> 0$: स्थानीय न्यूनतम, $ f(1,1) = -1$. (वैश्विक नहीं: $ f(x, 0) = x^3 \to
-\infty $.)
\end{example}

\begin{example}[संपूर्ण रूप से चार सूत्रीय अध्ययन]
\label{ex:b1:multivar:fourpoints}
$ f(x, y) = xy\,(3 - x - y) = 3xy - x^2y - xy^2$. ढाल:
\[
\frac{\partial f}{\partial x} = y\,(3 - 2x - y),
\qquad
\frac{\partial f}{\partial y} = x\,(3 - x - 2y).
\]
महत्वपूर्ण बिंदु: अगर $ y = 0$, दूसरा समीकरण देता है $ x \in
\{0, 3\}$; अगर $ x = 0$, पहला देता है $ y \in \{0, 3\}$; अगर $ xy
\neq 0$, हल करना $2x + y = 3$, $ x + 2y = 3$: $ x = y = 1$. चार
अंक: $(0,0)$, $(3,0)$, $(0,3)$, $(1,1)$. दूसरा डेरिवेटिव:
$ r = -2y $, $ s = 3 - 2x - 2y $, $ t = -2x $.
\begin{itemize}
  \item $(1,1)$: $ r = -2$, $ s = -1$, $ t = -2$: $ rt - s^2 = 3 >
        0$, $ r < 0$: स्थानीय अधिकतम, $ f(1,1) = 1$.
  \item $(0,0)$: $ r = t = 0$, $ s = 3$: $ rt - s^2 = -9 < 0$:
        काठी; वैसे ही $(3, 0)$ ($ s = -3$) और $(0, 3)$: तीन
        काठी
\end{itemize}
अधिकतम केवल स्थानीय है: $ f(-T, -T) = T^2(3 + 2T) \to
+\infty $. \omterm{def:b1:linmaps:projection}{समरूपता} जांच: $ f(x, y) = f(y, x)$, और वास्तव में
आलोचनात्मक \omterm{def:b1:logic:sets}{तय करना} और वर्गीकरण सममित है $ x
\leftrightarrow y $. व्याख्या: आयतों के बीच-ढीलेपन के साथ
$ x, y \geq 0$, $ x + y \leq 3$, उत्पाद $ xy(3 - x - y)$ का
के तीन ``भाग'' $3$ सबसे बड़ा तब होता है जब भाग बराबर हों
--- AM--GM सिद्धांत की दो-चर छाया।
\end{example}

\begin{remark}[सामान्य ख़तरे]
\label{rem:b1:multivar:pitfalls}
\emph{\omterm{def:b1:multivar:partial}{आंशिक व्युत्पन्न} के एक बिंदु पर मौजूद हो सकता है
असंततता}: रेडियल जाल $ g(x,y) = \frac{xy}{x^2+y^2}$ का
\cref{ex:b1:multivar:trap} है $\frac{\partial g}{\partial
x}(0,0) = \frac{\partial g}{\partial y}(0,0) = 0$ (दोनों अक्ष
प्रतिबंध \omterm{thm:b1:matrices:conjugation}{समान} रूप से गायब हो जाते हैं), फिर भी $ g $ की कोई \omterm{def:b1:topology:closure}{सीमा} नहीं है
उत्पत्ति --- आंशिक जांच केवल दो दिशाओं, निरंतरता जरूरत है
वे सभी; केवल $ C^1$ परिकल्पना व्यवस्था को पुनर्स्थापित करती है
(\cref{thm:b1:multivar:c1}).
\emph{रेखा \omterm{def:b1:topology:closure}{सीमा} कभी भी पर्याप्त नहीं होती}: \cref{exo:b1:multivar:3}का
फ़ंक्शन की \omterm{def:b1:topology:closure}{सीमा} होती है $0$ हर पंक्ति के साथ और फिर भी कोई \omterm{def:b1:topology:closure}{सीमा} नहीं ---
हमेशा परवलय (या \omterm{thm:b1:matrices:conjugation}{समान} रूप से मान्य ध्रुवीय सीमाएं) आज़माएं
$\theta $).
\emph{आलोचना आवश्यक है, पर्याप्त नहीं}: काठी प्रचुर मात्रा में है
(चार में से तीन अंक \cref{ex:b1:multivar:fourpoints}); और
प्रमेय कायम है \emph{\omterm{def:b1:topology:open}{खुला}} सेट केवल --- ए पर \omterm{def:b1:topology:closed}{बंद किया हुआ} डिस्क,
एक्स्ट्रेमा पर बैठ सकता है \omterm{def:b1:topology:closure}{सीमा} शून्येतर के साथ \omterm{def:b1:multivar:partial}{ग्रेडियेंट}
(\cref{exo:b1:multivar:12}).
\emph{मूक मामला $ rt - s^2 = 0$ वास्तव में चुप है}: $ x^4
+ y^4$ (न्यूनतम) और $ x^3 + y^3$ (न ही) दोनों के पास है $ r = s = t
= 0$ मूल पर; \omterm{def:b1:vspaces:sum}{प्रत्यक्ष} संकेत अध्ययन ही निर्णय करता है
(\cref{exo:b1:multivar:6}, समारोह $ h $).
\emph{ \omterm{def:b1:multivar:partial}{ग्रेडियेंट} है ओर्थोगोनल वक्रों को समतल करने के लिए, साथ में नहीं
उन्हें}: एक समोच्च रेखा का अनुसरण करने के लिए, लंबवत चलें $\nabla
f $; सबसे तेजी से चढ़ने के लिए, इसके साथ-साथ आगे बढ़ें --- दो उलटावों को मिलाते हुए
प्रत्येक समोच्च की ज्यामिति \omterm{def:b1:logic:map}{नक्शा}.
\end{remark}

\begin{remark}[जहां दो चर नेतृत्व करते हैं]
\label{rem:b1:multivar:whereused}
यह अध्याय एक द्वार है. \omterm{def:b1:multivar:partial}{ग्रेडियेंट} और श्रृंखला नियम का विस्तार होता है
शब्दशः करने के लिए $ n $ वर्ष~2 खंड में चर, जहां
$ o(\norm{(h,k)})$ का \cref{thm:b1:multivar:c1} बन जाता है
\emph{अंतर} और Monge परीक्षण के माध्यम से पूर्ण रूप से सिद्ध किया गया है
क्रम-दो टेलर सूत्र और द्विघात रूप। विशेष मामला
जिसका निपटारा किया जा सकता है \emph{यह} वर्ष --- द्विघात फलन,
जिसके लिए दूसरे क्रम का विस्तार सटीक है --- विषय है
सप्ताहांत की समस्या, और ऐसा ही होता है जो चलता है
विश्व की डेटा फिटिंग: \omterm{pb:b1:multivar:1}{कम से कम वर्गों} प्रतिगमन. विवश
एक्स्ट्रेमा (\cref{exo:b1:multivar:5} एक पूर्वावलोकन था) लैग्रेंज बन गया
वर्ष~2 में गुणक; हार्मोनिक कार्य
(\cref{exo:b1:multivar:7}) के जटिल विश्लेषण में वापसी
वर्ष~3 खंड.
\end{remark}

\begin{remark}[पुस्तक 3 के अंदर परिप्रेक्ष्य: वर्ष, बंद]
\label{rem:b1:multivar:perspectives}
यह अध्याय वह है जहां खंड के दो भाग हाथ मिलाते हैं। द
विश्लेषण आधे ने अपने उपकरण एक की आपूर्ति की यौगिक एक समय में:
माध्य मान प्रमेय ड्राइव \cref{thm:b1:multivar:c1}, टेलर
विस्तार चरम परीक्षणों को संचालित करते हैं, और $\varepsilon $ का
\cref{ch:b1:topology} की जगह गेंदों के साथ वापस आये
\omterm{prop:b1:reals:intervals}{अंतराल}. बीजगणित आधे ने ज्यामिति की आपूर्ति की: \omterm{def:b1:multivar:partial}{ग्रेडियेंट}
के माध्यम से पढ़ा जाता है \omterm{def:b1:euclid:def}{अंदरूनी प्रोडक्ट} का \cref{ch:b1:euclid}
(कौची-श्वार्ज़ इसे सबसे तीव्र दिशा बनाता है), मोंज
डेटा $(r, s, t)$ का एक सममित मैट्रिक्स है
\cref{ch:b1:matrices} के निर्धारक परीक्षण के साथ
\cref{ch:b1:det}, और सप्ताहांत समस्या चलती है \omterm{thm:b1:euclid:projection}{ओर्थोगोनल
प्रक्षेपण} डेटा वैक्टर पर. यहाँ तक कि वक्र भी
\cref{ch:b1:curves} स्तर के रूप में लौटें सेट. एक पाठक जो कर सकता है
इन पाँच हैंड-ऑफ़ कार्यों में से प्रत्येक का पुनर्निर्माण क्यों किया गया है
प्रभाव, पूरे वर्ष संशोधित --- जो वास्तविक उद्देश्य है
इस अंतिम अध्याय का.
\end{remark}

\section{अभ्यास}

\begin{exercise}[$\star $]\label{exo:b1:multivar:1}
की गणना करें \omterm{def:b1:multivar:partial}{आंशिक व्युत्पन्न}:
$ f(x,y) = x^2 y + \eu^{xy}$; $\;g(x,y) = \ln(x^2 + y^2)$ (पर
$\R^2\setminus\{0\}$); $\;h(x,y) = \arctan\frac yx $ (पर $ x > 0$).
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:multivar:2}
का अध्ययन करें निरंतरता पर $(0,0)$ (मूल्य के साथ $0$ वहाँ) का:
\[
f(x,y) = \frac{x^2 y}{x^2 + y^2},
\qquad
g(x,y) = \frac{xy}{x^2 + y^2},
\qquad
h(x,y) = \frac{x^3 + y^3}{x^2 + y^2}.
\]
\emph{(ध्रुवीय \omterm{prop:b1:vspaces:coordinates}{COORDINATES} $ x = \rho\cos\theta $, $ y = \rho\sin\theta $
सहायता: के एक कार्य से बंधा हुआ $\rho $ जब भी संभव हो अकेले।)}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:multivar:3}
होने देना $ f(x,y) = \dfrac{x^2 y}{x^4 + y^2}$ ($ f(0,0) = 0$). साबित करें कि
$ f $ \omterm{def:b1:topology:closure}{सीमा} है $0$ मूल पर \emph{हर सीधी रेखा के साथ},
लेकिन उस $ f\bigl(x, x^2\bigr) = \frac12$: $ f $ क्या नहीं है निरंतर पर
$(0,0)$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:multivar:4}
श्वार्ज के प्रमेय को हाथ से सत्यापित करें $ f(x, y) = x^3 y^2 + \sin(xy)$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:multivar:5}
होने देना $ f $ होना $ C^1$ पर $\R^2$ और $ g(t) = f(\cos t, \sin t)$. अभिव्यक्त करना
$ g'(t)$ श्रृंखला नियम के माध्यम से. उसका निष्कर्ष निकालो $ f $ इकाई तक सीमित है
वृत्त उन बिंदुओं पर चरम \omterm{def:b1:topology:closure}{सीमा} को प्राप्त करता है जहां $\nabla f $ के \omterm{thm:b1:matrices:conjugation}{समान}ांतर है
त्रिज्या वेक्टर.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:multivar:6}
खोजें और वर्गीकृत करें महत्वपूर्ण बिंदु का:
\[
f(x, y) = x^2 + xy + y^2 - 3x,
\qquad
g(x, y) = x^2 - y^2 + 4y,
\qquad
h(x, y) = x^4 + y^4 - 2(x - y)^2 .
\]
\emph{(के लिए $ h $, निर्धारक परीक्षण मूल में मौन है:
जाँच करें $ h(x, x)$ और $ h(x, -x)$.)}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:multivar:7}
एक समारोह $ f $ है \emph{लयबद्ध} कब $\frac{\partial^2 f}{\partial
x^2} + \frac{\partial^2 f}{\partial y^2} = 0$. जाँच करें कि $ x^2 -
y^2$, $ xy $, $\eu^x\cos y $ और $\ln(x^2 + y^2)$ (मूल से बाहर) हैं
हार्मोनिक.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b1:multivar:8}
समतल का बिंदु ज्ञात कीजिए $\{x + 2y - z = 4\} \subseteq \R^3$
मूल के निकटतम, दो तरीके: द्वारा ऑर्थोगोनल प्रक्षेपण
(\cref{ch:b1:euclid}), और दो-चर फ़ंक्शन को छोटा करके
$ f(x, y) = x^2 + y^2 + (x + 2y - 4)^2$ नष्ट करके प्राप्त किया गया $ z $.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b1:multivar:9}
(ध्रुवीय में लाप्लासियन \omterm{prop:b1:vspaces:coordinates}{COORDINATES}, पहला संपर्क) चलो $ f $ होना $ C^2$ पर
$\R^2 \setminus \{0\}$ और \emph{रेडियल}: $ f(x, y) =
\varphi\bigl(\sqrt{x^2+y^2}\bigr)$ साथ $\varphi $ कक्षा का $ C^2$ पर
$\intoo{0}{+\infty}$. साबित करें कि
\[
\frac{\partial^2 f}{\partial x^2} + \frac{\partial^2 f}{\partial
y^2} = \varphi''(\rho) + \frac{\varphi'(\rho)}{\rho},
\qquad \rho = \sqrt{x^2 + y^2},
\]
और छिद्रित तल पर सभी रेडियल हार्मोनिक फ़ंक्शन ढूंढें।
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:multivar:10}
ग्राफ़ के स्पर्शरेखा तल का समीकरण दीजिए $ z = xy $ पर
मुद्दा $(1, 1, 1)$. फिर ग्राफ के सभी बिंदु ज्ञात कीजिए
$ f(x,y) = x^3 + y^3 - 3xy $ जहां स्पर्शरेखा तल है
\emph{क्षैतिज}, और उत्तर को संबंधित करें
\cref{ex:b1:multivar:extrema}.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:multivar:11}
होने देना $ f $ होना $ C^1$ पर $\R^2$.
\begin{enumerate}
  \item अगर $\frac{\partial f}{\partial x} = 0$ हर जगह, साबित करो
        वह $ f(x, y)$ पर ही निर्भर करता है $ y $.
  \item सभी खोजें $ C^1$ समीकरण का समाधान
        $\frac{\partial f}{\partial x} = \frac{\partial
        f}{\partial y}$ पर $\R^2$. \emph{(\omterm{def:b1:logic:sets}{तय करना} $ g(u, v) =
        f\bigl(\frac{u+v}2, \frac{u-v}2\bigr)$ और गणना करें
        $\frac{\partial g}{\partial v}$ श्रृंखला नियम द्वारा.)}
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b1:multivar:12}
वैश्विक अधिकतम और न्यूनतम ज्ञात कीजिए $ f(x, y) = xy $ पर
\omterm{def:b1:topology:closed}{बंद किया हुआ} डिस्क $ x^2 + y^2 \leq 1$. \emph{(दो-चर को स्वीकार करें
चरम मूल्य प्रमेय: ए निरंतर पर कार्य करें \omterm{def:b1:topology:closed}{बंद किया हुआ} डिस्क
अपनी \omterm{def:b1:topology:closure}{सीमा} को प्राप्त करता है --- वर्ष~2 खंड में साबित हुआ। का इलाज करें
\omterm{def:b1:topology:open}{खुला} डिस्क द्वारा महत्वपूर्ण बिंदु और यह \omterm{def:b1:topology:closure}{सीमा} द्वारा घेरा
का पैरामीट्रिजेशन \cref{exo:b1:multivar:5}.)}
\end{exercise}

\section{समस्या: न्यूनतम वर्ग और प्रतिगमन रेखा}

\begin{problem}\label{pb:b1:multivar:1}
दिया गया $ n $ डेटा बिंदु, जो सीधी रेखा ``निकटतम'' से गुजरती है
वे सभी? लीजेंड्रे और गॉस ने उत्तर दिया: न्यूनतम करने वाली रेखा
का योग \emph{Squared} ऊर्ध्वाधर त्रुटियाँ --- क्योंकि वह
न्यूनतमकरण रैखिक बीजगणित द्वारा बिल्कुल हल करने योग्य है। यह समस्या
सबसे पहले Monge परीक्षण को सिद्ध करता है \cref{met:b1:multivar:monge}
ईमानदारी से द्विघात कार्यों के लिए (एक मामला जहां
दूसरे क्रम का विस्तार सटीक है), फिर बनाता है \omterm{pb:b1:multivar:1}{सामान्य
समीकरण}, द \omterm{pb:b1:multivar:1}{प्रतिगमन लाइन}, और सहसंबंध गुणांक
यूक्लिडियन ज्यामिति के शीर्ष पर
\cref{ch:b1:euclid}.\index{कम से कम वर्गों}\index{प्रतिगमन लाइन}
\index{सामान्य समीकरण}\index{सह - संबंध}

\medskip
\noindent\textbf{Part I --- Quadratic functions: the Monge test,
proved.} वास्तविक को ठीक करें $ r, s, t $ और चलो $ q(h, k) = r h^2 + 2s hk +
t k^2$.
\begin{enumerate}
  \item कल्पना करना $ r \neq 0$. पूर्ण-वर्गाकार प्रपत्र स्थापित करें
        \[
        q(h, k) = r\Bigl(h + \frac{s}{r}k\Bigr)^{\!2}
        + \frac{rt - s^2}{r}\,k^2 ,
        \]
        और निष्कर्ष निकालें: यदि $ rt - s^2 > 0$, $ q $ सख्त संकेत है
        का $ r $ मूल से बाहर; अगर $ rt - s^2 < 0$, $ q $ दोनों लेता है
        संकेत.
  \item शेष मामलों का निपटारा करें: $ r = 0$, $ t \neq 0$
        (सममित); और $ r = t = 0$ ($ q = 2shk $). निष्कर्ष:
        $ q $ दोनों संकेत लेता है iff $ rt - s^2 < 0$, और $ q $
        केवल मूल iff पर ही लुप्त हो जाता है $ rt - s^2 > 0$.
  \item अभी चलो $ f(x, y) = \frac12\bigl(r x^2 + 2s xy + t
        y^2\bigr) + \beta x + \gamma y + \delta $ द्विघात हो
        ए के साथ कार्य करें महत्वपूर्ण बिन्दू $ X_0 = (a, b)$. सिद्ध करो
        \emph{एकदम सही} विस्तार
        \[
        f(X_0 + (h,k)) = f(X_0) + \tfrac12\,q(h, k)
        \qquad (\text{no remainder}),
        \]
        और द्विघात के लिए मोंज वर्गीकरण निकालें
        फ़ंक्शंस: सख्त वैश्विक न्यूनतम यदि $ rt - s^2 > 0$, $ r >
        0$; सख्त वैश्विक अधिकतम यदि $ rt - s^2 > 0$, $ r < 0$;
        काठी अगर $ rt - s^2 < 0$.
  \item कल्पना करना $ rt - s^2 > 0$ और $ r > 0$ (ऐसा भी $ t > 0$).
        स्पष्ट ज़बरदस्ती को बाध्य साबित करें
        \[
        q(h, k) \geq c\,(h^2 + k^2),
        \qquad
        c = \frac{rt - s^2}{r + t} > 0 .
        \]
        \emph{(दिखाएँ कि द्विघात रूप $ q - c(h^2 + k^2)$,
        गुणांक के साथ $ r - c $, $ s $, $ t - c $, अभी भी है
        गैर-नकारात्मक विभेदक-परीक्षण डेटा।)}
  \item यह निष्कर्ष निकालें कि सकारात्मक निश्चितता वाला एक द्विघात फलन है
        द्विघात भाग की ओर प्रवृत्ति होती है $+\infty $ जैसा $\norm{(x,y)} \to
        \infty $, और इसलिए एक है \emph{अद्वितीय वैश्विक}
        मिनिमाइज़र: यह महत्वपूर्ण बिन्दू. के साथ तुलना करें
        \cref{ex:b1:multivar:extrema}, जहां स्थानीय न्यूनतम
        एक गैर-द्विघात फलन वैश्विक नहीं था।
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The \omterm{pb:b1:multivar:1}{सामान्य समीकरण}.} होने देना $ C_1,
C_2, b \in \R^n $ (विहित \omterm{def:b1:euclid:def}{अंदरूनी प्रोडक्ट}), और
\[
f(u, v) = \norm{u\,C_1 + v\,C_2 - b}^2 .
\]
\begin{enumerate}[resume]
  \item बढ़ाना $ f $ और दिखाएँ कि यह एक द्विघात फलन है $(u,
        v)$ द्विघात-भाग गुणांक के साथ $ r = 2\norm{C_1}^2$,
        $ s = 2\langle C_1, C_2\rangle $, $ t = 2\norm{C_2}^2$.
  \item बताते हैं कि $ rt - s^2 > 0$ अगर और केवल अगर $(C_1, C_2)$ है
        \omterm{def:b1:vspaces:free}{मुक्त} \emph{(कौची-श्वार्ज़ का समानता मामला,
        \cref{thm:b1:euclid:cs})}; यह अभी से मान लें.
  \item महत्वपूर्ण समीकरण दिखाएँ $\nabla f = 0$ हैं
        \emph{\omterm{pb:b1:multivar:1}{सामान्य समीकरण}}
        \[
        \begin{pmatrix}
        \norm{C_1}^2 & \langle C_1, C_2\rangle\\
        \langle C_1, C_2\rangle & \norm{C_2}^2
        \end{pmatrix}
        \begin{pmatrix} u\\ v\end{pmatrix}
        =
        \begin{pmatrix} \langle C_1, b\rangle\\ \langle C_2,
        b\rangle\end{pmatrix}
        \]
        --- का ग्राम मैट्रिक्स \cref{exo:b1:euclid:11} पर
        बाएँ --- और वे बिल्कुल यही कहते हैं: $ b - (uC_1 + vC_2)
        \perp C_1, C_2$. भाग I के साथ समापन करें
        \cref{thm:b1:euclid:projection}: अद्वितीय मिनिमाइज़र
        देता है $ p = uC_1 + vC_2 = $  ऑर्थोगोनल प्रक्षेपण का
        $ b $ पर $\operatorname{Vect}(C_1, C_2)$.
  \item दिखाएँ कि न्यूनतम मान है $\norm{b}^2 - \langle p,
        b\rangle = \norm{b - p}^2$, और पायथागॉरियन बनाएं
        चित्र.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The \omterm{pb:b1:multivar:1}{प्रतिगमन लाइन}.} डेटा बिंदु
$(x_1, y_1), \dots, (x_n, y_n)$, सभी नहीं $ x_i $ बराबर। छोटा करना
\[
E(\alpha, \beta) = \sum_{i=1}^{n}\bigl(y_i - \alpha - \beta
x_i\bigr)^2 .
\]
लिखना $\bar x = \frac1n\sum x_i $, $\bar y = \frac1n\sum y_i $,
$ v_x = \frac1n\sum x_i^2 - \bar x^2$, $ v_y = \frac1n\sum y_i^2 -
\bar y^2$, $ c_{xy} = \frac1n\sum x_i y_i - \bar x\bar y $.
\begin{enumerate}[resume]
  \item भाग II को पहचानें $ C_1 = (1, \dots, 1)$, $ C_2 =
        (x_1, \dots, x_n)$, $ b = (y_1, \dots, y_n)$, और लिखो
        द \omterm{pb:b1:multivar:1}{सामान्य समीकरण}
        \[
        n\alpha + \Bigl(\sum x_i\Bigr)\beta = \sum y_i,
        \qquad
        \Bigl(\sum x_i\Bigr)\alpha + \Bigl(\sum
        x_i^2\Bigr)\beta = \sum x_i y_i .
        \]
  \item उन्हें हल करें:
        \[
        \beta = \frac{c_{xy}}{v_x},
        \qquad
        \alpha = \bar y - \beta\,\bar x ,
        \]
        और निरीक्षण करें कि \emph{\omterm{pb:b1:multivar:1}{प्रतिगमन लाइन}} $ y = \alpha
        + \beta x $ माध्य बिंदु से होकर गुजरता है $(\bar x, \bar
        y)$.
  \item जाँच करना $ v_x > 0$ बिल्कुल जब $ x_i $ सभी नहीं हैं
        बराबर, और इसे प्रश्न 7 से मिलाएँ।
  \item सिद्ध कीजिए कि न्यूनतम त्रुटि है
        \[
        E_{\min} = n\Bigl(v_y - \frac{c_{xy}^2}{v_x}\Bigr)
        = n\,v_y\,(1 - \rho^2),
        \qquad
        \rho = \frac{c_{xy}}{\sqrt{v_x v_y}}
        \quad (v_y \neq 0).
        \]
        परिणाम निकालना $\abs\rho \leq 1$, साथ $\abs\rho = 1$ अगर और
        केवल तभी जब डेटा पूरी तरह से संरेखित हो।
  \item कार्यान्वित उदाहरण: डेटा के लिए $(0,0)$, $(1,1)$, $(2,3)$,
        $(3,4)$, गणना करें $\bar x, \bar y, v_x, c_{xy}$, द
        \omterm{pb:b1:multivar:1}{प्रतिगमन लाइन}, चार अवशेष और $ E_{\min}$.
  \item सिद्ध कीजिए कि अवशेष हैं $\varepsilon_i = y_i - \alpha -
        \beta x_i $ इष्टतम रेखा को संतुष्ट करें $\sum_i
        \varepsilon_i = 0$ और $\sum_i x_i\varepsilon_i = 0$,
        और ऑर्थोगोनैलिटी के माध्यम से दोनों की व्याख्या करें।
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Variations and applications.}
\begin{enumerate}[resume]
  \item (सर्वोत्तम स्थिरांक) दिखाएँ कि स्थिरांक $\alpha $
        कम से कम $\sum_i (y_i - \alpha)^2$ मतलब है $\bar
        y $, और वह न्यूनतम मूल्य है $ n\,v_y $: विचरण
        डेटा के स्थिर न होने की विफलता को मापता है।
  \item (मूल से होकर जाने वाली रेखा) दिखाएँ कि ढलान न्यूनतम हो रहा है
        $\sum_i (y_i - \beta x_i)^2$ है $\beta_0 = \frac{\sum
        x_iy_i}{\sum x_i^2}$, और डेटा पर एक शर्त दें
        के लिए $\beta_0$ ढलान के साथ मेल खाना
        $ c_{xy}/v_x $ प्रश्न 11 का.
  \item (एक पंक्ति से दूरी, फिर से) एक बिंदु के लिए $ M $ और रेखा
        $ D $ के माध्यम से $ P_0$ यूनिट वेक्टर द्वारा निर्देशित $ w $,
        छोटा करना $ g(\tau) = \norm{P_0 + \tau w - M}^2$ और
        निष्कर्ष निकालना $ d(M, D)^2 = \norm{M - P_0}^2 - \langle M - P_0,
        w\rangle^2$; सूत्र पुनर्प्राप्त करें $ d = \frac{\abs{a x_0 +
        b y_0 + c}}{\sqrt{a^2 + b^2}}$ लाइन के लिए $ ax + by + c
        = 0$ विमान में.
  \item (विचरण का विश्लेषण) अपघटन सिद्ध करें $ v_y =
        \beta^2 v_x + \frac{E_{\min}}n $: का विचरण
        $ y_i $ लाइन प्लस द्वारा समझाए गए भाग में विभाजित हो जाता है
        अवशिष्ट विचरण.
  \item (परवलयिक फ़िट) फ़िट होना $ y = a + bx + cx^2$, वह दिखाओ
        द \omterm{pb:b1:multivar:1}{सामान्य समीकरण} हैं $3 \times 3$ सिस्टम के साथ
        पल मैट्रिक्स
        \[
        \begin{pmatrix}
        n & \sum x_i & \sum x_i^2\\
        \sum x_i & \sum x_i^2 & \sum x_i^3\\
        \sum x_i^2 & \sum x_i^3 & \sum x_i^4
        \end{pmatrix},
        \]
        एक ग्राम मैट्रिक्स जो तीन के बराबर उलटा है
        द $ x_i $ विशिष्ट हैं \emph{(स्वतंत्रता $(1, X, X^2)$
        डेटा पर नमूना लिया गया, \cref{exo:b1:euclid:11}; तुलना करें
        सप्ताहांत की समस्या का क्षण मैट्रिक्स
        \cref{ch:b1:det})}.
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Robustness, and synthesis.}
\begin{enumerate}[resume]
  \item वर्गीकृत करें महत्वपूर्ण बिंदु तीन द्विघात में से
        कार्य
        \[
        x^2 + xy + y^2, \qquad x^2 + 3xy + y^2, \qquad
        x^2 + 2xy + y^2 ,
        \]
        भाग I द्वारा, पतित तीसरे मामले का हाथ से इलाज करना
        (न्यूनतम कहाँ प्राप्त हुआ है?)
  \item (बाहरी प्रयोग) बिंदु जोड़ें $(10, 0)$ को
        प्रश्न 14 का डेटा और ढलान की पुनः गणना करें $\beta $.
        क्या हुआ, और क्यों है \emph{Squared} त्रुटि तो
        एक दूर बिंदु के प्रति संवेदनशील?
  \item (भारित \omterm{pb:b1:multivar:1}{कम से कम वर्गों}) वजन दिया गया $ w_i > 0$,
        छोटा करना $\sum_i w_i(y_i - \alpha - \beta x_i)^2$: दिखाओ
        समाधान प्रश्न के \omterm{thm:b1:matrices:conjugation}{समान} सूत्रों द्वारा दिया गया है
        11 भारित साधन, विचरण और सहप्रसरण के साथ
        (उन्हें परिभाषित करें)।
  \item दिखाएँ कि इष्टतम रेखा के लिए, $ E_{\min} = 0$ बल
        सभी बिंदु बिल्कुल उस पर स्थित हों, और उससे जुड़ें
        समानता का मामला $\abs\rho = 1$ प्रश्न 13 का; पर परीक्षण करें
        संरेखित डेटा $(1,1), (2,3), (3,5)$.
  \item संश्लेषण, चार वाक्यों में: द्विघात कार्य क्यों
        एक ऐसा वर्ग है जहां इस अध्याय का दूसरा क्रम है
        परीक्षण के लिए किसी स्वीकृत प्रमेय की आवश्यकता नहीं है; कैसे \omterm{pb:b1:multivar:1}{सामान्य समीकरण}
        के साथ विश्लेषणात्मक न्यूनतमकरण की पहचान करें \omterm{thm:b1:euclid:projection}{ओर्थोगोनल
        प्रक्षेपण} का \cref{ch:b1:euclid}; क्या संबंध है
        गुणांक माप और कौन सी असमानता इसे बांधती है;
        और अध्याय 18-23 के कौन से टुकड़े (\omterm{def:b1:vspaces:free}{आधार}, ग्राम और
        पल मैट्रिक्स, अनुमान) पुनः प्रकट हुआ। नाम बताएं
        भाग II-III में अध्ययन की गई विधि और समीकरण।
\end{enumerate}
\end{problem}
