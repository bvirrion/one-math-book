\chapter{समतल वक्र}\label{ch:b1:curves}

की गणना \cref{ch:b1:derivative,ch:b1:taylor} के लिए बनाया गया था
कार्य $y = f(x)$; ज्यामिति और यांत्रिकी के अधिकांश वक्र ---
प्रक्षेपपथ, वृत्तों पर लुढ़के वृत्त, कक्षाएँ --- उस रूप को अस्वीकार करें
और इसके बजाय आओ \emph{पैरामीट्रिज्ड} घटता $ t \mapsto (x(t),
y(t))$ या जैसे \emph{ध्रुवीय} घटता $ r = r(\theta)$. यह शास्त्रीय
अध्याय दोनों का अध्ययन करता है, और इसके साथ समाप्त होता है शंकु.

\section{पैरामीट्रिज्ड वक्र}

\begin{definition}\label{def:b1:curves:def}
ए \emph{पैरामीट्रिज्ड वक्र}\index{पैरामीट्रिज्ड वक्र} एक है \omterm{def:b1:logic:map}{नक्शा} $ f
\colon I \to \R^2$, $ t \mapsto M(t) = (x(t), y(t))$, साथ $ x, y $ का
कक्षा $ C^1$ (कम से कम) पर \omterm{prop:b1:reals:intervals}{अंतराल} $ I $. \emph{वेग
वेक्टर} है $ f'(t) = (x'(t), y'(t))$; मुद्दा $ M(t_0)$ है
\emph{नियमित} कब $ f'(t_0) \neq (0,0)$, और यह \emph{स्पर्शरेखा}
वहां से होकर गुज़रने वाली लाइन है $ M(t_0)$ निर्देशक $ f'(t_0)$.
\end{definition}

\begin{proof}[क्यों $ f'$ स्पर्शरेखा को निर्देशित करता है]
टेलर द्वारा--युवा घटकवार, $ M(t) = M(t_0) + (t - t_0) f'(t_0) +
o(t - t_0)$: राग दिशा $\frac{M(t) - M(t_0)}{t - t_0}$
आदत है $ f'(t_0)$. नियमितता परिकल्पना वह है जो बनाती है
\omterm{def:b1:topology:closure}{सीमा} ए \emph{दिशा}: अगर $ f'(t_0) = 0$, डिस्प्ले ढह जाता है
को $ M(t) = M(t_0) + o(t - t_0)$ और कैसे के बारे में कुछ नहीं कहता
वक्र उस बिंदु को छोड़ देता है --- जहां से अलग उपचार होता है
नीचे दी गई विधि में एकवचन बिंदु, जहां पहला
\emph{न मिटनेवाला} यौगिक की भूमिका संभालता है $ f'$.
यह भी ध्यान दें कि स्पर्शरेखा रेखा एक ज्यामितीय वस्तु है: कोई भी
पुनर्\omterm{def:b1:counting:objects}{परिवर्तन} \omterm{def:b1:counting:objects}{परिवर्तन} $ f'$ एक शून्येतर अदिश कारक द्वारा और
लाइन को अपरिवर्तित छोड़ देता है.
\end{proof}

\begin{method}[एक पैरामीट्रिज्ड वक्र का अध्ययन]\label{met:b1:curves:study}
\begin{enumerate}
  \item \emph{डोमेन कम करें} \omterm{def:b1:linmaps:projection}{समरूपता} का उपयोग करना: बीच संबंध
        $ M(-t)$, $ M(t + T)$, \dots\ और $ M(t)$ (प्रतिबिंब,
        अनुवाद, आवधिकता), फिर केवल घटा हुआ भाग बनाएं।
  \item \emph{बदलाव}: के लक्षण सारणीबद्ध करें $ x'$ और $ y'$ संयुक्त रूप से;
        वक्र दाएं/बाएं की तरह चलता है $ x'$, ऊपर/नीचे जैसे $ y'$.
  \item \emph{उल्लेखनीय बिंदु}: क्षैतिज स्पर्शरेखा ($ y' = 0 \neq
        x'$), ऊर्ध्वाधर स्पर्शरेखा ($ x' = 0 \neq y'$); एक पर
        \emph{एकवचन} बिंदु ($ f' = 0$), उच्चतर जांच करें डेरिवेटिव
        स्पर्शरेखा दिशा के लिए.
  \item \emph{स्पर्शोन्मुख व्यवहार} के अंत में $ I $, फिर स्केच करें।
\end{enumerate}
\end{method}

\begin{example}[स्पर्शोन्मुख शाखाएँ, काम किया]
\label{ex:b1:curves:asymptotes}
$ x(t) = t $, $ y(t) = t + \dfrac1t $ पर $\intoo{0}{+\infty}$. जैसा $ t
\to 0^{+}$: $ x \to 0$ जबकि $ y \to +\infty $ --- वक्र चढ़ता है
साथ में \emph{ऊर्ध्वाधर अनंतस्पर्शी} $ x = 0$ (के लिए सीमित \omterm{def:b1:topology:closure}{सीमा})
एक समन्वय, दूसरे के लिए अनंत)। जैसा $ t \to +\infty $:
दोनों \omterm{prop:b1:vspaces:coordinates}{COORDINATES} उड़ाओ, इसलिए एक पंक्ति का परीक्षण करो: अंतर
\[
y(t) - x(t) = \frac1t \longrightarrow 0^{+}
\]
प्रदर्शित करता है \emph{परोक्ष अनंतस्पर्शी} $ y = x $, से संपर्क किया
ऊपर. बीच में, $ y' = 1 - \frac1{t^2}$ पर गायब हो जाता है $ t = 1$:
मुद्दा $(1, 2)$ शाखा का निचला बिंदु है, और $ x' = 1 >
0$ संपूर्ण, इसलिए वक्र हमेशा दाईं ओर बढ़ता है। एक चाप,
दो अनंतस्पर्शी, एक न्यूनतम: तीन से एक संपूर्ण चित्र
गणनाएँ --- नीचे दिखाई देने वाली सामान्य रेसिपी के साथ:
कब $ x, y \to \infty $ एक साथ, जांच करें $ y/x $ एक उम्मीदवार के लिए
ढलान (यहाँ) $\to 1$), तब $ y - (\text{slope})\,x $ के लिए
अवरोधन और दृष्टिकोण का पक्ष।
\end{example}

\begin{example}[एस्ट्रोइड]\label{ex:b1:curves:astroid}
$ x(t) = \cos^3 t $, $ y(t) = \sin^3 t $.\index{astroid} \omterm{def:b1:linmaps:projection}{समरूपता}:
$ M(t + 2\pi) = M(t)$ (एक \omterm{def:b1:vspaces:span}{अवधि} का अध्ययन करें); $ M(-t)$ का प्रतिबिम्ब है
$ M(t)$ में $ x $-अक्ष; $ M(\pi - t)$ में $ y $-अक्ष; $ M(\frac\pi2 -
t)$ विकर्ण में $ y = x $: यह अध्ययन करने के लिए पर्याप्त है $ t \in
\intcc{0}{\frac\pi4}$ और प्रकट करो.

वेग: $ f'(t) = 3\sin t\cos t\,(-\cos t, \sin t)$. पर
$\intoo{0}{\frac\pi2}$ सभी बिंदु निर्देशित स्पर्शरेखा के साथ नियमित हैं
द्वारा $(-\cos t, \sin t)$; पर $ t = 0$ (मुद्दा $(1,0)$) वेग
गायब हो जाता है: ए \emph{उभार}\index{उभार}, जहां वक्र उलट जाता है
स्पर्शरेखा दिशा $(-1, 0)\cdot(\pm)$ --- \omterm{def:b1:linmaps:projection}{समरूपता} से चार
cusps पर बैठो $(\pm1, 0), (0, \pm1)$.
\end{example}

\begin{example}[एक आंकड़ा-आठ, पूरा अध्ययन किया गया]
\label{ex:b1:curves:lissajous}
$ x(t) = \sin t $, $ y(t) = \sin 2t $ (ए \emph{लिसाजस वक्र}).
\emph{समानताएं}: $ M(t + \pi) = (-x(t), y(t))$ (प्रतिबिंब में
द $ y $-अक्ष), $ M(-t) = (-x(t), -y(t))$ (केंद्रीय \omterm{def:b1:linmaps:projection}{समरूपता}),
$ M(\pi - t) = (x(t), -y(t))$ (प्रतिबिंब में $ x $-अक्ष): यह
अध्ययन करने के लिए पर्याप्त है $ t \in \intcc{0}{\frac\pi2}$ और प्रकट करो.
\emph{बदलाव}: $ x' = \cos t \geq 0$ भर में, जबकि $ y' =
2\cos 2t $ पहले से सकारात्मक है $ t = \frac\pi4$ और इसके बाद नकारात्मक:
चाप दाहिनी ओर शिखर पर चढ़ता है
$\bigl(\frac{\sqrt2}2,\ 1\bigr)$ पर $ t = \frac\pi4$ (क्षैतिज
स्पर्शरेखा), फिर दाईं ओर उतरती है $(1, 0)$ पर $ t =
\frac\pi2$, कहाँ $ x' = 0 \neq y'$: ऊर्ध्वाधर स्पर्शरेखा.
\emph{दोहरा बिंदु}: $ M(0) = M(\pi) = (0,0)$, दो अलग-अलग के साथ
वेग
\[
f'(0) = (1,\ 2), \qquad f'(\pi) = (-1,\ 2) :
\]
दो \emph{नियमित} शाखाएँ मूल स्थान पर भिन्न-भिन्न स्थानों पर प्रतिच्छेद करती हैं
कोण --- एक दोहरा बिंदु, एक एकल बिंदु नहीं: प्रत्येक मार्ग है
पूरी तरह से सुचारू, दोनों मार्ग केवल अपना स्थान साझा करते हैं।
संपूर्ण वक्र नीचे चित्र-आठ है।
\end{example}

\section{ध्रुवीय वक्र}

\begin{definition}\label{def:b1:curves:polar}
ए \emph{ध्रुवीय वक्र}\index{ध्रुवीय वक्र} द्वारा दिया गया है $ r = r(\theta)$:
पैरामीटर का बिंदु $\theta $ है
\[
M(\theta) = r(\theta)\,\vec u(\theta),
\qquad \vec u(\theta) = (\cos\theta, \sin\theta).
\]
साथ $\vec v(\theta) = (-\sin\theta, \cos\theta) = \vec
u\,'(\theta)$, वेग है
\[
M'(\theta) = r'(\theta)\, \vec u(\theta) + r(\theta)\, \vec
v(\theta) .
\]
परिणाम: कहाँ $ r \neq 0$, बात नियमित है, और
स्पर्शरेखा किरण के साथ कोण बनाती है $ V $ द्वारा दिए गए $\tan V =
\frac{r}{r'}$ (बीच का कोण $ M'$ और $\vec u $); कहाँ $ r(\theta_0)
= 0$, वक्र मूल बिंदु से होकर गुजरता है \emph{किरण स्पर्शरेखा के साथ
$\theta = \theta_0$} (दिशा $\vec u(\theta_0)$, पढ़ लिया
$ M'(\theta_0) = r'(\theta_0)\vec u(\theta_0)$, या \omterm{def:b1:topology:closure}{सीमा} से
राग: से राग $ O $ को $ M(\theta)$ द्वारा ले जाया जाता है $\vec
u(\theta)$ स्वयं, जो प्रवृत्त होता है $\vec u(\theta_0)$ ---तो
नियम तब भी कायम रहता है जब $ r'(\theta_0) = 0$ और मुद्दा यह है
एकवचन, इसीलिए ध्रुवीय cusps मूल में, जैसे
कार्डियोइड, उनका स्पर्शरेखा निःशुल्क प्राप्त करें)।
\end{definition}

\begin{example}[कार्डियोइड]\label{ex:b1:curves:cardioid}
$ r(\theta) = 1 + \cos\theta $.\index{कारडायोड} \omterm{def:b1:linmaps:projection}{समरूपता}: $ r(-\theta)
= r(\theta)$: में प्रतिबिंब $ x $-अक्ष; अध्ययन $\theta \in
\intcc{0}{\pi}$. $ r $ से घट जाती है $2$ को $0$; पर $\theta = \pi $,
$ r = 0$: वक्र किरण की स्पर्शरेखीय रूप से मूल बिंदु तक पहुंचता है
$\theta = \pi $ (द $ x $-अक्ष), एक गठन \omterm{ex:b1:curves:astroid}{उभार} वहाँ---दिल का
बिंदु. पर स्पर्शरेखा $\theta = 0$: $ r' = 0$, इसलिए $\tan V = \infty $:
अक्ष के लंबवत.

कोना $ V $ एक और पढ़ने लायक. पर $\theta =
\frac\pi2$: $ r = 1$ और $ r' = -1$, इसलिए $\tan V = \frac{r}{r'} =
-1$: स्पर्श रेखा इसके साथ तीन-चौथाई समकोण बनाती है
बाहर जाने वाली किरण --- वक्र पहले से ही वापस उसकी ओर झुक रहा है
\omterm{ex:b1:curves:astroid}{उभार}. सूत्र $\tan V = r/r'$ स्पर्शरेखा निर्देश प्रदान करता है
बिना किसी गणना के पूरे वक्र के साथ $ M'(\theta)$
जो भी हो: यह ढलान को पढ़ने का ध्रुवीय एनालॉग है।
\end{example}

\begin{example}[ध्रुवीय से कार्तीय: एक छिपा हुआ वृत्त]
\label{ex:b1:curves:polarcircle}
क्या है \omterm{def:b1:curves:polar}{ध्रुवीय वक्र} $ r = 2\cos\theta $? गुणा करके $ r $:
$ r^2 = 2r\cos\theta $, यानी\ $ x^2 + y^2 = 2x $, यानी
\[
(x - 1)^2 + y^2 = 1 :
\]
केंद्र का घेरा $(1, 0)$ और त्रिज्या $1$, गुजर रहा है
मूल. बहीखाता मामले: जैसे $\theta $ पर चलाता है
$\intoc{-\frac\pi2}{\frac\pi2}$ पूरा घेरा बह गया है
बिल्कुल एक बार ($ r $ दोनों सिरों पर गायब हो जाता है), और पर $\theta =
\pm\frac\pi2$ नियम ``किरण के अनुदिश मूल बिंदु पर स्पर्शरेखा
$\theta = \theta_0$'' एक देता है \emph{खड़ा} वहाँ स्पर्शरेखा ---
ज्यामिति से मेल खाता है, क्योंकि ऊर्ध्वाधर अक्ष वास्तव में स्पर्शरेखा है
इस सर्कल में $ O $. के लिए $\theta $ उस \omterm{def:b1:topology:closure}{सीमा} से परे, $ r < 0$
उसी सर्कल को पुनः ट्रेस करता है: एक अनुस्मारक कि ए \omterm{def:b1:curves:polar}{ध्रुवीय वक्र} एक है
\emph{पैरामीट्रिज्ड} वस्तु को अपने ऊपर से गुजरने की अनुमति दी गई।
\end{example}

\end{omfigure}

\section{वक्राकार लंबाई}

\begin{definition}[एक चाप की लंबाई]\label{def:b1:curves:arclength}
 \emph{लंबाई}\index{लंबाई} एक का $ C^1$ आर्क $ f \colon
\intcc{a}{b} \to \R^2$ है \omterm{thm:b1:integration:def}{अभिन्न} गति का:
\[
L = \int_a^b \norm{f'(t)}\,\dd t
= \int_a^b \sqrt{x'(t)^2 + y'(t)^2}\;\dd t .
\]
(प्रेरणा: एक छोटे से पर \omterm{prop:b1:reals:intervals}{अंतराल}, $ M(t + h) \approx M(t) + h
f'(t)$, इसलिए चाप एक बहुभुज के करीब है जिसके खंड की लंबाई का योग है
एक रीमैन योग के लिए $\norm{f'}$,
\cref{thm:b1:integration:riemann}।) एक के लिए \omterm{def:b1:curves:polar}{ध्रुवीय वक्र} $ r =
r(\theta)$, वेग $ r'\vec u + r\vec v $ है ओर्थोगोनल
घटक, तो
\[
L = \int_{\theta_1}^{\theta_2}
\sqrt{r'(\theta)^2 + r(\theta)^2}\;\dd\theta .
\]
लंबाई (मोनोटोन,) पर निर्भर नहीं करती $ C^1$)
पैरामीट्रिजेशन चुना गया: प्रतिस्थापित करना $ t = \varphi(s)$ में
\omterm{thm:b1:integration:def}{अभिन्न} (\cref{thm:b1:integration:parts}) गुणा करता है $ f'$ द्वारा
$\varphi'$ और $\dd t $ द्वारा $\varphi'^{-1}$.
\end{definition}

\begin{example}[विवेक जांच: वृत्त]\label{ex:b1:curves:circlelength}
के लिए $ f(t) = (R\cos t, R\sin t)$ पर $\intcc{0}{2\pi}$: $\norm{f'} =
R $, इसलिए $ L = 2\pi R $ --- परिभाषा परिधि लौटाती है।
रिपेरामेट्रिज़ेशन परीक्षण: \omterm{def:b1:logic:map}{नक्शा} $ g(t) = (R\cos 2t, R\sin 2t)$ पर
$\intcc{0}{\pi}$ खींचता है \emph{वही} दोगुनी गति से चक्कर लगाओ
$\norm{g'} = 2R $, और
\[
\int_0^{\pi} 2R\,\dd t = 2\pi R
\]
फिर से: आधा समय, दोगुनी गति, वही \omterm{def:b1:curves:arclength}{लंबाई} --- द
परिभाषा में वादा किया गया अपरिवर्तनीयता, संख्याओं पर एक बार देखा गया।
(दौड़ना $ g $ सब पर $\intcc{0}{2\pi}$ देगा $4\pi R $: ए
दो बार घुमाया गया वक्र दोगुना लंबा होता है \emph{एक यात्रा के रूप में};
\omterm{def:b1:curves:arclength}{लंबाई} पैरामीटरयुक्त पथ और ईमानदार बहीखाता को मापता है
द \omterm{prop:b1:reals:intervals}{अंतराल} गणना का हिस्सा है।)
एस्ट्रोइड के लिए $(\cos^3t, \sin^3t)$: $\norm{f'} = 3\abs{\sin
t\cos t} = \tfrac32\abs{\sin 2t}$, और \omterm{def:b1:linmaps:projection}{समरूपता} द्वारा $ L = 4\int_0^{\pi/2}
\tfrac32\sin 2t\,\dd t = 6$: यूनिट सर्कल के अंदर खींचा गया एक वक्र,
का \omterm{def:b1:curves:arclength}{लंबाई} $6 < 2\pi $. सप्ताहांत की समस्या सबसे प्रसिद्ध मापती है
सभी का आर्क.
\end{example}

\section{कॉनिक्स}

\begin{definition}[फोकस--डायरेक्ट्रिक्स परिभाषा]\label{def:b1:curves:conics}
एक बिंदु तय करें $ F $ (\emph{केंद्र}), एक पंक्ति $ D $ के माध्यम से नहीं $ F $
(\emph{नियता}) और $ e > 0$ (\emph{सनक}).
\emph{शंकुधर}\index{शंकुधर} इनमें से डेटा है
\[
\mathcal{C} = \{M : d(M, F) = e\; d(M, D)\}:
\]
एक \emph{अंडाकार} के लिए $ e < 1$, ए \emph{परवलय} के लिए $ e = 1$, ए
\emph{अतिशयोक्ति} के लिए $ e > 1$. (वृत्त विकृत प्रतीत होता है
\omterm{def:b1:topology:closure}{सीमा} $ e \to 0$.)
\end{definition}

\begin{example}[परिभाषा, एक परवलय पर जाँची गई]
\label{ex:b1:curves:parabolacheck}
परवलय लीजिए $ y^2 = 4x $, यानी\ $2p = 4$: केंद्र $ F = (1, 0)$
और डायरेक्ट्रिक्स $ D : x = -1$ (घटा हुआ रूप
\cref{thm:b1:curves:reduced} उन्हें डालता है $\pm\frac p2$). पर
बिंदु $ M = (1, 2)$ वक्र का:
\[
MF = \sqrt{(1-1)^2 + 2^2} = 2,
\qquad
d(M, D) = 1 - (-1) = 2 :
\]
बराबर, जैसे $ e = 1$ मांग. पर $ M' = (4, 4)$: $ MF' = \sqrt{9 +
16} = 5$ और $ d(M', D) = 5$ दोबारा। फोकस--डायरेक्ट्रिक्स
परिभाषा एक अमूर्तता नहीं है: यह दूरियों की एक जोड़ी है
किसी भी बिंदु पर माप कर सकते हैं, और बीजगणित को कम कर सकते हैं
समीकरण एक बार और हमेशा के लिए किए गए माप के अलावा और कुछ नहीं है
सब.
\end{example}

\begin{theorem}[घटे समीकरण]\label{thm:b1:curves:reduced}
एक अच्छी तरह से चुने हुए में \omterm{def:b1:euclid:orthogonal}{ऑर्थोनॉर्मल} चौखटा:
\[
\text{ellipse: } \frac{x^2}{a^2} + \frac{y^2}{b^2} = 1
\quad (a \geq b > 0),
\qquad
\text{hyperbola: } \frac{x^2}{a^2} - \frac{y^2}{b^2} = 1,
\qquad
\text{parabola: } y^2 = 2px,
\]
दीर्घवृत्त के लिए, के साथ: foci पर $(\pm c, 0)$, $ c = \sqrt{a^2 - b^2}$,
$ e = \frac ca $, और द्विफोकल लक्षण वर्णन $ MF + MF' = 2a $;
अतिपरवलय के लिए: $ c = \sqrt{a^2 + b^2}$, $ e = \frac ca $,
$\abs{MF - MF'} = 2a $, स्पर्शोन्मुख $ y = \pm\frac ba x $.
\end{theorem}

\begin{proof}
मूल बिंदु और डायरेक्ट्रिक्स ऊर्ध्वाधर पर ध्यान केंद्रित करें, $ x = -h $
($ h > 0$): स्थिति $ MF^2 = e^2 d(M, D)^2$ पढ़ता $ x^2 + y^2 =
e^2 (x + h)^2$. के लिए $ e = 1$: $ y^2 = 2hx + h^2$, के बाद एक परवलय
बदलाव $ x \mapsto x - \frac h2$ (इसलिए $ p = h $). के लिए $ e \neq 1$:
में वर्ग पूरा करना $ x $,
\[
(1 - e^2)\Bigl(x - \frac{e^2 h}{1 - e^2}\Bigr)^{\!2} + y^2
= e^2h^2 + \frac{e^4 h^2}{1 - e^2}
= \frac{e^2 h^2}{1 - e^2} .
\]
\omterm{def:b1:logic:sets}{तय करना} $ X = x - \frac{e^2h}{1-e^2}$ (फ्रेम का एक बदलाव)। कब $ e < 1$,
सकारात्मक दाहिनी ओर से विभाजित करें:
$\frac{X^2}{a^2} + \frac{y^2}{b^2} = 1$ साथ $ a = \frac{eh}{1 -
e^2}$, $ b = \frac{eh}{\sqrt{1-e^2}}$; कब $ e > 1$, के दोनों तरफ
डिस्प्ले फ्लिप साइन उचित रूप से और \omterm{thm:b1:matrices:conjugation}{समान} विभाजन देता है
$\frac{X^2}{a^2} - \frac{y^2}{b^2} = 1$ साथ $ a = \frac{eh}{e^2 -
1}$, $ b = \frac{eh}{\sqrt{e^2 - 1}}$. के बताए गए मान $ c $
अनुसरण करना ($ c^2 = a^2 - b^2$ या $ a^2 + b^2$ देता है $ c = ea $ दोनों में
मामले, फोकस को सही ढंग से रखना), और बाइफोकल गुण हैं
घटे हुए समीकरणों पर \omterm{def:b1:vspaces:sum}{प्रत्यक्ष} सत्यापन। के लिए विस्तार से
दीर्घवृत्त, के साथ $ F' = (c, 0)$ और $ M = (X, y)$ वक्र के:
\[
MF'^2 = (X - c)^2 + y^2
= X^2 - 2cX + c^2 + b^2\Bigl(1 - \frac{X^2}{a^2}\Bigr)
= \frac{c^2}{a^2}X^2 - 2cX + a^2
= (a - eX)^2 ,
\]
का उपयोग करते हुए $ c^2 + b^2 = a^2$ और $ e = \frac ca $; तब से $\abs X \leq
a $ और $ e < 1$, $ a - eX > 0$, इसलिए $ MF' = a - eX $ बिना किसी वर्ग के
जड़ कभी निकाली गई. दर्पण गणना देती है $ MF = a + eX $,
कहाँ से $ MF + MF' = 2a $, स्थिर। अतिपरवलय के लिए भी यही बात है
बीजगणित की पैदावार $ MF = \abs{a + eX}$ और $ MF' = \abs{eX - a}$,
अंतर के साथ $\pm2a $ शाखा के अनुसार.
\end{proof}

\begin{example}\label{ex:b1:curves:conicexample}
$\frac{x^2}{25} + \frac{y^2}{9} = 1$: दीर्घवृत्त, $ a = 5$, $ b = 3$, $ c
= 4$: फोकस $(\pm4, 0)$, विलक्षणता $\frac45$. इसका पैरामीट्रिजेशन:
$(5\cos t, 3\sin t)$ --- अनिसोट्रोपिक रूप से फैला हुआ एक वृत्त; द
इसके किसी भी बिंदु की नाभि से दूरियों का योग है $10$.
\end{example}

\begin{example}[किसी कक्षा को उसके ध्रुवीय समीकरण से पढ़ना]
\label{ex:b1:curves:orbit}
ध्रुवीय \omterm{def:b1:curves:conics}{शंकुधर} $ r = \dfrac{1}{1 + \frac12\cos\theta}$
(\cref{exo:b1:curves:7} साथ $ p = 1$, $ e = \frac12$) एक है
ए के साथ दीर्घवृत्त \emph{मूल पर ध्यान केंद्रित करें} --- ए की ज्यामिति
सूर्य के साथ ग्रहों की कक्षा $ O $. से सब कुछ निकालें
$ p $ और $ e $: की कमी से
\cref{thm:b1:curves:reduced}, $ a = \dfrac{p}{1 - e^2} =
\dfrac{1}{3/4} = \dfrac43$ और $ c = ea = \dfrac23$. दो
एप्साइड्स बिना किसी सिद्धांत के इसकी जांच करते हैं:
\[
r(0) = \frac{1}{3/2} = \frac23 = a - c
\quad (\text{perihelion}),
\qquad
r(\pi) = \frac{1}{1/2} = 2 = a + c
\quad (\text{aphelion}),
\]
और उनका योग $\frac23 + 2 = \frac83 = 2a $ प्रमुख को पुनः प्राप्त करता है
अक्ष. जब भी फोकस होता है तो ध्रुवीय रूप स्वाभाविक होता है
शारीरिक रूप से प्रतिष्ठित; घटा हुआ कार्टेशियन रूप, जब भी
सममिति अक्ष हैं। दोनों के बीच कनवर्ट करना बिल्कुल सही है
प्रमेय की पूर्ण-वर्ग गणना क्या करती है।

एक डायल के रूप में विलक्षणता, अंततः: रखें $ p = 1$ और मुड़ें $ e $ में $ r
= \frac{1}{1 + e\cos\theta}$. पर $ e = 0$: वृत्त $ r = 1$. पर
$ e = \frac12$: दीर्घवृत्त का अभी अध्ययन किया गया, $ r $ के बीच झूल रहा है
$\frac23$ और $2$. पर $ e = 1$: $ r(\theta) \to \infty $ जैसा $\theta
\to \pi $ --- वक्र अब \omterm{def:b1:topology:closure}{बंद} नहीं होता: एक परवलय, यह
सबसे दूर बिंदु को अनंत तक धकेल दिया गया। पर $ e = 2$: हर
पर गायब हो जाता है $\cos\theta = -\frac12$, और केवल $\theta \in
\intoo{-\frac{2\pi}3}{\frac{2\pi}3}$ जीवित: ए की एक शाखा
हाइपरबोला, दो स्पर्शोन्मुख दिशाओं के साथ भाग रहा है। एक
सूत्र, संपूर्ण \omterm{def:b1:curves:conics}{शंकुधर} परिवार, और संक्रमण बिंदु $ e =
1$ उस क्षण के रूप में दिखाई देता है जब हर पहली बार शून्य पर पहुंचता है।
\end{example}

\begin{remark}[सामान्य ख़तरे]
\label{rem:b1:curves:pitfalls}
\emph{दोहरे अंक एकवचन बिंदु नहीं हैं}: ए पर
स्व-क्रॉसिंग, प्रत्येक शाखा नियमित है; ``एकवचन'' का तात्पर्य है
$ f'(t_0) = 0$ पैरामीटर के एक मान के लिए
(\cref{ex:b1:curves:lissajous} बनाम एस्ट्रोइड का cusps).
\emph{लंबवत स्पर्शरेखा बनाम \omterm{ex:b1:curves:astroid}{उभार}}: $ x'(t_0) = 0 \neq y'(t_0)$
ऊर्ध्वाधर स्पर्शरेखा वाला एक नियमित बिंदु है; केवल $ x' = y' = 0$
उच्च-क्रम विस्तार की मांग करता है।
\emph{विपरीत किरण पर ऋणात्मक त्रिज्या आलेख}: के लिए $ r(\theta) <
0$ बात यह है $-\abs{r}\,\vec u(\theta)$, कोण पर $\theta +
\pi $ (\cref{exo:b1:curves:5}); इसे भूलने से आंतरिक लूप खो जाते हैं
या वक्रों को दोगुना कर देता है।
\emph{वक्राकार \omterm{def:b1:curves:arclength}{लंबाई} एकीकृत $\norm{f'}$, नहीं $ f'$}: विभाजित करें
\omterm{thm:b1:integration:def}{अभिन्न} गति के शून्य पर --- क्षुद्र ग्रह के लिए,
एकीकरण $\frac32\sin 2t $ पूर्ण \omterm{def:b1:vspaces:span}{अवधि} के बिना निरपेक्षता के बिना
मान देता है $0$, नहीं $6$.
\emph{कॉनिक्स पढ़ने से पहले कम किया जाना चाहिए}: में $9x^2 +
25y^2 - 36x - 50y - 164 = 0$, न तो अक्ष, न ही केंद्र,
न ही वर्ग पूरा होने तक विलक्षणता दिखाई देती है
(\cref{exo:b1:curves:6}); और एक पर लुप्त हो रहा द्विघात भाग
\omterm{def:b1:counting:objects}{परिवर्तन}ीय का अर्थ परवलय है, न कि ``अपक्षयी दीर्घवृत्त''।
\end{remark}

\begin{remark}[ये मोड़ कहाँ जाते हैं]
\label{rem:b1:curves:whereused}
\omterm{def:b1:curves:def}{पैरामीट्रिज्ड वक्र} यांत्रिकी की भाषा हैं: प्रक्षेप पथ
वक्र हैं, वेग सदिश स्पर्शरेखा सदिश हैं, और वक्राकार \omterm{def:b1:curves:arclength}{लंबाई} का \cref{def:b1:curves:arclength} दूरी की दौड़ है.
शंकु जहाँ भी व्युत्क्रम-वर्ग नियम कार्य करता है --- ग्रह संबंधी, पुनः प्रकट होता है
कक्षाएँ सूर्य के फोकस पर दीर्घवृत्ताकार होती हैं। में
\cref{ch:b1:multivar}, वक्र स्तर बन जाते हैं सेट कार्यों का
दो चरों में से, और इस अध्याय की स्पर्शरेखा मिलती है
अगले की ढाल. वर्ष~2 की मात्रा वक्रता और जोड़ती है
वक्र का स्थानीय विहित रूप; सप्ताहांत की समस्या नीचे दी गई है
पहले से ही, केवल इस वर्ष के टूल के साथ, सब कुछ निकालता है
सत्रहवीं शताब्दी को इसके सबसे प्रसिद्ध वक्र के बारे में पता था।
\end{remark}

\begin{remark}[पुस्तक 3 के अंदर परिप्रेक्ष्य]
\label{rem:b1:curves:perspectives}
यह अध्याय खंड के पूरे प्रथम भाग का उपभोग करता है
अंतिम अध्याय खिलाता है. \emph{ग्रहण किया हुआ}: स्पर्शरेखा सदिश हैं
डेरिवेटिव (\cref{ch:b1:derivative}), वक्राकार \omterm{def:b1:curves:arclength}{लंबाई} और क्षेत्र हैं
\omterm{thm:b1:integration:def}{अभिन्न} (\cref{ch:b1:integration}), cusps द्वारा तय किये जाते हैं
टेलर विस्तार (\cref{ch:b1:taylor}), टॉटोक्रोन एक है
\omterm{def:b1:linmaps:def}{रेखीय} विभेदक समीकरण (\cref{ch:b1:diffeq}), और हर
दूरी और कोण यूक्लिडियन है (\cref{ch:b1:euclid}).
\emph{खिलाया}: में \cref{ch:b1:multivar}, एक वक्र $ t \mapsto (x(t),
y(t))$ एक स्तर के अंदर खींचा गया \omterm{def:b1:logic:sets}{तय करना} $\{f = c\}$ द्वारा विभेदित किया जाता है
श्रृंखला नियम, और परिणामी पहचान $\langle \nabla f,\
f'(t)\rangle = 0$ इस अध्याय के स्पर्शरेखा सदिशों से विवाह करता है
अगले एक के ढ़ाल---वक्र की स्पर्श रेखा और
सतह की सामान्य दिशा वही गणना देखी जाती है
दोनों बैंकों से.
\end{remark}

\begin{exercise}[$\star $]\label{exo:b1:curves:2}

\begin{exercise}[$\star $]\label{exo:b1:curves:3}

\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:curves:4}
स्केच करें \omterm{def:b1:curves:polar}{ध्रुवीय वक्र} $ r = \cos 2\theta $ (चार पंखुड़ियों वाला गुलाब) और

(एक पंक्ति पहचानें)।
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:curves:5}
का अध्ययन करें \omterm{def:b1:curves:polar}{ध्रुवीय वक्र} $ r = 1 + 2\cos\theta $ (एक लीमाçचालू): डोमेन

\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:curves:6}
की पहचान करें \omterm{def:b1:curves:conics}{शंकुधर} $9x^2 + 25y^2 - 36x - 50y - 164 = 0$: कम करें
वर्गों को पूरा करना, केंद्र, अर्ध-अक्ष, फोकस, विलक्षणता देना।
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:curves:7}
सिद्ध कीजिए कि a का ध्रुवीय समीकरण \omterm{def:b1:curves:conics}{शंकुधर} मूल पर ध्यान देने के साथ
है
\[
r = \frac{p}{1 + e\cos\theta}
\]
(दूरी पर लंबवत दिशा $\frac pe $ फोकस का दाहिना भाग)।
का कौन सा मान $\theta $ कब अनुमति है $ e > 1$?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:curves:8}
द्विफोकसी संपत्ति से $ MF + MF' = 2a $, ''माली'' का निष्कर्ष निकालें
दीर्घवृत्त का निर्माण'', और सिद्ध करें कि स्पर्शरेखा पर $ M $
के साथ \omterm{thm:b1:matrices:conjugation}{समान} कोण बनाता है $ MF $ और $ MF'$ \emph{(प्रतिबिंब संपत्ति;
अंतर करना $\norm{M(t) - F} + \norm{M(t) - F'} = 2a $ और
यूनिट-वेक्टर उत्पादों के लुप्त योग की व्याख्या करें)}.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b1:curves:9}
 \emph{बर्नौली का लेम्निस्केट} है \omterm{def:b1:curves:polar}{ध्रुवीय वक्र} $ r^2 = \cos
2\theta $ (लेना $ r = \sqrt{\cos2\theta}$ जहां परिभाषित किया गया है)।
\begin{enumerate}
  \item इसका डोमेन, सममिति, मूल बिंदु पर स्पर्शरेखा और बताएं
        इसे स्केच करें.
  \item साबित करें कि यह बिंदुओं का स्थान है $ M $ साथ $ MF \cdot MF'
        = \frac12$ कहाँ $ F, F' = \bigl(\pm\frac{1}{\sqrt2},
        0\bigr)$. \emph{(गणना करें $ MF^2\,MF'^2$ ध्रुवीय में
        \omterm{prop:b1:vspaces:coordinates}{COORDINATES}.)}
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:curves:10}
कुल की गणना करें \omterm{def:b1:curves:arclength}{लंबाई} कार्डियोइड का $ r = 1 + \cos\theta $
के ध्रुवीय सूत्र का उपयोग करना \cref{def:b1:curves:arclength}
\emph{(पहचान $1 + \cos\theta = 2\cos^2\frac\theta2$ बदल जाता है
वर्गमूल में $2\abs{\cos\frac\theta2}$)}.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:curves:11}
दिखाएँ कि दीर्घवृत्त की स्पर्श रेखा है $(a\cos t, b\sin t)$ पर
पैरामीटर का बिंदु $ t $ समीकरण है
\[
\frac{x\cos t}{a} + \frac{y\sin t}{b} = 1 ,
\]
और स्पर्शरेखा को घटाएं $\frac{x^2}{a^2} + \frac{y^2}{b^2} = 1$
एक बिंदु पर $(x_0, y_0)$ दीर्घवृत्त का: $\frac{x\,x_0}{a^2} +
\frac{y\,y_0}{b^2} = 1$ (''वर्गों का विभाजन'' नियम)।
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b1:curves:12}
 \emph{लघुगणकीय सर्पिल} है \omterm{def:b1:curves:polar}{ध्रुवीय वक्र} $ r =
\eu^{k\theta}$ ($ k > 0$ तय, $\theta \in \R $).
\begin{enumerate}
  \item वह कोण दिखाओ $ V $ त्रिज्या और के बीच
        स्पर्शरेखा है \emph{स्थिर} ($\tan V = \frac1k $) --- द
        सर्पिल प्रत्येक किरण को एक ही कोण पर पार करता है।
  \item सर्पिल को एक कोण पर घुमाते हुए दिखाएँ $ c $ एमएपीएस यह
        कारक द्वारा इसकी स्केलिंग पर $\eu^{-kc}$: हर
        सर्पिल का घूमना इसका आवर्धन है
        (स्वयं-समानता)।
  \item की गणना करें \omterm{def:b1:curves:arclength}{लंबाई} चाप का $\theta \in
        \intoc{-\infty}{\theta_0}$ (की एक \omterm{def:b1:topology:closure}{सीमा} के रूप में \omterm{def:b1:curves:arclength}{लंबाई} पर
        $\intcc{A}{\theta_0}$, $ A \to -\infty $) और उसका निरीक्षण करें
        यह परिमित है: एक वक्र जो अनंत बार सर्पिल होता है
        मूल के चारों ओर, परिमित का \omterm{def:b1:curves:arclength}{लंबाई}.
\end{enumerate}
\end{exercise}

\section{समस्या: चक्रज, वक्रों की रानी}

\begin{problem}\label{pb:b1:curves:1}
त्रिज्या का एक पहिया $ R $ बिना फिसले लुढ़कता है $ x $-अक्ष;
प्रारंभ में मूल बिंदु पर रिम का बिंदु निशान 
\emph{चक्रज}. सत्रहवीं शताब्दी में इस वक्र पर संघर्ष हुआ
--- गैलीलियो ने इसके पेपर कटआउट को तौला, व्रेन ने इसे मापा,
रोबरवाल ने इसके क्षेत्रफल की गणना की, ह्यूजेन्स ने इस पर घड़ियाँ बनाईं --- और
उनका प्रत्येक परिणाम इस अध्याय की पहुंच के भीतर है। हम
चार क्लासिक्स सिद्ध करें: स्पर्शरेखा निर्माण, रेन्स \omterm{def:b1:curves:arclength}{लंबाई}
$8R $, क्षेत्र $3\pi R^2$, और ह्यूजेन्स का टॉटोक्रोन
संपत्ति.\index{चक्रज}\index{tautochrone}

\medskip
\noindent\textbf{Part I --- Rolling and the tangent.}
\begin{enumerate}
  \item पहिये के एक कोण पर घूमने के बाद $ t $, इसका केंद्र
        पर बैठता है $\Omega(t) = (Rt, R)$ (बिना फिसले लुढ़कना:
        संपर्क दूरी $=$ चाप लुढ़का हुआ)। दिखाओ कि चिह्नित है
        बिंदु पर है
        \[
        M(t) = \bigl(R(t - \sin t),\; R(1 - \cos t)\bigr).
        \]
  \item गणना करना $ f'(t)$ और दिखाओ $\norm{f'(t)} =
        2R\,\abs{\sin\frac t2}$; एकल बिंदुओं का पता लगाएं
        (cusps --- सीएफ.\ \cref{exo:b1:curves:2}) और शीर्ष पर
        प्रत्येक मेहराब.
  \item होने देना $ C(t) = (Rt, 0)$ संपर्क बिंदु बनें और $ T(t) =
        (Rt, 2R)$ पहिये का शीर्ष. इसके लिए सिद्ध करें $0 < t <
        2\pi $ सदिश $ M - C $ है \emph{सामान्य} वक्र पर
        $ M(t)$ और वेक्टर $ T - M $ है \emph{स्पर्शरेखा}: खींचना
        एक चक्रवात के स्पर्शरेखा, बिंदु को शीर्ष से जोड़ें
        यह घूमता हुआ चक्र है. \emph{(हर चीज़ का गुणनखंड करें
        $\sin\frac t2$ और $\cos\frac t2$.)}
  \item प्रश्न 3 की गतिज व्याख्या करें: संपर्क बिंदु है
        द \emph{घूर्णन का तात्कालिक केंद्र}, और
        की गति $ M $ की दूरी के बराबर है $ C $ (इकाई कोणीय के लिए
        वेग). सत्यापित करें $\norm{M - C} = 2R\abs{\sin\frac t2}$.
  \item ऊंचाई-गति संबंध सिद्ध करें
        \[
        \norm{f'(t)}^2 = 2R\,y(t) :
        \]
        एक चक्रवात पर इकाई कोणीय वेग से यात्रा की जाती है
        प्रत्येक बिंदु पर गति बिल्कुल मुक्त-पतन गति है
        वर्तमान ऊँचाई के बराबर गिरना। (इसे अपने पास रखें
        भाग IV.)
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Wren's theorem: the arch has \omterm{def:b1:curves:arclength}{लंबाई}
$8R $.}
\begin{enumerate}[resume]
  \item का उपयोग करते हुए \cref{def:b1:curves:arclength}, गणना करें \omterm{def:b1:curves:arclength}{लंबाई}
        एक मेहराब का:
        \[
        L = \int_0^{2\pi} 2R\sin\frac t2\,\dd t = 8R
        \]
        (व्रेन्स प्रमेय, 1658)। चार पहिया व्यास, और नहीं
        $\pi $ कहीं भी.
  \item की गणना करें वक्राकार \omterm{def:b1:curves:arclength}{लंबाई} से \omterm{ex:b1:curves:astroid}{उभार}: $ s(t) =
        4R\bigl(1 - \cos\frac t2\bigr)$, और जांचें $ s(2\pi) =
        8R $.
  \item अब से चाप को मापें \emph{शीर्ष} $ t = \pi $:
        $\sigma(t) = \abs{4R\cos\frac t2}$. अन्तर्निहित को सिद्ध करें
        संबंध
        \[
        \sigma^2 = 8R\,\bigl(2R - y\bigr) :
        \]
        शीर्ष से वर्गाकार चाप की दूरी आनुपातिक है
        ऊँचाई शीर्ष से नीचे गिरती है।
  \item विवेक की जाँच: प्रश्न 8 से पुनर्प्राप्त करें कि अर्ध-आर्क
        शीर्ष से \omterm{ex:b1:curves:astroid}{उभार} है \omterm{def:b1:curves:arclength}{लंबाई} $4R $, और तुलना करें
        की एस्ट्रोइड गणना \cref{ex:b1:curves:circlelength}
        --- दोनों वक्र \omterm{pb:b1:logic:1}{बीजगणितीय} हैं, $\pi $-मुक्त \omterm{def:b1:curves:arclength}{लंबाई};
        स्पष्ट करें कि दोनों के होते हुए भी यह क्या संभव बनाता है
        वृत्तों से निर्मित. \emph{(के रूप को देखो
        $\norm{f'}$.)}
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Roberval's area: $3\pi R^2$.}
\begin{enumerate}[resume]
  \item औचित्य सिद्ध करें कि एक मेहराब और जमीन के बीच का क्षेत्र है
        $ A = \int_0^{2\pi} y(t)\,x'(t)\,\dd t $ \emph{(द
        प्रतिस्थापन $ x = x(t)$ में $\int y\,\dd x $,
        \cref{thm:b1:integration:parts}; $ x $ यह बढ़ रहा है)}.
  \item गणना करना
        \[
        A = R^2\int_0^{2\pi}(1 - \cos t)^2\,\dd t = 3\pi R^2 :
        \]
        बिल्कुल \emph{तीन} पहिये के क्षेत्रफल का गुना ---
        गैलीलियो ने तौलकर जिस अनुपात का अनुमान लगाया था।
  \item एस्ट्रोइड के लिए भी यही विधि $(\cos^3t, \sin^3t)$: दिखाओ
        वह घिरा हुआ क्षेत्र है $\frac{3\pi}8$
        \emph{(रैखिकीकरण करें $\sin^4 t\cos^2 t $; केवल स्थिरांक
        \omterm{def:b1:vspaces:span}{अवधि} पूरी \omterm{def:b1:vspaces:span}{अवधि} तक जीवित रहती है)}.
  \item विवेक ऊपरी इकाई पर प्रश्न 10 के सूत्र की जाँच करें
        अर्धवृत्त $(\cos t, \sin t)$, $ t $ से $\pi $ को $0$:
        क्या यह वापस आता है? $\frac\pi2$?
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Huygens' tautochrone.} पलटें
मेहराब: एक घर्षण रहित मनका गुरुत्वाकर्षण के तहत स्लाइड करता है $ g $, के अंदर
चक्राकार कटोरा
\[
x(t) = R(t + \sin t), \qquad y(t) = R(1 - \cos t)
\qquad (t \in \intcc{-\pi}{\pi}),
\]
जिसका निम्नतम बिंदु मूल बिंदु है ($ y $ ऊपर की ओर मापा गया)।
\begin{enumerate}[resume]
  \item गति की गणना करें $\norm{f'(t)} = 2R\cos\frac t2$, द
        वक्राकार \omterm{def:b1:curves:arclength}{लंबाई} नीचे से $ s(t) = 4R\sin\frac t2$, और
        \omterm{def:b1:arith:prime}{मुख्य} पहचान सिद्ध करें
        \[
        y = \frac{s^2}{8R} .
        \]
  \item मनका पैरामीटर के बिंदु से आराम की स्थिति में जारी किया गया
        $ t_0 > 0$ ऊर्जा संरक्षण का पालन करता है: यदि $ s(\tau)$ दर्शाता है
        समय पर इसकी चाप स्थिति $\tau $, तब $\frac12\bigl(
        \frac{\dd s}{\dd\tau}\bigr)^2 + g\,y = g\,y_0$. पुनः लिखें
        यह, प्रश्न 14 का उपयोग करते हुए, जैसे
        \[
        \Bigl(\frac{\dd s}{\dd\tau}\Bigr)^{\!2}
        = \frac{g}{4R}\,\bigl(s_0^2 - s^2\bigr),
        \qquad s_0 = s(t_0).
        \]
  \item के संबंध में भेद करें $\tau $ और प्राप्त करें
        हार्मोनिक थरथरानवाला
        \[
        \frac{\dd^2 s}{\dd\tau^2} = -\frac{g}{4R}\,s ;
        \]
        इसे हल करें \cref{thm:b1:diffeq:homogeneous2} और
        प्रारंभिक शर्तें: $ s(\tau) = s_0\cos(\omega\tau)$,
        $\omega = \sqrt{g/4R}$.
  \item निष्कर्ष निकालें \emph{tautochrone संपत्ति} (ह्यूजेंस, 1659):
        नीचे तक पहुँचने का समय,
        \[
        T_{\downarrow} = \frac{\pi}{2\omega}
        = \pi\sqrt{\frac Rg}\,,
        \]
        रिलीज बिंदु पर निर्भर नहीं करता --- मोतियों को जारी किया गया
        कटोरे की किन्हीं दो ऊँचाइयों से एक साथ आएँ
        एक साथ.
  \item प्रतिस्थापन द्वारा सत्यापित करें कि $ s(\tau) =
        s_0\cos\omega\tau $ प्रथम कोटि की ऊर्जा को संतुष्ट करता है
        प्रश्न 15 का समीकरण बिल्कुल (न केवल)
        विभेदित एक), और एक वाक्य में समझाएं कि क्यों ए
        गोलाकार पेंडुलम ही है \emph{लगभग}
        समकालिक है जबकि चक्रज बिल्कुल वैसा ही है।
  \item संख्यात्मक रूप से: क्या त्रिज्या $ R $ अवतरण का समय बनाता है
        ठीक एक सेकंड ($ g = 9.81$)? ध्यान दें कि कितना करीब है
        उत्तर है एक मीटर का, और कैसे पूरा दोलन
        \omterm{def:b1:vspaces:span}{अवधि} $2\pi\sqrt{4R/g}$ छोटे-कोण से तुलना करता है
        पेंडुलम सूत्र $2\pi\sqrt{\ell/g}$ के लिए $\ell = 4R $.
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Dividends, and synthesis.}
\begin{enumerate}[resume]
  \item प्रश्न 3 का स्पर्शरेखा नियम लागू करें $ t =
        \frac\pi2$ (लेना $ R = 1$): गणना करें $ M $, $ T $, इसकी दिशा $ MT $, और इसके विरुद्ध जाँच करें
        $ f'(\frac\pi2)$.
  \item (ट्रोकोइड्स) इसके बजाय दूरी पर एक बिंदु चिह्नित करें $ d $ से
        केंद्र ($ d \neq R $): वक्र है $ x = Rt -
        d\sin t $, $ y = R - d\cos t $. उसके लिए दिखाओ $ d < R $ द
        वक्र हर जगह नियमित है और संख्या का ग्राफ है
        विलक्षण व्यवहार ($ x' > 0$: यह आगे बढ़ता है), जबकि के लिए $ d
        > R $ भुज $ x'$ चिन्ह बदलता है और वक्र बनाता है
        लूप --- निकला हुआ किनारा वाला रेलवे पहिया जिसका रिम इंगित करता है
        यात्रा \emph{पीछे की ओर}.
  \item (ब्राचिस्टोक्रोन टीज़र) से \omterm{ex:b1:curves:astroid}{उभार} $(\pi R, 2R)$ का
        नीचे तक कटोरा, चक्रवात वंश समय की तुलना करें
        $\pi\sqrt{R/g}$ सीधी ढलान के साथ समय के साथ
        \omterm{thm:b1:matrices:conjugation}{समान} बिंदुओं को जोड़ना \emph{(निरंतर त्वरण
        $ g\sin\alpha $ तार के साथ)}: दिखाएँ कि तार लेता है
        $\sqrt{\pi^2 + 4}\,\sqrt{R/g} \approx 3.72\sqrt{R/g}$.
        वक्र रेखा को पार करता है --- यह, वास्तव में, सबसे तेज़ है
        सभी वक्रों का, विविधताओं की गणना का परिणाम है।
  \item उसे आर्च पर दिखाएँ ($0 < t < 2\pi $),
        \[
        \frac{\dd y}{\dd x} = \cot\frac t2,
        \qquad
        \frac{\dd^2 y}{\dd x^2} =
        -\frac{1}{4R\sin^4\frac t2} < 0 :
        \]
        मेहराब अवतल है, जिसकी ऊर्ध्वाधर स्पर्शरेखा बिल्कुल ठीक है
        द cusps.
  \item ऊर्ध्वाधर पुनः प्राप्त करें \omterm{ex:b1:curves:astroid}{उभार} की स्पर्शरेखा
        \cref{exo:b1:curves:2} ज्यामितीय रूप से: \omterm{def:b1:topology:closure}{सीमा} की गणना करें
        राग की दिशा $ MT $ प्रश्न 3 के रूप में $ t \to
        0^{+}$, बिना किसी विस्तार के।
  \item संश्लेषण, चार वाक्यों में: किन तीन ने प्रमेयों का नाम दिया
        यह समस्या (उनकी संख्या के साथ) साबित हुई $8R $, $3\pi
        R^2$, $\pi\sqrt{R/g}$ और उनके लेखक); कौन सा एकल
        कम्प्यूटेशनल डिवाइस ($\sin\frac t2$, $\cos\frac t2$
        गुणनखंडन) सभी भागों I, II और IV को संचालित करता है; कैसे
        आंतरिक संबंध $ y = s^2/8R$ परिवर्तित ज्यामिति
        एक में \omterm{def:b1:linmaps:def}{रेखीय} विभेदक समीकरण; और कौन सा अध्याय
        इस पुस्तक का प्रत्येक भाग इस पर निर्भर है।
\end{enumerate}
