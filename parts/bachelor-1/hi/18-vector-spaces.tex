\chapter{वेक्टर रिक्त स्थान}\label{ch:b1:vspaces}

रेखीय बीजगणित यहाँ से शुरू होता है: के अभिगृहीत वेक्टर रिक्त स्थान अलग करो क्या
$\R^2$, $\R^3$, \omterm{def:b1:poly:def}{बहुपद} रिक्त स्थान और फ़ंक्शन स्थान समान हैं
--- कोई जोड़ सकता है, और स्केल कर सकता है। दो अध्याय सिद्धांत का निर्माण करते हैं
(\cref{ch:b1:findim} आयाम जोड़ता है); वे भाषा \omterm{def:b1:logic:sets}{तय करना} ऊपर ---
\omterm{def:b1:vspaces:span}{अवधि}, \omterm{def:b1:vspaces:free}{मुक्त} परिवार, \omterm{def:b1:vspaces:free}{आधार}, \omterm{def:b1:vspaces:sum}{प्रत्यक्ष} योग ---हर एक की रोजी रोटी है
उनके बाद का अध्याय. लगातार, $ K $ अर्थ है $\R $ या $\C $
(\emph{अदिश}).

\section{परिभाषा और उदाहरण}

\begin{definition}[सदिश स्थल]\label{def:b1:vspaces:def}
ए \emph{$ K $-सदिश स्थल}\index{$ क $-सदिश स्थल} एक है \omterm{def:b1:logic:sets}{तय करना} $ E $ एक के साथ
अतिरिक्त बनाना $(E, +)$ एक \omterm{def:b1:structures:group}{एबेलियन} \omterm{def:b1:structures:group}{समूह} (शून्य लिखा है $0_E $ या
$0$), और एक अदिश गुणन $ K \times E \to E $ ऐसे कि, के लिए
सब $\lambda, \mu \in K $ और $ x, y \in E $:
\[
\lambda(x + y) = \lambda x + \lambda y,\quad
(\lambda + \mu) x = \lambda x + \mu x,\quad
\lambda(\mu x) = (\lambda\mu) x,\quad
1\,x = x .
\]
नतीजे: $0\,x = 0_E $, $\lambda\,0_E = 0_E $, $(-1)x = -x $, और
$\lambda x = 0_E \implies \lambda = 0$ या $ x = 0_E $ (गुणा करके
$\lambda^{-1}$).
\end{definition}

\begin{proof}[परिणामों का प्रमाण]
के लिए $0\,x = 0_E $: से $(0 + 0)x = 0x + 0x $ और $(0+0)x = 0x $,
रद्द करना $0x $ में \omterm{def:b1:structures:group}{समूह} $(E, +)$. के लिए $\lambda\,0_E $: वही
ट्रिक ऑन $\lambda(0_E + 0_E)$. के लिए $(-1)x $: जोड़ना $ x $,
\[
x + (-1)x = 1\,x + (-1)x = \bigl(1 + (-1)\bigr)x = 0\,x = 0_E ,
\]
इसलिए $(-1)x $ का योगात्मक व्युत्क्रम है $ x $. अंततः यदि $\lambda x
= 0_E $ साथ $\lambda \neq 0$: गुणा करके $\lambda^{-1}$ (द
अदिश रूप a \omterm{def:b1:structures:field}{मैदान}) और दो अभिगृहीतों का उपयोग करें
$\lambda^{-1}(\lambda x) = (\lambda^{-1}\lambda)x = 1x = x $
के साथ साथ $\lambda^{-1}0_E = 0_E $: $ x = 0_E $. उनके जैसा छोटा
हैं, इन चार नियमों का उपयोग प्रत्येक पृष्ठ पर चुपचाप किया जाता है
अनुसरण करता है --- और आखिरी बिल्कुल वहीं है खेत आवश्यक हैं:
अदिश राशि के ऊपर $\Z $, ``अंतरिक्ष'' $\Z/2\Z $ इसका उल्लंघन करेंगे
के साथ $2\,x = 0$.
\end{proof}

\begin{example}\label{ex:b1:vspaces:examples}
$ K^n $ (समन्वयानुसार संचालन); बहुआयामी पद $ K[X]$; द
कार्य $\mathcal{F}(A, K)$ किसी से \omterm{def:b1:logic:sets}{तय करना} $ A $ को $ K $ (बिंदुवार
संचालन) --- युक्त निरंतर कार्य, क्रम
$\mathcal{F}(\N, \R)$, वगैरह।; $\C $ एक के रूप में $\R $-सदिश स्थल। प्रत्येक में
मामले में स्वयंसिद्ध उनसे विरासत में मिले हैं $ K $.
\end{example}

\begin{definition}[उपस्पेस]\label{def:b1:vspaces:subspace}
$ F \subseteq E $ एक है \emph{उपस्पेस}\index{उपस्पेस} कब $0_E \in
F $ और $ F $ जोड़ और अदिश गुणन के तहत स्थिर है ---
समकक्ष:
\[
F \neq \emptyset
\qquad\text{and}\qquad
\forall x, y \in F,\ \forall \lambda \in K,\quad
x + \lambda y \in F .
\]
एक उपस्थान स्वयं एक है सदिश स्थल. उप-स्थानों का कोई प्रतिच्छेदन
एक उपस्थान है; एक संघ लगभग कभी नहीं होता है (समान प्रमाण)।
\cref{exo:b1:structures:6}).
\end{definition}

\begin{example}\label{ex:b1:vspaces:subspaces}
में $\mathcal{F}(\R, \R)$: द निरंतर कार्य,
विभेदक वाले, द बहुआयामी पद डिग्री का $\leq n $ (लिखा हुआ
$ K_n[X]$ अंदर $ K[X]$), एक सजातीय रैखिक के समाधान
विभेदक समीकरण (\cref{thm:b1:diffeq:homogeneous2} बस कहा
वह). गैर-उदाहरण: $\{f : f(0) = 1\}$ (कोई शून्य नहीं), बिल्कुल डिग्री
$ n $ (जोड़ के तहत स्थिर नहीं).
\end{example}

\begin{example}[सबस्पेस या नहीं: चार फैसले, तर्क दिए गए]
\label{ex:b1:vspaces:subspaceornot}
वास्तविक अनुक्रमों के स्थान में:
\begin{itemize}
  \item $\{u : u \text{ bounded}\}$ \emph{है} ए \omterm{def:b1:vspaces:subspace}{उपस्पेस}: $0$ है
        परिबद्ध, और यदि $\abs{u_n} \leq M $, $\abs{v_n} \leq M'$,
        तब $\abs{u_n + \lambda v_n} \leq M + \abs\lambda M'$.
  \item $\{u : u_n \to 1\}$ है \emph{नहीं}: शून्य अनुक्रम है
        गायब (और दो सदस्यों का योग होता है $2$).
  \item $\{u : u \text{ monotone}\}$ है \emph{नहीं}: $ u_n = n $
        और $ v_n = -n + (-1)^n $ एकरस हैं, उनका योग है
        $(-1)^n $ क्या नहीं है; जोड़ के अंतर्गत स्थिरता स्वयंसिद्ध है
        वह विफल रहता है, भले ही \omterm{def:b1:logic:sets}{तय करना} रोकना $0$ और सब
        इसके सदस्यों के अदिश गुणज।
  \item $\{u : u_{n+1} = u_n^2\}$ है \emph{नहीं}: इसमें है $0$
        लेकिन $2u $ जैसे ही भाग जाता है $ u $ एक अशून्य सदस्य है
        ($2u_{n+1} \neq (2u_n)^2$ सामान्य तौर पर) --- वर्ग करना है
        अरैखिकता.
\end{itemize}
कार्य क्रम हमेशा एक समान होता है: परीक्षण $0$ पहला (सबसे सस्ता),
फिर स्थिरता --- और खंडन करने के लिए, एक स्पष्ट प्रति उदाहरण
यह जोड़ी किसी भी संदेह को मात देती है।
\end{example}

\section{विस्तार, योग, प्रत्यक्ष योग}

\begin{definition}[रैखिक संयोजन, विस्तार]\label{def:b1:vspaces:span}
ए \emph{रैखिक संयोजन} परिवार की $(x_1, \dots, x_p)$ का
के वैक्टर $ E $ कोई है $\lambda_1 x_1 + \dots + \lambda_p x_p $
($\lambda_i \in K $). \omterm{def:b1:logic:sets}{तय करना} उन सभी का है
\emph{अवधि}\index{अवधि} $\operatorname{Vect}(x_1, \dots, x_p)$: यह
एक है \omterm{def:b1:vspaces:subspace}{उपस्पेस}, परिवार में सबसे छोटा।
\end{definition}

\begin{proof}[दो दावों का प्रमाण]
स्थिरता: दो रैखिक संयोजनों का योग $\sum\lambda_i x_i +
\sum\mu_i x_i = \sum(\lambda_i + \mu_i)x_i $ फिर से एक है, और इसी तरह
एक अदिश गुणज है $\mu\sum\lambda_i x_i = \sum(\mu\lambda_i)
x_i $; शून्य संयोजन दिखाता है $0$ संबंधित: द \omterm{def:b1:vspaces:span}{अवधि} एक है
\omterm{def:b1:vspaces:subspace}{उपस्पेस}. न्यूनतमता: चलो $ H $ कोई भी हो \omterm{def:b1:vspaces:subspace}{उपस्पेस} युक्त $ x_1,
\dots, x_p $. प्रत्येक अदिश गुणन के अंतर्गत स्थिरता द्वारा
$\lambda_i x_i \in H $, और उनके योग के तहत स्थिरता से
में निहित है $ H $: प्रत्येक रैखिक संयोजन का संबंध है $ H $, अर्थात\
$\operatorname{Vect}(x_1, \dots, x_p) \subseteq H $. इतना \omterm{def:b1:vspaces:span}{अवधि}
प्रत्येक में समाहित है \omterm{def:b1:vspaces:subspace}{उपस्पेस} परिवार युक्त: यह है
सबसे छोटा.
\end{proof}

\begin{definition}[योग, सीधा योग]\label{def:b1:vspaces:sum}
के लिए उपस्थान $ F, G $ का $ E $:
\[
F + G = \{\,u + v : u \in F,\ v \in G\,\}
\]
एक है \omterm{def:b1:vspaces:subspace}{उपस्पेस} (सबसे छोटा युक्त $ F \cup G $). योग है
\emph{प्रत्यक्ष}\index{प्रत्यक्ष}, लिखा हुआ $ F \oplus G $, जब हर
का तत्व $ F + G $ विघटित हो जाता है \emph{विशिष्ट} जैसा $ u + v $;
समकक्ष (नीचे देखें) जब $ F \cap G = \{0\}$. कब $ E = F \oplus
G $, द उपस्थान हैं \emph{पूरक}\index{पूरक} में $ E $.
\end{definition}

\begin{example}[दो पंक्तियों का योग]
\label{ex:b1:vspaces:sumoflines}
में $\R^3$, होने देना $ F = \operatorname{Vect}\bigl((1,0,1)\bigr)$ और
$ G = \operatorname{Vect}\bigl((0,1,1)\bigr)$. इनका योग है
\[
F + G = \{\,a(1,0,1) + b(0,1,1)\,\}
= \{(a,\ b,\ a + b)\} = \{(x, y, z) : z = x + y\},
\]
मूल बिंदु से गुजरने वाला वह तल जिसमें दोनों रेखाएँ हों। यह है
संघ से बिल्कुल बड़ा $ F \cup G $ (का मात्र क्रॉस
दो पंक्तियाँ): वेक्टर $(1, 1, 2) = (1,0,1) + (0,1,1)$ में निहित है
योग लेकिन किसी भी पंक्ति पर नहीं। और $ F \cap G = \{0\}$ (एक सामान्य
वेक्टर की आवश्यकता है $ a(1,0,1) = b(0,1,1)$, जिसके पहले दो
\omterm{prop:b1:vspaces:coordinates}{COORDINATES} बल $ a = b = 0$): योग \omterm{def:b1:vspaces:sum}{प्रत्यक्ष} है, और $ F \oplus
G $ बिल्कुल वही विमान है.
\end{example}

\begin{proposition}\label{prop:b1:vspaces:directsum}
$ F + G $ \omterm{def:b1:vspaces:sum}{प्रत्यक्ष} है यदि और केवल यदि $ F \cap G = \{0\}$.
\end{proposition}

\begin{proof}
यदि कुछ $ w \neq 0$ में निहित है $ F \cap G $: $ w = w + 0 = 0 + w $ दो हैं
का विघटन $ w $. इसके विपरीत, यदि $ u + v = u' + v'$ साथ $ u, u'
\in F $, $ v, v' \in G $, तब $ u - u' = v' - v $ से संबंधित $ F \cap G =
\{0\}$: अपघटन अद्वितीय हैं.
\end{proof}

\begin{method}[प्रमाणन $ E = F \oplus G $]
\label{met:b1:vspaces:directsum}
जांचने के लिए दो चीजें, प्रत्येक की अपनी मानक प्रारंभिक चाल के साथ।
\begin{enumerate}
  \item \emph{तुच्छ चौराहा.} लेना $ x \in F \cap G $,
        सदस्यता की दोनों शर्तें लिखें और निचोड़ें $ x =
        0$. (चित्र के \omterm{def:b1:vspaces:free}{आधार} पर कभी बहस न करें: cf.\ नीचे दिए गए नुकसान।)
  \item \emph{योग ही सब कुछ है.} एक मनमाना ले लो $ x \in E $
        और \emph{उत्पादन करना} विघटन $ x = f + g $ ---
        या तो अनुमान लगाकर $ f $ लक्ष्य से ($ f $ संतुष्ट होना चाहिए
        की परिभाषित संपत्ति $ F $, जो आमतौर पर इसे निर्देशित करता है
        सूत्र) या रैखिक प्रणाली को व्यक्त करके हल करके $ x $
        के जनरेटरों के विरुद्ध $ F $ और $ G $.
\end{enumerate}
जब अपघटन सूत्र का अनुमान लगाया जाता है, तो विशिष्टता होती है
चरण 1 से स्वचालित; जब केवल अस्तित्व ही अस्पष्ट हो, तो चरण 2 है
जहां काम रहता है. नीचे दिए गए दो उदाहरण विधि चलाते हैं: के लिए
सम/विषम कार्यों का सूत्र $ f $ मूल्यांकन करके बाध्य किया जाता है
पर वांछित पहचान $ x $ और $-x $; के लिए बहुआयामी पद लुप्त हो जाना
एक बिंदु पर, मूल्यांकन करके $ a $.
\end{method}

\begin{example}\label{ex:b1:vspaces:evenodd}
में $\mathcal{F}(\R, \R)$, सम कार्य करता है $\mathcal{P}$ और
अजीब कार्य $\mathcal{I}$ हैं \omterm{def:b1:vspaces:sum}{पूरक}: कोई $ f $ लिखते हैं
\[
f(x) = \underbrace{\frac{f(x) + f(-x)}{2}}_{\text{even}}
+ \underbrace{\frac{f(x) - f(-x)}{2}}_{\text{odd}},
\]
और एक फलन सम और विषम दोनों शून्य है। (के लिए आवेदन किया $\exp $, यह
जोड़ी है $(\cosh, \sinh)$ का \cref{ch:b1:functions}.)
\end{example}

\begin{example}[{$ K_n[X में एक पूरक जोड़ी]$}]
\label{ex:b1:vspaces:hyperplane}
हल करना $ a \in K $ और सेट करें $ F = \{P \in K_n[X] : P(a) = 0\}$, $ G =
\operatorname{Vect}(1)$ (स्थिरांक). तब $ K_n[X] = F \oplus
G $. वास्तव में $ F \cap G $ पर लुप्त होने वाले स्थिरांक शामिल हैं $ a $,
यानी\ $\{0\}$; और हर $ P $ के रूप में विघटित होता है
\[
P = \underbrace{\bigl(P - P(a)\bigr)}_{\in F}
+ \underbrace{P(a)}_{\in G} .
\]
अपघटन याद रखने योग्य है: a पर मान घटाना
बिंदु `` लुप्त हो रहे कार्यों पर प्रोजेक्ट करने का मानक तरीका है
$ a $''. ध्यान दें कि $ F $ एक बड़ा है \omterm{def:b1:vspaces:subspace}{उपस्पेस} और $ G $ एक छोटा सा; ए
\omterm{def:b1:vspaces:sum}{पूरक} जोड़ी को किसी भी दृष्टि से संतुलित करने की आवश्यकता नहीं है।
\end{example}

\begin{example}[एक पूरक उपस्थान कभी भी अद्वितीय नहीं होता है]
\label{ex:b1:vspaces:manysupplements}
में $\R^2$, होने देना $ F = \operatorname{Vect}\bigl((1,0)\bigr)$ (द
$ x $-अक्ष). दोनों $ G = \operatorname{Vect}\bigl((0,1)\bigr)$ और
$ G' = \operatorname{Vect}\bigl((1,1)\bigr)$ हैं \omterm{def:b1:vspaces:sum}{पूरक} को
$ F $: प्रत्येक मिलते हैं $ F $ केवल पर $0$, और प्रत्येक जोड़ी का योग है $\R^2$.
एक ही वेक्टर के अपघटन भिन्न होते हैं:
\[
(2,\ 1.5) = \underbrace{(2, 0)}_{\in F} +
\underbrace{(0, 1.5)}_{\in G}
= \underbrace{(0.5,\ 0)}_{\in F} +
\underbrace{(1.5,\ 1.5)}_{\in G'} .
\]
वास्तव में \emph{प्रत्येक} लाइन के अलावा अन्य $ F $ स्वयं एक है
\omterm{def:b1:vspaces:sum}{पूरक} का $ F $ में $\R^2$: \omterm{def:b1:vspaces:sum}{पूरक} प्रचुर मात्रा में हैं,
और ``द'' की बात कर रहे हैं \omterm{def:b1:vspaces:sum}{पूरक} जब तक कोई अर्थहीन न हो
अतिरिक्त संरचना (एक आंतरिक उत्पाद, \cref{ch:b1:euclid}) एकल
एक बाहर.
\end{example}

\section{मुक्त परिवार, उत्पादक परिवार, आधार}

\begin{definition}\label{def:b1:vspaces:free}
एक परिवार $(x_1, \dots, x_p)$ के सदिशों का $ E $ है:
\begin{itemize}
  \item \emph{उत्पादक} (का $ E $) कब
        $\operatorname{Vect}(x_1,\dots,x_p) = E $;
  \item \emph{मुक्त}\index{मुक्त} (इसके वेक्टर \emph{रैखिक रूप से स्वतंत्र}) कब
        \[
        \lambda_1 x_1 + \dots + \lambda_p x_p = 0
        \implies \lambda_1 = \dots = \lambda_p = 0 ;
        \]
        अन्यथा \emph{जुड़ा हुआ};
  \item ए \emph{आधार}\index{आधार} जब यह मुफ़्त और जनरेटिंग हो।
\end{itemize}
\end{definition}

\begin{proposition}[COORDINATES]\label{prop:b1:vspaces:coordinates}
$(e_1, \dots, e_n)$ एक है \omterm{def:b1:vspaces:free}{आधार} का $ E $ यदि और केवल यदि प्रत्येक $ x \in E $
है \emph{विशिष्ट} एक संयोजन $ x = \lambda_1 e_1 + \dots +
\lambda_n e_n $; अदिश $\lambda_i $ हैं
\emph{COORDINATES}\index{COORDINATES} का $ x $ में \omterm{def:b1:vspaces:free}{आधार}.
\end{proposition}

\begin{proof}
उत्पादक $=$ विघटन का अस्तित्व. विशिष्टता $=$
मुक्तता: एक ही के दो विघटन $ x $ एक संयोजन से भिन्न
के बराबर $0$; मुक्तता अपने सभी गुणांकों को बाध्य करती है ---
के मतभेद \omterm{prop:b1:vspaces:coordinates}{COORDINATES} --- गायब होना। इसके विपरीत, एक गैरतुच्छ
शून्य संयोजन दो विघटन देता है $0 = \sum \lambda_i
e_i = \sum 0\,e_i $.
\end{proof}

\begin{example}\label{ex:b1:vspaces:bases}
 \emph{कैनन का \omterm{def:b1:vspaces:free}{आधार}} का $ K^n $: $ e_i = (0, \dots, 1, \dots, 0)$
($1$ स्लॉट में $ i $). एकपदी $(1, X, X^2, \dots, X^n)$: ए \omterm{def:b1:vspaces:free}{आधार}
का $ K_n[X]$ (स्वतंत्रता: एक शून्य संयोजन शून्य है \omterm{def:b1:poly:def}{बहुपद}, तो
सभी गुणांक गायब हो जाते हैं, \cref{def:b1:poly:def}). में
$\C $ ऊपर $\R $: द \omterm{def:b1:vspaces:free}{आधार} $(1, \iu)$.
\end{example}

\begin{remark}[निर्देशांक एक टीम प्रयास है]
\label{rem:b1:vspaces:teameffort}
का पहला समन्वय $ x $ में एक \omterm{def:b1:vspaces:free}{आधार} $(e_1, \dots, e_n)$
पर निर्भर करता है \emph{सभी}  \omterm{def:b1:vspaces:free}{आधार} वैक्टर, बस नहीं $ e_1$. में
$\R^2$: वेक्टर $(3, 1)$ पहले समन्वय है $3$ में
विहित \omterm{def:b1:vspaces:free}{आधार}, लेकिन पहले समन्वय करें $2$ में \omterm{def:b1:vspaces:free}{आधार}
$\bigl((1,0), (1,1)\bigr)$ --- हल करना $(3,1) = a(1,0) + b(1,1)$:
$ b = 1$, $ a = 2$. एक बदल रहा हूँ \omterm{def:b1:vspaces:free}{आधार} वेक्टर फेरबदल
\emph{प्रत्येक} समन्वय; \cref{ch:b1:matrices} इसे पैकेज करेंगे
परिवर्तन-के-आधार मैट्रिक्स में फेरबदल।
\end{remark}

\begin{example}[उम्मीदवार के आधार पर परीक्षण, प्रारंभ से अंत तक]
\label{ex:b1:vspaces:testbasis}
है $\mathcal{F} = (1 + X,\ 1 + X^2,\ X + X^2)$ ए \omterm{def:b1:vspaces:free}{आधार} का
$\R_2[X]$? लिखना $ u_1, u_2, u_3$ तीन के लिए बहुआयामी पद.
\emph{उन्मुक्तता}: एक शून्य संयोजन $ a\,u_1 + b\,u_2 + c\,u_3 =
0$ देता है, गुणांक दर गुणांक,
\[
a + b = 0, \qquad a + c = 0, \qquad b + c = 0 ;
\]
पहले दो को घटाने पर, $ b = c $, फिर तीसरा देता है $2b =
0$: $ a = b = c = 0$, \omterm{def:b1:vspaces:free}{मुक्त}. \emph{उत्पादक}: समाधान करने के बजाय
तीन प्रणालियाँ, सममित संयोजन पर ध्यान दें
\[
u_1 + u_2 - u_3 = (1 + X) + (1 + X^2) - (X + X^2) = 2 ,
\]
इसलिए $1 = \frac12(u_1 + u_2 - u_3)$; तब
\[
X = u_1 - 1 = \tfrac12\bigl(u_1 - u_2 + u_3\bigr),
\qquad
X^2 = u_2 - 1 = \tfrac12\bigl(-u_1 + u_2 + u_3\bigr).
\]
मोनोमियल्स में स्थित हैं \omterm{def:b1:vspaces:span}{अवधि}, तो सब कुछ करता है: $\mathcal{F}$
एक है \omterm{def:b1:vspaces:free}{आधार}. एक बोनस के रूप में, तीन डिस्प्ले को असेंबल करने से मिलता है
\omterm{prop:b1:vspaces:coordinates}{COORDINATES} किसी का $ P = \alpha + \beta X + \gamma X^2$:
\[
P = \frac{\alpha + \beta - \gamma}{2}\,u_1
+ \frac{\alpha - \beta + \gamma}{2}\,u_2
+ \frac{-\alpha + \beta + \gamma}{2}\,u_3 .
\]
(स्वच्छता की जाँच करें $ P = X $: \omterm{prop:b1:vspaces:coordinates}{COORDINATES} $\bigl(\frac12,
-\frac12, \frac12\bigr)$, जैसा कि ऊपर पाया गया।) दो पाठ: समरूपता
परिवार में आमतौर पर एक शॉर्टकट संयोजन छिपा होता है; और एक बार
आयाम उपलब्ध है (\cref{ch:b1:findim}), संपूर्ण
इस काम का आधा हिस्सा जनरेट करने में आएगा \omterm{def:b1:vspaces:free}{मुक्त} प्रभारी --- तीन
\omterm{def:b1:vspaces:free}{मुक्त} ए के वैक्टर $3$-आयामी स्थान हमेशा एक बनता है \omterm{def:b1:vspaces:free}{आधार}.
\end{example}

\begin{proposition}[उपयोगी स्वतंत्रता मानदंड]\label{prop:b1:vspaces:freecriteria}
\begin{enumerate}
  \item का एक परिवार \emph{अशून्य बहुआयामी पद जोड़ीवार अलग-अलग
        डिग्री} है \omterm{def:b1:vspaces:free}{मुक्त}.
  \item a में एक वेक्टर जोड़ना \omterm{def:b1:vspaces:free}{मुक्त} परिवार रखता है \omterm{def:b1:vspaces:free}{मुक्त} यदि और केवल
        यदि वेक्टर बाहर है \omterm{def:b1:vspaces:span}{अवधि} परिवार की।
  \item ए का कोई भी उपपरिवार \omterm{def:b1:vspaces:free}{मुक्त} परिवार है \omterm{def:b1:vspaces:free}{मुक्त}; कोई भी परिवार
        जिसमें एक उत्पादक परिवार शामिल है, वह उत्पन्न कर रहा है।
\end{enumerate}
\end{proposition}

\begin{proof}
(1) अशक्त संयोजन में, उच्चतम डिग्री वर्तमान को देखें: इसका
गुणांक लुप्त हो जाना चाहिए (कुछ भी उस डिग्री को रद्द नहीं करता), और कैस्केड होना चाहिए
नीचे.

(2) यदि $ x \in \operatorname{Vect}(x_1, \dots, x_p)$, संबंध $ x
- \sum\lambda_i x_i = 0$ गैरतुच्छ है. इसके विपरीत, एक गैरतुच्छ
का शून्य संयोजन $(x_1, \dots, x_p, x)$ शामिल होना चाहिए $ x $ एक के साथ
शून्येतर गुणांक (अन्यथा यह छोटे की मुक्तता का खंडन करता है
परिवार), और समाधान के लिए $ x $ में डालता है \omterm{def:b1:vspaces:span}{अवधि}.

(3) उपपरिवार: उपपरिवार का एक अशक्त संयोजन एक है
लुप्त गुणांक वाला पूरा परिवार \omterm{def:b1:logic:sets}{तय करना} को $0$; उन्मुक्तता
बड़े परिवार के लोग उन सभी को मार देते हैं। सुपरफ़ैमिली: प्रत्येक वेक्टर
का $ E $ पहले से ही उत्पादक भाग का एक संयोजन है; दे दो
अतिरिक्त वैक्टर गुणांक $0$.
\end{proof}

\begin{example}[सीढ़ी सिद्धांत]
\label{ex:b1:vspaces:staircase}
होने देना $ P_0, P_1, \dots, P_n \in K_n[X]$ साथ $\deg P_k = k $ प्रत्येक के लिए
$ k $ (डिग्री की एक ``सीढ़ी'')। तब $(P_0, \dots, P_n)$ एक है
\omterm{def:b1:vspaces:free}{आधार} का $ K_n[X]$. उन्मुक्तता है
\cref{prop:b1:vspaces:freecriteria}~(1). पैदा करने के लिए
संपत्ति, डिग्री पर परिमित वंश द्वारा तर्क: चलो $ Q \in
K_n[X]$, $ Q \neq 0$, डिग्री का $ d $, अग्रणी गुणांक के साथ $ a $,
और चलो $ b \neq 0$ का अग्रणी गुणांक हो $ P_d $. तब $ Q -
\frac ab P_d $ की डिग्री है $< d $ (शीर्ष शर्तें रद्द करें); की जगह
$ Q $ इस अंतर और पुनरावृत्ति से, अधिक से अधिक के बाद $ n + 1$ कदम
कोई शून्य पर पहुंच जाता है \omterm{def:b1:poly:def}{बहुपद}, और घटाव को खोलना
अभिव्यक्त करता है $ Q $ के संयोजन के रूप में $ P_k $. दो सीढ़ियाँ
पहले ही मिल चुके हैं: स्थानांतरित शक्तियां $\bigl((X-a)^k\bigr)_{0 \leq k
\leq n}$ (\cref{exo:b1:vspaces:4}), और न्यूटन उत्पाद
$\bigl((X - x_0)(X - x_1)\cdots(X - x_{k-1})\bigr)_{0 \leq k \leq
n}$, सप्ताहांत की समस्या में काम पर लगाओ।
\end{example}

\begin{example}[कार्य स्थानों में स्वतंत्रता]\label{ex:b1:vspaces:functionfree}
में $\mathcal{F}(\R,\R)$, परिवार $(\eu^{a_1 x}, \dots, \eu^{a_p
x})$ साथ $ a_1 < \dots < a_p $ है \omterm{def:b1:vspaces:free}{मुक्त}: एक शून्य संयोजन को इससे विभाजित करें
$\eu^{a_p x}$ और चलो $ x \to +\infty $; अंतिम गुणांक मर जाता है,
और एक झरना नीचे गिरता है (\cref{exo:b1:vspaces:8} इसका विवरण और
वेरिएंट)। कार्यों की स्वतन्त्रता सिद्ध होती है \emph{का मूल्यांकन}:
अच्छी तरह से चुने गए बिंदुओं पर, अनंत पर, या विभेदन के बाद।
\end{example}

\begin{example}[एक छिपा हुआ रिश्ता एक अवधि को छोटा कर देता है]
\label{ex:b1:vspaces:hiddenrelation}
में $\mathcal{F}(\R, \R)$, क्या है
$\operatorname{Vect}(1,\ \cos^2,\ \sin^2)$? पहचान
$\cos^2 + \sin^2 = 1$ एक गैरतुच्छ शून्य संयोजन है
\[
1\cdot\mathbf{1} + (-1)\cos^2 + (-1)\sin^2 = 0 :
\]
परिवार जुड़ा हुआ है, और \omterm{def:b1:vspaces:span}{अवधि} द्वारा पहले ही उत्पन्न किया जा चुका है
$(1, \cos^2)$ अकेला ($\sin^2 = 1 - \cos^2$). वो छोटा सा परिवार
है \omterm{def:b1:vspaces:free}{मुक्त}: $ a + b\cos^2 x = 0$ सभी के लिए $ x $ देता है, पर $ x = 0$ और
$ x = \frac\pi2$: $ a + b = 0$ और $ a = 0$. इतना \omterm{def:b1:vspaces:span}{अवधि} एक है
\emph{विमान} फ़ंक्शन स्पेस के अंदर --- और इसमें यह भी शामिल है
$\cos 2x = 2\cos^2 x - 1$: रैखिक दिखने वाले परिवार
त्रिकोणमितीय कार्य नियमित रूप से पहचान के अंतर्गत ढह जाते हैं,
इसीलिए स्वतंत्रता अवश्य होनी चाहिए \emph{साबित}, से कभी नहीं माना गया
सूची की लंबाई.
\end{example}

\begin{remark}[सामान्य ख़तरे]
\label{rem:b1:vspaces:pitfalls}
चार शास्त्रीय जाल.
\emph{जोड़ीवार पर्याप्त नहीं है}: में $\R^2$, वैक्टर $(1,0)$,
$(0,1)$, $(1,1)$ जोड़ीवार गैर-आनुपातिक हैं, फिर भी जुड़े हुए हैं ---
उन्मुक्तता का गुण है \emph{साबुत} परिवार, एक द्वारा परीक्षण किया गया
वैश्विक संयोजन, कभी भी दो बटा दो नहीं।
\emph{शून्य वेक्टर हर चीज में जहर घोल देता है}: कोई भी परिवार युक्त
$0$ जुड़ा हुआ है ($1\cdot 0 = 0$ हालाँकि, यह एक गैर-तुच्छ संबंध है)।
अन्य वैक्टर निर्दोष।
\emph{मिलन योग नहीं है}: $ F \cup G $ लगभग कभी नहीं है \omterm{def:b1:vspaces:subspace}{उपस्पेस}
(\cref{def:b1:vspaces:subspace}); सबसे छोटा \omterm{def:b1:vspaces:subspace}{उपस्पेस}
दोनों से युक्त है $ F + G $, आमतौर पर संघ से बहुत बड़ा
--- में $\R^2$, दो अलग-अलग रेखाओं का मिलन एक क्रॉस है, योग करें
पूरा विमान.
\emph{\omterm{def:b1:vspaces:sum}{प्रत्यक्ष} के लिए तुच्छ प्रतिच्छेदन की आवश्यकता होती है, असम्बद्धता की नहीं}:
दो उपस्थान कभी भी असंयुक्त नहीं होते (दोनों समाहित होते हैं $0$); सही
शर्त है $ F \cap G = \{0\}$, और यह होना ही चाहिए \emph{साबित},
किसी चित्र को न पढ़ें --- cf.\
\cref{ex:b1:vspaces:manysupplements}, जहां कई अलग-अलग हैं $ G $
काम।
\emph{स्वतंत्रता अदिश राशि पर निर्भर करती है}: जोड़ी $(1, \iu)$ है
\omterm{def:b1:vspaces:free}{मुक्त} में $\C $ एक के रूप में देखा जाता है $\R $-वेक्टर स्पेस, लेकिन लिंक्ड इन $\C $
के रूप में देखा जाता है $\C $-सदिश स्थल ($\iu\cdot 1 + (-1)\cdot\iu = 0$).
हमेशा जानें कौन सा \omterm{def:b1:structures:field}{मैदान} परिवार घोषित करने से पहले कार्य कर रहा है \omterm{def:b1:vspaces:free}{मुक्त}
--- की सप्ताहांत समस्या \cref{ch:b1:findim} बिल्कुल बदल जाता है
अतार्किकता प्रमाणों में यह संवेदनशीलता।
\end{remark}

\begin{remark}[ये भाषा कहां जाती है]
\label{rem:b1:vspaces:whereused}
इस अध्याय के बाद की हर चीज़ भाषा बोलती है \omterm{def:b1:logic:sets}{तय करना} यहाँ ऊपर.
\cref{ch:b1:findim} गिनता \omterm{def:b1:vspaces:free}{आधार} वैक्टर और मोड़ ``\omterm{def:b1:vspaces:free}{मुक्त}'' और
एक पूर्णांक पर असमानताओं को ``उत्पन्न करना''
आयाम. \cref{ch:b1:linmaps} का अध्ययन करता है एमएपीएस के साथ संगत
दो ऑपरेशन; \omterm{def:b1:vspaces:sum}{प्रत्यक्ष} रकम वहां प्रोजेक्टर बनें.
\cref{ch:b1:matrices} वेक्टरों को उनके द्वारा एन्कोड करता है \omterm{prop:b1:vspaces:coordinates}{COORDINATES} में एक
\omterm{def:b1:vspaces:free}{आधार} --- \cref{prop:b1:vspaces:coordinates} का लाइसेंस है
वह एन्कोडिंग --- और \cref{ch:b1:euclid} लंबाई और कोण जोड़ता है
रैखिक संरचना के शीर्ष पर. वर्ष~2 खंड में वही
स्वयंसिद्ध, शब्दशः, मनमाने ढंग से चलते हैं खेत और अनंत में
आयाम; इस अध्याय में कहीं भी परिमितता का प्रयोग नहीं किया गया है।
\end{remark}

\begin{remark}[पुस्तक 3 के माध्यम से अनुसरण करने योग्य तीन सूत्र]
\label{rem:b1:vspaces:threads}
इस अध्याय के तीन विशिष्ट विचारों को विकसित होते हुए देखें।
\emph{सीढ़ी सिद्धांत}
(\cref{ex:b1:vspaces:staircase}) न्यूटन के रूप में पुनः प्रकट होता है \omterm{def:b1:vspaces:free}{आधार}
इस अध्याय की सप्ताहांत समस्या में, द्विपद के रूप में \omterm{def:b1:vspaces:free}{आधार}
$(B_k)$ वहाँ, और के रूप में \omterm{def:b1:poly:def}{बहुपद} में वैकल्पिक चाल
सप्ताहांत की समस्या \cref{ch:b1:det}: एक लेम्मा, तीन
निर्धा\omterm{def:b1:vspaces:free}{रक-मुक्त} लाभांश
\emph{एक स्वतंत्रता परीक्षण के रूप में मूल्यांकन}
(\cref{ex:b1:vspaces:functionfree}) अंतर्वेशन बन जाता है
की समरूपता \cref{ch:b1:linmaps}, फिर वेंडरमोंडे
की कसौटी \cref{ch:b1:det}, फिर ग्राम परीक्षण
\cref{ch:b1:euclid}: एक ही पलटा, तीन बार तेज।
\emph{\omterm{def:b1:vspaces:sum}{प्रत्यक्ष} रकम} (\cref{def:b1:vspaces:sum}) बन गया
प्रोजेक्टर में \cref{ch:b1:linmaps}, ऑर्थोगोनल विभाजन $ E =
F \oplus F^\perp $ में \cref{ch:b1:euclid}, और
में कम से कम वर्गों की व्याख्या-प्लस-अवशिष्ट अपघटन
सप्ताहांत की समस्या \cref{ch:b1:multivar}. यह बहुत कम है
पुस्तक, नीचे, इन तीन विचारों में से एक भी नई नहीं है
कपड़े.
\end{remark}

\begin{exercise}[$\star\star $]\label{exo:b1:vspaces:7}

और एक उदाहरण द्वारा दिखाएँ $\R^2$ वह अप्रतिबंधित

\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b1:vspaces:8}

\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b1:vspaces:9}
होने देना $ E $ एक हो $ K $-वेक्टर स्पेस और $ F, G, H $ उपस्थान साथ $ F + G =

परिकल्पना के बिना एक प्रतिउदाहरण दीजिए $ G \subseteq H $.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:vspaces:10}
में $\R[X]$, होने देना $\mathcal P $ हो \omterm{def:b1:logic:sets}{तय करना} का \emph{यहां तक की} बहुआयामी पद

\begin{exercise}[$\star\star $]\label{exo:b1:vspaces:11}
होने देना $(x_1, x_2, x_3)$ एक हो \omterm{def:b1:vspaces:free}{मुक्त} परिवार एक असली का सदिश स्थल $ E $.

\begin{exercise}[$\star\star\star $]\label{exo:b1:vspaces:12}
होने देना $ E $ एक हो सदिश स्थल ऊपर $\R $ (या $\C $) और $ F_1, \dots,
F_k $ \emph{उचित} उपस्थान का $ E $ (प्रत्येक $ F_i \neq E $).
\begin{enumerate}
  \item मामले का इलाज करें $ k = 2$ सीधे: यदि $ F_1 \not\subseteq
        F_2$ और $ F_2 \not\subseteq F_1$, चुनना $ x \in F_1
        \setminus F_2$ और $ y \in F_2 \setminus F_1$ और पता लगाएं $ x
        + y $.
  \item इसे सामान्य रूप से सिद्ध करें $ E \neq F_1 \cup \dots \cup F_k $: ए
        सदिश स्थल एक अनंत के ऊपर \omterm{def:b1:structures:field}{मैदान} कभी भी सीमित नहीं होता

        $ x \in F_1$ दूसरे के बाहर $ F_i $, चुनना $ y \notin F_1$,
        और लाइन का पालन करें $ t \mapsto y + tx $.)}
\end{enumerate}
\end{exercise}

\section{समस्या: प्रक्षेप, एक स्थान के लिए तीन आधार}

\begin{problem}\label{pb:b1:vspaces:1}
हल करना $ n + 1$ \emph{विशिष्ट} अंक $ x_0, x_1, \dots, x_n $ का $\R $.
यह समस्या दोबारा सामने आती है \omterm{thm:b1:poly:lagrange}{लैग्रेंज इंटरपोलेशन}
(\cref{thm:b1:poly:lagrange}) इस अध्याय की दृष्टि से: द
अंतरिक्ष $\R_n[X]$ तीन प्राकृतिक \omterm{def:b1:vspaces:free}{आधार} हैं --- लैग्रेंज,
न्यूटन, और (समान दूरी वाले बिंदुओं के लिए) द्विपद \omterm{def:b1:vspaces:free}{आधार} ---
और प्रत्येक \omterm{def:b1:vspaces:free}{आधार} एक प्रश्न को आसान बनाता है. सड़क वास्तविक पर समाप्त होती है
अंकगणित प्रमेय: पी\'ओलेआ का लक्षण वर्णन बहुआयामी पद
मानचित्रण $\Z $ में $\Z $.\index{लैग्रेंज इंटरपोलेशन}
\index{विभाजित मतभेद}\index{पूर्णांक-मूल्यवान बहुपद}

\medskip
\noindent\textbf{Part I --- The Lagrange \omterm{def:b1:vspaces:free}{आधार}.} के लिए $0 \leq i
\leq n $ \omterm{def:b1:logic:sets}{तय करना}
\[
L_i \;=\; \prod_{j \neq i} \frac{X - x_j}{x_i - x_j} \;\in\;
\R_n[X].
\]
\begin{enumerate}
  \item जाँच करें कि $\deg L_i = n $ ओर वो $ L_i(x_j) = 1$ अगर $ j =
        i $, और $0$ अगर $ j \neq i $.
  \item साबित करो कि परिवार $(L_0, \dots, L_n)$ है \omterm{def:b1:vspaces:free}{मुक्त}.
  \item इसे हर एक के लिए साबित करें $ P \in \R_n[X]$,
        \[
        P \;=\; \sum_{i=0}^{n} P(x_i)\, L_i ,
        \]
        और निष्कर्ष निकालो $(L_0, \dots, L_n)$ एक है \omterm{def:b1:vspaces:free}{आधार} का
        $\R_n[X]$. \emph{(दोनों पक्षों के अंतर पर विचार करें
        और इसकी जड़ें गिनें, \cref{cor:b1:poly:nroots}.)}
  \item प्रक्षेप प्रमेय निकालें: किसी भी मान के लिए $ y_0,
        \dots, y_n \in \R $ वहां एक है \emph{अद्वितीय} $ P \in
        \R_n[X]$ साथ $ P(x_i) = y_i $ सभी के लिए $ i $. लैग्रेंज में
        \omterm{def:b1:vspaces:free}{आधार}, क्या है \omterm{prop:b1:vspaces:coordinates}{COORDINATES} एक का \omterm{def:b1:poly:def}{बहुपद} $ P $?
  \item पहचान साबित करें
        \[
        \sum_{i=0}^{n} L_i = 1
        \qquad\text{and, for } 0 \leq k \leq n,\qquad
        \sum_{i=0}^{n} x_i^{k}\, L_i = X^{k} .
        \]
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The Newton \omterm{def:b1:vspaces:free}{आधार} and \omterm{pb:b1:vspaces:1}{विभाजित
मतभेद}.} \omterm{def:b1:logic:sets}{तय करना} $ N_0 = 1$ और $ N_k = (X - x_0)(X - x_1) \cdots
(X - x_{k-1})$ के लिए $1 \leq k \leq n $. एक समारोह के लिए $ f $ परिभाषित
नोड्स पर, परिभाषित करें \emph{\omterm{pb:b1:vspaces:1}{विभाजित मतभेद}} द्वारा
$ f[x_i] = f(x_i)$ और
\[
f[x_i, \dots, x_{i+k}] \;=\;
\frac{f[x_{i+1}, \dots, x_{i+k}] - f[x_i, \dots, x_{i+k-1}]}
{x_{i+k} - x_i} .
\]
\begin{enumerate}[resume]
  \item साबित करें कि $(N_0, N_1, \dots, N_n)$ एक है \omterm{def:b1:vspaces:free}{आधार} का
        $\R_n[X]$.
  \item गणना करना $ f[x_0, x_1]$ और $ f[x_0, x_1, x_2]$ के संदर्भ में
        के मूल्य $ f $, फिर सभी की गणना करें \omterm{pb:b1:vspaces:1}{विभाजित मतभेद} का
        $ f(x) = x^2$ तीन मनमाने नोड्स पर।
  \item (एटकेन की लेम्मा) चलो $ R $ बैठाना $ f $ पर $ x_0, \dots,
        x_{n-1}$ और $ Q $ बैठाना $ f $ पर $ x_1, \dots, x_n $,
        दोनों डिग्री के $\leq n - 1$. साबित करें कि
        \[
        S \;=\; \frac{(X - x_0)\,Q - (X - x_n)\,R}{x_n - x_0}
        \]
        प्रक्षेप करता है $ f $ पर $ x_0, x_1, \dots, x_n $.
  \item नोड्स की संख्या पर प्रेरण द्वारा, निष्कर्ष निकालें कि
        का गुणांक $ X^{k}$ के इंटरपोलेंट में $ f $ पर $ x_0,
        \dots, x_k $ बिलकुल है $ f[x_0, \dots, x_k]$.
  \item सिद्ध करना \emph{न्यूटन का प्रक्षेप सूत्र}: द
        का इंटरपोलेंट $ f $ पर $ x_0, \dots, x_n $ है
        \[
        P \;=\; \sum_{k=0}^{n} f[x_0, \dots, x_k]\, N_k ,
        \]
        और \omterm{def:b1:topology:closure}{बंद} सूत्र प्राप्त करें
        \[
        f[x_0, \dots, x_k] \;=\; \sum_{i=0}^{k}
        \frac{f(x_i)}{\prod_{j \neq i,\, j \leq k} (x_i - x_j)} ,
        \]
        जो यह दर्शाता है $ f[x_0, \dots, x_k]$ पर निर्भर नहीं है
        नोड्स का क्रम.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Equally spaced nodes: the difference
operator.} अब से नोड्स हैं $0, 1, 2, \dots $ और, एक के लिए
\omterm{def:b1:poly:def}{बहुपद} $ P $, हमलोग तैयार हैं
\[
\Delta P(X) = P(X + 1) - P(X),
\qquad
B_k = \frac{X(X-1)\cdots(X-k+1)}{k!} \quad (B_0 = 1).
\]
\begin{enumerate}[resume]
  \item दिखाओ अगर $\deg P = m \geq 1$ अग्रणी गुणांक के साथ
        $ a $, तब $\deg \Delta P = m - 1$ अग्रणी गुणांक के साथ
        $ m\,a $, ओर वो $\Delta $ स्थिरांक को मारता है.
  \item बताते हैं कि $(B_0, B_1, \dots, B_n)$ एक है \omterm{def:b1:vspaces:free}{आधार} का
        $\R_n[X]$ ओर वो $\Delta B_k = B_{k-1}$ के लिए $ k \geq 1$.
  \item (न्यूटन का अग्र-अंतर सूत्र) सिद्ध करें कि प्रत्येक $ P
        \in \R_n[X]$ संतुष्ट
        \[
        P \;=\; \sum_{k=0}^{n} \bigl(\Delta^{k} P\bigr)(0)\, B_k .
        \]
  \item इसे हर एक के लिए साबित करें $ k \geq 0$,
        \[
        \bigl(\Delta^{k} P\bigr)(0) \;=\;
        \sum_{j=0}^{k} (-1)^{k-j} \binom{k}{j} P(j) .
        \]
  \item दिखाओ अगर $\deg P = n $ अग्रणी गुणांक के साथ $ a_n $,
        तब $\Delta^{n} P $ स्थिरांक है $ n!\,a_n $ और
        $\Delta^{n+1} P = 0$.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- \omterm{pb:b1:vspaces:1}{पूर्णांक-मूल्यवान बहुपद}.} ए
\omterm{def:b1:poly:def}{बहुपद} $ P \in \R[X]$ है \emph{पूर्णांक-मूल्यवान} कब $ P(m) \in
\Z $ हरएक के लिए $ m \in \Z $.
\begin{enumerate}[resume]
  \item प्रत्येक को सिद्ध करें $ B_k $ पूर्णांक-मूल्यवान है. \emph{(इलाज $ m
        \geq k $, $0 \leq m < k $ और $ m < 0$ अलग से; के लिए $ m =
        -q < 0$, दिखाओ $ B_k(-q) = (-1)^k \binom{q + k - 1}{k}$.)}
  \item सिद्ध करना \emph{पी\'ओलेआ का लक्षण वर्णन}: $ P \in \R_n[X]$
        पूर्णांक-मूल्यवान है यदि और केवल यदि यह \omterm{prop:b1:vspaces:coordinates}{COORDINATES} में
        \omterm{def:b1:vspaces:free}{आधार} $(B_0, \dots, B_n)$ पूर्णांक हैं.
  \item निष्कर्ष निकालें: यदि $ P \in \R_n[X]$ पर पूर्णांक मान लेता है $ n + 1$
        \emph{लगातार} पूर्णांकों $ a, a+1, \dots, a+n $, तब $ P $
        पूर्णांक-मूल्यवान है. \emph{(शिफ्ट: अध्ययन को लागू करें $ Q(X)
        = P(X + a)$.)}
  \item प्रश्न 16 से यह निष्कर्ष निकालें कि का एक गुणनफल है $ k $ लगातार
        पूर्णांक हमेशा से विभाज्य होता है $ k!$.
  \item होने देना $ P = \dfrac{X(X+1)(2X+1)}{6}$. इसके न्यूटन की गणना कीजिए
        टेबल पर $0, 1, 2, 3$, लिखना $ P $ में \omterm{def:b1:vspaces:free}{आधार} $(B_k)$, और
        यह निष्कर्ष निकालें $ P $ पूर्णांक-मूल्यवान है, हालांकि इसका कोई भी नहीं
        एकपदी गुणांक एक पूर्णांक है। सत्यापित करें $\Delta P =
        (X+1)^2$ और निष्कर्ष निकालें $ P(m) = 1^2 + 2^2 + \dots + m^2$ के लिए
        $ m \in \N $.
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Dividends.}
\begin{enumerate}[resume]
  \item होने देना $ P \in \R_n[X]$ मूल्यों को प्रक्षेपित करें $2^i $ पर $ i =
        0, 1, \dots, n $. बताते हैं कि $ P = B_0 + B_1 + \dots + B_n $
        ओर वो $ P(n + 1) = 2^{n+1} - 1$: ``दोहरीकरण
        pattern'' always breaks at the very next point.
  \item (असतत प्रतिअवकलन) इसे सभी पूर्णांकों के लिए सिद्ध करें $ m
        \geq 1$ और $ k \geq 0$,
        \[
        \sum_{j=0}^{m-1} B_k(j) \;=\; B_{k+1}(m),
        \]
        यानी\ हॉकी-स्टिक की पहचान $\sum_{j=k}^{m-1}
        \binom{j}{k} = \binom{m}{k+1}$.
  \item बढ़ाना $ X^2$ और $ X^3$ में \omterm{def:b1:vspaces:free}{आधार} $(B_k)$ और निष्कर्ष निकालें
        के लिए \omterm{def:b1:topology:closure}{बंद} सूत्र $\sum_{j=0}^{m-1} j^2$ और
        $\sum_{j=0}^{m-1} j^3$; निकोमाचस की पहचान पुनः प्राप्त करें
        $1^3 + \dots + m^3 = (1 + \dots + m)^2$.
  \item लेना $ n = 2$ और नोड्स $0, 1, 2$. लिखना \omterm{prop:b1:vspaces:coordinates}{COORDINATES} का
        $ X^2$ इस समस्या के तीन \omterm{def:b1:vspaces:free}{आधार}ों में: एकपदी
        \omterm{def:b1:vspaces:free}{आधार}, लैग्रेंज \omterm{def:b1:vspaces:free}{आधार}, न्यूटन \omterm{def:b1:vspaces:free}{आधार}. की जाँच करें
        प्रश्न 4 और 9 के सामने तीन उत्तर।
  \item संश्लेषण। चार वाक्यों में: कौन सा वेक्टर-स्पेस अवधारणा
        प्रश्न 4 को स्वचालित बनाता है; न्यूटन क्यों? \omterm{def:b1:vspaces:free}{आधार} गणना करता है
        \omterm{prop:b1:vspaces:coordinates}{COORDINATES} \emph{रिकर्सिवली} जबकि लैग्रेंज \omterm{def:b1:vspaces:free}{आधार}
        उन्हें पढ़ता है \emph{तुरन्त}; कौन सी स्वतंत्रता की कसौटी
        दोनों \omterm{def:b1:vspaces:free}{आधार} साझा करते हैं; और किस सटीक अर्थ में पी\'ओलेया का
        theorem says that integrality of a \omterm{def:b1:poly:def}{बहुपद} एक है
        इसकी संपत्ति \omterm{prop:b1:vspaces:coordinates}{COORDINATES} \emph{दाईं ओर \omterm{def:b1:vspaces:free}{आधार}}.
\end{enumerate}
\end{problem}
