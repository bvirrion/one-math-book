\chapter{एक खंड पर एकीकरण}\label{ch:b1:integration}

 \omterm{thm:b1:integration:def}{अभिन्न} हाई स्कूल खंड की स्थापना लिए गए क्षेत्रों पर की गई थी
सहजता से। यह अध्याय इसका निर्माण करता है: सबसे पहले चरण कार्य,
जहां \omterm{thm:b1:integration:def}{अभिन्न} तो यह एक सीमित योग है निरंतर (और
टुकड़े-टुकड़े में निरंतर) एकसमान सन्निकटन द्वारा कार्य ---
वह स्थान जहाँ हेन की प्रमेय (\cref{thm:b1:continuity:heine}) कमाता है
यह रखना है. कैलकुलस का मौलिक प्रमेय फिर जोड़ता है
आदिम के लिए निर्माण, और \omterm{thm:b1:integration:riemann}{रीमैन रकम} इसे अलग से कनेक्ट करें
औसत.

लगातार, $a < b $ असली हैं.

\section{चरण कार्य}

\begin{definition}\label{def:b1:integration:step}
$\varphi \colon \intcc{a}{b} \to \R $ एक है \emph{समारोह की ओर कदम बढ़ाएं}\index{समारोह की ओर कदम बढ़ाएं} जब कोई उपविभाजन हो $ a = x_0
< x_1 < \dots < x_n = b $ ऐसा है कि $\varphi $ स्थिर है, बराबर है
$ c_i $, सभी के ऊपर \omterm{def:b1:topology:open}{खुला} \omterm{prop:b1:reals:intervals}{अंतराल} $\intoo{x_{i-1}}{x_i}$ (मूल्य पर
नोड्स अप्रतिबंधित हैं)। इसका \omterm{thm:b1:integration:def}{अभिन्न} है
\[
\int_a^b \varphi = \sum_{i=1}^{n} c_i\,(x_i - x_{i-1}),
\]
चुने गए उपखंड से स्वतंत्र (दो उपविभागों को परिष्कृत करें
उनका सामान्य एक: प्रत्येक पक्ष परिशोधन के तहत अपरिवर्तित है)।
\end{definition}

\begin{proposition}\label{prop:b1:integration:stepprops}
पर चरण कार्य, द \omterm{thm:b1:integration:def}{अभिन्न} रैखिक है, बढ़ रहा है ($\varphi \leq
\psi \implies \int\varphi \leq \int\psi $), और चैसल्स को संतुष्ट करता है
संबंध $\int_a^b = \int_a^c + \int_c^b $ के लिए $ a < c < b $.
\end{proposition}

\begin{proof}
इंजन है \emph{शोधन अपरिवर्तनशीलता}, में कहा गया है
परिभाषा: एक अतिरिक्त नोड सम्मिलित करना $ t \in \intoo{x_{i-1}}{x_i}$
एक उपविभाजन में शब्द को प्रतिस्थापित करता है $ c_i(x_i - x_{i-1})$ द्वारा
$ c_i(t - x_{i-1}) + c_i(x_i - t)$ --- वही संख्या --- तो
\omterm{thm:b1:integration:def}{अभिन्न} किसी भी परिमित शोधन से अपरिवर्तित है। अब लीजिए
$\varphi $ उपविभाजन के साथ $\sigma $ और $\psi $ उपविभाजन के साथ
$\sigma'$: सामान्य शोधन पर $\sigma \cup \sigma'$ दोनों
हैं चरण कार्य साथ \emph{वही} नोड्स, और प्रत्येक टुकड़े पर
$\varphi + \lambda\psi $ के बराबर स्थिरांक है $ c_i + \lambda
d_i $: रैखिकता परिमित योगों की रैखिकता में कम हो जाती है। बढ़ोतरी:
$ c_i \leq d_i $ प्रत्येक टुकड़े पर देता है $\sum c_i \Delta_i \leq \sum
d_i \Delta_i $ (लंबाई $\Delta_i \geq 0$). चैसल्स: डालें
नोड $ c $ और उस पर राशि विभाजित करें।
\end{proof}

\section{एक सतत कार्य का अभिन्न अंग}

\begin{theorem}[एकसमान सन्निकटन]\label{thm:b1:integration:approx}
होने देना $ f $ होना निरंतर पर $\intcc{a}{b}$. हरएक के लिए $\varepsilon > 0$
वहाँ हैं चरण कार्य $\varphi, \psi $ साथ
\[
\varphi \leq f \leq \psi
\qquad\text{and}\qquad
\psi - \varphi \leq \varepsilon \text{ on } \intcc{a}{b}.
\]
\end{theorem}

\begin{proof}
हेन के प्रमेय द्वारा (\cref{thm:b1:continuity:heine}), $ f $ है \omterm{def:b1:continuity:uniform}{समान रूप से
निरंतर}: चुनना $\delta $ के लिए $\varepsilon $, और का एक उपखंड
जाल $< \delta $ (समान दूरी पर, कहें, साथ $ n > \frac{b -
a}{\delta}$ टुकड़े)। सभी के ऊपर \omterm{def:b1:topology:closed}{बंद किया हुआ} टुकड़ा $\intcc{x_{i-1}}{x_i}$, $ f $
न्यूनतम प्राप्त करता है $ m_i $ और अधिकतम $ M_i $
(\cref{thm:b1:continuity:evt}), और $ M_i - m_i \leq \varepsilon $
(दो चरम बिंदु भीतर हैं $\delta $). परिभाषित करना $\varphi = m_i $
और $\psi = M_i $ पर $\intoo{x_{i-1}}{x_i}$ (और $\varphi = \psi = f $
नोड्स पर)।
\end{proof}

\begin{theorem}[अभिन्न की परिभाषा]\label{thm:b1:integration:def}
होने देना $ f $ होना निरंतर पर $\intcc{a}{b}$. दो नंबर
\[
I_-(f) = \sup\Bigl\{\int_a^b \varphi : \varphi \text{ step},\
\varphi \leq f\Bigr\},
\qquad
I_+(f) = \inf\Bigl\{\int_a^b \psi : \psi \text{ step},\ \psi \geq
f\Bigr\}
\]
समान हैं; उनका सामान्य मूल्य है
\emph{अभिन्न}\index{अभिन्न} $\int_a^b f $ (यह भी लिखा है $\int_a^b
f(t)\,\dd t $). यह पिछली धारणा से मेल खाता है कदम
कार्य, और टुकड़े-टुकड़े तक विस्तारित है निरंतर द्वारा कार्य करता है
बंटवारा $\intcc{a}{b}$ असंततता पर (चैसल्स एक के रूप में
वहां परिभाषा)।
\end{theorem}

\begin{proof}
दोनों सेट गैर-रिक्त हैं ($ f $ घिरा हुआ है) और हर निचला चरण \omterm{thm:b1:integration:def}{अभिन्न}
है $\leq $ प्रत्येक ऊपरी (एकरसता पर)। चरण कार्य): इसलिए
$ I_-(f) \leq I_+(f)$. द्वारा \cref{thm:b1:integration:approx}, हरएक के लिए
$\varepsilon $ के साथ एक जोड़ी है $\int\psi - \int\varphi \leq
\varepsilon(b - a)$: सुपर और इन्फ़ को एक साथ निचोड़ा जाता है, $ I_- =
I_+$.
\end{proof}

\begin{example}[टुकड़े-टुकड़े निरंतर, कोई नाटक नहीं]\label{ex:b1:integration:piecewise}
 फर्श समारोह पर $\intcc{0}{3}$ एक है \omterm{def:b1:integration:step}{समारोह की ओर कदम बढ़ाएं} में
भेष बदलना: अपनी छलांगों में बंट जाना,
\[
\int_0^3 \lfloor t \rfloor\,\dd t
= \int_0^1 0 + \int_1^2 1 + \int_2^3 2 = 0 + 1 + 2 = 3 ,
\]
और मूल्य \emph{पर} छलांग अंक $1, 2$ अप्रासंगिक हैं:
किसी फ़ंक्शन को सीमित रूप से कई बिंदुओं पर बदलने से संख्या में \omterm{def:b1:counting:objects}{परिवर्तन} होता है \omterm{thm:b1:integration:def}{अभिन्न}
(फ़्रेमिंग चरण कार्य अप्रभावित हैं)। यह संपूर्ण है
``टुकड़े-टुकड़े की सामग्री निरंतर''विस्तार: पर काटें
अंतत: अनेक असंततताएँ, प्रत्येक को एकीकृत करती हैं निरंतर टुकड़ा,
परिभाषा के रूप में --- चैसल्स जोड़ें।
\end{example}

\begin{example}[परिभाषा एक बार गणना करती है]\label{ex:b1:integration:defcompute}
होने देना $ f(x) = x $ पर $\intcc{0}{1}$ और काट लें $ n $ बराबर टुकड़े.
सर्वश्रेष्ठ चरण कार्य टुकड़ों पर स्थिर हैं $\varphi =
\frac{k-1}{n}$ और $\psi = \frac kn $ पर $ k $-वां टुकड़ा, साथ
\[
\int_0^1 \varphi = \sum_{k=1}^{n} \frac{k-1}{n}\cdot\frac1n
= \frac{n-1}{2n},
\qquad
\int_0^1 \psi = \sum_{k=1}^{n} \frac{k}{n}\cdot\frac1n
= \frac{n+1}{2n} .
\]
हर निचला \omterm{thm:b1:integration:def}{अभिन्न} है $\leq I_-(f) \leq I_+(f) \leq $ प्रत्येक
ऊपरी एक, तो $\frac{n-1}{2n} \leq I_-(f) \leq I_+(f) \leq
\frac{n+1}{2n}$ सभी के लिए $ n $: दोनों निचोड़ते हैं $\frac12$, और
$\int_0^1 x\,\dd x = \frac12$ सीधे परिभाषा से. द
समापन अंतर्दृष्टि: यह पहली और आखिरी बार है जब हम एकीकृत हुए हैं
परिभाषा से --- नीचे दिया गया मौलिक प्रमेय प्रतिस्थापित करता है
ऐसी सभी गणनाएँ एक एंटीडेरिवेटिव लुकअप द्वारा, जो है
इस अध्याय का संपूर्ण आर्थिक बिंदु।
\end{example}

\begin{theorem}[गुण]\label{thm:b1:integration:props}
के लिए $ f, g $ निरंतर (या टुकड़े-टुकड़े में निरंतर) पर $\intcc{a}{b}$ और
$\lambda \in \R $:
\begin{enumerate}
  \item रैखिकता: $\int (f + \lambda g) = \int f + \lambda \int g $;
  \item एकरसता: $ f \leq g \implies \int_a^b f \leq \int_a^b g $;
        और $\bigl|\int_a^b f\bigr| \leq \int_a^b \abs f \leq (b -
        a)\, \sup\abs f $;
  \item चासल्स: $\int_a^b = \int_a^c + \int_c^b $ (के साथ
        सम्मलेन $\int_b^a = -\int_a^b $, के किसी भी आदेश के लिए मान्य है
        \omterm{def:b1:topology:closure}{सीमा});
  \item सख्त सकारात्मकता: यदि $ f $ है निरंतर, $ f \geq 0$ और
        $\int_a^b f = 0$, तब $ f = 0$ हर जगह पर
        $\intcc{a}{b}$.
\end{enumerate}
\end{theorem}

\begin{proof}
(1)--(3) से गुजरना चरण कार्य के माध्यम से \omterm{def:b1:topology:closure}{सीमा} तक
सुपर/इन्फ परिभाषा. रैखिकता एक बार विवरण के योग्य है: दिया गया
$\varepsilon > 0$, चौखटा $\varphi_f \leq f \leq \psi_f $ और
$\varphi_g \leq g \leq \psi_g $ \omterm{prop:b1:reals:intervals}{अंतराल} के साथ $\leq \varepsilon $
(\cref{thm:b1:integration:approx}). के लिए $\lambda \geq 0$,
$\varphi_f + \lambda\varphi_g \leq f + \lambda g \leq \psi_f +
\lambda\psi_g $ द्वारा एक फ़्रेमिंग है चरण कार्य \omterm{prop:b1:reals:intervals}{अंतराल} के साथ $\leq
(1 + \lambda)\varepsilon $, और इसका चरण \omterm{thm:b1:integration:def}{अभिन्न} बराबर $\int
\varphi_f + \lambda\int\varphi_g $ वगैरह।\
(\cref{prop:b1:integration:stepprops}: देना $\varepsilon \to
0$ निचोड़ कर रख $\int(f + \lambda g)$ पर $\int f + \lambda\int g $.
के लिए $\lambda < 0$, से गुणा करना $\lambda $ \emph{पराजय} द
की फ़्रेमिंग $ g $ --- कम \omterm{def:b1:integration:step}{समारोह की ओर कदम बढ़ाएं} का $\lambda g $ है
$\lambda \psi_g $ --- और वही निचोड़ भूमिकाओं के साथ चलता है
अदला-बदली बाध्य $\abs{\int f} \leq \int \abs f $ से आता है
$-\abs f \leq f \leq \abs f $ और एकरसता.

(4) प्रतिधनात्मक: यदि $ f(x_0) = m > 0$, निरंतरता एक प्रदान करता है
लंबाई का उपअंतराल $\eta > 0$ जिस पर $ f \geq \frac m2$;
\omterm{def:b1:integration:step}{समारोह की ओर कदम बढ़ाएं} लायक $\frac m2$ वहाँ और $0$ अन्यत्र है $\leq f $,
इसलिए $\int f \geq \frac{m\eta}{2} > 0$.
\end{proof}

\begin{example}[काम पर चैसल्स: निरपेक्ष मूल्यों के साथ अभिन्नता]\label{ex:b1:integration:chasleswork}
निरपेक्ष मान को एकीकृत करने के लिए, जहां चिह्न बदलता है, वहां काटें।
\[
\int_0^2 \abs{x - 1}\,\dd x
= \int_0^1 (1 - x)\,\dd x + \int_1^2 (x - 1)\,\dd x
= \frac12 + \frac12 = 1 ,
\]
और, काटना $\intcc{0}{2\pi}$ पर $\pi $:
\[
\int_0^{2\pi} \abs{\sin t}\,\dd t
= \int_0^{\pi} \sin t\,\dd t - \int_{\pi}^{2\pi} \sin t\,\dd t
= 2 + 2 = 4 ,
\]
जबकि $\int_0^{2\pi} \sin t\,\dd t = 0$: रद्दीकरण वास्तविक है,
और यही कारण है कि सख्त सकारात्मकता \omterm{def:b1:logic:statement}{कथन}
(\cref{thm:b1:integration:props}~(4)) परिकल्पना करता है $ f
\geq 0$ --- इसके बिना, एक लुप्त हो जाना \omterm{thm:b1:integration:def}{अभिन्न} कुछ भी साबित नहीं होता
के बारे में $ f $. समापन अंतर्दृष्टि: $\int \abs f $ पैमाने
\emph{क्षेत्र}, $\int f $ पैमाने \emph{हस्ताक्षरित शेष}; द
असमानता $\abs{\int f} \leq \int\abs f $ का सटीक रिकॉर्ड है
रद्दीकरण क्या नष्ट कर सकता है.
\end{example}

\section{कलन का मौलिक प्रमेय}

\begin{theorem}[कलन का मौलिक प्रमेय]\label{thm:b1:integration:ftc}
होने देना $ f $ होना निरंतर एक पर \omterm{prop:b1:reals:intervals}{अंतराल} $ I $ और $ a \in I $. समारोह
\[
F(x) = \int_a^x f(t)\, \dd t
\]
वर्ग का है $ C^1$ पर $ I $, साथ $ F' = f $: प्रत्येक निरंतर समारोह
एक पर \omterm{prop:b1:reals:intervals}{अंतराल} आदिम हैं. नतीजतन, के लिए \emph{कोई}
प्राचीन $ G $ का $ f $:\index{कैलकुलस का मौलिक प्रमेय}
\[
\int_a^b f(t)\,\dd t = G(b) - G(a) .
\]
\end{theorem}

\begin{proof}
हल करना $ x_0 \in I $ और $\varepsilon > 0$; निरंतरता पर $ x_0$ प्रदान
$\delta $ साथ $\abs{f(t) - f(x_0)} \leq \varepsilon $ के लिए $\abs{t -
x_0} \leq \delta $. के लिए $0 < \abs{h} \leq \delta $ (और $ x_0 + h \in
I $), चैसल्स देता है
\[
\frac{F(x_0 + h) - F(x_0)}{h} - f(x_0)
= \frac 1h \int_{x_0}^{x_0+h} \bigl(f(t) - f(x_0)\bigr)\dd t ,
\]
जिसका निरपेक्ष मान है $\leq \frac{1}{\abs h}\cdot \abs h\,
\varepsilon = \varepsilon $ (बाध्य (2), किसी भी आदेश के लिए मान्य
सीमाएँ)। इसलिए $ F'(x_0) = f(x_0)$; $ F' = f $ है निरंतर: $ F $ है
$ C^1$. अगर $ G' = f $ फिर भी $(G - F)' = 0$ पर \omterm{prop:b1:reals:intervals}{अंतराल}, इसलिए $ G =
F + c $ (\cref{cor:b1:derivative:monotone}), और $ G(b) - G(a) = F(b)
- F(a) = \int_a^b f $.
\end{proof}

\begin{example}[गणना से पहले समरूपता]\label{ex:b1:integration:parity}
एक सममित पर \omterm{prop:b1:reals:intervals}{अंतराल}, समता कार्य करती है: यदि $ f $ अजीब है,
प्रतिस्थापन $ t \mapsto -t $ भेजता है $\int_{-a}^{0} f $ को
$-\int_0^a f $, इसलिए
\[
\int_{-a}^{a} f(t)\,\dd t = 0 ;
\qquad\text{if $ f $ is even,}\quad
\int_{-a}^{a} f = 2\int_0^a f .
\]
इस प्रकार $\int_{-1}^{1} \frac{t^3\cos t}{1 + t^4}\,\dd t = 0$ के साथ
दृष्टि में कोई आदिम नहीं है (इंटीग्रैंड विषम है), और
$\int_{-\pi}^{\pi} t^2\cos t\,\dd t = 2\int_0^\pi t^2\cos t\,
\dd t $. तकनीकों तक पहुंचने से पहले समरूपता की जांच करें:
सबसे तेज़ \omterm{thm:b1:integration:def}{अभिन्न} वह है जिसकी कभी गणना नहीं की गई।
\end{example}

\begin{example}[किसी व्युत्पन्न को देखते ही पहचानना]\label{ex:b1:integration:halfangle}
गणना करना $\displaystyle\int_0^{\pi/2} \frac{\dd x}{1 + \cos x}$.
अर्ध-कोण पहचान $1 + \cos x = 2\cos^2\frac x2$ बदल देता है
में एकीकृत $\frac{1}{2}\bigl(1 + \tan^2\frac
x2\bigr)$, जो वास्तव में है यौगिक का $\tan\frac x2$:
\[
\int_0^{\pi/2} \frac{\dd x}{1 + \cos x}
= \Bigl[\tan\frac x2\Bigr]_0^{\pi/2} = \tan\frac\pi4 = 1 .
\]
किसी प्रतिस्थापन मशीनरी की आवश्यकता नहीं थी --- केवल प्रतिबिम्ब की
किसी के रूप में एक इंटीग्रैंड पढ़ना यौगिक, मौलिक
प्रमेय बाकी काम कर रहा है। (ऐसे के पीछे व्यवस्थित उपकरण
त्रिकोणमितीय \omterm{thm:b1:integration:def}{अभिन्न}, प्रतिस्थापन $ t = \tan\frac x2$,
से निर्मित मानक टूलकिट से संबंधित है
\cref{thm:b1:integration:parts}~(2).)
\end{example}

\begin{example}[इंटीग्रल द्वारा परिभाषित कार्य]\label{ex:b1:integration:definedby}
मौलिक प्रमेय कार्यों का निर्माण करता है। होने देना
\[
F(x) = \int_0^x \eu^{-t^2}\,\dd t .
\]
शास्त्रीय कार्यों का कोई संयोजन नहीं है यौगिक
$\eu^{-t^2}$ (लिउविले का एक प्रमेय, स्वीकृत); अभी तक $ F $ मौजूद है,
है $ C^1$ साथ $ F'(x) = \eu^{-x^2} > 0$, सख्ती से बढ़ रहा है, विषम
(विकल्प $ t \mapsto -t $), और परिबद्ध: के लिए $ x \geq 1$,
\[
F(x) - F(1) = \int_1^x \eu^{-t^2}\dd t
\leq \int_1^x \eu^{-t}\dd t \leq \eu^{-1} ,
\]
इसलिए $ F \leq F(1) + \eu^{-1} \leq 1 + \eu^{-1}$. (सटीक \omterm{def:b1:topology:closure}{सीमा},
$\frac{\sqrt\pi}{2}$, की गणना डबल से की जाती है \omterm{thm:b1:integration:def}{अभिन्न} में
वर्ष 3 खंड।) गतिमान \omterm{def:b1:topology:closure}{सीमा} के लिए श्रृंखला नियम: $\frac{\dd}{\dd
x}\int_x^{x^2} \eu^{-t^2}\dd t = 2x\,\eu^{-x^4} - \eu^{-x^2}$.
समापन अंतर्दृष्टि: एकीकरण \emph{बनाता है} नये कार्य
पुराने से, उनके सभी गुणों के साथ पठनीय
इंटीग्रैंड --- जिस आदिम को आप लिख नहीं सकते वह अभी भी एक है
कार्य पर आपका पूरा नियंत्रण है।
\end{example}

\begin{example}[बिना मूल्यांकन किये अनुमान लगाना]\label{ex:b1:integration:estimate}
 \omterm{thm:b1:integration:def}{अभिन्न} $ R_n = \int_0^1 \frac{t^n}{1 + t}\,\dd t $ नहीं है
सुखद \omterm{def:b1:topology:closure}{बंद} रूप, फिर भी एकरसता उन्हें सटीक रूप से पिन करती है: पर
$\intcc{0}{1}$, $\frac12 \leq \frac{1}{1+t} \leq 1$, इसलिए
\[
\frac{1}{2(n+1)} = \frac12\int_0^1 t^n\,\dd t
\;\leq\; R_n \;\leq\; \int_0^1 t^n \,\dd t = \frac{1}{n+1} :
\]
क्षय का सटीक क्रम ($ R_n \sim $ का एक गुणक $\frac1n $,
वास्तव में $ R_n \sim \frac{1}{2n}$) दो पंक्तियों और संख्या के साथ
प्रतिव्युत्पन्न. इस अध्याय की सप्ताहांत समस्याएँ और
अगली बार बिल्कुल ऐसे फ़्रेमिंग पर चलें --- विश्लेषक का पहला
एक से पहले वृत्ति \omterm{thm:b1:integration:def}{अभिन्न} होना चाहिए \emph{इसे बांधो}, और केवल
फिर, यदि आवश्यक हो, तो इसकी गणना करें।
\end{example}

\begin{example}[औसत मान]\label{ex:b1:integration:average}
 \emph{अर्थ} एक का निरंतर $ f $ ऊपर $\intcc{a}{b}$ है
$\frac{1}{b-a}\int_a^b f $. साइन के आर्च के लिए:
\[
\frac{1}{\pi}\int_0^\pi \sin t\,\dd t
= \frac{1}{\pi}\bigl[-\cos t\bigr]_0^\pi = \frac{2}{\pi}
\approx 0.637 :
\]
एक पूर्ण सकारात्मक आर्च औसत नहीं है $\frac12$ लेकिन के लिए
$\frac2\pi $ --- वक्र त्रिभुज की तुलना में ऊंचाई पर अधिक समय व्यतीत करता है
होगा. द्वारा \cref{exo:b1:integration:11} (के लिए माध्य मान प्रमेय
\omterm{thm:b1:integration:def}{अभिन्न}, $ g = 1$), माध्य एक है \emph{कीमत}: $\sin c =
\frac2\pi $ कुछ के लिए $ c \in \intoo{0}{\pi}$. और द्वारा \omterm{thm:b1:integration:riemann}{रीमैन
रकम} इस अध्याय का माध्य सामान्य की \omterm{def:b1:topology:closure}{सीमा} है
का औसत $ n $ नमूने --- असतत के बीच का पुल
डेटा का माध्य और निरंतर सिग्नल का माध्य, जो कि कैसे होता है
द \omterm{thm:b1:integration:def}{अभिन्न} भौतिकी में प्रवेश करता है।
\end{example}

\begin{theorem}[भागों द्वारा एकीकरण; प्रतिस्थापन]\label{thm:b1:integration:parts}
\begin{enumerate}
  \item अगर $ u, v $ हैं $ C^1$ पर $\intcc{a}{b}$:\index{द्वारा एकीकरण
        भागों}
        \[
        \int_a^b u'v = \bigl[uv\bigr]_a^b - \int_a^b uv' .
        \]
  \item अगर $\varphi $ है $ C^1$ पर $\intcc{\alpha}{\beta}$ और $ f $
        निरंतर पर $\varphi(\intcc{\alpha}{\beta})$:
        \index{प्रतिस्थापन}
        \[
        \int_{\alpha}^{\beta} f\bigl(\varphi(t)\bigr)\,\varphi'(t)\,
        \dd t = \int_{\varphi(\alpha)}^{\varphi(\beta)} f(x)\, \dd x .
        \]
\end{enumerate}
\end{theorem}

\begin{proof}
(1) $(uv)' = u'v + uv'$; पर एकीकृत करें $\intcc{a}{b}$ और लागू करें
के लिए मौलिक प्रमेय $ C^1$ समारोह $ uv $.

(2) चलो $ F $ का आदिम हो $ f $ छवि पर \omterm{prop:b1:reals:intervals}{अंतराल}
(\cref{thm:b1:integration:ftc}). तब $(F \circ \varphi)' = (f
\circ \varphi)\,\varphi'$ (श्रृंखला नियम), इसलिए दोनों पक्ष बराबर हैं
$ F(\varphi(\beta)) - F(\varphi(\alpha))$.
\end{proof}

\begin{example}\label{ex:b1:integration:parts}
$\int_0^1 t\,\eu^t \dd t = \bigl[t\,\eu^t\bigr]_0^1 - \int_0^1
\eu^t\dd t = \eu - (\eu - 1) = 1$. और प्रतिस्थापन के साथ $ x =
\sin t $ ($ t \in \intcc{0}{\frac\pi2}$):
\[
\int_0^1 \sqrt{1 - x^2}\, \dd x
= \int_0^{\pi/2} \cos t \cdot \cos t \, \dd t
= \int_0^{\pi/2} \frac{1 + \cos 2t}{2}\, \dd t = \frac\pi4 ,
\]
--- यूनिट डिस्क का एक चौथाई, जैसा कि ज्यामिति की मांग है। (रैखिकीकरण
से \cref{met:b1:complex:trig} काम पर।)
\end{example}

\begin{remark}[इंटीग्रल कैलकुलस में सामान्य गड़बड़ियाँ]
(मैं) \emph{प्रतिस्थापन होना चाहिए $ C^1$ कुल मिलाकर \omterm{prop:b1:reals:intervals}{अंतराल}}:
\omterm{def:b1:counting:objects}{परिवर्तन} $ x = \frac1t $ पार अवैध है $0$; आँख मूँद कर लागू किया गया
को $\int_{-1}^{1}\frac{\dd x}{1 + x^2}$ यह ``साबित'' करता है कि
\omterm{thm:b1:integration:def}{अभिन्न} अपने स्वयं के नकारात्मक के बराबर है। जब किसी प्रतिस्थापन में a
विलक्षणता, काटो \omterm{prop:b1:reals:intervals}{अंतराल} प्रथम (चेसल्स), स्थानापन्न पर
प्रत्येक टुकड़ा, और उसके बाद ही पुनः संयोजित करें। (ii) \emph{लघुगणक
आदिमों को निरपेक्ष मूल्यों की आवश्यकता होती है}: $\int \frac{\dd x}{x - 2} =
\ln\abs{x - 2} + C $ के प्रत्येक पक्ष पर $2$ अलग से ---लिखना
$\ln(x - 2)$ पर $\intoo{0}{1}$ का लघुगणक लिख रहा है
ऋणात्मक संख्या; और स्थिरांक $ C $ दोनों पर मतभेद हो सकता है
विलक्षणता के पक्ष. (iii) \emph{एक लुप्तप्राय \omterm{thm:b1:integration:def}{अभिन्न} करता है
फ़ंक्शन को ख़त्म न करें}: $\int_0^{2\pi}\sin = 0$; की सकारात्मकता
निष्कर्ष से पहले इंटीग्रैंड आवश्यक है $ f = 0$
(\cref{ex:b1:integration:chasleswork}). (iv) \emph{\omterm{thm:b1:integration:riemann}{रीमैन रकम}
अंशांकित किया जाना चाहिए}: में $\frac{b-a}{n}\sum f\bigl(a +
k\frac{b-a}{n}\bigr)$, कदम बाहर और बिंदु अंदर
समान उपखंड से मेल खाना चाहिए --- बार-बार होने वाली त्रुटि का योग होता है
$\sum_{k=1}^{n} f\bigl(\frac kn\bigr)$ \emph{बिना} कारक
$\frac1n $, जो एकाग्र होने के बजाय अलग हो जाता है $\int_0^1
f $. आह्वान करने से पहले चेकलिस्ट \cref{thm:b1:integration:riemann}:
कारक निकालो $\frac1n $, सारांश को इस प्रकार पुनः लिखें $ f $ का $\frac kn $,
नाम $ f $ और इसकी जांच करें निरंतरता.
\end{remark}

\begin{example}[अनुमान लगाएं, अंतर करें, समायोजन करें]\label{ex:b1:integration:guessdiff}
क्या है $\int_1^x (\ln t)^2\,\dd t $? के एक आदिम का अनुमान लगाएं
प्रपत्र $ t\,P(\ln t)$ साथ $ P $ \omterm{def:b1:poly:def}{बहुपद} और अंतर करें:
\[
\bigl(t\,P(\ln t)\bigr)' = P(\ln t) + P'(\ln t) .
\]
ज़रुरत है $ P(u) + P'(u) = u^2$: लेना $ P(u) = u^2 - 2u + 2$
(मिलान गुणांक नीचे से $ u^2$). इस तरह
\[
\int_1^x (\ln t)^2\,\dd t
= \bigl[t\bigl((\ln t)^2 - 2\ln t + 2\bigr)\bigr]_1^x
= x(\ln x)^2 - 2x\ln x + 2x - 2 ,
\]
अन्यथा भागों द्वारा दो एकीकरणों द्वारा प्राप्त परिणाम। द
समापन अंतर्दृष्टि: फॉर्म के इंटीग्रैंड्स के लिए
(\omterm{def:b1:poly:def}{बहुपद} में $\ln t $) या (\omterm{def:b1:poly:def}{बहुपद} टाइम्स $\eu^{\lambda t}$),
आदिम का एक ही आकार होता है --- एक आकार को अलग करना
अनुमान एकीकरण को गुणांकों पर रैखिक बीजगणित में परिवर्तित करता है,
पुनरावृत्त भागों की तुलना में तेज़ और कम त्रुटि-प्रवण।
\end{example}

\begin{example}[बूमरैंग इंटीग्रल]\label{ex:b1:integration:boomerang}
गणना करना $ I = \int_0^{\pi/2} \eu^x \cos x\,\dd x $. द्वारा एकीकृत करें
दो बार भाग, हर बार त्रिकोणमितीय कारक को अलग करना:
\[
I = \bigl[\eu^x \sin x\bigr]_0^{\pi/2}
- \int_0^{\pi/2} \eu^x \sin x\,\dd x
= \eu^{\pi/2} - J,
\qquad
J = \bigl[-\eu^x\cos x\bigr]_0^{\pi/2}
+ \int_0^{\pi/2} \eu^x\cos x\,\dd x = 1 + I .
\]
 \omterm{thm:b1:integration:def}{अभिन्न} अपने आप में वापस आ गया है: $ I = \eu^{\pi/2} - 1 - I $,
कहाँ से
\[
I = \frac{\eu^{\pi/2} - 1}{2} .
\]
समापन अंतर्दृष्टि: जब इंटीग्रैंड दो का उत्पाद होता है
वे कार्य जो विभेदीकरण के तहत स्वयं को पुन: उत्पन्न करते हैं
($\eu^{ax}$, $\cos bx $, $\sin bx $), भागों द्वारा दो एकीकरण
अज्ञात के लिए एक रैखिक समीकरण तैयार करें \omterm{thm:b1:integration:def}{अभिन्न} --- इसे हल करें
एकीकृत करने के बजाय; समान रूप से, गुजरना
$\eu^{(a + \iu b)x}$ (\cref{ch:b1:complex}) और असली ले लो
भागों. दोनों सड़कें एक ही उत्तर देती हैं, और जाँच की जा रही है कि वे हैं
यह एक निःशुल्क विवेक परीक्षण है।
\end{example}

\section{रीमैन रकम}

\begin{theorem}[रीमैन रकम]\label{thm:b1:integration:riemann}
होने देना $ f $ होना निरंतर पर $\intcc{a}{b}$. तब\index{रीमैन रकम}
\[
S_n = \frac{b - a}{n} \sum_{k=0}^{n-1}
f\Bigl(a + k\,\frac{b-a}{n}\Bigr)
\xrightarrow[n \to \infty]{} \int_a^b f(t)\, \dd t ,
\]
और इसी तरह उपअंतराल के अंदर किसी भी मूल्यांकन बिंदु के साथ।
\end{theorem}

\begin{proof}
$ S_n $ है \omterm{thm:b1:integration:def}{अभिन्न} की \omterm{def:b1:integration:step}{समारोह की ओर कदम बढ़ाएं} $\varphi_n $ के बराबर
$ f(a + k\frac{b-a}{n})$ पर $ k $-वें उपअंतराल. दिया गया
$\varepsilon > 0$, एकसमान निरंतरता (हेन) प्रदान करता है $\delta $; के लिए
$ n > \frac{b-a}{\delta}$, उपअंतराल का प्रत्येक बिंदु भीतर है
$\delta $ इसके मूल्यांकन बिंदु का, इसलिए $\abs{f - \varphi_n} \leq
\varepsilon $ पर $\intcc{a}{b}$, कहाँ से
\[
\Bigl| \int_a^b f - S_n \Bigr|
= \Bigl| \int_a^b (f - \varphi_n) \Bigr|
\leq (b-a)\,\varepsilon . \qedhere
\]
\end{proof}

\begin{example}[अंशांकन के साथ दूसरा रीमैन योग]\label{ex:b1:integration:riemann2}
खोजो $\lim_{n\to\infty} \sum_{k=1}^{n} \dfrac{n}{(n+k)^2}$.
अंशांकन:
\[
\sum_{k=1}^{n} \frac{n}{(n+k)^2}
= \frac{1}{n}\sum_{k=1}^{n} \frac{n^2}{(n+k)^2}
= \frac1n \sum_{k=1}^{n} \frac{1}{\bigl(1 + \frac kn\bigr)^2} ,
\]
का एक रीमैन योग निरंतर $ f(x) = \frac{1}{(1+x)^2}$ पर
$\intcc{0}{1}$: हद है
\[
\int_0^1 \frac{\dd x}{(1 + x)^2}
= \Bigl[-\frac{1}{1+x}\Bigr]_0^1 = \frac12 .
\]
समापन अंतर्दृष्टि: संपूर्ण कला मध्य रेखा है ---
सारांश को आकार में बाध्य करें $ f(\frac kn)$ की कीमत पर
बिल्कुल एक कारक निकालना $\frac1n $; एक बार आकार है
सही है, प्रमेय विश्लेषण करता है और मौलिक प्रमेय
अंकगणित करता है.
\end{example}

\begin{remark}[जहां इंटीग्रल आगे काम करता है]
अध्याय की प्रत्येक रचना की एक अगली कड़ी है। \omterm{thm:b1:integration:riemann}{रीमैन रकम}
में लौटें \cref{ch:b1:series} श्रृंखला और के बीच पुल के रूप में
\omterm{thm:b1:integration:def}{अभिन्न} (की तुलना $\sum \frac{1}{n^\alpha}$ साथ $\int
\frac{\dd t}{t^\alpha}$); \omterm{thm:b1:integration:def}{अभिन्न} शेष सबसे तीव्र है
टेलर के सूत्र का रूप \cref{ch:b1:taylor};
$\sup $-आधारित परिभाषा लेब्सग्यू के लिए प्रोटोटाइप है
\omterm{thm:b1:integration:def}{अभिन्न} वर्ष 3 खंड का, जहां वही तीन संपत्तियां हैं
(रैखिकता, एकरसता, एक अभिसरण प्रमेय) को एक पर पुनर्निर्मित किया जाता है
कार्यों का कहीं बड़ा वर्ग। और नीचे सप्ताहांत की समस्या
बदल जाता है भागों द्वारा एकीकरण अंकगणित में: की अतार्किकता
$\pi^2$.
\end{remark}

\begin{exercise}[$\star $]\label{exo:b1:integration:2}
गणना करना $\displaystyle\int_0^{1} \frac{t}{(t^2+1)^2}\,\dd t $
(प्रतिस्थापन) और
$\displaystyle\int_0^{\pi/2} \cos^3 t\, \dd t $ (लिखना $\cos^3 =
\cos(1 - \sin^2)$).
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:integration:3}
जैसे, सीमाएँ ज्ञात कीजिए \omterm{thm:b1:integration:riemann}{रीमैन रकम}:
\[
u_n = \sum_{k=1}^{n} \frac{n}{n^2 + k^2},
\qquad
v_n = \frac{1}{n}\sqrt[n]{\frac{(2n)!}{n!\,n^n}}
\ \emph{(लघुगणक लें)}.
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:integration:4}
होने देना $ f $ होना निरंतर पर $\intcc{0}{1}$. गणना करना $\lim_{n\to\infty}
\int_0^1 x^n f(x)\,\dd x $. \emph{(काटना $\intcc{0}{1}$ पर $1 -
\delta $.)}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:integration:5}
(कॉची-श्वार्ज़) के लिए $ f, g $ निरंतर पर $\intcc{a}{b}$, सिद्ध करना
\[
\Bigl(\int_a^b fg\Bigr)^{\!2} \leq \int_a^b f^2 \cdot \int_a^b g^2 ,
\]
विस्तार करके $\int_a^b (f + \lambda g)^2 \geq 0$ में एक द्विघात के रूप में
$\lambda $. यह कब समानता है?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:integration:6}
होने देना $ f $ होना निरंतर पर $\R $, $ T $-आवधिक. साबित करें कि $\int_a^{a+T}
f $ पर निर्भर नहीं है $ a $, ओर वो $\frac1x \int_0^x f(t)\,\dd t
\to \frac 1T \int_0^T f $ जैसा $ x \to +\infty $.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:integration:7}
के लिए $ f $ निरंतर पर $\intcc{0}{1}$ साथ $\int_0^1 f = \frac12$,
साबित करें कि $ f $ में एक निश्चित बिंदु है $\intcc{0}{1}$. \emph{(एकीकृत करें
$ f(x) - x $ और सख्त सकारात्मकता का प्रयोग करें,
\cref{thm:b1:integration:props}~(4), इसके प्रतिधनात्मक के माध्यम से
मध्यवर्ती मूल्य प्रमेय के साथ संयुक्त।)}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b1:integration:8}
(वालिस \omterm{thm:b1:integration:def}{अभिन्न}) होने देना $ W_n = \int_0^{\pi/2} \sin^n t\,\dd t $.
\begin{enumerate}
  \item पुनरावृत्ति सिद्ध करें $ n W_n = (n-1) W_{n-2}$ ($ n \geq 2$) द्वारा
        भाग, और गणना करें $ W_0, W_1$, तब $ W_{2p}$ और $ W_{2p+1}$
        \omterm{def:b1:topology:closure}{बंद} रूप में.
  \item साबित करें कि $(W_n)$ के साथ घट रहा है $\frac{W_{n+1}}{W_n}
        \to 1$; सिद्ध कीजिए कि मात्रा $(n+1)\,W_{n+1} W_n $ है
        स्थिर, बराबर $\frac\pi2$; और समकक्ष निष्कर्ष निकालें
        $ W_n \sim \sqrt{\dfrac{\pi}{2n}}$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b1:integration:9}
(निवेन: $\pi $ तर्कहीन है) मान लीजिए $\pi = \frac ab $ साथ $ a, b \in
\N^*$, और \omterm{def:b1:logic:sets}{तय करना}, के लिए $ n $ चुना जाना है,
\[
P(x) = \frac{x^n (a - bx)^n}{n!},
\qquad
I_n = \int_0^{\pi} P(x)\sin x\, \dd x .
\]
\begin{enumerate}
  \item साबित करें कि $0 < I_n \leq \pi\,\frac{(\pi a)^n}{n!}$, जो है
        $< 1$ के लिए $ n $ बड़ा।
  \item साबित करें कि $ P $ और यह सब डेरिवेटिव लेना \emph{पूर्णांक}
        मूल्यों पर $0$ और कम से $\pi = \frac ab $. \emph{(द्विपद
        विस्तार: के गुणांक $ P $ टाइम्स $ k!$ पूर्णांक हैं
        के लिए $ k \geq n $; और $ P(\pi - x) = P(x)$.)}
  \item \omterm{def:b1:logic:sets}{तय करना} $ Q = P - P'' + P^{(4)} - \dots $ (एक सीमित राशि). जाँच करें कि $\bigl(Q'\sin x - Q\cos x\bigr)' = P \sin x $, और निष्कर्ष निकालें
        वह $ I_n = Q(\pi) + Q(0)$ एक पूर्णांक है.
  \item निष्कर्ष.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b1:integration:10}
होने देना $ f $ होना $ C^1$ पर $\intcc{a}{b}$. रीमैन--लेब्सग्यू-प्रकार को सिद्ध करें
\omterm{def:b1:topology:closure}{सीमा}
\[
\int_a^b f(t)\sin(\lambda t)\,\dd t \xrightarrow[\lambda \to
+\infty]{} 0
\]
भागों द्वारा एकीकृत करके. फिर इसे दोबारा साबित करें $ f $ केवल
निरंतर, के साथ एकसमान सन्निकटन द्वारा चरण कार्य
(\cref{thm:b1:integration:approx}).
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:integration:11}
(के लिए माध्य मान प्रमेय \omterm{thm:b1:integration:def}{अभिन्न}) होने देना $ f, g $ होना निरंतर पर
$\intcc{a}{b}$ साथ $ g \geq 0$. साबित करो कि वहाँ मौजूद है $ c \in
\intcc{a}{b}$ साथ
\[
\int_a^b f(t)\,g(t)\,\dd t = f(c)\int_a^b g(t)\,\dd t ,
\]
और एक उदाहरण द्वारा दिखाएँ कि परिकल्पना $ g \geq 0$ नहीं हो सकता
गिरा दिया.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b1:integration:12}
(क्षण शून्य को बल देते हैं) चलो $ f $ होना निरंतर पर $\intcc{a}{b}$
साथ
\[
\int_a^b f(t)\,t^k\,\dd t = 0 \qquad \text{for } k = 0, 1,
\dots, n .
\]
साबित करें कि $ f $ पर गायब हो जाता है $ n + 1$ के विशिष्ट बिंदु
$\intoo{a}{b}$. \emph{(अगर $ f $ \omterm{def:b1:counting:objects}{परिवर्तन} चिह्न केवल पर $ z_1 < \dots
< z_m $ साथ $ m \leq n $, एकीकृत करें $ f $ ख़िलाफ़ $ P(t) = (t - z_1)
\cdots (t - z_m)$ और सख्त सकारात्मकता का प्रयोग करें।)}
\end{exercise}

\begin{remark}[इस खंड के अंदर परिप्रेक्ष्य]
आगे के तीन अध्याय सीधे इसी पर निर्भर हैं।
\Cref{ch:b1:taylor} ले जाता है \omterm{thm:b1:integration:def}{अभिन्न} शेष ---
तीन टेलर फ़ॉर्मूलों में से सबसे तेज़ एक है भागों द्वारा एकीकरण
आवर्ती $ n $ बार. \Cref{ch:b1:series} की फ़्रेमिंग को परिवर्तित करता है
द्वारा रकम \omterm{thm:b1:integration:def}{अभिन्न} के लिए निर्णायक परीक्षा में $\sum n^{-\alpha}$,
और इसकी सप्ताहांत समस्या उस फ़्रेमिंग को यूलर में परिष्कृत करती है
स्थिर. \Cref{ch:b1:curves} इसे बनाएं \omterm{thm:b1:integration:def}{अभिन्न} ज्यामितीय:
एक पैरामीट्रिज्ड चाप की लंबाई है $\int \sqrt{x'(t)^2 +
y'(t)^2}\,\dd t $, एक \omterm{thm:b1:integration:def}{अभिन्न} एक का निरंतर ए पर कार्य करें
खंड --- सटीक रूप से यहां निर्मित वस्तु, कोई अनुचित सिद्धांत नहीं
जरुरत. सबसे अधिक पुन: उपयोग किया जाने वाला तथ्य सबसे विनम्र तथ्य होगा:
$\bigl|\int f\bigr| \leq (b - a)\sup\abs{f}$, असमानता
जो हर बिंदुवार अनुमान को एक में बदल देता है \omterm{thm:b1:integration:def}{अभिन्न} अनुमान लगाना।
\end{remark}

\section{समस्या: अभिन्न अतार्किकता मशीन}

\begin{problem}[{सप्ताहांत समस्या --- $\eu $ और $\pi^2$ हैं तर्कहीन, $\eu $ छह दशमलव तक, और $\frac{22}{7} > \pi $ के साथ प्रमाण}]
\label{pb:b1:integration:1}
एक तंत्र इस पूरी समस्या को नियंत्रित करता है: \emph{एक अभिव्यक्ति कि
एक धनात्मक पूर्णांक होना चाहिए, फिर भी यह संभवतः इससे छोटा है $1$,
अस्तित्व में नहीं रह सकता}. \Cref{exo:b1:integration:9} (निवेन) ने इसे एक बार चलाया
साबित करो $\pi \notin \Q $; यहां हम इसका औद्योगीकरण करते हैं। मशीन
तीन भागों की आवश्यकता है: a \emph{समाकलन} इनपुट (अंतबिंदु मान
अच्छी तरह से चुना हुआ बहुआयामी पद), ए \emph{लघुता} इनपुट (एक कारक
$\frac{1}{n!}$ को कुचलना \omterm{thm:b1:integration:def}{अभिन्न}), और ए \emph{पुल}
(भागों द्वारा एकीकरण) उन्हें जोड़ना। हम इसे साबित करते हैं $\eu $ है
तर्कहीन और प्रमाणित त्रुटि के साथ इसकी गणना करें, लीजेंड्रे को सिद्ध करें
तेज प्रमेय कि $\pi^2$ तर्कहीन है, और के साथ समाप्त होता है
सबसे आकर्षक \omterm{thm:b1:integration:def}{अभिन्न} विश्लेषण में: $\int_0^1 \frac{x^4(1 -
x)^4}{1 + x^2}\dd x = \frac{22}{7} - \pi $, कौन सा कोष्ठक $\pi $
हाथ से.\index{irrationality!of pi squared@of
$\pi^2$}\index{irrationality!of e@of $\eu $}

\medskip
\noindent\textbf{Part I --- Fuel.}
\begin{enumerate}
  \item साबित करें कि $\dfrac{c^{\,n}}{n!} \to 0$ प्रत्येक निश्चित के लिए $ c
        > 0$ \emph{(आगे $ n \geq 2c $, प्रत्येक चरण कम से कम आधा
        शब्द)}.
  \item (बीटा \omterm{thm:b1:integration:def}{अभिन्न}) प्रेरण द्वारा सिद्ध करें $ m $ साथ
        भागों द्वारा एकीकरण:
        \[
        \int_0^1 x^{\,k}\,(1 - x)^{\,m}\,\dd x =
        \frac{k!\,m!}{(k + m + 1)!}
        \qquad (k, m \in \N).
        \]
  \item परिणाम निकालना $\displaystyle\int_0^1 \bigl(x(1-x)\bigr)^n \dd x
        = \frac{1}{(2n+1)\binom{2n}{n}}$, और --- के साथ तुलना करना
        बाध्य $ x(1 - x) \leq \frac14$ --- अनुमान
        $\binom{2n}{n} \geq \dfrac{4^n}{2n+1}$, मेल मिलाना
        $\binom{2n}{n}^{1/n} \to 4$ से \cref{pb:b1:seq:1}.
  \item नीचे दो बार प्रयोग की गई लघुता प्रमेयिका को सिद्ध करें: प्रत्येक के लिए
        निरंतर $ g > 0$ पर $\intoo{0}{1}$,
        \[
        0 < \int_0^1 \bigl(x(1-x)\bigr)^n g(x)\,\dd x
        \leq \frac{\sup_{\intcc{0}{1}}\abs g}{4^{\,n}} .
        \]
\end{enumerate}

\medskip
\noindent\textbf{Part II --- $\eu $: irrationality, then six
decimals.} \omterm{def:b1:logic:sets}{तय करना} $ A_n = \displaystyle\int_0^1 x^n \eu^x \dd x $.
\begin{enumerate}[resume]
  \item गणना करना $ A_0$ और $ A_1$, पुनरावृत्ति सिद्ध करें $ A_n = \eu
        - n A_{n-1}$, और सीमाएँ $0 < A_n \leq
        \dfrac{\eu}{n+1}$.
  \item प्रेरण द्वारा दिखाएँ कि $ A_n = \alpha_n + \beta_n \eu $
        साथ $\alpha_n, \beta_n \in \Z $.
  \item उसका निष्कर्ष निकालो $\eu $ तर्कहीन है \emph{(अगर $\eu = \frac
        pq $, तब $ q A_n $ एक पूर्णांक फंसा हुआ है
        $\intoo{0}{1}$ के लिए $ n $ बड़ा)}. प्रमाण से तुलना करें
        का \cref{exo:b1:seq:9}: एक ही पंचलाइन, अलग ईंधन।
  \item प्रेरण द्वारा सिद्ध करें और भागों द्वारा एकीकरण, सटीक
        शेषफल सूत्र
        \[
        \eu = \sum_{k=0}^{n} \frac{1}{k!} + R_n,
        \qquad
        R_n = \frac{1}{n!}\int_0^1 (1 - t)^n\,\eu^{\,t}\,\dd t,
        \qquad
        \frac{1}{(n+1)!} \leq R_n \leq \frac{\eu}{(n+1)!} .
        \]
  \item लेना $ n = 9$: अवश्यंभावी $ R_9$ का उपयोग करते हुए $\eu < 2.75$ (से
        $ b_2 = 2.75$ में \cref{ex:b1:seq:e}), योग का मूल्यांकन करें,
        और प्रमाणित ब्रैकेटिंग समाप्त करें $2.7182818 \leq
        \eu \leq 2.7182823$ --- छह दशमलव, $\eu \approx
        2.718282$, सबूत के साथ.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Legendre's theorem: $\pi^2$ is
irrational.} होने देना $ f(x) = \dfrac{x^n (1 - x)^n}{n!}$, और मान लीजिए
$\pi^2 = \frac ab $ साथ $ a, b \in \N^*$.
\begin{enumerate}[resume]
  \item दिखाओ $ f(1 - x) = f(x)$ और $0 < f \leq
        \dfrac{1}{4^n\,n!}$ पर $\intoo{0}{1}$.
  \item बताते हैं कि $ f^{(k)}(0)$ और $ f^{(k)}(1)$ के लिए पूर्णांक हैं
        प्रत्येक $ k \geq 0$ \emph{(बढ़ाना $ x^n(1-x)^n $ पूर्णांक के साथ
        गुणांक; $\frac{k!}{n!} \in \Z $ के लिए $ k \geq n $;
        फिर समरूपता का उपयोग करें)}.
  \item परिभाषित करना
        \[
        G = b^{\,n} \sum_{k=0}^{n} (-1)^k\,
        \pi^{2n - 2k} f^{(2k)} .
        \]
        बताते हैं कि $ G(0)$ और $ G(1)$ पूर्णांक हैं \emph{(प्रत्येक
        $ b^n \pi^{2n-2k} = a^{\,n-k}\,b^{\,k}$)}.
  \item दूरबीन से सत्यापित करें $ G'' + \pi^2 G = b^n \pi^{2n+2} f
        = \pi^2 a^n f $, तब
        \[
        \frac{\dd}{\dd x}\Bigl(G'(x)\sin \pi x - \pi\,G(x)\cos
        \pi x\Bigr) = \pi^2 a^n f(x)\sin \pi x .
        \]
  \item अधिक एकीकृत करें $\intcc{0}{1}$ और निष्कर्ष निकालो
        \[
        \pi\,a^n \int_0^1 f(x)\sin(\pi x)\,\dd x = G(0) + G(1)
        \in \Z ,
        \]
        ए \emph{सकारात्मक} पूर्णांक से घिरा हुआ $\dfrac{\pi
        a^n}{4^n\,n!}$.
  \item प्रश्न 1 से निष्कर्ष निकालिए $\pi^2$ तर्कहीन है
        (लीजेंडर, 1794), और यह मजबूत होता है
        \cref{exo:b1:integration:9}: अतार्किकता क्यों होती है
        का $\pi^2$ का तात्पर्य है $\pi $, और इसके विपरीत नहीं?
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Understanding the machine.}
\begin{enumerate}[resume]
  \item दो विरोधी ताकतों (समाप्ति बिंदु की अभिन्नता) का पता लगाएँ
        डेटा; की विश्लेषणात्मक लघुता \omterm{thm:b1:integration:def}{अभिन्न}) और
        पुल, भाग II और III में। फिर बताएं कि कारक क्यों है
        $\frac{1}{n!}$ में $ f $ मूल बात यह है: यदि इसे हटा दिया जाए,
        अभिन्नता जीवित रहती है, लेकिन असमानता मर जाती है, और के लिए
        जिसने भिन्नों का दावा किया $\frac ab $ फिर प्रमाण करता है
        असफल?
  \item प्रभावशीलता: मान लीजिए कोई दावा करता है $\pi^2 = \frac ab $
        साथ $ a \leq 10$. दिखाएँ कि विरोधाभास पहले से ही है
        पर उतरता है $ n = 7$: गणना करें $\pi\,(10/4)^7/7! \approx
        0.38 < 1$. मशीन केवल खंडन नहीं करती; यह
        एक निश्चित, गणना योग्य चरण द्वारा खंडन करता है।
  \item विवेक बिना शर्त पुल की जाँच करें: दो से साबित करें
        भागों द्वारा एकीकरण
        \[
        \int_0^1 x(1 - x)\sin(\pi x)\,\dd x = \frac{4}{\pi^3},
        \]
        और प्रश्न 14 के साथ सामंजस्य स्थापित करें $ n = 1$ (रखना
        $\pi^2$ प्रतीकात्मक: दूरबीन की पहचान पढ़ती है $\pi^3
        \int_0^1 f_1 \sin \pi x = -(f_1''(0) + f_1''(1)) = 4$).
  \item क्या बनाता है $\eu^x $ और $\sin \pi x $ के रूप में पात्र हैं
        मशीन की गुठली? संपत्ति की पहचान करें (प्रत्येक संतुष्ट करता है)।
        स्थिरांक के साथ एक रैखिक अंतर समीकरण
        गुणांक, इसलिए दोहराया गया भागों द्वारा एकीकरण चक्र
        प्रारंभ में वापस), और \omterm{def:b1:topology:closure}{सीमा} को नाम दें: वही
        हरमाइट और लिंडमैन द्वारा परिष्कृत मशीन साबित होती है $\eu $
        और $\pi $ \emph{\omterm{pb:b1:logic:1}{ट्रान्सेंडैंटल}} --- इस मात्रा से परे.
\end{enumerate}

\medskip
\noindent\textbf{Part V --- $\frac{22}{7}$ versus $\pi $, and the
moral.}
\begin{enumerate}[resume]
  \item की स्थापना करें \omterm{def:b1:poly:def}{बहुपद} विभाजन
        \[
        \frac{x^4(1-x)^4}{1 + x^2}
        = x^6 - 4x^5 + 5x^4 - 4x^2 + 4 - \frac{4}{1 + x^2},
        \]
        और प्रतिष्ठित पहचान निकालें
        \[
        \int_0^1 \frac{x^4 (1-x)^4}{1 + x^2}\,\dd x
        = \frac{22}{7} - \pi .
        \]
  \item इंटीग्रैंड सकारात्मक है: निष्कर्ष निकालें $\pi <
        \frac{22}{7}$. फिर, बाउंडिंग $\frac{1}{1+x^2}$ बीच में
        $\frac12$ और $1$ और उपयोग कर रहे हैं $\int_0^1 (x(1-x))^4 =
        \frac{1}{630}$ (प्रश्न 3), सिद्ध करें
        \[
        \frac{22}{7} - \frac{1}{630} \;\leq\; \pi \;\leq\;
        \frac{22}{7} - \frac{1}{1260} ,
        \]
        यानी\ $3.14126 \leq \pi \leq 3.14207$: दो सही
        दशमलव, हाथ से.
  \item सामान्यीकरण: विभाजन $ x^{4m}(1-x)^{4m}$ द्वारा $1 + x^2$,
        दिखाएँ कि शेषफल स्थिरांक है $(-4)^m $ \emph{(काम
        modulo $ x^2 + 1$: $(1-x)^2 \equiv -2x $)}, निष्कर्ष निकालना
        तर्कसंगत $ r_m $ साथ
        \[
        \abs{\pi - r_m} \leq 4^{\,1 - 5m} ,
        \]
        और उसे जांचें $ m = 1$ प्रश्न 20-21 को पुन: प्रस्तुत करता है।
  \item सन्निकटन सिद्धांत के साथ इन तर्कसंगतताओं का सामना करें
        का \cref{pb:b1:derivative:1}: गणना करें $\abs{\pi -
        \frac{22}{7}} \approx 1.26\cdot10^{-3}$ के विरुद्ध
        डिरिचलेट गारंटी $\frac{1}{49}$, और उद्धरण
        $\abs{\pi - \frac{355}{113}} \approx 2.7\cdot10^{-7}$
        ख़िलाफ़ $\frac{1}{113^2} \approx 7.8\cdot10^{-5}$:
        इसके लिए असाधारण रूप से अच्छे तर्कसंगत सन्निकटन मौजूद हैं
        $\pi $ --- सुसंगत, तब से $\pi $ होना ज्ञात नहीं है
        बुरी तरह अनुमानित.
  \item (पूर्णांक जाल, सार) लेम्मा को सिद्ध करें
        सब कुछ एकीकृत करता है: \emph{अगर $ x \in \R $ और वहाँ मौजूद हैं
        पूर्णांक $ a_n, b_n $ साथ $0 < \abs{a_n + b_n x} \to 0$,
        तब $ x $ तर्कहीन है.} इसमें इसके उदाहरण सूचीबद्ध करें
        समस्या, में \cref{exo:b1:integration:9}, में
        \cref{exo:b1:seq:9}, और में \cref{pb:b1:derivative:1}.
  \item संश्लेषण, प्रत्येक एक वाक्य: (i) मशीन के तीन
        इसके भाग और प्रत्येक उपकरण बॉक्स में कहाँ रहता है
        अध्याय; (ii) क्या \omterm{thm:b1:integration:def}{अभिन्न} योगदान देता है कि
        का माध्य मान प्रमेय \cref{pb:b1:derivative:1} कुड नोट; (iii) निकाले गए परिणामों की सूची (दो)।
        अतार्किकताएँ, एक छह-दशमलव स्थिरांक, एक
        की ब्रैकेटिंग $\pi $, एक द्विपद बाउंड); (iv) द
        \omterm{def:b1:topology:closure}{सीमा}ंत (हर्माइट, लिंडमैन; और वही जाल, आगे बढ़ें
        $\zeta(2)$ और $\zeta(3)$, बीसवीं सदी में
        अंकगणित)।
\end{enumerate}
\end{problem}
