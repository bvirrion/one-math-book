\chapter{बीजगणितीय संरचनाएँ}\label{ch:b1:structures}

समान गणना नियम पुनः प्रकट होते रहते हैं: पूर्णांक, वास्तविक संख्याएँ,
सम्मिश्र संख्याएँ, \omterm{def:b1:arith:congruence}{अनुरूपता} कक्षाएं, और जल्द ही बहुपद
(\cref{ch:b1:poly}), सदिश और आव्यूह
(\cref{ch:b1:vspaces,ch:b1:matrices}). बीजगणित सामान्य को निकालता है
पैटर्न और उनके नाम: \emph{\omterm{def:b1:structures:group}{समूह}}, \emph{\omterm{def:b1:structures:ring}{अँगूठी}}, \emph{\omterm{def:b1:structures:field}{मैदान}}.
किसी तथ्य को संरचना के स्तर पर एक बार सिद्ध कर देना ही उसे सिद्ध कर देता है
हर उदाहरण एक बार में.

\section{रचना के नियम}

\begin{definition}\label{def:b1:structures:law}
ए \emph{रचना का नियम}\index{रचना का नियम} एक पर \omterm{def:b1:logic:sets}{तय करना} $E $ एक है \omterm{def:b1:logic:map}{नक्शा} $ E \times E \to E $, लिखा हुआ $(x, y) \mapsto x * y $. यह है
\emph{जोड़नेवाला} कब $(x*y)*z = x*(y*z)$ हमेशा, \emph{विनिमेय}
कब $ x * y = y * x $ हमेशा। तत्व $ e $ एक \emph{पहचान} कब
$ e * x = x * e = x $ सभी के लिए $ x $; तब $ x'$ एक \emph{श्लोक में} का $ x $
कब $ x * x' = x' * x = e $.
\end{definition}

\begin{proposition}[विशिष्टता]\label{prop:b1:structures:unique}
एक कानून की अधिकतम एक ही पहचान होती है; पहचान के साथ एक सहयोगी कानून के लिए,
प्रत्येक तत्व में अधिकतम एक व्युत्क्रम होता है।
\end{proposition}

\begin{proof}
अगर $ e $ और $ e'$ पहचान हैं: $ e = e * e' = e'$. अगर $ x'$ और $ x''$
औंधाना $ x $: $ x' = x' * e = x' * (x * x'') = (x' * x) * x'' = e * x''
= x''$.
\end{proof}

\section{समूह}

\begin{definition}[समूह]\label{def:b1:structures:group}
ए \emph{समूह}\index{समूह} $(G, *)$ एक है \omterm{def:b1:logic:sets}{तय करना} एक सहयोगी कानून के साथ
एक पहचान स्वीकार करना और जिसमें प्रत्येक तत्व का व्युत्क्रम होता है। द
समूह है \emph{एबेलियन}\index{एबेलियन} जब कानून है
क्रमविनिमेय
\end{definition}

\begin{example}\label{ex:b1:structures:groups}
$(\Z, +)$, $(\Q, +)$, $(\R, +)$, $(\C, +)$; $(\Q^*, \times)$,
$(\R^*, \times)$, $(\C^*, \times)$, $(\mathbb{U}_n, \times)$ (की जड़ें
एकता, \cref{def:b1:complex:unity}); \omterm{def:b1:logic:sets}{तय करना} $\mathfrak{S}(E)$ का
ए की आपत्तियां \omterm{def:b1:logic:sets}{तय करना} $ E $ स्वयं पर, रचना के अंतर्गत ---
\emph{सममित समूह}\index{सममित समूह} का $ E $, गै\omterm{def:b1:structures:group}{र-एबेलियन} जैसे ही $\abs E \geq 3$. नहीं \omterm{def:b1:structures:group}{समूह}: $(\N, +)$ (कोई व्युत्क्रम नहीं),
$(\Z, \times)$ (केवल $\pm 1$ उलटा)।
\end{example}

\begin{proposition}[गणना नियम]\label{prop:b1:structures:rules}
में एक \omterm{def:b1:structures:group}{समूह} $ G $ (गुणात्मक रूप से लिखा गया, पहचान $ e $):
\begin{enumerate}
  \item रद्दीकरण: $ ax = ay \implies x = y $ और $ xa = ya \implies
        x = y $;
  \item $(ab)^{-1} = b^{-1} a^{-1}$ और $(a^{-1})^{-1} = a $;
  \item के लिए $ a, b \in G $, प्रत्येक समीकरण $ ax = b $ और $ xa = b $ एक है
        अनोखा समाधान ($ x = a^{-1}b $, सम्मान।\ $ x = b a^{-1}$).
\end{enumerate}
\end{proposition}

\begin{proof}
(1) से गुणा करें $ a^{-1}$ उचित पक्ष पर, का उपयोग कर
साहचर्य. (2) $(b^{-1}a^{-1})(ab) = b^{-1}(a^{-1}a)b = b^{-1}b =
e $ और सममित रूप से; उलटे निष्कर्ष की विशिष्टता; दूसरा
बिंदु है \cref{prop:b1:structures:unique} के लिए आवेदन किया $ a^{-1}$. (3)
विशिष्टता के लिए (1) को प्रतिस्थापित और उपयोग करें।
\end{proof}

\begin{example}[एक आयत की समरूपता]\label{ex:b1:structures:klein}
एक (गैर-वर्गाकार) आयत ठीक चार सममिति स्वीकार करता है
स्वयं: पहचान $ e $, क्षैतिज-अक्ष प्रतिबिंब $ h $, द
ऊर्ध्वाधर-अक्ष प्रतिबिंब $ v $, और आधा मोड़ $ r $ के बारे में
केंद्र. रचना इसे चार-तत्व बनाती है \omterm{def:b1:logic:sets}{तय करना} ए \omterm{def:b1:structures:group}{समूह}: प्रत्येक
तत्व स्वयं का व्युत्क्रम है ($ h^2 = v^2 = r^2 = e $), और
किन्हीं दो अलग-अलग गैर-पहचान तत्वों का उत्पाद तीसरा है
($ hv = vh = r $: दोनों अक्षों में परावर्तित होना अर्ध-मोड़ है)। द
पूर्ण तालिका सममित है, इसलिए \omterm{def:b1:structures:group}{समूह} है \omterm{def:b1:structures:group}{एबेलियन} --- फिर भी यह है
\emph{नहीं} जो उसी \omterm{def:b1:structures:group}{समूह} घूर्णन के रूप में $\mathbb U_4$ का
\cref{ex:b1:structures:order}: वहाँ, $\iu $ है \omterm{def:b1:structures:order}{आदेश} $4$, जबकि
यहाँ हर तत्व है \omterm{def:b1:structures:order}{आदेश} $\leq 2$. दो \omterm{def:b1:structures:group}{समूह} उसी का
इस प्रकार आकार में वास्तव में भिन्न गुणन संरचनाएं हो सकती हैं
--- नीचे दिया गया चित्र दोनों तालिकाओं को एक साथ प्रदर्शित करता है। यह
चार तत्व \omterm{def:b1:structures:group}{समूह} के रूप में लौटता है $\{\pm1\} \times \{\pm1\}$, और
\cref{exo:b1:structures:7} समझाता है क्यों कोई \omterm{def:b1:structures:group}{समूह} सबके साथ
वर्ग तुच्छ, इस तरह होना चाहिए \omterm{def:b1:structures:group}{एबेलियन}.
\end{example}

\begin{definition}[उपसमूह]\label{def:b1:structures:subgroup}
एक उपसमुच्चय $ H $ एक का \omterm{def:b1:structures:group}{समूह} $ G $ एक है \emph{उपसमूह}\index{उपसमूह}
(लिखा हुआ $ H \leq G $) जब इसमें शामिल हो $ e $, कानून के तहत स्थिर है
और उलटा के तहत. तब $ H $ स्वयं एक है \omterm{def:b1:structures:group}{समूह}.

\emph{मानदंड:} एक गैररिक्त $ H \subseteq G $ यदि और केवल एक उपसमूह है
यदि
\[
\forall x, y \in H, \quad x y^{-1} \in H .
\]
\end{definition}

\begin{proof}[मानदंड का प्रमाण]
ए \omterm{def:b1:structures:subgroup}{उपसमूह} जाहिर तौर पर इसे संतुष्ट करता है. इसके विपरीत, चलो $ H \neq \emptyset $
इसे संतुष्ट करें, और चुनें $ x_0 \in H $. तब $ e = x_0 x_0^{-1} \in H $; के लिए
$ y \in H $, $ y^{-1} = e\,y^{-1} \in H $; और के लिए $ x, y \in H $, $ xy =
x (y^{-1})^{-1} \in H $.
\end{proof}

\begin{example}\label{ex:b1:structures:subgroups}
$\mathbb{U}_n \leq (\C^*, \times)$: गैररिक्त, और के लिए $ z, w \in
\mathbb{U}_n $, $(zw^{-1})^n = z^n (w^n)^{-1} = 1$. \omterm{def:b1:structures:subgroup}{उपसमूह}ों का
$(\Z, +)$ बिल्कुल वही हैं $ n\Z $ (साबित हुआ)
\cref{thm:b1:arith:gcd}). का एक चौराहा \omterm{def:b1:structures:subgroup}{उपसमूह}ों हमेशा एक है
\omterm{def:b1:structures:subgroup}{उपसमूह}, लेकिन एक संघ लगभग कभी नहीं होता है
(\cref{exo:b1:structures:6}).
\end{example}

\begin{remark}[संरचनाओं से जुड़ी सामान्य गड़बड़ियाँ]\label{rem:b1:structures:pitfalls}
\begin{enumerate}
  \item \emph{कानून के तहत स्थिरता पर्याप्त नहीं है.} $\N $ है
        अंदर जोड़ के तहत स्थिर $\Z $, रोकना $0$, अभी तक है
        नहीं \omterm{def:b1:structures:subgroup}{उपसमूह}: व्युत्क्रम गायब हैं। कसौटी
        $ xy^{-1} \in H $ हर चीज़ का एक ही बार में परीक्षण करता है --- लेकिन केवल
        जाँच के बाद $ H \neq \emptyset $.
  \item \emph{नॉनबेलियन रिफ्लेक्सिस।} सामान्य तौर पर \omterm{def:b1:structures:group}{समूह}, $(ab)^2 =
        abab $, जो है $ a^2b^2$ केवल जब $ a $ और $ b $ आना-जाना;
        वैसे ही $(ab)^{-1} = b^{-1}a^{-1}$, \omterm{def:b1:structures:order}{आदेश} उलटा।
        प्रत्येक पहचान स्कूल बीजगणित से आयातित होनी चाहिए
        स्वयंसिद्धों से पुनः व्युत्पन्न या क्रमविनिमेय के रूप में चिह्नित किया गया।
  \item \emph{गुठली बनाम छवि.} $\ker f $ में रहता है
        \emph{स्रोत}, $\operatorname{im} f $ लक्ष्य में;
        ``$ f $ \omterm{def:b1:logic:inj}{इंजेक्शन} आईएफएफ $\ker f $ मामूली''
        (\cref{prop:b1:structures:kernel}) का कोई एनालॉग नहीं है
        छवि ($\operatorname{im} f = G'$ प्रक्षेपात्मकता है)।
  \item \emph{रिंगों नहीं हैं \omterm{def:b1:structures:group}{समूह} के लिए $\times $.} में एक \omterm{def:b1:structures:ring}{अँगूठी}, अधिकांश
        तत्वों को उलटा करने और रद्द करने की आवश्यकता नहीं है $ a $
        आवश्यक है $ a $ एक इकाई होना या \omterm{def:b1:structures:ring}{अँगूठी} एक होना \omterm{def:b1:structures:field}{अभिन्न
        डोमेन}: में $\Z/12\Z $, $\overline3\,\overline2 =
        \overline3\,\overline6$ अभी तक $\overline2 \neq \overline6$
        (\cref{ex:b1:structures:zncomputation}).
\end{enumerate}
\end{remark}

\begin{definition}[समूह रूपवाद]\label{def:b1:structures:morphism}
होने देना $(G, *)$ और $(G', \star)$ होना \omterm{def:b1:structures:group}{समूह}. ए \omterm{def:b1:logic:map}{नक्शा} $ f \colon G \to G'$
एक है \emph{आकारिता}\index{आकारिता} कब
\[
\forall x, y \in G, \qquad f(x * y) = f(x) \star f(y).
\]
तब $ f(e_G) = e_{G'}$ और $ f(x^{-1}) = f(x)^{-1}$.
\emph{कर्नेल}\index{कर्नेल} और \emph{छवि} का $ f $ हैं
\[
\ker f = f^{-1}(\{e_{G'}\}) \leq G,
\qquad
\operatorname{im} f = f(G) \leq G' .
\]
ए \omterm{def:b1:logic:inj}{द्विभाजित} रूपवाद एक है \emph{समाकृतिकता}; यह उलटा है \omterm{def:b1:logic:map}{नक्शा} तब है
स्वचालित रूप से एक रूपवाद.
\end{definition}

\begin{proof}[दावे का प्रमाण]
$ f(e) = f(e * e) = f(e)\star f(e)$, और रद्द करना $ f(e)$ देता है
$ e_{G'} = f(e)$. तब $ f(x)\star f(x^{-1}) = f(x x^{-1}) = e_{G'}$
पहचान करता है $ f(x^{-1})$ उलटे के रूप में. गुठली: $ e \in \ker f $; अगर
$ x, y \in \ker f $, $ f(xy^{-1}) = f(x)f(y)^{-1} = e $; कसौटी
लागू होता है. छवि: साथ एक ही मानदंड $ f(x)f(y)^{-1} = f(xy^{-1})$.
एक समरूपता का व्युत्क्रम: के लिए $ u, v \in G'$, लिखना $ u = f(x)$, $ v =
f(y)$; तब $ f^{-1}(u \star v) = f^{-1}(f(xy)) = xy =
f^{-1}(u) f^{-1}(v)$.
\end{proof}

\begin{proposition}[कर्नेल के माध्यम से इंजेक्शन]\label{prop:b1:structures:kernel}
ए \omterm{def:b1:structures:group}{समूह} \omterm{def:b1:structures:morphism}{आकारिता} $ f $ है \omterm{def:b1:logic:inj}{इंजेक्शन} अगर और केवल अगर $\ker f =
\{e\}$.
\end{proposition}

\begin{proof}
अगर $ f $ है \omterm{def:b1:logic:inj}{इंजेक्शन}, $\ker f $ केवल एक को ही समाहित किया जा सकता है \omterm{def:b1:logic:map}{पूर्वछवि} का
$ e_{G'}$, जो है $ e $. इसके विपरीत, यदि $\ker f = \{e\}$ और $ f(x) =
f(y)$, तब $ f(xy^{-1}) = f(x) f(y)^{-1} = e_{G'}$, इसलिए $ xy^{-1} = e $,
यानी\ $ x = y $.
\end{proof}

\begin{example}\label{ex:b1:structures:expmorphism}
$\exp \colon (\R, +) \to (\R_+^*, \times)$ एक है \omterm{def:b1:structures:morphism}{आकारिता}
($\eu^{x+y} = \eu^x \eu^y $), \omterm{def:b1:logic:inj}{द्विभाजित}
(\cref{prop:b1:functions:expln}): योगात्मक और गुणक
संरचनाएं समरूपी हैं --- इसका ऐतिहासिक कारण
लघुगणक. एक और \omterm{def:b1:structures:morphism}{आकारिता}: $\theta \mapsto \eu^{\iu\theta}$ से
$(\R, +)$ यूनिट सर्कल पर $(\mathbb{U}, \times)$, साथ \omterm{def:b1:structures:morphism}{कर्नेल}
$2\pi\Z $.
\end{example}

\begin{example}[संकेत रूपवाद]\label{ex:b1:structures:signmorphism}
 \omterm{def:b1:logic:map}{नक्शा} $ s \colon (\R^*, \times) \to (\{\pm1\}, \times)$ भेज रहा हूँ
$ x $ इसका चिन्ह एक है \omterm{def:b1:structures:morphism}{आकारिता}: किसी उत्पाद का चिन्ह है
संकेतों का उत्पाद. इसका \omterm{def:b1:structures:morphism}{कर्नेल} है $\intoo0{+\infty}$ (ए
\omterm{def:b1:structures:subgroup}{उपसमूह}, जैसा \cref{def:b1:structures:morphism} वादे), यह
सभी की छवि $\{\pm1\}$: \omterm{def:b1:logic:inj}{विशेषण}, बड़े पैमाने पर गैर \omterm{def:b1:logic:inj}{इंजेक्शन}.
लघुचित्र में दो सामान्य पाठ। पहले एक \omterm{def:b1:structures:morphism}{आकारिता} कुचल सकता है
जानकारी: $ s $ कुछ भी याद नहीं है $ x $ लेकिन एक बिट, और वह
क्या इसका गुण है --- सांकेतिक तर्क बिल्कुल गणनाएँ हैं
उस कारक के माध्यम से $ s $. दूसरा, रूपवाद को $\{\pm1\}$ हैं
सरलतम ``अपरिवर्तनीय'': का हस्ताक्षर क्रमपरिवर्तन, में निर्मित
इस अध्याय की सप्ताहांत समस्या, पर भी वही घटना है
\omterm{def:b1:structures:group}{समूह} $\mathfrak S_n $, और समता का तर्क है कि यह सभी को शक्ति प्रदान करता है
ऐसे दो-मूल्य के माध्यम से उतरो \omterm{def:b1:structures:morphism}{आकारिता}.
\end{example}

\begin{definition}[शक्तियाँ, किसी तत्व का क्रम]\label{def:b1:structures:order}
में एक \omterm{def:b1:structures:group}{समूह} $ G $ (गुणक अंकन), सेट $ x^0 = e $, $ x^{k+1} =
x^k x $ और $ x^{-k} = (x^k)^{-1}$ के लिए $ k \in \N $; तब $ x^{k+l} = x^k
x^l $ सभी के लिए $ k, l \in \Z $, इसलिए $ k \mapsto x^k $ एक है \omterm{def:b1:structures:morphism}{आकारिता} $(\Z,
+) \to G $ किसकी छवि $\langle x \rangle = \{x^k : k \in \Z\}$ एक है
\omterm{def:b1:structures:subgroup}{उपसमूह}, द \omterm{def:b1:structures:subgroup}{उपसमूह} \emph{आप जेनरेट हुई} द्वारा $ x $.
\emph{आदेश}\index{आदेश} का $ x $ सबसे कम है $ m \geq
1$ साथ $ x^m = e $ यदि कोई मौजूद है (तो $\langle x\rangle = \{e, x,
\dots, x^{m-1}\}$ बिलकुल है $ m $ तत्व, और $ x^k = e \iff m
\mid k $), और $\infty $ अन्यथा।
\end{definition}

\begin{example}\label{ex:b1:structures:order}
में $(\C^*, \times)$: $\iu $ है \omterm{def:b1:structures:order}{आदेश} $4$, साथ $\langle \iu \rangle
= \{1, \iu, -1, -\iu\} = \mathbb{U}_4$; आम तौर पर अधिक $\omega =
\eu^{2\iu\pi/n}$ है \omterm{def:b1:structures:order}{आदेश} $ n $ और $\langle\omega\rangle =
\mathbb{U}_n $. में $(\Z, +)$, प्रत्येक $ x \neq 0$ अनंत है \omterm{def:b1:structures:order}{आदेश}.
परिभाषा में दावे क्यों मान्य हैं: यदि $ x $ है \omterm{def:b1:structures:order}{आदेश} $ m $, विभाजित करें
कोई भी $ k $ द्वारा $ m $ ($ k = mq + r $, $0 \leq r < m $,
\cref{thm:b1:arith:division}): $ x^k = (x^m)^q x^r = x^r $, तो
अवधि के साथ शक्तियों का चक्र $ m $, सूचीबद्ध तत्व जोड़ीवार हैं
की न्यूनतमता से भिन्न $ m $, और $ x^k = e $ बल $ r = 0$. \omterm{def:b1:structures:order}{आदेश}
का क्रमपरिवर्तन नीचे सप्ताहांत समस्या में गणना की गई है।
\end{example}

\begin{example}[अंदर आदेश $\mathbb U_{12}$]\label{ex:b1:structures:orderU12}
क्या है \omterm{def:b1:structures:order}{आदेश} का $\omega^k $ में $\mathbb U_n $, के लिए $\omega =
\eu^{2\iu\pi/n}$? एक के पास है $(\omega^k)^m = 1$ आईएफएफ $ n \mid km $, और
लेखन $ d = \gcd(n, k)$, $ n = dn'$, $ k = dk'$ साथ $\gcd(n', k')
= 1$: $ n \mid km \iff n' \mid k'm \iff n' \mid m $ (गॉस का
लेम्मा, \cref{thm:b1:arith:gauss}). ऐसा सबसे कम $ m \geq 1$ है
$ n' = \frac{n}{\gcd(n,k)}$. में $\mathbb U_{12}$ उदाहरण के लिए,
$\omega^8$ है \omterm{def:b1:structures:order}{आदेश} $\frac{12}{\gcd(12,8)} = 3$ (वास्तव में
$\omega^8 = \eu^{4\iu\pi/3} \in \mathbb U_3$), जबकि $\omega^5$
है \omterm{def:b1:structures:order}{आदेश} $12$: यह संपूर्ण उत्पन्न करता है \omterm{def:b1:structures:group}{समूह}, हालाँकि ऐसा नहीं है
``मानक'' जनरेटर। जनरेटरों की गिनती--- $ k $
साथ $\gcd(k, n) = 1$ --- पुनर्प्राप्त करता है \omterm{cor:b1:arith:bezout}{सह अभाज्य} की गिनती
\cref{ex:b1:counting:coprime}: \omterm{def:b1:structures:group}{समूह} सिद्धांत और गिनती मिलते हैं.
\end{example}

\section{अंगूठियाँ और खेत}

\begin{definition}[अँगूठी]\label{def:b1:structures:ring}
ए \emph{अँगूठी}\index{अँगूठी} $(A, +, \times)$ एक है \omterm{def:b1:logic:sets}{तय करना} ऐसे दो कानूनों के साथ
वह: $(A, +)$ एक \omterm{def:b1:structures:group}{एबेलियन} \omterm{def:b1:structures:group}{समूह} (पहचान $0$); $\times $ है
एक पहचान के साथ संबद्ध $1$; और $\times $ वितरित करता है $+$ पर
दोनों तरफ. अंगूठी है \emph{विनिमेय} कब $\times $ है। तत्व $ a $ है \emph{उलटी} (ए \emph{इकाई}) कब $ ab = ba = 1$
कुछ के लिए $ b $; इकाइयाँ बनती हैं a \omterm{def:b1:structures:group}{समूह} $(A^\times, \times)$.
\end{definition}

\begin{proof}[प्रमाण है कि इकाइयाँ एक समूह बनाती हैं]
स्थिरता: यदि $ a, a'$ व्युत्क्रम वाली इकाइयाँ हैं $ b, b'$, तब
\[
(aa')(b'b) = a(a'b')b = a\,1\,b = ab = 1,
\qquad (b'b)(aa') = 1
\]
सममित रूप से, तो $ aa'$ एक इकाई है. तत्व $1$ एक इकाई है (इसकी)
स्वयं का उलटा), साहचर्यता विरासत में मिली है $ A $, और
उलटा $ b $ एक इकाई का $ a $ स्वयं एक इकाई है (व्युत्क्रम के साथ)। $ a $).
इसलिए $(A^\times, \times)$ सभी को संतुष्ट करता है \omterm{def:b1:structures:group}{समूह} स्वयंसिद्ध. प्रत्येक \omterm{def:b1:structures:group}{समूह}
इस पुस्तक में जिसका निर्माण नहीं किया गया है क्रमपरिवर्तन इस प्रकार उत्पन्न होता है:
$\Q^* = \Q^\times $, $\R^*$, $\C^*$, की इकाइयाँ $\Z/n\Z $ नीचे,
और बाद में व्युत्क्रमणीय मैट्रिक्स (\cref{ch:b1:matrices}).
\end{proof}

\begin{example}\label{ex:b1:structures:rings}
$\Z, \Q, \R, \C $ क्रमविनिमेय हैं के छल्ले; $\Z^\times = \{1, -1\}$,
$\Q^\times = \Q^*$. बाद में: बहुपद के छल्ले $ K[X]$
(\cref{ch:b1:poly}), मैट्रिक्स के छल्ले (गैर-विनिमेय,
\cref{ch:b1:matrices}), और $\Z/n\Z $ नीचे। प्रत्येक \omterm{def:b1:structures:ring}{अँगूठी}, $0 \times
a = 0$ (वितरणशीलता से: $0a = (0+0)a = 0a + 0a $), और
$(-1)a = -a $.
\end{example}

\begin{example}[इडेम्पोटेंट्स: नए छल्लों में नई घटनाएं]\label{ex:b1:structures:idempotents}
में $\Z $, समीकरण $ x^2 = x $, यानी\ $ x(x - 1) = 0$, केवल है
समाधान $0$ और $1$. में $\Z/6\Z $, सभी वर्गों का परीक्षण:
$\overline0^2 = \overline0$, $\overline1^2 = \overline1$,
$\overline3^2 = \overline9 = \overline3$ और $\overline4^2 =
\overline{16} = \overline4$ --- \emph{चार} बेरोजगार दो
विदेशी शून्य विभाजक से आते हैं: $\overline3\,(\overline3 -
\overline1) = \overline3 \times \overline2 = \overline6 =
\overline0$ न तो कारक शून्य के साथ. ऐसी गणनाएँ अंशांकित होती हैं
किसी की सहज प्रवृत्ति: समीकरणों के बारे में परिचित तथ्य जीवित रहते हैं
\omterm{def:b1:structures:field}{अभिन्न डोमेन} और खेत, लेकिन एक जनरल \omterm{def:b1:structures:ring}{अँगूठी} कर सकता है और करता है
अलग व्यवहार करें --- बूलियन भी देखें के छल्ले का
\cref{exo:b1:structures:10}, कहाँ \emph{प्रत्येक} तत्व है
निष्क्रिय
\end{example}

\begin{proposition}[क्रमविनिमेय वलय में द्विपद प्रमेय]\label{prop:b1:structures:binomial}
अगर $ a, b $ एक क्रमविनिमेय के तत्व हैं \omterm{def:b1:structures:ring}{अँगूठी} (अधिक सामान्यतः, यदि
$ ab = ba $), फिर के लिए $ n \in \N $:
\[
(a+b)^n = \sum_{k=0}^n \binom nk a^k b^{n-k},
\qquad
a^n - b^n = (a - b) \sum_{k=0}^{n-1} a^k b^{\,n-1-k} .
\]
\end{proposition}

\begin{proof}
के प्रमाण \cref{thm:b1:counting:binomial} और ज्यामितीय का
पहचान केवल साहचर्य का उपयोग करती है, दो तत्वों की क्रमविनिमेयता,
और वितरण--- वे शब्दशः लागू होते हैं।
\end{proof}

\begin{example}[एक अपरिचित वलय में द्विपद प्रमेय]\label{ex:b1:structures:binomialring}
व्यापकता के दो त्वरित भुगतान। में $\Z/p\Z $ ($ p $ \omterm{def:b1:arith:prime}{मुख्य}), द
मध्य \omterm{def:b1:counting:objects}{द्विपद गुणांक} गायब होना
(\cref{thm:b1:arith:fermat}का पहला कदम), तो प्रमेय
में ढह जाता है \emph{नवसिखुआ का सपना}
\[
(a + b)^p = a^p + b^p \qquad \text{in } \Z/p\Z ,
\]
वहाँ एक वास्तविक पहचान है, चाहे वह कितनी भी आपराधिक क्यों न दिखती हो $\R $.
और किसी भी क्रमविनिमेय में \omterm{def:b1:structures:ring}{अँगूठी} एक तत्व युक्त $\varepsilon $
साथ $\varepsilon^2 = 0$, प्रमेय छोटा करता है:
$(a + \varepsilon)^n = a^n + n\,a^{n-1}\varepsilon $, सभी उच्चतर
एक कारक ले जाने वाले शब्द $\varepsilon^2 = 0$. गुणांक
$ n\,a^{n-1}$ का $\varepsilon $ का व्युत्पन्न है $ x^n $ ---
कोई दुर्घटना नहीं, और पहला संकेत कि व्युत्पन्न बीजगणित हैं
विश्लेषण के रूप में (औपचारिक व्युत्पन्न की तुलना करें)।
\cref{ch:b1:poly}).
\end{example}

\begin{definition}[इंटीग्रल डोमेन, फ़ील्ड]\label{def:b1:structures:field}
एक क्रमविनिमेय \omterm{def:b1:structures:ring}{अँगूठी} $ A \neq \{0\}$ एक \emph{अभिन्न
डोमेन}\index{अभिन्न
डोमेन} जब इसका कोई शून्य भाजक न हो: $ ab = 0
\implies a = 0$ या $ b = 0$. यह एक है \emph{मैदान}\index{मैदान} कब
प्रत्येक अशून्य तत्व व्युत्क्रमणीय है। प्रत्येक क्षेत्र एक अभिन्न अंग है
डोमेन ($ ab = 0$ और $ a \neq 0$ देना $ b = a^{-1}ab = 0$).
\end{definition}

\begin{example}
$\Q $, $\R $, $\C $ हैं खेत; $\Z $ एक \omterm{def:b1:structures:field}{अभिन्न डोमेन} लेकिन नहीं
\omterm{def:b1:structures:field}{मैदान}. एक में \omterm{def:b1:structures:field}{अभिन्न डोमेन}, रद्दीकरण कायम है $\times $: $ ab =
ac $ और $ a \neq 0$ संकेत करना $ b = c $.
\end{example}

\section{अंगूठी $\Z/n\Z $}

\begin{definition}\label{def:b1:structures:zn}
हल करना $ n \in \N^*$. \omterm{def:b1:arith:congruence}{अनुरूपता} कक्षाएं मॉड $ n $
(\cref{ex:b1:logic:congruence}) फॉर्म ए \omterm{def:b1:logic:sets}{तय करना}
$\Z/n\Z $\index{Z/nZ@$\Z/n\Z $} का $ n $ तत्व, लिखित $\overline 0,
\overline 1, \dots, \overline{n-1}$. संचालन
\[
\overline a + \overline b = \overline{a + b},
\qquad
\overline a \times \overline b = \overline{ab}
\]
अच्छी तरह से परिभाषित हैं --- परिणामों की श्रेणियां इस पर निर्भर नहीं करती हैं
प्रतिनिधि, ठीक इसलिए क्योंकि \omterm{def:b1:arith:congruence}{अनुरूपता} के साथ संगत है $+$
और $\times $ (\cref{def:b1:arith:congruence}) --- और बनाओ $\Z/n\Z $ ए
क्रमविनिमेय \omterm{def:b1:structures:ring}{अँगूठी}.
\end{definition}

\begin{theorem}[की इकाइयाँ $\Z/n\Z $; मैदान $\Z/p\Z $]\label{thm:b1:structures:zp}
\begin{enumerate}
  \item $\overline a $ में उलटा है $\Z/n\Z $ अगर और केवल अगर
        $\gcd(a, n) = 1$.
  \item $\Z/n\Z $ एक है \omterm{def:b1:structures:field}{मैदान} अगर और केवल अगर $ n $ है \omterm{def:b1:arith:prime}{मुख्य}.
\end{enumerate}
\end{theorem}

\begin{proof}
(1) है \cref{prop:b1:arith:invmod} कक्षाओं के साथ पुनः लिखा गया।

(2) यदि $ n = p $ है \omterm{def:b1:arith:prime}{मुख्य}, प्रत्येक $\overline a \neq \overline 0$ है $ p
\nmid a $, इसलिए $\gcd(a, p) = 1$: (1) द्वारा उलटा --- ए \omterm{def:b1:structures:field}{मैदान}. अगर $ n =
ab $ साथ $1 < a, b < n $, तब $\overline a\, \overline b = \overline
n = \overline 0$ साथ $\overline a, \overline b \neq \overline 0$:
शून्य भाजक, अत: एक भी नहीं \omterm{def:b1:structures:field}{अभिन्न डोमेन}; और $ n = 1$ देता है
शून्य \omterm{def:b1:structures:ring}{अँगूठी}, छोड़ा गया।
\end{proof}

\begin{example}[के कितने वर्गमूल $1$?]\label{ex:b1:structures:sqrtone}
हल करना $ x^2 = \overline 1$ में $\Z/8\Z $ और में $\Z/7\Z $. का परीक्षण
आठ वर्ग मॉड $8$: $1^2 = 1$, $3^2 = 9 \equiv 1$, $5^2 = 25
\equiv 1$, $7^2 = 49 \equiv 1$ --- \emph{चार} समाधान
$\{\overline1, \overline3, \overline5, \overline7\}$, भले ही
बहुपद $ X^2 - 1$ की डिग्री है $2$. में \omterm{def:b1:structures:field}{मैदान} $\Z/7\Z $,
इसके विपरीत, $ x^2 = \overline1$ मतलब $(x - \overline1)(x +
\overline1) = \overline0$, और ए \omterm{def:b1:structures:field}{मैदान} कोई शून्य भाजक नहीं है: $ x =
\pm\overline1$, केवल दो समाधान। विफलता मॉड $8$ है
पता लगाने योग्य: $(3-1)(3+1) = 2 \times 4 = 8 \equiv 0$ दोनों के बिना
कारक लुप्त होना. नैतिक: परिचित नियम ``एक डिग्री-$ d $
समीकरण अधिकतम है $ d $ जड़ों'' के बारे में एक प्रमेय है
\emph{\omterm{def:b1:structures:field}{अभिन्न डोमेन}} (\cref{cor:b1:poly:nroots} इसे खत्म साबित करता है
खेत); में के छल्ले शून्य भाजक के साथ यह चुपचाप विफल हो जाता है --- जो
यही कारण है कि विल्सन के प्रमेय का युग्म प्रमाण है
(\cref{exo:b1:arith:11}) आवश्यकता है $ p $ \omterm{def:b1:arith:prime}{मुख्य}.
\end{example}

\begin{example}[में कंप्यूटिंग $\Z/n\Z $]\label{ex:b1:structures:zncomputation}
में $\Z/12\Z $: इकाइयाँ हैं $\overline 1, \overline 5, \overline 7,
\overline{11}$ (कक्षाएं \omterm{cor:b1:arith:bezout}{सह अभाज्य} को $12$), और प्रत्येक का अपना है
उलटा ($5^2 = 25 \equiv 1$, $7^2 = 49 \equiv 1$, $11^2 = 121 \equiv
1$). समीकरण $\overline 3\, x = \overline 6$ है \emph{तीन}
समाधान ($ x \in \{\overline 2, \overline 6, \overline{10}\}$):
बिना उलटेपन के, बिना किसी रद्दीकरण के। में $\Z/11\Z $ इसके विपरीत,
हर समीकरण $\overline a x = \overline b $ साथ $\overline a \neq
\overline 0$ बिल्कुल एक ही समाधान है.
\end{example}

\begin{example}[एक समाधान लाइसेंस के रूप में समूह स्वयंसिद्ध]\label{ex:b1:structures:solving}
में \omterm{def:b1:structures:group}{समूह} $\bigl((\Z/7\Z)^*, \times\bigr)$, हल करना $\overline
3\,x = \overline 5$. द्वारा \cref{prop:b1:structures:rules}~(3) द
समाधान मौजूद है, अद्वितीय है, और बराबर है $\overline3^{-1}\,
\overline5$; तब से $\overline3 \times \overline5 = \overline{15}
= \overline1$, का उलटा $\overline 3$ है $\overline 5$, इसलिए
\[
x = \overline5 \times \overline5 = \overline{25} = \overline4,
\qquad\text{check: } \overline3 \times \overline4 =
\overline{12} = \overline5 .
\]
बात गारंटी से कम उत्तर की है: ए में \omterm{def:b1:structures:group}{समूह},
ऐसा प्रत्येक समीकरण विशिष्ट रूप से हल करने योग्य है \emph{पहले} कोई भी
गणना, इसलिए समाधान प्रक्रिया कभी भी ``नहीं'' में नहीं जा सकती
समाधान'' या ``कई''। तुलना करना $\overline3\,x = \overline6$
में $\Z/12\Z $ ऊपर, जहां गारंटी विफल हो जाती है --- यह जानते हुए भी
व्यक्ति जिस संरचना में है वह यह जानना है कि वह क्या मान सकता है।
\end{example}

\begin{example}[प्रत्यक्ष उत्पाद]\label{ex:b1:structures:product}
अगर $ G $ और $ H $ हैं \omterm{def:b1:structures:group}{समूह}, उत्पाद \omterm{def:b1:logic:sets}{तय करना} $ G \times H $ के साथ
घटकवार कानून $(g, h)(g', h') = (gg', hh')$ एक है \omterm{def:b1:structures:group}{समूह}:
अभिगृहीतों की जाँच समन्वय द्वारा समन्वय, पहचान के साथ की जाती है
$(e_G, e_H)$ और व्युत्क्रम $(g^{-1}, h^{-1})$. \omterm{def:b1:structures:order}{आदेश} द्वारा संयोजित करें
एलसीएम: $(g, h)^m = (g^m, h^m)$ पहचान है अगर \omterm{def:b1:structures:order}{आदेश}
का $ g $ और यह \omterm{def:b1:structures:order}{आदेश} का $ h $ दोनों बंट जाते हैं $ m $. इस प्रकार में $\Z/2\Z
\times \Z/2\Z $ (योजक) प्रत्येक अशून्य तत्व में होता है \omterm{def:b1:structures:order}{आदेश} $2$
--- यह बिल्कुल आयत है \omterm{def:b1:structures:group}{समूह} का
\cref{ex:b1:structures:klein} निर्देशांक में --- जबकि $\Z/4\Z $
का एक तत्व है \omterm{def:b1:structures:order}{आदेश} $4$: एक दूसरा, गणना-मुक्त प्रमाण
वह दो \omterm{def:b1:structures:group}{समूह} आकार का $4$ समरूपी नहीं हैं (ए
समरूपता संरक्षित करती है \omterm{def:b1:structures:order}{आदेश}). उत्पाद सबसे आसान तरीका हैं
नया निर्माण करें \omterm{def:b1:structures:group}{समूह} पुराने से, और विमान से $\R^2 = \R \times
\R $ का \cref{ch:b1:vspaces} निर्माण का सबसे महत्वपूर्ण है
उदाहरण.
\end{example}

\begin{remark}[फ़र्मेट, संरचनात्मक रूप से]
में \omterm{def:b1:structures:field}{मैदान} $\Z/p\Z $, अशून्य वर्ग एक गुणक बनाते हैं \omterm{def:b1:structures:group}{समूह}
साथ $ p - 1$ तत्व, और फ़र्मेट की छोटी प्रमेय
(\cref{thm:b1:arith:fermat}) कहता है: प्रत्येक तत्व $ x $ इस का \omterm{def:b1:structures:group}{समूह}
संतुष्ट $ x^{p-1} = \overline 1$. यह एक जनरल का उदाहरण है
परिमित के बारे में तथ्य \omterm{def:b1:structures:group}{समूह} (लैग्रेंज प्रमेय), दूसरे में सिद्ध हुआ
वर्ष; विल्सन के प्रमेय का युग्मन प्रमाण
(\cref{exo:b1:arith:11}) में पहले से ही यह समूह-सैद्धांतिक स्वाद था।
\end{remark}

\begin{remark}[इंटरल्यूड: अमूर्तता क्या खरीदती है]\label{rem:b1:structures:interlude}
यह पूछना उचित है कि सिद्ध करने से क्या प्राप्त हुआ, कहो,
\cref{prop:b1:structures:unique} बजाय एक अमूर्त कानून के लिए
संख्याओं के लिए. उत्तर उत्तोलन है. अब वह दो पंक्ति का तर्क
कवर, एक ही बार में: संरचना के तहत कार्यों के व्युत्क्रम
(\cref{thm:b1:logic:inverse}, जिसकी विशिष्टता का प्रमाण यह दोहराता है
शब्द दर शब्द), व्युत्क्रम मॉड $ n $
(\cref{prop:b1:arith:invmod}), गैर-शून्य वास्तविकताओं के व्युत्क्रम
किसी में भी इकाइयाँ \omterm{def:b1:structures:ring}{अँगूठी}, और --- अदृश्य दृष्टि --- उलटे की
के मैट्रिक्स \cref{ch:b1:matrices}, जहां की विशिष्टता $ A^{-1}$
प्रमाण की एक भी पंक्ति की आवश्यकता नहीं होगी। वही अर्थव्यवस्था कायम है
\cref{prop:b1:structures:kernel} (एक \omterm{def:b1:logic:inj}{इंजेक्शन} मानदंड,
रैखिक के लिए पुन: उपयोग किया गया एमएपीएस में \cref{ch:b1:linmaps}) और के लिए
\omterm{def:b1:structures:subgroup}{उपसमूह} कसौटी. यहां अमूर्तता इसके लिए व्यापकता नहीं है
स्वयं के लिए: यह एक ही प्रमेय को पांच बार सिद्ध करने से इंकार है
पांच नामों के तहत. कीमत --- किन सिद्धांतों पर नज़र रखना
प्रत्येक \omterm{def:b1:logic:statement}{कथन} वास्तव में प्रयोग किया जाता है --- बिल्कुल वही है जो अभ्यास का है
यह अध्याय ट्रेन.
\end{remark}

\begin{remark}[जहाँ इस अध्याय का प्रयोग किया जाता है]\label{rem:b1:structures:whereused}
इस अध्याय की शब्दावली शेष अध्याय का व्याकरण है
आयतन. रिंगों और खेत आयोजन \cref{ch:b1:poly} ($ K[X]$ एक है
\omterm{def:b1:structures:ring}{अँगूठी} नकल $\Z $) और \cref{ch:b1:fractions} ($ K(X)$ इसका है
\omterm{def:b1:structures:field}{मैदान} भिन्नों का); वेक्टर रिक्त स्थान (\cref{ch:b1:vspaces}) हैं
\omterm{def:b1:structures:group}{एबेलियन} \omterm{def:b1:structures:group}{समूह} के साथ \omterm{def:b1:structures:field}{मैदान} उन पर कार्रवाई करना; matrices
(\cref{ch:b1:matrices}) वॉल्यूम का पहला गंभीरता से निर्माण करें
गैर क्रमविनिमेय \omterm{def:b1:structures:ring}{अँगूठी}, और उनके उलटे तत्व a \omterm{def:b1:structures:group}{समूह} किसका
अध्ययन रैखिक बीजगणित ही है। आकृति विज्ञान और \omterm{def:b1:structures:morphism}{कर्नेल} के रूप में लौटें
रैखिक एमएपीएस और शून्य रिक्त स्थान \cref{ch:b1:linmaps} ---
\cref{prop:b1:structures:kernel} \emph{है} \omterm{def:b1:logic:inj}{इंजेक्शन}
उस अध्याय की कसौटी, यहाँ एक बार और सभी के लिए साबित हुई।
\omterm{ex:b1:structures:groups}{सममित समूह}, नीचे सप्ताहांत समस्या का सितारा, आपूर्ति करता है
हस्ताक्षर जिस पर निर्धारक बने होते हैं \cref{ch:b1:det}.
\end{remark}

\section{अभ्यास}

\begin{exercise}[$\star $]\label{exo:b1:structures:1}
पर $ E = \R \setminus \{1\}$, परिभाषित करना $ x * y = x + y - xy $. साबित करें कि
$(E, *)$ एक \omterm{def:b1:structures:group}{एबेलियन} \omterm{def:b1:structures:group}{समूह}. \emph{(पहचान पहचानें और
का उलटा $ x $; स्थिरता की जाँच करें: क्यों है $ x * y \neq 1$?)}
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:structures:2}
निम्नलिखित में से कौन से हैं \omterm{def:b1:structures:group}{समूह}?
\begin{enumerate}
  \item $(\intoo{0}{+\infty}, \times)$;
  \item $(\{-1, 0, 1\}, +)$;
  \item $(\Q^*, \times)$;
  \item  \omterm{def:b1:logic:sets}{तय करना} जोड़ के अंतर्गत विषम पूर्णांकों का।
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:structures:3}
की रचना तालिका लिखिए \omterm{ex:b1:structures:groups}{सममित समूह} $\mathfrak{S}_3$
का $\{1,2,3\}$ (छह आक्षेप: पहचान, तीन स्थानान्तरण, दो
$3$-साइकिल), और दो तत्व प्रदर्शित करते हैं जो आवागमन नहीं करते हैं।
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:structures:4}
साबित करें कि $ H = \{z \in \C^* : \abs z = 1\}$ एक है \omterm{def:b1:structures:subgroup}{उपसमूह} का $(\C^*,
\times)$, ओर वो $\R_+^*$ दूसरा है; है $ H \cup \R_+^*$ ए
\omterm{def:b1:structures:subgroup}{उपसमूह}?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:structures:5}
होने देना $ f \colon (\R, +) \to (\C^*, \times)$, $\theta \mapsto
\eu^{\iu\theta}$. साबित करें कि $ f $ एक है \omterm{def:b1:structures:morphism}{आकारिता}, गणना करें $\ker f $ और
$\operatorname{im} f $, और से निष्कर्ष निकालें
\cref{prop:b1:structures:kernel} वह $ f $ क्या नहीं है \omterm{def:b1:logic:inj}{इंजेक्शन}. प्रतिबंधित करें
इसे बनाने के लिए डोमेन \omterm{def:b1:logic:inj}{इंजेक्शन} यथासंभव बड़े अंतराल पर.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:structures:6}
होने देना $ H, K $ होना \omterm{def:b1:structures:subgroup}{उपसमूह}ों का $ G $. साबित करें कि $ H \cap K $ एक है \omterm{def:b1:structures:subgroup}{उपसमूह},
ओर वो $ H \cup K $ एक है \omterm{def:b1:structures:subgroup}{उपसमूह} \emph{केवल} कब $ H \subseteq K $ या
$ K \subseteq H $. \emph{(अगर $ h \in H \setminus K $ और $ k \in K
\setminus H $, कहाँ पर $ hk $ रहना?)}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:structures:7}
ए \omterm{def:b1:structures:group}{समूह} $ G $ संतुष्ट $ x^2 = e $ सभी के लिए $ x \in G $. साबित करें कि $ G $ है
\omterm{def:b1:structures:group}{एबेलियन}. \emph{(बढ़ाना $(xy)^2$.)}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:structures:8}
में $\Z/18\Z $: इकाइयों की सूची बनाएं और उनका व्युत्क्रम ज्ञात करें $\overline 5$;
हल करना $\overline 5\, x = \overline 7$; हल करना $\overline 6\, x =
\overline 3$ और $\overline 6\, x = \overline{12}$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:structures:9}
सिद्ध करें कि \omterm{def:b1:logic:sets}{तय करना} $\Z[\sqrt 2] = \{a + b\sqrt 2 : a, b \in \Z\}$ एक है \omterm{def:b1:structures:ring}{अँगूठी} (का एक उपरिंग $\R $), ओर वो $1 + \sqrt 2$ इसकी एक इकाई है
असीम रूप से कई विशिष्ट शक्तियों के साथ --- तो $\Z[\sqrt 2]^\times $ है
अनंत, विपरीत $\Z^\times $.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b1:structures:10}
(बूलियन के छल्ले) होने देना $ A $ एक हो \omterm{def:b1:structures:ring}{अँगूठी} जिसमें $ x^2 = x $ हरएक के लिए $ x $.
साबित करें कि $ x + x = 0$ सभी के लिए $ x $, ओर वो $ A $ क्रमविनिमेय है.
\emph{(बढ़ाना $(x+x)^2$ और $(x+y)^2$.)} ऐसा कोई उदाहरण दीजिये
\omterm{def:b1:structures:ring}{अँगूठी} साथ $\mathcal{P}(E)$, सममित अंतर को इस रूप में लेते हुए
गुणन के रूप में जोड़ और प्रतिच्छेदन।
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b1:structures:11}
होने देना $ G $ एक हो \omterm{def:b1:structures:group}{समूह} जिसमें, कुछ के लिए तय किया गया $ n \geq 1$, $(xy)^n =
x^n y^n $, $(xy)^{n+1} = x^{n+1}y^{n+1}$ और $(xy)^{n+2} =
x^{n+2}y^{n+2}$ सभी के लिए $ x, y $. साबित करें कि $ G $ है \omterm{def:b1:structures:group}{एबेलियन}.
\emph{(तीन पहचानों में से, पहले प्राप्त करें $ y^n x = x y^n $, तब
$ y^{n+1} x = x y^{n+1}$, और निष्कर्ष निकालें।)}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:structures:12}
\begin{enumerate}
  \item सभी का निर्धारण करें \omterm{def:b1:structures:group}{समूह} रूपवाद से $(\Z, +)$ को $(\Z, +)$.
  \item सिद्ध करें कि एकमात्र \omterm{def:b1:structures:group}{समूह} \omterm{def:b1:structures:morphism}{आकारिता} से $(\Q, +)$ को
        $(\Z, +)$ शून्य है \omterm{def:b1:structures:morphism}{आकारिता}. \emph{(के लिए $ x \in \Q $ और
        $ n \in \N^*$, तुलना करना $ f(x)$ और $ n\,f(x/n)$.)}
\end{enumerate}
\end{exercise}

\section{समस्या: सममित समूह और 8-पहेली}

\begin{problem}\label{pb:b1:structures:1}
 \omterm{def:b1:structures:group}{समूह} $\mathfrak S_n $ का क्रमपरिवर्तन का $\intint1n $ है
सबसे पुराना \omterm{def:b1:structures:group}{समूह} गणित में और अभी भी सबसे शिक्षाप्रद। यह
समस्या अपना संरचना सिद्धांत खरोंच से बनाती है --- चक्र,
स्थानान्तरण द्वारा पीढ़ी, द \emph{हस्ताक्षर} \omterm{def:b1:structures:morphism}{आकारिता}
$\varepsilon \colon \mathfrak S_n \to \{\pm1\}$ (जिसका अस्तित्व है
वास्तव में गैर-तुच्छ), और \omterm{pb:b1:structures:1}{वैकल्पिक समूह} $\mathfrak A_n $
द्वारा उत्पन्न $3$-साइकिल --- फिर इसे क्लासिकल में भुनाता है
पहेली: में $3 \times 3$ स्लाइडिंग-टाइल गेम, चालों का कोई क्रम नहीं
दो टाइलों की अदला-बदली कर सकते हैं और बाकी सभी चीज़ों को यथास्थान छोड़ सकते हैं।
\index{सममित समूह}\index{हस्ताक्षर}\index{वैकल्पिक समूह}
क्रमपरिवर्तन पर कार्रवाई $\intint1n $; उत्पादों $\sigma\tau $ मतलब
``लागू करो $\tau $ पहला''; $[\,v_1, \dots, v_n]$ को दर्शाता है
\omterm{def:b1:counting:objects}{परिवर्तन} भेज रहा हूँ $ i $ को $ v_i $.

\medskip
\noindent\textbf{Part I --- Cycles and transpositions.}
\begin{enumerate}
  \item औचित्य $\abs{\mathfrak S_n} = n!$
        (\cref{thm:b1:counting:counts}). में $\mathfrak S_3$,
        के दोनों उत्पादों की गणना करें $\sigma = [2, 3, 1]$ और $\tau =
        [1, 3, 2]$, और निष्कर्ष निकालो $\mathfrak S_3$ क्या नहीं है
        \omterm{def:b1:structures:group}{एबेलियन}.
  \item ए \emph{$ k $-चक्र} $(a_1\ a_2\ \dots\ a_k)$ ($ k \geq 2$,
        द $ a_i $ जोड़ीवार अलग) भेजता है $ a_1 \mapsto a_2
        \mapsto \dots \mapsto a_k \mapsto a_1$ और ठीक करता है
        बाकी सब कुछ; इसका \emph{सहायता} है $\{a_1, \dots,
        a_k\}$. साबित करें कि असंयुक्त समर्थन वाले दो चक्र
        आवागमन.
  \item प्रत्येक को सिद्ध करें $\sigma \in \mathfrak S_n $ एक उत्पाद है
        जोड़ीवार असंयुक्त समर्थन वाले चक्रों की, और वह यह
        अपघटन तक अद्वितीय है \omterm{def:b1:structures:order}{आदेश} कारकों का.
        \emph{(प्रत्येक के लिए विचार करें $ i $, क्रम $ i, \sigma(i),
        \sigma^2(i), \dots $: इसे वापस लौटना होगा $ i $; परिणामस्वरूप
        \emph{orbits} \omterm{thm:b1:logic:partition}{PARTITION} $\intint1n $, और $\sigma $ पर कार्य करता है
        प्रत्येक एक चक्र के रूप में।)}
  \item विघटित $\sigma = [4, 1, 5, 2, 3, 7, 8, 6] \in
        \mathfrak S_8$ असंयुक्त चक्रों में. को परिभाषित करना
        \emph{\omterm{def:b1:structures:order}{आदेश}} का $\sigma $ के रूप में
        \cref{def:b1:structures:order}, सिद्ध करें कि \omterm{def:b1:structures:order}{आदेश} एक का
        असंयुक्त चक्रों का उत्पाद उनकी लंबाई का एलसीएम है,
        और गणना करें \omterm{def:b1:structures:order}{आदेश} इस का $\sigma $.
  \item दूरबीन पहचान सिद्ध करें
        \[
        (a_1\ a_2\ \dots\ a_k)
        = (a_1\ a_k)(a_1\ a_{k-1})\cdots(a_1\ a_2) ,
        \]
        और निष्कर्ष निकालो कि प्रत्येक \omterm{def:b1:counting:objects}{परिवर्तन} का एक उत्पाद है
        स्थानान्तरण। लिखना $\sigma $ प्रश्न 4 के जैसे
        उत्पाद.
  \item आगे दिखाएँ कि \emph{नज़दीक} प्रतिस्थापन
        $(i\ \ i{+}1)$ पर्याप्त: के लिए $ a < b $,
        \[
        (a\ b) = (a\ \ a{+}1)(a{+}1\ \ a{+}2)\cdots(b{-}1\ \ b)
        \cdots(a{+}1\ \ a{+}2)(a\ \ a{+}1),
        \]
        का एक उत्पाद $2(b - a) - 1$ आसन्न स्थानान्तरण ---
        एक \emph{विषम} संख्या (यह समता दो बार मायने रखेगी
        नीचे)।
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The signature exists.} के लिए $\sigma
\in \mathfrak S_n $, होने देना
\[
N(\sigma) = \#\bigl\{(i, j) : i < j,\ \sigma(i) >
\sigma(j)\bigr\}
\]
इसकी संख्या हो \emph{इन्वर्ज़न}, और सेट करें $\varepsilon(\sigma) =
(-1)^{N(\sigma)}$.
\begin{enumerate}[resume]
  \item गणना करना $ N $ और $\varepsilon $ पहचान के लिए, ए के लिए
        स्थानान्तरण $(i\ \ i{+}1)$, और के लिए $[2, 3, 1]$.
  \item इसे हर एक के लिए साबित करें $\sigma $ और हर आसन्न
        स्थानान्तरण $\tau = (i\ \ i{+}1)$:
        $ N(\sigma\tau) = N(\sigma) \pm 1$. \emph{(के साथ रचना करना
        $\tau $ दाईं ओर स्थिति में मानों की अदला-बदली करता है $ i $ और
        $ i + 1$; ठीक एक जोड़ा अपना व्युत्क्रम बदलता है
        स्थिति.)}
  \item प्रश्न 6 का उपयोग करते हुए, यह निष्कर्ष निकालें \emph{कोई}
        स्थानांतरण $\tau $, $\varepsilon(\sigma\tau) =
        -\varepsilon(\sigma)$; निष्कर्ष निकालें कि यदि $\sigma $ एक है
        का उत्पाद $ p $ तब स्थानान्तरण $\varepsilon(\sigma)
        = (-1)^p $ --- विशेष रूप से की समता $ p $ पर ही निर्भर करता है
        पर $\sigma $, चुने हुए गुणनखंडन पर नहीं --- और वह
        $\varepsilon \colon \mathfrak S_n \to \{\pm 1\}$ एक है
        \omterm{def:b1:structures:group}{समूह} \omterm{def:b1:structures:morphism}{आकारिता}.
  \item दिखाओ कि ए $ k $-साइकिल पर हस्ताक्षर हैं $(-1)^{k-1}$, और
        वह सामान्य तौर पर $\varepsilon(\sigma) = (-1)^{n -
        c(\sigma)}$, कहाँ $ c(\sigma)$ की कक्षाओं की संख्या है
        $\sigma $ (निश्चित अंक शामिल)।
  \item  \emph{\omterm{pb:b1:structures:1}{वैकल्पिक समूह}} है $\mathfrak A_n =
        \ker\varepsilon $. औचित्य सिद्ध करें कि यह एक है \omterm{def:b1:structures:subgroup}{उपसमूह}, और साबित करो
        $\abs{\mathfrak A_n} = \frac{n!}2$ के लिए $ n \geq 2$.
        \emph{(स्थानांतरण ठीक करें $\tau_0$ और विचार करें $\sigma
        \mapsto \sigma\tau_0$.)}
  \item संगति की जाँच जारी है $\sigma = [4, 1, 5, 2, 3, 7, 8, 6]$:
        गणना करें $\varepsilon(\sigma)$ तीन तरीके--- गिनती करके
        व्युत्क्रम, चक्र प्रकार से प्रश्न 10 के माध्यम से, और से
        प्रश्न 5 में आपकी स्थानान्तरण गणना।
\end{enumerate}

\medskip
\noindent\textbf{Part III --- $\mathfrak A_n $ is generated by
$3$-cycles.}
\begin{enumerate}[resume]
  \item होने देना $ a, b, c, d $ जोड़ीवार अलग हो. दोनों को सत्यापित करें
        पहचान
        \[
        (a\ b)(a\ c) = (a\ c\ b),
        \qquad
        (a\ b)(c\ d) = (a\ c\ b)(a\ c\ d) .
        \]
  \item इसके लिए सिद्ध करें $ n \geq 3$, का प्रत्येक तत्व $\mathfrak
        A_n $ का एक उत्पाद है $3$-चक्र। \emph{(एक सम
        \omterm{def:b1:counting:objects}{परिवर्तन} की सम संख्या का गुणनफल है
        स्थानान्तरण; एक समय में उन दोनों को अवशोषित करें।)}
  \item लिखना $(1\ 2)(3\ 4)$ और यह $5$-चक्र $(1\ 2\ 3\ 4\ 5)$
        स्पष्ट रूप से के उत्पादों के रूप में $3$-चक्र।
  \item संयुग्मन सूत्र सिद्ध करें: प्रत्येक के लिए $\sigma \in
        \mathfrak S_n $,
        \[
        \sigma\,(a_1\ \dots\ a_k)\,\sigma^{-1}
        = \bigl(\sigma(a_1)\ \dots\ \sigma(a_k)\bigr) .
        \]
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The 8-puzzle.} टाइल्स $1, \dots, 8$
ए में स्लाइड करें $3 \times 3$ एक खाली सेल वाला फ़्रेम; ए \emph{कदम}
खाली सेल के निकट एक टाइल को इसमें स्लाइड करता है। कोशिकाओं को क्रमांकित करें
$1, \dots, 9$ (पंक्ति दर पंक्ति; हल की गई स्थिति में टाइल है $ i $ में
सेल $ i $ और सेल में खाली सेल $9$). खाली सेल को इस प्रकार समझें
एक नौवीं टाइल, तो एक स्थिति है एक \omterm{def:b1:counting:objects}{परिवर्तन} $\sigma \in \mathfrak
S_9$ (टाइल $\sigma(i)$ सेल में बैठता है $ i $).
\begin{enumerate}[resume]
  \item दिखाएँ कि एक चाल बदल जाती है $\sigma $ द्वारा $\sigma \circ \tau $
        कहाँ $\tau $ दो कोशिकाओं का स्थानान्तरण है
        सम्मिलित; यह निष्कर्ष निकालें कि प्रत्येक चाल फ़्लिप होती है
        $\varepsilon(\sigma)$.
  \item होने देना $ d(\sigma)$ टैक्सीकैब की दूरी (पंक्तियाँ प्लस) हो
        कॉलम) खाली सेल की वर्तमान सेल और उसके बीच
        होम सेल $9$. दिखाएँ कि प्रत्येक चाल बदलती है $ d $ द्वारा $\pm1$,
        इसलिए प्रत्येक चाल भी पलटती है $(-1)^{d(\sigma)}$. समाप्त करें कि
        \[
        I(\sigma) = \varepsilon(\sigma)\cdot(-1)^{d(\sigma)}
        \]
        है \emph{अचल} हर चाल के तहत.
  \item पहेली की शास्त्रीय असंभवता साबित करें: स्थिति
        जो टाइल्स की अदला-बदली करता है $7$ और $8$ और बाकी सब कुछ छोड़ देता है
        (खाली सेल सहित) स्थान तक नहीं पहुंचा जा सकता
        सुलझी हुई स्थिति से.
  \item हम इस वार्तालाप को स्वीकार करते हैं (इसका प्रमाण एक शिक्षाप्रद है लेकिन
        लंबा प्रेरण): हर स्थिति के साथ $ I = +1$ है
        पहुंच योग्य. ठीक उसका आधा भाग निकालें $8!$ पद
        घर पर खाली सेल का समाधान किया जा सकता है, यानी\
        $\frac{8!}2 = 20\,160$.
  \item प्रश्न 20 से निष्कर्ष निकालें कि पहुंच योग्य टाइल
        घर पर खाली सेल के साथ व्यवस्था बिल्कुल वैसी ही होती है
        \omterm{def:b1:structures:subgroup}{उपसमूह} $\mathfrak A_8 \leq \mathfrak S_8$.
  \item अपरिवर्तनीय के अनुप्रयोग: क्या कोई (ए) तक पहुंच सकता है
        वह स्थान जहाँ टाइलें हों $1, 2, 3$ चक्रीय रूप से क्रमपरिवर्तित हैं
        और बाकी सब कुछ, खाली सेल सहित, घर पर है?
        (बी) वह स्थिति जहां टाइल $5$ और खाली सेल है
        स्थानों का आदान-प्रदान हुआ और अन्य सभी टाइलें घर पर हैं? औचित्य सिद्ध करें
        दोनों उत्तर देते हैं $ I $.
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Synthesis.}
\begin{enumerate}[resume]
  \item इसके लिए सिद्ध करें $ n \geq 3$ केवल \omterm{def:b1:structures:group}{समूह} रूपवाद $ f
        \colon \mathfrak S_n \to \{\pm 1\}$ स्थिर हैं
        \omterm{def:b1:structures:morphism}{आकारिता} और $\varepsilon $. \emph{(प्रश्न 16 और का उपयोग करते हुए
        की क्रमपरिवर्तनशीलता $\{\pm1\}$, दिखाओ $ f $ वही लेता है
        सभी ट्रांसपोज़िशन पर मूल्य।)}
  \item वास्तव में समस्या का उपयोग कहां हुआ: (i) द \omterm{def:b1:structures:morphism}{आकारिता}
        अवधारणा और \cref{prop:b1:structures:kernel}; (ii) द
        गिनती के सिद्धांत \cref{ch:b1:counting}; (iii) द
        सुस्पष्टता मुद्दा जो प्रश्न 8-9 हल करता है? एक
        प्रत्येक वाक्य.
  \item संश्लेषण, एक संक्षिप्त पैराग्राफ में: एक समता फ़ंक्शन,
        एक बार अच्छी तरह से परिभाषित साबित, एक साथ व्यवस्थित करता है
        की आंतरिक संरचना $\mathfrak S_n $ (द \omterm{def:b1:structures:subgroup}{उपसमूह}
        $\mathfrak A_n $), एक भौतिक पहेली का निर्णय करता है, और ---
        सूत्र के माध्यम से $\det = \sum_\sigma
        \varepsilon(\sigma)\cdots$ --- निर्धारकों को परिभाषित करेगा
        में \cref{ch:b1:det}. आवर्ती पैटर्न पर टिप्पणी करें:
        अपरिवर्तनीय ``चालों के सभी अनुक्रमों को आज़माएँ'' को एक में बदल देते हैं
        गणना.
\end{enumerate}
\end{problem}
