\chapter{भेदभाव}\label{ch:b1:derivative}

संजात पूरे हाई स्कूल खंड में गणना की गई; क्या था
प्रमेयों की श्रृंखला गायब है जो गणना को बदल देती है
फ़ंक्शंस के बारे में जानकारी: रोले का प्रमेय, माध्य मान प्रमेय,
और उनके परिणाम --- एकरसता मानदंड, लिप्सचिट्ज़ सीमाएं,
उत्तलता इस अध्याय में सब कुछ परिभाषित कार्यों से संबंधित है
एक \omterm{prop:b1:reals:intervals}{अंतराल} $I $.

\section{व्युत्पन्न}

\begin{definition}\label{def:b1:derivative:def}
$ f \colon I \to \R $ है \emph{पर भिन्न $ x_0 \in
I $}\index{पर भिन्न $ x_0 \in
I $} जब अंतर भागफल
$\frac{f(x) - f(x_0)}{x - x_0}$ की एक (सीमित) \omterm{def:b1:topology:closure}{सीमा} है $ x \to x_0$;
\omterm{def:b1:topology:closure}{सीमा} लिखी है $ f'(x_0)$. समान रूप से:
\[
f(x_0 + h) = f(x_0) + f'(x_0)\,h + h\,\varepsilon(h),
\qquad \varepsilon(h) \xrightarrow[h \to 0]{} 0 ,
\]
ग्राफ तब स्पर्शरेखा रेखा को स्वीकार करता है $ y = f(x_0) + f'(x_0)(x -
x_0)$. भिन्नता पर $ x_0$ तात्पर्य निरंतरता पर $ x_0$ (पढ़ें
प्रदर्शन). $ f $ पर भिन्न है $ I $ जब यह हर बिंदु पर है;
$ f $ वर्ग का है $ C^1$ इसके अलावा जब $ f'$ है निरंतर, और का
कक्षा $ C^k $\index{सीके फ़ंक्शन@$ सी^के $ फ़ंक्शन} कब $ f $ हो सकता है
विभेदित $ k $ कई बार साथ $ f^{(k)}$ निरंतर.
\end{definition}

\begin{example}\label{ex:b1:derivative:notc1}
`` का विलोमविभेदक $\Rightarrow $ निरंतर'' विफल रहता है:
$\abs{\,\cdot\,}$ पर $0$. और भी आश्चर्य की बात है, विभेदक नहीं करता
संकेत करना $ C^1$: समारोह $ f(x) = x^2 \sin\frac 1x $ ($ f(0) = 0$) है
विभेदक हर जगह, साथ $ f'(0) = 0$, लेकिन $ f'(x) = 2x
\sin\frac1x - \cos\frac 1x $ की कोई \omterm{def:b1:topology:closure}{सीमा} नहीं है $0$
(\cref{exo:b1:derivative:2}).
\end{example}

\begin{example}[बिल्कुल एक बिंदु पर भिन्न]\label{ex:b1:derivative:onepoint}
होने देना $ f(x) = x^2$ के लिए $ x \in \Q $ और $ f(x) = 0$ के लिए $ x \notin
\Q $. पर $0$: $\bigl|\frac{f(h) - 0}{h}\bigr| \leq \abs h \to 0$,
इसलिए $ f $ है विभेदक पर $0$ साथ $ f'(0) = 0$. किसी पर $ x_0
\neq 0$, $ f $ सम नहीं है निरंतर: तर्कसंगत और तर्कहीन
अनुक्रम परिवर्तित हो रहे हैं $ x_0$ भेजना $ f $ को $ x_0^2 \neq 0$ और को
$0$ क्रमश (घनत्व, \cref{thm:b1:reals:density}). तो
भिन्नता वास्तव में है \emph{बिंदुवार} धारणा: यह हो सकता है
के एक बिंदु पर पकड़ो $\R $ और कहीं नहीं. के लिए नैतिक
अभ्यास: \omterm{def:b1:logic:statement}{कथन} एकरसता मानदंड या रोले की तरह
की आवश्यकता है यौगिक \emph{एक पर \omterm{prop:b1:reals:intervals}{अंतराल}} --- धारण करना
$ f'(x_0)$ पृथक बिंदुओं पर, चाहे कितने भी हों, कोई वैश्विक समर्थन नहीं करता है
निष्कर्ष जो भी हो.
\end{example}

\begin{theorem}[संचालन]\label{thm:b1:derivative:operations}
अगर $ f, g $ हैं विभेदक पर $ x_0$ (और जहां सूत्र बनते हैं
भाव):
\[
(f + g)' = f' + g', \qquad
(fg)' = f'g + fg', \qquad
\Bigl(\frac fg\Bigr)' = \frac{f'g - fg'}{g^2},
\]
और यदि $ g $ है विभेदक पर $ f(x_0)$:
$\;(g \circ f)'(x_0) = g'\bigl(f(x_0)\bigr)\, f'(x_0)$ (श्रृंखला
नियम\index{शृंखला नियम}).
\end{theorem}

\begin{proof}
योग: तत्काल. उत्पाद: लिखें
\[
f(x)g(x) - f(x_0)g(x_0)
= \bigl(f(x) - f(x_0)\bigr) g(x) + f(x_0)\bigl(g(x) - g(x_0)\bigr),
\]
द्वारा विभाजित करें $ x - x_0$ और चलो $ x \to x_0$ ($ g $ है निरंतर पर $ x_0$).
भागफल: इलाज करें $\frac 1g $ के जरिए $\frac{1/g(x) - 1/g(x_0)}{x - x_0} =
\frac{-1}{g(x)g(x_0)}\cdot\frac{g(x) - g(x_0)}{x - x_0}$, फिर आवेदन करें
उत्पाद नियम. श्रृंखला नियम: साथ $ y_0 = f(x_0)$, परिभाषित करना
$\theta(y) = \frac{g(y) - g(y_0)}{y - y_0}$ के लिए $ y \neq y_0$ और
$\theta(y_0) = g'(y_0)$: $\theta $ है निरंतर पर $ y_0$, और के लिए
$ x \neq x_0$,
\[
\frac{g(f(x)) - g(f(x_0))}{x - x_0}
= \theta\bigl(f(x)\bigr)\cdot \frac{f(x) - f(x_0)}{x - x_0}
\longrightarrow g'(y_0)\, f'(x_0),
\]
सीमाओं की संरचना द्वारा पहला कारक (यह उपकरण संभालता है
मामला $ f(x) = f(x_0)$ सफाई से, जहां भोले ``गुणा करें और विभाजित करें
द्वारा $ f(x) - f(x_0)$'' टूट जाता है)।
\end{proof}

\begin{theorem}[व्युत्क्रम फलन का व्युत्पन्न]\label{thm:b1:derivative:inverse}
होने देना $ f $ होना निरंतर और सख्ती से एकरस पर $ I $, विभेदक
पर $ x_0$ साथ $ f'(x_0) \neq 0$. तब $ f^{-1}$
(\cref{thm:b1:continuity:bijection}) है विभेदक पर $ y_0 =
f(x_0)$, साथ
\[
(f^{-1})'(y_0) = \frac{1}{f'(x_0)} = \frac{1}{f'\bigl(f^{-1}(y_0)\bigr)} .
\]
अगर $ f'(x_0) = 0$, व्युत्क्रम पर एक ऊर्ध्वाधर स्पर्शरेखा है $ y_0$.
\end{theorem}

\begin{proof}
के लिए $ y \to y_0$, \omterm{def:b1:logic:sets}{तय करना} $ x = f^{-1}(y)$: निरंतरता का $ f^{-1}$ देता है $ x
\to x_0$, और
\[
\frac{f^{-1}(y) - f^{-1}(y_0)}{y - y_0}
= \frac{x - x_0}{f(x) - f(x_0)}
= \frac{1}{\dfrac{f(x) - f(x_0)}{x - x_0}}
\longrightarrow \frac{1}{f'(x_0)} .
\]
लंबवत-स्पर्शरेखा दावा: यदि $ f'(x_0) = 0$, प्रदर्शित भागफल
किसी मात्रा का व्युत्क्रम है जिसकी प्रवृत्ति होती है $0$ रखते समय
एक स्थिर चिह्न (के लिए) $ f $ सख्ती से बढ़ रहा है, $\frac{f(x) -
f(x_0)}{x - x_0} > 0$ सभी के लिए $ x \neq x_0$): अंतर
का भागफल $ f^{-1}$ इसलिए प्रवृत्त होता है $+\infty $ (को $-\infty $
के लिए $ f $ घट रहा है)। उलटा रहता है निरंतर लेकिन नहीं है
विभेदक पर $ y_0$ --- इसका ग्राफ, का प्रतिबिंब $ f $का
विकर्ण के पार, ठीक वहीं लंबवत खड़ा है $ f $भाग गया है
क्षैतिज, जैसे $ x^{1/3}$ पर $0$ के विरुद्ध चित्रण करता है $ x^3$.
\end{proof}

\begin{example}[व्युत्क्रम व्युत्पन्न, दो बार]\label{ex:b1:derivative:inverseaction}
प्रमेय शास्त्रीय की पुनर्गणना करता है डेरिवेटिव बिना किसी \omterm{def:b1:topology:closure}{सीमा} के
काम. के लिए $\ln = \exp^{-1}$: पर $ y = \eu^x $,
\[
(\ln)'(y) = \frac{1}{\exp'(x)} = \frac{1}{\eu^{x}} = \frac1y ,
\]
प्रत्येक के लिए मान्य $ y > 0$ तब से $\exp' = \exp $ कभी गायब नहीं होता. के लिए
$\arctan = \tan^{-1}$: पर $ y = \tan x $,
\[
(\arctan)'(y) = \frac{1}{1 + \tan^2 x} = \frac{1}{1 + y^2} ,
\]
का उपयोग करते हुए $\tan' = 1 + \tan^2 > 0$. समापन अंतर्दृष्टि: सूत्र
किसी फ़ंक्शन के बारे में ज्ञान को उसके बारे में ज्ञान में परिवर्तित करता है
एक प्रतिस्थापन की कीमत पर उलटा --- और
प्रतिस्थापन ($ x = \ln y $, $ x = \arctan y $) बिलकुल यही है
\omterm{def:b1:logic:statement}{कथन} कि दो चर विपरीत दिशाओं में रहते हैं
आपत्ति.
\end{example}

\section{रोले और माध्य मान प्रमेय}

\begin{proposition}[आंतरिक चरम]\label{prop:b1:derivative:fermat}
अगर $ f $ है विभेदक एक भूरा \emph{\omterm{def:b1:topology:closure}{आंतरिक भाग}} बिंदु $ x_0$ का $ I $ और
तो फिर, वहां एक स्थानीय चरम है $ f'(x_0) = 0$.
\end{proposition}

\begin{proof}
स्थानीय अधिकतम कहें: वहाँ है $ r > 0$ साथ $ f(x) \leq f(x_0)$
के लिए $\abs{x - x_0} \leq r $, और आंतरिकता दोनों की गारंटी देती है
के पक्ष $ x_0$ के भीतर उपलब्ध हैं $ I $. के लिए $0 < h \leq r $ द
भागफल $\frac{f(x_0 + h) - f(x_0)}{h}$ अंश है $\leq 0$
और हर $> 0$: यह है $\leq 0$, और इसकी \omterm{def:b1:topology:closure}{सीमा} $ f'(x_0)$
inherits $\leq 0$ (व्यापक असमानताएँ \omterm{def:b1:topology:closure}{सीमा} तक पहुँच जाती हैं,
\cref{thm:b1:seq:order}); के लिए $-r \leq h < 0$ भागफल है
$\geq 0$, देना $ f'(x_0) \geq 0$. इस तरह $ f'(x_0) = 0$. (एट ए
समापन बिंदु, केवल एक संकेत उपलब्ध है: निष्कर्ष वहां विफल रहता है ---
सोचो $ x $ पर $\intcc{0}{1}$, अधिकतम पर $1$ साथ यौगिक
$1$.)
\end{proof}

\begin{theorem}[रोले]\label{thm:b1:derivative:rolle}
होने देना $ f $ होना निरंतर पर $\intcc{a}{b}$, विभेदक पर
$\intoo{a}{b}$, साथ $ f(a) = f(b)$. तब $ f'(c) = 0$ कुछ के लिए $ c \in
\intoo{a}{b}$.\index{रोले का प्रमेय}
\end{theorem}

\begin{proof}
चरम मूल्य प्रमेय द्वारा (\cref{thm:b1:continuity:evt}), $ f $
पर अपना अधिकतम और न्यूनतम प्राप्त करता है $\intcc{a}{b}$. यदि दोनों हैं
अंतिम बिंदु पर प्राप्त किया गया, तब से $ f(a) = f(b)$) अधिकतम $=$ मिनट और $ f $
स्थिर है: कोई भी \omterm{def:b1:topology:closure}{आंतरिक भाग} $ c $ काम करता है. अन्यथा एक अति है
एक पर प्राप्त किया \omterm{def:b1:topology:closure}{आंतरिक भाग} बिंदु $ c $, और
\cref{prop:b1:derivative:fermat} देता है $ f'(c) = 0$.
\end{proof}

\begin{theorem}[माध्य मान प्रमेय]\label{thm:b1:derivative:mvt}
होने देना $ f $ होना निरंतर पर $\intcc{a}{b}$, विभेदक पर
$\intoo{a}{b}$. वहाँ मौजूद है $ c \in \intoo{a}{b}$ साथ
\[
f(b) - f(a) = f'(c)\,(b - a) .
\]
\emph{औसत मूल्य असमानता:} यदि इसके अलावा $ m \leq f' \leq M $ पर
$\intoo{a}{b}$, तब $ m(b-a) \leq f(b) - f(a) \leq M(b-a)$; विशेष रूप से $\abs{f'} \leq K $ तात्पर्य यह है $ f $ है $ K $-लिप्सचिट्ज़.
\index{माध्य मान प्रमेय}
\end{theorem}

\begin{proof}
रोले लगाएं $ g(x) = f(x) - \frac{f(b) - f(a)}{b - a}(x - a)$: $ g $
है निरंतर पर $\intcc{a}{b}$, विभेदक अंदर, और $ g(a) =
f(a) = g(b)$. बिंदु पर $ c $ कहाँ $ g'(c) = 0$: $ f'(c) =
\frac{f(b)-f(a)}{b-a}$. असमानता सीमाबद्ध होकर चलती है $ f'(c)$;
लिप्सचिट्ज़ \omterm{def:b1:logic:statement}{कथन} इसे प्रत्येक जोड़ी बिंदुओं पर लागू करता है।
\end{proof}

\begin{example}[न्यूटन की विधि हीरोन की है]\label{ex:b1:derivative:newton}
हल करने की न्यूटन की विधि $ f(x) = 0$ द्वारा वक्र को प्रतिस्थापित करता है
यह वर्तमान अनुमान पर स्पर्शरेखा है $ x_n $ और स्पर्शज्या लेता है
अगले अनुमान के रूप में रूट करें:
\[
0 = f(x_n) + f'(x_n)(x_{n+1} - x_n)
\quad\Longrightarrow\quad
x_{n+1} = x_n - \frac{f(x_n)}{f'(x_n)} .
\]
इसे चालू करें $ f(x) = x^2 - 2$:
\[
x_{n+1} = x_n - \frac{x_n^2 - 2}{2x_n}
= \frac{x_n}{2} + \frac{1}{x_n}
= \frac12\Bigl(x_n + \frac{2}{x_n}\Bigr) :
\]
बिल्कुल बगुला की पुनरावृत्ति (\cref{ex:b1:seq:heron}), दो
सहस्राब्दी पहले। वहां देखी गई द्विघात गति अब है
स्पर्शरेखा चित्र द्वारा समझाया गया: एक साधारण जड़ के पास, वक्र और
स्पर्शरेखा दूसरे क्रम की त्रुटि से भिन्न होती है, इसलिए प्रत्येक चरण मोटे तौर पर होता है
त्रुटि का वर्ग करें --- सामान्य \omterm{def:b1:logic:statement}{कथन} से अनुसरण करता है
टेलर की \omterm{def:b1:topology:closure}{सीमा} \cref{ch:b1:taylor}. समापन अंतर्दृष्टि:
जहां द्विभाजन (\cref{ex:b1:continuity:dichotomy}) केवल उपयोग करता है
निरंतरता और प्रति चरण एक बिट प्राप्त करता है, न्यूटन एक खर्च करता है
यौगिक प्रति चरण सही अंकों की संख्या दोगुनी करने के लिए।
\end{example}

\begin{example}[एक अनुमानक के रूप में माध्य मान प्रमेय]\label{ex:b1:derivative:mvtestimate}
कितना बड़ा है $\sqrt{101}$? प्रमेय को लागू करें $ f(t) = \sqrt t $
पर $\intcc{100}{101}$: कुछ के लिए $ c \in \intoo{100}{101}$,
\[
\sqrt{101} - 10 = \frac{1}{2\sqrt c},
\qquad\text{so}\qquad
\frac{1}{2\sqrt{101}} < \sqrt{101} - 10 < \frac{1}{20} = 0.05 ,
\]
और तबसे $\sqrt{101} < 10.05$, बाईं \omterm{def:b1:topology:closure}{सीमा} से अधिक है
$\frac{1}{20.1} > 0.0497$: इस प्रकार $10.0497 < \sqrt{101} < 10.05$
(सही मूल्य $10.049875\dots $) --- एक से तीन सही दशमलव
यौगिक मूल्यांकन। वैसे ही $\abs{\sin a - \sin b} \leq
\abs{a - b}$ (अवश्यंभावी $\abs{\cos}\leq 1$): लिप्सचिट्ज़ का अनुमान
तब से उपयोग किया जाता है \cref{ch:b1:seq} ये सभी प्रमेय हैं. समापन
अंतर्दृष्टि: माध्य मान प्रमेय एक शून्य-क्रम टेलर सूत्र है
--- यह एक अज्ञात बिंदु पर व्यापार करता है $ c $ एक कठिन असमानता के लिए, और
\cref{ch:b1:taylor} बिल्कुल इसी व्यापार को दोहराएगा।
\end{example}

\begin{corollary}[एकरसता की कसौटी]\label{cor:b1:derivative:monotone}
होने देना $ f $ होना निरंतर पर $ I $, विभेदक पर \omterm{def:b1:topology:closure}{आंतरिक भाग}.
\begin{enumerate}
  \item $ f' \geq 0$ पर \omterm{def:b1:topology:closure}{आंतरिक भाग} $\iff $ $ f $ यह बढ़ रहा है; $ f' = 0$
        $\iff $ $ f $ स्थिर।
  \item अगर $ f' > 0$ बहुत से बिंदुओं को छोड़कर जहां यह लुप्त हो जाता है,
        $ f $ है \emph{कठोरता से} की बढ़ती।
\end{enumerate}
\end{corollary}

\begin{proof}
अगर $ f' \geq 0$: के लिए $ x < y $ में $ I $, माध्य मान प्रमेय पर
$\intcc{x}{y}$ देता है $ f(y) - f(x) = f'(c)(y - x) \geq 0$. इसके विपरीत,
एक बढ़ते फलन के अंतर भागफल हैं $\geq 0$, तो उनके
सीमाएं भी हैं. स्थिर मामला: पिछले को लागू करें $ f'$ और
$-f' \geq 0$. सख्त संस्करण: $ f $ यह बढ़ रहा है; समानता $ f(x) =
f(y)$ के लिए $ x < y $ जम जाएगा $ f $ पर $\intcc{x}{y}$, मजबूर करना $ f' =
0$ वहाँ --- अपरिमित रूप से अनेक बिंदु।
\end{proof}

\begin{example}[समान व्युत्पन्न, असमान कार्य]\label{ex:b1:derivative:intervalmatters}
पर $\R^* = \intoo{-\infty}{0} \cup \intoo{0}{+\infty}$, दोनों
$ f(x) = \ln\abs x $ और $ g(x) = \ln\abs x + \mathbf{1}_{x>0}$
(जोड़ना $1$ केवल दाहिनी आधी पंक्ति पर) संतुष्ट करें $ f' = g' =
\frac1x $. वे एक स्थिरांक से भिन्न नहीं हैं: मानदंड
``$ f' = 0 \implies f $ स्थिरांक'' एक है \emph{\omterm{prop:b1:reals:intervals}{अंतराल}} \omterm{def:b1:logic:statement}{कथन}
--- इसका प्रमाण दो बिंदुओं के बीच माध्य मान प्रमेय चलाता है,
जिसमें झूठ बोलने के लिए उन्हें जोड़ने वाले पूरे खंड की आवश्यकता होती है
डोमेन. प्रत्येक अर्ध-पंक्ति पर अलग-अलग, के आदिम
$\frac1x $ हैं $\ln\abs x + c $ प्रति आधी पंक्ति में एक स्थिरांक के साथ,
कुल मिलाकर दो स्वतंत्र स्थिरांक। \Cref{ch:b1:integration}
यह बढ़िया प्रिंट विरासत में मिला है: किसी फ़ंक्शन का ``'' आदिम है
एक स्थिरांक तक अच्छी तरह से परिभाषित \emph{सभी के ऊपर \omterm{prop:b1:reals:intervals}{अंतराल} इसके
डोमेन}, और एंटीडेरिवेटिव टेबल चुपचाप कनेक्टिविटी मान लेते हैं।
\end{example}

\begin{example}[सख्ती मुफ़्त में]\label{ex:b1:derivative:strictness}
$ x \mapsto x^3$ है \emph{कठोरता से} बढ़ता जा रहा है $\R $ यहां तक कि
हालाँकि यह यौगिक पर गायब हो जाता है $0$: मानदंड का खंड
``$ f' > 0$ बहुत से बिंदुओं को छोड़कर '' बिल्कुल डिज़ाइन किया गया है
ऐसे समतल बिंदुओं के लिए. इसके विपरीत, $ f' \geq 0$ अकेला ही देता है
व्यापक अर्थ में वृद्धि (एक स्थिर कार्य योग्य है), और
ए यौगिक संपूर्ण उपअंतराल पर गायब होने से यह स्थिर हो जाता है
वहां कार्य करें. व्यावहारिक नियम: सख्त दावा करना
एकरसता, शून्यों की सूची बनाएं $ f'$; निश्चित रूप से बहुत सारे (या अधिक)।
आम तौर पर, किसी भी उपअंतराल पर कोई भी) हानिरहित नहीं है, और \omterm{prop:b1:reals:intervals}{अंतराल} का
वे घातक हैं.
\end{example}

\begin{example}[एक संपूर्ण विविधता अध्ययन]\label{ex:b1:derivative:variations}
अध्ययन $ f(x) = x^3 - 3x + 1$ पर $\R $. यौगिक: $ f'(x) =
3(x^2 - 1)$, सकारात्मक पर $\intoo{-\infty}{-1}$, नकारात्मक पर
$\intoo{-1}{1}$, सकारात्मक पर $\intoo{1}{+\infty}$: द्वारा
एकरसता की कसौटी, $ f $ बढ़ता है, फिर घटता है, फिर
स्थानीय अधिकतम के साथ बढ़ता है $ f(-1) = 3$ और एक स्थानीय न्यूनतम
$ f(1) = -1$. सीमाएँ: $\mp\infty $ पर $\mp\infty $. परिणाम,
मध्यवर्ती मान के साथ भिन्नता तालिका को पढ़ें
प्रत्येक मोनोटोन शाखा पर प्रमेय: $ f $ ठीक एक बार गायब हो जाता है
प्रत्येक
\[
\intoo{-\infty}{-1}, \qquad \intoo{-1}{1}, \qquad
\intoo{1}{+\infty}
\]
(जंक्शनों पर मूल्यों के विपरीत संकेत हैं: $3 > 0 > -1$),
तो समीकरण $ x^3 - 3x + 1 = 0$ बिल्कुल तीन असली हैं
जड़ें; संख्यात्मक रूप से वे पास बैठते हैं $-1.88$, $0.35$, $1.53$. द
समापन अंतर्दृष्टि: एक भिन्नता तालिका एक है \emph{प्रमाण युक्ति},
कोई रेखाचित्र नहीं --- मोनोटोन शाखा और चिह्न \omterm{def:b1:counting:objects}{परिवर्तन} बराबर
बिल्कुल एक जड़, और तालिका शाखाओं की गणना करती है
संपूर्ण रूप से।
\end{example}

\begin{theorem}[लीबनिज सूत्र]\label{thm:b1:derivative:leibniz}
अगर $ f, g $ हैं $ n $ टाइम्स विभेदक, ऐसा ही है $ fg $,
और\index{लीबनिज सूत्र}
\[
(fg)^{(n)} = \sum_{k=0}^{n} \binom nk f^{(k)}\, g^{(n-k)} .
\]
\end{theorem}

\begin{proof}
प्रेरण चालू $ n $, द्विपद प्रमेय के बिल्कुल समानांतर। मामला $ n = 1$ उत्पाद नियम है. सूत्र को रैंक पर मानते हुए
$ n $, एक बार फिर अंतर करें:
\[
(fg)^{(n+1)} = \sum_{k=0}^{n} \binom nk
\Bigl( f^{(k+1)} g^{(n-k)} + f^{(k)} g^{(n-k+1)} \Bigr),
\]
फिर पहले योग को पुनः अनुक्रमित करें $ j = k + 1$ और इकट्ठा करें
का गुणांक $ f^{(j)} g^{(n+1-j)}$: यह है $\binom{n}{j-1} +
\binom nj = \binom{n+1}{j}$ पास्कल के नियम से
(\cref{prop:b1:counting:identities}), द \omterm{def:b1:topology:closure}{सीमा} शर्तें $ j = 0$
और $ j = n + 1$ भार उठाते $\binom{n+1}{0} = \binom{n+1}{n+1} =
1$ जैसा कि उन्हें करना चाहिए.
\end{proof}

\begin{example}[कार्रवाई में लीबनिज]\label{ex:b1:derivative:leibnizaction}
गणना करना $\bigl(x^2 \eu^x\bigr)^{(n)}$ के लिए $ n \geq 2$. लेना $ f =
x^2$, किसका डेरिवेटिव जल्दी मरो ($ f' = 2x $, $ f'' = 2$,
$ f^{(k)} = 0$ के लिए $ k \geq 3$), और $ g = \eu^x $: केवल तीन पद
लीबनिज राशि का जीवित रहना,
\[
\bigl(x^2\eu^x\bigr)^{(n)}
= \binom n0 x^2 \eu^x + \binom n1 (2x)\,\eu^x + \binom n2\,
2\,\eu^x
= \eu^x\bigl(x^2 + 2nx + n(n-1)\bigr).
\]
विवेक की जाँच $ n = 1$: $\eu^x(x^2 + 2x)$, जो वास्तव में है
$(x^2\eu^x)'$. समापन अंतर्दृष्टि: एक कारक होने पर लीबनिज़ का उपयोग करें
एक है \omterm{def:b1:poly:def}{बहुपद} --- तो योग ही है $\deg + 1$ शर्तें, और
सूत्र एक \omterm{def:b1:topology:closure}{बंद} रूप है, कोई अमूर्त पहचान नहीं। (दो के लिए
अनंत-जीवित कारक जैसे $\eu^x\sin x $, जटिल
से घातांक \cref{ch:b1:complex} बेहतर उपकरण हैं.)
\end{example}

\section{उत्तलता}

\begin{definition}\label{def:b1:derivative:convex}
$ f \colon I \to \R $ है \emph{उत्तल}\index{उत्तल} कब
प्रत्येक राग ग्राफ़ के ऊपर स्थित है:
\[
\forall x, y \in I,\ \forall t \in \intcc{0}{1}, \quad
f\bigl(tx + (1-t)y\bigr) \leq t f(x) + (1-t) f(y).
\]
($ f $ है \emph{नतोदर} कब $-f $ उत्तल है.)
\end{definition}

\begin{theorem}[विभेदक लक्षण वर्णन]\label{thm:b1:derivative:convexchar}
होने देना $ f $ होना विभेदक पर $ I $. निम्नलिखित समतुल्य हैं:
\begin{enumerate}
  \item $ f $ है \omterm{def:b1:derivative:convex}{उत्तल};
  \item $ f'$ बढ़ता जा रहा है $ I $;
  \item ग्राफ प्रत्येक स्पर्शरेखा के ऊपर स्थित है: $ f(y) \geq f(x) +
        f'(x)(y - x)$ सभी के लिए $ x, y \in I $.
\end{enumerate}
अगर $ f $ दो बार है विभेदक: $ f $ \omterm{def:b1:derivative:convex}{उत्तल} $\iff f'' \geq 0$.
\end{theorem}

\begin{proof}
(1 $\Rightarrow $ 3) उत्तलता के रूप में लिखा गया है $\frac{f(x + t(y-x)) -
f(x)}{t} \leq f(y) - f(x)$ के लिए $ t \in \intoc{0}{1}$; होने देना $ t \to 0^+$:
$ f'(x)(y - x) \leq f(y) - f(x)$.

(3 $\Rightarrow $ 2) के लिए $ x < y $, दो स्पर्शरेखा असमानताएँ $ x $
और कम से $ y $ देना $ f'(x)(y-x) \leq f(y) - f(x) \leq f'(y)(y - x)$,
इस तरह $ f'(x) \leq f'(y)$.

(2 $\Rightarrow $ 1) ठीक करें $ x < y $ और $ t \in \intoo{0}{1}$, और चलो $ z
= tx + (1-t)y \in \intoo{x}{y}$. माध्य मान प्रमेय द्वारा
$\intcc{x}{z}$ और $\intcc{z}{y}$: वहाँ हैं $ c_1 < z < c_2$ साथ
\[
\frac{f(z) - f(x)}{z - x} = f'(c_1) \leq f'(c_2)
= \frac{f(y) - f(z)}{y - z} ,
\]
और समाशोधन विभाजक ($ z - x = (1-t)(y-x)$, $ y - z = t(y-x)$)
बिल्कुल उत्तलता असमानता में पुनर्व्यवस्थित होता है।

दो बार विभेदक मामला: $ f'' \geq 0 \iff f'$ की बढ़ती
(\cref{cor:b1:derivative:monotone}).
\end{proof}

\begin{example}[शास्त्रीय उत्तलता असमानताएँ]\label{ex:b1:derivative:convexineq}
$\exp $ है \omterm{def:b1:derivative:convex}{उत्तल} ($\exp'' = \exp > 0$): यह स्पर्शरेखा है $0$ देता है
$\eu^x \geq 1 + x $ सभी के लिए $ x $. $\ln $ अवतल है: इसकी स्पर्शरेखा है $1$
देता है $\ln x \leq x - 1$; इसके तार देते हैं, के लिए $0 < a \leq b $, द
ज्यामितीय और अंकगणितीय साधनों के बीच असमानता: लेना $ t =
\frac12$ अवतलता में,
\[
\ln\frac{a + b}{2} \geq \frac{\ln a + \ln b}{2} = \ln\sqrt{ab},
\qquad\text{so}\qquad
\sqrt{ab} \leq \frac{a+b}{2} .
\]
सामान्य अंकगणित--ज्यामितीय असमानता है
\cref{exo:b1:derivative:9}.
\end{example}

\begin{example}[खरोंच से एक उत्तलता असमानता]\label{ex:b1:derivative:xlnx}
समारोह $ f(t) = t\ln t $ है \omterm{def:b1:derivative:convex}{उत्तल} पर $\intoo{0}{+\infty}$:
$ f''(t) = \frac1t > 0$. इसके मध्यबिंदु असमानता, से गुणा किया गया
$2$, पढ़ता है: सभी के लिए $ a, b > 0$,
\[
a\ln a + b\ln b \;\geq\; (a + b)\,\ln\frac{a + b}{2} ,
\]
समानता के साथ यदि $ a = b $ (सख्त उत्तलता)। टेस्ट ड्राइव: $ a =
1$, $ b = 3$ देता है $3\ln 3 = 3.296$ ख़िलाफ़ $4\ln 2 = 2.773$.
यह अहानिकर असमानता का दो-बिंदु मामला है
\emph{एन्ट्रापी} तुलना जो जेन्सेन के साथ फिर से प्रकट होती है
असमानता (\cref{exo:b1:derivative:9}) और में
वर्ष 3 खंड की सूचना-सैद्धांतिक स्पर्शोन्मुखता। द
समापन अंतर्दृष्टि: एक असमानता का निर्माण करने के लिए, एक फ़ंक्शन ढूंढें
जिसका दूसरा यौगिक एक चिन्ह है और लिखिए कि उत्तलता क्या है
कहते हैं --- विभेदक लक्षण वर्णन एक संकेत जाँच को बदल देता है
\omterm{def:b1:reals:bounds}{असीम} रूप से अनेक असमानताओं में।
\end{example}

\begin{remark}[डेरिवेटिव के साथ आम समस्याएं]
(मैं) \emph{एक सकारात्मक यौगिक एक बिंदु पर नहीं देता
इसके निकट एकरसता}: $ f(x) = \frac x2 + x^2\sin\frac1x $ (साथ
$ f(0) = 0$) है $ f'(0) = \frac12 > 0$, अभी तक
\[
f'(x) = \frac12 + 2x\sin\frac1x - \cos\frac1x
\]
के बराबर होती है $-\frac12$ हरेक पर $ x_n = \frac{1}{2\pi n}$: प्रत्येक
\omterm{def:b1:topology:open}{अड़ोस-पड़ोस} का $0$ अवरोह शामिल हैं। एकरसता की आवश्यकता है $ f'
\geq 0$ \emph{एक पर \omterm{prop:b1:reals:intervals}{अंतराल}}
(\cref{cor:b1:derivative:monotone}); केवल बिंदुवार संकेत
स्पर्शरेखा रेखा को पार करने को नियंत्रित करता है। (ii) \emph{रोले का
तीन परिकल्पनाएँ सभी सक्रिय हैं}: $\abs x $ पर $\intcc{-1}{1}$
(नहीं \omterm{def:b1:topology:closure}{आंतरिक भाग} भिन्नता), $ x $ पर $\intcc{0}{1}$ (नहीं समाप्त होता है
बराबर), और $ x - \lfloor x\rfloor $ पर $\intcc{0}{1}$ (निरंतरता
में विफल रहता है $1$) प्रत्येक बिल्कुल एक परिकल्पना को तोड़ता है और
निष्कर्ष. (iii) \emph{संजात असंतत हो सकता है, लेकिन
मनमाने ढंग से नहीं}: $ f'$ दोलन कर सकते हैं
(\cref{ex:b1:derivative:notc1}) फिर भी सदैव संतुष्ट करता है
मध्यवर्ती मूल्य संपत्ति (डार्बौक्स,
\cref{exo:b1:derivative:10}): ए यौगिक कभी नहीं कूदता --- अगर
आप एकतरफ़ा `` की गणना करते हैंयौगिक \omterm{def:b1:topology:closure}{सीमा}'' एक छलांग के साथ, आप
एक को विभेदित किया है गैर विभेदक समारोह। (iv) \emph{द
व्युत्क्रम सूत्र की आवश्यकता है $ f' \neq 0$}: $ x \mapsto x^3$ एक चिकना है
सख्ती से बढ़ रही आपत्ति जिसका उलटा $ x^{1/3}$ एक है
ऊर्ध्वाधर स्पर्शरेखा पर $0$ --- व्युत्क्रम की भिन्नता है
बिलकुल कहाँ खो गया $ f'$ गायब हो जाता है
(\cref{thm:b1:derivative:inverse}).
\end{remark}

\begin{remark}[जहां माध्य मान प्रमेय आगे काम करता है]
लगभग हर मात्रात्मक \omterm{def:b1:logic:statement}{कथन} अगले अध्यायों में से यह है
पोशाक में अध्याय का माध्य मान प्रमेय: मौलिक प्रमेय
कैलकुलस का (\cref{ch:b1:integration}) क्षेत्र को अलग करता है
कार्य और एकरसता मानदंड के साथ समाप्त होता है; द
टेलर--लैग्रेंज सूत्र (\cref{ch:b1:taylor}) माध्य मान है
प्रमेय को दोहराया गया $ n $ समय; न्यूटन की विधि का त्रुटि विश्लेषण
और निश्चित-बिंदु पुनरावृत्तियों की (\cref{exo:b1:derivative:11}) है
लिप्सचिट्ज़ फॉर्म; और इस अध्याय की सप्ताहांत समस्या
(\cref{pb:b1:derivative:1}) उसी लिप्सचिट्ज़ को बाउंड में बदल देता है
संख्या सिद्धांत --- के बीच एक प्रतिकर्षण असमानता बीजीय
नंबर और तर्कसंगत, पहले की उपज \omterm{pb:b1:derivative:1}{पारलौकिक संख्या}
इतिहास में. वर्ष 2 खंड में, माध्य मान
\emph{असमानता} समानता होने पर कई चरों में जीवित रहता है
नहीं करता.
\end{remark}

\begin{example}[अवतलता से यंग की असमानता]\label{ex:b1:derivative:young}
होने देना $ p, q > 1$ साथ $\frac1p + \frac1q = 1$. सभी के लिए $ a, b > 0$:
\[
ab \;\leq\; \frac{a^p}{p} + \frac{b^q}{q} .
\]
की अवतलता का एक अनुप्रयोग द्वारा प्रमाण $\ln $ वज़न के साथ
$\frac1p, \frac1q $ (दो-बिंदु जेन्सेन असमानता, जैसा कि
\cref{exo:b1:derivative:9}):
\[
\ln\Bigl(\frac{a^p}{p} + \frac{b^q}{q}\Bigr)
\;\geq\; \frac1p \ln(a^p) + \frac1q \ln(b^q)
= \ln a + \ln b = \ln(ab),
\]
और $\ln $ बढ़ने से लघुगणक की असमानता परिवर्तित हो जाती है
दावा; समानता यदि $ a^p = b^q $ (सख्त अवतलता)। मामला
$ p = q = 2$ अंकगणित-ज्यामितीय असमानता है $ ab \leq
\frac{a^2 + b^2}{2}$ छद्मवेश में. समापन अंतर्दृष्टि: यंग्स
असमानता H का \omterm{pb:b1:logic:1}{बीजगणितीय} बीज है\"वृद्ध और मिन्कोव्स्की
वर्ष 2 खंड की असमानताएँ --- एक अवतलता \omterm{def:b1:logic:statement}{कथन}
के बारे में $\ln $, मानदंडों के लिए काटा गया।
\end{example}

\begin{remark}[इस खंड के अंदर परिप्रेक्ष्य]
 यौगिक खंड समाप्त होने से पहले तीन नए जीवन प्राप्त करता है।
में \cref{ch:b1:taylor} यह पुनरावृत्त करता है: $ n $ डेरिवेटिव एक बिंदु पर
एक में संपीड़ित करें \omterm{def:b1:poly:def}{बहुपद} प्लस एक नियंत्रित त्रुटि, और
माध्य मान प्रमेय लैग्रेंज शेषफल बन जाता है। में
\cref{ch:b1:curves}, विभेदन ज्यामितीय हो जाता है: a के लिए
पैरामीट्रिज्ड वक्र $ t \mapsto (x(t), y(t))$, जोड़ी $(x'(t),
y'(t))$ एक वेग है \emph{वेक्टर}, स्पर्शरेखा बन जाती है
संरेखता, और महत्वपूर्ण बिंदु वर्गीकृत करने के लिए आधार बन जाते हैं। में
\cref{ch:b1:multivar}, एक समय में एक चर स्थिर होता है: आंशिक
डेरिवेटिव इस अध्याय को दो बार दोहराएँ, और स्पर्शरेखा रेखा
एक स्पर्शरेखा तल में बढ़ता है। सभी तीन अध्याय समान विरासत में मिले हैं
व्याकरण --- स्थानीय रैखिक सन्निकटन प्लस एक त्रुटि शब्द ---
सबसे पहले यहीं बात हुई.
\end{remark}

\section{अभ्यास}

\begin{exercise}[$\star $]\label{exo:b1:derivative:1}
अंतर करें (डोमेन निर्दिष्ट करना): $ x^x $;
$\;\ln\bigl(x + \sqrt{x^2+1}\bigr)$; $\;\arctan\frac{1}{x}$;
$\;\sqrt{1 + \eu^{2x}}$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:derivative:2}
पूरा \cref{ex:b1:derivative:notc1}: साबित करें कि $ f(x) = x^2
\sin\frac1x $, $ f(0) = 0$, है विभेदक पर $0$ साथ $ f'(0) = 0$,
ओर वो $ f'$ की कोई \omterm{def:b1:topology:closure}{सीमा} नहीं है $0$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:derivative:3}
माध्य मान प्रमेय या स्पर्शरेखा असमानताओं का उपयोग करके इसे सिद्ध करें
सबके लिए $ x > 0$:
\[
\frac{x}{1 + x} < \ln(1 + x) < x .
\]
परिणाम निकालना $\lim_{n\to\infty} \bigl(1 + \frac xn\bigr)^n = \eu^x $ हरएक के लिए $ x > 0$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:derivative:4}
होने देना $ P $ वास्तविक बनो \omterm{def:b1:poly:def}{बहुपद} साथ $ k $ विशिष्ट वास्तविक जड़ें। साबित करें कि
$ P'$ कम से कम है $ k - 1$ विशिष्ट वास्तविक जड़ें, उनसे जुड़ी हुई
का $ P $. इसका निष्कर्ष निकालें यदि $ P $ इसकी सभी जड़ें वास्तविक हैं, इसलिए यह भी है $ P'$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:derivative:5}
होने देना $ f $ होना विभेदक पर $\R $ साथ $ f' (x)\to \ell $ जैसा $ x \to
+\infty $. साबित करें कि $\frac{f(x)}{x} \to \ell $ \emph{(माध्य मान
प्रमेय पर $\intcc{A}{x}$)}. करता है $ f(x+1) - f(x) \to \ell $ भी पकड़ो?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:derivative:6}
(एक अलग रोले) चलो $ f $ होना $ n $ टाइम्स विभेदक पर $ I $ और
पर गायब हो जाओ $ n + 1$ विशिष्ट बिंदु. साबित करें कि $ f^{(n)}$ पर गायब हो जाता है
कम से कम एक बार. आवेदन: ए \omterm{def:b1:poly:def}{बहुपद} डिग्री का $\leq n $ पर लुप्त हो रहा है
$ n+1$ अंक शून्य है (फिर से)।
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:derivative:7}
होने देना $ f $ दो बार हो विभेदक पर $\intcc{a}{b}$ साथ $ f(a) = f(b)
= 0$ और $ f(x_0) > 0$ कुछ के लिए \omterm{def:b1:topology:closure}{आंतरिक भाग} $ x_0$. साबित करें कि $ f''(c) <
0$ कुछ के लिए $ c \in \intoo{a}{b}$. \emph{(दो माध्य मान प्रमेय और
ढलानों की तुलना।)}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:derivative:8}
फ़ंक्शन का अध्ययन करें $ f(x) = \dfrac{\ln x}{x}$ पर
$\intoo{0}{+\infty}$: विविधताएँ, सीमाएँ, अधिकतम। उसका निष्कर्ष निकालो $ a^b
> b^a $ सभी वास्तविकताओं के लिए $\eu \leq a < b $, और प्रसिद्ध विशेष का निपटारा करें
मामला: कौन सा $\eu^\pi $, $\pi^\eu $ बड़ा है? के विरुद्ध जाँच करें
छोटे पूर्णांक जोड़े $(2,3)$ और $(2,4)$: वे ऐसा व्यवहार क्यों करते हैं
अलग ढंग से?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:derivative:9}
(अंकगणित--ज्यामितीय असमानता) की अवतलता का उपयोग करना $\ln $ के साथ
सामान्य भार (जेन्सेन की असमानता के लिए $ n $ बिंदुओं द्वारा सिद्ध किया जाना है
प्रेरण पर $ n $), इसे सकारात्मक वास्तविकताओं के लिए दिखाएं $ a_1, \dots, a_n $:
\[
\sqrt[n]{a_1 a_2 \cdots a_n} \leq \frac{a_1 + \dots + a_n}{n},
\]
समानता के साथ यदि सभी $ a_i $ बराबर हैं.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b1:derivative:10}
(डार्बौक्स: डेरिवेटिव मध्यवर्ती मान लें) चलो $ f $ होना
विभेदक पर $ I $ और $ a < b $ में $ I $ साथ $ f'(a) < v < f'(b)$.
विचार करके $ g(x) = f(x) - vx $ और बिंदु कहां $ g $ प्राप्त करता है
यह न्यूनतम पर है $\intcc{a}{b}$, साबित करें कि $ f'(c) = v $ कुछ के लिए $ c
\in \intoo{a}{b}$ --- चाहे $ f'$ होने की आवश्यकता नहीं है निरंतर
(\cref{exo:b1:derivative:2}).
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b1:derivative:11}
होने देना $ f \colon \R \to \R $ होना विभेदक साथ $\abs{f'(x)} \leq k
< 1$ सभी के लिए $ x $ (ए \emph{सिकुड़न}). साबित करें कि $ f $ बिलकुल है
एक निश्चित बिंदु $\ell $, और वह हर क्रम $ u_{n+1} = f(u_n)$
में एकत्रित हो जाता है $\ell $ साथ $\abs{u_n - \ell} \leq k^n\abs{u_0 -
\ell}$. \emph{(अस्तित्व: मध्यवर्ती मूल्य प्रमेय को लागू करें
$ f(x) - x $ एक बड़े खंड पर, लिप्सचिट्ज़ बाउंड का उपयोग करके; या उपयोग करें
कॉची मानदंड के साथ पूर्णता।)}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b1:derivative:12}
(कॉची का माध्य मान प्रमेय और एल'अस्पताल का नियम)
\begin{enumerate}
  \item होने देना $ f, g $ होना निरंतर पर $\intcc{a}{b}$,
        विभेदक पर $\intoo{a}{b}$, साथ $ g'$ कभी शून्य नहीं
        वहाँ. साबित करें कि $ g(b) \neq g(a)$ और वह कुछ $ c \in
        \intoo{a}{b}$ संतुष्ट
        \[
        \frac{f(b) - f(a)}{g(b) - g(a)} = \frac{f'(c)}{g'(c)}
        \]
        \emph{(रोले को लागू करें $ h = f - \lambda g $ अधिकार के लिए
        स्थिर $\lambda $)}.
  \item में अस्पताल का नियम निकालें $\frac00$ ए पर फार्म
        बिंदु: यदि $ f(a) = g(a) = 0$ और
        $\frac{f'(x)}{g'(x)} \to \ell $ जैसा $ x \to a^+$, तब
        $\frac{f(x)}{g(x)} \to \ell $.
  \item दिखाएँ कि व्युत्क्रम विफल रहता है: के लिए $ f(x) = x^2\sin\frac1x $
        ($ f(0) = 0$) और $ g(x) = x $, भागफल
        $\frac{f}{g}$ पर एक \omterm{def:b1:topology:closure}{सीमा} है $0$ लेकिन
        $\frac{f'}{g'}$ कोई नहीं है.
\end{enumerate}
\end{exercise}

\section{समस्या: लिउविले की असमानता और पहली
पारलौकिक संख्या}

\begin{problem}[{सप्ताहांत समस्या --- \omterm{pb:b1:logic:1}{बीजगणितीय} संख्याएँ प्रतिकर्षित करना तर्कसंगत: $\abs{x - p/q} \geq C/q^d $, और का अतिक्रमण $\sum 10^{-n!}$}]
\label{pb:b1:derivative:1}
एक वास्तविक संख्या है \emph{\omterm{pb:b1:logic:1}{बीजगणितीय}} जब यह एक अशून्य की जड़ है
\omterm{def:b1:poly:def}{बहुपद} पूर्णांक गुणांक के साथ, और \emph{\omterm{pb:b1:logic:1}{ट्रान्सेंडैंटल}}
अन्यथा। 1844 में लिउविले ने अब तक का पहला नंबर तैयार किया
\emph{साबित} पारलौकिक, और उसके प्रमाण का इंजन यह है
अध्याय का माध्य मान प्रमेय: एक \omterm{pb:b1:derivative:1}{बीजगणितीय संख्या} डिग्री का $ d $
इससे बेहतर तर्कों द्वारा अनुमान नहीं लगाया जा सकता $ C/q^d $ ---तो ए
संख्या अनुमानित \emph{हर शक्ति से तेज़} नहीं हो सकता
बीजीय. यह समस्या असमानता पैदा करती है, निर्माण करती है
लिउविल का नंबर $ L = 0.110001000\dots $ (फैक्टोरियल वाले
स्थिति, अंक मशीनरी के माध्यम से \cref{pb:b1:reals:1}),
अपनी श्रेष्ठता सिद्ध करता है, और कैंटर के प्रतिद्वंद्वी प्रमाण के साथ समाप्त होता है
के लिए प्रभावी \omterm{def:b1:topology:closure}{सीमा} $\sqrt2$ और
$2^{1/3}$.\index{पारलौकिक संख्या}\index{लिउविले का
असमानता}\index{बीजगणितीय संख्या}

\medskip
\noindent\textbf{Part I --- How well can rationals be
approximated?}
\begin{enumerate}
  \item दिखाएँ कि दो भिन्न तर्कसंगतताएँ $\frac ab \neq \frac pq $
        (साथ लिखा है $ b, q \geq 1$) संतुष्ट $\bigl|\frac ab -
        \frac pq\bigr| \geq \frac{1}{bq}$. निष्कर्ष निकालें: यदि $ x =
        \frac ab $ और $0 < \bigl|x - \frac pq\bigr| <
        \frac{1}{bq}$, ऐसा कुछ नही $\frac pq $ मौजूद है --- एक तर्कसंगत
        पैमाने पर अन्य सभी तर्कसंगतताओं को अस्वीकार करता है $\frac 1q $.
  \item इसके लिए सिद्ध करें \emph{प्रत्येक} तर्कसंगत $\frac pq $ ($ q \geq
        1$): $\bigl|\sqrt2 - \frac pq\bigr| \geq
        \frac{1}{4q^2}$ \emph{(यदि दूरी अधिक हो $1$ यह
        स्पष्ट है; अन्यथा बाध्य $\abs{\sqrt2 + p/q} < 4$ और
        शून्येतर पूर्णांक का उपयोग करें $\abs{p^2 - 2q^2} \geq 1$)}.
  \item दूसरी दिशा में: उसकी जाँच करें $(p, q) \mapsto (p +
        2q, p + q)$ बरकरार रखता है $\abs{p^2 - 2q^2} = 1$, उत्पन्न करें
        से $(1,1)$ जोड़े $(3,2)$, $(7,5)$, $(17,12)$,
        $(41,29)$, $(99,70)$, और प्रत्येक को संतुष्ट दिखाएं
        \[
        \Bigl|\sqrt2 - \frac pq\Bigr| =
        \frac{1}{q^2\,(\sqrt2 + p/q)} < \frac{1}{2q^2} :
        \]
        क्रम के अनंत रूप से अनेक सन्निकटन $2$. साथ
        प्रश्न 2: का सन्निकटन प्रतिपादक $\sqrt 2$ है
        \emph{बिल्कुल} $2$.
  \item (डिरिचलेट) चलो $ x $ तर्कहीन हो और $ N \in \N^*$.
        पर विचार करें $ N + 1$ के आंशिक भाग $0, x, 2x,
        \dots, Nx $ में $ N $ बक्से $\intco{\frac kN}{\frac{k +
        1}{N}}$: पिजनहोल सिद्धांत द्वारा
        (\cref{cor:b1:counting:pigeonhole}), एक डिब्बे में दो गिरते हैं।
        परिणाम निकालना $ q \leq N $ और $ p $ साथ $\abs{qx - p} <
        \frac 1N $, इसलिए इसके साथ अपरिमित रूप से अनेक तर्कसंगतताएँ हैं
        $\bigl|x - \frac pq\bigr| < \frac{1}{q^2}$: \emph{प्रत्येक}
        तर्कहीन आदेश देने योग्य है $2$.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Liouville's inequality.} होने देना $ x $ हो
तर्कहीन और बीजीय.
\begin{enumerate}[resume]
  \item उसे शून्येतर पूर्णांकों में दिखाएँ बहुआयामी पद
        पर लुप्त हो रहा है $ x $ वहाँ एक है, कहो $ P $ डिग्री का $ d $,
        साथ \emph{कोई तर्कसंगत जड़ नहीं}; और जांचें $ d \geq 2$
        \emph{(एक कारक को विभाजित करें $ X - \frac ab $ ऊपर $\Q $ और
        स्पष्ट हर; डिग्री $1$ करना होगा $ x $
        तर्कसंगत)}.
  \item प्रत्येक तर्कसंगत के लिए इसे दिखाएँ $\frac pq $ ($ q \geq 1$):
        $\bigl|P\bigl(\frac pq\bigr)\bigr| \geq \frac{1}{q^d}$
        \emph{($ q^d P(p/q)$ एक शून्येतर पूर्णांक है)}.
  \item होने देना $ M = \max_{\intcc{x-1}{x+1}} \abs{P'}$
        (\cref{thm:b1:continuity:evt}). माध्य मान का उपयोग करना
        प्रमेय के बीच $ x $ और $\frac pq $, सिद्ध करना
        \emph{लिउविल की असमानता}: साथ $ C = \min\bigl(1,
        \frac 1M\bigr) > 0$,
        \[
        \Bigl| x - \frac pq \Bigr| \geq \frac{C}{q^{\,d}}
        \qquad\text{for every rational } \frac pq,\ q \geq 1 .
        \]
  \item पुकारना $ x $ ए \emph{लिउविले नंबर} जब प्रत्येक के लिए $ n \in
        \N $ एक तर्कसंगत है $\frac pq $ साथ $ q \geq 2$ और
        $0 < \bigl|x - \frac pq\bigr| < q^{-n}$. साबित करें कि ए
        लिउविल संख्या अपरिमेय है \emph{(प्रश्न 1: चुनें
        $ n $ साथ $2^{\,n-1} > b $)}.
  \item लिउविले के प्रमेय को सिद्ध करें: \emph{एक लिउविले नंबर है
        पारमार्थिक} \emph{(प्रश्न 7 और 8 को संयोजित करें:
        असमानता $ C < q^{\,d-n}$ बड़े पैमाने पर विफल रहता है $ n $)}.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The number $ L $.}
\begin{enumerate}[resume]
  \item होने देना $ L $ मूल्य हो (के अर्थ में)।
        \cref{pb:b1:reals:1}) दशमलव अंक स्ट्रिंग के साथ
        अंक $1$ पदों पर $ n!$ ($ n = 1, 2, 3, \dots $)
        और $0$ अन्यत्र, अर्थात्\ $ L = \sup_k t_k $ साथ $ t_k =
        \sum_{n=1}^{k} 10^{-n!}$. पहले लिखो $25$
        के अंक $ L $.
  \item प्रत्येक के लिए टेल ब्रैकेटिंग सिद्ध करें $ k \geq 1$:
        \[
        10^{-(k+1)!} \;\leq\; L - t_k \;\leq\;
        \frac{10}{9}\,10^{-(k+1)!} \;<\; 2\cdot 10^{-(k+1)!}
        \]
        \emph{(प्रत्येक आंशिक योग से परे बाध्य $ t_k $ एक परिमित द्वारा
        ज्यामितीय योग)}.
  \item लिखना $ t_k = \frac{p_k}{q_k}$ साथ $ q_k = 10^{k!}$.
        दिखाओ $0 < L - \frac{p_k}{q_k} < \frac{2}{q_k^{\,k+1}}$,
        और निष्कर्ष निकालो $ L $ इस अर्थ में एक लिउविल संख्या है
        प्रश्न 8 का.
  \item निष्कर्ष: $ L $ पारलौकिक है --- पहला स्पष्ट
        इतिहास में उदाहरण (लिउविल, 1844)। इसे क्रॉस चेक करें
        सीधे तौर पर अतार्किकता: इसके अंक अंततः नहीं होते हैं
        आवधिक (बढ़ते \omterm{prop:b1:reals:intervals}{अंतराल}, जैसे \cref{pb:b1:reals:1},
        प्रश्न 20).
  \item सामान्यीकरण: प्रत्येक अंक को बदलें $1$ एक मनमानी से
        शून्येतर अंक $ d_n \in \intint{1}{9}$. दिखाएँ कि मान क्या है
        अभी भी एक लिउविले संख्या है, और विकर्ण द्वारा --- घटाएं
        का तर्क \cref{pb:b1:reals:1} (प्रश्न 22) लागू
        इन अंकों के विकल्प --- कि अनगिनत हैं
        अनेक पारलौकिक संख्याएँ इस आकृति का.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The hierarchy of approximation
orders.} कहना $ x $ है \emph{ऑर्डर करने योग्य $\mu $} कब के लिए
कुछ स्थिर $ c > 0$ \omterm{def:b1:reals:bounds}{असीम} रूप से अनेक तर्कसंगत संतुष्ट होते हैं
$\bigl|x - \frac pq\bigr| < \frac{c}{q^{\mu}}$.
\begin{enumerate}[resume]
  \item भाग I--III से पदानुक्रम को इकट्ठा करें: तर्कसंगत हैं
        ऑर्डर करने योग्य $1$ और कोई बेहतर नहीं; $\sqrt 2$ ऑर्डर करने के लिए $2$ और कोई बेहतर नहीं; आदेश देने के लिए हर तर्कहीन
        कम से कम $2$; एक \omterm{pb:b1:derivative:1}{बीजगणितीय संख्या} डिग्री का $ d $ कोई आदेश नहीं
        परे $ d $; हर ऑर्डर के लिए लिउविले नंबर। औचित्य सिद्ध करें
        प्रत्येक दावा प्रासंगिक प्रश्न का हवाला देकर।
  \item बताते हैं कि $ L + r $ प्रत्येक के लिए एक लिउविले नंबर है
        तर्कसंगत $ r = \frac ab $ \emph{(अनुवाद करें
        सन्निकटन: नए हर हैं $ b\,q_k $)}.
        निष्कर्ष निकालें कि लिउविले --- इसलिए पारलौकिक ---
        संख्याएँ हैं \omterm{def:b1:topology:dense}{घना} में $\R $.
  \item (कैंटर, 1874) सिद्ध करें कि \omterm{def:b1:logic:sets}{तय करना} का \omterm{pb:b1:logic:1}{बीजगणितीय} संख्याएँ
        गणनीय है: परिमित रूप से अनेक पूर्णांक होते हैं
        बहुआयामी पद डिग्री प्लस योग के साथ $\abs{\text{coefficients}}$
        से घिरा $ h $, प्रत्येक अधिकतम के साथ $\deg $ जड़ें; ए
        का गणनीय संघ परिमित समुच्चय गणनीय है. चूंकि नहीं
        अनुक्रम निकास $\R $ (\cref{pb:b1:reals:1}, प्रश्न
        22), पारलौकिक संख्याएँ अस्तित्व --- वास्तव में एक रूप में
        बेशुमार \omterm{def:b1:logic:sets}{तय करना}. दो प्रमाणों की तुलना करें: क्या करता है
        लिउविल ने वह दिया जो कैंटर नहीं दे सकता?
  \item प्रश्न 2 से सीधे सिद्ध करें कि $\sqrt 2$ है
        \emph{नहीं} एक लिउविल नंबर \emph{(के लिए $ n \geq 3$,
        असमानता $ q^{-n} > \frac{1}{4q^2}$ \omterm{def:b1:topology:closure}{सीमा} $ q $;
        तब केवल सीमित रूप से बहुत से उम्मीदवार तर्क शेष रह जाते हैं
        से सकारात्मक दूरी पर $\sqrt2$)}. सामान्यीकरण:
        नहीं \omterm{pb:b1:derivative:1}{बीजगणितीय संख्या} लिउविल है.
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Effective constants.}
\begin{enumerate}[resume]
  \item पेल जोड़ी के लिए $(99, 70)$: सत्यापित करें $99^2 - 2\cdot70^2
        = 1$ और सटीक त्रुटि का मूल्यांकन करें
        \[
        \sqrt2 - \frac{99}{70}
        = \frac{-1}{70^2\,\bigl(\sqrt2 + \frac{99}{70}\bigr)},
        \qquad
        \Bigl|\sqrt 2 - \frac{99}{70}\Bigr| \approx 7.2\cdot
        10^{-5} :
        \]
        तीन अंकों वाले भिन्न से पाँच सही अंक।
  \item भाग II चालू करें $ x = 2^{1/3}$, $ P = X^3 - 2$: जाँच करना $ P $
        कोई तर्कसंगत जड़ नहीं है, बाध्य $ M = \max_{\intcc{x-1}{x+1}}
        3t^2 \leq 3\,(1 + 2^{1/3})^2 < 16$, और निष्कर्ष निकालें
        प्रभावी असमानता
        \[
        \Bigl| 2^{1/3} - \frac pq \Bigr| \geq
        \frac{1}{16\,q^3} \qquad \text{for all } \frac pq .
        \]
  \item अदायगी: दिखाएँ कि कोई भी तर्कसंगत अनुमान $2^{1/3}$
        अंदर $10^{-6}$ हर होना चाहिए $ q \geq 40$.
  \item वह आधार दिखाओ $10$ अप्रासंगिक है: बाइनरी
        अनुरूप $\sum_{n\geq1} 2^{-n!}$ (बाइनरी का मान
        फैक्टोरियल स्थिति वाले लोगों के साथ स्ट्रिंग) भी एक है
        लिउविले संख्या, इसलिए पारलौकिक।
\end{enumerate}

\medskip
\noindent\textbf{Part VI --- Frontiers and synthesis.}
\begin{enumerate}[resume]
  \item होने देना $ x^\dagger $ दशमलव स्ट्रिंग का मान हो
        बिल्कुल पदों पर हैं $3^k $ ($ k \geq 0$). दिखाओ
        $ x^\dagger $ ऑर्डर देना लगभग संभव है $3$, और निष्कर्ष निकालें
        लिउविले की असमानता से $ x^\dagger $ न ही है
        तर्कसंगत और न ही द्विघात अपरिमेय। स्पष्ट करें कि क्यों
        विधि वहां रुकती है: ऑर्डर करें $3$ के साथ संगत है
        डिग्री की बीजगणितीयता $\geq 3$, और उस अंतर को \omterm{def:b1:topology:closure}{बंद} करना
        (कोई भी प्रतिपादक $> 2$ प्रत्येक के लिए पर्याप्त है बीजीय
        संख्या) रोथ का प्रमेय है, जो इस खंड से बहुत आगे है।
  \item कैंटर को परिमाणित करें: दिखाएँ कि \omterm{pb:b1:logic:1}{बीजगणितीय} संख्याएँ का
        डिग्री $\leq d $ द्वारा दिए गए बहुआयामी पद गुणांक के साथ
        में $\intint{-H}{H}$ अधिकतम संख्या $ d\,(2H + 1)^{d+1}$.
        (यह सीमितता ही प्रश्न 17 को सफल बनाती है।)
  \item संश्लेषण, प्रत्येक एक वाक्य: (i) एकल का पता लगाएं
        लिउविले के प्रमाण का विश्लेषणात्मक घटक (कौन सा प्रमेय
        इस अध्याय का, जहाँ उपयोग किया गया है); (ii) तनाव बताएं
        जो इसे शक्ति प्रदान करता है (अखंडता बल)। $\abs{P(p/q)} \geq
        q^{-d}$, चिकनाई मना करती है $\abs{P(p/q)} > M\abs{x -
        p/q}$); (iii) लिउविले और कैंटर के प्रमाणों की तुलना करें
        के अस्तित्व का पारलौकिक संख्याएँ; (iv) नाम
        जहां यह खंड फिर से थीम से मिलता है --- सप्ताहांत
        की समस्या \cref{ch:b1:integration} को सिद्ध करता $\pi$
        समान अभिन्नता-बनाम-लघुता द्वारा तर्कहीन
        निचोड़ें, के स्थान पर इंटीग्रल के साथ डेरिवेटिव.
\end{enumerate}
\end{problem}
