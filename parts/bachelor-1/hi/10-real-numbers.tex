\chapter{वास्तविक संख्या}\label{ch:b1:reals}

सारा विश्लेषण एक गुण पर आधारित है जो भेद करता है $\R $ से
$\Q $: प्रत्येक अरिक्त \omterm{def:b1:logic:sets}{तय करना} ऊपर से घिरा हुआ एक है \emph{कम से कम} \omterm{def:b1:reals:bounds}{ऊपरी सीमा}. यह अध्याय इसे सटीक रूप से बताता है, इसका पहला निष्कर्ष निकालता है
परिणाम--- \omterm{thm:b1:reals:archimedes}{आर्किमिडीज़ संपत्ति}, द फर्श समारोह, द
परिमेय और अपरिमेय का घनत्व --- और सेट ऊपर
शब्दावली (sup, inf, max, min) का लगातार उपयोग किया जाता है
\cref{ch:b1:seq} आगे.

\section{ऊपरी सीमा वाली संपत्ति}

\begin{definition}[सीमाएँ, सुप और inf]\label{def:b1:reals:bounds}
होने देना $ A \subseteq \R $ शून्य न हो. एक असली $ M $ एक \emph{ऊपरी सीमा}\index{ऊपरी सीमा} का $ A $ कब $ a \leq M $ सभी के लिए $ a \in A $;
$ A $ है \emph{ऊपर से घिरा हुआ} जब इसकी ऊपरी सीमा होती है (इसी तरह)
नीचे, निचली सीमा के साथ; \emph{घिरे} दोनों का मतलब है)। ए
\emph{अधिकतम} का $ A $ से संबंधित एक ऊपरी सीमा है $ A $.

 \emph{सर्वोच्च} $\sup A $\index{सर्वोच्च} सबसे निचली ऊपरी सीमा है
का $ A $, जब यह अस्तित्व में है; \emph{असीम} $\inf A $\index{असीम} है
सबसे बड़ी निचली सीमा.
\end{definition}

\begin{theorem}[की पूर्णता स्वयंसिद्ध $\R $]\label{thm:b1:reals:sup}
$\R $ एक आदेश है \omterm{def:b1:structures:field}{मैदान} युक्त $\Q $ जिसमें \emph{प्रत्येक
ऊपर परिबद्ध गैर-रिक्त उपसमुच्चय में एक है \omterm{def:b1:reals:bounds}{सर्वोच्च}}.\index{completeness of
R@completeness of $\R $}
\end{theorem}
\admitted

\begin{remark}
हम इसे परिभाषित करने वाले सिद्धांत के रूप में लेते हैं $\R $; एक मॉडल का निर्माण (द्वारा)
डेडेकाइंड कटौती या तर्कसंगत के कॉची अनुक्रम द्वारा) और इसे साबित करना
विशिष्टता ईमानदार लेकिन लंबी है, और आगे के अध्ययन के लिए छोड़ दी गई है। ध्यान दें कि $\Q $ संपत्ति विफल हो जाती है: $\{x \in \Q : x^2 < 2\}$ परिबद्ध है
ऊपर लेकिन कोई न्यूनतम नहीं है \omterm{def:b1:reals:bounds}{ऊपरी सीमा} \emph{में $\Q $} ---इसके उम्मीदवार,
$\sqrt 2$, लापता है (\cref{ex:b1:logic:sqrt2}). पास करके
विपरीत ($\sup(-A) = -\inf A $), प्रत्येक गैर-रिक्त \omterm{def:b1:logic:sets}{तय करना} नीचे बंधा हुआ
एक है \omterm{def:b1:reals:bounds}{असीम}.
\end{remark}

\begin{proposition}[ $\varepsilon $-विशेषता]\label{prop:b1:reals:epsilon}
होने देना $ A \neq \emptyset $ ऊपर और से घिरा हो $ s \in \R $. तब $ s =
\sup A $ अगर और केवल अगर
\begin{enumerate}
  \item $ s $ एक \omterm{def:b1:reals:bounds}{ऊपरी सीमा}: $\forall a \in A $, $ a \leq s $; और
  \item इससे छोटा कुछ भी नहीं है: $\forall \varepsilon > 0$, $\exists a \in
        A $, $ a > s - \varepsilon $.
\end{enumerate}
\end{proposition}

\begin{proof}
अगर $ s = \sup A $: (1) परिभाषा के अनुसार, और (2) के लिए, $ s -
\varepsilon < s $ एक नहीं है \omterm{def:b1:reals:bounds}{ऊपरी सीमा}, जो वास्तव में है
का अस्तित्व $ a > s - \varepsilon $. इसके विपरीत, (1) कहता है $ s $ एक
\omterm{def:b1:reals:bounds}{ऊपरी सीमा}; (2) नहीं कहता $ t < s $ एक \omterm{def:b1:reals:bounds}{ऊपरी सीमा} (लेना $\varepsilon
= s - t $): $ s $ सबसे छोटा है.
\end{proof}

\begin{example}\label{ex:b1:reals:supexamples}
$\sup \intoo{0}{1} = 1$, प्राप्त नहीं (कोई अधिकतम नहीं);
$\sup \intcc{0}{1} = 1 = \max $. के लिए $ A = \{1 - \frac 1n : n \in
\N^*\}$: $\sup A = 1$, प्राप्त नहीं हुआ; $\inf A = \min A = 0$. एक अधिकतम,
जब यह अस्तित्व में है, है \omterm{def:b1:reals:bounds}{सर्वोच्च}; का पूरा बिंदु $\sup $ होना है
एक विकल्प जब अधिकतम मौजूद नहीं होता है।
\end{example}

\begin{example}[व्यवहार में सर्वोच्चता की गणना]\label{ex:b1:reals:supcompute}
के दो पूर्ण वर्कआउट \cref{prop:b1:reals:epsilon}.

\emph{ \omterm{def:b1:logic:sets}{तय करना} $ A = \{x + \frac1x : x > 0\}$.} हरएक के लिए $ x > 0$,
$ x + \frac1x - 2 = \frac{(\,\sqrt x - 1/\sqrt x\,)^2}{1} \geq 0$,
इसलिए $2$ निचली सीमा है; और $2 = 1 + \frac11 \in A $: इसलिए
$\inf A = \min A = 2$, पर प्राप्त किया गया $ x = 1$. ऊपर, $ A $ है
\omterm{def:b1:reals:bounds}{असीम}ित ($ x + \frac1x > x $ किसी से भी अधिक हो सकता है $ M $ द्वारा
\cref{thm:b1:reals:archimedes}): $\sup A $ में मौजूद नहीं है $\R $
(यह है $+\infty $ में $\overline\R $).

\emph{द \omterm{def:b1:logic:sets}{तय करना} $ B = \bigl\{\frac{m}{m + n} : m, n \in
\N^*\bigr\}$.} प्रत्येक तत्व में निहित है $\intoo{0}{1}$, इसलिए $0$ और
$1$ सीमाएँ हैं. न ही प्राप्त किया गया है: $\frac{m}{m+n} = 1$ होगा
बल $ n = 0$. के लिए \omterm{def:b1:reals:bounds}{सर्वोच्च}, जमाना $ n = 1$ और चलो $ m $ बढ़ना:
$\frac{m}{m+1} = 1 - \frac{1}{m+1} > 1 - \varepsilon $ जैसे ही
$ m + 1 > \frac1\varepsilon $ (आर्किमिडीज़): $\sup B = 1$.
सममित रूप से ($ m = 1$, $ n $ बड़ा), $\inf B = 0$. समापन
अंतर्दृष्टि: एक पिन करने के लिए \omterm{def:b1:reals:bounds}{सर्वोच्च}, एक अच्छी तरह से चुना गया \emph{एक-पैरामीटर
पथ} के अंदर \omterm{def:b1:logic:sets}{तय करना} पर्याप्त है --- यहीं रास्ता $ n = 1$ --- और यह $\varepsilon $-चरित्र-चित्रण और कुछ नहीं मांगता।
\end{example}

\begin{example}[अनंत दर्पण]\label{ex:b1:reals:infmirror}
 \omterm{def:b1:reals:bounds}{असीम} उसका अपना है $\varepsilon $-विशेषता, प्राप्त
से \cref{prop:b1:reals:epsilon} के माध्यम से $\inf A = -\sup(-A)$:
$ i = \inf A $ आईएफएफ $ i $ सीमा $ A $ नीचे और, प्रत्येक के लिए
$\varepsilon > 0$, कुछ $ a \in A $ है $ a < i + \varepsilon $. ए
एक साथ दोनों सीमाओं के साथ कसरत करें: चलो
\[
A = \Bigl\{(-1)^n + \frac1n : n \in \N^*\Bigr\}
= \Bigl\{0,\ \tfrac32,\ -\tfrac23,\ \tfrac54,\ -\tfrac45,\
\dots\Bigr\} .
\]
यहां तक कि सूचकांक भी देते हैं $1 + \frac1n \leq \frac32$, समानता के साथ
$ n = 2$: चूंकि विषम-सूचकांक मान भी हैं $\leq 0 <
\frac32$, हम पाते हैं $\sup A = \max A = \frac32$. विषम सूचकांक देते हैं
$-1 + \frac1n > -1$, की ओर घट रहा है $-1$: का प्रत्येक तत्व
$ A $ है $> -1$, और $-1 + \varepsilon $ द्वारा पीटा जाता है $-1 +
\frac1n $ विषम के लिए $ n > \frac1\varepsilon $: $\inf A = -1$, नहीं
हासिल किया. एक \omterm{def:b1:logic:sets}{तय करना}, प्रदर्शन पर सभी चार व्यवहार: a \omterm{def:b1:reals:bounds}{सर्वोच्च}
वह अधिकतम है, ए \omterm{def:b1:reals:bounds}{असीम} यह न्यूनतम नहीं है.
\end{example}

\begin{remark}[सुपर और इन्फ के साथ आम नुकसान]
अधिकांश खोए गए अंकों के लिए चार त्रुटियाँ जिम्मेदार हैं। (मैं) \emph{भ्रामक
$\sup $ और $\max $}: $\sup A $ से संबंधित होने की आवश्यकता नहीं है $ A $; लिखना $\max $
का एक तत्व प्रदर्शित करने के बाद ही $ A $ वह एक है \omterm{def:b1:reals:bounds}{ऊपरी सीमा}.
(ii) \emph{सख्त असमानताओं को पारित करना \omterm{def:b1:reals:bounds}{सर्वोच्च}}: अगर $ a <
b $ सभी के लिए $ a \in A $, कोई केवल निष्कर्ष निकाल सकता है $\sup A \leq b $ ---
गवाह $ A = \intoo{0}{1}$, $ b = 1$. (iii) \emph{लिखना $\sup A $
वैधता की जाँच करने से पहले}: प्रतीक की आवश्यकता है $ A $ गैररिक्त और
ऊपर सीमाबद्ध (\cref{met:b1:reals:supproofs}); $\sup \emptyset $
और $\sup \N $ में अपरिभाषित हैं $\R $ (के सम्मेलन
$\overline\R $ एक अलग, स्पष्ट कार्य हैं)। (iv) \emph{सेट
परिचालन}: $\sup(A \cup B) = \max(\sup A, \sup B)$ हमेशा, लेकिन
कुछ भी सामान्य नहीं है $ A \cap B $ --- यह खाली हो सकता है, और
तब भी जब यह नहीं है, $\sup(A \cap B)$ बहुत नीचे हो सकता है
$\min(\sup A, \sup B)$: लेना $ A = \{0, 2\}$ और $ B = \{0, 3\}$,
कहाँ $\sup(A \cap B) = 0$.
\end{remark}

\begin{example}[परिमित समुच्चयों में मैक्सिमा होता है --- चुपचाप इस्तेमाल किया जाने वाला एक लेम्मा]\label{ex:b1:reals:finitemax}
प्रत्येक परिमित अरिक्त $ F \subseteq \R $ अधिकतम (और ए) है
न्यूनतम)। तत्वों की संख्या पर प्रेरण: एक सिंगलटन
$\{a\}$ है $\max = a $; यदि दावा मान्य है $ n $-तत्व सेट
और $ F $ है $ n + 1$ तत्व, कोई भी चुनें $ a \in F $: द \omterm{def:b1:logic:sets}{तय करना} $ F
\setminus \{a\}$ अधिकतम है $ m $, और $\max F $ है $ m $ अगर $ a
\leq m $, अन्य $ a $. इसमें कोई पूर्णता शामिल नहीं है --- यह शुद्ध है
ऑर्डर प्लस इंडक्शन, पहले से ही मान्य है $\Q $ --- फिर भी लेम्मा
एक ईमानदार का हकदार है \omterm{def:b1:logic:statement}{कथन} क्योंकि आने वाले प्रमाण आह्वान करते हैं
यह चुपचाप: \omterm{thm:b1:reals:floor}{ज़मीन} नीचे निर्माण (''ए \omterm{def:b1:logic:sets}{तय करना} पूर्णांकों का
एक सीमित सीमा में फँसे हुए का एक महानतम तत्व है''), प्रत्येक
$\max(\abs{u_0}, \dots, \abs{u_{N-1}}, \dots)$ में बंधे
\cref{ch:b1:seq}, प्रत्येक ``अनेक में से सबसे बड़ा लें
$\delta $'' में \cref{ch:b1:continuity}. अनंत सेट कहाँ हैं
मैक्सिमा डाई और सुप्रीमा टेक ओवर: यह अध्याय मौजूद है
अनंत मामला.
\end{example}

\begin{theorem}[आर्किमिडीज़ संपत्ति]\label{thm:b1:reals:archimedes}
हरएक के लिए $ x \in \R $ वहाँ है $ n \in \N $ साथ $ n > x $. समान रूप से:
सबके लिए $\varepsilon > 0$ और $ y > 0$, कुछ एकाधिक $ n\varepsilon $
से अधिक है $ y $.\index{आर्किमिडीज़ संपत्ति}
\end{theorem}

\begin{proof}
मान लीजिए नहीं: कुछ $ x $ एक \omterm{def:b1:reals:bounds}{ऊपरी सीमा} का $\N $. तब $ s = \sup \N $
मौजूद है (\cref{thm:b1:reals:sup}). द्वारा
\cref{prop:b1:reals:epsilon}~(2)साथ $\varepsilon = 1$, वहाँ है $ n
\in \N $ साथ $ n > s - 1$; परन्तु फिर $ n + 1 \in \N $ और $ n + 1 > s $,
उसका खंडन कर रहा हूँ $ s $ एक \omterm{def:b1:reals:bounds}{ऊपरी सीमा}. दूसरे फॉर्म के लिए,
चलो $\varepsilon > 0$ और $ y > 0$: पहला फॉर्म जिस पर आवेदन किया गया है $ x
= \frac{y}{\varepsilon}$ का उत्पादन $ n \in \N $ साथ $ n >
\frac{y}{\varepsilon}$, और से गुणा करना $\varepsilon > 0$
(जो सख्त असमानताओं को बरकरार रखता है) देता है $ n\varepsilon > y $.
इसके विपरीत, दूसरा रूप $\varepsilon = 1$ और $ y = x $
के लिए पहला पुनर्प्राप्त करता है $ x > 0$, और $ n = 1$ संभालती है $ x \leq 0$:
दो \omterm{def:b1:logic:statement}{कथन} पूर्णतया समतुल्य हैं।
\end{proof}

\begin{example}[आर्किमिडीज़ काम पर]\label{ex:b1:reals:archimedeswork}
तीन तत्काल उपयोग, बाद में लगातार आवश्यकता। (मैं) \emph{नहीं
सकारात्मक वास्तविक प्रत्येक के नीचे है $\frac1n $}: अगर $0 < \varepsilon $,
चुनना $ n > \frac1\varepsilon $; तब $\frac1n < \varepsilon $. में
दूसरे शब्द, $\R $ इसमें कोई अपरिमेय-अनौपचारिक शामिल नहीं है
``$\frac1n $ मनमाने ढंग से छोटा हो जाता है'' बिल्कुल यही प्रमेय है।
(ii) \emph{स्पष्ट सीमाएँ}: कितना बड़ा होना चाहिए $ n $ के लिए हो
$\frac{1}{n^2} \leq 10^{-6}$? इतना ही काफी है $ n \geq 10^3$
--- आर्किमिडीज़ ऐसी गारंटी देता है $ n $ अस्तित्व में है, और बीजगणित
उनका पता लगाता है. (iii) \emph{शक्तियाँ किसी भी सीमा को हरा देती हैं}: $2^n \geq n +
1$ (प्रेरण), तो हर किसी के लिए $ M $ की कुछ शक्ति $2$ से अधिक है $ M $:
डायडिक्स के लिए उपयोग की जाने वाली ज्यामितीय वृद्धि
\cref{exo:b1:reals:8}. समापन अंतर्दृष्टि: आर्किमिडीयन
संपत्ति ``टेक'' फॉर्म के प्रत्येक वाक्यांश के पीछे लाइसेंस है
$ n $ काफी बड़ा'' --- अब से हम उस वाक्यांश का स्वतंत्र रूप से उपयोग करते हैं,
और यह उदाहरण इसका एकमुश्त औचित्य है।
\end{example}

\begin{theorem}[फ़्लोर फ़ंक्शन]\label{thm:b1:reals:floor}
हरएक के लिए $ x \in \R $ वहाँ बिल्कुल एक पूर्णांक है,
\emph{ज़मीन}\index{ज़मीन} $\lfloor x \rfloor $, साथ
\[
\lfloor x \rfloor \leq x < \lfloor x \rfloor + 1 .
\]
\end{theorem}

\begin{proof}
\emph{अस्तित्व।}  \omterm{def:b1:logic:sets}{तय करना} $ E = \{k \in \Z : k \leq x\}$ गैर-रिक्त है:
द्वारा \cref{thm:b1:reals:archimedes} वहाँ है $ m \in \N $ साथ $ m > -x $,
और तब $-m < x $, इसलिए $-m \in E $. यह ऊपर (किसी पूर्णांक से) घिरा हुआ है
$ n > x $, जो एक ही कारण से मौजूद है), इसलिए, एक होने के नाते \omterm{def:b1:logic:sets}{तय करना} का
पूर्णांक परिमित सीमा में फंसे हुए हैं $\intint{-m}{n}$, इसमें एक है
सबसे बड़ा तत्व $ k = \max E $. तब $ k \leq x $, और $ k + 1 \notin E $
मतलब $ x < k + 1$.

\emph{विशिष्टता.} अगर $ k $ और $ k'$ दोनों असमानताओं को संतुष्ट करते हैं,
फिर $ k \leq x < k' + 1$ देता है $ k \leq k'$, और सममित रूप से $ k'
\leq k $.
\end{proof}

\begin{example}[अभ्यास में फर्श]\label{ex:b1:reals:floorpractice}
$\lfloor 3.7 \rfloor = 3$, $\lfloor 5 \rfloor = 5$, और $\lfloor
-3.7 \rfloor = -4$: द \omterm{thm:b1:reals:floor}{ज़मीन} जाता है \emph{नीचे}, की ओर नहीं $0$.
विशिष्टता के दो परिणाम \cref{thm:b1:reals:floor} वह
हम चुपचाप उपयोग करेंगे. सबसे पहले, के लिए $ n \in \Z $,
\[
\lfloor x + n \rfloor = \lfloor x \rfloor + n ,
\]
क्योंकि $\lfloor x \rfloor + n $ दोनों को संतुष्ट करने वाला एक पूर्णांक है
असमानताओं को परिभाषित करना $ x + n $ --- और केवल एक पूर्णांक करता है।
दूसरा, $\lfloor \, \cdot \, \rfloor $ घट नहीं रहा है: यदि $ x \leq
y $ तब $\lfloor x \rfloor \leq x \leq y < \lfloor y \rfloor + 1$,
और एक पूर्णांक $< \lfloor y \rfloor + 1$ है $\leq \lfloor y
\rfloor $. हालाँकि, सावधान रहें $\lfloor 2x \rfloor \neq 2\lfloor
x \rfloor $ सामान्य तौर पर: $ x = 0.6$ देता है $\lfloor 1.2 \rfloor = 1
\neq 0 = 2\lfloor 0.6 \rfloor $.

क्या \emph{है} सत्य एक कार्यशील पहचान है जो बनाए रखने योग्य है (हर्माइट की,
इसके सरलतम मामले में): प्रत्येक वास्तविक के लिए $ x $,
\[
\lfloor x \rfloor + \Bigl\lfloor x + \frac12 \Bigr\rfloor
= \lfloor 2x \rfloor .
\]
लिखना $ x = \lfloor x\rfloor + u $ साथ $ u \in \intco{0}{1}$ और
दो मामलों को अलग करें. अगर $ u < \frac12$: बाईं ओर है $\lfloor
x\rfloor + \lfloor x\rfloor = 2\lfloor x\rfloor $, और $2x =
2\lfloor x\rfloor + 2u $ साथ $2u \in \intco{0}{1}$, तो सही है
पक्ष है $2\lfloor x\rfloor $ बहुत। अगर $ u \geq \frac12$: बायां
पक्ष है $\lfloor x\rfloor + (\lfloor x\rfloor + 1)$, और $2u \in
\intco{1}{2}$ दाहिनी ओर बनाता है $2\lfloor x\rfloor + 1$. द
समापन अंतर्दृष्टि: $\lfloor x + \frac12\rfloor $ है
\emph{गोलाई} का $ x $ निकटतम पूर्णांक तक, इसलिए पहचान
ऐसा कहते हैं \omterm{thm:b1:reals:floor}{ज़मीन} प्लस राउंडिंग बराबर \omterm{thm:b1:reals:floor}{ज़मीन} दोहरे का --- और
मामला भिन्नात्मक भाग में विभाजित हो गया $ u $ मानक है
प्रत्येक के पीछे तकनीक \omterm{thm:b1:reals:floor}{ज़मीन} पहचान
(\cref{exo:b1:reals:2,exo:b1:reals:3} उस पर भी चलाएँ)।
\end{example}

\begin{theorem}[का घनत्व $\Q $ और का $\R \setminus \Q $]\label{thm:b1:reals:density}
किन्हीं दो वास्तविकताओं के बीच $ x < y $ वहाँ एक तर्कसंगत और एक झूठ है
तर्कहीन.\index{घनत्व}
\end{theorem}

\begin{proof}
\emph{एक तर्कसंगत.} द्वारा \cref{thm:b1:reals:archimedes}, चुनना $ n \in
\N^*$ साथ $ n > \frac{1}{y - x}$, इसलिए $ ny - nx > 1$. होने देना $ m = \lfloor
nx \rfloor + 1$. एक तरफ पर, $ nx < \lfloor nx \rfloor + 1 = m $
(\cref{thm:b1:reals:floor}); दूसरे पर, $ m = \lfloor nx \rfloor +
1 \leq nx + 1 < ny $. द्वारा विभाजित करना $ n $: $ x < \frac mn < y $.

\emph{एक तर्कहीन.} पिछले बिंदु को जोड़ी पर लागू करें $ x -
\sqrt 2 < y - \sqrt 2$: कुछ तर्कसंगत $ q $ उनके बीच स्थित है, और फिर
$ q + \sqrt 2 \in \intoo{x}{y}$ तर्कहीन है (यदि $ q + \sqrt 2$ थे
तर्कसंगत, ऐसा ही होगा $\sqrt 2$).
\end{proof}

\begin{example}[घनत्व का प्रमाण चलाना]\label{ex:b1:reals:densityrun}
प्रमाण एक एल्गोरिथम है; आइए हम इस पर अमल करें $ x = 1.414$ और
$ y = \sqrt 2$. तब से $1.4142^2 = 1.99996164 < 2$, हमारे पास है $\sqrt 2
> 1.4142$, इसलिए $ y - x > 0.0002$ और $\frac{1}{y - x} < 5000$: विकल्प $ n = 5000$ वैध है. तब $ nx = 7070$, इसलिए $ m = \lfloor
7070 \rfloor + 1 = 7071$, और तर्कसंगत उत्पादन किया गया है
\[
\frac{m}{n} = \frac{7071}{5000} = 1.4142,
\qquad
1.414 < 1.4142 < \sqrt 2 .
\]
समापन अंतर्दृष्टि: प्रमाण की आवश्यकता है $ n $ से थोड़ा ही बड़ा
$\frac{1}{y-x}$, और का पहला गुणक लौटाता है $\frac 1n $
आगे $ x $. घनत्व कोई अमूर्त चमत्कार नहीं है --- यह लंबा है
भेस में विभाजन, एक विषय सप्ताहांत में विस्तार से विकसित हुआ
समस्या (\cref{pb:b1:reals:1}).
\end{example}

\begin{remark}[जहाँ आगे पूर्णता का प्रयोग किया जाता है]
\Cref{thm:b1:reals:sup} इसका एकमात्र गै\omterm{pb:b1:logic:1}{र-बीजगणितीय} सिद्धांत है
पुस्तक, और विश्लेषण का प्रत्येक अस्तित्व प्रमेय वह स्वयंसिद्ध है
अलग-अलग कपड़े पहनना: मोनोटोन अभिसरण प्रमेय
(\cref{ch:b1:seq}), बोल्ज़ानो--वीयरस्ट्रैस प्रमेय
(\cref{ch:b1:topology}), मध्यवर्ती मूल्य और चरम मूल्य
प्रमेय (\cref{ch:b1:continuity}), और की परिभाषा
एक के रूप में अभिन्न \omterm{def:b1:reals:bounds}{सर्वोच्च} कम रकम का (\cref{ch:b1:integration}).
वर्ष 3 खंड माप सिद्धांत और हिल्बर्ट रिक्त स्थान का निर्माण करता है
वही एकल स्वयंसिद्ध. जब आने वाले अध्यायों में कोई प्रमाण उत्पन्न होता है
वास्तविक संख्या हवा से बाहर, छुपे हुए की तलाश करें \omterm{def:b1:reals:bounds}{सर्वोच्च}.
\end{remark}

\begin{remark}[विसंगति और घनत्व के बीच]
$\Z $ और $\Q $ भीतर विपरीत छोर पर बैठो $\R $: प्रत्येक के आसपास
पूर्णांक में लंबाई का अंतर होता है $1$ जिसमें कोई अन्य न हो
(विवेक---यही बनाता है \omterm{thm:b1:reals:floor}{ज़मीन} अच्छी तरह से परिभाषित),
जबकि किन्हीं दो वास्तविकताओं के बीच अपरिमित रूप से कई तर्कसंगतताएँ होती हैं
(घनत्व). उल्लेखनीय रूप से, के लिए \emph{additive \omterm{def:b1:structures:subgroup}{उपसमूह}ों} का $\R $
बीच में कुछ भी नहीं है: \cref{exo:b1:reals:9} यह साबित करता है
ऐसा \omterm{def:b1:structures:subgroup}{उपसमूह} दोनों में से कोई एक रूप है $\alpha\Z $ (अलग) या
सघन --- एक द्वंद्व जो घनत्व को शक्ति प्रदान करता है $\{\sin n\}$ में
\cref{ch:b1:seq} और का रचनात्मक राक्षस
\cref{pb:b1:continuity:1}. सामान्य सेटबेशक, मिश्रण करें
स्वतंत्र रूप से व्यवहार करें: $\Z \cup \Q\cap\intcc{0}{1}$ दूर तक असतत है
दूर और बीच में घना।
\end{remark}

\begin{method}[सुप और इन्फ़ के साथ समानता सिद्ध करना]\label{met:b1:reals:supproofs}
साबित करना $\sup A = s $: जाँच करें कि $ s $ सीमा $ A $ ऊपर, फिर उत्पादन,
प्रत्येक के लिए $\varepsilon > 0$ (या एक अनुक्रम के लिए $\varepsilon = \frac
1n $), का एक तत्व $ A $ ऊपर $ s - \varepsilon $. सुप्रीमा की तुलना करने के लिए,
उपयोग करें: $ A \subseteq B \implies \sup A \leq \sup B $; और सभी के लिए $ a, b $:
$\sup(A + B) = \sup A + \sup B $, कहाँ $ A + B = \{a + b\}$
(\cref{exo:b1:reals:5}). कभी मत लिखो $\sup A $ जानने से पहले $ A $ है
गैर-रिक्त और ऊपर से घिरा हुआ।
\end{method}

\section{अंतराल}

\begin{proposition}[अंतरालों की विशेषता]\label{prop:b1:reals:intervals}
एक उपसमुच्चय $ I \subseteq \R $ एक \emph{अंतराल}\index{अंतराल} (एक
परिचित प्रकार के $\intoo{a}{b}$, $\intcc{a}{b}$, $\intco{a}{b}$,
$\intoc{a}{b}$, आधी पंक्तियाँ, $\R $, $\emptyset $, सिंगलटन) यदि और
केवल अगर यह है \emph{उत्तल}:
\[
\forall x, y \in I,\ \forall z \in \R, \quad
x \leq z \leq y \implies z \in I .
\]
\end{proposition}

\begin{proof}
प्रत्येक सूचीबद्ध प्रकार स्पष्ट रूप से उत्तल है। इसके विपरीत, चलो $ I $ उत्तल हो और
गैररिक्त. \omterm{def:b1:logic:sets}{तय करना} $ a = \inf I $ अगर $ I $ नीचे से घिरा हुआ है, अन्यथा $ a =
-\infty $; वैसे ही $ b = \sup I $ या $+\infty $. हम दावा करते हैं $\intoo{a}{b}
\subseteq I \subseteq \intcc{a}{b}$ (स्पष्ट सम्मेलनों के साथ
$\pm\infty $). दूसरा समावेशन सीमा की परिभाषा है। के लिए
पहले, चलो $ z \in \intoo{a}{b}$: तब से $ z > a $, $ z $ एक नहीं है
निचली सीमा (या $ a = -\infty $), इतना कुछ $ x \in I $ है $ x < z $;
इसी तरह कुछ $ y \in I $ है $ y > z $; उत्तलता डालता है $ z \in I $.

दोहरे समावेशन से प्रकार को पढ़ना बाकी है
$\intoo{a}{b} \subseteq I \subseteq \intcc{a}{b}$: द सेट
एक खुले के बीच निचोड़ा हुआ \omterm{prop:b1:reals:intervals}{अंतराल} और इसका बंद होना अलग है
$\intoo{a}{b}$ केवल (सीमित) की उपस्थिति या अनुपस्थिति से
समापनबिंदु. स्पष्ट रूप से: यदि $ a, b \in \R $, चार संभावनाएँ
के लिए $(a \in I,\ b \in I)$ देना $\intoo{a}{b}$, $\intco{a}{b}$,
$\intoc{a}{b}$, $\intcc{a}{b}$ (पतित मामलों सहित
$ a = b $: सिंगलटन यदि $ a \in I $); अगर $ a = -\infty $ और $ b \in
\R $, एक मिलता है $\intoo{-\infty}{b}$ या $\intoc{-\infty}{b}$;
सममित रूप से के लिए $ a \in \R $, $ b = +\infty $; और $ a = -\infty $,
$ b = +\infty $ देता है $ I = \R $. प्रत्येक मामला सूची में है: हो गया।
\end{proof}

\begin{remark}[उत्तलता सही परीक्षण क्यों है?]
प्रस्ताव परिवर्तित करता है a \emph{ज्यामितिक} परिभाषा (एक सूची
दस आकृतियों में से) a \emph{एक पंक्ति तार्किक} परीक्षण, और परीक्षण
वास्तव में कोई इसका उपयोग करता है: यह साबित करने के लिए कि a \omterm{def:b1:logic:sets}{तय करना} एक \omterm{prop:b1:reals:intervals}{अंतराल},
कभी भी इस बात का पीछा न करें कि यह दस आकृतियों में से कौन सी है --- उत्तलता सत्यापित करें
और प्रस्ताव को प्रकार से सुलझाने दें। मध्यवर्ती
का मूल्य प्रमेय \cref{ch:b1:continuity} सटीक रूप से बताया जाएगा
इस तरह (``एक की सतत छवि \omterm{prop:b1:reals:intervals}{अंतराल} एक
\omterm{prop:b1:reals:intervals}{अंतराल}''), और इसका प्रमाण उत्तलता उत्पन्न करता है, नहीं
आकार.
\end{remark}

\begin{remark}[विस्तारित वास्तविक रेखा]
दो प्रतीकों को जोड़कर काम करना सुविधाजनक है $\overline\R = \R
\cup \{-\infty, +\infty\}$, सम्मेलनों के साथ $\sup A = +\infty $
कब $ A $ ऊपर और से घिरा नहीं है $\sup \emptyset = -\infty $. तब
\emph{प्रत्येक} का उपसमुच्चय $\R $ एक \omterm{def:b1:reals:bounds}{सर्वोच्च} में $\overline\R $ --- ए
में सीमाओं के लिए सांकेतिक आराम का स्वतंत्र रूप से उपयोग किया जाता है \cref{ch:b1:seq}.
\end{remark}

\begin{example}[में कंप्यूटिंग $\overline\R $]\label{ex:b1:reals:extendedcompute}
लागू सम्मेलनों के साथ: $\sup \Z = +\infty $, $\inf \Z =
-\infty $; के लिए $ A = \{n + (-1)^n n : n \in \N\} = \{0, 4, 0, 8,
\dots\} \cup \{0\}$, $\sup A = +\infty $ (सम शर्तें $2n $ हैं
\omterm{def:b1:reals:bounds}{असीम}ित) और $\inf A = \min A = 0$; और $\sup\emptyset =
-\infty \leq \inf\emptyset = +\infty $ --- एक \omterm{def:b1:logic:sets}{तय करना} किसका
\omterm{def:b1:reals:bounds}{सर्वोच्च} है \emph{छोटे} इसके मुकाबले \omterm{def:b1:reals:bounds}{असीम}, एक अनुस्मारक कि
सम्मेलनों को बनाने के लिए चुना जाता है $\sup $ बढ़ रहा है और $\inf $
समावेशन के संबंध में कमी:
\[
A \subseteq B \implies \sup A \leq \sup B
\quad\text{and}\quad \inf A \geq \inf B ,
\]
अब कोई शून्यता चेतावनी के साथ मान्य है। सम्मेलन क्या करते हैं
\emph{नहीं} प्रदान अंकगणित है: $+\infty + (-\infty)$ और $0
\times (+\infty)$ अपरिभाषित रहें, और प्रत्येक \omterm{pb:b1:logic:1}{बीजगणितीय}
सुप्रीमा के हेरफेर को पहले जांचना चाहिए कि यह कभी भी इनका निर्माण नहीं करता है।
विस्तारित रेखा बहीखाता है, संख्या प्रणाली नहीं।
\end{example}

\begin{example}[सुप्रीमो जो बच गया $\Q $]\label{ex:b1:reals:supescape}
इस पर लौटे \omterm{def:b1:logic:sets}{तय करना} प्रारंभिक टिप्पणी का, $ A = \{x \in \Q : x^2 <
2\}$, और इसकी गणना करें \omterm{def:b1:reals:bounds}{सर्वोच्च} \emph{में $\R $}. यह शून्य नहीं है
($1 \in A $) और ऊपर से घिरा हुआ $1.5$ (अगर $ x > 1.5$ तब $ x^2 >
2.25 > 2$), इसलिए $ s = \sup A $ मौजूद है. हम दावा करते हैं $ s = \sqrt 2$ (द
वास्तविक संख्या निर्मित \cref{exo:b1:reals:12}). \omterm{def:b1:reals:bounds}{ऊपरी सीमा}: प्रत्येक
$ a \in A $ संतुष्ट $ a < \sqrt2$ --- के लिए $ a \leq 0$ यह है
स्पष्ट, और के लिए $ a > 0$, $ a \geq \sqrt2$ देगा $ a^2 \geq 2$.
इससे छोटा कोई काम नहीं: दिया गया $ t < \sqrt2$, घनत्व
(\cref{thm:b1:reals:density}) एक तर्कसंगतता प्रदान करता है $ q $ साथ
$\max(1, t) < q < \sqrt 2$, और तब $ q^2 < 2$, इसलिए $ q \in A $
से अधिक है $ t $. द्वारा \cref{prop:b1:reals:epsilon}, $ s = \sqrt2
\notin \Q $. समापन अंतर्दृष्टि: \omterm{def:b1:reals:bounds}{सर्वोच्च} एक का \omterm{def:b1:logic:sets}{तय करना} का
तर्कसंगतों को तर्कसंगत होने की आवश्यकता नहीं है --- पूर्णता बिल्कुल सही है
इसका वादा करो $\R $, विपरीत $\Q $, कभी नहीं होने देता \omterm{def:b1:reals:bounds}{सर्वोच्च} पलायन;
यह उदाहरण अध्याय की प्रारंभिक टिप्पणी है, जो अब सिद्ध हो चुका है
की ओर इशारा करने के बजाय।
\end{example}

\begin{remark}[इस खंड के अंदर परिप्रेक्ष्य]
अध्याय के तीन उपकरणों के आगे अलग-अलग करियर हैं।
\omterm{def:b1:reals:bounds}{सर्वोच्च} विश्लेषण आधा चलाता है: मोनोटोन सीमाएँ
(\cref{ch:b1:seq}), अभिन्न की परिभाषा
(\cref{ch:b1:integration}), और, की ज्यामिति में
\cref{ch:b1:euclid}, एक बिंदु से एक उपस्थान तक की दूरी ---
एक \omterm{def:b1:reals:bounds}{असीम} वह ऑर्थोगोनल प्रक्षेपण न्यूनतम में बदल जाता है।
फर्श समारोह जहां भी असतत मिलता है वहां लौट आता है
निरंतर: अंक विस्तार (इस अध्याय की सप्ताहांत समस्या),
डिरिचलेट का पिजनहोल सन्निकटन
(\cref{pb:b1:derivative:1}), योगों की अभिन्न तुलना
(\cref{ch:b1:series}). घनत्व तर्क एक विधि में अपग्रेड हो जाते हैं
में \cref{ch:b1:continuity}: सतत कार्यों की एक पहचान
केवल जाँच की आवश्यकता है $\Q $ --- कॉची की आधी कार्यक्षमता
समीकरण (\cref{pb:b1:continuity:1}) बिल्कुल वही चाल है। कब
इस बारे में संदेह है कि इस खंड में एक प्रमाण को अपना अस्तित्व कहाँ से मिलता है
\omterm{def:b1:logic:statement}{कथन}, उत्तर लगभग हमेशा होता है: यह अध्याय।
\end{remark}

\section{अभ्यास}

\begin{exercise}[$\star $]\label{exo:b1:reals:1}
निर्धारित करें (प्रमाण के साथ) सुपर, इंफ, मैक्स, मिन --- जब वे मौजूद हों ---:
\[
A = \Bigl\{\frac{1}{n} : n \in \N^*\Bigr\},
\qquad
B = \Bigl\{\frac{(-1)^n n}{n+1} : n \in \N\Bigr\},
\qquad
C = \{x \in \R : x^2 < 3\}.
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:reals:2}
इसे सभी के लिए साबित करें $ x, y \in \R $: $\lfloor x \rfloor + \lfloor y
\rfloor \leq \lfloor x + y \rfloor \leq \lfloor x \rfloor + \lfloor y
\rfloor + 1$, और यह कि दोनों सीमाएँ प्राप्त हो गई हैं।
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:reals:3}
इसे हर एक के लिए साबित करें $ x \in \R $ और $ n \in \N^*$:
$\Bigl\lfloor \frac{\lfloor nx \rfloor}{n} \Bigr\rfloor = \lfloor x
\rfloor $.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:reals:4}
होने देना $ A \subseteq B $ का गैर-रिक्त उपसमुच्चय हो $\R $, $ B $ घिरा हुआ. सिद्ध करना
$\inf B \leq \inf A \leq \sup A \leq \sup B $.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:reals:5}
गैर-रिक्त परिबद्ध के लिए $ A, B \subseteq \R $, परिभाषित करना $ A + B = \{a + b : a
\in A,\ b \in B\}$ और $-A = \{-a : a \in A\}$. सिद्ध करना:
\[
\sup(A + B) = \sup A + \sup B,
\qquad
\sup(-A) = -\inf A .
\]
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:reals:6}
होने देना $ f, g \colon E \to \R $ बंधे हुए कार्य हों। सिद्ध करना
\[
\sup_{x \in E}\, \bigl(f(x) + g(x)\bigr) \leq \sup_{x \in E} f(x) +
\sup_{x \in E} g(x),
\]
और एक उदाहरण दीजिए जहां असमानता सख्त है। ऐसा क्यों नहीं होता
विरोधाभास \cref{exo:b1:reals:5}?
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:reals:7}
साबित करें कि $\sqrt 2 + \sqrt 3$ तर्कहीन है. \emph{(इसे वर्गित करें और
की अतार्किकता का उपयोग करें $\sqrt 6$, के माध्यम से सिद्ध किया जाना है
\cref{exo:b1:arith:7}.)}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:reals:8}
सिद्ध करें कि \omterm{def:b1:logic:sets}{तय करना} $ D = \bigl\{\frac{m}{2^n} : m \in \Z,\ n \in
\N\bigr\}$ डायडिक परिमेय का घनत्व सघन है $\R $: किन्हीं दो के बीच
वास्तविकता एक द्विघात तर्कसंगत है।
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b1:reals:9}
होने देना $ G $ एक हो \omterm{def:b1:structures:subgroup}{उपसमूह} का $(\R, +)$ साथ $ G \neq \{0\}$. \omterm{def:b1:logic:sets}{तय करना} $\alpha =
\inf\,(G \cap \intoo{0}{+\infty})$. सिद्ध करना:
\begin{enumerate}
  \item अगर $\alpha > 0$, तब $ G = \alpha\Z $;
  \item अगर $\alpha = 0$, तब $ G $ में सघन है $\R $.
\end{enumerate}
उसका निष्कर्ष निकालो $\Z + \sqrt 2\,\Z $ में सघन है $\R $.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b1:reals:10}
के लिए $ A, B $ अरिक्त सेट सकारात्मक वास्तविकताओं की, चलो $ AB = \{ab : a \in
A, b \in B\}$. सिद्ध करना $\sup(AB) = \sup A \cdot \sup B $ (बाउंडेड केस),
और उदाहरण के द्वारा दिखाएँ कि सकारात्मकता आवश्यक है।
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:reals:11}
एक गैररिक्त सीमा के लिए $ A \subseteq \R $, को परिभाषित करो
\emph{व्यास}
\[
\operatorname{diam} A = \sup\,\{\abs{a - a'} : a, a' \in A\} .
\]
साबित करें कि $\operatorname{diam} A = \sup A - \inf A $, ओर वो
$\intcc{\inf A}{\sup A}$ सबसे छोटा बंद है \omterm{prop:b1:reals:intervals}{अंतराल}
युक्त $ A $.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b1:reals:12}
होने देना $ y > 0$ और $ E = \{x \geq 0 : x^2 \leq y\}$. साबित करें कि $ E $ है
गैर-रिक्त और ऊपर से घिरा हुआ, और वह $ s = \sup E $ संतुष्ट $ s^2 =
y $ \emph{(निकालें $ s^2 < y $ और $ s^2 > y $ प्रत्येक में प्रदर्शन करके
मामला, एक छोटा सा $ h > 0$ की परिभाषा का खंडन करता है
\omterm{def:b1:reals:bounds}{सर्वोच्च})}. यह निष्कर्ष निकालें कि प्रत्येक $ y > 0$ एक अद्वितीय वर्गमूल है
$\sqrt y > 0$ ओर वो $ y \mapsto \sqrt y $ बढ़ता जा रहा है
$\intoo{0}{+\infty}$.
\end{exercise}

\section{समस्या: अंकों का विस्तार और परिमेय की लय}

\begin{problem}[{सप्ताहांत समस्या --- $ b $-एडिक विस्तार: अस्तित्व, विशिष्टता और आवधिकता की विशेषताएँ $\Q $}]
\label{pb:b1:reals:1}
हर वास्तविक में $\intco{0}{1}$ प्रत्येक आधार में एक अंक का विस्तार होता है
$ b \geq 2$; एक बार अनुगामी तारों का विस्तार अद्वितीय है
अंक $ b - 1$ गैरकानूनी हैं; और अंततः यह वास्तव में आवधिक है
जब संख्या तर्कसंगत हो. यह समस्या तीनों तथ्यों को सिद्ध करती है
अकेले पूर्णता स्वयंसिद्ध से --- कोई क्रम नहीं, कोई शृंखला नहीं:
केवल \omterm{def:b1:reals:bounds}{सर्वोच्च}, द \omterm{thm:b1:reals:archimedes}{आर्किमिडीज़ संपत्ति} और यह फर्श समारोह
--- और अंकीय रूप में कैंटर के विकर्ण तर्क के साथ समाप्त होता है।
लगातार, $ b \geq 2$ एक निश्चित पूर्णांक है ( \emph{आधार}), ए
\emph{अंक} का एक तत्व है $\intint{0}{b-1}$, और एक अंक
स्ट्रिंग $(d_n)_{n \geq 1}$ है \emph{उचित} जब यह नहीं है
अंततः बराबर $ b - 1$ (अर्थात्: प्रत्येक के लिए $ N $ वहाँ है $ n >
N $ साथ $ d_n \leq b - 2$).\index{बी-एडिक विस्तार@$ बी $-एडिक
विस्तार}\index{दशमलव विस्तार}

\medskip
\noindent\textbf{Part I --- Digits by hand.} का लम्बा विभाजन $ p $
द्वारा $ q $ आधार में $ b $: वर्तमान शेष को इससे गुणा करें $ b $, विभाजित करें
द्वारा $ q $, भागफल को अगले अंक के रूप में रिकार्ड करें, शेषफल रखें।
\begin{enumerate}
  \item आधार में $10$, एल्गोरिथम चालू करें $\frac 18$ और पर
        $\frac 17$, प्रत्येक चरण पर अंक रिकॉर्ड करना \emph{और}
        शेष. जाँच करें कि शेष के लिए $\frac 17$
        के माध्यम से चक्र $1, 3, 2, 6, 4, 5$ और वह अंक
        $142857$ फिर हमेशा के लिए दोहराएँ.
  \item आधार की गणना करें-$2$ का विस्तार $\frac 13$ और का
        $\frac{5}{16}$, और आधार-$3$ का विस्तार $\frac 12$.
        ध्यान दें: एक संख्या समाप्त हो जाती है, अन्य दो दोहराती हैं ---
        और $\frac 12$, तो आधार में वश में $10$, हमेशा के लिए दोहराता है
        आधार $3$.
  \item के लिए $ x = \frac pq \in \intco{0}{1}$ सबसे कम शब्दों में, दिखाओ
        एल्गोरिदम द्वारा उत्पादित अंक अंततः हैं
        सब $0$ यदि और केवल यदि शेष $ b^N p \bmod q $
        कुछ के लिए गायब हो जाता है $ N $, अगर और केवल अगर $ q $ विभाजित कुछ
        शक्ति $ b^N $, यदि और केवल यदि प्रत्येक \omterm{def:b1:arith:prime}{मुख्य} का कारक $ q $
        विभाजित $ b $. जाँच करना: $\frac{1}{20}$ आधार पर समाप्त होता है
        $10$, आधार में नहीं $3$.
  \item को परिभाषित करो \emph{काट-छांट} $ s_n = \lfloor b^n x \rfloor
        / b^n $. के लिए $ x = \sqrt 2$ और $ b = 10$, गणना करें $ s_0,
        \dots, s_4$ प्रत्येक चरण पर यह सत्यापित करके कि दो
        लगातार वर्ग फैला हुआ $2$ (उदाहरण के लिए $1.4142^2
        = 1.99996164 < 2 < 2.00024449 = 1.4143^2$), और जांचें
        $ s_n \leq \sqrt 2 < s_n + 10^{-n}$ हर बार.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Existence, from the \omterm{def:b1:reals:bounds}{सर्वोच्च}.} हल करना
$ x \in \intco{0}{1}$ और सेट करें $ A_n = \lfloor b^n x \rfloor $ और
$ d_n = A_n - b\,A_{n-1}$ के लिए $ n \geq 1$.
\begin{enumerate}[resume]
  \item दिखाओ $ A_0 = 0$ और $ b\,A_{n-1} \leq A_n \leq b\,A_{n-1} +
        b - 1$; निष्कर्ष निकालें कि प्रत्येक $ d_n $ एक अंक है.
  \item बताते हैं कि $ s_n := A_n b^{-n}$ संतुष्ट
        \[
        s_n = \sum_{k=1}^{n} d_k\,b^{-k}
        \qquad\text{and}\qquad
        s_n \leq x < s_n + b^{-n} .
        \]
  \item सिद्ध करना $ b^n \geq n + 1$ प्रेरण द्वारा, फिर उसे दिखाओ
        $(s_n)$ घट नहीं रहा है और वह $ x = \sup_n s_n $
        \emph{(उपयोग \cref{prop:b1:reals:epsilon} और
        \cref{thm:b1:reals:archimedes})}.
  \item दिखाएँ कि स्ट्रिंग $(d_n)$ उचित है: यदि $ d_k = b - 1$
        सभी के लिए $ k > N $, गणना करें $ s_n $ के लिए $ n > N $ एक परिमित द्वारा
        ज्यामितीय योग और विरोधाभासी प्रश्न 6.
  \item इसके विपरीत, चलो $(e_n)_{n \geq 1}$ कोई भी उचित अंक हो
        स्ट्रिंग और $ t_n = \sum_{k=1}^n e_k b^{-k}$. बताते हैं कि $ y
        = \sup_n t_n $ मौजूद है, निहित है $\intco{0}{1}$, और
        संतुष्ट करता है $ t_n \leq y < t_n + b^{-n}$ हरएक के लिए $ n $
        \emph{(सख्त असमानता के लिए, एक अंक का उपयोग करें $ e_m \leq b
        - 2$ साथ $ m > n $)}. परिणाम निकालना $\lfloor b^n y \rfloor = b^n
        t_n $, तो वह अंक $ y $, के अर्थ में
        प्रश्न 5, बिल्कुल वही हैं $ e_n $.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Uniqueness, order, shift.}
\begin{enumerate}[resume]
  \item प्रश्न 5--9 को एकत्रित करें \emph{$ b $-आदिक विस्तार
        प्रमेय}: द एमएपीएस $ x \mapsto (d_n)$ और $(e_n) \mapsto
        \sup_n t_n $ के बीच परस्पर विपरीत आक्षेप हैं
        $\intco{0}{1}$ और यह \omterm{def:b1:logic:sets}{तय करना} उचित अंक स्ट्रिंग की. में
        विशेष रूप से कोई भी दो अलग-अलग उचित तार समान नहीं होते हैं
        मूल्य.
  \item अब अनुचित स्ट्रिंग्स की अनुमति दें। दिखाएँ कि एक स्ट्रिंग के साथ $ e_n
        = b - 1$ सभी के लिए $ n > M $ (साथ $ M \geq 0$ न्यूनतम) है
        मूल्य $ t_M + b^{-M}$; समाप्त करें कि $0.999\dots = 1$ में
        आधार $10$, और वह वास्तविक दो अंकों के साथ
        अभ्यावेदन बिल्कुल वैसा ही है $ b $-एडिक अंश
        $ m/b^N \in \intoo{0}{1}$ --- हर दूसरे असली बस है
        एक, अनुचित तारों के बीच भी।
  \item सिद्ध कीजिए कि प्रश्न 10 का आक्षेप है
        शब्दकोषीय क्रम के लिए आदेश-संरक्षण: यदि
        के उचित तार $ x $ और $ y $ सबसे पहले सूचकांक में भिन्नता $ m $,
        तब $ x < y $ अगर और केवल अगर $ d_m < e_m $.
  \item (शिफ्ट लेम्मा) चलो $ x \in \intco{0}{1}$ अंक हैं
        $(d_n)$. दिखाएँ कि भिन्नात्मक भाग $ bx $ अंक हैं
        $(d_{n+1})_{n \geq 1}$ \emph{(गणना करें $\lfloor b^n(bx -
        A_1)\rfloor $ का उपयोग करते हुए $\lfloor u - K \rfloor = \lfloor u
        \rfloor - K $ पूर्णांक के लिए $ K $)}, और प्रेरण द्वारा घटाएँ
        वह का भिन्नात्मक भाग $ b^m x $ अंक हैं
        $(d_{n+m})_{n \geq 1}$.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Rationality is periodicity.} होने देना $ x
= \frac pq \in \intco{0}{1}$ न्यूनतम शर्तों में हो और $ r_n = b^n p
\bmod q $ यूक्लिडियन प्रभाग का शेष भाग $ b^n p $ द्वारा
$ q $.
\begin{enumerate}[resume]
  \item दिखाओ $ A_n = \dfrac{b^n p - r_n}{q}$ और $ r_n =
        (b\,r_{n-1}) \bmod q $.
  \item दिखाओ $ d_n = \Bigl\lfloor \dfrac{b\,r_{n-1}}{q}
        \Bigr\rfloor $: प्रत्येक अंक पिछले का एक कार्य है
        अकेले रहना. यह बिल्कुल लम्बा विभाजन है
        भाग I
  \item पिजनहोल सिद्धांत लागू करें
        (\cref{cor:b1:counting:pigeonhole}) को $ r_0, \dots, r_q $
        और निष्कर्ष निकालें: प्रत्येक तर्कसंगत का विस्तार है
        अंततः आवधिक, प्रीपीरियड और अधिकतम अवधि के साथ
        $ q $.
  \item इसके विपरीत, मान लीजिए कि अंक $ y \in \intco{0}{1}$
        हैं \emph{विशुद्ध रूप से} आवधिक: $ d_{n+T} = d_n $ सभी के लिए $ n
        \geq 1$. शिफ्ट लेम्मा और विशिष्टता का उपयोग करना
        प्रश्न 10, दिखाएँ कि भिन्नात्मक भाग $ b^T y $
        के बराबर होती है $ y $, और निष्कर्ष निकालें $(b^T - 1)\,y \in \N $: इस प्रकार $ y $ है
        हर को विभाजित करने के साथ तर्कसंगत $ b^T - 1$. सत्यापित करें
        तंत्र चालू $0.(142857)$: $142857 \times 7 = 999999$.
  \item अंततः समय-समय पर मामले को शिफ्ट करके समझें और बताएं
        द \emph{आवधिकता मानदंड}: $ x \in \intco{0}{1}$ है
        तर्कसंगत यदि और केवल यदि यह उचित है $ b $-आदिक विस्तार है
        अंततः आवधिक --- एक आधार में यदि और केवल यदि में
        सब.
  \item के लिए $ x = \frac 1q $ साथ $\gcd(q, b) = 1$, दिखाओ कि
        विस्तार पूर्णतया आवधिक होता है तथा इसकी न्यूनतम अवधि होती है
        सबसे कम $ T \geq 1$ साथ $ b^T \equiv 1 \pmod q $ (द
        का गुणनात्मक क्रम $ b $ सापेक्ष $ q $). इसके लिए जाँच करें
        $ q = 7$, $ b = 10$ की शक्तियां $10$ सापेक्ष $7$ के माध्यम से चलना $3, 2, 6, 4, 5, 1$: आदेश $6$, मिलान प्रश्न
        1.
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Dividends and the diagonal.}
\begin{enumerate}[resume]
  \item होने देना $ x^*$ का असली हो $\intco{0}{1}$ जिसका आधार-$10$
        अंक हैं $1$ त्रिकोणीय स्थिति में $\frac{j(j +
        1)}{2}$ ($ j \geq 1$) और $0$ अन्यत्र: $ x^* =
        0.101001000100001\dots $ दिखाएँ कि इसका अंक स्ट्रिंग है
        उचित लेकिन अंततः आवधिक नहीं \emph{(एक अवधि तक $ T $
        अधिक से अधिक \omterm{prop:b1:reals:intervals}{अंतराल} पर लोगों को बाध्य करेगा $ T $, लेकिन दूरियां बढ़ती हैं)},
        और निष्कर्ष निकालो $ x^*$ तर्कहीन है: एक संख्या साबित हुई
        शुद्ध लय से अतार्किक.
  \item इसे प्रत्येक आधार के लिए दिखाएँ $ b $  \omterm{def:b1:logic:sets}{तय करना} $\{m/b^n : m \in \Z,
        n \in \N\}$ में सघन है $\R $ (सामान्यीकरण
        \cref{exo:b1:reals:8}), और वह हर तर्कसंगत $\frac pq
        \in \intoo{0}{1}$ एक \emph{समाप्त} में विस्तार
        आधार $ q $. नैतिक: समाप्ति जोड़ी का गुण है
        (संख्या, आधार); आवधिकता---तर्कसंगतता---है
        आंतरिक.
  \item (कैंटर का विकर्ण) चलो $ k \mapsto x_k $ कोई भी हो \omterm{def:b1:logic:map}{नक्शा} से
        $\N^*$ को $\intco{0}{1}$. अंक स्ट्रिंग को परिभाषित करें $ e_k =
        1$ यदि $ k $- का अंक $ x_k $ से मतभेद होना $1$, और
        $ e_k = 2$ अन्यथा। बताते हैं कि $(e_k)$ उचित है, वह
        इसका मूल्य $ y $ में निहित है $\intco{0}{1}$, ओर वो $ y \neq
        x_k $ हरएक के लिए $ k $. निष्कर्ष: नहीं \omterm{def:b1:logic:map}{नक्शा} $\N^* \to
        \intco{0}{1}$ है \omterm{def:b1:logic:inj}{विशेषण}. (की शब्दावली
        गणनीयता, और इस प्रमेय का उचित घर है
        \cref{ch:b1:topology}.)
  \item यदि उचित विस्तार हो तो दिखाएँ $ x $ और $ y $ सहमत
        सूचकांक तक $ n $ तब $\abs{x - y} < b^{-n}$, और अस्वीकृत करें
        के साथ बातचीत $ x = 0.1$, $ y = 0.0999$ आधार में $10$:
        संख्याओं की निकटता अंकों की सहमति को बाध्य नहीं करती है।
        किन वास्तविक लोगों को दोष देना है?
  \item भाग IV चालू करें $ x = \frac{1}{10}$ आधार में $ b = 2$:
        जब तक वे चक्रित न हो जाएं तब तक शेषफलों और अंकों की गणना करें, और
        निष्कर्ष $\frac{1}{10} = (0.0\overline{0011})_2$, के साथ
        पूर्वकाल $1$ और अवधि $4$. प्रश्न 3 के माध्यम से स्पष्ट करें कि क्यों
        कोई भी परिमित बाइनरी स्ट्रिंग कभी भी बराबर नहीं होगी $\frac{1}{10}$
        --- कंप्यूटर के फ़्लोटिंग-पॉइंट का कारण $0.1 + 0.2$ है
        बिलकुल नहीं $0.3$.
  \item संश्लेषण। एक-एक वाक्य में: कहाँ किया प्रमाण का प्रयोग
        (i) पूर्णता, (ii) द \omterm{thm:b1:reals:archimedes}{आर्किमिडीज़ संपत्ति}, (iii)
        की विशिष्टता खंड \omterm{thm:b1:reals:floor}{ज़मीन}, (iv) कबूतरखाना
        सिद्धांत? और नैतिक: $\intco{0}{1}$ निष्ठापूर्वक है
        उचित अंक स्ट्रिंग द्वारा कोडित, तर्कसंगतता के रूप में पढ़ा जाता है
        आवधिकता --- फिर भी विश्लेषण को प्राथमिकता देता है \omterm{def:b1:reals:bounds}{सर्वोच्च} को
        अंक. क्यों? (दो अंकों की स्ट्रिंग जोड़ने के बारे में सोचें।)
\end{enumerate}
\end{problem}
