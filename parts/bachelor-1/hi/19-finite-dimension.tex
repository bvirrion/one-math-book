\chapter{परिमित आयाम}\label{ch:b1:findim}

परिमित रूप से अनेक सदिशों द्वारा फैला हुआ स्थान एक अच्छी तरह से परिभाषित होता है
\emph{आयाम} --- इसके सभी \omterm{def:b1:vspaces:free}{आधार}ों का सामान्य आकार। सबूत
एक कॉम्बिनेटरियल इंजन से सभी प्रवाह के नीचे, एक्सचेंज लेम्मा:
ए \omterm{def:b1:vspaces:free}{मुक्त} परिवार किसी उत्पादक से अधिक संख्या कभी नहीं हो सकती। आयाम के साथ
इसके बाद हर जगह उपयोग किए जाने वाले उपकरण आते हैं: अधूरा \omterm{def:b1:vspaces:free}{आधार}
प्रमेय, \omterm{def:b1:findim:rank}{रैंक} एक परिवार का, ग्रासमैन का सूत्र।

\section{आधारों का अस्तित्व}

\begin{definition}\label{def:b1:findim:def}
$E $ है \emph{परिमित-आयामी}\index{परिमित-आयामी} जब यह है
एक सीमित उत्पादक परिवार। (अन्यथा \emph{अनंत-आयामी}:
ऐसा ही है $ K[X]$, जिनके सीमित परिवार ही हैं \omterm{def:b1:vspaces:span}{अवधि} बहुआयामी पद परिबद्ध का
डिग्री.)
\end{definition}

\begin{theorem}[एक्सचेंज लेम्मा]\label{thm:b1:findim:exchange}
होने देना $(g_1, \dots, g_n)$ उत्पन्न $ E $ और $(f_1, \dots, f_p)$ होना \omterm{def:b1:vspaces:free}{मुक्त}
में $ E $. तब $ p \leq n $.
\end{theorem}

\begin{proof}
हम प्रेरण द्वारा सिद्ध करते हैं $ k \leq p $: \emph{पुनः क्रमांकित करने के बाद
$ g_j $, परिवार $(f_1, \dots, f_k, g_{k+1}, \dots, g_n)$ उत्पन्न करता है
$ E $} ---जो मजबूर करता है $ k \leq n $ प्रत्येक चरण में, और $ p \leq n $ अंत में।

$ k = 0$: परिकल्पना। चरण: इसके लिए मान लें $ k - 1$; तब $ f_k $ है
का एक संयोजन $(f_1, \dots, f_{k-1},$ $ g_k, \dots, g_n)$. में
यह संयोजन कुछ $ g_j $ ($ j \geq k $) का शून्येतर गुणांक है
---
अन्यथा $ f_k $ का एक संयोजन होगा $ f_1, \dots, f_{k-1}$,
स्वतंत्रता का खंडन. पुनः क्रमांकित करें ताकि $ j = k $, और के लिए हल करें
$ g_k $: $ g_k $ का एक संयोजन है $(f_1, \dots, f_k, g_{k+1}, \dots,
g_n)$. का प्रत्येक सदिश $ E $, के माध्यम से व्यक्त किया गया $(k-1)$-परिवार,
फिर के माध्यम से पुनः व्यक्त किया जा सकता है $ k $-परिवार: यह उत्पन्न करता है।
(अगर $ k - 1 = n $, नहीं $ g $ रहता है और $ f_k $ का एक संयोजन होगा
द $ f_i $अकेला है: असंभव; इसलिए $ k \leq n $.)
\end{proof}

\begin{example}[विनिमय, एक बार देखा]
\label{ex:b1:findim:exchangerun}
में $\R^2$, उत्पादक परिवार को लीजिए $\bigl(g_1, g_2\bigr) =
\bigl((1,0), (0,1)\bigr)$ और यह \omterm{def:b1:vspaces:free}{मुक्त} परिवार $\bigl(f_1,
f_2\bigr) = \bigl((1,2), (3,4)\bigr)$. कदम $1$: $ f_1 = 1\cdot
g_1 + 2\cdot g_2$; का गुणांक $ g_2$ शून्येतर है, इसलिए
आदान-प्रदान $ g_2$ के लिए $ f_1$: परिवार $\bigl(f_1, g_1\bigr)$
अभी भी उत्पन्न करता है ($ g_2 = \frac12(f_1 - g_1)$). कदम $2$: $ f_2 =
(3,4) = 2\,f_1 + 1\cdot g_1$; शेष का गुणांक
$ g_1$ शून्येतर है (यह होना चाहिए: $ f_2$ का गुणज नहीं है $ f_1$),
तो फिर से विनिमय करें: $\bigl(f_1, f_2\bigr)$ उत्पन्न करता है $\R^2$. था
वहाँ एक तिहाई हो गया \omterm{def:b1:vspaces:free}{मुक्त} वेक्टर $ f_3$, नहीं $ g $ तक रहेगा
इसे अवशोषित करें --- जो कि लेम्मा बिल्कुल इसी तरह से मना करता है $3$ \omterm{def:b1:vspaces:free}{मुक्त}
वेक्टर में $\R^2$. उपरोक्त प्रमाण में किया गया यह बहीखाता है
सामान्य.
\end{example}

\begin{theorem}[परिमित आयाम में आधार]\label{thm:b1:findim:bases}
होने देना $ E \neq \{0\}$ होना \omterm{def:b1:findim:def}{परिमित-आयामी}.
\begin{enumerate}
  \item प्रत्येक परिमित उत्पादक परिवार से कोई एक निकाल सकता है \omterm{def:b1:vspaces:free}{आधार}.
  \item (अपूर्ण \omterm{def:b1:vspaces:free}{आधार} प्रमेय\index{अपूर्ण आधार प्रमेय})
        प्रत्येक \omterm{def:b1:vspaces:free}{मुक्त} परिवार ए तक फैला हुआ है \omterm{def:b1:vspaces:free}{आधार}, किसी के वैक्टर का उपयोग करना
        चुना हुआ उत्पादक परिवार।
  \item के सभी \omterm{def:b1:vspaces:free}{आधार} $ E $ समान संख्या के साथ परिमित हैं
        तत्व: \emph{आयाम} $\dim E $. (सम्मेलन:
        $\dim\{0\} = 0$.)
\end{enumerate}
\end{theorem}

\begin{proof}
(1) Discard, one at a time, any vector that is a combination of
अन्य; the family stays generating, since in any expression
using the discarded vector one may substitute its combination of
बचे हुए लोग. प्रक्रिया समाप्त हो जाती है --- प्रत्येक चरण सिकुड़ता है
finite family by one --- and it halts exactly when no remaining
वेक्टर दूसरों का एक संयोजन है। अंतिम परिवार अभी भी है
उत्पन्न करना, और यह है \omterm{def:b1:vspaces:free}{मुक्त}: एक गैर-तुच्छ शून्य संयोजन होगा
कुछ गैर-शून्य गुणांक रखें, और इसके द्वारा विभाजित करने से हल हो जाएगा
दूसरों के संदर्भ में संबंधित वेक्टर के लिए, इसे बनाते हुए
आख़िरकार ख़ारिज करने योग्य --- इस प्रक्रिया के विपरीत
रुका हुआ.

(2) चलो $(f_1, \dots, f_p)$ होना \omterm{def:b1:vspaces:free}{मुक्त}, $(g_1, \dots, g_n)$ उत्पन्न करना।
के माध्यम से चलना $ g_1, \dots, g_n $, जोड़ना $ g_j $ वर्तमान परिवार के लिए
जब भी यह पहले से ही इसमें नहीं है \omterm{def:b1:vspaces:span}{अवधि}
(\cref{prop:b1:vspaces:freecriteria}~(2) परिवार रखता है \omterm{def:b1:vspaces:free}{मुक्त}).
अंतिम परिवार है \omterm{def:b1:vspaces:free}{मुक्त}, और उत्पन्न करना: प्रत्येक $ g_j $ इसमें निहित है
\omterm{def:b1:vspaces:span}{अवधि} --- या तो इसे जोड़ा गया था, या यह पहले से ही एक संयोजन था।

(3) प्रत्येक के दो-दो \omterm{def:b1:vspaces:free}{आधार} हैं \omterm{def:b1:vspaces:free}{मुक्त} और प्रत्येक उत्पन्न करने वाला: एक्सचेंज लेम्मा
दोनों को उनकी प्रमुखताओं के बीच असमानताएँ देता है। (परिमितता:
ए \omterm{def:b1:vspaces:free}{आधार} है \omterm{def:b1:vspaces:free}{मुक्त}, इसलिए विनिमय द्वारा परिमित से बड़ा नहीं
परिवार उत्पन्न करना।)
\end{proof}

\begin{example}\label{ex:b1:findim:dims}
$\dim K^n = n $ (विहित \omterm{def:b1:vspaces:free}{आधार}); $\dim K_n[X] = n + 1$ (एकपदी);
$\dim_\R \C = 2$; का समाधान स्थान $ y'' + ay' + by = 0$ है
आयाम $2$ (\cref{thm:b1:diffeq:homogeneous2}: समाधान हैं
पैरामीट्रिज्ड वस्तुनिष्ठ रूप से और रैखिक रूप से $(\lambda, \mu) \in K^2$).
\end{example}

\begin{example}[एक ही सेट, दो आयाम]
\label{ex:b1:findim:twodimensions}
 \omterm{def:b1:logic:sets}{तय करना} $\C^2$ सम्मिश्र संख्याओं के युग्मों का एक है $\C $-वेक्टर स्थान
आयाम का $2$ (विहित \omterm{def:b1:vspaces:free}{आधार} $ e_1, e_2$) --- और एक
$\R $-आयाम का वेक्टर स्थान $4$, साथ \omterm{def:b1:vspaces:free}{आधार}
\[
(1, 0),\quad (\iu, 0),\quad (0, 1),\quad (0, \iu):
\]
प्रत्येक $(z, w) = (a + \iu b,\ c + \iu d)$ असली है \omterm{prop:b1:vspaces:coordinates}{COORDINATES}
$(a, b, c, d)$, विशिष्ट रूप से। आयाम का गुण नहीं है \omterm{def:b1:logic:sets}{तय करना}
अकेले सदिशों की: यह स्वतंत्रता की कोटि को गिनता है \emph{रिश्तेदार
अनुमत अदिशों के लिए}, और अदिश आपूर्ति को आधा करना $\C $
को $\R $ गिनती दोगुनी हो जाती है. (सप्ताहांत की समस्या इसका फायदा उठाती है
इस संवेदनशीलता का चरम मामला, जिसमें अदिश सभी सिकुड़ गए
का रास्ता $\Q $.)
\end{example}

\begin{example}[पूर्णता एल्गोरिथ्म चला रहा है]
\label{ex:b1:findim:completion}
को पूर्ण करो \omterm{def:b1:vspaces:free}{मुक्त} परिवार $\bigl((1,1,1)\bigr)$ एक में \omterm{def:b1:vspaces:free}{आधार} का
$\R^3$ कैनोनिकल वैक्टर का उपयोग करना। का प्रमाण चलाएँ
\cref{thm:b1:findim:bases}~(2) उत्पादक परिवार पर $(e_1,
e_2, e_3)$: है $ e_1 \in \operatorname{Vect}(1,1,1)$? नहीं (बहु
का $(1,1,1)$ बराबर है \omterm{prop:b1:vspaces:coordinates}{COORDINATES}) --- इसे जोड़ें। है $ e_2 \in
\operatorname{Vect}\bigl((1,1,1), e_1\bigr)$? एक संयोजन
$\alpha(1,1,1) + \beta e_1$ दूसरा और तीसरा बराबर है
\omterm{prop:b1:vspaces:coordinates}{COORDINATES}, और $ e_2$ नहीं --- इसे जोड़ें। परिवार
$\bigl((1,1,1), e_1, e_2\bigr)$ है \omterm{def:b1:vspaces:free}{मुक्त} साथ $3$ वेक्टर: रुकें,
यह एक है \omterm{def:b1:vspaces:free}{आधार} (\cref{prop:b1:findim:twoofthree} इसे बनायेंगे
रिफ्लेक्स अधिकारी)। ध्यान दें कि परिणाम किस क्रम पर निर्भर करता है
द $ g_j $ स्कैन किए गए हैं: पूर्णता एक एल्गोरिदम है, कोई सूत्र नहीं।
\end{example}

\begin{proposition}[तीन में से दो का नियम]\label{prop:b1:findim:twoofthree}
होने देना $\dim E = n $ और $\mathcal{F}$ वास्तव में एक परिवार $ n $ वेक्टर
का $ E $. तब
\[
\mathcal{F} \text{ free} \iff \mathcal{F} \text{ generating} \iff
\mathcal{F} \text{ basis}.
\]
इसके अलावा कोई भी \omterm{def:b1:vspaces:free}{मुक्त} परिवार है $\leq n $ वैक्टर, कोई भी उत्पादक परिवार
$\geq n $.
\end{proposition}

\begin{proof}
 \omterm{def:b1:counting:card}{प्रमुखता} सीमाएं ए के विरुद्ध विनिमय लेम्मा हैं \omterm{def:b1:vspaces:free}{आधार}. अगर
$\mathcal{F}$ (आकार $ n $) है \omterm{def:b1:vspaces:free}{मुक्त} लेकिन उत्पन्न नहीं हो रहा है, कुछ $ x $ झूठ
इसके बाहर \omterm{def:b1:vspaces:span}{अवधि}; जोड़कर $ x $ एक देता है \omterm{def:b1:vspaces:free}{मुक्त} परिवार का $ n + 1$
वेक्टर: असंभव. अगर $\mathcal{F}$ पैदा हो रहा है लेकिन नहीं \omterm{def:b1:vspaces:free}{मुक्त},
ए निकालना \omterm{def:b1:vspaces:free}{आधार} (\cref{thm:b1:findim:bases}~(1)) एक देता है \omterm{def:b1:vspaces:free}{आधार} का
$< n $ वेक्टर: असंभव.
\end{proof}

\begin{example}[तीन में से दो, आधा काम बचा रहा है]
\label{ex:b1:findim:twoofthreerun}
है $\bigl((1,1,0), (0,1,1), (1,0,1)\bigr)$ ए \omterm{def:b1:vspaces:free}{आधार} का $\R^3$?
गणना: तीन सदिश, आयाम तीन --- तो अकेले मुक्तता
निर्णय करता है. एक शून्य संयोजन देता है $ a + c = 0$, $ a + b = 0$, $ b +
c = 0$; तीनों को जोड़ने पर, $2(a + b + c) = 0$, और घटाना
प्रत्येक मूल समीकरण से $ a + b + c = 0$ पत्तियों $ b = c = a =
0$: \omterm{def:b1:vspaces:free}{मुक्त}, इसलिए ए \omterm{def:b1:vspaces:free}{आधार}, का आधा हिस्सा उत्पन्न करने के साथ
प्रमेय द्वारा सत्यापन निःशुल्क प्रदान किया गया। तुलना करना
\cref{ex:b1:vspaces:testbasis}, जहां वही दोहरा
सत्यापन हाथ से करना पड़ता था --- सिद्धांत का एक अध्याय
प्रत्येक पर बिल्कुल उसी बचत में परिवर्तित हो जाता है \omterm{def:b1:vspaces:free}{आधार} के लिए जाँच करें
बाकी किताब.
\end{example}

\begin{method}[एक आयाम की गणना]
\label{met:b1:findim:computedim}
आवृत्ति के घटते क्रम में तीन मानक मार्ग।
\begin{enumerate}
  \item \emph{पैरामीटराइज़ करें, फिर पढ़ें \omterm{def:b1:vspaces:free}{आधार}.} हल करें
        बाधाओं को परिभाषित करते हुए, सामान्य तत्व को व्यक्त करें
        जीवित मापदंडों में रैखिक रूप से, और जांचें कि
        मापदंडों को गुणा करने वाले वेक्टर हैं \omterm{def:b1:vspaces:free}{मुक्त}: द
        आयाम पैरामीटरों की संख्या है. (नीचे ए पर चलाएँ
        ठोस \omterm{def:b1:vspaces:subspace}{उपस्पेस} का $\R^4$.)
  \item \emph{प्रदर्शनी एक \omterm{def:b1:logic:inj}{द्विभाजित} रैखिक पैरामीट्रिजेशन.} कब
        तत्व परिमित रूप से अनेक मानों द्वारा निर्धारित होते हैं ---
        पुनरावृत्ति की प्रारंभिक स्थितियाँ
        (\cref{exo:b1:findim:10}), एक समाधान के गुणांक
        सूत्र (\cref{ex:b1:findim:dims}) --- आयाम है
        उन मानों की संख्या.
  \item \emph{सूत्रों का प्रयोग करें.} चौराहों के लिए ग्रासमैन और
        रकम, \omterm{def:b1:findim:rank}{रैंक} के लिए तक फैला, और बाद में रैंक--शून्यता के लिए
        गुठली और छवियाँ: आयाम आमतौर पर होते हैं
        \emph{गणना}, अनुमान नहीं लगाया गया।
\end{enumerate}
तीनों मार्गों में, तीन में से दो का नियम समापनकर्ता है:
एक बार जब गिनती मेल खाती है, तो स्वतंत्रता या पीढ़ी अकेले ही समाप्त हो जाती है।
\end{method}

\begin{example}[मार्ग 1, पूर्णतः]
\label{ex:b1:findim:parametrize}
का आयाम $ H = \{(x, y, z, t) \in \R^4 : x + y + z + t = 0
\text{ and } x = t\}$. हल करना: $ t = x $ और $ y + z = -2x $, इसलिए $ z =
-2x - y $ साथ $ x, y $ \omterm{def:b1:vspaces:free}{मुक्त}:
\[
(x,\ y,\ -2x - y,\ x) = x\,(1, 0, -2, 1) + y\,(0, 1, -1, 0) .
\]
दो वेक्टर हैं \omterm{def:b1:vspaces:free}{मुक्त} (पहले दो को देखें \omterm{prop:b1:vspaces:coordinates}{COORDINATES}:
$(x, y) = (0,0)$), इसलिए वे एक बनाते हैं \omterm{def:b1:vspaces:free}{आधार} का $ H $ और $\dim H = 2$.
गिनती पूर्वानुमानित थी --- दो स्वतंत्र रैखिक बाधाएँ
में $\R^4$ क्या प्रत्येक को एक आयाम खाना चाहिए --- लेकिन
पैरामीट्रिजेशन इसे सिद्ध करता है \emph{और} एक पर हाथ \omterm{def:b1:vspaces:free}{आधार}, जो
अकेले भविष्यवाणी कभी नहीं करती; \cref{exo:b1:findim:9} बनाता है
``प्रत्येक समीकरण अधिकतम एक आयाम खाता है'' नारा एक में बदल जाता है
प्रमेय.
\end{example}

\begin{example}[रूट 2, पूर्णतः]
\label{ex:b1:findim:route2}
का आयाम $ W = \{P \in \R_n[X] : P(1) = P(2) = 0\}$ (के लिए $ n
\geq 2$). कारक प्रमेय द्वारा दो बार लागू किया गया
(\cref{thm:b1:poly:factor}; जड़ें $1$ और $2$ विशिष्ट हैं),
$ P \in W $ बिल्कुल कब $ P = (X - 1)(X - 2)\,Q $ साथ $\deg Q \leq
n - 2$. पत्राचार $ Q \mapsto (X-1)(X-2)Q $ रैखिक है,
सभी तक पहुंचता है $ W $, और है \omterm{def:b1:logic:inj}{इंजेक्शन} (कोई उत्पाद तभी शून्य होता है जब
$ Q = 0$): $ W $ पैरामीट्रिज्ड है वस्तुनिष्ठ रूप से और रैखिक रूप से
$\R_{n-2}[X]$, इसलिए
\[
\dim W = \dim \R_{n-2}[X] = n - 1 .
\]
ए \omterm{def:b1:vspaces:free}{आधार} पैरामीट्रिजेशन के साथ आता है: की छवियां
एकपदी, $\bigl((X-1)(X-2),\ (X-1)(X-2)X,\ \dots,\
(X-1)(X-2)X^{n-2}\bigr)$. प्रत्येक नई मूल्यांकन बाधा ए पर
ताजा बिंदु की लागत बिल्कुल एक आयाम --- गिनती है
की रीढ़ \omterm{thm:b1:poly:lagrange}{लैग्रेंज इंटरपोलेशन}
(\cref{thm:b1:poly:lagrange}).
\end{example}

\begin{example}[एक अभिन्न बाधा की कीमत एक आयाम पर भी होती है]
\label{ex:b1:findim:integralconstraint}
का आयाम $ H = \{P \in \R_2[X] : \int_0^1 P = 0\}$. लिखना
$ P = a + bX + cX^2$, बाधा पढ़ता है $ a + \frac b2 + \frac
c3 = 0$; के लिए हल करें $ a $ और पैरामीट्रिज:
\[
P = b\Bigl(X - \frac12\Bigr) + c\Bigl(X^2 - \frac13\Bigr),
\]
इसलिए $ H = \operatorname{Vect}\bigl(X - \frac12,\ X^2 -
\frac13\bigr)$, आयाम $2$ (दो बहुआयामी पद विशिष्ट हैं
डिग्री: \omterm{def:b1:vspaces:free}{मुक्त}). एक रेखीय स्थिति --- चाहे एक मूल्यांकन,
एक \omterm{thm:b1:integration:def}{अभिन्न}, या कोई अन्य रैखिक नुस्खा --- अधिकतम एक को हटा देता है
आयाम, और बिल्कुल एक जैसे ही स्थिति नहीं है
समान रूप से शून्य. \cref{ch:b1:linmaps} ऐसी रेसिपीज के नाम बताएंगे
\emph{रैखिक रूप} और उनका समाधान सेट \emph{हाइपरप्लेन};
 \omterm{def:b1:vspaces:free}{आधार} यहां पाए गए वेक्टर पुनः प्रकट होते हैं
\cref{ch:b1:euclid} लीजेंड्रे परिवार की शुरुआत के रूप में।
\end{example}

\section{उपस्थान, रैंक, ग्रासमैन}

\begin{theorem}[उपस्थान]\label{thm:b1:findim:subspaces}
होने देना $ E $ होना \omterm{def:b1:findim:def}{परिमित-आयामी} और $ F $ ए \omterm{def:b1:vspaces:subspace}{उपस्पेस}. तब $ F $ है
\omterm{def:b1:findim:def}{परिमित-आयामी}, $\dim F \leq \dim E $, समानता के साथ यदि और केवल
यदि $ F = E $. इसके अलावा हर \omterm{def:b1:vspaces:subspace}{उपस्पेस} एक \omterm{def:b1:vspaces:sum}{पूरक} \omterm{def:b1:vspaces:subspace}{उपस्पेस}.
\end{theorem}

\begin{proof}
\omterm{def:b1:vspaces:free}{मुक्त} के परिवार $ F $ अधिक से अधिक है $\dim E $ वैक्टर (एक्सचेंज लेम्मा
में $ E $: ए \omterm{def:b1:vspaces:free}{मुक्त} परिवार का $ F $ विशेष रूप से है \omterm{def:b1:vspaces:free}{मुक्त} में $ E $, और $ E $
एक सीमित उत्पादक परिवार है)। बिच में \omterm{def:b1:vspaces:free}{मुक्त} के परिवार $ F $
इनमें से एक चुनें \emph{अधिक से अधिक} आकार $ p $ --- आकार के बाद से संभव है
से घिरे हुए पूर्णांक हैं $\dim E $. यह उत्पन्न करता है $ F $: अन्यथा कुछ
$ x \in F $ इसके बाहर पड़ा रहेगा \omterm{def:b1:vspaces:span}{अवधि}, और जोड़ना $ x $ देगा
ए \omterm{def:b1:vspaces:free}{मुक्त} परिवार का $ F $ आकार का $ p + 1$
(\cref{prop:b1:vspaces:freecriteria}~(2)), विरोधाभासी
अधिकतमता प्राणी \omterm{def:b1:vspaces:free}{मुक्त} और उत्पन्न करना, यह एक है \omterm{def:b1:vspaces:free}{आधार} का $ F $, और
$\dim F = p \leq \dim E $. अगर $ p = \dim E = n $: ए \omterm{def:b1:vspaces:free}{मुक्त} परिवार का $ n $ के वैक्टर
$ E $ एक है \omterm{def:b1:vspaces:free}{आधार} का $ E $ (\cref{prop:b1:findim:twoofthree}), इसलिए $ F
\supseteq \operatorname{span} = E $. \omterm{def:b1:vspaces:sum}{पूरक}: पूर्ण ए \omterm{def:b1:vspaces:free}{आधार}
$(f_1, \dots, f_p)$ का $ F $ एक में \omterm{def:b1:vspaces:free}{आधार} $(f_1, \dots, f_p, g_{p+1},
\dots, g_n)$ का $ E $ (अपूर्ण \omterm{def:b1:vspaces:free}{आधार} प्रमेय); तब $ G =
\operatorname{Vect}(g_{p+1}, \dots, g_n)$ संतुष्ट $ E = F \oplus G $
(विघटनों का अस्तित्व और विशिष्टता = \omterm{prop:b1:vspaces:coordinates}{COORDINATES} में
बड़ा \omterm{def:b1:vspaces:free}{आधार}).
\end{proof}

\begin{definition}[एक परिवार का दर्जा]\label{def:b1:findim:rank}
 \emph{रैंक}\index{रैंक} के एक सीमित परिवार का
वैक्टर इसका आयाम है \omterm{def:b1:vspaces:span}{अवधि}:
$\operatorname{rk}(x_1, \dots, x_p) = \dim
\operatorname{Vect}(x_1, \dots, x_p) \leq \min(p, \dim E)$,
समानता के साथ $ p $ अगर परिवार है \omterm{def:b1:vspaces:free}{मुक्त}.
\end{definition}

\begin{example}[उन्मूलन द्वारा रैंक की गणना करना]
\label{ex:b1:findim:rankcomputation}
पद का $\bigl((1,2,3), (2,3,4), (3,4,5), (1,1,1)\bigr)$ में
$\R^3$. \omterm{def:b1:vspaces:span}{अवधि} जब कोई सदिश a से घटाता है तो अपरिवर्तित रहता है
दूसरों का संयोजन (दोनों परिवार)। \omterm{def:b1:vspaces:span}{अवधि} वही
संयोजन): बदलें $(2,3,4)$ द्वारा $(2,3,4) - (1,2,3) = (1,1,1)$
और $(3,4,5)$ द्वारा $(3,4,5) - (1,2,3) = (2,2,2)$. \omterm{def:b1:vspaces:span}{अवधि} अभी है
$\operatorname{Vect}\bigl((1,2,3), (1,1,1), (2,2,2),
(1,1,1)\bigr) = \operatorname{Vect}\bigl((1,2,3), (1,1,1)\bigr)$,
और ये दोनों सदिश आनुपातिक नहीं हैं: \omterm{def:b1:findim:rank}{रैंक} है $2$. यह
घटाना-और-छोड़ना प्रक्रिया को गाऊसी के रूप में व्यवस्थित किया गया है
में उन्मूलन \cref{ch:b1:det}.
\end{example}

\begin{example}[संयोजन द्वारा संक्षेपण]
\label{ex:b1:findim:sumconcat}
ले लेना $\R^3$,
\[
F = \operatorname{Vect}\bigl((1,2,3),\ (1,1,1)\bigr),
\qquad
G = \operatorname{Vect}\bigl((2,3,4)\bigr) .
\]
योग $ F + G $ सभी के सम्मिलित परिवार द्वारा फैलाया गया है
तीन जनरेटर, और
\[
(2, 3, 4) = (1, 2, 3) + (1, 1, 1)
\]
दिखाता है कि तीसरा अनावश्यक है: $ F + G = F $, आयाम का $2$ ---
समकक्ष $ G \subseteq F $, जो संबंध प्रदर्शित करता है।
ग्रासमैन पुष्टि करते हैं: $\dim(F \cap G) = 2 + 1 - 2 = 1 = \dim G $.
योगों की गणना जनरेटरों को जोड़कर और फिर घटाकर की जाती है
द्वारा ढेर \omterm{def:b1:findim:rank}{रैंक} एल्गोरिथ्म; किसी नई तकनीक की कभी जरूरत नहीं पड़ती.
\end{example}

\begin{theorem}[ग्रासमैन का सूत्र]\label{thm:b1:findim:grassmann}
के लिए \omterm{def:b1:findim:def}{परिमित-आयामी} उपस्थान $ F, G $ का
$ E $:\index{ग्रासमैन का सूत्र}
\[
\dim(F + G) = \dim F + \dim G - \dim(F \cap G) .
\]
विशेष रूप से $ F + G $ \omterm{def:b1:vspaces:sum}{प्रत्यक्ष} iff है $\dim(F + G) = \dim F + \dim G $.
\end{theorem}

\begin{proof}
ए से शुरू करें \omterm{def:b1:vspaces:free}{आधार} $(e_1, \dots, e_r)$ का $ F \cap G $; इसे पूरा करो
एक में \omterm{def:b1:vspaces:free}{आधार} $(e_1, \dots, e_r, f_1, \dots, f_s)$ का $ F $ और एक में
\omterm{def:b1:vspaces:free}{आधार} $(e_1, \dots, e_r, g_1, \dots, g_t)$ का $ G $
(\cref{thm:b1:findim:bases}~(2)). हम दावा करते हैं
\[
\mathcal{B} = (e_1, \dots, e_r, f_1, \dots, f_s, g_1, \dots, g_t)
\]
एक है \omterm{def:b1:vspaces:free}{आधार} का $ F + G $; गणना द्वारा सूत्र इस प्रकार है: $(r + s) +
(r + t) - r = r + s + t $.

$\mathcal{B}$ उत्पन्न करता है $ F + G $: कोई $ u + v $ ($ u \in F $, $ v \in G $)
इसके माध्यम से विस्तार होता है। उन्मुक्तता: मान लीजिए $\sum \alpha_i e_i + \sum
\beta_j f_j + \sum \gamma_k g_k = 0$. सदिश $ w = \sum \gamma_k
g_k = -\sum\alpha_i e_i - \sum\beta_j f_j $ में निहित है $ G \cap F $, तो यह
का विस्तार होता है $(e_i)$ अकेला; लेकिन $ w $ का भी विस्तार होता है $(g_k)$, और इसमें \omterm{def:b1:vspaces:free}{आधार} का $ G $ इन दो अभिव्यक्तियों का मेल होना चाहिए: सभी
$\gamma_k = 0$ (और यह $ e $-निर्देशांक मिलान)। रिश्ता कम हो जाता है
को $\sum\alpha_i e_i + \sum\beta_j f_j = 0$, में एक रिश्ता \omterm{def:b1:vspaces:free}{आधार}
का $ F $: शेष सभी गुणांक गायब हो जाते हैं।
\end{proof}

\begin{example}\label{ex:b1:findim:planes}
दो अलग-अलग विमान $ F, G $ (आयाम $2$) का $\R^3$ संतुष्ट $ F + G
= \R^3$ (उनके योग में निश्चित रूप से एक विमान शामिल है), इसलिए $\dim(F \cap G) =
2 + 2 - 3 = 1$: वे हमेशा एक रेखा के अनुदिश प्रतिच्छेद करते हैं --- नहीं
मूल के माध्यम से ``समानांतर तल''।
\end{example}

\begin{example}[कार्रवाई में ग्रासमैन, में $\R^4$]
\label{ex:b1:findim:grassmannrun}
होने देना $ F = \operatorname{Vect}\bigl(e_1,\ e_2,\ (1,1,1,0)\bigr)$
और $ G = \operatorname{Vect}(e_3, e_4)$ में $\R^4$. आयाम:
$\dim F = 3$ (तीसरे जनरेटर में एक गैर-शून्य तीसरा समन्वय है,
बाहर $\operatorname{Vect}(e_1, e_2)$) और $\dim G = 2$. जोड़:
$ F + G $ रोकना $ e_1, e_2, e_4$ और $ e_3 = (1,1,1,0) - e_1 -
e_2$: यह सब है $\R^4$. फिर ग्रासमैन \emph{गणना करता है} द
चौराहे का आकार बिना किसी उन्मूलन के:
\[
\dim(F \cap G) = 3 + 2 - 4 = 1 .
\]
रेखा को पहचानने के लिए अंदर देखें $ G $: एक वेक्टर $(0, 0, c, d)$
में निहित है $ F $ बिल्कुल जब यह है $ a e_1 + b e_2 + \lambda(1,1,1,0)$,
जबरदस्ती $\lambda = -a = -b $ और $ d = 0$: चौराहा है
$\operatorname{Vect}(e_3)$ --- सुसंगत, तब से $ e_3$ था
में प्रदर्शित किया गया $ F $ ऊपर और अंदर निहित है $ G $ परिभाषा से। विशिष्ट
श्रम का विभाजन: ग्रासमैन भविष्यवाणी करता है \emph{कितना} देखना
इसके लिए, रैखिक प्रणाली तब मिलती है \emph{क्या}.
\end{example}

\begin{example}[समीकरणों द्वारा प्रतिच्छेद करना]
\label{ex:b1:findim:intersectequations}
कब दोनों उपस्थान समाधान के रूप में आओ सेट, प्रतिच्छेद करना बस है
समीकरणों को व्यवस्थित करना। में $\R^3$: $ F = \{x + y + z = 0\}$ और
$ G = \{x = y\}$ देना
\[
F \cap G = \{x = y,\ 2x + z = 0\}
= \{(x,\ x,\ -2x)\} = \operatorname{Vect}(1, 1, -2),
\]
एक पंक्ति. ग्रासमैन द्वारा क्रॉस-चेक: $ F + G = \R^3$ (विमान हैं
अलग, इसलिए उनके योग में सख्ती से एक विमान शामिल है), इसलिए
$\dim(F\cap G) = 2 + 2 - 3 = 1$. ए के दो विवरण
\omterm{def:b1:vspaces:subspace}{उपस्पेस} --- समीकरणों द्वारा, जनरेटरों द्वारा --- प्रत्येक एक बनाता है
ऑपरेशन तुच्छ: समीकरण स्टैकिंग, जनरेटर द्वारा प्रतिच्छेद करते हैं
जोड़कर योग; उनके बीच रूपांतरण बिल्कुल वही है
एक रैखिक प्रणाली को हल करने का अर्थ है (\cref{ch:b1:det}).
\end{example}

\begin{remark}[सामान्य ख़तरे]
\label{rem:b1:findim:pitfalls}
\emph{आयाम योग के साथ नहीं जुड़ते} जब तक कि योग \omterm{def:b1:vspaces:sum}{प्रत्यक्ष} न हो:
के दो विमान $\R^3$ पास होना $\dim(F + G) = 3$, नहीं $4$; हमेशा
प्रतिच्छेदन पद (ग्रासमैन) द्वारा सही।
\emph{समावेशन के लिए आयाम की आवश्यकता होती है \emph{and} समावेशन}:
$\dim F = \dim G $ अकेला कभी नहीं देता $ F = G $ (दो अलग-अलग पंक्तियाँ
का $\R^2$); की समानता का मामला
\cref{thm:b1:findim:subspaces} आवश्यक है $ F \subseteq G $ पहला।
\emph{मापदंडों की गिनती अभी तक प्रमाण नहीं है}: ``दो समीकरण
$\R^4$, तो आयाम $2$'' विफल रहता है जब समीकरण होते हैं
आश्रित ($ x + y = 0$ और $2x + 2y = 0$ आयाम छोड़ें $3$);
केवल एक वास्तविक पैरामीट्रिजेशन या ए \omterm{def:b1:findim:rank}{रैंक} गणना निर्णय लेती है।
\emph{ए के आयाम की बात मत करो गैर-उपस्थान}: समाधान
सेट का \emph{अमानवीय} सिस्टम चूक गए $0$; उनका ``आयाम''
संबद्ध सजातीय समाधान स्थान का है
(\cref{ch:b1:det} इसे सटीक बनाता है)।
\emph{अनंत आयाम मौजूद है}: $ K[X]$ रोकना \omterm{def:b1:vspaces:free}{मुक्त} परिवार
हर आकार (एकपदी) का, इसलिए कोई सीमित उत्पादक परिवार नहीं है
मौजूद हो सकता है --- \omterm{def:b1:logic:statement}{कथन} जैसे दो-में-तीन नियम हैं
सख्ती से \omterm{def:b1:findim:def}{परिमित-आयामी} और बुरी तरह असफल हो गए $ K[X]$
(\cref{cor:b1:linmaps:samedim} के लिए वही दिखाएगा
इंजेक्टिविटी/सजेक्टिविटी)।
\emph{पद के बारे में है \omterm{def:b1:vspaces:span}{अवधि}, सूची नहीं}: एक वेक्टर को दोहराते हुए,
गैर-शून्य स्थिरांकों द्वारा पुन: व्यवस्थित करना, या पुनः स्केल करना छोड़ देता है \omterm{def:b1:findim:rank}{रैंक}
अपरिवर्तित, और $\operatorname{rk} = p $ (वेक्टर की संख्या)
एक है \emph{साबित करने के लिए संपत्ति} ---यह बिल्कुल उन्मुक्तता है. ए
का परिवार $5$ के वैक्टर \omterm{def:b1:findim:rank}{रैंक} $2$ तीन सदिशों का मान रखता है
अतिरेक का, जो उन्मूलन
(\cref{ex:b1:findim:rankcomputation}) स्पष्ट रूप से पता लगाता है।
\end{remark}

\begin{remark}[जहां आयाम काम करने जाता है]
\label{rem:b1:findim:whereused}
यहां से आयाम पुस्तक का पसंदीदा गणना तर्क है।
\cref{ch:b1:linmaps} रैंक-शून्यता प्रमेय को सिद्ध करता है
ग्रासमैन के सूत्र का कार्यात्मक संस्करण; \cref{ch:b1:matrices}
गणना करता है \omterm{def:b1:findim:rank}{रैंक} पंक्ति में कमी से; \cref{ch:b1:det} `` बदल जाता है $ n$
के वैक्टर $ K^n $ फॉर्म ए \omterm{def:b1:vspaces:free}{आधार}'' एक संख्या में शून्येतर होना। द
नीचे दी गई सप्ताहांत समस्या आयाम कार्य को दर्शाती है \emph{अंकगणित}:
के ऊपर आयामों की गिनती \omterm{def:b1:structures:field}{मैदान} $\Q $ अतार्किकता सिद्ध करता है
\omterm{def:b1:logic:statement}{कथन} जो हाथ से अछूते लगते हैं। वर्ष~3 खंड में
समान आयाम मायने रखता है, द्वारा परिष्कृत \omterm{def:b1:structures:group}{समूह} सिद्धांत, तय करें कौन सा
शास्त्रीय निर्माण समस्याएं हल करने योग्य हैं --- वह कहानी है
गैलोइस सिद्धांत.
\end{remark}

\begin{remark}[पुस्तक 3 के अंदर परिप्रेक्ष्य]
\label{rem:b1:findim:perspectives}
आयाम इस आयतन के शेष भाग की संरक्षित मात्रा है,
और संरक्षण कानूनों का नाम पहले से रखना उचित है।
\cref{ch:b1:linmaps} को सिद्ध करता $\dim E = \dim\ker u +
\operatorname{rk} u $: क्या रेखीय है \omterm{def:b1:logic:map}{नक्शा} कुचलता है प्लस यह क्या है
हमेशा स्रोत का योग रखता है। \cref{ch:b1:det} इसे परिष्कृत करता है
समाधान की संरचना में सेट: $ p $ अज्ञात शून्य
$\operatorname{rk} A $ धुरी का आयाम छोड़ देता है
समाधान स्थान, जिसे गाऊसी उन्मूलन के रूप में प्रदर्शित किया जाता है \omterm{def:b1:vspaces:free}{मुक्त}
पैरामीटर. \cref{ch:b1:euclid} फूट $\dim E = \dim F + \dim
F^\perp $ ऑर्थोगोनली, और सप्ताहांत की समस्या
\cref{ch:b1:multivar} बिल्कुल इतना बजट खर्च करता है: $ n $ डेटा
अंक, $2$ पैरामीटर फिट, $ n - 2$ अवशिष्ट के आयाम.
जब भी कोई गिनती बाद के अध्याय में संतुलन बनाने से इंकार कर देती है, तो
त्रुटि एक भूला हुआ कर्नेल या एक गै\omterm{def:b1:vspaces:sum}{र-प्रत्यक्ष} योग है --- वापस आएँ
ग्रासमैन का सूत्र प्रथम.
\end{remark}

\section{अभ्यास}

\begin{exercise}[$\star $]\label{exo:b1:findim:1}
दे दो \omterm{def:b1:vspaces:free}{आधार} और इसका आयाम:
\begin{enumerate}
  \item $ F = \{(x,y,z) \in \R^3 : x + y + z = 0\}$;
  \item $ G = \{(x,y,z,t) \in \R^4 : x = y,\ z = 2t\}$;
  \item $ H = \{P \in \R_3[X] : P(1) = P'(1) = 0\}$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:findim:2}
की गणना करें \omterm{def:b1:findim:rank}{रैंक} परिवार की
$\bigl((1,1,1), (1,2,3), (3,5,7), (0,1,2)\bigr)$ में $\R^3$, और
ए निकालें \omterm{def:b1:vspaces:free}{आधार} उसके जैसा \omterm{def:b1:vspaces:span}{अवधि}.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:findim:3}
को पूर्ण करो \omterm{def:b1:vspaces:free}{मुक्त} परिवार $\bigl((1,1,0,0), (0,0,1,1)\bigr)$ एक में
\omterm{def:b1:vspaces:free}{आधार} का $\R^4$ विहित सदिशों का उपयोग करना, और औचित्य सिद्ध करना।
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:findim:4}
साबित करें कि $\bigl(1, X, X(X-1), X(X-1)(X-2)\bigr)$ एक है \omterm{def:b1:vspaces:free}{आधार} का
$\R_3[X]$, और खोजें \omterm{prop:b1:vspaces:coordinates}{COORDINATES} का $ X^3$ इस में।
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:findim:5}
होने देना $ F $ और $ G $ होना उपस्थान आयामों का $4$ और $5$ एक स्थान का
$ E $ साथ $\dim E = 7$. के संभावित मान क्या हैं $\dim(F \cap
G)$? प्रत्येक मान को साकार करने वाला एक उदाहरण दीजिए $ E = \R^7$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:findim:6}
होने देना $ H = \{P \in \R_n[X] : P(1) = 0\}$. साबित करें कि $ H $ एक है
का हाइपरप्लेन $\R_n[X]$ (ए \omterm{def:b1:vspaces:subspace}{उपस्पेस} आयाम का $ n $), प्रदर्शनी एक
\omterm{def:b1:vspaces:free}{आधार} का $ H $ \emph{(कारक प्रमेय के बारे में सोचें: $ P = (X-1)Q $)}, और
एक दे दो \omterm{def:b1:vspaces:sum}{पूरक} रेखा।
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:findim:7}
होने देना $ u_1, \dots, u_p $ के सदिश बनें \omterm{def:b1:findim:rank}{रैंक} $ r $. उसे हटाकर सिद्ध करो
एक वेक्टर से एक परिवार उत्पन्न होता है \omterm{def:b1:findim:rank}{रैंक} $ r $ या $ r - 1$, और वह
एक वेक्टर जोड़ने से परिणाम प्राप्त होते हैं \omterm{def:b1:findim:rank}{रैंक} $ r $ या $ r + 1$. उसका निष्कर्ष निकालो \omterm{def:b1:findim:rank}{रैंक}
अधिक से अधिक \omterm{def:b1:counting:objects}{परिवर्तन} $1$ किसी एकल सम्मिलन या विलोपन के अंतर्गत।
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b1:findim:8}
होने देना $ F_1 \subseteq F_2 \subseteq \dots \subseteq F_k $ होना उपस्थान
का $ E $ ($\dim E = n $) साथ $ F_i \neq F_{i+1}$ सभी के लिए $ i $. सिद्ध करना $ k
\leq n + 1$. यह निष्कर्ष निकालें कि एक सख्ती से बढ़ती हुई श्रृंखला है उपस्थान
का $\R^n $ अधिकतम लंबाई होती है $ n + 1$, और अधिकतम में से एक को प्रदर्शित करें
लंबाई.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b1:findim:9}
होने देना $ E $ आयाम का हो $ n $ और $ F $, $ G $ दो हाइपरप्लेन (आयाम
$ n - 1$), $ F \neq G $. गणना करना $\dim(F \cap G)$. सामान्यीकरण: द
का चौराहा $ k $ हाइपरप्लेन का आयाम होता है $\geq n - k $.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:findim:10}
होने देना $ E $ हो \omterm{def:b1:logic:sets}{तय करना} वास्तविक अनुक्रम संतोषजनक हैं $ u_{n+2} =
3u_{n+1} - 2u_n $ सभी के लिए $ n $.
\begin{enumerate}
  \item बताते हैं कि $ E $ एक है \omterm{def:b1:vspaces:subspace}{उपस्पेस} अनुक्रमों के स्थान का, और
        कि एक क्रम $ E $ पूरी तरह से निर्धारित है, रैखिक रूप से,
        जोड़ी द्वारा $(u_0, u_1)$; परिणाम निकालना $\dim E = 2$.
  \item जांचें कि निरंतर अनुक्रम $(1)$ और ज्यामितीय
        क्रम $(2^n)$ रिहायश $ E $ और फॉर्म ए \omterm{def:b1:vspaces:free}{आधार} का $ E $.
  \item का क्रम ज्ञात कीजिए $ E $ साथ $ u_0 = 0$, $ u_1 = 1$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:findim:11}
होने देना $ E $ आयाम का हो $ n $.
\begin{enumerate}
  \item अगर $ F, G $ हैं उपस्थान साथ $\dim F + \dim G > n $, सिद्ध करना
        $ F \cap G \neq \{0\}$. चित्रण: दो उपस्थान का
        आयाम $51$ और $50$ का $\R^{100}$ हमेशा एक साझा करें
        शून्येतर सदिश.
  \item अगर $ H $ एक हाइपरप्लेन है और $ F $ ए \omterm{def:b1:vspaces:subspace}{उपस्पेस} साथ $ F \cap H =
        \{0\}$, सिद्ध करना $\dim F \leq 1$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b1:findim:12}
(सामान्य \omterm{def:b1:vspaces:sum}{पूरक}) होने देना $ F, G $ होना उपस्थान का $ E $ (परिमित
आयाम) साथ $\dim F = \dim G $. साबित करें कि $ F $ और $ G $ स्वीकार करें ए
\emph{सामान्य} \omterm{def:b1:vspaces:sum}{पूरक}: वहां एक है \omterm{def:b1:vspaces:subspace}{उपस्पेस} $ S $ साथ $ E = F
\oplus S = G \oplus S $. \emph{(नीचे की ओर प्रेरित करें $\dim F $: अगर $ F
\neq G \neq E $, चुनना $ x \notin F \cup G $ ---
\cref{exo:b1:vspaces:12} इसकी अनुमति देता है --- और विचार करें $ F \oplus
\operatorname{Vect}(x)$ और $ G \oplus \operatorname{Vect}(x)$.)}
\end{exercise}

\section{समस्या: डेडेकाइंड का टावर कानून}

\begin{problem}\label{pb:b1:findim:1}
अध्याय 18-19 में किसी भी अदिश राशि के बारे में कुछ भी इस्तेमाल नहीं किया गया है
द \omterm{def:b1:structures:field}{मैदान} स्वयंसिद्ध (\cref{def:b1:structures:field}): कोई भी हो सकता है
इसलिए ले लो $ K = \Q $ और मापें सेट का \emph{वास्तविक संख्या}
के मापदण्ड के साथ $\Q $-आयाम। यह समस्या गणना करती है
का आयाम $\Q(\sqrt2, \sqrt3)$, सिद्ध करता है \emph{डेडेकाइंड का \omterm{pb:b1:findim:1}{टावर
कानून}} $\dim_\Q M = \dim_\Q K \cdot \dim_K M $, और फसल
शुद्ध आयाम गणना द्वारा अतार्किकता प्रमेय --- नहीं
$\varepsilon $, कोई दशमलव नहीं, केवल \omterm{def:b1:vspaces:free}{आधार}।\index{टावर कानून}
\index{field!subfield of $\R $}\index{अपरिमेय संख्या}

\medskip
\noindent\textbf{Part I --- Rational scalars.}
\begin{enumerate}
  \item जाँच करें कि $\Q $ एक है \omterm{def:b1:structures:field}{मैदान} और वह हर परिभाषा और
        अध्याय 18-19 का प्रमाण केवल इसका उपयोग करता है \omterm{def:b1:structures:field}{मैदान} के स्वयंसिद्ध
        अदिश; समाप्त करें कि $\R $ एक है $\Q $-वेक्टर स्पेस और
        वह एक्सचेंज लेम्मा, \omterm{def:b1:vspaces:free}{आधार} प्रमेय और
        ग्रासमैन का फार्मूला कायम है $\Q $. एक कदम बताएं
        के प्रमाण का \cref{thm:b1:findim:exchange} कहाँ
        एक गैरशून्य अदिश द्वारा विभाजन किया जाता है।
  \item बताते हैं कि $(1, \sqrt2)$ है \omterm{def:b1:vspaces:free}{मुक्त} ऊपर $\Q $ लेकिन ऊपर से जुड़ा हुआ है
        $\R $. \omterm{def:b1:logic:sets}{तय करना} $\Q(\sqrt2) = \operatorname{Vect}_\Q(1,
        \sqrt2) = \{a + b\sqrt2 : a, b \in \Q\}$; क्या है
        $\dim_\Q \Q(\sqrt2)$?
  \item बताते हैं कि $\Q(\sqrt2)$ गुणन के अंतर्गत स्थिर है।
        के लिए $ u = a + b\sqrt2$ \omterm{def:b1:logic:sets}{तय करना} $\sigma(u) = a - b\sqrt2$ और
        $ N(u) = u\,\sigma(u) = a^2 - 2b^2$. दिखाओ $\sigma(uv) =
        \sigma(u)\sigma(v)$, निष्कर्ष निकालना $ N(uv) = N(u)N(v)$, और दिखाओ
        $ N(u) \neq 0$ जब कभी भी $ u \neq 0$.
  \item यह निष्कर्ष निकालें कि प्रत्येक अशून्य $ u \in \Q(\sqrt2)$ उसके पास है
        में उलटा $\Q(\sqrt2)$, अर्थात् $ u^{-1} =
        \sigma(u)/N(u)$: $\Q(\sqrt2)$ का एक उपक्षेत्र है $\R $.
        गणना करना $\dfrac1{3 + 2\sqrt2}$ और $\dfrac1{1 + \sqrt2}$.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Adjoining $\sqrt3$.}
\begin{enumerate}[resume]
  \item साबित करें कि $\sqrt6 \notin \Q $ \emph{(प्रतिपादक की तुलना करें
        का $2$ के दोनों तरफ $6q^2 = p^2$, के रूप में
        \cref{exo:b1:arith:7})}, फिर उस $\sqrt3 \notin
        \Q(\sqrt2)$ \emph{(वर्ग $\sqrt3 = a + b\sqrt2$ और
        मामलों पर चर्चा करें $ ab \neq 0$, $ b = 0$, $ a = 0$)}.
  \item होने देना $ M = \operatorname{Vect}_\Q(1, \sqrt2, \sqrt3,
        \sqrt6)$. बताते हैं कि $ M $ गुणन के अंतर्गत स्थिर है
        \emph{(के उत्पादों की एक तालिका \omterm{def:b1:vspaces:free}{आधार} वेक्टर
        पर्याप्त)}.
  \item लिखना $ K = \Q(\sqrt2)$. बताते हैं कि $(1, \sqrt3)$ है \omterm{def:b1:vspaces:free}{मुक्त}
        \emph{ऊपर \omterm{def:b1:structures:field}{मैदान} $ K $}, और उसका निष्कर्ष निकालो $ M = K +
        K\sqrt3$ एक है $ K $-आयाम का वेक्टर स्थान $2$ साथ
        \omterm{def:b1:vspaces:free}{आधार} $(1, \sqrt3)$.
  \item साबित करें कि $(1, \sqrt2, \sqrt3, \sqrt6)$ है \omterm{def:b1:vspaces:free}{मुक्त} ऊपर
        $\Q $, इस तरह $\dim_\Q M = 4$. \emph{(एक शून्य संबंध \omterm{def:b1:structures:group}{समूह}ित करें
        जैसे $(a + b\sqrt2) + (c + d\sqrt2)\sqrt3 = 0$ और आवेदन करें
        प्रश्न 7 फिर 2.)}
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The \omterm{pb:b1:findim:1}{टावर कानून}.} होने देना $\Q \subseteq K
\subseteq M \subseteq \R $ कहाँ $ K $ और $ M $ उपक्षेत्र हैं, $(e_1,
\dots, e_m)$ एक है \omterm{def:b1:vspaces:free}{आधार} का $ K $ के तौर पर $\Q $-वेक्टर स्पेस, और $(f_1,
\dots, f_n)$ ए \omterm{def:b1:vspaces:free}{आधार} का $ M $ के तौर पर $ K $-सदिश स्थल।
\begin{enumerate}[resume]
  \item दिखाओ कि $ mn $ उत्पादों $(e_i f_j)$ उत्पन्न $ M $ ऊपर
        $\Q $.
  \item दिखाओ कि परिवार $(e_i f_j)$ है \omterm{def:b1:vspaces:free}{मुक्त} ऊपर $\Q $.
        \emph{(शून्य को पुनर्गठित करें $\Q $-संयोजन के रूप में $\sum_j
        \bigl(\sum_i \lambda_{ij} e_i\bigr) f_j $, जिसका आंतरिक
        गुणांक में रहते हैं $ K $.)}
  \item के साथ समापन करें \emph{डेडेकाइंड का \omterm{pb:b1:findim:1}{टावर कानून}}:
        \[
        \dim_\Q M \;=\; \dim_\Q K \,\cdot\, \dim_K M ,
        \]
        और इसकी जांच करें $\Q \subseteq \Q(\sqrt2) \subseteq M $
        प्रश्न 7 एवं 8 के विरूद्ध।
  \item (खेत निःशुल्क) चलो $ A \subseteq \R $ एक हो
        \omterm{def:b1:findim:def}{परिमित-आयामी} $\Q $-उपस्थान युक्त $1$ और
        गुणन के अंतर्गत स्थिर, और चलो $ u \in A $, $ u \neq
        0$. दिखाओ अगर $(a_1, \dots, a_d)$ एक है \omterm{def:b1:vspaces:free}{आधार} का $ A $,
        तब $(u a_1, \dots, u a_d)$ फिर से एक है \omterm{def:b1:vspaces:free}{आधार} का $ A $;
        उसका निष्कर्ष निकालो $ u $ एक उलटा है \emph{में $ A $}: $ A $ एक है
        का उपक्षेत्र $\R $. कौन सा पूर्व प्रश्न यह करता है
        ठीक हो जाओ?
  \item इसे हर एक के लिए दिखाएँ $ x \in A $ (जैसा कि प्रश्न 12 में है, $\dim_\Q
        A = d $) परिवार $(1, x, x^2, \dots, x^{d})$ जुड़ा हुआ है:
        का प्रत्येक तत्व $ A $ एक अशून्य की जड़ है \omterm{def:b1:poly:def}{बहुपद}
        तर्कसंगत गुणांकों के साथ, अधिकतम डिग्री का $ d $.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- One number generates everything.}
\omterm{def:b1:logic:sets}{तय करना} $ s = \sqrt2 + \sqrt3$.
\begin{enumerate}[resume]
  \item की गणना करें \omterm{prop:b1:vspaces:coordinates}{COORDINATES} का $ s^2$, $ s^3$ और $ s^4$ में
        \omterm{def:b1:vspaces:free}{आधार} $(1, \sqrt2, \sqrt3, \sqrt6)$ का $ M $.
  \item बताते हैं कि $(1, s, s^2, s^3)$ है \omterm{def:b1:vspaces:free}{मुक्त} ऊपर $\Q $, और निष्कर्ष निकालें
        $\operatorname{Vect}_\Q(1, s, s^2, s^3) = M $: हर
        का तत्व $ M $ एक तर्कसंगत है \omterm{def:b1:poly:def}{बहुपद} में $ s $. अभिव्यक्त करना
        $\sqrt2$ और $\sqrt3$ इस प्रकार बहुआयामी पद.
  \item सत्यापित करें $ s^4 - 10s^2 + 1 = 0$, और दिखाओ $ X^4 - 10X^2
        + 1$ है \emph{\omterm{def:b1:poly:def}{राक्षसी} \omterm{def:b1:poly:def}{बहुपद} न्यूनतम डिग्री का}
        पर लुप्त हो रहा है $ s $. इसकी चार वास्तविक जड़ें ज्ञात कीजिए।
  \item निष्कर्ष: $1/s = 10s - s^3$; इस संख्या को पहचानें. बताते हैं कि $ a + b\sqrt2 + c\sqrt3 + d\sqrt6$ ($ a,b,c,d \in \Q $)
        तर्कसंगत है यदि और केवल यदि $ b = c = d = 0$; विशेष रूप से $\sqrt2 + \sqrt3 + \sqrt6$ तर्कहीन है
        (मजबूत करना \cref{exo:b1:reals:7}).
\end{enumerate}

\medskip
\noindent\textbf{Part V --- The cube root stays outside.} \omterm{def:b1:logic:sets}{तय करना} $ t
= 2^{1/3}$.
\begin{enumerate}[resume]
  \item बताते हैं कि $ X^3 - 2$ कोई तर्कसंगत जड़ नहीं है \emph{(द
        का तर्कसंगत-मूल मानदंड \cref{exo:b1:poly:5}, या ए
        मूल्यांकन गणना)}; उसका निष्कर्ष निकालो $(1, t)$ है \omterm{def:b1:vspaces:free}{मुक्त} ऊपर
        $\Q $.
  \item बताते हैं कि $(1, t, t^2)$ है \omterm{def:b1:vspaces:free}{मुक्त} ऊपर $\Q $. \emph{(यदि कुछ
        शून्येतर $ P \in \Q[X]$ डिग्री का $\leq 2$ मारता $ t $, ले लो
        कम से कम डिग्री और विभाजन में से एक $ X^3 - 2$ इसके द्वारा,
        \cref{thm:b1:poly:division}; समाप्त करें कि $ X^3 - 2$
        एक तर्कसंगत जड़ होगी।)} उसका निष्कर्ष निकालो $ A =
        \operatorname{Vect}_\Q(1, t, t^2)$ आयाम है $3$, है
        गुणन के अंतर्गत स्थिर, और का एक उपक्षेत्र है $\R $.
  \item साबित करें कि $ t \notin M $: अन्यथा $ M $ यह एक होगा सदिश स्थल ऊपर \omterm{def:b1:structures:field}{मैदान} $ A $, और यह \omterm{pb:b1:findim:1}{टावर कानून} जबरदस्ती करेगा
        $3 \mid 4$. इसलिए $2^{1/3}$ तर्कसंगत संयोजन नहीं है
        का $1, \sqrt2, \sqrt3, \sqrt6$.
  \item उसका निष्कर्ष निकालो $ t \notin \Q(\sqrt2)$, वह $ t + \sqrt2$ है
        तर्कहीन, और वह कोई तर्कसंगत नहीं $ a, b $ संतुष्ट
        $2^{1/3} = a + b\sqrt3$.
\end{enumerate}

\medskip
\noindent\textbf{Part VI --- All the subfields, and synthesis.}
\begin{enumerate}[resume]
  \item सिद्ध करो \omterm{def:b1:arith:divides}{भाजकत्व} डिग्री की: यदि $\Q \subseteq A
        \subseteq B \subseteq \R $ के साथ उपक्षेत्र हैं $\dim_\Q B $
        तो फिर परिमित $\dim_\Q A $ विभाजित $\dim_\Q B $. क्या हैं?
        के उपक्षेत्रों के संभावित आयाम $ M $?
  \item ठानना \emph{सभी} के उपक्षेत्र $ M $ आयाम का $2$
        ऊपर $\Q $. \emph{(खोजना कम करें $ w = a + b\sqrt2
        + c\sqrt3 + d\sqrt6 \in M \setminus \Q $ साथ $ w^2 \in
        \Q $: बढ़ाना $ w^2$ और तीन अतार्किक को नष्ट कर दो
        \omterm{prop:b1:vspaces:coordinates}{COORDINATES}.)}
  \item अभिव्यक्त करना $\dfrac1{1 + \sqrt2 + \sqrt3}$ में \omterm{def:b1:vspaces:free}{आधार} $(1,
        \sqrt2, \sqrt3, \sqrt6)$ \emph{(अच्छी तरह से चुने गए से गुणा करें
        \omterm{def:b1:complex:field}{संयुग्मित})}.
  \item संश्लेषण, चार वाक्यों में: क्यों $\Q$-आयाम एक है
        \emph{अंकगणित} ए का अपरिवर्तनीय \omterm{def:b1:logic:sets}{तय करना} यथार्थ का; क्या
        आयाम की परिमितता प्रत्येक तत्व पर बल देती है
        (प्रश्न 13); दो-में-तीन नियम कैसे उत्पन्न हुआ
        पतली हवा से उलटा (प्रश्न 12); और क्या
        \omterm{pb:b1:findim:1}{टावर कानून} निषेध (प्रश्न 20)। सिद्ध प्रमेय का नाम बताइए
        भाग III में.
\end{enumerate}
\end{problem}
