\chapter{मैट्रिसेस}\label{ch:b1:matrices}

एक मैट्रिक्स एक है \omterm{def:b1:linmaps:def}{रेखीय} मानचित्र में लिखा है \omterm{prop:b1:vspaces:coordinates}{COORDINATES}. यह अध्याय सेट
शब्दकोष के ऊपर --- रचना मैट्रिक्स उत्पाद, वस्तुनिष्ठता बन जाती है
उलटापन हो जाता है, \omterm{def:b1:vspaces:free}{आधार} का \omterm{def:b1:counting:objects}{परिवर्तन} संयुग्मन बन जाता है --- और
एल्गोरिथम पक्ष: पंक्ति संचालन, रैंकों की गणना और
व्युत्क्रम। पहली बार हाई स्कूल खंड में मिले, मैट्रिसेस अब हैं
के सिद्धांत पर \omterm{def:b1:vspaces:free}{आधार}ित है
\cref{ch:b1:vspaces,ch:b1:findim,ch:b1:linmaps}.

\section{मैट्रिसेस और रैखिक मानचित्र}

\begin{definition}\label{def:b1:matrices:def}
$\mathcal{M}_{n,p}(K)$ है सदिश स्थल का $ n \times p $ सरणियों $ A
= (a_{ij})$ अदिश का ($ i $: पंक्ति, $ j $: स्तंभ), आयाम का $ np $
(\omterm{def:b1:vspaces:free}{आधार}: मैट्रिक्स $ E_{ij}$ एक के साथ $1$). दिए गए \omterm{def:b1:vspaces:free}{आधार}
$\mathcal{B} = (e_1, \dots, e_p)$ का $ E $ और $\mathcal{C}$ का $ F $
($\dim F = n $), द \emph{का मैट्रिक्स $ u \in \mathcal{L}(E, F)$} है
वह सारणी जिसका $ j $-वां कॉलम सूचीबद्ध करता है \omterm{prop:b1:vspaces:coordinates}{COORDINATES} का $ u(e_j)$ में
$\mathcal{C}$:
\[
\operatorname{Mat}_{\mathcal{B},\mathcal{C}}(u) = (a_{ij}),
\qquad u(e_j) = \sum_{i=1}^{n} a_{ij}\, f_i .
\]
 \omterm{def:b1:logic:map}{नक्शा} $ u \mapsto \operatorname{Mat}_{\mathcal{B},\mathcal{C}}(u)$
से एक \omterm{def:b1:linmaps:projection}{समरूपता} है $\mathcal{L}(E, F)$ पर
$\mathcal{M}_{n,p}(K)$ (\cref{prop:b1:linmaps:basis}: ए \omterm{def:b1:linmaps:def}{रेखीय} मानचित्र
बिल्कुल एक विकल्प है इमेजिस की $ e_j $).
\end{definition}

\begin{example}[मैट्रिक्स के रूप में व्युत्पन्न]
\label{ex:b1:matrices:derivative}
होने देना $ D(P) = P'$ पर $\R_3[X]$. एकपदी में \omterm{def:b1:vspaces:free}{आधार} $(1, X, X^2,
X^3)$: $ D(1) = 0$, $ D(X) = 1$, $ D(X^2) = 2X $, $ D(X^3) = 3X^2$, इसलिए
\[
\operatorname{Mat}(D) =
\begin{pmatrix}
0 & 1 & 0 & 0\\
0 & 0 & 2 & 0\\
0 & 0 & 0 & 3\\
0 & 0 & 0 & 0
\end{pmatrix}.
\]
में \emph{विभाजित} \omterm{def:b1:vspaces:free}{आधार} $\bigl(1,\ X,\ \frac{X^2}2,\
\frac{X^3}6\bigr)$, प्रत्येक \omterm{def:b1:vspaces:free}{आधार} वेक्टर एमएपीएस पिछले वाले को
($ D\bigl(\frac{X^k}{k!}\bigr) = \frac{X^{k-1}}{(k-1)!}$), और
मैट्रिक्स शुद्ध बदलाव बन जाता है: सुपरडायगोनल पर, शून्य
अन्यत्र. दो नैतिकताएँ: मैट्रिक्स का संबंध है \emph{जोड़ा}
(\omterm{def:b1:logic:map}{नक्शा}, \omterm{def:b1:vspaces:free}{आधार}), को नहीं \omterm{def:b1:logic:map}{नक्शा} अकेला; और एक अच्छा \omterm{def:b1:vspaces:free}{आधार} बनाता है
संरचना एक नज़र में दिखाई देती है --- शिफ्ट फॉर्म तुरंत दिखाई देता है
वह $ D^4 = 0$ पर $\R_3[X]$, मैट्रिक्स की प्रत्येक शक्ति इसे आगे बढ़ा रही है
लोगों का विकर्ण एक कदम आगे।
\end{example}

\begin{definition}[उत्पाद]\label{def:b1:matrices:product}
के लिए $ A \in \mathcal{M}_{n,p}$ और $ B \in \mathcal{M}_{p,q}$:
\[
(AB)_{ik} = \sum_{j=1}^{p} a_{ij}\, b_{jk}
\qquad (1 \leq i \leq n,\ 1 \leq k \leq q).
\]
यह बिल्कुल रचना का मैट्रिक्स है:
$\operatorname{Mat}(v \circ u) = \operatorname{Mat}(v)\,
\operatorname{Mat}(u)$ (\omterm{def:b1:vspaces:free}{आधार} बीच में मेल खाते हैं)। इसी प्रकार,
यदि $ X $ का स्तम्भ है \omterm{prop:b1:vspaces:coordinates}{COORDINATES} का $ x $, का स्तंभ $ u(x)$ है
$ AX $.
\end{definition}

\begin{proof}[रचना सूत्र का प्रमाण]
\[
v(u(e_k)) = v\Bigl(\sum_j b_{jk} f_j\Bigr) = \sum_j b_{jk}\, v(f_j)
= \sum_j b_{jk} \sum_i a_{ij}\, g_i
= \sum_i \Bigl(\sum_j a_{ij} b_{jk}\Bigr) g_i . \qedhere
\]
\end{proof}

\begin{proposition}[बीजगणित $\mathcal{M}_n(K)$]\label{prop:b1:matrices:ring}
वर्गाकार आव्यूह $\mathcal{M}_n(K)$ प्रपत्र a (के लिए गैर-विनिमेय) $ n
\geq 2$) \omterm{def:b1:structures:ring}{अँगूठी}, पहचान के साथ $ I_n $; इसका \omterm{def:b1:structures:group}{समूह} इकाइयों का है
\emph{सामान्य रैखिक समूह}\index{सामान्य रैखिक समूह}
$ GL_n(K)$, तदनुसार \omterm{def:b1:logic:inj}{द्विभाजित} एंडोमोर्फिज्म। के लिए $ A, B \in
\mathcal{M}_n(K)$:
\[
AB = I_n \implies A \in GL_n(K) \text{ and } B = A^{-1}
\]
(एकतरफा व्युत्क्रम दोतरफा होते हैं
\cref{cor:b1:linmaps:samedim}).
\end{proposition}

\begin{proof}
\omterm{def:b1:structures:ring}{अँगूठी} स्वयंसिद्धों से परिवहन होता है $\mathcal{L}(E)$ के माध्यम से
की \omterm{def:b1:linmaps:projection}{समरूपता} \cref{def:b1:matrices:def}: यह रूपांतरित करता है
रचना को उत्पाद में और योग को योग में, इसलिए साहचर्यता,
वितरणशीलता और भूमिका $ I_n $ से विरासत में मिले हैं
के बारे में संबंधित तथ्य एमएपीएस, बिना किसी प्रविष्टिवार सत्यापन के।
गैर-कम्यूटेटिविटी: $ E_{12}E_{21} = E_{11} \neq E_{22}
= E_{21}E_{12}$. अगर $ AB = I_n $: एंडोमोर्फिज्म $ a $ का $ A $
संतुष्ट $ a \circ b = \mathrm{id}$, इसलिए $ a $ है \omterm{def:b1:logic:inj}{विशेषण} ($ x =
a(b(x))$ एक प्रदर्शित करता है \omterm{def:b1:logic:map}{पूर्वछवि} हरेक का $ x $), इस तरह \omterm{def:b1:logic:inj}{द्विभाजित} में
परिमित आयाम (\cref{cor:b1:linmaps:samedim}); लिखना $ a
\circ b = \mathrm{id}$ साथ $ a^{-1}$ बाईं ओर देता है $ b =
a^{-1}$, और तब $ b\circ a = \mathrm{id}$ भी: एकतरफ़ा
उलटा हर समय दोतरफा था --- एक सख्ती से
\omterm{def:b1:findim:def}{परिमित-आयामी} कृपादृष्टि।
\end{proof}

\begin{definition}[स्थानांतरण; पता लगाना]\label{def:b1:matrices:transpose}
 \emph{खिसकाना} का $ A = (a_{ij}) \in \mathcal{M}_{n,p}$ है $ A^{\mathsf T}
= (a_{ji}) \in \mathcal{M}_{p,n}$; यह संतुष्ट करता है $(AB)^{\mathsf T} = B^{\mathsf T}
A^{\mathsf T}$ और $(A^{\mathsf T})^{\mathsf T} = A $.
\emph{पता लगाना}\index{पता लगाना} एक वर्ग मैट्रिक्स का है $\operatorname{tr} A
= \sum_i a_{ii}$; यह है \omterm{def:b1:linmaps:def}{रेखीय}, और
\[
\operatorname{tr}(AB) = \operatorname{tr}(BA)
\qquad (A \in \mathcal{M}_{n,p},\ B \in \mathcal{M}_{p,n}).
\]
\end{definition}

\begin{proof}[ट्रेस पहचान का प्रमाण]
$\operatorname{tr}(AB) = \sum_i \sum_j a_{ij} b_{ji}$ और
$\operatorname{tr}(BA) = \sum_j \sum_i b_{ji} a_{ij}$: वही
दोगुना योग.
\end{proof}

\begin{example}[काम पर निशान]
\label{ex:b1:matrices:tracework}
 अनुमान का \cref{ch:b1:linmaps} पर
$\operatorname{Vect}(1,1)$ साथ में $\operatorname{Vect}(0,1)$,
$ p(x, y) = (x, x)$, मैट्रिक्स है $ A = \begin{pmatrix} 1 & 0\\ 1 &
0\end{pmatrix}$ विहित में \omterm{def:b1:vspaces:free}{आधार}: वास्तव में $ A^2 = A $, और
\[
\operatorname{tr} A = 1 = \operatorname{rk} A ,
\]
illustrating \cref{exo:b1:matrices:8}: बेरोजगारों के लिए \omterm{def:b1:matrices:transpose}{पता लगाना}
\emph{गिनता} का आयाम \omterm{def:b1:linmaps:kerim}{छवि}, जो भी तिरछा हो \omterm{def:b1:vspaces:free}{आधार}
मैट्रिक्स में लिखा गया है। अपरिवर्तनीय तंत्र है
पहचान $\operatorname{tr}(AB) = \operatorname{tr}(BA)$:
\[
\operatorname{tr}\bigl(P^{-1}(AP)\bigr) =
\operatorname{tr}\bigl((AP)P^{-1}\bigr) = \operatorname{tr} A ,
\]
अतः सभी आव्यूह \omterm{thm:b1:matrices:conjugation}{समान} हैं $ A $ इसे साझा करें \omterm{def:b1:matrices:transpose}{पता लगाना} --- पहला
\emph{संख्यात्मक अपरिवर्तनीय} एक एंडोमोर्फिज्म का, जिसमें शामिल होना है
में निर्धारक \cref{ch:b1:det} (जोड़ी $(s, p)$ का
सप्ताहांत समस्या नीचे दी गई है)।
\end{example}

\begin{example}[सममित प्लस एंटीसिमेट्रिक]
\label{ex:b1:matrices:symsplit}
पुकारना $ A $ \emph{सममित} कब $ A^{\mathsf T} = A $,
\emph{antisymmetric} कब $ A^{\mathsf T} = -A $. हर चौक
मैट्रिक्स विशिष्ट रूप से एक प्लस दूसरे के रूप में विभाजित होता है:
\[
A = \underbrace{\frac{A + A^{\mathsf T}}{2}}_{\text{symmetric}}
+ \underbrace{\frac{A - A^{\mathsf T}}{2}}_{\text{antisymmetric}},
\]
और एक मैट्रिक्स जो दोनों शून्य है ($ A = -A $): दो सेट हैं
\omterm{def:b1:vspaces:sum}{पूरक} उपस्थान का $\mathcal{M}_n(K)$ --- सटीक
कार्यों के सम/विषम विभाजन का एनालॉग
(\cref{ex:b1:vspaces:evenodd}), ट्रांसपोज़िशन प्लेइंग के साथ
की भूमिका $ x \mapsto -x $. आयाम: एक सममित मैट्रिक्स है \omterm{def:b1:vspaces:free}{मुक्त}
विकर्ण पर और ऊपर, एक एंटीसिमेट्रिक सख्ती से ऊपर
(शून्य विकर्ण):
\[
\frac{n(n+1)}{2} + \frac{n(n-1)}{2} = n^2 ,
\]
और गिनती संतुलन ग्रासमैन की पुष्टि है
प्रत्यक्षता. के लिए $ n = 2$: $\begin{pmatrix} 1 & 5\\ 1 &
2\end{pmatrix} = \begin{pmatrix} 1 & 3\\ 3 & 2\end{pmatrix} +
\begin{pmatrix} 0 & 2\\ -2 & 0\end{pmatrix}$. सममित
मैट्रिक्स के रूप में वापस आते हैं द्वितीय-व्युत्पन्न का डेटा
\cref{ch:b1:multivar} (मोंज ट्रिपल $ r, s, t $), और
सममित-ऑर्थोगोनल को वर्गीकृत किया गया है
\cref{exo:b1:euclid:12}.
\end{example}

\section{आधार परिवर्तन}

\begin{definition}\label{def:b1:matrices:changeofbasis}
होने देना $\mathcal{B}, \mathcal{B}'$ के \omterm{def:b1:vspaces:free}{आधार} बनें $ E $. \emph{का परिवर्तन
आधार मैट्रिक्स}\index{का परिवर्तन
आधार मैट्रिक्स} $ P =
P_{\mathcal{B}\to\mathcal{B}'}$ कॉलम के लिए है \omterm{prop:b1:vspaces:coordinates}{COORDINATES} की \emph{नया} \omterm{def:b1:vspaces:free}{आधार} वेक्टर में \emph{पुराना} \omterm{def:b1:vspaces:free}{आधार}. यह है
उलटा, $ P^{-1} = P_{\mathcal{B}'\to\mathcal{B}}$, और
\omterm{prop:b1:vspaces:coordinates}{COORDINATES} द्वारा रूपांतरित करें $ X = PX'$ (पुराना $=$ $ P\,\cdot $ नया)।
\end{definition}

\begin{example}[परिवर्तन-के-आधार मैट्रिक्स को पढ़ना]
\label{ex:b1:matrices:changerun}
में $\R^2$, विहित से $\mathcal B $ को $\mathcal B' =
\bigl((1,1), (1,-1)\bigr)$:
\[
P = P_{\mathcal B\to\mathcal B'} =
\begin{pmatrix} 1 & 1\\ 1 & -1 \end{pmatrix}
\]
(नए वेक्टर पुराने में लिखे गए हैं \omterm{prop:b1:vspaces:coordinates}{COORDINATES}, कॉलम दर कॉलम)। द
पुराने का वेक्टर \omterm{prop:b1:vspaces:coordinates}{COORDINATES} $ X = (3, 1)^{\mathsf T}$ नया है
\omterm{prop:b1:vspaces:coordinates}{COORDINATES} $ X' = P^{-1}X = \frac12(3 + 1,\ 3 - 1)^{\mathsf T} =
(2, 1)^{\mathsf T}$: वास्तव में $2(1,1) + 1(1,-1) = (3,1)$. ध्यान रखें
दिशा --- मैट्रिक्स $ P $ से निर्मित है \emph{नया} \omterm{def:b1:vspaces:free}{आधार}
लेकिन धर्म \omterm{def:b1:counting:objects}{परिवर्तन} करता है \emph{नये से पुराने तक} \omterm{prop:b1:vspaces:coordinates}{COORDINATES} ($ X = PX'$); गुजर रहा है
पुराने से नए की लागत विपरीत होती है। विवेक जांच लिखना
$2(1,1) + (1,-1) = (3,1)$ प्रत्येक रूपांतरण के बाद पकड़ लेता है
उलटा-$ P $ त्रुटि, जो कि सबसे आम गलती है
अध्याय.
\end{example}

\begin{theorem}[मानचित्र के आधार में परिवर्तन]\label{thm:b1:matrices:conjugation}
होने देना $ u \in \mathcal{L}(E)$ मैट्रिक्स के साथ $ A $ में $\mathcal{B}$ और $ A'$
में $\mathcal{B}'$, और $ P = P_{\mathcal{B}\to\mathcal{B}'}$. तब
\[
A' = P^{-1} A\, P .
\]
इस प्रकार संबंधित दो आव्यूह हैं \emph{समान}\index{समान}. (के लिए $ u \colon E \to F $ \omterm{def:b1:vspaces:free}{आधार}ों के दो जोड़े के साथ,
सूत्र है $ A' = Q^{-1} A P $ --- \emph{समकक्ष}
मैट्रिक्स\index{समतुल्य आव्यूह}.)
\end{theorem}

\begin{proof}
किसी के लिए $ x $: $ X = PX'$ और यह \omterm{def:b1:linmaps:kerim}{छवि} संतुष्ट $ Y = AX $, $ Y = PY'$.
इसलिए $ PY' = APX'$, यानी\ $ Y' = (P^{-1}AP)X'$ सभी के लिए $ X'$: मैट्रिक्स
का $ u $ नए में \omterm{def:b1:vspaces:free}{आधार} है $ P^{-1}AP $ (के लिए ले $ X'$ विहित
कॉलम)।
\end{proof}

\begin{example}[एक अच्छा आधार मानचित्र को पारदर्शी बनाता है]
\label{ex:b1:matrices:conjugationrun}
होने देना $ u(x, y) = (y, x)$ (स्वैप), मैट्रिक्स के साथ $ A = \begin{pmatrix} 0
& 1\\ 1 & 0\end{pmatrix}$ विहित में \omterm{def:b1:vspaces:free}{आधार}. में \omterm{def:b1:vspaces:free}{आधार}
$\mathcal B' = \bigl((1,1), (1,-1)\bigr)$:
\[
P = \begin{pmatrix} 1 & 1\\ 1 & -1 \end{pmatrix},
\qquad
P^{-1} = \frac12\begin{pmatrix} 1 & 1\\ 1 & -1 \end{pmatrix},
\qquad
P^{-1} A P = \begin{pmatrix} 1 & 0\\ 0 & -1 \end{pmatrix}.
\]
वास्तव में किसी मैट्रिक्स उत्पाद की आवश्यकता नहीं थी: $ u $ ठीक करता है $(1,1)$ और
उलटा $(1,-1)$, तो में $\mathcal B'$ इसका मैट्रिक्स \emph{अवश्य} होना
$\operatorname{diag}(1, -1)$ --- अदला-बदली प्रतिबिंब है
रेखा के पार $ y = x $. किसी दिए गए एंडोमोर्फिज्म के लिए, ढूँढना, ए
\omterm{def:b1:vspaces:free}{आधार} जिसमें इसका मैट्रिक्स विकर्ण हो जाता है वह केंद्रीय समस्या है
वर्ष का ~2 खंड (कमी सिद्धांत); सप्ताहांत की समस्या
नीचे दिखाया गया है कि कितनी दूर है \omterm{def:b1:poly:def}{बहुपद} अकेले पहचान पहले से ही चली जाती है।
\end{example}

\begin{example}[आधार बदलना, उलटा चलाना]
\label{ex:b1:matrices:reverserun}
 अनुमान पर $ F = \operatorname{Vect}(1,1)$ साथ में $ G =
\operatorname{Vect}(1,-1)$ अनुकूलित में है \omterm{def:b1:vspaces:free}{आधार} $\mathcal
B' = \bigl((1,1),(1,-1)\bigr)$, पारदर्शी मैट्रिक्स $ A' =
\operatorname{diag}(1, 0)$. इसे पाने के लिए विहि\omterm{def:b1:vspaces:free}{त-आधार} मैट्रिक्स,
भागो \cref{thm:b1:matrices:conjugation} पीछे की ओर, $ A = P A'
P^{-1}$:
\[
P = \begin{pmatrix} 1 & 1\\ 1 & -1\end{pmatrix},
\quad
P^{-1} = \frac12\begin{pmatrix} 1 & 1\\ 1 & -1\end{pmatrix},
\quad
A = P\begin{pmatrix} 1 & 0\\ 0 & 0\end{pmatrix}P^{-1}
= \frac12\begin{pmatrix} 1 & 1\\ 1 & 1\end{pmatrix}.
\]
जाँच करना: $ A^2 = A $ (निष्क्रिय), $\operatorname{tr} A = 1 =
\operatorname{rk} A $, और $ A\binom{1}{1} = \binom11$,
$ A\binom{1}{-1} = 0$, के रूप में निर्धारित। यह उल्टी दिशा---
मैट्रिक्स को अच्छे से डिज़ाइन करें \omterm{def:b1:vspaces:free}{आधार}, फिर वापस \omterm{def:b1:complex:field}{संयुग्मित} --- है
कैसे घूर्णन, परावर्तन और अनुमान मैट्रिक्स वास्तव में हैं
व्यवहार में उत्पादित.
\end{example}

\begin{theorem}[रैंक सामान्य रूप]\label{thm:b1:matrices:rank}
 \emph{रैंक} एक मैट्रिक्स की (इसके स्तंभों की रैंक, समकक्ष
सम्बंधित का \omterm{def:b1:linmaps:def}{रेखीय} मानचित्र) समतुल्यता का एकमात्र अपरिवर्तनीय है:
प्रत्येक $ A \in \mathcal{M}_{n,p}$ रैंक का $ r $ के बराबर है
\[
J_r = \begin{pmatrix} I_r & 0 \\ 0 & 0 \end{pmatrix},
\]
और $\operatorname{rk}(A^{\mathsf T}) = \operatorname{rk}(A)$: पंक्ति रैंक
कॉलम रैंक के बराबर है।
\end{theorem}

\begin{proof}
होने देना $ u \colon E \to F $ रैंक है $ r $. एक विकल्प चुनें \omterm{def:b1:vspaces:sum}{पूरक} $ S $ का
$\ker u $ ($\dim S = r $, \cref{thm:b1:linmaps:ranknullity}) साथ
\omterm{def:b1:vspaces:free}{आधार} $(e_1, \dots, e_r)$, ए द्वारा पूरा किया गया \omterm{def:b1:vspaces:free}{आधार} का $\ker u $ एक में
\omterm{def:b1:vspaces:free}{आधार} का $ E $; इमेजिस $ f_i = u(e_i)$, $ i \leq r $, फॉर्म ए \omterm{def:b1:vspaces:free}{आधार} का
$\operatorname{im} u $ (प्रतिबंध एक \omterm{def:b1:linmaps:projection}{समरूपता} है), पूरा हुआ
एक में \omterm{def:b1:vspaces:free}{आधार} का $ F $. इन \omterm{def:b1:vspaces:free}{आधार}ों में का मैट्रिक्स $ u $ बिलकुल है
$ J_r $. इसलिए $ A = Q J_r P^{-1}$ उलटे के लिए $ P, Q $.

ट्रांसपोज़िंग: $ A^{\mathsf T} = (P^{-1})^{\mathsf T} J_r^{\mathsf T} Q^{\mathsf T}$
साथ $ J_r^{\mathsf T}$ एक ही रूप का (रैंक) $ r $) और बाहरी
व्युत्क्रमणीय कारक (खिसकाना व्युत्क्रमणीय का व्युत्क्रमणीय है, से
$(AB)^{\mathsf T} = B^{\mathsf T}A^{\mathsf T}$ के लिए आवेदन किया $ AA^{-1} = I $):
$\operatorname{rk} A^{\mathsf T} = r $.
\end{proof}

\section{पंक्ति संचालन}

\begin{method}[मैट्रिक्स पर गाऊसी उन्मूलन]\label{met:b1:matrices:gauss}
तीन \emph{प्रारंभिक पंक्ति संचालन}\index{प्रारंभिक पंक्ति संचालन} ---
दो पंक्तियों की अदला-बदली करें, एक पंक्ति को इससे गुणा करें $\lambda \neq 0$, का गुणज जोड़ें
एक पंक्ति से दूसरी पंक्ति --- रैंक न बदलें (प्रत्येक को छोड़ दिया गया है
एक व्युत्क्रमणीय मैट्रिक्स द्वारा गुणा)। एल्गोरिथम: एक धुरी बनाएं
(सबसे बायीं ओर शून्येतर प्रविष्टि), नीचे इसका कॉलम साफ़ करें, अगले पर जाएँ
पंक्ति और स्तंभ; परिणामी सोपानक रूप की धुरी की संख्या
रैंक है.

\emph{उलटा गणना:} ब्लॉक पर एल्गोरिदम चलाएँ $(A \mid
I_n)$ जब तक बायां ब्लॉक नहीं बन जाता $ I_n $ (यदि संभव हो तो) $ A $ है
उलटा); सही ब्लॉक तो है $ A^{-1}$ --- वास्तव में उत्पाद
उपयोग किए गए प्राथमिक आव्यूहों का उपयोग बराबर होता है $ A^{-1}$.
\end{method}

\begin{example}\label{ex:b1:matrices:inverse}
$ A = \begin{pmatrix} 1 & 2 \\ 3 & 4 \end{pmatrix}$: कम करना $(A \mid
I_2)$:
\[
\begin{pmatrix} 1 & 2 & 1 & 0\\ 3 & 4 & 0 & 1 \end{pmatrix}
\to
\begin{pmatrix} 1 & 2 & 1 & 0\\ 0 & -2 & -3 & 1 \end{pmatrix}
\to
\begin{pmatrix} 1 & 0 & -2 & 1\\ 0 & 1 & \tfrac32 & -\tfrac12
\end{pmatrix},
\]
(संचालन: $ L_2 \leftarrow L_2 - 3L_1$; तब $ L_1 \leftarrow L_1
+ L_2$, $ L_2 \leftarrow -\frac12 L_2$). इसलिए $ A^{-1} =
\begin{pmatrix} -2 & 1 \\ \tfrac32 & -\tfrac12\end{pmatrix}$.
\emph{जाँच करना:} $ AA^{-1} = I_2$.
\end{example}

\begin{example}[केवल पंक्तियों के आधार पर एक पैरामीटर के साथ रैंक करें]
\label{ex:b1:matrices:rankparameter}
के लिए $ m \in \R $, का पद
$ M_m = \begin{pmatrix} 1 & 1 & m\\ 1 & m & 1\\ m & 1 &
1\end{pmatrix}$. कम करना: $ L_2 \leftarrow L_2 - L_1$ और $ L_3
\leftarrow L_3 - mL_1$ पंक्तियाँ दें
\[
(1,\ 1,\ m), \qquad (0,\ m - 1,\ 1 - m), \qquad
(0,\ 1 - m,\ 1 - m^2).
\]
\emph{मामला $ m = 1$}: अंतिम दो पंक्तियाँ गायब हो जाती हैं --- एक धुरी,
$\operatorname{rk} M_1 = 1$ (सभी तीन मूल पंक्तियाँ थीं
बराबर)। \emph{मामला $ m \neq 1$}: पैमाना $ L_2$ द्वारा $\frac1{m-1}$ और
$ L_3$ द्वारा $\frac1{1-m}$ पाने के $(0, 1, -1)$ और $(0, 1, 1 + m)$,
तब $ L_3 \leftarrow L_3 - L_2 = (0, 0, m + 2)$. अगर $ m = -2$:
दो धुरी, रैंक $2$; अन्यथा तीन धुरी, रैंक $3$. सारांश:
\[
\operatorname{rk} M_m =
\begin{cases}
1 & m = 1,\\
2 & m = -2,\\
3 & \text{otherwise}.
\end{cases}
\]
\omterm{thm:b1:matrices:conjugation}{समान} सीमाएँ एक निर्धारक गणना से बाहर हो जाएंगी
में \cref{ch:b1:det} (द \omterm{def:b1:poly:def}{बहुपद} $-(m+2)(m-1)^2$ का
\cref{exo:b1:det:7}) --- लेकिन ध्यान दें कि उन्मूलन क्या देता है
निर्धारक नहीं है: \emph{कीमत} में रैंक का
पतित मामले, केवल तथ्य यह नहीं कि यह गिरा।
\end{example}

\begin{example}[कंप्यूटिंग शक्तियाँ]\label{ex:b1:matrices:powers}
$ A = \begin{pmatrix} 1 & 1 \\ 0 & 1\end{pmatrix} = I + N $ साथ $ N =
E_{12}$, $ N^2 = 0$. तब से $ I $ और $ N $ लघुकरण, द्विपद प्रमेय
(\cref{prop:b1:structures:binomial}) काटता है:
\[
A^k = I + kN = \begin{pmatrix} 1 & k \\ 0 & 1 \end{pmatrix}
\qquad (k \in \N, \text{ and } k \in \Z \text{ using } A^{-1} = I -
N).
\]
\end{example}

\begin{method}[कम्प्यूटिंग $ A^n $: तीन मार्ग]
\label{met:b1:matrices:powers}
\begin{enumerate}
  \item \emph{द्विपद मार्ग}: अगर $ A = \lambda I + N $ साथ $ N $
        निलपोटेंट, द्विपद प्रमेय छोटा हो जाता है
        (\cref{ex:b1:matrices:powers}, \cref{exo:b1:matrices:5});
        यह लागू होता है क्योंकि $\lambda I $ हर चीज़ के साथ आवागमन करता है।
  \item \emph{\omterm{def:b1:poly:def}{बहुपद} मार्ग}: लगता है \omterm{def:b1:poly:def}{बहुपद} पहचान
        से संतुष्ट $ A $ (आयाम में $2$, हमेशा $ A^2 = sA -
        pI $) और कम करें $ X^n $ मॉड्यूलो यह; सप्ताहांत की समस्या
        नीचे इस मार्ग को पूरी तरह से बनाया गया है।
  \item \emph{समानता मार्ग}: एक उलटा खोजें $ P $ साथ
        $ P^{-1}AP = D $ सरल (विकर्ण, या शिफ्ट), गणना करें
        $ D^n $, और पूर्ववत करें: $ A^n = P D^n P^{-1}$
        (\cref{thm:b1:matrices:conjugation},
        \cref{ex:b1:matrices:conjugationrun}); व्यवस्थित
        ऐसे खोजें $ P $ वर्ष~2 का न्यूनीकरण सिद्धांत है।
\end{enumerate}
मार्ग जो भी हो, परिणाम की जाँच करें $ n = 0, 1, 2$: तीन
सस्ते परीक्षण जो लगभग हर गलती को पकड़ लेते हैं।
\end{method}

\begin{remark}[सामान्य नुकसान: गैर-कम्यूटेटिविटी की कीमत]
\label{rem:b1:matrices:pitfalls}
अदिश बीजगणित की प्रत्येक पहचान जिसका प्रमाण कारकों को पुन: व्यवस्थित करता है
में मर जाता है $\mathcal{M}_n(K)$, $ n \geq 2$.
\emph{वर्गों}: $(A + B)^2 = A^2 + AB + BA + B^2$, और
बीच का पतन हो जाता है $2AB $ केवल $ AB = BA $
(\cref{exo:b1:matrices:1}).
\emph{उत्पादों की शक्तियाँ}: $(AB)^k $ है $ ABAB\cdots $, नहीं
$ A^kB^k $.
\emph{शून्य भाजक}: $ E_{12}E_{12} = 0$ साथ $ E_{12} \neq 0$;
फलस्वरूप \emph{कोई रद्दीकरण नहीं}: $ AB = AC $ तात्पर्य $ B = C $
केवल जब $ A $ उलटा है (से गुणा करें)। $ A^{-1}$ --- पर
सही पक्ष)।
\emph{निशान}: $\operatorname{tr}(AB) = \operatorname{tr}(BA)$
हमेशा, लेकिन $\operatorname{tr}(AB) \neq
\operatorname{tr}A\operatorname{tr}B $ सामान्य तौर पर (ले लो $ A = B =
I_2$: $2 \neq 4$), और $\operatorname{tr}(ABC) =
\operatorname{tr}(BCA)$ (चक्रीय) जबकि $\operatorname{tr}(ACB)$
भिन्न हो सकता है.
\emph{स्थानान्तरण रिवर्स}: $(AB)^{\mathsf T} = B^{\mathsf
T}A^{\mathsf T}$ --- उलटाव को भूल जाना सबसे आम है
ऑर्थोगोनैलिटी संगणना में त्रुटि (\cref{ch:b1:euclid}).
संदेह होने पर, किसी भी दावा की गई पहचान का परीक्षण करें $ E_{12}$ और
$ E_{21}$: सबसे छोटी गैर-यात्रा जोड़ी सबसे झूठ का खंडन करती है
एक पंक्ति में सूत्र.
\end{remark}

\begin{remark}[शब्दकोश कहाँ जाता है]
\label{rem:b1:matrices:whereused}
इसके प्रत्येक शेष पृष्ठ पर मैट्रिक्स डिक्शनरी का उपयोग किया गया है
आयतन: \cref{ch:b1:det} प्रत्येक वर्ग मैट्रिक्स को एक एकल जोड़ता है
संख्या निर्णय लेने वाली व्युत्क्रमणीयता, और हल करती है $ AX = B $
व्यवस्थित रूप से; \cref{ch:b1:euclid} मैट्रिक्स को अलग करता है
लंबाई का संरक्षण (ऑर्थोगोनल मैट्रिक्स); और में
\cref{ch:b1:multivar}, किसी फ़ंक्शन का दूसरे क्रम का व्यवहार
दो चर एक सममित है $2 \times 2$ मैट्रिक्स. \omterm{def:b1:matrices:transpose}{पता लगाना},
लगभग पारित होते ही ऊपर प्रस्तुत किया गया, एक शक्तिशाली अपरिवर्तनीय बन जाता है:
\cref{exo:b1:matrices:6,exo:b1:matrices:8} पहले स्वाद दो,
और वर्ष~2 खंड इस पर स्वदेशी सिद्धांत का निर्माण करता है। सप्ताहांत
समस्या अन्य वर्कहॉर्स को विकसित करती है: \omterm{def:b1:poly:def}{बहुपद} पहचान
एक मैट्रिक्स से संतुष्ट, जो की गणना को बदल देता है $ A^n $ में
एक दो \omterm{def:b1:vspaces:span}{अवधि} \omterm{pb:b1:matrices:1}{रैखिक पुनरावृत्ति}.
\end{remark}

\begin{remark}[पुस्तक 3 के अंदर परिप्रेक्ष्य]
\label{rem:b1:matrices:perspectives}
यहां पेश किए गए तीन मैट्रिक्स परिवारों की बाद में नियुक्तियां होंगी
यह वॉल्यूम. \emph{सममित मैट्रिक्स}
(\cref{ex:b1:matrices:symsplit}) दूसरे क्रम का डेटा ले जाएं
दो चरों के कार्य: स्पंज परीक्षण
\cref{ch:b1:multivar} एक है \omterm{def:b1:logic:statement}{कथन} के संकेत व्यवहार के बारे में
एक सममित $2\times2$ मैट्रिक्स, और इसके निर्धारक $ rt - s^2$
द्वारा गणना की जाती है \cref{ch:b1:det}की मशीनरी. \emph{ओर्थोगोनल
matrices} ($ A^{\mathsf T}A = I $) की आइसोमेट्री हैं
\cref{ch:b1:euclid}, जहां खिसकाना अंततः उसे प्राप्त कर लेता है
ज्यामितीय अर्थ: यह भीतर की \omterm{pb:b1:logic:1}{बीजगणितीय} छाया है
उत्पाद. \emph{व्युत्क्रमणीय मैट्रिक्स} उनके व्यावहारिक परीक्षण को पूरा करें
में \cref{ch:b1:det} --- एक नंबर, $\det A \neq 0$ --- समापन
इस अध्याय की खोज पंक्ति कटौती के साथ शुरू हुई। \omterm{def:b1:matrices:transpose}{पता लगाना} और
निर्धारक तब अपरिवर्तनीय जोड़ी के रूप में यात्रा करता है $(s, p)$ का
सप्ताहांत की समस्या, आइजेनवैल्यू सिद्धांत तक सभी तरह से
वर्ष~2.
\end{remark}

\section{अभ्यास}

\begin{exercise}[$\star $]\label{exo:b1:matrices:1}
होने देना $ A = \begin{pmatrix} 1 & 2 \\ 0 & 1 \end{pmatrix}$ और $ B =
\begin{pmatrix} 0 & 1 \\ 1 & 0\end{pmatrix}$. गणना करना $ AB $, $ BA $,
$ A^2 - B^2$ और $(A+B)(A-B)$; स्पष्ट करें कि अंतिम दो भिन्न क्यों हैं।
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:matrices:2}
की रैंक की गणना करें
\[
M = \begin{pmatrix}
1 & 2 & 3\\
2 & 4 & 6\\
1 & 1 & 1
\end{pmatrix},
\qquad
N = \begin{pmatrix}
1 & 1 & 0 & 2\\
0 & 1 & 1 & 1\\
1 & 2 & 1 & 3
\end{pmatrix}.
\]
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:matrices:3}
उलटा, पंक्ति में कमी से, $ A = \begin{pmatrix} 1 & 0 & 1\\ 2 & 1 &
1\\ 1 & 1 & 1 \end{pmatrix}$, और एक उत्पाद पर जाँच करें।
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:matrices:4}
मैट्रिक्स को विहित में लिखें \omterm{def:b1:vspaces:free}{आधार} का $\R_2[X]$, का
एंडोमोर्फिज्म $ u(P) = P(X + 1)$. बिना हिसाब लगाए स्पष्ट करें कि क्यों
यह उलटा है, और इसका मैट्रिक्स दें $ u^{-1}$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:matrices:5}
होने देना $ A = \begin{pmatrix} 2 & 1 \\ 0 & 2\end{pmatrix}$. लिखना $ A =
2I + N $, गणना करें $ N^2$, और निष्कर्ष निकालें $ A^k $ सभी के लिए $ k \in \N $ द्वारा
द्विपद प्रमेय.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:matrices:6}
साबित करें कि कोई मैट्रिक्स नहीं हैं $ A, B \in \mathcal{M}_n(K)$ (साथ
$ K = \R $ या $\C $) ऐसा है कि $ AB - BA = I_n $. \emph{(लेना निशान.)}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:matrices:7}
एक मैट्रिक्स $ A $ है \emph{निलपोटेंट} कब $ A^m = 0$ कुछ के लिए $ m $. साबित करें कि $ I - A $ फिर उलटा है, साथ
\[
(I - A)^{-1} = I + A + A^2 + \dots + A^{m-1} .
\]
अनुप्रयोग: उलटा $\begin{pmatrix} 1 & 2 & 3\\ 0 & 1 & 2\\ 0 & 0 &
1\end{pmatrix}$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:matrices:8}
होने देना $ A \in \mathcal{M}_n(\R)$ संतुष्ट $ A^2 = A $ (निष्क्रिय) साबित करें कि $\operatorname{tr} A = \operatorname{rk} A $. \emph{(व्याख्या
$ A $ के तौर पर अनुमान और एक अनुकूलित चुनें \omterm{def:b1:vspaces:free}{आधार};
\cref{thm:b1:matrices:conjugation} का कहना है \omterm{def:b1:matrices:transpose}{पता लगाना} है
आधार-स्वतंत्र तब से $\operatorname{tr}(P^{-1}MP) =
\operatorname{tr} M $.)}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b1:matrices:9}
होने देना $ J \in \mathcal{M}_n(\R)$ सभी का मैट्रिक्स बनें। गणना करना $ J^2$,
और निष्कर्ष निकालें, के लिए $ a, b \in \R $, की व्युत्क्रमणीयता की स्थिति $ M
= aI + bJ $ के साथ साथ $ M^{-1}$ \emph{(इसके विपरीत की तलाश करें
एक ही रूप $\alpha I + \beta J $)}.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b1:matrices:10}
(रैंक असमानताएं) के लिए $ A, B \in \mathcal{M}_n(K)$, सिद्ध करना
\[
\operatorname{rk}(A + B) \leq \operatorname{rk} A +
\operatorname{rk} B,
\qquad
\operatorname{rk}(AB) \geq \operatorname{rk} A + \operatorname{rk}
B - n .
\]
\emph{(दूसरे के लिए --- सिल्वेस्टर की असमानता --- लागू करें
रैंक - के प्रतिबंध के लिए शून्यता \omterm{def:b1:logic:map}{नक्शा} का $ A $ को
$\operatorname{im} B $.)}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:matrices:11}
होने देना $ D = \operatorname{diag}(d_1, \dots, d_n)$ साथ $ d_i $
\emph{जोड़ीवार अलग}.
\begin{enumerate}
  \item इसे एक मैट्रिक्स सिद्ध करें $ A $ के साथ आवागमन करता है $ D $ अगर और केवल अगर
        $ A $ विकर्ण है. \emph{(तुलना करें $(i,j)$ की प्रविष्टियाँ
        $ AD $ और $ DA $.)}
  \item निष्कर्ष निकालें \emph{केंद्र} का $\mathcal{M}_n(K)$: द
        मैट्रिसेस के साथ आवागमन \emph{प्रत्येक} मैट्रिक्स बिल्कुल हैं
        अदिश मैट्रिक्स $\lambda I_n $. \emph{(के विरुद्ध परीक्षण करें
        $ D $, फिर मैट्रिक्स के विरुद्ध $ E_{ij}$.)}
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b1:matrices:12}
(रैंक-वन मैट्रिसेस) चलो $ A \in \mathcal{M}_n(K)$, $ A \neq 0$.
\begin{enumerate}
  \item साबित करें कि $\operatorname{rk} A = 1$ अगर और केवल अगर $ A =
        CL $ एक शून्येतर स्तंभ के लिए $ C \in \mathcal{M}_{n,1}$ और ए
        शून्येतर पंक्ति $ L \in \mathcal{M}_{1,n}$.
  \item ऐसे के लिए $ A $, सिद्ध करना $ A^2 = (\operatorname{tr} A)\,A $;
        यह निष्कर्ष निकालें कि रैंक-वन मैट्रिक्स शून्य है यदि और केवल यदि
        इसका \omterm{def:b1:matrices:transpose}{पता लगाना} शून्य है.
  \item अगर $\operatorname{tr} A \neq -1$, साबित करें कि $ I_n + A $ है
        के साथ उलटा
        \[
        (I_n + A)^{-1} = I_n - \frac{1}{1 + \operatorname{tr}
        A}\,A ,
        \]
        ओर वो $ I_n + A $ है \emph{नहीं} उलटा जब
        $\operatorname{tr} A = -1$. \emph{(मारे गए वेक्टर को ढूंढें
        द्वारा $ I_n + A $.)}
\end{enumerate}
\end{exercise}

\section{समस्या: बहुपद विभाजन द्वारा मैट्रिक्स की शक्तियाँ}

\begin{problem}\label{pb:b1:matrices:1}
कम्प्यूटिंग $ A^{100}$ प्रवेश द्वारा प्रवेश निराशाजनक है; इसकी गणना करना
ए के माध्यम से \omterm{def:b1:poly:def}{बहुपद} पहचान से संतुष्ट $ A $ तीन पंक्तियाँ है.
यह समस्या स्क्रैच से विधि बनाती है: यूक्लिडियन डिवीजन
का $ X^n $, पहचान $ A^2 - sA + pI = 0$ प्रत्येक द्वारा सत्यापित $2
\times 2$ मैट्रिक्स (केली--हैमिल्टन प्रमेय आयाम में
$2$), और मैट्रिक्स शक्तियों के बीच शब्दकोश और \omterm{pb:b1:matrices:1}{रैखिक
पुनरावृत्ति} --- साथ \omterm{pb:b1:matrices:1}{फाइबोनैचि संख्याएँ} चल रहे उदाहरण के रूप में।
\index{केली--हैमिल्टन प्रमेय (आयाम 2)}
\index{फाइबोनैचि संख्याएँ}\index{रैखिक पुनरावृत्ति}

\medskip
\noindent\textbf{Part I --- The remainder calculus.} हल करना $ s, p
\in K $ और $ D = X^2 - sX + p $.
\begin{enumerate}
  \item प्रत्येक के लिए इसका औचित्य सिद्ध करें $ n \in \N $ अद्वितीय हैं $ Q_n
        \in K[X]$ और $(a_n, b_n) \in K^2$ साथ
        \[
        X^n = Q_n\,D + a_n X + b_n ,
        \]
        और गणना करें $(a_0, b_0)$ और $(a_1, b_1)$.
  \item से गुणा करना $ X $ और पुनः विभाजित करके स्थापित करें
        पुनरावृत्ति
        \[
        a_{n+1} = s\,a_n + b_n,
        \qquad
        b_{n+1} = -p\,a_n ,
        \]
        और निष्कर्ष निकालें $ a_{n+2} = s\,a_{n+1} - p\,a_n $: द
        गुणांक अनुक्रम का पालन करता है \omterm{pb:b1:matrices:1}{रैखिक पुनरावृत्ति} से जुड़ा $ D $.
  \item कल्पना करना $ D $ इसकी दो अलग-अलग जड़ें हैं $\lambda \neq \mu $.
        विभाजन की पहचान का मूल्यांकन करते हुए सिद्ध करें
        \[
        a_n = \frac{\lambda^n - \mu^n}{\lambda - \mu},
        \qquad
        b_n = \frac{\lambda\mu^n - \mu\lambda^n}{\lambda - \mu} .
        \]
  \item कल्पना करना $ D = (X - \lambda)^2$. का उपयोग यौगिक का
        विभाजन की पहचान, साबित करो $ a_n = n\lambda^{n-1}$ और
        $ b_n = (1 - n)\lambda^{n}$.
  \item दिखाएँ कि एक निश्चित मैट्रिक्स को प्रतिस्थापित करना $ M \in
        \mathcal{M}_k(K)$ में बहुआयामी पद रकम का सम्मान करता है और
        उत्पाद: $(PQ)(M) = P(M)\,Q(M)$. इसका निष्कर्ष निकालें यदि $ D(M) =
        0$, तब
        \[
        M^n = a_n\,M + b_n\,I \qquad (n \in \N).
        \]
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Dimension 2: \omterm{def:b1:matrices:transpose}{पता लगाना}, determinant
number, Cayley--Hamilton.} के लिए $ A = \begin{pmatrix} a & b\\ c &
d\end{pmatrix}$ \omterm{def:b1:logic:sets}{तय करना} $ s = a + d = \operatorname{tr} A $ और $ p = ad
- bc $ (वह संख्या \cref{ch:b1:det} का नाम देगा
निर्धारक)।
\begin{enumerate}[resume]
  \item \omterm{def:b1:vspaces:sum}{प्रत्यक्ष} गणना द्वारा सत्यापित करें \emph{केली--हैमिल्टन
        आयाम में पहचान $2$}:
        \[
        A^2 - s\,A + p\,I_2 = 0 .
        \]
  \item इसे \omterm{def:b1:vspaces:sum}{प्रत्यक्ष} विस्तार से सिद्ध करें $ p $ गुणक है:
        स्पष्ट संकेतन के साथ, $ p(AB) = p(A)\,p(B)$. तो दिखाओ:
        $ A $ उलटा है यदि और केवल यदि $ p \neq 0$, किस स्थिति में
        \[
        A^{-1} = \frac1p\,\bigl(s\,I_2 - A\bigr).
        \]
  \item होने देना $ A = \begin{pmatrix} 1 & 1\\ 0 & 2\end{pmatrix}$.
        गणना करना $ s $, $ p $, की जड़ें $ D $, और एक निष्कर्ष निकालें \omterm{def:b1:topology:closed}{बंद किया हुआ}
        के लिए सूत्र $ A^n $; इसे \omterm{def:b1:vspaces:sum}{प्रत्यक्ष} गणना के विरुद्ध जाँचें
        का $ A^2$.
  \item होने देना $ A = \begin{pmatrix} 3 & 1\\ -1 & 1\end{pmatrix}$.
        बताते हैं कि $ D $ इसमें डबल रूट और कंप्यूट है $ A^n $; जांचें
        पर $ n = 2$.
  \item होने देना $ F = \begin{pmatrix} 1 & 1\\ 1 & 0\end{pmatrix}$ और
        परिभाषित करें \omterm{pb:b1:matrices:1}{फाइबोनैचि संख्याएँ} द्वारा $ F_0 = 0$, $ F_1 = 1$,
        $ F_{n+2} = F_{n+1} + F_n $. सिद्ध करना
        \[
        F^n = \begin{pmatrix} F_{n+1} & F_n\\ F_n &
        F_{n-1}\end{pmatrix} \quad (n \geq 1),
        \]
        बिनेट का सूत्र निकालें $ F_n = \dfrac{\varphi^n -
        \psi^n}{\sqrt5}$ कहाँ $\varphi = \frac{1 + \sqrt5}2$,
        $\psi = \frac{1 - \sqrt5}2$, और, प्रश्न 7 का उपयोग करते हुए,
        कैसिनी की पहचान $ F_{n+1}F_{n-1} - F_n^2 = (-1)^n $.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- \omterm{pb:b1:matrices:1}{रैखिक पुनरावृत्ति}, structurally.}
हल करना $ s, p \in K $ साथ $ p \neq 0$, और चलो $ E_D $ हो \omterm{def:b1:logic:sets}{तय करना} का
अनुक्रम के साथ $ u_{n+2} = s\,u_{n+1} - p\,u_n $ सभी के लिए $ n $.
\begin{enumerate}[resume]
  \item बताते हैं कि $ E_D $ एक है सदिश स्थल आयाम का $2$
        (अनुकूल बनाना \cref{exo:b1:findim:10}).
  \item वह क्रम दिखाओ $(a_n)$ भाग I का तत्व है
        का $ E_D $ प्रारंभिक मूल्यों के साथ $0, 1$, और वह हर $ u
        \in E_D $ संतुष्ट
        \[
        u_n = u_1\,a_n + u_0\,b_n \qquad (n \in \N),
        \]
        साथ $(b_n)$ जैसा कि भाग I में है: विभाजन शेष का समाधान करें
        \emph{सभी} एक ही बार में पुनरावृत्ति हो जाती है।
  \item अगर $\lambda \neq \mu $ की जड़ें हैं $ D $, बताते हैं कि
        $\bigl((\lambda^n), (\mu^n)\bigr)$ एक है \omterm{def:b1:vspaces:free}{आधार} का $ E_D $;
        अगर $ D = (X-\lambda)^2$ साथ $\lambda \neq 0$, बताते हैं कि
        $\bigl((\lambda^n), (n\lambda^n)\bigr)$ एक है.
  \item पूर्णतः हल करें: $ u_{n+2} = u_{n+1} + 6u_n $, $ u_0 = 1$,
        $ u_1 = 8$; उत्तर की जाँच करें $ u_2$ और $ u_3$.
  \item होने देना $ C = \begin{pmatrix} 0 & 1\\ -p & s\end{pmatrix}$
        (द \emph{साथी मैट्रिक्स} का $ D $). बताते हैं कि
        \[
        \begin{pmatrix} u_{n}\\ u_{n+1}\end{pmatrix}
        = C^n \begin{pmatrix} u_0\\ u_1\end{pmatrix}
        \quad (u \in E_D),
        \]
        ओर वो $\operatorname{tr} C = s $ और $ p(C) = p $: द
        पुनरावृत्ति और मैट्रिक्स \omterm{thm:b1:matrices:conjugation}{समान} होते हैं \omterm{def:b1:poly:def}{बहुपद} $ D $.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Degree three.} होने देना $ D_3 = X^3 -
\alpha X^2 - \beta X - \gamma $ और
\[
C_3 = \begin{pmatrix} 0 & 1 & 0\\ 0 & 0 & 1\\ \gamma & \beta &
\alpha \end{pmatrix}.
\]
\begin{enumerate}[resume]
  \item बताते हैं कि $ D_3(C_3) = 0$. \emph{(गणना करें इमेजिस का
        विहित \omterm{def:b1:vspaces:free}{आधार} की शक्तियों के अंतर्गत वेक्टर $ C_3$: द
        \omterm{def:b1:logic:map}{नक्शा} का $ C_3$ भेजता है $ e_1 \mapsto \dots \mapsto $ ए
        अंतिम पंक्ति द्वारा मजबूर संयोजन।)}
  \item दिखाओ अगर $ D_3$ इसकी तीन अलग-अलग जड़ें हैं $\lambda_1,
        \lambda_2, \lambda_3$, शेष $ R_n $ का $ X^n $
        द्वारा विभाजित $ D_3$ है \emph{लैग्रेंज इंटरपोलेंट} का
        मूल्य $\lambda_i^n $ नोड्स पर $\lambda_i $
        (\cref{thm:b1:poly:lagrange}); प्रत्येक प्रविष्टि का निष्कर्ष निकालें
        का $ C_3^{\,n}$ का एक निश्चित रैखिक संयोजन है
        $\lambda_1^n, \lambda_2^n, \lambda_3^n $.
  \item हल करना: $ u_{n+3} = 2u_{n+2} + u_{n+1} - 2u_n $ साथ $ u_0 =
        0$, $ u_1 = 1$, $ u_2 = 1$. \emph{(कारक $ D_3 = (X - 1)(X
        + 1)(X - 2)$.)} पर जाँचा $ u_3$.
  \item के शेष की गणना करें $ X^n $ सापेक्ष $(X -
        \lambda)^3$ \emph{(टेलर का विस्तार $ X^n $ पर
        $\lambda $)}, और इसके लिए एक सूत्र निकालें $(\lambda I +
        N)^n $ कब $ N^3 = 0$ और $ N $ हर चीज़ के साथ आवागमन करता है
        दृष्टि; इसे द्विपद प्रमेय से जांचें।
  \item उसके लिए दिखाओ $ D_3$ विशिष्ट जड़ों के साथ, सामान्य
        आदेश का समाधान-$3$ पुनरावृत्ति है $ u_n = c_1
        \lambda_1^n + c_2\lambda_2^n + c_3\lambda_3^n $: साबित करो
        कि तीन ज्यामितीय अनुक्रम एक बनाते हैं \omterm{def:b1:vspaces:free}{आधार} का
        समाधान स्थान. \emph{(स्वतंत्रता के लिए, शून्य का मूल्यांकन करें
        पर संयोजन $ n = 0, 1, 2$ और एक को पहचानो
        अलग-अलग नोड्स पर इंटरपोलेशन प्रणाली
        $\lambda_i $.)}
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Fibonacci dividends, and synthesis.}
\begin{enumerate}[resume]
  \item सिद्ध करना $ F_1 + F_2 + \dots + F_n = F_{n+2} - 1$.
  \item से $ F^{m+n} = F^m F^n $, जोड़ सूत्र प्राप्त करें
        \[
        F_{m+n} = F_{m+1}F_n + F_m F_{n-1},
        \]
        और निष्कर्ष निकालें $ F_{2n} = F_n(F_{n+1} + F_{n-1})$.
  \item साबित करें कि $ F_n $ के निकटतम पूर्णांक है
        $\varphi^n/\sqrt5$ हरएक के लिए $ n \geq 0$.
  \item होने देना $ t_n = \operatorname{tr}(F^n) = F_{n+1} + F_{n-1}$
        (द \emph{लुकास संख्या} $ L_n $). दिखाओ $ t_{n+2} =
        t_{n+1} + t_n $, $ t_1 = 1$, $ t_2 = 3$, वह $ L_n =
        \varphi^n + \psi^n $, और ठीक हो जाओ $ F_{2n} = F_n L_n $.
  \item संश्लेषण, चार वाक्यों में: ए की शक्तियाँ क्यों $2
        \times 2$ मैट्रिक्स विमान में रहते हैं
        $\operatorname{Vect}(I, A)$ का $\mathcal{M}_2(K)$ (जो
        आयाम तर्क एक द्विघात पहचान की गारंटी देता है, और
        भाग II ने कौन सी स्पष्ट पहचान प्रस्तुत की); कैसे यूक्लिडियन
        विभाजन घातांक को दो-पदों में परिवर्तित करता है
        पुनरावृत्ति; कौन \omterm{def:b1:logic:statement}{कथन} इस समस्या का है $ n =
        2$ सभी आयामों में मान्य एक प्रमेय का मामला (इसे नाम दें,
        और कहिये इसमें कहाँ सिद्ध है? \omterm{def:b1:series:def}{शृंखला}); और क्या
        साथी मैट्रिक्स निर्माण चित्र में जोड़ता है।
\end{enumerate}
\end{problem}
