\chapter{वास्तविक रेखा की टोपोलॉजी}\label{ch:b1:topology}

सीमाएँ समान ज्यामितीय शब्दावली का संदर्भ देती रहती हैं: बिंदु
``करीब'' ए \omterm{def:b1:logic:sets}{तय करना}, सेट ``बिना \omterm{def:b1:topology:closure}{सीमा} लीक'', \omterm{prop:b1:reals:intervals}{अंतराल} में
कौन से अनुक्रम बच नहीं सकते. यह अध्याय उस शब्दावली को ठीक करता है ---
\omterm{def:b1:topology:open}{खुला} और \omterm{def:b1:topology:closure}{बंद} सेट, \omterm{def:b1:topology:closure}{आंतरिक भाग} और \omterm{def:b1:topology:closure}{बंद}, घनत्व ---असली पर
रेखा, और उसके अनुक्रमिक रूप में खंडों की सघनता को सिद्ध करती है।
मानक वेक्टर स्थानों में समान धारणाएँ, दूसरे वर्ष की सामग्री हैं;
पर $\R $ वे पहुंच के भीतर हैं और तुरंत उपयोगी हैं
\cref{ch:b1:continuity}.

\section{खुले सेट, बंद सेट}

\begin{definition}[पड़ोस, खुला सेट]\label{def:b1:topology:open}
ए \omterm{def:b1:logic:sets}{तय करना} $ V \subseteq \R $ एक है \emph{अड़ोस-पड़ोस}\index{अड़ोस-पड़ोस}
का $ x \in \R $ जब इसमें एक शामिल हो \omterm{prop:b1:reals:intervals}{अंतराल} $\intoo{x - r}{x + r}$ के लिए
कुछ $ r > 0$. ए \omterm{def:b1:logic:sets}{तय करना} $ U \subseteq \R $ है \emph{खुला}\index{खुला}
जब यह इसके प्रत्येक बिंदु का पड़ोस हो:
\[
\forall x \in U,\ \exists r > 0, \quad \intoo{x - r}{x + r}
\subseteq U .
\]
\end{definition}

\begin{example}\label{ex:b1:topology:open}
\omterm{def:b1:topology:open}{खुला} \omterm{prop:b1:reals:intervals}{अंतराल} हैं \omterm{def:b1:topology:open}{खुला}: के लिए $ x \in \intoo{a}{b}$, लेना $ r = \min(x -
a,\, b - x) > 0$. आधा लाइनों $\intoo{a}{+\infty}$ हैं \omterm{def:b1:topology:open}{खुला}; $\R $ और
$\emptyset $ हैं \omterm{def:b1:topology:open}{खुला} (उत्तरार्द्ध शून्य रूप से)। $\intcc{0}{1}$ है
\emph{नहीं} \omterm{def:b1:topology:open}{खुला}: नहीं \omterm{prop:b1:reals:intervals}{अंतराल} आस-पास $0$ अंदर रहता है.
\end{example}

\begin{proposition}[खुले सेटों की स्थिरता]\label{prop:b1:topology:openstable}
का कोई भी संघ खुले सेट है \omterm{def:b1:topology:open}{खुला}; ए \emph{परिमित} का चौराहा \omterm{def:b1:topology:open}{खुला}
सेट है \omterm{def:b1:topology:open}{खुला}. अनंत चौराहे विफल हो सकते हैं:
$\bigcap_{n \geq 1} \intoo{-\frac 1n}{\frac 1n} = \{0\}$, नहीं \omterm{def:b1:topology:open}{खुला}.
\end{proposition}

\begin{proof}
संघ: यदि $ x \in \bigcup_i U_i $, तब $ x \in U_{i_0}$ कुछ के लिए $ i_0$,
और यह \omterm{prop:b1:reals:intervals}{अंतराल} द्वारा उपलब्ध कराया गया $ U_{i_0}$ यूनियन के अंदर बैठता है. परिमित
चौराहा: यदि $ x \in U_1 \cap \dots \cap U_k $, लेना $ r = \min(r_1,
\dots, r_k) > 0$ प्रत्येक द्वारा प्रदान की गई त्रिज्या की $ U_j $. के लिए
प्रतिउदाहरण: कोई भी \omterm{prop:b1:reals:intervals}{अंतराल} आस-पास $0$ कुछ शामिल है $\frac{1}{n}$
(आर्किमिडीज़), इसलिए चौराहा छोड़ देता है।
\end{proof}

\begin{example}[स्पष्ट त्रिज्या के साथ खुलेपन को प्रमाणित करना]\label{ex:b1:topology:radii}
है $ U = \{x \in \R : x^2 > 2\}$ \omterm{def:b1:topology:open}{खुला}? हाँ, और प्रमाणपत्र
नीचे लिखा जा सकता है: $ U = \intoo{-\infty}{-\sqrt2} \cup
\intoo{\sqrt2}{+\infty}$, दो का मिलन \omterm{def:b1:topology:open}{खुला} आधी पंक्तियाँ, \omterm{def:b1:topology:open}{खुला}
द्वारा \cref{prop:b1:topology:openstable}. वैकल्पिक रूप से, बहस करें
बिंदु दर बिंदु: के लिए $ x \in U $ साथ $ x > \sqrt 2$, लेना $ r = x -
\sqrt2 > 0$: प्रत्येक $ y \in \intoo{x - r}{x + r}$ संतुष्ट $ y >
\sqrt 2$, इस तरह $ y^2 > 2$; बाईं ओर सममित रूप से. दोनों
शैलियाँ मायने रखती हैं --- संरचनात्मक एक (ज्ञात से निर्माण)। \omterm{def:b1:topology:open}{खुला}
सेट संघों और परिमित प्रतिच्छेदनों द्वारा) तराजू बेहतर,
$\varepsilon $-स्टाइल तब काम करता है जब कोई संरचना दिखाई नहीं देती; और
\cref{ch:b1:continuity} एक तीसरा जोड़ देगा, सबसे शक्तिशाली:
$ U $ है \omterm{def:b1:logic:map}{पूर्वछवि} की \omterm{def:b1:topology:open}{खुला} $\intoo{2}{+\infty}$ के अंतर्गत
निरंतर $ x \mapsto x^2$.
\end{example}

\begin{definition}[बंद सेट]\label{def:b1:topology:closed}
ए \omterm{def:b1:logic:sets}{तय करना} $ F \subseteq \R $ है \emph{बंद किया हुआ}\index{बंद किया हुआ} जब यह
पूरक $\R \setminus F $ है \omterm{def:b1:topology:open}{खुला}. डी मॉर्गन द्वारा और
\cref{prop:b1:topology:openstable}: कोई भी चौराहा \omterm{def:b1:topology:closure}{बंद} सेट
\omterm{def:b1:topology:closure}{बंद} है, \omterm{def:b1:topology:closure}{बंद} की परिमित यूनियनें सेट \omterm{def:b1:topology:closure}{बंद} हो जाती हैं।
\end{definition}

\begin{theorem}[बंद सेटों का अनुक्रमिक लक्षण वर्णन]\label{thm:b1:topology:seqclosed}
$ F $ है \omterm{def:b1:topology:closed}{बंद किया हुआ} यदि और केवल यदि: प्रत्येक अनुक्रम के लिए $(u_n)$ के अंक
$ F $ कुछ में परिवर्तित होना $\ell \in \R $, \omterm{def:b1:topology:closure}{सीमा} $\ell $ से संबंधित
$ F $. (``\omterm{def:b1:topology:closed}{बंद किया हुआ}'' $=$ ``\omterm{def:b1:topology:closure}{सीमा} के अंतर्गत स्थिर''।)
\end{theorem}

\begin{proof}
($\Rightarrow $) होने देना $ F $ होना \omterm{def:b1:topology:closed}{बंद किया हुआ}, $ u_n \in F $, $ u_n \to \ell $, और
मान लीजिये $\ell \notin F $. पूरक है \omterm{def:b1:topology:open}{खुला}: कुछ
$\intoo{\ell - r}{\ell + r}$ बचा जाता है $ F $. लेकिन अभिसरण डालता है $ u_n $
के कारण से \omterm{prop:b1:reals:intervals}{अंतराल} बड़े के लिए $ n $: के साथ विरोधाभास $ u_n \in F $.

($\Leftarrow $) कल्पना करना $ F $ क्या नहीं है \omterm{def:b1:topology:closed}{बंद किया हुआ}: पूरक नहीं है \omterm{def:b1:topology:open}{खुला},
इतना कुछ $ x \notin F $ है कोई \omterm{prop:b1:reals:intervals}{अंतराल} $\intoo{x - r}{x + r}$ अंदर
पूरक; ले रहा हूँ $ r = \frac{1}{n+1}$, चुनना $ u_n \in F $ साथ
$\abs{u_n - x} < \frac{1}{n+1}$. तब $ u_n \in F $, $ u_n \to x \notin
F $: अनुक्रमिक संपत्ति विफल हो जाती है।
\end{proof}

\begin{example}[अनुक्रमिक परीक्षण, दोनों तरीकों से]\label{ex:b1:topology:seqtest}
\emph{\omterm{def:b1:topology:closed}{बंद किया हुआ}:} $ F = \Z \cup \bigl\{n + \frac1n : n \geq 2\bigr\}$.
होने देना $ u_k \in F $ साथ $ u_k \to \ell $. खिड़की $\intcc{\ell -
1}{\ell + 1}$ इसमें केवल सीमित रूप से कई बिंदु शामिल हैं $ F $ (अंततः
अनेक पूर्णांक, परिमित रूप से अनेक $ n + \frac1n $), और कुछ रैंक से परे
सब $ u_k $ इसमें झूठ बोलें: यह क्रम फिर बहुत सारे लेता है
मूल्य, और, अभिसरण, अंततः स्थिर है (जैसा कि)
\cref{exo:b1:topology:3}): $\ell \in F $. \omterm{def:b1:topology:closed}{बंद किया हुआ} --- चाहे
$ F $ दूरी पर बिंदुओं के जोड़े शामिल हैं $\frac1n $, मनमाने ढंग से
\omterm{def:b1:topology:closure}{बंद} करें.

\emph{नहीं \omterm{def:b1:topology:closed}{बंद किया हुआ}:} $ G = \bigl\{\frac1m + \frac1n : m, n \in
\N^*\bigr\}$. क्रम $\frac1n + \frac1n \in G $ आदत है
$0$, और $0 \notin G $ (दो सकारात्मक शब्दों का योग): द
अनुक्रमिक परीक्षण विफल, $ G $ क्या नहीं है \omterm{def:b1:topology:closed}{बंद किया हुआ}. दिलचस्प बात यह है कि प्रत्येक
$\frac1m $ \emph{करता है} के संबंधित $\overline G \cap G $: वास्तव में
$\frac1m = \frac{1}{m+1} + \frac{1}{m(m+1)} \in G $. समापन
अंतर्दृष्टि: बंदता साबित करने के लिए, नियंत्रण \emph{सभी} अभिसारी
एक ही बार में अनुक्रम (आमतौर पर स्थानीय परिमितता या ए के माध्यम से)।
बंद-सूत्र तर्क); इसका खंडन करने के लिए, एक अच्छा चुना हुआ भाग रहा है
अनुक्रम पर्याप्त है --- विषमता नकारात्मक दिशा बनाती है
आसान एक, और इस अध्याय के सभी प्रति उदाहरण हैं
यह एक-पंक्ति आकार.
\end{example}

\begin{example}\label{ex:b1:topology:closed}
सेगमेंट $\intcc{a}{b}$, आधी पंक्तियाँ $\intco{a}{+\infty}$, परिमित
सेट, $\Z $ (पूर्णांकों का अभिसरण क्रम अंततः स्थिर होता है)
हैं \omterm{def:b1:topology:closed}{बंद किया हुआ}. $\intoc{0}{1}$ नहीं है \omterm{def:b1:topology:open}{खुला} (पर विफल रहता है $1$) और न \omterm{def:b1:topology:closed}{बंद किया हुआ}
($\frac 1n \to 0 \notin $  \omterm{def:b1:logic:sets}{तय करना}): अधिकांश सेट दोनों ही नहीं हैं. $\R $ और
$\emptyset $ दोनों \omterm{def:b1:topology:open}{खुला} और \omterm{def:b1:topology:closed}{बंद किया हुआ} --- और वे एकमात्र ऐसे हैं
के उपसमुच्चय $\R $ (\cref{exo:b1:topology:9}).
\end{example}

\begin{example}[अनगिनत टुकड़ों से इकट्ठा किया गया एक खुला सेट]\label{ex:b1:topology:rminusz}
$\R \setminus \Z = \bigcup_{n \in \Z} \intoo{n}{n+1}$: एक
का अनंत मिलन \omterm{def:b1:topology:open}{खुला} \omterm{prop:b1:reals:intervals}{अंतराल}, \omterm{def:b1:topology:open}{खुला} द्वारा
\cref{prop:b1:topology:openstable} --- इसलिए $\Z $ है \omterm{def:b1:topology:closed}{बंद किया हुआ} नहीं के साथ
अनुक्रमिक तर्क की आवश्यकता. श्रम के विभाजन पर ध्यान दें
स्थिरता नियम: \emph{यूनियन} का खुले सेट मनमाना हो सकता है
(प्रत्येक बिंदु को केवल अपने स्वयं के प्रमाणपत्र की आवश्यकता होती है, जो उसके द्वारा प्रदान किया जाता है
\omterm{def:b1:logic:sets}{तय करना} इसमें शामिल है), जबकि \emph{चौराहों} परिमित रहना चाहिए
(प्रमाणपत्रों को प्रतिच्छेद किया जाना चाहिए, और अनंत रूप से कई त्रिज्याएँ हो सकती हैं
कुछ भी नहीं सिकुड़ना)। \Cref{exo:b1:topology:10} ये दिखाएंगे
उदाहरण सामान्य आकार है: प्रत्येक \omterm{def:b1:topology:open}{खुला} का उपसमुच्चय $\R $ एक है
का गणनीय असंयुक्त संघ \omterm{def:b1:topology:open}{खुला} \omterm{prop:b1:reals:intervals}{अंतराल}.
\end{example}

\section{आंतरिक, समापन, घनत्व}

\begin{definition}[आंतरिक, समापन, सीमा]\label{def:b1:topology:closure}
होने देना $ A \subseteq \R $.
\begin{itemize}
  \item एक बिंदु $ x $ है \emph{आंतरिक भाग} को $ A $ कब $ A $ एक है
        \omterm{def:b1:topology:open}{अड़ोस-पड़ोस} का $ x $; \emph{आंतरिक भाग}
        $\mathring{A}$\index{आंतरिक भाग} है \omterm{def:b1:logic:sets}{तय करना} आंतरिक बिंदुओं का.
  \item एक बिंदु $ x $ है \emph{पक्षपाती} को $ A $ जब हर \omterm{def:b1:topology:open}{अड़ोस-पड़ोस}
        का $ x $ की बैठक $ A $; \emph{बंद}
        $\overline{A}$\index{बंद} है \omterm{def:b1:logic:sets}{तय करना} अनुवर्ती बिंदुओं का.
  \item  \emph{सीमा} है $\partial A = \overline A \setminus
        \mathring A $.\index{सीमा}
\end{itemize}
तब $\mathring A \subseteq A \subseteq \overline A $.
\end{definition}

\begin{proposition}[मुख्य गुण]\label{prop:b1:topology:closureprops}
\begin{enumerate}
  \item $\mathring A $ सबसे बड़ा है \omterm{def:b1:topology:open}{खुला} सेट में निहित $ A $; $ A $
        है \omterm{def:b1:topology:open}{खुला} आईएफएफ $ A = \mathring A $.
  \item $\overline A $ सबसे छोटा है \omterm{def:b1:topology:closure}{बंद} सेट युक्त $ A $; $ A $
        है \omterm{def:b1:topology:closed}{बंद किया हुआ} आईएफएफ $ A = \overline A $.
  \item (अनुपालन का अनुक्रमिक लक्षण वर्णन) $ x \in \overline
        A $ अगर और केवल अगर $ x $ के बिंदुओं के अनुक्रम की \omterm{def:b1:topology:closure}{सीमा} है
        $ A $.
  \item कार्यान्वयन विचारों का आदान-प्रदान करता है: $\R \setminus
        \overline A = \bigl(\R \setminus A\bigr)^{\!\circ}$.
\end{enumerate}
\end{proposition}

\begin{proof}
(4) $ x \notin \overline A $ $\iff $ कुछ \omterm{def:b1:topology:open}{अड़ोस-पड़ोस} का $ x $ बचा जाता है $ A $
$\iff $ कुछ \omterm{prop:b1:reals:intervals}{अंतराल} आस-पास $ x $ में निहित है $\R \setminus A $ $\iff $ $ x $
है \omterm{def:b1:topology:closure}{आंतरिक भाग} को $\R \setminus A $.

(1) $\mathring A $ है \omterm{def:b1:topology:open}{खुला}: अगर $ x \in \mathring A $, कुछ
$\intoo{x-r}{x+r} \subseteq A $; हर बिंदु $ y $ उसमें से \omterm{prop:b1:reals:intervals}{अंतराल} है
एक छोटा \omterm{prop:b1:reals:intervals}{अंतराल} इसके चारों ओर इसके अंदर, इसलिए अंदर $ A $: संपूर्ण
\omterm{prop:b1:reals:intervals}{अंतराल} में है $\mathring A $. कोई \omterm{def:b1:topology:open}{खुला} $ U \subseteq A $ के होते हैं
\omterm{def:b1:topology:closure}{आंतरिक भाग} के अंक $ A $, इसलिए $ U \subseteq \mathring A $: सबसे बड़ा. द
खुलेपन का लक्षण वर्णन इस प्रकार है।

(2) विस्तार से. द्वारा (4), $\R \setminus \overline A $ है \omterm{def:b1:topology:closure}{आंतरिक भाग}
का $\R \setminus A $, एक \omterm{def:b1:topology:open}{खुला} सेट द्वारा (1): तो $\overline A $ है
\omterm{def:b1:topology:closed}{बंद किया हुआ}, और इसमें शामिल है $ A $. न्यूनतमता: चलो $ F \supseteq A $ होना
\omterm{def:b1:topology:closed}{बंद किया हुआ}. तब $\R \setminus F $ है \omterm{def:b1:topology:open}{खुला} और इसमें समाहित है $\R
\setminus A $, इसलिए (1) में अधिकतमता से,
\[
\R \setminus F \subseteq \bigl(\R \setminus A\bigr)^{\!\circ}
= \R \setminus \overline A ,
\]
और फिर से पूरक लेना: $\overline A \subseteq F $. इस प्रकार
$\overline A $ सबसे छोटा है \omterm{def:b1:topology:closed}{बंद किया हुआ} सुपरसेट विशेषता:
यदि $ A = \overline A $ तब $ A $ है \omterm{def:b1:topology:closed}{बंद किया हुआ} (अभी दिखाया गया); अगर $ A $ है
\omterm{def:b1:topology:closed}{बंद किया हुआ}, यह स्वयं एक है \omterm{def:b1:topology:closed}{बंद किया हुआ} का सुपरसेट $ A $, इतनी न्यूनता
बल $\overline A \subseteq A $, और समानता।

(3) यदि $ u_n \in A $, $ u_n \to x $: प्रत्येक \omterm{def:b1:topology:open}{अड़ोस-पड़ोस} का $ x $ शामिल है
कुछ $ u_n \in A $, इसलिए $ x \in \overline A $. इसके विपरीत, यदि $ x \in
\overline A $: प्रत्येक \omterm{prop:b1:reals:intervals}{अंतराल} $\intoo{x - \frac{1}{n+1}}{x +
\frac{1}{n+1}}$ की बैठक $ A $ कुछ $ u_n $, और $ u_n \to x $.
\end{proof}

\begin{example}\label{ex:b1:topology:closureexamples}
$\overline{\intoo{0}{1}} = \intcc{0}{1}$;
$\mathring{\intcc{0}{1}} = \intoo{0}{1}$; $\partial\intoo{0}{1} =
\{0, 1\}$. के लिए $ A = \{\frac 1n : n \in \N^*\}$: $\overline A = A
\cup \{0\}$, $\mathring A = \emptyset $, $\partial A = A \cup \{0\}$.
के लिए $\Q $: द्वारा घनत्व (\cref{thm:b1:reals:density}) प्रत्येक वास्तविक है
का अनुयायी $\Q $, इसलिए $\overline{\Q} = \R $ जबकि $\mathring{\Q} =
\emptyset $ (प्रत्येक \omterm{prop:b1:reals:intervals}{अंतराल} इसमें अपरिमेयताएं शामिल हैं): द \omterm{def:b1:topology:closure}{सीमा} का
$\Q $ सब कुछ है $\R $.
\end{example}

\begin{example}[एक पूर्ण शरीर रचना]\label{ex:b1:topology:anatomy}
होने देना $ A = \intoc{0}{1} \,\cup\, \bigl(\Q \cap \intoo{2}{3}\bigr)
\,\cup\, \{4\}$. हम तीनों की गणना करते हैं सेट का
\cref{def:b1:topology:closure}, टुकड़ा दर टुकड़ा।

\emph{\omterm{def:b1:topology:closure}{आंतरिक भाग}.} का एक बिंदु $\intoo{0}{1}$ एक संपूर्ण है \omterm{prop:b1:reals:intervals}{अंतराल}
अंदर $ A $: \omterm{def:b1:topology:closure}{आंतरिक भाग}. बात $1$: प्रत्येक \omterm{prop:b1:reals:intervals}{अंतराल} इसके चारों ओर
का अधिकार लीक $1$, कहाँ $ A $ तक कुछ नहीं है $2$: नहीं
\omterm{def:b1:topology:closure}{आंतरिक भाग}. का कोई मतलब नहीं $\Q \cap \intoo{2}{3}$ है \omterm{def:b1:topology:closure}{आंतरिक भाग} (प्रत्येक
\omterm{prop:b1:reals:intervals}{अंतराल} इसमें तर्कहीन बातें शामिल हैं, \cref{thm:b1:reals:density});
न ही अलग-थलग है $4$. इसलिए $\mathring A = \intoo{0}{1}$.

\emph{समापन.} के अंक की \omterm{def:b1:topology:closure}{सीमा} $ A $: के सभी $\intcc{0}{1}$
($0 = \lim \frac1n $ साथ $\frac 1n \in A $); के सभी
$\intcc{2}{3}$ (प्रत्येक वास्तविक की तर्कसंगतता की एक \omterm{def:b1:topology:closure}{सीमा} होती है
\omterm{prop:b1:reals:intervals}{अंतराल}, घनत्व दोबारा); और $4$. और कुछ नहीं: एक बिंदु
बाहर $\intcc{0}{1} \cup \intcc{2}{3} \cup \{4\}$ सकारात्मक है
उससे दूरी \omterm{def:b1:topology:closure}{बंद} सेट. इसलिए $\overline A = \intcc{0}{1} \cup
\intcc{2}{3} \cup \{4\}$.

\emph{\omterm{def:b1:topology:closure}{सीमा}.} $\partial A = \overline A \setminus \mathring A
= \{0, 1\} \cup \intcc{2}{3} \cup \{4\}$.

समापन अंतर्दृष्टि: तीन ऑपरेशन अधिनियम \emph{स्थानीय रूप से} ---
का प्रत्येक टुकड़ा $ A $ अपनी प्रकृति के अनुसार योगदान देता है (a
ठोस \omterm{prop:b1:reals:intervals}{अंतराल} अपने अंदर रखता है, एक घना-लेकिन-छिद्रपूर्ण टुकड़ा बदल जाता है
पूरी तरह से में \omterm{def:b1:topology:closure}{सीमा}, एक पृथक बिंदु शुद्ध है \omterm{def:b1:topology:closure}{सीमा}), और
का दो-पंक्ति चित्रण $ A $ किसी भी प्रमाण से पहले प्रत्येक उत्तर की भविष्यवाणी करता है
लिखा है.
\end{example}

\begin{remark}[बिंदु-निर्धारित तर्क में सामान्य ख़तरे]
(मैं) \emph{``नहीं \omterm{def:b1:topology:open}{खुला}'' का मतलब `` नहीं है\omterm{def:b1:topology:closed}{बंद किया हुआ}''}: अधिकांश सेट हैं
न तो ($\intoc{0}{1}$), और दो सेट दोनों
($\emptyset $, $\R $) --- \omterm{def:b1:topology:open}{खुला} और \omterm{def:b1:topology:closed}{बंद किया हुआ} विपरीत नहीं हैं लेकिन
पूरकता के माध्यम से दोहरे। (ii) \emph{\omterm{def:b1:topology:closure}{आंतरिक भाग} और \omterm{def:b1:topology:closure}{बंद}
आवागमन न करें}: के लिए $ A = \Q $,
\[
\overline{\mathring A} = \overline\emptyset = \emptyset
\qquad\text{while}\qquad
\bigl(\,\overline A\,\bigr)^{\!\circ} = \mathring \R = \R :
\]
दोनों पुनरावृत्त ऑपरेटर उतने ही भिन्न हैं सेट कर सकना। (iii)
\emph{के अनंत संघ \omterm{def:b1:topology:closure}{बंद} सेट होने में असफल हो सकता है \omterm{def:b1:topology:closed}{बंद किया हुआ}}:
$\bigcup_{n\geq1} \intcc{\frac1n}{1} = \intoc{0}{1}$ --- द
प्रतिच्छेदन का दर्पण प्रतिउदाहरण
\cref{prop:b1:topology:openstable}. (iv) \emph{\omterm{def:b1:topology:dense}{घना} नहीं करता
मतलब बड़ा}: $\Q $ है \omterm{def:b1:topology:dense}{घना}, गणनीय, खाली के साथ \omterm{def:b1:topology:closure}{आंतरिक भाग}, और
इसका पूरक है \omterm{def:b1:topology:dense}{घना} बहुत; घनत्व कहते हैं ``मनमाने ढंग से करीब
सब कुछ'', न कि ``लगभग सब कुछ'' --- सप्ताहांत की समस्या
\omterm{pb:b1:topology:1}{कैंटर सेट} (\cref{pb:b1:topology:1}) विपरीत बिंदु बनाता है,
टोपोलॉजिकल रूप से छोटा \omterm{def:b1:logic:sets}{तय करना} वह बेशुमार बड़ा है.
\end{remark}

\begin{example}[एक समापन की सटीक गणना की गई]\label{ex:b1:topology:closurecomputed}
होने देना $ G = \bigl\{\frac1m + \frac1n : m, n \in \N^*\bigr\}$ (से
\cref{ex:b1:topology:seqtest}). दावा करना:
\[
\overline G = G \,\cup\, \Bigl\{\frac1m : m \in \N^*\Bigr\}
\,\cup\, \{0\} .
\]
($\supseteq $) $\frac1m = \lim_n \bigl(\frac1m + \frac1n\bigr)$
और $0 = \lim_n \frac2n $: अनुक्रम से अनुवर्ती
लक्षण वर्णन ($\subseteq $) होने देना $ x = \lim_k \bigl(
\frac{1}{m_k} + \frac{1}{n_k}\bigr)$; प्रत्येक जोड़ी को इस प्रकार ऑर्डर करें
$ m_k \leq n_k $. अगर $(m_k)$ \omterm{def:b1:reals:bounds}{असीम}ित है, ए \omterm{def:b1:seq:subsequence}{परिणाम को} है $ m_k
\to \infty $, इस तरह $ n_k \to \infty $ भी और $ x = 0$. अन्यथा
$(m_k)$ मान लीजिए, उनमें से एक, सीमित रूप से कई मान लेता है $ m $,
अनंत बार; उसके साथ \omterm{def:b1:seq:subsequence}{परिणाम को} $\frac{1}{n_k} \to x -
\frac1m $: अगर $(n_k)$ परिबद्ध है तो इसमें कुछ मूल्य लगता है $ n $
अनंत बार और $ x = \frac1m + \frac1n \in G $; यदि नहीं, $ x =
\frac1m $. हर मामला घोषित में उतरता है \omterm{def:b1:logic:sets}{तय करना}. समापन
अंतर्दृष्टि: कंप्यूटिंग ए \omterm{def:b1:topology:closure}{बंद} एक कॉम्पैक्टनेस-शैली का मामला है
सूचकांकों पर विश्लेषण --- परिबद्ध सूचकांक का अर्थ है सीमित रूप से अनेक
मान (पिजनहोल), अनबाउंड इंडेक्स का मतलब है कि एक \omterm{def:b1:topology:closure}{सीमा} बच जाती है ---
और उत्तर की विशिष्ट दो-परत संरचना प्रदर्शित करता है
\omterm{def:b1:topology:closure}{सीमा} बिंदु: \omterm{def:b1:logic:sets}{तय करना}, इसकी पहली पीढ़ी की सीमाएँ, और उनकी
\omterm{def:b1:topology:closure}{सीमा} $0$.
\end{example}

\begin{definition}[घनत्व, स्थलाकृतिक रूप]\label{def:b1:topology:dense}
$ A $ है \emph{घना}\index{घना} में $\R $ कब $\overline A = \R $
--- समान रूप से, प्रत्येक गैर-रिक्त \omterm{def:b1:topology:open}{खुला} \omterm{prop:b1:reals:intervals}{अंतराल} की बैठक $ A $;
समकक्ष रूप से (द्वारा \cref{prop:b1:topology:closureprops}~(3)), प्रत्येक
वास्तविक के तत्वों की एक \omterm{def:b1:topology:closure}{सीमा} है $ A $. उदाहरण: $\Q $, $\R \setminus
\Q $, डायडिक्स (\cref{exo:b1:reals:8}), घना \omterm{def:b1:structures:subgroup}{उपसमूह}ों
(\cref{exo:b1:reals:9}).
\end{definition}

\begin{example}[घनत्व सापेक्ष है]\label{ex:b1:topology:relativedensity}
``\omterm{def:b1:topology:dense}{घना}''जैसा कि यहां परिभाषित किया गया है इसका अर्थ है \omterm{def:b1:topology:dense}{घना} \emph{में $\R $}; ए \omterm{def:b1:logic:sets}{तय करना} कर सकते हैं
इसके बजाय हो \omterm{def:b1:topology:dense}{घना} केवल पंक्ति के एक भाग में. के डायडिक्स
$\intcc{0}{1}$, यानी\ $ D \cap \intcc{0}{1}$
(\cref{exo:b1:reals:8}), प्रत्येक से मिलें \omterm{def:b1:topology:open}{खुला} \omterm{prop:b1:reals:intervals}{अंतराल} सम्मिलित
$\intcc{0}{1}$ लेकिन निश्चित रूप से चूक गए $\intoo{2}{3}$ पूरी तरह से: वे
हैं \omterm{def:b1:topology:dense}{घना} \emph{में} $\intcc{0}{1}$, अर्थ $\overline{D \cap
\intcc{0}{1}} = \intcc{0}{1}$. सामान्य वाक्यांश ``$ A $ है \omterm{def:b1:topology:dense}{घना}
में $ B $'' संक्षिप्तीकरण $ B \subseteq \overline A $ --- हमेशा नाम
परिवेश \omterm{def:b1:logic:sets}{तय करना}, चूँकि सप्ताहांत समस्या के अंतिम बिंदु हैं \omterm{def:b1:topology:dense}{घना}
\emph{में \omterm{pb:b1:topology:1}{कैंटर सेट}} जबकि कहीं नहीं है \omterm{def:b1:topology:dense}{घना} \emph{में
$\R $}: जो उसी \omterm{def:b1:logic:sets}{तय करना}, दो सत्य और विपरीत ध्वनि वाले
विवरण.
\end{example}

\begin{example}[घनत्व संभालना]\label{ex:b1:topology:densitytricks}
तीन त्वरित चालें जो लगातार दोहराई जाती हैं। \emph{एनलार्जिंग}: अगर
$ A $ है \omterm{def:b1:topology:dense}{घना} और $ A \subseteq B $, तब $ B $ है \omterm{def:b1:topology:dense}{घना} (प्रत्येक
\omterm{prop:b1:reals:intervals}{अंतराल} पहले से ही मिलता है $ A $). \emph{परिवहन}: अगर $ A $ है
\omterm{def:b1:topology:dense}{घना}, ऐसा ही है $\lambda A + \mu $ के लिए $\lambda \neq 0$ --- एक
\omterm{prop:b1:reals:intervals}{अंतराल} $ I $ की बैठक $\lambda A + \mu $ यदि \omterm{prop:b1:reals:intervals}{अंतराल}
$\frac{I - \mu}{\lambda}$ की बैठक $ A $; इस प्रकार के विषम गुणज
$10^{-9}$, कहते हैं, हैं \omterm{def:b1:topology:dense}{घना}. \emph{अंतर्विभाजक विफल रहता है}: दो \omterm{def:b1:topology:dense}{घना}
सेट एक दूसरे को पूरी तरह से मिस कर सकते हैं ($\Q $ और $\R \setminus \Q $):
घनत्व मिलन और स्नेह से बचे रहते हैं एमएपीएस, कभी भी चौराहे नहीं।
\end{example}

\section{खंडों की सघनता}

\begin{theorem}[खंड क्रमिक रूप से सघन हैं]\label{thm:b1:topology:compact}
होने देना $ a \leq b $. के बिंदुओं का हर क्रम $\intcc{a}{b}$ एक
\omterm{def:b1:seq:subsequence}{परिणाम को} एक बिंदु पर एकत्रित होना \emph{का}
$\intcc{a}{b}$.\index{सघनता}

अधिक सामान्यतः, के उपसमुच्चय $\R $ इस संपत्ति के साथ (प्रत्येक
अनुक्रम में एक है \omterm{def:b1:seq:subsequence}{परिणाम को} में एकत्रित होना \omterm{def:b1:logic:sets}{तय करना}) बिलकुल हैं
\emph{\omterm{def:b1:topology:closed}{बंद किया हुआ} और घिरा हुआ} सेट.
\end{theorem}

\begin{proof}
में एक क्रम $\intcc{a}{b}$ घिरा हुआ है, इसलिए बोल्ज़ानो--वीयरस्ट्रैस
(\cref{thm:b1:seq:bw}) एक अभिसरण निकालता है \omterm{def:b1:seq:subsequence}{परिणाम को}; इसकी \omterm{def:b1:topology:closure}{सीमा}
में रहता है $\intcc{a}{b}$ क्योंकि खंड हैं \omterm{def:b1:topology:closed}{बंद किया हुआ}
(\cref{thm:b1:topology:seqclosed}).

सामान्य मामला. \emph{(\omterm{def:b1:topology:closed}{बंद किया हुआ} घिरे $\Rightarrow $ सघन)}: होने देना
$ F $ होना \omterm{def:b1:topology:closed}{बंद किया हुआ} और घिरा हुआ, और $(u_n)$ में एक क्रम $ F $.
की \omterm{def:b1:topology:closure}{सीमा} $ F $ अनुक्रम को सीमित करता है, इसलिए बोलजानो--वीयरस्ट्रैस
अर्क $ u_{\varphi(n)} \to \ell $; और $\ell \in F $ क्योंकि $ F $
है \omterm{def:b1:topology:closed}{बंद किया हुआ} और यह \omterm{def:b1:seq:subsequence}{परिणाम को} बिंदुओं का एक अभिसारी क्रम है
का $ F $ (\cref{thm:b1:topology:seqclosed}): दो परिकल्पनाएँ
एक-एक का उपभोग किया जाता है, \omterm{def:b1:topology:closure}{सीमा} के अस्तित्व के लिए बाध्यता,
इसकी सदस्यता के लिए बंदता। \emph{(कॉम्पैक्ट $\Rightarrow $ \omterm{def:b1:topology:closed}{बंद किया हुआ} और
घिरा हुआ)}: अगर $ F $ \omterm{def:b1:reals:bounds}{असीम}ित है, उठाओ $ u_n \in F $ साथ $\abs{u_n} \geq
n $; प्रत्येक \omterm{def:b1:seq:subsequence}{परिणाम को} \omterm{def:b1:reals:bounds}{असीम}ित है, इसलिए भिन्न है
(\cref{prop:b1:seq:first}): कोई अभिसरण नहीं \omterm{def:b1:seq:subsequence}{परिणाम को} बिल्कुल भी। अगर $ F $
क्या नहीं है \omterm{def:b1:topology:closed}{बंद किया हुआ}, लेना $ u_n \in F $ साथ $ u_n \to \ell \notin F $
(\cref{thm:b1:topology:seqclosed}): प्रत्येक \omterm{def:b1:seq:subsequence}{परिणाम को} में एकत्रित हो जाता है
$\ell \notin F $, तो नहीं \omterm{def:b1:seq:subsequence}{परिणाम को} अभिसरण \emph{में} $ F $.
\end{proof}

\begin{example}[नेस्टेड कॉम्पैक्ट सेट]\label{ex:b1:topology:nested}
प्रमेय के लिए पहली कसरत. होने देना $ K_0 \supseteq K_1 \supseteq
K_2 \supseteq \dots $ गैर-रिक्त कॉम्पैक्ट हो (\omterm{def:b1:topology:closed}{बंद किया हुआ} परिबद्ध) उपसमुच्चय
का $\R $. तब $\bigcap_n K_n \neq \emptyset $. वास्तव में, उठाओ $ x_n
\in K_n $ प्रत्येक के लिए $ n $: अनुक्रम कॉम्पैक्ट में रहता है $ K_0$,
तो ए \omterm{def:b1:seq:subsequence}{परिणाम को} $ x_{\varphi(n)}$ कुछ में परिवर्तित हो जाता है $ x $
(\cref{thm:b1:topology:compact}). प्रत्येक निश्चित के लिए $ m $, शर्तें
$ x_{\varphi(n)}$ साथ $\varphi(n) \geq m $ सब झूठ बोलते हैं \omterm{def:b1:topology:closed}{बंद किया हुआ}
\omterm{def:b1:logic:sets}{तय करना} $ K_m $, तो \omterm{def:b1:topology:closure}{सीमा} $ x $ में निहित है $ K_m $
(\cref{thm:b1:topology:seqclosed}); जैसा $ m $ मनमाना था, $ x \in
\bigcap_n K_n $. एक उपयोगी साथी: यदि एक \omterm{def:b1:topology:open}{खुला} सेट $ U $ रोकना
$\bigcap_n K_n $, तब $ U \supseteq K_n $ कुछ के लिए $ n $ --- लागू करें
बिंदुओं के लिए वही तर्क $ x_n \in K_n \setminus U $; \omterm{def:b1:topology:closure}{सीमा}
$ x $ में पड़ा रहेगा $\bigcap K_n \subseteq U $, अभी तक $ U $ \omterm{def:b1:topology:open}{खुला} बल
$ x_{\varphi(n)} \in U $ अंततः, एक विरोधाभास. दोनों
\omterm{def:b1:logic:statement}{कथन} सघनता के बिना विफल: $\bigcap_n \intoo{0}{\frac
1n} = \emptyset $ और $\bigcap_n \intco{n}{+\infty} = \emptyset $.
समापन अंतर्दृष्टि: सघनता एक अनंत श्रृंखला को परिवर्तित करती है
एक \omterm{def:b1:topology:closure}{सीमा} बिंदु में गैर-शून्यता का दावा --- यह है
उपकरण जो अनंत चौराहे तक जाने से बचता है, और
सप्ताहांत समस्या (\cref{pb:b1:topology:1}) इस पर दो बार झुकेंगे।
\end{example}

\begin{remark}
यह चरम मूल्य प्रमेय के पीछे का इंजन है
(\cref{ch:b1:continuity}) और एकसमान निरंतरता पर हेन का प्रमेय।
``कॉम्पैक्ट'' नाम इसकी सामान्य (कवरिंग) परिभाषा प्राप्त कर लेगा
दूसरे वर्ष में; पर $\R $, अनुक्रमिक सघनता ही वह सब कुछ है जिसकी हमें आवश्यकता है,
और ``कॉम्पैक्ट $=$ \omterm{def:b1:topology:closed}{बंद किया हुआ} $+$ परिबद्ध'' है \omterm{def:b1:logic:statement}{कथन} याद करने के लिए।
\end{remark}

\begin{example}[यूनियनों के अंतर्गत सीमाएँ]\label{ex:b1:topology:boundaryunion}
हमेशा $\partial(A \cup B) \subseteq \partial A \cup \partial
B $: का एक बिंदु $\partial(A \cup B)$ प्रत्येक के पास है \omterm{def:b1:topology:open}{अड़ोस-पड़ोस}
बैठक $ A \cup B $ (इस तरह $ A $ या $ B $, \omterm{def:b1:reals:bounds}{असीम} रूप से अक्सर में से एक
उन्हें) और के पूरक को पूरा करना $ A \cup B $, जिसमें निहित है
दोनों पूरक हैं --- एक संक्षिप्त जांच फिर बिंदु को अंदर रखती है
$\partial A $ या $\partial B $. समावेशन शानदार हो सकता है
सख्त: साथ $ A = \Q $ और $ B = \R\setminus\Q $,
\[
\partial(A \cup B) = \partial \R = \emptyset ,
\qquad
\partial A \cup \partial B = \R \cup \R = \R :
\]
दो फटे हुए सेट उनकी सीमाओं को एक निर्बाध रूप में जोड़ा जा सकता है
एक दूसरे को नष्ट करना. समापन अंतर्दृष्टि: आंतरिक सज्जा और
\omterm{def:b1:topology:closure}{बंद} व्यवहार \emph{होगा-} यूनियनों के तहत और
चौराहे, लेकिन सीमाएँ --- व्यवहार नहीं करतीं $\partial $ एक के रूप में
व्युत्पन्न मात्रा ($\overline A \setminus \mathring A $), कभी नहीं
एक संचालिका जिसके पास स्वयं का बीजगणित है।
\end{example}

\begin{remark}[इस खंड के अंदर परिप्रेक्ष्य]
यहां निर्मित शब्दावली का इस पुस्तक में दोगुने से अधिक उपयोग किया गया है।
में \cref{ch:b1:continuity}, प्रत्येक प्रमेय एक टोपोलॉजी है
\omterm{def:b1:logic:statement}{कथन} भेस में: मध्यवर्ती मूल्य प्रमेय कहता है
निरंतर एमएपीएस को सुरक्षित रखें \omterm{prop:b1:reals:intervals}{अंतराल} संपत्ति, चरम
मूल्य प्रमेय कहता है कि वे सघनता --- और प्रमाणों को संरक्षित करते हैं
बुलाओ \cref{thm:b1:topology:seqclosed,thm:b1:topology:compact}
नाम से. में \cref{ch:b1:multivar}, वही परिभाषाएँ हैं
में पुनः पढ़ें $\R^2$ के स्थान पर डिस्क के साथ \omterm{prop:b1:reals:intervals}{अंतराल}: खुले सेट,
\omterm{def:b1:topology:closure}{बंद} और सघनता शब्द दर शब्द स्थानांतरण, और
दो-चर चरम मान प्रमेय फिर से लागू होता है
बोलजानो--वीयरस्ट्रैस (प्रत्येक निर्देशांक पर उद्धरण)। एक
धारणा जो करती है \emph{नहीं} दर्द रहित तरीके से सामान्यीकरण करना है
\omterm{prop:b1:reals:intervals}{अंतराल} स्वयं --- समतल में, जुड़ाव प्रतिस्थापित हो जाता है
उत्तलता, एक कहानी जो शुरू होती है
\cref{exo:b1:topology:9}'केवल' है $\emptyset $ और $\R $ हैं \omterm{def:b1:topology:open}{खुला}
और \omterm{def:b1:topology:closed}{बंद किया हुआ}''.
\end{remark}

\section{अभ्यास}

\begin{exercise}[$\star $]\label{exo:b1:topology:1}
प्रत्येक के लिए \omterm{def:b1:logic:sets}{तय करना}, कहो क्या यह है \omterm{def:b1:topology:open}{खुला}, \omterm{def:b1:topology:closed}{बंद किया हुआ}, दोनों, या न ही (साथ)
औचित्य): $\intoo{0}{1} \cup \intoo{2}{3}$; $\;\intco{0}{1}$;
$\;\{0\} \cup \intcc{1}{2}$; $\;\R \setminus \Z $; $\;\Q \cap
\intoo{0}{1}$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:topology:2}
ठानना $\mathring A $, $\overline A $ और $\partial A $ के लिए:
$ A = \intoc{0}{1} \cup \{2\}$; $\;A = \R \setminus \Q $;
$\;A = \bigl\{\frac{(-1)^n n}{n+1} : n \in \N\bigr\}$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:topology:3}
साबित करें कि ए परिमित सेट है \omterm{def:b1:topology:closed}{बंद किया हुआ}, पहले पूरक के माध्यम से, फिर इसके माध्यम से
अनुक्रमिक लक्षण वर्णन.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:topology:4}
किसी के लिए भी इसे साबित करें $ A, B \subseteq \R $: $\overline{A \cup B} =
\overline A \cup \overline B $. उदाहरण देकर दिखाओ
$\overline{A \cap B}$ से भिन्न हो सकता है $\overline A \cap \overline
B $.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:topology:5}
होने देना $ u_n \to \ell $ में $\R $. सिद्ध करें कि \omterm{def:b1:logic:sets}{तय करना} $\{u_n : n \in \N\}
\cup \{\ell\}$ है \omterm{def:b1:topology:closed}{बंद किया हुआ} (इसलिए अगर हम इसे जोड़ दें तो कॉम्पैक्ट है
परिबद्ध --- जो यह है)।
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:topology:6}
होने देना $ U $ होना \omterm{def:b1:topology:open}{खुला} और $ A $ मनमाना। साबित करें कि $ U + A = \{u + a\}$ है
\omterm{def:b1:topology:open}{खुला}. यह निष्कर्ष निकालें कि a का योग है \omterm{def:b1:topology:open}{खुला} सेट और कोई भी \omterm{def:b1:logic:sets}{तय करना} है \omterm{def:b1:topology:open}{खुला}, और
कंट्रास्ट: दो प्रदर्शित करें \emph{\omterm{def:b1:topology:closed}{बंद किया हुआ}} सेट जिसका योग नहीं है \omterm{def:b1:topology:closed}{बंद किया हुआ}.
\emph{(कोशिश $\Z $ और $\sqrt 2\,\Z $, साथ \cref{exo:b1:reals:9}.)}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:topology:7}
एक बिंदु $ x \in A $ है \emph{एकाकी} में $ A $ जब कुछ \omterm{def:b1:topology:open}{अड़ोस-पड़ोस} का
$ x $ की बैठक $ A $ केवल पर $ x $. के प्रत्येक बिन्दु को सिद्ध करें $\Z $ पृथक है
में $\Z $, वह $ A = \{\frac1n\}$ इसके सभी बिंदुओं को अभी तक अलग कर दिया गया है
$\overline A \neq A $, और वह ए \omterm{def:b1:logic:sets}{तय करना} जिसके सभी बिंदु पृथक हैं
खाली है \omterm{def:b1:topology:closure}{आंतरिक भाग}.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:topology:8}
सिद्ध करें कि \omterm{def:b1:topology:closure}{बंद} एक सीमाबद्ध की \omterm{def:b1:logic:sets}{तय करना} घिरा हुआ है, और वह
$\sup \overline A = \sup A $ के लिए $ A $ ऊपर गैर-रिक्त सीमाबद्ध। निष्कर्ष निकालें
वह $\sup A \in \overline A $: द \omterm{def:b1:reals:bounds}{सर्वोच्च} सदैव अनुवर्ती है.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b1:topology:9}
सिद्ध करें कि इसका एकमात्र उपसमुच्चय है $\R $ वह दोनों हैं \omterm{def:b1:topology:open}{खुला} और \omterm{def:b1:topology:closed}{बंद किया हुआ} हैं
$\emptyset $ और $\R $. \emph{संकेत: मान लीजिये $ A $ है \omterm{def:b1:topology:open}{खुला}, \omterm{def:b1:topology:closed}{बंद किया हुआ}, साथ
$ A \neq \emptyset $ और $\R \setminus A \neq \emptyset $; चुनना $ a \in
A $, $ b \notin A $, कहना $ a < b $, और विचार करें $ s = \sup\,(A \cap
\intcc{a}{b})$; तय करो की $ s $ का हो सकता है $ A $ या इसके लिए
पूरक.}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b1:topology:10}
(की संरचना खुले सेट) होने देना $ U \subseteq \R $ होना \omterm{def:b1:topology:open}{खुला}, गैररिक्त. के लिए
$ x \in U $, होने देना $ I_x $ सबका मिलन हो \omterm{def:b1:topology:open}{खुला} \omterm{prop:b1:reals:intervals}{अंतराल} युक्त
$ x $ और इसमें समाहित है $ U $. साबित करें कि $ I_x $ एक \omterm{def:b1:topology:open}{खुला} \omterm{prop:b1:reals:intervals}{अंतराल}, वह
दो सेट $ I_x $, $ I_y $ समान या असंयुक्त हैं, और वह $ U $ एक संघ है
का \emph{अनगिनत} जोड़ीवार विसंयुक्त \omterm{def:b1:topology:open}{खुला} \omterm{prop:b1:reals:intervals}{अंतराल} \emph{(चुनें
प्रत्येक में एक तर्कसंगत)}.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:topology:11}
एक बिंदु $ x \in \R $ एक \emph{संचय बिंदु} का $ A $ कब
प्रत्येक \omterm{def:b1:topology:open}{अड़ोस-पड़ोस} का $ x $ की बैठक $ A \setminus \{x\}$; उनका \omterm{def:b1:logic:sets}{तय करना} है
द \emph{व्युत्पन्न \omterm{def:b1:logic:sets}{तय करना}} $ A'$. साबित करें कि $\overline A = A \cup A'$,
ओर वो $ A $ है \omterm{def:b1:topology:closed}{बंद किया हुआ} अगर और केवल अगर $ A' \subseteq A $. ठानना
$ A'$ के लिए $ A = \{\frac 1n : n \in \N^*\}$, के लिए $ A = \Z $, और के लिए
$ A = \Q $.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b1:topology:12}
के लिए $ A \subseteq \R $ गैर-रिक्त, परिभाषित करें $ d_A(x) = \inf\{\abs{x - a}
: a \in A\}$. सिद्ध करना:
\begin{enumerate}
  \item $\abs{d_A(x) - d_A(y)} \leq \abs{x - y}$ सभी के लिए $ x, y $
        ($ d_A $ है $1$-लिप्सचिट्ज़);
  \item $ d_A(x) = 0$ अगर और केवल अगर $ x \in \overline A $; में
        विशेष रूप से, यदि $ F $ है \omterm{def:b1:topology:closed}{बंद किया हुआ} और $ x \notin F $, तब
        $ d_F(x) > 0$;
  \item हरएक के लिए $\varepsilon > 0$, द \omterm{def:b1:logic:sets}{तय करना} $ V_\varepsilon =
        \{x : d_F(x) < \varepsilon\}$ है \omterm{def:b1:topology:open}{खुला}, रोकना $ F $, और
        $\bigcap_{\varepsilon > 0} V_\varepsilon = F $ के लिए $ F $
        \omterm{def:b1:topology:closed}{बंद किया हुआ}: प्रत्येक \omterm{def:b1:topology:closure}{बंद} सेट का एक गणनीय प्रतिच्छेदन है
        खुले सेट.
\end{enumerate}
\end{exercise}

\section{समस्या: कैंटर सेट, एक साथ छोटा और विशाल}

\begin{problem}[{सप्ताहांत समस्या --- कैंटर मध्य-तिहाई \omterm{def:b1:logic:sets}{तय करना}: लंबाई शून्य, बेशुमार, उत्तम, और $ C + C = \intcc{0}{2}$}]
\label{pb:b1:topology:1}
से हटाने $\intcc{0}{1}$ इसका \omterm{def:b1:topology:open}{खुला} बीच का तीसरा, फिर बीच का
प्रत्येक शेष खंड का तीसरा, और हमेशा के लिए दोहराएँ: क्या
जीवित रहता है \emph{\omterm{pb:b1:topology:1}{कैंटर सेट}} $ C $, मौलिक
विश्लेषण का प्रति उदाहरण कारखाना। यह समस्या इसका निर्माण करती है,
इसे आधार के माध्यम से पढ़ता है-$3$ की मशीनरी \cref{pb:b1:reals:1},
और अपना विरोधाभासी चित्र स्थापित करता है: कुल लंबाई शून्य, फिर भी
बेशुमार; खाली \omterm{def:b1:topology:closure}{आंतरिक भाग}, फिर भी कोई पृथक बिंदु नहीं; पूरी तरह से
अभी भी डिस्कनेक्ट किया गया है $ C + C $ पूरे खंड को भर देता है
$\intcc{0}{2}$.\index{कैंटर सेट} औपचारिक रूप से: $ C_0 = \intcc{0}{1}$,
और $ C_{n+1}$ से प्राप्त किया जाता है $ C_n $ को हटाकर \omterm{def:b1:topology:open}{खुला} मध्य
के प्रत्येक खंड का तीसरा $ C_n $; अंत में $ C = \bigcap_{n \geq 0}
C_n $. पूरे, ए \emph{टर्नरी कोड} का $ x \in \intcc{0}{1}$
कोई अंक स्ट्रिंग है $(d_k)_{k\geq1}$ साथ $ d_k \in \{0, 1, 2\}$
जिसका मूल्य $\sup_n \sum_{k=1}^n d_k 3^{-k}$ के बराबर होती है $ x $ ---
अनुचित कोड (आखिरकार $2$) अनुमति दी जाती है; द्वारा
\cref{pb:b1:reals:1} (प्रश्न 9-11), प्रत्येक $ x \in
\intcc{0}{1}$ एक या दो कोड होते हैं, दो बिल्कुल कब $ x = m/3^N
\in \intoo{0}{1}$.

\medskip
\noindent\textbf{Part I --- The construction.}
\begin{enumerate}
  \item वर्णन करना $ C_1$ और $ C_2$ स्पष्ट रूप से यूनियनों के रूप में
        खंड, और प्रेरण द्वारा सिद्ध करें: $ C_n $ एक असंयुक्त है
        का संघ $2^n $ \omterm{def:b1:topology:closed}{बंद किया हुआ} खंड, प्रत्येक की लंबाई $3^{-n}$.
  \item बताते हैं कि $ C $ है \omterm{def:b1:topology:closed}{बंद किया हुआ}, परिबद्ध --- अत: सघन
        (\cref{thm:b1:topology:compact}) --- गैर-रिक्त, और वह
        प्रत्येक के प्रत्येक खंड का प्रत्येक समापन बिंदु $ C_n $ से संबंधित $ C $.
  \item की कुल लम्बाई $ C_n $ है $\bigl(\frac23\bigr)^n $.
        प्रत्येक के लिए यह निष्कर्ष निकालें $\varepsilon > 0$, द \omterm{def:b1:logic:sets}{तय करना} $ C $ कर सकते हैं
        कुल लंबाई के बहुत से खंडों द्वारा कवर किया जाना चाहिए
        $\leq \varepsilon $: द \omterm{pb:b1:topology:1}{कैंटर सेट} है \emph{लंबाई
        शून्य}.
  \item दिखाओ कि एक \omterm{prop:b1:reals:intervals}{अंतराल} में निहित $ C $ लंबाई है $\leq
        3^{-n}$ हरएक के लिए $ n $, इसलिए एक सिंगलटन या खाली है:
        $\mathring C = \emptyset $. प्राणी \omterm{def:b1:topology:closed}{बंद किया हुआ} खाली के साथ
        \omterm{def:b1:topology:closure}{आंतरिक भाग}, $ C $ है \emph{कहीं नहीं \omterm{def:b1:topology:dense}{घना}}.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The ternary code.}
\begin{enumerate}[resume]
  \item स्व-समानता पुनरावर्तन सिद्ध करें
        \[
        C_{n+1} = \tfrac13 C_n \,\cup\,
        \bigl(\tfrac23 + \tfrac13 C_n\bigr),
        \qquad\text{hence}\qquad
        C = \tfrac13 C \,\cup\, \bigl(\tfrac23 + \tfrac13
        C\bigr),
        \]
        दो टुकड़े अलग-अलग हो रहे हैं: $ C $ की दो प्रतियाँ हैं
        खुद पैमाने पर $\frac13$.
  \item प्रेरण द्वारा सिद्ध करें $ n $: $ x \in C_n $ अगर और केवल अगर
        $ x $ एक टर्नरी कोड है जिसका पहला $ n $ अंक झूठ बोलते हैं
        $\{0, 2\}$. इस तथ्य का उपयोग करके निष्कर्ष निकालें $ x $ अधिकतम है
        दो कोड: $ x \in C $ अगर और केवल अगर $ x $ के साथ एक कोड है
        \emph{के बराबर कोई अंक नहीं $1$} (ए \emph{$1$-मुक्त} कोड).
  \item कार्रवाई में कोड: दे $1$-के लिए निःशुल्क कोड $0$, $1$,
        $\frac13$, $\frac23$; दिखाओ $\frac14 = (0.\overline{02})_3$
        और $\frac34 = (0.\overline{20})_3$, तो दोनों के हैं
        $ C $; और उसे जांचें $\frac14$ है \emph{नहीं} एक समापन बिंदु
        किसी का $ C_n $ (एंडपॉइंट्स का फॉर्म है $ m/3^n $).
  \item प्रत्येक को दिखाओ $ x \in C $ है \emph{बिल्कुल एक} $1$-मुक्त
        कोड \emph{(कब $ x $ इसके दो कोड हैं, इसे सटीक रूप से सिद्ध करें
        जोड़ी में से एक में अंक शामिल है $1$)}. निष्कर्ष:
        मूल्य \omterm{def:b1:logic:map}{नक्शा} से एक आक्षेप है $\{0,2\}$-स्ट्रिंग्स पर
        $ C $.
  \item (विकर्ण) चलो $ k \mapsto x_k $ कोई भी हो \omterm{def:b1:logic:map}{नक्शा} $\N^* \to C $.
        एक निर्माण करें $\{0, 2\}$-स्ट्रिंग सूचकांक पर भिन्न है $ k $ से
        का कोड $ x_k $, और निष्कर्ष निकालो $ C $ बेशुमार है
        --- जबकि, इसके विपरीत, प्रश्न 3 कहता है कि यह है
        मीट्रिक रूप से नगण्य.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Topological portrait.}
\begin{enumerate}[resume]
  \item अब तक का रिकॉर्ड इकट्ठा करें: $ C $ सघन है, बेशुमार है,
        लंबाई शून्य की, कहीं नहीं \omterm{def:b1:topology:dense}{घना}. कौन सा एकल रोकथाम
        $ C \subseteq C_n $ प्रत्येक संपत्ति वहन करती है?
  \item ($ C $ एकदम सही है) चलो $ x \in C $ साथ $1$-निःशुल्क कोड
        $(d_k)$. अंक पलटना $ d_n $ ($0 \leftrightarrow
        2$) पैदा करता है $ x_n \in C $ साथ $\abs{x_n - x} = 2 \cdot
        3^{-n}$. समाप्त करें कि $ C $ कोई पृथक बिंदु नहीं है: प्रत्येक
        का बिंदु $ C $ की एक \omterm{def:b1:topology:closure}{सीमा} है \emph{अन्य} के अंक $ C $.
  \item दिखाएँ कि प्रश्न 2 के अंतिम बिंदु एक गणनीय बनाते हैं
        \omterm{def:b1:topology:dense}{घना} का उपसमुच्चय $ C $ \emph{(बाद में कोड को छोटा करें $ n $
        अंक और जारी रखें $0$ एस; गणनीयता जैसे कि
        \cref{exo:b1:topology:10})}. निष्कर्ष: विशिष्ट बिंदु
        का $C $ --- पसंद $\frac14$ --- है \emph{नहीं} एक समापन बिंदु:
        समापनबिंदु एक बेशुमार के अंदर एक गणनीय कंकाल हैं
        शरीर.
  \item (पूरी तरह से डिस्कनेक्ट) चलो $ x < y $ में $ C $. चुनना $ n $
        साथ $3^{-n} < y - x $ और एक बिंदु उत्पन्न करें $ z \in
        \intoo{x}{y}$ साथ $ z \notin C $. निष्कर्ष निकालें कि एकमात्र
        गैररिक्त \omterm{prop:b1:reals:intervals}{अंतराल} में निहित $ C $ सिंगलटन हैं.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Arithmetic of $ C $.}
\begin{enumerate}[resume]
  \item दिखाओ $1 - C = C $ \emph{(क्या करता है $ x \mapsto 1 - x $ करो
        एक को $1$-मुफ़्त कोड? याद करना $1 = (0.\overline{2})_3$)}.
  \item (कोड जोड़ना) यदि दिखाएँ $ x $, $ x'$ कोड हैं
        $(a_k)$, $(b_k)$, तब $ x + x' = \lim_n\,(t_n + t'_n)$
        कहाँ $ t_n, t'_n $ आंशिक योग हैं. निष्कर्ष निकालें: प्रत्येक $ y
        \in \intcc{0}{1}$ है \emph{मध्य} दो बिंदुओं का
        का $ C $ --- एक कोड दिया गया $(e_k)$ का $ y $, अंक चुनें
        $ a_k, b_k \in \{0, 2\}$ साथ $\frac{a_k + b_k}{2} =
        e_k $.
  \item निष्कर्ष निकालना $ C + C = \intcc{0}{2}$ और, प्रश्न 14 के साथ,
        $ C - C = \intcc{-1}{1}$. ठोस उदाहरण: लिखें $1$ जैसे
        प्रश्न 7 में पाए गए दो गैर-समाप्ति बिंदुओं का योग।
  \item प्रतिबिंबित करें: ए \omterm{def:b1:logic:sets}{तय करना} लंबाई का शून्य जिसका अंतर \omterm{def:b1:logic:sets}{तय करना} भरण
        $\intcc{-1}{1}$. के बीच कोई विरोधाभास क्यों नहीं है?
        ``$ C $ मीट्रिक रूप से नगण्य है'' और ``$ C + C $ भरा हुआ है
        लम्बाई''? (एक वाक्य; सोचो कितनी लंबाई है
        नियंत्रित करता है और यह क्या नहीं करता है।)
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Members, rational and irrational.}
\begin{enumerate}[resume]
  \item प्रश्न 8 को आवधिकता मानदंड के साथ मिलाएं
        \cref{pb:b1:reals:1}: का एक बिंदु $ C $ तर्कसंगत है यदि और
        केवल अगर यह $1$-मुक्त कोड अंततः आवधिक होता है। भागो
        आधार-$3$ जाँच करने के लिए लंबा विभाजन $\frac1{13} =
        (0.\overline{002})_3 \in C $.
  \item स्पष्ट रूप से उत्पादन करें \emph{तर्कहीन} के सदस्य $ C $:
        कोड का मान $ d_k = 2$ त्रिकोणीय पर
        पद $ k = \frac{j(j+1)}{2}$ और $ d_k = 0$
        अन्यत्र. बढ़ते \omterm{prop:b1:reals:intervals}{अंतराल}ों द्वारा अतार्किकता को उचित ठहराएँ
        का तर्क \cref{pb:b1:reals:1} (प्रश्न 20).
  \item (पूरे खंड पर) विचार करें $ h $ के बिंदु का मानचित्रण
        $ C $ साथ $1$-निःशुल्क कोड $(d_k)$ के मूल्य के लिए
        \emph{द्विआधारी} डोरी $\bigl(\frac{d_k}2\bigr)$, अर्थात\
        $ h(x) = \sup_n \sum_{k=1}^n \frac{d_k}{2}\,2^{-k}$.
        बताते हैं कि $ h $ एमएपीएस $ C $ \emph{पर} $\intcc{0}{1}$. तो
        नगण्य $ C $ पूर्ण के एक खंड पर प्रक्षेपित होता है
        लंबाई --- इसका दूसरा प्रमाण $ C $ बेशुमार है.
  \item (स्वयं समान लंबाई) मान लीजिए लंबाई की कुछ धारणा $ L $
        के लिए परिभाषित किए गए थे $ C $ और इसकी सिकुड़ी हुई प्रतियों का सम्मान करते हुए
        स्केलिंग ($ L(\lambda A) = \lambda L(A)$), अनुवाद
        अपरिवर्तनशीलता, और असंयुक्त अपघटन पर संवेदनशीलता
        प्रश्न 5 का। फिर दिखाओ $ L(C) = \frac23\,L(C)$,
        जबरदस्ती $ L(C) = 0$: पहले से ही अकेले आत्म-समानता
        वाक्य $ C $ लंबाई शून्य तक.
\end{enumerate}

\medskip
\noindent\textbf{Part VI --- A fat cousin, and the moral.}
\begin{enumerate}[resume]
  \item (मोटा \omterm{pb:b1:topology:1}{कैंटर सेट}) निर्माण दोहराएँ, लेकिन स्तर पर
        $ n $ ($ n = 0, 1, 2, \dots $) प्रत्येक से हटाएँ $2^n $
        वर्तमान खंड एक केंद्रीय \omterm{def:b1:topology:open}{खुला} \omterm{prop:b1:reals:intervals}{अंतराल} लम्बाई का
        $4^{-(n+1)}$ केवल। दिखाएँ कि खंड की लंबाई है $ l_n $
        आज्ञा का पालन करना $ l_{n+1} = \frac{l_n - 4^{-(n+1)}}{2}$, $ l_n =
        \frac{2^n + 1}{2\cdot 4^n} > 0$, जो परिणामी है
        $ K = \bigcap K_n $ खाली के साथ सघन है \omterm{def:b1:topology:closure}{आंतरिक भाग}, और
        कि हटाई गई कुल लंबाई है $\sum_{n\geq0} 2^n
        4^{-(n+1)} = \frac12$. स्वीकार करना (सहज ज्ञान, वर्ष 3)
        के परिमित संघों के लिए लंबाई की संवेदनशीलता \omterm{prop:b1:reals:intervals}{अंतराल}, और
        दोनों का उपयोग करना \omterm{def:b1:logic:statement}{कथन} का \cref{ex:b1:topology:nested},
        दिखाएँ कि कोई भी परिमित परिवार \omterm{def:b1:topology:open}{खुला} \omterm{prop:b1:reals:intervals}{अंतराल} कवर
        $ K $ कुल लंबाई है $\geq \frac12$: $ K $ कहीं नहीं है
        \omterm{def:b1:topology:dense}{घना} लेकिन \emph{नहीं} नगण्य. छोटेपन के अनेक होते हैं
        असमान अर्थ.
  \item (दूरियाँ) इसे गैर-रिक्त के लिए दिखाएँ \omterm{def:b1:topology:closed}{बंद किया हुआ} $ F \subseteq
        \R $ और $ x \in \R $, द \omterm{def:b1:reals:bounds}{असीम} $ d(x, F)$ है
        \emph{प्राप्त} \emph{(अनुक्रम को न्यूनतम करना प्लस
        बोल्ज़ानो--वीयरस्ट्रैस)}. फिर गणना करें
        \[
        \max_{y \in \intcc{0}{1}} d(y, C) = \frac16 ,
        \]
        ठीक केंद्र पर पहुंच गया $ y = \frac12$ \emph{(ए
        मंच पर बने \omterm{prop:b1:reals:intervals}{अंतराल} का बिंदु $ n $ भीतर है
        $\frac{3^{-n}}{2}$ \omterm{prop:b1:reals:intervals}{अंतराल} के अंतिम बिंदुओं में से, जो अंदर स्थित हैं
        $ C $)}.
  \item (प्रत्येक बिंदु एक अनुवर्ती \omterm{def:b1:topology:closure}{सीमा} है) प्रश्न 12 का उपयोग करना
        और 2, एक एकल अनुक्रम उत्पन्न करें $ C $ किसका \omterm{def:b1:logic:sets}{तय करना} का
        अनुवर्ती \omterm{def:b1:topology:closure}{सीमा} है \emph{सभी} का $ C $. (तुलना करें:
        एक अभिसरण अनुक्रम के लिए कि \omterm{def:b1:logic:sets}{तय करना} एक बिंदु है ---
        $ C $ कॉम्पैक्ट के बीच विपरीत चरम का एहसास होता है।)
  \item संश्लेषण, प्रत्येक एक वाक्य: (i) इसका कौन सा प्रमेय है
        अध्याय क्या निर्माण वास्तव में उपभोग (स्थिरता) करता है
        का \omterm{def:b1:topology:closure}{बंद} सेट, सघनता, अनुक्रमिक
        लक्षण वर्णन)? (ii) चार विरोधाभासों की सूची बनाएं
        चित्र की जोड़ी (लंबाई शून्य/बेशुमार,
        \omterm{def:b1:topology:closed}{बंद किया हुआ}/खाली \omterm{def:b1:topology:closure}{आंतरिक भाग}, पूर्ण/पूरी तरह से असंबद्ध,
        नगण्य/$ C+C $ भरा हुआ); (iii) कहाँ होता है $ C $ पुनः सतह पर आना
        बाद में (शैतान की सीढ़ी बनी $ h $ सिद्धांत में
        निरंतरता का, और वर्ष का माप सिद्धांत 3
        मात्रा, कहाँ $ C$ ``गणनीय'' को अलग करता है
        ``नगण्य'')?
\end{enumerate}
\end{problem}
