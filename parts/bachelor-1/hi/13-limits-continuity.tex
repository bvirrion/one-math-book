\chapter{सीमाएँ और निरंतरता}\label{ch:b1:continuity}

मध्यवर्ती मूल्य प्रमेय और चरम मूल्य प्रमेय थे
दृश्य विश्वास पर हाई स्कूल स्तर पर उपयोग किया जाता है। अनुक्रमों के साथ
(\cref{ch:b1:seq}) और की टोपोलॉजी $\R $ (\cref{ch:b1:topology})
हाथ में, यह अध्याय उन्हें सिद्ध करता है --- और सिद्धांत को पूरा करता है
मोनोटोन आक्षेप प्रमेय (जो वैध बनाता है $\arcsin $,
$\operatorname{arcosh}$ और उनके परिजन \cref{ch:b1:functions})
और हेन की प्रमेय पर एकसमान निरंतरता.

लगातार, $ I $ एक \omterm{prop:b1:reals:intervals}{अंतराल} और $ f \colon I \to \R $; ``$ x_0 \in
\overline I $'' अंतिम बिंदुओं पर \omterm{def:b1:topology:closure}{सीमा} की अनुमति देता है।

\section{कार्यों की सीमाएँ}

\begin{definition}[एक बिंदु पर सीमित करें]\label{def:b1:continuity:limit}
होने देना $ x_0 \in \overline{I}$ और $\ell \in \R $. तब $ f(x) \to \ell $ जैसा
$ x \to x_0$ कब\index{किसी फ़ंक्शन की सीमा!}
\[
\forall \varepsilon > 0,\ \exists \delta > 0,\ \forall x \in I,
\qquad \abs{x - x_0} \leq \delta \implies \abs{f(x) - \ell} \leq
\varepsilon .
\]
पर \omterm{def:b1:topology:closure}{सीमा} $\pm\infty $ और अनंत सीमाएँ उसी से परिभाषित होती हैं
पैटर्न ($\abs{x - x_0} \leq \delta $ बन जाता है $ x \geq M $; $\abs{f(x) -
\ell} \leq \varepsilon $ बन जाता है $ f(x) \geq M'$). एकतरफ़ा \omterm{def:b1:topology:closure}{सीमा}
प्रतिबंधित $ x $ को $ x > x_0$ (लिखा हुआ $ x \to x_0^+$) या $ x < x_0$. द
\omterm{def:b1:topology:closure}{सीमा} अस्तित्व में होने पर अद्वितीय होती है (समान प्रमाण)।
\cref{prop:b1:seq:first}).
\end{definition}

\begin{example}[निचोड़ के माध्यम से एक सीमा]\label{ex:b1:continuity:floorlimit}
गणना करना $\lim_{x \to 0} x\,\bigl\lfloor \frac1x \bigr\rfloor $.
 \omterm{thm:b1:reals:floor}{ज़मीन} ब्रैकेटिंग $\frac1x - 1 < \lfloor \frac1x \rfloor \leq
\frac1x $ से गुणा करने के बाद देता है $ x $ (संकेत पर ध्यान दें!):
\[
1 - x < x\Bigl\lfloor \frac1x \Bigr\rfloor \leq 1 \quad (x > 0),
\qquad
1 \leq x\Bigl\lfloor \frac1x \Bigr\rfloor < 1 - x \quad (x < 0),
\]
और दोनों एकतरफा निचोड़ \omterm{def:b1:topology:closure}{बंद} हो गए $1$: हद है $1$. नोट
क्या हुआ: $\lfloor \frac1x\rfloor $ अकेले के पास जंगली छलांगें हैं
$0$, लेकिन कारक $ x $ प्रत्येक छलांग को नियंत्रित करता है ($ x $ कई बार एक इकाई की छलांग होती है
छोटा), और केवल ब्रैकेटिंग बची है। समापन अंतर्दृष्टि:
परिबद्ध-दोलन के साथ एक छोटे कारक के उत्पादों की \omterm{def:b1:topology:closure}{सीमा}
कारक निचोड़ समस्याएं हैं, कभी भी ऑपरेशन-प्रमेय समस्याएं नहीं ---
संचालन प्रमेय की आवश्यकता है \emph{दोनों} कारकों का अभिसरण होना।
\end{example}

\begin{theorem}[अनुक्रमिक लक्षण वर्णन]\label{thm:b1:continuity:seqchar}
$ f(x) \to \ell $ जैसा $ x \to x_0$ यदि और केवल यदि: के लिए \emph{प्रत्येक}
अनुक्रम $(u_n)$ के अंक $ I $ साथ $ u_n \to x_0$, एक के पास है
$ f(u_n) \to \ell $.
\end{theorem}

\begin{proof}
($\Rightarrow $) होने देना $ u_n \to x_0$ और $\varepsilon > 0$. लेना
$\delta $ परिभाषा से, तो $ N $ साथ $\abs{u_n - x_0} \leq
\delta $ के लिए $ n \geq N $: आगे $ N $, $\abs{f(u_n) - \ell} \leq
\varepsilon $.

($\Leftarrow $) गर्भनिरोधक। अगर $ f \not\to \ell $: कुछ
$\varepsilon_0 > 0$ हर को हरा देता है $\delta $; का चयन $\delta =
\frac{1}{n+1}$ का उत्पादन $ u_n \in I $ साथ $\abs{u_n - x_0} \leq
\frac{1}{n+1}$ और $\abs{f(u_n) - \ell} > \varepsilon_0$. तब $ u_n
\to x_0$ लेकिन $ f(u_n) \not\to \ell $.
\end{proof}

\begin{corollary}[संचालन, रचना, क्रम]\label{cor:b1:continuity:operations}
\omterm{def:b1:topology:closure}{सीमा} के योग, गुणनफल, भागफल (शून्येतर \omterm{def:b1:topology:closure}{सीमा} नीचे) व्यवहार करते हैं
जहाँ तक अनुक्रमों की बात है; अगर $ f \to \ell $ पर $ x_0$ और $ g \to m $ पर $\ell $,
और $ g $ चारों ओर परिभाषित किया गया है $\ell $ साथ $ g(\ell) = m $ या $ f \neq \ell $
पास में $ x_0$, तब $ g \circ f \to m $ पर $ x_0$; सीमाएँ विस्तृत रहती हैं
असमानताएँ, और निचोड़ प्रमेय कायम है।
\end{corollary}

\begin{proof}
प्रत्येक \omterm{def:b1:logic:statement}{कथन} के माध्यम से स्थानान्तरित होता है
\cref{thm:b1:continuity:seqchar} इसके अनुक्रम एनालॉग के लिए
(\cref{thm:b1:seq:operations,thm:b1:seq:order}). रचना के लिए,
डायरेक्ट चेनिंग एक बार लिखने लायक है: दिया गया $\varepsilon >
0$, की \omterm{def:b1:topology:closure}{सीमा} $ g $ पर $\ell $ प्रदान $\eta > 0$ साथ
\[
\abs{y - \ell} \leq \eta \implies \abs{g(y) - m} \leq
\varepsilon \quad (y \text{ in the domain of } g),
\]
मामला कहां है $ y = \ell $ कवर किया गया है क्योंकि $ g(\ell) = m $; फिर
की \omterm{def:b1:topology:closure}{सीमा} $ f $ पर $ x_0$ प्रदान $\delta > 0$ साथ $\abs{x -
x_0} \leq \delta \implies \abs{f(x) - \ell} \leq \eta $; जंजीर बनाना,
$\abs{x - x_0} \leq \delta $ देता है $\abs{g(f(x)) - m} \leq
\varepsilon $. वैकल्पिक परिकल्पना में ($ f \neq \ell $ पास में
$ x_0$), मूल्य $ y = \ell $ कभी नहीं खिलाया जाता $ g $ और इसका मूल्य
अप्रासंगिक है.
\end{proof}

\begin{example}[रचना परंतुक क्यों मौजूद है?]\label{ex:b1:continuity:compositiontrap}
होने देना $ g(y) = 0$ के लिए $ y \neq 0$ और $ g(0) = 1$, और चलो $ f $ हो
समान रूप से $0$. तब $ f(x) \to 0$ जैसा $ x \to 0$, और $ g(y) \to 0$
जैसा $ y \to 0$; अभी तक $ g(f(x)) = g(0) = 1$ हरएक के लिए $ x $, इसलिए $ g
\circ f \to 1 \neq 0$. आंतरिक कार्य बैठता है \emph{बिलकुल चालू}
निषिद्ध मूल्य $\ell = 0$ हमेशा के लिए और $ g $की \omterm{def:b1:topology:closure}{सीमा} पर है $0$
क्या नजरअंदाज करता है $ g $ पर करता है $0$. का प्रावधान
\cref{cor:b1:continuity:operations} --- दोनों में से एक $ g(\ell) = m $
(अर्थात्\ $ g $ निरंतर पर $\ell $), या $ f \neq \ell $ पास में $ x_0$ ---
वास्तव में यही इसे नियमबद्ध करता है। समापन अंतर्दृष्टि: में
अभ्यास से कोई रचना करता है \emph{निरंतर} कार्य और
परन्तुक निःशुल्क है; यह तभी काटता है जब सीमाएं साथ ली जाती हैं
छिद्रित पड़ोस, यही कारण है कि की परिभाषा
$\lim_{x \to x_0}$ इस पुस्तक में प्रयुक्त वह बिंदु शामिल है जब यह
डोमेन में है.
\end{example}

\section{निरंतरता}

\begin{definition}\label{def:b1:continuity:continuous}
$ f $ है \emph{पर निरंतर $ x_0 \in I $}\index{पर निरंतर $ x_0 \in I $} कब $ f(x)
\to f(x_0)$ जैसा $ x \to x_0$; \emph{निरंतर चालू $ I $} जब यह है
हर बिंदु पर निरंतर. द्वारा
\cref{thm:b1:continuity:seqchar}: $ f $ पर निरंतर है $ x_0$ आईएफएफ
$ f(u_n) \to f(x_0)$ हर अनुक्रम के लिए $ u_n \to x_0$ में $ I $.
\end{definition}

\begin{example}[असंततताओं का एक वर्गीकरण]\label{ex:b1:continuity:taxonomy}
बढ़ती गंभीरता में, एक बिंदु पर असफल होने के तीन तरीके।
\emph{हटाने योग्य}: $ f(x) = \frac{\sin x}{x}$ पर $\R^*$ \omterm{def:b1:topology:closure}{सीमा} है
$1$ पर $0$; परिभाषित $ f(0) = 1$ इसकी मरम्मत करता है --- असंततता
एक छेद था, कोई विशेषता नहीं। \emph{कूदना}: $\lfloor x \rfloor $ पर
एक पूर्णांक की अलग-अलग एकतरफ़ा सीमाएँ होती हैं ($ n - 1$ और $ n $); नहीं
मूल्य का चुनाव उनमें सामंजस्य स्थापित कर सकता है, लेकिन दोनों आधी-सीमाएँ मौजूद हैं।
\emph{आवश्यक}: $\sin\frac1x $ पर $0$ की कोई एकतरफ़ा \omterm{def:b1:topology:closure}{सीमा} नहीं है
सभी (\cref{exo:b1:continuity:1}) ---बिना दोलन
बस्ती. मोनोटोन फ़ंक्शन केवल मध्य प्रकार का उत्पादन कर सकते हैं
(उनकी एकतरफा सीमाएं हमेशा मौजूद रहती हैं, सुप्रीमा और इनफिमा),
यही कारण है कि उनका असंतत होना सेट अधिकतम गणनीय हैं ---
प्रति छलांग एक तर्कसंगत। डेरबौक्स प्रमेय द्वारा व्युत्पन्न
(\cref{exo:b1:derivative:10}), केवल अंतिम प्रकार का उत्पादन कर सकता है:
जंप डिसकंटीनिटी वाला फ़ंक्शन कभी भी व्युत्पन्न नहीं होता है
कुछ भी.
\end{example}

\begin{proposition}\label{prop:b1:continuity:algebra}
योग, उत्पाद, भागफल (जहाँ परिभाषित) और रचनाएँ
निरंतर कार्य हैं निरंतर. \omterm{def:b1:poly:def}{बहुपद}ों, तर्कसंगत भिन्न
(उनके बाहर \omterm{def:b1:fractions:field}{डंडे}), $\abs{\,\cdot\,}$, $\exp $, $\ln $, द
त्रिकोणमितीय और अतिशयोक्तिपूर्ण कार्य और उनके व्युत्क्रम
(\cref{ch:b1:functions}) हैं निरंतर उनके डोमेन पर.
\end{proposition}

\begin{proof}
संचालन: \cref{cor:b1:continuity:operations}. स्थिरांक और
पहचान हैं निरंतर सीधे परिभाषा से
($\delta = \varepsilon $ पहचान की सेवा करता है, कोई भी $\delta $ द
स्थिरांक); के उत्पाद निरंतर कार्य किया जा रहा है निरंतर,
प्रत्येक एकपदी $ a_k x^k $ पर प्रेरण द्वारा अनुसरण करता है $ k $, और रकम
ख़त्म करो बहुआयामी पद; ए \omterm{def:b1:fractions:field}{तर्कसंगत अंश} दो का भागफल है
बहुआयामी पद, निरंतर जहां भी हर नहीं है
गायब हो जाना. $\abs{\,\cdot\,}$: उलटा त्रिकोण असमानता,
$\bigl|\abs{f(x)} - \abs{f(x_0)}\bigr| \leq \abs{f(x) -
f(x_0)}$, तो वही $\delta $ काम करता है. शास्त्रीय के लिए
कार्य जो हम प्रदान करते हैं निरंतरता यहाँ; भिन्नता (में सिद्ध)
\cref{ch:b1:derivative}) अधिक बलवान है।
\end{proof}

\begin{example}[सतत कार्यों का अधिकतम और न्यूनतम]\label{ex:b1:continuity:maxmin}
अगर $ f, g $ हैं निरंतर, तो हैं $\max(f, g)$ और $\min(f, g)$:
किसी मामले के विश्लेषण की आवश्यकता नहीं है, पहचान के लिए धन्यवाद
\[
\max(f, g) = \frac{f + g + \abs{f - g}}{2},
\qquad
\min(f, g) = \frac{f + g - \abs{f - g}}{2},
\]
और यह निरंतरता रकम का और का $\abs{\,\cdot\,}$
(\cref{prop:b1:continuity:algebra}). विशेष रूप से $ f^+ =
\max(f, 0)$ और $ f^- = \max(-f, 0)$ हैं निरंतर साथ $ f =
f^+ - f^-$: श्रृंखला के लिए प्रयुक्त चिह्न-विभाजन
(\cref{ch:b1:series}) और, पूर्ण पैमाने पर, एकीकरण में
वर्ष 3 खंड का सिद्धांत, नियमितता में कुछ भी खर्च नहीं करता है।
\end{example}

\begin{theorem}[मध्यवर्ती मूल्य प्रमेय]\label{thm:b1:continuity:ivt}
होने देना $ f $ होना निरंतर पर $\intcc{a}{b}$ साथ $ f(a) \leq 0 \leq
f(b)$. तब $ f(c) = 0$ कुछ के लिए $ c \in \intcc{a}{b}$.
नतीजतन, ए निरंतर एक पर कार्य करें \omterm{prop:b1:reals:intervals}{अंतराल} हर मान लेता है
इसके किन्हीं दो मानों के बीच: $ f(I)$ एक
\omterm{prop:b1:reals:intervals}{अंतराल}.\index{मध्यवर्ती मूल्य प्रमेय}
\end{theorem}

\begin{proof}
द्विभाजन। \omterm{def:b1:logic:sets}{तय करना} $ a_0 = a $, $ b_0 = b $. दिया गया $\intcc{a_n}{b_n}$ साथ
$ f(a_n) \leq 0 \leq f(b_n)$, होने देना $ m $ मध्यबिंदु बनें: यदि $ f(m) \leq
0$ रखना $\intcc{m}{b_n}$, अन्यथा रखो $\intcc{a_n}{m}$; संकेत
स्थितियाँ बनी रहती हैं. क्रम $(a_n), (b_n)$ आसन्न हैं ($ b_n -
a_n = \frac{b-a}{2^n}$), सामान्य \omterm{def:b1:topology:closure}{सीमा} के साथ $ c $
(\cref{thm:b1:seq:adjacent}). द्वारा निरंतरता और
\cref{thm:b1:seq:order}: $ f(c) = \lim f(a_n) \leq 0$ और $ f(c) =
\lim f(b_n) \geq 0$, इसलिए $ f(c) = 0$.

परिणाम के लिए: दिए गए मान $ f(u) < v < f(w)$, उपरोक्त लागू करें
को $ x \mapsto f(x) - v $ समापन बिंदु वाले खंड पर $ u $ और $ w $;
इस प्रकार $ f(I)$ उत्तल है, अर्थात\ एक \omterm{prop:b1:reals:intervals}{अंतराल}
(\cref{prop:b1:reals:intervals}).
\end{proof}

\begin{example}[एक समीकरण, पूरा प्रोटोकॉल]\label{ex:b1:continuity:protocol}
हल करना $\eu^x = 3 - x $ ऊपर $\R $: अस्तित्व, विशिष्टता,
स्थान. \omterm{def:b1:logic:sets}{तय करना} $ g(x) = \eu^x + x - 3$, निरंतर. स्थान और
अस्तित्व: $ g(0) = -2 < 0$ और $ g(1) = \eu - 2 > 0$, तो
मध्यवर्ती मूल्य प्रमेय एक समाधान प्रस्तुत करता है
$\intoo{0}{1}$. विशिष्टता: $ g $ सख्ती का योग है
बढ़ रहा है $\eu^x $ और $ x - 3$, इसलिए सख्ती से बढ़ रहा है
$\R $; एक सख्ती से मोनोटोन फ़ंक्शन प्रत्येक मान को अधिकतम लेता है
एक बार, इसलिए समाधान सभी पर अद्वितीय है $\R $ (सिर्फ में नहीं
\omterm{prop:b1:reals:intervals}{अंतराल} जांच की गई)। प्रोटोकॉल --- को पुनर्व्यवस्थित करें $ g = 0$, संकेत
अस्तित्व के लिए \omterm{def:b1:counting:objects}{परिवर्तन}, विशिष्टता के लिए एकरसता --- स्थिर हो जाती है
तीन पंक्तियों में अधिकांश ``कितने समाधान'' प्रश्न, और
\cref{ch:b1:derivative}की विविधता सारणी
इसे गैर-मोनोटोन तक विस्तारित करें $ g $ काटने से $\R $ एकस्वर में
शाखाएँ.
\end{example}

\begin{example}[एक एल्गोरिथम के रूप में द्विभाजन]\label{ex:b1:continuity:dichotomy}
का प्रमाण \cref{thm:b1:continuity:ivt} गणना करता है. लेना $ f(x) =
x^3 + x - 1$: $ f(0) = -1 < 0 < 1 = f(1)$, तो एक जड़ निहित है
$\intoo{0}{1}$. आधा करना:
\[
f(0.5) = -0.375 < 0, \qquad
f(0.75) = 0.171875 > 0, \qquad
f(0.625) = -0.130859375 < 0 ,
\]
इसलिए जड़ क्रमिक रूप से फंस जाती है $\intoo{0.5}{1}$, तब
$\intoo{0.5}{0.75}$, तब $\intoo{0.625}{0.75}$ (सही मूल्य:
$ c \approx 0.6823$). बाद $ n $ चरण त्रुटि अधिकतम है
$\frac{b - a}{2^n}$: दस कदम तीन दशमलव देते हैं, बीस कदम देते हैं
छह. समापन अंतर्दृष्टि: मध्यवर्ती मूल्य प्रमेय नहीं है
केवल एक अस्तित्व \omterm{def:b1:logic:statement}{कथन} --- इसका द्विभाजन प्रमाण है
गारंटीकृत, यदि धीमा है, रूट-फाइंडिंग एल्गोरिदम, जिसके विरुद्ध
तेज़ लेकिन स्थानीय न्यूटन विधि \cref{ch:b1:derivative} चाहिए
मापा जाए.
\end{example}

\begin{theorem}[चरम मूल्य प्रमेय]\label{thm:b1:continuity:evt}
ए निरंतर एक खंड पर कार्य करें $\intcc{a}{b}$ घिरा हुआ है और
अपनी \omterm{def:b1:topology:closure}{सीमा} को प्राप्त करता है: वहाँ हैं $ c, d \in \intcc{a}{b}$ साथ
\[
f(c) = \inf_{\intcc{a}{b}} f,
\qquad
f(d) = \sup_{\intcc{a}{b}} f .
\]
के साथ संयुक्त \cref{thm:b1:continuity:ivt}: \emph{ निरंतर
एक खंड की छवि एक खंड है} $\intcc{f(c)}{f(d)}$.
\index{चरम मूल्य प्रमेय}
\end{theorem}

\begin{proof}
\emph{ऊपर परिबद्ध:} अन्यथा चुनें $ u_n $ साथ $ f(u_n) \geq n $. द्वारा
खंड की सघनता (\cref{thm:b1:topology:compact}), ए
\omterm{def:b1:seq:subsequence}{परिणाम को} $ u_{\varphi(n)} \to x \in \intcc{a}{b}$; निरंतरता देता है
$ f(u_{\varphi(n)}) \to f(x)$, लेकिन $ f(u_{\varphi(n)}) \geq \varphi(n)
\to +\infty $: विरोधाभास।

\emph{सुपर प्राप्त:} होने देना $ M = \sup f $ और चुनें $ v_n $ साथ $ f(v_n) >
M - \frac{1}{n+1}$ (\cref{prop:b1:reals:epsilon}). निकालना
$ v_{\varphi(n)} \to d \in \intcc{a}{b}$: तब $ f(d) = \lim
f(v_{\varphi(n)}) = M $ निचोड़ने से. \omterm{def:b1:reals:bounds}{असीम} द्वारा प्रबंधित किया जाता है
$-f $.
\end{proof}

\begin{example}[एक खंड पर सकारात्मक न्यूनतम]\label{ex:b1:continuity:posmin}
होने देना $ f $ होना निरंतर पर $\intcc{0}{1}$ साथ $ f(x) > 0$ हरएक के लिए $ x $. तब $\inf f > 0$: चरम मूल्य प्रमेय द्वारा
\omterm{def:b1:reals:bounds}{असीम} एक है \emph{कीमत} $ f(c)$, और $ f(c) > 0$ परिकल्पना द्वारा.
तो ए निरंतर किसी खंड पर सकारात्मक कार्य सीमित है
से $0$ --- दो पंक्ति का तर्क एक दर्जन बार इस्तेमाल किया गया
आने वाले अध्याय (नियंत्रण में विभाजक, चरण-कार्य)।
फ़्रेमिंग, त्रुटि सीमाएं)। एक गैर-कॉम्पैक्ट पर \omterm{prop:b1:reals:intervals}{अंतराल} यह विफल रहता है
शानदार ढंग से: $ f(x) = x $ पर $\intoc{0}{1}$ है निरंतर और
के साथ सकारात्मक $\inf f = 0$, प्राप्त नहीं हुआ. समापन अंतर्दृष्टि:
``सकारात्मक'' को ``समान रूप से सकारात्मक'' में अपग्रेड किया जाता है, ठीक उसी समय जब
डोमेन सघन है; चरम मूल्य प्रमेय की प्रत्येक परिकल्पना
भार वहन करने वाला है.
\end{example}

\begin{remark}[तीन प्रमेयों से संबंधित सामान्य गलतियाँ]
(मैं) \emph{निरंतर इमेजिस}: केवल \emph{खंडों} मजबूत हैं.
 निरंतर एक की छवि \omterm{def:b1:topology:open}{खुला} \omterm{prop:b1:reals:intervals}{अंतराल} होने की आवश्यकता नहीं है \omterm{def:b1:topology:open}{खुला}
($ x \mapsto x^2$ एमएपीएस $\intoo{-1}{1}$ पर $\intco{0}{1}$), द
एक की छवि \omterm{def:b1:topology:closure}{बंद} सेट होने की आवश्यकता नहीं है \omterm{def:b1:topology:closed}{बंद किया हुआ} ($\arctan $ एमएपीएस 
\omterm{def:b1:topology:closed}{बंद किया हुआ} $\R $ पर \omterm{def:b1:topology:open}{खुला}
$\intoo{-\frac\pi2}{\frac\pi2}$); लेकिन एक खंड की छवि है
एक खंड (\cref{thm:b1:continuity:evt}). (ii) \emph{द
मध्यवर्ती मूल्य प्रमेय की आवश्यकता है \omterm{prop:b1:reals:intervals}{अंतराल}}: समारोह $ x
\mapsto \frac1x $, निरंतर पर $\intco{-1}{0} \cup
\intoc{0}{1}$, मान लेता है $-1$ और $1$ फिर भी कभी नहीं
गायब हो जाता है --- इसका डोमेन दो असंयुक्त टुकड़े और मूल्य है
$0$ \omterm{prop:b1:reals:intervals}{अंतराल} में गिर जाता है; हमेशा नाम दें \omterm{prop:b1:reals:intervals}{अंतराल} जिस पर
प्रमेय लागू किया गया है. (iii) \emph{एकसमान निरंतरता की एक संपत्ति है
जोड़ी (फ़ंक्शन, \omterm{def:b1:logic:sets}{तय करना})}: $ x^2$ है \omterm{def:b1:continuity:uniform}{समान रूप से निरंतर} हर पर
खंड अभी तक चालू नहीं है $\R $
(\cref{ex:b1:continuity:notuniform}) --- शब्द ``\omterm{def:b1:continuity:uniform}{समान रूप से
निरंतर}''डोमेन के बिना अर्थहीन हैं। (iv)
\emph{निरंतरता व्युत्क्रम औपचारिक नहीं है}: यह कायम रहता है
\omterm{prop:b1:reals:intervals}{अंतराल} एकरसता के माध्यम से
(\cref{thm:b1:continuity:bijection}), लेकिन ए निरंतर आपत्ति
के संघों के बीच \omterm{prop:b1:reals:intervals}{अंतराल} एक असंतत व्युत्क्रम हो सकता है ---
उस प्रमेय के बाद की टिप्पणी इसलिए है क्योंकि छात्र इसे उद्धृत करते हैं
के बिना \omterm{prop:b1:reals:intervals}{अंतराल} परिकल्पना।
\end{remark}

\section{मोनोटोन फ़ंक्शन और व्युत्क्रम फ़ंक्शन}

\begin{theorem}[मोनोटोन आक्षेप प्रमेय]\label{thm:b1:continuity:bijection}
होने देना $ f $ होना निरंतर और एक पर सख्ती से एकरस \omterm{prop:b1:reals:intervals}{अंतराल} $ I $. तब
\begin{enumerate}
  \item $ f $ से एक आक्षेप है $ I $ पर \omterm{prop:b1:reals:intervals}{अंतराल} $ J = f(I)$;
  \item उलटा $ f^{-1} \colon J \to I $ सख्ती से मोनोटोनिक (में) है
        एक ही दिशा) और \emph{निरंतर}.
\end{enumerate}
\end{theorem}

\begin{proof}
कहना $ f $ सख्ती से बढ़ रहा है. (1) \omterm{def:b1:logic:inj}{इंजेक्शन} तत्काल से है
सख्त एकरसता; प्रक्षेपात्मकता पर $ f(I)$ तुच्छ है, और $ f(I)$
एक \omterm{prop:b1:reals:intervals}{अंतराल} द्वारा \cref{thm:b1:continuity:ivt}.

(2) $ f^{-1}$ सख्ती से बढ़ रहा है: यदि $ y < y'$ में $ J $ लेकिन $ f^{-1}(y)
\geq f^{-1}(y')$, वृद्धि को लागू करना $ f $ देता है $ y \geq y'$,
बेतुका। निरंतरता का $ f^{-1}$ पर $ y_0 = f(x_0) \in J $: होने देना
$\varepsilon > 0$. पहले मान लीजिए $ x_0$ \omterm{def:b1:topology:closure}{आंतरिक भाग} को $ I $, और
सिकुड़ना $\varepsilon $ ताकि $ x_0 \pm \varepsilon \in I $: उनका
छवियां संतुष्ट करती हैं $ f(x_0 - \varepsilon) < y_0 < f(x_0 +
\varepsilon)$. लेना
$\delta = \min\bigl(y_0 - f(x_0 - \varepsilon),\, f(x_0 +
\varepsilon) - y_0\bigr) > 0$: के लिए $\abs{y - y_0} \leq \delta $,
की एकरसता $ f^{-1}$ निचोड़ कर रख $ f^{-1}(y)$ बीच में $ x_0 -
\varepsilon $ और $ x_0 + \varepsilon $. अगर $ x_0$ मान लीजिए, बाईं ओर है
का समापन बिंदु $ I $, केवल $ x_0 + \varepsilon $ उपलब्ध है: फिर
$ y_0 = \min J $ (एकरसता), प्रत्येक $ y \in J $ साथ $ y - y_0
\leq f(x_0 + \varepsilon) - y_0$ संतुष्ट $ x_0 \leq f^{-1}(y)
\leq x_0 + \varepsilon $, और एकतरफा अनुमान बिल्कुल सही है
निरंतरता एक अंतिम बिंदु पर; सही समापन बिंदु सममित है।
(टिप्पणी: निरंतरता का $ f^{-1}$
था \emph{नहीं} उससे निकाला गया $ f $ समरूपता से --- यह है
एक पर एकरसता \omterm{prop:b1:reals:intervals}{अंतराल} वह काम करता है.)
\end{proof}

\begin{remark}
यह प्रमेय क्या है \cref{def:b1:functions:arc} और
\cref{prop:b1:functions:invhyp} चुपचाप उपयोग किया गया: $\arcsin $, $\arccos $,
$\arctan $, $\operatorname{arsinh}$, \dots\ हैं निरंतर. ए
पूरक (\cref{exo:b1:continuity:10}): ए निरंतर
\emph{\omterm{def:b1:logic:inj}{इंजेक्शन}} एक पर कार्य करें \omterm{prop:b1:reals:intervals}{अंतराल} स्वचालित रूप से सख्ती है
मोनोटोनिक, इसलिए मोनोटोनिटी परिकल्पना की कोई लागत नहीं है।
\end{remark}

\begin{example}[सभी आदेशों की जड़ें]\label{ex:b1:continuity:nthroot}
के लिए $ n \in \N^*$, समारोह $ f(x) = x^n $ है निरंतर और
सख्ती से बढ़ रहा है $\intco{0}{+\infty}$, साथ $ f(0) = 0$ और
$ f(x) \to +\infty $: इसकी छवि है $\intco{0}{+\infty}$
(\cref{thm:b1:continuity:ivt} के लिए \omterm{prop:b1:reals:intervals}{अंतराल} संरचना)। द
मोनोटोन आक्षेप प्रमेय तब, एक झटके में, वितरित करता है
सख्ती से बढ़ रहा है \emph{निरंतर} श्लोक में
\[
x \mapsto x^{1/n} \colon \intco{0}{+\infty} \to
\intco{0}{+\infty} :
\]
अस्तित्व, विशिष्टता और निरंतरता का $ n $-वें जड़ें, नहीं के साथ
गणना. साथ तुलना करें \cref{exo:b1:reals:12}, जो बनाया गया
$\sqrt y $ हाथ से \omterm{def:b1:reals:bounds}{सर्वोच्च}: सिद्धांत का एक अध्याय है
कार्य के उस पृष्ठ को दो पंक्तियों में, और उसी दो पंक्तियों में संपीड़ित किया
पंक्तियों को वैध किया गया $\arcsin $, $\arctan $ और
$\operatorname{arsinh}$ उनके पहले। समापन अंतर्दृष्टि: एक अच्छा
प्रमेय संग्रहीत श्रम है।
\end{example}

\section{एकसमान निरंतरता}

\begin{definition}\label{def:b1:continuity:uniform}
$ f \colon I \to \R $ है \emph{समान रूप से निरंतर}\index{समान रूप से निरंतर} कब
\[
\forall \varepsilon > 0,\ \exists \delta > 0,\ \forall x, y \in I,
\qquad \abs{x - y} \leq \delta \implies \abs{f(x) - f(y)} \leq
\varepsilon .
\]
मुद्दा: $\delta $ पर ही निर्भर करता है $\varepsilon $, पर नहीं
में स्थान $ I $. वर्दी निरंतरता तात्पर्य निरंतरता; ए
\emph{लिप्सचिट्ज़} समारोह ($\abs{f(x) - f(y)} \leq k \abs{x - y}$)
समान रूप से है निरंतर ($\delta = \varepsilon/k $).
\end{definition}

\begin{example}\label{ex:b1:continuity:notuniform}
$ x \mapsto x^2$ है निरंतर पर $\R $ लेकिन \emph{नहीं} \omterm{def:b1:continuity:uniform}{समान रूप से
निरंतर}: $\abs{(n + \frac 1n)^2 - n^2} = 2 + \frac{1}{n^2} \geq 2$
हालाँकि तर्क दूरी पर हैं $\frac 1n \to 0$. किसी पर
घिरा हुआ \omterm{prop:b1:reals:intervals}{अंतराल} इसलिए, यह लिप्सचिट्ज़ है \omterm{def:b1:continuity:uniform}{समान रूप से निरंतर} ---
नीचे हेन के प्रमेय के अनुरूप।
\end{example}

\begin{example}[समान मॉड्यूल, स्पष्ट रूप से]\label{ex:b1:continuity:moduli}
एक खंड पर, हेन एक वर्दी की गारंटी देता है $\delta $; अक्सर कोई भी कर सकता है
भी \emph{गणना} यह। के लिए $ f(x) = x^2$ पर $\intcc{0}{10}$:
\[
\abs{x^2 - y^2} = \abs{x + y}\,\abs{x - y} \leq 20\,\abs{x - y},
\]
इसलिए $\delta = \frac{\varepsilon}{20}$ समान रूप से काम करता है (ए
लिप्सचिट्ज़ \omterm{def:b1:complex:field}{मापांक}, रैखिक में $\varepsilon $). के लिए $\sqrt x $ पर
$\intcc{0}{1}$: $\abs{\sqrt x - \sqrt y} \leq \sqrt{\abs{x -
y}}$ (\cref{exo:b1:continuity:9}), इसलिए $\delta = \varepsilon^2$
काम करता है --- एकसमान लेकिन \emph{नहीं} रैखिक: निकट $0$ वर्ग
जड़ खड़ी है, और कीमत के प्रतिपादक में प्रकट होता है
$\varepsilon $, एकरूपता की विफलता में नहीं. समापन
अंतर्दृष्टि: एकसमान निरंतरता एक स्पेक्ट्रम है, हाँ/नहीं ---
समारोह $\delta(\varepsilon)$, कहा जाता है \omterm{def:b1:complex:field}{मापांक}, उपाय
एकरूपता कितनी महंगी है, और लिप्सचिट्ज़ बस इसकी है
सर्वोत्तम ग्रेड.
\end{example}

\begin{theorem}[हेन]\label{thm:b1:continuity:heine}
ए निरंतर ए पर कार्य करें \emph{खंड} $\intcc{a}{b}$ है \omterm{def:b1:continuity:uniform}{समान रूप से
निरंतर}.\index{हेन का प्रमेय}
\end{theorem}

\begin{proof}
विरोधाभास से: कुछ मान लीजिए $\varepsilon_0 > 0$ हर को हरा देता है
$\delta $. साथ $\delta = \frac{1}{n+1}$, चुनना $ x_n, y_n \in
\intcc{a}{b}$ साथ $\abs{x_n - y_n} \leq \frac{1}{n+1}$ और
$\abs{f(x_n) - f(y_n)} > \varepsilon_0$. सघनता
(\cref{thm:b1:topology:compact}) अर्क $ x_{\varphi(n)} \to c \in
\intcc{a}{b}$; तब $ y_{\varphi(n)} \to c $ भी (निचोड़ें)। $\abs{x -
y}$). निरंतरता पर $ c $ देता है $ f(x_{\varphi(n)}) \to f(c)$ और
$ f(y_{\varphi(n)}) \to f(c)$, इसलिए $\abs{f(x_{\varphi(n)}) -
f(y_{\varphi(n)})} \to 0 < \varepsilon_0$: विरोधाभास।
\end{proof}

\begin{example}[बंधा हुआ, सतत, फिर भी समान रूप से नहीं]\label{ex:b1:continuity:sinx2}
समारोह $ f(x) = \sin(x^2)$ है निरंतर और सीमाबद्ध है
$\R $, लेकिन नहीं \omterm{def:b1:continuity:uniform}{समान रूप से निरंतर}. लेना
\[
x_n = \sqrt{2\pi n}, \qquad y_n = \sqrt{2\pi n + \tfrac\pi2}:
\qquad
y_n - x_n = \frac{\pi/2}{\sqrt{2\pi n + \frac\pi2} +
\sqrt{2\pi n}} \longrightarrow 0 ,
\]
अभी तक $ f(y_n) - f(x_n) = \sin\bigl(2\pi n + \frac\pi2\bigr) -
\sin(2\pi n) = 1 - 0 = 1$ हरएक के लिए $ n $: एक भी नही $\delta $ सेवा कर सकता $\varepsilon = \frac12$ हर जगह. ज्यामितीय रूप से,
के दोलन $\sin(x^2)$ \emph{तेजी लाने}: ग्राफ
छोटी और छोटी खिड़कियों पर एक पूर्ण तरंग पूरी करता है, इसलिए
क्षैतिज पैमाना दिया गया $\varepsilon $ शून्य पर सिकुड़न की आवश्यकता है
जैसे $ x $ बढ़ता है. समापन अंतर्दृष्टि: बाध्यता से खरीदारी नहीं होती
एकरूपता (यह उदाहरण), और \omterm{def:b1:reals:bounds}{असीम}ितता रोकता नहीं है
यह ($\sqrt x $, \cref{exo:b1:continuity:9}); जो निर्णय लेता है वह है
\emph{\omterm{def:b1:complex:field}{मापांक} दोलन का}, और हेन का प्रमेय कॉम्पैक्ट कहता है
डोमेन इसे स्वचालित रूप से अनुशासित करते हैं।
\end{example}

\begin{example}[कैलकुलेटर प्रयोग, समझाया गया]\label{ex:b1:continuity:cositeration}
कैलकुलेटर में कोई भी संख्या टाइप करें और दबाएँ $\cos $ बार-बार:
डिस्प्ले लॉक हो जाता है $0.7390851\dots $ क्यों? एक के बाद दबाएँ
मूल्य निहित है $\intcc{-1}{1}$, दो बजे के बाद $\intcc{\cos 1}{1}
\subseteq \intcc{0.54}{1}$, एक स्थिर \omterm{prop:b1:reals:intervals}{अंतराल} के लिए $\cos $. उस पर,
$\abs{\cos a - \cos b} \leq \sin(1)\,\abs{a - b}$ साथ $\sin 1 =
0.841\dots < 1$ (उत्पाद-से-योग की \omterm{def:b1:topology:closure}{सीमा}
\cref{pb:b1:seq:1}, या का औसत मूल्य असमानता
\cref{ch:b1:derivative}): पुनरावृत्ति अनुबंध, इसलिए द्वारा
त्रुटि-नियंत्रण चरण \cref{met:b1:seq:recurrent},
\[
\abs{u_n - c} \leq (0.842)^{\,n-2}\,\abs{u_2 - c}
\longrightarrow 0 ,
\]
कहाँ $ c $ अद्वितीय निश्चित बिंदु है $\cos c = c $
(\cref{exo:b1:continuity:6}). के बारे में $40$ प्रेस तीन खरीदते हैं
दशमलव ($0.842^{40} \approx 10^{-3}$) --- एक ज्यामितीय दर,
द्विभाजन की तुलना में हल्का $2^{-n}$ प्रति चरण, लेकिन प्रत्येक प्रेस की लागत
एक कीस्ट्रोक जबकि प्रत्येक द्विभाजन चरण में एक पूर्ण चिह्न खर्च होता है
मूल्यांकन. समापन अंतर्दृष्टि: का निश्चित-बिंदु चित्र
\cref{ch:b1:seq} और इस अध्याय के अस्तित्व प्रमेय हैं
एक कहानी के दो हिस्से --- आईवीटी ने पाया $ c $, संकुचन पहुँचता है
यह.
\end{example}

\begin{remark}[ये प्रमेय आगे कहां काम करते हैं]
इस अध्याय के तीन स्तंभ प्रत्येक बाद वाले को शक्ति प्रदान करते हैं। द
मध्यवर्ती मूल्य प्रमेय प्रत्येक अस्तित्व-समाधान को पोषित करता है
तर्क और मोनोटोन आक्षेप प्रमेय; चरम मूल्य
प्रमेय अनुकूलन समस्याओं को प्रमेयों (रोल और) में बदल देता है
माध्य मान प्रमेय \cref{ch:b1:derivative} बिल्कुल शुरू करो
वहाँ); हेन का प्रमेय इसका कारण है निरंतर पर कार्य करता है
खंडों को एकीकृत किया जा सकता है \cref{ch:b1:integration} --- गणवेश $\delta $ यही वह चीज़ है जो रीमैन के योगों को एकाकार बनाती है। वर्ष में
2 वॉल्यूम के साथ वही तिकड़ी मानक वेक्टर स्थानों में फिर से दिखाई देती है
कॉम्पैक्टनेस वह काम कर रही है जो सेगमेंट यहां करते हैं।
\end{remark}

\begin{remark}[इस खंड के अंदर परिप्रेक्ष्य]
निरंतरता वह पिछड़ने वाला है लेकिन कभी सेवानिवृत्त नहीं हुआ।
\Cref{ch:b1:derivative} इसे भिन्नता के लिए मजबूत करता है और
एहसान लौटाता है (विभेदक का तात्पर्य है निरंतर);
\cref{ch:b1:integration} हेइन के माध्यम से, इस पर दो बार टिकी हुई है
निर्माण और मौलिक प्रमेय के माध्यम से, जिसका
केंद्रीय वस्तु $ x \mapsto \int_a^x f $ एक मात्र उन्नयन
निरंतर $ f $ एक को $ C^1$ प्राचीन। में
\cref{ch:b1:multivar}, निरंतरता दो चर में एक जाल रखता है
पूर्वावलोकन के लायक: फ़ंक्शन $\frac{xy}{x^2 + y^2}$ (विस्तारित
द्वारा $0$) है निरंतर में $ x $ प्रत्येक के लिए निश्चित $ y $ और में $ y $ के लिए
प्रत्येक निश्चित $ x $, अभी नहीं निरंतर मूल पर --- साथ में
विकर्ण $ x = y $ यह निरंतर है $\frac12$. अलग निरंतरता
से सख्ती से कमजोर है निरंतरता: अनुक्रमिक
चरित्र-चित्रण इस कदम से बच जाता है $\R^2$, लेकिन अनुक्रम अवश्य होना चाहिए
से आने की अनुमति दी जाए \emph{प्रत्येक} दिशा, न केवल
कुल्हाड़ियों के साथ.
\end{remark}

\section{अभ्यास}

\begin{exercise}[$\star $]\label{exo:b1:continuity:1}
अनुक्रमिक लक्षण वर्णन का प्रयोग करते हुए इसे सिद्ध करें $ x \mapsto
\sin\frac 1x $ की कोई \omterm{def:b1:topology:closure}{सीमा} नहीं है $0^+$ \emph{(दो अनुक्रम प्रदर्शित करें)}.
करता है $ x \mapsto x \sin\frac 1x $ एक लो?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:continuity:2}
का अध्ययन करें निरंतरता पर $\R $ का $ f(x) = \lfloor x \rfloor $ और का
$ g(x) = x - \lfloor x \rfloor $, और का $ h(x) = \lfloor x \rfloor +
(x - \lfloor x\rfloor)^2$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:continuity:3}
उस समीकरण को सिद्ध करें $ x^5 - 3x + 1 = 0$ कम से कम तीन वास्तविक हैं
समाधान \emph{(अच्छी तरह से चुने गए बिंदुओं पर मूल्यांकन करें और आवेदन करें
\cref{thm:b1:continuity:ivt} तीन असंयुक्त खंडों पर)}.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:continuity:4}
प्रत्येक को सिद्ध करें \omterm{def:b1:poly:def}{बहुपद} विषम डिग्री की वास्तविक जड़ होती है।
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:continuity:5}
(निश्चित बिंदु) चलिए $ f \colon \intcc{0}{1} \to \intcc{0}{1}$ होना
निरंतर. साबित करें कि $ f $ एक निश्चित बिंदु है: $ f(c) = c $ कुछ के लिए
$ c $. उदाहरण दें कि दोनों में से कोई भी नहीं निरंतरता न ही खंड हो सकता है
गिरा दिया.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:continuity:6}
उस समीकरण को सिद्ध करें $\cos x = x $ बिल्कुल एक वास्तविक समाधान है,
और वह इसमें निहित है $\intoo{0}{1}$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:continuity:7}
होने देना $ f \colon \R \to \R $ होना निरंतर साथ $ f(x) \to +\infty $ जैसा $ x
\to \pm\infty $. साबित करें कि $ f $ पर वैश्विक न्यूनतम प्राप्त करता है $\R $.
\emph{(एक उपस्तर वाले खंड को कम करें \omterm{def:b1:logic:sets}{तय करना}.)}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:continuity:8}
होने देना $ f \colon \R \to \R $ होना निरंतर और आवधिक (अवधि का)। $ T >
0$). साबित करें कि $ f $ बंधा हुआ है और अपनी \omterm{def:b1:topology:closure}{सीमा} को प्राप्त करता है, और वह वहां है
मौजूद है $ c $ साथ $ f(c + \frac T2) = f(c)$. \emph{(दूसरे के लिए
बिंदु, अध्ययन $ g(x) = f(x + \frac T2) - f(x)$ एक अवधि में.)}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:continuity:9}
साबित करें कि $ x \mapsto \sqrt x $ है \omterm{def:b1:continuity:uniform}{समान रूप से निरंतर} पर
$\intco{0}{+\infty}$, हालाँकि यह लिप्सचिट्ज़ के निकट नहीं है $0$.
\emph{(साबित करें और प्रयोग करें $\abs{\sqrt x - \sqrt y} \leq \sqrt{\abs{x -
y}}$.)}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b1:continuity:10}
होने देना $ f $ होना निरंतर और \omterm{def:b1:logic:inj}{इंजेक्शन} एक पर \omterm{prop:b1:reals:intervals}{अंतराल} $ I $. साबित करें कि
$ f $ सख्ती से एकरस है. \emph{संकेत: यदि नहीं, तो हैं $ a < b < c $
साथ, कहो, $ f(b) > f(a)$ और $ f(b) > f(c)$; मध्यवर्ती लागू करें
मूल्य प्रमेय के बीच एक मूल्य $\max(f(a), f(c))$ और $ f(b)$ पर
के दोनों तरफ $ b $.}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b1:continuity:11}
(कॉची का \omterm{pb:b1:continuity:1}{कार्यात्मक समीकरण}, निरंतर केस) चलो $ f \colon \R \to
\R $ होना निरंतर साथ $ f(x + y) = f(x) + f(y)$ सभी के लिए $ x, y $.
साबित करें कि $ f(x) = f(1)\,x $ सभी के लिए $ x $: सबसे पहले $\N $, $\Z $, $\Q $
(अकेले योगात्मकता द्वारा), फिर आगे $\R $ द्वारा निरंतरता और घनत्व
(\cref{thm:b1:reals:density}).
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b1:continuity:12}
होने देना $ f \colon \intco{0}{+\infty} \to \R $ होना निरंतर साथ
$ f(x) \to \ell \in \R $ जैसा $ x \to +\infty $. साबित करें कि $ f $ है
\emph{समान रूप से} निरंतर पर $\intco{0}{+\infty}$. \emph{(ए पर काटें
बड़ा $ A $: हेन चालू $\intcc{0}{A+1}$, \omterm{def:b1:topology:closure}{सीमा} पार $ A $; बनाओ
दो शासन व्यवस्थाएँ ओवरलैप होती हैं।)}
\end{exercise}

\section{समस्या: कॉची का कार्यात्मक समीकरण और उसकी बहनें}

\begin{problem}[{सप्ताहांत समस्या --- $ f(x + y) = f(x) + f(y)$: नियमितता रैखिकता को बल देती है, और राक्षसों का चित्र}]
\label{pb:b1:continuity:1}
कौन से कार्य संतुष्ट करते हैं $ f(x + y) = f(x) + f(y)$ सभी वास्तविक के लिए $ x,
y $? कॉची ने 1821 में प्रश्न उठाया; उत्तर एक प्रतिमान है.
\Cref{exo:b1:continuity:11} दिखाता है कि ऐसा \emph{additive}
फ़ंक्शन रैखिक है $\Q $ और वह पूर्ण निरंतरता बल $ f(x)
= cx $. यह समस्या नाटकीय रूप से परिकल्पना को तेज करती है ---
निरंतरता एक पर \emph{अकेला} बिंदु, या एकरसता, या मात्र
एक छोटे से पर \omterm{def:b1:topology:closure}{सीमा} \omterm{prop:b1:reals:intervals}{अंतराल}, प्रत्येक पर्याप्त है --- फिर पेंट करता है
एक काल्पनिक अरेखीय समाधान का चित्र (इसका ग्राफ)।
विमान को भरता है), बहन समीकरणों की विशेषता को हल करता है
$\eu^{cx}$, $ c\ln x $, $ x^c $ और $ cx^2$, और जेन्सेन के साथ समाप्त होता है
समीकरण और प्रमेय \emph{मध्यबिंदु-उत्तल $+$ निरंतर
$\implies $ उत्तल}.\index{कॉची कार्यात्मक है
समीकरण}\index{कार्यात्मक समीकरण} लगातार, \emph{additive}
मतलब: $ f(x + y) = f(x) + f(y)$ सभी के लिए $ x, y \in \R $.

\medskip
\noindent\textbf{Part I --- $\Q $-linearity, and one point of
निरंतरता.}
\begin{enumerate}
  \item होने देना $ f $ योगात्मक बनें. से \cref{exo:b1:continuity:11},
        $ f(r) = f(1)\,r $ तर्कसंगत के लिए $ r $. बेहतर साबित करो
        \omterm{def:b1:logic:statement}{कथन} नीचे उपयोग किया गया: के लिए \emph{प्रत्येक} $ x \in \R $ और $ r
        \in \Q $, $ f(rx) = r\,f(x)$ ($ f $ है
        $\Q $-\emph{रेखीय}).
  \item मान लीजिए योजक $ f $ है निरंतर एक ही बार में
        बिंदु $ x_0$. बताते हैं कि $ f $ है निरंतर हर जगह
        \emph{(गणना करें $ f(x + h) - f(x)$ के अनुसार $ f(x_0 + h)
        - f(x_0)$)}, इस तरह $ f(x) = f(1)\,x $.
  \item दिखाएँ कि एक योगात्मक फलन इसके द्वारा निर्धारित होता है
        किसी पर प्रतिबंध \omterm{def:b1:topology:dense}{घना} \omterm{def:b1:structures:subgroup}{उपसमूह}: यदि दो योगात्मक
        कार्यों पर सहमति है $\Z + \sqrt2\,\Z $ और दोनों हैं
        निरंतर, वे समान हैं --- जबकि बिना निरंतरता,
        निर्धारित करना $ f(1) = 0$ और $ f(\sqrt 2) = 1$ सुसंगत है
        के साथ $\Q $-पर रैखिकता \omterm{def:b1:structures:subgroup}{उपसमूह}. गणना करना $ f(m +
        n\sqrt2)$ इस नुस्खे के लिए.
  \item होने देना $ f $ योगात्मक हो और ऊपर से घिरा हो $ M $ कुछ पर
        \omterm{prop:b1:reals:intervals}{अंतराल} $\intcc{a}{b}$ साथ $ a < b $. बताते हैं कि $ f $ है
        ऊपर से घिरा हुआ $\intcc{0}{\ell}$, $\ell = b - a $
        \emph{(द्वारा अनुवाद करें) $ a $)}.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The regularity ladder.}
\begin{enumerate}[resume]
  \item सतत प्रश्न 4: उपयोग करना $ f(\ell - t) + f(t) =
        f(\ell)$, बताते हैं कि $ f $ भी बाध्य है \emph{नीचे} पर
        $\intcc{0}{\ell}$: $\abs f \leq C $ वहाँ।
  \item बताते हैं कि $\abs{f(t)} \leq \frac{C}{n}$ के लिए $ t \in
        \intcc{0}{\ell/n}$, और उसका निष्कर्ष निकालो $ f $ है निरंतर पर
        $0$ \emph{(विषमता बाईं ओर संभालती है)}, इसलिए
        हर जगह (प्रश्न 2): एक योगात्मक फलन से घिरा हुआ
        एक \omterm{prop:b1:reals:intervals}{अंतराल} रैखिक है.
  \item मोनोटोन केस निकालें: एक योगात्मक कार्य
        कुछ पर घट नहीं रहा $\intcc{a}{b}$ ($ a < b $) है $ f(x)
        = cx $ साथ $ c \geq 0$.
  \item इकट्ठा करो \emph{नियमितता सीढ़ी}: एक योज्य के लिए
        $ f $, निम्नलिखित समतुल्य हैं --- (ए) $ f(x) = cx $;
        (बी) $ f $ निरंतर; (सी) $ f $ निरंतर एक बिंदु पर; (डी)
        $ f $ कुछ नॉनडिजनरेट पर मोनोटोन \omterm{prop:b1:reals:intervals}{अंतराल}; (ई) $ f $
        कुछ नपुंसक पर बँधा हुआ \omterm{prop:b1:reals:intervals}{अंतराल}. व्यवस्थित करें
        निहितार्थ ताकि प्रत्येक या तो तुच्छ हो या पहले से ही हो
        साबित हुआ.
  \item सत्यापित करें कि प्रश्न 4-6 में केवल एक \omterm{def:b1:topology:closure}{सीमा} का उपयोग किया गया है
        \emph{ऊपर}: एक योगात्मक फलन ऊपर एक से घिरा हुआ है
        अविक्षिप्त \omterm{prop:b1:reals:intervals}{अंतराल} पहले से ही रैखिक है. निष्कर्ष निकालें
        दर्पण \omterm{def:b1:logic:statement}{कथन} नीचे एक बाउंड के लिए, और रिकॉर्ड करें
        इस प्रकार सीढ़ी के पायदान का सबसे मजबूत रूप (ई) प्राप्त हुआ।
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Portrait of a monster.} मान लीजिए
अभी $ f $ योगात्मक है लेकिन \emph{नहीं} रैखिक.
\begin{enumerate}[resume]
  \item दिखाएँ कि वहाँ शून्येतर वास्तविकताएँ हैं $ u, v $ साथ
        $\dfrac{f(u)}{u} \neq \dfrac{f(v)}{v}$, और वह
        वेक्टर $(u, f(u))$ और $(v, f(v))$ विमान को फैलाएं (उनका
        निर्धारक $ u f(v) - v f(u)$ शून्येतर है)।
  \item का ग्राफ दिखाएँ $ f $ इसमें सभी बिंदु शामिल हैं
        \[
        r\,(u, f(u)) + s\,(v, f(v)), \qquad r, s \in \Q ,
        \]
        और यह निष्कर्ष निकालें कि ग्राफ़ है \emph{\omterm{def:b1:topology:dense}{घना} में $\R^2$}: के लिए
        हर बिंदु $(x_0, y_0)$ विमान का और हर
        $\varepsilon > 0$, कुछ $(x, f(x))$ भीतर है
        $\varepsilon $ इसका \emph{(हल करें $2\times2$ असली
        सिस्टम, फिर वास्तविक गुणांक का अनुमान लगाएं
        तर्कसंगत)}.
  \item प्रश्न 11 से पूरा चित्र निकालें: एक अरेखीय
        प्रत्येक नॉनडीजेनरेट पर योगात्मक कार्य \omterm{def:b1:reals:bounds}{असीम}ित है
        \omterm{prop:b1:reals:intervals}{अंतराल}, हर बिंदु पर असंतत, नहीं पर एकरस
        \omterm{prop:b1:reals:intervals}{अंतराल}, और किसी की इसकी छवि \omterm{prop:b1:reals:intervals}{अंतराल} है \omterm{def:b1:topology:dense}{घना} में
        $\R $. प्रश्न 8 से सामंजस्य स्थापित करें।
  \item राक्षस मौजूद हैं --- एक पर \omterm{def:b1:topology:dense}{घना} \omterm{def:b1:structures:subgroup}{उपसमूह}, रचनात्मक रूप से:
        पर $ G = \Z + \sqrt2\,\Z $ परिभाषित करना $ f(m + n\sqrt2) = n $.
        दिखाओ $ f $ अच्छी तरह से परिभाषित और योगात्मक है $ G $, ओर वो
        $ f $ पर \omterm{def:b1:reals:bounds}{असीम}ित है $ G \cap \intoo{0}{\varepsilon}$ हरएक के लिए $\varepsilon > 0$ \emph{(निश्चित के लिए $ N $, केवल
        निश्चित रूप से अनेक $ g = m + n\sqrt2 \in \intoo{0}{1}$ पास होना
        $\abs n \leq N $; अभी तक $ G \cap \intoo{0}{\varepsilon}$ है
        अनंत)}. एक पैराग्राफ में बताएं कि ऐसा क्यों बढ़ाया जा रहा है
        एक $ f $ सभी को $\R $ के आधार की आवश्यकता है $\R $ के तौर पर
        $\Q $-वेक्टर स्पेस (ए \emph{हामेल आधार}), जिसका
        अस्तित्व इससे परे एक स्वयंसिद्ध पसंद का मामला है
        आयतन.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The sister equations.} सब
यहाँ कार्य हैं निरंतर.
\begin{enumerate}[resume]
  \item होने देना $ f \colon \R \to \R $ होना निरंतर, समान रूप से नहीं
        $0$, साथ $ f(x + y) = f(x)f(y)$. दिखाओ $ f(x) =
        f\bigl(\frac x2\bigr)^2 \geq 0$, तब $ f > 0$ हर जगह,
        फिर $ f(x) = \eu^{cx}$ कुछ के लिए $ c $: घातांक
        बिल्कुल वही हैं निरंतर रूपवाद से $(\R, +)$ को
        $(\R^*, \times)$.
  \item होने देना $ f \colon \intoo{0}{+\infty} \to \R $ होना निरंतर
        साथ $ f(xy) = f(x) + f(y)$. दिखाओ $ f(x) = c\ln x $
        \emph{(परिवहन द्वारा $\exp $)}.
  \item होने देना $ f \colon \intoo{0}{+\infty} \to \intoo{0}{+\infty}$
        होना निरंतर साथ $ f(xy) = f(x)f(y)$. दिखाओ $ f(x) = x^c $.
  \item सभी खोजें निरंतर $ f \colon \R \to \R $ साथ
        \[
        f(x + y) = f(x) + f(y) + f(x)f(y) .
        \]
        \emph{(अध्ययन $ h = 1 + f $; विकृत मामले का इलाज करें
        अलग से।)}
  \item (समानांतर चतुर्भुज) सभी खोजें निरंतर $ f \colon \R \to \R $
        साथ $ f(x + y) + f(x - y) = 2f(x) + 2f(y)$: दिखाओ $ f $ है
        यहां तक कि, $ f(0) = 0$, $ f(nx) = n^2 f(x)$ प्रेरण द्वारा, फिर
        $ f(x) = f(1)\,x^2$. (यह समीकरण का फिंगरप्रिंट है
        द्विघात रूप --- समांतर चतुर्भुज नियम जो पता लगाता है,
        वर्ष 2 खंड में, कौन से मानदंड भीतर से आते हैं
        उत्पाद.)
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Jensen and midpoint convexity.}
\begin{enumerate}[resume]
  \item (जेन्सेन का समीकरण) चलो $ f \colon \R \to \R $ होना
        निरंतर साथ $ f\bigl(\frac{x+y}{2}\bigr) = \frac{f(x)
        + f(y)}{2}$. बताते हैं कि $ g = f - f(0)$ संतुष्ट
        $ g\bigl(\frac x2\bigr) = \frac{g(x)}{2}$, वह निष्कर्ष निकालो
        $ g $ योगात्मक है, और निष्कर्ष निकालता है $ f(x) = cx + d $.
  \item मान लीजिए अब केवल \emph{असमानता}: $ f $ निरंतर
        साथ
        \[
        f\Bigl(\frac{x + y}{2}\Bigr) \leq \frac{f(x) +
        f(y)}{2} \qquad (x, y \in \R).
        \]
        प्रेरण द्वारा सिद्ध करें $ n $ वह सभी डायडिक भारों के लिए
        $\lambda = \frac{k}{2^n} \in \intcc{0}{1}$:
        \[
        f\bigl(\lambda x + (1 - \lambda)y\bigr) \leq \lambda
        f(x) + (1 - \lambda) f(y) .
        \]
  \item द्वारा बढ़ाएँ निरंतरता और घनत्व डायडिक्स का
        (\cref{exo:b1:reals:8}) प्रत्येक को $\lambda \in
        \intcc{0}{1}$: ए निरंतर मध्यबिंदु-उत्तल फ़ंक्शन
        पूर्ण उत्तलता असमानता (धारणा) को संतुष्ट करता है
        में व्यवस्थित रूप से अध्ययन किया गया \cref{ch:b1:derivative}).
  \item दिखाओ निरंतरता छोड़ा नहीं जा सकता: एक अरैखिक योज्य
        $ f $ को संतुष्ट करता है \emph{मध्यबिंदु समानता} प्रश्न का
        19 अभी तक किसी पर कोई उत्तलता असमानता नहीं है \omterm{prop:b1:reals:intervals}{अंतराल} (प्रश्न
        12). नैतिक: मध्यबिंदु उत्तलता एक गणनीय-चरण है
        संपत्ति (डायडिक्स), उत्तलता एक सातत्य;
        निरंतरता क्या यह पुल बिल्कुल वैसा ही है जैसा कि भाग I--II में है।
\end{enumerate}

\medskip
\noindent\textbf{Part VI --- Last variations and synthesis.}
\begin{enumerate}[resume]
  \item सभी खोजें निरंतर $ f \colon \R \to \R $ साथ $ f(x + y)
        = f(x) + f(y) + xy $ \emph{(विशेष घटाएं
        समाधान $\frac{x^2}{2}$)}.
  \item सिद्ध करें: यदि $ f \colon \R \to \R $ है निरंतर और
        केवल एक पर योगात्मक \emph{\omterm{def:b1:topology:dense}{घना} \omterm{def:b1:structures:subgroup}{उपसमूह}} $ G $ (अर्थात्\
        $ f(g + g') = f(g) + f(g')$ के लिए $ g, g' \in G $), तब $ f $
        पर योगात्मक है $\R $. अधिक सामान्यतः, दो निरंतर
        ऐसे कार्य जो a पर सहमत हों \omterm{def:b1:topology:dense}{घना} का उपसमुच्चय $\R $ हैं
        बराबर.
  \item संश्लेषण, प्रत्येक एक वाक्य: (i) नियमितता बताएं
        स्मृति से प्रश्न 8 की सीढ़ी; (ii) क्यों स्पष्ट करें
        ``ग्राफ \omterm{def:b1:topology:dense}{घना} विमान में'' सही मानसिक छवि है
        नियमितता की विफलता के लिए; (iii) पांचों की सूची बनाएं
        भाग IV-V और में वर्णित शास्त्रीय कार्य
        एकल विधि जिसने उन सभी को पकड़ लिया; (iv) दोनों के नाम बताएं
        वे स्थान जहाँ घनत्व का $\Q $ (या डायडिक्स के) में $\R$
        तर्क को आगे बढ़ाया, और वह स्थान जहां यह नहीं हो सका
        (प्रश्न 13).
\end{enumerate}
\end{problem}
