\chapter{मानक कार्य}\label{ch:b1:functions}

विश्लेषण केवल उतना ही उपयोगी है जितना कार्यों का भंडार जिसमें कोई महारत हासिल कर लेता है। को
माध्यमिक विद्यालय से विरासत में मिला संग्रह --- शक्तियाँ, घातीय,
लघुगणक, त्रिकोणमितीय फलन --- यह अध्याय उनका व्युत्क्रम जोड़ता है
कार्य ($\arcsin $, $\arccos $, $\arctan $) और अतिशयोक्तिपूर्ण परिवार।
हाई स्कूल स्तर पर डेरिवेटिव का स्वतंत्र रूप से उपयोग किया जाता है; द
व्युत्क्रम फलनों के पीछे का सिद्धांत पूरा हो गया है
\cref{ch:b1:continuity,ch:b1:derivative}.

\section{घातांक, लघुगणक, शक्तियाँ}

\begin{proposition}[पुनर्कथन और लक्षण वर्णन]\label{prop:b1:functions:expln}
$\exp \colon \R \to \intoo{0}{+\infty}$ और $\ln \colon
\intoo{0}{+\infty} \to \R $ सख्ती से, पारस्परिक आक्षेप हैं
बढ़ रहा है, के साथ
\[
\exp(x + y) = \exp x \exp y, \qquad \ln(xy) = \ln x + \ln y,
\qquad \exp' = \exp, \qquad \ln'(x) = \frac 1x .
\]
इसके अतिरिक्त $\exp $ है \emph{केवल} विभेदनीय कार्य $ f \colon
\R \to \R $ साथ $ f' = f $ और $ f(0) = 1$.
\end{proposition}

\begin{proof}
पुनर्\omterm{def:b1:logic:statement}{कथन} हाई स्कूल सामग्री है। विशिष्टता के लिए, चलो $ f' = f $,
$ f(0) = 1$, और सेट करें $ g(x) = f(x)\,\eu^{-x}$. तब $ g' = f'\eu^{-x} -
f\eu^{-x} = 0$, इसलिए $ g $ के बराबर स्थिरांक है $ g(0) = 1$: $ f = \exp $.
\end{proof}

\begin{definition}[सामान्य शक्तियाँ]\label{def:b1:functions:powers}
के लिए $ x > 0$ और $\alpha \in \R $:
$\;x^\alpha = \eu^{\alpha \ln x}$\index{शक्ति समारोह}. के लिए $ a > 0$,
$ a \neq 1$, द \emph{आधार से लघुगणक $ a $} है $\log_a x =
\frac{\ln x}{\ln a}$, का उलटा $ x \mapsto a^x $.
\end{definition}

\begin{example}[घातांकीय समीकरणों को हल करना]\label{ex:b1:functions:expequations}
हल करना $2^x = 5^{\,x-1}$ में $\R $. दोनों पक्ष सकारात्मक हैं, इसलिए लें
लघुगणक --- एक उलटा कदम:
\[
x\ln2 = (x - 1)\ln5
\iff x(\ln2 - \ln5) = -\ln5
\iff x = \frac{\ln5}{\ln5 - \ln2} = \frac{\ln 5}{\ln\frac52}
\approx 1.756 .
\]
अगला, हल करें $ x^{\sqrt2} = 3$ के लिए $ x > 0$: शक्ति तक बढ़ाएँ
$\frac1{\sqrt2}$ (अर्थात्\ पारस्परिक आपत्ति लागू करें):
$ x = 3^{1/\sqrt2} = \eu^{(\ln 3)/\sqrt2} \approx 2.175$. द
अंतर्दृष्टि: शक्तियों को मिलाने वाला हर समीकरण खुलता है $\ln $ और
$\exp $, क्योंकि परिभाषा $ x^\alpha = \eu^{\alpha\ln x}$
घातांक के अंकगणित में सभी शक्ति जोड़-तोड़ को कम कर देता है ---
लेकिन केवल डोमेन पर $ x > 0$ जहां वह परिभाषा रहती है.
\end{example}

\begin{example}[दोगुना समय]\label{ex:b1:functions:doubling}
एक मात्रा बढ़ती है $3\%$ प्रति चरण: बाद में $ n $ यह कदम है
से गुणा किया गया $(1.03)^n $. यह कब दोगुना हो जाता है? हल करना $(1.03)^n
\geq 2$:
\[
n\ln(1.03) \geq \ln2
\iff n \geq \frac{\ln 2}{\ln 1.03}
= \frac{0.6931}{0.02956} \approx 23.4 ,
\]
इसलिए पहला दोहरीकरण चरण पर होता है $24$. (फाइनेंसरों का
``का नियम $72$'', जो दोगुना होने के समय का अनुमान लगाता है
$72$ दर को प्रतिशत में विभाजित करने पर यह गणना होती है
सन्निकटन $\ln(1 + x) \approx x $, में मात्राबद्ध
\cref{ch:b1:taylor}.) घातीय
प्रक्रियाओं के बारे में उनके लघुगणक के माध्यम से सबसे अच्छा तर्क दिया जाता है: पर
वह पैमाना, विकास रैखिक है और प्रश्न विभाजन बन जाते हैं।
\end{example}

\begin{example}[कितना लंबा $2^{2026}$?]\label{ex:b1:functions:digits}
किसी पूर्णांक के दशमलव अंकों की संख्या $ N \geq 1$ है
$\floor{\log_{10} N} + 1$ (वास्तव में $ N $ है $ d $ अंक बिल्कुल कब
$10^{d-1} \leq N < 10^d $, यानी\ $ d - 1 \leq \log_{10}N < d $). के लिए
$ N = 2^{2026}$:
\[
\log_{10} 2^{2026} = 2026\,\log_{10}2
= 2026 \times 0.301030 = 609.887 ,
\]
इसलिए $2^{2026}$ है $610$ अंक. भिन्नात्मक भाग में a होता है
बोनस: $10^{0.887} \approx 7.7$, तो संख्या \emph{शुरू होता है} साथ
$7$. एक गुणन से किसी संख्या के बारे में एक प्रश्न का उत्तर मिल गया
कभी लिखूंगा --- लघुगणक गुणात्मक आकार को संपीड़ित करता है
योगात्मक आकार में, जो उनका संपूर्ण ऐतिहासिक बिंदु है
(\cref{ex:b1:structures:expmorphism} उस वाक्य को सटीक बनाता है)।
\end{example}

\begin{proposition}[शक्ति नियम और विकास तुलना]\label{prop:b1:functions:powerrules}
के लिए $ x, y > 0$ और $\alpha, \beta \in \R $:
\[
x^{\alpha+\beta} = x^\alpha x^\beta, \quad
(x^\alpha)^\beta = x^{\alpha\beta}, \quad
(xy)^\alpha = x^\alpha y^\alpha, \quad
(x^\alpha)' = \alpha\, x^{\alpha - 1}.
\]
विकास का पैमाना\index{विकास तुलना} जैसा $ x \to +\infty $, हरएक के लिए
$\alpha > 0$ और $\beta > 0$:
\[
\frac{(\ln x)^\beta}{x^\alpha} \longrightarrow 0,
\qquad
\frac{x^\alpha}{\eu^{\beta x}} \longrightarrow 0 .
\]
\end{proposition}

\begin{proof}
पहचान उन लोगों को प्रतिलेखित करती है $\exp $ और $\ln $ के माध्यम से
परिभाषा. दूसरे के लिए विस्तार से (सबसे कम स्पष्ट): $ x^\alpha
> 0$ और $\ln(x^\alpha) = \alpha\ln x $ (आवेदन करना $\ln $ को
परिभाषा), तो
\[
(x^\alpha)^\beta = \eu^{\beta\ln(x^\alpha)}
= \eu^{\beta\alpha\ln x} = x^{\alpha\beta} ;
\]
पहली और तीसरी समान दो-पंक्ति वाली अनवाइंडिंग हैं
$\exp(u + v) = \exp u\exp v $ और $\ln(xy) = \ln x + \ln y $. द
व्युत्पन्न श्रृंखला नियम है: $(\eu^{\alpha\ln x})' =
\frac{\alpha}{x} \eu^{\alpha\ln x}$.

तुलना: से $\frac{\ln t}{t} \to 0$ (हाई में साबित हुआ
स्कूल की मात्रा), स्थानापन्न $ t = x^{\alpha/\beta}$:
$\frac{\ln x}{x^{\alpha/\beta}} \to 0$, फिर शक्ति तक बढ़ाएँ
$\beta $. दूसरे के लिए, स्थानापन्न $ x = \ln u $ पहले में।
\end{proof}

\begin{example}[प्रत्येक पाठक के पास दो सीमाएँ होनी चाहिए]\label{ex:b1:functions:xx}
क्या हैं $\lim_{x \to 0^+} x^x $ और $\lim_{x \to +\infty}
x^{1/x}$? दोनों शक्तियों को घातांक के माध्यम से परिभाषित किया गया है
पहले घातांक को व्यवस्थित करें:
\[
x^x = \eu^{x\ln x}, \qquad
x\ln x = -\frac{\ln(1/x)}{1/x} \longrightarrow 0
\quad (x \to 0^+),
\qquad\text{so } x^x \longrightarrow \eu^0 = 1 ;
\]
और $ x^{1/x} = \eu^{(\ln x)/x} \to \eu^0 = 1$ जैसा $ x \to +\infty $,
सीधे द्वारा \omterm{prop:b1:functions:powerrules}{विकास तुलना}. अंतर्दृष्टि: एक अनिश्चित
शक्ति (आकार की) $0^0$ या $\infty^0$) है \emph{हमेशा} द्वारा संभाला गया
पुनर्लेखन $ u^v = \eu^{v\ln u}$ और उत्पाद का विश्लेषण करना $ v \ln u $
--- कभी भी आधार और घातांक से अलग-अलग अनुमान लगाकर नहीं। द
समारोह $ x \mapsto \frac{\ln x}{x}$ इससे दोनों की सीमा तय हो गई है
इस अध्याय की सप्ताहांत समस्या का विस्तृत अध्ययन किया गया।
\end{example}

\begin{remark}[घातों और लघुगणक के साथ सामान्य गड़बड़ियाँ]\label{rem:b1:functions:pitfalls}
\begin{enumerate}
  \item \emph{डोमेन.} $ x^\alpha $ तर्कहीन के लिए $\alpha $ आवश्यकताओं
        $ x > 0$; $(-8)^{1/3}$ ``के पक्ष में इससे बचना सबसे अच्छा है
        का वास्तविक घनमूल $-8$'', क्योंकि नियम
        $(x^\alpha)^\beta = x^{\alpha\beta}$ चुपचाप विफल रहता है
        नकारात्मक: $\bigl((-8)^2\bigr)^{1/6} = 2 \neq -2$.
  \item \emph{$\sqrt{x^2} = \abs x $}, नहीं $ x $: भूल जाना
        निरपेक्ष मूल्य खोई हुई नकारात्मकता का क्लासिक स्रोत है
        समाधान.
  \item \emph{$\ln(xy) = \ln x + \ln y $ आवश्यक है $ x, y > 0$.} के लिए
        नकारात्मक तर्क, $\ln(xy)$ जबकि परिभाषित किया जा सकता है
        दाहिनी ओर नहीं है.
  \item \emph{कौन सा कार्य बढ़ता है?} में $ x^\alpha $ 
        \emph{आधार} बदलता रहता है; में $ a^x $  \emph{प्रतिपादक} करता है।
        उनकी वृद्धि बेहद अलग है
        (\cref{prop:b1:functions:powerrules}), और संकर
        जैसे भाव $ x^{\ln x}$ या $ x^{1/x}$ होना ही चाहिए
        के रूप में पुनः लिखा गया $\eu^{(\cdot)}$ किसी भी तर्क से पहले ---
        जैसा कि \cref{ex:b1:functions:xx}.
\end{enumerate}
\end{remark}

\section{व्युत्क्रम त्रिकोणमितीय फलन}

\begin{definition}[$\arcsin $, $\arccos $, $\arctan $]\label{def:b1:functions:arc}
प्रतिबंध
\[
\sin \colon \intcc{-\tfrac\pi2}{\tfrac\pi2} \to \intcc{-1}{1},
\qquad
\cos \colon \intcc{0}{\pi} \to \intcc{-1}{1},
\qquad
\tan \colon \intoo{-\tfrac\pi2}{\tfrac\pi2} \to \R
\]
पूरी तरह से एकरस आक्षेप हैं। उनका उलटा एमएपीएस लिखा गया है
$\arcsin $\index{आर्कसीन}, $\arccos $\index{कोटिकोज्या} और
$\arctan $\index{आर्कटिक स्पर्शरेखा}. इस प्रकार, उदाहरण के लिए, $ y = \arcsin x $
($ x \in \intcc{-1}{1}$) है \emph{} में कोण
$\intcc{-\frac\pi2}{\frac\pi2}$ जिसका साइन है $ x $.
\end{definition}

\begin{proposition}[संजात]\label{prop:b1:functions:arcderiv}
उनके डोमेन के आंतरिक भाग पर:
\[
\arcsin' x = \frac{1}{\sqrt{1 - x^2}},
\qquad
\arccos' x = \frac{-1}{\sqrt{1 - x^2}},
\qquad
\arctan' x = \frac{1}{1 + x^2}.
\]
\end{proposition}

\begin{proof}
व्युत्क्रम-फ़ंक्शन नियम का पूर्वानुमान सिद्ध हुआ
\cref{ch:b1:derivative}: अगर $ f $ एक अलग आक्षेप है और
$ f' \neq 0$, तब $(f^{-1})'(x) = \frac{1}{f'(f^{-1}(x))}$. के लिए
$\arcsin $: साथ $ y = \arcsin x $,
\[
\arcsin' x = \frac{1}{\cos y}
= \frac{1}{\sqrt{1 - \sin^2 y}} = \frac{1}{\sqrt{1 - x^2}},
\]
कहाँ $\cos y = +\sqrt{1 - \sin^2 y}$ क्योंकि $ y \in
\intoo{-\frac\pi2}{\frac\pi2}$ बल $\cos y > 0$. उसी प्रकार
$\arccos' x = \frac{1}{-\sin y}$ साथ $\sin y > 0$ पर
$\intoo{0}{\pi}$, और $\arctan' x = \frac{1}{1 + \tan^2 y} =
\frac{1}{1 + x^2}$ का उपयोग करते हुए $\tan' = 1 + \tan^2$.
\end{proof}

\begin{example}[एक छिपे हुए स्थिरांक को पहचानना]\label{ex:b1:functions:hiddenconst}
अध्ययन $ g(x) = \arctan\dfrac{1 - x}{1 + x}$ पर $\intoo{-1}
{+\infty}$. श्रृंखला नियम और एक संक्षिप्त गणना:
\[
g'(x) = \frac{1}{1 + \bigl(\frac{1-x}{1+x}\bigr)^2}\cdot
\frac{-(1+x) - (1-x)}{(1+x)^2}
= \frac{-2}{(1+x)^2 + (1-x)^2}
= \frac{-1}{1 + x^2} ,
\]
तब से $(1+x)^2 + (1-x)^2 = 2 + 2x^2$. इसलिए $ g' = -\arctan'$: द
समारोह $ g + \arctan $ पर शून्य व्युत्पन्न है \emph{अंतराल}
$\intoo{-1}{+\infty}$, इसलिए वहां स्थिर है; इसका मूल्य पर $ x =
0$ है $\arctan 1 + \arctan 0 = \frac\pi4$. निष्कर्ष:
\[
\arctan\frac{1 - x}{1 + x} = \frac\pi4 - \arctan x
\qquad (x > -1) .
\]
पर $\intoo{-\infty}{-1}$ वही व्युत्पन्न गणना लागू होती है लेकिन
स्थिरांक अलग है ($-\frac{3\pi}4$: सीमा का मूल्यांकन इस प्रकार करें
$ x \to -\infty $). अंतर्दृष्टि: ``शून्य व्युत्पन्न का तात्पर्य है
स्थिरांक'' एक अंतराल-दर-अंतराल है \omterm{def:b1:logic:statement}{कथन} --- बिलकुल
सूक्ष्मता का शोषण किया गया \cref{exo:b1:functions:6} और
\cref{exo:b1:functions:10}.
\end{example}

\begin{proposition}[मानक पहचान]\label{prop:b1:functions:arcidentities}
\begin{enumerate}
  \item के लिए $ x \in \intcc{-1}{1}$: $\arcsin x + \arccos x =
        \dfrac{\pi}{2}$.
  \item के लिए $ x > 0$: $\arctan x + \arctan\dfrac 1x = \dfrac{\pi}{2}$
        (और $-\frac\pi2$ के लिए $ x < 0$).
  \item $\sin(\arccos x) = \cos(\arcsin x) = \sqrt{1 - x^2}$;
        $\;\tan(\arcsin x) = \dfrac{x}{\sqrt{1 - x^2}}$ के लिए
        $\abs x < 1$.
\end{enumerate}
\end{proposition}

\begin{proof}
(1) की व्युत्पत्ति $ x \mapsto \arcsin x + \arccos x $ शून्य पर है
$\intoo{-1}{1}$ (\cref{prop:b1:functions:arcderiv}), तो फ़ंक्शन
वहाँ स्थिर है, उसके मान के बराबर $\frac\pi2$ पर $0$; समापन बिंदु
मूल्य $\pm 1$ सीधे जाँच की जाती है ($\frac\pi2 + 0$ और $-\frac\pi2 +
\pi $).

(2) वही विधि चालू $\intoo{0}{+\infty}$: व्युत्पन्न है
$\frac{1}{1+x^2} + \frac{-1/x^2}{1 + 1/x^2} = \frac{1}{1+x^2} -
\frac{1}{x^2 + 1} = 0$, और कम से $ x = 1$ योग है $2\arctan 1 =
\frac\pi2$. के लिए $ x < 0$, की विषमता का प्रयोग करें $\arctan $.

(3)साथ $ y = \arccos x \in \intcc{0}{\pi}$: $\sin y \geq 0$, इसलिए
$\sin y = \sqrt{1 - \cos^2 y} = \sqrt{1 - x^2}$; इसी तरह के लिए
$\cos(\arcsin x)$, और स्पर्शरेखा सूत्र भागफल है।
\end{proof}

\begin{remark}
$\arcsin(\sin\theta) = \theta $ रखती है \emph{केवल} के लिए $\theta \in
\intcc{-\frac\pi2}{\frac\pi2}$: उदाहरण के लिए
$\arcsin(\sin\pi) = 0 \neq \pi $. दूसरे में रचना
दिशा, $\sin(\arcsin x) = x $, सभी पर मान्य है
$\intcc{-1}{1}$.
\end{remark}

\begin{example}[मुख्य अंतराल पर वापस लपेटना]
\label{ex:b1:functions:wrap}
गणना करना
\[
\arctan\Bigl(\tan\frac{3\pi}4\Bigr)
\qquad\text{and}\qquad
\arctan\Bigl(\tan\frac{17\pi}5\Bigr).
\]
नुस्खा: बदलें
के अद्वितीय कोण द्वारा कोण $\intoo{-\frac\pi2}{\frac\pi2}$ के साथ
वही स्पर्शरेखा, यानी\ का सही गुणज घटाएँ $\pi $
(की अवधि $\tan $). पहला: $\frac{3\pi}4 - \pi =
-\frac\pi4$, तो उत्तर है $-\frac\pi4$. दूसरा: $\frac{17\pi}
5 - 3\pi = \frac{2\pi}5 \in \intoo{-\frac\pi2}{\frac\pi2}$, तो
उत्तर है $\frac{2\pi}5$. गणना यूक्लिडियन है
द्वारा कोण का विभाजन $\pi $ भेस में --- और अनुरूप
के लिए नुस्खे $\arcsin $ (प्रतिबिंबित करें
$\intcc{-\frac\pi2}{\frac\pi2}$, अवधि $2\pi $) और $\arccos $
(प्रतिबिंबित करें $\intcc0\pi $) गाड़ी चलाना \cref{exo:b1:functions:1} और
का टुकड़ेवार जवाब \cref{exo:b1:functions:10}.
\end{example}

\begin{example}[आर्कटिक स्पर्शरेखाओं को सुरक्षित रूप से जोड़ना]\label{ex:b1:functions:arctansum}
आइए साबित करें
\[
\arctan\frac12 + \arctan\frac15 + \arctan\frac18 = \frac\pi4 .
\]
दो अवयव: स्पर्शरेखा जोड़ सूत्र, और --- चरण
शुरुआती लोग भूल जाते हैं --- ए \emph{स्थानीयकरण} राशि का. पहला,
$\tan(u + v) = \frac{\tan u + \tan v}{1 - \tan u\tan v}$ साथ
$ u = \arctan\frac12$, $ v = \arctan\frac15$ देता है
\[
\tan(u + v) = \frac{\frac12 + \frac15}{1 - \frac1{10}}
= \frac{7/10}{9/10} = \frac79,
\qquad\text{then}\qquad
\tan\Bigl(u + v + \arctan\frac18\Bigr)
= \frac{\frac79 + \frac18}{1 - \frac7{72}}
= \frac{65/72}{65/72} = 1 .
\]
दूसरा, तीनों कोणों में से प्रत्येक कोण में स्थित है $\intoo0{\frac\pi4}$
(उनके तर्क कम हैं $1$), तो योग निहित है
$\intoo0{\frac{3\pi}4}$; स्पर्शरेखा के साथ वहां एकमात्र कोण है $1$ है
$\frac\pi4$. स्थानीयकरण के बिना निष्कर्ष होगा
``$\frac\pi4$ के गुणक तक $\pi $'' --- आधा प्रमाण। द
वही दो-चरणीय अनुशासन चलता है \cref{exo:b1:functions:8} और
\cref{exo:b1:functions:12}.
\end{example}

\begin{example}[एक चापस्पर्शरेखा समीकरण]\label{ex:b1:functions:arctaneq}
हल करना $\arctan x + \arctan 2x = \dfrac\pi4$. पहला स्थानीयकरण:
बायीं ओर का चिन्ह है $ x $ (दोनों पद करते हैं), इसलिए कोई भी समाधान है
$ x > 0$, और फिर योग निहित है $\intoo0\pi $. स्पर्शरेखाएँ लीजिए
(\omterm{def:b1:logic:inj}{इंजेक्शन} लंबाई के किसी अंतराल पर नहीं $\pi $, लेकिन के साथ संयुक्त
स्थानीयकरण यह पर्याप्त होगा): अतिरिक्त सूत्र देता है
\[
\tan\bigl(\arctan x + \arctan 2x\bigr)
= \frac{3x}{1 - 2x^2} = 1
\iff 2x^2 + 3x - 1 = 0
\iff x = \frac{-3 \pm \sqrt{17}}4 .
\]
स्थानीयकरण द्वारा नकारात्मक जड़ को त्याग दिया जाता है; के लिए
सकारात्मक उम्मीदवार $ x_0 = \frac{\sqrt{17} - 3}4 \approx 0.28$,
राशि निहित है $\intoo0\pi $ स्पर्शरेखा के साथ $1$, और एकमात्र ऐसा
कोण है $\frac\pi4$ (में $\intoo{\frac\pi2}\pi $ स्पर्शरेखा है
नकारात्मक): $ x_0$ अद्वितीय समाधान है. के आकार पर ध्यान दें
तर्क: स्पर्शरेखाएँ लेने से समाधान बन सकते हैं, उन्हें कभी न खोएँ,
तो कोई बहुपद समीकरण को हल करता है \emph{तब} द्वारा फ़िल्टर करता है
स्थानीयकरण --- स्क्वेरिंग के समान ही फॉरवर्ड-चेक अनुशासन
एक समीकरण.
\end{example}

\section{अतिशयोक्तिपूर्ण कार्य}

\begin{definition}[$\cosh $, $\sinh $, $\tanh $]\label{def:b1:functions:hyperbolic}
के लिए $ x \in \R $:
\[
\cosh x = \frac{\eu^x + \eu^{-x}}{2},
\qquad
\sinh x = \frac{\eu^x - \eu^{-x}}{2},
\qquad
\tanh x = \frac{\sinh x}{\cosh x}
\]
(\emph{हाइपरबोलिक कोसाइन, साइन, स्पर्शज्या}\index{हाइपरबोलिक कोसाइन, साइन, स्पर्शज्या}). $\cosh $ सम है, $\sinh $ और $\tanh $ अजीब हैं.
\end{definition}

\begin{proposition}[बुनियादी गुण]\label{prop:b1:functions:hyprules}
सभी के लिए $ x, y \in \R $:
\begin{enumerate}
  \item $\cosh^2 x - \sinh^2 x = 1$;
  \item $\cosh' = \sinh $, $\;\sinh' = \cosh $, $\;\tanh' = 1 - \tanh^2
        = \dfrac{1}{\cosh^2}$;
  \item $\cosh(x+y) = \cosh x\cosh y + \sinh x \sinh y $ और
        $\sinh(x+y) = \sinh x \cosh y + \cosh x \sinh y $;
  \item $\sinh $ की सख्ती से बढ़ती हुई आपत्ति है $\R $ पर $\R $;
        $\cosh $ तक सीमित $\R_+$ सख्ती से बढ़ रहा है
        पर आपत्ति $\intco{1}{+\infty}$; $\tanh $ एक सख्ती है
        की बढ़ती आपत्ति $\R $ पर $\intoo{-1}{1}$.
\end{enumerate}
\end{proposition}

\begin{proof}
(1) $\bigl(\frac{\eu^x + \eu^{-x}}{2}\bigr)^2 - \bigl(\frac{\eu^x -
\eu^{-x}}{2}\bigr)^2 = \frac{(\eu^{2x} + 2 + \eu^{-2x}) - (\eu^{2x} -
2 + \eu^{-2x})}{4} = 1$.

(2) परिभाषित सूत्रों को अलग करें; के लिए $\tanh $, भागफल
नियम देता है $\frac{\cosh^2 - \sinh^2}{\cosh^2}$, जो दोनों है
$1 - \tanh^2$ और $\frac{1}{\cosh^2}$ द्वारा (1).

(3) परिभाषाओं का उपयोग करके दाहिनी ओर का विस्तार करें। विस्तार से
पहले सूत्र के लिए:
\[
\cosh x\cosh y + \sinh x\sinh y
= \frac{(\eu^x + \eu^{-x})(\eu^y + \eu^{-y})
+ (\eu^x - \eu^{-x})(\eu^y - \eu^{-y})}{4} ;
\]
आठ उत्पादों में से, चार ``मिश्रित'' उत्पाद
($\eu^{x-y}$, $\eu^{y-x}$) जोड़े में रद्द करें, जबकि
$\eu^{x+y}$ और $\eu^{-(x+y)}$ प्रत्येक दो बार प्रकट होता है:
कुल है $\frac{2\eu^{x+y} + 2\eu^{-(x+y)}}{4} = \cosh(x+y)$.
साइन सूत्र जीवित मिश्रित पदों के समान है
इसके बजाय.

(4) $\sinh' = \cosh \geq 1 > 0$, इसलिए $\sinh $ सख्ती से बढ़ रहा है,
सीमा के साथ $\pm\infty $ (प्रमुख शब्द $\pm\frac{\eu^{\abs
x}}{2}$); आपत्ति \omterm{def:b1:logic:statement}{कथन} फिर मध्यवर्ती से अनुसरण करता है
मूल्य प्रमेय (हाई स्कूल स्तर पर प्रयुक्त; व्यवस्थित उपचार
\cref{ch:b1:continuity}). पर $\R_+$, $\cosh' = \sinh \geq 0$
पर ही लुप्त हो रहा है $0$: से सख्ती से बढ़ रहा है $\cosh 0 = 1$ को
$+\infty $. और $\tanh' > 0$ सीमा के साथ $\pm 1$ पर $\pm\infty $: विस्तार से,
\[
\tanh x = \frac{\eu^x - \eu^{-x}}{\eu^x + \eu^{-x}}
= \frac{1 - \eu^{-2x}}{1 + \eu^{-2x}}
\xrightarrow[x \to +\infty]{} 1 ,
\]
अंश और हर को विभाजित करने के बाद $\eu^x $, और विचित्रता
सीमा देता है $-1$ पर $-\infty $; सख्ती से बढ़ रहा है
समारोह $\tanh $ इसलिए एमएपीएस $\R $ पर $\intoo{-1}1$.
\end{proof}

\begin{remark}[``अतिपरवलयिक'' क्यों?]
बात $(\cosh t, \sinh t)$ शाखा के साथ चलता है $ x > 0$ का
अतिपरवलय $ x^2 - y^2 = 1$ (द्वारा
\cref{prop:b1:functions:hyprules}~(1)), बिल्कुल वैसा ही $(\cos t, \sin
t)$ वृत्त के साथ चलता है $ x^2 + y^2 = 1$. हर गोलाकार पहचान है
एक अतिशयोक्तिपूर्ण सहोदर, जिसके द्वारा नियंत्रित चिह्न \omterm{def:b1:counting:objects}{परिवर्तन} होते हैं
$\sinh^2 \leftrightarrow -\sin^2$.
\end{remark}

\begin{example}[के लिए अतिरिक्त सूत्र $\tanh $]\label{ex:b1:functions:tanhadd}
के दो जोड़ सूत्रों को विभाजित करना
\cref{prop:b1:functions:hyprules}~(3) द्वारा $\cosh x\cosh y $:
\[
\tanh(x + y)
= \frac{\sinh x\cosh y + \cosh x\sinh y}
{\cosh x\cosh y + \sinh x\sinh y}
= \frac{\tanh x + \tanh y}{1 + \tanh x\,\tanh y} ,
\]
स्पर्शरेखा योग सूत्र का अतिपरवलयिक सहोदर --- a के साथ
$+$ जहां त्रिकोणमिति में एक है $-$. एक लाभांश: तब से $\abs{\tanh}
< 1$, दाहिनी ओर एक ``वेग जोड़'' नियम है
कभी नहीं छोड़ता $\intoo{-1}1$: अगर $ u, v \in \intoo{-1}1$ तब
$\frac{u + v}{1 + uv} \in \intoo{-1}1$ साथ ही (लिखें $ u = \tanh
a $, $ v = \tanh b $, वस्तुनिष्ठता से संभव है, और सूत्र पढ़ें
पीछे की ओर)। \omterm{pb:b1:logic:1}{बीजगणितीय} रूप से जाँच करना, बिना अतिशयोक्तिपूर्ण
कार्य, थोड़ा कष्टदायक व्यायाम है; द्वारा पैरामीट्रिज़िंग
$\tanh $ इसे एक पंक्ति बनाता है --- वही रणनीति जो परिपत्र है
फ़ंक्शन यूनिट सर्कल के लिए प्रदान करते हैं।
\end{example}

\begin{proposition}[व्युत्क्रम अतिपरवलयिक फलन]\label{prop:b1:functions:invhyp}
द्वारा दिए गए व्युत्क्रम \cref{prop:b1:functions:hyprules}~(4) है
बंद प्रपत्र:
\[
\operatorname{arsinh} x = \ln\bigl(x + \sqrt{x^2 + 1}\bigr)
\ (x \in \R),
\qquad
\operatorname{arcosh} x = \ln\bigl(x + \sqrt{x^2 - 1}\bigr)
\ (x \geq 1),
\]
\[
\operatorname{artanh} x = \frac 12 \ln\frac{1 + x}{1 - x}
\ (\abs x < 1),
\]
डेरिवेटिव के साथ $\frac{1}{\sqrt{x^2+1}}$, $\frac{1}{\sqrt{x^2-1}}$
($ x > 1$) और $\frac{1}{1 - x^2}$ क्रमश।
\end{proposition}

\begin{proof}
के लिए $\operatorname{arsinh}$: हल करना $ x = \sinh y = \frac{\eu^y -
\eu^{-y}}{2}$. सेटिंग $ u = \eu^y > 0$: $ u^2 - 2xu - 1 = 0$, इसलिए $ u = x
+ \sqrt{x^2 + 1}$ (जड़ $ x - \sqrt{x^2+1}$ नकारात्मक है), और $ y =
\ln(x + \sqrt{x^2+1})$. अन्य दो समान गणनाएँ हैं
($ u^2 - 2xu + 1 = 0$ के लिए $\operatorname{arcosh}$, जड़ रखते हुए
$\geq 1$; के लिए दो-पंक्ति पुनर्व्यवस्था $\operatorname{artanh}$).
व्युत्पन्न: लघुगणकीय अभिव्यक्तियों में अंतर करें, जैसे
\[
\bigl(\ln(x + \sqrt{x^2+1})\bigr)'
= \frac{1 + \frac{x}{\sqrt{x^2+1}}}{x + \sqrt{x^2+1}}
= \frac{1}{\sqrt{x^2 + 1}} . \qedhere
\]
\end{proof}

\begin{example}[काम पर बंद फॉर्म]\label{ex:b1:functions:arcoshtwo}
का समाधान $\cosh t = 2$ साथ $ t \geq 0$ है, बंद द्वारा
रूप, $ t = \operatorname{arcosh} 2 = \ln(2 + \sqrt3) \approx
1.317$; दूसरा उपाय है $-t $, समता से --- और वास्तव में
$\ln(2 - \sqrt3) = \ln\frac1{2 + \sqrt3} = -\ln(2 + \sqrt3)$: द
दो जड़ें $ u = \eu^t $ द्विघात का $ u^2 - 4u + 1 = 0$ हैं
पारस्परिक, उनके उत्पाद के रूप में $1$ (विएटा) मांग करता है। यह छोटा
गणना सामान्य पैटर्न प्रदर्शित करती है: अतिशयोक्तिपूर्ण समीकरण
में द्विघात में परिवर्तित करें $\eu^t $, और समरूपता $ t \mapsto -t $
समरूपता के रूप में प्रकट होता है $ u \mapsto 1/u $ द्विघात का ---
हल करते समय याद रखने योग्य \cref{exo:b1:functions:7}.
\end{example}

\begin{method}[सही आदिम रूप का चयन करना]\label{met:b1:functions:primitive}
एकीकरण के लिए याद रखने योग्य तीन व्युत्पन्न पैटर्न
(\cref{ch:b1:integration}):
\[
\int \frac{\dd x}{1 + x^2} = \arctan x + C,
\qquad
\int \frac{\dd x}{\sqrt{1 - x^2}} = \arcsin x + C,
\qquad
\int \frac{\dd x}{\sqrt{x^2 + 1}} = \operatorname{arsinh} x + C,
\]
और एक सामान्य द्विघात के लिए, वर्ग पूरा करके इन्हें कम करें
और पुनर्स्केलिंग।
\end{method}

\begin{remark}[जहाँ इस अध्याय का प्रयोग किया जाता है]\label{rem:b1:functions:whereused}
यह अध्याय आने वाले सभी विश्लेषणों की कार्यशील शब्दावली है।
के विकास का पैमाना \cref{prop:b1:functions:powerrules} निर्णय करता है
सर्वत्र अभिसरण प्रश्न
\cref{ch:b1:seq,ch:b1:series}; के व्युत्पन्न सूत्र
\cref{prop:b1:functions:arcderiv} और
\cref{prop:b1:functions:invhyp} प्रायः आदिम होते हैं
में जरूरत है \cref{ch:b1:integration}, के जरिए
\cref{met:b1:functions:primitive}; अतिशयोक्तिपूर्ण कार्य
समीकरण के समाधानों को पैरामीट्रिज करें $ y'' = y $ में
\cref{ch:b1:diffeq} बिल्कुल वैसे ही जैसे वृत्ताकार कार्य पैरामीट्रिज करते हैं
के $ y'' = -y $. का लक्षण वर्णन $\exp $ द्वारा $ f' = f $,
$ f(0) = 1$ (\cref{prop:b1:functions:expln}) एक आयामी है
रैखिक अंतर समीकरणों के सिद्धांत का बीज, और
कैटेनरी वक्र $ y = \cosh x $ के समतल वक्रों के बीच लौटता है
\cref{ch:b1:curves}.
\end{remark}

\begin{remark}[इंटरल्यूड: व्युत्क्रम-कार्य कार्यक्रम]\label{rem:b1:functions:interlude}
इस अध्याय ने एक प्रोग्राम को चार बार चलाया: एक फ़ंक्शन को प्रतिबंधित करें
जब तक यह पूरी तरह से एकरस आक्षेप न बन जाए, तब तक व्युत्क्रम का नाम बताइए,
इसके माध्यम से प्रत्येक सूत्र का परिवहन करें। क्या था \emph{इस्तेमाल किया गया} प्रत्येक पर
चरण --- कि एक पर एक सतत सख्ती से एकरस कार्य
अंतराल एक अंतराल पर एक आक्षेप है, और इसका व्युत्क्रम है
सतत, फिर महत्वपूर्ण बिंदुओं से दूर अवकलनीय --- था
हाई स्कूल क्रेडिट पर उधार लिया गया। इसके अंदर कर्ज चुकाया जाता है
आयतन: \cref{ch:b1:continuity} आपत्ति को सिद्ध करता है \omterm{def:b1:logic:statement}{कथन}
(मोनोटोन व्युत्क्रम प्रमेय, मध्यवर्ती मूल्य पर आधारित है
प्रमेय), और \cref{ch:b1:derivative} व्युत्पन्न नियम सिद्ध करता है
$(f^{-1})' = 1/(f' \circ f^{-1})$ जिसने चुपचाप हर एक का उत्पादन किया
का सूत्र \cref{prop:b1:functions:arcderiv} और
\cref{prop:b1:functions:invhyp}. उन अध्यायों को साथ में पढ़ना
$\arcsin $ और $\operatorname{arsinh}$ मन में --- जैसा काम किया
उदाहरण यह सिद्धांत उचित ठहराने के लिए बनाया गया था --- अभीष्ट तरीका है
स्पष्ट गोलाकारता के चारों ओर।
\end{remark}

\section{अभ्यास}

\begin{exercise}[$\star $]\label{exo:b1:functions:1}
कैलकुलेटर के बिना गणना करें: $\arcsin\frac{\sqrt 3}{2}$;
$\;\arccos\bigl(-\frac 12\bigr)$; $\;\arctan(-1)$;
$\;\arcsin\bigl(\sin\frac{5\pi}{6}\bigr)$;
$\;\arccos\bigl(\cos\frac{7\pi}{4}\bigr)$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:functions:2}
की परिभाषा का क्षेत्र बताइये $ f(x) = \arcsin(2x - 1)$ और गणना करें
$ f'$ जहां परिभाषित किया गया है. के लिए वही प्रश्न $ g(x) = \arctan\sqrt{x}$.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:functions:3}
इसे सभी के लिए साबित करें $ x \in \R $: $\cosh x + \sinh x = \eu^x $ और
$(\cosh x + \sinh x)^n = \cosh nx + \sinh nx $ हरएक के लिए $ n \in \Z $
(एक अतिशयोक्तिपूर्ण डी मोइवर सूत्र)।
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:functions:4}
सरल $\cos(2\arcsin x)$ और $\sin(2\arctan x)$ बीजगणित में
की अभिव्यक्तियाँ $ x $.
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:functions:5}
आदेश, बड़े के लिए $ x $, कार्य
$ x^{100}$, $\;\eu^{\sqrt x}$, $\;(\ln x)^{1000}$, $\;\eu^{x/100}$,
$\;x^{\ln x}$, सबसे धीमी से सबसे तेज़ वृद्धि तक, औचित्य के साथ
पर आधारित \cref{prop:b1:functions:powerrules}.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:functions:6}
फ़ंक्शन का अध्ययन करें $ f(x) = \arctan\dfrac{2x}{1 - x^2}$ इसके डोमेन पर:
गणना करें $ f'$, साथ तुलना करें $(2\arctan x)'$, और व्यक्त करें $ f(x)$ के अनुसार $\arctan x $ डोमेन के तीन अंतरालों में से प्रत्येक पर।
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:functions:7}
में हल करें $\R $: $\;\cosh x = 2$; तब $5\cosh x - 4 \sinh x = 3$
(लघुगणक के साथ समाधान व्यक्त करें)। \emph{दूसरे के लिए संकेत:
के साथ सब कुछ लिखें $ u = \eu^x $.}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:functions:8}
पहचान साबित करें $\arctan\frac{1}{2} + \arctan\frac{1}{3} =
\frac{\pi}{4}$, फिर मशीन का सूत्र\index{मशीन का सूत्र}
\[
4\arctan\frac 15 - \arctan\frac{1}{239} = \frac{\pi}{4}.
\]
\emph{संकेत: जोड़ का उपयोग करके दोनों पक्षों की स्पर्शरेखा की गणना करें
सूत्र, और नियंत्रण जिसमें प्रत्येक पक्ष किस अंतराल में स्थित है।}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:functions:9}
इसे सभी के लिए साबित करें $ x \geq 0$: $\;x - \dfrac{x^3}{6} \leq \sin x \leq
x $ \emph{(अंतरों के क्रमिक व्युत्पन्नों का अध्ययन करें)}, और
निष्कर्ष निकालना $\lim_{x \to 0^+} \frac{x - \sin x}{x^3} = \frac 16$ है
प्रशंसनीय --- सीमा स्वयं में स्थापित है
\cref{ch:b1:taylor}.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b1:functions:10}
के लिए $ x \in \intcc{-1}{1}$, \omterm{def:b1:logic:sets}{तय करना}
$ h(x) = \arcsin\bigl(2x\sqrt{1 - x^2}\,\bigr) - 2\arcsin x $. कोई हो सकता है
उम्मीद $ h = 0$ पहचान से $\sin 2y = 2\sin y\cos y $ --- लेकिन
$ h $ है \emph{नहीं} समान रूप से शून्य. गणना करना $ h'$ खुले अंतराल पर
यह कहां मौजूद है, मूल्यांकन करें $ h $ अच्छी तरह से चुने गए बिंदुओं पर, और पूरा दें
का टुकड़ा-टुकड़ा-निरंतर विवरण $ h $ पर $\intcc{-1}{1}$.
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b1:functions:11}
(गुडरमैनियन) चलो $ g(x) = \arctan(\sinh x)$ के लिए $ x \in \R $. साबित करें कि $ g $ एक अजीब, सख्ती से बढ़ती हुई आपत्ति है $\R $ पर
$\intoo{-\frac\pi2}{\frac\pi2}$, वह $ g'(x) = \frac{1}{\cosh x}$,
ओर वो $\tan g(x) = \sinh x $, $\;\sin g(x) = \tanh x $,
$\;\cos g(x) = \frac{1}{\cosh x}$: समारोह $ g $ लिंक परिपत्र
और सम्मिश्र संख्याओं के बिना अतिशयोक्तिपूर्ण त्रिकोणमिति।
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:functions:12}
साबित करें कि
\[
\arctan 1 + \arctan 2 + \arctan 3 = \pi .
\]
\emph{संकेत: गणना करें $\arctan 2 + \arctan 3$ सबसे पहले, स्थानीयकरण
के रूप में योग करें \cref{ex:b1:functions:arctansum}; सावधान रहें कि
यहाँ योग सूत्र का हर ऋणात्मक है।}
\end{exercise}

\section{समस्या: समीकरण $ x^y = y^x $}

\begin{problem}\label{pb:b1:functions:1}
धनात्मक संख्याओं के कौन से जोड़े संतुष्ट करते हैं $ x^y = y^x $? हर कोई जानता है
एक आकस्मिक दिखने वाला उदाहरण, $2^4 = 4^2 = 16$; यह समस्या दिखती है
इसके बारे में कुछ भी आकस्मिक नहीं है। पूरा समीकरण नियंत्रित है
एकल फ़ंक्शन की विविधताओं द्वारा
\[
f(t) = \frac{\ln t}{t} \qquad (t > 0),
\]
जिसके अध्ययन से निष्कर्ष निकलता है: सम्पूर्ण समाधान \omterm{def:b1:logic:sets}{तय करना} (एक विकर्ण प्लस एक
के माध्यम से घुमावदार शाखा $(\eu, \eu)$), का एक तर्कसंगत पैरामीटरीकरण
शाखा, तथ्य यह है कि $(2, 4)$ इसका है \emph{केवल} पूर्णांक बिंदु
और इसके सभी तर्कसंगत बिंदुओं का वर्गीकरण --- प्लस, जैसा
लाभांश, की तुलना $\eu^\pi $ साथ $\pi^\eu $ और
का एकस्वर अभिसरण $\bigl(1 + \frac1n\bigr)^n $ को $\eu $.
\index{शक्ति समारोह}

\medskip
\noindent\textbf{Part I --- The function $ f(t) = \ln t / t $.}
\begin{enumerate}
  \item उसे उचित ठहराओ $ f $ पर भिन्न है $\intoo0{+\infty}$,
        गणना करें $ f'$, और विविधता तालिका बनाएं: $ f $
        पर सख्ती बढ़ती है $\intoc0\eu $, सख्ती से घटता है
        $\intco\eu{+\infty}$, अधिकतम के साथ $ f(\eu) = \frac1\eu $.
  \item की सीमा निर्धारित करें $ f $ पर $0^+$ और $+\infty $
        (\cref{prop:b1:functions:powerrules}), का चिन्ह $ f $
        (नकारात्मक पर $\intoo01$, शून्य पर $1$, सकारात्मक परे),
        और ग्राफ को स्केच करें।
  \item प्रत्यक्ष गणना द्वारा जाँच करें कि $ f(2) = f(4)$. (इसे रखें
        मन में समानता: सारी समस्या इसी से उत्पन्न होती है।)
  \item होने देना $ c \in \R $. के मान के अनुसार चर्चा करें $ c $, द
        के समाधानों की संख्या $ f(t) = c $: बिल्कुल एक के लिए $ c \leq
        0$; बिल्कुल दो (एक इंच) $\intoo1\eu $, एक में
        $\intoo\eu{+\infty}$) के लिए $0 < c < \frac1\eu $; बिल्कुल एक
        के लिए $ c = \frac1\eu $; के लिए कोई नहीं $ c > \frac1\eu $.
  \item उसका निष्कर्ष निकालो $\eu^x > x^\eu $ हरएक के लिए $ x > 0$ साथ $ x \neq
        \eu $, और विशेष रूप से किसका निपटान करें $\eu^\pi $ और
        $\pi^\eu $ बड़ा है.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The equation and its curve.} इस हिस्से में $ x, y > 0$.
\begin{enumerate}[resume]
  \item बताते हैं कि $ x^y = y^x \iff f(x) = f(y)$.
  \item समाधान की संरचना निकालें \omterm{def:b1:logic:sets}{तय करना}: सभी विकर्ण
        जोड़े $(x, x)$; और यह \emph{गैरतुच्छ} जोड़े ($ x \neq
        y $), जो संतुष्ट करता है: दोनों निर्देशांक हैं $> 1$, और एक
        में निहित है $\intoo1\eu $ जबकि दूसरा अंदर है
        $\intoo\eu{+\infty}$.
  \item गैर-तुच्छ जोड़ियों को पैरामीट्रिज़ करें: लेखन $ y = tx $ साथ $ t
        > 0$, $ t \neq 1$, बताते हैं कि $ x^y = y^x $ बल
        \[
        x(t) = t^{\frac1{t-1}},
        \qquad
        y(t) = t\,x(t) = t^{\frac{t}{t-1}},
        \]
        और इसके विपरीत ऐसी प्रत्येक जोड़ी एक समाधान है।
  \item सत्यापित करें कि $ t = 2$ देता है $(2, 4)$, और समरूपता सिद्ध करें
        $ x(1/t) = y(t)$, $ y(1/t) = x(t)$: पैरामीटर को उलटना
        दो निर्देशांकों की अदला-बदली करता है।
  \item की सीमा निर्धारित करें $ x(t)$ और $ y(t)$ जैसा $ t \to 1$
        (दोनों की प्रवृत्ति होती है $\eu $), जैसा $ t \to +\infty $ ($ x \to 1$, $ y
        \to +\infty $) और के रूप में $ t \to 0^+$ ($ x \to +\infty $, $ y \to
        1$). परिणामी शाखा का वर्णन करें: एक वक्र स्पर्शोन्मुख
        पंक्तियाँ $ x = 1$ और $ y = 1$, पर विकर्ण को पार करना
        $(\eu, \eu)$.
  \item बताते हैं कि $ t \mapsto x(t)$ पर सख्ती से कमी आ रही है
        $\intoo1{+\infty}$. \emph{(अध्ययन $ h(t) = 1 - \frac1t - \ln
        t $, जिसका चिह्न व्युत्पन्न को नियंत्रित करता है $\ln x(t) =
        \frac{\ln t}{t - 1}$.)}
  \item निष्कर्ष निकालें कि गैर-तुच्छ समाधान सख्ती से परिभाषित करते हैं
        घटती आपत्ति $\varphi \colon \intoo1\eu \to
        \intoo\eu{+\infty}$ साथ $ x^{\varphi(x)} = \varphi(x)^x $,
        ओर वो $\varphi $ शाखा का एक समावेश है:
        $\varphi(\varphi(x)) = x $ जहाँ भी दोनों पक्षों को परिभाषित किया गया है।
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Integer and rational points.}
\begin{enumerate}[resume]
  \item साबित करें कि $(2, 4)$ और $(4, 2)$ एकमात्र गैरतुच्छ हैं
        \emph{पूर्णांक} के समाधान $ x^y = y^x $.
  \item के लिए $ n \in \N^*$, के साथ पैरामीट्रिजेशन लागू करें $ t = 1 +
        \frac1n $ और दिखाओ
        \[
        x_n = \Bigl(1 + \frac1n\Bigr)^{n},
        \qquad
        y_n = \Bigl(1 + \frac1n\Bigr)^{n+1}
        \]
        एक गैरतुच्छ है \emph{तर्कसंगत} हर एक के लिए समाधान $ n $,
        साथ $(x_1, y_1) = (2, 4)$.
  \item इसके विपरीत, चलो $(x, y)$ एक गैर-तुच्छ तर्कसंगत समाधान बनें
        के साथ $ y > x $, और लिखा $ t = y/x = 1 + \frac rs $ सबसे कम में
        शर्तें ($ r, s \in \N^*$). का उपयोग करते हुए $ x = t^{s/r}$ और स्वीकार कर रहा हूँ
        अभाज्य गुणनखंडन की विशिष्टता (से परिचित)।
        स्कूल; में साबित हुआ \cref{ch:b1:arith}), बताते हैं कि $ x $
        तर्कसंगत बल दोनों $ s $ और $ s + r $ होना $ r $-वें शक्तियां
        पूर्णांकों का.
  \item द्विपद प्रमेय से दिखाएँ कि $ a^r - b^r = r $ नहीं है
        पूर्णांक समाधान $ a > b \geq 1$ कब $ r \geq 2$, और
        निष्कर्ष निकालें: के तर्कसंगत समाधान $ x^y = y^x $ हैं
        बिल्कुल जोड़े $(x_n, y_n)$ प्रश्न 14 का (और उनका)
        अदला-बदली)।
  \item मामले का सत्यापन करें $ n = 2$ संख्यात्मक रूप से: गणना करें $ f(9/4)$ और
        $ f(27/8)$ दशमलव के चार स्थानों तक जाएँ और जाँचें कि वे सहमत हैं।
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Dividends.}
\begin{enumerate}[resume]
  \item साबित करने के लिए प्रश्न 7 और 11 का उपयोग करें, आगे कुछ नहीं
        गणना, कि अनुक्रम $ x_n = \bigl(1 +
        \frac1n\bigr)^n $ के साथ सख्ती से बढ़ रहा है $ x_n < \eu $,
        वह $ y_n = \bigl(1 + \frac1n\bigr)^{n+1}$ सख्ती से है
        के साथ घट रहा है $ y_n > \eu $, और वह दोनों एक हो जाते हैं
        $\eu $. (पैरामीटर $ t_n = 1 + \frac1n $ तक कम हो जाता है
        $1$.)
  \item के लिए सामान्य तुलना नियम स्थापित करें $1 < a < b $:
        अगर $\eu \leq a $ तब $ a^b > b^a $; अगर $ b \leq \eu $ तब
        $ a^b < b^a $; और दो उदाहरणों द्वारा दिखाएँ $(2, 3)$ और
        $(2, 5)$ वह मिश्रित मामले में $ a < \eu < b $ दोनों
        परिणाम वास्तव में घटित होते हैं।
  \item मान लीजिए कि शामिल है $\varphi $ प्रश्न 12 का है
        पर भिन्न $\eu $ (यह है)। अंतर करें
        पहचान $\varphi(\varphi(x)) = x $ पर $ x = \eu $ और निष्कर्ष निकालें
        $\varphi'(\eu) = -1$: शाखा विकर्ण को पार करती है
        इसके समकोण.
  \item के साथ अद्वितीय समाधान युग्म खोजें $ y = 3x $, बंद में
        फॉर्म बनाएं, और इसे चार दशमलव तक संख्यात्मक रूप से जांचें $ f $.
  \item संख्याओं के बीच $\sqrt2$, $\sqrt[3]3$, $\sqrt[4]4$,
        $\sqrt[5]5$, सबसे बड़ा निर्धारित करें और दो की पहचान करें
        वह बराबर हैं. \emph{(तुलना करना $ n^{1/n} =
        \eu^{f(n)}$.)}
\end{enumerate}

\noindent\textbf{Part V --- Synthesis.}
\begin{enumerate}[resume]
  \item संपूर्ण समाधान बताएं \omterm{def:b1:logic:sets}{तय करना} का $ x^y = y^x $ में
        क्वार्टर-प्लेन $ x, y > 0$ --- विकर्ण प्लस शाखा, उनका
        चौराहा $(\eu, \eu)$, अनंतस्पर्शी, पूर्णांक
        बिंदु $(2, 4)$, पर तर्कसंगत बिंदु जमा हो रहे हैं
        $(\eu, \eu)$ --- एक ऐसे रूप में जिसे आप स्मृति से स्केच कर सकते हैं।
  \item वास्तव में समस्या का उपयोग कहां हुआ: (i) द \omterm{prop:b1:functions:powerrules}{विकास
        तुलना} का \cref{prop:b1:functions:powerrules}; (ii)
        मध्यवर्ती मूल्य प्रमेय (आक्षेप के माध्यम से)।
        \omterm{def:b1:logic:statement}{कथन}); (iii) प्राइम की स्वीकृत विशिष्टता
        गुणनखंडन? एक-एक वाक्य.
  \item नैतिक, एक संक्षिप्त अनुच्छेद में: एक एकल भिन्नता तालिका
        दो अज्ञात में एक समीकरण हल किया, उसका वर्गीकरण किया
        तर्कसंगत बिंदु, और के एकरस अभिसरण को साबित किया
