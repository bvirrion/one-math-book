\chapter{गिनती}\label{ch:b1:counting}

गिनती परिमित समुच्चय प्राथमिक लगता है --- और शीघ्र ही सूक्ष्म हो जाता है।
यह अध्याय परिभाषित करता है \omterm{def:b1:counting:card}{प्रमुखता} ठीक से (आक्षेपों के माध्यम से, में
की भावना \cref{ch:b1:logic}), मुट्ठी भर गिनती स्थापित करता है
ऐसे सिद्धांत जिनसे हर चीज़ अनुसरण करती है, और शास्त्रीयता प्राप्त करती है
गिनती: सूचियाँ, क्रमपरिवर्तन, उपसमुच्चय, \omterm{def:b1:counting:objects}{द्विपद गुणांक}.

\section{परिमित समुच्चयों की प्रमुखता}

\begin{definition}[परिमित समुच्चय, प्रमुखता]\label{def:b1:counting:card}
के लिए $n \in \N^*$, लिखना $\intint{1}{n} = \{1, 2, \dots,
n\}$. ए \omterm{def:b1:logic:sets}{तय करना} $ E $ है \emph{परिमित}\index{परिमित} कब $ E = \emptyset $
या वहां से कोई आपत्ति है $\intint{1}{n}$ पर $ E $ के लिए
कुछ $ n \in \N^*$; यह $ n $ निराला है (\cref{thm:b1:counting:welldef})
और है \emph{प्रमुखता}\index{प्रमुखता} का $ E $, लिखा हुआ
$\abs{E}$ (साथ $\abs{\emptyset} = 0$).
\end{definition}

\begin{theorem}[कार्डिनलिटी अच्छी तरह से परिभाषित है]\label{thm:b1:counting:welldef}
अगर $ m \neq n $, कोई आपत्ति नहीं है $\intint{1}{m}$
पर $\intint{1}{n}$. अधिक सटीक रूप से, यदि $ m > n $ वहाँ है
से कोई \omterm{def:b1:logic:inj}{इंजेक्शन} नहीं $\intint{1}{m}$ में $\intint{1}{n}$.
\end{theorem}

\begin{proof}
हम प्रेरण द्वारा सिद्ध करते हैं $ n $  \omterm{def:b1:logic:statement}{कथन}: \emph{सभी के लिए $ m > n $,
कोई \omterm{def:b1:logic:inj}{इंजेक्शन} नहीं है $\intint{1}{m} \to \intint{1}{n}$}. के लिए $ n = 0$ लक्ष्य खाली है और $ m \geq 1$: नहीं \omterm{def:b1:logic:map}{नक्शा}
बिल्कुल मौजूद है. मान लीजिये \omterm{def:b1:logic:statement}{कथन} के लिए $ n $, और मान लीजिए $ f \colon
\intint{1}{m} \to \intint{1}{n+1}$ एक है
के साथ \omterm{def:b1:logic:inj}{इंजेक्शन} $ m > n + 1$. यदि मान $ n + 1$ प्राप्त नहीं हुआ है, $ f $
में एक \omterm{def:b1:logic:inj}{इंजेक्शन} है $\intint{1}{n}$, का खंडन करते हुए
प्रेरण परिकल्पना. अन्यथा $ f(a) = n + 1$ बिल्कुल एक के लिए $ a $;
बदलना $ f(a)$ और $ f(m)$ (औपचारिक रूप से: स्थानांतरण के साथ रचना करें
दो मान), ताकि नया \omterm{def:b1:logic:inj}{इंजेक्शन} $ g $ है $ g(m) = n + 1$.
फिर का प्रतिबंध $ g $ को $\intint{1}{m-1}$ एक है
में \omterm{def:b1:logic:inj}{इंजेक्शन} $\intint{1}{n}$ साथ $ m - 1 > n $ ---
फिर से विरोधाभास.
\end{proof}

\begin{corollary}[कबूतर का छेद सिद्धांत]\label{cor:b1:counting:pigeonhole}
अगर $\abs{E} > \abs{F}$, नहीं \omterm{def:b1:logic:map}{नक्शा} $ f \colon E \to F $ है \omterm{def:b1:logic:inj}{इंजेक्शन}: कुछ
के दो तत्व $ E $ उनकी छवि साझा करें\index{कबूतरखाने का सिद्धांत}.
\end{corollary}

\begin{proof}
लिखना $\abs E = m $, $\abs F = n $ साथ $ m > n $, और आक्षेप चुनें
$ u \colon \intint1m \to E $ और $ v \colon F \to \intint1n $. अगर $ f $
थे \omterm{def:b1:logic:inj}{इंजेक्शन}, $ v \circ f \circ u $ से एक \omterm{def:b1:logic:inj}{इंजेक्शन} होगा
$\intint1m $ में $\intint1n $ (\omterm{def:b1:logic:inj}{इंजेक्शन} की एक संरचना,
\cref{prop:b1:logic:comp}), विरोधाभासी
\cref{thm:b1:counting:welldef}.
\end{proof}

\begin{remark}[अंतराल: प्रमेय के प्रमाण में अदला-बदली क्यों?]\label{rem:b1:counting:swapinterlude}
का प्रमाण \cref{thm:b1:counting:welldef} शामिल है
अध्याय का पहला वास्तव में चतुर कदम, धीरे-धीरे दोबारा दोहराने लायक।
बाधा: प्रेरण परिकल्पना को लागू करना जो कोई चाहता है
अंतिम बिंदु हटाएँ $ m $ स्रोत का \emph{और} अंतिम बिंदु
$ n+1$ लक्ष्य का, लेकिन $ f $ कोई और बिंदु भेज सकते हैं $ a $ को $ n
+ 1$, और फिर लक्ष्य बिंदु को हटाने से नुकसान होता है \omterm{def:b1:logic:map}{नक्शा}
अन्यत्र. इलाज: रचना करें $ f $ के स्थानांतरण के साथ
दो \emph{मान} $ f(a)$ और $ f(m)$ --- का एक आक्षेप
लक्ष्य, इसलिए \omterm{def:b1:logic:inj}{इंजेक्शन} को संरक्षित किया जाता है --- जिसके बाद
परेशानी भरा मूल्य $ n + 1$ हानिरहित स्थिति में बैठता है $ m $, और
दोनों विलोपन साफ़ हैं. यह ``पहले सामान्य करें, फिर काटें''
पैटर्न पुनरावर्ती होता है: इस प्रकार विक्षोभ पुनरावृत्ति पुनर्निर्देशित होती है
$\sigma^{-1}(n+1)$ इस अध्याय की सप्ताहांत समस्या में, और
कैसे क्रमपरिवर्तन भर में पैच किए गए हैं
\cref{ch:b1:structures}सममित समूह पर समस्या.
\end{remark}

\begin{proposition}[इंजेक्शन, अनुमान और कार्डिनैलिटी]\label{prop:b1:counting:injsur}
होने देना $ E, F $ होना परिमित समुच्चय साथ $\abs{E} = \abs{F}$, और $ f \colon E
\to F $. तब
\[
f \text{ injective} \iff f \text{ surjective} \iff f \text{ bijective}.
\]
\end{proposition}

\begin{proof}
कल्पना करना $ f $ \omterm{def:b1:logic:inj}{इंजेक्शन}. तब $ f $ से एक आक्षेप है $ E $ पर $ f(E)$,
इसलिए $\abs{f(E)} = \abs{E} = \abs{F}$. अगर $ f(E)$ एक बिंदु चूक गया $ y_0$ का
$ F $, तब $ f $ का एक \omterm{def:b1:logic:inj}{इंजेक्शन} होगा $ E $ में $ F \setminus \{y_0\}$,
ए \omterm{def:b1:logic:sets}{तय करना} का \omterm{def:b1:counting:card}{प्रमुखता} $\abs{F} - 1 < \abs{E}$ --- द्वारा असंभव
कबूतरखाने का सिद्धांत. इसलिए $ f(E) = F $: $ f $ है \omterm{def:b1:logic:inj}{विशेषण}, इस तरह
\omterm{def:b1:logic:inj}{द्विभाजित}.

कल्पना करना $ f $ \omterm{def:b1:logic:inj}{विशेषण}. प्रत्येक के लिए चुनें $ y \in F $ एक \omterm{def:b1:logic:map}{पूर्वछवि} $ s(y)
\in E $; तब $ f \circ s = \mathrm{id}_F $, इसलिए $ s $ है \omterm{def:b1:logic:inj}{इंजेक्शन}
(\cref{prop:b1:logic:comp}). पिछले पैराग्राफ द्वारा लागू किया गया $ s $
(कार्डिनैलिटी बराबर हैं), $ s $ है \omterm{def:b1:logic:inj}{द्विभाजित}. से $ f \circ s =
\mathrm{id}_F $ हम पाते हैं $ f = \mathrm{id}_F \circ s^{-1} = s^{-1}$, इसलिए
$ f $ है \omterm{def:b1:logic:inj}{द्विभाजित}. अंततः, ए \omterm{def:b1:logic:inj}{द्विभाजित} \omterm{def:b1:logic:map}{नक्शा} परिभाषा के अनुसार दोनों हैं
\omterm{def:b1:logic:inj}{इंजेक्शन} और \omterm{def:b1:logic:inj}{विशेषण}, जो निहितार्थ के चक्र को बंद कर देता है।
\end{proof}

\begin{example}[परिमितता आवश्यक है]\label{ex:b1:counting:finiteonly}
एक पर \emph{परिमित} \omterm{def:b1:logic:sets}{तय करना}, \cref{prop:b1:counting:injsur} एक शक्तिशाली है
शॉर्टकट: कोई भी \omterm{def:b1:logic:inj}{इंजेक्शन} \omterm{def:b1:logic:map}{नक्शा} से $ E $ स्वयं के लिए स्वचालित रूप से एक है
\omterm{def:b1:counting:objects}{परिवर्तन} का $ E $ --- वस्तुपरकता का आधा हिस्सा मुफ़्त में मिलता है। दोनों
निहितार्थ अनंत पर गिरते हैं सेट: $ n \mapsto n + 1$ है
\omterm{def:b1:logic:inj}{इंजेक्शन} से $\N $ को $\N $ लेकिन चूक जाता है $0$, और यह \omterm{def:b1:logic:map}{नक्शा} $\N \to \N $
भेज रहा हूँ $0 \mapsto 0$ और $ n \mapsto n - 1$ के लिए $ n \geq 1$ है
\omterm{def:b1:logic:inj}{विशेषण} लेकिन नहीं \omterm{def:b1:logic:inj}{इंजेक्शन}. जब भी यह प्रस्ताव लागू किया जाता है,
परिमितता परिकल्पना वास्तविक कार्य कर रही है --- एक विषय जो कि
सप्ताहांत की समस्या \cref{ch:b1:logic} दूसरी ओर से अन्वेषण करता है,
जहां अनंत सेट क्या वास्तव में वही लोग ऐसा स्वीकार कर रहे हैं स्व-मानचित्र.
\end{example}

\begin{example}[आधा काम, मुफ़्त]\label{ex:b1:counting:mod7}
पर विचार करें \omterm{def:b1:logic:map}{नक्शा} $ f $ पर $\{0, 1, \dots, 6\}$ भेज रहा हूँ $ k $ को
का शेष $3k $ द्वारा विभाजित करने पर $7$; इसके मानों की तालिका है
\[
0,\ 3,\ 6,\ 2,\ 5,\ 1,\ 4 .
\]
है $ f $ एक आपत्ति? केवल \omterm{def:b1:logic:inj}{इंजेक्शन} ही पर्याप्त है
(\cref{prop:b1:counting:injsur}): अगर $3k $ और $3k'$ समान है
शेष, $7$ विभाजित $3(k - k')$, और तबसे $7$ प्रधान है और
विभाजित नहीं करता $3$, यह विभाजित करता है $ k - k'$ (यूक्लिड का लेम्मा, यहां प्रयोग किया जाता है
यहां हाई स्कूल स्तर और साबित किया \cref{ch:b1:arith}); साथ
$\abs{k - k'} \leq 6$ यह बल देता है $ k = k'$. प्रक्षेपात्मकता आती है
निःशुल्क --- हल करने की कोई आवश्यकता नहीं $3k \equiv c $ प्रत्येक के लिए $ c $, हालांकि
तालिका पुष्टि करती है कि प्रत्येक मान ठीक एक बार दिखाई देता है। शॉर्टकट है
एक वर्कहॉर्स: यह मॉड्यूलर की व्युत्क्रमणीयता को सिद्ध करता है
गुणन (\cref{ch:b1:arith}), युग्मन को शक्ति प्रदान करता है
विल्सन का प्रमेय, और रैखिक बीजगणित में ``एएन'' के रूप में लौटता है
एक परिमित-आयामी स्थान की एंडोमोर्फिज्म है \omterm{def:b1:logic:inj}{इंजेक्शन} आईएफएफ
\omterm{def:b1:logic:inj}{विशेषण}'' (\cref{ch:b1:findim}).
\end{example}

\section{गिनती के सिद्धांत}

\begin{proposition}[योग और उत्पाद नियम]\label{prop:b1:counting:rules}
होने देना $ E, F $ होना परिमित समुच्चय.
\begin{enumerate}
  \item अगर $ E \cap F = \emptyset $, तब $\abs{E \cup F} = \abs{E} +
        \abs{F}$; अधिक आम तौर पर, ए के लिए \omterm{thm:b1:logic:partition}{PARTITION} का $ E $ टुकड़ों में
        $ E_1, \dots, E_k $, $\abs{E} = \sum_i \abs{E_i}$.
  \item सामान्य तौर पर, $\abs{E \cup F} = \abs{E} + \abs{F} - \abs{E \cap
        F}$.
  \item $\abs{E \times F} = \abs{E} \times \abs{F}$.
  \item  \omterm{def:b1:logic:sets}{तय करना} $ F^E $ के सभी एमएपीएस से $ E $ को $ F $ संतुष्ट
        $\abs{F^E} = \abs{F}^{\abs{E}}$.
  \item $\abs{\mathcal{P}(E)} = 2^{\abs{E}}$.
\end{enumerate}
\end{proposition}

\begin{proof}
(1) सिलसिलेवार गणनाएँ: यदि $ E = \{x_1, \dots, x_m\}$ और $ F =
\{y_1, \dots, y_n\}$ फिर, बिना किसी दोहराव के $ x_1, \dots, x_m, y_1,
\dots, y_n $ विश्लेषण करता है $ E \cup F $ दोहराव के बिना (असंगति)।
प्रेरण इसका विस्तार करता है $ k $ टुकड़े।

(2) $ E \cup F $ का असंयुक्त संघ है $ E $ और $ F \setminus E $, और
$ F $ का असंयुक्त संघ है $ F \cap E $ और $ F \setminus E $; इसलिए
$\abs{E \cup F} = \abs{E} + \abs{F \setminus E} = \abs{E} + \abs{F} -
\abs{E \cap F}$.

(3) $ E \times F $ असंयुक्त संघ है, खत्म $ x \in E $, की सेट
$\{x\} \times F $, की प्रत्येक \omterm{def:b1:counting:card}{प्रमुखता} $\abs{F}$; लागू करें (1).

(4) ए \omterm{def:b1:logic:map}{नक्शा} से $ E = \{x_1, \dots, x_m\}$ को $ F $ बिल्कुल पसंद है
का $ m $-टुपल $(f(x_1), \dots, f(x_m)) \in F^m $; यह
पत्राचार एक आक्षेप है, और $\abs{F^m} = \abs{F}^m $ द्वारा (3) और
प्रेरण.

(5) के उपसमुच्चय $ E $ अनुरूप वस्तुनिष्ठ रूप से को एमएपीएस $ E \to \{0, 1\}$
(भेजना $ A $ इसके सूचक कार्य के लिए); लागू करें (4).
\end{proof}

\begin{example}[पूरक गिनती]\label{ex:b1:counting:complement}
कितने $4$-डिजिट पिन कोड (अंक $0$--$9$, आदेश मायने रखता है,
पुनरावृत्ति की अनुमति है) शामिल है \emph{कम से कम एक} बार-बार अंक?
उन्हें गिनने का सीधा अर्थ है मामलों की बाजीगरी करना ``बिल्कुल एक जोड़ी,
दो जोड़े, एक तिगुना, एक चौगुना'' --- पांच अतिव्यापी
विन्यास. इसके बजाय पूरक की गणना करें: सभी कोड संख्या
$10^4 = 10\,000$ (उत्पाद नियम), चार अलग-अलग कोड
अंक संख्या $10 \times 9 \times 8 \times 7 = 5\,040$
($4$-व्यवस्थाएँ), तो उत्तर है
\[
10^4 - 10 \cdot 9 \cdot 8 \cdot 7 = 10\,000 - 5\,040 = 4\,960 .
\]
सभी पिनों में से लगभग आधे एक अंक दोहराते हैं। अंतर्दृष्टि: जब भी ए
गिनती को ``कम से कम'' या ``सभी नहीं'' के साथ परिभाषित किया गया है, प्रयास करें
पहले पूरक करें --- योग नियम इसकी गारंटी देता है
$\abs{A} = \abs{E} - \abs{\overline A}$, और पूरक है
प्रायः एक ही स्वच्छ विन्यास।
\end{example}

\begin{example}[जालीदार रास्ते]\label{ex:b1:counting:paths}
कोने से सबसे छोटे रास्ते गिनें $(0,0)$ कोने में
$(4, 3)$ एक ग्रिड का, केवल एक कदम दाएँ (R) या एक कदम ऊपर जाना
(यू) एक समय में. ऐसा हर रास्ता सटीक रूप से चलता है $7$ चरण, जिनमें से
$4$ आर और हैं $3$ क्या आप; इसके विपरीत, लंबाई का कोई भी शब्द $7$ में
चार आर के साथ अक्षर आर, यू बिल्कुल एक पथ का वर्णन करते हैं। रास्ते
इसलिए पत्र-व्यवहार करें वस्तुनिष्ठ रूप से पदों के चयन के लिए
आर के:
\[
\binom{7}{4} = 35 .
\]
अंतर्दृष्टि है \emph{एन्कोडिंग}: गिनती तुच्छ हो गई
पल-पल प्रत्येक पथ का एक शब्द में अनुवाद किया गया, अर्थात।\ का एक उपसमुच्चय
पद --- इस नारे का एक और उदाहरण कि एक सही गिनती
एक प्रच्छन्न आक्षेप है (\cref{met:b1:counting:which}).
\end{example}

\section{सूचियाँ, क्रमपरिवर्तन, उपसमुच्चय}

\begin{definition}[व्यवस्थाएँ, क्रमपरिवर्तन, संयोजन]\label{def:b1:counting:objects}
होने देना $ E $ एक हो \omterm{def:b1:logic:sets}{तय करना} साथ $\abs{E} = n $ और चलो $0 \leq k \leq n $.
\begin{itemize}
  \item ए \emph{$ k $-व्यवस्था} का $ E $ एक \omterm{def:b1:logic:inj}{इंजेक्शन} $ k $-टुपल का
        के तत्व $ E $ (दोहराव के बिना एक आदेशित चयन);
  \item ए \emph{परिवर्तन}\index{परिवर्तन} का $ E $ एक आक्षेप है
        से $ E $ स्वयं के लिए --- समकक्ष, एक $ n $-व्यवस्था;
  \item ए \emph{$ k $-संयोजन} का एक उपसमुच्चय है $ E $ साथ $ k $ तत्व
        (दोहराव के बिना एक अव्यवस्थित चयन)। उनकी संख्या है
        लिखा हुआ $\binom{n}{k}$, पढ़ें ``$ n $ चुनना $ k $''
        \index{द्विपद गुणांक}.
\end{itemize}
\end{definition}

\begin{theorem}[तीन मायने रखता है]\label{thm:b1:counting:counts}
साथ $ n = \abs{E}$ और $0 \leq k \leq n $:
\begin{enumerate}
  \item की संख्या $ k $-की व्यवस्था $ E $ है $ n (n-1) \cdots
        (n-k+1) = \dfrac{n!}{(n-k)!}$;
  \item की संख्या क्रमपरिवर्तन का $ E $ है $ n!$;
  \item $\dbinom{n}{k} = \dfrac{n!}{k!\,(n-k)!}$.
\end{enumerate}
\end{theorem}

\begin{proof}
(1) पहला निर्देशांक चुनें ($ n $ तरीके), फिर दूसरा ($ n - 1$
शेष विकल्प), \dots, फिर $ k $-वां ($ n - k + 1$ विकल्प)।
औपचारिक रूप से, शामिल करें $ k $. के लिए $ k = 1$ वहाँ हैं $ n $ एक कार्यकाल
\omterm{def:b1:logic:inj}{इंजेक्शन} टुपल्स के लिए गिनती मान लें $ k - 1$. प्रत्येक
$ k $-व्यवस्था $(x_1, \dots, x_k)$ बिल्कुल एक से प्राप्त होता है
$(k-1)$-व्यवस्था---इसकी काट-छाँट $(x_1, \dots, x_{k-1})$
--- बाहर अंतिम निर्देशांक जोड़कर $\{x_1, \dots,
x_{k-1}\}$, जिसके लिए बिल्कुल $ n - (k - 1)$ मान उपलब्ध हैं.
 $ k $-व्यवस्थाओं को इस प्रकार काट-छाँट द्वारा विभाजित किया जाता है
सामान्य आकार की कक्षाएं $ n - k + 1$ द्वारा अनुक्रमित
$(k-1)$-व्यवस्थाएं, और योग नियम देता है
\[
\frac{n!}{(n-k+1)!}\;(n - k + 1) = \frac{n!}{(n-k)!} .
\]

(2) (1) के साथ है $ k = n $.

(3) प्रत्येक $ k $-सबसेट आदेश में $ k!$ विशिष्ट $ k $-व्यवस्थाएं, और
प्रत्येक $ k $-व्यवस्था बिल्कुल एक उपसमुच्चय से उत्पन्न होती है: इसलिए
$\frac{n!}{(n-k)!} = \binom nk \cdot k!$.
\end{proof}

\begin{example}[गोल मेज़: समरूपता द्वारा भागफल]\label{ex:b1:counting:roundtable}
कितने तरीकों से कर सकते हैं $ n $ मेहमान एक गोल मेज के चारों ओर बैठते हैं, दो
जब प्रत्येक अतिथि के पास समान बाएँ और समान सीटें हों तो सीटें समान होती हैं
सही पड़ोसी --- यानी\ घूर्णन तक? प्रत्येक गोलाकार बैठने की जगह
बिल्कुल मेल खाता है $ n $ रैखिक सीटें (किसी भी स्थान पर वृत्त काटें
का $ n $ स्थान), तो $ n!$ समूहों में रैखिक क्रम ध्वस्त हो जाते हैं
का $ n $:
\[
\frac{n!}{n} = (n-1)! \quad\text{circular seatings.}
\]
समान रूप से: एक विशिष्ट अतिथि को कहीं भी बैठाएं (हत्या करें)।
घूर्णी स्वतंत्रता), फिर शेष का आदेश दें $ n - 1$ मेहमान
दक्षिणावर्त. के लिए $ n = 6$: $120$ टेबल. दो समाधान दर्शाते हैं
अत्यधिक गिनती के दो मानक इलाज: सटीक से विभाजित करें
दोहराव की संख्या, या \emph{समरूपता तोड़ो} एक को पिन करके
आपत्ति नीचे. दोनों को समूह-पुनरावृत्ति आकार की आवश्यकता होती है
प्रत्येक कॉन्फ़िगरेशन के लिए समान --- जो सूत्र का प्रमाण है
$\binom nk = \frac{n!}{k!\,(n-k)!}$ ऊपर भी प्रयोग किया जाता है, साथ में
$ k!$ की जगह $ n $.
\end{example}

\begin{example}[एक बाधा जोड़ना]\label{ex:b1:counting:constraint}
गोलमेज को जारी रखना: बीच में $(n-1)!$ की तालिकाएँ $ n \geq
3$ अतिथि, कितनी सीट दो दिए गए अतिथि $ A $ और $ B $ \emph{अलग}
(आसन्न नहीं)? पूरक की गणना करें. टेबल्स कहाँ $ A $ और $ B $
एक साथ बैठें: उन्हें एक ही ब्लॉक में चिपका दें --- $ n - 1$ वस्तुएं
मेज के चारों ओर, यानी\ $(n-2)!$ वृत्ताकार व्यवस्था --- फिर
जोड़ी को उसके ब्लॉक के अंदर ऑर्डर करें ($2$ तौर तरीकों): $2\,(n-2)!$ आसन्न
टेबल. इस तरह
\[
(n-1)! - 2\,(n-2)! = (n-2)!\,\bigl((n - 1) - 2\bigr)
= (n-3)\,(n-2)!
\]
टेबल उन्हें अलग रखती हैं। स्वच्छता जांच: $ n = 3$ देता है $0$ (चारों ओर
एक त्रिकोण, हर कोई हर किसी को छूता है) और $ n = 4$ देता है $2$,
आसानी से हाथ से सूचीबद्ध। चिपकाने की तरकीब --- एक मजबूर ब्लॉक का इलाज करें
एक वस्तु के रूप में, फिर उसकी आंतरिक व्यवस्था को गिनें --- है
निकटवर्ती बाधाओं, रैखिक या गोलाकार के लिए मानक इलाज।
\end{example}

\begin{proposition}[बुनियादी पहचान]\label{prop:b1:counting:identities}
के लिए $0 \leq k \leq n $:
\[
\binom{n}{k} = \binom{n}{n-k},
\qquad
\binom{n}{k} = \binom{n-1}{k-1} + \binom{n-1}{k}
\quad (1 \leq k \leq n-1),
\qquad
\sum_{k=0}^{n} \binom{n}{k} = 2^n .
\]
\end{proposition}

\begin{proof}
पहली पहचान: $ A \mapsto E \setminus A $ के बीच एक आपत्ति है
$ k $-उपसमुच्चय और $(n-k)$-उपसमुच्चय. पास्कल का नियम\index{पास्कल का नियम}:
एक तत्व ठीक करें $ a \in E $; $ k $-उपसमुच्चय युक्त उपसमुच्चय में विभाजित हो जाते हैं
$ a $ (चुने $ k - 1$ अन्य: $\binom{n-1}{k-1}$) और जो परहेज कर रहे हैं
$ a $ ($\binom{n-1}{k}$). तीसरी पहचान: दोनों पक्ष सभी उपसमूहों को गिनते हैं
का $ E $, बाईं ओर आकार के अनुसार विभाजित
(\cref{prop:b1:counting:rules}~(1) और (5)).
\end{proof}

\begin{theorem}[द्विपद प्रमेय]\label{thm:b1:counting:binomial}
सभी के लिए $ a, b $ एक क्रमविनिमेय वलय में (कहें) $\R $ या $\C $) और $ n \in \N $:
\[
(a + b)^n = \sum_{k=0}^{n} \binom{n}{k} a^k b^{\,n-k} .
\]
\end{theorem}

\begin{proof}
विस्तार $(a+b)(a+b)\cdots(a+b)$ वितरणात्मक रूप से प्रति एक पद प्राप्त होता है
विकल्प, प्रत्येक कारक में, दोनों में से $ a $ या $ b $: शब्द $ a^k b^{n-k}$
जिसे चुनने के प्रत्येक तरीके के लिए एक बार प्रकट होता है $ k $ की $ n $ कारक
योगदान करें $ a $ --- वह है, $\binom nk $ बार. (वैकल्पिक रूप से: शामिल करें
पर $ n $ पास्कल के नियम का उपयोग करते हुए।)
\end{proof}

\begin{example}\label{ex:b1:counting:binomial}
दो शास्त्रीय विशेषज्ञताएँ: $ a = b = 1$ ठीक $\sum_k \binom nk
= 2^n $; $ a = -1$, $ b = 1$ देता है $\sum_{k} (-1)^k \binom nk = 0$ के लिए
$ n \geq 1$: एक गैर-रिक्त के उपसमुच्चय के बीच \omterm{def:b1:logic:sets}{तय करना}, बिलकुल आधा है
यहां तक कि \omterm{def:b1:counting:card}{प्रमुखता}.
\end{example}

\begin{example}[एक पहचान, दो सबूत]\label{ex:b1:counting:threen}
विशेषज्ञता $ a = 2$, $ b = 1$ द्विपद प्रमेय का पाठ
\[
\sum_{k=0}^{n} \binom nk\,2^k = 3^n .
\]
यहां बिल्कुल भी बीजगणित के बिना वही पहचान है। दाहिनी ओर
लंबाई के शब्द गिनता है $ n $ वर्णमाला के ऊपर $\{0, 1, 2\}$
(प्रॉडक्ट नियम)। प्रत्येक शब्द को इसके अनुसार वर्गीकृत करें \omterm{def:b1:logic:sets}{तय करना} $ K $ पदों का
एक शून्येतर अक्षर रखना: चुनना $ K $ साथ $\abs K = k $ लागत
$\binom nk $, फिर प्रत्येक स्थिति $ K $ स्वतंत्र रूप से वहन करता है $1$
या $2$: $2^k $ तौर तरीकों। योग नियम खत्म $ k $ बाईं ओर देता है.
समझौते की ख़ुशी से परे, दोनों के प्रमाण अलग-अलग हैं
गुण: \omterm{pb:b1:logic:1}{बीजगणितीय} किसी भी मूल्य का सामान्यीकरण करता है $ a $, द
संयोजक एक \emph{बताते हैं} सूत्र और अनुकूलन
बाधाएं (पत्र को प्रतिबंधित करें) $2$ अंतिम स्थिति में, कहें) वह
कोई प्रतिस्थापन कैप्चर नहीं. दोनों तकनीकों को सक्रिय रखना है
यह अध्याय व्यावहारिक कौशल को प्रशिक्षित करता है।
\end{example}

\begin{method}[कौन सी गिनती लागू होती है?]\label{met:b1:counting:which}
गणना करने से पहले, चयन के बारे में दो प्रश्नों के उत्तर दें: क्या
\emph{आदेश} बात है, और हैं \emph{repetitions} अनुमत?
\begin{center}
\renewcommand{\arraystretch}{1.4}
\begin{tabular}{l|c|c}
 & order matters & order does not matter \\
\hline
no repetition & $\dfrac{n!}{(n-k)!}$ & $\dbinom{n}{k}$ \\[6pt]
repetition allowed & $ n^k $ & (\cref{exo:b1:counting:10}) \\
\end{tabular}
\end{center}
फिर किसी आपत्ति या ए की तलाश करें \omterm{thm:b1:logic:partition}{PARTITION} इनसे समस्या कम हो रही है
मॉडल मायने रखता है; सही गिनती एक छिपी हुई आपत्ति है।
\end{method}

\begin{remark}[गिनती में सामान्य गड़बड़ियाँ]\label{rem:b1:counting:pitfalls}
\begin{enumerate}
  \item \emph{गैर-विच्छेदित मामलों का सारांश।} योग नियम के लिए आवश्यक है a
        \omterm{thm:b1:logic:partition}{PARTITION}; यदि कॉन्फ़िगरेशन दो मामलों को संतुष्ट कर सकता है
        एक बार, उन्हें दो बार गिना जाता है --- इलाज है
        समावेश--बहिष्करण (\cref{thm:b1:counting:inclexcl}) या
        एक बेहतर मामला विभाजन.
  \item \emph{आदेशित बनाम अव्यवस्थित।} ''की एक समिति'' का चयन करना
        दो'' है $\binom n2$, नहीं $ n(n-1)$: \omterm{def:b1:logic:sets}{तय करना} \emph{पहले
        कंप्यूटिंग} क्या चयन में कोई आदेश है, और यदि
        एक क्रमबद्ध गणना आसान है, की संख्या से विभाजित करें
        अंत में आदेश --- लेकिन केवल तभी जब प्रत्येक अव्यवस्थित हो
        वस्तु से उत्पन्न होती है \emph{वही} ऑर्डर किए गए लोगों की संख्या.
  \item \emph{बहु-मंचीय विकल्प जो स्वतंत्र नहीं हैं।}
        उत्पाद नियम के लिए प्रत्येक चरण में विकल्पों की संख्या की आवश्यकता होती है
        पिछले विकल्पों से स्वतंत्र होना। ``ए चुनें
        कप्तान, फिर एक अलग उप-कप्तान'' ठीक है ($ n(n-1)$);
        ``ऐसे दो खिलाड़ी चुनें जो आपस में तालमेल बिठा सकें'' यह कोई दो चरण की बात नहीं है
        बिल्कुल उत्पाद.
  \item \emph{निर्माण द्वारा दोहरी गिनती।} प्रत्येक का निर्माण
        वस्तु दो बार --- उदा.\ हाथ गिनते हुए \emph{कम से कम}
        एक इक्का (एक इक्का चुनें) $\times $ (चुनना $4$ अधिक
        कार्ड) --- दो इक्के वाले हाथों की अधिक गिनती। ``कम से कम''
        लगभग हमेशा पूरक की मांग करता है
        (\cref{ex:b1:counting:complement}).
\end{enumerate}
\end{remark}

\begin{example}[पोकर शैली की गिनती]\label{ex:b1:counting:poker}
ए से $52$-कार्ड डेक, की संख्या $5$-कार्ड हाथ है
$\binom{52}{5} = 2\,598\,960$. हाथ जिसमें बिल्कुल एक इक्का हो:
इक्का चुनें ($4$ तरीके) फिर $4$ के बीच कार्ड $48$ गैर-इक्के:
$4 \binom{48}{4} = 778\,320$. उत्पाद नियम लागू होता है क्योंकि
चुनाव स्वतंत्र चरणों में विभाजित हो जाता है।
\end{example}

\begin{method}[दोहरी गिनती]\label{met:b1:counting:doublecount}
दो गिनती के भावों के बीच एक पहचान सिद्ध करने के लिए, एक एकल खोजें
परिमित सेट जिसे दोनों पक्ष गिनते हैं --- आम तौर पर ए \omterm{def:b1:logic:sets}{तय करना} का \emph{जोड़े}
--- और इसका मूल्यांकन करें \omterm{def:b1:counting:card}{प्रमुखता} दो अलग-अलग क्रम में. द
प्रोटोटाइप है \emph{हाथ मिलाना लेम्मा}: किसी पार्टी में, जोड़े गिनें
(व्यक्ति, हाथ मिलाया). लोगों पर योग देना देता है $\sum_p d_p $ (हर व्यक्ति $ p $हाथ मिलाने की संख्या); हाथ मिलाने पर योग मिलता है
हाथ मिलाने की संख्या दोगुनी (प्रत्येक में दो लोग शामिल होते हैं)। इस तरह
$\sum_p d_p $ सम है --- तो विषम को हिलाने वाले लोगों की संख्या
हाथों की संख्या हमेशा सम होती है, एक गैर-तुच्छ निष्कर्ष प्राप्त होता है
बिना किसी फ़ॉर्मूले के. वही इंजन चलता है
\cref{exo:b1:counting:12} और सप्ताहांत के कई प्रश्न
नीचे समस्या.
\end{method}

\begin{example}[औसत उपसमुच्चय]\label{ex:b1:counting:averagesubset}
औसत क्या है \omterm{def:b1:counting:card}{प्रमुखता} एक के एक उपसमुच्चय का $ n $-तत्व \omterm{def:b1:logic:sets}{तय करना}
$ E $, सभी $2^n $ उपसमुच्चय समान रूप से संभावित हैं? दोहरी गिनती करें
जोड़े $(A, a)$ साथ $ a \in A $: उपसमुच्चय का योग देता है
$\sum_A \abs A $, कुल हम चाहते हैं; तत्वों का योग देता है
$ n \cdot 2^{n-1}$ (हरेक $ n $ तत्व ठीक आधे में हैं
उपसमुच्चय में से प्रत्येक को जोड़े $ A $ युक्त $ a $ साथ $ A
\setminus \{a\}$). इस तरह
\[
\frac{1}{2^n}\sum_{A \subseteq E} \abs A
= \frac{n\,2^{n-1}}{2^n} = \frac n2 :
\]
उपसमुच्चय, औसतन, आधे-भरे होते हैं --- समरूपता के रूप में $ A
\leftrightarrow \overline A $ (कौन से जोड़े का आकार $ k $ और $ n - k $)
भविष्यवाणी भी करता है. दो प्रमाण, एक उत्तर और दोनों प्रत्यक्ष से बचते हैं
गणना $\sum_k k\binom nk $ का \cref{exo:b1:counting:5}: ए
अच्छी तरह से चुनी गई जोड़ी अक्सर एक पहचान को बदल देती है।
\end{example}

\section{समावेश--बहिष्करण}

\begin{theorem}[समावेश--बहिष्करण]\label{thm:b1:counting:inclexcl}
के लिए परिमित समुच्चय $ A_1, \dots, A_p $:
\[
\Bigl|\, \bigcup_{i=1}^{p} A_i \,\Bigr|
= \sum_{\emptyset \neq I \subseteq \intint{1}{p}}
(-1)^{\abs{I}+1} \Bigl|\, \bigcap_{i \in I} A_i \,\Bigr| .
\]
के लिए $ p = 3$:
$\abs{A \cup B \cup C} = \abs A + \abs B + \abs C - \abs{A \cap B} -
\abs{A \cap C} - \abs{B \cap C} + \abs{A \cap B \cap C}$.
\end{theorem}

\begin{proof}
एक तत्व ठीक करें $ x $ संघ का और उसके योगदान को गिनें
दाहिनी ओर. होने देना $ J = \{i : x \in A_i\}$, का \omterm{def:b1:counting:card}{प्रमुखता} $ m \geq
1$. तत्व $ x $ में एक बार गिना जाता है $\abs{\bigcap_{i \in I} A_i}$
बिल्कुल कब $\emptyset \neq I \subseteq J $, संकेत के साथ
$(-1)^{\abs I + 1}$; इसका कुल योगदान है
\[
\sum_{k=1}^{m} \binom{m}{k} (-1)^{k+1}
= 1 - \sum_{k=0}^{m} \binom mk (-1)^k = 1 - 0 = 1
\]
द्वारा \cref{ex:b1:counting:binomial}. तो संघ का प्रत्येक तत्व है
बिल्कुल एक बार गिना गया।
\end{proof}

\begin{example}[सहअभाज्य पूर्णांकों की गिनती]\label{ex:b1:counting:coprime}
के कितने पूर्णांक हैं $\intint1{120}$ सहअभाज्य हैं $120 = 2^3
\times 3 \times 5$? एक पूर्णांक एक गुणनखंड साझा करता है $120$ बिलकुल
जब यह विभाज्य हो $2$, $3$ या $5$, इसलिए पूरक की गिनती करें
का $ A_2 \cup A_3 \cup A_5$, कहाँ $ A_d $ के गुणज एकत्रित करता है
$ d $. अंदर $\intint1{120}$, के गुणज $ d $ संख्या $120/d $
जब कभी भी $ d $ विभाजित $120$ --- किसी फ़्लोर फ़ंक्शंस की आवश्यकता नहीं --- और
$ A_2 \cap A_3 = A_6$, आदि। समावेश--बहिष्करण:
\[
\abs{A_2 \cup A_3 \cup A_5}
= 60 + 40 + 24 - 20 - 12 - 8 + 4 = 88 ,
\]
इसलिए $120 - 88 = 32$ पूर्णांक सहअभाज्य हैं $120$. यह है
गणना को उत्पाद के रूप में पुन: समूहित करने का निर्देश:
\[
120 - 88 = 120\Bigl(1 - \frac12\Bigr)\Bigl(1 -
\frac13\Bigr)\Bigl(1 - \frac15\Bigr) = 120 \cdot \frac12 \cdot
\frac23 \cdot \frac45 = 32 :
\]
तीन कोष्ठकों का विस्तार करने पर ठीक आठ को पुन: उत्पन्न होता है
समावेशन की हस्ताक्षरित शर्तें--बहिष्करण, प्रति उपसमूह में से एक $\{2, 3,
5\}$. यह उत्पाद प्रपत्र यूलर के टोटिएंट फ़ंक्शन को परिभाषित करता है, जिसका
अंकगणितीय भूमिका के साथ प्रकट होता है \cref{ch:b1:arith}की सर्वांगसमताएं और
वर्ष 2 खंड में विकसित किया गया है।
\end{example}

\begin{example}[अपविन्यास]\label{ex:b1:counting:derangement}
ए \emph{गड़बड़ी} एक है \omterm{def:b1:counting:objects}{परिवर्तन} बिना किसी निश्चित बिंदु के. होने देना $ A_i $
हो \omterm{def:b1:logic:sets}{तय करना} का क्रमपरिवर्तन का $\intint{1}{n}$ फिक्सिंग $ i $;
तब $\abs{\bigcap_{i \in I} A_i} = (n - \abs I)!$, और
समावेश--बहिष्करण मायने रखता है क्रमपरिवर्तन कम से कम एक निश्चित के साथ
बिंदु; विक्षोभ संख्या
\[
D_n = n! \sum_{k=0}^{n} \frac{(-1)^k}{k!} .
\]
तब से $\sum (-1)^k / k! \to \eu^{-1}$ (देखना \cref{ch:b1:series}), के बारे में
$37\%$ के सभी क्रमपरिवर्तन ये विक्षोभ हैं, जो भी हों $ n $ है।
\end{example}

\begin{remark}[जहाँ इस अध्याय का प्रयोग किया जाता है]\label{rem:b1:counting:whereused}
\omterm{def:b1:counting:objects}{द्विपद गुणांक} इनमें से सबसे अधिक पुन: उपयोग की जाने वाली वस्तुएँ हैं
अध्याय: वे द्विपद प्रमेय को आगे बढ़ाते हैं \cref{ch:b1:poly}
(का विस्तार $(X + a)^n $), के लिए लीबनिज सूत्र $ n $-वें
किसी उत्पाद का व्युत्पन्न \cref{ch:b1:derivative}, और
टेलर विस्तार के गुणांक \cref{ch:b1:taylor}.
क्रमपरिवर्तन एक समूह के रूप में लौटें --- से निर्मित हस्ताक्षर के साथ
व्युत्क्रम गिनती --- में \cref{ch:b1:structures}, और
हस्ताक्षर बदले में निर्धारकों को परिभाषित करता है \cref{ch:b1:det}.
समावेश--बहिष्करण और गिनती के सिद्धांत हैं परिमित
असतत संभाव्यता की रीढ़, वर्ष 2 खंड में विकसित;
की विक्षोभ संख्या \cref{ex:b1:counting:derangement} हैं
नीचे सप्ताहांत समस्या का गहराई से अध्ययन किया गया है।
\end{remark}

\section{अभ्यास}

\begin{exercise}[$\star $]\label{exo:b1:counting:1}
एक लाइसेंस प्लेट में दो अक्षर (A--Z), फिर तीन अंक होते हैं
दो अक्षर. कितनी प्लेटें संभव हैं? बिना दोहराए कितने
चारों में से पत्र?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:counting:2}
कितने विपर्यय (अक्षरों की पुनर्व्यवस्था, सार्थक या नहीं)
शब्द है \textsc{orange}? और \textsc{banana}?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:counting:3}
की एक समिति $4$ लोगों में से चुना जाता है $7$ महिलाएं और $5$ पुरुष. कैसे
अनेक समितियाँ: कुल मिलाकर? बिल्कुल के साथ $2$ औरत? कम से कम एक के साथ
यार?
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:counting:4}
के किसी भी समूह में इसे सिद्ध करें $13$ लोग, दो लोग अपना जन्म महीना साझा करते हैं;
और वह भी किसी के बीच $ n + 1$ पूर्णांकों में से चुना गया $\intint{1}{2n}$, दो लगातार हैं। \emph{(कबूतर का छेद दोनों बार: नाम
बक्से.)}
\end{exercise}

\begin{exercise}[$\star $]\label{exo:b1:counting:5}
गणना करना $\sum_{k=0}^{n} k \binom{n}{k}$. \emph{संकेत: अंतर करें
$(1 + x)^n $, या उपयोग करें $ k \binom nk = n \binom{n-1}{k-1}$ (इसे साबित करो)।}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:counting:6}
कितनी सख्ती से बढ़ रही है एमएपीएस वहाँ से हैं $\intint{1}{k}$ को $\intint{1}{n}$? की संख्या निकालें
बढ़ रहा है (जरूरी नहीं कि सख्ती से) एमएपीएस. \emph{दूसरे के लिए संकेत
गिनती: $ f $ की बढ़ती $\mapsto $ $ g(i) = f(i) + i - 1$.}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:counting:7}
(वैंडरमोंडे) गिनकर साबित करें $ k $-ए के उपसमुच्चय \omterm{def:b1:logic:sets}{तय करना} दो हिस्सों में बंट गया
आकार के ब्लॉक $ m $ और $ n $:
\[
\binom{m+n}{k} = \sum_{j=0}^{k} \binom{m}{j} \binom{n}{k-j} .
\]
परिणाम निकालना $\sum_{j=0}^{n} \binom nj^2 = \binom{2n}{n}$.
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:counting:8}
कितने पूर्णांक हैं $\intint{1}{1000}$ से विभाज्य हैं
$2$ या $3$ या $5$? (समावेशन--बहिष्करण; $\lfloor 1000/6 \rfloor $
के गुणजों को गिनता है $6$, वगैरह।)
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:counting:9}
ए से अनुमानों की गणना करें \omterm{def:b1:logic:sets}{तय करना} का $4$ ए पर तत्व \omterm{def:b1:logic:sets}{तय करना} का $2$
तत्व; फिर ए पर \omterm{def:b1:logic:sets}{तय करना} का $3$ तत्व. \emph{संकेत: गिनती करें
गै\omterm{def:b1:logic:inj}{र-विशेषण}ात्मक एमएपीएस समावेशन के साथ- छूटे हुए मूल्यों पर बहिष्करण।}
\end{exercise}

\begin{exercise}[$\star\star $]\label{exo:b1:counting:10}
(सितारे और बार) साबित करें कि संख्या $ k $-का चयन $ n $
वस्तुओं \emph{साथ} पुनरावृत्ति, क्रम की अनदेखी --- समकक्ष, द
की संख्या $(x_1, \dots, x_n) \in \N^n $ साथ $ x_1 + \dots + x_n = k $
--- है $\binom{n + k - 1}{k}$. \emph{संकेत: समाधान को एक पंक्ति के रूप में एन्कोड करें
का $ k $ सितारे और $ n - 1$ बार.}
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b1:counting:11}
का सूत्र सिद्ध करें \cref{ex:b1:counting:derangement} के लिए $ D_n $ में
विवरण, और निष्कर्ष निकालें $ n! = \sum_{k=0}^{n} \binom{n}{k} D_{n-k}$ (भी
इस पहचान को सीधे वर्गीकृत करके सिद्ध करें क्रमपरिवर्तन उनके द्वारा
निश्चित-बिंदु \omterm{def:b1:logic:sets}{तय करना}).
\end{exercise}

\begin{exercise}[$\star\star\star $]\label{exo:b1:counting:12}
के लिए $ n \in \N^*$, जोड़ियों की दोहरी गिनती से सिद्ध करें (उपसमुच्चय, चिह्नित)।
तत्त्व):
\[
\sum_{k=1}^{n} k \binom{n}{k} = n\, 2^{n-1},
\qquad\text{then}\qquad
\sum_{k=1}^{n} k^2 \binom{n}{k} = n(n+1)\, 2^{n-2} .
\]
\emph{दूसरे के लिए: चिह्नित तत्वों के जोड़े गिनें, चाहे वे बराबर हों या नहीं।}
\end{exercise}

\section{समस्या: अव्यवस्थाएं, या गलत पते वाले पत्र}

\begin{problem}\label{pb:b1:counting:1}
एक सचिव डालता है $ n $ में पत्र $ n $ बेतरतीब ढंग से संबोधित लिफाफे:
इसकी क्या संभावना है \emph{कोई नहीं} सही पत्र प्राप्त होता है?
यह शास्त्रीय प्रश्न (मोंटमॉर्ट, 1708) विक्षिप्तता की ओर ले जाता है
नंबर $ D_n $ का \cref{ex:b1:counting:derangement}. द
समावेश--बहिष्करण सूत्र केवल प्रारंभिक कदम है: यह समस्या
गणना करने वाली पुनरावृत्तियों को विकसित करता है $ D_n $, दो और स्वतंत्र
सूत्र के प्रमाण, हड़ताली प्रमेय $ D_n $ है
के निकटतम पूर्णांक $ n!/\eu $, निश्चित बिंदुओं का पूर्ण वितरण
एक यादृच्छिक का \omterm{def:b1:counting:objects}{परिवर्तन}, और अनुक्रम का जिज्ञासु अंकगणित
$(D_n)$.\index{गड़बड़ी}
लगातार, $ D_n $ विक्षोभों की संख्या (निश्चित-बिंदु) को दर्शाता है
निःशुल्क क्रमपरिवर्तन) का $\intint1n $, सम्मेलन के साथ $ D_0 = 1$
(द एम्प्टी \omterm{def:b1:counting:objects}{परिवर्तन} कोई निश्चित बिंदु नहीं है)।

\medskip
\noindent\textbf{Part I --- Small cases and the fixed-point
census.}
\begin{enumerate}
  \item गणना करना $ D_1, D_2, D_3$ सीधे, और $ D_4$ सूचीबद्ध करके
        की अव्यवस्थाएं $\{1, 2, 3, 4\}$ के मान के आधार पर समूहीकृत किया गया
        $\sigma(1)$. (आपको ढूंढना चाहिए $ D_4 = 9$.)
  \item के लिए $0 \leq k \leq n $, वह संख्या दिखाओ $ P_k(n)$ का
        क्रमपरिवर्तन का $\intint1n $ साथ \emph{बिल्कुल} $ k $ निश्चित
        अंक है $\binom nk D_{n-k}$.
  \item के लिए जनगणना सत्यापित करें $ n = 4$: गणना करें $ P_0(4), \dots,
        P_4(4)$ और जांचें कि उनका योग कितना है $4! = 24$. जो अधिक है
        चार अक्षरों की संभावना: कोई मिलान नहीं, या बिल्कुल एक मिलान?
  \item दोहरी गिनती से (\cref{met:b1:counting:doublecount})
        जोड़े $(\sigma, i)$ साथ $\sigma(i) = i $, बताते हैं कि
        \[
        \sum_{\sigma} \abs{\mathrm{Fix}(\sigma)} = n! :
        \]
        औसतन, एक यादृच्छिक \omterm{def:b1:counting:objects}{परिवर्तन} है \emph{बिल्कुल एक}
        निश्चित बिंदु, जो भी हो $ n \geq 1$ है।
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Two recurrences and two new proofs of
the formula.}
\begin{enumerate}[resume]
  \item के लिए, संयुक्त रूप से सिद्ध करें $ n \geq 1$:
        \[
        D_{n+1} = n\,(D_n + D_{n-1}) .
        \]
        (विकृतियों को वर्गीकृत करें $\sigma $ का $\intint1{n+1}$ द्वारा
        $ j = \sigma(n+1)$, फिर चाहे द्वारा $\sigma(j) = n + 1$; यदि $\sigma(j) \neq n+1$, के साथ एक आक्षेप का निर्माण करें
        की अव्यवस्थाएं $\intint1n $ को पुनर्निर्देशित करके \omterm{def:b1:logic:map}{पूर्वछवि} का
        $ n + 1$ को $ j $.) तक पुनरावृत्ति की संख्यात्मक जाँच करें
        $ D_6$.
  \item सेटिंग $ u_n = D_n - n D_{n-1}$, प्रश्न 5 से निष्कर्ष निकालें
        वह $ u_{n+1} = -u_n $, और दूसरी पुनरावृत्ति समाप्त करें:
        \[
        D_n = n D_{n-1} + (-1)^n \qquad (n \geq 1).
        \]
  \item प्रश्न 6 से, का सूत्र आगमन द्वारा सिद्ध कीजिए
        \cref{ex:b1:counting:derangement},
        \[
        D_n = n! \sum_{k=0}^{n} \frac{(-1)^k}{k!},
        \]
        --- समावेशन-बहिष्करण से पूर्णतः स्वतंत्र प्रमाण।
  \item (द्विपद व्युत्क्रम) चलो $(a_n)$ और $(b_n)$ दो हो
        क्रम ऐसे कि $ a_n = \sum_{k=0}^n \binom nk b_k $ सभी के लिए $ n $. साबित करें कि
        \[
        b_n = \sum_{k=0}^{n} (-1)^{n-k} \binom nk a_k
        \qquad (n \in \N).
        \]
        (पहले स्थापित करें \emph{त्रिपद पुनरीक्षण}
        $\binom nk \binom kj = \binom nj \binom{n-j}{k-j}$, फिर
        के वैकल्पिक पंक्ति योग का उपयोग करें
        \cref{ex:b1:counting:binomial}.)
  \item प्रश्न 8 को पहचान पर लागू करें $ n! = \sum_k \binom nk
        D_{n-k}$ का \cref{exo:b1:counting:11} एक प्राप्त करने के लिए
        \emph{तीसरा} के लिए सूत्र का प्रमाण $ D_n $.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- The nearest integer to
$ n!/\eu $.} इस भाग के लिए स्वीकार करें --- सिद्धांत अंतर्निहित है
\cref{ch:b1:series} --- वह
$\eu^{-1} = \lim_{n \to \infty} s_n $ कहाँ $ s_n = \sum_{k=0}^{n}
\frac{(-1)^k}{k!}$, सख्त वैकल्पिक-श्रृंखला के साथ बाध्य
$\abs{\eu^{-1} - s_n} < \frac1{(n+1)!}$ हरएक के लिए $ n $.
\begin{enumerate}[resume]
  \item बताते हैं कि $\bigl| D_n - n!/\eu \bigr| < \frac1{n+1}$ सभी के लिए $ n \in \N $.
  \item शीर्षक प्रमेय निकालें: \emph{हरएक के लिए $ n \geq 1$,
        $ D_n $ के निकटतम पूर्णांक है $ n!/\eu $}. क्यों करता है
        तर्क की आवश्यकता $ n \geq 1$?
  \item त्रुटि का चिह्न निर्धारित करें: उसे दिखाएँ $ D_n > n!/\eu $
        बिल्कुल कब $ n $ सम है. (पहले उपेक्षित शब्द का पता लगाएं
        वैकल्पिक श्रृंखला का।)
  \item गणना करना $ D_7$ के माध्यम से $ D_{10}$ की पुनरावृत्ति के साथ
        प्रश्न 5, फिर जाँचें $ D_{10}$ ख़िलाफ़ $10!/\eu $
        ($10! = 3\,628\,800$, $\eu \approx 2.718281828$).
  \item (हैट-चेक संभावना) चलो $ p_n = D_n/n!$ हो
        संभावना है कि एक समान रूप से यादृच्छिक \omterm{def:b1:counting:objects}{परिवर्तन} एक है
        विक्षोभ. दिखाओ $\abs{p_n - \eu^{-1}} < \frac1{(n+1)!}$
        और गणना करें $ p_6$ पाँच दशमलव स्थानों तक. टिप्पणी: क्यों है
        मॉन्टमॉर्ट के प्रश्न का उत्तर अनिवार्य रूप से स्वतंत्र है
        का $ n $ --- पहले से ही एक दर्जन पत्रों के लिए?
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- The distribution of fixed points.}
\begin{enumerate}[resume]
  \item हल करना $ k \in \N $. दिखाएँ कि का अनुपात क्रमपरिवर्तन
        का $\intint1n $ बिल्कुल के साथ $ k $ निश्चित बिंदु संतुष्ट करते हैं
        \[
        \frac{P_k(n)}{n!} = \frac{s_{n-k}}{k!}
        \;\xrightarrow[n \to \infty]{}\; \frac{\eu^{-1}}{k!} .
        \]
        (ये सीमित मूल्य, संक्षेप में $1$, इससे
        \emph{पॉइसन वितरण} पैरामीटर का $1$, एक केंद्रीय
        वर्ष 2 खंड में संभाव्यता पाठ्यक्रम का उद्देश्य।)
  \item त्रिगुणों की दुगुनी गिनती करके $(\sigma, i, j)$ कहाँ $ i
        \neq j $ दोनों द्वारा तय किए गए हैं $\sigma $, बताते हैं कि
        $\sum_\sigma \abs{\mathrm{Fix}(\sigma)}\,
        (\abs{\mathrm{Fix}(\sigma)} - 1) = n!$ के लिए $ n \geq 2$.
        प्रश्न 4 के साथ संयुक्त: का औसत
        $\abs{\mathrm{Fix}}^2$ है $2$, तो ``फैलाव'' (भिन्नता)
        निश्चित बिंदुओं की संख्या के बराबर है $1$ --- फिर से
        से स्वतंत्र $ n $, फिर से पॉइसन कानून से मेल खाता है।
  \item के अनुपात की गणना करें क्रमपरिवर्तन कम से कम एक होना
        के लिए निश्चित बिंदु $ n = 4, 5, 6$ (अंशों और चार के रूप में
        दशमलव स्थान), और तुलना करें $1 - \eu^{-1} \approx
        0.6321$.
  \item सीधे दिखाएँ --- कोई सीमा आवश्यक नहीं --- वह $ s_{n+2} - s_n
        = (-1)^{n+1}\bigl(\frac1{(n+1)!} - \frac1{(n+2)!}\bigr)$,
        और संभावनाएँ निकालें $ p_n = s_n $ का
        प्रश्न 14 दोलन: $ p_0 > p_2 > p_4 > \dots $ और
        $ p_1 < p_3 < p_5 < \dots $, सम (सम्मान)\ विषम) मान
        घट रहा है (सम्मान)\ सामान्य सीमा की ओर बढ़ रहा है)।
        $\eu^{-1}$.
  \item (गुप्त सैंटा) $ n $ प्रत्येक व्यक्ति एक टोपी से एक नाम खींचता है;
        यदि कोई अपना स्वयं का नाम बनाता है, तो \emph{साबुत} ड्राइंग है
        शून्य से पुनः प्रारंभ किया गया. मानक तथ्य का उपयोग करते हुए कि ए
        संभाव्यता की घटना $ p $ औसत लेता है $1/p $ प्रयास,
        आवश्यक पूर्ण चित्रों की औसत संख्या का अनुमान लगाएं,
        और निष्कर्ष निकालें कि इस प्रक्रिया में लगभग लागत आती है $\eu \approx
        2.72$ औसतन चित्र, अनिवार्य रूप से स्वतंत्र रूप से
        $ n $.
\end{enumerate}

\medskip
\noindent\textbf{Part V --- The arithmetic of $ D_n $, and a
synthesis.}
\begin{enumerate}[resume]
  \item प्रश्न 5 को परिष्कृत करें: उसे निश्चित के लिए दिखाएँ $ j \in
        \intint2n $, की विक्षिप्तताएँ $\intint1n $ साथ
        $\sigma(1) = j $ बिल्कुल सही संख्या $ D_{n-1} + D_{n-2}$,
        का स्वतंत्र रूप से $ j $. उसका निष्कर्ष निकालो $ n - 1$ विभाजित $ D_n $
        हरएक के लिए $ n \geq 2$.
  \item साबित करें कि $ D_n $ विषम है यदि और केवल यदि $ n $ सम है. (कार्य
        modulo $2$ प्रश्न 6 की पुनरावृत्ति में)
  \item साबित करें कि $ D_n \equiv (-1)^n \pmod n $ के लिए $ n \geq 1$, और
        के अंतिम अंक पर सर्वांगसमता की जाँच करें $ D_{10}$.
  \item प्रश्न 6 से दिखाएँ कि $\dfrac{D_n}{D_{n-1}} = n +
        \dfrac{(-1)^n}{D_{n-1}}$ के लिए $ n \geq 3$, तो का अनुपात
        लगातार विक्षोभ संख्या है \emph{लगभग बिलकुल}
        $ n $; एक वाक्य में स्पष्ट करें कि यह सुसंगत क्यों है
        $ D_n \approx n!/\eu $.
  \item वास्तव में इस समस्या का उपयोग कहां हुआ: (i) उत्पाद और
        योग नियम; (ii) दोहरी गिनती; (iii) द्विपद
        प्रमेय; (iv) स्वीकृत वैकल्पिक-श्रृंखला बाध्य? एक
        प्रत्येक वाक्य.
  \item संश्लेषण। के लिए सूत्र $ D_n $ अब तीन प्रमाण हैं
        (समावेशन--बहिष्करण, पुनरावृत्ति प्लस प्रेरण, द्विपद
        उलटा)। एक संक्षिप्त अनुच्छेद में, प्रत्येक प्रमाण की तुलना करें
        \emph{बताते हैं}: कौन सबसे तेज़ गणना करता है, कौन सा
        अन्य निश्चित-बिंदु गणनाओं को सामान्यीकृत करता है, और कौन सा
        पता चलता है क्यों $\eu$ लिफ़ाफ़ों के बारे में एक समस्या में प्रकट होता है.
\end{enumerate}
\end{problem}
