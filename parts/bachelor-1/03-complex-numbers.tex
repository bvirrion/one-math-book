\chapter{Complex Numbers}\label{ch:b1:complex}

Complex numbers were met in \cref{ch:g12:complex} as a computational
device for quadratic equations. This chapter treats them as a central
object: the exponential form and its consequences (de Moivre, $n$-th
roots, roots of unity), the systematic translation between $\C$ and
plane geometry, and the use of $\eu^{\iu\theta}$ as a machine for
proving trigonometric identities.

\section{The field $\C$, modulus and conjugate}

\begin{definition}[The complex field]\label{def:b1:complex:field}
$\C = \{a + \iu b : a, b \in \R\}$ with the usual addition and the
multiplication determined by $\iu^2 = -1$. Every nonzero $z = a + \iu
b$ has an inverse: $z^{-1} = \frac{a - \iu b}{a^2 + b^2}$ --- in the
language of \cref{ch:b1:structures}, $\C$ is a field. One writes $a =
\Re(z)$, $b = \Im(z)$, $\conj{z} = a - \iu b$ (the
\emph{conjugate}\index{conjugate}) and $\abs{z} = \sqrt{a^2 + b^2}$
(the \emph{modulus}\index{modulus}).
\end{definition}

\begin{proposition}[Rules for conjugate and modulus]\label{prop:b1:complex:rules}
For $z, w \in \C$:
\begin{enumerate}
  \item $\conj{z + w} = \conj z + \conj w$, $\conj{zw} = \conj z\,
        \conj w$, $\conj{\conj z} = z$;
  \item $z \conj z = \abs z^2$; $\;\Re(z) = \frac{z + \conj z}{2}$,
        $\Im(z) = \frac{z - \conj z}{2\iu}$;
  \item $\abs{zw} = \abs z\, \abs w$, and $\abs{z^{-1}} =
        \abs{z}^{-1}$ for $z \neq 0$;
  \item (triangle inequality\index{triangle inequality}) $\abs{z + w}
        \leq \abs z + \abs w$, with equality if and only if $z$ and
        $w$ lie on a common ray from $0$ ($w = \lambda z$ or
        $z = \lambda w$ with $\lambda \geq 0$);
  \item (reverse triangle inequality) $\bigl|\abs z - \abs w\bigr|
        \leq \abs{z - w}$.
\end{enumerate}
\end{proposition}

\begin{proof}
(1) and (2) are direct computations on real and imaginary parts. (3):
$\abs{zw}^2 = zw\,\conj{zw} = z\conj z\, w \conj w = \abs z^2 \abs
w^2$, then take square roots; apply to $z \cdot z^{-1} = 1$ for the
inverse.

(4) Both sides are nonnegative, so compare squares:
\[
\abs{z+w}^2 = (z+w)(\conj z + \conj w)
= \abs z^2 + \abs w^2 + 2\,\Re(z \conj w),
\]
and $\Re(z\conj w) \leq \abs{z \conj w} = \abs z \abs w$ gives
$\abs{z+w}^2 \leq (\abs z + \abs w)^2$. Equality forces $\Re(z \conj
w) = \abs{z \conj w}$, i.e.\ $z \conj w \in \R_+$; if $w \neq 0$ this
gives $z = \frac{z\conj w}{\abs w^2}\, w = \lambda w$ with $\lambda
\geq 0$ (and the case $w = 0$ is trivial).

(5) $\abs z = \abs{(z - w) + w} \leq \abs{z-w} + \abs w$ gives $\abs z
- \abs w \leq \abs{z - w}$; swap $z$ and $w$ for the other sign.
\end{proof}

\section{Exponential form}

\begin{definition}[Complex exponential of an imaginary argument]\label{def:b1:complex:expi}
For $\theta \in \R$ one defines
\[
\eu^{\iu\theta} = \cos\theta + \iu \sin\theta .
\]
Every $z \neq 0$ can be written $z = r\,\eu^{\iu\theta}$ with $r =
\abs z > 0$; $\theta$ is an \emph{argument}\index{argument} of $z$,
determined up to adding a multiple of $2\pi$. The value in
$\intoc{-\pi}{\pi}$ is the \emph{principal argument}, written
$\arg z$.
\end{definition}

\begin{theorem}[The functional equation]\label{thm:b1:complex:funceq}
For all $\theta, \varphi \in \R$:
\[
\eu^{\iu\theta}\, \eu^{\iu\varphi} = \eu^{\iu(\theta + \varphi)},
\qquad
\abs{\eu^{\iu\theta}} = 1,
\qquad
\conj{\eu^{\iu\theta}} = \eu^{-\iu\theta} = (\eu^{\iu\theta})^{-1}.
\]
Consequently $\abs{zw} = \abs z \abs w$ carries the arguments too:
$\arg(zw) \equiv \arg z + \arg w \pmod{2\pi}$.
\end{theorem}

\begin{proof}
Expand the product and use the addition formulas:
\[
(\cos\theta + \iu\sin\theta)(\cos\varphi + \iu\sin\varphi)
= (\cos\theta\cos\varphi - \sin\theta\sin\varphi)
+ \iu\,(\sin\theta\cos\varphi + \cos\theta\sin\varphi),
\]
which is $\cos(\theta+\varphi) + \iu\sin(\theta+\varphi)$. The modulus
is $\sqrt{\cos^2\theta + \sin^2\theta} = 1$, and the conjugate formula
is the parity of cosine and sine; it inverts $\eu^{\iu\theta}$ since
$\eu^{\iu\theta}\eu^{-\iu\theta} = \eu^0 = 1$.
\end{proof}

\begin{corollary}[de Moivre's formula]\label{cor:b1:complex:demoivre}
For $\theta \in \R$ and $n \in \Z$:
$\;(\cos\theta + \iu\sin\theta)^n = \cos(n\theta) +
\iu\sin(n\theta)$.\index{de Moivre's formula}
\end{corollary}

\begin{proof}
For $n \geq 0$, induction using \cref{thm:b1:complex:funceq}; for
$n < 0$, take inverses.
\end{proof}

\begin{proposition}[Euler's formulas]\label{prop:b1:complex:euler}
\[
\cos\theta = \frac{\eu^{\iu\theta} + \eu^{-\iu\theta}}{2},
\qquad
\sin\theta = \frac{\eu^{\iu\theta} - \eu^{-\iu\theta}}{2\iu}.
\]
\end{proposition}

\begin{proof}
Add, respectively subtract, $\eu^{\iu\theta} = \cos\theta +
\iu\sin\theta$ and $\eu^{-\iu\theta} = \cos\theta - \iu\sin\theta$.
\end{proof}

\begin{method}[Trigonometry via exponentials]\label{met:b1:complex:trig}
\begin{enumerate}
  \item \emph{Linearize} $\cos^p\theta \sin^q\theta$ (turn powers into
        a sum of $\cos k\theta$, $\sin k\theta$): substitute Euler's
        formulas, expand with the binomial theorem
        (\cref{thm:b1:counting:binomial}), regroup conjugate terms.
  \item \emph{Expand} $\cos n\theta$ as a polynomial in $\cos\theta$:
        write $\cos n\theta = \Re\bigl((\cos\theta +
        \iu\sin\theta)^n\bigr)$, expand, and convert even powers of
        $\sin$ via $\sin^2 = 1 - \cos^2$.
  \item \emph{Sum trigonometric series} like $\sum_k \cos k\theta$:
        recognize the real part of a geometric sum $\sum_k
        (\eu^{\iu\theta})^k$.
  \item The \emph{half-angle factorization}: for all $p, q$,
        \[
        \eu^{\iu p} + \eu^{\iu q}
        = 2 \cos\tfrac{p - q}{2}\; \eu^{\iu \frac{p+q}{2}},
        \qquad
        \eu^{\iu p} - \eu^{\iu q}
        = 2\iu \sin\tfrac{p - q}{2}\; \eu^{\iu \frac{p+q}{2}} .
        \]
\end{enumerate}
\end{method}

\begin{example}[Linearization]\label{ex:b1:complex:linearize}
\[
\cos^3\theta
= \Bigl(\frac{\eu^{\iu\theta} + \eu^{-\iu\theta}}{2}\Bigr)^{\!3}
= \frac{\eu^{3\iu\theta} + 3\eu^{\iu\theta} + 3\eu^{-\iu\theta} +
\eu^{-3\iu\theta}}{8}
= \frac{\cos 3\theta + 3\cos\theta}{4}.
\]
This form integrates immediately --- the very reason linearization
matters in \cref{ch:b1:integration}.
\end{example}

\section{Roots of complex numbers}

\begin{theorem}[$n$-th roots]\label{thm:b1:complex:roots}
Let $a = r\,\eu^{\iu\alpha} \neq 0$ and $n \in \N^*$. The equation
$z^n = a$ has exactly $n$ solutions:
\[
z_k = r^{1/n}\, \eu^{\iu\left(\frac{\alpha}{n} +
\frac{2k\pi}{n}\right)},
\qquad k = 0, 1, \dots, n - 1 .
\]
\end{theorem}

\begin{proof}
Write $z = \rho\,\eu^{\iu\theta}$ ($\rho > 0$). Then $z^n =
\rho^n \eu^{\iu n\theta} = r \eu^{\iu\alpha}$ if and only if $\rho^n
= r$ (moduli) and $n\theta \equiv \alpha \pmod{2\pi}$ (arguments),
i.e.\ $\rho = r^{1/n}$ and $\theta = \frac{\alpha}{n} +
\frac{2k\pi}{n}$ for some $k \in \Z$. Two integers $k$ and $k'$ give
the same solution exactly when $n$ divides $k - k'$, so $k = 0, \dots,
n-1$ lists each solution once.
\end{proof}

\begin{definition}[Roots of unity]\label{def:b1:complex:unity}
The \emph{$n$-th roots of unity}\index{roots of unity} are the
solutions of $z^n = 1$:
\[
\mathbb{U}_n = \bigl\{\, \omega^k : k = 0, \dots, n-1 \,\bigr\},
\qquad \omega = \eu^{2\iu\pi/n}.
\]
They form a group under multiplication (\cref{ch:b1:structures}) and
sit at the vertices of a regular $n$-gon inscribed in the unit circle.
\end{definition}

\begin{proposition}[Sum of the roots of unity]\label{prop:b1:complex:sumroots}
For $n \geq 2$, the $n$-th roots of unity sum to zero:
$\sum_{k=0}^{n-1} \omega^k = 0$.
\end{proposition}

\begin{proof}
Geometric sum with ratio $\omega \neq 1$:
$\sum_{k=0}^{n-1} \omega^k = \frac{\omega^n - 1}{\omega - 1} = 0$
since $\omega^n = 1$.
\end{proof}

\begin{example}[Square roots in algebraic form]\label{ex:b1:complex:sqrt}
To solve $z^2 = 3 + 4\iu$ without trigonometry, set $z = x + \iu y$:
\[
x^2 - y^2 = 3, \qquad 2xy = 4, \qquad x^2 + y^2 = \abs{3 + 4\iu} = 5 .
\]
Adding the first and last: $x^2 = 4$, so $x = \pm 2$, then $y = 2/x =
\pm 1$ with the \emph{same} sign pairing ($xy = 2 > 0$): $z = \pm(2 +
\iu)$. Combined with the usual formula, this solves every quadratic
equation with complex coefficients
(\cref{exo:b1:complex:7}).
\end{example}

\section{Complex numbers and plane geometry}

\begin{proposition}[Geometric dictionary]\label{prop:b1:complex:geometry}
Identify the point $M(x, y)$ of the plane with its \emph{affix} $z = x
+ \iu y$\index{affix}. For distinct points $A, B, C$ of affixes $a, b,
c$:
\begin{enumerate}
  \item $\abs{b - a}$ is the distance $AB$;
  \item $\arg\dfrac{c - a}{b - a}$ is the angle between the vectors
        $\vect{AB}$ and $\vect{AC}$ (mod $2\pi$);
  \item $A, B, C$ are aligned if and only if $\dfrac{c - a}{b - a} \in
        \R$; the lines $AB$ and $AC$ are perpendicular if and only if
        $\dfrac{c - a}{b - a} \in \iu\R$.
\end{enumerate}
\end{proposition}

\begin{proof}
(1) is the definition of the modulus applied to $b - a$, the affix of
$\vect{AB}$. (2): write $b - a = r\eu^{\iu\theta}$, $c - a =
s\eu^{\iu\varphi}$; then $\frac{c-a}{b-a} = \frac sr
\eu^{\iu(\varphi - \theta)}$ has argument $\varphi - \theta$, the
angle from $\vect{AB}$ to $\vect{AC}$. (3): alignment means angle $0$
or $\pi$, i.e.\ argument in $\pi\Z$, i.e.\ the quotient is real;
perpendicularity means angle $\pm\frac\pi2$, i.e.\ the quotient is
purely imaginary. (The quotient is nonzero since $C \neq A$.)
\end{proof}

\begin{proposition}[Maps $z \mapsto az + b$]\label{prop:b1:complex:similitude}
Let $a \in \C^*$, $b \in \C$. The transformation $f(z) = az + b$ of
the plane:
\begin{itemize}
  \item is a translation if $a = 1$;
  \item otherwise has a unique fixed point $\zeta = \frac{b}{1-a}$,
        and $f(z) - \zeta = a\,(z - \zeta)$: $f$ is the rotation of
        center $\zeta$ and angle $\arg a$, composed with the scaling
        of center $\zeta$ and ratio $\abs a$.
\end{itemize}
In particular $z \mapsto \eu^{\iu\theta} z$ is the rotation of angle
$\theta$ about the origin, and $\conj z$ is the reflection in the real
axis.
\end{proposition}

\begin{proof}
If $a = 1$, $f(z) = z + b$ translates by the vector of affix $b$. If
$a \neq 1$, the fixed-point equation $z = az + b$ has the unique
solution $\zeta = \frac{b}{1-a}$, and then $f(z) - \zeta = az + b -
(a\zeta + b) = a(z - \zeta)$. Writing $a = \abs a\, \eu^{\iu\arg a}$,
multiplication by $a$ scales distances to $\zeta$ by $\abs a$ and adds
$\arg a$ to angles at $\zeta$, which is the announced composition.
\end{proof}

\begin{example}\label{ex:b1:complex:equilateral}
$A, B, C$ (affixes $a, b, c$, pairwise distinct) form an equilateral
triangle with the vertices in direct (counterclockwise) order if and
only if $\frac{c - a}{b - a} = \eu^{\iu\pi/3}$: the rotation of
center $A$ and angle $\frac\pi3$ sends $B$ to $C$. Both orientations
together are captured by the symmetric equation $a^2 + b^2 + c^2 = ab
+ bc + ca$ (\cref{exo:b1:complex:10}).
\end{example}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:b1:complex:1}
Put in exponential form: $1 + \iu$; $\;\sqrt 3 - \iu$; $\;-5$;
$\;\dfrac{1 + \iu}{\sqrt 3 - \iu}$. Deduce
$\cos\frac{5\pi}{12}$ and $\sin\frac{5\pi}{12}$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:complex:2}
Compute $(1 + \iu)^{20}$. For which $n \in \N$ is $(1 + \iu)^n$ a
real number?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:complex:3}
Describe geometrically the set of $z \in \C$ such that:
$\abs{z - 2} = \abs{z + \iu}$; $\;\abs{z - 1} = 2$;
$\;\dfrac{z - 1}{z + 1} \in \iu\R$ (for $z \neq -1$).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:complex:4}
Linearize $\sin^4\theta$, and expand $\cos 4\theta$ as a polynomial in
$\cos\theta$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:complex:5}
Solve $z^3 = 8\iu$ and place the solutions on a picture. Solve
$z^4 = -4$ and factor $X^4 + 4$ into two real quadratic polynomials.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:complex:6}
For $\theta \in \R$ with $\eu^{\iu\theta} \neq 1$ and $n \in \N$,
compute
\[
C_n = \sum_{k=0}^{n} \cos k\theta
\qquad\text{and}\qquad
S_n = \sum_{k=0}^{n} \sin k\theta
\]
by summing the geometric series $\sum_k \eu^{\iu k\theta}$ and using
the half-angle factorization.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:complex:7}
Solve in $\C$: $z^2 - (3 + 4\iu) z + (-1 + 5\iu) = 0$. \emph{(Compute
the discriminant, extract its square roots as in
\cref{ex:b1:complex:sqrt}.)}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:complex:8}
Let $\omega = \eu^{2\iu\pi/5}$.
\begin{enumerate}
  \item Justify $1 + \omega + \omega^2 + \omega^3 + \omega^4 = 0$.
  \item Set $u = \omega + \omega^4$ and $v = \omega^2 + \omega^3$.
        Compute $u + v$ and $uv$, and deduce that $u$ and $v$ are the
        roots of $X^2 + X - 1$.
  \item Conclude that $\cos\frac{2\pi}{5} = \frac{\sqrt 5 - 1}{4}$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:complex:9}
Prove that for all $z, w \in \C$ (parallelogram identity):
\[
\abs{z + w}^2 + \abs{z - w}^2 = 2\abs z^2 + 2\abs w^2 ,
\]
and interpret it geometrically in the parallelogram with vertices $0,
z, w, z + w$.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:complex:10}
Prove that three pairwise distinct points of affixes $a, b, c$ form an
equilateral triangle (either orientation) if and only if
\[
a^2 + b^2 + c^2 = ab + bc + ca .
\]
\emph{Hint: the direct case is $\frac{c-a}{b-a} = -j^2$ and the
indirect case $\frac{c-a}{b-a} = -j$, where $j = \eu^{2\iu\pi/3}$
satisfies $j^2 + j + 1 = 0$; or factor $a + jb + j^2c$ and $a + j^2 b
+ jc$.}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:complex:11}
For $n \in \N^*$, evaluate $P = \prod_{k=1}^{n-1}
\bigl(1 - \omega^k\bigr)$ where $\omega = \eu^{2\iu\pi/n}$.
\emph{Hint: $X^n - 1 = \prod_{k=0}^{n-1} (X - \omega^k)$; divide by
$X - 1$ and evaluate at $X = 1$.} Deduce
$\prod_{k=1}^{n-1} \sin\frac{k\pi}{n} = \dfrac{n}{2^{n-1}}$.
\end{exercise}
