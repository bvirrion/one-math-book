\chapter{Complex Numbers}\label{ch:b1:complex}

Complex numbers were met in the High School volume as a computational
device for quadratic equations. This chapter treats them as a central
object: the exponential form and its consequences (de Moivre, $n$-th
roots, roots of unity), the systematic translation between $\C$ and
plane geometry, and the use of $\eu^{\iu\theta}$ as a machine for
proving trigonometric identities.

\section{The field $\C$, modulus and conjugate}

\begin{definition}[The complex field]\label{def:b1:complex:field}
$\C = \{a + \iu b : a, b \in \R\}$ with the usual addition and the
multiplication determined by $\iu^2 = -1$. Every nonzero $z = a + \iu
b$ has an inverse: $z^{-1} = \frac{a - \iu b}{a^2 + b^2}$ --- in the
language of \cref{ch:b1:structures}, $\C$ is a field. One writes $a =
\Re(z)$, $b = \Im(z)$, $\conj{z} = a - \iu b$ (the
\emph{conjugate}\index{conjugate}) and $\abs{z} = \sqrt{a^2 + b^2}$
(the \emph{modulus}\index{modulus}).
\end{definition}

\begin{proposition}[Rules for conjugate and modulus]\label{prop:b1:complex:rules}
For $z, w \in \C$:
\begin{enumerate}
  \item $\conj{z + w} = \conj z + \conj w$, $\conj{zw} = \conj z\,
        \conj w$, $\conj{\conj z} = z$;
  \item $z \conj z = \abs z^2$; $\;\Re(z) = \frac{z + \conj z}{2}$,
        $\Im(z) = \frac{z - \conj z}{2\iu}$;
  \item $\abs{zw} = \abs z\, \abs w$, and $\abs{z^{-1}} =
        \abs{z}^{-1}$ for $z \neq 0$;
  \item (triangle inequality\index{triangle inequality}) $\abs{z + w}
        \leq \abs z + \abs w$, with equality if and only if $z$ and
        $w$ lie on a common ray from $0$ ($w = \lambda z$ or
        $z = \lambda w$ with $\lambda \geq 0$);
  \item (reverse triangle inequality) $\bigl|\abs z - \abs w\bigr|
        \leq \abs{z - w}$.
\end{enumerate}
\end{proposition}

\begin{proof}
(1) and (2) are direct computations on real and imaginary parts. (3):
$\abs{zw}^2 = zw\,\conj{zw} = z\conj z\, w \conj w = \abs z^2 \abs
w^2$, then take square roots; apply to $z \cdot z^{-1} = 1$ for the
inverse.

(4) Both sides are nonnegative, so compare squares:
\[
\abs{z+w}^2 = (z+w)(\conj z + \conj w)
= \abs z^2 + \abs w^2 + 2\,\Re(z \conj w),
\]
and $\Re(z\conj w) \leq \abs{z \conj w} = \abs z \abs w$ gives
$\abs{z+w}^2 \leq (\abs z + \abs w)^2$. Equality forces $\Re(z \conj
w) = \abs{z \conj w}$, i.e.\ $z \conj w \in \R_+$; if $w \neq 0$ this
gives $z = \frac{z\conj w}{\abs w^2}\, w = \lambda w$ with $\lambda
\geq 0$ (and the case $w = 0$ is trivial).

(5) $\abs z = \abs{(z - w) + w} \leq \abs{z-w} + \abs w$ gives $\abs z
- \abs w \leq \abs{z - w}$; swap $z$ and $w$ for the other sign.
\end{proof}

\begin{example}[A complete modulus-and-argument workout]\label{ex:b1:complex:workout}
Put $w = \dfrac{3 + 4\iu}{1 - 2\iu}$ in algebraic form and compute
its modulus twice. Multiplying by the conjugate of the
denominator:
\[
w = \frac{(3 + 4\iu)(1 + 2\iu)}{(1 - 2\iu)(1 + 2\iu)}
= \frac{3 + 6\iu + 4\iu - 8}{1 + 4}
= \frac{-5 + 10\iu}{5} = -1 + 2\iu .
\]
Directly: $\abs w = \sqrt{1 + 4} = \sqrt5$. Via the quotient rule
(\cref{prop:b1:complex:rules}~(3)): $\abs w = \frac{\abs{3 +
4\iu}}{\abs{1 - 2\iu}} = \frac5{\sqrt5} = \sqrt5$ --- same answer,
no algebraic form needed. The lesson generalizes: moduli and
arguments travel well through products and quotients, real and
imaginary parts travel well through sums. Choose the
representation that matches the operations at hand, and convert
only when forced.
\end{example}

\begin{example}[Equations involving the conjugate]\label{ex:b1:complex:conjeq}
Solve in $\C$: $\;z + 2\conj z = 6 + 2\iu$. An equation mixing $z$
and $\conj z$ is \emph{not} polynomial in $z$; the reliable move
is to split into real coordinates. With $z = x + \iu y$:
\[
z + 2\conj z = 3x - \iu y ,
\]
so the equation reads $3x = 6$ and $-y = 2$: the unique solution
is $z = 2 - 2\iu$. (Check: $(2 - 2\iu) + 2(2 + 2\iu) = 6 +
2\iu$.) Alternatively, conjugate the whole equation to get
$\conj z + 2z = 6 - 2\iu$ and solve the linear system in the
unknowns $z, \conj z$ --- same answer, and a useful trick when
the coefficients are complex. Equations in $z$ and $\conj z$ are
really systems of two real equations; expecting ``degree $1$, one
solution'' is safe here, but $z\conj z = -1$ (no solution) shows
the polynomial intuition failing as soon as products appear.
\end{example}

\section{Exponential form}

\begin{definition}[Complex exponential of an imaginary argument]\label{def:b1:complex:expi}
For $\theta \in \R$ one defines
\[
\eu^{\iu\theta} = \cos\theta + \iu \sin\theta .
\]
Every $z \neq 0$ can be written $z = r\,\eu^{\iu\theta}$ with $r =
\abs z > 0$; $\theta$ is an \emph{argument}\index{argument} of $z$,
determined up to adding a multiple of $2\pi$. The value in
$\intoc{-\pi}{\pi}$ is the \emph{principal argument}, written
$\arg z$.
\end{definition}

\begin{example}[First values, and one famous identity]\label{ex:b1:complex:eulerid}
Reading the definition at the cardinal angles:
\[
\eu^{\iu\pi/2} = \iu, \qquad
\eu^{\iu\pi} = -1, \qquad
\eu^{2\iu\pi} = 1, \qquad
\eu^{\iu\pi/4} = \frac{\sqrt2}2\,(1 + \iu) .
\]
The second one, rearranged as $\eu^{\iu\pi} + 1 = 0$, is Euler's
celebrated identity linking $\eu$, $\iu$, $\pi$, $1$ and $0$; at
this stage of the book it is a definition unwinding rather than a
theorem, and its real content --- why the analytic exponential
function of \cref{ch:b1:functions}, extended to imaginary
arguments, deserves the same name --- is settled by the power
series of \cref{ch:b1:series}. Meanwhile the display above is
worth memorizing as a conversion table: it is used silently every
time an argument is read off a picture.
\end{example}

\begin{theorem}[The functional equation]\label{thm:b1:complex:funceq}
For all $\theta, \varphi \in \R$:
\[
\eu^{\iu\theta}\, \eu^{\iu\varphi} = \eu^{\iu(\theta + \varphi)},
\qquad
\abs{\eu^{\iu\theta}} = 1,
\qquad
\conj{\eu^{\iu\theta}} = \eu^{-\iu\theta} = (\eu^{\iu\theta})^{-1}.
\]
Consequently $\abs{zw} = \abs z \abs w$ carries the arguments too:
$\arg(zw) \equiv \arg z + \arg w \pmod{2\pi}$.
\end{theorem}

\begin{proof}
Expand the product and use the addition formulas:
\[
(\cos\theta + \iu\sin\theta)(\cos\varphi + \iu\sin\varphi)
= (\cos\theta\cos\varphi - \sin\theta\sin\varphi)
+ \iu\,(\sin\theta\cos\varphi + \cos\theta\sin\varphi),
\]
which is $\cos(\theta+\varphi) + \iu\sin(\theta+\varphi)$. The modulus
is $\sqrt{\cos^2\theta + \sin^2\theta} = 1$, and the conjugate formula
is the parity of cosine and sine; it inverts $\eu^{\iu\theta}$ since
$\eu^{\iu\theta}\eu^{-\iu\theta} = \eu^0 = 1$.
\end{proof}

\begin{corollary}[de Moivre's formula]\label{cor:b1:complex:demoivre}
For $\theta \in \R$ and $n \in \Z$:
$\;(\cos\theta + \iu\sin\theta)^n = \cos(n\theta) +
\iu\sin(n\theta)$.\index{de Moivre's formula}
\end{corollary}

\begin{proof}
For $n \geq 0$, induct. The case $n = 0$ reads $1 = 1$. Assuming
the formula for $n$, the functional equation
(\cref{thm:b1:complex:funceq}) gives
\[
\bigl(\eu^{\iu\theta}\bigr)^{n+1}
= \bigl(\eu^{\iu\theta}\bigr)^{n}\,\eu^{\iu\theta}
= \eu^{\iu n\theta}\,\eu^{\iu\theta}
= \eu^{\iu(n+1)\theta} ,
\]
which is the formula at rank $n + 1$. For $n < 0$, write $n = -m$
with $m > 0$: since $\eu^{\iu m\theta}$ has inverse
$\eu^{-\iu m\theta}$ (\cref{thm:b1:complex:funceq} again),
\[
\bigl(\eu^{\iu\theta}\bigr)^{-m}
= \bigl(\eu^{\iu m\theta}\bigr)^{-1}
= \eu^{\iu(-m)\theta} . \qedhere
\]
\end{proof}

\begin{example}[Expanding $\cos 3\theta$ by de Moivre]\label{ex:b1:complex:cos3}
Write $c = \cos\theta$, $s = \sin\theta$. De Moivre and the binomial
theorem give
\[
\cos 3\theta + \iu \sin 3\theta = (c + \iu s)^3
= c^3 - 3cs^2 + \iu\,(3c^2 s - s^3),
\]
and identifying real parts, then substituting $s^2 = 1 - c^2$:
\[
\cos 3\theta = c^3 - 3c(1 - c^2) = 4\cos^3\theta - 3\cos\theta .
\]
The imaginary part yields $\sin 3\theta = 3\sin\theta -
4\sin^3\theta$ for free: one complex identity always carries
\emph{two} real ones. Read backwards, the boxed identity is the
key to the classical trisection equation: constructing
$\cos(\theta/3)$ from $\cos\theta$ means solving the cubic $4x^3 -
3x = \cos\theta$, which is where the algebra of
\cref{ch:b1:poly} takes over.
\end{example}

\begin{proposition}[Euler's formulas]\label{prop:b1:complex:euler}
\[
\cos\theta = \frac{\eu^{\iu\theta} + \eu^{-\iu\theta}}{2},
\qquad
\sin\theta = \frac{\eu^{\iu\theta} - \eu^{-\iu\theta}}{2\iu}.
\]
\end{proposition}

\begin{proof}
Add, respectively subtract, $\eu^{\iu\theta} = \cos\theta +
\iu\sin\theta$ and $\eu^{-\iu\theta} = \cos\theta - \iu\sin\theta$.
\end{proof}

\begin{method}[Trigonometry via exponentials]\label{met:b1:complex:trig}
\begin{enumerate}
  \item \emph{Linearize} $\cos^p\theta \sin^q\theta$ (turn powers into
        a sum of $\cos k\theta$, $\sin k\theta$): substitute Euler's
        formulas, expand with the binomial theorem
        (\cref{thm:b1:counting:binomial}), regroup conjugate terms.
  \item \emph{Expand} $\cos n\theta$ as a polynomial in $\cos\theta$:
        write $\cos n\theta = \Re\bigl((\cos\theta +
        \iu\sin\theta)^n\bigr)$, expand, and convert even powers of
        $\sin$ via $\sin^2 = 1 - \cos^2$.
  \item \emph{Sum trigonometric series} like $\sum_k \cos k\theta$:
        recognize the real part of a geometric sum $\sum_k
        (\eu^{\iu\theta})^k$.
  \item The \emph{half-angle factorization}: for all $p, q$,
        \[
        \eu^{\iu p} + \eu^{\iu q}
        = 2 \cos\tfrac{p - q}{2}\; \eu^{\iu \frac{p+q}{2}},
        \qquad
        \eu^{\iu p} - \eu^{\iu q}
        = 2\iu \sin\tfrac{p - q}{2}\; \eu^{\iu \frac{p+q}{2}} .
        \]
\end{enumerate}
\end{method}

\begin{example}[Linearization]\label{ex:b1:complex:linearize}
\[
\cos^3\theta
= \Bigl(\frac{\eu^{\iu\theta} + \eu^{-\iu\theta}}{2}\Bigr)^{\!3}
= \frac{\eu^{3\iu\theta} + 3\eu^{\iu\theta} + 3\eu^{-\iu\theta} +
\eu^{-3\iu\theta}}{8}
= \frac{\cos 3\theta + 3\cos\theta}{4}.
\]
This form integrates immediately --- the very reason linearization
matters in \cref{ch:b1:integration}. A mixed product works the
same way, only with both Euler formulas at once:
\[
\sin^2\theta\cos^2\theta
= \Bigl(\frac{\sin2\theta}2\Bigr)^{\!2}
= \frac{1}{4}\cdot
\Bigl(\frac{\eu^{2\iu\theta} - \eu^{-2\iu\theta}}{2\iu}
\Bigr)^{\!2}
= \frac{2 - \eu^{4\iu\theta} - \eu^{-4\iu\theta}}{16}
= \frac{1 - \cos4\theta}{8} ,
\]
where the double-angle shortcut in the first step saved one
binomial expansion --- always worth scanning for before
mechanizing.
\end{example}

\begin{example}[A binomial trigonometric sum]\label{ex:b1:complex:binomsum}
For $n \in \N$ and $\theta \in \R$, evaluate $S =
\sum_{k=0}^{n}\binom nk \cos k\theta$. Recognize the real part of
a binomial expansion:
\[
S = \Re\sum_{k=0}^n \binom nk \bigl(\eu^{\iu\theta}\bigr)^k
= \Re\bigl(1 + \eu^{\iu\theta}\bigr)^n ,
\]
then factor the half-angle (\cref{met:b1:complex:trig}~(4)): $1 +
\eu^{\iu\theta} = 2\cos\frac\theta2\,\eu^{\iu\theta/2}$, so
\[
S = \Re\Bigl(2^n\cos^n\frac\theta2\;\eu^{\iu n\theta/2}\Bigr)
= 2^n \cos^n\frac\theta2\,\cos\frac{n\theta}2 .
\]
The imaginary part yields $\sum_k\binom nk\sin k\theta =
2^n\cos^n\frac\theta2\sin\frac{n\theta}2$ for free. Sanity checks:
$\theta = 0$ recovers $\sum\binom nk = 2^n$, and $\theta = \pi$
gives $S = 0$ for $n \geq 1$ (each factor $\cos\frac\pi2$
vanishes), i.e.\ the alternating row sum of
\cref{ex:b1:counting:binomial}. The method --- ``see the cosine
sum as the shadow of a complex power, then factor half-angles''
--- is exactly that of \cref{exo:b1:complex:6}, with the binomial
theorem replacing the geometric series.
\end{example}

\section{Roots of complex numbers}

\begin{theorem}[$n$-th roots]\label{thm:b1:complex:roots}
Let $a = r\,\eu^{\iu\alpha} \neq 0$ and $n \in \N^*$. The equation
$z^n = a$ has exactly $n$ solutions:
\[
z_k = r^{1/n}\, \eu^{\iu\left(\frac{\alpha}{n} +
\frac{2k\pi}{n}\right)},
\qquad k = 0, 1, \dots, n - 1 .
\]
\end{theorem}

\begin{proof}
Write $z = \rho\,\eu^{\iu\theta}$ ($\rho > 0$). Then $z^n =
\rho^n \eu^{\iu n\theta} = r \eu^{\iu\alpha}$ if and only if $\rho^n
= r$ (moduli) and $n\theta \equiv \alpha \pmod{2\pi}$ (arguments),
i.e.\ $\rho = r^{1/n}$ and $\theta = \frac{\alpha}{n} +
\frac{2k\pi}{n}$ for some $k \in \Z$. It remains to see when two
integers $k, k'$ yield the same number: this happens exactly when
the angles differ by a multiple of $2\pi$,
\[
\frac{2k\pi}n - \frac{2k'\pi}n \in 2\pi\Z
\iff \frac{k - k'}n \in \Z
\iff n \mid k - k' .
\]
By Euclidean division, every $k \in \Z$ is congruent mod $n$ to
exactly one element of $\{0, 1, \dots, n-1\}$, so this range lists
each solution once and the count is exactly $n$. (The same
argument, run inside $\mathbb U_n$, shows the roots form a regular
$n$-gon: consecutive values of $k$ rotate by the fixed angle
$\frac{2\pi}n$.)
\end{proof}

\begin{example}[Cube roots of $-27$]\label{ex:b1:complex:cuberoots}
Solve $z^3 = -27$. Exponential form of the right side: $-27 =
27\,\eu^{\iu\pi}$, so the three roots are
\[
z_k = 3\,\eu^{\iu(\frac\pi3 + \frac{2k\pi}3)}, \quad k = 0, 1, 2 :
\qquad
z_0 = 3\eu^{\iu\pi/3} = \frac32 + \frac{3\sqrt3}2\,\iu,
\quad
z_1 = -3,
\quad
z_2 = \conj{z_0} .
\]
Two checks. First, the real root $-3$ is the obvious one, and the
other two are its rotations by $\pm\frac{2\pi}3$ --- equivalently
$-3j$ and $-3j^2$. Second, algebra confirms: $z^3 + 27 = (z +
3)(z^2 - 3z + 9)$, and the quadratic has discriminant $9 - 36 =
-27 < 0$ with roots $\frac{3 \pm 3\iu\sqrt3}2 = z_0, \conj{z_0}$.
The insight: for real right-hand sides, the non-real roots always
come in conjugate pairs, so a picture of the solution set is
symmetric about the real axis --- a preview of the real
factorization theorem of \cref{ch:b1:poly}.
\end{example}

\begin{example}[A non-real right-hand side]\label{ex:b1:complex:fourthroots}
Solve $z^4 = -8 + 8\iu\sqrt3$. Exponential form of the right side:
modulus $\sqrt{64 + 192} = 16$, argument $\theta$ with
$\cos\theta = -\frac12$, $\sin\theta = \frac{\sqrt3}2$, i.e.\
$\theta = \frac{2\pi}3$. The four roots are
\[
z_k = 2\,\eu^{\iu(\frac\pi6 + \frac{k\pi}2)}, \quad k = 0, 1, 2,
3 :
\qquad
z_0 = \sqrt3 + \iu,\quad z_1 = \iu z_0 = -1 + \iu\sqrt3,
\]
\[
z_2 = -z_0 = -\sqrt3 - \iu,\qquad z_3 = -\iu z_0 = 1 - \iu\sqrt3 .
\]
(Check: $z_0^2 = 2 + 2\iu\sqrt3$, so $z_0^4 = (2 +
2\iu\sqrt3)^2 = 4 - 12 + 8\iu\sqrt3 = -8 + 8\iu\sqrt3$.) Once
\emph{one} root is found, the other three come free: they are its
successive rotations by $\frac\pi2$, i.e.\ its products with the
fourth roots of unity --- the general structure behind
\cref{thm:b1:complex:roots}, worth exploiting before recomputing
each root from scratch. No conjugate symmetry this time: the
right-hand side is not real.
\end{example}

\begin{definition}[Roots of unity]\label{def:b1:complex:unity}
The \emph{$n$-th roots of unity}\index{roots of unity} are the
solutions of $z^n = 1$:
\[
\mathbb{U}_n = \bigl\{\, \omega^k : k = 0, \dots, n-1 \,\bigr\},
\qquad \omega = \eu^{2\iu\pi/n}.
\]
They form a group under multiplication (\cref{ch:b1:structures}) and
sit at the vertices of a regular $n$-gon inscribed in the unit circle.
\end{definition}

\begin{proposition}[Sum of the roots of unity]\label{prop:b1:complex:sumroots}
For $n \geq 2$, the $n$-th roots of unity sum to zero:
$\sum_{k=0}^{n-1} \omega^k = 0$.
\end{proposition}

\begin{proof}
Geometric sum with ratio $\omega \neq 1$:
$\sum_{k=0}^{n-1} \omega^k = \frac{\omega^n - 1}{\omega - 1} = 0$
since $\omega^n = 1$.
\end{proof}

\begin{example}[Reading real and imaginary parts]\label{ex:b1:complex:sumrootsparts}
Splitting $\sum_{k=0}^{n-1}\omega^k = 0$ into real and imaginary
parts yields two trigonometric identities for free:
\[
\sum_{k=0}^{n-1}\cos\frac{2k\pi}n = 0,
\qquad
\sum_{k=0}^{n-1}\sin\frac{2k\pi}n = 0
\qquad (n \geq 2).
\]
Geometrically: the centroid of a regular $n$-gon inscribed in the
unit circle is its center --- the vertices balance out exactly.
For $n = 5$ the first identity gives $1 + 2\cos\frac{2\pi}5 +
2\cos\frac{4\pi}5 = 0$ (pairing $k$ with $n - k$), the starting
point of \cref{exo:b1:complex:8}'s computation of
$\cos\frac{2\pi}5$.
\end{example}

\begin{example}[Square roots in algebraic form]\label{ex:b1:complex:sqrt}
To solve $z^2 = 3 + 4\iu$ without trigonometry, set $z = x + \iu y$:
\[
x^2 - y^2 = 3, \qquad 2xy = 4, \qquad x^2 + y^2 = \abs{3 + 4\iu} = 5 .
\]
Adding the first and last: $x^2 = 4$, so $x = \pm 2$, then $y = 2/x =
\pm 1$ with the \emph{same} sign pairing ($xy = 2 > 0$): $z = \pm(2 +
\iu)$. Combined with the usual formula, this solves every quadratic
equation with complex coefficients
(\cref{exo:b1:complex:7}).
\end{example}

\section{Complex numbers and plane geometry}

\begin{proposition}[Geometric dictionary]\label{prop:b1:complex:geometry}
Identify the point $M(x, y)$ of the plane with its \emph{affix} $z = x
+ \iu y$\index{affix}. For distinct points $A, B, C$ of affixes $a, b,
c$:
\begin{enumerate}
  \item $\abs{b - a}$ is the distance $AB$;
  \item $\arg\dfrac{c - a}{b - a}$ is the angle between the vectors
        $\vect{AB}$ and $\vect{AC}$ (mod $2\pi$);
  \item $A, B, C$ are aligned if and only if $\dfrac{c - a}{b - a} \in
        \R$; the lines $AB$ and $AC$ are perpendicular if and only if
        $\dfrac{c - a}{b - a} \in \iu\R$.
\end{enumerate}
\end{proposition}

\begin{proof}
(1) is the definition of the modulus applied to $b - a$, the affix of
$\vect{AB}$. (2): write $b - a = r\eu^{\iu\theta}$, $c - a =
s\eu^{\iu\varphi}$; then $\frac{c-a}{b-a} = \frac sr
\eu^{\iu(\varphi - \theta)}$ has argument $\varphi - \theta$, the
angle from $\vect{AB}$ to $\vect{AC}$. (3): alignment means angle $0$
or $\pi$, i.e.\ argument in $\pi\Z$, i.e.\ the quotient is real;
perpendicularity means angle $\pm\frac\pi2$, i.e.\ the quotient is
purely imaginary. (The quotient is nonzero since $C \neq A$.)
\end{proof}

\begin{remark}[Interlude: $\C$ is the plane with a multiplication]\label{rem:b1:complex:interlude}
It is worth pausing on what makes this chapter possible at all:
the plane $\R^2$ carries additions in every direction, but no
God-given multiplication --- and $\C$ \emph{is} the plane
equipped with one, in which multiplying by a fixed number
rotates and scales. This one structure will be milked three more
times in this volume. In \cref{ch:b1:matrices}, multiplication by
$a + \iu b$ reappears as the $2 \times 2$ matrix with rows $(a,
-b)$ and $(b, a)$: complex arithmetic is a first, entirely
concrete family of matrix products. In \cref{ch:b1:euclid}, the
formula $\Re(\conj z\,w)$ turns out to be the dot product, and
$\abs z$ the Euclidean norm: the triangle inequality proved here
is the model for the Cauchy--Schwarz story there. And in
\cref{ch:b1:curves}, a moving point is best written $t \mapsto
z(t)$, so that velocity and acceleration become complex-valued
derivatives --- circular motion, for one, is simply $z(t) =
R\,\eu^{\iu\omega t}$. One good multiplication, four chapters of
dividends.
\end{remark}

\begin{proposition}[Maps $z \mapsto az + b$]\label{prop:b1:complex:similitude}
Let $a \in \C^*$, $b \in \C$. The transformation $f(z) = az + b$ of
the plane:
\begin{itemize}
  \item is a translation if $a = 1$;
  \item otherwise has a unique fixed point $\zeta = \frac{b}{1-a}$,
        and $f(z) - \zeta = a\,(z - \zeta)$: $f$ is the rotation of
        center $\zeta$ and angle $\arg a$, composed with the scaling
        of center $\zeta$ and ratio $\abs a$.
\end{itemize}
In particular $z \mapsto \eu^{\iu\theta} z$ is the rotation of angle
$\theta$ about the origin, and $\conj z$ is the reflection in the real
axis.
\end{proposition}

\begin{proof}
If $a = 1$, $f(z) = z + b$ translates by the vector of affix $b$. If
$a \neq 1$, the fixed-point equation $z = az + b$ has the unique
solution $\zeta = \frac{b}{1-a}$, and then $f(z) - \zeta = az + b -
(a\zeta + b) = a(z - \zeta)$. Writing $a = \abs a\, \eu^{\iu\arg a}$,
multiplication by $a$ scales distances to $\zeta$ by $\abs a$ and adds
$\arg a$ to angles at $\zeta$, which is the announced composition.
\end{proof}

\begin{example}[Classifying a map $z \mapsto az+b$]\label{ex:b1:complex:classify}
Take $f(z) = \iu z + 1$. Here $a = \iu \neq 1$: the fixed point is
\[
\zeta = \frac{b}{1 - a} = \frac1{1 - \iu}
= \frac{1 + \iu}{2},
\]
and since $\abs a = 1$, there is no scaling at all: $f$ is the pure
rotation of center $\frac{1+\iu}2$ and angle $\arg \iu = \frac\pi2$.
Check: $f(\zeta) = \iu\,\frac{1+\iu}2 + 1 = \frac{\iu - 1}2 + 1 =
\frac{1 + \iu}2 = \zeta$, and $f(0) = 1$, $f(1) = 1 + \iu$, $f(1 +
\iu) = \iu\,(1 + \iu) + 1 = \iu$: the four points $0, 1, 1+\iu, \iu$
of the unit square cycle around their center $\zeta$, one quarter
turn at a time --- exactly what a rotation by $\frac\pi2$ about the
square's center must do.
\end{example}

\begin{remark}[Common pitfalls with moduli and arguments]\label{rem:b1:complex:pitfalls}
\begin{enumerate}
  \item \emph{No inequalities in $\C$.} Writing $z \leq w$ for
        non-real numbers is meaningless; only moduli, real parts
        and imaginary parts can be compared.
  \item \emph{$\sqrt{\phantom z}$ is reserved for $\R_+$.} Every
        nonzero complex number has \emph{two} square roots and
        neither is privileged: write ``let $\delta$ be a square
        root of $\Delta$'' (computed as in
        \cref{ex:b1:complex:sqrt}), never $\sqrt\Delta$ ---
        the rule $\sqrt{ab} = \sqrt a\sqrt b$ already fails at
        $a = b = -1$.
  \item \emph{Arguments live modulo $2\pi$.} From
        $\eu^{\iu\alpha} = \eu^{\iu\beta}$ conclude $\alpha
        \equiv \beta \pmod{2\pi}$, not $\alpha = \beta$;
        forgetting the $2k\pi$ is how solution sets of $z^n = a$
        lose $n - 1$ of their $n$ elements.
  \item \emph{$\abs{z + w}$ is not $\abs z + \abs w$.} Equality
        in the triangle inequality is the exceptional aligned
        case (\cref{prop:b1:complex:rules}~(4)); in general the
        modulus of a sum must be estimated, not computed.
\end{enumerate}
\end{remark}

\begin{example}\label{ex:b1:complex:equilateral}
$A, B, C$ (affixes $a, b, c$, pairwise distinct) form an equilateral
triangle with the vertices in direct (counterclockwise) order if and
only if $\frac{c - a}{b - a} = \eu^{\iu\pi/3}$: the rotation of
center $A$ and angle $\frac\pi3$ sends $B$ to $C$. Both orientations
together are captured by the symmetric equation $a^2 + b^2 + c^2 = ab
+ bc + ca$ (\cref{exo:b1:complex:10}).
\end{example}

\begin{remark}[Where this chapter is used]\label{rem:b1:complex:whereused}
The exponential form is the single most reused computational device
of the volume. Roots of unity become the standard example of a
cyclic group in \cref{ch:b1:structures} and drive the factorization
of $X^n - 1$ in \cref{ch:b1:poly}; grouping conjugate roots there
produces the real factorizations used by partial fractions in
\cref{ch:b1:fractions}. Linearization (\cref{met:b1:complex:trig})
is the standard preparation for integrating trigonometric powers in
\cref{ch:b1:integration}, and the characteristic equations of
\cref{ch:b1:diffeq} have complex roots whose real and imaginary
parts produce the oscillating solutions $\eu^{\lambda t}\cos\omega
t$. The geometric dictionary returns in matrix clothing: rotations
and similarities become the orthogonal matrices of
\cref{ch:b1:matrices,ch:b1:euclid}, and parametrized curves in
\cref{ch:b1:curves} are often best written as maps $t \mapsto z(t)
\in \C$.
\end{remark}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:b1:complex:1}
Put in exponential form: $1 + \iu$; $\;\sqrt 3 - \iu$; $\;-5$;
$\;\dfrac{1 + \iu}{\sqrt 3 - \iu}$. Deduce
$\cos\frac{5\pi}{12}$ and $\sin\frac{5\pi}{12}$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:complex:2}
Compute $(1 + \iu)^{20}$. For which $n \in \N$ is $(1 + \iu)^n$ a
real number?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:complex:3}
Describe geometrically the set of $z \in \C$ such that:
$\abs{z - 2} = \abs{z + \iu}$; $\;\abs{z - 1} = 2$;
$\;\dfrac{z - 1}{z + 1} \in \iu\R$ (for $z \neq -1$).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:complex:4}
Linearize $\sin^4\theta$, and expand $\cos 4\theta$ as a polynomial in
$\cos\theta$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:complex:5}
Solve $z^3 = 8\iu$ and place the solutions on a picture. Solve
$z^4 = -4$ and factor $X^4 + 4$ into two real quadratic polynomials.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:complex:6}
For $\theta \in \R$ with $\eu^{\iu\theta} \neq 1$ and $n \in \N$,
compute
\[
C_n = \sum_{k=0}^{n} \cos k\theta
\qquad\text{and}\qquad
S_n = \sum_{k=0}^{n} \sin k\theta
\]
by summing the geometric series $\sum_k \eu^{\iu k\theta}$ and using
the half-angle factorization.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:complex:7}
Solve in $\C$: $z^2 - (3 + 4\iu) z + (-1 + 5\iu) = 0$. \emph{(Compute
the discriminant, extract its square roots as in
\cref{ex:b1:complex:sqrt}.)}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:complex:8}
Let $\omega = \eu^{2\iu\pi/5}$.
\begin{enumerate}
  \item Justify $1 + \omega + \omega^2 + \omega^3 + \omega^4 = 0$.
  \item Set $u = \omega + \omega^4$ and $v = \omega^2 + \omega^3$.
        Compute $u + v$ and $uv$, and deduce that $u$ and $v$ are the
        roots of $X^2 + X - 1$.
  \item Conclude that $\cos\frac{2\pi}{5} = \frac{\sqrt 5 - 1}{4}$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:complex:9}
Prove that for all $z, w \in \C$ (parallelogram identity):
\[
\abs{z + w}^2 + \abs{z - w}^2 = 2\abs z^2 + 2\abs w^2 ,
\]
and interpret it geometrically in the parallelogram with vertices $0,
z, w, z + w$.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:complex:10}
Prove that three pairwise distinct points of affixes $a, b, c$ form an
equilateral triangle (either orientation) if and only if
\[
a^2 + b^2 + c^2 = ab + bc + ca .
\]
\emph{Hint: the direct case is $\frac{c-a}{b-a} = -j^2$ and the
indirect case $\frac{c-a}{b-a} = -j$, where $j = \eu^{2\iu\pi/3}$
satisfies $j^2 + j + 1 = 0$; or factor $a + jb + j^2c$ and $a + j^2 b
+ jc$.}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:complex:11}
For $n \in \N^*$, evaluate $P = \prod_{k=1}^{n-1}
\bigl(1 - \omega^k\bigr)$ where $\omega = \eu^{2\iu\pi/n}$.
\emph{Hint: $X^n - 1 = \prod_{k=0}^{n-1} (X - \omega^k)$; divide by
$X - 1$ and evaluate at $X = 1$.} Deduce
$\prod_{k=1}^{n-1} \sin\frac{k\pi}{n} = \dfrac{n}{2^{n-1}}$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:complex:12}
Let $n \geq 2$. Solve the equation $(z + 1)^n = (z - 1)^n$ in $\C$:
show that it has exactly $n - 1$ solutions, all purely imaginary,
namely
\[
z_k = -\iu\,\frac{\cos\frac{k\pi}{n}}{\sin\frac{k\pi}{n}},
\qquad k = 1, \dots, n - 1 .
\]
\emph{Hint: $z = 1$ is not a solution, so divide and use the roots of
unity; then apply the half-angle factorization of
\cref{met:b1:complex:trig}.}
\end{exercise}

\section{Problem: Napoleon's theorem}

\begin{problem}\label{pb:b1:complex:1}
Erect an equilateral triangle outward on each side of an
\emph{arbitrary} triangle: the three centers of those triangles
always form an equilateral triangle. This is Napoleon's theorem
\index{Napoleon's theorem} --- a statement with no visible reason to
be true, which the algebra of $j = \eu^{2\iu\pi/3}$ proves in three
lines of computation. This problem builds the full toolkit
(rotations, direct similarities, the $j$-criterion of
\cref{exo:b1:complex:10}), proves Napoleon's theorem inner and
outer, locates the degenerate case, and closes with a second jewel
of the same school: van Schooten's theorem on equilateral triangles
and Ptolemy's inequality. Throughout, points of the plane are
identified with their affixes.

\medskip
\noindent\textbf{Part I --- The number $j$ and rotations.}
\begin{enumerate}
  \item Compute the algebraic form of $j$, then $j^2$, $j^3$, $1 + j
        + j^2$, $\conj j$ and $j^{-1}$. Place $1$, $j$, $j^2$ on a
        sketch of the unit circle.
  \item Show that the rotation of center $a$ and angle $\theta$ is
        $r(z) = a + \eu^{\iu\theta}(z - a)$, and that the
        composition of two rotations, of angles $\theta$ and
        $\theta'$, is a rotation of angle $\theta + \theta'$ if
        $\theta + \theta' \notin 2\pi\Z$, and a translation
        otherwise. (Use \cref{prop:b1:complex:similitude}.)
  \item Let $b \neq c$. Show that there are exactly two points $p$
        making $(b, c, p)$ equilateral, namely $p = b +
        \eu^{\pm\iu\pi/3}(c - b)$. For a \emph{direct} triangle
        $(a, b, c)$, check on the example $a = 0$, $b = 1$, $c =
        \iu$ that the choice $\eu^{-\iu\pi/3}$ is the one lying on
        the far side of the line $BC$ from $a$ --- the
        \emph{outward} apex.
  \item From the solution of \cref{exo:b1:complex:10}, $(a, b, c)$
        is direct equilateral iff $a + jb + j^2c = 0$. Prove the two
        rotation identities
        \[
        b + jc + j^2a = j^2\,(a + jb + j^2c),
        \qquad
        c + ja + j^2b = j\,(a + jb + j^2c),
        \]
        and deduce that the criterion is invariant under cyclic
        permutation of $(a, b, c)$.
  \item Show that if $a + jb + j^2c = 0$ and two of the three points
        coincide, then all three coincide. (So the criterion exactly
        characterizes: direct equilateral triangle, or a single
        point.)
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Direct similarities.}
\begin{enumerate}[resume]
  \item Show that the maps $f(z) = az + b$ with $a \in \C^*$
        (\emph{direct similarities}) are stable under composition
        and inversion: the composition of two of them, and the
        inverse of any of them, is again of this form. (In the
        language of \cref{ch:b1:structures}: they form a group.)
  \item Given $z_1 \neq z_2$ and $w_1 \neq w_2$, show there is a
        \emph{unique} direct similarity $f$ with $f(z_1) = w_1$ and
        $f(z_2) = w_2$, and give $a$ and $b$ explicitly.
  \item Show that a direct similarity $f(z) = az + b$ multiplies
        all distances by $\abs a$ and preserves the \emph{shape}
        $\frac{c - a'}{b' - a'}$ of any triangle $(a', b', c')$
        \emph{(numerator and denominator are both multiplied by
        $a$)}. Deduce that two triangles are directly similar
        exactly when their shapes are equal.
  \item Determine completely the direct similarity with $f(0) = 1$
        and $f(1) = \iu$: give $a$, $b$, the fixed point, the ratio
        and the angle.
  \item Let $\alpha + \beta = 1$. Show that the ``weighted point''
        $g(a', b') = \alpha a' + \beta b'$ commutes with every
        direct similarity: $f(\alpha a' + \beta b') = \alpha f(a')
        + \beta f(b')$. Deduce that centroids, midpoints, and the
        Napoleon centers below are all transported by similarities
        --- the algebraic license behind every ``without loss of
        generality, place the circumcircle at the unit circle''
        argument.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Napoleon's theorem.} Let $(a, b, c)$
be a direct triangle. On each side erect the outward equilateral
triangle (question 3) and let $n_a$, $n_b$, $n_c$ be the centers
(centroids) of the triangles erected on $[b, c]$, $[c, a]$,
$[a, b]$ respectively.
\begin{enumerate}[resume]
  \item Show that
        \[
        n_a = \alpha b + \beta c, \qquad
        n_b = \alpha c + \beta a, \qquad
        n_c = \alpha a + \beta b,
        \qquad\text{with}\quad
        \alpha = \frac{3 + \iu\sqrt3}{6},\ \beta = \conj\alpha .
        \]
  \item Check $\alpha + \beta = 1$, and deduce that the triangle
        $(n_a, n_b, n_c)$ has the \emph{same centroid} as
        $(a, b, c)$.
  \item Using the identities of question 4, show that
        \[
        n_a + j\,n_b + j^2 n_c
        = (\alpha j^2 + \beta j)\,(a + jb + j^2c) .
        \]
  \item Compute $\alpha j + \beta$ and conclude: $\alpha j^2 +
        \beta j = j(\alpha j + \beta) = 0$, so $n_a + j n_b + j^2
        n_c = 0$ for \emph{every} triangle: the outer Napoleon
        triangle is direct equilateral (or a point). Napoleon's
        theorem is proved.
  \item Erect the equilateral triangles \emph{inward} instead
        (choice $\eu^{+\iu\pi/3}$ in question 3) and let $m_a, m_b,
        m_c$ be their centers. Show $m_a = \beta b + \alpha c$ (and
        cyclically), then prove $m_a + j^2 m_b + j\,m_c = 0$: the
        inner Napoleon triangle is equilateral too, with the
        opposite orientation.
  \item Locate the degeneracy: show that $n_a = n_b = n_c$ happens
        exactly when $a + j^2 b + jc = 0$, i.e.\ when $(a, b, c)$
        is an \emph{indirect} equilateral triangle. \emph{(Two
        points among $n_a, n_b, n_c$ coincide iff all three do, by
        question 5; then use $n_a + j^2 n_b + j n_c = j(\alpha +
        \beta j)(a + j^2b + jc)$ and check $\alpha + \beta j \neq
        0$.)}
  \item Carry out the whole computation on the triangle $a = 0$,
        $b = 1$, $c = \iu$: give $n_a, n_b, n_c$ exactly, and
        verify by direct computation of the three squared side
        lengths that the triangle is equilateral, of squared side
        $\frac{2 + \sqrt3}{3}$.
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Ptolemy and van Schooten.}
\begin{enumerate}[resume]
  \item Prove the identity, valid for all complex $a, b, c, d$:
        \[
        (a - b)(c - d) + (a - d)(b - c) = (a - c)(b - d) .
        \]
  \item Deduce \emph{Ptolemy's inequality}\index{Ptolemy's
        inequality}: for any four points $A, B, C, D$,
        \[
        AC \cdot BD \;\leq\; AB \cdot CD + AD \cdot BC ,
        \]
        with equality if and only if $(a-b)(c-d)$ and
        $(a-d)(b-c)$ lie on a common ray from $0$.
  \item Show that for $\theta, \varphi \in \R$,
        $\abs{\eu^{\iu\theta} - \eu^{\iu\varphi}} =
        2\,\abs{\sin\frac{\theta - \varphi}2}$.
  \item (van Schooten's theorem\index{van Schooten's theorem}) Let
        $(a, b, c) = (1, j, j^2)$ --- by Part II this loses no
        generality among direct equilateral triangles --- and let
        $p = \eu^{\iu\theta}$ with $\theta \in
        \intoo{2\pi/3}{4\pi/3}$, a point of the circumcircle on the
        arc $BC$ not containing $A$. Prove
        \[
        PA = PB + PC .
        \]
        \emph{(Express the three distances with question 20 and use
        a sum-to-product formula.)}
  \item Verify at $p = -1$ that this is exactly the equality case
        of Ptolemy's inequality for the cyclic order $A, B, P, C$:
        compute $(a - b)(p - c)$ and $(a - c)(b - p)$ and check
        that their ratio is a positive real.
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Synthesis.}
\begin{enumerate}[resume]
  \item Compute the outer Napoleon triangle of the equilateral
        triangle $(1, j, j^2)$ itself, and describe the result
        geometrically.
  \item Where exactly did the problem use: (i) multiplication as
        rotation; (ii) the group structure of similarities and the
        shape invariant; (iii) the half-angle factorization of
        \cref{met:b1:complex:trig}? One sentence each.
  \item In a short paragraph, articulate the moral of the problem:
        what does the dictionary between plane geometry and the
        algebra of $\C$ buy, what does it cost, and which of the
        two proofsteps --- the identity $\alpha j + \beta = 0$ or
        the classical picture --- \emph{explains} Napoleon's
        theorem better in your view? Mention one place where the
        dictionary will reappear in matrix form later in this
        volume.
\end{enumerate}
\end{problem}
