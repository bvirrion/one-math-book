\chapter{Logic, Sets and Maps}\label{ch:b1:logic}

Up to now, proofs have been carried out with an informal but honest idea
of what ``proving'' means. This first chapter of undergraduate
mathematics makes the rules of the game explicit: what a mathematical
statement is, how the connectives and quantifiers combine statements,
which moves are legal in a proof --- and then builds, on this footing,
the two universal languages of mathematics: sets and maps.

\section{Statements and connectives}

\begin{definition}[Statement, connectives]\label{def:b1:logic:statement}
A \emph{statement}\index{statement} (or \emph{proposition}) is a
sentence which is either true (T) or false (F) --- exactly one of the
two. From statements $P$ and $Q$ one forms:
\begin{itemize}
  \item the \emph{negation} $\lnot P$ (``not $P$''), true exactly when
        $P$ is false;
  \item the \emph{conjunction} $P \land Q$ (``$P$ and $Q$''), true
        exactly when both are true;
  \item the \emph{disjunction} $P \lor Q$ (``$P$ or $Q$''), true
        exactly when at least one is true (this ``or'' is inclusive);
  \item the \emph{implication} $P \implies Q$, false exactly when $P$
        is true and $Q$ is false;
  \item the \emph{equivalence} $P \iff Q$, true exactly when $P$ and
        $Q$ have the same truth value.
\end{itemize}
\end{definition}

\begin{remark}
The truth table of $P \implies Q$ deserves a pause: when $P$ is false,
$P \implies Q$ is \emph{true}, whatever $Q$ is. ``If $2 < 1$ then
$0 = 5$'' is a true implication. An implication asserts nothing about
what happens when its hypothesis fails.
\end{remark}

\begin{proposition}[Rules of computation on statements]\label{prop:b1:logic:rules}
For all statements $P$, $Q$, $R$:
\begin{enumerate}
  \item $\lnot(\lnot P) \iff P$;
  \item De Morgan's laws\index{De Morgan's laws}:
        $\lnot(P \land Q) \iff (\lnot P) \lor (\lnot Q)$ and
        $\lnot(P \lor Q) \iff (\lnot P) \land (\lnot Q)$;
  \item $(P \implies Q) \iff \bigl((\lnot P) \lor Q\bigr)$, hence
        $\lnot(P \implies Q) \iff P \land (\lnot Q)$;
  \item contraposition\index{contraposition}:
        $(P \implies Q) \iff \bigl((\lnot Q) \implies (\lnot P)\bigr)$;
  \item $(P \iff Q) \iff \bigl((P \implies Q) \land (Q \implies
        P)\bigr)$;
  \item distributivity: $P \land (Q \lor R) \iff (P \land Q) \lor
        (P \land R)$ and $P \lor (Q \land R) \iff (P \lor Q) \land
        (P \lor R)$.
\end{enumerate}
\end{proposition}

\begin{proof}
Each equivalence is checked by comparing truth tables: two compound
statements built from $P$, $Q$, $R$ are equivalent exactly when they
take the same truth value in each of the (four or eight) cases. Let us
display one table in full, for the first De Morgan law:
\begin{center}
\begin{tabular}{cc|c|c|cc|c}
$P$ & $Q$ & $P \land Q$ & $\lnot(P \land Q)$ & $\lnot P$ & $\lnot Q$ &
$(\lnot P) \lor (\lnot Q)$ \\
\hline
T & T & T & F & F & F & F \\
T & F & F & T & F & T & T \\
F & T & F & T & T & F & T \\
F & F & F & T & T & T & T \\
\end{tabular}
\end{center}
Columns $4$ and $7$ coincide, which proves the law. For contraposition
a verbal shortcut is quicker: $P \implies Q$ is false exactly in the
case ($P$ true, $Q$ false), and $(\lnot Q) \implies (\lnot P)$ is false
exactly in the case ($\lnot Q$ true, $\lnot P$ false), i.e.\ ($Q$
false, $P$ true) --- the same single case, so the two implications have
identical tables. The remaining rules are checked the same way; note
that (3) reduces every implication to a disjunction, so that (2)
mechanically produces the negation rule $\lnot(P \implies Q) \iff P
\land (\lnot Q)$: to contradict an implication one must exhibit a case
where the hypothesis holds and the conclusion fails.
\end{proof}

\section{Quantifiers}

\begin{definition}[Quantifiers]\label{def:b1:logic:quantifiers}
Let $P(x)$ be a property of an element $x$ of a set $E$.
\begin{itemize}
  \item $\forall x \in E,\ P(x)$ (``for all $x$ in $E$, $P(x)$'') is
        true when every element of $E$ satisfies $P$;
  \item $\exists x \in E,\ P(x)$ (``there exists $x$ in $E$ such that
        $P(x)$'') is true when at least one element of $E$ satisfies
        $P$.
\end{itemize}
One writes $\exists!$ for ``there exists a unique''.
\end{definition}

\begin{proposition}[Negation of quantifiers]\label{prop:b1:logic:negquant}
\[
\lnot\bigl(\forall x \in E,\ P(x)\bigr) \iff
\exists x \in E,\ \lnot P(x),
\qquad
\lnot\bigl(\exists x \in E,\ P(x)\bigr) \iff
\forall x \in E,\ \lnot P(x).
\]
\end{proposition}

\begin{proof}
Let us argue the first equivalence in both directions; the second
is symmetric. If $\forall x \in E,\ P(x)$ is false, then not every
element satisfies $P$: the set $A = \{x \in E : \lnot P(x)\}$
cannot be empty, and any of its elements witnesses $\exists x \in
E,\ \lnot P(x)$. Conversely, if some $x_0 \in E$ satisfies $\lnot
P(x_0)$, then $x_0$ is a counterexample and the universal statement
fails. For the second rule: ``no $x$ satisfies $P$'' means the set
$\{x : P(x)\}$ is empty, i.e.\ every $x$ lies in its complement
$A$. Applied in cascade to a nested prefix of quantifiers, the two
rules give the mechanical procedure of \cref{ex:b1:logic:limit}:
the negation walks left to right, flipping every $\forall$ into
$\exists$ and every $\exists$ into $\forall$, and finally negates
the innermost predicate.
\end{proof}

\begin{example}[Negating everyday mathematical sentences]\label{ex:b1:logic:negpractice}
Let $f \colon \R \to \R$. The sentence ``$f$ is increasing'' reads
\[
\forall x \in \R,\ \forall y \in \R,\quad
x \leq y \implies f(x) \leq f(y) ,
\]
and its negation, by \cref{prop:b1:logic:negquant} plus the rule
$\lnot(P \implies Q) \iff P \land \lnot Q$:
\[
\exists x \in \R,\ \exists y \in \R,\quad
x \leq y \ \text{ and }\ f(x) > f(y) :
\]
a single witnessing pair suffices. Likewise ``$f$ is bounded'' is
$\exists M \in \R,\ \forall x \in \R,\ \abs{f(x)} \leq M$, with
negation
\[
\forall M \in \R,\ \exists x \in \R,\quad \abs{f(x)} > M :
\]
\emph{whatever} bound is proposed, some point beats it. The
insight: a correct negation never contains ``not'' applied to a
quantified block --- it is a new positive statement, in which the
roles are exchanged: one now produces the witnesses one previously
received.
\end{example}

\begin{example}[Order of quantifiers]\label{ex:b1:logic:order}
The order of different quantifiers matters:
\[
\forall x \in \R,\ \exists y \in \R,\ y > x
\quad\text{is true (take } y = x+1\text{),}
\]
\[
\exists y \in \R,\ \forall x \in \R,\ y > x
\quad\text{is false (no real number exceeds all reals).}
\]
In the first statement $y$ may depend on $x$; in the second, one single
$y$ must work for all $x$. Two identical quantifiers, on the other
hand, always commute.
\end{example}

\begin{example}[Reading a definition with three quantifiers]\label{ex:b1:logic:limit}
The sentence ``the sequence $(u_n)$ converges to $\ell$'' will be
written in \cref{ch:b1:seq} as
\[
\forall \varepsilon > 0,\ \exists N \in \N,\ \forall n \geq N,\quad
\abs{u_n - \ell} \leq \varepsilon .
\]
Its negation, by \cref{prop:b1:logic:negquant} applied three times, is
\[
\exists \varepsilon > 0,\ \forall N \in \N,\ \exists n \geq N,\quad
\abs{u_n - \ell} > \varepsilon .
\]
Being able to negate such sentences mechanically, without thinking
about what they mean, is a genuine skill: it separates the logical work
from the mathematical work.
\end{example}

\section{Proof techniques}

\begin{method}[The standard proof patterns]\label{met:b1:logic:proofs}
To prove\dots
\begin{enumerate}
  \item \emph{an implication $P \implies Q$ directly}: assume $P$,
        deduce $Q$;
  \item \emph{by contraposition}: assume $\lnot Q$, deduce $\lnot P$
        --- valid by \cref{prop:b1:logic:rules}~(4);
  \item \emph{by contradiction}: assume the statement is false, derive
        a contradiction;
  \item \emph{an equivalence}: prove both implications separately (or
        chain known equivalences);
  \item \emph{a ``for all'' statement}: pick an \emph{arbitrary} $x$
        in $E$ (``let $x \in E$'') and prove $P(x)$;
  \item \emph{a ``there exists'' statement}: exhibit a witness, or
        prove existence indirectly;
  \item \emph{by induction}: see \cref{thm:b1:logic:induction}.
\end{enumerate}
When proving a statement about a well-chosen but arbitrary element,
never give the element extra properties: ``let $x \in \R$'' followed by
``since $x > 0$\dots'' proves nothing about negative $x$.
\end{method}

\begin{remark}[Common pitfalls in proofs]\label{rem:b1:logic:pitfalls}
Four classical traps, all worth naming once.
\begin{enumerate}
  \item \emph{Converse instead of contrapositive.} $Q \implies P$
        is \emph{not} equivalent to $P \implies Q$; only $\lnot Q
        \implies \lnot P$ is. ``If it rains, the street is wet''
        does not entitle one to conclude rain from a wet street.
  \item \emph{Proving an equivalence by one implication.} An
        ``iff'' claim is two theorems; announce which direction is
        being proved, and prove both. Chains of $\iff$ are legal
        only if \emph{every} link is genuinely reversible ---
        squaring an equation, for instance, is not.
  \item \emph{Backwards proofs.} Starting from the desired
        conclusion and deducing a true statement proves nothing
        (from $-1 = 1$ one derives the true $1 = 1$ by squaring).
        A computation may be \emph{discovered} backwards, but it
        must be \emph{written} forwards, or with explicit
        equivalences.
  \item \emph{Fixed witness versus arbitrary element.} To prove
        $\exists x,\ P(x)$, one may exhibit a single cleverly
        chosen $x$; to prove $\forall x,\ P(x)$, the chosen $x$
        must remain arbitrary. Mixing the two --- checking a
        universal claim on an example --- is the most common
        error in beginners' copies.
\end{enumerate}
\end{remark}

\begin{example}[Contraposition and contradiction at work]\label{ex:b1:logic:sqrt2}
\emph{For $n \in \N$: if $n^2$ is even then $n$ is even.} By
contraposition: if $n$ is odd, $n = 2k+1$, then
$n^2 = 4k^2 + 4k + 1$ is odd.

\emph{$\sqrt 2$ is irrational.} By contradiction: suppose
$\sqrt 2 = p/q$ with $p, q \in \N^*$ and the fraction in lowest terms.
Then $p^2 = 2q^2$ is even, so $p$ is even (previous point), $p = 2r$;
then $q^2 = 2r^2$ is even, so $q$ is even --- contradicting lowest
terms.
\end{example}

\begin{theorem}[Induction]\label{thm:b1:logic:induction}
Let $P(n)$ be a property of the integer $n$. If
\begin{enumerate}
  \item $P(0)$ is true, and
  \item for all $n \in \N$, $P(n) \implies P(n+1)$,
\end{enumerate}
then $P(n)$ is true for all $n \in \N$.

\emph{Strong induction:} conclusion unchanged if (2) is replaced by:
for all $n$, $\bigl(P(0) \land \dots \land P(n)\bigr) \implies
P(n+1)$.
\end{theorem}

\begin{proof}
This is a property of $\N$ itself, equivalent to: \emph{every nonempty
subset of $\N$ has a least element} (which we take as known). Indeed,
suppose (1) and (2) hold and let $A = \{n \in \N : P(n)
\text{ false}\}$. If $A \neq \emptyset$, it has a least element $m$;
$m \neq 0$ by (1); then $m - 1 \notin A$, so $P(m-1)$ holds, and (2)
gives $P(m)$ --- contradiction. So $A = \emptyset$. For strong
induction, apply the same argument: $P(0), \dots, P(m-1)$ all hold
since $m$ is least in $A$.
\end{proof}

\begin{example}[Proving unique existence]\label{ex:b1:logic:uniqueexistence}
A statement $\exists!\,x,\ P(x)$ is \emph{two} statements, proved
separately: existence (exhibit or construct some $x_0$ with
$P(x_0)$) and uniqueness (assume $P(x)$ and $P(x')$, deduce $x =
x'$). Sample: \emph{there is a unique real $x$ with $x^3 + x =
2$.} Existence: $x_0 = 1$ works, since $1 + 1 = 2$. Uniqueness:
if $x^3 + x = x'^3 + x'$, then
\[
0 = (x^3 - x'^3) + (x - x')
= (x - x')\,\bigl(x^2 + xx' + x'^2 + 1\bigr),
\]
and the second factor is positive (it equals $\bigl(x +
\tfrac{x'}2\bigr)^2 + \tfrac34 x'^2 + 1 \geq 1$), so $x = x'$.
Note the division of labor: existence used a lucky guess,
uniqueness used algebra valid for \emph{arbitrary} solutions ---
neither argument does the other's job, and forgetting the second
half is a standing temptation once a solution has been found.
\end{example}

\begin{example}\label{ex:b1:logic:induction}
For all $n \in \N^*$: $\;\sum_{k=1}^n k = \frac{n(n+1)}{2}$. Base case
$n = 1$: both sides equal $1$. Step: assuming the formula for $n$,
\[
\sum_{k=1}^{n+1} k = \frac{n(n+1)}{2} + (n+1)
= (n+1)\Bigl(\frac n2 + 1\Bigr) = \frac{(n+1)(n+2)}{2}. \qedhere
\]
\end{example}

\begin{example}[Strong induction at work]\label{ex:b1:logic:strongind}
\emph{Every integer $n \geq 2$ is a product of prime numbers} (a prime
being an integer $\geq 2$ whose only divisors $\geq 1$ are $1$ and
itself; primes are studied for their own sake in \cref{ch:b1:arith}).
Ordinary induction is helpless here: knowing that $95 = 5 \times 19$
factors says nothing about $96$. Strong induction fits exactly. Base
case: $2$ is prime, hence a (one-factor) product of primes. Step: let
$n \geq 2$ and assume every integer $m$ with $2 \leq m \leq n$ is a
product of primes. If $n + 1$ is prime, done. Otherwise $n + 1 = ab$
with $2 \leq a, b \leq n$; by the strong hypothesis both $a$ and $b$
are products of primes, hence so is $n + 1$. The insight: strong
induction is the right tool whenever the ``reason'' for $P(n+1)$ lives
at some unpredictable earlier rank, not at rank $n$.
\end{example}

\section{Sets}

\begin{definition}[Set operations]\label{def:b1:logic:sets}
We take the notion of \emph{set}\index{set} and the membership relation
$x \in E$ as primitive. For sets $A, B$ inside an ambient set $E$:
\begin{itemize}
  \item \emph{inclusion}: $A \subseteq B$ when $\forall x,\ x \in A
        \implies x \in B$; equality $A = B$ when $A \subseteq B$ and
        $B \subseteq A$;
  \item \emph{union} $A \cup B$, \emph{intersection} $A \cap B$,
        \emph{difference} $A \setminus B = \{x \in A : x \notin B\}$,
        \emph{complement} $\overline{A} = E \setminus A$;
  \item the \emph{empty set} $\emptyset$, contained in every set;
  \item the \emph{power set}\index{power set} $\mathcal{P}(E)$: the set
        of all subsets of $E$;
  \item the \emph{product} $E \times F$: the set of ordered pairs
        $(x, y)$ with $x \in E$, $y \in F$.
\end{itemize}
\end{definition}

\begin{example}[Getting used to the power set]\label{ex:b1:logic:powerset}
For $E = \{a, b\}$:
\[
\mathcal P(E) = \bigl\{\, \emptyset,\ \{a\},\ \{b\},\ \{a, b\}
\,\bigr\},
\]
four elements --- and note the type discipline: $a \in E$ but
$\{a\} \in \mathcal P(E)$; the statements $a \in \mathcal P(E)$
and $\{a\} \subseteq \mathcal P(E)$ are both false as written
(the second would require $a$ to be a \emph{subset} of $E$).
Iterating from nothing: $\mathcal P(\emptyset) = \{\emptyset\}$
has one element, $\mathcal P(\mathcal P(\emptyset)) = \{\emptyset,
\{\emptyset\}\}$ has two, the next has four --- sets of sets are
ordinary sets, and \cref{ch:b1:counting} will confirm the doubling
pattern: $\abs{\mathcal P(E)} = 2^{\abs E}$. Keeping the levels
($x$, $\{x\}$, $\{\{x\}\}$) straight is half the battle in
exercises like \cref{exo:b1:logic:11,exo:b1:logic:12}.
\end{example}

\begin{proposition}[Set algebra]\label{prop:b1:logic:setalgebra}
For subsets $A, B, C$ of $E$:
\begin{enumerate}
  \item $A \cap (B \cup C) = (A \cap B) \cup (A \cap C)$ and
        $A \cup (B \cap C) = (A \cup B) \cap (A \cup C)$;
  \item De Morgan: $\overline{A \cup B} = \overline{A} \cap
        \overline{B}$ and $\overline{A \cap B} = \overline{A} \cup
        \overline{B}$;
  \item $A \subseteq B \iff \overline{B} \subseteq \overline{A}$.
\end{enumerate}
\end{proposition}

\begin{proof}
Each identity translates a rule of \cref{prop:b1:logic:rules} through
the dictionary ($\in A$ or not) $\leftrightarrow$ (statement true or
false): e.g.\ $x \in \overline{A \cup B} \iff \lnot(x \in A \lor x \in
B) \iff (x \notin A) \land (x \notin B) \iff x \in \overline{A} \cap
\overline{B}$. Point (3) is contraposition. As a second sample, the
first distributivity law in full:
\[
x \in A \cap (B \cup C)
\iff (x \in A) \land \bigl(x \in B \lor x \in C\bigr)
\iff \bigl(x \in A \land x \in B\bigr) \lor
\bigl(x \in A \land x \in C\bigr),
\]
by the distributivity of \cref{prop:b1:logic:rules}~(6), and the
last statement reads $x \in (A \cap B) \cup (A \cap C)$. Every set
identity of this kind is provable by this one mechanical
translation --- which is why none of them needs memorizing.
\end{proof}

\begin{method}[Proving set equalities]\label{met:b1:logic:doubleinc}
To prove $A = B$, prove the two inclusions: let $x \in A$, show $x \in
B$; then let $x \in B$, show $x \in A$. Alternatively, chain
equivalences $x \in A \iff \dots \iff x \in B$ when each step really is
an equivalence.
\end{method}

\section{Maps}

\begin{definition}[Map, image, preimage]\label{def:b1:logic:map}
A \emph{map}\index{map} (or \emph{function}) $f \colon E \to F$
assigns to every element $x$ of the set $E$ (the \emph{domain}) exactly
one element $f(x)$ of the set $F$ (the \emph{codomain}). For $A
\subseteq E$ and $B \subseteq F$:
\[
f(A) = \{f(x) : x \in A\} \subseteq F,
\qquad
f^{-1}(B) = \{x \in E : f(x) \in B\} \subseteq E
\]
are the \emph{direct image} of $A$ and the \emph{preimage}
\index{preimage} of $B$. The \emph{composition} of $f \colon E \to F$
and $g \colon F \to G$ is $g \circ f \colon E \to G$, $x \mapsto
g(f(x))$.
\end{definition}

\begin{remark}
The notation $f^{-1}(B)$ does \emph{not} presuppose an inverse map:
$f^{-1}(B)$ is defined for every $f$. Preimages behave better than
images: $f^{-1}$ preserves unions, intersections and complements,
whereas $f(A \cap A') \subseteq f(A) \cap f(A')$ can be strict
(\cref{exo:b1:logic:8}).
\end{remark}

\begin{example}[Computing images and preimages]\label{ex:b1:logic:images}
Let $f \colon \R \to \R$, $x \mapsto x^2$. Then:
\[
f\bigl(\intcc{-1}{2}\bigr) = \intcc04, \qquad
f^{-1}\bigl(\intcc14\bigr) = \intcc{-2}{-1} \cup \intcc12, \qquad
f^{-1}(\{-1\}) = \emptyset .
\]
For the first: every $x \in \intcc{-1}2$ has $x^2 \in \intcc04$, and
every $y \in \intcc04$ is attained as $y = (\sqrt y)^2$ with $\sqrt y
\in \intcc02 \subseteq \intcc{-1}2$ --- note that the image is
\emph{not} $\intcc14 = \{(-1)^2, 2^2\}$: images of intervals are not
computed from endpoints alone. For the second: $1 \leq x^2 \leq 4 \iff
1 \leq \abs x \leq 2$, which splits into two pieces. The third
illustrates that a preimage may be empty --- $f^{-1}(B)$ always makes
sense, however small the intersection of $B$ with the image. Finally
observe on this example the strictness phenomenon of the remark above:
with $A = \intcc{-1}0$ and $A' = \intcc01$, one has $f(A \cap A') =
f(\{0\}) = \{0\}$, while $f(A) \cap f(A') = \intcc01$.
\end{example}

\begin{definition}[Injective, surjective, bijective]\label{def:b1:logic:inj}
A map $f \colon E \to F$ is:
\begin{itemize}
  \item \emph{injective}\index{injective} when distinct elements have
        distinct images: $\forall x, x' \in E,\ f(x) = f(x') \implies
        x = x'$;
  \item \emph{surjective}\index{surjective} when every element of $F$
        is attained: $\forall y \in F,\ \exists x \in E,\ f(x) = y$;
  \item \emph{bijective}\index{bijective} when it is both, i.e.\ every
        $y \in F$ has exactly one preimage.
\end{itemize}
\end{definition}

\begin{theorem}[Inverse map]\label{thm:b1:logic:inverse}
A map $f \colon E \to F$ is bijective if and only if there is a map
$g \colon F \to E$ with $g \circ f = \mathrm{id}_E$ and $f \circ g =
\mathrm{id}_F$. In that case $g$ is unique; it is written $f^{-1}$ and
called the \emph{inverse} of $f$, and $f^{-1}$ is itself bijective with
$(f^{-1})^{-1} = f$.
\end{theorem}

\begin{proof}
($\Rightarrow$) If $f$ is bijective, every $y \in F$ has a unique
preimage; define $g(y)$ to be that preimage. Then $f(g(y)) = y$ by
construction, and $g(f(x)) = x$ because $x$ is \emph{the} preimage of
$f(x)$.

($\Leftarrow$) Suppose such a $g$ exists. If $f(x) = f(x')$, applying
$g$ gives $x = x'$: $f$ is injective. For $y \in F$, $x = g(y)$
satisfies $f(x) = y$: $f$ is surjective.

Uniqueness: if $g$ and $h$ both work, then $g = g \circ \mathrm{id}_F
= g \circ (f \circ h) = (g \circ f) \circ h = h$. Finally the pair of
identities is symmetric in $f$ and $g$, so $g = f^{-1}$ is bijective
with inverse $f$.
\end{proof}

\begin{example}[Computing an inverse in practice]\label{ex:b1:logic:inversecomp}
Let $f \colon \R \to \intoo0{+\infty}$, $f(x) = \eu^{2x+1}$. To
invert, solve $y = f(x)$ for a given $y > 0$:
\[
y = \eu^{2x+1} \iff \ln y = 2x + 1 \iff x = \frac{\ln y - 1}2 ,
\]
each step being reversible on the announced domains. The
computation delivers everything at once: for each $y$ in the
codomain there is exactly one solution $x$, so $f$ is bijective,
and
\[
f^{-1} \colon \intoo0{+\infty} \to \R,
\qquad
f^{-1}(y) = \frac{\ln y - 1}2 .
\]
A quick check of both compositions ($f^{-1}(f(x)) = \frac{(2x+1) -
1}2 = x$ and $f(f^{-1}(y)) = \eu^{\ln y} = y$) confirms
\cref{thm:b1:logic:inverse}'s criterion. The insight: ``solve for
$x$ and watch the equivalences'' is simultaneously the existence
proof, the uniqueness proof, and the formula --- but it only works
if the codomain was announced correctly ($f$ is \emph{not}
surjective onto $\R$).
\end{example}

\begin{proposition}[Composition and the three properties]\label{prop:b1:logic:comp}
Let $f \colon E \to F$ and $g \colon F \to G$.
\begin{enumerate}
  \item If $f$ and $g$ are injective (resp.\ surjective, bijective), so
        is $g \circ f$; and then $(g \circ f)^{-1} = f^{-1} \circ
        g^{-1}$ in the bijective case.
  \item If $g \circ f$ is injective, then $f$ is injective. If
        $g \circ f$ is surjective, then $g$ is surjective.
\end{enumerate}
\end{proposition}

\begin{proof}
(1) If $g(f(x)) = g(f(x'))$, injectivity of $g$ gives $f(x) = f(x')$,
then injectivity of $f$ gives $x = x'$. If $z \in G$, surjectivity of
$g$ gives $y$ with $g(y) = z$, then surjectivity of $f$ gives $x$ with
$f(x) = y$, so $g(f(x)) = z$. In the bijective case one checks directly
that $f^{-1} \circ g^{-1}$ is a two-sided inverse of $g \circ f$, and
uniqueness in \cref{thm:b1:logic:inverse} concludes.

(2) If $f(x) = f(x')$ then $g(f(x)) = g(f(x'))$, and injectivity of
$g \circ f$ gives $x = x'$. If $z \in G$, surjectivity of $g \circ f$
gives $x$ with $g(f(x)) = z$: then $y = f(x)$ satisfies $g(y) = z$.
\end{proof}

\begin{example}[Point (2) is sharp]\label{ex:b1:logic:compsharp}
In \cref{prop:b1:logic:comp}~(2), one cannot upgrade the
conclusions: $g \circ f$ bijective does \emph{not} force $f$
surjective or $g$ injective. Take $E = G = \{1\}$, $F = \{1,
2\}$, with $f(1) = 1$ and $g(1) = g(2) = 1$: then $g \circ f =
\mathrm{id}_E$ is bijective, yet $f$ misses the element $2$ and
$g$ collapses both elements. The moral is a precise bookkeeping
rule: composition information flows to the \emph{inner} map for
injectivity and to the \emph{outer} map for surjectivity, never
the other way. (\cref{exo:b1:logic:9} builds the same phenomenon
with infinite sets, where it is the engine behind one-sided
inverses.)
\end{example}

\begin{example}\label{ex:b1:logic:injexamples}
$f \colon \R \to \R$, $x \mapsto x^2$ is neither injective ($f(-1) =
f(1)$) nor surjective ($-1$ has no preimage). Restricting the domain
and codomain, $f \colon \R_+ \to \R_+$, $x \mapsto x^2$ is bijective,
with inverse $y \mapsto \sqrt y$. A map's injectivity or surjectivity
depends on the announced domain and codomain, not only on the formula.
\end{example}

\section{Relations}

\begin{definition}[Equivalence relation]\label{def:b1:logic:equiv}
A \emph{binary relation} $\mathcal{R}$ on a set $E$ is an
\emph{equivalence relation}\index{equivalence relation} when it is:
\emph{reflexive} ($x \mathbin{\mathcal{R}} x$ for all $x$),
\emph{symmetric} ($x \mathbin{\mathcal{R}} y \implies y
\mathbin{\mathcal{R}} x$) and \emph{transitive} ($x
\mathbin{\mathcal{R}} y$ and $y \mathbin{\mathcal{R}} z$ imply $x
\mathbin{\mathcal{R}} z$). The \emph{equivalence class} of $x$ is
$\mathrm{cl}(x) = \{y \in E : x \mathbin{\mathcal{R}} y\}$.
\end{definition}

\begin{example}[Checking the three axioms]\label{ex:b1:logic:fracpart}
On $\R$, declare $x \mathbin{\mathcal{R}} y$ when $x - y \in \Z$.
\emph{Reflexive:} $x - x = 0 \in \Z$. \emph{Symmetric:} if $x - y
\in \Z$ then $y - x = -(x - y) \in \Z$. \emph{Transitive:} if $x -
y \in \Z$ and $y - z \in \Z$, then $x - z = (x - y) + (y - z) \in
\Z$ (a sum of integers). So $\mathcal R$ is an equivalence
relation, and $\mathrm{cl}(x) = x + \Z = \{x + k : k \in \Z\}$:
each class contains exactly one representative in $\intco01$, its
\emph{fractional part}. By contrast, the relation ``$\abs{x - y}
\leq 1$'' on $\R$ is reflexive and symmetric but \emph{not}
transitive ($0 \mathbin{\mathcal R} 1$ and $1 \mathbin{\mathcal R}
2$, yet $\abs{0 - 2} > 1$): closeness does not propagate, and no
partition into classes exists --- a useful counterexample to keep
in mind when verifying the axioms feels routine.
\end{example}

\begin{theorem}[Classes form a partition]\label{thm:b1:logic:partition}
Let $\mathcal{R}$ be an equivalence relation on $E$. Then the
equivalence classes are nonempty, pairwise disjoint or equal, and their
union is $E$: they form a \emph{partition}\index{partition} of $E$.
Conversely, every partition of $E$ arises this way from exactly one
equivalence relation (``being in the same piece'').
\end{theorem}

\begin{proof}
$x \in \mathrm{cl}(x)$ by reflexivity, so classes are nonempty with
union $E$. Suppose $\mathrm{cl}(x) \cap \mathrm{cl}(y) \neq \emptyset$,
say $z$ lies in both. Then $x \mathbin{\mathcal{R}} z$ and $y
\mathbin{\mathcal{R}} z$, so by symmetry and transitivity $x
\mathbin{\mathcal{R}} y$. Now for any $t \in \mathrm{cl}(y)$,
transitivity gives $t \in \mathrm{cl}(x)$, and symmetrically: the two
classes are equal. For the converse, let $(E_i)_{i \in I}$ be a
partition of $E$ and define $x \mathbin{\mathcal S} y$ to mean
``some piece contains both $x$ and $y$''. \emph{Reflexive:} $x$
lies in some piece, which then contains $x$ twice over.
\emph{Symmetric:} the defining condition is symmetric in $x$ and
$y$. \emph{Transitive:} if $x, y \in E_i$ and $y, z \in E_j$, then
$y \in E_i \cap E_j$, so $E_i = E_j$ (distinct pieces are
disjoint) and $x, z$ share a piece. The $\mathcal S$-class of $x$
is exactly the piece containing $x$, so the classes are the given
pieces. Finally the relation is determined by its classes: two
equivalence relations with the same classes relate the same pairs,
since each relates $x$ and $y$ exactly when $y$ belongs to the
class of $x$ --- whence the uniqueness claim.
\end{proof}

\begin{example}\label{ex:b1:logic:congruence}
On $\Z$, congruence modulo $n$ ($x \equiv y \pmod n$ when $n$ divides
$x - y$) is an equivalence relation; its classes are the $n$ sets of
integers with a given remainder upon division by $n$. This example
becomes the ring $\Z/n\Z$ in \cref{ch:b1:structures}.
\end{example}

\begin{definition}[Order relation]\label{def:b1:logic:order}
A relation $\preceq$ on $E$ is an \emph{order}\index{order relation}
when it is reflexive, \emph{antisymmetric} ($x \preceq y$ and $y
\preceq x$ imply $x = y$) and transitive. The order is \emph{total}
when any two elements are comparable, \emph{partial} otherwise. An
element $M \in A \subseteq E$ is a \emph{greatest element} of $A$ when
$a \preceq M$ for all $a \in A$; greatest (and least) elements are
unique when they exist.
\end{definition}

\begin{example}\label{ex:b1:logic:orders}
$(\R, \leq)$ is totally ordered. $(\mathcal{P}(E), \subseteq)$ is
partially ordered as soon as $E$ has two elements: $\{a\}$ and
$\{b\}$ are not comparable. The subset $A = \{\{a\}, \{b\}\}$ of
$\mathcal{P}(\{a,b\})$ has no greatest element, yet has an upper bound
$\{a, b\}$: the distinction between greatest elements and upper bounds
returns, for $\R$, in \cref{ch:b1:reals}.
\end{example}

\begin{example}[Two orders on the grid $\N^2$]\label{ex:b1:logic:lexorder}
On pairs of naturals, compare componentwise: $(a, b) \preceq (a',
b')$ when $a \leq a'$ \emph{and} $b \leq b'$ (the \emph{product
order}). This is an order --- each axiom is inherited coordinate
by coordinate --- but a partial one: $(1, 3)$ and $(2, 0)$ are
incomparable. Now compare like a dictionary: $(a, b)
\preceq_{\mathrm{lex}} (a', b')$ when $a < a'$, or $a = a'$ and $b
\leq b'$ (the \emph{lexicographic order}). Transitivity requires
a two-case check but holds, and any two pairs are now comparable:
the order is total. The two orders rank the same set differently
--- $(0, 100) \preceq_{\mathrm{lex}} (1, 0)$ though the product
order says nothing --- a reminder that an order is a structure
one \emph{chooses}, not a property of the set. Lexicographic
comparison is also the standard trick for turning several sorting
criteria into one.
\end{example}

\begin{remark}[Interlude: size as bijection]\label{rem:b1:logic:interlude}
A quiet theme of this chapter deserves a spotlight: bijections are
the mathematician's notion of ``same size''. For finite sets this
becomes the counting calculus of \cref{ch:b1:counting}, where
every formula is secretly a bijection; for infinite sets it
becomes the weekend problem below, where $\N$, $\Q$ and $\R$ turn
out to have genuinely different sizes. The same dictionary
reappears twice more in this volume in refined forms: sequences
(\cref{ch:b1:seq}) are nothing but maps $\N \to \R$, so statements
about sequences are statements about a set of maps; and linear
algebra will measure vector spaces not by bijections but by
\emph{linear} bijections, whose existence is governed by a single
number, the dimension (\cref{ch:b1:findim}). Whenever a new
``sameness'' appears --- equipotence, isomorphism of groups
(\cref{ch:b1:structures}), linear isomorphism --- the pattern of
\cref{thm:b1:logic:inverse} repeats: sameness is an invertible,
structure-respecting map.
\end{remark}

\begin{remark}[Where this chapter is used]\label{rem:b1:logic:whereused}
Everywhere --- but a few places deserve flagging. The three-quantifier
gymnastics of \cref{ex:b1:logic:limit} is the daily bread of
\cref{ch:b1:seq,ch:b1:continuity}: every limit proof is a game played
against an arbitrary $\varepsilon$. Equivalence classes reappear as the
congruence classes of $\Z/n\Z$ in \cref{ch:b1:structures}, where the
partition of \cref{thm:b1:logic:partition} acquires an algebraic
structure of its own. Order relations, upper bounds and least upper
bounds become the axiomatic heart of $\R$ in \cref{ch:b1:reals}.
Injections, surjections and bijections return as the linear maps of
\cref{ch:b1:linmaps}, where injectivity can be tested on a single
vector (the kernel); and the weekend problem below turns the bare
notion of bijection into a theory of the \emph{sizes of infinite
sets}, whose conclusions (countability of $\Q$, uncountability of
$\R$) resurface in \cref{ch:b1:reals,ch:b1:topology}.
\end{remark}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:b1:logic:1}
Write the negation of each statement, without using the word ``not'':
\begin{enumerate}
  \item $\forall x \in \R,\ \exists y \in \R,\ x + y > 0$;
  \item $\exists x \in \R,\ \forall y \in \R,\ xy = 0$;
  \item $\forall \varepsilon > 0,\ \exists \delta > 0,\ \forall x \in
        \R,\ \abs{x} \leq \delta \implies \abs{f(x)} \leq \varepsilon$
        (for a fixed map $f \colon \R \to \R$).
\end{enumerate}
Then decide whether statements (1) and (2) are true.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:logic:2}
Let $P, Q$ be statements. Using truth tables, prove that
$\lnot(P \implies Q) \iff P \land (\lnot Q)$, and deduce the negation
of: ``if a function is differentiable then it is continuous''.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:logic:3}
Prove by contraposition: for $x \in \R$, if $x^3 + x \geq 2$ then
$x \geq 1$. Then prove by contradiction: there is no smallest strictly
positive real number.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:logic:4}
Prove by induction that for all $n \in \N$:
\begin{enumerate}
  \item $\sum_{k=0}^{n} 2^k = 2^{n+1} - 1$;
  \item $4^n + 5$ is divisible by $3$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:logic:5}
Find the flaw in the following ``proof'' that all pencils have the same
color. \emph{Let $P(n)$: ``in every set of $n$ pencils, all pencils
have the same color''. $P(1)$ is clear. Assume $P(n)$ and take $n+1$
pencils; removing the last one, the first $n$ share their color;
removing the first one, the last $n$ share their color; hence all
$n+1$ share their color.}
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:logic:6}
Let $A, B, C$ be subsets of $E$. Prove:
\begin{enumerate}
  \item $A \setminus B = A \cap \overline{B}$;
  \item $(A \cup B) \setminus C = (A \setminus C) \cup (B \setminus
        C)$;
  \item $A \subseteq B \iff A \cup B = B \iff A \cap B = A$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:logic:7}
For each map, decide (with proof) whether it is injective, surjective,
bijective:
\begin{enumerate}
  \item $f \colon \N \to \N$, $n \mapsto n + 1$;
  \item $g \colon \Z \to \Z$, $n \mapsto n + 1$;
  \item $h \colon \R \setminus \{1\} \to \R$, $x \mapsto
        \frac{x+1}{x-1}$.
\end{enumerate}
For $h$, adjust the codomain to make it bijective and compute the
inverse.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:logic:8}
Let $f \colon E \to F$, let $A, A' \subseteq E$ and $B, B' \subseteq
F$.
\begin{enumerate}
  \item Prove $f^{-1}(B \cap B') = f^{-1}(B) \cap f^{-1}(B')$ and
        $f(A \cup A') = f(A) \cup f(A')$.
  \item Prove $f(A \cap A') \subseteq f(A) \cap f(A')$ and give an
        example where the inclusion is strict.
  \item Prove: $f$ is injective if and only if $f(A \cap A') = f(A)
        \cap f(A')$ for all $A, A'$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:logic:9}
Let $f \colon E \to F$ and $g \colon F \to E$ satisfy $g \circ f =
\mathrm{id}_E$. Prove that $f$ is injective and $g$ is surjective.
Give an example where neither $f$ nor $g$ is bijective.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:logic:10}
On $\R$, define $x \mathbin{\mathcal{R}} y \iff x^2 - y^2 = x - y$.
Prove that $\mathcal{R}$ is an equivalence relation and describe the
equivalence class of each real $x$. Which classes have exactly one
element?
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:logic:11}
(Cantor) Let $E$ be a set. Prove that there is no surjection from $E$
onto $\mathcal{P}(E)$. \emph{Hint: given $f \colon E \to
\mathcal{P}(E)$, consider $D = \{x \in E : x \notin f(x)\}$.}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:logic:12}
Let $f \colon E \to F$ be a map. Define $\Phi \colon \mathcal{P}(F)
\to \mathcal{P}(E)$ by $\Phi(B) = f^{-1}(B)$.
\begin{enumerate}
  \item Prove that $f$ is surjective if and only if $\Phi$ is
        injective.
  \item Prove that $f$ is injective if and only if $\Phi$ is
        surjective.
\end{enumerate}
\end{exercise}

\section{Problem: Comparing infinities}

\begin{problem}\label{pb:b1:logic:1}
When do two sets have ``the same number of elements''? Cantor's
answer --- when there is a bijection between them --- turns out to be
usable even for infinite sets, and it splits infinity into genuinely
different sizes. This problem builds the whole toolbox from the bare
definitions of this chapter: the Cantor--Schr\"oder--Bernstein theorem
(two injections manufacture a bijection), the countability of $\Q$,
the uncountability of $\R$ by the diagonal argument, and Cantor's
astonishing 1874 conclusion: \emph{transcendental numbers exist, and
massively so}, without exhibiting a single one.
\index{countable set}\index{Cantor--Schroder--Bernstein theorem}
Throughout, for sets $E$ and $F$, write $E \preceq F$ when there
exists an injection from $E$ into $F$, and $E \approx F$ (``$E$ and
$F$ are \emph{equipotent}'') when there exists a bijection from $E$
onto $F$.

\medskip
\noindent\textbf{Part I --- The vocabulary of comparison.}
\begin{enumerate}
  \item Show that $\approx$ behaves like an equivalence relation:
        $E \approx E$; if $E \approx F$ then $F \approx E$; if
        $E \approx F$ and $F \approx G$ then $E \approx G$. (Quote
        precisely \cref{thm:b1:logic:inverse} and
        \cref{prop:b1:logic:comp}.)
  \item Show that $\preceq$ is transitive, and that an injection
        $f \colon E \to F$ always induces $E \approx f(E)$.
  \item Let $E \neq \emptyset$. Show that $E \preceq F$ if and only
        if there exists a surjection from $F$ onto $E$.
  \item Verify that $n \mapsto n + 1$ is a bijection from $\N$ onto
        $\N^* = \N \setminus \{0\}$, and that
        \[
        \sigma(n) = \frac n2 \ \ (n \text{ even}), \qquad
        \sigma(n) = -\frac{n+1}2 \ \ (n \text{ odd})
        \]
        is a bijection from $\N$ onto $\Z$. Hence removing a point,
        or doubling into negatives, does not change the size of
        $\N$.
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The Cantor--Schr\"oder--Bernstein
theorem.} Let $f \colon E \to F$ and $g \colon F \to E$ be two
injections. Define
\[
C_0 = E \setminus g(F), \qquad C_{n+1} = g\bigl(f(C_n)\bigr)
\ \ (n \in \N), \qquad C = \bigcup_{n \in \N} C_n,
\]
and let $h \colon E \to F$ send $x \in C$ to $f(x)$, and $x \notin C$
to the unique $y \in F$ with $g(y) = x$.
\begin{enumerate}[resume]
  \item Check that $h$ is well defined: if $x \notin C$ then $x \in
        g(F)$, and the element $y$ with $g(y) = x$ is unique.
  \item Show that $g\bigl(f(C)\bigr) = \bigcup_{n \geq 1} C_n
        \subseteq C$. (Direct images commute with unions:
        \cref{exo:b1:logic:8}.)
  \item Show that $h$ is injective. (Three cases; in the mixed case
        $x \in C$, $x' \notin C$, show that $h(x) = h(x')$ would
        force $x' \in g(f(C)) \subseteq C$.)
  \item Show that $h$ is surjective: given $y \in F$, distinguish
        the cases $g(y) \notin C$ and $g(y) \in C_n$ for some $n
        \geq 1$ (why is $g(y) \in C_0$ impossible?), and exhibit a
        preimage of $y$ in each case.
  \item Conclude with the \emph{Cantor--Schr\"oder--Bernstein
        theorem}: if $E \preceq F$ and $F \preceq E$, then $E
        \approx F$. Comment in one sentence on what makes this
        statement nontrivial.
  \item Two applications. (a) Show $\intcc01 \approx \intoo01$.
        (b) Show that $\varphi(p, q) = 2^p(2q + 1) - 1$ defines a
        bijection from $\N \times \N$ onto $\N$ --- injectivity by a
        parity argument, surjectivity by strong induction
        (\cref{thm:b1:logic:induction}). Hence $\N \times \N \approx
        \N$: the plane of integer points is no bigger than the
        line.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Countable sets.} Call a set $E$
\emph{at most countable} when $E \preceq \N$, and \emph{countable}
when $E \approx \N$.
\begin{enumerate}[resume]
  \item Show that every infinite subset $A \subseteq \N$ is
        countable. (Define $\varphi(n)$ recursively as the least
        element of $A \setminus \{\varphi(0), \dots,
        \varphi(n-1)\}$; show $\varphi$ is strictly increasing,
        satisfies $\varphi(n) \geq n$, and attains every element of
        $A$.)
  \item Deduce that a set is at most countable if and only if it is
        finite or countable, and observe that question 9 gives the
        shortcut: if $E \preceq \N$ and $\N \preceq E$, then $E$ is
        countable.
  \item Show that if $E$ and $F$ are at most countable, so is $E
        \times F$. Deduce that $\Z \times \N^*$ is countable.
  \item Show that $\Q$ is countable. (Inject $\Q$ into $\Z \times
        \N^*$ by writing each rational in lowest terms with positive
        denominator --- the uniqueness of that representation is
        proved in \cref{ch:b1:arith}; then apply question 12.)
  \item Show that a countable union of at most countable sets is at
        most countable: if each $E_n$ ($n \in \N$) is at most
        countable, so is $\bigcup_{n \in \N} E_n$. (Send $x$ to the
        pair $(n, f_n(x))$ where $n$ is the \emph{least} index with
        $x \in E_n$.)
  \item Show that the set of \emph{finite} subsets of $\N$ is
        countable. (Map a finite subset $F$ to $\sum_{i \in F} 2^i$;
        prove injectivity by comparing the largest element where two
        finite sets differ, using $\sum_{k=0}^{m-1} 2^k = 2^m - 1$
        from \cref{exo:b1:logic:4}.)
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Diagonalization.} Let $\{0,1\}^{\N}$
denote the set of all maps $u \colon \N \to \{0, 1\}$, i.e.\ the set
of binary sequences.
\begin{enumerate}[resume]
  \item Construct a bijection between $\mathcal{P}(\N)$ and
        $\{0,1\}^{\N}$ (indicator functions).
  \item (The diagonal argument) Let $\Phi \colon \N \to
        \{0,1\}^{\N}$ be any map. Consider the sequence $d$ defined
        by $d(n) = 1 - \Phi(n)(n)$. Show that $d$ is not in the
        image of $\Phi$, and conclude that $\{0,1\}^{\N}$ is
        \emph{not} at most countable. Explain in one sentence why,
        through question 17, this is exactly Cantor's theorem
        (\cref{exo:b1:logic:11}) for $E = \N$.
  \item Admit --- as familiar from school, and established
        rigorously in \cref{ch:b1:reals} --- that every $x \in
        \intco01$ has a unique \emph{proper} decimal expansion $x =
        0.d_1 d_2 d_3\dots$ (one not ending in an infinite string of
        $9$s). Given any sequence $(x_n)_{n \geq 1}$ of elements of
        $\intco01$, construct $x \in \intco01$ with $x \neq x_n$ for
        all $n$: choose its $n$-th digit to be $5$ if the $n$-th
        digit of $x_n$ differs from $5$, and $6$ otherwise. Justify
        carefully that $x$ is proper and avoids every $x_n$, and
        conclude that $\intco01$ is not at most countable.
  \item Deduce that $\R$ is uncountable, and that the set $\R
        \setminus \Q$ of irrational numbers is uncountable as well.
        In what precise sense are ``most'' real numbers irrational?
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Cantor's 1874 theorem: transcendental
numbers exist.} A real number $x$ is \emph{algebraic}
\index{algebraic number} when $P(x) = 0$ for some nonzero polynomial
$P$ with integer coefficients, and \emph{transcendental}
\index{transcendental number} otherwise. Admit for this part --- it
is proved in \cref{ch:b1:poly} --- that a nonzero polynomial of
degree $n$ has at most $n$ real roots.
\begin{enumerate}[resume]
  \item Show that every rational number is algebraic, and find
        explicit polynomials with integer coefficients annihilating
        $\sqrt 2$ and $\sqrt 2 + \sqrt 3$.
  \item For fixed $n \in \N$, show that the set of polynomials of
        degree at most $n$ with integer coefficients is countable.
        (Inject it into $\Z^{n+1}$ and induct on $n$ with
        question 13.)
  \item Deduce that the set of \emph{all} polynomials with integer
        coefficients is countable.
  \item Prove \emph{Cantor's theorem on algebraic numbers}: the set
        $\mathcal{A}$ of algebraic real numbers is countable.
  \item Conclude: transcendental real numbers exist, and the set of
        transcendental numbers is uncountable. Then take stock of
        the whole problem in a few sentences: the chain $\N \approx
        \Z \approx \Q \approx \mathcal{A}$, the strict jump to $\R
        \approx$ (essentially) $\mathcal{P}(\N)$, where each tool
        (Cantor--Schr\"oder--Bernstein, countable unions, the
        diagonal) was decisive --- and the philosophical punch of
        proving that uncountably many transcendental numbers exist
        without naming even one. (Proving a \emph{specific} number
        such as $\pi$ transcendental is far harder and beyond this
        volume.)
\end{enumerate}
\end{problem}
