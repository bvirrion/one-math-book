\chapter{Courbes planes}\label{ch:b1:curves}

Le calcul différentiel et intégral des \cref{ch:b1:derivative,ch:b1:taylor}
a été construit pour les fonctions $y = f(x)$~; la plupart des
courbes de la géométrie et de la mécanique --- trajectoires,
cercles roulant sur des cercles, orbites --- refusent cette forme
et se présentent plutôt comme des courbes \emph{paramétrées} $t
\mapsto (x(t), y(t))$ ou comme des courbes \emph{en polaires} $r =
r(\theta)$. Ce chapitre classique étudie les deux, et se termine
par les \omterm{def:b1:curves:conics}{coniques}.

\section{Courbes paramétrées}

\begin{definition}\label{def:b1:curves:def}
Une \emph{courbe paramétrée}\index{courbe paramétrée} est une
\omterm{def:b1:logic:map}{application} $f \colon I \to \R^2$, $t \mapsto M(t) = (x(t), y(t))$,
où $x$ et $y$ sont de classe $C^1$ (au moins) sur l'\omterm{prop:b1:reals:intervals}{intervalle} $I$.
Le \emph{vecteur vitesse} est $f'(t) = (x'(t), y'(t))$~; le point
$M(t_0)$ est \emph{régulier} lorsque $f'(t_0) \neq (0,0)$, et la
\emph{tangente} en ce point est la droite passant par $M(t_0)$ et
dirigée par $f'(t_0)$.
\end{definition}

\begin{proof}[Pourquoi $f'$ dirige la tangente]
Par la formule de Taylor--Young appliquée composante par composante,
$M(t) = M(t_0) + (t - t_0) f'(t_0) + o(t - t_0)$~: la direction de
la corde $\frac{M(t) - M(t_0)}{t - t_0}$ tend vers $f'(t_0)$.
C'est l'hypothèse de régularité qui fait de cette limite une
\emph{direction}~: si $f'(t_0) = 0$, la formule se réduit
à $M(t) = M(t_0) + o(t - t_0)$ et ne dit rien sur la façon dont la
courbe quitte le point --- d'où le traitement séparé des points
singuliers dans la méthode ci-dessous, où la première \omterm{def:b1:derivative:def}{dérivée}
\emph{non nulle} reprend le rôle de $f'$.
Remarquons aussi que la tangente est un objet géométrique~: tout
reparamétrage multiplie $f'$ par un facteur scalaire non nul et
laisse la droite inchangée.
\end{proof}

\begin{method}[Étude d'une courbe paramétrée]\label{met:b1:curves:study}
\begin{enumerate}
  \item \emph{Réduire l'\omterm{prop:b1:reals:intervals}{intervalle} d'étude} grâce aux symétries~:
        relations entre $M(-t)$, $M(t + T)$, \dots\ et $M(t)$
        (réflexions, translations, périodicité), puis ne tracer que
        la partie réduite.
  \item \emph{Variations}~: dresser conjointement le tableau des
        signes de $x'$ et de $y'$~; la courbe va vers la droite ou
        vers la gauche selon $x'$, vers le haut ou vers le bas selon $y'$.
  \item \emph{Points remarquables}~: tangente horizontale ($y' = 0
        \neq x'$), tangente verticale ($x' = 0 \neq y'$)~; en un
        point \emph{singulier} ($f' = 0$), examiner les \omterm{def:b1:derivative:def}{dérivées}
        d'ordre supérieur pour obtenir la direction de la tangente.
  \item \emph{Comportement asymptotique} aux bornes de $I$, puis tracé.
\end{enumerate}
\end{method}

\begin{example}[Branches asymptotiques, un exemple traité]
\label{ex:b1:curves:asymptotes}
$x(t) = t$, $y(t) = t + \dfrac1t$ sur $\intoo{0}{+\infty}$. Quand $t
\to 0^{+}$~: $x \to 0$ tandis que $y \to +\infty$ --- la courbe
grimpe le long de l'\emph{asymptote verticale} $x = 0$ (limite finie
pour une coordonnée, infinie pour l'autre). Quand $t \to +\infty$~:
les deux \omterm{prop:b1:vspaces:coordinates}{coordonnées} explosent~; on teste alors une droite~: la
différence
\[
y(t) - x(t) = \frac1t \longrightarrow 0^{+}
\]
met en évidence l'\emph{asymptote oblique} $y = x$, approchée par
valeurs supérieures. Entre les deux, $y' = 1 - \frac1{t^2}$ s'annule en $t = 1$~:
le point $(1, 2)$ est le point bas de la branche, et $x' = 1 >
0$ partout, donc la courbe avance toujours vers la droite. Un arc,
deux asymptotes, un minimum~: une image complète à partir de trois
calculs --- avec la recette générale visible en dessous~:
lorsque $x, y \to \infty$ \omterm{def:b1:logic:sets}{ensemble}, examiner $y/x$ pour une pente
candidate (ici $\to 1$), puis $y - (\text{pente})\,x$ pour
l'ordonnée à l'origine et le côté d'approche.
\end{example}

\begin{example}[L'astroïde]\label{ex:b1:curves:astroid}
$x(t) = \cos^3 t$, $y(t) = \sin^3 t$.\index{astroïde} Symétries~:
$M(t + 2\pi) = M(t)$ (on étudie une période)~; $M(-t)$ est le
symétrique de $M(t)$ par rapport à l'axe des $x$~; $M(\pi - t)$ par
rapport à l'axe des $y$~; $M(\frac\pi2 - t)$ par rapport à la
diagonale $y = x$~: il suffit d'étudier $t \in
\intcc{0}{\frac\pi4}$ et de déplier.

Vitesse~: $f'(t) = 3\sin t\cos t\,(-\cos t, \sin t)$. Sur
$\intoo{0}{\frac\pi2}$ tous les points sont réguliers, de tangente
dirigée par $(-\cos t, \sin t)$~; en $t = 0$ (le point $(1,0)$) la
vitesse s'annule~: c'est un \emph{point de
rebroussement}\index{point de rebroussement}, où la courbe rebrousse
chemin le long de la direction tangente $(-1, 0)\cdot(\pm)$ --- par
symétrie, les quatre \omterm{ex:b1:curves:astroid}{points de rebroussement} sont en $(\pm1, 0), (0, \pm1)$.
\end{example}

\begin{example}[Une courbe en huit, étudiée complètement]
\label{ex:b1:curves:lissajous}
$x(t) = \sin t$, $y(t) = \sin 2t$ (une \emph{courbe de Lissajous}).
\emph{Symétries}~: $M(t + \pi) = (-x(t), y(t))$ (réflexion par
rapport à l'axe des $y$), $M(-t) = (-x(t), -y(t))$ (symétrie
centrale), $M(\pi - t) = (x(t), -y(t))$ (réflexion par rapport à
l'axe des $x$)~: il suffit d'étudier $t \in \intcc{0}{\frac\pi2}$
et de déplier.
\emph{Variations}~: $x' = \cos t \geq 0$ partout, tandis que $y' =
2\cos 2t$ est positif avant $t = \frac\pi4$ et négatif après~:
l'arc monte vers la droite jusqu'au sommet
$\bigl(\frac{\sqrt2}2,\ 1\bigr)$ en $t = \frac\pi4$ (tangente
horizontale), puis descend vers la droite jusqu'à $(1, 0)$ en $t =
\frac\pi2$, où $x' = 0 \neq y'$~: tangente verticale.
\emph{Point double}~: $M(0) = M(\pi) = (0,0)$, avec deux vitesses
différentes
\[
f'(0) = (1,\ 2), \qquad f'(\pi) = (-1,\ 2) :
\]
deux branches \emph{régulières} se croisant à l'origine selon des
angles distincts --- un point double, non un point singulier~: chaque
passage est parfaitement lisse, les deux passages partagent
simplement leur position. La courbe entière est le huit ci-dessous.
\end{example}

\section{Courbes en polaires}

\begin{definition}\label{def:b1:curves:polar}
Une \emph{courbe en polaires}\index{courbe en polaires} est donnée
par $r = r(\theta)$~: le point de paramètre $\theta$ est
\[
M(\theta) = r(\theta)\,\vec u(\theta),
\qquad \vec u(\theta) = (\cos\theta, \sin\theta).
\]
Avec $\vec v(\theta) = (-\sin\theta, \cos\theta) = \vec
u\,'(\theta)$, la vitesse est
\[
M'(\theta) = r'(\theta)\, \vec u(\theta) + r(\theta)\, \vec
v(\theta) .
\]
Conséquences~: là où $r \neq 0$, le point est régulier, et la
tangente fait avec le rayon vecteur l'angle $V$ donné par $\tan V =
\frac{r}{r'}$ (angle entre $M'$ et $\vec u$)~; là où $r(\theta_0)
= 0$, la courbe passe par l'origine \emph{avec pour tangente le
rayon $\theta = \theta_0$} (direction $\vec u(\theta_0)$, lue sur
$M'(\theta_0) = r'(\theta_0)\vec u(\theta_0)$, ou sur la corde
limite~: la corde de $O$ à $M(\theta)$ est portée par $\vec
u(\theta)$ lui-même, qui tend vers $\vec u(\theta_0)$ --- la règle
vaut donc même lorsque $r'(\theta_0) = 0$ et que le point est
singulier, ce qui explique pourquoi les \omterm{ex:b1:curves:astroid}{points de rebroussement}
polaires à l'origine, comme celui de la cardioïde, obtiennent leur
tangente gratuitement).
\end{definition}

\begin{example}[La cardioïde]\label{ex:b1:curves:cardioid}
$r(\theta) = 1 + \cos\theta$.\index{cardioïde} Symétrie~: $r(-\theta)
= r(\theta)$~: réflexion par rapport à l'axe des $x$~; on étudie
$\theta \in \intcc{0}{\pi}$. $r$ décroît de $2$ à $0$~; en $\theta =
\pi$, $r = 0$~: la courbe atteint l'origine tangentiellement au rayon
$\theta = \pi$ (l'axe des $x$), y formant un \omterm{ex:b1:curves:astroid}{point de rebroussement}
--- la pointe du cœur. Tangente en $\theta = 0$~: $r' = 0$, donc
$\tan V = \infty$~: perpendiculaire à l'axe.

L'angle $V$ mérite une lecture de plus. En $\theta =
\frac\pi2$~: $r = 1$ et $r' = -1$, donc $\tan V = \frac{r}{r'} =
-1$~: la tangente fait les trois quarts d'un angle droit avec le
rayon sortant --- la courbe s'infléchit déjà vers son \omterm{ex:b1:curves:astroid}{point de
rebroussement}. La formule $\tan V = r/r'$ fournit les directions
tangentes tout au long de la courbe sans aucun calcul de
$M'(\theta)$~: c'est l'analogue polaire de la lecture d'une pente.
\end{example}

\begin{example}[Des polaires aux cartésiennes~: un cercle caché]
\label{ex:b1:curves:polarcircle}
Quelle est la \omterm{def:b1:curves:polar}{courbe en polaires} $r = 2\cos\theta$~? Multiplions par $r$~:
$r^2 = 2r\cos\theta$, c'est-à-dire $x^2 + y^2 = 2x$, c'est-à-dire
\[
(x - 1)^2 + y^2 = 1 :
\]
le cercle de centre $(1, 0)$ et de rayon $1$, passant par
l'origine. La comptabilité importe~: lorsque $\theta$ parcourt
$\intoc{-\frac\pi2}{\frac\pi2}$, le cercle entier est décrit
exactement une fois ($r$ s'annule aux deux bornes), et en $\theta =
\pm\frac\pi2$ la règle «~tangente à l'origine le long du rayon
$\theta = \theta_0$~» y donne une tangente \emph{verticale} ---
en accord avec la géométrie, puisque l'axe vertical est bien tangent
à ce cercle en $O$. Pour $\theta$ au-delà de cet \omterm{prop:b1:reals:intervals}{intervalle}, $r < 0$
redécrit le même cercle~: rappel qu'une \omterm{def:b1:curves:polar}{courbe en polaires} est un
objet \emph{paramétré}, autorisé à repasser sur lui-même.
\end{example}

\section{Longueur d'arc}

\begin{definition}[Longueur d'un arc]\label{def:b1:curves:arclength}
La \emph{longueur}\index{longueur d'arc} d'un arc $C^1$ $f \colon
\intcc{a}{b} \to \R^2$ est l'\omterm{thm:b1:integration:def}{intégrale} de la vitesse scalaire~:
\[
L = \int_a^b \norm{f'(t)}\,\dd t
= \int_a^b \sqrt{x'(t)^2 + y'(t)^2}\;\dd t .
\]
(Motivation~: sur un petit \omterm{prop:b1:reals:intervals}{intervalle}, $M(t + h) \approx M(t) + h
f'(t)$, de sorte que l'arc est proche d'une ligne polygonale dont les
longueurs des segments s'additionnent en une somme de Riemann de
$\norm{f'}$, \cref{thm:b1:integration:riemann}.) Pour une \omterm{def:b1:curves:polar}{courbe en
polaires} $r = r(\theta)$, la vitesse $r'\vec u + r\vec v$ a des
composantes \omterm{def:b1:euclid:orthogonal}{orthogonales}, donc
\[
L = \int_{\theta_1}^{\theta_2}
\sqrt{r'(\theta)^2 + r(\theta)^2}\;\dd\theta .
\]
La longueur ne dépend pas du paramétrage (monotone, $C^1$)
choisi~: la substitution $t = \varphi(s)$ dans l'\omterm{thm:b1:integration:def}{intégrale}
(\cref{thm:b1:integration:parts}) multiplie $f'$ par
$\varphi'$ et $\dd t$ par $\varphi'^{-1}$.
\end{definition}

\begin{example}[Vérification~: le cercle]\label{ex:b1:curves:circlelength}
Pour $f(t) = (R\cos t, R\sin t)$ sur $\intcc{0}{2\pi}$~: $\norm{f'} =
R$, donc $L = 2\pi R$ --- la définition rend bien la circonférence.
Test de reparamétrage~: l'\omterm{def:b1:logic:map}{application} $g(t) = (R\cos 2t, R\sin 2t)$ sur
$\intcc{0}{\pi}$ décrit le \emph{même} cercle à vitesse doublée
$\norm{g'} = 2R$, et
\[
\int_0^{\pi} 2R\,\dd t = 2\pi R
\]
de nouveau~: deux fois moins de temps, deux fois plus vite, même
\omterm{def:b1:curves:arclength}{longueur} --- l'invariance promise dans la définition, constatée une
fois sur les nombres.
(Faire parcourir $g$ sur tout $\intcc{0}{2\pi}$ donnerait $4\pi R$~: une
courbe parcourue deux fois est deux fois plus longue \emph{en tant que
trajet}~; la \omterm{def:b1:curves:arclength}{longueur} mesure le chemin paramétré, et une comptabilité
honnête de l'\omterm{prop:b1:reals:intervals}{intervalle} fait partie du calcul.)
Pour l'astroïde $(\cos^3t, \sin^3t)$~: $\norm{f'} = 3\abs{\sin
t\cos t} = \tfrac32\abs{\sin 2t}$, et par symétrie $L = 4\int_0^{\pi/2}
\tfrac32\sin 2t\,\dd t = 6$~: une courbe tracée à l'intérieur du cercle
unité, de \omterm{def:b1:curves:arclength}{longueur} $6 < 2\pi$. Le devoir maison mesure la plus célèbre
des arches.
\end{example}

\section{Coniques}

\begin{definition}[Définition par foyer et directrice]\label{def:b1:curves:conics}
Fixons un point $F$ (le \emph{foyer}), une droite $D$ ne passant pas
par $F$ (la \emph{directrice}) et $e > 0$ (l'\emph{excentricité}). La
\emph{conique}\index{conique} associée à ces données est
\[
\mathcal{C} = \{M : d(M, F) = e\; d(M, D)\}:
\]
c'est une \emph{ellipse} si $e < 1$, une \emph{parabole} si $e = 1$,
une \emph{hyperbole} si $e > 1$. (Le cercle apparaît comme cas limite
dégénéré $e \to 0$.)
\end{definition}

\begin{example}[La définition, vérifiée sur une parabole]
\label{ex:b1:curves:parabolacheck}
Prenons la parabole $y^2 = 4x$, c'est-à-dire $2p = 4$~: foyer $F = (1, 0)$
et directrice $D : x = -1$ (la forme réduite du
\cref{thm:b1:curves:reduced} les place en $\pm\frac p2$). Au
point $M = (1, 2)$ de la courbe~:
\[
MF = \sqrt{(1-1)^2 + 2^2} = 2,
\qquad
d(M, D) = 1 - (-1) = 2 :
\]
égaux, comme $e = 1$ l'exige. En $M' = (4, 4)$~: $MF' = \sqrt{9 +
16} = 5$ et $d(M', D) = 5$ également. La définition par foyer et
directrice n'est pas une abstraction~: c'est un couple de distances
que l'on peut mesurer en tout point, et l'algèbre des équations
réduites n'est rien d'autre que cette mesure faite une fois pour
toutes.
\end{example}

\begin{theorem}[Équations réduites]\label{thm:b1:curves:reduced}
Dans un repère orthonormal bien choisi~:
\[
\text{ellipse~: } \frac{x^2}{a^2} + \frac{y^2}{b^2} = 1
\quad (a \geq b > 0),
\qquad
\text{hyperbole~: } \frac{x^2}{a^2} - \frac{y^2}{b^2} = 1,
\qquad
\text{parabole~: } y^2 = 2px,
\]
avec, pour l'ellipse~: foyers en $(\pm c, 0)$, $c = \sqrt{a^2 - b^2}$,
$e = \frac ca$, et la caractérisation bifocale $MF + MF' = 2a$~;
pour l'hyperbole~: $c = \sqrt{a^2 + b^2}$, $e = \frac ca$,
$\abs{MF - MF'} = 2a$, asymptotes $y = \pm\frac ba x$.
\end{theorem}

\begin{proof}
Plaçons le foyer à l'origine et la directrice verticale, $x = -h$
($h > 0$)~: la condition $MF^2 = e^2 d(M, D)^2$ s'écrit $x^2 + y^2 =
e^2 (x + h)^2$. Pour $e = 1$~: $y^2 = 2hx + h^2$, une parabole après la
translation $x \mapsto x - \frac h2$ (donc $p = h$). Pour $e \neq 1$~:
en complétant le carré en $x$,
\[
(1 - e^2)\Bigl(x - \frac{e^2 h}{1 - e^2}\Bigr)^{\!2} + y^2
= e^2h^2 + \frac{e^4 h^2}{1 - e^2}
= \frac{e^2 h^2}{1 - e^2} .
\]
Posons $X = x - \frac{e^2h}{1-e^2}$ (une translation du repère). Lorsque $e < 1$,
divisons par le membre de droite, qui est positif~:
$\frac{X^2}{a^2} + \frac{y^2}{b^2} = 1$ avec $a = \frac{eh}{1 -
e^2}$, $b = \frac{eh}{\sqrt{1-e^2}}$~; lorsque $e > 1$, les deux membres
de l'égalité changent de signe comme il convient et la même division donne
$\frac{X^2}{a^2} - \frac{y^2}{b^2} = 1$ avec $a = \frac{eh}{e^2 -
1}$, $b = \frac{eh}{\sqrt{e^2 - 1}}$. Les valeurs annoncées de $c$
s'en déduisent ($c^2 = a^2 - b^2$ ou $a^2 + b^2$ donne $c = ea$ dans les
deux cas, ce qui place correctement le foyer), et les propriétés bifocales sont
des vérifications directes sur les équations réduites. En détail pour
l'ellipse, avec $F' = (c, 0)$ et $M = (X, y)$ sur la courbe~:
\[
MF'^2 = (X - c)^2 + y^2
= X^2 - 2cX + c^2 + b^2\Bigl(1 - \frac{X^2}{a^2}\Bigr)
= \frac{c^2}{a^2}X^2 - 2cX + a^2
= (a - eX)^2 ,
\]
en utilisant $c^2 + b^2 = a^2$ et $e = \frac ca$~; comme $\abs X \leq
a$ et $e < 1$, on a $a - eX > 0$, donc $MF' = a - eX$ sans jamais extraire
de racine carrée. Le calcul symétrique donne $MF = a + eX$,
d'où $MF + MF' = 2a$, constante. Pour l'hyperbole, la même
algèbre donne $MF = \abs{a + eX}$ et $MF' = \abs{eX - a}$,
de différence $\pm2a$ selon la branche.
\end{proof}

\begin{example}\label{ex:b1:curves:conicexample}
$\frac{x^2}{25} + \frac{y^2}{9} = 1$~: ellipse, $a = 5$, $b = 3$, $c
= 4$~: foyers $(\pm4, 0)$, excentricité $\frac45$. Son paramétrage~:
$(5\cos t, 3\sin t)$ --- un cercle étiré de façon anisotrope~; la
somme des distances aux foyers de n'importe lequel de ses points vaut $10$.
\end{example}

\begin{example}[Lire une orbite sur son équation polaire]
\label{ex:b1:curves:orbit}
La \omterm{def:b1:curves:conics}{conique} en polaires $r = \dfrac{1}{1 + \frac12\cos\theta}$
(\cref{exo:b1:curves:7} avec $p = 1$, $e = \frac12$) est une
ellipse dont un \emph{foyer est à l'origine} --- la géométrie d'une
orbite planétaire avec le soleil en $O$. Tout s'extrait de
$p$ et $e$~: d'après la réduction du
\cref{thm:b1:curves:reduced}, $a = \dfrac{p}{1 - e^2} =
\dfrac{1}{3/4} = \dfrac43$ et $c = ea = \dfrac23$. Les deux
apsides le confirment sans la moindre théorie~:
\[
r(0) = \frac{1}{3/2} = \frac23 = a - c
\quad (\text{périhélie}),
\qquad
r(\pi) = \frac{1}{1/2} = 2 = a + c
\quad (\text{aphélie}),
\]
et leur somme $\frac23 + 2 = \frac83 = 2a$ redonne le grand
axe. La forme polaire est la forme naturelle dès qu'un foyer est
physiquement distingué~; la forme cartésienne réduite l'est dès que
les axes de symétrie le sont. Passer de l'une à l'autre est
exactement ce que fait le calcul par complétion du carré du théorème.

L'excentricité comme un bouton de réglage, enfin~: gardons $p = 1$ et
faisons tourner $e$ dans $r = \frac{1}{1 + e\cos\theta}$. En $e = 0$~:
le cercle $r = 1$. En $e = \frac12$~: l'ellipse qu'on vient d'étudier,
$r$ oscillant entre $\frac23$ et $2$. En $e = 1$~: $r(\theta) \to \infty$
quand $\theta \to \pi$ --- la courbe ne se referme plus~: une parabole, son
point le plus lointain repoussé à l'infini. En $e = 2$~: le dénominateur
s'annule en $\cos\theta = -\frac12$, et seul $\theta \in
\intoo{-\frac{2\pi}3}{\frac{2\pi}3}$ subsiste~: une branche d'une
hyperbole, s'échappant selon deux directions asymptotiques. Une seule
formule, toute la famille des \omterm{def:b1:curves:conics}{coniques}, et le point de transition $e =
1$ visible comme l'instant où le dénominateur atteint zéro pour la première fois.
\end{example}

\begin{remark}[Pièges classiques]
\label{rem:b1:curves:pitfalls}
\emph{Les points doubles ne sont pas des points singuliers}~: en un
point de croisement, chaque branche est régulière~; «~singulier~» renvoie à
$f'(t_0) = 0$ pour une valeur du paramètre
(\cref{ex:b1:curves:lissajous} face aux \omterm{ex:b1:curves:astroid}{points de rebroussement} de l'astroïde).
\emph{Tangente verticale contre \omterm{ex:b1:curves:astroid}{point de rebroussement}}~: $x'(t_0) = 0 \neq y'(t_0)$
est un point régulier à tangente verticale~; seul $x' = y' = 0$
exige le développement d'ordre supérieur.
\emph{Les rayons négatifs se placent sur le rayon opposé}~: pour $r(\theta) <
0$ le point est $-\abs{r}\,\vec u(\theta)$, d'angle $\theta +
\pi$ (\cref{exo:b1:curves:5})~; l'oublier fait perdre les boucles intérieures
ou dédouble les courbes.
\emph{La \omterm{def:b1:curves:arclength}{longueur d'arc} intègre $\norm{f'}$, pas $f'$}~: on découpe
l'\omterm{thm:b1:integration:def}{intégrale} aux zéros de la vitesse --- pour l'astroïde,
intégrer $\frac32\sin 2t$ sur une période entière sans valeur
absolue donne $0$, et non $6$.
\emph{Les \omterm{def:b1:curves:conics}{coniques} doivent être réduites avant d'être lues}~: dans $9x^2 +
25y^2 - 36x - 50y - 164 = 0$, ni les axes, ni le centre,
ni l'excentricité ne sont visibles avant d'avoir complété les carrés
(\cref{exo:b1:curves:6})~; et une partie quadratique nulle en une
variable signifie parabole, non «~ellipse dégénérée~».
\end{remark}

\begin{remark}[Où mènent ces courbes]
\label{rem:b1:curves:whereused}
Les \omterm{def:b1:curves:def}{courbes paramétrées} sont le langage de la mécanique~: les trajectoires
sont des courbes, les vecteurs vitesse sont des vecteurs tangents, et la
\omterm{def:b1:curves:arclength}{longueur d'arc} de la \cref{def:b1:curves:arclength} est la distance parcourue. Les
\omterm{def:b1:curves:conics}{coniques} réapparaissent partout où agit une loi en carré inverse --- les
orbites planétaires sont des ellipses avec le soleil en un foyer. Au
\cref{ch:b1:multivar}, les courbes deviennent les lignes de niveau de fonctions
de deux variables, et la tangente de ce chapitre rencontre le
gradient du suivant. Le volume de Licence~2 ajoute la courbure et la
forme canonique locale d'une courbe~; le devoir maison ci-dessous
extrait déjà, avec les seuls outils de cette année, tout ce que le
dix-septième siècle savait de sa courbe la plus célèbre.
\end{remark}

\begin{remark}[Perspectives dans le livre 3]
\label{rem:b1:curves:perspectives}
Ce chapitre consomme toute la première moitié du volume et
nourrit le dernier chapitre. \emph{Consommé}~: les vecteurs tangents sont
des \omterm{def:b1:derivative:def}{dérivées} (\cref{ch:b1:derivative}), la \omterm{def:b1:curves:arclength}{longueur d'arc} et les aires sont
des \omterm{thm:b1:integration:def}{intégrales} (\cref{ch:b1:integration}), les \omterm{ex:b1:curves:astroid}{points de rebroussement} se règlent par
des développements de Taylor (\cref{ch:b1:taylor}), la tautochrone est une
équation différentielle \omterm{def:b1:linmaps:def}{linéaire} (\cref{ch:b1:diffeq}), et toute
distance et tout angle sont euclidiens (\cref{ch:b1:euclid}).
\emph{Nourri}~: au \cref{ch:b1:multivar}, une courbe $t \mapsto (x(t),
y(t))$ tracée dans une ligne de niveau $\{f = c\}$ se dérive par
la \omterm{thm:b1:derivative:operations}{règle de la chaîne}, et l'identité obtenue $\langle \nabla f,\
f'(t)\rangle = 0$ marie les vecteurs tangents de ce chapitre aux
gradients du suivant --- la tangente de la courbe et la
direction normale de la surface sont le même calcul vu
des deux rives.
\end{remark}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:b1:curves:1}
Étudier et tracer la courbe $x(t) = t^2$, $y(t) = t^3$ (symétries,
variations, comportement au point singulier $t = 0$).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:curves:2}
Pour la cycloïde $x(t) = t - \sin t$, $y(t) = 1 - \cos t$ (la
trajectoire d'un point d'une roue de rayon $1$ qui roule)~: identifier
la symétrie de translation par période, les points singuliers, et la
direction de la tangente en $t = 0$ \emph{(développer $x$ et $y$ aux premiers
ordres non nuls)}.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:curves:3}
Donner la tangente à la courbe $(\cos^3 t, \sin^3 t)$ en $t =
\frac\pi4$, et vérifier que le segment de cette tangente découpé par les
axes est de \omterm{def:b1:curves:arclength}{longueur} $1$ --- une propriété célèbre de l'astroïde
(vraie en tout point régulier).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:curves:4}
Tracer les \omterm{def:b1:curves:polar}{courbes en polaires} $r = \cos 2\theta$ (rose à quatre pétales) et
$r = \frac{1}{\cos\theta}$ sur $\intoo{-\frac\pi2}{\frac\pi2}$
(reconnaître une droite).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:curves:5}
Étudier la \omterm{def:b1:curves:polar}{courbe en polaires} $r = 1 + 2\cos\theta$ (un limaçon)~: domaine
où $r \geq 0$ et où $r < 0$ (les points tracés avec un rayon négatif
se placent sur le rayon opposé), passage par l'origine, boucle intérieure,
tracé.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:curves:6}
Identifier la \omterm{def:b1:curves:conics}{conique} $9x^2 + 25y^2 - 36x - 50y - 164 = 0$~: la réduire en
complétant les carrés, donner centre, demi-axes, foyers, excentricité.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:curves:7}
Démontrer que l'équation polaire d'une \omterm{def:b1:curves:conics}{conique} dont un foyer est à l'origine
est
\[
r = \frac{p}{1 + e\cos\theta}
\]
(directrice verticale, à distance $\frac pe$ à droite du foyer).
Quelles valeurs de $\theta$ sont permises lorsque $e > 1$~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:curves:8}
À partir de la propriété bifocale $MF + MF' = 2a$, déduire la «~construction
du jardinier~» de l'ellipse, et démontrer que la tangente en $M$
fait des angles égaux avec $MF$ et $MF'$ \emph{(propriété de réflexion~;
dériver $\norm{M(t) - F} + \norm{M(t) - F'} = 2a$ et
interpréter la somme nulle des produits de vecteurs unitaires)}.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:curves:9}
La \emph{lemniscate de Bernoulli} est la \omterm{def:b1:curves:polar}{courbe en polaires} $r^2 = \cos
2\theta$ (prendre $r = \sqrt{\cos2\theta}$ là où c'est défini).
\begin{enumerate}
  \item Donner son domaine, ses symétries, les tangentes à l'origine, et
        la tracer.
  \item Démontrer que c'est le lieu des points $M$ tels que $MF \cdot MF'
        = \frac12$, où $F, F' = \bigl(\pm\frac{1}{\sqrt2},
        0\bigr)$. \emph{(Calculer $MF^2\,MF'^2$ en \omterm{prop:b1:vspaces:coordinates}{coordonnées}
        polaires.)}
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:curves:10}
Calculer la \omterm{def:b1:curves:arclength}{longueur} totale de la cardioïde $r = 1 + \cos\theta$
à l'aide de la formule polaire de la \cref{def:b1:curves:arclength}
\emph{(l'identité $1 + \cos\theta = 2\cos^2\frac\theta2$ transforme
la racine carrée en $2\abs{\cos\frac\theta2}$)}.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:curves:11}
Montrer que la tangente à l'ellipse $(a\cos t, b\sin t)$ au
point de paramètre $t$ a pour équation
\[
\frac{x\cos t}{a} + \frac{y\sin t}{b} = 1 ,
\]
et en déduire la tangente à $\frac{x^2}{a^2} + \frac{y^2}{b^2} = 1$
en un point $(x_0, y_0)$ de l'ellipse~: $\frac{x\,x_0}{a^2} +
\frac{y\,y_0}{b^2} = 1$ (la règle du «~dédoublement des carrés~»).
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:curves:12}
La \emph{spirale logarithmique} est la \omterm{def:b1:curves:polar}{courbe en polaires} $r =
\eu^{k\theta}$ ($k > 0$ fixé, $\theta \in \R$).
\begin{enumerate}
  \item Montrer que l'angle $V$ entre le rayon vecteur et la
        tangente est \emph{constant} ($\tan V = \frac1k$) --- la
        spirale coupe tout rayon sous le même angle.
  \item Montrer que faire tourner la spirale d'un angle $c$ l'envoie
        sur son \omterm{def:b1:linmaps:kerim}{image} par l'homothétie de rapport $\eu^{-kc}$~: toute
        rotation de la spirale en est un agrandissement
        (autosimilarité).
  \item Calculer la \omterm{def:b1:curves:arclength}{longueur} de l'arc $\theta \in
        \intoc{-\infty}{\theta_0}$ (comme limite des \omterm{def:b1:curves:arclength}{longueurs} sur
        $\intcc{A}{\theta_0}$, $A \to -\infty$) et observer qu'elle
        est finie~: une courbe s'enroulant une infinité de fois
        autour de l'origine, de \omterm{def:b1:curves:arclength}{longueur} finie.
\end{enumerate}
\end{exercise}

\section{Problème~: la cycloïde, reine des courbes}

\begin{problem}\label{pb:b1:curves:1}
Une roue de rayon $R$ roule sans glisser le long de l'axe des $x$~;
le point de la jante initialement à l'origine décrit la
\emph{cycloïde}. Le dix-septième siècle s'est battu autour de cette courbe
--- Galilée en a pesé des découpes de papier, Wren l'a mesurée,
Roberval en a calculé l'aire, Huygens en a fait des horloges --- et
chacun de leurs résultats est à la portée de ce chapitre. Nous
démontrons les quatre classiques~: la construction de la tangente, la \omterm{def:b1:curves:arclength}{longueur}
$8R$ de Wren, l'aire $3\pi R^2$, et la propriété de tautochronisme
de Huygens.\index{cycloïde}\index{tautochrone}

\medskip
\noindent\textbf{Partie I --- Le roulement et la tangente.}
\begin{enumerate}
  \item Après que la roue a tourné d'un angle $t$, son centre
        est en $\Omega(t) = (Rt, R)$ (roulement sans glissement~:
        distance de contact $=$ arc déroulé). Montrer que le point
        marqué est en
        \[
        M(t) = \bigl(R(t - \sin t),\; R(1 - \cos t)\bigr).
        \]
  \item Calculer $f'(t)$ et montrer que $\norm{f'(t)} =
        2R\,\abs{\sin\frac t2}$~; localiser les points singuliers
        (\omterm{ex:b1:curves:astroid}{points de rebroussement} --- cf.\ \cref{exo:b1:curves:2}) et le sommet
        de chaque arche.
  \item Soit $C(t) = (Rt, 0)$ le point de contact et $T(t) =
        (Rt, 2R)$ le sommet de la roue. Démontrer que pour $0 < t <
        2\pi$ le vecteur $M - C$ est \emph{normal} à la courbe en
        $M(t)$ et que le vecteur $T - M$ lui est \emph{tangent}~: pour tracer
        la tangente à une cycloïde, joindre le point au sommet de
        son cercle roulant. \emph{(Tout factoriser par
        $\sin\frac t2$ et $\cos\frac t2$.)}
  \item Interpréter la question 3 cinématiquement~: le point de contact est
        le \emph{centre instantané de rotation}, et la
        vitesse de $M$ est égale à sa distance à $C$ (pour une vitesse
        angulaire unité). Vérifier $\norm{M - C} = 2R\abs{\sin\frac t2}$.
  \item Démontrer la relation hauteur--vitesse
        \[
        \norm{f'(t)}^2 = 2R\,y(t) :
        \]
        sur une cycloïde parcourue à vitesse angulaire unité, la
        vitesse en chaque point est exactement la vitesse de chute \omterm{def:b1:vspaces:free}{libre} pour une
        descente égale à la hauteur courante. (À garder pour
        la partie IV.)
\end{enumerate}

\medskip
\noindent\textbf{Partie II --- Le théorème de Wren~: l'arche a pour \omterm{def:b1:curves:arclength}{longueur}
$8R$.}
\begin{enumerate}[resume]
  \item À l'aide de la \cref{def:b1:curves:arclength}, calculer la \omterm{def:b1:curves:arclength}{longueur}
        d'une arche~:
        \[
        L = \int_0^{2\pi} 2R\sin\frac t2\,\dd t = 8R
        \]
        (théorème de Wren, 1658). Quatre diamètres de roue, et pas de
        $\pi$ nulle part.
  \item Calculer la \omterm{def:b1:curves:arclength}{longueur d'arc} depuis le \omterm{ex:b1:curves:astroid}{point de rebroussement}~: $s(t) =
        4R\bigl(1 - \cos\frac t2\bigr)$, et vérifier $s(2\pi) =
        8R$.
  \item Mesurer maintenant l'arc depuis le \emph{sommet} $t = \pi$~:
        $\sigma(t) = \abs{4R\cos\frac t2}$. Démontrer la relation
        intrinsèque
        \[
        \sigma^2 = 8R\,\bigl(2R - y\bigr) :
        \]
        le carré de la distance d'arc au sommet est proportionnel à
        la hauteur de descente sous le sommet.
  \item Vérifications~: retrouver à partir de la question 8 que la demi-arche
        du sommet au \omterm{ex:b1:curves:astroid}{point de rebroussement} est de \omterm{def:b1:curves:arclength}{longueur} $4R$, et comparer avec le
        calcul sur l'astroïde de l'\cref{ex:b1:curves:circlelength}
        --- les deux courbes ont des \omterm{def:b1:curves:arclength}{longueurs} algébriques, sans $\pi$~;
        expliquer ce qui rend cela possible bien que les deux soient
        construites à partir de cercles. \emph{(Regarder la forme de
        $\norm{f'}$.)}
\end{enumerate}

\medskip
\noindent\textbf{Partie III --- L'aire de Roberval~: $3\pi R^2$.}
\begin{enumerate}[resume]
  \item Justifier que l'aire entre une arche et le sol est
        $A = \int_0^{2\pi} y(t)\,x'(t)\,\dd t$ \emph{(la
        substitution $x = x(t)$ dans $\int y\,\dd x$,
        \cref{thm:b1:integration:parts}~; $x$ est croissante)}.
  \item Calculer
        \[
        A = R^2\int_0^{2\pi}(1 - \cos t)^2\,\dd t = 3\pi R^2 :
        \]
        exactement \emph{trois} fois l'aire de la roue ---
        le rapport que Galilée avait deviné en pesant.
  \item Même méthode pour l'astroïde $(\cos^3t, \sin^3t)$~: montrer
        que l'aire enfermée vaut $\frac{3\pi}8$
        \emph{(linéariser $\sin^4 t\cos^2 t$~; seul le terme
        constant survit sur une période entière)}.
  \item Vérifier la formule de la question 10 sur le demi-cercle unité
        supérieur $(\cos t, \sin t)$, $t$ allant de $\pi$ à $0$~:
        rend-elle bien $\frac\pi2$~?
\end{enumerate}

\medskip
\noindent\textbf{Partie IV --- La tautochrone de Huygens.} Retournons
l'arche~: une bille sans frottement glisse, sous la gravité $g$, dans la
cuvette cycloïdale
\[
x(t) = R(t + \sin t), \qquad y(t) = R(1 - \cos t)
\qquad (t \in \intcc{-\pi}{\pi}),
\]
dont le point le plus bas est l'origine ($y$ mesuré vers le haut).
\begin{enumerate}[resume]
  \item Calculer la vitesse $\norm{f'(t)} = 2R\cos\frac t2$, la
        \omterm{def:b1:curves:arclength}{longueur d'arc} depuis le fond $s(t) = 4R\sin\frac t2$, et
        démontrer l'identité clé
        \[
        y = \frac{s^2}{8R} .
        \]
  \item La bille lâchée sans vitesse initiale depuis le point de paramètre
        $t_0 > 0$ obéit à la conservation de l'énergie~: si $s(\tau)$ désigne
        son abscisse curviligne à l'instant $\tau$, alors $\frac12\bigl(
        \frac{\dd s}{\dd\tau}\bigr)^2 + g\,y = g\,y_0$. Réécrire
        ceci, à l'aide de la question 14, sous la forme
        \[
        \Bigl(\frac{\dd s}{\dd\tau}\Bigr)^{\!2}
        = \frac{g}{4R}\,\bigl(s_0^2 - s^2\bigr),
        \qquad s_0 = s(t_0).
        \]
  \item Dériver par rapport à $\tau$ et obtenir l'oscillateur
        harmonique
        \[
        \frac{\dd^2 s}{\dd\tau^2} = -\frac{g}{4R}\,s ;
        \]
        le résoudre avec le \cref{thm:b1:diffeq:homogeneous2} et les
        conditions initiales~: $s(\tau) = s_0\cos(\omega\tau)$,
        $\omega = \sqrt{g/4R}$.
  \item En déduire la \emph{propriété de tautochronisme} (Huygens, 1659)~:
        le temps de descente jusqu'au fond,
        \[
        T_{\downarrow} = \frac{\pi}{2\omega}
        = \pi\sqrt{\frac Rg}\,,
        \]
        ne dépend pas du point de lâcher --- des billes lâchées
        \omterm{def:b1:logic:sets}{ensemble} depuis deux hauteurs quelconques de la cuvette arrivent
        \omterm{def:b1:logic:sets}{ensemble}.
  \item Vérifier par substitution que $s(\tau) =
        s_0\cos\omega\tau$ satisfait exactement l'équation d'énergie
        du premier ordre de la question 15 (et pas seulement l'équation
        \omterm{def:b1:derivative:def}{dérivée}), et expliquer en une phrase pourquoi un
        pendule circulaire n'est qu'\emph{approximativement}
        isochrone alors que la cycloïde l'est exactement.
  \item Numériquement~: quel rayon $R$ rend le temps de descente
        exactement égal à une seconde ($g = 9.81$)~? Noter à quel point la
        réponse est proche d'un mètre, et comment la période complète
        d'oscillation $2\pi\sqrt{4R/g}$ se compare à la formule du
        pendule aux petits angles $2\pi\sqrt{\ell/g}$ pour $\ell = 4R$.
\end{enumerate}

\medskip
\noindent\textbf{Partie V --- Dividendes et synthèse.}
\begin{enumerate}[resume]
  \item Appliquer la règle de la tangente de la question 3 en $t =
        \frac\pi2$ (prendre $R = 1$)~: calculer $M$, $T$, la
        direction de $MT$, et la confronter à
        $f'(\frac\pi2)$.
  \item (Trochoïdes) Marquer plutôt un point à distance $d$ du
        centre ($d \neq R$)~: la courbe est $x = Rt -
        d\sin t$, $y = R - d\cos t$. Montrer que pour $d < R$ la
        courbe est partout régulière et ne présente aucun
        comportement singulier ($x' > 0$~: elle avance), tandis que pour $d
        > R$ l'abscisse $x'$ change de signe et la courbe fait
        des boucles --- la roue de chemin de fer à boudin dont les points de
        la jante voyagent \emph{à reculons}.
  \item (Avant-goût de la brachistochrone) Du \omterm{ex:b1:curves:astroid}{point de rebroussement} $(\pi R, 2R)$ de
        la cuvette jusqu'au fond, comparer le temps de descente sur la cycloïde
        $\pi\sqrt{R/g}$ avec le temps le long de la glissière rectiligne
        joignant les mêmes points \emph{(accélération constante
        $g\sin\alpha$ le long de la corde)}~: montrer que la corde demande
        $\sqrt{\pi^2 + 4}\,\sqrt{R/g} \approx 3.72\sqrt{R/g}$.
        La courbe bat la droite --- elle est, de fait, la plus rapide
        de toutes les courbes, résultat du calcul des variations.
  \item Montrer que sur l'arche ($0 < t < 2\pi$),
        \[
        \frac{\dd y}{\dd x} = \cot\frac t2,
        \qquad
        \frac{\dd^2 y}{\dd x^2} =
        -\frac{1}{4R\sin^4\frac t2} < 0 :
        \]
        l'arche est concave, avec des tangentes verticales exactement aux
        \omterm{ex:b1:curves:astroid}{points de rebroussement}.
  \item Retrouver géométriquement la tangente verticale au \omterm{ex:b1:curves:astroid}{point de rebroussement} de
        l'\cref{exo:b1:curves:2}~: calculer la direction limite
        de la corde $MT$ de la question 3 quand $t \to
        0^{+}$, sans aucun développement.
  \item Synthèse, en quatre phrases~: quels trois théorèmes nommés
        ce problème a démontrés (avec leurs valeurs $8R$, $3\pi
        R^2$, $\pi\sqrt{R/g}$ et leurs auteurs)~; quel unique
        procédé de calcul (les factorisations par $\sin\frac t2$, $\cos\frac t2$)
        a alimenté toutes les parties I, II et IV~; comment
        la relation intrinsèque $y = s^2/8R$ a converti la géométrie
        en une équation différentielle \omterm{def:b1:linmaps:def}{linéaire}~; et sur quel chapitre
        de ce livre chaque partie s'est appuyée.
\end{enumerate}
\end{problem}
