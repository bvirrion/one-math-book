\chapter{Séries numériques}\label{ch:b1:series}

Sommer une infinité de nombres, c'est prendre la limite des sommes
partielles --- ni plus, ni moins. Ce chapitre met en place les
définitions et les critères de convergence utilisables en première
année~: comparaison et équivalents pour les termes positifs, la \omterm{thm:b1:series:ratio}{règle
de d'Alembert}, la comparaison \omterm{thm:b1:integration:def}{série-intégrale} qui donne les \omterm{thm:b1:series:riemann}{séries de
Riemann}, la \omterm{thm:b1:series:absolute}{convergence absolue} et le critère spécial des \omterm{thm:b1:series:alternating}{séries
alternées}. La théorie plus fine (produits de \omterm{def:b1:series:def}{séries}, sommation par
paquets, \omterm{def:b1:series:def}{séries} de fonctions) relève de la deuxième année.

\section{Généralités}

\begin{definition}\label{def:b1:series:def}
Étant donnée une suite $(u_n)$, la \emph{série}\index{série} $\sum u_n$
est la suite des \emph{sommes partielles} $S_N = \sum_{n=0}^{N} u_n$.
La série \emph{converge} lorsque $(S_N)$ converge~; la limite est la
\emph{somme} $\sum_{n=0}^{\infty} u_n$, et $R_N = \sum_{n > N} u_n =
S - S_N$ est le \emph{reste}, qui tend vers $0$.
\end{definition}

\begin{example}[Série géométrique]\label{ex:b1:series:geometric}
Pour $q \in \C$~: $\;S_N = \sum_{n=0}^{N} q^n = \frac{1 -
q^{N+1}}{1-q}$ ($q \neq 1$). La \omterm{def:b1:series:def}{série} converge si et seulement si
$\abs q < 1$ (\cref{exo:b1:seq:3}), avec\index{série géométrique}
\[
\sum_{n=0}^{\infty} q^n = \frac{1}{1 - q} .
\]
\end{example}

\begin{example}[Les décimaux périodiques sont des séries géométriques]\label{ex:b1:series:periodicdecimal}
Que vaut $0.363636\dots$~? Son écriture même est une \omterm{def:b1:series:def}{série}~:
\[
0.\overline{36} = \sum_{k=1}^{\infty} \frac{36}{100^k}
= 36\cdot\frac{1/100}{1 - 1/100} = \frac{36}{99} =
\frac{4}{11} ,
\]
par la somme géométrique de raison $q = \frac{1}{100}$. En général,
un bloc $B$ de $p$ chiffres répété indéfiniment vaut
$\frac{B}{10^p - 1}$ --- le mécanisme derrière le critère de
périodicité du \cref{pb:b1:reals:1}, que le langage de ce chapitre
énonce enfin en une ligne~: un \omterm{pb:b1:reals:1}{développement décimal} est une \omterm{def:b1:series:def}{série}
convergente, périodique à partir d'un certain rang exactement
lorsque sa somme est rationnelle. La machinerie des chiffres du
chapitre 10, bâtie là-bas avec de simples \omterm{def:b1:reals:bounds}{bornes supérieures}, était
de la théorie des \omterm{def:b1:series:def}{séries} voyageant incognito.
\end{example}

\begin{proposition}[Premiers faits]\label{prop:b1:series:first}
\begin{enumerate}
  \item Si $\sum u_n$ converge, alors $u_n \to 0$. (La réciproque est
        \emph{fausse}~: la \omterm{def:b1:series:def}{série} harmonique.)
  \item Linéarité~: les \omterm{def:b1:series:def}{séries} convergentes s'ajoutent et se
        multiplient par un scalaire, avec les sommes attendues.
  \item (Télescopage\index{série télescopique}) $\sum (v_{n+1} -
        v_n)$ converge si et seulement si $(v_n)$ converge, de somme
        $\lim v_n - v_0$.
  \item Modifier un nombre fini de termes ne change pas la nature de
        la \omterm{def:b1:series:def}{série} (seulement la somme).
\end{enumerate}
\end{proposition}

\begin{proof}
(1) $u_N = S_N - S_{N-1} \to S - S = 0$. La \omterm{def:b1:series:def}{série} harmonique a
$u_n = \frac1n \to 0$ et diverge pourtant (\cref{exo:b1:seq:5}). (2)
Opérations sur les limites. (3) $S_N = v_{N+1} - v_0$. (4) Les sommes
partielles sont modifiées d'une quantité constante à partir d'un
certain rang.
\end{proof}

\begin{example}[Prévoir les décimales avec le reste géométrique]\label{ex:b1:series:georemainder}
Pour $\abs q < 1$, le reste de la \omterm{ex:b1:series:geometric}{série géométrique} est
explicite~:
\[
R_N = \sum_{n = N+1}^{\infty} q^n = \frac{q^{N+1}}{1 - q} .
\]
Cela convertit un objectif de précision en un nombre de termes,
avant tout calcul. Pour évaluer $\sum_{n\geq0} \bigl(\frac13\bigr)^n =
\frac32$ à $10^{-10}$ près~: il faut $\frac{(1/3)^{N+1}}{2/3} \leq
10^{-10}$, c'est-à-dire $3^{N} \geq \frac{3}{2}\cdot 10^{10}$,
c'est-à-dire $N \geq 22$ (car $3^{22} \approx 3.1\cdot10^{10}$)~:
vingt-trois termes, connus à l'avance. Toute estimation à vitesse
géométrique des devoirs maison (la \omterm{def:b1:series:def}{série} en $\frac13$ pour $\ln 2$,
les \omterm{def:b1:functions:arc}{arctangentes} de Machin dans le \cref{pb:b1:taylor:1}) est ce
budget de deux lignes en habit de professionnel.
\end{example}

\begin{example}[Un télescope plus long]\label{ex:b1:series:telescope3}
Calculons $\sum_{n\geq1} \frac{1}{n(n+1)(n+2)}$. Décomposition en
éléments simples (\cref{ch:b1:fractions})~:
\[
\frac{1}{n(n+1)(n+2)}
= \frac{1/2}{n} - \frac{1}{n+1} + \frac{1/2}{n+2}
= \frac12\Bigl(\frac{1}{n(n+1)} - \frac{1}{(n+1)(n+2)}\Bigr),
\]
où la seconde forme --- une différence de valeurs consécutives de
$w_n = \frac{1}{n(n+1)}$ --- est celle qui se télescope. D'où
\[
\sum_{n=1}^{N} \frac{1}{n(n+1)(n+2)}
= \frac12\Bigl(w_1 - w_{N+1}\Bigr)
= \frac12\Bigl(\frac12 - \frac{1}{(N+1)(N+2)}\Bigr)
\longrightarrow \frac14 .
\]
L'idée à retenir~: les éléments simples à trois termes se
télescopent rarement tels quels~; il faut d'abord les regrouper en
une différence $w_n - w_{n+1}$ --- la récompense n'est pas seulement
la convergence, mais la somme exacte, qu'aucun critère de
comparaison ne fournit jamais.
\end{example}

\section{Séries à termes positifs}

\begin{theorem}[Sommes partielles bornées]\label{thm:b1:series:positive}
Si $u_n \geq 0$ pour tout $n$, les sommes partielles croissent, donc~:
$\sum u_n$ converge $\iff$ ses sommes partielles sont majorées. D'où
le \emph{critère de comparaison}~: si $0 \leq u_n \leq v_n$ pour tout
$n$ (assez grand),
\[
\sum v_n \text{ converge} \implies \sum u_n \text{ converge},
\qquad
\sum u_n \text{ diverge} \implies \sum v_n \text{ diverge}.
\]
Et le \emph{critère des équivalents}~: si $u_n \sim v_n$ avec $v_n \geq
0$, les deux \omterm{def:b1:series:def}{séries} sont de même nature.
\end{theorem}

\begin{proof}
\omterm{thm:b1:seq:monotone}{Théorème de la limite monotone} (\cref{thm:b1:seq:monotone}) pour le
premier point~; comparaison des sommes partielles pour le second.
Équivalents~: pour $n$ grand, $\frac12 v_n \leq u_n \leq 2 v_n$
(définition de $\sim$ avec $\varepsilon = \frac12$), et la comparaison
s'applique dans les deux sens.
\end{proof}

\begin{example}[Un équivalent qui prouve la divergence]\label{ex:b1:series:equivdiverge}
Nature de $\sum_{n\geq1} n\sin\dfrac{1}{n^2}$~? Comme
$\frac{1}{n^2} \to 0$ et $\sin h \sim h$ en $0$~:
\[
n\sin\frac{1}{n^2} \;\sim\; n\cdot\frac{1}{n^2} = \frac1n ,
\]
et le critère des équivalents transfère la divergence de la \omterm{def:b1:series:def}{série}
harmonique~: divergente --- alors même que les termes tendent vers
$0$. Un développement, une échelle, un verdict~; le même schéma en
deux temps (un équivalent, puis une lecture dans l'échelle de
Riemann ou géométrique) décide les quatre \omterm{def:b1:series:def}{séries} de
l'\cref{exo:b1:series:3}.
\end{example}

\begin{example}[Le critère des équivalents en une ligne]\label{ex:b1:series:equivtest}
Nature de $\sum_{n \geq 1} \frac{\sqrt{n+1} - \sqrt n}{n}$~?
Multiplions le numérateur par sa quantité \omterm{def:b1:complex:field}{conjuguée}~:
\[
\frac{\sqrt{n+1} - \sqrt n}{n}
= \frac{1}{n\,(\sqrt{n+1} + \sqrt n)}
\sim \frac{1}{2\,n^{3/2}} ,
\]
une échelle de Riemann convergente ($\alpha = \frac32 > 1$)~: la
\omterm{def:b1:series:def}{série} converge. Toute la décision a tenu en un équivalent et une
lecture --- à condition que les termes soient positifs, ce qu'ils
sont. L'idée à retenir~: pour les \omterm{def:b1:series:def}{séries} positives, toute la théorie
de la convergence est un \emph{dictionnaire d'échelles}
($n^{-\alpha}$, $q^n$, $\frac{1}{n(\ln n)^\alpha}$) plus le droit de
remplacer un terme par un équivalent~; le travail analytique est
dans l'asymptotique (\cref{ch:b1:taylor}), jamais dans la sommation.
\end{example}

\begin{theorem}[Comparaison série-intégrale~; séries de Riemann]\label{thm:b1:series:riemann}
Soit $f$ \omterm{def:b1:continuity:continuous}{continue}, positive et \emph{décroissante} sur
$\intco{1}{+\infty}$. Alors
\[
\int_1^{N+1} f(t)\,\dd t \;\leq\; \sum_{n=1}^{N} f(n) \;\leq\; f(1)
+ \int_1^{N} f(t)\,\dd t ,
\]
donc $\sum f(n)$ converge si et seulement si $\bigl(\int_1^x f\bigr)$
est bornée. En particulier, pour $\alpha \in \R$~:\index{séries de Riemann}
\[
\sum_{n \geq 1} \frac{1}{n^\alpha} \text{ converge}
\iff \alpha > 1,
\]
et $\sum_{n=1}^{N} \frac1n = \ln N + O(1)$.
\end{theorem}

\begin{proof}
Pour $n \leq t \leq n+1$, la monotonie donne $f(n+1) \leq f(t)
\leq f(n)$~; en intégrant sur $\intcc{n}{n+1}$ (un segment de
longueur $1$)~:
\[
f(n+1) \;\leq\; \int_n^{n+1} f(t)\,\dd t \;\leq\; f(n) .
\]
En sommant les inégalités de droite pour $n = 1, \dots, N-1$, on
obtient $\int_1^{N} f \leq \sum_{n=1}^{N-1} f(n)$, d'où
l'encadrement supérieur après avoir ajouté $f(N) \leq f(1)$~; en
sommant celles de gauche pour $n = 1, \dots, N$, on obtient
$\sum_{n=2}^{N+1} f(n) \leq \int_1^{N+1} f$, qui après réindexation
est l'encadrement inférieur. Convergence~: les sommes partielles et
les \omterm{thm:b1:integration:def}{intégrales} $\int_1^x f$ se majorent mutuellement à la constante
$f(1)$ près, et toutes deux sont croissantes, donc l'une est bornée
si et seulement si l'autre l'est (\cref{thm:b1:series:positive}).
Pour $f(t) = t^{-\alpha}$ ($\alpha \neq 1$)~: $\int_1^x
t^{-\alpha}\dd t = \frac{x^{1-\alpha} - 1}{1 - \alpha}$, bornée si et
seulement si $\alpha > 1$~; pour $\alpha = 1$ l'\omterm{thm:b1:integration:def}{intégrale} vaut
$\ln x \to \infty$, et l'encadrement donne $\ln(N+1) \leq H_N \leq 1
+ \ln N$. Pour $\alpha \leq 0$ les termes ne tendent pas vers $0$.
\end{proof}

\begin{example}[La pile harmonique]\label{ex:b1:series:harmonicstack}
Combien de termes la \omterm{def:b1:series:def}{série} harmonique doit-elle accumuler pour
dépasser $20$~? L'encadrement $\ln(N+1) \leq H_N \leq 1 + \ln N$
répond sans aucune sommation~: $H_N \geq 20$ exige $1 + \ln N \geq
20$, c'est-à-dire $N \geq \eu^{19} \approx 1.8\cdot10^{8}$, et est
garanti dès que $\ln(N + 1) \geq 20$, c'est-à-dire $N \approx
\eu^{20} \approx 4.9\cdot10^{8}$. (Le devoir maison affine cela en
$N \approx \eu^{20 - \gamma} \approx 2.7\cdot10^{8}$ grâce à la
\omterm{ex:b1:series:oddtelescope}{constante d'Euler}.) L'idée à retenir~: la comparaison
\omterm{thm:b1:integration:def}{série-intégrale} ne se contente pas de décider la convergence --- elle
\emph{localise} les sommes partielles avec une précision
logarithmique, transformant un calcul désespéré (des centaines de
millions de termes) en une estimation de deux lignes.
\end{example}

\begin{theorem}[Règle de d'Alembert]\label{thm:b1:series:ratio}
Soit $u_n > 0$ avec $\frac{u_{n+1}}{u_n} \to \ell$.\index{règle de d'Alembert}
\begin{itemize}
  \item Si $\ell < 1$~: $\sum u_n$ converge~;
  \item si $\ell > 1$~: $u_n \to +\infty$, divergence~;
  \item si $\ell = 1$~: pas de conclusion ($\sum \frac1n$ diverge, $\sum
        \frac{1}{n^2}$ converge).
\end{itemize}
\end{theorem}

\begin{proof}
Si $\ell < 1$, fixons $q \in \intoo{\ell}{1}$~: au-delà d'un certain
$N$, $u_{n+1} \leq q\,u_n$, donc $u_n \leq u_N q^{\,n-N}$ par
récurrence~: comparaison avec une \omterm{ex:b1:series:geometric}{série géométrique}. Si $\ell > 1$~:
au-delà d'un certain $N$ la suite $(u_n)$ est croissante, donc elle
ne peut tendre vers $0$ (sa limite, si elle existe, est $\geq u_N >
0$)~; par la \cref{prop:b1:series:first}~(1), divergence --- et de fait
$u_n \geq u_N q^{n-N}$ avec $q > 1$ donne $u_n \to \infty$.
\end{proof}

\begin{example}\label{ex:b1:series:ratio}
$\sum \frac{x^n}{n!}$ converge pour tout $x > 0$~: le rapport vaut
$\frac{x}{n+1} \to 0$. Sa somme est $\eu^x$~: par l'inégalité de
Taylor--Lagrange (\cref{thm:b1:taylor:lagrange}) sur $\intcc{0}{x}$,
\[
\Bigl| \eu^x - \sum_{k=0}^{n} \frac{x^k}{k!} \Bigr|
\leq \eu^{x}\, \frac{x^{n+1}}{(n+1)!} \xrightarrow[n\to\infty]{} 0 ,
\]
la borne tendant vers $0$ parce que la factorielle l'emporte
(l'\cref{exo:b1:integration:9}~(1) utilisait le même fait). Le même
argument somme les \omterm{def:b1:series:def}{séries} de $\sin$, $\cos$, $\sinh$, $\cosh$ sur
$\R$ tout entier.
\end{example}

\begin{example}[La règle de d'Alembert est suffisante, non nécessaire]\label{ex:b1:series:rationotnecessary}
Soit $u_n = 2^{-n}$ pour $n$ pair et $u_n = 2^{-n-2}$ pour $n$
impair. Les rapports consécutifs oscillent entre $\frac{1}{8}$ et
$\frac12\cdot4 = 2$, de sorte que $\frac{u_{n+1}}{u_n}$ n'a pas de
limite et que d'Alembert reste muet --- pourtant $u_n \leq 2^{-n}$
et le critère de comparaison règle la convergence instantanément.
L'hypothèse de la règle (le rapport \emph{converge}) est une vraie
restriction~: elle convient aux termes dotés d'une seule structure
multiplicative dominante (factorielles, puissances), et échoue sur
tout ce qui respire. Quand les rapports se conduisent mal, revenez à
la comparaison avec une enveloppe géométrique --- ce que la \omterm{thm:b1:series:ratio}{règle de
d'Alembert} n'a jamais cessé d'être, comme le montre sa démonstration.
\end{example}

\begin{example}[La règle de d'Alembert dans les batailles de factorielles]\label{ex:b1:series:ratiobinom}
Nature de $\sum_{n\geq0} \dfrac{(n!)^2}{(2n)!}$ (inverses des
coefficients binomiaux centraux, au facteur $n + 1$ près)~? Le
rapport fait s'effondrer les factorielles~:
\[
\frac{u_{n+1}}{u_n}
= \frac{((n+1)!)^2}{(n!)^2}\cdot\frac{(2n)!}{(2n+2)!}
= \frac{(n+1)^2}{(2n+1)(2n+2)}
\longrightarrow \frac14 < 1 :
\]
convergente, et avec de la marge --- les termes décroissent pour
l'essentiel comme $4^{-n}$, ce qui est cohérent avec
$\binom{2n}{n} \geq \frac{4^n}{2n+1}$ du \cref{pb:b1:integration:1}.
L'idée à retenir~: les quotients de factorielles sont exactement ce
que la \omterm{thm:b1:series:ratio}{règle de d'Alembert} digère --- chaque factorielle se simplifie
en une \omterm{def:b1:fractions:field}{fraction rationnelle} en $n$, dont la limite se lit sur les
termes dominants.
\end{example}

\section{Convergence absolue~; séries alternées}

\begin{theorem}[Convergence absolue]\label{thm:b1:series:absolute}
Si $\sum \abs{u_n}$ converge (\emph{convergence
absolue}\index{convergence absolue}), alors $\sum u_n$ converge, et
$\bigl|\sum u_n\bigr| \leq \sum \abs{u_n}$. Ceci vaut pour des termes
réels ou complexes.
\end{theorem}

\begin{proof}
Les sommes partielles vérifient, pour $M > N$ (critère de Cauchy,
\cref{thm:b1:seq:complete})~:
\[
\abs{S_M - S_N} = \Bigl| \sum_{n=N+1}^{M} u_n \Bigr|
\leq \sum_{n=N+1}^{M} \abs{u_n},
\]
quantité petite pour $N$ grand puisque les sommes partielles de
$\sum\abs{u_n}$ forment une \omterm{def:b1:seq:cauchy}{suite de Cauchy}. Donc $(S_N)$ est de
Cauchy, donc convergente. L'inégalité passe à la limite depuis
l'\omterm{prop:b1:complex:rules}{inégalité triangulaire} finie.
\end{proof}

\begin{example}[Convergence absolue, réelle et complexe]\label{ex:b1:series:absexamples}
$\sum_{n\geq1} \frac{\sin n}{n^2}$~: les termes changent de signe de
façon erratique (de fait $(\sin n)$ est \omterm{def:b1:topology:dense}{dense} dans $\intcc{-1}{1}$,
\cref{exo:b1:seq:12}), et aucune structure alternée n'est en vue. La
\omterm{thm:b1:series:absolute}{convergence absolue} sauve tout d'un coup~:
$\bigl|\frac{\sin n}{n^2}\bigr| \leq \frac{1}{n^2}$, une échelle
convergente, donc la \omterm{def:b1:series:def}{série} converge. Le même bouclier fonctionne sur
$\C$~: $\sum_{n\geq1}\frac{\eu^{\iu n}}{n^2}$ converge parce que
$\bigl|\frac{\eu^{\iu n}}{n^2}\bigr| = \frac{1}{n^2}$ --- les
configurations de signes, fussent-elles bidimensionnelles, sont sans
importance dès que les \omterm{def:b1:complex:field}{modules} sont sommables. L'idée à retenir~: la
\omterm{thm:b1:series:absolute}{convergence absolue} est le seul outil de ce chapitre qui ne demande
jamais comment les signes sont organisés~; essayez-la d'abord
(\cref{met:b1:series:decide}), et réservez les critères délicats aux
\omterm{def:b1:series:def}{séries} qui lui échappent.
\end{example}

\begin{theorem}[Critère spécial des séries alternées]\label{thm:b1:series:alternating}
Soit $(a_n)$ décroissante avec $a_n \to 0$. Alors la \omterm{thm:b1:series:alternating}{série alternée}
$\sum (-1)^n a_n$ converge~; sa somme est comprise entre deux sommes
partielles consécutives quelconques, et\index{série alternée}
\[
\abs{R_N} = \Bigl| \sum_{n > N} (-1)^n a_n \Bigr| \leq a_{N+1} .
\]
\end{theorem}

\begin{proof}
Les sommes partielles paires et impaires sont adjacentes~: $S_{2p+2}
- S_{2p} = a_{2p+2} - a_{2p+1} \leq 0$ (décroissance), $S_{2p+1} -
S_{2p-1} = a_{2p} - a_{2p+1} \geq 0$ (croissance), et $S_{2p} -
S_{2p+1} = a_{2p+1} \to 0$. Par le \cref{thm:b1:seq:adjacent} elles ont
une limite commune $S$, dont le critère des deux \omterm{def:b1:seq:subsequence}{suites extraites}
(\cref{prop:b1:seq:subsequences}) fait la limite de $(S_N)$~; de
plus $S$ est piégée entre deux sommes partielles consécutives, et
$\abs{S - S_N}$ vaut au plus l'écart à la suivante, $a_{N+1}$.
\end{proof}

\begin{example}[Série harmonique alternée]\label{ex:b1:series:ln2}
$\sum_{n \geq 1} \frac{(-1)^{n-1}}{n}$ converge (critère spécial)
mais pas absolument (\omterm{def:b1:series:def}{série} harmonique). Sa somme vaut $\ln 2$~: de
l'identité géométrique finie $\frac{1}{1+t} = \sum_{k=0}^{n-1} (-t)^k
+ \frac{(-t)^n}{1+t}$, on intègre sur $\intcc{0}{1}$~:
\[
\ln 2 = \sum_{k=1}^{n} \frac{(-1)^{k-1}}{k}
+ (-1)^n \int_0^1 \frac{t^n}{1+t}\,\dd t ,
\qquad
0 \leq \int_0^1 \frac{t^n}{1+t}\,\dd t \leq \frac{1}{n+1} \to 0 .
\]
La convergence est d'une lenteur pénible ($R_N \approx \frac{1}{N}$)
--- les \omterm{thm:b1:series:alternating}{séries alternées} convergent par compensation, non par
petitesse.
\end{example}

\begin{remark}[Pièges classiques avec les séries]
(i) \emph{Le critère des équivalents exige un signe}~: posons $v_n =
\frac{(-1)^n}{\sqrt n}$ et $u_n = v_n + \frac1n$. Alors
$\frac{u_n}{v_n} = 1 + \frac{(-1)^n}{\sqrt n} \to 1$, donc $u_n
\sim v_n$~; pourtant $\sum v_n$ converge (critère spécial) tandis
que $\sum u_n = \sum v_n + \sum \frac1n$ diverge. L'équivalence
contrôle la \emph{taille} des termes, et pour les \omterm{def:b1:series:def}{séries} de signe
quelconque la taille ne fait pas le destin --- le critère est
énoncé, et vrai, pour des termes positifs (à partir d'un certain
rang) seulement. (ii) \emph{$u_n \to 0$ ne prouve rien}~: la \omterm{def:b1:series:def}{série}
harmonique est l'éternel contre-exemple~; le sens direct
(\cref{prop:b1:series:first}~(1)) n'est qu'un test rapide de
divergence. (iii) \emph{Un rapport de limite $1$ est un silence,
pas une convergence}~: $\sum\frac1n$ et $\sum\frac{1}{n^2}$ ont
toutes deux un rapport $\to 1$~; passez aux échelles de Riemann ou
à la comparaison \omterm{thm:b1:integration:def}{série-intégrale}. (iv) \emph{Alterner ne suffit pas,
il faut décroître}~: $\sum \frac{(-1)^n}{n + (-1)^n}$ a l'air
alternée et ne se traite que par développement
(\cref{exo:b1:series:5})~; le devoir maison du \cref{ch:b1:taylor}
(question 23 là-bas) montre que le critère peut échouer purement et
simplement sans monotonie. (v) \emph{Grouper et réordonner n'est pas
gratuit}~: insérer des parenthèses est inoffensif pour une \omterm{def:b1:series:def}{série}
convergente mais peut créer de la convergence à partir de la
divergence ($1 - 1 + 1 - \cdots$ groupé par paires), et réordonner
peut changer la somme elle-même --- le drame mis en scène dans le
devoir maison de ce chapitre (\cref{pb:b1:series:1}).
\end{remark}

\begin{method}[Décider de la nature d'une série]\label{met:b1:series:decide}
\begin{enumerate}
  \item A-t-on $u_n \to 0$~? Sinon, divergence, on s'arrête.
  \item Termes positifs~: cherchez un équivalent de $u_n$
        (développements, \cref{ch:b1:taylor}~!), comparez avec les
        échelles de Riemann ou géométriques~; factorielles et
        puissances appellent la \omterm{thm:b1:series:ratio}{règle de d'Alembert}~; un $f(n)$
        décroissant appelle la comparaison \omterm{thm:b1:integration:def}{série-intégrale}.
  \item Signes variables~: essayez d'abord la \omterm{thm:b1:series:absolute}{convergence absolue}~;
        si elle échoue, le critère spécial des \omterm{thm:b1:series:alternating}{séries alternées}
        (vérifiez soigneusement la \emph{décroissance})~; au-delà,
        les outils de deuxième année.
\end{enumerate}
\end{method}

\begin{example}[Dénominateurs impairs, un demi-télescope]\label{ex:b1:series:oddtelescope}
Calculons $\sum_{n \geq 1} \dfrac{1}{4n^2 - 1}$. Éléments
simples~: $\frac{1}{(2n-1)(2n+1)} =
\frac12\bigl(\frac{1}{2n-1} - \frac{1}{2n+1}\bigr)$, donc
\[
\sum_{n=1}^{N} \frac{1}{4n^2 - 1}
= \frac12\Bigl(1 - \frac{1}{2N+1}\Bigr)
\longrightarrow \frac12 .
\]
Comparez avec $\sum \frac{1}{n(n+1)} = 1$
(\cref{exo:b1:series:1})~: même squelette télescopique, mais les
termes consécutifs sont ici distants de deux dans les nombres
impairs, et le facteur $\frac12$ enregistre le pas. L'idée à
retenir~: le télescopage est un changement de point de vue, non une
astuce --- dès que le terme général est une différence $w_n -
w_{n+1}$ d'une suite ayant une limite, la somme vaut $w_1 - \lim w$,
exactement la \cref{prop:b1:series:first}~(3).
\end{example}

\begin{remark}[La chaîne de l'analyse, rétrospectivement]
Ce chapitre est le point de convergence de l'analyse du volume, et
chaque critère nomme son ancêtre. Les sommes partielles bornées,
c'est le \omterm{thm:b1:seq:monotone}{théorème de la limite monotone} (\cref{ch:b1:seq}), lui-même
l'axiome de complétude du \cref{ch:b1:reals}~; la \omterm{thm:b1:series:absolute}{convergence absolue},
c'est le critère de Cauchy~; le critère intégral, c'est
l'encadrement des aires du \cref{ch:b1:integration}~; les équivalents
de termes généraux, ce sont les développements du
\cref{ch:b1:taylor}~; et le critère spécial des \omterm{thm:b1:series:alternating}{séries alternées},
c'est le lemme des \omterm{thm:b1:seq:adjacent}{suites adjacentes} endimanché. Lue à l'envers, la
chaîne explique à quoi chaque chapitre \emph{servait} --- et les
devoirs maison qui la traversent (développements en base $b$,
Cesàro--Stolz, les machines à irrationalité, la \omterm{ex:b1:series:oddtelescope}{constante d'Euler})
sont les mêmes quelques idées se retrouvant à une altitude toujours
plus haute. L'algèbre linéaire qui suit change de sujet, pas
d'exigence~: l'habitude des énoncés exacts avec erreur certifiée
survit au passage des limites aux dimensions.
\end{remark}

\begin{remark}[Où vont les séries ensuite]
Ce chapitre clôt l'analyse du volume et ouvre trois portes. Dans le
volume de Licence 2, les \omterm{def:b1:series:def}{séries} acquièrent une variable ($\sum
a_n x^n$~: les \omterm{def:b1:series:def}{séries} entières, avec leur rayon de convergence) puis
une théorie à valeurs fonctions (les \omterm{def:b1:series:def}{séries} de Fourier)~; la
dichotomie \omterm{thm:b1:series:absolute}{convergence absolue} contre convergence conditionnelle,
dramatisée dans le devoir maison ci-dessous, devient la clé de voûte
des deux. En probabilités (volume de Licence 3), les espérances de
variables aléatoires discrètes \emph{sont} des \omterm{def:b1:series:def}{séries}, et c'est la
\omterm{thm:b1:series:absolute}{convergence absolue} qui les rend bien définies. Et la \omterm{def:b1:series:def}{série} de
Riemann $\sum n^{-s}$, poussée à des $s$ complexes, devient la
fonction zêta --- la \omterm{def:b1:series:def}{série} la plus étudiée de toutes les
mathématiques.
\end{remark}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:b1:series:1}
Nature (et somme, en cas de télescopage) de~:
\[
\sum_{n\geq1} \frac{1}{n(n+1)},
\qquad
\sum_{n\geq2} \ln\Bigl(1 - \frac{1}{n^2}\Bigr),
\qquad
\sum_{n\geq0} \frac{3^n + 4^n}{5^n} .
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:series:2}
Nature de~: $\;\sum \dfrac{n^2}{2^n}$~; $\;\sum \dfrac{n!}{n^n}$~;
$\;\sum \dfrac{2^n\,n!}{n^n}$~; $\;\sum \dfrac{3^n\,n!}{n^n}$.
\emph{(\omterm{thm:b1:series:ratio}{Règle de d'Alembert}~; on rappelle que $\bigl(1 + \frac1n\bigr)^n \to \eu$.)}
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:series:3}
Nature de~: $\;\sum \sin\dfrac{1}{n^2}$~; $\;\sum
\Bigl(1 - \cos\dfrac1n\Bigr)$~; $\;\sum \dfrac{1}{\sqrt{n(n+1)}}$~;
$\;\sum \dfrac{\ln n}{n^2}$ \emph{(comparer avec $n^{-3/2}$)}.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:series:4}
Montrer que $\sum_{n\geq1} \frac{1}{n^2}$ converge, de somme $\leq 2$,
en utilisant $\frac{1}{n^2} \leq \frac{1}{n(n-1)}$ pour $n \geq 2$ et
une majoration télescopique.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:series:5}
Nature de $\;\sum \dfrac{(-1)^n}{\sqrt n}$, de $\;\sum
\dfrac{(-1)^n}{n + (-1)^n}$ \emph{(développer~: le critère spécial ne
s'applique pas directement --- pourquoi~?)}, et de $\;\sum
\sin\bigl(\pi\sqrt{n^2+1}\,\bigr)$ \emph{(réduire modulo $\pi$~:
$\sqrt{n^2+1} = n + \frac{1}{2n} + O(n^{-3})$)}.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:series:6}
Pour quels $\alpha > 0$ la \omterm{def:b1:series:def}{série} $\sum_{n \geq 2}
\dfrac{1}{n (\ln n)^{\alpha}}$ converge-t-elle~? \emph{(Comparaison
\omterm{thm:b1:integration:def}{série-intégrale}~; poser $u = \ln t$.)}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:series:7}
Soit $u_n = \dfrac{1}{n} - \ln\Bigl(1 + \dfrac1n\Bigr)$. Montrer que
$0 \leq u_n \leq \dfrac{1}{2n^2}$, que $\sum u_n$ converge, et en
déduire l'existence de la \omterm{ex:b1:series:oddtelescope}{constante d'Euler}\index{constante d'Euler}~:
\[
\gamma = \lim_{N \to \infty} \Bigl( \sum_{n=1}^{N} \frac 1n - \ln N
\Bigr) .
\]
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:series:8}
Calculer les sommes
\[
\sum_{n=1}^{\infty} \frac{1}{n(n+2)}
\qquad\text{et}\qquad
\sum_{n=0}^{\infty} \frac{n}{2^n} .
\]
\emph{(Pour la première~: éléments simples. Pour la seconde~: calculer
$\sum_{n=1}^{N} n x^{n-1}$ sous forme close et faire $N \to \infty$ en
$x = \frac12$.)}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:series:9}
(Condensation de Cauchy) Soit $(u_n)$ positive et décroissante.
Montrer que
\[
\sum_{n \geq 1} u_n \text{ converge}
\iff
\sum_{k \geq 0} 2^k\, u_{2^k} \text{ converge},
\]
en comparant des paquets de termes entre puissances consécutives de
$2$. En redéduire le critère de Riemann et
l'\cref{exo:b1:series:6}.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:series:10}
En utilisant l'identité \omterm{thm:b1:integration:def}{intégrale} de l'\cref{ex:b1:series:ln2}
adaptée à $\frac{1}{1+t^2}$, démontrer la \omterm{thm:b1:derivative:leibniz}{formule de Leibniz}
\[
\frac{\pi}{4} = \sum_{n=0}^{\infty} \frac{(-1)^n}{2n+1}
= 1 - \frac13 + \frac15 - \frac17 + \cdots
\]
avec la majoration de l'erreur $\abs{R_N} \leq \frac{1}{2N+3}$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:series:11}
Nature de $\displaystyle\sum_{n \geq 1} \frac{1}{n^{1 + 1/n}}$.
\emph{(Calculer la limite de $n^{1/n}$ et trouver un équivalent du
terme général~: le critère de Riemann exige un exposant
\emph{fixe}.)}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:series:12}
Soit $(u_n)$ positive et \emph{décroissante} avec $\sum u_n$
convergente. Montrer que $n\,u_n \to 0$ \emph{(majorer $n\,u_{2n}$
par une tranche $\sum_{k=n+1}^{2n} u_k$ et utiliser le critère de
Cauchy)}. Montrer que la réciproque est fausse, et que l'hypothèse
de monotonie ne peut pas être supprimée.
\end{exercise}

\section{Problème~: la constante d'Euler et la série qui change
de somme}

\begin{problem}[{Devoir maison --- $H_n = \ln n + \gamma +
\frac{1}{2n} + O(n^{-2})$, et le réarrangement de $1 - \frac12 +
\frac13 - \dots$ en $\frac{\ln 2}{2}$}]
\label{pb:b1:series:1}
Deux histoires se partagent la \omterm{def:b1:series:def}{série} harmonique. D'abord, la
comptabilité exacte de sa divergence~: $H_n - \ln n$ converge vers la
\omterm{ex:b1:series:oddtelescope}{constante d'Euler} $\gamma$ (\cref{exo:b1:series:7}), et ce problème
affine l'énoncé en une loi à deux côtés $\frac{1}{2(n+1)}
\leq H_n - \ln n - \gamma \leq \frac{1}{2n}$, qui certifie $\gamma
= 0.5772\dots$ à la main. Ensuite, le scandale de la convergence
conditionnelle~: la \omterm{def:b1:series:def}{série} harmonique alternée a pour somme $\ln 2$
(\cref{ex:b1:series:ln2}), et pourtant \emph{les mêmes termes, dans
un autre ordre}, ont pour somme $\frac{\ln 2}{2}$ --- ou $\ln 2 +
\frac12\ln\frac pq$ pour tous $p, q$, ou n'importe quel réel
(Riemann). Les deux histoires n'en font qu'une~: les sommes
réarrangées se calculent \emph{avec} la loi du
$\gamma$.\index{constante d'Euler}\index{réarrangement d'une série}

\medskip
\noindent\textbf{Partie I --- $\gamma$, encadré.} Posons $a_n = H_n
- \ln n$ et $b_n = H_n - \ln(n+1)$.
\begin{enumerate}
  \item En utilisant $\frac{t}{1+t} \leq \ln(1 + t) \leq t$, montrer
        que $(a_n)$ décroît, que $(b_n)$ croît, et qu'elles sont
        adjacentes~; leur limite commune est $\gamma$, avec $b_n \leq
        \gamma \leq a_n$ pour tout $n$.
  \item Premier tir numérique~: à partir de $H_{10} = 2.928968\dots$,
        encadrer $\gamma$ entre $b_{10} = 0.5311$ et $a_{10} =
        0.6264$. Quel $n$ ce grossier encadrement exigerait-il pour
        quatre décimales~?
  \item Montrer la représentation exacte du reste $a_n - \gamma =
        \sum_{k \geq n} w_k$ (limite des sommes partielles), où
        \[
        w_k = a_k - a_{k+1}
        = \ln\Bigl(1 + \frac1k\Bigr) - \frac{1}{k+1}
        = \int_k^{k+1} \frac{(k + 1 - t)}{t\,(k+1)}\,\dd t ,
        \]
        et déduire de la forme intégrale l'encadrement
        $\dfrac{1}{2(k+1)^2} \leq w_k \leq \dfrac{1}{2k(k+1)}$.
\end{enumerate}

\medskip
\noindent\textbf{Partie II --- La loi du $\frac{1}{2n}$.}
\begin{enumerate}[resume]
  \item Sommer les bornes de la question 3 (les deux côtés se
        télescopent ou se comparent à des télescopages) et conclure
        la loi~:
        \[
        \frac{1}{2(n+1)} \;\leq\; H_n - \ln n - \gamma \;\leq\;
        \frac{1}{2n} \qquad (n \geq 1).
        \]
  \item En déduire $H_n = \ln n + \gamma + \frac{1}{2n} +
        O\bigl(\frac{1}{n^2}\bigr)$~; précisément, montrer que
        $\gamma_n = H_n - \ln n - \frac{1}{2n}$ vérifie
        $-\frac{1}{2n(n+1)} \leq \gamma_n - \gamma \leq 0$.
  \item Certifier quatre décimales avec $n = 100$~: sachant que
        $H_{100} = 5.1873775\dots$, calculer $\gamma_{100} =
        0.577207\dots$ et conclure $\gamma = 0.5772 \pm 5\cdot10^{-5}$
        (valeur exacte $0.5772156\dots$).
  \item Deux dividendes de la loi, tous deux utiles plus loin~: quand
        $m \to \infty$,
        \[
        H_{2m} - H_m = \ln 2 - \frac{1}{4m} +
        O\Bigl(\frac{1}{m^2}\Bigr),
        \qquad
        \sum_{j=1}^{m} \frac{1}{2j-1} = \frac{\ln m}{2} + \ln 2
        + \frac\gamma2 + o(1) ,
        \]
        la seconde via $\sum_{j \leq m} \frac{1}{2j-1} = H_{2m}
        - \frac12 H_m$, et de même $\sum_{j=1}^{m}
        \frac{1}{2j} = \frac{\ln m}{2} + \frac\gamma2 + o(1)$.
\end{enumerate}

\medskip
\noindent\textbf{Partie III --- La \omterm{def:b1:series:def}{série} harmonique alternée,
au second ordre.}
\begin{enumerate}[resume]
  \item Montrer (par récurrence, ou par regroupement) l'identité
        $\sum_{k=1}^{2m} \frac{(-1)^{k-1}}{k} = H_{2m} - H_m$,
        et en déduire à la fois la somme $\ln 2$ (encore) et la
        vitesse exacte~:
        \[
        \sum_{k=1}^{2m} \frac{(-1)^{k-1}}{k}
        = \ln 2 - \frac{1}{4m} + O\Bigl(\frac{1}{m^2}\Bigr) .
        \]
  \item En déduire l'erreur asymptotique de la \omterm{def:b1:series:def}{série} harmonique
        alternée à \emph{tout} indice~: $S - S_N \sim
        \frac{(-1)^N}{2N}$ --- deux fois plus petite que la borne du
        pire cas $a_{N+1} \approx \frac1N$ du
        \cref{thm:b1:series:alternating}.
  \item (Accélération gratuite) Montrer que les sommes moyennées
        $\tilde S_N = \frac{S_N + S_{N+1}}{2}$ vérifient $\tilde
        S_N = \ln 2 + O\bigl(\frac{1}{N^2}\bigr)$. Vérification~:
        $S_{10} = 0.64563$, $S_{11} = 0.73654$, $\tilde S_{10} =
        0.69109$, contre $\ln 2 = 0.69315$~: une moyenne achète
        deux décimales.
  \item Expliquer en deux phrases pourquoi aucune astuce de ce genre
        ne peut aider un phénomène de queue divergente
        \emph{positive} comme l'encadrement de la question 2~:
        l'erreur alternée oscille (signe $(-1)^N$), donc la moyenne
        annule son terme dominant, tandis que l'erreur
        $\frac{1}{2n}$ de l'encadrement du $\gamma$ est de signe
        constant. (Moyenner $a_n$ et $b_n$, en revanche, \emph{aide}~:
        relier $\frac{a_n + b_n}{2}$ à l'estimation par le point
        milieu $H_n - \ln\bigl(n + \frac12\bigr)$ et montrer que son
        erreur est $O\bigl(\frac{1}{n^2}\bigr)$.)
\end{enumerate}

\medskip
\noindent\textbf{Partie IV --- La rigidité et sa faillite.}
\begin{enumerate}[resume]
  \item Montrer que la partie positive $\sum \frac{1}{2j-1}$ et la
        partie négative $\sum \frac{1}{2j}$ de la \omterm{def:b1:series:def}{série} harmonique
        alternée divergent toutes deux --- la signature de la
        convergence \emph{conditionnelle}.
  \item Démontrer l'énoncé général derrière la question 12~: si
        $\sum u_n$ converge mais $\sum \abs{u_n}$ diverge, alors la
        \omterm{def:b1:series:def}{série} des parties positives $\sum u_n^+$ et celle des parties
        négatives $\sum u_n^-$ divergent toutes deux \emph{(à partir
        de $u_n^\pm = \frac{\abs{u_n} \pm u_n}{2}$~: si l'une
        convergeait, l'autre aussi, donc $\sum\abs{u_n}$)}.
        Cet inépuisable réservoir de masse positive et négative est
        ce que la recette de Riemann va dépenser.
  \item (Rigidité) Démontrer~: si $\sum u_n$ converge
        \emph{absolument} et si $\sigma \colon \N \to \N$ est une
        \omterm{def:b1:logic:inj}{bijection}, alors $\sum u_{\sigma(n)}$ converge vers la même
        somme \emph{(pour $N$ grand, les $M$ premiers termes
        réarrangés contiennent $u_0, \dots, u_N$~; comparer les
        sommes partielles à travers le reste $\sum_{n >
        N}\abs{u_n}$)}.
  \item (La recette de Riemann) Soit $t \in \R$. Décrire le
        réarrangement glouton de la \omterm{def:b1:series:def}{série} harmonique alternée~:
        prendre des termes positifs $1, \frac13, \frac15, \dots$
        jusqu'à ce que la somme partielle dépasse $t$ pour la
        première fois, puis des termes négatifs
        $-\frac12, -\frac14, \dots$ jusqu'à ce qu'elle passe sous
        $t$ pour la première fois, et recommencer. Montrer que
        chaque terme est utilisé exactement une fois, qu'après le
        premier croisement les sommes partielles restent à distance
        de $t$ au plus le dernier terme utilisé, et conclure que la
        \omterm{def:b1:series:def}{série} réarrangée converge vers $t$~: \emph{toute} somme
        prescrite est atteignable.
\end{enumerate}

\medskip
\noindent\textbf{Partie V --- La formule $(p, q)$.} Fixons des
entiers $p, q \geq 1$. Réarrangeons la \omterm{def:b1:series:def}{série} harmonique alternée par
blocs~: $p$ termes positifs (les inverses impairs suivants), puis $q$
termes négatifs (les inverses pairs suivants), et on recommence.
\begin{enumerate}[resume]
  \item Vérifier qu'il s'agit bien d'un réarrangement (chaque terme
        exactement une fois), et que pour $(p, q) = (1, 2)$ il
        s'écrit
        \[
        1 - \frac12 - \frac14 + \frac13 - \frac16 - \frac18 +
        \frac15 - \frac1{10} - \frac1{12} + \dots
        \]
  \item (La division exacte par deux) Pour $(p, q) = (1, 2)$,
        démontrer l'identité de bloc
        \[
        \frac{1}{2k-1} - \frac{1}{4k-2} - \frac{1}{4k}
        = \frac12\Bigl(\frac{1}{2k-1} - \frac{1}{2k}\Bigr),
        \]
        et en déduire la relation \emph{exacte} $T_{3K} = \frac12
        S_{2K}$ entre les sommes partielles réarrangées et les
        originales~: la division de la somme par deux est visible à
        chaque étape finie, et pas seulement à la limite.
  \item Montrer que la somme partielle après $K$ blocs complets vaut
        $\sum_{j=1}^{pK} \frac{1}{2j-1} - \sum_{j=1}^{qK}
        \frac{1}{2j}$, et calculer sa limite à l'aide de la question
        7~:
        \[
        \ln 2 + \frac12 \ln\frac pq .
        \]
  \item Contrôler les sommes partielles \emph{à l'\omterm{def:b1:topology:closure}{intérieur}} d'un
        bloc (les termes tendent vers $0$) et conclure que la \omterm{def:b1:series:def}{série}
        réarrangée selon $(p, q)$ converge vers $\ln 2 + \frac12\ln
        \frac pq$. En particulier $(1, 2)$ donne $\frac{\ln
        2}{2}$~: vérifier sur les neuf premiers termes, $T_9 =
        0.3083$, qui rampe vers $0.3466$.
  \item Vérifications et portée~: $(1,1)$ redonne $\ln 2$~;
        $(2,1)$ donne $\frac32\ln 2$~; quelles sommes sont
        atteignables par des blocs $(p, q)$, et comment ce menu
        dénombrable se compare-t-il à la carte complète de Riemann
        (question 14)~?
\end{enumerate}

\medskip
\noindent\textbf{Partie VI --- Épilogue~: $\gamma$ à l'œuvre, et
synthèse.}
\begin{enumerate}[resume]
  \item Identifier la somme de la \omterm{def:b1:series:def}{série} convergente
        $\sum_{k\geq1} \bigl(\frac1k - \ln\frac{k+1}{k}\bigr)$
        (\cref{exo:b1:series:7})~: montrer qu'elle vaut $\gamma$.
  \item Faire tourner la recette de Riemann (question 14) pour la
        cible $t = 1$ et lister les douze premiers termes produits
        ($1, \frac13, -\frac12, \frac15, -\frac14, \frac17,
        \frac19, -\frac16, \frac1{11}, \frac1{13}, -\frac18,
        \frac1{15}$), en calculant la somme partielle ($\approx
        0.980$) --- regardez l'algorithme respirer autour de sa cible.
  \item Montrer qu'un certain réarrangement de la \omterm{def:b1:series:def}{série} harmonique
        alternée diverge vers $+\infty$ \emph{(des blocs de termes
        positifs assez longs pour gagner $1$ à chaque fois, en
        utilisant la question 12, séparés par des termes négatifs
        isolés)}.
  \item Affiner l'\cref{ex:b1:series:harmonicstack} avec la loi du
        $\gamma$~: montrer que le premier indice tel que $H_N \geq
        20$ vérifie $N = \eu^{\,20 - \gamma}\,(1 + o(1))
        \approx 2.7\cdot10^{8}$ --- la \omterm{ex:b1:series:oddtelescope}{constante d'Euler} est
        exactement la correction qui manquait à l'encadrement
        grossier.
  \item Synthèse, une phrase pour chaque point~: (i) la loi du
        $\gamma$ et ce qu'apporte chacun de ses trois morceaux
        ($\ln n$, $\gamma$, $\frac{1}{2n}$)~; (ii) pourquoi la
        convergence conditionnelle rend la somme dépendante de
        l'ordre alors que la \omterm{thm:b1:series:absolute}{convergence absolue} l'interdit~; (iii)
        en quoi la formule $(p,q)$ était un \emph{calcul} avec la loi
        du $\gamma$ plutôt qu'une affirmation d'existence abstraite~;
        (iv) où ces fils se poursuivent --- \omterm{def:b1:series:def}{séries} entières et
        produits de \omterm{def:b1:series:def}{séries} dans le volume de Licence 2, et le devoir
        maison du volume de Licence 3 sur la formule de Stirling, où
        la même comptabilité somme-contre-intégrale tourne à plein
        régime.
\end{enumerate}
\end{problem}
