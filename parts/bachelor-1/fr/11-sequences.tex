\chapter{Suites}\label{ch:b1:seq}

Les suites ont été manipulées dans le volume de Terminale (lycée) avec
une notion de limite admise pour moitié sur parole. Ici la théorie est
reconstruite sur la complétude de $\R$ (\cref{ch:b1:reals})~: chaque
théorème classique --- convergence monotone, \omterm{thm:b1:seq:adjacent}{suites adjacentes},
Bolzano--Weierstrass, critère de Cauchy --- est une face de ce seul
axiome. Le chapitre s'achève par l'étude pratique des suites définies
par $u_{n+1} = f(u_n)$.

\section{Convergence}

\begin{definition}[Limite d'une suite]\label{def:b1:seq:limit}
Une suite $(u_n)$ de réels \emph{converge} vers $\ell \in \R$
lorsque\index{limite!d'une suite}
\[
\forall \varepsilon > 0,\ \exists N \in \N,\ \forall n \geq N,
\qquad \abs{u_n - \ell} \leq \varepsilon .
\]
On écrit $u_n \to \ell$ ou $\lim u_n = \ell$. Une suite qui ne
converge pas (vers aucun réel) \emph{diverge}. Divergence \emph{vers
$+\infty$}~: $\forall M,\ \exists N,\ \forall n \geq N,\ u_n \geq M$
(de même pour $-\infty$).
\end{definition}

\begin{example}[Une démonstration en $\varepsilon$--$N$, rédigée une fois]\label{ex:b1:seq:epsilonN}
Affirmation~: $u_n = \dfrac{n^2 + 1}{2n^2 - 3} \to \dfrac12$.
Isolons d'abord l'erreur~:
\[
\Bigl| u_n - \frac12 \Bigr|
= \Bigl| \frac{2(n^2 + 1) - (2n^2 - 3)}{2(2n^2 - 3)} \Bigr|
= \frac{5}{2\,\abs{2n^2 - 3}}
= \frac{5}{2\,(2n^2 - 3)} \quad (n \geq 2).
\]
Puis dominons-la par quelque chose de simple~: pour $n \geq 2$, $2n^2
- 3 \geq n^2$, donc l'erreur est $\leq \frac{5}{2n^2} \leq \frac
5{2n}$. Étant donné $\varepsilon > 0$, la \omterm{thm:b1:reals:archimedes}{propriété d'Archimède}
fournit $N \geq \max\bigl(2, \frac{5}{2\varepsilon}\bigr)$~; pour $n
\geq N$ l'erreur est $\leq \varepsilon$. C'est fini. L'idée à
retenir~: une démonstration en $\varepsilon$--$N$ comporte exactement
trois gestes --- calculer l'erreur, la majorer par une expression
élémentaire décroissante, résoudre en le seuil --- et après les
théorèmes de ce chapitre (opérations, encadrement) on n'écrit
pratiquement plus jamais une telle démonstration~: les théorèmes
emballent les trois gestes une fois pour toutes.
\end{example}

\begin{example}[Divergence vers l'infini, certifiée]\label{ex:b1:seq:toinfinity}
Affirmation~: $u_n = n^2 - 100n \to +\infty$. Factorisons le terme
dominant~: $u_n = n^2\bigl(1 - \frac{100}{n}\bigr) \geq
\frac{n^2}{2}$ pour $n \geq 200$. Étant donné $M$, prenons $N =
\max\bigl(200, \lceil\sqrt{2M}\rceil\bigr)$~: pour $n \geq N$, $u_n
\geq \frac{n^2}{2} \geq M$. Deux habitudes sont ici à l'œuvre~: la
factorisation par le terme dominant transforme une compétition
($n^2$ contre $-100n$) en une seule échelle multipliée par un
facteur tendant vers $1$~; et le seuil peut être énorme ($u_{100} =
0$, la suite est même négative avant $n = 100$) --- la divergence
vers $+\infty$ est une \omterm{def:b1:logic:statement}{assertion} sur la queue, indifférente à toute
quantité finie de mauvaise conduite.
\end{example}

\begin{proposition}[Premières propriétés]\label{prop:b1:seq:first}
\begin{enumerate}
  \item La limite, si elle existe, est unique.
  \item Une suite convergente est bornée.
  \item Si $u_n \to \ell$, toute modification d'un nombre fini de
        termes laisse la convergence et la limite inchangées.
\end{enumerate}
\end{proposition}

\begin{proof}
(1) Si $u_n \to \ell$ et $u_n \to \ell'$ avec $\ell \neq \ell'$,
prenons $\varepsilon = \frac{\abs{\ell - \ell'}}{3}$~: au-delà des
deux seuils, $\abs{\ell - \ell'} \leq \abs{\ell - u_n} + \abs{u_n -
\ell'} \leq 2\varepsilon = \frac23 \abs{\ell - \ell'}$, absurde.

(2) Avec $\varepsilon = 1$~: au-delà de $N$, $\abs{u_n} \leq \abs\ell
+ 1$~; les termes antérieurs, en nombre fini, sont bornés eux aussi,
donc $\abs{u_n} \leq \max(\abs{u_0}, \dots, \abs{u_{N-1}}, \abs\ell +
1)$.

(3) En détail~: supposons $v_n = u_n$ pour $n \geq n_0$ et $u_n \to
\ell$. Étant donné $\varepsilon > 0$, prenons le seuil $N$ de
$(u_n)$~: pour $n \geq \max(N, n_0)$, $\abs{v_n - \ell} = \abs{u_n
- \ell} \leq \varepsilon$. Donc $v_n \to \ell$~: la définition ne
quantifie que sur $n \geq N$, et tout préfixe fini peut être
réécrit au prix d'un agrandissement du seuil. (C'est pourquoi les
hypothèses «~pour tout $n$ assez grand~» suffisent partout dans ce
chapitre.)
\end{proof}

\begin{theorem}[Opérations sur les limites]\label{thm:b1:seq:operations}
Si $u_n \to \ell$ et $v_n \to m$, alors
\[
u_n + v_n \to \ell + m, \qquad u_n v_n \to \ell m, \qquad
\frac{u_n}{v_n} \to \frac{\ell}{m} \ (\text{si } m \neq 0),
\qquad \abs{u_n} \to \abs\ell .
\]
\end{theorem}

\begin{proof}
\emph{Somme~:} $\abs{(u_n + v_n) - (\ell + m)} \leq \abs{u_n - \ell} +
\abs{v_n - m} \leq 2\varepsilon$ au-delà du plus grand des deux
seuils. \emph{Produit~:} écrivons
\[
u_n v_n - \ell m = (u_n - \ell)\,v_n + \ell\,(v_n - m);
\]
$(v_n)$ est bornée par un certain $B$ (\cref{prop:b1:seq:first}),
donc le membre de droite est $\leq B\abs{u_n - \ell} +
\abs{\ell}\,\abs{v_n - m}$, arbitrairement petit. \emph{Quotient~:} il
suffit de traiter $\frac 1{v_n}$. Avec $\varepsilon = \frac{\abs
m}{2}$~: au-delà d'un certain $N_0$, $\abs{v_n} \geq \frac{\abs
m}{2}$, donc
\[
\Bigl| \frac{1}{v_n} - \frac 1m \Bigr|
= \frac{\abs{m - v_n}}{\abs{v_n m}}
\leq \frac{2}{m^2}\,\abs{v_n - m} \longrightarrow 0 .
\]
\emph{Valeur absolue~:} $\bigl|\abs{u_n} - \abs\ell\bigr| \leq
\abs{u_n - \ell}$ (deuxième \omterm{prop:b1:complex:rules}{inégalité triangulaire},
\cref{prop:b1:complex:rules}).
\end{proof}

\begin{example}[Les opérations, plus une astuce algébrique]\label{ex:b1:seq:conjugatetrick}
Calculons $\lim\,\bigl(\sqrt{n^2 + n} - n\bigr)$. Les deux morceaux
tendent séparément vers $+\infty$~: le théorème des opérations ne dit
rien de leur différence (une \emph{forme indéterminée}). Multiplions
par la quantité \omterm{def:b1:complex:field}{conjuguée}~:
\[
\sqrt{n^2 + n} - n
= \frac{(n^2 + n) - n^2}{\sqrt{n^2 + n} + n}
= \frac{n}{\sqrt{n^2+n} + n}
= \frac{1}{\sqrt{1 + \frac1n} + 1} .
\]
Maintenant tout converge~: $\sqrt{1 + \frac1n} \to 1$, car
$0 \leq \sqrt{1 + h} - 1 = \frac{h}{\sqrt{1+h} + 1} \leq h$
(\omterm{def:b1:complex:field}{conjuguée} encore, puis encadrement avec $h = \frac1n$)~; le
théorème des opérations donne alors la limite $\frac{1}{1 + 1} =
\frac12$. L'idée à retenir~: le théorème des opérations n'est pas une
calculatrice universelle des limites --- les formes indéterminées
($\infty - \infty$, $\frac00$, $0 \times \infty$, $1^\infty$) doivent
d'abord être \emph{transformées} par l'algèbre (quantités conjuguées,
factorisation du terme dominant) jusqu'à ce que chaque morceau
converge~; la machine systématique pour les cas résistants est le
développement limité du \cref{ch:b1:taylor}.
\end{example}

\begin{theorem}[Limites et ordre]\label{thm:b1:seq:order}
\begin{enumerate}
  \item Si $u_n \leq v_n$ pour tout $n$ assez grand, et si les deux
        convergent, alors $\lim u_n \leq \lim v_n$. (Les inégalités
        strictes ne passent \emph{pas} à la limite~: $\frac 1n > 0$
        mais $\lim = 0$.)
  \item (\omterm{thm:b1:seq:order}{Théorème d'encadrement}\index{théorème d'encadrement}) Si
        $u_n \leq w_n \leq v_n$ pour tout $n$ assez grand et $u_n,
        v_n \to \ell$, alors $w_n \to \ell$.
  \item Si $u_n \to \ell > 0$, alors $u_n > \frac\ell2 > 0$ pour tout
        $n$ assez grand.
\end{enumerate}
\end{theorem}

\begin{proof}
(1) Supposons $\ell = \lim u_n > m = \lim v_n$~; avec $\varepsilon =
\frac{\ell - m}{3}$, les termes de rang assez grand vérifient $v_n
\leq m + \varepsilon < \ell - \varepsilon \leq u_n$, ce qui contredit
$u_n \leq v_n$.

(2) Au-delà des seuils~: $\ell - \varepsilon \leq u_n \leq w_n \leq
v_n \leq \ell + \varepsilon$.

(3) c'est la \cref{def:b1:seq:limit} avec $\varepsilon = \frac\ell2$.
\end{proof}

\begin{example}[Deux encadrements]\label{ex:b1:seq:squeezes}
(i) $\dfrac{\sin n}{n} \to 0$~: à partir de $-\frac1n \leq
\frac{\sin n}{n} \leq \frac1n$, les deux murs s'effondrant sur $0$
--- inutile de comprendre le numérateur erratique. (ii) $(2^n +
3^n)^{1/n} \to 3$~: encadrons l'intérieur,
\[
3^n \leq 2^n + 3^n \leq 2\cdot3^n
\quad\Longrightarrow\quad
3 \leq (2^n + 3^n)^{1/n} \leq 3\cdot 2^{1/n} ,
\]
et $2^{1/n} = \eu^{\frac{\ln 2}{n}} \to 1$ (comme pour $5^{1/n}$ dans
l'\cref{exo:b1:seq:2})~: l'encadrement livre $3$. L'idée à
retenir~: une somme d'exponentielles en compétition se comporte comme
son \emph{plus grand} terme --- les autres sont absorbées par un
facteur constant inoffensif, que la racine $n$-ième efface ensuite.
\end{example}

\section{Suites monotones}

\begin{theorem}[Théorème de la limite monotone]\label{thm:b1:seq:monotone}
Une suite croissante et majorée converge, vers $\sup\{u_n : n \in
\N\}$~; une suite croissante non majorée diverge vers $+\infty$.
(Énoncés miroirs pour les suites
décroissantes.)\index{théorème de la limite monotone}
\end{theorem}

\begin{proof}
Posons $s = \sup\{u_n\}$ (\cref{thm:b1:reals:sup}). Étant donné
$\varepsilon > 0$, la caractérisation en $\varepsilon$
(\cref{prop:b1:reals:epsilon}) fournit $N$ avec $u_N > s -
\varepsilon$~; par monotonie, $s - \varepsilon < u_N \leq u_n \leq
s$ pour tout $n \geq N$~: convergence vers $s$. Si la suite n'est pas
majorée~: pour tout $M$ il existe $u_N > M$, et la monotonie maintient
tous les termes suivants au-dessus de $M$.
\end{proof}

\begin{example}[Le théorème monotone comme machine à existence]\label{ex:b1:seq:existencemachine}
Soit $u_n = \prod_{k=1}^{n} \bigl(1 + \frac{1}{2^k}\bigr)$. Chaque
facteur dépasse $1$, donc $(u_n)$ est croissante. Majorée~? Prenons
les logarithmes et utilisons $\ln(1 + x) \leq x$
(l'\cref{ex:b1:derivative:convexineq} l'anticipe~; ou bien
l'inégalité grossière $1 + x \leq \eu^x$ du volume de Terminale)~:
\[
\ln u_n = \sum_{k=1}^{n} \ln\Bigl(1 + \frac{1}{2^k}\Bigr)
\leq \sum_{k=1}^{n} \frac{1}{2^k} < 1 ,
\]
donc $u_n < \eu$. Croissante et bornée~: $(u_n)$ converge vers un
certain $\ell \in \intoc{u_1}{\eu}$ --- un nombre réel parfaitement
défini, sans la moindre forme close en vue ($\ell = 2.384\dots$).
L'idée à retenir~: le \omterm{thm:b1:seq:monotone}{théorème de la limite monotone} est la machine à
existence la moins chère de l'analyse~; c'est lui qui a nommé $\eu$
lui-même (l'\cref{ex:b1:seq:e} ci-dessous), et au
\cref{ch:b1:series} il décidera de la convergence de toute série à
termes positifs par simple caractère borné.
\end{example}

\begin{theorem}[Suites adjacentes]\label{thm:b1:seq:adjacent}
Soient $(a_n)$ croissante, $(b_n)$ décroissante, avec $b_n - a_n \to
0$. Alors les deux convergent, vers une limite \emph{commune} $\ell$,
et $a_n \leq \ell \leq b_n$ pour tout $n$.\index{suites adjacentes}
\end{theorem}

\begin{proof}
D'abord, $a_n \leq b_n$ pour tout $n$~: la suite $(b_n - a_n)$ est
décroissante et tend vers $0$, donc elle est $\geq 0$ (un terme
négatif la figerait sous $0$). Ensuite $(a_n)$ est croissante et
majorée par $b_0$~: elle converge vers un certain $\ell$ (le
\cref{thm:b1:seq:monotone})~; de même $(b_n) \to \ell'$~; et $\ell' -
\ell = \lim (b_n - a_n) = 0$. Les inégalités $a_n \leq \ell \leq b_n$
découlent de la monotonie ($\ell = \sup a_k \geq a_n$, etc.).
\end{proof}

\begin{example}[Le nombre $\eu$]\label{ex:b1:seq:e}
Posons $a_n = \sum_{k=0}^{n} \frac{1}{k!}$ et $b_n = a_n + \frac{1}{n
\cdot n!}$ ($n \geq 1$). Alors $(a_n)$ croît~; et
\[
b_{n+1} - b_n = \frac{1}{(n+1)!} + \frac{1}{(n+1)(n+1)!} -
\frac{1}{n\,n!}
= \frac{n(n+1) + n - (n+1)^2}{n(n+1)(n+1)!}
= \frac{-1}{n(n+1)(n+1)!} < 0 ,
\]
donc $(b_n)$ décroît, et $b_n - a_n \to 0$~: elles sont adjacentes.
Leur limite commune est (par définition ici) le nombre $\eu \approx
2.71828$~; les inégalités $a_n < \eu < b_n$ sont assez fines pour
prouver que $\eu \notin \Q$ (l'\cref{exo:b1:seq:9}).
\end{example}

\section{Suites extraites et théorème de Bolzano--Weierstrass}

\begin{definition}[Suite extraite]\label{def:b1:seq:subsequence}
Une \emph{suite extraite}\index{suite extraite} de $(u_n)$ est une
suite $(u_{\varphi(n)})$ où $\varphi \colon \N \to \N$ est
strictement croissante (noter que $\varphi(n) \geq n$, par
récurrence).
\end{definition}

\begin{proposition}\label{prop:b1:seq:subsequences}
Si $u_n \to \ell$ ($\ell \in \R$ ou $\pm\infty$), toute \omterm{def:b1:seq:subsequence}{suite extraite}
tend vers $\ell$. Par conséquent, une suite possédant deux \omterm{def:b1:seq:subsequence}{suites
extraites} de limites différentes diverge. Réciproquement, si
$(u_{2n})$ et $(u_{2n+1})$ convergent toutes deux vers le
\emph{même} $\ell$, alors $u_n \to \ell$.
\end{proposition}

\begin{proof}
Au-delà du seuil $N$ de $(u_n)$, tous les indices $\varphi(n) \geq n
\geq N$ conviennent (l'inégalité $\varphi(n) \geq n$ est la
récurrence signalée dans la \cref{def:b1:seq:subsequence}~:
$\varphi(0) \geq 0$, et $\varphi(n+1) > \varphi(n) \geq n$ force
$\varphi(n+1) \geq n + 1$). Pour la réciproque~: étant donné
$\varepsilon$, prenons les deux seuils $N_0$ (pairs) et $N_1$
(impairs)~; un indice quelconque $n \geq \max(2N_0, 2N_1 + 1)$ est
soit pair, $n = 2k$ avec $k \geq N_0$, soit impair, $n = 2k+1$ avec
$k \geq N_1$ --- dans les deux cas $\abs{u_n - \ell} \leq
\varepsilon$~: tout indice est couvert par l'une des deux \omterm{def:b1:seq:subsequence}{suites
extraites}, et c'est là tout le propos.
\end{proof}

\begin{example}[Valeurs d'adhérence]\label{ex:b1:seq:sublimits}
Pour $u_n = (-1)^n \frac{n}{n+1}$~: la \omterm{def:b1:seq:subsequence}{suite extraite} des indices
pairs tend vers $1$, celle des indices impairs vers $-1$, donc la
suite diverge --- mais elle le fait de façon organisée, en
s'agglutinant autour des deux valeurs $\pm 1$. Pour $u_n =
\cos\frac{2\pi n}{3}$~: les trois \omterm{def:b1:seq:subsequence}{suites extraites} d'indices $3k$, $3k
+ 1$, $3k + 2$ sont constantes, égales à $1$, $-\frac12$,
$-\frac12$~; l'\omterm{def:b1:logic:sets}{ensemble} des valeurs d'adhérence est $\{1,
-\frac12\}$. L'idée à retenir~: une suite \emph{bornée} converge
exactement lorsqu'elle a une unique valeur d'adhérence
(l'\cref{exo:b1:seq:8})~; la divergence d'une suite bornée signifie
toujours au moins deux paquets, et Bolzano--Weierstrass ci-dessous
garantit qu'il y en a au moins un.
\end{example}

\begin{theorem}[Bolzano--Weierstrass]\label{thm:b1:seq:bw}
De toute suite bornée de réels on peut extraire une suite
convergente.\index{théorème de Bolzano--Weierstrass}
\end{theorem}

\begin{proof}
Soit $u_n \in \intcc{a}{b}$ pour tout $n$. Construisons des segments
emboîtés par dichotomie~: posons $\intcc{a_0}{b_0} = \intcc{a}{b}$~;
étant donné $\intcc{a_k}{b_k}$ contenant $u_n$ pour une infinité de
$n$, l'une de ses deux moitiés contient encore $u_n$ pour une infinité
de $n$ --- appelons-la $\intcc{a_{k+1}}{b_{k+1}}$. Les suites $(a_k)$,
$(b_k)$ sont adjacentes ($b_k - a_k = \frac{b-a}{2^k} \to 0$), de
limite commune $\ell$ (le \cref{thm:b1:seq:adjacent}).

Extrayons~: choisissons $\varphi(0)$ avec $u_{\varphi(0)} \in
\intcc{a_0}{b_0}$, puis, par récurrence, $\varphi(k+1) > \varphi(k)$
avec $u_{\varphi(k+1)} \in \intcc{a_{k+1}}{b_{k+1}}$ --- c'est
possible puisque ce segment contient une infinité de termes. Alors
$a_k \leq u_{\varphi(k)} \leq b_k$, et le \omterm{thm:b1:seq:order}{théorème d'encadrement} donne
$u_{\varphi(k)} \to \ell$.
\end{proof}

\begin{remark}[Ce que Bolzano--Weierstrass dit, et ne dit pas]
Il \emph{dit}~: du seul caractère borné, une \omterm{def:b1:seq:subsequence}{suite extraite} converge
--- une existence sans formule, comme la démonstration par dichotomie
le montre bien (rien ne nous dit \emph{quels} indices survivent). Il
ne dit \emph{pas} que la limite est unique~: $((-1)^n)$ a des \omterm{def:b1:seq:subsequence}{suites
extraites} convergeant vers $1$ et vers $-1$, et l'\omterm{def:b1:logic:sets}{ensemble} des valeurs
d'adhérence peut même être infini (l'\cref{ex:b1:seq:sublimits}, et
tout l'\omterm{def:b1:logic:sets}{ensemble} de Cantor dans le \cref{pb:b1:topology:1}). Il ne
survit pas à l'absence de bornes~: $(n)$ n'a aucune \omterm{def:b1:seq:subsequence}{suite extraite}
convergente --- même si l'on peut toujours extraire d'une suite non
bornée une suite tendant vers $+\infty$ ou vers $-\infty$ (choisir
$\varphi(k)$ avec $u_{\varphi(k)} \geq k$, par exemple). Bien
utilisé, le théorème est une \emph{pompe à existence}~: il apparaît
au cœur du critère de Cauchy ci-dessous, du théorème de Heine et du
théorème des bornes atteintes --- toujours pour produire un point
qu'aucune construction explicite n'offre.
\end{remark}

\section{Suites de Cauchy et complétude}

\begin{definition}\label{def:b1:seq:cauchy}
Une suite $(u_n)$ est une \emph{suite de Cauchy}\index{suite de
Cauchy} lorsque ses termes deviennent arbitrairement proches
\emph{les uns des autres}~:
\[
\forall \varepsilon > 0,\ \exists N,\ \forall p, q \geq N,
\qquad \abs{u_p - u_q} \leq \varepsilon .
\]
\end{definition}

\begin{example}[Vérifier la propriété de Cauchy à la main]\label{ex:b1:seq:cauchyhand}
Soit $u_n = \sum_{k=0}^{n} \frac{\cos k}{2^k}$ --- aucune monotonie,
aucune limite devinable. Pour $p > q$~:
\[
\abs{u_p - u_q}
= \Bigl| \sum_{k=q+1}^{p} \frac{\cos k}{2^k} \Bigr|
\leq \sum_{k=q+1}^{p} \frac{1}{2^k}
< \frac{1}{2^{q}} ,
\]
par l'\omterm{prop:b1:complex:rules}{inégalité triangulaire}, $\abs{\cos k} \leq 1$ et une somme
géométrique finie. Étant donné $\varepsilon > 0$, choisissons $N$ avec
$2^{-N} \leq \varepsilon$~: tous les écarts au-delà de $N$ sont $\leq
\varepsilon$, la suite est de Cauchy, donc converge --- vers une
limite que personne ne sait nommer sous forme close, et c'est
précisément le propos. L'idée à retenir~: la domination géométrique
des accroissements est la façon standard de gagner la propriété de
Cauchy, et le \cref{ch:b1:series} mettra l'argument en bouteille sous
le nom «~la convergence absolue implique la convergence~».
\end{example}

\begin{theorem}[Complétude de $\R$]\label{thm:b1:seq:complete}
Une suite de réels converge si et seulement si c'est une \omterm{def:b1:seq:cauchy}{suite de
Cauchy}.\index{completude de R@complétude de $\R$}
\end{theorem}

\begin{proof}
($\Rightarrow$) Si $u_n \to \ell$~: au-delà du seuil associé à
$\frac\varepsilon2$, $\abs{u_p - u_q} \leq \abs{u_p - \ell} +
\abs{\ell - u_q} \leq \varepsilon$.

($\Leftarrow$) Soit $(u_n)$ de Cauchy. \emph{Elle est bornée}~: avec
$\varepsilon = 1$, au-delà de $N$ tous les termes sont à distance
$\leq 1$ de $u_N$, et la tête est finie. \emph{Extrayons}~: par le
\cref{thm:b1:seq:bw}, une \omterm{def:b1:seq:subsequence}{suite extraite} $u_{\varphi(n)} \to \ell$.
\emph{Concluons}~: étant donné $\varepsilon > 0$, prenons $N$ (de
Cauchy, pour $\frac\varepsilon2$) et $n \geq N$ avec
$\abs{u_{\varphi(n)} - \ell} \leq \frac\varepsilon2$ et $\varphi(n)
\geq N$~; alors pour tout $p \geq N$~:
\[
\abs{u_p - \ell} \leq \abs{u_p - u_{\varphi(n)}} +
\abs{u_{\varphi(n)} - \ell} \leq \varepsilon . \qedhere
\]
\end{proof}

\begin{remark}
La valeur du critère~: il certifie la convergence \emph{sans nommer
la limite}. Il tombe en défaut sur $\Q$ (les troncatures décimales de
$\sqrt 2$ forment une \omterm{def:b1:seq:cauchy}{suite de Cauchy} de rationnels sans limite
rationnelle)~: la complétude est une propriété de $\R$, équivalente à
l'axiome de la \omterm{def:b1:reals:bounds}{borne supérieure}. C'est aussi le cheval de trait qui
tire la convergence des séries (\cref{ch:b1:series}).
\end{remark}

\begin{example}[Une suite de Cauchy à limite invisible]\label{ex:b1:seq:basel}
Soit $S_n = \sum_{k=1}^{n} \frac{1}{k^2}$. Pour $p > q \geq 1$~:
\[
S_p - S_q = \sum_{k=q+1}^{p} \frac{1}{k^2}
\leq \sum_{k=q+1}^{p} \frac{1}{k(k-1)}
= \sum_{k=q+1}^{p} \Bigl(\frac{1}{k-1} - \frac 1k\Bigr)
= \frac 1q - \frac 1p < \frac 1q ,
\]
donc au-delà de $N > \frac1\varepsilon$ tous les écarts sont $\leq
\varepsilon$~: $(S_n)$ est de Cauchy, donc converge. Remarquons ce
qui vient de se passer~: nous avons prouvé qu'un nombre réel précis
existe sans en avoir le moindre nom. (C'est $\frac{\pi^2}{6}$ --- une
célèbre identité d'Euler, démontrée dans le volume de Licence 2~;
rien dans ce chapitre ne pouvait nous l'apprendre.) Cette division du
travail --- l'existence maintenant, l'identification plus tard, si
jamais --- est tout le propos du critère de Cauchy, et le moteur de
la théorie des séries au \cref{ch:b1:series}.
\end{example}

\section{Suites récurrentes}

\begin{method}[Étudier $u_{n+1} = f(u_n)$]\label{met:b1:seq:recurrent}
Étant donnés $f$ et un point de départ $u_0$~:\index{suite récurrente}
\begin{enumerate}
  \item \emph{\omterm{prop:b1:reals:intervals}{Intervalle} stable~:} trouver un \omterm{prop:b1:reals:intervals}{intervalle} $I$ avec
        $f(I) \subseteq I$ contenant $u_0$~: alors tous les $u_n \in
        I$ (par récurrence).
  \item \emph{Limites candidates~:} si $u_n \to \ell \in I$ et si $f$
        est continue en $\ell$ (\cref{ch:b1:continuity}), alors $\ell$
        est un \emph{point fixe}~: $f(\ell) = \ell$. Résoudre $f(x) =
        x$.
  \item \emph{Monotonie~:} si $f$ est croissante sur $I$, alors
        $(u_n)$ est monotone (croissante si $u_1 \geq u_0$,
        décroissante sinon)~; combinée au caractère borné, le
        \cref{thm:b1:seq:monotone} conclut. Si $f$ est décroissante,
        étudier les deux \omterm{def:b1:seq:subsequence}{suites extraites} $(u_{2n})$ et $(u_{2n+1})$,
        qui sont monotones pour $f \circ f$.
  \item \emph{Contrôle de l'erreur~:} une inégalité $\abs{f(x) - \ell}
        \leq k\abs{x - \ell}$ avec $k < 1$ donne directement
        $\abs{u_n - \ell} \leq k^n \abs{u_0 - \ell} \to 0$.
\end{enumerate}
\end{method}

\begin{example}[Méthode de Héron]\label{ex:b1:seq:heron}
Soient $u_0 = 2$ et $u_{n+1} = \dfrac12\Bigl(u_n +
\dfrac{2}{u_n}\Bigr)$~: l'antique algorithme de calcul de
$\sqrt 2$.\index{méthode de Héron}
\begin{itemize}
  \item \emph{Stabilité~:} pour $x > 0$, l'inégalité
        arithmético-géométrique donne $\frac12(x + \frac2x) \geq
        \sqrt{x \cdot \frac 2x} = \sqrt 2$~; donc $I =
        \intco{\sqrt 2}{+\infty}$ est stable et contient $u_1$ (en
        effet $u_1 = \frac32 \geq \sqrt 2$).
  \item \emph{Monotonie~:} pour $x \geq \sqrt 2$, $\;x - f(x) =
        \frac{x^2 - 2}{2x} \geq 0$~: la suite décroît à partir de
        $u_1$, et elle est minorée par $\sqrt 2$~: elle converge.
  \item \emph{Limite~:} les points fixes vérifient $x = \frac12(x +
        \frac2x)$, c'est-à-dire $x^2 = 2$~: sur $I$, $\ell = \sqrt 2$.
  \item \emph{Vitesse~:} $u_{n+1} - \sqrt 2 = \frac{(u_n -
        \sqrt2)^2}{2u_n} \leq \frac{(u_n - \sqrt 2)^2}{2\sqrt 2}$~:
        le nombre de décimales exactes \emph{double} à peu près à
        chaque étape (convergence quadratique).
\end{itemize}
\end{example}

\begin{remark}[Pièges classiques autour des limites]
Quatre grands classiques. (i) \emph{Des petits pas n'impliquent pas
la convergence}~: $u_{n+1} - u_n \to 0$ est bien plus faible que la
propriété de Cauchy --- les sommes harmoniques $H_n$ ont des pas
$\frac{1}{n+1} \to 0$ et divergent pourtant vers $+\infty$
(l'\cref{exo:b1:seq:5})~; la condition de Cauchy contrôle $\abs{u_p -
u_q}$ pour \emph{tous} les couples de rang assez grand, pas
seulement les couples consécutifs. (ii) \emph{Les inégalités strictes
meurent à la limite}~: de $u_n < v_n$ pour tout $n$ on ne tire que
$\lim u_n \leq \lim v_n$ (le \cref{thm:b1:seq:order})~; $\frac1n > 0$
et pourtant $\lim = 0$. (iii) \emph{Bornée n'est pas convergente}~:
$((-1)^n)$ est bornée et diverge~; le caractère borné plus la
\emph{monotonie} donne la convergence, le caractère borné seul ne
garantit qu'une \omterm{def:b1:seq:subsequence}{suite extraite} convergente (le
\cref{thm:b1:seq:bw}). (iv) \emph{L'équation du point fixe vient en
second, pas en premier}~: pour $u_{n+1} = f(u_n)$, résoudre $f(\ell)
= \ell$ identifie la limite \emph{seulement après} que la convergence
a été prouvée. La récurrence $u_{n+1} = 2u_n$ a l'unique point fixe
$\ell = 0$, et pourtant, partie de $u_0 = 1$, la suite file vers
$+\infty$~: l'équation $\ell = 2\ell$ n'avait jamais droit à une
limite. Ordre des opérations, toujours~: l'existence d'abord (la
\cref{met:b1:seq:recurrent}, étapes 1 à 3), l'identification ensuite.
\end{remark}

\begin{example}[Un $f$ décroissant~: la récurrence dorée]\label{ex:b1:seq:golden}
Soient $u_0 = 1$ et $u_{n+1} = \dfrac{1}{1 + u_n}$. Ici $f(x) =
\frac{1}{1+x}$ est \emph{décroissante}, donc la suite n'est pas
monotone (elle alterne autour de sa limite)~; l'étape de contraction
de la \cref{met:b1:seq:recurrent} est le bon outil. Stabilité~: si
$x \in \intcc{\frac12}{1}$ alors $1 + x \in
\intcc{\frac32}{2}$, donc $f(x) \in \intcc{\frac12}{\frac23}
\subseteq \intcc{\frac12}{1}$, et $u_1 = \frac12$ y place toute la
suite. Point fixe~: $\ell = \frac{1}{1+\ell}$ avec $\ell > 0$ donne
$\ell^2 + \ell - 1 = 0$, c'est-à-dire
\[
\ell = \frac{\sqrt5 - 1}{2} = 0.6180\dots
\]
(l'inverse du nombre d'or). Contraction~: pour $x, y \in
\intcc{\frac12}{1}$,
\[
\abs{f(x) - f(y)} = \frac{\abs{x - y}}{(1+x)(1+y)}
\leq \frac{\abs{x-y}}{(3/2)^2} = \frac49\,\abs{x - y} ,
\]
donc $\abs{u_n - \ell} \leq \bigl(\frac49\bigr)^{n-1}\abs{u_1 -
\ell} \to 0$~: convergence, à vitesse géométrique, sans aucune
monotonie. L'idée à retenir~: les méthodes monotones et les méthodes
de contraction se partagent le monde des \omterm{met:b1:seq:recurrent}{suites récurrentes} --- un $f$
croissant donne des orbites monotones, un $f$ décroissant des orbites
alternées apprivoisées par une constante de Lipschitz $< 1$ (la
théorie systématique est l'\cref{exo:b1:derivative:11}).
\end{example}

\begin{remark}[Perspectives à l'intérieur de ce volume]
Les suites sont l'instrument de mesure que tout le reste du volume
approche de chaque objet. Au \cref{ch:b1:topology} elles
\emph{caractérisent} les fermés et les compacts~; au
\cref{ch:b1:continuity} elles transportent les limites de fonctions~;
au \cref{ch:b1:integration} les sommes de Riemann sont des suites
convergeant vers l'intégrale~; le \cref{ch:b1:series} \emph{est} la
théorie d'une classe particulière de suites, les sommes partielles.
Même les chapitres d'algèbre les consomment~: les itérées d'une
matrice au \cref{ch:b1:matrices} forment des suites dont le
comportement (convergence de $A^n$) est une question d'algèbre
linéaire posée avec le vocabulaire de ce chapitre. Les deux théorèmes
à emporter partout~: la limite monotone (l'existence à partir de
l'ordre) et Bolzano--Weierstrass (l'existence à partir du caractère
borné) --- entre eux deux, presque toute limite de ce livre naît.
\end{remark}

\begin{remark}[Suites complexes]
Une suite $(z_n)$ de nombres complexes converge vers $\ell$ lorsque
$\abs{z_n - \ell} \to 0$~; de façon équivalente, lorsque $\Re(z_n) \to
\Re(\ell)$ et $\Im(z_n) \to \Im(\ell)$ (comparer $\abs{z}$ à
$\abs{\Re z} + \abs{\Im z}$). Les théorèmes qui ne font pas intervenir
l'ordre --- opérations, Bolzano--Weierstrass (extraire deux fois),
critère de Cauchy --- se transposent mot pour mot.
\end{remark}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:b1:seq:1}
Directement à partir de la \cref{def:b1:seq:limit}, montrer que
$\dfrac{2n+1}{n+3} \to 2$, et que $(u_n) = ((-1)^n)$ diverge.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:seq:2}
Calculer les limites~:
\[
\frac{n^2 - 3n + 1}{2n^2 + 5},
\qquad
\sqrt{n+1} - \sqrt n,
\qquad
\frac{2^n + n^3}{3^n - n^2},
\qquad
\sqrt[n]{5}\ \Bigl(= 5^{1/n}\Bigr).
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:seq:3}
Démontrer la comparaison standard~: si $\abs{q} < 1$ alors $q^n \to 0$
\emph{(écrire $\frac{1}{\abs q} = 1 + h$, $h > 0$, et utiliser
l'inégalité de Bernoulli $(1+h)^n \geq 1 + nh$, à démontrer par
récurrence)}. Quels sont les comportements pour $q = 1$, $q = -1$,
$\abs q > 1$~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:seq:4}
Soit $u_{n+1} = \frac{u_n + 3}{2}$, $u_0 = 0$. Trouver le point fixe
$\ell$, montrer que $v_n = u_n - \ell$ est géométrique, et donner une
formule explicite ainsi que la limite de $(u_n)$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:seq:5}
(Série harmonique) Soit $H_n = \sum_{k=1}^{n} \frac 1k$. Montrer que
$H_{2n} - H_n \geq \frac12$ pour tout $n \geq 1$, et en déduire que
$(H_n)$ n'est \emph{pas} une \omterm{def:b1:seq:cauchy}{suite de Cauchy}, donc diverge (vers
$+\infty$, puisqu'elle est croissante).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:seq:6}
On suppose que $(u_{2n})$, $(u_{2n+1})$ et $(u_{3n})$ convergent
toutes trois. Montrer que $(u_n)$ converge. \emph{(Trouver des \omterm{def:b1:seq:subsequence}{suites
extraites} communes pour égaler les limites.)}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:seq:7}
Étudier la suite $u_0 = 0$, $u_{n+1} = \sqrt{2 + u_n}$~: stabilité,
monotonie, limite. Démontrer ensuite la majoration de l'erreur
$\abs{u_n - 2} \leq \dfrac{2}{3^{\,n}}$ \emph{(montrer que $2 -
u_{n+1} = \dfrac{2 - u_n}{2 + \sqrt{2 + u_n}}$ et minorer le
dénominateur par $3$)}.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:seq:8}
Soit $(u_n)$ bornée, telle que toute \omterm{def:b1:seq:subsequence}{suite extraite} convergente de
$(u_n)$ ait la \emph{même} limite $\ell$. Montrer que $u_n \to \ell$.
\emph{(Raisonnement par l'absurde plus Bolzano--Weierstrass.)}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:seq:9}
Avec les notations de l'\cref{ex:b1:seq:e}, supposons $\eu = \frac pq$
avec $p, q \in \N^*$. En utilisant $a_q < \eu < b_q = a_q + \frac{1}{q\,
q!}$, multiplier par $q!$ et en tirer une contradiction entre deux
entiers. Conclure~: $\eu$ est irrationnel.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:seq:10}
(Moyennes de Cesàro) Pour une suite $(u_n)_{n \geq 1}$, posons $c_n =
\frac{u_1 + \dots + u_n}{n}$.
\begin{enumerate}
  \item Montrer que $u_n \to \ell$ entraîne $c_n \to \ell$
        \emph{(couper la somme à un seuil $N$~; majorer la tête par
        une quantité fixe divisée par $n$, la queue par
        $\varepsilon$)}.
  \item Montrer par un exemple que la réciproque est fausse.
  \item En déduire que si $u_{n+1} - u_n \to \ell$, alors
        $\frac{u_n}{n} \to \ell$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:seq:11}
Soit $(u_n)$ vérifiant $0 \leq u_{m+n} \leq u_m + u_n$ pour tous $m,
n$ (sous-additivité). Montrer que $\bigl(\frac{u_n}{n}\bigr)$ converge
vers $\inf_{n \geq 1} \frac{u_n}{n}$. \emph{(Pour $m$ fixé, écrire $n
= qm + r$ et majorer $\frac{u_n}{n}$ à l'aide de $u_n \leq q\,u_m +
u_r$.)}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:seq:12}
En utilisant la densité du \omterm{def:b1:structures:subgroup}{sous-groupe} $\Z + 2\pi\Z$ de $(\R, +)$
(l'\cref{exo:b1:reals:9}), montrer que la suite $(\sin n)_{n \in
\N}$ est dense dans $\intcc{-1}{1}$ --- en particulier elle diverge.
\end{exercise}

\section{Problème~: Cesàro, Stolz et la lente décroissance du sinus}

\begin{problem}[{Devoir maison --- le \omterm{pb:b1:seq:1}{théorème de Cesàro--Stolz}
et l'asymptotique $u_n \sim \sqrt{3/n}$ pour $u_{n+1} = \sin
u_n$}]
\label{pb:b1:seq:1}
Le \omterm{pb:b1:seq:1}{théorème de Cesàro--Stolz} est la règle de l'Hospital discrète~:
pour trouver la limite d'un quotient $a_n/b_n$, il suffit de trouver
la limite du quotient des \emph{accroissements} $(a_{n+1} -
a_n)/(b_{n+1} - b_n)$. Ce problème démontre le théorème, en récolte
des limites classiques, puis le braque sur une cible célèbre~: la
suite $u_{n+1} = \sin u_n$, qui rampe vers $0$ à la vitesse
exactement calculable $u_n \sim \sqrt{3/n}$.\index{theoreme de Cesaro--Stolz@théorème de Cesàro--Stolz}
Deux faits du volume de Terminale sont admis ici et honnêtement
redémontrés plus loin dans ce volume~: l'inégalité de la tangente
\[
\tag{G1} \eu^{u} \geq 1 + u \quad (u \in \R),
\]
redémontrée par convexité au \cref{ch:b1:derivative}, et
l'encadrement du sinus
\[
\tag{G2} x - \frac{x^3}{6} \;\leq\; \sin x \;\leq\; x -
\frac{x^3}{6} + \frac{x^5}{120} \quad (0 \leq x \leq 1), \qquad
\abs{\sin x} \leq \abs{x} \quad (x \in \R),
\]
redémontré par la formule de Taylor au \cref{ch:b1:taylor}.

\medskip
\noindent\textbf{Partie I --- Des sommes sans formule close.}
\begin{enumerate}
  \item En utilisant $1 + 2 + \dots + n = \frac{n(n+1)}{2}$ et $1^2 +
        \dots + n^2 = \frac{n(n+1)(2n+1)}{6}$, calculer
        $\lim \frac{1 + 2 + \dots + n}{n^2}$ et $\lim \frac{1^2 +
        \dots + n^2}{n^3}$.
  \item Soit $T_n = \sum_{k=1}^n \sqrt k$, pour laquelle il n'existe
        aucune formule close. Démontrer l'encadrement
        \[
        \frac{1}{2\sqrt 2}\,n^{3/2} \;\leq\; T_n \;\leq\;
        n^{3/2}
        \]
        \emph{(ne garder que les termes $k > \frac n2$ pour la
        minoration)}. Ainsi $T_n$ est de l'ordre de $n^{3/2}$ ---
        mais avec quelle constante~? Garder la question pour la
        question 8.
  \item (Lemme télescopique) Soit $(b_n)$ strictement croissante et
        supposons que, pour tout $k \geq N$,
        \[
        m \;\leq\; \frac{a_{k+1} - a_k}{b_{k+1} - b_k} \;\leq\;
        M .
        \]
        Montrer que $m \leq \dfrac{a_n - a_N}{b_n - b_N} \leq M$
        pour tout $n > N$.
\end{enumerate}

\medskip
\noindent\textbf{Partie II --- Le \omterm{pb:b1:seq:1}{théorème de Cesàro--Stolz}.} Soit
$(b_n)$ strictement croissante avec $b_n \to +\infty$, et supposons
$\dfrac{a_{n+1} - a_n}{b_{n+1} - b_n} \to \ell \in \R$.
\begin{enumerate}[resume]
  \item Fixons $\varepsilon > 0$. Montrer qu'il existe $N$ tel que
        $\ell - \varepsilon \leq \dfrac{a_n - a_N}{b_n - b_N}
        \leq \ell + \varepsilon$ pour tout $n > N$.
  \item Établir, pour $n > N$, l'identité
        \[
        \frac{a_n}{b_n} - \ell
        = \frac{a_N - \ell\,b_N}{b_n}
        + \Bigl(1 - \frac{b_N}{b_n}\Bigr)
        \Bigl(\frac{a_n - a_N}{b_n - b_N} - \ell\Bigr),
        \]
        et en conclure le théorème~: $\dfrac{a_n}{b_n} \to \ell$.
  \item Démontrer la variante $+\infty$~: si $\dfrac{a_{n+1} -
        a_n}{b_{n+1} - b_n} \to +\infty$ (mêmes hypothèses sur
        $(b_n)$), alors $\dfrac{a_n}{b_n} \to +\infty$.
  \item Prendre $b_n = n$~: retrouver le théorème de la moyenne de
        Cesàro de l'\cref{exo:b1:seq:10}. Montrer ensuite que la
        réciproque de Cesàro--Stolz est fausse~: pour $a_n = (-1)^n$,
        $b_n = n$, le quotient $a_n/b_n$ converge alors que le
        quotient des accroissements n'a pas de limite. Stolz est une
        rue à sens unique.
\end{enumerate}

\medskip
\noindent\textbf{Partie III --- Premiers dividendes.}
\begin{enumerate}[resume]
  \item Démontrer $(1+h)^{3/2} - 1 = \dfrac{3h + 3h^2 +
        h^3}{(1+h)^{3/2} + 1}$ par conjugaison, en déduire
        $n\bigl((1 + \tfrac1n)^{3/2} - 1\bigr) \to \tfrac32$,
        et conclure par Cesàro--Stolz~:
        \[
        T_n = \sum_{k=1}^{n} \sqrt k \;\sim\; \tfrac23\,
        n^{3/2} ,
        \]
        ce qui résout le suspense de la question 2.
  \item À partir de (G1) seule, obtenir l'encadrement du logarithme
        \[
        \frac{t}{1 + t} \;\leq\; \ln(1 + t) \;\leq\; t
        \qquad (t > -1)
        \]
        \emph{(appliquer (G1) en $u = \ln(1+t)$ puis en $u =
        -t/(1+t)$)}.
  \item Montrer que $b_n = \ln n$ est strictement croissante avec
        $\ln n \to +\infty$, et démontrer par Cesàro--Stolz et la
        question 9 que
        \[
        H_n = \sum_{k=1}^{n} \frac 1k \;\sim\; \ln n .
        \]
        (La structure plus fine $H_n = \ln n + \gamma + o(1)$ est le
        devoir maison du \cref{ch:b1:series}.)
  \item (Des rapports aux racines) Soit $u_n > 0 $ avec
        $\frac{u_{n+1}}{u_n} \to L > 0$. À l'aide de la question 9,
        montrer que $\ln\frac{u_{n+1}}{u_n} \to \ln L$~; appliquer
        Cesàro pour en déduire $\frac{\ln u_n}{n} \to \ln L$, puis,
        avec (G1), que $u_n^{1/n} \to L$. Application~: calculer
        $\lim\,\binom{2n}{n}^{1/n}$.
\end{enumerate}

\medskip
\noindent\textbf{Partie IV --- La lente décroissance du sinus.} Soient
$u_0 \in \R$ et $u_{n+1} = \sin u_n$.
\begin{enumerate}[resume]
  \item À partir de (G2), montrer que $0 < \sin x < x$ pour $0 < x
        \leq 1$. En déduire~: $u_1 \in \intcc{-1}{1}$~; si $u_1 = 0$
        la suite est nulle à partir du rang $1$~; et si $u_1 > 0$ (le
        cas $u_1 < 0$ étant symétrique, $\sin$ étant impaire), alors
        $(u_n)_{n \geq 1}$ est strictement décroissante, positive, et
        converge vers $0$ \emph{(identifier la limite via $\ell =
        \sin \ell$, en utilisant $\abs{\sin a - \sin b} \leq \abs{a -
        b}$, elle-même conséquence de (G2) et de la formule de
        transformation produit-somme)}.
  \item On suppose désormais $u_1 \in \intoc{0}{1}$. Montrer par
        encadrement, en utilisant (G2)~:
        \[
        \frac{\sin u_n}{u_n} \to 1
        \qquad\text{et}\qquad
        \frac{u_n - \sin u_n}{u_n^{3}} \to \frac16 .
        \]
  \item Démontrer la factorisation
        \[
        w_n := \frac{1}{u_{n+1}^{2}} - \frac{1}{u_n^{2}}
        = \frac{u_n - \sin u_n}{u_n^{3}} \cdot
        \frac{u_n + \sin u_n}{u_n} \cdot
        \Bigl(\frac{u_n}{\sin u_n}\Bigr)^{2},
        \]
        et en déduire $w_n \to \frac13$.
  \item Conclure avec l'\cref{exo:b1:seq:10} (version
        accroissements) que $\frac{1}{n\,u_n^{2}} \to \frac13$, puis,
        par un argument de conjugaison pour la racine carrée, le
        résultat vedette~:
        \[
        \sqrt n\;u_n \longrightarrow \sqrt 3 ,
        \qquad\text{c.-à-d.}\qquad
        u_n \sim \sqrt{\frac 3n} .
        \]
  \item Quantifier la lenteur~: montrer qu'à partir d'un certain rang
        $\sqrt{2/n} \leq u_n \leq 2/\sqrt n$, de sorte qu'atteindre
        $u_n \leq 10^{-2}$ demande plus de $20\,000$ itérations
        (environ $30\,000$, d'après l'asymptotique). Comparer avec la
        \omterm{ex:b1:seq:heron}{méthode de Héron} (l'\cref{ex:b1:seq:heron}) et expliquer la
        raison structurelle~: au point fixe $0$, la pente de $\sin$
        vaut $1$ (un point fixe \emph{neutre}), alors que les
        itérations qui divisent l'erreur par deux réclament une pente
        de \omterm{def:b1:complex:field}{module} $< 1$.
  \item Montrer que pour \emph{tout} point de départ $u_0 \in \R$,
        ou bien $u_n = 0$ à partir du rang $1$, ou bien $\abs{u_n}
        \sim \sqrt{3/n}$ --- la chute est universelle, seul le signe
        se souvient de $u_0$.
\end{enumerate}

\medskip
\noindent\textbf{Partie V --- Le principe général.} Le sinus n'est
qu'un exemplaire d'une machine.
\begin{enumerate}[resume]
  \item Soit $u_n > 0$, $u_n \to 0$, et $\dfrac{u_n -
        u_{n+1}}{u_n^{2}} \to a > 0$. Démontrer successivement~:
        $\frac{u_{n+1}}{u_n} \to 1$~; puis $\frac{1}{u_{n+1}} -
        \frac{1}{u_n} \to a$~; puis $n\,u_n \to \frac1a$.
  \item (Modèle exact) Pour $u_{n+1} = \dfrac{u_n}{1 + u_n}$, $u_0
        > 0$~: montrer que $\frac{1}{u_n}$ est arithmétique, résoudre
        exactement, et confronter la conclusion de la question 18 à la
        formule exacte.
  \item Pour $u_{n+1} = u_n \eu^{-u_n}$, $u_0 > 0$~: montrer que $u_n
        \to 0$, utiliser (G1) pour encadrer $\frac{1 - \eu^{-t}}{t}$
        entre $\frac{1}{1+t}$ et $1$ pour $t > 0$, et conclure $u_n
        \sim \frac 1n$.
  \item (Contact cubique, télescope au carré) Soit $u_n > 0$, $u_n
        \to 0$, $\dfrac{u_n - u_{n+1}}{u_n^{3}} \to a > 0$. Adapter
        la factorisation de la question 14 pour montrer que
        $\frac{1}{u_{n+1}^2} - \frac{1}{u_n^2} \to 2a$, et conclure
        $n\,u_n^{2} \to \frac{1}{2a}$. Vérifier que $a =
        \frac16$ redonne la partie IV.
\end{enumerate}

\medskip
\noindent\textbf{Partie VI --- Limites de la méthode, et morales.}
\begin{enumerate}[resume]
  \item Montrer que l'hypothèse $b_n \to +\infty$ ne peut pas être
        supprimée~: pour $a_n = 2 - 2^{-n}$ et $b_n = 1 - 2^{-n}$,
        le quotient des accroissements tend vers $1$ tandis que
        $\frac{a_n} {b_n} \to 2$. Désigner la ligne exacte de la
        démonstration de la question 5 qui casse.
  \item (Stolz deux fois) Démontrer $\sum_{k=1}^{n} H_k \sim n \ln n$
        \emph{(une \omterm{def:b1:logic:map}{application} de Cesàro--Stolz, puis la question
        10~; encadrer $(n+1)\ln(n+1) - n\ln n$ à l'aide de la
        question 9)}.
  \item (Moyennes géométriques) Si $u_n > 0$ et $u_n \to \ell > 0$,
        montrer que $(u_1 u_2 \cdots u_n)^{1/n} \to \ell$~; si $u_n
        \to +\infty$, montrer que $(u_1 \cdots u_n)^{1/n} \to
        +\infty$. En déduire $(n!)^{1/n} \to +\infty$.
  \item Synthèse, une phrase pour chaque point~: (i) où exactement la
        complétude est-elle intervenue dans ce problème~; (ii) en
        quel sens Cesàro--Stolz est-il une règle de l'Hospital
        discrète (son jumeau différentiel repose sur le théorème des
        accroissements finis du \cref{ch:b1:derivative})~; (iii)
        énoncer l'heuristique qui relie l'ordre de contact de $f$ en
        un point fixe neutre à l'exposant de décroissance de $u_{n+1}
        = f(u_n)$~; (iv) remonter la constante $3$ de $\sqrt{3/n}$ le
        long de la chaîne $\frac16 \to \frac13 \to 3$.
\end{enumerate}
\end{problem}
