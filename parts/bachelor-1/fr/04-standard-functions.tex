\chapter{Fonctions usuelles}\label{ch:b1:functions}

L'analyse ne vaut que par le stock de fonctions que l'on maîtrise. À
la collection héritée du lycée --- puissances, exponentielle,
logarithme, fonctions trigonométriques --- ce chapitre ajoute leurs
fonctions réciproques ($\arcsin$, $\arccos$, $\arctan$) et la famille
hyperbolique. Les dérivées sont utilisées librement au niveau du
lycée~; la théorie qui sous-tend les fonctions réciproques est
complétée aux \cref{ch:b1:continuity,ch:b1:derivative}.

\section{Exponentielle, logarithme, puissances}

\begin{proposition}[Rappels et caractérisation]\label{prop:b1:functions:expln}
$\exp \colon \R \to \intoo{0}{+\infty}$ et $\ln \colon
\intoo{0}{+\infty} \to \R$ sont des \omterm{def:b1:logic:inj}{bijections} réciproques l'une de
l'autre, strictement croissantes, et
\[
\exp(x + y) = \exp x \exp y, \qquad \ln(xy) = \ln x + \ln y,
\qquad \exp' = \exp, \qquad \ln'(x) = \frac 1x .
\]
De plus, $\exp$ est la \emph{seule} fonction dérivable $f \colon
\R \to \R$ telle que $f' = f$ et $f(0) = 1$.
\end{proposition}

\begin{proof}
Les rappels relèvent du lycée. Pour l'unicité, soit $f$ telle que
$f' = f$ et $f(0) = 1$, et posons $g(x) = f(x)\,\eu^{-x}$. Alors
$g' = f'\eu^{-x} - f\eu^{-x} = 0$, donc $g$ est constante, égale à
$g(0) = 1$~: $f = \exp$.
\end{proof}

\begin{definition}[Puissances générales]\label{def:b1:functions:powers}
Pour $x > 0$ et $\alpha \in \R$~:
$\;x^\alpha = \eu^{\alpha \ln x}$\index{fonction puissance}. Pour
$a > 0$, $a \neq 1$, le \emph{logarithme de base $a$} est $\log_a x =
\frac{\ln x}{\ln a}$, réciproque de $x \mapsto a^x$.
\end{definition}

\begin{example}[Résoudre des équations exponentielles]\label{ex:b1:functions:expequations}
Résolvons $2^x = 5^{\,x-1}$ dans $\R$. Les deux membres sont
positifs~: on peut donc prendre les logarithmes --- étape réversible~:
\[
x\ln2 = (x - 1)\ln5
\iff x(\ln2 - \ln5) = -\ln5
\iff x = \frac{\ln5}{\ln5 - \ln2} = \frac{\ln 5}{\ln\frac52}
\approx 1.756 .
\]
Résolvons ensuite $x^{\sqrt2} = 3$ pour $x > 0$~: élevons à la
puissance $\frac1{\sqrt2}$ (c'est-à-dire appliquons la \omterm{def:b1:logic:inj}{bijection}
réciproque)~:
$x = 3^{1/\sqrt2} = \eu^{(\ln 3)/\sqrt2} \approx 2.175$. L'idée à
retenir~: toute équation mêlant des puissances se dénoue par $\ln$
et $\exp$, car la définition $x^\alpha = \eu^{\alpha\ln x}$
ramène toute manipulation de puissances à l'arithmétique des
exposants --- mais seulement sur le domaine $x > 0$ où cette
définition a un sens.
\end{example}

\begin{example}[Temps de doublement]\label{ex:b1:functions:doubling}
Une quantité croît de $3\,\%$ à chaque étape~: après $n$ étapes, elle a
été multipliée par $(1.03)^n$. Quand double-t-elle~? Résolvons
$(1.03)^n \geq 2$~:
\[
n\ln(1.03) \geq \ln2
\iff n \geq \frac{\ln 2}{\ln 1.03}
= \frac{0.6931}{0.02956} \approx 23.4 ,
\]
donc le premier doublement a lieu à l'étape $24$. (La «~règle de
$72$~» des financiers, qui estime le temps de doublement par
$72$ divisé par le taux en pourcentage, n'est rien d'autre que ce
calcul avec l'approximation $\ln(1 + x) \approx x$, quantifiée au
\cref{ch:b1:taylor}.) On raisonne toujours mieux sur les processus
exponentiels à travers leurs logarithmes~: à cette échelle, la
croissance est linéaire et les questions deviennent des divisions.
\end{example}

\begin{example}[Quelle est la longueur de $2^{2026}$~?]\label{ex:b1:functions:digits}
Le nombre de chiffres décimaux d'un entier $N \geq 1$ est
$\floor{\log_{10} N} + 1$ (en effet, $N$ a $d$ chiffres exactement
quand $10^{d-1} \leq N < 10^d$, c'est-à-dire $d - 1 \leq \log_{10}N <
d$). Pour $N = 2^{2026}$~:
\[
\log_{10} 2^{2026} = 2026\,\log_{10}2
= 2026 \times 0.301030 = 609.887 ,
\]
donc $2^{2026}$ a $610$ chiffres. La partie fractionnaire offre une
prime~: $10^{0.887} \approx 7.7$, donc le nombre \emph{commence} par
$7$. Une seule multiplication a répondu à une question portant sur un
nombre que personne n'écrira jamais --- les logarithmes compriment la
taille multiplicative en taille additive, ce qui est tout leur intérêt
historique (l'\cref{ex:b1:structures:expmorphism} rend cette phrase
précise).
\end{example}

\begin{proposition}[Règles de calcul et croissances comparées]\label{prop:b1:functions:powerrules}
Pour $x, y > 0$ et $\alpha, \beta \in \R$~:
\[
x^{\alpha+\beta} = x^\alpha x^\beta, \quad
(x^\alpha)^\beta = x^{\alpha\beta}, \quad
(xy)^\alpha = x^\alpha y^\alpha, \quad
(x^\alpha)' = \alpha\, x^{\alpha - 1}.
\]
Échelle de croissance\index{croissances comparées} quand $x \to
+\infty$, pour tous $\alpha > 0$ et $\beta > 0$~:
\[
\frac{(\ln x)^\beta}{x^\alpha} \longrightarrow 0,
\qquad
\frac{x^\alpha}{\eu^{\beta x}} \longrightarrow 0 .
\]
\end{proposition}

\begin{proof}
Les identités transcrivent celles de $\exp$ et de $\ln$ à travers la
définition. En détail pour la deuxième (la moins évidente)~: $x^\alpha
> 0$ et $\ln(x^\alpha) = \alpha\ln x$ (appliquer $\ln$ à la
définition), donc
\[
(x^\alpha)^\beta = \eu^{\beta\ln(x^\alpha)}
= \eu^{\beta\alpha\ln x} = x^{\alpha\beta} ;
\]
la première et la troisième sont les mêmes déroulements en deux
lignes, via $\exp(u + v) = \exp u\exp v$ et $\ln(xy) = \ln x + \ln y$.
La dérivée est la règle de dérivation des fonctions composées~:
$(\eu^{\alpha\ln x})' = \frac{\alpha}{x} \eu^{\alpha\ln x}$.

Comparaisons~: partant de $\frac{\ln t}{t} \to 0$ (démontré dans le
volume de Terminale), substituons $t = x^{\alpha/\beta}$~:
$\frac{\ln x}{x^{\alpha/\beta}} \to 0$, puis élevons à la puissance
$\beta$. Pour la seconde, substituons $x = \ln u$ dans la première.
\end{proof}

\begin{example}[Deux limites que tout lecteur doit posséder]\label{ex:b1:functions:xx}
Que valent $\lim_{x \to 0^+} x^x$ et $\lim_{x \to +\infty}
x^{1/x}$~? Les deux puissances sont définies via l'exponentielle~:
réglons donc d'abord le sort de l'exposant~:
\[
x^x = \eu^{x\ln x}, \qquad
x\ln x = -\frac{\ln(1/x)}{1/x} \longrightarrow 0
\quad (x \to 0^+),
\qquad\text{donc } x^x \longrightarrow \eu^0 = 1 ;
\]
et $x^{1/x} = \eu^{(\ln x)/x} \to \eu^0 = 1$ quand $x \to +\infty$,
directement par les \omterm{prop:b1:functions:powerrules}{croissances comparées}. L'idée à retenir~: une
puissance indéterminée (de forme $0^0$ ou $\infty^0$) se traite
\emph{toujours} en réécrivant $u^v = \eu^{v\ln u}$ et en analysant le
produit $v \ln u$ --- jamais en devinant à partir de la base et de
l'exposant pris séparément. La fonction $x \mapsto \frac{\ln x}{x}$
qui a décidé des deux limites est étudiée exhaustivement dans le
devoir maison de ce chapitre.
\end{example}

\begin{remark}[Pièges classiques avec les puissances et les logarithmes]\label{rem:b1:functions:pitfalls}
\begin{enumerate}
  \item \emph{Les domaines.} Pour $\alpha$ irrationnel, la puissance
        $x^\alpha$ exige $x > 0$~; mieux vaut éviter $(-8)^{1/3}$ au profit de
        «~la racine cubique réelle de $-8$~», car la règle
        $(x^\alpha)^\beta = x^{\alpha\beta}$ tombe silencieusement en
        défaut sur les négatifs~:
        $\bigl((-8)^2\bigr)^{1/6} = 2 \neq -2$.
  \item \emph{$\sqrt{x^2} = \abs x$}, et non $x$~: oublier la valeur
        absolue est la source classique de solutions négatives
        perdues.
  \item \emph{$\ln(xy) = \ln x + \ln y$ exige $x, y > 0$.} Pour des
        arguments négatifs, $\ln(xy)$ peut être défini alors que le
        membre de droite ne l'est pas.
  \item \emph{Quelle fonction croît~?} Dans $x^\alpha$, c'est la
        \emph{base} qui varie~; dans $a^x$, c'est l'\emph{exposant}.
        Leurs croissances sont radicalement différentes
        (\cref{prop:b1:functions:powerrules}), et les expressions
        hybrides comme $x^{\ln x}$ ou $x^{1/x}$ doivent être
        réécrites sous la forme $\eu^{(\cdot)}$ avant tout
        raisonnement --- comme dans l'\cref{ex:b1:functions:xx}.
\end{enumerate}
\end{remark}

\section{Fonctions trigonométriques réciproques}

\begin{definition}[$\arcsin$, $\arccos$, $\arctan$]\label{def:b1:functions:arc}
Les restrictions
\[
\sin \colon \intcc{-\tfrac\pi2}{\tfrac\pi2} \to \intcc{-1}{1},
\qquad
\cos \colon \intcc{0}{\pi} \to \intcc{-1}{1},
\qquad
\tan \colon \intoo{-\tfrac\pi2}{\tfrac\pi2} \to \R
\]
sont des \omterm{def:b1:logic:inj}{bijections} strictement monotones. Leurs réciproques se notent
$\arcsin$\index{arcsinus}, $\arccos$\index{arccosinus} et
$\arctan$\index{arctangente}. Ainsi, par exemple, $y = \arcsin x$
($x \in \intcc{-1}{1}$) est \emph{l'}unique angle de
$\intcc{-\frac\pi2}{\frac\pi2}$ dont le sinus vaut $x$.
\end{definition}

\begin{proposition}[Dérivées]\label{prop:b1:functions:arcderiv}
À l'intérieur de leur \omterm{def:b1:logic:sets}{ensemble} de définition~:
\[
\arcsin' x = \frac{1}{\sqrt{1 - x^2}},
\qquad
\arccos' x = \frac{-1}{\sqrt{1 - x^2}},
\qquad
\arctan' x = \frac{1}{1 + x^2}.
\]
\end{proposition}

\begin{proof}
Anticipons la règle de dérivation de la réciproque, démontrée au
\cref{ch:b1:derivative}~: si $f$ est une \omterm{def:b1:logic:inj}{bijection} dérivable avec
$f' \neq 0$, alors $(f^{-1})'(x) = \frac{1}{f'(f^{-1}(x))}$. Pour
$\arcsin$~: avec $y = \arcsin x$,
\[
\arcsin' x = \frac{1}{\cos y}
= \frac{1}{\sqrt{1 - \sin^2 y}} = \frac{1}{\sqrt{1 - x^2}},
\]
où $\cos y = +\sqrt{1 - \sin^2 y}$ car $y \in
\intoo{-\frac\pi2}{\frac\pi2}$ impose $\cos y > 0$. De même,
$\arccos' x = \frac{1}{-\sin y}$ avec $\sin y > 0$ sur
$\intoo{0}{\pi}$, et $\arctan' x = \frac{1}{1 + \tan^2 y} =
\frac{1}{1 + x^2}$ en utilisant $\tan' = 1 + \tan^2$.
\end{proof}

\begin{example}[Reconnaître une constante cachée]\label{ex:b1:functions:hiddenconst}
Étudions $g(x) = \arctan\dfrac{1 - x}{1 + x}$ sur $\intoo{-1}
{+\infty}$. Dérivation des fonctions composées et court calcul~:
\[
g'(x) = \frac{1}{1 + \bigl(\frac{1-x}{1+x}\bigr)^2}\cdot
\frac{-(1+x) - (1-x)}{(1+x)^2}
= \frac{-2}{(1+x)^2 + (1-x)^2}
= \frac{-1}{1 + x^2} ,
\]
puisque $(1+x)^2 + (1-x)^2 = 2 + 2x^2$. Donc $g' = -\arctan'$~: la
fonction $g + \arctan$ a une dérivée nulle sur l'\emph{intervalle}
$\intoo{-1}{+\infty}$, elle y est donc constante~; sa valeur en $x =
0$ vaut $\arctan 1 + \arctan 0 = \frac\pi4$. Conclusion~:
\[
\arctan\frac{1 - x}{1 + x} = \frac\pi4 - \arctan x
\qquad (x > -1) .
\]
Sur $\intoo{-\infty}{-1}$, le même calcul de dérivée vaut, mais la
constante est différente ($-\frac{3\pi}4$~: évaluer la limite quand
$x \to -\infty$). L'idée à retenir~: «~dérivée nulle implique
constante~» est un énoncé intervalle par intervalle --- exactement la
subtilité exploitée dans l'\cref{exo:b1:functions:6} et
l'\cref{exo:b1:functions:10}.
\end{example}

\begin{proposition}[Identités usuelles]\label{prop:b1:functions:arcidentities}
\begin{enumerate}
  \item Pour $x \in \intcc{-1}{1}$~: $\arcsin x + \arccos x =
        \dfrac{\pi}{2}$.
  \item Pour $x > 0$~: $\arctan x + \arctan\dfrac 1x = \dfrac{\pi}{2}$
        (et $-\frac\pi2$ pour $x < 0$).
  \item $\sin(\arccos x) = \cos(\arcsin x) = \sqrt{1 - x^2}$~;
        $\;\tan(\arcsin x) = \dfrac{x}{\sqrt{1 - x^2}}$ pour
        $\abs x < 1$.
\end{enumerate}
\end{proposition}

\begin{proof}
(1) La dérivée de $x \mapsto \arcsin x + \arccos x$ est nulle sur
$\intoo{-1}{1}$ (\cref{prop:b1:functions:arcderiv}), donc la fonction
y est constante, égale à sa valeur $\frac\pi2$ en $0$~; les valeurs
aux extrémités $\pm 1$ se vérifient directement ($\frac\pi2 + 0$ et
$-\frac\pi2 + \pi$).

(2) Même méthode sur $\intoo{0}{+\infty}$~: la dérivée vaut
$\frac{1}{1+x^2} + \frac{-1/x^2}{1 + 1/x^2} = \frac{1}{1+x^2} -
\frac{1}{x^2 + 1} = 0$, et en $x = 1$ la somme vaut $2\arctan 1 =
\frac\pi2$. Pour $x < 0$, utiliser l'imparité de $\arctan$.

(3) Avec $y = \arccos x \in \intcc{0}{\pi}$~: $\sin y \geq 0$, donc
$\sin y = \sqrt{1 - \cos^2 y} = \sqrt{1 - x^2}$~; de même pour
$\cos(\arcsin x)$, et la formule de la tangente est le quotient.
\end{proof}

\begin{remark}
$\arcsin(\sin\theta) = \theta$ n'est vrai \emph{que} pour $\theta \in
\intcc{-\frac\pi2}{\frac\pi2}$~: par exemple
$\arcsin(\sin\pi) = 0 \neq \pi$. La composition dans l'autre
sens, $\sin(\arcsin x) = x$, est valable sur tout
$\intcc{-1}{1}$.
\end{remark}

\begin{example}[Ramener un angle à l'intervalle principal]
\label{ex:b1:functions:wrap}
Calculons
\[
\arctan\Bigl(\tan\frac{3\pi}4\Bigr)
\qquad\text{et}\qquad
\arctan\Bigl(\tan\frac{17\pi}5\Bigr).
\]
La recette~: remplacer l'angle par l'unique angle de
$\intoo{-\frac\pi2}{\frac\pi2}$ ayant la même tangente, c'est-à-dire
lui retrancher le bon multiple de $\pi$ (la période de $\tan$).
D'abord~: $\frac{3\pi}4 - \pi = -\frac\pi4$, donc la réponse est
$-\frac\pi4$. Ensuite~: $\frac{17\pi} 5 - 3\pi = \frac{2\pi}5 \in
\intoo{-\frac\pi2}{\frac\pi2}$, donc la réponse est $\frac{2\pi}5$.
Le calcul est une division euclidienne de l'angle par $\pi$ déguisée
--- et les recettes analogues pour $\arcsin$ (se ramener à
$\intcc{-\frac\pi2}{\frac\pi2}$, période $2\pi$) et $\arccos$
(se ramener à $\intcc0\pi$) gouvernent l'\cref{exo:b1:functions:1} et
la réponse par morceaux de l'\cref{exo:b1:functions:10}.
\end{example}

\begin{example}[Additionner des arctangentes sans se tromper]\label{ex:b1:functions:arctansum}
Démontrons que
\[
\arctan\frac12 + \arctan\frac15 + \arctan\frac18 = \frac\pi4 .
\]
Deux ingrédients~: la formule d'addition de la tangente et --- l'étape
que les débutants oublient --- une \emph{localisation} de la somme.
D'abord, $\tan(u + v) = \frac{\tan u + \tan v}{1 - \tan u\tan v}$ avec
$u = \arctan\frac12$, $v = \arctan\frac15$ donne
\[
\tan(u + v) = \frac{\frac12 + \frac15}{1 - \frac1{10}}
= \frac{7/10}{9/10} = \frac79,
\qquad\text{puis}\qquad
\tan\Bigl(u + v + \arctan\frac18\Bigr)
= \frac{\frac79 + \frac18}{1 - \frac7{72}}
= \frac{65/72}{65/72} = 1 .
\]
Ensuite, chacun des trois angles appartient à $\intoo0{\frac\pi4}$
(leurs arguments sont plus petits que $1$), donc la somme appartient à
$\intoo0{\frac{3\pi}4}$~; le seul angle de cet intervalle dont la
tangente vaut $1$ est $\frac\pi4$. Sans la localisation, la conclusion
serait «~$\frac\pi4$ modulo un multiple de $\pi$~» --- une
demi-démonstration. La même discipline en deux temps gouverne
l'\cref{exo:b1:functions:8} et l'\cref{exo:b1:functions:12}.
\end{example}

\begin{example}[Une équation en arctangente]\label{ex:b1:functions:arctaneq}
Résolvons $\arctan x + \arctan 2x = \dfrac\pi4$. Localisons d'abord~:
le membre de gauche a le signe de $x$ (les deux termes l'ont), donc
toute solution vérifie $x > 0$, et la somme appartient alors à
$\intoo0\pi$. Prenons les tangentes (la tangente n'est \omterm{def:b1:logic:inj}{injective} sur
aucun intervalle de longueur $\pi$, mais combinée à la localisation
cela suffira)~: la formule d'addition donne
\[
\tan\bigl(\arctan x + \arctan 2x\bigr)
= \frac{3x}{1 - 2x^2} = 1
\iff 2x^2 + 3x - 1 = 0
\iff x = \frac{-3 \pm \sqrt{17}}4 .
\]
La racine négative est écartée par la localisation~; pour le candidat
positif $x_0 = \frac{\sqrt{17} - 3}4 \approx 0.28$, la somme
appartient à $\intoo0\pi$ et a pour tangente $1$, et le seul angle de
ce type est $\frac\pi4$ (sur $\intoo{\frac\pi2}\pi$ la tangente est
négative)~: $x_0$ est l'unique solution. Notons la forme de
l'argument~: prendre les tangentes peut créer des solutions, jamais en
perdre~; on résout donc l'équation polynomiale, \emph{puis} on filtre
par la localisation --- la même discipline de vérification a
posteriori que lorsqu'on élève une équation au carré.
\end{example}

\section{Fonctions hyperboliques}

\begin{definition}[$\cosh$, $\sinh$, $\tanh$]\label{def:b1:functions:hyperbolic}
Pour $x \in \R$~:
\[
\cosh x = \frac{\eu^x + \eu^{-x}}{2},
\qquad
\sinh x = \frac{\eu^x - \eu^{-x}}{2},
\qquad
\tanh x = \frac{\sinh x}{\cosh x}
\]
(\emph{cosinus, sinus, tangente hyperboliques}\index{fonctions
hyperboliques}). $\cosh$ est paire, $\sinh$ et $\tanh$ sont impaires.
\end{definition}

\begin{proposition}[Propriétés fondamentales]\label{prop:b1:functions:hyprules}
Pour tous $x, y \in \R$~:
\begin{enumerate}
  \item $\cosh^2 x - \sinh^2 x = 1$~;
  \item $\cosh' = \sinh$, $\;\sinh' = \cosh$, $\;\tanh' = 1 - \tanh^2
        = \dfrac{1}{\cosh^2}$~;
  \item $\cosh(x+y) = \cosh x\cosh y + \sinh x \sinh y$ et
        $\sinh(x+y) = \sinh x \cosh y + \cosh x \sinh y$~;
  \item $\sinh$ est une \omterm{def:b1:logic:inj}{bijection} strictement croissante de $\R$ sur
        $\R$~; $\cosh$ restreinte à $\R_+$ est une \omterm{def:b1:logic:inj}{bijection}
        strictement croissante sur $\intco{1}{+\infty}$~; $\tanh$ est
        une \omterm{def:b1:logic:inj}{bijection} strictement croissante de $\R$ sur
        $\intoo{-1}{1}$.
\end{enumerate}
\end{proposition}

\begin{proof}
(1) $\bigl(\frac{\eu^x + \eu^{-x}}{2}\bigr)^2 - \bigl(\frac{\eu^x -
\eu^{-x}}{2}\bigr)^2 = \frac{(\eu^{2x} + 2 + \eu^{-2x}) - (\eu^{2x} -
2 + \eu^{-2x})}{4} = 1$.

(2) Dériver les formules de définition~; pour $\tanh$, la règle de
dérivation d'un quotient donne $\frac{\cosh^2 - \sinh^2}{\cosh^2}$,
qui vaut à la fois $1 - \tanh^2$ et $\frac{1}{\cosh^2}$ d'après (1).

(3) Développer les membres de droite à l'aide des définitions. En
détail pour la première formule~:
\[
\cosh x\cosh y + \sinh x\sinh y
= \frac{(\eu^x + \eu^{-x})(\eu^y + \eu^{-y})
+ (\eu^x - \eu^{-x})(\eu^y - \eu^{-y})}{4} ;
\]
des huit produits, les quatre termes «~croisés~»
($\eu^{x-y}$, $\eu^{y-x}$) s'annulent deux à deux, tandis que
$\eu^{x+y}$ et $\eu^{-(x+y)}$ apparaissent chacun deux fois~:
le total vaut $\frac{2\eu^{x+y} + 2\eu^{-(x+y)}}{4} = \cosh(x+y)$.
La formule du sinus est identique, ce sont cette fois les termes
croisés qui survivent.

(4) $\sinh' = \cosh \geq 1 > 0$, donc $\sinh$ est strictement
croissante, de limites $\pm\infty$ (terme dominant $\pm\frac{\eu^{\abs
x}}{2}$)~; le caractère \omterm{def:b1:logic:inj}{bijectif} résulte alors du théorème des valeurs
intermédiaires (utilisé au niveau du lycée~; traitement systématique
au \cref{ch:b1:continuity}). Sur $\R_+$, $\cosh' = \sinh \geq 0$ ne
s'annule qu'en $0$~: strictement croissante de $\cosh 0 = 1$ à
$+\infty$. Et $\tanh' > 0$, de limites $\pm 1$ en $\pm\infty$~: en
détail,
\[
\tanh x = \frac{\eu^x - \eu^{-x}}{\eu^x + \eu^{-x}}
= \frac{1 - \eu^{-2x}}{1 + \eu^{-2x}}
\xrightarrow[x \to +\infty]{} 1 ,
\]
après division du numérateur et du dénominateur par $\eu^x$, et
l'imparité donne la limite $-1$ en $-\infty$~; la fonction
strictement croissante $\tanh$ envoie donc $\R$ sur $\intoo{-1}1$.
\end{proof}

\begin{remark}[Pourquoi «~hyperbolique~»~?]
Le point $(\cosh t, \sinh t)$ parcourt la branche $x > 0$ de
l'hyperbole $x^2 - y^2 = 1$ (d'après la
\cref{prop:b1:functions:hyprules}~(1)), exactement comme $(\cos t,
\sin t)$ parcourt le cercle $x^2 + y^2 = 1$. Toute identité circulaire
a une sœur hyperbolique, les changements de signe étant régis par
$\sinh^2 \leftrightarrow -\sin^2$.
\end{remark}

\begin{example}[La formule d'addition pour $\tanh$]\label{ex:b1:functions:tanhadd}
En divisant les deux formules d'addition de la
\cref{prop:b1:functions:hyprules}~(3) par $\cosh x\cosh y$~:
\[
\tanh(x + y)
= \frac{\sinh x\cosh y + \cosh x\sinh y}
{\cosh x\cosh y + \sinh x\sinh y}
= \frac{\tanh x + \tanh y}{1 + \tanh x\,\tanh y} ,
\]
sœur hyperbolique de la formule d'addition de la tangente --- avec un
$+$ là où la trigonométrie a un $-$. Un dividende~: comme
$\abs{\tanh} < 1$, le membre de droite est une règle d'«~addition des
vitesses~» qui ne quitte jamais $\intoo{-1}1$~: si $u, v \in
\intoo{-1}1$, alors $\frac{u + v}{1 + uv} \in \intoo{-1}1$ également
(écrire $u = \tanh a$, $v = \tanh b$, ce qui est possible par
\omterm{def:b1:logic:inj}{bijectivité}, et lire la formule à l'envers). Le vérifier
algébriquement, sans les \omterm{def:b1:functions:hyperbolic}{fonctions hyperboliques}, est un exercice un
peu pénible~; le paramétrage par $\tanh$ le réduit à une ligne --- la
stratégie même que les fonctions circulaires offrent pour le cercle
unité.
\end{example}

\begin{proposition}[Fonctions hyperboliques réciproques]\label{prop:b1:functions:invhyp}
Les réciproques fournies par la
\cref{prop:b1:functions:hyprules}~(4) admettent des expressions
explicites~:
\[
\operatorname{arsinh} x = \ln\bigl(x + \sqrt{x^2 + 1}\bigr)
\ (x \in \R),
\qquad
\operatorname{arcosh} x = \ln\bigl(x + \sqrt{x^2 - 1}\bigr)
\ (x \geq 1),
\]
\[
\operatorname{artanh} x = \frac 12 \ln\frac{1 + x}{1 - x}
\ (\abs x < 1),
\]
de dérivées respectives $\frac{1}{\sqrt{x^2+1}}$,
$\frac{1}{\sqrt{x^2-1}}$ ($x > 1$) et $\frac{1}{1 - x^2}$.
\end{proposition}

\begin{proof}
Pour $\operatorname{arsinh}$~: résolvons $x = \sinh y = \frac{\eu^y -
\eu^{-y}}{2}$. En posant $u = \eu^y > 0$~: $u^2 - 2xu - 1 = 0$, donc
$u = x + \sqrt{x^2 + 1}$ (la racine $x - \sqrt{x^2+1}$ est négative),
et $y = \ln(x + \sqrt{x^2+1})$. Les deux autres sont des calculs
identiques ($u^2 - 2xu + 1 = 0$ pour $\operatorname{arcosh}$, en
gardant la racine $\geq 1$~; une réorganisation en deux lignes pour
$\operatorname{artanh}$). Dérivées~: dériver les expressions
logarithmiques, par exemple
\[
\bigl(\ln(x + \sqrt{x^2+1})\bigr)'
= \frac{1 + \frac{x}{\sqrt{x^2+1}}}{x + \sqrt{x^2+1}}
= \frac{1}{\sqrt{x^2 + 1}} . \qedhere
\]
\end{proof}

\begin{example}[Les expressions explicites à l'œuvre]\label{ex:b1:functions:arcoshtwo}
La solution de $\cosh t = 2$ avec $t \geq 0$ est, d'après
l'expression explicite, $t = \operatorname{arcosh} 2 = \ln(2 +
\sqrt3) \approx 1.317$~; l'autre solution est $-t$, par parité --- et
en effet $\ln(2 - \sqrt3) = \ln\frac1{2 + \sqrt3} = -\ln(2 +
\sqrt3)$~: les deux racines $u = \eu^t$ du trinôme $u^2 - 4u + 1 = 0$
sont inverses l'une de l'autre, comme l'exige leur produit égal à $1$
(relations de Viète). Ce minuscule calcul exhibe le schéma général~:
les équations hyperboliques se ramènent à des équations du second
degré en $\eu^t$, et la symétrie $t \mapsto -t$ apparaît comme la
symétrie $u \mapsto 1/u$ du trinôme --- à retenir pour résoudre
l'\cref{exo:b1:functions:7}.
\end{example}

\begin{method}[Choisir la bonne forme de primitive]\label{met:b1:functions:primitive}
Les trois schémas de dérivée à mémoriser pour l'intégration
(\cref{ch:b1:integration})~:
\[
\int \frac{\dd x}{1 + x^2} = \arctan x + C,
\qquad
\int \frac{\dd x}{\sqrt{1 - x^2}} = \arcsin x + C,
\qquad
\int \frac{\dd x}{\sqrt{x^2 + 1}} = \operatorname{arsinh} x + C,
\]
et pour un trinôme du second degré quelconque, on s'y ramène en le
mettant sous forme canonique puis en changeant d'échelle.
\end{method}

\begin{remark}[Où ce chapitre est utilisé]\label{rem:b1:functions:whereused}
Ce chapitre est le vocabulaire de travail de toute l'analyse à venir.
L'échelle de croissance de la \cref{prop:b1:functions:powerrules}
tranche des questions de convergence tout au long des
\cref{ch:b1:seq,ch:b1:series}~; les formules de dérivée de la
\cref{prop:b1:functions:arcderiv} et de la
\cref{prop:b1:functions:invhyp} sont les primitives dont on a le plus
souvent besoin au \cref{ch:b1:integration}, via la
\cref{met:b1:functions:primitive}~; les \omterm{def:b1:functions:hyperbolic}{fonctions hyperboliques}
paramètrent les solutions de l'équation $y'' = y$ au
\cref{ch:b1:diffeq} exactement comme les fonctions circulaires
paramètrent celles de $y'' = -y$. La caractérisation de $\exp$ par
$f' = f$, $f(0) = 1$ (\cref{prop:b1:functions:expln}) est la graine,
en dimension un, de la théorie des équations différentielles
linéaires, et la chaînette $y = \cosh x$ reviendra parmi les courbes
planes du \cref{ch:b1:curves}.
\end{remark}

\begin{remark}[Interlude~: le programme des fonctions réciproques]\label{rem:b1:functions:interlude}
Ce chapitre a déroulé quatre fois le même programme~: restreindre une
fonction jusqu'à ce qu'elle devienne une \omterm{def:b1:logic:inj}{bijection} strictement
monotone, nommer sa réciproque, transporter à travers elle toutes les
formules. Ce qui a été \emph{utilisé} à chaque étape --- qu'une
fonction continue strictement monotone sur un intervalle est une
\omterm{def:b1:logic:inj}{bijection} sur un intervalle, que sa réciproque est continue, puis
dérivable hors des points critiques --- l'a été à crédit sur le lycée.
La dette est remboursée dans ce volume~: le \cref{ch:b1:continuity}
démontre l'énoncé de \omterm{def:b1:logic:inj}{bijectivité} (le théorème de la \omterm{def:b1:logic:inj}{bijection}
monotone, qui repose sur le théorème des valeurs intermédiaires), et
le \cref{ch:b1:derivative} démontre la règle de dérivation
$(f^{-1})' = 1/(f' \circ f^{-1})$ qui a silencieusement produit toutes
les formules de la \cref{prop:b1:functions:arcderiv} et de la
\cref{prop:b1:functions:invhyp}. Lire ces chapitres en gardant
$\arcsin$ et $\operatorname{arsinh}$ en tête --- comme les exemples
travaillés que la théorie a été bâtie pour justifier --- est la façon
prévue de contourner la circularité apparente.
\end{remark}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:b1:functions:1}
Calculer sans calculatrice~: $\arcsin\frac{\sqrt 3}{2}$~;
$\;\arccos\bigl(-\frac 12\bigr)$~; $\;\arctan(-1)$~;
$\;\arcsin\bigl(\sin\frac{5\pi}{6}\bigr)$~;
$\;\arccos\bigl(\cos\frac{7\pi}{4}\bigr)$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:functions:2}
Donner l'\omterm{def:b1:logic:sets}{ensemble} de définition de $f(x) = \arcsin(2x - 1)$ et
calculer $f'$ là où elle est définie. Mêmes questions pour
$g(x) = \arctan\sqrt{x}$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:functions:3}
Démontrer que, pour tout $x \in \R$~: $\cosh x + \sinh x = \eu^x$ et
$(\cosh x + \sinh x)^n = \cosh nx + \sinh nx$ pour tout $n \in \Z$
(une formule de Moivre hyperbolique).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:functions:4}
Simplifier $\cos(2\arcsin x)$ et $\sin(2\arctan x)$ en expressions
algébriques de $x$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:functions:5}
Ordonner, pour $x$ grand, les fonctions
$x^{100}$, $\;\eu^{\sqrt x}$, $\;(\ln x)^{1000}$, $\;\eu^{x/100}$,
$\;x^{\ln x}$, de la croissance la plus lente à la plus rapide, avec
des justifications fondées sur la \cref{prop:b1:functions:powerrules}.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:functions:6}
Étudier la fonction $f(x) = \arctan\dfrac{2x}{1 - x^2}$ sur son
\omterm{def:b1:logic:sets}{ensemble} de définition~: calculer $f'$, comparer avec
$(2\arctan x)'$, et exprimer $f(x)$ à l'aide de $\arctan x$ sur chacun
des trois intervalles du domaine.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:functions:7}
Résoudre dans $\R$~: $\;\cosh x = 2$~; puis $5\cosh x - 4 \sinh x = 3$
(exprimer les solutions à l'aide de logarithmes). \emph{Indication
pour la deuxième~: tout écrire avec $u = \eu^x$.}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:functions:8}
Démontrer l'identité $\arctan\frac{1}{2} + \arctan\frac{1}{3} =
\frac{\pi}{4}$, puis la \omterm{rem:b1:functions:interlude}{formule de Machin}\index{formule de Machin}
\[
4\arctan\frac 15 - \arctan\frac{1}{239} = \frac{\pi}{4}.
\]
\emph{Indication~: calculer la tangente des deux membres à l'aide de
la formule d'addition, et contrôler dans quel intervalle se situe
chaque membre.}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:functions:9}
Démontrer que, pour tout $x \geq 0$~: $\;x - \dfrac{x^3}{6} \leq
\sin x \leq x$ \emph{(étudier les dérivées successives des
différences)}, et en déduire que $\lim_{x \to 0^+} \frac{x - \sin
x}{x^3} = \frac 16$ est plausible --- la limite elle-même est établie
au \cref{ch:b1:taylor}.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:functions:10}
Pour $x \in \intcc{-1}{1}$, posons
$h(x) = \arcsin\bigl(2x\sqrt{1 - x^2}\,\bigr) - 2\arcsin x$. On
pourrait s'attendre à $h = 0$ d'après l'identité $\sin 2y = 2\sin
y\cos y$ --- mais $h$ n'est \emph{pas} identiquement nulle. Calculer
$h'$ sur les intervalles ouverts où elle existe, évaluer $h$ en des
points bien choisis, et donner la description complète, constante par
morceaux, de $h$ sur $\intcc{-1}{1}$.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:functions:11}
(Fonction de Gudermann) Soit $g(x) = \arctan(\sinh x)$ pour $x \in
\R$. Démontrer que $g$ est une \omterm{def:b1:logic:inj}{bijection} impaire, strictement
croissante, de $\R$ sur $\intoo{-\frac\pi2}{\frac\pi2}$, que
$g'(x) = \frac{1}{\cosh x}$, et que $\tan g(x) = \sinh x$,
$\;\sin g(x) = \tanh x$, $\;\cos g(x) = \frac{1}{\cosh x}$~: la
fonction $g$ relie trigonométrie circulaire et trigonométrie
hyperbolique sans passer par les nombres complexes.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:functions:12}
Démontrer que
\[
\arctan 1 + \arctan 2 + \arctan 3 = \pi .
\]
\emph{Indication~: calculer d'abord $\arctan 2 + \arctan 3$, en
localisant la somme comme dans l'\cref{ex:b1:functions:arctansum}~;
attention, le dénominateur de la formule d'addition est ici négatif.}
\end{exercise}

\section{Problème~: l'équation $x^y = y^x$}

\begin{problem}\label{pb:b1:functions:1}
Quels couples de nombres positifs vérifient $x^y = y^x$~? Tout le
monde en connaît un exemple d'apparence accidentelle, $2^4 = 4^2 =
16$~; ce problème montre que rien n'y est accidentel. L'équation tout
entière est gouvernée par les variations d'une seule fonction,
\[
f(t) = \frac{\ln t}{t} \qquad (t > 0),
\]
dont l'étude livre~: l'\omterm{def:b1:logic:sets}{ensemble} complet des solutions (une diagonale
et une branche courbe passant par $(\eu, \eu)$), un paramétrage
rationnel de la branche, le fait que $(2, 4)$ en est l'\emph{unique}
point entier et la classification de tous ses points rationnels ---
plus, en dividendes, la comparaison de $\eu^\pi$ et $\pi^\eu$ et la
convergence monotone de $\bigl(1 + \frac1n\bigr)^n$ vers $\eu$.
\index{fonction puissance}

\medskip
\noindent\textbf{Partie I --- La fonction $f(t) = \ln t / t$.}
\begin{enumerate}
  \item Justifier que $f$ est dérivable sur $\intoo0{+\infty}$,
        calculer $f'$ et dresser le tableau de variations~: $f$
        croît strictement sur $\intoc0\eu$, décroît strictement sur
        $\intco\eu{+\infty}$, avec pour maximum $f(\eu) = \frac1\eu$.
  \item Déterminer les limites de $f$ en $0^+$ et en $+\infty$
        (\cref{prop:b1:functions:powerrules}), le signe de $f$
        (négatif sur $\intoo01$, nul en $1$, positif au-delà), et
        tracer l'allure du graphe.
  \item Vérifier par un calcul direct que $f(2) = f(4)$. (Garder
        cette égalité en tête~: tout le problème en découle.)
  \item Soit $c \in \R$. Discuter, selon la valeur de $c$, le
        nombre de solutions de $f(t) = c$~: exactement une pour
        $c \leq 0$~; exactement deux (une dans $\intoo1\eu$, une
        dans $\intoo\eu{+\infty}$) pour $0 < c < \frac1\eu$~;
        exactement une pour $c = \frac1\eu$~; aucune pour
        $c > \frac1\eu$.
  \item En déduire que $\eu^x > x^\eu$ pour tout $x > 0$ tel que
        $x \neq \eu$, et trancher en particulier lequel de $\eu^\pi$
        et $\pi^\eu$ est le plus grand.
\end{enumerate}

\medskip
\noindent\textbf{Partie II --- L'équation et sa courbe.} Dans cette
partie, $x, y > 0$.
\begin{enumerate}[resume]
  \item Montrer que $x^y = y^x \iff f(x) = f(y)$.
  \item En déduire la structure de l'\omterm{def:b1:logic:sets}{ensemble} des solutions~: tous
        les couples diagonaux $(x, x)$~; et les couples \emph{non
        triviaux} ($x \neq y$), qui vérifient~: les deux coordonnées
        sont $> 1$, et l'une appartient à $\intoo1\eu$ tandis que
        l'autre appartient à $\intoo\eu{+\infty}$.
  \item Paramétrer les couples non triviaux~: en écrivant $y = tx$
        avec $t > 0$, $t \neq 1$, montrer que $x^y = y^x$ impose
        \[
        x(t) = t^{\frac1{t-1}},
        \qquad
        y(t) = t\,x(t) = t^{\frac{t}{t-1}},
        \]
        et que, réciproquement, tout couple de cette forme est
        solution.
  \item Vérifier que $t = 2$ donne $(2, 4)$, et démontrer la
        symétrie $x(1/t) = y(t)$, $y(1/t) = x(t)$~: inverser le
        paramètre échange les deux coordonnées.
  \item Déterminer les limites de $x(t)$ et $y(t)$ quand $t \to 1$
        (toutes deux tendent vers $\eu$), quand $t \to +\infty$
        ($x \to 1$, $y \to +\infty$) et quand $t \to 0^+$ ($x \to
        +\infty$, $y \to 1$). Décrire la branche obtenue~: une
        courbe asymptote aux droites $x = 1$ et $y = 1$, coupant la
        diagonale en $(\eu, \eu)$.
  \item Montrer que $t \mapsto x(t)$ est strictement décroissante
        sur $\intoo1{+\infty}$. \emph{(Étudier $h(t) = 1 - \frac1t -
        \ln t$, dont le signe contrôle la dérivée de $\ln x(t) =
        \frac{\ln t}{t - 1}$.)}
  \item Conclure que les solutions non triviales définissent une
        \omterm{def:b1:logic:inj}{bijection} strictement décroissante $\varphi \colon
        \intoo1\eu \to \intoo\eu{+\infty}$ vérifiant
        $x^{\varphi(x)} = \varphi(x)^x$, et que $\varphi$ est une
        involution de la branche~: $\varphi(\varphi(x)) = x$ partout
        où les deux membres sont définis.
\end{enumerate}

\medskip
\noindent\textbf{Partie III --- Points entiers et points rationnels.}
\begin{enumerate}[resume]
  \item Démontrer que $(2, 4)$ et $(4, 2)$ sont les seules solutions
        \emph{entières} non triviales de $x^y = y^x$.
  \item Pour $n \in \N^*$, appliquer le paramétrage avec $t = 1 +
        \frac1n$ et montrer que
        \[
        x_n = \Bigl(1 + \frac1n\Bigr)^{n},
        \qquad
        y_n = \Bigl(1 + \frac1n\Bigr)^{n+1}
        \]
        est une solution \emph{rationnelle} non triviale pour tout
        $n$, avec $(x_1, y_1) = (2, 4)$.
  \item Réciproquement, soit $(x, y)$ une solution rationnelle non
        triviale avec $y > x$, et écrivons $t = y/x = 1 + \frac rs$
        sous forme irréductible ($r, s \in \N^*$). En utilisant
        $x = t^{s/r}$ et en admettant l'unicité de la décomposition
        en facteurs premiers (familière depuis le lycée~; démontrée
        au \cref{ch:b1:arith}), montrer que la rationalité de $x$
        impose que $s$ et $s + r$ soient tous deux des puissances
        $r$-ièmes d'entiers.
  \item Montrer, à l'aide de la formule du binôme, que $a^r - b^r =
        r$ n'a pas de solution entière $a > b \geq 1$ lorsque
        $r \geq 2$, et conclure~: les solutions rationnelles de
        $x^y = y^x$ sont exactement les couples $(x_n, y_n)$ de la
        question 14 (et leurs échangés).
  \item Vérifier numériquement le cas $n = 2$~: calculer $f(9/4)$ et
        $f(27/8)$ à quatre décimales et constater qu'ils coïncident.
\end{enumerate}

\medskip
\noindent\textbf{Partie IV --- Dividendes.}
\begin{enumerate}[resume]
  \item Utiliser les questions 7 et 11 pour démontrer, sans aucun
        calcul supplémentaire, que la suite $x_n = \bigl(1 +
        \frac1n\bigr)^n$ est strictement croissante avec $x_n < \eu$,
        que $y_n = \bigl(1 + \frac1n\bigr)^{n+1}$ est strictement
        décroissante avec $y_n > \eu$, et que toutes deux convergent
        vers $\eu$. (Le paramètre $t_n = 1 + \frac1n$ décroît vers
        $1$.)
  \item Établir la règle de comparaison générale pour $1 < a < b$~:
        si $\eu \leq a$, alors $a^b > b^a$~; si $b \leq \eu$, alors
        $a^b < b^a$~; et montrer, sur les deux exemples $(2, 3)$ et
        $(2, 5)$, que dans le cas mixte $a < \eu < b$ les deux
        issues se produisent effectivement.
  \item Supposons l'involution $\varphi$ de la question 12
        dérivable en $\eu$ (elle l'est). Dériver l'identité
        $\varphi(\varphi(x)) = x$ en $x = \eu$ et en déduire
        $\varphi'(\eu) = -1$~: la branche coupe la diagonale
        perpendiculairement.
  \item Trouver l'unique couple solution vérifiant $y = 3x$, sous
        forme explicite, et le vérifier numériquement à quatre
        décimales à l'aide de $f$.
  \item Parmi les nombres $\sqrt2$, $\sqrt[3]3$, $\sqrt[4]4$,
        $\sqrt[5]5$, déterminer le plus grand et identifier les deux
        qui sont égaux. \emph{(Comparer $n^{1/n} =
        \eu^{f(n)}$.)}
\end{enumerate}

\medskip
\noindent\textbf{Partie V --- Synthèse.}
\begin{enumerate}[resume]
  \item Décrire l'\omterm{def:b1:logic:sets}{ensemble} complet des solutions de $x^y = y^x$ dans
        le quart de plan $x, y > 0$ --- diagonale et branche, leur
        intersection $(\eu, \eu)$, les asymptotes, le point entier
        $(2, 4)$, les points rationnels s'accumulant en
        $(\eu, \eu)$ --- sous une forme que l'on pourrait esquisser
        de mémoire.
  \item Où exactement le problème a-t-il utilisé~: (i) les
        \omterm{prop:b1:functions:powerrules}{croissances comparées} de la
        \cref{prop:b1:functions:powerrules}~; (ii) le théorème des
        valeurs intermédiaires (à travers les énoncés de
        \omterm{def:b1:logic:inj}{bijectivité})~; (iii) l'unicité admise de la décomposition en
        facteurs premiers~? Une phrase pour chacun.
  \item Morale, en un court paragraphe~: un unique tableau de
        variations a résolu une équation à deux inconnues, classifié
        ses points rationnels et démontré la convergence monotone de
        $\bigl(1 + \frac1n\bigr)^n$ --- commenter cette économie de
        moyens, et dire où chaque fil est industrialisé plus loin
        dans le volume (\cref{ch:b1:seq} pour la suite,
        \cref{ch:b1:derivative} pour les tableaux de variations,
        \cref{ch:b1:taylor} pour la précision qui manque au tableau).
\end{enumerate}
\end{problem}
