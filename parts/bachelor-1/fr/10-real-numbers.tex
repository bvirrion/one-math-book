\chapter{Nombres réels}\label{ch:b1:reals}

Toute l'analyse repose sur une propriété qui distingue $\R$ de
$\Q$~: toute partie non vide et majorée admet un \emph{plus petit}
\omterm{def:b1:reals:bounds}{majorant}. Ce chapitre l'énonce précisément, en tire les premières
conséquences --- la \omterm{thm:b1:reals:archimedes}{propriété d'Archimède}, la \omterm{thm:b1:reals:floor}{fonction partie
entière}, la densité des rationnels et des irrationnels --- et met en
place le vocabulaire (sup, inf, max, min) constamment utilisé à
partir du \cref{ch:b1:seq}.

\section{La propriété de la borne supérieure}

\begin{definition}[Bornes, sup et inf]\label{def:b1:reals:bounds}
Soit $A \subseteq \R$ une partie non vide. Un réel $M$ est un
\emph{majorant}\index{majorant} de $A$ lorsque $a \leq M$ pour tout
$a \in A$~; $A$ est \emph{majorée} lorsqu'elle admet un majorant (de
même en dessous, avec les minorants~; \emph{bornée} signifie les
deux). Un \emph{maximum} de $A$ est un majorant appartenant à $A$.

La \emph{borne supérieure} $\sup A$\index{borne supérieure} est le
plus petit des majorants de $A$, lorsqu'il existe~; la \emph{borne
inférieure} $\inf A$\index{borne inférieure} est le plus grand des
minorants.
\end{definition}

\begin{theorem}[Axiome de complétude de $\R$]\label{thm:b1:reals:sup}
$\R$ est un \omterm{def:b1:structures:field}{corps} totalement ordonné contenant $\Q$ dans lequel
\emph{toute partie non vide et majorée admet une \omterm{def:b1:reals:bounds}{borne
supérieure}}.\index{complétude de R@complétude de $\R$}
\end{theorem}
\admitted

\begin{remark}
Nous le prenons comme axiome définissant $\R$~; construire un modèle
(par les coupures de Dedekind ou par les suites de Cauchy de
rationnels) et démontrer son unicité est honnête mais long, et nous
le laissons pour plus tard. Remarquons que $\Q$ ne possède pas cette
propriété~: $\{x \in \Q : x^2 < 2\}$ est majorée mais n'a pas de plus
petit \omterm{def:b1:reals:bounds}{majorant} \emph{dans $\Q$} --- son candidat, $\sqrt 2$, manque à
l'appel (\cref{ex:b1:logic:sqrt2}). En passant aux opposés
($\sup(-A) = -\inf A$), toute partie non vide et minorée admet une
\omterm{def:b1:reals:bounds}{borne inférieure}.
\end{remark}

\begin{proposition}[La caractérisation par $\varepsilon$]\label{prop:b1:reals:epsilon}
Soient $A \neq \emptyset$ une partie majorée et $s \in \R$. Alors $s =
\sup A$ si et seulement si
\begin{enumerate}
  \item $s$ est un \omterm{def:b1:reals:bounds}{majorant}~: $\forall a \in A$, $a \leq s$~; et
  \item rien de plus petit ne l'est~: $\forall \varepsilon > 0$,
        $\exists a \in A$, $a > s - \varepsilon$.
\end{enumerate}
\end{proposition}

\begin{proof}
Si $s = \sup A$~: (1) vaut par définition, et pour (2), $s -
\varepsilon < s$ n'est pas un \omterm{def:b1:reals:bounds}{majorant}, ce qui est exactement
l'existence d'un $a > s - \varepsilon$. Réciproquement, (1) dit que
$s$ est un \omterm{def:b1:reals:bounds}{majorant}~; (2) dit qu'aucun $t < s$ n'est un \omterm{def:b1:reals:bounds}{majorant}
(prendre $\varepsilon = s - t$)~: $s$ est le plus petit.
\end{proof}

\begin{example}\label{ex:b1:reals:supexamples}
$\sup \intoo{0}{1} = 1$, non atteinte (pas de maximum)~;
$\sup \intcc{0}{1} = 1 = \max$. Pour $A = \{1 - \frac 1n : n \in
\N^*\}$~: $\sup A = 1$, non atteinte~; $\inf A = \min A = 0$. Un
maximum, lorsqu'il existe, est la \omterm{def:b1:reals:bounds}{borne supérieure}~; tout l'intérêt
de $\sup$ est de disposer d'un substitut quand le maximum n'existe
pas.
\end{example}

\begin{example}[Calculer des bornes supérieures en pratique]\label{ex:b1:reals:supcompute}
Deux mises en pratique complètes de la
\cref{prop:b1:reals:epsilon}.

\emph{L'\omterm{def:b1:logic:sets}{ensemble} $A = \{x + \frac1x : x > 0\}$.} Pour tout $x > 0$,
$x + \frac1x - 2 = \frac{(\,\sqrt x - 1/\sqrt x\,)^2}{1} \geq 0$,
donc $2$ est un minorant~; et $2 = 1 + \frac11 \in A$~: par
conséquent $\inf A = \min A = 2$, atteinte en $x = 1$. Au-dessus,
$A$ n'est pas majorée ($x + \frac1x > x$ peut dépasser tout $M$
d'après le \cref{thm:b1:reals:archimedes})~: $\sup A$ n'existe pas
dans $\R$ (elle vaut $+\infty$ dans $\overline\R$).

\emph{L'\omterm{def:b1:logic:sets}{ensemble} $B = \bigl\{\frac{m}{m + n} : m, n \in
\N^*\bigr\}$.} Tout élément appartient à $\intoo{0}{1}$, donc $0$
et $1$ encadrent $B$. Aucun des deux n'est atteint~:
$\frac{m}{m+n} = 1$ forcerait $n = 0$. Pour la \omterm{def:b1:reals:bounds}{borne supérieure},
figeons $n = 1$ et laissons croître $m$~:
$\frac{m}{m+1} = 1 - \frac{1}{m+1} > 1 - \varepsilon$ dès que
$m + 1 > \frac1\varepsilon$ (Archimède)~: $\sup B = 1$.
Symétriquement ($m = 1$, $n$ grand), $\inf B = 0$. L'idée à
retenir~: pour épingler une \omterm{def:b1:reals:bounds}{borne supérieure}, un seul
\emph{chemin à un paramètre} bien choisi à l'intérieur de
l'\omterm{def:b1:logic:sets}{ensemble} suffit --- ici le chemin $n = 1$ --- et la
caractérisation par $\varepsilon$ n'en demande pas davantage.
\end{example}

\begin{example}[Le miroir de la borne inférieure]\label{ex:b1:reals:infmirror}
La \omterm{def:b1:reals:bounds}{borne inférieure} possède sa propre caractérisation par
$\varepsilon$, obtenue à partir de la
\cref{prop:b1:reals:epsilon} via $\inf A = -\sup(-A)$~:
$i = \inf A$ si et seulement si $i$ minore $A$ et si, pour tout
$\varepsilon > 0$, un certain $a \in A$ vérifie $a < i +
\varepsilon$. Une mise en pratique avec les deux bornes à la
fois~: posons
\[
A = \Bigl\{(-1)^n + \frac1n : n \in \N^*\Bigr\}
= \Bigl\{0,\ \tfrac32,\ -\tfrac23,\ \tfrac54,\ -\tfrac45,\
\dots\Bigr\} .
\]
Les indices pairs donnent $1 + \frac1n \leq \frac32$, avec
égalité en $n = 2$~: comme de plus les valeurs d'indice impair
sont $\leq 0 < \frac32$, on obtient $\sup A = \max A =
\frac32$. Les indices impairs donnent $-1 + \frac1n > -1$,
décroissant vers $-1$~: tout élément de $A$ est $> -1$, et
$-1 + \varepsilon$ est dépassé par $-1 + \frac1n$ pour $n$
impair $> \frac1\varepsilon$~: $\inf A = -1$, non atteinte. Un
seul \omterm{def:b1:logic:sets}{ensemble}, et les quatre comportements exposés~: une \omterm{def:b1:reals:bounds}{borne
supérieure} qui est un maximum, une \omterm{def:b1:reals:bounds}{borne inférieure} qui n'est pas
un minimum.
\end{example}

\begin{remark}[Pièges courants avec sup et inf]
Quatre erreurs expliquent l'essentiel des points perdus. (i)
\emph{Confondre $\sup$ et $\max$}~: $\sup A$ n'appartient pas
nécessairement à $A$~; n'écrire $\max$ qu'après avoir exhibé un
élément de $A$ qui soit un \omterm{def:b1:reals:bounds}{majorant}. (ii) \emph{Passer des
inégalités strictes à la \omterm{def:b1:reals:bounds}{borne supérieure}}~: si $a < b$ pour
tout $a \in A$, on ne peut conclure que $\sup A \leq b$ ---
témoin $A = \intoo{0}{1}$, $b = 1$. (iii) \emph{Écrire $\sup A$
avant d'en avoir le droit}~: le symbole exige $A$ non vide et
majorée (\cref{met:b1:reals:supproofs})~; $\sup \emptyset$
et $\sup \N$ ne sont pas définis dans $\R$ (les conventions de
$\overline\R$ relèvent d'un acte séparé et explicite). (iv)
\emph{Opérations ensemblistes}~: $\sup(A \cup B) = \max(\sup A,
\sup B)$ toujours, mais rien de général ne vaut pour $A \cap B$
--- il peut être vide, et même lorsqu'il ne l'est pas,
$\sup(A \cap B)$ peut être très en dessous de
$\min(\sup A, \sup B)$~: prendre $A = \{0, 2\}$ et $B = \{0, 3\}$,
où $\sup(A \cap B) = 0$.
\end{remark}

\begin{example}[Les ensembles finis ont un maximum --- un lemme utilisé sans le dire]\label{ex:b1:reals:finitemax}
Toute partie finie non vide $F \subseteq \R$ admet un maximum (et
un minimum). Par récurrence sur le nombre d'éléments~: un
singleton $\{a\}$ a pour $\max$ la valeur $a$~; si l'\omterm{def:b1:logic:statement}{assertion}
vaut pour les \omterm{def:b1:logic:sets}{ensembles} à $n$ éléments et si $F$ en a $n + 1$,
choisissons $a \in F$~: l'\omterm{def:b1:logic:sets}{ensemble} $F \setminus \{a\}$ a un
maximum $m$, et $\max F$ vaut $m$ si $a \leq m$, et $a$ sinon.
Aucune complétude n'intervient --- c'est de l'ordre pur plus une
récurrence, déjà valable dans $\Q$ --- mais le lemme mérite un
énoncé honnête, car les démonstrations à venir l'invoquent sans le
dire~: la construction de la \omterm{thm:b1:reals:floor}{partie entière} ci-dessous
(«~un \omterm{def:b1:logic:sets}{ensemble} d'entiers piégé dans un \omterm{prop:b1:reals:intervals}{intervalle} borné a un plus
grand élément~»), toute majoration $\max(\abs{u_0}, \dots,
\abs{u_{N-1}}, \dots)$ du \cref{ch:b1:seq}, tout «~prenons le plus
grand des $\delta$ en nombre fini~» du \cref{ch:b1:continuity}.
C'est dans les \omterm{def:b1:logic:sets}{ensembles} infinis que les maxima meurent et que les
\omterm{def:b1:reals:bounds}{bornes supérieures} prennent le relais~: ce chapitre existe pour le
cas infini.
\end{example}

\begin{theorem}[Propriété d'Archimède]\label{thm:b1:reals:archimedes}
Pour tout $x \in \R$ il existe $n \in \N$ avec $n > x$. De façon
équivalente~: pour tous $\varepsilon > 0$ et $y > 0$, un multiple
$n\varepsilon$ dépasse $y$.\index{propriété d'Archimède}
\end{theorem}

\begin{proof}
Supposons le contraire~: un certain $x$ majore $\N$, de sorte que
$s = \sup \N$ existe (\cref{thm:b1:reals:sup}). D'après la
\cref{prop:b1:reals:epsilon}~(2) avec $\varepsilon = 1$, il existe $n
\in \N$ avec $n > s - 1$~; mais alors $n + 1 \in \N$ et $n + 1 > s$,
ce qui contredit le fait que $s$ soit un \omterm{def:b1:reals:bounds}{majorant}. Pour la seconde
forme, soient $\varepsilon > 0$ et $y > 0$~: la première forme
appliquée à $x = \frac{y}{\varepsilon}$ produit $n \in \N$ avec
\[
n > \frac{y}{\varepsilon},
\]
et la multiplication par $\varepsilon > 0$ (qui préserve les
inégalités strictes) donne $n\varepsilon > y$.
Réciproquement, la seconde forme avec $\varepsilon = 1$ et $y = x$
redonne la première pour $x > 0$, et $n = 1$ règle le cas $x \leq 0$~:
les deux énoncés sont strictement équivalents.
\end{proof}

\begin{example}[Archimède à l'œuvre]\label{ex:b1:reals:archimedeswork}
Trois usages immédiats, constamment nécessaires par la suite. (i)
\emph{Aucun réel strictement positif n'est en dessous de tous les
$\frac1n$}~: si $0 < \varepsilon$, choisissons $n >
\frac1\varepsilon$~; alors $\frac1n < \varepsilon$. Autrement dit,
$\R$ ne contient pas d'infiniment petits --- l'énoncé informel
«~$\frac1n$ devient arbitrairement petit~» est exactement ce
théorème. (ii) \emph{Seuils explicites}~: quelle taille doit avoir
$n$ pour que $\frac{1}{n^2} \leq 10^{-6}$~? Il suffit que $n \geq
10^3$ --- Archimède garantit que de tels $n$ existent, et l'algèbre
les localise. (iii) \emph{Les puissances battent toute borne}~:
$2^n \geq n + 1$ (récurrence), donc pour tout $M$ une puissance de
$2$ dépasse $M$~: c'est la croissance géométrique utilisée pour les
dyadiques dans l'\cref{exo:b1:reals:8}. L'idée à retenir~: la
\omterm{thm:b1:reals:archimedes}{propriété d'Archimède} est ce qui autorise toute phrase de la forme
«~prenons $n$ assez grand~» --- désormais nous userons librement de
cette formule, et cet exemple en est la justification, faite une
fois pour toutes.
\end{example}

\begin{theorem}[Fonction partie entière]\label{thm:b1:reals:floor}
Pour tout $x \in \R$ il existe exactement un entier, la \emph{partie
entière}\index{fonction partie entière} $\lfloor x \rfloor$, tel que
\[
\lfloor x \rfloor \leq x < \lfloor x \rfloor + 1 .
\]
\end{theorem}

\begin{proof}
\emph{Existence.} L'\omterm{def:b1:logic:sets}{ensemble} $E = \{k \in \Z : k \leq x\}$ est non
vide~: d'après le \cref{thm:b1:reals:archimedes} il existe $m \in \N$
avec $m > -x$, et alors $-m < x$, donc $-m \in E$. Il est majoré (par
tout entier $n > x$, qui existe pour la même raison), donc, étant un
\omterm{def:b1:logic:sets}{ensemble} d'entiers piégé dans l'\omterm{prop:b1:reals:intervals}{intervalle} fini $\intint{-m}{n}$, il
possède un plus grand élément $k = \max E$. Alors $k \leq x$, et $k +
1 \notin E$ signifie $x < k + 1$.

\emph{Unicité.} Si $k$ et $k'$ vérifient tous deux les inégalités,
alors $k \leq x < k' + 1$ donne $k \leq k'$, et symétriquement $k'
\leq k$.
\end{proof}

\begin{example}[La partie entière en pratique]\label{ex:b1:reals:floorpractice}
$\lfloor 3.7 \rfloor = 3$, $\lfloor 5 \rfloor = 5$, et $\lfloor
-3.7 \rfloor = -4$~: la \omterm{thm:b1:reals:floor}{partie entière} va vers le \emph{bas}, et
non vers $0$. Deux conséquences de l'unicité dans le
\cref{thm:b1:reals:floor} que nous utiliserons sans le dire.
D'abord, pour $n \in \Z$,
\[
\lfloor x + n \rfloor = \lfloor x \rfloor + n ,
\]
car $\lfloor x \rfloor + n$ est un entier vérifiant les deux
inégalités qui définissent celle de $x + n$ --- et un seul entier
le fait. Ensuite, $\lfloor \, \cdot \, \rfloor$ est croissante au
sens large~: si $x \leq y$ alors $\lfloor x \rfloor \leq x \leq y <
\lfloor y \rfloor + 1$, et un entier $< \lfloor y \rfloor + 1$ est
$\leq \lfloor y \rfloor$. Attention toutefois~: $\lfloor 2x \rfloor
\neq 2\lfloor x \rfloor$ en général~; $x = 0.6$ donne $\lfloor 1.2
\rfloor = 1 \neq 0 = 2\lfloor 0.6 \rfloor$.

Ce qui \emph{est} vrai est une identité à garder en réserve (celle
d'Hermite, dans son cas le plus simple)~: pour tout réel $x$,
\[
\lfloor x \rfloor + \Bigl\lfloor x + \frac12 \Bigr\rfloor
= \lfloor 2x \rfloor .
\]
Écrivons $x = \lfloor x\rfloor + u$ avec $u \in \intco{0}{1}$ et
séparons deux cas. Si $u < \frac12$~: le membre de gauche vaut
$\lfloor x\rfloor + \lfloor x\rfloor = 2\lfloor x\rfloor$, et $2x =
2\lfloor x\rfloor + 2u$ avec $2u \in \intco{0}{1}$, donc le membre
de droite vaut $2\lfloor x\rfloor$ également. Si $u \geq \frac12$~:
le membre de gauche vaut $\lfloor x\rfloor + (\lfloor x\rfloor +
1)$, et $2u \in \intco{1}{2}$ fait valoir $2\lfloor x\rfloor + 1$ au
membre de droite. L'idée à retenir~: $\lfloor x + \frac12\rfloor$
est l'\emph{arrondi} de $x$ à l'entier le plus proche, de sorte que
l'identité dit que \omterm{thm:b1:reals:floor}{partie entière} plus arrondi égale \omterm{thm:b1:reals:floor}{partie entière}
du double --- et la disjonction de cas sur la partie fractionnaire
$u$ est la technique standard derrière toute identité de \omterm{thm:b1:reals:floor}{partie
entière} (les \cref{exo:b1:reals:2,exo:b1:reals:3} en vivent aussi).
\end{example}

\begin{theorem}[Densité de $\Q$ et de $\R \setminus \Q$]\label{thm:b1:reals:density}
Entre deux réels quelconques $x < y$ se trouvent un rationnel et un
irrationnel.\index{densité}
\end{theorem}

\begin{proof}
\emph{Un rationnel.} D'après le \cref{thm:b1:reals:archimedes},
choisissons $n \in \N^*$ avec $n > \frac{1}{y - x}$, de sorte que $ny
- nx > 1$. Posons $m = \lfloor nx \rfloor + 1$. D'un côté, $nx <
\lfloor nx \rfloor + 1 = m$ (\cref{thm:b1:reals:floor})~; de l'autre,
$m = \lfloor nx \rfloor + 1 \leq nx + 1 < ny$. En divisant par $n$~:
$x < \frac mn < y$.

\emph{Un irrationnel.} Appliquons le point précédent au couple $x -
\sqrt 2 < y - \sqrt 2$~: un rationnel $q$ se trouve entre eux, et
alors $q + \sqrt 2 \in \intoo{x}{y}$ est irrationnel (si $q + \sqrt
2$ était rationnel, $\sqrt 2$ le serait aussi).
\end{proof}

\begin{example}[La démonstration de la densité, exécutée]\label{ex:b1:reals:densityrun}
La démonstration est un algorithme~; exécutons-le sur $x = 1.414$
et $y = \sqrt 2$. Comme $1.4142^2 = 1.99996164 < 2$, on a $\sqrt 2
> 1.4142$, donc $y - x > 0.0002$ et $\frac{1}{y - x} < 5000$~: le
choix $n = 5000$ est légitime. Alors $nx = 7070$, donc $m = \lfloor
7070 \rfloor + 1 = 7071$, et le rationnel produit est
\[
\frac{m}{n} = \frac{7071}{5000} = 1.4142,
\qquad
1.414 < 1.4142 < \sqrt 2 .
\]
L'idée à retenir~: la démonstration ne demande qu'un $n$ à peine
plus grand que $\frac{1}{y-x}$, et renvoie le premier multiple de
$\frac 1n$ au-delà de $x$. La densité n'est pas un miracle abstrait
--- c'est une division euclidienne déguisée, thème longuement
développé dans le devoir maison (\cref{pb:b1:reals:1}).
\end{example}

\begin{remark}[Où la complétude servira ensuite]
Le \cref{thm:b1:reals:sup} est l'unique axiome non algébrique de ce
livre, et tout théorème d'existence de l'analyse n'est que cet
axiome habillé autrement~: le théorème de la limite monotone
(\cref{ch:b1:seq}), le théorème de Bolzano--Weierstrass
(\cref{ch:b1:topology}), le théorème des valeurs intermédiaires et
celui des bornes atteintes (\cref{ch:b1:continuity}), et la
définition même de l'intégrale comme \omterm{def:b1:reals:bounds}{borne supérieure} des sommes
inférieures (\cref{ch:b1:integration}). Le volume de Licence 3
construit la théorie de la mesure et les espaces de Hilbert sur ce
même unique axiome. Lorsqu'une démonstration des chapitres à venir
produit un nombre réel sorti de nulle part, cherchez la \omterm{def:b1:reals:bounds}{borne
supérieure} cachée.
\end{remark}

\begin{remark}[Entre discrétude et densité]
$\Z$ et $\Q$ occupent des positions opposées à l'intérieur de
$\R$~: autour de chaque entier il y a un trou de longueur $1$ ne
contenant aucun autre entier (la discrétude --- c'est elle qui rend
la \omterm{thm:b1:reals:floor}{partie entière} bien définie), tandis qu'entre deux réels
quelconques se trouvent une infinité de rationnels (la densité).
Remarquablement, pour les \emph{\omterm{def:b1:structures:subgroup}{sous-groupes} additifs} de $\R$ il
n'y a rien entre les deux~: l'\cref{exo:b1:reals:9} démontre qu'un
tel \omterm{def:b1:structures:subgroup}{sous-groupe} est soit de la forme $\alpha\Z$ (discret), soit
dense --- une dichotomie qui gouverne la densité de $\{\sin n\}$ au
\cref{ch:b1:seq} et le monstre constructif du
\cref{pb:b1:continuity:1}. Les \omterm{def:b1:logic:sets}{ensembles} généraux, bien sûr, mêlent
librement les deux comportements~: $\Z \cup \Q\cap\intcc{0}{1}$ est
discret au loin et dense au milieu.
\end{remark}

\begin{method}[Démontrer des égalités avec sup et inf]\label{met:b1:reals:supproofs}
Pour démontrer $\sup A = s$~: vérifier que $s$ majore $A$, puis
produire, pour chaque $\varepsilon > 0$ (ou pour une suite
$\varepsilon = \frac 1n$), un élément de $A$ au-dessus de $s -
\varepsilon$. Pour comparer des \omterm{def:b1:reals:bounds}{bornes supérieures}, utiliser~: $A
\subseteq B \implies \sup A \leq \sup B$~; et pour tous $a, b$~:
$\sup(A + B) = \sup A + \sup B$, où $A + B = \{a + b\}$
(\cref{exo:b1:reals:5}). Ne jamais écrire $\sup A$ avant de savoir
que $A$ est non vide et majorée.
\end{method}

\section{Intervalles}

\begin{proposition}[Caractérisation des intervalles]\label{prop:b1:reals:intervals}
Une partie $I \subseteq \R$ est un \emph{intervalle}\index{intervalle}
(l'un des types familiers $\intoo{a}{b}$, $\intcc{a}{b}$,
$\intco{a}{b}$, $\intoc{a}{b}$, demi-droites, $\R$, $\emptyset$,
singletons) si et seulement si elle est \emph{convexe}~:
\[
\forall x, y \in I,\ \forall z \in \R, \quad
x \leq z \leq y \implies z \in I .
\]
\end{proposition}

\begin{proof}
Chacun des types listés est manifestement convexe. Réciproquement,
soit $I$ convexe et non vide. Posons $a = \inf I$ si $I$ est minorée,
et $a = -\infty$ sinon~; de même $b = \sup I$ ou $+\infty$. Montrons
que $\intoo{a}{b} \subseteq I \subseteq \intcc{a}{b}$ (avec les
conventions évidentes en $\pm\infty$). La seconde inclusion est la
définition des bornes. Pour la première, soit $z \in \intoo{a}{b}$~:
comme $z > a$, $z$ n'est pas un minorant (ou $a = -\infty$), donc un
certain $x \in I$ vérifie $x < z$~; de même un certain $y \in I$
vérifie $y > z$~; la convexité place $z \in I$.

Il reste à lire le type sur la double inclusion
$\intoo{a}{b} \subseteq I \subseteq \intcc{a}{b}$~: les \omterm{def:b1:logic:sets}{ensembles}
coincés entre un \omterm{prop:b1:reals:intervals}{intervalle} ouvert et son adhérence ne diffèrent
de $\intoo{a}{b}$ que par la présence ou l'absence des extrémités
(finies). Explicitement~: si $a, b \in \R$, les quatre
possibilités pour $(a \in I,\ b \in I)$ donnent $\intoo{a}{b}$,
$\intco{a}{b}$, $\intoc{a}{b}$, $\intcc{a}{b}$ (y compris les cas
dégénérés $a = b$~: un singleton si $a \in I$)~; si $a = -\infty$
et $b \in \R$, on obtient $\intoo{-\infty}{b}$ ou
$\intoc{-\infty}{b}$~; symétriquement pour $a \in \R$, $b =
+\infty$~; et $a = -\infty$, $b = +\infty$ donne $I = \R$. Chaque
cas figure dans la liste~: c'est fini.
\end{proof}

\begin{remark}[Pourquoi la convexité est le bon critère]
La proposition convertit une définition \emph{géométrique} (une
liste de dix formes) en un test \emph{logique d'une ligne}, et
c'est ce test que l'on utilise réellement~: pour démontrer qu'un
\omterm{def:b1:logic:sets}{ensemble} est un \omterm{prop:b1:reals:intervals}{intervalle}, ne jamais courir après celle des dix
formes qui convient --- vérifier la convexité et laisser la
proposition trancher le type. Le théorème des valeurs
intermédiaires du \cref{ch:b1:continuity} sera énoncé exactement
ainsi («~l'image continue d'un \omterm{prop:b1:reals:intervals}{intervalle} est un \omterm{prop:b1:reals:intervals}{intervalle}~»),
et sa démonstration produit la convexité, non la forme.
\end{remark}

\begin{remark}[Droite réelle achevée]
Il est commode d'adjoindre deux symboles et de travailler dans
$\overline\R = \R \cup \{-\infty, +\infty\}$, avec les conventions
$\sup A = +\infty$ lorsque $A$ n'est pas majorée et $\sup \emptyset
= -\infty$. Alors \emph{toute} partie de $\R$ admet une \omterm{def:b1:reals:bounds}{borne
supérieure} dans $\overline\R$ --- un confort de notation dont on use
librement pour les limites au \cref{ch:b1:seq}.
\end{remark}

\begin{example}[Calculer dans $\overline\R$]\label{ex:b1:reals:extendedcompute}
Avec ces conventions en vigueur~: $\sup \Z = +\infty$, $\inf \Z =
-\infty$~; pour $A = \{n + (-1)^n n : n \in \N\} = \{0, 4, 0, 8,
\dots\} \cup \{0\}$, $\sup A = +\infty$ (les termes pairs $2n$ ne
sont pas majorés) et $\inf A = \min A = 0$~; et $\sup\emptyset =
-\infty \leq \inf\emptyset = +\infty$ --- le seul \omterm{def:b1:logic:sets}{ensemble} dont la
\omterm{def:b1:reals:bounds}{borne supérieure} soit \emph{plus petite} que sa \omterm{def:b1:reals:bounds}{borne inférieure},
rappel que les conventions sont choisies pour rendre $\sup$
croissante et $\inf$ décroissante vis-à-vis de l'inclusion~:
\[
A \subseteq B \implies \sup A \leq \sup B
\quad\text{et}\quad \inf A \geq \inf B ,
\]
désormais valable sans réserve de non-vacuité. Ce que les
conventions ne fournissent \emph{pas}, c'est une arithmétique~:
$+\infty + (-\infty)$ et $0 \times (+\infty)$ restent indéfinis, et
toute manipulation algébrique de \omterm{def:b1:reals:bounds}{bornes supérieures} doit d'abord
vérifier qu'elle ne les forme jamais. La droite achevée est une
comptabilité, pas un système de nombres.
\end{example}

\begin{example}[La borne supérieure qui s'est échappée de $\Q$]\label{ex:b1:reals:supescape}
Revenons à l'\omterm{def:b1:logic:sets}{ensemble} de la remarque initiale, $A = \{x \in \Q :
x^2 < 2\}$, et calculons sa \omterm{def:b1:reals:bounds}{borne supérieure} \emph{dans $\R$}. Il
est non vide ($1 \in A$) et majoré par $1.5$ (si $x > 1.5$ alors
$x^2 > 2.25 > 2$), donc $s = \sup A$ existe. Montrons que $s =
\sqrt 2$ (le réel construit dans l'\cref{exo:b1:reals:12}).
\omterm{def:b1:reals:bounds}{Majorant}~: tout $a \in A$ vérifie $a < \sqrt2$ --- pour $a \leq 0$
c'est clair, et pour $a > 0$, $a \geq \sqrt2$ donnerait $a^2 \geq
2$. Rien de plus petit ne convient~: étant donné $t < \sqrt2$, la
densité (\cref{thm:b1:reals:density}) fournit un rationnel $q$ avec
$\max(1, t) < q < \sqrt 2$, et alors $q^2 < 2$, donc $q \in A$
dépasse $t$. D'après la \cref{prop:b1:reals:epsilon}, $s = \sqrt2
\notin \Q$. L'idée à retenir~: la \omterm{def:b1:reals:bounds}{borne supérieure} d'un \omterm{def:b1:logic:sets}{ensemble} de
rationnels n'est pas nécessairement rationnelle --- la complétude
est précisément la promesse que $\R$, contrairement à $\Q$, ne
laisse jamais une \omterm{def:b1:reals:bounds}{borne supérieure} s'échapper~; cet exemple est la
remarque initiale du chapitre, désormais démontrée et non plus
seulement montrée du doigt.
\end{example}

\begin{remark}[Perspectives dans ce volume]
Les trois outils de ce chapitre ont des carrières distinctes
devant eux. La \omterm{def:b1:reals:bounds}{borne supérieure} mène la moitié «~analyse~»~:
limites monotones (\cref{ch:b1:seq}), définition même de
l'intégrale (\cref{ch:b1:integration}), et, dans la géométrie du
\cref{ch:b1:euclid}, la distance d'un point à un sous-espace ---
une \omterm{def:b1:reals:bounds}{borne inférieure} que la projection orthogonale transforme en
minimum. La \omterm{thm:b1:reals:floor}{fonction partie entière} revient partout où le discret
rencontre le continu~: développements en base $b$ (le devoir maison
de ce chapitre), l'approximation de Dirichlet par les tiroirs
(\cref{pb:b1:derivative:1}), les comparaisons intégrales de sommes
(\cref{ch:b1:series}). Les arguments de densité deviendront une
méthode au \cref{ch:b1:continuity}~: une identité entre fonctions
continues n'a besoin d'être vérifiée que sur $\Q$ --- la moitié de
l'équation fonctionnelle de Cauchy (\cref{pb:b1:continuity:1}) est
exactement ce mouvement. Lorsqu'on se demande d'où une
démonstration de ce volume tire ses énoncés d'existence, la réponse
est presque toujours~: de ce chapitre.
\end{remark}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:b1:reals:1}
Déterminer (avec démonstrations) sup, inf, max, min --- lorsqu'ils
existent --- de~:
\[
A = \Bigl\{\frac{1}{n} : n \in \N^*\Bigr\},
\qquad
B = \Bigl\{\frac{(-1)^n n}{n+1} : n \in \N\Bigr\},
\qquad
C = \{x \in \R : x^2 < 3\}.
\]
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:reals:2}
Démontrer que pour tous $x, y \in \R$~: $\lfloor x \rfloor + \lfloor y
\rfloor \leq \lfloor x + y \rfloor \leq \lfloor x \rfloor + \lfloor y
\rfloor + 1$, et que les deux bornes sont atteintes.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:reals:3}
Démontrer que pour tout $x \in \R$ et tout $n \in \N^*$~:
$\Bigl\lfloor \frac{\lfloor nx \rfloor}{n} \Bigr\rfloor = \lfloor x
\rfloor$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:reals:4}
Soient $A \subseteq B$ des parties non vides de $\R$, $B$ bornée.
Démontrer $\inf B \leq \inf A \leq \sup A \leq \sup B$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:reals:5}
Pour $A, B \subseteq \R$ non vides et bornées, on pose $A + B = \{a +
b : a \in A,\ b \in B\}$ et $-A = \{-a : a \in A\}$. Démontrer~:
\[
\sup(A + B) = \sup A + \sup B,
\qquad
\sup(-A) = -\inf A .
\]
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:reals:6}
Soient $f, g \colon E \to \R$ des fonctions bornées. Démontrer
\[
\sup_{x \in E}\, \bigl(f(x) + g(x)\bigr) \leq \sup_{x \in E} f(x) +
\sup_{x \in E} g(x),
\]
et donner un exemple où l'inégalité est stricte. Pourquoi cela ne
contredit-il pas l'\cref{exo:b1:reals:5}~?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:reals:7}
Démontrer que $\sqrt 2 + \sqrt 3$ est irrationnel. \emph{(L'élever au
carré et utiliser l'irrationalité de $\sqrt 6$, à démontrer via
l'\cref{exo:b1:arith:7}.)}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:reals:8}
Démontrer que l'\omterm{def:b1:logic:sets}{ensemble} $D = \bigl\{\frac{m}{2^n} : m \in \Z,\ n \in
\N\bigr\}$ des rationnels dyadiques est dense dans $\R$~: entre deux
réels quelconques se trouve un rationnel dyadique.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:reals:9}
Soit $G$ un \omterm{def:b1:structures:subgroup}{sous-groupe} de $(\R, +)$ avec $G \neq \{0\}$. Posons
$\alpha = \inf\,(G \cap \intoo{0}{+\infty})$. Démontrer~:
\begin{enumerate}
  \item si $\alpha > 0$, alors $G = \alpha\Z$~;
  \item si $\alpha = 0$, alors $G$ est dense dans $\R$.
\end{enumerate}
En déduire que $\Z + \sqrt 2\,\Z$ est dense dans $\R$.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:reals:10}
Pour $A, B$ des \omterm{def:b1:logic:sets}{ensembles} non vides de réels strictement positifs, on
pose $AB = \{ab : a \in A, b \in B\}$. Démontrer $\sup(AB) = \sup A
\cdot \sup B$ (cas borné), et montrer sur un exemple que la
positivité est essentielle.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:reals:11}
Pour $A \subseteq \R$ non vide et bornée, on définit le
\emph{diamètre}
\[
\operatorname{diam} A = \sup\,\{\abs{a - a'} : a, a' \in A\} .
\]
Démontrer que $\operatorname{diam} A = \sup A - \inf A$, et que
$\intcc{\inf A}{\sup A}$ est le plus petit \omterm{prop:b1:reals:intervals}{intervalle} fermé
contenant $A$.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:reals:12}
Soient $y > 0$ et $E = \{x \geq 0 : x^2 \leq y\}$. Démontrer que $E$
est non vide et majoré, et que $s = \sup E$ vérifie $s^2 = y$
\emph{(écarter $s^2 < y$ et $s^2 > y$ en exhibant, dans chaque cas,
un petit $h > 0$ contredisant la définition de la \omterm{def:b1:reals:bounds}{borne
supérieure})}. En déduire que tout $y > 0$ possède une unique racine
carrée $\sqrt y > 0$ et que $y \mapsto \sqrt y$ est croissante sur
$\intoo{0}{+\infty}$.
\end{exercise}

\section{Problème~: développements en base $b$ et le rythme des rationnels}

\begin{problem}[{Devoir maison --- développements en base $b$~:
existence, unicité, et la périodicité caractérise $\Q$}]
\label{pb:b1:reals:1}
Tout réel de $\intco{0}{1}$ possède un développement en chiffres dans
toute base $b \geq 2$~; le développement est unique dès que l'on
proscrit les suites finissant par une infinité de chiffres $b - 1$~;
et il est périodique à partir d'un certain rang exactement lorsque le
nombre est rationnel. Ce problème démontre ces trois faits à partir du
seul axiome de complétude --- pas de suites, pas de séries~: rien que
la \omterm{def:b1:reals:bounds}{borne supérieure}, la \omterm{thm:b1:reals:archimedes}{propriété d'Archimède} et la \omterm{thm:b1:reals:floor}{fonction partie
entière} --- et se clôt sur l'argument diagonal de Cantor en version
chiffres. Dans tout ce qui suit, $b \geq 2$ est un entier fixé (la
\emph{base}), un \emph{chiffre} est un élément de $\intint{0}{b-1}$,
et une suite de chiffres $(d_n)_{n \geq 1}$ est \emph{propre}
lorsqu'elle n'est pas égale à $b - 1$ à partir d'un certain rang
(c'est-à-dire~: pour tout $N$ il existe $n > N$ avec $d_n \leq b -
2$).\index{développement en base b@développement en base $b$}\index{développement décimal}

\medskip
\noindent\textbf{Partie I --- Les chiffres à la main.} Division de $p$
par $q$ en base $b$~: multiplier le reste courant par $b$, diviser par
$q$, noter le quotient comme chiffre suivant, garder le reste.
\begin{enumerate}
  \item En base $10$, exécuter l'algorithme sur $\frac 18$ et sur
        $\frac 17$, en notant à chaque étape le chiffre \emph{et}
        le reste. Vérifier que les restes de $\frac 17$ parcourent
        le cycle $1, 3, 2, 6, 4, 5$ et que les chiffres $142857$
        se répètent ensuite indéfiniment.
  \item Calculer les développements en base $2$ de $\frac 13$ et de
        $\frac{5}{16}$, et le développement en base $3$ de $\frac
        12$. Observer~: l'un des nombres a un développement fini,
        les deux autres se répètent --- et $\frac 12$, si docile en
        base $10$, se répète indéfiniment en base $3$.
  \item Pour $x = \frac pq \in \intco{0}{1}$ sous forme
        irréductible, montrer que les chiffres produits par
        l'algorithme sont tous nuls à partir d'un certain rang si
        et seulement si le reste $b^N p \bmod q$ s'annule pour un
        certain $N$, si et seulement si $q$ \omterm{def:b1:arith:divides}{divise} une puissance
        $b^N$, si et seulement si tout facteur premier de $q$
        \omterm{def:b1:arith:divides}{divise} $b$. Vérifier~: $\frac{1}{20}$ a un développement
        fini en base $10$, mais pas en base $3$.
  \item On définit la \emph{troncature} $s_n = \lfloor b^n x \rfloor
        / b^n$. Pour $x = \sqrt 2$ et $b = 10$, calculer $s_0,
        \dots, s_4$ en vérifiant à chaque étape que deux carrés
        consécutifs encadrent $2$ (par exemple $1.4142^2
        = 1.99996164 < 2 < 2.00024449 = 1.4143^2$), et contrôler
        chaque fois que $s_n \leq \sqrt 2 < s_n + 10^{-n}$.
\end{enumerate}

\medskip
\noindent\textbf{Partie II --- L'existence, par la \omterm{def:b1:reals:bounds}{borne
supérieure}.} Fixons $x \in \intco{0}{1}$ et posons $A_n = \lfloor b^n
x \rfloor$ et $d_n = A_n - b\,A_{n-1}$ pour $n \geq 1$.
\begin{enumerate}[resume]
  \item Montrer que $A_0 = 0$ et $b\,A_{n-1} \leq A_n \leq
        b\,A_{n-1} + b - 1$~; en conclure que chaque $d_n$ est un
        chiffre.
  \item Montrer que $s_n := A_n b^{-n}$ vérifie
        \[
        s_n = \sum_{k=1}^{n} d_k\,b^{-k}
        \qquad\text{et}\qquad
        s_n \leq x < s_n + b^{-n} .
        \]
  \item Démontrer $b^n \geq n + 1$ par récurrence, puis montrer que
        $(s_n)$ est croissante au sens large et que $x = \sup_n s_n$
        \emph{(utiliser la \cref{prop:b1:reals:epsilon} et le
        \cref{thm:b1:reals:archimedes})}.
  \item Montrer que la suite $(d_n)$ est propre~: si $d_k = b - 1$
        pour tout $k > N$, calculer $s_n$ pour $n > N$ par une somme
        géométrique finie et contredire la question 6.
  \item Réciproquement, soit $(e_n)_{n \geq 1}$ une suite de chiffres
        propre quelconque et $t_n = \sum_{k=1}^n e_k b^{-k}$. Montrer
        que $y = \sup_n t_n$ existe, appartient à $\intco{0}{1}$, et
        vérifie $t_n \leq y < t_n + b^{-n}$ pour tout $n$
        \emph{(pour l'inégalité stricte, utiliser un chiffre $e_m \leq
        b - 2$ avec $m > n$)}. En déduire $\lfloor b^n y \rfloor = b^n
        t_n$, puis que les chiffres de $y$, au sens de la question 5,
        sont exactement les $e_n$.
\end{enumerate}

\medskip
\noindent\textbf{Partie III --- Unicité, ordre, décalage.}
\begin{enumerate}[resume]
  \item Assembler les questions 5 à 9 en le \emph{théorème du
        développement en base $b$}~: les \omterm{def:b1:logic:map}{applications} $x \mapsto
        (d_n)$ et $(e_n) \mapsto \sup_n t_n$ sont des \omterm{def:b1:logic:inj}{bijections}
        réciproques l'une de l'autre entre $\intco{0}{1}$ et
        l'\omterm{def:b1:logic:sets}{ensemble} des suites de chiffres propres. En particulier,
        deux suites propres distinctes n'ont jamais la même valeur.
  \item Autorisons maintenant les suites impropres. Montrer qu'une
        suite avec $e_n = b - 1$ pour tout $n > M$ (avec $M \geq 0$
        minimal) a pour valeur $t_M + b^{-M}$~; en conclure que
        $0.999\dots = 1$ en base $10$, et que les réels ayant deux
        représentations en chiffres sont exactement les fractions
        $b$-adiques $m/b^N \in \intoo{0}{1}$ --- tout autre réel
        n'en a qu'une, même parmi les suites impropres.
  \item Démontrer que la \omterm{def:b1:logic:inj}{bijection} de la question 10 respecte
        l'ordre lexicographique~: si les suites propres de $x$ et de
        $y$ diffèrent pour la première fois à l'indice $m$, alors
        $x < y$ si et seulement si $d_m < e_m$.
  \item (Lemme de décalage) Soit $x \in \intco{0}{1}$ de chiffres
        $(d_n)$. Montrer que la partie fractionnaire de $bx$ a pour
        chiffres $(d_{n+1})_{n \geq 1}$ \emph{(calculer $\lfloor
        b^n(bx - A_1)\rfloor$ à l'aide de $\lfloor u - K \rfloor =
        \lfloor u \rfloor - K$ pour $K$ entier)}, et en déduire par
        récurrence que la partie fractionnaire de $b^m x$ a pour
        chiffres $(d_{n+m})_{n \geq 1}$.
\end{enumerate}

\medskip
\noindent\textbf{Partie IV --- Être rationnel, c'est être
périodique.} Soient $x = \frac pq \in \intco{0}{1}$ sous forme
irréductible et $r_n = b^n p \bmod q$ le reste de la division
euclidienne de $b^n p$ par $q$.
\begin{enumerate}[resume]
  \item Montrer que $A_n = \dfrac{b^n p - r_n}{q}$ et $r_n =
        (b\,r_{n-1}) \bmod q$.
  \item Montrer que $d_n = \Bigl\lfloor \dfrac{b\,r_{n-1}}{q}
        \Bigr\rfloor$~: chaque chiffre est une fonction du seul reste
        précédent. C'est exactement la division posée de la
        partie I.
  \item Appliquer le \omterm{cor:b1:counting:pigeonhole}{principe des tiroirs}
        (\cref{cor:b1:counting:pigeonhole}) à $r_0, \dots, r_q$
        et en conclure~: le développement de tout rationnel est
        périodique à partir d'un certain rang, avec prépériode et
        période au plus $q$.
  \item Réciproquement, supposons les chiffres de $y \in
        \intco{0}{1}$ \emph{purement} périodiques~: $d_{n+T} = d_n$
        pour tout $n \geq 1$. À l'aide du lemme de décalage et de
        l'unicité de la question 10, montrer que la partie
        fractionnaire de $b^T y$ vaut $y$, et en déduire $(b^T -
        1)\,y \in \N$~: ainsi $y$ est rationnel, de dénominateur
        divisant $b^T - 1$. Vérifier le mécanisme sur
        $0.(142857)$~: $142857 \times 7 = 999999$.
  \item Traiter le cas périodique à partir d'un certain rang par
        décalage, et énoncer le \emph{critère de périodicité}~: $x
        \in \intco{0}{1}$ est rationnel si et seulement si son
        développement propre en base $b$ est périodique à partir d'un
        certain rang --- dans une base si et seulement si dans
        toutes.
  \item Pour $x = \frac 1q$ avec $\gcd(q, b) = 1$, montrer que le
        développement est purement périodique et que sa plus petite
        période est le plus petit $T \geq 1$ tel que $b^T \equiv 1
        \pmod q$ (l'ordre multiplicatif de $b$ modulo $q$). Vérifier
        que pour $q = 7$, $b = 10$ les puissances de $10$ modulo $7$
        parcourent $3, 2, 6, 4, 5, 1$~: ordre $6$, ce qui rejoint la
        question 1.
\end{enumerate}

\medskip
\noindent\textbf{Partie V --- Dividendes et diagonale.}
\begin{enumerate}[resume]
  \item Soit $x^*$ le réel de $\intco{0}{1}$ dont les chiffres en
        base $10$ valent $1$ aux positions triangulaires $\frac{j(j
        + 1)}{2}$ ($j \geq 1$) et $0$ ailleurs~: $x^* =
        0.101001000100001\dots$ Montrer que sa suite de chiffres est
        propre mais non périodique à partir d'un certain rang
        \emph{(une période $T$ forcerait des $1$ séparés par des
        écarts au plus $T$, or les écarts croissent)}, et en
        conclure que $x^*$ est irrationnel~: un nombre dont
        l'irrationalité se lit dans le seul rythme.
  \item Montrer que pour toute base $b$ l'\omterm{def:b1:logic:sets}{ensemble} $\{m/b^n : m \in
        \Z, n \in \N\}$ est dense dans $\R$ (généralisant
        l'\cref{exo:b1:reals:8}), et que tout rationnel $\frac pq
        \in \intoo{0}{1}$ a un développement \emph{fini} en base
        $q$. Morale~: être fini est une propriété du couple
        (nombre, base)~; la périodicité --- la rationalité --- est
        intrinsèque.
  \item (Diagonale de Cantor) Soit $k \mapsto x_k$ une \omterm{def:b1:logic:map}{application}
        quelconque de $\N^*$ dans $\intco{0}{1}$. Définissons la
        suite de chiffres $e_k = 1$ si le $k$-ième chiffre de $x_k$
        est différent de $1$, et $e_k = 2$ sinon. Montrer que
        $(e_k)$ est propre, que sa valeur $y$ appartient à
        $\intco{0}{1}$, et que $y \neq x_k$ pour tout $k$. Conclure~:
        aucune \omterm{def:b1:logic:map}{application} $\N^* \to \intco{0}{1}$ n'est \omterm{def:b1:logic:inj}{surjective}.
        (Le vocabulaire de la dénombrabilité, et le vrai foyer de ce
        théorème, c'est le \cref{ch:b1:topology}.)
  \item Montrer que si les développements propres de $x$ et de $y$
        coïncident jusqu'à l'indice $n$ alors $\abs{x - y} < b^{-n}$,
        et réfuter la réciproque avec $x = 0.1$, $y = 0.0999$ en base
        $10$~: la proximité des nombres ne force pas la coïncidence
        des chiffres. Qui sont les coupables~?
  \item Exécuter la partie IV sur $x = \frac{1}{10}$ en base $b =
        2$~: calculer les restes et les chiffres jusqu'au cycle, et
        conclure $\frac{1}{10} = (0.0\overline{0011})_2$, avec
        prépériode $1$ et période $4$. Expliquer, via la question 3,
        pourquoi aucune chaîne binaire finie ne vaudra jamais
        $\frac{1}{10}$ --- la raison pour laquelle le $0.1 + 0.2$
        d'un ordinateur en virgule flottante ne vaut pas exactement
        $0.3$.
  \item Synthèse. En une phrase chacun~: où la démonstration a-t-elle
        utilisé (i) la complétude, (ii) la \omterm{thm:b1:reals:archimedes}{propriété d'Archimède},
        (iii) la clause d'unicité de la \omterm{thm:b1:reals:floor}{partie entière}, (iv) le
        \omterm{cor:b1:counting:pigeonhole}{principe des tiroirs}~? Et la morale~: $\intco{0}{1}$ est
        fidèlement codé par les suites de chiffres propres, la
        rationalité s'y lit comme une périodicité --- et pourtant
        l'analyse préfère la \omterm{def:b1:reals:bounds}{borne supérieure} aux chiffres.
        Pourquoi~? (Penser à l'addition de deux suites de chiffres.)
\end{enumerate}
\end{problem}
