\chapter{Équations différentielles linéaires}\label{ch:b1:diffeq}

Rencontrées pour la première fois dans le volume de Terminale, les
équations différentielles sont traitées ici avec des démonstrations
complètes et dans une plus grande généralité~: équations linéaires du
premier ordre à coefficients variables (entièrement résolues par la
méthode de \omterm{thm:b1:diffeq:voc}{variation de la constante}) et équations linéaires du second
ordre à coefficients constants, le modèle des oscillations. Les deux
cas présentent la même structure~: \emph{solution générale $=$ une
solution particulière $+$ solution générale de l'équation homogène}.

\section{Équations linéaires du premier ordre}

\begin{definition}\label{def:b1:diffeq:linear1}
Soient $I$ un intervalle et $a, b \colon I \to \R$ (ou $\C$) deux
fonctions continues. L'équation
\[
(E)\colon\quad y' + a(x)\,y = b(x),
\]
d'inconnue la fonction dérivable $y \colon I \to \R$ (ou $\C$), est
une \emph{équation différentielle linéaire du premier
ordre}\index{équation différentielle!linéaire du premier ordre}.
L'équation $(H)\colon y' + a(x) y = 0$ en est l'équation
\emph{homogène}.
\end{definition}

\begin{theorem}[Résolution de l'équation homogène]\label{thm:b1:diffeq:homogeneous1}
Soit $A$ une primitive de $a$ sur $I$ (elle existe~:
\cref{ch:b1:integration}). Les solutions de $(H)$ sur $I$ sont
exactement les fonctions
\[
y(x) = \lambda\, \eu^{-A(x)}, \qquad \lambda \in \R \text{ (ou } \C).
\]
\end{theorem}

\begin{proof}
Ces fonctions sont solutions~: $y' = -\lambda A' \eu^{-A} = -a y$.
Réciproquement, soit $y$ une solution de $(H)$~; posons $z(x) = y(x)\,
\eu^{A(x)}$. Alors
\[
z' = y' \eu^{A} + y\, a\, \eu^{A} = (y' + ay)\, \eu^{A} = 0 ,
\]
donc $z$ est constante sur l'intervalle $I$, disons $z = \lambda$~: $y
= \lambda \eu^{-A}$. (Notons la logique~: aucune solution n'est
perdue, car \emph{toute} solution a été écrite sous la forme
annoncée.)
\end{proof}

\begin{example}[Une équation homogène à coefficient variable]\label{ex:b1:diffeq:varcoeff}
Résolvons $y' + (\cos x)\,y = 0$ sur $\R$. Une primitive de $a(x) =
\cos x$ est $A(x) = \sin x$, donc les solutions sont
\[
y(x) = \lambda\,\eu^{-\sin x}, \qquad \lambda \in \R .
\]
Deux lectures. Toute solution est périodique (de période $2\pi$) et ne
s'annule jamais sauf si $\lambda = 0$ --- le signe de $\lambda$ est
pour toujours le signe de $y$, puisqu'une exponentielle ne peut
traverser zéro. Et la solution passant par $y(0) = y_0$ est
$y_0\eu^{-\sin x}$~: exactement une courbe de la famille par chaque
point initial, l'image en dimension un du
\cref{thm:b1:diffeq:voc}~(2).
\end{example}

\begin{theorem}[Variation de la constante~; problème de Cauchy]\label{thm:b1:diffeq:voc}
Avec les notations ci-dessus~:
\begin{enumerate}
  \item Les solutions de $(E)$ sur $I$ sont exactement
        \[
        y(x) = \Bigl(\lambda + \int_{x_0}^{x} b(t)\, \eu^{A(t)} \dd t
        \Bigr)\, \eu^{-A(x)},
        \qquad \lambda \in \R,
        \]
        où $x_0 \in I$ est fixé. Autrement dit~: solution générale de
        $(H)$ plus une solution particulière de $(E)$.
  \item Pour tout $x_0 \in I$ et tout $y_0$, le \emph{problème de
        Cauchy}\index{problème de Cauchy} «~$(E)$ et $y(x_0) = y_0$~»
        admet exactement une solution sur $I$.
\end{enumerate}
\end{theorem}

\begin{proof}
(1) Suivant la méthode dite de \emph{variation de la
constante}\index{variation de la constante}, cherchons les solutions
sous la forme $y = \mu(x)\, \eu^{-A(x)}$ avec $\mu$ dérivable --- on
ne perd aucune généralité, puisque toute fonction sur $I$ peut
s'écrire ainsi ($\mu = y\,\eu^{A}$). En substituant,
\[
y' + ay = \mu' \eu^{-A} - \mu a \eu^{-A} + a\mu\eu^{-A}
= \mu'\, \eu^{-A},
\]
donc $y$ est solution de $(E)$ si et seulement si $\mu'(x) =
b(x)\,\eu^{A(x)}$, si et seulement si $\mu(x) = \lambda + \int_{x_0}^x
b(t)\eu^{A(t)}\dd t$ pour une certaine constante $\lambda$ (deux
primitives d'une même fonction continue sur un intervalle diffèrent
d'une constante).

(2) Dans la formule, $y(x_0) = \lambda\,\eu^{-A(x_0)}$~: la condition
$y(x_0) = y_0$ détermine $\lambda = y_0 \eu^{A(x_0)}$ de manière
unique.
\end{proof}

\begin{example}\label{ex:b1:diffeq:order1}
Résolvons $y' + \dfrac{y}{x} = x^2$ sur $I = \intoo{0}{+\infty}$. Ici
$a(x) = \frac 1x$, $A(x) = \ln x$, $\eu^{-A(x)} = \frac 1x$. Solutions
homogènes~: $\frac{\lambda}{x}$. \omterm{thm:b1:diffeq:voc}{Variation de la constante}~:
$\mu'(x) = x^2 \cdot x = x^3$, donc $\mu = \frac{x^4}{4} + \lambda$, et
\[
y(x) = \frac{x^3}{4} + \frac{\lambda}{x}, \qquad \lambda \in \R .
\]
Avec la condition initiale $y(1) = 0$~: $\lambda = -\frac14$.
\emph{Vérification~:} $y' + \frac yx = \frac{3x^2}{4} -
\frac{\lambda}{x^2} + \frac{x^2}{4} + \frac{\lambda}{x^2} = x^2$.
\end{example}

\begin{example}[Deviner vaut mieux qu'intégrer]\label{ex:b1:diffeq:guess}
Résolvons $y' + 2x\,y = x$ sur $\R$. La \omterm{thm:b1:diffeq:voc}{variation de la constante}
fonctionne ($A = x^2$, $\mu' = x\,\eu^{x^2}$, $\mu = \frac12\eu^{x^2}
+ \lambda$), mais remarquer que la \emph{constante} $y_p = \frac12$
est solution de l'équation ($0 + 2x\cdot\frac12 = x$) est plus rapide.
Avec les solutions homogènes $\lambda\,\eu^{-x^2}$~:
\[
y(x) = \frac12 + \lambda\,\eu^{-x^2}, \qquad \lambda \in \R .
\]
Toute solution converge vers $\frac12$ extrêmement vite quand $x \to
\pm\infty$~: la solution particulière constante est un
\emph{équilibre} que toutes les solutions rejoignent. L'idée à
retenir~: avant de lancer la méthode générale, consacrez dix secondes
à chercher une solution particulière évidente (constante, monôme,
multiple du second membre)~; le théorème de structure achève ensuite
le travail.
\end{example}

\begin{remark}[Les intervalles comptent]
Le théorème vit sur un \emph{intervalle} où $a$ et $b$ sont continues.
Pour $y' + \frac yx = 0$ sur $\R^*$, les solutions sont
$\frac{\lambda}{x}$ sur $\intoo{0}{+\infty}$ et $\frac{\mu}{x}$ sur
$\intoo{-\infty}{0}$ avec des constantes \emph{indépendantes}~: il n'y
a aucune raison qu'une formule unique recolle à travers la singularité
en $0$.
\end{remark}

\begin{example}[Un second membre complexe, deux réponses réelles]\label{ex:b1:diffeq:complexrhs}
Résolvons $y' - y = \cos x$ et $y' - y = \sin x$ d'un seul coup.
Travaillons dans $\C$ avec le second membre $\eu^{\iu x}$~: en
essayant $y_p = c\,\eu^{\iu x}$, on obtient $c(\iu - 1)\eu^{\iu x} =
\eu^{\iu x}$, donc
\[
c = \frac1{\iu - 1} = \frac{-1 - \iu}2,
\qquad
y_p = -\frac{(1 + \iu)(\cos x + \iu\sin x)}2
= \frac{\sin x - \cos x}2 + \iu\,\frac{-\sin x - \cos x}2 .
\]
Comme l'équation est à coefficients réels, les parties réelle et
imaginaire se séparent~: $\frac{\sin x - \cos x}2$ est solution de
$y' - y = \cos x$, et $-\frac{\sin x + \cos x}2$ est solution de
$y' - y = \sin x$ (vérifions la première~: sa dérivée $\frac{\cos x +
\sin x}2$, moins la fonction, donne $\cos x$). Une seule ligne
complexe a remplacé deux passages par la \omterm{thm:b1:diffeq:voc}{variation de la constante} ---
la même économie que la \cref{met:b1:diffeq:particular} systématise au
second ordre, et un dividende récurrent du \cref{ch:b1:complex}.
\end{example}

\section{Équations linéaires du second ordre à coefficients constants}

\begin{definition}\label{def:b1:diffeq:linear2}
Soient $a, b \in \R$ et $f \colon I \to \R$ continue. L'équation
\[
(E)\colon\quad y'' + a\,y' + b\,y = f(x)
\]
est une \emph{équation différentielle linéaire du second ordre à
coefficients constants}\index{équation différentielle!linéaire du
second ordre}~; $(H)\colon y'' + ay' + by = 0$ en est l'équation
homogène, et $\chi(r) = r^2 + ar + b$ son \emph{polynôme
caractéristique}\index{polynôme caractéristique}.
\end{definition}

\begin{theorem}[Solutions homogènes]\label{thm:b1:diffeq:homogeneous2}
Soit $\Delta = a^2 - 4b$ le discriminant de $\chi$. Les solutions
réelles de $(H)$ sur $\R$ sont~:
\begin{enumerate}
  \item si $\Delta > 0$, avec $r_1 \neq r_2$ les deux racines réelles~:
        $\;y = \lambda\, \eu^{r_1 x} + \mu\, \eu^{r_2 x}$~;
  \item si $\Delta = 0$, avec $r_0$ la racine double~:
        $\;y = (\lambda + \mu x)\, \eu^{r_0 x}$~;
  \item si $\Delta < 0$, avec les racines $\alpha \pm \iu\omega$
        ($\omega > 0$)~:
        $\;y = \eu^{\alpha x} (\lambda \cos\omega x + \mu
        \sin\omega x)$~;
\end{enumerate}
dans chaque cas, $(\lambda, \mu)$ décrivant $\R^2$.
\end{theorem}

\begin{proof}
Observons d'abord que pour $r \in \C$, $x \mapsto \eu^{rx}$ est
solution de $(H)$ si et seulement si $\chi(r) = 0$ (substituons~:
$(r^2 + ar + b)\eu^{rx} = 0$). C'est pourquoi les exponentielles sont
le premier essai naturel~: la dérivation agit sur $\eu^{rx}$ comme la
multiplication par le nombre $r$, de sorte que l'équation
différentielle devient l'équation numérique $\chi(r) = 0$ --- tout le
problème analytique est comprimé dans la recherche des racines d'un
trinôme.

L'étape clé est un changement d'inconnue qui abaisse l'ordre. Soit $r$
une racine (éventuellement complexe) de $\chi$ et écrivons $y = z\,
\eu^{rx}$, ce qui ne perd aucune généralité. Alors
\[
y'' + ay' + by
= \bigl(z'' + (2r + a) z' + \chi(r) z\bigr)\eu^{rx}
= \bigl(z'' + (2r + a) z'\bigr)\eu^{rx},
\]
donc $(H)$ devient l'équation \emph{du premier ordre} $u' + (2r + a) u
= 0$ pour $u = z'$.

\emph{Cas $\Delta \neq 0$~:} choisissons $r = r_1$~; alors $2r_1 + a =
r_1 - r_2$ (puisque $r_1 + r_2 = -a$). D'après
le \cref{thm:b1:diffeq:homogeneous1}, $z' = c\,\eu^{(r_2 - r_1)x}$ pour
une certaine constante $c$~; en intégrant sur $\R$, $z = \mu\,
\eu^{(r_2 - r_1)x} + \lambda$ avec $\mu = \frac{c}{r_2 - r_1}$, et
donc $y = z\, \eu^{r_1 x} = \lambda\,\eu^{r_1x} + \mu\,\eu^{r_2x}$.
Lorsque $\Delta < 0$, les racines sont $\alpha \pm \iu\omega$ et les
solutions complexes sont $y = c_1\eu^{(\alpha+\iu\omega)x} +
c_2\eu^{(\alpha-\iu\omega)x}$ avec $c_1, c_2 \in \C$. Lesquelles sont
à valeurs réelles~? Comme $\conj{\eu^{(\alpha+\iu\omega)x}} =
\eu^{(\alpha-\iu\omega)x}$, le \omterm{def:b1:complex:field}{conjugué} de $y$ est $\conj{c_2}\,
\eu^{(\alpha+\iu\omega)x} + \conj{c_1}\,\eu^{(\alpha-\iu\omega)x}$, et
$y = \conj y$ pour tout $x$ impose $c_2 = \conj{c_1}$ (les deux
exponentielles sont linéairement indépendantes~: évaluons en deux
points, ou comparons en $x=0$ après division par $\eu^{\alpha x}$). En
écrivant $c_1 = \frac{\lambda - \iu\mu}2$ avec $\lambda, \mu$ réels~:
\[
y = 2\,\Re\Bigl(c_1\,\eu^{\alpha x}(\cos\omega x +
\iu\sin\omega x)\Bigr)
= \eu^{\alpha x}\bigl(\lambda\cos\omega x + \mu\sin\omega
x\bigr),
\]
et réciproquement toute fonction de cette forme est solution (partie
réelle d'une solution complexe d'une équation réelle)~: l'espace des
solutions réelles est bien celui annoncé.

\emph{Cas $\Delta = 0$~:} $r = r_0$, $2r_0 + a = 0$, donc $z'' = 0$~:
$z = \lambda + \mu x$ et $y = (\lambda + \mu x)\eu^{r_0 x}$.
\end{proof}

\begin{example}[Un problème de Cauchy, du début à la fin]\label{ex:b1:diffeq:cauchy2}
Résolvons $y'' - 3y' + 2y = 0$ avec $y(0) = 0$, $y'(0) = 1$. Le
\omterm{def:b1:diffeq:linear2}{polynôme caractéristique} $r^2 - 3r + 2 = (r - 1)(r - 2)$ a les racines
réelles $1$ et $2$~: solution générale $y = \lambda\eu^{x} +
\mu\eu^{2x}$. Les deux conditions donnent le système linéaire
\[
\lambda + \mu = 0, \qquad \lambda + 2\mu = 1 ,
\]
donc $\mu = 1$, $\lambda = -1$~:
\[
y(x) = \eu^{2x} - \eu^{x} .
\]
Vérification~: $y(0) = 0$~; $y' = 2\eu^{2x} - \eu^x$ vérifie $y'(0) =
1$~; et $y'' - 3y' + 2y = (4 - 6 + 2)\eu^{2x} + (-1 + 3 - 2)\eu^x =
0$. Notons la forme de la réponse~: au voisinage de $-\infty$ c'est le
mode lent $-\eu^x$ qui domine~; au voisinage de $+\infty$, le mode
rapide $\eu^{2x}$. Lire les solutions comme des superpositions de
modes ayant des taux de décroissance ou de croissance différents est
l'habitude payante --- c'est ainsi qu'est organisée la séparation
transitoire/régime permanent du devoir maison.
\end{example}

\begin{theorem}[Structure et problème de Cauchy]\label{thm:b1:diffeq:structure2}
\begin{enumerate}
  \item Si $y_p$ est une solution particulière de $(E)$, les solutions
        de $(E)$ sont exactement les $y_p + y_h$, $y_h$ décrivant les
        solutions de $(H)$.
  \item (Superposition) Si $y_1$ est solution de $y'' + ay' + by = f_1$
        et $y_2$ solution de $y'' + ay' + by = f_2$, alors $y_1 + y_2$
        est solution de l'équation de second membre $f_1 + f_2$.
  \item Pour tout $x_0 \in I$ et tout $(y_0, y_0')$, le \omterm{thm:b1:diffeq:voc}{problème de
        Cauchy} «~$(E)$, $y(x_0) = y_0$, $y'(x_0) = y_0'$~» admet
        exactement une solution sur $I$. \emph{(Existence sous réserve
        d'une solution particulière~; unicité sans réserve.)}
\end{enumerate}
\end{theorem}

\begin{proof}
(1) $y$ est solution de $(E)$ si et seulement si $y - y_p$ est
solution de $(H)$, par linéarité de $y \mapsto y'' + ay' + by$. (2)
relève de la même linéarité.

(3) D'après (1), il suffit de montrer que les constantes $(\lambda,
\mu)$ peuvent toujours être ajustées, de manière unique, à n'importe
quelles données $(y_0, y_0')$. Quitte à translater la variable,
supposons $x_0 = 0$. Dans le cas (1) du
\cref{thm:b1:diffeq:homogeneous2}, $y = \lambda\eu^{r_1x} +
\mu\eu^{r_2x}$ donne
\[
y(0) = \lambda + \mu, \qquad y'(0) = r_1\lambda + r_2\mu :
\]
un système linéaire en $(\lambda, \mu)$ dont le déterminant vaut $r_2
- r_1 \neq 0$~; en le résolvant explicitement, $\mu = \frac{y_0' -
r_1y_0}{r_2 - r_1}$ et $\lambda = y_0 - \mu$~: exactement une
solution. Dans le cas (2), $y(0) = \lambda$ et $y'(0) = r_0\lambda +
\mu$~: le système est triangulaire, de déterminant $1$, résolu par
$\lambda = y_0$, $\mu = y_0' - r_0y_0$. Dans le cas (3), $y(0) =
\lambda$ et $y'(0) = \alpha\lambda + \omega\mu$~: déterminant $\omega
\neq 0$, résolu par $\lambda = y_0$, $\mu = \frac{y_0' - \alpha
y_0}\omega$. Dans chaque cas, l'\omterm{def:b1:logic:map}{application} $(\lambda, \mu) \mapsto
(y(x_0), y'(x_0))$ est une \omterm{def:b1:logic:inj}{bijection} linéaire --- le langage du
\cref{ch:b1:linmaps} comprimera cette discussion par cas en une seule
phrase.
\end{proof}

\begin{method}[Solution particulière pour $f(x) = P(x)\,\eu^{\gamma x}$]\label{met:b1:diffeq:particular}
Lorsque le second membre est $P(x)\,\eu^{\gamma x}$ avec $P$ un
polynôme et $\gamma \in \R$ (ce qui couvre les polynômes, les
exponentielles, et, via un $\gamma$ complexe ou la superposition, le
$\cos$ et le $\sin$)~: cherchons une solution particulière de la forme
\[
y_p(x) = x^{m}\, Q(x)\, \eu^{\gamma x},
\qquad
m = \text{multiplicité de } \gamma \text{ comme racine de } \chi
\ (m = 0, 1 \text{ ou } 2),
\]
avec $Q$ un polynôme de même degré que $P$, dont les coefficients
s'obtiennent par substitution et identification. Pour $f = K\cos\omega
x$ (ou $\sin$), résolvons avec le second membre $K\eu^{\iu\omega x}$
et prenons la partie réelle (resp.\ imaginaire).
\end{method}

\begin{example}[La superposition à l'œuvre]\label{ex:b1:diffeq:superposition}
Résolvons $y'' - y = \eu^{x} + 4$ sur $\R$. Équation homogène~:
$\chi(r) = r^2 - 1$, racines $\pm1$, donc $y_h = \lambda\eu^x +
\mu\eu^{-x}$. Séparons le second membre et traitons chaque morceau par
l'encadré de méthode. \emph{Morceau $\eu^x$~:} ici $\gamma = 1$ est
racine simple de $\chi$, essayons donc $y_1 = c\,x\,\eu^x$~: alors
$y_1'' - y_1 = c(x + 2)\eu^x - cx\eu^x = 2c\,\eu^x$, d'où $c =
\frac12$. \emph{Morceau $4$~:} $\gamma = 0$ n'est pas racine~; la
constante $y_2 = -4$ convient. Par superposition
(\cref{thm:b1:diffeq:structure2}~(2))~:
\[
y = \frac{x\,\eu^x}2 - 4 + \lambda\,\eu^x + \mu\,\eu^{-x},
\qquad (\lambda, \mu) \in \R^2 .
\]
Notons que les deux morceaux ont exigé des formes \emph{différentes}
($m = 1$ contre $m = 0$)~: le test de multiplicité s'applique à chaque
exposant séparément, ce qui est tout l'intérêt de séparer le second
membre avant de deviner.
\end{example}

\begin{example}[La règle de multiplicité à l'œuvre]\label{ex:b1:diffeq:multiplicity}
Résolvons $y'' + y' = x$ sur $\R$. Le second membre est
$P(x)\eu^{0\cdot x}$ avec $P(x) = x$, et $\gamma = 0$ est racine
\emph{simple} de $\chi(r) = r^2 + r = r(r + 1)$~: donc $m = 1$, et le
bon essai est $y_p = x\,(\alpha x + \beta) = \alpha x^2 + \beta x$, de
degré un de plus que $P$. En substituant~:
\[
y_p'' + y_p' = 2\alpha + (2\alpha x + \beta)
= 2\alpha x + (2\alpha + \beta) ,
\]
et l'identification avec $x$ donne $\alpha = \frac12$, $\beta = -1$~:
$y_p = \frac{x^2}2 - x$. Solution générale~: $y = \frac{x^2}2 - x +
\lambda + \mu\,\eu^{-x}$. Si nous avions essayé $y_p = \alpha x +
\beta$ (en ignorant la multiplicité), la substitution aurait donné
$y_p'' + y_p' = \alpha$, une constante --- aucun choix de $\alpha,
\beta$ ne peut égaler $x$, et l'échec est structurel~: les constantes
sont déjà solutions de l'équation homogène, donc elles sont invisibles
pour le membre de gauche. Le facteur $x^m$ existe précisément pour
sortir de l'espace des solutions homogènes.
\end{example}

\begin{remark}[L'assurance à trente secondes]\label{rem:b1:diffeq:check}
Chaque équation résolue dans ce chapitre se termine par une
vérification par substitution, et ce n'est pas décoratif. Un calcul
d'équation différentielle enchaîne beaucoup de petites étapes (une
primitive, une dérivée de produit, deux constantes), et une seule
erreur de signe se propage invisiblement~; substituer la formule
finale dans l'équation les attrape presque toutes, au prix d'une seule
dérivation. Cultivez le réflexe sur trois niveaux~: vérifiez la
\emph{solution particulière} seule (la partie homogène s'annule de
toute façon), vérifiez les \emph{conditions initiales} sur la solution
complète, et lorsqu'un paramètre est présent, vérifiez une \emph{valeur
dégénérée} (la formule pour $\Omega$ général redonne-t-elle la réponse
connue en $\Omega = 0$~?). L'habitude coûte une demi-minute~; elle
convertit «~probablement juste~» en «~vérifié~».
\end{remark}

\begin{remark}[Pièges classiques]\label{rem:b1:diffeq:pitfalls}
\begin{enumerate}
  \item \emph{Normaliser d'abord.} Les formules supposent que
        l'équation s'écrit $y' + a(x)y = b(x)$ --- coefficient $1$
        devant $y'$. Pour $xy' - 2y = x^3$, divisez par $x$ (sur un
        intervalle évitant $0$) avant d'identifier $a$ et $b$, comme
        dans l'\cref{exo:b1:diffeq:2}.
  \item \emph{Une constante par dimension, fixée à la fin.} La
        solution générale du premier ordre porte une constante, celle
        du second ordre deux~; les conditions initiales s'imposent sur
        la solution \emph{complète} $y_p + y_h$, jamais sur $y_h$ seule
        --- les imposer avant d'ajouter $y_p$ est l'erreur structurelle
        la plus fréquente.
  \item \emph{Attention à la multiplicité.} Un essai de solution
        particulière qui est solution de l'équation homogène est
        invisible pour le membre de gauche~; le facteur $x^m$ de
        la \cref{met:b1:diffeq:particular} n'est pas optionnel
        (\cref{ex:b1:diffeq:multiplicity}).
  \item \emph{Les intervalles font partie de la réponse.} Les
        solutions vivent sur des intervalles où les coefficients sont
        continus~; recoller à travers une singularité peut créer des
        constantes parasites (\cref{exo:b1:diffeq:12}) ou détruire
        l'unicité. «~Résoudre sur $\R^*$~» signifie deux problèmes
        indépendants.
\end{enumerate}
\end{remark}

\begin{example}[Excitation hors résonance]\label{ex:b1:diffeq:offresonance}
Résolvons $y'' + 4y = \sin x$ sur $\R$. Pulsation propre $2$,
pulsation d'excitation $1$~: puisque $\iu$ n'est \emph{pas} racine de
$\chi(r) = r^2 + 4$, la multiplicité est $m = 0$ et une simple
sinusoïde suffit. En essayant $y_p = \alpha\sin x$ (aucun cosinus
n'est nécessaire~: l'équation n'a pas de terme en $y'$, et le $\sin$
régénère le $\sin$)~:
\[
y_p'' + 4y_p = -\alpha\sin x + 4\alpha\sin x = 3\alpha\sin x ,
\]
donc $\alpha = \frac13$ et la solution générale est
\[
y = \frac{\sin x}3 + \lambda\cos 2x + \mu\sin 2x .
\]
Toute solution reste bornée~: c'est une superposition de deux
oscillations, aux pulsations $1$ (forcée) et $2$ (propre). À comparer
avec l'exemple suivant, où une excitation \emph{à} la pulsation propre
change la forme même de la réponse.
\end{example}

\begin{example}[Une oscillation forcée]\label{ex:b1:diffeq:oscillation}
Résolvons $y'' + y = \cos x$, $y(0) = 0$, $y'(0) = 0$.

\emph{Équation homogène~:} $\chi(r) = r^2 + 1$, racines $\pm\iu$~:
$y_h = \lambda\cos x + \mu \sin x$.

\emph{Solution particulière~:} second membre $\Re(\eu^{\iu x})$ avec
$\gamma = \iu$ racine simple de $\chi$~: essayons $z_p = c\, x\,
\eu^{\iu x}$ ($c \in \C$). Alors $z_p'' + z_p = c\,(2\iu)\eu^{\iu x}$,
qui vaut $\eu^{\iu x}$ pour $c = \frac{1}{2\iu} = -\frac\iu2$. Donc
$z_p = -\frac{\iu}{2} x (\cos x + \iu \sin x)$ et $y_p = \Re(z_p) =
\frac{x \sin x}{2}$.

\emph{Solution générale~:} $y = \frac{x\sin x}{2} + \lambda\cos x +
\mu\sin x$. Conditions~: $y(0) = \lambda = 0$~; $y' = \frac{\sin x +
x\cos x}{2} + \mu\cos x$, donc $y'(0) = \mu = 0$. Réponse~: $y =
\frac{x\sin x}{2}$ --- une oscillation dont l'amplitude croît
linéairement~: le phénomène de \emph{résonance}\index{résonance},
causé par l'excitation du système à sa pulsation propre.
\end{example}

\begin{remark}[Interlude~: la linéarité est une géométrie]\label{rem:b1:diffeq:interlude}
Reprenons la forme de chaque \omterm{def:b1:logic:sets}{ensemble} de solutions de ce chapitre~:
une solution particulière plus un espace de solutions homogènes avec
une constante libre (premier ordre) ou deux (second ordre). Les
chapitres d'algèbre linéaire
(\cref{ch:b1:vspaces,ch:b1:findim,ch:b1:linmaps}) en fourniront le
vocabulaire exact~: l'\omterm{def:b1:logic:map}{application} $L(y) = y'' + ay' + by$ est
\emph{linéaire}, ses solutions homogènes forment le \emph{noyau} de
$L$, un espace vectoriel dont la \emph{dimension} est égale à l'ordre
de l'équation --- c'est là le contenu honnête de «~une constante par
ordre~» --- et l'\omterm{def:b1:logic:sets}{ensemble} des solutions de $L(y) = f$ est un
\emph{sous-espace affine}, un translaté du noyau. Même l'\omterm{def:b1:logic:map}{application} de
Cauchy $(\lambda, \mu) \mapsto (y(x_0), y'(x_0))$ du
\cref{thm:b1:diffeq:structure2} est une \omterm{def:b1:logic:inj}{bijection} linéaire entre deux
plans, c'est-à-dire un système $2 \times 2$ inversible
(\cref{ch:b1:matrices}). Rien de ce chapitre ne sera à refaire --- il
suffira de le renommer, et ce renommage est le meilleur échauffement
possible pour l'algèbre linéaire~: chaque définition abstraite y a
déjà gagné sa vie ici.
\end{remark}

\begin{remark}[Où ce chapitre est utilisé]\label{rem:b1:diffeq:whereused}
Le théorème de structure --- les solutions de $(E)$ sont «~une
solution particulière plus les solutions de $(H)$~» --- est la
première apparition d'un schéma que les \cref{ch:b1:vspaces,ch:b1:linmaps}
nommeront~: l'\omterm{def:b1:logic:sets}{ensemble} des solutions de $(H)$ est le \emph{noyau} de
l'\omterm{def:b1:logic:map}{application} linéaire $y \mapsto y'' + ay' + by$, et l'\omterm{def:b1:logic:sets}{ensemble} des
solutions de $(E)$ en est un translaté affine. Le \omterm{def:b1:diffeq:linear2}{polynôme
caractéristique} réapparaît comme \omterm{def:b1:diffeq:linear2}{polynôme caractéristique} d'une matrice
au \cref{ch:b1:matrices}~: une équation du second ordre est un système
$2 \times 2$ du premier ordre déguisé, point de vue que le volume de
Licence 2 systématise. Les intégrales exigées par la \omterm{thm:b1:diffeq:voc}{variation de la
constante} sont fournies par le \cref{ch:b1:integration}, et le devoir
maison ci-dessous --- l'oscillateur amorti forcé --- est le cas modèle
de toute question d'oscillation dans les sciences, des circuits
électriques aux ponts suspendus.
\end{remark}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:b1:diffeq:1}
Résoudre sur $\R$~: $\;y' + 2y = \eu^{3x}$~; puis le \omterm{thm:b1:diffeq:voc}{problème de Cauchy}
$y(0) = 1$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:diffeq:2}
Résoudre sur $\intoo{0}{+\infty}$~: $\;x y' - 2y = x^3$ \emph{(mettre
d'abord l'équation sous forme normalisée)}.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:diffeq:3}
Résoudre sur $\R$, en donnant la solution générale réelle~:
$\;y'' - 3y' + 2y = 0$~; $\;y'' + 4y' + 4y = 0$~; $\;y'' - 2y' + 5y =
0$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:diffeq:4}
Résoudre $y'' - y = x^2$ sur $\R$, puis le \omterm{thm:b1:diffeq:voc}{problème de Cauchy} $y(0) =
0$, $y'(0) = 1$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:diffeq:5}
Résoudre sur $\intoo{-\frac\pi2}{\frac\pi2}$~:
$\;y' + y\tan x = \sin 2x$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:diffeq:6}
Résoudre $y'' - 4y' + 3y = (2x + 1)\,\eu^{x}$ sur $\R$.
\emph{(Attention à la multiplicité~: $1$ est-il racine du \omterm{def:b1:diffeq:linear2}{polynôme
caractéristique}~?)}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:diffeq:7}
Résoudre $y'' + 4y = \sin 2x + x$ sur $\R$ \emph{(superposition~:
traiter chaque second membre séparément)}.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:diffeq:8}
Une tasse de café à la température $T_0 = 80\,^\circ$C est posée dans
une pièce à $20\,^\circ$C. La loi de refroidissement de Newton énonce
que $T' = -k\,(T - 20)$ avec $k > 0$. Résoudre en $T(t)$ et, sachant
que le café est à $50\,^\circ$C au bout de $10$ minutes, déterminer
quand il atteint $25\,^\circ$C.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:diffeq:9}
(Oscillateur amorti) Pour $\varepsilon \geq 0$, on considère
$y'' + 2\varepsilon y' + y = 0$.
\begin{enumerate}
  \item Résoudre pour $\varepsilon \in \intco{0}{1}$, $\varepsilon =
        1$, et $\varepsilon > 1$.
  \item Montrer que pour $\varepsilon > 0$ toute solution tend vers $0$
        en $+\infty$, et que pour $\varepsilon = 0$ les solutions non
        nulles n'y tendent pas.
  \item Pour $\varepsilon \in \intoo{0}{1}$, montrer que les zéros
        d'une solution non nulle sont régulièrement espacés, avec un
        écart de $\frac{\pi}{\sqrt{1 - \varepsilon^2}}$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:diffeq:10}
Déterminer toutes les fonctions $f \colon \R \to \R$, deux fois
dérivables, telles que
\[
\forall x, y \in \R, \qquad f(x + y) + f(x - y) = 2 f(x) f(y),
\]
avec $f(0) \ne 0$ et $f$ non constante.
\emph{Indication~: fixer $y$, dériver deux fois par rapport à $x$ en
$0$~; montrer que $f(0) = 1$ et $f'' = c f$ pour une certaine constante
$c$~; puis résoudre selon le signe de $c$ et vérifier quelles solutions
satisfont l'équation fonctionnelle.}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:diffeq:11}
(Équation d'Euler) Résoudre $x^2 y'' - x y' + y = 0$ sur
$\intoo{0}{+\infty}$. \emph{Indication~: poser $z(t) = y(\eu^t)$,
c'est-à-dire substituer $x = \eu^t$, et montrer que $z$ satisfait une
équation linéaire à coefficients constants.}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:diffeq:12}
On considère l'équation $x\,y' = 2y$ sur la droite réelle tout
entière, d'inconnue la fonction dérivable $y \colon \R \to \R$.
\begin{enumerate}
  \item Résoudre sur $\intoo{0}{+\infty}$ et sur $\intoo{-\infty}{0}$.
  \item Montrer que pour \emph{toutes} constantes $a, b \in \R$, la
        fonction égale à $ax^2$ pour $x \geq 0$ et à $bx^2$ pour
        $x < 0$ est dérivable sur $\R$ et est solution de l'équation
        partout.
  \item En déduire que l'\omterm{def:b1:logic:sets}{ensemble} des solutions sur $\R$ est une
        famille à deux paramètres, et expliquer pourquoi cela ne
        contredit pas l'unicité du \cref{thm:b1:diffeq:voc}.
\end{enumerate}
\end{exercise}

\section{Problème~: l'oscillateur amorti forcé}

\begin{problem}\label{pb:b1:diffeq:1}
Une seule équation gouverne une masse au bout d'un ressort dans un
milieu visqueux, la charge dans un circuit RLC, et un immeuble
oscillant sous le vent~:
\[
(E_\Omega)\colon\quad
x'' + 2\lambda x' + \omega_0^2\,x = A\cos(\Omega t),
\]
avec $\lambda \geq 0$ l'amortissement, $\omega_0 > 0$ la pulsation
propre, et $A > 0$, $\Omega > 0$ l'amplitude et la pulsation de
l'excitation. Ce problème en extrait le comportement complet~: la
décroissance des transitoires, l'unique régime permanent périodique,
la courbe de \omterm{ex:b1:diffeq:oscillation}{résonance} et sa finesse (le \emph{\omterm{pb:b1:diffeq:1}{facteur de qualité}}),
les \omterm{pb:b1:diffeq:1}{battements} du cas non amorti, et le bilan d'énergie qui entretient
l'oscillation.\index{résonance}\index{oscillateur}
Sauf mention contraire, $0 < \lambda < \omega_0$ (régime
sous-amorti) et l'on pose $\omega_d = \sqrt{\omega_0^2 - \lambda^2}$.

\medskip
\noindent\textbf{Partie I --- L'oscillateur libre.} Ici $A = 0$.
\begin{enumerate}
  \item Résoudre l'équation homogène $(H)$ pour $0 < \lambda <
        \omega_0$, puis pour $\lambda = 0$. (Les régimes $\lambda
        \geq \omega_0$ ont été traités à l'\cref{exo:b1:diffeq:9}~;
        les citer.)
  \item Montrer que pour tout $\lambda > 0$, toutes les solutions de
        $(H)$ tendent vers $0$ en $+\infty$ --- dans les trois
        régimes.
  \item Définir l'énergie $\mathcal E(t) = \frac12 x'(t)^2 +
        \frac12\omega_0^2\,x(t)^2$ le long d'une solution de $(H)$.
        Montrer que $\mathcal E'(t) = -2\lambda\,x'(t)^2 \leq 0$, et
        en déduire (sans rien résoudre) que le \omterm{thm:b1:diffeq:voc}{problème de Cauchy}
        «~$(H)$, $x(t_0) = x'(t_0) = 0$~» n'a que la solution nulle,
        et ce pour tout $\lambda \geq 0$.
  \item Pour $0 < \lambda < \omega_0$, écrire la solution non nulle
        sous la forme $x(t) = R\,\eu^{-\lambda t}\cos(\omega_d t -
        \varphi)$ et poser $T_d = \frac{2\pi}{\omega_d}$ la
        pseudo-période. Montrer que $x(t + T_d) = \eu^{-\lambda
        T_d}\,x(t)$~: chaque oscillation est la précédente réduite du
        facteur constant $\eu^{-\delta}$, $\delta =
        \frac{2\pi\lambda}{\omega_d}$ (le \emph{décrément
        logarithmique}). Calculer $\delta$ pour $\omega_0 = 1$,
        $\lambda = 0.1$.
  \item Définir le \emph{\omterm{pb:b1:diffeq:1}{facteur de qualité}} $Q =
        \dfrac{\omega_0}{2\lambda}$\index{facteur de qualité}.
        Montrer qu'au bout du temps $\frac1\lambda$ (une division de
        l'amplitude par $\eu$), l'oscillateur a accompli
        $\frac{\omega_d}{2\pi\lambda}$ pseudo-périodes, ce qui, pour
        un amortissement faible ($\lambda \ll \omega_0$), vaut
        approximativement $\frac Q\pi$~: le \omterm{pb:b1:diffeq:1}{facteur de qualité} compte,
        à $\pi$ près, le nombre d'oscillations accomplies avant que
        l'amplitude ne soit divisée par $\eu$.
\end{enumerate}

\medskip
\noindent\textbf{Partie II --- Le régime permanent.} Désormais $A > 0$
et $\lambda > 0$.
\begin{enumerate}[resume]
  \item Chercher une solution particulière comme partie réelle de
        $z\,\eu^{\iu\Omega t}$ avec $z \in \C$
        (\cref{met:b1:diffeq:particular}). Montrer que cela fonctionne
        avec
        \[
        z = \frac{A}{\omega_0^2 - \Omega^2 + 2\iu\lambda\Omega} .
        \]
  \item En déduire le régime permanent sous forme
        amplitude--phase~: $x_p(t) = R(\Omega)\cos\bigl(\Omega t -
        \varphi(\Omega)\bigr)$ avec
        \[
        R(\Omega) = \frac{A}{\sqrt{(\omega_0^2 - \Omega^2)^2 +
        4\lambda^2\Omega^2}},
        \qquad
        \tan\varphi = \frac{2\lambda\Omega}{\omega_0^2 -
        \Omega^2},
        \quad \varphi \in \intoo0\pi .
        \]
  \item Interpréter les deux régimes extrêmes~: calculer les limites
        de $R$ et $\varphi$ quand $\Omega \to 0^+$ (réponse
        quasi-statique $A/\omega_0^2$, phase $0$) et quand $\Omega \to
        +\infty$ ($R \sim A/\Omega^2 \to 0$, phase $\to \pi$~: la
        masse se déplace à l'opposé d'une excitation trop rapide).
  \item Montrer que \emph{toute} solution de $(E_\Omega)$ est $x_p$
        plus une solution de $(H)$, donc converge vers le régime
        permanent $x_p$ quand $t \to +\infty$, quelles que soient les
        conditions initiales~: une fois le transitoire éteint,
        l'oscillateur n'a plus aucun souvenir de son démarrage.
  \item Montrer que $x_p$ est la \emph{seule} solution périodique de
        $(E_\Omega)$.
  \item Traiter un \omterm{thm:b1:diffeq:voc}{problème de Cauchy} jusqu'au bout~: pour $x'' + 2x'
        + 2x = \cos t$ avec $x(0) = x'(0) = 0$, montrer que la
        solution est
        \[
        x(t) = \frac{\cos t + 2\sin t}5
        - \eu^{-t}\,\frac{\cos t + 3\sin t}5 ,
        \]
        et identifier les parties transitoire et permanente.
\end{enumerate}

\medskip
\noindent\textbf{Partie III --- La courbe de \omterm{ex:b1:diffeq:oscillation}{résonance}.} Étude de
$\Omega \mapsto R(\Omega)$ sur $\intoo0{+\infty}$.
\begin{enumerate}[resume]
  \item En posant $u = \Omega^2$ et $g(u) = (\omega_0^2 - u)^2 +
        4\lambda^2 u$, montrer~: si $2\lambda^2 < \omega_0^2$, alors
        $R$ atteint un maximum strict à la \emph{pulsation de
        \omterm{ex:b1:diffeq:oscillation}{résonance}} $\Omega_r = \sqrt{\omega_0^2 - 2\lambda^2}$, avec
        \[
        R_{\max} = R(\Omega_r)
        = \frac{A}{2\lambda\sqrt{\omega_0^2 - \lambda^2}} .
        \]
  \item Montrer que $\dfrac{R_{\max}}{R(0)} = Q\,\bigl(1 -
        \tfrac{\lambda^2}{\omega_0^2}\bigr)^{-1/2} \geq Q$~: à la
        \omterm{ex:b1:diffeq:oscillation}{résonance}, l'excitation est amplifiée (pour l'essentiel) par
        le \omterm{pb:b1:diffeq:1}{facteur de qualité}.
  \item Démontrer que l'amplitude de la \emph{vitesse} $V(\Omega) =
        \Omega\,R(\Omega)$ est maximale exactement en $\Omega =
        \omega_0$ (et non en $\Omega_r$), et que la phase y vaut
        $\varphi(\omega_0) = \frac\pi2$~: en $\Omega = \omega_0$ la
        vitesse est exactement en phase avec la force.
  \item (Bande passante) Résoudre exactement $g(u) = 2\,g(u_r)$, où
        $u_r = \omega_0^2 - 2\lambda^2$, et en déduire que les deux
        pulsations $\Omega_\pm$ où $R = R_{\max}/\sqrt2$ vérifient
        $\Omega_+^2 - \Omega_-^2 = 4\lambda\sqrt{\omega_0^2 -
        \lambda^2}$~; conclure que pour un amortissement faible la
        bande passante vaut $\Omega_+ - \Omega_- \approx 2\lambda$,
        c'est-à-dire $Q \approx \frac{\omega_0}{\Omega_+ -
        \Omega_-}$~: les pics de \omterm{ex:b1:diffeq:oscillation}{résonance} aigus sont les systèmes à
        grand $Q$.
  \item Portrait numérique pour $\omega_0 = 1$, $\lambda = 0.05$
        ($Q = 10$), $A = 1$~: calculer $\Omega_r$, $R_{\max}$, la
        réponse statique $R(0)$, et la bande passante approchée.
  \item Montrer que si $2\lambda^2 \geq \omega_0^2$, alors $R$ est
        strictement décroissante sur $\intoo0{+\infty}$~: les systèmes
        fortement amortis n'ont aucun pic de \omterm{ex:b1:diffeq:oscillation}{résonance}.
\end{enumerate}

\medskip
\noindent\textbf{Partie IV --- Sans amortissement~: \omterm{pb:b1:diffeq:1}{battements} et
\omterm{ex:b1:diffeq:oscillation}{résonance}.} Ici $\lambda = 0$.
\begin{enumerate}[resume]
  \item Pour $\Omega \neq \omega_0$, déterminer la solution générale
        de $x'' + \omega_0^2 x = A\cos(\Omega t)$.
  \item Résoudre le \omterm{thm:b1:diffeq:voc}{problème de Cauchy} $x(0) = x'(0) = 0$ et
        transformer la réponse en la forme produit
        \[
        x(t) = \frac{2A}{\omega_0^2 - \Omega^2}\,
        \sin\Bigl(\frac{(\omega_0 - \Omega)t}2\Bigr)
        \sin\Bigl(\frac{(\omega_0 + \Omega)t}2\Bigr) .
        \]
  \item Pour $\Omega$ proche de $\omega_0$, lire le produit comme une
        oscillation rapide à la pulsation $\frac{\omega_0 + \Omega}2$
        modulée par une enveloppe lente de pulsation
        $\frac{\abs{\omega_0 - \Omega}}2$~: les \emph{battements}
        \index{battements}. Donner la période de l'enveloppe et
        l'amplitude maximale, et remarquer comment toutes deux
        explosent quand $\Omega \to \omega_0$.
  \item Fixer $t$ et faire tendre $\Omega \to \omega_0$ dans la
        formule de la question 19~: montrer que la limite est
        \[
        x_\infty(t) = \frac{A\,t\,\sin(\omega_0 t)}{2\omega_0} ,
        \]
        et vérifier directement que $x_\infty$ est solution de
        l'équation résonante $x'' + \omega_0^2 x = A\cos(\omega_0 t)$
        avec $x(0) = x'(0) = 0$ (à comparer avec
        l'\cref{ex:b1:diffeq:oscillation})~: la \omterm{ex:b1:diffeq:oscillation}{résonance} est la limite
        de \omterm{pb:b1:diffeq:1}{battements} toujours plus lents et toujours plus grands.
  \item Opposer les deux destins de la \omterm{ex:b1:diffeq:oscillation}{résonance}~: croissance linéaire
        $\frac{At}{2\omega_0}$ sans amortissement, contre saturation à
        $R_{\max} \approx Q\,\frac{A}{\omega_0^2}$ avec un
        amortissement faible. En une phrase~: quel mécanisme physique
        convertit le premier en le second~?
\end{enumerate}

\medskip
\noindent\textbf{Partie V --- Bilan d'énergie et synthèse.}
\begin{enumerate}[resume]
  \item Dans le régime permanent de la partie II, calculer la moyenne
        sur une période $\frac{2\pi}\Omega$ (a) de la puissance
        injectée par l'excitation, $P_{\mathrm{in}}(t) = A\cos(\Omega
        t) \cdot x_p'(t)$, et (b) de la puissance dissipée par
        l'amortissement, $P_{\mathrm{diss}}(t) = 2\lambda\,x_p'(t)^2$.
        Montrer que les deux moyennes valent $\lambda\,R^2\Omega^2$~:
        l'excitation fournit exactement ce que l'amortissement brûle
        --- c'est pourquoi le régime permanent est permanent.
  \item Où exactement le problème a-t-il utilisé~: (i) le théorème de
        structure \cref{thm:b1:diffeq:structure2}~; (ii) la méthode de
        l'exponentielle complexe~; (iii) une étude de fonction d'une
        variable réelle dans le style du \cref{ch:b1:functions}~? Une
        phrase pour chacun.
  \item Synthèse~: décrire la carte complète des comportements de
        $(E_\Omega)$ --- libre contre forcé, amorti contre non amorti,
        le rôle de $Q$ comme unique cadran sans dimension réglant la
        hauteur du pic, la bande passante et la durée de vie du
        transitoire --- et indiquer où l'histoire se poursuit~:
        systèmes $2 \times 2$ du premier ordre (\cref{ch:b1:matrices}
        et le volume de Licence 2) et décomposition d'une excitation
        périodique générale en sinusoïdes (séries de Fourier, dans le
        volume de Licence 3), pour laquelle le cas sinusoïdal de ce
        problème est la brique fondamentale.
\end{enumerate}
