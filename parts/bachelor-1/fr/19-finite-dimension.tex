\chapter{Dimension finie}\label{ch:b1:findim}

Un espace engendré par un nombre fini de vecteurs possède une
\emph{dimension} bien définie --- la taille commune de toutes ses
\omterm{def:b1:vspaces:free}{bases}. Les démonstrations qui suivent découlent toutes d'un même
moteur combinatoire, le lemme d'échange~: une \omterm{def:b1:vspaces:free}{famille libre} ne peut
jamais être plus nombreuse qu'une famille génératrice. Avec la
dimension viennent les outils utilisés partout ensuite~: le \omterm{thm:b1:findim:bases}{théorème
de la base incomplète}, le \omterm{def:b1:findim:rank}{rang} d'une famille, la \omterm{thm:b1:findim:grassmann}{formule de Grassmann}.

\section{Existence de bases}

\begin{definition}\label{def:b1:findim:def}
$E$ est \emph{de dimension finie}\index{dimension finie} lorsqu'il
possède une famille génératrice finie. (Sinon il est \emph{de dimension
infinie}~: c'est le cas de $K[X]$, dont les familles finies
n'engendrent que des \omterm{def:b1:poly:def}{polynômes} de degré borné.)
\end{definition}

\begin{theorem}[Lemme d'échange]\label{thm:b1:findim:exchange}
Soit $(g_1, \dots, g_n)$ une famille génératrice de $E$ et $(f_1,
\dots, f_p)$ une \omterm{def:b1:vspaces:free}{famille libre} de $E$. Alors $p \leq n$.
\end{theorem}

\begin{proof}
Montrons par récurrence sur $k \leq p$~: \emph{quitte à renuméroter
les $g_j$, la famille}
\[
(f_1, \dots, f_k, g_{k+1}, \dots, g_n)
\]
\emph{engendre $E$} --- ce qui force $k \leq n$ à chaque étape, et
$p \leq n$ à l'arrivée.

$k = 0$~: c'est l'hypothèse. Étape~: supposons le résultat pour $k -
1$~; alors $f_k$ est combinaison de $(f_1, \dots, f_{k-1},$ $g_k,
\dots, g_n)$. Dans cette combinaison, un certain $g_j$ ($j \geq k$) a
un coefficient non nul
---
sinon $f_k$ serait combinaison de $f_1, \dots, f_{k-1}$, contredisant
la liberté. Renumérotons de sorte que $j = k$, et résolvons en $g_k$~:
$g_k$ est combinaison de $(f_1, \dots, f_k, g_{k+1}, \dots,
g_n)$. Tout vecteur de $E$, exprimé au moyen de la famille à $k-1$
termes, se réexprime alors au moyen de celle à $k$ termes~: elle
engendre. (Si $k - 1 = n$, il ne reste plus aucun $g$ et $f_k$ serait
combinaison des seuls $f_i$~: impossible~; donc $k \leq n$.)
\end{proof}

\begin{example}[L'échange, vu une fois en action]
\label{ex:b1:findim:exchangerun}
Dans $\R^2$, prenons la famille génératrice $\bigl(g_1, g_2\bigr) =
\bigl((1,0), (0,1)\bigr)$ et la \omterm{def:b1:vspaces:free}{famille libre} $\bigl(f_1,
f_2\bigr) = \bigl((1,2), (3,4)\bigr)$. Étape $1$~: $f_1 = 1\cdot
g_1 + 2\cdot g_2$~; le coefficient de $g_2$ est non nul, donc on
échange $g_2$ contre $f_1$~: la famille $\bigl(f_1, g_1\bigr)$
engendre encore ($g_2 = \frac12(f_1 - g_1)$). Étape $2$~: $f_2 =
(3,4) = 2\,f_1 + 1\cdot g_1$~; le coefficient du $g_1$ restant est
non nul (il doit l'être~: $f_2$ n'est pas multiple de $f_1$), donc
on échange à nouveau~: $\bigl(f_1, f_2\bigr)$ engendre $\R^2$. S'il
y avait eu un troisième vecteur \omterm{def:b1:vspaces:free}{libre} $f_3$, aucun $g$ n'aurait
subsisté pour l'absorber --- et c'est exactement ainsi que le lemme
interdit $3$ vecteurs \omterm{def:b1:vspaces:free}{libres} dans $\R^2$. La démonstration ci-dessus
n'est que cette comptabilité menée en général.
\end{example}

\begin{theorem}[Bases en dimension finie]\label{thm:b1:findim:bases}
Soit $E \neq \{0\}$ \omterm{def:b1:findim:def}{de dimension finie}.
\begin{enumerate}
  \item De toute famille génératrice finie on peut extraire une \omterm{def:b1:vspaces:free}{base}.
  \item (\omterm{thm:b1:findim:bases}{Théorème de la base incomplète}\index{théorème de la base incomplète})
        Toute \omterm{def:b1:vspaces:free}{famille libre} se complète en une \omterm{def:b1:vspaces:free}{base}, à l'aide de
        vecteurs d'une famille génératrice choisie à l'avance.
  \item Toutes les \omterm{def:b1:vspaces:free}{bases} de $E$ sont finies et ont le même nombre
        d'éléments~: la \emph{dimension} $\dim E$. (Convention~:
        $\dim\{0\} = 0$.)
\end{enumerate}
\end{theorem}

\begin{proof}
(1) Éliminons, un à un, tout vecteur qui est combinaison des autres~;
la famille reste génératrice, puisque dans toute expression
utilisant le vecteur éliminé on peut substituer sa combinaison des
survivants. Le procédé s'arrête --- chaque étape fait décroître
d'une unité une famille finie --- et il s'arrête exactement quand
aucun vecteur restant n'est combinaison des autres. La famille
finale est encore génératrice, et elle est \omterm{def:b1:vspaces:free}{libre}~: une combinaison
nulle non triviale porterait un coefficient non nul, et en divisant
par lui on résoudrait en le vecteur correspondant à l'aide des
autres, qui serait donc encore éliminable --- contredisant l'arrêt
du procédé.

(2) Soient $(f_1, \dots, f_p)$ \omterm{def:b1:vspaces:free}{libre} et $(g_1, \dots, g_n)$
génératrice. Parcourons $g_1, \dots, g_n$ en adjoignant $g_j$ à la
famille courante chaque fois qu'il n'appartient pas déjà à son
\omterm{def:b1:vspaces:span}{sous-espace engendré}
(la \cref{prop:b1:vspaces:freecriteria}~(2) garde la \omterm{def:b1:vspaces:free}{famille libre}).
La famille finale est \omterm{def:b1:vspaces:free}{libre}, et génératrice~: chaque $g_j$ appartient
à son \omterm{def:b1:vspaces:span}{sous-espace engendré} --- soit il a été adjoint, soit il en
était déjà une combinaison.

(3) Deux \omterm{def:b1:vspaces:free}{bases} sont chacune \omterm{def:b1:vspaces:free}{libre} et chacune génératrice~: le lemme
d'échange donne les deux inégalités entre leurs cardinaux. (Finitude~:
une \omterm{def:b1:vspaces:free}{base} est \omterm{def:b1:vspaces:free}{libre}, donc, par échange, pas plus grande qu'une famille
génératrice finie.)
\end{proof}

\begin{example}\label{ex:b1:findim:dims}
$\dim K^n = n$ (\omterm{def:b1:vspaces:free}{base} canonique)~; $\dim K_n[X] = n + 1$ (les
monômes)~; $\dim_\R \C = 2$~; l'espace des solutions de $y'' + ay' +
by = 0$ est de dimension $2$ (\cref{thm:b1:diffeq:homogeneous2}~: les
solutions sont paramétrées bijectivement et linéairement par
$(\lambda, \mu) \in K^2$).
\end{example}

\begin{example}[Le même ensemble, deux dimensions]
\label{ex:b1:findim:twodimensions}
L'\omterm{def:b1:logic:sets}{ensemble} $\C^2$ des couples de nombres complexes est un
$\C$-espace vectoriel de dimension $2$ (\omterm{def:b1:vspaces:free}{base} canonique $e_1, e_2$)
--- et un $\R$-espace vectoriel de dimension $4$, de \omterm{def:b1:vspaces:free}{base}
\[
(1, 0),\quad (\iu, 0),\quad (0, 1),\quad (0, \iu):
\]
tout $(z, w) = (a + \iu b,\ c + \iu d)$ a pour \omterm{prop:b1:vspaces:coordinates}{coordonnées} réelles
$(a, b, c, d)$, de façon unique. La dimension n'est pas une
propriété du seul \omterm{def:b1:logic:sets}{ensemble} des vecteurs~: elle compte les degrés de
liberté \emph{relativement aux scalaires autorisés}, et diviser par
deux la réserve de scalaires en passant de $\C$ à $\R$ double le
compte. (Le devoir maison exploite le cas extrême de cette
sensibilité, avec des scalaires réduits jusqu'à $\Q$.)
\end{example}

\begin{example}[L'algorithme de complétion en action]
\label{ex:b1:findim:completion}
Complétons la \omterm{def:b1:vspaces:free}{famille libre} $\bigl((1,1,1)\bigr)$ en une \omterm{def:b1:vspaces:free}{base} de
$\R^3$ à l'aide des vecteurs canoniques. Déroulons la démonstration
du \cref{thm:b1:findim:bases}~(2) sur la famille génératrice $(e_1,
e_2, e_3)$~: a-t-on $e_1 \in \operatorname{Vect}(1,1,1)$~? Non (les
multiples de $(1,1,1)$ ont des \omterm{prop:b1:vspaces:coordinates}{coordonnées} égales) --- on l'adjoint.
A-t-on $e_2 \in \operatorname{Vect}\bigl((1,1,1), e_1\bigr)$~? Une
combinaison $\alpha(1,1,1) + \beta e_1$ a ses deuxième et troisième
\omterm{prop:b1:vspaces:coordinates}{coordonnées} égales, ce qui n'est pas le cas de $e_2$ --- on
l'adjoint. La famille $\bigl((1,1,1), e_1, e_2\bigr)$ est \omterm{def:b1:vspaces:free}{libre} à
$3$ vecteurs~: on s'arrête, c'est une \omterm{def:b1:vspaces:free}{base}
(la \cref{prop:b1:findim:twoofthree} officialisera ce réflexe).
Remarquons que le résultat dépend de l'ordre dans lequel les $g_j$
sont parcourus~: la complétion est un algorithme, pas une formule.
\end{example}

\begin{proposition}[La règle «~deux sur trois~»]\label{prop:b1:findim:twoofthree}
Soient $\dim E = n$ et $\mathcal{F}$ une famille de $n$ vecteurs
exactement de $E$. Alors
\[
\mathcal{F} \text{ libre} \iff \mathcal{F} \text{ génératrice} \iff
\mathcal{F} \text{ base}.
\]
De plus toute \omterm{def:b1:vspaces:free}{famille libre} a $\leq n$ vecteurs, toute famille
génératrice $\geq n$.
\end{proposition}

\begin{proof}
Les bornes de \omterm{def:b1:counting:card}{cardinal} sont le lemme d'échange confronté à une \omterm{def:b1:vspaces:free}{base}.
Si $\mathcal{F}$ (de taille $n$) est \omterm{def:b1:vspaces:free}{libre} sans être génératrice, un
certain $x$ est hors de son \omterm{def:b1:vspaces:span}{sous-espace engendré}~; en l'adjoignant on
obtient une \omterm{def:b1:vspaces:free}{famille libre} de $n + 1$ vecteurs~: impossible. Si
$\mathcal{F}$ est génératrice sans être \omterm{def:b1:vspaces:free}{libre}, en extraire une \omterm{def:b1:vspaces:free}{base}
(\cref{thm:b1:findim:bases}~(1)) fournit une \omterm{def:b1:vspaces:free}{base} de $< n$ vecteurs~:
impossible.
\end{proof}

\begin{example}[«~Deux sur trois~», ou la moitié du travail économisée]
\label{ex:b1:findim:twoofthreerun}
La famille
\[
\bigl((1,1,0), (0,1,1), (1,0,1)\bigr)
\]
est-elle une \omterm{def:b1:vspaces:free}{base}
de $\R^3$~? Comptons~: trois vecteurs, dimension trois --- la seule
liberté décide donc. Une combinaison nulle donne $a + c = 0$, $a + b
= 0$, $b + c = 0$~; en additionnant les trois, $2(a + b + c) = 0$, et
en retranchant chaque équation initiale de $a + b + c = 0$ il reste
$b = c = a = 0$~: \omterm{def:b1:vspaces:free}{libre}, donc \omterm{def:b1:vspaces:free}{base}, la moitié «~génératrice~» de la
vérification étant offerte par le théorème. À comparer avec
l'\cref{ex:b1:vspaces:testbasis}, où la même double vérification
devait être menée à la main --- un chapitre de théorie se convertit
exactement en cette économie, sur chaque vérification de \omterm{def:b1:vspaces:free}{base}
jusqu'à la fin du livre.
\end{example}

\begin{method}[Calculer une dimension]
\label{met:b1:findim:computedim}
Trois voies standard, par ordre de fréquence décroissante.
\begin{enumerate}
  \item \emph{Paramétrer, puis lire une \omterm{def:b1:vspaces:free}{base}.} Résoudre les
        contraintes de définition, exprimer l'élément général
        linéairement en fonction des paramètres survivants, et
        vérifier que les vecteurs multipliant les paramètres sont
        \omterm{def:b1:vspaces:free}{libres}~: la dimension est le nombre de paramètres. (Déroulée
        ci-dessous sur un \omterm{def:b1:vspaces:subspace}{sous-espace} concret de $\R^4$.)
  \item \emph{Exhiber un paramétrage linéaire \omterm{def:b1:logic:inj}{bijectif}.} Lorsque
        les éléments sont déterminés par un nombre fini de valeurs
        --- conditions initiales d'une récurrence
        (\cref{exo:b1:findim:10}), coefficients d'une formule de
        solution (\cref{ex:b1:findim:dims}) --- la dimension est le
        nombre de ces valeurs.
  \item \emph{Utiliser les formules.} Grassmann pour les
        intersections et les sommes, le \omterm{def:b1:findim:rank}{rang} pour les \omterm{def:b1:vspaces:span}{sous-espaces
        engendrés}, et plus tard le théorème du \omterm{def:b1:findim:rank}{rang} pour les noyaux
        et les images~: les dimensions se \emph{calculent}
        d'ordinaire, elles ne se devinent pas.
\end{enumerate}
Dans les trois voies, la règle «~deux sur trois~» donne le coup de
grâce~: dès que le compte est bon, la liberté ou le caractère
générateur suffit à conclure.
\end{method}

\begin{example}[Voie 1, en entier]
\label{ex:b1:findim:parametrize}
Dimension de $H = \{(x, y, z, t) \in \R^4 : x + y + z + t = 0
\text{ et } x = t\}$. Résolvons~: $t = x$ et $y + z = -2x$, donc $z =
-2x - y$ avec $x, y$ \omterm{def:b1:vspaces:free}{libres}~:
\[
(x,\ y,\ -2x - y,\ x) = x\,(1, 0, -2, 1) + y\,(0, 1, -1, 0) .
\]
Les deux vecteurs sont \omterm{def:b1:vspaces:free}{libres} (regarder les deux premières
\omterm{prop:b1:vspaces:coordinates}{coordonnées}~: $(x, y) = (0,0)$), donc ils forment une \omterm{def:b1:vspaces:free}{base} de $H$ et
$\dim H = 2$. Le compte était prévisible --- deux contraintes
linéaires indépendantes dans $\R^4$ devraient manger chacune une
dimension --- mais le paramétrage le démontre \emph{et} livre une
\omterm{def:b1:vspaces:free}{base}, ce que la prévision seule ne fait jamais~;
l'\cref{exo:b1:findim:9} transforme le slogan «~chaque équation mange
au plus une dimension~» en théorème.
\end{example}

\begin{example}[Voie 2, en entier]
\label{ex:b1:findim:route2}
Dimension de $W = \{P \in \R_n[X] : P(1) = P(2) = 0\}$ (pour $n
\geq 2$). Par le théorème du facteur appliqué deux fois
(\cref{thm:b1:poly:factor}~; les racines $1$ et $2$ sont distinctes),
$P \in W$ exactement lorsque $P = (X - 1)(X - 2)\,Q$ avec $\deg Q \leq
n - 2$. La correspondance $Q \mapsto (X-1)(X-2)Q$ est linéaire,
atteint tout $W$, et est \omterm{def:b1:logic:inj}{injective} (un produit n'est nul que si
$Q = 0$)~: $W$ est paramétré bijectivement et linéairement par
$\R_{n-2}[X]$, donc
\[
\dim W = \dim \R_{n-2}[X] = n - 1 .
\]
Une \omterm{def:b1:vspaces:free}{base} vient avec le paramétrage~: les images des monômes,
$\bigl((X-1)(X-2),\ (X-1)(X-2)X,\ \dots,\
(X-1)(X-2)X^{n-2}\bigr)$. Chaque nouvelle contrainte d'évaluation en
un point neuf coûte exactement une dimension --- c'est l'ossature
comptable de l'\omterm{thm:b1:poly:lagrange}{interpolation de Lagrange}
(\cref{thm:b1:poly:lagrange}).
\end{example}

\begin{example}[Une contrainte intégrale coûte elle aussi une dimension]
\label{ex:b1:findim:integralconstraint}
Dimension de $H = \{P \in \R_2[X] : \int_0^1 P = 0\}$. En écrivant
$P = a + bX + cX^2$, la contrainte s'écrit $a + \frac b2 + \frac
c3 = 0$~; résolvons en $a$ et paramétrons~:
\[
P = b\Bigl(X - \frac12\Bigr) + c\Bigl(X^2 - \frac13\Bigr),
\]
donc $H = \operatorname{Vect}\bigl(X - \frac12,\ X^2 -
\frac13\bigr)$, de dimension $2$ (les deux \omterm{def:b1:poly:def}{polynômes} ont des degrés
distincts~: \omterm{def:b1:vspaces:free}{libres}). Une condition linéaire --- que ce soit une
évaluation, une \omterm{thm:b1:integration:def}{intégrale}, ou toute autre recette linéaire ---
retire au plus une dimension, et exactement une dès que la condition
n'est pas identiquement nulle. Le \cref{ch:b1:linmaps} nommera de
telles recettes \emph{formes linéaires} et leurs \omterm{def:b1:logic:sets}{ensembles} de
solutions \emph{hyperplans}~; les vecteurs de \omterm{def:b1:vspaces:free}{base} trouvés ici
réapparaissent au \cref{ch:b1:euclid} comme le début de la famille de
Legendre.
\end{example}

\section{Sous-espaces, rang, formule de Grassmann}

\begin{theorem}[Sous-espaces]\label{thm:b1:findim:subspaces}
Soient $E$ \omterm{def:b1:findim:def}{de dimension finie} et $F$ un \omterm{def:b1:vspaces:subspace}{sous-espace vectoriel}. Alors
$F$ est \omterm{def:b1:findim:def}{de dimension finie}, $\dim F \leq \dim E$, avec égalité si et
seulement si $F = E$. De plus tout \omterm{def:b1:vspaces:subspace}{sous-espace vectoriel} admet un
\omterm{def:b1:vspaces:sum}{supplémentaire}.
\end{theorem}

\begin{proof}
Les \omterm{def:b1:vspaces:free}{familles libres} de $F$ ont au plus $\dim E$ vecteurs (lemme
d'échange dans $E$~: une \omterm{def:b1:vspaces:free}{famille libre} de $F$ est en particulier
\omterm{def:b1:vspaces:free}{libre} dans $E$, et $E$ a une famille génératrice finie). Parmi les
\omterm{def:b1:vspaces:free}{familles libres} de $F$, choisissons-en une de taille \emph{maximale}
$p$ --- c'est possible, les tailles étant des entiers majorés par
$\dim E$. Elle engendre $F$~: sinon un certain $x \in F$ serait hors
de son \omterm{def:b1:vspaces:span}{sous-espace engendré}, et en adjoignant $x$ on obtiendrait une
\omterm{def:b1:vspaces:free}{famille libre} de $F$ de taille $p + 1$
(\cref{prop:b1:vspaces:freecriteria}~(2)), contredisant la
maximalité. \omterm{def:b1:vspaces:free}{Libre} et génératrice, c'est une \omterm{def:b1:vspaces:free}{base} de $F$, et
$\dim F = p \leq \dim E$. Si $p = \dim E = n$~: une \omterm{def:b1:vspaces:free}{famille libre} de $n$
vecteurs de $E$ est une \omterm{def:b1:vspaces:free}{base} de $E$
(\cref{prop:b1:findim:twoofthree}), donc $F$ contient le \omterm{def:b1:vspaces:span}{sous-espace
engendré} par cette famille, qui vaut $E$. \omterm{def:b1:vspaces:sum}{Supplémentaire}~: complétons une \omterm{def:b1:vspaces:free}{base}
$(f_1, \dots, f_p)$ de $F$ en une \omterm{def:b1:vspaces:free}{base} $(f_1, \dots, f_p, g_{p+1},
\dots, g_n)$ de $E$ (\omterm{thm:b1:findim:bases}{théorème de la base incomplète})~; alors $G =
\operatorname{Vect}(g_{p+1}, \dots, g_n)$ vérifie $E = F \oplus G$
(existence et unicité des décompositions = \omterm{prop:b1:vspaces:coordinates}{coordonnées} dans la
grande \omterm{def:b1:vspaces:free}{base}).
\end{proof}

\begin{definition}[Rang d'une famille]\label{def:b1:findim:rank}
Le \emph{rang}\index{rang!d'une famille} d'une famille finie de
vecteurs est la dimension du \omterm{def:b1:vspaces:subspace}{sous-espace} qu'elle engendre~:
$\operatorname{rk}(x_1, \dots, x_p) = \dim
\operatorname{Vect}(x_1, \dots, x_p) \leq \min(p, \dim E)$,
avec égalité à $p$ si et seulement si la famille est \omterm{def:b1:vspaces:free}{libre}.
\end{definition}

\begin{example}[Calcul d'un rang par élimination]
\label{ex:b1:findim:rankcomputation}
\omterm{def:b1:findim:rank}{Rang} de $\bigl((1,2,3), (2,3,4), (3,4,5), (1,1,1)\bigr)$ dans
$\R^3$. Le \omterm{def:b1:vspaces:span}{sous-espace engendré} ne change pas quand on retranche à
un vecteur une combinaison des autres (les deux familles engendrent
les mêmes combinaisons)~: remplaçons $(2,3,4)$ par $(2,3,4) - (1,2,3)
= (1,1,1)$ et $(3,4,5)$ par $(3,4,5) - (1,2,3) = (2,2,2)$. Le
\omterm{def:b1:vspaces:span}{sous-espace engendré} est désormais
$\operatorname{Vect}\bigl((1,2,3), (1,1,1), (2,2,2),
(1,1,1)\bigr) = \operatorname{Vect}\bigl((1,2,3), (1,1,1)\bigr)$,
et ces deux vecteurs ne sont pas proportionnels~: le \omterm{def:b1:findim:rank}{rang} vaut $2$.
Cette procédure de soustraction-élimination est systématisée sous le
nom de méthode du pivot de Gauss au \cref{ch:b1:det}.
\end{example}

\begin{example}[Sommer par concaténation]
\label{ex:b1:findim:sumconcat}
Prenons, dans $\R^3$,
\[
F = \operatorname{Vect}\bigl((1,2,3),\ (1,1,1)\bigr),
\qquad
G = \operatorname{Vect}\bigl((2,3,4)\bigr) .
\]
La somme $F + G$ est engendrée par la famille concaténée des trois
générateurs, et
\[
(2, 3, 4) = (1, 2, 3) + (1, 1, 1)
\]
montre que le troisième est redondant~: $F + G = F$, de dimension $2$
--- autrement dit $G \subseteq F$, ce que la relation affiche.
Grassmann confirme~: $\dim(F \cap G) = 2 + 1 - 2 = 1 = \dim G$.
Les sommes se calculent en concaténant les générateurs puis en
réduisant le tas par l'algorithme du \omterm{def:b1:findim:rank}{rang}~; aucune technique nouvelle
n'est jamais nécessaire.
\end{example}

\begin{theorem}[Formule de Grassmann]\label{thm:b1:findim:grassmann}
Pour des \omterm{def:b1:vspaces:subspace}{sous-espaces vectoriels} $F, G$ \omterm{def:b1:findim:def}{de dimension finie} de
$E$~:\index{formule de Grassmann}
\[
\dim(F + G) = \dim F + \dim G - \dim(F \cap G) .
\]
En particulier la somme $F + G$ est directe si et seulement si
$\dim(F + G) = \dim F + \dim G$.
\end{theorem}

\begin{proof}
Partons d'une \omterm{def:b1:vspaces:free}{base} $(e_1, \dots, e_r)$
de $F \cap G$~; complétons-la en une
\omterm{def:b1:vspaces:free}{base}
\[
(e_1, \dots, e_r, f_1, \dots, f_s) \text{ de } F
\quad\text{et}\quad
(e_1, \dots, e_r, g_1, \dots, g_t) \text{ de } G
\]
(\cref{thm:b1:findim:bases}~(2)). Montrons que
\[
\mathcal{B} = (e_1, \dots, e_r, f_1, \dots, f_s, g_1, \dots, g_t)
\]
est une \omterm{def:b1:vspaces:free}{base} de $F + G$~; la formule suit par comptage~: $(r + s) +
(r + t) - r = r + s + t$.

$\mathcal{B}$ engendre $F + G$~: tout $u + v$ ($u \in F$, $v \in G$)
s'y décompose. Liberté~: supposons $\sum \alpha_i e_i + \sum
\beta_j f_j + \sum \gamma_k g_k = 0$. Le vecteur $w = \sum \gamma_k
g_k = -\sum\alpha_i e_i - \sum\beta_j f_j$ appartient à $G \cap F$, donc
il se décompose sur les seuls $(e_i)$~; mais $w$ se décompose aussi
sur les $(g_k)$, et dans la \omterm{def:b1:vspaces:free}{base} de $G$ ces deux expressions doivent
coïncider~: tous les
$\gamma_k = 0$ (et les \omterm{prop:b1:vspaces:coordinates}{coordonnées} en $e$ se correspondent). La relation se
réduit à $\sum\alpha_i e_i + \sum\beta_j f_j = 0$, une relation dans la
\omterm{def:b1:vspaces:free}{base} de $F$~: tous les coefficients restants sont nuls.
\end{proof}

\begin{example}\label{ex:b1:findim:planes}
Deux plans distincts $F, G$ (de dimension $2$) de $\R^3$ vérifient
$F + G = \R^3$ (leur somme contient strictement un plan), donc
$\dim(F \cap G) = 2 + 2 - 3 = 1$~: ils se coupent toujours suivant une
droite --- pas de «~plans parallèles~» passant par l'origine.
\end{example}

\begin{example}[Grassmann à l'œuvre, dans $\R^4$]
\label{ex:b1:findim:grassmannrun}
Soient $F = \operatorname{Vect}\bigl(e_1,\ e_2,\ (1,1,1,0)\bigr)$
et $G = \operatorname{Vect}(e_3, e_4)$ dans $\R^4$. Dimensions~:
$\dim F = 3$ (le troisième générateur a une troisième coordonnée non
nulle, il est hors de $\operatorname{Vect}(e_1, e_2)$) et $\dim G = 2$.
Somme~: $F + G$ contient $e_1, e_2, e_4$ et $e_3 = (1,1,1,0) - e_1 -
e_2$~: c'est $\R^4$ tout entier. Grassmann \emph{calcule} alors la
taille de l'intersection sans la moindre élimination~:
\[
\dim(F \cap G) = 3 + 2 - 4 = 1 .
\]
Pour identifier la droite, regardons dans $G$~: un vecteur $(0, 0, c, d)$
appartient à $F$ exactement quand il vaut $a e_1 + b e_2 + \lambda(1,1,1,0)$,
ce qui force $\lambda = -a = -b$ et $d = 0$~: l'intersection est
$\operatorname{Vect}(e_3)$ --- cohérent, puisque $e_3$ a été exhibé
dans $F$ ci-dessus et appartient à $G$ par définition. Division du
travail typique~: Grassmann prédit \emph{combien} il faut chercher,
le système linéaire trouve ensuite \emph{quoi}.
\end{example}

\begin{example}[Intersecter par les équations]
\label{ex:b1:findim:intersectequations}
Quand les deux \omterm{def:b1:vspaces:subspace}{sous-espaces} sont donnés comme \omterm{def:b1:logic:sets}{ensembles} de solutions,
intersecter revient simplement à empiler les équations. Dans $\R^3$~:
$F = \{x + y + z = 0\}$ et $G = \{x = y\}$ donnent
\[
F \cap G = \{x = y,\ 2x + z = 0\}
= \{(x,\ x,\ -2x)\} = \operatorname{Vect}(1, 1, -2),
\]
une droite. Contre-vérification par Grassmann~: $F + G = \R^3$ (les
plans sont distincts, donc leur somme contient strictement un plan),
d'où $\dim(F\cap G) = 2 + 2 - 3 = 1$. Les deux descriptions d'un
\omterm{def:b1:vspaces:subspace}{sous-espace} --- par équations, par générateurs --- rendent chacune une
opération triviale~: les équations s'intersectent en s'empilant, les
générateurs se somment en se concaténant~; passer des unes aux autres
est exactement ce que signifie résoudre un système linéaire
(\cref{ch:b1:det}).
\end{example}

\begin{remark}[Pièges courants]
\label{rem:b1:findim:pitfalls}
\emph{Les dimensions ne s'ajoutent pas le long des sommes} sauf si la
somme est directe~: deux plans de $\R^3$ vérifient $\dim(F + G) = 3$,
et non $4$~; toujours corriger par le terme d'intersection
(Grassmann).
\emph{Une inclusion demande la dimension \emph{et} l'inclusion}~:
$\dim F = \dim G$ seul ne donne jamais $F = G$ (deux droites
distinctes de $\R^2$)~; le cas d'égalité du
\cref{thm:b1:findim:subspaces} exige d'abord $F \subseteq G$.
\emph{Compter des paramètres n'est pas encore une démonstration}~:
«~deux équations dans $\R^4$, donc dimension $2$~» est faux quand les
équations sont dépendantes ($x + y = 0$ et $2x + 2y = 0$ laissent la
dimension $3$)~; seul un paramétrage effectif ou un calcul de \omterm{def:b1:findim:rank}{rang}
décide.
\emph{Ne pas parler de la dimension d'un \omterm{def:b1:logic:sets}{ensemble} qui n'est pas un
\omterm{def:b1:vspaces:subspace}{sous-espace}}~: les \omterm{def:b1:logic:sets}{ensembles} de solutions des systèmes
\emph{non homogènes} ne contiennent pas $0$~; leur «~dimension~» est
celle de l'espace des solutions du système homogène associé
(le \cref{ch:b1:det} rendra cela précis).
\emph{La dimension infinie existe}~: $K[X]$ contient des \omterm{def:b1:vspaces:free}{familles
libres} de toutes les tailles (les monômes), donc aucune famille
génératrice finie ne peut exister --- des énoncés comme la règle
«~deux sur trois~» sont strictement \omterm{def:b1:findim:def}{de dimension finie} et échouent
lourdement sur $K[X]$
(le \cref{cor:b1:linmaps:samedim} montrera la même chose pour
l'\omterm{def:b1:logic:inj}{injectivité} et la \omterm{def:b1:logic:inj}{surjectivité}).
\emph{Le \omterm{def:b1:findim:rank}{rang} porte sur le \omterm{def:b1:vspaces:span}{sous-espace engendré}, pas sur la liste}~:
répéter un vecteur, réordonner, ou multiplier par des constantes non
nulles laisse le \omterm{def:b1:findim:rank}{rang} inchangé, et $\operatorname{rk} = p$ (le nombre
de vecteurs) est une \emph{propriété à démontrer} --- c'est
exactement la liberté. Une famille de $5$ vecteurs de \omterm{def:b1:findim:rank}{rang} $2$ porte
l'équivalent de trois vecteurs de redondance, que l'élimination
(\cref{ex:b1:findim:rankcomputation}) localise explicitement.
\end{remark}

\begin{remark}[Où la dimension se met au travail]
\label{rem:b1:findim:whereused}
La dimension est désormais l'argument de comptage préféré du livre.
Le \cref{ch:b1:linmaps} démontre le théorème du \omterm{def:b1:findim:rank}{rang}, version
fonctionnelle de la \omterm{thm:b1:findim:grassmann}{formule de Grassmann}~; le \cref{ch:b1:matrices}
calcule les \omterm{def:b1:findim:rank}{rangs} par opérations sur les lignes~; le \cref{ch:b1:det}
transforme «~$n$ vecteurs de $K^n$ forment une \omterm{def:b1:vspaces:free}{base}~» en un seul
nombre non nul. Le devoir maison ci-dessous montre la dimension
faisant de l'\emph{arithmétique}~: compter des dimensions sur le \omterm{def:b1:structures:field}{corps}
$\Q$ démontre des énoncés d'irrationalité qui paraissent inattaquables
à la main. Dans le volume de Licence~3, les mêmes comptes de
dimension, affinés par la théorie des \omterm{def:b1:structures:group}{groupes}, décident quels
problèmes classiques de construction sont solubles --- cette histoire
s'appelle la théorie de Galois.
\end{remark}

\begin{remark}[Perspectives à l'intérieur du livre 3]
\label{rem:b1:findim:perspectives}
La dimension est la quantité conservée de tout le reste de ce volume,
et il vaut la peine d'en nommer les lois de conservation à l'avance.
Le \cref{ch:b1:linmaps} démontre $\dim E = \dim\ker u +
\operatorname{rk} u$~: ce qu'une \omterm{def:b1:logic:map}{application} linéaire écrase plus ce
qu'elle conserve fait toujours l'espace de départ. Le
\cref{ch:b1:det} raffine cela en la structure des \omterm{def:b1:logic:sets}{ensembles} de
solutions~: $p$ inconnues moins $\operatorname{rk} A$ pivots laissent
la dimension de l'espace des solutions, que le pivot de Gauss exhibe
comme paramètres \omterm{def:b1:vspaces:free}{libres}. Le \cref{ch:b1:euclid} scinde $\dim E = \dim
F + \dim F^\perp$ orthogonalement, et le devoir maison du
\cref{ch:b1:multivar} dépense exactement ce budget~: $n$ points de
données, $2$ paramètres ajustés, $n - 2$ dimensions de résidus.
Chaque fois qu'un compte refuse de tomber juste dans un chapitre
ultérieur, l'erreur est un noyau oublié ou une somme non directe ---
revenir d'abord à la \omterm{thm:b1:findim:grassmann}{formule de Grassmann}.
\end{remark}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:b1:findim:1}
Donner une \omterm{def:b1:vspaces:free}{base} et la dimension de~:
\begin{enumerate}
  \item $F = \{(x,y,z) \in \R^3 : x + y + z = 0\}$~;
  \item $G = \{(x,y,z,t) \in \R^4 : x = y,\ z = 2t\}$~;
  \item $H = \{P \in \R_3[X] : P(1) = P'(1) = 0\}$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:findim:2}
Calculer le \omterm{def:b1:findim:rank}{rang} de la famille
$\bigl((1,1,1), (1,2,3), (3,5,7), (0,1,2)\bigr)$ dans $\R^3$, et
extraire une \omterm{def:b1:vspaces:free}{base} du \omterm{def:b1:vspaces:subspace}{sous-espace} qu'elle engendre.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:findim:3}
Compléter la \omterm{def:b1:vspaces:free}{famille libre} $\bigl((1,1,0,0), (0,0,1,1)\bigr)$ en une
\omterm{def:b1:vspaces:free}{base} de $\R^4$ à l'aide de vecteurs canoniques, et justifier.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:findim:4}
Démontrer que $\bigl(1, X, X(X-1), X(X-1)(X-2)\bigr)$ est une \omterm{def:b1:vspaces:free}{base} de
$\R_3[X]$, et trouver les \omterm{prop:b1:vspaces:coordinates}{coordonnées} de $X^3$ dans cette \omterm{def:b1:vspaces:free}{base}.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:findim:5}
Soient $F$ et $G$ des \omterm{def:b1:vspaces:subspace}{sous-espaces vectoriels} de dimensions $4$ et $5$
d'un espace $E$ avec $\dim E = 7$. Quelles sont les valeurs possibles
de $\dim(F \cap G)$~? Donner un exemple réalisant chaque valeur avec
$E = \R^7$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:findim:6}
Soit $H = \{P \in \R_n[X] : P(1) = 0\}$. Démontrer que $H$ est un
hyperplan de $\R_n[X]$ (un \omterm{def:b1:vspaces:subspace}{sous-espace} de dimension $n$), exhiber une
\omterm{def:b1:vspaces:free}{base} de $H$ \emph{(penser au théorème du facteur~: $P = (X-1)Q$)}, et
donner une droite \omterm{def:b1:vspaces:sum}{supplémentaire}.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:findim:7}
Soient $u_1, \dots, u_p$ des vecteurs de \omterm{def:b1:findim:rank}{rang} $r$. Démontrer que
retirer un vecteur donne une famille de \omterm{def:b1:findim:rank}{rang} $r$ ou $r - 1$, et
qu'adjoindre un vecteur donne un \omterm{def:b1:findim:rank}{rang} $r$ ou $r + 1$. En déduire que
le \omterm{def:b1:findim:rank}{rang} varie d'au plus $1$ sous toute insertion ou suppression
isolée.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:findim:8}
Soient $F_1 \subseteq F_2 \subseteq \dots \subseteq F_k$ des
\omterm{def:b1:vspaces:subspace}{sous-espaces} de $E$ ($\dim E = n$) avec $F_i \neq F_{i+1}$ pour tout
$i$. Démontrer que $k \leq n + 1$. En déduire qu'une chaîne
strictement croissante de \omterm{def:b1:vspaces:subspace}{sous-espaces} de $\R^n$ a une longueur au
plus $n + 1$, et en exhiber une de longueur maximale.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:findim:9}
Soient $E$ de dimension $n$ et $F$, $G$ deux hyperplans (de dimension
$n - 1$), $F \neq G$. Calculer $\dim(F \cap G)$. Généraliser~:
l'intersection de $k$ hyperplans a une dimension $\geq n - k$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:findim:10}
Soit $E$ l'\omterm{def:b1:logic:sets}{ensemble} des suites réelles vérifiant $u_{n+2} =
3u_{n+1} - 2u_n$ pour tout $n$.
\begin{enumerate}
  \item Montrer que $E$ est un \omterm{def:b1:vspaces:subspace}{sous-espace} de l'espace des suites, et
        qu'une suite de $E$ est entièrement déterminée, linéairement,
        par le couple $(u_0, u_1)$~; en déduire $\dim E = 2$.
  \item Vérifier que la suite constante $(1)$ et la suite géométrique
        $(2^n)$ appartiennent à $E$ et forment une \omterm{def:b1:vspaces:free}{base} de $E$.
  \item Trouver la suite de $E$ telle que $u_0 = 0$, $u_1 = 1$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:findim:11}
Soit $E$ de dimension $n$.
\begin{enumerate}
  \item Si $F, G$ sont des \omterm{def:b1:vspaces:subspace}{sous-espaces} avec $\dim F + \dim G > n$,
        démontrer que $F \cap G \neq \{0\}$. Illustrer~: deux
        \omterm{def:b1:vspaces:subspace}{sous-espaces} de dimensions $51$ et $50$ de $\R^{100}$
        partagent toujours un vecteur non nul.
  \item Si $H$ est un hyperplan et $F$ un \omterm{def:b1:vspaces:subspace}{sous-espace} avec $F \cap H =
        \{0\}$, démontrer que $\dim F \leq 1$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:findim:12}
(\omterm{def:b1:vspaces:sum}{Supplémentaire} commun) Soient $F, G$ des \omterm{def:b1:vspaces:subspace}{sous-espaces} de $E$ (\omterm{def:b1:findim:def}{de
dimension finie}) avec $\dim F = \dim G$. Démontrer que $F$ et $G$
admettent un \omterm{def:b1:vspaces:sum}{supplémentaire} \emph{commun}~: il existe un \omterm{def:b1:vspaces:subspace}{sous-espace}
$S$ tel que $E = F \oplus S = G \oplus S$. \emph{(Récurrence
descendante sur $\dim F$~: si $F \neq G \neq E$, choisir $x \notin F
\cup G$ --- l'\cref{exo:b1:vspaces:12} l'autorise --- et considérer $F
\oplus \operatorname{Vect}(x)$ et $G \oplus \operatorname{Vect}(x)$.)}
\end{exercise}

\section{Problème~: la formule des degrés de Dedekind}

\begin{problem}\label{pb:b1:findim:1}
Rien, dans les chapitres 18 et 19, n'a fait appel aux scalaires au-delà
des axiomes de \omterm{def:b1:structures:field}{corps} (\cref{def:b1:structures:field})~: on peut donc
prendre $K = \Q$ et mesurer des \omterm{def:b1:logic:sets}{ensembles} de \emph{nombres réels} à
l'aune de la $\Q$-dimension. Ce problème calcule la
dimension de $\Q(\sqrt2, \sqrt3)$, démontre la \emph{\omterm{pb:b1:findim:1}{formule des
degrés} de Dedekind} $\dim_\Q M = \dim_\Q K \cdot \dim_K M$, et
récolte des théorèmes d'irrationalité par pur comptage de dimensions
--- pas de $\varepsilon$, pas de décimales, rien que des
\omterm{def:b1:vspaces:free}{bases}.\index{formule des degrés}
\index{corps!sous-corps de $\R$}\index{nombre irrationnel}

\medskip
\noindent\textbf{Partie I --- Scalaires rationnels.}
\begin{enumerate}
  \item Vérifier que $\Q$ est un \omterm{def:b1:structures:field}{corps} et que toutes les définitions
        et démonstrations des chapitres 18 et 19 n'utilisent que les
        axiomes de \omterm{def:b1:structures:field}{corps} des scalaires~; en conclure que $\R$ est un
        $\Q$-espace vectoriel et que le lemme d'échange, les théorèmes
        sur les \omterm{def:b1:vspaces:free}{bases} et la \omterm{thm:b1:findim:grassmann}{formule de Grassmann} valent sur $\Q$.
        Repérer l'unique étape de la démonstration du
        \cref{thm:b1:findim:exchange} où l'on divise par un scalaire
        non nul.
  \item Montrer que $(1, \sqrt2)$ est \omterm{def:b1:vspaces:free}{libre} sur $\Q$ mais liée sur
        $\R$. Poser $\Q(\sqrt2) = \operatorname{Vect}_\Q(1,
        \sqrt2) = \{a + b\sqrt2 : a, b \in \Q\}$~; que vaut
        $\dim_\Q \Q(\sqrt2)$~?
  \item Montrer que $\Q(\sqrt2)$ est stable par multiplication.
        Pour $u = a + b\sqrt2$, poser $\sigma(u) = a - b\sqrt2$ et
        $N(u) = u\,\sigma(u) = a^2 - 2b^2$. Montrer que $\sigma(uv) =
        \sigma(u)\sigma(v)$, en déduire $N(uv) = N(u)N(v)$, et montrer
        que $N(u) \neq 0$ dès que $u \neq 0$.
  \item En déduire que tout $u \in \Q(\sqrt2)$ non nul a son inverse
        dans $\Q(\sqrt2)$, à savoir $u^{-1} =
        \sigma(u)/N(u)$~: $\Q(\sqrt2)$ est un \omterm{def:b1:structures:field}{sous-corps} de $\R$.
        Calculer $\dfrac1{3 + 2\sqrt2}$ et $\dfrac1{1 + \sqrt2}$.
\end{enumerate}

\medskip
\noindent\textbf{Partie II --- Adjonction de $\sqrt3$.}
\begin{enumerate}[resume]
  \item Démontrer que $\sqrt6 \notin \Q$ \emph{(comparer l'exposant
        de $2$ des deux côtés de $6q^2 = p^2$, comme dans
        l'\cref{exo:b1:arith:7})}, puis que $\sqrt3 \notin
        \Q(\sqrt2)$ \emph{(élever $\sqrt3 = a + b\sqrt2$ au carré et
        discuter les cas $ab \neq 0$, $b = 0$, $a = 0$)}.
  \item Soit $M = \operatorname{Vect}_\Q(1, \sqrt2, \sqrt3,
        \sqrt6)$. Montrer que $M$ est stable par multiplication
        \emph{(une table des produits des vecteurs de \omterm{def:b1:vspaces:free}{base}
        suffit)}.
  \item On pose $K = \Q(\sqrt2)$. Montrer que $(1, \sqrt3)$ est \omterm{def:b1:vspaces:free}{libre}
        \emph{sur le \omterm{def:b1:structures:field}{corps} $K$}, et en déduire que $M = K +
        K\sqrt3$ est un $K$-espace vectoriel de dimension $2$, de
        \omterm{def:b1:vspaces:free}{base} $(1, \sqrt3)$.
  \item Démontrer que $(1, \sqrt2, \sqrt3, \sqrt6)$ est \omterm{def:b1:vspaces:free}{libre} sur
        $\Q$, donc $\dim_\Q M = 4$. \emph{(Regrouper une relation
        nulle sous la forme $(a + b\sqrt2) + (c + d\sqrt2)\sqrt3 = 0$
        et appliquer les questions 7 puis 2.)}
\end{enumerate}

\medskip
\noindent\textbf{Partie III --- La \omterm{pb:b1:findim:1}{formule des degrés}.} Soient $\Q
\subseteq K \subseteq M \subseteq \R$ où $K$ et $M$ sont des
\omterm{def:b1:structures:field}{sous-corps}, $(e_1, \dots, e_m)$ une \omterm{def:b1:vspaces:free}{base} de $K$ comme $\Q$-espace
vectoriel, et $(f_1, \dots, f_n)$ une \omterm{def:b1:vspaces:free}{base} de $M$ comme $K$-espace
vectoriel.
\begin{enumerate}[resume]
  \item Montrer que les $mn$ produits $(e_i f_j)$ engendrent $M$ sur
        $\Q$.
  \item Montrer que la famille $(e_i f_j)$ est \omterm{def:b1:vspaces:free}{libre} sur $\Q$.
        \emph{(Réorganiser une $\Q$-combinaison nulle sous la forme
        $\sum_j \bigl(\sum_i \lambda_{ij} e_i\bigr) f_j$, dont les
        coefficients \omterm{def:b1:topology:closure}{intérieurs} vivent dans $K$.)}
  \item Conclure par la \emph{\omterm{pb:b1:findim:1}{formule des degrés} de Dedekind}~:
        \[
        \dim_\Q M \;=\; \dim_\Q K \,\cdot\, \dim_K M ,
        \]
        et la vérifier sur $\Q \subseteq \Q(\sqrt2) \subseteq M$
        à l'aide des questions 7 et 8.
  \item (Des \omterm{def:b1:structures:field}{corps} gratuitement) Soit $A \subseteq \R$ un
        $\Q$-sous-espace \omterm{def:b1:findim:def}{de dimension finie} contenant $1$ et
        stable par multiplication, et soit $u \in A$, $u \neq
        0$. Montrer que si $(a_1, \dots, a_d)$ est une \omterm{def:b1:vspaces:free}{base} de $A$,
        alors $(u a_1, \dots, u a_d)$ en est encore une~; en déduire
        que $u$ a un inverse \emph{dans $A$}~: $A$ est un \omterm{def:b1:structures:field}{sous-corps}
        de $\R$. Quelles questions antérieures ce résultat
        redémontre-t-il~?
  \item Montrer que pour tout $x \in A$ (comme à la question 12,
        $\dim_\Q A = d$) la famille $(1, x, x^2, \dots, x^{d})$ est
        liée~: tout élément de $A$ est racine d'un \omterm{def:b1:poly:def}{polynôme} non nul à
        coefficients rationnels, de degré au plus $d$.
\end{enumerate}

\medskip
\noindent\textbf{Partie IV --- Un seul nombre engendre tout.}
Posons $s = \sqrt2 + \sqrt3$.
\begin{enumerate}[resume]
  \item Calculer les \omterm{prop:b1:vspaces:coordinates}{coordonnées} de $s^2$, $s^3$ et $s^4$ dans la
        \omterm{def:b1:vspaces:free}{base} $(1, \sqrt2, \sqrt3, \sqrt6)$ de $M$.
  \item Montrer que $(1, s, s^2, s^3)$ est \omterm{def:b1:vspaces:free}{libre} sur $\Q$, et en
        déduire que $\operatorname{Vect}_\Q(1, s, s^2, s^3) = M$~:
        tout élément de $M$ est un \omterm{def:b1:poly:def}{polynôme} à coefficients rationnels
        en $s$. Exprimer $\sqrt2$ et $\sqrt3$ comme de tels \omterm{def:b1:poly:def}{polynômes}.
  \item Vérifier que $s^4 - 10s^2 + 1 = 0$, et montrer que $X^4 -
        10X^2 + 1$ est le \emph{\omterm{def:b1:poly:def}{polynôme unitaire} de plus petit
        degré} s'annulant en $s$. Déterminer ses quatre racines
        réelles.
  \item En déduire~: $1/s = 10s - s^3$~; identifier ce nombre. Montrer
        que $a + b\sqrt2 + c\sqrt3 + d\sqrt6$ ($a,b,c,d \in \Q$)
        est rationnel si et seulement si $b = c = d = 0$~; en
        particulier $\sqrt2 + \sqrt3 + \sqrt6$ est irrationnel
        (ce qui renforce l'\cref{exo:b1:reals:7}).
\end{enumerate}

\medskip
\noindent\textbf{Partie V --- La racine cubique reste dehors.} Posons
$t = 2^{1/3}$.
\begin{enumerate}[resume]
  \item Montrer que $X^3 - 2$ n'a pas de racine rationnelle \emph{(le
        critère de racine rationnelle de l'\cref{exo:b1:poly:5}, ou un
        comptage de valuations)}~; en déduire que $(1, t)$ est \omterm{def:b1:vspaces:free}{libre}
        sur $\Q$.
  \item Montrer que $(1, t, t^2)$ est \omterm{def:b1:vspaces:free}{libre} sur $\Q$. \emph{(Si un
        $P \in \Q[X]$ non nul de degré $\leq 2$ annule $t$, en prendre
        un de plus petit degré et diviser $X^3 - 2$ par lui,
        \cref{thm:b1:poly:division}~; conclure que $X^3 - 2$ aurait
        une racine rationnelle.)} En déduire que $A =
        \operatorname{Vect}_\Q(1, t, t^2)$ est de dimension $3$, est
        stable par multiplication, et est un \omterm{def:b1:structures:field}{sous-corps} de $\R$.
  \item Démontrer que $t \notin M$~: sinon $M$ serait un \omterm{def:b1:vspaces:def}{espace
        vectoriel} sur le \omterm{def:b1:structures:field}{corps} $A$, et la \omterm{pb:b1:findim:1}{formule des degrés} forcerait
        $3 \mid 4$. Donc $2^{1/3}$ n'est pas une combinaison
        rationnelle de $1, \sqrt2, \sqrt3, \sqrt6$.
  \item En déduire que $t \notin \Q(\sqrt2)$, que $t + \sqrt2$ est
        irrationnel, et qu'aucun couple de rationnels $a, b$ ne vérifie
        $2^{1/3} = a + b\sqrt3$.
\end{enumerate}

\medskip
\noindent\textbf{Partie VI --- Tous les \omterm{def:b1:structures:field}{sous-corps}, et synthèse.}
\begin{enumerate}[resume]
  \item Démontrer la \omterm{def:b1:arith:divides}{divisibilité} des degrés~: si $\Q \subseteq A
        \subseteq B \subseteq \R$ sont des \omterm{def:b1:structures:field}{sous-corps} avec $\dim_\Q B$
        finie, alors $\dim_\Q A$ \omterm{def:b1:arith:divides}{divise} $\dim_\Q B$. Quelles sont les
        dimensions possibles des \omterm{def:b1:structures:field}{sous-corps} de $M$~?
  \item Déterminer \emph{tous} les \omterm{def:b1:structures:field}{sous-corps} de $M$ de dimension $2$
        sur $\Q$. \emph{(Se ramener à trouver les $w = a + b\sqrt2
        + c\sqrt3 + d\sqrt6 \in M \setminus \Q$ tels que $w^2 \in
        \Q$~: développer $w^2$ et annuler les trois \omterm{prop:b1:vspaces:coordinates}{coordonnées}
        irrationnelles.)}
  \item Exprimer $\dfrac1{1 + \sqrt2 + \sqrt3}$ dans la \omterm{def:b1:vspaces:free}{base} $(1,
        \sqrt2, \sqrt3, \sqrt6)$ \emph{(multiplier par des conjugués
        bien choisis)}.
  \item Synthèse, en quatre phrases~: pourquoi la $\Q$-dimension est
        un invariant \emph{arithmétique} d'un \omterm{def:b1:logic:sets}{ensemble} de réels~; ce
        que la finitude de la dimension impose à chaque élément
        (question 13)~; comment la règle «~deux sur trois~» a produit
        des inverses à partir de rien (question 12)~; et ce que la
        \omterm{pb:b1:findim:1}{formule des degrés} interdit (question 20). Nommer le théorème
        démontré dans la partie III.
\end{enumerate}
\end{problem}
