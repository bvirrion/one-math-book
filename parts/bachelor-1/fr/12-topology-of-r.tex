\chapter{Topologie de la droite réelle}\label{ch:b1:topology}

Les limites renvoient sans cesse au même vocabulaire géométrique~: des
points «~proches~» d'un \omterm{def:b1:logic:sets}{ensemble}, des \omterm{def:b1:logic:sets}{ensembles} «~sans fuite par la
\omterm{def:b1:topology:closure}{frontière}~», des \omterm{prop:b1:reals:intervals}{intervalles} dont les suites ne peuvent s'échapper. Ce
chapitre fixe ce vocabulaire --- \omterm{def:b1:topology:open}{ouverts} et \omterm{def:b1:topology:closed}{fermés}, \omterm{def:b1:topology:closure}{intérieur} et
\omterm{def:b1:topology:closure}{adhérence}, \omterm{def:b1:topology:dense}{densité} --- sur la droite réelle, et démontre la compacité
des segments sous sa forme séquentielle. Les mêmes notions, dans les
espaces vectoriels normés, relèvent de la deuxième année~; sur $\R$
elles sont à notre portée et immédiatement utiles pour
le \cref{ch:b1:continuity}.

\section{Ouverts, fermés}

\begin{definition}[Voisinage, ouvert]\label{def:b1:topology:open}
Une partie $V \subseteq \R$ est un \emph{voisinage}\index{voisinage}
de $x \in \R$ lorsqu'elle contient un \omterm{prop:b1:reals:intervals}{intervalle} $\intoo{x - r}{x + r}$
pour un certain $r > 0$. Une partie $U \subseteq \R$ est
\emph{ouverte}\index{ouvert} lorsqu'elle est un voisinage de chacun de
ses points~:
\[
\forall x \in U,\ \exists r > 0, \quad \intoo{x - r}{x + r}
\subseteq U .
\]
\end{definition}

\begin{example}\label{ex:b1:topology:open}
Les \omterm{prop:b1:reals:intervals}{intervalles} \omterm{def:b1:topology:open}{ouverts} sont des \omterm{def:b1:topology:open}{ouverts}~: pour $x \in \intoo{a}{b}$,
prendre $r = \min(x - a,\, b - x) > 0$. Les demi-droites
$\intoo{a}{+\infty}$ sont \omterm{def:b1:topology:open}{ouvertes}~; $\R$ et $\emptyset$ sont \omterm{def:b1:topology:open}{ouverts}
(le second trivialement). $\intcc{0}{1}$ n'est \emph{pas} \omterm{def:b1:topology:open}{ouvert}~:
aucun \omterm{prop:b1:reals:intervals}{intervalle} autour de $0$ n'y reste contenu.
\end{example}

\begin{proposition}[Stabilité des ouverts]\label{prop:b1:topology:openstable}
Toute réunion d'\omterm{def:b1:topology:open}{ouverts} est un \omterm{def:b1:topology:open}{ouvert}~; une intersection \emph{finie}
d'\omterm{def:b1:topology:open}{ouverts} est un \omterm{def:b1:topology:open}{ouvert}. Les intersections infinies peuvent échouer~:
$\bigcap_{n \geq 1} \intoo{-\frac 1n}{\frac 1n} = \{0\}$, qui n'est pas \omterm{def:b1:topology:open}{ouvert}.
\end{proposition}

\begin{proof}
Réunion~: si $x \in \bigcup_i U_i$, alors $x \in U_{i_0}$ pour un
certain $i_0$, et l'\omterm{prop:b1:reals:intervals}{intervalle} fourni par $U_{i_0}$ est contenu dans la
réunion. Intersection finie~: si $x \in U_1 \cap \dots \cap U_k$,
prendre $r = \min(r_1, \dots, r_k) > 0$ parmi les rayons fournis par
chaque $U_j$. Pour le contre-exemple~: tout \omterm{prop:b1:reals:intervals}{intervalle} autour de $0$
contient un $\frac{1}{n}$ (Archimède), donc sort de l'intersection.
\end{proof}

\begin{example}[Certifier l'ouverture par des rayons explicites]\label{ex:b1:topology:radii}
L'\omterm{def:b1:logic:sets}{ensemble} $U = \{x \in \R : x^2 > 2\}$ est-il \omterm{def:b1:topology:open}{ouvert}~? Oui, et le
certificat peut s'écrire explicitement~: $U = \intoo{-\infty}{-\sqrt2}
\cup \intoo{\sqrt2}{+\infty}$, réunion de deux demi-droites
\omterm{def:b1:topology:open}{ouvertes}, \omterm{def:b1:topology:open}{ouverte} d'après la \cref{prop:b1:topology:openstable}. On peut
aussi raisonner point par point~: pour $x \in U$ avec $x > \sqrt 2$,
prendre $r = x - \sqrt2 > 0$~: tout $y \in \intoo{x - r}{x + r}$
vérifie $y > \sqrt 2$, donc $y^2 > 2$~; symétriquement à gauche.
Les deux styles comptent --- le style structurel (construire à
partir d'\omterm{def:b1:topology:open}{ouverts} connus par réunions et intersections finies) passe
mieux à l'échelle, le style en $\varepsilon$ fonctionne lorsque aucune
structure n'est visible~; et le \cref{ch:b1:continuity} en ajoutera un
troisième, le plus puissant~: $U$ est l'\omterm{def:b1:logic:map}{image réciproque} de l'\omterm{def:b1:topology:open}{ouvert}
$\intoo{2}{+\infty}$ par l'\omterm{def:b1:logic:map}{application} continue $x \mapsto x^2$.
\end{example}

\begin{definition}[Fermé]\label{def:b1:topology:closed}
Une partie $F \subseteq \R$ est \emph{fermée}\index{fermé} lorsque son
complémentaire $\R \setminus F$ est \omterm{def:b1:topology:open}{ouvert}. Par les lois de de Morgan
et la \cref{prop:b1:topology:openstable}~: toute intersection de fermés
est fermée, toute réunion finie de fermés est fermée.
\end{definition}

\begin{theorem}[Caractérisation séquentielle des fermés]\label{thm:b1:topology:seqclosed}
$F$ est \omterm{def:b1:topology:closed}{fermé} si et seulement si~: pour toute suite $(u_n)$ de points
de $F$ convergeant vers un $\ell \in \R$, la limite $\ell$ appartient à
$F$. («~\omterm{def:b1:topology:closed}{Fermé}~» $=$ «~stable par passage à la limite~».)
\end{theorem}

\begin{proof}
($\Rightarrow$) Soit $F$ \omterm{def:b1:topology:closed}{fermé}, $u_n \in F$, $u_n \to \ell$, et
supposons $\ell \notin F$. Le complémentaire est \omterm{def:b1:topology:open}{ouvert}~: un certain
$\intoo{\ell - r}{\ell + r}$ évite $F$. Mais la convergence place $u_n$
dans cet \omterm{prop:b1:reals:intervals}{intervalle} pour $n$ grand~: contradiction avec $u_n \in F$.

($\Leftarrow$) Supposons $F$ non \omterm{def:b1:topology:closed}{fermé}~: le complémentaire n'est pas
\omterm{def:b1:topology:open}{ouvert}, donc il existe $x \notin F$ tel qu'aucun \omterm{prop:b1:reals:intervals}{intervalle}
$\intoo{x - r}{x + r}$ ne soit contenu dans le complémentaire~; en
prenant $r = \frac{1}{n+1}$, choisissons $u_n \in F$ avec
$\abs{u_n - x} < \frac{1}{n+1}$. Alors $u_n \in F$, $u_n \to x \notin
F$~: la propriété séquentielle est en défaut.
\end{proof}

\begin{example}[Le test séquentiel, dans les deux sens]\label{ex:b1:topology:seqtest}
\emph{\omterm{def:b1:topology:closed}{Fermé}~:} $F = \Z \cup \bigl\{n + \frac1n : n \geq 2\bigr\}$.
Soit $u_k \in F$ avec $u_k \to \ell$. La fenêtre $\intcc{\ell -
1}{\ell + 1}$ ne contient qu'un nombre fini de points de $F$ (un
nombre fini d'entiers, un nombre fini de $n + \frac1n$), et à
partir d'un certain rang tous les $u_k$ y sont~: la suite ne prend
alors qu'un nombre fini de valeurs et, convergeant, est
stationnaire (comme dans l'\cref{exo:b1:topology:3})~: $\ell \in F$.
\omterm{def:b1:topology:closed}{Fermé} --- bien que $F$ contienne des couples de points à distance
$\frac1n$, arbitrairement proches.

\emph{Non \omterm{def:b1:topology:closed}{fermé}~:} $G = \bigl\{\frac1m + \frac1n : m, n \in
\N^*\bigr\}$. La suite $\frac1n + \frac1n \in G$ tend vers $0$, et
$0 \notin G$ (somme de deux termes strictement positifs)~: le test
séquentiel échoue, $G$ n'est pas \omterm{def:b1:topology:closed}{fermé}. Fait remarquable, chaque
$\frac1m$ appartient \emph{bel et bien} à $\overline G \cap G$~: en
effet $\frac1m = \frac{1}{m+1} + \frac{1}{m(m+1)} \in G$. L'idée à
retenir~: pour démontrer qu'un \omterm{def:b1:logic:sets}{ensemble} est \omterm{def:b1:topology:closed}{fermé}, il faut
contrôler \emph{toutes} les suites convergentes d'un coup (en
général par un argument de finitude locale ou de formule \omterm{def:b1:topology:closed}{fermée})~;
pour le réfuter, une seule suite fuyante bien choisie suffit ---
cette asymétrie fait du sens négatif le plus facile, et les
contre-exemples de ce chapitre ont tous cette forme en une ligne.
\end{example}

\begin{example}\label{ex:b1:topology:closed}
Les segments $\intcc{a}{b}$, les demi-droites $\intco{a}{+\infty}$,
les \omterm{def:b1:counting:card}{ensembles finis}, $\Z$ (une suite convergente d'entiers est
stationnaire) sont \omterm{def:b1:topology:closed}{fermés}. $\intoc{0}{1}$ n'est ni \omterm{def:b1:topology:open}{ouvert} (cela
échoue en $1$) ni \omterm{def:b1:topology:closed}{fermé} ($\frac 1n \to 0 \notin$ l'\omterm{def:b1:logic:sets}{ensemble})~: la
plupart des parties ne sont ni l'un ni l'autre. $\R$ et $\emptyset$
sont à la fois \omterm{def:b1:topology:open}{ouverts} et \omterm{def:b1:topology:closed}{fermés} --- et ce sont les seules parties de
$\R$ dans ce cas (\cref{exo:b1:topology:9}).
\end{example}

\begin{example}[Un ouvert assemblé à partir d'une infinité de morceaux]\label{ex:b1:topology:rminusz}
$\R \setminus \Z = \bigcup_{n \in \Z} \intoo{n}{n+1}$~: une
réunion infinie d'\omterm{prop:b1:reals:intervals}{intervalles} \omterm{def:b1:topology:open}{ouverts}, \omterm{def:b1:topology:open}{ouverte} d'après
la \cref{prop:b1:topology:openstable} --- donc $\Z$ est \omterm{def:b1:topology:closed}{fermé} sans le
moindre argument séquentiel. Notons le partage des tâches dans
les règles de stabilité~: les \emph{réunions} d'\omterm{def:b1:topology:open}{ouverts} peuvent
être quelconques (chaque point n'a besoin que de son propre
certificat, fourni par l'unique \omterm{def:b1:logic:sets}{ensemble} qui le contient), tandis
que les \emph{intersections} doivent rester finies (les
certificats doivent être intersectés, et une infinité de rayons
peuvent se contracter jusqu'à rien). \Cref{exo:b1:topology:10}
montrera que cet exemple est la forme générale~: tout \omterm{def:b1:topology:open}{ouvert} de
$\R$ est une réunion dénombrable disjointe d'\omterm{prop:b1:reals:intervals}{intervalles} \omterm{def:b1:topology:open}{ouverts}.
\end{example}

\section{Intérieur, adhérence, densité}

\begin{definition}[Intérieur, adhérence, frontière]\label{def:b1:topology:closure}
Soit $A \subseteq \R$.
\begin{itemize}
  \item Un point $x$ est \emph{intérieur} à $A$ lorsque $A$ est un
        \omterm{def:b1:topology:open}{voisinage} de $x$~; l'\emph{intérieur}
        $\mathring{A}$\index{intérieur} est l'\omterm{def:b1:logic:sets}{ensemble} des points
        intérieurs.
  \item Un point $x$ est \emph{adhérent} à $A$ lorsque tout \omterm{def:b1:topology:open}{voisinage}
        de $x$ rencontre $A$~; l'\emph{adhérence}
        $\overline{A}$\index{adhérence} est l'\omterm{def:b1:logic:sets}{ensemble} des points
        adhérents.
  \item La \emph{frontière} est $\partial A = \overline A \setminus
        \mathring A$.\index{frontière}
\end{itemize}
On a alors $\mathring A \subseteq A \subseteq \overline A$.
\end{definition}

\begin{proposition}[Propriétés principales]\label{prop:b1:topology:closureprops}
\begin{enumerate}
  \item $\mathring A$ est le plus grand \omterm{def:b1:topology:open}{ouvert} contenu dans $A$~; $A$
        est \omterm{def:b1:topology:open}{ouvert} si et seulement si $A = \mathring A$.
  \item $\overline A$ est le plus petit \omterm{def:b1:topology:closed}{fermé} contenant $A$~; $A$
        est \omterm{def:b1:topology:closed}{fermé} si et seulement si $A = \overline A$.
  \item (Caractérisation séquentielle de l'\omterm{def:b1:topology:closure}{adhérence}) $x \in \overline
        A$ si et seulement si $x$ est limite d'une suite de points de
        $A$.
  \item Le passage au complémentaire échange les notions~: $\R \setminus
        \overline A = \bigl(\R \setminus A\bigr)^{\!\circ}$.
\end{enumerate}
\end{proposition}

\begin{proof}
(4) $x \notin \overline A$ $\iff$ un \omterm{def:b1:topology:open}{voisinage} de $x$ évite $A$
$\iff$ un \omterm{prop:b1:reals:intervals}{intervalle} autour de $x$ est contenu dans $\R \setminus A$
$\iff$ $x$ est \omterm{def:b1:topology:closure}{intérieur} à $\R \setminus A$.

(1) $\mathring A$ est \omterm{def:b1:topology:open}{ouvert}~: si $x \in \mathring A$, un certain
$\intoo{x-r}{x+r} \subseteq A$~; tout point $y$ de cet \omterm{prop:b1:reals:intervals}{intervalle}
admet autour de lui un \omterm{prop:b1:reals:intervals}{intervalle} plus petit contenu dedans, donc
contenu dans $A$~: l'\omterm{prop:b1:reals:intervals}{intervalle} tout entier est dans $\mathring A$.
Tout \omterm{def:b1:topology:open}{ouvert} $U \subseteq A$ est formé de points \omterm{def:b1:topology:closure}{intérieurs} à $A$, donc
$U \subseteq \mathring A$~: c'est bien le plus grand. La
caractérisation des \omterm{def:b1:topology:open}{ouverts} s'en déduit.

(2) En détail. D'après (4), $\R \setminus \overline A$ est
l'\omterm{def:b1:topology:closure}{intérieur} de $\R \setminus A$, un \omterm{def:b1:topology:open}{ouvert} d'après (1)~: donc
$\overline A$ est \omterm{def:b1:topology:closed}{fermé}, et il contient $A$. Minimalité~: soit
$F \supseteq A$ un \omterm{def:b1:topology:closed}{fermé}. Alors $\R \setminus F$ est \omterm{def:b1:topology:open}{ouvert} et contenu
dans $\R \setminus A$, donc par la maximalité de (1),
\[
\R \setminus F \subseteq \bigl(\R \setminus A\bigr)^{\!\circ}
= \R \setminus \overline A ,
\]
et en repassant au complémentaire~: $\overline A \subseteq F$. Ainsi
$\overline A$ est le plus petit \omterm{def:b1:topology:closed}{fermé} contenant $A$. Caractérisation~:
si $A = \overline A$ alors $A$ est \omterm{def:b1:topology:closed}{fermé} (ce qu'on vient de voir)~; si
$A$ est \omterm{def:b1:topology:closed}{fermé}, il est lui-même un \omterm{def:b1:topology:closed}{fermé} contenant $A$, donc la
minimalité impose $\overline A \subseteq A$, d'où l'égalité.

(3) Si $u_n \in A$, $u_n \to x$~: tout \omterm{def:b1:topology:open}{voisinage} de $x$ contient un
$u_n \in A$, donc $x \in \overline A$. Réciproquement, si $x \in
\overline A$~: chaque \omterm{prop:b1:reals:intervals}{intervalle} $\intoo{x - \frac{1}{n+1}}{x +
\frac{1}{n+1}}$ rencontre $A$ en un point $u_n$, et $u_n \to x$.
\end{proof}

\begin{example}\label{ex:b1:topology:closureexamples}
$\overline{\intoo{0}{1}} = \intcc{0}{1}$~;
$\mathring{\intcc{0}{1}} = \intoo{0}{1}$~; $\partial\intoo{0}{1} =
\{0, 1\}$. Pour $A = \{\frac 1n : n \in \N^*\}$~: $\overline A = A
\cup \{0\}$, $\mathring A = \emptyset$, $\partial A = A \cup \{0\}$.
Pour $\Q$~: par \omterm{def:b1:topology:dense}{densité} (\cref{thm:b1:reals:density}) tout réel est
adhérent à $\Q$, donc $\overline{\Q} = \R$ tandis que $\mathring{\Q} =
\emptyset$ (tout \omterm{prop:b1:reals:intervals}{intervalle} contient des irrationnels)~: la \omterm{def:b1:topology:closure}{frontière}
de $\Q$ est $\R$ tout entier.
\end{example}

\begin{example}[Une anatomie complète]\label{ex:b1:topology:anatomy}
Soit $A = \intoc{0}{1} \,\cup\, \bigl(\Q \cap \intoo{2}{3}\bigr)
\,\cup\, \{4\}$. Calculons les trois \omterm{def:b1:logic:sets}{ensembles} de
la \cref{def:b1:topology:closure}, morceau par morceau.

\emph{\omterm{def:b1:topology:closure}{Intérieur}.} Un point de $\intoo{0}{1}$ possède tout un
\omterm{prop:b1:reals:intervals}{intervalle} contenu dans $A$~: il est \omterm{def:b1:topology:closure}{intérieur}. Le point $1$~:
tout \omterm{prop:b1:reals:intervals}{intervalle} autour de lui déborde à droite de $1$, où $A$ ne
contient rien jusqu'à $2$~: il n'est pas \omterm{def:b1:topology:closure}{intérieur}. Aucun point de
$\Q \cap \intoo{2}{3}$ n'est \omterm{def:b1:topology:closure}{intérieur} (tout \omterm{prop:b1:reals:intervals}{intervalle} contient
des irrationnels, \cref{thm:b1:reals:density})~; le point isolé
$4$ non plus. Donc $\mathring A = \intoo{0}{1}$.

\emph{\omterm{def:b1:topology:closure}{Adhérence}.} Les limites de points de $A$~: tout
$\intcc{0}{1}$ ($0 = \lim \frac1n$ avec $\frac 1n \in A$)~; tout
$\intcc{2}{3}$ (tout réel de cet \omterm{prop:b1:reals:intervals}{intervalle} est limite de
rationnels de l'\omterm{prop:b1:reals:intervals}{intervalle}, \omterm{def:b1:topology:dense}{densité} encore)~; et $4$. Rien
d'autre~: un point hors de $\intcc{0}{1} \cup \intcc{2}{3} \cup
\{4\}$ est à distance strictement positive de ce \omterm{def:b1:topology:closed}{fermé}. Donc
$\overline A = \intcc{0}{1} \cup \intcc{2}{3} \cup \{4\}$.

\emph{\omterm{def:b1:topology:closure}{Frontière}.} $\partial A = \overline A \setminus \mathring A
= \{0, 1\} \cup \intcc{2}{3} \cup \{4\}$.

L'idée à retenir~: les trois opérations agissent \emph{localement}
--- chaque morceau de $A$ contribue selon sa nature propre (un
\omterm{prop:b1:reals:intervals}{intervalle} plein garde son \omterm{def:b1:topology:closure}{intérieur}, un morceau \omterm{def:b1:topology:dense}{dense} mais poreux
se transforme entièrement en \omterm{def:b1:topology:closure}{frontière}, un point isolé est
\omterm{def:b1:topology:closure}{frontière} pure), et un croquis de $A$ en deux traits prédit toutes
les réponses avant qu'aucune démonstration ne soit écrite.
\end{example}

\begin{remark}[Pièges classiques du raisonnement topologique]
(i) \emph{«~Non \omterm{def:b1:topology:open}{ouvert}~» ne signifie pas «~\omterm{def:b1:topology:closed}{fermé}~»}~: la plupart
des parties ne sont ni l'un ni l'autre ($\intoc{0}{1}$), et deux
parties sont les deux à la fois ($\emptyset$, $\R$) --- \omterm{def:b1:topology:open}{ouvert} et
\omterm{def:b1:topology:closed}{fermé} ne sont pas des contraires mais des duaux par passage au
complémentaire. (ii) \emph{L'\omterm{def:b1:topology:closure}{intérieur} et l'\omterm{def:b1:topology:closure}{adhérence} ne commutent
pas}~: pour $A = \Q$,
\[
\overline{\mathring A} = \overline\emptyset = \emptyset
\qquad\text{tandis que}\qquad
\bigl(\,\overline A\,\bigr)^{\!\circ} = \mathring \R = \R :
\]
les deux opérateurs itérés diffèrent autant que deux \omterm{def:b1:logic:sets}{ensembles} le
peuvent. (iii) \emph{Une réunion infinie de \omterm{def:b1:topology:closed}{fermés} peut ne pas être
\omterm{def:b1:topology:closed}{fermée}}~: $\bigcup_{n\geq1} \intcc{\frac1n}{1} = \intoc{0}{1}$ ---
le miroir du contre-exemple sur les intersections de
la \cref{prop:b1:topology:openstable}. (iv) \emph{\omterm{def:b1:topology:dense}{Dense} ne veut pas
dire gros}~: $\Q$ est \omterm{def:b1:topology:dense}{dense}, dénombrable, d'\omterm{def:b1:topology:closure}{intérieur} vide, et son
complémentaire est \omterm{def:b1:topology:dense}{dense} lui aussi~; la \omterm{def:b1:topology:dense}{densité} dit
«~arbitrairement proche de tout~», et non «~presque tout~» ---
l'\omterm{pb:b1:topology:1}{ensemble de Cantor} du devoir maison (\cref{pb:b1:topology:1})
illustre le point opposé, un \omterm{def:b1:logic:sets}{ensemble} topologiquement petit et
pourtant non dénombrable, donc énorme.
\end{remark}

\begin{example}[Une adhérence calculée exactement]\label{ex:b1:topology:closurecomputed}
Soit $G = \bigl\{\frac1m + \frac1n : m, n \in \N^*\bigr\}$ (venu de
l'\cref{ex:b1:topology:seqtest}). Affirmation~:
\[
\overline G = G \,\cup\, \Bigl\{\frac1m : m \in \N^*\Bigr\}
\,\cup\, \{0\} .
\]
($\supseteq$) $\frac1m = \lim_n \bigl(\frac1m + \frac1n\bigr)$
et $0 = \lim_n \frac2n$~: adhérents par la caractérisation
séquentielle. ($\subseteq$) Soit $x = \lim_k \bigl(
\frac{1}{m_k} + \frac{1}{n_k}\bigr)$~; ordonnons chaque couple de
sorte que $m_k \leq n_k$. Si $(m_k)$ n'est pas bornée, une \omterm{def:b1:seq:subsequence}{suite
extraite} vérifie $m_k \to \infty$, donc $n_k \to \infty$ aussi et
$x = 0$. Sinon $(m_k)$ ne prend qu'un nombre fini de valeurs,
dont l'une, disons $m$, une infinité de fois~; le long de cette
\omterm{def:b1:seq:subsequence}{suite extraite}, $\frac{1}{n_k} \to x - \frac1m$~: si $(n_k)$ est
bornée, elle prend une valeur $n$ une infinité de fois et $x =
\frac1m + \frac1n \in G$~; sinon, $x = \frac1m$. Tous les cas
retombent dans l'\omterm{def:b1:logic:sets}{ensemble} annoncé. L'idée à retenir~: calculer
une \omterm{def:b1:topology:closure}{adhérence}, c'est mener une disjonction de cas sur les indices
à la manière de la compacité --- un indice borné signifie un
nombre fini de valeurs (\omterm{cor:b1:counting:pigeonhole}{principe des tiroirs}), un indice non
borné signifie qu'une limite s'échappe --- et la réponse fait
apparaître la structure typique à deux étages des points
limites~: l'\omterm{def:b1:logic:sets}{ensemble}, ses limites de première génération, et leur
limite $0$.
\end{example}

\begin{definition}[Densité, forme topologique]\label{def:b1:topology:dense}
$A$ est \emph{dense}\index{densité} dans $\R$ lorsque $\overline A = \R$
--- de façon équivalente, tout \omterm{prop:b1:reals:intervals}{intervalle} \omterm{def:b1:topology:open}{ouvert} non vide rencontre
$A$~; de façon équivalente encore (par
\cref{prop:b1:topology:closureprops}~(3)), tout réel est limite
d'éléments de $A$. Exemples~: $\Q$, $\R \setminus \Q$, les nombres
dyadiques (\cref{exo:b1:reals:8}), les \omterm{def:b1:structures:subgroup}{sous-groupes} denses
(\cref{exo:b1:reals:9}).
\end{definition}

\begin{example}[La densité est relative]\label{ex:b1:topology:relativedensity}
«~\omterm{def:b1:topology:dense}{Dense}~» au sens défini ici signifie \omterm{def:b1:topology:dense}{dense} \emph{dans $\R$}~; un
\omterm{def:b1:logic:sets}{ensemble} peut au contraire n'être \omterm{def:b1:topology:dense}{dense} que dans une partie de la
droite. Les dyadiques de $\intcc{0}{1}$, c'est-à-dire $D \cap
\intcc{0}{1}$ (\cref{exo:b1:reals:8}), rencontrent tout \omterm{prop:b1:reals:intervals}{intervalle}
\omterm{def:b1:topology:open}{ouvert} inclus dans $\intcc{0}{1}$ mais manquent bien sûr
entièrement $\intoo{2}{3}$~: ils sont \omterm{def:b1:topology:dense}{denses} \emph{dans}
$\intcc{0}{1}$, au sens où $\overline{D \cap \intcc{0}{1}} =
\intcc{0}{1}$. La formule générale «~$A$ est \omterm{def:b1:topology:dense}{dense} dans $B$~»
abrège $B \subseteq \overline A$ --- nommez toujours l'\omterm{def:b1:logic:sets}{ensemble}
ambiant, car les extrémités du devoir maison sont \omterm{def:b1:topology:dense}{denses}
\emph{dans l'\omterm{pb:b1:topology:1}{ensemble de Cantor}} tout en étant nulle part \omterm{def:b1:topology:dense}{denses}
\emph{dans $\R$}~: le même \omterm{def:b1:logic:sets}{ensemble}, deux descriptions véridiques
et d'apparence opposée.
\end{example}

\begin{example}[Manipuler la densité]\label{ex:b1:topology:densitytricks}
Trois gestes rapides qui reviennent constamment. \emph{Agrandir}~:
si $A$ est \omterm{def:b1:topology:dense}{dense} et $A \subseteq B$, alors $B$ est \omterm{def:b1:topology:dense}{dense} (tout
\omterm{prop:b1:reals:intervals}{intervalle} rencontre déjà $A$). \emph{Transporter}~: si $A$ est
\omterm{def:b1:topology:dense}{dense}, alors $\lambda A + \mu$ l'est aussi pour $\lambda \neq 0$
--- un \omterm{prop:b1:reals:intervals}{intervalle} $I$ rencontre $\lambda A + \mu$ si et seulement
si l'\omterm{prop:b1:reals:intervals}{intervalle} $\frac{I - \mu}{\lambda}$ rencontre $A$~; ainsi
les multiples impairs de $10^{-9}$, par exemple, sont \omterm{def:b1:topology:dense}{denses}.
\emph{Intersecter échoue}~: deux \omterm{def:b1:logic:sets}{ensembles} \omterm{def:b1:topology:dense}{denses} peuvent se
manquer entièrement ($\Q$ et $\R \setminus \Q$)~: la \omterm{def:b1:topology:dense}{densité}
survit aux réunions et aux \omterm{def:b1:logic:map}{applications} affines, jamais aux
intersections.
\end{example}

\section{Compacité des segments}

\begin{theorem}[Les segments sont séquentiellement compacts]\label{thm:b1:topology:compact}
Soient $a \leq b$. De toute suite de points de $\intcc{a}{b}$ on peut
extraire une suite convergeant vers un point \emph{de}
$\intcc{a}{b}$.\index{compacité}

Plus généralement, les parties de $\R$ possédant cette propriété (de
toute suite on peut extraire une suite convergeant dans la partie)
sont exactement les parties \emph{\omterm{def:b1:topology:closed}{fermées} et bornées}.
\end{theorem}

\begin{proof}
Une suite à valeurs dans $\intcc{a}{b}$ est bornée, donc le
\omterm{thm:b1:seq:bw}{théorème de Bolzano--Weierstrass} permet d'en
extraire une suite convergente (\cref{thm:b1:seq:bw})~; sa limite reste
dans $\intcc{a}{b}$ car les segments sont
\omterm{def:b1:topology:closed}{fermés}
(\cref{thm:b1:topology:seqclosed}).

Cas général. \emph{(\omterm{def:b1:topology:closed}{Fermé} borné $\Rightarrow$ compact)}~: soit
$F$ \omterm{def:b1:topology:closed}{fermé} et borné, et $(u_n)$ une suite de $F$. Le caractère
borné de $F$ borne la suite, donc Bolzano--Weierstrass en extrait
$u_{\varphi(n)} \to \ell$~; et $\ell \in F$ car $F$ est \omterm{def:b1:topology:closed}{fermé} et
la \omterm{def:b1:seq:subsequence}{suite extraite} est une suite convergente de points de $F$
(\cref{thm:b1:topology:seqclosed})~: les deux hypothèses sont
consommées une par une, le caractère borné pour l'existence de la
limite, le caractère \omterm{def:b1:topology:closed}{fermé} pour son appartenance.
\emph{(Compact $\Rightarrow$ \omterm{def:b1:topology:closed}{fermé} et borné)}~: si $F$ n'est pas
borné, choisissons $u_n \in F$ avec $\abs{u_n} \geq n$~; toute
\omterm{def:b1:seq:subsequence}{suite extraite} est non bornée, donc divergente
(\cref{prop:b1:seq:first})~: aucune \omterm{def:b1:seq:subsequence}{suite extraite} ne converge. Si $F$
n'est pas \omterm{def:b1:topology:closed}{fermé}, prenons $u_n \in F$ avec $u_n \to \ell \notin F$
(\cref{thm:b1:topology:seqclosed})~: toute \omterm{def:b1:seq:subsequence}{suite extraite} converge
vers $\ell \notin F$, donc aucune \omterm{def:b1:seq:subsequence}{suite extraite} ne converge
\emph{dans} $F$.
\end{proof}

\begin{example}[Compacts emboîtés]\label{ex:b1:topology:nested}
Un premier exercice d'application du théorème. Soient $K_0 \supseteq
K_1 \supseteq K_2 \supseteq \dots$ des parties compactes (\omterm{def:b1:topology:closed}{fermées}
bornées) non vides de $\R$. Alors $\bigcap_n K_n \neq \emptyset$. En
effet, choisissons $x_n \in K_n$ pour chaque $n$~: la suite vit
dans le compact $K_0$, donc une \omterm{def:b1:seq:subsequence}{suite extraite} $x_{\varphi(n)}$
converge vers un certain $x$ (\cref{thm:b1:topology:compact}). Pour
tout $m$ fixé, les termes $x_{\varphi(n)}$ avec $\varphi(n) \geq m$
appartiennent tous au \omterm{def:b1:topology:closed}{fermé} $K_m$, donc la limite $x$ appartient à
$K_m$ (\cref{thm:b1:topology:seqclosed})~; comme $m$ était
quelconque, $x \in \bigcap_n K_n$. Un compagnon utile~: si un \omterm{def:b1:topology:open}{ouvert}
$U$ contient $\bigcap_n K_n$, alors $U \supseteq K_n$ pour un certain
$n$ --- appliquer le même argument à des points $x_n \in K_n
\setminus U$~; la limite $x$ appartiendrait à $\bigcap K_n \subseteq
U$, or $U$ \omterm{def:b1:topology:open}{ouvert} force $x_{\varphi(n)} \in U$ à partir d'un certain
rang, contradiction. Les deux énoncés tombent en défaut sans
compacité~: $\bigcap_n \intoo{0}{\frac 1n} = \emptyset$ et
$\bigcap_n \intco{n}{+\infty} = \emptyset$.
L'idée à retenir~: la compacité convertit une chaîne infinie
d'\omterm{def:b1:logic:statement}{assertions} de non-vacuité en un unique point limite --- c'est
l'outil qui survit au passage à l'intersection infinie, et le devoir
maison (\cref{pb:b1:topology:1}) s'y appuiera deux fois.
\end{example}

(\cref{ch:b1:continuity}) et du théorème de Heine sur la continuité

\begin{example}[Frontières et réunions]\label{ex:b1:topology:boundaryunion}
On a toujours $\partial(A \cup B) \subseteq \partial A \cup \partial
B$~: un point de $\partial(A \cup B)$ a tous ses \omterm{def:b1:topology:open}{voisinages} qui
rencontrent $A \cup B$ (donc $A$ ou $B$, l'un des deux une
infinité de fois) et qui rencontrent le complémentaire de $A \cup
B$, lequel est contenu dans les deux complémentaires --- une brève
vérification place alors le point dans $\partial A$ ou dans
$\partial B$. L'inclusion peut être spectaculairement stricte~:
avec $A = \Q$ et $B = \R\setminus\Q$,
\[
\partial(A \cup B) = \partial \R = \emptyset ,
\qquad
\partial A \cup \partial B = \R \cup \R = \R :
\]
deux \omterm{def:b1:logic:sets}{ensembles} déchiquetés peuvent se recoller en un \omterm{def:b1:logic:sets}{ensemble} sans
couture, leurs \omterm{def:b1:topology:closure}{frontières} s'annihilant l'une l'autre. L'idée à
retenir~: l'\omterm{def:b1:topology:closure}{intérieur} et l'\omterm{def:b1:topology:closure}{adhérence} se comportent
\emph{monotonement} vis-à-vis des réunions et des intersections,
mais pas les \omterm{def:b1:topology:closure}{frontières} --- traitez $\partial$ comme une quantité
dérivée ($\overline A \setminus \mathring A$), jamais comme un
opérateur doté d'une algèbre propre.
\end{example}

\begin{remark}[Perspectives à l'intérieur de ce volume]
Le vocabulaire construit ici est réemployé deux fois encore dans
ce livre. Dans le \cref{ch:b1:continuity}, chaque théorème est un
énoncé de topologie déguisé~: le théorème des valeurs
intermédiaires dit que les \omterm{def:b1:logic:map}{applications} continues préservent la
propriété d'\omterm{prop:b1:reals:intervals}{intervalle}, le théorème des bornes atteintes qu'elles
préservent la compacité --- et les démonstrations appellent
les \cref{thm:b1:topology:seqclosed,thm:b1:topology:compact} par leur
nom. Dans le \cref{ch:b1:multivar}, les mêmes définitions sont
relues dans $\R^2$ avec des disques à la place des \omterm{prop:b1:reals:intervals}{intervalles}~:
\omterm{def:b1:topology:open}{ouverts}, \omterm{def:b1:topology:closure}{adhérences} et compacité se transposent mot pour mot, et
le théorème des bornes atteintes à deux variables repose de
nouveau sur Bolzano--Weierstrass (extraire sur chaque
coordonnée). La seule notion qui ne se généralise \emph{pas} sans
douleur est l'\omterm{prop:b1:reals:intervals}{intervalle} lui-même --- dans le plan, la connexité
remplace la convexité, une histoire qui commence avec le
«~seuls $\emptyset$ et $\R$ sont à la fois \omterm{def:b1:topology:open}{ouverts} et \omterm{def:b1:topology:closed}{fermés}~» de
l'\cref{exo:b1:topology:9}.
\end{remark}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:b1:topology:1}
Pour chacune des parties suivantes, dire si elle est \omterm{def:b1:topology:open}{ouverte}, \omterm{def:b1:topology:closed}{fermée},
les deux, ou ni l'une ni l'autre (avec justification)~:
$\intoo{0}{1} \cup \intoo{2}{3}$~; $\;\intco{0}{1}$~;
$\;\{0\} \cup \intcc{1}{2}$~; $\;\R \setminus \Z$~; $\;\Q \cap
\intoo{0}{1}$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:topology:2}
Déterminer $\mathring A$, $\overline A$ et $\partial A$ pour~:
$A = \intoc{0}{1} \cup \{2\}$~; $\;A = \R \setminus \Q$~;
$\;A = \bigl\{\frac{(-1)^n n}{n+1} : n \in \N\bigr\}$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:topology:3}
Démontrer qu'un \omterm{def:b1:counting:card}{ensemble fini} est \omterm{def:b1:topology:closed}{fermé}, d'abord par le
complémentaire, puis par la caractérisation séquentielle.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:topology:4}
Démontrer que, pour toutes parties $A, B \subseteq \R$~:
$\overline{A \cup B} = \overline A \cup \overline B$. Montrer sur un
exemple que $\overline{A \cap B}$ peut différer de $\overline A \cap
\overline B$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:topology:5}
Soit $u_n \to \ell$ dans $\R$. Démontrer que l'\omterm{def:b1:logic:sets}{ensemble} $\{u_n : n \in
\N\} \cup \{\ell\}$ est \omterm{def:b1:topology:closed}{fermé} (donc compact si l'on ajoute qu'il est
borné --- ce qu'il est).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:topology:6}
Soient $U$ un \omterm{def:b1:topology:open}{ouvert} et $A$ une partie quelconque. Démontrer que
$U + A = \{u + a\}$ est \omterm{def:b1:topology:open}{ouvert}. En déduire que la somme d'un \omterm{def:b1:topology:open}{ouvert} et
d'une partie quelconque est \omterm{def:b1:topology:open}{ouverte}, et contraster~: exhiber deux
\emph{\omterm{def:b1:topology:closed}{fermés}} dont la somme n'est pas \omterm{def:b1:topology:closed}{fermée}.
\emph{(Essayer $\Z$ et $\sqrt 2\,\Z$, avec l'\cref{exo:b1:reals:9}.)}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:topology:7}
Un point $x \in A$ est \emph{isolé} dans $A$ lorsqu'un \omterm{def:b1:topology:open}{voisinage} de
$x$ rencontre $A$ en $x$ seulement. Démontrer que tout point de $\Z$
est isolé dans $\Z$, que $A = \{\frac1n\}$ a tous ses points isolés
alors que $\overline A \neq A$, et qu'un \omterm{def:b1:logic:sets}{ensemble} dont tous les points
sont isolés est d'\omterm{def:b1:topology:closure}{intérieur} vide.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:topology:8}
Démontrer que l'\omterm{def:b1:topology:closure}{adhérence} d'une partie bornée est bornée, et que
$\sup \overline A = \sup A$ pour $A$ non vide et majorée. En déduire
que $\sup A \in \overline A$~: la \omterm{def:b1:reals:bounds}{borne supérieure} est toujours adhérente.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:topology:9}
Démontrer que les seules parties de $\R$ à la fois \omterm{def:b1:topology:open}{ouvertes} et \omterm{def:b1:topology:closed}{fermées}
sont $\emptyset$ et $\R$. \emph{Indication~: supposer $A$ \omterm{def:b1:topology:open}{ouvert},
\omterm{def:b1:topology:closed}{fermé}, avec $A \neq \emptyset$ et $\R \setminus A \neq \emptyset$~;
choisir $a \in A$, $b \notin A$, disons $a < b$, et considérer $s =
\sup\,(A \cap \intcc{a}{b})$~; décider si $s$ peut appartenir à $A$ ou
à son complémentaire.}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:topology:10}
(Structure des \omterm{def:b1:topology:open}{ouverts}) Soit $U \subseteq \R$ un \omterm{def:b1:topology:open}{ouvert} non vide. Pour
$x \in U$, soit $I_x$ la réunion de tous les \omterm{prop:b1:reals:intervals}{intervalles} \omterm{def:b1:topology:open}{ouverts}
contenant $x$ et contenus dans $U$. Démontrer que $I_x$ est un
\omterm{prop:b1:reals:intervals}{intervalle} \omterm{def:b1:topology:open}{ouvert}, que deux \omterm{def:b1:logic:sets}{ensembles} $I_x$, $I_y$ sont égaux ou
disjoints, et que $U$ est une réunion \emph{dénombrable} d'\omterm{prop:b1:reals:intervals}{intervalles}
\omterm{def:b1:topology:open}{ouverts} deux à deux disjoints \emph{(choisir un rationnel dans
chacun)}.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:topology:11}
Un point $x \in \R$ est un \emph{point d'accumulation} de $A$ lorsque
tout \omterm{def:b1:topology:open}{voisinage} de $x$ rencontre $A \setminus \{x\}$~; leur \omterm{def:b1:logic:sets}{ensemble}
est l'\emph{\omterm{def:b1:logic:sets}{ensemble} dérivé} $A'$. Démontrer que $\overline A = A \cup A'$,
et que $A$ est \omterm{def:b1:topology:closed}{fermé} si et seulement si $A' \subseteq A$. Déterminer
$A'$ pour $A = \{\frac 1n : n \in \N^*\}$, pour $A = \Z$, et pour
$A = \Q$.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:topology:12}
Pour $A \subseteq \R$ non vide, on pose $d_A(x) = \inf\{\abs{x - a}
: a \in A\}$. Démontrer~:
\begin{enumerate}
  \item $\abs{d_A(x) - d_A(y)} \leq \abs{x - y}$ pour tous $x, y$
        ($d_A$ est $1$-lipschitzienne)~;
  \item $d_A(x) = 0$ si et seulement si $x \in \overline A$~; en
        particulier, si $F$ est \omterm{def:b1:topology:closed}{fermé} et $x \notin F$, alors
        $d_F(x) > 0$~;
  \item pour tout $\varepsilon > 0$, l'\omterm{def:b1:logic:sets}{ensemble} $V_\varepsilon =
        \{x : d_F(x) < \varepsilon\}$ est \omterm{def:b1:topology:open}{ouvert}, contient $F$, et
        $\bigcap_{\varepsilon > 0} V_\varepsilon = F$ lorsque $F$ est
        \omterm{def:b1:topology:closed}{fermé}~: tout \omterm{def:b1:topology:closed}{fermé} est une intersection dénombrable
        d'\omterm{def:b1:topology:open}{ouverts}.
\end{enumerate}
\end{exercise}

\section{Problème~: l'ensemble de Cantor, minuscule et énorme à la fois}

\begin{problem}[{Devoir maison --- l'\omterm{def:b1:logic:sets}{ensemble} triadique de Cantor~:
longueur nulle, non dénombrable, parfait, et $C + C =
\intcc{0}{2}$}]
\label{pb:b1:topology:1}
Retirer de $\intcc{0}{1}$ son tiers médian \omterm{def:b1:topology:open}{ouvert}, puis le tiers
médian de chaque segment restant, et recommencer indéfiniment~: ce
qui survit est l'\emph{\omterm{pb:b1:topology:1}{ensemble de Cantor}} $C$, la fabrique
fondamentale de contre-exemples de l'analyse. Ce problème le
construit, le lit à travers la machinerie en base $3$ du
\cref{pb:b1:reals:1}, et établit son portrait paradoxal~: longueur
totale nulle, et pourtant non dénombrable~; \omterm{def:b1:topology:closure}{intérieur} vide, et
pourtant aucun point isolé~; totalement discontinu, et pourtant
$C + C$ remplit le segment $\intcc{0}{2}$ tout
entier.\index{ensemble de Cantor} Formellement~: $C_0 = \intcc{0}{1}$,
et $C_{n+1}$ s'obtient à partir de $C_n$ en supprimant le tiers
médian \omterm{def:b1:topology:open}{ouvert} de chaque segment de $C_n$~; finalement $C =
\bigcap_{n \geq 0} C_n$. Dans tout le problème, un \emph{code
ternaire} de $x \in \intcc{0}{1}$ est une suite de chiffres
$(d_k)_{k\geq1}$ avec $d_k \in \{0, 1, 2\}$ dont la valeur
$\sup_n \sum_{k=1}^n d_k 3^{-k}$ vaut $x$ --- les codes impropres
(égaux à $2$ à partir d'un rang) sont autorisés~; d'après
le \cref{pb:b1:reals:1} (questions 9 à 11), tout $x \in
\intcc{0}{1}$ possède un ou deux codes, deux exactement lorsque
$x = m/3^N \in \intoo{0}{1}$.

\medskip
\noindent\textbf{Partie I --- La construction.}
\begin{enumerate}
  \item Décrire explicitement $C_1$ et $C_2$ comme réunions de
        segments, et démontrer par récurrence~: $C_n$ est une
        réunion disjointe de $2^n$ segments \omterm{def:b1:topology:closed}{fermés}, chacun de
        longueur $3^{-n}$.
  \item Montrer que $C$ est \omterm{def:b1:topology:closed}{fermé}, borné --- donc compact
        (\cref{thm:b1:topology:compact}) --- non vide, et que
        toute extrémité de tout segment de tout $C_n$ appartient
        à $C$.
  \item La longueur totale de $C_n$ vaut $\bigl(\frac23\bigr)^n$.
        En déduire que, pour tout $\varepsilon > 0$, l'\omterm{def:b1:logic:sets}{ensemble}
        $C$ peut être recouvert par un nombre fini de segments de
        longueur totale $\leq \varepsilon$~: l'\omterm{pb:b1:topology:1}{ensemble de Cantor}
        est \emph{de longueur nulle}.
  \item Montrer qu'un \omterm{prop:b1:reals:intervals}{intervalle} contenu dans $C$ a une longueur
        $\leq 3^{-n}$ pour tout $n$, donc est vide ou réduit à un
        point~: $\mathring C = \emptyset$. Étant \omterm{def:b1:topology:closed}{fermé} et
        d'\omterm{def:b1:topology:closure}{intérieur} vide, $C$ est \emph{nulle part \omterm{def:b1:topology:dense}{dense}}.
\end{enumerate}

\medskip
\noindent\textbf{Partie II --- Le code ternaire.}
\begin{enumerate}[resume]
  \item Démontrer la relation d'autosimilarité
        \[
        C_{n+1} = \tfrac13 C_n \,\cup\,
        \bigl(\tfrac23 + \tfrac13 C_n\bigr),
        \qquad\text{d'où}\qquad
        C = \tfrac13 C \,\cup\, \bigl(\tfrac23 + \tfrac13
        C\bigr),
        \]
        les deux morceaux étant disjoints~: $C$ est formé de deux
        copies de lui-même à l'échelle $\frac13$.
  \item Démontrer par récurrence sur $n$~: $x \in C_n$ si et
        seulement si $x$ possède un code ternaire dont les $n$
        premiers chiffres sont dans $\{0, 2\}$. En déduire, en
        utilisant le fait que $x$ possède au plus deux codes~:
        $x \in C$ si et seulement si $x$ possède un code
        \emph{sans aucun chiffre égal à $1$} (un code \emph{sans
        $1$}).
  \item Les codes en action~: donner des codes sans $1$ pour $0$,
        $1$, $\frac13$, $\frac23$~; montrer que
        $\frac14 = (0.\overline{02})_3$ et
        $\frac34 = (0.\overline{20})_3$, de sorte que tous deux
        appartiennent à $C$~; et vérifier que $\frac14$ n'est
        \emph{pas} une extrémité d'un $C_n$ (les extrémités sont
        de la forme $m/3^n$).
  \item Montrer que chaque $x \in C$ possède \emph{exactement un}
        code sans $1$ \emph{(lorsque $x$ possède deux codes,
        démontrer qu'exactement l'un des deux contient le chiffre
        $1$)}. Conclure~: l'\omterm{def:b1:logic:map}{application} «~valeur~» est une
        \omterm{def:b1:logic:inj}{bijection} de l'\omterm{def:b1:logic:sets}{ensemble} des suites $\{0,2\}$ sur $C$.
  \item (Diagonale) Soit $k \mapsto x_k$ une \omterm{def:b1:logic:map}{application}
        quelconque $\N^* \to C$. Construire une suite $\{0, 2\}$
        différant à l'indice $k$ du code de $x_k$, et en conclure
        que $C$ n'est pas dénombrable --- alors que, par
        contraste, la question 3 le dit métriquement négligeable.
\end{enumerate}

\medskip
\noindent\textbf{Partie III --- Portrait topologique.}
\begin{enumerate}[resume]
  \item Rassembler le bilan obtenu~: $C$ est compact, non
        dénombrable, de longueur nulle, nulle part \omterm{def:b1:topology:dense}{dense}. Pour
        chaque propriété, dire si la seule inclusion $C \subseteq
        C_n$ suffit à la porter.
  \item ($C$ est parfait) Soit $x \in C$ de code sans $1$ égal à
        $(d_k)$. Échanger le chiffre $d_n$ ($0 \leftrightarrow
        2$) produit $x_n \in C$ avec $\abs{x_n - x} = 2 \cdot
        3^{-n}$. En conclure que $C$ n'a aucun point isolé~: tout
        point de $C$ est limite d'\emph{autres} points de $C$.
  \item Montrer que les extrémités de la question 2 forment une
        partie dénombrable \omterm{def:b1:topology:dense}{dense} de $C$ \emph{(tronquer le code
        après $n$ chiffres et poursuivre par des $0$~;
        dénombrabilité comme dans l'\cref{exo:b1:topology:10})}.
        Conclure~: le point typique de $C$ --- comme $\frac14$ ---
        n'est \emph{pas} une extrémité~: les extrémités forment un
        squelette dénombrable à l'intérieur d'un corps non
        dénombrable.
  \item (Totalement discontinu) Soient $x < y$ dans $C$. Choisir
        $n$ tel que $3^{-n} < y - x$ et produire un point $z \in
        \intoo{x}{y}$ avec $z \notin C$. En conclure que les
        seuls \omterm{prop:b1:reals:intervals}{intervalles} non vides contenus dans $C$ sont les
        singletons.
\end{enumerate}

\medskip
\noindent\textbf{Partie IV --- Arithmétique de $C$.}
\begin{enumerate}[resume]
  \item Montrer que $1 - C = C$ \emph{(que fait $x \mapsto 1 - x$
        à un code sans $1$~? rappelons que $1 = (0.\overline{2})_3$)}.
  \item (Addition des codes) Montrer que si $x$, $x'$ ont pour
        codes $(a_k)$, $(b_k)$, alors $x + x' = \lim_n\,(t_n + t'_n)$
        où $t_n, t'_n$ sont les sommes partielles. En déduire~:
        tout $y \in \intcc{0}{1}$ est le \emph{milieu} de deux
        points de $C$ --- étant donné un code $(e_k)$ de $y$,
        choisir des chiffres $a_k, b_k \in \{0, 2\}$ tels que
        $\frac{a_k + b_k}{2} = e_k$.
  \item En conclure que $C + C = \intcc{0}{2}$ et, avec la
        question 14, que $C - C = \intcc{-1}{1}$. Cas concret~:
        écrire $1$ comme somme des deux points non extrémités
        trouvés à la question 7.
  \item Réfléchir~: un \omterm{def:b1:logic:sets}{ensemble} de longueur nulle dont l'\omterm{def:b1:logic:sets}{ensemble}
        des différences remplit $\intcc{-1}{1}$. Pourquoi n'y
        a-t-il pas de contradiction entre «~$C$ est métriquement
        négligeable~» et «~$C + C$ est de longueur pleine~»~? (Une
        phrase~; réfléchir à ce que la longueur contrôle et à ce
        qu'elle ne contrôle pas.)
\end{enumerate}

\medskip
\noindent\textbf{Partie V --- Des éléments, rationnels et irrationnels.}
\begin{enumerate}[resume]
  \item Combiner la question 8 avec le critère de périodicité du
        \cref{pb:b1:reals:1}~: un point de $C$ est rationnel si et
        seulement si son code sans $1$ est périodique à partir
        d'un certain rang. Effectuer la division en base $3$ pour
        vérifier que $\frac1{13} = (0.\overline{002})_3 \in C$.
  \item Produire un élément explicitement \emph{irrationnel} de
        $C$~: la valeur du code valant $d_k = 2$ aux positions
        triangulaires $k = \frac{j(j+1)}{2}$ et $d_k = 0$
        ailleurs. Justifier l'irrationalité par l'argument des
        écarts croissants du \cref{pb:b1:reals:1} (question 20).
  \item (Sur un segment tout entier) Considérer $h$ qui envoie le
        point de $C$ de code sans $1$ égal à $(d_k)$ sur la
        valeur de la suite \emph{binaire} $\bigl(\frac{d_k}2\bigr)$,
        c'est-à-dire $h(x) = \sup_n \sum_{k=1}^n \frac{d_k}{2}\,2^{-k}$.
        Montrer que $h$ envoie $C$ \emph{sur} $\intcc{0}{1}$ tout
        entier. Ainsi le négligeable $C$ se surjecte sur un
        segment de longueur pleine --- une seconde démonstration
        du fait que $C$ n'est pas dénombrable.
  \item (Longueur et autosimilarité) Supposons qu'une notion de
        longueur $L$ soit définie pour $C$ et ses copies
        réduites, respectant l'homothétie ($L(\lambda A) = \lambda L(A)$),
        l'invariance par translation et l'additivité sur la
        décomposition disjointe de la question 5. Montrer
        qu'alors $L(C) = \frac23\,L(C)$, ce qui force $L(C) = 0$~:
        l'autosimilarité seule condamne déjà $C$ à la longueur
        nulle.
\end{enumerate}

\medskip
\noindent\textbf{Partie VI --- Un cousin gras, et la morale.}
\begin{enumerate}[resume]
  \item (\omterm{pb:b1:topology:1}{Ensemble de Cantor} gras) Reprendre la construction, mais
        en retirant à l'étape $n$ ($n = 0, 1, 2, \dots$) de
        chacun des $2^n$ segments courants un \omterm{prop:b1:reals:intervals}{intervalle} \omterm{def:b1:topology:open}{ouvert}
        central de longueur $4^{-(n+1)}$ seulement. Montrer que
        les longueurs $l_n$ des segments vérifient
        $l_{n+1} = \frac{l_n - 4^{-(n+1)}}{2}$, $l_n =
        \frac{2^n + 1}{2\cdot 4^n} > 0$, que l'\omterm{def:b1:logic:sets}{ensemble}
        $K = \bigcap K_n$ ainsi obtenu est compact d'\omterm{def:b1:topology:closure}{intérieur}
        vide, et que la longueur totale retirée vaut
        $\sum_{n\geq0} 2^n 4^{-(n+1)} = \frac12$. En admettant
        l'additivité (intuitive, Licence 3) de la longueur pour
        les réunions finies d'\omterm{prop:b1:reals:intervals}{intervalles}, et en utilisant les
        deux énoncés de l'\cref{ex:b1:topology:nested}, montrer que
        toute famille finie d'\omterm{prop:b1:reals:intervals}{intervalles} \omterm{def:b1:topology:open}{ouverts} recouvrant $K$
        a une longueur totale $\geq \frac12$~: $K$ est nulle part
        \omterm{def:b1:topology:dense}{dense} mais \emph{non} négligeable. La petitesse a
        plusieurs sens non équivalents.
  \item (Distances) Montrer que, pour un \omterm{def:b1:topology:closed}{fermé} non vide $F \subseteq
        \R$ et $x \in \R$, la \omterm{def:b1:reals:bounds}{borne inférieure} $d(x, F)$ est
        \emph{atteinte} \emph{(suite minimisante et
        Bolzano--Weierstrass)}. Calculer ensuite
        \[
        \max_{y \in \intcc{0}{1}} d(y, C) = \frac16 ,
        \]
        maximum atteint exactement au centre $y = \frac12$ \emph{(un
        point d'un trou créé à l'étape $n$ est à distance au plus
        $\frac{3^{-n}}{2}$ des extrémités du trou, lesquelles
        appartiennent à $C$)}.
  \item (Tout point est valeur d'\omterm{def:b1:topology:closure}{adhérence}) En utilisant les
        questions 12 et 2, produire une seule suite de $C$ dont
        l'\omterm{def:b1:logic:sets}{ensemble} des valeurs d'\omterm{def:b1:topology:closure}{adhérence} est $C$ \emph{tout
        entier}. (À comparer~: pour une suite convergente cet
        \omterm{def:b1:logic:sets}{ensemble} est réduit à un point --- $C$ réalise l'extrême
        opposé parmi les compacts.)
  \item Synthèse, une phrase pour chaque point~: (i) quels
        théorèmes de ce chapitre la construction a-t-elle
        réellement consommés (stabilité des \omterm{def:b1:topology:closed}{fermés}, compacité,
        caractérisations séquentielles)~? (ii) énumérer les quatre
        couples paradoxaux du portrait (longueur nulle/non
        dénombrable, \omterm{def:b1:topology:closed}{fermé}/\omterm{def:b1:topology:closure}{intérieur} vide, parfait/totalement
        discontinu, négligeable/$C+C$ plein)~; (iii) où $C$
        refait-il surface plus tard (l'escalier du diable
        construit sur $h$ dans la théorie de la continuité, et la
        théorie de la mesure du volume de Licence 3, où $C$ sépare
        «~dénombrable~» de «~négligeable~»)~?
\end{enumerate}
\end{problem}
