\chapter{Nombres complexes}\label{ch:b1:complex}

Les nombres complexes ont été rencontrés dans le volume de Terminale
(lycée) comme un outil de calcul pour les équations du second degré.
Ce chapitre en fait un objet central~: la forme exponentielle et ses
conséquences (de Moivre, racines $n$-ièmes, \omterm{def:b1:complex:unity}{racines de l'unité}), la
traduction systématique entre $\C$ et la géométrie plane, et l'usage
de $\eu^{\iu\theta}$ comme machine à démontrer les identités
trigonométriques.

\section{Le corps $\C$, module et conjugué}

\begin{definition}[Le corps des nombres complexes]\label{def:b1:complex:field}
$\C = \{a + \iu b : a, b \in \R\}$, muni de l'addition usuelle et de
la multiplication déterminée par $\iu^2 = -1$. Tout $z = a + \iu b$
non nul admet un inverse~: $z^{-1} = \frac{a - \iu b}{a^2 + b^2}$ ---
dans le langage du \cref{ch:b1:structures}, $\C$ est un corps. On
note $a = \Re(z)$, $b = \Im(z)$, $\conj{z} = a - \iu b$ (le
\emph{conjugué}\index{conjugué}) et $\abs{z} = \sqrt{a^2 + b^2}$ (le
\emph{module}\index{module}).
\end{definition}

\begin{proposition}[Règles de calcul sur le conjugué et le module]\label{prop:b1:complex:rules}
Pour $z, w \in \C$~:
\begin{enumerate}
  \item $\conj{z + w} = \conj z + \conj w$, $\conj{zw} = \conj z\,
        \conj w$, $\conj{\conj z} = z$~;
  \item $z \conj z = \abs z^2$~; $\;\Re(z) = \frac{z + \conj z}{2}$,
        $\Im(z) = \frac{z - \conj z}{2\iu}$~;
  \item $\abs{zw} = \abs z\, \abs w$, et $\abs{z^{-1}} =
        \abs{z}^{-1}$ pour $z \neq 0$~;
  \item (\omterm{prop:b1:complex:rules}{inégalité triangulaire}\index{inégalité triangulaire})
        $\abs{z + w} \leq \abs z + \abs w$, avec égalité si et
        seulement si $z$ et $w$ sont portés par une même demi-droite
        issue de $0$ ($w = \lambda z$ ou $z = \lambda w$ avec
        $\lambda \geq 0$)~;
  \item (\omterm{prop:b1:complex:rules}{inégalité triangulaire} renversée) $\bigl|\abs z - \abs w\bigr|
        \leq \abs{z - w}$.
\end{enumerate}
\end{proposition}

\begin{proof}
(1) et (2) sont des calculs directs sur les parties réelle et
imaginaire. (3)~: $\abs{zw}^2 = zw\,\conj{zw} = z\conj z\, w \conj w =
\abs z^2 \abs w^2$, puis on prend les racines carrées~; on applique
cela à $z \cdot z^{-1} = 1$ pour l'inverse.

(4) Les deux membres sont positifs, comparons donc les carrés~:
\[
\abs{z+w}^2 = (z+w)(\conj z + \conj w)
= \abs z^2 + \abs w^2 + 2\,\Re(z \conj w),
\]
et $\Re(z\conj w) \leq \abs{z \conj w} = \abs z \abs w$ donne
$\abs{z+w}^2 \leq (\abs z + \abs w)^2$. Le cas d'égalité impose
$\Re(z \conj w) = \abs{z \conj w}$, c'est-à-dire $z \conj w \in \R_+$~;
si $w \neq 0$, il vient $z = \frac{z\conj w}{\abs w^2}\, w = \lambda w$
avec $\lambda \geq 0$ (et le cas $w = 0$ est trivial).

(5) $\abs z = \abs{(z - w) + w} \leq \abs{z-w} + \abs w$ donne $\abs z
- \abs w \leq \abs{z - w}$~; on échange $z$ et $w$ pour l'autre signe.
\end{proof}

\begin{example}[Un entraînement complet aux modules et arguments]\label{ex:b1:complex:workout}
Mettons $w = \dfrac{3 + 4\iu}{1 - 2\iu}$ sous forme algébrique et
calculons son \omterm{def:b1:complex:field}{module} de deux façons. En multipliant par le
\omterm{def:b1:complex:field}{conjugué} du dénominateur~:
\[
w = \frac{(3 + 4\iu)(1 + 2\iu)}{(1 - 2\iu)(1 + 2\iu)}
= \frac{3 + 6\iu + 4\iu - 8}{1 + 4}
= \frac{-5 + 10\iu}{5} = -1 + 2\iu .
\]
Directement~: $\abs w = \sqrt{1 + 4} = \sqrt5$. Par la règle du
quotient (\cref{prop:b1:complex:rules}~(3))~: $\abs w = \frac{\abs{3 +
4\iu}}{\abs{1 - 2\iu}} = \frac5{\sqrt5} = \sqrt5$ --- même réponse,
sans passer par la forme algébrique. La leçon se généralise~:
\omterm{def:b1:complex:field}{modules} et \omterm{def:b1:complex:expi}{arguments} passent bien aux produits et aux quotients,
parties réelle et imaginaire passent bien aux sommes. On choisit la
représentation adaptée aux opérations en jeu, et l'on ne convertit
que lorsqu'on y est contraint.
\end{example}

\begin{example}[Équations faisant intervenir le conjugué]\label{ex:b1:complex:conjeq}
Résolvons dans $\C$~: $\;z + 2\conj z = 6 + 2\iu$. Une équation
mêlant $z$ et $\conj z$ n'est \emph{pas} polynomiale en $z$~; le
réflexe fiable est de passer aux coordonnées réelles. Avec $z = x
+ \iu y$~:
\[
z + 2\conj z = 3x - \iu y ,
\]
de sorte que l'équation s'écrit $3x = 6$ et $-y = 2$~: l'unique
solution est $z = 2 - 2\iu$. (Vérification~: $(2 - 2\iu) + 2(2 +
2\iu) = 6 + 2\iu$.) On peut aussi conjuguer l'équation tout
entière, ce qui donne $\conj z + 2z = 6 - 2\iu$, et résoudre le
système linéaire d'inconnues $z, \conj z$ --- même réponse, et une
astuce utile lorsque les coefficients sont complexes. Les
équations en $z$ et $\conj z$ sont en réalité des systèmes de deux
équations réelles~; l'attente «~degré $1$, une solution~» est ici
sans danger, mais $z\conj z = -1$ (sans solution) montre
l'intuition polynomiale tomber en défaut dès que des produits
apparaissent.
\end{example}

\section{Forme exponentielle}

\begin{definition}[Exponentielle complexe d'un argument imaginaire pur]\label{def:b1:complex:expi}
Pour $\theta \in \R$, on pose
\[
\eu^{\iu\theta} = \cos\theta + \iu \sin\theta .
\]
Tout $z \neq 0$ peut s'écrire $z = r\,\eu^{\iu\theta}$ avec $r =
\abs z > 0$~; $\theta$ est un \emph{argument}\index{argument} de $z$,
déterminé à l'addition près d'un multiple de $2\pi$. La valeur
appartenant à $\intoc{-\pi}{\pi}$ est l'\emph{argument principal},
noté $\arg z$.
\end{definition}

\begin{example}[Premières valeurs, et une identité célèbre]\label{ex:b1:complex:eulerid}
En lisant la définition aux angles remarquables~:
\[
\eu^{\iu\pi/2} = \iu, \qquad
\eu^{\iu\pi} = -1, \qquad
\eu^{2\iu\pi} = 1, \qquad
\eu^{\iu\pi/4} = \frac{\sqrt2}2\,(1 + \iu) .
\]
La deuxième, réécrite $\eu^{\iu\pi} + 1 = 0$, est la célèbre
identité d'Euler reliant $\eu$, $\iu$, $\pi$, $1$ et $0$~; à ce
stade du livre, c'est un déroulement de la définition plutôt qu'un
théorème, et son véritable contenu --- pourquoi la fonction
exponentielle analytique du \cref{ch:b1:functions}, prolongée aux
\omterm{def:b1:complex:expi}{arguments} imaginaires, mérite le même nom --- sera réglé par les
séries entières du \cref{ch:b1:series}. En attendant, la formule
ci-dessus mérite d'être sue par cœur comme table de conversion~:
on l'utilise silencieusement chaque fois qu'un \omterm{def:b1:complex:expi}{argument} se lit sur
une figure.
\end{example}

\begin{theorem}[L'équation fonctionnelle]\label{thm:b1:complex:funceq}
Pour tous $\theta, \varphi \in \R$~:
\[
\eu^{\iu\theta}\, \eu^{\iu\varphi} = \eu^{\iu(\theta + \varphi)},
\qquad
\abs{\eu^{\iu\theta}} = 1,
\qquad
\conj{\eu^{\iu\theta}} = \eu^{-\iu\theta} = (\eu^{\iu\theta})^{-1}.
\]
Par conséquent, $\abs{zw} = \abs z \abs w$ s'accompagne du transport
des \omterm{def:b1:complex:expi}{arguments}~: $\arg(zw) \equiv \arg z + \arg w \pmod{2\pi}$.
\end{theorem}

\begin{proof}
Développons le produit et utilisons les formules d'addition~:
\[
(\cos\theta + \iu\sin\theta)(\cos\varphi + \iu\sin\varphi)
= (\cos\theta\cos\varphi - \sin\theta\sin\varphi)
+ \iu\,(\sin\theta\cos\varphi + \cos\theta\sin\varphi),
\]
c'est-à-dire $\cos(\theta+\varphi) + \iu\sin(\theta+\varphi)$. Le
\omterm{def:b1:complex:field}{module} vaut $\sqrt{\cos^2\theta + \sin^2\theta} = 1$, et la formule du
\omterm{def:b1:complex:field}{conjugué} traduit la parité du cosinus et l'imparité du sinus~; elle
fournit l'inverse de $\eu^{\iu\theta}$ puisque
$\eu^{\iu\theta}\eu^{-\iu\theta} = \eu^0 = 1$.
\end{proof}

\begin{corollary}[Formule de de Moivre]\label{cor:b1:complex:demoivre}
Pour $\theta \in \R$ et $n \in \Z$~:
$\;(\cos\theta + \iu\sin\theta)^n = \cos(n\theta) +
\iu\sin(n\theta)$.\index{formule de de Moivre}
\end{corollary}

\begin{proof}
Pour $n \geq 0$, on procède par récurrence. Le cas $n = 0$ s'écrit
$1 = 1$. En supposant la formule au rang $n$, l'équation
fonctionnelle (\cref{thm:b1:complex:funceq}) donne
\[
\bigl(\eu^{\iu\theta}\bigr)^{n+1}
= \bigl(\eu^{\iu\theta}\bigr)^{n}\,\eu^{\iu\theta}
= \eu^{\iu n\theta}\,\eu^{\iu\theta}
= \eu^{\iu(n+1)\theta} ,
\]
ce qui est la formule au rang $n + 1$. Pour $n < 0$, écrivons $n =
-m$ avec $m > 0$~: comme $\eu^{\iu m\theta}$ a pour inverse
$\eu^{-\iu m\theta}$ (\cref{thm:b1:complex:funceq} à nouveau),
\[
\bigl(\eu^{\iu\theta}\bigr)^{-m}
= \bigl(\eu^{\iu m\theta}\bigr)^{-1}
= \eu^{\iu(-m)\theta} . \qedhere
\]
\end{proof}

\begin{example}[Développement de $\cos 3\theta$ par de Moivre]\label{ex:b1:complex:cos3}
Posons $c = \cos\theta$, $s = \sin\theta$. La \omterm{cor:b1:complex:demoivre}{formule de de Moivre} et
le binôme de Newton donnent
\[
\cos 3\theta + \iu \sin 3\theta = (c + \iu s)^3
= c^3 - 3cs^2 + \iu\,(3c^2 s - s^3),
\]
et, en identifiant les parties réelles puis en substituant $s^2 = 1
- c^2$~:
\[
\cos 3\theta = c^3 - 3c(1 - c^2) = 4\cos^3\theta - 3\cos\theta .
\]
La partie imaginaire fournit gratuitement $\sin 3\theta =
3\sin\theta - 4\sin^3\theta$~: une identité complexe en porte
toujours \emph{deux} réelles. Lue à l'envers, l'identité obtenue
est la clé de l'équation classique de la trisection~: construire
$\cos(\theta/3)$ à partir de $\cos\theta$ revient à résoudre
l'équation du troisième degré $4x^3 - 3x = \cos\theta$, où
l'algèbre du \cref{ch:b1:poly} prend le relais.
\end{example}

\begin{proposition}[Formules d'Euler]\label{prop:b1:complex:euler}
\[
\cos\theta = \frac{\eu^{\iu\theta} + \eu^{-\iu\theta}}{2},
\qquad
\sin\theta = \frac{\eu^{\iu\theta} - \eu^{-\iu\theta}}{2\iu}.
\]
\end{proposition}

\begin{proof}
On ajoute, respectivement on soustrait, $\eu^{\iu\theta} = \cos\theta +
\iu\sin\theta$ et $\eu^{-\iu\theta} = \cos\theta - \iu\sin\theta$.
\end{proof}

\begin{method}[La trigonométrie par les exponentielles]\label{met:b1:complex:trig}
\begin{enumerate}
  \item \emph{Linéariser} $\cos^p\theta \sin^q\theta$ (transformer les
        puissances en une somme de $\cos k\theta$, $\sin k\theta$)~:
        on substitue les formules d'Euler, on développe par le binôme
        de Newton (\cref{thm:b1:counting:binomial}), on regroupe les
        termes \omterm{def:b1:complex:field}{conjugués}.
  \item \emph{Développer} $\cos n\theta$ en polynôme en $\cos\theta$~:
        on écrit $\cos n\theta = \Re\bigl((\cos\theta +
        \iu\sin\theta)^n\bigr)$, on développe, et l'on convertit les
        puissances paires de $\sin$ grâce à $\sin^2 = 1 - \cos^2$.
  \item \emph{Sommer des séries trigonométriques} comme $\sum_k \cos
        k\theta$~: on reconnaît la partie réelle d'une somme
        géométrique $\sum_k (\eu^{\iu\theta})^k$.
  \item La \emph{factorisation par l'angle moitié}~: pour tous $p, q$,
        \[
        \eu^{\iu p} + \eu^{\iu q}
        = 2 \cos\tfrac{p - q}{2}\; \eu^{\iu \frac{p+q}{2}},
        \qquad
        \eu^{\iu p} - \eu^{\iu q}
        = 2\iu \sin\tfrac{p - q}{2}\; \eu^{\iu \frac{p+q}{2}} .
        \]
\end{enumerate}
\end{method}

\begin{example}[Linéarisation]\label{ex:b1:complex:linearize}
\[
\cos^3\theta
= \Bigl(\frac{\eu^{\iu\theta} + \eu^{-\iu\theta}}{2}\Bigr)^{\!3}
= \frac{\eu^{3\iu\theta} + 3\eu^{\iu\theta} + 3\eu^{-\iu\theta} +
\eu^{-3\iu\theta}}{8}
= \frac{\cos 3\theta + 3\cos\theta}{4}.
\]
Cette forme s'intègre immédiatement --- c'est précisément la
raison d'être de la linéarisation au \cref{ch:b1:integration}. Un
produit mixte se traite de la même façon, en utilisant simplement
les deux formules d'Euler à la fois~:
\[
\sin^2\theta\cos^2\theta
= \Bigl(\frac{\sin2\theta}2\Bigr)^{\!2}
= \frac{1}{4}\cdot
\Bigl(\frac{\eu^{2\iu\theta} - \eu^{-2\iu\theta}}{2\iu}
\Bigr)^{\!2}
= \frac{2 - \eu^{4\iu\theta} - \eu^{-4\iu\theta}}{16}
= \frac{1 - \cos4\theta}{8} ,
\]
où le raccourci de l'angle double, à la première étape, a économisé
un développement du binôme --- il vaut toujours la peine de le
chercher avant de mécaniser.
\end{example}

\begin{example}[Une somme trigonométrique binomiale]\label{ex:b1:complex:binomsum}
Pour $n \in \N$ et $\theta \in \R$, calculons $S =
\sum_{k=0}^{n}\binom nk \cos k\theta$. On y reconnaît la partie
réelle d'un développement du binôme~:
\[
S = \Re\sum_{k=0}^n \binom nk \bigl(\eu^{\iu\theta}\bigr)^k
= \Re\bigl(1 + \eu^{\iu\theta}\bigr)^n ,
\]
puis on factorise par l'angle moitié
(\cref{met:b1:complex:trig}~(4))~: $1 + \eu^{\iu\theta} =
2\cos\frac\theta2\,\eu^{\iu\theta/2}$, d'où
\[
S = \Re\Bigl(2^n\cos^n\frac\theta2\;\eu^{\iu n\theta/2}\Bigr)
= 2^n \cos^n\frac\theta2\,\cos\frac{n\theta}2 .
\]
La partie imaginaire fournit gratuitement $\sum_k\binom nk\sin
k\theta = 2^n\cos^n\frac\theta2\sin\frac{n\theta}2$.
Vérifications~: $\theta = 0$ redonne $\sum\binom nk = 2^n$, et
$\theta = \pi$ donne $S = 0$ pour $n \geq 1$ (chaque facteur
$\cos\frac\pi2$ s'annule), c'est-à-dire la somme alternée d'une
ligne du triangle de Pascal (\cref{ex:b1:counting:binomial}). La
méthode --- «~voir la somme de cosinus comme l'ombre d'une
puissance complexe, puis factoriser par l'angle moitié~» --- est
exactement celle de l'\cref{exo:b1:complex:6}, le binôme de Newton
remplaçant la série géométrique.
\end{example}

\section{Racines des nombres complexes}

\begin{theorem}[Racines $n$-ièmes]\label{thm:b1:complex:roots}
Soient $a = r\,\eu^{\iu\alpha} \neq 0$ et $n \in \N^*$. L'équation
$z^n = a$ admet exactement $n$ solutions~:
\[
z_k = r^{1/n}\, \eu^{\iu\left(\frac{\alpha}{n} +
\frac{2k\pi}{n}\right)},
\qquad k = 0, 1, \dots, n - 1 .
\]
\end{theorem}

\begin{proof}
Écrivons $z = \rho\,\eu^{\iu\theta}$ ($\rho > 0$). Alors $z^n =
\rho^n \eu^{\iu n\theta} = r \eu^{\iu\alpha}$ si et seulement si
$\rho^n = r$ (\omterm{def:b1:complex:field}{modules}) et $n\theta \equiv \alpha \pmod{2\pi}$
(\omterm{def:b1:complex:expi}{arguments}), c'est-à-dire $\rho = r^{1/n}$ et $\theta =
\frac{\alpha}{n} + \frac{2k\pi}{n}$ pour un certain $k \in \Z$. Reste
à voir quand deux entiers $k, k'$ donnent le même nombre~: cela se
produit exactement lorsque les angles diffèrent d'un multiple de
$2\pi$,
\[
\frac{2k\pi}n - \frac{2k'\pi}n \in 2\pi\Z
\iff \frac{k - k'}n \in \Z
\iff n \mid k - k' .
\]
Par division euclidienne, tout $k \in \Z$ est congru modulo $n$ à
exactement un élément de $\{0, 1, \dots, n-1\}$~: cette plage énumère
donc chaque solution une fois et une seule, et le compte est
exactement $n$. (Le même argument, mené dans $\mathbb U_n$, montre que
les racines forment un $n$-gone régulier~: deux valeurs consécutives
de $k$ correspondent à une rotation de l'angle fixe $\frac{2\pi}n$.)
\end{proof}

\begin{example}[Racines cubiques de $-27$]\label{ex:b1:complex:cuberoots}
Résolvons $z^3 = -27$. Forme exponentielle du second membre~: $-27
= 27\,\eu^{\iu\pi}$, si bien que les trois racines sont
\[
z_k = 3\,\eu^{\iu(\frac\pi3 + \frac{2k\pi}3)}, \quad k = 0, 1, 2 :
\qquad
z_0 = 3\eu^{\iu\pi/3} = \frac32 + \frac{3\sqrt3}2\,\iu,
\quad
z_1 = -3,
\quad
z_2 = \conj{z_0} .
\]
Deux vérifications. D'abord, la racine réelle $-3$ est la racine
évidente, et les deux autres en sont les images par les rotations
d'angles $\pm\frac{2\pi}3$ --- autrement dit $-3j$ et $-3j^2$.
Ensuite, l'algèbre confirme~: $z^3 + 27 = (z + 3)(z^2 - 3z + 9)$,
et le trinôme a pour discriminant $9 - 36 = -27 < 0$, de racines
$\frac{3 \pm 3\iu\sqrt3}2 = z_0, \conj{z_0}$. L'idée à retenir~:
pour un second membre réel, les racines non réelles vont toujours
par paires de \omterm{def:b1:complex:field}{conjuguées}, de sorte que la figure de l'\omterm{def:b1:logic:sets}{ensemble} des
solutions est symétrique par rapport à l'axe réel --- un avant-goût
du théorème de factorisation réelle du \cref{ch:b1:poly}.
\end{example}

\begin{example}[Un second membre non réel]\label{ex:b1:complex:fourthroots}
Résolvons $z^4 = -8 + 8\iu\sqrt3$. Forme exponentielle du second
membre~: \omterm{def:b1:complex:field}{module} $\sqrt{64 + 192} = 16$, \omterm{def:b1:complex:expi}{argument} $\theta$ avec
$\cos\theta = -\frac12$, $\sin\theta = \frac{\sqrt3}2$,
c'est-à-dire $\theta = \frac{2\pi}3$. Les quatre racines sont
\[
z_k = 2\,\eu^{\iu(\frac\pi6 + \frac{k\pi}2)}, \quad k = 0, 1, 2,
3 :
\qquad
z_0 = \sqrt3 + \iu,\quad z_1 = \iu z_0 = -1 + \iu\sqrt3,
\]
\[
z_2 = -z_0 = -\sqrt3 - \iu,\qquad z_3 = -\iu z_0 = 1 - \iu\sqrt3 .
\]
(Vérification~: $z_0^2 = 2 + 2\iu\sqrt3$, donc $z_0^4 = (2 +
2\iu\sqrt3)^2 = 4 - 12 + 8\iu\sqrt3 = -8 + 8\iu\sqrt3$.) Dès
qu'\emph{une} racine est trouvée, les trois autres viennent
gratuitement~: ce sont ses images par les rotations successives
d'angle $\frac\pi2$, autrement dit ses produits par les racines
quatrièmes de l'unité --- c'est la structure générale sous-jacente
au \cref{thm:b1:complex:roots}, qu'il vaut mieux exploiter que de
recalculer chaque racine depuis le début. Pas de symétrie par
conjugaison cette fois~: le second membre n'est pas réel.
\end{example}

\begin{definition}[Racines de l'unité]\label{def:b1:complex:unity}
Les \emph{racines $n$-ièmes de l'unité}\index{racines de l'unité}
sont les solutions de $z^n = 1$~:
\[
\mathbb{U}_n = \bigl\{\, \omega^k : k = 0, \dots, n-1 \,\bigr\},
\qquad \omega = \eu^{2\iu\pi/n}.
\]
Elles forment un groupe pour la multiplication
(\cref{ch:b1:structures}) et occupent les sommets d'un $n$-gone
régulier inscrit dans le cercle unité.
\end{definition}

\begin{proposition}[Somme des racines de l'unité]\label{prop:b1:complex:sumroots}
Pour $n \geq 2$, la somme des racines $n$-ièmes de l'unité est nulle~:
$\sum_{k=0}^{n-1} \omega^k = 0$.
\end{proposition}

\begin{proof}
Somme géométrique de raison $\omega \neq 1$~:
$\sum_{k=0}^{n-1} \omega^k = \frac{\omega^n - 1}{\omega - 1} = 0$
puisque $\omega^n = 1$.
\end{proof}

\begin{example}[Lecture des parties réelle et imaginaire]\label{ex:b1:complex:sumrootsparts}
En séparant $\sum_{k=0}^{n-1}\omega^k = 0$ en parties réelle et
imaginaire, on obtient gratuitement deux identités
trigonométriques~:
\[
\sum_{k=0}^{n-1}\cos\frac{2k\pi}n = 0,
\qquad
\sum_{k=0}^{n-1}\sin\frac{2k\pi}n = 0
\qquad (n \geq 2).
\]
Géométriquement~: l'isobarycentre d'un $n$-gone régulier inscrit
dans le cercle unité est son centre --- les sommets s'équilibrent
exactement. Pour $n = 5$, la première identité donne $1 +
2\cos\frac{2\pi}5 + 2\cos\frac{4\pi}5 = 0$ (en appariant $k$ et $n
- k$), point de départ du calcul de $\cos\frac{2\pi}5$ mené dans
l'\cref{exo:b1:complex:8}.
\end{example}

\begin{example}[Racines carrées sous forme algébrique]\label{ex:b1:complex:sqrt}
Pour résoudre $z^2 = 3 + 4\iu$ sans trigonométrie, posons $z = x +
\iu y$~:
\[
x^2 - y^2 = 3, \qquad 2xy = 4, \qquad x^2 + y^2 = \abs{3 + 4\iu} = 5 .
\]
En ajoutant la première et la dernière~: $x^2 = 4$, donc $x = \pm 2$,
puis $y = 2/x = \pm 1$ avec le \emph{même} appariement de signes ($xy
= 2 > 0$)~: $z = \pm(2 + \iu)$. Combiné à la formule usuelle, cela
résout toute équation du second degré à coefficients complexes
(\cref{exo:b1:complex:7}).
\end{example}

\section{Nombres complexes et géométrie plane}

\begin{proposition}[Dictionnaire géométrique]\label{prop:b1:complex:geometry}
Identifions le point $M(x, y)$ du plan à son \emph{affixe} $z = x +
\iu y$\index{affixe}. Pour des points distincts $A, B, C$ d'affixes
$a, b, c$~:
\begin{enumerate}
  \item $\abs{b - a}$ est la distance $AB$~;
  \item $\arg\dfrac{c - a}{b - a}$ est l'angle entre les vecteurs
        $\vect{AB}$ et $\vect{AC}$ (modulo $2\pi$)~;
  \item $A, B, C$ sont alignés si et seulement si $\dfrac{c - a}{b -
        a} \in \R$~; les droites $AB$ et $AC$ sont perpendiculaires si
        et seulement si $\dfrac{c - a}{b - a} \in \iu\R$.
\end{enumerate}
\end{proposition}

\begin{proof}
(1) n'est que la définition du \omterm{def:b1:complex:field}{module} appliquée à $b - a$, affixe de
$\vect{AB}$. (2)~: écrivons $b - a = r\eu^{\iu\theta}$, $c - a =
s\eu^{\iu\varphi}$~; alors $\frac{c-a}{b-a} = \frac sr
\eu^{\iu(\varphi - \theta)}$ a pour \omterm{def:b1:complex:expi}{argument} $\varphi - \theta$,
l'angle de $\vect{AB}$ à $\vect{AC}$. (3)~: l'alignement signifie un
angle $0$ ou $\pi$, c'est-à-dire un \omterm{def:b1:complex:expi}{argument} dans $\pi\Z$,
c'est-à-dire un quotient réel~; la perpendicularité signifie un angle
$\pm\frac\pi2$, c'est-à-dire un quotient imaginaire pur. (Le quotient
est non nul puisque $C \neq A$.)
\end{proof}

\begin{remark}[Interlude~: $\C$ est le plan muni d'une multiplication]\label{rem:b1:complex:interlude}
Il vaut la peine de s'arrêter sur ce qui rend ce chapitre
possible~: le plan $\R^2$ porte des additions dans toutes les
directions, mais aucune multiplication donnée d'avance --- et $\C$
\emph{est} le plan muni d'une telle multiplication, dans laquelle
multiplier par un nombre fixé fait tourner et dilate. Cette unique
structure sera exploitée trois fois encore dans ce volume. Au
\cref{ch:b1:matrices}, la multiplication par $a + \iu b$
réapparaît sous la forme de la matrice $2 \times 2$ de lignes $(a,
-b)$ et $(b, a)$~: l'arithmétique complexe est une première
famille, entièrement concrète, de produits matriciels. Au
\cref{ch:b1:euclid}, la formule $\Re(\conj z\,w)$ se révèle être le
produit scalaire, et $\abs z$ la norme euclidienne~: l'\omterm{prop:b1:complex:rules}{inégalité
triangulaire} démontrée ici est le modèle de l'histoire de
Cauchy--Schwarz qu'on y racontera. Et au \cref{ch:b1:curves}, un
point mobile s'écrit au mieux $t \mapsto z(t)$, de sorte que
vitesse et accélération deviennent des dérivées à valeurs
complexes --- le mouvement circulaire, par exemple, s'écrit
simplement $z(t) = R\,\eu^{\iu\omega t}$. Une bonne multiplication,
quatre chapitres de dividendes.
\end{remark}

\begin{proposition}[Les applications $z \mapsto az + b$]\label{prop:b1:complex:similitude}
Soient $a \in \C^*$ et $b \in \C$. La transformation $f(z) = az + b$
du plan~:
\begin{itemize}
  \item est une translation si $a = 1$~;
  \item sinon, admet un unique point fixe $\zeta = \frac{b}{1-a}$,
        et $f(z) - \zeta = a\,(z - \zeta)$~: $f$ est la rotation de
        centre $\zeta$ et d'angle $\arg a$, composée avec
        l'homothétie de centre $\zeta$ et de rapport $\abs a$.
\end{itemize}
En particulier, $z \mapsto \eu^{\iu\theta} z$ est la rotation d'angle
$\theta$ autour de l'origine, et $\conj z$ est la symétrie par rapport
à l'axe réel.
\end{proposition}

\begin{proof}
Si $a = 1$, $f(z) = z + b$ est la translation de vecteur d'affixe $b$.
Si $a \neq 1$, l'équation aux points fixes $z = az + b$ a pour unique
solution $\zeta = \frac{b}{1-a}$, et alors $f(z) - \zeta = az + b -
(a\zeta + b) = a(z - \zeta)$. En écrivant $a = \abs a\, \eu^{\iu\arg
a}$, la multiplication par $a$ multiplie les distances à $\zeta$ par
$\abs a$ et ajoute $\arg a$ aux angles en $\zeta$~: c'est la composée
annoncée.
\end{proof}

\begin{example}[Classification d'une application $z \mapsto az+b$]\label{ex:b1:complex:classify}
Prenons $f(z) = \iu z + 1$. Ici $a = \iu \neq 1$~: le point fixe est
\[
\zeta = \frac{b}{1 - a} = \frac1{1 - \iu}
= \frac{1 + \iu}{2},
\]
et comme $\abs a = 1$, il n'y a aucune homothétie~: $f$ est la
rotation pure de centre $\frac{1+\iu}2$ et d'angle $\arg \iu =
\frac\pi2$. Vérification~: $f(\zeta) = \iu\,\frac{1+\iu}2 + 1 =
\frac{\iu - 1}2 + 1 = \frac{1 + \iu}2 = \zeta$, et $f(0) = 1$, $f(1) =
1 + \iu$, $f(1 + \iu) = \iu\,(1 + \iu) + 1 = \iu$~: les quatre points
$0, 1, 1+\iu, \iu$ du carré unité tournent autour de leur centre
$\zeta$, d'un quart de tour à chaque fois --- exactement ce que doit
faire une rotation d'angle $\frac\pi2$ autour du centre du carré.
\end{example}

\begin{remark}[Pièges classiques sur les modules et les arguments]\label{rem:b1:complex:pitfalls}
\begin{enumerate}
  \item \emph{Pas d'inégalités dans $\C$.} Écrire $z \leq w$ pour
        des nombres non réels n'a aucun sens~; seuls les \omterm{def:b1:complex:field}{modules},
        les parties réelles et les parties imaginaires se
        comparent.
  \item \emph{$\sqrt{\phantom z}$ est réservé à $\R_+$.} Tout
        nombre complexe non nul a \emph{deux} racines carrées, et
        aucune n'est privilégiée~: on écrit «~soit $\delta$ une
        racine carrée de $\Delta$~» (calculée comme dans
        l'\cref{ex:b1:complex:sqrt}), jamais $\sqrt\Delta$ --- la
        règle $\sqrt{ab} = \sqrt a\sqrt b$ tombe déjà en défaut
        pour $a = b = -1$.
  \item \emph{Les \omterm{def:b1:complex:expi}{arguments} vivent modulo $2\pi$.} De
        $\eu^{\iu\alpha} = \eu^{\iu\beta}$ on conclut $\alpha
        \equiv \beta \pmod{2\pi}$, et non $\alpha = \beta$~;
        oublier le $2k\pi$ est la façon dont les \omterm{def:b1:logic:sets}{ensembles} de
        solutions de $z^n = a$ perdent $n - 1$ de leurs $n$
        éléments.
  \item \emph{$\abs{z + w}$ n'est pas $\abs z + \abs w$.}
        L'égalité dans l'\omterm{prop:b1:complex:rules}{inégalité triangulaire} est le cas
        exceptionnel des vecteurs alignés
        (\cref{prop:b1:complex:rules}~(4))~; en général, le \omterm{def:b1:complex:field}{module}
        d'une somme se majore, il ne se calcule pas.
\end{enumerate}
\end{remark}

\begin{example}\label{ex:b1:complex:equilateral}
$A, B, C$ (d'affixes $a, b, c$, deux à deux distinctes) forment un
triangle équilatéral dont les sommets sont dans l'ordre direct (sens
trigonométrique) si et seulement si $\frac{c - a}{b - a} =
\eu^{\iu\pi/3}$~: la rotation de centre $A$ et d'angle $\frac\pi3$
envoie $B$ sur $C$. Les deux orientations à la fois sont saisies par
l'équation symétrique $a^2 + b^2 + c^2 = ab + bc + ca$
(\cref{exo:b1:complex:10}).
\end{example}

\begin{remark}[Où ce chapitre est utilisé]\label{rem:b1:complex:whereused}
La forme exponentielle est l'outil de calcul le plus réutilisé de
tout le volume. Les \omterm{def:b1:complex:unity}{racines de l'unité} deviennent l'exemple standard
de groupe cyclique au \cref{ch:b1:structures} et gouvernent la
factorisation de $X^n - 1$ au \cref{ch:b1:poly}~; en y regroupant
les racines \omterm{def:b1:complex:field}{conjuguées}, on obtient les factorisations réelles
utilisées par les éléments simples au \cref{ch:b1:fractions}. La
linéarisation (\cref{met:b1:complex:trig}) est la préparation
standard à l'intégration des puissances trigonométriques au
\cref{ch:b1:integration}, et les équations caractéristiques du
\cref{ch:b1:diffeq} ont des racines complexes dont les parties
réelle et imaginaire produisent les solutions oscillantes
$\eu^{\lambda t}\cos\omega t$. Le dictionnaire géométrique revient
en habit matriciel~: rotations et similitudes deviennent les
matrices orthogonales des \cref{ch:b1:matrices,ch:b1:euclid}, et les
courbes paramétrées du \cref{ch:b1:curves} s'écrivent souvent au
mieux comme des \omterm{def:b1:logic:map}{applications} $t \mapsto z(t) \in \C$.
\end{remark}

\section{Exercices}

\begin{exercise}[$\star$]\label{exo:b1:complex:1}
Mettre sous forme exponentielle~: $1 + \iu$~; $\;\sqrt 3 - \iu$~;
$\;-5$~; $\;\dfrac{1 + \iu}{\sqrt 3 - \iu}$. En déduire
$\cos\frac{5\pi}{12}$ et $\sin\frac{5\pi}{12}$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:complex:2}
Calculer $(1 + \iu)^{20}$. Pour quels $n \in \N$ le nombre $(1 +
\iu)^n$ est-il réel~?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:complex:3}
Décrire géométriquement l'\omterm{def:b1:logic:sets}{ensemble} des $z \in \C$ tels que~:
$\abs{z - 2} = \abs{z + \iu}$~; $\;\abs{z - 1} = 2$~;
$\;\dfrac{z - 1}{z + 1} \in \iu\R$ (pour $z \neq -1$).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:complex:4}
Linéariser $\sin^4\theta$, et développer $\cos 4\theta$ en polynôme en
$\cos\theta$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:complex:5}
Résoudre $z^3 = 8\iu$ et placer les solutions sur une figure. Résoudre
$z^4 = -4$ et factoriser $X^4 + 4$ en produit de deux polynômes réels
du second degré.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:complex:6}
Pour $\theta \in \R$ tel que $\eu^{\iu\theta} \neq 1$ et $n \in \N$,
calculer
\[
C_n = \sum_{k=0}^{n} \cos k\theta
\qquad\text{et}\qquad
S_n = \sum_{k=0}^{n} \sin k\theta
\]
en sommant la série géométrique $\sum_k \eu^{\iu k\theta}$ et en
utilisant la factorisation par l'angle moitié.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:complex:7}
Résoudre dans $\C$~: $z^2 - (3 + 4\iu) z + (-1 + 5\iu) = 0$.
\emph{(Calculer le discriminant, en extraire les racines carrées comme
dans l'\cref{ex:b1:complex:sqrt}.)}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:complex:8}
Soit $\omega = \eu^{2\iu\pi/5}$.
\begin{enumerate}
  \item Justifier que $1 + \omega + \omega^2 + \omega^3 + \omega^4 =
        0$.
  \item On pose $u = \omega + \omega^4$ et $v = \omega^2 + \omega^3$.
        Calculer $u + v$ et $uv$, et en déduire que $u$ et $v$ sont
        les racines de $X^2 + X - 1$.
  \item En conclure que $\cos\frac{2\pi}{5} = \frac{\sqrt 5 - 1}{4}$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:complex:9}
Démontrer que, pour tous $z, w \in \C$ (identité du
parallélogramme)~:
\[
\abs{z + w}^2 + \abs{z - w}^2 = 2\abs z^2 + 2\abs w^2 ,
\]
et l'interpréter géométriquement dans le parallélogramme de sommets
$0, z, w, z + w$.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:complex:10}
Démontrer que trois points deux à deux distincts, d'affixes $a, b, c$,
forment un triangle équilatéral (quelle que soit l'orientation) si et
seulement si
\[
a^2 + b^2 + c^2 = ab + bc + ca .
\]
\emph{Indication~: le cas direct s'écrit $\frac{c-a}{b-a} = -j^2$ et
le cas indirect $\frac{c-a}{b-a} = -j$, où $j = \eu^{2\iu\pi/3}$
vérifie $j^2 + j + 1 = 0$~; ou bien factoriser $a + jb + j^2c$ et $a +
j^2 b + jc$.}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:complex:11}
Pour $n \in \N^*$, calculer $P = \prod_{k=1}^{n-1}
\bigl(1 - \omega^k\bigr)$ où $\omega = \eu^{2\iu\pi/n}$.
\emph{Indication~: $X^n - 1 = \prod_{k=0}^{n-1} (X - \omega^k)$~;
diviser par $X - 1$ et évaluer en $X = 1$.} En déduire
$\prod_{k=1}^{n-1} \sin\frac{k\pi}{n} = \dfrac{n}{2^{n-1}}$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:complex:12}
Soit $n \geq 2$. Résoudre l'équation $(z + 1)^n = (z - 1)^n$ dans
$\C$~: montrer qu'elle admet exactement $n - 1$ solutions, toutes
imaginaires pures, à savoir
\[
z_k = -\iu\,\frac{\cos\frac{k\pi}{n}}{\sin\frac{k\pi}{n}},
\qquad k = 1, \dots, n - 1 .
\]
\emph{Indication~: $z = 1$ n'est pas solution, on peut donc diviser et
utiliser les \omterm{def:b1:complex:unity}{racines de l'unité}~; appliquer ensuite la factorisation
par l'angle moitié de la \cref{met:b1:complex:trig}.}
\end{exercise}

\section{Problème~: le théorème de Napoléon}

\begin{problem}\label{pb:b1:complex:1}
Construisons vers l'extérieur un triangle équilatéral sur chaque
côté d'un triangle \emph{quelconque}~: les trois centres de ces
triangles forment toujours un triangle équilatéral. C'est le
\omterm{pb:b1:complex:1}{théorème de Napoléon}\index{théorème de Napoléon} --- un énoncé sans
raison visible d'être vrai, que l'algèbre de $j = \eu^{2\iu\pi/3}$
démontre en trois lignes de calcul. Ce problème construit tout
l'outillage (rotations, similitudes directes, le critère en $j$ de
l'\cref{exo:b1:complex:10}), démontre le \omterm{pb:b1:complex:1}{théorème de Napoléon}
intérieur et extérieur, localise le cas dégénéré, et se referme sur
un second joyau de la même école~: le \omterm{pb:b1:complex:1}{théorème de van Schooten} sur
les triangles équilatéraux et l'\omterm{pb:b1:complex:1}{inégalité de Ptolémée}. Dans tout le
problème, les points du plan sont identifiés à leurs affixes.

\medskip
\noindent\textbf{Partie I --- Le nombre $j$ et les rotations.}
\begin{enumerate}
  \item Calculer la forme algébrique de $j$, puis $j^2$, $j^3$, $1 +
        j + j^2$, $\conj j$ et $j^{-1}$. Placer $1$, $j$, $j^2$ sur
        un croquis du cercle unité.
  \item Montrer que la rotation de centre $a$ et d'angle $\theta$
        est $r(z) = a + \eu^{\iu\theta}(z - a)$, et que la composée
        de deux rotations, d'angles $\theta$ et $\theta'$, est une
        rotation d'angle $\theta + \theta'$ si $\theta + \theta'
        \notin 2\pi\Z$, et une translation sinon. (Utiliser la
        \cref{prop:b1:complex:similitude}.)
  \item Soit $b \neq c$. Montrer qu'il existe exactement deux points
        $p$ rendant $(b, c, p)$ équilatéral, à savoir $p = b +
        \eu^{\pm\iu\pi/3}(c - b)$. Pour un triangle \emph{direct}
        $(a, b, c)$, vérifier sur l'exemple $a = 0$, $b = 1$, $c =
        \iu$ que le choix $\eu^{-\iu\pi/3}$ est celui qui se situe
        de l'autre côté de la droite $BC$ par rapport à $a$ --- le
        sommet \emph{extérieur}.
  \item D'après la solution de l'\cref{exo:b1:complex:10}, $(a, b,
        c)$ est équilatéral direct si et seulement si $a + jb + j^2c
        = 0$. Démontrer les deux identités de rotation
        \[
        b + jc + j^2a = j^2\,(a + jb + j^2c),
        \qquad
        c + ja + j^2b = j\,(a + jb + j^2c),
        \]
        et en déduire que le critère est invariant par \omterm{def:b1:counting:objects}{permutation}
        circulaire de $(a, b, c)$.
  \item Montrer que si $a + jb + j^2c = 0$ et si deux des trois
        points coïncident, alors les trois coïncident. (Le critère
        caractérise donc exactement~: triangle équilatéral direct,
        ou point unique.)
\end{enumerate}

\medskip
\noindent\textbf{Partie II --- Les similitudes directes.}
\begin{enumerate}[resume]
  \item Montrer que les \omterm{def:b1:logic:map}{applications} $f(z) = az + b$ avec $a \in
        \C^*$ (les \emph{similitudes directes}) sont stables par
        composition et par passage à la réciproque~: la composée de
        deux d'entre elles, et la réciproque de l'une quelconque
        d'entre elles, est encore de cette forme. (Dans le langage
        du \cref{ch:b1:structures}~: elles forment un groupe.)
  \item Étant donnés $z_1 \neq z_2$ et $w_1 \neq w_2$, montrer qu'il
        existe une \emph{unique} similitude directe $f$ telle que
        $f(z_1) = w_1$ et $f(z_2) = w_2$, et donner $a$ et $b$
        explicitement.
  \item Montrer qu'une similitude directe $f(z) = az + b$ multiplie
        toutes les distances par $\abs a$ et conserve la
        \emph{forme} $\frac{c - a'}{b' - a'}$ de tout triangle $(a',
        b', c')$ \emph{(numérateur et dénominateur sont tous deux
        multipliés par $a$)}. En déduire que deux triangles sont
        directement semblables exactement lorsque leurs formes sont
        égales.
  \item Déterminer complètement la similitude directe telle que
        $f(0) = 1$ et $f(1) = \iu$~: donner $a$, $b$, le point fixe,
        le rapport et l'angle.
  \item Soit $\alpha + \beta = 1$. Montrer que le «~point pondéré~»
        $g(a', b') = \alpha a' + \beta b'$ commute avec toute
        similitude directe~: $f(\alpha a' + \beta b') = \alpha f(a')
        + \beta f(b')$. En déduire que les isobarycentres, les
        milieux et les centres de Napoléon ci-dessous sont tous
        transportés par les similitudes --- c'est la licence
        algébrique qui se cache derrière tout \omterm{def:b1:complex:expi}{argument} du type
        «~sans perte de généralité, plaçons le cercle circonscrit
        sur le cercle unité~».
\end{enumerate}

\medskip
\noindent\textbf{Partie III --- Le \omterm{pb:b1:complex:1}{théorème de Napoléon}.} Soit $(a,
b, c)$ un triangle direct. Sur chaque côté, construisons vers
l'extérieur le triangle équilatéral (question 3) et notons $n_a$,
$n_b$, $n_c$ les centres (isobarycentres) des triangles construits
respectivement sur $[b, c]$, $[c, a]$, $[a, b]$.
\begin{enumerate}[resume]
  \item Montrer que
        \[
        n_a = \alpha b + \beta c, \qquad
        n_b = \alpha c + \beta a, \qquad
        n_c = \alpha a + \beta b,
        \qquad\text{avec}\quad
        \alpha = \frac{3 + \iu\sqrt3}{6},\ \beta = \conj\alpha .
        \]
  \item Vérifier que $\alpha + \beta = 1$, et en déduire que le
        triangle $(n_a, n_b, n_c)$ a le \emph{même isobarycentre}
        que $(a, b, c)$.
  \item À l'aide des identités de la question 4, montrer que
        \[
        n_a + j\,n_b + j^2 n_c
        = (\alpha j^2 + \beta j)\,(a + jb + j^2c) .
        \]
  \item Calculer $\alpha j + \beta$ et conclure~: $\alpha j^2 +
        \beta j = j(\alpha j + \beta) = 0$, donc $n_a + j n_b + j^2
        n_c = 0$ pour \emph{tout} triangle~: le triangle de Napoléon
        extérieur est équilatéral direct (ou réduit à un point). Le
        \omterm{pb:b1:complex:1}{théorème de Napoléon} est démontré.
  \item Construire cette fois les triangles équilatéraux vers
        l'\emph{intérieur} (choix $\eu^{+\iu\pi/3}$ à la question 3)
        et noter $m_a, m_b, m_c$ leurs centres. Montrer que $m_a =
        \beta b + \alpha c$ (et de même par \omterm{def:b1:counting:objects}{permutation} circulaire),
        puis démontrer que $m_a + j^2 m_b + j\,m_c = 0$~: le
        triangle de Napoléon intérieur est équilatéral lui aussi,
        avec l'orientation opposée.
  \item Localiser la dégénérescence~: montrer que $n_a = n_b = n_c$
        se produit exactement lorsque $a + j^2 b + jc = 0$,
        c'est-à-dire lorsque $(a, b, c)$ est un triangle équilatéral
        \emph{indirect}. \emph{(Deux des points $n_a, n_b, n_c$
        coïncident si et seulement si les trois coïncident, d'après
        la question 5~; utiliser ensuite $n_a + j^2 n_b + j n_c =
        j(\alpha + \beta j)(a + j^2b + jc)$ et vérifier que $\alpha
        + \beta j \neq 0$.)}
  \item Mener tout le calcul sur le triangle $a = 0$, $b = 1$, $c =
        \iu$~: donner $n_a, n_b, n_c$ exactement, et vérifier par le
        calcul direct des trois carrés des longueurs des côtés que
        le triangle est équilatéral, de côté au carré $\frac{2 +
        \sqrt3}{3}$.
\end{enumerate}

\medskip
\noindent\textbf{Partie IV --- Ptolémée et van Schooten.}
\begin{enumerate}[resume]
  \item Démontrer l'identité, valable pour tous complexes $a, b, c,
        d$~:
        \[
        (a - b)(c - d) + (a - d)(b - c) = (a - c)(b - d) .
        \]
  \item En déduire l'\emph{inégalité de
        Ptolémée}\index{inégalité de Ptolémée}~: pour quatre points
        quelconques $A, B, C, D$,
        \[
        AC \cdot BD \;\leq\; AB \cdot CD + AD \cdot BC ,
        \]
        avec égalité si et seulement si $(a-b)(c-d)$ et
        $(a-d)(b-c)$ sont portés par une même demi-droite issue de
        $0$.
  \item Montrer que, pour $\theta, \varphi \in \R$,
        $\abs{\eu^{\iu\theta} - \eu^{\iu\varphi}} =
        2\,\abs{\sin\frac{\theta - \varphi}2}$.
  \item (\omterm{pb:b1:complex:1}{théorème de van Schooten}\index{théorème de van Schooten})
        Soit $(a, b, c) = (1, j, j^2)$ --- d'après la partie II,
        cela ne restreint pas la généralité parmi les triangles
        équilatéraux directs --- et soit $p = \eu^{\iu\theta}$ avec
        $\theta \in \intoo{2\pi/3}{4\pi/3}$, point du cercle
        circonscrit situé sur l'arc $BC$ ne contenant pas $A$.
        Démontrer que
        \[
        PA = PB + PC .
        \]
        \emph{(Exprimer les trois distances à l'aide de la question
        20 et utiliser une formule de transformation de somme en
        produit.)}
  \item Vérifier en $p = -1$ qu'il s'agit exactement du cas
        d'égalité de l'\omterm{pb:b1:complex:1}{inégalité de Ptolémée} pour l'ordre cyclique
        $A, B, P, C$~: calculer $(a - b)(p - c)$ et $(a - c)(b - p)$
        et vérifier que leur rapport est un réel strictement
        positif.
\end{enumerate}

\medskip
\noindent\textbf{Partie V --- Synthèse.}
\begin{enumerate}[resume]
  \item Calculer le triangle de Napoléon extérieur du triangle
        équilatéral $(1, j, j^2)$ lui-même, et décrire
        géométriquement le résultat.
  \item Où exactement le problème a-t-il utilisé~: (i) la
        multiplication comme rotation~; (ii) la structure de groupe
        des similitudes et l'invariant de forme~; (iii) la
        factorisation par l'angle moitié de la
        \cref{met:b1:complex:trig}~? Une phrase pour chacun.
  \item En un court paragraphe, dégager la morale du problème~:
        qu'apporte le dictionnaire entre la géométrie plane et
        l'algèbre de $\C$, que coûte-t-il, et laquelle des deux
        démarches --- l'identité $\alpha j + \beta = 0$ ou la figure
        classique --- \emph{explique}-t-elle le mieux le \omterm{pb:b1:complex:1}{théorème de
        Napoléon}, à votre avis~? Citer un endroit où le dictionnaire
        réapparaîtra sous forme matricielle plus loin dans ce
        volume.
\end{enumerate}
\end{problem}
