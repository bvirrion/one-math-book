\chapter{Linear Maps}\label{ch:b1:linmaps}

The maps worth studying between vector spaces are those compatible
with the structure: \emph{linear maps}. Their two fundamental
subspaces --- kernel and image --- measure injectivity and
surjectivity, and in finite dimension the rank--nullity theorem
binds their sizes into one conservation law. Projections and
symmetries, then linear forms and hyperplanes, close the chapter.

\section{Definitions and first properties}

\begin{definition}\label{def:b1:linmaps:def}
Let $E, F$ be $K$-vector spaces. A map $u \colon E \to F$ is
\emph{linear}\index{linear map} when
\[
\forall x, y \in E,\ \forall \lambda \in K, \qquad
u(x + \lambda y) = u(x) + \lambda u(y).
\]
Then $u(0) = 0$ and $u(\sum \lambda_i x_i) = \sum \lambda_i u(x_i)$.
The set $\mathcal{L}(E, F)$ of linear maps is itself a vector space;
composition of linear maps is linear, and bilinear in the pair. An
\emph{endomorphism} is a linear $E \to E$; an \emph{isomorphism} is
a bijective linear map (its inverse is then automatically linear);
$u^{-1}$ of an isomorphism, compositions of isomorphisms, are
isomorphisms.
\end{definition}

\begin{proposition}[A linear map is known on a basis]\label{prop:b1:linmaps:basis}
Let $(e_1, \dots, e_n)$ be a basis of $E$ and $(v_1, \dots, v_n)$
arbitrary vectors of $F$. There is exactly one linear map $u \colon
E \to F$ with $u(e_i) = v_i$ for all $i$. Moreover:
\[
u \text{ injective} \iff (v_i) \text{ free};
\qquad
u \text{ surjective} \iff (v_i) \text{ generates } F .
\]
\end{proposition}

\begin{proof}
Existence/uniqueness: every $x$ has unique coordinates $x = \sum
\lambda_i e_i$ (\cref{prop:b1:vspaces:coordinates}); linearity
forces $u(x) = \sum \lambda_i v_i$, and this formula does define a
linear map. Injectivity: $u(\sum\lambda_i e_i) = 0$ means
$\sum\lambda_i v_i = 0$; ``only the trivial one'' on the left
(injectivity, via \cref{prop:b1:linmaps:kernel} below) matches
``only the trivial one'' on the right (freeness). Surjectivity: the
image of $u$ is exactly $\operatorname{Vect}(v_1, \dots, v_n)$.
\end{proof}

\begin{definition}[Kernel and image]\label{def:b1:linmaps:kerim}
For $u \in \mathcal{L}(E, F)$:
\[
\ker u = \{x \in E : u(x) = 0\} \subseteq E,
\qquad
\operatorname{im} u = u(E) \subseteq F ,
\]
both subspaces (direct verification with the
criterion).\index{kernel}\index{image}
\end{definition}

\begin{proposition}\label{prop:b1:linmaps:kernel}
$u$ is injective $\iff$ $\ker u = \{0\}$; $u$ is surjective $\iff$
$\operatorname{im} u = F$.
\end{proposition}

\begin{proof}
As for groups (\cref{prop:b1:structures:kernel}): $u(x) = u(y) \iff
u(x - y) = 0 \iff x - y \in \ker u$. The second point is the
definition.
\end{proof}

\section{The rank--nullity theorem}

\begin{definition}\label{def:b1:linmaps:rank}
The \emph{rank}\index{rank!of a linear map} of $u \in \mathcal{L}(E,
F)$ (with $E$ finite-dimensional) is $\operatorname{rk} u = \dim
\operatorname{im} u$ --- also the rank of the family
$\bigl(u(e_1), \dots, u(e_n)\bigr)$ for any basis $(e_i)$ of $E$.
\end{definition}

\begin{theorem}[Rank--nullity]\label{thm:b1:linmaps:ranknullity}
Let $E$ be finite-dimensional and $u \in \mathcal{L}(E, F)$.
Then\index{rank--nullity theorem}
\[
\dim E = \dim \ker u + \operatorname{rk} u .
\]
More precisely, if $S$ is any supplementary subspace of $\ker u$ in
$E$, then $u$ restricts to an \emph{isomorphism} from $S$ onto
$\operatorname{im} u$.
\end{theorem}

\begin{proof}
Let $S$ satisfy $E = \ker u \oplus S$
(\cref{thm:b1:findim:subspaces}), and let $v \colon S \to
\operatorname{im} u$ be the restriction of $u$.

$v$ is injective: $\ker v = S \cap \ker u = \{0\}$.

$v$ is surjective: any $u(x)$ with $x = k + s$ ($k \in \ker u$, $s
\in S$) equals $u(s) = v(s)$.

So $v$ is an isomorphism; an isomorphism sends a basis to a basis
(\cref{prop:b1:linmaps:basis}), so $\dim S = \dim\operatorname{im}
u$, and $\dim E = \dim\ker u + \dim S$ concludes.
\end{proof}

\begin{corollary}\label{cor:b1:linmaps:samedim}
If $\dim E = \dim F$ (finite), then for $u \in \mathcal{L}(E, F)$:
\[
u \text{ injective} \iff u \text{ surjective} \iff u \text{
bijective}.
\]
In particular this holds for endomorphisms in finite dimension. (It
fails in infinite dimension: on $K[X]$, the derivative is surjective
but not injective, and $P \mapsto XP$ is injective but not
surjective.)
\end{corollary}

\begin{proof}
Injective $\iff \dim\ker u = 0 \iff \operatorname{rk} u = \dim E =
\dim F \iff \operatorname{im} u = F$ (a subspace of full dimension
is everything, \cref{thm:b1:findim:subspaces}) $\iff$ surjective.
\end{proof}

\begin{example}[Interpolation, structurally]\label{ex:b1:linmaps:interpolation}
Fix distinct $x_0, \dots, x_n$ and let $u \colon \R_n[X] \to
\R^{n+1}$, $P \mapsto (P(x_0), \dots, P(x_n))$: linear. Its kernel
is $\{P : \deg P \leq n,\ n+1 \text{ roots}\} = \{0\}$
(\cref{cor:b1:poly:nroots}). Equal dimensions $n + 1$: $u$ is an
isomorphism --- existence \emph{and} uniqueness of the Lagrange
interpolant (\cref{thm:b1:poly:lagrange}) in one line.
\end{example}

\section{Projections and symmetries}

\begin{definition}\label{def:b1:linmaps:projection}
Let $E = F \oplus G$. The \emph{projection onto $F$ along
$G$}\index{projection} maps $x = f + g$ (unique decomposition) to
$p(x) = f$; the associated \emph{symmetry}\index{symmetry} is $s(x)
= f - g$. Both are linear, and $s = 2p - \mathrm{id}$.
\end{definition}

\begin{theorem}[Algebraic characterization]\label{thm:b1:linmaps:projchar}
\begin{enumerate}
  \item An endomorphism $p$ is a projection (onto some $F$ along
        some $G$) if and only if $p \circ p = p$; then $F =
        \operatorname{im} p = \ker(p - \mathrm{id})$ and $G = \ker
        p$.
  \item An endomorphism $s$ is a symmetry if and only if $s \circ s
        = \mathrm{id}$; then $E = \ker(s - \mathrm{id}) \oplus
        \ker(s + \mathrm{id})$.
\end{enumerate}
\end{theorem}

\begin{proof}
(1) A projection satisfies $p(f + g) = f$ and $p(f) = f$: $p^2 = p$.
Conversely, let $p^2 = p$; set $F = \operatorname{im} p$, $G = \ker
p$. Every $x$ writes $x = p(x) + (x - p(x))$ with $p(x) \in F$ and
$p\bigl(x - p(x)\bigr) = p(x) - p^2(x) = 0$: $E = F + G$. If $y \in
F \cap G$: $y = p(z)$ and $p(y) = 0$, so $y = p(z) = p^2(z) = p(y) =
0$: direct sum, and $p$ is the projection onto $F$ along $G$.
Finally on $F$: $y = p(z)$ gives $p(y) = y$, so $F \subseteq
\ker(p - \mathrm{id})$, and conversely $p(y) = y$ puts $y$ in the
image.

(2) $s = 2p - \mathrm{id}$ is bijective algebra: $s^2 = \mathrm{id}
\iff 4p^2 - 4p + \mathrm{id} = \mathrm{id} \iff p^2 = p$ where $p =
\frac{s + \mathrm{id}}{2}$. The eigenspace decomposition is $F =
\ker(s - \mathrm{id})$, $G = \ker(s + \mathrm{id})$ under the
correspondence.
\end{proof}

\section{Linear forms and hyperplanes}

\begin{definition}\label{def:b1:linmaps:forms}
A \emph{linear form}\index{linear form} on $E$ is a linear map
$\varphi \colon E \to K$. A \emph{hyperplane}\index{hyperplane} of
$E$ ($\dim E = n$) is a subspace of dimension $n - 1$.
\end{definition}

\begin{theorem}\label{thm:b1:linmaps:hyperplanes}
The hyperplanes of $E$ are exactly the kernels of nonzero linear
forms. Two nonzero forms have the same kernel if and only if they
are proportional.
\end{theorem}

\begin{proof}
If $\varphi \neq 0$: $\operatorname{rk}\varphi = 1$ (image a nonzero
subspace of $K$), so $\dim\ker\varphi = n - 1$: a hyperplane.
Conversely, let $H$ be a hyperplane, $(e_1, \dots, e_{n-1})$ a basis
of $H$ completed by $e_n$: the form ``last coordinate'' has kernel
$H$.

Proportional forms share their kernel. Conversely, suppose
$\ker\varphi = \ker\psi = H$ and pick $a \notin H$: every $x$ writes
$x = h + \lambda a$ (as $E = H \oplus Ka$), and
\[
\varphi(x) = \lambda \varphi(a), \qquad \psi(x) = \lambda\psi(a):
\]
so $\varphi = \frac{\varphi(a)}{\psi(a)}\,\psi$.
\end{proof}

\begin{example}\label{ex:b1:linmaps:hyperplane}
In $K^n$, a hyperplane is a solution set $\{a_1 x_1 + \dots + a_n
x_n = 0\}$ with not all $a_i$ zero --- the familiar equation of a
plane through the origin in $\R^3$. In function spaces, evaluation
forms $P \mapsto P(1)$ or $f \mapsto \int_0^1 f$ define hyperplanes
of $\R_n[X]$, of $C(\intcc{0}{1})$ (cf.\ \cref{exo:b1:findim:6}).
\end{example}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:b1:linmaps:1}
Which maps are linear?
\begin{enumerate}
  \item $\R^2 \to \R^2$, $(x, y) \mapsto (x + y, x - 2y)$;
  \item $\R^2 \to \R$, $(x, y) \mapsto xy$;
  \item $\R[X] \to \R[X]$, $P \mapsto P' + XP$;
  \item $\mathcal{F}(\R,\R) \to \R$, $f \mapsto f(3)$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:linmaps:2}
Let $u \colon \R^3 \to \R^3$, $(x,y,z) \mapsto (x + y - z,\; 2x + y
+ z,\; 3x + 2y)$. Determine $\ker u$ (basis, dimension),
$\operatorname{rk} u$, and a basis of $\operatorname{im} u$. Is $u$
injective? surjective?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:linmaps:3}
Let $u \colon \R_n[X] \to \R_n[X]$, $P \mapsto P - P'$. Prove that
$u$ is an isomorphism: once via $\ker u$, once by exhibiting the
inverse \emph{(consider $P + P' + P'' + \dots$)}.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:linmaps:4}
Let $u \in \mathcal{L}(E, F)$ and $v \in \mathcal{L}(F, G)$, spaces
finite-dimensional. Prove:
\[
\operatorname{rk}(v \circ u) \leq
\min\bigl(\operatorname{rk} u,\ \operatorname{rk} v\bigr).
\]
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:linmaps:5}
Let $u$ be an endomorphism of $E$ (finite-dimensional) with $u^2 =
0$. Prove that $\operatorname{im} u \subseteq \ker u$, hence
$\operatorname{rk} u \leq \frac{\dim E}{2}$. For $E = \R^2$, give an
example with equality.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:linmaps:6}
Let $p, q$ be projections of $E$ with $p \circ q = q \circ p$.
Prove that $p \circ q$ is a projection, with
\[
\operatorname{im}(pq) = \operatorname{im} p \cap \operatorname{im}
q ,
\qquad
\ker (pq) = \ker p + \ker q .
\]
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:linmaps:7}
Let $u \in \mathcal{L}(E)$, $E$ finite-dimensional. Prove the
equivalence of:
\begin{enumerate}
  \item $E = \ker u \oplus \operatorname{im} u$;
  \item $\ker u = \ker u^2$;
  \item $\operatorname{im} u = \operatorname{im} u^2$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:linmaps:8}
Let $\varphi, \psi$ be linear forms on $E$ with $\ker\varphi
\subseteq \ker\psi$. Prove $\psi = \lambda\varphi$ for some
$\lambda \in K$ (including the degenerate cases).
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:linmaps:9}
Let $u \in \mathcal{L}(E)$ with $\dim E = n$, and suppose $u^n = 0$
but $u^{n-1} \neq 0$ (a \emph{maximally nilpotent} endomorphism).
Pick $x$ with $u^{n-1}(x) \neq 0$; prove that $\bigl(x, u(x), \dots,
u^{n-1}(x)\bigr)$ is a basis of $E$. \emph{(Apply powers of $u$ to a
null combination, starting with $u^{n-1}$.)}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:linmaps:10}
Let $f \in \mathcal{L}(\R^n)$ with $f \circ f = -\mathrm{id}$.
\begin{enumerate}
  \item Prove that $f$ is an isomorphism and that no $x \neq 0$
        satisfies $f(x) = \lambda x$ with $\lambda \in \R$.
  \item Prove that $n$ is even. \emph{Hint: pick $x_1 \neq 0$; show
        $\operatorname{Vect}(x_1, f(x_1))$ is a plane stable under
        $f$; choose $x_2$ outside it and iterate, proving that
        $\bigl(x_1, f(x_1), x_2, f(x_2), \dots\bigr)$ stays free.}
\end{enumerate}
\end{exercise}
