\chapter{Rational Fractions}\label{ch:b1:fractions}

A \omterm{def:b1:fractions:field}{rational fraction} is a quotient of \omterm{def:b1:poly:def}{polynomials}. The central theorem
of this short chapter --- partial fraction
decomposition\index{partial fraction decomposition} --- breaks any
such quotient into a sum of elementary bricks $\frac{c}{(X - a)^k}$.
Beyond its algebraic interest, it is the standard machine for
integrating rational functions (\cref{ch:b1:integration}) and for
summing certain series (\cref{ch:b1:series}).

\section{The field $K(X)$}

\begin{definition}\label{def:b1:fractions:field}
A \emph{rational fraction}\index{rational fraction} over $K$ ($= \R$
or $\C$) is a quotient $F = \frac{A}{B}$ with $A, B \in K[X]$, $B \neq
0$; two quotients $\frac AB$ and $\frac{A'}{B'}$ are identified when
$AB' = A'B$. Every fraction has a \emph{reduced form} with
$\gcd(A, B) = 1$, unique up to constants. With the natural operations,
the \omterm{def:b1:logic:sets}{set} $K(X)$ of rational fractions is a \omterm{def:b1:structures:field}{field}.

The \emph{poles}\index{pole} of $F$ (in reduced form) are the roots of
$B$; the \emph{order} of a pole is its \omterm{def:b1:poly:derivative}{multiplicity} as a root of $B$.
The \emph{degree} of $F$ is $\deg F = \deg A - \deg B \in \Z \cup
\{-\infty\}$.
\end{definition}

\begin{example}[Reading off poles, orders and degree]\label{ex:b1:fractions:vocab}
Let $F = \dfrac{X^3 - X}{X^4 - 2X^3 + X^2}$. Factor both layers:
numerator $X(X-1)(X+1)$, denominator $X^2(X-1)^2$; cancel the
common factor $X(X-1)$:
\[
F = \frac{X + 1}{X(X - 1)} \quad\text{(reduced form).}
\]
\omterm{def:b1:fractions:field}{Poles}: $0$ and $1$, both \emph{simple} --- the orders are read on
the reduced denominator, so the apparent double roots of the
original one are irrelevant. Degree: $\deg F = 1 - 2 = -1$,
visible asymptotically ($xF(x) \to 1$ as $x \to \infty$). The
degree behaves like the \omterm{def:b1:poly:def}{polynomial} degree ($\deg FG = \deg F +
\deg G$, $\deg(F + G) \leq \max$), a bookkeeping rule used
constantly in the coefficient hunts below: each ``limit of
$xF(x)$'' argument is a degree count in disguise.
\end{example}

\begin{proposition}[Integer part]\label{prop:b1:fractions:integerpart}
Every $F = \frac AB \in K(X)$ is uniquely $F = E + \frac RB$ with $E
\in K[X]$ (the \emph{integer part}, or \omterm{def:b1:poly:def}{polynomial} part, of $F$) and
$\deg R < \deg B$. One has $E \neq 0$ iff $\deg F \geq 0$.
\end{proposition}

\begin{proof}
Euclidean division $A = BE + R$ (\cref{thm:b1:poly:division}), divided
by $B$. Uniqueness: if $E + \frac RB = E' + \frac{R'}{B}$, then $(E -
E')B = R' - R$ with $\deg(R' - R) < \deg B$, forcing $E = E'$ then $R
= R'$.
\end{proof}

\begin{example}[Reduce first, then divide]\label{ex:b1:fractions:integerpart}
Find the integer part of $F = \dfrac{X^3 + 1}{X^2 - 1}$. Dividing
blindly: $X^3 + 1 = (X^2 - 1)X + (X + 1)$, so $F = X +
\frac{X + 1}{X^2 - 1}$. But the fraction was not reduced: $X^3 + 1
= (X + 1)(X^2 - X + 1)$ and $X^2 - 1 = (X+1)(X-1)$ share the factor
$X + 1$, and
\[
F = \frac{X^2 - X + 1}{X - 1} = X + \frac{1}{X - 1} :
\]
the same integer part $X$, but the fractional part collapses to a
single brick, and the ``\omterm{def:b1:fractions:field}{pole}'' at $-1$ was never a \omterm{def:b1:fractions:field}{pole} at all.
Always reduce to lowest terms before hunting \omterm{def:b1:fractions:field}{poles}: the \omterm{def:b1:fractions:field}{poles} of
$F$ are the roots of the \emph{reduced} denominator. (The integer
part is insensitive to the simplification, as the uniqueness in
\cref{prop:b1:fractions:integerpart} guarantees.)
\end{example}

\section{Partial fraction decomposition over $\C$}

\begin{theorem}[Decomposition over $\C$]\label{thm:b1:fractions:complex}
Let $F = \frac AB \in \C(X)$ be in reduced form, with $B = c\,(X -
a_1)^{m_1} \cdots (X - a_r)^{m_r}$. Then $F$ is, in a unique way,
\[
F = E + \sum_{i=1}^{r} \sum_{k=1}^{m_i}
\frac{c_{i,k}}{(X - a_i)^k},
\qquad E \in \C[X],\ c_{i,k} \in \C ,
\]
$E$ being the integer part of $F$.
\end{theorem}

\begin{proof}
By \cref{prop:b1:fractions:integerpart} we may assume $\deg A < \deg
B$ and prove the sum decomposition with $E = 0$.

\emph{Splitting the \omterm{def:b1:fractions:field}{poles}.} Write $B = (X - a_1)^{m_1} B_1$ with
$B_1(a_1) \neq 0$. The \omterm{def:b1:poly:def}{polynomials} $(X-a_1)^{m_1}$ and $B_1$ are
\omterm{cor:b1:arith:bezout}{coprime} (no common root), so by Bézout in $\C[X]$ (see the remark in
\cref{ch:b1:poly}) there are $U_0, V_0$ with $U_0 (X-a_1)^{m_1} +
V_0 B_1 = 1$; multiplying by $A$ and setting $U = AU_0$, $V =
AV_0$, then dividing by $B$:
\[
\frac AB = \frac{V}{(X - a_1)^{m_1}} + \frac{U}{B_1} .
\]
The degree conditions can be enforced: divide $V$ by
$(X-a_1)^{m_1}$, say $V = (X-a_1)^{m_1}Q + V_1$ with $\deg V_1 <
m_1$, and absorb the quotient into the second term ($U_1 = U +
QB_1$):
\[
\frac AB = \frac{V_1}{(X - a_1)^{m_1}} + \frac{U_1}{B_1},
\qquad \deg V_1 < m_1 ;
\]
comparing degrees ($\deg A < \deg B$ and $\deg V_1 < m_1$) forces
$\deg U_1 < \deg B_1$ as well. Iterating on $\frac{U_1}{B_1}$,
\omterm{def:b1:fractions:field}{pole} after \omterm{def:b1:fractions:field}{pole}, reduces everything to the \omterm{def:b1:fractions:field}{one-pole} case below.

\emph{One \omterm{def:b1:fractions:field}{pole}.} For $\frac{A}{(X-a)^m}$ with $\deg A < m$: expand $A$
in powers of $(X - a)$, $A = \sum_{j=0}^{m-1} \alpha_j (X - a)^j$
(Taylor expansion of a \omterm{def:b1:poly:def}{polynomial}, as in the proof of
\cref{prop:b1:poly:multiplicity}); dividing gives exactly the bricks
$\frac{\alpha_j}{(X-a)^{m-j}}$. Concretely, for $\frac{X^2 +
1}{(X+2)^3}$: substituting $X = Y - 2$,
\[
X^2 + 1 = (Y - 2)^2 + 1 = Y^2 - 4Y + 5 ,
\qquad\text{so}\qquad
\frac{X^2+1}{(X+2)^3} = \frac1{Y} - \frac4{Y^2} + \frac5{Y^3}
\]
with $Y = X + 2$: the three bricks appear by mere division of the
shifted expansion --- the fastest route whenever a single
high-order \omterm{def:b1:fractions:field}{pole} is involved, and the recommended one for
\cref{exo:b1:fractions:3}.

\emph{Uniqueness.} Suppose two decompositions coincide; their
difference is an identity $0 = \sum_{i,k} \frac{d_{i,k}}{(X -
a_i)^k}$. Multiply through by $(X - a_1)^{m_1}$: every term
acquires a factor vanishing at $a_1$ \emph{except} the one with
$i = 1$, $k = m_1$, whose coefficient becomes $d_{1,m_1}$ plus
terms carrying at least one factor $(X - a_1)$. Evaluating at
$a_1$ (legitimate: after the multiplication, no \omterm{def:b1:fractions:field}{pole} remains at
$a_1$) gives $d_{1,m_1} = 0$. With the top coefficient gone,
repeat with $(X - a_1)^{m_1 - 1}$, and so on downward to $k = 1$;
then move to the next \omterm{def:b1:fractions:field}{pole}. All $d_{i,k}$ vanish: the
decomposition is unique.
\end{proof}

\begin{method}[Computing the coefficients]\label{met:b1:fractions:compute}
In practice, avoid Bézout; combine:
\begin{enumerate}
  \item \emph{cover-up for the highest power}: the coefficient of
        $\frac{1}{(X-a)^m}$ (with $m$ the order of the \omterm{def:b1:fractions:field}{pole} $a$) is
        \[
        c_{a,m} = \Bigl[\,(X - a)^m F\,\Bigr]_{X = a};
        \]
        for a \emph{simple} \omterm{def:b1:fractions:field}{pole} of $F = \frac AB$, this is
        $\frac{A(a)}{B'(a)}$;
  \item \emph{evaluations} at convenient points and \emph{limits} of
        $xF(x)$ as $x \to \infty$ to collect linear relations for the
        remaining coefficients;
  \item \emph{parity} or conjugation symmetries, when present, to cut
        the work in half.
\end{enumerate}
\end{method}

\begin{remark}[Common pitfalls with partial fractions]\label{rem:b1:fractions:pitfalls}
\begin{enumerate}
  \item \emph{Skipping the integer part.} The brick decomposition
        applies to fractions of degree $< 0$; when $\deg A \geq
        \deg B$, divide first
        (\cref{prop:b1:fractions:integerpart}), or the
        coefficient hunt will produce contradictions.
  \item \emph{Cover-up beyond its scope.} Multiplying by
        $(X - a)^k$ and evaluating at $a$ yields the coefficient
        only for $k = m$, the \emph{full} order of the \omterm{def:b1:fractions:field}{pole}; the
        lower-order coefficients require other relations (limits,
        evaluations) --- see \cref{ex:b1:fractions:multiple}.
  \item \emph{Forgetting to reduce.} \omterm{def:b1:fractions:field}{Poles} are read on the
        \emph{reduced} form; a common factor between numerator
        and denominator creates phantom \omterm{def:b1:fractions:field}{poles}
        (\cref{ex:b1:fractions:integerpart}).
  \item \emph{Wrong brick shapes over $\R$.} Above an irreducible
        quadratic the numerators are \emph{affine} ($\alpha X +
        \beta$), not constant; writing $\frac{c}{X^2 + 1}$ alone
        loses solutions --- the correct shapes are dictated by
        \cref{thm:b1:fractions:real}, never improvised.
\end{enumerate}
\end{remark}

\begin{proof}[Proof of the simple-pole formula]
Near a simple \omterm{def:b1:fractions:field}{pole} $a$: $B = (X - a) Q$ with $Q(a) \neq 0$, and $B' =
Q + (X - a) Q'$, so $B'(a) = Q(a)$. The cover-up value is
$\frac{A(a)}{Q(a)} = \frac{A(a)}{B'(a)}$.
\end{proof}

\begin{example}\label{ex:b1:fractions:simple}
Decompose $F = \dfrac{1}{X(X-1)(X-2)}$. Three simple \omterm{def:b1:fractions:field}{poles}; cover-up
at each:
\[
c_0 = \frac{1}{(0-1)(0-2)} = \frac12,
\quad
c_1 = \frac{1}{1 \times (1 - 2)} = -1,
\quad
c_2 = \frac{1}{2 \times 1} = \frac12,
\]
so $F = \dfrac{1/2}{X} - \dfrac{1}{X-1} + \dfrac{1/2}{X-2}$.
\emph{Check at $X = 3$:} directly, $F(3) = \frac{1}{3\cdot2\cdot1} =
\frac16$; from the decomposition, $\frac{1/2}{3} - \frac{1}{2} +
\frac{1/2}{1} = \frac16 - \frac12 + \frac12 = \frac16$.
\end{example}

\begin{example}[Multiple pole]\label{ex:b1:fractions:multiple}
Decompose $F = \dfrac{X}{(X-1)^2 (X+1)}$. Shape:
$\frac{a}{(X-1)^2} + \frac{b}{X-1} + \frac{c}{X+1}$.
\begin{itemize}
  \item Cover-up at the double \omterm{def:b1:fractions:field}{pole}: $a = \bigl[\frac{X}{X+1}\bigr]_{X=1}
        = \frac12$.
  \item Cover-up at $-1$: $c = \bigl[\frac{X}{(X-1)^2}\bigr]_{X=-1} =
        -\frac14$.
  \item Limit of $xF(x)$ at $\infty$: $0 = b + c$, so $b = \frac14$.
\end{itemize}
\[
F = \frac{1/2}{(X-1)^2} + \frac{1/4}{X-1} - \frac{1/4}{X+1} .
\]
\emph{Check at $X = 0$:} $F(0) = 0$ and $\frac12 - \frac14 - \frac14
= 0$.
\end{example}

\section{Decomposition over $\R$}

\begin{theorem}[Decomposition over $\R$]\label{thm:b1:fractions:real}
Let $F \in \R(X)$ be in reduced form with denominator
\[
B = c \prod_i (X - a_i)^{m_i} \prod_j (X^2 + p_j X + q_j)^{n_j}
\qquad (p_j^2 - 4q_j < 0).
\]
Then $F$ decomposes uniquely as its integer part plus terms
\[
\frac{c_{i,k}}{(X - a_i)^k} \quad (1 \leq k \leq m_i),
\qquad
\frac{\alpha_{j,l}\, X + \beta_{j,l}}{(X^2 + p_j X + q_j)^{l}}
\quad (1 \leq l \leq n_j),
\]
with real coefficients.
\end{theorem}

\begin{proof}
Decompose over $\C$ (\cref{thm:b1:fractions:complex}). Since $F$ is
real, the coefficient over the \omterm{def:b1:fractions:field}{pole} $\conj a$ (at each order) is the
conjugate of the coefficient over $a$ (apply conjugation to the
decomposition and invoke uniqueness). Group each conjugate pair:
\[
\frac{c}{(X - z)^l} + \frac{\conj c}{(X - \conj z)^l}
= \frac{c\,(X - \conj z)^l + \conj c\,(X - z)^l}{\bigl(X^2 - 2\Re(z)X
+ \abs z^2\bigr)^{l}},
\]
whose numerator is its own conjugate, hence real, of degree $\leq l$;
splitting off multiples of the real quadratic lowers it to degree
$\leq 1$ at each level $l$ (a small downward induction). Real \omterm{def:b1:fractions:field}{poles}
keep their real coefficients (conjugation fixes them). Uniqueness
follows from uniqueness over $\C$.
\end{proof}

\begin{example}[Watching the conjugate pairing]\label{ex:b1:fractions:pairing}
The mechanism of the proof, on the smallest case: over $\C$, the
\omterm{def:b1:fractions:field}{poles} of $\frac1{X^2+1}$ are $\pm\iu$, with cover-up coefficients
$\frac1{2\iu}$ at $\iu$ and $\frac1{-2\iu}$ at $-\iu$ ---
conjugates of each other, as the theorem predicts:
\[
\frac{1}{X^2 + 1}
= \frac{1/(2\iu)}{X - \iu} - \frac{1/(2\iu)}{X + \iu} .
\]
Recombining over the common denominator:
\[
\frac{1}{2\iu}\cdot\frac{(X + \iu) - (X - \iu)}{X^2 + 1}
= \frac{1}{2\iu}\cdot\frac{2\iu}{X^2 + 1} = \frac{1}{X^2+1} :
\]
the imaginary parts cancel and the real brick reappears intact.
For real integrands one usually never leaves $\R$ --- but when
evaluating \emph{sums} at complex points (as the weekend problem
does with \omterm{def:b1:complex:unity}{roots of unity}), the complex bricks are the natural
currency, and this pairing is the exchange rate between the two
decompositions.
\end{example}

\begin{example}\label{ex:b1:fractions:real}
Decompose $F = \dfrac{4}{(X^2+1)(X-1)^2}$ over $\R$. Shape:
$\frac{aX + b}{X^2 + 1} + \frac{c}{(X-1)^2} + \frac{d}{X - 1}$.
Cover-up at the double \omterm{def:b1:fractions:field}{pole}: $c = \bigl[\frac{4}{X^2+1}\bigr]_{X=1} =
2$. Cover-up at the complex \omterm{def:b1:fractions:field}{pole} $\iu$ (numerator over $X^2 + 1$
evaluated via the complex decomposition, or directly): multiply by
$X^2 + 1$ and set $X = \iu$:
\[
a\iu + b = \frac{4}{(\iu - 1)^2} = \frac{4}{-2\iu} = 2\iu ,
\]
so $a = 2$, $b = 0$. Limit of $xF(x)$ at infinity: $0 = a + d$, so $d
= -2$. Hence
\[
F = \frac{2X}{X^2+1} + \frac{2}{(X-1)^2} - \frac{2}{X-1} .
\]
\emph{Check at $X = 0$:} $F(0) = 4$ and $0 + 2 + 2 = 4$.
\end{example}

\begin{example}[A cubic denominator, start to finish]\label{ex:b1:fractions:cubic}
Decompose $F = \dfrac{1}{X^3 + 1}$ over $\R$. Factor first: $X^3 +
1 = (X + 1)(X^2 - X + 1)$, the quadratic having discriminant $-3 <
0$. Shape: $\frac{a}{X+1} + \frac{bX + c}{X^2 - X + 1}$. Cover-up
at the simple \omterm{def:b1:fractions:field}{pole} $-1$: $a = \bigl[\frac1{X^2 - X +
1}\bigr]_{X=-1} = \frac13$. Limit of $xF(x)$ at infinity: $0 = a +
b$, so $b = -\frac13$. Evaluation at $X = 0$: $1 = a + c$, so $c =
\frac23$. Hence
\[
\frac{1}{X^3+1} = \frac13\Bigl(\frac{1}{X+1}
+ \frac{-X + 2}{X^2 - X + 1}\Bigr) ,
\]
confirmed at $X = 1$: left side $\frac12$, right side
$\frac13\bigl(\frac12 + 1\bigr) = \frac12$. Note the economy:
three unknowns, three cheap linear facts (one cover-up, one
limit, one evaluation), no expansion of anything --- the workflow
of \cref{met:b1:fractions:compute} in its pure form.
\end{example}

\begin{remark}[What it is for]
Once decomposed, a rational function integrates term by term: bricks
$\frac{1}{(x-a)^k}$ have elementary primitives, and bricks
$\frac{\alpha x + \beta}{(x^2 + px + q)^l}$ reduce to $\ln$ and
$\arctan$ (\cref{ch:b1:integration}). Telescoping sums are the other
standard application (\cref{exo:b1:fractions:8}).
\end{remark}

\begin{remark}[Where this chapter is used]\label{rem:b1:fractions:whereused}
Partial fractions are above all a \emph{preprocessing} step: the
integration chapter (\cref{ch:b1:integration}) feeds every rational
integrand through \cref{thm:b1:fractions:real} before integrating,
and the series chapter (\cref{ch:b1:series}) telescopes rational
terms exactly as in \cref{exo:b1:fractions:8} and in the weekend
problem below --- which pushes the technique all the way to
$\sum 1/k^2 = \pi^2/6$. The logarithmic derivative $P'/P = \sum
m_i/(X - a_i)$ (\cref{exo:b1:poly:11}) reappears whenever root
locations are studied. Beyond this volume, the decomposition of
$1/\chi(X)$ for a characteristic \omterm{def:b1:poly:def}{polynomial} $\chi$ underlies the
computation of matrix powers and of Laplace transforms in the Year
2 volume: the bricks $\frac{c}{(X - a)^k}$ are the algebraic shadow
of the solutions $t^{k-1}\eu^{at}$ met in \cref{ch:b1:diffeq}.
\end{remark}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:b1:fractions:1}
Decompose over $\R$: $\dfrac{1}{X^2 - 1}$;
$\;\dfrac{X}{X^2 - 3X + 2}$; $\;\dfrac{X^2 + 1}{X(X-1)}$ \emph{(mind
the integer part)}.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:fractions:2}
Decompose over $\R$: $\dfrac{1}{X(X^2 + 1)}$ and
$\dfrac{X^3}{X^2 + X + 1}$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:fractions:3}
Decompose $\dfrac{1}{X^2(X - 1)}$ and $\dfrac{X + 1}{(X - 1)^3}$
\emph{(for the second, substitute $Y = X - 1$)}.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:fractions:4}
Decompose over $\C$, then over $\R$: $\dfrac{1}{X^4 - 1}$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:fractions:5}
Decompose over $\R$: $\dfrac{X^2}{(X^2 + 1)^2}$, and deduce a
primitive of $x \mapsto \dfrac{x^2}{(x^2+1)^2}$ given
$\int \frac{\dd x}{(x^2+1)^2} = \frac12\bigl(\arctan x +
\frac{x}{x^2+1}\bigr) + C$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:fractions:6}
For $n \in \N^*$, decompose $F_n = \dfrac{n!}{X(X+1)\cdots(X+n)}$
\emph{(simple \omterm{def:b1:fractions:field}{poles} at $0, -1, \dots, -n$; use the cover-up formula
and recognize \omterm{def:b1:counting:objects}{binomial coefficients})}.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:fractions:7}
Using the identity of \cref{exo:b1:poly:11} for $P = X^n - 1$,
prove that
\[
\sum_{k=0}^{n-1} \frac{1}{X - \omega^k} = \frac{n X^{n-1}}{X^n - 1},
\qquad \omega = \eu^{2\iu\pi/n},
\]
and evaluate both sides at $X = 2$ for $n = 4$ as a check.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:fractions:8}
Decompose $\dfrac{1}{k(k+1)(k+2)}$ and compute
\[
S_n = \sum_{k=1}^{n} \frac{1}{k(k+1)(k+2)},
\qquad\text{then}\qquad
\lim_{n \to \infty} S_n .
\]
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:fractions:9}
Let $P \in \R[X]$ be \omterm{def:b1:poly:def}{monic} of degree $n$ with $n$ distinct real roots
$x_1 < \dots < x_n$. Prove that
\[
\sum_{i=1}^{n} \frac{1}{P'(x_i)} = 0 \quad (n \geq 2),
\qquad
\sum_{i=1}^{n} \frac{x_i^{\,n-1}}{P'(x_i)} = 1 .
\]
\emph{Hint: decompose $\frac{X^m}{P}$ for $m \leq n - 1$ and look at
the coefficient decay at infinity --- or use \omterm{thm:b1:poly:lagrange}{Lagrange interpolation}
(\cref{thm:b1:poly:lagrange}) of the monomial $X^m$ at the nodes
$x_i$.}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:fractions:10}
Decompose $\dfrac{1}{X(X+1)^2}$ over $\R$. Admitting the value
$\sum_{k \geq 1} \frac1{k^2} = \frac{\pi^2}6$ (proved in this
chapter's weekend problem), deduce
\[
\sum_{n=1}^{\infty} \frac{1}{n(n+1)^2} = 2 - \frac{\pi^2}{6} .
\]
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:fractions:11}
Decompose over $\R$: $\dfrac{1}{(X^2+1)(X^2+4)}$, then
$\dfrac{X^2}{(X^2+1)(X^2+4)}$. \emph{Hint: both denominators are
\omterm{def:b1:poly:def}{polynomials} in $X^2$: decompose $\frac1{(Y+1)(Y+4)}$ first.}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:fractions:12}
(Secular equations) Let $F = \sum_{i=1}^{r} \frac{c_i}{X - p_i}$
with $p_1 < p_2 < \dots < p_r$ real and all $c_i > 0$.
\begin{enumerate}
  \item Show that $F$ is strictly decreasing on each interval of
        its domain, and give its limits at $\pm\infty$ and on both
        sides of each \omterm{def:b1:fractions:field}{pole}.
  \item Deduce that for every $\lambda > 0$, the equation $F(x) =
        \lambda$ has exactly $r$ real solutions, one in each
        interval $\intoo{p_i}{p_{i+1}}$ and one beyond $p_r$.
        (Such equations govern eigenvalue perturbations; the
        intermediate value theorem is used at High School level
        here and proved in \cref{ch:b1:continuity}.)
\end{enumerate}
\end{exercise}

\section{Problem: Partial fractions as an engine}

\begin{problem}\label{pb:b1:fractions:1}
Partial fraction decomposition looks like bookkeeping; this problem
shows it is an engine. Fed with the fraction
$\frac1{X(X+1)\cdots(X+k)}$, it telescopes whole families of sums
in closed form; fed with $\frac1{X^n - 1}$, it produces
trigonometric identities such as
\[
\sum_{k=1}^{n-1}\frac{1}{\sin^2\frac{k\pi}{n}} = \frac{n^2-1}{3} ;
\]
and, pushed one step further, that identity squeezes out one of the
most celebrated formulas in mathematics, Euler's
\[
\sum_{k=1}^{\infty}\frac1{k^2} = \frac{\pi^2}{6}
\]
--- here obtained with nothing beyond this chapter's algebra and
high-school trigonometry.\index{telescoping}\index{Basel problem}
Throughout, $\omega = \eu^{2\iu\pi/n}$; limits of sequences are
used at High School level (\cref{ch:b1:seq} formalizes them).

\medskip
\noindent\textbf{Part I --- The telescope.}
\begin{enumerate}
  \item Decompose $\frac1{X(X+1)}$ and compute $\sum_{n=1}^{N}
        \frac1{n(n+1)}$ exactly; conclude that the sum tends to
        $1$.
  \item Same for $\frac1{X(X+2)}$: show $\sum_{n=1}^{N}
        \frac1{n(n+2)} = \frac12\bigl(\frac32 - \frac1{N+1} -
        \frac1{N+2}\bigr) \to \frac34$. (With a gap, \emph{two}
        boundary terms survive at each end.)
  \item Formalize the mechanism: if $F(X) = G(X) - G(X+1)$ for
        some rational $G$ without \omterm{def:b1:fractions:field}{poles} in $\intco1{+\infty}$,
        then $\sum_{n=1}^N F(n) = G(1) - G(N+1)$. Recover the
        value $\frac14$ of \cref{exo:b1:fractions:8} by exhibiting
        the witness $G$ for $F = \frac1{X(X+1)(X+2)}$.
  \item Prove the general factorial telescope: for $k \geq 1$,
        \[
        \frac{1}{X(X+1)\cdots(X+k)}
        = \frac1k\biggl(\frac{1}{X(X+1)\cdots(X+k-1)}
        - \frac{1}{(X+1)\cdots(X+k)}\biggr),
        \]
        and deduce
        \[
        \sum_{n=1}^{\infty}\frac{1}{n(n+1)\cdots(n+k)}
        = \frac{1}{k \cdot k!} .
        \]
        Check the case $k = 2$ against question 3.
  \item Evaluate the decomposition of \cref{exo:b1:fractions:6}
        at well-chosen points to prove
        \[
        \sum_{j=0}^{n}(-1)^j\binom nj\,\frac1{j+1}
        = \frac1{n+1},
        \qquad
        \sum_{j=0}^{n}(-1)^j\binom nj\,\frac1{j+2}
        = \frac1{(n+1)(n+2)} .
        \]
\end{enumerate}

\medskip
\noindent\textbf{Part II --- The fraction $1/(X^n - 1)$.}
\begin{enumerate}[resume]
  \item Show by the cover-up formula that
        \[
        \frac{1}{X^n - 1}
        = \frac1n\sum_{k=0}^{n-1}\frac{\omega^k}{X - \omega^k} .
        \]
  \item Two sanity checks: verify the formula directly for $n =
        2$, and show that the sum of the $n$ coefficients
        vanishes for $n \geq 2$ --- explain why it must (consider
        $xF(x)$ as $x \to \infty$).
  \item Re-derive by cover-up the identity of
        \cref{exo:b1:fractions:7}:
        $\frac{nX^{n-1}}{X^n-1} = \sum_k \frac1{X - \omega^k}$.
  \item Group conjugate \omterm{def:b1:fractions:field}{poles} to prove the real decomposition:
        with $\theta_k = \frac{2k\pi}n$,
        \[
        \frac{\omega^k}{X - \omega^k} +
        \frac{\omega^{n-k}}{X - \omega^{n-k}}
        = \frac{2\cos\theta_k\,X - 2}
        {X^2 - 2\cos\theta_k\,X + 1} ,
        \]
        and write down the full real decomposition of
        $\frac1{X^n-1}$ (distinguish $n$ odd and $n$ even).
  \item Specialize to $n = 4$ and check against
        \cref{exo:b1:fractions:4}.
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Trigonometric sums, and Euler's
$\pi^2/6$.} Let $P = 1 + X + \dots + X^{n-1}$, whose roots are
$\omega, \omega^2, \dots, \omega^{n-1}$ (all simple).
\begin{enumerate}[resume]
  \item Using $\frac{P'}P = \sum_{k=1}^{n-1}\frac1{X - \omega^k}$
        (\cref{exo:b1:poly:11}), prove
        \[
        \sum_{k=1}^{n-1}\frac{1}{1 - \omega^k} = \frac{n-1}2 .
        \]
  \item Prove $\dfrac1{1 - \eu^{\iu\theta}} = \dfrac12 +
        \dfrac\iu2\,\frac{\cos(\theta/2)}{\sin(\theta/2)}$ for
        $\theta \notin 2\pi\Z$ (half-angle factorization,
        \cref{met:b1:complex:trig}), and deduce from question 11
        that $\sum_{k=1}^{n-1}
        \frac{\cos(k\pi/n)}{\sin(k\pi/n)} = 0$ --- also visible
        by the symmetry $k \leftrightarrow n - k$.
  \item Differentiating the identity of question 11 (i.e.\ using
        $\bigl(\frac{P'}P\bigr)' = \frac{P''}P -
        \bigl(\frac{P'}P\bigr)^2$ evaluated at $X = 1$), prove
        \[
        \sum_{k=1}^{n-1}\frac{1}{(1 - \omega^k)^2}
        = \frac{(n-1)(5-n)}{12} .
        \]
  \item Writing $\cot t = \frac{\cos t}{\sin t}$, deduce from
        questions 12--13 the two closed forms
        \[
        \sum_{k=1}^{n-1}\cot^2\frac{k\pi}n = \frac{(n-1)(n-2)}3,
        \qquad
        \sum_{k=1}^{n-1}\frac1{\sin^2\frac{k\pi}n}
        = \frac{n^2-1}3 .
        \]
  \item Verify both formulas by hand for $n = 3$ and $n = 4$.
  \item Prove the inequalities $\cot t < \frac1t < \frac1{\sin t}$
        for $t \in \intoo0{\frac\pi2}$ (from $\sin t < t < \tan
        t$), and deduce, for $n = 2m + 1$ and $1 \leq k \leq m$:
        \[
        \cot^2\frac{k\pi}n \;<\; \frac{n^2}{k^2\pi^2}
        \;<\; \frac1{\sin^2\frac{k\pi}n} .
        \]
  \item Sum these inequalities for $k = 1, \dots, m$ (using the
        symmetry $k \leftrightarrow n - k$ to halve the formulas
        of question 14) and squeeze:
        \[
        \sum_{k=1}^{\infty}\frac1{k^2} = \frac{\pi^2}6 .
        \]
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Higher bricks.}
\begin{enumerate}[resume]
  \item By squaring the decomposition of $\frac1{X^2-1}$ and
        re-decomposing the cross term, prove
        \[
        \frac1{(X^2-1)^2}
        = \frac14\Bigl(\frac1{(X-1)^2} + \frac1{(X+1)^2}\Bigr)
        - \frac14\Bigl(\frac1{X-1} - \frac1{X+1}\Bigr),
        \]
        and check it at $X = 0$.
  \item Combine question 18, the telescope, and Euler's value
        (question 17) to prove
        \[
        \sum_{n=2}^{\infty}\frac1{(n^2-1)^2}
        = \frac{\pi^2}{12} - \frac{11}{16} ,
        \]
        and confirm the value numerically to three decimals.
  \item Differentiate the identity of question 8 to obtain a
        closed form for $\sum_{k=0}^{n-1}\frac1{(X -
        \omega^k)^2}$, and check it at $X = 2$, $n = 2$.
  \item Prove that for $k \geq 2$,
        \[
        \sum_{n=k}^{\infty}\frac{1}{\binom nk} = \frac{k}{k-1} .
        \]
        \emph{(Reduce to question 4 by writing $1/\binom nk$ with
        factorials.)}
  \item For $k = 3$, give the exact partial sum
        $\sum_{n=3}^{N}\frac1{\binom n3}$ and its limit.
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Synthesis.}
\begin{enumerate}[resume]
  \item As a capstone computation, write out the complete real
        decomposition of $\dfrac1{X^6 - 1}$ and check it at $X =
        0$.
  \item Where exactly did the problem use: (i) the uniqueness of
        the decomposition; (ii) the \omterm{def:b1:complex:unity}{roots of unity} from
        \cref{ch:b1:complex}; (iii) the logarithmic derivative
        from \cref{exo:b1:poly:11}? One sentence each.
  \item Synthesis, in a short paragraph: one algebraic identity
        --- breaking a fraction into bricks --- generated exact
        sums, trigonometric identities, and $\pi^2/6$. Comment on
        the division of labor between algebra (exact
        decompositions, valid everywhere) and analysis (limits,
        squeezing), and point to where each thread is
        industrialized: telescoping and comparison in
        \cref{ch:b1:series}, integration of the bricks in
        \cref{ch:b1:integration}.
\end{enumerate}
\end{problem}
