\section*{अध्याय \ref{ch:b1:integration} --- एक खंड पर एकीकरण}

\begin{solution}{exo:b1:integration:1}
$\dfrac{1}{x^2-4} = \dfrac{1/4}{x - 2} - \dfrac{1/4}{x+2}$
(कवर-अप), इसलिए
\[
\int_0^1 \frac{\dd x}{x^2 - 4}
= \frac14\Bigl[\ln\abs{x-2} - \ln\abs{x+2}\Bigr]_0^1
= \frac14\Bigl(\ln\frac{1}{3} - \ln 1\Bigr) = -\frac{\ln 3}{4}.
\]

भागों द्वारा ($ u' = 1$, $ v = \ln t $): $\int_1^{\eu} \ln t\,\dd t =
\bigl[t\ln t\bigr]_1^{\eu} - \int_1^{\eu} \dd t = \eu - (\eu - 1) =
1$.

भागों द्वारा ($ u' = \sin t $, $ v = t $): $\int_0^\pi t\sin t\,\dd t =
\bigl[-t\cos t\bigr]_0^\pi + \int_0^\pi \cos t\,\dd t = \pi + 0 =
\pi $.
\end{solution}

\begin{solution}{exo:b1:integration:2}
प्रतिस्थापन $ u = t^2 + 1$, $\dd u = 2t\,\dd t $:
\[
\int_0^1 \frac{t\,\dd t}{(t^2+1)^2}
= \frac12 \int_1^2 \frac{\dd u}{u^2}
= \frac12\Bigl[-\frac1u\Bigr]_1^2 = \frac14 .
\]
साथ $ u = \sin t $: $\int_0^{\pi/2} \cos^3 t\,\dd t = \int_0^{\pi/2}
(1 - \sin^2 t)\cos t\,\dd t = \bigl[\sin t - \frac{\sin^3
t}{3}\bigr]_0^{\pi/2} = 1 - \frac13 = \frac23$.
\end{solution}

\begin{solution}{exo:b1:integration:3}
$ u_n = \dfrac1n \sum_{k=1}^{n} \dfrac{1}{1 + (k/n)^2}$: एक रीमैन योग
का $ x \mapsto \frac{1}{1+x^2}$ पर $\intcc{0}{1}$, इसलिए $ u_n \to
\int_0^1 \frac{\dd x}{1+x^2} = \arctan 1 = \dfrac\pi4$.

$\ln v_n = \dfrac1n \sum_{k=1}^{n} \ln\dfrac{n+k}{n} = \dfrac 1n
\sum_{k=1}^n \ln\Bigl(1 + \dfrac kn\Bigr) \to \int_0^1 \ln(1+x)\,\dd
x = \bigl[(1+x)\ln(1+x) - x\bigr]_0^1 = 2\ln 2 - 1$. इस तरह $ v_n \to
\eu^{2\ln 2 - 1} = \dfrac 4\eu $. (पहचान की जाँच करें:
$\frac{(2n)!}{n!\,n^n} = \prod_{k=1}^{n} \frac{n+k}{n}$.)
\end{solution}

\begin{solution}{exo:b1:integration:4}
हद है $0$. होने देना $ M = \sup \abs f $ और $\varepsilon \in
\intoo{0}{1}$. पर काटें $1 - \varepsilon $:
\[
\Bigl| \int_0^1 x^n f \Bigr|
\leq \int_0^{1 - \varepsilon} x^n \abs f + \int_{1-\varepsilon}^1
x^n \abs f
\leq M\,(1-\varepsilon)^n + M\varepsilon .
\]
तब से $(1 - \varepsilon)^n \to 0$ (\cref{exo:b1:seq:3}), लिम्सअप
बायीं ओर का है $\leq M\varepsilon $ हरएक के लिए $\varepsilon $:
द \omterm{thm:b1:integration:def}{अभिन्न} आदत है $0$.
\end{solution}

\begin{solution}{exo:b1:integration:5}
$ Q(\lambda) = \int_a^b (f + \lambda g)^2 = \int f^2 + 2\lambda \int
fg + \lambda^2 \int g^2 \geq 0$ सभी के लिए $\lambda $. अगर $\int g^2 =
0$, तब $ g = 0$ (सख्त सकारात्मकता,
\cref{thm:b1:integration:props}~(4)) और असमानता है $0 \leq
0$. अन्यथा $ Q $ हर जगह एक वास्तविक द्विघात है $\geq 0$: इसका
विभेदक है $\leq 0$, यानी\ $\bigl(\int fg\bigr)^2 \leq \int
f^2 \int g^2$.

समानता यदि भेदभाव करने वाला मिट जाता है यदि $ Q(\lambda_0) = 0$ के लिए
कुछ $\lambda_0$, यानी\ $\int (f + \lambda_0 g)^2 = 0$, यानी\ (सख्त
सकारात्मकता फिर से) $ f = -\lambda_0 g $: समानता बिल्कुल तभी कायम होती है
$ f $ और $ g $ आनुपातिक हैं.
\end{solution}

\begin{solution}{exo:b1:integration:6}
होने देना $\Phi(a) = \int_a^{a+T} f $. मौलिक प्रमेय द्वारा
(\cref{thm:b1:integration:ftc}), $\Phi $ है विभेदक साथ
$\Phi'(a) = f(a + T) - f(a) = 0$: स्थिर।

के लिए $ x > 0$, लिखना $ x = nT + r $, $0 \leq r < T $ ($ n = \lfloor x/T
\rfloor $). चासल्स:
\[
\int_0^x f = n \int_0^T f + \int_{nT}^{nT + r} f,
\qquad
\Bigl| \int_{nT}^{nT+r} f \Bigr| \leq T \sup_{\intcc{0}{T}} \abs f
= C .
\]
तब $\frac 1x \int_0^x f = \frac{nT}{x}\cdot\frac 1T \int_0^T f +
O\bigl(\frac 1x\bigr)$, और $\frac{nT}{x} \to 1$: हद है
$\frac1T \int_0^T f $.
\end{solution}

\begin{solution}{exo:b1:integration:7}
होने देना $ g(x) = f(x) - x $: निरंतर, साथ
\[
\int_0^1 g = \int_0^1 f - \frac12 = 0 .
\]
अगर $ g $ कभी गायब नहीं हुआ, मध्यवर्ती मूल्य प्रमेय एक को बाध्य करेगा
स्थिर चिह्न (ए निरंतर एक पर कार्य करें \omterm{prop:b1:reals:intervals}{अंतराल} दोनों संकेत ले रहा हूँ

(\cref{thm:b1:integration:props}~(4) पर लागू किया गया $ g > 0$, देना
$\int g > 0$): के साथ विरोधाभास $\int g = 0$. इसलिए $ g(c) = 0$ के लिए

\end{solution}

\begin{solution}{exo:b1:integration:8}
\begin{enumerate}
  \item भागों द्वारा $ u' = \sin t $, $ v = \sin^{n-1} t $:
        \[
        W_n = \bigl[-\cos t\sin^{n-1}t\bigr]_0^{\pi/2}
        + (n-1)\int_0^{\pi/2} \cos^2 t\,\sin^{n-2} t\,\dd t
        = (n-1)(W_{n-2} - W_n),
        \]
        इसलिए $ nW_n = (n-1)W_{n-2}$. से $ W_0 = \frac\pi2$, $ W_1 = 1$:
        \[
        W_{2p} = \frac{(2p-1)(2p-3)\cdots 1}{(2p)(2p-2)\cdots 2}\,
        \frac{\pi}{2} = \frac{(2p)!}{4^p (p!)^2}\,\frac\pi2,
        \qquad
        W_{2p+1} = \frac{(2p)(2p-2)\cdots 2}{(2p+1)(2p-1)\cdots 3}
        = \frac{4^p (p!)^2}{(2p+1)!} .
        \]
  \item पर $\intoo{0}{\frac\pi2}$, $0 < \sin t < 1$, इसलिए $\sin^{n+1} <
        \sin^n $ और $(W_n)$ (सख्ती से) घट रहा है, सकारात्मक।
        पुनरावृत्ति के साथ सैंडविचिंग:
        \[
        \frac{n}{n+1} = \frac{W_{n+1}}{W_{n-1}}
        \leq \frac{W_{n+1}}{W_n} \leq 1
        \quad\implies\quad \frac{W_{n+1}}{W_n} \to 1 .
        \]
        अपरिवर्तनीय: $ a_n = (n+1)W_{n+1}W_n $ संतुष्ट $ a_n = a_{n-1}$
        पुनरावृत्ति द्वारा $(n+1)W_{n+1} = nW_{n-1}$, इसलिए $ a_n = a_0 =
        1 \cdot W_1 W_0 = \frac\pi2$. तब
        \[
        n W_n^2 \sim (n+1) W_{n+1} W_n = \frac\pi2
        \quad\implies\quad
        W_n \sim \sqrt{\frac{\pi}{2n}} .
        \]
\end{enumerate}
\end{solution}

\begin{solution}{exo:b1:integration:9}
\begin{enumerate}
  \item पर $\intoo{0}{\pi}$: $ x > 0$, $ a - bx = b(\frac ab - x) =
        b(\pi - x) > 0$ और $\sin x > 0$, तो इंटीग्रैंड है $> 0$
        और $ I_n > 0$ (सख्त सकारात्मकता). बंधा हुआ: चालू
        $\intcc{0}{\pi}$, $ x \leq \pi $ और $ a - bx \leq a $, इसलिए $ P
        \leq \frac{\pi^n a^n}{n!}$ और $ I_n \leq
        \pi\,\frac{(\pi a)^n}{n!}$, जिसकी प्रवृत्ति होती है $0$ (द
        फैक्टोरियल ज्यामितीय शब्द को मात देता है: यह सामान्य शब्द है
        अभिसारी घातीय श्रृंखला का, cf.\
        \cref{ex:b1:seq:e}); विशेष रूप से $ I_n < 1$ बड़े के लिए $ n $.
  \item बढ़ाना $ x^n(a - bx)^n = \sum_{j=0}^{n} \binom nj a^{n-j}
        (-b)^j x^{n+j}$: इसलिए $ P = \frac{1}{n!}\sum_j c_j x^{n+j}$
        पूर्णांक के साथ $ c_j $. तब $ P^{(k)}(0) = 0$ के लिए $ k < n $
        (मूल्यांकन) और, के लिए $ n \leq k \leq 2n $, $ P^{(k)}(0) =
        \frac{k!}{n!} c_{k-n}$, तब से एक पूर्णांक $ n! \mid k!$.
        इसके अतिरिक्त $ P(\pi - x) = P(x)$ (विकल्प: $\pi - x $ अदला-बदली
        कारक, का उपयोग करना $ a - b(\pi - x) = bx $), इसलिए $ P^{(k)}(\pi)
        = \pm P^{(k)}(0)$: पूर्णांक भी।
  \item साथ $ Q = P - P'' + P^{(4)} - \dots $ (परिमित: $ P $ की डिग्री है
        $2n $): $ Q + Q'' = P $, और
        \[
        \bigl(Q'\sin x - Q\cos x\bigr)' = (Q + Q'')\sin x = P\sin x .
        \]
        इस तरह $ I_n = \bigl[Q'\sin x - Q\cos x\bigr]_0^{\pi} = Q(\pi)
        + Q(0)$, मूल्यों का योग $ P^{(2k)}$ पर $0$ और $\pi $: एक
        (2) से पूर्णांक.
  \item के लिए $ n $ बड़ा, $ I_n $ के साथ एक पूर्णांक है $0 < I_n < 1$:
        असंभव। कल्पना $\pi = \frac ab $ विफल रहता है: $\pi $ है
        तर्कहीन.
\end{enumerate}
\end{solution}

\begin{solution}{exo:b1:integration:10}
$ C^1$ मामला: भागों द्वारा,
\[
\int_a^b f(t)\sin\lambda t\,\dd t
= \Bigl[-f(t)\frac{\cos\lambda t}{\lambda}\Bigr]_a^b
+ \frac{1}{\lambda}\int_a^b f'(t)\cos\lambda t\,\dd t,
\]
द्वारा निरपेक्ष मान में परिबद्ध $\frac{2\sup\abs f + (b - a)\sup\abs{f'}}
{\lambda} \to 0$.

निरंतर मामला: चलो $\varepsilon > 0$ और एक चुनें \omterm{def:b1:integration:step}{समारोह की ओर कदम बढ़ाएं}
$\varphi $ साथ $\abs{f - \varphi} \leq \varepsilon $
(\cref{thm:b1:integration:approx} प्रदान $\varphi \leq f \leq
\psi $ \omterm{prop:b1:reals:intervals}{अंतराल} के साथ $\leq\varepsilon $; लेना $\varphi $). तब
\[
\Bigl|\int_a^b f\sin\lambda t\Bigr|
\leq \int_a^b \abs{f - \varphi}
+ \Bigl|\int_a^b \varphi \sin\lambda t\Bigr|
\leq (b-a)\varepsilon
+ \sum_i \abs{c_i}\,\Bigl|\int_{x_{i-1}}^{x_i} \sin\lambda t\,\dd
t\Bigr| ,
\]
और प्रत्येक $\bigl|\int \sin \lambda t\bigr| = \bigl|\frac{\cos\lambda
x_{i-1} - \cos\lambda x_i}{\lambda}\bigr| \leq \frac{2}{\lambda}$:
दूसरा पद प्रवृत्त होता है $0$. इसलिए लिमसुप है $\leq
(b-a)\varepsilon $ हरएक के लिए $\varepsilon $: हद है $0$.
\end{solution}

\begin{solution}{exo:b1:integration:11}
होने देना $ m = \min f $ और $ M = \max f $, चरम मूल्य द्वारा प्राप्त किया गया
प्रमेय. तब से $ g \geq 0$: $ m\,g \leq fg \leq M\,g $, तो द्वारा
एकरसता
\[
m \int_a^b g \;\leq\; \int_a^b fg \;\leq\; M \int_a^b g .
\]
अगर $\int_a^b g = 0$: सख्त सकारात्मकता
(\cref{thm:b1:integration:props}~(4)) बल $ g \equiv 0$, दोनों
पक्ष गायब हो जाते हैं, और कोई भी $ c $ काम करता है. अन्यथा $ t = \frac{\int
fg}{\int g}$ में निहित है $\intcc{m}{M} = f(\intcc{a}{b})$
(\cref{thm:b1:continuity:evt,thm:b1:continuity:ivt}), इसलिए $ t =
f(c)$ कुछ के लिए $ c $.

साइन मायने रखता है: चालू $\intcc{-1}{1}$ साथ $ f(t) = g(t) = t $: $\int
fg = \int_{-1}^1 t^2 = \frac23$, जबकि $ f(c)\int_{-1}^1 t\,\dd t
= 0$ हरएक के लिए $ c $.
\end{solution}

\begin{solution}{exo:b1:integration:12}
अगर $ f \equiv 0$ दावा निरर्थक है (प्रत्येक बिंदु शून्य है)। तो
मान लीजिए $ f \not\equiv 0$ और मान लीजिए कि यह अधिकतम है $ n $ विशिष्ट
शून्य में $\intoo{a}{b}$; होने देना $ z_1 < \dots < z_m $ ($ m \leq n $) हो
वो शून्य कहाँ $ f $ \emph{संकेत बदलता है} (संभवतः कोई नहीं). \omterm{def:b1:logic:sets}{तय करना}
$ P(t) = \prod_{i=1}^{m}(t - z_i)$ (खाली उत्पाद $= 1$), का
डिग्री $ m \leq n $. प्रत्येक उपअंतराल पर द्वारा कटौती की जाती है $ z_i $, दोनों
$ f $ और $ P $ स्थिर चिह्न होता है, और पार करते समय दोनों फ़्लिप चिह्न होते हैं
कुछ $ z_i $: उत्पाद $ fP $ सभी पर एक स्थिर चिह्न है
$\intoo{a}{b}$. प्राणी निरंतर, समान रूप से शून्य नहीं, का
निरंतर संकेत, यह है $\bigl|\int_a^b fP\bigr| > 0$ (सख्त
सकारात्मकता लागू होती है $\abs{fP}$). लेकिन $\int f P $ एक रैखिक है
क्षणों का संयोजन $\int f\,t^k $, $ k \leq n $, सभी शून्य:
विरोधाभास. इस तरह $ f $ कम से कम है $ n + 1$ में विशिष्ट शून्य
$\intoo{a}{b}$.
\end{solution}

\begin{solution}{pb:b1:integration:1}
\textbf{1.} होने देना $ N = \lceil 2c \rceil $. के लिए $ n \geq N $:
$\frac{c^{n+1}/(n+1)!}{c^n/n!} = \frac{c}{n+1} \leq \frac12$, इसलिए
$0 < \frac{c^n}{n!} \leq \frac{c^N}{N!}\,2^{-(n - N)} \to 0$:
निचोड़ना।

\textbf{2.} हल करना $ k $; प्रेरण पर $ m $. के लिए $ m = 0$: $\int_0^1
x^k = \frac{1}{k+1} = \frac{k!\,0!}{(k+1)!}$. चरण दर चरण, भागों द्वारा
($ u = (1-x)^m $, $ v' = x^k $):
\[
\int_0^1 x^k (1-x)^m \dd x
= \Bigl[\frac{x^{k+1}}{k+1}(1-x)^m\Bigr]_0^1
+ \frac{m}{k+1}\int_0^1 x^{k+1}(1-x)^{m-1}\dd x
= \frac{m}{k+1}\cdot\frac{(k+1)!\,(m-1)!}{(k+m+1)!} ,
\]
जो है $\frac{k!\,m!}{(k+m+1)!}$.

\textbf{3.} $ k = m = n $: $\int_0^1 (x(1-x))^n =
\frac{(n!)^2}{(2n+1)!} = \frac{1}{(2n+1)\binom{2n}{n}}$. तब से
$ x(1-x) \leq \frac14$ पर $\intcc{0}{1}$, द \omterm{thm:b1:integration:def}{अभिन्न} है $\leq
4^{-n}$, कहाँ से $\binom{2n}{n} \geq \frac{4^n}{2n+1}$ ---
के साथ संगत $\binom{2n}{n}^{1/n} \to 4$
(\cref{pb:b1:seq:1}).

\textbf{4.} इंटीग्रैंड है निरंतर, $\geq 0$, और सकारात्मक
पर $\intoo{0}{1}$, इसलिए समान रूप से शून्य नहीं: यह \omterm{thm:b1:integration:def}{अभिन्न} है
$> 0$ (\cref{thm:b1:integration:props}~(4)). \omterm{def:b1:reals:bounds}{ऊपरी सीमा}:
$(x(1-x))^n \leq 4^{-n}$ और $ g \leq \sup\abs g $, फिर
एकरसता.

\textbf{5.} $ A_0 = \eu - 1$; $ A_1 = [x\eu^x]_0^1 - \int_0^1
\eu^x = \eu - (\eu - 1) = 1$. भागों द्वारा: $ A_n = [x^n \eu^x]_0^1 -
n\int_0^1 x^{n-1}\eu^x = \eu - n A_{n-1}$. सीमाएँ: एकीकृत
सकारात्मक है, इसलिए $ A_n > 0$; और $\eu^x \leq \eu $ देता है $ A_n \leq
\eu\int_0^1 x^n = \frac{\eu}{n+1}$.

\textbf{6.} $ A_0 = -1 + 1\cdot\eu $. अगर $ A_{n-1} = \alpha_{n-1} +
\beta_{n-1}\eu $ पूर्णांक प्रविष्टियों के साथ, फिर
\[
A_n = \eu - n\alpha_{n-1} - n\beta_{n-1}\eu
= \underbrace{(-n\,\alpha_{n-1})}_{\alpha_n}
+ \underbrace{(1 - n\,\beta_{n-1})}_{\beta_n}\,\eu ,
\]
दोनों पूर्णांक.

\textbf{7.} अगर $\eu = \frac pq $: $ q A_n = q\alpha_n + p\beta_n
\in \Z $, और $0 < qA_n \leq \frac{q\eu}{n+1} < 1$ के लिए $ n $ बड़ा:
एक पूर्णांक के बीच सख्ती से $0$ और $1$ --- असंभव। इसलिए $\eu
\notin \Q $. में \cref{exo:b1:seq:9} फंसा हुआ पूर्णांक था
$ q!\,\frac pq - q!\,a_q $; यह रहा $ qA_n $: वही जाल,
\omterm{thm:b1:integration:def}{अभिन्न} ईंधन।

\textbf{8.} $ n = 0$: $ R_0 = \int_0^1 \eu^t = \eu - 1$, इसलिए $\eu =
1 + R_0$. भागों द्वारा ($ u = \eu^t $, $ v = -\frac{(1-t)^{n+1}}{n+1}$):
\[
R_n = \frac{1}{n!}\Bigl(\Bigl[-\frac{(1-t)^{n+1}}{n+1}
\eu^t\Bigr]_0^1 + \frac{1}{n+1}\int_0^1 (1-t)^{n+1}\eu^t\Bigr)
= \frac{1}{(n+1)!} + R_{n+1} ,
\]
तो सूत्र से प्रचारित होता है $ n $ को $ n + 1$. सीमाएँ: $1 \leq
\eu^t \leq \eu $ पर $\intcc{0}{1}$ और $\int_0^1 (1-t)^n =
\frac{1}{n+1}$ देना $\frac{1}{(n+1)!} \leq R_n \leq
\frac{\eu}{(n+1)!}$.

\textbf{9.} $\sum_{k=0}^{9} \frac{1}{k!} = \frac{986410}{362880}
= 2.71828152\dots $, और
\[
\frac{1}{10!} = 2.76\cdot10^{-7} \leq R_9 \leq
\frac{2.75}{10!} = 7.58\cdot10^{-7} ,
\]
इसलिए $2.7182818 \leq \eu \leq 2.7182823$: प्रमाण सहित, $\eu =
2.718282$ छह दशमलव तक (सही मान $2.7182818\dots $).

\textbf{10.} $ f(1 - x) = \frac{(1-x)^n x^n}{n!} = f(x)$. पर
$\intoo{0}{1}$: $0 < x(1-x) \leq \frac14$, इसलिए $0 < f \leq
\frac{1}{4^n\,n!}$.

\textbf{11.} $ x^n(1-x)^n = \sum_{j=0}^{n} (-1)^j\binom
nj\,x^{n+j}$, इसलिए $ f = \frac{1}{n!}\sum_j c_j\,x^{n+j}$ साथ
$ c_j \in \Z $. इस तरह $ f^{(k)}(0) = 0$ के लिए $ k < n $ या $ k > 2n $,
और के लिए $ n \leq k \leq 2n $: $ f^{(k)}(0) = \frac{k!}{n!}\,
c_{k-n}$, एक पूर्णांक क्योंकि $ n! \mid k!$. समरूपता देता है
$ f^{(k)}(1) = (-1)^k f^{(k)}(0) \in \Z $.

\textbf{12.} $ b^n \pi^{2n-2k} = b^n\bigl(\frac
ab\bigr)^{\,n-k} = a^{\,n-k}\,b^{\,k} \in \Z $, इसलिए $ G(0) =
\sum_k (-1)^k a^{n-k} b^k f^{(2k)}(0)$ और इसी तरह $ G(1)$ हैं
प्रश्न 11 द्वारा पूर्णांक.

\textbf{13.} में $\pi^2 G + G''$, शब्द $ k $ का $\pi^2 G $
किया जाता है $\pi^{2n-2k+2}f^{(2k)}$ और शब्द $ j = k - 1$ का
$ G''$ किया जाता है $(-1)^{k-1}\pi^{2n-2k+2}f^{(2k)}$: सब कुछ
को छोड़कर रद्द करता है $ k = 0$ पहले योग में और $ j = n $ में
दूसरा, यानी
\[
G'' + \pi^2 G = b^n\bigl(\pi^{2n+2} f + (-1)^n f^{(2n+2)}\bigr)
= b^n \pi^{2n+2} f = \pi^2 a^n f
\]
($ f $ की डिग्री है $2n $, इसलिए $ f^{(2n+2)} = 0$; और $ b^n\pi^{2n} =
a^n $). तब
\[
\bigl(G'\sin\pi x - \pi G\cos\pi x\bigr)'
= (G'' + \pi^2 G)\sin \pi x = \pi^2 a^n f\sin\pi x .
\]

\textbf{14.} एकीकरण हो रहा है $\intcc{0}{1}$:
\[
\pi^2 a^n \int_0^1 f\sin(\pi x)\,\dd x
= \bigl[G'\sin\pi x - \pi G\cos\pi x\bigr]_0^1
= \pi\bigl(G(1) + G(0)\bigr) ,
\]
इसलिए $\pi a^n \int_0^1 f\sin\pi x = G(0) + G(1) \in \Z $. पर
$\intoo{0}{1}$, $ f > 0$ और $\sin\pi x > 0$: बाईं ओर है
सकारात्मक, तो $ G(0) + G(1) \geq 1$; और $\sin \leq 1$ के साथ
प्रश्न 10 इसे सीमित करता है $\frac{\pi a^n}{4^n\,n!}$.

\textbf{15.} प्रश्न 1 द्वारा (साथ $ c = \frac a4$),
$\frac{\pi a^n}{4^n n!} \to 0$: बड़े के लिए $ n $ यह है $< 1$,
विरोधाभासी $ G(0) + G(1) \geq 1$. तो कोई अंश नहीं $\frac ab $
के बराबर होती है $\pi^2$: लीजेंड्रे का प्रमेय। अगर $\pi $ तर्कसंगत थे,
$\pi^2$ भी होगा: तो $\pi \notin \Q $ --- और निहितार्थ
केवल इसी तरह चलता है ($\sqrt2$ तर्कसंगत के साथ तर्कहीन है
वर्ग), यही कारण है $\pi^2 \notin \Q $ सख्ती से मजबूत है
से \cref{exo:b1:integration:9}.

\textbf{16.} अखण्डता: प्रश्न 11-12 (अंतबिंदु)
डेरिवेटिव); लघुता: प्रश्न 10 और 1; पुल: प्रश्न
13--14 (दूरबीन डबल भागों द्वारा एकीकरण). बिना
$\frac{1}{n!}$, समापन बिंदु डेटा पूर्णांक (और भी अधिक) रहता है
आसानी से), लेकिन बाध्य हो जाता है $\frac{\pi a^n}{4^n}$, जो
की ओर प्रवृत्त होता है $0$ केवल जब $ a < 4$ --- और हर उम्मीदवार के पास है $ a =
b\,\pi^2 > 9$. फैक्टोरियल बिल्कुल वही है जो इससे आगे निकल जाता है
ज्यामितीय वृद्धि $ a^n $: कोई तथ्यात्मक नहीं, कोई प्रमेय नहीं।

\textbf{17.} के लिए $ a \leq 10$, पूर्णांक $ G(0) + G(1)$ है
सकारात्मक और अधिकतम $\pi\,(10/4)^n/n!$. पर $ n = 7$:
$2.5^7 = 610.35\dots $, तो बाध्य है $\frac{\pi \times
610.35}{5040} \approx 0.38 < 1$ (पर $ n = 6$ यह अब भी है
$1.07$): विरोधाभास सातवें चरण पर है,
स्पष्ट रूप से.

\textbf{18.} भागों द्वारा दो एकीकरण:
\[
\int_0^1 x(1-x)\sin\pi x\,\dd x
= \frac1\pi\int_0^1 (1 - 2x)\cos\pi x\,\dd x
= \frac{2}{\pi^2}\int_0^1 \sin\pi x\,\dd x
= \frac{2}{\pi^2}\cdot\frac{2}{\pi} = \frac{4}{\pi^3}
\]
(ब्रैकेट शब्द गायब हो जाते हैं: $ x(1-x)$ पर $0, 1$, और $\sin\pi x $
पर $0, 1$). प्रतीकात्मक रूप से, $ n = 1$ दूरबीन (कोई धारणा नहीं)
पर $\pi $) पढ़ता है $\pi^3\int_0^1 f_1\sin\pi x = -(f_1''(0) +
f_1''(1))$ साथ $ f_1 = x(1 - x)$, $ f_1'' = -2$: दाहिनी ओर $4$
--- दोनों गणनाएँ सहमत हैं।

\textbf{19.} $\eu^x $ हल करती है $ y' = y $ और $\sin\pi x $ हल करती है
$ y'' = -\pi^2 y $: स्थिर गुणांक वाले रैखिक समीकरण,
तो भागों द्वारा एकीकरण कर्नेल को वापस अपने पास चक्रित करता है और
सब रखता है \omterm{def:b1:topology:closure}{सीमा} अंदर डेटा $\Z + \Z\eu $ (सम्मान.\ पूर्णांक
बहुआयामी पद में $\pi^2$). वह \omterm{def:b1:topology:closure}{बंद} संपत्ति क्या है
मशीन की जरूरत. कई बिंदुओं पर अनुकूलित गुठली के साथ परिष्कृत
एक बार, वही तंत्र हर्मिट प्रमेय उत्पन्न करता है ($\eu $
\omterm{pb:b1:logic:1}{ट्रान्सेंडैंटल}, 1873) और लिंडमैन ($\pi $ पारलौकिक,
1882) --- इस खंड से परे।

\textbf{20.} \omterm{def:b1:poly:def}{बहुपद} विभाजन (या वापस गुणा करें और जांचें):
\[
x^4(1-x)^4 = (x^6 - 4x^5 + 5x^4 - 4x^2 + 4)(1 + x^2) - 4 .
\]
द्वारा विभाजित प्रदर्शित पहचान को एकीकृत करना $1 + x^2$:
\[
\int_0^1 \frac{x^4(1-x)^4}{1+x^2}\dd x
= \Bigl(\frac17 - \frac46 + 1 - \frac43 + 4\Bigr)
- 4\arctan 1
= \frac{22}{7} - \pi ,
\]
का उपयोग करते हुए $\frac17 - \frac23 + 1 - \frac43 + 4 = 3 + \frac17$ और
$\arctan 1 = \frac\pi4$.

\textbf{21.} इंटीग्रैंड है निरंतर, सकारात्मक पर
$\intoo{0}{1}$: द \omterm{thm:b1:integration:def}{अभिन्न} है $> 0$, इसलिए $\pi < \frac{22}{7}$.
इसके अतिरिक्त $\frac12 \leq \frac{1}{1+x^2} \leq 1$ पर
$\intcc{0}{1}$ और $\int_0^1 (x(1-x))^4 = \frac{(4!)^2}{9!} =
\frac{1}{630}$ (प्रश्न 3):
\[
\frac{1}{1260} \leq \frac{22}{7} - \pi \leq \frac{1}{630}
\quad\Longrightarrow\quad
3.14126 < \frac{22}{7} - \frac{1}{630} \leq \pi \leq
\frac{22}{7} - \frac{1}{1260} < 3.14207 .
\]

\textbf{22.} सापेक्ष $ x^2 + 1$: $ x^2 \equiv -1$, इसलिए $ x^{4m} =
(x^2)^{2m} \equiv 1$ और $(1 - x)^2 = 1 - 2x + x^2 \equiv -2x $,
इस तरह $(1-x)^{4m} \equiv (-2x)^{2m} = 4^m (x^2)^m \equiv
(-4)^m $: शेषफल अचर है $(-4)^m $, और
भागफल $ Q_m $ पूर्णांक गुणांक (ए से विभाजन) है \omterm{def:b1:poly:def}{राक्षसी}
पूर्णांक \omterm{def:b1:poly:def}{बहुपद}). द्वारा पहचान को विभाजित करना $1 + x^2$ और
एकीकृत करना:
\[
J_m := \int_0^1 \frac{(x(1-x))^{4m}}{1+x^2}\dd x
= s_m + (-4)^m\,\frac{\pi}{4},
\qquad s_m = \int_0^1 Q_m \in \Q .
\]
के लिए समाधान $\pi $: साथ $ r_m = (-1)^{m+1}\,4^{\,1-m} s_m \in
\Q $, $\;\abs{\pi - r_m} = 4^{\,1-m} J_m \leq
4^{\,1-m}\cdot4^{-4m} = 4^{\,1-5m}$. के लिए $ m = 1$: $ s_1 =
\frac{22}{7}$, $ r_1 = \frac{22}{7}$, अवश्यंभावी $4^{-4} =
\frac{1}{256}$ ---प्रश्न 20--21 पुनः।

\textbf{23.} $\bigl|\pi - \frac{22}{7}\bigr| = \frac{22}{7} -
\pi \approx 1.26\cdot10^{-3}$, से सोलह गुना बेहतर
आदेश-$2$ बेंचमार्क $\frac{1}{7^2} \approx 2.0\cdot10^{-2}$
डिरिचलेट द्वारा गारंटीकृत (\cref{pb:b1:derivative:1}, प्रश्न 4);
और $\bigl|\pi - \frac{355}{113}\bigr| \approx 2.7\cdot10^{-7}$
\omterm{pb:b1:diffeq:1}{धड़कता है} $\frac{1}{113^2} \approx 7.8\cdot10^{-5}$ एक कारक द्वारा
$\approx 300$. किसी भी बात से कोई विरोधाभास साबित नहीं हुआ: लिउविले
असमानताएँ \emph{निचली \omterm{def:b1:topology:closure}{सीमा}} केवल सन्निकटन त्रुटियों के लिए
\omterm{pb:b1:logic:1}{बीजगणितीय} संख्याएँ, और इसके लिए कोई बाध्यता नहीं है $\pi $ पर उपलब्ध है
यह स्तर --- $\pi $ शानदार ढंग से अनुमान लगाने के लिए स्वतंत्र है
अच्छा.

\textbf{24.} लेम्मा: मान लीजिए $ x = \frac pq $ और $0 < \abs{a_n +
b_n x} \to 0$. तब $\abs{a_n + b_n x} = \frac{\abs{q a_n + p
b_n}}{q}$, साथ $ q a_n + p b_n $ एक शून्येतर पूर्णांक (शून्येतर
क्योंकि निरपेक्ष मान है $> 0$): इसलिए $\abs{a_n + b_n x}
\geq \frac1q $ हरएक के लिए $ n $, के अभिसरण का खंडन
$0$. उदाहरण: प्रश्न 7 ($ x = \eu $, $ a_n = \alpha_n $, $ b_n =
\beta_n $); \cref{exo:b1:seq:9} ($ x = \eu $ फिर से, साथ $ a_q =
-q!\sum_{k \leq q}\frac{1}{k!}$, $ b_q = q!$); और
\cref{pb:b1:derivative:1}, प्रश्न 1, इसका ज्यामितीय रूप है।
भाग III और में \cref{exo:b1:integration:9} जाल चलता है
\emph{अंदर} विरोधाभास: तर्कसंगतता मानने से \omterm{def:b1:counting:objects}{परिवर्तन} होता है
एक पूर्णांक में एक अभिव्यक्ति, जिसे विश्लेषण तब निचोड़ता है
में $\intoo{0}{1}$ --- वही सिद्धांत, स्थानांतरित।

\textbf{25.} (i) समग्रता अंतिम बिंदु गणना में रहती है
