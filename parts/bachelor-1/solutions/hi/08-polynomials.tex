\section*{अध्याय \ref{ch:b1:poly} --- बहुपदों}

\begin{solution}{exo:b1:poly:1}
$X^5 - 1 = (X^2 + X + 1)(X^3 - X^2 + 1) + (-X - 2)$. चरण: घटाना
$ X^3 B $, तब $-X^2 B $, तब $ B $; शेष $-X - 2$ की डिग्री है
$1 < 2$. \emph{पर जांचें $ X = 1$:} $\;0 = 3 \times 1 + (-3)$.

$2X^4 + X^3 - X + 3 = (X^2 - 2)(2X^2 + X + 4) + (X + 11)$. \emph{जांचें
पर $ X = 0$:} $\;3 = (-2)(4) + 11$.
\end{solution}

\begin{solution}{exo:b1:poly:2}
$ X^2 + X + 1 = (X - j)(X - j^2)$ साथ $ j = \eu^{2\iu\pi/3}$, $ j^3 =
1$. यह विभाजित $ Q_n = X^{2n} + X^n + 1$ आईएफएफ $ j $ और $ j^2$ की जड़ें हैं
$ Q_n $; तब से $ Q_n $ वास्तविक गुणांक हैं, $ Q_n(j^2) =
\conj{Q_n(j)}$, तो शर्त बस है $ Q_n(j) = 0$. अब $ Q_n(j) =
j^{2n} + j^n + 1$ पर निर्भर करता है $ n $ आधुनिक $3$:
\begin{itemize}
  \item $ n \equiv 0$: $ Q_n(j) = 1 + 1 + 1 = 3 \neq 0$;
  \item $ n \equiv 1$: $ Q_n(j) = j^2 + j + 1 = 0$;
  \item $ n \equiv 2$: $ Q_n(j) = j^4 + j^2 + 1 = j + j^2 + 1 = 0$.
\end{itemize}
इसलिए $ X^2 + X + 1 \mid X^{2n} + X^n + 1$ बिल्कुल कब $3 \nmid n $.
\end{solution}

\begin{solution}{exo:b1:poly:3}
द्वारा \cref{prop:b1:poly:multiplicity}, $(X-1)^2 \mid P $ आईएफएफ $ P(1) =
P'(1) = 0$:
\[
P(1) = 2 + a + b = 0, \qquad P'(1) = 4 + 3a + 2b = 0 .
\]
हल करना: $ b = -a - 2$ और $4 + 3a - 2a - 4 = a = 0$, इसलिए $ a = 0$, $ b =
-2$: $ P = X^4 - 2X^2 + 1 = (X^2 - 1)^2 = (X-1)^2 (X+1)^2$, जो है
वास्तविक गुणनखंडन.
\end{solution}

\begin{solution}{exo:b1:poly:4}
$ X^3 - 1 = (X - 1)(X - j)(X - j^2)$ ऊपर $\C $ ($ j = \eu^{2\iu\pi/3}$),
और $(X - 1)(X^2 + X + 1)$ ऊपर $\R $.

$ X^4 + X^2 + 1 = (X^2 + X + 1)(X^2 - X + 1)$ ऊपर $\R $ (गुणा करो,
या ध्यान दें $ X^4 + X^2 + 1 = (X^2+1)^2 - X^2$); ऊपर $\C $, प्रत्येक द्विघात
विभाजन: जड़ें $ j, j^2$ और $-j, -j^2$, यानी\ $\eu^{\pm 2\iu\pi/3},
\eu^{\pm\iu\pi/3}$.

$ X^6 - 1 = \prod_{k=0}^{5} (X - \eu^{\iu k\pi/3})$ ऊपर $\C $, और खत्म होता है
$\R $:
\[
X^6 - 1 = (X-1)(X+1)(X^2 + X + 1)(X^2 - X + 1),
\]
संयुग्मी जोड़ियों को \omterm{def:b1:structures:group}{समूह}ीकृत करना $\eu^{\pm 2\iu\pi/3}$ और $\eu^{\pm
\iu\pi/3}$.
\end{solution}

\begin{solution}{exo:b1:poly:5}
\begin{enumerate}
  \item होने देना $ p/q $ (निम्नतम पद) का मूल बनें \omterm{def:b1:poly:def}{राक्षसी} पूर्णांक
        \omterm{def:b1:poly:def}{बहुपद} $ X^3 + \dots + c_0$: हरों को साफ़ करना
        $ P(p/q) = 0$ देता है $ p^3 = -q\,(\text{integer})$, इसलिए $ q \mid
        p^3$; सहप्रधानता बल $ q = \pm 1$: मूल एक पूर्णांक है
        $ p $, और $ p \mid c_0$ (पृथक करें $ c_0$). यहां उम्मीदवार हैं
        बाँटना $6$: परीक्षण, $ P(1) = 0$, $ P(2) = 0$, $ P(3) = 0$. इसलिए
        $ P = (X-1)(X-2)(X-3)$.
  \item विएटा: $ s_1 = 6$, $ s_2 = 11$, $ s_3 = 6$. वर्गों का योग:
        $ s_1^2 - 2s_2 = 36 - 22 = 14 = 1 + 4 + 9$, आशा के अनुसार। व्युत्क्रमों का योग:
        $\frac{s_2}{s_3} = \frac{11}{6} = 1 + \frac12 + \frac13$, आशा के अनुसार।
\end{enumerate}
\end{solution}

\begin{solution}{exo:b1:poly:6}
का प्रथम श्रेणी चरण \omterm{met:b1:arith:euclid}{यूक्लिडियन एल्गोरिथ्म}:
\[
(X + 1)(X^3 - X^2 + X - 1)
= X^4 - X^3 + X^2 - X + X^3 - X^2 + X - 1 = X^4 - 1 ,
\]
तो का विभाजन $ X^4 - 1$ द्वारा $ X^3 - X^2 + X - 1$ सटीक है
(भागफल $ X + 1$, शेष $0$), और एल्गोरिथ्म तुरंत बंद हो जाता है:
\[
\gcd(X^4 - 1,\; X^3 - X^2 + X - 1) = X^3 - X^2 + X - 1
\]
(पहले से \omterm{def:b1:poly:def}{राक्षसी}). बेज़ाउट संबंध तुच्छ है: $\gcd = 0
\cdot (X^4 - 1) + 1 \cdot (X^3 - X^2 + X - 1)$. संगति की जाँच करें
फैक्टरिंग: $ X^3 - X^2 + X - 1 = (X - 1)(X^2 + 1)$, जो वास्तव में है
के सामान्य अघुलनशील कारकों का उत्पाद $ X^4 - 1 =
(X-1)(X+1)(X^2+1)$.
\end{solution}

\begin{solution}{exo:b1:poly:7}
तब से $ P \geq 0$ पर $\R $, इसकी असली जड़ें भी हैं \omterm{def:b1:poly:derivative}{बहुलता} (ए पर
विषम की जड़ \omterm{def:b1:poly:derivative}{बहुलता}, $ P $ संकेत बदलता है)। का उपयोग करते हुए
\cref{cor:b1:poly:factorization} और युग्मित करें, लिखें
\[
P = c \prod_i (X - a_i)^{2k_i} \prod_j \bigl((X - z_j)(X - \conj
z_j)\bigr)^{n_j},
\]
साथ $ c > 0$ (व्यवहार पर $+\infty $). होने देना
\[
S = \sqrt c\, \prod_i (X - a_i)^{k_i} \prod_j (X - z_j)^{n_j}
\in \C[X],
\]
ताकि $ P = S\,\conj S $ कहाँ $\conj S $ \omterm{def:b1:complex:field}{संयुग्मित} है
गुणांक. विभाजित करना $ S = A + \iu B $ साथ $ A, B \in \R[X]$: तब
\[
P = (A + \iu B)(A - \iu B) = A^2 + B^2 .
\]
\end{solution}

\begin{solution}{exo:b1:poly:8}
लैग्रेंज (\cref{thm:b1:poly:lagrange}) नोड्स के साथ $0, 1, 2$:
\[
P = 1\cdot\frac{(X-1)(X-2)}{(0-1)(0-2)} + 3\cdot\frac{X(X-2)}{1\cdot(1-2)}
+ 2\cdot\frac{X(X-1)}{2\cdot 1}
= \frac{(X-1)(X-2)}{2} - 3X(X-2) + X(X-1).
\]
विस्तार: $\frac{X^2 - 3X + 2}{2} - 3X^2 + 6X + X^2 - X =
-\frac{3}{2}X^2 + \frac{7}{2}X + 1$.

प्रणाली: $ P = aX^2 + bX + c $ साथ $ c = 1$; $ a + b + 1 = 3$; $4a + 2b +
1 = 2$. तीसरे में से दूसरे को दोगुना घटाने पर: $2a - 1 = -4$,
इसलिए $ a = -\frac32$, $ b = \frac72$. वही \omterm{def:b1:poly:def}{बहुपद}:
$ P = -\frac32 X^2 + \frac72 X + 1$. (जाँच करना $ P(2) = -6 + 7 + 1 = 2$.)
\end{solution}

\begin{solution}{exo:b1:poly:9}
$ P = X^{2n+1} - 1$: $ P' = (2n+1)X^{2n} \geq 0$, इतना \omterm{def:b1:poly:def}{बहुपद}
फ़ंक्शन बढ़ रहा है (सख्ती से छोड़कर $0$), सीमा के साथ
$\mp\infty $: यह बिल्कुल एक बार गायब हो जाता है $\R $ (पर $ x = 1$).

होने देना $ E_n = \sum_{k=0}^{n} \frac{X^k}{k!}$. तब $ E_n' = E_{n-1} = E_n
- \frac{X^n}{n!}$. एक बहुमूल $ a $ संतुष्ट करेगा $ E_n(a) =
E_n'(a) = 0$ (\cref{prop:b1:poly:multiplicity}), इस तरह
$\frac{a^n}{n!} = E_n(a) - E_n'(a) = 0$, इसलिए $ a = 0$; लेकिन $ E_n(0) = 1
\neq 0$. कोई एकाधिक जड़ नहीं.
\end{solution}

\begin{solution}{exo:b1:poly:10}
\begin{enumerate}
  \item इंडक्शन (दो बेस केस होल्ड)। का उपयोग करते हुए $\cos(n+1)\theta +
        \cos(n-1)\theta = 2\cos\theta\cos n\theta $:
        \[
        T_{n+1}(\cos\theta) = 2\cos\theta \cos n\theta -
        \cos(n-1)\theta = \cos(n+1)\theta .
        \]
  \item $ T_n(\cos\theta) = 0$ आईएफएफ $ n\theta \equiv \frac\pi2 \pmod
        \pi $: संख्या
        \[
        x_k = \cos\Bigl(\frac{(2k+1)\pi}{2n}\Bigr),
        \qquad k = 0, 1, \dots, n-1,
        \]
        हैं $ n $ के विशिष्ट बिंदु $\intoo{-1}{1}$ (कोण झूठ बोलते हैं
        $\intoo{0}{\pi}$ कहाँ $\cos $ है \omterm{def:b1:logic:inj}{इंजेक्शन}), की सभी जड़ें
        $ T_n $; तब से $\deg T_n = n $ (पुनरावृत्ति से, साथ
        अग्रणी गुणांक $2^{n-1}$ के लिए $ n \geq 1$, प्रेरण द्वारा),
        ये हैं \emph{सभी} जड़ें, प्रत्येक सरल.
  \item के लिए $ x = \cos\theta \in \intcc{-1}{1}$: $\abs{T_n(x)} =
        \abs{\cos n\theta} \leq 1$, समानता के साथ यदि $ n\theta \equiv
        0 \pmod\pi $, यानी\ पर $ n+1$ अंक $ y_k =
        \cos\frac{k\pi}{n}$, $ k = 0, \dots, n $, कहाँ $ T_n(y_k) =
        (-1)^k $. (यह संतुलन बनाता है $2^{1-n}T_n $  \omterm{def:b1:poly:def}{राक्षसी}
        डिग्री-$ n $ \omterm{def:b1:poly:def}{बहुपद} सबसे छोटे सुपर-मानदंड पर
        $\intcc{-1}{1}$ --- इस अध्याय के सप्ताहांत में साबित हुआ
        समस्या.)
\end{enumerate}
\end{solution}

\begin{solution}{exo:b1:poly:11}
लिखना $ P = c\prod_i (X - a_i)^{m_i}$. उत्पाद नियम (तक विस्तारित)
कई कारक) देता है
\[
P' = c\sum_{i} m_i (X - a_i)^{m_i - 1} \prod_{k \neq i} (X -
a_k)^{m_k},
\]
और से विभाजित करना $ P $: $\frac{P'}{P} = \sum_i \frac{m_i}{X - a_i}$
(तर्कसंगत कार्यों के रूप में, यानी\ जड़ों से दूर)।

होने देना $ w $ की जड़ हो $ P'$. अगर $ w $ में से एक है $ a_i $, यह में निहित है
उत्तल पतवार तुच्छ। अन्यथा, पर मूल्यांकन $ w $:
\[
0 = \sum_i \frac{m_i}{w - a_i}
= \sum_i m_i\, \frac{\conj w - \conj a_i}{\abs{w - a_i}^2} .
\]
संयुग्मन: $\sum_i \lambda_i (w - a_i) = 0$ कहाँ $\lambda_i =
\frac{m_i}{\abs{w - a_i}^2} > 0$. इस तरह
\[
w = \frac{\sum_i \lambda_i a_i}{\sum_i \lambda_i} :
\]
एक उत्तल संयोजन (सकारात्मक भार का योग)। $1$ के बाद
जड़ों का सामान्यीकरण)। $ a_i $. तो की हर जड़ $ P'$ में निहित है
की जड़ों का उत्तल आवरण $ P $.
\end{solution}

\begin{solution}{exo:b1:poly:12}
के मूल्यांकन का योग करें $(1 + X)^n $ तीन घन पर की जड़ें
एकता:
\[
2^n + (1 + j)^n + (1 + j^2)^n
= \sum_{k=0}^n \binom nk\,\bigl(1 + j^k + j^{2k}\bigr)
= 3\sum_{k\,:\,3\mid k}\binom nk ,
\]
तब से $1 + j^k + j^{2k}$ के बराबर एक ज्यामितीय योग है $3$ कब $3
\mid k $ और को $\frac{j^{3k} - 1}{j^k - 1} = 0$ अन्यथा। अब $1 +
j = \frac12 + \iu\frac{\sqrt3}2 = \eu^{\iu\pi/3}$ और $1 + j^2 =
\conj{1 + j} = \eu^{-\iu\pi/3}$, इसलिए $(1+j)^n + (1+j^2)^n =
2\cos\frac{n\pi}3$ और
\[
\sum_{k\geq0}\binom n{3k} = \frac{2^n + 2\cos\frac{n\pi}3}{3} .
\]
जाँच: $ n = 3$: $\frac{8 + 2\cos\pi}3 = 2 = \binom30 + \binom33$;
$ n = 6$: $\frac{64 + 2}3 = 22 = 1 + 20 + 1$.
\end{solution}

\begin{solution}{pb:b1:poly:1}
\textbf{1.} $ T_2 = 2X^2 - 1$; $ T_3 = 2X(2X^2 - 1) - X = 4X^3 -
3X $; $ T_4 = 2X\,T_3 - T_2 = 8X^4 - 8X^2 + 1$; $ T_5 = 2X\,T_4 - T_3
= 16X^5 - 20X^3 + 5X $. पहचान $ T_3(\cos\theta) = \cos3\theta $
बिलकुल है $\cos3\theta = 4\cos^3\theta - 3\cos\theta $ से
\cref{ex:b1:complex:cos3}.

\textbf{2.} के लिए सच है $ n = 1, 2$. अगर $ T_{n-1}$, $ T_n $ डिग्री हैं
$ n-1$, $ n $ और अग्रणी गुणांक $2^{n-2}$, $2^{n-1}$, तब
$2X\,T_n $ की डिग्री है $ n+1$ और अग्रणी गुणांक $2^n $, जबकि
$ T_{n-1}$ निम्न डिग्री है: $ T_{n+1}$ की डिग्री है $ n + 1$, नेतृत्व गुणांक $2^n $. समता: यदि $ T_{n-1}$ की समता है $ n - 1$
और $ T_n $ की है कि $ n $, तब $2X\,T_n $ और $ T_{n-1}$ दोनों के पास है
की समता $ n + 1$, इसलिए ऐसा होता है $ T_{n+1}$.

\textbf{3.} अगर $ P(\cos\theta) = \cos n\theta $ सभी के लिए $\theta $,
तब $ P $ और $ T_n $ के हर बिंदु पर सहमत हैं $\intcc{-1}1$ --- एक
अनंत \omterm{def:b1:logic:sets}{तय करना} --- इसलिए $ P - T_n $ इसकी अनंत रूप से अनेक जड़ें हैं और यह है
शून्य \omterm{def:b1:poly:def}{बहुपद} (\cref{cor:b1:poly:nroots}).

\textbf{4.} के लिए $ x = \cos\theta $: $ T_m(T_n(\cos\theta)) =
T_m(\cos n\theta) = \cos(mn\theta) = T_{mn}(\cos\theta)$, और
$2T_mT_n(\cos\theta) = 2\cos m\theta\cos n\theta = \cos(m+n)\theta
+ \cos\abs{m - n}\theta $. दोनों की पहचान कायम है $\intcc{-1}1$,
इसलिए जैसे \omterm{def:b1:poly:def}{बहुपद} प्रश्न 3 के तर्क द्वारा पहचान।

\textbf{5.}  $ x_k $ हैं $ n $ विशिष्ट सरल जड़ें और
अग्रणी गुणांक है $2^{n-1}$:
\[
T_n = 2^{n-1}\prod_{k=0}^{n-1}
\Bigl(X - \cos\frac{(2k+1)\pi}{2n}\Bigr) .
\]
इंटरलेसिंग: कोण $0 < \frac{\pi}{2n} < \frac\pi n <
\frac{3\pi}{2n} < \frac{2\pi}n < \dots < \pi $ के बीच वैकल्पिक
द $ y $-कोण $\frac{k\pi}n $ और यह $ x $-कोण
$\frac{(2k+1)\pi}{2n}$; तब से $\cos $ पर सख्ती से कमी आ रही है
$\intcc0\pi $, मान उल्टे क्रम में जुड़ते हैं: $ y_n <
x_{n-1} < y_{n-1} < \dots < x_0 < y_0$. लगातार दो के बीच
एक्स्ट्रेमा ठीक एक जड़ पर बैठता है, जैसा कि चित्र में दिखाया गया है $\cos n\theta $
सुझाव देता है.

\textbf{6.} के साथ प्रेरण $2\cosh a\cosh b = \cosh(a + b) +
\cosh(a - b)$ (\cref{prop:b1:functions:hyprules}): $ T_{n+1}(\cosh
t) = 2\cosh t\cosh nt - \cosh(n-1)t = \cosh(n+1)t $. के लिए $ x \geq
1$, लिखना $ x = \cosh t $ साथ $ t \geq 0$; तब $\eu^t = x +
\sqrt{x^2 - 1}$ और $\eu^{-t} = x - \sqrt{x^2 - 1}$, इसलिए
\[
T_n(x) = \cosh(nt)
= \frac{(x + \sqrt{x^2-1})^n + (x - \sqrt{x^2-1})^n}2 .
\]
के लिए $ x > 1$ पहला पद अधिक है $\frac12(1)^n $ सख्ती से और
ज्यामितीय रूप से बढ़ता है: $ T_n(x) > 1$.

\textbf{7.} डी मोइवरे: $\cos n\theta = \Re\bigl((\cos\theta +
\iu\sin\theta)^n\bigr) = \sum_{2j \leq n}\binom n{2j}
\cos^{n-2j}\theta\,(\iu\sin\theta)^{2j}$, और $(\iu\sin\theta)^{2j}
= (-\sin^2\theta)^j = (\cos^2\theta - 1)^j $. स्थानापन्न $ x =
\cos\theta $ और प्रश्न 3 का आह्वान:
\[
T_n(x) = \sum_{0\leq 2j\leq n}\binom n{2j}x^{n-2j}(x^2 - 1)^j .
\]
के लिए $ n = 3$: $\binom30 x^3 + \binom32 x(x^2 - 1) = x^3 + 3x^3 -
3x = 4x^3 - 3x $, जैसा कि प्रश्न 1 में है।

\textbf{8.} $ T_n(1) = \cos(n\cdot0) = 1$; $ T_n(-1) = \cos(n\pi) =
(-1)^n $; $ T_n(0) = \cos\frac{n\pi}2$, जो है $0$ विषम के लिए $ n $ और
$(-1)^{n/2}$ एक जैसे के लिए $ n $.

\textbf{9.} के लिए प्रेरण $ U_n(\cos\theta) =
\frac{\sin(n+1)\theta}{\sin\theta}$: के लिए सच है $ U_0 = 1$ और $ U_1 =
2X $ ($\sin2\theta = 2\sin\theta\cos\theta $); कदम है
योग-से-उत्पाद पहचान $\sin(n+2)\theta = 2\cos\theta\,
\sin(n+1)\theta - \sin n\theta $. अब अंतर करें
$ T_n(\cos\theta) = \cos n\theta $ में $\theta $:
$-\sin\theta\,T_n'(\cos\theta) = -n\sin n\theta $, अभीतक के लिए तो $\theta
\notin \pi\Z $:
\[
T_n'(\cos\theta) = n\,\frac{\sin n\theta}{\sin\theta}
= n\,U_{n-1}(\cos\theta) ,
\]
और यह बहुआयामी पद $ T_n'$ और $ nU_{n-1}$, पर सहमति
$\intoo{-1}1$, बराबर हैं.

\textbf{10.} $\abs{\sin(n+1)\theta} = \abs{\sin n\theta\cos\theta
+ \cos n\theta\sin\theta} \leq \abs{\sin n\theta} +
\abs{\sin\theta}$, और प्रेरण देता है $\abs{\sin n\theta} \leq
n\abs{\sin\theta}$. इस तरह $\abs{U_{n-1}} \leq n $ पर $\intoo{-1}1$
और $\abs{T_n'} = n\abs{U_{n-1}} \leq n^2$ वहाँ; पर $\pm1$ द
बाउंड सीमाएँ लेकर विस्तारित होता है (या सीधे: $ U_{n-1}(1) = n $ से
पुनरावृत्ति, $ U_n(1) = n + 1$ प्रेरण द्वारा, और समता देता है
$ U_{n-1}(-1) = (-1)^{n-1}n $). इस प्रकार $ T_n'(1) = n^2$ और $ T_n'(-1)
= (-1)^{n-1}n^2$: बाध्य $ n^2$ अंतिम बिंदु पर प्राप्त किया जाता है।

\textbf{11.} अंतर $\sin\theta\,T_n'(\cos\theta) = n\sin
n\theta $ (प्रश्न 9) के संबंध में $\theta $:
\[
\cos\theta\,T_n'(\cos\theta) - \sin^2\theta\,T_n''(\cos\theta)
= n^2\cos n\theta = n^2\,T_n(\cos\theta) .
\]
साथ $ x = \cos\theta $ और $\sin^2\theta = 1 - x^2$: $ x\,T_n' - (1
- x^2)T_n'' = n^2T_n $ पर $\intcc{-1}1$, इसलिए हर जगह:
$(1 - x^2)y'' - xy' + n^2y = 0$ के लिए $ y = T_n $. की जाँच करें $ T_2 =
2x^2 - 1$: $(1 - x^2)(4) - x(4x) + 4(2x^2 - 1) = 4 - 4x^2 - 4x^2
+ 8x^2 - 4 = 0$.

\textbf{12.} $\widetilde T_n $ है \omterm{def:b1:poly:def}{राक्षसी} (प्रश्न 2) और
$\abs{\widetilde T_n} = 2^{1-n}\abs{T_n} \leq 2^{1-n}$ पर
$\intcc{-1}1$, साथ $\widetilde T_n(y_k) = (-1)^k2^{1-n}$ पर
$ n + 1$ अंक $ y_k $ (\cref{exo:b1:poly:10}): आदर्श बिल्कुल वैसा ही है
$2^{1-n}$, वैकल्पिक संकेतों के साथ प्राप्त किया गया।

\textbf{13.} $\widetilde T_n $ और $ P $ दोनों \omterm{def:b1:poly:def}{राक्षसी} डिग्री का
$ n $, इसलिए प्रमुख शर्तें रद्द करें: $\deg D \leq n - 1$. पर $ y_k $:
$ D(y_k) = (-1)^k2^{1-n} - P(y_k)$, और $\abs{P(y_k)} \leq \norm
P_\infty < 2^{1-n}$ का संकेत मजबूर करता है $ D(y_k)$ का होना
$(-1)^k2^{1-n}$, कठोरता से।

\textbf{14.} $ D $ के बीच चिन्ह बदलता है $ y_{k+1}$ और $ y_k $ प्रत्येक के लिए
$ k = 0, \dots, n-1$: मध्यवर्ती मूल्य संपत्ति द्वारा, $ D $ एक है
इनमें से प्रत्येक में जड़ $ n $ जोड़ीवार असंयुक्त खुले अंतराल --- $ n $
एक शून्येतर के लिए अलग जड़ें \omterm{def:b1:poly:def}{बहुपद} डिग्री का $\leq n - 1$,
असंभव। और $ D = 0$ असंभव भी है (मानदंड अलग-अलग हैं)।
विरोधाभास: नहीं \omterm{def:b1:poly:def}{राक्षसी} $ P $ डिग्री का $ n $ है $\norm P_\infty <
2^{1-n}$, जो चेबीशेव का प्रमेय है।

\textbf{15.} अब $\abs{P(y_k)} \leq 2^{1-n}$ केवल, तो $(-1)^k
D(y_k) = 2^{1-n} - (-1)^kP(y_k) \geq 2^{1-n} - \abs{P(y_k)} \geq
0$. कल्पना करना $ D(y_k) = 0$ एक इंटीरियर में $ y_k $ ($0 < k < n $): तब
$ P(y_k) = (-1)^k2^{1-n}$, इसलिए $\abs P $ अपने सर्वोच्च को प्राप्त करता है
$2^{1-n}$ आंतरिक बिंदु पर $ y_k $, कहाँ से $ P'(y_k) = 0$
(आंतरिक चरम); और $ T_n'(y_k) = nU_{n-1}(y_k) = 0$ तब से
$\sin(n\cdot\frac{k\pi}n) = 0$ --- इसलिए $\widetilde T_n'(y_k) = 0$
भी, और $ D'(y_k) = 0$: $ y_k $ की जड़ है $ D $ का \omterm{def:b1:poly:derivative}{बहुलता} कम से कम $2$.

\textbf{16.} की जड़ें गिनें $ D $ साथ \omterm{def:b1:poly:derivative}{बहुलता}. होने देना $ z $ हो
आंतरिक बिंदुओं की संख्या $ y_k $ साथ $ D(y_k) = 0$ (प्रत्येक एक डबल
जड़, प्रश्न 15 द्वारा) और $ e \in \{0, 1, 2\}$ की संख्या
समापनबिंदु ($ y_0$ या $ y_n $) साथ $ D = 0$ (प्रत्येक एक साधारण जड़ है
कम से कम)। एक अन्तराल $(y_{k+1}, y_k)$ जिसके दो समापन बिंदु दोनों हैं $ D
\neq 0$ सख्ती से वैकल्पिक संकेत रखता है, इसलिए एक इंटीरियर
जड़. प्रत्येक लुप्त आंतरिक बिंदु अधिकतम दो को खराब करता है
आसन्न अंतराल, प्रत्येक लुप्त अंतबिंदु अधिकतम एक अंतराल पर: कम से कम
$ n - 2z - e $ अंतराल अभी भी एक-एक जड़ का योगदान करते हैं, सभी अलग-अलग
से $ y $-जड़ें. कुल: कम से कम $(n - 2z - e) + 2z + e = n $
जड़ों के साथ \omterm{def:b1:poly:derivative}{बहुलता}, एक के लिए \omterm{def:b1:poly:def}{बहुपद} डिग्री का $\leq n - 1$:
इसलिए $ D = 0$ और $ P = \widetilde T_n $. मिनिमाइज़र अद्वितीय है.

\textbf{17.} एफ़िन \omterm{def:b1:logic:map}{नक्शा} $ t \mapsto x = \frac{a+b}2 +
\frac{b-a}2\,t $ एक आक्षेप है $\intcc{-1}1 \to \intcc ab $. अगर $ P $
है \omterm{def:b1:poly:def}{राक्षसी} डिग्री का $ n $, तब $ Q(t) = P(x(t))$ एक है \omterm{def:b1:poly:def}{बहुपद} में
$ t $ अग्रणी गुणांक के साथ $\bigl(\frac{b-a}2\bigr)^n $, और
$\sup_{\intcc ab}\abs P = \sup_{\intcc{-1}1}\abs Q $. \omterm{def:b1:poly:def}{राक्षसी}
\omterm{def:b1:poly:def}{बहुपद} $ Q/\bigl(\frac{b-a}2\bigr)^n $ सुपर-मानदंड है $\geq
2^{1-n}$ (प्रश्न 13-14), अतः
\[
\sup_{\intcc ab}\abs P \geq \Bigl(\frac{b-a}2\Bigr)^n 2^{1-n}
= 2\Bigl(\frac{b-a}4\Bigr)^n ,
\]
बिल्कुल समानता के साथ $ P(x) = \bigl(\frac{b-a}2\bigr)^n
\widetilde T_n\bigl(t(x)\bigr)$ (प्रश्न 16).

\textbf{18.} $\widetilde T_3 = \frac{T_3}4 = X^3 - \frac34X $;
$\widetilde T_3{}' = 3X^2 - \frac34$ पर गायब हो जाता है $\pm\frac12$.
मान: $\widetilde T_3(-1) = -\frac14$, $\widetilde
T_3(-\tfrac12) = \frac14$, $\widetilde T_3(\tfrac12) = -\frac14$,
$\widetilde T_3(1) = \frac14$: के चार वैकल्पिक चरम
निरपेक्ष मूल्य $\frac14$ --- इसलिए $\norm{\widetilde T_3}_\infty =
\frac14$, और चेबीशेव के प्रमेय संख्या द्वारा \omterm{def:b1:poly:def}{राक्षसी} घन छोटा है
सुपर-मानदंड पर $\intcc{-1}1$.

\textbf{19.} $\omega $ है \omterm{def:b1:poly:def}{राक्षसी} डिग्री का $ n + 1$, इसलिए
$\norm\omega_\infty \geq 2^{-n}$ चेबीशेव के प्रमेय द्वारा (डिग्री
$ n+1$), समानता के साथ यदि और केवल यदि $\omega = \widetilde T_{n+1}
= 2^{-n}T_{n+1}$ (प्रश्न 16), अर्थात\ यदि और केवल यदि नोड्स
हैं $ n + 1$ की जड़ें $ T_{n+1}$. चेबीशेव नोड्स के साथ त्रुटि हुई
कारक $\norm\omega_\infty $ है $2^{-n}$ --- न्यूनतम संभव.

\textbf{20.} $5 \times 36^\circ = 180^\circ $, इसलिए $ T_5(c) =
\cos180^\circ = -1$: $16c^5 - 20c^3 + 5c + 1 = 0$. परीक्षण $ x =
-1$: $-16 + 20 - 5 + 1 = 0$, और विस्तार पुष्टि करता है
\[
16x^5 - 20x^3 + 5x + 1 = (x + 1)\bigl(4x^2 - 2x - 1\bigr)^2 .
\]
तब से $ c = \cos36^\circ \neq -1$, $ c $ की जड़ है $4x^2 - 2x -
1$, जिनकी जड़ें हैं $\frac{1 \pm \sqrt5}4$; जैसा $ c > 0$,
\[
\cos36^\circ = \frac{1 + \sqrt5}4 .
\]
स्थिरता: $\cos72^\circ = T_2(c) = 2c^2 - 1 = 2\cdot\frac{3 +
\sqrt5}8 - 1 = \frac{\sqrt5 - 1}4$, में पाया गया मान
\cref{exo:b1:complex:8}.

\textbf{21.} $\sqrt{1.1^2 - 1} = \sqrt{0.21} \approx 0.458$, इसलिए
$ x + \sqrt{x^2-1} \approx 1.558$ और $(1.558)^{10} \approx 84.5$,
जबकि $(1.1 - 0.458)^{10} \approx 0.01$: $ T_{10}(1.1) \approx
\frac{84.5 + 0.01}2 \approx 42$. ए \omterm{def:b1:poly:def}{बहुपद} में फंस गया
$\intcc{-1}1$ अंतराल पहले ही बढ़ चुका है $40$ एक
इसके किनारे से दसवां परे: एक खंड पर सीमाबद्धता कुछ नहीं कहती है
इसके बाहर इंच.

\textbf{22.} प्रश्न 7 के सूत्र में $ T_p $, शब्द $ j = 0$
है $ X^p $; हर दूसरा पद वहन करता है $\binom p{2j}$ साथ $0 < 2j <
p $ (टिप्पणी $2j \neq p $ तब से $ p $ विषम है), जो विभाज्य है $ p $
के प्रमाण के प्रथम चरण द्वारा \cref{thm:b1:arith:fermat}.
इसलिए प्रत्येक गुणांक $ T_p - X^p $ का गुणज है $ p $.
जाँच: $ T_3 - X^3 = 3X^3 - 3X = 3(X^3 - X)$; $ T_5 - X^5 = 15X^5
- 20X^3 + 5X = 5(3X^5 - 4X^3 + X)$.

\textbf{23.} प्रश्न 17 द्वारा $\intcc ab = \intcc01$ और $ n =
2$: न्यूनतम विचलन $2\bigl(\frac14\bigr)^2 = \frac18$,
द्वारा प्राप्त किया गया $\bigl(\frac12\bigr)^2\widetilde T_2(2x - 1) =
\frac14\bigl((2x-1)^2 - \frac12\bigr) = x^2 - x + \frac18$.
\omterm{def:b1:poly:def}{राक्षसी} द्विघात पर शून्य के निकटतम $\intcc01$ है $ x^2 - x +
\frac18$, सुपर-मानदंड के साथ $\frac18$.

\textbf{24.} (i) कठोरता --- ए \omterm{def:b1:poly:def}{बहुपद} से अधिक जड़ों के साथ
इसकी डिग्री शून्य है --- विशिष्टता सिद्धांत को संचालित करती है
(प्रश्न 3), त्रिकोणमितीय सर्वसमिकाओं का स्थानांतरण
\omterm{def:b1:poly:def}{बहुपद} सर्वसमिकाएँ (प्रश्न 4, 7, 9, 11), और दोनों
चरमता प्रमाण के मूल-गिनती तर्क (प्रश्न 14,
16). (ii) त्रिकोणमिति \cref{ch:b1:complex} (डी मोइवरे,
योग-से-उत्पाद) और अतिशयोक्तिपूर्ण कार्य का
\cref{ch:b1:functions} परिवार के पीछे हर पहचान प्रदान की;
प्रतिस्थापन $ x = \cos\theta $ पुल है. (iii) द
\omterm{def:b1:arith:divides}{भाजकत्व} $ p \mid \binom p{2j}$ से \cref{ch:b1:arith} बदल गया
में गुणांक सूत्र \omterm{def:b1:arith:congruence}{अनुरूपता} प्रश्न 22 का.

\textbf{25.} चेबीशेव का प्रमेय एक अनुकूलन को एक में बदल देता है
अनंत-आयामी परिवार (सभी) \omterm{def:b1:poly:def}{राक्षसी} \omterm{def:b1:poly:def}{बहुपद}) परिमित में
कॉम्बिनेटरिक्स: एक प्रतियोगी से बेहतर $\widetilde T_n $ होगा
उससे निम्न-डिग्री से भिन्न है \omterm{def:b1:poly:def}{बहुपद} साइन बदलने के लिए मजबूर किया गया
$ n $ समय --- अपनी डिग्री की अनुमति से एक अधिक जड़। द
इस प्रकार, समद्विबाहु पैटर्न एक जिज्ञासा नहीं है, बल्कि वास्तव में है
इष्टतमता का प्रमाण पत्र, और समानता का मामला तेज होता है
\omterm{def:b1:poly:derivative}{बहुलता} के साथ जड़-गणना। प्रतिस्थापन $ x =
\cos\theta $ अंतिम शब्द का हकदार है: यह कठोर का परिवहन करता है,
की अलग दुनिया बहुआयामी पद की आवधिक दुनिया में
त्रिकोणमिति, जहां जड़ें और चरम $ T_n $ बस हैं
का नियमित ग्रिड $\cos n\theta$. दो विश्लेषण तथ्य उधार लिए गए
--- मध्यवर्ती मूल्य संपत्ति (प्रश्न 14; में सिद्ध हुआ
\cref{ch:b1:continuity}) और एक पर लुप्त होने वाला व्युत्पन्न
आंतरिक चरम (प्रश्न 15; सिद्ध हुआ \cref{ch:b1:derivative})
--- ये बिल्कुल वही उपकरण हैं जिन्हें बाद के अध्याय वापस सौंपेंगे,
लूप बंद करना.
