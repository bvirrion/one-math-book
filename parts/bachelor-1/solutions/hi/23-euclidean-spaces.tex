\begin{solution}{exo:b1:euclid:4}
$ F^\perp = \{(x,y,z) : x + y + z = 0\}$, \omterm{def:b1:vspaces:free}{आधार} $\bigl((1,-1,0),
(1,0,-1)\bigr)$. साथ $ e = \frac{1}{\sqrt3}(1,1,1)$: $ p_F(x) =

\operatorname{Mat}(p_F) = \frac13

\begin{solution}{exo:b1:euclid:5}
होने देना $ p = \alpha(1,0,1) + \beta(0,1,1) = (\alpha, \beta, \alpha +
\beta)$. की रूढ़िवादिता $ x - p $ जेनरेटर के लिए:
\[
\langle x - p, (1,0,1)\rangle = (1 - \alpha) + (4 - \alpha - \beta)
= 0,
\qquad
\langle x - p, (0,1,1)\rangle = (1 - \beta) + (4 - \alpha - \beta)
= 0,
\]
यानी\ $2\alpha + \beta = 5$ और $\alpha + 2\beta = 5$: $\alpha =
\beta = \frac53$. इसलिए $ p_F(x) = \bigl(\frac53, \frac53,
\frac{10}{3}\bigr)$ और
\[
d(x, F) = \norm{x - p} =
\norm{\bigl(-\tfrac23, -\tfrac23, \tfrac23\bigr)} =
\frac{2}{\sqrt 3} .
\]
ग्राम--श्मिट अनावश्यक है क्योंकि इसकी परिभाषित संपत्ति है
अनुमान --- $ x - p \perp F $ --- स्वयं एक है \omterm{def:b1:det:system}{रैखिक प्रणाली}
(``सामान्य समीकरण'') में गुणांक पर \emph{कोई} उत्पन्न करना
परिवार.
\end{solution}

\begin{solution}{exo:b1:euclid:6}

\[
\Bigl(\int_0^1 f \cdot 1\Bigr)^2 \leq \int_0^1 f^2 \cdot \int_0^1
1^2 = \int_0^1 f^2 ,

\begin{solution}{exo:b1:euclid:7}

\end{solution}

\begin{solution}{exo:b1:euclid:9}
मात्रा है $\norm{f - (a + bX)}^2$ में $ C(\intcc{0}{1})$ साथ \omterm{thm:b1:integration:def}{अभिन्न} \omterm{def:b1:euclid:def}{अंदरूनी प्रोडक्ट}: बिल्कुल न्यूनतम \omterm{thm:b1:euclid:projection}{ओर्थोगोनल
प्रक्षेपण} $ a + bX = p_F(f) = X - \frac16$
(\cref{ex:b1:euclid:bestapprox}). न्यूनतम है
\[
\norm{f - p_F(f)}^2 = \int_0^1 \Bigl(x^2 - x + \frac16\Bigr)^{\!2}
\dd x .
\]
बढ़ाना: $\int_0^1 (x^2 - x + \frac16)^2 = \int_0^1 \bigl(x^4 - 2x^3
+ \frac{4}{3}x^2 - \frac13 x + \frac{1}{36}\bigr)\dd x = \frac15 -
\frac12 + \frac49 - \frac16 + \frac{1}{36} = \frac{1}{180}$.
तो न्यूनतम बराबर है $\dfrac{1}{180}$.
\end{solution}

\begin{solution}{exo:b1:euclid:10}
होने देना $ y \in \ker(u - \mathrm{id})$ (अर्थात्\ $ u(y) = y $) और $ x \in E $.
फिर, के संरक्षण का उपयोग करना \omterm{def:b1:euclid:def}{अंदरूनी प्रोडक्ट} ($\langle u(a),
u(b)\rangle = \langle a, b\rangle $):
\[
\langle x - u(x),\, y\rangle
= \langle x, y\rangle - \langle u(x), y\rangle
= \langle x, y\rangle - \langle u(x), u(y)\rangle
= \langle x, y\rangle - \langle x, y\rangle = 0 :
\]
के हर वेक्टर $\operatorname{im}(u - \mathrm{id})$ है ओर्थोगोनल
प्रत्येक निश्चित वेक्टर के लिए।

इसलिए $\operatorname{im}(u - \mathrm{id}) \subseteq \ker(u -
\mathrm{id})^{\perp}$, और रैंक के अनुसार--शून्यता प्लस
\cref{exo:b1:euclid:7}, दोनों का आयाम है $\dim E - \dim\ker(u -
\mathrm{id})$: वे बराबर हैं. तब $ E = \ker(u - \mathrm{id})
\oplus \ker(u - \mathrm{id})^\perp = \ker(u - \mathrm{id}) \oplus
\operatorname{im}(u - \mathrm{id})$.
\end{solution}

\begin{solution}{pb:b1:euclid:1}
\textbf{1.} $\norm{t_a(x) - t_a(y)} = \norm{x - y}$; ए \omterm{def:b1:linmaps:def}{रेखीय}
\omterm{def:b1:euclid:isometry}{आइसोमेट्री} मानदंडों को संरक्षित करता है, इसलिए दूरियां ($\norm{g(x) - g(y)} =
\norm{g(x - y)} = \norm{x-y}$); और की एक रचना
दूरी बनाए रखना एमएपीएस दूरियाँ बरकरार रखता है. प्रत्येक \omterm{def:b1:euclid:isometry}{आइसोमेट्री} है
\omterm{def:b1:logic:inj}{इंजेक्शन} (विशिष्ट बिंदु सकारात्मक दूरी पर रहते हैं) और, द्वारा
नीचे दिया गया वर्गीकरण, \omterm{def:b1:logic:inj}{द्विभाजित}; पहचान और व्युत्क्रम हैं
\omterm{def:b1:euclid:isometry}{आइसोमेट्री}: ए \omterm{def:b1:structures:group}{समूह}.

\textbf{2.} $\norm{f(x)} = \norm{f(x) - f(0)} = \norm{x - 0} =
\norm x $. ध्रुवीकरण:
\[
\langle f(x), f(y)\rangle
= \frac{\norm{f(x)}^2 + \norm{f(y)}^2 - \norm{f(x) -
f(y)}^2}{2}
= \frac{\norm x^2 + \norm y^2 - \norm{x - y}^2}{2}
= \langle x, y\rangle .
\]

\textbf{3.} प्रश्न 2 के साथ विस्तार (प्रत्येक उत्पाद) $ f $का
तर्कों के उत्पाद के बराबर):
\begin{align*}
\norm{f(x+y) - f(x) - f(y)}^2
&= \norm{x+y}^2 + \norm x^2 + \norm y^2\\
&\quad - 2\langle x+y, x\rangle - 2\langle x+y, y\rangle
+ 2\langle x, y\rangle = 0 ,
\end{align*}
जैसा कि \omterm{def:b1:vspaces:sum}{प्रत्यक्ष} जांच से पता चलता है; इसी तरह $\norm{f(\lambda x) -
\lambda f(x)}^2 = \norm{\lambda x}^2 - 2\lambda\langle \lambda
x, x\rangle + \lambda^2\norm x^2 = 0$. इसलिए $ f(x + y) = f(x) +
f(y)$ और $ f(\lambda x) = \lambda f(x)$: $ f $ है \omterm{def:b1:linmaps:def}{रेखीय}, और
मानक-संरक्षण: $ f \in O(\R^2)$.

\textbf{4.} \omterm{def:b1:logic:sets}{तय करना} $ a = f(0)$ और $ g = t_{-a}\circ f $: एक \omterm{def:b1:euclid:isometry}{आइसोमेट्री}
फिक्सिंग $0$, इसलिए ए \omterm{def:b1:linmaps:def}{रेखीय} \omterm{def:b1:euclid:isometry}{आइसोमेट्री} (प्रश्न 3), और $ f = t_a
\circ g $. विशिष्टता: $ t_a \circ g = t_{a'}\circ g'$ मूल्यांकन किया गया
पर $0$ देता है $ a = a'$, तब $ g = g'$.

\textbf{5.} $(t_a\circ g)\circ(t_{a'}\circ g') = t_{a +
g(a')}\circ(g\circ g')$, तब से $ g\,t_{a'} = t_{g(a')}\,g $ के लिए
\omterm{def:b1:linmaps:def}{रेखीय} $ g $. इतना \omterm{def:b1:linmaps:def}{रेखीय} एक रचना का हिस्सा है $ g \circ g'$,
और $\det(gg') = \det g \det g'$: \omterm{def:b1:vspaces:sum}{प्रत्यक्ष} $\circ$ \omterm{def:b1:vspaces:sum}{प्रत्यक्ष} $=$
अप्रत्यक्ष $\circ$ अप्रत्यक्ष $=$ \omterm{def:b1:vspaces:sum}{प्रत्यक्ष},
\omterm{def:b1:vspaces:sum}{प्रत्यक्ष} $\circ$ अप्रत्यक्ष $=$ अप्रत्यक्ष---का संकेत नियम $\pm1$.

\textbf{6.} $ z \mapsto \eu^{\iu\theta}z $ भेजता है $1$ को
$(\cos\theta, \sin\theta)$ और $\iu $ को
$\iu\eu^{\iu\theta} = (-\sin\theta, \cos\theta)$: के कॉलम
$ R_\theta $. और $ z \mapsto \eu^{\iu\theta}\conj z $ भेजता है $1$ को
$(\cos\theta, \sin\theta)$ और $\iu $ को $-\iu\eu^{\iu\theta} =
(\sin\theta, -\cos\theta)$: के कॉलम $ S_\theta $.

\textbf{7.} $ a = 1$: $ f = t_b $. $ a \neq 1$: निश्चित-बिंदु
समीकरण $ az_0 + b = z_0$ अद्वितीय समाधान है $ z_0 = b/(1 -
a)$, और तब
\[
f(z) - z_0 = az + b - (az_0 + b) = a(z - z_0):
\]
केन्द्रित फ्रेम में $ z_0$, $ f $ से गुणा है $ a
= \eu^{\iu\arg a}$: केंद्र का घूर्णन $ z_0$ और कोण
$\arg a $.

\textbf{8.} $ f(f(z)) = a\conj{(a\conj z + b)} + b = a\conj a\,z
+ a\conj b + b = z + v $: $ f^2 = t_v $. रूपान्तरण: $ f(z + v) =
a\conj z + a\conj v + b $ और $ a\conj v = a\conj{(a\conj b + b)}
= b + a\conj b = v $, इसलिए $ f\circ t_v = t_v\circ f $.

\emph{मामला $ v = 0$}: $ f^2 = \mathrm{id}$. किसी के लिए $ z $, द
मध्यबिंदु $ m = \frac{z + f(z)}2$ संतुष्ट करता है (आइसोमेट्रीज़ किया जा रहा है
एफ़िन, भाग I) $ f(m) = \frac{f(z) + f^2(z)}{2} = m $: निश्चित
अंक मौजूद हैं. अनुवाद द्वारा एक निश्चित बिंदु पर \omterm{def:b1:complex:field}{संयुग्मित} करना,
$ f $ एक बन जाता है \omterm{def:b1:linmaps:def}{रेखीय} अप्रत्यक्ष \omterm{def:b1:euclid:isometry}{आइसोमेट्री}, यानी\ कुछ $ S_\theta $
(\cref{thm:b1:euclid:plane}): एक पंक्ति में एक प्रतिबिंब.

\emph{मामला $ v \neq 0$}: $ r = t_{-v/2}\circ f $ अप्रत्यक्ष है और
\[
r^2 = t_{-v/2}\,f\,t_{-v/2}\,f = t_{-v/2}\,t_{-v/2}\,f^2
= t_{-v}\,t_v = \mathrm{id}
\]
(कम्यूटेशन का उपयोग करके), इसलिए $ r $ एक प्रतिबिंब है, और $ f =
t_{v/2}\circ r $. इसकी धुरी \omterm{thm:b1:matrices:conjugation}{समान}ांतर है $ v $: $ r $ के साथ आवागमन करता है
$ t_v $ (दोनों $ f $ और $ t_{v/2}$ ऐसा करो $ t_v $ एमएपीएस अक्ष (द
की निश्चित लाइन $ r $) अपने आप को, मजबूर करना $ v $ इसे निर्देशित करने के लिए. इस तरह
$ f $ एक है \omterm{pb:b1:euclid:1}{सरकना प्रतिबिंब} ग्लाइड वेक्टर के साथ $ v/2$. प्रत्येक
\omterm{def:b1:euclid:isometry}{आइसोमेट्री} अब वर्गीकृत किया गया है: अनुवाद या रोटेशन (\omterm{def:b1:vspaces:sum}{प्रत्यक्ष}),
प्रतिबिंब या \omterm{pb:b1:euclid:1}{सरकना प्रतिबिंब} (अप्रत्यक्ष)।

\textbf{9.} रोटेशन $ f(z) = az + b $, $ g(z) = a'z + b'$ साथ
$ a = \eu^{\iu\alpha}$, $ a' = \eu^{\iu\beta}$: $ g\circ f(z) =
a'az + (a'b + b')$ है \omterm{def:b1:linmaps:def}{रेखीय} गुणक
$\eu^{\iu(\alpha+\beta)}$: कोण का एक घूर्णन $\alpha + \beta $
प्रश्न 7 द्वारा, या अनुवाद कब $\eu^{\iu(\alpha+\beta)} =
1$. कुछ विचार $ r_i(z) = a_i\conj z + b_i $ साथ $ a_i =
\eu^{\iu\theta_i}$ (कोण पर अक्ष $\theta_i/2$):
\[
r_2\circ r_1(z) = a_2\conj{a_1}\,z + (a_2\conj{b_1} + b_2),
\]
कोण के साथ \omterm{def:b1:vspaces:sum}{प्रत्यक्ष} $\theta_2 - \theta_1 = 2\bigl(\tfrac{\theta_2}2
- \tfrac{\theta_1}2\bigr)$: अक्षों के बीच के कोण का दोगुना;
\omterm{thm:b1:matrices:conjugation}{समान}ांतर अक्ष ($\theta_1 = \theta_2$) अनुवाद दीजिए.

\textbf{10.} केंद्र का एक घूर्णन $ c $ और कोण $\theta $ है
के माध्यम से लाइनों में दो प्रतिबिंबों का उत्पाद $ c $ कोण बनाना
$\theta/2$ (प्रश्न 9, उल्टा पढ़ें); एक अनुवाद $ t_v $ है
\omterm{thm:b1:matrices:conjugation}{समान}ांतर रेखाओं में दो प्रतिबिंबों का उत्पाद ओर्थोगोनल को
$ v $ दूरी पर $\norm v/2$; एक प्रतिबिंब एक प्रतिबिंब है; ए
\omterm{pb:b1:euclid:1}{सरकना प्रतिबिंब} अनुवाद के साथ रचा गया एक प्रतिबिंब है,
इसलिए तीन. अधिकतम: तीन.

\textbf{11.} $ a = \iu $, $ b = 1 + \iu $: $ v = a\conj b + b =
\iu(1 - \iu) + 1 + \iu = (1 + \iu) + (1 + \iu) = 2 + 2\iu \neq
0$: ए \omterm{pb:b1:euclid:1}{सरकना प्रतिबिंब} ग्लाइड वेक्टर के साथ $ v/2 = 1 + \iu $. प्रतिबिंब $ r = t_{-(1+\iu)}\circ f $ है $ r(z) = \iu\conj z $,
जिसका अक्ष कोण पर रेखा है $\frac12\arg\iu = \frac\pi4$:
रेखा $ y = x $. इसलिए $ f $ है \omterm{pb:b1:euclid:1}{सरकना प्रतिबिंब} अक्ष का $ y =
x $ और वेक्टर $(1, 1)$.

\textbf{12.} $ f(z) = \iu z $ और $ g(z) = 1 + \iu(z - 1) = \iu z
+ 1 - \iu $, इसलिए
\[
g\circ f(z) = \iu(\iu z) + 1 - \iu = -z + 1 - \iu :
\]
\omterm{def:b1:linmaps:def}{रेखीय} गुणक $-1 = \eu^{\iu\pi}$, कोण का एक घूर्णन
$\pi $ (आधा मोड़), केंद्र का $ z_0 = \frac{1 - \iu}{1 - (-1)}
= \frac{1 - \iu}{2}$.

\textbf{13.} $ r_2\circ r_1(z) = \iu\,\conj{\conj z} = \iu z $:
कोण का घूमना $\frac\pi2$ के बारे में $0$. कुल्हाड़ियाँ (असली
अक्ष, कोण $0$; रेखा $ y = x $, कोण $\frac\pi4$) पर मिलें
कोण $\frac\pi4$, और वह दोगुना है $\frac\pi2$: प्रश्न 9
पुष्टि की गई.

\textbf{14.} साथ $ f = t_a\circ g $, $ g $ \omterm{def:b1:linmaps:def}{रेखीय}, और $\sum_i
\lambda_i = 1$:
\[
f\Bigl(\sum_i \lambda_i x_i\Bigr)
= a + \sum_i \lambda_i\,g(x_i)
= \sum_i \lambda_i\bigl(a + g(x_i)\bigr)
= \sum_i \lambda_i f(x_i) .
\]

\textbf{15.} के लिए $ h \in G $, प्रश्न 14 (गुणांक) का उपयोग करते हुए
$\frac1{\abs G}$ का योग $1$):
\[
h(c) = \frac{1}{\abs G}\sum_{g \in G} h\bigl(g(x_0)\bigr)
= \frac{1}{\abs G}\sum_{g' \in G} g'(x_0) = c ,
\]
क्योंकि $ g \mapsto hg $ का आक्षेप है $ G $ खुद पर.

\textbf{16.} संयुग्म $ t_{-c}\,g\,t_{c}$ ($ g \in G $) रूप
ए \omterm{def:b1:structures:group}{समूह} \omterm{def:b1:euclid:isometry}{आइसोमेट्री} को ठीक करना $0$, इसलिए का \emph{\omterm{def:b1:linmaps:def}{रेखीय}}
आइसोमेट्रीज़ (प्रश्न 3): एक परिमित \omterm{def:b1:structures:subgroup}{उपसमूह} का $ O(\R^2)$. यदि कुछ $ s \in G $ अप्रत्यक्ष है, \omterm{def:b1:logic:map}{नक्शा} $ g \mapsto sg $ भेजता है $ G^+$
\omterm{def:b1:logic:inj}{इंजेक्शन} से में $ G\setminus G^+$ और $ h \mapsto s^{-1}h $ इसका है
उलटा (प्रश्न 5 का संकेत नियम): $\abs G = 2\,\abs{G^+}$;
अन्यथा $ G = G^+$.

\textbf{17.} अगर $ G^{+} = \{\mathrm{id}\}$, यह है $ C_1$.
अन्यथा इसके तत्वों को इस प्रकार लिखें $ R_\theta $, $\theta \in
\intco{0}{2\pi}$, और चलो $\theta_0$ सबसे छोटा सकारात्मक बनें
घटित होने वाला कोण. के लिए $ R_\theta \in G^{+}$, यूक्लिडियन डिवीजन
$\theta = k\theta_0 + \rho $ साथ $0 \leq \rho < \theta_0$ देता है
$ R_\rho = R_\theta R_{\theta_0}^{-k} \in G^{+}$, इसलिए $\rho = 0$
न्यूनतमता द्वारा: $ G^{+} = \langle R_{\theta_0}\rangle $. डिवाइडिंग
$2\pi $ द्वारा $\theta_0$ वैसे ही दिखाता है $2\pi = n\theta_0$:
$ G^{+} \cong C_n $, कोण के घूमने से उत्पन्न होता है
$\frac{2\pi}n $.

\textbf{18.} $ G = G^{+} \cup sG^{+}$ की गिनती से
प्रश्न 16. साथ में $ s(z) = a\conj z $ और $ r(z) = \omega z $
(\omterm{def:b1:linmaps:forms}{रैखिक रूप}, प्रश्न 16 के बाद; $\abs a = \abs\omega = 1$):
\[
s r s(z) = a\,\conj{\omega\,a\conj z}
= a\conj\omega\conj a\,z = \conj\omega\,z = r^{-1}(z) .
\]
प्रत्येक तत्व $ sr^k $ का $ sG^{+}$ अप्रत्यक्ष है और संतुष्ट करता है
$(sr^k)^2 = (sr^ks)\,r^k = r^{-k}r^k = \mathrm{id}$: एक
अप्रत्यक्ष अप्रत्यक्ष \omterm{def:b1:euclid:isometry}{आइसोमेट्री} फिक्सिंग $ c $, यानी\ एक प्रतिबिंब. इसलिए
$ G $ के होते हैं $ n $ घूर्णन और $ n $ के साथ प्रतिबिंब
रिश्ते $ r^n = s^2 = \mathrm{id}$, $ srs = r^{-1}$: द
\omterm{pb:b1:euclid:1}{डायहेड्रल समूह} $ D_n $.

\textbf{19.} संयोजन: एक परिमित \omterm{def:b1:structures:group}{समूह} समतल सममिति का
एक बिंदु तय करता है $ c $ (प्रश्न 15), एक परिमित तक कम हो जाता है \omterm{def:b1:structures:subgroup}{उपसमूह}
का $ O(2)$ (प्रश्न 16), और है $ C_n $ यदि इसमें केवल शामिल है
घूर्णन (प्रश्न 17), $ D_n $ अन्यथा (प्रश्न 18):
\emph{लियोनार्डो का प्रमेय}.

\textbf{20.} एक अनुवाद $ t_v \neq \mathrm{id}$ शक्तियां हैं
$ t_{kv}$, सभी अलग ($ kv $ के लिए जोड़ीवार अलग $ v \neq 0$):
अनंत क्रम. ए \omterm{pb:b1:euclid:1}{सरकना प्रतिबिंब} $ f $ है $ f^2 = t_v $ साथ $ v
\neq 0$: पुनः अनंत क्रम. कोई भी परिमित में फिट नहीं बैठता \omterm{def:b1:structures:group}{समूह}
--- लगातार प्रश्न 15--19 के साथ, जो केवल उत्पन्न हुआ
घूर्णन और प्रतिबिंब.

\textbf{21.}  $2n $ एमएपीएस शीर्ष को सुरक्षित रखें \omterm{def:b1:logic:sets}{तय करना}: $\omega^k
\omega^j = \omega^{k+j}$ और $\omega^k\conj{\omega^j} =
\omega^{k-j}$; सममिति होने के कारण, वे बहुभुज को सुरक्षित रखते हैं
(खंड खंडों में जाते हैं)। इसके विपरीत, एक \omterm{def:b1:linmaps:projection}{समरूपता} संरक्षित करती है
बैरीसेंटर $0$ शीर्षों का (प्रश्न 14), अत: है \omterm{def:b1:linmaps:def}{रेखीय},
और शीर्षों को क्रमपरिवर्तित करता है (वे बहुभुज के बिंदु हैं
से अधिकतम दूरी पर $0$). ए \omterm{def:b1:linmaps:def}{रेखीय} \omterm{def:b1:euclid:isometry}{आइसोमेट्री} भेज रहा हूँ
शिखर $1$ को $\omega^k $ है $ z \mapsto \omega^k z $ यदि \omterm{def:b1:vspaces:sum}{प्रत्यक्ष},
$ z \mapsto \omega^k\conj z $ यदि अप्रत्यक्ष (इसका मैट्रिक्स है
एक कॉलम और चिह्न द्वारा निर्धारित): अधिक से अधिक $2n $
\omterm{def:b1:linmaps:projection}{समरूपता}, तो बिल्कुल $ D_n $ ऊपर।

\textbf{22.} $ C_1$: एक विषमकोण त्रिभुज (कोई गैर-तुच्छ नहीं
\omterm{def:b1:linmaps:projection}{समरूपता})। $ D_1$: एक समद्विबाहु, गैर-समबाहु त्रिभुज (एक)।
प्रतिबिंब)। $ D_2$: एक गैर-वर्ग आयत (पहचान, the
केंद्र के बारे में आधा मोड़, दो अक्ष प्रतिबिंब)। $ C_3$: ए
ट्रिस्केलियन --- तीन सर्वांगसम मुड़ी हुई भुजाएँ जुड़ी हुई $120$-डिग्री
\omterm{prop:b1:reals:intervals}{अंतराल}; मोड़ हर प्रतिबिंब को मार देता है लेकिन छोड़ देता है
क्रम का घुमाव $3$.

\textbf{23.} साथ $\omega = \iu $: घूर्णन $ z \mapsto z $,
$\iu z $, $-z $, $-\iu z $ (कोण $0, \frac\pi2, \pi,
\frac{3\pi}2$), और प्रतिबिंब
\[
z \mapsto \conj z \ (\text{axis } y = 0), \quad
\iu\conj z \ (y = x), \quad
-\conj z \ (x = 0), \quad
-\iu\conj z \ (y = -x):
\]
वर्ग के दो विकर्ण और दो माध्यिकाएँ ---
\omterm{pb:b1:euclid:1}{डायहेड्रल समूह} $ D_4$, आदेश की $8$.

\textbf{24.} लिखना $ a = \eu^{\iu\theta} \neq 1$: $ f(z) = az +
c_1(1 - a)$ और $ g(z) = az + c_2(1 - a)$. तब
\[
f\circ g(z) - g\circ f(z)
= (1-a)\bigl(ac_2 + c_1 - ac_1 - c_2\bigr)
= -(1 - a)^2(c_2 - c_1) \neq 0 ,
\]
इसलिए $ f g \neq g f $. दोनों रचनाएँ हैं \omterm{def:b1:linmaps:def}{रेखीय} गुणक
$ a^2$, इसलिए $(fg)\circ(gf)^{-1}$ है \omterm{def:b1:linmaps:def}{रेखीय} गुणक $1$: यह है
द्वारा अनुवाद $ fg(z) - gf(z)$, ऊपर शून्येतर स्थिरांक।
ए \omterm{def:b1:structures:group}{समूह} युक्त $ f $ और $ g$ इसमें यह अनुवाद और शामिल है
इसकी सभी शक्तियाँ: यह अनंत है। दो घूर्णन केंद्र भी एक हैं
एक परिमित के लिए अनेक \omterm{def:b1:structures:group}{समूह} --- लियोनार्डो का ज्यामितीय हृदय
प्रमेय.

\textbf{25.} (i) प्रश्न 3-4 प्रत्येक दूरी-संरक्षण को दर्शाता है
\omterm{def:b1:logic:map}{नक्शा} से जुड़ा हुआ है ओर्थोगोनल \omterm{def:b1:linmaps:def}{रेखीय} भाग, और प्रश्न 15 पिन
हर परिमित \omterm{def:b1:structures:group}{समूह} एक निश्चित बिंदु पर: दोनों परिणाम परिवर्तित हो जाते हैं
मूल में रैखिक बीजगणित में मीट्रिक ज्यामिति। (ii) द
पूरी सूची है अनुवाद, घूर्णन, प्रतिबिंब, \omterm{pb:b1:euclid:1}{सरकना
प्रतिबिंब}; के निर्धारक \omterm{def:b1:linmaps:def}{रेखीय} भाग \omterm{def:b1:vspaces:sum}{प्रत्यक्ष} को अलग करता है
अप्रत्यक्ष से, और निश्चित बिंदुओं का अस्तित्व अलग हो जाता है
प्रत्येक समता के भीतर दो प्रकार। (iii) प्रश्न 9, दो द्वारा
प्रतिबिंब किसी भी घूर्णन या अनुवाद की रचना करते हैं, इसलिए
प्रतिबिंब समग्रता उत्पन्न करते हैं \omterm{def:b1:structures:group}{समूह} ---अधिकतम तीन पर्याप्त होंगे
किसी के लिए \omterm{def:b1:euclid:isometry}{आइसोमेट्री}. (iv) सीमित पैमाने पर केवल दो परिवार
जीवित रहें, चक्रीय और \omterm{pb:b1:euclid:1}{डायहेड्रल समूह}, यही कारण है कि रोसेट
आभूषण बिल्कुल दो प्रकार के होते हैं (दर्पण के साथ या दर्पण के बिना)।
कुल्हाड़ियाँ)। भाग II ने साबित कर दिया \emph{विमान का वर्गीकरण
सममिति}; भाग IV सिद्ध हुआ \emph{लियोनार्डो का प्रमेय}.
\end{solution}
