\begin{solution}{exo:b1:topology:1}
$\intoo{0}{1} \cup \intoo{2}{3}$: \omterm{def:b1:topology:open}{खुला} (संघ का खुले सेट), नहीं
\omterm{def:b1:topology:closed}{बंद किया हुआ} ($\frac 1n \to 0$ बाहर)।

$\intco{0}{1}$: कोई भी नहीं। नहीं \omterm{def:b1:topology:open}{खुला} (नहीं \omterm{prop:b1:reals:intervals}{अंतराल} आस-पास $0$ अंदर);
नहीं \omterm{def:b1:topology:closed}{बंद किया हुआ} ($1 - \frac1n \to 1 \notin $ \omterm{def:b1:logic:sets}{तय करना}).

$\{0\} \cup \intcc{1}{2}$: \omterm{def:b1:topology:closed}{बंद किया हुआ} (का परिमित मिलन \omterm{def:b1:topology:closure}{बंद} सेट), नहीं
\omterm{def:b1:topology:open}{खुला} (पर विफल रहता है $0$).

$\R \setminus \Z $: \omterm{def:b1:topology:open}{खुला} ($\Z $ है \omterm{def:b1:topology:closed}{बंद किया हुआ}), नहीं \omterm{def:b1:topology:closed}{बंद किया हुआ}: क्रम
$\bigl(\frac 1n\bigr)$ इसमें निहित है, लेकिन इसकी \omterm{def:b1:topology:closure}{सीमा} है $0$ से संबंधित $\Z $,
यानी\ से बच जाता है \omterm{def:b1:logic:sets}{तय करना}.

$\Q \cap \intoo{0}{1}$: कोई भी नहीं। नहीं \omterm{def:b1:topology:open}{खुला}: प्रत्येक \omterm{prop:b1:reals:intervals}{अंतराल} एक के आसपास
तर्कसंगत में तर्कहीन शामिल हैं। नहीं \omterm{def:b1:topology:closed}{बंद किया हुआ}: इसमें अनुक्रम शामिल हैं
तर्कहीन की ओर झुकाव $\frac{\sqrt 2}{2}$ (घनत्व).
\end{solution}

\begin{solution}{exo:b1:topology:2}
$ A = \intoc{0}{1} \cup \{2\}$: $\mathring A = \intoo{0}{1}$,
$\overline A = \intcc{0}{1} \cup \{2\}$, $\partial A = \{0, 1, 2\}$.

$ A = \R \setminus \Q $: $\mathring A = \emptyset $ (प्रत्येक \omterm{prop:b1:reals:intervals}{अंतराल}
तर्कसंगत शामिल हैं), $\overline A = \R $ (घनत्व का
तर्कहीन), $\partial A = \R $.

$ A = \bigl\{\frac{(-1)^n n}{n+1}\bigr\}$: सम पदों की प्रवृत्ति होती है $1$,
के लिए अजीब शर्तें $-1$, न ही किसी का है $ A $. $\mathring A =
\emptyset $ (पृथक बिंदु), $\overline A = A \cup \{-1, 1\}$,
$\partial A = \overline A $.
\end{solution}

\begin{solution}{exo:b1:topology:3}
\emph{पूरक.} $ F = \{a_1 < a_2 < \dots < a_k\}$: पूरक
का संघ है \omterm{def:b1:topology:open}{खुला} \omterm{prop:b1:reals:intervals}{अंतराल} $\intoo{-\infty}{a_1}$,
$\intoo{a_i}{a_{i+1}}$, $\intoo{a_k}{+\infty}$ --- \omterm{def:b1:topology:open}{खुला} द्वारा
\cref{prop:b1:topology:openstable}.

\emph{अनुक्रम.} होने देना $ u_n \in F $, $ u_n \to \ell $. साथ $\varepsilon
= \min\{\abs{a_i - a_j} : i \neq j\}/2 > 0$ (या कोई भी $\varepsilon $ अगर
$ F $ एक सिंगलटन है): कुछ रैंक से परे, सभी पद भीतर हैं
$\varepsilon $ का $\ell $, इसलिए भीतर $2\varepsilon $ एक दूसरे का,
जो उन्हें एक होने पर मजबूर करता है $ a_i $ उस पद से आगे; तब
$\ell = a_i \in F $.
\end{solution}

\begin{solution}{exo:b1:topology:4}
$\subseteq $: $\overline A \cup \overline B $ है \omterm{def:b1:topology:closed}{बंद किया हुआ} (परिमित मिलन)
और शामिल है $ A \cup B $, इसलिए इसमें शामिल है \emph{सबसे छोटा} \omterm{def:b1:topology:closed}{बंद किया हुआ}
सुपरसेट $\overline{A \cup B}$. $\supseteq $: $ A \subseteq A \cup B $
देता है $\overline A \subseteq \overline{A \cup B}$ (\omterm{def:b1:topology:closure}{बंद} है
मोनोटोन: के अनुवर्ती बिंदु $ A $ बड़े के अनुयायी हैं \omterm{def:b1:logic:sets}{तय करना}), और
इसी तरह के लिए $ B $.

चौराहों के लिए प्रतिउदाहरण: $ A = \intoo{0}{1}$, $ B =
\intoo{1}{2}$: $\overline{A \cap B} = \overline\emptyset = \emptyset $
लेकिन $\overline A \cap \overline B = \{1\}$.
\end{solution}

\begin{solution}{exo:b1:topology:5}
अनुक्रमिक लक्षण वर्णन का प्रयोग करें
(\cref{thm:b1:topology:seqclosed}). होने देना $ S = \{u_n\} \cup \{\ell\}$
और $(v_k)$ में एक क्रम $ S $ साथ $ v_k \to m $; दिखाओ $ m \in S $. दो
मामले. यदि कुछ मूल्य $ v \in S $ द्वारा लिया जाता है $(v_k)$ अनंत बार,
फिर एक स्थिरांक \omterm{def:b1:seq:subsequence}{परिणाम को} देता है $ m = v \in S $. अन्यथा हर
मूल्य प्रायः परिमित रूप से लिया जाता है; विशेष रूप से, प्रत्येक के लिए $ n $, शब्द
$ u_n $ बहुत बार प्रकट होता है, और $\ell $ बहुत। फिर हर एक के लिए $ N $, द
सूचकांक $ k $ साथ $ v_k \in \{u_0, \dots, u_N, \ell\}$ परिमित हैं
अनेक: शेष $ v_k $ शर्तें हैं $ u_n $ साथ $ n > N $. दिया गया
$\varepsilon > 0$, चुनना $ N $ साथ $\abs{u_n - \ell} \leq
\varepsilon $ के लिए $ n > N $: सभी लेकिन सीमित रूप से अनेक $ v_k $ संतुष्ट
$\abs{v_k - \ell} \leq \varepsilon $. इस तरह $ v_k \to \ell $, इसलिए $ m =
\ell \in S $.
\end{solution}

\begin{solution}{exo:b1:topology:6}
$ U + A = \bigcup_{a \in A} (U + a)$, और प्रत्येक अनुवाद करें $ U + a $ है
\omterm{def:b1:topology:open}{खुला} (अनुवाद कर रहा हूँ \omterm{prop:b1:reals:intervals}{अंतराल} प्रमाण पत्र)। का एक संघ खुले सेट है
\omterm{def:b1:topology:open}{खुला} (\cref{prop:b1:topology:openstable}).

\omterm{def:b1:topology:closure}{बंद} सेट: $\Z $ और $\sqrt 2\,\Z $ हैं \omterm{def:b1:topology:closed}{बंद किया हुआ} (जैसा $\alpha\Z $:
अभिसारी अनुक्रम अंततः स्थिर होते हैं, cf.\
\cref{ex:b1:topology:closed}). उनका योग $\Z + \sqrt 2\,\Z $ है \omterm{def:b1:topology:dense}{घना}
में $\R $ (\cref{exo:b1:reals:9}) लेकिन नहीं है $\R $ (यह गणनीय है, या
बस: $\frac{\sqrt 2}{2} \notin \Z + \sqrt2\Z $, अन्य $\sqrt 2$
तर्कसंगत होगा---लिखना $\frac{\sqrt2}{2} = m + n\sqrt 2$
बल $(2n - 1)\sqrt 2 = -2m $, इसलिए $\sqrt 2 \in \Q $ जब तक $ n =
\frac12$, असंभव)। ए \omterm{def:b1:topology:dense}{घना} उचित उपसमुच्चय नहीं है \omterm{def:b1:topology:closed}{बंद किया हुआ}: इसका
\omterm{def:b1:topology:closure}{बंद} है $\R \neq $ स्वयं.
\end{solution}

\begin{solution}{exo:b1:topology:7}
$\Z $: द \omterm{def:b1:topology:open}{अड़ोस-पड़ोस} $\intoo{n - \frac12}{n + \frac12}$ का $ n $
की बैठक $\Z $ केवल पर $ n $.

$ A = \{\frac1n : n \in \N^*\}$: आस-पास $\frac 1n $, द \omterm{prop:b1:reals:intervals}{अंतराल} का
त्रिज्या $\frac{1}{n} - \frac{1}{n+1} = \frac{1}{n(n+1)}$ (आधा, कहो)
इसे अपने पड़ोसियों से अलग कर देता है --- सभी बिंदुओं को अलग कर देता है। अभी तक $0 \in
\overline A \setminus A $: पृथक बिंदु अनुयायी को नहीं रोकते
बाहरी लोग

यदि सभी बिंदु $ A $ अलग-थलग हैं: कोई मतलब नहीं $ A $ है \omterm{def:b1:topology:closure}{आंतरिक भाग}, तब से
एक \omterm{def:b1:topology:closure}{आंतरिक भाग} बिंदु में एक संपूर्ण है \omterm{prop:b1:reals:intervals}{अंतराल} का $ A $-इसके आस-पास के पड़ोसी
(ए.एन \omterm{prop:b1:reals:intervals}{अंतराल} अनंत है), अलगाव का खंडन करता है। इसलिए $\mathring A
= \emptyset $.
\end{solution}

\begin{solution}{exo:b1:topology:8}
अगर $ A \subseteq \intcc{-M}{M}$, द \omterm{def:b1:topology:closure}{बंद} सेट $\intcc{-M}{M}$
रोकना $ A $, इसलिए शामिल है $\overline A $ (सबसे छोटा \omterm{def:b1:topology:closed}{बंद किया हुआ}
सुपरसेट): $\overline A $ घिरा है।

होने देना $ s = \sup A $ (परिमित). तब से $ A \subseteq \overline A $, $\sup
\overline A \geq s $. इसके विपरीत, $\overline A \subseteq
\intoc{-\infty}{s}$: आधी पंक्ति है \omterm{def:b1:topology:closed}{बंद किया हुआ} और शामिल है $ A $; तो
का प्रत्येक तत्व $\overline A $ है $\leq s $, देना $\sup\overline A
\leq s $. समानता.

$ s \in \overline A $: से $\varepsilon $-विशेषता
(\cref{prop:b1:reals:epsilon}), प्रत्येक \omterm{prop:b1:reals:intervals}{अंतराल} $\intoo{s -
\varepsilon}{s + \varepsilon}$ का एक तत्व शामिल है $ A $: $ s $ है
अनुयायी.
\end{solution}

\begin{solution}{exo:b1:topology:9}
कल्पना करना $ A $ है \omterm{def:b1:topology:open}{खुला} और \omterm{def:b1:topology:closed}{बंद किया हुआ}, साथ $ a \in A $ और $ b \in \R
\setminus A $; व्यापकता की हानि के बिना $ a < b $. \omterm{def:b1:logic:sets}{तय करना} $ B = A \cap
\intcc{a}{b}$ गैर-रिक्त है ($ a $), परिबद्ध: चलो $ s = \sup B $. द्वारा
\cref{exo:b1:topology:8}, $ s \in \overline B \subseteq \overline A =
A $ ($ A $ \omterm{def:b1:topology:closed}{बंद किया हुआ}). टिप्पणी $ s \leq b $, और तबसे $ b \notin A $: $ s < b $.
अब $ A $ है \omterm{def:b1:topology:open}{खुला}: कुछ \omterm{prop:b1:reals:intervals}{अंतराल} $\intoo{s - r}{s + r}$ में निहित है $ A $,
और हम ले सकते हैं $ r < b - s $. तब $ s + \frac{r}{2}$ से संबंधित $ A
\cap \intcc{a}{b} = B $ और अधिक है $ s $ --- विरोधाभासी $ s = \sup
B $. इसलिए ऐसी कोई जोड़ी नहीं $(a, b)$ मौजूद है: में से एक $ A $, $\R \setminus A $
खाली है.
\end{solution}

\begin{solution}{exo:b1:topology:10}
$ I_x $ का एक संघ है \omterm{def:b1:topology:open}{खुला} \omterm{prop:b1:reals:intervals}{अंतराल} सभी युक्त $ x $: यह है \omterm{def:b1:topology:open}{खुला},
और यह एक है \omterm{prop:b1:reals:intervals}{अंतराल}, उत्तल होना --- यदि $ u < z < v $ साथ $ u, v \in
I_x $, तब $ u $ और $ v $ रिहायश \omterm{def:b1:topology:open}{खुला} उपअंतराल $ J_u \ni x $, $ J_v \ni
x $ का $ U $, और $ J_u \cup J_v $ एक \omterm{prop:b1:reals:intervals}{अंतराल} (दोनों शामिल हैं $ x $)
अंदर $ U $ युक्त $ z $; इसलिए $ z \in I_x $
(\cref{prop:b1:reals:intervals}).

अगर $ I_x \cap I_y \neq \emptyset $: $ I_x \cup I_y $ फिर एक है \omterm{def:b1:topology:open}{खुला}
\omterm{prop:b1:reals:intervals}{अंतराल} (उत्तल: दो अतिव्यापी \omterm{prop:b1:reals:intervals}{अंतराल}) में निहित $ U $
युक्त $ x $ और $ y $, इसलिए $ I_x \cup I_y \subseteq I_x $ और
$\subseteq I_y $ प्रत्येक की अधिकतमता से: $ I_x = I_y $.

इसलिए $ U $ विशिष्ट का असंयुक्त मिलन है सेट $ I_x $ (प्रत्येक $ x \in
U $ अपने आप में निहित है $ I_x $). गणनीयता: प्रत्येक गैर-रिक्त \omterm{def:b1:topology:open}{खुला} \omterm{prop:b1:reals:intervals}{अंतराल}
$ I $ परिवार में एक तर्कसंगतता शामिल है $ q_I $
(\cref{thm:b1:reals:density}), और विशिष्ट असंबद्धता \omterm{prop:b1:reals:intervals}{अंतराल} प्राप्त करें
विशिष्ट तर्क: परिवार सम्मिलित करता है $\Q $, जो गणनीय है
(यह पूर्णांकों के जोड़े द्वारा अनुक्रमित है)। अत: अधिक से अधिक संख्या में
\omterm{prop:b1:reals:intervals}{अंतराल}.
\end{solution}

\begin{solution}{exo:b1:topology:11}
$\overline A = A \cup A'$. ($\supseteq $) $ A \subseteq \overline
A $ हमेशा; और यदि $ x \in A'$, प्रत्येक \omterm{def:b1:topology:open}{अड़ोस-पड़ोस} का $ x $ की बैठक $ A
\setminus \{x\} \subseteq A $, इसलिए $ x $ अनुयायी है. ($\subseteq $)
होने देना $ x \in \overline A $. अगर $ x \in A $, हो गया। अगर $ x \notin A $,
प्रत्येक \omterm{def:b1:topology:open}{अड़ोस-पड़ोस} का $ x $ की बैठक $ A = A \setminus \{x\}$: $ x \in
A'$.

फलस्वरूप $ A $ \omterm{def:b1:topology:closed}{बंद किया हुआ} $\iff $ $ A = \overline A = A \cup A'$
$\iff $ $ A' \subseteq A $.

$ A = \{\frac 1n\}$: $0$ एक संचय बिंदु है ($\frac 1n \to
0$, शर्तें $\neq 0$); प्रत्येक $\frac 1n $ पृथक है
(\cref{exo:b1:topology:7}), तो अंदर नहीं $ A'$; और एक बिंदु $ x
\notin A \cup \{0\}$ एक संपूर्ण है \omterm{prop:b1:reals:intervals}{अंतराल} टाल $ A $ (बीच में)
के दो पड़ोसी $ x $ में $ A \cup \{0\}$, या उससे आगे $1$).
इस तरह $ A' = \{0\}$.

$\Z' = \emptyset $: प्रत्येक पूर्णांक पृथक है, प्रत्येक गैर-पूर्णांक
एक है \omterm{def:b1:topology:open}{अड़ोस-पड़ोस} अंदर $\R \setminus \Z $.

$\Q' = \R $: प्रत्येक \omterm{prop:b1:reals:intervals}{अंतराल} किसी भी वास्तविक के चारों ओर \omterm{def:b1:reals:bounds}{असीम} रूप से समाहित है
कई तर्क (\cref{thm:b1:reals:density}), विशेष रूप से एक
केंद्र से भिन्न.
\end{solution}

\begin{solution}{exo:b1:topology:12}
\begin{enumerate}
  \item हरएक के लिए $ a \in A $: $\abs{x - a} \leq \abs{x - y} +
        \abs{y - a}$, इसलिए $ d_A(x) \leq \abs{x - y} + \abs{y - a}$;
        लेना \omterm{def:b1:reals:bounds}{असीम} ऊपर $ a $: $ d_A(x) \leq \abs{x - y} +
        d_A(y)$. का आदान प्रदान $ x $ और $ y $ दूसरा देता है
        असमानता: $\abs{d_A(x) - d_A(y)} \leq \abs{x - y}$.
  \item $ d_A(x) = 0$ $\iff $ हरएक के लिए $\varepsilon > 0$ वहाँ है
        $ a \in A $ साथ $\abs{x - a} < \varepsilon $ $\iff $ प्रत्येक
        \omterm{prop:b1:reals:intervals}{अंतराल} आस-पास $ x $ की बैठक $ A $ $\iff $ $ x \in \overline A $.
        अगर $ F $ है \omterm{def:b1:topology:closed}{बंद किया हुआ} और $ x \notin F = \overline F $, तब
        $ d_F(x) \neq 0$, यानी\ $ d_F(x) > 0$.
  \item अगर $ d_F(x) < \varepsilon $, \omterm{def:b1:logic:sets}{तय करना} $ r = \varepsilon -
        d_F(x) > 0$: के लिए $\abs{y - x} < r $, भाग (1) देता है
        $ d_F(y) \leq d_F(x) + \abs{x - y} < \varepsilon $: द
        \omterm{prop:b1:reals:intervals}{अंतराल} $\intoo{x - r}{x + r}$ में निहित है $ V_\varepsilon $,
        जो इसलिए है \omterm{def:b1:topology:open}{खुला}; इसमें है $ F $ तब से $ d_F = 0$
        वहाँ। अंत में $ x \in \bigcap_{\varepsilon>0}
        V_\varepsilon $ $\iff $ $ d_F(x) < \varepsilon $ सभी के लिए
        $\varepsilon $ $\iff $ $ d_F(x) = 0$ $\iff $ $ x \in \overline
        F = F $. तब से $\bigcap_{\varepsilon > 0} V_\varepsilon =
        \bigcap_{n \geq 1} V_{1/n}$, प्रत्येक \omterm{def:b1:topology:closure}{बंद} सेट एक है
        का गणनीय प्रतिच्छेदन खुले सेट.
\end{enumerate}
\end{solution}

\]
प्रेरण: यदि $ C_n $ का एक असंयुक्त संघ है $2^n $ \omterm{def:b1:topology:closed}{बंद किया हुआ} खंड
लम्बाई का $3^{-n}$, हटा रहा हूँ \omterm{def:b1:topology:open}{खुला} प्रत्येक का मध्य तीसरा
दो छोड़ देता है \omterm{def:b1:topology:closed}{बंद किया हुआ} लंबाई के खंड $3^{-n-1}$ प्रति माता-पिता:
$2^{n+1}$ खंड, जोड़ीवार असंयुक्त (विभिन्न के बच्चे
माता-पिता अलग हो गए हैं क्योंकि माता-पिता अलग हो गए थे; एक के बच्चे
माता-पिता को हटाए गए अंतर से अलग किया जाता है)।

\textbf{2.} प्रत्येक $ C_n $ इसलिए, खंडों का एक सीमित संघ है
\omterm{def:b1:topology:closed}{बंद किया हुआ}; $ C = \bigcap C_n $ का एक चौराहा है \omterm{def:b1:topology:closure}{बंद} सेट:
\omterm{def:b1:topology:closed}{बंद किया हुआ} (\cref{def:b1:topology:closed}); घिरा हुआ ($\subseteq
\intcc{0}{1}$): द्वारा कॉम्पैक्ट \cref{thm:b1:topology:compact}.
गैररिक्त: $0$ प्रत्येक के सबसे बाएं खंड में स्थित है $ C_n $. होने देना
$ a $ एक खंड का समापन बिंदु बनें $ S $ का $ C_n $. के लिए $ m \leq n $,
$ a \in C_n \subseteq C_m $. बाद के चरणों के लिए: मध्य-तीसरा
विलोपन कभी भी समापन बिंदु को नहीं हटाता है, और $ a $ यह फिर से एक समापन बिंदु है
के दो बच्चों में से एक का $ S $ (बच्चा छू रहा है $ a $); द्वारा
प्रेरण $ a \in C_m $ सभी के लिए $ m \geq n $: $ a \in C $.

\textbf{3.} की कुल लम्बाई $ C_n $: $2^n \cdot 3^{-n} =
(\frac23)^n \to 0$. दिया गया $\varepsilon > 0$, चुनना $ n $ साथ
$(\frac23)^n \leq \varepsilon $: तब $ C \subseteq C_n $, एक संघ
कुल लंबाई के बहुत से खंडों का $\leq \varepsilon $.

\textbf{4.} होने देना $ I \subseteq C $ एक हो \omterm{prop:b1:reals:intervals}{अंतराल} दो अलग के साथ
अंक. हरएक के लिए $ n $: $ I \subseteq C_n $, और $ I $, उत्तल होना,
ए के अंदर झूठ बोलना चाहिए \emph{अकेला} का खंड $ C_n $ (दो से मुलाकात
खंड बल देंगे $ I $ बीच के \omterm{prop:b1:reals:intervals}{अंतराल} का एक बिंदु समाहित करना
उन्हें, जो बाहर है $ C_n $). इसलिए की लंबाई $ I $ है $\leq
3^{-n}$ सभी के लिए $ n $: विरोधाभास। तो एकमात्र \omterm{prop:b1:reals:intervals}{अंतराल}
अंदर $ C $ खाली या सिंगलटन हैं; विशेष रूप से नहीं
$\intoo{x-r}{x+r}$ अंदर फिट बैठता है $ C $: $\mathring C = \emptyset $.
जैसा $ C $ है \omterm{def:b1:topology:closed}{बंद किया हुआ}, $\overline C = C $ खाली है \omterm{def:b1:topology:closure}{आंतरिक भाग}: $ C $ है
कहीं नहीं \omterm{def:b1:topology:dense}{घना}.

\textbf{5.} लिखना $\varphi_0(x) = \frac x3$ और $\varphi_2(x) =
\frac{2 + x}{3}$, की आत्मीय आपत्तियाँ बढ़ रही हैं $\intcc{0}{1}$
पर $\intcc{0}{\frac13}$ और $\intcc{\frac23}{1}$. दावा करना:
$ C_{n+1} = \varphi_0(C_n) \cup \varphi_2(C_n)$. के लिए $ n = 0$ यह
प्रश्न 1 है। प्रेरण: एक बढ़ता हुआ आकर्षण \omterm{def:b1:logic:map}{नक्शा} भेजता है
एक खंड के मध्य तीसरे से छवि के मध्य तीसरे तक
खंड, इसलिए मध्य तिहाई को हटाने से आवागमन होता है $\varphi_0$ और
$\varphi_2$; विलोपन चरण को लागू करना $ C_{n+1} =
\varphi_0(C_n) \cup \varphi_2(C_n)$ पैदावार $ C_{n+2} =
\varphi_0(C_{n+1}) \cup \varphi_2(C_{n+1})$. खत्म हो रहा है
$ n $: के लिए $ x \leq \frac13$, $ x \in C \iff x \in \varphi_0(C_n)$
सभी के लिए $ n $ $\iff 3x \in \bigcap C_n = C $; इसी तरह चालू
$\intcc{\frac23}{1}$; और कोई मतलब नहीं $\intoo{\frac13}{\frac23}$
में निहित है $ C_1$. इस तरह $ C = \varphi_0(C) \cup \varphi_2(C)$,
असम्बद्ध रूप से।

\textbf{6.} प्रेरण चालू $ n $; मामला $ n = 0$ हर कहता है $ x \in
\intcc{0}{1}$ एक कोड है, जो है \cref{pb:b1:reals:1}
(प्रश्न 9 के लिए $ x < 1$; $1 = (0.\overline 2)_3$). मान लीजिये
पद पर समानता $ n $. अगर $ x \in C_{n+1}$: प्रश्न 5 से, $ x =
\varphi_i(z)$ साथ $ z \in C_n $ और $ i \in \{0, 2\}$; अगर $(e_k)$
का एक कोड है $ z $ पहले के साथ $ n $ अंक $1$-फिर मुफ़्त $(i, e_1,
e_2, \dots)$ आंशिक रकम है $\frac i3 + \frac13\sum_{k\leq m}
e_k 3^{-k} \to \varphi_i(z) = x $: का एक कोड $ x $ पहले के साथ $ n +
1$ अंक $1$-मुक्त। इसके विपरीत, यदि $ x $ एक कोड है $(d_k)$ साथ
$ d_1, \dots, d_{n+1} \in \{0, 2\}$: स्थानांतरित स्ट्रिंग $(d_2,
d_3, \dots)$ कुछ मूल्य है $ z \in \intcc{0}{1}$, यह पहला है $ n $
अंक हैं $1$-मुक्त, और आंशिक-योग गणना पढ़ें
पीछे की ओर देता है $ x = \varphi_{d_1}(z)$; प्रेरण द्वारा $ z \in
C_n $, इसलिए $ x \in C_{n+1}$ प्रश्न 5 से। अंत में: एक पूर्णतः
$1$-फ्री कोड डालता है $ x $ प्रत्येक $ C_n $, इसलिए में $ C $; इसके विपरीत
यदि $ x \in C $, फिर प्रत्येक के लिए $ n $ के अधिकतम दो कोडों में से एक
$ x $ (\cref{pb:b1:reals:1}, प्रश्न 11) का पहला है $ n $ अंक
$1$-मुक्त; एक निश्चित कोड को मनमाने ढंग से बड़े पैमाने पर काम करना चाहिए $ n $
(दो कोड के बीच कबूतर का छेद), और एक कोड जिसका पहला $ n $
अंक हैं $1$- मनमाने ढंग से बड़े के लिए मुफ़्त $ n $ है $1$-मुक्त
एकमुश्त.

\textbf{7.} $0 = (0.\overline 0)_3$, $1 = (0.\overline 2)_3$,
$\frac13 = (0.0\overline{2})_3$ (अनुचित जुड़वां $(0.1)_3$),
$\frac23 = (0.2\overline{0})_3$. ज्यामितीय योग:
\[
(0.\overline{02})_3 = \sum_{j\geq1} \frac{2}{9^{\,j}}
= \frac{2/9}{1 - 1/9} = \frac14 ,
\qquad
(0.\overline{20})_3 = \sum_{j\geq1} \frac{2}{3\cdot 9^{\,j-1}}
= \frac{2/3}{1 - 1/9} = \frac34 ,
\]
दोनों $1$-मुक्त: $\frac14, \frac34 \in C $. (अनंत रकम
आंशिक रकम का संक्षिप्त \omterm{def:b1:reals:bounds}{सर्वोच्च}, जैसा कि \cref{pb:b1:reals:1}.)
के खंडों के समापन बिंदु $ C_n $ स्वरूप के हैं $ m/3^n $
(प्रेरण: बच्चों के समापन बिंदु मूल समापन बिंदु हैं या भिन्न हैं
एक से अनेक तक $3^{-n-1}$). अगर $\frac14 =
\frac{m}{3^n}$ तब $3^n = 4m $, और $4 \nmid 3^n $: असंभव।
इसलिए $\frac14 \in C $ बिना किसी समापन बिंदु के।

\textbf{8.} कल्पना करना $ x $ दो अलग थे $1$-निःशुल्क कोड. होना
हर तरह से दो कोड (\cref{pb:b1:reals:1}, प्रश्न 11, आधार
$3$) वह $ x = m/3^N \in \intoo{0}{1}$ और दो कोड हैं:
अंतिम गैर-शून्य अंक के साथ समाप्त होने वाला $ d_N \in \{1, 2\}$
के बाद $0$ एस, और इसके जुड़वां, के साथ $d_N - 1$ पद पर $ N $
के बाद $2$ एस। अगर $d_N = 1$ पहले में एक शामिल है $1$; अगर $ d_N
= 2$ जुड़वां वहन करता है $ d_N - 1 = 1$. किसी भी तरह से अधिक से अधिक एक
जोड़ी है $1$-मुक्त: विरोधाभास. तो प्रत्येक $ x \in C $ है
बिल्कुल एक $1$-मुक्त कोड (प्रश्न 6 द्वारा अस्तित्व), और
विशिष्ट $\{0,2\}$-स्ट्रिंग्स के अलग-अलग मान हैं। प्रत्येक
$\{0,2\}$-स्ट्रिंग में मान है $\intcc{0}{1}$ (आंशिक रकम $\leq
1$) सभी उपसर्गों के साथ $1$-मुक्त, इसलिए प्रत्येक में मूल्य $ C_n $,
यानी\ में $ C $: मूल्य \omterm{def:b1:logic:map}{नक्शा} से एक आक्षेप है
$\{0,2\}$-स्ट्रिंग्स पर $ C $.

\textbf{9.} होने देना $(d^{(k)})$ हो $1$- का निःशुल्क कोड $ x_k $ और सेट करें $ e_k = 2 - d^{(k)}_k \in \{0, 2\}$: ए $\{0,2\}$-स्ट्रिंग किसकी
मूल्य $ y $ में निहित है $ C $ और हैं $(e_k)$ क्योंकि यह अद्वितीय है $1$-मुक्त
कोड (प्रश्न 8). प्रत्येक के लिए $ k $ के कोड $ y $ और $ x_k $
स्थिति में भिन्नता $ k $, इसलिए $ y \neq x_k $: नहीं \omterm{def:b1:logic:map}{नक्शा} $\N^* \to C $ है
\omterm{def:b1:logic:inj}{विशेषण}. एक बेशुमार \omterm{def:b1:logic:sets}{तय करना} लंबाई का शून्य: बड़ाई में
\omterm{def:b1:counting:card}{प्रमुखता}, माप में लघुता --- एक साथ।

\textbf{10.} सघनता: अनंत की बंदता
प्रतिच्छेदन प्लस सीमाबद्धता (प्रश्न 2) --- एक संपत्ति
एक भी रोकथाम द्वारा नहीं किया गया। लंबाई शून्य: $ C \subseteq
C_n $ कुल लंबाई के साथ $(\frac23)^n $ (प्रश्न 3). कहीं नहीं
घनत्व: $ C \subseteq C_n $ बल \omterm{prop:b1:reals:intervals}{अंतराल} अंदर $ C $ होना
लंबाई $\leq 3^{-n}$ (प्रश्न 4). बेशुमारता नंबर पर सवार है
बिल्कुल रोकथाम: इसे पूर्ण प्रतिच्छेदन संरचना की आवश्यकता है,
प्रश्न 8 के आक्षेप में एन्कोड किया गया।

\textbf{11.} पलटना $ d_n $ को $2 - d_n $: नई स्ट्रिंग अभी भी है
$\{0,2\}$-स्ट्रिंग, तो इसका मूल्य $ x_n $ में निहित है $ C $; आंशिक
रैंक से अधिक रकम $ n $ बिल्कुल भिन्न $2\cdot3^{-n}$, इसलिए
$\abs{x_n - x} = 2\cdot3^{-n}$. इस प्रकार $ x_n \neq x $ और $ x_n \to
x $: का हर बिंदु $ C $ के अन्य बिन्दुओं की एक \omterm{def:b1:topology:closure}{सीमा} है $ C $ --- $ C $
है \emph{उत्तम}, बिना किसी पृथक बिंदु के।

\textbf{12.} प्रश्न 5 के शामिल होने से, के खंड
$ C_n $ बिल्कुल वही हैं $\intcc{t}{t + 3^{-n}}$ कहाँ $ t $ चलता है
लंबाई के मानों पर-$ n $ $\{0,2\}$-स्ट्रिंग्स. दिया गया $ x \in C $
कोड के साथ $(d_k)$, कटाव $ t_n $ (अंक $ d_1 \dots d_n $
तब $0$ s) इसलिए एक बायां समापन बिंदु है, और $0 \leq x - t_n
\leq 3^{-n}$: समापनबिंदु हैं \omterm{def:b1:topology:dense}{घना} में $ C $. वे का एक उपसमुच्चय बनाते हैं
$\{m/3^n : m, n\}$, ए \omterm{def:b1:logic:sets}{तय करना} पूर्णांकों के जोड़े द्वारा अनुक्रमित, इसलिए
गणनीय (जैसा कि) $\Q $ में \cref{exo:b1:topology:10}). तब से $ C $
बेशुमार है (प्रश्न 9), लेकिन गिनती के अलावा सभी कई बिंदु हैं
$ C $ समापनबिंदु नहीं हैं --- $\frac14$ (प्रश्न 7) दृश्यमान है
उस हिमशैल का सिरा.

\textbf{13.} चुनना $ n $ साथ $3^{-n} < y - x $. दोनों $ x, y \in
C_n $, और वे एक ही खंड (लंबाई) में नहीं रह सकते $3^{-n} <
y - x $): उनके खंडों के बीच हटाया गया अंतर प्रदान करता है $ z $
साथ $ x < z < y $ और $ z \notin C_n \supseteq C $. अत: कोई दो
के अंक $ C $ पूरक द्वारा अलग किए गए हैं: एकमात्र उत्तल
के उपसमुच्चय $ C $ सिंगलटन हैं --- $ C $ पूरी तरह से डिस्कनेक्ट हो गया है.

\textbf{14.} अगर $(d_k)$ है $1$- का निःशुल्क कोड $ x $, डोर
$(2 - d_k)$ फिर से एक है $\{0,2\}$-स्ट्रिंग, आंशिक योग के साथ
\[
\sum_{k=1}^{n} (2 - d_k)3^{-k} = (1 - 3^{-n}) - \sum_{k=1}^n
d_k 3^{-k} \longrightarrow 1 - x :
\]
इसलिए $1 - x \in C $. इस प्रकार $1 - C \subseteq C $, और लागू करना \omterm{def:b1:logic:map}{नक्शा}
दो बार देता है $1 - C = C $: द \omterm{pb:b1:topology:1}{कैंटर सेट} के बारे में सममित है
$\frac12$.

\textbf{15.} आंशिक रकम $ t_n \to x $ और $ t'_n \to x'$
(बढ़ते क्रम उनके साथ जुड़ते हैं \omterm{def:b1:reals:bounds}{सर्वोच्च}, यानी\ द
मूल्य), तो $ t_n + t'_n \to x + x'$ द्वारा
\cref{thm:b1:seq:operations}. दिया गया $ y \in \intcc{0}{1}$ के साथ
कोड $(e_k)$, चुनना $(a_k, b_k) = (0,0), (0,2), (2,2)$
के अनुसार $ e_k = 0, 1, 2$: तब $ a_k + b_k = 2e_k $, द
तार $(a_k)$, $(b_k)$ हैं $\{0,2\}$-मूल्यों के साथ तार $ x,
x' \in C $, और
\[
x + x' = \lim_n\,(t_n + t'_n) = \lim_n 2\sum_{k=1}^n e_k 3^{-k}
= 2y :
\]
प्रत्येक $ y \in \intcc{0}{1}$ के दो बिंदुओं का मध्यबिंदु है $ C $.

\textbf{16.} प्रश्न 15 देता है $\intcc{0}{2} = 2\,\intcc{0}{1}
\subseteq C + C $, और $ C + C \subseteq \intcc{0}{1} +
\intcc{0}{1} = \intcc{0}{2}$: समानता. फिर, का उपयोग कर $1 - C = C $:
\[
C - C = C + (C - 1) = (C + C) - 1 = \intcc{-1}{1} .
\]
ठोस उदाहरण: $1 = \frac14 + \frac34$, दो का योग
के गैर-समाप्ति बिंदु सदस्य $ C $.

\textbf{17.} लंबाई मापती है कि रेखा का कितना भाग है \omterm{def:b1:logic:sets}{तय करना} स्वयं
कब्जा कर लेता है; यह इसके बारे में कुछ नहीं कहता \omterm{def:b1:logic:sets}{तय करना} का \emph{रकम}, जो है
दो-पैरामीटर परिवार की छवि $ C \times C $ अंतर्गत $(x,
x') \mapsto x + x'$ --- दो अंकीय तार चुने गए हैं
स्वतंत्र रूप से, और वह स्वतंत्रता ही वास्तव में भरती है
$\intcc{0}{2}$. कोई भी प्रमेय किसी समुच्चय की लंबाई को इससे सीमित नहीं करता
सारांश की लंबाई, और $ C $ यह प्रमाण है कि कोई नहीं कर सकता।

\textbf{18.} द्वारा \cref{pb:b1:reals:1} (प्रश्न 18), $ x $ है
तर्कसंगत यदि इसका समुचित विस्तार अंततः आवधिक हो।
$1$- का निःशुल्क कोड $ x \in C $ या तो वह उचित विस्तार है या
एक समाप्त होने वाले का अनुचित जुड़वां; एक समाप्ति स्ट्रिंग और
यह जुड़वां (अंततः स्थिर) है $2$ s) अंततः दोनों हैं
आवधिक, तो की आवधिकता $1$-फ्री कोड के बराबर है
की तर्कसंगतता $ x $. का लम्बा विभाजन $\frac1{13}$ आधार में $3$
($ r_0 = 1$): $3 = 13\cdot0 + 3$, $9 = 13\cdot0 + 9$, $27 =
13\cdot2 + 1$, और शेष वापस आ जाता है $1$: अंक
$\overline{002}$, इसलिए $\frac1{13} = (0.\overline{002})_3$,
$1$-मुक्त और आवधिक: का एक तर्कसंगत सदस्य $ C $. (जाँच करना:
$\frac{2/27}{1 - 1/27} = \frac{2}{26} = \frac1{13}$.)

\textbf{19.} के साथ स्ट्रिंग $ d_k = 2$ त्रिकोणीय पर
पद $ k = \frac{j(j+1)}{2}$ और $0$ अन्यत्र एक है
$\{0,2\}$-स्ट्रिंग, तो इसका मूल्य $ x^*$ से संबंधित $ C $ (प्रश्न
8). इसमें अनन्त रूप से अनेक हैं $2$ \omterm{prop:b1:reals:intervals}{अंतराल} के साथ है $j + 1 \to \infty $
लगातार के बीच, इसलिए यह अंततः आवधिक नहीं है (ए)।
अवधि $ T $ अंततः मजबूर करेगा $2$ \omterm{prop:b1:reals:intervals}{अंतराल} पर है $\leq T $: द
बढ़ते-अंतराल का तर्क \cref{pb:b1:reals:1}, प्रश्न 20); द्वारा
प्रश्न 18, $ x^* \notin \Q $. और प्रश्न 9 प्लस द्वारा
की गणनीयता $\Q $, के सभी लेकिन गिनती के कई सदस्य $ C $ हैं
तर्कहीन: $ x^*$ यह आदर्श है, अपवाद नहीं।

\textbf{20.} होने देना $ y \in \intcc{0}{1}$: इसमें एक बाइनरी कोड है
$(c_k)$ साथ $ c_k \in \{0, 1\}$ (\cref{pb:b1:reals:1}, प्रश्न
9, आधार $2$; $ y = 1$ सब कुछ लेता है-$1$ एस स्ट्रिंग). तब $(2c_k)$
एक है $\{0,2\}$-स्ट्रिंग, इसका मूल्य $ x $ में निहित है $ C $, और $ h(x)$ है
का मूल्य $(c_k)$, अर्थात् $ y $: $ h $ एमएपीएस $ C $ पर
$\intcc{0}{1}$. अगर $ C $ की छवि थे \omterm{def:b1:logic:map}{नक्शा} से $\N^*$,
के साथ रचना करना $ h $ सभी की सूची बनाऊंगा $\intco{0}{1}$,
के विकर्ण प्रमेय का खंडन \cref{pb:b1:reals:1}
(प्रश्न 22): $ C $ फिर से, बेशुमार है. एक लंबाई-शून्य \omterm{def:b1:logic:sets}{तय करना}
एक पूर्ण खंड पर अनुमान लगाना।

\textbf{21.} प्रश्न 5 से, $ C $ का असंयुक्त संघ है
$\varphi_0(C)$ और $\varphi_2(C)$, प्रत्येक का अनुवाद
स्केल्ड कॉपी $\frac13 C $. एडिटिविटी, स्केलिंग और अनुवाद
अपरिवर्तनीय देना
\[
L(C) = L(\varphi_0(C)) + L(\varphi_2(C))
= \tfrac13 L(C) + \tfrac13 L(C) = \tfrac23\,L(C),
\]
इसलिए $\frac13 L(C) = 0$: $ L(C) = 0$. अकेले आत्म-समानता
वाक्य $ C $ लंबाई शून्य तक --- प्रश्न 3 को केवल निष्पादित किया गया
वाक्य.

\textbf{22.} लंबाई का एक खंड $ l_n $ एक केंद्रीय खो देता है \omterm{prop:b1:reals:intervals}{अंतराल}
लम्बाई का $4^{-(n+1)}$, लंबाई के दो खंड छोड़कर $ l_{n+1}
= \frac{l_n - 4^{-(n+1)}}{2}$; से $ l_0 = 1$, प्रेरण
पुष्टि करता है $ l_n = \frac{2^n + 1}{2\cdot4^n}$: वास्तव में
$\frac12\Bigl(\frac{2^n+1}{2\cdot4^n} - \frac{1}{4^{n+1}}\Bigr)
= \frac{2(2^n + 1) - 1}{2\cdot4^{n+1}} = \frac{2^{n+1} +
1}{2\cdot4^{n+1}}$, और $ l_n > 0$ हमेशा: निर्माण कभी नहीं
भूखा मरता है. $ K = \bigcap K_n $ है \omterm{def:b1:topology:closed}{बंद किया हुआ} और घिरा हुआ, इसलिए सघन;
एक \omterm{prop:b1:reals:intervals}{अंतराल} अंदर $ K $ के एक खंड में स्थित है $ K_n $, लंबाई का
$ l_n \to 0$: खाली \omterm{def:b1:topology:closure}{आंतरिक भाग}. हटाई गई लंबाई: $\sum_{n\geq0} 2^n
\cdot 4^{-(n+1)} = \frac14\sum_{n\geq0}\bigl(\frac12\bigr)^n =
\frac12$, और प्रत्येक $ K_n $ कुल लंबाई है $2^n l_n = \frac{2^n +
1}{2^{n+1}} > \frac12$. अब बहुत कुछ होने दो \omterm{def:b1:topology:open}{खुला} \omterm{prop:b1:reals:intervals}{अंतराल}
मिलन है $ U \supseteq K $. साथी द्वारा \omterm{def:b1:logic:statement}{कथन} का
\cref{ex:b1:topology:nested}, $ U \supseteq K_n $ कुछ के लिए $ n $;
के परिमित संघों पर लंबाई की योगात्मकता को स्वीकार करना \omterm{prop:b1:reals:intervals}{अंतराल},
आवरण की कुल लंबाई \omterm{prop:b1:reals:intervals}{अंतराल} कम से कम यही है
$ K_n $, जो अधिक है $\frac12$. इसलिए $ K $ कहीं नहीं है \omterm{def:b1:topology:dense}{घना}, फिर भी नहीं
सस्ता कवर मौजूद है: टोपोलॉजिकल छोटापन (कहीं नहीं \omterm{def:b1:topology:dense}{घना}) और
मीट्रिक छोटापन (लंबाई शून्य) वास्तव में अलग-अलग धारणाएं हैं,
और $ K $ उन्हें अलग करता है.

\textbf{23.} प्राप्ति: चलो $ d = d(x, F)$ और चुनें $ a_k \in F $
साथ $\abs{x - a_k} \leq d + \frac1k $: द $ a_k $ बंधे हुए हैं, इसलिए
बोल्ज़ानो--वीयरस्ट्रैस (\cref{thm:b1:seq:bw}) अर्क
$ a_{\varphi(k)} \to a $, साथ $ a \in F $ ($ F $ \omterm{def:b1:topology:closed}{बंद किया हुआ},
\cref{thm:b1:topology:seqclosed}) और $\abs{x - a} = \lim
\abs{x - a_{\varphi(k)}} = d $. अब अधिकतम: यदि $ y \in C $,
$ d(y, C) = 0$; अन्यथा $ y $ किसी स्तर पर हटाए गए \omterm{prop:b1:reals:intervals}{अंतराल} में निहित है
$ n \geq 1$, एक \omterm{def:b1:topology:open}{खुला} \omterm{prop:b1:reals:intervals}{अंतराल} लम्बाई का $3^{-n}$ जिनके दो
समापनबिंदु से संबंधित हैं $ C $ (प्रश्न 2), तो $ d(y, C) \leq
\frac{3^{-n}}{2} \leq \frac16$, समानता की आवश्यकता के साथ $ n = 1$
और $ y $ \omterm{prop:b1:reals:intervals}{अंतराल} के केंद्र में $\intoo{\frac13}{\frac23}$,
यानी\ $ y = \frac12$; सचमुच $ d\bigl(\frac12, C\bigr) =
\frac16$ तब से $ C \cap \intoo{\frac13}{\frac23} = \emptyset $ और
$\frac13, \frac23 \in C $. इस तरह $\max_{y\in\intcc{0}{1}} d(y, C)
= \frac16$, ठीक उसी समय प्राप्त हुआ $\frac12$.

\textbf{24.} अंतिमबिंदु एक गणनीय बनाते हैं \omterm{def:b1:topology:dense}{घना} का उपसमुच्चय $ C $
(प्रश्न 12): उन्हें एक अनुक्रम के रूप में सूचीबद्ध करें $(e_j)_{j\geq1}$,
में एक क्रम $ C $. इसकी सभी आगामी सीमाएं निहित हैं $ C $ ($ C $
\omterm{def:b1:topology:closed}{बंद किया हुआ}). इसके विपरीत, ठीक करें $ x \in C $: प्रत्येक के लिए $ n $, खंड
का $ C_m $ युक्त $ x $ ($ m \geq n $) के अपने समापन बिंदु हैं
भीतर $3^{-m} \leq 3^{-n}$ का $ x $, तो \omterm{def:b1:reals:bounds}{असीम} रूप से कई अलग
समापनबिंदु भीतर स्थित हैं $3^{-n}$ का $ x $; सूचकांक चुनें $ j_1 < j_2
< \dots $ साथ $\abs{e_{j_n} - x} \leq 3^{-n}$: ए \omterm{def:b1:seq:subsequence}{परिणाम को}
में एकत्रित होना $ x $. इतना \omterm{def:b1:logic:sets}{तय करना} की अनुवर्ती सीमाओं की $(e_j)$
बिलकुल है $ C $ --- एक क्रम असंख्यों पर एकत्रित हो रहा है
बिंदु, एक अभिसरण अनुक्रम से विपरीत चरम, जिसका
क्लस्टर \omterm{def:b1:logic:sets}{तय करना} एक सिंगलटन है.

\textbf{25.} (i) निर्माण में खपत: की स्थिरता \omterm{def:b1:topology:closure}{बंद}
सेट मनमाने ढंग से प्रतिच्छेदन के तहत (अस्तित्व) $ C $ के तौर पर \omterm{def:b1:topology:closure}{बंद}
सेट), सघनता प्रमेय \cref{thm:b1:topology:compact}
(प्रश्न 2, 22, 23), और अनुक्रमिक लक्षण वर्णन
बंदता और पालन (नेस्टेड-कॉम्पैक्ट तर्क)।
\cref{ex:b1:topology:nested} और प्रश्न 23). (ii) चार
युग्म: लंबाई शून्य फिर भी बेशुमार (प्रश्न 3, 9); \omterm{def:b1:topology:closed}{बंद किया हुआ}
फिर भी खाली के साथ \omterm{def:b1:topology:closure}{आंतरिक भाग} (प्रश्न 4); उत्तम --- पृथक नहीं
बिंदु --- फिर भी पूरी तरह से असंबद्ध (प्रश्न 11, 13);
अभी तक नगण्य है $ C + C = \intcc{0}{2}$ (प्रश्न 16). (iii)
आक्षेप $ h $ प्रश्न 20 का, निरंतर बनाया गया और
न घटनेवाला, सिद्धांत में शैतान की सीढ़ी बन जाता है
निरंतर कार्य; और वर्ष 3 खंड के माप सिद्धांत में,
$ C$ मानक गवाह है कि ``नगण्य'' का मतलब नहीं है
``गणनीय'', इसके मोटे चचेरे भाई (प्रश्न 22) के साथ अलग हो रहा है
``कहीं नहीं \omterm{def:b1:topology:dense}{घना}'' ''नगण्य'' से।
\end{solution}
