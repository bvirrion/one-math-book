\section*{अध्याय \ref{ch:b1:complex} --- जटिल संख्याएँ}

\begin{solution}{exo:b1:complex:1}
$1 + \iu = \sqrt 2\, \eu^{\iu\pi/4}$;
$\;\sqrt 3 - \iu = 2\, \eu^{-\iu\pi/6}$;
$\;-5 = 5\, \eu^{\iu\pi}$;
\[
\frac{1 + \iu}{\sqrt 3 - \iu}
= \frac{\sqrt 2}{2}\, \eu^{\iu(\pi/4 + \pi/6)}
= \frac{\sqrt 2}{2}\, \eu^{5\iu\pi/12}.
\]
समान भागफल की \omterm{pb:b1:logic:1}{बीजगणितीय} रूप से गणना करना (से गुणा करना)। \omterm{def:b1:complex:field}{संयुग्मित}):
\[
\frac{(1 + \iu)(\sqrt 3 + \iu)}{4}
= \frac{(\sqrt 3 - 1) + \iu(\sqrt 3 + 1)}{4}.
\]
से पहचान $\frac{\sqrt 2}{2}(\cos\frac{5\pi}{12} +
\iu\sin\frac{5\pi}{12})$:
\[
\cos\frac{5\pi}{12} = \frac{\sqrt 6 - \sqrt 2}{4},
\qquad
\sin\frac{5\pi}{12} = \frac{\sqrt 6 + \sqrt 2}{4}.
\]
\end{solution}

\begin{solution}{exo:b1:complex:2}
$(1+\iu)^2 = 2\iu $, इसलिए $(1+\iu)^{20} = (2\iu)^{10} = 2^{10}\,\iu^{10}
= 1024 \times (-1) = -1024$.

घातीय रूप में $(1+\iu)^n = 2^{n/2}\, \eu^{\iu n\pi/4}$, असली अगर
और केवल यदि $\sin\frac{n\pi}{4} = 0$, यानी\ $4 \mid n $. इसलिए $(1+\iu)^n
\in \R $ बिल्कुल के गुणकों के लिए $4$ (कीमत $(-4)^{n/4}$).
\end{solution}

\begin{solution}{exo:b1:complex:3}
$\abs{z - 2} = \abs{z + \iu}$: बिंदुओं से समान दूरी $2$ और
$-\iu $ --- जुड़ने वाले खंड का लंबवत समद्विभाजक $(2, 0)$
और $(0, -1)$.

$\abs{z - 1} = 2$: केंद्र का वृत्त $1$ और त्रिज्या $2$.

$\frac{z-1}{z+1} \in \iu\R $: द्वारा
\cref{prop:b1:complex:geometry}~(3), अंक $ M(z)$, $ A(1)$,
$ B(-1)$ संतुष्ट करें: पंक्तियाँ $ MA $ और $ MB $ लंबवत (या $ z = 1$, कहाँ
भागफल है $0 \in \iu\R $). \omterm{def:b1:logic:sets}{तय करना} व्यास का वृत्त है
$[-1, 1]$ (इकाई वृत्त), बिंदु घटा $-1$ जहां भागफल
अपरिभाषित है. \emph{गणना द्वारा जांचें:} $ z = \eu^{\iu\theta}$ देता है
$\frac{z-1}{z+1} = \frac{\eu^{\iu\theta/2}(\eu^{\iu\theta/2} -
\eu^{-\iu\theta/2})}{\eu^{\iu\theta/2}(\eu^{\iu\theta/2} +
\eu^{-\iu\theta/2})} = \iu\tan\frac\theta2 \in \iu\R $.
\end{solution}

\begin{solution}{exo:b1:complex:4}
रैखिककरण:
\[
\sin^4\theta
= \Bigl(\frac{\eu^{\iu\theta} - \eu^{-\iu\theta}}{2\iu}\Bigr)^{\!4}
= \frac{\eu^{4\iu\theta} - 4\eu^{2\iu\theta} + 6 - 4\eu^{-2\iu\theta}
+ \eu^{-4\iu\theta}}{16}
= \frac{\cos 4\theta - 4\cos 2\theta + 3}{8}.
\]
विस्तार: डी मोइवरे द्वारा, $\cos 4\theta = \Re\bigl((c +
\iu s)^4\bigr) = c^4 - 6c^2 s^2 + s^4$ साथ $ c = \cos\theta $, $ s =
\sin\theta $; प्रतिस्थापन $ s^2 = 1 - c^2$:
\[
\cos 4\theta = c^4 - 6c^2(1 - c^2) + (1 - c^2)^2
= 8c^4 - 8c^2 + 1 .
\]
\end{solution}

\begin{solution}{exo:b1:complex:5}
$8\iu = 8\,\eu^{\iu\pi/2}$, तो घनमूल हैं $2\,\eu^{\iu(\pi/6 +
2k\pi/3)}$, $ k = 0, 1, 2$:
\[
z_0 = 2\eu^{\iu\pi/6} = \sqrt 3 + \iu,\quad
z_1 = 2\eu^{5\iu\pi/6} = -\sqrt 3 + \iu,\quad
z_2 = 2\eu^{3\iu\pi/2} = -2\iu :
\]
त्रिज्या के वृत्त पर एक समबाहु त्रिभुज $2$.

$-4 = 4\eu^{\iu\pi}$, इसलिए $ z^4 = -4$ समाधान है $\sqrt 2\,
\eu^{\iu(\pi/4 + k\pi/2)}$: $\;1 + \iu $, $-1 + \iu $, $-1 - \iu $,
$1 - \iu $. बाँधना \omterm{def:b1:complex:field}{संयुग्मित} जड़ें:
\[
X^4 + 4 = \bigl(X^2 - 2X + 2\bigr)\bigl(X^2 + 2X + 2\bigr),
\]
तब से $(X - (1+\iu))(X - (1-\iu)) = X^2 - 2X + 2$ और इसी तरह के लिए
दूसरी जोड़ी. (जांचने के लिए विस्तार करें।)
\end{solution}

\begin{solution}{exo:b1:complex:6}
$ C_n + \iu S_n = \sum_{k=0}^{n} \eu^{\iu k\theta} =
\frac{\eu^{\iu(n+1)\theta} - 1}{\eu^{\iu\theta} - 1}$ (ज्यामितीय योग,
अनुपात $\eu^{\iu\theta} \neq 1$). अर्ध-कोण गुणनखंडन
(\cref{met:b1:complex:trig}~(4)) अंश और हर पर:
\[
\frac{2\iu\sin\frac{(n+1)\theta}{2}\, \eu^{\iu(n+1)\theta/2}}
{2\iu\sin\frac{\theta}{2}\, \eu^{\iu\theta/2}}
= \frac{\sin\frac{(n+1)\theta}{2}}{\sin\frac{\theta}{2}}\,
\eu^{\iu n\theta/2}.
\]
वास्तविक और काल्पनिक भाग लेना:
\[
C_n = \frac{\sin\frac{(n+1)\theta}{2}}{\sin\frac{\theta}{2}}
\cos\frac{n\theta}{2},
\qquad
S_n = \frac{\sin\frac{(n+1)\theta}{2}}{\sin\frac{\theta}{2}}
\sin\frac{n\theta}{2}.
\]
\end{solution}

z = \frac{(3 + 4\iu) \pm (1 + 2\iu)}{2}

\emph{जाँच करना:} जोड़ $= 3 + 4\iu $ और उत्पाद $(2+3\iu)(1+\iu) = -1 +

\end{solution}

\begin{solution}{exo:b1:complex:9}
के प्रमाण के अनुसार दोनों वर्गों का विस्तार करें
\cref{prop:b1:complex:rules}~(4):
\[
\abs{z + w}^2 = \abs z^2 + \abs w^2 + 2\Re(z\conj w),
\qquad
\abs{z - w}^2 = \abs z^2 + \abs w^2 - 2\Re(z\conj w),
\]
और जोड़ें. ज्यामितीय रूप से, $\abs{z+w}$ और $\abs{z-w}$ की लंबाई हैं
शीर्षों सहित समांतर चतुर्भुज के दो विकर्ण $0, z, z+w, w $,
जबकि $\abs z $ और $\abs w $ भुजाओं की लंबाई हैं: का योग
विकर्णों का वर्ग चारों के वर्गों के योग के बराबर होता है

\end{solution}

\begin{solution}{exo:b1:complex:10}
होने देना $ j = \eu^{2\iu\pi/3}$, इसलिए $ j^2 + j + 1 = 0$ और $\eu^{\iu\pi/3} =
-j^2$, $\eu^{-\iu\pi/3} = -j $. त्रिभुज समबाहु सीधी iff है
$ c - a = -j^2 (b - a)$, अप्रत्यक्ष आईएफएफ $ c - a = -j(b - a)$
(\cref{ex:b1:complex:equilateral}).

विचार करना $ P = a + jb + j^2 c $ और $ Q = a + j^2 b + jc $. का उपयोग करते हुए $1 +
j^2 = -j $ और $ j^3 = 1$:
\[
c - a + j^2(b - a) = c + j^2 b - (1 + j^2)\,a = c + j^2 b + ja
= j\,(a + jb + j^2 c) = jP,
\]
तो सीधा मामला पढ़ता है $ jP = 0 \iff P = 0$; वही गणना
के साथ $ j $ की जगह $ j^2$ देता है $ c - a + j(b - a) = j^2 Q $, तो
अप्रत्यक्ष मामला पढ़ता है $ Q = 0$. इसलिए: समबाहु (कोई भी अभिविन्यास)
$\iff PQ = 0$. विस्तार, के साथ $ j + j^2 = -1$:
\[
PQ = a^2 + b^2 + c^2 + (j + j^2)(ab + bc + ca)
= a^2 + b^2 + c^2 - (ab + bc + ca).
\]
अतः त्रिभुज समबाहु है यदि और केवल यदि $ a^2 + b^2 + c^2 = ab +
bc + ca $.
\end{solution}

\begin{solution}{exo:b1:complex:11}
के बाद से $\omega^k $, $ k = 0, \dots, n-1$, बिल्कुल वही हैं $ n $ जड़ें
का $ X^n - 1$ (\cref{thm:b1:complex:roots}), और बहुपद है
राक्षसी:
\[
X^n - 1 = \prod_{k=0}^{n-1} (X - \omega^k)
= (X - 1) \prod_{k=1}^{n-1} (X - \omega^k).
\]
द्वारा विभाजित करना $ X - 1$: $\;1 + X + \dots + X^{n-1} = \prod_{k=1}^{n-1}
(X - \omega^k)$. पर मूल्यांकन कर रहे हैं $ X = 1$ देता है $ P = n $.

अब $1 - \omega^k = -\eu^{\iu k\pi/n}\bigl(\eu^{\iu k\pi/n} -
\eu^{-\iu k\pi/n}\bigr) = -2\iu\,\eu^{\iu k\pi/n} \sin\frac{k\pi}{n}$,
इसलिए मॉड्यूलि को अंदर ले जा रहे हैं $ P = n $ (प्रत्येक $\sin\frac{k\pi}n > 0$ के लिए $1 \leq k
\leq n-1$):
\[
n = \abs P = \prod_{k=1}^{n-1} 2\sin\frac{k\pi}{n}
= 2^{n-1} \prod_{k=1}^{n-1} \sin\frac{k\pi}{n},
\qquad\text{hence}\qquad
\prod_{k=1}^{n-1} \sin\frac{k\pi}{n} = \frac{n}{2^{n-1}} .
\]
\end{solution}

\begin{solution}{exo:b1:complex:12}
$ z = 1$ कोई समाधान नहीं है ($2^n \neq 0$), तो समीकरण है
के बराबर $\bigl(\frac{z+1}{z-1}\bigr)^n = 1$, अर्थात\
$\frac{z+1}{z-1} = \omega^k $ साथ $\omega = \eu^{2\iu\pi/n}$ और
$ k \in \{0, \dots, n-1\}$. मूल्य $ k = 0$ बहिष्कृत है ($ z + 1 =
z - 1$ असंभव है)। के लिए $1 \leq k \leq n - 1$, सुलझाना
$ z + 1 = \omega^k(z - 1)$ देता है $ z(1 - \omega^k) = -1 - \omega^k $,
के साथ $\varphi = \frac{2k\pi}{n}$ और आधा कोण
गुणनखंडीकरण $1 + \eu^{\iu\varphi} = 2\cos\frac\varphi2\,
\eu^{\iu\varphi/2}$, $\eu^{\iu\varphi} - 1 = 2\iu\sin\frac\varphi2
\,\eu^{\iu\varphi/2}$:
\[
z_k = \frac{1 + \omega^k}{\omega^k - 1}
= \frac{2\cos\frac{k\pi}{n}}{2\iu\,\sin\frac{k\pi}{n}}
= -\iu\,\frac{\cos\frac{k\pi}{n}}{\sin\frac{k\pi}{n}} ,
\]
जैसा कि दावा किया गया है यह पूरी तरह से काल्पनिक है। \omterm{def:b1:logic:map}{नक्शा} $ t \mapsto \cos t/\sin t $ है
\omterm{def:b1:logic:inj}{इंजेक्शन} पर $\intoo0\pi $ (यह सख्ती से घट रहा है), इसलिए
$ n - 1$ मान $ z_k $ जोड़ीवार भिन्न हैं: समीकरण, डिग्री का
$ n - 1$ एक बार विस्तारित ( $ z^n $ शर्तें रद्द), वास्तव में ये हैं
$ n - 1$ समाधान.
\end{solution}

\begin{solution}{pb:b1:complex:1}
\textbf{1.} $ j = \cos\frac{2\pi}3 + \iu\sin\frac{2\pi}3 = -\frac12
+ \iu\frac{\sqrt3}2$; $ j^2 = \eu^{4\iu\pi/3} = -\frac12 -
\iu\frac{\sqrt3}2 = \conj j $; $ j^3 = 1$; $1 + j + j^2 = 0$ (का योग
घन एकता की जड़ें, \cref{prop:b1:complex:sumroots}); $ j^{-1}
= j^2$ (तब से $ j \cdot j^2 = 1$). यूनिट सर्कल पर, $1$, $ j $,
$ j^2$ एक सीधे समबाहु त्रिभुज के शीर्ष हैं।

\textbf{2.} केन्द्र का घूर्णन $ a $ और कोण $\theta $ ठीक करता है $ a $
और से जारी प्रत्येक वेक्टर को बदल देता है $ a $ द्वारा $\theta $: $ r(z) - a =
\eu^{\iu\theta}(z - a)$, यानी\ $ r(z) = a + \eu^{\iu\theta}(z - a)$.
लिखना $ r(z) = a + \eu^{\iu\theta}(z-a)$ और $ r'(z) = a' +
\eu^{\iu\theta'}(z-a')$:
\[
r' \circ r\,(z) = \eu^{\iu(\theta + \theta')} z +
\text{constant},
\]
ए \omterm{def:b1:logic:map}{नक्शा} रूप का $ Az + B $ साथ $ A = \eu^{\iu(\theta+\theta')}$ का
\omterm{def:b1:complex:field}{मापांक} $1$. द्वारा \cref{prop:b1:complex:similitude}, यह एक चक्र है
कोण का $\arg A = \theta + \theta'$ अगर $ A \neq 1$, और ए
अनुवाद यदि $ A = 1$, यानी\ अगर $\theta + \theta' \in 2\pi\Z $.

\textbf{3.} $(b, c, p)$ समबाहु है iff $\abs{p - b} = \abs{c -
b} = \abs{p - c}$. लिखना $ q = \frac{p - b}{c - b}$, पहला
समानता कहती है $\abs q = 1$ और दूसरा $\abs{q - 1} = 1$;
एक साथ, $ q = \eu^{\pm\iu\pi/3}$ (के दो प्रतिच्छेदन बिंदु
वृत्त $\abs q = 1$ और $\abs{q - 1} = 1$ हैं $\frac12 \pm
\iu\frac{\sqrt3}2$). इस तरह $ p = b + \eu^{\pm\iu\pi/3}(c - b)$. के लिए
$ a = 0$, $ b = 1$, $ c = \iu $ (एक सीधा त्रिकोण): विकल्प
$\eu^{-\iu\pi/3}$ देता है
\[
p = 1 + \Bigl(\tfrac12 - \iu\tfrac{\sqrt3}2\Bigr)(\iu - 1)
= \tfrac{1 + \sqrt3}2\,(1 + \iu) \approx 1.37 + 1.37\,\iu ,
\]
जो रेखा के सबसे दूर स्थित है $ BC $ ($ x + y = 1$) से $ a
= 0$: बाहर की ओर. विकल्प $\eu^{+\iu\pi/3}$ देता है $ p \approx -0.37
- 0.37\,\iu $, उसी तरफ $ a $: भीतर की ओर.

\textbf{4.} गुणा $ P = a + jb + j^2c $ द्वारा $ j^2$: $ j^2 P = j^2 a +
j^3 b + j^4 c = b + jc + j^2 a $ (का उपयोग करते हुए $ j^3 = 1$, $ j^4 = j $); और तक $ j $: $ jP = ja + j^2 b + c $. ये दो पहचान हैं. अगर $ P
= 0$ तब $ j^2 P = jP = 0$: मानदंड कायम है $(b, c, a)$ और
$(c, a, b)$ साथ ही --- चक्रीय अपरिवर्तन (जैसा कि यह होना चाहिए: an
समबाहु त्रिभुज को इसकी परवाह नहीं है कि कौन सा शीर्ष पहले सूचीबद्ध है)।

\textbf{5.} अगर $ a = b $: $0 = a(1 + j) + j^2 c = -j^2 a + j^2 c $
(का उपयोग करते हुए $1 + j = -j^2$), इसलिए $ c = a $. अगर $ b = c $: $0 = a + b(j + j^2)
= a - b $, इसलिए $ a = b $. अगर $ a = c $: $0 = a(1 + j^2) + jb = -ja + jb $,
इसलिए $ a = b $. प्रत्येक मामले में सभी तीन बिंदु मेल खाते हैं।

\textbf{6.} $(a'z + b') \circ (az + b) = a'a\,z + (a'b + b')$ साथ
$ a'a \neq 0$: एक ही रूप. का उलटा $ z \mapsto az + b $ है $ z
\mapsto \frac1a z - \frac ba $, फिर से उसी रूप का। के साथ
पहचान \omterm{def:b1:logic:map}{नक्शा} तटस्थ तत्व के रूप में, प्रत्यक्ष समानताएँ बनती हैं
रचना के अंतर्गत समूह.

\textbf{7.} $ f(z) = az + b $ संतुष्ट $ f(z_1) = w_1$, $ f(z_2) =
w_2$ आईएफएफ $ a z_1 + b = w_1$ और $ a z_2 + b = w_2$; घटाना,
$ a(z_2 - z_1) = w_2 - w_1$, इसलिए
\[
a = \frac{w_2 - w_1}{z_2 - z_1} \;(\neq 0), \qquad
b = w_1 - a z_1
\]
मजबूर हैं, और इसके विपरीत यह विकल्प काम करता है: अस्तित्व और
विशिष्टता.

\textbf{8.} $\abs{f(z) - f(w)} = \abs{a(z - w)} = \abs a\,\abs{z -
w}$: सभी दूरियों को गुणा किया जाता है $\abs a $. आकृति के लिए:
\[
\frac{f(c') - f(a')}{f(b') - f(a')} = \frac{a(c' -
a')}{a(b' - a')} = \frac{c' - a'}{b' - a'} .
\]
यदि दो त्रिभुज $(a', b', c')$ और $(a'', b'', c'')$ बराबर है
आकार, चलो $ f $ के साथ अद्वितीय प्रत्यक्ष समानता हो $ f(a') = a''$
और $ f(b') = b''$ (प्रश्न 7); फिर का आकार $(a'', b'',
f(c'))$ के बराबर है $(a', b', c')$, इसलिए उस का $(a'', b'',
c'')$, और आकृति पहले से तीसरा शीर्ष निर्धारित करती है
दो: $ f(c') = c''$. इसके विपरीत समान आकृतियाँ अनुसरण करती हैं
अपरिवर्तनशीलता प्रदर्शित की गई।

\textbf{9.} $ b = f(0) = 1$; $ a + b = f(1) = \iu $ देता है $ a = \iu -
1$. अनुपात $\abs a = \sqrt2$, कोण $\arg(\iu - 1) =
\frac{3\pi}4$. नियत बिन्दु:
\[
\zeta = \frac{b}{1 - a} = \frac1{2 - \iu} = \frac{2 + \iu}5 .
\]
इसलिए $ f $ केंद्र की प्रत्यक्ष समानता है $\frac{2+\iu}5$, अनुपात
$\sqrt2$ और कोण $\frac{3\pi}4$.

\textbf{10.} साथ $\alpha + \beta = 1$ और $ f(z) = az + b $:
\[
f(\alpha a' + \beta b') = a\alpha a' + a\beta b' + b
= \alpha(a a' + b) + \beta(a b' + b)
= \alpha f(a') + \beta f(b') ,
\]
कुंजी है $ b = (\alpha + \beta)b $. मध्यबिंदु ($\alpha = \beta =
\frac12$), सेंट्रोइड्स (पुनरावृत्त), और नेपोलियन केंद्र $\alpha b
+ \beta c $ इसलिए नीचे हैं \emph{समतुल्य}: रूपांतरित करना
त्रिभुज उन्हें तदनुसार रूपांतरित कर देता है। अत: सिद्ध हो रहा है कि ए
समानता-अपरिवर्तनीय \omterm{def:b1:logic:statement}{कथन} एक अच्छी तरह से स्थित त्रिकोण के लिए साबित होता है
यह सभी के लिए --- प्रश्न 21 में प्रयुक्त लाइसेंस।

\textbf{11.} बाहरी शीर्ष पर $[b, c]$ है $ p_a = b +
\eu^{-\iu\pi/3}(c - b)$ (प्रश्न 3), तो केंद्र है
\[
n_a = \frac{b + c + p_a}3
= \frac{2b + c + \frac{1 - \iu\sqrt3}2\,(c - b)}3
= \frac{(3 + \iu\sqrt3)\,b + (3 - \iu\sqrt3)\,c}6
= \alpha b + \beta c ,
\]
साथ $\alpha = \frac{3 + \iu\sqrt3}6$ और $\beta = \frac{3 -
\iu\sqrt3}6 = \conj\alpha $. पक्षों पर भी यही गणना
$[c, a]$ और $[a, b]$ देता है $ n_b = \alpha c + \beta a $ और $ n_c =
\alpha a + \beta b $ (भूमिकाओं का चक्रीय बदलाव)।

\textbf{12.} $\alpha + \beta = \frac{3 + \iu\sqrt3 + 3 -
\iu\sqrt3}6 = 1$. इस तरह
\[
\frac{n_a + n_b + n_c}3
= \frac{(\alpha + \beta)(a + b + c)}3 = \frac{a + b + c}3 :
\]
दोनों त्रिभुज अपना केन्द्रक साझा करते हैं।

\textbf{13.} द्वारा समूह बनाएं $\alpha $ और $\beta $ और प्रश्न 4 लागू करें
की पहचान $ P = a + jb + j^2c $:
\[
n_a + j n_b + j^2 n_c
= \alpha\,(b + jc + j^2 a) + \beta\,(c + ja + j^2 b)
= \alpha\,j^2 P + \beta\,j P
= (\alpha j^2 + \beta j)\,P .
\]

\textbf{14.} साथ $ j = \frac{-1 + \iu\sqrt3}2$:
\[
\alpha j = \frac{(3 + \iu\sqrt3)(-1 + \iu\sqrt3)}{12}
= \frac{-3 + 3\iu\sqrt3 - \iu\sqrt3 - 3}{12}
= \frac{-3 + \iu\sqrt3}6 = -\beta ,
\]
इसलिए $\alpha j + \beta = 0$, इस तरह $\alpha j^2 + \beta j = j(\alpha j
+ \beta) = 0$, और प्रश्न 13 देता है $ n_a + jn_b + j^2n_c = 0$ के लिए
हर त्रिकोण $(a, b, c)$. कसौटी के अनुसार (प्रश्न 4--5), द
केंद्र एक सीधा समबाहु त्रिभुज या एक बिंदु बनाते हैं:
नेपोलियन का प्रमेय.

\textbf{15.} भीतर का शीर्ष है $ b + \eu^{+\iu\pi/3}(c - b)$, और
प्रश्न 11 की गणना $\eu^{+\iu\pi/3} = \frac{1 +
\iu\sqrt3}2$ स्वैप $\alpha $ और $\beta $: $ m_a = \beta b + \alpha
c $, $ m_b = \beta c + \alpha a $, $ m_c = \beta a + \alpha b $. उपयोग करना
समान पहचान $ b + j^2c + ja = j\,Q $ और $ c + j^2a + jb
= j^2 Q $ के लिए $ Q = a + j^2b + jc $:
\[
m_a + j^2 m_b + j m_c
= \beta(b + j^2 c + ja) + \alpha(c + j^2 a + jb)
= (\beta j + \alpha j^2)\, Q = j(\beta + \alpha j)\,Q = 0 ,
\]
तब से $\alpha j = -\beta $ (प्रश्न 14). तो आंतरिक केंद्र
को संतुष्ट करें \emph{अप्रत्यक्ष} समबाहु मानदंड: समबाहु के साथ
विपरीत दिशा (या एक बिंदु)।

\textbf{16.} प्रश्न 5 द्वारा त्रिगुण पर लागू होता है $(n_a, n_b, n_c)$
(जो प्रत्यक्ष मानदंड को पूरा करता है), दो केंद्र मेल खाते हैं यदि
तीनों करते हैं; और तीनों संपाती हैं यदि \emph{दोनों} मानदंड कायम,
यानी\ यदि इसके अतिरिक्त $ n_a + j^2 n_b + j n_c = 0$. के रूप में गणना करें
प्रश्न 13, पहचान के साथ $ b + j^2c + ja = jQ $, $ c + j^2a +
jb = j^2Q $:
\[
n_a + j^2 n_b + j n_c
= \alpha\,jQ + \beta\,j^2 Q = j(\alpha + \beta j)\,Q .
\]
सीधे: $\beta j = \frac{(3 -
\iu\sqrt3)(-1 + \iu\sqrt3)}{12} = \frac{-3 + 3\iu\sqrt3 +
\iu\sqrt3 + 3}{12} = \frac{\iu\sqrt3}3$, इसलिए $\alpha + \beta j =
\frac{3 + \iu\sqrt3 + 2\iu\sqrt3}6 = \frac{1 + \iu\sqrt3}2 \neq
0$. इसलिए बाहरी नेपोलियन त्रिभुज iff का ह्रास होता है $ Q = a +
j^2b + jc = 0$, यानी\ आईएफएफ $(a, b, c)$ एक अप्रत्यक्ष समबाहु है
त्रिभुज --- उस स्थिति में ``बाहर की ओर'' निर्माण सभी बिंदु हैं
त्रिभुज के परिबद्ध क्षेत्र में और एक केंद्र साझा करें।

\textbf{17.} साथ $ a = 0$, $ b = 1$, $ c = \iu $:
\[
n_a = \alpha + \beta\iu = \frac{(3 + \sqrt3)(1 + \iu)}6,
\qquad
n_b = \alpha\iu = \frac{-\sqrt3 + 3\iu}6,
\qquad
n_c = \beta = \frac{3 - \iu\sqrt3}6 .
\]
वर्गाकार भुजाएँ: $ n_a - n_b = \frac{(3 + 2\sqrt3) +
\iu\sqrt3}6$ देता है $\abs{n_a - n_b}^2 = \frac{(3 + 2\sqrt3)^2 +
3}{36} = \frac{24 + 12\sqrt3}{36} = \frac{2 + \sqrt3}3$; $ n_b -
n_c = \frac{(3 + \sqrt3)(-1 + \iu)}6$ देता है $\abs{n_b - n_c}^2 =
\frac{2(3 + \sqrt3)^2}{36} = \frac{24 + 12\sqrt3}{36}$; $ n_c - n_a
= \frac{-\sqrt3 - \iu(3 + 2\sqrt3)}6$ पुनः वही मान देता है।
तीनों वर्गाकार भुजाएँ बराबर हैं $\frac{2 + \sqrt3}3$: समबाहु,
जैसा कि वादा किया गया था.

\textbf{18.} बढ़ाना:
\[
(a - b)(c - d) + (a - d)(b - c)
= (ac - ad - bc + bd) + (ab - ac - bd + cd)
= ab - ad - bc + cd ,
\]
और $(a - c)(b - d) = ab - ad - bc + cd $: बराबर।

\textbf{19.} प्रश्न 18 में मोडुली लें और त्रिभुज लगाएं
असमानता (\cref{prop:b1:complex:rules}~(4)):
\[
AC \cdot BD = \abs{(a-b)(c-d) + (a-d)(b-c)}
\leq \abs{a-b}\,\abs{c-d} + \abs{a-d}\,\abs{b-c}
= AB \cdot CD + AD \cdot BC ,
\]
समानता के साथ यदि दोनों योग एक उभयनिष्ठ किरण पर स्थित हों $0$
(त्रिभुज असमानता का समानता मामला)।

\textbf{20.} अर्ध-कोण गुणनखंडन
(\cref{met:b1:complex:trig}~(4)):
$\eu^{\iu\theta} - \eu^{\iu\varphi} = 2\iu\,\sin\frac{\theta -
\varphi}2\;\eu^{\iu(\theta + \varphi)/2}$, और मोडुली लेने से मौत हो जाती है
यूनिमॉड्यूलर कारक: $\abs{\eu^{\iu\theta} - \eu^{\iu\varphi}}
= 2\,\abs{\sin\frac{\theta - \varphi}2}$.

\textbf{21.} साथ $ p = \eu^{\iu\theta}$, $\theta \in
\intoo{2\pi/3}{4\pi/3}$, प्रश्न 20 देता है
\[
PA = 2\,\abs{\sin\tfrac\theta2},
\quad
PB = 2\,\abs{\sin\bigl(\tfrac\theta2 - \tfrac\pi3\bigr)},
\quad
PC = 2\,\abs{\sin\bigl(\tfrac\theta2 - \tfrac{2\pi}3\bigr)} .
\]
चाप पर, $\frac\theta2 \in \intoo{\pi/3}{2\pi/3}$: तब
$\sin\frac\theta2 > 0$; $\frac\theta2 - \frac\pi3 \in
\intoo0{\pi/3}$, इसलिए दूसरी ज्या धनात्मक है; $\frac\theta2 -
\frac{2\pi}3 \in \intoo{-\pi/3}0$, तो तीसरा नकारात्मक है और
$ PC = 2\sin\bigl(\frac{2\pi}3 - \frac\theta2\bigr)$. योग करने के लिए
उत्पाद:
\[
\sin\Bigl(\frac\theta2 - \frac\pi3\Bigr) +
\sin\Bigl(\frac{2\pi}3 - \frac\theta2\Bigr)
= 2\,\sin\frac\pi6\,\cos\Bigl(\frac\theta2 - \frac\pi2\Bigr)
= \sin\frac\theta2 ,
\]
इसलिए $ PB + PC = 2\sin\frac\theta2 = PA $: वैन शूटेन का प्रमेय।

\textbf{22.} पर $ p = -1$: $ PA = 2$, $ PB = PC = 1$ और सभी तरफ
समबाहु त्रिभुज की लंबाई होती है $\sqrt3$, तो दोनों तरफ
टॉलेमी ने पढ़ा $2\sqrt3$. \omterm{pb:b1:logic:1}{बीजगणितीय} रूप से, प्रयोग करते हुए $-1 - j^2 = j $ और
$1 + j = -j^2$:
\[
(a - b)(p - c) = (1 - j)\,(-1 - j^2) = (1 - j)j = j - j^2
= \iu\sqrt3 ,
\]
\[
(a - c)(b - p) = (1 - j^2)(j + 1) = 1 + j - j^2 - j^3
= j - j^2 = \iu\sqrt3 .
\]
दोनों पद बराबर हैं, इसलिए उनका अनुपात है
$1 \in \R_{>0}$: प्रश्न 19 का समानता मामला, मिलान $ PA
\cdot BC = PB \cdot AC + PC \cdot AB $ बिल्कुल।

\textbf{23.} के लिए $(a, b, c) = (1, j, j^2)$: $ n_a = \alpha j +
\beta j^2 = -\beta + \beta j^2$ (प्रश्न 14) $= \beta(j^2 - 1)$;
संख्यानुसार $\beta(j^2 - 1) = \frac{(3 - \iu\sqrt3)}6 \cdot
\bigl(-\frac32 - \iu\frac{\sqrt3}2\bigr) = -1$. वैसे ही $ n_b =
\alpha j^2 + \beta = -j $ और $ n_c = \alpha + \beta j = -j^2$. द
बाहरी नेपोलियन त्रिकोण $(1, j, j^2)$ है $(-1, -j, -j^2)$: द
मूल त्रिभुज अपने केन्द्रक से परावर्तित होता है $0$ --- वही
आकार, आधा-मोड़ घुमाया गया। समबाहु त्रिभुज एक निश्चित आकृति है
नेपोलियन के निर्माण की, सिकुड़ती हुई सीमा नहीं।

\textbf{24.} (i) रोटेशन के रूप में एक यूनिमॉड्यूलर संख्या से गुणा
शीर्षों और केन्द्रों का निर्माण किया (प्रश्न 2--3, 11) और परिवर्तित किया
दूरियों को ज्याओं में वृत्त करें (प्रश्न 20)। (ii) समूह
संरचना और आकार अपरिवर्तनीय ने सामान्यीकरण को उचित ठहराया
इकाई वृत्त को परिवृत्त और त्रिभुज को $(1, j, j^2)$
प्रश्न 21 में, प्रश्न 10 के समतुल्य के माध्यम से। (iii) द
अर्ध-कोण गुणनखंडन प्रश्न 20 और दोनों को संचालित करता है
योग-से-उत्पाद चरण फिनिशिंग वैन शूटेन।

\textbf{25.} शब्दकोश ज्यामितीय रूपांतरित करता है \omterm{def:b1:logic:statement}{कथन} में
बहुपद सर्वसमिकाएँ जिनमें प्रत्येक परिकल्पना एक समीकरण है:
यह जो खरीदता है वह मशीनीकरण है --- नेपोलियन का प्रमेय कम हो गया है
$\alpha j + \beta = 0$, अंकगणित की एक पंक्ति
$\Q(\iu\sqrt3)$; इसकी कीमत ज्यामितीय दृश्यता है ---
गणना प्रमाणित करती है परंतु प्रदर्शित नहीं करती \emph{क्यों} केन्द्रों
एक समबाहु त्रिभुज में बंद करें। इसका उचित उत्तर यह है कि
पहचान प्रमेय की व्याख्या करती है \emph{अनिवार्यता} (यह धारण करता है
समान रूप से में $ a, b, c $, इसलिए कॉन्फ़िगरेशन की कोई चतुराई नहीं है
शामिल), जबकि चित्र इसकी व्याख्या करता है \emph{सामग्री}. वैसा ही
जब घुमाव बन जाते हैं तो शब्दकोश मैट्रिक्स रूप में वापस आ जाता है
ओर्थोगोनल $2 \times 2$ के मैट्रिक्स \cref{ch:b1:matrices} और
\cref{ch:b1:euclid}, जहां `` गुणा
$\eu^{\iu\theta}$'' एक रैखिक आइसोमेट्री का प्रोटोटाइप है।
\end{solution}
