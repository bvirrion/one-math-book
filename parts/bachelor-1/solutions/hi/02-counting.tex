\section*{अध्याय \ref{ch:b1:counting} --- गिनती}

\begin{solution}{exo:b1:counting:1}
स्वतंत्र चरण और उत्पाद नियम: $26^2 \times 10^3 \times 26^2
= 26^4 \times 1000 = 456\,976\,000$ प्लेटें. चार अक्षरों के साथ
जोड़ीवार अलग-अलग, अक्षर चरण एक बनाते हैं $4$-की व्यवस्था
वर्णमाला: $26 \times 25 \times 24 \times 23 = 358\,800$ तरीके, तो
$358\,800 \times 1000 = 358\,800\,000$ प्लेटें.
\end{solution}

\begin{solution}{exo:b1:counting:2}
\textsc{orange} है $6$ विशिष्ट अक्षर: $6! = 720$ विपर्यय.
\textsc{banana} है $6$ दोहराव वाले अक्षर ($3$ जैसा, $2$ n का,
$1$ बी): प्रत्येक अनाग्राम ए की स्थिति से निर्धारित होता है
($\binom 63$ विकल्प), फिर शेष में से n का $3$ धब्बे
($\binom 32$), बी अंतिम स्थान ले रहा है:
$\binom{6}{3}\binom{3}{2} = 20 \times 3 = 60$ विपर्यय
(समान रूप से $6!/(3!\,2!\,1!) = 60$).
\end{solution}

\begin{solution}{exo:b1:counting:3}
कुल: $\binom{12}{4} = 495$. बिल्कुल $2$ महिलाएं: उन्हें चुनें
($\binom 72 = 21$) और $2$ पुरुष ($\binom 52 = 10$): $210$ समितियाँ
कम से कम एक आदमी: ``नो मैन'' का पूरक,
$\binom{12}{4} - \binom{7}{4} = 495 - 35 = 460$.
\end{solution}

\begin{solution}{exo:b1:counting:4}
\emph{जनमदि की:} बक्से हैं $12$ महीने; $13$ लोगों में $12$
बक्से एक ही बक्से में दो को मजबूर करते हैं
(\cref{cor:b1:counting:pigeonhole}).

\emph{लगातार पूर्णांक:} बक्से हैं $ n $ जोड़े $\{1,2\},
\{3,4\}, \dots, \{2n-1, 2n\}$, कौन \omterm{thm:b1:logic:partition}{PARTITION} $\intint{1}{2n}$.
का चयन $ n + 1$ पूर्णांक दो को एक ही युग्म में रखता है, और दो
किसी युग्म के तत्व क्रमागत होते हैं।
\end{solution}

\begin{solution}{exo:b1:counting:5}
के लिए $1 \leq k \leq n $,
\[
k \binom nk = k\,\frac{n!}{k!\,(n-k)!}
= n\,\frac{(n-1)!}{(k-1)!\,(n-k)!} = n \binom{n-1}{k-1}.
\]
के साथ सारांश और पुनः अनुक्रमणिका $ j = k - 1$:
\[
\sum_{k=0}^{n} k \binom nk = n \sum_{j=0}^{n-1} \binom{n-1}{j}
= n\, 2^{n-1}
\]
द्वारा \cref{prop:b1:counting:identities}. (वैकल्पिक: अंतर करें
$(1+x)^n = \sum_k \binom nk x^k $ और सेट करें $ x = 1$.)
\end{solution}

\begin{solution}{exo:b1:counting:6}
एक सख्ती से बढ़ रही है \omterm{def:b1:logic:map}{नक्शा} $ f \colon \intint{1}{k} \to \intint{1}{n}$
इसकी छवि से निर्धारित होता है, a $ k $-का सबसेट $\intint{1}{n}$ (सूचीबद्ध करें
बढ़ते क्रम में उपसमुच्चय); इसके विपरीत प्रत्येक $ k $-सबसेट बिल्कुल देता है
एक ऐसा \omterm{def:b1:logic:map}{नक्शा}. इस तरह $\binom nk $ सख्ती से बढ़ रहा है एमएपीएस.

अगर $ f $ केवल बढ़ रहा है, सेट $ g(i) = f(i) + i - 1$. तब $ g $ है
सख्ती से बढ़ रहा है (लगातार तर्कों के बीच, $ f $ गेन्स $\geq 0$
और $ i - 1$ गेन्स $1$) में मानों के साथ $\intint{1}{n + k - 1}$; और
$ f(i) = g(i) - i + 1$ ठीक $ f $ किसी भी सख्ती से बढ़ रही से $ g $
में $\intint{1}{n+k-1}$. यह एक आक्षेप है, इसलिए हैं
$\binom{n + k - 1}{k}$ की बढ़ती एमएपीएस.
\end{solution}

\begin{solution}{exo:b1:counting:7}
विभाजित करें ए \omterm{def:b1:logic:sets}{तय करना} $ E $ साथ $ m + n $ तत्वों को ब्लॉक में $ M $ ($ m $ तत्व)
और $ N $ ($ n $ तत्व)। ए $ k $-का सबसेट $ E $ कुछ शामिल है $ j $ घटक $ M $ ($0 \leq j \leq k $) और $ k - j $ का $ N $; निश्चित के लिए $ j $ वहाँ हैं
$\binom mj \binom{n}{k-j}$ ऐसे उपसमुच्चय, और मामले $ j = 0, \dots,
k $ \omterm{thm:b1:logic:partition}{PARTITION}  $ k $-उपसमुच्चय. योग नियम वेंडरमोंडे को देता है
पहचान\index{वेंडरमोंडे की पहचान}.

साथ $ m = n = k $: $\binom{2n}{n} = \sum_{j=0}^{n} \binom nj
\binom{n}{n-j} = \sum_{j=0}^{n} \binom nj^2$, का उपयोग करना $\binom{n}{n-j} =
\binom nj $.
\end{solution}

\begin{solution}{exo:b1:counting:8}
होने देना $ A_d $ के गुणज हों $ d $ में $\intint{1}{1000}$, इसलिए
$\abs{A_d} = \lfloor 1000/d \rfloor $. समावेश--बहिष्करण
(\cref{thm:b1:counting:inclexcl}) साथ $ A_2, A_3, A_5$, टिप्पण
$ A_2 \cap A_3 = A_6$ वगैरह।:
\[
500 + 333 + 200 - 166 - 100 - 66 + 33 = 734 .
\]
इसलिए $734$ पूर्णांकों से विभाज्य होते हैं $2$, $3$ या $5$.
\end{solution}

\begin{solution}{exo:b1:counting:9}
पर $2$ तत्व: सभी $2^4 = 16$ एमएपीएस सिवाय $2$ स्थिर एमएपीएस:
$14$ अनुमान.

पर $3$ तत्व: समावेशन द्वारा - छूटे हुए मूल्यों पर बहिष्करण
की संख्या एमएपीएस ए से $4$-ए पर सेट करें $3$-सेट में कम से कम एक मान गुम है
है $\binom 31 2^4 - \binom 32 1^4 = 48 - 3 = 45$; कुल एमएपीएस $3^4 =
81$; अनुमान: $81 - 45 = 36$. (जाँचें: से एक अनुमान $4$ पर
$3$ तत्व ठीक एक मान को दोगुना कर देते हैं: दोगुना मान चुनें
($3$), जोड़ी इसे मैप करती है ($\binom 42 = 6$), और के लिए एक आपत्ति
बाकी ($2$): $3 \times 6 \times 2 = 36$.)
\end{solution}

\begin{solution}{exo:b1:counting:10}
का एक समाधान $ x_1 + \dots + x_n = k $ में $\N^n $ की एक पंक्ति के रूप में एन्कोड करता है
$ k $ सितारे और $ n - 1$ बार्स: लिखें $ x_1$ सितारे, एक बार, $ x_2$ सितारे, ए
बार, \dots, के साथ समाप्त हो रहा है $ x_n $ सितारे. यह पर आक्षेप है
लंबाई के शब्द $ k + n - 1$ का उपयोग करते हुए $ k $ सितारे और $ n - 1$ बार, और
वे शब्द तारों की स्थिति से निर्धारित होते हैं:
$\binom{n + k - 1}{k}$. पुनरावृत्ति के साथ चयन के अनुरूप हैं
समीकरण के समाधान ($ x_i $ = वस्तु की प्रतियों की संख्या $ i $), तो
गिनती वही है.
\end{solution}

\begin{solution}{exo:b1:counting:11}
साथ $ A_i = \{\sigma : \sigma(i) = i\}$, ए \omterm{def:b1:counting:objects}{परिवर्तन} में
$\bigcap_{i \in I} A_i $ प्रत्येक को ठीक करता है $ i \in I $ और दूसरे को क्रमपरिवर्तित करता है
$ n - \abs I $ स्वतंत्र रूप से अंक: $\abs{\bigcap_{i \in I} A_i} =
(n - \abs I)!$. समावेश--बहिष्करण:
\[
\Bigl|\bigcup_i A_i\Bigr|
= \sum_{k=1}^{n} (-1)^{k+1} \binom nk (n-k)!
= \sum_{k=1}^{n} (-1)^{k+1} \frac{n!}{k!} ,
\]
क्योंकि वहां हैं $\binom nk $ सबसेट $ I $ आकार का $ k $. इस तरह
\[
D_n = n! - \Bigl|\bigcup_i A_i\Bigr|
= n!\Bigl(1 - \sum_{k=1}^{n} \frac{(-1)^{k+1}}{k!}\Bigr)
= n! \sum_{k=0}^{n} \frac{(-1)^k}{k!} .
\]
दूसरी पहचान के लिए: वर्गीकृत करें क्रमपरिवर्तन $\sigma $ का
$\intint{1}{n}$ उनके निश्चित बिंदु से \omterm{def:b1:logic:sets}{तय करना} $ F(\sigma)$. एक निश्चित के लिए
$ k $-उपसमुच्चय $ F $, द क्रमपरिवर्तन साथ $ F(\sigma) = F $ बिल्कुल वही हैं
पूरक की विसंगतियाँ: $ D_{n-k}$ उनमें से. संक्षेप में कहें तो
$\binom nk $ के विकल्प $ F $ प्रत्येक के लिए $ k $:
$ n! = \sum_{k=0}^{n} \binom nk D_{n-k}$.
\end{solution}

\begin{solution}{exo:b1:counting:12}
\emph{पहली पहचान.} जोड़े गिनें $(A, a)$ कहाँ $ A \subseteq E $
($\abs E = n $) और $ a \in A $. के आकार के अनुसार $ A $: $\sum_k \binom nk k $
जोड़े. पहले चिह्नित तत्व को चुनकर: $ n $ के लिए विकल्प $ a $, फिर
शेष का कोई उपसमुच्चय $ n - 1$ पूरा करने के लिए तत्व $ A $:
$ n\,2^{n-1}$ जोड़े.

\emph{दूसरी पहचान.} त्रिगुणों की गिनती करें $(A, a, b)$ साथ $ a, b \in
A $ (संभवतः $ a = b $). आकार के अनुसार: $\sum_k k^2 \binom nk $. सीधे:
या तो $ a = b $ ($ n\,2^{n-1}$ त्रिगुण, पिछली गिनती) या $ a \neq b $
($ n(n-1)$ क्रमबद्ध विकल्प, फिर दूसरे का कोई उपसमुच्चय $ n - 2$
तत्व: $ n(n-1)\,2^{n-2}$). कुल
\[
n\,2^{n-1} + n(n-1)\,2^{n-2} = n\,2^{n-2}\,(2 + n - 1)
= n(n+1)\,2^{n-2} .
\]
\end{solution}

\begin{solution}{pb:b1:counting:1}
\textbf{1.} $ D_1 = 0$ (केवल \omterm{def:b1:counting:objects}{परिवर्तन} ठीक करता है $1$), $ D_2 = 1$
(स्वैप), $ D_3 = 2$ (एक-पंक्ति संकेतन में: $231$ और $312$). के लिए
$ n = 4$, द्वारा समूह बनाएं $\sigma(1)$: साथ $\sigma(1) = 2$ विक्षोभ
हैं $2143$, $2341$, $2413$; साथ $\sigma(1) = 3$: $3142$, $3412$,
$3421$; साथ $\sigma(1) = 4$: $4123$, $4312$, $4321$. प्रत्येक में तीन
समूह: $ D_4 = 9$.

\textbf{2.} ए \omterm{def:b1:counting:objects}{परिवर्तन} बिल्कुल के साथ $ k $ निश्चित बिंदु है
इसके निश्चित-बिंदु की पसंद से निर्धारित होता है \omterm{def:b1:logic:sets}{तय करना} $ F $ ($\binom nk $
तरीके) एक साथ पूरक के लिए इसके प्रतिबंध के साथ, जो होना चाहिए
एक हो \omterm{def:b1:counting:objects}{परिवर्तन} का $ n - k $ के साथ अंक \emph{नहीं} नियत बिन्दु
($ D_{n-k}$ तौर तरीकों)। दो विकल्प स्वतंत्र हैं और
पत्राचार है \omterm{def:b1:logic:inj}{द्विभाजित}: $ P_k(n) = \binom nk D_{n-k}$.

\textbf{3.} $ P_0(4) = D_4 = 9$; $ P_1(4) = \binom41 D_3 = 4 \times 2
= 8$; $ P_2(4) = \binom42 D_2 = 6$; $ P_3(4) = \binom43 D_1 = 0$
(तीन निश्चित बिंदु एक चौथाई को बाध्य करते हैं); $ P_4(4) = 1$. जोड़: $9 + 8 + 6
+ 0 + 1 = 24 = 4!$. कोई मेल नहीं ($9$ केस) बिल्कुल एक मैच हराता है
($8$ मामले) --- संकीर्ण रूप से।

\textbf{4.} जोड़े गिनें $(\sigma, i)$ साथ $\sigma(i) = i $. के लिए
निश्चित $ i $, द क्रमपरिवर्तन फिक्सिंग $ i $ हैं क्रमपरिवर्तन का
अन्य $ n - 1$ अंक: $(n-1)!$ उनमें से. इसलिए जोड़ियों की संख्या
है $ n \cdot (n-1)! = n!$, और यह संख्या भी है
$\sum_\sigma \abs{\mathrm{Fix}(\sigma)}$. संख्या से विभाजित करना
$ n!$ का क्रमपरिवर्तन: निश्चित बिंदुओं की औसत संख्या है
बिलकुल $1$, हरएक के लिए $ n \geq 1$.

\textbf{5.} होने देना $\sigma $ का विक्षोभ हो $\intint1{n+1}$ और
$ j = \sigma(n+1) \in \intint1n $: $ n $ संभावित मान। \emph{मामला
$\sigma(j) = n+1$:} बिन्दु $ j $ और $ n+1$ स्वैप, और $\sigma $
शेष तक ही सीमित है $ n - 1$ अंक एक मनमाना है
उनका अपमान: $ D_{n-1}$ संभावनाएं. \emph{मामला $\sigma(j)
\neq n+1$:} होने देना $ i_0 = \sigma^{-1}(n+1)$; यहाँ $ i_0 \neq j $ और
$ i_0 \leq n $. परिभाषित करना $\tau $ पर $\intint1n $ द्वारा $\tau(i) = \sigma(i)$
के लिए $ i \neq i_0$ और $\tau(i_0) = j $. तब $\tau $ एक है \omterm{def:b1:counting:objects}{परिवर्तन}
का $\intint1n $ (मूल्य $ n+1$ लापता द्वारा प्रतिस्थापित किया गया है
मूल्य $ j $), और यह एक विक्षोभ है: $\tau(i_0) = j \neq i_0$, और
$\tau(i) = \sigma(i) \neq i $ अन्यत्र. इसके विपरीत, ए से
विक्षोभ $\tau $ का $\intint1n $ और मूल्य $ j $, एक ठीक हो जाता है
$\sigma $ सेटिंग द्वारा $\sigma(n+1) = j $, $\sigma(\tau^{-1}(j)) = n+1$
और $\sigma = \tau $ अन्यत्र: एक आक्षेप, देना $ D_n $
संभावनाएं. संक्षेप में कहना $ j $: $ D_{n+1} = n(D_n + D_{n-1})$.
संख्यात्मक रूप से: $ D_5 = 4(9 + 2) = 44$, $ D_6 = 5(44 + 9) = 265$.

\textbf{6.} प्रश्न 5 से, $ D_{n+1} = nD_n + nD_{n-1}$, इसलिए
\[
u_{n+1} = D_{n+1} - (n+1)D_n = nD_n + nD_{n-1} - (n+1)D_n
= -(D_n - nD_{n-1}) = -u_n .
\]
तब से $ u_1 = D_1 - 1 \cdot D_0 = -1$, प्रेरण देता है $ u_n =
(-1)^n $, यानी\ $ D_n = nD_{n-1} + (-1)^n $ के लिए $ n \geq 1$.

\textbf{7.} प्रेरण चालू $ n $. आधार: $ D_0 = 1 = 0!\,s_0$. चरण:
मान रहा हूँ $ D_{n-1} = (n-1)!\,s_{n-1}$,
\[
D_n = nD_{n-1} + (-1)^n = n!\,s_{n-1} + (-1)^n
= n!\Bigl(s_{n-1} + \frac{(-1)^n}{n!}\Bigr) = n!\,s_n ,
\]
जो सूत्र है. कोई समावेशन--बहिष्करण का उपयोग नहीं किया गया: केवल
प्रश्न 5 की संयुक्त पुनरावृत्ति।

\textbf{8.} भाज्य द्वारा त्रिपद पुनरीक्षण:
\[
\binom nk \binom kj
= \frac{n!}{k!\,(n-k)!} \cdot \frac{k!}{j!\,(k-j)!}
= \frac{n!}{j!\,(n-j)!} \cdot \frac{(n-j)!}{(k-j)!\,(n-k)!}
= \binom nj \binom{n-j}{k-j} .
\]
अब स्थानापन्न $ a_k = \sum_j \binom kj b_j $ और दोनों का आदान-प्रदान करें
परिमित रकम:
\[
\sum_{k=0}^{n} (-1)^{n-k} \binom nk a_k
= \sum_{j=0}^{n} b_j \binom nj
\sum_{k=j}^{n} (-1)^{n-k} \binom{n-j}{k-j}
= \sum_{j=0}^{n} b_j \binom nj
\sum_{i=0}^{n-j} (-1)^{(n-j)-i} \binom{n-j}{i} .
\]
आंतरिक योग का विस्तार है $(1 + (-1))^{n-j} = 0^{n-j}$
(द्विपद प्रमेय, \cref{thm:b1:counting:binomial}): यह गायब हो जाता है
के लिए $ j < n $ और बराबर है $1$ के लिए $ j = n $. केवल $ j = n $ जीवित रहता है, और
दायीं ओर है $ b_n $, जैसा कि दावा किया गया है।

\textbf{9.} समरूपता से $\binom nk = \binom n{n-k}$, द
की पहचान \cref{exo:b1:counting:11} के रूप में पुनः लिखता है $ n! =
\sum_{k=0}^n \binom nk D_k $. प्रश्न 8 को साथ में लागू करें $ a_n = n!$ और
$ b_k = D_k $:
\[
D_n = \sum_{k=0}^{n} (-1)^{n-k} \binom nk k!
= \sum_{k=0}^{n} (-1)^{n-k} \frac{n!}{(n-k)!}
= n! \sum_{j=0}^{n} \frac{(-1)^j}{j!} ,
\]
द्वारा पुनः अनुक्रमणिका $ j = n - k $: सूत्र तीसरी बार।

\textbf{10.} $ D_n = n!\,s_n $ (प्रश्न 7), अतः
\[
\Bigl| D_n - \frac{n!}{\eu} \Bigr|
= n!\,\abs{s_n - \eu^{-1}} < \frac{n!}{(n+1)!} = \frac1{n+1} .
\]

\textbf{11.} के लिए $ n \geq 1$, $\frac1{n+1} \leq \frac12$, और
प्रश्न 10 की असमानता सख्त है: $ D_n $ दूरी पर स्थित है
$< \frac12$ से $ n!/\eu $, इसलिए यह अद्वितीय निकटतम पूर्णांक है। के लिए
$ n = 0$ सीमा केवल दूरी देती है $< 1$, और वास्तव में दावा
वहाँ विफल रहता है: $0!/\eu \approx 0.368$ निकटतम पूर्णांक है $0$, जबकि
$ D_0 = 1$.

\textbf{12.} $\eu^{-1} - s_n = \sum_{k \geq n+1} (-1)^k/k!$ एक है
कड़ाई से घटते पदों के साथ श्रृंखला को वैकल्पिक करना, इसलिए इसका संकेत है
इसके पहले कार्यकाल का संकेत $(-1)^{n+1}/(n+1)!$. इस तरह $ s_n -
\eu^{-1}$ का चिन्ह है $(-1)^n $: के लिए $ n $ यहां तक की, $ s_n > \eu^{-1}$
और $ D_n = n!\,s_n > n!/\eu $; के लिए $ n $ विषम, $ D_n < n!/\eu $.

\textbf{13.} $ D_7 = 6(265 + 44) = 6 \times 309 = 1854$; $ D_8 =
7(1854 + 265) = 7 \times 2119 = 14\,833$; $ D_9 = 8(14\,833 + 1854)
= 8 \times 16\,687 = 133\,496$; $ D_{10} = 9(133\,496 + 14\,833) =
9 \times 148\,329 = 1\,334\,961$. जाँच करना: $10!/\eu = 3\,628\,800 /
2.718281828 \approx 1\,334\,960.92$, जिसका निकटतम पूर्णांक है
$1\,334\,961$ --- और $ D_{10} > 10!/\eu $, जैसा कि प्रश्न 12 भविष्यवाणी करता है
सम के लिए $ n $.

\textbf{14.} $\abs{p_n - \eu^{-1}} = \abs{s_n - \eu^{-1}} <
\frac1{(n+1)!}$. के लिए $ n = 6$: $ p_6 = 265/720 = 0.36806$ (पांच
दशमलव), विरुद्ध $\eu^{-1} = 0.36788$; अंतर नीचे है $1/7! =
1/5040 < 2 \times 10^{-4}$. बाध्य $1/(n+1)!$ इतनी तेजी से ढह जाता है
कि प्रायिकता a के लिए पहले से ही कई दशमलव स्थानों पर निश्चित है
दर्जन पत्र: उत्तर ``के बारे में $36.8\%$'' है, प्रत्येक के लिए
व्यावहारिक उद्देश्य, से स्वतंत्र $ n $ --- का प्रसिद्ध आश्चर्य
समस्या.

\textbf{15.} प्रश्न 2 और द्वारा $ D_m = m!\,s_m $:
\[
\frac{P_k(n)}{n!} = \frac{\binom nk D_{n-k}}{n!}
= \frac{D_{n-k}}{k!\,(n-k)!} = \frac{s_{n-k}}{k!}
\;\longrightarrow\; \frac{\eu^{-1}}{k!}
\]
जैसा $ n \to \infty $ साथ $ k $ तय हो गया, तब से $ s_{n-k} \to \eu^{-1}$.
सीमा मान $\eu^{-1}/k!$ ($ k \in \N $) के वजन हैं
पैरामीटर का पॉइसन वितरण $1$.

\textbf{16.} त्रिगुणों की गिनती करें $(\sigma, i, j)$ साथ $ i \neq j $,
$\sigma(i) = i $, $\sigma(j) = j $. पहले ऑर्डर किया गया जोड़ा चुनना:
$ n(n-1)$ तौर तरीकों; क्रमपरिवर्तन दोनों को ठीक करना $ i $ और $ j $ हैं
क्रमपरिवर्तन शेष का $ n - 2$ अंक: $(n-2)!$ उनमें से.
कुल: $ n(n-1)(n-2)! = n!$. इसके बजाय संक्षेपण समाप्त करें $\sigma $ प्रथम
प्रत्येक के लिए मायने रखता है $\sigma $, अलग-अलग निश्चित के क्रमित जोड़े
अंक: $\abs{\mathrm{Fix}(\sigma)}(\abs{\mathrm{Fix}(\sigma)}-1)$.
इसलिए बताई गई पहचान; द्वारा विभाजित करना $ n!$, का औसत
$\abs{\mathrm{Fix}}(\abs{\mathrm{Fix}} - 1)$ है $1$, तो औसत
का $\abs{\mathrm{Fix}}^2$ है $1 + 1 = 2$ और भिन्नता है $2 -
1^2 = 1$.

\textbf{17.} अनुपात $1 - p_n $: के लिए $ n = 4$, $1 - \frac
9{24} = \frac{15}{24} = 0.6250$; के लिए $ n = 5$, $1 - \frac{44}{120} =
\frac{76}{120} = 0.6333$; के लिए $ n = 6$, $1 - \frac{265}{720} =
\frac{455}{720} = 0.6319$. सभी एक प्रतिशत के भीतर $1 - \eu^{-1}
\approx 0.6321$, इसके चारों ओर घूम रहा है।

\textbf{18.} सीधे:
\[
s_{n+2} - s_n = \frac{(-1)^{n+1}}{(n+1)!} +
\frac{(-1)^{n+2}}{(n+2)!}
= (-1)^{n+1}\Bigl(\frac1{(n+1)!} - \frac1{(n+2)!}\Bigr),
\]
और कोष्ठक है $> 0$. के लिए $ n $ यहां तक कि अंतर भी है
नकारात्मक: $ s_{n+2} < s_n $, इसलिए $ p_0 > p_2 > p_4 > \dots $; के लिए $ n $
अजीब है यह सकारात्मक है: $ p_1 < p_3 < p_5 < \dots $ के साथ संयुक्त
प्रश्न 12 (सम उपरोक्त)। $\eu^{-1}$, संभावनाएँ नीचे) और प्रश्न 14
(से दूरी $\eu^{-1}$ आदत है $0$): दो सीढ़ियाँ सिकुड़ती हैं
$\eu^{-1}$ उन दोनों के बीच।

\textbf{19.} एक पूर्ण ड्राइंग एक समान यादृच्छिक है \omterm{def:b1:counting:objects}{परिवर्तन},
वैध जब यह एक विक्षोभ है: संभाव्यता $ p_n \approx \eu^{-1}$.
उद्धृत तथ्य के अनुसार, सफलता मिलने तक रेखाचित्रों की औसत संख्या है
$1/p_n $, और प्रश्न 14 देता है $1/p_n \approx \eu $ एक त्रुटि तक
यह छोटे लोगों के लिए पहले से ही नगण्य है $ n $. तो एक गुप्त सांता के साथ
पुनः आरंभ करने में औसतन लगभग लागत आती है $\eu \approx 2.72$ पूर्ण
चित्र---चाहे कार्यालय है $6$ लोग या $600$.

\textbf{20.} हल करना $ j \geq 2$ और प्रश्न 5 का वर्गीकरण चालू रखें
मूल्य $\sigma(1) = j $. अगर $\sigma(j) = 1$: शेष
$ n - 2$ अंक एक मनमाना विक्षोभ लेकर आते हैं, $ D_{n-2}$ तौर तरीकों। अगर
$\sigma(j) \neq 1$: पुनर्निर्देशित करें \omterm{def:b1:logic:map}{पूर्वछवि} $ i_0 = \sigma^{-1}(1)$ को
$ j $ बिलकुल प्रश्न 5 की तरह; यह के साथ एक आक्षेप है
की अव्यवस्थाएं $ n - 1$ अंक $\{2, \dots, n\}$: $ D_{n-1}$
तौर तरीकों। कुल $ D_{n-1} + D_{n-2}$, हर किसी के लिए समान $ j $. संक्षेप
के ऊपर $ n - 1$ के मान $ j $: $ D_n = (n-1)(D_{n-1} + D_{n-2})$,
जो कारक प्रदर्शित करता है $ n - 1$: $(n-1) \mid D_n $.

\textbf{21.} दावा करना: $ D_n $ अजीब है अगर $ n $ सम है. प्रेरण का उपयोग करना
$ D_n = nD_{n-1} + (-1)^n $, यानी\ $ D_n \equiv nD_{n-1} + 1 \pmod 2$.
आधार: $ D_1 = 0$ सम है, $ n = 1$ अजीब: दावा कायम है. अगर $ n $ सम है,
$ nD_{n-1}$ सम है और $ D_n \equiv 1$: अजीब, जैसा दावा किया गया। अगर $ n $ है
फिर अजीब है $ n - 1$ सम है, इसलिए $ D_{n-1}$ परिकल्पना से अजीब है, और
$ D_n \equiv D_{n-1} + 1 \equiv 0$: यहां तक की। इंडक्शन बंद हो जाता है.

\textbf{22.} कमी $ D_n = nD_{n-1} + (-1)^n $ सापेक्ष $ n $ मारता है
पहला पद: $ D_n \equiv (-1)^n \pmod n $. के लिए $ n = 10$:
$(-1)^{10} = 1$, सचमुच $ D_{10} = 1\,334\,961$ में समाप्त होता है
अंक $1$.

\textbf{23.} के लिए $ n \geq 3$, $ D_{n-1} \geq 1$ और का विभाजन
प्रश्न 6 की पुनरावृत्ति द्वारा $ D_{n-1}$ देता है $ D_n/D_{n-1} = n +
(-1)^n/D_{n-1}$, साथ $\abs{(-1)^n/D_{n-1}} \leq 1$ और तेजी से
की ओर रुझान $0$. संगति: यदि $ D_n \approx n!/\eu $, तब
$ D_n/D_{n-1} \approx n!/(n-1)! = n $ --- कारक $\eu $ रद्द करता है
अनुपात में, और पुनरावृत्ति इसकी सटीकता की पुष्टि करती है
$1/D_{n-1}$.

\textbf{24.} (i) उत्पाद और योग नियम प्रत्येक गणना के अंतर्गत आते हैं:
प्रश्न 2 और 5 \omterm{thm:b1:logic:partition}{PARTITION} सेट का क्रमपरिवर्तन स्वतंत्र में
चरणों. (ii) दोहरी गणना से माध्य (प्रश्न 4) और प्राप्त हुआ
संख्या के साथ निश्चित बिंदुओं की संख्या का विचरण (प्रश्न 16)।
के लिए सूत्र $ D_n $ बिल्कुल भी। (iii) द्विपद प्रमेय का मूल्यांकन किया गया
वैकल्पिक आंतरिक योग $(1-1)^{n-j}$ जो द्विपद व्युत्क्रम बनाता है
कार्य (प्रश्न 8). (iv) प्रत्यावर्ती-श्रृंखला बाध्य ने परिवर्तित कर दिया
सटीक लेकिन अपारदर्शी योग $ n!\,s_n $ पारदर्शी में \omterm{def:b1:logic:statement}{कथन}
``निकटतम पूर्णांक $ n!/\eu $'' (प्रश्न 10-14).

\textbf{25.} समावेश--बहिष्करण
(\cref{ex:b1:counting:derangement} और \cref{exo:b1:counting:11})
वैचारिक प्रमाण है: यह वैकल्पिक योग की व्याख्या इस प्रकार करता है
ओवरकाउंटिंग सुधार, और यह गिनती को शब्दशः सामान्यीकृत करता है
``बुरे'' के किसी भी परिवार से परहेज करने वाले तत्व सेट. पुनरावृत्ति मार्ग
(प्रश्न 5-7) सबसे तेज़ --- रैखिक समय, सटीक पूर्णांक की गणना करता है
अंकगणित, कोई भाज्य नहीं --- और अंकगणित का स्रोत है
भाग V के तथ्य। द्विपद व्युत्क्रम (प्रश्न 8--9) स्थान देता है
एक सामान्य \omterm{def:b1:counting:objects}{परिवर्तन} के अंदर का सूत्र जो कहीं भी फिर से प्रकट होगा
पहचान की दो त्रिकोणीय प्रणालियाँ एक दूसरे के सामने हैं। और
की उपस्थिति $\eu $ सूत्र द्वारा ही सर्वोत्तम रूप से समझाया गया है: द
विक्षोभों का अनुपात आंशिक योग है $ s_n $ श्रृंखला का
के लिए $\eu^{-1}$, तो मॉन्टमॉर्ट के लिफाफे तीन दशक पहले थे
यूलर का अंकन, पहले से ही संख्या की गणना कर रहा है $\eu$.
\end{solution}
