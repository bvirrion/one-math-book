\section*{अध्याय \ref{ch:b1:fractions} --- तर्कसंगत भिन्न}

\begin{solution}{exo:b1:fractions:1}
$\dfrac{1}{X^2 - 1}$: सरल \omterm{def:b1:fractions:field}{डंडे} $\pm 1$; कवर अप:
$\dfrac{1/2}{X-1} - \dfrac{1/2}{X+1}$.

$\dfrac{X}{X^2 - 3X + 2} = \dfrac{X}{(X-1)(X-2)}$: कवर-अप देता है
$\frac{1}{1-2} = -1$ पर $1$ और $\frac{2}{2-1} = 2$ पर $2$:
$\dfrac{-1}{X-1} + \dfrac{2}{X-2}$.

$\dfrac{X^2+1}{X(X-1)}$: डिग्री है $0$, तो एक पूर्णांक है
भाग: विभाजन, $ X^2 + 1 = (X^2 - X) + (X + 1)$, इसलिए $ F = 1 +
\frac{X+1}{X(X-1)}$. शेष पर पर्दा डालना: $\frac{1}{-1} = -1$
पर $0$, $\frac{2}{1} = 2$ पर $1$:
\[
F = 1 - \frac{1}{X} + \frac{2}{X-1} .
\]
\end{solution}

\begin{solution}{exo:b1:fractions:2}
$\dfrac{1}{X(X^2+1)}$: आकार $\frac aX + \frac{bX + c}{X^2 + 1}$.
पर पर्दा डालना $0$: $ a = 1$. की सीमा $ xF $: $0 = a + b $, इसलिए $ b = -1$.
पर मूल्यांकन $ X = 1$: $\frac12 = 1 + \frac{c - 1}{2}$, इसलिए $ c = 0$:
\[
\frac{1}{X(X^2+1)} = \frac 1X - \frac{X}{X^2+1} .
\]

$\dfrac{X^3}{X^2+X+1}$: विभाजन: $ X^3 = (X^2+X+1)(X - 1) + 1$, इसलिए
\[
\frac{X^3}{X^2+X+1} = X - 1 + \frac{1}{X^2 + X + 1} ,
\]
पहले से ही वास्तविक विघटित रूप में (द्विघात में ऋणात्मक है
विभेदक)।
\end{solution}

\begin{solution}{exo:b1:fractions:3}
$\dfrac{1}{X^2(X-1)}$: आकार $\frac{a}{X^2} + \frac bX +
\frac{c}{X-1}$. डबल पर कवर-अप खंभा $0$:
$ a = \bigl[\frac{1}{X-1}\bigr]_{0} = -1$. पर पर्दा डालना $1$: $ c = 1$.
की सीमा $ xF $: $0 = b + c $, इसलिए $ b = -1$:
\[
\frac{1}{X^2(X-1)} = -\frac{1}{X^2} - \frac1X + \frac{1}{X-1} .
\]

$\dfrac{X+1}{(X-1)^3}$: साथ $ Y = X - 1$, अंश है $ Y + 2$:
\[
\frac{Y + 2}{Y^3} = \frac{1}{Y^2} + \frac{2}{Y^3}
= \frac{1}{(X-1)^2} + \frac{2}{(X-1)^3} .
\]
\end{solution}

\begin{solution}{exo:b1:fractions:4}
 \omterm{def:b1:fractions:field}{डंडे} चौथे हैं एकता की जड़ें $1, \iu, -1, -\iu $, सब
सरल. सरल-ध्रुव सूत्र के साथ $ B' = 4X^3$: गुणांक पर $ a $ है
$\frac{1}{4a^3} = \frac{a}{4a^4} = \frac a4$ (का उपयोग करते हुए $ a^4 = 1$). तो,
खत्म $\C $:
\[
\frac{1}{X^4 - 1}
= \frac{1/4}{X - 1} - \frac{1/4}{X + 1}
+ \frac{\iu/4}{X - \iu} - \frac{\iu/4}{X + \iu} .
\]
संयुग्मी जोड़ी को \omterm{def:b1:structures:group}{समूह}ीकृत करना (सामान्य हर)। $ X^2 + 1$):
$\frac{\iu}{4}\bigl(\frac{1}{X-\iu} - \frac{1}{X+\iu}\bigr) =
\frac{\iu}{4}\cdot\frac{2\iu}{X^2+1} = \frac{-1/2}{X^2+1}$. ऊपर
$\R $:
\[
\frac{1}{X^4 - 1}
= \frac{1/4}{X-1} - \frac{1/4}{X+1} - \frac{1/2}{X^2 + 1} .
\]
\emph{पर जांचें $ X = 0$:} $-1 = -\frac14 - \frac14 - \frac12$.
\end{solution}

\begin{solution}{exo:b1:fractions:5}
$\dfrac{X^2}{(X^2+1)^2} = \dfrac{(X^2 + 1) - 1}{(X^2+1)^2} =
\dfrac{1}{X^2+1} - \dfrac{1}{(X^2+1)^2}$.

इसलिए एक आदिम:
\[
\int \frac{x^2\,\dd x}{(x^2+1)^2}
= \arctan x - \frac12\Bigl(\arctan x + \frac{x}{x^2+1}\Bigr) + C
= \frac12\arctan x - \frac{x}{2(x^2+1)} + C .
\]
\end{solution}

\begin{solution}{exo:b1:fractions:6}
 \omterm{def:b1:fractions:field}{डंडे} $0, -1, \dots, -n $ सरल हैं. पर पर्दा डालना $-k $:
\[
c_k = \frac{n!}{\prod_{j \neq k} (-k + j)}
= \frac{n!}{\bigl(\prod_{j=0}^{k-1}(j - k)\bigr)
\bigl(\prod_{j=k+1}^{n}(j-k)\bigr)}
= \frac{n!}{(-1)^k k!\,(n-k)!} = (-1)^k \binom nk .
\]
इसलिए
\[
\frac{n!}{X(X+1)\cdots(X+n)}
= \sum_{k=0}^{n} \frac{(-1)^k \binom nk}{X + k} .
\]
(स्वच्छता की जाँच करें $ n = 1$: $\frac{1}{X(X+1)} = \frac1X -
\frac{1}{X+1}$.)
\end{solution}

\begin{solution}{exo:b1:fractions:7}
$ P = X^n - 1$ के पास है $ n $ सरल जड़ें $\omega^k $
(\cref{thm:b1:complex:roots}), इसलिए लघुगणक-व्युत्पन्न पहचान
का \cref{exo:b1:poly:11} पढ़ता
\[
\sum_{k=0}^{n-1} \frac{1}{X - \omega^k}
= \frac{P'(X)}{P(X)} = \frac{n X^{n-1}}{X^n - 1} .
\]
पर $ X = 2$, $ n = 4$: दाहिनी ओर $= \frac{4 \times 8}{15} =
\frac{32}{15}$. बाईं तरफ: $\frac{1}{2-1} + \frac{1}{2+1} +
\frac{1}{2 - \iu} + \frac{1}{2 + \iu} = 1 + \frac13 + \frac{4}{5} =
\frac{15 + 5 + 12}{15} = \frac{32}{15}$, का उपयोग करना $\frac{1}{2-\iu} +
\frac{1}{2+\iu} = \frac{4}{5}$.
\end{solution}

\begin{solution}{exo:b1:fractions:8}
कवर अप: $\dfrac{1}{k(k+1)(k+2)} = \dfrac{1/2}{k} - \dfrac{1}{k+1} +
\dfrac{1/2}{k+2}$. टेलीस्कोपिंग अंतर के रूप में पुनः लिखें:
\[
\frac{1}{k(k+1)(k+2)}
= \frac12\Bigl(\frac{1}{k(k+1)} - \frac{1}{(k+1)(k+2)}\Bigr),
\]
(जांचने के लिए विस्तार करें --- या दो अपघटनों को घटाएं)। सारांश:
\[
S_n = \frac12\Bigl(\frac{1}{1 \times 2} - \frac{1}{(n+1)(n+2)}\Bigr)
= \frac14 - \frac{1}{2(n+1)(n+2)}
\xrightarrow[n \to \infty]{} \frac14 .
\]
\end{solution}

\begin{solution}{exo:b1:fractions:9}
विघटित करें, के लिए $0 \leq m \leq n - 1$, अंश $\frac{X^m}{P}$
(डिग्री $ m - n \leq -1$, \omterm{def:b1:fractions:field}{डंडे} सरल): द सरल ध्रुव सूत्र देता है
\[
\frac{X^m}{P} = \sum_{i=1}^{n} \frac{x_i^m / P'(x_i)}{X - x_i} .
\]
गुणा करके $ X $ और चलो $ X \to +\infty $: बायां भाग की ओर झुकता है
की सीमा $ X^{m+1}/P $, जो है $0$ अगर $ m \leq n - 2$ और $1$ अगर $ m =
n - 1$ ($ P $ है \omterm{def:b1:poly:def}{राक्षसी} डिग्री का $ n $); दाहिनी ओर की प्रवृत्ति होती है
$\sum_i \frac{x_i^m}{P'(x_i)}$. इस तरह
\[
\sum_{i=1}^{n} \frac{x_i^{m}}{P'(x_i)} =
\begin{cases}
0 & \text{for } 0 \leq m \leq n-2,\\
1 & \text{for } m = n - 1,
\end{cases}
\]
जिसमें दोनों घोषित पहचान शामिल हैं ($ m = 0$ आवश्यक है $ n \geq
2$). (के माध्यम से व्याख्या \cref{thm:b1:poly:lagrange}: ये योग हैं
लैग्रेंज इंटरपोलेंट्स के प्रमुख गुणांक $ X^m $, और
प्रक्षेप ए \omterm{def:b1:poly:def}{बहुपद} डिग्री का $\leq n-1$ पर $ n $ अंक
इसे हूबहू पुन: प्रस्तुत करता है।)
\end{solution}

\begin{solution}{exo:b1:fractions:10}
आकार $\frac aX + \frac b{X+1} + \frac c{(X+1)^2}$. पर पर्दा डालना
$0$: $ a = 1$; दोहरे पर पर्दा डालना खंभा: $ c = \bigl[\frac1X
\bigr]_{X=-1} = -1$; की सीमा $ xF(x)$ अनंत पर: $0 = a + b $, इसलिए
$ b = -1$:
\[
\frac{1}{X(X+1)^2} = \frac1X - \frac1{X+1} - \frac1{(X+1)^2} .
\]
के लिए सारांश $ n = 1, \dots, N $: पहली दो ईंटें दूरबीन
$1 - \frac1{N+1}$, और तीसरा योगदान देता है $-\sum_{k=2}^{N+1}
\frac1{k^2}$. दे $ N \to \infty $ और उपयोग कर रहे हैं $\sum_{k\geq1}
\frac1{k^2} = \frac{\pi^2}6$:
\[
\sum_{n=1}^{\infty}\frac{1}{n(n+1)^2}
= 1 - \Bigl(\frac{\pi^2}6 - 1\Bigr) = 2 - \frac{\pi^2}6
\approx 0.355 .
\]
\end{solution}

\begin{solution}{exo:b1:fractions:11}
चर में $ Y $: $\frac1{(Y+1)(Y+4)} = \frac{1/3}{Y+1} -
\frac{1/3}{Y+4}$ (पर पर्दा डालना $-1$ और $-4$), इसलिए
\[
\frac{1}{(X^2+1)(X^2+4)}
= \frac13\,\frac1{X^2+1} - \frac13\,\frac1{X^2+4} .
\]
वैसे ही $\frac{Y}{(Y+1)(Y+4)} = \frac{-1/3}{Y+1} +
\frac{4/3}{Y+4}$, इसलिए
\[
\frac{X^2}{(X^2+1)(X^2+4)}
= -\frac13\,\frac1{X^2+1} + \frac43\,\frac1{X^2+4} .
\]
(पर जांचें $ X = 0$: $0 = \frac13(-1 + 1)$.) ये पहले से ही हैं
वास्तविक अपघटन: अघुलनशील द्विघातों पर अंश
स्थिरांक होते हैं.
\end{solution}

\begin{solution}{exo:b1:fractions:12}
\begin{enumerate}
  \item प्रत्येक अंतराल पर परहेज करना \omterm{def:b1:fractions:field}{डंडे}, $ F'(x) = -\sum_i
        \frac{c_i}{(x - p_i)^2} < 0$: सख्ती से घट रहा है। जैसा $ x
        \to \pm\infty $, हर ईंट की प्रवृत्ति होती है $0$: $ F \to 0$, से
        ऊपर पर $+\infty $ (वहां सभी ईंटें सकारात्मक हैं) और से
        नीचे पर $-\infty $. जैसा $ x \to p_i^+$, ईंट
        $\frac{c_i}{x - p_i}$ तक उड़ा देता है $+\infty $ और अन्य
        बंधे रहें: $ F \to +\infty $; वैसे ही $ F \to -\infty $ जैसा
        $ x \to p_i^-$.
  \item हल करना $\lambda > 0$. पर $\intoo{-\infty}{p_1}$: $ F $
        से घट जाती है $0^-$ को $-\infty $, इसलिए $ F < 0 < \lambda $:
        कोई समाधान नहीं. सभी के ऊपर $\intoo{p_i}{p_{i+1}}$ ($1 \leq i
        \leq r-1$): $ F $ से घट जाती है $+\infty $ को $-\infty $,
        इसलिए मूल्य लेता है $\lambda $ ठीक एक बार (मध्यवर्ती)
        मूल्य संपत्ति प्लस सख्त एकरसता)। पर
        $\intoo{p_r}{+\infty}$: $ F $ से घट जाती है $+\infty $ को
        $0^+$, फिर से बिल्कुल एक समाधान। कुल: बिलकुल $ r $
        समाधान, के साथ अंतर्संबंधित \omterm{def:b1:fractions:field}{डंडे}.
\end{enumerate}
\end{solution}

\begin{solution}{pb:b1:fractions:1}
\textbf{1.} कवर अप: $\frac1{X(X+1)} = \frac1X - \frac1{X+1}$.
दूरबीनों का योग:
\[
\sum_{n=1}^{N}\Bigl(\frac1n - \frac1{n+1}\Bigr)
= 1 - \frac1{N+1} \longrightarrow 1 .
\]

\textbf{2.} $\frac1{X(X+2)} = \frac{1/2}X - \frac{1/2}{X+2}$.
संक्षेप में, शर्तें $\frac1n $ के लिए जीवित रहें $ n = 1, 2$ और शर्तें
$-\frac1{n+2}$ के लिए जीवित रहें $ n = N-1, N $:
\[
\sum_{n=1}^{N}\frac1{n(n+2)}
= \frac12\Bigl(1 + \frac12 - \frac1{N+1} - \frac1{N+2}\Bigr)
\longrightarrow \frac34 .
\]

\textbf{3.} अगर $ F(X) = G(X) - G(X+1)$, तब $\sum_{n=1}^N F(n) =
\sum_{n=1}^N\bigl(G(n) - G(n+1)\bigr) = G(1) - G(N+1)$: सभी
मध्यवर्ती मान जोड़े में रद्द हो जाते हैं। के लिए $ F =
\frac1{X(X+1)(X+2)}$, साक्षी है $ G(X) = \frac1{2X(X+1)}$:
\[
G(X) - G(X+1)
= \frac{(X+2) - X}{2X(X+1)(X+2)} = F(X) ,
\]
इसलिए $\sum_{n=1}^N F(n) = \frac14 - \frac1{2(N+1)(N+2)} \to
\frac14$, का मान है \cref{exo:b1:fractions:8}.

\textbf{4.} सामान्य विभाजक के ऊपर दाहिना भाग रखें
$ X(X+1)\cdots(X+k)$:
\[
\frac1k\cdot\frac{(X + k) - X}{X(X+1)\cdots(X+k)}
= \frac{1}{X(X+1)\cdots(X+k)} ,
\]
जो पहचान है. इसलिए $ F_k(X) = \frac1k\bigl(G_k(X) -
G_k(X+1)\bigr)$ साथ $ G_k(X) = \frac1{X(X+1)\cdots(X+k-1)}$, और
प्रश्न 3 का तंत्र देता है
\[
\sum_{n=1}^{N}\frac1{n(n+1)\cdots(n+k)}
= \frac1k\Bigl(\frac1{k!} - G_k(N+1)\Bigr)
\longrightarrow \frac1{k\cdot k!} ,
\]
तब से $ G_k(1) = \frac1{k!}$ और $ G_k(N+1) \to 0$. के लिए $ k = 2$:
$\frac1{2\cdot2} = \frac14$, मिलान प्रश्न 3.

\textbf{5.} \cref{exo:b1:fractions:6} देता है
$\frac{n!}{X(X+1)\cdots(X+n)} = \sum_{j=0}^n
\frac{(-1)^j\binom nj}{X + j}$. पर मूल्यांकन करें $ X = 1$: बाईं ओर
है $\frac{n!}{(n+1)!} = \frac1{n+1}$, दाहिनी ओर है
$\sum_j(-1)^j\binom nj\frac1{1+j}$: पहली पहचान. पर $ X = 2$:
बाईं ओर है $\frac{n!}{2\cdot3\cdots(n+2)} =
\frac{n!\cdot1}{(n+2)!} = \frac1{(n+1)(n+2)}$, दाहिनी ओर
$\sum_j(-1)^j\binom nj\frac1{2+j}$: दूसरी पहचान.

\textbf{6.}  \omterm{def:b1:fractions:field}{डंडे} $\omega^k $ सरल हैं, और सरल ध्रुव
का सूत्र \cref{met:b1:fractions:compute} गुणांक देता है
\[
\frac{1}{\bigl(nX^{n-1}\bigr)_{X = \omega^k}}
= \frac{1}{n\,\omega^{k(n-1)}}
= \frac{\omega^k}{n\,\omega^{kn}} = \frac{\omega^k}n ,
\]
का उपयोग करते हुए $\omega^{kn} = 1$. इस तरह $\frac1{X^n-1} = \frac1n\sum_k
\frac{\omega^k}{X - \omega^k}$.

\textbf{7.} के लिए $ n = 2$ ($\omega = -1$): $\frac12\bigl(
\frac1{X-1} - \frac1{X+1}\bigr) = \frac12\cdot\frac{2}{X^2-1} =
\frac1{X^2-1}$: सही। गुणांकों का योग है
$\frac1n\sum_k\omega^k = 0$ के लिए $ n \geq 2$
(\cref{prop:b1:complex:sumroots}). उनको जरूर: $ x\,F(x) \to
\sum_k c_k $ जैसा $ x \to \infty $ सरल के साथ किसी भी अपघटन के लिए
\omterm{def:b1:fractions:field}{डंडे}, जबकि यहाँ $ xF(x) = \frac{x}{x^n-1} \to 0$ तब से $ n \geq
2$.

\textbf{8.} के लिए कवर-अप $\frac{X^{n-1}}{X^n - 1}$ पर $\omega^k $:
$\frac{A(\omega^k)}{B'(\omega^k)} =
\frac{\omega^{k(n-1)}}{n\omega^{k(n-1)}} = \frac1n $, इसलिए
$\frac{X^{n-1}}{X^n-1} = \frac1n\sum_k\frac1{X - \omega^k}$ ---
की पहचान \cref{exo:b1:fractions:7} दोबारा।

\textbf{9.} साथ $ c = \omega^k $, $\conj c = \omega^{n-k}$ और
$ c\conj c = 1$, $ c + \conj c = 2\cos\theta_k $:
\[
\frac{c}{X - c} + \frac{\conj c}{X - \conj c}
= \frac{c(X - \conj c) + \conj c(X - c)}
{(X - c)(X - \conj c)}
= \frac{2\cos\theta_k\,X - 2}{X^2 - 2\cos\theta_k\,X + 1} .
\]
समूहन $ k $ साथ $ n - k $ प्रश्न 6 में: विषम के लिए $ n $,
\[
\frac1{X^n - 1} = \frac1n\Biggl(\frac1{X-1} +
\sum_{k=1}^{(n-1)/2}
\frac{2\cos\theta_k X - 2}{X^2 - 2\cos\theta_k X + 1}\Biggr) ;
\]
एक जैसे के लिए $ n $, अतिरिक्त स्व-युग्मित खंभा $\omega^{n/2} = -1$
योगदान $\frac{-1}{X+1}$ कोष्ठक और युग्म के अंदर
योग तक चलता है $\frac n2 - 1$.

\textbf{10.} $ n = 4$: $\theta_1 = \frac\pi2$, $\cos\theta_1 = 0$,
अतः युग्म पद है $\frac{-2}{X^2+1}$ और
\[
\frac1{X^4-1} = \frac14\Bigl(\frac1{X-1} - \frac1{X+1}
- \frac2{X^2+1}\Bigr) ,
\]
का विघटन \cref{exo:b1:fractions:4}.

\textbf{11.} $ P = \prod_{k=1}^{n-1}(X - \omega^k)$ (विभाजित करना $ X^n -
1$ द्वारा $ X - 1$), तो द्वारा \cref{exo:b1:poly:11}, $\frac{P'}P =
\sum_{k=1}^{n-1}\frac1{X-\omega^k}$. पर $ X = 1$: $ P(1) = n $ और
$ P'(1) = \sum_{j=1}^{n-1}j = \frac{n(n-1)}2$, कहाँ से
\[
\sum_{k=1}^{n-1}\frac1{1 - \omega^k}
= \frac{P'(1)}{P(1)} = \frac{n-1}2 .
\]

\textbf{12.} अर्ध-कोण: $1 - \eu^{\iu\theta} =
-2\iu\sin\frac\theta2\,\eu^{\iu\theta/2}$, इसलिए
\[
\frac1{1 - \eu^{\iu\theta}}
= \frac{\eu^{-\iu\theta/2}}{-2\iu\sin\frac\theta2}
= \frac{\iu\cos\frac\theta2 + \sin\frac\theta2}
{2\sin\frac\theta2}
= \frac12 + \frac\iu2\,
\frac{\cos\frac\theta2}{\sin\frac\theta2} .
\]
साथ $\theta = \theta_k = \frac{2k\pi}n $: $\frac1{1-\omega^k} =
\frac12 + \frac\iu2\cot\frac{k\pi}n $. संक्षेप में कहना $ k = 1, \dots,
n-1$ और वास्तविक मूल्य से तुलना करना $\frac{n-1}2$ का
प्रश्न 11: वास्तविक भाग पहले से ही हर चीज़ का हिसाब रखते हैं, इसलिए
$\sum_k\cot\frac{k\pi}n = 0$ --- समरूपता के रूप में $\cot\frac{(n-k)
\pi}n = -\cot\frac{k\pi}n $ भी दिखाता है.

\textbf{13.} फर्क $\frac{P'}P = \sum_k\frac1{X -
\omega^k}$:
\[
\frac{P''}P - \Bigl(\frac{P'}P\Bigr)^2
= -\sum_{k=1}^{n-1}\frac1{(X - \omega^k)^2} .
\]
पर $ X = 1$: $ P''(1) = \sum_{j=2}^{n-1}j(j-1) = 2\binom n3 =
\frac{n(n-1)(n-2)}3$ (हॉकी-स्टिक पहचान, या प्रेरण), तो
\[
\sum_{k=1}^{n-1}\frac1{(1 - \omega^k)^2}
= \Bigl(\frac{n-1}2\Bigr)^2 - \frac{(n-1)(n-2)}3
= \frac{(n-1)\bigl(3(n-1) - 4(n-2)\bigr)}{12}
= \frac{(n-1)(5-n)}{12} .
\]

\textbf{14.} वर्ग प्रश्न 12 का सूत्र, के साथ $ c_k =
\cot\frac{k\pi}n $:
\[
\frac1{(1-\omega^k)^2}
= \Bigl(\frac12 + \frac\iu2 c_k\Bigr)^2
= \frac14 - \frac{c_k^2}4 + \frac\iu2\,c_k .
\]
संक्षेपण एवं प्रयोग $\sum c_k = 0$ (प्रश्न 12) और प्रश्न 13:
$\frac{n-1}4 - \frac14\sum_k c_k^2 = \frac{(n-1)(5-n)}{12}$, इसलिए
\[
\sum_{k=1}^{n-1}\cot^2\frac{k\pi}n
= (n-1) - \frac{(n-1)(5-n)}3 = \frac{(n-1)(n-2)}3 .
\]
तब $\frac1{\sin^2t} = 1 + \cot^2t $ देता है
$\sum_k\frac1{\sin^2\frac{k\pi}n} = (n-1) +
\frac{(n-1)(n-2)}3 = \frac{n^2-1}3$.

\textbf{15.} $ n = 3$: $\cot^2\frac\pi3 + \cot^2\frac{2\pi}3 =
\frac13 + \frac13 = \frac23 = \frac{2\cdot1}3$; और
$\frac1{\sin^2} $ का योग है $\frac43 + \frac43 = \frac83 =
\frac{9-1}3$. $ n = 4$: $1 + 0 + 1 = 2 = \frac{3\cdot2}3$; और $2
+ 1 + 2 = 5 = \frac{16-1}3$. दोनों सूत्र जाँचते हैं।

\textbf{16.} के लिए $ t \in \intoo0{\frac\pi2}$, शास्त्रीय
तुलना $\sin t < t < \tan t $ (क्षेत्र या उत्तलता तर्क,
हाई स्कूल से परिचित) पैदावार, व्युत्क्रम लेते हुए, $\cot t <
\frac1t < \frac1{\sin t}$, वहां तीनों पॉजिटिव; चुकता करना
व्यवस्था को सुरक्षित रखता है. साथ $ t = \frac{k\pi}n $, $1 \leq k \leq m $,
$ n = 2m+1$ (इसलिए $ t < \frac\pi2$):
\[
\cot^2\frac{k\pi}n < \frac{n^2}{k^2\pi^2} <
\frac1{\sin^2\frac{k\pi}n} .
\]

\textbf{17.} समरूपता से $\cot^2\frac{(n-k)\pi}n =
\cot^2\frac{k\pi}n $ और इसी तरह के लिए $\sin^2$, प्रश्न 14 का योग
आधा: $\sum_{k=1}^{m}\cot^2\frac{k\pi}n = \frac{(n-1)(n-2)}6 =
\frac{m(2m-1)}3$ और $\sum_{k=1}^m\frac1{\sin^2\frac{k\pi}n} =
\frac{n^2-1}6 = \frac{2m(m+1)}3$. सारांश प्रश्न 16 समाप्त $ k =
1, \dots, m $ और से गुणा करना $\frac{\pi^2}{n^2}$:
\[
\frac{\pi^2}{(2m+1)^2}\cdot\frac{m(2m-1)}3
\;<\; \sum_{k=1}^{m}\frac1{k^2} \;<\;
\frac{\pi^2}{(2m+1)^2}\cdot\frac{2m(m+1)}3 .
\]
दोनों सीमाएँ प्रवृत्त होती हैं $\frac{\pi^2}6$ जैसा $ m \to \infty $ (अनुपात
$\frac{m(2m-1)}{(2m+1)^2}$ और $\frac{2m(m+1)}{(2m+1)^2}$ दोनों
करते हैं $\frac12$), इसलिए निचोड़ने से आंशिक रकम बढ़ती है
अभिसरण और
\[
\sum_{k=1}^{\infty}\frac1{k^2} = \frac{\pi^2}6 .
\]

\textbf{18.} वर्ग $\frac1{X^2-1} = \frac12\bigl(\frac1{X-1} -
\frac1{X+1}\bigr)$:
\[
\frac1{(X^2-1)^2} = \frac14\Bigl(\frac1{(X-1)^2} +
\frac1{(X+1)^2}\Bigr) - \frac12\cdot\frac1{(X-1)(X+1)} ,
\]
और क्रॉस टर्म को फिर से विघटित करें $\frac1{(X-1)(X+1)} =
\frac12\bigl(\frac1{X-1} - \frac1{X+1}\bigr)$ कहा गया प्राप्त करने के लिए
प्रपत्र. पर $ X = 0$: बाईं तरफ $1$; दाहिनी ओर $\frac14(1 + 1) -
\frac14(-1 - 1) = \frac12 + \frac12 = 1$.

\textbf{19.} योग प्रश्न 18 ओवर $ n \geq 2$. साथ $ S =
\sum_{k\geq1}\frac1{k^2} = \frac{\pi^2}6$: $\sum_{n\geq2}
\frac1{(n-1)^2} = S $; $\sum_{n\geq2}\frac1{(n+1)^2} = S - 1 -
\frac14$; और गैप-टू टेलीस्कोप $\sum_{n\geq2}\bigl(
\frac1{n-1} - \frac1{n+1}\bigr) = 1 + \frac12 = \frac32$. इस तरह
\[
\sum_{n=2}^{\infty}\frac1{(n^2-1)^2}
= \frac14\Bigl(2S - \frac54\Bigr) - \frac14\cdot\frac32
= \frac S2 - \frac{11}{16}
= \frac{\pi^2}{12} - \frac{11}{16} \approx 0.135 .
\]
संख्यात्मक रूप से: $\frac19 + \frac1{64} + \frac1{225} + \frac1{576} +
\frac1{1225} + \dots \approx 0.1111 + 0.0156 + 0.0044 + 0.0017 +
0.0008 + \dots \approx 0.135$: सुसंगत।

\textbf{20.} प्रश्न 8 की पहचान में अंतर करें
$\sum_k\frac1{X-\omega^k} = \frac{nX^{n-1}}{X^n-1}$:
\[
\sum_{k=0}^{n-1}\frac1{(X - \omega^k)^2}
= \frac{n^2X^{2n-2} - n(n-1)X^{n-2}(X^n - 1)}{(X^n - 1)^2} .
\]
पर जांचें $ n = 2$, $ X = 2$: दाहिनी ओर $\frac{4\cdot4 -
2\cdot1\cdot3}{9} = \frac{10}9$; बाईं तरफ $\frac1{(2-1)^2} +
\frac1{(2+1)^2} = 1 + \frac19 = \frac{10}9$.

\textbf{21.} $\frac1{\binom nk} = \frac{k!\,(n-k)!}{n!} =
\frac{k!}{(n-k+1)(n-k+2)\cdots n}$, का एक उत्पाद $ k $ लगातार
नीचे पूर्णांक. स्थानापन्न $ j = n - k + 1$ (इसलिए $ j $ पर चलाता है $\N^*$ जैसा $ n $ से चलती है $ k $):
\[
\sum_{n=k}^{\infty}\frac1{\binom nk}
= k!\sum_{j=1}^{\infty}\frac1{j(j+1)\cdots(j+k-1)}
= \frac{k!}{(k-1)\,(k-1)!} = \frac{k}{k-1} ,
\]
प्रश्न 4 के साथ लागू किया गया $ k - 1$ की जगह $ k $ (तब से वैध
$ k - 1 \geq 1$).

\textbf{22.} के लिए $ k = 3$: $\frac1{\binom n3} =
\frac6{(n-2)(n-1)n}$, अतः प्रश्न 3 के आंशिक योग (स्थानांतरित) से,
\[
\sum_{n=3}^{N}\frac1{\binom n3}
= 6\sum_{j=1}^{N-2}\frac1{j(j+1)(j+2)}
= 6\Bigl(\frac14 - \frac1{2(N-1)N}\Bigr)
= \frac32 - \frac3{(N-1)N}
\longrightarrow \frac32 ,
\]
मूल्य $\frac k{k-1} = \frac32$ प्रश्न 21 का.

\textbf{23.} $ n = 6$: जोड़े $ k = 1$ ($\theta_1 = \frac\pi3$,
$2\cos\theta_1 = 1$) और $ k = 2$ ($\theta_2 = \frac{2\pi}3$,
$2\cos\theta_2 = -1$), प्लस असली \omterm{def:b1:fractions:field}{डंडे} $\pm1$:
\[
\frac1{X^6-1} = \frac16\Bigl(\frac1{X-1} - \frac1{X+1}
+ \frac{X - 2}{X^2 - X + 1}
+ \frac{-X - 2}{X^2 + X + 1}\Bigr) .
\]
पर $ X = 0$: बाईं तरफ $-1$; दाहिनी ओर $\frac16(-1 - 1 - 2 - 2)
= -1$: सही।

\textbf{24.} (i) विशिष्टता प्रत्येक पहचान को वैध बनाती है
गुणांक --- कवर-अप, संयुग्म-जोड़ी समूहन
प्रश्न 9, और विभेदन युक्तियाँ (प्रश्न 13, 20)
सब इसी पर टिके हैं. (ii) एकता की जड़ें की आपूर्ति की \omterm{def:b1:fractions:field}{डंडे} का $ X^n -
1$, उनकी समरूपता ($ k \leftrightarrow n-k $), और
प्रश्न 12 का अर्ध-कोण बीजगणित
(\cref{met:b1:complex:trig}). (iii) लघुगणकीय व्युत्पन्न
$\frac{P'}P = \sum\frac{m_i}{X-a_i}$ के बारे में परिवर्तित जानकारी
की जड़ें $ P = 1 + X + \dots + X^{n-1}$ संख्यात्मक में
प्रश्न 11 और 13 का योग --- भाग II और के बीच का बंधन
तृतीय.

\textbf{25.} अपघटन एक विशुद्ध \omterm{pb:b1:logic:1}{बीजगणितीय} पहचान है,
चर के प्रत्येक मान के लिए एक ही बार में सत्य; वही बनाता है
यह एक इंजन है. पूर्णांकों को प्रतिस्थापित करने और उनका योग करने से यह बदल गया
दूरबीन (भाग I); प्रतिस्थापन एकता की जड़ें और \omterm{def:b1:structures:group}{समूह}ीकरण
संयुग्मों ने इसे त्रिकोणमितीय सर्वसमिकाओं (भाग II और) में बदल दिया
III); और केवल अंतिम चरण में ही विश्लेषण दर्ज किया गया --- a
दो बंद रूपों के बीच निचोड़ें --- वितरित करने के लिए $\frac{\pi^2}6$,
ए \omterm{def:b1:logic:statement}{कथन} कोई भी परिमित प्रतिस्थापन नहीं पहुँच सका। का यह विभाजन
श्रम (बीजगणित सटीक परिमित पहचान उत्पन्न करता है, विश्लेषण पास होता है
सीमा तक) के लिए टेम्पलेट है \cref{ch:b1:series}, कहाँ
दूरबीन और तुलना व्यवस्थित हो जाती है, और इसके लिए
\cref{ch:b1:integration}, जहां प्रत्येक ईंट एक आदिम प्राप्त करती है
और वही अपघटन योगों के बजाय अभिन्नों की गणना करते हैं।
\end{solution}
