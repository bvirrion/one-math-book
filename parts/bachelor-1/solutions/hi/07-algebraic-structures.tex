\section*{अध्याय \ref{ch:b1:structures} --- बीजगणितीय संरचनाएँ}

\begin{solution}{exo:b1:structures:1}
\emph{स्थिरता:} $x * y = 1 \iff x + y - xy = 1 \iff (1-x)(1-y) = 0$,
के लिए असंभव $ x, y \neq 1$. वास्तव में प्रमुख पहचान है
\[
1 - x * y = (1 - x)(1 - y):
\]
 \omterm{def:b1:logic:map}{नक्शा} $\varphi(x) = 1 - x $ भेजता है $(E, *)$ को $(\R^*, \times)$ साथ
$\varphi(x * y) = \varphi(x)\varphi(y)$ --- ए \omterm{def:b1:logic:inj}{द्विभाजित} \omterm{def:b1:structures:morphism}{आकारिता}. सब
अभिगृहीत अब परिवहन करते हैं: साहचर्यता और क्रमविनिमेयता का अनुसरण होता है
के $\times $; पहचान है $\varphi^{-1}(1) = 0$ (जाँच करना: $ x *
0 = x $); का उलटा $ x $ है $\varphi^{-1}\bigl((1-x)^{-1}\bigr) =
1 - \frac{1}{1-x} = \frac{x}{x - 1}$ (जो है $\neq 1$). इसलिए $(E, *)$
एक \omterm{def:b1:structures:group}{एबेलियन} \omterm{def:b1:structures:group}{समूह}.
\end{solution}

\begin{solution}{exo:b1:structures:3}

\tau_{23}$ (दो नामित बिंदुओं की अदला-बदली), और चक्र $ c =
(1\,2\,3)$ (अर्थात्\ $1 \mapsto 2 \mapsto 3 \mapsto 1$) और $ c^2 =

\begin{center}

$\tau_{13}$ & $\tau_{23}$ \\
\hline
$\mathrm{id}$ & $\mathrm{id}$ & $ c $ & $ c^2$ & $\tau_{12}$ &
$\tau_{13}$ & $\tau_{23}$ \\
$ c $ & $ c $ & $ c^2$ & $\mathrm{id}$ & $\tau_{13}$ & $\tau_{23}$ &
$\tau_{12}$ \\
$ c^2$ & $ c^2$ & $\mathrm{id}$ & $ c $ & $\tau_{23}$ & $\tau_{12}$ &
$\tau_{13}$ \\
$\tau_{12}$ & $\tau_{12}$ & $\tau_{23}$ & $\tau_{13}$ & $\mathrm{id}$

$\tau_{13}$ & $\tau_{13}$ & $\tau_{12}$ & $\tau_{23}$ & $ c $ &
$\mathrm{id}$ & $ c^2$ \\
$\tau_{23}$ & $\tau_{23}$ & $\tau_{13}$ & $\tau_{12}$ & $ c^2$ & $ c $ &

\end{tabular}
\end{center}
गैर-यात्रा जोड़ी: $\tau_{12}\tau_{13} = c^2$ जबकि

\end{solution}

\begin{solution}{exo:b1:structures:6}

$ xy^{-1}$ दोनों में $ H $ और $ K $. संघ: यदि $ H \subseteq K $ संघ है

K $, तब $ h = (hk)k^{-1} \in K $: विरोधाभास। इसलिए $ H \cup K $ नहीं है कोई \omterm{def:b1:structures:subgroup}{उपसमूह}.
\end{solution}

\begin{solution}{exo:b1:structures:8}
की इकाइयाँ $\Z/18\Z $: कक्षाएं \omterm{cor:b1:arith:bezout}{सह अभाज्य} को $18 = 2 \times 3^2$:
$\overline 1, \overline 5, \overline 7, \overline{11}, \overline{13},
\overline{17}$. का उलटा $\overline 5$: $5 \times 11 = 55 = 3\times
18 + 1$, इसलिए $\overline 5^{-1} = \overline{11}$.

$\overline 5 x = \overline 7$: गुणा करके $\overline{11}$: $ x =
\overline{77} = \overline 5$ (तब से $77 = 4\times 18 + 5$). अनोखा
समाधान.

$\overline 6 x = \overline 3$: समीकरण $6x \equiv 3 \pmod{18}$
मतलब $18 \mid 6x - 3$. लेकिन $6x - 3 = 3(2x - 1)$ अजीब है, जबकि $18$
सम है: एक सम संख्या किसी विषम संख्या को विभाजित नहीं कर सकती। कोई समाधान नहीं.

$\overline 6 x = \overline{12}$: $6x \equiv 12 \pmod{18} \iff x
\equiv 2 \pmod 3$: समाधान $ x \in \{\overline 2, \overline 5,
\overline 8, \overline{11}, \overline{14}, \overline{17}\}$ --- छह
उनमें से.
\end{solution}

\begin{solution}{exo:b1:structures:9}
$\Z[\sqrt 2]$ रोकना $0$ और $1$, और घटाव के तहत स्थिर है
और उत्पाद:
\[
(a + b\sqrt 2)(c + d\sqrt 2) = (ac + 2bd) + (ad + bc)\sqrt 2 ,
\]
तो यह का एक उपरिंग है $\R $ (कम्यूटेटिविटी, एसोसिएटिविटी,
वितरणशीलता विरासत में मिली है)। इकाई: $(1 + \sqrt 2)(-1 + \sqrt 2) =
2 - 1 = 1$, इसलिए $1 + \sqrt 2$ व्युत्क्रम के साथ व्युत्क्रमणीय है $\sqrt 2 - 1
\in \Z[\sqrt 2]$. इसकी शक्तियां $(1 + \sqrt 2)^n $ सख्ती से हैं
बढ़ रहा है (आधार है $> 1$), इसलिए जोड़ीवार अलग है, और प्रत्येक है
एक इकाई ($\bigl((1+\sqrt2)^n\bigr)^{-1} = (\sqrt 2 - 1)^n $): द \omterm{def:b1:structures:group}{समूह}
इकाइयों की संख्या अनंत है.
\end{solution}

\begin{solution}{exo:b1:structures:10}
$ x + x = (x + x)^2 = x^2 + x^2 + x^2 + x^2 = 4x^2 = 4x $ --- इसलिए $2x =
4x $, देना $2x = 0$, यानी\ $ x + x = 0$ (प्रत्येक तत्व अपना है
योगात्मक व्युत्क्रम)। तब
\[
x + y = (x+y)^2 = x^2 + xy + yx + y^2 = x + xy + yx + y ,
\]
इसलिए $ xy + yx = 0$, यानी\ $ xy = -yx = yx $ (का उपयोग करते हुए $-z = z $). इस तरह $ A $ है
क्रमविनिमेय

उदाहरण: चालू $\mathcal{P}(E)$, परिभाषित करना $ A + B = (A \cup B) \setminus (A
\cap B)$ (सममित अंतर) और $ A \times B = A \cap B $. एक
जाँच: $(\mathcal{P}(E), +)$ एक \omterm{def:b1:structures:group}{एबेलियन} \omterm{def:b1:structures:group}{समूह} पहचान के साथ
$\emptyset $ और प्रत्येक \omterm{def:b1:logic:sets}{तय करना} इसका अपना उलटा; $\cap $ सहयोगी है,
क्रमविनिमेय, पहचान के साथ $ E $; वितरणशीलता $ A \cap (B + C) = (A
\cap B) + (A \cap C)$ धारण करता है (एक तत्व बाईं ओर स्थित है यदि यह है
में है $ A $ और बिल्कुल एक में $ B, C $). और $ A \cap A = A $: हर
आवश्यकतानुसार तत्व निष्क्रिय है।
\end{solution}

\begin{solution}{exo:b1:structures:11}
के लिए परिकल्पना लिखें $ n $ और $ n+1$:
\[
(xy)^{n+1} = x^{n+1} y^{n+1}
\quad\text{and}\quad
(xy)^{n+1} = (xy)(xy)^n = xy\,x^n y^n .
\]
बराबर करना: $ x^{n+1} y^{n+1} = x\,y\,x^n\,y^n $; रद्द करना $ x $ बाईं ओर
और $ y^n $ दायीं तरफ: $ x^n y = y x^n $. वही गणना एक
उच्च डिग्री ($ n+1$ और $ n+2$) देता है $ x^{n+1} y = y x^{n+1}$. तब
\[
y\,x^{n+1} = x^{n+1} y = x\,(x^n y) = x\,y\,x^n ,
\]
और रद्द करना $ x^n $ के दाईं ओर $ y x \cdot x^n = x y \cdot x^n $:
$ yx = xy $. इसलिए $ G $ है \omterm{def:b1:structures:group}{एबेलियन}.
\end{solution}

\begin{solution}{exo:b1:structures:12}
\begin{enumerate}
  \item होने देना $ f \colon \Z \to \Z $ additive हो और $ a = f(1)$. द्वारा
        प्रेरण $ f(k) = ka $ के लिए $ k \in \N $, और $ f(-k) = -f(k) =
        -ka $: इसलिए $ f $ से गुणा है $ a $. इसके विपरीत
        प्रत्येक \omterm{def:b1:logic:map}{नक्शा} $ k \mapsto ak $ एक है \omterm{def:b1:structures:morphism}{आकारिता}: द रूपवाद
        $(\Z,+) \to (\Z,+)$ वास्तव में a से गुणन हैं
        निश्चित पूर्णांक.
  \item होने देना $ f \colon \Q \to \Z $ एक हो \omterm{def:b1:structures:morphism}{आकारिता}, $ x \in \Q $ और
        $ n \in \N^*$. तब
        \[
        f(x) = f\Bigl(\underbrace{\tfrac xn + \dots +
        \tfrac xn}_{n}\Bigr) = n\,f\Bigl(\frac xn\Bigr) ,
        \]
        तो पूर्णांक $ f(x)$ प्रत्येक से विभाज्य है $ n \geq 1$.
        एकमात्र ऐसा पूर्णांक है $0$: $ f \equiv 0$.
\end{enumerate}
\end{solution}

\begin{solution}{pb:b1:structures:1}
\textbf{1.} ए \omterm{def:b1:counting:objects}{परिवर्तन} का आक्षेप है $\intint1n $, यानी\ एक
$ n $-की व्यवस्था $ n $ वस्तुएँ: वहाँ हैं $ n!$ उनमें से
(\cref{thm:b1:counting:counts}). साथ $\sigma = [2,3,1]$, $\tau =
[1,3,2]$: $\sigma\tau $ भेजता है $1 \mapsto 1 \mapsto 2$, $2 \mapsto 3
\mapsto 1$, $3 \mapsto 2 \mapsto 3$: $\sigma\tau = [2,1,3]$; और
$\tau\sigma $ भेजता है $1 \mapsto 2 \mapsto 3$, $2 \mapsto 3 \mapsto
2$, $3 \mapsto 1 \mapsto 1$: $\tau\sigma = [3,2,1] \neq
\sigma\tau $.

\textbf{2.} होने देना $\gamma, \gamma'$ असंयुक्त समर्थन है $ S, S'$.
के लिए $ x \in S $: $\gamma'(x) = x $ और $\gamma(x) \in S $, इसलिए
$\gamma\gamma'(x) = \gamma(x) = \gamma'\gamma(x)$. सममित रूप से
के लिए $ x \in S'$; और दोनों पक्ष प्रत्येक को ठीक करते हैं $ x \notin S \cup S'$. इसलिए
$\gamma\gamma' = \gamma'\gamma $.

\textbf{3.} के लिए $ i \in \intint1n $, मूल्य $ i, \sigma(i),
\sigma^2(i), \dots $ एक में रहते हैं परिमित सेट, इसलिए $\sigma^a(i) =
\sigma^b(i)$ कुछ के लिए $ a < b $; \omterm{def:b1:logic:inj}{इंजेक्शन} देता है $\sigma^{b-a}(i)
= i $: अनुक्रम वापस आता है $ i $. पुकारना \emph{कक्षा} का $ i $ द
सेट $\{i, \sigma(i), \dots, \sigma^{k-1}(i)\}$ साथ $ k \geq 1$
न्यूनतम ऐसा कि $\sigma^k(i) = i $. दो कक्षाएँ एक में मिल रही हैं
बिंदु संपाती है (दोनों आगे हैं $\sigma $-उसकी छवियाँ
बिंदु), इसलिए कक्षाएँ \omterm{thm:b1:logic:partition}{PARTITION} $\intint1n $; $\sigma $ प्रत्येक पर कार्य करता है
आकार की कक्षा $ k \geq 2$ के रूप में $ k $-चक्र $(i\ \sigma(i)\ \cdots\
\sigma^{k-1}(i))$ और सिंगलटन को ठीक करता है। इन्हीं का उत्पाद
असंयुक्त चक्र से सहमत हैं $\sigma $ हर जगह. विशिष्टता: में
असंयुक्त चक्रों में कोई भी विघटन, चक्र के माध्यम से $ i $ अवश्य
हो $(i\ \sigma(i)\ \cdots)$ ---चक्र को मजबूर होना पड़ता है
अपनी प्रेरित क्रिया के साथ परिक्रमा करता है।

\textbf{4.} कक्षाओं का अनुसरण: $1 \to 4 \to 2 \to 1$, $3 \to 5
\to 3$, $6 \to 7 \to 8 \to 6$:
\[
\sigma = (1\ 4\ 2)(3\ 5)(6\ 7\ 8) .
\]
अगर $\sigma = \gamma_1\cdots\gamma_r $ के असंयुक्त चक्रों के साथ
लंबाई $ k_1, \dots, k_r $, रूपान्तरण (प्रश्न 2) देता है
$\sigma^m = \gamma_1^m\cdots\gamma_r^m $, और चूंकि समर्थन करता है
असंयुक्त हैं, $\sigma^m = \mathrm{id}$ यदि प्रत्येक $\gamma_i^m =
\mathrm{id}$ आईएफएफ $ k_i \mid m $ सभी के लिए $ i $ (ए $ k $-साइकिल है \omterm{def:b1:structures:order}{आदेश}
$ k $: $\gamma^m $ भेजता है $ a_1$ को $ a_{1 + (m \bmod k)}$). सबसे कम
ऐसे $ m $ है $\operatorname{lcm}(k_1, \dots, k_r)$. यहाँ:
$\operatorname{lcm}(3, 2, 3) = 6$.

\textbf{5.} प्रत्येक बिंदु पर दाहिनी ओर, सबसे दाहिनी ओर लागू करें
कारक पहले. $ a_1 \mapsto a_2$ द्वारा $(a_1\ a_2)$, फिर हर बाद में
कारक ठीक करता है $ a_2$: जाल $ a_1 \mapsto a_2$. के लिए $2 \leq i < k $:
$ a_i $ तक अछूता है $(a_1\ a_i)$ को भेजता है $ a_1$, और
बिल्कुल अगला कारक $(a_1\ a_{i+1})$ भेजता है $ a_1$ को $ a_{i+1}$, के बाद
जो कुछ भी इसे स्थानांतरित नहीं करता है: नेट $ a_i \mapsto a_{i+1}$. अंत में $ a_k $
सभी कारकों द्वारा तय किया जाता है, लेकिन सबसे बाईं ओर, जो इसे भेजता है $ a_1$.
बिल्कुल यही चक्र है. चूंकि प्रत्येक \omterm{def:b1:counting:objects}{परिवर्तन} का एक उत्पाद है
चक्र (प्रश्न 3), यह स्थानान्तरण का एक उत्पाद है। के लिए
प्रश्न 4 $\sigma $:
\[
\sigma = (1\ 2)(1\ 4)\;(3\ 5)\;(6\ 8)(6\ 7),
\]
पाँच स्थानान्तरण।

\textbf{6.} प्रेरण चालू $ b - a $. के लिए $ b = a + 1$ पहचान है
तुच्छ ($1 = 2\cdot1 - 1$ कारक)। के लिए $ b > a + 1$, सत्यापित करें
सीधे तौर पर वह $(a\ b) = (a\ \ a{+}1)\,(a{+}1\ \ b)\,(a\ \ a{+}1)$:
दाईं ओर भेजता है $ a \mapsto a{+}1 \mapsto b \mapsto b $, $ b
\mapsto b \mapsto a{+}1 \mapsto a $, $ a{+}1 \mapsto a \mapsto a
\mapsto a{+}1$, और बाकी को ठीक करता है। प्रेरण द्वारा $(a{+}1\ \ b)$ है
का एक पैलिंड्रोमिक उत्पाद $2(b - a - 1) - 1$ आसन्न
ट्रांसपोज़िशन, इसलिए $(a\ b)$ में से एक है $2(b - a) - 1$: एक अजीब
संख्या.

\textbf{7.} $ N(\mathrm{id}) = 0$, $\varepsilon = +1$. के लिए $(i\ \
i{+}1)$, एकमात्र उलटा जोड़ा है $(i, i+1)$: $ N = 1$,
$\varepsilon = -1$. के लिए $[2, 3, 1]$: उल्टे जोड़े हैं $(1,
3)$ (मान $2 > 1$) और $(2, 3)$ (मान $3 > 1$): $ N = 2$,
$\varepsilon = +1$.

\textbf{8.} की मूल्य सूचियाँ $\sigma $ और $\sigma\tau $ भिन्न
केवल पदों की अदला-बदली से $ i $ और $ i + 1$. की एक जोड़ी के लिए
पद शामिल नहीं हैं $ i, i+1$, कुछ भी नहीं बदलता. के लिए $ k < i $,
दो जोड़े $(k, i)$ और $(k, i+1)$ उनके व्युत्क्रम का आदान-प्रदान करें
स्थितियाँ (समान दो मानों की तुलना की जाती है $\sigma(k)$, में
दूसरा \omterm{def:b1:structures:order}{आदेश} पदों का): उनका कुल योगदान है
अपरिवर्तित; इसी तरह के लिए $ k > i + 1$. एक बची हुई जोड़ी $(i,
i+1)$ इसकी स्थिति पलट जाती है। इस तरह $ N(\sigma\tau) = N(\sigma) \pm 1$.

\textbf{9.} होने देना $\tau = (a\ b)$ कोई भी स्थानान्तरण हो: द्वारा
प्रश्न 6 यह आसन्न की विषम संख्या का गुणनफल है
स्थानान्तरण, इसलिए सही-गुणा करना $\tau $ \omterm{def:b1:counting:objects}{परिवर्तन} $ N $ एक द्वारा
विषम योग (प्रश्न 8, बार-बार लागू): $\varepsilon(\sigma
\tau) = -\varepsilon(\sigma)$. अब यदि $\sigma = \tau_1\cdots
\tau_p $ (ट्रांसपोज़िशन), इसे पहचान से बनाएं $ p $
सही-गुणा: $\varepsilon(\sigma) =
(-1)^p\varepsilon(\mathrm{id}) = (-1)^p $. तब से
$\varepsilon(\sigma)$ व्युत्क्रम द्वारा परिभाषित किया गया है --- स्वतंत्र रूप से
किसी भी गुणनखंड की --- की समता $ p $ का एक अपरिवर्तनीय है
$\sigma $. \omterm{def:b1:structures:morphism}{आकारिता}: लिखना $\sigma $ साथ $ p $ और $\sigma'$ साथ
$ q $ स्थानांतरण, $\sigma\sigma'$ उपयोग $ p + q $ उनमें से:
$\varepsilon(\sigma\sigma') = (-1)^{p+q} =
\varepsilon(\sigma)\varepsilon(\sigma')$.

\textbf{10.} ए $ k $-साइकिल का एक उत्पाद है $ k - 1$ स्थानान्तरण
(प्रश्न 5): $\varepsilon = (-1)^{k-1}$. सामान्य के लिए $\sigma $
आकार की कक्षाओं के साथ $ k_1, \dots, k_r $ ($ k_i \geq 2$) प्लस $ f $
निश्चित बिंदु, $ c(\sigma) = r + f $ और $ n = k_1 + \dots + k_r + f $,
इसलिए
\[
\varepsilon(\sigma) = \prod_{i=1}^r (-1)^{k_i - 1}
= (-1)^{\sum_i k_i - r} = (-1)^{n - f - r} = (-1)^{n -
c(\sigma)} .
\]

\textbf{11.} $\mathfrak A_n = \ker\varepsilon $ एक है \omterm{def:b1:structures:subgroup}{उपसमूह} जैसे
द \omterm{def:b1:structures:morphism}{कर्नेल} एक का \omterm{def:b1:structures:morphism}{आकारिता} (\cref{def:b1:structures:morphism}). ठीक करो
एक स्थानांतरण $\tau_0$ (के लिए मौजूद है $ n \geq 2$). \omterm{def:b1:logic:map}{नक्शा} $\sigma
\mapsto \sigma\tau_0$ का आक्षेप है $\mathfrak S_n $ (यह अपना है
उलटा) आदान-प्रदान $\mathfrak A_n $ साथ \omterm{def:b1:logic:sets}{तय करना} विषम का
क्रमपरिवर्तन (प्रश्न 9). दो सेट \omterm{thm:b1:logic:partition}{PARTITION} $\mathfrak S_n $
और समान आकार है: $\abs{\mathfrak A_n} = \frac{n!}2$.

\textbf{12.} \emph{इन्वर्ज़न} का $[4, 1, 5, 2, 3, 7, 8, 6]$:से
मूल्य $4$: ऊपर $1, 2, 3$: तीन; से $5$: ऊपर $2, 3$: दो;
से $7$: ऊपर $6$: एक; से $8$: ऊपर $6$: एक। $ N = 7$,
$\varepsilon = -1$. \emph{साइकिल का प्रकार}: $ c = 3$ कक्षाएँ, $ n = 8$:
$\varepsilon = (-1)^{8-3} = -1$. \emph{ट्रांसपोज़िशन गिनती}: पांच
प्रश्न 5 में स्थानान्तरण: $(-1)^5 = -1$. तीनों सहमत हैं.

\textbf{13.} $(a\ b)(a\ c)$ (सबसे दाहिना पहला): $ a \mapsto c
\mapsto c $; $ c \mapsto a \mapsto b $; $ b \mapsto b \mapsto a $:
द $3$-चक्र $(a\ c\ b)$. और $(a\ c\ b)(a\ c\ d)$: $ a \mapsto c
\mapsto b $; $ b \mapsto b \mapsto a $; $ c \mapsto d \mapsto d $; $ d
\mapsto a \mapsto c $: वह है $(a\ b)(c\ d)$, जैसा कि दावा किया गया है।

\textbf{14.} होने देना $\sigma \in \mathfrak A_n $: प्रश्न 9 से,
$\sigma = \tau_1\cdots\tau_{2m}$ की सम संख्या के साथ
स्थानान्तरण। उन्हें लगातार जोड़ियों में \omterm{def:b1:structures:group}{समूह}ित करें
$\tau_{2i-1}\tau_{2i}$: यदि दोनों बराबर हैं, तो जोड़ी है
पहचान और गायब हो जाता है; यदि वे बिल्कुल एक बिंदु साझा करते हैं, तो
प्रश्न 13 की पहली पहचान जोड़ी को एक के रूप में लिखती है $3$-चक्र;
यदि वे असंयुक्त हैं, तो दूसरी पहचान इसे दो के रूप में लिखती है
$3$-चक्र। इस तरह $\sigma $ का एक उत्पाद है $3$-चक्र (या
पहचान, एक खाली उत्पाद --- और के लिए $ n \geq 3$ भी $(1\ 2\
3)^3$).

\textbf{15.} $(1\ 2)(3\ 4) = (1\ 3\ 2)(1\ 3\ 4)$ (प्रश्न 13
के साथ $ a{=}1, b{=}2, c{=}3, d{=}4$). के लिए $5$-साइकिल: द्वारा
प्रश्न 5, $(1\ 2\ 3\ 4\ 5) = (1\ 5)(1\ 4)(1\ 3)(1\ 2)$, और
जोड़ी बनाना: $(1\ 5)(1\ 4) = (1\ 4\ 5)$, $(1\ 3)(1\ 2) = (1\ 2\ 3)$:
\[
(1\ 2\ 3\ 4\ 5) = (1\ 4\ 5)(1\ 2\ 3) .
\]
(पर जाँचा $3$: $(1\ 2\ 3)$ भेजता है $3 \to 1$, तब $(1\ 4\ 5)$
भेजता है $1 \to 4$: जाल $3 \to 4$, सही।)

\textbf{16.} दोनों पक्षों को एक मनमाने बिंदु पर लागू करें। के लिए $ i =
\sigma(a_j)$: बाईं ओर देता है $\sigma\bigl((a_1\ \dots\
a_k)(a_j)\bigr) = \sigma(a_{j+1})$ (सूचकांक मॉड $ k $), जो है
दाहिना भाग क्या करता है $\sigma(a_j)$. के लिए $ i $ इसका नहीं
फार्म: $\sigma^{-1}(i)$ समर्थन के बाहर है, इसलिए बाईं ओर
ठीक करता है $ i $, और दाहिनी ओर भी ऐसा ही है। सर्वत्र समान।

\textbf{17.} सेल की टाइल को खिसकाना $ c'$ खाली सेल में
$ c $ कोशिकाओं की सामग्री की अदला-बदली करता है $ c $ और $ c'$ (टाइल $9$, द
रिक्त, की ओर ले जाता है $ c'$). यदि टाइल $\sigma(i)$ कोठरी में बैठ गया $ i $, द
नई स्थिति है $\sigma' = \sigma \circ (c\ c')$: समान सामग्री
सिवाय उस कोशिकाओं के $ c, c'$ एक दूसरे की पूर्व सामग्री पढ़ें। द्वारा
प्रश्न 9, $\varepsilon(\sigma') = -\varepsilon(\sigma)$.

\textbf{18.} एक चाल खाली सेल को आसन्न सेल में भेजती है: इसकी
पंक्ति या उसका स्तंभ बिल्कुल बदलता है $1$, तो टैक्सीकैब दूरी
$ d $ सेल के लिए $9$ द्वारा \omterm{def:b1:counting:objects}{परिवर्तन} $\pm1$, और $(-1)^d $ फ़्लिप. प्रत्येक के बाद से
दोनों को पलटें $\varepsilon(\sigma)$ और $(-1)^{d(\sigma)}$,
उनका उत्पाद $ I(\sigma)$ हर चाल से अपरिवर्तित है: एक
अपरिवर्तनीय.

\textbf{19.} हल स्थिति है $\varepsilon = +1$, $ d = 0$:
$ I = +1$. लक्ष्य (टाइल्स $7, 8$ अदला-बदली, खाली घर) है
कोशिकाओं की सामग्री का स्थानान्तरण $7$ और $8$: $\varepsilon
= -1$, $ d = 0$: $ I = -1$. तब से $ I $ अपरिवर्तनीय है और दो
मान भिन्न-भिन्न होते हैं, चालों का कोई क्रम उनसे नहीं जुड़ता।

\textbf{20.} घर पर रिक्त स्थान वाली एक स्थिति है \omterm{def:b1:counting:objects}{परिवर्तन}
की $8$ कोशिकाओं के बीच टाइलें $1, \dots, 8$, यानी\ का एक तत्व
$\mathfrak S_8$; यह है $ d = 0$, इसलिए $ I = \varepsilon(\sigma)$.
पहुंच योग्य बल $ I = +1$, यानी\ $\sigma \in \mathfrak A_8$; द
स्वीकृत वार्तालाप सब कुछ कहता है $\mathfrak A_8$ पहुंच गया है. गिनती करना:
$\abs{\mathfrak A_8} = \frac{8!}2 = 20\,160$ (प्रश्न 11).

\textbf{21.} प्रश्न 20 तक पहुंच योग्य रिक्त-घर पर
व्यवस्थाएँ बिल्कुल सही ढंग से बनती हैं $\mathfrak A_8$ --- विशेष रूप से ए
\emph{\omterm{def:b1:structures:subgroup}{उपसमूह}} का $\mathfrak S_8$: दो हल करने योग्य रचनाएँ
हाथापाई, या किसी को उलट देना, हल करने योग्य रहता है, जो कि बहुत दूर है
शुद्ध पहेली तर्क से स्पष्ट।

\textbf{22.} (ए) ए $3$-खाली घर के साथ टाइल्स का चक्र:
$\varepsilon = +1$ (प्रश्न 10), $ d = 0$, इसलिए $ I = +1$: पहुंच योग्य
(स्वीकृत वार्तालाप के अनुसार) --- एक व्यक्ति तीन टाइलें चक्रित कर सकता है।
(बी) टाइल $5$ और रिक्त स्थान का आदान-प्रदान: स्थिति है
कोशिकाओं की सामग्री का स्थानान्तरण $5$ और $9$, इसलिए
$\varepsilon = -1$; रिक्त स्थान टैक्सी कैब के मध्य में स्थित है
दूरी $ d = 2$ घर से, तो $(-1)^d = +1$ और $ I = -1$:
अगम्य. कोई बस ``खाली जगह को बीच में पार्क नहीं कर सकता''
टाइल्स को अन्यथा क्रमबद्ध छोड़ देना।

\textbf{23.} होने देना $ f \colon \mathfrak S_n \to \{\pm1\}$ एक हो
\omterm{def:b1:structures:morphism}{आकारिता}. किन्हीं दो स्थानान्तरणों के लिए $\tau, \tau'$, प्रश्न 16
प्रदान करता है $\sigma $ साथ $\sigma\tau\sigma^{-1} = \tau'$ (\omterm{def:b1:logic:map}{नक्शा} द
दो अन्य बिंदुओं पर दो स्थानांतरित बिंदु; $ n \geq 3$ कमरे की गारंटी देता है
ऐसा करने के लिए, यद्यपि भी $ n = 2$ यहाँ तुच्छ है)। तब $ f(\tau') =
f(\sigma)f(\tau)f(\sigma)^{-1} = f(\tau)$ तब से $\{\pm1\}$ है
\omterm{def:b1:structures:group}{एबेलियन}: $ f $ स्थानान्तरण पर स्थिर है। यदि वह स्थिरांक है
$+1$, तब $ f = 1$ स्थानान्तरण के सभी उत्पादों पर, अर्थात्\
हर जगह (प्रश्न 5). अगर यह है $-1$, तब $ f(\sigma) =
(-1)^p = \varepsilon(\sigma)$ के एक उत्पाद पर $ p $ स्थानान्तरण।
इसलिए $ f \in \{1, \varepsilon\}$.

\textbf{24.} (i) द \omterm{def:b1:structures:morphism}{आकारिता} की संपत्ति $\varepsilon $ और यह
\omterm{def:b1:structures:morphism}{कर्नेल} मशीनरी दी $\mathfrak A_n $ इसका \omterm{def:b1:structures:subgroup}{उपसमूह} संरचना और
इसका आकार, और \cref{prop:b1:structures:kernel}-शैली तर्क
प्रश्न 11 और 21 से गुजरता है। (ii) गिनती:
$\abs{\mathfrak S_n} = n!$, प्रश्न 11 का आधा तर्क,
और गिनती $20\,160$ प्रश्न 20 के हैं
\cref{ch:b1:counting} काम पर। (iii) प्रश्न 8--9 हल करें a
वास्तविक सु-परिभाषित समस्या --- ``संख्या की समता
ट्रांसपोज़िशन का'' यह मानता है कि यह समता निर्भर नहीं करती है
गुणनखंडन पर, बिल्कुल वैसे ही $\Z/n\Z $ के संचालन की आवश्यकता है
प्रतिनिधि-स्वतंत्रता में \cref{def:b1:structures:zn}.

\textbf{25.} हस्ताक्षर एकल है $\{\pm1\}$-मूल्यांकित
गणना, एक बार अच्छी तरह से परिभाषित साबित हुई, और यह तीन करती है
एक बार में नौकरियाँ: आंतरिक रूप से, यह कटौती करता है $\mathfrak S_n $ आधे में और
पृथक करता है $\mathfrak A_n $ के साथ $3$-साइकिल जनरेटर;
बाह्य रूप से, यह एक पंक्ति में एक प्रश्न का निर्णय करता है (``ये दोनों कर सकते हैं
टाइलों की अदला-बदली की जाएगी?'') वह भोली खोज कभी भी व्यवस्थित नहीं हो सकी, तब से
असफल चाल अनुक्रमों की कोई सीमित सूची असंभवता साबित नहीं करती; और
संरचनात्मक रूप से, यह अंदर का अल्टरनेटिंग-साइन इंजन है
सूत्र $\det A = \sum_\sigma \varepsilon(\sigma)\,
a_{1\sigma(1)}\cdots a_{n\sigma(n)}$ का \cref{ch:b1:det}. द
पैटर्न --- प्रत्येक प्रारंभिक चाल द्वारा संरक्षित मात्रा ज्ञात करें,
प्रारंभ में और लक्ष्य पर इसकी गणना करें --- है
``क्या यह संभव है?'' के विरुद्ध गणितज्ञ का मानक हथियार
प्रश्न, और यह जब भी वापस आएगा \omterm{def:b1:structures:group}{समूह} ए पर कार्य करता है \omterm{def:b1:logic:sets}{तय करना} का
राज्य.
