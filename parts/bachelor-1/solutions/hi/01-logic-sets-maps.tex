\section*{अध्याय \ref{ch:b1:logic} --- तर्क, सेट और मानचित्र}

\begin{solution}{exo:b1:logic:1}
नकारना, धक्का देना $\lnot $ प्रत्येक परिमाणक के माध्यम से
(\cref{prop:b1:logic:negquant}) और उपयोग कर रहे हैं $\lnot(P \implies Q) \iff
P \land \lnot Q $:
\begin{enumerate}
  \item $\exists x \in \R,\ \forall y \in \R,\ x + y \leq 0$;
  \item $\forall x \in \R,\ \exists y \in \R,\ xy \neq 0$;
  \item $\exists \varepsilon > 0,\ \forall \delta > 0,\ \exists x \in
        \R,\ \abs{x} \leq \delta \text{ and } \abs{f(x)} >
        \varepsilon $.
\end{enumerate}
\omterm{def:b1:logic:statement}{कथन} (1) सत्य है: दिया गया $ x $, लेना $ y = -x + 1$; तब $ x + y = 1 >
0$. \omterm{def:b1:logic:statement}{कथन} (2) सत्य है: $ x = 0$ संतुष्ट $ xy = 0$ सभी के लिए $ y $.
\end{solution}

\begin{solution}{exo:b1:logic:2}
सत्य तालिका, चार मामलों के लिए टी/एफ लिखना $(P, Q)$:
\begin{center}
\begin{tabular}{cc|c|c|c|c}
$ P $ & $ Q $ & $ P \implies Q $ & $\lnot(P \implies Q)$ & $\lnot Q $ &
$ P \land \lnot Q $ \\
\hline
T & T & T & F & F & F \\
T & F & F & T & T & T \\
F & T & T & F & F & F \\
F & F & T & F & T & F \\
\end{tabular}
\end{center}
कॉलम $4$ और $6$ संयोग, तुल्यता सिद्ध करना। का निषेध
``यदि कोई फलन अवकलनीय है तो वह सतत है'' इसलिए है:
``एक ऐसा फलन है जो अवकलनीय है और सतत नहीं है''
(एक झूठ \omterm{def:b1:logic:statement}{कथन}, जैसा कि होता है: मूल निहितार्थ सत्य है,
देखिये \cref{ch:b1:derivative}).
\end{solution}

\begin{solution}{exo:b1:logic:3}
\emph{विरोधाभास.} मान लीजिए $ x < 1$. तब $ x^3 < 1$ (घन
कार्य बढ़ रहा है) और $ x < 1$, इसलिए $ x^3 + x < 2$. इससे यह सिद्ध होता है
गर्भनिरोधक, इसलिए \omterm{def:b1:logic:statement}{कथन}.

\emph{विरोधाभास।} कल्पना करना $ a > 0$ सबसे छोटा सख्ती से सकारात्मक है
असली. तब $ a/2$ पूरी तरह से सकारात्मक है और $ a/2 < a $ (तब से $ a > 0$),
न्यूनतमता का खंडन. अत: ऐसा कुछ नहीं $ a $ मौजूद है.
\end{solution}

\begin{solution}{exo:b1:logic:4}
\begin{enumerate}
  \item बेस केस $ n = 0$: $2^0 = 1 = 2^1 - 1$. चरण: मान लेना
        के लिए पहचान $ n $,
        \[
        \sum_{k=0}^{n+1} 2^k = (2^{n+1} - 1) + 2^{n+1}
        = 2 \cdot 2^{n+1} - 1 = 2^{n+2} - 1 .
        \]
  \item बेस केस $ n = 0$: $4^0 + 5 = 6 = 3 \times 2$. चरण: यदि
        $4^n + 5 = 3m $, तब
        \[
        4^{n+1} + 5 = 4(4^n + 5) - 15 = 3(4m - 5),
        \]
        द्वारा विभाज्य $3$.
\end{enumerate}
\end{solution}

\begin{solution}{exo:b1:logic:5}
आगमनात्मक चरण चुपचाप मानता है कि दो समूह (``पहला
$ n $'' और ``आखिरी $ n $'') ओवरलैप करें, ताकि साझा पेंसिलें चल सकें
एक समूह से दूसरे समूह में रंग। के लिए $ n + 1 = 2$ दो समूह
हैं $\{$ पहली पेंसिल $\}$ और $\{$ दूसरी पेंसिल $\}$: वे असंयुक्त हैं,
और तर्क टूट जाता है. इसलिए $ P(1) \implies P(2)$ कभी साबित नहीं हुआ, और
इंडक्शन ढह जाता है --- भले ही $ P(n) \implies P(n+1)$ है
प्रत्येक के लिए मान्य $ n \geq 2$.
\end{solution}

\begin{solution}{exo:b1:logic:6}
\begin{enumerate}
  \item $ x \in A \setminus B \iff x \in A \land x \notin B \iff x \in
        A \land x \in \overline{B} \iff x \in A \cap \overline{B}$.
  \item (1) और वितरण का उपयोग करना
        (\cref{prop:b1:logic:setalgebra}):
        $(A \cup B) \cap \overline{C} = (A \cap \overline{C}) \cup
        (B \cap \overline{C})$.
  \item कल्पना करना $ A \subseteq B $. तब $ A \cup B \subseteq B $ (दोनों
        टुकड़े पड़े हैं $ B $) और $ B \subseteq A \cup B $ ऐसे ही हमेशा
        $ A \cup B = B $. कल्पना करना $ A \cup B = B $: तब $ A \cap B
        \subseteq A $ हमेशा, और $ A \subseteq A \cup B = B $ देता है
        $ A \subseteq A \cap B $, इसलिए $ A \cap B = A $. कल्पना करना $ A \cap B =
        A $: तब $ A = A \cap B \subseteq B $. तीन शर्तें हैं
        इस प्रकार समतुल्य (हमने निहितार्थों का एक चक्र सिद्ध किया)।
\end{enumerate}
\end{solution}

\begin{solution}{exo:b1:logic:7}
\begin{enumerate}
  \item $ f $ है \omterm{def:b1:logic:inj}{इंजेक्शन} ($ n + 1 = m + 1 \implies n = m $) लेकिन नहीं
        \omterm{def:b1:logic:inj}{विशेषण}: $0$ है कोई \omterm{def:b1:logic:map}{पूर्वछवि} में $\N $.
  \item $ g $ है \omterm{def:b1:logic:inj}{द्विभाजित}: $ n \mapsto n - 1$ पर दोतरफा उलटा है
        $\Z $.
  \item $ h $ है \omterm{def:b1:logic:inj}{इंजेक्शन}: $\frac{x+1}{x-1} = \frac{x'+1}{x'-1}$ देता है
        $(x+1)(x'-1) = (x'+1)(x-1)$, यानी\ $ xx' - x + x' - 1 = xx' -
        x' + x - 1$, इसलिए $2x' = 2x $. यह नहीं है \omterm{def:b1:logic:inj}{विशेषण} पर $\R $:
        सुलझाना $ y = \frac{x+1}{x-1}$ देता है $ x(y - 1) = y + 1$, जो
        कब कोई समाधान नहीं है $ y = 1$ (समीकरण पढ़ता है $0 = 2$).
        कोडोमेन के साथ $\R \setminus \{1\}$, वही गणना देता है
        अद्वितीय \omterm{def:b1:logic:map}{पूर्वछवि} $ x = \frac{y+1}{y-1}$, इसलिए
        $ h \colon \R \setminus \{1\} \to \R \setminus \{1\}$ है
        \omterm{def:b1:logic:inj}{द्विभाजित} और $ h^{-1}(y) = \frac{y+1}{y-1} = h(y)$: $ h $ इसका है
        खुद का उलटा.
\end{enumerate}
\end{solution}

\begin{solution}{exo:b1:logic:8}
\begin{enumerate}
  \item $ x \in f^{-1}(B \cap B') \iff f(x) \in B \cap B' \iff f(x) \in
        B \land f(x) \in B' \iff x \in f^{-1}(B) \cap f^{-1}(B')$.
        छवियों के लिए: $ y \in f(A \cup A')$ आईएफएफ $ y = f(x)$ कुछ के लिए $ x $
        में $ A $ या में $ A'$, यदि $ y \in f(A)$ या $ y \in f(A')$.
  \item अगर $ y \in f(A \cap A')$, तब $ y = f(x)$ साथ $ x \in A $ और
        $ x \in A'$, इसलिए $ y \in f(A)$ और $ y \in f(A')$. कठोरता:
        ले लो $ f \colon \R \to \R $, $ x \mapsto x^2$, $ A = \{-1\}$,
        $ A' = \{1\}$: तब $ f(A \cap A') = f(\emptyset) = \emptyset $
        लेकिन $ f(A) \cap f(A') = \{1\}$.
  \item ($\Leftarrow $) साथ $ A = \{x\}$, $ A' = \{x'\}$ के लिए $ x \neq
        x'$: अगर $ f(x) = f(x')$, तब $ f(A) \cap f(A') = \{f(x)\}$
        जबकि $ f(A \cap A') = \emptyset $, कल्पित का खंडन
        समानता; इसलिए $ f $ है \omterm{def:b1:logic:inj}{इंजेक्शन}. ($\Rightarrow $) होने देना $ f $ होना
        \omterm{def:b1:logic:inj}{इंजेक्शन} और $ y \in f(A) \cap f(A')$: $ y = f(x) = f(x')$ साथ
        $ x \in A $, $ x' \in A'$; \omterm{def:b1:logic:inj}{इंजेक्शन} देता है $ x = x' \in A \cap
        A'$, इसलिए $ y \in f(A \cap A')$. (2) के साथ, समानता कायम है।
\end{enumerate}
\end{solution}

\begin{solution}{exo:b1:logic:9}
$ g \circ f = \mathrm{id}_E $ है \omterm{def:b1:logic:inj}{इंजेक्शन} और \omterm{def:b1:logic:inj}{विशेषण}, तो द्वारा
\cref{prop:b1:logic:comp}~(2), $ f $ है \omterm{def:b1:logic:inj}{इंजेक्शन} और $ g $ है \omterm{def:b1:logic:inj}{विशेषण}.
उदाहरण: $ E = \N $, $ F = \Z $, $ f $ समावेशन $ n \mapsto n $, और
$ g \colon \Z \to \N $, $ g(n) = n $ के लिए $ n \geq 0$ और $ g(n) = 0$ के लिए
$ n < 0$. तब $ g(f(n)) = n $ सभी के लिए $ n \in \N $, लेकिन $ f $ क्या नहीं है
\omterm{def:b1:logic:inj}{विशेषण} और $ g $ क्या नहीं है \omterm{def:b1:logic:inj}{इंजेक्शन}.
\end{solution}

\begin{solution}{exo:b1:logic:10}
$ x^2 - y^2 = x - y \iff (x - y)(x + y) = x - y \iff (x - y)(x + y - 1)
= 0 \iff y = x $ या $ y = 1 - x $. \emph{प्रतिवर्ती:} $ y = x $ काम करता है.
\emph{सममित:} शर्त ``$ y = x $ या $ y = 1 - x $'' है
में सममित $ x $ और $ y $ (अगर $ y = 1 - x $ तब $ x = 1 - y $).
\emph{सकर्मक:} कल्पना करना $ x \mathbin{\mathcal{R}} y $ और $ y
\mathbin{\mathcal{R}} z $; चार मामलों से गुज़रते हुए, $ z $ के बराबर होती है $ x $
या $1 - x $ हर बार (उदा.\ $ y = 1 - x $ और $ z = 1 - y $ देना $ z = x $).
इसलिए $\mathcal{R}$ एक \omterm{def:b1:logic:equiv}{समतुल्य संबंध} और $\mathrm{cl}(x) =
\{x,\, 1 - x\}$. इस वर्ग में बिल्कुल एक तत्व है जब $ x = 1 - x $,
यानी\ के लिए $ x = \frac12$.
\end{solution}

\begin{solution}{exo:b1:logic:11}
होने देना $ f \colon E \to \mathcal{P}(E)$ कोई भी हो \omterm{def:b1:logic:map}{नक्शा} और सेट करें
$ D = \{x \in E : x \notin f(x)\} \in \mathcal{P}(E)$. कल्पना करना $ D =
f(a)$ कुछ के लिए $ a \in E $. अगर $ a \in D $, फिर की परिभाषा के अनुसार $ D $,
$ a \notin f(a) = D $: विरोधाभास। अगर $ a \notin D $, तब $ a \notin
f(a)$, तो की परिभाषा के अनुसार $ D $, $ a \in D $: विरोधाभास। इस तरह $ D $ है
की छवि में नहीं $ f $, और $ f $ क्या नहीं है \omterm{def:b1:logic:inj}{विशेषण}. (विशेष रूप से नहीं
\omterm{def:b1:logic:sets}{तय करना} इसके साथ आपत्ति में है \omterm{def:b1:logic:sets}{सत्ता स्थापित}: के ``अधिक'' उपसमुच्चय हैं
$\N $ पूर्णांकों से अधिक.)
\end{solution}

\begin{solution}{exo:b1:logic:12}
\begin{enumerate}
  \item ($\Rightarrow $) होने देना $ f $ होना \omterm{def:b1:logic:inj}{विशेषण} और $\Phi(B) =
        \Phi(B')$. के लिए $ y \in B $, चुनना $ x $ साथ $ f(x) = y $; तब $ x
        \in f^{-1}(B) = f^{-1}(B')$, इसलिए $ y = f(x) \in B'$. इस तरह $ B
        \subseteq B'$, और सममित रूप से $ B' \subseteq B $: $\Phi $ है
        \omterm{def:b1:logic:inj}{इंजेक्शन}. ($\Leftarrow $) अगर $ f $ क्या नहीं है \omterm{def:b1:logic:inj}{विशेषण}, चुनना $ y_0
        \in F $ छवि के बाहर; तब $ f^{-1}(\{y_0\}) = \emptyset =
        f^{-1}(\emptyset)$ साथ $\{y_0\} \neq \emptyset $, इसलिए $\Phi $ क्या नहीं है \omterm{def:b1:logic:inj}{इंजेक्शन}.
  \item ($\Rightarrow $) होने देना $ f $ होना \omterm{def:b1:logic:inj}{इंजेक्शन} और $ A \subseteq E $. \omterm{def:b1:logic:sets}{तय करना}
        $ B = f(A)$; तब $ f^{-1}(B) = \{x : f(x) \in f(A)\}$, और
        \omterm{def:b1:logic:inj}{इंजेक्शन} देता है $ f(x) \in f(A) \iff x \in A $, इसलिए $\Phi(B) =
        A $: $\Phi $ है \omterm{def:b1:logic:inj}{विशेषण}. ($\Leftarrow $) अगर $ f $ क्या नहीं है
        \omterm{def:b1:logic:inj}{इंजेक्शन}, लेना $ x \neq x'$ साथ $ f(x) = f(x')$. प्रत्येक
        \omterm{def:b1:logic:map}{पूर्वछवि} \omterm{def:b1:logic:sets}{तय करना} $ f^{-1}(B)$ रोकना $ x $ यदि और केवल यदि यह
        शामिल है $ x'$; इस तरह $\{x\}$ स्वरूप का नहीं है $\Phi(B)$, और
        $\Phi $ क्या नहीं है \omterm{def:b1:logic:inj}{विशेषण}.
\end{enumerate}
\end{solution}

\begin{solution}{pb:b1:logic:1}
\textbf{1.} \emph{प्रतिवर्ती:} $\mathrm{id}_E $ का आक्षेप है $ E $
खुद पर. \emph{सममित:} अगर $ f \colon E \to F $ है \omterm{def:b1:logic:inj}{द्विभाजित},
\cref{thm:b1:logic:inverse} प्रदान $ f^{-1} \colon F \to E $, स्वयं
\omterm{def:b1:logic:inj}{द्विभाजित}. \emph{सकर्मक:} अगर $ f \colon E \to F $ और $ g \colon F
\to G $ आक्षेप हैं, \cref{prop:b1:logic:comp}~(1) कहते हैं $ g \circ f
\colon E \to G $ एक आक्षेप है. (यह केवल ``पसंद'' है समतुल्यता
संबंध: सभी का संग्रह सेट स्वयं एक नहीं है \omterm{def:b1:logic:sets}{तय करना}, द्वारा
यह विरोधाभास है \cref{exo:b1:logic:11} संकेत करता है; तीन गुण
क्या मायने रखता है.)

\textbf{2.} अगर $ f \colon E \to F $ और $ g \colon F \to G $ हैं
\omterm{def:b1:logic:inj}{इंजेक्शन}, $ g \circ f $ है \omterm{def:b1:logic:inj}{इंजेक्शन} द्वारा \cref{prop:b1:logic:comp}~(1):
$ E \preceq G $. दूसरे बिंदु के लिए, कोरस्ट्रिक्ट $ f $ इसकी छवि के लिए:
\omterm{def:b1:logic:map}{नक्शा} $\tilde f \colon E \to f(E)$, $ x \mapsto f(x)$, है \omterm{def:b1:logic:inj}{विशेषण} द्वारा
का निर्माण $ f(E)$ और \omterm{def:b1:logic:inj}{इंजेक्शन} क्योंकि $ f $ है, इसलिए \omterm{def:b1:logic:inj}{द्विभाजित}:
$ E \approx f(E)$.

\textbf{3.} ($\Rightarrow $) होने देना $ f \colon E \to F $ होना \omterm{def:b1:logic:inj}{इंजेक्शन} और
ठीक करो $ a \in E $ ($ E \neq \emptyset $). परिभाषित करना $ s \colon F \to E $ द्वारा:
$ s(y)$ अद्वितीय है $ x $ साथ $ f(x) = y $ कब $ y \in f(E)$
(\omterm{def:b1:logic:inj}{इंजेक्शन} द्वारा विशिष्टता), और $ s(y) = a $ अन्यथा। हरएक के लिए $ x
\in E $, $ s(f(x)) = x $, तो हर $ x $ प्राप्त होता है: $ s $ है \omterm{def:b1:logic:inj}{विशेषण}.
($\Leftarrow $) होने देना $ s \colon F \to E $ होना \omterm{def:b1:logic:inj}{विशेषण}. प्रत्येक के लिए $ x \in
E $ एक का चयन $ y_x \in F $ साथ $ s(y_x) = x $, और सेट करें $ u(x) = y_x $. अगर
$ u(x) = u(x')$ तब $ x = s(u(x)) = s(u(x')) = x'$: $ u \colon E \to F $
है \omterm{def:b1:logic:inj}{इंजेक्शन}.

\textbf{4.} $ n \mapsto n + 1$ एमएपीएस $\N $ में $\N^*$, है \omterm{def:b1:logic:inj}{इंजेक्शन}
($ n + 1 = m + 1 \implies n = m $) और \omterm{def:b1:logic:inj}{विशेषण} (प्रत्येक $ m \geq 1$ है
$(m - 1) + 1$ साथ $ m - 1 \in \N $). के लिए $\sigma $: यह एमएपीएस सम संख्या $0, 2, 4, \dots $ को $0, 1, 2, \dots $ और विषम संख्याएँ $1, 3,
5, \dots $ को $-1, -2, -3, \dots $ इंजेक्टिविटी: सम इनपुट अंदर आते हैं
$\N $ ($\sigma(n) = n/2 \geq 0$) और विषम इनपुट इसमें उतरते हैं
पूर्णतः ऋणात्मक पूर्णांक ($\sigma(n) = -(n+1)/2 \leq -1$), तो ए
टकराव एक समता वर्ग के भीतर होना चाहिए, जहां $\sigma $ है
सख्ती से एकरस ($ n/2 = m/2$ या $(n+1)/2 = (m+1)/2$ बल $ n =
m $). \omterm{def:b1:logic:inj}{विशेषण}: $ k \geq 0$ है $\sigma(2k)$; $ k \leq -1$ है
$\sigma(-2k - 1)$ साथ $-2k - 1 \geq 1$ विषम। इसलिए $\N \approx \N^*$ और
$\N \approx \Z $.

\textbf{5.} $ C_0 = E \setminus g(F) \subseteq C $, इसलिए $ x \notin C $
तात्पर्य $ x \notin C_0$, यानी\ $ x \in g(F)$: कुछ $ y \in F $ संतुष्ट
$ g(y) = x $. यदि भी $ g(y') = x $, की injectivity $ g $ देता है $ y' = y $.
इसलिए की परिभाषा का दूसरा खंड $ h $ एक अद्वितीय चुनता है,
अच्छी तरह से परिभाषित तत्व $ g^{-1}(x)$.

\textbf{6.} प्रत्यक्ष छवियाँ यूनियनों के साथ चलती हैं
(\cref{exo:b1:logic:8}~(1), पर लागू $ f $ फिर को $ g $):
\[
g\bigl(f(C)\bigr)
= g\Bigl(f\Bigl(\bigcup_{n \in \N} C_n\Bigr)\Bigr)
= \bigcup_{n \in \N} g\bigl(f(C_n)\bigr)
= \bigcup_{n \in \N} C_{n+1}
= \bigcup_{n \geq 1} C_n \subseteq C .
\]

\textbf{7.} होने देना $ x \neq x'$ में $ E $. यदि दोनों झूठ बोलते हैं $ C $, तब $ h(x) =
f(x) \neq f(x') = h(x')$ के \omterm{def:b1:logic:inj}{इंजेक्शन} द्वारा $ f $. यदि दोनों में से कोई भी झूठ नहीं बोलता है
$ C $, तब $ g(h(x)) = x \neq x' = g(h(x'))$, इसलिए $ h(x) \neq h(x')$. अगर
$ x \in C $ और $ x' \notin C $ (मिश्रित मामला, नामों की अदला-बदली तक):
मान लीजिये $ h(x) = h(x')$, यानी\ $ f(x) = g^{-1}(x')$. को लागू करने $ g $:
$ x' = g(f(x)) \in g(f(C))$, और प्रश्न 6 देता है $ x' \in C $ ---
विरोधाभास। इसलिए $ h(x) \neq h(x')$ सभी मामलों में: $ h $ है \omterm{def:b1:logic:inj}{इंजेक्शन}.

\textbf{8.} होने देना $ y \in F $. \emph{केस 1: $ g(y) \notin C $.} तब
$ h(g(y)) = g^{-1}(g(y)) = y $: तत्व $ g(y)$ एक है \omterm{def:b1:logic:map}{पूर्वछवि}.
\emph{केस 2: $ g(y) \in C $}, कहना $ g(y) \in C_n $. तब से $ g(y) \in
g(F)$, हमारे पास है $ g(y) \notin C_0 = E \setminus g(F)$, इसलिए $ n \geq 1$
और $ g(y) \in C_n = g(f(C_{n-1}))$: वहाँ है $ x \in C_{n-1}$ साथ
$ g(y) = g(f(x))$. का \omterm{def:b1:logic:inj}{इंजेक्शन} $ g $ देता है $ y = f(x)$, और $ x \in
C_{n-1} \subseteq C $, इसलिए $ h(x) = f(x) = y $. दोनों ही मामलों में $ y $ है
प्राप्त: $ h $ है \omterm{def:b1:logic:inj}{विशेषण}, इस तरह \omterm{def:b1:logic:inj}{द्विभाजित}.

\textbf{9.} अगर $ E \preceq F $ और $ F \preceq E $, \omterm{def:b1:logic:inj}{इंजेक्शन} उठाओ $ f
\colon E \to F $ और $ g \colon F \to E $; प्रश्न 5-8 निर्माण ए
आपत्ति $ h \colon E \to F $, इसलिए $ E \approx F $. \omterm{def:b1:logic:statement}{कथन} है
गैर-तुच्छ क्योंकि दिए गए दोनों \omterm{def:b1:logic:inj}{इंजेक्शन} असंबंधित हैं --- न तो
होना चाहिए \omterm{def:b1:logic:inj}{विशेषण}, और कोई फार्मूला मिश्रण नहीं $ f $ और $ g $ भोलेपन से परिभाषित करता है
ए \omterm{def:b1:logic:map}{नक्शा}: संपूर्ण सामग्री है \omterm{thm:b1:logic:partition}{PARTITION} का $ E $ क्षेत्र में $ C $
(जहाँ कोई नकल करता है $ f $) और इसका पूरक (जहां कोई दौड़ता है $ g $
पीछे की ओर)।

\textbf{10.} (ए) समावेशन $\intoo01 \to \intcc01$ है \omterm{def:b1:logic:inj}{इंजेक्शन};
और $ x \mapsto \frac{x + 1}3$ एमएपीएस $\intcc01$ \omterm{def:b1:logic:inj}{इंजेक्शन} से में
$\intcc{\frac13}{\frac23} \subseteq \intoo01$ (यह इसके साथ जुड़ा हुआ है
शून्येतर ढलान)। प्रश्न 9 से, $\intcc01 \approx \intoo01$ --- ए
आपत्ति जिसे स्पष्ट रूप से लिखना काफी अप्रिय है।
(बी) \emph{\omterm{def:b1:logic:inj}{इंजेक्शन}।} कल्पना करना $2^p(2q + 1) = 2^{p'}(2q' + 1)$ के साथ,
कहो, $ p \leq p'$. द्वारा विभाजित करना $2^p $: $2q + 1 = 2^{p' - p}(2q' + 1)$.
अगर $ p' > p $ दायीं ओर सम है और बायीं ओर विषम ---
असंभव; इसलिए $ p = p'$, तब $2q + 1 = 2q' + 1$ और $ q = q'$.
\emph{प्रक्षेपात्मकता।} हम मजबूत प्रेरण द्वारा दिखाते हैं कि प्रत्येक पूर्णांक
$ m \geq 1$ स्वरूप का है $2^p(2q + 1)$. के लिए $ m = 1$: $ p = q = 0$.
होने देना $ m \geq 1$ और सभी पूर्णांकों के लिए दावा मान लें
$\intint1m $. अगर $ m + 1$ अजीब है, $ m + 1 = 2q + 1$ साथ $ p = 0$. अगर
$ m + 1$ सम है, $ m + 1 = 2m'$ साथ $1 \leq m' \leq m $; द्वारा
परिकल्पना $ m' = 2^p(2q + 1)$, इसलिए $ m + 1 = 2^{p+1}(2q + 1)$. इस तरह
$\varphi(p, q) = 2^p(2q + 1) - 1$ हर हिट $ n \in \N $, और
$\varphi $ एक आक्षेप है $\N \times \N \to \N $.

\textbf{11.} तब से $ A $ अनंत है, $ A \setminus \{\varphi(0), \dots,
\varphi(n - 1)\}$ कभी खाली नहीं होता, और सबसे कम तत्व वाला गुण होता है
$\N $ (साबित करने के लिए प्रयोग किया जाता है \cref{thm:b1:logic:induction}) बनाता है
पुनरावर्ती परिभाषा वैध. \emph{सख्ती से बढ़ रहा है:}
$\varphi(n + 1)$ से संबंधित $ A \setminus \{\varphi(0), \dots,
\varphi(n)\} \subseteq A \setminus \{\varphi(0), \dots, \varphi(n -
1)\}$, जिसका न्यूनतम है $\varphi(n)$; इसलिए $\varphi(n + 1) \geq
\varphi(n)$, और समानता को बाहर रखा गया है, जहां से $\varphi(n+1) >
\varphi(n)$. \emph{$\varphi(n) \geq n $:} प्रेरण द्वारा, $\varphi(0)
\geq 0$, और $\varphi(n + 1) \geq \varphi(n) + 1 \geq n + 1$.
\emph{\omterm{def:b1:logic:inj}{इंजेक्शन}} सख्त एकरसता से अनुसरण करता है।
\emph{पर \omterm{def:b1:logic:inj}{विशेषण} $ A $:} मान लीजिए कुछ $ a \in A $ कभी नहीं है
हासिल किया. तब से $\varphi(a + 1) \geq a + 1 > a $, द \omterm{def:b1:logic:sets}{तय करना} का $ n $ साथ
$\varphi(n) > a $ गैर-रिक्त है; होने देना $ n $ इसका सबसे छोटा तत्व हो. हरएक के लिए
$ k < n $, $\varphi(k) \leq a $, इस तरह $\varphi(k) < a $ ($ a $ नहीं है
प्राप्त) तब $ a $ में निहित है $ A \setminus \{\varphi(0), \dots,
\varphi(n - 1)\}$ और $ a < \varphi(n)$, न्यूनतमता का खंडन
परिभाषित करना $\varphi(n)$. इसलिए $\varphi $ एक आक्षेप है $\N \to A $, और
$ A $ गणनीय है.

\textbf{12.} होने देना $ E \preceq \N $ एक \omterm{def:b1:logic:inj}{इंजेक्शन} के माध्यम से $ f $; तब $ E \approx
f(E)$ (प्रश्न 2). अगर $ f(E)$ परिमित है, $ E $ परिमित है; अगर $ f(E)$ है
अनंत, प्रश्न 11 देता है $ f(E) \approx \N $, इसलिए $ E \approx \N $ द्वारा
परिवर्तनशीलता (प्रश्न 1). इसके विपरीत परिमित सेट और \omterm{pb:b1:logic:1}{गणनीय समुच्चय}
स्पष्ट रूप से इंजेक्ट करें $\N $. शॉर्टकट: $ E \preceq \N $ और $\N
\preceq E $ देना $ E \approx \N $ सीधे द्वारा
कैंटर--श्र\"ओडर--बर्नस्टीन--- किसी गणना तर्क की आवश्यकता नहीं है।

\textbf{13.} होने देना $ f \colon E \to \N $ और $ g \colon F \to \N $ हो
\omterm{def:b1:logic:inj}{इंजेक्शन}. तब $(x, y) \mapsto \varphi\bigl(f(x), g(y)\bigr)$ एक है
\omterm{def:b1:logic:inj}{इंजेक्शन} $ E \times F \to \N $: यदि छवियां मेल खाती हैं, तो \omterm{def:b1:logic:inj}{इंजेक्शन} की क्षमता
$\varphi $ (प्रश्न 10) देता है $ f(x) = f(x')$ और $ g(y) = g(y')$,
तब $ x = x'$, $ y = y'$. के लिए $\Z \times \N^*$: दोनों कारक हैं
गणनीय (प्रश्न 4), अतः $\Z \times \N^* \preceq \N $; यह है
अनंत (इसमें शामिल है $\{0\} \times \N^*$), इसलिए गणनीय है
प्रश्न 12.

\textbf{14.} हर तर्कसंगत $ r $ एक अद्वितीय प्रतिनिधित्व है $ r =
p/q $ साथ $ p \in \Z $, $ q \in \N^*$ और भिन्न न्यूनतम पदों में
(अद्वितीयता सिद्ध होती है \cref{ch:b1:arith}; के लिए $ r = 0$ लेना
$0/1$). \omterm{def:b1:logic:map}{नक्शा} $ r \mapsto (p, q)$ तब है \omterm{def:b1:logic:inj}{इंजेक्शन}: जोड़ी
निर्धारित करता है $ r = p/q $. इस तरह $\Q \preceq \Z \times \N^* \preceq \N $
प्रश्न 13 द्वारा. चूँकि $\N \subseteq \Q $ देता है $\N \preceq \Q $,
प्रश्न 12 (या कैंटर--श्र.)\"ओडर--बर्नस्टीन सीधे) दिखाता है $\Q
\approx \N $: परिमेय गणनीय हैं।

\textbf{15.} प्रत्येक के लिए $ n $ एक \omterm{def:b1:logic:inj}{इंजेक्शन} ठीक करो $ f_n \colon E_n \to \N $.
के लिए $ x \in \bigcup_n E_n $, होने देना $ n(x)$ हो \emph{कम से कम} $ n $ साथ
$ x \in E_n $, और सेट करें $ u(x) = \varphi\bigl(n(x), f_{n(x)}(x)\bigr)
\in \N $. अगर $ u(x) = u(x')$, की injectivity $\varphi $ देता है $ n(x) =
n(x') = n $ और $ f_n(x) = f_n(x')$, इस तरह $ x = x'$ के \omterm{def:b1:logic:inj}{इंजेक्शन} द्वारा
$ f_n $. तो संघ इसमें प्रवेश करता है $\N $: यह अधिकतम गणनीय है।

\textbf{16.} होने देना $\Psi(F) = \sum_{i \in F} 2^i $ के लिए $ F \subseteq \N $
परिमित ($\Psi(\emptyset) = 0$). कल्पना करना $ F \neq F'$ और चलो $ m $ हो
सबसे बड़ा तत्व जिस पर वे भिन्न हैं, कहते हैं $ m \in F \setminus
F'$ (यदि आवश्यक हो तो नाम बदलें)। तत्वों $> m $ दोनों के हैं या
न तो, इसलिए वे दोनों राशियों में समान रूप से योगदान करते हैं; तुलना कर रहे हैं
तत्वों का योगदान $\leq m $:
\[
\sum_{i \in F,\, i \leq m} 2^i \geq 2^m
> 2^m - 1 = \sum_{k=0}^{m-1} 2^k
\geq \sum_{i \in F',\, i \leq m} 2^i ,
\]
के ज्यामितीय योग का उपयोग करना \cref{exo:b1:logic:4}. इस तरह $\Psi(F)
\neq \Psi(F')$: $\Psi $ है \omterm{def:b1:logic:inj}{इंजेक्शन} और यह \omterm{def:b1:logic:sets}{तय करना} परिमित उपसमुच्चय का
का $\N $ अधिकतम गणनीय है; यह अनंत है (इसमें सब कुछ समाहित है)।
सिंगलटन), इसलिए गणनीय।

\textbf{17.} भेजना $ A \subseteq \N $ इसके सूचक को $\mathbf 1_A
\colon \N \to \{0,1\}$, $\mathbf 1_A(n) = 1$ अगर $ n \in A $ और $0$
अन्यथा; भेजना $ u \in \{0,1\}^{\N}$ को $ A_u = \{n \in \N : u(n) =
1\}$. दो एमएपीएस परस्पर विपरीत हैं: $ A_{\mathbf 1_A} = A $ और
$\mathbf 1_{A_u} = u $ (प्रत्येक पर मूल्य की जाँच करें $ n $). द्वारा
\cref{thm:b1:logic:inverse}, प्रत्येक एक आक्षेप है: $\mathcal{P}(\N)
\approx \{0,1\}^{\N}$.

\textbf{18.} हरएक के लिए $ n $, $ d(n) = 1 - \Phi(n)(n) \neq \Phi(n)(n)$,

कोई \omterm{def:b1:logic:inj}{इंजेक्शन} नहीं $\{0,1\}^{\N} \to \N $ दोनों में से एक: $\{0,1\}^{\N}$ पर नहीं है
सर्वाधिक गणनीय. प्रश्न 17 के शब्दकोष के माध्यम से ए \omterm{def:b1:logic:map}{नक्शा} $\Phi
\colon \N \to \{0,1\}^{\N}$ एक है \omterm{def:b1:logic:map}{नक्शा} $ f \colon \N \to

\cref{exo:b1:logic:11} के लिए $ E = \N $.

\textbf{19.} लिखना $ x_n = 0.d_1(n)\,d_2(n)\,d_3(n)\dots $ उचित में
रूप दें और परिभाषित करें $\delta_n = 5$ अगर $ d_n(n) \neq 5$, $\delta_n = 6$
अगर $ d_n(n) = 5$, तब $ x = 0.\delta_1\delta_2\delta_3\dots $ यह

\textbf{20.} $\intco01 \subseteq \R $, तो एक \omterm{def:b1:logic:inj}{इंजेक्शन} $\R \to \N $
एक पर ही सीमित होगा $\intco01$, विरोधाभासी प्रश्न 19:

विरोधाभास। अतः अतार्किकताएँ अनगिनत हैं। सटीक रूप से:

प्रेरण पर $ n $: $\Z^1 = \Z $ गणनीय है (प्रश्न 4), और
$\Z^{n+2} \approx \Z^{n+1} \times \Z $ द्वारा अधिकतम गणनीय है
प्रश्न 13. तो प्रत्येक \omterm{def:b1:logic:sets}{तय करना} परिबद्ध-डिग्री पूर्णांक बहुपद का है
अधिक से अधिक गणनीय; यह अनंत है (इसमें स्थिरांक शामिल हैं), इसलिए
प्रश्न 12 द्वारा गणनीय.

\textbf{23.}  \omterm{def:b1:logic:sets}{तय करना} सभी पूर्णांक बहुपदों का है $\bigcup_{n \in
\N} \{P : \deg P \leq n,\ P \text{ has integer coefficients}\}$, ए
का गणनीय संघ \omterm{pb:b1:logic:1}{गणनीय समुच्चय}: अधिक से अधिक गणनीय
प्रश्न 15, अनंत, अत: गणनीय।

\textbf{24.} प्रत्येक शून्येतर पूर्णांक बहुपद के लिए $ P $, जड़ \omterm{def:b1:logic:sets}{तय करना}
$ R_P = \{x \in \R : P(x) = 0\}$ परिमित है (अधिकतम) $\deg P $
तत्व, स्वीकृत)। प्रश्न 23 द्वारा शून्येतर पूर्णांक बहुपद
गिनाया जा सकता है $ P_0, P_1, P_2, \dots $; तब $\mathcal{A} =
\bigcup_{n \in \N} R_{P_n}$ परिमित का एक गणनीय संघ है (इसलिए
अधिक से अधिक गणनीय) सेट: अधिक से अधिक प्रश्न 15 द्वारा गणनीय
शामिल है $\Q $ (प्रश्न 21), इसलिए यह अनंत है: $\mathcal{A}$ है
गणनीय.

\textbf{25.} अगर $\R \setminus \mathcal{A}$ अधिकतम गणनीय थे,
$\R = \mathcal{A} \cup (\R \setminus \mathcal{A})$ अधिक से अधिक होगा
गणनीय (प्रश्न 15), प्रश्न 20 का खंडन करता है। इसलिए
पारलौकिक संख्याएँ अस्तित्व में है और यहां तक कि बेशुमार भी बनता है \omterm{def:b1:logic:sets}{तय करना},
जब \omterm{pb:b1:logic:1}{बीजगणितीय} संख्याएँ --- जिसमें प्रत्येक निर्मित संख्या शामिल है
पूर्णांकों से मूलांकों तक --- एक मात्र गणनीय कंकाल बनाते हैं
अंदर $\R $. वास्तुकला का सारांश: प्रश्न 1--3 \omterm{def:b1:logic:sets}{तय करना} ऊपर
तुलना भाषा; कैंटर--श्र\"ओडर--बर्नस्टीन (प्रश्न)।
5--9) आइए इसके बजाय दो आसान \omterm{def:b1:logic:inj}{इंजेक्शन}ों द्वारा समशक्ति सिद्ध करें
एक चतुर आक्षेप, और इसके लिए उपयोग किया गया था $\intcc01 \approx \intoo01$,
के लिए $\Q $ और पूरे भाग V में; जोड़ी आपत्ति (प्रश्न)
10) संचालित उत्पाद और गणनीय यूनियनें (प्रश्न 13, 15),
जो बदले में संचालित होता है $\Q $, पूर्णांक बहुपद और
$\mathcal{A}$; विकर्ण तर्क (प्रश्न 18--19) प्रदान किया गया
एक सख्त असमानता $\N \prec \R $ इससे पूरी कहानी बन जाती है
गैरतुच्छ. कैंटर का निष्कर्ष दार्शनिक रूप से आश्चर्यजनक है:
सबूत प्रदर्शित करता है नं पारलौकिक संख्या बिल्कुल भी, फिर भी वह दिखाता है
समशक्ति की भावना \emph{लगभग हर} वास्तविक संख्या है
पारमार्थिक। एक विशिष्ट पारलौकिक नामकरण --- $\pi $ या
$\eu$ --- पूरी तरह से अलग गणित और दशकों की आवश्यकता है
काम.
\end{solution}
