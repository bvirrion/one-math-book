\begin{solution}{exo:b1:functions:2}
$ f(x) = \arcsin(2x - 1)$ आवश्यक है $-1 \leq 2x - 1 \leq 1$, यानी\ $ x
\in \intcc{0}{1}$. पर $\intoo{0}{1}$, श्रृंखला नियम और
\cref{prop:b1:functions:arcderiv} देना
\[
f'(x) = \frac{2}{\sqrt{1 - (2x-1)^2}} = \frac{2}{\sqrt{4x - 4x^2}}
= \frac{1}{\sqrt{x(1 - x)}} .
\]
$ g(x) = \arctan\sqrt x $ के लिए परिभाषित किया गया है $ x \geq 0$, और के लिए $ x > 0$:
\[
g'(x) = \frac{1}{1 + x} \cdot \frac{1}{2\sqrt x}
= \frac{1}{2\sqrt x\,(1 + x)} .
\]
\end{solution}

\begin{solution}{exo:b1:functions:3}
$\cosh x + \sinh x = \frac{\eu^x + \eu^{-x}}{2} + \frac{\eu^x -
\eu^{-x}}{2} = \eu^x $. इसलिए के लिए $ n \in \Z $:
\[
(\cosh x + \sinh x)^n = (\eu^x)^n = \eu^{nx}
= \cosh nx + \sinh nx .
\]
\end{solution}

\begin{solution}{exo:b1:functions:4}
साथ $ y = \arcsin x $: $\cos 2y = 1 - 2\sin^2 y = 1 - 2x^2$, इसलिए
$\cos(2\arcsin x) = 1 - 2x^2$.

साथ $ y = \arctan x $: $\sin 2y = 2 \sin y \cos y = 2 \tan y \cos^2 y =
\frac{2\tan y}{1 + \tan^2 y}$, इसलिए $\sin(2\arctan x) =
\dfrac{2x}{1 + x^2}$.
\end{solution}

\begin{solution}{exo:b1:functions:5}
सबसे धीमे से सबसे तेज़ तक:
\[
(\ln x)^{1000} \;\ll\; x^{100} \;\ll\; x^{\ln x} \;\ll\;
\eu^{\sqrt x} \;\ll\; \eu^{x/100} .
\]
औचित्य: $(\ln x)^{1000}/x^{100} \to 0$ द्वारा
\cref{prop:b1:functions:powerrules} (लॉग शक्तियों से हार जाते हैं)।
$ x^{100} = \eu^{100\ln x}$ और $ x^{\ln x} = \eu^{(\ln x)^2}$: तब से
$(\ln x)^2 - 100\ln x \to +\infty $, दूसरा जीतता है। $ x^{\ln x} =
\eu^{(\ln x)^2} \ll \eu^{\sqrt x}$ क्योंकि $(\ln x)^2/\sqrt x \to 0$
(लॉग शक्ति खो देते हैं $ x^{1/4}$, चुकता)। अंत में $\sqrt x - x/100
\to -\infty $, इसलिए $\eu^{\sqrt x} \ll \eu^{x/100}$.
\end{solution}

\begin{solution}{exo:b1:functions:6}
कार्यक्षेत्र: $ x \neq \pm 1$, तीन अंतराल. सभी के ऊपर,
\[
f'(x) = \frac{\bigl(\frac{2x}{1-x^2}\bigr)'}{1 +
\frac{4x^2}{(1-x^2)^2}}
= \frac{\frac{2(1-x^2) + 4x^2}{(1-x^2)^2}}
{\frac{(1-x^2)^2 + 4x^2}{(1-x^2)^2}}
= \frac{2(1 + x^2)}{(1 + x^2)^2} = \frac{2}{1 + x^2}
= (2\arctan x)' .
\]
इसलिए $ f(x) - 2\arctan x $ प्रत्येक अंतराल पर स्थिर रहता है। मान: पर $ x =
0$, $ f(0) = 0$: स्थिरांक है $0$ पर $\intoo{-1}{1}$. जैसा $ x \to
+\infty $, $\frac{2x}{1-x^2} \to 0^-$ इसलिए $ f \to 0$, जबकि $2\arctan x
\to \pi $: स्थिरांक है $-\pi $ पर $\intoo{1}{+\infty}$. विचित्रता से,
यह है $+\pi $ पर $\intoo{-\infty}{-1}$. सारांश: $ f = 2\arctan x $ पर
$\intoo{-1}{1}$, $ f = 2\arctan x - \pi $ के लिए $ x > 1$, $ f = 2\arctan x +
\pi $ के लिए $ x < -1$. (यह स्पर्श रेखा के लिए द्विकोण सूत्र है,
पूरा पढ़ें $\arctan $.)
\end{solution}

\begin{solution}{exo:b1:functions:7}
$\cosh x = 2$: साथ $ u = \eu^x > 0$, $ u + u^{-1} = 4$, इसलिए $ u^2 - 4u +
1 = 0$, $ u = 2 \pm \sqrt 3$: $ x = \ln(2 + \sqrt 3)$ या $ x = \ln(2 -
\sqrt 3) = -\ln(2 + \sqrt 3)$ (दोनों समाधान विपरीत हैं, जैसे
$\cosh $ सम है; दोनों वैध हैं)। इसके तुल्य $ x = \pm
\operatorname{arcosh} 2$.

$5\cosh x - 4\sinh x = 3$: घातीय परिभाषाओं को प्रतिस्थापित करते हुए,
$\frac{5(u + u^{-1}) - 4(u - u^{-1})}{2} = 3$, यानी\ $ u + 9u^{-1} =
6$, यानी\ $ u^2 - 6u + 9 = (u - 3)^2 = 0$: $ u = 3$, एक हल $ x =
\ln 3$.
\end{solution}

\begin{solution}{exo:b1:functions:8}
होने देना $\alpha = \arctan\frac12 + \arctan\frac13$. अतिरिक्त सूत्र:
\[
\tan\alpha = \frac{\frac12 + \frac13}{1 - \frac12\cdot\frac13}
= \frac{5/6}{5/6} = 1 .
\]
दोनों आर्कटिक स्पर्शरेखाएँ रिहायश $\intoo{0}{\frac\pi4}$ (उनके तर्क अंदर हैं
$\intoo{0}{1}$), इसलिए $\alpha \in \intoo{0}{\frac\pi2}$; एकमात्र कोण
वहाँ स्पर्शरेखा के साथ $1$ है $\frac\pi4$.

मशीन: चलो $\beta = \arctan\frac15$. दोहरा कोण दो बार:
\[
\tan 2\beta = \frac{2/5}{1 - 1/25} = \frac{5}{12},
\qquad
\tan 4\beta = \frac{2 \cdot 5/12}{1 - 25/144} = \frac{120}{119}.
\]
तब
\[
\tan\Bigl(4\beta - \frac\pi4\Bigr)
= \frac{\frac{120}{119} - 1}{1 + \frac{120}{119}}
= \frac{1/119}{239/119} = \frac{1}{239}.
\]
जगह: $\beta < \arctan 1 = \frac\pi4$ और वास्तव में $\tan 4\beta =
\frac{120}{119}$ के करीब $1$ साथ $4\beta \in \intoo{0}{\frac\pi2}$
(तब से $\beta < \frac\pi8$, जैसा $\tan\frac\pi8 = \sqrt 2 - 1 > \frac15$);
इसलिए $4\beta - \frac\pi4 \in \intoo{-\frac\pi4}{\frac\pi4}$, कहाँ
$\arctan $ उलट $\tan $: $4\beta - \frac\pi4 =
\arctan\frac{1}{239}$, जो मशीन का सूत्र है।
\end{solution}

\begin{solution}{exo:b1:functions:9}
होने देना $ f(x) = x - \sin x $: $ f(0) = 0$ और $ f'(x) = 1 - \cos x \geq 0$,
इसलिए $ f \geq 0$ पर $\R_+$: $\sin x \leq x $.

होने देना $ g(x) = \sin x - x + \frac{x^3}{6}$: $ g(0) = 0$, $ g'(x) = \cos x -
1 + \frac{x^2}{2}$, $ g'(0) = 0$, $ g''(x) = -\sin x + x = f(x) \geq 0$
पर $\R_+$. इसलिए $ g'$ के साथ बढ़ रहा है $ g'(0) = 0$, इस तरह $ g' \geq 0$,
इस तरह $ g $ के साथ बढ़ रहा है $ g(0) = 0$: $ g \geq 0$, यानी\ $ x -
\frac{x^3}{6} \leq \sin x $ पर $\R_+$.

फलस्वरूप $0 \leq \frac{x - \sin x}{x^3} \leq \frac 16$ के लिए $ x >
0$: अनुपात पास फंसा हुआ है $\frac16$ का पैमाना, और
\cref{ch:b1:taylor} दिखाता है कि इसकी सीमा बिल्कुल सही है $\frac 16$.
\end{solution}

\begin{solution}{exo:b1:functions:10}
लिखना $ y = \arcsin x $, इसलिए $2x\sqrt{1 - x^2} = 2\sin y\cos y = \sin
2y $ और $ h(x) = \arcsin(\sin 2y) - 2y $.

के लिए $\abs x \leq \frac{\sqrt 2}{2}$: $2y \in
\intcc{-\frac\pi2}{\frac\pi2}$, इसलिए $\arcsin(\sin 2y) = 2y $ और $ h =
0$.

के लिए $ x \in \intoc{\frac{\sqrt 2}{2}}{1}$: $2y \in
\intoc{\frac\pi2}{\pi}$, और का कोण
$\intcc{-\frac\pi2}{\frac\pi2}$ साइन के साथ $\sin 2y $ है $\pi - 2y $:
इसलिए $ h(x) = \pi - 4\arcsin x $. (व्युत्पन्न द्वारा जांचें: वहां
$ h'(x) = \frac{2(1 - 2x^2)}{\abs{1 - 2x^2}\sqrt{1 - x^2}} -
\frac{2}{\sqrt{1-x^2}} = \frac{-4}{\sqrt{1 - x^2}}$, व्युत्पन्न
का $-4\arcsin x $; और कम से $ x = 1$, $ h(1) = \arcsin 0 - 2\cdot\frac\pi2
= -\pi = \pi - 4\cdot\frac\pi2$.)

के लिए $ x \in \intco{-1}{-\frac{\sqrt 2}{2}}$, की विचित्रता से $ h $:
$ h(x) = -\pi - 4\arcsin x $.
\end{solution}

\begin{solution}{exo:b1:functions:11}
$ g = \arctan \circ \sinh $ विषम, सख्ती से एक रचना है
कार्यों में वृद्धि, इसलिए यह अजीब और सख्ती से बढ़ रही है; जैसा $ x \to
+\infty $, $\sinh x \to +\infty $ इसलिए $ g(x) \to \frac\pi2$, और $ g $ एक है
पर आपत्ति $\intoo{-\frac\pi2}{\frac\pi2}$ (निरंतरता प्लस)
मध्यवर्ती मूल्य प्रमेय)। चेन नियम के साथ
\cref{prop:b1:functions:hyprules}~(1):
\[
g'(x) = \frac{\cosh x}{1 + \sinh^2 x} = \frac{\cosh x}{\cosh^2 x}
= \frac{1}{\cosh x} .
\]
निर्माण द्वारा $\tan g(x) = \sinh x $. तब से $ g(x) \in
\intoo{-\frac\pi2}{\frac\pi2}$ धनात्मक कोज्या है,
\begin{align*}
\cos g(x) &= \frac{1}{\sqrt{1 + \tan^2 g(x)}}
= \frac{1}{\sqrt{1 + \sinh^2 x}} = \frac{1}{\cosh x},\\
\sin g(x) &= \tan g(x) \cos g(x) = \frac{\sinh x}{\cosh x} = \tanh x .
\end{align*}
\end{solution}

\begin{solution}{exo:b1:functions:12}
\omterm{def:b1:logic:sets}{तय करना} $ u = \arctan 2$ और $ v = \arctan 3$; दोनों झूठ बोलते हैं
$\intoo{\frac\pi4}{\frac\pi2}$ (उनके तर्क अधिक हैं $1$), इसलिए $ u
+ v \in \intoo{\frac\pi2}{\pi}$. जोड़ सूत्र देता है
\[
\tan(u + v) = \frac{2 + 3}{1 - 6} = -1 ,
\]
और का अनोखा कोण $\intoo{\frac\pi2}{\pi}$ स्पर्शरेखा के साथ $-1$
है $\frac{3\pi}4$: इसलिए $\arctan 2 + \arctan 3 = \frac{3\pi}4$.
(आँख मूंदकर आवेदन करना $\arctan $ स्पर्शरेखा को दे देंगे
$-\frac\pi4$, द्वारा बंद $\pi $ --- स्थानीयकरण वही है जो बचाता है

संकेत है कि राशि निकल गई है
$\intoo{-\frac\pi2}{\frac\pi2}$.)जोड़ना $\arctan 1 = \frac\pi4$:
\[
\arctan 1 + \arctan 2 + \arctan 3 = \frac\pi4 + \frac{3\pi}4 = \pi .
\]
\end{solution}

\begin{solution}{pb:b1:functions:1}
\textbf{1.} $ f $ के साथ भिन्न-भिन्न कार्यों का एक भागफल है
गैर-लुप्त होने वाला हर $\intoo0{+\infty}$, और
\[
f'(t) = \frac{\frac1t \cdot t - \ln t}{t^2} = \frac{1 - \ln
t}{t^2} ,
\]
के लिए सकारात्मक $ t < \eu $, शून्य पर $ t = \eu $, के लिए नकारात्मक $ t > \eu $:
$ f $ पर सख्ती बढ़ती है $\intoc0\eu $, सख्ती से घटता है
$\intco\eu{+\infty}$, अधिकतम के साथ $ f(\eu) = \frac1\eu $.

\textbf{2.} जैसा $ t \to 0^+$: $\ln t \to -\infty $ और $\frac1t \to
+\infty $, इसलिए $ f(t) \to -\infty $. जैसा $ t \to +\infty $: $ f(t) \to 0$
से \omterm{prop:b1:functions:powerrules}{विकास तुलना} का \cref{prop:b1:functions:powerrules}
($\beta = \alpha = 1$). संकेत: का $\ln t $, इसलिए $ f < 0$ पर
$\intoo01$, $ f(1) = 0$, $ f > 0$ पर $\intoo1{+\infty}$. ग्राफ
से चढ़ता है $-\infty $, शून्य को पार करता है $1$, शिखर पर $(\eu,
\frac1\eu)$, फिर क्षय हो जाता है $0^+$.

\textbf{3.} $ f(4) = \frac{\ln 4}4 = \frac{2\ln 2}4 = \frac{\ln 2}2
= f(2)$.

\textbf{4.} भिन्नता तालिका और मध्यवर्ती मान द्वारा
प्रमेय (हाई स्कूल स्तर पर प्रयुक्त; औपचारिक रूप से
\cref{ch:b1:continuity}). के लिए $ c < 0$: समाधान केवल वहीं मौजूद हैं
$ f < 0$, यानी\ में $\intoo01$ कहाँ $ f $ सख्ती से बढ़ रहा है
पर आपत्ति $\intoo{-\infty}0$: बिल्कुल एक. के लिए $ c = 0$: केवल
$ t = 1$. के लिए $0 < c < \frac1\eu $: पर $\intoo1\eu $, $ f $ बढ़ता है
से $0$ को $\frac1\eu $: एक हल; पर $\intoo\eu{+\infty}$,
$ f $ से घट जाती है $\frac1\eu $ को $0$: और एक; कुल दो. के लिए
$ c = \frac1\eu $: केवल अधिकतम बिंदु $ t = \eu $. के लिए $ c >
\frac1\eu $: कोई नहीं।

\textbf{5.} के लिए $ x \neq \eu $, सख्त अधिकतमता देता है $ f(x) <
f(\eu)$, यानी\ $\frac{\ln x}x < \frac1\eu $, यानी\ $\eu\ln x < x $,
यानी\ $\ln(x^\eu) < \ln(\eu^x)$: $ x^\eu < \eu^x $. साथ $ x = \pi $:
$\eu^\pi > \pi^\eu $ (संख्यात्मक रूप से $23.14 > 22.46$).

\textbf{6.} के लिए $ x, y > 0$, दोनों पक्ष सकारात्मक हैं, इसलिए
\[
x^y = y^x \iff y\ln x = x\ln y \iff \frac{\ln x}x = \frac{\ln y}y
\iff f(x) = f(y) ,
\]
द्वारा विभाजित करना $ xy > 0$.

\textbf{7.} होने देना $ f(x) = f(y) = c $ साथ $ x \neq y $. अगर $ c \leq 0$ या
$ c = \frac1\eu $, प्रश्न 4 समीकरण कहता है $ f = c $ एक सिंगल है
समाधान: असंभव. इसलिए $0 < c < \frac1\eu $, और फिर से
प्रश्न 4 दो समाधान एक बिंदु हैं $\intoo1\eu $ और एक
का $\intoo\eu{+\infty}$: दोनों निर्देशांक अधिक हैं $1$ और वे
पैर फैलाकर बैठना $\eu $.

\textbf{8.} स्थानापन्न $ y = tx $ में $ y\ln x = x\ln y $:
\[
tx\ln x = x(\ln t + \ln x)
\iff (t - 1)\ln x = \ln t
\iff \ln x = \frac{\ln t}{t - 1} ,
\]
इसलिए $ x = t^{1/(t-1)}$ और $ y = tx = t^{1 + \frac1{t-1}} =
t^{t/(t-1)}$. इसके विपरीत, इन मूल्यों के लिए, $\ln y = \frac{t\ln
t}{t-1} = t\ln x $ और $ y = tx $, इसलिए $\frac{\ln y}y = \frac{t\ln
x}{tx} = \frac{\ln x}x $: एक समाधान। हर गैरतुच्छ समाधान है
कुछ अनुपात $ t = y/x \in \intoo0{+\infty} \setminus \{1\}$, तो
पैरामीट्रिजेशन पूरा हो गया है.

\textbf{9.} $ t = 2$: $ x = 2^{1/1} = 2$, $ y = 2^{2/1} = 4$. और
\[
x(1/t) = (1/t)^{\frac1{\frac1t - 1}}
= (t^{-1})^{\frac{t}{1 - t}}
= t^{\frac{t}{t - 1}} = y(t) ,
\]
तब $ y(1/t) = \frac1t\,x(1/t) = \frac{y(t)}t = x(t)$: द
पैरामीटर \omterm{def:b1:counting:objects}{परिवर्तन} $ t \mapsto 1/t $ निर्देशांकों की अदला-बदली करता है, जैसे कि
समीकरण की समरूपता मांग करती है।

\textbf{10.} जैसा $ t \to 1$: $\frac{\ln t}{t-1} \to 1$ (यह है
का अंतर भागफल $\ln $ पर $1$), इसलिए $ x \to \eu^1 = \eu $ और
$ y = tx \to \eu $. जैसा $ t \to +\infty $: $\ln x = \frac{\ln t}{t-1}
\to 0$ इसलिए $ x \to 1$, जबकि $\ln y = \frac{t\ln t}{t-1} \to
+\infty $ इसलिए $ y \to +\infty $. जैसा $ t \to 0^+$: $\ln x = \frac{\ln
t}{t-1} \to \frac{-\infty}{-1} = +\infty $ इसलिए $ x \to +\infty $,
जबकि $\ln y = \frac{t\ln t}{t-1} \to \frac{0}{-1} = 0$ इसलिए $ y \to
1$ (का उपयोग करते हुए $ t\ln t \to 0$, \cref{ex:b1:functions:xx}). शाखा
इसलिए अनंतस्पर्शी से चलता है $ y = 1$ (सबसे दांयीं ओर), ऊपर तक
$(\eu, \eu)$ विकर्ण पर, और अनंतस्पर्शी के अनुदिश $ x = 1$
(दूर शीर्ष) --- प्रश्न 9 द्वारा विकर्ण के बारे में सममित।

\textbf{11.} $\ln x(t) = g(t) = \frac{\ln t}{t-1}$, और
\[
g'(t) = \frac{\frac{t-1}t - \ln t}{(t-1)^2}
= \frac{h(t)}{(t-1)^2},
\qquad h(t) = 1 - \frac1t - \ln t .
\]
$ h(1) = 0$ और $ h'(t) = \frac1{t^2} - \frac1t = \frac{1 -
t}{t^2} < 0$ के लिए $ t > 1$: $ h < 0$ पर $\intoo1{+\infty}$, इसलिए $ g' <
0$ और $ x = \eu^g $ वहाँ (से) सख्ती से कम हो रहा है $\eu $ को
$1$, प्रश्न 10 से)।

\textbf{12.} प्रश्न 11 द्वारा, $ t \mapsto x(t)$ एक सख्ती है
से आपत्ति कम हो रही है $\intoo1{+\infty}$ पर $\intoo1\eu $;
लिखना $ t(x)$ इसके व्युत्क्रम (भी सख्ती से घटते हुए) और सेट के लिए
$\varphi(x) = y(t(x))$. टिप्पणी $ h < 0$ पर भी $\intoo01$ (वहाँ
$ h' > 0$ और $ h(1) = 0$), इसलिए $ x(\cdot)$ पर कम हो रहा है
$\intoo01$ भी; इस तरह $ y(t) = x(1/t)$ (प्रश्न 9) है
\emph{की बढ़ती} में $ t $ पर $\intoo1{+\infty}$, से $\eu $ को
$+\infty $. रचना: $\varphi = y \circ t(\cdot)$ सख्ती से है
से घट रहा है $\intoo1\eu $ पर $\intoo\eu{+\infty}$, और
$ x^{\varphi(x)} = \varphi(x)^x $ प्रश्न 8 द्वारा। अंत में, ए के लिए
सामान्य मूल्य $ c = f(x) \in \intoo0{\frac1\eu}$, जोड़ी $\{x,
\varphi(x)\}$ है \emph{} दो-तत्व समाधान \omterm{def:b1:logic:sets}{तय करना} का $ f = c $
(प्रश्न 4); का विस्तार $\varphi $ को $\intoo\eu{+\infty}$ के रूप में
उलटा \omterm{def:b1:logic:map}{नक्शा}, $\varphi(\varphi(x))$ दूसरे पर लौटता है (अर्थात्\
मूल) तत्व: $\varphi \circ \varphi = \mathrm{id}$.

\textbf{13.} अगर $(x, y)$ के साथ एक गैरतुच्छ पूर्णांक समाधान है $ x
< y $, प्रश्न 7 डालता है $ x \in \intoo1\eu $: वहाँ एकमात्र पूर्णांक है
है $ x = 2$. तब $ f(y) = f(2) = \frac{\ln2}2$ साथ $ y > \eu $; द्वारा
प्रश्न 4 समीकरण $ f = \frac{\ln2}2$ बिल्कुल एक है
परे समाधान $\eu $, और प्रश्न 3 इसे प्रदर्शित करता है: $ y = 4$. इस तरह
$(2, 4)$, और इसकी अदला-बदली ही एकमात्र चीज़ है।

\textbf{14.} $ t = 1 + \frac1n = \frac{n+1}n $ देता है $\frac1{t - 1}
= n $ और $\frac t{t-1} = n + 1$, इसलिए
\[
x_n = \Bigl(\frac{n+1}n\Bigr)^{n},
\qquad
y_n = \Bigl(\frac{n+1}n\Bigr)^{n+1},
\]
स्पष्ट रूप से तर्कसंगत, विशिष्ट ($ y_n = t\,x_n \neq x_n $), और $ n =
1$ देता है $ x_1 = 2$, $ y_1 = 4$.

\textbf{15.} $ t = y/x $ तर्कसंगत है और $> 1$; लिखना $ t - 1 =
\frac rs $ सबसे कम शब्दों में, तो $ t = \frac{s + r}s $ साथ $\gcd(s +
r, s) = \gcd(r, s) = 1$. प्रश्न 8 देता है $ x = t^{s/r}$, इस तरह
\[
x^r = t^s = \frac{(s+r)^s}{s^s} ,
\]
न्यूनतम पदों में एक भिन्न (कोई भी अभाज्य दोनों को विभाजित नहीं करता है $ s + r $ और
$ s $). लिखना $ x = \frac pq $ सबसे कम शब्दों में, $ x^r = \frac{p^r}
{q^r}$ निम्नतम शर्तों में भी है, और कम की विशिष्टता से
प्रतिनिधित्व: $ p^r = (s+r)^s $ और $ q^r = s^s $. तुलना करें
किसी भी अभाज्य का प्रतिपादक $\ell $ में $ q^r = s^s $: $ r \cdot
v_\ell(q) = s \cdot v_\ell(s)$, इसलिए $ r $ विभाजित $ s\,v_\ell(s)$;
तब से $\gcd(r, s) = 1$, $ r $ विभाजित $ v_\ell(s)$ प्रत्येक प्राइम के लिए
$\ell $ (अद्वितीय गुणनखंडन, स्वीकार किया गया; सिद्ध किया गया
\cref{ch:b1:arith}), इसलिए $ s = b^r $ एक पूर्णांक के लिए $ b $. वैसा ही
पर तर्क $ p^r = (s+r)^s $ देता है $ s + r = a^r $.

\textbf{16.} कल्पना करना $ a^r - b^r = r $ पूर्णांकों के साथ $ a > b \geq 1$
और $ r \geq 2$. तब $ a \geq b + 1$, तो द्विपद प्रमेय द्वारा
\[
a^r - b^r \geq (b+1)^r - b^r
= \sum_{k=0}^{r-1}\binom rk b^k
\geq \binom r0 + \binom r{r-1} b^{r-1}
\geq 1 + r > r ,
\]
एक विरोधाभास. प्रश्न 15 से, $ a^r - b^r = (s + r) - s = r $
बल $ r = 1$: $ t = 1 + \frac1s $, और समाधान है $(x_s,
y_s)$ प्रश्न 14 का। बदली हुई जोड़ियों के साथ,
वर्गीकरण पूरा हो गया है.

\textbf{17.} $ f\bigl(\tfrac94\bigr) = \frac{\ln 2.25}{2.25} =
\frac{0.81093}{2.25} = 0.3604$ और $ f\bigl(\tfrac{27}8\bigr) =
\frac{\ln 3.375}{3.375} = \frac{1.21640}{3.375} = 0.3604$ (चार
दशमलव): बराबर, जैसा कि निर्माण का वादा है --- इसलिए
$\bigl(\tfrac94\bigr)^{27/8} = \bigl(\tfrac{27}8\bigr)^{9/4}$.

\textbf{18.} पैरामीटर $ t_n = 1 + \frac1n $ सख्ती से कमी करें
को $1$. तब से $ x(\cdot)$ पर सख्ती से कमी आ रही है
$\intoo1{+\infty}$ (प्रश्न 11), $ x_n = x(t_n)$ सख्ती से है
\emph{की बढ़ती}; तब से $ y(\cdot)$ वहां सख्ती बढ़ती जा रही है
(प्रश्न 12), $ y_n = y(t_n)$ सख्ती से है \emph{घटते}. द्वारा
प्रश्न 7, $ x_n \in \intoo1\eu $ और $ y_n \in
\intoo\eu{+\infty}$: इसलिए $ x_n < \eu < y_n $ हरएक के लिए $ n $. अंत में
$ t_n \to 1$ और प्रश्न 10 दीजिए $ x_n \to \eu $ और $ y_n \to \eu $.
का शास्त्रीय एकस्वर अभिसरण $\bigl(1 +
\frac1n\bigr)^n $ शाखा की ज्यामिति से बाहर हो जाता है, नहीं के साथ
नई असमानता.

\textbf{19.} अगर $\eu \leq a < b $: $ f $ सख्ती से कम हो रहा है
$\intco\eu{+\infty}$ देता है $ f(a) > f(b)$, यानी\ $ b\ln a > a\ln
b $, यानी\ $ a^b > b^a $. अगर $1 < a < b \leq \eu $: $ f $ सख्ती से
वृद्धि देता है $ f(a) < f(b)$, इस तरह $ a^b < b^a $. मिश्रित मामला:
$2 < \eu < 3$ और $2^3 = 8 < 9 = 3^2$; लेकिन $2 < \eu < 5$ और
$2^5 = 32 > 25 = 5^2$: दोनों परिणाम घटित होते हैं, इसका निर्णय किस पक्ष द्वारा किया जाता है
शाखा का बिंदु $(a, b)$ गिरता है.

\textbf{20.} शाखा विकर्ण पर मिलती है $\eu $ और $\varphi $
वहां निरंतर विस्तार होता है $\varphi(\eu) = \eu $.
फर्क $\varphi(\varphi(x)) = x $ पर $ x = \eu $ द्वारा
श्रृंखला नियम: $\varphi'(\varphi(\eu))\,\varphi'(\eu) =
\varphi'(\eu)^2 = 1$, इसलिए $\varphi'(\eu) = \pm1$; $\varphi $ है
घट रहा है, इसलिए $\varphi'(\eu) = -1$. शाखा पार करती है
ढलान के साथ विकर्ण $-1$: लंबवत्.

\textbf{21.} $ y = 3x $ मामला यह है $ t = 3$: $ x = 3^{1/2} = \sqrt3$
और $ y = 3^{3/2} = 3\sqrt3$. जाँच करना: $ f(\sqrt3) =
\frac{0.54931}{1.73205} = 0.3171$ और $ f(3\sqrt3) =
\frac{\ln 5.19615}{5.19615} = \frac{1.64792}{5.19615} = 0.3171$:
चार दशमलव के बराबर, तो $(\sqrt3)^{3\sqrt3} =
(3\sqrt3)^{\sqrt3}$.

\textbf{22.} $ n^{1/n} = \eu^{f(n)}$, और $\exp $ बढ़ रहा है, इसलिए
आदेश वह है $ f(n)$: $ f(3) = \frac{\ln3}3 \approx
0.3662$ धड़कता है $ f(2) = f(4) = \frac{\ln2}2 \approx 0.3466$ (प्रश्न
3) और $ f(5) \approx 0.3219$. सबसे बड़ा है $\sqrt[3]3$, और
बराबर जोड़ी है $\sqrt2 = \sqrt[4]4$ ---पूर्णांक युग्म $(2, 4)$
एक और भेष में.

\textbf{23.} समाधान \omterm{def:b1:logic:sets}{तय करना} विकर्ण का मिलन है $\{(x,
x) : x > 0\}$ और एक शाखा, विकर्ण के बारे में सममित,
अनंतस्पर्शी से सख्ती से घट रहा है $ x = 1$ (जैसा $ y \to
+\infty $) अनंतस्पर्शी को $ y = 1$ (जैसा $ x \to +\infty $), पार करना
विकर्ण बिल्कुल एक बार, पर $(\eu, \eu)$, ढलान के साथ $-1$
वहाँ। शाखा पर पूर्णांक बिंदु बैठें $(2, 4)$ और $(4, 2)$
--- केवल वही --- और तर्कसंगत बिंदु $(x_n, y_n)$,
जो शाखा के साथ-साथ नीरस रूप से आगे बढ़ते हैं $(\eu, \eu)$
उस तक पहुंचे बिना ($\eu $ तर्कहीन है; शाखा का
तर्कसंगत बिंदु एक अपरिमेय कोने पर जमा होते हैं)।

\textbf{24.} (मैं) \omterm{prop:b1:functions:powerrules}{विकास तुलना} की सीमा दी $ f $ पर
$+\infty $ (प्रश्न 2) और $ t\ln t \to 0$ (प्रश्न 10),
विविधता तालिका और अनंतस्पर्शी दोनों को आकार देना। (ii) द
आक्षेप के माध्यम से मध्यवर्ती मूल्य प्रमेय \omterm{def:b1:logic:statement}{कथन},
भिन्नता तालिका को सटीक समाधान गणना में परिवर्तित किया
(प्रश्न 4) और व्युत्क्रम के अस्तित्व में \omterm{def:b1:logic:map}{नक्शा} $ t(x)$
(प्रश्न 12). (iii) अद्वितीय गुणनखंडन ने दोनों को संचालित किया
प्रश्न 15, अंकगणितीय हृदय की निम्नतम-शर्तों की पहचान
तर्कसंगत-बिंदु वर्गीकरण का.

\textbf{25.} एक व्युत्पन्न गणना --- का संकेत $1 - \ln
t $ --- सब कुछ उत्पन्न किया: प्रत्येक स्तर के लिए समाधान गणना
$ c $, शाखा का आकार और स्पर्शोन्मुख, चरम
असमानता $\eu^x \geq x^\eu $, पूर्णांक का वर्गीकरण और
तर्कसंगत समाधान, और एकस्वर अभिसरण $\bigl(1 +
\frac1n\bigr)^n $. यह विविधता के साथ सोचने की अर्थव्यवस्था है
तालिकाएँ: एक-आयामी अध्ययन दो-चर समीकरण को हल करता है
क्योंकि समीकरण एक ही फ़ंक्शन के माध्यम से कारक बनता है।
\cref{ch:b1:seq} का अभिसरण फिर से करेगा $(x_n)$ के साथ
मोनोटोन अनुक्रमों का सामान्य सिद्धांत; \cref{ch:b1:derivative}
माध्य मान प्रमेय पर कठोरता से भिन्नता तालिकाएँ ढूँढता है; और
\cref{ch:b1:taylor} वह आपूर्ति करता है जो तालिका नहीं दे सकती ---
\emph{रफ़्तार} अभिसरण का $ x_n \to \eu $ (यह व्यवस्थित है
$1/n$).
\end{solution}
