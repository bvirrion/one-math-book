\begin{solution}{exo:b1:multivar:1}
$\dfrac{\partial f}{\partial x} = 2xy + y\,\eu^{xy}$,
$\dfrac{\partial f}{\partial y} = x^2 + x\,\eu^{xy}$.

$\dfrac{\partial g}{\partial x} = \dfrac{2x}{x^2+y^2}$,
$\dfrac{\partial g}{\partial y} = \dfrac{2y}{x^2+y^2}$.

$\dfrac{\partial h}{\partial x} = \dfrac{-y/x^2}{1 + y^2/x^2} =

$\dfrac{\partial h}{\partial y} = \dfrac{x}{x^2+y^2}$.
\end{solution}

\begin{solution}{exo:b1:multivar:3}
साथ में $ y = mx $: $ f(x, mx) = \dfrac{m x^3}{x^4 + m^2 x^2} =
\dfrac{mx}{x^2 + m^2} \to 0$ (के लिए $ m \neq 0$; साथ में $ y = 0$ और यह
$ y $-अक्ष, $ f = 0$). तो हर सीधी-रेखा की \omterm{def:b1:topology:closure}{सीमा} है $0$. लेकिन पर
परवलय $ y = x^2$:
\[
f(x, x^2) = \frac{x^4}{x^4 + x^4} = \frac12 :
\]
क्रम $\bigl(\frac1n, \frac{1}{n^2}\bigr) \to (0,0)$ है
$ f \to \frac12 \neq 0$. नहीं निरंतर: पंक्तियाँ पर्याप्त नहीं हैं
दो-चर सीमाओं का परीक्षण करें।
\end{solution}

\begin{solution}{exo:b1:multivar:5}
द्वारा \cref{thm:b1:multivar:chain} साथ $(x, y) = (\cos t, \sin t)$:
\[
g'(t) = -\sin t\,\frac{\partial f}{\partial x}(\cos t, \sin t)
+ \cos t\,\frac{\partial f}{\partial y}(\cos t, \sin t)
= \bigl\langle \nabla f,\ (-\sin t, \cos t)\bigr\rangle .
\]
की चरम \omterm{def:b1:topology:closure}{सीमा} पर $ g $, $ g'(t) = 0$: $\nabla f $ है ओर्थोगोनल को
स्पर्शरेखा सदिश $(-\sin t, \cos t)$ वृत्त के, अत: \omterm{thm:b1:matrices:conjugation}{समान}ांतर
त्रिज्या वेक्टर $(\cos t, \sin t)$ (विमान ओर्थोगोनल
\omterm{def:b1:vspaces:sum}{पूरक} एक यूनिट वेक्टर की वह रेखा है तक फैला, लिया गया
लंबवत)। यह लैग्रेंज का सबसे सरल उदाहरण है
गुणक.
\end{solution}

\begin{solution}{exo:b1:multivar:6}
$ f $: $\nabla f = (2x + y - 3,\; x + 2y) = 0$: $ y = -\frac x2$ और
$2x - \frac x2 = 3$: $ x = 2$, $ y = -1$. दूसरा आदेश: $ r = 2$, $ s =
1$, $ t = 2$: $ rt - s^2 = 3 > 0$, $ r > 0$: स्थानीय (वास्तव में वैश्विक ---
द्विघात) न्यूनतम पर $(2, -1)$, कीमत $ f(2,-1) = -3$.

$ g $: $\nabla g = (2x,\; -2y + 4) = 0$: बिंदु $(0, 2)$; $ r = 2$, $ s
= 0$, $ t = -2$: $ rt - s^2 = -4 < 0$: काठी.

$ h $: $\nabla h = \bigl(4x^3 - 4(x - y),\; 4y^3 + 4(x-y)\bigr) = 0$.
जोड़ना: $ x^3 + y^3 = 0$, इसलिए $ y = -x $; प्रतिस्थापित करना: $4x^3 - 8x = 0$:
$ x \in \{0, \pm\sqrt2\}$. महत्वपूर्ण बिंदु: $(0,0)$,
$(\sqrt2, -\sqrt2)$, $(-\sqrt2, \sqrt2)$.
पर $(\pm\sqrt2, \mp\sqrt2)$: $ r = 12x^2 - 4 = 20$, $ s = 4$, $ t =
20$: $ rt - s^2 > 0$, $ r > 0$: स्थानीय मिनिमा (मान $ h = 4 + 4 - 16 =
-8$). पर $(0,0)$: $ r = t = -4$, $ s = 4$: $ rt - s^2 = 0$: चुपचाप।
परीक्षण करना: $ h(x, x) = 2x^4 > 0$ और $ h(x, -x) = 2x^4 - 8x^2 < 0$ के लिए
छोटा $ x \neq 0$: प्रत्येक में दोनों चिह्न \omterm{def:b1:topology:open}{अड़ोस-पड़ोस} --- कोई चरम \omterm{def:b1:topology:closure}{सीमा} नहीं
मूल पर.
\end{solution}

\begin{solution}{exo:b1:multivar:7}

दूसरा भाग गायब हो जाता है। $\eu^x\cos y $: $\frac{\partial^2}{\partial
x^2} = \eu^x\cos y $, $\frac{\partial^2}{\partial y^2} =

= \frac{2(y^2 - x^2)}{(x^2+y^2)^2},
\]
और यह $ y $-संस्करण इसका विपरीत है: योग $0$.
\end{solution}

\begin{solution}{exo:b1:multivar:8}
\emph{प्रक्षेपण:} विमान $ P $ सामान्य है $ n = (1, 2, -1)$ और
से गुजरता है $ A = (4, 0, 0)$. उद्गम का निकटतम बिंदु है
$ O $ प्रक्षेपित: $ p = O + \frac{\langle A - O, n\rangle}{\norm
n^2}\,n = \frac{4}{6}(1,2,-1) = \bigl(\frac23, \frac43,
-\frac23\bigr)$, दूरी पर $\frac{4}{\sqrt 6}$. (सूत्र: दूरी
उत्पत्ति से तल तक $\langle X, n \rangle = c $ है
$\frac{\abs c}{\norm n}$ साथ $ c = 4$.)

\emph{न्यूनतमकरण:} $ f(x,y) = x^2 + y^2 + (x + 2y - 4)^2$. लिखना
$ w = x + 2y - 4$ अंतिम कारक के लिए:
\[
\nabla f = \bigl(2x + 2w,\; 2y + 4w\bigr) = 0
\iff x = -w \text{ and } y = -2w .
\]
की परिभाषा में प्रतिस्थापित करना $ w $: $ w = -w - 4w - 4$, इसलिए $ w =
-\frac23$, देना $ x = \frac23$, $ y = \frac43$, और $ z = w =
-\frac23$. के \omterm{thm:b1:matrices:conjugation}{समान} बिंदु अनुमान, दूरी पर
$\sqrt{\frac49 + \frac{16}{9} + \frac49} = \frac{2\sqrt 6}{3} =
\frac{4}{\sqrt6}$; तब से यह न्यूनतम है $ f \to \infty $ अनंत पर
(एक सकारात्मक-निश्चित द्विघात प्लस \omterm{def:b1:linmaps:def}{रेखीय} शर्तें)।
\end{solution}

\begin{solution}{exo:b1:multivar:9}
साथ $\rho = \sqrt{x^2+y^2}$: $\frac{\partial \rho}{\partial x} =
\frac{x}{\rho}$, इसलिए
\[
\frac{\partial f}{\partial x} = \varphi'(\rho)\,\frac{x}{\rho},
\qquad
\frac{\partial^2 f}{\partial x^2}
= \varphi''(\rho)\,\frac{x^2}{\rho^2}
+ \varphi'(\rho)\,\Bigl(\frac{1}{\rho} -
\frac{x^2}{\rho^3}\Bigr),
\]
(भागफल और श्रृंखला नियम)। सममिति जोड़ना $ y $-अभिव्यक्ति:
द $\varphi''$ शर्तें एकत्रित करें $\frac{x^2 + y^2}{\rho^2} = 1$, द
$\varphi'$ शर्तें एकत्रित करें $\frac{2}{\rho} - \frac{x^2 +
y^2}{\rho^3} = \frac{1}{\rho}$:
\[
\Delta f = \varphi'' + \frac{\varphi'}{\rho} .
\]
रेडियल हार्मोनिक: हल करें $\varphi'' + \frac{\varphi'}{\rho} = 0$: द
प्रथम-क्रम समीकरण $ u' + \frac u\rho = 0$ के लिए $ u = \varphi'$
देता है $ u = \frac{c}{\rho}$ (\cref{thm:b1:diffeq:homogeneous1}),
तब $\varphi = c\ln\rho + d $. रेडियल हार्मोनिक कार्य हैं
$ c\,\ln\sqrt{x^2 + y^2} + d $ --- की लघुगणकीय क्षमता
\cref{exo:b1:multivar:7} और स्थिरांक, और कुछ नहीं।
\end{solution}

\begin{solution}{exo:b1:multivar:11}
\begin{enumerate}
  \item तय के लिए $ y $, एक-चर फ़ंक्शन $ x \mapsto
        f(x,y)$ शून्य है यौगिक पर $\R $, इसलिए स्थिर है
        (\cref{thm:b1:derivative:mvt}): $ f(x, y) = f(0, y)$ सभी के लिए $ x $: $ f $ पर ही निर्भर करता है $ y $.
  \item होने देना $ g(u, v) = f\bigl(\frac{u+v}2, \frac{u-v}2\bigr)$.
        \omterm{def:b1:series:def}{शृंखला} नियम द्वारा (\cref{thm:b1:multivar:chain}, लागू किया गया
        चर में $ v $ साथ $ u $ जमा हुआ):
        \[
        \frac{\partial g}{\partial v}
        = \frac12\,\frac{\partial f}{\partial x}
        - \frac12\,\frac{\partial f}{\partial y} = 0 .
        \]
        द्वारा (1), $ g $ पर ही निर्भर करता है $ u $: $ g(u, v) = \varphi(u)$
        साथ $\varphi $ कक्षा का $ C^1$, और \omterm{def:b1:counting:objects}{परिवर्तन} को उलट देना
        चर का ($ u = x + y $, $ v = x - y $),
        \[
        f(x, y) = \varphi(x + y).
        \]
        इसके विपरीत प्रत्येक ऐसे $ f $ संतुष्ट $ f_x = f_y =
        \varphi'$: समाधान बिल्कुल वही हैं $ C^1$ कार्य
        का $ x + y $.
\end{enumerate}
\end{solution}

\begin{solution}{pb:b1:multivar:1}
\textbf{1.} दाहिनी ओर का विस्तार:
\[
r\Bigl(h + \frac srk\Bigr)^{\!2} + \frac{rt - s^2}{r}k^2
= rh^2 + 2shk + \frac{s^2}{r}k^2 + \frac{rt - s^2}{r}k^2
= q(h,k) .
\]
अगर $ rt - s^2 > 0$: दोनों वर्गों पर कारक चिन्ह अंकित है $ r $,
और $ q(h,k) = 0$ बल $ k = 0$ तब $ h = 0$: का सख्त संकेत $ r $
मूल से बाहर. अगर $ rt - s^2 < 0$: $ q(1, 0) = r $ जबकि
$ q\bigl(-\frac sr, 1\bigr) = \frac{rt - s^2}{r}$ के पास है
विपरीत चिह्न: दोनों चिह्न घटित होते हैं।

\textbf{2.} अगर $ r = 0 \neq t $, की भूमिकाओं का आदान-प्रदान करें $ h $ और $ k $
(पूर्ण वर्ग इंच) $ k $), \omterm{thm:b1:matrices:conjugation}{समान} निष्कर्षों के साथ; टिप्पणी
$ rt - s^2 = -s^2 < 0$ जैसे ही $ s \neq 0$, सचमुच $ q =
2shk + tk^2$ तब दोनों संकेत लेता है ($ k $ के विरुद्ध छोटा $ h $). अगर
$ r = t = 0$: $ q = 2shk $, दोनों संकेत iff $ s \neq 0$, यानी\ आईएफएफ
$ rt - s^2 = -s^2 < 0$. सारांश: $ q $ दोनों संकेत लेता है $\iff rt -
s^2 < 0$; $ q $ केवल मूल पर ही लुप्त हो जाता है (निश्चित) $\iff rt -
s^2 > 0$ (जो मजबूर करता है $ r \neq 0$, तब से $ r = 0$ देता है $ q(1,0) =
0$).

\textbf{3.} साथ $ f_x = rx + sy + \beta $ और $ f_y = sx + ty +
\gamma $, सीधा विस्तार देता है
\[
f(a + h, b + k) = f(a, b) + h\,f_x(a,b) + k\,f_y(a,b)
+ \tfrac12 q(h, k),
\]
बिना किसी उच्चतर पद के (फ़ंक्शन है a \omterm{def:b1:poly:def}{बहुपद} डिग्री का
$2$). ए पर महत्वपूर्ण बिन्दू  \omterm{def:b1:linmaps:def}{रेखीय} भाग गायब हो जाता है: $ f(X_0 + H)
- f(X_0) = \frac12 q(H)$ \emph{बिल्कुल}, तो का संकेत अध्ययन
प्रश्न 1-2 वर्गीकृत करता है: सख्त वैश्विक न्यूनतम ($ rt - s^2 >
0$, $ r > 0$), सख्त वैश्विक अधिकतम ($ rt - s^2 > 0$, $ r < 0$),
काठी--- प्रत्येक में दोनों चिन्ह \omterm{def:b1:topology:open}{अड़ोस-पड़ोस} --- कब $ rt - s^2 <
0$. ये साबित होता है \cref{met:b1:multivar:monge} द्विघात के लिए
कार्य.

\textbf{4.} फार्म $ q - c(h^2 + k^2)$ डेटा है $ r - c $, $ s $,
$ t - c $. साथ $ c = \frac{rt - s^2}{r + t}$ (टिप्पणी $ t > 0$ तब से
$ rt > s^2 \geq 0$ और $ r > 0$):
\[
(r - c)(t - c) - s^2 = rt - s^2 - c(r + t) + c^2 = c^2 \geq 0,
\]
और $ r - c \geq 0$ ($ c \leq r \iff rt - s^2 \leq r^2 + rt $,
सत्य)। अगर $ r - c > 0$, प्रश्न 1 का पूरा वर्ग दिखाता है
$ q - c(h^2 + k^2) \geq 0$; अगर $ r - c = 0$, तब $ s = 0$ (से
$-s^2 = -c^2 \leq \dots $ प्रदर्शित मात्रा को बाध्य करना
$\geq 0$ पहले कारक के साथ $0$) और फॉर्म है $(t - c)k^2 \geq
0$. दोनों ही मामलों में $ q(h,k) \geq c(h^2 + k^2)$.

\textbf{5.} प्रश्न 3-4 से, $ f(X) = f(X_0) + \frac12 q(X -
X_0) \geq f(X_0) + \frac c2\norm{X - X_0}^2 \to +\infty $ जैसा
$\norm X \to \infty $: $ f $ ज़बरदस्ती है, और असमानता है
के लिए सख्त $ X \neq X_0$: द महत्वपूर्ण बिन्दू अद्वितीय वैश्विक है
न्यूनतम करनेवाला। घन के लिए $ f = x^3 + y^3 - 3xy $ का
\cref{ex:b1:multivar:extrema}, ऐसा कोई निष्कर्ष नहीं है: $ f(x,0)
= x^3 \to -\infty $, और स्थानीय न्यूनतम पर $(1,1)$ नहीं है
वैश्विक --- द्विघात विस्तार की सटीकता विफल रही है।

\textbf{6.} चुकता मानदंड का विस्तार:
\[
f(u,v) = u^2\norm{C_1}^2 + 2uv\,\langle C_1, C_2\rangle +
v^2\norm{C_2}^2 - 2u\,\langle C_1, b\rangle - 2v\,\langle C_2,
b\rangle + \norm b^2 ,
\]
का एक द्विघात कार्य $(u, v)$ जिसका द्विघात भाग है
$\frac12(ru^2 + 2suv + tv^2)$ साथ $ r = 2\norm{C_1}^2$, $ s =
2\langle C_1, C_2\rangle $, $ t = 2\norm{C_2}^2$.

\textbf{7.} $ rt - s^2 = 4\bigl(\norm{C_1}^2\norm{C_2}^2 -
\langle C_1, C_2\rangle^2\bigr) \geq 0$ कॉची-श्वार्ज़ द्वारा
(\cref{thm:b1:euclid:cs}), बिल्कुल समानता के साथ जब $ C_1, C_2$
आनुपातिक हैं (या एक शून्य है), यानी\ जब जोड़ी है
जुड़ा हुआ. इसलिए $ rt - s^2 > 0 \iff (C_1, C_2)$ \omterm{def:b1:vspaces:free}{मुक्त}.

\textbf{8.} $\nabla f = 0$ पढ़ता
\[
\norm{C_1}^2 u + \langle C_1, C_2\rangle v = \langle C_1,
b\rangle,
\qquad
\langle C_1, C_2\rangle u + \norm{C_2}^2 v = \langle C_2,
b\rangle,
\]
 \omterm{pb:b1:multivar:1}{सामान्य समीकरण} बाईं ओर ग्राम मैट्रिक्स के साथ; कहते हैं
$\langle uC_1 + vC_2 - b,\ C_i\rangle = 0$ के लिए $ i = 1, 2$, अर्थात\
$ b - p \perp \operatorname{Vect}(C_1, C_2)$ के लिए $ p = uC_1 +
vC_2$. भाग I (प्रश्न 3,5) द्वारा अद्वितीय महत्वपूर्ण बिन्दू है
अद्वितीय वैश्विक मिनिमाइज़र, और द्वारा
\cref{thm:b1:euclid:projection} लक्षण वर्णन ``$ p \in
F $, $ b - p \perp F $''पहचानता है $ p $ के रूप में \omterm{thm:b1:euclid:projection}{ओर्थोगोनल
प्रक्षेपण} का $ b $ पर $ F = \operatorname{Vect}(C_1, C_2)$.

\textbf{9.} $\norm{b - p}^2 = \norm b^2 - 2\langle b, p\rangle +
\norm p^2$, और $\langle p, b - p\rangle = 0$ देता है $\norm p^2 =
\langle p, b\rangle $: न्यूनतम बराबर है $\norm b^2 - \langle p,
b\rangle $. पाइथागोरस: $\norm b^2 = \norm p^2 + \norm{b - p}^2$
--- डेटा वेक्टर अपने स्पष्ट भाग में विभाजित हो जाता है $ p $ और उसके
अवशिष्ट भाग $ b - p $, ओर्थोगोनल एक दूसरे से।

\textbf{10.} $ E(\alpha, \beta) = \norm{\alpha C_1 + \beta C_2 -
b}^2$ बताए गए के साथ $ C_1, C_2, b $; \omterm{pb:b1:multivar:1}{सामान्य समीकरण} का
प्रश्न 8 प्रवेशवार हैं,
\[
n\alpha + \Bigl(\sum x_i\Bigr)\beta = \sum y_i,
\qquad
\Bigl(\sum x_i\Bigr)\alpha + \Bigl(\sum x_i^2\Bigr)\beta = \sum
x_i y_i .
\]

\textbf{11.} द्वारा विभाजित करना $ n $: $\alpha + \beta\bar x = \bar y $
और $\alpha\bar x + \beta\bigl(v_x + \bar x^2\bigr) = c_{xy} +
\bar x\bar y $. स्थानापन्न $\alpha = \bar y - \beta\bar x $ में
दूसरा: $\beta v_x = c_{xy}$, इसलिए
\[
\beta = \frac{c_{xy}}{v_x}, \qquad \alpha = \bar y - \beta\bar
x ,
\]
और पहला समीकरण बिल्कुल यही कहता है $(\bar x, \bar y)$
लाइन पर पड़ा है.

\textbf{12.} $ v_x = \frac1n\sum(x_i - \bar x)^2 \geq 0$, शून्य
यदि हर $ x_i $ के बराबर होती है $\bar x $. और $ rt - s^2 = 4\bigl(n\sum
x_i^2 - (\sum x_i)^2\bigr) = 4n^2 v_x $: प्रश्न 7 की स्वतंत्रता
शर्त है $ v_x > 0$, यानी\  $ x_i $ सभी \omterm{thm:b1:matrices:conjugation}{समान} नहीं.

\textbf{13.} साथ $\alpha = \bar y - \beta\bar x $, केन्द्र
डेटा ($\tilde x_i = x_i - \bar x $, $\tilde y_i = y_i - \bar y $):
\[
E(\alpha, \beta) = \sum_i(\tilde y_i - \beta\tilde x_i)^2
= n\,v_y - 2\beta\,n\,c_{xy} + \beta^2 n\,v_x ,
\]
पर न्यूनतम किया गया $\beta = c_{xy}/v_x $ मूल्य के साथ $ E_{\min} =
n\bigl(v_y - \frac{c_{xy}^2}{v_x}\bigr) = n v_y(1 - \rho^2)$.
तब से $ E_{\min} \geq 0$ और $ v_y > 0$: $\rho^2 \leq 1$, और
$\abs\rho = 1$ आईएफएफ $ E_{\min} = 0$, यानी\ यदि प्रत्येक अवशिष्ट
गायब हो जाता है: बिंदु बिल्कुल रेखा पर स्थित होते हैं।

\textbf{14.} $ n = 4$: $\bar x = \frac64 = 1.5$, $\bar y = 2$,
$\sum x_i^2 = 14$ इसलिए $ v_x = 3.5 - 2.25 = 1.25$, $\sum x_iy_i =
0 + 1 + 6 + 12 = 19$ इसलिए $ c_{xy} = 4.75 - 3 = 1.75$. इस तरह
\[
\beta = \frac{1.75}{1.25} = 1.4,
\qquad
\alpha = 2 - 1.4\times1.5 = -0.1 :
\qquad y = 1.4\,x - 0.1 .
\]
उपयुक्त मान $-0.1,\ 1.3,\ 2.7,\ 4.1$; बच गया $0.1,\ -0.3,\
0.3,\ -0.1$; $ E_{\min} = 0.01 + 0.09 + 0.09 + 0.01 = 0.2$
(जाँच करना: $ v_y = \frac{26}4 - 4 = 2.5$ और $4(2.5 -
\frac{1.75^2}{1.25}) = 4(2.5 - 2.45) = 0.2$).

\textbf{15.} दो \omterm{pb:b1:multivar:1}{सामान्य समीकरण} प्रश्न 10 के हैं
बिलकुल $\sum_i\varepsilon_i = 0$ और $\sum_i x_i\varepsilon_i =
0$: अवशिष्ट वेक्टर है ओर्थोगोनल को $ C_1 = (1, \dots, 1)$
और को $ C_2 = (x_i)$ --- पूरे मॉडल स्थान के लिए. में
विशेष रूप से इष्टतम रेखा हमेशा अपनी त्रुटियों को संतुलित करती है: वे
शून्य का योग.

\textbf{16.} $\frac{\dd}{\dd\alpha}\sum(y_i - \alpha)^2 =
-2\sum(y_i - \alpha) = 0$ देता है $\alpha = \bar y $ (और दूसरा यौगिक $2n > 0$ इसे वैश्विक न्यूनतम बनाता है
फलन एक चर में एक प्रबल द्विघात है)। न्यूनतम
मूल्य $\sum(y_i - \bar y)^2 = n v_y $: विचरण है
एक स्थिर मॉडल की अघुलनशील द्विघात त्रुटि।

\textbf{17.} $\frac{\dd}{\dd\beta}\sum(y_i - \beta x_i)^2 =
-2\sum x_i(y_i - \beta x_i) = 0$ देता है $\beta_0 = \frac{\sum
x_iy_i}{\sum x_i^2}$. दोनों ढलानों को केन्द्र के ऊपर लिखना
मात्राएँ: $\beta_0 = \frac{c_{xy} + \bar x\bar y}{v_x + \bar
x^2}$ के बराबर होती है $\frac{c_{xy}}{v_x}$ आईएफएफ $ c_{xy}\bar x^2 = \bar
x\bar y\,v_x $, यानी\ आईएफएफ $\bar x = 0$ (केंद्रित एब्सिस्सा) या
$\bar y = \beta\bar x $ --- बाद वाला अर्थ $\alpha = 0$: द
पूर्ण \omterm{pb:b1:multivar:1}{प्रतिगमन लाइन} पहले से ही मूल से होकर गुजरता है.

\textbf{18.} $ g(\tau) = \norm{M - P_0}^2 - 2\tau\langle M - P_0,
w\rangle + \tau^2$ न्यूनतम है $\tau^* = \langle M - P_0,
w\rangle $, मूल्य के साथ $ d(M,D)^2 = \norm{M - P_0}^2 - \langle M -
P_0, w\rangle^2$. विमान में, पूरा $ w $ एक में \omterm{def:b1:euclid:orthogonal}{ऑर्थोनॉर्मल}
\omterm{def:b1:vspaces:free}{आधार} $(w, n)$ साथ $ n = \frac{(a, b)}{\sqrt{a^2+b^2}}$ (इकाई
का सामान्य $ D $): तब $\norm{M - P_0}^2 = \langle M - P_0,
w\rangle^2 + \langle M - P_0, n\rangle^2$, इसलिए $ d(M, D) =
\abs{\langle M - P_0, n\rangle}$, और साथ में $ aP_{0x} + bP_{0y} =
-c $:
\[
d = \frac{\abs{a x_0 + b y_0 + c}}{\sqrt{a^2 + b^2}} .
\]

\textbf{19.} से $ E_{\min}/n = v_y - \frac{c_{xy}^2}{v_x}$ और
$\beta^2 v_x = \frac{c_{xy}^2}{v_x}$:
\[
v_y = \beta^2 v_x + \frac{E_{\min}}n :
\]
कुल विचरण $=$ फिट लाइन के साथ भिन्नता $+$ अवशिष्ट
विचरण. (से विभाजित करना $ v_y $: $1 = \rho^2 + (1 - \rho^2)$, द
विचरण का हिस्सा लाइन द्वारा ``समझाया गया'' है $\rho^2$.)

\textbf{20.} मॉडल स्थान है
$\operatorname{Vect}\bigl(C_1, C_2, C_3\bigr)$ साथ $ C_1 = (1)$,
$ C_2 = (x_i)$, $ C_3 = (x_i^2)$; कम से कम $\norm{aC_1 + bC_2 +
cC_3 - b}^2$ बिलकुल प्रश्न 8 की तरह, की ओर ले जाता है $3\times3$
ग्राम प्रणाली, जिसके मैट्रिक्स में प्रविष्टियाँ हैं $\langle C_i, C_j\rangle =
\sum_k x_k^{\,i+j-2}$: प्रदर्शित क्षण मैट्रिक्स। यह है
उलटा यदि $(C_1, C_2, C_3)$ है \omterm{def:b1:vspaces:free}{मुक्त}
(\cref{exo:b1:euclid:11}); एक रिश्ता $ a + bx_k + cx_k^2 = 0$
सभी के लिए $ k $ हर बनाता है $ x_k $ एक की एक जड़ \omterm{def:b1:poly:def}{बहुपद} का
डिग्री $\leq 2$, जब तक कि तीन अलग-अलग मानों के साथ असंभव न हो $ a
= b = c = 0$. सप्ताहांत समस्या के आघूर्ण मैट्रिक्स की तुलना करें
का \cref{ch:b1:det}, जहां उनके निर्धारकों का वर्ग किया गया था
वेंडरमोंडे मूल्य.

\textbf{21.} पहला: $ r = 2, s = 1, t = 2$, $ rt - s^2 = 3 > 0$,
$ r > 0$: मूल में सख्त वैश्विक न्यूनतम। दूसरा: $ r = 2, s
= 3, t = 2$, $ rt - s^2 = -5 < 0$: काठी. तीसरा: $ r = s = t =
2$, $ rt - s^2 = 0$: पतित--- परंतु $ x^2 + 2xy + y^2 = (x +
y)^2 \geq 0$ पूरी लाइन पर गायब हो जाता है $ y = -x $: ए
(गैर-सख्त) वैश्विक न्यूनतम एक रेखा के साथ प्राप्त किया गया, जो अदृश्य है
निर्धारक परीक्षण.

\textbf{22.} नई रकम ($ n = 5$): $\bar x = 3.2$, $\bar y =
1.6$, $\sum x_iy_i = 19$ इसलिए $ c_{xy} = 3.8 - 5.12 = -1.32$,
$\sum x_i^2 = 114$ इसलिए $ v_x = 22.8 - 10.24 = 12.56$. नई ढलान:
\[
\beta = \frac{-1.32}{12.56} \approx -0.105 :
\]
एक बिंदु ने स्पष्ट रूप से बढ़ती प्रवृत्ति को बदल दिया ($\beta = 1.4$) में
थोड़ा कम हो रहा है. चुकता त्रुटि एक अवशिष्ट शुल्क लेती है
$\varepsilon^2$, तो एक दूर का बिंदु --- बड़ा
$\abs{x_{i} - \bar x}$ \emph{और} बड़ा अवशिष्ट --- हावी है
दोनों $ c_{xy}$ और $ v_x $: \omterm{pb:b1:multivar:1}{कम से कम वर्गों} कुशल है लेकिन नहीं
मजबूत.

\textbf{23.} \omterm{def:b1:logic:sets}{तय करना} $ W = \sum w_i $ और परिभाषित करें $\bar x_w =
\frac1W\sum w_ix_i $, $\bar y_w $ वैसे ही, $ v_x^w = \frac1W\sum
w_ix_i^2 - \bar x_w^2$, $ c^w_{xy} = \frac1W\sum w_ix_iy_i - \bar
x_w\bar y_w $. \omterm{def:b1:logic:map}{नक्शा} $\langle u, v\rangle_w = \sum_i w_iu_iv_i $
एक \omterm{def:b1:euclid:def}{अंदरूनी प्रोडक्ट} पर $\R^n $ ($ w_i > 0$ निश्चितता देता है), इसलिए
भाग II शब्दशः लागू होता है और \omterm{pb:b1:multivar:1}{सामान्य समीकरण} द्वारा विभाजित करें $ W $
में
\[
\alpha + \beta\bar x_w = \bar y_w,
\qquad
\alpha\bar x_w + \beta(v^w_x + \bar x_w^2) = c^w_{xy} + \bar
x_w\bar y_w ,
\]
जहां से $\beta = c^w_{xy}/v^w_x $ और $\alpha = \bar y_w -
\beta\bar x_w $: वही सूत्र, प्रत्येक औसत भार के साथ।

\textbf{24.} $ E_{\min} = \sum\varepsilon_i^2 = 0$ हर एक को मजबूर करता है
$\varepsilon_i = 0$: सभी बिंदु बिल्कुल रेखा पर; प्रश्न द्वारा
13 यह मामला है $\abs\rho = 1$. परीक्षण के लिए $(1,1), (2,3),
(3,5)$: $\bar x = 2$, $\bar y = 3$, $ v_x = \frac{14}3 - 4 =
\frac23$, $ c_{xy} = \frac{22}3 - 6 = \frac43$: $\beta = 2$,
$\alpha = -1$, सचमुच $ y_i = 2x_i - 1$ तीनों बिंदुओं के लिए;
$ v_y = \frac{35}3 - 9 = \frac83$ और $\rho^2 =
\frac{(4/3)^2}{(2/3)(8/3)} = 1$.

\textbf{25.} (i) द्विघात फलनों के लिए क्रम-दो विस्तार
एक है \emph{पहचान}, अतः द्विघात रूप का सांकेतिक अध्ययन
--- शुद्ध बीजगणित, प्रश्न 1--2 --- वर्गीकृत करता है महत्वपूर्ण बिंदु
बिना स्वीकृत टेलर प्रमेय के। (ii) द \omterm{pb:b1:multivar:1}{सामान्य समीकरण} कहो
``अवशिष्ट ओर्थोगोनल मॉडल स्पेस के लिए'', इसलिए विश्लेषणात्मक
न्यूनतम \emph{है}  ऑर्थोगोनल प्रक्षेपण डेटा वेक्टर का:
कैलकुलस और यूक्लिडियन ज्यामिति एक ही वस्तु की गणना करते हैं। (iii)
सहसंबंध $\rho = c_{xy}/\sqrt{v_xv_y}$ हिस्सेदारी मापता है
$\rho^2$ रेखा द्वारा समझाए गए विचरण का, और
कॉची--श्वार्ज़ इसे सीमित करता है: $\abs\rho \leq 1$, केवल समानता के साथ
संरेखित डेटा के लिए. (iv) \omterm{def:b1:vspaces:free}{आधार} और मुक्तता (अध्याय 18), ग्राम
और आघूर्ण आव्यूह और उनके निर्धारक (अध्याय 22-23),
और ऑर्थोगोनल प्रक्षेपण (अध्याय 23) सभी पुनः प्रकट हुए
एक एल्गोरिथ्म के कार्यशील भाग। भाग II--III का विकास होता है
\emph{की विधि \omterm{pb:b1:multivar:1}{कम से कम वर्गों}} (लीजेंडर, गॉस) इसके माध्यम से
\emph{\omterm{pb:b1:multivar:1}{सामान्य समीकरण}}.
\end{solution}
