\section*{अध्याय \ref{ch:b1:series} --- संख्यात्मक श्रृंखला}

\begin{solution}{exo:b1:series:1}
$\dfrac{1}{n(n+1)} = \dfrac1n - \dfrac{1}{n+1}$: दूरबीन, $ S_N =
1 - \frac{1}{N+1} \to 1$. अभिसरण, योग $1$.

$\ln\bigl(1 - \frac{1}{n^2}\bigr) = \ln\frac{(n-1)(n+1)}{n^2} =
\ln\frac{n-1}{n} - \ln\frac{n}{n+1}$: दूरबीन फिर से, $ S_N =
\ln\frac12 - \ln\frac{N}{N+1} \to -\ln 2$. अभिसरण, योग $-\ln 2$.

$\dfrac{3^n + 4^n}{5^n} = \bigl(\frac35\bigr)^n +
\bigl(\frac45\bigr)^n $: दो अभिसरण \omterm{ex:b1:series:geometric}{जियोमीट्रिक श्रंखला}, जोड़
$\frac{1}{1 - 3/5} + \frac{1}{1 - 4/5} = \frac52 + 5 = \frac{15}{2}$.
\end{solution}

\begin{solution}{exo:b1:series:2}
\omterm{thm:b1:series:ratio}{अनुपात परीक्षण} लगातार।

$\frac{u_{n+1}}{u_n} = \frac{(n+1)^2}{2n^2} \to \frac12 < 1$:
अभिसारी।

$\frac{u_{n+1}}{u_n} = \frac{(n+1)!\,n^n}{n!\,(n+1)^{n+1}} =
\bigl(\frac{n}{n+1}\bigr)^n = \bigl(1 + \frac1n\bigr)^{-n} \to
\frac1\eu < 1$: अभिसरण.

कारक के साथ $2^n $: अनुपात $\to \frac2\eu < 1$: अभिसरण.

साथ $3^n $: अनुपात $\to \frac3\eu > 1$: भिन्न (शब्दों की प्रवृत्ति होती है
$+\infty $).
\end{solution}

\begin{solution}{exo:b1:series:3}
सभी गैर-नकारात्मक शब्द; समकक्षों का उपयोग करें
(\cref{thm:b1:series:positive}).

$\sin\frac{1}{n^2} \sim \frac{1}{n^2}$: अभिसरण (रीमैन $\alpha
= 2$).

$1 - \cos\frac1n \sim \frac{1}{2n^2}$: अभिसरण.

$\frac{1}{\sqrt{n(n+1)}} \sim \frac1n $: भिन्न।

$\frac{\ln n}{n^2} = \frac{1}{n^{3/2}}\cdot\frac{\ln n}{n^{1/2}}$
और $\frac{\ln n}{\sqrt n} \to 0$
(\cref{prop:b1:functions:powerrules}): इसलिए $\frac{\ln n}{n^2} \leq
\frac{1}{n^{3/2}}$ बड़े के लिए $ n $: अभिसरण.
\end{solution}

\begin{solution}{exo:b1:series:4}
के लिए $ n \geq 2$: $\frac{1}{n^2} \leq \frac{1}{n(n-1)} =
\frac{1}{n-1} - \frac1n $. इस तरह
\[
\sum_{n=1}^{N} \frac{1}{n^2} \leq 1 + \sum_{n=2}^{N}
\Bigl(\frac{1}{n-1} - \frac1n\Bigr) = 1 + 1 - \frac1N < 2 :
\]
आंशिक रकम बढ़ती और सीमित होती है $2$: अभिसरण
(\cref{thm:b1:series:positive}), जोड़ $\leq 2$. (सटीक मान
$\frac{\pi^2}{6}$ यह दूसरे वर्ष का उत्सव है।)
\end{solution}

\begin{solution}{exo:b1:series:5}
$\sum \frac{(-1)^n}{\sqrt n}$: के साथ बारी-बारी से $\frac{1}{\sqrt n}
\downarrow 0$: अभिसरण (\cref{thm:b1:series:alternating}); नहीं
बिल्कुल ($\alpha = \frac12 \leq 1$).

$\sum \frac{(-1)^n}{n + (-1)^n}$: क्रम $\frac{1}{n +
(-1)^n}$ है \emph{नहीं} घट रहा है ($\frac{1}{n+1}$ तब
$\frac{1}{n}$ बुरी तरह से वैकल्पिक), इसलिए परीक्षण सीधे लागू नहीं होता है।
बढ़ाना:
\[
\frac{(-1)^n}{n + (-1)^n}
= \frac{(-1)^n}{n}\cdot\frac{1}{1 + \frac{(-1)^n}{n}}
= \frac{(-1)^n}{n} - \frac{1}{n^2} +
O\Bigl(\frac{1}{n^3}\Bigr):
\]
पहला \omterm{def:b1:series:def}{शृंखला} अभिसरण (वैकल्पिक), $\sum \frac{1}{n^2}$
अभिसरण, $ O(n^{-3})$ बिल्कुल अभिसरण: तीन का योग
अभिसारी \omterm{def:b1:series:def}{शृंखला} जुटता है.

$\sin\bigl(\pi\sqrt{n^2+1}\bigr)$: लिखना $\sqrt{n^2 + 1} = n +
\frac{1}{2n} + \varepsilon_n $ साथ $\varepsilon_n = O(n^{-3})$;
फिर, द्वारा $\pi $-आवधिकता $\sin $ हस्ताक्षर करने तक,
\[
\sin\bigl(\pi\sqrt{n^2+1}\bigr)
= (-1)^n \sin\Bigl(\frac{\pi}{2n} + \pi\varepsilon_n\Bigr) .
\]
\omterm{def:b1:logic:sets}{तय करना} $\theta_n = \frac{\pi}{2n} + \pi\varepsilon_n $ और $ a_n =
\sin\theta_n $. बड़े के लिए $ n $, $\theta_n \in \intoo{0}{\frac\pi2}$
और
\[
\theta_n - \theta_{n+1} = \frac{\pi}{2n(n+1)} +
\pi(\varepsilon_n - \varepsilon_{n+1})
= \frac{\pi}{2n^2} + O\Bigl(\frac{1}{n^3}\Bigr) > 0
\]
अंततः, ऐसा $(\theta_n)$ तक कम हो जाता है $0$; तब से $\sin $ है
बढ़ता जा रहा है $\intcc{0}{\frac\pi2}$, $(a_n)$ तक कम हो जाता है $0$ भी। वैकल्पिक परीक्षण लागू होता है: अभिसरण --- बिल्कुल नहीं,
तब से $ a_n \sim \frac{\pi}{2n}$.
\end{solution}

\begin{solution}{exo:b1:series:6}
$ f(t) = \frac{1}{t(\ln t)^\alpha}$ सकारात्मक है, निरंतर,
पर घट रहा है $\intco{2}{+\infty}$. स्थानापन्न $ u = \ln t $:
\[
\int_2^x \frac{\dd t}{t(\ln t)^\alpha}
= \int_{\ln 2}^{\ln x} \frac{\dd u}{u^\alpha},
\]
के रूप में परिबद्ध $ x \to \infty $ आईएफएफ $\alpha > 1$
(\cref{thm:b1:series:riemann}की गणना). द्वारा \omterm{thm:b1:integration:def}{अभिन्न}
तुलना: अभिसरण आईएफएफ $\alpha > 1$. (इन \emph{बर्ट्रेंड-प्रकार}
\omterm{def:b1:series:def}{शृंखला} दिखाओ कितना अच्छा है \omterm{def:b1:topology:closure}{सीमा} अभिसरण का है: $ n\ln n $
विचलन, $ n(\ln n)^{1.01}$ अभिसरण।)
\end{solution}

\begin{solution}{exo:b1:series:7}
की स्पर्शरेखा-रेखा \omterm{def:b1:topology:closure}{सीमा} से \cref{exo:b1:derivative:3} पुनः लिखा गया
विस्तार के माध्यम से: के लिए $ x = \frac1n \in \intoc{0}{1}$, टेलर--लैग्रेंज
के लिए $\ln(1+x)$ क्रम में $1$ देता है $\ln(1 + x) = x - \frac{x^2}{2(1 +
c)^2}$ कुछ के लिए $ c \in \intoo{0}{x}$, इसलिए
\[
0 \leq u_n = \frac1n - \ln\Bigl(1 + \frac1n\Bigr) \leq
\frac{1}{2n^2} .
\]
के साथ तुलना \omterm{thm:b1:series:riemann}{रीमैन श्रृंखला}: $\sum u_n $ जुटता है. यह आंशिक है
योग दूरबीन लघुगणक:
\[
\sum_{n=1}^{N} u_n = H_N - \ln(N+1)
\]
(तब से $\sum_{n\leq N} \ln\frac{n+1}{n} = \ln(N+1)$). इसलिए $ H_N -
\ln(N+1)$ अभिसरण; जोड़ना $\ln\frac{N+1}{N} \to 0$, क्रम
$ H_N - \ln N $ जुटता है. इसकी \omterm{def:b1:topology:closure}{सीमा} है $\gamma \approx 0.5772$.
\end{solution}

\begin{solution}{exo:b1:series:8}
$\dfrac{1}{n(n+2)} = \dfrac{1/2}{n} - \dfrac{1/2}{n+2}$: आंशिक
के \omterm{prop:b1:reals:intervals}{अंतराल} के साथ दूरबीनों का योग $2$,
\[
S_N = \frac12\Bigl(1 + \frac12 - \frac{1}{N+1} -
\frac{1}{N+2}\Bigr) \longrightarrow \frac34 .
\]

$\sum \frac{n}{2^n}$: के लिए $\abs x < 1$, परिमित को अलग करना
ज्यामितीय योग और \omterm{def:b1:topology:closure}{सीमा} तक गुजरना (सभी)। \omterm{def:b1:series:def}{शृंखला} यहां जुटते हैं
बिल्कुल, \omterm{thm:b1:series:ratio}{अनुपात परीक्षण}): से $\sum_{n\geq0} x^n = \frac{1}{1-x}$,
आंशिक रकम के साथ सीधी गणना से प्राप्त होता है
\[
\sum_{n=1}^{N} n x^{n-1}
= \frac{1 - (N+1)x^N + N x^{N+1}}{(1 - x)^2}
\xrightarrow[N\to\infty]{} \frac{1}{(1-x)^2}
\quad (\abs x < 1),
\]
(द \omterm{def:b1:topology:closure}{सीमा} शर्तें $ N x^N \to 0$). पर $ x = \frac12$:
$\sum_{n\geq1} n\bigl(\frac12\bigr)^{n-1} = 4$, इसलिए $\sum_{n\geq0}
\frac{n}{2^n} = \frac12 \times 4 = 2$.
\end{solution}

\begin{solution}{exo:b1:series:9}
की शर्तों को \omterm{def:b1:structures:group}{समूह}ित करें $\sum u_n $ की शक्तियों के बीच पैकेट में $2$.
ऊपरी पैकेट: के लिए $2^k \leq n < 2^{k+1}$ वहाँ हैं $2^k $ शर्तें,
प्रत्येक $\leq u_{2^k}$:
\[
\sum_{n=1}^{2^{K+1}-1} u_n
= \sum_{k=0}^{K} \sum_{n=2^k}^{2^{k+1}-1} u_n
\leq \sum_{k=0}^{K} 2^k u_{2^k} .
\]
निचला पैकेट: प्रत्येक पद एक ही पैकेट का होता है $\geq u_{2^{k+1}}$,
इसलिए $\sum_{n=2^k}^{2^{k+1}-1} u_n \geq 2^k u_{2^{k+1}} = \frac12
\cdot 2^{k+1} u_{2^{k+1}}$, कहाँ से
\[
\sum_{n=1}^{2^{K+1}-1} u_n \geq \frac12 \sum_{k=1}^{K+1} 2^{k}
u_{2^{k}} .
\]
दोनों आंशिक-योग तुलनाएँ दोनों तरीकों से जाती हैं (गैर-नकारात्मक शब्द,
\cref{thm:b1:series:positive}): दो \omterm{def:b1:series:def}{शृंखला} एक ही स्वभाव है.

रीमैन: $ u_n = n^{-\alpha}$ देता है $2^k u_{2^k} =
2^{k(1-\alpha)}$, ए \omterm{ex:b1:series:geometric}{जियोमीट्रिक श्रंखला}, अभिसरण आईएफएफ $2^{1 - \alpha}
< 1$ आईएफएफ $\alpha > 1$. बर्ट्रेंड
(\cref{exo:b1:series:6}): $ u_n = \frac{1}{n(\ln n)^\alpha}$ देता है
$2^k u_{2^k} = \frac{1}{(k\ln 2)^\alpha}$, ए \omterm{thm:b1:series:riemann}{रीमैन श्रृंखला} में $ k $:
अभिसरण आईएफएफ $\alpha > 1$.
\end{solution}

\begin{solution}{exo:b1:series:10}
अनुपात के साथ परिमित ज्यामितीय पहचान $-t^2$:
\[
\frac{1}{1 + t^2} = \sum_{k=0}^{n-1} (-1)^k t^{2k}
+ \frac{(-1)^n t^{2n}}{1 + t^2} .
\]
अधिक एकीकृत करें $\intcc{0}{1}$ (बाईं ओर एकीकृत होता है $\arctan
1 = \frac\pi4$, \cref{prop:b1:functions:arcderiv}):
\[
\frac{\pi}{4} = \sum_{k=0}^{n-1} \frac{(-1)^k}{2k+1}
+ (-1)^n \int_0^1 \frac{t^{2n}}{1+t^2}\,\dd t,
\qquad
0 \leq \int_0^1 \frac{t^{2n}}{1+t^2}\,\dd t \leq \int_0^1 t^{2n}\dd
t = \frac{1}{2n+1} .
\]
दे $ n \to \infty $ सूत्र को सिद्ध करता है, और प्रदर्शित बाउंड
पर \omterm{thm:b1:integration:def}{अभिन्न} वास्तव में शेषफल बंधा हुआ है: योग करने के बाद
$ N $ (अर्थात्\ $ n = N + 1$ शर्तें), $\abs{R_N} \leq \frac{1}{2N + 3}$.
\end{solution}

\begin{solution}{exo:b1:series:11}
$ n^{1/n} = \eu^{\frac{\ln n}{n}} \to \eu^0 = 1$
(\cref{prop:b1:functions:powerrules}). इस तरह
\[
\frac{1}{n^{1 + 1/n}} = \frac{1}{n}\,\eu^{-\frac{\ln n}{n}}
\sim \frac{1}{n} ,
\]
और समकक्ष परीक्षण (\cref{thm:b1:series:positive})
अपसारी हार्मोनिक के साथ तुलना करता है \omterm{def:b1:series:def}{शृंखला}: \emph{विभिन्न},
यद्यपि प्रत्येक प्रतिपादक $1 + \frac1n $ से अधिक है $1$. रीमैन
मानदंड की चिंता ए \emph{तय} प्रतिपादक $\alpha $; एक प्रतिपादक
नीचे की ओर खिसकना $1$ यहाँ की तरह अपना सारा मार्जिन खो सकता है।
\end{solution}

\begin{solution}{exo:b1:series:12}
होने देना $\varepsilon > 0$. कॉची मानदंड के अनुसार
अभिसारी \omterm{def:b1:series:def}{शृंखला} (\cref{thm:b1:seq:complete} पर लागू किया गया
आंशिक रकम), वहाँ है $ N $ साथ $\sum_{k=n+1}^{2n} u_k \leq
\varepsilon $ के लिए $ n \geq N $. इनमें से प्रत्येक की एकरसता से $ n $
शर्तें है $\geq u_{2n}$:
\[
n\,u_{2n} \leq \sum_{k=n+1}^{2n} u_k \leq \varepsilon
\quad\Longrightarrow\quad 2n\,u_{2n} \leq 2\varepsilon ,
\]
और विषम सूचकांकों के लिए $(2n+1)\,u_{2n+1} \leq (2n+1)\,u_{2n} \leq
2\bigl(2n\,u_{2n}\bigr) \leq 4\varepsilon $ के लिए $ n \geq N $: में
दोनों समानताएं, $ n u_n \to 0$.

बातचीत झूठी: $ u_n = \frac{1}{n\ln n}$ है $ n u_n =
\frac{1}{\ln n} \to 0$, फिर भी \omterm{def:b1:series:def}{शृंखला} विचलन
(\cref{exo:b1:series:6}, $\alpha = 1$). एकरसता आवश्यक:
चलो $ u_n = \frac1n $ कब $ n $ एक पूर्ण वर्ग है और $ u_n =
2^{-n}$ अन्यथा \omterm{def:b1:series:def}{शृंखला} अभिसरण (वर्ग शब्दों का योग)।
जैसे $\sum \frac{1}{k^2}$, बाकी ज्यामितीय रूप से), लेकिन $ n u_n =
1$ चौकों के साथ.
\end{solution}

\begin{solution}{pb:b1:series:1}
\textbf{1.} $ a_{n+1} - a_n = \frac{1}{n+1} - \ln\frac{n+1}{n}
\leq 0$ क्योंकि $\ln(1 + \frac1n) \geq \frac{1/n}{1 + 1/n} =
\frac{1}{n+1}$; और $ b_{n+1} - b_n = \frac{1}{n+1} -
\ln\frac{n+2}{n+1} \geq 0$ क्योंकि $\ln(1 + \frac{1}{n+1}) \leq
\frac{1}{n+1}$. उनका अंतर $ a_n - b_n = \ln(1 + \frac1n) \to 0$:
नज़दीक (\cref{thm:b1:seq:adjacent}), सामान्य \omterm{def:b1:topology:closure}{सीमा} के साथ $\lim
a_n = \gamma $ (\cref{exo:b1:series:7}), और $ b_n \leq \gamma
\leq a_n $.

\textbf{2.} $ b_{10} = 2.928968 - \ln 11 = 0.5311$ और $ a_{10} =
2.928968 - \ln 10 = 0.6264$: इसलिए $\gamma \in
\intcc{0.5311}{0.6264}$. अंतर है $\ln 1.1 \approx 0.095$ और
जैसे सिकुड़ जाता है $\frac1n $: चार दशमलव ($\text{gap} \leq
10^{-4}$) जरूरत पड़ेगी $ n \approx 10^4$ --- कोष्ठक हैं
सही लेकिन धीमा.

\textbf{3.} अंतःसर्पण $ a_n - a_{m+1} = \sum_{k=n}^{m} w_k $ और
देना $ m \to \infty $: $ a_n - \gamma = \sum_{k\geq n} w_k $
(आंशिक रकम की \omterm{def:b1:topology:closure}{सीमा}). इसके अतिरिक्त
\[
w_k = \int_k^{k+1} \frac{\dd t}{t} - \frac{1}{k+1}
= \int_k^{k+1} \Bigl(\frac1t - \frac{1}{k+1}\Bigr)\dd t
= \int_k^{k+1} \frac{k + 1 - t}{t\,(k+1)}\,\dd t .
\]
पर $\intcc{k}{k+1}$: $\frac{1}{(k+1)^2} \leq \frac{1}{t(k+1)}
\leq \frac{1}{k(k+1)}$, और $\int_k^{k+1}(k + 1 - t)\dd t =
\frac12$: इस तरह $\frac{1}{2(k+1)^2} \leq w_k \leq
\frac{1}{2k(k+1)}$.

\textbf{4.} ऊपरी: $\sum_{k \geq n} \frac{1}{2k(k+1)} =
\frac12\sum_{k\geq n}\bigl(\frac1k - \frac{1}{k+1}\bigr) =
\frac{1}{2n}$ (दूरबीन)। निचला: $\frac{1}{2(k+1)^2} \geq
\frac{1}{2(k+1)(k+2)}$, जिसका योग दूरबीनों को
$\frac{1}{2(n+1)}$. प्रश्न 3 के साथ:
\[
\frac{1}{2(n+1)} \leq H_n - \ln n - \gamma \leq \frac{1}{2n} .
\]

\textbf{5.} घटाना $\frac{1}{2n}$: $\gamma_n - \gamma \in
\intcc{\frac{1}{2(n+1)} - \frac{1}{2n}}{0} =
\intcc{-\frac{1}{2n(n+1)}}{0}$: संशोधित अनुमान सटीक है
को $ O\bigl(\frac{1}{n^2}\bigr)$, और हमेशा नीचे से।

\textbf{6.} $\gamma_{100} = 5.1873775 - \ln 100 - 0.005 =
0.5772073$, साथ $0 \leq \gamma - \gamma_{100} \leq
\frac{1}{20200} < 5\cdot10^{-5}$: इस तरह $0.577207 \leq \gamma
\leq 0.577257$, यानी\ $\gamma = 0.5772 \pm 5\cdot10^{-5}$ (सच
मूल्य $0.5772156\dots $) --- ए से चार प्रमाणित दशमलव
प्रश्न 2 के लिए दस हजार के मुकाबले सौ पद।

\textbf{7.} पहला लाभांश:
\[
H_{2m} - H_m = \Bigl(\ln 2m + \gamma + \frac{1}{4m}\Bigr)
- \Bigl(\ln m + \gamma + \frac{1}{2m}\Bigr) +
O\Bigl(\frac{1}{m^2}\Bigr)
= \ln 2 - \frac{1}{4m} + O\Bigl(\frac{1}{m^2}\Bigr).
\]
दूसरा: सम व्युत्क्रम तक $2m $ का योग $\frac12 H_m $, इसलिए
$\sum_{j=1}^{m}\frac{1}{2j-1} = H_{2m} - \frac12 H_m = \frac12
\ln m + \ln 2 + \frac\gamma2 + o(1)$, और $\sum_{j=1}^m
\frac{1}{2j} = \frac12\ln m + \frac\gamma2 + o(1)$.

\textbf{8.} सम पदों को दो बार विभाजित करना:
$\sum_{k=1}^{2m}\frac{(-1)^{k-1}}{k} = H_{2m} - 2\cdot\frac12
H_m = H_{2m} - H_m $. प्रश्न 7 से यह बराबर है $\ln 2 -
\frac{1}{4m} + O(m^{-2})$: योग है $\ln 2$
(\cref{ex:b1:series:ln2} दोबारा) \emph{साथ} इसकी गति.

\textbf{9.} के लिए $ N = 2m $: $ S - S_N = \frac{1}{4m} + O(m^{-2}) =
\frac{1}{2N} + O(N^{-2})$. के लिए $ N = 2m + 1$: $ S_{N} = S_{2m} +
\frac{1}{2m+1}$, इसलिए
\[
S - S_N = \Bigl(\frac{1}{4m} - \frac{1}{2m+1}\Bigr) +
O\Bigl(\frac{1}{m^2}\Bigr)
= -\frac{1}{4m} + O\Bigl(\frac{1}{m^2}\Bigr)
= -\frac{1}{2N} + O\Bigl(\frac{1}{N^2}\Bigr).
\]
दोनों ही मामलों में $ S - S_N \sim \frac{(-1)^N}{2N}$: आधा
सबसे ख़राब स्थिति वाला $ a_{N+1}$, एक ज्ञात, वैकल्पिक संकेत के साथ।

\textbf{10.} औसत दोलनशील अग्रणी पद को समाप्त कर देता है:
\[
S - \tilde S_N = \frac{(S - S_N) + (S - S_{N+1})}{2}
= \frac{(-1)^N}{2}\Bigl(\frac{1}{2N} - \frac{1}{2(N+1)}\Bigr)
+ O\Bigl(\frac{1}{N^2}\Bigr) = O\Bigl(\frac{1}{N^2}\Bigr).
\]
संख्यात्मक रूप से: $ S_{10} = 0.645635$, $ S_{11} = 0.736544$, $\tilde
S_{10} = 0.691089$, और $\ln 2 = 0.693147$: त्रुटि समाप्त हो जाती है
$4.8\cdot10^{-2}$ को $2.1\cdot10^{-3}$ --- एक जोड़, बीस
कई गुना बेहतर.

\textbf{11.} प्रत्यावर्ती त्रुटि हर कदम पर संकेत बदलती है,
इसलिए लगातार आंशिक योग \omterm{def:b1:topology:closure}{सीमा} और उनके माध्य का विस्तार करते हैं
प्रथम-क्रम अवधि को रद्द करता है; ब्रैकेट त्रुटि $ H_n - \ln n -
\gamma \approx \frac{1}{2n}$ स्थिर चिह्न है, इसलिए कोई औसत नहीं है
साथ में $ n $ इसे रद्द कर सकते हैं. दोनों का औसत निकालना \emph{कोष्ठक} करता है
सहायता: $\frac{a_n + b_n}{2} = H_n - \ln\sqrt{n(n+1)}$, और तबसे
$\ln\sqrt{n(n+1)} = \ln\bigl(n + \frac12\bigr) + O(n^{-2})$,
\[
H_n - \ln\Bigl(n + \frac12\Bigr)
= \Bigl(H_n - \ln n - \frac{1}{2n}\Bigr) + \frac{1}{8n^2} +
O\Bigl(\frac{1}{n^3}\Bigr) = \gamma + O\Bigl(\frac{1}{n^2}\Bigr)
\]
(प्रश्न 5 और $\ln(1 + \frac{1}{2n}) = \frac{1}{2n} -
\frac{1}{8n^2} + O(n^{-3})$). पर जांचें $ n = 10$: $ H_{10} -
\ln 10.5 = 0.57759$, पहले से ही भीतर $4\cdot10^{-4}$ का
$\gamma $.

\textbf{12.} $\sum_{j\leq m} \frac{1}{2j-1} \geq \sum_{j \leq m}
\frac{1}{2j} = \frac12 H_m \to +\infty $: दोनों सकारात्मक और
प्रत्यावर्ती हार्मोनिक का नकारात्मक भाग \omterm{def:b1:series:def}{शृंखला} विचलन

\textbf{13.} लिखना $ u_n^\pm = \frac{\abs{u_n} \pm u_n}{2} \geq
0$, इसलिए $ u_n = u_n^+ - u_n^-$ और $\abs{u_n} = u_n^+ + u_n^-$. अगर
$\sum u_n^+$ फिर एकत्रित हुए $\sum u_n^- = \sum (u_n^+ - u_n)$
अभिसरण होगा (अभिसरण का अंतर \omterm{def:b1:series:def}{शृंखला}), इस तरह $\sum
\abs{u_n}$ भी: सशर्त अभिसरण के साथ विरोधाभास. द्वारा
समरूपता दोनों $\sum u_n^\pm $ विचलन (को.) $+\infty $): एक अनंत
सकारात्मक और नकारात्मक द्रव्यमान का भंडार।

\textbf{14.} होने देना $ S = \sum u_n $, $\varepsilon > 0$, और $ N $ साथ
$\sum_{n > N} \abs{u_n} \leq \varepsilon $ (के लिए कॉची मानदंड
$\sum\abs{u_n}$). होने देना $ M_0$ इतना बड़ा हो कि $\sigma(\{0,
\dots, M_0\}) \supseteq \{0, \dots, N\}$. के लिए $ M \geq M_0$, के अंतर $\sum_{m \leq M} u_{\sigma(m)} - \sum_{n \leq N} u_n $
का एक सीमित योग है \emph{विशिष्ट} शर्तें $ u_n $ साथ $ n > N $,
इसलिए पूर्ण मूल्य का $\leq \varepsilon $; और $\abs{S -
\sum_{n\leq N} u_n} \leq \varepsilon $ भी। तो पुनर्व्यवस्थित
आंशिक रकम भीतर हैं $2\varepsilon $ का $ S $ अंततः:
$\sum u_{\sigma(n)} = S $. पूर्ण अभिसरण है
पुनर्व्यवस्था-प्रमाण.

\textbf{15.} लालची प्रक्रिया का प्रत्येक चरण समाप्त हो जाता है
परिमित रूप से अनेक पद, क्योंकि शेष सकारात्मक
(क्रमशः नकारात्मक) पदों में अकेले आंशिक योग भिन्न होते हैं
(प्रश्न 12): चल रहा योग अंततः पार होना चाहिए $ t $. द
इसलिए प्रक्रिया अनंत रूप से कई परिमित चरणों को वैकल्पिक करती है,
सकारात्मक शब्दों को क्रम में और नकारात्मक शब्दों को क्रम में लेते हुए
क्रम: प्रत्येक शब्द का प्रयोग बिल्कुल एक बार किया जाता है --- एक पुनर्व्यवस्था।
पहली क्रॉसिंग के बाद, लगातार दो क्रॉसिंग के बीच
आंशिक रकमें नीरस रूप से आगे बढ़ती हैं $ t $, और एक चौराहे पर
वे अधिक से अधिक अभी जोड़े गए शब्द से आगे निकल जाते हैं; शर्तों के बाद से
में प्रयोग किया जाता है $ j $-वें क्रॉसिंग में कम से कम इंडेक्स है $ j $ उनके में
वर्ग, ये ओवरशूट होते हैं $0$. इसलिए आंशिक रकम
में जुटना $ t $: प्रत्येक वास्तविक संख्या कुछ का योग है
पुनर्व्यवस्था.

\textbf{16.} सकारात्मक स्लॉट प्राप्त होते हैं $\frac{1}{2j-1}$ के लिए
$ j = 1, 2, \dots $ क्रम में, नकारात्मक स्लॉट $\frac{1}{2j}$ में
क्रम: प्रत्यावर्ती हार्मोनिक का प्रत्येक पद \omterm{def:b1:series:def}{शृंखला} प्रकट होता है
बिल्कुल एक बार. के लिए $(p, q) = (1, 2)$, ब्लॉक हैं
$\bigl(1, -\frac12, -\frac14\bigr)$, $\bigl(\frac13, -\frac16,
-\frac18\bigr)$, $\bigl(\frac15, -\frac1{10},
-\frac1{12}\bigr)$, \dots\ --- प्रदर्शित \omterm{def:b1:series:def}{शृंखला}.

\textbf{17.} तब से $\frac{1}{4k-2} = \frac12\cdot\frac{1}{2k-1}$:
\[
\frac{1}{2k-1} - \frac{1}{4k-2} - \frac{1}{4k}
= \frac12\,\frac{1}{2k-1} - \frac12\,\frac{1}{2k}
= \frac12\Bigl(\frac{1}{2k-1} - \frac{1}{2k}\Bigr).
\]
संक्षेप में कहना $ k = 1, \dots, K $: $ T_{3K} = \frac12
\sum_{k=1}^{K}\bigl(\frac{1}{2k-1} - \frac{1}{2k}\bigr) =
\frac12 S_{2K}$: प्रत्येक तीसरे आंशिक योग पर, पुनर्व्यवस्थित
\omterm{def:b1:series:def}{शृंखला} है \emph{बिल्कुल} मूल का आधा.

\textbf{18.} बाद $ K $ पूर्ण ब्लॉक, पुनर्व्यवस्थित आंशिक
योग है $\sum_{j=1}^{pK}\frac{1}{2j-1} -
\sum_{j=1}^{qK}\frac{1}{2j}$, और प्रश्न 7 इसका मूल्यांकन करता है:
\[
\Bigl(\frac{\ln(pK)}{2} + \ln 2 + \frac\gamma2\Bigr)
- \Bigl(\frac{\ln(qK)}{2} + \frac\gamma2\Bigr) + o(1)
= \ln 2 + \frac12\ln\frac pq + o(1) :
\]
 $\gamma $ रद्द करें, $\ln K $ रद्द करें, अनुपात
$\frac pq $ जीवित रहता है.

\textbf{19.} ब्लॉक के अंदर एक आंशिक योग $ K + 1$ से भिन्न है
$ K $- अधिकतम राशि ब्लॉक करें $ p + q $ शर्तें, प्रत्येक पूर्ण मूल्य की
$\leq \frac{1}{2qK}$-ईश, इसलिए द्वारा $ O\bigl(\frac1K\bigr) \to 0$:
आंशिक योगों के पूर्ण अनुक्रम की \omterm{def:b1:topology:closure}{सीमा} समान होती है $\ln 2 +
\frac12\ln\frac pq $. के लिए $(1, 2)$: $\ln 2 + \frac12\ln\frac12 =
\frac{\ln 2}{2} = 0.34657\dots $, सचमुच $ T_9 = 0.30833$
इसकी ओर रेंगता है: प्रश्न 17 द्वारा, $ T_{3K} = \frac12 S_{2K}$
प्रत्यावर्ती-हार्मोनिक त्रुटि के बिल्कुल आधे भाग के साथ अभिसरण होता है।
शर्तें समान, योग आधा।

\textbf{20.} $(1,1)$: $\ln 2 + \frac12\ln 1 = \ln 2$ --- द
मूल क्रम, स्थिरता. $(2,1)$: $\frac32\ln 2 \approx
1.0397$. $(p,q)$ मेनू बिल्कुल गणनीय तक पहुंचता है \omterm{def:b1:topology:dense}{घना}
परिवार $\ln 2 + \frac12\ln r $, $ r \in \Q_{>0}$; रीमैन लालची है
नुस्खा (प्रश्न 15) तक पहुँचता है \emph{प्रत्येक} असली। संरचना खरीदता है
सूत्र; लालच समग्रता खरीद लेता है.

\textbf{21.} आंशिक रकम दूरबीन: $\sum_{k=1}^{N}
\bigl(\frac1k - \ln\frac{k+1}{k}\bigr) = H_N - \ln(N+1) = b_N
\to \gamma $: द \omterm{def:b1:series:def}{शृंखला} का \cref{exo:b1:series:7} बिल्कुल सही योग
को \omterm{ex:b1:series:oddtelescope}{यूलर स्थिरांक}.

\textbf{22.} के लिए लालची $ t = 1$: पहला सकारात्मक पद लाता है
बिल्कुल योग $1$, इससे परे नहीं, इसलिए दूसरा सकारात्मक है
पार करने के लिए ले जाया गया: $1, \frac13$ (जोड़ $1.3333 > 1$), तब
$-\frac12$ ($0.8333$), $\frac15$ ($1.0333$), $-\frac14$
($0.7833$), $\frac17, \frac19$ ($1.0373$), $-\frac16$
($0.8706$), $\frac1{11}, \frac1{13}$ ($1.0384$), $-\frac18$
($0.9134$), $\frac1{15}$ ($0.9801$), \dots\ --- रकम सांस लेती है
चारों ओर $1$ हमेशा छोटे आयाम के साथ, अब दो सकारात्मकताएँ
प्रति चक्र आवश्यक है क्योंकि नकारात्मक बड़े हैं।

\textbf{23.} ब्लॉक बनाएं: स्टेज पर $ j $, पर्याप्त अप्रयुक्त जोड़ें
आंशिक राशि को कम से कम बढ़ाने के लिए सकारात्मक शर्तें $1$
(संभव: शेष सकारात्मक पदों का योग अलग-अलग है,
प्रश्न 12), फिर एकल नकारात्मक पद जोड़ें $-\frac{1}
{2j}$. प्रत्येक सकारात्मक शब्द का अंततः उपयोग किया जाता है (प्रत्येक चरण में उपयोग किया जाता है)।
कम से कम एक), प्रत्येक नकारात्मक भी (प्रति चरण एक): a
पुनर्व्यवस्था. प्रत्येक चरण में योग बदलता है $\geq 1 -
\frac{1}{2j} \geq \frac12$: आंशिक रकम अधिक है
$\frac{j}{2}$ चरण के बाद $ j $ और एक चरण के भीतर वेतन वृद्धि
अंतिम को छोड़कर सकारात्मक हैं, से बंधे हुए हैं $\frac{1}{2j} \to 0$:
से विचलन $+\infty $.

\textbf{24.} कायदे से, $ H_N \geq 20 \iff \ln N \geq 20 -
\gamma - \frac{1}{2N} + O(N^{-2})$: दहलीज $ N^*$
संतुष्ट $\ln N^* = 20 - \gamma + o(1)$, यानी\ $ N^* =
\eu^{20 - \gamma}(1 + o(1)) \approx \eu^{19.4228} \approx
2.72\cdot10^{8}$ --- कच्ची खिड़की के अंदर
$\intcc{1.8\cdot10^8}{4.9\cdot10^8}$ का
\cref{ex:b1:series:harmonicstack}, और द्वारा पिन किया गया $\gamma $.

\textbf{25.} (i)इन $ H_n = \ln n + \gamma + \frac{1}{2n} +
O(n^{-2})$: द $\ln n $ है \omterm{thm:b1:integration:def}{अभिन्न}, $\gamma $ की कीमत
किसी राशि को a से प्रतिस्थापित करना \omterm{thm:b1:integration:def}{अभिन्न} (वास्तव में एक नया स्थिरांक
विश्लेषण), और $\frac{1}{2n}$ पहला सुधार---
समलम्ब चतुर्भुज की छाया. (ii) सशर्त अभिसरण पर निर्भर करता है
दो अनंत जलाशयों के बीच रद्दीकरण (प्रश्न 13), इसलिए
पुन: व्यवस्थित करने से जलाशयों का वजन फिर से बढ़ जाता है; पूर्ण अभिसरण है
परिमित कुल द्रव्यमान, और प्रश्न 14 का पूँछ अनुमान है
आदेश-अंधा। (iii) द $(p,q)$ रकम थे \emph{गणना}: द
$\gamma $-कानून ने प्रत्येक पुनर्व्यवस्थित आंशिक योग को बदल दिया $\ln 2 +
\frac12\ln\frac pq + o(1)$, साथ $\gamma $ स्वयं रद्द हो रहा है ---
एक स्पर्शोन्मुख बहीखाता पद्धति, कोई अमूर्त तर्क नहीं।
(iv) अगला: उत्पाद और बिजली के लिए बिना शर्त योग्यता
\omterm{def:b1:series:def}{शृंखला} वर्ष 2 खंड में; और वर्ष 3 खंड का सप्ताहांत
स्टर्लिंग के सूत्र पर समस्या, जहां योग-बना\omterm{thm:b1:integration:def}{म-अभिन्न}
बहीखाता, एक ऑर्डर को आगे बढ़ाया, उत्पादन किया $\sqrt{2\pi}$
स्वयं.
\end{solution}
