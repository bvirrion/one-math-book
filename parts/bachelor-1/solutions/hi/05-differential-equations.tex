\section*{अध्याय \ref{ch:b1:diffeq} --- रैखिक विभेदक समीकरण}

\begin{solution}{exo:b1:diffeq:1}
सजातीय: $y_h = \lambda\,\eu^{-2x}$. विशेष: प्रयास करें $ y_p =
c\,\eu^{3x}$ ($3$ की जड़ नहीं है $ r + 2$): $3c + 2c = 1$, $ c =
\frac15$. सामान्य समाधान: $ y = \frac{\eu^{3x}}{5} +
\lambda\,\eu^{-2x}$. साथ $ y(0) = 1$: $\frac15 + \lambda = 1$,
$\lambda = \frac45$, इसलिए $ y = \frac{\eu^{3x} + 4\,\eu^{-2x}}{5}$.
\end{solution}

\begin{solution}{exo:b1:diffeq:2}
पर $\intoo{0}{+\infty}$, से विभाजित करें $ x $: $ y' - \frac{2}{x}\,y = x^2$.
यहाँ $ A(x) = -2\ln x $, $\eu^{-A(x)} = x^2$: सजातीय समाधान
$\lambda x^2$. स्थिरांकों का \omterm{def:b1:counting:objects}{परिवर्तन}: $\mu'(x) = x^2 \cdot x^{-2} =
1$, इसलिए $\mu = x + \lambda $ और
\[
y(x) = x^3 + \lambda x^2, \qquad \lambda \in \R .
\]
\emph{जाँच करना:} $ x(3x^2 + 2\lambda x) - 2(x^3 + \lambda x^2) = x^3$.
\end{solution}

\begin{solution}{exo:b1:diffeq:3}
$ y'' - 3y' + 2y = 0$: जड़ें $1$ और $2$; $\;y = \lambda\,\eu^{x} +
\mu\,\eu^{2x}$.

$ y'' + 4y' + 4y = 0$: डबल रूट $-2$; $\;y = (\lambda + \mu
x)\,\eu^{-2x}$.

$ y'' - 2y' + 5y = 0$: जड़ें $1 \pm 2\iu $; $\;y = \eu^{x}(\lambda\cos
2x + \mu\sin 2x)$.
\end{solution}

\begin{solution}{exo:b1:diffeq:4}
सजातीय: जड़ें $\pm 1$, $ y_h = \lambda\,\eu^x + \mu\,\eu^{-x}$.
दाहिनी ओर बहुपद के साथ विशेष ($\gamma = 0$ जड़ नहीं):
$ y_p = ax^2 + bx + c $; प्रतिस्थापित करना, $2a - (ax^2 + bx + c) = x^2$
देता है $ a = -1$, $ b = 0$, $ c = 2a = -2$: $ y_p = -x^2 - 2$. सामान्य
समाधान $ y = -x^2 - 2 + \lambda\eu^x + \mu\eu^{-x}$.

कॉची: $ y(0) = -2 + \lambda + \mu = 0$ और $ y'(0) = \lambda - \mu =
1$: $\lambda = \frac32$, $\mu = \frac12$. इसलिए
$ y = -x^2 - 2 + \frac{3\eu^x + \eu^{-x}}{2}$.
\end{solution}

\begin{solution}{exo:b1:diffeq:5}
$ a(x) = \tan x $, $ A(x) = -\ln(\cos x)$ (वैध: $\cos > 0$ पर
अंतराल), $\eu^{-A} = \cos x $: सजातीय समाधान $\lambda\cos x $.
स्थिरांकों का \omterm{def:b1:counting:objects}{परिवर्तन}: $\mu'(x) = \sin 2x \cdot \frac{1}{\cos x} =
2\sin x $, इसलिए $\mu = -2\cos x + \lambda $ और
\[
y(x) = -2\cos^2 x + \lambda \cos x .
\]
\emph{जाँच करना:} $ y' = 4\cos x \sin x - \lambda\sin x $ और $ y\tan x =
-2\cos x\sin x + \lambda \sin x $; उनका योग है $2\cos x\sin x =
\sin 2x $, आवश्यकता अनुसार।
\end{solution}

\begin{solution}{exo:b1:diffeq:6}
$\chi(r) = r^2 - 4r + 3 = (r-1)(r-3)$: $\gamma = 1$ एक सरल जड़ है
($ m = 1$). कोशिश $ y_p = x(ax + b)\,\eu^x $. साथ $ u = ax^2 + bx $,
\[
y_p'' - 4y_p' + 3y_p = \bigl(u'' + (2 - 4)u' + \chi(1) u\bigr)\eu^x
= \bigl(2a - 2(2ax + b)\bigr)\eu^x .
\]
साथ पहचान $(2x + 1)\eu^x $: $-4a = 2$ और $2a - 2b = 1$, इसलिए $ a =
-\frac12$, $ b = -1$. सामान्य समाधान:
\[
y = -\Bigl(\frac{x^2}{2} + x\Bigr)\eu^{x} + \lambda\,\eu^{x} +
\mu\,\eu^{3x}, \qquad (\lambda, \mu) \in \R^2 .
\]
\end{solution}

\begin{solution}{exo:b1:diffeq:7}
सजातीय: $ y_h = \lambda\cos 2x + \mu\sin 2x $.

दाहिनी ओर $ x $ ($\gamma = 0$ जड़ नहीं): $ y_1 = ax + b $ साथ
$4(ax + b) = x $: $ y_1 = \frac x4$.

दाहिनी ओर $\sin 2x = \Im(\eu^{2\iu x})$, $2\iu $ की सरल जड़
$ r^2 + 4$: कोशिश $ z = c\,x\,\eu^{2\iu x}$; तब $ z'' + 4z =
4\iu c\,\eu^{2\iu x}$, के बराबर $\eu^{2\iu x}$ के लिए $ c =
\frac{1}{4\iu} = -\frac{\iu}{4}$. इसलिए $ z = -\frac{\iu x}{4}(\cos 2x +
\iu \sin 2x)$ और $ y_2 = \Im(z) = -\frac{x\cos 2x}{4}$.

सुपरपोजिशन द्वारा:
\[
y = \frac{x}{4} - \frac{x\cos 2x}{4} + \lambda\cos 2x + \mu\sin 2x .
\]
\end{solution}

\begin{solution}{exo:b1:diffeq:8}
समीकरण $ T' + kT = 20k $ निरंतर विशेष समाधान है $20$ और
सजातीय समाधान $\lambda\eu^{-kt}$: $ T(t) = 20 +
\lambda\,\eu^{-kt}$, और $ T(0) = 80$ देता है $\lambda = 60$:
\[
T(t) = 20 + 60\,\eu^{-kt} .
\]
$ T(10) = 50$: $\eu^{-10k} = \frac12$, इसलिए $ k = \frac{\ln 2}{10}$. तब
$ T(t) = 25$ आवश्यक है $\eu^{-kt} = \frac{5}{60} = \frac{1}{12}$, यानी
\[
t = \frac{\ln 12}{k} = 10\,\frac{\ln 12}{\ln 2} \approx 35.8
\text{ minutes.}
\]
\end{solution}

\begin{solution}{exo:b1:diffeq:9}
\begin{enumerate}
  \item $\chi(r) = r^2 + 2\varepsilon r + 1$, $\Delta = 4(\varepsilon^2
        - 1)$.
        के लिए $\varepsilon \in \intco{0}{1}$: जड़ें $-\varepsilon \pm
        \iu\sqrt{1 - \varepsilon^2}$, इसलिए
        $ y = \eu^{-\varepsilon t}\bigl(\lambda\cos\omega t +
        \mu\sin\omega t\bigr)$ साथ $\omega = \sqrt{1 -
        \varepsilon^2}$.
        के लिए $\varepsilon = 1$: डबल रूट $-1$, $ y = (\lambda + \mu
        t)\,\eu^{-t}$.
        के लिए $\varepsilon > 1$: वास्तविक जड़ें $ r_\pm = -\varepsilon \pm
        \sqrt{\varepsilon^2 - 1}$, दोनों $< 0$, और $ y =
        \lambda\eu^{r_+t} + \mu\eu^{r_-t}$.
  \item के लिए $\varepsilon \in \intoo{0}{1}$: $\abs y \leq
        \eu^{-\varepsilon t}(\abs\lambda + \abs\mu) \to 0$. के लिए
        $\varepsilon = 1$: $(\lambda + \mu t)\eu^{-t} \to 0$
        (घातांक बहुपद को हराता है,
        \cref{prop:b1:functions:powerrules}). के लिए $\varepsilon > 1$:
        तब से दोनों घातांक क्षय हो गए हैं $ r_\pm < 0$ (वास्तव में
        $\sqrt{\varepsilon^2 - 1} < \varepsilon $). के लिए $\varepsilon =
        0$: $ y = \lambda\cos t + \mu\sin t $ निरंतर आयाम है
        $\sqrt{\lambda^2 + \mu^2} \neq 0$ जब तक $ y = 0$.
  \item लिखना $\lambda\cos\omega t + \mu\sin\omega t =
        R\cos(\omega t - \varphi)$ साथ $ R = \sqrt{\lambda^2 + \mu^2}
        > 0$. के शून्य $ y $ के हैं $\cos(\omega t -
        \varphi)$ (कारक $\eu^{-\varepsilon t}$ कभी नहीं
        गायब हो जाता है): $\omega t - \varphi \equiv \frac\pi2 \pmod \pi $,
        अंतराल के साथ एक अंकगणितीय प्रगति $\frac{\pi}{\omega} =
        \frac{\pi}{\sqrt{1 - \varepsilon^2}}$.
\end{enumerate}
\end{solution}

\begin{solution}{exo:b1:diffeq:10}
\omterm{def:b1:logic:sets}{तय करना} $ x = y = 0$: $2f(0) = 2f(0)^2$, और $ f(0) \neq 0$ देता है $ f(0) =
1$. हल करना $ x $ और समीकरण के संबंध में दो बार अंतर करें $ y $:
\[
f''(x+y) + f''(x-y) = 2 f(x) f''(y) .
\]
सेटिंग $ y = 0$: $\;2f''(x) = 2 f(x) f''(0)$, वह है
\[
f''(x) = c\,f(x), \qquad c = f''(0).
\]
\emph{मामला $ c = \omega^2 > 0$:} $ f(x) = \lambda\cosh\omega x +
\mu\sinh\omega x $; $ f(0) = 1$ देता है $\lambda = 1$. में प्लगिंग
कार्यात्मक समीकरण और योग सूत्रों का उपयोग करना
(\cref{prop:b1:functions:hyprules}), समीकरण बल $\mu = 0$
(गुणांकों की तुलना करें $\sinh\omega x \sinh\omega y $ या
पर मूल्यांकन करें $ x = y $): $ f = \cosh\omega x $, जो संतुष्ट करता है
$\cosh(x+y) + \cosh(x-y) = 2\cosh x\cosh y $.

\emph{मामला $ c = -\omega^2 < 0$:} इसी तरह $ f(x) = \cos\omega x $
($\omega \neq 0$), जो समीकरण को संतुष्ट करता है।

\emph{मामला $ c = 0$:} $ f $ के साथ जुड़ें $ f(0) = 1$: $ f(x) = 1 + \mu x $;
समीकरण बल $\mu = 0$, छोड़ा गया ($ f $ स्थिर नहीं)

निष्कर्ष: समाधान हैं $ f(x) = \cos\omega x $ और $ f(x) =
\cosh\omega x $, $\omega > 0$.
\end{solution}

\begin{solution}{exo:b1:diffeq:11}
\omterm{def:b1:logic:sets}{तय करना} $ z(t) = y(\eu^t)$, ताकि $ y(x) = z(\ln x)$ के लिए $ x > 0$. तब
\[
y'(x) = \frac{z'(\ln x)}x,
\qquad
y''(x) = \frac{z''(\ln x) - z'(\ln x)}{x^2} ,
\]
और समीकरण में प्रतिस्थापित:
\[
x^2y'' - xy' + y = \bigl(z'' - z'\bigr) - z' + z
= z'' - 2z' + z = 0 .
\]
अभिलक्षणिक बहुपद $(r - 1)^2$: डबल रूट $1$, इसलिए $ z(t) =
(\lambda + \mu t)\,\eu^t $ और, वापस वेरिएबल में $ x = \eu^t $:
\[
y(x) = (\lambda + \mu\ln x)\,x,
\qquad \lambda, \mu \in \R .
\]
\end{solution}

\begin{solution}{exo:b1:diffeq:12}
\begin{enumerate}
  \item सामान्यीकृत रूप में $ y' - \frac2x\,y = 0$ प्रत्येक अंतराल पर:
        $ A(x) = -2\ln\abs x $, तो समाधान हैं $ y = a x^2$ पर
        $\intoo0{+\infty}$ और $ y = b x^2$ पर $\intoo{-\infty}0$,
        स्वतंत्र स्थिरांक के साथ
        (\cref{thm:b1:diffeq:homogeneous1}).
  \item होने देना $ y = ax^2$ के लिए $ x \geq 0$ और $ bx^2$ के लिए $ x < 0$. पर
        प्रत्येक खुली आधी पंक्ति $ y $ से भिन्न है $ xy' = 2y $.
        पर $0$: अंतर भागफल $\frac{y(h) - y(0)}h =
        ah $ या $ bh $ प्रवृत्त $0$, इसलिए $ y'(0) = 0$ मौजूद है, और
        समीकरण पर $ x = 0$ पढ़ता $0 \cdot y'(0) = 2y(0) = 0$:
        संतुष्ट। इसलिए $ y $ सभी पर समीकरण हल करता है $\R $.
  \item समाधान चालू $\R $ वास्तव में ये चिपके हुए कार्य हैं: a
        \emph{दो}-प्रथम-क्रम समीकरण के लिए पैरामीटर परिवार।
        के साथ कोई विरोधाभास नहीं है \cref{thm:b1:diffeq:voc},
        जिनकी परिकल्पनाएँ यहाँ विफल होती हैं: के रूप में लिखा गया है $ y' + a(x)y = 0$,
        गुणांक $ a(x) = -\frac2x $ पर सतत नहीं है $0$
        --- वास्तव में परिभाषित नहीं --- तो $\R $ पर कोई अंतराल नहीं है
        जो प्रमेय लागू होता है. पर विलक्षणता $0$
        दो अर्ध-पंक्तियों और मान को अलग कर देता है $ y(0) = 0$
        कोई जानकारी नहीं ले जाने के लिए मजबूर किया जाता है। हर कॉची
        डेटम एट $ x_0 \neq 0$ पर ही समाधान निर्धारित करता है
        अर्ध-पंक्ति युक्त $ x_0$.
\end{enumerate}
\end{solution}

\begin{solution}{pb:b1:diffeq:1}
\textbf{1.} $\chi(r) = r^2 + 2\lambda r + \omega_0^2$, $\Delta =
4(\lambda^2 - \omega_0^2) < 0$ के लिए $0 < \lambda < \omega_0$: जड़ें
$-\lambda \pm \iu\omega_d $, तो द्वारा
\cref{thm:b1:diffeq:homogeneous2}
\[
x(t) = \eu^{-\lambda t}\bigl(\lambda_1\cos\omega_d t +
\mu_1\sin\omega_d t\bigr), \qquad (\lambda_1, \mu_1) \in \R^2 .
\]
के लिए $\lambda = 0$: $ x = \lambda_1\cos\omega_0 t +
\mu_1\sin\omega_0 t $. महत्वपूर्ण ($\lambda = \omega_0$) और
अत्यधिक नमीयुक्त ($\lambda > \omega_0$) शासन वे हैं
\cref{exo:b1:diffeq:9} (समय घटाने के बाद): $(\lambda_1 +
\mu_1 t)\eu^{-\lambda t}$, सम्मान।\ के संयोजन $\eu^{r_\pm t}$
साथ $ r_\pm = -\lambda \pm \sqrt{\lambda^2 - \omega_0^2}$.

\textbf{2.} कम नमीयुक्त: $\abs x \leq \eu^{-\lambda
t}(\abs{\lambda_1} + \abs{\mu_1}) \to 0$. गंभीर: $(\lambda_1 +
\mu_1 t)\eu^{-\lambda t} \to 0$ चूँकि घातांक हराते हैं
बहुपद (\cref{prop:b1:functions:powerrules}). अत्यधिक नमीयुक्त:
$ r_- < r_+ = -\lambda + \sqrt{\lambda^2 - \omega_0^2} < 0$
क्योंकि $\sqrt{\lambda^2 - \omega_0^2} < \lambda $; दोनों
घातीय क्षय.

\textbf{3.} के समाधान के साथ $(H)$, का उपयोग करना $ x'' = -2\lambda x' -
\omega_0^2 x $:
\[
\mathcal E'(t) = x'x'' + \omega_0^2 x x'
= x'\bigl(-2\lambda x' - \omega_0^2 x\bigr) + \omega_0^2 xx'
= -2\lambda\,x'^2 \leq 0 .
\]
अगर $ x(t_0) = x'(t_0) = 0$ तब $\mathcal E(t_0) = 0$; $\mathcal E $
गैर-नकारात्मक और गैर-वृद्धिशील है, इसलिए $\mathcal E \equiv 0$ पर
$\intco{t_0}{+\infty}$, मजबूर करना $ x \equiv 0$ वहाँ; के लिए $ t \leq
t_0$, वही तर्क चलाएँ $\tilde x(t) = x(2t_0 - t)$, जो
अवमंदन के साथ समीकरण को हल करता है $-\lambda $ लेकिन अभी भी है
$\tilde{\mathcal E}(t_0) = 0$ और $\tilde{\mathcal E}' =
+2\lambda\tilde x'^2 \geq 0$ साथ $\tilde{\mathcal E} \geq 0$;
गैर-नकारात्मक, गैर-घटता हुआ और दाएँ छोर पर शून्य
$\intoc{-\infty}{t_0}$ मतलब सर्वत्र शून्य. इसलिए $ x \equiv 0$ पर
$\R $ --- विशिष्टता का एक ऊर्जा प्रमाण, सभी के लिए मान्य $\lambda
\geq 0$.

\textbf{4.} $ x(t + T_d) = R\,\eu^{-\lambda t}\eu^{-\lambda
T_d}\cos(\omega_d t + 2\pi - \varphi) = \eu^{-\lambda T_d}x(t)$.
प्रति छद्म अवधि में सिकुड़न कारक है $\eu^{-\delta}$ साथ
$\delta = \lambda T_d = \frac{2\pi\lambda}{\omega_d}$. के लिए
$\omega_0 = 1$, $\lambda = 0.1$: $\omega_d = \sqrt{0.99} =
0.99499$, इसलिए $\delta = \frac{0.62832}{0.99499} = 0.6315$: प्रत्येक
झूलता रहता है $\eu^{-0.63} \approx 53\%$ इसके आयाम का.

\textbf{5.} आयाम कारक है $\eu^{-\lambda t}$, जो
द्वारा क्षय होता है $\eu $ ऊपर $ t = \frac1\lambda $. उस अंतराल में शामिल है
$\frac{1/\lambda}{T_d} = \frac{\omega_d}{2\pi\lambda}$
छद्म काल. के लिए $\lambda \ll \omega_0$, $\omega_d \approx
\omega_0$ और यह है $\approx \frac{\omega_0}{2\pi\lambda} =
\frac Q\pi $. ए $ Q = 300$ गिटार के तार लगभग सौ तक बजते हैं
अवधि; ए $ Q = 1$ डोर डैम्पर एक को पूरा नहीं करता है।

\textbf{6.} स्थानापन्न $ z\,\eu^{\iu\Omega t}$ बाएँ हाथ में
पक्ष देता है $ z\,(-\Omega^2 + 2\iu\lambda\Omega +
\omega_0^2)\,\eu^{\iu\Omega t}$, जो बराबर है $ A\,\eu^{\iu\Omega
t}$ बिल्कुल के लिए $ z = \frac{A}{\omega_0^2 - \Omega^2 +
2\iu\lambda\Omega}$ (हर अशून्य है: यह काल्पनिक है
भाग है $2\lambda\Omega > 0$). चूँकि गुणांक वास्तविक हैं,
असली हिस्सा $ x_p = \Re\bigl(z\eu^{\iu\Omega t}\bigr)$ हल करता है
दाहिनी ओर वाला समीकरण $\Re\bigl(A\eu^{\iu\Omega t}\bigr) =
A\cos\Omega t $.

\textbf{7.} लिखना $\omega_0^2 - \Omega^2 + 2\iu\lambda\Omega =
\sqrt{D}\,\eu^{\iu\varphi}$ साथ $ D = (\omega_0^2 - \Omega^2)^2 +
4\lambda^2\Omega^2$ और $\varphi \in \intoo0\pi $ (काल्पनिक
भाग $2\lambda\Omega $ सकारात्मक है), इसलिए $\tan\varphi =
\frac{2\lambda\Omega}{\omega_0^2 - \Omega^2}$. तब $ z =
\frac{A}{\sqrt D}\eu^{-\iu\varphi}$ और
\[
x_p(t) = \Re\Bigl(\frac A{\sqrt D}\,\eu^{\iu(\Omega t -
\varphi)}\Bigr) = R\cos(\Omega t - \varphi),
\qquad R = \frac A{\sqrt D} .
\]

\textbf{8.} जैसा $\Omega \to 0^+$: $ D \to \omega_0^4$, इसलिए $ R \to
A/\omega_0^2$ और $\tan\varphi \to 0^+$ साथ $\varphi \in
\intoo0{\frac\pi2}$: $\varphi \to 0$. द्रव्यमान बल का अनुसरण करता है
अर्ध-स्थैतिक रूप से, बल/कठोरता से विस्थापित। जैसा $\Omega \to
+\infty $: $ D \sim \Omega^4$, इसलिए $ R \sim A/\Omega^2 \to 0$, और
$\varphi \to \pi $ (जटिल संख्या $\omega_0^2 - \Omega^2 +
2\iu\lambda\Omega $ तर्क के साथ दूसरे चतुर्थांश में जाता है $\to
\pi $): द्रव्यमान मुश्किल से चलता है, और चरण के विरोध में ---
जड़ता हावी है.

\textbf{9.} द्वारा \cref{thm:b1:diffeq:structure2}~(1), हर समाधान
है $ x = x_p + x_h $ साथ $ x_h $ सुलझाने $(H)$; प्रश्न 2 से, $ x_h(t)
\to 0$, इसलिए $ x(t) - x_p(t) \to 0$: सभी समाधान एकाग्र हो जाते हैं
वही स्थिर अवस्था. प्रारंभिक स्थितियाँ ही क्षणिक को आकार देती हैं।

\textbf{10.} अगर $ x $ एक आवधिक समाधान है, $ x - x_p = x_h $ एक है
का आवधिक समाधान $(H)$ जिसकी प्रवृत्ति होती है $0$ पर $+\infty $; ए
सीमा के साथ आवधिक कार्य $0$ समान रूप से है $0$ (इसके मान चालू हैं
एक अवधि हमेशा के लिए दोहराई जाती है, इसलिए प्रत्येक मान एक की सीमा है
इसके बाद की ओर रुझान $0$). इस तरह $ x = x_p $.

\textbf{11.} यहाँ $\lambda = 1$, $\omega_0^2 = 2$, $\Omega = 1$,
$ A = 1$: $ z = \frac1{2 - 1 + 2\iu} = \frac{1 - 2\iu}5$, इसलिए
\[
x_p = \Re\Bigl(\frac{(1 - 2\iu)(\cos t + \iu\sin t)}5\Bigr)
= \frac{\cos t + 2\sin t}5 .
\]
सजातीय: की जड़ें $ r^2 + 2r + 2$ हैं $-1 \pm \iu $: $ x_h =
\eu^{-t}(C\cos t + S\sin t)$. स्थितियाँ: $ x(0) = \frac15 + C = 0$
देता है $ C = -\frac15$; विभेद करना, $ x'(0) = \frac25 - C + S =
0$ देता है $ S = C - \frac25 = -\frac35$. इस तरह
\[
x(t) = \underbrace{\frac{\cos t + 2\sin t}5}_{\text{steady}}
- \underbrace{\eu^{-t}\,\frac{\cos t + 3\sin
t}5}_{\text{transient}} ,
\]
क्षणभंगुर मरने जैसा $\eu^{-t}$.

\textbf{12.} विस्तार, $ g(u) = u^2 - 2(\omega_0^2 - 2\lambda^2)u
+ \omega_0^4 = (u - u_r)^2 + g(u_r)$ साथ $ u_r = \omega_0^2 -
2\lambda^2$ और
\[
g(u_r) = \omega_0^4 - u_r^2 = (\omega_0^2 - u_r)(\omega_0^2 +
u_r) = 2\lambda^2\bigl(2\omega_0^2 - 2\lambda^2\bigr) =
4\lambda^2(\omega_0^2 - \lambda^2) .
\]
अगर $2\lambda^2 < \omega_0^2$, तब $ u_r > 0$ एक स्वीकार्य है
वर्ग आवृत्ति: $ g $ वहां सख्त न्यूनतम है, इसलिए $ R = A/\sqrt
g $ पर सख्त अधिकतम है $\Omega_r = \sqrt{u_r} =
\sqrt{\omega_0^2 - 2\lambda^2}$, साथ $ R_{\max} = A/\sqrt{g(u_r)}
= \frac{A}{2\lambda\sqrt{\omega_0^2 - \lambda^2}}$.

\textbf{13.} $ R(0) = A/\omega_0^2$, इसलिए
\[
\frac{R_{\max}}{R(0)}
= \frac{\omega_0^2}{2\lambda\sqrt{\omega_0^2 - \lambda^2}}
= \frac{\omega_0}{2\lambda}\cdot
\frac{\omega_0}{\sqrt{\omega_0^2 - \lambda^2}}
= Q\,\Bigl(1 - \frac{\lambda^2}{\omega_0^2}\Bigr)^{-1/2}
\geq Q .
\]
कमज़ोर अवमंदन के लिए सुधार कारक निकट है $1$: \omterm{ex:b1:diffeq:oscillation}{गूंज}
स्थैतिक विस्थापन को अनिवार्य रूप से गुणा करता है $ Q $.

\textbf{14.} $ V(\Omega)^2 = \frac{A^2 u}{g(u)}$ साथ $ u =
\Omega^2$. इसके व्युत्पत्ति का चिन्ह है $ g(u) - u\,g'(u) =
(\omega_0^2 - u)^2 + 4\lambda^2 u - u\bigl(2(u - \omega_0^2) +
4\lambda^2\bigr) = (\omega_0^2 - u)^2 + 2u(\omega_0^2 - u) =
(\omega_0^2 - u)(\omega_0^2 + u)$, के लिए सकारात्मक $ u < \omega_0^2$
और उससे परे नकारात्मक: बिल्कुल सख्त अधिकतम $\Omega =
\omega_0$, प्रत्येक अवमंदन के लिए। वहाँ $\tan\varphi $ के साथ विस्फोट करता है
$\varphi \in \intoo0\pi $: $\varphi = \frac\pi2$, और $ x_p'(t) =
-R\Omega\sin(\Omega t - \frac\pi2) = R\Omega\cos(\Omega t)$ है
बिल्कुल बल के चरण में: इष्टतम शक्ति हस्तांतरण।

\textbf{15.} $ R = R_{\max}/\sqrt2 \iff g(u) = 2g(u_r) \iff (u -
u_r)^2 = g(u_r) = 4\lambda^2(\omega_0^2 - \lambda^2)$, देना
\[
u_\pm = u_r \pm 2\lambda\sqrt{\omega_0^2 - \lambda^2},
\qquad
\Omega_+^2 - \Omega_-^2 = 4\lambda\sqrt{\omega_0^2 - \lambda^2} .
\]
तब $\Omega_+ - \Omega_- = \frac{\Omega_+^2 -
\Omega_-^2}{\Omega_+ + \Omega_-}$, और के लिए $\lambda \ll \omega_0$
दोनों $\Omega_\pm \approx \omega_0$: $\Omega_+ - \Omega_- \approx
\frac{4\lambda\omega_0}{2\omega_0} = 2\lambda $, इसलिए
$\frac{\omega_0}{\Omega_+ - \Omega_-} \approx
\frac{\omega_0}{2\lambda} = Q $. मापना ए \omterm{ex:b1:diffeq:oscillation}{गूंज} शिखर की चौड़ाई
इसका माप करता है \omterm{pb:b1:diffeq:1}{गुणवत्ता कारक}.

\textbf{16.} $ Q = 10$; $\Omega_r = \sqrt{1 - 2(0.05)^2} =
\sqrt{0.995} = 0.9975$; $ R_{\max} = \frac1{2 \times 0.05
\sqrt{1 - 0.0025}} = \frac1{0.1 \times 0.99875} = 10.01$; स्थैतिक
प्रतिक्रिया $ R(0) = 1$; बैंडविड्थ $\approx 2\lambda = 0.1$. एक लंबा
ऊंचाई की पतली कील $\approx Q $ ऊंचाई के एक पठार के ऊपर $1$.

\textbf{17.} अगर $2\lambda^2 \geq \omega_0^2$ तब $ u_r \leq 0$ और
$ g'(u) = 2(u - u_r) > 0$ सभी के लिए $ u > 0$: $ g $ सख्ती से बढ़ता है
पर $\intoo0{+\infty}$, इसलिए $ R = A/\sqrt g $ से सख्ती से घटता है
$ R(0) = A/\omega_0^2$: प्रतिक्रिया शून्य आवृत्ति पर सबसे बड़ी है
और कोई शिखर नहीं है.

\textbf{18.} $\gamma = \iu\Omega $ की जड़ नहीं है $ r^2 +
\omega_0^2$ (जैसा $\Omega \neq \omega_0$), इसलिए
\cref{met:b1:diffeq:particular} साथ $ m = 0$ देता है $ x_p =
\frac{A\cos\Omega t}{\omega_0^2 - \Omega^2}$ (स्थानापन्न और
जांचें: $-\Omega^2 + \omega_0^2$ कोसाइन का गुना)। सामान्य
समाधान:
\[
x(t) = \frac{A\cos\Omega t}{\omega_0^2 - \Omega^2}
+ \lambda_1\cos\omega_0 t + \mu_1\sin\omega_0 t .
\]

\textbf{19.} $ x(0) = 0$ बल $\lambda_1 = -\frac A{\omega_0^2 -
\Omega^2}$, और $ x'(0) = 0$ बल $\mu_1 = 0$:
\[
x(t) = \frac{A}{\omega_0^2 - \Omega^2}\,
\bigl(\cos\Omega t - \cos\omega_0 t\bigr)
= \frac{2A}{\omega_0^2 - \Omega^2}\,
\sin\Bigl(\frac{(\omega_0 - \Omega)t}2\Bigr)
\sin\Bigl(\frac{(\omega_0 + \Omega)t}2\Bigr) ,
\]
उत्पाद सूत्र द्वारा $\cos a - \cos b =
2\sin\frac{b - a}2\sin\frac{b + a}2$ के साथ लागू किया गया $ a = \Omega t $,
$ b = \omega_0 t $.

\textbf{20.} के लिए $\Omega $ पास में $\omega_0$, दूसरी साइन
तीव्र आवृत्ति पर दोलन करता है $\frac{\omega_0 + \Omega}2
\approx \omega_0$, जबकि पहला आवृत्ति का एक धीमा आवरण है
$\frac{\abs{\omega_0 - \Omega}}2$: तेज का आयाम
लिफाफा अवधि के साथ दोलन बढ़ता और घटता है
$\frac{2\pi}{\abs{\omega_0 - \Omega}}$ (प्रति लिफ़ाफ़ा दो बीट
अवधि), अधिकतम तक पहुँचना $\frac{2A}{\abs{\omega_0^2 - \Omega^2}}$.
जैसा $\Omega \to \omega_0$, धड़कनें धीमी हो जाती हैं (अवधि)।
$\to \infty $) और लंबा (आयाम $\to \infty $).

\textbf{21.} हल करना $ t $. जैसा $\Omega \to \omega_0$:
\[
\frac{2A}{\omega_0^2 - \Omega^2}\sin\Bigl(\frac{(\omega_0 -
\Omega)t}2\Bigr)
= \frac{2A}{\omega_0 + \Omega}\cdot
\frac{\sin\bigl(\frac{(\omega_0 - \Omega)t}2\bigr)}
{\omega_0 - \Omega}
\longrightarrow \frac{2A}{2\omega_0}\cdot\frac t2
= \frac{At}{2\omega_0} ,
\]
जबकि $\sin\bigl(\frac{(\omega_0 + \Omega)t}2\bigr) \to
\sin(\omega_0 t)$: हद है $ x_\infty(t) =
\frac{At\sin\omega_0 t}{2\omega_0}$. सीधी जांच: साथ $ C =
\frac A{2\omega_0}$, $ x_\infty = Ct\sin\omega_0 t $ है
$ x_\infty'' = 2C\omega_0\cos\omega_0 t - C\omega_0^2 t
\sin\omega_0 t $, इसलिए $ x_\infty'' + \omega_0^2 x_\infty =
2C\omega_0\cos\omega_0 t = A\cos\omega_0 t $, साथ $ x_\infty(0) =
0$ और $ x_\infty'(0) = C\sin 0 + C\omega_0 \cdot 0 \cdot \cos 0 =
0$ --- के लिए $\omega_0 = A = 1$ यह है
बिलकुल \cref{ex:b1:diffeq:oscillation}. \omterm{ex:b1:diffeq:oscillation}{गूंज} है
धड़कनों का पतन: लिफाफे का पहला उभार, फैला हुआ
अनंत लंबाई तक.

\textbf{22.} अवमंदन के बिना गुंजयमान आयाम रैखिक रूप से बढ़ता है
और बिना बंधन के; भीगने के साथ $\lambda > 0$ विकास संतृप्त होता है
पर $ R_{\max} \approx Q\,\frac A{\omega_0^2}$. तंत्र:
अपव्यय आयाम के साथ बढ़ती दर से ऊर्जा को हटाता है
(प्रश्न 23), इसलिए भिगोना होने पर निर्माण बिल्कुल रुक जाता है
ऊर्जा उतनी ही तेजी से जलती है जितनी तेजी से बल उसे आपूर्ति करता है।

\textbf{23.} साथ $ x_p' = -R\Omega\sin(\Omega t - \varphi)$, खत्म
एक अवधि औसत $\langle\cos^2\rangle =
\langle\sin^2\rangle = \frac12$ और $\langle\sin\cos\rangle = 0$
देना:
\[
\langle P_{\mathrm{diss}}\rangle
= 2\lambda\,R^2\Omega^2\,\langle\sin^2(\Omega t -
\varphi)\rangle = \lambda R^2\Omega^2 ;
\]
और विस्तार $\sin(\Omega t - \varphi) = \sin\Omega t\cos\varphi
- \cos\Omega t\sin\varphi $:
\[
\langle P_{\mathrm{in}}\rangle
= -AR\Omega\,\bigl\langle\cos\Omega t\,\sin(\Omega t -
\varphi)\bigr\rangle
= AR\Omega\,\frac{\sin\varphi}2 .
\]
तब से $\sin\varphi = \frac{2\lambda\Omega}{\sqrt D} =
\frac{2\lambda\Omega R}A $, यह है $\frac{AR\Omega}2 \cdot
\frac{2\lambda\Omega R}A = \lambda R^2\Omega^2$: \omterm{def:b1:logic:inj}{इंजेक्शन} और
बिखरी हुई शक्तियां बिल्कुल संतुलित होती हैं --- ए की परिभाषित संपत्ति
स्थिर शासन.

\textbf{24.} (i) संरचना प्रमेय प्रत्येक समाधान को विभाजित करता है
स्थिर अवस्था प्लस क्षणिक (प्रश्न 9-11) और घटाया गया
सजातीय समस्या के लिए विशिष्टता. (ii) जटिल विधि
एक विशेष समाधान की खोज को एक विभाजन में बदल दिया
सम्मिश्र संख्याएँ (प्रश्न 6), आयाम और चरण के साथ पढ़ें
\omterm{def:b1:complex:field}{मापांक} और एक तर्क. (iii) द \omterm{ex:b1:diffeq:oscillation}{गूंज} वक्र एक शुद्ध है
फ़ंक्शन अध्ययन --- एक द्विघात में $ u = \Omega^2$, यह न्यूनतम है,
इसका स्तर सेट --- की शैली में \cref{ch:b1:functions}
(प्रश्न 12-17).

\textbf{25.} मुक्त और नम: क्षयकारी छद्म-दोलन,
जीवनकाल $\frac1\lambda $, के बारे में $\frac Q\pi $ झूले. मजबूर और
नम: क्षणिक मर जाते हैं, और एक अद्वितीय साइनसॉइडल स्थिर अवस्था
बलपूर्वक आवृत्ति पर जीवित रहता है, आयाम निकट चरम पर होता है
$\omega_0$ (ऊंचाई $\approx Q \times $ स्थिर, चौड़ाई $\approx
\frac{\omega_0}Q $) और चरण स्वीपिंग से $0$ को $\pi $ के माध्यम से
$\frac\pi2$ पर $\omega_0$. स्वतंत्र और अविचल: शाश्वत
दोलन. बलपूर्वक और अवमंदित: \omterm{pb:b1:diffeq:1}{धड़कता है}, पतित होता जाता है
रैखिक रूप से बढ़ रहा है \omterm{ex:b1:diffeq:oscillation}{गूंज} सटीक ट्यूनिंग पर. एक आयामहीन
संख्या, $ Q = \frac{\omega_0}{2\lambda}$, सब कुछ धुनता है ---
चरम ऊंचाई, बैंडविड्थ और क्षणिक जीवनकाल तीन रीडिंग हैं
एक ही डायल का. अगली कड़ी: पुनर्लेखन $ x'' + 2\lambda x' +
\omega_0^2x$ प्रथम-क्रम प्रणाली के मैट्रिक्स तरीकों को खोलता है
\cref{ch:b1:matrices} और वर्ष 2 खंड, और विघटित करना
साइनसोइड्स में मनमाना आवधिक बल (फूरियर श्रृंखला, वर्ष 3
वॉल्यूम) इस समस्या का एकल-आवृत्ति विश्लेषण बनाता है
यूनिवर्सल बिल्डिंग ब्लॉक: प्रत्येक आवृत्ति के लिए हल करें, सुपरपोज़ करें।
\end{solution}
