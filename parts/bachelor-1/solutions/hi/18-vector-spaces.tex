\begin{solution}{exo:b1:vspaces:1}
\begin{enumerate}
  \item हाँ: शामिल है $0$, और परिभाषित समीकरण रैखिक है
        (नीचे स्थिर $ x + \lambda y $).
  \item नहीं: शामिल नहीं है $(0,0,0)$.
  \item नहीं: $(1, 0)$ और $(0, -1)$ संबंधित ($ xy = 0$), उनका योग $(1,
        -1)$ नहीं ($ xy = -1 < 0$).
  \item हाँ: $0$ पर गायब हो जाता है $1$; $(P + \lambda Q)(1) = P(1) +
        \lambda Q(1) = 0$.
  \item हाँ: शून्य फ़ंक्शन परिबद्ध है; अगर $\abs f \leq M $ और
        $\abs g \leq M'$, तब $\abs{f + \lambda g} \leq M +
        \abs\lambda M'$.
\end{enumerate}
\end{solution}

\begin{solution}{exo:b1:vspaces:2}
$(1,2,1) = a(1,0,1) + b(1,1,0)$ आवश्यक है $ a + b = 1$, $ b = 2$, $ a =
1$: असंगत ($ a + b = 3 \neq 1$): में नहीं \omterm{def:b1:vspaces:span}{अवधि}. $(2,1,1) =
a(1,0,1) + b(1,1,0)$: $ b = 1$, $ a = 1$, $ a + b = 2$: सुसंगत, इसलिए
$(2,1,1) = (1,0,1) + (1,1,0)$, में \omterm{def:b1:vspaces:span}{अवधि}.

समीकरण: $(x, y, z) = (a + b, b, a)$ मतलब $ x = y + z $: द \omterm{def:b1:vspaces:span}{अवधि} है
विमान $\{x - y - z = 0\}$.
\end{solution}

\begin{solution}{exo:b1:vspaces:3}
पहला परिवार: $\lambda(1,1,0) + \mu(1,0,1) + \nu(0,1,1) = 0$ देता है
$\lambda + \mu = 0$, $\lambda + \nu = 0$, $\mu + \nu = 0$: जोड़ना,
$2(\lambda + \mu + \nu) = 0$, और प्रत्येक समीकरण को घटाने पर,
$\lambda = \mu = \nu = 0$: \omterm{def:b1:vspaces:free}{मुक्त}.

दूसरा: $(2,4,6) = 2(1,2,3)$: जुड़ा हुआ.

तीसरा: चार सदिश $\R^3$ --- आवश्यक रूप से एक बार आयाम जुड़ा हुआ है
उपलब्ध है (\cref{ch:b1:findim}); सीधे: $(0,1,1) = -(1,0,0) +
0\cdot(1,1,0) + (1,1,1)$, वास्तव में $(-1,0,0) + (1,1,1) = (0,1,1)$: ए
गैरतुच्छ संबंध.
\end{solution}

\begin{solution}{exo:b1:vspaces:4}
\omterm{def:b1:poly:def}{बहुपद}ों $1, (X-1), (X-1)^2, (X-1)^3$ अलग-अलग डिग्री हैं $0,
1, 2, 3$: \omterm{def:b1:vspaces:free}{मुक्त} (\cref{prop:b1:vspaces:freecriteria}~(1)); चार \omterm{def:b1:vspaces:free}{मुक्त}
वैक्टर उत्पन्न करना (प्रत्येक $ P \in \R_3[X]$ की शक्तियों में विस्तार होता है $ X -
1$, उदा.\ टेलर द्वारा बहुआयामी पद, सी.एफ.\ का प्रमाण
\cref{prop:b1:poly:multiplicity}): ए \omterm{def:b1:vspaces:free}{आधार}. के लिए $ X^3$, टेलर पर
$1$: $ P = X^3$, $ P(1) = 1$, $ P'(1) = 3$, $ P''(1) = 6$, $ P'''(1) =
6$:
\[
X^3 = 1 + 3(X - 1) + 3(X-1)^2 + (X-1)^3 ,
\]
\omterm{prop:b1:vspaces:coordinates}{COORDINATES} $(1, 3, 3, 1)$ (पास्कल की पंक्ति, जैसी कि अपेक्षा थी $ X^3 =
((X-1)+1)^3$).
\end{solution}

\begin{solution}{exo:b1:vspaces:5}
$ F \cap G $: शर्तें $ x = y = z $ और $ x = t = 0$ एक साथ दे
$ x = 0$, इस तरह $ y = z = 0$, और $ t = 0$: चौराहा है
$\{0\}$. योग: दिया गया $(x,y,z,t)$, तलाश $(a,a,a,b) \in F $ और

z - x $: सदैव संभव. इसलिए $\R^4 = F \oplus G $, और
\[
(1,2,3,4) = (1,1,1,4) + (0,1,2,0) .
\]
\end{solution}

\begin{solution}{exo:b1:vspaces:6}
का एक तत्व $ G $ है $\lambda u $; यदि यह अभिसरण होता है, तो (जैसे)
$\lambda u_n = \lambda(-1)^n $ इसकी दो अनुवर्ती सीमाएँ हैं $\pm
\lambda $) अनिवार्य रूप से $\lambda = 0$: $ F \cap G = \{0\}$.

$ F + G $ सब कुछ नहीं है: इसमें रूप के अनुक्रम शामिल हैं
$ c_n + \lambda(-1)^n $ साथ $(c_n)$ अभिसारी। क्रम $ v_n =
n $ इस रूप का नहीं है ($ v_n - \lambda(-1)^n $ \omterm{def:b1:reals:bounds}{असीम}ित है, कदापि नहीं
अभिसरण)। इसलिए $ F \oplus G \subsetneq $ (सभी अनुक्रमों का स्थान)।
\end{solution}

\begin{solution}{exo:b1:vspaces:7}
($\supseteq $) दोनों $ F \cap G $ और $ F \cap H $ रिहायश $ F $, और उनके
योग निहित है $ G + (F \cap H)$: समावेश बाईं ओर से होता है
एक है \omterm{def:b1:vspaces:subspace}{उपस्पेस} दोनों टुकड़ों से युक्त --- ठोस रूप से, एक तत्व $ g +
h $ साथ $ g \in F\cap G $, $ h \in F \cap H $ में है $ F $ (दो का योग
के तत्व $ F $) और में $ G + (F \cap H)$.

($\subseteq $) होने देना $ x \in F $ साथ $ x = g + h $, $ g \in G $, $ h \in F
\cap H $. तब $ g = x - h \in F $ (तत्वों का अंतर $ F $), इसलिए
$ g \in F \cap G $, और $ x = g + h \in (F \cap G) + (F \cap H)$.

में पूर्ण वितरण का प्रति उदाहरण $\R^2$: $ F =
\operatorname{Vect}(1,1)$, $ G = \operatorname{Vect}(1,0)$, $ H =
\operatorname{Vect}(0,1)$. तब $ G + H = \R^2$, इसलिए $ F \cap (G+H) =
F $, जबकि $ F \cap G = F \cap H = \{0\}$: दाहिनी ओर है $\{0\}
\neq F $.
\end{solution}

\begin{solution}{exo:b1:vspaces:8}
\begin{enumerate}
  \item कल्पना करना $\sum_{i} \lambda_i \eu^{a_i x} = 0$ सभी के लिए $ x $.
        गुणा करके $\eu^{-a_p x}$: $\lambda_p + \sum_{i < p}
        \lambda_i \eu^{(a_i - a_p)x} \to \lambda_p $ जैसा $ x \to
        +\infty $ (प्रत्येक प्रतिपादक $ a_i - a_p < 0$). बाईं ओर है
        समान रूप से $0$, इसलिए $\lambda_p = 0$; नीचे की ओर पुनरावृत्त करें.
  \item होने देना $ a\cos x + b \sin x + c \cos 2x + d\sin 2x = 0$ सभी के लिए
        $ x $. पर मूल्यांकन करें $ x = 0$: $ a + c = 0$; पर $ x = \pi $: $-a + c
        = 0$; इस तरह $ a = c = 0$, और संबंध कम हो जाता है $ b\sin
        x + d \sin 2x = 0$. पर मूल्यांकन करें $ x = \frac\pi2$: $ b = 0$;
        तो फिर $ x = \frac\pi4$: $ d = 0$.
  \item कल्पना करना $\sum \lambda_i \abs{x - a_i} = 0$ सभी के लिए $ x $. द
        समारोह $\sum_{i \neq j} \lambda_i\abs{x - a_i}$ है
        विभेदक पर $ a_j $ (प्रत्येक पद अपने आप से दूर है
        कोने), तो $-\lambda_j \abs{x - a_j}$, उनका अंतर,
        होना ही चाहिए विभेदक पर $ a_j $ भी --- जो मजबूर करता है
        $\lambda_j = 0$ ($\abs{\,\cdot\,}$ एक कोना है)। यह कायम है
        प्रत्येक के लिए $ j $.
\end{enumerate}
\end{solution}

\begin{solution}{exo:b1:vspaces:9}
होने देना $ h \in H $. तब से $ h \in H \subseteq F + H = F + G $, लिखना $ h = f
+ g $ साथ $ f \in F $, $ g \in G $. तब $ f = h - g \in H $ (दोनों शर्तें
में $ H $, का उपयोग करना $ G \subseteq H $), इसलिए $ f \in F \cap H = F \cap G
\subseteq G $, और $ h = f + g \in G $. इस तरह $ H \subseteq G $, और साथ में
परिकल्पना $ G \subseteq H $: समानता.

बिना प्रति उदाहरण $ G \subseteq H $: में $\R^2$, लेना $ F =
\operatorname{Vect}(1,0)$, $ G = \operatorname{Vect}(0,1)$, $ H =
\operatorname{Vect}(1,1)$: तब $ F + G = F + H = \R^2$ और $ F \cap G
= F \cap H = \{0\}$, अभी तक $ G \neq H $.
\end{solution}

\begin{solution}{exo:b1:vspaces:10}
दोनों सेट हैं उपस्थान (परिभाषित स्थितियां रैखिक हैं और
के लिए रुको $0$). अपघटन: के लिए $ P \in \R[X]$,
\[
P(X) = \underbrace{\frac{P(X) + P(-X)}{2}}_{\in\,\mathcal P}
+ \underbrace{\frac{P(X) - P(-X)}{2}}_{\in\,\mathcal I},
\]
और ए \omterm{def:b1:poly:def}{बहुपद} सम और विषम दोनों संतुष्ट करते हैं $ P = -P $, इसलिए $ P = 0$:
योग सीधा और बराबर है $\R[X]$.

अब चलो $ P = \sum_k a_k X^k $ सम हो. तब $ P(X) - P(-X) = 2
\sum_{k \text{ odd}} a_k X^k $ शून्य है \omterm{def:b1:poly:def}{बहुपद}, तो हर
विषम-डिग्री गुणांक गायब हो जाता है (\cref{def:b1:poly:def}): $ P \in
\operatorname{Vect}(1, X^2, X^4, \dots)$, यानी\ $ P = Q(X^2)$ एक के लिए
\omterm{def:b1:poly:def}{बहुपद} $ Q $. इसके विपरीत प्रत्येक \omterm{def:b1:poly:def}{बहुपद} में $ X^2$ सम है.
\end{solution}

\begin{solution}{exo:b1:vspaces:11}
कल्पना करना $ a(x_1 + x_2) + b(x_2 + x_3) + c(x_3 + x_1) = 0$.
पर पुनः \omterm{def:b1:structures:group}{समूह} बनाना \omterm{def:b1:vspaces:free}{मुक्त} परिवार $(x_1, x_2, x_3)$:
\[
(a + c)\,x_1 + (a + b)\,x_2 + (b + c)\,x_3 = 0
\implies a + c = a + b = b + c = 0 .
\]
पहले दो समीकरण घटाने पर प्राप्त होता है $ c = b $; फिर तीसरा
देता है $2b = 0$, इसलिए $ b = c = 0$, तब $ a = 0$: परिवार है \omterm{def:b1:vspaces:free}{मुक्त}.

चार सदिशों के लिए अनुरूप परिवार है \emph{हमेशा} जुड़ा हुआ:
\[
(x_1 + x_2) - (x_2 + x_3) + (x_3 + x_4) - (x_4 + x_1) = 0
\]
एक गैर-तुच्छ शून्य संयोजन (गुणांक) है $1, -1, 1, -1$),
जो कुछ भी $(x_1, x_2, x_3, x_4)$ है। चक्र की लंबाई की समता
निर्णय करता है.
\end{solution}

\begin{solution}{exo:b1:vspaces:12}
\begin{enumerate}
  \item अगर $ F_1 \subseteq F_2$ या $ F_2 \subseteq F_1$, संघ
        दोनों में से एक है, इसलिए उचित है। अन्यथा चुनें $ x \in F_1
        \setminus F_2$ और $ y \in F_2 \setminus F_1$, और विचार करें
        $ x + y $. अगर $ x + y \in F_1$, तब $ y = (x + y) - x \in
        F_1$: विरोधाभास। अगर $ x + y \in F_2$, तब $ x \in F_2$:
        विरोधाभास। इसलिए $ x + y \notin F_1 \cup F_2$, और $ E \neq
        F_1 \cup F_2$.
  \item विरोधाभास के लिए मान लीजिए कि $ E = F_1 \cup \dots \cup
        F_k $, साथ $ k $ चुना गया \emph{न्यूनतम} ऐसे सबके बीच
        आवरण. न्यूनतमता वर्जित करती है $ F_1 \subseteq F_2 \cup
        \dots \cup F_k $ (अन्यथा छोड़ें $ F_1$), इसलिय वहाँ है $ x \in F_1$
        साथ $ x \notin F_i $ सभी के लिए $ i \geq 2$. तब से $ F_1$ है
        उचित, उठाओ $ y \notin F_1$. प्रत्येक अदिश के लिए $ t $, द
        वेक्टर $ y + t x $ कुछ में झूठ है $ F_i $. यह कभी भी झूठ नहीं बोलता
        $ F_1$: अन्यथा $ y = (y + tx) - tx \in F_1$ (जैसा $ x \in
        F_1$). \omterm{def:b1:structures:field}{मैदान} अनंत है, इसलिए चुनें $ k $ विशिष्ट
        अदिश $ t_1, \dots, t_k $: द $ k $ वैक्टर $ y + t_j x $
        में गिरना $ k - 1$ उपस्थान $ F_2, \dots, F_k $, और दो
        उनमें से, कहते हैं $ y + t x $ और $ y + t' x $ साथ $ t \neq t'$,
        उसी में पड़े रहो $ F_i $ ($ i \geq 2$). फिर उनका अंतर
        $(t - t')x \in F_i $, इसलिए $ x \in F_i $: विरोधाभास। इसलिए
        उचित द्वारा कोई सीमित आवरण नहीं उपस्थान मौजूद है.
\end{enumerate}
\end{solution}

\begin{solution}{pb:b1:vspaces:1}
\textbf{1.} $ L_i $ का एक उत्पाद है $ n $ रैखिक कारकों को a से विभाजित किया गया है
शून्येतर स्थिरांक ( $ x_i $ विशिष्ट हैं), इसलिए $\deg L_i = n $.
पर मूल्यांकन कर रहे हैं $ x_j $ साथ $ j \neq i $: कारक $ X - x_j $ का
अंश-गणक गायब हो जाता है, इसलिए $ L_i(x_j) = 0$. पर $ x_i $, अंश और
हर संपाती: $ L_i(x_i) = 1$.

\textbf{2.} कल्पना करना $\sum_i \lambda_i L_i = 0$. पर मूल्यांकन करें $ x_j $:
सिवाय सभी पद मर जाते हैं $\lambda_j L_j(x_j) = \lambda_j $, इसलिए
$\lambda_j = 0$ हरएक के लिए $ j $: परिवार है \omterm{def:b1:vspaces:free}{मुक्त}.

\textbf{3.} होने देना $ D = P - \sum_i P(x_i) L_i $. तब $\deg D \leq n $
और, प्रश्न 1 से, $ D(x_j) = P(x_j) - P(x_j) = 0$ के लिए $ n + 1$
विशिष्ट बिंदु $ x_0, \dots, x_n $. एक शून्येतर \omterm{def:b1:poly:def}{बहुपद} डिग्री का
$\leq n $ अधिकतम है $ n $ जड़ें (\cref{cor:b1:poly:nroots}), इसलिए $ D
= 0$. इस प्रकार प्रत्येक $ P \in \R_n[X]$ का एक संयोजन है $ L_i $:
परिवार पैदा कर रहा है, और प्रश्न 2 के साथ, ए \omterm{def:b1:vspaces:free}{आधार}.

\textbf{4.} दिया गया $ y_0, \dots, y_n $, द \omterm{def:b1:poly:def}{बहुपद} $ P = \sum_i
y_i L_i $ की डिग्री है $\leq n $ और अंतर्विष्ट करता है. विशिष्टता: एक
प्रक्षेप $ P $ प्रश्न 3 से, \omterm{prop:b1:vspaces:coordinates}{COORDINATES} $(P(x_0), \dots,
P(x_n)) = (y_0, \dots, y_n)$ में \omterm{def:b1:vspaces:free}{आधार} $(L_i)$, और \omterm{prop:b1:vspaces:coordinates}{COORDINATES}
में एक \omterm{def:b1:vspaces:free}{आधार} विशिष्ट हैं (\cref{prop:b1:vspaces:coordinates}).
\omterm{prop:b1:vspaces:coordinates}{COORDINATES} का $ P $ लैग्रेंज में \omterm{def:b1:vspaces:free}{आधार} इसके हैं \emph{मूल्यों पर
नोड्स} ---यह है पूरा मुद्दा \omterm{def:b1:vspaces:free}{आधार}.

\textbf{5.} प्रश्न 3 को लागू करें $ P = X^k $ ($0 \leq k \leq n $):
\[
X^k = \sum_{i=0}^{n} x_i^{k} L_i ,
\]
और $ k = 0$ देता है $\sum_i L_i = 1$.

\textbf{6.} $\deg N_k = k $ बिल्कुल: परिवार $(N_0, \dots, N_n)$
में डिग्रियों की एक सीढ़ी है $\R_n[X]$, इसलिए ए \omterm{def:b1:vspaces:free}{आधार} द्वारा
\cref{ex:b1:vspaces:staircase} (से \omterm{def:b1:vspaces:free}{मुक्त}ि
\cref{prop:b1:vspaces:freecriteria}~(1), पीढ़ी दर पीढ़ी
डिग्री पर उतरना)।

\textbf{7.} पुनरावृत्ति से,
\[
f[x_0, x_1] = \frac{f(x_1) - f(x_0)}{x_1 - x_0},
\qquad
f[x_0, x_1, x_2] = \frac{f[x_1, x_2] - f[x_0, x_1]}{x_2 - x_0}.
\]
के लिए $ f(x) = x^2$:
\[
f[x_0, x_1] = \frac{x_1^2 - x_0^2}{x_1 - x_0} = x_0 + x_1,
\]
और तब
\[
f[x_0, x_1, x_2]
= \frac{(x_1 + x_2) - (x_0 + x_1)}{x_2 - x_0}
= \frac{x_2 - x_0}{x_2 - x_0} = 1 .
\]

\textbf{8.} $\deg S \leq n $ तब से $ Q, R $ डिग्री है $\leq n - 1$.
पर $ x_0$: $ S(x_0) = \frac{-(x_0 - x_n) R(x_0)}{x_n - x_0} =
R(x_0) = f(x_0)$. पर $ x_n $: $ S(x_n) = \frac{(x_n - x_0)
Q(x_n)}{x_n - x_0} = Q(x_n) = f(x_n)$. एक भूरा \omterm{def:b1:topology:closure}{आंतरिक भाग} नोड $ x_i $
($1 \leq i \leq n-1$), दोनों $ Q $ और $ R $ मूल्य ले लो $ f(x_i)$,
इसलिए
\[
S(x_i) = \frac{(x_i - x_0) - (x_i - x_n)}{x_n - x_0}\, f(x_i)
= f(x_i) .
\]

\textbf{9.} नोड्स की संख्या पर प्रेरण. एक नोड:
इंटरपोलेंट स्थिरांक है $ f(x_0) = f[x_0]$. दावा मान लीजिए
के लिए $ k $ नोड्स और चलो $ S $ पर प्रक्षेप करें $ x_0, \dots, x_k $; द्वारा
विशिष्टता (प्रश्न 4), $ S $ ऐटकेन लेम्मा द्वारा दिया गया है $ R $
(नोड्स $ x_0, \dots, x_{k-1}$) और $ Q $ (नोड्स $ x_1, \dots, x_k $).
का गुणांक $ X^{k}$ में $ S $ है
\[
\frac{[X^{k-1}]\,Q - [X^{k-1}]\,R}{x_k - x_0}
= \frac{f[x_1, \dots, x_k] - f[x_0, \dots, x_{k-1}]}{x_k - x_0}
= f[x_0, \dots, x_k]
\]
प्रेरण परिकल्पना और परिभाषित पुनरावृत्ति द्वारा।

\textbf{10.} होने देना $ P_k $ बैठाना $ f $ पर $ x_0, \dots, x_k $. के अंतर $ P_k - P_{k-1}$ की डिग्री है $\leq k $ और गायब हो जाता है
$ x_0, \dots, x_{k-1}$, इसलिए कारक प्रमेय द्वारा लागू किया गया $ k $ बार
(\cref{thm:b1:poly:factor}) यह बराबर है $ c\,N_k $ एक स्थिरांक के लिए $ c $;
के गुणांकों की तुलना करना $ X^{k}$ और प्रश्न 9 का उपयोग करते हुए, $ c =
f[x_0, \dots, x_k]$. टेलीस्कोपिंग से $ P_0 = f(x_0) N_0$ देता है
न्यूटन का सूत्र. \omterm{def:b1:topology:closure}{बंद} प्रपत्र के लिए लिखें $ P_k = \sum_{i \leq
k} f(x_i) L_i $ (लैग्रेंज, नोड्स पर $ x_0, \dots, x_k $) और पढ़ें
का गुणांक $ X^{k}$: प्रत्येक $ L_i $ योगदान
$\frac{1}{\prod_{j \neq i}(x_i - x_j)}$, कहाँ से
\[
f[x_0, \dots, x_k] = \sum_{i=0}^{k}
\frac{f(x_i)}{\prod_{j \neq i,\, j \leq k}(x_i - x_j)} .
\]
दाहिना पक्ष किसी के भी अंतर्गत अपरिवर्तनीय है \omterm{def:b1:counting:objects}{परिवर्तन} नोड्स की, तो
विभाजित अंतर उनके क्रम पर निर्भर नहीं करता है।

\textbf{11.} अगर $ P = a X^m + (\text{lower degrees})$, द्विपद
प्रमेय देता है
\[
\Delta P = a\bigl((X+1)^m - X^m\bigr) + \dots
= a\,m\,X^{m-1} + (\text{lower degrees}),
\]
क्योंकि $(X+1)^m - X^m = m X^{m-1} + \dots $ और निम्न-डिग्री
का हिस्सा $ P $ डिग्री का योगदान देता है $\leq m - 2$ बाद $\Delta $ (या
डिग्री $\leq m-2$ शर्तें)। इसलिए $\deg \Delta P = m - 1$ अग्रणी के साथ
गुणांक $ m a $. एक स्थिरांक $ c $ देता है $\Delta c = c - c = 0$.

\textbf{12.} $\deg B_k = k $: सीढ़ी, इसलिए ए \omterm{def:b1:vspaces:free}{आधार} का
$\R_n[X]$ (\cref{ex:b1:vspaces:staircase}). के लिए $\Delta B_k $
($ k \geq 1$), सामान्य उत्पाद का कारक:
\begin{align*}
k!\,\Delta B_k
&= (X+1)X\cdots(X-k+2) - X(X-1)\cdots(X-k+1) \\
&= X(X-1)\cdots(X-k+2)\,\bigl[(X+1) - (X-k+1)\bigr] \\
&= k\,X(X-1)\cdots(X-k+2),
\end{align*}
इसलिए $\Delta B_k = \frac{X(X-1)\cdots(X-k+2)}{(k-1)!} = B_{k-1}$.

\textbf{13.} लिखना $ P = \sum_{k=0}^{n} c_k B_k $ (\omterm{def:b1:vspaces:free}{आधार}, प्रश्न
12). आवेदन करना $\Delta^{j}$: प्रश्न 12 द्वारा, $\Delta^{j} P = \sum_{k
\geq j} c_k B_{k-j}$. पर मूल्यांकन करें $0$: $ B_0(0) = 1$ और $ B_m(0) =
0$ के लिए $ m \geq 1$ (कारक $ X $ गायब हो जाता है), तो
$\bigl(\Delta^{j}P\bigr)(0) = c_j $. यही आगे का अंतर है
सूत्र.

\textbf{14.} प्रेरण चालू $ k $. के लिए $ k = 0$ पहचान पढ़ती है
$ P(0) = P(0)$. इसके लिए मान लीजिये $ k $ और इसे लागू करें $\Delta P $:
\[
\bigl(\Delta^{k+1} P\bigr)(0)
= \sum_{j=0}^{k} (-1)^{k-j}\binom kj \bigl(P(j+1) - P(j)\bigr).
\]
का गुणांक एकत्रित करें $ P(i)$: यह है $(-1)^{k-i+1}\binom
k{i-1}\cdot(-1)^{0}$ पहले योग से (स्थानांतरित) और
$-(-1)^{k-i}\binom ki $ दूसरे से --- एक साथ
\[
(-1)^{k+1-i}\Bigl(\binom k{i-1} + \binom ki\Bigr)
= (-1)^{k+1-i}\binom{k+1}i
\]
पास्कल के नियम से, जो रैंक पर पहचान है $ k + 1$.

\textbf{15.} डिग्री से प्रश्न 11 को दोहराते हुए $ n $, नेतृत्व गुणांक $ a_n $: एक के बाद एक $\Delta $, डिग्री $ n-1$ और अग्रणी
गुणांक $ n a_n $; दो के बाद, $ n(n-1)a_n $; बाद $ n $ कदम,
डिग्री $0$ और मूल्य $ n(n-1)\cdots 1\, a_n = n!\,a_n $, एक स्थिरांक.
और एक $\Delta $ इसे मारता है: $\Delta^{n+1}P = 0$.

\textbf{16.} अगर $ m \geq k $: $ B_k(m) = \binom mk \in \N $. अगर $0
\leq m < k $: का एक कारक $ m(m-1)\cdots(m-k+1)$ शून्य है, इसलिए
$ B_k(m) = 0$. अगर $ m = -q $ साथ $ q \geq 1$:
\[
B_k(-q) = \frac{(-q)(-q-1)\cdots(-q-k+1)}{k!}
= (-1)^k\,\frac{q(q+1)\cdots(q+k-1)}{k!}
= (-1)^k \binom{q+k-1}{k},
\]
पूर्णांक। तो हर $ B_k $ एमएपीएस $\Z $ में $\Z $.

\textbf{17.} ($\Leftarrow $) अगर $ P = \sum_k c_k B_k $ साथ $ c_k \in
\Z $, फिर के लिए $ m \in \Z $, $ P(m) = \sum_k c_k B_k(m) \in \Z $ द्वारा
प्रश्न 16.($\Rightarrow $) अगर $ P $ पूर्णांक-मूल्यवान है, इसका
\omterm{prop:b1:vspaces:coordinates}{COORDINATES} हैं $ c_k = \bigl(\Delta^k P\bigr)(0) = \sum_{j=0}^k
(-1)^{k-j}\binom kj P(j)$ (प्रश्न 13 और 14), एक पूर्णांक
पूर्णांकों का संयोजन $ P(0), \dots, P(k)$. ये है
पी\'ओलेआ का लक्षण वर्णन \omterm{pb:b1:vspaces:1}{पूर्णांक-मूल्यवान बहुपद}.

\textbf{18.} \omterm{def:b1:logic:sets}{तय करना} $ Q(X) = P(X + a)$, ए \omterm{def:b1:poly:def}{बहुपद} डिग्री का $\leq
n $ साथ $ Q(0), Q(1), \dots, Q(n) \in \Z $. इसका \omterm{prop:b1:vspaces:coordinates}{COORDINATES} में
$(B_k)_{k \leq n}$ हैं $ c_k = \sum_{j \leq k}(-1)^{k-j}\binom kj
Q(j) \in \Z $ (प्रश्न 14 केवल मानों का उपयोग करता है $0, \dots, k
\leq n $). प्रश्न 17 द्वारा ($\Leftarrow $), $ Q $ पूर्णांक-मान पर है
सभी $\Z $, इसलिए ऐसा है $ P(X) = Q(X - a)$.

\textbf{19.} का एक उत्पाद $ k $ क्रमागत पूर्णांक है $ m(m-1)
\cdots(m-k+1) = k!\,B_k(m)$ कुछ के लिए $ m \in \Z $, और $ B_k(m) \in
\Z $ प्रश्न 16 से: उत्पाद विभाज्य है $ k!$.

\textbf{20.} के मान $ P = \frac{X(X+1)(2X+1)}{6}$ पर $0,1,2,3$:
$0, 1, 5, 14$. अंतर तालिका: $\Delta $ पंक्ति $1, 4, 9$;
$\Delta^2$ पंक्ति $3, 5$; $\Delta^3$ पंक्ति $2$. अत: प्रश्न 13 से,
\[
P = 0\cdot B_0 + 1\cdot B_1 + 3\,B_2 + 2\,B_3 ,
\]
पूर्णांक के साथ \omterm{prop:b1:vspaces:coordinates}{COORDINATES}: $ P $ पूर्णांक-मान है (प्रश्न 17),
जबकि इसके एकपदी गुणांक $\frac13, \frac12, \frac16$ हैं
पूर्णांक नहीं. \omterm{def:b1:vspaces:sum}{प्रत्यक्ष} गणना:
\begin{align*}
\Delta P &= \frac{(X+1)(X+2)(2X+3) - X(X+1)(2X+1)}{6} \\
&= \frac{(X+1)\bigl[(X+2)(2X+3) - X(2X+1)\bigr]}{6}
= \frac{(X+1)(6X+6)}{6} = (X+1)^2 .
\end{align*}
अंतःसर्पण $ P(m) = \sum_{j=0}^{m-1}\Delta P(j) = \sum_{j=1}^{m}
j^2$ (साथ $ P(0) = 0$): वर्गों का योग सूत्र.

\textbf{21.} मूल्य $2^i $ पर $ i = 0, \dots, n $ अंतर है
टेबल लगातार $1$ बाएँ किनारे पर: $\Delta^k $ अनुक्रम का
$(2^i)$ है $(2^i)$ फिर से (तब से) $2^{i+1} - 2^i = 2^i $), इसलिए
$\bigl(\Delta^k P\bigr)(0) = 2^0 = 1$ सभी के लिए $ k \leq n $, और $ P =
B_0 + B_1 + \dots + B_n $ प्रश्न 13 द्वारा. फिर
\[
P(n+1) = \sum_{k=0}^{n}\binom{n+1}{k}
= 2^{n+1} - \binom{n+1}{n+1} = 2^{n+1} - 1 \neq 2^{n+1}:
\]
पैटर्न पहले अनियंत्रित बिंदु पर टूट जाता है।

\textbf{22.} प्रश्न 12 द्वारा, $ B_k = \Delta B_{k+1}$, इसलिए
\[
\sum_{j=0}^{m-1} B_k(j)
= \sum_{j=0}^{m-1}\bigl(B_{k+1}(j+1) - B_{k+1}(j)\bigr)
= B_{k+1}(m) - B_{k+1}(0) = B_{k+1}(m).
\]
के लिए $ j < k $ शर्तें $ B_k(j)$ गायब हो जाता है, इसलिए योग वास्तव में शुरू होता है
$ j = k $: $\sum_{j=k}^{m-1}\binom jk = \binom m{k+1}$, द
हॉकी-स्टिक की पहचान.

\textbf{23.} अंतर तालिकाएँ (या \omterm{def:b1:vspaces:sum}{प्रत्यक्ष} विस्तार) देती हैं
\[
X^2 = B_1 + 2 B_2, \qquad X^3 = B_1 + 6 B_2 + 6 B_3
\]
(जाँच करना: $ B_1 + 2B_2 = X + X(X-1) = X^2$; पर $ X = 1, 2, 3$ द
दूसरा देता है $1, 8, 27$). प्रश्न 22 फिर उपज
\[
\sum_{j=0}^{m-1} j^2 = B_2(m) + 2B_3(m)
= \binom m2 + 2\binom m3 = \frac{m(m-1)(2m-1)}{6},
\]
\[
\sum_{j=0}^{m-1} j^3 = B_2(m) + 6B_3(m) + 6B_4(m)
= \binom m2 + 6\binom m3 + 6\binom m4 .
\]
अंतिम अभिव्यक्ति का विस्तार: $\binom m2 + 6\binom m3 + 6\binom
m4 = \frac{m(m-1)}{2}\bigl[1 + 2(m-2) +
\frac{(m-2)(m-3)}{2}\bigr] = \frac{m^2(m-1)^2}{4} = \binom m2^2$.
की जगह $ m $ द्वारा $ m + 1$: $1^3 + \dots + m^3 =
\bigl(\frac{m(m+1)}2\bigr)^2 = (1 + \dots + m)^2$, निकोमाचस'
पहचान.

\textbf{24.} नोड्स $0, 1, 2$, \omterm{def:b1:poly:def}{बहुपद} $ X^2$. एकपदीय \omterm{def:b1:vspaces:free}{आधार}
$(1, X, X^2)$: \omterm{prop:b1:vspaces:coordinates}{COORDINATES} $(0, 0, 1)$. लैग्रेंज \omterm{def:b1:vspaces:free}{आधार}:
\omterm{prop:b1:vspaces:coordinates}{COORDINATES} मूल्य हैं $(0, 1, 4)$ (प्रश्न 4). न्यूटन \omterm{def:b1:vspaces:free}{आधार}
$(1, X, X(X-1))$: \omterm{pb:b1:vspaces:1}{विभाजित मतभेद} $ f[0] = 0$, $ f[0,1] = 1$,
$ f[0,1,2] = \frac{3 - 1}{2} = 1$ (प्रश्न 9), अतः \omterm{prop:b1:vspaces:coordinates}{COORDINATES}
$(0, 1, 1)$ --- वास्तव में $ X + X(X-1) = X^2$. तीन \omterm{def:b1:vspaces:free}{आधार}, तीन
समन्वय सदिश, एक \omterm{def:b1:poly:def}{बहुपद}.

\textbf{25.} (i) प्रश्न 4 स्वचालित है क्योंकि $(L_i)$ एक है
\emph{\omterm{def:b1:vspaces:free}{आधार}}: प्रक्षेप का अस्तित्व एवं विशिष्टता है
बिल्कुल अस्तित्व और विशिष्टता \omterm{prop:b1:vspaces:coordinates}{COORDINATES}. (ii) न्यूटन
\omterm{def:b1:vspaces:free}{आधार} एक सीढ़ी है, इसलिए \omterm{prop:b1:vspaces:coordinates}{COORDINATES} क्रमिक रूप से गणना की जाती है
विभाजन --- प्रत्येक नया नोड बिना छेड़छाड़ किए एक शब्द जोड़ता है
पिछले वाले --- जबकि लैग्रेंज \omterm{prop:b1:vspaces:coordinates}{COORDINATES} का $ P $ हैं
मूल्य $ P(x_i)$, बिना किसी गणना के उपलब्ध है। (iii) दोनों
\omterm{def:b1:vspaces:free}{आधार} हैं \omterm{def:b1:vspaces:free}{मुक्त} के समान दो मानदंडों द्वारा
\cref{prop:b1:vspaces:freecriteria}: न्यूटन के लिए विशिष्ट डिग्री,
लैग्रेंज के लिए नोड्स पर मूल्यांकन। (iv) पी\'ओलेआ का प्रमेय कहता है
वह ``$ P(\Z) \subseteq \Z $'', मूल्यों की एक संपत्ति, समतुल्य है
की अभिन्नता के लिए \omterm{prop:b1:vspaces:coordinates}{COORDINATES} में \omterm{def:b1:vspaces:free}{आधार} $(B_k)$ --- द
ए का अंकगणित \omterm{def:b1:poly:def}{बहुपद} में ही दृश्यमान हो जाता है \omterm{def:b1:vspaces:free}{आधार}
प्रश्न के अनुरूप।
