\begin{solution}{exo:b1:findim:3}
जोड़ने का प्रयास करें $ e_1 = (1,0,0,0)$ और $ e_3 = (0,0,1,0)$. परिवार
$\bigl((1,1,0,0), (0,0,1,1), e_1, e_3\bigr)$ है \omterm{def:b1:vspaces:free}{मुक्त}: एक शून्य
संयोजन $\alpha(1,1,0,0) + \beta(0,0,1,1) + \gamma e_1 + \delta
e_3 = 0$ पढ़ता $(\alpha + \gamma, \alpha, \beta + \delta, \beta) =
0$, इसलिए $\alpha = \beta = 0$, तब $\gamma = \delta = 0$. चार \omterm{def:b1:vspaces:free}{मुक्त}
आयाम में वेक्टर $4$: ए \omterm{def:b1:vspaces:free}{आधार}
(\cref{prop:b1:findim:twoofthree}).
\end{solution}

\begin{solution}{exo:b1:findim:4}
डिग्री $0, 1, 2, 3$ जोड़ीवार अलग: \omterm{def:b1:vspaces:free}{मुक्त}
(\cref{prop:b1:vspaces:freecriteria}), आयाम में चार वेक्टर
$4$: \omterm{def:b1:vspaces:free}{आधार}. के लिए $ X^3$: नीचे की ओर विस्तार करें,
\[
X(X-1)(X-2) = X^3 - 3X^2 + 2X,
\qquad
X(X-1) = X^2 - X,
\]
इसलिए $ X^3 - X(X-1)(X-2) = 3X^2 - 2X $; और $3X^2 - 2X = 3(X^2 - X) + X
= 3\,X(X-1) + X $. इस तरह
\[
X^3 = X(X-1)(X-2) + 3\,X(X-1) + 1\cdot X + 0\cdot 1 :
\]
\omterm{prop:b1:vspaces:coordinates}{COORDINATES} $(0, 1, 3, 1)$ पर $\bigl(1, X, X(X-1),
X(X-1)(X-2)\bigr)$. (ये गुप्त रूप से स्टर्लिंग नंबर हैं।)
\end{solution}

\begin{solution}{exo:b1:findim:5}
ग्रासमैन: $\dim(F \cap G) = 4 + 5 - \dim(F + G)$, और $ F + G $ एक है
\omterm{def:b1:vspaces:subspace}{उपस्पेस} का $ E $ युक्त $ G $: $5 \leq \dim(F+G) \leq 7$. इस तरह
$\dim(F \cap G) \in \{2, 3, 4\}$. में अहसास $\R^7$ के साथ
विहित \omterm{def:b1:vspaces:free}{आधार} $(e_1, \dots, e_7)$, ले रहा हूँ $ G =
\operatorname{Vect}(e_1, \dots, e_5)$:
\begin{itemize}
  \item $ F = \operatorname{Vect}(e_1, e_2, e_6, e_7)$: $ F + G = \R^7$,
        चौराहा $\operatorname{Vect}(e_1, e_2)$, आयाम $2$;
  \item $ F = \operatorname{Vect}(e_1, e_2, e_3, e_6)$: चौराहा
        आयाम का $3$;
  \item $ F = \operatorname{Vect}(e_1, e_2, e_3, e_4) \subseteq G $:
        आयाम $4$.
\end{itemize}
\end{solution}

\begin{solution}{exo:b1:findim:6}
$ H $ एक है \omterm{def:b1:vspaces:subspace}{उपस्पेस} (\cref{exo:b1:vspaces:1}~(4) शब्दशः). द्वारा
कारक प्रमेय (\cref{thm:b1:poly:factor}), $ P \in H \iff P =
(X-1)Q $ साथ $\deg Q \leq n - 1$: द \omterm{def:b1:logic:map}{नक्शा} $ Q \mapsto (X-1)Q $ एक है
से रैखिक आक्षेप $\R_{n-1}[X]$ पर $ H $, तो ए \omterm{def:b1:vspaces:free}{आधार} का $ H $ है
\[
\bigl((X-1),\ (X-1)X,\ (X-1)X^2,\ \dots,\ (X-1)X^{n-1}\bigr),
\qquad \dim H = n .
\]
ए \omterm{def:b1:vspaces:sum}{पूरक} रेखा: $\operatorname{Vect}(1)$ (स्थिरांक)। वास्तव में
$ H \cap \operatorname{Vect}(1) = \{0\}$ (एक गैरशून्य स्थिरांक नहीं होता है
पर गायब हो जाओ $1$) और आयाम जुड़ते हैं $ n + 1$: ग्रासमैन द्वारा, $ H
\oplus \operatorname{Vect}(1) = \R_n[X]$.
\end{solution}

\end{solution}

\begin{solution}{exo:b1:findim:8}
सख्ती से बढ़ती श्रृंखला के साथ, आयाम सख्ती से बढ़ते हैं
($ F_i \subseteq F_{i+1}$, $ F_i \neq F_{i+1}$ और
\cref{thm:b1:findim:subspaces}: आयामों की समानता बल देगी
रिक्त स्थान की समानता)। इसलिए $\dim F_1 < \dim F_2 < \dots < \dim F_k $ है
पूर्णांकों का सख्ती से बढ़ता क्रम $\intint{0}{n}$: अधिक से अधिक $ n + 1$ मूल्य, $ k \leq n + 1$. अधिकतम श्रृंखला में $\R^n $:
\[
\{0\} \subsetneq \operatorname{Vect}(e_1) \subsetneq
\operatorname{Vect}(e_1, e_2) \subsetneq \dots \subsetneq \R^n ,
\]
बिलकुल लंबाई का $ n + 1$.
\end{solution}

\begin{solution}{exo:b1:findim:9}
$ F + G $ सख्ती से शामिल है $ F $ (तब से $ G \not\subseteq F $), इसलिए
$\dim(F + G) = n $ और ग्रासमैन देता है $\dim(F \cap G) = (n-1) +
(n-1) - n = n - 2$.

सामान्य दावा, प्रेरण द्वारा $ k $: चौराहा $ I_k $ का $ k $
हाइपरप्लेन के पास है $\dim I_k \geq n - k $. के लिए सच है $ k = 1$. कदम: $ I_{k+1}
= I_k \cap H_{k+1}$, और ग्रासमैन अंदर $ E $:
\[
\dim(I_k \cap H_{k+1}) = \dim I_k + (n - 1) - \dim(I_k + H_{k+1})
\geq \dim I_k + (n-1) - n \geq n - k - 1 . \qedhere
\]
\end{solution}

\begin{solution}{exo:b1:findim:10}
\begin{enumerate}
  \item स्थिति $ u_{n+2} - 3u_{n+1} + 2u_n = 0$ रैखिक है
        और शून्य अनुक्रम से संतुष्ट: $ E $ एक है \omterm{def:b1:vspaces:subspace}{उपस्पेस}. द्वारा
        प्रेरण, $ u_0$ और $ u_1$ प्रत्येक का निर्धारण करें $ u_n $, और
        निर्भरता रैखिक है (प्रत्येक चरण एक रैखिक संयोजन है
        पिछले दो मान); इसके विपरीत प्रत्येक जोड़ी $(a, b)$
        के ठीक एक क्रम से उत्पन्न होता है $ E $ (परिभाषित करना $ u_n $ द्वारा
        पुनरावृत्ति)। के रूप में \cref{ex:b1:findim:dims}, $ E $ है
        पैरामीट्रिज्ड वस्तुनिष्ठ रूप से और रैखिक रूप से $(u_0, u_1) \in
        \R^2$: $\dim E = 2$.
  \item स्थिरांक: $3\cdot1 - 2\cdot1 = 1$. ज्यामितीय: $3\cdot
        2^{n+1} - 2\cdot 2^n = (6 - 2)2^n = 2^{n+2}$. दोनों पड़े हैं
        $ E $. उन्मुक्तता: $ a\cdot 1 + b\cdot 2^n = 0$ सभी के लिए $ n $
        देता है, पर $ n = 0$ और $ n = 1$: $ a + b = 0$, $ a + 2b = 0$,
        इसलिए $ a = b = 0$. दो \omterm{def:b1:vspaces:free}{मुक्त} आयाम में वेक्टर $2$: ए \omterm{def:b1:vspaces:free}{आधार}
        (\cref{prop:b1:findim:twoofthree}).
  \item हल करना $ a + b = 0$, $ a + 2b = 1$: $ b = 1$, $ a = -1$, इसलिए
        $ u_n = 2^n - 1$ (मेरसेन अनुक्रम)।
\end{enumerate}
\end{solution}

\begin{solution}{exo:b1:findim:11}
\begin{enumerate}
  \item ग्रासमैन: $\dim(F \cap G) = \dim F + \dim G - \dim(F +

\begin{solution}{exo:b1:findim:12}
नीचे की ओर प्रेरण $ d = \dim F = \dim G $, से $ d = n $ को $ d =
0$. अगर $ d = n $: $ F = G = E $ और $ S = \{0\}$ काम करता है. मान लीजिये
\omterm{def:b1:logic:statement}{कथन} के लिए रखता है \omterm{def:b1:vspaces:subspace}{उपस्पेस} आयाम के जोड़े $ d + 1 \leq n $,
और चलो $\dim F = \dim G = d < n $.

अगर $ F = G $: के लिए ले $ S $ कोई \omterm{def:b1:vspaces:sum}{पूरक} का $ F $
(\cref{thm:b1:findim:subspaces}). अगर $ F \neq G $: दोनों उचित हैं,
तो द्वारा \cref{exo:b1:vspaces:12} वहाँ मौजूद है $ x \notin F \cup G $.
रकम $ F' = F \oplus \operatorname{Vect}(x)$ और $ G' = G \oplus
\operatorname{Vect}(x)$ \omterm{def:b1:vspaces:sum}{प्रत्यक्ष} हैं ($ x \notin F $, $ x \notin G $)
और आयाम है $ d + 1$; प्रेरण परिकल्पना द्वारा वे
एक आम स्वीकार करें \omterm{def:b1:vspaces:sum}{पूरक} $ S'$: $ E = F' \oplus S' = G' \oplus
S'$. \omterm{def:b1:logic:sets}{तय करना} $ S = \operatorname{Vect}(x) \oplus S'$ --- \omterm{def:b1:vspaces:sum}{प्रत्यक्ष},
क्योंकि $ S' \cap \operatorname{Vect}(x) \subseteq S' \cap F' =
\{0\}$.

तब $ F + S = F + \operatorname{Vect}(x) + S' = F' + S' = E $, और
$ F \cap S = \{0\}$: अगर $ f = \lambda x + s'$ साथ $ f \in F $, $ s'
\in S'$, तब $ s' = f - \lambda x \in F' \cap S' = \{0\}$, इसलिए $ f
= \lambda x $, मजबूर करना $\lambda = 0$ ($ x \notin F $) और $ f = 0$.
इस तरह $ E = F \oplus S $, और सममित रूप से $ E = G \oplus S $.
\end{solution}

\begin{solution}{pb:b1:findim:1}
\textbf{1.} $\Q $ रोकना $0 \neq 1$, जोड़ के तहत स्थिर है,
गुणन और विपरीत, और प्रत्येक गैर-शून्य तर्कसंगत में एक होता है
तर्कसंगत उलटा: ए \omterm{def:b1:structures:field}{मैदान} (\cref{def:b1:structures:field}). द
की परिभाषाएँ \omterm{def:b1:vspaces:span}{अवधि}, उन्मुक्तता, \omterm{def:b1:vspaces:free}{आधार} और अध्यायों के प्रमाण
18--19 केवल वेक्टर जोड़, वितरण और अदिश का उपयोग करें
ए में अंकगणित \omterm{def:b1:structures:field}{मैदान} --- कभी भी पूर्ण मूल्य, ऑर्डर या ए नहीं
\omterm{def:b1:topology:closure}{सीमा}. इसलिए $\R $, अपने स्वयं के जोड़ और गुणा के साथ
$\Q \times \R \to \R $, एक है $\Q $-वेक्टर स्पेस, और सभी
प्रमेय लागू होते हैं. किसी अदिश द्वारा विभाजन प्रमाण में एक बार होता है
\cref{thm:b1:findim:exchange}: ``के लिए हल करें $ g_k $'' एक विभाजित
इसके शून्येतर गुणांक द्वारा.

\textbf{2.} अगर $ a + b\sqrt2 = 0$ साथ $ a, b \in \Q $ दोनों नहीं
शून्य: $ b \neq 0$ देगा $\sqrt2 = -a/b \in \Q $, विरोधाभासी
की अतार्किकता $\sqrt2$ (\cref{exo:b1:arith:7} साथ $ p =
2$); इसलिए $ b = 0$, तब $ a = 0$: \omterm{def:b1:vspaces:free}{मुक्त} ऊपर $\Q $. ऊपर $\R $ द
संबंध $\sqrt2\cdot 1 + (-1)\cdot\sqrt2 = 0$ गैरतुच्छ है:
जुड़ा हुआ. इस तरह $\dim_\Q \Q(\sqrt2) = 2$, अद्वितीय के साथ \omterm{prop:b1:vspaces:coordinates}{COORDINATES}
$(a, b)$.

\textbf{3.} $(a + b\sqrt2)(c + d\sqrt2) = (ac + 2bd) + (ad +
bc)\sqrt2 \in \Q(\sqrt2)$. तब
\[
\sigma(uv) = (ac + 2bd) - (ad + bc)\sqrt2
= (a - b\sqrt2)(c - d\sqrt2) = \sigma(u)\sigma(v),
\]
इसलिए $ N(uv) = uv\,\sigma(uv) = u\sigma(u)\,v\sigma(v) = N(u)N(v)$.
अगर $ N(u) = a^2 - 2b^2 = 0$ साथ $ u \neq 0$: $ b \neq 0$ देगा
$(a/b)^2 = 2$, का एक तर्कसंगत वर्गमूल $2$; इसलिए $ b = 0$, तब $ a
= 0$ और $ u = 0$: विरोधाभास। इस तरह $ N(u) \neq 0$ के लिए $ u \neq
0$.

\textbf{4.} $ u \cdot \dfrac{\sigma(u)}{N(u)} = \dfrac{N(u)}{N(u)}
= 1$, और $\sigma(u)/N(u) \in \Q(\sqrt2)$: प्रत्येक अशून्य तत्व
अंदर उलटा है $\Q(\sqrt2)$, जो इसलिए एक उपक्षेत्र है
का $\R $. उदाहरण: $ N(3 + 2\sqrt2) = 9 - 8 = 1$, इसलिए
\[
\frac1{3 + 2\sqrt2} = 3 - 2\sqrt2 ;
\qquad
N(1 + \sqrt2) = -1, \quad
\frac1{1 + \sqrt2} = \sqrt2 - 1 .
\]

\textbf{5.} अगर $\sqrt6 = p/q $ तब $6q^2 = p^2$; का प्रतिपादक
$2$ है $1 + 2v_2(q)$, विषम, बाईं ओर, और $2v_2(p)$, यहां तक कि, पर
सही: असंभव. अब मान लीजिए $\sqrt3 = a + b\sqrt2$ साथ
$ a, b \in \Q $. वर्ग बनाना: $3 = a^2 + 2b^2 + 2ab\sqrt2$. अगर $ ab
\neq 0$, तब $\sqrt2 = (3 - a^2 - 2b^2)/(2ab) \in \Q $:
असंभव। अगर $ b = 0$: $\sqrt3 = a \in \Q $, विरोधाभासी
\cref{exo:b1:arith:7} ($ p = 3$). अगर $ a = 0$: $\sqrt3 = b\sqrt2$,
और से गुणा करना $\sqrt2$: $\sqrt6 = 2b \in \Q $: असंभव।
इसलिए $\sqrt3 \notin \Q(\sqrt2)$.

\textbf{6.} के उत्पाद \omterm{def:b1:vspaces:free}{आधार} वेक्टर हैं
\[
\sqrt2\,\sqrt3 = \sqrt6,\quad
\sqrt2\,\sqrt6 = 2\sqrt3,\quad
\sqrt3\,\sqrt6 = 3\sqrt2,\quad
(\sqrt2)^2 = 2,\ (\sqrt3)^2 = 3,\ (\sqrt6)^2 = 6,
\]
सभी में $ M $. के दो तत्वों का एक उत्पाद $ M $ द्वारा फैलता है
इनके तर्कसंगत संयोजनों में द्विरेखीयता: $ M $ स्थिर है
गुणन के अंतर्गत.

\textbf{7.} होने देना $ x + y\sqrt3 = 0$ साथ $ x, y \in K = \Q(\sqrt2)$.
अगर $ y \neq 0$, तब $\sqrt3 = -x/y \in K $ (प्रश्न 4: $ K $ एक है
\omterm{def:b1:structures:field}{मैदान}), विरोधाभासी प्रश्न 5. तो $ y = 0$, तब $ x = 0$: $(1,
\sqrt3)$ है \omterm{def:b1:vspaces:free}{मुक्त} ऊपर $ K $. यह उत्पन्न करता है: $ K + K\sqrt3 =
\{(a + b\sqrt2) + (c + d\sqrt2)\sqrt3\} = \operatorname{Vect}_\Q
(1, \sqrt2, \sqrt3, \sqrt6) = M $. इस तरह $\dim_K M = 2$.

\textbf{8.} एक रिश्ता $ a + b\sqrt2 + c\sqrt3 + d\sqrt6 = 0$
के रूप में पुनः \omterm{def:b1:structures:group}{समूह}ित होता है $(a + b\sqrt2) + (c + d\sqrt2)\sqrt3 = 0$ के साथ
में गुणांक $ K $; प्रश्न 7 से दोनों गायब हो जाते हैं, और प्रश्न 2 से
$ a = b = 0$ और $ c = d = 0$. तो परिवार है \omterm{def:b1:vspaces:free}{मुक्त} और $\dim_\Q M
= 4$.

\textbf{9.} प्रत्येक $ x \in M $ लिखते हैं $ x = \sum_j y_j f_j $ साथ $ y_j
\in K $ (\omterm{def:b1:vspaces:free}{आधार} का $ M $ ऊपर $ K $), और प्रत्येक $ y_j = \sum_i
\lambda_{ij} e_i $ साथ $\lambda_{ij} \in \Q $ (\omterm{def:b1:vspaces:free}{आधार} का $ K $ ऊपर
$\Q $); प्रतिस्थापित करना, $ x = \sum_{i,j} \lambda_{ij}\, e_i f_j $: द
उत्पाद उत्पन्न करते हैं $ M $ ऊपर $\Q $.

\textbf{10.} कल्पना करना $\sum_{i,j} \lambda_{ij}\, e_i f_j = 0$ साथ
$\lambda_{ij} \in \Q $. पुनः\omterm{def:b1:structures:group}{समूह}ित करें: $\sum_j \bigl(\sum_i \lambda_{ij}
e_i\bigr) f_j = 0$, और आंतरिक रकम का संबंध है $ K $. की उन्मुक्तता
$(f_j)$ ऊपर $ K $ देता है $\sum_i \lambda_{ij} e_i = 0$ प्रत्येक के लिए
$ j $; की आज़ादी $(e_i)$ ऊपर $\Q $ फिर देता है $\lambda_{ij} = 0$
सभी के लिए $ i, j $.

\textbf{11.} प्रश्न 9 एवं 10 द्वारा, $(e_i f_j)_{i \leq m,\, j
\leq n}$ एक है \omterm{def:b1:vspaces:free}{आधार} का $ M $ ऊपर $\Q $ साथ $ mn $ तत्व:
\[
\dim_\Q M = m\,n = \dim_\Q K \cdot \dim_K M .
\]
जाँच करना: $\dim_\Q K = 2$ और $\dim_K M = 2$ (प्रश्न 2 और 7)
देना $\dim_\Q M = 4$, जो कि प्रश्न 8 है।

\textbf{12.} वैक्टर $ u a_1, \dots, u a_d $ के संबंधित $ A $
(स्थिरता). वे हैं \omterm{def:b1:vspaces:free}{मुक्त}: अगर $\sum_i \lambda_i\, u a_i = 0$,
तब $ u \sum_i \lambda_i a_i = 0$ में $\R $, और $ u \neq 0$ बल
$\sum_i \lambda_i a_i = 0$, इस तरह $\lambda_i = 0$ (द $ a_i $ प्रपत्र
ए \omterm{def:b1:vspaces:free}{आधार}). इसलिए $(u a_1, \dots, u a_d)$ एक है \omterm{def:b1:vspaces:free}{मुक्त} परिवार का $ d $
वेक्टर में $ A $, $\dim_\Q A = d $: ए \omterm{def:b1:vspaces:free}{आधार}
(\cref{prop:b1:findim:twoofthree}). विशेष रूप से $1 \in A $
के रूप में विघटित होता है $1 = \sum_i \mu_i\, u a_i = u\,v $ साथ $ v = \sum_i
\mu_i a_i \in A $: का उलटा $ u $ में निहित है $ A $. इससे ठीक हो जाता है
प्रश्न 4 ($ A = \Q(\sqrt2)$) और एक झटके में यह साबित कर देता है $ M $
का एक उपक्षेत्र है $\R $ (प्रश्न 6 के साथ)।

\textbf{13.}  $ d + 1$ वैक्टर $1, x, x^2, \dots, x^{d}$ सब
लेट जाओ $ A $ (उत्पादों के अंतर्गत स्थिरता); ए \omterm{def:b1:vspaces:free}{मुक्त} परिवार का $ A $ है
अधिक से अधिक $ d $ वेक्टर (\cref{prop:b1:findim:twoofthree}), तो वे
जुड़े हुए हैं: तर्कसंगत हैं $\lambda_0, \dots, \lambda_d $,
सभी शून्य नहीं, साथ $\sum_k \lambda_k x^k = 0$. \omterm{def:b1:poly:def}{बहुपद} $ P
= \sum_k \lambda_k X^k $ शून्येतर है, तर्कसंगत गुणांक है,
डिग्री $\leq d $, और $ P(x) = 0$.

\textbf{14.} $ s^2 = 2 + 2\sqrt6 + 3 = 5 + 2\sqrt6$: \omterm{prop:b1:vspaces:coordinates}{COORDINATES}
$(5, 0, 0, 2)$. तब
\[
s^3 = s\,s^2 = (\sqrt2 + \sqrt3)(5 + 2\sqrt6)
= 5\sqrt2 + 2\sqrt{12} + 5\sqrt3 + 2\sqrt{18}
= 11\sqrt2 + 9\sqrt3 ,
\]
\omterm{prop:b1:vspaces:coordinates}{COORDINATES} $(0, 11, 9, 0)$ (का उपयोग करते हुए $\sqrt{12} = 2\sqrt3$,
$\sqrt{18} = 3\sqrt2$). अंत में $ s^4 = (s^2)^2 = 25 + 20\sqrt6 +
24 = 49 + 20\sqrt6$: \omterm{prop:b1:vspaces:coordinates}{COORDINATES} $(49, 0, 0, 20)$.

\textbf{15.} कल्पना करना $ a + bs + cs^2 + ds^3 = 0$. चार पढ़ना
\omterm{prop:b1:vspaces:coordinates}{COORDINATES} पर $(1, \sqrt2, \sqrt3, \sqrt6)$:
\[
a + 5c = 0,\qquad b + 11d = 0,\qquad b + 9d = 0,\qquad 2c = 0 .
\]
इसलिए $ c = 0$, तब $ a = 0$; मध्य समीकरणों को घटाना, $2d =
0$, तब $ b = 0$: $(1, s, s^2, s^3)$ है \omterm{def:b1:vspaces:free}{मुक्त}. चार \omterm{def:b1:vspaces:free}{मुक्त} वेक्टर
में $ M $ आयाम का $4$: ए \omterm{def:b1:vspaces:free}{आधार}, इसलिए
$\operatorname{Vect}_\Q(1, s, s^2, s^3) = M $. प्रश्न 14 से,
$ s^3 - 9s = 2\sqrt2$ और $11s - s^3 = 2\sqrt3$:
\[
\sqrt2 = \frac{s^3 - 9s}{2},
\qquad
\sqrt3 = \frac{11s - s^3}{2} .
\]

\textbf{16.} $ s^4 - 10s^2 + 1 = (49 + 20\sqrt6) - 10(5 +
2\sqrt6) + 1 = 0$. ए \omterm{def:b1:poly:def}{राक्षसी} \omterm{def:b1:poly:def}{बहुपद} डिग्री का $\leq 3$
पर लुप्त हो रहा है $ s $ का एक गैर-तुच्छ शून्य संयोजन उत्पन्न करेगा
$(1, s, s^2, s^3)$, विरोधाभासी प्रश्न 15: $ X^4 - 10X^2 + 1$
सबसे कम डिग्री है. इसकी जड़ें: $ X^2 = 5 \pm 2\sqrt6 = (\sqrt3 \pm
\sqrt2)^2$, तो चार वास्तविक जड़ें हैं $\pm(\sqrt3 + \sqrt2)$ और
$\pm(\sqrt3 - \sqrt2)$, यानी\ $\pm\sqrt2 \pm \sqrt3$.

\textbf{17.} से $ s^4 - 10s^2 + 1 = 0$: $ s\,(10s - s^3) = 1$, इसलिए
\[
\frac1s = 10s - s^3 = 10(\sqrt2 + \sqrt3) - (11\sqrt2 + 9\sqrt3)
= \sqrt3 - \sqrt2 ,
\]
के साथ संगत $(\sqrt3 + \sqrt2)(\sqrt3 - \sqrt2) = 1$. अगर $ a +
b\sqrt2 + c\sqrt3 + d\sqrt6 = r \in \Q $, तब $(a - r) + b\sqrt2
+ c\sqrt3 + d\sqrt6 = 0$, और उन्मुक्तता (प्रश्न 8) बल $ b = c
= d = 0$ (और $ a = r $). के लिए $\sqrt2 + \sqrt3 + \sqrt6$ 
\omterm{prop:b1:vspaces:coordinates}{COORDINATES} $(0, 1, 1, 1)$ इस रूप के नहीं हैं: तर्कहीन ---
ए \omterm{def:b1:logic:statement}{कथन} की तुलना में मजबूत \cref{exo:b1:reals:7}, बिना प्राप्त किया गया
कोई चुकता चाल.

\textbf{18.} एक तर्कसंगत जड़ $ p/q $ (न्यूनतम शर्तें) की $ X^3 - 2$
संतुष्ट $ p^3 = 2q^3$, और की कसौटी
\cref{exo:b1:poly:5} देता है $ p \mid 2$, $ q \mid 1$: उम्मीदवार
$\pm1, \pm2$, जिनके घन हैं $\pm1, \pm8 \neq 2$. कोई तर्कसंगत नहीं
जड़; विशेष रूप से $ t = 2^{1/3} \notin \Q $, इसलिए $(1, t)$ है \omterm{def:b1:vspaces:free}{मुक्त}
ऊपर $\Q $ (जैसा कि प्रश्न 2 में है)।

\textbf{19.} कल्पना करना $(1, t, t^2)$ जुड़ा हुआ: कुछ अशून्य $ P \in
\Q[X]$ साथ $\deg P \leq 2$ है $ P(t) = 0$; ऐसा चुनें $ P $ का
\emph{कम से कम} डिग्री $ d \geq 1$. प्रश्न 18 द्वारा, $ d \neq 1$, इसलिए $ d
= 2$. यूक्लिडियन डिवीजन (\cref{thm:b1:poly:division}): $ X^3 - 2
= PQ + R $ साथ $ Q, R \in \Q[X]$, $\deg R < 2$. पर मूल्यांकन कर रहे हैं $ t $:
$0 = P(t)Q(t) + R(t) = R(t)$, इसलिए $ R $ पर गायब हो जाता है $ t $; न्यूनता
का $ d $ बल $ R = 0$. तब $ X^3 - 2 = PQ $ साथ $\deg Q = 1$: द
की तर्कसंगत जड़ $ Q $ का तर्कसंगत मूल है $ X^3 - 2$,
विरोधाभासी प्रश्न 18. अत: $(1, t, t^2)$ है \omterm{def:b1:vspaces:free}{मुक्त} और
$\dim_\Q A = 3$. स्थिरता: $ t^3 = 2$ के प्रत्येक उत्पाद को कम करता है
$1, t, t^2$ उनमें से एक संयोजन के लिए ($ t\cdot t^2 = 2$, $ t^2\cdot
t^2 = 2t $); $ A $ रोकना $1$: प्रश्न 12 द्वारा, $ A $ एक उपक्षेत्र है
का $\R $.

\textbf{20.} कल्पना करना $ t \in M $. तब $ t^2 \in M $ (स्थिरता), तो
$ A \subseteq M $, और $ M $ एक है सदिश स्थल ऊपर \omterm{def:b1:structures:field}{मैदान} $ A $:
अभिगृहीत वे हैं $\R $-अंकगणित, प्रतिबंधित। यह है
\omterm{def:b1:findim:def}{परिमित-आयामी} ऊपर $ A $ (एक परिमित $\Q $-परिवार का सृजन करना
एक फ़ोर्टिओरी ओवर उत्पन्न करता है $ A \supseteq \Q $). \omterm{pb:b1:findim:1}{टावर कानून} के लिए
$\Q \subseteq A \subseteq M $ देता है
\[
4 = \dim_\Q M = \dim_\Q A \cdot \dim_A M = 3\,\dim_A M ,
\]
असंभव: $3$ विभाजित नहीं करता $4$. इसलिए $2^{1/3} \notin M $: नहीं
का तर्कसंगत संयोजन $1, \sqrt2, \sqrt3, \sqrt6$ के बराबर होती है
$2^{1/3}$.

\textbf{21.} $\Q(\sqrt2) \subseteq M $, इसलिए $ t \notin \Q(\sqrt2)$.
अगर $ t + \sqrt2 = r \in \Q $, तब $ t = r - \sqrt2 \in \Q(\sqrt2)$:
विरोधाभास --- $2^{1/3} + \sqrt2$ तर्कहीन है. अगर $ t = a +
b\sqrt3$, तब $ t \in \operatorname{Vect}_\Q(1, \sqrt3) \subseteq
M $: फिर से विरोधाभास.

\textbf{22.} $ B $ एक है सदिश स्थल ऊपर \omterm{def:b1:structures:field}{मैदान} $ A $
(स्केलर का प्रतिबंध), \omterm{def:b1:findim:def}{परिमित-आयामी} तब से $\dim_\Q B $ है
परिमित. \omterm{pb:b1:findim:1}{टावर कानून} के लिए $\Q \subseteq A \subseteq B $ देता है
$\dim_\Q B = \dim_\Q A \cdot \dim_A B $: बायां कारक विभाजित.
के उपक्षेत्र $ M $ इसलिए है $\Q $-आयाम $1$, $2$ या $4$:
आयाम $1$ है $\Q $ स्वयं, आयाम $4$ है $ M $.

\textbf{23.} होने देना $ F \subseteq M $ के साथ एक उपक्षेत्र बनें $\dim_\Q F =
2$, और $ w \in F \setminus \Q $: $(1, w)$ है \omterm{def:b1:vspaces:free}{मुक्त}, इसलिए ए \omterm{def:b1:vspaces:free}{आधार}
का $ F $. प्रश्न 13 द्वारा ($ d = 2$), $ w^2 = \alpha + \beta w $ के लिए
तर्कसंगत $\alpha, \beta $. सेटिंग $ v = w - \beta/2 \in F
\setminus \Q $:
\[
v^2 = w^2 - \beta w + \frac{\beta^2}4 = \alpha + \frac{\beta^2}4
\;\in\; \Q,
\]
और $ F = \operatorname{Vect}(1, v)$. अब लिखें $ v = a + b\sqrt2 +
c\sqrt3 + d\sqrt6$ और विस्तार करें:
\[
v^2 = (a^2 + 2b^2 + 3c^2 + 6d^2)
+ (2ab + 6cd)\sqrt2 + (2ac + 4bd)\sqrt3 + (2ad + 2bc)\sqrt6 .
\]
तीन अतार्किक \omterm{prop:b1:vspaces:coordinates}{COORDINATES} गायब होना: $ ab + 3cd = ac + 2bd =
ad + bc = 0$. अगर $ a \neq 0$: $ b = -3cd/a $, और दूसरा
स्थिति बन जाती है $ c(a^2 - 6d^2) = 0$; तब से $ a^2 = 6d^2$ साथ $ d
\neq 0$ करना होगा $\sqrt6 = \abs{a/d}$ तर्कसंगत, या तो $ c = 0$
या $ d = 0$, और दोनों ही स्थितियों में शेष स्थितियाँ बाध्य होती हैं $ b =
c = d = 0$, यानी\ $ v \in \Q $: छोड़ा गया। इसलिए $ a = 0$, और
शर्तें पढ़ें $3cd = 2bd = bc = 0$: अधिक से अधिक एक $ b, c, d $ है
शून्येतर. इस तरह $ v $ का एक तर्कसंगत गुणज है $\sqrt2$, $\sqrt3$
या $\sqrt6$, और
\[
F \in \bigl\{\Q(\sqrt2),\ \Q(\sqrt3),\ \Q(\sqrt6)\bigr\} ,
\]
जिनमें से प्रत्येक वास्तव में एक है $2$-आयामी उपक्षेत्र (स्थिरता के रूप में)।
प्रश्न 6 में, प्रश्न 12 से व्युत्क्रम): बिल्कुल तीन द्विघात
उपक्षेत्र।

\textbf{24.} $(1 + \sqrt2 + \sqrt3)(1 + \sqrt2 - \sqrt3) = (1 +
\sqrt2)^2 - 3 = 2\sqrt2$, इसलिए
\[
\frac1{1 + \sqrt2 + \sqrt3}
= \frac{1 + \sqrt2 - \sqrt3}{2\sqrt2}
= \frac{\sqrt2 + 2 - \sqrt6}{4}
= \frac12 + \frac14\sqrt2 + 0\cdot\sqrt3 - \frac14\sqrt6 .
\]
(जाँच करना: $(1 + \sqrt2 + \sqrt3)(2 + \sqrt2 - \sqrt6) = 4$ के बाद
विस्तार.)

\textbf{25.} (मैं) $\Q $-आयाम एक जोड़ता है \emph{पूर्णांक} को
के प्रत्येक उपक्षेत्र $\R $, और यह \omterm{pb:b1:findim:1}{टावर कानून} ये पूर्णांक बनाता है
समावेशन के साथ गुणा करें: आयाम एक अंकगणित की तरह व्यवहार करता है
अपरिवर्तनीय, और \omterm{def:b1:arith:divides}{भाजकत्व} बाधाएँ असंभव हो जाती हैं
सबूत (ii) परिमित आयाम प्रत्येक तत्व को a को संतुष्ट करने के लिए बाध्य करता है
शून्येतर तर्कसंगत \omterm{def:b1:poly:def}{बहुपद} अधिकतम डिग्री का समीकरण
आयाम: परिमितता का अर्थ है बीजगणित। (iii) द
तीन में से दो का नियम ``गुणा'' में बदल गया $ u $ एमएपीएस ए \omterm{def:b1:vspaces:free}{आधार}
एक को \omterm{def:b1:vspaces:free}{मुक्त} परिवार अधिकतम आकार का'' प्रक्षेपात्मकता में, उत्पादन
$ u^{-1}$ बिना किसी सूत्र के: व्युत्क्रम गिनती से आए। (iv) द
\omterm{pb:b1:findim:1}{टावर कानून} ए को मना करता है $3$ आयामी \omterm{def:b1:structures:field}{मैदान} अंदर ए
$4$-आयामी एक, यही कारण है $2^{1/3}$ पहुंचा नहीं जा सकता
से $\sqrt2$ और $\sqrt3$. भाग III का प्रमेय है
\emph{डेडेकाइंड का \omterm{pb:b1:findim:1}{टावर कानून}}गैलोज़ सिद्धांत का प्रारंभिक कदम,
वर्ष~3 खंड में विकसित किया गया।
\end{solution}
