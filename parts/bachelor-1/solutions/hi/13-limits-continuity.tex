\section*{अध्याय \ref{ch:b1:continuity} --- सीमाएँ और निरंतरता}

\begin{solution}{exo:b1:continuity:1}
लेना $u_n = \frac{1}{2\pi n + \pi/2}$ और $ v_n = \frac{1}{2\pi n}$:
दोनों की प्रवृत्ति होती है $0^+$, अभी तक $\sin\frac{1}{u_n} = 1$ और
$\sin\frac{1}{v_n} = 0$. दो क्रम, दो अलग-अलग सीमाएँ
छवियाँ: द्वारा \cref{thm:b1:continuity:seqchar}, कोई \omterm{def:b1:topology:closure}{सीमा} नहीं $0^+$.

$ x \sin\frac1x $: द्वारा निचोड़ा हुआ $\abs{x\sin\frac1x} \leq \abs x \to 0$,
तो \omterm{def:b1:topology:closure}{सीमा} पर $0$ मौजूद है और बराबर है $0$.
\end{solution}

\begin{solution}{exo:b1:continuity:2}
$ f = \lfloor\cdot\rfloor $ है निरंतर पर $\R \setminus \Z $
(स्थानीय रूप से स्थिर) और प्रत्येक पर असंतत $ n \in \Z $: बाईं \omterm{def:b1:topology:closure}{सीमा}
$ n - 1$, कीमत $ n $.

$ g(x) = x - \lfloor x\rfloor $: समान असंततता बिंदु (द
पहचान है निरंतर, इसलिए $ g $ की छलांग विरासत में मिली है $ f $); पर $ n \in
\Z $, बाईं \omterm{def:b1:topology:closure}{सीमा} $1 \neq 0 = g(n)$.

$ h(x) = \lfloor x\rfloor + (x - \lfloor x\rfloor)^2$: पर $\intco{n}{n
+ 1}$, $ h(x) = n + (x - n)^2$, निरंतर वहाँ; पर $ x = n $, बायां
\omterm{def:b1:topology:closure}{सीमा} है $(n-1) + 1 = n = h(n)$: छलांग रद्द हो जाती है। $ h $ है निरंतर
पर $\R $ (और सख्ती से बढ़ रहा है)।
\end{solution}

\begin{solution}{exo:b1:continuity:3}
$ P(x) = x^5 - 3x + 1$: $ P(-2) = -32 + 6 + 1 = -25 < 0$; $ P(0) = 1 >
0$; $ P(1) = -1 < 0$; $ P(2) = 27 > 0$. पर तीन चिन्ह \omterm{def:b1:counting:objects}{परिवर्तन}
असंयुक्त खंड $\intcc{-2}{0}$, $\intcc{0}{1}$, $\intcc{1}{2}$:
द्वारा \cref{thm:b1:continuity:ivt}, कम से कम तीन जड़ें। (का होना
डिग्री $5$, $ P $ अधिकतम पाँच हैं; एक भिन्नता अध्ययन दिखाएगा
बिल्कुल तीन.)
\end{solution}

\begin{solution}{exo:b1:continuity:4}
होने देना $ P = a_{2m+1}X^{2m+1} + \dots $ साथ $ a_{2m+1} > 0$ (अन्यथा प्रतिस्थापित करें
$ P $ द्वारा $-P $). प्रमुख शब्द का गुणनखंडन करते हुए, $ P(x) = a_{2m+1}
x^{2m+1}\bigl(1 + o(1)\bigr)$ जैसा $ x \to \pm\infty $: इसलिए $ P(x) \to
+\infty $ पर $+\infty $ और $-\infty $ पर $-\infty $. चुनना $ a $ साथ
$ P(a) < 0$ और $ b $ साथ $ P(b) > 0$: मध्यवर्ती मूल्य प्रमेय पर
$\intcc{a}{b}$ एक जड़ प्रदान करता है.
\end{solution}

\begin{solution}{exo:b1:continuity:5}
होने देना $ g(x) = f(x) - x $, निरंतर पर $\intcc{0}{1}$. तब से $ f $ एमएपीएस
में $\intcc{0}{1}$: $ g(0) = f(0) \geq 0$ और $ g(1) = f(1) - 1 \leq
0$. द्वारा \cref{thm:b1:continuity:ivt}, $ g(c) = 0$ कुछ के लिए $ c $: ए
निश्चित बिंदु.

परिकल्पनाओं की आवश्यकता: पर $\intcc{0}{1}$, असंतत \omterm{def:b1:logic:map}{नक्शा}
$ f(x) = 1$ के लिए $ x \leq \frac12$, $ f(x) = 0$ के लिए $ x > \frac12$ नहीं है
निश्चित बिंदु; पर \omterm{prop:b1:reals:intervals}{अंतराल} $\intoo{0}{1}$ (खंड नहीं), $ f(x) =
\frac x2$ है निरंतर में $\intoo{0}{1}$ बिना किसी निश्चित बिंदु के
(उम्मीदवार $0$ लापता है); पर $\R $, $ f(x) = x + 1$.
\end{solution}

\begin{solution}{exo:b1:continuity:6}
$ g(x) = \cos x - x $ है निरंतर, $ g(0) = 1 > 0$, $ g(1) = \cos 1 -
1 < 0$: एक समाधान मौजूद है $\intoo{0}{1}$
(\cref{thm:b1:continuity:ivt}). विशिष्टता: $ g $ सख्ती से है
पर घट रहा है $\R $ --- के लिए $ x \leq 0$, $ g(x) \geq 1 - x > 0$ नहीं है
वैसे भी शून्य; और $ g'(x) = -\sin x - 1 \leq 0$ केवल समानता के साथ
पृथक बिंदु ($ x \equiv -\frac\pi2 \bmod 2\pi $), इसलिए $ g $ है
सख्ती से घट रही है (\cref{ch:b1:derivative}; वैकल्पिक रूप से: चालू
$\intcc{0}{1}$, $\cos $ सख्ती से कम हो रहा है और $-x $ भी इसलिए $ g $
है)। एक सख्त मोनोटोन फ़ंक्शन अधिकतम एक बार गायब हो जाता है।
\end{solution}

\begin{solution}{exo:b1:continuity:7}
हल करना $ M = f(0) + 1$. वहाँ है $ A > 0$ साथ $ f(x) \geq M $ के लिए $\abs x
\geq A $ (दो अनंत सीमाओं की परिभाषा; बड़ा लें
दहलीज)। खंड पर $\intcc{-A}{A}$, चरम मूल्य प्रमेय
(\cref{thm:b1:continuity:evt}) प्रदान करता है $ c $ साथ $ f(c) =
\inf_{\intcc{-A}{A}} f \leq f(0)$. के लिए $\abs x \geq A $: $ f(x) \geq M
> f(0) \geq f(c)$. इसलिए $ f(c)$ वैश्विक न्यूनतम है.
\end{solution}

\begin{solution}{exo:b1:continuity:8}
खंड पर $\intcc{0}{T}$, $ f $ बंधा हुआ है और अपनी \omterm{def:b1:topology:closure}{सीमा} को प्राप्त करता है
(\cref{thm:b1:continuity:evt}); आवधिकता के अनुसार, ये सीमाएँ हैं
सब पर $\R $, अभी भी प्राप्त है।

होने देना $ g(x) = f(x + \frac T2) - f(x)$, निरंतर. तब
\[
g(0) + g\bigl(\tfrac T2\bigr)
= \bigl(f(\tfrac T2) - f(0)\bigr) + \bigl(f(T) - f(\tfrac T2)\bigr)
= f(T) - f(0) = 0 :
\]
$ g(0)$ और $ g(\frac T2)$ विपरीत संकेत हैं (या कोई गायब हो जाता है), इसलिए
मध्यवर्ती मूल्य प्रमेय पर $\intcc{0}{T/2}$ देता है $ c $ साथ
$ g(c) = 0$, यानी\ $ f(c + \frac T2) = f(c)$.
\end{solution}

\begin{solution}{exo:b1:continuity:9}
सबसे पहले असमानता: के लिए $0 \leq y \leq x $,
\[
\bigl(\sqrt y + \sqrt{x - y}\bigr)^2 = x + 2\sqrt{y(x-y)} \geq x,
\]
इसलिए $\sqrt x \leq \sqrt y + \sqrt{x - y}$, यानी\ $\sqrt x - \sqrt y
\leq \sqrt{x - y}$. इस तरह $\abs{\sqrt x - \sqrt y} \leq
\sqrt{\abs{x - y}}$ सभी के लिए $ x, y \geq 0$.

एकसमान निरंतरता: दिया गया $\varepsilon > 0$, लेना $\delta =
\varepsilon^2$; तब $\abs{x - y} \leq \delta $ तात्पर्य $\abs{\sqrt x
- \sqrt y} \leq \sqrt\delta = \varepsilon $.

लिप्सचिट्ज़ निकट नहीं $0$: $\frac{\sqrt x - \sqrt 0}{x - 0} =
\frac{1}{\sqrt x} \to +\infty $ जैसा $ x \to 0^+$, इसलिए कोई स्थिरांक नहीं $ k $
सभी अंतर भागफल पर हावी हो सकता है।
\end{solution}

\begin{solution}{exo:b1:continuity:10}
कल्पना करना $ f $ है \omterm{def:b1:logic:inj}{इंजेक्शन}, निरंतर, और सख्ती से एकरस नहीं।
फिर हैं $ a < b < c $ में $ I $ साथ $ f(b)$ बीच में नहीं $ f(a)$ और
$ f(c)$ --- वास्तव में, यदि के लिए \emph{सभी} मध्य मान तीन गुना था
बाहरी लोगों के बीच, $ f $ मोनोटोनिक होगा (किन्हीं दो की तुलना करें)।
जोड़े; एक संक्षिप्त केस जांच)। कहना $ f(b) > \max(f(a), f(c))$ (दूसरा
मामला सममित है, बदलें $ f $ द्वारा $-f $). चुनना $ v $ साथ $\max(f(a),
f(c)) < v < f(b)$. मध्यवर्ती मूल्य प्रमेय द्वारा पर लागू
$\intcc{a}{b}$ और पर $\intcc{b}{c}$, वहाँ हैं $ u_1 \in
\intoo{a}{b}$ और $ u_2 \in \intoo{b}{c}$ साथ $ f(u_1) = v = f(u_2)$:
समान छवियों वाले दो अलग-अलग बिंदु, इंजेक्टिविटी का खंडन करते हुए।
\end{solution}

\begin{solution}{exo:b1:continuity:11}
$ f(0) = f(0+0) = 2f(0)$ देता है $ f(0) = 0$; $ f(-x) = -f(x)$ से $0 =
f(x - x)$. \omterm{def:b1:logic:sets}{तय करना} $\alpha = f(1)$. प्रेरण: $ f(n) = n\alpha $ के लिए $ n
\in \N $, फिर के लिए $ n \in \Z $ विचित्रता से. के लिए $ q \in \N^*$:
$ q\,f(\frac pq) = f(p) = p\alpha $ (जोड़ना $\frac pq $ खुद को $ q $
समय), तो $ f(\frac pq) = \alpha\frac pq $: $ f = \alpha\,
\mathrm{id}$ पर $\Q $.

अभी चलो $ x \in \R $ और $(r_n)$ तर्कसंगतता का एक क्रम $ r_n \to
x $ (घनत्व, \cref{thm:b1:reals:density}, नेस्टेड में लागू किया गया
\omterm{prop:b1:reals:intervals}{अंतराल}; या $ r_n = \frac{\lfloor nx\rfloor}{n}$). निरंतरता:
$ f(x) = \lim f(r_n) = \lim \alpha r_n = \alpha x $.
\end{solution}

\begin{solution}{exo:b1:continuity:12}
होने देना $\varepsilon > 0$. \omterm{def:b1:topology:closure}{सीमा} से $+\infty $, वहाँ है $ A $ साथ
$\abs{f(x) - \ell} \leq \frac\varepsilon2$ के लिए $ x \geq A $; इसलिए के लिए
$ x, y \geq A $: $\abs{f(x) - f(y)} \leq \varepsilon $ (कोई निकटता नहीं
आवश्यकता है)।

खंड पर $\intcc{0}{A + 1}$, हेन का प्रमेय
(\cref{thm:b1:continuity:heine}) देता है $\delta_0 > 0$ इसके लिए
$\varepsilon $; \omterm{def:b1:logic:sets}{तय करना} $\delta = \min(\delta_0, 1)$.

अब कोई भी ले लो $ x, y \geq 0$ साथ $\abs{x - y} \leq \delta $, कहना $ x
\leq y $. अगर $ y \leq A + 1$: दोनों खंड में स्थित हैं, और हेइन
$\delta_0$ लागू होता है. अन्यथा $ y > A + 1$, और तब $ x \geq y - 1 >
A $: दोनों झूठ बोलते हैं $\intco{A}{+\infty}$, जहां \omterm{def:b1:topology:closure}{सीमा} तर्क
लागू होता है. दोनों ही मामलों में $\abs{f(x) - f(y)} \leq \varepsilon $: वर्दी
निरंतरता.
\end{solution}

\begin{solution}{pb:b1:continuity:1}
\textbf{1.} के लिए $ n \in \N $: $ f(nx) = n f(x)$ प्रेरण द्वारा
($ f((n+1)x) = f(nx) + f(x)$). भी $ f(0) = 2f(0)$ देता है $ f(0) =
0$, और $0 = f(x - x) = f(x) + f(-x)$ विचित्रता देता है, इसलिए $ f(nx) =
nf(x)$ के लिए $ n \in \Z $. के लिए $ r = \frac pq $: $ q\,f\bigl(\tfrac
pq x\bigr) = f(px) = p\,f(x)$, इसलिए $ f(rx) = r f(x)$: $ f $ है
$\Q $-रैखिक.

\textbf{2.} एडिटिविटी देता है, सभी के लिए $ x $ और $ h $:
\[
f(x + h) - f(x) = f(h) = f(x_0 + h) - f(x_0) .
\]
जैसा $ h \to 0$, दाहिनी ओर की प्रवृत्ति होती है $0$ द्वारा निरंतरता पर
$ x_0$; इस तरह $ f(x + h) \to f(x)$: निरंतरता प्रत्येक पर $ x $. तब
\cref{exo:b1:continuity:11} पैदावार $ f(x) = f(1)\,x $.

\textbf{3.} दो निरंतर योगात्मक कार्य रूप के होते हैं
$ cx $ और $ c'x $ (प्रश्न 2); यदि वे इस पर सहमत हैं \omterm{def:b1:topology:dense}{घना} \omterm{def:b1:structures:subgroup}{उपसमूह}
$\Z + \sqrt2\,\Z $ (\cref{exo:b1:reals:9}), तब $ c\,g = c'g $ के लिए
कुछ $ g \neq 0$ इस में: $ c = c'$, कार्य समान हैं। बिना
निरंतरता: $\Q $-रैखिकता केवल मूल्यों को बांधती है
$\Q $-संयोजन, और $1, \sqrt2$ हैं $\Q $-स्वतंत्र ($\sqrt2
\notin \Q $), इसलिए $ f(1) = 0$, $ f(\sqrt2) = 1$ सुसंगत है और
बलों, पर \omterm{def:b1:structures:subgroup}{उपसमूह},
\[
f(m + n\sqrt2) = m\,f(1) + n\,f(\sqrt2) = n .
\]

\textbf{4.} के लिए $ t \in \intcc{0}{\ell}$: $ a + t \in
\intcc{a}{b}$, इसलिए $ f(t) = f(a + t) - f(a) \leq M - f(a) =: M'$.

\textbf{5.} के लिए $ t \in \intcc{0}{\ell}$, भी $\ell - t \in
\intcc{0}{\ell}$, और additivity देता है $ f(t) = f(\ell) - f(\ell
- t) \geq f(\ell) - M'$. इस तरह $\abs{f} \leq C $ पर
$\intcc{0}{\ell}$ साथ $ C = \max\bigl(\abs{M'}, \abs{f(\ell) -
M'}\bigr)$.

\textbf{6.} के लिए $ t \in \intcc{0}{\ell/n}$: $ nt \in
\intcc{0}{\ell}$ और $ f(t) = \frac{f(nt)}{n}$ (प्रश्न 1), तो
$\abs{f(t)} \leq \frac Cn $. दिया गया $\varepsilon > 0$, चुनना $ n >
\frac C\varepsilon $: के लिए $0 \leq h \leq \frac\ell n $,
$\abs{f(h)} \leq \varepsilon $, और नकारात्मक के लिए $ h $ उपयोग $ f(h) =
-f(-h)$. इसलिए $ f(h) \to 0 = f(0)$ जैसा $ h \to 0$: निरंतरता पर $0$,
इसलिए सर्वत्र (प्रश्न 2), इसलिए $ f(x) = f(1)x $.

\textbf{7.} अगर $ f $ पर घट नहीं रहा है $\intcc{a}{b}$, तब
$ f(a) \leq f(x) \leq f(b)$ वहाँ: प्रश्न से घिरा हुआ, इतना रैखिक
6, $ f(x) = cx $; और $ c(b - a) = f(b) - f(a) \geq 0$ बल $ c
\geq 0$.

\textbf{8.} (ए)$\Rightarrow $(बी)$\Rightarrow $(सी): तुच्छ.
(सी)$\Rightarrow $(ए): प्रश्न 2. (ए)$\Rightarrow $(डी): एक रैखिक
कार्य हर जगह एकरस है। (डी)$\Rightarrow $(ई): एक नीरस
पर कार्य करें $\intcc{a}{b}$ वहां इसके समापन बिंदु से घिरा हुआ है
मूल्य. (ई)$\Rightarrow $(ए): प्रश्न 4-6. पांच
\omterm{def:b1:logic:statement}{कथन} समतुल्य हैं --- नियमितता सीढ़ी।

\textbf{9.} प्रश्न 4 में केवल का उपयोग किया गया \omterm{def:b1:reals:bounds}{ऊपरी सीमा} $ M $; प्रश्न 5
\emph{व्युत्पन्न} निचला भाग ऊपर वाले से होकर गुजरता है
प्रतिबिंब $ f(t) = f(\ell) - f(\ell - t)$; प्रश्न 6 फिर चला
$\abs f \leq C $. तो: योगात्मक और परिबद्ध \emph{ऊपर} एक पर
अविक्षिप्त \omterm{prop:b1:reals:intervals}{अंतराल} पहले से ही रैखिक का तात्पर्य है। एक \omterm{def:b1:topology:closure}{सीमा} के लिए
नीचे, इसे लागू करें $-f $ (योगात्मक, ऊपर परिबद्ध)। सबसे मजबूत
पायदान (ई): एक तरफा एक पर बंधा हुआ \omterm{prop:b1:reals:intervals}{अंतराल} पर्याप्त.

\textbf{10.} अगर $\frac{f(x)}{x}$ एक स्थिरांक थे $ c $ सभी के लिए
$ x \neq 0$, $ f $ रैखिक होगा; अतः शून्येतर हैं $ u, v $
साथ $\frac{f(u)}u \neq \frac{f(v)}v $, यानी\ $ u f(v) - v f(u)
\neq 0$: सदिशों का निर्धारक $(u, f(u))$, $(v,
f(v))$ शून्येतर है, और वे फैले हुए हैं $\R^2$.

\textbf{11.} के लिए $ r, s \in \Q $: $ f(ru + sv) = r f(u) + s f(v)$
(प्रश्न 1 दो बार प्लस एडिटिविटी), इसलिए ग्राफ़ में शामिल है
\[
\bigl(ru + sv,\; r f(u) + s f(v)\bigr)
= r\,(u, f(u)) + s\,(v, f(v)) .
\]
दिया गया $(x_0, y_0)$ और $\varepsilon > 0$: द $2 \times 2$
प्रणाली $ a(u, f(u)) + b(v, f(v)) = (x_0, y_0)$ एक (अद्वितीय) है
वास्तविक समाधान $(a, b)$ चूँकि सारणिक अशून्य है। उठाओ
तर्कसंगत $ r_n \to a $, $ s_n \to b $: तब $ r_n(u, f(u)) +
s_n(v, f(v)) \to (x_0, y_0)$ समन्वय के अनुसार, और इनमें से प्रत्येक
बिंदु ग्राफ़ पर स्थित हैं: ग्राफ़ है \omterm{def:b1:topology:dense}{घना} में $\R^2$.

\textbf{12.} होने देना $ I $ नपुंसक बनो \omterm{prop:b1:reals:intervals}{अंतराल}, $ x_0$ इसका
मध्यबिंदु, $ M $ मनमाना: घनत्व भीतर एक ग्राफ़ बिंदु प्रदान करता है
$\min\bigl(\frac{\abs I}{2}, 1\bigr)$ का $(x_0, M + 1)$, अर्थात\
$ x \in I $ साथ $ f(x) > M $: \omterm{def:b1:reals:bounds}{असीम}ित पर $ I $, इसलिए (प्रश्न 8)
हर बिंदु पर असंतत और नहीं पर एकरस \omterm{prop:b1:reals:intervals}{अंतराल}; और
किसी भी लक्ष्य के लिए $ y_0$, ग्राफ़ बिंदु निकट $(x_0, y_0)$ देना $ f(x)$
मनमाने ढंग से करीब $ y_0$ साथ $ x \in I $: $ f(I)$ है \omterm{def:b1:topology:dense}{घना} में
$\R $. यह प्रश्न 8 है जिसे पीछे की ओर पढ़ें: चूंकि सभी पायदान हैं
समतुल्य, एक अरेखीय योगात्मक फ़ंक्शन विफल होना चाहिए \emph{सभी}
उनमें से, हर जगह.

\textbf{13.} अच्छी तरह से परिभाषित: $ m + n\sqrt2 = m' + n'\sqrt2$ बल
$(n - n')\sqrt2 = m' - m \in \Z $, इसलिए $ n = n'$ (अन्य $\sqrt2 \in
\Q $) और $ m = m'$. एडिटिविटी चालू $ G $ तब स्पष्ट समन्वय होता है
समन्वय करें. असीमता निकट $0^+$: हल करना $\varepsilon \in
\intoo{0}{1}$ और $ N \in \N $. प्रत्येक के लिए निश्चित $ n $ साथ $\abs n
\leq N $, स्थिति $ m + n\sqrt2 \in \intoo{0}{1}$ पिंस $ m $
एक के अंदर \omterm{prop:b1:reals:intervals}{अंतराल} लम्बाई का $1$: अधिकतम एक पूर्णांक $ m $ प्रति
$ n $, तो अधिक से अधिक $2N + 1$ घटक $ G \cap \intoo{0}{1}$ पास होना
$\abs{f} \leq N $. लेकिन $ G \cap \intoo{0}{\varepsilon}$ है
अनंत ($ G $ है \omterm{def:b1:topology:dense}{घना}, \cref{exo:b1:reals:9}); इसलिए यह
कुछ शामिल है $ g $ साथ $\abs{f(g)} > N $: $ f $ पर \omterm{def:b1:reals:bounds}{असीम}ित है
हर अधिकार \omterm{def:b1:topology:open}{अड़ोस-पड़ोस} का $0$. विस्तार $ f $ एक योजक के लिए
पर कार्य करें $\R $ का अर्थ है एक परिवार पर सुसंगत रूप से मूल्यों का चयन करना
वास्तविक वह है $\Q $-रैखिक रूप से स्वतंत्र और फैला हुआ $\R $ ऊपर $\Q $
--- एक हैमेल आधार; किसी का उत्पादन करने के लिए पसंद के सिद्धांत की आवश्यकता होती है,
और कोई स्पष्ट सूत्र यह नहीं कर सकता: रचनात्मक रूप से हम इसके स्वामी हैं
राक्षस केवल पर $ G $.

\textbf{14.} $ f(x) = f(\frac x2 + \frac x2) = f(\frac x2)^2 \geq
0$. अगर $ f(x_0) = 0$, तब $ f(x) = f(x - x_0)f(x_0) = 0$ सभी के लिए
$ x $: छोड़ा गया। इसलिए $ f > 0$ और $ g = \ln \circ f $ है निरंतर
(\cref{prop:b1:continuity:algebra}) साथ $ g(x + y) = g(x) +
g(y)$: द्वारा \cref{exo:b1:continuity:11}, $ g(x) = cx $, इसलिए $ f(x) =
\eu^{cx}$. इसके विपरीत प्रत्येक $\eu^{cx}$ काम करता है: निरंतर
रूपवाद $(\R, +) \to (\R^*, \times)$ बिल्कुल वही हैं
घातांक.

\textbf{15.} $ g(u) = f(\eu^u)$ है निरंतर और $ g(u + v) =
f(\eu^u \eu^v) = g(u) + g(v)$: $ g(u) = cu $, और हर $ x > 0$
लिखते हैं $ x = \eu^u $ साथ $ u = \ln x $: $ f(x) = c\ln x $.

\textbf{16.} $ h = \ln \circ f $ है निरंतर पर
$\intoo{0}{+\infty}$ साथ $ h(xy) = h(x) + h(y)$: प्रश्न 15 द्वारा,
$ h(x) = c\ln x $, इसलिए $ f(x) = \eu^{c\ln x} = x^c $.

\textbf{17.} $ x = y = 0$: $ f(0) = 2f(0) + f(0)^2$, इसलिए $ f(0)(1 +
f(0)) = 0$. अगर $ f(0) = -1$: सेटिंग $ y = 0$, $ f(x) = f(x) +
f(0) + f(x)f(0) = -1$ सभी के लिए $ x $: स्थिरांक $ f \equiv -1$
(जो वास्तव में समीकरण को संतुष्ट करता है)। अन्यथा $ f(0) = 0$; $ h
= 1 + f $ है निरंतर, $ h(0) = 1$, और
\[
h(x + y) = 1 + f(x) + f(y) + f(x)f(y) = h(x)\,h(y) :
\]
प्रश्न 14 द्वारा, $ h(x) = \eu^{cx}$, यानी\ $ f(x) = \eu^{cx} - 1$
(मामला $ c = 0$ दे रही है $ f \equiv 0$). पूरी लिस्ट: $ f
\equiv -1$ और $ f(x) = \eu^{cx} - 1$, $ c \in \R $.

\textbf{18.} $ x = y = 0$: $2f(0) = 4f(0)$, इसलिए $ f(0) = 0$. $ x =
0$: $ f(y) + f(-y) = 2f(y)$, इसलिए $ f $ सम है. $ y = x $: $ f(2x) =
4f(x)$. प्रेरण का उपयोग करना $(x, y) \to (nx, x)$:
\[
f((n{+}1)x) = 2f(nx) + 2f(x) - f((n{-}1)x)
= (2n^2 + 2 - (n-1)^2) f(x) = (n+1)^2 f(x) .
\]
तब $ f(x) = f\bigl(q\cdot\frac xq\bigr) = q^2 f\bigl(\frac
xq\bigr)$ देता है $ f\bigl(\frac pq x\bigr) = \frac{p^2}{q^2}f(x)$:
$ f(r) = r^2 f(1)$ पर $\Q $ (समता संकेतों को संभालती है)। दो
निरंतर कार्य $ f $ और $ x \mapsto f(1)x^2$ पर सहमत हूँ
\omterm{def:b1:topology:dense}{घना} \omterm{def:b1:logic:sets}{तय करना} $\Q $, इसलिए हर जगह (प्रश्न 24): $ f(x) =
f(1)\,x^2$; प्रत्येक $ cx^2$ समीकरण को संतुष्ट करता है.

\textbf{19.} $ g = f - f(0)$ है निरंतर, $ g(0) = 0$, और
जेन्सेन के समीकरण को संतुष्ट करता है (स्थिरांक रद्द)। लेना $ y =
0$: $ g\bigl(\frac x2\bigr) = \frac{g(x)}{2}$. फिर सबके लिए $ x,
y $:
\[
\frac{g(x + y)}{2} = g\Bigl(\frac{x+y}{2}\Bigr)
= \frac{g(x) + g(y)}{2} ,
\]
इसलिए $ g $ योगात्मक है और निरंतर: $ g(x) = cx $ (प्रश्न 2), और
$ f(x) = cx + d $ साथ $ d = f(0)$. सभी एफ़िन फ़ंक्शंस संतुष्ट हैं
जेन्सेन: सूची पूरी है.

\textbf{20.} प्रेरण चालू $ n $. के लिए $ n = 0$: $\lambda \in \{0,
1\}$, मामूली। सभी भारों के लिए असमानता मानें
$\frac{k}{2^n}$. एक वजन $\lambda = \frac{k}{2^{n+1}}$ साथ $ k $
स्तर तक भी कम कर देता है $ n $; के लिए $ k = 2j + 1$, $\lambda $ है
का मध्यबिंदु $\lambda_1 = \frac{j}{2^n}$ और $\lambda_2 =
\frac{j+1}{2^n}$. साथ $ z_i = \lambda_i x + (1 - \lambda_i)y $:
$\lambda x + (1-\lambda)y = \frac{z_1 + z_2}{2}$, इसलिए
\[
f\bigl(\lambda x + (1{-}\lambda)y\bigr)
\leq \frac{f(z_1) + f(z_2)}{2}
\leq \frac{(\lambda_1 + \lambda_2)f(x) + (2 - \lambda_1 -
\lambda_2)f(y)}{2}
= \lambda f(x) + (1 - \lambda)f(y) .
\]

\textbf{21.} हल करना $ x, y $. एमएपीएस $\lambda \mapsto f(\lambda x +
(1 - \lambda)y)$ और $\lambda \mapsto \lambda f(x) + (1 -
\lambda)f(y)$ हैं निरंतर पर $\intcc{0}{1}$ (रचना और
बीजगणित, \cref{prop:b1:continuity:algebra}). असमानता
डायडिक भार को धारण करता है, जो हैं \omterm{def:b1:topology:dense}{घना} में $\intcc{0}{1}$
(\cref{exo:b1:reals:8}); मनमानी के लिए $\lambda $ डायडिक्स ले लो
$\lambda_n \to \lambda $ और \omterm{def:b1:topology:closure}{सीमा} से गुजरें
(\cref{thm:b1:continuity:seqchar} और \cref{thm:b1:seq:order}):
उत्तलता असमानता प्रत्येक के लिए है $\lambda \in
\intcc{0}{1}$ --- मध्यबिंदु उत्तलता प्लस निरंतरता बराबर
उत्तलता (की धारणा) \cref{ch:b1:derivative}).

\textbf{22.} एक अरैखिक योज्य $ f $ संतुष्ट
$ f\bigl(\frac{x+y}{2}\bigr) = \frac{f(x) + f(y)}{2}$ बिलकुल
(प्रश्न 1 के साथ $ r = \frac12$, फिर additivity): यह है
मध्यबिंदु-उत्तल, यहाँ तक कि मध्यबिंदु-एफ़िन भी। अगर यह पूर्ण रूप से संतुष्ट है
कुछ पर उत्तलता असमानता \omterm{prop:b1:reals:intervals}{अंतराल} $\intcc{x}{y}$, फिर के लिए
$\lambda \in \intcc{0}{1}$: $ f(\lambda x + (1-\lambda)y) \leq
\max(f(x), f(y))$ --- ऊपर एक गैर-अपक्षयी पर घिरा हुआ \omterm{prop:b1:reals:intervals}{अंतराल},
इसलिए प्रश्न 9 द्वारा रैखिक: विरोधाभास। इसलिए निरंतरता में
प्रश्न 21 कोई विलासिता नहीं है: इसके बिना, मध्यबिंदु उत्तलता
केवल गणनीय डायडिक कंकाल और सातत्य को नियंत्रित करता है
बीच-बीच में जंगली दौड़ता है।

\textbf{23.} $ p(x) = \frac{x^2}{2}$ संतुष्ट $ p(x+y) = p(x) +
p(y) + xy $. अगर $ f $ कोई है निरंतर समाधान, $ g = f - p $ है
निरंतर और additive, तो $ g(x) = cx $:
\[
f(x) = \frac{x^2}{2} + cx , \qquad c \in \R ,
\]
और इनमें से प्रत्येक एक समाधान है: सूची पूरी है।

\textbf{24.} सामान्य सिद्धांत: यदि $ u, v $ हैं निरंतर और
ए पर सहमत हूं \omterm{def:b1:topology:dense}{घना} $ D \subseteq \R $, फिर के लिए $ x \in \R $ चुनना
$ d_n \in D $ साथ $ d_n \to x $
(\cref{prop:b1:topology:closureprops}); $ u(x) = \lim u(d_n) =
\lim v(d_n) = v(x)$. अब चलो $ f $ होना निरंतर और additive पर
द \omterm{def:b1:topology:dense}{घना} \omterm{def:b1:structures:subgroup}{उपसमूह} $ G $. हल करना $ x, y \in \R $ और ले लो $ g_n \to x $,
$ g'_n \to y $ साथ $ g_n, g'_n \in G $; तब $ g_n + g'_n \to x + y $
और, अनुक्रमिक रूप से निरंतरता पर $ x + y $, $ x $ और $ y $:
\[
f(x + y) = \lim f(g_n + g'_n) = \lim\bigl(f(g_n) +
f(g'_n)\bigr) = f(x) + f(y) :
\]
$ f $ सभी पर योगात्मक है $\R $ (इसलिए रैखिक, प्रश्न 2 से)।

\textbf{25.} (i) योगात्मक के लिए $ f $: रैखिक $\iff $ निरंतर
$\iff $ निरंतर एक बिंदु पर $\iff $ कुछ पर एकरसता \omterm{prop:b1:reals:intervals}{अंतराल}
$\iff $ कुछ पर सीमित (एकतरफ़ा भी)। \omterm{prop:b1:reals:intervals}{अंतराल}. (ii) घनत्व
विमान में ग्राफ से पता चलता है कि विफलता स्थानीय नहीं है
दोष लेकिन एक वैश्विक विस्फोट: प्रत्येक उपअंतराल के ऊपर
मूल्य सब पर हावी हो जाते हैं $\R $, इसलिए प्रत्येक नियमितता संपत्ति
एक ही बार में हर जगह विफल हो जाता है। (iii) पकड़: $ cx $, $\eu^{cx}$,
$ c\ln x $, $ x^c $, $ cx^2$, और $ cx + d $ --- छह लक्षण,
एक विधि: समीकरण को कॉची तक पहुँचाएँ, सिद्ध करें
$\Q $- प्रेरण द्वारा कंकाल, उन्नयन $\R $ द्वारा घनत्व प्लस
निरंतरता. (iv) घनत्व का $\Q$ (या डायडिक्स) ने किया
में उन्नयन \cref{exo:b1:continuity:11} और प्रश्न 21 में; यह
प्रश्न 13 में कुछ भी नहीं रखा, क्योंकि बिना निरंतरता
मान a से प्रसारित नहीं होते हैं \omterm{def:b1:topology:dense}{घना} \omterm{def:b1:logic:sets}{तय करना} इसके लिए \omterm{def:b1:topology:closure}{बंद} ---
घनत्व केवल जानकारी स्थानांतरित करता है निरंतरता.
\end{solution}
