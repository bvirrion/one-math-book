\section*{अध्याय \ref{ch:b1:curves} --- समतल वक्र}

\begin{solution}{exo:b1:curves:1}
$M(-t) = (t^2, -t^3)$: में प्रतिबिंब $ x $-अक्ष; अध्ययन $ t \geq 0$.
दोनों $ x' = 2t $ और $ y' = 3t^2$ हैं $\geq 0$: शाखा चलती है
दाएं और ऊपर, से $(0,0)$ अनंत की ओर। पर $ t = 0$ वेग
गायब हो जाता है; विस्तार $ x = t^2$, $ y = t^3$ दिखाएँ स्पर्शरेखा है
$ x $-अक्ष ($ y/x = t \to 0$) साथ $ y $ जबकि संकेत बदल रहा है $ x \geq 0$:
ए \emph{\omterm{ex:b1:curves:astroid}{उभार}} बायीं ओर इशारा करते हुए. वक्र अर्ध-घनाकार परवलय है
$ y^2 = x^3$.
\end{solution}

\begin{solution}{exo:b1:curves:2}
$ M(t + 2\pi) = M(t) + (2\pi, 0)$: वक्र दोहराता है, द्वारा अनुवादित
एक पहिया परिधि; एक \omterm{def:b1:vspaces:span}{अवधि} का अध्ययन करें. $ x'(t) = 1 - \cos t \geq
0$, $ y'(t) = \sin t $: मेहराब ऊपर उठता है $\intoo{0}{\pi}$, इस पर गिरना
$\intoo{\pi}{2\pi}$, पर समापन $(\pi, 2)$. एकवचन बिंदु
कहाँ $ x' = y' = 0$: $ t \in 2\pi\Z $, जमीन पर. पास में $ t = 0$:
\[
x(t) = \frac{t^3}{6} + o(t^3),
\qquad
y(t) = \frac{t^2}{2} + o(t^2):
\]
$ x $ संकेत बदलता है, $ y \geq 0$, और $\frac{x}{y^{3/2}}$ परिबद्ध:
स्पर्शरेखा है \emph{खड़ा} (दिशा $(0,1)$), ए \omterm{ex:b1:curves:astroid}{उभार} कहाँ
ट्रैक किए गए बिंदु पर क्षण भर के लिए शून्य गति होती है --- भौतिक हस्ताक्षर
बिना फिसले लुढ़कने का.
\end{solution}

\begin{solution}{exo:b1:curves:3}
पर $ t = \frac\pi4$: $ M = \bigl(\frac{\sqrt2}{4},
\frac{\sqrt2}{4}\bigr)$, स्पर्शरेखा दिशा $(-\cos t, \sin t) =
\frac{1}{\sqrt2}(-1, 1)$ (\cref{ex:b1:curves:astroid}). स्पर्श रेखा: $ y - \frac{\sqrt2}{4} = -(x - \frac{\sqrt2}{4})$, यानी\ $ x +
y = \frac{\sqrt2}{2}$. अवरोधन: $\bigl(\frac{\sqrt2}{2},
0\bigr)$ और $\bigl(0, \frac{\sqrt2}{2}\bigr)$; के बीच का खंड
उनके पास है \omterm{def:b1:curves:arclength}{लंबाई} $\sqrt{\frac12 + \frac12} = 1$.

(सामान्य बिंदु $ t $: पर स्पर्शरेखा $(\cos^3 t, \sin^3 t)$ काटता है
कुल्हाड़ियों पर $(\cos t, 0)$ और $(0, \sin t)$ --- जाँच करें कि लाइन
इन बिन्दुओं के माध्यम से दिशा मिलती है $(-\cos t, \sin t)$ और गुजरता है
के माध्यम से $ M(t)$ --- और कट खंड है \omterm{def:b1:curves:arclength}{लंबाई} $\sqrt{\cos^2 t +
\sin^2 t} = 1$.)
\end{solution}

\begin{solution}{exo:b1:curves:4}
$ r = \cos 2\theta $: \omterm{def:b1:vspaces:span}{अवधि} $\pi $ में $\theta $, दोनों में सममिति
कुल्हाड़ी; $ r $ पर गायब हो जाता है $\theta = \pm\frac\pi4$ (पर स्पर्शरेखा

$ r\cos\theta = 1$ समीकरण है $ x = 1$: द \omterm{def:b1:curves:polar}{ध्रुवीय वक्र} $ r =

\end{solution}

\begin{solution}{exo:b1:curves:5}
$ r(\theta) = 1 + 2\cos\theta $ के लिए गायब हो जाता है $\cos\theta = -\frac12$:
$\theta = \pm\frac{2\pi}{3}$. में \omterm{def:b1:linmaps:projection}{समरूपता} $ x $-अक्ष; अध्ययन
$\theta \in \intcc{0}{\pi}$. के लिए $\theta \in
\intco{0}{\frac{2\pi}{3}}$: $ r > 0$, से घट रहा है $3$ को $0$:
बाहरी चाप, किरण की मूल स्पर्शरेखा में प्रवेश करता है $\theta =
\frac{2\pi}{3}$. के लिए $\theta \in \intoc{\frac{2\pi}{3}}{\pi}$: $ r <
0$: बिन्दु $ r\vec u(\theta)$ विपरीत किरण (कोण) पर बैठें
$\theta - \pi \in \intoc{-\frac\pi3}{0}$), दूरी के साथ $\abs r $
से बढ़ रहा है $0$ को $1$: यह एक छोटा सा चित्र बनाता है \emph{आंतरिक फंदे}
उत्पत्ति के माध्यम से. पर $\theta = \pi $, $ r = -1$ और मुद्दा यह है
$-\vec u(\pi) = (1, 0)$: लूप सकारात्मक पर \omterm{def:b1:topology:closure}{बंद} होता है $ x $-अक्ष.
रेखाचित्र: एक बड़ा दिल जैसा बाहरी वक्र
अधिकतम पहुंच के साथ $3$ पर $\theta = 0$, साथ ही इसके अंदर एक लूप
के माध्यम से $ O $ और $(1,0)$.
\end{solution}

\begin{solution}{exo:b1:curves:6}
पूर्ण वर्ग:
\[
9(x^2 - 4x) + 25(y^2 - 2y) = 164
\iff 9(x-2)^2 + 25(y-1)^2 = 164 + 36 + 25 = 225 .
\]
द्वारा विभाजित करना $225$: $\frac{(x-2)^2}{25} + \frac{(y-1)^2}{9} = 1$: एक
केंद्र का दीर्घवृत्त $(2, 1)$, अर्ध-अक्ष $ a = 5$ (क्षैतिज), $ b =
3$; $ c = \sqrt{25 - 9} = 4$: फोकस $(2 \pm 4,\, 1) = (-2, 1)$ और
$(6, 1)$; सनक $ e = \frac45$.
\end{solution}

\begin{solution}{exo:b1:curves:7}
मूल, डायरेक्ट्रिक्स पर ध्यान दें $ D: x = d $ साथ $ d = \frac pe > 0$.
एक बिंदु के लिए $ M = (r\cos\theta, r\sin\theta)$ साथ $ r > 0$:
\[
MF = r,
\qquad
d(M, D) = \abs{d - r\cos\theta},
\]
और यह \omterm{def:b1:curves:conics}{शंकुधर} स्थिति $ MF = e\,d(M, D)$, शासन में जहां $ M $
डायरेक्ट्रिक्स के फोकस पक्ष पर है ($ r\cos\theta < d $), पढ़ता है $ r
= e(d - r\cos\theta)$, यानी
\[
r(1 + e\cos\theta) = ed = p,
\qquad
r = \frac{p}{1 + e\cos\theta} .
\]
के लिए $ e < 1$ हर कभी लुप्त नहीं होता: सब $\theta $ अनुमति
(दीर्घवृत्त)। के लिए $ e = 1$: $\theta \neq \pi $ (परवलय, की ओर \omterm{def:b1:topology:open}{खुला}
डायरेक्ट्रिक्स का सुदूर भाग)। के लिए $ e > 1$: ज़रूरत $1 + e\cos\theta > 0$,
यानी\ $\theta \in \intoo{-\theta_0}{\theta_0}$ साथ $\theta_0 =
\arccos\bigl(-\frac1e\bigr)$: एक \emph{शाखा} अतिपरवलय का
(दूसरी शाखा चिन्ह चयन से मेल खाती है $ r < 0$, या को
दूसरा फोकस)।
\end{solution}

\begin{solution}{exo:b1:curves:8}
\emph{माली:} की एक स्ट्रिंग संलग्न करें \omterm{def:b1:curves:arclength}{लंबाई} $2a $ दो दांव के लिए $ F,
F'$ और इसे अनुरेखण बिंदु के साथ तना हुआ रखें $ M $:बाधा है
बिलकुल $ MF + MF' = 2a $, इसलिए अनुरेखित वक्र दीर्घवृत्त है।

\emph{प्रतिबिंब संपत्ति:} होने देना $ t \mapsto M(t)$ नियमित रहो
पैरामीट्रिजेशन और $ u(t) = \frac{M(t) - F}{\norm{M(t) - F}}$,
$ u'(t)$ अनुरूप इकाई वेक्टर $ F'$. फर्क
$\norm{M - F} + \norm{M - F'} = 2a $, का उपयोग करना $\frac{\dd}{\dd
t}\norm{M - F} = \bigl\langle \frac{M - F}{\norm{M-F}},\,
M'\bigr\rangle $ (श्रृंखला नियम चालू $\sqrt{\langle\cdot,\cdot\rangle}$):
\[
\bigl\langle u + u',\, M'\bigr\rangle = 0 .
\]
तो स्पर्शरेखा दिशा $ M'$ है ओर्थोगोनल द्विभाजक को
दिशा $ u + u'$ दो फोकल किरणों में से: स्पर्शरेखा बराबर बनाती है
के साथ कोण $ MF $ और $ MF'$. (एक फोकस से प्रकाश की किरण परावर्तित होती है
दीर्घवृत्त से दूसरे फ़ोकस तक।)
\end{solution}

\begin{solution}{exo:b1:curves:9}
\begin{enumerate}
  \item कार्यक्षेत्र: $\cos 2\theta \geq 0$: $\theta \in
        \intcc{-\frac\pi4}{\frac\pi4} \cup
        \intcc{\frac{3\pi}{4}}{\frac{5\pi}{4}}$. \omterm{def:b1:linmaps:projection}{समरूपता}: $\theta
        \mapsto -\theta $ ($ x $-अक्ष) और $\theta \mapsto \pi -
        \theta $ ($ y $-अक्ष): अध्ययन $\intcc{0}{\frac\pi4}$; $ r $
        से घट जाती है $1$ को $0$. पर $\theta = \pm\frac\pi4$: $ r =
        0$, विकर्णों के साथ मूल बिंदु पर स्पर्शरेखाएँ। वक्र है
        द $\infty $ प्रतीक: दो सममित लूप मिल रहे हैं $ O $,
        पहुँचना $(\pm 1, 0)$.
  \item साथ $ F, F' = (\pm c, 0)$, $ c = \frac{1}{\sqrt 2}$, और $ M
        = (r\cos\theta, r\sin\theta)$:
        \[
        MF^2\, MF'^2
        = \bigl((r\cos\theta - c)^2 + r^2\sin^2\theta\bigr)
        \bigl((r\cos\theta + c)^2 + r^2\sin^2\theta\bigr)
        = (r^2 + c^2)^2 - (2rc\cos\theta)^2 ,
        \]
        पहचान से $(A - B)(A + B) = A^2 - B^2$ साथ $ A = r^2
        + c^2$, $ B = 2rc\cos\theta $. साथ $ c^2 = \frac12$:
        \[
        MF^2 MF'^2 = \Bigl(r^2 + \frac12\Bigr)^2 - 2r^2\cos^2\theta
        = r^4 + r^2\bigl(1 - 2\cos^2\theta\bigr) + \frac14
        = r^4 - r^2\cos 2\theta + \frac14 .
        \]
        लेम्निस्केट पर, $ r^2 = \cos 2\theta $: पहले दो पद
        रद्द करना, जा रहा हूँ $ MF^2MF'^2 = \frac14$, यानी\ $ MF \cdot MF' =
        \frac12$. इसके विपरीत, गणना को पीछे की ओर पढ़ने से पता चलता है
        लोकस समीकरण $ MF\,MF' = \frac12$ है $ r^2 = \cos
        2\theta $ (के लिए $ r \neq 0$; और $ O $ दोनों को संतुष्ट करता है)।
\end{enumerate}
\end{solution}

\begin{solution}{exo:b1:curves:10}
$ r = 1 + \cos\theta $, $ r' = -\sin\theta $:
\[
r^2 + r'^2 = 1 + 2\cos\theta + \cos^2\theta + \sin^2\theta
= 2 + 2\cos\theta = 4\cos^2\frac\theta2 ,
\]
इसलिए $\sqrt{r^2 + r'^2} = 2\abs{\cos\frac\theta2}$ और, द्वारा
में \omterm{def:b1:linmaps:projection}{समरूपता} $ x $-अक्ष,
\[
L = \int_0^{2\pi} 2\Bigl|\cos\frac\theta2\Bigr|\,\dd\theta
= 2\int_0^{\pi} 2\cos\frac\theta2\,\dd\theta
= 2\Bigl[4\sin\frac\theta2\Bigr]_0^{\pi} = 8 .
\]
एक और \omterm{pb:b1:logic:1}{बीजगणितीय}, $\pi $-मुक्त परिधि.
\end{solution}

\begin{solution}{exo:b1:curves:11}
बात $(a\cos t, b\sin t)$ समीकरण को संतुष्ट करता है:
$\cos^2 t + \sin^2 t = 1$. रेखा का सामान्य सदिश है
$\bigl(\frac{\cos t}a, \frac{\sin t}b\bigr)$, और इसका उत्पाद
वेग के साथ $(-a\sin t, b\cos t)$ है $-\sin t\cos t +
\sin t\cos t = 0$: रेखा बिंदु से होकर गुजरती है
स्पर्शरेखा दिशा --- यह स्पर्शरेखा है। के लिए $(x_0, y_0)$ पर
दीर्घवृत्त, लिखें $\cos t = \frac{x_0}a $, $\sin t = \frac{y_0}b $ और
स्थानापन्न:
\[
\frac{x\,x_0}{a^2} + \frac{y\,y_0}{b^2} = 1 ,
\]
``विभाजन'' द्वारा दीर्घवृत्त समीकरण से प्राप्त किया गया $ x^2 \mapsto
x\,x_0$ और $ y^2 \mapsto y\,y_0$.
\end{solution}

\begin{solution}{exo:b1:curves:12}
\begin{enumerate}
  \item $ r' = k\eu^{k\theta}$, इसलिए $\tan V = \frac{r}{r'} =

        से गुणा करता है $\eu^{\iu c}$:

        \eu^{k\theta}\eu^{\iu(\theta + c)}
        = \eu^{-kc}\,\eu^{k(\theta + c)}\eu^{\iu(\theta+c)} ,

        \eu^{kA}\bigr)$, और के रूप में $ A \to -\infty $:

        L = \frac{\sqrt{1 + k^2}}{k}\,\eu^{k\theta_0} < \infty :

\end{enumerate}
\end{solution}

\begin{solution}{pb:b1:curves:1}
\textbf{1.} बिना फिसले लुढ़कने का अर्थ है संपर्क बिंदु
पहिए के चाप के बराबर दूरी तय की गई, जो अनियंत्रित हुई: के बाद
द्वारा मुड़ना $ t $, केंद्र पर है $(Rt, R)$. चिन्हित बिंदु बैठता है
कोण पर रिम पर $ t $ \emph{पीछे} नीचे की ओर लंबवत
(आगे बढ़ते समय पहिया दक्षिणावर्त घूमता है):
\[
M(t) = \Omega(t) + R(-\sin t, -\cos t)
= \bigl(R(t - \sin t),\ R(1 - \cos t)\bigr),
\]
जो कि सही है $ t = 0$ ($ M = (0,0)$) और कम से $ t = \pi $ ($ M =
(\pi R, 2R)$, शीर्ष)।

\textbf{2.} $ f'(t) = R(1 - \cos t,\ \sin t)$ और
\[
\norm{f'}^2 = R^2\bigl((1-\cos t)^2 + \sin^2 t\bigr)
= 2R^2(1 - \cos t) = 4R^2\sin^2\frac t2 ,
\]
इसलिए $\norm{f'} = 2R\abs{\sin\frac t2}$. पर एकवचन बिंदु $ t \in
2\pi\Z $: द cusps \omterm{thm:b1:reals:floor}{ज़मीन} पर (\cref{exo:b1:curves:2}); द
मेहराब का शीर्ष है $ t = \pi $, जहां गति $2R $ अधिकतम है.

\textbf{3.} अर्ध-कोण गुणनखंडन:
\[
f'(t) = 2R\sin\frac t2\,\Bigl(\sin\frac t2,\ \cos\frac t2\Bigr),
\quad
M - C = 2R\sin\frac t2\,\Bigl(-\cos\frac t2,\ \sin\frac t2\Bigr),
\]
\[
T - M = \bigl(R\sin t,\ R(1 + \cos t)\bigr)
= 2R\cos\frac t2\,\Bigl(\sin\frac t2,\ \cos\frac t2\Bigr).
\]
के लिए $0 < t < 2\pi $, $\sin\frac t2 \neq 0$: $ M - C $ है ओर्थोगोनल
स्पर्शरेखा दिशा की ओर $\bigl(\sin\frac t2, \cos\frac
t2\bigr)$ (उनका उत्पाद है $-\sin\frac t2\cos\frac t2 +
\sin\frac t2\cos\frac t2 = 0$), और $ T - M $ इसके \omterm{thm:b1:matrices:conjugation}{समान}ांतर है.
पहिये के शीर्ष तक का तार \emph{है} स्पर्शरेखा; द
संपर्क बिंदु पर तार सामान्य है।

\textbf{4.} प्रत्येक क्षण में पहिया अपने संपर्क के चारों ओर घूमता है
बिंदु (उस बिंदु का वेग शून्य है: बिना फिसले लुढ़कना),
इसलिए पहिये का प्रत्येक कठोर बिंदु गति करता है ओर्थोगोनली लाइन तक
इससे जुड़ना $ C $, गति (कोणीय वेग) के साथ $1$) के बराबर
वह दूरी. जाँच करना: $\norm{M - C} = 2R\abs{\sin\frac t2} =
\norm{f'(t)}$.

\textbf{5.} $2R\,y(t) = 2R\cdot 2R\sin^2\frac t2 =
4R^2\sin^2\frac t2 = \norm{f'(t)}^2$. ऊंचाई पर गति $ y $ है
$\sqrt{2R\,y}$ --- औपचारिक रूप से कानून $ v = \sqrt{2gh}$ का \omterm{def:b1:vspaces:free}{मुक्त}
गिरना, \omterm{thm:b1:reals:floor}{ज़मीन} के छत से टकराने के साथ; भाग IV इसे बदल देता है
घड़ी की कल में अवलोकन.

\textbf{6.} द्वारा \cref{def:b1:curves:arclength} और प्रश्न 2
($\sin\frac t2 \geq 0$ पर $\intcc{0}{2\pi}$):
\[
L = \int_0^{2\pi} 2R\sin\frac t2\,\dd t
= 2R\Bigl[-2\cos\frac t2\Bigr]_0^{2\pi} = 2R(2 + 2) = 8R .
\]
व्रेन का प्रमेय: बिल्कुल चार व्यास।

\textbf{7.} $ s(t) = \int_0^t 2R\sin\frac u2\,\dd u =
4R\bigl(1 - \cos\frac t2\bigr)$; $ s(2\pi) = 4R(1+1) = 8R $,
सुसंगत।

\textbf{8.} $\sigma(t) = \abs{s(t) - s(\pi)} = \abs{4R(1 -
\cos\frac t2) - 4R} = 4R\abs{\cos\frac t2}$, इसलिए $\sigma^2 =
16R^2\cos^2\frac t2$; और $2R - y = R(1 + \cos t) =
2R\cos^2\frac t2$, कहाँ से $8R(2R - y) = 16R^2\cos^2\frac t2 =
\sigma^2$.

\textbf{9.} पर \omterm{ex:b1:curves:astroid}{उभार} $ t = 0$ (या $2\pi $): $\sigma = 4R $, का आधा $8R $: शीर्ष भाग मेहराब को आधा कर देता है। दोनों गणनाओं में
गति है $\abs{\text{trigonometric polynomial in } t/2}$, किसका
प्रतिअवकलन फिर से त्रिकोणमितीय है: द \omterm{def:b1:curves:arclength}{लंबाई} एक है
का अंतर \emph{मान} कोसाइन की --- परिमेय संख्याएँ
बार $ R $ --- मापने के लिए वृत्त का कोई चाप नहीं है, इसलिए नहीं $\pi $.
वृत्त में स्वयं स्थिर गति है, इसलिए यह है \omterm{def:b1:curves:arclength}{लंबाई} \omterm{thm:b1:integration:def}{अभिन्न}
पूरा उत्पादन करता है \omterm{prop:b1:reals:intervals}{अंतराल} \omterm{def:b1:curves:arclength}{लंबाई} $2\pi $; चक्रवात की गति
सिरों पर लुप्त हो जाता है और \omterm{pb:b1:logic:1}{बीजगणितीय} रूप से एकीकृत हो जाता है।

\textbf{10.} मेहराब के साथ बह गया है $ x $ से बढ़ रहा है $0$ को
$2\pi R $ ($ x' = R(1 - \cos t) \geq 0$, केवल पर लुप्त हो रहा है
पृथक बिंदु)। स्थानापन्न $ x = x(t)$ क्षेत्र में \omterm{thm:b1:integration:def}{अभिन्न}
$\int_0^{2\pi R} y\,\dd x $ (\cref{thm:b1:integration:parts})
देता है $ A = \int_0^{2\pi} y(t)\,x'(t)\,\dd t $.

\textbf{11.}
\[
A = \int_0^{2\pi} R(1 - \cos t)\cdot R(1 - \cos t)\,\dd t
= R^2\int_0^{2\pi}\bigl(1 - 2\cos t + \cos^2 t\bigr)\dd t
= R^2\Bigl(2\pi - 0 + \pi\Bigr) = 3\pi R^2 ,
\]
का उपयोग करते हुए $\int_0^{2\pi}\cos^2 = \pi $. बिल्कुल तीन पहिया क्षेत्र:
गैलीलियो के संतुलन ने कहा ``के बारे में $3$''; रोबर्वल की गणना
``बिल्कुल'' कहता है।

\textbf{12.} साथ $ x = \cos^3 t $, $ y = \sin^3 t $: $ y\,x' =
-3\sin^4 t\cos^2 t $. रैखिक बनाना:
\[
\sin^4 t\cos^2 t = \frac{1 - \cos 2t}{2}\cdot\frac{\sin^2
2t}{4} = \frac{\sin^2 2t}{8} - \frac{\sin^2 2t\cos 2t}{8} ,
\]
और खत्म होता है $\intcc{0}{2\pi}$: $\int\sin^2 2t = \pi $, $\int \sin^2
2t\cos 2t = \bigl[\frac{\sin^3 2t}{6}\bigr] = 0$. इसलिए $\oint
y\,\dd x = -3\cdot\frac\pi8$, और संलग्न क्षेत्र है
$\frac{3\pi}8$ (चिह्न वामावर्त रिकॉर्ड करता है
ओरिएंटेशन)।

\textbf{13.} के लिए $(\cos t, \sin t)$ साथ $ t $ से $\pi $ को $0$,
$ x $ से बढ़ता है $-1$ को $1$ और
\[
\int_\pi^0 \sin t\cdot(-\sin t)\,\dd t
= \int_0^\pi \sin^2 t\,\dd t = \frac\pi2 :
\]
सूत्र ऊपरी आधे डिस्क का क्षेत्रफल लौटाता है, जैसा कि यह होना चाहिए।

\textbf{14.} $ x' = R(1 + \cos t) = 2R\cos^2\frac t2$, $ y' =
R\sin t = 2R\sin\frac t2\cos\frac t2$, इसलिए $\norm{f'} =
2R\cos\frac t2$ (पर गैरनकारात्मक $\intcc{-\pi}{\pi}$). आर्क से
निचला: $ s(t) = \int_0^t 2R\cos\frac u2\,\dd u =
4R\sin\frac t2$. तब
\[
y = R(1 - \cos t) = 2R\sin^2\frac t2
= 2R\Bigl(\frac{s}{4R}\Bigr)^{\!2} = \frac{s^2}{8R} .
\]

\textbf{15.} ऊर्जा संरक्षण के साथ $ v = \frac{\dd s}{\dd\tau}$
और $ y = \frac{s^2}{8R}$, $ y_0 = \frac{s_0^2}{8R}$:
\[
\Bigl(\frac{\dd s}{\dd\tau}\Bigr)^{\!2} = 2g(y_0 - y)
= \frac{2g}{8R}\bigl(s_0^2 - s^2\bigr)
= \frac{g}{4R}\bigl(s_0^2 - s^2\bigr).
\]

\textbf{16.} में भेद करना $\tau $: $2s's'' =
-\frac{g}{4R}\,2ss'$, तो कहीं भी $ s' \neq 0$ (इसलिए हर जगह
द्वारा निरंतरता), $ s'' = -\frac{g}{4R}s $: हार्मोनिक थरथरानवाला.
द्वारा \cref{thm:b1:diffeq:homogeneous2}, $ s(\tau) =
A\cos\omega\tau + B\sin\omega\tau $ साथ $\omega =
\sqrt{g/(4R)}$; प्रारंभिक शर्तें $ s(0) = s_0$, $ s'(0) = 0$
देना $ s(\tau) = s_0\cos\omega\tau $.

\textbf{17.} मनका जब नीचे पहुँचता है $ s = 0$, यानी\ पर
$\omega\tau = \frac\pi2$:
\[
T_{\downarrow} = \frac{\pi}{2\omega}
= \frac\pi2\sqrt{\frac{4R}{g}} = \pi\sqrt{\frac Rg}\,,
\]
से स्वतंत्र $ s_0$: कटोरे में कहीं से भी जारी, सभी
मोती एक ही पल में आ जाते हैं --- टॉटोक्रोन।

\textbf{18.} स्थानापन्न $ s = s_0\cos\omega\tau $: बाईं ओर
है $ s_0^2\omega^2\sin^2\omega\tau $ और दाहिनी ओर है
$\frac{g}{4R}s_0^2(1 - \cos^2\omega\tau) =
s_0^2\omega^2\sin^2\omega\tau $: सटीक समानता, इसलिए कोई नकली नहीं
समाधान पेश किया गया। एक गोलाकार चाप के लिए, $ y $ नहीं है
के आनुपातिक $ s^2$ ($ y = \ell(1 - \cos\frac s\ell) =
\frac{s^2}{2\ell} - \frac{s^4}{24\ell^3} + \dots $): द
पुनर्स्थापना \omterm{def:b1:vspaces:span}{अवधि} ही है \emph{लगभग} \omterm{def:b1:linmaps:def}{रेखीय}, तो
एक गोलाकार पेंडुलम की \omterm{def:b1:vspaces:span}{अवधि} आयाम के साथ बहती है, जबकि
साइक्लोइड सख्ती से स्थिर है।

\textbf{19.} $\pi\sqrt{R/g} = 1$ देता है $ R = \frac{g}{\pi^2}
\approx \frac{9.81}{9.87} \approx 0.994$ मी --- लगभग बिल्कुल
एक मीटर. पूरा दोलन लेता है $\frac{2\pi}\omega =
2\pi\sqrt{4R/g}$, जो सटीक रूप से लघु-कोण सूत्र है
$2\pi\sqrt{\ell/g}$ के एक पेंडुलम के लिए \omterm{def:b1:curves:arclength}{लंबाई} $\ell = 4R $:
ह्यूजेन्स ने अपने पेंडुलम को बिल्कुल साइक्लोइडल गालों से निलंबित कर दिया
वह अनुपात.

\textbf{20.} पर $ t = \frac\pi2$, $ R = 1$: $ M = \bigl(\frac\pi2 -
1,\ 1\bigr)$, $ T = \bigl(\frac\pi2,\ 2\bigr)$, इसलिए $ T - M = (1,
1)$. और $ f'(\frac\pi2) = (1 - 0,\ 1) = (1, 1)$: को राग
जैसा कि वादा किया गया था, शीर्ष पर वेग है।

\textbf{21.} ट्रोचॉइड के लिए, $ x'(t) = R - d\cos t $ और $ y'(t)
= d\sin t $. अगर $ d < R $: $ x' \geq R - d > 0$, बात हमेशा
अग्रिम; वेग कभी लुप्त नहीं होता (इसका पहला घटक है
सकारात्मक): कोई एकवचन बिंदु नहीं, एक नियमित तरंग। अगर $ d > R $: $ x'(0)
= R - d < 0 < R + d = x'(\pi)$, इसलिए बिंदु पीछे की ओर चला जाता है
संपर्क क्षणों के पास और अन्यत्र आगे: वक्र
लूपों में स्वयं को पार करता है। रेलवे के किनारे पर एक बिंदु
पहिया (रेल हेड के नीचे, $ d > R $) हर समय पीछे की ओर यात्रा करता है
बारी.

\textbf{22.} से राग \omterm{ex:b1:curves:astroid}{उभार} $(\pi R, 2R)$ मूल की ओर
है \omterm{def:b1:curves:arclength}{लंबाई} $ L = R\sqrt{\pi^2 + 4}$ और ढलान कोण $\alpha $ साथ
$\sin\alpha = \frac{2R}{L}$. स्थिरांक के साथ आराम से फिसलना
त्वरण $ g\sin\alpha $: $ L = \frac12 g\sin\alpha\,T^2$, इसलिए
\[
T = \sqrt{\frac{2L}{g\sin\alpha}} = \sqrt{\frac{2L^2}{2Rg}}
= \frac{L}{\sqrt{Rg}} = \sqrt{\pi^2 + 4}\,\sqrt{\frac Rg}
\approx 3.72\sqrt{\frac Rg}\,,
\]
ख़िलाफ़ $\pi\sqrt{R/g} \approx 3.14\sqrt{R/g}$ चक्रवात के लिए:
घुमावदार रास्ता तेज़ है. यह वास्तव में सबसे तेज़ संभव है
--- ब्रैचिस्टोक्रोन --- कैलकुलस का एक प्रमेय
इस मात्रा से परे विविधताएँ।

\textbf{23.} $\dfrac{\dd y}{\dd x} = \dfrac{y'}{x'} =
\dfrac{\sin t}{1 - \cos t} = \cot\dfrac t2$ (आधा कोण
सूत्र). तब
\[
\frac{\dd^2y}{\dd x^2}
= \frac{\dd}{\dd t}\Bigl(\cot\frac t2\Bigr)\cdot\frac{1}{x'(t)}
= -\frac{1}{2\sin^2\frac t2}\cdot\frac{1}{2R\sin^2\frac t2}
= -\frac{1}{4R\sin^4\frac t2} < 0 :
\]
पूरे मेहराब में अवतल; जैसा $ t \to 0^+$ या $2\pi^-$ ढाल $\cot\frac t2 \to \pm\infty $: पर ऊर्ध्वाधर स्पर्श रेखाएँ
cusps.

\textbf{24.} तार दिशा है $ T - M \parallel
\bigl(\sin\frac t2, \cos\frac t2\bigr)$, जिसकी प्रवृत्ति होती है $(0, 1)$
जैसा $ t \to 0^{+}$: पर स्पर्शरेखा \omterm{ex:b1:curves:astroid}{उभार} लंबवत है ---
बिना किसी टेलर विस्तार के, शुद्ध ज्यामिति से पुनर्प्राप्त।

\textbf{25.} (i) व्रेन का प्रमेय, $ L = 8R $; रोबरवाल का क्षेत्र, $ A =
3\pi R^2$ (गैलीलियो का अनुमानित अनुपात $3$); ह्यूजेन्स'
टॉटोक्रोन, $ T_\downarrow = \pi\sqrt{R/g}$. (ii) प्रत्येक
गणना अर्ध-कोण गुणनखंडों पर चलती है $ f' =
2R\sin\frac t2(\sin\frac t2, \cos\frac t2)$ और उनका कटोरा
एनालॉग --- एक पहचान शक्ति स्पर्शरेखा, \omterm{def:b1:curves:arclength}{लंबाई} और घड़ी
एक जैसे (iii) आंतरिक संबंध $ y = s^2/8R $ परिवर्तित कर दिया
ऊर्जा समीकरण में $ s'' = -\frac{g}{4R}s$, ए \omterm{def:b1:linmaps:def}{रेखीय} समीकरण
स्थिर गुणांकों के साथ जिनके समाधान बिल्कुल हैं
समकालिक। (iv) भाग I में डिफरेंशियल कैलकुलस का उपयोग किया गया
\cref{ch:b1:derivative}, भाग II--III \omterm{thm:b1:integration:def}{अभिन्न} का
\cref{ch:b1:integration}, भाग IV के अंतर समीकरण
\cref{ch:b1:diffeq} --- साइक्लोइड इस पुस्तक का पाठ्यक्रम है
एक वक्र में लुढ़का हुआ।
\end{solution}
