\section*{अध्याय \ref{ch:b1:arith} --- पूर्णांक अंकगणित}

\begin{solution}{exo:b1:arith:1}
$1001 = 1 \times 777 + 224$; $777 = 3 \times 224 + 105$; $224 = 2
\times 105 + 14$; $105 = 7 \times 14 + 7$; $14 = 2 \times 7 + 0$. इसलिए
$\gcd(1001, 777) = 7$. पीछे की ओर:
\[
7 = 105 - 7 \times 14
= 105 - 7(224 - 2\times 105) = 15 \times 105 - 7 \times 224
\]
\[
= 15(777 - 3\times 224) - 7\times 224 = 15 \times 777 - 52 \times 224
= 15 \times 777 - 52(1001 - 777) = 67 \times 777 - 52 \times 1001 .
\]
जाँच करना: $67 \times 777 = 52\,059$ और $52 \times 1001 = 52\,052$;
अंतर $7$. बेज़आउट जोड़ी: $(u, v) = (-52, 67)$ के लिए $1001u + 777v =
7$.
\end{solution}

\begin{solution}{exo:b1:arith:2}
तब से $10 \equiv 1 \pmod 9$: $10^k \equiv 1$, इसलिए $\sum_k d_k 10^k
\equiv \sum_k d_k \pmod 9$. तब से $10 \equiv -1 \pmod{11}$: $10^k
\equiv (-1)^k $, इसलिए पूर्णांक प्रत्यावर्ती योग के सर्वांगसम है
$\sum_k (-1)^k d_k $ आधुनिक $11$ (से शुरू हो रहा है \emph{इकाइयां} अंक
संकेत के साथ $+$).

$123\,456\,789$: अंक योग $45 \equiv 0 \pmod 9$. से वैकल्पिक योग
इकाइयाँ: $9 - 8 + 7 - 6 + 5 - 4 + 3 - 2 + 1 = 5$, तो संख्या है
$\equiv 5 \pmod{11}$.
\end{solution}

\begin{solution}{exo:b1:arith:3}
यूक्लिड: $237 = 2 \times 91 + 55$; $91 = 1 \times 55 + 36$; $55 = 1
\times 36 + 19$; $36 = 1 \times 19 + 17$; $19 = 1 \times 17 + 2$; $17
= 8 \times 2 + 1$. पीछे की ओर:
\[
1 = 17 - 8\times 2 = 17 - 8(19 - 17) = 9\times 17 - 8\times 19
= 9(36 - 19) - 8\times 19 = 9\times 36 - 17\times 19
\]
\[
= 9\times 36 - 17(55 - 36) = 26\times 36 - 17\times 55
= 26(91 - 55) - 17\times 55 = 26\times 91 - 43\times 55
\]
\[
= 26\times 91 - 43(237 - 2\times 91) = 112 \times 91 - 43 \times 237.
\]
इसलिए $91 \times 112 \equiv 1 \pmod{237}$: समाधान हैं $ x \equiv
112 \pmod{237}$. (जाँच करना: $91 \times 112 = 10\,192 = 43 \times 237 +
1$.)
\end{solution}

\begin{solution}{exo:b1:arith:4}
$\gcd(17, 39) = 1$: यूक्लिड देता है $39 = 2\times 17 + 5$, $17 = 3\times
5 + 2$, $5 = 2\times 2 + 1$, और पीछे की ओर
\[
1 = 5 - 2\times 2 = 5 - 2(17 - 3\times 5) = 7\times 5 - 2\times 17
= 7(39 - 2\times 17) - 2\times 17 = 7\times 39 - 16\times 17 .
\]
विशेष समाधान $(x_0, y_0) = (-16, 7)$. का सामान्य समाधान
सजातीय समीकरण $17x + 39y = 0$: $ x = 39k $, $ y = -17k $ (तब से
$17 \mid 39y $ और $\gcd(17,39) = 1$ बल $17 \mid y $ ---
\omterm{thm:b1:arith:gauss}{गॉस की लेम्मा})। इस तरह
\[
(x, y) = (-16 + 39k,\; 7 - 17k), \qquad k \in \Z .
\]
दाहिनी ओर के लिए $5$, विशेष समाधान को इससे गुणा करें $5$:
$(x, y) = (-80 + 39k,\; 35 - 17k)$, $ k \in \Z $.
\end{solution}

\begin{solution}{exo:b1:arith:5}
हरएक के लिए \omterm{def:b1:arith:prime}{मुख्य} $ p $, साथ $\alpha = v_p(a)$ और $\beta = v_p(b)$:
\[
v_p\bigl(\gcd(a,b)\bigr) + v_p\bigl(\operatorname{lcm}(a,b)\bigr)
= \min(\alpha, \beta) + \max(\alpha, \beta)
= \alpha + \beta = v_p(ab) .
\]
प्रत्येक पर समान मूल्यांकन वाले दो धनात्मक पूर्णांक \omterm{def:b1:arith:prime}{मुख्य} बराबर हैं
(\cref{prop:b1:arith:valuation}), इसलिए $\gcd(a,b)\operatorname{lcm}(a,b)
= ab $.
\end{solution}

\begin{solution}{exo:b1:arith:6}
मूल्यांकन: $\gcd(a, b) = 2^{\min(10,6)} 3^{\min(4,7)} 5^{\min(2,0)}
7^{\min(0,1)} = 2^6\, 3^4 = 5184$;
$\operatorname{lcm}(a,b) = 2^{10}\, 3^7\, 5^2\, 7$.

भाजक गणना: का एक सकारात्मक भाजक $ n = \prod p_i^{\alpha_i}$ है
बिल्कुल एक विकल्प $\prod p_i^{\beta_i}$ साथ $0 \leq \beta_i \leq
\alpha_i $ (\cref{prop:b1:arith:valuation}); विकल्प हैं
स्वतंत्र, इसलिए हैं $\prod_i (\alpha_i + 1)$ विभाजक. के लिए $ a $:
$(10+1)(4+1)(2+1) = 165$.
\end{solution}

\begin{solution}{exo:b1:arith:7}
कल्पना करना $\sqrt p = \frac rq $ साथ $ r, q \in \N^*$, यानी\ $ p q^2 =
r^2$. आवेदन करना $ v_p $: $ v_p(pq^2) = 1 + 2v_p(q)$ अजीब है, जबकि $ v_p(r^2)
= 2 v_p(r)$ सम है. एक पूर्णांक में एक विषम और एक सम संख्या नहीं हो सकती
$ p $-एक बार में मूल्यांकन: विरोधाभास. इसलिए $\sqrt p \notin \Q $.
\end{solution}

\begin{solution}{exo:b1:arith:8}
\emph{सामान्य तथ्य.} साथ $\gcd(m,n) = 1$, बेज़आउट देता है $ mu + nv = 1$.
\omterm{def:b1:logic:sets}{तय करना} $ x_0 = b\,mu + a\,nv $. तब $ x_0 \equiv a\,nv \equiv a(1 - mu)
\equiv a \pmod m $ और इसी तरह $ x_0 \equiv b \pmod n $: अस्तित्व। अगर
$ x $ और $ x'$ दो समाधान हैं, $ m $ और $ n $ विभाजित करना $ x - x'$, इसलिए
$ mn \mid x - x'$ (\cref{thm:b1:arith:gauss}~(2)): विशिष्टता मॉड
$ mn $.

\emph{संख्यात्मक रूप से:} $ m = 7$, $ n = 11$: $7 \times (-3) + 11 \times 2 =
1$. इसलिए $ x_0 = 5 \times 7 \times (-3) + 2 \times 11 \times 2 = -105 +
44 = -61 \equiv 16 \pmod{77}$. जाँच करना: $16 = 2\times 7 + 2 \equiv 2
\pmod 7$; $16 = 11 + 5 \equiv 5 \pmod{11}$. समाधान: $ x \equiv 16
\pmod{77}$.
\end{solution}

\begin{solution}{exo:b1:arith:9}
आधुनिक $7$: फर्मेट देता है $3^6 \equiv 1$, और $1000 = 6 \times 166 + 4$,
इसलिए $3^{1000} \equiv 3^4 = 81 \equiv 4 \pmod 7$.

के अंतिम दो अंक $7^{100}$: कार्य मॉड $4$ और मॉड $25$. आधुनिक $4$: $7
\equiv -1$, इसलिए $7^{100} \equiv 1$. आधुनिक $25$: $7^2 = 49 \equiv -1$, इसलिए
$7^4 \equiv 1$ और $7^{100} = (7^4)^{25} \equiv 1$. चीनियों द्वारा
शेषफल प्रमेय (\cref{exo:b1:arith:8}), $7^{100} \equiv 1
\pmod{100}$: अंतिम दो अंक हैं $01$.
\end{solution}

\begin{solution}{exo:b1:arith:10}
लिखना $ m = nq + r $, $0 \leq r < n $. तब
\[
2^m - 1 = 2^r\bigl(2^{nq} - 1\bigr) + 2^r - 1,
\]
और $2^n - 1$ विभाजित $2^{nq} - 1 = (2^n - 1)(2^{n(q-1)} + \dots +
1)$. तो मॉड $2^n - 1$, $\;2^m - 1 \equiv 2^r - 1$, और तबसे $0 \leq
2^r - 1 < 2^n - 1$, यह \emph{है} यूक्लिडियन शेष.

इसलिए \omterm{met:b1:arith:euclid}{यूक्लिडियन एल्गोरिथ्म} जोड़ी पर $(2^m - 1, 2^n - 1)$
दर्पण, प्रतिपादक द्वारा प्रतिपादक, एल्गोरिथ्म पर $(m, n)$: प्रत्येक
विभाजन चरण प्रतिस्थापित करता है $(m, n)$ द्वारा $(n, r)$ ऊपर और $(2^m - 1,
2^n - 1)$ द्वारा $(2^n - 1, 2^r - 1)$ नीचे. एल्गोरिथ्म ऊपर
पर समाप्त होता है $\gcd(m,n)$, इसलिए नीचे यह समाप्त होता है
$2^{\gcd(m,n)} - 1$.
\end{solution}

\begin{solution}{exo:b1:arith:11}
पहले समाधान करें $ x^2 \equiv 1 \pmod p $: $ p \mid (x-1)(x+1)$, तो द्वारा
यूक्लिड की प्रमेयिका $ x \equiv 1$ या $ x \equiv -1 \pmod p $.

उत्पाद में $(p-1)! = 1 \times 2 \times \dots \times (p-1)$, हर
कारक $ a $ उलटा मॉड है $ p $, और इसका उलटा $ a^{-1}$ फिर से है
कारकों में से एक (\cref{prop:b1:arith:invmod}). प्रत्येक को जोड़े $ a $ साथ
$ a^{-1}$: जोड़े को गुणा किया जाता है $1$, स्व-युग्मित कारकों को छोड़कर
($ a = a^{-1}$, यानी\ $ a^2 \equiv 1$) अकेले खड़े रहें --- और ये हैं
बिलकुल $1$ और $ p - 1$. इस तरह
\[
(p-1)! \equiv 1 \times (p - 1) \equiv -1 \pmod p .
\]
(के लिए $ p = 2$: $1! = 1 \equiv -1 \pmod 2$; युग्म तर्क
पतित हो जाता है लेकिन परिणाम कायम रहता है।)

\emph{बातचीत.} होने देना $ n \geq 2$ समग्र हो, $ n = ab $ साथ $1 < a
\leq b < n $. अगर $ a < b $, दोनों अलग-अलग कारकों के रूप में दिखाई देते हैं $(n-1)!$,
इसलिए $ n \mid (n-1)!$ और $(n-1)! \equiv 0 \not\equiv -1$. अगर $ a = b $
(अर्थात्\ $ n = a^2$): के लिए $ a \geq 3$, दोनों $ a $ और $2a $ हैं $< n $, इसलिए
$ n = a^2 \mid a \times 2a \mid (n-1)!$, वही निष्कर्ष; के लिए $ n = 4$,
$(n-1)! = 6 \equiv 2 \not\equiv -1 \pmod 4$.
\end{solution}

\begin{solution}{exo:b1:arith:12}
\begin{enumerate}
  \item प्रेरण। के लिए $ n = 1$: $ F_0 = 3 = F_1 - 2 = 5 - 2$.
        यह मानते हुए $ F_0\cdots F_{n-1} = F_n - 2$:
        \[
        F_0 \cdots F_n = (F_n - 2)F_n
        = \bigl(2^{2^n} - 1\bigr)\bigl(2^{2^n} + 1\bigr)
        = 2^{2^{n+1}} - 1 = F_{n+1} - 2 .
        \]
  \item होने देना $ m < n $ और $ d = \gcd(F_m, F_n)$. द्वारा (1), $ F_m $
        विभाजित $ F_n - 2$, इसलिए $ d $ विभाजित दोनों $ F_n $ और $ F_n -
        2$, इस तरह विभाजित $2$. लेकिन प्रत्येक फ़र्मेट संख्या विषम है, इसलिए
        $ d = 1$.
  \item प्रत्येक $ F_n \geq 3$ एक \omterm{def:b1:arith:prime}{मुख्य} भाजक $ p_n $
        (\cref{thm:b1:arith:euclidprimes}का पहला कदम)। अगर $ m
        \neq n $, तब $ p_m \neq p_n $, एक आम के बाद से \omterm{def:b1:arith:prime}{मुख्य} होगा
        बाँटना $\gcd(F_m, F_n) = 1$. \omterm{def:b1:logic:map}{नक्शा} $ n \mapsto p_n $ इसलिए \omterm{def:b1:logic:inj}{इंजेक्शन} से $\N $ में अभाज्य: वहाँ हैं
        असीम रूप से अनेक अभाज्य.
\end{enumerate}
\end{solution}

\begin{solution}{pb:b1:arith:1}
\textbf{1.} $10! = 3\,628\,800$: दो अनुगामी शून्य. मूल्यांकन
कारक दर कारक: की शक्तियाँ $2$ से आते हैं $2, 4 = 2^2, 6, 8 = 2^3,
10$, कुल मिलाकर $ v_2(10!) = 1 + 2 + 1 + 3 + 1 = 8$; की शक्तियां $5$
से $5$ और $10$: $ v_5(10!) = 2$. अनुगामी शून्य $=
\min(v_2, v_5) = 2$, सुसंगत।

\textbf{2.} यूक्लिडियन प्रभाग लिखिए $\lfloor x\rfloor = nq +
r $, $0 \leq r \leq n - 1$. तब $ x = nq + r + \{x\}$ साथ $0 \leq
r + \{x\} < n $, इसलिए $\lfloor x/n \rfloor = q = \bigl\lfloor \lfloor
x \rfloor / n \bigr\rfloor $.

\textbf{3.} होने देना $\alpha = v_p(a) \leq \beta = v_p(b)$ (यदि स्वैप करें
आवश्यकता है) और लिखें $ a = p^\alpha a'$, $ b = p^\beta b'$ साथ $ p
\nmid a', b'$. तब $ a + b = p^\alpha\bigl(a' + p^{\beta -
\alpha}b'\bigr)$, इसलिए $ v_p(a + b) \geq \alpha = \min $. अगर $\alpha <
\beta $, कोष्ठक है $ a' + p^{\beta-\alpha}b' \equiv a'
\not\equiv 0 \pmod p $: मूल्यांकन बिल्कुल सही है $\alpha $.

\textbf{4.} के गुणज $ m $ में $\intint1n $ हैं $ m, 2m, \dots,
qm $ कहाँ $ q $ के साथ सबसे बड़ा पूर्णांक है $ qm \leq n $, यानी\ $ q =
\lfloor n/m \rfloor $.

\textbf{5.} अद्वितीय गुणनखंडन द्वारा, $ v_p(n!) = \sum_{j=1}^{n}
v_p(j)$. अलग-अलग गिनें: प्रत्येक $ j $ योगदान $ v_p(j) =
\#\{k \geq 1 : p^k \mid j\}$, इसलिए
\[
v_p(n!) = \sum_{j=1}^n \#\{k : p^k \mid j\}
= \sum_{k\geq1} \#\{j \leq n : p^k \mid j\}
= \sum_{k\geq1} \Bigl\lfloor \frac n{p^k} \Bigr\rfloor
\]
प्रश्न 4 से --- \omterm{pb:b1:arith:1}{लीजेंड्रे का सूत्र}। योग परिमित है: शर्तें
के साथ $ p^k > n $ गायब होना।

\textbf{6.} $ v_5(1000!) = 200 + 40 + 8 + 1 = 249$ (द्वारा विभाजन
$5, 25, 125, 625$); $ v_2(1000!) = 500 + 250 + 125 + 62 + 31 + 15 +
7 + 3 + 1 = 994$. का अनुगामी शून्य $1000!$: प्रत्येक शून्य उपभोग करता है
एक $2$ और एक $5$, तो वहाँ हैं $\min(994, 249) = 249$ उनमें से.

\textbf{7.} साथ $ n = \sum_i a_ip^i $, प्रश्न 2 देता है $\lfloor
n/p^k \rfloor = \sum_{i \geq k} a_ip^{i-k}$ (आधार को छोटा करें-$ p $
विस्तार)। संक्षेप में कहना $ k \geq 1$ और दो परिमित का आदान-प्रदान
रकम:
\[
v_p(n!) = \sum_{i\geq1} a_i \sum_{k=1}^{i} p^{i-k}
= \sum_{i\geq0} a_i\,\frac{p^i - 1}{p - 1}
= \frac{n - s_p(n)}{p - 1} .
\]

\textbf{8.} के लिए $ p = 2$: $ v_2(n!) = n - s_2(n)$. तब से $ n \geq 1$
है $ s_2(n) \geq 1$, हमेशा $ v_2(n!) \leq n - 1 < n $: $2^n \nmid
n!$. और $ v_2(n!) = n - 1$ आईएफएफ $ s_2(n) = 1$ आईएफएफ $ n $ की एक शक्ति है
$2$.

\textbf{9.} $ n $ है $\lfloor \log_p n \rfloor + 1$ आधार-$ p $
अंक, प्रत्येक अधिकतम $ p - 1$, इसलिए $1 \leq s_p(n) \leq
(p-1)\bigl(\log_p(n) + 1\bigr)$. प्रश्न 7 में प्रतिस्थापित:
\[
\frac n{p-1} - \log_p(n) - 1 \;\leq\; v_p(n!) \;<\; \frac n{p-1},
\]
और से विभाजित करना $ n $: $\frac{v_p(n!)}n \to \frac1{p-1}$.

\textbf{10.} $ Z(n) - Z(n-1) = v_5(n!/(n-1)!) = v_5(n)$: गिनती
पीछे चल रहे शून्यों की संख्या छलाँग लगाती है $ v_5(n)$ के प्रत्येक गुणज पर $5$ और है
बीच में स्थिर. $ Z(24) = \lfloor24/5\rfloor = 4$ और $ Z(25) =
5 + 1 = 6$: पर $ n = 25$ गिनती उछल जाती है $4$ सीधे करने के लिए $6$
($ v_5(25) = 2$), और तबसे $ Z $ के साथ घट नहीं रहा है $ Z \leq 4$
पहले और $ Z \geq 6$ बाद में, मूल्य $5$ कभी प्राप्त नहीं होता: नहीं
फैक्टोरियल ठीक पाँच शून्य पर समाप्त होता है।

\textbf{11.} लिखना $ x = \lfloor x\rfloor + \{x\}$: $\lfloor x +
y\rfloor = \lfloor x\rfloor + \lfloor y\rfloor + \lfloor \{x\} +
\{y\}\rfloor $, और $0 \leq \{x\} + \{y\} < 2$ आखिरी मंजिल बनाता है
$0$ या $1$. फिर, लीजेंड्रे द्वारा तीन बार आवेदन किया गया,
\[
v_p\binom{m+n}m = v_p\bigl((m{+}n)!\bigr) - v_p(m!) - v_p(n!)
= \sum_{k\geq1}\Bigl(
\Bigl\lfloor\frac{m+n}{p^k}\Bigr\rfloor
- \Bigl\lfloor\frac{m}{p^k}\Bigr\rfloor
- \Bigl\lfloor\frac{n}{p^k}\Bigr\rfloor\Bigr),
\]
का एक सीमित योग $0$ रेत $1$ s (पहला दावा लागू करें $x =
m/p^k $, $ y = n/p^k $).

\textbf{12.} हल करना $ k \geq 1$ और लिखा $ m = p^km_1 + m_0$, $ n =
p^kn_1 + n_0$ साथ $0 \leq m_0, n_0 < p^k $ (यूक्लिडियन प्रभाग:
$ m_0$ से बनी संख्या है $ k $ के निम्न अंक $ m $). तब
\[
\Bigl\lfloor\frac{m+n}{p^k}\Bigr\rfloor
- \Bigl\lfloor\frac m{p^k}\Bigr\rfloor
- \Bigl\lfloor\frac n{p^k}\Bigr\rfloor
= \Bigl\lfloor\frac{m_0 + n_0}{p^k}\Bigr\rfloor ,
\]
जो है $1$ अगर $ m_0 + n_0 \geq p^k $ और $0$ अन्यथा। लेकिन $ m_0 +
n_0 \geq p^k $ सटीक रूप से कहता है कि जोड़ना $ k $ के निम्न अंक $ m $
और $ n $ स्थिति में बह जाता है $ k $ --- स्थिति में ले जाना $ k $
स्कूल की किताब जोड़ने के एल्गोरिदम में। संक्षेप में कहना $ k $:
$ v_p\binom{m+n}m $ आधार में कैरी की संख्या है-$ p $
जोड़ना $ m + n $. (कुमेर, 1852.)

\textbf{13.} कुम्हार लगाएं $ m = j $, $ n = p^k - j $, जोड़ $ p^k =
(1\underbrace{0\cdots0}_{k})_p $. होने देना $ a = v_p(j)$, तो आधार-$ p $
के अंक $ j $ पदों पर $0, \dots, a-1$ हैं $0$ और अंक
पद पर $ a $ शून्येतर है. के अंक $ p^k - j $ नीचे की स्थिति
$ a $ हैं $0$ बहुत ($ p^k - j = p^a(p^{k-a} - j/p^a)$). पद पर
$ a $, दो शून्येतर अंकों का योग अवश्य होना चाहिए $ p $ (परिणाम अंक $0$):
एक ले जाना; प्रत्येक पद पर $ a+1, \dots, k-1$, अंक प्लस
आने वाली कैरी राशि $ p $ (परिणाम अंक $0$ फिर से): ले जाना
प्रचारित करता है. कुल: $ k - a $ ले जाता है, इसलिए $ v_p\binom{p^k}j = k -
v_p(j)$. के लिए $ k = 1$: $ v_p\binom pj = 1$ के लिए $0 < j < p $, द
\omterm{def:b1:arith:divides}{भाजकत्व} में प्रयोग किया जाता है \cref{thm:b1:arith:fermat}.

\textbf{14.} डिजिटल फॉर्म (प्रश्न 7) द्वारा, का उपयोग करते हुए $ s_2(2n) =
s_2(n)$ (शून्य अंक जोड़कर):
\[
v_2\binom{2n}n = \bigl(2n - s_2(2n)\bigr) - 2\bigl(n -
s_2(n)\bigr) = 2s_2(n) - s_2(2n) = s_2(n) \geq 1 :
\]
$\binom{2n}n $ हमेशा सम होता है, और $ v_2 = 1$ (अर्थात्\ $\binom{2n}n
\equiv 2 \pmod 4$) बिल्कुल कब $ s_2(n) = 1$, यानी\ कब $ n $ एक है
की शक्ति $2$.

\textbf{15.} वेंडरमोंडे के साथ $ m = n = k = p $: $\binom{2p}p =
\sum_{j=0}^p \binom pj\binom p{p-j} = \sum_{j=0}^p \binom pj^2$.
के लिए $0 < j < p $, $ p \mid \binom pj $ (प्रश्न 13), अतः $\binom
pj^2 \equiv 0 \pmod p $; अंतिम शर्तें देते हैं $1 + 1$:
$\binom{2p}p \equiv 2 \pmod p $.

\textbf{16.} आधार $3$: $500 = 486 + 9 + 3 + 2$, अंक (कम से
उच्च) $(2, 1, 1, 0, 0, 2)$, इसलिए $ s_3(500) = 6$; और $1000 = 729 +
243 + 27 + 1$, अंक $(1, 0, 0, 1, 0, 1, 1)$, इसलिए $ s_3(1000) = 4$.
\emph{कुमेर:} जोड़ना $500 + 500$ आधार में $3$: पद $0$: $2 + 2 =
4$, अंक $1$ ढोना $1$; पद $1$: $1 + 1 + 1 = 3$, अंक $0$
ढोना $1$; पद $2$: $1 + 1 + 1 = 3$, अंक $0$ ढोना $1$;
पद $3$: $0 + 0 + 1 = 1$, कोई कैरी नहीं; पद $4$: $0$;
पद $5$: $2 + 2 = 4$, अंक $1$ ढोना $1$; पद $6$:
भूमि ले जाना: अंक $1$. चार वहन: $ v_3\binom{1000}{500} = 4$.
\emph{किंवदंती:} $ v_3(1000!) = \frac{1000 - 4}2 = 498$ और
$ v_3(500!) = \frac{500 - 6}2 = 247$, इसलिए $ v_3\binom{1000}{500} =
498 - 2\times247 = 4$. दोनों गणनाएँ सहमत हैं --- और
अतिरिक्त अंक $(1, 0, 0, 1, 0, 1, 1)$ पुन: पेश $1000$, जैसे वे
अवश्य.

\textbf{17.} कुमेर द्वारा ($ p = 2$, $ m = k $, $ n' = n - k $):
$\binom nk $ यदि जोड़ अजीब है $ k + (n - k)$ आधार में $2$ है
कोई कैरी नहीं, यदि प्रत्येक स्थिति में अंक संतुष्ट हों $ k_i + (n -
k)_i = n_i $; उस मामले में $ k_i \leq n_i $ सभी के लिए $ i $. इसके विपरीत,
यदि $ k_i \leq n_i $ सभी के लिए $ i $, फिर अंकों वाली संख्या $ n_i -
k_i $ है $ n - k $ और अतिरिक्त कैरी-फ़्री है. आधार में वही प्रमाण
$ p $: $ p \nmid \binom nk $ यदि हर आधार-$ p $ का अंक $ k $ पर है
का सबसे संगत अंक $ n $.

\textbf{18.} की गिनती $ k \in \intint0n $ जिनके अंक आज्ञापालन करते हैं
$ k_i \leq n_i $: का प्रत्येक अंक $ k $ के बीच स्वतंत्र रूप से चुना जाता है
$ n_i + 1$ मूल्य, देना $\prod_i (n_i + 1)$ विकल्प; आधार में $2$
यह है $2^{\#\{i : n_i = 1\}} = 2^{s_2(n)}$. पंक्ति $4 =
(100)_2$: $2^1 = 2$ अजीब प्रविष्टियाँ --- वास्तव में $1, 4, 6, 4, 1$ अजीब है
प्रविष्टियाँ केवल अंत में। पंक्ति $5 = (101)_2$: $2^2 = 4$ --- वास्तव में
$1, 5, 10, 10, 5, 1$.

\textbf{19.} सभी आंतरिक प्रविष्टियाँ भी $\iff $ पंक्ति बिल्कुल सही है
$2$ विषम प्रविष्टियाँ (दोनों सिरे हमेशा विषम होते हैं) $\iff 2^{s_2(n)} =
2 \iff s_2(n) = 1 \iff n $ की एक शक्ति है $2$.

\textbf{20.} प्रश्न 11 के योग में $ k $-वाँ पद लुप्त हो जाता है
जैसे ही $ p^k > m + n $ (तब तीनों मंजिलें बराबर हैं, वास्तव में
पहला है $0$ कब $ p^k > m+n $; प्रत्येक पद अधिक सरल है $0$).
अत: अधिक से अधिक $\lfloor \log_p(m+n)\rfloor $ प्रत्येक पद अशून्य है
लायक $1$: $ a = v_p\binom{m+n}m \leq \log_p(m+n)$, यानी\ $ p^a \leq
m + n $.

\textbf{21.} हरएक के लिए \omterm{def:b1:arith:prime}{मुख्य} $ p $, $ v_p\bigl(\operatorname{lcm}(1,
\dots, 2n)\bigr) = \lfloor\log_p(2n)\rfloor $ (की सबसे बड़ी शक्ति
$ p $ जो निम्न से अधिक नहीं है $2n $ के बीच दिखाई देता है $1, \dots, 2n $). प्रश्न 20
के साथ $ m = n $ देता है $ v_p\binom{2n}n \leq \lfloor\log_p(2n)\rfloor $
हरएक के लिए $ p $: द्वारा \cref{prop:b1:arith:valuation}, $\binom{2n}n
\mid \operatorname{lcm}(1, \dots, 2n)$. आकार के लिए: अनुपात
$\binom{2n}{k+1}/\binom{2n}k = \frac{2n-k}{k+1} \geq 1$ बिलकुल
के लिए $ k < n $, इसलिए केंद्रीय प्रविष्टि सबसे बड़ी है $2n + 1$
पंक्ति की प्रविष्टियाँ $2n $, कहाँ से $4^n = \sum_k \binom{2n}k \leq
(2n+1)\binom{2n}n $. संयोजन:
\[
\operatorname{lcm}(1, \dots, 2n) \geq \binom{2n}n \geq
\frac{4^n}{2n + 1} .
\]
अगर कुछ थे अभाज्य नीचे $2n $, एलसीएम यह नहीं हो सकता
बड़ा: एलसीएम की घातीय वृद्धि का एक मात्रात्मक निशान है
की प्रचुरता अभाज्य.

\textbf{22.} $ Z(n) = \sum_k\lfloor n/5^k\rfloor \approx \frac
n4$, इसलिए निकट लक्ष्य रखें $ n = 4 \times 2026 = 8104$: $ Z(8104) = 1620 +
324 + 64 + 12 + 2 = 2022$. के गुणकों द्वारा कदम बढ़ाएँ $5$: $ Z(8110)
= 2024$, $ Z(8115) = 2025$, और
\[
Z(8120) = 1624 + 324 + 64 + 12 + 2 = 2026 .
\]
तब से $ Z $ के गुणजों के बीच स्थिर है $5$ और $ Z(8119) =
Z(8115) = 2025$, सबसे छोटा $ n $ कम से कम के साथ $2026$ पीछे चल रहा है
शून्य है $ n = 8120$.

\textbf{23.} आधार $7$: $50 = 49 + 1$, अंक (निम्न से उच्च) $(1, 0,
1)$. जोड़ा जा रहा है $50 + 50$: पद $0$: $1 + 1 = 2 < 7$, कोई कैरी नहीं;
पद $1$: $0 + 0 = 0$; पद $2$: $1 + 1 = 2 < 7$, नहीं
ले जाना. कैरी-फ़्री, इसलिए कुमेर द्वारा $ v_7\binom{100}{50} = 0$: $7 \nmid
\binom{100}{50}$. लीजेंड्रे सहमत हैं: $ v_7(100!) = \lfloor 100/7
\rfloor + \lfloor 100/49 \rfloor = 14 + 2 = 16$ और $ v_7(50!) = 7
+ 1 = 8$, इसलिए $ v_7\binom{100}{50} = 16 - 2\times8 = 0$.

\textbf{24.} (i) अद्वितीय गुणनखंडन ही इसका आधार है
की परिभाषा $ v_p $ और इसकी संवेदनशीलता, इसलिए \omterm{pb:b1:arith:1}{लीजेंड्रे का सूत्र}
और हर \omterm{def:b1:arith:divides}{भाजकत्व} निष्कर्ष
(\cref{prop:b1:arith:valuation}). (ii) यूक्लिडियन विभाजन उत्पन्न हुआ
प्रश्न 2 की काट-छाँट पहचान और विभाजन $ m = p^km_1 +
m_0$ जो कैरी को अलग करता है (प्रश्न 12)। (iii) गिनती: द
का गुणज-$ m $ गिनती (प्रश्न 4), अंक-पसंद उत्पाद
(प्रश्न 18), और पंक्ति-योग बाउंड $4^n \leq
(2n+1)\binom{2n}n $ (प्रश्न 21) सभी हैं
\cref{ch:b1:counting}-शैली के तर्क।

\textbf{25.} लेजेन्ड्रे ``किसकी शक्ति'' को परिवर्तित करता है $ p $ विभाजित
$ n!$'' आधार में-$ p $ अंक अंकगणित; कुमेर संपीड़ित करता है
के लिए उत्तर \omterm{def:b1:counting:objects}{द्विपद गुणांक} एकल के वहन में
जोड़ --- \omterm{def:b1:arith:divides}{भाजकत्व}, प्रतीत होता है कि विशाल की एक वैश्विक संपत्ति है
संख्याओं को स्थानीय स्तर पर, अंक दर अंक पढ़ा जाता है। प्रश्न 16 है
प्रतिमान: हाथ से गणना की गई चार कैरीज़, सटीक निर्धारित करती हैं
की शक्ति $3$ सैकड़ों अंकों वाली एक संख्या में. और प्रश्न 21
गहरे पानी में विचारों के समान चक्र को दर्शाता है: एक
के लिए घातीय निचली सीमा $\operatorname{lcm}(1, \dots, 2n)$ है
की ओर पहला, पूरी तरह से प्रारंभिक कदम प्रधान संख्या प्रमेय,
जिसका प्रमाण इस खंड से कहीं परे है। संपूर्ण टूलकिट ---
विभाजन, जीसीडी, मूल्यांकन --- को बहुपदों के लिए दोबारा दोहराया जाता है
\cref{ch:b1:poly}, जहां एक अंक विस्तार का एनालॉग है
की शक्तियों में विस्तार $(X - a)$.
