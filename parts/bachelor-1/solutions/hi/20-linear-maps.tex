\begin{solution}{exo:b1:linmaps:2}

(ले रहा हूँ $ x = 2$), आयाम $1$.

रैंक--शून्यता: $\operatorname{rk} u = 3 - 1 = 2$. \omterm{def:b1:linmaps:kerim}{छवि}: द्वारा फैलाया गया
द इमेजिस विहित का \omterm{def:b1:vspaces:free}{आधार}, $ u(e_1) = (1,2,3)$, $ u(e_2) =

नहीं \omterm{def:b1:logic:inj}{इंजेक्शन} ($\ker \neq \{0\}$), नहीं \omterm{def:b1:logic:inj}{विशेषण} (\omterm{def:b1:linmaps:rank}{रैंक} $2 < 3$):

\end{solution}

\begin{solution}{exo:b1:linmaps:3}
\emph{गुठली:} $ P = P'$ बल $\deg P = \deg P'$ जब तक $ P = 0$;
लेकिन $\deg P' < \deg P $ के लिए $ P \neq 0$: इसलिए $\ker u = \{0\}$, और $ u $,
एक \omterm{def:b1:logic:inj}{इंजेक्शन} की एंडोमोर्फिज्म \omterm{def:b1:findim:def}{परिमित-आयामी} $\R_n[X]$, एक है
\omterm{def:b1:linmaps:projection}{समरूपता} (\cref{cor:b1:linmaps:samedim}).

\emph{श्लोक में:} होने देना $ v(P) = P + P' + P'' + \dots + P^{(n)}$ (ए
पर सीमित राशि $\R_n[X]$). तब
\[
v\bigl(u(P)\bigr) = \sum_{k=0}^{n} (P - P')^{(k)}
= \sum_{k=0}^{n} P^{(k)} - \sum_{k=0}^{n} P^{(k+1)}
= P - P^{(n+1)} = P ,
\]
दूरबीन, तब से $ P^{(n+1)} = 0$. इसलिए $ v = u^{-1}$.
\end{solution}

\begin{solution}{exo:b1:linmaps:5}
$ u^2 = 0$ मतलब $ u(u(x)) = 0$ सभी के लिए $ x $: प्रत्येक $ u(x)$ में निहित है
$\ker u $, यानी\ $\operatorname{im} u \subseteq \ker u $. फिर

\[
\dim E = \dim\ker u + \operatorname{rk} u \geq 2\operatorname{rk} u .
\]

\begin{solution}{exo:b1:linmaps:6}
$(pq)^2 = pqpq = ppqq = pq $ (कम्यूटेशन): ए अनुमान
(\cref{thm:b1:linmaps:projchar}).

\omterm{def:b1:linmaps:kerim}{छवि}: $\operatorname{im}(pq) \subseteq \operatorname{im} p $, और
$= \operatorname{im}(qp) \subseteq \operatorname{im} q $: निहित
चौराहे में. इसके विपरीत, यदि $ x \in \operatorname{im} p \cap
\operatorname{im} q $, तब $ p(x) = x $ और $ q(x) = x $ (निर्धारित बिंदु
की विशेषता बताएं \omterm{def:b1:linmaps:kerim}{छवि} एक का अनुमान), इसलिए $ pq(x) = x $: $ x \in
\operatorname{im}(pq)$.

गुठली: $\ker p \subseteq \ker(qp) = \ker(pq)$ और इसी तरह $\ker q
\subseteq \ker(pq)$: राशि निहित है. इसके विपरीत, चलो $ pq(x) =
0$, और लिखा
\[
x = \underbrace{q(x)}_{\in\, \ker p} +
\underbrace{(x - q(x))}_{\in\, \ker q} :
\]
पहला पद संतुष्ट करता है $ p(q(x)) = 0$, तो यह निहित है $\ker p $; द
दूसरा अंदर है $\ker q $ तब से $ q(x - q(x)) = q(x) - q^2(x) = 0$. इस तरह
$ x \in \ker p + \ker q $.
\end{solution}

\begin{solution}{exo:b1:linmaps:7}
पहले सामान्य समावेशन पर ध्यान दें $\ker u \subseteq \ker u^2$ और
$\operatorname{im} u^2 \subseteq \operatorname{im} u $, और द्वारा
पद--शून्यता, (2) $\iff $ (3) (बराबर \omterm{def:b1:linmaps:kerim}{कर्नेल} $\iff $ बराबर \omterm{def:b1:linmaps:rank}{रैंक}
$\iff $ बराबर इमेजिस, समावेशन को देखते हुए)।

(1 $\Rightarrow $ 2): चलो $ u^2(x) = 0$; तब $ u(x) \in \ker u \cap
\operatorname{im} u = \{0\}$, इसलिए $ x \in \ker u $.

(2 $\Rightarrow $ 1): ग्रासमैन और रैंक--शून्यता द्वारा, $\dim(\ker u +
\operatorname{im} u) = \dim\ker u + \operatorname{rk} u -
\dim(\ker u \cap \operatorname{im} u) = \dim E - \dim(\ker u \cap
\operatorname{im} u)$: योग है $ E $ यदि चौराहा है
$\{0\}$. होने देना $ y \in \ker u \cap \operatorname{im} u $: $ y = u(x)$ और
$ u(y) = 0$, इसलिए $ u^2(x) = 0$, तो (द्वारा (2)) $ u(x) = 0$: $ y = 0$. इस तरह
$ E = \ker u \oplus \operatorname{im} u $.
\end{solution}

\begin{solution}{exo:b1:linmaps:10}
\begin{enumerate}
  \item $ f \circ f = -\mathrm{id}$ है \omterm{def:b1:logic:inj}{द्विभाजित}, इसलिए $ f $ है
        (\cref{prop:b1:logic:comp} अनुकूलित: $ f $ दो तरफा है
        उलटा $-f $). अगर $ f(x) = \lambda x $ साथ $ x \neq 0$:
        आवेदन करना $ f $, $-x = \lambda^2 x $, इसलिए $\lambda^2 = -1$:
        में असंभव $\R $.
  \item लोभपूर्वक परिवार का निर्माण करें। लेना $ x_1 \neq 0$: $(x_1,
        f(x_1))$ है \omterm{def:b1:vspaces:free}{मुक्त} द्वारा (1). अगर $\operatorname{Vect}$ का
        वर्तमान परिवार $\bigl(x_1, f(x_1), \dots, x_k,
        f(x_k)\bigr)$, इसे कहते हैं $ V_k $ --- ए \omterm{def:b1:vspaces:subspace}{उपस्पेस} के अंतर्गत स्थिर $ f $
        (प्रत्येक जनरेटर एमएपीएस किसी अन्य जनरेटर या उसके नकारात्मक के लिए:
        $ f(f(x_i)) = -x_i $) --- सब कुछ नहीं है $ E $, चुनना $ x_{k+1}
        \notin V_k $. \emph{दावा: बढ़ा हुआ परिवार है \omterm{def:b1:vspaces:free}{मुक्त}.}
        कल्पना करना $\alpha x_{k+1} + \beta f(x_{k+1}) + v = 0$ साथ $ v
        \in V_k $ और $(\alpha, \beta) \neq (0,0)$. आवेदन करना $ f $:
        $\alpha f(x_{k+1}) - \beta x_{k+1} + f(v) = 0$ साथ $ f(v)
        \in V_k $. हटाना $ f(x_{k+1})$ दोनों रिश्तों के बीच
        (पहले को इससे गुणा करें $\alpha $, दूसरा द्वारा $-\beta $, और
        जोड़ें):
        \[
        (\alpha^2 + \beta^2)\, x_{k+1} \in V_k ,
        \]
        और $\alpha^2 + \beta^2 \neq 0$ बल $ x_{k+1} \in V_k $:
        विरोधाभास। इसलिए वेक्टर जोड़ते हुए निर्माण जारी है
        \emph{एक समय में दो}, जब तक $ V_k = E $:अंतिम परिवार एक है
        \omterm{def:b1:vspaces:free}{आधार} समान आकार का, और $ n $ सम है.
\end{enumerate}
\end{solution}

\begin{solution}{exo:b1:linmaps:11}
\omterm{def:b1:reals:bounds}{ऊपरी सीमा}: हरएक के लिए $ x $, $(u + v)(x) = u(x) + v(x) \in
\operatorname{im} u + \operatorname{im} v $, इसलिए
\[
\operatorname{rk}(u + v)
\leq \dim(\operatorname{im} u + \operatorname{im} v)
\leq \operatorname{rk} u + \operatorname{rk} v
\]
(ग्रासमैन, \cref{thm:b1:findim:grassmann}). निचली \omterm{def:b1:topology:closure}{सीमा}: लागू करें
द \omterm{def:b1:reals:bounds}{ऊपरी सीमा} जोड़ी को $(u + v, -v)$, जिसका योग है $ u $:
\[
\operatorname{rk} u \leq \operatorname{rk}(u + v) +
\operatorname{rk}(-v) = \operatorname{rk}(u + v) +
\operatorname{rk} v,
\]
इसलिए $\operatorname{rk} u - \operatorname{rk} v \leq
\operatorname{rk}(u+v)$; का आदान प्रदान $ u $ और $ v $ पूर्ण देता है
मूल्य.
\end{solution}

\begin{solution}{exo:b1:linmaps:12}
\emph{सटीक सूत्र.} होने देना $ v'$ का प्रतिबंध हो $ v $ तक
\omterm{def:b1:vspaces:subspace}{उपस्पेस} $\operatorname{im} w $. इसका \omterm{def:b1:linmaps:kerim}{छवि} है $ v(w(F)) =
\operatorname{im}(v \circ w)$, और इसके \omterm{def:b1:linmaps:kerim}{कर्नेल} है $\ker v \cap
\operatorname{im} w $. रैंक--के लिए शून्यता $ v'$ अंतरिक्ष पर
$\operatorname{im} w $:
\[
\operatorname{rk} w = \dim\operatorname{im} w
= \operatorname{rk}(v \circ w) + \dim(\ker v \cap
\operatorname{im} w) .
\]

\emph{फ्रोबेनियस.} सटीक सूत्र को दो बार लागू करें $ w $ और को
$ w \circ u $:
\[
\operatorname{rk} w - \operatorname{rk}(vw)
= \dim\bigl(\ker v \cap \operatorname{im} w\bigr),
\qquad
\operatorname{rk}(wu) - \operatorname{rk}(vwu)
= \dim\bigl(\ker v \cap \operatorname{im}(wu)\bigr) .
\]
तब से $\operatorname{im}(w \circ u) \subseteq \operatorname{im}
w $, दूसरा चौराहा पहले में समाहित है, और उसका
आयाम बड़ा नहीं है:
\[
\operatorname{rk}(wu) - \operatorname{rk}(vwu)
\;\leq\; \operatorname{rk} w - \operatorname{rk}(vw) ,
\]
जो फ्रोबेनियस की असमानता में पुनर्व्यवस्थित होता है। साथ $ w =
\mathrm{id}_F $ (\omterm{def:b1:linmaps:rank}{रैंक} $\dim F $, और $\operatorname{im}\,
\mathrm{id}_F = F $): $\operatorname{rk} v + \operatorname{rk} u
\leq \dim F + \operatorname{rk}(vu)$, सिल्वेस्टर की असमानता ---
फिर से साबित हुआ, मैट्रिक्स रूप में, में \cref{exo:b1:matrices:10}.
\end{solution}

\begin{solution}{pb:b1:linmaps:1}
\textbf{1.} $(\mathrm{id} - p)^2 = \mathrm{id} - 2p + p^2 =
\mathrm{id} - p $: एक प्रोजेक्टर. अगर $ y = x - p(x)$, तब $ p(y) =
p(x) - p^2(x) = 0$, और इसके विपरीत $ x \in \ker p $ देता है $ x =
(\mathrm{id} - p)(x)$: $\operatorname{im}(\mathrm{id} - p) = \ker
p $. और $(\mathrm{id} - p)(x) = 0 \iff p(x) = x \iff x \in
\operatorname{im} p $ (निश्चित अंक,
\cref{thm:b1:linmaps:projchar}): $\ker(\mathrm{id} - p) =
\operatorname{im} p $.

\textbf{2.} $(\lambda\,\mathrm{id} + \mu p)^2 =
\lambda^2\,\mathrm{id} + (2\lambda\mu + \mu^2)\,p $. जोड़ी
$(\mathrm{id}, p)$ है \omterm{def:b1:vspaces:free}{मुक्त} में $\mathcal{L}(E)$: $ p = c\,
\mathrm{id}$ देगा $ c^2 = c $, इसलिए $ p = 0$ या $\mathrm{id}$,
छोड़ा गया। गुणांक की पहचान, \omterm{def:b1:logic:map}{नक्शा} एक प्रोजेक्टर है iff
$\lambda^2 = \lambda $ और $2\lambda\mu + \mu^2 = \mu $. के लिए
$\lambda = 0$: $\mu \in \{0, 1\}$. के लिए $\lambda = 1$: $\mu^2 +
\mu = 0$, $\mu \in \{0, -1\}$. बिल्कुल चार प्रोजेक्टर
हवाई जहाज़: $0$, $ p $, $\mathrm{id}$, $\mathrm{id} - p $.

\textbf{3.} $(\lambda\,\mathrm{id} + \mu p)(\lambda'\,\mathrm{id}
+ \mu' p) = \lambda\lambda'\,\mathrm{id} + (\lambda\mu' +
\mu\lambda' + \mu\mu')\,p $: विमान नीचे स्थिर है
रचना. तब से $ p^k = p $ हरएक के लिए $ k \geq 1$, के लिए $ Q =
\sum_k a_k X^k $:
\[
Q(p) = a_0\,\mathrm{id} + \Bigl(\sum_{k \geq 1} a_k\Bigr) p
= Q(0)\,\mathrm{id} + \bigl(Q(1) - Q(0)\bigr)\,p .
\]

\textbf{4.} पर $\operatorname{im} p $ (कहाँ $ p $ के रूप में कार्य करता है
पहचान), $\lambda\,\mathrm{id} + \mu p $ से गुणा करता है $\lambda
+ \mu $; पर $\ker p $, द्वारा $\lambda $. जैसा $ E = \operatorname{im} p
\oplus \ker p $, द \omterm{def:b1:logic:map}{नक्शा} है \omterm{def:b1:logic:inj}{द्विभाजित} आईएफएफ $\lambda \neq 0$ और
$\lambda + \mu \neq 0$. हल $\lambda\alpha = 1$,
$\lambda\beta + \mu\alpha + \mu\beta = 0$ रचना नियम में
प्रश्न 3 का:
\[
(\lambda\,\mathrm{id} + \mu p)^{-1}
= \frac1\lambda\,\mathrm{id} -
\frac{\mu}{\lambda(\lambda + \mu)}\,p ,
\]
जिसकी कार्रवाई है $1/\lambda $ पर $\ker p $ और $1/(\lambda +
\mu)$ पर $\operatorname{im} p $, जैसा कि यह होना चाहिए।

\textbf{5.} लिखना $ F = \operatorname{im} p = \operatorname{im}
p'$. हरएक के लिए $ x $, $ p'(x) \in F $ और $ p $ ठीक करता है $ F $ बिंदुवार:
$ p(p'(x)) = p'(x)$, यानी\ $ p\,p' = p'$; संतुलित $ p'\,p =
p $. जब दो अनुमान उन्हें साझा करें \omterm{def:b1:linmaps:kerim}{छवि}, जिसने आवेदन किया
\emph{पहला} निर्णय लेता है: इसका आउटपुट पहले से ही निहित है $ F $, जहां
बाहरी अनुमान पहचान के रूप में कार्य करता है और कुछ भी नहीं बदलता है।

\textbf{6.} $(p + q)^2 = p^2 + pq + qp + q^2 = (p + q) + pq +
qp $, इसलिए $ p + q $ प्रोजेक्टर बल $ pq + qp = 0$. पर लिखें
साथ छोड़ दिया $ p $: $ pq + pqp = 0$; के साथ दाईं ओर $ p $: $ pqp + qp
= 0$. घटाना, $ pq = qp $; तब $ pq + qp = 2pq = 0$ और
विशेषता नहीं है $2$: $ pq = qp = 0$.

\textbf{7.} साथ $ pq = qp = 0$, वही विस्तार देता है $(p +
q)^2 = p + q $. \omterm{def:b1:linmaps:kerim}{छवि}: $\operatorname{im}(p + q) \subseteq
\operatorname{im} p + \operatorname{im} q $ हमेशा। इसके विपरीत,
के लिए $ x \in \operatorname{im} p $: $ q(x) = q(p(x)) = 0$, इसलिए $(p +
q)(x) = p(x) = x $ और $ x \in \operatorname{im}(p+q)$; के लिए समान
$\operatorname{im} q $. प्रत्यक्षता: $ x \in \operatorname{im} p
\cap \operatorname{im} q $ देता है $ x = p(x) = p(q(x)) = 0$.
गुठली: अगर $ p(x) + q(x) = 0$, आवेदन करना $ p $ देता है $ p(x) +
p(q(x)) = p(x) = 0$, और आवेदन करना $ q $ देता है $ q(x) = 0$: $\ker(p
+ q) = \ker p \cap \ker q $ (रिवर्स समावेशन स्पष्ट है)।

\textbf{8.} $ p - q $ एक प्रोजेक्टर है iff $\mathrm{id} - (p - q) =
(\mathrm{id} - p) + q $ एक है (प्रश्न 1 दो बार)। सवालों से
6-7 प्रोजेक्टर पर लागू किया गया $\mathrm{id} - p $ और $ q $, यह
यदि रखता है $(\mathrm{id} - p)q = q(\mathrm{id} - p) = 0$, अर्थात\
यदि $ pq = q $ और $ qp = q $.

\textbf{9.} $ pq = q $ मतलब $ p $ प्रत्येक को ठीक करता है $ q(x)$, अर्थात\
$\operatorname{im} q \subseteq \ker(p - \mathrm{id}) =
\operatorname{im} p $. और $ qp = q $ मतलब $ q\bigl((\mathrm{id} -
p)(x)\bigr) = 0$ सभी के लिए $ x $, यानी\ $ q $ पर गायब हो जाता है
$\operatorname{im}(\mathrm{id} - p) = \ker p $: $\ker p \subseteq
\ker q $. दोनों चरण समतुल्य हैं: क्रम $ q \leq p $ कहते हैं
$ q $ छोटे पर प्रोजेक्ट करता है \omterm{def:b1:linmaps:kerim}{छवि}, एक बड़े के साथ \omterm{def:b1:linmaps:kerim}{कर्नेल}.

\textbf{10.} विस्तार, $(\mathrm{id} - p)(\mathrm{id} - q) =
\mathrm{id} - p - q + pq = \mathrm{id} - r $. प्रोजेक्टर
$\mathrm{id} - p $ और $\mathrm{id} - q $ आवागमन, तो द्वारा
\cref{exo:b1:linmaps:6} उनका उत्पाद $\mathrm{id} - r $ है
प्रोजेक्टर चालू $\operatorname{im}(\mathrm{id} - p) \cap
\operatorname{im}(\mathrm{id} - q) = \ker p \cap \ker q $ साथ में
$\ker(\mathrm{id} - p) + \ker(\mathrm{id} - q) =
\operatorname{im} p + \operatorname{im} q $. प्रश्न 1 से, $ r =
\mathrm{id} - (\mathrm{id} - r)$ फिर प्रोजेक्टर के साथ है
$\operatorname{im} r = \operatorname{im} p + \operatorname{im}
q $ और $\ker r = \ker p \cap \ker q $.

\textbf{11.} $ p_i $ अच्छी तरह से परिभाषित है (की विशिष्टता)
अपघटन) और \omterm{def:b1:linmaps:def}{रेखीय} (का अपघटन $ x + \lambda y $
विघटितों का योग है, फिर से विशिष्टता द्वारा)। के लिए $ x_i
\in F_i $ विघटन है $ x_i $ खुद, तो $ p_i(x_i) = x_i $:
$ p_i^2 = p_i $, और $ p_j(x_i) = 0$ के लिए $ j \neq i $: $ p_i p_j = 0$
($ p_j(x) \in F_j $). घटकों को संक्षेप में, $\sum_i p_i =
\mathrm{id}$. अंत में $\operatorname{im} p_i = F_i $ और $\ker
p_i = \bigoplus_{j \neq i} F_j $.

\textbf{12.} $ p_i = p_i \circ \mathrm{id} = p_i\sum_j p_j =
p_i^2 + \sum_{j \neq i} p_i p_j = p_i^2$: प्रत्येक $ p_i $ एक है
प्रोजेक्टर. प्रत्येक $ x = \mathrm{id}(x) = \sum_i p_i(x)$ में निहित है
$\sum_i \operatorname{im} p_i $: द इमेजिस का योग $ E $.
प्रत्यक्षता: मान लीजिए $ y_1 + \dots + y_k = 0$ साथ $ y_i \in
\operatorname{im} p_i $, इसलिए $ p_i(y_i) = y_i $. आवेदन करना $ p_j $: $ p_j
(y_i) = p_j p_i (y_i) = 0$ के लिए $ i \neq j $, इसलिए $0 = p_j\bigl(\sum
y_i\bigr) = y_j $, हरएक के लिए $ j $. इस तरह $ E = \bigoplus_i
\operatorname{im} p_i $.

\textbf{13.} $ q = \mathrm{id} - p $, और प्रश्न 1 देता है $ pq =
p - p^2 = 0 = qp $ सीधे: दो प्रोजेक्टर के लिए, संक्षेप में
पहचान पहले से ही जोड़ी की रूढ़िवादिता को मजबूर करती है।

\textbf{14.} $ p + q = \mathrm{id} - r $ साथ $ r $ एक प्रोजेक्टर, और
$(\mathrm{id} - r)$ एक प्रोजेक्टर है (प्रश्न 1): तो $ p + q $ एक है
प्रोजेक्टर, और प्रश्न 6 देता है $ pq = qp = 0$. \omterm{def:b1:linmaps:projection}{समरूपता} द्वारा ($ q +
r = \mathrm{id} - p $ और $ p + r = \mathrm{id} - q $), सभी
जोड़ीवार उत्पाद गायब हो जाते हैं, और प्रश्न 12 समाप्त होता है: $ E =
\operatorname{im} p \oplus \operatorname{im} q \oplus
\operatorname{im} r $, खुद ब खुद।

\textbf{15.} \emph{लेम्मा.} ग्रासमैन के साथ प्रेरण द्वारा
(\cref{thm:b1:findim:grassmann}):
\[
\dim(F_1 + \dots + F_k) \leq \dim(F_1 + \dots + F_{k-1}) + \dim
F_k \leq \dots \leq \sum_i \dim F_i .
\]
यदि कुल एक समानता है, तो प्रत्येक चरण है: $(F_1 + \dots +
F_{j-1}) \cap F_j = \{0\}$ हरएक के लिए $ j $, और एक रिश्ता $ y_1 +
\dots + y_k = 0$ ($ y_i \in F_i $) दाहिनी ओर से ढह जाता है: $ y_k
\in (F_1 + \dots + F_{k-1}) \cap F_k = \{0\}$, तब $ y_{k-1} =
0$, आदि: योग \omterm{def:b1:vspaces:sum}{प्रत्यक्ष} है। इसके विपरीत ए \omterm{def:b1:vspaces:sum}{प्रत्यक्ष} योग है
योगात्मक आयाम (संक्षिप्त \omterm{def:b1:vspaces:free}{आधार})। \emph{आवेदन पत्र:}
$ x = \sum_i p_i(x)$ दिखाता है $ E = \sum_i \operatorname{im} p_i $,
इसलिए $ n \leq \sum_i \operatorname{rk} p_i $; परिकल्पना देती है
समानता, इसलिए प्रत्यक्षता. उत्पाद: ठीक करें $ j $ और $ y \in
\operatorname{im} p_j $. तब $ y = \sum_i p_i(y)$ साथ $ p_i(y)
\in \operatorname{im} p_i $, जबकि $ y = y $ एक भी है
अपघटन (घटक $ j $ अकेला); विशिष्टता बल $ p_i(y)
= 0$ के लिए $ i \neq j $. के लिए आवेदन किया $ y = p_j(x)$: $ p_i p_j = 0$.

\textbf{16.} अगर $ u^k(x) = 0$ तब $ u^{k+1}(x) = u(0) = 0$. और
$\operatorname{im} u^{k+1} = u^k\bigl(u(E)\bigr) \subseteq
u^k(E) = \operatorname{im} u^k $.

\textbf{17.} मान लीजिए $\ker u^{r} = \ker u^{r+1}$ और चलो $ x \in
\ker u^{r+2}$: तब $ u(x) \in \ker u^{r+1} = \ker u^{r}$, इसलिए
$ u^{r+1}(x) = 0$: $ x \in \ker u^{r+1}$. प्रश्न 16 के साथ,
$\ker u^{r+1} = \ker u^{r+2}$, और बाद में प्रेरण द्वारा
\omterm{def:b1:linmaps:kerim}{कर्नेल} साथ मेल खाना $\ker u^{r}$. के लिए इमेजिस: पद--शून्यता
देता है $\dim\operatorname{im} u^k = n - \dim\ker u^k $, इतना
\omterm{def:b1:linmaps:kerim}{छवि} आयाम ठीक उसी समय स्थिर हो जाते हैं जब \omterm{def:b1:linmaps:kerim}{कर्नेल} आयाम करते हैं,
और प्रश्न 16 के समावेशन के साथ, समान आयाम का मतलब है
बराबर उपस्थान (\cref{thm:b1:findim:subspaces}).

\textbf{18.} क्रम $\bigl(\dim\ker u^k\bigr)_k $ है
में मानों के साथ न घटनेवाला $\intint{0}{n}$; यह बढ़ नहीं सकता
सख्ती से $ n + 1$ कई बार, तो कुछ $\dim\ker u^{r} = \dim\ker
u^{r+1}$ साथ $ r \leq n $, इस तरह $\ker u^{r} = \ker u^{r+1}$
(समावेश प्लस समान आयाम)। लेना $ r $ सबसे छोटा; प्रश्न 17
से सब कुछ जमा देता है $ r $ पर, इमेजिस शामिल.

\textbf{19.} प्रतिच्छेदन: चलो $ x \in \ker u^{r} \cap
\operatorname{im} u^{r}$, कहना $ x = u^{r}(y)$ साथ $ u^{r}(x) =
0$. तब $ u^{2r}(y) = 0$, और $\ker u^{2r} = \ker u^{r}$
(प्रश्न 17), अतः $ x = u^{r}(y) = 0$. आयाम: रैंक--शून्यता
के लिए $ u^{r}$ देता है $\dim\ker u^{r} + \dim\operatorname{im} u^{r}
= n $; तुच्छ प्रतिच्छेदन के साथ, ग्रासमैन योग a बनाता है
\omterm{def:b1:vspaces:subspace}{उपस्पेस} आयाम का $ n $: $ E = \ker u^{r} \oplus
\operatorname{im} u^{r}$.

\textbf{20.} स्थिरता: $ u^{r}(u(x)) = u(u^{r}(x)) = 0$ के लिए $ x
\in \ker u^{r}$; और $ u(u^{r}(y)) = u^{r}(u(y)) \in
\operatorname{im} u^{r}$. पर $ N = \ker u^{r}$: $(u|_N)^{r} = 0$
की परिभाषा के अनुसार $ N $: निलपोटेंट. पर $ I = \operatorname{im}
u^{r}$: $\ker(u|_I) = \ker u \cap I \subseteq \ker u^{r} \cap I
= \{0\}$, इसलिए $ u|_I $ एक \omterm{def:b1:logic:inj}{इंजेक्शन} की एंडोमोर्फिज्म
\omterm{def:b1:findim:def}{परिमित-आयामी} $ I $, इस तरह \omterm{def:b1:logic:inj}{द्विभाजित}
(\cref{cor:b1:linmaps:samedim}).

\textbf{21.} होने देना $ x = a + b $ साथ $ a \in N $, $ b \in I $. तब
$ u(x) = u(a) + u(b)$ साथ $ u(a) \in N $ और $ u(b) \in I $
(प्रश्न 20): यह \emph{है} का विघटन $ u(x)$, इसलिए
$\pi(u(x)) = u(a) = u(\pi(x))$: $\pi u = u\pi $.

\textbf{22.} $ u^2(x,y,z) = u(y, 0, z) = (0, 0, z)$ और
$ u^3(x,y,z) = u(0,0,z) = (0,0,z) = u^2(x,y,z)$. \omterm{def:b1:linmaps:kerim}{कर्नेल}: $\ker u
= \{y = z = 0\} = \operatorname{Vect}(e_1)$, $\ker u^2 = \{z =
0\} = \operatorname{Vect}(e_1, e_2)$, $\ker u^3 = \ker u^2$:
पर स्थिरीकरण $ r = 2$. इमेजिस: $\operatorname{im} u =
\operatorname{Vect}(e_1, e_3)$, $\operatorname{im} u^2 =
\operatorname{Vect}(e_3)$. फिटिंग: $\R^3 =
\operatorname{Vect}(e_1, e_2) \oplus \operatorname{Vect}(e_3)$,
और $\pi(x, y, z) = (x, y, 0)$. जाँच करना: $\pi u(x,y,z) = \pi(y, 0,
z) = (y, 0, 0)$ और $ u\pi(x,y,z) = u(x, y, 0) = (y, 0, 0)$:
बराबर। पहले पहलू पर $ u(x, y, 0) = (y, 0, 0)$, किसका
वर्ग है $0$: निलपोटेंट; दूसरे पर $ u(0,0,z) = (0,0,z)$:
पहचान, \omterm{def:b1:logic:inj}{द्विभाजित}.

\textbf{23.} अगर $ u^m = 0$ तब $\ker u^m = E $; तब से \omterm{def:b1:linmaps:kerim}{कर्नेल} हैं
से जमे हुए $ r $ पर, $\ker u^{r} = \ker u^{\max(m, r)} = E $.
इसके विपरीत $\ker u^{r} = E $ मतलब $ u^{r} = 0$. और $\ker u^{r} =
E \iff $ फिटिंग प्रोजेक्टर चालू है $ E $ साथ में $\{0\}$, अर्थात\
$\pi = \mathrm{id}$. अंत में $ r \leq n $ (प्रश्न 18) देता है:
प्रत्येक निलपोटेंट एंडोमोर्फिज्म संतुष्ट करता है $ u^{n} = 0$ --- द
निलपोटेंसी इंडेक्स कभी भी आयाम से अधिक नहीं होता है।

\textbf{24.} होने देना $ m $ का एक निलपोटेंसी सूचकांक हो $ u|_A $: $ A
\subseteq \ker u^{m} \subseteq \ker u^{\max(m,r)} = \ker u^{r}$.
तब से $ u|_B $ है \omterm{def:b1:logic:inj}{द्विभाजित}, $ B = u(B) = u^{k}(B) \subseteq
\operatorname{im} u^{k}$ हरएक के लिए $ k $, विशेष रूप से $ B
\subseteq \operatorname{im} u^{r}$. तब
\[
n = \dim A + \dim B \leq \dim\ker u^{r} +
\dim\operatorname{im} u^{r} = n :
\]
दोनों समावेशन आयामों की समानता हैं, इसलिए
उपस्थान: $ A = \ker u^{r}$, $ B = \operatorname{im} u^{r}$.

\textbf{25.} (i) भाग III एक शब्दकोश है: विभाजन $ E = F_1
\oplus \dots \oplus F_k $ के परिवारों से बिल्कुल मेल खाता है
प्रोजेक्टर के साथ $\sum p_i = \mathrm{id}$ और $ p_i p_j = 0$, द
$ F_i $ होने के नाते इमेजिस. (ii) के लिए $ k = 3$ \omterm{def:b1:vspaces:sum}{पूरक}
$\mathrm{id} - p_i $ स्वयं प्रोजेक्टर हैं, जिन्होंने \omterm{def:b1:topology:closure}{बंद} कर दिया
बिना किसी अतिरिक्त परिकल्पना के तर्क; सामान्य के लिए $ k $ हमें चाहिए
$\sum_i \operatorname{rk} p_i \leq n $, एक असमानता वर्ष~2
ट्रेस मुफ्त में देता है ($\operatorname{tr} p = \operatorname{rk}
p $ एक प्रोजेक्टर के लिए, और निशान जुड़ते हैं $\operatorname{tr}
\mathrm{id} = n $). (iii) \cref{exo:b1:linmaps:7} फिटिंग का है
पहले से ही स्थिर मामले में लेम्मा $ r \leq 1$; सामान्य तौर पर एक
चलो \omterm{def:b1:linmaps:kerim}{कर्नेल} और \omterm{def:b1:linmaps:kerim}{छवि} जंजीरें जम जाती हैं, जिसमें अधिक से अधिक समय लगता है $ n $
कदम. (iv) वर्ष~2 खंड में, लागू किया गया $ u - \lambda\,
\mathrm{id}$, शून्यप्रभावी कारक सामान्यीकृत हो जाता है
eigenspace पर $\lambda$ और भाग III के प्रोजेक्टर बन जाते हैं
कमी सिद्धांत के वर्णक्रमीय प्रोजेक्टर। का प्रमेय
भाग IV है \emph{\omterm{pb:b1:linmaps:1}{फिटिंग की लेम्मा}}.
\end{solution}
