\section*{Chapter \ref{ch:b1:findim} --- Finite Dimension}

\begin{solution}{exo:b1:findim:1}
\begin{enumerate}
  \item $z = -x - y$: $F = \{(x, y, -x-y)\} =
        \operatorname{Vect}\bigl((1,0,-1), (0,1,-1)\bigr)$; the two
        vectors are free (coordinates): $\dim F = 2$.
  \item $G = \{(x, x, 2t, t)\} =
        \operatorname{Vect}\bigl((1,1,0,0), (0,0,2,1)\bigr)$: free,
        $\dim G = 2$.
  \item $P(1) = P'(1) = 0$ means $(X-1)^2 \mid P$
        (\cref{prop:b1:poly:multiplicity}): $P = (X-1)^2(aX + b)$.
        Basis $\bigl((X-1)^2, X(X-1)^2\bigr)$, dimension $2$.
\end{enumerate}
\end{solution}

\begin{solution}{exo:b1:findim:2}
$(3,5,7) = (1,1,1) + 2(1,2,3)$ and $(0,1,2) = (1,2,3) - (1,1,1)$:
both are combinations of the first two, which are free (not
proportional). Rank $2$; basis of the span:
$\bigl((1,1,1), (1,2,3)\bigr)$.
\end{solution}

\begin{solution}{exo:b1:findim:3}
Try appending $e_1 = (1,0,0,0)$ and $e_3 = (0,0,1,0)$. The family
$\bigl((1,1,0,0), (0,0,1,1), e_1, e_3\bigr)$ is free: a null
combination $\alpha(1,1,0,0) + \beta(0,0,1,1) + \gamma e_1 + \delta
e_3 = 0$ reads $(\alpha + \gamma, \alpha, \beta + \delta, \beta) =
0$, so $\alpha = \beta = 0$, then $\gamma = \delta = 0$. Four free
vectors in dimension $4$: a basis
(\cref{prop:b1:findim:twoofthree}).
\end{solution}

\begin{solution}{exo:b1:findim:4}
Degrees $0, 1, 2, 3$ pairwise distinct: free
(\cref{prop:b1:vspaces:freecriteria}), four vectors in dimension
$4$: basis. For $X^3$: expand downward,
\[
X(X-1)(X-2) = X^3 - 3X^2 + 2X,
\qquad
X(X-1) = X^2 - X,
\]
so $X^3 - X(X-1)(X-2) = 3X^2 - 2X$; and $3X^2 - 2X = 3(X^2 - X) + X
= 3\,X(X-1) + X$. Hence
\[
X^3 = X(X-1)(X-2) + 3\,X(X-1) + 1\cdot X + 0\cdot 1 :
\]
coordinates $(0, 1, 3, 1)$ on $\bigl(1, X, X(X-1),
X(X-1)(X-2)\bigr)$. (These are Stirling numbers in disguise.)
\end{solution}

\begin{solution}{exo:b1:findim:5}
Grassmann: $\dim(F \cap G) = 4 + 5 - \dim(F + G)$, and $F + G$ is a
subspace of $E$ containing $G$: $5 \leq \dim(F+G) \leq 7$. Hence
$\dim(F \cap G) \in \{2, 3, 4\}$. Realizations in $\R^7$ with
canonical basis $(e_1, \dots, e_7)$, taking $G =
\operatorname{Vect}(e_1, \dots, e_5)$:
\begin{itemize}
  \item $F = \operatorname{Vect}(e_1, e_2, e_6, e_7)$: $F + G = \R^7$,
        intersection $\operatorname{Vect}(e_1, e_2)$, dimension $2$;
  \item $F = \operatorname{Vect}(e_1, e_2, e_3, e_6)$: intersection
        of dimension $3$;
  \item $F = \operatorname{Vect}(e_1, e_2, e_3, e_4) \subseteq G$:
        dimension $4$.
\end{itemize}
\end{solution}

\begin{solution}{exo:b1:findim:6}
$H$ is a subspace (\cref{exo:b1:vspaces:1}~(4) verbatim). By the
factor theorem (\cref{thm:b1:poly:factor}), $P \in H \iff P =
(X-1)Q$ with $\deg Q \leq n - 1$: the map $Q \mapsto (X-1)Q$ is a
linear bijection from $\R_{n-1}[X]$ onto $H$, so a basis of $H$ is
\[
\bigl((X-1),\ (X-1)X,\ (X-1)X^2,\ \dots,\ (X-1)X^{n-1}\bigr),
\qquad \dim H = n .
\]
A supplementary line: $\operatorname{Vect}(1)$ (constants). Indeed
$H \cap \operatorname{Vect}(1) = \{0\}$ (a nonzero constant does not
vanish at $1$) and dimensions add up to $n + 1$: by Grassmann, $H
\oplus \operatorname{Vect}(1) = \R_n[X]$.
\end{solution}

\begin{solution}{exo:b1:findim:7}
Deletion: removing $u_p$, the span can only shrink; and it shrinks
by at most one dimension, since adding $u_p$ back to a basis of the
smaller span gives a generating family of the larger with at most
one extra vector. Symmetrically, appending a vector $v$: the new
span contains the old with at most one extra generator, so its
dimension is $r$ (if $v$ was already in the span) or $r + 1$
(otherwise, by \cref{prop:b1:vspaces:freecriteria}~(2) a basis
extends). Both statements together give the ``rank is
$1$-Lipschitz'' conclusion.
\end{solution}

\begin{solution}{exo:b1:findim:8}
Along a strictly increasing chain, dimensions strictly increase
($F_i \subseteq F_{i+1}$, $F_i \neq F_{i+1}$ and
\cref{thm:b1:findim:subspaces}: equality of dimensions would force
equality of spaces). So $\dim F_1 < \dim F_2 < \dots < \dim F_k$ is
a strictly increasing sequence of integers in $\intint{0}{n}$: at
most $n + 1$ values, $k \leq n + 1$. Maximal chain in $\R^n$:
\[
\{0\} \subsetneq \operatorname{Vect}(e_1) \subsetneq
\operatorname{Vect}(e_1, e_2) \subsetneq \dots \subsetneq \R^n ,
\]
of length exactly $n + 1$.
\end{solution}

\begin{solution}{exo:b1:findim:9}
$F + G$ strictly contains $F$ (since $G \not\subseteq F$), so
$\dim(F + G) = n$ and Grassmann gives $\dim(F \cap G) = (n-1) +
(n-1) - n = n - 2$.

General claim, by induction on $k$: the intersection $I_k$ of $k$
hyperplanes has $\dim I_k \geq n - k$. True for $k = 1$. Step: $I_{k+1}
= I_k \cap H_{k+1}$, and Grassmann inside $E$:
\[
\dim(I_k \cap H_{k+1}) = \dim I_k + (n - 1) - \dim(I_k + H_{k+1})
\geq \dim I_k + (n-1) - n \geq n - k - 1 . \qedhere
\]
\end{solution}
