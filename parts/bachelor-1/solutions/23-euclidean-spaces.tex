\begin{solution}{exo:b1:euclid:4}

\begin{solution}{exo:b1:euclid:5}
Let $p = \alpha(1,0,1) + \beta(0,1,1) = (\alpha, \beta, \alpha +
\beta)$. Orthogonality of $x - p$ to the generators:
\[
\langle x - p, (1,0,1)\rangle = (1 - \alpha) + (4 - \alpha - \beta)
= 0,
\qquad
\langle x - p, (0,1,1)\rangle = (1 - \beta) + (4 - \alpha - \beta)
= 0,
\]
i.e.\ $2\alpha + \beta = 5$ and $\alpha + 2\beta = 5$: $\alpha =
\beta = \frac53$. So $p_F(x) = \bigl(\frac53, \frac53,
\frac{10}{3}\bigr)$ and
\[
d(x, F) = \norm{x - p} =
\norm{\bigl(-\tfrac23, -\tfrac23, \tfrac23\bigr)} =
\frac{2}{\sqrt 3} .
\]
Gram--Schmidt is unnecessary because the defining property of the
\omterm{def:b1:linmaps:projection}{projection} --- $x - p \perp F$ --- is itself a \omterm{def:b1:det:system}{linear system}
(``normal equations'') on the coefficients in \emph{any} generating
family.
\end{solution}

\begin{solution}{exo:b1:euclid:6}
Cauchy--Schwarz with $g = 1$:
\[
\Bigl(\int_0^1 f \cdot 1\Bigr)^2 \leq \int_0^1 f^2 \cdot \int_0^1
1^2 = \int_0^1 f^2 ,
\]
equality iff $f$ is proportional to $1$, i.e.\ constant. Discrete
version: in $\R^n$ with $x = (a_1, \dots, a_n)$, $y = (1, \dots,
1)$:
$\bigl(\sum a_i\bigr)^2 \leq \norm x^2\, \norm y^2 = n \sum a_i^2$,
equality iff all $a_i$ equal.
\end{solution}

\begin{solution}{exo:b1:euclid:7}
By \cref{thm:b1:euclid:projection}, $E = F \oplus F^\perp$, so
dimensions add: $\dim F^\perp = \dim E - \dim F$. Inclusion $F
\subseteq (F^\perp)^\perp$ is immediate (vectors of $F$ are
\omterm{def:b1:euclid:orthogonal}{orthogonal} to everything in $F^\perp$). Dimensions:
$\dim(F^\perp)^\perp = \dim E - \dim F^\perp = \dim F$; an
inclusion with equal (finite) dimensions is an equality
(\cref{thm:b1:findim:subspaces}).
\end{solution}

\begin{solution}{exo:b1:euclid:9}
The quantity is $\norm{f - (a + bX)}^2$ in $C(\intcc{0}{1})$ with
the \omterm{thm:b1:integration:def}{integral} \omterm{def:b1:euclid:def}{inner product}: minimal exactly at the \omterm{thm:b1:euclid:projection}{orthogonal
projection} $a + bX = p_F(f) = X - \frac16$
(\cref{ex:b1:euclid:bestapprox}). The minimum is
\[
\norm{f - p_F(f)}^2 = \int_0^1 \Bigl(x^2 - x + \frac16\Bigr)^{\!2}
\dd x .
\]
Expand: $\int_0^1 (x^2 - x + \frac16)^2 = \int_0^1 \bigl(x^4 - 2x^3
+ \frac{4}{3}x^2 - \frac13 x + \frac{1}{36}\bigr)\dd x = \frac15 -
\frac12 + \frac49 - \frac16 + \frac{1}{36} = \frac{1}{180}$.
So the minimum equals $\dfrac{1}{180}$.
\end{solution}

\begin{solution}{exo:b1:euclid:10}
Let $y \in \ker(u - \mathrm{id})$ (i.e.\ $u(y) = y$) and $x \in E$.
Then, using preservation of the \omterm{def:b1:euclid:def}{inner product} ($\langle u(a),
u(b)\rangle = \langle a, b\rangle$):
\[
\langle x - u(x),\, y\rangle
= \langle x, y\rangle - \langle u(x), y\rangle
= \langle x, y\rangle - \langle u(x), u(y)\rangle
= \langle x, y\rangle - \langle x, y\rangle = 0 :
\]
every vector of $\operatorname{im}(u - \mathrm{id})$ is \omterm{def:b1:euclid:orthogonal}{orthogonal}
to every fixed vector.

Therefore $\operatorname{im}(u - \mathrm{id}) \subseteq \ker(u -
\mathrm{id})^{\perp}$, and by rank--nullity plus
\cref{exo:b1:euclid:7}, both have dimension $\dim E - \dim\ker(u -
\mathrm{id})$: they are equal. Then $E = \ker(u - \mathrm{id})
\oplus \ker(u - \mathrm{id})^\perp = \ker(u - \mathrm{id}) \oplus
\operatorname{im}(u - \mathrm{id})$.
\end{solution}

\begin{solution}{exo:b1:euclid:12}
\begin{enumerate}
  \item $A^2 = A A = A^{\mathsf T} A = I$, so $u^2 =
        \mathrm{id}$: $u$ is a symmetry, and
        \cref{thm:b1:linmaps:projchar}~(2) gives $E = \ker(u -

        symmetric: the line reflections. These are exactly the

\begin{solution}{pb:b1:euclid:1}
\textbf{1.} $\norm{t_a(x) - t_a(y)} = \norm{x - y}$; a \omterm{def:b1:linmaps:def}{linear}
\omterm{def:b1:euclid:isometry}{isometry} preserves norms, hence distances ($\norm{g(x) - g(y)} =
\norm{g(x - y)} = \norm{x-y}$); and a composition of
distance-preserving \omterm{def:b1:logic:map}{maps} preserves distances. Every \omterm{def:b1:euclid:isometry}{isometry} is
\omterm{def:b1:logic:inj}{injective} (distinct points stay at positive distance) and, by
the classification below, \omterm{def:b1:logic:inj}{bijective}; identity and inverses are
isometries: a \omterm{def:b1:structures:group}{group}.

\textbf{2.} $\norm{f(x)} = \norm{f(x) - f(0)} = \norm{x - 0} =
\norm x$. Polarization:
\[
\langle f(x), f(y)\rangle
= \frac{\norm{f(x)}^2 + \norm{f(y)}^2 - \norm{f(x) -
f(y)}^2}{2}
= \frac{\norm x^2 + \norm y^2 - \norm{x - y}^2}{2}
= \langle x, y\rangle .
\]

\textbf{3.} Expanding with question 2 (each product of $f$'s
equals the product of the arguments):
\begin{align*}
\norm{f(x+y) - f(x) - f(y)}^2
&= \norm{x+y}^2 + \norm x^2 + \norm y^2\\
&\quad - 2\langle x+y, x\rangle - 2\langle x+y, y\rangle
+ 2\langle x, y\rangle = 0 ,
\end{align*}
as a direct check shows; similarly $\norm{f(\lambda x) -
\lambda f(x)}^2 = \norm{\lambda x}^2 - 2\lambda\langle \lambda
x, x\rangle + \lambda^2\norm x^2 = 0$. So $f(x + y) = f(x) +
f(y)$ and $f(\lambda x) = \lambda f(x)$: $f$ is \omterm{def:b1:linmaps:def}{linear}, and
norm-preserving: $f \in O(\R^2)$.

\textbf{4.} Set $a = f(0)$ and $g = t_{-a}\circ f$: an \omterm{def:b1:euclid:isometry}{isometry}
fixing $0$, hence a \omterm{def:b1:linmaps:def}{linear} \omterm{def:b1:euclid:isometry}{isometry} (question 3), and $f = t_a
\circ g$. Uniqueness: $t_a \circ g = t_{a'}\circ g'$ evaluated
at $0$ gives $a = a'$, then $g = g'$.

\textbf{5.} $(t_a\circ g)\circ(t_{a'}\circ g') = t_{a +
g(a')}\circ(g\circ g')$, since $g\,t_{a'} = t_{g(a')}\,g$ for
\omterm{def:b1:linmaps:def}{linear} $g$. So the \omterm{def:b1:linmaps:def}{linear} part of a composition is $g \circ g'$,
and $\det(gg') = \det g \det g'$: direct$\circ$direct $=$
indirect$\circ$indirect $=$ direct,
direct$\circ$indirect $=$ indirect --- the sign rule of $\pm1$.

\textbf{6.} $z \mapsto \eu^{\iu\theta}z$ sends $1$ to
$(\cos\theta, \sin\theta)$ and $\iu$ to
$\iu\eu^{\iu\theta} = (-\sin\theta, \cos\theta)$: the columns of
$R_\theta$. And $z \mapsto \eu^{\iu\theta}\conj z$ sends $1$ to
$(\cos\theta, \sin\theta)$ and $\iu$ to $-\iu\eu^{\iu\theta} =
(\sin\theta, -\cos\theta)$: the columns of $S_\theta$.

\textbf{7.} $a = 1$: $f = t_b$. $a \neq 1$: the fixed-point
equation $az_0 + b = z_0$ has the unique solution $z_0 = b/(1 -
a)$, and then
\[
f(z) - z_0 = az + b - (az_0 + b) = a(z - z_0):
\]
in the frame centered at $z_0$, $f$ is the multiplication by $a
= \eu^{\iu\arg a}$: the rotation of center $z_0$ and angle
$\arg a$.

\textbf{8.} $f(f(z)) = a\conj{(a\conj z + b)} + b = a\conj a\,z
+ a\conj b + b = z + v$: $f^2 = t_v$. Commutation: $f(z + v) =
a\conj z + a\conj v + b$ and $a\conj v = a\conj{(a\conj b + b)}
= b + a\conj b = v$, so $f\circ t_v = t_v\circ f$.

\emph{Case $v = 0$}: $f^2 = \mathrm{id}$. For any $z$, the
midpoint $m = \frac{z + f(z)}2$ satisfies (isometries being
affine, Part I) $f(m) = \frac{f(z) + f^2(z)}{2} = m$: fixed
points exist. Conjugating by the translation to a fixed point,
$f$ becomes a \omterm{def:b1:linmaps:def}{linear} indirect \omterm{def:b1:euclid:isometry}{isometry}, i.e.\ some $S_\theta$
(\cref{thm:b1:euclid:plane}): a reflection in a line.

\emph{Case $v \neq 0$}: $r = t_{-v/2}\circ f$ is indirect and
\[
r^2 = t_{-v/2}\,f\,t_{-v/2}\,f = t_{-v/2}\,t_{-v/2}\,f^2
= t_{-v}\,t_v = \mathrm{id}
\]
(using the commutation), so $r$ is a reflection, and $f =
t_{v/2}\circ r$. Its axis is parallel to $v$: $r$ commutes with
$t_v$ (both $f$ and $t_{v/2}$ do), so $t_v$ \omterm{def:b1:logic:map}{maps} the axis (the
fixed line of $r$) to itself, forcing $v$ to direct it. Hence
$f$ is a \omterm{pb:b1:euclid:1}{glide reflection} with glide vector $v/2$. Every
\omterm{def:b1:euclid:isometry}{isometry} is now classified: translation or rotation (direct),
reflection or \omterm{pb:b1:euclid:1}{glide reflection} (indirect).

\textbf{9.} Rotations $f(z) = az + b$, $g(z) = a'z + b'$ with
$a = \eu^{\iu\alpha}$, $a' = \eu^{\iu\beta}$: $g\circ f(z) =
a'az + (a'b + b')$ has \omterm{def:b1:linmaps:def}{linear} coefficient
$\eu^{\iu(\alpha+\beta)}$: a rotation of angle $\alpha + \beta$
by question 7, or a translation when $\eu^{\iu(\alpha+\beta)} =
1$. Reflections $r_i(z) = a_i\conj z + b_i$ with $a_i =
\eu^{\iu\theta_i}$ (axis at angle $\theta_i/2$):
\[
r_2\circ r_1(z) = a_2\conj{a_1}\,z + (a_2\conj{b_1} + b_2),
\]
direct with angle $\theta_2 - \theta_1 = 2\bigl(\tfrac{\theta_2}2
- \tfrac{\theta_1}2\bigr)$: twice the angle between the axes;
parallel axes ($\theta_1 = \theta_2$) give a translation.

\textbf{10.} A rotation of center $c$ and angle $\theta$ is the
product of two reflections in lines through $c$ making angle
$\theta/2$ (question 9, read backwards); a translation $t_v$ is
the product of two reflections in parallel lines \omterm{def:b1:euclid:orthogonal}{orthogonal} to
$v$ at distance $\norm v/2$; a reflection is one reflection; a
\omterm{pb:b1:euclid:1}{glide reflection} is a reflection composed with a translation,
hence three. Maximum: three.

\textbf{11.} $a = \iu$, $b = 1 + \iu$: $v = a\conj b + b =
\iu(1 - \iu) + 1 + \iu = (1 + \iu) + (1 + \iu) = 2 + 2\iu \neq
0$: a \omterm{pb:b1:euclid:1}{glide reflection} with glide vector $v/2 = 1 + \iu$. The
reflection $r = t_{-(1+\iu)}\circ f$ is $r(z) = \iu\conj z$,
whose axis is the line at angle $\frac12\arg\iu = \frac\pi4$:
the line $y = x$. So $f$ is the \omterm{pb:b1:euclid:1}{glide reflection} of axis $y =
x$ and vector $(1, 1)$.

\textbf{12.} $f(z) = \iu z$ and $g(z) = 1 + \iu(z - 1) = \iu z
+ 1 - \iu$, so
\[
g\circ f(z) = \iu(\iu z) + 1 - \iu = -z + 1 - \iu :
\]
\omterm{def:b1:linmaps:def}{linear} coefficient $-1 = \eu^{\iu\pi}$, a rotation of angle
$\pi$ (a half-turn), of center $z_0 = \frac{1 - \iu}{1 - (-1)}
= \frac{1 - \iu}{2}$.

\textbf{13.} $r_2\circ r_1(z) = \iu\,\conj{\conj z} = \iu z$:
the rotation of angle $\frac\pi2$ about $0$. The axes (the real
axis, angle $0$; the line $y = x$, angle $\frac\pi4$) meet at
angle $\frac\pi4$, and twice that is $\frac\pi2$: question 9
confirmed.

\textbf{14.} With $f = t_a\circ g$, $g$ \omterm{def:b1:linmaps:def}{linear}, and $\sum_i
\lambda_i = 1$:
\[
f\Bigl(\sum_i \lambda_i x_i\Bigr)
= a + \sum_i \lambda_i\,g(x_i)
= \sum_i \lambda_i\bigl(a + g(x_i)\bigr)
= \sum_i \lambda_i f(x_i) .
\]

\textbf{15.} For $h \in G$, using question 14 (the coefficients
$\frac1{\abs G}$ sum to $1$):
\[
h(c) = \frac{1}{\abs G}\sum_{g \in G} h\bigl(g(x_0)\bigr)
= \frac{1}{\abs G}\sum_{g' \in G} g'(x_0) = c ,
\]
because $g \mapsto hg$ is a bijection of $G$ onto itself.

\textbf{16.} The conjugates $t_{-c}\,g\,t_{c}$ ($g \in G$) form
a \omterm{def:b1:structures:group}{group} of isometries fixing $0$, hence of \emph{\omterm{def:b1:linmaps:def}{linear}}
isometries (question 3): a finite \omterm{def:b1:structures:subgroup}{subgroup} of $O(\R^2)$. If
some $s \in G$ is indirect, the \omterm{def:b1:logic:map}{map} $g \mapsto sg$ sends $G^+$
\omterm{def:b1:logic:inj}{injectively} into $G\setminus G^+$ and $h \mapsto s^{-1}h$ is its
inverse (sign rule of question 5): $\abs G = 2\,\abs{G^+}$;
otherwise $G = G^+$.

\textbf{17.} If $G^{+} = \{\mathrm{id}\}$, it is $C_1$.
Otherwise write its elements as $R_\theta$, $\theta \in
\intco{0}{2\pi}$, and let $\theta_0$ be the smallest positive
angle occurring. For $R_\theta \in G^{+}$, euclidean division
$\theta = k\theta_0 + \rho$ with $0 \leq \rho < \theta_0$ gives
$R_\rho = R_\theta R_{\theta_0}^{-k} \in G^{+}$, so $\rho = 0$
by minimality: $G^{+} = \langle R_{\theta_0}\rangle$. Dividing
$2\pi$ by $\theta_0$ the same way shows $2\pi = n\theta_0$:
$G^{+} \cong C_n$, generated by the rotation of angle
$\frac{2\pi}n$.

\textbf{18.} $G = G^{+} \cup sG^{+}$ by the counting of
question 16. With $s(z) = a\conj z$ and $r(z) = \omega z$
(\omterm{def:b1:linmaps:forms}{linear forms}, after question 16; $\abs a = \abs\omega = 1$):
\[
s r s(z) = a\,\conj{\omega\,a\conj z}
= a\conj\omega\conj a\,z = \conj\omega\,z = r^{-1}(z) .
\]
Each element $sr^k$ of $sG^{+}$ is indirect and satisfies
$(sr^k)^2 = (sr^ks)\,r^k = r^{-k}r^k = \mathrm{id}$: an
involutive indirect \omterm{def:b1:euclid:isometry}{isometry} fixing $c$, i.e.\ a reflection. So
$G$ consists of $n$ rotations and $n$ reflections with the
relations $r^n = s^2 = \mathrm{id}$, $srs = r^{-1}$: the
\omterm{pb:b1:euclid:1}{dihedral group} $D_n$.

\textbf{19.} Combining: a finite \omterm{def:b1:structures:group}{group} of plane isometries
fixes a point $c$ (question 15), reduces to a finite \omterm{def:b1:structures:subgroup}{subgroup}
of $O(2)$ (question 16), and is $C_n$ if it contains only
rotations (question 17), $D_n$ otherwise (question 18):
\emph{Leonardo's theorem}.

\textbf{20.} A translation $t_v \neq \mathrm{id}$ has powers
$t_{kv}$, all distinct ($kv$ pairwise distinct for $v \neq 0$):
infinite order. A \omterm{pb:b1:euclid:1}{glide reflection} $f$ has $f^2 = t_v$ with $v
\neq 0$: again infinite order. Neither fits in a finite \omterm{def:b1:structures:group}{group}
--- consistently with questions 15--19, which produced only
rotations and reflections.

\textbf{21.} The $2n$ \omterm{def:b1:logic:map}{maps} preserve the vertex \omterm{def:b1:logic:sets}{set}: $\omega^k
\omega^j = \omega^{k+j}$ and $\omega^k\conj{\omega^j} =
\omega^{k-j}$; being isometries, they preserve the polygon
(segments go to segments). Conversely, a symmetry preserves the
barycenter $0$ of the vertices (question 14), hence is \omterm{def:b1:linmaps:def}{linear},
and permutes the vertices (they are the points of the polygon
at maximal distance from $0$). A \omterm{def:b1:linmaps:def}{linear} \omterm{def:b1:euclid:isometry}{isometry} sending the
vertex $1$ to $\omega^k$ is $z \mapsto \omega^k z$ if direct,
$z \mapsto \omega^k\conj z$ if indirect (its matrix is
determined by one column and the sign): at most $2n$
symmetries, so exactly the $D_n$ above.

\textbf{22.} $C_1$: a scalene triangle (no nontrivial
symmetry). $D_1$: an isoceles, non-equilateral triangle (one
reflection). $D_2$: a non-square rectangle (identity, the
half-turn about the center, the two axis reflections). $C_3$: a
triskelion --- three congruent bent arms attached at $120$-degree
\omterm{prop:b1:reals:intervals}{intervals}; the bend kills every reflection but leaves the
rotations of order $3$.

\textbf{23.} With $\omega = \iu$: the rotations $z \mapsto z$,
$\iu z$, $-z$, $-\iu z$ (angles $0, \frac\pi2, \pi,
\frac{3\pi}2$), and the reflections
\[
z \mapsto \conj z \ (\text{axis } y = 0), \quad
\iu\conj z \ (y = x), \quad
-\conj z \ (x = 0), \quad
-\iu\conj z \ (y = -x):
\]
the two diagonals and the two medians of the square --- the
\omterm{pb:b1:euclid:1}{dihedral group} $D_4$, of order $8$.

\textbf{24.} Write $a = \eu^{\iu\theta} \neq 1$: $f(z) = az +
c_1(1 - a)$ and $g(z) = az + c_2(1 - a)$. Then
\[
f\circ g(z) - g\circ f(z)
= (1-a)\bigl(ac_2 + c_1 - ac_1 - c_2\bigr)
= -(1 - a)^2(c_2 - c_1) \neq 0 ,
\]
so $f g \neq g f$. Both compositions have \omterm{def:b1:linmaps:def}{linear} coefficient
$a^2$, so $(fg)\circ(gf)^{-1}$ has \omterm{def:b1:linmaps:def}{linear} coefficient $1$: it is
the translation by $fg(z) - gf(z)$, the nonzero constant above.
A \omterm{def:b1:structures:group}{group} containing $f$ and $g$ contains this translation and
all its powers: it is infinite. Two rotation centers are one too
many for a finite \omterm{def:b1:structures:group}{group} --- the geometric heart of Leonardo's
theorem.

\textbf{25.} (i) Questions 3--4 show every distance-preserving
\omterm{def:b1:logic:map}{map} is affine with \omterm{def:b1:euclid:orthogonal}{orthogonal} \omterm{def:b1:linmaps:def}{linear} part, and question 15 pins
every finite \omterm{def:b1:structures:group}{group} to a fixed point: both results convert
metric geometry into linear algebra at the origin. (ii) The
complete list is translation, rotation, reflection, \omterm{pb:b1:euclid:1}{glide
reflection}; the determinant of the \omterm{def:b1:linmaps:def}{linear} part separates direct
from indirect, and the existence of fixed points separates the
two types within each parity. (iii) By question 9, two
reflections compose to any rotation or translation, so
reflections generate the whole \omterm{def:b1:structures:group}{group} --- at most three suffice
for any \omterm{def:b1:euclid:isometry}{isometry}. (iv) On the finite scale only two families
survive, the cyclic and \omterm{pb:b1:euclid:1}{dihedral groups}, which is why rosette
ornaments come in exactly two kinds (with or without mirror
axes). Part II proved the \emph{classification of plane
isometries}; Part IV proved \emph{Leonardo's theorem}.
