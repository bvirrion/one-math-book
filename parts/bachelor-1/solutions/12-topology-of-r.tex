\section*{Chapter \ref{ch:b1:topology} --- Topology of the Real Line}

\begin{solution}{exo:b1:topology:1}
$\intoo{0}{1} \cup \intoo{2}{3}$: \omterm{def:b1:topology:open}{open} (union of \omterm{def:b1:topology:open}{open sets}), not
\omterm{def:b1:topology:closed}{closed} ($\frac 1n \to 0$ outside).

$\intco{0}{1}$: neither. Not \omterm{def:b1:topology:open}{open} (no \omterm{prop:b1:reals:intervals}{interval} around $0$ inside);
not \omterm{def:b1:topology:closed}{closed} ($1 - \frac1n \to 1 \notin$ \omterm{def:b1:logic:sets}{set}).

$\{0\} \cup \intcc{1}{2}$: \omterm{def:b1:topology:closed}{closed} (finite union of \omterm{def:b1:topology:closed}{closed sets}), not
\omterm{def:b1:topology:open}{open} (fails at $0$).

$\R \setminus \Z$: \omterm{def:b1:topology:open}{open} ($\Z$ is \omterm{def:b1:topology:closed}{closed}), not \omterm{def:b1:topology:closed}{closed}: the sequence
$\bigl(\frac 1n\bigr)$ lies in it, but its limit $0$ belongs to $\Z$,
i.e.\ escapes the \omterm{def:b1:logic:sets}{set}.

$\Q \cap \intoo{0}{1}$: neither. Not \omterm{def:b1:topology:open}{open}: every \omterm{prop:b1:reals:intervals}{interval} around a
rational contains irrationals. Not \omterm{def:b1:topology:closed}{closed}: it contains sequences
tending to the irrational $\frac{\sqrt 2}{2}$ (\omterm{def:b1:topology:dense}{density}).
\end{solution}

\begin{solution}{exo:b1:topology:2}
$A = \intoc{0}{1} \cup \{2\}$: $\mathring A = \intoo{0}{1}$,
$\overline A = \intcc{0}{1} \cup \{2\}$, $\partial A = \{0, 1, 2\}$.

$A = \R \setminus \Q$: $\mathring A = \emptyset$ (every \omterm{prop:b1:reals:intervals}{interval}
contains rationals), $\overline A = \R$ (\omterm{def:b1:topology:dense}{density} of the
irrationals), $\partial A = \R$.

$A = \bigl\{\frac{(-1)^n n}{n+1}\bigr\}$: the even terms tend to $1$,
the odd terms to $-1$, neither belongs to $A$. $\mathring A =
\emptyset$ (isolated points), $\overline A = A \cup \{-1, 1\}$,
$\partial A = \overline A$.
\end{solution}

\begin{solution}{exo:b1:topology:3}
\emph{Complements.} $F = \{a_1 < a_2 < \dots < a_k\}$: the complement
is the union of the \omterm{def:b1:topology:open}{open} \omterm{prop:b1:reals:intervals}{intervals} $\intoo{-\infty}{a_1}$,
$\intoo{a_i}{a_{i+1}}$, $\intoo{a_k}{+\infty}$ --- \omterm{def:b1:topology:open}{open} by
\cref{prop:b1:topology:openstable}.

\emph{Sequences.} Let $u_n \in F$, $u_n \to \ell$. With $\varepsilon
= \min\{\abs{a_i - a_j} : i \neq j\}/2 > 0$ (or any $\varepsilon$ if
$F$ is a singleton): beyond some rank, all terms are within
$\varepsilon$ of $\ell$, hence within $2\varepsilon$ of each other,
which forces them to be one single $a_i$ from that rank on; then
$\ell = a_i \in F$.
\end{solution}

\begin{solution}{exo:b1:topology:4}
$\subseteq$: $\overline A \cup \overline B$ is \omterm{def:b1:topology:closed}{closed} (finite union)
and contains $A \cup B$, so it contains the \emph{smallest} \omterm{def:b1:topology:closed}{closed}
superset $\overline{A \cup B}$. $\supseteq$: $A \subseteq A \cup B$
gives $\overline A \subseteq \overline{A \cup B}$ (\omterm{def:b1:topology:closure}{closure} is
monotone: adherent points of $A$ are adherent to the larger \omterm{def:b1:logic:sets}{set}), and
likewise for $B$.

Counterexample for intersections: $A = \intoo{0}{1}$, $B =
\intoo{1}{2}$: $\overline{A \cap B} = \overline\emptyset = \emptyset$
but $\overline A \cap \overline B = \{1\}$.
\end{solution}

\begin{solution}{exo:b1:topology:5}
Use the sequential characterization
(\cref{thm:b1:topology:seqclosed}). Let $S = \{u_n\} \cup \{\ell\}$
and $(v_k)$ a sequence in $S$ with $v_k \to m$; show $m \in S$. Two
cases. If some value $v \in S$ is taken by $(v_k)$ infinitely often,
then a constant \omterm{def:b1:seq:subsequence}{subsequence} gives $m = v \in S$. Otherwise every
value is taken finitely often; in particular, for each $n$, the term
$u_n$ appears finitely often, and $\ell$ too. Then for every $N$, the
indices $k$ with $v_k \in \{u_0, \dots, u_N, \ell\}$ are finitely
many: the remaining $v_k$ are terms $u_n$ with $n > N$. Given
$\varepsilon > 0$, choose $N$ with $\abs{u_n - \ell} \leq
\varepsilon$ for $n > N$: all but finitely many $v_k$ satisfy
$\abs{v_k - \ell} \leq \varepsilon$. Hence $v_k \to \ell$, so $m =
\ell \in S$.
\end{solution}

\begin{solution}{exo:b1:topology:6}
$U + A = \bigcup_{a \in A} (U + a)$, and each translate $U + a$ is
\omterm{def:b1:topology:open}{open} (translating the \omterm{prop:b1:reals:intervals}{interval} certificates). A union of \omterm{def:b1:topology:open}{open sets} is
\omterm{def:b1:topology:open}{open} (\cref{prop:b1:topology:openstable}).

\omterm{def:b1:topology:closed}{Closed sets}: $\Z$ and $\sqrt 2\,\Z$ are \omterm{def:b1:topology:closed}{closed} (as $\alpha\Z$:
convergent sequences are eventually constant, cf.\
\cref{ex:b1:topology:closed}). Their sum $\Z + \sqrt 2\,\Z$ is \omterm{def:b1:topology:dense}{dense}
in $\R$ (\cref{exo:b1:reals:9}) but is not $\R$ (it is countable, or
simply: $\frac{\sqrt 2}{2} \notin \Z + \sqrt2\Z$, else $\sqrt 2$
would be rational --- writing $\frac{\sqrt2}{2} = m + n\sqrt 2$
forces $(2n - 1)\sqrt 2 = -2m$, so $\sqrt 2 \in \Q$ unless $n =
\frac12$, impossible). A \omterm{def:b1:topology:dense}{dense} proper subset is not \omterm{def:b1:topology:closed}{closed}: its
\omterm{def:b1:topology:closure}{closure} is $\R \neq$ itself.
\end{solution}

\begin{solution}{exo:b1:topology:7}
$\Z$: the \omterm{def:b1:topology:open}{neighborhood} $\intoo{n - \frac12}{n + \frac12}$ of $n$
meets $\Z$ only at $n$.

$A = \{\frac1n : n \in \N^*\}$: around $\frac 1n$, the \omterm{prop:b1:reals:intervals}{interval} of
radius $\frac{1}{n} - \frac{1}{n+1} = \frac{1}{n(n+1)}$ (halved, say)
isolates it from its neighbors --- all points isolated. Yet $0 \in
\overline A \setminus A$: isolated points do not prevent adherent
outsiders.

If all points of $A$ are isolated: no point of $A$ is \omterm{def:b1:topology:closure}{interior}, since
an \omterm{def:b1:topology:closure}{interior} point has a whole \omterm{prop:b1:reals:intervals}{interval} of $A$-neighbors around it
(an \omterm{prop:b1:reals:intervals}{interval} is infinite), contradicting isolation. So $\mathring A
= \emptyset$.
\end{solution}

\begin{solution}{exo:b1:topology:8}
If $A \subseteq \intcc{-M}{M}$, the \omterm{def:b1:topology:closed}{closed set} $\intcc{-M}{M}$
contains $A$, hence contains $\overline A$ (smallest \omterm{def:b1:topology:closed}{closed}
superset): $\overline A$ is bounded.

Let $s = \sup A$ (finite). Since $A \subseteq \overline A$, $\sup
\overline A \geq s$. Conversely, $\overline A \subseteq
\intoc{-\infty}{s}$: the half-line is \omterm{def:b1:topology:closed}{closed} and contains $A$; so
every element of $\overline A$ is $\leq s$, giving $\sup\overline A
\leq s$. Equality.

$s \in \overline A$: by the $\varepsilon$-characterization
(\cref{prop:b1:reals:epsilon}), every \omterm{prop:b1:reals:intervals}{interval} $\intoo{s -
\varepsilon}{s + \varepsilon}$ contains an element of $A$: $s$ is
adherent.
\end{solution}

\begin{solution}{exo:b1:topology:9}
Suppose $A$ is \omterm{def:b1:topology:open}{open} and \omterm{def:b1:topology:closed}{closed}, with $a \in A$ and $b \in \R
\setminus A$; without loss of generality $a < b$. The \omterm{def:b1:logic:sets}{set} $B = A \cap
\intcc{a}{b}$ is nonempty ($a$), bounded: let $s = \sup B$. By
\cref{exo:b1:topology:8}, $s \in \overline B \subseteq \overline A =
A$ ($A$ \omterm{def:b1:topology:closed}{closed}). Note $s \leq b$, and since $b \notin A$: $s < b$.
Now $A$ is \omterm{def:b1:topology:open}{open}: some \omterm{prop:b1:reals:intervals}{interval} $\intoo{s - r}{s + r}$ lies in $A$,
and we may take $r < b - s$. Then $s + \frac{r}{2}$ belongs to $A
\cap \intcc{a}{b} = B$ and exceeds $s$ --- contradicting $s = \sup
B$. Hence no such pair $(a, b)$ exists: one of $A$, $\R \setminus A$
is empty.
\end{solution}

\begin{solution}{exo:b1:topology:10}
$I_x$ is a union of \omterm{def:b1:topology:open}{open} \omterm{prop:b1:reals:intervals}{intervals} all containing $x$: it is \omterm{def:b1:topology:open}{open},
and it is an \omterm{prop:b1:reals:intervals}{interval}, being convex --- if $u < z < v$ with $u, v \in
I_x$, then $u$ and $v$ lie in \omterm{def:b1:topology:open}{open} subintervals $J_u \ni x$, $J_v \ni
x$ of $U$, and $J_u \cup J_v$ is an \omterm{prop:b1:reals:intervals}{interval} (both contain $x$)
inside $U$ containing $z$; so $z \in I_x$
(\cref{prop:b1:reals:intervals}).

If $I_x \cap I_y \neq \emptyset$: $I_x \cup I_y$ is then an \omterm{def:b1:topology:open}{open}
\omterm{prop:b1:reals:intervals}{interval} (convex: two overlapping \omterm{prop:b1:reals:intervals}{intervals}) contained in $U$
containing $x$ and $y$, so $I_x \cup I_y \subseteq I_x$ and
$\subseteq I_y$ by maximality of each: $I_x = I_y$.

So $U$ is the disjoint union of the distinct \omterm{def:b1:logic:sets}{sets} $I_x$ (each $x \in
U$ lies in its own $I_x$). Countability: each nonempty \omterm{def:b1:topology:open}{open} \omterm{prop:b1:reals:intervals}{interval}
$I$ of the family contains a rational $q_I$
(\cref{thm:b1:reals:density}), and distinct disjoint \omterm{prop:b1:reals:intervals}{intervals} get
distinct rationals: the family injects into $\Q$, which is countable
(it is indexed by pairs of integers). Hence at most countably many
\omterm{prop:b1:reals:intervals}{intervals}.
\end{solution}

\begin{solution}{exo:b1:topology:11}
$\overline A = A \cup A'$. ($\supseteq$) $A \subseteq \overline
A$ always; and if $x \in A'$, every \omterm{def:b1:topology:open}{neighborhood} of $x$ meets $A
\setminus \{x\} \subseteq A$, so $x$ is adherent. ($\subseteq$)
Let $x \in \overline A$. If $x \in A$, done. If $x \notin A$,
every \omterm{def:b1:topology:open}{neighborhood} of $x$ meets $A = A \setminus \{x\}$: $x \in
A'$.

Consequently $A$ \omterm{def:b1:topology:closed}{closed} $\iff$ $A = \overline A = A \cup A'$
$\iff$ $A' \subseteq A$.

$A = \{\frac 1n\}$: $0$ is an accumulation point ($\frac 1n \to
0$, terms $\neq 0$); each $\frac 1n$ is isolated
(\cref{exo:b1:topology:7}), so not in $A'$; and a point $x
\notin A \cup \{0\}$ has a whole \omterm{prop:b1:reals:intervals}{interval} avoiding $A$ (between
the two neighbors of $x$ in $A \cup \{0\}$, or beyond $1$).
Hence $A' = \{0\}$.

$\Z' = \emptyset$: every integer is isolated, every non-integer
has a \omterm{def:b1:topology:open}{neighborhood} inside $\R \setminus \Z$.

$\Q' = \R$: every \omterm{prop:b1:reals:intervals}{interval} around any real contains infinitely
many rationals (\cref{thm:b1:reals:density}), in particular one
different from the center.
\end{solution}

\begin{solution}{exo:b1:topology:12}
\begin{enumerate}
  \item For every $a \in A$: $\abs{x - a} \leq \abs{x - y} +
        \abs{y - a}$, so $d_A(x) \leq \abs{x - y} + \abs{y - a}$;
        taking the \omterm{def:b1:reals:bounds}{infimum} over $a$: $d_A(x) \leq \abs{x - y} +
        d_A(y)$. Exchanging $x$ and $y$ gives the other
        inequality: $\abs{d_A(x) - d_A(y)} \leq \abs{x - y}$.
  \item $d_A(x) = 0$ $\iff$ for every $\varepsilon > 0$ there is
        $a \in A$ with $\abs{x - a} < \varepsilon$ $\iff$ every
        \omterm{prop:b1:reals:intervals}{interval} around $x$ meets $A$ $\iff$ $x \in \overline A$.
        If $F$ is \omterm{def:b1:topology:closed}{closed} and $x \notin F = \overline F$, then
        $d_F(x) \neq 0$, i.e.\ $d_F(x) > 0$.
  \item If $d_F(x) < \varepsilon$, set $r = \varepsilon -
        d_F(x) > 0$: for $\abs{y - x} < r$, part (1) gives
        $d_F(y) \leq d_F(x) + \abs{x - y} < \varepsilon$: the
        \omterm{prop:b1:reals:intervals}{interval} $\intoo{x - r}{x + r}$ lies in $V_\varepsilon$,
        which is therefore \omterm{def:b1:topology:open}{open}; it contains $F$ since $d_F = 0$
        there. Finally $x \in \bigcap_{\varepsilon>0}
        V_\varepsilon$ $\iff$ $d_F(x) < \varepsilon$ for all
        $\varepsilon$ $\iff$ $d_F(x) = 0$ $\iff$ $x \in \overline
        F = F$. Since $\bigcap_{\varepsilon > 0} V_\varepsilon =
        \bigcap_{n \geq 1} V_{1/n}$, every \omterm{def:b1:topology:closed}{closed set} is a
        countable intersection of \omterm{def:b1:topology:open}{open sets}.
\end{enumerate}
\end{solution}

\begin{solution}{pb:b1:topology:1}
\textbf{1.} $C_1 = \intcc{0}{\frac13} \cup \intcc{\frac23}{1}$
and
\[
C_2 = \intcc{0}{\tfrac19} \cup \intcc{\tfrac29}{\tfrac13}
\cup \intcc{\tfrac23}{\tfrac79} \cup \intcc{\tfrac89}{1} .
\]
Induction: if $C_n$ is a disjoint union of $2^n$ \omterm{def:b1:topology:closed}{closed} segments
of length $3^{-n}$, deleting the \omterm{def:b1:topology:open}{open} middle third of each
leaves two \omterm{def:b1:topology:closed}{closed} segments of length $3^{-n-1}$ per parent:
$2^{n+1}$ segments, pairwise disjoint (children of distinct
parents are separated because the parents were; children of one
parent are separated by the removed gap).

\textbf{2.} Each $C_n$ is a finite union of segments, hence
\omterm{def:b1:topology:closed}{closed}; $C = \bigcap C_n$ is an intersection of \omterm{def:b1:topology:closed}{closed sets}:
\omterm{def:b1:topology:closed}{closed} (\cref{def:b1:topology:closed}); bounded ($\subseteq
\intcc{0}{1}$): compact by \cref{thm:b1:topology:compact}.
Nonempty: $0$ lies in the leftmost segment of every $C_n$. Let
$a$ be an endpoint of a segment $S$ of $C_n$. For $m \leq n$,
$a \in C_n \subseteq C_m$. For the later stages: the middle-third
deletion never removes an endpoint, and $a$ is again an endpoint
of one of the two children of $S$ (the child touching $a$); by
induction $a \in C_m$ for all $m \geq n$: $a \in C$.

\textbf{3.} Total length of $C_n$: $2^n \cdot 3^{-n} =
(\frac23)^n \to 0$. Given $\varepsilon > 0$, choose $n$ with
$(\frac23)^n \leq \varepsilon$: then $C \subseteq C_n$, a union
of finitely many segments of total length $\leq \varepsilon$.

\textbf{4.} Let $I \subseteq C$ be an \omterm{prop:b1:reals:intervals}{interval} with two distinct
points. For every $n$: $I \subseteq C_n$, and $I$, being convex,
must lie inside a \emph{single} segment of $C_n$ (meeting two
segments would force $I$ to contain a point of the gap between
them, which is outside $C_n$). Hence the length of $I$ is $\leq
3^{-n}$ for all $n$: contradiction. So the only \omterm{prop:b1:reals:intervals}{intervals}
inside $C$ are empty or singletons; in particular no
$\intoo{x-r}{x+r}$ fits inside $C$: $\mathring C = \emptyset$.
As $C$ is \omterm{def:b1:topology:closed}{closed}, $\overline C = C$ has empty \omterm{def:b1:topology:closure}{interior}: $C$ is
nowhere \omterm{def:b1:topology:dense}{dense}.

\textbf{5.} Write $\varphi_0(x) = \frac x3$ and $\varphi_2(x) =
\frac{2 + x}{3}$, increasing affine bijections of $\intcc{0}{1}$
onto $\intcc{0}{\frac13}$ and $\intcc{\frac23}{1}$. Claim:
$C_{n+1} = \varphi_0(C_n) \cup \varphi_2(C_n)$. For $n = 0$ this
is question 1. Induction: an increasing affine \omterm{def:b1:logic:map}{map} sends the
middle third of a segment to the middle third of the image
segment, so deleting middle thirds commutes with $\varphi_0$ and
$\varphi_2$; applying the deletion step to $C_{n+1} =
\varphi_0(C_n) \cup \varphi_2(C_n)$ yields $C_{n+2} =
\varphi_0(C_{n+1}) \cup \varphi_2(C_{n+1})$. Intersecting over
$n$: for $x \leq \frac13$, $x \in C \iff x \in \varphi_0(C_n)$
for all $n$ $\iff 3x \in \bigcap C_n = C$; likewise on
$\intcc{\frac23}{1}$; and no point of $\intoo{\frac13}{\frac23}$
lies in $C_1$. Hence $C = \varphi_0(C) \cup \varphi_2(C)$,
disjointly.

\textbf{6.} Induction on $n$; the case $n = 0$ says every $x \in
\intcc{0}{1}$ has a code, which is \cref{pb:b1:reals:1}
(question 9 for $x < 1$; $1 = (0.\overline 2)_3$). Suppose the
equivalence at rank $n$. If $x \in C_{n+1}$: by question 5, $x =
\varphi_i(z)$ with $z \in C_n$ and $i \in \{0, 2\}$; if $(e_k)$
is a code of $z$ with first $n$ digits $1$-free, then $(i, e_1,
e_2, \dots)$ has partial sums $\frac i3 + \frac13\sum_{k\leq m}
e_k 3^{-k} \to \varphi_i(z) = x$: a code of $x$ with first $n +
1$ digits $1$-free. Conversely, if $x$ has a code $(d_k)$ with
$d_1, \dots, d_{n+1} \in \{0, 2\}$: the shifted string $(d_2,
d_3, \dots)$ has some value $z \in \intcc{0}{1}$, its first $n$
digits are $1$-free, and the partial-sum computation read
backwards gives $x = \varphi_{d_1}(z)$; by induction $z \in
C_n$, so $x \in C_{n+1}$ by question 5. Finally: a fully
$1$-free code puts $x$ in every $C_n$, hence in $C$; conversely
if $x \in C$, then for every $n$ one of the at most two codes of
$x$ (\cref{pb:b1:reals:1}, question 11) has its first $n$ digits
$1$-free; one fixed code must work for arbitrarily large $n$
(pigeonhole between two codes), and a code whose first $n$
digits are $1$-free for arbitrarily large $n$ is $1$-free
outright.

\textbf{7.} $0 = (0.\overline 0)_3$, $1 = (0.\overline 2)_3$,
$\frac13 = (0.0\overline{2})_3$ (the improper twin of $(0.1)_3$),
$\frac23 = (0.2\overline{0})_3$. Geometric sums:
\[
(0.\overline{02})_3 = \sum_{j\geq1} \frac{2}{9^{\,j}}
= \frac{2/9}{1 - 1/9} = \frac14 ,
\qquad
(0.\overline{20})_3 = \sum_{j\geq1} \frac{2}{3\cdot 9^{\,j-1}}
= \frac{2/3}{1 - 1/9} = \frac34 ,
\]
both $1$-free: $\frac14, \frac34 \in C$. (The infinite sums
abbreviate suprema of partial sums, as in \cref{pb:b1:reals:1}.)
Endpoints of segments of $C_n$ are of the form $m/3^n$
(induction: children endpoints are parent endpoints or differ
from one by a multiple of $3^{-n-1}$). If $\frac14 =
\frac{m}{3^n}$ then $3^n = 4m$, and $4 \nmid 3^n$: impossible.
So $\frac14 \in C$ without ever being an endpoint.

\textbf{8.} Suppose $x$ had two distinct $1$-free codes. Having
two codes at all means (\cref{pb:b1:reals:1}, question 11, base
$3$) that $x = m/3^N \in \intoo{0}{1}$ and the two codes are:
the terminating one, with last nonzero digit $d_N \in \{1, 2\}$
followed by $0$s, and its twin, with $d_N - 1$ at position $N$
followed by $2$s. If $d_N = 1$ the first contains a $1$; if $d_N
= 2$ the twin carries $d_N - 1 = 1$. Either way at most one of
the pair is $1$-free: contradiction. So each $x \in C$ has
exactly one $1$-free code (existence by question 6), and
distinct $\{0,2\}$-strings have distinct values. Every
$\{0,2\}$-string has value in $\intcc{0}{1}$ (partial sums $\leq
1$) with all prefixes $1$-free, hence value in every $C_n$,
i.e.\ in $C$: the value \omterm{def:b1:logic:map}{map} is a bijection from
$\{0,2\}$-strings onto $C$.

\textbf{9.} Let $(d^{(k)})$ be the $1$-free code of $x_k$ and
set $e_k = 2 - d^{(k)}_k \in \{0, 2\}$: a $\{0,2\}$-string whose
value $y$ lies in $C$ and has $(e_k)$ as its unique $1$-free
code (question 8). For each $k$ the codes of $y$ and $x_k$
differ at position $k$, so $y \neq x_k$: no \omterm{def:b1:logic:map}{map} $\N^* \to C$ is
\omterm{def:b1:logic:inj}{surjective}. An uncountable \omterm{def:b1:logic:sets}{set} of length zero: bigness in
\omterm{def:b1:counting:card}{cardinality}, smallness in measure --- simultaneously.

\textbf{10.} Compactness: closedness of the infinite
intersection plus boundedness (question 2) --- the one property
not carried by a single containment. Length zero: $C \subseteq
C_n$ with total length $(\frac23)^n$ (question 3). Nowhere
\omterm{def:b1:topology:dense}{density}: $C \subseteq C_n$ forces \omterm{prop:b1:reals:intervals}{intervals} inside $C$ to have
length $\leq 3^{-n}$ (question 4). Uncountability rides on no
containment at all: it needs the full intersection structure,
encoded in the bijection of question 8.

\textbf{11.} Flip $d_n$ to $2 - d_n$: the new string is still a
$\{0,2\}$-string, so its value $x_n$ lies in $C$; the partial
sums beyond rank $n$ differ by exactly $2\cdot3^{-n}$, so
$\abs{x_n - x} = 2\cdot3^{-n}$. Thus $x_n \neq x$ and $x_n \to
x$: every point of $C$ is a limit of other points of $C$ --- $C$
is \emph{perfect}, with no isolated point.

\textbf{12.} By the induction of question 5, the segments of
$C_n$ are exactly the $\intcc{t}{t + 3^{-n}}$ where $t$ runs
over the values of length-$n$ $\{0,2\}$-strings. Given $x \in C$
with code $(d_k)$, the truncation $t_n$ (digits $d_1 \dots d_n$
then $0$s) is therefore a left endpoint, and $0 \leq x - t_n
\leq 3^{-n}$: endpoints are \omterm{def:b1:topology:dense}{dense} in $C$. They form a subset of
$\{m/3^n : m, n\}$, a \omterm{def:b1:logic:sets}{set} indexed by pairs of integers, hence
countable (as for $\Q$ in \cref{exo:b1:topology:10}). Since $C$
is uncountable (question 9), all but countably many points of
$C$ are not endpoints --- $\frac14$ (question 7) is the visible
tip of that iceberg.

\textbf{13.} Pick $n$ with $3^{-n} < y - x$. Both $x, y \in
C_n$, and they cannot lie in the same segment (length $3^{-n} <
y - x$): the removed gap between their segments provides $z$
with $x < z < y$ and $z \notin C_n \supseteq C$. Hence any two
points of $C$ are separated by the complement: the only convex
subsets of $C$ are singletons --- $C$ is totally disconnected.

\textbf{14.} If $(d_k)$ is the $1$-free code of $x$, the string
$(2 - d_k)$ is again a $\{0,2\}$-string, with partial sums
\[
\sum_{k=1}^{n} (2 - d_k)3^{-k} = (1 - 3^{-n}) - \sum_{k=1}^n
d_k 3^{-k} \longrightarrow 1 - x :
\]
so $1 - x \in C$. Thus $1 - C \subseteq C$, and applying the \omterm{def:b1:logic:map}{map}
twice gives $1 - C = C$: the \omterm{pb:b1:topology:1}{Cantor set} is symmetric about
$\frac12$.

\textbf{15.} The partial sums $t_n \to x$ and $t'_n \to x'$
(increasing sequences converge to their \omterm{def:b1:reals:bounds}{supremum}, i.e.\ the
value), so $t_n + t'_n \to x + x'$ by
\cref{thm:b1:seq:operations}. Given $y \in \intcc{0}{1}$ with
code $(e_k)$, choose $(a_k, b_k) = (0,0), (0,2), (2,2)$
according as $e_k = 0, 1, 2$: then $a_k + b_k = 2e_k$, the
strings $(a_k)$, $(b_k)$ are $\{0,2\}$-strings with values $x,
x' \in C$, and
\[
x + x' = \lim_n\,(t_n + t'_n) = \lim_n 2\sum_{k=1}^n e_k 3^{-k}
= 2y :
\]
every $y \in \intcc{0}{1}$ is the midpoint of two points of $C$.

\textbf{16.} Question 15 gives $\intcc{0}{2} = 2\,\intcc{0}{1}
\subseteq C + C$, and $C + C \subseteq \intcc{0}{1} +
\intcc{0}{1} = \intcc{0}{2}$: equality. Then, using $1 - C = C$:
\[
C - C = C + (C - 1) = (C + C) - 1 = \intcc{-1}{1} .
\]
Concrete instance: $1 = \frac14 + \frac34$, a sum of two
non-endpoint members of $C$.

\textbf{17.} Length measures how much of the line the \omterm{def:b1:logic:sets}{set} itself
occupies; it says nothing about the \omterm{def:b1:logic:sets}{set} of \emph{sums}, which is
the image of the two-parameter family $C \times C$ under $(x,
x') \mapsto x + x'$ --- the two digit strings are chosen
independently, and that freedom is exactly what fills
$\intcc{0}{2}$. No theorem bounds the length of a sumset by the
lengths of the summands, and $C$ is the proof that none can.

\textbf{18.} By \cref{pb:b1:reals:1} (question 18), $x$ is
rational iff its proper expansion is eventually periodic. The
$1$-free code of $x \in C$ is either that proper expansion or
the improper twin of a terminating one; a terminating string and
its twin (eventually constant $2$s) are both eventually
periodic, so periodicity of the $1$-free code is equivalent to
rationality of $x$. Long division of $\frac1{13}$ in base $3$
($r_0 = 1$): $3 = 13\cdot0 + 3$, $9 = 13\cdot0 + 9$, $27 =
13\cdot2 + 1$, and the remainder returns to $1$: digits
$\overline{002}$, so $\frac1{13} = (0.\overline{002})_3$,
$1$-free and periodic: a rational member of $C$. (Check:
$\frac{2/27}{1 - 1/27} = \frac{2}{26} = \frac1{13}$.)

\textbf{19.} The string with $d_k = 2$ at the triangular
positions $k = \frac{j(j+1)}{2}$ and $0$ elsewhere is a
$\{0,2\}$-string, so its value $x^*$ belongs to $C$ (question
8). It has infinitely many $2$s with gaps $j + 1 \to \infty$
between consecutive ones, so it is not eventually periodic (a
period $T$ would eventually force $2$s at gaps $\leq T$: the
growing-gaps argument of \cref{pb:b1:reals:1}, question 20); by
question 18, $x^* \notin \Q$. And by question 9 plus the
countability of $\Q$, all but countably many members of $C$ are
irrational: $x^*$ is the norm, not the exception.

\textbf{20.} Let $y \in \intcc{0}{1}$: it has a binary code
$(c_k)$ with $c_k \in \{0, 1\}$ (\cref{pb:b1:reals:1}, question
9, base $2$; $y = 1$ takes the all-$1$s string). Then $(2c_k)$
is a $\{0,2\}$-string, its value $x$ lies in $C$, and $h(x)$ is
the value of $(c_k)$, namely $y$: $h$ \omterm{def:b1:logic:map}{maps} $C$ onto
$\intcc{0}{1}$. If $C$ were the image of a \omterm{def:b1:logic:map}{map} from $\N^*$,
composing with $h$ would list all of $\intco{0}{1}$,
contradicting the diagonal theorem of \cref{pb:b1:reals:1}
(question 22): $C$ is uncountable, again. A length-zero \omterm{def:b1:logic:sets}{set}
surjecting onto a full segment.

\textbf{21.} By question 5, $C$ is the disjoint union of
$\varphi_0(C)$ and $\varphi_2(C)$, each a translate of the
scaled copy $\frac13 C$. Additivity, scaling and translation
invariance give
\[
L(C) = L(\varphi_0(C)) + L(\varphi_2(C))
= \tfrac13 L(C) + \tfrac13 L(C) = \tfrac23\,L(C),
\]
so $\frac13 L(C) = 0$: $L(C) = 0$. Self-similarity alone
sentences $C$ to length zero --- question 3 merely executed the
sentence.

\textbf{22.} A segment of length $l_n$ loses a central \omterm{prop:b1:reals:intervals}{interval}
of length $4^{-(n+1)}$, leaving two segments of length $l_{n+1}
= \frac{l_n - 4^{-(n+1)}}{2}$; from $l_0 = 1$, induction
confirms $l_n = \frac{2^n + 1}{2\cdot4^n}$: indeed
$\frac12\Bigl(\frac{2^n+1}{2\cdot4^n} - \frac{1}{4^{n+1}}\Bigr)
= \frac{2(2^n + 1) - 1}{2\cdot4^{n+1}} = \frac{2^{n+1} +
1}{2\cdot4^{n+1}}$, and $l_n > 0$ always: the construction never
starves. $K = \bigcap K_n$ is \omterm{def:b1:topology:closed}{closed} and bounded, hence compact;
an \omterm{prop:b1:reals:intervals}{interval} inside $K$ lies in one segment of $K_n$, of length
$l_n \to 0$: empty \omterm{def:b1:topology:closure}{interior}. Removed length: $\sum_{n\geq0} 2^n
\cdot 4^{-(n+1)} = \frac14\sum_{n\geq0}\bigl(\frac12\bigr)^n =
\frac12$, and each $K_n$ has total length $2^n l_n = \frac{2^n +
1}{2^{n+1}} > \frac12$. Now let finitely many \omterm{def:b1:topology:open}{open} \omterm{prop:b1:reals:intervals}{intervals}
have union $U \supseteq K$. By the companion \omterm{def:b1:logic:statement}{statement} of
\cref{ex:b1:topology:nested}, $U \supseteq K_n$ for some $n$;
admitting additivity of length on finite unions of \omterm{prop:b1:reals:intervals}{intervals},
the total length of the covering \omterm{prop:b1:reals:intervals}{intervals} is at least that of
$K_n$, which exceeds $\frac12$. So $K$ is nowhere \omterm{def:b1:topology:dense}{dense}, yet no
cheap cover exists: topological smallness (nowhere \omterm{def:b1:topology:dense}{dense}) and
metric smallness (length zero) are genuinely different notions,
and $K$ separates them.

\textbf{23.} Attainment: let $d = d(x, F)$ and pick $a_k \in F$
with $\abs{x - a_k} \leq d + \frac1k$: the $a_k$ are bounded, so
Bolzano--Weierstrass (\cref{thm:b1:seq:bw}) extracts
$a_{\varphi(k)} \to a$, with $a \in F$ ($F$ \omterm{def:b1:topology:closed}{closed},
\cref{thm:b1:topology:seqclosed}) and $\abs{x - a} = \lim
\abs{x - a_{\varphi(k)}} = d$. Now the maximum: if $y \in C$,
$d(y, C) = 0$; otherwise $y$ lies in a gap removed at some stage
$n \geq 1$, an \omterm{def:b1:topology:open}{open} \omterm{prop:b1:reals:intervals}{interval} of length $3^{-n}$ whose two
endpoints belong to $C$ (question 2), so $d(y, C) \leq
\frac{3^{-n}}{2} \leq \frac16$, with equality requiring $n = 1$
and $y$ at the center of the gap $\intoo{\frac13}{\frac23}$,
i.e.\ $y = \frac12$; and indeed $d\bigl(\frac12, C\bigr) =
\frac16$ since $C \cap \intoo{\frac13}{\frac23} = \emptyset$ and
$\frac13, \frac23 \in C$. Hence $\max_{y\in\intcc{0}{1}} d(y, C)
= \frac16$, attained exactly at $\frac12$.

\textbf{24.} The endpoints form a countable \omterm{def:b1:topology:dense}{dense} subset of $C$
(question 12): list them as a single sequence $(e_j)_{j\geq1}$,
a sequence in $C$. Its subsequential limits all lie in $C$ ($C$
\omterm{def:b1:topology:closed}{closed}). Conversely, fix $x \in C$: for each $n$, the segments
of the $C_m$ containing $x$ ($m \geq n$) have their endpoints
within $3^{-m} \leq 3^{-n}$ of $x$, so infinitely many distinct
endpoints lie within $3^{-n}$ of $x$; choose indices $j_1 < j_2
< \dots$ with $\abs{e_{j_n} - x} \leq 3^{-n}$: a \omterm{def:b1:seq:subsequence}{subsequence}
converging to $x$. So the \omterm{def:b1:logic:sets}{set} of subsequential limits of $(e_j)$
is exactly $C$ --- one sequence clustering at uncountably many
points, the opposite extreme from a convergent sequence, whose
cluster \omterm{def:b1:logic:sets}{set} is a singleton.

\textbf{25.} (i) The construction consumed: stability of \omterm{def:b1:topology:closed}{closed
sets} under arbitrary intersection (existence of $C$ as a \omterm{def:b1:topology:closed}{closed
set}), the compactness theorem \cref{thm:b1:topology:compact}
(questions 2, 22, 23), and the sequential characterizations of
closedness and adherence (the nested-compacts argument of
\cref{ex:b1:topology:nested} and question 23). (ii) The four
pairings: length zero yet uncountable (questions 3, 9); \omterm{def:b1:topology:closed}{closed}
yet with empty \omterm{def:b1:topology:closure}{interior} (question 4); perfect --- no isolated
point --- yet totally disconnected (questions 11, 13);
negligible yet with $C + C = \intcc{0}{2}$ (question 16). (iii)
The surjection $h$ of question 20, made continuous and
nondecreasing, becomes the devil's staircase in the theory of
continuous functions; and in the Year 3 volume's measure theory,
$C$ is the standard witness that ``negligible'' does not mean
``countable'', with its fat cousin (question 22) separating
``nowhere \omterm{def:b1:topology:dense}{dense}'' from ``negligible''.
\end{solution}
