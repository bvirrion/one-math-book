\section*{Capítulo \ref{ch:b1:fractions} --- Frações Racionais}

\begin{solution}{exo:b1:fractions:1}
$\dfrac{1}{X^2 - 1}$: simples postes $\pm 1$; cobrir:
$\dfrac{1/2}{X-1} - \dfrac{1/2}{X+1}$.

$\dfrac{X}{X^2 - 3X + 2} = \dfrac{X}{(X-1)(X-2)}$: encobrimento dá
$\frac{1}{1-2} = -1$ no $1$ e $\frac{2}{2-1} = 2$ no $2$:
$\dfrac{-1}{X-1} + \dfrac{2}{X-2}$.

$\dfrac{X^2+1}{X(X-1)}$: o grau é $0$, então existe um número inteiro
parte: \omterm{thm:b1:logic:partition}{divisão}, $ X^2 + 1 = (X^2 - X) + (X + 1)$, então $ F = 1 +
\frac{X+1}{X(X-1)}$. Encobrimento do restante: $\frac{1}{-1} = -1$
no $0$, $\frac{2}{1} = 2$ no $1$:
\[
F = 1 - \frac{1}{X} + \frac{2}{X-1} .
\]
\end{solution}

\begin{solution}{exo:b1:fractions:2}
$\dfrac{1}{X(X^2+1)}$: forma $\frac aX + \frac{bX + c}{X^2 + 1}$.
Encobrimento em $0$: $ a = 1$. Limite de $ xF $: $0 = a + b $, então $ b = -1$.
Avaliação em $ X = 1$: $\frac12 = 1 + \frac{c - 1}{2}$, então $ c = 0$:
\[
\frac{1}{X(X^2+1)} = \frac 1X - \frac{X}{X^2+1} .
\]

$\dfrac{X^3}{X^2+X+1}$: \omterm{thm:b1:logic:partition}{divisão}: $ X^3 = (X^2+X+1)(X - 1) + 1$, então
\[
\frac{X^3}{X^2+X+1} = X - 1 + \frac{1}{X^2 + X + 1} ,
\]
já na forma real decomposta (a quadrática tem negativo
discriminante).
\end{solution}

\begin{solution}{exo:b1:fractions:3}
$\dfrac{1}{X^2(X-1)}$: forma $\frac{a}{X^2} + \frac bX +
\frac{c}{X-1}$. Encobrimento em dobro pólo $0$:
$ a = \bigl[\frac{1}{X-1}\bigr]_{0} = -1$. Encobrimento em $1$: $ c = 1$.
Limite de $ xF $: $0 = b + c $, então $ b = -1$:
\[
\frac{1}{X^2(X-1)} = -\frac{1}{X^2} - \frac1X + \frac{1}{X-1} .
\]

$\dfrac{X+1}{(X-1)^3}$: com $ Y = X - 1$, o numerador é $ Y + 2$:
\[
\frac{Y + 2}{Y^3} = \frac{1}{Y^2} + \frac{2}{Y^3}
= \frac{1}{(X-1)^2} + \frac{2}{(X-1)^3} .
\]
\end{solution}

\begin{solution}{exo:b1:fractions:4}
O postes são os quartos raízes da unidade $1, \iu, -1, -\iu $, tudo
simples. Pólo simples fórmula com $ B' = 4X^3$: coeficiente em $ a $ é
$\frac{1}{4a^3} = \frac{a}{4a^4} = \frac a4$ (usando $ a^4 = 1$). Então,
acabou $\C $:
\[
\frac{1}{X^4 - 1}
= \frac{1/4}{X - 1} - \frac{1/4}{X + 1}
+ \frac{\iu/4}{X - \iu} - \frac{\iu/4}{X + \iu} .
\]
Agrupando o par \omterm{def:b1:complex:field}{conjugado} (denominador comum $ X^2 + 1$):
$\frac{\iu}{4}\bigl(\frac{1}{X-\iu} - \frac{1}{X+\iu}\bigr) =
\frac{\iu}{4}\cdot\frac{2\iu}{X^2+1} = \frac{-1/2}{X^2+1}$. Sobre
$\R $:
\[
\frac{1}{X^4 - 1}
= \frac{1/4}{X-1} - \frac{1/4}{X+1} - \frac{1/2}{X^2 + 1} .
\]
\emph{Verifique em $ X = 0$:} $-1 = -\frac14 - \frac14 - \frac12$.
\end{solution}

\begin{solution}{exo:b1:fractions:5}
$\dfrac{X^2}{(X^2+1)^2} = \dfrac{(X^2 + 1) - 1}{(X^2+1)^2} =
\dfrac{1}{X^2+1} - \dfrac{1}{(X^2+1)^2}$.

Daí um primitivo:
\[
\int \frac{x^2\,\dd x}{(x^2+1)^2}
= \arctan x - \frac12\Bigl(\arctan x + \frac{x}{x^2+1}\Bigr) + C
= \frac12\arctan x - \frac{x}{2(x^2+1)} + C .
\]
\end{solution}

\begin{solution}{exo:b1:fractions:6}
O postes $0, -1, \dots, -n $ são simples. Encobrimento em $-k $:
\[
c_k = \frac{n!}{\prod_{j \neq k} (-k + j)}
= \frac{n!}{\bigl(\prod_{j=0}^{k-1}(j - k)\bigr)
\bigl(\prod_{j=k+1}^{n}(j-k)\bigr)}
= \frac{n!}{(-1)^k k!\,(n-k)!} = (-1)^k \binom nk .
\]
Então
\[
\frac{n!}{X(X+1)\cdots(X+n)}
= \sum_{k=0}^{n} \frac{(-1)^k \binom nk}{X + k} .
\]
(Verificação de sanidade para $ n = 1$: $\frac{1}{X(X+1)} = \frac1X -
\frac{1}{X+1}$.)
\end{solution}

\begin{solution}{exo:b1:fractions:7}
$ P = X^n - 1$ tem o $ n $ raízes simples $\omega^k $
(\cref{thm:b1:complex:roots}), então a identidade logarítmica-derivada
de \cref{exo:b1:poly:11} lê
\[
\sum_{k=0}^{n-1} \frac{1}{X - \omega^k}
= \frac{P'(X)}{P(X)} = \frac{n X^{n-1}}{X^n - 1} .
\]
No $ X = 2$, $ n = 4$: lado direito $= \frac{4 \times 8}{15} =
\frac{32}{15}$. Lado esquerdo: $\frac{1}{2-1} + \frac{1}{2+1} +
\frac{1}{2 - \iu} + \frac{1}{2 + \iu} = 1 + \frac13 + \frac{4}{5} =
\frac{15 + 5 + 12}{15} = \frac{32}{15}$, usando $\frac{1}{2-\iu} +
\frac{1}{2+\iu} = \frac{4}{5}$.
\end{solution}

\begin{solution}{exo:b1:fractions:8}
Cobrir: $\dfrac{1}{k(k+1)(k+2)} = \dfrac{1/2}{k} - \dfrac{1}{k+1} +
\dfrac{1/2}{k+2}$. Reescreva como uma diferença telescópica:
\[
\frac{1}{k(k+1)(k+2)}
= \frac12\Bigl(\frac{1}{k(k+1)} - \frac{1}{(k+1)(k+2)}\Bigr),
\]
(expandir para verificar --- ou subtrair as duas decomposições). Resumindo:
\[
S_n = \frac12\Bigl(\frac{1}{1 \times 2} - \frac{1}{(n+1)(n+2)}\Bigr)
= \frac14 - \frac{1}{2(n+1)(n+2)}
\xrightarrow[n \to \infty]{} \frac14 .
\]
\end{solution}

\begin{solution}{exo:b1:fractions:9}
Decompor, por $0 \leq m \leq n - 1$, a fração $\frac{X^m}{P}$
(grau $ m - n \leq -1$, postes simples): o pólo simples fórmula dá
\[
\frac{X^m}{P} = \sum_{i=1}^{n} \frac{x_i^m / P'(x_i)}{X - x_i} .
\]
Multiplicar por $ X $ e deixe $ X \to +\infty $: o lado esquerdo tende para o
limite de $ X^{m+1}/P $, que é $0$ se $ m \leq n - 2$ e $1$ se $ m =
n - 1$ ($ P $ é \omterm{def:b1:poly:def}{Mônica} de grau $ n $); o lado direito tende a
$\sum_i \frac{x_i^m}{P'(x_i)}$. Por isso
\[
\sum_{i=1}^{n} \frac{x_i^{m}}{P'(x_i)} =
\begin{cases}
0 & \text{for } 0 \leq m \leq n-2,\\
1 & \text{for } m = n - 1,
\end{cases}
\]
que contém ambas as identidades anunciadas ($ m = 0$ requer $ n \geq
2$). (Interpretação via \cref{thm:b1:poly:lagrange}: essas somas são
os coeficientes principais dos interpolantes de Lagrange de $ X^m $ e
interpolando um polinomial de grau $\leq n-1$ no $ n $ pontos
reproduz exatamente.)
\end{solution}

\begin{solution}{exo:b1:fractions:10}
Forma $\frac aX + \frac b{X+1} + \frac c{(X+1)^2}$. Encobrimento em
$0$: $ a = 1$; encobrimento em dobro pólo: $ c = \bigl[\frac1X
\bigr]_{X=-1} = -1$; limite de $ xF(x)$ no infinito: $0 = a + b $, então
$ b = -1$:
\[
\frac{1}{X(X+1)^2} = \frac1X - \frac1{X+1} - \frac1{(X+1)^2} .
\]
Resumindo para $ n = 1, \dots, N $: os primeiros dois tijolos do telescópio a
$1 - \frac1{N+1}$, e o terceiro contribui $-\sum_{k=2}^{N+1}
\frac1{k^2}$. Deixando $ N \to \infty $ e usando $\sum_{k\geq1}
\frac1{k^2} = \frac{\pi^2}6$:
\[
\sum_{n=1}^{\infty}\frac{1}{n(n+1)^2}
= 1 - \Bigl(\frac{\pi^2}6 - 1\Bigr) = 2 - \frac{\pi^2}6
\approx 0.355 .
\]
\end{solution}

\begin{solution}{exo:b1:fractions:11}
Na variável $ Y $: $\frac1{(Y+1)(Y+4)} = \frac{1/3}{Y+1} -
\frac{1/3}{Y+4}$ (encobrimento em $-1$ e $-4$), então
\[
\frac{1}{(X^2+1)(X^2+4)}
= \frac13\,\frac1{X^2+1} - \frac13\,\frac1{X^2+4} .
\]
Da mesma maneira $\frac{Y}{(Y+1)(Y+4)} = \frac{-1/3}{Y+1} +
\frac{4/3}{Y+4}$, então
\[
\frac{X^2}{(X^2+1)(X^2+4)}
= -\frac13\,\frac1{X^2+1} + \frac43\,\frac1{X^2+4} .
\]
(Verifique em $ X = 0$: $0 = \frac13(-1 + 1)$.) Esses já são os
decomposições reais: os numeradores sobre as quadráticas irredutíveis
acontecer de serem constantes.
\end{solution}

\begin{solution}{exo:b1:fractions:12}
\begin{enumerate}
  \item Em cada intervalo evitando o postes, $ F'(x) = -\sum_i
        \frac{c_i}{(x - p_i)^2} < 0$: estritamente decrescente. Como $ x
        \to \pm\infty $, todo tijolo tende a $0$: $ F \to 0$, de
        acima em $+\infty $ (todos os tijolos positivos ali) e de
        abaixo em $-\infty $. Como $ x \to p_i^+$, o tijolo
        $\frac{c_i}{x - p_i}$ explode para $+\infty $ e os outros
        fique limitado: $ F \to +\infty $; da mesma maneira $ F \to -\infty $ como
        $ x \to p_i^-$.
  \item Consertar $\lambda > 0$. Sobre $\intoo{-\infty}{p_1}$: $ F $
        diminui de $0^-$ para $-\infty $, então $ F < 0 < \lambda $:
        nenhuma solução. Em cada $\intoo{p_i}{p_{i+1}}$ ($1 \leq i
        \leq r-1$): $ F $ diminui de $+\infty $ para $-\infty $,
        daí assume o valor $\lambda $ exatamente uma vez (intermediário
        propriedade de valor mais monotonicidade estrita). Sobre
        $\intoo{p_r}{+\infty}$: $ F $ diminui de $+\infty $ para
        $0^+$, novamente exatamente uma solução. Total: exatamente $ r $
        soluções, entrelaçadas com postes.
\end{enumerate}
\end{solution}

\begin{solution}{pb:b1:fractions:1}
\textbf{1.} Cobrir: $\frac1{X(X+1)} = \frac1X - \frac1{X+1}$.
A soma dos telescópios:
\[
\sum_{n=1}^{N}\Bigl(\frac1n - \frac1{n+1}\Bigr)
= 1 - \frac1{N+1} \longrightarrow 1 .
\]

\textbf{2.} $\frac1{X(X+2)} = \frac{1/2}X - \frac{1/2}{X+2}$.
Resumindo, os termos $\frac1n $ sobreviver por $ n = 1, 2$ e os termos
$-\frac1{n+2}$ sobreviver por $ n = N-1, N $:
\[
\sum_{n=1}^{N}\frac1{n(n+2)}
= \frac12\Bigl(1 + \frac12 - \frac1{N+1} - \frac1{N+2}\Bigr)
\longrightarrow \frac34 .
\]

\textbf{3.} Se $ F(X) = G(X) - G(X+1)$, então $\sum_{n=1}^N F(n) =
\sum_{n=1}^N\bigl(G(n) - G(n+1)\bigr) = G(1) - G(N+1)$: todos
valores intermediários são cancelados em pares. Para $ F =
\frac1{X(X+1)(X+2)}$, a testemunha é $ G(X) = \frac1{2X(X+1)}$:
\[
G(X) - G(X+1)
= \frac{(X+2) - X}{2X(X+1)(X+2)} = F(X) ,
\]
então $\sum_{n=1}^N F(n) = \frac14 - \frac1{2(N+1)(N+2)} \to
\frac14$, o valor de \cref{exo:b1:fractions:8}.

\textbf{4.} Coloque o lado direito sobre o denominador comum
$ X(X+1)\cdots(X+k)$:
\[
\frac1k\cdot\frac{(X + k) - X}{X(X+1)\cdots(X+k)}
= \frac{1}{X(X+1)\cdots(X+k)} ,
\]
qual é a identidade. Então $ F_k(X) = \frac1k\bigl(G_k(X) -
G_k(X+1)\bigr)$ com $ G_k(X) = \frac1{X(X+1)\cdots(X+k-1)}$ e
o mecanismo da pergunta 3 dá
\[
\sum_{n=1}^{N}\frac1{n(n+1)\cdots(n+k)}
= \frac1k\Bigl(\frac1{k!} - G_k(N+1)\Bigr)
\longrightarrow \frac1{k\cdot k!} ,
\]
desde $ G_k(1) = \frac1{k!}$ e $ G_k(N+1) \to 0$. Para $ k = 2$:
$\frac1{2\cdot2} = \frac14$, correspondente à pergunta 3.

\textbf{5.} \cref{exo:b1:fractions:6} dá
$\frac{n!}{X(X+1)\cdots(X+n)} = \sum_{j=0}^n
\frac{(-1)^j\binom nj}{X + j}$. Avaliar em $ X = 1$: o lado esquerdo
é $\frac{n!}{(n+1)!} = \frac1{n+1}$, o lado direito é
$\sum_j(-1)^j\binom nj\frac1{1+j}$: primeira identidade. No $ X = 2$:
o lado esquerdo é $\frac{n!}{2\cdot3\cdots(n+2)} =
\frac{n!\cdot1}{(n+2)!} = \frac1{(n+1)(n+2)}$, o lado direito
$\sum_j(-1)^j\binom nj\frac1{2+j}$: segunda identidade.

\textbf{6.} O postes $\omega^k $ são simples e os pólo simples
fórmula de \cref{met:b1:fractions:compute} dá o coeficiente
\[
\frac{1}{\bigl(nX^{n-1}\bigr)_{X = \omega^k}}
= \frac{1}{n\,\omega^{k(n-1)}}
= \frac{\omega^k}{n\,\omega^{kn}} = \frac{\omega^k}n ,
\]
usando $\omega^{kn} = 1$. Por isso $\frac1{X^n-1} = \frac1n\sum_k
\frac{\omega^k}{X - \omega^k}$.

\textbf{7.} Para $ n = 2$ ($\omega = -1$): $\frac12\bigl(
\frac1{X-1} - \frac1{X+1}\bigr) = \frac12\cdot\frac{2}{X^2-1} =
\frac1{X^2-1}$: correto. A soma dos coeficientes é
$\frac1n\sum_k\omega^k = 0$ para $ n \geq 2$
(\cref{prop:b1:complex:sumroots}). Eles devem: $ x\,F(x) \to
\sum_k c_k $ como $ x \to \infty $ para qualquer decomposição com simples
postes, enquanto aqui $ xF(x) = \frac{x}{x^n-1} \to 0$ desde $ n \geq
2$.

\textbf{8.} Encobrimento para $\frac{X^{n-1}}{X^n - 1}$ no $\omega^k $:
$\frac{A(\omega^k)}{B'(\omega^k)} =
\frac{\omega^{k(n-1)}}{n\omega^{k(n-1)}} = \frac1n $, então
$\frac{X^{n-1}}{X^n-1} = \frac1n\sum_k\frac1{X - \omega^k}$ ---
a identidade de \cref{exo:b1:fractions:7} de novo.

\textbf{9.} Com $ c = \omega^k $, $\conj c = \omega^{n-k}$ e
$ c\conj c = 1$, $ c + \conj c = 2\cos\theta_k $:
\[
\frac{c}{X - c} + \frac{\conj c}{X - \conj c}
= \frac{c(X - \conj c) + \conj c(X - c)}
{(X - c)(X - \conj c)}
= \frac{2\cos\theta_k\,X - 2}{X^2 - 2\cos\theta_k\,X + 1} .
\]
Agrupamento $ k $ com $ n - k $ na questão 6: para ímpar $ n $,
\[
\frac1{X^n - 1} = \frac1n\Biggl(\frac1{X-1} +
\sum_{k=1}^{(n-1)/2}
\frac{2\cos\theta_k X - 2}{X^2 - 2\cos\theta_k X + 1}\Biggr) ;
\]
até mesmo $ n $, o auto-pareado extra pólo $\omega^{n/2} = -1$
contribui $\frac{-1}{X+1}$ dentro dos parênteses e o par
soma corre para $\frac n2 - 1$.

\textbf{10.} $ n = 4$: $\theta_1 = \frac\pi2$, $\cos\theta_1 = 0$,
então o termo do par é $\frac{-2}{X^2+1}$ e
\[
\frac1{X^4-1} = \frac14\Bigl(\frac1{X-1} - \frac1{X+1}
- \frac2{X^2+1}\Bigr) ,
\]
a decomposição de \cref{exo:b1:fractions:4}.

\textbf{11.} $ P = \prod_{k=1}^{n-1}(X - \omega^k)$ (dividir $ X^n -
1$ por $ X - 1$), então por \cref{exo:b1:poly:11}, $\frac{P'}P =
\sum_{k=1}^{n-1}\frac1{X-\omega^k}$. No $ X = 1$: $ P(1) = n $ e
$ P'(1) = \sum_{j=1}^{n-1}j = \frac{n(n-1)}2$, de onde
\[
\sum_{k=1}^{n-1}\frac1{1 - \omega^k}
= \frac{P'(1)}{P(1)} = \frac{n-1}2 .
\]

\textbf{12.} Meio ângulo: $1 - \eu^{\iu\theta} =
-2\iu\sin\frac\theta2\,\eu^{\iu\theta/2}$, então
\[
\frac1{1 - \eu^{\iu\theta}}
= \frac{\eu^{-\iu\theta/2}}{-2\iu\sin\frac\theta2}
= \frac{\iu\cos\frac\theta2 + \sin\frac\theta2}
{2\sin\frac\theta2}
= \frac12 + \frac\iu2\,
\frac{\cos\frac\theta2}{\sin\frac\theta2} .
\]
Com $\theta = \theta_k = \frac{2k\pi}n $: $\frac1{1-\omega^k} =
\frac12 + \frac\iu2\cot\frac{k\pi}n $. Resumindo $ k = 1, \dots,
n-1$ e comparando com o valor real $\frac{n-1}2$ de
questão 11: as partes reais já dão conta de tudo, então
$\sum_k\cot\frac{k\pi}n = 0$ --- como a simetria $\cot\frac{(n-k)
\pi}n = -\cot\frac{k\pi}n $ também mostra.

\textbf{13.} Diferenciando $\frac{P'}P = \sum_k\frac1{X -
\omega^k}$:
\[
\frac{P''}P - \Bigl(\frac{P'}P\Bigr)^2
= -\sum_{k=1}^{n-1}\frac1{(X - \omega^k)^2} .
\]
No $ X = 1$: $ P''(1) = \sum_{j=2}^{n-1}j(j-1) = 2\binom n3 =
\frac{n(n-1)(n-2)}3$ (identidade do taco de hóquei ou indução), então
\[
\sum_{k=1}^{n-1}\frac1{(1 - \omega^k)^2}
= \Bigl(\frac{n-1}2\Bigr)^2 - \frac{(n-1)(n-2)}3
= \frac{(n-1)\bigl(3(n-1) - 4(n-2)\bigr)}{12}
= \frac{(n-1)(5-n)}{12} .
\]

\textbf{14.} Fórmula de quadratura da questão 12, com $ c_k =
\cot\frac{k\pi}n $:
\[
\frac1{(1-\omega^k)^2}
= \Bigl(\frac12 + \frac\iu2 c_k\Bigr)^2
= \frac14 - \frac{c_k^2}4 + \frac\iu2\,c_k .
\]
Somando e usando $\sum c_k = 0$ (questão 12) e questão 13:
$\frac{n-1}4 - \frac14\sum_k c_k^2 = \frac{(n-1)(5-n)}{12}$, então
\[
\sum_{k=1}^{n-1}\cot^2\frac{k\pi}n
= (n-1) - \frac{(n-1)(5-n)}3 = \frac{(n-1)(n-2)}3 .
\]
Então $\frac1{\sin^2t} = 1 + \cot^2t $ dá
$\sum_k\frac1{\sin^2\frac{k\pi}n} = (n-1) +
\frac{(n-1)(n-2)}3 = \frac{n^2-1}3$.

\textbf{15.} $ n = 3$: $\cot^2\frac\pi3 + \cot^2\frac{2\pi}3 =
\frac13 + \frac13 = \frac23 = \frac{2\cdot1}3$; e
$\frac1{\sin^2} $ somas para $\frac43 + \frac43 = \frac83 =
\frac{9-1}3$. $ n = 4$: $1 + 0 + 1 = 2 = \frac{3\cdot2}3$; e $2
+ 1 + 2 = 5 = \frac{16-1}3$. Ambas as fórmulas são verificadas.

\textbf{16.} Para $ t \in \intoo0{\frac\pi2}$, o clássico
comparação $\sin t < t < \tan t $ (argumento de área ou convexidade,
familiar desde o ensino médio) rende, recebendo recíprocos, $\cot t <
\frac1t < \frac1{\sin t}$, todos os três positivos ali; quadratura
preserva a ordem. Com $ t = \frac{k\pi}n $, $1 \leq k \leq m $,
$ n = 2m+1$ (então $ t < \frac\pi2$):
\[
\cot^2\frac{k\pi}n < \frac{n^2}{k^2\pi^2} <
\frac1{\sin^2\frac{k\pi}n} .
\]

\textbf{17.} Pelas simetrias $\cot^2\frac{(n-k)\pi}n =
\cot^2\frac{k\pi}n $ e da mesma forma para $\sin^2$, somas da questão 14
metade: $\sum_{k=1}^{m}\cot^2\frac{k\pi}n = \frac{(n-1)(n-2)}6 =
\frac{m(2m-1)}3$ e $\sum_{k=1}^m\frac1{\sin^2\frac{k\pi}n} =
\frac{n^2-1}6 = \frac{2m(m+1)}3$. Resumindo a questão 16 $ k =
1, \dots, m $ e multiplicando por $\frac{\pi^2}{n^2}$:
\[
\frac{\pi^2}{(2m+1)^2}\cdot\frac{m(2m-1)}3
\;<\; \sum_{k=1}^{m}\frac1{k^2} \;<\;
\frac{\pi^2}{(2m+1)^2}\cdot\frac{2m(m+1)}3 .
\]
Ambos os limites tendem a $\frac{\pi^2}6$ como $ m \to \infty $ (as proporções
$\frac{m(2m-1)}{(2m+1)^2}$ e $\frac{2m(m+1)}{(2m+1)^2}$ ambos
tendem a $\frac12$), então, pela compressão, as somas parciais crescentes
convergem e
\[
\sum_{k=1}^{\infty}\frac1{k^2} = \frac{\pi^2}6 .
\]

\textbf{18.} Quadrado $\frac1{X^2-1} = \frac12\bigl(\frac1{X-1} -
\frac1{X+1}\bigr)$:
\[
\frac1{(X^2-1)^2} = \frac14\Bigl(\frac1{(X-1)^2} +
\frac1{(X+1)^2}\Bigr) - \frac12\cdot\frac1{(X-1)(X+1)} ,
\]
e re-decompor o termo cruzado $\frac1{(X-1)(X+1)} =
\frac12\bigl(\frac1{X-1} - \frac1{X+1}\bigr)$ para obter o declarado
formulário. No $ X = 0$: lado esquerdo $1$; lado direito $\frac14(1 + 1) -
\frac14(-1 - 1) = \frac12 + \frac12 = 1$.

\textbf{19.} Soma a questão 18 $ n \geq 2$. Com $ S =
\sum_{k\geq1}\frac1{k^2} = \frac{\pi^2}6$: $\sum_{n\geq2}
\frac1{(n-1)^2} = S $; $\sum_{n\geq2}\frac1{(n+1)^2} = S - 1 -
\frac14$; e o telescópio gap-two $\sum_{n\geq2}\bigl(
\frac1{n-1} - \frac1{n+1}\bigr) = 1 + \frac12 = \frac32$. Por isso
\[
\sum_{n=2}^{\infty}\frac1{(n^2-1)^2}
= \frac14\Bigl(2S - \frac54\Bigr) - \frac14\cdot\frac32
= \frac S2 - \frac{11}{16}
= \frac{\pi^2}{12} - \frac{11}{16} \approx 0.135 .
\]
Numericamente: $\frac19 + \frac1{64} + \frac1{225} + \frac1{576} +
\frac1{1225} + \dots \approx 0.1111 + 0.0156 + 0.0044 + 0.0017 +
0.0008 + \dots \approx 0.135$: consistente.

\textbf{20.} Diferencie a identidade da questão 8
$\sum_k\frac1{X-\omega^k} = \frac{nX^{n-1}}{X^n-1}$:
\[
\sum_{k=0}^{n-1}\frac1{(X - \omega^k)^2}
= \frac{n^2X^{2n-2} - n(n-1)X^{n-2}(X^n - 1)}{(X^n - 1)^2} .
\]
Verifique em $ n = 2$, $ X = 2$: lado direito $\frac{4\cdot4 -
2\cdot1\cdot3}{9} = \frac{10}9$; lado esquerdo $\frac1{(2-1)^2} +
\frac1{(2+1)^2} = 1 + \frac19 = \frac{10}9$.

\textbf{21.} $\frac1{\binom nk} = \frac{k!\,(n-k)!}{n!} =
\frac{k!}{(n-k+1)(n-k+2)\cdots n}$, um produto de $ k $ consecutivo
inteiros lá embaixo. Substituindo $ j = n - k + 1$ (então $ j $ corre
acabou $\N^*$ como $ n $ corre de $ k $):
\[
\sum_{n=k}^{\infty}\frac1{\binom nk}
= k!\sum_{j=1}^{\infty}\frac1{j(j+1)\cdots(j+k-1)}
= \frac{k!}{(k-1)\,(k-1)!} = \frac{k}{k-1} ,
\]
pela questão 4 aplicada com $ k - 1$ no lugar de $ k $ (válido desde
$ k - 1 \geq 1$).

\textbf{22.} Para $ k = 3$: $\frac1{\binom n3} =
\frac6{(n-2)(n-1)n}$, então pela soma parcial da questão 3 (deslocada),
\[
\sum_{n=3}^{N}\frac1{\binom n3}
= 6\sum_{j=1}^{N-2}\frac1{j(j+1)(j+2)}
= 6\Bigl(\frac14 - \frac1{2(N-1)N}\Bigr)
= \frac32 - \frac3{(N-1)N}
\longrightarrow \frac32 ,
\]
o valor $\frac k{k-1} = \frac32$ da questão 21.

\textbf{23.} $ n = 6$: pares $ k = 1$ ($\theta_1 = \frac\pi3$,
$2\cos\theta_1 = 1$) e $ k = 2$ ($\theta_2 = \frac{2\pi}3$,
$2\cos\theta_2 = -1$), mais o real postes $\pm1$:
\[
\frac1{X^6-1} = \frac16\Bigl(\frac1{X-1} - \frac1{X+1}
+ \frac{X - 2}{X^2 - X + 1}
+ \frac{-X - 2}{X^2 + X + 1}\Bigr) .
\]
No $ X = 0$: lado esquerdo $-1$; lado direito $\frac16(-1 - 1 - 2 - 2)
= -1$: correto.

\textbf{24.} (i) A singularidade legitima toda identificação de
coeficientes --- encobrimento, o agrupamento de pares \omterm{def:b1:complex:field}{conjugados} de
questão 9 e os truques de diferenciação (questões 13, 20)
todos descansam sobre isso. (ii) Raízes da unidade forneceu o postes de $ X^n -
1$, suas simetrias ($ k \leftrightarrow n-k $) e o
álgebra de meio ângulo da questão 12
(\cref{met:b1:complex:trig}). (iii) A derivada logarítmica
$\frac{P'}P = \sum\frac{m_i}{X-a_i}$ informações convertidas sobre
as raízes de $ P = 1 + X + \dots + X^{n-1}$ no numérico
somas das questões 11 e 13 --- a articulação entre as Partes II e
III.

\textbf{25.} A decomposição é uma identidade puramente algébrica,
true para todos os valores da variável de uma só vez; é isso que faz
é um motor. Substituir números inteiros e somar transformou-o em
telescópios (Parte I); substituindo raízes da unidade e agrupamento
\omterm{def:b1:complex:field}{conjugados} transformaram-no em identidades trigonométricas (Partes II e
III); e somente na última etapa a análise entrou --- um
espremer entre dois formulários fechados --- para entregar $\frac{\pi^2}6$,
um \omterm{def:b1:logic:statement}{declaração} nenhuma substituição finita poderia alcançar. Esta \omterm{thm:b1:logic:partition}{divisão} de
trabalho (a álgebra produz identidades finitas exatas, a análise passa
até o limite) é o modelo para \cref{ch:b1:series}, onde
telescópica e comparação tornam-se sistemáticas, e para
\cref{ch:b1:integration}, onde cada tijolo adquire um primitivo
e as mesmas decomposições calculam integrais em vez de somas.
\end{solution}
