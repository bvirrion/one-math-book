\section*{Capítulo \ref{ch:b1:series} --- Série Numérica}

\begin{solution}{exo:b1:series:1}
$\dfrac{1}{n(n+1)} = \dfrac1n - \dfrac{1}{n+1}$: telescópico, $ S_N =
1 - \frac{1}{N+1} \to 1$. Convergente, soma $1$.

$\ln\bigl(1 - \frac{1}{n^2}\bigr) = \ln\frac{(n-1)(n+1)}{n^2} =
\ln\frac{n-1}{n} - \ln\frac{n}{n+1}$: telescópica novamente, $ S_N =
\ln\frac12 - \ln\frac{N}{N+1} \to -\ln 2$. Convergente, soma $-\ln 2$.

$\dfrac{3^n + 4^n}{5^n} = \bigl(\frac35\bigr)^n +
\bigl(\frac45\bigr)^n $: dois convergentes \omterm{ex:b1:series:geometric}{série geométrica}, soma
$\frac{1}{1 - 3/5} + \frac{1}{1 - 4/5} = \frac52 + 5 = \frac{15}{2}$.
\end{solution}

\begin{solution}{exo:b1:series:2}
Teste de proporção por todo.

$\frac{u_{n+1}}{u_n} = \frac{(n+1)^2}{2n^2} \to \frac12 < 1$:
convergente.

$\frac{u_{n+1}}{u_n} = \frac{(n+1)!\,n^n}{n!\,(n+1)^{n+1}} =
\bigl(\frac{n}{n+1}\bigr)^n = \bigl(1 + \frac1n\bigr)^{-n} \to
\frac1\eu < 1$: convergente.

Com o fator $2^n $: razão $\to \frac2\eu < 1$: convergente.

Com $3^n $: razão $\to \frac3\eu > 1$: divergente (os termos tendem a
$+\infty $).
\end{solution}

\begin{solution}{exo:b1:series:3}
Todos os termos não negativos; usar equivalentes
(\cref{thm:b1:series:positive}).

$\sin\frac{1}{n^2} \sim \frac{1}{n^2}$: convergente (Riemann $\alpha
= 2$).

$1 - \cos\frac1n \sim \frac{1}{2n^2}$: convergente.

$\frac{1}{\sqrt{n(n+1)}} \sim \frac1n $: divergente.

$\frac{\ln n}{n^2} = \frac{1}{n^{3/2}}\cdot\frac{\ln n}{n^{1/2}}$
e $\frac{\ln n}{\sqrt n} \to 0$
(\cref{prop:b1:functions:powerrules}): então $\frac{\ln n}{n^2} \leq
\frac{1}{n^{3/2}}$ para grande $ n $: convergente.
\end{solution}

\begin{solution}{exo:b1:series:4}
Para $ n \geq 2$: $\frac{1}{n^2} \leq \frac{1}{n(n-1)} =
\frac{1}{n-1} - \frac1n $. Por isso
\[
\sum_{n=1}^{N} \frac{1}{n^2} \leq 1 + \sum_{n=2}^{N}
\Bigl(\frac{1}{n-1} - \frac1n\Bigr) = 1 + 1 - \frac1N < 2 :
\]
somas parciais crescentes e limitadas por $2$: convergência
(\cref{thm:b1:series:positive}), soma $\leq 2$. (O valor exato
$\frac{\pi^2}{6}$ é uma celebração do segundo ano.)
\end{solution}

\begin{solution}{exo:b1:series:5}
$\sum \frac{(-1)^n}{\sqrt n}$: alternando com $\frac{1}{\sqrt n}
\downarrow 0$: convergente (\cref{thm:b1:series:alternating}); não
absolutamente ($\alpha = \frac12 \leq 1$).

$\sum \frac{(-1)^n}{n + (-1)^n}$: a sequência $\frac{1}{n +
(-1)^n}$ é \emph{não} diminuindo ($\frac{1}{n+1}$ então
$\frac{1}{n}$ alternam mal), então o teste não se aplica diretamente.
Expandir:
\[
\frac{(-1)^n}{n + (-1)^n}
= \frac{(-1)^n}{n}\cdot\frac{1}{1 + \frac{(-1)^n}{n}}
= \frac{(-1)^n}{n} - \frac{1}{n^2} +
O\Bigl(\frac{1}{n^3}\Bigr):
\]
o primeiro \omterm{def:b1:series:def}{série} converge (alternando), $\sum \frac{1}{n^2}$
converge, o $ O(n^{-3})$ converge absolutamente: a soma de três
convergente \omterm{def:b1:series:def}{série} converge.

$\sin\bigl(\pi\sqrt{n^2+1}\bigr)$: escrever $\sqrt{n^2 + 1} = n +
\frac{1}{2n} + \varepsilon_n $ com $\varepsilon_n = O(n^{-3})$;
então, por $\pi $- periodicidade de $\sin $ até assinar,
\[
\sin\bigl(\pi\sqrt{n^2+1}\bigr)
= (-1)^n \sin\Bigl(\frac{\pi}{2n} + \pi\varepsilon_n\Bigr) .
\]
\omterm{def:b1:logic:sets}{Definir} $\theta_n = \frac{\pi}{2n} + \pi\varepsilon_n $ e $ a_n =
\sin\theta_n $. Para grandes $ n $, $\theta_n \in \intoo{0}{\frac\pi2}$
e
\[
\theta_n - \theta_{n+1} = \frac{\pi}{2n(n+1)} +
\pi(\varepsilon_n - \varepsilon_{n+1})
= \frac{\pi}{2n^2} + O\Bigl(\frac{1}{n^3}\Bigr) > 0
\]
eventualmente, então $(\theta_n)$ diminui para $0$; desde $\sin $ é
aumentando em $\intcc{0}{\frac\pi2}$, $(a_n)$ diminui para $0$ também. O teste alternado se aplica: convergente --- não absolutamente,
desde $ a_n \sim \frac{\pi}{2n}$.
\end{solution}

\begin{solution}{exo:b1:series:6}
$ f(t) = \frac{1}{t(\ln t)^\alpha}$ é positivo, contínuo,
diminuindo em $\intco{2}{+\infty}$. Substituindo $ u = \ln t $:
\[
\int_2^x \frac{\dd t}{t(\ln t)^\alpha}
= \int_{\ln 2}^{\ln x} \frac{\dd u}{u^\alpha},
\]
limitado como $ x \to \infty $ se $\alpha > 1$
(\cref{thm:b1:series:riemann}cálculo). Por integrante
comparação: convergência sse $\alpha > 1$. (Esses \emph{Tipo Bertrand}
\omterm{def:b1:series:def}{série} mostrar o quão bom é o \omterm{def:b1:topology:closure}{limite} de convergência é: $ n\ln n $
diverge, $ n(\ln n)^{1.01}$ converge.)
\end{solution}

\begin{solution}{exo:b1:series:7}
Pelos \omterm{def:b1:topology:closure}{limites} da linha tangente de \cref{exo:b1:derivative:3} reescrito
via expansões: para $ x = \frac1n \in \intoc{0}{1}$, Taylor-Lagrange
para $\ln(1+x)$ por ordem $1$ dá $\ln(1 + x) = x - \frac{x^2}{2(1 +
c)^2}$ para alguns $ c \in \intoo{0}{x}$, então
\[
0 \leq u_n = \frac1n - \ln\Bigl(1 + \frac1n\Bigr) \leq
\frac{1}{2n^2} .
\]
Comparação com o \omterm{thm:b1:series:riemann}{Série Riemann}: $\sum u_n $ converge. É parcial
soma telescópios os logaritmos:
\[
\sum_{n=1}^{N} u_n = H_N - \ln(N+1)
\]
(desde $\sum_{n\leq N} \ln\frac{n+1}{n} = \ln(N+1)$). Então $ H_N -
\ln(N+1)$ converge; adicionando $\ln\frac{N+1}{N} \to 0$, a sequência
$ H_N - \ln N $ converge. Seu \omterm{def:b1:topology:closure}{limite} é $\gamma \approx 0.5772$.
\end{solution}

\begin{solution}{exo:b1:series:8}
$\dfrac{1}{n(n+2)} = \dfrac{1/2}{n} - \dfrac{1/2}{n+2}$: o parcial
soma telescópios com um atraso de $2$,
\[
S_N = \frac12\Bigl(1 + \frac12 - \frac{1}{N+1} -
\frac{1}{N+2}\Bigr) \longrightarrow \frac34 .
\]

$\sum \frac{n}{2^n}$: para $\abs x < 1$, diferenciando o finito
soma geométrica e passando ao \omterm{def:b1:topology:closure}{limite} (todos \omterm{def:b1:series:def}{série} aqui convergem
absolutamente, \omterm{thm:b1:series:ratio}{teste de razão}): de $\sum_{n\geq0} x^n = \frac{1}{1-x}$,
obtém-se por cálculo direto com somas parciais
\[
\sum_{n=1}^{N} n x^{n-1}
= \frac{1 - (N+1)x^N + N x^{N+1}}{(1 - x)^2}
\xrightarrow[N\to\infty]{} \frac{1}{(1-x)^2}
\quad (\abs x < 1),
\]
(o \omterm{def:b1:topology:closure}{limite} termos $ N x^N \to 0$). No $ x = \frac12$:
$\sum_{n\geq1} n\bigl(\frac12\bigr)^{n-1} = 4$, então $\sum_{n\geq0}
\frac{n}{2^n} = \frac12 \times 4 = 2$.
\end{solution}

\begin{solution}{exo:b1:series:9}
Agrupe os termos de $\sum u_n $ em pacotes entre potências de $2$.
Pacotes superiores: para $2^k \leq n < 2^{k+1}$ há $2^k $ termos,
cada $\leq u_{2^k}$:
\[
\sum_{n=1}^{2^{K+1}-1} u_n
= \sum_{k=0}^{K} \sum_{n=2^k}^{2^{k+1}-1} u_n
\leq \sum_{k=0}^{K} 2^k u_{2^k} .
\]
Pacotes inferiores: cada termo do mesmo pacote é $\geq u_{2^{k+1}}$,
então $\sum_{n=2^k}^{2^{k+1}-1} u_n \geq 2^k u_{2^{k+1}} = \frac12
\cdot 2^{k+1} u_{2^{k+1}}$, de onde
\[
\sum_{n=1}^{2^{K+1}-1} u_n \geq \frac12 \sum_{k=1}^{K+1} 2^{k}
u_{2^{k}} .
\]
Ambas as comparações de soma parcial vão em ambos os sentidos (termos não negativos,
\cref{thm:b1:series:positive}): os dois \omterm{def:b1:series:def}{série} têm a mesma natureza.

Riemann: $ u_n = n^{-\alpha}$ dá $2^k u_{2^k} =
2^{k(1-\alpha)}$, um \omterm{ex:b1:series:geometric}{série geométrica}, convergente se sef $2^{1 - \alpha}
< 1$ se $\alpha > 1$. Bertrand
(\cref{exo:b1:series:6}): $ u_n = \frac{1}{n(\ln n)^\alpha}$ dá
$2^k u_{2^k} = \frac{1}{(k\ln 2)^\alpha}$, um \omterm{thm:b1:series:riemann}{Série Riemann} em $ k $:
convergente se sef $\alpha > 1$.
\end{solution}

\begin{solution}{exo:b1:series:10}
Identidade geométrica finita com proporção $-t^2$:
\[
\frac{1}{1 + t^2} = \sum_{k=0}^{n-1} (-1)^k t^{2k}
+ \frac{(-1)^n t^{2n}}{1 + t^2} .
\]
Integrar $\intcc{0}{1}$ (o lado esquerdo se integra a $\arctan
1 = \frac\pi4$, \cref{prop:b1:functions:arcderiv}):
\[
\frac{\pi}{4} = \sum_{k=0}^{n-1} \frac{(-1)^k}{2k+1}
+ (-1)^n \int_0^1 \frac{t^{2n}}{1+t^2}\,\dd t,
\qquad
0 \leq \int_0^1 \frac{t^{2n}}{1+t^2}\,\dd t \leq \int_0^1 t^{2n}\dd
t = \frac{1}{2n+1} .
\]
Deixando $ n \to \infty $ prova a fórmula e o \omterm{def:b1:topology:closure}{limite} exibido
no integrante é exatamente o resto ligado: depois de somar
$ N $ (ou seja,\ $ n = N + 1$ termos), $\abs{R_N} \leq \frac{1}{2N + 3}$.
\end{solution}

\begin{solution}{exo:b1:series:11}
$ n^{1/n} = \eu^{\frac{\ln n}{n}} \to \eu^0 = 1$
(\cref{prop:b1:functions:powerrules}). Por isso
\[
\frac{1}{n^{1 + 1/n}} = \frac{1}{n}\,\eu^{-\frac{\ln n}{n}}
\sim \frac{1}{n} ,
\]
e o teste de equivalentes (\cref{thm:b1:series:positive})
compara com o harmônico divergente \omterm{def:b1:series:def}{série}: \emph{divergente},
embora cada expoente $1 + \frac1n $ excede $1$. O Riemann
critério diz respeito a um \emph{fixo} expoente $\alpha $; um expoente
deslizando para baixo para $1$ pode perder toda a sua margem, como aqui.
\end{solution}

\begin{solution}{exo:b1:series:12}
Deixar $\varepsilon > 0$. Pelo critério de Cauchy para o
convergente \omterm{def:b1:series:def}{série} (\cref{thm:b1:seq:complete} aplicado ao
somas parciais), há $ N $ com $\sum_{k=n+1}^{2n} u_k \leq
\varepsilon $ para $ n \geq N $. Por monotonicidade cada um desses $ n $
termos é $\geq u_{2n}$:
\[
n\,u_{2n} \leq \sum_{k=n+1}^{2n} u_k \leq \varepsilon
\quad\Longrightarrow\quad 2n\,u_{2n} \leq 2\varepsilon ,
\]
e para índices ímpares $(2n+1)\,u_{2n+1} \leq (2n+1)\,u_{2n} \leq
2\bigl(2n\,u_{2n}\bigr) \leq 4\varepsilon $ para $ n \geq N $: em
ambas as paridades, $ n u_n \to 0$.

Converse falso: $ u_n = \frac{1}{n\ln n}$ tem $ n u_n =
\frac{1}{\ln n} \to 0$, ainda o \omterm{def:b1:series:def}{série} diverge
(\cref{exo:b1:series:6}, $\alpha = 1$). Monotonicidade necessária:
deixe $ u_n = \frac1n $ quando $ n $ é um quadrado perfeito e $ u_n =
2^{-n}$ caso contrário: o \omterm{def:b1:series:def}{série} converge (a soma dos termos quadrados
gosto $\sum \frac{1}{k^2}$, o resto geometricamente), mas $ n u_n =
1$ ao longo dos quadrados.
\end{solution}

\begin{solution}{pb:b1:series:1}
\textbf{1.} $ a_{n+1} - a_n = \frac{1}{n+1} - \ln\frac{n+1}{n}
\leq 0$ porque $\ln(1 + \frac1n) \geq \frac{1/n}{1 + 1/n} =
\frac{1}{n+1}$; e $ b_{n+1} - b_n = \frac{1}{n+1} -
\ln\frac{n+2}{n+1} \geq 0$ porque $\ln(1 + \frac{1}{n+1}) \leq
\frac{1}{n+1}$. A lacuna deles $ a_n - b_n = \ln(1 + \frac1n) \to 0$:
adjacente (\cref{thm:b1:seq:adjacent}), com \omterm{def:b1:topology:closure}{limite} comum $\lim
a_n = \gamma $ (\cref{exo:b1:series:7}), e $ b_n \leq \gamma
\leq a_n $.

\textbf{2.} $ b_{10} = 2.928968 - \ln 11 = 0.5311$ e $ a_{10} =
2.928968 - \ln 10 = 0.6264$: então $\gamma \in
\intcc{0.5311}{0.6264}$. A lacuna é $\ln 1.1 \approx 0.095$ e
encolhe como $\frac1n $: quatro casas decimais ($\text{gap} \leq
10^{-4}$) precisaria $ n \approx 10^4$ --- os colchetes são
correto, mas lento.

\textbf{3.} Telescópico $ a_n - a_{m+1} = \sum_{k=n}^{m} w_k $ e
deixando $ m \to \infty $: $ a_n - \gamma = \sum_{k\geq n} w_k $
(\omterm{def:b1:topology:closure}{limite} de somas parciais). Além disso
\[
w_k = \int_k^{k+1} \frac{\dd t}{t} - \frac{1}{k+1}
= \int_k^{k+1} \Bigl(\frac1t - \frac{1}{k+1}\Bigr)\dd t
= \int_k^{k+1} \frac{k + 1 - t}{t\,(k+1)}\,\dd t .
\]
Sobre $\intcc{k}{k+1}$: $\frac{1}{(k+1)^2} \leq \frac{1}{t(k+1)}
\leq \frac{1}{k(k+1)}$, e $\int_k^{k+1}(k + 1 - t)\dd t =
\frac12$: por isso $\frac{1}{2(k+1)^2} \leq w_k \leq
\frac{1}{2k(k+1)}$.

\textbf{4.} Superior: $\sum_{k \geq n} \frac{1}{2k(k+1)} =
\frac12\sum_{k\geq n}\bigl(\frac1k - \frac{1}{k+1}\bigr) =
\frac{1}{2n}$ (telescópio). Mais baixo: $\frac{1}{2(k+1)^2} \geq
\frac{1}{2(k+1)(k+2)}$, cuja soma é telescópica para
$\frac{1}{2(n+1)}$. Com a pergunta 3:
\[
\frac{1}{2(n+1)} \leq H_n - \ln n - \gamma \leq \frac{1}{2n} .
\]

\textbf{5.} Subtrair $\frac{1}{2n}$: $\gamma_n - \gamma \in
\intcc{\frac{1}{2(n+1)} - \frac{1}{2n}}{0} =
\intcc{-\frac{1}{2n(n+1)}}{0}$: a estimativa corrigida é exata
para $ O\bigl(\frac{1}{n^2}\bigr)$, e sempre por baixo.

\textbf{6.} $\gamma_{100} = 5.1873775 - \ln 100 - 0.005 =
0.5772073$, com $0 \leq \gamma - \gamma_{100} \leq
\frac{1}{20200} < 5\cdot10^{-5}$: por isso $0.577207 \leq \gamma
\leq 0.577257$, ou seja\ $\gamma = 0.5772 \pm 5\cdot10^{-5}$ (verdadeiro
valor $0.5772156\dots $) --- quatro casas decimais certificadas de um
cem termos, contra dez mil da questão 2.

\textbf{7.} Primeiro dividendo:
\[
H_{2m} - H_m = \Bigl(\ln 2m + \gamma + \frac{1}{4m}\Bigr)
- \Bigl(\ln m + \gamma + \frac{1}{2m}\Bigr) +
O\Bigl(\frac{1}{m^2}\Bigr)
= \ln 2 - \frac{1}{4m} + O\Bigl(\frac{1}{m^2}\Bigr).
\]
Segundo: os recíprocos pares até $2m $ soma para $\frac12 H_m $, então
$\sum_{j=1}^{m}\frac{1}{2j-1} = H_{2m} - \frac12 H_m = \frac12
\ln m + \ln 2 + \frac\gamma2 + o(1)$, e $\sum_{j=1}^m
\frac{1}{2j} = \frac12\ln m + \frac\gamma2 + o(1)$.

\textbf{8.} Dividindo os termos pares duas vezes:
$\sum_{k=1}^{2m}\frac{(-1)^{k-1}}{k} = H_{2m} - 2\cdot\frac12
H_m = H_{2m} - H_m $. Pela questão 7 isso é igual $\ln 2 -
\frac{1}{4m} + O(m^{-2})$: a soma é $\ln 2$
(\cref{ex:b1:series:ln2} de novo) \emph{com} sua velocidade.

\textbf{9.} Para $ N = 2m $: $ S - S_N = \frac{1}{4m} + O(m^{-2}) =
\frac{1}{2N} + O(N^{-2})$. Para $ N = 2m + 1$: $ S_{N} = S_{2m} +
\frac{1}{2m+1}$, então
\[
S - S_N = \Bigl(\frac{1}{4m} - \frac{1}{2m+1}\Bigr) +
O\Bigl(\frac{1}{m^2}\Bigr)
= -\frac{1}{4m} + O\Bigl(\frac{1}{m^2}\Bigr)
= -\frac{1}{2N} + O\Bigl(\frac{1}{N^2}\Bigr).
\]
Em ambos os casos $ S - S_N \sim \frac{(-1)^N}{2N}$: metade do
\omterm{def:b1:topology:closure}{limite} do pior caso $ a_{N+1}$, com um sinal conhecido e alternado.

\textbf{10.} A média elimina o termo principal oscilante:
\[
S - \tilde S_N = \frac{(S - S_N) + (S - S_{N+1})}{2}
= \frac{(-1)^N}{2}\Bigl(\frac{1}{2N} - \frac{1}{2(N+1)}\Bigr)
+ O\Bigl(\frac{1}{N^2}\Bigr) = O\Bigl(\frac{1}{N^2}\Bigr).
\]
Numericamente: $ S_{10} = 0.645635$, $ S_{11} = 0.736544$, $\tilde
S_{10} = 0.691089$, e $\ln 2 = 0.693147$: o erro cai de
$4.8\cdot10^{-2}$ para $2.1\cdot10^{-3}$ --- uma adição, vinte
vezes \omterm{def:b1:arith:prime}{melhor}.

\textbf{11.} O erro alternado muda de sinal a cada passo,
então somas parciais consecutivas ultrapassam o \omterm{def:b1:topology:closure}{limite} e sua média
cancela o termo de primeira ordem; o erro do colchete $ H_n - \ln n -
\gamma \approx \frac{1}{2n}$ tem sinal constante, então não há média
junto $ n $ pode cancelá-lo. Média dos dois \emph{colchetes} faz
ajuda: $\frac{a_n + b_n}{2} = H_n - \ln\sqrt{n(n+1)}$, e desde
$\ln\sqrt{n(n+1)} = \ln\bigl(n + \frac12\bigr) + O(n^{-2})$,
\[
H_n - \ln\Bigl(n + \frac12\Bigr)
= \Bigl(H_n - \ln n - \frac{1}{2n}\Bigr) + \frac{1}{8n^2} +
O\Bigl(\frac{1}{n^3}\Bigr) = \gamma + O\Bigl(\frac{1}{n^2}\Bigr)
\]
(questão 5 e $\ln(1 + \frac{1}{2n}) = \frac{1}{2n} -
\frac{1}{8n^2} + O(n^{-3})$). Verifique em $ n = 10$: $ H_{10} -
\ln 10.5 = 0.57759$, já dentro $4\cdot10^{-4}$ de
$\gamma $.

\textbf{12.} $\sum_{j\leq m} \frac{1}{2j-1} \geq \sum_{j \leq m}
\frac{1}{2j} = \frac12 H_m \to +\infty $: tanto o positivo quanto o
a parte negativa do harmônico alternado \omterm{def:b1:series:def}{série} divergem.

\textbf{13.} Escrever $ u_n^\pm = \frac{\abs{u_n} \pm u_n}{2} \geq
0$, então $ u_n = u_n^+ - u_n^-$ e $\abs{u_n} = u_n^+ + u_n^-$. Se
$\sum u_n^+$ convergiu, então $\sum u_n^- = \sum (u_n^+ - u_n)$
convergiria (diferença de convergente \omterm{def:b1:series:def}{série}), por isso $\sum
\abs{u_n}$ também: contradição com convergência condicional. Por
simetria ambos $\sum u_n^\pm $ divergir (para $+\infty $): um infinito
reservatório de massa positiva e negativa.

\textbf{14.} Deixar $ S = \sum u_n $, $\varepsilon > 0$, e $ N $ com
$\sum_{n > N} \abs{u_n} \leq \varepsilon $ (Critério de Cauchy para
$\sum\abs{u_n}$). Deixar $ M_0$ ser grande o suficiente para que $\sigma(\{0,
\dots, M_0\}) \supseteq \{0, \dots, N\}$. Para $ M \geq M_0$, a diferença $\sum_{m \leq M} u_{\sigma(m)} - \sum_{n \leq N} u_n $
é uma soma finita de \emph{distinto} termos $ u_n $ com $ n > N $,
portanto de valor absoluto $\leq \varepsilon $; e $\abs{S -
\sum_{n\leq N} u_n} \leq \varepsilon $ também. Então o reorganizado
somas parciais estão dentro $2\varepsilon $ de $ S $ eventualmente:
$\sum u_{\sigma(n)} = S $. Convergência absoluta é
à prova de rearranjo.

\textbf{15.} Cada fase do procedimento ganancioso termina após
um número finito de termos, porque o restante positivo
somente os termos (respectivamente negativos) têm somas parciais divergentes
(questão 12): a soma acumulada deve eventualmente ultrapassar $ t $. O
o procedimento, portanto, alterna infinitas fases finitas,
consumindo os termos positivos em ordem e os termos negativos em
ordem: cada termo é usado exatamente uma vez --- um rearranjo.
Após a primeira travessia, entre duas travessias consecutivas o
somas parciais movem-se monotonicamente em direção $ t $, e em um cruzamento
eles ultrapassam, no máximo, o termo recém-adicionado; já que os termos
usado no $ j $-º cruzamento tem índice pelo menos $ j $ em seu
classe, esses excessos tendem a $0$. Daí as somas parciais
convergir para $ t $: todo número real é a soma de alguns
rearranjo.

\textbf{16.} Os slots positivos recebem $\frac{1}{2j-1}$ para
$ j = 1, 2, \dots $ em ordem, os slots negativos $\frac{1}{2j}$ em
ordem: cada termo do harmônico alternado \omterm{def:b1:series:def}{série} aparece
exatamente uma vez. Para $(p, q) = (1, 2)$, os blocos são
$\bigl(1, -\frac12, -\frac14\bigr)$, $\bigl(\frac13, -\frac16,
-\frac18\bigr)$, $\bigl(\frac15, -\frac1{10},
-\frac1{12}\bigr)$, \dots\ --- o exibido \omterm{def:b1:series:def}{série}.

\textbf{17.} Desde $\frac{1}{4k-2} = \frac12\cdot\frac{1}{2k-1}$:
\[
\frac{1}{2k-1} - \frac{1}{4k-2} - \frac{1}{4k}
= \frac12\,\frac{1}{2k-1} - \frac12\,\frac{1}{2k}
= \frac12\Bigl(\frac{1}{2k-1} - \frac{1}{2k}\Bigr).
\]
Resumindo $ k = 1, \dots, K $: $ T_{3K} = \frac12
\sum_{k=1}^{K}\bigl(\frac{1}{2k-1} - \frac{1}{2k}\bigr) =
\frac12 S_{2K}$: a cada terceira soma parcial, o reorganizado
\omterm{def:b1:series:def}{série} é \emph{exatamente} metade do original.

\textbf{18.} Depois $ K $ blocos completos, os parciais reorganizados
soma é $\sum_{j=1}^{pK}\frac{1}{2j-1} -
\sum_{j=1}^{qK}\frac{1}{2j}$, e a questão 7 avalia:
\[
\Bigl(\frac{\ln(pK)}{2} + \ln 2 + \frac\gamma2\Bigr)
- \Bigl(\frac{\ln(qK)}{2} + \frac\gamma2\Bigr) + o(1)
= \ln 2 + \frac12\ln\frac pq + o(1) :
\]
o $\gamma $ é cancelar, o $\ln K $ s cancelar, a proporção
$\frac pq $ sobrevive.

\textbf{19.} Uma soma parcial dentro do bloco $ K + 1$ difere do
$ K $-bloquear soma no máximo $ p + q $ termos, cada um de valor absoluto
$\leq \frac{1}{2qK}$-ish, portanto por $ O\bigl(\frac1K\bigr) \to 0$:
a sequência completa de somas parciais tem o mesmo \omterm{def:b1:topology:closure}{limite} $\ln 2 +
\frac12\ln\frac pq $. Para $(1, 2)$: $\ln 2 + \frac12\ln\frac12 =
\frac{\ln 2}{2} = 0.34657\dots $, e de fato $ T_9 = 0.30833$
rasteja em direção a ele: pela pergunta 17, $ T_{3K} = \frac12 S_{2K}$
converge com exatamente metade do erro harmônico alternado.
Mesmos termos, metade da soma.

\textbf{20.} $(1,1)$: $\ln 2 + \frac12\ln 1 = \ln 2$ --- o
ordem original, consistência. $(2,1)$: $\frac32\ln 2 \approx
1.0397$. O $(p,q)$ menu atinge exatamente o contável \omterm{def:b1:topology:dense}{denso}
família $\ln 2 + \frac12\ln r $, $ r \in \Q_{>0}$; O ganancioso de Riemann
receita (questão 15) atinge \emph{todo} real. Compras de estrutura
fórmulas; a ganância compra a totalidade.

\textbf{21.} O telescópio de somas parciais: $\sum_{k=1}^{N}
\bigl(\frac1k - \ln\frac{k+1}{k}\bigr) = H_N - \ln(N+1) = b_N
\to \gamma $: o \omterm{def:b1:series:def}{série} de \cref{exo:b1:series:7} soma exatamente
para \omterm{ex:b1:series:oddtelescope}{Constante de Euler}.

\textbf{22.} Ganancioso por $ t = 1$: o primeiro termo positivo traz
a soma exatamente para $1$, não além dele, então um segundo positivo é
levado para atravessar: $1, \frac13$ (soma $1.3333 > 1$), então
$-\frac12$ ($0.8333$), $\frac15$ ($1.0333$), $-\frac14$
($0.7833$), $\frac17, \frac19$ ($1.0373$), $-\frac16$
($0.8706$), $\frac1{11}, \frac1{13}$ ($1.0384$), $-\frac18$
($0.9134$), $\frac1{15}$ ($0.9801$), \dots\ --- as somas respiram
ao redor $1$ com amplitude cada vez menor, dois positivos agora
necessário por ciclo, uma vez que os negativos são maiores.

\textbf{23.} Blocos de construção: em estágio $ j $, anexe o suficiente não utilizado
termos positivos para aumentar a soma parcial em pelo menos $1$
(possível: os restantes termos positivos têm somas divergentes,
questão 12), em seguida, acrescente o único termo negativo $-\frac{1}
{2j}$. Todo termo positivo é eventualmente usado (cada estágio usa
pelo menos um), todos os negativos também (um por estágio): um
rearranjo. Cada estágio altera a soma em $\geq 1 -
\frac{1}{2j} \geq \frac12$: as somas parciais excedem
$\frac{j}{2}$ depois da fase $ j $ e os incrementos dentro de um estágio
são positivos, exceto o último, limitado por $\frac{1}{2j} \to 0$:
divergência para $+\infty $.

\textbf{24.} Pela lei, $ H_N \geq 20 \iff \ln N \geq 20 -
\gamma - \frac{1}{2N} + O(N^{-2})$: o \omterm{def:b1:topology:closure}{limite} $ N^*$
satisfaz $\ln N^* = 20 - \gamma + o(1)$, ou seja\ $ N^* =
\eu^{20 - \gamma}(1 + o(1)) \approx \eu^{19.4228} \approx
2.72\cdot10^{8}$ --- dentro da janela bruta
$\intcc{1.8\cdot10^8}{4.9\cdot10^8}$ de
\cref{ex:b1:series:harmonicstack}e fixado por $\gamma $.

\textbf{25.} (eu) Em $ H_n = \ln n + \gamma + \frac{1}{2n} +
O(n^{-2})$: o $\ln n $ é o integrante, $\gamma $ o preço de
substituindo uma soma por uma integrante (uma constante genuinamente nova de
análise) e $\frac{1}{2n}$ a primeira correção --- a
sombra do trapézio. (ii) A convergência condicional depende de
cancelamento entre dois reservatórios infinitos (questão 13), então
o reordenamento repondera os reservatórios; convergência absoluta tem
massa total finita, e a estimativa da cauda da questão 14 é
cego para ordens. (iii) O $(p,q)$ as somas eram \emph{computado}: o
$\gamma $-law transformou cada soma parcial reorganizada em $\ln 2 +
\frac12\ln\frac pq + o(1)$, com $\gamma $ cancelando-se ---
um exercício de contabilidade assintótica, não um argumento abstrato.
(iv) Próximo: produtos e soma incondicional de potência
\omterm{def:b1:series:def}{série} no volume do Ano 2; e o fim de semana do volume do Ano 3
problema na fórmula de Stirling, onde soma versus integral
contabilidade, empurrou um pedido adiante, produz $\sqrt{2\pi}$
em si.
