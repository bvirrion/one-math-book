\section*{Capítulo \ref{ch:b1:topology} --- Topologia da Linha Real}

\begin{solution}{exo:b1:topology:1}
$\intoo{0}{1} \cup \intoo{2}{3}$: \omterm{def:b1:topology:open}{abrir} (união de conjuntos abertos), não
\omterm{def:b1:topology:closed}{fechado} ($\frac 1n \to 0$ fora).

$\intco{0}{1}$: nenhum. Não \omterm{def:b1:topology:open}{abrir} (não \omterm{prop:b1:reals:intervals}{intervalo} em volta $0$ dentro);
não \omterm{def:b1:topology:closed}{fechado} ($1 - \frac1n \to 1 \notin $ \omterm{def:b1:logic:sets}{definir}).

$\{0\} \cup \intcc{1}{2}$: \omterm{def:b1:topology:closed}{fechado} (união finita de conjuntos \omterm{def:b1:topology:closed}{fechados}), não
\omterm{def:b1:topology:open}{abrir} (falha em $0$).

$\R \setminus \Z $: \omterm{def:b1:topology:open}{abrir} ($\Z $ é \omterm{def:b1:topology:closed}{fechado}), não \omterm{def:b1:topology:closed}{fechado}: a sequência
$\bigl(\frac 1n\bigr)$ está nele, mas seu \omterm{def:b1:topology:closure}{limite} $0$ pertence a $\Z $,
ou seja,\ escapa do \omterm{def:b1:logic:sets}{definir}.

$\Q \cap \intoo{0}{1}$: nenhum. Não \omterm{def:b1:topology:open}{abrir}: todo \omterm{prop:b1:reals:intervals}{intervalo} em torno de um
racional contém irracionais. Não \omterm{def:b1:topology:closed}{fechado}: contém sequências
tendendo ao irracional $\frac{\sqrt 2}{2}$ (densidade).
\end{solution}

\begin{solution}{exo:b1:topology:2}
$ A = \intoc{0}{1} \cup \{2\}$: $\mathring A = \intoo{0}{1}$,
$\overline A = \intcc{0}{1} \cup \{2\}$, $\partial A = \{0, 1, 2\}$.

$ A = \R \setminus \Q $: $\mathring A = \emptyset $ (todo \omterm{prop:b1:reals:intervals}{intervalo}
contém racionais), $\overline A = \R $ (densidade do
irracionais), $\partial A = \R $.

$ A = \bigl\{\frac{(-1)^n n}{n+1}\bigr\}$: os termos pares tendem a $1$,
os termos estranhos para $-1$, nem pertence a $ A $. $\mathring A =
\emptyset $ (pontos isolados), $\overline A = A \cup \{-1, 1\}$,
$\partial A = \overline A $.
\end{solution}

\begin{solution}{exo:b1:topology:3}
\emph{Complementos.} $ F = \{a_1 < a_2 < \dots < a_k\}$: o complemento
é a união do \omterm{def:b1:topology:open}{abrir} \omterm{prop:b1:reals:intervals}{intervalos} $\intoo{-\infty}{a_1}$,
$\intoo{a_i}{a_{i+1}}$, $\intoo{a_k}{+\infty}$ --- \omterm{def:b1:topology:open}{abrir} por
\cref{prop:b1:topology:openstable}.

\emph{Sequências.} Deixar $ u_n \in F $, $ u_n \to \ell $. Com $\varepsilon
= \min\{\abs{a_i - a_j} : i \neq j\}/2 > 0$ (ou qualquer $\varepsilon $ se
$ F $ é um singleton): além de alguma classificação, todos os termos estão dentro
$\varepsilon $ de $\ell $, portanto, dentro $2\varepsilon $ um do outro,
o que os obriga a ser um único $ a_i $ dessa classificação em diante; então
$\ell = a_i \in F $.
\end{solution}

\begin{solution}{exo:b1:topology:4}
$\subseteq $: $\overline A \cup \overline B $ é \omterm{def:b1:topology:closed}{fechado} (união finita)
e contém $ A \cup B $, portanto contém o \emph{menor} \omterm{def:b1:topology:closed}{fechado}
superconjunto $\overline{A \cup B}$. $\supseteq $: $ A \subseteq A \cup B $
dá $\overline A \subseteq \overline{A \cup B}$ (\omterm{def:b1:topology:closure}{encerramento} é
monótono: pontos aderentes de $ A $ são adeptos do maior \omterm{def:b1:logic:sets}{definir}) e
da mesma forma para $ B $.

Contra-exemplo para interseções: $ A = \intoo{0}{1}$, $ B =
\intoo{1}{2}$: $\overline{A \cap B} = \overline\emptyset = \emptyset $
mas $\overline A \cap \overline B = \{1\}$.
\end{solution}

\begin{solution}{exo:b1:topology:5}
Use a caracterização sequencial
(\cref{thm:b1:topology:seqclosed}). Deixar $ S = \{u_n\} \cup \{\ell\}$
e $(v_k)$ uma sequência em $ S $ com $ v_k \to m $; mostrar $ m \in S $. Dois
casos. Se algum valor $ v \in S $ é levado por $(v_k)$ infinitamente frequentemente,
então uma constante \omterm{def:b1:seq:subsequence}{subsequência} dá $ m = v \in S $. Caso contrário, cada
o valor é obtido com uma frequência finita; em particular, para cada $ n $, O termo
$ u_n $ aparece com uma frequência finita e $\ell $ também. Então para cada $ N $, o
índices $ k $ com $ v_k \in \{u_0, \dots, u_N, \ell\}$ são finitamente
muitos: o restante $ v_k $ são termos $ u_n $ com $ n > N $. Dado
$\varepsilon > 0$, escolher $ N $ com $\abs{u_n - \ell} \leq
\varepsilon $ para $ n > N $: todos, exceto um número finito $ v_k $ satisfazer
$\abs{v_k - \ell} \leq \varepsilon $. Por isso $ v_k \to \ell $, então $ m =
\ell \in S $.
\end{solution}

\begin{solution}{exo:b1:topology:6}
$ U + A = \bigcup_{a \in A} (U + a)$, e cada um traduz $ U + a $ é
\omterm{def:b1:topology:open}{abrir} (traduzindo o \omterm{prop:b1:reals:intervals}{intervalo} certificados). Uma união de conjuntos abertos é
\omterm{def:b1:topology:open}{abrir} (\cref{prop:b1:topology:openstable}).

Conjuntos \omterm{def:b1:topology:closed}{fechados}: $\Z $ e $\sqrt 2\,\Z $ são \omterm{def:b1:topology:closed}{fechado} (como $\alpha\Z $:
sequências convergentes são eventualmente constantes, cf.\
\cref{ex:b1:topology:closed}). A soma deles $\Z + \sqrt 2\,\Z $ é \omterm{def:b1:topology:dense}{denso}
em $\R $ (\cref{exo:b1:reals:9}) mas não é $\R $ (é contável, ou
simplesmente: $\frac{\sqrt 2}{2} \notin \Z + \sqrt2\Z $, outro $\sqrt 2$
seria racional --- escrever $\frac{\sqrt2}{2} = m + n\sqrt 2$
forças $(2n - 1)\sqrt 2 = -2m $, então $\sqrt 2 \in \Q $ a menos que $ n =
\frac12$, impossível). UM \omterm{def:b1:topology:dense}{denso} subconjunto adequado não é \omterm{def:b1:topology:closed}{fechado}: isso é
\omterm{def:b1:topology:closure}{encerramento} é $\R \neq $ em si.
\end{solution}

\begin{solution}{exo:b1:topology:7}
$\Z $: o \omterm{def:b1:topology:open}{vizinhança} $\intoo{n - \frac12}{n + \frac12}$ de $ n $
conhece $\Z $ apenas em $ n $.

$ A = \{\frac1n : n \in \N^*\}$: em volta $\frac 1n $, o \omterm{prop:b1:reals:intervals}{intervalo} de
raio $\frac{1}{n} - \frac{1}{n+1} = \frac{1}{n(n+1)}$ (metade, digamos)
isola-o de seus vizinhos --- todos os pontos isolados. Ainda $0 \in
\overline A \setminus A $: pontos isolados não impedem a aderência
estranhos.

Se todos os pontos de $ A $ estão isolados: nenhum ponto de $ A $ é \omterm{def:b1:topology:closure}{interior}, desde
um \omterm{def:b1:topology:closure}{interior} ponto tem um todo \omterm{prop:b1:reals:intervals}{intervalo} de $ A $-vizinhos ao redor
(um \omterm{prop:b1:reals:intervals}{intervalo} é infinito), contradizendo o isolamento. Então $\mathring A
= \emptyset $.
\end{solution}

\begin{solution}{exo:b1:topology:8}
Se $ A \subseteq \intcc{-M}{M}$, o conjunto \omterm{def:b1:topology:closed}{fechado} $\intcc{-M}{M}$
contém $ A $, portanto contém $\overline A $ (menor \omterm{def:b1:topology:closed}{fechado}
superconjunto): $\overline A $ é limitado.

Deixar $ s = \sup A $ (finito). Desde $ A \subseteq \overline A $, $\sup
\overline A \geq s $. Por outro lado, $\overline A \subseteq
\intoc{-\infty}{s}$: a meia-linha é \omterm{def:b1:topology:closed}{fechado} e contém $ A $; então
cada elemento de $\overline A $ é $\leq s $, dando $\sup\overline A
\leq s $. Igualdade.

$ s \in \overline A $: pelo $\varepsilon $-caracterização
(\cref{prop:b1:reals:epsilon}), todo \omterm{prop:b1:reals:intervals}{intervalo} $\intoo{s -
\varepsilon}{s + \varepsilon}$ contém um elemento de $ A $: $ s $ é
aderente.
\end{solution}

\begin{solution}{exo:b1:topology:9}
Suponha $ A $ é \omterm{def:b1:topology:open}{abrir} e \omterm{def:b1:topology:closed}{fechado}, com $ a \in A $ e $ b \in \R
\setminus A $; sem perda de generalidade $ a < b $. O \omterm{def:b1:logic:sets}{definir} $ B = A \cap
\intcc{a}{b}$ não está vazio ($ a $), limitado: deixe $ s = \sup B $. Por
\cref{exo:b1:topology:8}, $ s \in \overline B \subseteq \overline A =
A $ ($ A $ \omterm{def:b1:topology:closed}{fechado}). Observação $ s \leq b $, e desde $ b \notin A $: $ s < b $.
Agora $ A $ é \omterm{def:b1:topology:open}{abrir}: alguns \omterm{prop:b1:reals:intervals}{intervalo} $\intoo{s - r}{s + r}$ reside em $ A $,
e podemos levar $ r < b - s $. Então $ s + \frac{r}{2}$ pertence a $ A
\cap \intcc{a}{b} = B $ e excede $ s $ --- contraditório $ s = \sup
B $. Portanto, não existe tal par $(a, b)$ existe: um dos $ A $, $\R \setminus A $
está vazio.
\end{solution}

\begin{solution}{exo:b1:topology:10}
$ I_x $ é uma união de \omterm{def:b1:topology:open}{abrir} \omterm{prop:b1:reals:intervals}{intervalos} todos contendo $ x $: isso é \omterm{def:b1:topology:open}{abrir},
e é um \omterm{prop:b1:reals:intervals}{intervalo}, sendo convexo --- se $ u < z < v $ com $ u, v \in
I_x $, então $ u $ e $ v $ deitar em \omterm{def:b1:topology:open}{abrir} subintervalos $ J_u \ni x $, $ J_v \ni
x $ de $ U $, e $ J_u \cup J_v $ é um \omterm{prop:b1:reals:intervals}{intervalo} (ambos contêm $ x $)
dentro $ U $ contendo $ z $; então $ z \in I_x $
(\cref{prop:b1:reals:intervals}).

Se $ I_x \cap I_y \neq \emptyset $: $ I_x \cup I_y $ é então um \omterm{def:b1:topology:open}{abrir}
\omterm{prop:b1:reals:intervals}{intervalo} (convexo: dois sobrepostos \omterm{prop:b1:reals:intervals}{intervalos}) contido em $ U $
contendo $ x $ e $ y $, então $ I_x \cup I_y \subseteq I_x $ e
$\subseteq I_y $ pela maximidade de cada um: $ I_x = I_y $.

Então $ U $ é a união disjunta do distinto conjuntos $ I_x $ (cada $ x \in
U $ reside em si mesmo $ I_x $). Contabilidade: cada não vazio \omterm{def:b1:topology:open}{abrir} \omterm{prop:b1:reals:intervals}{intervalo}
$ I $ da família contém uma racionalidade $ q_I $
(\cref{thm:b1:reals:density}) e disjunto distinto \omterm{prop:b1:reals:intervals}{intervalos} obter
racionais distintos: a família injeta $\Q $, que é contável
(é indexado por pares de inteiros). Portanto, no máximo, muitos contáveis
\omterm{prop:b1:reals:intervals}{intervalos}.
\end{solution}

\begin{solution}{exo:b1:topology:11}
$\overline A = A \cup A'$. ($\supseteq $) $ A \subseteq \overline
A $ sempre; e se $ x \in A'$, todo \omterm{def:b1:topology:open}{vizinhança} de $ x $ conhece $ A
\setminus \{x\} \subseteq A $, então $ x $ é aderente. ($\subseteq $)
Deixar $ x \in \overline A $. Se $ x \in A $, feito. Se $ x \notin A $,
todo \omterm{def:b1:topology:open}{vizinhança} de $ x $ conhece $ A = A \setminus \{x\}$: $ x \in
A'$.

Consequentemente $ A $ \omterm{def:b1:topology:closed}{fechado} $\iff $ $ A = \overline A = A \cup A'$
$\iff $ $ A' \subseteq A $.

$ A = \{\frac 1n\}$: $0$ é um ponto de acumulação ($\frac 1n \to
0$, termos $\neq 0$); cada $\frac 1n $ está isolado
(\cref{exo:b1:topology:7}), então não em $ A'$; e um ponto $ x
\notin A \cup \{0\}$ tem um todo \omterm{prop:b1:reals:intervals}{intervalo} evitando $ A $ (entre
os dois vizinhos de $ x $ em $ A \cup \{0\}$, ou além $1$).
Por isso $ A' = \{0\}$.

$\Z' = \emptyset $: todo número inteiro é isolado, todo não inteiro
tem um \omterm{def:b1:topology:open}{vizinhança} dentro $\R \setminus \Z $.

$\Q' = \R $: todo \omterm{prop:b1:reals:intervals}{intervalo} em torno de qualquer real contém infinitamente
muitos racionais (\cref{thm:b1:reals:density}), em especial um
diferente do centro.
\end{solution}

\begin{solution}{exo:b1:topology:12}
\begin{enumerate}
  \item Para cada $ a \in A $: $\abs{x - a} \leq \abs{x - y} +
        \abs{y - a}$, então $ d_A(x) \leq \abs{x - y} + \abs{y - a}$;
        tomando o \omterm{def:b1:reals:bounds}{ínfimo} sobre $ a $: $ d_A(x) \leq \abs{x - y} +
        d_A(y)$. Troca $ x $ e $ y $ dá o outro
        desigualdade: $\abs{d_A(x) - d_A(y)} \leq \abs{x - y}$.
  \item $ d_A(x) = 0$ $\iff $ para cada $\varepsilon > 0$ há
        $ a \in A $ com $\abs{x - a} < \varepsilon $ $\iff $ todo
        \omterm{prop:b1:reals:intervals}{intervalo} em volta $ x $ conhece $ A $ $\iff $ $ x \in \overline A $.
        Se $ F $ é \omterm{def:b1:topology:closed}{fechado} e $ x \notin F = \overline F $, então
        $ d_F(x) \neq 0$, ou seja\ $ d_F(x) > 0$.
  \item Se $ d_F(x) < \varepsilon $, \omterm{def:b1:logic:sets}{definir} $ r = \varepsilon -
        d_F(x) > 0$: para $\abs{y - x} < r $, parte (1) dá
        $ d_F(y) \leq d_F(x) + \abs{x - y} < \varepsilon $: o
        \omterm{prop:b1:reals:intervals}{intervalo} $\intoo{x - r}{x + r}$ reside em $ V_\varepsilon $,
        o que é portanto \omterm{def:b1:topology:open}{abrir}; contém $ F $ desde $ d_F = 0$
        lá. Finalmente $ x \in \bigcap_{\varepsilon>0}
        V_\varepsilon $ $\iff $ $ d_F(x) < \varepsilon $ para todos
        $\varepsilon $ $\iff $ $ d_F(x) = 0$ $\iff $ $ x \in \overline
        F = F $. Desde $\bigcap_{\varepsilon > 0} V_\varepsilon =
        \bigcap_{n \geq 1} V_{1/n}$, todo conjunto \omterm{def:b1:topology:closed}{fechado} é um
        interseção contável de conjuntos abertos.
\end{enumerate}
\end{solution}

\begin{solution}{pb:b1:topology:1}
\textbf{1.} $ C_1 = \intcc{0}{\frac13} \cup \intcc{\frac23}{1}$
e
\[
C_2 = \intcc{0}{\tfrac19} \cup \intcc{\tfrac29}{\tfrac13}
\cup \intcc{\tfrac23}{\tfrac79} \cup \intcc{\tfrac89}{1} .
\]
Indução: se $ C_n $ é uma união disjunta de $2^n $ \omterm{def:b1:topology:closed}{fechado} segmentos
de comprimento $3^{-n}$, excluindo o \omterm{def:b1:topology:open}{abrir} terço médio de cada
deixa dois \omterm{def:b1:topology:closed}{fechado} segmentos de comprimento $3^{-n-1}$ por pai:
$2^{n+1}$ segmentos, disjuntos aos pares (filhos de distintos
os pais estão separados porque os pais estavam; filhos de um
pai são separados pela lacuna removida).

\textbf{2.} Cada $ C_n $ é uma união finita de segmentos, portanto
\omterm{def:b1:topology:closed}{fechado}; $ C = \bigcap C_n $ é uma intersecção de conjuntos \omterm{def:b1:topology:closed}{fechados}:
\omterm{def:b1:topology:closed}{fechado} (\cref{def:b1:topology:closed}); limitado ($\subseteq
\intcc{0}{1}$): compacto por \cref{thm:b1:topology:compact}.
Não vazio: $0$ encontra-se no segmento mais à esquerda de cada $ C_n $. Deixar
$ a $ ser um ponto final de um segmento $ S $ de $ C_n $. Para $ m \leq n $,
$ a \in C_n \subseteq C_m $. Para as fases posteriores: o terço médio
a exclusão nunca remove um endpoint e $ a $ é novamente um ponto final
de um dos dois filhos de $ S $ (a criança tocando $ a $); por
indução $ a \in C_m $ para todos $ m \geq n $: $ a \in C $.

\textbf{3.} Comprimento total de $ C_n $: $2^n \cdot 3^{-n} =
(\frac23)^n \to 0$. Dado $\varepsilon > 0$, escolher $ n $ com
$(\frac23)^n \leq \varepsilon $: então $ C \subseteq C_n $, um sindicato
de um número finito de segmentos de comprimento total $\leq \varepsilon $.

\textbf{4.} Deixar $ I \subseteq C $ feijão \omterm{prop:b1:reals:intervals}{intervalo} com dois distintos
pontos. Para cada $ n $: $ I \subseteq C_n $, e $ I $, sendo convexo,
deve estar dentro de um \emph{solteiro} segmento de $ C_n $ (encontro dois
segmentos forçariam $ I $ para conter um ponto da lacuna entre
eles, que está fora $ C_n $). Daí o comprimento de $ I $ é $\leq
3^{-n}$ para todos $ n $: contradição. Então o único \omterm{prop:b1:reals:intervals}{intervalos}
dentro $ C $ estão vazios ou singletons; em particular não
$\intoo{x-r}{x+r}$ cabe dentro $ C $: $\mathring C = \emptyset $.
Como $ C $ é \omterm{def:b1:topology:closed}{fechado}, $\overline C = C $ está vazio \omterm{def:b1:topology:closure}{interior}: $ C $ é
lugar nenhum \omterm{def:b1:topology:dense}{denso}.

\textbf{5.} Escrever $\varphi_0(x) = \frac x3$ e $\varphi_2(x) =
\frac{2 + x}{3}$, aumentando as bijeções afins de $\intcc{0}{1}$
para $\intcc{0}{\frac13}$ e $\intcc{\frac23}{1}$. Alegar:
$ C_{n+1} = \varphi_0(C_n) \cup \varphi_2(C_n)$. Para $ n = 0$ isso
é a questão 1. Indução: um afim crescente \omterm{def:b1:logic:map}{mapa} envia o
terço médio de um segmento até o terço médio da imagem
segmento, portanto, a exclusão dos terços médios comuta com $\varphi_0$ e
$\varphi_2$; aplicando a etapa de exclusão para $ C_{n+1} =
\varphi_0(C_n) \cup \varphi_2(C_n)$ rendimentos $ C_{n+2} =
\varphi_0(C_{n+1}) \cup \varphi_2(C_{n+1})$. Cruzando
$ n $: para $ x \leq \frac13$, $ x \in C \iff x \in \varphi_0(C_n)$
para todos $ n $ $\iff 3x \in \bigcap C_n = C $; da mesma forma em
$\intcc{\frac23}{1}$; e nenhum ponto de $\intoo{\frac13}{\frac23}$
reside em $ C_1$. Por isso $ C = \varphi_0(C) \cup \varphi_2(C)$,
de forma desconexa.

\textbf{6.} Indução ativada $ n $; o caso $ n = 0$ diz cada $ x \in
\intcc{0}{1}$ tem um código, que é \cref{pb:b1:reals:1}
(pergunta 9 para $ x < 1$; $1 = (0.\overline 2)_3$). Suponha que
equivalência na classificação $ n $. Se $ x \in C_{n+1}$: pela pergunta 5, $ x =
\varphi_i(z)$ com $ z \in C_n $ e $ i \in \{0, 2\}$; se $(e_k)$
é um código de $ z $ com primeiro $ n $ dígitos $1$-livre, então $(i, e_1,
e_2, \dots)$ tem somas parciais $\frac i3 + \frac13\sum_{k\leq m}
e_k 3^{-k} \to \varphi_i(z) = x $: um código de $ x $ com primeiro $ n +
1$ dígitos $1$-livre. Por outro lado, se $ x $ tem um código $(d_k)$ com
$ d_1, \dots, d_{n+1} \in \{0, 2\}$: a string deslocada $(d_2,
d_3, \dots)$ tem algum valor $ z \in \intcc{0}{1}$, seu primeiro $ n $
dígitos são $1$-free, e o cálculo da soma parcial é lido
para trás dá $ x = \varphi_{d_1}(z)$; por indução $ z \in
C_n $, então $ x \in C_{n+1}$ pela pergunta 5. Finalmente: uma avaliação totalmente
$1$- colocações de código grátis $ x $ em cada $ C_n $, portanto em $ C $; inversamente
se $ x \in C $, então para cada $ n $ um dos no máximo dois códigos de
$ x $ (\cref{pb:b1:reals:1}, questão 11) tem seu primeiro $ n $ dígitos
$1$-livre; um código fixo deve funcionar para valores arbitrariamente grandes $ n $
(categoria entre dois códigos) e um código cujo primeiro $ n $
dígitos são $1$-free para arbitrariamente grande $ n $ é $1$-grátis
imediatamente.

\textbf{7.} $0 = (0.\overline 0)_3$, $1 = (0.\overline 2)_3$,
$\frac13 = (0.0\overline{2})_3$ (o gêmeo impróprio de $(0.1)_3$),
$\frac23 = (0.2\overline{0})_3$. Somas geométricas:
\[
(0.\overline{02})_3 = \sum_{j\geq1} \frac{2}{9^{\,j}}
= \frac{2/9}{1 - 1/9} = \frac14 ,
\qquad
(0.\overline{20})_3 = \sum_{j\geq1} \frac{2}{3\cdot 9^{\,j-1}}
= \frac{2/3}{1 - 1/9} = \frac34 ,
\]
ambos $1$-livre: $\frac14, \frac34 \in C $. (As somas infinitas
abreviar a supremacia das somas parciais, como em \cref{pb:b1:reals:1}.)
Pontos finais de segmentos de $ C_n $ são da forma $ m/3^n $
(indução: os endpoints filhos são endpoints pais ou diferem
de um por um múltiplo de $3^{-n-1}$). Se $\frac14 =
\frac{m}{3^n}$ então $3^n = 4m $, e $4 \nmid 3^n $: impossível.
Então $\frac14 \in C $ sem nunca ser um ponto final.

\textbf{8.} Suponha $ x $ tinha dois distintos $1$-códigos gratuitos. Tendo
dois códigos de qualquer maneira (\cref{pb:b1:reals:1}, questão 11, base
$3$) que $ x = m/3^N \in \intoo{0}{1}$ e os dois códigos são:
o final, com o último dígito diferente de zero $ d_N \in \{1, 2\}$
seguido pela $0$ s, e seu gêmeo, com $ d_N - 1$ na posição $ N $
seguido pela $2$ S. Se $ d_N = 1$ o primeiro contém um $1$; se $ d_N
= 2$ o gêmeo carrega $ d_N - 1 = 1$. De qualquer forma, no máximo um dos
o par é $1$-livre: contradição. Então cada um $ x \in C $ tem
exatamente um $1$-código livre (existência pela questão 6), e
distinto $\{0,2\}$-strings têm valores distintos. Todo
$\{0,2\}$-string tem valor em $\intcc{0}{1}$ (somas parciais $\leq
1$) com todos os prefixos $1$-livre, portanto, valor em cada $ C_n $,
ou seja,\ em $ C $: o valor \omterm{def:b1:logic:map}{mapa} é uma bijeção de
$\{0,2\}$-strings em $ C $.

\textbf{9.} Deixar $(d^{(k)})$ seja o $1$-código gratuito de $ x_k $ e
\omterm{def:b1:logic:sets}{definir} $ e_k = 2 - d^{(k)}_k \in \{0, 2\}$: um $\{0,2\}$-string cujo
valor $ y $ reside em $ C $ e tem $(e_k)$ como é único $1$-grátis
código (questão 8). Para cada $ k $ os códigos de $ y $ e $ x_k $
diferem na posição $ k $, então $ y \neq x_k $: não \omterm{def:b1:logic:map}{mapa} $\N^* \to C $ é
\omterm{def:b1:logic:inj}{sobrejetivo}. Um incontável \omterm{def:b1:logic:sets}{definir} de comprimento zero: grandeza em
\omterm{def:b1:counting:card}{cardinalidade}, pequenez em medida --- simultaneamente.

\textbf{10.} Compacidade: fechamento do infinito
interseção mais limitação (questão 2) --- a única propriedade
não transportado por uma única contenção. Comprimento zero: $ C \subseteq
C_n $ com comprimento total $(\frac23)^n $ (questão 3). Em lugar nenhum
densidade: $ C \subseteq C_n $ forças \omterm{prop:b1:reals:intervals}{intervalos} dentro $ C $ ter
comprimento $\leq 3^{-n}$ (questão 4). A incontável viagem não
contenção: ela precisa de toda a estrutura de interseção,
codificado na bijeção da questão 8.

\textbf{11.} Virar $ d_n $ para $2 - d_n $: a nova string ainda é um
$\{0,2\}$-string, então seu valor $ x_n $ reside em $ C $; o parcial
somas além da classificação $ n $ diferem exatamente $2\cdot3^{-n}$, então
$\abs{x_n - x} = 2\cdot3^{-n}$. Por isso $ x_n \neq x $ e $ x_n \to
x $: cada ponto de $ C $ é um \omterm{def:b1:topology:closure}{limite} de outros pontos de $ C $ --- $ C $
é \emph{perfeito}, sem ponto isolado.

\textbf{12.} Pela indução da questão 5, os segmentos de
$ C_n $ são exatamente os $\intcc{t}{t + 3^{-n}}$ onde $ t $ corre
sobre os valores de comprimento $ n $ $\{0,2\}$-cordas. Dado $ x \in C $
com código $(d_k)$, o truncamento $ t_n $ (dígitos $ d_1 \dots d_n $
então $0$ s) é, portanto, um ponto final à esquerda, e $0 \leq x - t_n
\leq 3^{-n}$: os pontos finais são \omterm{def:b1:topology:dense}{denso} em $ C $. Eles formam um subconjunto de
$\{m/3^n : m, n\}$, um \omterm{def:b1:logic:sets}{definir} indexado por pares de inteiros, portanto
contável (como para $\Q $ em \cref{exo:b1:topology:10}). Desde $ C $
é incontável (pergunta 9), todos, exceto contáveis muitos pontos de
$ C $ não são pontos finais --- $\frac14$ (questão 7) é o visível
ponta desse iceberg.

\textbf{13.} Escolha $ n $ com $3^{-n} < y - x $. Ambos $ x, y \in
C_n $, e eles não podem estar no mesmo segmento (comprimento $3^{-n} <
y - x $): a lacuna removida entre seus segmentos fornece $ z $
com $ x < z < y $ e $ z \notin C_n \supseteq C $. Portanto, quaisquer dois
pontos de $ C $ são separados pelo complemento: o único convexo
subconjuntos de $ C $ são solteiros --- $ C $ está totalmente desconectado.

\textbf{14.} Se $(d_k)$ é o $1$-código gratuito de $ x $, a corda
$(2 - d_k)$ é novamente um $\{0,2\}$-string, com somas parciais
\[
\sum_{k=1}^{n} (2 - d_k)3^{-k} = (1 - 3^{-n}) - \sum_{k=1}^n
d_k 3^{-k} \longrightarrow 1 - x :
\]
então $1 - x \in C $. Por isso $1 - C \subseteq C $, e aplicando o \omterm{def:b1:logic:map}{mapa}
duas vezes dá $1 - C = C $: o \omterm{pb:b1:topology:1}{Conjunto de cantores} é simétrico em relação a
$\frac12$.

\textbf{15.} As somas parciais $ t_n \to x $ e $ t'_n \to x'$
(sequências crescentes convergem para seus \omterm{def:b1:reals:bounds}{supremo}, ou seja\ o
valor), então $ t_n + t'_n \to x + x'$ por
\cref{thm:b1:seq:operations}. Dado $ y \in \intcc{0}{1}$ com
código $(e_k)$, escolher $(a_k, b_k) = (0,0), (0,2), (2,2)$
de acordo com $ e_k = 0, 1, 2$: então $ a_k + b_k = 2e_k $, o
cordas $(a_k)$, $(b_k)$ são $\{0,2\}$-strings com valores $ x,
x' \in C $, e
\[
x + x' = \lim_n\,(t_n + t'_n) = \lim_n 2\sum_{k=1}^n e_k 3^{-k}
= 2y :
\]
todo $ y \in \intcc{0}{1}$ é o ponto médio de dois pontos de $ C $.

\textbf{16.} A questão 15 dá $\intcc{0}{2} = 2\,\intcc{0}{1}
\subseteq C + C $, e $ C + C \subseteq \intcc{0}{1} +
\intcc{0}{1} = \intcc{0}{2}$: igualdade. Então, usando $1 - C = C $:
\[
C - C = C + (C - 1) = (C + C) - 1 = \intcc{-1}{1} .
\]
Instância concreta: $1 = \frac14 + \frac34$, uma soma de dois
membros não terminais de $ C $.

\textbf{17.} O comprimento mede quanto da linha o \omterm{def:b1:logic:sets}{definir} em si
ocupa; não diz nada sobre o \omterm{def:b1:logic:sets}{definir} de \emph{somas}, que é
a imagem da família de dois parâmetros $ C \times C $ sob $(x,
x') \mapsto x + x'$ --- as strings de dois dígitos são escolhidas
de forma independente, e essa liberdade é exatamente o que preenche
$\intcc{0}{2}$. Nenhum teorema limita o comprimento de um conjunto pela
comprimentos dos summands, e $ C $ é a prova de que ninguém pode.

\textbf{18.} Por \cref{pb:b1:reals:1} (questão 18), $ x $ é
racional se sua expansão adequada for eventualmente periódica. O
$1$-código gratuito de $ x \in C $ é essa expansão adequada ou
o gêmeo impróprio de um terminal; uma string de terminação e
seu gêmeo (eventualmente constante $2$ s) são ambos eventualmente
periódico, então a periodicidade do $1$-código livre é equivalente a
racionalidade de $ x $. Longa \omterm{thm:b1:logic:partition}{divisão} de $\frac1{13}$ na base $3$
($ r_0 = 1$): $3 = 13\cdot0 + 3$, $9 = 13\cdot0 + 9$, $27 =
13\cdot2 + 1$, e o restante retorna para $1$: dígitos
$\overline{002}$, então $\frac1{13} = (0.\overline{002})_3$,
$1$-livre e periódico: um membro racional de $ C $. (Verificar:
$\frac{2/27}{1 - 1/27} = \frac{2}{26} = \frac1{13}$.)

\textbf{19.} A corda com $ d_k = 2$ no triangular
posições $ k = \frac{j(j+1)}{2}$ e $0$ em outro lugar é um
$\{0,2\}$-string, então seu valor $ x^*$ pertence a $ C $ (pergunta
8). Tem infinitos $2$ está com lacunas $ j + 1 \to \infty $
entre consecutivos, portanto não é eventualmente periódico (um
período $ T $ eventualmente forçaria $2$ está em lacunas $\leq T $: o
argumento das lacunas crescentes de \cref{pb:b1:reals:1}, questão 20); por
questão 18, $ x^* \notin \Q $. E pela questão 9 mais o
contabilização de $\Q $, todos, exceto muitos membros contáveis $ C $ são
irracional: $ x^*$ é a norma, não a exceção.

\textbf{20.} Deixar $ y \in \intcc{0}{1}$: tem um código binário
$(c_k)$ com $ c_k \in \{0, 1\}$ (\cref{pb:b1:reals:1}, pergunta
9, básico $2$; $ y = 1$ leva tudo-$1$ corda). Então $(2c_k)$
é um $\{0,2\}$-string, seu valor $ x $ reside em $ C $, e $ h(x)$ é
o valor de $(c_k)$, nomeadamente $ y $: $ h $ \omterm{def:b1:logic:map}{mapas} $ C $ para
$\intcc{0}{1}$. Se $ C $ eram a imagem de um \omterm{def:b1:logic:map}{mapa} de $\N^*$,
compondo com $ h $ listaria todos $\intco{0}{1}$,
contradizendo o teorema da diagonal de \cref{pb:b1:reals:1}
(questão 22): $ C $ é incontável, novamente. Um comprimento zero \omterm{def:b1:logic:sets}{definir}
sobrepondo-se a um segmento completo.

\textbf{21.} Pela pergunta 5, $ C $ é a união disjunta de
$\varphi_0(C)$ e $\varphi_2(C)$, cada um uma tradução do
cópia em escala $\frac13 C $. Aditividade, dimensionamento e tradução
invariância dar
\[
L(C) = L(\varphi_0(C)) + L(\varphi_2(C))
= \tfrac13 L(C) + \tfrac13 L(C) = \tfrac23\,L(C),
\]
então $\frac13 L(C) = 0$: $ L(C) = 0$. Auto-similaridade sozinha
sentenças $ C $ ao comprimento zero --- a questão 3 apenas executou o
sentença.

\textbf{22.} Um segmento de comprimento $ l_n $ perde um central \omterm{prop:b1:reals:intervals}{intervalo}
de comprimento $4^{-(n+1)}$, deixando dois segmentos de comprimento $ l_{n+1}
= \frac{l_n - 4^{-(n+1)}}{2}$; de $ l_0 = 1$, indução
confirma $ l_n = \frac{2^n + 1}{2\cdot4^n}$: de fato
$\frac12\Bigl(\frac{2^n+1}{2\cdot4^n} - \frac{1}{4^{n+1}}\Bigr)
= \frac{2(2^n + 1) - 1}{2\cdot4^{n+1}} = \frac{2^{n+1} +
1}{2\cdot4^{n+1}}$, e $ l_n > 0$ sempre: a construção nunca
morre de fome. $ K = \bigcap K_n $ é \omterm{def:b1:topology:closed}{fechado} e limitado, portanto compacto;
um \omterm{prop:b1:reals:intervals}{intervalo} dentro $ K $ está em um segmento de $ K_n $, de comprimento
$ l_n \to 0$: vazio \omterm{def:b1:topology:closure}{interior}. Comprimento removido: $\sum_{n\geq0} 2^n
\cdot 4^{-(n+1)} = \frac14\sum_{n\geq0}\bigl(\frac12\bigr)^n =
\frac12$, e cada $ K_n $ tem comprimento total $2^n l_n = \frac{2^n +
1}{2^{n+1}} > \frac12$. Agora deixe um número finito \omterm{def:b1:topology:open}{abrir} \omterm{prop:b1:reals:intervals}{intervalos}
ter união $ U \supseteq K $. Pelo companheiro \omterm{def:b1:logic:statement}{declaração} de
\cref{ex:b1:topology:nested}, $ U \supseteq K_n $ para alguns $ n $;
admitindo aditividade de comprimento em uniões finitas de \omterm{prop:b1:reals:intervals}{intervalos},
o comprimento total da cobertura \omterm{prop:b1:reals:intervals}{intervalos} é pelo menos o de
$ K_n $, que ultrapassa $\frac12$. Então $ K $ não está em lugar nenhum \omterm{def:b1:topology:dense}{denso}, ainda não
existe uma cobertura barata: pequenez topológica (em nenhum lugar \omterm{def:b1:topology:dense}{denso}) e
pequenez métrica (comprimento zero) são noções genuinamente diferentes,
e $ K $ os separa.

\textbf{23.} Realização: deixe $ d = d(x, F)$ e escolha $ a_k \in F $
com $\abs{x - a_k} \leq d + \frac1k $: o $ a_k $ são limitados, então
Bolzano-Weierstrass (\cref{thm:b1:seq:bw}) extratos
$ a_{\varphi(k)} \to a $, com $ a \in F $ ($ F $ \omterm{def:b1:topology:closed}{fechado},
\cref{thm:b1:topology:seqclosed}) e $\abs{x - a} = \lim
\abs{x - a_{\varphi(k)}} = d $. Agora o máximo: se $ y \in C $,
$ d(y, C) = 0$; de outra forma $ y $ reside em uma lacuna removida em algum momento
$ n \geq 1$, um \omterm{def:b1:topology:open}{abrir} \omterm{prop:b1:reals:intervals}{intervalo} de comprimento $3^{-n}$ cujos dois
pontos finais pertencem a $ C $ (questão 2), então $ d(y, C) \leq
\frac{3^{-n}}{2} \leq \frac16$, com a igualdade exigindo $ n = 1$
e $ y $ no centro da lacuna $\intoo{\frac13}{\frac23}$,
ou seja\ $ y = \frac12$; e de fato $ d\bigl(\frac12, C\bigr) =
\frac16$ desde $ C \cap \intoo{\frac13}{\frac23} = \emptyset $ e
$\frac13, \frac23 \in C $. Por isso $\max_{y\in\intcc{0}{1}} d(y, C)
= \frac16$, alcançado exatamente em $\frac12$.

\textbf{24.} Os pontos finais formam um contável \omterm{def:b1:topology:dense}{denso} subconjunto de $ C $
(questão 12): liste-os como uma única sequência $(e_j)_{j\geq1}$,
uma sequência em $ C $. Seus \omterm{def:b1:topology:closure}{limites} subseqüentes estão todos em $ C $ ($ C $
\omterm{def:b1:topology:closed}{fechado}). Por outro lado, consertar $ x \in C $: para cada $ n $, os segmentos
do $ C_m $ contendo $ x $ ($ m \geq n $) têm seus pontos finais
dentro $3^{-m} \leq 3^{-n}$ de $ x $, então infinitamente muitos distintos
os pontos finais estão dentro $3^{-n}$ de $ x $; escolha índices $ j_1 < j_2
< \dots $ com $\abs{e_{j_n} - x} \leq 3^{-n}$: um \omterm{def:b1:seq:subsequence}{subsequência}
convergindo para $ x $. Então o \omterm{def:b1:logic:sets}{definir} de \omterm{def:b1:topology:closure}{limites} subsequentes de $(e_j)$
é exatamente $ C $ --- um agrupamento de sequência em incontáveis
pontos, o extremo oposto de uma sequência convergente, cujo
aglomerado \omterm{def:b1:logic:sets}{definir} é um único.

\textbf{25.} (i) A construção consumida: estabilidade de \omterm{def:b1:topology:closed}{fechado}
conjuntos sob interseção arbitrária (existência de $ C $ como um \omterm{def:b1:topology:closed}{fechado}
\omterm{def:b1:logic:sets}{definir}), o teorema da compacidade \cref{thm:b1:topology:compact}
(questões 2, 22, 23), e as caracterizações sequenciais de
fechamento e adesão (o argumento dos compactos aninhados de
\cref{ex:b1:topology:nested} e questão 23). (ii) Os quatro
emparelhamentos: comprimento zero mas incontável (questões 3, 9); \omterm{def:b1:topology:closed}{fechado}
ainda com vazio \omterm{def:b1:topology:closure}{interior} (questão 4); perfeito --- não isolado
ponto --- ainda totalmente desconectado (questões 11, 13);
insignificante ainda com $ C + C = \intcc{0}{2}$ (questão 16). (iii)
A sobrejeção $ h $ da questão 20, feita contínua e
não decrescente, torna-se a escada do diabo na teoria da
funções contínuas; e na teoria da medida do volume do Ano 3,
$ C$ é a testemunha padrão de que ``desprezível'' não significa
``contável'', com seu primo gordo (questão 22) separando
``em lugar nenhum \omterm{def:b1:topology:dense}{denso}'' de ``insignificante''.
\end{solution}
