\begin{solution}{exo:b1:findim:1}
\begin{enumerate}
  \item $z = -x - y $: $ F = \{(x, y, -x-y)\} =
        \operatorname{Vect}\bigl((1,0,-1), (0,1,-1)\bigr)$; os dois
        vetores são \omterm{def:b1:vspaces:free}{livre} (\omterm{prop:b1:vspaces:coordinates}{coordenadas}): $\dim F = 2$.
  \item $ G = \{(x, x, 2t, t)\} =
        \operatorname{Vect}\bigl((1,1,0,0), (0,0,2,1)\bigr)$: \omterm{def:b1:vspaces:free}{livre},

  \item $ P(1) = P'(1) = 0$ significa $(X-1)^2 \mid P $
        (\cref{prop:b1:poly:multiplicity}): $ P = (X-1)^2(aX + b)$.
        \omterm{def:b1:vspaces:free}{Base} $\bigl((X-1)^2, X(X-1)^2\bigr)$, dimensão $2$.
\end{enumerate}
\end{solution}

\begin{solution}{exo:b1:findim:3}
Tente anexar $ e_1 = (1,0,0,0)$ e $ e_3 = (0,0,1,0)$. A família
$\bigl((1,1,0,0), (0,0,1,1), e_1, e_3\bigr)$ é \omterm{def:b1:vspaces:free}{livre}: um nulo
combinação $\alpha(1,1,0,0) + \beta(0,0,1,1) + \gamma e_1 + \delta
e_3 = 0$ lê $(\alpha + \gamma, \alpha, \beta + \delta, \beta) =
0$, então $\alpha = \beta = 0$, então $\gamma = \delta = 0$. Quatro \omterm{def:b1:vspaces:free}{livre}
vetores em dimensão $4$: um \omterm{def:b1:vspaces:free}{base}
(\cref{prop:b1:findim:twoofthree}).
\end{solution}

\begin{solution}{exo:b1:findim:4}
Graus $0, 1, 2, 3$ distintos aos pares: \omterm{def:b1:vspaces:free}{livre}
(\cref{prop:b1:vspaces:freecriteria}), quatro vetores em dimensão
$4$: \omterm{def:b1:vspaces:free}{base}. Para $ X^3$: expandir para baixo,
\[
X(X-1)(X-2) = X^3 - 3X^2 + 2X,
\qquad
X(X-1) = X^2 - X,
\]
então $ X^3 - X(X-1)(X-2) = 3X^2 - 2X $; e $3X^2 - 2X = 3(X^2 - X) + X
= 3\,X(X-1) + X $. Por isso
\[
X^3 = X(X-1)(X-2) + 3\,X(X-1) + 1\cdot X + 0\cdot 1 :
\]
\omterm{prop:b1:vspaces:coordinates}{coordenadas} $(0, 1, 3, 1)$ sobre $\bigl(1, X, X(X-1),
X(X-1)(X-2)\bigr)$. (Estes são números de Stirling disfarçados.)
\end{solution}

\begin{solution}{exo:b1:findim:5}
Grassman: $\dim(F \cap G) = 4 + 5 - \dim(F + G)$, e $ F + G $ é um
\omterm{def:b1:vspaces:subspace}{subespaço} de $ E $ contendo $ G $: $5 \leq \dim(F+G) \leq 7$. Por isso
$\dim(F \cap G) \in \{2, 3, 4\}$. Realizações em $\R^7$ com
canônico \omterm{def:b1:vspaces:free}{base} $(e_1, \dots, e_7)$, tirando $ G =
\operatorname{Vect}(e_1, \dots, e_5)$:
\begin{itemize}
  \item $ F = \operatorname{Vect}(e_1, e_2, e_6, e_7)$: $ F + G = \R^7$,
        interseção $\operatorname{Vect}(e_1, e_2)$, dimensão $2$;
  \item $ F = \operatorname{Vect}(e_1, e_2, e_3, e_6)$: interseção
        de dimensão $3$;
  \item $ F = \operatorname{Vect}(e_1, e_2, e_3, e_4) \subseteq G $:
        dimensão $4$.
\end{itemize}
\end{solution}

\begin{solution}{exo:b1:findim:6}
$ H $ é um \omterm{def:b1:vspaces:subspace}{subespaço} (\cref{exo:b1:vspaces:1}~(4) literalmente). Pelo
teorema do fator (\cref{thm:b1:poly:factor}), $ P \in H \iff P =
(X-1)Q $ com $\deg Q \leq n - 1$: o \omterm{def:b1:logic:map}{mapa} $ Q \mapsto (X-1)Q $ é um
bijeção linear de $\R_{n-1}[X]$ para $ H $, então um \omterm{def:b1:vspaces:free}{base} de $ H $ é
\[
\bigl((X-1),\ (X-1)X,\ (X-1)X^2,\ \dots,\ (X-1)X^{n-1}\bigr),
\qquad \dim H = n .
\]
UM \omterm{def:b1:vspaces:sum}{suplementar} linha: $\operatorname{Vect}(1)$ (constantes). De fato
$ H \cap \operatorname{Vect}(1) = \{0\}$ (uma constante diferente de zero não
desaparecer em $1$) e as dimensões somam $ n + 1$: por Grassmann, $ H
\oplus \operatorname{Vect}(1) = \R_n[X]$.
\end{solution}

\begin{solution}{exo:b1:findim:7}

menor \omterm{def:b1:vspaces:span}{período} dá uma família geradora maior com no máximo

(caso contrário, por \cref{prop:b1:vspaces:freecriteria}~(2) um \omterm{def:b1:vspaces:free}{base}
estende). Ambos declarações juntos dão o ``\omterm{def:b1:findim:rank}{classificação} é
$1$-Conclusão de Lipschitz.
\end{solution}

\begin{solution}{exo:b1:findim:8}
Ao longo de uma cadeia estritamente crescente, as dimensões aumentam estritamente
($ F_i \subseteq F_{i+1}$, $ F_i \neq F_{i+1}$ e
\cref{thm:b1:findim:subspaces}: igualdade de dimensões forçaria
igualdade de espaços). Então $\dim F_1 < \dim F_2 < \dots < \dim F_k $ é
uma sequência estritamente crescente de inteiros em $\intint{0}{n}$: em
mais $ n + 1$ valores, $ k \leq n + 1$. Cadeia máxima em $\R^n $:
\[
\{0\} \subsetneq \operatorname{Vect}(e_1) \subsetneq
\operatorname{Vect}(e_1, e_2) \subsetneq \dots \subsetneq \R^n ,
\]
de comprimento exatamente $ n + 1$.
\end{solution}

\begin{solution}{exo:b1:findim:9}
$ F + G $ contém estritamente $ F $ (desde $ G \not\subseteq F $), então
$\dim(F + G) = n $ e Grassmann dá $\dim(F \cap G) = (n-1) +
(n-1) - n = n - 2$.

Afirmação geral, por indução em $ k $: a interseção $ I_k $ de $ k $
hiperplanos tem $\dim I_k \geq n - k $. Verdadeiro para $ k = 1$. Etapa: $ I_{k+1}
= I_k \cap H_{k+1}$, e Grassmann dentro $ E $:
\[
\dim(I_k \cap H_{k+1}) = \dim I_k + (n - 1) - \dim(I_k + H_{k+1})
\geq \dim I_k + (n-1) - n \geq n - k - 1 . \qedhere
\]
\end{solution}

\begin{solution}{exo:b1:findim:10}
\begin{enumerate}
  \item A condição $ u_{n+2} - 3u_{n+1} + 2u_n = 0$ é linear
        e satisfeito pela sequência zero: $ E $ é um \omterm{def:b1:vspaces:subspace}{subespaço}. Por
        indução, $ u_0$ e $ u_1$ determinar cada $ u_n $, e o
        dependência é linear (cada etapa é uma combinação linear de
        os dois valores anteriores); inversamente cada par $(a, b)$
        surge de exatamente uma sequência de $ E $ (\omterm{def:b1:logic:sets}{definir} $ u_n $ por
        a recorrência). Como em \cref{ex:b1:findim:dims}, $ E $ é
        parametrizado bijetivamente e linearmente por $(u_0, u_1) \in
        \R^2$: $\dim E = 2$.
  \item Constantes: $3\cdot1 - 2\cdot1 = 1$. Geométrico: $3\cdot
        2^{n+1} - 2\cdot 2^n = (6 - 2)2^n = 2^{n+2}$. Ambos mentem
        $ E $. Liberdade: $ a\cdot 1 + b\cdot 2^n = 0$ para todos $ n $
        dá, em $ n = 0$ e $ n = 1$: $ a + b = 0$, $ a + 2b = 0$,
        então $ a = b = 0$. Dois \omterm{def:b1:vspaces:free}{livre} vetores em dimensão $2$: um \omterm{def:b1:vspaces:free}{base}
        (\cref{prop:b1:findim:twoofthree}).
  \item Resolver $ a + b = 0$, $ a + 2b = 1$: $ b = 1$, $ a = -1$, então
        $ u_n = 2^n - 1$ (a sequência de Mersenne).
\end{enumerate}
\end{solution}

\begin{solution}{exo:b1:findim:11}
\begin{enumerate}
  \item Grassman: $\dim(F \cap G) = \dim F + \dim G - \dim(F +
        G) \geq \dim F + \dim G - n > 0$, então $ F \cap G \neq
        \{0\}$. Com $ n = 100$: $51 + 50 - 100 = 1 > 0$, o
        interseção contém uma linha.
  \item Se $ F \cap H = \{0\}$, a soma é direta e $\dim F +
        (n - 1) = \dim(F \oplus H) \leq n $, então $\dim F \leq 1$.
        (Por outro lado, uma linha não contida em $ H $ satisfaz
        isto: os hiperplanos não perdem quase nada.)
\end{enumerate}
\end{solution}

\begin{solution}{exo:b1:findim:12}
Indução descendente ativada $ d = \dim F = \dim G $, de $ d = n $ para $ d =
0$. Se $ d = n $: $ F = G = E $ e $ S = \{0\}$ funciona. Suponha que
\omterm{def:b1:logic:statement}{declaração} vale para \omterm{def:b1:vspaces:subspace}{subespaço} pares de dimensão $ d + 1 \leq n $,
e deixe $\dim F = \dim G = d < n $.

Se $ F = G $: levar para $ S $ qualquer \omterm{def:b1:vspaces:sum}{suplementar} de $ F $
(\cref{thm:b1:findim:subspaces}). Se $ F \neq G $: ambos são adequados,
então por \cref{exo:b1:vspaces:12} existe $ x \notin F \cup G $.
As somas $ F' = F \oplus \operatorname{Vect}(x)$ e $ G' = G \oplus
\operatorname{Vect}(x)$ são \omterm{def:b1:vspaces:sum}{diretos} ($ x \notin F $, $ x \notin G $)
e tem dimensão $ d + 1$; pela hipótese de indução eles
admitir um comum \omterm{def:b1:vspaces:sum}{suplementar} $ S'$: $ E = F' \oplus S' = G' \oplus
S'$. \omterm{def:b1:logic:sets}{Definir} $ S = \operatorname{Vect}(x) \oplus S'$ --- \omterm{def:b1:vspaces:sum}{direto},
porque $ S' \cap \operatorname{Vect}(x) \subseteq S' \cap F' =
\{0\}$.

Então $ F + S = F + \operatorname{Vect}(x) + S' = F' + S' = E $, e
$ F \cap S = \{0\}$: se $ f = \lambda x + s'$ com $ f \in F $, $ s'
\in S'$, então $ s' = f - \lambda x \in F' \cap S' = \{0\}$, então $ f
= \lambda x $, forçando $\lambda = 0$ ($ x \notin F $) e $ f = 0$.
Por isso $ E = F \oplus S $ e simetricamente $ E = G \oplus S $.
\end{solution}

\begin{solution}{pb:b1:findim:1}
\textbf{1.} $\Q $ contém $0 \neq 1$, é estável sob adição,
multiplicação e opostos, e todo racional diferente de zero tem um
inverso racional: a \omterm{def:b1:structures:field}{campo} (\cref{def:b1:structures:field}). O
definições de \omterm{def:b1:vspaces:span}{período}, liberdade, \omterm{def:b1:vspaces:free}{base} e as provas dos capítulos
18--19 usam apenas adição vetorial, distributividade e escalar
aritmética em um \omterm{def:b1:structures:field}{campo} --- nunca um valor absoluto, uma ordem ou um
\omterm{def:b1:topology:closure}{limite}. Então $\R $, com sua própria adição e a multiplicação
$\Q \times \R \to \R $, é um $\Q $-espaço vetorial e todos os
teoremas se aplicam. A \omterm{thm:b1:logic:partition}{divisão} por um escalar ocorre uma vez na prova de
\cref{thm:b1:findim:exchange}: para ``resolver para $ g_k $'' um \omterm{def:b1:arith:divides}{divide}
pelo seu coeficiente diferente de zero.

\textbf{2.} Se $ a + b\sqrt2 = 0$ com $ a, b \in \Q $ não ambos
zero: $ b \neq 0$ daria $\sqrt2 = -a/b \in \Q $, contradizendo
a irracionalidade de $\sqrt2$ (\cref{exo:b1:arith:7} com $ p =
2$); então $ b = 0$, então $ a = 0$: \omterm{def:b1:vspaces:free}{livre} sobre $\Q $. Sobre $\R $ o
relação $\sqrt2\cdot 1 + (-1)\cdot\sqrt2 = 0$ não é trivial:
vinculado. Por isso $\dim_\Q \Q(\sqrt2) = 2$, com exclusivo \omterm{prop:b1:vspaces:coordinates}{coordenadas}
$(a, b)$.

\textbf{3.} $(a + b\sqrt2)(c + d\sqrt2) = (ac + 2bd) + (ad +
bc)\sqrt2 \in \Q(\sqrt2)$. Então
\[
\sigma(uv) = (ac + 2bd) - (ad + bc)\sqrt2
= (a - b\sqrt2)(c - d\sqrt2) = \sigma(u)\sigma(v),
\]
então $ N(uv) = uv\,\sigma(uv) = u\sigma(u)\,v\sigma(v) = N(u)N(v)$.
Se $ N(u) = a^2 - 2b^2 = 0$ com $ u \neq 0$: $ b \neq 0$ daria
$(a/b)^2 = 2$, uma raiz quadrada racional de $2$; então $ b = 0$, então $ a
= 0$ e $ u = 0$: contradição. Por isso $ N(u) \neq 0$ para $ u \neq
0$.

\textbf{4.} $ u \cdot \dfrac{\sigma(u)}{N(u)} = \dfrac{N(u)}{N(u)}
= 1$, e $\sigma(u)/N(u) \in \Q(\sqrt2)$: todo elemento diferente de zero
é invertível por dentro $\Q(\sqrt2)$, que é portanto um subcampo
de $\R $. Exemplos: $ N(3 + 2\sqrt2) = 9 - 8 = 1$, então
\[
\frac1{3 + 2\sqrt2} = 3 - 2\sqrt2 ;
\qquad
N(1 + \sqrt2) = -1, \quad
\frac1{1 + \sqrt2} = \sqrt2 - 1 .
\]

\textbf{5.} Se $\sqrt6 = p/q $ então $6q^2 = p^2$; o expoente de
$2$ é $1 + 2v_2(q)$, ímpar, à esquerda, e $2v_2(p)$, mesmo, em
o certo: impossível. Agora suponha $\sqrt3 = a + b\sqrt2$ com
$ a, b \in \Q $. Quadratura: $3 = a^2 + 2b^2 + 2ab\sqrt2$. Se $ ab
\neq 0$, então $\sqrt2 = (3 - a^2 - 2b^2)/(2ab) \in \Q $:
impossível. Se $ b = 0$: $\sqrt3 = a \in \Q $, contradizendo
\cref{exo:b1:arith:7} ($ p = 3$). Se $ a = 0$: $\sqrt3 = b\sqrt2$,
e multiplicando por $\sqrt2$: $\sqrt6 = 2b \in \Q $: impossível.
Então $\sqrt3 \notin \Q(\sqrt2)$.

\textbf{6.} Os produtos de \omterm{def:b1:vspaces:free}{base} vetores são
\[
\sqrt2\,\sqrt3 = \sqrt6,\quad
\sqrt2\,\sqrt6 = 2\sqrt3,\quad
\sqrt3\,\sqrt6 = 3\sqrt2,\quad
(\sqrt2)^2 = 2,\ (\sqrt3)^2 = 3,\ (\sqrt6)^2 = 6,
\]
tudo dentro $ M $. Um produto de dois elementos de $ M $ expande por
bilinearidade em combinações racionais destes: $ M $ é estável
sob multiplicação.

\textbf{7.} Deixar $ x + y\sqrt3 = 0$ com $ x, y \in K = \Q(\sqrt2)$.
Se $ y \neq 0$, então $\sqrt3 = -x/y \in K $ (pergunta 4: $ K $ é um
\omterm{def:b1:structures:field}{campo}), contradizendo a questão 5. Então $ y = 0$, então $ x = 0$: $(1,
\sqrt3)$ é \omterm{def:b1:vspaces:free}{livre} sobre $ K $. Ele gera: $ K + K\sqrt3 =
\{(a + b\sqrt2) + (c + d\sqrt2)\sqrt3\} = \operatorname{Vect}_\Q
(1, \sqrt2, \sqrt3, \sqrt6) = M $. Por isso $\dim_K M = 2$.

\textbf{8.} Uma relação $ a + b\sqrt2 + c\sqrt3 + d\sqrt6 = 0$
reagrupa como $(a + b\sqrt2) + (c + d\sqrt2)\sqrt3 = 0$ com
coeficientes em $ K $; pela pergunta 7 ambos desaparecem, e pela pergunta 2
$ a = b = 0$ e $ c = d = 0$. Então a família é \omterm{def:b1:vspaces:free}{livre} e $\dim_\Q M
= 4$.

\textbf{9.} Todo $ x \in M $ escreve $ x = \sum_j y_j f_j $ com $ y_j
\in K $ (\omterm{def:b1:vspaces:free}{base} de $ M $ sobre $ K $) e cada $ y_j = \sum_i
\lambda_{ij} e_i $ com $\lambda_{ij} \in \Q $ (\omterm{def:b1:vspaces:free}{base} de $ K $ sobre
$\Q $); substituindo, $ x = \sum_{i,j} \lambda_{ij}\, e_i f_j $: o
produtos geram $ M $ sobre $\Q $.

\textbf{10.} Suponha $\sum_{i,j} \lambda_{ij}\, e_i f_j = 0$ com
$\lambda_{ij} \in \Q $. Reagrupar: $\sum_j \bigl(\sum_i \lambda_{ij}
e_i\bigr) f_j = 0$, e as somas internas pertencem a $ K $. Liberdade de
$(f_j)$ sobre $ K $ dá $\sum_i \lambda_{ij} e_i = 0$ para cada
$ j $; liberdade de $(e_i)$ sobre $\Q $ então dá $\lambda_{ij} = 0$
para todos $ i, j $.

\textbf{11.} Pelas questões 9 e 10, $(e_i f_j)_{i \leq m,\, j
\leq n}$ é um \omterm{def:b1:vspaces:free}{base} de $ M $ sobre $\Q $ com $ mn $ elementos:
\[
\dim_\Q M = m\,n = \dim_\Q K \cdot \dim_K M .
\]
Verificar: $\dim_\Q K = 2$ e $\dim_K M = 2$ (questões 2 e 7)
dar $\dim_\Q M = 4$, que é a questão 8.

\textbf{12.} Os vetores $ u a_1, \dots, u a_d $ pertencer a $ A $
(estabilidade). Eles são \omterm{def:b1:vspaces:free}{livre}: se $\sum_i \lambda_i\, u a_i = 0$,
então $ u \sum_i \lambda_i a_i = 0$ em $\R $, e $ u \neq 0$ forças
$\sum_i \lambda_i a_i = 0$, por isso $\lambda_i = 0$ (o $ a_i $ formulário
um \omterm{def:b1:vspaces:free}{base}). Então $(u a_1, \dots, u a_d)$ é um família \omterm{def:b1:vspaces:free}{livre} de $ d $
vetores em $ A $, $\dim_\Q A = d $: um \omterm{def:b1:vspaces:free}{base}
(\cref{prop:b1:findim:twoofthree}). Em particular $1 \in A $
se decompõe como $1 = \sum_i \mu_i\, u a_i = u\,v $ com $ v = \sum_i
\mu_i a_i \in A $: o inverso de $ u $ reside em $ A $. Isso se recupera
questão 4 ($ A = \Q(\sqrt2)$) e prova de uma só vez que $ M $
é um subcampo de $\R $ (com a pergunta 6).

\textbf{13.} O $ d + 1$ vetores $1, x, x^2, \dots, x^{d}$ tudo
deitar $ A $ (estabilidade sob produtos); um família \omterm{def:b1:vspaces:free}{livre} de $ A $ tem
no máximo $ d $ vetores (\cref{prop:b1:findim:twoofthree}), então eles
estão ligados: existem racionais $\lambda_0, \dots, \lambda_d $,
nem todos zero, com $\sum_k \lambda_k x^k = 0$. O polinomial $ P
= \sum_k \lambda_k X^k $ é diferente de zero, tem coeficientes racionais,
grau $\leq d $, e $ P(x) = 0$.

\textbf{14.} $ s^2 = 2 + 2\sqrt6 + 3 = 5 + 2\sqrt6$: \omterm{prop:b1:vspaces:coordinates}{coordenadas}
$(5, 0, 0, 2)$. Então
\[
s^3 = s\,s^2 = (\sqrt2 + \sqrt3)(5 + 2\sqrt6)
= 5\sqrt2 + 2\sqrt{12} + 5\sqrt3 + 2\sqrt{18}
= 11\sqrt2 + 9\sqrt3 ,
\]
\omterm{prop:b1:vspaces:coordinates}{coordenadas} $(0, 11, 9, 0)$ (usando $\sqrt{12} = 2\sqrt3$,
$\sqrt{18} = 3\sqrt2$). Finalmente $ s^4 = (s^2)^2 = 25 + 20\sqrt6 +
24 = 49 + 20\sqrt6$: \omterm{prop:b1:vspaces:coordinates}{coordenadas} $(49, 0, 0, 20)$.

\textbf{15.} Suponha $ a + bs + cs^2 + ds^3 = 0$. Lendo os quatro
\omterm{prop:b1:vspaces:coordinates}{coordenadas} sobre $(1, \sqrt2, \sqrt3, \sqrt6)$:
\[
a + 5c = 0,\qquad b + 11d = 0,\qquad b + 9d = 0,\qquad 2c = 0 .
\]
Então $ c = 0$, então $ a = 0$; subtraindo as equações do meio, $2d =
0$, então $ b = 0$: $(1, s, s^2, s^3)$ é \omterm{def:b1:vspaces:free}{livre}. Quatro \omterm{def:b1:vspaces:free}{livre} vetores
em $ M $ de dimensão $4$: um \omterm{def:b1:vspaces:free}{base}, então
$\operatorname{Vect}_\Q(1, s, s^2, s^3) = M $. Da pergunta 14,
$ s^3 - 9s = 2\sqrt2$ e $11s - s^3 = 2\sqrt3$:
\[
\sqrt2 = \frac{s^3 - 9s}{2},
\qquad
\sqrt3 = \frac{11s - s^3}{2} .
\]

\textbf{16.} $ s^4 - 10s^2 + 1 = (49 + 20\sqrt6) - 10(5 +
2\sqrt6) + 1 = 0$. UM \omterm{def:b1:poly:def}{polinômio} mônico de grau $\leq 3$
desaparecendo em $ s $ produziria uma combinação nula não trivial de
$(1, s, s^2, s^3)$, contradizendo a questão 15: $ X^4 - 10X^2 + 1$
tem menor grau. Suas raízes: $ X^2 = 5 \pm 2\sqrt6 = (\sqrt3 \pm
\sqrt2)^2$, então as quatro raízes reais são $\pm(\sqrt3 + \sqrt2)$ e
$\pm(\sqrt3 - \sqrt2)$, ou seja\ $\pm\sqrt2 \pm \sqrt3$.

\textbf{17.} De $ s^4 - 10s^2 + 1 = 0$: $ s\,(10s - s^3) = 1$, então
\[
\frac1s = 10s - s^3 = 10(\sqrt2 + \sqrt3) - (11\sqrt2 + 9\sqrt3)
= \sqrt3 - \sqrt2 ,
\]
consistente com $(\sqrt3 + \sqrt2)(\sqrt3 - \sqrt2) = 1$. Se $ a +
b\sqrt2 + c\sqrt3 + d\sqrt6 = r \in \Q $, então $(a - r) + b\sqrt2
+ c\sqrt3 + d\sqrt6 = 0$ e forças de liberdade (pergunta 8) $ b = c
= d = 0$ (e $ a = r $). Para $\sqrt2 + \sqrt3 + \sqrt6$ o
\omterm{prop:b1:vspaces:coordinates}{coordenadas} $(0, 1, 1, 1)$ não são desta forma: irracionais ---
um \omterm{def:b1:logic:statement}{declaração} mais forte que \cref{exo:b1:reals:7}, obtido sem
quaisquer truques de quadratura.

\textbf{18.} Uma raiz racional $ p/q $ (termos mais baixos) de $ X^3 - 2$
satisfaz $ p^3 = 2q^3$, e o critério de
\cref{exo:b1:poly:5} dá $ p \mid 2$, $ q \mid 1$: candidatos
$\pm1, \pm2$, cujos cubos são $\pm1, \pm8 \neq 2$. Não racional
raiz; em particular $ t = 2^{1/3} \notin \Q $, então $(1, t)$ é \omterm{def:b1:vspaces:free}{livre}
sobre $\Q $ (como na questão 2).

\textbf{19.} Suponha $(1, t, t^2)$ vinculado: algum diferente de zero $ P \in
\Q[X]$ com $\deg P \leq 2$ tem $ P(t) = 0$; escolha tal $ P $ de
\emph{ao menos} grau $ d \geq 1$. Pela pergunta 18, $ d \neq 1$, então $ d
= 2$. \omterm{thm:b1:logic:partition}{Divisão} euclidiana (\cref{thm:b1:poly:division}): $ X^3 - 2
= PQ + R $ com $ Q, R \in \Q[X]$, $\deg R < 2$. Avaliando em $ t $:
$0 = P(t)Q(t) + R(t) = R(t)$, então $ R $ desaparece em $ t $; minimalidade
de $ d $ forças $ R = 0$. Então $ X^3 - 2 = PQ $ com $\deg Q = 1$: o
raiz racional de $ Q $ é uma raiz racional de $ X^3 - 2$,
contradizendo a questão 18. Portanto $(1, t, t^2)$ é \omterm{def:b1:vspaces:free}{livre} e
$\dim_\Q A = 3$. Estabilidade: $ t^3 = 2$ reduz todo produto de
$1, t, t^2$ para uma combinação deles ($ t\cdot t^2 = 2$, $ t^2\cdot
t^2 = 2t $); $ A $ contém $1$: pela pergunta 12, $ A $ é um subcampo
de $\R $.

\textbf{20.} Suponha $ t \in M $. Então $ t^2 \in M $ (estabilidade), então
$ A \subseteq M $, e $ M $ é um espaço vetorial sobre o \omterm{def:b1:structures:field}{campo} $ A $:
os axiomas são os de $\R $-aritmética, restrita. Isso é
\omterm{def:b1:findim:def}{dimensão finita} sobre $ A $ (um finito $\Q $-gerando família
gera a fortiori sobre $ A \supseteq \Q $). O \omterm{pb:b1:findim:1}{lei da torre} para
$\Q \subseteq A \subseteq M $ dá
\[
4 = \dim_\Q M = \dim_\Q A \cdot \dim_A M = 3\,\dim_A M ,
\]
impossível: $3$ não \omterm{def:b1:arith:divides}{divide} $4$. Então $2^{1/3} \notin M $: não
combinação racional de $1, \sqrt2, \sqrt3, \sqrt6$ é igual
$2^{1/3}$.

\textbf{21.} $\Q(\sqrt2) \subseteq M $, então $ t \notin \Q(\sqrt2)$.
Se $ t + \sqrt2 = r \in \Q $, então $ t = r - \sqrt2 \in \Q(\sqrt2)$:
contradição --- $2^{1/3} + \sqrt2$ é irracional. Se $ t = a +
b\sqrt3$, então $ t \in \operatorname{Vect}_\Q(1, \sqrt3) \subseteq
M $: contradição novamente.

\textbf{22.} $ B $ é um espaço vetorial sobre o \omterm{def:b1:structures:field}{campo} $ A $
(restrição de escalares), \omterm{def:b1:findim:def}{dimensão finita} desde $\dim_\Q B $ é
finito. O \omterm{pb:b1:findim:1}{lei da torre} para $\Q \subseteq A \subseteq B $ dá
$\dim_\Q B = \dim_\Q A \cdot \dim_A B $: o fator esquerdo \omterm{def:b1:arith:divides}{divide}.
Subcampos de $ M $ portanto, tem $\Q $-dimensão $1$, $2$ ou $4$:
dimensão $1$ é $\Q $ em si, dimensão $4$ é $ M $.

\textbf{23.} Deixar $ F \subseteq M $ ser um subcampo com $\dim_\Q F =
2$, e $ w \in F \setminus \Q $: $(1, w)$ é \omterm{def:b1:vspaces:free}{livre}, daí um \omterm{def:b1:vspaces:free}{base}
de $ F $. Pela pergunta 13 ($ d = 2$), $ w^2 = \alpha + \beta w $ para
racionais $\alpha, \beta $. Contexto $ v = w - \beta/2 \in F
\setminus \Q $:
\[
v^2 = w^2 - \beta w + \frac{\beta^2}4 = \alpha + \frac{\beta^2}4
\;\in\; \Q,
\]
e $ F = \operatorname{Vect}(1, v)$. Agora escreva $ v = a + b\sqrt2 +
c\sqrt3 + d\sqrt6$ e expandir:
\[
v^2 = (a^2 + 2b^2 + 3c^2 + 6d^2)
+ (2ab + 6cd)\sqrt2 + (2ac + 4bd)\sqrt3 + (2ad + 2bc)\sqrt6 .
\]
Os três irracionais \omterm{prop:b1:vspaces:coordinates}{coordenadas} desaparecer: $ ab + 3cd = ac + 2bd =
ad + bc = 0$. Se $ a \neq 0$: $ b = -3cd/a $, e o segundo
condição se torna $ c(a^2 - 6d^2) = 0$; desde $ a^2 = 6d^2$ com $ d
\neq 0$ faria $\sqrt6 = \abs{a/d}$ racional, quer $ c = 0$
ou $ d = 0$, e em ambos os casos as condições restantes forçam $ b =
c = d = 0$, ou seja\ $ v \in \Q $: excluído. Então $ a = 0$, e o
condições lidas $3cd = 2bd = bc = 0$: no máximo um dos $ b, c, d $ é
diferente de zero. Por isso $ v $ é um múltiplo racional de $\sqrt2$, $\sqrt3$
ou $\sqrt6$, e
\[
F \in \bigl\{\Q(\sqrt2),\ \Q(\sqrt3),\ \Q(\sqrt6)\bigr\} ,
\]
cada um dos quais é de fato um $2$ subcampo -dimensional (estabilidade como
na questão 6, inverso da questão 12): exatamente três quadráticos
subcampos.

\textbf{24.} $(1 + \sqrt2 + \sqrt3)(1 + \sqrt2 - \sqrt3) = (1 +
\sqrt2)^2 - 3 = 2\sqrt2$, então
\[
\frac1{1 + \sqrt2 + \sqrt3}
= \frac{1 + \sqrt2 - \sqrt3}{2\sqrt2}
= \frac{\sqrt2 + 2 - \sqrt6}{4}
= \frac12 + \frac14\sqrt2 + 0\cdot\sqrt3 - \frac14\sqrt6 .
\]
(Verificar: $(1 + \sqrt2 + \sqrt3)(2 + \sqrt2 - \sqrt6) = 4$ depois
expansão.)

\textbf{25.} (eu) $\Q $-dimension atribui um \emph{inteiro} para
cada subcampo de $\R $, e o \omterm{pb:b1:findim:1}{lei da torre} faz esses números inteiros
multiplique ao longo das inclusões: a dimensão se comporta como uma aritmética
invariante, e \omterm{def:b1:arith:divides}{divisibilidade} restrições tornam-se impossibilidade
provas. (ii) \omterm{def:b1:findim:def}{Dimensão finita} força cada elemento a satisfazer uma
racional diferente de zero polinomial equação de grau no máximo o
dimensão: finitude significa algebricidade. (iii) O
regra dois de três virou ``multiplicação por $ u $ \omterm{def:b1:logic:map}{mapas} um \omterm{def:b1:vspaces:free}{base}
para um família \omterm{def:b1:vspaces:free}{livre} de tamanho máximo'' em sobrejetividade, produzindo
$ u^{-1}$ sem fórmula: os inversos vieram da contagem. (iv) O
\omterm{pb:b1:findim:1}{lei da torre} proíbe um $3$-dimensional \omterm{def:b1:structures:field}{campo} dentro de um
$4$-dimensional, e é por isso $2^{1/3}$ não pode ser alcançado
de $\sqrt2$ e $\sqrt3$. O teorema da Parte III é
\emph{Dedekind \omterm{pb:b1:findim:1}{lei da torre}}, o movimento de abertura da teoria de Galois,
desenvolvido no volume do Ano ~ 3.
\end{solution}
