\section*{Capítulo \ref{ch:b1:linmaps} --- Mapas Lineares}

\begin{solution}{exo:b1:linmaps:1}
(1) \omterm{def:b1:linmaps:def}{Linear}: \omterm{prop:b1:vspaces:coordinates}{coordenadas} são expressões \omterm{def:b1:linmaps:def}{lineares}. (2) Não \omterm{def:b1:linmaps:def}{linear}:
$u(2(1,1)) = 4 \neq 2 = 2u(1,1)$. (3) \omterm{def:b1:linmaps:def}{Linear}: diferenciação e
multiplicação por $ X $ são, e somas de \omterm{def:b1:logic:map}{mapas} \omterm{def:b1:linmaps:def}{lineares} são. (4) \omterm{def:b1:linmaps:def}{Linear}:
a avaliação respeita operações pontuais.
\end{solution}

\begin{solution}{exo:b1:linmaps:2}
\omterm{def:b1:linmaps:kerim}{Núcleo}: resolver $ x + y - z = 0$, $2x + y + z = 0$, $3x + 2y = 0$.
A partir do terceiro, $ y = -\frac{3x}{2}$; o primeiro dá $ z = x + y =
-\frac x2$; verifique no segundo: $2x - \frac{3x}{2} - \frac x2 = 0$:
satisfeito. Então $\ker u = \operatorname{Vect}\bigl((2, -3, -1)\bigr)$
(tirando $ x = 2$), dimensão $1$.

\omterm{def:b1:linmaps:rank}{Classificação} - nulidade: $\operatorname{rk} u = 3 - 1 = 2$. \omterm{def:b1:linmaps:kerim}{Imagem}: abrangido por
o imagens do canônico \omterm{def:b1:vspaces:free}{base}, $ u(e_1) = (1,2,3)$, $ u(e_2) =
(1,1,2)$, $ u(e_3) = (-1,1,0)$; os dois primeiros são \omterm{def:b1:vspaces:free}{livre}, e o \omterm{def:b1:linmaps:rank}{classificação}
é $2$: \omterm{def:b1:vspaces:free}{base} $\bigl((1,2,3), (1,1,2)\bigr)$.

Não \omterm{def:b1:logic:inj}{injetivo} ($\ker \neq \{0\}$), não \omterm{def:b1:logic:inj}{sobrejetivo} (\omterm{def:b1:linmaps:rank}{classificação} $2 < 3$):
consistente com \cref{cor:b1:linmaps:samedim}.
\end{solution}

\begin{solution}{exo:b1:linmaps:3}
\emph{\omterm{def:b1:linmaps:kerim}{Núcleo}:} $ P = P'$ forças $\deg P = \deg P'$ a menos que $ P = 0$;
mas $\deg P' < \deg P $ para $ P \neq 0$: então $\ker u = \{0\}$, e $ u $,
um \omterm{def:b1:logic:inj}{injetivo} endomorfismo do \omterm{def:b1:findim:def}{dimensão finita} $\R_n[X]$, é um
isomorfismo (\cref{cor:b1:linmaps:samedim}).

\emph{Inverso:} deixar $ v(P) = P + P' + P'' + \dots + P^{(n)}$ (um
soma finita em $\R_n[X]$). Então
\[
v\bigl(u(P)\bigr) = \sum_{k=0}^{n} (P - P')^{(k)}
= \sum_{k=0}^{n} P^{(k)} - \sum_{k=0}^{n} P^{(k+1)}
= P - P^{(n+1)} = P ,
\]
telescópica, uma vez que $ P^{(n+1)} = 0$. Então $ v = u^{-1}$.
\end{solution}

\begin{solution}{exo:b1:linmaps:4}
$\operatorname{im}(v \circ u) = v(\operatorname{im} u) \subseteq
\operatorname{im} v $: \omterm{def:b1:linmaps:rank}{classificação} $\leq \operatorname{rk} v $. E $ v $
restrito a $\operatorname{im} u $ tem \omterm{def:b1:linmaps:kerim}{imagem}
$\operatorname{im}(vu)$ com \omterm{def:b1:linmaps:rank}{classificação} - nulidade dentro
$\operatorname{im} u $: $\operatorname{rk}(vu) \leq
\dim\operatorname{im} u = \operatorname{rk} u $.
\end{solution}

\begin{solution}{exo:b1:linmaps:5}
$ u^2 = 0$ significa $ u(u(x)) = 0$ para todos $ x $: todo $ u(x)$ reside em
$\ker u $, ou seja\ $\operatorname{im} u \subseteq \ker u $. Então
\omterm{def:b1:linmaps:rank}{classificação} - nulidade:
\[
\dim E = \dim\ker u + \operatorname{rk} u \geq 2\operatorname{rk} u .
\]
Exemplo de igualdade em $\R^2$: $ u(x, y) = (y, 0)$: $ u^2 = 0$,
$\operatorname{rk} u = 1 = \frac{\dim E}{2}$.
\end{solution}

\begin{solution}{exo:b1:linmaps:6}
$(pq)^2 = pqpq = ppqq = pq $ (comutação): uma projeção
(\cref{thm:b1:linmaps:projchar}).

\omterm{def:b1:linmaps:kerim}{Imagem}: $\operatorname{im}(pq) \subseteq \operatorname{im} p $, e
$= \operatorname{im}(qp) \subseteq \operatorname{im} q $: contido
no cruzamento. Por outro lado, se $ x \in \operatorname{im} p \cap
\operatorname{im} q $, então $ p(x) = x $ e $ q(x) = x $ (pontos fixos
caracterizar o \omterm{def:b1:linmaps:kerim}{imagem} de um projeção), então $ pq(x) = x $: $ x \in
\operatorname{im}(pq)$.

\omterm{def:b1:linmaps:kerim}{Núcleo}: $\ker p \subseteq \ker(qp) = \ker(pq)$ e da mesma forma $\ker q
\subseteq \ker(pq)$: a soma está contida. Por outro lado, deixe $ pq(x) =
0$ e escreva
\[
x = \underbrace{q(x)}_{\in\, \ker p} +
\underbrace{(x - q(x))}_{\in\, \ker q} :
\]
o primeiro termo satisfaz $ p(q(x)) = 0$, então está em $\ker p $; o
o segundo está em $\ker q $ desde $ q(x - q(x)) = q(x) - q^2(x) = 0$. Por isso
$ x \in \ker p + \ker q $.
\end{solution}

\begin{solution}{exo:b1:linmaps:7}
Observe primeiro as inclusões gerais $\ker u \subseteq \ker u^2$ e
$\operatorname{im} u^2 \subseteq \operatorname{im} u $, e por
\omterm{def:b1:linmaps:rank}{classificação} - nulidade, (2) $\iff $ (3) (igual grãos $\iff $ igual classificações
$\iff $ igual imagens, dadas as inclusões).

(1 $\Rightarrow $ 2): deixe $ u^2(x) = 0$; então $ u(x) \in \ker u \cap
\operatorname{im} u = \{0\}$, então $ x \in \ker u $.

(2 $\Rightarrow $ 1): por Grassmann e \omterm{def:b1:linmaps:rank}{classificação} - nulidade, $\dim(\ker u +
\operatorname{im} u) = \dim\ker u + \operatorname{rk} u -
\dim(\ker u \cap \operatorname{im} u) = \dim E - \dim(\ker u \cap
\operatorname{im} u)$: a soma é $ E $ se a interseção for
$\{0\}$. Deixar $ y \in \ker u \cap \operatorname{im} u $: $ y = u(x)$ e
$ u(y) = 0$, então $ u^2(x) = 0$, então (por (2)) $ u(x) = 0$: $ y = 0$. Por isso
$ E = \ker u \oplus \operatorname{im} u $.
\end{solution}

\begin{solution}{exo:b1:linmaps:8}
Se $\varphi = 0$: então $\ker\varphi = E \subseteq \ker\psi $ forças
$\psi = 0 = 0\cdot\varphi $. Se $\varphi \neq 0$: $\ker\varphi $ é um
\omterm{def:b1:linmaps:forms}{hiperplano}; escolha $ a \notin \ker\varphi $. \omterm{def:b1:logic:sets}{Definir} $\lambda =
\frac{\psi(a)}{\varphi(a)}$. O formulário $\psi - \lambda\varphi $
desaparece em $\ker\varphi $ (ambos fazem, pela inclusão) e em $ a $:
desaparece em $\ker\varphi \oplus Ka = E $. Então $\psi =
\lambda\varphi $.
\end{solution}

\begin{solution}{exo:b1:linmaps:9}
Suponha $\lambda_0 x + \lambda_1 u(x) + \dots + \lambda_{n-1}
u^{n-1}(x) = 0$. Aplicar $ u^{n-1}$: todos os termos com um fator $ u^{\geq
n}$ morrer, deixando $\lambda_0 u^{n-1}(x) = 0$, então $\lambda_0 = 0$.
Aplicar $ u^{n-2}$ para a relação restante: $\lambda_1 u^{n-1}(x) =
0$, então $\lambda_1 = 0$; e assim por diante. A família é \omterm{def:b1:vspaces:free}{livre}; sendo de tamanho
$ n = \dim E $, é um \omterm{def:b1:vspaces:free}{base} (\cref{prop:b1:findim:twoofthree}). (Em
isso \omterm{def:b1:vspaces:free}{base}, $ u $ atua como uma mudança --- o modelo de máximo
nulpotência.)
\end{solution}

\begin{solution}{exo:b1:linmaps:10}
\begin{enumerate}
  \item $ f \circ f = -\mathrm{id}$ é \omterm{def:b1:logic:inj}{bijetivo}, então $ f $ é
        (\cref{prop:b1:logic:comp} adaptado: $ f $ tem os dois lados
        inverso $-f $). Se $ f(x) = \lambda x $ com $ x \neq 0$:
        aplicando $ f $, $-x = \lambda^2 x $, então $\lambda^2 = -1$:
        impossível em $\R $.
  \item Construa a família com avidez. Pegar $ x_1 \neq 0$: $(x_1,
        f(x_1))$ é \omterm{def:b1:vspaces:free}{livre} por (1). Se $\operatorname{Vect}$ do
        família atual $\bigl(x_1, f(x_1), \dots, x_k,
        f(x_k)\bigr)$, chame-o $ V_k $ --- um \omterm{def:b1:vspaces:subspace}{subespaço} estável sob $ f $
        (cada gerador \omterm{def:b1:logic:map}{mapas} para outro gerador ou seu negativo:
        $ f(f(x_i)) = -x_i $) --- não é tudo $ E $, escolher $ x_{k+1}
        \notin V_k $. \emph{Alegação: a família alargada é \omterm{def:b1:vspaces:free}{livre}.}
        Suponha $\alpha x_{k+1} + \beta f(x_{k+1}) + v = 0$ com $ v
        \in V_k $ e $(\alpha, \beta) \neq (0,0)$. Aplicar $ f $:
        $\alpha f(x_{k+1}) - \beta x_{k+1} + f(v) = 0$ com $ f(v)
        \in V_k $. Eliminar $ f(x_{k+1})$ entre as duas relações
        (multiplique o primeiro por $\alpha $, o segundo por $-\beta $ e
        adicionar):
        \[
        (\alpha^2 + \beta^2)\, x_{k+1} \in V_k ,
        \]
        e $\alpha^2 + \beta^2 \neq 0$ forças $ x_{k+1} \in V_k $:
        contradição. Então a construção continua, adicionando vetores
        \emph{dois de cada vez}, até $ V_k = E $: a família final é uma
        \omterm{def:b1:vspaces:free}{base} de tamanho uniforme e $ n $ é par.
\end{enumerate}
\end{solution}

\begin{solution}{exo:b1:linmaps:11}
\omterm{def:b1:topology:closure}{Limite} superior: para cada $ x $, $(u + v)(x) = u(x) + v(x) \in
\operatorname{im} u + \operatorname{im} v $, então
\[
\operatorname{rk}(u + v)
\leq \dim(\operatorname{im} u + \operatorname{im} v)
\leq \operatorname{rk} u + \operatorname{rk} v
\]
(Grasmann, \cref{thm:b1:findim:grassmann}). \omterm{def:b1:topology:closure}{Limite} inferior: aplicar
o \omterm{def:b1:topology:closure}{limite} superior para o par $(u + v, -v)$, cuja soma é $ u $:
\[
\operatorname{rk} u \leq \operatorname{rk}(u + v) +
\operatorname{rk}(-v) = \operatorname{rk}(u + v) +
\operatorname{rk} v,
\]
então $\operatorname{rk} u - \operatorname{rk} v \leq
\operatorname{rk}(u+v)$; trocando $ u $ e $ v $ dá o absoluto
valor.
\end{solution}

\begin{solution}{exo:b1:linmaps:12}
\emph{Fórmula exata.} Deixar $ v'$ ser a restrição de $ v $ para o
\omterm{def:b1:vspaces:subspace}{subespaço} $\operatorname{im} w $. Isso é \omterm{def:b1:linmaps:kerim}{imagem} é $ v(w(F)) =
\operatorname{im}(v \circ w)$, e seu \omterm{def:b1:linmaps:kerim}{núcleo} é $\ker v \cap
\operatorname{im} w $. \omterm{def:b1:linmaps:rank}{Classificação} - nulidade para $ v'$ no espaço
$\operatorname{im} w $:
\[
\operatorname{rk} w = \dim\operatorname{im} w
= \operatorname{rk}(v \circ w) + \dim(\ker v \cap
\operatorname{im} w) .
\]

\emph{Frobenius.} Aplique a fórmula exata duas vezes, para $ w $ e para
$ w \circ u $:
\[
\operatorname{rk} w - \operatorname{rk}(vw)
= \dim\bigl(\ker v \cap \operatorname{im} w\bigr),
\qquad
\operatorname{rk}(wu) - \operatorname{rk}(vwu)
= \dim\bigl(\ker v \cap \operatorname{im}(wu)\bigr) .
\]
Desde $\operatorname{im}(w \circ u) \subseteq \operatorname{im}
w $, a segunda intersecção está contida na primeira, e seu
dimensão não é maior:
\[
\operatorname{rk}(wu) - \operatorname{rk}(vwu)
\;\leq\; \operatorname{rk} w - \operatorname{rk}(vw) ,
\]
que se reorganiza na desigualdade de Frobenius. Com $ w =
\mathrm{id}_F $ (\omterm{def:b1:linmaps:rank}{classificação} $\dim F $, e $\operatorname{im}\,
\mathrm{id}_F = F $): $\operatorname{rk} v + \operatorname{rk} u
\leq \dim F + \operatorname{rk}(vu)$, Desigualdade de Sylvester ---
provado novamente, em forma de matriz, em \cref{exo:b1:matrices:10}.
\end{solution}

\begin{solution}{pb:b1:linmaps:1}
\textbf{1.} $(\mathrm{id} - p)^2 = \mathrm{id} - 2p + p^2 =
\mathrm{id} - p $: um projetor. Se $ y = x - p(x)$, então $ p(y) =
p(x) - p^2(x) = 0$, e inversamente $ x \in \ker p $ dá $ x =
(\mathrm{id} - p)(x)$: $\operatorname{im}(\mathrm{id} - p) = \ker
p $. E $(\mathrm{id} - p)(x) = 0 \iff p(x) = x \iff x \in
\operatorname{im} p $ (pontos fixos,
\cref{thm:b1:linmaps:projchar}): $\ker(\mathrm{id} - p) =
\operatorname{im} p $.

\textbf{2.} $(\lambda\,\mathrm{id} + \mu p)^2 =
\lambda^2\,\mathrm{id} + (2\lambda\mu + \mu^2)\,p $. O par
$(\mathrm{id}, p)$ é \omterm{def:b1:vspaces:free}{livre} em $\mathcal{L}(E)$: $ p = c\,
\mathrm{id}$ daria $ c^2 = c $, então $ p = 0$ ou $\mathrm{id}$,
excluído. Identificando coeficientes, o \omterm{def:b1:logic:map}{mapa} é um projetor se
$\lambda^2 = \lambda $ e $2\lambda\mu + \mu^2 = \mu $. Para
$\lambda = 0$: $\mu \in \{0, 1\}$. Para $\lambda = 1$: $\mu^2 +
\mu = 0$, $\mu \in \{0, -1\}$. Exatamente quatro projetores no
avião: $0$, $ p $, $\mathrm{id}$, $\mathrm{id} - p $.

\textbf{3.} $(\lambda\,\mathrm{id} + \mu p)(\lambda'\,\mathrm{id}
+ \mu' p) = \lambda\lambda'\,\mathrm{id} + (\lambda\mu' +
\mu\lambda' + \mu\mu')\,p $: o avião está estável sob
composição. Desde $ p^k = p $ para cada $ k \geq 1$, para $ Q =
\sum_k a_k X^k $:
\[
Q(p) = a_0\,\mathrm{id} + \Bigl(\sum_{k \geq 1} a_k\Bigr) p
= Q(0)\,\mathrm{id} + \bigl(Q(1) - Q(0)\bigr)\,p .
\]

\textbf{4.} Sobre $\operatorname{im} p $ (onde $ p $ atua como o
identidade), $\lambda\,\mathrm{id} + \mu p $ multiplica por $\lambda
+ \mu $; sobre $\ker p $, por $\lambda $. Como $ E = \operatorname{im} p
\oplus \ker p $, o \omterm{def:b1:logic:map}{mapa} é \omterm{def:b1:logic:inj}{bijetivo} se $\lambda \neq 0$ e
$\lambda + \mu \neq 0$. Resolvendo $\lambda\alpha = 1$,
$\lambda\beta + \mu\alpha + \mu\beta = 0$ na regra de composição
da questão 3:
\[
(\lambda\,\mathrm{id} + \mu p)^{-1}
= \frac1\lambda\,\mathrm{id} -
\frac{\mu}{\lambda(\lambda + \mu)}\,p ,
\]
cuja ação é por $1/\lambda $ sobre $\ker p $ e $1/(\lambda +
\mu)$ sobre $\operatorname{im} p $, como deve ser.

\textbf{5.} Escrever $ F = \operatorname{im} p = \operatorname{im}
p'$. Para cada $ x $, $ p'(x) \in F $ e $ p $ correções $ F $ pontualmente:
$ p(p'(x)) = p'(x)$, ou seja\ $ p\,p' = p'$; simetricamente $ p'\,p =
p $. Quando dois projeções compartilhe seus \omterm{def:b1:linmaps:kerim}{imagem}, aquele aplicado
\emph{primeiro} decide: sua saída já está em $ F $, onde o
exterior projeção atua como a identidade e não muda nada.

\textbf{6.} $(p + q)^2 = p^2 + pq + qp + q^2 = (p + q) + pq +
qp $, então $ p + q $ forças do projetor $ pq + qp = 0$. Componha no
saiu com $ p $: $ pq + pqp = 0$; à direita com $ p $: $ pqp + qp
= 0$. Subtraindo, $ pq = qp $; então $ pq + qp = 2pq = 0$ e o
característica não é $2$: $ pq = qp = 0$.

\textbf{7.} Com $ pq = qp = 0$, a mesma expansão dá $(p +
q)^2 = p + q $. \omterm{def:b1:linmaps:kerim}{Imagem}: $\operatorname{im}(p + q) \subseteq
\operatorname{im} p + \operatorname{im} q $ sempre. Por outro lado,
para $ x \in \operatorname{im} p $: $ q(x) = q(p(x)) = 0$, então $(p +
q)(x) = p(x) = x $ e $ x \in \operatorname{im}(p+q)$; o mesmo para
$\operatorname{im} q $. Direção: $ x \in \operatorname{im} p
\cap \operatorname{im} q $ dá $ x = p(x) = p(q(x)) = 0$.
\omterm{def:b1:linmaps:kerim}{Núcleo}: se $ p(x) + q(x) = 0$, aplicando $ p $ dá $ p(x) +
p(q(x)) = p(x) = 0$ e aplicando $ q $ dá $ q(x) = 0$: $\ker(p
+ q) = \ker p \cap \ker q $ (a inclusão reversa é clara).

\textbf{8.} $ p - q $ é um projetor se $\mathrm{id} - (p - q) =
(\mathrm{id} - p) + q $ é um (pergunta 1 duas vezes). Por perguntas
6--7 aplicado aos projetores $\mathrm{id} - p $ e $ q $, isso
mantém se $(\mathrm{id} - p)q = q(\mathrm{id} - p) = 0$, ou seja\
se $ pq = q $ e $ qp = q $.

\textbf{9.} $ pq = q $ significa $ p $ corrige cada $ q(x)$, ou seja\
$\operatorname{im} q \subseteq \ker(p - \mathrm{id}) =
\operatorname{im} p $. E $ qp = q $ significa $ q\bigl((\mathrm{id} -
p)(x)\bigr) = 0$ para todos $ x $, ou seja\ $ q $ desaparece em
$\operatorname{im}(\mathrm{id} - p) = \ker p $: $\ker p \subseteq
\ker q $. Ambas as etapas são equivalências: a ordem $ q \leq p $ diz
$ q $ projetos em um menor \omterm{def:b1:linmaps:kerim}{imagem}, ao longo de um maior \omterm{def:b1:linmaps:kerim}{núcleo}.

\textbf{10.} Expandindo, $(\mathrm{id} - p)(\mathrm{id} - q) =
\mathrm{id} - p - q + pq = \mathrm{id} - r $. Os projetores
$\mathrm{id} - p $ e $\mathrm{id} - q $ comutar, então por
\cref{exo:b1:linmaps:6} seu produto $\mathrm{id} - r $ é o
projetor em $\operatorname{im}(\mathrm{id} - p) \cap
\operatorname{im}(\mathrm{id} - q) = \ker p \cap \ker q $ junto
$\ker(\mathrm{id} - p) + \ker(\mathrm{id} - q) =
\operatorname{im} p + \operatorname{im} q $. Pela pergunta 1, $ r =
\mathrm{id} - (\mathrm{id} - r)$ é então o projetor com
$\operatorname{im} r = \operatorname{im} p + \operatorname{im}
q $ e $\ker r = \ker p \cap \ker q $.

\textbf{11.} $ p_i $ está bem definido (singularidade do
decomposição) e \omterm{def:b1:linmaps:def}{linear} (a decomposição de $ x + \lambda y $
é a soma das decomposições, novamente por unicidade). Para $ x_i
\in F_i $ a decomposição é $ x_i $ em si, então $ p_i(x_i) = x_i $:
$ p_i^2 = p_i $, e $ p_j(x_i) = 0$ para $ j \neq i $: $ p_i p_j = 0$
($ p_j(x) \in F_j $). Somando os componentes, $\sum_i p_i =
\mathrm{id}$. Finalmente $\operatorname{im} p_i = F_i $ e $\ker
p_i = \bigoplus_{j \neq i} F_j $.

\textbf{12.} $ p_i = p_i \circ \mathrm{id} = p_i\sum_j p_j =
p_i^2 + \sum_{j \neq i} p_i p_j = p_i^2$: cada $ p_i $ é um
projetor. Todo $ x = \mathrm{id}(x) = \sum_i p_i(x)$ reside em
$\sum_i \operatorname{im} p_i $: o imagens soma para $ E $.
Direção: suponha $ y_1 + \dots + y_k = 0$ com $ y_i \in
\operatorname{im} p_i $, então $ p_i(y_i) = y_i $. Aplicar $ p_j $: $ p_j
(y_i) = p_j p_i (y_i) = 0$ para $ i \neq j $, então $0 = p_j\bigl(\sum
y_i\bigr) = y_j $, para cada $ j $. Por isso $ E = \bigoplus_i
\operatorname{im} p_i $.

\textbf{13.} $ q = \mathrm{id} - p $, e a pergunta 1 dá $ pq =
p - p^2 = 0 = qp $ diretamente: para dois projetores, somando o
a identidade já força a ortogonalidade do par.

\textbf{14.} $ p + q = \mathrm{id} - r $ com $ r $ um projetor e
$(\mathrm{id} - r)$ é um projetor (pergunta 1): então $ p + q $ é um
projetor, e a pergunta 6 dá $ pq = qp = 0$. Por simetria ($ q +
r = \mathrm{id} - p $ e $ p + r = \mathrm{id} - q $), todos
os produtos aos pares desaparecem e a questão 12 conclui: $ E =
\operatorname{im} p \oplus \operatorname{im} q \oplus
\operatorname{im} r $, automaticamente.

\textbf{15.} \emph{Lema.} Por indução com Grassmann
(\cref{thm:b1:findim:grassmann}):
\[
\dim(F_1 + \dots + F_k) \leq \dim(F_1 + \dots + F_{k-1}) + \dim
F_k \leq \dots \leq \sum_i \dim F_i .
\]
Se o total for uma igualdade, cada passo é: $(F_1 + \dots +
F_{j-1}) \cap F_j = \{0\}$ para cada $ j $, e uma relação $ y_1 +
\dots + y_k = 0$ ($ y_i \in F_i $) entra em colapso da direita: $ y_k
\in (F_1 + \dots + F_{k-1}) \cap F_k = \{0\}$, então $ y_{k-1} =
0$, etc.: a soma é direta. Por outro lado, um soma direta tem
dimensões aditivas (\omterm{def:b1:vspaces:free}{bases} concatenadas). \emph{Aplicativo:}
$ x = \sum_i p_i(x)$ mostra $ E = \sum_i \operatorname{im} p_i $,
então $ n \leq \sum_i \operatorname{rk} p_i $; a hipótese dá
igualdade e, portanto, franqueza. Produtos: consertar $ j $ e $ y \in
\operatorname{im} p_j $. Então $ y = \sum_i p_i(y)$ com $ p_i(y)
\in \operatorname{im} p_i $, enquanto $ y = y $ também é um
decomposição (componente $ j $ sozinho); forças de singularidade $ p_i(y)
= 0$ para $ i \neq j $. Aplicado a $ y = p_j(x)$: $ p_i p_j = 0$.

\textbf{16.} Se $ u^k(x) = 0$ então $ u^{k+1}(x) = u(0) = 0$. E
$\operatorname{im} u^{k+1} = u^k\bigl(u(E)\bigr) \subseteq
u^k(E) = \operatorname{im} u^k $.

\textbf{17.} Assumir $\ker u^{r} = \ker u^{r+1}$ e deixe $ x \in
\ker u^{r+2}$: então $ u(x) \in \ker u^{r+1} = \ker u^{r}$, então
$ u^{r+1}(x) = 0$: $ x \in \ker u^{r+1}$. Com a questão 16,
$\ker u^{r+1} = \ker u^{r+2}$, e por indução tudo mais tarde
grãos coincidir com $\ker u^{r}$. Para imagens: \omterm{def:b1:linmaps:rank}{classificação} - nulidade
dá $\dim\operatorname{im} u^k = n - \dim\ker u^k $, então o
\omterm{def:b1:linmaps:kerim}{imagem} dimensões congelam exatamente quando o \omterm{def:b1:linmaps:kerim}{núcleo} dimensões fazem,
e com as inclusões da questão 16, dimensões iguais significam
igual \omterm{def:b1:vspaces:subspace}{subespaços} (\cref{thm:b1:findim:subspaces}).

\textbf{18.} A sequência $\bigl(\dim\ker u^k\bigr)_k $ é
não decrescente com valores em $\intint{0}{n}$; não pode aumentar
estritamente $ n + 1$ vezes, então alguns $\dim\ker u^{r} = \dim\ker
u^{r+1}$ com $ r \leq n $, por isso $\ker u^{r} = \ker u^{r+1}$
(inclusão mais igual dimensão). Pegar $ r $ menor; questão 17
congela tudo desde $ r $ sobre, imagens incluído.

\textbf{19.} Interseção: deixe $ x \in \ker u^{r} \cap
\operatorname{im} u^{r}$, dizer $ x = u^{r}(y)$ com $ u^{r}(x) =
0$. Então $ u^{2r}(y) = 0$, e $\ker u^{2r} = \ker u^{r}$
(questão 17), então $ x = u^{r}(y) = 0$. Dimensões: \omterm{def:b1:linmaps:rank}{classificação} - nulidade
para $ u^{r}$ dá $\dim\ker u^{r} + \dim\operatorname{im} u^{r}
= n $; com interseção trivial, Grassmann torna a soma um
\omterm{def:b1:vspaces:subspace}{subespaço} de dimensão $ n $: $ E = \ker u^{r} \oplus
\operatorname{im} u^{r}$.

\textbf{20.} Estabilidade: $ u^{r}(u(x)) = u(u^{r}(x)) = 0$ para $ x
\in \ker u^{r}$; e $ u(u^{r}(y)) = u^{r}(u(y)) \in
\operatorname{im} u^{r}$. Sobre $ N = \ker u^{r}$: $(u|_N)^{r} = 0$
por definição de $ N $: nilpotente. Sobre $ I = \operatorname{im}
u^{r}$: $\ker(u|_I) = \ker u \cap I \subseteq \ker u^{r} \cap I
= \{0\}$, então $ u|_I $ é um \omterm{def:b1:logic:inj}{injetivo} endomorfismo do
\omterm{def:b1:findim:def}{dimensão finita} $ I $, por isso \omterm{def:b1:logic:inj}{bijetivo}
(\cref{cor:b1:linmaps:samedim}).

\textbf{21.} Deixar $ x = a + b $ com $ a \in N $, $ b \in I $. Então
$ u(x) = u(a) + u(b)$ com $ u(a) \in N $ e $ u(b) \in I $
(questão 20): isso \emph{é} a decomposição de $ u(x)$, então
$\pi(u(x)) = u(a) = u(\pi(x))$: $\pi u = u\pi $.

\textbf{22.} $ u^2(x,y,z) = u(y, 0, z) = (0, 0, z)$ e
$ u^3(x,y,z) = u(0,0,z) = (0,0,z) = u^2(x,y,z)$. \omterm{def:b1:linmaps:kerim}{Núcleos}: $\ker u
= \{y = z = 0\} = \operatorname{Vect}(e_1)$, $\ker u^2 = \{z =
0\} = \operatorname{Vect}(e_1, e_2)$, $\ker u^3 = \ker u^2$:
estabilização em $ r = 2$. Imagens: $\operatorname{im} u =
\operatorname{Vect}(e_1, e_3)$, $\operatorname{im} u^2 =
\operatorname{Vect}(e_3)$. Montagem: $\R^3 =
\operatorname{Vect}(e_1, e_2) \oplus \operatorname{Vect}(e_3)$,
e $\pi(x, y, z) = (x, y, 0)$. Verificar: $\pi u(x,y,z) = \pi(y, 0,
z) = (y, 0, 0)$ e $ u\pi(x,y,z) = u(x, y, 0) = (y, 0, 0)$:
igual. No primeiro fator $ u(x, y, 0) = (y, 0, 0)$, cujo
quadrado é $0$: nilpotente; no segundo $ u(0,0,z) = (0,0,z)$:
a identidade, \omterm{def:b1:logic:inj}{bijetivo}.

\textbf{23.} Se $ u^m = 0$ então $\ker u^m = E $; desde grãos são
congelado de $ r $ sobre, $\ker u^{r} = \ker u^{\max(m, r)} = E $.
Por outro lado $\ker u^{r} = E $ significa $ u^{r} = 0$. E $\ker u^{r} =
E \iff $ o projetor de montagem está ligado $ E $ junto $\{0\}$, ou seja\
$\pi = \mathrm{id}$. Finalmente $ r \leq n $ (questão 18) dá:
todo endomorfismo nilpotente satisfaz $ u^{n} = 0$ --- o
o índice de nilpotência nunca excede a dimensão.

\textbf{24.} Deixar $ m $ ser um índice de nilpotência de $ u|_A $: $ A
\subseteq \ker u^{m} \subseteq \ker u^{\max(m,r)} = \ker u^{r}$.
Desde $ u|_B $ é \omterm{def:b1:logic:inj}{bijetivo}, $ B = u(B) = u^{k}(B) \subseteq
\operatorname{im} u^{k}$ para cada $ k $, em particular $ B
\subseteq \operatorname{im} u^{r}$. Então
\[
n = \dim A + \dim B \leq \dim\ker u^{r} +
\dim\operatorname{im} u^{r} = n :
\]
ambas as inclusões são igualdades de dimensões, portanto de
\omterm{def:b1:vspaces:subspace}{subespaços}: $ A = \ker u^{r}$, $ B = \operatorname{im} u^{r}$.

\textbf{25.} (i) A Parte III é um dicionário: divisões $ E = F_1
\oplus \dots \oplus F_k $ correspondem exatamente às famílias de
projetores com $\sum p_i = \mathrm{id}$ e $ p_i p_j = 0$, o
$ F_i $ sendo o imagens. (ii) Para $ k = 3$ os complementos
$\mathrm{id} - p_i $ são eles próprios projetores, o que fechou o
argumento sem hipótese extra; para geral $ k $ é preciso
$\sum_i \operatorname{rk} p_i \leq n $, uma desigualdade no ano ~ 2
trace dá de graça ($\operatorname{tr} p = \operatorname{rk}
p $ para um projetor, e os traços somam $\operatorname{tr}
\mathrm{id} = n $). (iii) \cref{exo:b1:linmaps:7} é do Fitting
lema no caso já estabilizado $ r \leq 1$; em geral um
deixa o \omterm{def:b1:linmaps:kerim}{núcleo} e \omterm{def:b1:linmaps:kerim}{imagem} as correntes congelam, o que leva no máximo $ n $
passos. (iv) No volume Ano~2, aplicado a $ u - \lambda\,
\mathrm{id}$, o fator nilpotente torna-se o generalizado
autoespaço em $\lambda$ e os projetores da Parte III tornam-se
os projetores espectrais da teoria da redução. O teorema de
A Parte IV é \emph{Lema de Fitting}.
\end{solution}
