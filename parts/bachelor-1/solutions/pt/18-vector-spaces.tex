\begin{solution}{exo:b1:vspaces:1}
\begin{enumerate}
  \item Sim: contém $0$, e a equação definidora é linear
        (estável sob $ x + \lambda y $).
  \item Não: não contém $(0,0,0)$.
  \item Não: $(1, 0)$ e $(0, -1)$ pertencer ($ xy = 0$), sua soma $(1,

  \item Sim: $0$ desaparece em $1$; $(P + \lambda Q)(1) = P(1) +

\end{enumerate}
\end{solution}

\begin{solution}{exo:b1:vspaces:2}
$(1,2,1) = a(1,0,1) + b(1,1,0)$ requer $ a + b = 1$, $ b = 2$, $ a =
1$: incompatível ($ a + b = 3 \neq 1$): não no \omterm{def:b1:vspaces:span}{período}. $(2,1,1) =
a(1,0,1) + b(1,1,0)$: $ b = 1$, $ a = 1$, $ a + b = 2$: consistente, então
$(2,1,1) = (1,0,1) + (1,1,0)$, no \omterm{def:b1:vspaces:span}{período}.

Equação: $(x, y, z) = (a + b, b, a)$ significa $ x = y + z $: o \omterm{def:b1:vspaces:span}{período} é
o avião $\{x - y - z = 0\}$.
\end{solution}

\begin{solution}{exo:b1:vspaces:3}
Primeira família: $\lambda(1,1,0) + \mu(1,0,1) + \nu(0,1,1) = 0$ dá
$\lambda + \mu = 0$, $\lambda + \nu = 0$, $\mu + \nu = 0$: adicionando,
$2(\lambda + \mu + \nu) = 0$, e subtraindo cada equação,
$\lambda = \mu = \nu = 0$: \omterm{def:b1:vspaces:free}{livre}.

Segundo: $(2,4,6) = 2(1,2,3)$: vinculado.

Terceiro: quatro vetores em $\R^3$ --- necessariamente vinculado uma vez dimensão
está disponível (\cref{ch:b1:findim}); diretamente: $(0,1,1) = -(1,0,0) +
0\cdot(1,1,0) + (1,1,1)$, de fato $(-1,0,0) + (1,1,1) = (0,1,1)$: um
relação não trivial.
\end{solution}

\begin{solution}{exo:b1:vspaces:4}
\omterm{def:b1:poly:def}{Polinômios} $1, (X-1), (X-1)^2, (X-1)^3$ têm graus distintos $0,
1, 2, 3$: \omterm{def:b1:vspaces:free}{livre} (\cref{prop:b1:vspaces:freecriteria}~(1)); quatro \omterm{def:b1:vspaces:free}{livre}
vetores gerando (cada $ P \in \R_3[X]$ se expande em poderes de $ X -
1$, por exemplo\ por Taylor para \omterm{def:b1:poly:def}{polinômios}, cf.\ a prova de
\cref{prop:b1:poly:multiplicity}): um \omterm{def:b1:vspaces:free}{base}. Para $ X^3$, Taylor em
$1$: $ P = X^3$, $ P(1) = 1$, $ P'(1) = 3$, $ P''(1) = 6$, $ P'''(1) =
6$:
\[
X^3 = 1 + 3(X - 1) + 3(X-1)^2 + (X-1)^3 ,
\]
\omterm{prop:b1:vspaces:coordinates}{coordenadas} $(1, 3, 3, 1)$ (linha de Pascal, como esperado de $ X^3 =
((X-1)+1)^3$).
\end{solution}

\begin{solution}{exo:b1:vspaces:5}
$ F \cap G $: as condições $ x = y = z $ e $ x = t = 0$ juntos dar
$ x = 0$, por isso $ y = z = 0$, e $ t = 0$: a interseção é
$\{0\}$. Soma: dada $(x,y,z,t)$, procurar $(a,a,a,b) \in F $ e
$(0,c,d,0) \in G $ resumindo: $ a = x $, $ b = t $, $ c = y - x $, $ d =
z - x $: sempre possível. Então $\R^4 = F \oplus G $, e
\[
(1,2,3,4) = (1,1,1,4) + (0,1,2,0) .
\]
\end{solution}

\begin{solution}{exo:b1:vspaces:6}
Um elemento de $ G $ é $\lambda u $; se convergir, então (como
$\lambda u_n = \lambda(-1)^n $ tem os dois \omterm{def:b1:topology:closure}{limites} subsequentes $\pm
\lambda $) necessariamente $\lambda = 0$: $ F \cap G = \{0\}$.

$ F + G $ não é tudo: consiste em sequências da forma
$ c_n + \lambda(-1)^n $ com $(c_n)$ convergente. A sequência $ v_n =
n $ não é desta forma ($ v_n - \lambda(-1)^n $ é ilimitado, nunca
convergente). Então $ F \oplus G \subsetneq $ (espaço de todas as sequências).
\end{solution}

\begin{solution}{exo:b1:vspaces:7}
($\supseteq $) Ambos $ F \cap G $ e $ F \cap H $ deitar em $ F $, e seus
a soma está em $ G + (F \cap H)$: a inclusão segue desde o lado esquerdo
é um \omterm{def:b1:vspaces:subspace}{subespaço} contendo ambas as peças --- concretamente, um elemento $ g +
h $ com $ g \in F\cap G $, $ h \in F \cap H $ está em $ F $ (soma de dois
elementos de $ F $) e em $ G + (F \cap H)$.

($\subseteq $) Deixar $ x \in F $ com $ x = g + h $, $ g \in G $, $ h \in F
\cap H $. Então $ g = x - h \in F $ (diferença de elementos de $ F $), então
$ g \in F \cap G $, e $ x = g + h \in (F \cap G) + (F \cap H)$.

Contra-exemplo para distributividade total em $\R^2$: $ F =
\operatorname{Vect}(1,1)$, $ G = \operatorname{Vect}(1,0)$, $ H =
\operatorname{Vect}(0,1)$. Então $ G + H = \R^2$, então $ F \cap (G+H) =
F $, enquanto $ F \cap G = F \cap H = \{0\}$: o lado direito é $\{0\}
\neq F $.
\end{solution}

\begin{solution}{exo:b1:vspaces:8}
\begin{enumerate}
  \item Suponha $\sum_{i} \lambda_i \eu^{a_i x} = 0$ para todos $ x $.
        Multiplicar por $\eu^{-a_p x}$: $\lambda_p + \sum_{i < p}
        \lambda_i \eu^{(a_i - a_p)x} \to \lambda_p $ como $ x \to
        +\infty $ (cada expoente $ a_i - a_p < 0$). O lado esquerdo é
        de forma idêntica $0$, então $\lambda_p = 0$; iterar para baixo.
  \item Deixar $ a\cos x + b \sin x + c \cos 2x + d\sin 2x = 0$ para todos
        $ x $. Avaliar em $ x = 0$: $ a + c = 0$; no $ x = \pi $: $-a + c
        = 0$; por isso $ a = c = 0$, e a relação se reduz a $ b\sin
        x + d \sin 2x = 0$. Avaliar em $ x = \frac\pi2$: $ b = 0$;
        então em $ x = \frac\pi4$: $ d = 0$.
  \item Suponha $\sum \lambda_i \abs{x - a_i} = 0$ para todos $ x $. O
        função $\sum_{i \neq j} \lambda_i\abs{x - a_i}$ é
        diferenciável no $ a_j $ (cada termo é, longe de seu próprio
        canto), então $-\lambda_j \abs{x - a_j}$, sua diferença,
        deve ser diferenciável no $ a_j $ também --- quais forças
        $\lambda_j = 0$ ($\abs{\,\cdot\,}$ tem um canto). Isso vale
        para cada $ j $.
\end{enumerate}
\end{solution}

\begin{solution}{exo:b1:vspaces:9}
Deixar $ h \in H $. Desde $ h \in H \subseteq F + H = F + G $, escrever $ h = f
+ g $ com $ f \in F $, $ g \in G $. Então $ f = h - g \in H $ (ambos os termos
em $ H $, usando $ G \subseteq H $), então $ f \in F \cap H = F \cap G
\subseteq G $, e $ h = f + g \in G $. Por isso $ H \subseteq G $, e com
a hipótese $ G \subseteq H $: igualdade.

Contraexemplo sem $ G \subseteq H $: em $\R^2$, pegar $ F =
\operatorname{Vect}(1,0)$, $ G = \operatorname{Vect}(0,1)$, $ H =
\operatorname{Vect}(1,1)$: então $ F + G = F + H = \R^2$ e $ F \cap G
= F \cap H = \{0\}$, ainda $ G \neq H $.
\end{solution}

\begin{solution}{exo:b1:vspaces:10}
Ambos conjuntos são \omterm{def:b1:vspaces:subspace}{subespaços} (as condições definidoras são lineares e
espere por $0$). Decomposição: para $ P \in \R[X]$,
\[
P(X) = \underbrace{\frac{P(X) + P(-X)}{2}}_{\in\,\mathcal P}
+ \underbrace{\frac{P(X) - P(-X)}{2}}_{\in\,\mathcal I},
\]
e um polinomial tanto pares quanto ímpares satisfazem $ P = -P $, então $ P = 0$:
a soma é direta e igual $\R[X]$.

Agora deixe $ P = \sum_k a_k X^k $ esteja equilibrado. Então $ P(X) - P(-X) = 2
\sum_{k \text{ odd}} a_k X^k $ é o zero polinomial, então cada
coeficiente de grau ímpar desaparece (\cref{def:b1:poly:def}): $ P \in
\operatorname{Vect}(1, X^2, X^4, \dots)$, ou seja\ $ P = Q(X^2)$ por um
polinomial $ Q $. Por outro lado, cada polinomial em $ X^2$ é par.
\end{solution}

\begin{solution}{exo:b1:vspaces:11}
Suponha $ a(x_1 + x_2) + b(x_2 + x_3) + c(x_3 + x_1) = 0$.
Reagrupamento no família \omterm{def:b1:vspaces:free}{livre} $(x_1, x_2, x_3)$:
\[
(a + c)\,x_1 + (a + b)\,x_2 + (b + c)\,x_3 = 0
\implies a + c = a + b = b + c = 0 .
\]
Subtraindo as duas primeiras equações dá $ c = b $; o terceiro então
dá $2b = 0$, então $ b = c = 0$, então $ a = 0$: a família é \omterm{def:b1:vspaces:free}{livre}.

Para quatro vetores a família análoga é \emph{sempre} vinculado:
\[
(x_1 + x_2) - (x_2 + x_3) + (x_3 + x_4) - (x_4 + x_1) = 0
\]
é uma combinação nula não trivial (coeficientes $1, -1, 1, -1$),
qualquer que seja $(x_1, x_2, x_3, x_4)$ é. A paridade da duração do ciclo
decide.
\end{solution}

\begin{solution}{exo:b1:vspaces:12}
\begin{enumerate}
  \item Se $ F_1 \subseteq F_2$ ou $ F_2 \subseteq F_1$, o sindicato
        é um dos dois, portanto adequado. Caso contrário, escolha $ x \in F_1
        \setminus F_2$ e $ y \in F_2 \setminus F_1$, e considere
        $ x + y $. Se $ x + y \in F_1$, então $ y = (x + y) - x \in
        F_1$: contradição. Se $ x + y \in F_2$, então $ x \in F_2$:
        contradição. Então $ x + y \notin F_1 \cup F_2$, e $ E \neq
        F_1 \cup F_2$.
  \item Suponha por contradição que $ E = F_1 \cup \dots \cup
        F_k $, com $ k $ escolhido \emph{mínimo} entre todos esses
        coberturas. A minimalidade proíbe $ F_1 \subseteq F_2 \cup
        \dots \cup F_k $ (caso contrário, solte $ F_1$), então há $ x \in F_1$
        com $ x \notin F_i $ para todos $ i \geq 2$. Desde $ F_1$ é
        adequado, escolha $ y \notin F_1$. Para cada escalar $ t $, o
        vetor $ y + t x $ reside em alguns $ F_i $. Nunca mente
        $ F_1$: de outra forma $ y = (y + tx) - tx \in F_1$ (como $ x \in
        F_1$). O \omterm{def:b1:structures:field}{campo} é infinito, então escolha $ k $ distinto
        escalares $ t_1, \dots, t_k $: o $ k $ vetores $ y + t_j x $
        cair no $ k - 1$ \omterm{def:b1:vspaces:subspace}{subespaços} $ F_2, \dots, F_k $ e dois
        deles, digamos $ y + t x $ e $ y + t' x $ com $ t \neq t'$,
        deitar no mesmo $ F_i $ ($ i \geq 2$). Então a diferença deles
        $(t - t')x \in F_i $, então $ x \in F_i $: contradição. Portanto
        nenhuma cobertura finita por adequada \omterm{def:b1:vspaces:subspace}{subespaços} existe.
\end{enumerate}
\end{solution}

\begin{solution}{pb:b1:vspaces:1}
\textbf{1.} $ L_i $ é um produto de $ n $ fatores lineares divididos por um
constante diferente de zero (o $ x_i $ são distintos), então $\deg L_i = n $.
Avaliando em $ x_j $ com $ j \neq i $: o fator $ X - x_j $ do
numerador desaparece, então $ L_i(x_j) = 0$. No $ x_i $, numerador e
denominador coincide: $ L_i(x_i) = 1$.

\textbf{2.} Suponha $\sum_i \lambda_i L_i = 0$. Avaliar em $ x_j $:
todos os termos morrem, exceto $\lambda_j L_j(x_j) = \lambda_j $, então
$\lambda_j = 0$ para cada $ j $: a família é \omterm{def:b1:vspaces:free}{livre}.

\textbf{3.} Deixar $ D = P - \sum_i P(x_i) L_i $. Então $\deg D \leq n $
e, pela pergunta 1, $ D(x_j) = P(x_j) - P(x_j) = 0$ para o $ n + 1$
pontos distintos $ x_0, \dots, x_n $. Um diferente de zero polinomial de grau
$\leq n $ tem no máximo $ n $ raízes (\cref{cor:b1:poly:nroots}), então $ D
= 0$. Assim cada $ P \in \R_n[X]$ é uma combinação do $ L_i $:
a família está gerando, e com a questão 2, um \omterm{def:b1:vspaces:free}{base}.

\textbf{4.} Dado $ y_0, \dots, y_n $, o polinomial $ P = \sum_i
y_i L_i $ tem diploma $\leq n $ e interpola. Singularidade: um
interpolando $ P $ tem, pela pergunta 3, \omterm{prop:b1:vspaces:coordinates}{coordenadas} $(P(x_0), \dots,
P(x_n)) = (y_0, \dots, y_n)$ no \omterm{def:b1:vspaces:free}{base} $(L_i)$, e \omterm{prop:b1:vspaces:coordinates}{coordenadas}
em um \omterm{def:b1:vspaces:free}{base} são únicos (\cref{prop:b1:vspaces:coordinates}). O
\omterm{prop:b1:vspaces:coordinates}{coordenadas} de $ P $ em Lagrange \omterm{def:b1:vspaces:free}{base} são seus \emph{valores em
os nós} --- este é o ponto principal do \omterm{def:b1:vspaces:free}{base}.

\textbf{5.} Aplique a questão 3 a $ P = X^k $ ($0 \leq k \leq n $):
\[
X^k = \sum_{i=0}^{n} x_i^{k} L_i ,
\]
e $ k = 0$ dá $\sum_i L_i = 1$.

\textbf{6.} $\deg N_k = k $ exatamente: a família $(N_0, \dots, N_n)$
é uma escada de graus em $\R_n[X]$, daí um \omterm{def:b1:vspaces:free}{base} por
\cref{ex:b1:vspaces:staircase} (liberdade de
\cref{prop:b1:vspaces:freecriteria}~(1), geração por finito
descida no grau).

\textbf{7.} Da recorrência,
\[
f[x_0, x_1] = \frac{f(x_1) - f(x_0)}{x_1 - x_0},
\qquad
f[x_0, x_1, x_2] = \frac{f[x_1, x_2] - f[x_0, x_1]}{x_2 - x_0}.
\]
Para $ f(x) = x^2$:
\[
f[x_0, x_1] = \frac{x_1^2 - x_0^2}{x_1 - x_0} = x_0 + x_1,
\]
e então
\[
f[x_0, x_1, x_2]
= \frac{(x_1 + x_2) - (x_0 + x_1)}{x_2 - x_0}
= \frac{x_2 - x_0}{x_2 - x_0} = 1 .
\]

\textbf{8.} $\deg S \leq n $ desde $ Q, R $ ter diploma $\leq n - 1$.
No $ x_0$: $ S(x_0) = \frac{-(x_0 - x_n) R(x_0)}{x_n - x_0} =
R(x_0) = f(x_0)$. No $ x_n $: $ S(x_n) = \frac{(x_n - x_0)
Q(x_n)}{x_n - x_0} = Q(x_n) = f(x_n)$. Em um \omterm{def:b1:topology:closure}{interior} nó $ x_i $
($1 \leq i \leq n-1$), ambos $ Q $ e $ R $ pegue o valor $ f(x_i)$,
então
\[
S(x_i) = \frac{(x_i - x_0) - (x_i - x_n)}{x_n - x_0}\, f(x_i)
= f(x_i) .
\]

\textbf{9.} Indução no número de nós. Um nó: o
interpolante é a constante $ f(x_0) = f[x_0]$. Assuma a afirmação
para $ k $ nós e deixe $ S $ interpolar em $ x_0, \dots, x_k $; por
singularidade (questão 4), $ S $ é dado pelo lema de Aitken de $ R $
(nós $ x_0, \dots, x_{k-1}$) e $ Q $ (nós $ x_1, \dots, x_k $).
O coeficiente de $ X^{k}$ em $ S $ é
\[
\frac{[X^{k-1}]\,Q - [X^{k-1}]\,R}{x_k - x_0}
= \frac{f[x_1, \dots, x_k] - f[x_0, \dots, x_{k-1}]}{x_k - x_0}
= f[x_0, \dots, x_k]
\]
pela hipótese de indução e pela recorrência definidora.

\textbf{10.} Deixar $ P_k $ interpolar $ f $ no $ x_0, \dots, x_k $. A diferença $ P_k - P_{k-1}$ tem diploma $\leq k $ e desaparece em
$ x_0, \dots, x_{k-1}$, então pelo teorema do fator aplicado $ k $ vezes
(\cref{thm:b1:poly:factor}) é igual $ c\,N_k $ para uma constante $ c $;
comparando coeficientes de $ X^{k}$ e usando a questão 9, $ c =
f[x_0, \dots, x_k]$. Telescópica de $ P_0 = f(x_0) N_0$ dá
Fórmula de Newton. Para o formulário \omterm{def:b1:topology:closed}{fechado}, escreva $ P_k = \sum_{i \leq
k} f(x_i) L_i $ (Lagrange, nos nós $ x_0, \dots, x_k $) e leia
o coeficiente de $ X^{k}$: cada $ L_i $ contribui
$\frac{1}{\prod_{j \neq i}(x_i - x_j)}$, de onde
\[
f[x_0, \dots, x_k] = \sum_{i=0}^{k}
\frac{f(x_i)}{\prod_{j \neq i,\, j \leq k}(x_i - x_j)} .
\]
O lado direito é invariante sob qualquer \omterm{def:b1:counting:objects}{permutação} dos nós, então
a diferença dividida não depende de sua ordenação.

\textbf{11.} Se $ P = a X^m + (\text{lower degrees})$, o binômio
teorema dá
\[
\Delta P = a\bigl((X+1)^m - X^m\bigr) + \dots
= a\,m\,X^{m-1} + (\text{lower degrees}),
\]
porque $(X+1)^m - X^m = m X^{m-1} + \dots $ e o grau inferior
parte de $ P $ contribui com grau $\leq m - 2$ depois $\Delta $ (ou
grau $\leq m-2$ termos). Então $\deg \Delta P = m - 1$ com liderança
coeficiente $ m a $. Uma constante $ c $ dá $\Delta c = c - c = 0$.

\textbf{12.} $\deg B_k = k $: escada, daí um \omterm{def:b1:vspaces:free}{base} de
$\R_n[X]$ (\cref{ex:b1:vspaces:staircase}). Para $\Delta B_k $
($ k \geq 1$), fatore o produto comum:
\begin{align*}
k!\,\Delta B_k
&= (X+1)X\cdots(X-k+2) - X(X-1)\cdots(X-k+1) \\
&= X(X-1)\cdots(X-k+2)\,\bigl[(X+1) - (X-k+1)\bigr] \\
&= k\,X(X-1)\cdots(X-k+2),
\end{align*}
então $\Delta B_k = \frac{X(X-1)\cdots(X-k+2)}{(k-1)!} = B_{k-1}$.

\textbf{13.} Escrever $ P = \sum_{k=0}^{n} c_k B_k $ (\omterm{def:b1:vspaces:free}{base}, pergunta
12). Aplicar $\Delta^{j}$: pela pergunta 12, $\Delta^{j} P = \sum_{k
\geq j} c_k B_{k-j}$. Avaliar em $0$: $ B_0(0) = 1$ e $ B_m(0) =
0$ para $ m \geq 1$ (o fator $ X $ desaparece), então
$\bigl(\Delta^{j}P\bigr)(0) = c_j $. Esta é a diferença direta
fórmula.

\textbf{14.} Indução ativada $ k $. Para $ k = 0$ a identidade lê
$ P(0) = P(0)$. Assuma isso por $ k $ e aplique-o $\Delta P $:
\[
\bigl(\Delta^{k+1} P\bigr)(0)
= \sum_{j=0}^{k} (-1)^{k-j}\binom kj \bigl(P(j+1) - P(j)\bigr).
\]
Colete o coeficiente de $ P(i)$: isso é $(-1)^{k-i+1}\binom
k{i-1}\cdot(-1)^{0}$ da primeira soma (deslocada) e
$-(-1)^{k-i}\binom ki $ do segundo --- juntos
\[
(-1)^{k+1-i}\Bigl(\binom k{i-1} + \binom ki\Bigr)
= (-1)^{k+1-i}\binom{k+1}i
\]
pela regra de Pascal, que é a identidade na classificação $ k + 1$.

\textbf{15.} Iterando a questão 11 do grau $ n $, liderando
coeficiente $ a_n $: depois de um $\Delta $, grau $ n-1$ e liderando
coeficiente $ n a_n $; depois das duas, $ n(n-1)a_n $; depois $ n $ passos,
grau $0$ e valor $ n(n-1)\cdots 1\, a_n = n!\,a_n $, uma constante.
Mais um $\Delta $ mata: $\Delta^{n+1}P = 0$.

\textbf{16.} Se $ m \geq k $: $ B_k(m) = \binom mk \in \N $. Se $0
\leq m < k $: um fator de $ m(m-1)\cdots(m-k+1)$ é zero, então
$ B_k(m) = 0$. Se $ m = -q $ com $ q \geq 1$:
\[
B_k(-q) = \frac{(-q)(-q-1)\cdots(-q-k+1)}{k!}
= (-1)^k\,\frac{q(q+1)\cdots(q+k-1)}{k!}
= (-1)^k \binom{q+k-1}{k},
\]
um número inteiro. Então cada $ B_k $ \omterm{def:b1:logic:map}{mapas} $\Z $ em $\Z $.

\textbf{17.} ($\Leftarrow $) Se $ P = \sum_k c_k B_k $ com $ c_k \in
\Z $, então para $ m \in \Z $, $ P(m) = \sum_k c_k B_k(m) \in \Z $ por
questão 16. ($\Rightarrow $) Se $ P $ tem valor inteiro, seu
\omterm{prop:b1:vspaces:coordinates}{coordenadas} são $ c_k = \bigl(\Delta^k P\bigr)(0) = \sum_{j=0}^k
(-1)^{k-j}\binom kj P(j)$ (questões 13 e 14), um número inteiro
combinação dos inteiros $ P(0), \dots, P(k)$. Isto é
Pćaracterização de olya de \omterm{def:b1:poly:def}{polinômios} com valor inteiro.

\textbf{18.} \omterm{def:b1:logic:sets}{Definir} $ Q(X) = P(X + a)$, um polinomial de grau $\leq
n $ com $ Q(0), Q(1), \dots, Q(n) \in \Z $. Isso é \omterm{prop:b1:vspaces:coordinates}{coordenadas} em
$(B_k)_{k \leq n}$ são $ c_k = \sum_{j \leq k}(-1)^{k-j}\binom kj
Q(j) \in \Z $ (a questão 14 utiliza apenas os valores em $0, \dots, k
\leq n $). Pela pergunta 17 ($\Leftarrow $), $ Q $ tem valor inteiro em
todos $\Z $, portanto, assim é $ P(X) = Q(X - a)$.

\textbf{19.} Um produto de $ k $ inteiros consecutivos é $ m(m-1)
\cdots(m-k+1) = k!\,B_k(m)$ para alguns $ m \in \Z $, e $ B_k(m) \in
\Z $ pela questão 16: o produto é divisível por $ k!$.

\textbf{20.} Valores de $ P = \frac{X(X+1)(2X+1)}{6}$ no $0,1,2,3$:
$0, 1, 5, 14$. Tabela de diferenças: $\Delta $ linha $1, 4, 9$;
$\Delta^2$ linha $3, 5$; $\Delta^3$ linha $2$. Portanto, pela questão 13,
\[
P = 0\cdot B_0 + 1\cdot B_1 + 3\,B_2 + 2\,B_3 ,
\]
com número inteiro \omterm{prop:b1:vspaces:coordinates}{coordenadas}: $ P $ tem valor inteiro (questão 17),
enquanto seus coeficientes monomiais $\frac13, \frac12, \frac16$ são
não inteiros. Cálculo \omterm{def:b1:vspaces:sum}{direto}:
\begin{align*}
\Delta P &= \frac{(X+1)(X+2)(2X+3) - X(X+1)(2X+1)}{6} \\
&= \frac{(X+1)\bigl[(X+2)(2X+3) - X(2X+1)\bigr]}{6}
= \frac{(X+1)(6X+6)}{6} = (X+1)^2 .
\end{align*}
Telescópico $ P(m) = \sum_{j=0}^{m-1}\Delta P(j) = \sum_{j=1}^{m}
j^2$ (com $ P(0) = 0$): a fórmula da soma dos quadrados.

\textbf{21.} Os valores $2^i $ no $ i = 0, \dots, n $ tem diferença
mesa constantemente $1$ na borda esquerda: $\Delta^k $ da sequência
$(2^i)$ é $(2^i)$ novamente (desde $2^{i+1} - 2^i = 2^i $), então
$\bigl(\Delta^k P\bigr)(0) = 2^0 = 1$ para todos $ k \leq n $, e $ P =
B_0 + B_1 + \dots + B_n $ pela pergunta 13. Então
\[
P(n+1) = \sum_{k=0}^{n}\binom{n+1}{k}
= 2^{n+1} - \binom{n+1}{n+1} = 2^{n+1} - 1 \neq 2^{n+1}:
\]
o padrão quebra no primeiro ponto não controlado.

\textbf{22.} Pela pergunta 12, $ B_k = \Delta B_{k+1}$, então
\[
\sum_{j=0}^{m-1} B_k(j)
= \sum_{j=0}^{m-1}\bigl(B_{k+1}(j+1) - B_{k+1}(j)\bigr)
= B_{k+1}(m) - B_{k+1}(0) = B_{k+1}(m).
\]
Para $ j < k $ os termos $ B_k(j)$ desaparece, então a soma realmente começa em
$ j = k $: $\sum_{j=k}^{m-1}\binom jk = \binom m{k+1}$, o
identidade do taco de hóquei.

\textbf{23.} Tabelas de diferenças (ou expansão direta) fornecem
\[
X^2 = B_1 + 2 B_2, \qquad X^3 = B_1 + 6 B_2 + 6 B_3
\]
(verificar: $ B_1 + 2B_2 = X + X(X-1) = X^2$; no $ X = 1, 2, 3$ o
segundo dá $1, 8, 27$). A questão 22 então produz
\[
\sum_{j=0}^{m-1} j^2 = B_2(m) + 2B_3(m)
= \binom m2 + 2\binom m3 = \frac{m(m-1)(2m-1)}{6},
\]
\[
\sum_{j=0}^{m-1} j^3 = B_2(m) + 6B_3(m) + 6B_4(m)
= \binom m2 + 6\binom m3 + 6\binom m4 .
\]
Expandindo a última expressão: $\binom m2 + 6\binom m3 + 6\binom
m4 = \frac{m(m-1)}{2}\bigl[1 + 2(m-2) +
\frac{(m-2)(m-3)}{2}\bigr] = \frac{m^2(m-1)^2}{4} = \binom m2^2$.
Substituindo $ m $ por $ m + 1$: $1^3 + \dots + m^3 =
\bigl(\frac{m(m+1)}2\bigr)^2 = (1 + \dots + m)^2$, Nicômaco'
identidade.

\textbf{24.} Nós $0, 1, 2$, polinomial $ X^2$. Monômio \omterm{def:b1:vspaces:free}{base}
$(1, X, X^2)$: \omterm{prop:b1:vspaces:coordinates}{coordenadas} $(0, 0, 1)$. Lagrange \omterm{def:b1:vspaces:free}{base}:
\omterm{prop:b1:vspaces:coordinates}{coordenadas} são os valores $(0, 1, 4)$ (questão 4). Newton \omterm{def:b1:vspaces:free}{base}
$(1, X, X(X-1))$: \omterm{pb:b1:vspaces:1}{diferenças divididas} $ f[0] = 0$, $ f[0,1] = 1$,
$ f[0,1,2] = \frac{3 - 1}{2} = 1$ (questão 9), então \omterm{prop:b1:vspaces:coordinates}{coordenadas}
$(0, 1, 1)$ --- de fato $ X + X(X-1) = X^2$. Três \omterm{def:b1:vspaces:free}{bases}, três
vetores coordenados, um polinomial.

\textbf{25.} (i) A questão 4 é automática porque $(L_i)$ é um
\emph{\omterm{def:b1:vspaces:free}{base}}: a existência e a singularidade da interpolação são
exatamente a existência e a singularidade de \omterm{prop:b1:vspaces:coordinates}{coordenadas}. (ii) O Newton
\omterm{def:b1:vspaces:free}{base} é uma escada, então \omterm{prop:b1:vspaces:coordinates}{coordenadas} são computados por sucessivos
divisões --- cada novo nó adiciona um termo sem perturbar o
anteriores --- enquanto o Lagrange \omterm{prop:b1:vspaces:coordinates}{coordenadas} de $ P $ são os
valores $ P(x_i)$, disponível sem nenhum cálculo. (iii) Ambos
\omterm{def:b1:vspaces:free}{bases} são \omterm{def:b1:vspaces:free}{livre} pelos mesmos dois critérios de
\cref{prop:b1:vspaces:freecriteria}: graus distintos para Newton,
avaliação nos nós de Lagrange. (iv)Pó teorema de olya diz
que ``$ P(\Z) \subseteq \Z $'', uma propriedade de valores, é equivalente
à integralidade do \omterm{prop:b1:vspaces:coordinates}{coordenadas} no \omterm{def:b1:vspaces:free}{base} $(B_k)$ --- o
aritmética de um polinomial torna-se visível apenas no \omterm{def:b1:vspaces:free}{base}
adaptado à questão.
\end{solution}
