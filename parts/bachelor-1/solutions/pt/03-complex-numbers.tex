\begin{solution}{exo:b1:complex:1}
$1 + \iu = \sqrt 2\, \eu^{\iu\pi/4}$;
$\;\sqrt 3 - \iu = 2\, \eu^{-\iu\pi/6}$;
$\;-5 = 5\, \eu^{\iu\pi}$;
\[
\frac{1 + \iu}{\sqrt 3 - \iu}

\iu\sin\frac{5\pi}{12})$:
\[
\cos\frac{5\pi}{12} = \frac{\sqrt 6 - \sqrt 2}{4},
\qquad
\sin\frac{5\pi}{12} = \frac{\sqrt 6 + \sqrt 2}{4}.
\]
\end{solution}

\begin{solution}{exo:b1:complex:3}

\begin{solution}{exo:b1:complex:4}
LinearizaĂ§ĂŁo:
\[
\sin^4\theta
= \Bigl(\frac{\eu^{\iu\theta} - \eu^{-\iu\theta}}{2\iu}\Bigr)^{\!4}
= \frac{\eu^{4\iu\theta} - 4\eu^{2\iu\theta} + 6 - 4\eu^{-2\iu\theta}
+ \eu^{-4\iu\theta}}{16}
= \frac{\cos 4\theta - 4\cos 2\theta + 3}{8}.
\]
ExpansĂŁo: por de Moivre, $\cos 4\theta = \Re\bigl((c +
\iu s)^4\bigr) = c^4 - 6c^2 s^2 + s^4$ com $ c = \cos\theta $, $ s =
\sin\theta $; substituindo $ s^2 = 1 - c^2$:
\[
\cos 4\theta = c^4 - 6c^2(1 - c^2) + (1 - c^2)^2
= 8c^4 - 8c^2 + 1 .
\]
\end{solution}

\begin{solution}{exo:b1:complex:10}
Deixar $ j = \eu^{2\iu\pi/3}$, entĂŁo $ j^2 + j + 1 = 0$ e $\eu^{\iu\pi/3} =
-j^2$, $\eu^{-\iu\pi/3} = -j $. O triĂąngulo Ă© equilĂĄtero direto se
$ c - a = -j^2 (b - a)$, indireto se sef $ c - a = -j(b - a)$
(\cref{ex:b1:complex:equilateral}).

Considerar $ P = a + jb + j^2 c $ e $ Q = a + j^2 b + jc $. Usando $1 +
j^2 = -j $ e $ j^3 = 1$:
\[
c - a + j^2(b - a) = c + j^2 b - (1 + j^2)\,a = c + j^2 b + ja
= j\,(a + jb + j^2 c) = jP,
\]
entĂŁo o caso direto diz $ jP = 0 \iff P = 0$; o mesmo cĂĄlculo
com $ j $ no lugar de $ j^2$ dĂĄ $ c - a + j(b - a) = j^2 Q $, entĂŁo o
leituras indiretas de casos $ Q = 0$. Portanto: equilĂĄtero (qualquer orientaĂ§ĂŁo)
$\iff PQ = 0$. Expandindo, com $ j + j^2 = -1$:
\[
PQ = a^2 + b^2 + c^2 + (j + j^2)(ab + bc + ca)
= a^2 + b^2 + c^2 - (ab + bc + ca).
\]
Portanto, o triĂąngulo Ă© equilĂĄtero se e somente se $ a^2 + b^2 + c^2 = ab +
bc + ca $.
\end{solution}

\begin{solution}{exo:b1:complex:11}
Desde o $\omega^k $, $ k = 0, \dots, n-1$, sĂŁo exatamente os $ n $ raĂ­zes
de $ X^n - 1$ (\cref{thm:b1:complex:roots}), e o polinĂŽmio Ă©
MĂŽnica:
\[
X^n - 1 = \prod_{k=0}^{n-1} (X - \omega^k)
= (X - 1) \prod_{k=1}^{n-1} (X - \omega^k).
\]
Dividindo por $ X - 1$: $\;1 + X + \dots + X^{n-1} = \prod_{k=1}^{n-1}
(X - \omega^k)$. Avaliando em $ X = 1$ dĂĄ $ P = n $.

Agora $1 - \omega^k = -\eu^{\iu k\pi/n}\bigl(\eu^{\iu k\pi/n} -
\eu^{-\iu k\pi/n}\bigr) = -2\iu\,\eu^{\iu k\pi/n} \sin\frac{k\pi}{n}$,
entĂŁo tomando \omterm{def:b1:complex:field}{mĂłdulos} $ P = n $ (cada $\sin\frac{k\pi}n > 0$ para $1 \leq k
\leq n-1$):
\[
n = \abs P = \prod_{k=1}^{n-1} 2\sin\frac{k\pi}{n}
= 2^{n-1} \prod_{k=1}^{n-1} \sin\frac{k\pi}{n},
\qquad\text{hence}\qquad
\prod_{k=1}^{n-1} \sin\frac{k\pi}{n} = \frac{n}{2^{n-1}} .
\]
\end{solution}

\begin{solution}{pb:b1:complex:1}
\textbf{1.} $ j = \cos\frac{2\pi}3 + \iu\sin\frac{2\pi}3 = -\frac12
+ \iu\frac{\sqrt3}2$; $ j^2 = \eu^{4\iu\pi/3} = -\frac12 -
\iu\frac{\sqrt3}2 = \conj j $; $ j^3 = 1$; $1 + j + j^2 = 0$ (soma de
o cubo raĂ­zes da unidade, \cref{prop:b1:complex:sumroots}); $ j^{-1}
= j^2$ (desde $ j \cdot j^2 = 1$). No cĂ­rculo unitĂĄrio, $1$, $ j $,
$ j^2$ sĂŁo os vĂ©rtices de um triĂąngulo equilĂĄtero direto.

\textbf{2.} A rotaĂ§ĂŁo do centro $ a $ e Ăąngulo $\theta $ correĂ§Ă”es $ a $
e transforma cada vetor emitido de $ a $ por $\theta $: $ r(z) - a =
\eu^{\iu\theta}(z - a)$, ou seja\ $ r(z) = a + \eu^{\iu\theta}(z - a)$.
ComposiĂ§ĂŁo $ r(z) = a + \eu^{\iu\theta}(z-a)$ e $ r'(z) = a' +
\eu^{\iu\theta'}(z-a')$:
\[
r' \circ r\,(z) = \eu^{\iu(\theta + \theta')} z +
\text{constant},
\]
um \omterm{def:b1:logic:map}{mapa} da forma $ Az + B $ com $ A = \eu^{\iu(\theta+\theta')}$ de
\omterm{def:b1:complex:field}{mĂłdulo} $1$. Por \cref{prop:b1:complex:similitude}, Ă© uma rotaĂ§ĂŁo
de Ăąngulo $\arg A = \theta + \theta'$ se $ A \neq 1$, e um
traduĂ§ĂŁo se $ A = 1$, ou seja,\ se $\theta + \theta' \in 2\pi\Z $.

\textbf{3.} $(b, c, p)$ Ă© equilĂĄtero se $\abs{p - b} = \abs{c -
b} = \abs{p - c}$. Escrita $ q = \frac{p - b}{c - b}$, o primeiro
igualdade diz $\abs q = 1$ e o segundo $\abs{q - 1} = 1$;
junto, $ q = \eu^{\pm\iu\pi/3}$ (os dois pontos de intersecĂ§ĂŁo de
os cĂ­rculos $\abs q = 1$ e $\abs{q - 1} = 1$ sĂŁo $\frac12 \pm
\iu\frac{\sqrt3}2$). Por isso $ p = b + \eu^{\pm\iu\pi/3}(c - b)$. Para
$ a = 0$, $ b = 1$, $ c = \iu $ (um triĂąngulo direto): a escolha
$\eu^{-\iu\pi/3}$ dĂĄ
\[
p = 1 + \Bigl(\tfrac12 - \iu\tfrac{\sqrt3}2\Bigr)(\iu - 1)
= \tfrac{1 + \sqrt3}2\,(1 + \iu) \approx 1.37 + 1.37\,\iu ,
\]
que fica do outro lado da linha $ BC $ ($ x + y = 1$) de $ a
= 0$: para fora. A escolha $\eu^{+\iu\pi/3}$ dĂĄ $ p \approx -0.37
- 0.37\,\iu $, do mesmo lado que $ a $: para dentro.

\textbf{4.} Multiplicar $ P = a + jb + j^2c $ por $ j^2$: $ j^2 P = j^2 a +
j^3 b + j^4 c = b + jc + j^2 a $ (usando $ j^3 = 1$, $ j^4 = j $); e
por $ j $: $ jP = ja + j^2 b + c $. Estas sĂŁo as duas identidades. Se $ P
= 0$ entĂŁo $ j^2 P = jP = 0$: o critĂ©rio vale para $(b, c, a)$ e
$(c, a, b)$ tambĂ©m --- invariĂąncia cĂ­clica (como deve ser: um
triĂąngulo equilĂĄtero nĂŁo se importa com qual vĂ©rtice Ă© listado primeiro).

\textbf{5.} Se $ a = b $: $0 = a(1 + j) + j^2 c = -j^2 a + j^2 c $
(usando $1 + j = -j^2$), entĂŁo $ c = a $. Se $ b = c $: $0 = a + b(j + j^2)
= a - b $, entĂŁo $ a = b $. Se $ a = c $: $0 = a(1 + j^2) + jb = -ja + jb $,
entĂŁo $ a = b $. Em cada caso, todos os trĂȘs pontos coincidem.

\textbf{6.} $(a'z + b') \circ (az + b) = a'a\,z + (a'b + b')$ com
$ a'a \neq 0$: mesma forma. O inverso de $ z \mapsto az + b $ Ă© $ z
\mapsto \frac1a z - \frac ba $, novamente da mesma forma. Com o
identidade \omterm{def:b1:logic:map}{mapa} como elemento neutro, as semelhanĂ§as diretas formam um
grupo sob composiĂ§ĂŁo.

\textbf{7.} $ f(z) = az + b $ satisfaz $ f(z_1) = w_1$, $ f(z_2) =
w_2$ se $ a z_1 + b = w_1$ e $ a z_2 + b = w_2$; subtraindo,
$ a(z_2 - z_1) = w_2 - w_1$, entĂŁo
\[
a = \frac{w_2 - w_1}{z_2 - z_1} \;(\neq 0), \qquad
b = w_1 - a z_1
\]
sĂŁo forĂ§ados e, inversamente, esta escolha funciona: a existĂȘncia e
singularidade.

\textbf{8.} $\abs{f(z) - f(w)} = \abs{a(z - w)} = \abs a\,\abs{z -
w}$: todas as distĂąncias sĂŁo multiplicadas por $\abs a $. Para a forma:
\[
\frac{f(c') - f(a')}{f(b') - f(a')} = \frac{a(c' -
a')}{a(b' - a')} = \frac{c' - a'}{b' - a'} .
\]
Se dois triĂąngulos $(a', b', c')$ e $(a'', b'', c'')$ tem igual
formas, deixe $ f $ ser a Ășnica semelhanĂ§a direta com $ f(a') = a''$
e $ f(b') = b''$ (questĂŁo 7); entĂŁo a forma de $(a'', b'',
f(c'))$ Ă© igual ao de $(a', b', c')$, daĂ­ o de $(a'', b'',
c'')$, e a forma determina o terceiro vĂ©rtice do primeiro
dois: $ f(c') = c''$. Por outro lado, formas iguais decorrem do
invariĂąncia exibida.

\textbf{9.} $ b = f(0) = 1$; $ a + b = f(1) = \iu $ dĂĄ $ a = \iu -
1$. RazĂŁo $\abs a = \sqrt2$, Ăąngulo $\arg(\iu - 1) =
\frac{3\pi}4$. Ponto fixo:
\[
\zeta = \frac{b}{1 - a} = \frac1{2 - \iu} = \frac{2 + \iu}5 .
\]
EntĂŁo $ f $ Ă© a semelhanĂ§a direta do centro $\frac{2+\iu}5$, razĂŁo
$\sqrt2$ e Ăąngulo $\frac{3\pi}4$.

\textbf{10.} Com $\alpha + \beta = 1$ e $ f(z) = az + b $:
\[
f(\alpha a' + \beta b') = a\alpha a' + a\beta b' + b
= \alpha(a a' + b) + \beta(a b' + b)
= \alpha f(a') + \beta f(b') ,

\frac12$), centrĂłides (iterado) e os centros de NapoleĂŁo $\alpha b
+ \beta c $ abaixo sĂŁo, portanto, \emph{equivariante}: transformando o
triĂąngulo os transforma de acordo. DaĂ­ provar uma
invariante por similaridade \omterm{def:b1:logic:statement}{declaraĂ§ĂŁo} para um triĂąngulo bem colocado prova
para todos --- a licenĂ§a usada na questĂŁo 21.

\textbf{11.} O ĂĄpice externo em $[b, c]$ Ă© $ p_a = b +
\eu^{-\iu\pi/3}(c - b)$ (questĂŁo 3), entĂŁo o centro Ă©
\[

\textbf{12.} $\alpha + \beta = \frac{3 + \iu\sqrt3 + 3 -

\[

\]
os dois triĂąngulos compartilham seu centrĂłide.

\textbf{13.} Agrupar por $\alpha $ e $\beta $ e aplique a questĂŁo 4

\textbf{14.} Com $ j = \frac{-1 + \iu\sqrt3}2$:

\textbf{15.} O ĂĄpice interno Ă© $ b + \eu^{+\iu\pi/3}(c - b)$ e
