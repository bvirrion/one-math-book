\begin{solution}{exo:b1:curves:1}
$M(-t) = (t^2, -t^3)$: reflexﾃ｣o no $ x $-eixo; estudar $ t \geq 0$.
Ambos $ x' = 2t $ e $ y' = 3t^2$ sﾃ｣o $\geq 0$: o ramo se move
para a direita e para cima, de $(0,0)$ ao infinito. No $ t = 0$ a velocidade
desaparece; expansﾃｵes $ x = t^2$, $ y = t^3$ mostre que a tangente ﾃｩ
$ x $-eixo ($ y/x = t \to 0$) com $ y $ mudando de sinal enquanto $ x \geq 0$:
um \emph{\omterm{ex:b1:curves:astroid}{cﾃｺspide}} apontando para a esquerda. A curva ﾃｩ a parﾃ｡bola semicﾃｺbica
$ y^2 = x^3$.
\end{solution}

\begin{solution}{exo:b1:curves:2}
$ M(t + 2\pi) = M(t) + (2\pi, 0)$: a curva se repete, traduzida por
circunferﾃｪncia de uma roda; estudar um \omterm{def:b1:vspaces:span}{perﾃｭodo}. $ x'(t) = 1 - \cos t \geq
0$, $ y'(t) = \sin t $: o arco sobe $\intoo{0}{\pi}$, cai sobre
$\intoo{\pi}{2\pi}$, culminando em $(\pi, 2)$. Pontos singulares
onde $ x' = y' = 0$: $ t \in 2\pi\Z $, no \omterm{thm:b1:reals:floor}{chﾃ｣o}. Aproximar $ t = 0$:
\[
x(t) = \frac{t^3}{6} + o(t^3),
\qquad
y(t) = \frac{t^2}{2} + o(t^2):
\]
$ x $ muda de sinal, $ y \geq 0$, e $\frac{x}{y^{3/2}}$ delimitado: o
tangente ﾃｩ \emph{vertical} (direﾃｧﾃ｣o $(0,1)$), um \omterm{ex:b1:curves:astroid}{cﾃｺspide} onde o
o ponto rastreado tem momentaneamente velocidade zero --- a assinatura fﾃｭsica
de rolar sem escorregar.
\end{solution}

\begin{solution}{exo:b1:curves:3}
No $ t = \frac\pi4$: $ M = \bigl(\frac{\sqrt2}{4},

\begin{solution}{exo:b1:curves:7}
Foco na origem, diretriz $ D: x = d $ com $ d = \frac pe > 0$.
Por um ponto $ M = (r\cos\theta, r\sin\theta)$ com $ r > 0$:
\[
MF = r,
\qquad
d(M, D) = \abs{d - r\cos\theta},
\]
e o \omterm{def:b1:curves:conics}{cﾃｴnico} doenﾃｧa $ MF = e\,d(M, D)$, no regime em que $ M $
estﾃ｡ no lado do foco da diretriz ($ r\cos\theta < d $), lﾃｪ $ r
= e(d - r\cos\theta)$, ou seja
\[
r(1 + e\cos\theta) = ed = p,
\qquad
r = \frac{p}{1 + e\cos\theta} .
\]
Para $ e < 1$ o denominador nunca desaparece: todos $\theta $ permitido
(elipse). Para $ e = 1$: $\theta \neq \pi $ (parﾃ｡bola, aberta em direﾃｧﾃ｣o
o outro lado da diretriz). Para $ e > 1$: precisar $1 + e\cos\theta > 0$,
ou seja\ $\theta \in \intoo{-\theta_0}{\theta_0}$ com $\theta_0 =
\arccos\bigl(-\frac1e\bigr)$: um \emph{filial} da hipﾃｩrbole
(o outro ramo corresponde ﾃ escolha do sinal $ r < 0$, ou para o
segundo foco).
\end{solution}

\begin{solution}{exo:b1:curves:8}
\emph{Jardineiro:} anexe uma corda de \omterm{def:b1:curves:arclength}{comprimento} $2a $ para duas apostas $ F,
F'$ e mantenha-o esticado com o ponto de rastreamento $ M $: a restriﾃｧﾃ｣o ﾃｩ
exatamente $ MF + MF' = 2a $, entﾃ｣o a curva traﾃｧada ﾃｩ a elipse.

\emph{Propriedade de reflexﾃ｣o:} deixar $ t \mapsto M(t)$ seja regular
parametrizaﾃｧﾃ｣o e $ u(t) = \frac{M(t) - F}{\norm{M(t) - F}}$,
$ u'(t)$ o vetor unitﾃ｡rio anﾃ｡logo a $ F'$. Diferenciando
$\norm{M - F} + \norm{M - F'} = 2a $, usando $\frac{\dd}{\dd
t}\norm{M - F} = \bigl\langle \frac{M - F}{\norm{M-F}},\,
M'\bigr\rangle $ (regra da cadeia em $\sqrt{\langle\cdot,\cdot\rangle}$):
\[
\bigl\langle u + u',\, M'\bigr\rangle = 0 .
\]
Entﾃ｣o a direﾃｧﾃ｣o tangente $ M'$ ﾃｩ ortogonal para a bissetriz
direﾃｧﾃ｣o $ u + u'$ dos dois raios focais: a tangente faz igual
ﾃ｢ngulos com $ MF $ e $ MF'$. (Um raio de luz de um foco reflete
fora da elipse para o outro foco.)
\end{solution}

\begin{solution}{exo:b1:curves:9}
\begin{enumerate}
  \item Domﾃｭnio: $\cos 2\theta \geq 0$: $\theta \in
        \intcc{-\frac\pi4}{\frac\pi4} \cup
        \intcc{\frac{3\pi}{4}}{\frac{5\pi}{4}}$. Simetrias: $\theta
        \mapsto -\theta $ ($ x $-eixo) e $\theta \mapsto \pi -
        \theta $ ($ y $-eixo): estudo $\intcc{0}{\frac\pi4}$; $ r $
        diminui de $1$ para $0$. No $\theta = \pm\frac\pi4$: $ r =
        0$, tangentes na origem ao longo das diagonais. A curva ﾃｩ
        o $\infty $ sﾃｭmbolo: dois loops simﾃｩtricos reunidos em $ O $,
        alcanﾃｧando $(\pm 1, 0)$.
  \item Com $ F, F' = (\pm c, 0)$, $ c = \frac{1}{\sqrt 2}$, e $ M
        = (r\cos\theta, r\sin\theta)$:
        \[
        MF^2\, MF'^2
        = \bigl((r\cos\theta - c)^2 + r^2\sin^2\theta\bigr)
        \bigl((r\cos\theta + c)^2 + r^2\sin^2\theta\bigr)
        = (r^2 + c^2)^2 - (2rc\cos\theta)^2 ,
        \]
        pela identidade $(A - B)(A + B) = A^2 - B^2$ com $ A = r^2
        + c^2$, $ B = 2rc\cos\theta $. Com $ c^2 = \frac12$:
        \[
        MF^2 MF'^2 = \Bigl(r^2 + \frac12\Bigr)^2 - 2r^2\cos^2\theta
        = r^4 + r^2\bigl(1 - 2\cos^2\theta\bigr) + \frac14
        = r^4 - r^2\cos 2\theta + \frac14 .
        \]
        Na lemniscata, $ r^2 = \cos 2\theta $: os dois primeiros termos
        cancelar, sair $ MF^2MF'^2 = \frac14$, ou seja\ $ MF \cdot MF' =
        \frac12$. Por outro lado, o cﾃ｡lculo lido de trﾃ｡s para frente mostra
        a equaﾃｧﾃ｣o do lugar geomﾃｩtrico $ MF\,MF' = \frac12$ ﾃｩ $ r^2 = \cos
        2\theta $ (para $ r \neq 0$; e $ O $ satisfaz ambos).
\end{enumerate}
\end{solution}

\begin{solution}{exo:b1:curves:10}
$ r = 1 + \cos\theta $, $ r' = -\sin\theta $:
\[
r^2 + r'^2 = 1 + 2\cos\theta + \cos^2\theta + \sin^2\theta

\end{solution}

\begin{solution}{exo:b1:curves:11}
O ponto $(a\cos t, b\sin t)$ satisfaz a equaﾃｧﾃ｣o:
$\cos^2 t + \sin^2 t = 1$. O vetor normal da reta ﾃｩ
$\bigl(\frac{\cos t}a, \frac{\sin t}b\bigr)$, e seu produto
com a velocidade $(-a\sin t, b\cos t)$ ﾃｩ $-\sin t\cos t +
\sin t\cos t = 0$: a reta passa pelo ponto com o
direﾃｧﾃ｣o tangente --- ﾃｩ a tangente. Para $(x_0, y_0)$ no
elipse, escreva $\cos t = \frac{x_0}a $, $\sin t = \frac{y_0}b $ e
substituto:
\[
\frac{x\,x_0}{a^2} + \frac{y\,y_0}{b^2} = 1 ,
\]
obtido da equaﾃｧﾃ｣o da elipse por ``\omterm{thm:b1:logic:partition}{divisﾃ｣o}'' $ x^2 \mapsto
x\,x_0$ e $ y^2 \mapsto y\,y_0$.
\end{solution}

\begin{solution}{exo:b1:curves:12}
\begin{enumerate}
  \item $ r' = k\eu^{k\theta}$, entﾃ｣o $\tan V = \frac{r}{r'} =
        \frac1k $: constante. A espiral corta todos os raios do
        origem no mesmo ﾃ｢ngulo $ V = \arctan\frac1k $.
  \item Em notaﾃｧﾃ｣o complexa, a espiral ﾃｩ $\{\eu^{k\theta}

        = \eu^{-kc}\,\eu^{k(\theta + c)}\eu^{\iu(\theta+c)} ,

        outra curva suave, mas linhas e cﾃｭrculos tem isso

\end{enumerate}
\end{solution}

\begin{solution}{pb:b1:curves:1}
\textbf{1.} Rolar sem escorregar significa que o ponto de contato
percorreu uma distﾃ｢ncia igual ao arco da roda desenrolada: depois
passando $ t $, o centro estﾃ｡ em $(Rt, R)$. O ponto marcado fica
no aro em ﾃ｢ngulo $ t $ \emph{atrﾃ｡s} a vertical descendente
(a roda gira no sentido horﾃ｡rio enquanto avanﾃｧa):
\[
M(t) = \Omega(t) + R(-\sin t, -\cos t)
= \bigl(R(t - \sin t),\ R(1 - \cos t)\bigr),
\]
o que estﾃ｡ correto em $ t = 0$ ($ M = (0,0)$) e em $ t = \pi $ ($ M =
(\pi R, 2R)$, o topo).

\textbf{2.} $ f'(t) = R(1 - \cos t,\ \sin t)$ e
\[
\norm{f'}^2 = R^2\bigl((1-\cos t)^2 + \sin^2 t\bigr)
= 2R^2(1 - \cos t) = 4R^2\sin^2\frac t2 ,
\]
entﾃ｣o $\norm{f'} = 2R\abs{\sin\frac t2}$. Pontos singulares em $ t \in
2\pi\Z $: o \omterm{ex:b1:curves:astroid}{cﾃｺspides} no \omterm{thm:b1:reals:floor}{chﾃ｣o} (\cref{exo:b1:curves:2}); o
topo do arco ﾃｩ $ t = \pi $, onde a velocidade $2R $ ﾃｩ mﾃ｡ximo.

\textbf{3.} Fatoraﾃｧﾃｵes de meio ﾃ｢ngulo:
\[
f'(t) = 2R\sin\frac t2\,\Bigl(\sin\frac t2,\ \cos\frac t2\Bigr),
\quad
M - C = 2R\sin\frac t2\,\Bigl(-\cos\frac t2,\ \sin\frac t2\Bigr),
\]
\[
T - M = \bigl(R\sin t,\ R(1 + \cos t)\bigr)
= 2R\cos\frac t2\,\Bigl(\sin\frac t2,\ \cos\frac t2\Bigr).
\]
Para $0 < t < 2\pi $, $\sin\frac t2 \neq 0$: $ M - C $ ﾃｩ ortogonal
para a direﾃｧﾃ｣o tangente $\bigl(\sin\frac t2, \cos\frac
t2\bigr)$ (seu produto ﾃｩ $-\sin\frac t2\cos\frac t2 +
\sin\frac t2\cos\frac t2 = 0$), e $ T - M $ ﾃｩ paralelo a ele.
O acorde no topo da roda \emph{ﾃｩ} a tangente; o
corda ao ponto de contato ﾃｩ normal.

\textbf{4.} A cada instante a roda gira em torno do seu contato
ponto (esse ponto tem velocidade zero: rolando sem escorregar),
entﾃ｣o cada ponto rﾃｭgido da roda se move ortogonalmente para a linha
juntando-se a isso $ C $, com velocidade (velocidade angular $1$) igual a
essa distﾃ｢ncia. Verificar: $\norm{M - C} = 2R\abs{\sin\frac t2} =
\norm{f'(t)}$.

\textbf{5.} $2R\,y(t) = 2R\cdot 2R\sin^2\frac t2 =
4R^2\sin^2\frac t2 = \norm{f'(t)}^2$. A velocidade em altura $ y $ ﾃｩ
$\sqrt{2R\,y}$ --- formalmente a lei $ v = \sqrt{2gh}$ de \omterm{def:b1:vspaces:free}{livre}
cair, com o \omterm{thm:b1:reals:floor}{chﾃ｣o} tocando o teto; A Parte IV transforma isso
observaﾃｧﾃ｣o em um relﾃｳgio.

\textbf{6.} Por \cref{def:b1:curves:arclength} e pergunta 2
($\sin\frac t2 \geq 0$ sobre $\intcc{0}{2\pi}$):
\[
L = \int_0^{2\pi} 2R\sin\frac t2\,\dd t
= 2R\Bigl[-2\cos\frac t2\Bigr]_0^{2\pi} = 2R(2 + 2) = 8R .
\]
Teorema de Wren: exatamente quatro diﾃ｢metros.

\textbf{7.} $ s(t) = \int_0^t 2R\sin\frac u2\,\dd u =
4R\bigl(1 - \cos\frac t2\bigr)$; $ s(2\pi) = 4R(1+1) = 8R $,
consistente.

\textbf{8.} $\sigma(t) = \abs{s(t) - s(\pi)} = \abs{4R(1 -
\cos\frac t2) - 4R} = 4R\abs{\cos\frac t2}$, entﾃ｣o $\sigma^2 =
16R^2\cos^2\frac t2$; e $2R - y = R(1 + \cos t) =
2R\cos^2\frac t2$, de onde $8R(2R - y) = 16R^2\cos^2\frac t2 =
\sigma^2$.

\textbf{9.} No \omterm{ex:b1:curves:astroid}{cﾃｺspide} $ t = 0$ (ou $2\pi $): $\sigma = 4R $, metade
de $8R $: o ﾃ｡pice \omterm{def:b1:arith:divides}{divide} o arco pela metade. Em ambos os cﾃ｡lculos o
velocidade ﾃｩ $\abs{\text{trigonometric polynomial in } t/2}$, cujo
a antiderivada ﾃｩ novamente trigonomﾃｩtrica: o \omterm{def:b1:curves:arclength}{comprimento} ﾃｩ um
diferenﾃｧa de \emph{valores} de cossenos --- nﾃｺmeros racionais
vezes $ R $ --- sem arco de cﾃｭrculo para medir, portanto nﾃ｣o $\pi $.
O cﾃｭrculo em si tem velocidade constante, entﾃ｣o ﾃｩ \omterm{def:b1:curves:arclength}{comprimento} integrante
produz o completo \omterm{prop:b1:reals:intervals}{intervalo} \omterm{def:b1:curves:arclength}{comprimento} $2\pi $; a velocidade da ciclﾃｳide
desaparece nas extremidades e integra-se algebricamente.

\textbf{10.} O arco ﾃｩ varrido com $ x $ aumentando de $0$ para
$2\pi R $ ($ x' = R(1 - \cos t) \geq 0$, desaparecendo apenas em
pontos isolados). Substituindo $ x = x(t)$ na ﾃ｡rea integrante
$\int_0^{2\pi R} y\,\dd x $ (\cref{thm:b1:integration:parts})
dﾃ｡ $ A = \int_0^{2\pi} y(t)\,x'(t)\,\dd t $.

\textbf{11.}
\[
A = \int_0^{2\pi} R(1 - \cos t)\cdot R(1 - \cos t)\,\dd t
= R^2\int_0^{2\pi}\bigl(1 - 2\cos t + \cos^2 t\bigr)\dd t
= R^2\Bigl(2\pi - 0 + \pi\Bigr) = 3\pi R^2 ,
\]
usando $\int_0^{2\pi}\cos^2 = \pi $. Exatamente trﾃｪs ﾃ｡reas de roda:
O balanﾃｧo de Galileu dizia 窶徭obre $3$''; Cﾃ｡lculo de Roberval
diz ``exatamente''.

\textbf{12.} Com $ x = \cos^3 t $, $ y = \sin^3 t $: $ y\,x' =
-3\sin^4 t\cos^2 t $. Linearizar:
\[
\sin^4 t\cos^2 t = \frac{1 - \cos 2t}{2}\cdot\frac{\sin^2
2t}{4} = \frac{\sin^2 2t}{8} - \frac{\sin^2 2t\cos 2t}{8} ,
\]
e mais $\intcc{0}{2\pi}$: $\int\sin^2 2t = \pi $, $\int \sin^2
2t\cos 2t = \bigl[\frac{\sin^3 2t}{6}\bigr] = 0$. Entﾃ｣o $\oint
y\,\dd x = -3\cdot\frac\pi8$, e a ﾃ｡rea fechada ﾃｩ
$\frac{3\pi}8$ (o sinal registra o sentido anti-horﾃ｡rio
orientaﾃｧﾃ｣o).

\textbf{13.} Para $(\cos t, \sin t)$ com $ t $ de $\pi $ para $0$,
$ x $ aumenta de $-1$ para $1$ e
\[
\int_\pi^0 \sin t\cdot(-\sin t)\,\dd t
= \int_0^\pi \sin^2 t\,\dd t = \frac\pi2 :
\]
a fﾃｳrmula retorna a ﾃ｡rea do meio disco superior, como deveria.

\textbf{14.} $ x' = R(1 + \cos t) = 2R\cos^2\frac t2$, $ y' =
R\sin t = 2R\sin\frac t2\cos\frac t2$, entﾃ｣o $\norm{f'} =
2R\cos\frac t2$ (nﾃ｣o negativo em $\intcc{-\pi}{\pi}$). Arco de
a parte inferior: $ s(t) = \int_0^t 2R\cos\frac u2\,\dd u =
4R\sin\frac t2$. Entﾃ｣o
\[
y = R(1 - \cos t) = 2R\sin^2\frac t2
= 2R\Bigl(\frac{s}{4R}\Bigr)^{\!2} = \frac{s^2}{8R} .
\]

\textbf{15.} Conservaﾃｧﾃ｣o de energia com $ v = \frac{\dd s}{\dd\tau}$
e $ y = \frac{s^2}{8R}$, $ y_0 = \frac{s_0^2}{8R}$:
\[
\Bigl(\frac{\dd s}{\dd\tau}\Bigr)^{\!2} = 2g(y_0 - y)
= \frac{2g}{8R}\bigl(s_0^2 - s^2\bigr)
= \frac{g}{4R}\bigl(s_0^2 - s^2\bigr).
\]

\textbf{16.} Diferenciando em $\tau $: $2s's'' =
-\frac{g}{4R}\,2ss'$, entﾃ｣o onde quer que $ s' \neq 0$ (portanto em todos os lugares
por continuidade), $ s'' = -\frac{g}{4R}s $: o oscilador harmﾃｴnico.
Por \cref{thm:b1:diffeq:homogeneous2}, $ s(\tau) =
