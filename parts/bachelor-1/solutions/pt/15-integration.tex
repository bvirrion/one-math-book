\begin{solution}{exo:b1:integration:1}
$\dfrac{1}{x^2-4} = \dfrac{1/4}{x - 2} - \dfrac{1/4}{x+2}$
(encobrimento), entĂŁo
\[
\int_0^1 \frac{\dd x}{x^2 - 4}
= \frac14\Bigl[\ln\abs{x-2} - \ln\abs{x+2}\Bigr]_0^1
= \frac14\Bigl(\ln\frac{1}{3} - \ln 1\Bigr) = -\frac{\ln 3}{4}.
\]

Por partes ($ u' = 1$, $ v = \ln t $): $\int_1^{\eu} \ln t\,\dd t =
\bigl[t\ln t\bigr]_1^{\eu} - \int_1^{\eu} \dd t = \eu - (\eu - 1) =
1$.

Por partes ($ u' = \sin t $, $ v = t $): $\int_0^\pi t\sin t\,\dd t =
\bigl[-t\cos t\bigr]_0^\pi + \int_0^\pi \cos t\,\dd t = \pi + 0 =
\pi $.
\end{solution}

\begin{solution}{exo:b1:integration:2}
SubstituiĂ§ĂŁo $ u = t^2 + 1$, $\dd u = 2t\,\dd t $:
\[
\int_0^1 \frac{t\,\dd t}{(t^2+1)^2}
= \frac12 \int_1^2 \frac{\dd u}{u^2}
= \frac12\Bigl[-\frac1u\Bigr]_1^2 = \frac14 .
\]
Com $ u = \sin t $: $\int_0^{\pi/2} \cos^3 t\,\dd t = \int_0^{\pi/2}
(1 - \sin^2 t)\cos t\,\dd t = \bigl[\sin t - \frac{\sin^3
t}{3}\bigr]_0^{\pi/2} = 1 - \frac13 = \frac23$.
\end{solution}

\begin{solution}{exo:b1:integration:3}
$ u_n = \dfrac1n \sum_{k=1}^{n} \dfrac{1}{1 + (k/n)^2}$: uma soma de Riemann
de $ x \mapsto \frac{1}{1+x^2}$ sobre $\intcc{0}{1}$, entĂŁo $ u_n \to
\int_0^1 \frac{\dd x}{1+x^2} = \arctan 1 = \dfrac\pi4$.

$\ln v_n = \dfrac1n \sum_{k=1}^{n} \ln\dfrac{n+k}{n} = \dfrac 1n
\sum_{k=1}^n \ln\Bigl(1 + \dfrac kn\Bigr) \to \int_0^1 \ln(1+x)\,\dd
x = \bigl[(1+x)\ln(1+x) - x\bigr]_0^1 = 2\ln 2 - 1$. Por isso $ v_n \to
\eu^{2\ln 2 - 1} = \dfrac 4\eu $. (VerificaĂ§ĂŁo da identificaĂ§ĂŁo:
$\frac{(2n)!}{n!\,n^n} = \prod_{k=1}^{n} \frac{n+k}{n}$.)
\end{solution}

\begin{solution}{exo:b1:integration:4}
O \omterm{def:b1:topology:closure}{limite} ĂŠ $0$. Deixar $ M = \sup \abs f $ e $\varepsilon \in
\intoo{0}{1}$. Cortar em $1 - \varepsilon $:
\[
\Bigl| \int_0^1 x^n f \Bigr|
\leq \int_0^{1 - \varepsilon} x^n \abs f + \int_{1-\varepsilon}^1
x^n \abs f
\leq M\,(1-\varepsilon)^n + M\varepsilon .
\]
Desde $(1 - \varepsilon)^n \to 0$ (\cref{exo:b1:seq:3}), o limsup
do lado esquerdo ĂŠ $\leq M\varepsilon $ para cada $\varepsilon $:
o integrante tende a $0$.
\end{solution}

\begin{solution}{exo:b1:integration:5}
$ Q(\lambda) = \int_a^b (f + \lambda g)^2 = \int f^2 + 2\lambda \int
fg + \lambda^2 \int g^2 \geq 0$ para todos $\lambda $. Se $\int g^2 =
0$, entĂŁo $ g = 0$ (positividade estrita,
\cref{thm:b1:integration:props}~(4)) e a desigualdade ĂŠ $0 \leq
0$. De outra forma $ Q $ ĂŠ uma quadrĂĄtica genuĂ­na, em todos os lugares $\geq 0$: ĂŠ
discriminante ĂŠ $\leq 0$, ou seja\ $\bigl(\int fg\bigr)^2 \leq \int
f^2 \int g^2$.

Igualdade se o discriminante desaparecer se $ Q(\lambda_0) = 0$ para
alguns $\lambda_0$, ou seja\ $\int (f + \lambda_0 g)^2 = 0$, ou seja,\ (estrito
positividade novamente) $ f = -\lambda_0 g $: a igualdade vale exatamente quando
$ f $ e $ g $ sĂŁo proporcionais.
\end{solution}

\begin{solution}{exo:b1:integration:6}
Deixar $\Phi(a) = \int_a^{a+T} f $. Pelo teorema fundamental
(\cref{thm:b1:integration:ftc}), $\Phi $ ĂŠ diferenciĂĄvel com
$\Phi'(a) = f(a + T) - f(a) = 0$: constante.

Para $ x > 0$, escrever $ x = nT + r $, $0 \leq r < T $ ($ n = \lfloor x/T
\rfloor $). Castelos:
\[
\int_0^x f = n \int_0^T f + \int_{nT}^{nT + r} f,
\qquad
\Bigl| \int_{nT}^{nT+r} f \Bigr| \leq T \sup_{\intcc{0}{T}} \abs f
= C .
\]
EntĂŁo $\frac 1x \int_0^x f = \frac{nT}{x}\cdot\frac 1T \int_0^T f +
O\bigl(\frac 1x\bigr)$, e $\frac{nT}{x} \to 1$: o \omterm{def:b1:topology:closure}{limite} ĂŠ
$\frac1T \int_0^T f $.
\end{solution}

\begin{solution}{exo:b1:integration:7}

\end{solution}

\begin{solution}{exo:b1:integration:8}
\begin{enumerate}
  \item Por partes com $ u' = \sin t $, $ v = \sin^{n-1} t $:
        \[
        W_n = \bigl[-\cos t\sin^{n-1}t\bigr]_0^{\pi/2}
        + (n-1)\int_0^{\pi/2} \cos^2 t\,\sin^{n-2} t\,\dd t
        = (n-1)(W_{n-2} - W_n),
        \]
        entĂŁo $ nW_n = (n-1)W_{n-2}$. De $ W_0 = \frac\pi2$, $ W_1 = 1$:
        \[
        W_{2p} = \frac{(2p-1)(2p-3)\cdots 1}{(2p)(2p-2)\cdots 2}\,
        \frac{\pi}{2} = \frac{(2p)!}{4^p (p!)^2}\,\frac\pi2,
        \qquad
        W_{2p+1} = \frac{(2p)(2p-2)\cdots 2}{(2p+1)(2p-1)\cdots 3}
        = \frac{4^p (p!)^2}{(2p+1)!} .
        \]
  \item Sobre $\intoo{0}{\frac\pi2}$, $0 < \sin t < 1$, entĂŁo $\sin^{n+1} <
        \sin^n $ e $(W_n)$ ĂŠ (estritamente) decrescente, positivo.
        Sanduichando com a recorrĂŞncia:
        \[
        \frac{n}{n+1} = \frac{W_{n+1}}{W_{n-1}}
        \leq \frac{W_{n+1}}{W_n} \leq 1
        \quad\implies\quad \frac{W_{n+1}}{W_n} \to 1 .
        \]
        Invariante: $ a_n = (n+1)W_{n+1}W_n $ satisfaz $ a_n = a_{n-1}$
        pela recorrĂŞncia $(n+1)W_{n+1} = nW_{n-1}$, entĂŁo $ a_n = a_0 =
        1 \cdot W_1 W_0 = \frac\pi2$. EntĂŁo
        \[
        n W_n^2 \sim (n+1) W_{n+1} W_n = \frac\pi2
        \quad\implies\quad
        W_n \sim \sqrt{\frac{\pi}{2n}} .
        \]
\end{enumerate}
\end{solution}

\begin{solution}{exo:b1:integration:9}
\begin{enumerate}
  \item Sobre $\intoo{0}{\pi}$: $ x > 0$, $ a - bx = b(\frac ab - x) =
        b(\pi - x) > 0$ e $\sin x > 0$, entĂŁo o integrando ĂŠ $> 0$
        e $ I_n > 0$ (positividade estrita). \omterm{def:b1:topology:closure}{Limite}: ativado
        $\intcc{0}{\pi}$, $ x \leq \pi $ e $ a - bx \leq a $, entĂŁo $ P
        \leq \frac{\pi^n a^n}{n!}$ e $ I_n \leq
        \pi\,\frac{(\pi a)^n}{n!}$, o que tende a $0$ (o
        fatorial vence o termo geomĂŠtrico: ĂŠ o termo geral
        da sĂŠrie exponencial convergente, cf.\
        \cref{ex:b1:seq:e}); em particular $ I_n < 1$ para grande $ n $.
  \item Expandir $ x^n(a - bx)^n = \sum_{j=0}^{n} \binom nj a^{n-j}
        (-b)^j x^{n+j}$: entĂŁo $ P = \frac{1}{n!}\sum_j c_j x^{n+j}$
        com nĂşmero inteiro $ c_j $. EntĂŁo $ P^{(k)}(0) = 0$ para $ k < n $
        (avaliaĂ§ĂŁo) e, para $ n \leq k \leq 2n $, $ P^{(k)}(0) =
        \frac{k!}{n!} c_{k-n}$, um nĂşmero inteiro desde $ n! \mid k!$.
        AlĂŠm disso $ P(\pi - x) = P(x)$ (substituir: $\pi - x $ trocas
        os fatores, usando $ a - b(\pi - x) = bx $), entĂŁo $ P^{(k)}(\pi)
        = \pm P^{(k)}(0)$: inteiros tambĂŠm.
  \item Com $ Q = P - P'' + P^{(4)} - \dots $ (finito: $ P $ tem diploma
        $2n $): $ Q + Q'' = P $, e
        \[
        \bigl(Q'\sin x - Q\cos x\bigr)' = (Q + Q'')\sin x = P\sin x .
        \]
        Por isso $ I_n = \bigl[Q'\sin x - Q\cos x\bigr]_0^{\pi} = Q(\pi)
        + Q(0)$, uma soma de valores $ P^{(2k)}$ no $0$ e $\pi $: um
        inteiro por (2).
  \item Para $ n $ grande, $ I_n $ ĂŠ um nĂşmero inteiro com $0 < I_n < 1$:
        impossĂ­vel. A suposiĂ§ĂŁo $\pi = \frac ab $ falha: $\pi $ ĂŠ
        irracional.
\end{enumerate}
\end{solution}

\begin{solution}{exo:b1:integration:10}
$ C^1$ caso: por partes,
\[
\int_a^b f(t)\sin\lambda t\,\dd t
= \Bigl[-f(t)\frac{\cos\lambda t}{\lambda}\Bigr]_a^b
+ \frac{1}{\lambda}\int_a^b f'(t)\cos\lambda t\,\dd t,
\]
limitado em valor absoluto por $\frac{2\sup\abs f + (b - a)\sup\abs{f'}}
{\lambda} \to 0$.

ContĂ­nuo caso: deixe $\varepsilon > 0$ e escolha um funĂ§ĂŁo de passo
$\varphi $ com $\abs{f - \varphi} \leq \varepsilon $
(\cref{thm:b1:integration:approx} fornece $\varphi \leq f \leq
\psi $ com lacuna $\leq\varepsilon $; pegar $\varphi $). EntĂŁo
\[
\Bigl|\int_a^b f\sin\lambda t\Bigr|
\leq \int_a^b \abs{f - \varphi}
+ \Bigl|\int_a^b \varphi \sin\lambda t\Bigr|
\leq (b-a)\varepsilon
+ \sum_i \abs{c_i}\,\Bigl|\int_{x_{i-1}}^{x_i} \sin\lambda t\,\dd
t\Bigr| ,
\]
e cada um $\bigl|\int \sin \lambda t\bigr| = \bigl|\frac{\cos\lambda
x_{i-1} - \cos\lambda x_i}{\lambda}\bigr| \leq \frac{2}{\lambda}$:
o segundo termo tende a $0$. Portanto, o limsup ĂŠ $\leq
(b-a)\varepsilon $ para cada $\varepsilon $: o \omterm{def:b1:topology:closure}{limite} ĂŠ $0$.
\end{solution}

\begin{solution}{exo:b1:integration:11}
Deixar $ m = \min f $ e $ M = \max f $, alcanĂ§ado pelo valor extremo
teorema. Desde $ g \geq 0$: $ m\,g \leq fg \leq M\,g $, entĂŁo por
monotonicidade
\[
m \int_a^b g \;\leq\; \int_a^b fg \;\leq\; M \int_a^b g .
\]
Se $\int_a^b g = 0$: positividade estrita
(\cref{thm:b1:integration:props}~(4)) forĂ§as $ g \equiv 0$, ambos
lados desaparecem e qualquer $ c $ funciona. De outra forma $ t = \frac{\int
fg}{\int g}$ reside em $\intcc{m}{M} = f(\intcc{a}{b})$
(\cref{thm:b1:continuity:evt,thm:b1:continuity:ivt}), entĂŁo $ t =
f(c)$ para alguns $ c $.

Assuntos de sinalizaĂ§ĂŁo: ativado $\intcc{-1}{1}$ com $ f(t) = g(t) = t $: $\int
fg = \int_{-1}^1 t^2 = \frac23$, enquanto $ f(c)\int_{-1}^1 t\,\dd t
= 0$ para cada $ c $.
\end{solution}

\begin{solution}{exo:b1:integration:12}
Se $ f \equiv 0$ a afirmaĂ§ĂŁo ĂŠ vazia (cada ponto ĂŠ zero). EntĂŁo
assumir $ f \not\equiv 0$ e suponha que tenha no mĂĄximo $ n $ distinto
zeros em $\intoo{a}{b}$; deixar $ z_1 < \dots < z_m $ ($ m \leq n $) ser
aqueles zeros onde $ f $ \emph{sinal de mudanĂ§as} (possivelmente nenhum). \omterm{def:b1:logic:sets}{Definir}
$ P(t) = \prod_{i=1}^{m}(t - z_i)$ (produto vazio $= 1$), de
grau $ m \leq n $. Em cada subintervalo cortado pelo $ z_i $, ambos
$ f $ e $ P $ tĂŞm sinal constante e ambos invertem o sinal ao cruzar
alguns $ z_i $: o produto $ fP $ tem um sinal constante em todos
$\intoo{a}{b}$. Ser contĂ­nuo, nĂŁo identicamente zero, de
sinal constante, tem $\bigl|\int_a^b fP\bigr| > 0$ (estrito
positividade aplicada a $\abs{fP}$). Mas $\int f P $ ĂŠ linear
combinaĂ§ĂŁo dos momentos $\int f\,t^k $, $ k \leq n $, todos zero:
contradiĂ§ĂŁo. Por isso $ f $ tem pelo menos $ n + 1$ zeros distintos em
$\intoo{a}{b}$.
\end{solution}

\begin{solution}{pb:b1:integration:1}
\textbf{1.} Deixar $ N = \lceil 2c \rceil $. Para $ n \geq N $:
$\frac{c^{n+1}/(n+1)!}{c^n/n!} = \frac{c}{n+1} \leq \frac12$, entĂŁo
$0 < \frac{c^n}{n!} \leq \frac{c^N}{N!}\,2^{-(n - N)} \to 0$:
espremer.

\textbf{2.} Consertar $ k $; induĂ§ĂŁo ligada $ m $. Para $ m = 0$: $\int_0^1
x^k = \frac{1}{k+1} = \frac{k!\,0!}{(k+1)!}$. Passo a passo, por partes
($ u = (1-x)^m $, $ v' = x^k $):
\[
\int_0^1 x^k (1-x)^m \dd x
= \Bigl[\frac{x^{k+1}}{k+1}(1-x)^m\Bigr]_0^1
+ \frac{m}{k+1}\int_0^1 x^{k+1}(1-x)^{m-1}\dd x
= \frac{m}{k+1}\cdot\frac{(k+1)!\,(m-1)!}{(k+m+1)!} ,
\]
o que ĂŠ $\frac{k!\,m!}{(k+m+1)!}$.

\textbf{3.} $ k = m = n $: $\int_0^1 (x(1-x))^n =
\frac{(n!)^2}{(2n+1)!} = \frac{1}{(2n+1)\binom{2n}{n}}$. Desde
$ x(1-x) \leq \frac14$ sobre $\intcc{0}{1}$, o integrante ĂŠ $\leq
4^{-n}$, de onde $\binom{2n}{n} \geq \frac{4^n}{2n+1}$ ---
consistente com $\binom{2n}{n}^{1/n} \to 4$
(\cref{pb:b1:seq:1}).

\textbf{4.} O integrando ĂŠ contĂ­nuo, $\geq 0$ e positivo
ligado $\intoo{0}{1}$, portanto, nĂŁo ĂŠ identicamente zero: ĂŠ integrante ĂŠ
$> 0$ (\cref{thm:b1:integration:props}~(4)). \omterm{def:b1:topology:closure}{Limite} superior:
$(x(1-x))^n \leq 4^{-n}$ e $ g \leq \sup\abs g $, entĂŁo
monotonicidade.

\textbf{5.} $ A_0 = \eu - 1$; $ A_1 = [x\eu^x]_0^1 - \int_0^1
\eu^x = \eu - (\eu - 1) = 1$. Por partes: $ A_n = [x^n \eu^x]_0^1 -
n\int_0^1 x^{n-1}\eu^x = \eu - n A_{n-1}$. \omterm{def:b1:topology:closure}{Limites}: o integrando
ĂŠ positivo, entĂŁo $ A_n > 0$; e $\eu^x \leq \eu $ dĂĄ $ A_n \leq
\eu\int_0^1 x^n = \frac{\eu}{n+1}$.

\textbf{6.} $ A_0 = -1 + 1\cdot\eu $. Se $ A_{n-1} = \alpha_{n-1} +
\beta_{n-1}\eu $ com entradas inteiras, entĂŁo
\[
A_n = \eu - n\alpha_{n-1} - n\beta_{n-1}\eu
= \underbrace{(-n\,\alpha_{n-1})}_{\alpha_n}
+ \underbrace{(1 - n\,\beta_{n-1})}_{\beta_n}\,\eu ,
\]
ambos os nĂşmeros inteiros.

\textbf{7.} Se $\eu = \frac pq $: $ q A_n = q\alpha_n + p\beta_n
\in \Z $, e $0 < qA_n \leq \frac{q\eu}{n+1} < 1$ para $ n $ grande:
um nĂşmero inteiro estritamente entre $0$ e $1$ --- impossĂ­vel. EntĂŁo $\eu
\notin \Q $. Em \cref{exo:b1:seq:9} o inteiro preso foi
$ q!\,\frac pq - q!\,a_q $; aqui estĂĄ $ qA_n $: mesma armadilha,
integrante combustĂ­vel.

\textbf{8.} $ n = 0$: $ R_0 = \int_0^1 \eu^t = \eu - 1$, entĂŁo $\eu =
1 + R_0$. Por partes ($ u = \eu^t $, $ v = -\frac{(1-t)^{n+1}}{n+1}$):
\[
R_n = \frac{1}{n!}\Bigl(\Bigl[-\frac{(1-t)^{n+1}}{n+1}
\eu^t\Bigr]_0^1 + \frac{1}{n+1}\int_0^1 (1-t)^{n+1}\eu^t\Bigr)
= \frac{1}{(n+1)!} + R_{n+1} ,
\]
entĂŁo a fĂłrmula se propaga de $ n $ para $ n + 1$. \omterm{def:b1:topology:closure}{Limites}: $1 \leq
\eu^t \leq \eu $ sobre $\intcc{0}{1}$ e $\int_0^1 (1-t)^n =
\frac{1}{n+1}$ dar $\frac{1}{(n+1)!} \leq R_n \leq
\frac{\eu}{(n+1)!}$.

\textbf{9.} $\sum_{k=0}^{9} \frac{1}{k!} = \frac{986410}{362880}
= 2.71828152\dots $, e
\[
\frac{1}{10!} = 2.76\cdot10^{-7} \leq R_9 \leq
\frac{2.75}{10!} = 7.58\cdot10^{-7} ,
\]
entĂŁo $2.7182818 \leq \eu \leq 2.7182823$: com prova, $\eu =
2.718282$ com seis casas decimais (valor verdadeiro $2.7182818\dots $).

\textbf{10.} $ f(1 - x) = \frac{(1-x)^n x^n}{n!} = f(x)$. Sobre
$\intoo{0}{1}$: $0 < x(1-x) \leq \frac14$, entĂŁo $0 < f \leq
\frac{1}{4^n\,n!}$.

\textbf{11.} $ x^n(1-x)^n = \sum_{j=0}^{n} (-1)^j\binom
nj\,x^{n+j}$, entĂŁo $ f = \frac{1}{n!}\sum_j c_j\,x^{n+j}$ com
$ c_j \in \Z $. Por isso $ f^{(k)}(0) = 0$ para $ k < n $ ou $ k > 2n $,
e para $ n \leq k \leq 2n $: $ f^{(k)}(0) = \frac{k!}{n!}\,
c_{k-n}$, um nĂşmero inteiro porque $ n! \mid k!$. A simetria dĂĄ
$ f^{(k)}(1) = (-1)^k f^{(k)}(0) \in \Z $.

\textbf{12.} $ b^n \pi^{2n-2k} = b^n\bigl(\frac
ab\bigr)^{\,n-k} = a^{\,n-k}\,b^{\,k} \in \Z $, entĂŁo $ G(0) =
\sum_k (-1)^k a^{n-k} b^k f^{(2k)}(0)$ e da mesma forma $ G(1)$ sĂŁo
inteiros pela questĂŁo 11.

\textbf{13.} Em $\pi^2 G + G''$, O termo $ k $ de $\pi^2 G $
carrega $\pi^{2n-2k+2}f^{(2k)}$ e o termo $ j = k - 1$ de
$ G''$ carrega $(-1)^{k-1}\pi^{2n-2k+2}f^{(2k)}$: tudo
cancela exceto $ k = 0$ na primeira soma e $ j = n $ no
segundo, ou seja,
\[
G'' + \pi^2 G = b^n\bigl(\pi^{2n+2} f + (-1)^n f^{(2n+2)}\bigr)
= b^n \pi^{2n+2} f = \pi^2 a^n f
\]
($ f $ tem diploma $2n $, entĂŁo $ f^{(2n+2)} = 0$; e $ b^n\pi^{2n} =
a^n $). EntĂŁo
\[
\bigl(G'\sin\pi x - \pi G\cos\pi x\bigr)'
= (G'' + \pi^2 G)\sin \pi x = \pi^2 a^n f\sin\pi x .
\]

\textbf{14.} Integrando mais $\intcc{0}{1}$:
\[
\pi^2 a^n \int_0^1 f\sin(\pi x)\,\dd x
= \bigl[G'\sin\pi x - \pi G\cos\pi x\bigr]_0^1
= \pi\bigl(G(1) + G(0)\bigr) ,
\]
entĂŁo $\pi a^n \int_0^1 f\sin\pi x = G(0) + G(1) \in \Z $. Sobre
$\intoo{0}{1}$, $ f > 0$ e $\sin\pi x > 0$: o lado esquerdo ĂŠ
positivo, entĂŁo $ G(0) + G(1) \geq 1$; e $\sin \leq 1$ com
a questĂŁo 10 limita por $\frac{\pi a^n}{4^n\,n!}$.

\textbf{15.} Pela pergunta 1 (com $ c = \frac a4$),
$\frac{\pi a^n}{4^n n!} \to 0$: para grandes $ n $ isso ĂŠ $< 1$,
contradizendo $ G(0) + G(1) \geq 1$. EntĂŁo nĂŁo hĂĄ fraĂ§ĂŁo $\frac ab $
ĂŠ igual $\pi^2$: Teorema de Legendre. Se $\pi $ eram racionais,
$\pi^2$ seria tambĂŠm: entĂŁo $\pi \notin \Q $ --- e a implicaĂ§ĂŁo
sĂł funciona assim ($\sqrt2$ ĂŠ irracional com racional
quadrado), e ĂŠ por isso $\pi^2 \notin \Q $ ĂŠ estritamente mais forte
do que \cref{exo:b1:integration:9}.

\textbf{16.} Integralidade: questĂľes 11 a 12 (ponto final
derivados); pequenez: questĂľes 10 e 1; ponte: perguntas
13--14 (o duplo telescĂłpico integraĂ§ĂŁo por partes). Sem
$\frac{1}{n!}$, os dados do endpoint permanecem inteiros (ainda mais

crescimento geomĂŠtrico $ a^n $: sem fatorial, sem teorema.
