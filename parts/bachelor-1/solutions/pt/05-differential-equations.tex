\begin{solution}{exo:b1:diffeq:10}
\omterm{def:b1:logic:sets}{Definir} $ x = y = 0$: $2f(0) = 2f(0)^2$, e $ f(0) \neq 0$ dĂˇ $ f(0) =

\emph{Caso $ c = 0$:} $ f $ afim com $ f(0) = 1$: $ f(x) = 1 + \mu x $;
as forĂ§as da equaĂ§ĂŁo $\mu = 0$, excluĂ­do ($ f $ nĂŁo constante).

ConclusĂŁo: as soluĂ§Ăµes sĂŁo $ f(x) = \cos\omega x $ e $ f(x) =
\cosh\omega x $, $\omega > 0$.
\end{solution}

\begin{solution}{pb:b1:diffeq:1}
\textbf{1.} $\chi(r) = r^2 + 2\lambda r + \omega_0^2$, $\Delta =

\cref{thm:b1:diffeq:homogeneous2}
\[

\mu_1\sin\omega_d t\bigr), \qquad (\lambda_1, \mu_1) \in \R^2 .
\]
Para $\lambda = 0$: $ x = \lambda_1\cos\omega_0 t +
\mu_1\sin\omega_0 t $. A crĂ­tica ($\lambda = \omega_0$) e
superamortecido ($\lambda > \omega_0$) regimes sĂŁo aqueles de
\cref{exo:b1:diffeq:9} (apĂłs o tempo de redimensionamento): $(\lambda_1 +
\mu_1 t)\eu^{-\lambda t}$, resp.\ combinaĂ§Ăµes de $\eu^{r_\pm t}$
com $ r_\pm = -\lambda \pm \sqrt{\lambda^2 - \omega_0^2}$.

\textbf{2.} Subamortecido: $\abs x \leq \eu^{-\lambda
t}(\abs{\lambda_1} + \abs{\mu_1}) \to 0$. CrĂ­tico: $(\lambda_1 +
\mu_1 t)\eu^{-\lambda t} \to 0$ desde que as exponenciais bateram
polinĂ´mios (\cref{prop:b1:functions:powerrules}). Sobreamortecido:
$ r_- < r_+ = -\lambda + \sqrt{\lambda^2 - \omega_0^2} < 0$
porque $\sqrt{\lambda^2 - \omega_0^2} < \lambda $; ambos
decadĂŞncia exponencial.

\textbf{3.} Ao longo de uma soluĂ§ĂŁo de $(H)$, usando $ x'' = -2\lambda x' -
\omega_0^2 x $:
\[
\mathcal E'(t) = x'x'' + \omega_0^2 x x'
= x'\bigl(-2\lambda x' - \omega_0^2 x\bigr) + \omega_0^2 xx'
= -2\lambda\,x'^2 \leq 0 .

Se $ x(t_0) = x'(t_0) = 0$ entĂŁo $\mathcal E(t_0) = 0$; $\mathcal E $
Ă© nĂŁo negativo e nĂŁo crescente, entĂŁo $\mathcal E \equiv 0$ sobre
$\intco{t_0}{+\infty}$, forĂ§ando $ x \equiv 0$ lĂˇ; para $ t \leq
t_0$, execute o mesmo argumento em $\tilde x(t) = x(2t_0 - t)$, que
resolve a equaĂ§ĂŁo com amortecimento $-\lambda $ mas ainda tem
$\tilde{\mathcal E}(t_0) = 0$ e $\tilde{\mathcal E}' =
+2\lambda\tilde x'^2 \geq 0$ com $\tilde{\mathcal E} \geq 0$;
nĂŁo negativo, nĂŁo decrescente e zero na extremidade direita de
$\intoc{-\infty}{t_0}$ significa zero por toda parte. EntĂŁo $ x \equiv 0$ sobre
$\R $ --- uma prova energĂ©tica de unicidade, vĂˇlida para todos $\lambda
\geq 0$.

\textbf{4.} $ x(t + T_d) = R\,\eu^{-\lambda t}\eu^{-\lambda
T_d}\cos(\omega_d t + 2\pi - \varphi) = \eu^{-\lambda T_d}x(t)$.
O fator de reduĂ§ĂŁo por pseudoperĂ­odo Ă© $\eu^{-\delta}$ com
$\delta = \lambda T_d = \frac{2\pi\lambda}{\omega_d}$. Para

\textbf{5.} O fator de amplitude Ă© $\eu^{-\lambda t}$, que
