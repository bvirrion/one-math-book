\section*{Chapter \ref{ch:b1:reals} --- Real Numbers}

\begin{solution}{exo:b1:reals:1}
$A$: every element is $\leq 1$ and $1 \in A$: $\sup A = \max A = 1$.
Lower bounds: $0$ bounds below; for $\varepsilon > 0$, Archimedes
provides $n$ with $\frac 1n < \varepsilon$, so no positive number
bounds $A$ below: $\inf A = 0$, not attained (no min).

$B$: terms $0, -\frac12, \frac23, -\frac34, \frac45, \dots$ The even
terms $\frac{n}{n+1}$ ($n$ even) increase to $1$ without reaching it;
the odd terms $-\frac{n}{n+1}$ decrease to $-1$. So $\sup B = 1$ and
$\inf B = -1$, neither attained: no max, no min. (Bounds: $\abs{b}
< 1$ for all $b \in B$; and $\frac{n}{n+1} = 1 - \frac{1}{n+1} > 1 -
\varepsilon$ for $n$ large, similarly below.)

$C = \intoo{-\sqrt 3}{\sqrt 3}$: $\sup C = \sqrt 3$, $\inf C =
-\sqrt 3$, neither attained.
\end{solution}

\begin{solution}{exo:b1:reals:2}
Write $x = \lfloor x \rfloor + u$, $y = \lfloor y \rfloor + v$ with
$u, v \in \intco{0}{1}$. Then $x + y = \lfloor x \rfloor + \lfloor y
\rfloor + (u + v)$ with $u + v \in \intco{0}{2}$. If $u + v < 1$,
$\lfloor x + y\rfloor = \lfloor x\rfloor + \lfloor y \rfloor$; if
$1 \leq u + v < 2$, $\lfloor x+y \rfloor = \lfloor x \rfloor +
\lfloor y \rfloor + 1$. Both cases occur: $(x, y) = (0.2,\, 0.3)$
gives the left equality, $(0.7,\, 0.8)$ the right one.
\end{solution}

\begin{solution}{exo:b1:reals:3}
Let $k = \lfloor x \rfloor$, so $k \leq x < k + 1$. Multiplying by
$n$: $nk \leq nx < nk + n$, and taking floors (an increasing
operation on integers' side): $nk \leq \lfloor nx \rfloor \leq nk + n
- 1$. Dividing by $n$: $k \leq \frac{\lfloor nx \rfloor}{n} < k + 1$,
so the outer floor is $k$.
\end{solution}

\begin{solution}{exo:b1:reals:4}
Every element of $A$ is in $B$, so $\sup B$ bounds $A$ above:
therefore $\sup A \leq \sup B$ ($\sup A$ is the \emph{least} upper
bound). Symmetrically $\inf B \leq \inf A$. Finally $\inf A \leq
\sup A$ because $A$ is nonempty: any $a \in A$ sits between them.
\end{solution}

\begin{solution}{exo:b1:reals:5}
Let $s = \sup A$, $t = \sup B$. Every $a + b \leq s + t$: upper
bound. For $\varepsilon > 0$, choose $a > s - \frac\varepsilon2$ and
$b > t - \frac\varepsilon2$ (\cref{prop:b1:reals:epsilon}): then $a +
b > s + t - \varepsilon$. By the $\varepsilon$-characterization,
$\sup(A+B) = s + t$.

For $-A$: $m$ bounds $-A$ above $\iff$ $-m$ bounds $A$ below; the
least upper bound of $-A$ therefore corresponds to the greatest lower
bound of $A$: $\sup(-A) = -\inf A$.
\end{solution}

\begin{solution}{exo:b1:reals:6}
For every $x$: $f(x) + g(x) \leq \sup f + \sup g$; taking the sup of
the left side gives the inequality. Strict example: $E = \{0, 1\}$,
$f = \mathbf{1}_{\{0\}}$ (value $1$ at $0$, else $0$), $g =
\mathbf{1}_{\{1\}}$: $\sup(f + g) = 1 < 2 = \sup f + \sup g$.

No contradiction with \cref{exo:b1:reals:5}: there, $a \in A$ and $b
\in B$ vary \emph{independently}; here the same $x$ feeds both $f$
and $g$ --- the set $\{f(x) + g(x) : x \in E\}$ is smaller than the
set $\{f(x) + g(y) : x, y \in E\}$.
\end{solution}

\begin{solution}{exo:b1:reals:7}
$\sqrt 6$ is irrational: $6 = 2 \times 3$ is not a perfect square,
and $v_2(6q^2) = 1 + 2v_2(q)$ odd prevents $6q^2 = r^2$ (as in
\cref{exo:b1:arith:7}). Now suppose $x = \sqrt 2 + \sqrt 3 \in \Q$.
Then $x^2 = 5 + 2\sqrt 6 \in \Q$, so $\sqrt 6 = \frac{x^2 - 5}{2} \in
\Q$: contradiction. Hence $\sqrt 2 + \sqrt 3 \notin \Q$.
\end{solution}

\begin{solution}{exo:b1:reals:8}
Let $x < y$. Pick $n \in \N$ with $2^n > \frac{1}{y - x}$ (Archimedes:
$2^n \geq n + 1$ by an easy induction, so some power of $2$ exceeds
any real). Then, as in the proof of \cref{thm:b1:reals:density} with
$2^n$ in place of $n$: $m = \lfloor 2^n x \rfloor + 1$ satisfies $x <
\frac{m}{2^n} < y$. So $D$ is dense.
\end{solution}

\begin{solution}{exo:b1:reals:9}
\begin{enumerate}
  \item Suppose $\alpha > 0$. First, $\alpha \in G$. Suppose not: by
        the $\varepsilon$-characterization of the infimum with
        $\varepsilon = \alpha$, there is $g \in G$ with $\alpha < g <
        2\alpha$ (strict on the left since $\alpha \notin G$); then,
        with $\varepsilon = g - \alpha$, there is $h \in G$ with
        $\alpha < h < g$. Now $g - h \in G$ and $0 < g - h < g -
        \alpha < \alpha$: an element of $G \cap \intoo{0}{+\infty}$
        below its infimum, absurd. So $\alpha \in G$, and $\alpha\Z
        \subseteq G$ ($G$ is a group). Conversely, for $x \in G$, let
        $k = \lfloor x/\alpha \rfloor$: then $x - k\alpha \in G$ and
        $0 \leq x - k\alpha < \alpha$, and the definition of $\alpha$
        forces $x - k\alpha = 0$. Hence $G = \alpha\Z$.
  \item Suppose $\alpha = 0$, and let $x < y$. There is $g \in G$
        with $0 < g < y - x$. The multiple $kg$ with $k = \lfloor x/g
        \rfloor + 1$ satisfies $x < kg \leq x + g < y$, and $kg \in
        G$: density.
\end{enumerate}
$G = \Z + \sqrt 2\,\Z$ is a subgroup of $(\R, +)$. It is not of the
form $\alpha\Z$: otherwise $1 = p\alpha$ and $\sqrt 2 = q\alpha$
($p, q \in \Z$) would give $\sqrt 2 = \frac qp \in \Q$,
contradiction. By the dichotomy, $G$ is dense in $\R$.
\end{solution}

\begin{solution}{exo:b1:reals:10}
Let $s = \sup A > 0$, $t = \sup B > 0$. For $a \in A$, $b \in B$: $ab
\leq st$ (multiplying inequalities between \emph{positive} numbers).
For $0 < \varepsilon < \min(s, t)$: choose $a > s - \varepsilon$ and
$b > t - \varepsilon$; then
\[
ab > (s - \varepsilon)(t - \varepsilon)
= st - \varepsilon(s + t) + \varepsilon^2
> st - \varepsilon (s + t),
\]
and $\varepsilon(s+t)$ can be made arbitrarily small: by the
$\varepsilon$-characterization (in the form: no number $< st$ bounds
$AB$ above), $\sup AB = st$.

Positivity is essential: $A = B = \{-1, 0\}$ gives $AB = \{0, 1\}$,
$\sup AB = 1$, while $\sup A \cdot \sup B = 0 \times 0 = 0$.
\end{solution}
