\begin{solution}{exo:b1:linmaps:2}
\omterm{def:b1:linmaps:kerim}{Kernel}: solve $x + y - z = 0$, $2x + y + z = 0$, $3x + 2y = 0$.
From the third, $y = -\frac{3x}{2}$; the first gives $z = x + y =
-\frac x2$; check in the second: $2x - \frac{3x}{2} - \frac x2 = 0$:
satisfied. So $\ker u = \operatorname{Vect}\bigl((2, -3, -1)\bigr)$
(taking $x = 2$), dimension $1$.

Rank--nullity: $\operatorname{rk} u = 3 - 1 = 2$. \omterm{def:b1:linmaps:kerim}{Image}: spanned by
the \omterm{def:b1:linmaps:kerim}{images} of the canonical \omterm{def:b1:vspaces:free}{basis}, $u(e_1) = (1,2,3)$, $u(e_2) =
(1,1,2)$, $u(e_3) = (-1,1,0)$; the first two are \omterm{def:b1:vspaces:free}{free}, and the \omterm{def:b1:linmaps:rank}{rank}
is $2$: \omterm{def:b1:vspaces:free}{basis} $\bigl((1,2,3), (1,1,2)\bigr)$.

Not \omterm{def:b1:logic:inj}{injective} ($\ker \neq \{0\}$), not \omterm{def:b1:logic:inj}{surjective} (\omterm{def:b1:linmaps:rank}{rank} $2 < 3$):
consistent with \cref{cor:b1:linmaps:samedim}.
\end{solution}

\begin{solution}{exo:b1:linmaps:3}
\emph{\omterm{def:b1:linmaps:kerim}{Kernel}:} $P = P'$ forces $\deg P = \deg P'$ unless $P = 0$;
but $\deg P' < \deg P$ for $P \neq 0$: so $\ker u = \{0\}$, and $u$,
an \omterm{def:b1:logic:inj}{injective} endomorphism of the \omterm{def:b1:findim:def}{finite-dimensional} $\R_n[X]$, is an
isomorphism (\cref{cor:b1:linmaps:samedim}).

\emph{Inverse:} let $v(P) = P + P' + P'' + \dots + P^{(n)}$ (a
finite sum on $\R_n[X]$). Then
\[
v\bigl(u(P)\bigr) = \sum_{k=0}^{n} (P - P')^{(k)}
= \sum_{k=0}^{n} P^{(k)} - \sum_{k=0}^{n} P^{(k+1)}
= P - P^{(n+1)} = P ,
\]
telescoping, since $P^{(n+1)} = 0$. So $v = u^{-1}$.
\end{solution}

\begin{solution}{exo:b1:linmaps:4}
$\operatorname{im}(v \circ u) = v(\operatorname{im} u) \subseteq
\operatorname{im} v$: \omterm{def:b1:linmaps:rank}{rank} $\leq \operatorname{rk} v$. And $v$
restricted to $\operatorname{im} u$ has \omterm{def:b1:linmaps:kerim}{image}
$\operatorname{im}(vu)$ with rank--nullity inside
$\operatorname{im} u$: $\operatorname{rk}(vu) \leq
\dim\operatorname{im} u = \operatorname{rk} u$.
\end{solution}

\begin{solution}{exo:b1:linmaps:5}
$u^2 = 0$ means $u(u(x)) = 0$ for all $x$: every $u(x)$ lies in
$\ker u$, i.e.\ $\operatorname{im} u \subseteq \ker u$. Then
rank--nullity:
\[
\dim E = \dim\ker u + \operatorname{rk} u \geq 2\operatorname{rk} u .
\]
Equality example in $\R^2$: $u(x, y) = (y, 0)$: $u^2 = 0$,
$\operatorname{rk} u = 1 = \frac{\dim E}{2}$.
\end{solution}

\begin{solution}{exo:b1:linmaps:6}
$(pq)^2 = pqpq = ppqq = pq$ (commutation): a \omterm{def:b1:linmaps:projection}{projection}
(\cref{thm:b1:linmaps:projchar}).

\omterm{def:b1:linmaps:kerim}{Image}: $\operatorname{im}(pq) \subseteq \operatorname{im} p$, and
$= \operatorname{im}(qp) \subseteq \operatorname{im} q$: contained
in the intersection. Conversely, if $x \in \operatorname{im} p \cap
\operatorname{im} q$, then $p(x) = x$ and $q(x) = x$ (fixed points
characterize the \omterm{def:b1:linmaps:kerim}{image} of a \omterm{def:b1:linmaps:projection}{projection}), so $pq(x) = x$: $x \in
\operatorname{im}(pq)$.

\omterm{def:b1:linmaps:kerim}{Kernel}: $\ker p \subseteq \ker(qp) = \ker(pq)$ and likewise $\ker q
\subseteq \ker(pq)$: the sum is contained. Conversely, let $pq(x) =
0$, and write
\[
x = \underbrace{q(x)}_{\in\, \ker p} +
\underbrace{(x - q(x))}_{\in\, \ker q} :
\]
the first term satisfies $p(q(x)) = 0$, so it lies in $\ker p$; the
second is in $\ker q$ since $q(x - q(x)) = q(x) - q^2(x) = 0$. Hence
$x \in \ker p + \ker q$.
\end{solution}

\begin{solution}{exo:b1:linmaps:7}
Note first the general inclusions $\ker u \subseteq \ker u^2$ and
$\operatorname{im} u^2 \subseteq \operatorname{im} u$, and by
rank--nullity, (2) $\iff$ (3) (equal \omterm{def:b1:linmaps:kerim}{kernels} $\iff$ equal \omterm{def:b1:linmaps:rank}{ranks}
$\iff$ equal \omterm{def:b1:linmaps:kerim}{images}, given the inclusions).

(1 $\Rightarrow$ 2): let $u^2(x) = 0$; then $u(x) \in \ker u \cap
\operatorname{im} u = \{0\}$, so $x \in \ker u$.

(2 $\Rightarrow$ 1): by Grassmann and rank--nullity, $\dim(\ker u +
\operatorname{im} u) = \dim\ker u + \operatorname{rk} u -
\dim(\ker u \cap \operatorname{im} u) = \dim E - \dim(\ker u \cap
\operatorname{im} u)$: the sum is $E$ iff the intersection is
$\{0\}$. Let $y \in \ker u \cap \operatorname{im} u$: $y = u(x)$ and
$u(y) = 0$, so $u^2(x) = 0$, so (by (2)) $u(x) = 0$: $y = 0$. Hence
$E = \ker u \oplus \operatorname{im} u$.
\end{solution}

\begin{solution}{exo:b1:linmaps:8}
If $\varphi = 0$: then $\ker\varphi = E \subseteq \ker\psi$ forces
$\psi = 0 = 0\cdot\varphi$. If $\varphi \neq 0$: $\ker\varphi$ is a
\omterm{def:b1:linmaps:forms}{hyperplane}; pick $a \notin \ker\varphi$. Set $\lambda =
\frac{\psi(a)}{\varphi(a)}$. The form $\psi - \lambda\varphi$
vanishes on $\ker\varphi$ (both do, by the inclusion) and at $a$:
it vanishes on $\ker\varphi \oplus Ka = E$. So $\psi =
\lambda\varphi$.
\end{solution}

\begin{solution}{exo:b1:linmaps:9}
Suppose $\lambda_0 x + \lambda_1 u(x) + \dots + \lambda_{n-1}
u^{n-1}(x) = 0$. Apply $u^{n-1}$: all terms with a factor $u^{\geq
n}$ die, leaving $\lambda_0 u^{n-1}(x) = 0$, so $\lambda_0 = 0$.
Apply $u^{n-2}$ to the remaining relation: $\lambda_1 u^{n-1}(x) =
0$, so $\lambda_1 = 0$; and so on. The family is \omterm{def:b1:vspaces:free}{free}; being of size
$n = \dim E$, it is a \omterm{def:b1:vspaces:free}{basis} (\cref{prop:b1:findim:twoofthree}). (In
this \omterm{def:b1:vspaces:free}{basis}, $u$ acts as a shift --- the model of maximal
nilpotence.)
\end{solution}

\begin{solution}{exo:b1:linmaps:10}
\begin{enumerate}
  \item $f \circ f = -\mathrm{id}$ is \omterm{def:b1:logic:inj}{bijective}, so $f$ is
        (\cref{prop:b1:logic:comp} adapted: $f$ has the two-sided
        inverse $-f$). If $f(x) = \lambda x$ with $x \neq 0$:
        applying $f$, $-x = \lambda^2 x$, so $\lambda^2 = -1$:
        impossible in $\R$.
  \item Build the family greedily. Take $x_1 \neq 0$: $(x_1,
        f(x_1))$ is \omterm{def:b1:vspaces:free}{free} by (1). If $\operatorname{Vect}$ of the
        current family $\bigl(x_1, f(x_1), \dots, x_k,
        f(x_k)\bigr)$, call it $V_k$ --- a \omterm{def:b1:vspaces:subspace}{subspace} stable under $f$
        (each generator \omterm{def:b1:logic:map}{maps} to another generator or its negative:
        $f(f(x_i)) = -x_i$) --- is not all of $E$, choose $x_{k+1}
        \notin V_k$. \emph{Claim: the enlarged family is \omterm{def:b1:vspaces:free}{free}.}
        Suppose $\alpha x_{k+1} + \beta f(x_{k+1}) + v = 0$ with $v
        \in V_k$ and $(\alpha, \beta) \neq (0,0)$. Apply $f$:
        $\alpha f(x_{k+1}) - \beta x_{k+1} + f(v) = 0$ with $f(v)
        \in V_k$. Eliminate $f(x_{k+1})$ between the two relations
        (multiply the first by $\alpha$, the second by $-\beta$, and
        add):
        \[
        (\alpha^2 + \beta^2)\, x_{k+1} \in V_k ,
        \]
        and $\alpha^2 + \beta^2 \neq 0$ forces $x_{k+1} \in V_k$:
        contradiction. So the construction continues, adding vectors
        \emph{two at a time}, until $V_k = E$: the final family is a
        \omterm{def:b1:vspaces:free}{basis} of even size, and $n$ is even.
\end{enumerate}
\end{solution}

\begin{solution}{exo:b1:linmaps:11}
\omterm{def:b1:reals:bounds}{Upper bound}: for every $x$, $(u + v)(x) = u(x) + v(x) \in
\operatorname{im} u + \operatorname{im} v$, so
\[
\operatorname{rk}(u + v)
\leq \dim(\operatorname{im} u + \operatorname{im} v)
\leq \operatorname{rk} u + \operatorname{rk} v
\]
(Grassmann, \cref{thm:b1:findim:grassmann}). Lower bound: apply
the \omterm{def:b1:reals:bounds}{upper bound} to the pair $(u + v, -v)$, whose sum is $u$:
\[
\operatorname{rk} u \leq \operatorname{rk}(u + v) +
\operatorname{rk}(-v) = \operatorname{rk}(u + v) +
\operatorname{rk} v,
\]
so $\operatorname{rk} u - \operatorname{rk} v \leq
\operatorname{rk}(u+v)$; exchanging $u$ and $v$ gives the absolute
value.
\end{solution}

\begin{solution}{exo:b1:linmaps:12}
\emph{Exact formula.} Let $v'$ be the restriction of $v$ to the
\omterm{def:b1:vspaces:subspace}{subspace} $\operatorname{im} w$. Its \omterm{def:b1:linmaps:kerim}{image} is $v(w(F)) =
\operatorname{im}(v \circ w)$, and its \omterm{def:b1:linmaps:kerim}{kernel} is $\ker v \cap
\operatorname{im} w$. Rank--nullity for $v'$ on the space
$\operatorname{im} w$:
\[
\operatorname{rk} w = \dim\operatorname{im} w
= \operatorname{rk}(v \circ w) + \dim(\ker v \cap
\operatorname{im} w) .
\]

\emph{Frobenius.} Apply the exact formula twice, to $w$ and to
$w \circ u$:
\[
\operatorname{rk} w - \operatorname{rk}(vw)
= \dim\bigl(\ker v \cap \operatorname{im} w\bigr),
\qquad
\operatorname{rk}(wu) - \operatorname{rk}(vwu)
= \dim\bigl(\ker v \cap \operatorname{im}(wu)\bigr) .
\]
Since $\operatorname{im}(w \circ u) \subseteq \operatorname{im}
w$, the second intersection is contained in the first, and its
dimension is no larger:
\[
\operatorname{rk}(wu) - \operatorname{rk}(vwu)
\;\leq\; \operatorname{rk} w - \operatorname{rk}(vw) ,
\]
which rearranges into Frobenius' inequality. With $w =
\mathrm{id}_F$ (\omterm{def:b1:linmaps:rank}{rank} $\dim F$, and $\operatorname{im}\,
\mathrm{id}_F = F$): $\operatorname{rk} v + \operatorname{rk} u
\leq \dim F + \operatorname{rk}(vu)$, Sylvester's inequality ---
proved again, in matrix form, in \cref{exo:b1:matrices:10}.
\end{solution}

\begin{solution}{pb:b1:linmaps:1}
\textbf{1.} $(\mathrm{id} - p)^2 = \mathrm{id} - 2p + p^2 =
\mathrm{id} - p$: a projector. If $y = x - p(x)$, then $p(y) =
p(x) - p^2(x) = 0$, and conversely $x \in \ker p$ gives $x =
(\mathrm{id} - p)(x)$: $\operatorname{im}(\mathrm{id} - p) = \ker
p$. And $(\mathrm{id} - p)(x) = 0 \iff p(x) = x \iff x \in
\operatorname{im} p$ (fixed points,
\cref{thm:b1:linmaps:projchar}): $\ker(\mathrm{id} - p) =
\operatorname{im} p$.

\textbf{2.} $(\lambda\,\mathrm{id} + \mu p)^2 =
\lambda^2\,\mathrm{id} + (2\lambda\mu + \mu^2)\,p$. The pair
$(\mathrm{id}, p)$ is \omterm{def:b1:vspaces:free}{free} in $\mathcal{L}(E)$: $p = c\,
\mathrm{id}$ would give $c^2 = c$, so $p = 0$ or $\mathrm{id}$,
excluded. Identifying coefficients, the \omterm{def:b1:logic:map}{map} is a projector iff
$\lambda^2 = \lambda$ and $2\lambda\mu + \mu^2 = \mu$. For
$\lambda = 0$: $\mu \in \{0, 1\}$. For $\lambda = 1$: $\mu^2 +
\mu = 0$, $\mu \in \{0, -1\}$. Exactly four projectors in the
plane: $0$, $p$, $\mathrm{id}$, $\mathrm{id} - p$.

\textbf{3.} $(\lambda\,\mathrm{id} + \mu p)(\lambda'\,\mathrm{id}
+ \mu' p) = \lambda\lambda'\,\mathrm{id} + (\lambda\mu' +
\mu\lambda' + \mu\mu')\,p$: the plane is stable under
composition. Since $p^k = p$ for every $k \geq 1$, for $Q =
\sum_k a_k X^k$:
\[
Q(p) = a_0\,\mathrm{id} + \Bigl(\sum_{k \geq 1} a_k\Bigr) p
= Q(0)\,\mathrm{id} + \bigl(Q(1) - Q(0)\bigr)\,p .
\]

\textbf{4.} On $\operatorname{im} p$ (where $p$ acts as the
identity), $\lambda\,\mathrm{id} + \mu p$ multiplies by $\lambda
+ \mu$; on $\ker p$, by $\lambda$. As $E = \operatorname{im} p
\oplus \ker p$, the \omterm{def:b1:logic:map}{map} is \omterm{def:b1:logic:inj}{bijective} iff $\lambda \neq 0$ and
$\lambda + \mu \neq 0$. Solving $\lambda\alpha = 1$,
$\lambda\beta + \mu\alpha + \mu\beta = 0$ in the composition rule
of question 3:
\[
(\lambda\,\mathrm{id} + \mu p)^{-1}
= \frac1\lambda\,\mathrm{id} -
\frac{\mu}{\lambda(\lambda + \mu)}\,p ,
\]
whose action is by $1/\lambda$ on $\ker p$ and $1/(\lambda +
\mu)$ on $\operatorname{im} p$, as it must be.

\textbf{5.} Write $F = \operatorname{im} p = \operatorname{im}
p'$. For every $x$, $p'(x) \in F$ and $p$ fixes $F$ pointwise:
$p(p'(x)) = p'(x)$, i.e.\ $p\,p' = p'$; symmetrically $p'\,p =
p$. When two \omterm{def:b1:linmaps:projection}{projections} share their \omterm{def:b1:linmaps:kerim}{image}, the one applied
\emph{first} decides: its output already lies in $F$, where the
outer \omterm{def:b1:linmaps:projection}{projection} acts as the identity and changes nothing.

\textbf{6.} $(p + q)^2 = p^2 + pq + qp + q^2 = (p + q) + pq +
qp$, so $p + q$ projector forces $pq + qp = 0$. Compose on the
left with $p$: $pq + pqp = 0$; on the right with $p$: $pqp + qp
= 0$. Subtracting, $pq = qp$; then $pq + qp = 2pq = 0$ and the
characteristic is not $2$: $pq = qp = 0$.

\textbf{7.} With $pq = qp = 0$, the same expansion gives $(p +
q)^2 = p + q$. \omterm{def:b1:linmaps:kerim}{Image}: $\operatorname{im}(p + q) \subseteq
\operatorname{im} p + \operatorname{im} q$ always. Conversely,
for $x \in \operatorname{im} p$: $q(x) = q(p(x)) = 0$, so $(p +
q)(x) = p(x) = x$ and $x \in \operatorname{im}(p+q)$; same for
$\operatorname{im} q$. Directness: $x \in \operatorname{im} p
\cap \operatorname{im} q$ gives $x = p(x) = p(q(x)) = 0$.
\omterm{def:b1:linmaps:kerim}{Kernel}: if $p(x) + q(x) = 0$, applying $p$ gives $p(x) +
p(q(x)) = p(x) = 0$, and applying $q$ gives $q(x) = 0$: $\ker(p
+ q) = \ker p \cap \ker q$ (the reverse inclusion is clear).

\textbf{8.} $p - q$ is a projector iff $\mathrm{id} - (p - q) =
(\mathrm{id} - p) + q$ is one (question 1 twice). By questions
6--7 applied to the projectors $\mathrm{id} - p$ and $q$, this
holds iff $(\mathrm{id} - p)q = q(\mathrm{id} - p) = 0$, i.e.\
iff $pq = q$ and $qp = q$.

\textbf{9.} $pq = q$ means $p$ fixes every $q(x)$, i.e.\
$\operatorname{im} q \subseteq \ker(p - \mathrm{id}) =
\operatorname{im} p$. And $qp = q$ means $q\bigl((\mathrm{id} -
p)(x)\bigr) = 0$ for all $x$, i.e.\ $q$ vanishes on
$\operatorname{im}(\mathrm{id} - p) = \ker p$: $\ker p \subseteq
\ker q$. Both steps are equivalences: the order $q \leq p$ says
$q$ projects onto a smaller \omterm{def:b1:linmaps:kerim}{image}, along a larger \omterm{def:b1:linmaps:kerim}{kernel}.

\textbf{10.} Expanding, $(\mathrm{id} - p)(\mathrm{id} - q) =
\mathrm{id} - p - q + pq = \mathrm{id} - r$. The projectors
$\mathrm{id} - p$ and $\mathrm{id} - q$ commute, so by
\cref{exo:b1:linmaps:6} their product $\mathrm{id} - r$ is the
projector onto $\operatorname{im}(\mathrm{id} - p) \cap
\operatorname{im}(\mathrm{id} - q) = \ker p \cap \ker q$ along
$\ker(\mathrm{id} - p) + \ker(\mathrm{id} - q) =
\operatorname{im} p + \operatorname{im} q$. By question 1, $r =
\mathrm{id} - (\mathrm{id} - r)$ is then the projector with
$\operatorname{im} r = \operatorname{im} p + \operatorname{im}
q$ and $\ker r = \ker p \cap \ker q$.

\textbf{11.} $p_i$ is well defined (uniqueness of the
decomposition) and \omterm{def:b1:linmaps:def}{linear} (the decomposition of $x + \lambda y$
is the sum of the decompositions, again by uniqueness). For $x_i
\in F_i$ the decomposition is $x_i$ itself, so $p_i(x_i) = x_i$:
$p_i^2 = p_i$, and $p_j(x_i) = 0$ for $j \neq i$: $p_i p_j = 0$
($p_j(x) \in F_j$). Summing the components, $\sum_i p_i =
\mathrm{id}$. Finally $\operatorname{im} p_i = F_i$ and $\ker
p_i = \bigoplus_{j \neq i} F_j$.

\textbf{12.} $p_i = p_i \circ \mathrm{id} = p_i\sum_j p_j =
p_i^2 + \sum_{j \neq i} p_i p_j = p_i^2$: each $p_i$ is a
projector. Every $x = \mathrm{id}(x) = \sum_i p_i(x)$ lies in
$\sum_i \operatorname{im} p_i$: the \omterm{def:b1:linmaps:kerim}{images} sum to $E$.
Directness: suppose $y_1 + \dots + y_k = 0$ with $y_i \in
\operatorname{im} p_i$, so $p_i(y_i) = y_i$. Apply $p_j$: $p_j
(y_i) = p_j p_i (y_i) = 0$ for $i \neq j$, so $0 = p_j\bigl(\sum
y_i\bigr) = y_j$, for every $j$. Hence $E = \bigoplus_i
\operatorname{im} p_i$.

\textbf{13.} $q = \mathrm{id} - p$, and question 1 gives $pq =
p - p^2 = 0 = qp$ directly: for two projectors, summing to the
identity already forces orthogonality of the pair.

\textbf{14.} $p + q = \mathrm{id} - r$ with $r$ a projector, and
$(\mathrm{id} - r)$ is a projector (question 1): so $p + q$ is a
projector, and question 6 gives $pq = qp = 0$. By symmetry ($q +
r = \mathrm{id} - p$ and $p + r = \mathrm{id} - q$), all
pairwise products vanish, and question 12 concludes: $E =
\operatorname{im} p \oplus \operatorname{im} q \oplus
\operatorname{im} r$, automatically.

\textbf{15.} \emph{Lemma.} By induction with Grassmann
(\cref{thm:b1:findim:grassmann}):
\[
\dim(F_1 + \dots + F_k) \leq \dim(F_1 + \dots + F_{k-1}) + \dim
F_k \leq \dots \leq \sum_i \dim F_i .
\]
If the total is an equality, each step is: $(F_1 + \dots +
F_{j-1}) \cap F_j = \{0\}$ for every $j$, and a relation $y_1 +
\dots + y_k = 0$ ($y_i \in F_i$) collapses from the right: $y_k
\in (F_1 + \dots + F_{k-1}) \cap F_k = \{0\}$, then $y_{k-1} =
0$, etc.: the sum is direct. Conversely a \omterm{def:b1:vspaces:sum}{direct sum} has
additive dimensions (concatenate bases). \emph{Application:}
$x = \sum_i p_i(x)$ shows $E = \sum_i \operatorname{im} p_i$,
so $n \leq \sum_i \operatorname{rk} p_i$; the hypothesis gives
equality, hence directness. Products: fix $j$ and $y \in
\operatorname{im} p_j$. Then $y = \sum_i p_i(y)$ with $p_i(y)
\in \operatorname{im} p_i$, while $y = y$ is also a
decomposition (component $j$ alone); uniqueness forces $p_i(y)
= 0$ for $i \neq j$. Applied to $y = p_j(x)$: $p_i p_j = 0$.

\textbf{16.} If $u^k(x) = 0$ then $u^{k+1}(x) = u(0) = 0$. And
$\operatorname{im} u^{k+1} = u^k\bigl(u(E)\bigr) \subseteq
u^k(E) = \operatorname{im} u^k$.

\textbf{17.} Assume $\ker u^{r} = \ker u^{r+1}$ and let $x \in
\ker u^{r+2}$: then $u(x) \in \ker u^{r+1} = \ker u^{r}$, so
$u^{r+1}(x) = 0$: $x \in \ker u^{r+1}$. With question 16,
$\ker u^{r+1} = \ker u^{r+2}$, and by induction all later
\omterm{def:b1:linmaps:kerim}{kernels} coincide with $\ker u^{r}$. For \omterm{def:b1:linmaps:kerim}{images}: rank--nullity
gives $\dim\operatorname{im} u^k = n - \dim\ker u^k$, so the
\omterm{def:b1:linmaps:kerim}{image} dimensions freeze exactly when the \omterm{def:b1:linmaps:kerim}{kernel} dimensions do,
and with the inclusions of question 16, equal dimensions mean
equal \omterm{def:b1:vspaces:subspace}{subspaces} (\cref{thm:b1:findim:subspaces}).

\textbf{18.} The sequence $\bigl(\dim\ker u^k\bigr)_k$ is
nondecreasing with values in $\intint{0}{n}$; it cannot increase
strictly $n + 1$ times, so some $\dim\ker u^{r} = \dim\ker
u^{r+1}$ with $r \leq n$, hence $\ker u^{r} = \ker u^{r+1}$
(inclusion plus equal dimension). Take $r$ smallest; question 17
freezes everything from $r$ on, \omterm{def:b1:linmaps:kerim}{images} included.

\textbf{19.} Intersection: let $x \in \ker u^{r} \cap
\operatorname{im} u^{r}$, say $x = u^{r}(y)$ with $u^{r}(x) =
0$. Then $u^{2r}(y) = 0$, and $\ker u^{2r} = \ker u^{r}$
(question 17), so $x = u^{r}(y) = 0$. Dimensions: rank--nullity
for $u^{r}$ gives $\dim\ker u^{r} + \dim\operatorname{im} u^{r}
= n$; with trivial intersection, Grassmann makes the sum a
\omterm{def:b1:vspaces:subspace}{subspace} of dimension $n$: $E = \ker u^{r} \oplus
\operatorname{im} u^{r}$.

\textbf{20.} Stability: $u^{r}(u(x)) = u(u^{r}(x)) = 0$ for $x
\in \ker u^{r}$; and $u(u^{r}(y)) = u^{r}(u(y)) \in
\operatorname{im} u^{r}$. On $N = \ker u^{r}$: $(u|_N)^{r} = 0$
by definition of $N$: nilpotent. On $I = \operatorname{im}
u^{r}$: $\ker(u|_I) = \ker u \cap I \subseteq \ker u^{r} \cap I
= \{0\}$, so $u|_I$ is an \omterm{def:b1:logic:inj}{injective} endomorphism of the
\omterm{def:b1:findim:def}{finite-dimensional} $I$, hence \omterm{def:b1:logic:inj}{bijective}
(\cref{cor:b1:linmaps:samedim}).

\textbf{21.} Let $x = a + b$ with $a \in N$, $b \in I$. Then
$u(x) = u(a) + u(b)$ with $u(a) \in N$ and $u(b) \in I$
(question 20): this \emph{is} the decomposition of $u(x)$, so
$\pi(u(x)) = u(a) = u(\pi(x))$: $\pi u = u\pi$.

\textbf{22.} $u^2(x,y,z) = u(y, 0, z) = (0, 0, z)$ and
$u^3(x,y,z) = u(0,0,z) = (0,0,z) = u^2(x,y,z)$. \omterm{def:b1:linmaps:kerim}{Kernels}: $\ker u
= \{y = z = 0\} = \operatorname{Vect}(e_1)$, $\ker u^2 = \{z =
0\} = \operatorname{Vect}(e_1, e_2)$, $\ker u^3 = \ker u^2$:
stabilization at $r = 2$. \omterm{def:b1:linmaps:kerim}{Images}: $\operatorname{im} u =
\operatorname{Vect}(e_1, e_3)$, $\operatorname{im} u^2 =
\operatorname{Vect}(e_3)$. Fitting: $\R^3 =
\operatorname{Vect}(e_1, e_2) \oplus \operatorname{Vect}(e_3)$,
and $\pi(x, y, z) = (x, y, 0)$. Check: $\pi u(x,y,z) = \pi(y, 0,
z) = (y, 0, 0)$ and $u\pi(x,y,z) = u(x, y, 0) = (y, 0, 0)$:
equal. On the first factor $u(x, y, 0) = (y, 0, 0)$, whose
square is $0$: nilpotent; on the second $u(0,0,z) = (0,0,z)$:
the identity, \omterm{def:b1:logic:inj}{bijective}.

\textbf{23.} If $u^m = 0$ then $\ker u^m = E$; since \omterm{def:b1:linmaps:kerim}{kernels} are
frozen from $r$ on, $\ker u^{r} = \ker u^{\max(m, r)} = E$.
Conversely $\ker u^{r} = E$ means $u^{r} = 0$. And $\ker u^{r} =
E \iff$ the Fitting projector is onto $E$ along $\{0\}$, i.e.\
$\pi = \mathrm{id}$. Finally $r \leq n$ (question 18) gives:
every nilpotent endomorphism satisfies $u^{n} = 0$ --- the
nilpotency index never exceeds the dimension.

\textbf{24.} Let $m$ be a nilpotency index of $u|_A$: $A
\subseteq \ker u^{m} \subseteq \ker u^{\max(m,r)} = \ker u^{r}$.
Since $u|_B$ is \omterm{def:b1:logic:inj}{bijective}, $B = u(B) = u^{k}(B) \subseteq
\operatorname{im} u^{k}$ for every $k$, in particular $B
\subseteq \operatorname{im} u^{r}$. Then
\[
n = \dim A + \dim B \leq \dim\ker u^{r} +
\dim\operatorname{im} u^{r} = n :
\]
both inclusions are equalities of dimensions, hence of
\omterm{def:b1:vspaces:subspace}{subspaces}: $A = \ker u^{r}$, $B = \operatorname{im} u^{r}$.

\textbf{25.} (i) Part III is a dictionary: splittings $E = F_1
\oplus \dots \oplus F_k$ correspond exactly to families of
projectors with $\sum p_i = \mathrm{id}$ and $p_i p_j = 0$, the
$F_i$ being the \omterm{def:b1:linmaps:kerim}{images}. (ii) For $k = 3$ the complements
$\mathrm{id} - p_i$ are themselves projectors, which closed the
argument with no extra hypothesis; for general $k$ one needs
$\sum_i \operatorname{rk} p_i \leq n$, an inequality the Year~2
trace gives for free ($\operatorname{tr} p = \operatorname{rk}
p$ for a projector, and traces add up to $\operatorname{tr}
\mathrm{id} = n$). (iii) \cref{exo:b1:linmaps:7} is Fitting's
lemma in the already-stabilized case $r \leq 1$; in general one
lets the \omterm{def:b1:linmaps:kerim}{kernel} and \omterm{def:b1:linmaps:kerim}{image} chains freeze, which takes at most $n$
steps. (iv) In the Year~2 volume, applied to $u - \lambda\,
\mathrm{id}$, the nilpotent factor becomes the generalized
eigenspace at $\lambda$ and the projectors of Part III become
the spectral projectors of the reduction theory. The theorem of
Part IV is \emph{Fitting's lemma}.
\end{solution}
