\section*{Chapter \ref{ch:b1:linmaps} --- Linear Maps}

\begin{solution}{exo:b1:linmaps:1}
(1) Linear: coordinates are linear expressions. (2) Not linear:
$u(2(1,1)) = 4 \neq 2 = 2u(1,1)$. (3) Linear: differentiation and
multiplication by $X$ are, and sums of linear maps are. (4) Linear:
evaluation respects pointwise operations.
\end{solution}

\begin{solution}{exo:b1:linmaps:2}
Kernel: solve $x + y - z = 0$, $2x + y + z = 0$, $3x + 2y = 0$.
From the third, $y = -\frac{3x}{2}$; the first gives $z = x + y =
-\frac x2$; check in the second: $2x - \frac{3x}{2} - \frac x2 = 0$:
satisfied. So $\ker u = \operatorname{Vect}\bigl((2, -3, -1)\bigr)$
(taking $x = 2$), dimension $1$.

Rank--nullity: $\operatorname{rk} u = 3 - 1 = 2$. Image: spanned by
the images of the canonical basis, $u(e_1) = (1,2,3)$, $u(e_2) =
(1,1,2)$, $u(e_3) = (-1,1,0)$; the first two are free, and the rank
is $2$: basis $\bigl((1,2,3), (1,1,2)\bigr)$.

Not injective ($\ker \neq \{0\}$), not surjective (rank $2 < 3$):
consistent with \cref{cor:b1:linmaps:samedim}.
\end{solution}

\begin{solution}{exo:b1:linmaps:3}
\emph{Kernel:} $P = P'$ forces $\deg P = \deg P'$ unless $P = 0$;
but $\deg P' < \deg P$ for $P \neq 0$: so $\ker u = \{0\}$, and $u$,
an injective endomorphism of the finite-dimensional $\R_n[X]$, is an
isomorphism (\cref{cor:b1:linmaps:samedim}).

\emph{Inverse:} let $v(P) = P + P' + P'' + \dots + P^{(n)}$ (a
finite sum on $\R_n[X]$). Then
\[
v\bigl(u(P)\bigr) = \sum_{k=0}^{n} (P - P')^{(k)}
= \sum_{k=0}^{n} P^{(k)} - \sum_{k=0}^{n} P^{(k+1)}
= P - P^{(n+1)} = P ,
\]
telescoping, since $P^{(n+1)} = 0$. So $v = u^{-1}$.
\end{solution}

\begin{solution}{exo:b1:linmaps:4}
$\operatorname{im}(v \circ u) = v(\operatorname{im} u) \subseteq
\operatorname{im} v$: rank $\leq \operatorname{rk} v$. And $v$
restricted to $\operatorname{im} u$ has image
$\operatorname{im}(vu)$ with rank--nullity inside
$\operatorname{im} u$: $\operatorname{rk}(vu) \leq
\dim\operatorname{im} u = \operatorname{rk} u$.
\end{solution}

\begin{solution}{exo:b1:linmaps:5}
$u^2 = 0$ means $u(u(x)) = 0$ for all $x$: every $u(x)$ lies in
$\ker u$, i.e.\ $\operatorname{im} u \subseteq \ker u$. Then
rank--nullity:
\[
\dim E = \dim\ker u + \operatorname{rk} u \geq 2\operatorname{rk} u .
\]
Equality example in $\R^2$: $u(x, y) = (y, 0)$: $u^2 = 0$,
$\operatorname{rk} u = 1 = \frac{\dim E}{2}$.
\end{solution}

\begin{solution}{exo:b1:linmaps:6}
$(pq)^2 = pqpq = ppqq = pq$ (commutation): a projection
(\cref{thm:b1:linmaps:projchar}).

Image: $\operatorname{im}(pq) \subseteq \operatorname{im} p$, and
$= \operatorname{im}(qp) \subseteq \operatorname{im} q$: contained
in the intersection. Conversely, if $x \in \operatorname{im} p \cap
\operatorname{im} q$, then $p(x) = x$ and $q(x) = x$ (fixed points
characterize the image of a projection), so $pq(x) = x$: $x \in
\operatorname{im}(pq)$.

Kernel: $\ker p \subseteq \ker(qp) = \ker(pq)$ and likewise $\ker q
\subseteq \ker(pq)$: the sum is contained. Conversely, let $pq(x) =
0$, and write
\[
x = \underbrace{q(x)}_{\in\, \ker p} +
\underbrace{(x - q(x))}_{\in\, \ker q} :
\]
the first term satisfies $p(q(x)) = 0$, so it lies in $\ker p$; the
second is in $\ker q$ since $q(x - q(x)) = q(x) - q^2(x) = 0$. Hence
$x \in \ker p + \ker q$.
\end{solution}

\begin{solution}{exo:b1:linmaps:7}
Note first the general inclusions $\ker u \subseteq \ker u^2$ and
$\operatorname{im} u^2 \subseteq \operatorname{im} u$, and by
rank--nullity, (2) $\iff$ (3) (equal kernels $\iff$ equal ranks
$\iff$ equal images, given the inclusions).

(1 $\Rightarrow$ 2): let $u^2(x) = 0$; then $u(x) \in \ker u \cap
\operatorname{im} u = \{0\}$, so $x \in \ker u$.

(2 $\Rightarrow$ 1): by Grassmann and rank--nullity, $\dim(\ker u +
\operatorname{im} u) = \dim\ker u + \operatorname{rk} u -
\dim(\ker u \cap \operatorname{im} u) = \dim E - \dim(\ker u \cap
\operatorname{im} u)$: the sum is $E$ iff the intersection is
$\{0\}$. Let $y \in \ker u \cap \operatorname{im} u$: $y = u(x)$ and
$u(y) = 0$, so $u^2(x) = 0$, so (by (2)) $u(x) = 0$: $y = 0$. Hence
$E = \ker u \oplus \operatorname{im} u$.
\end{solution}

\begin{solution}{exo:b1:linmaps:8}
If $\varphi = 0$: then $\ker\varphi = E \subseteq \ker\psi$ forces
$\psi = 0 = 0\cdot\varphi$. If $\varphi \neq 0$: $\ker\varphi$ is a
hyperplane; pick $a \notin \ker\varphi$. Set $\lambda =
\frac{\psi(a)}{\varphi(a)}$. The form $\psi - \lambda\varphi$
vanishes on $\ker\varphi$ (both do, by the inclusion) and at $a$:
it vanishes on $\ker\varphi \oplus Ka = E$. So $\psi =
\lambda\varphi$.
\end{solution}

\begin{solution}{exo:b1:linmaps:9}
Suppose $\lambda_0 x + \lambda_1 u(x) + \dots + \lambda_{n-1}
u^{n-1}(x) = 0$. Apply $u^{n-1}$: all terms with a factor $u^{\geq
n}$ die, leaving $\lambda_0 u^{n-1}(x) = 0$, so $\lambda_0 = 0$.
Apply $u^{n-2}$ to the remaining relation: $\lambda_1 u^{n-1}(x) =
0$, so $\lambda_1 = 0$; and so on. The family is free; being of size
$n = \dim E$, it is a basis (\cref{prop:b1:findim:twoofthree}). (In
this basis, $u$ acts as a shift --- the model of maximal
nilpotence.)
\end{solution}

\begin{solution}{exo:b1:linmaps:10}
\begin{enumerate}
  \item $f \circ f = -\mathrm{id}$ is bijective, so $f$ is
        (\cref{prop:b1:logic:comp} adapted: $f$ has the two-sided
        inverse $-f$). If $f(x) = \lambda x$ with $x \neq 0$:
        applying $f$, $-x = \lambda^2 x$, so $\lambda^2 = -1$:
        impossible in $\R$.
  \item Build the family greedily. Take $x_1 \neq 0$: $(x_1,
        f(x_1))$ is free by (1). If $\operatorname{Vect}$ of the
        current family $\bigl(x_1, f(x_1), \dots, x_k,
        f(x_k)\bigr)$, call it $V_k$ --- a subspace stable under $f$
        (each generator maps to another generator or its negative:
        $f(f(x_i)) = -x_i$) --- is not all of $E$, choose $x_{k+1}
        \notin V_k$. \emph{Claim: the enlarged family is free.}
        Suppose $\alpha x_{k+1} + \beta f(x_{k+1}) + v = 0$ with $v
        \in V_k$ and $(\alpha, \beta) \neq (0,0)$. Apply $f$:
        $\alpha f(x_{k+1}) - \beta x_{k+1} + f(v) = 0$ with $f(v)
        \in V_k$. Eliminate $f(x_{k+1})$ between the two relations
        (multiply the first by $\alpha$, the second by $-\beta$, and
        add):
        \[
        (\alpha^2 + \beta^2)\, x_{k+1} \in V_k ,
        \]
        and $\alpha^2 + \beta^2 \neq 0$ forces $x_{k+1} \in V_k$:
        contradiction. So the construction continues, adding vectors
        \emph{two at a time}, until $V_k = E$: the final family is a
        basis of even size, and $n$ is even.
\end{enumerate}
\end{solution}
