\section*{Chapitre \ref{ch:b1:structures} --- Structures algébriques}

\begin{solution}{exo:b1:structures:1}
\emph{Stabilité~:} $x * y = 1 \iff x + y - xy = 1 \iff (1-x)(1-y) = 0$,
impossible pour $x, y \neq 1$. En effet, l'identité clé est
\[
1 - x * y = (1 - x)(1 - y):
\]
l'\omterm{def:b1:logic:map}{application} $\varphi(x) = 1 - x$ envoie $(E, *)$ sur $(\R^*, \times)$
avec $\varphi(x * y) = \varphi(x)\varphi(y)$ --- c'est un \omterm{def:b1:structures:morphism}{morphisme}
\omterm{def:b1:logic:inj}{bijectif}. Tous les axiomes se transportent alors~: associativité et
commutativité découlent de celles de $\times$~; l'élément neutre est
$\varphi^{-1}(1) = 0$ (vérification~: $x * 0 = x$)~; l'inverse de $x$
est $\varphi^{-1}\bigl((1-x)^{-1}\bigr) =
1 - \frac{1}{1-x} = \frac{x}{x - 1}$ (qui est $\neq 1$). Donc $(E, *)$
est un \omterm{def:b1:structures:group}{groupe abélien}.
\end{solution}

\begin{solution}{exo:b1:structures:3}
Notons $\mathrm{id}$, les transpositions $\tau_{12}, \tau_{13},
\tau_{23}$ (qui échangent les deux points indiqués), et les cycles $c =
(1\,2\,3)$ (c'est-à-dire $1 \mapsto 2 \mapsto 3 \mapsto 1$) et $c^2 =
(1\,3\,2)$. La table de $\sigma\rho$ (ligne $\sigma$, colonne $\rho$,
on applique $\rho$ d'abord)~:
\begin{center}
\renewcommand{\arraystretch}{1.2}
\begin{tabular}{c|cccccc}
$\sigma\backslash\rho$ & $\mathrm{id}$ & $c$ & $c^2$ & $\tau_{12}$ &
$\tau_{13}$ & $\tau_{23}$ \\
\hline
$\mathrm{id}$ & $\mathrm{id}$ & $c$ & $c^2$ & $\tau_{12}$ &
$\tau_{13}$ & $\tau_{23}$ \\
$c$ & $c$ & $c^2$ & $\mathrm{id}$ & $\tau_{13}$ & $\tau_{23}$ &
$\tau_{12}$ \\
$c^2$ & $c^2$ & $\mathrm{id}$ & $c$ & $\tau_{23}$ & $\tau_{12}$ &
$\tau_{13}$ \\
$\tau_{12}$ & $\tau_{12}$ & $\tau_{23}$ & $\tau_{13}$ & $\mathrm{id}$
& $c^2$ & $c$ \\
$\tau_{13}$ & $\tau_{13}$ & $\tau_{12}$ & $\tau_{23}$ & $c$ &
$\mathrm{id}$ & $c^2$ \\
$\tau_{23}$ & $\tau_{23}$ & $\tau_{13}$ & $\tau_{12}$ & $c^2$ & $c$ &
$\mathrm{id}$ \\
\end{tabular}
\end{center}
Couple ne commutant pas~: $\tau_{12}\tau_{13} = c^2$ tandis que
$\tau_{13}\tau_{12} = c$. (Vérification d'une case~:
$\tau_{12}\tau_{13}$ envoie $1 \xmapsto{\tau_{13}} 3
\xmapsto{\tau_{12}} 3$, $3 \mapsto 1 \mapsto 2$, $2 \mapsto 2 \mapsto
1$~: c'est $1 \mapsto 3 \mapsto 2 \mapsto 1$, le cycle $c^2 =
(1\,3\,2)$.)
\end{solution}

\begin{solution}{exo:b1:structures:4}
$H$~: $1 \in H$~; pour $z, w \in H$, $\abs{zw^{-1}} = \abs z / \abs w =
1$~: le critère s'applique. $\R_+^*$~: de même, la positivité
remplaçant le \omterm{def:b1:complex:field}{module}. Réunion~: $\iu \in H$ et $2 \in \R_+^*$, mais
$2\iu$ a pour \omterm{def:b1:complex:field}{module} $2 \neq 1$ et n'est pas un réel strictement
positif~: $2\iu \notin H \cup \R_+^*$, donc la réunion n'est pas stable
--- ce n'est pas un \omterm{def:b1:structures:subgroup}{sous-groupe} (comme le prévoit
\cref{exo:b1:structures:6}, aucun des deux \omterm{def:b1:structures:subgroup}{sous-groupes} ne contient
l'autre).
\end{solution}

\begin{solution}{exo:b1:structures:5}
\omterm{def:b1:structures:morphism}{Morphisme}~: $\eu^{\iu(\theta + \varphi)} =
\eu^{\iu\theta}\eu^{\iu\varphi}$ (\cref{thm:b1:complex:funceq}).
\omterm{def:b1:structures:morphism}{Noyau}~: $\eu^{\iu\theta} = 1 \iff \theta \in 2\pi\Z$, donc $\ker f =
2\pi\Z \neq \{0\}$~: $f$ n'est pas \omterm{def:b1:logic:inj}{injectif}. Image~: tout nombre
complexe de \omterm{def:b1:complex:field}{module} $1$ s'écrit $\eu^{\iu\theta}$ pour un certain
$\theta$ (forme polaire), donc $\operatorname{im} f = \mathbb{U}$, le
cercle unité. La restriction de $f$ à un intervalle semi-ouvert de
longueur $2\pi$, comme $\intco{0}{2\pi}$ ou $\intoc{-\pi}{\pi}$, est
\omterm{def:b1:logic:inj}{injective} (deux angles de même image diffèrent d'un multiple de $2\pi$,
et un seul représentant de chaque classe tient dans l'intervalle)~;
aucun intervalle de longueur supérieure ne convient, puisqu'il contient
deux points distants de $2\pi$.
\end{solution}

\begin{solution}{exo:b1:structures:6}
Intersection~: $e \in H \cap K$, et $x, y \in H \cap K$ donnent
$xy^{-1}$ à la fois dans $H$ et dans $K$. Réunion~: si $H \subseteq K$,
la réunion est $K$, qui est un \omterm{def:b1:structures:subgroup}{sous-groupe} (et symétriquement).
Réciproquement, supposons qu'aucune des deux inclusions n'ait lieu~:
prenons $h \in H \setminus K$ et $k \in K \setminus H$, et supposons
que $H \cup K$ soit un \omterm{def:b1:structures:subgroup}{sous-groupe}~; alors $hk \in H \cup K$.
Si $hk \in H$, alors $k = h^{-1}(hk) \in H$~: contradiction. Si $hk \in
K$, alors $h = (hk)k^{-1} \in K$~: contradiction. Donc $H \cup K$ n'est
pas un \omterm{def:b1:structures:subgroup}{sous-groupe}.
\end{solution}

\begin{solution}{exo:b1:structures:7}
Remarquons d'abord que $x^2 = e$ signifie $x^{-1} = x$ pour tout $x$.
Alors, pour $x, y \in G$~:
\[
xy = (xy)^{-1} = y^{-1} x^{-1} = yx ,
\]
en utilisant \cref{prop:b1:structures:rules}~(2). Donc $G$ est \omterm{def:b1:structures:group}{abélien}.
\end{solution}

\begin{solution}{exo:b1:structures:8}
Inversibles de $\Z/18\Z$~: les classes premières avec $18 = 2 \times
3^2$~: $\overline 1, \overline 5, \overline 7, \overline{11},
\overline{13}, \overline{17}$. Inverse de $\overline 5$~: $5 \times 11
= 55 = 3\times 18 + 1$, donc $\overline 5^{-1} = \overline{11}$.

$\overline 5 x = \overline 7$~: on multiplie par $\overline{11}$~: $x =
\overline{77} = \overline 5$ (car $77 = 4\times 18 + 5$). Solution
unique.

$\overline 6 x = \overline 3$~: l'équation $6x \equiv 3 \pmod{18}$
signifie $18 \mid 6x - 3$. Or $6x - 3 = 3(2x - 1)$ est impair, tandis
que $18$ est pair~: un nombre pair ne peut pas diviser un nombre impair.
Pas de solution.

$\overline 6 x = \overline{12}$~: $6x \equiv 12 \pmod{18} \iff x
\equiv 2 \pmod 3$~: solutions $x \in \{\overline 2, \overline 5,
\overline 8, \overline{11}, \overline{14}, \overline{17}\}$ --- il y en
a six.
\end{solution}

\begin{solution}{exo:b1:structures:9}
$\Z[\sqrt 2]$ contient $0$ et $1$, et est stable par soustraction et par
produit~:
\[
(a + b\sqrt 2)(c + d\sqrt 2) = (ac + 2bd) + (ad + bc)\sqrt 2 ,
\]
c'est donc un \omterm{def:b1:structures:ring}{sous-anneau} de $\R$ (commutativité, associativité et
distributivité sont héritées). Inversible~: $(1 + \sqrt 2)(-1 + \sqrt 2)
= 2 - 1 = 1$, donc $1 + \sqrt 2$ est inversible, d'inverse $\sqrt 2 - 1
\in \Z[\sqrt 2]$. Ses puissances $(1 + \sqrt 2)^n$ forment une suite
strictement croissante (la base est $> 1$), donc sont deux à deux
distinctes, et chacune est inversible ($\bigl((1+\sqrt2)^n\bigr)^{-1} =
(\sqrt 2 - 1)^n$)~: le \omterm{def:b1:structures:group}{groupe} des inversibles est infini.
\end{solution}

\begin{solution}{exo:b1:structures:10}
$x + x = (x + x)^2 = x^2 + x^2 + x^2 + x^2 = 4x^2 = 4x$ --- donc $2x =
4x$, d'où $2x = 0$, c'est-à-dire $x + x = 0$ (chaque élément est son
propre opposé). Ensuite
\[
x + y = (x+y)^2 = x^2 + xy + yx + y^2 = x + xy + yx + y ,
\]
donc $xy + yx = 0$, c'est-à-dire $xy = -yx = yx$ (en utilisant
$-z = z$). Ainsi $A$ est commutatif.

Exemple~: sur $\mathcal{P}(E)$, posons $A + B = (A \cup B) \setminus (A
\cap B)$ (différence symétrique) et $A \times B = A \cap B$. On
vérifie~: $(\mathcal{P}(E), +)$ est un \omterm{def:b1:structures:group}{groupe abélien} de neutre
$\emptyset$, chaque partie étant son propre opposé~; $\cap$ est
associative, commutative, de neutre $E$~; la distributivité $A \cap (B +
C) = (A \cap B) + (A \cap C)$ a lieu (un élément appartient au membre de
gauche si et seulement s'il est dans $A$ et dans exactement l'un des
deux \omterm{def:b1:logic:sets}{ensembles} $B, C$). Et $A \cap A = A$~: tout élément est idempotent,
comme voulu.
\end{solution}

\begin{solution}{exo:b1:structures:11}
Écrivons l'hypothèse pour $n$ et $n+1$~:
\[
(xy)^{n+1} = x^{n+1} y^{n+1}
\quad\text{et}\quad
(xy)^{n+1} = (xy)(xy)^n = xy\,x^n y^n .
\]
En identifiant~: $x^{n+1} y^{n+1} = x\,y\,x^n\,y^n$~; on simplifie par
$x$ à gauche et par $y^n$ à droite~: $x^n y = y x^n$. Le même calcul un
cran plus haut ($n+1$ et $n+2$) donne $x^{n+1} y = y x^{n+1}$. Alors
\[
y\,x^{n+1} = x^{n+1} y = x\,(x^n y) = x\,y\,x^n ,
\]
et en simplifiant par $x^n$ à droite dans $y x \cdot x^n = x y \cdot
x^n$~: $yx = xy$. Donc $G$ est \omterm{def:b1:structures:group}{abélien}.
\end{solution}

\begin{solution}{exo:b1:structures:12}
\begin{enumerate}
  \item Soit $f \colon \Z \to \Z$ additive et $a = f(1)$. Par
        récurrence, $f(k) = ka$ pour $k \in \N$, et $f(-k) = -f(k) =
        -ka$~: $f$ est donc la multiplication par $a$. Réciproquement,
        toute \omterm{def:b1:logic:map}{application} $k \mapsto ak$ est un \omterm{def:b1:structures:morphism}{morphisme}~: les
        \omterm{def:b1:structures:morphism}{morphismes} $(\Z,+) \to (\Z,+)$ sont exactement les
        multiplications par un entier fixé.
  \item Soit $f \colon \Q \to \Z$ un \omterm{def:b1:structures:morphism}{morphisme}, $x \in \Q$ et
        $n \in \N^*$. Alors
        \[
        f(x) = f\Bigl(\underbrace{\tfrac xn + \dots +
        \tfrac xn}_{n}\Bigr) = n\,f\Bigl(\frac xn\Bigr) ,
        \]
        donc l'entier $f(x)$ est divisible par tout $n \geq 1$.
        Le seul entier de ce type est $0$~: $f \equiv 0$.
\end{enumerate}
\end{solution}

\begin{solution}{pb:b1:structures:1}
\textbf{1.} Une \omterm{def:b1:counting:objects}{permutation} est une \omterm{def:b1:logic:inj}{bijection} de $\intint1n$,
c'est-à-dire un arrangement des $n$ objets pris $n$ à $n$~: il y en a
$n!$ (\cref{thm:b1:counting:counts}). Avec $\sigma = [2,3,1]$, $\tau =
[1,3,2]$~: $\sigma\tau$ envoie $1 \mapsto 1 \mapsto 2$, $2 \mapsto 3
\mapsto 1$, $3 \mapsto 2 \mapsto 3$~: $\sigma\tau = [2,1,3]$~; et
$\tau\sigma$ envoie $1 \mapsto 2 \mapsto 3$, $2 \mapsto 3 \mapsto
2$, $3 \mapsto 1 \mapsto 1$~: $\tau\sigma = [3,2,1] \neq
\sigma\tau$.

\textbf{2.} Soient $\gamma, \gamma'$ de supports disjoints $S, S'$.
Pour $x \in S$~: $\gamma'(x) = x$ et $\gamma(x) \in S$, donc
$\gamma\gamma'(x) = \gamma(x) = \gamma'\gamma(x)$. Symétriquement
pour $x \in S'$~; et les deux membres fixent tout $x \notin S \cup S'$.
Donc $\gamma\gamma' = \gamma'\gamma$.

\textbf{3.} Pour $i \in \intint1n$, les valeurs $i, \sigma(i),
\sigma^2(i), \dots$ vivent dans un ensemble fini, donc $\sigma^a(i) =
\sigma^b(i)$ pour certains $a < b$~; l'\omterm{def:b1:logic:inj}{injectivité} donne
$\sigma^{b-a}(i) = i$~: la suite revient à $i$. Appelons \emph{orbite}
de $i$ l'\omterm{def:b1:logic:sets}{ensemble} $\{i, \sigma(i), \dots, \sigma^{k-1}(i)\}$, où
$k \geq 1$ est minimal tel que $\sigma^k(i) = i$. Deux orbites qui se
rencontrent en un point coïncident (chacune est l'\omterm{def:b1:logic:sets}{ensemble} des images
successives de ce point par $\sigma$), donc les orbites forment une
\omterm{thm:b1:logic:partition}{partition} de $\intint1n$~; $\sigma$ agit sur chaque orbite de taille
$k \geq 2$ comme le $k$-cycle $(i\ \sigma(i)\ \cdots\
\sigma^{k-1}(i))$ et fixe les singletons. Le produit de ces cycles
disjoints coïncide partout avec $\sigma$. Unicité~: dans toute
décomposition en cycles disjoints, le cycle passant par $i$ ne peut
être que $(i\ \sigma(i)\ \cdots)$ --- les cycles sont forcés d'être les
orbites munies de l'action induite.

\textbf{4.} En suivant les orbites~: $1 \to 4 \to 2 \to 1$, $3 \to 5
\to 3$, $6 \to 7 \to 8 \to 6$~:
\[
\sigma = (1\ 4\ 2)(3\ 5)(6\ 7\ 8) .
\]
Si $\sigma = \gamma_1\cdots\gamma_r$ avec des cycles disjoints de
longueurs $k_1, \dots, k_r$, la commutation (question 2) donne
$\sigma^m = \gamma_1^m\cdots\gamma_r^m$, et comme les supports sont
disjoints, $\sigma^m = \mathrm{id}$ si et seulement si chaque
$\gamma_i^m = \mathrm{id}$, si et seulement si $k_i \mid m$ pour tout
$i$ (un $k$-cycle est d'\omterm{def:b1:structures:order}{ordre} $k$~: $\gamma^m$ envoie $a_1$ sur
$a_{1 + (m \bmod k)}$). Le plus petit tel $m$ est
$\operatorname{lcm}(k_1, \dots, k_r)$. Ici~:
$\operatorname{lcm}(3, 2, 3) = 6$.

\textbf{5.} Appliquons le membre de droite à chaque point, en
commençant par le facteur le plus à droite. $a_1 \mapsto a_2$ par
$(a_1\ a_2)$, puis tous les facteurs suivants fixent $a_2$~: au total
$a_1 \mapsto a_2$. Pour $2 \leq i < k$~: $a_i$ n'est pas touché jusqu'à
ce que $(a_1\ a_i)$ l'envoie sur $a_1$, et le facteur immédiatement
suivant $(a_1\ a_{i+1})$ envoie $a_1$ sur $a_{i+1}$, après quoi plus
rien ne le bouge~: au total $a_i \mapsto a_{i+1}$. Enfin $a_k$ est fixé
par tous les facteurs sauf le plus à gauche, qui l'envoie sur $a_1$.
C'est exactement le cycle. Comme toute \omterm{def:b1:counting:objects}{permutation} est un produit de
cycles (question 3), c'est un produit de transpositions. Pour le
$\sigma$ de la question 4~:
\[
\sigma = (1\ 2)(1\ 4)\;(3\ 5)\;(6\ 8)(6\ 7),
\]
soit cinq transpositions.

\textbf{6.} Récurrence sur $b - a$. Pour $b = a + 1$ l'identité est
triviale ($1 = 2\cdot1 - 1$ facteur). Pour $b > a + 1$, on vérifie
directement que $(a\ b) = (a\ \ a{+}1)\,(a{+}1\ \ b)\,(a\ \ a{+}1)$~:
le membre de droite envoie $a \mapsto a{+}1 \mapsto b \mapsto b$, $b
\mapsto b \mapsto a{+}1 \mapsto a$, $a{+}1 \mapsto a \mapsto a
\mapsto a{+}1$, et fixe le reste. Par récurrence, $(a{+}1\ \ b)$ est
un produit palindromique de $2(b - a - 1) - 1$ transpositions
adjacentes, donc $(a\ b)$ en est un de $2(b - a) - 1$~: un nombre
impair.

\textbf{7.} $N(\mathrm{id}) = 0$, $\varepsilon = +1$. Pour $(i\ \
i{+}1)$, la seule paire inversée est $(i, i+1)$~: $N = 1$,
$\varepsilon = -1$. Pour $[2, 3, 1]$~: les paires inversées sont $(1,
3)$ (valeurs $2 > 1$) et $(2, 3)$ (valeurs $3 > 1$)~: $N = 2$,
$\varepsilon = +1$.

\textbf{8.} Les listes de valeurs de $\sigma$ et de $\sigma\tau$ ne
diffèrent que par l'échange des positions $i$ et $i + 1$. Pour une
paire de positions ne faisant intervenir ni $i$ ni $i+1$, rien ne
change. Pour $k < i$, les deux paires $(k, i)$ et $(k, i+1)$ échangent
leurs statuts d'inversion (les deux mêmes valeurs sont comparées à
$\sigma(k)$, dans l'\omterm{def:b1:structures:order}{ordre} inverse des positions)~: leur contribution
totale est inchangée~; de même pour $k > i + 1$. La seule paire
restante, $(i, i+1)$, change de statut. D'où $N(\sigma\tau) = N(\sigma)
\pm 1$.

\textbf{9.} Soit $\tau = (a\ b)$ une transposition quelconque~: par la
question 6, c'est un produit d'un nombre impair de transpositions
adjacentes, donc la multiplication à droite par $\tau$ modifie $N$ d'un
total impair (question 8, appliquée de façon répétée)~:
$\varepsilon(\sigma \tau) = -\varepsilon(\sigma)$. Maintenant, si
$\sigma = \tau_1\cdots \tau_p$ (des transpositions), construisons-le à
partir de l'identité par $p$ multiplications à droite~:
$\varepsilon(\sigma) = (-1)^p\varepsilon(\mathrm{id}) = (-1)^p$. Comme
$\varepsilon(\sigma)$ est défini par les inversions --- indépendamment
de toute factorisation --- la parité de $p$ est un invariant de
$\sigma$. \omterm{def:b1:structures:morphism}{Morphisme}~: en écrivant $\sigma$ avec $p$ transpositions et
$\sigma'$ avec $q$, $\sigma\sigma'$ en utilise $p + q$~:
$\varepsilon(\sigma\sigma') = (-1)^{p+q} =
\varepsilon(\sigma)\varepsilon(\sigma')$.

\textbf{10.} Un $k$-cycle est un produit de $k - 1$ transpositions
(question 5)~: $\varepsilon = (-1)^{k-1}$. Pour un $\sigma$ général
d'orbites de tailles $k_1, \dots, k_r$ ($k_i \geq 2$) plus $f$ points
fixes, on a $c(\sigma) = r + f$ et $n = k_1 + \dots + k_r + f$, donc
\[
\varepsilon(\sigma) = \prod_{i=1}^r (-1)^{k_i - 1}
= (-1)^{\sum_i k_i - r} = (-1)^{n - f - r} = (-1)^{n -
c(\sigma)} .
\]

\textbf{11.} $\mathfrak A_n = \ker\varepsilon$ est un \omterm{def:b1:structures:subgroup}{sous-groupe}, en
tant que \omterm{def:b1:structures:morphism}{noyau} d'un \omterm{def:b1:structures:morphism}{morphisme} (\cref{def:b1:structures:morphism}).
Fixons une transposition $\tau_0$ (il en existe pour $n \geq 2$).
L'\omterm{def:b1:logic:map}{application} $\sigma \mapsto \sigma\tau_0$ est une \omterm{def:b1:logic:inj}{bijection} de
$\mathfrak S_n$ (elle est sa propre réciproque) qui échange
$\mathfrak A_n$ et l'\omterm{def:b1:logic:sets}{ensemble} des \omterm{def:b1:counting:objects}{permutations} impaires (question 9).
Ces deux \omterm{def:b1:logic:sets}{ensembles} forment une \omterm{thm:b1:logic:partition}{partition} de $\mathfrak S_n$ et ont même
\omterm{def:b1:counting:card}{cardinal}~: $\abs{\mathfrak A_n} = \frac{n!}2$.

\textbf{12.} \emph{Inversions} de $[4, 1, 5, 2, 3, 7, 8, 6]$~: la
valeur $4$ domine $1, 2, 3$~: trois~; la valeur $5$ domine $2, 3$~:
deux~; la valeur $7$ domine $6$~: une~; la valeur $8$ domine $6$~: une.
$N = 7$, $\varepsilon = -1$. \emph{Type de cycles}~: $c = 3$ orbites,
$n = 8$~: $\varepsilon = (-1)^{8-3} = -1$. \emph{Nombre de
transpositions}~: cinq transpositions à la question 5~: $(-1)^5 = -1$.
Les trois calculs concordent.

\textbf{13.} $(a\ b)(a\ c)$ (le facteur le plus à droite d'abord)~: $a
\mapsto c \mapsto c$~; $c \mapsto a \mapsto b$~; $b \mapsto b \mapsto
a$~: c'est le $3$-cycle $(a\ c\ b)$. Et $(a\ c\ b)(a\ c\ d)$~: $a
\mapsto c \mapsto b$~; $b \mapsto b \mapsto a$~; $c \mapsto d \mapsto
d$~; $d \mapsto a \mapsto c$~: c'est $(a\ b)(c\ d)$, comme annoncé.

\textbf{14.} Soit $\sigma \in \mathfrak A_n$~: par la question 9,
$\sigma = \tau_1\cdots\tau_{2m}$ avec un nombre pair de
transpositions. Groupons-les par paires consécutives
$\tau_{2i-1}\tau_{2i}$~: si les deux sont égales, la paire est
l'identité et disparaît~; si elles ont exactement un point commun, la
première identité de la question 13 écrit la paire comme un unique
$3$-cycle~; si elles sont disjointes, la seconde identité l'écrit
comme deux $3$-cycles. Ainsi $\sigma$ est un produit de $3$-cycles (ou
l'identité, produit vide --- et, pour $n \geq 3$, également
$(1\ 2\ 3)^3$).

\textbf{15.} $(1\ 2)(3\ 4) = (1\ 3\ 2)(1\ 3\ 4)$ (question 13
avec $a{=}1, b{=}2, c{=}3, d{=}4$). Pour le $5$-cycle~: par la
question 5, $(1\ 2\ 3\ 4\ 5) = (1\ 5)(1\ 4)(1\ 3)(1\ 2)$, et en
appariant~: $(1\ 5)(1\ 4) = (1\ 4\ 5)$, $(1\ 3)(1\ 2) = (1\ 2\ 3)$~:
\[
(1\ 2\ 3\ 4\ 5) = (1\ 4\ 5)(1\ 2\ 3) .
\]
(Vérification sur $3$~: $(1\ 2\ 3)$ envoie $3 \to 1$, puis
$(1\ 4\ 5)$ envoie $1 \to 4$~: au total $3 \to 4$, ce qui est correct.)

\textbf{16.} Appliquons les deux membres à un point quelconque. Pour
$i = \sigma(a_j)$~: le membre de gauche donne $\sigma\bigl((a_1\ \dots\
a_k)(a_j)\bigr) = \sigma(a_{j+1})$ (indices modulo $k$), ce qui est
précisément l'image de $\sigma(a_j)$ par le membre de droite. Pour $i$
qui n'est pas de cette forme~: $\sigma^{-1}(i)$ est hors du support,
donc le membre de gauche fixe $i$, et le membre de droite aussi.
Égalité partout.

\textbf{17.} Faire glisser la pièce de la case $c'$ dans la case vide
$c$ échange les contenus des cases $c$ et $c'$ (la pièce $9$, le
vide, va en $c'$). Si la pièce $\sigma(i)$ occupait la case $i$, la
nouvelle position est $\sigma' = \sigma \circ (c\ c')$~: mêmes
contenus, sauf que les cases $c, c'$ portent chacune l'ancien contenu
de l'autre. Par la question 9, $\varepsilon(\sigma') =
-\varepsilon(\sigma)$.

\textbf{18.} Un coup envoie la case vide sur une case adjacente~: sa
ligne ou sa colonne change d'exactement $1$, donc la distance de
Manhattan $d$ à la case $9$ varie de $\pm1$, et $(-1)^d$ change de
signe. Puisque chaque coup change le signe de $\varepsilon(\sigma)$ et
celui de $(-1)^{d(\sigma)}$, leur produit $I(\sigma)$ est inchangé par
tout coup~: c'est un invariant.

\textbf{19.} La position résolue a $\varepsilon = +1$, $d = 0$~:
$I = +1$. La position visée (pièces $7, 8$ échangées, vide à sa place)
est la transposition des contenus des cases $7$ et $8$~: $\varepsilon
= -1$, $d = 0$~: $I = -1$. Comme $I$ est invariant et que les deux
valeurs diffèrent, aucune suite de coups ne les relie.

\textbf{20.} Une position ayant le vide à sa place est une \omterm{def:b1:counting:objects}{permutation}
des $8$ pièces parmi les cases $1, \dots, 8$, c'est-à-dire un élément
de $\mathfrak S_8$~; elle vérifie $d = 0$, donc $I = \varepsilon(\sigma)$.
Être atteignable impose $I = +1$, c'est-à-dire $\sigma \in \mathfrak
A_8$~; la réciproque admise dit que tout $\mathfrak A_8$ est atteint.
Dénombrement~: $\abs{\mathfrak A_8} = \frac{8!}2 = 20\,160$
(question 11).

\textbf{21.} Par la question 20, les dispositions atteignables avec le
vide à sa place forment exactement $\mathfrak A_8$ --- en particulier un
\emph{\omterm{def:b1:structures:subgroup}{sous-groupe}} de $\mathfrak S_8$~: composer deux mélanges
solubles, ou inverser l'un d'eux, reste soluble, ce qui est loin d'être
évident par un raisonnement direct sur le taquin.

\textbf{22.} (a) Un $3$-cycle de pièces avec le vide à sa place~:
$\varepsilon = +1$ (question 10), $d = 0$, donc $I = +1$~: atteignable
(par la réciproque admise) --- on peut permuter circulairement trois
pièces. (b) Pièce $5$ et vide échangés~: la position est la
transposition des contenus des cases $5$ et $9$, donc $\varepsilon =
-1$~; le vide se trouve au centre, à distance de Manhattan $d = 2$ de
sa case d'origine, donc $(-1)^d = +1$ et $I = -1$~: inatteignable. On ne
peut pas simplement «~garer le vide au milieu~» en laissant les pièces
par ailleurs rangées.

\textbf{23.} Soit $f \colon \mathfrak S_n \to \{\pm1\}$ un
\omterm{def:b1:structures:morphism}{morphisme}. Pour deux transpositions quelconques $\tau, \tau'$, la
question 16 fournit un $\sigma$ tel que $\sigma\tau\sigma^{-1} = \tau'$
(envoyer les deux points déplacés sur les deux autres~; $n \geq 3$
garantit la place nécessaire, même si $n = 2$ est ici trivial). Alors
$f(\tau') = f(\sigma)f(\tau)f(\sigma)^{-1} = f(\tau)$ puisque
$\{\pm1\}$ est \omterm{def:b1:structures:group}{abélien}~: $f$ est constant sur les transpositions. Si
cette constante vaut $+1$, alors $f = 1$ sur tous les produits de
transpositions, c'est-à-dire partout (question 5). Si elle vaut $-1$,
alors $f(\sigma) = (-1)^p = \varepsilon(\sigma)$ sur un produit de $p$
transpositions. Donc $f \in \{1, \varepsilon\}$.

\textbf{24.} (i) La propriété de \omterm{def:b1:structures:morphism}{morphisme} de $\varepsilon$ et la
machinerie du \omterm{def:b1:structures:morphism}{noyau} ont donné à $\mathfrak A_n$ sa structure de
\omterm{def:b1:structures:subgroup}{sous-groupe} et son \omterm{def:b1:counting:card}{cardinal}, et un raisonnement du type
\cref{prop:b1:structures:kernel} traverse les questions 11 et 21.
(ii) Dénombrement~: $\abs{\mathfrak S_n} = n!$, l'argument de moitié de
la question 11 et le compte $20\,160$ de la question 20 sont le
\cref{ch:b1:counting} à l'œuvre. (iii) Les questions 8 et 9 résolvent
un véritable problème de bonne définition --- «~la parité du nombre de
transpositions~» présuppose que cette parité ne dépend pas de la
factorisation, exactement comme les opérations de $\Z/n\Z$ exigeaient
l'indépendance vis-à-vis des représentants dans
la \cref{def:b1:structures:zn}.

\textbf{25.} La signature est un unique calcul à valeurs dans
$\{\pm1\}$, dont on a démontré une fois pour toutes qu'il est bien
défini, et il remplit trois fonctions à la fois~: à l'intérieur, il coupe
$\mathfrak S_n$ en deux et isole $\mathfrak A_n$ avec ses générateurs
$3$-cycles~; à l'extérieur, il tranche en une ligne une question
(«~ces deux pièces peuvent-elles être échangées~?~») qu'une recherche
naïve ne pourrait jamais régler, puisqu'aucune liste finie de suites
de coups infructueuses ne démontre l'impossibilité~; et
structurellement, c'est le moteur à signes alternés à l'intérieur de la
formule $\det A = \sum_\sigma \varepsilon(\sigma)\,
a_{1\sigma(1)}\cdots a_{n\sigma(n)}$ du \cref{ch:b1:det}. Le schéma ---
trouver une quantité conservée par tout coup élémentaire, la calculer
au départ et à l'arrivée --- est l'arme standard du mathématicien contre
les questions «~est-ce possible~?~», et il reviendra chaque fois qu'un
\omterm{def:b1:structures:group}{groupe} agira sur un \omterm{def:b1:logic:sets}{ensemble} d'états.
\end{solution}
