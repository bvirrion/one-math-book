\section*{Chapitre \ref{ch:b1:diffeq} --- Équations différentielles linéaires}

\begin{solution}{exo:b1:diffeq:1}
Équation homogène~: $y_h = \lambda\,\eu^{-2x}$. Solution
particulière~: essayons $y_p = c\,\eu^{3x}$ ($3$ n'est pas racine de
$r + 2$)~: $3c + 2c = 1$, $c = \frac15$. Solution générale~: $y =
\frac{\eu^{3x}}{5} + \lambda\,\eu^{-2x}$. Avec $y(0) = 1$~: $\frac15 +
\lambda = 1$, $\lambda = \frac45$, donc $y = \frac{\eu^{3x} +
4\,\eu^{-2x}}{5}$.
\end{solution}

\begin{solution}{exo:b1:diffeq:2}
Sur $\intoo{0}{+\infty}$, divisons par $x$~: $y' - \frac{2}{x}\,y =
x^2$. Ici $A(x) = -2\ln x$, $\eu^{-A(x)} = x^2$~: solutions homogènes
$\lambda x^2$. \omterm{thm:b1:diffeq:voc}{Variation de la constante}~: $\mu'(x) = x^2 \cdot x^{-2}
= 1$, donc $\mu = x + \lambda$ et
\[
y(x) = x^3 + \lambda x^2, \qquad \lambda \in \R .
\]
\emph{Vérification~:} $x(3x^2 + 2\lambda x) - 2(x^3 + \lambda x^2) =
x^3$.
\end{solution}

\begin{solution}{exo:b1:diffeq:3}
$y'' - 3y' + 2y = 0$~: racines $1$ et $2$~; $\;y = \lambda\,\eu^{x} +
\mu\,\eu^{2x}$.

$y'' + 4y' + 4y = 0$~: racine double $-2$~; $\;y = (\lambda + \mu
x)\,\eu^{-2x}$.

$y'' - 2y' + 5y = 0$~: racines $1 \pm 2\iu$~; $\;y =
\eu^{x}(\lambda\cos 2x + \mu\sin 2x)$.
\end{solution}

\begin{solution}{exo:b1:diffeq:4}
Équation homogène~: racines $\pm 1$, $y_h = \lambda\,\eu^x +
\mu\,\eu^{-x}$. Solution particulière à second membre polynomial
($\gamma = 0$ n'est pas racine)~: $y_p = ax^2 + bx + c$~; en
substituant, $2a - (ax^2 + bx + c) = x^2$ donne $a = -1$, $b = 0$, $c
= 2a = -2$~: $y_p = -x^2 - 2$. Solution générale $y = -x^2 - 2 +
\lambda\eu^x + \mu\eu^{-x}$.

Cauchy~: $y(0) = -2 + \lambda + \mu = 0$ et $y'(0) = \lambda - \mu =
1$~: $\lambda = \frac32$, $\mu = \frac12$. Donc
$y = -x^2 - 2 + \frac{3\eu^x + \eu^{-x}}{2}$.
\end{solution}

\begin{solution}{exo:b1:diffeq:5}
$a(x) = \tan x$, $A(x) = -\ln(\cos x)$ (licite~: $\cos > 0$ sur
l'intervalle), $\eu^{-A} = \cos x$~: solutions homogènes
$\lambda\cos x$. \omterm{thm:b1:diffeq:voc}{Variation de la constante}~: $\mu'(x) = \sin 2x \cdot
\frac{1}{\cos x} = 2\sin x$, donc $\mu = -2\cos x + \lambda$ et
\[
y(x) = -2\cos^2 x + \lambda \cos x .
\]
\emph{Vérification~:} $y' = 4\cos x \sin x - \lambda\sin x$ et
$y\tan x = -2\cos x\sin x + \lambda \sin x$~; leur somme vaut $2\cos
x\sin x = \sin 2x$, comme voulu.
\end{solution}

\begin{solution}{exo:b1:diffeq:6}
$\chi(r) = r^2 - 4r + 3 = (r-1)(r-3)$~: $\gamma = 1$ est racine simple
($m = 1$). Essayons $y_p = x(ax + b)\,\eu^x$. Avec $u = ax^2 + bx$,
\[
y_p'' - 4y_p' + 3y_p = \bigl(u'' + (2 - 4)u' + \chi(1) u\bigr)\eu^x
= \bigl(2a - 2(2ax + b)\bigr)\eu^x .
\]
Identifions avec $(2x + 1)\eu^x$~: $-4a = 2$ et $2a - 2b = 1$, donc $a
= -\frac12$, $b = -1$. Solution générale~:
\[
y = -\Bigl(\frac{x^2}{2} + x\Bigr)\eu^{x} + \lambda\,\eu^{x} +
\mu\,\eu^{3x}, \qquad (\lambda, \mu) \in \R^2 .
\]
\end{solution}

\begin{solution}{exo:b1:diffeq:7}
Équation homogène~: $y_h = \lambda\cos 2x + \mu\sin 2x$.

Second membre $x$ ($\gamma = 0$ n'est pas racine)~: $y_1 = ax + b$ avec
$4(ax + b) = x$~: $y_1 = \frac x4$.

Second membre $\sin 2x = \Im(\eu^{2\iu x})$, $2\iu$ racine simple de
$r^2 + 4$~: essayons $z = c\,x\,\eu^{2\iu x}$~; alors $z'' + 4z =
4\iu c\,\eu^{2\iu x}$, qui vaut $\eu^{2\iu x}$ pour $c =
\frac{1}{4\iu} = -\frac{\iu}{4}$. Donc $z = -\frac{\iu x}{4}(\cos 2x +
\iu \sin 2x)$ et $y_2 = \Im(z) = -\frac{x\cos 2x}{4}$.

Par superposition~:
\[
y = \frac{x}{4} - \frac{x\cos 2x}{4} + \lambda\cos 2x + \mu\sin 2x .
\]
\end{solution}

\begin{solution}{exo:b1:diffeq:8}
L'équation $T' + kT = 20k$ admet la solution particulière constante
$20$ et les solutions homogènes $\lambda\eu^{-kt}$~: $T(t) = 20 +
\lambda\,\eu^{-kt}$, et $T(0) = 80$ donne $\lambda = 60$~:
\[
T(t) = 20 + 60\,\eu^{-kt} .
\]
$T(10) = 50$~: $\eu^{-10k} = \frac12$, donc $k = \frac{\ln 2}{10}$.
Ensuite $T(t) = 25$ exige $\eu^{-kt} = \frac{5}{60} = \frac{1}{12}$,
c'est-à-dire
\[
t = \frac{\ln 12}{k} = 10\,\frac{\ln 12}{\ln 2} \approx 35.8
\text{ minutes.}
\]
\end{solution}

\begin{solution}{exo:b1:diffeq:9}
\begin{enumerate}
  \item $\chi(r) = r^2 + 2\varepsilon r + 1$, $\Delta = 4(\varepsilon^2
        - 1)$.
        Pour $\varepsilon \in \intco{0}{1}$~: racines $-\varepsilon \pm
        \iu\sqrt{1 - \varepsilon^2}$, donc
        $y = \eu^{-\varepsilon t}\bigl(\lambda\cos\omega t +
        \mu\sin\omega t\bigr)$ avec $\omega = \sqrt{1 -
        \varepsilon^2}$.
        Pour $\varepsilon = 1$~: racine double $-1$, $y = (\lambda +
        \mu t)\,\eu^{-t}$.
        Pour $\varepsilon > 1$~: racines réelles $r_\pm = -\varepsilon
        \pm \sqrt{\varepsilon^2 - 1}$, toutes deux $< 0$, et $y =
        \lambda\eu^{r_+t} + \mu\eu^{r_-t}$.
  \item Pour $\varepsilon \in \intoo{0}{1}$~: $\abs y \leq
        \eu^{-\varepsilon t}(\abs\lambda + \abs\mu) \to 0$. Pour
        $\varepsilon = 1$~: $(\lambda + \mu t)\eu^{-t} \to 0$
        (l'exponentielle l'emporte sur le polynôme,
        \cref{prop:b1:functions:powerrules}). Pour $\varepsilon > 1$~:
        les deux exponentielles décroissent puisque $r_\pm < 0$ (en
        effet $\sqrt{\varepsilon^2 - 1} < \varepsilon$). Pour
        $\varepsilon = 0$~: $y = \lambda\cos t + \mu\sin t$ a une
        amplitude constante $\sqrt{\lambda^2 + \mu^2} \neq 0$ sauf si
        $y = 0$.
  \item Écrivons $\lambda\cos\omega t + \mu\sin\omega t =
        R\cos(\omega t - \varphi)$ avec $R = \sqrt{\lambda^2 + \mu^2}
        > 0$. Les zéros de $y$ sont ceux de $\cos(\omega t -
        \varphi)$ (le facteur $\eu^{-\varepsilon t}$ ne s'annule
        jamais)~: $\omega t - \varphi \equiv \frac\pi2 \pmod \pi$, une
        progression arithmétique d'écart $\frac{\pi}{\omega} =
        \frac{\pi}{\sqrt{1 - \varepsilon^2}}$.
\end{enumerate}
\end{solution}

\begin{solution}{exo:b1:diffeq:10}
Posons $x = y = 0$~: $2f(0) = 2f(0)^2$, et $f(0) \neq 0$ donne $f(0) =
1$. Fixons $x$ et dérivons deux fois l'équation par rapport à $y$~:
\[
f''(x+y) + f''(x-y) = 2 f(x) f''(y) .
\]
En posant $y = 0$~: $\;2f''(x) = 2 f(x) f''(0)$, c'est-à-dire
\[
f''(x) = c\,f(x), \qquad c = f''(0).
\]
\emph{Cas $c = \omega^2 > 0$~:} $f(x) = \lambda\cosh\omega x +
\mu\sinh\omega x$~; $f(0) = 1$ donne $\lambda = 1$. En reportant dans
l'équation fonctionnelle et en utilisant les formules d'addition
(\cref{prop:b1:functions:hyprules}), l'équation impose $\mu = 0$
(comparer les coefficients de $\sinh\omega x \sinh\omega y$ ou évaluer
en $x = y$)~: $f = \cosh\omega x$, qui satisfait bien
$\cosh(x+y) + \cosh(x-y) = 2\cosh x\cosh y$.

\emph{Cas $c = -\omega^2 < 0$~:} de même $f(x) = \cos\omega x$
($\omega \neq 0$), qui satisfait l'équation.

\emph{Cas $c = 0$~:} $f$ est affine avec $f(0) = 1$~: $f(x) = 1 + \mu
x$~; l'équation impose $\mu = 0$, exclu ($f$ n'est pas constante).

Conclusion~: les solutions sont $f(x) = \cos\omega x$ et $f(x) =
\cosh\omega x$, $\omega > 0$.
\end{solution}

\begin{solution}{exo:b1:diffeq:11}
Posons $z(t) = y(\eu^t)$, de sorte que $y(x) = z(\ln x)$ pour $x > 0$.
Alors
\[
y'(x) = \frac{z'(\ln x)}x,
\qquad
y''(x) = \frac{z''(\ln x) - z'(\ln x)}{x^2} ,
\]
et en substituant dans l'équation~:
\[
x^2y'' - xy' + y = \bigl(z'' - z'\bigr) - z' + z
= z'' - 2z' + z = 0 .
\]
\omterm{def:b1:diffeq:linear2}{Polynôme caractéristique} $(r - 1)^2$~: racine double $1$, donc $z(t) =
(\lambda + \mu t)\,\eu^t$ et, de retour dans la variable $x =
\eu^t$~:
\[
y(x) = (\lambda + \mu\ln x)\,x,
\qquad \lambda, \mu \in \R .
\]
\end{solution}

\begin{solution}{exo:b1:diffeq:12}
\begin{enumerate}
  \item Sous forme normalisée $y' - \frac2x\,y = 0$ sur chaque
        intervalle~: $A(x) = -2\ln\abs x$, donc les solutions sont $y =
        a x^2$ sur $\intoo0{+\infty}$ et $y = b x^2$ sur
        $\intoo{-\infty}0$, avec des constantes indépendantes
        (\cref{thm:b1:diffeq:homogeneous1}).
  \item Soit $y = ax^2$ pour $x \geq 0$ et $bx^2$ pour $x < 0$. Sur
        chaque demi-droite ouverte, $y$ est dérivable et vérifie $xy'
        = 2y$. En $0$~: les taux d'accroissement $\frac{y(h) -
        y(0)}h = ah$ ou $bh$ tendent vers $0$, donc $y'(0) = 0$
        existe, et l'équation en $x = 0$ s'écrit $0 \cdot y'(0) =
        2y(0) = 0$~: elle est satisfaite. Donc $y$ est solution de
        l'équation sur $\R$ tout entier.
  \item Les solutions sur $\R$ sont exactement ces fonctions
        recollées~: une famille à \emph{deux} paramètres pour une
        équation du premier ordre. Il n'y a pas de contradiction avec
        le \cref{thm:b1:diffeq:voc}, dont les hypothèses tombent en
        défaut ici~: une fois l'équation écrite sous la forme
        $y' + a(x)y = 0$, le coefficient $a(x) = -\frac2x$ n'est pas
        continu en $0$ ---
        il n'y est même pas défini --- de sorte que $\R$ n'est pas un
        intervalle sur lequel le théorème s'applique. La singularité en
        $0$ déconnecte les deux demi-droites, et la valeur $y(0) = 0$
        est imposée, sans transmettre aucune information d'un côté à
        l'autre. Toute donnée de Cauchy en $x_0 \neq 0$ ne détermine la
        solution que sur la demi-droite contenant $x_0$.
\end{enumerate}
\end{solution}

\begin{solution}{pb:b1:diffeq:1}
\textbf{1.} $\chi(r) = r^2 + 2\lambda r + \omega_0^2$, $\Delta =
4(\lambda^2 - \omega_0^2) < 0$ pour $0 < \lambda < \omega_0$~: racines
$-\lambda \pm \iu\omega_d$, donc d'après
le \cref{thm:b1:diffeq:homogeneous2}
\[
x(t) = \eu^{-\lambda t}\bigl(\lambda_1\cos\omega_d t +
\mu_1\sin\omega_d t\bigr), \qquad (\lambda_1, \mu_1) \in \R^2 .
\]
Pour $\lambda = 0$~: $x = \lambda_1\cos\omega_0 t +
\mu_1\sin\omega_0 t$. Les régimes critique ($\lambda = \omega_0$) et
suramorti ($\lambda > \omega_0$) sont ceux de
l'\cref{exo:b1:diffeq:9} (après changement d'échelle de temps)~:
$(\lambda_1 + \mu_1 t)\eu^{-\lambda t}$, resp.\ des combinaisons de
$\eu^{r_\pm t}$ avec $r_\pm = -\lambda \pm \sqrt{\lambda^2 -
\omega_0^2}$.

\textbf{2.} Sous-amorti~: $\abs x \leq \eu^{-\lambda
t}(\abs{\lambda_1} + \abs{\mu_1}) \to 0$. Critique~: $(\lambda_1 +
\mu_1 t)\eu^{-\lambda t} \to 0$ puisque les exponentielles l'emportent
sur les polynômes (\cref{prop:b1:functions:powerrules}). Suramorti~:
$r_- < r_+ = -\lambda + \sqrt{\lambda^2 - \omega_0^2} < 0$ car
$\sqrt{\lambda^2 - \omega_0^2} < \lambda$~; les deux exponentielles
décroissent.

\textbf{3.} Le long d'une solution de $(H)$, en utilisant $x'' =
-2\lambda x' - \omega_0^2 x$~:
\[
\mathcal E'(t) = x'x'' + \omega_0^2 x x'
= x'\bigl(-2\lambda x' - \omega_0^2 x\bigr) + \omega_0^2 xx'
= -2\lambda\,x'^2 \leq 0 .
\]
Si $x(t_0) = x'(t_0) = 0$ alors $\mathcal E(t_0) = 0$~; $\mathcal E$
est positive et décroissante, donc $\mathcal E \equiv 0$ sur
$\intco{t_0}{+\infty}$, ce qui force $x \equiv 0$ sur cet
intervalle~; pour $t \leq t_0$, appliquons le même raisonnement à
$\tilde x(t) = x(2t_0 - t)$, qui est solution de l'équation
d'amortissement $-\lambda$ mais vérifie toujours
$\tilde{\mathcal E}(t_0) = 0$ et $\tilde{\mathcal E}' =
+2\lambda\tilde x'^2 \geq 0$ avec $\tilde{\mathcal E} \geq 0$~;
positive, croissante et nulle à l'extrémité droite de
$\intoc{-\infty}{t_0}$ signifie nulle partout. Donc $x \equiv 0$ sur
$\R$ --- une démonstration énergétique de l'unicité, valable pour tout
$\lambda \geq 0$.

\textbf{4.} $x(t + T_d) = R\,\eu^{-\lambda t}\eu^{-\lambda
T_d}\cos(\omega_d t + 2\pi - \varphi) = \eu^{-\lambda T_d}x(t)$. Le
facteur de réduction par pseudo-période est $\eu^{-\delta}$ avec
$\delta = \lambda T_d = \frac{2\pi\lambda}{\omega_d}$. Pour
$\omega_0 = 1$, $\lambda = 0.1$~: $\omega_d = \sqrt{0.99} =
0.99499$, donc $\delta = \frac{0.62832}{0.99499} = 0.6315$~: chaque
oscillation conserve $\eu^{-0.63} \approx 53~\%$ de son amplitude.

\textbf{5.} Le facteur d'amplitude est $\eu^{-\lambda t}$, qui est
divisé par $\eu$ sur une durée $t = \frac1\lambda$. Cet intervalle
contient $\frac{1/\lambda}{T_d} = \frac{\omega_d}{2\pi\lambda}$
pseudo-périodes. Pour $\lambda \ll \omega_0$, $\omega_d \approx
\omega_0$ et cela vaut $\approx \frac{\omega_0}{2\pi\lambda} =
\frac Q\pi$. Une corde de guitare de $Q = 300$ vibre pendant environ
une centaine de périodes~; un ferme-porte de $Q = 1$ n'en achève même
pas une.

\textbf{6.} En substituant $z\,\eu^{\iu\Omega t}$ dans le membre de
gauche, on obtient $z\,(-\Omega^2 + 2\iu\lambda\Omega +
\omega_0^2)\,\eu^{\iu\Omega t}$, qui vaut $A\,\eu^{\iu\Omega t}$
exactement pour $z = \frac{A}{\omega_0^2 - \Omega^2 +
2\iu\lambda\Omega}$ (le dénominateur est non nul~: sa partie
imaginaire vaut $2\lambda\Omega > 0$). Comme les coefficients sont
réels, la partie réelle $x_p = \Re\bigl(z\eu^{\iu\Omega t}\bigr)$ est
solution de l'équation de second membre $\Re\bigl(A\eu^{\iu\Omega
t}\bigr) = A\cos\Omega t$.

\textbf{7.} Écrivons $\omega_0^2 - \Omega^2 + 2\iu\lambda\Omega =
\sqrt{D}\,\eu^{\iu\varphi}$ avec $D = (\omega_0^2 - \Omega^2)^2 +
4\lambda^2\Omega^2$ et $\varphi \in \intoo0\pi$ (la partie imaginaire
$2\lambda\Omega$ est positive), de sorte que $\tan\varphi =
\frac{2\lambda\Omega}{\omega_0^2 - \Omega^2}$. Alors $z =
\frac{A}{\sqrt D}\eu^{-\iu\varphi}$ et
\[
x_p(t) = \Re\Bigl(\frac A{\sqrt D}\,\eu^{\iu(\Omega t -
\varphi)}\Bigr) = R\cos(\Omega t - \varphi),
\qquad R = \frac A{\sqrt D} .
\]

\textbf{8.} Quand $\Omega \to 0^+$~: $D \to \omega_0^4$, donc $R \to
A/\omega_0^2$ et $\tan\varphi \to 0^+$ avec $\varphi \in
\intoo0{\frac\pi2}$~: $\varphi \to 0$. La masse suit la force de
manière quasi-statique, déplacée de force/raideur. Quand $\Omega \to
+\infty$~: $D \sim \Omega^4$, donc $R \sim A/\Omega^2 \to 0$, et
$\varphi \to \pi$ (le nombre complexe $\omega_0^2 - \Omega^2 +
2\iu\lambda\Omega$ gagne le deuxième quadrant avec un argument $\to
\pi$)~: la masse bouge à peine, et en opposition de phase --- l'inertie
domine.

\textbf{9.} D'après le \cref{thm:b1:diffeq:structure2}~(1), toute
solution s'écrit $x = x_p + x_h$ avec $x_h$ solution de $(H)$~;
d'après la question 2, $x_h(t) \to 0$, donc $x(t) - x_p(t) \to 0$~:
toutes les solutions convergent vers le même régime permanent. Les
conditions initiales ne façonnent que le transitoire.

\textbf{10.} Si $x$ est une solution périodique, $x - x_p = x_h$ est
une solution périodique de $(H)$ qui tend vers $0$ en $+\infty$~; une
fonction périodique de limite $0$ est identiquement nulle (ses valeurs
sur une période se répètent indéfiniment, donc toute valeur est limite
d'une suite extraite tendant vers $0$). Donc $x = x_p$.

\textbf{11.} Ici $\lambda = 1$, $\omega_0^2 = 2$, $\Omega = 1$,
$A = 1$~: $z = \frac1{2 - 1 + 2\iu} = \frac{1 - 2\iu}5$, donc
\[
x_p = \Re\Bigl(\frac{(1 - 2\iu)(\cos t + \iu\sin t)}5\Bigr)
= \frac{\cos t + 2\sin t}5 .
\]
Équation homogène~: les racines de $r^2 + 2r + 2$ sont $-1 \pm \iu$~:
$x_h = \eu^{-t}(C\cos t + S\sin t)$. Conditions~: $x(0) = \frac15 + C
= 0$ donne $C = -\frac15$~; en dérivant, $x'(0) = \frac25 - C + S = 0$
donne $S = C - \frac25 = -\frac35$. D'où
\[
x(t) = \underbrace{\frac{\cos t + 2\sin t}5}_{\text{permanent}}
- \underbrace{\eu^{-t}\,\frac{\cos t + 3\sin
t}5}_{\text{transitoire}} ,
\]
le transitoire s'éteignant comme $\eu^{-t}$.

\textbf{12.} En développant, $g(u) = u^2 - 2(\omega_0^2 - 2\lambda^2)u
+ \omega_0^4 = (u - u_r)^2 + g(u_r)$ avec $u_r = \omega_0^2 -
2\lambda^2$ et
\[
g(u_r) = \omega_0^4 - u_r^2 = (\omega_0^2 - u_r)(\omega_0^2 +
u_r) = 2\lambda^2\bigl(2\omega_0^2 - 2\lambda^2\bigr) =
4\lambda^2(\omega_0^2 - \lambda^2) .
\]
Si $2\lambda^2 < \omega_0^2$, alors $u_r > 0$ est un carré de
pulsation admissible~: $g$ y admet un minimum strict, donc $R = A/\sqrt
g$ admet un maximum strict en $\Omega_r = \sqrt{u_r} =
\sqrt{\omega_0^2 - 2\lambda^2}$, avec $R_{\max} = A/\sqrt{g(u_r)}
= \frac{A}{2\lambda\sqrt{\omega_0^2 - \lambda^2}}$.

\textbf{13.} $R(0) = A/\omega_0^2$, donc
\[
\frac{R_{\max}}{R(0)}
= \frac{\omega_0^2}{2\lambda\sqrt{\omega_0^2 - \lambda^2}}
= \frac{\omega_0}{2\lambda}\cdot
\frac{\omega_0}{\sqrt{\omega_0^2 - \lambda^2}}
= Q\,\Bigl(1 - \frac{\lambda^2}{\omega_0^2}\Bigr)^{-1/2}
\geq Q .
\]
Pour un amortissement faible, le facteur correctif est proche de $1$~:
la \omterm{ex:b1:diffeq:oscillation}{résonance} multiplie le déplacement statique par essentiellement
$Q$.

\textbf{14.} $V(\Omega)^2 = \frac{A^2 u}{g(u)}$ avec $u =
\Omega^2$. Sa dérivée a le signe de $g(u) - u\,g'(u) =
(\omega_0^2 - u)^2 + 4\lambda^2 u - u\bigl(2(u - \omega_0^2) +
4\lambda^2\bigr) = (\omega_0^2 - u)^2 + 2u(\omega_0^2 - u) =
(\omega_0^2 - u)(\omega_0^2 + u)$, positif pour $u < \omega_0^2$ et
négatif au-delà~: maximum strict exactement en $\Omega = \omega_0$,
et ce pour tout amortissement. Là, $\tan\varphi$ explose avec $\varphi
\in \intoo0\pi$~: $\varphi = \frac\pi2$, et $x_p'(t) =
-R\Omega\sin(\Omega t - \frac\pi2) = R\Omega\cos(\Omega t)$ est
exactement en phase avec la force~: transfert de puissance optimal.

\textbf{15.} $R = R_{\max}/\sqrt2 \iff g(u) = 2g(u_r) \iff (u -
u_r)^2 = g(u_r) = 4\lambda^2(\omega_0^2 - \lambda^2)$, d'où
\[
u_\pm = u_r \pm 2\lambda\sqrt{\omega_0^2 - \lambda^2},
\qquad
\Omega_+^2 - \Omega_-^2 = 4\lambda\sqrt{\omega_0^2 - \lambda^2} .
\]
Ensuite $\Omega_+ - \Omega_- = \frac{\Omega_+^2 -
\Omega_-^2}{\Omega_+ + \Omega_-}$, et pour $\lambda \ll \omega_0$ on a
$\Omega_\pm \approx \omega_0$ pour les deux~: $\Omega_+ - \Omega_-
\approx \frac{4\lambda\omega_0}{2\omega_0} = 2\lambda$, donc
$\frac{\omega_0}{\Omega_+ - \Omega_-} \approx
\frac{\omega_0}{2\lambda} = Q$. Mesurer la largeur d'un pic de
\omterm{ex:b1:diffeq:oscillation}{résonance}, c'est mesurer son \omterm{pb:b1:diffeq:1}{facteur de qualité}.

\textbf{16.} $Q = 10$~; $\Omega_r = \sqrt{1 - 2(0.05)^2} =
\sqrt{0.995} = 0.9975$~; $R_{\max} = \frac1{2 \times 0.05
\sqrt{1 - 0.0025}} = \frac1{0.1 \times 0.99875} = 10.01$~; réponse
statique $R(0) = 1$~; bande passante $\approx 2\lambda = 0.1$. Une
pointe haute et fine de hauteur $\approx Q$ au-dessus d'un plateau de
hauteur $1$.

\textbf{17.} Si $2\lambda^2 \geq \omega_0^2$ alors $u_r \leq 0$ et
$g'(u) = 2(u - u_r) > 0$ pour tout $u > 0$~: $g$ croît strictement sur
$\intoo0{+\infty}$, donc $R = A/\sqrt g$ décroît strictement à partir
de $R(0) = A/\omega_0^2$~: la réponse est maximale à pulsation nulle
et il n'y a aucun pic.

\textbf{18.} $\gamma = \iu\Omega$ n'est pas racine de $r^2 +
\omega_0^2$ (puisque $\Omega \neq \omega_0$), donc
la \cref{met:b1:diffeq:particular} avec $m = 0$ donne $x_p =
\frac{A\cos\Omega t}{\omega_0^2 - \Omega^2}$ (substituer et
vérifier~: $-\Omega^2 + \omega_0^2$ fois le cosinus). Solution
générale~:
\[
x(t) = \frac{A\cos\Omega t}{\omega_0^2 - \Omega^2}
+ \lambda_1\cos\omega_0 t + \mu_1\sin\omega_0 t .
\]

\textbf{19.} $x(0) = 0$ impose $\lambda_1 = -\frac A{\omega_0^2 -
\Omega^2}$, et $x'(0) = 0$ impose $\mu_1 = 0$~:
\[
x(t) = \frac{A}{\omega_0^2 - \Omega^2}\,
\bigl(\cos\Omega t - \cos\omega_0 t\bigr)
= \frac{2A}{\omega_0^2 - \Omega^2}\,
\sin\Bigl(\frac{(\omega_0 - \Omega)t}2\Bigr)
\sin\Bigl(\frac{(\omega_0 + \Omega)t}2\Bigr) ,
\]
par la formule de factorisation $\cos a - \cos b =
2\sin\frac{b - a}2\sin\frac{b + a}2$ appliquée avec $a = \Omega t$,
$b = \omega_0 t$.

\textbf{20.} Pour $\Omega$ proche de $\omega_0$, le second sinus
oscille à la pulsation rapide $\frac{\omega_0 + \Omega}2 \approx
\omega_0$, tandis que le premier est une enveloppe lente de pulsation
$\frac{\abs{\omega_0 - \Omega}}2$~: l'amplitude de l'oscillation
rapide croît et décroît avec une période d'enveloppe
$\frac{2\pi}{\abs{\omega_0 - \Omega}}$ (deux \omterm{pb:b1:diffeq:1}{battements} par période
d'enveloppe), atteignant des maxima $\frac{2A}{\abs{\omega_0^2 -
\Omega^2}}$. Quand $\Omega \to \omega_0$, les \omterm{pb:b1:diffeq:1}{battements} deviennent à
la fois plus lents (période $\to \infty$) et plus amples (amplitude
$\to \infty$).

\textbf{21.} Fixons $t$. Quand $\Omega \to \omega_0$~:
\[
\frac{2A}{\omega_0^2 - \Omega^2}\sin\Bigl(\frac{(\omega_0 -
\Omega)t}2\Bigr)
= \frac{2A}{\omega_0 + \Omega}\cdot
\frac{\sin\bigl(\frac{(\omega_0 - \Omega)t}2\bigr)}
{\omega_0 - \Omega}
\longrightarrow \frac{2A}{2\omega_0}\cdot\frac t2
= \frac{At}{2\omega_0} ,
\]
tandis que $\sin\bigl(\frac{(\omega_0 + \Omega)t}2\bigr) \to
\sin(\omega_0 t)$~: la limite est $x_\infty(t) =
\frac{At\sin\omega_0 t}{2\omega_0}$. Vérification directe~: avec $C =
\frac A{2\omega_0}$, $x_\infty = Ct\sin\omega_0 t$ vérifie
$x_\infty'' = 2C\omega_0\cos\omega_0 t - C\omega_0^2 t
\sin\omega_0 t$, donc $x_\infty'' + \omega_0^2 x_\infty =
2C\omega_0\cos\omega_0 t = A\cos\omega_0 t$, avec $x_\infty(0) =
0$ et $x_\infty'(0) = C\sin 0 + C\omega_0 \cdot 0 \cdot \cos 0 =
0$ --- pour $\omega_0 = A = 1$, c'est exactement
l'\cref{ex:b1:diffeq:oscillation}. La \omterm{ex:b1:diffeq:oscillation}{résonance} est la dégénérescence
des \omterm{pb:b1:diffeq:1}{battements}~: la première enflure de l'enveloppe, étirée jusqu'à
une longueur infinie.

\textbf{22.} Sans amortissement, l'amplitude résonante croît
linéairement et sans borne~; avec un amortissement $\lambda > 0$, la
croissance sature à $R_{\max} \approx Q\,\frac A{\omega_0^2}$. Le
mécanisme~: la dissipation retire de l'énergie à un rythme qui croît
avec l'amplitude (question 23), de sorte que la montée s'arrête
exactement lorsque l'amortissement brûle l'énergie aussi vite que
l'excitation la fournit.

\textbf{23.} Avec $x_p' = -R\Omega\sin(\Omega t - \varphi)$, sur une
période les moyennes $\langle\cos^2\rangle = \langle\sin^2\rangle =
\frac12$ et $\langle\sin\cos\rangle = 0$ donnent~:
\[
\langle P_{\mathrm{diss}}\rangle
= 2\lambda\,R^2\Omega^2\,\langle\sin^2(\Omega t -
\varphi)\rangle = \lambda R^2\Omega^2 ;
\]
et en développant $\sin(\Omega t - \varphi) = \sin\Omega t\cos\varphi
- \cos\Omega t\sin\varphi$~:
\[
\langle P_{\mathrm{in}}\rangle
= -AR\Omega\,\bigl\langle\cos\Omega t\,\sin(\Omega t -
\varphi)\bigr\rangle
= AR\Omega\,\frac{\sin\varphi}2 .
\]
Comme $\sin\varphi = \frac{2\lambda\Omega}{\sqrt D} =
\frac{2\lambda\Omega R}A$, cela vaut $\frac{AR\Omega}2 \cdot
\frac{2\lambda\Omega R}A = \lambda R^2\Omega^2$~: les puissances
injectée et dissipée s'équilibrent exactement --- la propriété
caractéristique d'un régime permanent.

\textbf{24.} (i) Le théorème de structure a séparé chaque solution en
régime permanent plus transitoire (questions 9 à 11) et a ramené
l'unicité au problème homogène. (ii) La méthode complexe a transformé
la recherche d'une solution particulière en une seule division de
nombres complexes (question 6), l'amplitude et la phase se lisant sur
un \omterm{def:b1:complex:field}{module} et un argument. (iii) La courbe de \omterm{ex:b1:diffeq:oscillation}{résonance} est une pure
étude de fonction --- un trinôme en $u = \Omega^2$, son minimum, ses
lignes de niveau --- dans le style du \cref{ch:b1:functions}
(questions 12 à 17).

\textbf{25.} Libre et amorti~: pseudo-oscillations décroissantes, durée
de vie $\frac1\lambda$, environ $\frac Q\pi$ oscillations. Forcé et
amorti~: les transitoires meurent, et un unique régime permanent
sinusoïdal survit à la pulsation d'excitation, avec une amplitude qui
culmine près de $\omega_0$ (hauteur $\approx Q \times$ la réponse
statique, largeur $\approx \frac{\omega_0}Q$) et une phase qui balaie
de $0$ à $\pi$ en passant par $\frac\pi2$ en $\omega_0$. Libre et non
amorti~: oscillation perpétuelle. Forcé et non amorti~: des
\omterm{pb:b1:diffeq:1}{battements}, dégénérant en une \omterm{ex:b1:diffeq:oscillation}{résonance} à croissance linéaire à
l'accord exact. Un seul nombre sans dimension, $Q =
\frac{\omega_0}{2\lambda}$, règle tout --- hauteur du pic, bande
passante et durée de vie du transitoire sont trois lectures du même
cadran. La suite~: réécrire $x'' + 2\lambda x' + \omega_0^2x$ comme un
système du premier ordre ouvre les méthodes matricielles du
\cref{ch:b1:matrices} et du volume de Licence 2, et décomposer une
excitation périodique quelconque en sinusoïdes (séries de Fourier,
volume de Licence 3) fait de l'analyse à une seule fréquence de ce
problème la brique universelle~: résoudre pour chaque fréquence, puis
superposer.
\end{solution}
