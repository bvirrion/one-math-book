\section*{Chapitre \ref{ch:b1:derivative} --- Dérivation}

\begin{solution}{exo:b1:derivative:1}
$x^x = \eu^{x\ln x}$ sur $\intoo{0}{+\infty}$~: \omterm{def:b1:derivative:def}{dérivée} $(\ln x +
1)\,x^x$.

$\ln(x + \sqrt{x^2+1})$ sur $\R$ (l'argument est toujours $> 0$)~:
\omterm{def:b1:derivative:def}{dérivée} $\frac{1}{\sqrt{x^2+1}}$ (calculée dans la
\cref{prop:b1:functions:invhyp} --- c'est
$\operatorname{arsinh}$).

$\arctan\frac1x$ sur $\R^*$~: \omterm{def:b1:derivative:def}{dérivée} $\frac{-1/x^2}{1 + 1/x^2} =
\frac{-1}{1 + x^2}$ (cohérent avec la
\cref{prop:b1:functions:arcidentities}~(2)~: la fonction vaut
$\pm\frac\pi2 - \arctan x$ sur chaque demi-droite).

$\sqrt{1 + \eu^{2x}}$ sur $\R$~: \omterm{def:b1:derivative:def}{dérivée}
$\frac{\eu^{2x}}{\sqrt{1 + \eu^{2x}}}$.
\end{solution}

\begin{solution}{exo:b1:derivative:2}
En $0$~: $\bigl|\frac{f(h) - 0}{h}\bigr| = \abs{h \sin\frac1h} \leq
\abs h \to 0$, donc $f'(0) = 0$. Pour $x \neq 0$, les règles usuelles donnent
$f'(x) = 2x\sin\frac1x - \cos\frac1x$. Le long de $x_n = \frac{1}{2\pi
n}$~: $f'(x_n) = 0 - 1 \to -1$~; le long de $y_n = \frac{1}{(2n+1)\pi}$~:
$f'(y_n) = 0 + 1 \to 1$. Deux suites tendant vers $0$ avec des limites
différentes de $f'$~: pas de limite (\cref{thm:b1:continuity:seqchar}), donc $f'$
n'est pas \omterm{def:b1:continuity:continuous}{continue} en $0$ et $f$ est \omterm{def:b1:derivative:def}{dérivable} sans être
$C^1$.
\end{solution}

\begin{solution}{exo:b1:derivative:3}
$\ln(1+x) < x$ pour $x > 0$~: inégalité de tangente issue de la concavité en $0$
(stricte hors du point de contact puisque $\ln$ est strictement concave~;
ou bien appliquer le \omterm{thm:b1:derivative:mvt}{théorème des accroissements finis}~: $\ln(1+x) = \frac{x}{1+c}$ pour un certain
$c \in \intoo{0}{x}$, et $\frac{x}{1+c} < x$). La même identité des
accroissements finis donne la minoration~: $\frac{x}{1+c} > \frac{x}{1+x}$.

Conséquence~: avec $x/n$ à la place de $x$,
\[
\frac{x/n}{1 + x/n} < \ln\Bigl(1 + \frac xn\Bigr) < \frac xn
\quad\implies\quad
\frac{x}{1 + x/n} < n \ln\Bigl(1 + \frac xn\Bigr) < x .
\]
Le membre de gauche tend vers $x$~: par encadrement, $n\ln(1 + \frac xn)
\to x$, et par \omterm{def:b1:continuity:continuous}{continuité} de $\exp$, $\bigl(1 + \frac xn\bigr)^n =
\eu^{n\ln(1 + x/n)} \to \eu^x$.
\end{solution}

\begin{solution}{exo:b1:derivative:4}
Soient $x_1 < x_2 < \dots < x_k$ des racines distinctes de $P$. Sur chaque
$\intcc{x_i}{x_{i+1}}$, Rolle (\cref{thm:b1:derivative:rolle})
produit $c_i \in \intoo{x_i}{x_{i+1}}$ avec $P'(c_i) = 0$~: cela fait
$k - 1$ racines de $P'$, distinctes car les \omterm{prop:b1:reals:intervals}{intervalles} \omterm{def:b1:topology:open}{ouverts} sont
disjoints --- et entrelacées par construction.

Si $P$ (de degré $n$) a toutes ses racines réelles, écrivons-les avec
leurs \omterm{def:b1:poly:derivative}{multiplicités} $m_1 + \dots + m_k = n$. Chaque racine de \omterm{def:b1:poly:derivative}{multiplicité}
$m_i \geq 2$ est racine de $P'$ de \omterm{def:b1:poly:derivative}{multiplicité} $m_i - 1$
(\cref{prop:b1:poly:multiplicity}), ce qui contribue pour $\sum (m_i - 1) = n
- k$~; Rolle contribue pour $k - 1$ de plus, toutes distinctes de celles-là. Total
$\geq n - 1 = \deg P'$~: toutes les racines de $P'$ sont réelles.
\end{solution}

\begin{solution}{exo:b1:derivative:5}
Fixons $\varepsilon > 0$ et $A$ tel que $\abs{f'(t) - \ell} \leq
\varepsilon$ pour $t \geq A$. Pour $x > A$, le \omterm{thm:b1:derivative:mvt}{théorème des accroissements finis} sur
$\intcc{A}{x}$ donne $c \in \intoo{A}{x}$ avec
\[
f(x) = f(A) + f'(c)(x - A),
\qquad\text{donc}\qquad
\Bigl|\frac{f(x)}{x} - \ell\Bigr|
\leq \frac{\abs{f(A)} + \abs\ell A}{x} + \abs{f'(c) - \ell}
\cdot\frac{x - A}{x} \leq \frac{C_A}{x} + \varepsilon .
\]
Pour $x$ grand, $\frac{C_A}{x} \leq \varepsilon$~: d'où
$\frac{f(x)}{x} \to \ell$.

Oui~: $f(x+1) - f(x) = f'(c_x)$ avec $c_x \in \intoo{x}{x+1}$ (\omterm{thm:b1:derivative:mvt}{théorème des
accroissements finis} sur $\intcc{x}{x+1}$), et $c_x \to +\infty$, donc
$f(x+1) - f(x) \to \ell$.
\end{solution}

\begin{solution}{exo:b1:derivative:6}
Récurrence sur $n$. Pour $n = 1$~: Rolle. Si l'énoncé vaut pour $n - 1$~:
$f$ s'annule en $n+1$ points, donc, Rolle appliqué sur les $n$ \omterm{prop:b1:reals:intervals}{intervalles}
intermédiaires, $f'$ s'annule en $n$ points distincts~; l'hypothèse de récurrence
appliquée à $f'$ ($n-1$ fois \omterm{def:b1:derivative:def}{dérivable}, $n$ zéros) fait
s'annuler $(f')^{(n-1)} = f^{(n)}$ quelque part.

Application~: si $P$ de degré $\leq n$ s'annule en $n+1$ points, alors
$P^{(n)}$, constante égale à $n!$ fois le coefficient dominant,
s'annule~: le coefficient dominant est $0$, et l'on conclut par
récurrence descendante (ou directement~: tous les coefficients s'annulent).
\end{solution}

\begin{solution}{exo:b1:derivative:7}
Par le \omterm{thm:b1:derivative:mvt}{théorème des accroissements finis} sur $\intcc{a}{x_0}$ et sur
$\intcc{x_0}{b}$~:
\[
f'(c_1) = \frac{f(x_0) - f(a)}{x_0 - a} = \frac{f(x_0)}{x_0 - a} > 0,
\qquad
f'(c_2) = \frac{f(b) - f(x_0)}{b - x_0} = \frac{-f(x_0)}{b - x_0} < 0,
\]
avec $c_1 < x_0 < c_2$. Alors le \omterm{thm:b1:derivative:mvt}{théorème des accroissements finis} appliqué à $f'$
sur $\intcc{c_1}{c_2}$ donne $c$ tel que
\[
f''(c) = \frac{f'(c_2) - f'(c_1)}{c_2 - c_1} < 0 . \qedhere
\]
\end{solution}

\begin{solution}{exo:b1:derivative:8}
$f'(x) = \frac{1 - \ln x}{x^2}$~: $f$ croît sur $\intoc{0}{\eu}$,
décroît sur $\intco{\eu}{+\infty}$, avec pour maximum $f(\eu) =
\frac1\eu$~; limites $-\infty$ en $0^+$ et $0$ en $+\infty$ (\omterm{prop:b1:functions:powerrules}{croissances
comparées}).

Pour $\eu \leq a < b$~: $f$ y étant strictement décroissante, on a
$\frac{\ln a}{a} > \frac{\ln b}{b}$, c'est-à-dire $b \ln a > a \ln b$,
c'est-à-dire $a^b > b^a$.

Avec $a = \eu < b = \pi$~: $\eu^\pi > \pi^\eu$.

Petits couples~: $(2, 3)$~: $2^3 = 8 < 9 = 3^2$ --- inversé~! La raison~:
$2 < \eu$, et sur $\intoo{0}{\eu}$ la fonction $f$ est
\emph{croissante}, de sorte que la comparaison bascule lorsque les deux nombres sont
en dessous de $\eu$, et est imprévisible de part et d'autre de $\eu$ ($f(2) = f(4)$
explique l'égalité $2^4 = 4^2 = 16$).
\end{solution}

\begin{solution}{exo:b1:derivative:9}
\emph{Jensen pour $\ln$, par récurrence sur $n$.} Affirmation~: pour des
$x_i$ strictement positifs et des poids $\lambda_i > 0$ avec $\sum \lambda_i = 1$~:
$\ln\bigl(\sum \lambda_i x_i\bigr) \geq \sum \lambda_i \ln x_i$. Pour
$n = 2$, c'est la concavité. Hérédité~: avec $\Lambda = \lambda_1 + \dots +
\lambda_{n-1} = 1 - \lambda_n$ et $y = \sum_{i<n}
\frac{\lambda_i}{\Lambda} x_i$,
\[
\ln\Bigl(\sum_{i \leq n} \lambda_i x_i\Bigr)
= \ln\bigl(\Lambda y + \lambda_n x_n\bigr)
\geq \Lambda \ln y + \lambda_n \ln x_n
\geq \Lambda \sum_{i<n} \frac{\lambda_i}{\Lambda}\ln x_i
+ \lambda_n \ln x_n,
\]
en utilisant la concavité ($n = 2$) puis l'hypothèse de récurrence.

Avec $\lambda_i = \frac 1n$ et $x_i = a_i$~: $\ln\frac{\sum a_i}{n}
\geq \frac 1n \sum \ln a_i = \ln\sqrt[n]{a_1\cdots a_n}$~;
il ne reste qu'à passer à l'exponentielle. Égalité~: $\ln$ est \emph{strictement} concave ($\ln'' < 0$),
donc l'égalité à chaque étape force les points moyennés à coïncider ---
c'est-à-dire tous les $a_i$ égaux~; et s'ils sont tous égaux, l'égalité est claire.
\end{solution}

\begin{solution}{exo:b1:derivative:10}
Posons $g(x) = f(x) - vx$~: \omterm{def:b1:derivative:def}{dérivable}, avec $g'(a) = f'(a) - v < 0$
et $g'(b) = f'(b) - v > 0$. Par le \omterm{thm:b1:continuity:evt}{théorème des bornes atteintes}, $g$
atteint son minimum sur $\intcc{a}{b}$ en un certain $c$. Ce n'est pas en $a$~:
puisque $g'(a) < 0$, les points juste à droite de $a$ vérifient $g < g(a)$. Ce n'est
pas en $b$~: puisque $g'(b) > 0$, les points juste à gauche de $b$ vérifient $g <
g(b)$. Donc $c$ est \omterm{def:b1:topology:closure}{intérieur}, et la \cref{prop:b1:derivative:fermat}
donne $g'(c) = 0$, c'est-à-dire $f'(c) = v$.
\end{solution}

\begin{solution}{exo:b1:derivative:11}
\emph{Unicité~:} deux points fixes $\ell \neq \ell'$ donneraient
$\abs{\ell - \ell'} = \abs{f(\ell) - f(\ell')} \leq k\abs{\ell -
\ell'} < \abs{\ell - \ell'}$, absurde.

\emph{Existence~:} $g(x) = f(x) - x$ vérifie, par l'inégalité des
accroissements finis, $f(x) \leq f(0) + k\abs x$~; donc pour $x \geq
\frac{\abs{f(0)}}{1 - k}$, $g(x) \leq f(0) + kx - x \leq 0$, et
symétriquement $g(-x) \geq 0$ pour $x$ grand. Le \omterm{thm:b1:continuity:ivt}{théorème des valeurs
intermédiaires} donne un zéro $\ell$ de $g$~: un point fixe.

\emph{Convergence~:} de nouveau l'inégalité des accroissements finis~:
\[
\abs{u_{n+1} - \ell} = \abs{f(u_n) - f(\ell)} \leq k\abs{u_n - \ell},
\]
donc par récurrence $\abs{u_n - \ell} \leq k^n \abs{u_0 - \ell} \to 0$.
\end{solution}

\begin{solution}{exo:b1:derivative:12}
\begin{enumerate}
  \item Si $g(b) = g(a)$, Rolle donnerait un zéro \omterm{def:b1:topology:closure}{intérieur} de
        $g'$~: exclu. Posons $\lambda = \frac{f(b) - f(a)}{g(b) -
        g(a)}$ et $h = f - \lambda g$~: $h$ est \omterm{def:b1:continuity:continuous}{continue} sur
        $\intcc{a}{b}$, \omterm{def:b1:derivative:def}{dérivable} à l'\omterm{def:b1:topology:closure}{intérieur}, et $h(b) - h(a)
        = f(b) - f(a) - \lambda(g(b) - g(a)) = 0$. Rolle
        fournit $c$ tel que $h'(c) = 0$, c'est-à-dire $f'(c) =
        \lambda\,g'(c)$~; il ne reste qu'à diviser par $g'(c) \neq 0$.
  \item Pour $x > a$ proche de $a$, la partie (1) sur $\intcc{a}{x}$
        (où $g' \neq 0$) donne $g(x) \neq 0$ et $c_x \in
        \intoo{a}{x}$ avec
        \[
        \frac{f(x)}{g(x)} = \frac{f(x) - f(a)}{g(x) - g(a)}
        = \frac{f'(c_x)}{g'(c_x)} .
        \]
        Quand $x \to a^+$, $c_x \to a^+$ (encadrement), donc le membre
        de droite tend vers $\ell$~: $\frac{f}{g} \to \ell$.
  \item $\frac{f(x)}{g(x)} = x\sin\frac1x \to 0$, tandis que
        $\frac{f'(x)}{g'(x)} = 2x\sin\frac1x - \cos\frac1x$ n'a
        pas de limite en $0$ (\cref{exo:b1:derivative:2})~: la
        règle de l'Hospital ne transfère l'information que de
        $\frac{f'}{g'}$ vers $\frac fg$, jamais en sens inverse.
\end{enumerate}
\end{solution}

\begin{solution}{pb:b1:derivative:1}
\textbf{1.} $\bigl|\frac ab - \frac pq\bigr| = \frac{\abs{aq -
bp}}{bq}$, et $aq - bp$ est un entier non nul lorsque les fractions
diffèrent~: la distance est $\geq \frac{1}{bq}$. Donc aucun rationnel
autre que $x$ lui-même n'entre dans l'\omterm{prop:b1:reals:intervals}{intervalle} épointé de rayon
$\frac{1}{bq}$ autour de $x = \frac ab$.

\textbf{2.} Si $\bigl|\sqrt2 - \frac pq\bigr| \geq 1 \geq
\frac{1}{4q^2}$, c'est fini. Sinon $\frac pq \in \intoo{\sqrt2 -
1}{\sqrt2 + 1}$, donc $0 < \sqrt2 + \frac pq < 2\sqrt2 + 1 < 4$.
Puisque $\sqrt 2 \notin \Q$, $p^2 - 2q^2$ est un entier non nul, et
\[
\Bigl|\sqrt2 - \frac pq\Bigr|
= \frac{\abs{2q^2 - p^2}}{q^2\,\bigl(\sqrt2 + \frac pq\bigr)}
\geq \frac{1}{4q^2} .
\]

\textbf{3.} $(p + 2q)^2 - 2(p + q)^2 = -(p^2 - 2q^2)$~: la valeur
$\pm1$ se propage. À partir de $(1,1)$~:
\[
(3,2),\ (7,5),\ (17,12),\ (41,29),\ (99,70),
\]
avec $p^2 - 2q^2$ alternant $-1, +1, \dots$
Pour ces couples, $\frac pq \geq 1$, donc $\sqrt2 + \frac pq > 2$ et
\[
\Bigl|\sqrt2 - \frac pq\Bigr| =
\frac{1}{q^2(\sqrt2 + p/q)} < \frac{1}{2q^2} ,
\]
avec $q \to \infty$~: une infinité d'approximations d'ordre $2$.
Avec la question 2, l'exposant $2$ est exact pour $\sqrt 2$.

\textbf{4.} Les $N + 1$ nombres $kx - \lfloor kx\rfloor$ ($0
\leq k \leq N$) sont dans les $N$ boîtes $\intco{\frac
jN}{\frac{j+1}{N}}$~: deux partagent une boîte
(\cref{cor:b1:counting:pigeonhole}), disons pour $i < j$. Avec $q =
j - i \leq N$ et $p = \lfloor jx\rfloor - \lfloor ix\rfloor$~:
$\abs{qx - p} < \frac1N$, donc $\bigl|x - \frac pq\bigr| <
\frac{1}{Nq} \leq \frac{1}{q^2}$. En faisant $N \to \infty$~: puisque
$x$ est irrationnel, chaque fraction fixée est à distance strictement positive de
$x$, tandis que $\frac{1}{Nq} \leq \frac 1N \to 0$ force de nouvelles
fractions à apparaître~: une infinité de $\frac pq$ distincts avec
$\bigl|x - \frac pq\bigr| < \frac{1}{q^2}$.

\textbf{5.} Partons d'un \omterm{def:b1:poly:def}{polynôme} non nul $P_0$ à coefficients entiers avec $P_0(x) =
0$. Si $P_0$ a une racine rationnelle $\frac ab$, le théorème de factorisation
(\cref{thm:b1:poly:factor}) écrit $P_0 = \bigl(X - \frac
ab\bigr)Q$ avec $Q \in \Q[X]$~;
puisque $x \neq \frac ab$ ($x$ irrationnel), $Q(x) = 0$, et
chasser les dénominateurs donne un \omterm{def:b1:poly:def}{polynôme} \emph{à coefficients entiers} non nul
de degré plus petit s'annulant en $x$. Le degré décroît à chaque
étape, donc le processus s'arrête~: on atteint $P \in \Z[X]$, $P(x) = 0$,
sans racine rationnelle, d'un certain degré $d$. Si $d \leq 1$, $P =
uX + v$ rendrait $x = -\frac vu$ rationnel~: donc $d \geq 2$.

\textbf{6.} $q^d\,P\bigl(\frac pq\bigr) = a_d p^d + a_{d-1}
p^{d-1} q + \dots + a_0 q^d$ est un entier, et il est non nul
parce que $P$ n'a pas de racine rationnelle~: $\bigl|P\bigl(\frac
pq\bigr)\bigr| \geq q^{-d}$.

\textbf{7.} Remarquons que $M > 0$~: $P'$ est un \omterm{def:b1:poly:def}{polynôme} non nul ($d \geq
2$), il ne peut donc pas s'annuler identiquement sur
$\intcc{x-1}{x+1}$. Si $\bigl|x - \frac pq\bigr| > 1$, alors la quantité
dépasse trivialement $\frac{C}{q^d}$. Sinon $\frac pq \in
\intcc{x-1}{x+1}$ et le \omterm{thm:b1:derivative:mvt}{théorème des accroissements finis}
(\cref{thm:b1:derivative:mvt}) donne $c$ entre $x$ et
$\frac pq$ tel que
\[
\Bigl|P\Bigl(\frac pq\Bigr)\Bigr|
= \Bigl|P\Bigl(\frac pq\Bigr) - P(x)\Bigr|
= \abs{P'(c)}\,\Bigl|x - \frac pq\Bigr|
\leq M\,\Bigl|x - \frac pq\Bigr| ,
\]
donc, avec la question 6~: $\bigl|x - \frac pq\bigr| \geq
\frac{1}{Mq^d} \geq \frac{C}{q^d}$.

\textbf{8.} Supposons que $x = \frac ab$ soit de Liouville. Choisissons $n$ tel que
$2^{n-1} > b$ et le $\frac pq$ correspondant, $q \geq 2$~:
\[
0 < \Bigl|x - \frac pq\Bigr| < \frac{1}{q^n}
= \frac{1}{q^{n-1}\,q} \leq \frac{1}{2^{n-1} q} <
\frac{1}{bq} ,
\]
en contradiction avec la question 1. Les nombres de Liouville sont donc irrationnels.

\textbf{9.} Si un nombre de Liouville $x$ était algébrique~: il est irrationnel
(question 8), donc les questions 5 à 7 fournissent $d \geq 2$ et $C > 0$
avec $\bigl|x - \frac pq\bigr| \geq \frac{C}{q^d}$ toujours. Pour
chaque $n$, l'approximant de Liouville donne $\frac{C}{q^d} <
q^{-n}$, c'est-à-dire $C < q^{d-n} \leq 2^{d-n}$ (car $q \geq 2$). Pour
$n$ grand, $2^{d-n} < C$~: contradiction. Les nombres de Liouville sont
\omterm{pb:b1:logic:1}{transcendants}.

\textbf{10.} Des uns aux positions $1, 2, 6, 24$~; tous les autres chiffres
parmi les $25$ premiers sont nuls~:
\[
L = 0.1100010000\,0000000000\,00010\dots
\]

\textbf{11.} Pour $m > k$, les positions $n!$ avec $n > k$ sont des
entiers distincts $\geq (k+1)!$, donc la somme géométrique finie
donne
\[
t_m - t_k = \sum_{n=k+1}^{m} 10^{-n!}
\leq \sum_{j = (k+1)!}^{m!} 10^{-j}
< 10^{-(k+1)!}\,\frac{1}{1 - \frac1{10}}
= \frac{10}{9}\,10^{-(k+1)!} ;
\]
en prenant la \omterm{def:b1:reals:bounds}{borne supérieure} sur $m$~: $L - t_k \leq
\frac{10}{9}10^{-(k+1)!} < 2\cdot10^{-(k+1)!}$. Minoration~: $L
\geq t_{k+1} = t_k + 10^{-(k+1)!}$.

\textbf{12.} $p_k = 10^{k!}\,t_k \in \N$, $q_k = 10^{k!}$, et
$(k+1)! = (k+1)\,k!$ donne $10^{-(k+1)!} = q_k^{-(k+1)}$~:
la question 11 s'écrit
\[
0 < L - \frac{p_k}{q_k} < \frac{2}{q_k^{\,k+1}} .
\]
Étant donné $n$~: pour $k \geq n$, $2\,q_k^{-(k+1)} \leq q_k^{-n}$
(en effet $q_k^{\,k+1-n} \geq q_k \geq 10 > 2$), et $q_k \geq 2$~:
la définition de la question 8 est satisfaite. $L$ est un nombre de Liouville.

\textbf{13.} Par la question 9, $L$ est \omterm{pb:b1:logic:1}{transcendant} --- le premier
nombre de l'histoire dont la transcendance ait été démontrée (Liouville, 1844).
Recoupement par les chiffres~: la suite a une infinité de uns avec des
écarts consécutifs $(k+1)! - k! = k\cdot k! \to \infty$, elle n'est donc
pas périodique à partir d'un certain rang, et $L \notin \Q$ par le critère
de périodicité du \cref{pb:b1:reals:1} (question 18) --- cohérent.

\textbf{14.} Avec des chiffres $d_n \in \intint{1}{9}$ aux
positions factorielles~: la majoration du reste de la question 11 est multipliée par au
plus $9$~: $0 < L' - t'_k \leq 9\cdot\frac{10}{9}\,10^{-(k+1)!} =
10\,q_k^{-(k+1)}$ (positivité parce que le chiffre à la position
$(k+1)!$ est non nul). Pour $k \geq n$~: $10\,q_k^{-(k+1)} \leq
q_k^{-n}$ puisque $q_k^{\,k+1-n} \geq 10$~: de nouveau un nombre de
Liouville, donc \omterm{pb:b1:logic:1}{transcendant}. Ces valeurs sont deux à deux
distinctes pour des choix de chiffres distincts (les suites sont propres ---
les zéros y abondent --- et les suites propres déterminent leur valeur,
\cref{pb:b1:reals:1}, question 10). Étant donné une liste quelconque $k \mapsto
x_k$ de ces nombres, choisissons le $k$-ième chiffre factoriel dans
$\intint{1}{9}$ différent de celui de $x_k$~: on obtient un nombre de
la même forme absent de la liste. Une infinité non dénombrable de
\omterm{pb:b1:logic:1}{transcendants} explicites.

\textbf{15.} D'abord un lemme~: \emph{si $\bigl|x - \frac pq\bigr|
\geq \frac{C}{q^s}$ pour tout $\frac pq \neq x$, alors $x$ n'est
approchable à aucun ordre $\mu > s$.} En effet, une infinité de
$\frac pq \neq x$ avec $\bigl|x - \frac pq\bigr| <
\frac{c}{q^\mu}$ forcerait $\frac{C}{q^s} < \frac{c}{q^\mu}$,
c'est-à-dire $q^{\mu - s} < \frac cC$~: les $q$ sont bornés, et seul un nombre
borné de fractions se trouve à distance au plus $1$ de $x$ ---
un nombre fini de candidats, non une infinité. Passons à la
hiérarchie~: les rationnels sont approchables à l'ordre $1$ ($\frac
pq$ avec $p = \lfloor qx\rfloor + 1$ donne une erreur $\leq \frac1q <
\frac2q$) et à aucun ordre $\mu > 1$ (la question 1 donne
l'hypothèse du lemme avec $s = 1$, $C = \frac1b$)~; $\sqrt2$~:
ordre $2$ (question 3) et pas plus (question 2 et le lemme)~;
tout irrationnel~: au moins $2$ (question 4)~; un algébrique de degré
$d$~: au plus $d$ (question 7 et le lemme)~; les nombres de Liouville~:
tout ordre (la formule affichée de la question 12, avec $c = 2$).

\textbf{16.} Avec $r = \frac ab$~: $\frac{p_k}{q_k} + \frac ab =
\frac{b p_k + a q_k}{b q_k} =: \frac{P_k}{Q_k}$, $Q_k = b q_k
\geq 2$, et
\[
\Bigl|(L + r) - \frac{P_k}{Q_k}\Bigr| = L - \frac{p_k}{q_k}
< 2\,q_k^{-(k+1)} = 2\,b^{\,k+1} Q_k^{-(k+1)} .
\]
Étant donné $n$~: pour $k$ grand, $Q_k^{\,k+1-n} \geq Q_k = b\,10^{k!}
\geq 2\,b^{\,k+1}$ (la factorielle écrase la puissance), donc
l'erreur est $< Q_k^{-n}$~: $L + r$ est de Liouville. Puisque $\Q$ est \omterm{def:b1:topology:dense}{dense}
et que chaque $L + r$ est \omterm{pb:b1:logic:1}{transcendant}, les \omterm{pb:b1:logic:1}{nombres transcendants} sont
\omterm{def:b1:topology:dense}{denses} dans $\R$.

\textbf{17.} Pour $h \geq 1$ il n'y a qu'un nombre fini de $P \in
\Z[X]$ avec $\deg P + \sum_i \abs{a_i} \leq h$ (degré $\leq h$
et chaque coefficient dans $\intint{-h}{h}$~: au plus
$(2h+1)^{h+1}$). Tout \omterm{def:b1:poly:def}{polynôme} non nul à coefficients entiers a une telle
hauteur, et a au plus $\deg P$ racines réelles~: les \omterm{pb:b1:logic:1}{nombres
algébriques} forment une réunion dénombrable (sur $h$) d'\omterm{def:b1:counting:card}{ensembles finis}, ils
peuvent donc être listés en une seule suite. Si les \omterm{pb:b1:logic:1}{transcendants} pouvaient
eux aussi être listés, entrelacer les deux listes listerait $\R$,
en contradiction avec le \cref{pb:b1:reals:1} (question 22). Donc les \omterm{pb:b1:logic:1}{nombres
transcendants} forment un \omterm{def:b1:logic:sets}{ensemble} non dénombrable. Comparaison~:
Cantor démontre que \emph{la plupart} des réels sont \omterm{pb:b1:logic:1}{transcendants} sans en exhiber
aucun~; Liouville en exhibe un, avec des constantes effectives (partie V)
--- existence par abondance contre existence par construction.

\textbf{18.} D'après la question 2, l'hypothèse du lemme est vérifiée avec $s
= 2$, $C = \frac14$. Si $\sqrt2$ était de Liouville, alors pour $n =
3$~: $\frac{1}{4q^2} < q^{-3}$ force $q < 4$, donc $q \in \{2,
3\}$~; seul un nombre fini de $\frac pq$ avec ces $q$ se trouve à distance
au plus $1$ de $\sqrt2$, chacun à une distance strictement positive
$\geq \varepsilon_0$ ($\sqrt2$ irrationnel)~; choisir $n$ tel que
$2^{-n} < \varepsilon_0$ ne laisse plus aucun $\frac pq$ admissible~:
contradiction. Le même argument avec $\frac{C}{q^d}$ montre qu'aucun
\omterm{pb:b1:logic:1}{nombre algébrique} n'est de Liouville --- la question 9 en habits effectifs.

\textbf{19.} $99^2 - 2\cdot70^2 = 9801 - 9800 = 1$. D'où
\[
\sqrt2 - \frac{99}{70} =
\frac{2 - (99/70)^2}{\sqrt2 + 99/70}
= \frac{-1}{4900\,\bigl(\sqrt2 + \tfrac{99}{70}\bigr)} ,
\qquad
\Bigl|\sqrt2 - \frac{99}{70}\Bigr|
= \frac{1}{4900 \times 2.8284\dots} \approx 7.2\cdot10^{-5} :
\]
$\frac{99}{70} = 1.414285\dots$ contre $\sqrt2 =
1.414213\dots$ --- cinq chiffres corrects.

\textbf{20.} Test des racines rationnelles pour $P = X^3 - 2$~: les candidats
$\pm1, \pm2, \pm\frac12$ ne sont pas racines. Donc $d = 3$ et la partie II
s'applique à $x = 2^{1/3} = 1.2599\dots$ Sur $\intcc{x - 1}{x + 1}
\subseteq \intcc{0.25}{2.26}$~: $\abs{P'(t)} = 3t^2 \leq 3\,(1 +
2^{1/3})^2 < 3\times(2.26)^2 = 15.32 < 16$, donc $M < 16$ et $C
\geq \frac{1}{16}$~:
\[
\Bigl|2^{1/3} - \frac pq\Bigr| \geq \frac{1}{16\,q^3}
\qquad\text{pour tout rationnel.}
\]

\textbf{21.} Si $\bigl|2^{1/3} - \frac pq\bigr| < 10^{-6}$, alors
$\frac{1}{16 q^3} < 10^{-6}$, c'est-à-dire $q^3 > \frac{10^6}{16} =
62\,500$~; puisque $39^3 = 59\,319 < 62\,500 \leq 64\,000 = 40^3$~:
$q \geq 40$.

\textbf{22.} Refaisons tourner la partie III en base $2$~: $B = \sup_k \sum_{n\leq
k} 2^{-n!}$, $q_k = 2^{k!}$, et le reste géométrique (de raison
$\frac12$) donne $0 < B - \frac{p_k}{q_k} \leq 2\cdot2^{-(k+1)!}
= 2\,q_k^{-(k+1)} \leq q_k^{-n}$ pour $k \geq n$. Donc $B$ est de
Liouville, donc \omterm{pb:b1:logic:1}{transcendant}~: rien dans l'argument n'est
décimal.

\textbf{23.} Avec des uns aux positions $3^k$~: $q_k = 10^{3^k}$ et
la majoration du reste donne $0 < x^\dagger - \frac{p_k}{q_k} <
2\cdot10^{-3^{k+1}} = 2\,q_k^{-3}$ (car $3^{k+1} = 3\cdot3^k$)~:
une infinité d'approximations d'ordre $3$. Par le lemme de la
question 15~: l'ordre $3 > 1$ exclut la rationalité, et l'ordre $3 >
2$ exclut d'être un irrationnel quadratique (dont l'\omterm{pb:b1:derivative:1}{inégalité de
Liouville} a $s = d = 2$). Mais un \omterm{pb:b1:logic:1}{nombre algébrique} de degré
$\geq 3$ n'est repoussé qu'à l'ordre $d \geq 3$~: la méthode de
Liouville ne peut pas séparer $x^\dagger$ des cubiques. L'écart est
comblé par le théorème de Roth --- tout irrationnel algébrique a un
ordre d'approximation exactement égal à $2$ --- un résultat du vingtième siècle
bien au-delà de ce volume~; en l'admettant, $x^\dagger$ est lui aussi
\omterm{pb:b1:logic:1}{transcendant}.

\textbf{24.} Il y a au plus $(2H+1)^{d+1}$ uplets $(a_0,
\dots, a_d)$ à entrées dans $\intint{-H}{H}$, et chaque \omterm{def:b1:poly:def}{polynôme}
non nul parmi eux a au plus $d$ racines réelles~: au plus
$d\,(2H+1)^{d+1}$ \omterm{pb:b1:logic:1}{nombres algébriques} apparaissent --- la finitude
qui a permis à la question 17 de tous les énumérer.

\textbf{25.} (i) L'unique ingrédient analytique est le \omterm{thm:b1:derivative:mvt}{théorème des
accroissements finis}, à la question 7, qui convertit l'annulation $P(x) =
0$ en la répulsion lipschitzienne $\abs{P(p/q)} \leq M\abs{x -
p/q}$. (ii) La tension~: le caractère entier pousse $\abs{P(p/q)}$ vers le haut,
jusqu'à $q^{-d}$, la régularité le tire vers le bas, jusqu'à $M\abs{x - p/q}$ --- un
rationnel trop proche de $x$ serait écrasé entre les deux.
(iii) Liouville construit un \omterm{pb:b1:logic:1}{transcendant} avec des constantes effectives~;
Cantor montre que presque tous les réels sont \omterm{pb:b1:logic:1}{transcendants} sans en nommer
aucun~: construction contre cardinalité. (iv) Le
devoir maison du \cref{ch:b1:integration} démontre
l'irrationalité de $\pi$ par le même étau --- une intégrale qui
serait un entier strictement positif tout en étant piégée dans $\intoo{0}{1}$ ---
avec l'intégration remplaçant la dérivation comme moitié analytique.
\end{solution}
