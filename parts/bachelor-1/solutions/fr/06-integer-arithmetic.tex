\section*{Chapitre \ref{ch:b1:arith} --- Arithmétique des entiers}

\begin{solution}{exo:b1:arith:1}
$1001 = 1 \times 777 + 224$~; $777 = 3 \times 224 + 105$~; $224 = 2
\times 105 + 14$~; $105 = 7 \times 14 + 7$~; $14 = 2 \times 7 + 0$. Donc
$\gcd(1001, 777) = 7$. En remontant~:
\[
7 = 105 - 7 \times 14
= 105 - 7(224 - 2\times 105) = 15 \times 105 - 7 \times 224
\]
\[
= 15(777 - 3\times 224) - 7\times 224 = 15 \times 777 - 52 \times 224
= 15 \times 777 - 52(1001 - 777) = 67 \times 777 - 52 \times 1001 .
\]
Vérification~: $67 \times 777 = 52\,059$ et $52 \times 1001 = 52\,052$~;
différence $7$. Couple de Bézout~: $(u, v) = (-52, 67)$ pour $1001u + 777v =
7$.
\end{solution}

\begin{solution}{exo:b1:arith:2}
Comme $10 \equiv 1 \pmod 9$~: $10^k \equiv 1$, donc $\sum_k d_k 10^k
\equiv \sum_k d_k \pmod 9$. Comme $10 \equiv -1 \pmod{11}$~: $10^k
\equiv (-1)^k$, donc l'entier est \omterm{def:b1:arith:congruence}{congru} à la somme alternée
$\sum_k (-1)^k d_k$ modulo $11$ (en partant du chiffre des \emph{unités}
avec le signe $+$).

$123\,456\,789$~: somme des chiffres $45 \equiv 0 \pmod 9$. Somme alternée à partir
des unités~: $9 - 8 + 7 - 6 + 5 - 4 + 3 - 2 + 1 = 5$, donc le nombre est
$\equiv 5 \pmod{11}$.
\end{solution}

\begin{solution}{exo:b1:arith:3}
Euclide~: $237 = 2 \times 91 + 55$~; $91 = 1 \times 55 + 36$~; $55 = 1
\times 36 + 19$~; $36 = 1 \times 19 + 17$~; $19 = 1 \times 17 + 2$~; $17
= 8 \times 2 + 1$. En remontant~:
\[
1 = 17 - 8\times 2 = 17 - 8(19 - 17) = 9\times 17 - 8\times 19
= 9(36 - 19) - 8\times 19 = 9\times 36 - 17\times 19
\]
\[
= 9\times 36 - 17(55 - 36) = 26\times 36 - 17\times 55
= 26(91 - 55) - 17\times 55 = 26\times 91 - 43\times 55
\]
\[
= 26\times 91 - 43(237 - 2\times 91) = 112 \times 91 - 43 \times 237.
\]
Donc $91 \times 112 \equiv 1 \pmod{237}$~: les solutions sont $x \equiv
112 \pmod{237}$. (Vérification~: $91 \times 112 = 10\,192 = 43 \times 237 +
1$.)
\end{solution}

\begin{solution}{exo:b1:arith:4}
$\gcd(17, 39) = 1$~: Euclide donne $39 = 2\times 17 + 5$, $17 = 3\times
5 + 2$, $5 = 2\times 2 + 1$, et en remontant
\[
1 = 5 - 2\times 2 = 5 - 2(17 - 3\times 5) = 7\times 5 - 2\times 17
= 7(39 - 2\times 17) - 2\times 17 = 7\times 39 - 16\times 17 .
\]
Solution particulière $(x_0, y_0) = (-16, 7)$. Solution générale de
l'équation homogène $17x + 39y = 0$~: $x = 39k$, $y = -17k$ (car
$17 \mid 39y$ et $\gcd(17,39) = 1$ imposent $17 \mid y$ ---
\omterm{thm:b1:arith:gauss}{lemme de Gauss}). D'où
\[
(x, y) = (-16 + 39k,\; 7 - 17k), \qquad k \in \Z .
\]
Pour le second membre $5$, on multiplie la solution particulière par $5$~:
$(x, y) = (-80 + 39k,\; 35 - 17k)$, $k \in \Z$.
\end{solution}

\begin{solution}{exo:b1:arith:5}
Pour tout \omterm{def:b1:arith:prime}{nombre premier} $p$, avec $\alpha = v_p(a)$ et $\beta = v_p(b)$~:
\[
v_p\bigl(\gcd(a,b)\bigr) + v_p\bigl(\operatorname{lcm}(a,b)\bigr)
= \min(\alpha, \beta) + \max(\alpha, \beta)
= \alpha + \beta = v_p(ab) .
\]
Deux entiers strictement positifs ayant la même valuation en tout \omterm{def:b1:arith:prime}{nombre premier} sont égaux
(\cref{prop:b1:arith:valuation}), donc $\gcd(a,b)\operatorname{lcm}(a,b)
= ab$.
\end{solution}

\begin{solution}{exo:b1:arith:6}
Valuations~: $\gcd(a, b) = 2^{\min(10,6)} 3^{\min(4,7)} 5^{\min(2,0)}
7^{\min(0,1)} = 2^6\, 3^4 = 5184$~;
$\operatorname{lcm}(a,b) = 2^{10}\, 3^7\, 5^2\, 7$.

Nombre de diviseurs~: un diviseur positif de $n = \prod p_i^{\alpha_i}$ est
exactement un choix $\prod p_i^{\beta_i}$ avec $0 \leq \beta_i \leq
\alpha_i$ (\cref{prop:b1:arith:valuation})~; les choix sont
indépendants, donc il y a $\prod_i (\alpha_i + 1)$ diviseurs. Pour $a$~:
$(10+1)(4+1)(2+1) = 165$.
\end{solution}

\begin{solution}{exo:b1:arith:7}
Supposons $\sqrt p = \frac rq$ avec $r, q \in \N^*$, c'est-à-dire $p q^2 =
r^2$. Appliquons $v_p$~: $v_p(pq^2) = 1 + 2v_p(q)$ est impair, tandis que $v_p(r^2)
= 2 v_p(r)$ est pair. Un entier ne peut avoir à la fois une valuation
$p$-adique paire et impaire~: contradiction. Donc $\sqrt p \notin \Q$.
\end{solution}

\begin{solution}{exo:b1:arith:8}
\emph{Fait général.} Avec $\gcd(m,n) = 1$, Bézout donne $mu + nv = 1$.
Posons $x_0 = b\,mu + a\,nv$. Alors $x_0 \equiv a\,nv \equiv a(1 - mu)
\equiv a \pmod m$ et de même $x_0 \equiv b \pmod n$~: existence. Si
$x$ et $x'$ sont deux solutions, $m$ et $n$ \omterm{def:b1:arith:divides}{divisent} $x - x'$, donc
$mn \mid x - x'$ (\cref{thm:b1:arith:gauss}~(2))~: unicité modulo
$mn$.

\emph{Numériquement~:} $m = 7$, $n = 11$~: $7 \times (-3) + 11 \times 2 =
1$. Donc $x_0 = 5 \times 7 \times (-3) + 2 \times 11 \times 2 = -105 +
44 = -61 \equiv 16 \pmod{77}$. Vérification~: $16 = 2\times 7 + 2 \equiv 2
\pmod 7$~; $16 = 11 + 5 \equiv 5 \pmod{11}$. Solutions~: $x \equiv 16
\pmod{77}$.
\end{solution}

\begin{solution}{exo:b1:arith:9}
Modulo $7$~: Fermat donne $3^6 \equiv 1$, et $1000 = 6 \times 166 + 4$,
donc $3^{1000} \equiv 3^4 = 81 \equiv 4 \pmod 7$.

Deux derniers chiffres de $7^{100}$~: travaillons modulo $4$ et modulo $25$. Modulo $4$~: $7
\equiv -1$, donc $7^{100} \equiv 1$. Modulo $25$~: $7^2 = 49 \equiv -1$, donc
$7^4 \equiv 1$ et $7^{100} = (7^4)^{25} \equiv 1$. D'après le \omterm{rem:b1:arith:interlude}{théorème des
restes chinois} (\cref{exo:b1:arith:8}), $7^{100} \equiv 1
\pmod{100}$~: les deux derniers chiffres sont $01$.
\end{solution}

\begin{solution}{exo:b1:arith:10}
Écrivons $m = nq + r$, $0 \leq r < n$. Alors
\[
2^m - 1 = 2^r\bigl(2^{nq} - 1\bigr) + 2^r - 1,
\]
et $2^n - 1$ \omterm{def:b1:arith:divides}{divise} $2^{nq} - 1 = (2^n - 1)(2^{n(q-1)} + \dots +
1)$. Donc, modulo $2^n - 1$, $\;2^m - 1 \equiv 2^r - 1$, et comme $0 \leq
2^r - 1 < 2^n - 1$, c'est bien \emph{le} reste euclidien.

Par conséquent, l'\omterm{met:b1:arith:euclid}{algorithme d'Euclide} appliqué au couple $(2^m - 1, 2^n - 1)$
reproduit, exposant par exposant, l'algorithme appliqué à $(m, n)$~: chaque
étape de division remplace $(m, n)$ par $(n, r)$ à l'étage du haut et $(2^m - 1,
2^n - 1)$ par $(2^n - 1, 2^r - 1)$ à l'étage du bas. En haut, l'algorithme
se termine sur $\gcd(m,n)$, donc en bas il se termine sur
$2^{\gcd(m,n)} - 1$.
\end{solution}

\begin{solution}{exo:b1:arith:11}
Résolvons d'abord $x^2 \equiv 1 \pmod p$~: $p \mid (x-1)(x+1)$, donc, par
le lemme d'Euclide, $x \equiv 1$ ou $x \equiv -1 \pmod p$.

Dans le produit $(p-1)! = 1 \times 2 \times \dots \times (p-1)$, tout
facteur $a$ est inversible modulo $p$, et son inverse $a^{-1}$ est encore
l'un des facteurs (\cref{prop:b1:arith:invmod}). Apparions chaque $a$ avec
$a^{-1}$~: les paires ont pour produit $1$, sauf les facteurs appariés à
eux-mêmes ($a = a^{-1}$, c'est-à-dire $a^2 \equiv 1$) qui restent seuls --- et ce sont
exactement $1$ et $p - 1$. D'où
\[
(p-1)! \equiv 1 \times (p - 1) \equiv -1 \pmod p .
\]
(Pour $p = 2$~: $1! = 1 \equiv -1 \pmod 2$~; l'argument d'appariement
dégénère mais le résultat subsiste.)

\emph{Réciproque.} Soit $n \geq 2$ composé, $n = ab$ avec $1 < a
\leq b < n$. Si $a < b$, tous deux apparaissent comme facteurs distincts de $(n-1)!$,
donc $n \mid (n-1)!$ et $(n-1)! \equiv 0 \not\equiv -1$. Si $a = b$
(c'est-à-dire $n = a^2$)~: pour $a \geq 3$, $a$ et $2a$ sont tous deux $< n$, donc
$n = a^2 \mid a \times 2a \mid (n-1)!$, même conclusion~; pour $n = 4$,
$(n-1)! = 6 \equiv 2 \not\equiv -1 \pmod 4$.
\end{solution}

\begin{solution}{exo:b1:arith:12}
\begin{enumerate}
  \item Récurrence. Pour $n = 1$~: $F_0 = 3 = F_1 - 2 = 5 - 2$.
        En supposant $F_0\cdots F_{n-1} = F_n - 2$~:
        \[
        F_0 \cdots F_n = (F_n - 2)F_n
        = \bigl(2^{2^n} - 1\bigr)\bigl(2^{2^n} + 1\bigr)
        = 2^{2^{n+1}} - 1 = F_{n+1} - 2 .
        \]
  \item Soient $m < n$ et $d = \gcd(F_m, F_n)$. D'après (1), $F_m$
        \omterm{def:b1:arith:divides}{divise} $F_n - 2$, donc $d$ \omterm{def:b1:arith:divides}{divise} à la fois $F_n$ et $F_n -
        2$, et \omterm{def:b1:arith:divides}{divise} donc $2$. Mais tout nombre de Fermat est impair, donc
        $d = 1$.
  \item Chaque $F_n \geq 3$ a un diviseur \omterm{def:b1:arith:prime}{premier} $p_n$
        (première étape du \cref{thm:b1:arith:euclidprimes}). Si $m
        \neq n$, alors $p_m \neq p_n$, car un \omterm{def:b1:arith:prime}{premier} commun
        diviserait $\gcd(F_m, F_n) = 1$. L'\omterm{def:b1:logic:map}{application} $n \mapsto p_n$ est
        donc \omterm{def:b1:logic:inj}{injective} de $\N$ dans l'\omterm{def:b1:logic:sets}{ensemble} des \omterm{def:b1:arith:prime}{nombres premiers}~: il existe une
        infinité de \omterm{def:b1:arith:prime}{nombres premiers}.
\end{enumerate}
\end{solution}

\begin{solution}{pb:b1:arith:1}
\textbf{1.} $10! = 3\,628\,800$~: deux zéros terminaux. Valuations
facteur par facteur~: les puissances de $2$ proviennent de $2, 4 = 2^2, 6, 8 = 2^3,
10$, ce qui totalise $v_2(10!) = 1 + 2 + 1 + 3 + 1 = 8$~; les puissances de $5$
proviennent de $5$ et $10$~: $v_5(10!) = 2$. Zéros terminaux $=
\min(v_2, v_5) = 2$, cohérent.

\textbf{2.} Écrivons la division euclidienne $\lfloor x\rfloor = nq +
r$, $0 \leq r \leq n - 1$. Alors $x = nq + r + \{x\}$ avec $0 \leq
r + \{x\} < n$, donc $\lfloor x/n \rfloor = q = \bigl\lfloor \lfloor
x \rfloor / n \bigr\rfloor$.

\textbf{3.} Soit $\alpha = v_p(a) \leq \beta = v_p(b)$ (quitte à échanger)
et écrivons $a = p^\alpha a'$, $b = p^\beta b'$ avec $p
\nmid a', b'$. Alors $a + b = p^\alpha\bigl(a' + p^{\beta -
\alpha}b'\bigr)$, donc $v_p(a + b) \geq \alpha = \min$. Si $\alpha <
\beta$, la parenthèse vaut $a' + p^{\beta-\alpha}b' \equiv a'
\not\equiv 0 \pmod p$~: la valuation est exactement $\alpha$.

\textbf{4.} Les multiples de $m$ dans $\intint1n$ sont $m, 2m, \dots,
qm$ où $q$ est le plus grand entier tel que $qm \leq n$, c'est-à-dire $q =
\lfloor n/m \rfloor$.

\textbf{5.} Par unicité de la factorisation, $v_p(n!) = \sum_{j=1}^{n}
v_p(j)$. Comptons autrement~: chaque $j$ contribue pour $v_p(j) =
\#\{k \geq 1 : p^k \mid j\}$, donc
\[
v_p(n!) = \sum_{j=1}^n \#\{k : p^k \mid j\}
= \sum_{k\geq1} \#\{j \leq n : p^k \mid j\}
= \sum_{k\geq1} \Bigl\lfloor \frac n{p^k} \Bigr\rfloor
\]
d'après la question 4 --- c'est la \omterm{pb:b1:arith:1}{formule de Legendre}. La somme est finie~: les termes
avec $p^k > n$ s'annulent.

\textbf{6.} $v_5(1000!) = 200 + 40 + 8 + 1 = 249$ (divisions par
$5, 25, 125, 625$)~; $v_2(1000!) = 500 + 250 + 125 + 62 + 31 + 15 +
7 + 3 + 1 = 994$. Zéros terminaux de $1000!$~: chaque zéro consomme
un $2$ et un $5$, il y en a donc $\min(994, 249) = 249$.

\textbf{7.} Avec $n = \sum_i a_ip^i$, la question 2 donne $\lfloor
n/p^k \rfloor = \sum_{i \geq k} a_ip^{i-k}$ (on tronque le développement
en base $p$). En sommant sur $k \geq 1$ et en échangeant les deux sommes
finies~:
\[
v_p(n!) = \sum_{i\geq1} a_i \sum_{k=1}^{i} p^{i-k}
= \sum_{i\geq0} a_i\,\frac{p^i - 1}{p - 1}
= \frac{n - s_p(n)}{p - 1} .
\]

\textbf{8.} Pour $p = 2$~: $v_2(n!) = n - s_2(n)$. Comme $n \geq 1$
vérifie $s_2(n) \geq 1$, on a toujours $v_2(n!) \leq n - 1 < n$~: $2^n \nmid
n!$. Et $v_2(n!) = n - 1$ si et seulement si $s_2(n) = 1$, si et seulement si $n$ est une puissance
de $2$.

\textbf{9.} $n$ a $\lfloor \log_p n \rfloor + 1$ chiffres en base
$p$, chacun au plus $p - 1$, donc $1 \leq s_p(n) \leq
(p-1)\bigl(\log_p(n) + 1\bigr)$. En substituant dans la question 7~:
\[
\frac n{p-1} - \log_p(n) - 1 \;\leq\; v_p(n!) \;<\; \frac n{p-1},
\]
et en divisant par $n$~: $\frac{v_p(n!)}n \to \frac1{p-1}$.

\textbf{10.} $Z(n) - Z(n-1) = v_5(n!/(n-1)!) = v_5(n)$~: le nombre
de zéros terminaux fait un saut de $v_5(n)$ à chaque multiple de $5$ et est
constant entre deux. $Z(24) = \lfloor24/5\rfloor = 4$ et $Z(25) =
5 + 1 = 6$~: en $n = 25$, le compte saute de $4$ directement à $6$
($v_5(25) = 2$), et comme $Z$ est croissante avec $Z \leq 4$
avant et $Z \geq 6$ après, la valeur $5$ n'est jamais atteinte~: aucune
factorielle ne se termine par exactement cinq zéros.

\textbf{11.} Écrivons $x = \lfloor x\rfloor + \{x\}$~: $\lfloor x +
y\rfloor = \lfloor x\rfloor + \lfloor y\rfloor + \lfloor \{x\} +
\{y\}\rfloor$, et $0 \leq \{x\} + \{y\} < 2$ fait que la dernière partie entière
vaut $0$ ou $1$. Puis, par Legendre appliqué trois fois,
\[
v_p\binom{m+n}m = v_p\bigl((m{+}n)!\bigr) - v_p(m!) - v_p(n!)
= \sum_{k\geq1}\Bigl(
\Bigl\lfloor\frac{m+n}{p^k}\Bigr\rfloor
- \Bigl\lfloor\frac{m}{p^k}\Bigr\rfloor
- \Bigl\lfloor\frac{n}{p^k}\Bigr\rfloor\Bigr),
\]
une somme finie de $0$ et de $1$ (on applique la première \omterm{def:b1:logic:statement}{assertion} à $x =
m/p^k$, $y = n/p^k$).

\textbf{12.} Fixons $k \geq 1$ et écrivons $m = p^km_1 + m_0$, $n =
p^kn_1 + n_0$ avec $0 \leq m_0, n_0 < p^k$ (division euclidienne~:
$m_0$ est le nombre formé des $k$ chiffres de poids faible de $m$). Alors
\[
\Bigl\lfloor\frac{m+n}{p^k}\Bigr\rfloor
- \Bigl\lfloor\frac m{p^k}\Bigr\rfloor
- \Bigl\lfloor\frac n{p^k}\Bigr\rfloor
= \Bigl\lfloor\frac{m_0 + n_0}{p^k}\Bigr\rfloor ,
\]
qui vaut $1$ si $m_0 + n_0 \geq p^k$ et $0$ sinon. Or $m_0 +
n_0 \geq p^k$ signifie précisément que l'addition des $k$ chiffres de poids faible de $m$
et de $n$ déborde en position $k$ --- une retenue en position $k$
dans l'algorithme d'addition appris à l'école. En sommant sur $k$~:
$v_p\binom{m+n}m$ est le nombre de retenues de l'addition $m + n$ en
base $p$. (Kummer, 1852.)

\textbf{13.} Appliquons Kummer à $m = j$, $n = p^k - j$, de somme $p^k =
(1\underbrace{0\cdots0}_{k})_p$. Posons $a = v_p(j)$~: les chiffres en base $p$
de $j$ aux positions $0, \dots, a-1$ sont nuls et le chiffre
en position $a$ est non nul. Les chiffres de $p^k - j$ en dessous de la position
$a$ sont nuls eux aussi ($p^k - j = p^a(p^{k-a} - j/p^a)$). En position
$a$, les deux chiffres non nuls doivent avoir pour somme $p$ (chiffre résultat $0$)~:
une retenue~; à chacune des positions $a+1, \dots, k-1$, les chiffres augmentés de la
retenue entrante ont pour somme $p$ (chiffre résultat $0$ de nouveau)~: la retenue
se propage. Au total~: $k - a$ retenues, donc $v_p\binom{p^k}j = k -
v_p(j)$. Pour $k = 1$~: $v_p\binom pj = 1$ pour $0 < j < p$, la
\omterm{def:b1:arith:divides}{divisibilité} utilisée au \cref{thm:b1:arith:fermat}.

\textbf{14.} Par la forme en chiffres (question 7), en utilisant $s_2(2n) =
s_2(n)$ (on ajoute un chiffre $0$)~:
\[
v_2\binom{2n}n = \bigl(2n - s_2(2n)\bigr) - 2\bigl(n -
s_2(n)\bigr) = 2s_2(n) - s_2(2n) = s_2(n) \geq 1 :
\]
$\binom{2n}n$ est toujours pair, et $v_2 = 1$ (c'est-à-dire $\binom{2n}n
\equiv 2 \pmod 4$) exactement lorsque $s_2(n) = 1$, c'est-à-dire lorsque $n$ est une
puissance de $2$.

\textbf{15.} Vandermonde avec $m = n = k = p$~: $\binom{2p}p =
\sum_{j=0}^p \binom pj\binom p{p-j} = \sum_{j=0}^p \binom pj^2$.
Pour $0 < j < p$, $p \mid \binom pj$ (question 13), donc $\binom
pj^2 \equiv 0 \pmod p$~; les termes extrêmes donnent $1 + 1$~:
$\binom{2p}p \equiv 2 \pmod p$.

\textbf{16.} Base $3$~: $500 = 486 + 9 + 3 + 2$, chiffres (des poids faibles aux
poids forts) $(2, 1, 1, 0, 0, 2)$, donc $s_3(500) = 6$~; et $1000 = 729 +
243 + 27 + 1$, chiffres $(1, 0, 0, 1, 0, 1, 1)$, donc $s_3(1000) = 4$.
\emph{Kummer~:} additionnons $500 + 500$ en base $3$~: position $0$~: $2 + 2 =
4$, chiffre $1$, retenue $1$~; position $1$~: $1 + 1 + 1 = 3$, chiffre $0$,
retenue $1$~; position $2$~: $1 + 1 + 1 = 3$, chiffre $0$, retenue $1$~;
position $3$~: $0 + 0 + 1 = 1$, pas de retenue~; position $4$~: $0$~;
position $5$~: $2 + 2 = 4$, chiffre $1$, retenue $1$~; position $6$~:
la retenue tombe~: chiffre $1$. Quatre retenues~: $v_3\binom{1000}{500} = 4$.
\emph{Legendre~:} $v_3(1000!) = \frac{1000 - 4}2 = 498$ et
$v_3(500!) = \frac{500 - 6}2 = 247$, donc $v_3\binom{1000}{500} =
498 - 2\times247 = 4$. Les deux calculs concordent --- et les chiffres de
l'addition $(1, 0, 0, 1, 0, 1, 1)$ redonnent $1000$, comme il se
doit.

\textbf{17.} Par Kummer ($p = 2$, $m = k$, $n' = n - k$)~:
$\binom nk$ est impair si et seulement si l'addition $k + (n - k)$ en base $2$ n'a
aucune retenue, si et seulement si, à chaque position, les chiffres vérifient $k_i + (n -
k)_i = n_i$~; dans ce cas $k_i \leq n_i$ pour tout $i$. Réciproquement,
si $k_i \leq n_i$ pour tout $i$, alors le nombre dont les chiffres sont $n_i -
k_i$ vaut $n - k$, et l'addition est sans retenue. Même démonstration en base
$p$~: $p \nmid \binom nk$ si et seulement si chaque chiffre en base $p$ de $k$ est au
plus égal au chiffre correspondant de $n$.

\textbf{18.} En comptant les $k \in \intint0n$ dont les chiffres vérifient
$k_i \leq n_i$~: chaque chiffre de $k$ est choisi indépendamment parmi
$n_i + 1$ valeurs, ce qui donne $\prod_i (n_i + 1)$ choix~; en base $2$
cela vaut $2^{\#\{i : n_i = 1\}} = 2^{s_2(n)}$. Ligne $4 =
(100)_2$~: $2^1 = 2$ coefficients impairs --- en effet $1, 4, 6, 4, 1$ n'a de
coefficients impairs qu'aux extrémités. Ligne $5 = (101)_2$~: $2^2 = 4$ --- en effet
$1, 5, 10, 10, 5, 1$.

\textbf{19.} Tous les coefficients intérieurs sont pairs $\iff$ la ligne a exactement
$2$ coefficients impairs (les deux extrémités le sont toujours) $\iff 2^{s_2(n)} =
2 \iff s_2(n) = 1 \iff n$ est une puissance de $2$.

\textbf{20.} Dans la somme de la question 11, le $k$-ième terme s'annule dès
que $p^k > m + n$ (les trois parties entières sont alors égales, la
première valant même $0$ quand $p^k > m+n$~; plus simplement, chaque terme est nul).
Donc au plus $\lfloor \log_p(m+n)\rfloor$ termes sont non nuls, chacun
valant $1$~: $a = v_p\binom{m+n}m \leq \log_p(m+n)$, c'est-à-dire $p^a \leq
m + n$.

\textbf{21.} Pour tout \omterm{def:b1:arith:prime}{nombre premier} $p$, $v_p\bigl(\operatorname{lcm}(1,
\dots, 2n)\bigr) = \lfloor\log_p(2n)\rfloor$ (la plus grande puissance de
$p$ n'excédant pas $2n$ figure parmi $1, \dots, 2n$). La question 20
avec $m = n$ donne $v_p\binom{2n}n \leq \lfloor\log_p(2n)\rfloor$
pour tout $p$~: d'après la \cref{prop:b1:arith:valuation}, $\binom{2n}n
\mid \operatorname{lcm}(1, \dots, 2n)$. Pour la taille~: le rapport
$\binom{2n}{k+1}/\binom{2n}k = \frac{2n-k}{k+1} \geq 1$ exactement
pour $k < n$, donc le coefficient central est le plus grand des $2n + 1$
coefficients de la ligne $2n$, d'où $4^n = \sum_k \binom{2n}k \leq
(2n+1)\binom{2n}n$. En combinant~:
\[
\operatorname{lcm}(1, \dots, 2n) \geq \binom{2n}n \geq
\frac{4^n}{2n + 1} .
\]
S'il y avait peu de \omterm{def:b1:arith:prime}{nombres premiers} en dessous de $2n$, le PPCM ne pourrait pas être aussi
grand~: la croissance exponentielle du PPCM est une trace quantitative de
l'abondance des \omterm{def:b1:arith:prime}{nombres premiers}.

\textbf{22.} $Z(n) = \sum_k\lfloor n/5^k\rfloor \approx \frac
n4$, on vise donc autour de $n = 4 \times 2026 = 8104$~: $Z(8104) = 1620 +
324 + 64 + 12 + 2 = 2022$. Montons de multiple de $5$ en multiple de $5$~: $Z(8110)
= 2024$, $Z(8115) = 2025$, et
\[
Z(8120) = 1624 + 324 + 64 + 12 + 2 = 2026 .
\]
Comme $Z$ est constante entre deux multiples de $5$ et que $Z(8119) =
Z(8115) = 2025$, le plus petit $n$ ayant au moins $2026$ zéros
terminaux est $n = 8120$.

\textbf{23.} Base $7$~: $50 = 49 + 1$, chiffres (des poids faibles aux poids forts) $(1, 0,
1)$. Additionnons $50 + 50$~: position $0$~: $1 + 1 = 2 < 7$, pas de retenue~;
position $1$~: $0 + 0 = 0$~; position $2$~: $1 + 1 = 2 < 7$, pas de
retenue. Sans retenue, donc par Kummer $v_7\binom{100}{50} = 0$~: $7 \nmid
\binom{100}{50}$. Legendre confirme~: $v_7(100!) = \lfloor 100/7
\rfloor + \lfloor 100/49 \rfloor = 14 + 2 = 16$ et $v_7(50!) = 7
+ 1 = 8$, donc $v_7\binom{100}{50} = 16 - 2\times8 = 0$.

\textbf{24.} (i) L'unicité de la factorisation sous-tend la définition
même de $v_p$ et son additivité, donc la \omterm{pb:b1:arith:1}{formule de Legendre}
et toutes les conclusions de \omterm{def:b1:arith:divides}{divisibilité}
(\cref{prop:b1:arith:valuation}). (ii) La division euclidienne a produit
l'identité de troncature de la question 2 et la décomposition $m = p^km_1 +
m_0$ qui isole la retenue (question 12). (iii) Dénombrement~: le
comptage des multiples de $m$ (question 4), le produit des choix de chiffres
(question 18) et la majoration par la somme d'une ligne $4^n \leq
(2n+1)\binom{2n}n$ (question 21) sont tous des arguments dans l'esprit du
\cref{ch:b1:counting}.

\textbf{25.} Legendre convertit «~quelle puissance de $p$ \omterm{def:b1:arith:divides}{divise}
$n!$~» en arithmétique des chiffres en base $p$~; Kummer comprime la
réponse, pour les coefficients binomiaux, dans les retenues d'une seule
addition --- la \omterm{def:b1:arith:divides}{divisibilité}, apparemment propriété globale d'énormes
nombres, se lit localement, chiffre par chiffre. La question 16 en est
le paradigme~: quatre retenues, calculées à la main, déterminent la puissance
exacte de $3$ dans un nombre de plusieurs centaines de chiffres. Et la question 21
montre le même cercle d'idées effleurant des eaux profondes~: une
minoration exponentielle de $\operatorname{lcm}(1, \dots, 2n)$ est
un premier pas, entièrement élémentaire, vers le théorème des \omterm{def:b1:arith:prime}{nombres premiers},
dont la démonstration dépasse de loin ce volume. Toute la boîte à outils ---
division, PGCD, valuations --- est rejouée pour les polynômes au
\cref{ch:b1:poly}, où l'analogue du développement en chiffres est le
développement suivant les puissances de $(X - a)$.
\end{solution}
