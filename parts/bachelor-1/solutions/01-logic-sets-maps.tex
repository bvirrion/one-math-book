\section*{Chapter \ref{ch:b1:logic} --- Logic, Sets and Maps}

\begin{solution}{exo:b1:logic:1}
Negations, pushing $\lnot$ through each quantifier
(\cref{prop:b1:logic:negquant}) and using $\lnot(P \implies Q) \iff
P \land \lnot Q$:
\begin{enumerate}
  \item $\exists x \in \R,\ \forall y \in \R,\ x + y \leq 0$;
  \item $\forall x \in \R,\ \exists y \in \R,\ xy \neq 0$;
  \item $\exists \varepsilon > 0,\ \forall \delta > 0,\ \exists x \in
        \R,\ \abs{x} \leq \delta \text{ and } \abs{f(x)} >
        \varepsilon$.
\end{enumerate}
Statement (1) is true: given $x$, take $y = -x + 1$; then $x + y = 1 >
0$. Statement (2) is true: $x = 0$ satisfies $xy = 0$ for all $y$.
\end{solution}

\begin{solution}{exo:b1:logic:2}
Truth table, writing T/F for the four cases $(P, Q)$:
\begin{center}
\begin{tabular}{cc|c|c|c|c}
$P$ & $Q$ & $P \implies Q$ & $\lnot(P \implies Q)$ & $\lnot Q$ &
$P \land \lnot Q$ \\
\hline
T & T & T & F & F & F \\
T & F & F & T & T & T \\
F & T & T & F & F & F \\
F & F & T & F & T & F \\
\end{tabular}
\end{center}
Columns $4$ and $6$ coincide, proving the equivalence. The negation of
``if a function is differentiable then it is continuous'' is therefore:
``there is a function which is differentiable and not continuous''
(a false statement, as it happens: the original implication is true,
see \cref{ch:b1:derivative}).
\end{solution}

\begin{solution}{exo:b1:logic:3}
\emph{Contraposition.} Assume $x < 1$. Then $x^3 < 1$ (the cube
function is increasing) and $x < 1$, so $x^3 + x < 2$. This proves the
contrapositive, hence the statement.

\emph{Contradiction.} Suppose $a > 0$ is the smallest strictly positive
real. Then $a/2$ is strictly positive and $a/2 < a$ (since $a > 0$),
contradicting minimality. Hence no such $a$ exists.
\end{solution}

\begin{solution}{exo:b1:logic:4}
\begin{enumerate}
  \item Base case $n = 0$: $2^0 = 1 = 2^1 - 1$. Step: assuming the
        identity for $n$,
        \[
        \sum_{k=0}^{n+1} 2^k = (2^{n+1} - 1) + 2^{n+1}
        = 2 \cdot 2^{n+1} - 1 = 2^{n+2} - 1 .
        \]
  \item Base case $n = 0$: $4^0 + 5 = 6 = 3 \times 2$. Step: if
        $4^n + 5 = 3m$, then
        \[
        4^{n+1} + 5 = 4(4^n + 5) - 15 = 3(4m - 5),
        \]
        divisible by $3$.
\end{enumerate}
\end{solution}

\begin{solution}{exo:b1:logic:5}
The inductive step silently assumes that the two groups (``the first
$n$'' and ``the last $n$'') overlap, so that the shared pencils carry
the color from one group to the other. For $n + 1 = 2$ the two groups
are $\{$first pencil$\}$ and $\{$second pencil$\}$: they are disjoint,
and the argument breaks. So $P(1) \implies P(2)$ was never proved, and
the induction collapses --- even though $P(n) \implies P(n+1)$ is
valid for every $n \geq 2$.
\end{solution}

\begin{solution}{exo:b1:logic:6}
\begin{enumerate}
  \item $x \in A \setminus B \iff x \in A \land x \notin B \iff x \in
        A \land x \in \overline{B} \iff x \in A \cap \overline{B}$.
  \item Using (1) and distributivity
        (\cref{prop:b1:logic:setalgebra}):
        $(A \cup B) \cap \overline{C} = (A \cap \overline{C}) \cup
        (B \cap \overline{C})$.
  \item Suppose $A \subseteq B$. Then $A \cup B \subseteq B$ (both
        pieces lie in $B$) and $B \subseteq A \cup B$ always, so
        $A \cup B = B$. Suppose $A \cup B = B$: then $A \cap B
        \subseteq A$ always, and $A \subseteq A \cup B = B$ gives
        $A \subseteq A \cap B$, so $A \cap B = A$. Suppose $A \cap B =
        A$: then $A = A \cap B \subseteq B$. The three conditions are
        thus equivalent (we proved a cycle of implications).
\end{enumerate}
\end{solution}

\begin{solution}{exo:b1:logic:7}
\begin{enumerate}
  \item $f$ is injective ($n + 1 = m + 1 \implies n = m$) but not
        surjective: $0$ has no preimage in $\N$.
  \item $g$ is bijective: $n \mapsto n - 1$ is a two-sided inverse on
        $\Z$.
  \item $h$ is injective: $\frac{x+1}{x-1} = \frac{x'+1}{x'-1}$ gives
        $(x+1)(x'-1) = (x'+1)(x-1)$, i.e.\ $xx' - x + x' - 1 = xx' -
        x' + x - 1$, so $2x' = 2x$. It is not surjective onto $\R$:
        solving $y = \frac{x+1}{x-1}$ gives $x(y - 1) = y + 1$, which
        has no solution when $y = 1$ (the equation reads $0 = 2$).
        With codomain $\R \setminus \{1\}$, the same computation gives
        the unique preimage $x = \frac{y+1}{y-1}$, so
        $h \colon \R \setminus \{1\} \to \R \setminus \{1\}$ is
        bijective and $h^{-1}(y) = \frac{y+1}{y-1} = h(y)$: $h$ is its
        own inverse.
\end{enumerate}
\end{solution}

\begin{solution}{exo:b1:logic:8}
\begin{enumerate}
  \item $x \in f^{-1}(B \cap B') \iff f(x) \in B \cap B' \iff f(x) \in
        B \land f(x) \in B' \iff x \in f^{-1}(B) \cap f^{-1}(B')$.
        For images: $y \in f(A \cup A')$ iff $y = f(x)$ for some $x$
        in $A$ or in $A'$, iff $y \in f(A)$ or $y \in f(A')$.
  \item If $y \in f(A \cap A')$, then $y = f(x)$ with $x \in A$ and
        $x \in A'$, so $y \in f(A)$ and $y \in f(A')$. Strictness:
        take $f \colon \R \to \R$, $x \mapsto x^2$, $A = \{-1\}$,
        $A' = \{1\}$: then $f(A \cap A') = f(\emptyset) = \emptyset$
        but $f(A) \cap f(A') = \{1\}$.
  \item ($\Leftarrow$) With $A = \{x\}$, $A' = \{x'\}$ for $x \neq
        x'$: if $f(x) = f(x')$, then $f(A) \cap f(A') = \{f(x)\}$
        while $f(A \cap A') = \emptyset$, contradicting the assumed
        equality; so $f$ is injective. ($\Rightarrow$) Let $f$ be
        injective and $y \in f(A) \cap f(A')$: $y = f(x) = f(x')$ with
        $x \in A$, $x' \in A'$; injectivity gives $x = x' \in A \cap
        A'$, so $y \in f(A \cap A')$. With (2), equality holds.
\end{enumerate}
\end{solution}

\begin{solution}{exo:b1:logic:9}
$g \circ f = \mathrm{id}_E$ is injective and surjective, so by
\cref{prop:b1:logic:comp}~(2), $f$ is injective and $g$ is surjective.
Example: $E = \N$, $F = \Z$, $f$ the inclusion $n \mapsto n$, and
$g \colon \Z \to \N$, $g(n) = n$ for $n \geq 0$ and $g(n) = 0$ for
$n < 0$. Then $g(f(n)) = n$ for all $n \in \N$, but $f$ is not
surjective and $g$ is not injective.
\end{solution}

\begin{solution}{exo:b1:logic:10}
$x^2 - y^2 = x - y \iff (x - y)(x + y) = x - y \iff (x - y)(x + y - 1)
= 0 \iff y = x$ or $y = 1 - x$. \emph{Reflexive:} $y = x$ works.
\emph{Symmetric:} the condition ``$y = x$ or $y = 1 - x$'' is
symmetric in $x$ and $y$ (if $y = 1 - x$ then $x = 1 - y$).
\emph{Transitive:} suppose $x \mathbin{\mathcal{R}} y$ and $y
\mathbin{\mathcal{R}} z$; going through the four cases, $z$ equals $x$
or $1 - x$ each time (e.g.\ $y = 1 - x$ and $z = 1 - y$ give $z = x$).
So $\mathcal{R}$ is an equivalence relation and $\mathrm{cl}(x) =
\{x,\, 1 - x\}$. This class has one element exactly when $x = 1 - x$,
i.e.\ for $x = \frac12$.
\end{solution}

\begin{solution}{exo:b1:logic:11}
Let $f \colon E \to \mathcal{P}(E)$ be any map and set
$D = \{x \in E : x \notin f(x)\} \in \mathcal{P}(E)$. Suppose $D =
f(a)$ for some $a \in E$. If $a \in D$, then by definition of $D$,
$a \notin f(a) = D$: contradiction. If $a \notin D$, then $a \notin
f(a)$, so by definition of $D$, $a \in D$: contradiction. Hence $D$ is
not in the image of $f$, and $f$ is not surjective. (In particular no
set is in bijection with its power set: there are ``more'' subsets of
$\N$ than integers.)
\end{solution}

\begin{solution}{exo:b1:logic:12}
\begin{enumerate}
  \item ($\Rightarrow$) Let $f$ be surjective and $\Phi(B) =
        \Phi(B')$. For $y \in B$, pick $x$ with $f(x) = y$; then $x
        \in f^{-1}(B) = f^{-1}(B')$, so $y = f(x) \in B'$. Hence $B
        \subseteq B'$, and symmetrically $B' \subseteq B$: $\Phi$ is
        injective. ($\Leftarrow$) If $f$ is not surjective, pick $y_0
        \in F$ outside the image; then $f^{-1}(\{y_0\}) = \emptyset =
        f^{-1}(\emptyset)$ with $\{y_0\} \neq \emptyset$, so $\Phi$ is
        not injective.
  \item ($\Rightarrow$) Let $f$ be injective and $A \subseteq E$. Set
        $B = f(A)$; then $f^{-1}(B) = \{x : f(x) \in f(A)\}$, and
        injectivity gives $f(x) \in f(A) \iff x \in A$, so $\Phi(B) =
        A$: $\Phi$ is surjective. ($\Leftarrow$) If $f$ is not
        injective, take $x \neq x'$ with $f(x) = f(x')$. Every
        preimage set $f^{-1}(B)$ contains $x$ if and only if it
        contains $x'$; hence $\{x\}$ is not of the form $\Phi(B)$, and
        $\Phi$ is not surjective.
\end{enumerate}
\end{solution}
