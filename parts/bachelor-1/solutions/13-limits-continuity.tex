\section*{Chapter \ref{ch:b1:continuity} --- Limits and Continuity}

\begin{solution}{exo:b1:continuity:1}
Take $u_n = \frac{1}{2\pi n + \pi/2}$ and $v_n = \frac{1}{2\pi n}$:
both tend to $0^+$, yet $\sin\frac{1}{u_n} = 1$ and
$\sin\frac{1}{v_n} = 0$. Two sequences, two different limits of
images: by \cref{thm:b1:continuity:seqchar}, no limit at $0^+$.

$x \sin\frac1x$: squeezed by $\abs{x\sin\frac1x} \leq \abs x \to 0$,
so the limit at $0$ exists and equals $0$.
\end{solution}

\begin{solution}{exo:b1:continuity:2}
$f = \lfloor\cdot\rfloor$ is \omterm{def:b1:continuity:continuous}{continuous} on $\R \setminus \Z$
(locally constant) and discontinuous at each $n \in \Z$: left limit
$n - 1$, value $n$.

$g(x) = x - \lfloor x\rfloor$: same discontinuity points (the
identity is \omterm{def:b1:continuity:continuous}{continuous}, so $g$ inherits the jumps of $f$); at $n \in
\Z$, left limit $1 \neq 0 = g(n)$.

$h(x) = \lfloor x\rfloor + (x - \lfloor x\rfloor)^2$: on $\intco{n}{n
+ 1}$, $h(x) = n + (x - n)^2$, \omterm{def:b1:continuity:continuous}{continuous} there; at $x = n$, the left
limit is $(n-1) + 1 = n = h(n)$: the jumps cancel. $h$ is \omterm{def:b1:continuity:continuous}{continuous}
on $\R$ (and strictly increasing).
\end{solution}

\begin{solution}{exo:b1:continuity:3}
$P(x) = x^5 - 3x + 1$: $P(-2) = -32 + 6 + 1 = -25 < 0$; $P(0) = 1 >
0$; $P(1) = -1 < 0$; $P(2) = 27 > 0$. Three sign changes on the
disjoint segments $\intcc{-2}{0}$, $\intcc{0}{1}$, $\intcc{1}{2}$:
by \cref{thm:b1:continuity:ivt}, at least three roots. (Being of
degree $5$, $P$ has at most five; a variation study would show
exactly three.)
\end{solution}

\begin{solution}{exo:b1:continuity:4}
Let $P = a_{2m+1}X^{2m+1} + \dots$ with $a_{2m+1} > 0$ (else replace
$P$ by $-P$). Factoring the dominant term, $P(x) = a_{2m+1}
x^{2m+1}\bigl(1 + o(1)\bigr)$ as $x \to \pm\infty$: so $P(x) \to
+\infty$ at $+\infty$ and $-\infty$ at $-\infty$. Pick $a$ with
$P(a) < 0$ and $b$ with $P(b) > 0$: the intermediate value theorem on
$\intcc{a}{b}$ provides a root.
\end{solution}

\begin{solution}{exo:b1:continuity:5}
Let $g(x) = f(x) - x$, \omterm{def:b1:continuity:continuous}{continuous} on $\intcc{0}{1}$. Since $f$ \omterm{def:b1:logic:map}{maps}
into $\intcc{0}{1}$: $g(0) = f(0) \geq 0$ and $g(1) = f(1) - 1 \leq
0$. By \cref{thm:b1:continuity:ivt}, $g(c) = 0$ for some $c$: a
fixed point.

Necessity of the hypotheses: on $\intcc{0}{1}$, the discontinuous \omterm{def:b1:logic:map}{map}
$f(x) = 1$ for $x \leq \frac12$, $f(x) = 0$ for $x > \frac12$ has no
fixed point; on the \omterm{prop:b1:reals:intervals}{interval} $\intoo{0}{1}$ (not a segment), $f(x) =
\frac x2$ is \omterm{def:b1:continuity:continuous}{continuous} into $\intoo{0}{1}$ with no fixed point
(the candidate $0$ is missing); on $\R$, $f(x) = x + 1$.
\end{solution}

\begin{solution}{exo:b1:continuity:6}
$g(x) = \cos x - x$ is \omterm{def:b1:continuity:continuous}{continuous}, $g(0) = 1 > 0$, $g(1) = \cos 1 -
1 < 0$: a solution exists in $\intoo{0}{1}$
(\cref{thm:b1:continuity:ivt}). Uniqueness: $g$ is strictly
decreasing on $\R$ --- for $x \leq 0$, $g(x) \geq 1 - x > 0$ has no
zero anyway; and $g'(x) = -\sin x - 1 \leq 0$ with equality only at
isolated points ($x \equiv -\frac\pi2 \bmod 2\pi$), so $g$ is
strictly decreasing (\cref{ch:b1:derivative}; alternatively: on
$\intcc{0}{1}$, $\cos$ is strictly decreasing and $-x$ too, so $g$
is). A strictly monotone function vanishes at most once.
\end{solution}

\begin{solution}{exo:b1:continuity:7}
Fix $M = f(0) + 1$. There is $A > 0$ with $f(x) \geq M$ for $\abs x
\geq A$ (definition of the two infinite limits; take the larger
threshold). On the segment $\intcc{-A}{A}$, the extreme value theorem
(\cref{thm:b1:continuity:evt}) provides $c$ with $f(c) =
\inf_{\intcc{-A}{A}} f \leq f(0)$. For $\abs x \geq A$: $f(x) \geq M
> f(0) \geq f(c)$. So $f(c)$ is the global minimum.
\end{solution}

\begin{solution}{exo:b1:continuity:8}
On the segment $\intcc{0}{T}$, $f$ is bounded and attains its bounds
(\cref{thm:b1:continuity:evt}); by periodicity, these are the bounds
on all of $\R$, still attained.

Let $g(x) = f(x + \frac T2) - f(x)$, \omterm{def:b1:continuity:continuous}{continuous}. Then
\[
g(0) + g\bigl(\tfrac T2\bigr)
= \bigl(f(\tfrac T2) - f(0)\bigr) + \bigl(f(T) - f(\tfrac T2)\bigr)
= f(T) - f(0) = 0 :
\]
$g(0)$ and $g(\frac T2)$ have opposite signs (or one vanishes), so
the intermediate value theorem on $\intcc{0}{T/2}$ gives $c$ with
$g(c) = 0$, i.e.\ $f(c + \frac T2) = f(c)$.
\end{solution}

\begin{solution}{exo:b1:continuity:9}
First the inequality: for $0 \leq y \leq x$,
\[
\bigl(\sqrt y + \sqrt{x - y}\bigr)^2 = x + 2\sqrt{y(x-y)} \geq x,
\]
so $\sqrt x \leq \sqrt y + \sqrt{x - y}$, i.e.\ $\sqrt x - \sqrt y
\leq \sqrt{x - y}$. Hence $\abs{\sqrt x - \sqrt y} \leq
\sqrt{\abs{x - y}}$ for all $x, y \geq 0$.

\omterm{def:b1:continuity:uniform}{Uniform continuity}: given $\varepsilon > 0$, take $\delta =
\varepsilon^2$; then $\abs{x - y} \leq \delta$ implies $\abs{\sqrt x
- \sqrt y} \leq \sqrt\delta = \varepsilon$.

Not Lipschitz near $0$: $\frac{\sqrt x - \sqrt 0}{x - 0} =
\frac{1}{\sqrt x} \to +\infty$ as $x \to 0^+$, so no constant $k$
can dominate all difference quotients.
\end{solution}

\begin{solution}{exo:b1:continuity:10}
Suppose $f$ is \omterm{def:b1:logic:inj}{injective}, \omterm{def:b1:continuity:continuous}{continuous}, and not strictly monotonic.
Then there are $a < b < c$ in $I$ with $f(b)$ not between $f(a)$ and
$f(c)$ --- indeed, if for \emph{all} triples the middle value were
between the outer ones, $f$ would be monotonic (compare any two
pairs; a short case check). Say $f(b) > \max(f(a), f(c))$ (the other
case is symmetric, replace $f$ by $-f$). Choose $v$ with $\max(f(a),
f(c)) < v < f(b)$. By the intermediate value theorem applied on
$\intcc{a}{b}$ and on $\intcc{b}{c}$, there are $u_1 \in
\intoo{a}{b}$ and $u_2 \in \intoo{b}{c}$ with $f(u_1) = v = f(u_2)$:
two distinct points with equal images, contradicting injectivity.
\end{solution}

\begin{solution}{exo:b1:continuity:11}
$f(0) = f(0+0) = 2f(0)$ gives $f(0) = 0$; $f(-x) = -f(x)$ from $0 =
f(x - x)$. Set $\alpha = f(1)$. Induction: $f(n) = n\alpha$ for $n
\in \N$, then for $n \in \Z$ by oddness. For $q \in \N^*$:
$q\,f(\frac pq) = f(p) = p\alpha$ (add $\frac pq$ to itself $q$
times), so $f(\frac pq) = \alpha\frac pq$: $f = \alpha\,
\mathrm{id}$ on $\Q$.

Let now $x \in \R$ and $(r_n)$ a sequence of rationals with $r_n \to
x$ (\omterm{def:b1:topology:dense}{density}, \cref{thm:b1:reals:density}, applied in nested
\omterm{prop:b1:reals:intervals}{intervals}; or $r_n = \frac{\lfloor nx\rfloor}{n}$). \omterm{def:b1:continuity:continuous}{Continuity}:
$f(x) = \lim f(r_n) = \lim \alpha r_n = \alpha x$.
\end{solution}

\begin{solution}{exo:b1:continuity:12}
Let $\varepsilon > 0$. By the limit at $+\infty$, there is $A$ with
$\abs{f(x) - \ell} \leq \frac\varepsilon2$ for $x \geq A$; hence for
$x, y \geq A$: $\abs{f(x) - f(y)} \leq \varepsilon$ (no closeness
needed).

On the segment $\intcc{0}{A + 1}$, Heine's theorem
(\cref{thm:b1:continuity:heine}) gives $\delta_0 > 0$ for this
$\varepsilon$; set $\delta = \min(\delta_0, 1)$.

Now take any $x, y \geq 0$ with $\abs{x - y} \leq \delta$, say $x
\leq y$. If $y \leq A + 1$: both lie in the segment, and the Heine
$\delta_0$ applies. Otherwise $y > A + 1$, and then $x \geq y - 1 >
A$: both lie in $\intco{A}{+\infty}$, where the limit argument
applies. In both cases $\abs{f(x) - f(y)} \leq \varepsilon$: \omterm{def:b1:continuity:uniform}{uniform
continuity}.
\end{solution}

\begin{solution}{pb:b1:continuity:1}
\textbf{1.} For $n \in \N$: $f(nx) = n f(x)$ by induction
($f((n+1)x) = f(nx) + f(x)$). Also $f(0) = 2f(0)$ gives $f(0) =
0$, and $0 = f(x - x) = f(x) + f(-x)$ gives oddness, so $f(nx) =
nf(x)$ for $n \in \Z$. For $r = \frac pq$: $q\,f\bigl(\tfrac
pq x\bigr) = f(px) = p\,f(x)$, so $f(rx) = r f(x)$: $f$ is
$\Q$-linear.

\textbf{2.} Additivity gives, for all $x$ and $h$:
\[
f(x + h) - f(x) = f(h) = f(x_0 + h) - f(x_0) .
\]
As $h \to 0$, the right side tends to $0$ by \omterm{def:b1:continuity:continuous}{continuity} at
$x_0$; hence $f(x + h) \to f(x)$: \omterm{def:b1:continuity:continuous}{continuity} at every $x$. Then
\cref{exo:b1:continuity:11} yields $f(x) = f(1)\,x$.

\textbf{3.} Two \omterm{def:b1:continuity:continuous}{continuous} additive functions are of the form
$cx$ and $c'x$ (question 2); if they agree on the \omterm{def:b1:topology:dense}{dense} \omterm{def:b1:structures:subgroup}{subgroup}
$\Z + \sqrt2\,\Z$ (\cref{exo:b1:reals:9}), then $c\,g = c'g$ for
some $g \neq 0$ in it: $c = c'$, the functions are equal. Without
\omterm{def:b1:continuity:continuous}{continuity}: $\Q$-linearity only ties values at
$\Q$-combinations, and $1, \sqrt2$ are $\Q$-independent ($\sqrt2
\notin \Q$), so $f(1) = 0$, $f(\sqrt2) = 1$ is consistent and
forces, on the \omterm{def:b1:structures:subgroup}{subgroup},
\[
f(m + n\sqrt2) = m\,f(1) + n\,f(\sqrt2) = n .
\]

\textbf{4.} For $t \in \intcc{0}{\ell}$: $a + t \in
\intcc{a}{b}$, so $f(t) = f(a + t) - f(a) \leq M - f(a) =: M'$.

\textbf{5.} For $t \in \intcc{0}{\ell}$, also $\ell - t \in
\intcc{0}{\ell}$, and additivity gives $f(t) = f(\ell) - f(\ell
- t) \geq f(\ell) - M'$. Hence $\abs{f} \leq C$ on
$\intcc{0}{\ell}$ with $C = \max\bigl(\abs{M'}, \abs{f(\ell) -
M'}\bigr)$.

\textbf{6.} For $t \in \intcc{0}{\ell/n}$: $nt \in
\intcc{0}{\ell}$ and $f(t) = \frac{f(nt)}{n}$ (question 1), so
$\abs{f(t)} \leq \frac Cn$. Given $\varepsilon > 0$, choose $n >
\frac C\varepsilon$: for $0 \leq h \leq \frac\ell n$,
$\abs{f(h)} \leq \varepsilon$, and for negative $h$ use $f(h) =
-f(-h)$. So $f(h) \to 0 = f(0)$ as $h \to 0$: \omterm{def:b1:continuity:continuous}{continuity} at $0$,
hence everywhere (question 2), hence $f(x) = f(1)x$.

\textbf{7.} If $f$ is nondecreasing on $\intcc{a}{b}$, then
$f(a) \leq f(x) \leq f(b)$ there: bounded, so linear by question
6, $f(x) = cx$; and $c(b - a) = f(b) - f(a) \geq 0$ forces $c
\geq 0$.

\textbf{8.} (a)$\Rightarrow$(b)$\Rightarrow$(c): trivial.
(c)$\Rightarrow$(a): question 2. (a)$\Rightarrow$(d): a linear
function is monotone everywhere. (d)$\Rightarrow$(e): a monotone
function on $\intcc{a}{b}$ is bounded there by its endpoint
values. (e)$\Rightarrow$(a): questions 4--6. The five
\omterm{def:b1:logic:statement}{statements} are equivalent --- the regularity ladder.

\textbf{9.} Question 4 used only the \omterm{def:b1:reals:bounds}{upper bound} $M$; question 5
\emph{derived} the lower bound from the upper one via the
reflection $f(t) = f(\ell) - f(\ell - t)$; question 6 then ran on
$\abs f \leq C$. So: additive and bounded \emph{above} on one
nondegenerate \omterm{prop:b1:reals:intervals}{interval} already implies linear. For a bound
below, apply this to $-f$ (additive, bounded above). Strongest
rung (e): a one-sided bound on one \omterm{prop:b1:reals:intervals}{interval} suffices.

\textbf{10.} If $\frac{f(x)}{x}$ were one constant $c$ for all
$x \neq 0$, $f$ would be linear; so there are nonzero $u, v$
with $\frac{f(u)}u \neq \frac{f(v)}v$, i.e.\ $u f(v) - v f(u)
\neq 0$: the determinant of the vectors $(u, f(u))$, $(v,
f(v))$ is nonzero, and they span $\R^2$.

\textbf{11.} For $r, s \in \Q$: $f(ru + sv) = r f(u) + s f(v)$
(question 1 twice plus additivity), so the graph contains
\[
\bigl(ru + sv,\; r f(u) + s f(v)\bigr)
= r\,(u, f(u)) + s\,(v, f(v)) .
\]
Given $(x_0, y_0)$ and $\varepsilon > 0$: the $2 \times 2$
system $a(u, f(u)) + b(v, f(v)) = (x_0, y_0)$ has a (unique)
real solution $(a, b)$ since the determinant is nonzero. Pick
rationals $r_n \to a$, $s_n \to b$: then $r_n(u, f(u)) +
s_n(v, f(v)) \to (x_0, y_0)$ coordinatewise, and each of these
points lies on the graph: the graph is \omterm{def:b1:topology:dense}{dense} in $\R^2$.

\textbf{12.} Let $I$ be a nondegenerate \omterm{prop:b1:reals:intervals}{interval}, $x_0$ its
midpoint, $M$ arbitrary: \omterm{def:b1:topology:dense}{density} provides a graph point within
$\min\bigl(\frac{\abs I}{2}, 1\bigr)$ of $(x_0, M + 1)$, i.e.\
$x \in I$ with $f(x) > M$: unbounded on $I$, hence (question 8)
discontinuous at every point and monotone on no \omterm{prop:b1:reals:intervals}{interval}; and
for any target $y_0$, graph points near $(x_0, y_0)$ give $f(x)$
arbitrarily close to $y_0$ with $x \in I$: $f(I)$ is \omterm{def:b1:topology:dense}{dense} in
$\R$. This is question 8 read backwards: since all rungs are
equivalent, a nonlinear additive function must fail \emph{all}
of them, everywhere.

\textbf{13.} Well defined: $m + n\sqrt2 = m' + n'\sqrt2$ forces
$(n - n')\sqrt2 = m' - m \in \Z$, so $n = n'$ (else $\sqrt2 \in
\Q$) and $m = m'$. Additivity on $G$ is then clear coordinate by
coordinate. Unboundedness near $0^+$: fix $\varepsilon \in
\intoo{0}{1}$ and $N \in \N$. For each fixed $n$ with $\abs n
\leq N$, the condition $m + n\sqrt2 \in \intoo{0}{1}$ pins $m$
inside an \omterm{prop:b1:reals:intervals}{interval} of length $1$: at most one integer $m$ per
$n$, so at most $2N + 1$ elements of $G \cap \intoo{0}{1}$ have
$\abs{f} \leq N$. But $G \cap \intoo{0}{\varepsilon}$ is
infinite ($G$ is \omterm{def:b1:topology:dense}{dense}, \cref{exo:b1:reals:9}); hence it
contains some $g$ with $\abs{f(g)} > N$: $f$ is unbounded on
every right \omterm{def:b1:topology:open}{neighborhood} of $0$. Extending $f$ to an additive
function on $\R$ means choosing values coherently on a family of
reals that is $\Q$-linearly independent and spans $\R$ over $\Q$
--- a Hamel basis; producing one requires the axiom of choice,
and no explicit formula can do it: constructively we own the
monster only on $G$.

\textbf{14.} $f(x) = f(\frac x2 + \frac x2) = f(\frac x2)^2 \geq
0$. If $f(x_0) = 0$, then $f(x) = f(x - x_0)f(x_0) = 0$ for all
$x$: excluded. So $f > 0$ and $g = \ln \circ f$ is \omterm{def:b1:continuity:continuous}{continuous}
(\cref{prop:b1:continuity:algebra}) with $g(x + y) = g(x) +
g(y)$: by \cref{exo:b1:continuity:11}, $g(x) = cx$, so $f(x) =
\eu^{cx}$. Conversely each $\eu^{cx}$ works: the \omterm{def:b1:continuity:continuous}{continuous}
\omterm{def:b1:structures:morphism}{morphisms} $(\R, +) \to (\R^*, \times)$ are exactly the
exponentials.

\textbf{15.} $g(u) = f(\eu^u)$ is \omterm{def:b1:continuity:continuous}{continuous} and $g(u + v) =
f(\eu^u \eu^v) = g(u) + g(v)$: $g(u) = cu$, and every $x > 0$
writes $x = \eu^u$ with $u = \ln x$: $f(x) = c\ln x$.

\textbf{16.} $h = \ln \circ f$ is \omterm{def:b1:continuity:continuous}{continuous} on
$\intoo{0}{+\infty}$ with $h(xy) = h(x) + h(y)$: by question 15,
$h(x) = c\ln x$, so $f(x) = \eu^{c\ln x} = x^c$.

\textbf{17.} $x = y = 0$: $f(0) = 2f(0) + f(0)^2$, so $f(0)(1 +
f(0)) = 0$. If $f(0) = -1$: setting $y = 0$, $f(x) = f(x) +
f(0) + f(x)f(0) = -1$ for all $x$: the constant $f \equiv -1$
(which indeed satisfies the equation). Otherwise $f(0) = 0$; $h
= 1 + f$ is \omterm{def:b1:continuity:continuous}{continuous}, $h(0) = 1$, and
\[
h(x + y) = 1 + f(x) + f(y) + f(x)f(y) = h(x)\,h(y) :
\]
by question 14, $h(x) = \eu^{cx}$, i.e.\ $f(x) = \eu^{cx} - 1$
(the case $c = 0$ giving $f \equiv 0$). Complete list: $f
\equiv -1$ and $f(x) = \eu^{cx} - 1$, $c \in \R$.

\textbf{18.} $x = y = 0$: $2f(0) = 4f(0)$, so $f(0) = 0$. $x =
0$: $f(y) + f(-y) = 2f(y)$, so $f$ is even. $y = x$: $f(2x) =
4f(x)$. Induction using $(x, y) \to (nx, x)$:
\[
f((n{+}1)x) = 2f(nx) + 2f(x) - f((n{-}1)x)
= (2n^2 + 2 - (n-1)^2) f(x) = (n+1)^2 f(x) .
\]
Then $f(x) = f\bigl(q\cdot\frac xq\bigr) = q^2 f\bigl(\frac
xq\bigr)$ gives $f\bigl(\frac pq x\bigr) = \frac{p^2}{q^2}f(x)$:
$f(r) = r^2 f(1)$ on $\Q$ (evenness handles signs). The two
\omterm{def:b1:continuity:continuous}{continuous} functions $f$ and $x \mapsto f(1)x^2$ agree on the
\omterm{def:b1:topology:dense}{dense} \omterm{def:b1:logic:sets}{set} $\Q$, hence everywhere (question 24): $f(x) =
f(1)\,x^2$; every $cx^2$ satisfies the equation.

\textbf{19.} $g = f - f(0)$ is \omterm{def:b1:continuity:continuous}{continuous}, $g(0) = 0$, and
satisfies Jensen's equation (the constants cancel). Taking $y =
0$: $g\bigl(\frac x2\bigr) = \frac{g(x)}{2}$. Then for all $x,
y$:
\[
\frac{g(x + y)}{2} = g\Bigl(\frac{x+y}{2}\Bigr)
= \frac{g(x) + g(y)}{2} ,
\]
so $g$ is additive and \omterm{def:b1:continuity:continuous}{continuous}: $g(x) = cx$ (question 2), and
$f(x) = cx + d$ with $d = f(0)$. All affine functions satisfy
Jensen: the list is complete.

\textbf{20.} Induction on $n$. For $n = 0$: $\lambda \in \{0,
1\}$, trivial. Assume the inequality for all weights
$\frac{k}{2^n}$. A weight $\lambda = \frac{k}{2^{n+1}}$ with $k$
even reduces to level $n$; for $k = 2j + 1$, $\lambda$ is the
midpoint of $\lambda_1 = \frac{j}{2^n}$ and $\lambda_2 =
\frac{j+1}{2^n}$. With $z_i = \lambda_i x + (1 - \lambda_i)y$:
$\lambda x + (1-\lambda)y = \frac{z_1 + z_2}{2}$, so
\[
f\bigl(\lambda x + (1{-}\lambda)y\bigr)
\leq \frac{f(z_1) + f(z_2)}{2}
\leq \frac{(\lambda_1 + \lambda_2)f(x) + (2 - \lambda_1 -
\lambda_2)f(y)}{2}
= \lambda f(x) + (1 - \lambda)f(y) .
\]

\textbf{21.} Fix $x, y$. The \omterm{def:b1:logic:map}{maps} $\lambda \mapsto f(\lambda x +
(1 - \lambda)y)$ and $\lambda \mapsto \lambda f(x) + (1 -
\lambda)f(y)$ are \omterm{def:b1:continuity:continuous}{continuous} on $\intcc{0}{1}$ (composition and
algebra, \cref{prop:b1:continuity:algebra}). The inequality
holds on the dyadic weights, which are \omterm{def:b1:topology:dense}{dense} in $\intcc{0}{1}$
(\cref{exo:b1:reals:8}); for arbitrary $\lambda$ take dyadics
$\lambda_n \to \lambda$ and pass to the limit
(\cref{thm:b1:continuity:seqchar} and \cref{thm:b1:seq:order}):
the convexity inequality holds for every $\lambda \in
\intcc{0}{1}$ --- midpoint convexity plus \omterm{def:b1:continuity:continuous}{continuity} equals
convexity (the notion of \cref{ch:b1:derivative}).

\textbf{22.} A nonlinear additive $f$ satisfies
$f\bigl(\frac{x+y}{2}\bigr) = \frac{f(x) + f(y)}{2}$ exactly
(question 1 with $r = \frac12$, then additivity): it is
midpoint-convex, even midpoint-affine. If it satisfied the full
convexity inequality on some \omterm{prop:b1:reals:intervals}{interval} $\intcc{x}{y}$, then for
$\lambda \in \intcc{0}{1}$: $f(\lambda x + (1-\lambda)y) \leq
\max(f(x), f(y))$ --- bounded above on a nondegenerate \omterm{prop:b1:reals:intervals}{interval},
hence linear by question 9: contradiction. So \omterm{def:b1:continuity:continuous}{continuity} in
question 21 is not a luxury: without it, midpoint convexity
controls only the countable dyadic skeleton, and the continuum
in between runs wild.

\textbf{23.} $p(x) = \frac{x^2}{2}$ satisfies $p(x+y) = p(x) +
p(y) + xy$. If $f$ is any \omterm{def:b1:continuity:continuous}{continuous} solution, $g = f - p$ is
\omterm{def:b1:continuity:continuous}{continuous} and additive, so $g(x) = cx$:
\[
f(x) = \frac{x^2}{2} + cx , \qquad c \in \R ,
\]
and each of these is a solution: the list is complete.

\textbf{24.} General principle: if $u, v$ are \omterm{def:b1:continuity:continuous}{continuous} and
agree on a \omterm{def:b1:topology:dense}{dense} $D \subseteq \R$, then for $x \in \R$ choose
$d_n \in D$ with $d_n \to x$
(\cref{prop:b1:topology:closureprops}); $u(x) = \lim u(d_n) =
\lim v(d_n) = v(x)$. Now let $f$ be \omterm{def:b1:continuity:continuous}{continuous} and additive on
the \omterm{def:b1:topology:dense}{dense} \omterm{def:b1:structures:subgroup}{subgroup} $G$. Fix $x, y \in \R$ and take $g_n \to x$,
$g'_n \to y$ with $g_n, g'_n \in G$; then $g_n + g'_n \to x + y$
and, by sequential \omterm{def:b1:continuity:continuous}{continuity} at $x + y$, $x$ and $y$:
\[
f(x + y) = \lim f(g_n + g'_n) = \lim\bigl(f(g_n) +
f(g'_n)\bigr) = f(x) + f(y) :
\]
$f$ is additive on all of $\R$ (hence linear, by question 2).

\textbf{25.} (i) For additive $f$: linear $\iff$ \omterm{def:b1:continuity:continuous}{continuous}
$\iff$ \omterm{def:b1:continuity:continuous}{continuous} at one point $\iff$ monotone on some \omterm{prop:b1:reals:intervals}{interval}
$\iff$ bounded (even one-sidedly) on some \omterm{prop:b1:reals:intervals}{interval}. (ii) \omterm{def:b1:topology:dense}{Density}
of the graph in the plane shows the failure is not a local
defect but a global explosion: above every subinterval the
values smear over all of $\R$, so every regularity property
fails everywhere at once. (iii) The catch: $cx$, $\eu^{cx}$,
$c\ln x$, $x^c$, $cx^2$, and $cx + d$ --- six characterizations,
one method: transport the equation to Cauchy's, prove the
$\Q$-skeleton by induction, upgrade to $\R$ by \omterm{def:b1:topology:dense}{density} plus
\omterm{def:b1:continuity:continuous}{continuity}. (iv) \omterm{def:b1:topology:dense}{Density} of $\Q$ (or the dyadics) carried the
upgrades in \cref{exo:b1:continuity:11} and in question 21; it
carried nothing in question 13, because without \omterm{def:b1:continuity:continuous}{continuity}
values do not propagate from a \omterm{def:b1:topology:dense}{dense} \omterm{def:b1:logic:sets}{set} to its \omterm{def:b1:topology:closure}{closure} ---
\omterm{def:b1:topology:dense}{density} transfers information only along \omterm{def:b1:continuity:continuous}{continuity}.
