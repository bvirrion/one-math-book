\section*{Chapter \ref{ch:b1:derivative} --- Differentiation}

\begin{solution}{exo:b1:derivative:1}
$x^x = \eu^{x\ln x}$ on $\intoo{0}{+\infty}$: derivative $(\ln x +
1)\,x^x$.

$\ln(x + \sqrt{x^2+1})$ on $\R$ (the argument is always $> 0$):
derivative $\frac{1}{\sqrt{x^2+1}}$ (computed in
\cref{prop:b1:functions:invhyp} --- it is
$\operatorname{arsinh}$).

$\arctan\frac1x$ on $\R^*$: derivative $\frac{-1/x^2}{1 + 1/x^2} =
\frac{-1}{1 + x^2}$ (consistent with
\cref{prop:b1:functions:arcidentities}~(2): the function is
$\pm\frac\pi2 - \arctan x$ on each half-line).

$\sqrt{1 + \eu^{2x}}$ on $\R$: derivative
$\frac{\eu^{2x}}{\sqrt{1 + \eu^{2x}}}$.
\end{solution}

\begin{solution}{exo:b1:derivative:2}
At $0$: $\bigl|\frac{f(h) - 0}{h}\bigr| = \abs{h \sin\frac1h} \leq
\abs h \to 0$, so $f'(0) = 0$. For $x \neq 0$, the usual rules give
$f'(x) = 2x\sin\frac1x - \cos\frac1x$. Along $x_n = \frac{1}{2\pi
n}$: $f'(x_n) = 0 - 1 \to -1$; along $y_n = \frac{1}{(2n+1)\pi}$:
$f'(y_n) = 0 + 1 \to 1$. Two sequences tending to $0$ with different
limits of $f'$: no limit (\cref{thm:b1:continuity:seqchar}), so $f'$
is not continuous at $0$ and $f$ is differentiable without being
$C^1$.
\end{solution}

\begin{solution}{exo:b1:derivative:3}
$\ln(1+x) < x$ for $x > 0$: concavity tangent inequality at $0$
(strict away from the contact point since $\ln$ is strictly concave;
or apply the mean value theorem: $\ln(1+x) = \frac{x}{1+c}$ for some
$c \in \intoo{0}{x}$, and $\frac{x}{1+c} < x$). The same mean value
identity gives the lower bound: $\frac{x}{1+c} > \frac{x}{1+x}$.

Consequence: with $x/n$ in place of $x$,
\[
\frac{x/n}{1 + x/n} < \ln\Bigl(1 + \frac xn\Bigr) < \frac xn
\quad\implies\quad
\frac{x}{1 + x/n} < n \ln\Bigl(1 + \frac xn\Bigr) < x .
\]
The left member tends to $x$: by the squeeze, $n\ln(1 + \frac xn)
\to x$, and by continuity of $\exp$, $\bigl(1 + \frac xn\bigr)^n =
\eu^{n\ln(1 + x/n)} \to \eu^x$.
\end{solution}

\begin{solution}{exo:b1:derivative:4}
Let $x_1 < x_2 < \dots < x_k$ be distinct roots of $P$. On each
$\intcc{x_i}{x_{i+1}}$, Rolle (\cref{thm:b1:derivative:rolle})
produces $c_i \in \intoo{x_i}{x_{i+1}}$ with $P'(c_i) = 0$: that is
$k - 1$ roots of $P'$, distinct because the open intervals are
disjoint --- and interlaced by construction.

If $P$ (degree $n$) has all roots real, write them with
multiplicities $m_1 + \dots + m_k = n$. Each root of multiplicity
$m_i \geq 2$ is a root of $P'$ of multiplicity $m_i - 1$
(\cref{prop:b1:poly:multiplicity}), contributing $\sum (m_i - 1) = n
- k$; Rolle contributes $k - 1$ more, all distinct from these. Total
$\geq n - 1 = \deg P'$: all roots of $P'$ are real.
\end{solution}

\begin{solution}{exo:b1:derivative:5}
Fix $\varepsilon > 0$ and $A$ with $\abs{f'(t) - \ell} \leq
\varepsilon$ for $t \geq A$. For $x > A$, the mean value theorem on
$\intcc{A}{x}$ gives $c \in \intoo{A}{x}$ with
\[
f(x) = f(A) + f'(c)(x - A),
\qquad\text{so}\qquad
\Bigl|\frac{f(x)}{x} - \ell\Bigr|
\leq \frac{\abs{f(A)} + \abs\ell A}{x} + \abs{f'(c) - \ell}
\cdot\frac{x - A}{x} \leq \frac{C_A}{x} + \varepsilon .
\]
For $x$ large, $\frac{C_A}{x} \leq \varepsilon$: hence
$\frac{f(x)}{x} \to \ell$.

Yes: $f(x+1) - f(x) = f'(c_x)$ with $c_x \in \intoo{x}{x+1}$ (mean
value theorem on $\intcc{x}{x+1}$), and $c_x \to +\infty$, so
$f(x+1) - f(x) \to \ell$.
\end{solution}

\begin{solution}{exo:b1:derivative:6}
Induction on $n$. For $n = 1$: Rolle. If the claim holds for $n - 1$:
$f$ vanishes at $n+1$ points, so by Rolle applied on the $n$ gaps,
$f'$ vanishes at $n$ distinct points; the induction hypothesis
applied to $f'$ ($n-1$ times differentiable, $n$ zeros) makes
$(f')^{(n-1)} = f^{(n)}$ vanish somewhere.

Application: if $P$ of degree $\leq n$ vanishes at $n+1$ points, then
$P^{(n)}$, a constant equal to $n!$ times the leading coefficient,
vanishes: the leading coefficient is $0$, and one concludes by
downward induction (or directly: all coefficients vanish).
\end{solution}

\begin{solution}{exo:b1:derivative:7}
By the mean value theorem on $\intcc{a}{x_0}$ and on
$\intcc{x_0}{b}$:
\[
f'(c_1) = \frac{f(x_0) - f(a)}{x_0 - a} = \frac{f(x_0)}{x_0 - a} > 0,
\qquad
f'(c_2) = \frac{f(b) - f(x_0)}{b - x_0} = \frac{-f(x_0)}{b - x_0} < 0,
\]
with $c_1 < x_0 < c_2$. Then the mean value theorem applied to $f'$
on $\intcc{c_1}{c_2}$ gives $c$ with
\[
f''(c) = \frac{f'(c_2) - f'(c_1)}{c_2 - c_1} < 0 . \qedhere
\]
\end{solution}

\begin{solution}{exo:b1:derivative:8}
$f'(x) = \frac{1 - \ln x}{x^2}$: $f$ increases on $\intoc{0}{\eu}$,
decreases on $\intco{\eu}{+\infty}$, with maximum $f(\eu) =
\frac1\eu$; limits $-\infty$ at $0^+$ and $0$ at $+\infty$ (growth
comparison).

For $\eu \leq a < b$: $f$ strictly decreasing there gives
$\frac{\ln a}{a} > \frac{\ln b}{b}$, i.e.\ $b \ln a > a \ln b$,
i.e.\ $a^b > b^a$.

With $a = \eu < b = \pi$: $\eu^\pi > \pi^\eu$.

Small pairs: $(2, 3)$: $2^3 = 8 < 9 = 3^2$ --- reversed! The reason:
$2 < \eu$, and on $\intoo{0}{\eu}$ the function $f$ is
\emph{increasing}, so the comparison flips when both numbers sit
below $\eu$, and is unpredictable across $\eu$ ($f(2) = f(4)$
explains the tie $2^4 = 4^2 = 16$).
\end{solution}

\begin{solution}{exo:b1:derivative:9}
\emph{Jensen for $\ln$, by induction on $n$.} Claim: for positive
$x_i$ and weights $\lambda_i > 0$ with $\sum \lambda_i = 1$:
$\ln\bigl(\sum \lambda_i x_i\bigr) \geq \sum \lambda_i \ln x_i$. For
$n = 2$ this is concavity. Step: with $\Lambda = \lambda_1 + \dots +
\lambda_{n-1} = 1 - \lambda_n$ and $y = \sum_{i<n}
\frac{\lambda_i}{\Lambda} x_i$,
\[
\ln\Bigl(\sum_{i \leq n} \lambda_i x_i\Bigr)
= \ln\bigl(\Lambda y + \lambda_n x_n\bigr)
\geq \Lambda \ln y + \lambda_n \ln x_n
\geq \Lambda \sum_{i<n} \frac{\lambda_i}{\Lambda}\ln x_i
+ \lambda_n \ln x_n,
\]
using concavity ($n = 2$) then the induction hypothesis.

With $\lambda_i = \frac 1n$ and $x_i = a_i$: $\ln\frac{\sum a_i}{n}
\geq \frac 1n \sum \ln a_i = \ln\sqrt[n]{a_1\cdots a_n}$;
exponentiate. Equality: $\ln$ is \emph{strictly} concave ($\ln'' < 0$),
so equality at each step forces the averaged points to coincide ---
i.e.\ all $a_i$ equal; and if all are equal, equality is clear.
\end{solution}

\begin{solution}{exo:b1:derivative:10}
Let $g(x) = f(x) - vx$: differentiable, with $g'(a) = f'(a) - v < 0$
and $g'(b) = f'(b) - v > 0$. By the extreme value theorem, $g$
attains its minimum on $\intcc{a}{b}$ at some $c$. It is not at $a$:
since $g'(a) < 0$, points just right of $a$ have $g < g(a)$. It is
not at $b$: since $g'(b) > 0$, points just left of $b$ have $g <
g(b)$. So $c$ is interior, and \cref{prop:b1:derivative:fermat}
gives $g'(c) = 0$, i.e.\ $f'(c) = v$.
\end{solution}

\begin{solution}{exo:b1:derivative:11}
\emph{Uniqueness:} two fixed points $\ell \neq \ell'$ would give
$\abs{\ell - \ell'} = \abs{f(\ell) - f(\ell')} \leq k\abs{\ell -
\ell'} < \abs{\ell - \ell'}$, absurd.

\emph{Existence:} $g(x) = f(x) - x$ satisfies, by the mean value
inequality, $f(x) \leq f(0) + k\abs x$; so for $x \geq
\frac{\abs{f(0)}}{1 - k}$, $g(x) \leq f(0) + kx - x \leq 0$, and
symmetrically $g(-x) \geq 0$ for $x$ large. The intermediate value
theorem gives a zero $\ell$ of $g$: a fixed point.

\emph{Convergence:} the mean value inequality again:
\[
\abs{u_{n+1} - \ell} = \abs{f(u_n) - f(\ell)} \leq k\abs{u_n - \ell},
\]
so by induction $\abs{u_n - \ell} \leq k^n \abs{u_0 - \ell} \to 0$.
\end{solution}
