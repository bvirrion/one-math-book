\section*{Hoofdstuk \ref{ch:b1:functions} --- Standaardfuncties}

\begin{solution}{exo:b1:functions:1}
$\arcsin\frac{\sqrt 3}{2} = \frac\pi3$;
$\;\arccos\bigl(-\frac12\bigr) = \frac{2\pi}{3}$;
$\;\arctan(-1) = -\frac\pi4$.

$\arcsin\bigl(\sin\frac{5\pi}{6}\bigr)$: $\sin\frac{5\pi}{6} =
\frac12$, en de hoek van $\intcc{-\frac\pi2}{\frac\pi2}$ met sinus
$\frac12$ is $\frac\pi6$ (niet $\frac{5\pi}{6}$).

$\arccos\bigl(\cos\frac{7\pi}{4}\bigr)$: $\cos\frac{7\pi}{4} =
\frac{\sqrt 2}{2}$, en de hoek van $\intcc{0}{\pi}$ met die cosinus
is $\frac\pi4$.
\end{solution}

\begin{solution}{exo:b1:functions:2}
$f(x) = \arcsin(2x - 1)$ vereist $-1 \leq 2x - 1 \leq 1$, d.w.z.\ $x
\in \intcc{0}{1}$. Op $\intoo{0}{1}$ geven de kettingregel en
\cref{prop:b1:functions:arcderiv}
\[
f'(x) = \frac{2}{\sqrt{1 - (2x-1)^2}} = \frac{2}{\sqrt{4x - 4x^2}}
= \frac{1}{\sqrt{x(1 - x)}} .
\]
$g(x) = \arctan\sqrt x$ is gedefinieerd voor $x \geq 0$, en voor $x > 0$:
\[
g'(x) = \frac{1}{1 + x} \cdot \frac{1}{2\sqrt x}
= \frac{1}{2\sqrt x\,(1 + x)} .
\]
\end{solution}

\begin{solution}{exo:b1:functions:3}
$\cosh x + \sinh x = \frac{\eu^x + \eu^{-x}}{2} + \frac{\eu^x -
\eu^{-x}}{2} = \eu^x$. Bijgevolg voor $n \in \Z$:
\[
(\cosh x + \sinh x)^n = (\eu^x)^n = \eu^{nx}
= \cosh nx + \sinh nx .
\]
\end{solution}

\begin{solution}{exo:b1:functions:4}
Met $y = \arcsin x$: $\cos 2y = 1 - 2\sin^2 y = 1 - 2x^2$, dus
$\cos(2\arcsin x) = 1 - 2x^2$.

Met $y = \arctan x$: $\sin 2y = 2 \sin y \cos y = 2 \tan y \cos^2 y =
\frac{2\tan y}{1 + \tan^2 y}$, dus $\sin(2\arctan x) =
\dfrac{2x}{1 + x^2}$.
\end{solution}

\begin{solution}{exo:b1:functions:5}
Van traagste naar snelste:
\[
(\ln x)^{1000} \;\ll\; x^{100} \;\ll\; x^{\ln x} \;\ll\;
\eu^{\sqrt x} \;\ll\; \eu^{x/100} .
\]
Verantwoordingen: $(\ln x)^{1000}/x^{100} \to 0$ via
\cref{prop:b1:functions:powerrules} (logaritmen verliezen van machten).
$x^{100} = \eu^{100\ln x}$ en $x^{\ln x} = \eu^{(\ln x)^2}$: aangezien
$(\ln x)^2 - 100\ln x \to +\infty$, wint de tweede. $x^{\ln x} =
\eu^{(\ln x)^2} \ll \eu^{\sqrt x}$ omdat $(\ln x)^2/\sqrt x \to 0$
(logaritmen verliezen van de macht $x^{1/4}$, gekwadrateerd). Ten slotte $\sqrt x - x/100
\to -\infty$, dus $\eu^{\sqrt x} \ll \eu^{x/100}$.
\end{solution}

\begin{solution}{exo:b1:functions:6}
Domein: $x \neq \pm 1$, drie intervallen. Op elk,
\[
f'(x) = \frac{\bigl(\frac{2x}{1-x^2}\bigr)'}{1 +
\frac{4x^2}{(1-x^2)^2}}
= \frac{\frac{2(1-x^2) + 4x^2}{(1-x^2)^2}}
{\frac{(1-x^2)^2 + 4x^2}{(1-x^2)^2}}
= \frac{2(1 + x^2)}{(1 + x^2)^2} = \frac{2}{1 + x^2}
= (2\arctan x)' .
\]
Dus $f(x) - 2\arctan x$ is constant op elk interval. Waarden: in $x =
0$ is $f(0) = 0$: de constante is $0$ op $\intoo{-1}{1}$. Als $x \to
+\infty$, $\frac{2x}{1-x^2} \to 0^-$ dus $f \to 0$, terwijl $2\arctan x
\to \pi$: de constante is $-\pi$ op $\intoo{1}{+\infty}$. Door de oneven pariteit
is ze $+\pi$ op $\intoo{-\infty}{-1}$. Samenvatting: $f = 2\arctan x$ op
$\intoo{-1}{1}$, $f = 2\arctan x - \pi$ voor $x > 1$, $f = 2\arctan x +
\pi$ voor $x < -1$. (Dit is de verdubbelingsformule voor de tangens,
gelezen via $\arctan$.)
\end{solution}

\begin{solution}{exo:b1:functions:7}
$\cosh x = 2$: met $u = \eu^x > 0$, $u + u^{-1} = 4$, dus $u^2 - 4u +
1 = 0$, $u = 2 \pm \sqrt 3$: $x = \ln(2 + \sqrt 3)$ of $x = \ln(2 -
\sqrt 3) = -\ln(2 + \sqrt 3)$ (de twee oplossingen zijn tegengesteld, aangezien
$\cosh$ even is; beide zijn geldig). Equivalent, $x = \pm
\operatorname{arcosh} 2$.

$5\cosh x - 4\sinh x = 3$: door de exponentiële definities te substitueren,
$\frac{5(u + u^{-1}) - 4(u - u^{-1})}{2} = 3$, d.w.z.\ $u + 9u^{-1} =
6$, d.w.z.\ $u^2 - 6u + 9 = (u - 3)^2 = 0$: $u = 3$, één oplossing $x =
\ln 3$.
\end{solution}

\begin{solution}{exo:b1:functions:8}
Zij $\alpha = \arctan\frac12 + \arctan\frac13$. Optelformule:
\[
\tan\alpha = \frac{\frac12 + \frac13}{1 - \frac12\cdot\frac13}
= \frac{5/6}{5/6} = 1 .
\]
Beide boogtangensen liggen in $\intoo{0}{\frac\pi4}$ (hun argumenten liggen in
$\intoo{0}{1}$), dus $\alpha \in \intoo{0}{\frac\pi2}$; de enige hoek
daar met tangens $1$ is $\frac\pi4$.

Machin: zij $\beta = \arctan\frac15$. Verdubbeling tweemaal:
\[
\tan 2\beta = \frac{2/5}{1 - 1/25} = \frac{5}{12},
\qquad
\tan 4\beta = \frac{2 \cdot 5/12}{1 - 25/144} = \frac{120}{119}.
\]
Vervolgens
\[
\tan\Bigl(4\beta - \frac\pi4\Bigr)
= \frac{\frac{120}{119} - 1}{1 + \frac{120}{119}}
= \frac{1/119}{239/119} = \frac{1}{239}.
\]
Locatie: $\beta < \arctan 1 = \frac\pi4$ en in feite $\tan 4\beta =
\frac{120}{119}$ dicht bij $1$ met $4\beta \in \intoo{0}{\frac\pi2}$
(aangezien $\beta < \frac\pi8$, want $\tan\frac\pi8 = \sqrt 2 - 1 > \frac15$);
dus $4\beta - \frac\pi4 \in \intoo{-\frac\pi4}{\frac\pi4}$, waar
$\arctan$ de $\tan$ inverteert: $4\beta - \frac\pi4 =
\arctan\frac{1}{239}$, wat de formule van Machin is.
\end{solution}

\begin{solution}{exo:b1:functions:9}
Zij $f(x) = x - \sin x$: $f(0) = 0$ en $f'(x) = 1 - \cos x \geq 0$,
dus $f \geq 0$ op $\R_+$: $\sin x \leq x$.

Zij $g(x) = \sin x - x + \frac{x^3}{6}$: $g(0) = 0$, $g'(x) = \cos x -
1 + \frac{x^2}{2}$, $g'(0) = 0$, $g''(x) = -\sin x + x = f(x) \geq 0$
op $\R_+$. Dus $g'$ is stijgend met $g'(0) = 0$, dus $g' \geq 0$,
dus $g$ is stijgend met $g(0) = 0$: $g \geq 0$, d.w.z.\ $x -
\frac{x^3}{6} \leq \sin x$ op $\R_+$.

Bijgevolg $0 \leq \frac{x - \sin x}{x^3} \leq \frac 16$ voor $x >
0$: de verhouding is gevangen nabij de schaal van $\frac16$, en
\cref{ch:b1:taylor} toont dat haar limiet precies $\frac 16$ is.
\end{solution}

\begin{solution}{exo:b1:functions:10}
Schrijf $y = \arcsin x$, zodat $2x\sqrt{1 - x^2} = 2\sin y\cos y = \sin
2y$ en $h(x) = \arcsin(\sin 2y) - 2y$.

Voor $\abs x \leq \frac{\sqrt 2}{2}$: $2y \in
\intcc{-\frac\pi2}{\frac\pi2}$, dus $\arcsin(\sin 2y) = 2y$ en $h =
0$.

Voor $x \in \intoc{\frac{\sqrt 2}{2}}{1}$: $2y \in
\intoc{\frac\pi2}{\pi}$, en de hoek van
$\intcc{-\frac\pi2}{\frac\pi2}$ met sinus $\sin 2y$ is $\pi - 2y$:
dus $h(x) = \pi - 4\arcsin x$. (Controle via de afgeleide: daar is
$h'(x) = \frac{2(1 - 2x^2)}{\abs{1 - 2x^2}\sqrt{1 - x^2}} -
\frac{2}{\sqrt{1-x^2}} = \frac{-4}{\sqrt{1 - x^2}}$, de afgeleide
van $-4\arcsin x$; en in $x = 1$, $h(1) = \arcsin 0 - 2\cdot\frac\pi2
= -\pi = \pi - 4\cdot\frac\pi2$.)

Voor $x \in \intco{-1}{-\frac{\sqrt 2}{2}}$, door de oneven pariteit van $h$:
$h(x) = -\pi - 4\arcsin x$.
\end{solution}

\begin{solution}{exo:b1:functions:11}
$g = \arctan \circ \sinh$ is een samenstelling van oneven, strikt
stijgende functies, dus is ze oneven en strikt stijgend; als $x \to
+\infty$ is $\sinh x \to +\infty$ dus $g(x) \to \frac\pi2$, en $g$ is een
bijectie op $\intoo{-\frac\pi2}{\frac\pi2}$ (continuïteit plus de
tussenwaardestelling). Kettingregel met
\cref{prop:b1:functions:hyprules}~(1):
\[
g'(x) = \frac{\cosh x}{1 + \sinh^2 x} = \frac{\cosh x}{\cosh^2 x}
= \frac{1}{\cosh x} .
\]
Per constructie is $\tan g(x) = \sinh x$. Dan, aangezien $g(x) \in
\intoo{-\frac\pi2}{\frac\pi2}$ positieve cosinus heeft,
\begin{align*}
\cos g(x) &= \frac{1}{\sqrt{1 + \tan^2 g(x)}}
= \frac{1}{\sqrt{1 + \sinh^2 x}} = \frac{1}{\cosh x},\\
\sin g(x) &= \tan g(x) \cos g(x) = \frac{\sinh x}{\cosh x} = \tanh x .
\end{align*}
\end{solution}

\begin{solution}{exo:b1:functions:12}
Stel $u = \arctan 2$ en $v = \arctan 3$; beide liggen in
$\intoo{\frac\pi4}{\frac\pi2}$ (hun argumenten overschrijden $1$), dus $u
+ v \in \intoo{\frac\pi2}{\pi}$. De optelformule geeft
\[
\tan(u + v) = \frac{2 + 3}{1 - 6} = -1 ,
\]
en de unieke hoek van $\intoo{\frac\pi2}{\pi}$ met tangens $-1$
is $\frac{3\pi}4$: dus $\arctan 2 + \arctan 3 = \frac{3\pi}4$.
(Blindelings $\arctan$ toepassen op de tangens zou
$-\frac\pi4$ geven, $\pi$ ernaast --- de lokalisering is wat de
berekening redt, en de negatieve noemer $1 - 6$ is precies het
signaal dat de som
$\intoo{-\frac\pi2}{\frac\pi2}$ heeft verlaten.) Optellen van $\arctan 1 = \frac\pi4$:
\[
\arctan 1 + \arctan 2 + \arctan 3 = \frac\pi4 + \frac{3\pi}4 = \pi .
\]
\end{solution}

\begin{solution}{pb:b1:functions:1}
\textbf{1.} $f$ is een quotiënt van afleidbare functies met
niet-nul noemer op $\intoo0{+\infty}$, en
\[
f'(t) = \frac{\frac1t \cdot t - \ln t}{t^2} = \frac{1 - \ln
t}{t^2} ,
\]
positief voor $t < \eu$, nul in $t = \eu$, negatief voor $t > \eu$:
$f$ stijgt strikt op $\intoc0\eu$, daalt strikt op
$\intco\eu{+\infty}$, met maximum $f(\eu) = \frac1\eu$.

\textbf{2.} Als $t \to 0^+$: $\ln t \to -\infty$ en $\frac1t \to
+\infty$, dus $f(t) \to -\infty$. Als $t \to +\infty$: $f(t) \to 0$
via de \omterm{prop:b1:functions:powerrules}{groeivergelijking} van \cref{prop:b1:functions:powerrules}
($\beta = \alpha = 1$). Teken: dat van $\ln t$, dus $f < 0$ op
$\intoo01$, $f(1) = 0$, $f > 0$ op $\intoo1{+\infty}$. De grafiek
klimt van $-\infty$, kruist nul in $1$, piekt in $(\eu,
\frac1\eu)$, en vervalt dan naar $0^+$.

\textbf{3.} $f(4) = \frac{\ln 4}4 = \frac{2\ln 2}4 = \frac{\ln 2}2
= f(2)$.

\textbf{4.} Via de variatietabel en de tussenwaarde-
stelling (gebruikt op middelbareschoolniveau; geformaliseerd in
\cref{ch:b1:continuity}). Voor $c < 0$: oplossingen bestaan enkel waar
$f < 0$, d.w.z.\ in $\intoo01$ waar $f$ een strikt stijgende
bijectie is op $\intoo{-\infty}0$: precies één. Voor $c = 0$: enkel
$t = 1$. Voor $0 < c < \frac1\eu$: op $\intoo1\eu$ stijgt $f$
van $0$ naar $\frac1\eu$: één oplossing; op $\intoo\eu{+\infty}$
daalt $f$ van $\frac1\eu$ naar $0$: nog één; totaal twee. Voor
$c = \frac1\eu$: enkel het maximumpunt $t = \eu$. Voor $c >
\frac1\eu$: geen.

\textbf{5.} Voor $x \neq \eu$ geeft strikte maximaliteit $f(x) <
f(\eu)$, d.w.z.\ $\frac{\ln x}x < \frac1\eu$, d.w.z.\ $\eu\ln x < x$,
d.w.z.\ $\ln(x^\eu) < \ln(\eu^x)$: $x^\eu < \eu^x$. Met $x = \pi$:
$\eu^\pi > \pi^\eu$ (numeriek $23.14 > 22.46$).

\textbf{6.} Voor $x, y > 0$ zijn beide leden positief, dus
\[
x^y = y^x \iff y\ln x = x\ln y \iff \frac{\ln x}x = \frac{\ln y}y
\iff f(x) = f(y) ,
\]
delend door $xy > 0$.

\textbf{7.} Zij $f(x) = f(y) = c$ met $x \neq y$. Als $c \leq 0$ of
$c = \frac1\eu$, zegt vraag 4 dat de vergelijking $f = c$ een enkele
oplossing heeft: onmogelijk. Dus $0 < c < \frac1\eu$, en opnieuw via
vraag 4 zijn de twee oplossingen één punt van $\intoo1\eu$ en één
van $\intoo\eu{+\infty}$: beide coördinaten overschrijden $1$ en ze
liggen aan weerszijden van $\eu$.

\textbf{8.} Door $y = tx$ te substitueren in $y\ln x = x\ln y$:
\[
tx\ln x = x(\ln t + \ln x)
\iff (t - 1)\ln x = \ln t
\iff \ln x = \frac{\ln t}{t - 1} ,
\]
dus $x = t^{1/(t-1)}$ en $y = tx = t^{1 + \frac1{t-1}} =
t^{t/(t-1)}$. Omgekeerd, voor deze waarden, $\ln y = \frac{t\ln
t}{t-1} = t\ln x$ en $y = tx$, dus $\frac{\ln y}y = \frac{t\ln
x}{tx} = \frac{\ln x}x$: een oplossing. Elke niet-triviale oplossing heeft
een verhouding $t = y/x \in \intoo0{+\infty} \setminus \{1\}$, dus de
parametrisatie is volledig.

\textbf{9.} $t = 2$: $x = 2^{1/1} = 2$, $y = 2^{2/1} = 4$. En
\[
x(1/t) = (1/t)^{\frac1{\frac1t - 1}}
= (t^{-1})^{\frac{t}{1 - t}}
= t^{\frac{t}{t - 1}} = y(t) ,
\]
dan $y(1/t) = \frac1t\,x(1/t) = \frac{y(t)}t = x(t)$: de
parameterverandering $t \mapsto 1/t$ verwisselt de coördinaten, zoals de
symmetrie van de vergelijking eist.

\textbf{10.} Als $t \to 1$: $\frac{\ln t}{t-1} \to 1$ (het is het
differentiequotiënt van $\ln$ in $1$), dus $x \to \eu^1 = \eu$ en
$y = tx \to \eu$. Als $t \to +\infty$: $\ln x = \frac{\ln t}{t-1}
\to 0$ dus $x \to 1$, terwijl $\ln y = \frac{t\ln t}{t-1} \to
+\infty$ dus $y \to +\infty$. Als $t \to 0^+$: $\ln x = \frac{\ln
t}{t-1} \to \frac{-\infty}{-1} = +\infty$ dus $x \to +\infty$,
terwijl $\ln y = \frac{t\ln t}{t-1} \to \frac{0}{-1} = 0$ dus $y \to
1$ (gebruikmakend van $t\ln t \to 0$, \cref{ex:b1:functions:xx}). De tak
loopt dus van de asymptoot $y = 1$ (uiterst rechts), omhoog door
$(\eu, \eu)$ op de diagonaal, en weg langs de asymptoot $x = 1$
(uiterst boven) --- symmetrisch om de diagonaal via vraag 9.

\textbf{11.} $\ln x(t) = g(t) = \frac{\ln t}{t-1}$, en
\[
g'(t) = \frac{\frac{t-1}t - \ln t}{(t-1)^2}
= \frac{h(t)}{(t-1)^2},
\qquad h(t) = 1 - \frac1t - \ln t .
\]
$h(1) = 0$ en $h'(t) = \frac1{t^2} - \frac1t = \frac{1 -
t}{t^2} < 0$ voor $t > 1$: $h < 0$ op $\intoo1{+\infty}$, dus $g' <
0$ en $x = \eu^g$ is er strikt dalend (van $\eu$ naar
$1$, via vraag 10).

\textbf{12.} Via vraag 11 is $t \mapsto x(t)$ een strikt
dalende bijectie van $\intoo1{+\infty}$ op $\intoo1\eu$;
schrijf $t(x)$ voor haar inverse (ook strikt dalend) en stel
$\varphi(x) = y(t(x))$. Merk op dat $h < 0$ ook op $\intoo01$ (daar is
$h' > 0$ en $h(1) = 0$), dus $x(\cdot)$ is dalend op
$\intoo01$ eveneens; bijgevolg is $y(t) = x(1/t)$ (vraag 9)
\emph{stijgend} in $t$ op $\intoo1{+\infty}$, van $\eu$ naar
$+\infty$. Samenstellen: $\varphi = y \circ t(\cdot)$ is strikt
dalend van $\intoo1\eu$ op $\intoo\eu{+\infty}$, en
$x^{\varphi(x)} = \varphi(x)^x$ via vraag 8. Ten slotte, voor een
gemeenschappelijke waarde $c = f(x) \in \intoo0{\frac1\eu}$, is het paar $\{x,
\varphi(x)\}$ \emph{de} oplossingsverzameling met twee elementen van $f = c$
(vraag 4); door $\varphi$ uit te breiden naar $\intoo\eu{+\infty}$ als de
inverse \omterm{def:b1:logic:map}{afbeelding} keert $\varphi(\varphi(x))$ terug naar het andere (d.w.z.\
oorspronkelijke) element: $\varphi \circ \varphi = \mathrm{id}$.

\textbf{13.} Als $(x, y)$ een niet-triviale geheeltallige oplossing is met $x
< y$, plaatst vraag 7 $x \in \intoo1\eu$: het enige gehele getal daar
is $x = 2$. Dan is $f(y) = f(2) = \frac{\ln2}2$ met $y > \eu$; via
vraag 4 heeft de vergelijking $f = \frac{\ln2}2$ precies één
oplossing voorbij $\eu$, en vraag 3 toont ze: $y = 4$. Bijgevolg
zijn $(2, 4)$, en zijn verwisseling, de enige.

\textbf{14.} $t = 1 + \frac1n = \frac{n+1}n$ geeft $\frac1{t - 1}
= n$ en $\frac t{t-1} = n + 1$, dus
\[
x_n = \Bigl(\frac{n+1}n\Bigr)^{n},
\qquad
y_n = \Bigl(\frac{n+1}n\Bigr)^{n+1},
\]
duidelijk rationaal, verschillend ($y_n = t\,x_n \neq x_n$), en $n =
1$ geeft $x_1 = 2$, $y_1 = 4$.

\textbf{15.} $t = y/x$ is rationaal en $> 1$; schrijf $t - 1 =
\frac rs$ in onvereenvoudigbare vorm, dus $t = \frac{s + r}s$ met $\gcd(s +
r, s) = \gcd(r, s) = 1$. Vraag 8 geeft $x = t^{s/r}$, dus
\[
x^r = t^s = \frac{(s+r)^s}{s^s} ,
\]
een onvereenvoudigbare breuk (geen priemgetal deelt zowel $s + r$ als
$s$). Door $x = \frac pq$ in onvereenvoudigbare vorm te schrijven, is $x^r = \frac{p^r}
{q^r}$ ook onvereenvoudigbaar, en door de uniciteit van de gereduceerde
voorstelling: $p^r = (s+r)^s$ en $q^r = s^s$. Vergelijk de
exponent van een willekeurig priemgetal $\ell$ in $q^r = s^s$: $r \cdot
v_\ell(q) = s \cdot v_\ell(s)$, dus $r$ deelt $s\,v_\ell(s)$;
aangezien $\gcd(r, s) = 1$, deelt $r$ $v_\ell(s)$ voor elk priemgetal
$\ell$ (unieke ontbinding, aangenomen; bewezen in
\cref{ch:b1:arith}), dus $s = b^r$ voor een geheel getal $b$. Hetzelfde
argument op $p^r = (s+r)^s$ geeft $s + r = a^r$.

\textbf{16.} Veronderstel $a^r - b^r = r$ met gehele getallen $a > b \geq 1$
en $r \geq 2$. Dan $a \geq b + 1$, dus via het binomium van Newton
\[
a^r - b^r \geq (b+1)^r - b^r
= \sum_{k=0}^{r-1}\binom rk b^k
\geq \binom r0 + \binom r{r-1} b^{r-1}
\geq 1 + r > r ,
\]
een tegenspraak. Uit vraag 15 dwingt $a^r - b^r = (s + r) - s = r$
dat $r = 1$: $t = 1 + \frac1s$, en de oplossing is $(x_s,
y_s)$ van vraag 14. Samen met de verwisselde paren is de
classificatie volledig.

\textbf{17.} $f\bigl(\tfrac94\bigr) = \frac{\ln 2.25}{2.25} =
\frac{0.81093}{2.25} = 0.3604$ en $f\bigl(\tfrac{27}8\bigr) =
\frac{\ln 3.375}{3.375} = \frac{1.21640}{3.375} = 0.3604$ (vier
decimalen): gelijk, zoals de constructie belooft --- dus
$\bigl(\tfrac94\bigr)^{27/8} = \bigl(\tfrac{27}8\bigr)^{9/4}$.

\textbf{18.} De parameters $t_n = 1 + \frac1n$ dalen strikt
naar $1$. Aangezien $x(\cdot)$ strikt dalend is op
$\intoo1{+\infty}$ (vraag 11), is $x_n = x(t_n)$ strikt
\emph{stijgend}; aangezien $y(\cdot)$ er strikt stijgend is
(vraag 12), is $y_n = y(t_n)$ strikt \emph{dalend}. Via
vraag 7, $x_n \in \intoo1\eu$ en $y_n \in
\intoo\eu{+\infty}$: dus $x_n < \eu < y_n$ voor elke $n$. Ten slotte
geven $t_n \to 1$ en vraag 10 dat $x_n \to \eu$ en $y_n \to \eu$.
De klassieke monotone convergentie van $\bigl(1 +
\frac1n\bigr)^n$ valt uit de meetkunde van de tak, zonder
nieuwe ongelijkheid.

\textbf{19.} Als $\eu \leq a < b$: $f$ strikt dalend op
$\intco\eu{+\infty}$ geeft $f(a) > f(b)$, d.w.z.\ $b\ln a > a\ln
b$, d.w.z.\ $a^b > b^a$. Als $1 < a < b \leq \eu$: $f$ strikt
stijgend geeft $f(a) < f(b)$, dus $a^b < b^a$. Gemengd geval:
$2 < \eu < 3$ en $2^3 = 8 < 9 = 3^2$; maar $2 < \eu < 5$ en
$2^5 = 32 > 25 = 5^2$: beide uitkomsten treden op, beslist door welke zijde
van de tak het punt $(a, b)$ valt.

\textbf{20.} De tak ontmoet de diagonaal in $\eu$ en $\varphi$
breidt er continu uit met $\varphi(\eu) = \eu$.
Door $\varphi(\varphi(x)) = x$ af te leiden in $x = \eu$ via de
kettingregel: $\varphi'(\varphi(\eu))\,\varphi'(\eu) =
\varphi'(\eu)^2 = 1$, dus $\varphi'(\eu) = \pm1$; $\varphi$ is
dalend, dus $\varphi'(\eu) = -1$. De tak kruist de
diagonaal met helling $-1$: loodrecht.

\textbf{21.} $y = 3x$ is het geval $t = 3$: $x = 3^{1/2} = \sqrt3$
en $y = 3^{3/2} = 3\sqrt3$. Controle: $f(\sqrt3) =
\frac{0.54931}{1.73205} = 0.3171$ en $f(3\sqrt3) =
\frac{\ln 5.19615}{5.19615} = \frac{1.64792}{5.19615} = 0.3171$:
gelijk tot op vier decimalen, dus $(\sqrt3)^{3\sqrt3} =
(3\sqrt3)^{\sqrt3}$.

\textbf{22.} $n^{1/n} = \eu^{f(n)}$, en $\exp$ is stijgend, dus
de ordening is die van $f(n)$: $f(3) = \frac{\ln3}3 \approx
0.3662$ verslaat $f(2) = f(4) = \frac{\ln2}2 \approx 0.3466$ (vraag
3) en $f(5) \approx 0.3219$. Het grootste is $\sqrt[3]3$, en het
gelijke paar is $\sqrt2 = \sqrt[4]4$ --- het geheeltallige paar $(2, 4)$
in nog een andere vermomming.

\textbf{23.} De oplossingsverzameling is de unie van de diagonaal $\{(x,
x) : x > 0\}$ en een enkele tak, symmetrisch om de diagonaal,
strikt dalend van de asymptoot $x = 1$ (als $y \to
+\infty$) naar de asymptoot $y = 1$ (als $x \to +\infty$), die
de diagonaal precies eenmaal kruist, in $(\eu, \eu)$, met helling $-1$
daar. Op de tak liggen de geheeltallige punten $(2, 4)$ en $(4, 2)$
--- de enige --- en de rationale punten $(x_n, y_n)$,
die monotoon langs de tak marcheren naar $(\eu, \eu)$
zonder het ooit te bereiken ($\eu$ is irrationaal; de rationale
punten van de tak accumuleren in een irrationale hoek).

\textbf{24.} (i) \omterm{prop:b1:functions:powerrules}{Groeivergelijkingen} gaven de limieten van $f$ in
$+\infty$ (vraag 2) en $t\ln t \to 0$ (vraag 10),
wat zowel de variatietabel als de asymptoten vormgaf. (ii) De
tussenwaardestelling, via de bijectie-uitspraken,
zette de variatietabel om in exacte oplossingsaantallen
(vraag 4) en in het bestaan van de inverse \omterm{def:b1:logic:map}{afbeelding} $t(x)$
(vraag 12). (iii) Unieke ontbinding dreef de twee
identificaties in onvereenvoudigbare vorm van vraag 15 aan, het rekenkundige hart
van de classificatie der rationale punten.

\textbf{25.} Eén berekening van de afgeleide --- het teken van $1 - \ln
t$ --- genereerde alles: het oplossingsaantal voor elk niveau
$c$, de vorm en asymptoten van de tak, de extremale
ongelijkheid $\eu^x \geq x^\eu$, de classificatie van geheeltallige en
rationale oplossingen, en de monotone convergentie van $\bigl(1 +
\frac1n\bigr)^n$. Dit is de zuinigheid van het denken met variatie-
tabellen: een eendimensionale studie lost een tweevariabelenvergelijking op
omdat de vergelijking door een enkele functie factoriseert.
\cref{ch:b1:seq} zal de convergentie van $(x_n)$ overdoen met de
algemene theorie van monotone rijen; \cref{ch:b1:derivative}
grondvest variatietabellen streng op de middelwaardestelling; en
\cref{ch:b1:taylor} levert wat de tabel niet kan --- de
\emph{snelheid} van de convergentie $x_n \to \eu$ (ze is van orde
$1/n$).
\end{solution}
