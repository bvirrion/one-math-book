\section*{Hoofdstuk \ref{ch:b1:series} --- Numerieke reeksen}

\begin{solution}{exo:b1:series:1}
$\dfrac{1}{n(n+1)} = \dfrac1n - \dfrac{1}{n+1}$: telescoperend, $S_N =
1 - \frac{1}{N+1} \to 1$. Convergent, som $1$.

$\ln\bigl(1 - \frac{1}{n^2}\bigr) = \ln\frac{(n-1)(n+1)}{n^2} =
\ln\frac{n-1}{n} - \ln\frac{n}{n+1}$: opnieuw telescoperend, $S_N =
\ln\frac12 - \ln\frac{N}{N+1} \to -\ln 2$. Convergent, som $-\ln 2$.

$\dfrac{3^n + 4^n}{5^n} = \bigl(\frac35\bigr)^n +
\bigl(\frac45\bigr)^n$: twee convergente meetkundige reeksen, som
$\frac{1}{1 - 3/5} + \frac{1}{1 - 4/5} = \frac52 + 5 = \frac{15}{2}$.
\end{solution}

\begin{solution}{exo:b1:series:2}
Overal het \omterm{thm:b1:series:ratio}{quotiëntcriterium}.

$\frac{u_{n+1}}{u_n} = \frac{(n+1)^2}{2n^2} \to \frac12 < 1$:
convergent.

$\frac{u_{n+1}}{u_n} = \frac{(n+1)!\,n^n}{n!\,(n+1)^{n+1}} =
\bigl(\frac{n}{n+1}\bigr)^n = \bigl(1 + \frac1n\bigr)^{-n} \to
\frac1\eu < 1$: convergent.

Met de factor $2^n$: quotiënt $\to \frac2\eu < 1$: convergent.

Met $3^n$: quotiënt $\to \frac3\eu > 1$: divergent (termen streven
naar $+\infty$).
\end{solution}

\begin{solution}{exo:b1:series:3}
Alle termen niet-negatief; gebruik equivalenten
(\cref{thm:b1:series:positive}).

$\sin\frac{1}{n^2} \sim \frac{1}{n^2}$: convergent (Riemann $\alpha
= 2$).

$1 - \cos\frac1n \sim \frac{1}{2n^2}$: convergent.

$\frac{1}{\sqrt{n(n+1)}} \sim \frac1n$: divergent.

$\frac{\ln n}{n^2} = \frac{1}{n^{3/2}}\cdot\frac{\ln n}{n^{1/2}}$
en $\frac{\ln n}{\sqrt n} \to 0$
(\cref{prop:b1:functions:powerrules}): dus $\frac{\ln n}{n^2} \leq
\frac{1}{n^{3/2}}$ voor grote $n$: convergent.
\end{solution}

\begin{solution}{exo:b1:series:4}
Voor $n \geq 2$: $\frac{1}{n^2} \leq \frac{1}{n(n-1)} =
\frac{1}{n-1} - \frac1n$. Dus
\[
\sum_{n=1}^{N} \frac{1}{n^2} \leq 1 + \sum_{n=2}^{N}
\Bigl(\frac{1}{n-1} - \frac1n\Bigr) = 1 + 1 - \frac1N < 2 :
\]
partiële sommen stijgend en begrensd door $2$: convergentie
(\cref{thm:b1:series:positive}), som $\leq 2$. (De exacte waarde
$\frac{\pi^2}{6}$ is een feest voor het tweede jaar.)
\end{solution}

\begin{solution}{exo:b1:series:5}
$\sum \frac{(-1)^n}{\sqrt n}$: alternerend met $\frac{1}{\sqrt n}
\downarrow 0$: convergent (\cref{thm:b1:series:alternating}); niet
absoluut ($\alpha = \frac12 \leq 1$).

$\sum \frac{(-1)^n}{n + (-1)^n}$: de rij $\frac{1}{n +
(-1)^n}$ is \emph{niet} dalend ($\frac{1}{n+1}$ dan
$\frac{1}{n}$ wisselen ongunstig af), dus het criterium is niet
rechtstreeks toepasbaar. Ontwikkel:
\[
\frac{(-1)^n}{n + (-1)^n}
= \frac{(-1)^n}{n}\cdot\frac{1}{1 + \frac{(-1)^n}{n}}
= \frac{(-1)^n}{n} - \frac{1}{n^2} +
O\Bigl(\frac{1}{n^3}\Bigr):
\]
de eerste \omterm{def:b1:series:def}{reeks} convergeert (alternerend), $\sum \frac{1}{n^2}$
convergeert, de $O(n^{-3})$ convergeert absoluut: de som van drie
convergente reeksen convergeert.

$\sin\bigl(\pi\sqrt{n^2+1}\bigr)$: schrijf $\sqrt{n^2 + 1} = n +
\frac{1}{2n} + \varepsilon_n$ met $\varepsilon_n = O(n^{-3})$;
dan, door de $\pi$-periodiciteit van $\sin$ op het teken na,
\[
\sin\bigl(\pi\sqrt{n^2+1}\bigr)
= (-1)^n \sin\Bigl(\frac{\pi}{2n} + \pi\varepsilon_n\Bigr) .
\]
Stel $\theta_n = \frac{\pi}{2n} + \pi\varepsilon_n$ en $a_n =
\sin\theta_n$. Voor grote $n$, $\theta_n \in \intoo{0}{\frac\pi2}$
en
\[
\theta_n - \theta_{n+1} = \frac{\pi}{2n(n+1)} +
\pi(\varepsilon_n - \varepsilon_{n+1})
= \frac{\pi}{2n^2} + O\Bigl(\frac{1}{n^3}\Bigr) > 0
\]
uiteindelijk, dus $(\theta_n)$ daalt naar $0$; aangezien $\sin$
stijgend is op $\intcc{0}{\frac\pi2}$, daalt $(a_n)$ ook naar $0$. Het
criterium voor alternerende reeksen is toepasbaar: convergent --- niet
absoluut, aangezien $a_n \sim \frac{\pi}{2n}$.
\end{solution}

\begin{solution}{exo:b1:series:6}
$f(t) = \frac{1}{t(\ln t)^\alpha}$ is positief, \omterm{def:b1:continuity:continuous}{continu},
dalend op $\intco{2}{+\infty}$. Met de substitutie $u = \ln t$:
\[
\int_2^x \frac{\dd t}{t(\ln t)^\alpha}
= \int_{\ln 2}^{\ln x} \frac{\dd u}{u^\alpha},
\]
begrensd als $x \to \infty$ dan en slechts dan als $\alpha > 1$
(de berekening van \cref{thm:b1:series:riemann}). Via
integraalvergelijking: convergentie dan en slechts dan als $\alpha >
1$. (Deze reeksen \emph{van het Bertrand-type} tonen hoe fijn de grens
van convergentie is: $n\ln n$ divergeert, $n(\ln n)^{1.01}$
convergeert.)
\end{solution}

\begin{solution}{exo:b1:series:7}
Via de raaklijngrenzen van \cref{exo:b1:derivative:3} herschreven
met ontwikkelingen: voor $x = \frac1n \in \intoc{0}{1}$ geeft
Taylor--Lagrange voor $\ln(1+x)$ in orde $1$ dat $\ln(1 + x) = x -
\frac{x^2}{2(1 + c)^2}$ voor een zekere $c \in \intoo{0}{x}$, dus
\[
0 \leq u_n = \frac1n - \ln\Bigl(1 + \frac1n\Bigr) \leq
\frac{1}{2n^2} .
\]
Vergelijking met de \omterm{thm:b1:series:riemann}{Riemann-reeks}: $\sum u_n$ convergeert. Haar
partiële som laat de logaritmen telescoperen:
\[
\sum_{n=1}^{N} u_n = H_N - \ln(N+1)
\]
(aangezien $\sum_{n\leq N} \ln\frac{n+1}{n} = \ln(N+1)$). Dus $H_N -
\ln(N+1)$ convergeert; door $\ln\frac{N+1}{N} \to 0$ toe te voegen,
convergeert de rij $H_N - \ln N$. Haar limiet is $\gamma \approx
0.5772$.
\end{solution}

\begin{solution}{exo:b1:series:8}
$\dfrac{1}{n(n+2)} = \dfrac{1/2}{n} - \dfrac{1/2}{n+2}$: de partiële
som telescopeert met een verschuiving van $2$,
\[
S_N = \frac12\Bigl(1 + \frac12 - \frac{1}{N+1} -
\frac{1}{N+2}\Bigr) \longrightarrow \frac34 .
\]

$\sum \frac{n}{2^n}$: voor $\abs x < 1$, door de eindige
meetkundige som af te leiden en over te gaan naar de limiet (alle
reeksen hier convergeren absoluut, \omterm{thm:b1:series:ratio}{quotiëntcriterium}): uit
$\sum_{n\geq0} x^n = \frac{1}{1-x}$ verkrijgt men door directe
berekening met partiële sommen
\[
\sum_{n=1}^{N} n x^{n-1}
= \frac{1 - (N+1)x^N + N x^{N+1}}{(1 - x)^2}
\xrightarrow[N\to\infty]{} \frac{1}{(1-x)^2}
\quad (\abs x < 1),
\]
(de randtermen $N x^N \to 0$). Bij $x = \frac12$:
$\sum_{n\geq1} n\bigl(\frac12\bigr)^{n-1} = 4$, dus $\sum_{n\geq0}
\frac{n}{2^n} = \frac12 \times 4 = 2$.
\end{solution}

\begin{solution}{exo:b1:series:9}
Groepeer de termen van $\sum u_n$ in pakketten tussen machten van $2$.
Bovenste pakketten: voor $2^k \leq n < 2^{k+1}$ zijn er $2^k$ termen,
elk $\leq u_{2^k}$:
\[
\sum_{n=1}^{2^{K+1}-1} u_n
= \sum_{k=0}^{K} \sum_{n=2^k}^{2^{k+1}-1} u_n
\leq \sum_{k=0}^{K} 2^k u_{2^k} .
\]
Onderste pakketten: elke term van hetzelfde pakket is $\geq
u_{2^{k+1}}$, dus $\sum_{n=2^k}^{2^{k+1}-1} u_n \geq 2^k u_{2^{k+1}} =
\frac12 \cdot 2^{k+1} u_{2^{k+1}}$, waaruit
\[
\sum_{n=1}^{2^{K+1}-1} u_n \geq \frac12 \sum_{k=1}^{K+1} 2^{k}
u_{2^{k}} .
\]
Beide vergelijkingen van partiële sommen gelden in beide richtingen
(niet-negatieve termen, \cref{thm:b1:series:positive}): de twee
reeksen hebben dezelfde aard.

Riemann: $u_n = n^{-\alpha}$ geeft $2^k u_{2^k} =
2^{k(1-\alpha)}$, een \omterm{ex:b1:series:geometric}{meetkundige reeks}, convergent dan en slechts
dan als $2^{1 - \alpha} < 1$ dan en slechts dan als $\alpha > 1$.
Bertrand (\cref{exo:b1:series:6}): $u_n = \frac{1}{n(\ln n)^\alpha}$
geeft $2^k u_{2^k} = \frac{1}{(k\ln 2)^\alpha}$, een \omterm{thm:b1:series:riemann}{Riemann-reeks} in
$k$: convergent dan en slechts dan als $\alpha > 1$.
\end{solution}

\begin{solution}{exo:b1:series:10}
Eindige meetkundige identiteit met quotiënt $-t^2$:
\[
\frac{1}{1 + t^2} = \sum_{k=0}^{n-1} (-1)^k t^{2k}
+ \frac{(-1)^n t^{2n}}{1 + t^2} .
\]
Integreer over $\intcc{0}{1}$ (de linkerzijde integreert tot $\arctan
1 = \frac\pi4$, \cref{prop:b1:functions:arcderiv}):
\[
\frac{\pi}{4} = \sum_{k=0}^{n-1} \frac{(-1)^k}{2k+1}
+ (-1)^n \int_0^1 \frac{t^{2n}}{1+t^2}\,\dd t,
\qquad
0 \leq \int_0^1 \frac{t^{2n}}{1+t^2}\,\dd t \leq \int_0^1 t^{2n}\dd
t = \frac{1}{2n+1} .
\]
Door $n \to \infty$ te laten gaan is de formule bewezen, en de
getoonde grens op de \omterm{thm:b1:integration:def}{integraal} is precies de restgrens: na sommatie
tot $N$ (d.w.z.\ $n = N + 1$ termen), $\abs{R_N} \leq \frac{1}{2N +
3}$.
\end{solution}

\begin{solution}{exo:b1:series:11}
$n^{1/n} = \eu^{\frac{\ln n}{n}} \to \eu^0 = 1$
(\cref{prop:b1:functions:powerrules}). Dus
\[
\frac{1}{n^{1 + 1/n}} = \frac{1}{n}\,\eu^{-\frac{\ln n}{n}}
\sim \frac{1}{n} ,
\]
en het equivalentiecriterium (\cref{thm:b1:series:positive})
vergelijkt met de divergente harmonische \omterm{def:b1:series:def}{reeks}: \emph{divergent},
hoewel elke exponent $1 + \frac1n$ groter is dan $1$. Het
Riemann-criterium betreft een \emph{vaste} exponent $\alpha$; een
exponent die naar $1$ afglijdt kan al zijn marge verliezen, zoals
hier.
\end{solution}

\begin{solution}{exo:b1:series:12}
Zij $\varepsilon > 0$. Volgens het Cauchy-criterium voor de
convergente \omterm{def:b1:series:def}{reeks} (\cref{thm:b1:seq:complete} toegepast op de
partiële sommen) bestaat er een $N$ met $\sum_{k=n+1}^{2n} u_k \leq
\varepsilon$ voor $n \geq N$. Door de monotonie is elk van deze $n$
termen $\geq u_{2n}$:
\[
n\,u_{2n} \leq \sum_{k=n+1}^{2n} u_k \leq \varepsilon
\quad\Longrightarrow\quad 2n\,u_{2n} \leq 2\varepsilon ,
\]
en voor oneven indices $(2n+1)\,u_{2n+1} \leq (2n+1)\,u_{2n} \leq
2\bigl(2n\,u_{2n}\bigr) \leq 4\varepsilon$ voor $n \geq N$: in
beide pariteiten, $n u_n \to 0$.

Omkering onwaar: $u_n = \frac{1}{n\ln n}$ heeft $n u_n =
\frac{1}{\ln n} \to 0$, en toch divergeert de \omterm{def:b1:series:def}{reeks}
(\cref{exo:b1:series:6}, $\alpha = 1$). Monotonie noodzakelijk:
zij $u_n = \frac1n$ wanneer $n$ een volkomen kwadraat is en $u_n =
2^{-n}$ anders: de \omterm{def:b1:series:def}{reeks} convergeert (de kwadraattermen sommeren
als $\sum \frac{1}{k^2}$, de rest meetkundig), maar $n u_n =
1$ langs de kwadraten.
\end{solution}

\begin{solution}{pb:b1:series:1}
\textbf{1.} $a_{n+1} - a_n = \frac{1}{n+1} - \ln\frac{n+1}{n}
\leq 0$ omdat $\ln(1 + \frac1n) \geq \frac{1/n}{1 + 1/n} =
\frac{1}{n+1}$; en $b_{n+1} - b_n = \frac{1}{n+1} -
\ln\frac{n+2}{n+1} \geq 0$ omdat $\ln(1 + \frac{1}{n+1}) \leq
\frac{1}{n+1}$. Hun verschil $a_n - b_n = \ln(1 + \frac1n) \to 0$:
ingesloten (\cref{thm:b1:seq:adjacent}), met gemeenschappelijke limiet
$\lim a_n = \gamma$ (\cref{exo:b1:series:7}), en $b_n \leq \gamma
\leq a_n$.

\textbf{2.} $b_{10} = 2.928968 - \ln 11 = 0.5311$ en $a_{10} =
2.928968 - \ln 10 = 0.6264$: dus $\gamma \in
\intcc{0.5311}{0.6264}$. Het verschil is $\ln 1.1 \approx 0.095$ en
krimpt als $\frac1n$: vier decimalen ($\text{verschil} \leq
10^{-4}$) zouden $n \approx 10^4$ vereisen --- de inkaderingen zijn
correct maar traag.

\textbf{3.} Telescoperend $a_n - a_{m+1} = \sum_{k=n}^{m} w_k$ en
$m \to \infty$ latend gaan: $a_n - \gamma = \sum_{k\geq n} w_k$
(limiet van partiële sommen). Bovendien
\[
w_k = \int_k^{k+1} \frac{\dd t}{t} - \frac{1}{k+1}
= \int_k^{k+1} \Bigl(\frac1t - \frac{1}{k+1}\Bigr)\dd t
= \int_k^{k+1} \frac{k + 1 - t}{t\,(k+1)}\,\dd t .
\]
Op $\intcc{k}{k+1}$: $\frac{1}{(k+1)^2} \leq \frac{1}{t(k+1)}
\leq \frac{1}{k(k+1)}$, en $\int_k^{k+1}(k + 1 - t)\dd t =
\frac12$: vandaar $\frac{1}{2(k+1)^2} \leq w_k \leq
\frac{1}{2k(k+1)}$.

\textbf{4.} Boven: $\sum_{k \geq n} \frac{1}{2k(k+1)} =
\frac12\sum_{k\geq n}\bigl(\frac1k - \frac{1}{k+1}\bigr) =
\frac{1}{2n}$ (telescoperend). Onder: $\frac{1}{2(k+1)^2} \geq
\frac{1}{2(k+1)(k+2)}$, waarvan de som telescopeert tot
$\frac{1}{2(n+1)}$. Met vraag 3:
\[
\frac{1}{2(n+1)} \leq H_n - \ln n - \gamma \leq \frac{1}{2n} .
\]

\textbf{5.} Trek $\frac{1}{2n}$ af: $\gamma_n - \gamma \in
\intcc{\frac{1}{2(n+1)} - \frac{1}{2n}}{0} =
\intcc{-\frac{1}{2n(n+1)}}{0}$: de gecorrigeerde schatting is exact
tot op $O\bigl(\frac{1}{n^2}\bigr)$, en steeds van onderaf.

\textbf{6.} $\gamma_{100} = 5.1873775 - \ln 100 - 0.005 =
0.5772073$, met $0 \leq \gamma - \gamma_{100} \leq
\frac{1}{20200} < 5\cdot10^{-5}$: vandaar $0.577207 \leq \gamma
\leq 0.577257$, d.w.z.\ $\gamma = 0.5772 \pm 5\cdot10^{-5}$ (ware
waarde $0.5772156\dots$) --- vier gecertificeerde decimalen uit een
honderd termen, tegenover tienduizend voor vraag 2.

\textbf{7.} Eerste dividend:
\[
H_{2m} - H_m = \Bigl(\ln 2m + \gamma + \frac{1}{4m}\Bigr)
- \Bigl(\ln m + \gamma + \frac{1}{2m}\Bigr) +
O\Bigl(\frac{1}{m^2}\Bigr)
= \ln 2 - \frac{1}{4m} + O\Bigl(\frac{1}{m^2}\Bigr).
\]
Tweede: de even omgekeerden tot $2m$ sommeren tot $\frac12 H_m$, dus
$\sum_{j=1}^{m}\frac{1}{2j-1} = H_{2m} - \frac12 H_m = \frac12
\ln m + \ln 2 + \frac\gamma2 + o(1)$, en $\sum_{j=1}^m
\frac{1}{2j} = \frac12\ln m + \frac\gamma2 + o(1)$.

\textbf{8.} Twee keer de even termen afsplitsen:
$\sum_{k=1}^{2m}\frac{(-1)^{k-1}}{k} = H_{2m} - 2\cdot\frac12
H_m = H_{2m} - H_m$. Volgens vraag 7 is dit gelijk aan $\ln 2 -
\frac{1}{4m} + O(m^{-2})$: de som is $\ln 2$
(\cref{ex:b1:series:ln2} opnieuw) \emph{met} zijn snelheid.

\textbf{9.} Voor $N = 2m$: $S - S_N = \frac{1}{4m} + O(m^{-2}) =
\frac{1}{2N} + O(N^{-2})$. Voor $N = 2m + 1$: $S_{N} = S_{2m} +
\frac{1}{2m+1}$, dus
\[
S - S_N = \Bigl(\frac{1}{4m} - \frac{1}{2m+1}\Bigr) +
O\Bigl(\frac{1}{m^2}\Bigr)
= -\frac{1}{4m} + O\Bigl(\frac{1}{m^2}\Bigr)
= -\frac{1}{2N} + O\Bigl(\frac{1}{N^2}\Bigr).
\]
In beide gevallen $S - S_N \sim \frac{(-1)^N}{2N}$: de helft van de
slechtst-mogelijke grens $a_{N+1}$, met een bekend, alternerend teken.

\textbf{10.} Middeling doodt de oscillerende leidende term:
\[
S - \tilde S_N = \frac{(S - S_N) + (S - S_{N+1})}{2}
= \frac{(-1)^N}{2}\Bigl(\frac{1}{2N} - \frac{1}{2(N+1)}\Bigr)
+ O\Bigl(\frac{1}{N^2}\Bigr) = O\Bigl(\frac{1}{N^2}\Bigr).
\]
Numeriek: $S_{10} = 0.645635$, $S_{11} = 0.736544$, $\tilde
S_{10} = 0.691089$, en $\ln 2 = 0.693147$: de fout daalt van
$4.8\cdot10^{-2}$ naar $2.1\cdot10^{-3}$ --- één optelling, twintig
keer beter.

\textbf{11.} De alternerende fout wisselt bij elke stap van teken,
dus opeenvolgende partiële sommen omspannen de limiet en hun
gemiddelde doet de eersteordeterm wegvallen; de inkaderingsfout $H_n -
\ln n - \gamma \approx \frac{1}{2n}$ heeft een constant teken, dus
geen enkele middeling langs $n$ kan haar wegwerken. Het middelen van
de twee \emph{inkaderingen} helpt wél: $\frac{a_n + b_n}{2} =
H_n - \ln\sqrt{n(n+1)}$, en aangezien
$\ln\sqrt{n(n+1)} = \ln\bigl(n + \frac12\bigr) + O(n^{-2})$,
\[
H_n - \ln\Bigl(n + \frac12\Bigr)
= \Bigl(H_n - \ln n - \frac{1}{2n}\Bigr) + \frac{1}{8n^2} +
O\Bigl(\frac{1}{n^3}\Bigr) = \gamma + O\Bigl(\frac{1}{n^2}\Bigr)
\]
(vraag 5 en $\ln(1 + \frac{1}{2n}) = \frac{1}{2n} -
\frac{1}{8n^2} + O(n^{-3})$). Controle bij $n = 10$: $H_{10} -
\ln 10.5 = 0.57759$, reeds binnen $4\cdot10^{-4}$ van
$\gamma$.

\textbf{12.} $\sum_{j\leq m} \frac{1}{2j-1} \geq \sum_{j \leq m}
\frac{1}{2j} = \frac12 H_m \to +\infty$: zowel het positieve als
het negatieve deel van de alternerende harmonische \omterm{def:b1:series:def}{reeks} divergeren.

\textbf{13.} Schrijf $u_n^\pm = \frac{\abs{u_n} \pm u_n}{2} \geq
0$, dus $u_n = u_n^+ - u_n^-$ en $\abs{u_n} = u_n^+ + u_n^-$. Als
$\sum u_n^+$ zou convergeren, dan zou $\sum u_n^- = \sum (u_n^+ - u_n)$
convergeren (verschil van convergente reeksen), en dus $\sum
\abs{u_n}$ ook: tegenspraak met de voorwaardelijke convergentie. Door
symmetrie divergeren beide $\sum u_n^\pm$ (naar $+\infty$): een
oneindig reservoir van positieve en van negatieve massa.

\textbf{14.} Zij $S = \sum u_n$, $\varepsilon > 0$, en $N$ met
$\sum_{n > N} \abs{u_n} \leq \varepsilon$ (Cauchy-criterium voor
$\sum\abs{u_n}$). Zij $M_0$ groot genoeg opdat $\sigma(\{0,
\dots, M_0\}) \supseteq \{0, \dots, N\}$. Voor $M \geq M_0$ is de
verschil $\sum_{m \leq M} u_{\sigma(m)} - \sum_{n \leq N} u_n$
een eindige som van \emph{verschillende} termen $u_n$ met $n > N$,
dus van absolute waarde $\leq \varepsilon$; en $\abs{S -
\sum_{n\leq N} u_n} \leq \varepsilon$ eveneens. Dus de herschikte
partiële sommen liggen uiteindelijk binnen $2\varepsilon$ van $S$:
$\sum u_{\sigma(n)} = S$. \omterm{thm:b1:series:absolute}{Absolute convergentie} is bestand tegen
herschikking.

\textbf{15.} Elke fase van de gulzige procedure eindigt na
eindig veel termen, omdat de resterende positieve
(respectievelijk negatieve) termen op zich divergerende partiële
sommen hebben (vraag 12): de lopende som moet uiteindelijk $t$
overschrijden. De procedure wisselt daarom oneindig veel eindige
fasen af, waarbij de positieve termen op volgorde en de negatieve
termen op volgorde verbruikt worden: elke term wordt precies één keer
gebruikt --- een herschikking. Na de eerste overschrijding bewegen de
partiële sommen tussen twee opeenvolgende overschrijdingen monotoon
naar $t$, en bij een overschrijding schieten ze hoogstens de zojuist
toegevoegde term door; aangezien de termen die bij de $j$-de
overschrijding gebruikt worden een index van minstens $j$ in hun
klasse hebben, streven deze overschrijdingen naar $0$. Vandaar
convergeren de partiële sommen naar $t$: elk reëel getal is de som van
een of andere herschikking.

\textbf{16.} De positieve plaatsen ontvangen $\frac{1}{2j-1}$ voor
$j = 1, 2, \dots$ op volgorde, de negatieve plaatsen $\frac{1}{2j}$
op volgorde: elke term van de alternerende harmonische \omterm{def:b1:series:def}{reeks}
verschijnt precies één keer. Voor $(p, q) = (1, 2)$ zijn de blokken
$\bigl(1, -\frac12, -\frac14\bigr)$, $\bigl(\frac13, -\frac16,
-\frac18\bigr)$, $\bigl(\frac15, -\frac1{10},
-\frac1{12}\bigr)$, \dots\ --- de getoonde \omterm{def:b1:series:def}{reeks}.

\textbf{17.} Aangezien $\frac{1}{4k-2} = \frac12\cdot\frac{1}{2k-1}$:
\[
\frac{1}{2k-1} - \frac{1}{4k-2} - \frac{1}{4k}
= \frac12\,\frac{1}{2k-1} - \frac12\,\frac{1}{2k}
= \frac12\Bigl(\frac{1}{2k-1} - \frac{1}{2k}\Bigr).
\]
Sommeren over $k = 1, \dots, K$: $T_{3K} = \frac12
\sum_{k=1}^{K}\bigl(\frac{1}{2k-1} - \frac{1}{2k}\bigr) =
\frac12 S_{2K}$: bij elke derde partiële som is de herschikte
\omterm{def:b1:series:def}{reeks} \emph{exact} de helft van de oorspronkelijke.

\textbf{18.} Na $K$ volledige blokken is de herschikte partiële
som $\sum_{j=1}^{pK}\frac{1}{2j-1} -
\sum_{j=1}^{qK}\frac{1}{2j}$, en vraag 7 evalueert ze:
\[
\Bigl(\frac{\ln(pK)}{2} + \ln 2 + \frac\gamma2\Bigr)
- \Bigl(\frac{\ln(qK)}{2} + \frac\gamma2\Bigr) + o(1)
= \ln 2 + \frac12\ln\frac pq + o(1) :
\]
de $\gamma$'s vallen weg, de $\ln K$'s vallen weg, de verhouding
$\frac pq$ overleeft.

\textbf{19.} Een partiële som binnen blok $K + 1$ verschilt van de
$K$-bloksom met hoogstens $p + q$ termen, elk van absolute waarde
ongeveer $\leq \frac{1}{2qK}$, en dus met $O\bigl(\frac1K\bigr) \to 0$:
de volledige rij partiële sommen heeft dezelfde limiet $\ln 2 +
\frac12\ln\frac pq$. Voor $(1, 2)$: $\ln 2 + \frac12\ln\frac12 =
\frac{\ln 2}{2} = 0.34657\dots$, en inderdaad kruipt $T_9 = 0.30833$
ernaartoe: volgens vraag 17 convergeert $T_{3K} = \frac12 S_{2K}$
met exact de helft van de alternerend-harmonische fout.
Dezelfde termen, de halve som.

\textbf{20.} $(1,1)$: $\ln 2 + \frac12\ln 1 = \ln 2$ --- de
oorspronkelijke volgorde, consistentie. $(2,1)$: $\frac32\ln 2 \approx
1.0397$. Het $(p,q)$-menu bereikt precies de aftelbare dichte
familie $\ln 2 + \frac12\ln r$, $r \in \Q_{>0}$; het gulzige recept
van Riemann (vraag 15) bereikt \emph{elk} reëel getal. Structuur koopt
formules; gulzigheid koopt totaliteit.

\textbf{21.} De partiële sommen telescoperen: $\sum_{k=1}^{N}
\bigl(\frac1k - \ln\frac{k+1}{k}\bigr) = H_N - \ln(N+1) = b_N
\to \gamma$: de \omterm{def:b1:series:def}{reeks} van \cref{exo:b1:series:7} sommeert exact
tot de \omterm{pb:b1:series:1}{constante van Euler}.

\textbf{22.} Gulzig voor $t = 1$: de eerste positieve term brengt
de som exact tot $1$, niet erboven, dus een tweede positieve wordt
genomen om te overschrijden: $1, \frac13$ (som $1.3333 > 1$), dan
$-\frac12$ ($0.8333$), $\frac15$ ($1.0333$), $-\frac14$
($0.7833$), $\frac17, \frac19$ ($1.0373$), $-\frac16$
($0.8706$), $\frac1{11}, \frac1{13}$ ($1.0384$), $-\frac18$
($0.9134$), $\frac1{15}$ ($0.9801$), \dots\ --- de sommen ademen
rond $1$ met steeds kleinere amplitude, nu twee positieven per cyclus
nodig aangezien de negatieven groter zijn.

\textbf{23.} Bouw blokken: in fase $j$, voeg genoeg ongebruikte
positieve termen toe om de partiële som met minstens $1$ te verhogen
(mogelijk: de resterende positieve termen hebben divergerende sommen,
vraag 12), voeg dan de enkele negatieve term $-\frac{1}
{2j}$ toe. Elke positieve term wordt uiteindelijk gebruikt (elke
fase gebruikt er minstens één), elke negatieve ook (één per fase): een
herschikking. Elke fase verandert de som met $\geq 1 -
\frac{1}{2j} \geq \frac12$: de partiële sommen overschrijden
$\frac{j}{2}$ na fase $j$ en de aangroeiingen binnen een fase
zijn positief behalve de laatste, begrensd door $\frac{1}{2j} \to 0$:
divergentie naar $+\infty$.

\textbf{24.} Volgens de wet, $H_N \geq 20 \iff \ln N \geq 20 -
\gamma - \frac{1}{2N} + O(N^{-2})$: de drempel $N^*$
voldoet aan $\ln N^* = 20 - \gamma + o(1)$, d.w.z.\ $N^* =
\eu^{20 - \gamma}(1 + o(1)) \approx \eu^{19.4228} \approx
2.72\cdot10^{8}$ --- binnen het ruwe venster
$\intcc{1.8\cdot10^8}{4.9\cdot10^8}$ van
\cref{ex:b1:series:harmonicstack}, en vastgepind door $\gamma$.

\textbf{25.} (i) In $H_n = \ln n + \gamma + \frac{1}{2n} +
O(n^{-2})$: de $\ln n$ is de \omterm{thm:b1:integration:def}{integraal}, $\gamma$ de prijs voor
het vervangen van een som door een \omterm{thm:b1:integration:def}{integraal} (een werkelijk nieuwe
constante van de analyse), en $\frac{1}{2n}$ de eerste correctie ---
de schaduw van het trapezium. (ii) Voorwaardelijke convergentie leunt
op het wegvallen tussen twee oneindige reservoirs (vraag 13), dus
herordenen herweegt de reservoirs; \omterm{thm:b1:series:absolute}{absolute convergentie} heeft een
eindige totale massa, en de staartschatting van vraag 14 is
volgorde-blind. (iii) De $(p,q)$-sommen werden \emph{berekend}: de
$\gamma$-wet veranderde elke herschikte partiële som in $\ln 2 +
\frac12\ln\frac pq + o(1)$, waarbij $\gamma$ zelf wegvalt ---
een asymptotische boekhoudoefening, geen abstract argument.
(iv) Vervolgens: producten en onvoorwaardelijke sommeerbaarheid voor
machtreeksen in het boekdeel van het tweede jaar; en de weekendopgave
van het boekdeel van het derde jaar over de formule van Stirling,
waar de boekhouding som-tegen-integraal, één orde verder doorgetrokken,
$\sqrt{2\pi}$ zelf voortbrengt.
\end{solution}
