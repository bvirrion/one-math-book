\section*{Hoofdstuk \ref{ch:b1:complex} --- Complexe getallen}

\begin{solution}{exo:b1:complex:1}
$1 + \iu = \sqrt 2\, \eu^{\iu\pi/4}$;
$\;\sqrt 3 - \iu = 2\, \eu^{-\iu\pi/6}$;
$\;-5 = 5\, \eu^{\iu\pi}$;
\[
\frac{1 + \iu}{\sqrt 3 - \iu}
= \frac{\sqrt 2}{2}\, \eu^{\iu(\pi/4 + \pi/6)}
= \frac{\sqrt 2}{2}\, \eu^{5\iu\pi/12}.
\]
Hetzelfde quotiënt algebraïsch berekenen (vermenigvuldig met de \omterm{def:b1:complex:field}{toegevoegde}):
\[
\frac{(1 + \iu)(\sqrt 3 + \iu)}{4}
= \frac{(\sqrt 3 - 1) + \iu(\sqrt 3 + 1)}{4}.
\]
Identificeren met $\frac{\sqrt 2}{2}(\cos\frac{5\pi}{12} +
\iu\sin\frac{5\pi}{12})$:
\[
\cos\frac{5\pi}{12} = \frac{\sqrt 6 - \sqrt 2}{4},
\qquad
\sin\frac{5\pi}{12} = \frac{\sqrt 6 + \sqrt 2}{4}.
\]
\end{solution}

\begin{solution}{exo:b1:complex:2}
$(1+\iu)^2 = 2\iu$, dus $(1+\iu)^{20} = (2\iu)^{10} = 2^{10}\,\iu^{10}
= 1024 \times (-1) = -1024$.

In exponentiële vorm $(1+\iu)^n = 2^{n/2}\, \eu^{\iu n\pi/4}$, reëel dan
en slechts dan als $\sin\frac{n\pi}{4} = 0$, d.w.z.\ $4 \mid n$. Dus $(1+\iu)^n
\in \R$ precies voor de veelvouden van $4$ (waarde $(-4)^{n/4}$).
\end{solution}

\begin{solution}{exo:b1:complex:3}
$\abs{z - 2} = \abs{z + \iu}$: gelijke afstand tot de punten $2$ en
$-\iu$ --- de middelloodlijn van het lijnstuk dat $(2, 0)$
en $(0, -1)$ verbindt.

$\abs{z - 1} = 2$: de cirkel met middelpunt $1$ en straal $2$.

$\frac{z-1}{z+1} \in \iu\R$: volgens
\cref{prop:b1:complex:geometry}~(3) voldoen de punten $M(z)$, $A(1)$,
$B(-1)$ aan: rechten $MA$ en $MB$ loodrecht (of $z = 1$, waar
het quotiënt $0 \in \iu\R$ is). De \omterm{def:b1:logic:sets}{verzameling} is de cirkel met middellijn
$[-1, 1]$ (de eenheidscirkel), min het punt $-1$ waar het quotiënt
ongedefinieerd is. \emph{Controle door berekening:} $z = \eu^{\iu\theta}$ geeft
$\frac{z-1}{z+1} = \frac{\eu^{\iu\theta/2}(\eu^{\iu\theta/2} -
\eu^{-\iu\theta/2})}{\eu^{\iu\theta/2}(\eu^{\iu\theta/2} +
\eu^{-\iu\theta/2})} = \iu\tan\frac\theta2 \in \iu\R$.
\end{solution}

\begin{solution}{exo:b1:complex:4}
Linearisatie:
\[
\sin^4\theta
= \Bigl(\frac{\eu^{\iu\theta} - \eu^{-\iu\theta}}{2\iu}\Bigr)^{\!4}
= \frac{\eu^{4\iu\theta} - 4\eu^{2\iu\theta} + 6 - 4\eu^{-2\iu\theta}
+ \eu^{-4\iu\theta}}{16}
= \frac{\cos 4\theta - 4\cos 2\theta + 3}{8}.
\]
Ontwikkeling: door de Moivre, $\cos 4\theta = \Re\bigl((c +
\iu s)^4\bigr) = c^4 - 6c^2 s^2 + s^4$ met $c = \cos\theta$, $s =
\sin\theta$; door $s^2 = 1 - c^2$ te substitueren:
\[
\cos 4\theta = c^4 - 6c^2(1 - c^2) + (1 - c^2)^2
= 8c^4 - 8c^2 + 1 .
\]
\end{solution}

\begin{solution}{exo:b1:complex:5}
$8\iu = 8\,\eu^{\iu\pi/2}$, dus de derdemachtswortels zijn $2\,\eu^{\iu(\pi/6 +
2k\pi/3)}$, $k = 0, 1, 2$:
\[
z_0 = 2\eu^{\iu\pi/6} = \sqrt 3 + \iu,\quad
z_1 = 2\eu^{5\iu\pi/6} = -\sqrt 3 + \iu,\quad
z_2 = 2\eu^{3\iu\pi/2} = -2\iu :
\]
een gelijkzijdige driehoek op de cirkel met straal $2$.

$-4 = 4\eu^{\iu\pi}$, dus $z^4 = -4$ heeft oplossingen $\sqrt 2\,
\eu^{\iu(\pi/4 + k\pi/2)}$: $\;1 + \iu$, $-1 + \iu$, $-1 - \iu$,
$1 - \iu$. \omterm{def:b1:complex:field}{Toegevoegde} wortels paren:
\[
X^4 + 4 = \bigl(X^2 - 2X + 2\bigr)\bigl(X^2 + 2X + 2\bigr),
\]
aangezien $(X - (1+\iu))(X - (1-\iu)) = X^2 - 2X + 2$ en analoog voor
het andere paar. (Ontwikkel ter controle.)
\end{solution}

\begin{solution}{exo:b1:complex:6}
$C_n + \iu S_n = \sum_{k=0}^{n} \eu^{\iu k\theta} =
\frac{\eu^{\iu(n+1)\theta} - 1}{\eu^{\iu\theta} - 1}$ (meetkundige som,
reden $\eu^{\iu\theta} \neq 1$). Halvehoekontbinding
(\cref{met:b1:complex:trig}~(4)) op teller en noemer:
\[
\frac{2\iu\sin\frac{(n+1)\theta}{2}\, \eu^{\iu(n+1)\theta/2}}
{2\iu\sin\frac{\theta}{2}\, \eu^{\iu\theta/2}}
= \frac{\sin\frac{(n+1)\theta}{2}}{\sin\frac{\theta}{2}}\,
\eu^{\iu n\theta/2}.
\]
Reële en imaginaire delen nemen:
\[
C_n = \frac{\sin\frac{(n+1)\theta}{2}}{\sin\frac{\theta}{2}}
\cos\frac{n\theta}{2},
\qquad
S_n = \frac{\sin\frac{(n+1)\theta}{2}}{\sin\frac{\theta}{2}}
\sin\frac{n\theta}{2}.
\]
\end{solution}

\begin{solution}{exo:b1:complex:7}
Discriminant: $\Delta = (3 + 4\iu)^2 - 4(-1 + 5\iu) = 9 + 24\iu - 16 +
4 - 20\iu = -3 + 4\iu$. Vierkantswortels van $-3 + 4\iu$: los $x^2 - y^2
= -3$, $2xy = 4$, $x^2 + y^2 = 5$ op; dan $x^2 = 1$, $y = 2/x$, dezelfde
tekens: $\delta = \pm(1 + 2\iu)$. Bijgevolg
\[
z = \frac{(3 + 4\iu) \pm (1 + 2\iu)}{2}
\in \{\, 2 + 3\iu,\; 1 + \iu \,\}.
\]
\emph{Controle:} som $= 3 + 4\iu$ en product $(2+3\iu)(1+\iu) = -1 +
5\iu$, zoals vereist door de coëfficiënten.
\end{solution}

\begin{solution}{exo:b1:complex:8}
\begin{enumerate}
  \item \cref{prop:b1:complex:sumroots} met $n = 5$.
  \item $u + v = \omega + \omega^2 + \omega^3 + \omega^4 = -1$ door (1).
        Voor het product, ontwikkel en reduceer de exponenten modulo $5$:
        \[
        uv = (\omega + \omega^4)(\omega^2 + \omega^3)
        = \omega^3 + \omega^4 + \omega^6 + \omega^7
        = \omega^3 + \omega^4 + \omega + \omega^2 = -1 .
        \]
        Dus $u$ en $v$ hebben som $-1$ en product $-1$: het zijn de
        twee wortels van $X^2 + X - 1$.
  \item $u = \omega + \conj\omega = 2\cos\frac{2\pi}{5} > 0$ (de
        hoek is scherp), en de positieve wortel van $X^2 + X - 1$ is
        $\frac{-1 + \sqrt 5}{2}$. Bijgevolg $\cos\frac{2\pi}{5} =
        \frac{\sqrt 5 - 1}{4}$.
\end{enumerate}
\end{solution}

\begin{solution}{exo:b1:complex:9}
Ontwikkel beide kwadraten zoals in het bewijs van
\cref{prop:b1:complex:rules}~(4):
\[
\abs{z + w}^2 = \abs z^2 + \abs w^2 + 2\Re(z\conj w),
\qquad
\abs{z - w}^2 = \abs z^2 + \abs w^2 - 2\Re(z\conj w),
\]
en tel op. Meetkundig zijn $\abs{z+w}$ en $\abs{z-w}$ de lengten van
de twee diagonalen van het parallellogram met hoekpunten $0, z, z+w, w$,
terwijl $\abs z$ en $\abs w$ de zijdelengten zijn: de som van de
kwadraten van de diagonalen is gelijk aan de som van de kwadraten van de vier
zijden.
\end{solution}

\begin{solution}{exo:b1:complex:10}
Zij $j = \eu^{2\iu\pi/3}$, zodat $j^2 + j + 1 = 0$ en $\eu^{\iu\pi/3} =
-j^2$, $\eu^{-\iu\pi/3} = -j$. De driehoek is gelijkzijdig direct d.e.s.d.a.\
$c - a = -j^2 (b - a)$, indirect d.e.s.d.a.\ $c - a = -j(b - a)$
(\cref{ex:b1:complex:equilateral}).

Beschouw $P = a + jb + j^2 c$ en $Q = a + j^2 b + jc$. Met behulp van $1 +
j^2 = -j$ en $j^3 = 1$:
\[
c - a + j^2(b - a) = c + j^2 b - (1 + j^2)\,a = c + j^2 b + ja
= j\,(a + jb + j^2 c) = jP,
\]
dus het directe geval luidt $jP = 0 \iff P = 0$; dezelfde berekening
met $j$ in plaats van $j^2$ geeft $c - a + j(b - a) = j^2 Q$, dus het
indirecte geval luidt $Q = 0$. Bijgevolg: gelijkzijdig (beide oriëntaties)
$\iff PQ = 0$. Uitwerken, met $j + j^2 = -1$:
\[
PQ = a^2 + b^2 + c^2 + (j + j^2)(ab + bc + ca)
= a^2 + b^2 + c^2 - (ab + bc + ca).
\]
Dus de driehoek is gelijkzijdig dan en slechts dan als $a^2 + b^2 + c^2 = ab +
bc + ca$.
\end{solution}

\begin{solution}{exo:b1:complex:11}
Aangezien de $\omega^k$, $k = 0, \dots, n-1$, precies de $n$ wortels
van $X^n - 1$ zijn (\cref{thm:b1:complex:roots}), en de veelterm
monisch is:
\[
X^n - 1 = \prod_{k=0}^{n-1} (X - \omega^k)
= (X - 1) \prod_{k=1}^{n-1} (X - \omega^k).
\]
Delen door $X - 1$: $\;1 + X + \dots + X^{n-1} = \prod_{k=1}^{n-1}
(X - \omega^k)$. Evalueren in $X = 1$ geeft $P = n$.

Nu $1 - \omega^k = -\eu^{\iu k\pi/n}\bigl(\eu^{\iu k\pi/n} -
\eu^{-\iu k\pi/n}\bigr) = -2\iu\,\eu^{\iu k\pi/n} \sin\frac{k\pi}{n}$,
dus door moduli te nemen in $P = n$ (elke $\sin\frac{k\pi}n > 0$ voor $1 \leq k
\leq n-1$):
\[
n = \abs P = \prod_{k=1}^{n-1} 2\sin\frac{k\pi}{n}
= 2^{n-1} \prod_{k=1}^{n-1} \sin\frac{k\pi}{n},
\qquad\text{dus}\qquad
\prod_{k=1}^{n-1} \sin\frac{k\pi}{n} = \frac{n}{2^{n-1}} .
\]
\end{solution}

\begin{solution}{exo:b1:complex:12}
$z = 1$ is geen oplossing ($2^n \neq 0$), dus de vergelijking is
equivalent met $\bigl(\frac{z+1}{z-1}\bigr)^n = 1$, d.w.z.\
$\frac{z+1}{z-1} = \omega^k$ met $\omega = \eu^{2\iu\pi/n}$ en
$k \in \{0, \dots, n-1\}$. De waarde $k = 0$ is uitgesloten ($z + 1 =
z - 1$ is onmogelijk). Voor $1 \leq k \leq n - 1$ geeft het oplossen van
$z + 1 = \omega^k(z - 1)$ dat $z(1 - \omega^k) = -1 - \omega^k$,
dus met $\varphi = \frac{2k\pi}{n}$ en de halvehoek-
ontbindingen $1 + \eu^{\iu\varphi} = 2\cos\frac\varphi2\,
\eu^{\iu\varphi/2}$, $\eu^{\iu\varphi} - 1 = 2\iu\sin\frac\varphi2
\,\eu^{\iu\varphi/2}$:
\[
z_k = \frac{1 + \omega^k}{\omega^k - 1}
= \frac{2\cos\frac{k\pi}{n}}{2\iu\,\sin\frac{k\pi}{n}}
= -\iu\,\frac{\cos\frac{k\pi}{n}}{\sin\frac{k\pi}{n}} ,
\]
zuiver imaginair zoals beweerd. De \omterm{def:b1:logic:map}{afbeelding} $t \mapsto \cos t/\sin t$ is
\omterm{def:b1:logic:inj}{injectief} op $\intoo0\pi$ (ze is strikt dalend), dus de
$n - 1$ waarden $z_k$ zijn paarsgewijs verschillend: de vergelijking, van graad
$n - 1$ eenmaal uitgewerkt (de $z^n$-termen heffen elkaar op), heeft precies deze
$n - 1$ oplossingen.
\end{solution}

\begin{solution}{pb:b1:complex:1}
\textbf{1.} $j = \cos\frac{2\pi}3 + \iu\sin\frac{2\pi}3 = -\frac12
+ \iu\frac{\sqrt3}2$; $j^2 = \eu^{4\iu\pi/3} = -\frac12 -
\iu\frac{\sqrt3}2 = \conj j$; $j^3 = 1$; $1 + j + j^2 = 0$ (som van
de derdemachtswortels van de eenheid, \cref{prop:b1:complex:sumroots}); $j^{-1}
= j^2$ (aangezien $j \cdot j^2 = 1$). Op de eenheidscirkel zijn $1$, $j$,
$j^2$ de hoekpunten van een directe gelijkzijdige driehoek.

\textbf{2.} De rotatie met middelpunt $a$ en hoek $\theta$ houdt $a$ vast
en draait elke vector die uit $a$ vertrekt over $\theta$: $r(z) - a =
\eu^{\iu\theta}(z - a)$, d.w.z.\ $r(z) = a + \eu^{\iu\theta}(z - a)$.
Samenstellen van $r(z) = a + \eu^{\iu\theta}(z-a)$ en $r'(z) = a' +
\eu^{\iu\theta'}(z-a')$:
\[
r' \circ r\,(z) = \eu^{\iu(\theta + \theta')} z +
\text{constante},
\]
een \omterm{def:b1:logic:map}{afbeelding} van de vorm $Az + B$ met $A = \eu^{\iu(\theta+\theta')}$ van
\omterm{def:b1:complex:field}{modulus} $1$. Volgens \cref{prop:b1:complex:similitude} is het een rotatie
met hoek $\arg A = \theta + \theta'$ als $A \neq 1$, en een
translatie als $A = 1$, d.w.z.\ als $\theta + \theta' \in 2\pi\Z$.

\textbf{3.} $(b, c, p)$ is gelijkzijdig d.e.s.d.a.\ $\abs{p - b} = \abs{c -
b} = \abs{p - c}$. Schrijf $q = \frac{p - b}{c - b}$; de eerste
gelijkheid zegt $\abs q = 1$ en de tweede $\abs{q - 1} = 1$;
samen, $q = \eu^{\pm\iu\pi/3}$ (de twee snijpunten van
de cirkels $\abs q = 1$ en $\abs{q - 1} = 1$ zijn $\frac12 \pm
\iu\frac{\sqrt3}2$). Bijgevolg $p = b + \eu^{\pm\iu\pi/3}(c - b)$. Voor
$a = 0$, $b = 1$, $c = \iu$ (een directe driehoek): de keuze
$\eu^{-\iu\pi/3}$ geeft
\[
p = 1 + \Bigl(\tfrac12 - \iu\tfrac{\sqrt3}2\Bigr)(\iu - 1)
= \tfrac{1 + \sqrt3}2\,(1 + \iu) \approx 1.37 + 1.37\,\iu ,
\]
wat aan de overkant van de rechte $BC$ ($x + y = 1$) ligt ten opzichte van $a
= 0$: naar buiten. De keuze $\eu^{+\iu\pi/3}$ geeft $p \approx -0.37
- 0.37\,\iu$, aan dezelfde kant als $a$: naar binnen.

\textbf{4.} Vermenigvuldig $P = a + jb + j^2c$ met $j^2$: $j^2 P = j^2 a +
j^3 b + j^4 c = b + jc + j^2 a$ (met $j^3 = 1$, $j^4 = j$); en
met $j$: $jP = ja + j^2 b + c$. Dit zijn de twee identiteiten. Als $P
= 0$ dan $j^2 P = jP = 0$: het criterium geldt ook voor $(b, c, a)$ en
$(c, a, b)$ --- cyclische invariantie (zoals het moet: een
gelijkzijdige driehoek geeft er niet om welk hoekpunt eerst genoemd wordt).

\textbf{5.} Als $a = b$: $0 = a(1 + j) + j^2 c = -j^2 a + j^2 c$
(met $1 + j = -j^2$), dus $c = a$. Als $b = c$: $0 = a + b(j + j^2)
= a - b$, dus $a = b$. Als $a = c$: $0 = a(1 + j^2) + jb = -ja + jb$,
dus $a = b$. In elk geval vallen alle drie de punten samen.

\textbf{6.} $(a'z + b') \circ (az + b) = a'a\,z + (a'b + b')$ met
$a'a \neq 0$: dezelfde vorm. De inverse van $z \mapsto az + b$ is $z
\mapsto \frac1a z - \frac ba$, opnieuw van dezelfde vorm. Met de
identieke \omterm{def:b1:logic:map}{afbeelding} als neutraal element vormen de directe gelijkvormigheden een
groep onder samenstelling.

\textbf{7.} $f(z) = az + b$ voldoet aan $f(z_1) = w_1$, $f(z_2) =
w_2$ d.e.s.d.a.\ $a z_1 + b = w_1$ en $a z_2 + b = w_2$; door af te trekken,
$a(z_2 - z_1) = w_2 - w_1$, dus
\[
a = \frac{w_2 - w_1}{z_2 - z_1} \;(\neq 0), \qquad
b = w_1 - a z_1
\]
zijn afgedwongen, en omgekeerd werkt deze keuze: bestaan en
uniciteit.

\textbf{8.} $\abs{f(z) - f(w)} = \abs{a(z - w)} = \abs a\,\abs{z -
w}$: alle afstanden worden vermenigvuldigd met $\abs a$. Voor de vorm:
\[
\frac{f(c') - f(a')}{f(b') - f(a')} = \frac{a(c' -
a')}{a(b' - a')} = \frac{c' - a'}{b' - a'} .
\]
Als twee driehoeken $(a', b', c')$ en $(a'', b'', c'')$ gelijke
vormen hebben, zij $f$ de unieke directe gelijkvormigheid met $f(a') = a''$
en $f(b') = b''$ (vraag 7); dan is de vorm van $(a'', b'',
f(c'))$ gelijk aan die van $(a', b', c')$, dus aan die van $(a'', b'',
c'')$, en de vorm bepaalt het derde hoekpunt uit de eerste
twee: $f(c') = c''$. Omgekeerd volgen gelijke vormen uit de
weergegeven invariantie.

\textbf{9.} $b = f(0) = 1$; $a + b = f(1) = \iu$ geeft $a = \iu -
1$. Verhouding $\abs a = \sqrt2$, hoek $\arg(\iu - 1) =
\frac{3\pi}4$. Vast punt:
\[
\zeta = \frac{b}{1 - a} = \frac1{2 - \iu} = \frac{2 + \iu}5 .
\]
Dus $f$ is de directe gelijkvormigheid met middelpunt $\frac{2+\iu}5$, verhouding
$\sqrt2$ en hoek $\frac{3\pi}4$.

\textbf{10.} Met $\alpha + \beta = 1$ en $f(z) = az + b$:
\[
f(\alpha a' + \beta b') = a\alpha a' + a\beta b' + b
= \alpha(a a' + b) + \beta(a b' + b)
= \alpha f(a') + \beta f(b') ,
\]
waarbij de sleutel $b = (\alpha + \beta)b$ is. Middens ($\alpha = \beta =
\frac12$), zwaartepunten (itereer), en de Napoleon-middelpunten $\alpha b
+ \beta c$ hieronder zijn daarom \emph{equivariant}: de driehoek
transformeren transformeert ze dienovereenkomstig. Bijgevolg bewijst het bewijzen van een
gelijkvormigheidsinvariante \omterm{def:b1:logic:statement}{uitspraak} voor één goed geplaatste driehoek
haar voor alle --- de vrijbrief gebruikt in vraag 21.

\textbf{11.} De naar buiten gerichte top op $[b, c]$ is $p_a = b +
\eu^{-\iu\pi/3}(c - b)$ (vraag 3), dus het middelpunt is
\[
n_a = \frac{b + c + p_a}3
= \frac{2b + c + \frac{1 - \iu\sqrt3}2\,(c - b)}3
= \frac{(3 + \iu\sqrt3)\,b + (3 - \iu\sqrt3)\,c}6
= \alpha b + \beta c ,
\]
met $\alpha = \frac{3 + \iu\sqrt3}6$ en $\beta = \frac{3 -
\iu\sqrt3}6 = \conj\alpha$. Dezelfde berekening op de zijden
$[c, a]$ en $[a, b]$ geeft $n_b = \alpha c + \beta a$ en $n_c =
\alpha a + \beta b$ (cyclische verschuiving van de rollen).

\textbf{12.} $\alpha + \beta = \frac{3 + \iu\sqrt3 + 3 -
\iu\sqrt3}6 = 1$. Bijgevolg
\[
\frac{n_a + n_b + n_c}3
= \frac{(\alpha + \beta)(a + b + c)}3 = \frac{a + b + c}3 :
\]
de twee driehoeken delen hun zwaartepunt.

\textbf{13.} Groepeer volgens $\alpha$ en $\beta$ en pas de identiteiten van vraag 4
toe op $P = a + jb + j^2c$:
\[
n_a + j n_b + j^2 n_c
= \alpha\,(b + jc + j^2 a) + \beta\,(c + ja + j^2 b)
= \alpha\,j^2 P + \beta\,j P
= (\alpha j^2 + \beta j)\,P .
\]

\textbf{14.} Met $j = \frac{-1 + \iu\sqrt3}2$:
\[
\alpha j = \frac{(3 + \iu\sqrt3)(-1 + \iu\sqrt3)}{12}
= \frac{-3 + 3\iu\sqrt3 - \iu\sqrt3 - 3}{12}
= \frac{-3 + \iu\sqrt3}6 = -\beta ,
\]
dus $\alpha j + \beta = 0$, bijgevolg $\alpha j^2 + \beta j = j(\alpha j
+ \beta) = 0$, en vraag 13 geeft $n_a + jn_b + j^2n_c = 0$ voor
elke driehoek $(a, b, c)$. Volgens het criterium (vragen 4--5) vormen de
middelpunten een directe gelijkzijdige driehoek of één enkel punt:
de stelling van Napoleon.

\textbf{15.} De naar binnen gerichte top is $b + \eu^{+\iu\pi/3}(c - b)$, en
de berekening van vraag 11 met $\eu^{+\iu\pi/3} = \frac{1 +
\iu\sqrt3}2$ verwisselt $\alpha$ en $\beta$: $m_a = \beta b + \alpha
c$, $m_b = \beta c + \alpha a$, $m_c = \beta a + \alpha b$. Met behulp van
de analoge identiteiten $b + j^2c + ja = j\,Q$ en $c + j^2a + jb
= j^2 Q$ voor $Q = a + j^2b + jc$:
\[
m_a + j^2 m_b + j m_c
= \beta(b + j^2 c + ja) + \alpha(c + j^2 a + jb)
= (\beta j + \alpha j^2)\, Q = j(\beta + \alpha j)\,Q = 0 ,
\]
aangezien $\alpha j = -\beta$ (vraag 14). Dus de inwendige middelpunten
voldoen aan het \emph{indirecte} gelijkzijdige criterium: gelijkzijdig met
de tegengestelde oriëntatie (of een punt).

\textbf{16.} Volgens vraag 5 toegepast op het drietal $(n_a, n_b, n_c)$
(dat aan het directe criterium voldoet) vallen twee middelpunten samen d.e.s.d.a.\
alle drie samenvallen; en alle drie vallen samen d.e.s.d.a.\ \emph{beide} criteria gelden,
d.w.z.\ d.e.s.d.a.\ bovendien $n_a + j^2 n_b + j n_c = 0$. Bereken zoals in
vraag 13, met de identiteiten $b + j^2c + ja = jQ$, $c + j^2a +
jb = j^2Q$:
\[
n_a + j^2 n_b + j n_c
= \alpha\,jQ + \beta\,j^2 Q = j(\alpha + \beta j)\,Q .
\]
Rechtstreeks: $\beta j = \frac{(3 -
\iu\sqrt3)(-1 + \iu\sqrt3)}{12} = \frac{-3 + 3\iu\sqrt3 +
\iu\sqrt3 + 3}{12} = \frac{\iu\sqrt3}3$, dus $\alpha + \beta j =
\frac{3 + \iu\sqrt3 + 2\iu\sqrt3}6 = \frac{1 + \iu\sqrt3}2 \neq
0$. Bijgevolg ontaardt de uitwendige Napoleon-driehoek d.e.s.d.a.\ $Q = a +
j^2b + jc = 0$, d.w.z.\ d.e.s.d.a.\ $(a, b, c)$ een indirecte gelijkzijdige
driehoek is --- in dat geval wijzen de ``naar buiten gerichte'' constructies allemaal
in het omgeschreven gebied van de driehoek en delen ze één middelpunt.

\textbf{17.} Met $a = 0$, $b = 1$, $c = \iu$:
\[
n_a = \alpha + \beta\iu = \frac{(3 + \sqrt3)(1 + \iu)}6,
\qquad
n_b = \alpha\iu = \frac{-\sqrt3 + 3\iu}6,
\qquad
n_c = \beta = \frac{3 - \iu\sqrt3}6 .
\]
Gekwadrateerde zijden: $n_a - n_b = \frac{(3 + 2\sqrt3) +
\iu\sqrt3}6$ geeft $\abs{n_a - n_b}^2 = \frac{(3 + 2\sqrt3)^2 +
3}{36} = \frac{24 + 12\sqrt3}{36} = \frac{2 + \sqrt3}3$; $n_b -
n_c = \frac{(3 + \sqrt3)(-1 + \iu)}6$ geeft $\abs{n_b - n_c}^2 =
\frac{2(3 + \sqrt3)^2}{36} = \frac{24 + 12\sqrt3}{36}$; $n_c - n_a
= \frac{-\sqrt3 - \iu(3 + 2\sqrt3)}6$ geeft opnieuw dezelfde waarde.
Alle drie de gekwadrateerde zijden zijn gelijk aan $\frac{2 + \sqrt3}3$: gelijkzijdig,
zoals beloofd.

\textbf{18.} Uitwerken:
\[
(a - b)(c - d) + (a - d)(b - c)
= (ac - ad - bc + bd) + (ab - ac - bd + cd)
= ab - ad - bc + cd ,
\]
en $(a - c)(b - d) = ab - ad - bc + cd$: gelijk.

\textbf{19.} Neem moduli in vraag 18 en pas de driehoeks-
ongelijkheid toe (\cref{prop:b1:complex:rules}~(4)):
\[
AC \cdot BD = \abs{(a-b)(c-d) + (a-d)(b-c)}
\leq \abs{a-b}\,\abs{c-d} + \abs{a-d}\,\abs{b-c}
= AB \cdot CD + AD \cdot BC ,
\]
met gelijkheid d.e.s.d.a.\ de twee sommanden op een gemeenschappelijke halfrechte vanuit $0$ liggen
(het gelijkheidsgeval van de driehoeksongelijkheid).

\textbf{20.} Halvehoekontbinding
(\cref{met:b1:complex:trig}~(4)):
$\eu^{\iu\theta} - \eu^{\iu\varphi} = 2\iu\,\sin\frac{\theta -
\varphi}2\;\eu^{\iu(\theta + \varphi)/2}$, en moduli nemen doodt
de unimodulaire factoren: $\abs{\eu^{\iu\theta} - \eu^{\iu\varphi}}
= 2\,\abs{\sin\frac{\theta - \varphi}2}$.

\textbf{21.} Met $p = \eu^{\iu\theta}$, $\theta \in
\intoo{2\pi/3}{4\pi/3}$, geeft vraag 20
\[
PA = 2\,\abs{\sin\tfrac\theta2},
\quad
PB = 2\,\abs{\sin\bigl(\tfrac\theta2 - \tfrac\pi3\bigr)},
\quad
PC = 2\,\abs{\sin\bigl(\tfrac\theta2 - \tfrac{2\pi}3\bigr)} .
\]
Op de boog, $\frac\theta2 \in \intoo{\pi/3}{2\pi/3}$: dan
$\sin\frac\theta2 > 0$; $\frac\theta2 - \frac\pi3 \in
\intoo0{\pi/3}$, dus de tweede sinus is positief; $\frac\theta2 -
\frac{2\pi}3 \in \intoo{-\pi/3}0$, dus de derde is negatief en
$PC = 2\sin\bigl(\frac{2\pi}3 - \frac\theta2\bigr)$. Som naar
product:
\[
\sin\Bigl(\frac\theta2 - \frac\pi3\Bigr) +
\sin\Bigl(\frac{2\pi}3 - \frac\theta2\Bigr)
= 2\,\sin\frac\pi6\,\cos\Bigl(\frac\theta2 - \frac\pi2\Bigr)
= \sin\frac\theta2 ,
\]
dus $PB + PC = 2\sin\frac\theta2 = PA$: de stelling van van Schooten.

\textbf{22.} Bij $p = -1$: $PA = 2$, $PB = PC = 1$ en alle zijden van
de gelijkzijdige driehoek hebben lengte $\sqrt3$, dus beide leden van
Ptolemaeus luiden $2\sqrt3$. Algebraïsch, met $-1 - j^2 = j$ en
$1 + j = -j^2$:
\[
(a - b)(p - c) = (1 - j)\,(-1 - j^2) = (1 - j)j = j - j^2
= \iu\sqrt3 ,
\]
\[
(a - c)(b - p) = (1 - j^2)(j + 1) = 1 + j - j^2 - j^3
= j - j^2 = \iu\sqrt3 .
\]
De twee termen zijn gelijk, dus hun verhouding is
$1 \in \R_{>0}$: het gelijkheidsgeval van vraag 19, dat exact overeenkomt met $PA
\cdot BC = PB \cdot AC + PC \cdot AB$.

\textbf{23.} Voor $(a, b, c) = (1, j, j^2)$: $n_a = \alpha j +
\beta j^2 = -\beta + \beta j^2$ (vraag 14) $= \beta(j^2 - 1)$;
numeriek $\beta(j^2 - 1) = \frac{(3 - \iu\sqrt3)}6 \cdot
\bigl(-\frac32 - \iu\frac{\sqrt3}2\bigr) = -1$. Evenzo $n_b =
\alpha j^2 + \beta = -j$ en $n_c = \alpha + \beta j = -j^2$. De
uitwendige Napoleon-driehoek van $(1, j, j^2)$ is $(-1, -j, -j^2)$: de
oorspronkelijke driehoek gespiegeld door zijn zwaartepunt $0$ --- dezelfde
grootte, over een halve slag gedraaid. Een gelijkzijdige driehoek is een vaste vorm
van de Napoleon-constructie, geen krimpende limiet.

\textbf{24.} (i) Vermenigvuldiging met een unimodulair getal als rotatie
bouwde de toppen en middelpunten (vragen 2--3, 11) en zette
cirkelafstanden om in sinussen (vraag 20). (ii) De groeps-
structuur en de vorminvariant verantwoordden het normaliseren van de
omgeschreven cirkel tot de eenheidscirkel en de driehoek tot $(1, j, j^2)$
in vraag 21, via de equivariantie van vraag 10. (iii) De
halvehoekontbinding dreef zowel vraag 20 als de
som-naar-product-stap die van Schooten afrondt.

\textbf{25.} Het woordenboek zet meetkundige uitspraken om in
polynomiale identiteiten waarin elke hypothese een vergelijking is:
wat het je oplevert is mechanisatie --- de stelling van Napoleon herleid tot
$\alpha j + \beta = 0$, één regel rekenkunde in
$\Q(\iu\sqrt3)$; wat het kost is meetkundige zichtbaarheid --- de
berekening certificeert maar toont niet \emph{waarom} de middelpunten
zich sluiten tot een gelijkzijdige driehoek. Een eerlijk antwoord is dat de
identiteit de \emph{onvermijdelijkheid} van de stelling verklaart (ze geldt
identiek in $a, b, c$, dus er is geen vernuft van configuratie bij
betrokken), terwijl de tekening haar \emph{inhoud} verklaart. Hetzelfde
woordenboek keert terug in matrixvorm wanneer rotaties de
orthogonale $2 \times 2$-matrices van \cref{ch:b1:matrices} en
\cref{ch:b1:euclid} worden, waar ``vermenigvuldiging met
$\eu^{\iu\theta}$'' het prototype is van een lineaire isometrie.
\end{solution}
