\section*{Hoofdstuk \ref{ch:b1:diffeq} --- Lineaire differentiaalvergelijkingen}

\begin{solution}{exo:b1:diffeq:1}
Homogeen: $y_h = \lambda\,\eu^{-2x}$. Particulier: probeer $y_p =
c\,\eu^{3x}$ ($3$ is geen wortel van $r + 2$): $3c + 2c = 1$, $c =
\frac15$. Algemene oplossing: $y = \frac{\eu^{3x}}{5} +
\lambda\,\eu^{-2x}$. Met $y(0) = 1$: $\frac15 + \lambda = 1$,
$\lambda = \frac45$, dus $y = \frac{\eu^{3x} + 4\,\eu^{-2x}}{5}$.
\end{solution}

\begin{solution}{exo:b1:diffeq:2}
Op $\intoo{0}{+\infty}$, deel door $x$: $y' - \frac{2}{x}\,y = x^2$.
Hier is $A(x) = -2\ln x$, $\eu^{-A(x)} = x^2$: homogene oplossingen
$\lambda x^2$. \omterm{thm:b1:diffeq:voc}{Variatie van constanten}: $\mu'(x) = x^2 \cdot x^{-2} =
1$, dus $\mu = x + \lambda$ en
\[
y(x) = x^3 + \lambda x^2, \qquad \lambda \in \R .
\]
\emph{Controle:} $x(3x^2 + 2\lambda x) - 2(x^3 + \lambda x^2) = x^3$.
\end{solution}

\begin{solution}{exo:b1:diffeq:3}
$y'' - 3y' + 2y = 0$: wortels $1$ en $2$; $\;y = \lambda\,\eu^{x} +
\mu\,\eu^{2x}$.

$y'' + 4y' + 4y = 0$: dubbele wortel $-2$; $\;y = (\lambda + \mu
x)\,\eu^{-2x}$.

$y'' - 2y' + 5y = 0$: wortels $1 \pm 2\iu$; $\;y = \eu^{x}(\lambda\cos
2x + \mu\sin 2x)$.
\end{solution}

\begin{solution}{exo:b1:diffeq:4}
Homogeen: wortels $\pm 1$, $y_h = \lambda\,\eu^x + \mu\,\eu^{-x}$.
Particulier met veelterm-rechterlid ($\gamma = 0$ geen wortel):
$y_p = ax^2 + bx + c$; door substitutie geeft $2a - (ax^2 + bx + c) = x^2$
$a = -1$, $b = 0$, $c = 2a = -2$: $y_p = -x^2 - 2$. Algemene
oplossing $y = -x^2 - 2 + \lambda\eu^x + \mu\eu^{-x}$.

Cauchy: $y(0) = -2 + \lambda + \mu = 0$ en $y'(0) = \lambda - \mu =
1$: $\lambda = \frac32$, $\mu = \frac12$. Dus
$y = -x^2 - 2 + \frac{3\eu^x + \eu^{-x}}{2}$.
\end{solution}

\begin{solution}{exo:b1:diffeq:5}
$a(x) = \tan x$, $A(x) = -\ln(\cos x)$ (geldig: $\cos > 0$ op het
interval), $\eu^{-A} = \cos x$: homogene oplossingen $\lambda\cos x$.
\omterm{thm:b1:diffeq:voc}{Variatie van constanten}: $\mu'(x) = \sin 2x \cdot \frac{1}{\cos x} =
2\sin x$, dus $\mu = -2\cos x + \lambda$ en
\[
y(x) = -2\cos^2 x + \lambda \cos x .
\]
\emph{Controle:} $y' = 4\cos x \sin x - \lambda\sin x$ en $y\tan x =
-2\cos x\sin x + \lambda \sin x$; hun som is $2\cos x\sin x =
\sin 2x$, zoals vereist.
\end{solution}

\begin{solution}{exo:b1:diffeq:6}
$\chi(r) = r^2 - 4r + 3 = (r-1)(r-3)$: $\gamma = 1$ is een enkelvoudige wortel
($m = 1$). Probeer $y_p = x(ax + b)\,\eu^x$. Met $u = ax^2 + bx$,
\[
y_p'' - 4y_p' + 3y_p = \bigl(u'' + (2 - 4)u' + \chi(1) u\bigr)\eu^x
= \bigl(2a - 2(2ax + b)\bigr)\eu^x .
\]
Identificeer met $(2x + 1)\eu^x$: $-4a = 2$ en $2a - 2b = 1$, dus $a =
-\frac12$, $b = -1$. Algemene oplossing:
\[
y = -\Bigl(\frac{x^2}{2} + x\Bigr)\eu^{x} + \lambda\,\eu^{x} +
\mu\,\eu^{3x}, \qquad (\lambda, \mu) \in \R^2 .
\]
\end{solution}

\begin{solution}{exo:b1:diffeq:7}
Homogeen: $y_h = \lambda\cos 2x + \mu\sin 2x$.

Rechterlid $x$ ($\gamma = 0$ geen wortel): $y_1 = ax + b$ met
$4(ax + b) = x$: $y_1 = \frac x4$.

Rechterlid $\sin 2x = \Im(\eu^{2\iu x})$, $2\iu$ enkelvoudige wortel van
$r^2 + 4$: probeer $z = c\,x\,\eu^{2\iu x}$; dan $z'' + 4z =
4\iu c\,\eu^{2\iu x}$, gelijk aan $\eu^{2\iu x}$ voor $c =
\frac{1}{4\iu} = -\frac{\iu}{4}$. Dus $z = -\frac{\iu x}{4}(\cos 2x +
\iu \sin 2x)$ en $y_2 = \Im(z) = -\frac{x\cos 2x}{4}$.

Door superpositie:
\[
y = \frac{x}{4} - \frac{x\cos 2x}{4} + \lambda\cos 2x + \mu\sin 2x .
\]
\end{solution}

\begin{solution}{exo:b1:diffeq:8}
De vergelijking $T' + kT = 20k$ heeft constante particuliere oplossing $20$ en
homogene oplossingen $\lambda\eu^{-kt}$: $T(t) = 20 +
\lambda\,\eu^{-kt}$, en $T(0) = 80$ geeft $\lambda = 60$:
\[
T(t) = 20 + 60\,\eu^{-kt} .
\]
$T(10) = 50$: $\eu^{-10k} = \frac12$, dus $k = \frac{\ln 2}{10}$. Dan
vereist $T(t) = 25$ dat $\eu^{-kt} = \frac{5}{60} = \frac{1}{12}$, d.w.z.
\[
t = \frac{\ln 12}{k} = 10\,\frac{\ln 12}{\ln 2} \approx 35.8
\text{ minuten.}
\]
\end{solution}

\begin{solution}{exo:b1:diffeq:9}
\begin{enumerate}
  \item $\chi(r) = r^2 + 2\varepsilon r + 1$, $\Delta = 4(\varepsilon^2
        - 1)$.
        Voor $\varepsilon \in \intco{0}{1}$: wortels $-\varepsilon \pm
        \iu\sqrt{1 - \varepsilon^2}$, dus
        $y = \eu^{-\varepsilon t}\bigl(\lambda\cos\omega t +
        \mu\sin\omega t\bigr)$ met $\omega = \sqrt{1 -
        \varepsilon^2}$.
        Voor $\varepsilon = 1$: dubbele wortel $-1$, $y = (\lambda + \mu
        t)\,\eu^{-t}$.
        Voor $\varepsilon > 1$: reële wortels $r_\pm = -\varepsilon \pm
        \sqrt{\varepsilon^2 - 1}$, beide $< 0$, en $y =
        \lambda\eu^{r_+t} + \mu\eu^{r_-t}$.
  \item Voor $\varepsilon \in \intoo{0}{1}$: $\abs y \leq
        \eu^{-\varepsilon t}(\abs\lambda + \abs\mu) \to 0$. Voor
        $\varepsilon = 1$: $(\lambda + \mu t)\eu^{-t} \to 0$
        (exponentiële functie verslaat veelterm,
        \cref{prop:b1:functions:powerrules}). Voor $\varepsilon > 1$:
        beide exponentiële functies nemen af aangezien $r_\pm < 0$ (inderdaad
        $\sqrt{\varepsilon^2 - 1} < \varepsilon$). Voor $\varepsilon =
        0$: $y = \lambda\cos t + \mu\sin t$ heeft constante amplitude
        $\sqrt{\lambda^2 + \mu^2} \neq 0$ tenzij $y = 0$.
  \item Schrijf $\lambda\cos\omega t + \mu\sin\omega t =
        R\cos(\omega t - \varphi)$ met $R = \sqrt{\lambda^2 + \mu^2}
        > 0$. De nulpunten van $y$ zijn die van $\cos(\omega t -
        \varphi)$ (de factor $\eu^{-\varepsilon t}$ wordt nooit
        nul): $\omega t - \varphi \equiv \frac\pi2 \pmod \pi$,
        een rekenkundige rij met afstand $\frac{\pi}{\omega} =
        \frac{\pi}{\sqrt{1 - \varepsilon^2}}$.
\end{enumerate}
\end{solution}

\begin{solution}{exo:b1:diffeq:10}
Stel $x = y = 0$: $2f(0) = 2f(0)^2$, en $f(0) \neq 0$ geeft $f(0) =
1$. Houd $x$ vast en differentieer de vergelijking tweemaal naar $y$:
\[
f''(x+y) + f''(x-y) = 2 f(x) f''(y) .
\]
Stellend $y = 0$: $\;2f''(x) = 2 f(x) f''(0)$, dat wil zeggen
\[
f''(x) = c\,f(x), \qquad c = f''(0).
\]
\emph{Geval $c = \omega^2 > 0$:} $f(x) = \lambda\cosh\omega x +
\mu\sinh\omega x$; $f(0) = 1$ geeft $\lambda = 1$. Ingevuld in de
functievergelijking en met gebruik van de somformules
(\cref{prop:b1:functions:hyprules}), dwingt de vergelijking $\mu = 0$ af
(vergelijk de coëfficiënten van $\sinh\omega x \sinh\omega y$ of
evalueer in $x = y$): $f = \cosh\omega x$, wat inderdaad voldoet aan
$\cosh(x+y) + \cosh(x-y) = 2\cosh x\cosh y$.

\emph{Geval $c = -\omega^2 < 0$:} analoog $f(x) = \cos\omega x$
($\omega \neq 0$), wat aan de vergelijking voldoet.

\emph{Geval $c = 0$:} $f$ affien met $f(0) = 1$: $f(x) = 1 + \mu x$;
de vergelijking dwingt $\mu = 0$ af, uitgesloten ($f$ niet constant).

Conclusie: de oplossingen zijn $f(x) = \cos\omega x$ en $f(x) =
\cosh\omega x$, $\omega > 0$.
\end{solution}

\begin{solution}{exo:b1:diffeq:11}
Stel $z(t) = y(\eu^t)$, zodat $y(x) = z(\ln x)$ voor $x > 0$. Dan
\[
y'(x) = \frac{z'(\ln x)}x,
\qquad
y''(x) = \frac{z''(\ln x) - z'(\ln x)}{x^2} ,
\]
en door substitutie in de vergelijking:
\[
x^2y'' - xy' + y = \bigl(z'' - z'\bigr) - z' + z
= z'' - 2z' + z = 0 .
\]
\omterm{def:b1:diffeq:linear2}{Karakteristieke veelterm} $(r - 1)^2$: dubbele wortel $1$, dus $z(t) =
(\lambda + \mu t)\,\eu^t$ en, terug in de variabele $x = \eu^t$:
\[
y(x) = (\lambda + \mu\ln x)\,x,
\qquad \lambda, \mu \in \R .
\]
\end{solution}

\begin{solution}{exo:b1:diffeq:12}
\begin{enumerate}
  \item In genormaliseerde vorm $y' - \frac2x\,y = 0$ op elk interval:
        $A(x) = -2\ln\abs x$, dus de oplossingen zijn $y = a x^2$ op
        $\intoo0{+\infty}$ en $y = b x^2$ op $\intoo{-\infty}0$,
        met onafhankelijke constanten
        (\cref{thm:b1:diffeq:homogeneous1}).
  \item Zij $y = ax^2$ voor $x \geq 0$ en $bx^2$ voor $x < 0$. Op
        elke open halfrechte is $y$ afleidbaar met $xy' = 2y$.
        In $0$: de differentiequotiënten $\frac{y(h) - y(0)}h =
        ah$ of $bh$ naderen $0$, dus $y'(0) = 0$ bestaat, en de
        vergelijking in $x = 0$ luidt $0 \cdot y'(0) = 2y(0) = 0$:
        voldaan. Dus $y$ lost de vergelijking op op heel $\R$.
  \item De oplossingen op $\R$ zijn precies deze aaneengelijmde functies: een
        familie met \emph{twee} parameters voor een vergelijking van de eerste orde.
        Er is geen tegenspraak met \cref{thm:b1:diffeq:voc},
        waarvan de hypothesen hier falen: geschreven als $y' + a(x)y = 0$,
        is de coëfficiënt $a(x) = -\frac2x$ niet continu in $0$
        --- zelfs niet gedefinieerd --- dus $\R$ is geen interval waarop
        de stelling van toepassing is. De singulariteit in $0$
        ontkoppelt de twee halfrechten, en de waarde $y(0) = 0$
        is afgedwongen en draagt geen informatie over. Elk Cauchy-
        gegeven in $x_0 \neq 0$ bepaalt de oplossing alleen op de
        halfrechte die $x_0$ bevat.
\end{enumerate}
\end{solution}

\begin{solution}{pb:b1:diffeq:1}
\textbf{1.} $\chi(r) = r^2 + 2\lambda r + \omega_0^2$, $\Delta =
4(\lambda^2 - \omega_0^2) < 0$ voor $0 < \lambda < \omega_0$: wortels
$-\lambda \pm \iu\omega_d$, dus volgens
\cref{thm:b1:diffeq:homogeneous2}
\[
x(t) = \eu^{-\lambda t}\bigl(\lambda_1\cos\omega_d t +
\mu_1\sin\omega_d t\bigr), \qquad (\lambda_1, \mu_1) \in \R^2 .
\]
Voor $\lambda = 0$: $x = \lambda_1\cos\omega_0 t +
\mu_1\sin\omega_0 t$. De kritieke ($\lambda = \omega_0$) en
overgedempte ($\lambda > \omega_0$) regimes zijn die van
\cref{exo:b1:diffeq:9} (na herschaling van de tijd): $(\lambda_1 +
\mu_1 t)\eu^{-\lambda t}$, resp.\ combinaties van $\eu^{r_\pm t}$
met $r_\pm = -\lambda \pm \sqrt{\lambda^2 - \omega_0^2}$.

\textbf{2.} Ondergedempt: $\abs x \leq \eu^{-\lambda
t}(\abs{\lambda_1} + \abs{\mu_1}) \to 0$. Kritiek: $(\lambda_1 +
\mu_1 t)\eu^{-\lambda t} \to 0$ aangezien exponentiële functies
veeltermen verslaan (\cref{prop:b1:functions:powerrules}). Overgedempt:
$r_- < r_+ = -\lambda + \sqrt{\lambda^2 - \omega_0^2} < 0$
omdat $\sqrt{\lambda^2 - \omega_0^2} < \lambda$; beide
exponentiële functies nemen af.

\textbf{3.} Langs een oplossing van $(H)$, met $x'' = -2\lambda x' -
\omega_0^2 x$:
\[
\mathcal E'(t) = x'x'' + \omega_0^2 x x'
= x'\bigl(-2\lambda x' - \omega_0^2 x\bigr) + \omega_0^2 xx'
= -2\lambda\,x'^2 \leq 0 .
\]
Als $x(t_0) = x'(t_0) = 0$ dan $\mathcal E(t_0) = 0$; $\mathcal E$
is niet-negatief en niet-stijgend, dus $\mathcal E \equiv 0$ op
$\intco{t_0}{+\infty}$, wat $x \equiv 0$ daar afdwingt; voor $t \leq
t_0$, pas hetzelfde argument toe op $\tilde x(t) = x(2t_0 - t)$, die
de vergelijking oplost met demping $-\lambda$ maar nog steeds
$\tilde{\mathcal E}(t_0) = 0$ en $\tilde{\mathcal E}' =
+2\lambda\tilde x'^2 \geq 0$ met $\tilde{\mathcal E} \geq 0$ heeft;
niet-negatief, niet-dalend en nul aan het rechteruiteinde van
$\intoc{-\infty}{t_0}$ betekent overal nul. Dus $x \equiv 0$ op
$\R$ --- een energiebewijs van eenduidigheid, geldig voor alle $\lambda
\geq 0$.

\textbf{4.} $x(t + T_d) = R\,\eu^{-\lambda t}\eu^{-\lambda
T_d}\cos(\omega_d t + 2\pi - \varphi) = \eu^{-\lambda T_d}x(t)$.
De krimpfactor per pseudoperiode is $\eu^{-\delta}$ met
$\delta = \lambda T_d = \frac{2\pi\lambda}{\omega_d}$. Voor
$\omega_0 = 1$, $\lambda = 0.1$: $\omega_d = \sqrt{0.99} =
0.99499$, dus $\delta = \frac{0.62832}{0.99499} = 0.6315$: elke
zwaai behoudt $\eu^{-0.63} \approx 53\%$ van zijn amplitude.

\textbf{5.} De amplitudefactor is $\eu^{-\lambda t}$, die
afneemt met $\eu$ over $t = \frac1\lambda$. Dat interval bevat
$\frac{1/\lambda}{T_d} = \frac{\omega_d}{2\pi\lambda}$
pseudoperioden. Voor $\lambda \ll \omega_0$, $\omega_d \approx
\omega_0$ en dit is $\approx \frac{\omega_0}{2\pi\lambda} =
\frac Q\pi$. Een gitaarsnaar met $Q = 300$ klinkt ongeveer honderd
perioden door; een deurdemper met $Q = 1$ voltooit er niet één.

\textbf{6.} Door $z\,\eu^{\iu\Omega t}$ te substitueren in het linkerlid
krijgen we $z\,(-\Omega^2 + 2\iu\lambda\Omega +
\omega_0^2)\,\eu^{\iu\Omega t}$, wat exact gelijk is aan $A\,\eu^{\iu\Omega
t}$ voor $z = \frac{A}{\omega_0^2 - \Omega^2 +
2\iu\lambda\Omega}$ (de noemer is niet nul: zijn imaginaire
deel is $2\lambda\Omega > 0$). Aangezien de coëfficiënten reëel zijn,
lost het reële deel $x_p = \Re\bigl(z\eu^{\iu\Omega t}\bigr)$ de
vergelijking op met rechterlid $\Re\bigl(A\eu^{\iu\Omega t}\bigr) =
A\cos\Omega t$.

\textbf{7.} Schrijf $\omega_0^2 - \Omega^2 + 2\iu\lambda\Omega =
\sqrt{D}\,\eu^{\iu\varphi}$ met $D = (\omega_0^2 - \Omega^2)^2 +
4\lambda^2\Omega^2$ en $\varphi \in \intoo0\pi$ (het imaginaire
deel $2\lambda\Omega$ is positief), dus $\tan\varphi =
\frac{2\lambda\Omega}{\omega_0^2 - \Omega^2}$. Dan $z =
\frac{A}{\sqrt D}\eu^{-\iu\varphi}$ en
\[
x_p(t) = \Re\Bigl(\frac A{\sqrt D}\,\eu^{\iu(\Omega t -
\varphi)}\Bigr) = R\cos(\Omega t - \varphi),
\qquad R = \frac A{\sqrt D} .
\]

\textbf{8.} Als $\Omega \to 0^+$: $D \to \omega_0^4$, dus $R \to
A/\omega_0^2$ en $\tan\varphi \to 0^+$ met $\varphi \in
\intoo0{\frac\pi2}$: $\varphi \to 0$. De massa volgt de kracht
quasistatisch, verplaatst met kracht/stijfheid. Als $\Omega \to
+\infty$: $D \sim \Omega^4$, dus $R \sim A/\Omega^2 \to 0$, en
$\varphi \to \pi$ (het complexe getal $\omega_0^2 - \Omega^2 +
2\iu\lambda\Omega$ gaat naar het tweede kwadrant met argument $\to
\pi$): de massa beweegt nauwelijks, en in tegenfase ---
traagheid domineert.

\textbf{9.} Volgens \cref{thm:b1:diffeq:structure2}~(1) is elke oplossing
$x = x_p + x_h$ met $x_h$ een oplossing van $(H)$; volgens vraag 2, $x_h(t)
\to 0$, dus $x(t) - x_p(t) \to 0$: alle oplossingen convergeren naar
dezelfde steady state. De beginvoorwaarden vormen alleen de transiënt.

\textbf{10.} Als $x$ een periodieke oplossing is, is $x - x_p = x_h$ een
periodieke oplossing van $(H)$ die naar $0$ nadert in $+\infty$; een
periodieke functie met limiet $0$ is identiek $0$ (haar waarden op
één periode herhalen zich eeuwig, dus elke waarde is een limiet van een
deelrij die naar $0$ nadert). Bijgevolg $x = x_p$.

\textbf{11.} Hier is $\lambda = 1$, $\omega_0^2 = 2$, $\Omega = 1$,
$A = 1$: $z = \frac1{2 - 1 + 2\iu} = \frac{1 - 2\iu}5$, dus
\[
x_p = \Re\Bigl(\frac{(1 - 2\iu)(\cos t + \iu\sin t)}5\Bigr)
= \frac{\cos t + 2\sin t}5 .
\]
Homogeen: wortels van $r^2 + 2r + 2$ zijn $-1 \pm \iu$: $x_h =
\eu^{-t}(C\cos t + S\sin t)$. Voorwaarden: $x(0) = \frac15 + C = 0$
geeft $C = -\frac15$; door te differentiëren, $x'(0) = \frac25 - C + S =
0$ geeft $S = C - \frac25 = -\frac35$. Bijgevolg
\[
x(t) = \underbrace{\frac{\cos t + 2\sin t}5}_{\text{steady}}
- \underbrace{\eu^{-t}\,\frac{\cos t + 3\sin
t}5}_{\text{transiënt}} ,
\]
waarbij de transiënt sterft als $\eu^{-t}$.

\textbf{12.} Door uit te werken, $g(u) = u^2 - 2(\omega_0^2 - 2\lambda^2)u
+ \omega_0^4 = (u - u_r)^2 + g(u_r)$ met $u_r = \omega_0^2 -
2\lambda^2$ en
\[
g(u_r) = \omega_0^4 - u_r^2 = (\omega_0^2 - u_r)(\omega_0^2 +
u_r) = 2\lambda^2\bigl(2\omega_0^2 - 2\lambda^2\bigr) =
4\lambda^2(\omega_0^2 - \lambda^2) .
\]
Als $2\lambda^2 < \omega_0^2$, dan is $u_r > 0$ een toelaatbare
gekwadrateerde frequentie: $g$ heeft daar een strikt minimum, dus $R = A/\sqrt
g$ heeft een strikt maximum in $\Omega_r = \sqrt{u_r} =
\sqrt{\omega_0^2 - 2\lambda^2}$, met $R_{\max} = A/\sqrt{g(u_r)}
= \frac{A}{2\lambda\sqrt{\omega_0^2 - \lambda^2}}$.

\textbf{13.} $R(0) = A/\omega_0^2$, dus
\[
\frac{R_{\max}}{R(0)}
= \frac{\omega_0^2}{2\lambda\sqrt{\omega_0^2 - \lambda^2}}
= \frac{\omega_0}{2\lambda}\cdot
\frac{\omega_0}{\sqrt{\omega_0^2 - \lambda^2}}
= Q\,\Bigl(1 - \frac{\lambda^2}{\omega_0^2}\Bigr)^{-1/2}
\geq Q .
\]
Voor zwakke demping is de correctiefactor dicht bij $1$: \omterm{ex:b1:diffeq:oscillation}{resonantie}
vermenigvuldigt de statische verplaatsing met in wezen $Q$.

\textbf{14.} $V(\Omega)^2 = \frac{A^2 u}{g(u)}$ met $u =
\Omega^2$. Zijn afgeleide heeft het teken van $g(u) - u\,g'(u) =
(\omega_0^2 - u)^2 + 4\lambda^2 u - u\bigl(2(u - \omega_0^2) +
4\lambda^2\bigr) = (\omega_0^2 - u)^2 + 2u(\omega_0^2 - u) =
(\omega_0^2 - u)(\omega_0^2 + u)$, positief voor $u < \omega_0^2$
en negatief daarboven: strikt maximum precies in $\Omega =
\omega_0$, voor elke demping. Daar blaast $\tan\varphi$ op met
$\varphi \in \intoo0\pi$: $\varphi = \frac\pi2$, en $x_p'(t) =
-R\Omega\sin(\Omega t - \frac\pi2) = R\Omega\cos(\Omega t)$ is
exact in fase met de kracht: optimale vermogensoverdracht.

\textbf{15.} $R = R_{\max}/\sqrt2 \iff g(u) = 2g(u_r) \iff (u -
u_r)^2 = g(u_r) = 4\lambda^2(\omega_0^2 - \lambda^2)$, wat geeft
\[
u_\pm = u_r \pm 2\lambda\sqrt{\omega_0^2 - \lambda^2},
\qquad
\Omega_+^2 - \Omega_-^2 = 4\lambda\sqrt{\omega_0^2 - \lambda^2} .
\]
Dan $\Omega_+ - \Omega_- = \frac{\Omega_+^2 -
\Omega_-^2}{\Omega_+ + \Omega_-}$, en voor $\lambda \ll \omega_0$
zijn beide $\Omega_\pm \approx \omega_0$: $\Omega_+ - \Omega_- \approx
\frac{4\lambda\omega_0}{2\omega_0} = 2\lambda$, dus
$\frac{\omega_0}{\Omega_+ - \Omega_-} \approx
\frac{\omega_0}{2\lambda} = Q$. De breedte van een resonantiepiek meten
meet de kwaliteitsfactor.

\textbf{16.} $Q = 10$; $\Omega_r = \sqrt{1 - 2(0.05)^2} =
\sqrt{0.995} = 0.9975$; $R_{\max} = \frac1{2 \times 0.05
\sqrt{1 - 0.0025}} = \frac1{0.1 \times 0.99875} = 10.01$; statische
respons $R(0) = 1$; bandbreedte $\approx 2\lambda = 0.1$. Een hoge
dunne piek van hoogte $\approx Q$ boven een plateau van hoogte $1$.

\textbf{17.} Als $2\lambda^2 \geq \omega_0^2$ dan $u_r \leq 0$ en
$g'(u) = 2(u - u_r) > 0$ voor alle $u > 0$: $g$ stijgt strikt
op $\intoo0{+\infty}$, dus $R = A/\sqrt g$ daalt strikt vanaf
$R(0) = A/\omega_0^2$: de respons is het grootst bij nulfrequentie
en er is geen piek.

\textbf{18.} $\gamma = \iu\Omega$ is geen wortel van $r^2 +
\omega_0^2$ (aangezien $\Omega \neq \omega_0$), dus
\cref{met:b1:diffeq:particular} met $m = 0$ geeft $x_p =
\frac{A\cos\Omega t}{\omega_0^2 - \Omega^2}$ (substitueer en
controleer: $-\Omega^2 + \omega_0^2$ maal de cosinus). Algemene
oplossing:
\[
x(t) = \frac{A\cos\Omega t}{\omega_0^2 - \Omega^2}
+ \lambda_1\cos\omega_0 t + \mu_1\sin\omega_0 t .
\]

\textbf{19.} $x(0) = 0$ dwingt $\lambda_1 = -\frac A{\omega_0^2 -
\Omega^2}$ af, en $x'(0) = 0$ dwingt $\mu_1 = 0$ af:
\[
x(t) = \frac{A}{\omega_0^2 - \Omega^2}\,
\bigl(\cos\Omega t - \cos\omega_0 t\bigr)
= \frac{2A}{\omega_0^2 - \Omega^2}\,
\sin\Bigl(\frac{(\omega_0 - \Omega)t}2\Bigr)
\sin\Bigl(\frac{(\omega_0 + \Omega)t}2\Bigr) ,
\]
door de productformule $\cos a - \cos b =
2\sin\frac{b - a}2\sin\frac{b + a}2$ toegepast met $a = \Omega t$,
$b = \omega_0 t$.

\textbf{20.} Voor $\Omega$ nabij $\omega_0$ oscilleert de tweede sinus
op de snelle frequentie $\frac{\omega_0 + \Omega}2
\approx \omega_0$, terwijl de eerste een trage omhullende is van frequentie
$\frac{\abs{\omega_0 - \Omega}}2$: de amplitude van de snelle
oscillatie wast en neemt af met omhullende-periode
$\frac{2\pi}{\abs{\omega_0 - \Omega}}$ (twee \omterm{pb:b1:diffeq:1}{zwevingen} per omhullende-
periode), en bereikt maxima $\frac{2A}{\abs{\omega_0^2 - \Omega^2}}$.
Als $\Omega \to \omega_0$, worden de \omterm{pb:b1:diffeq:1}{zwevingen} zowel trager (periode
$\to \infty$) als hoger (amplitude $\to \infty$).

\textbf{21.} Houd $t$ vast. Als $\Omega \to \omega_0$:
\[
\frac{2A}{\omega_0^2 - \Omega^2}\sin\Bigl(\frac{(\omega_0 -
\Omega)t}2\Bigr)
= \frac{2A}{\omega_0 + \Omega}\cdot
\frac{\sin\bigl(\frac{(\omega_0 - \Omega)t}2\bigr)}
{\omega_0 - \Omega}
\longrightarrow \frac{2A}{2\omega_0}\cdot\frac t2
= \frac{At}{2\omega_0} ,
\]
terwijl $\sin\bigl(\frac{(\omega_0 + \Omega)t}2\bigr) \to
\sin(\omega_0 t)$: de limiet is $x_\infty(t) =
\frac{At\sin\omega_0 t}{2\omega_0}$. Directe controle: met $C =
\frac A{2\omega_0}$, $x_\infty = Ct\sin\omega_0 t$ heeft
$x_\infty'' = 2C\omega_0\cos\omega_0 t - C\omega_0^2 t
\sin\omega_0 t$, dus $x_\infty'' + \omega_0^2 x_\infty =
2C\omega_0\cos\omega_0 t = A\cos\omega_0 t$, met $x_\infty(0) =
0$ en $x_\infty'(0) = C\sin 0 + C\omega_0 \cdot 0 \cdot \cos 0 =
0$ --- voor $\omega_0 = A = 1$ is dit
precies \cref{ex:b1:diffeq:oscillation}. \omterm{ex:b1:diffeq:oscillation}{Resonantie} is de
degeneratie van \omterm{pb:b1:diffeq:1}{zwevingen}: de eerste opzwelling van de omhullende, uitgerekt
tot oneindige lengte.

\textbf{22.} Zonder demping groeit de resonante amplitude lineair
en zonder begrenzing; met demping $\lambda > 0$ verzadigt de groei
bij $R_{\max} \approx Q\,\frac A{\omega_0^2}$. Het mechanisme:
dissipatie verwijdert energie tegen een snelheid die groeit met de amplitude
(vraag 23), dus de opbouw stopt precies wanneer de demping
energie verbrandt zo snel als de aandrijving ze aanlevert.

\textbf{23.} Met $x_p' = -R\Omega\sin(\Omega t - \varphi)$, over
een periode geven de gemiddelden $\langle\cos^2\rangle =
\langle\sin^2\rangle = \frac12$ en $\langle\sin\cos\rangle = 0$:
\[
\langle P_{\mathrm{diss}}\rangle
= 2\lambda\,R^2\Omega^2\,\langle\sin^2(\Omega t -
\varphi)\rangle = \lambda R^2\Omega^2 ;
\]
en door $\sin(\Omega t - \varphi) = \sin\Omega t\cos\varphi
- \cos\Omega t\sin\varphi$ uit te werken:
\[
\langle P_{\mathrm{in}}\rangle
= -AR\Omega\,\bigl\langle\cos\Omega t\,\sin(\Omega t -
\varphi)\bigr\rangle
= AR\Omega\,\frac{\sin\varphi}2 .
\]
Aangezien $\sin\varphi = \frac{2\lambda\Omega}{\sqrt D} =
\frac{2\lambda\Omega R}A$, is dit $\frac{AR\Omega}2 \cdot
\frac{2\lambda\Omega R}A = \lambda R^2\Omega^2$: geïnjecteerd en
gedissipeerd vermogen balanceren exact --- de bepalende eigenschap van een
steady regime.

\textbf{24.} (i) De structuurstelling splitste elke oplossing in
steady state plus transiënt (vragen 9--11) en reduceerde
eenduidigheid tot het homogene probleem. (ii) De complexe methode
maakte van de zoektocht naar een particuliere oplossing één deling van
complexe getallen (vraag 6), met amplitude en fase afgelezen van een
\omterm{def:b1:complex:field}{modulus} en een argument. (iii) De resonantiekromme is een pure
functiestudie --- een kwadratische in $u = \Omega^2$, haar minimum,
haar niveauverzamelingen --- in de stijl van \cref{ch:b1:functions}
(vragen 12--17).

\textbf{25.} Vrij en gedempt: afnemende pseudo-oscillaties,
levensduur $\frac1\lambda$, ongeveer $\frac Q\pi$ zwaaien. Geforceerd en
gedempt: transiënten sterven, en een unieke sinusoïdale steady state
overleeft op de aandrijvingsfrequentie, met amplitude die piekt nabij
$\omega_0$ (hoogte $\approx Q \times$ statisch, breedte $\approx
\frac{\omega_0}Q$) en fase die veegt van $0$ naar $\pi$ door
$\frac\pi2$ in $\omega_0$. Vrij en ongedempt: eeuwigdurende
oscillatie. Geforceerd en ongedempt: \omterm{pb:b1:diffeq:1}{zwevingen}, degenererend in
lineair groeiende \omterm{ex:b1:diffeq:oscillation}{resonantie} bij exacte afstemming. Eén dimensieloos
getal, $Q = \frac{\omega_0}{2\lambda}$, stemt alles af ---
piekhoogte, bandbreedte, en transiëntlevensduur zijn drie aflezingen
van dezelfde knop. Het vervolg: het herschrijven van $x'' + 2\lambda x' +
\omega_0^2x$ als een stelsel van de eerste orde opent de matrixmethoden van
\cref{ch:b1:matrices} en het deel van jaar 2, en het ontbinden van een
willekeurige periodieke aandrijving in sinusoïden (Fourierreeksen, deel van jaar 3)
maakt de enkelvoudige-frequentieanalyse van dit probleem de
universele bouwsteen: los op voor elke frequentie, superponeer.
\end{solution}
