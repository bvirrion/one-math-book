\begin{solution}{exo:b1:vspaces:1}
\begin{enumerate}
  \item Ja: bevat $0$, en de definiĂ«rende vergelijking is lineair
        (stabiel onder $x + \lambda y$).
  \item Nee: bevat $(0,0,0)$ niet.
  \item Nee: $(1, 0)$ en $(0, -1)$ horen erbij ($xy = 0$), hun som

\begin{solution}{exo:b1:vspaces:2}
$(1,2,1) = a(1,0,1) + b(1,1,0)$ vereist $a + b = 1$, $b = 2$, $a =
1$: onverenigbaar ($a + b = 3 \neq 1$): niet in het \omterm{def:b1:vspaces:span}{opspansel}.
$(2,1,1) = a(1,0,1) + b(1,1,0)$: $b = 1$, $a = 1$, $a + b = 2$:
consistent, dus $(2,1,1) = (1,0,1) + (1,1,0)$, in het \omterm{def:b1:vspaces:span}{opspansel}.

Vergelijking: $(x, y, z) = (a + b, b, a)$ betekent $x = y + z$: het
\omterm{def:b1:vspaces:span}{opspansel} is het vlak $\{x - y - z = 0\}$.
\end{solution}

\begin{solution}{exo:b1:vspaces:3}
Eerste familie: $\lambda(1,1,0) + \mu(1,0,1) + \nu(0,1,1) = 0$
geeft $\lambda + \mu = 0$, $\lambda + \nu = 0$, $\mu + \nu = 0$:
optellend, $2(\lambda + \mu + \nu) = 0$, en elke vergelijking
aftrekkend, $\lambda = \mu = \nu = 0$: \omterm{def:b1:vspaces:free}{vrij}.

Tweede: $(2,4,6) = 2(1,2,3)$: gebonden.

Derde: vier vectoren in $\R^3$ --- noodzakelijk gebonden zodra de
dimensie beschikbaar is (\cref{ch:b1:findim}); rechtstreeks:
$(0,1,1) = -(1,0,0) + 0\cdot(1,1,0) + (1,1,1)$, inderdaad $(-1,0,0)
+ (1,1,1) = (0,1,1)$: een niet-triviale relatie.
\end{solution}

\begin{solution}{exo:b1:vspaces:4}
De \omterm{def:b1:poly:def}{veeltermen} $1, (X-1), (X-1)^2, (X-1)^3$ hebben verschillende
graden $0, 1, 2, 3$: \omterm{def:b1:vspaces:free}{vrij}
(\cref{prop:b1:vspaces:freecriteria}~(1)); vier vrije vectoren die
voortbrengen (elke $P \in \R_3[X]$ ontwikkelt in machten van $X -
1$, bijv.\ via Taylor voor \omterm{def:b1:poly:def}{veeltermen}, vgl.\ het bewijs van
\cref{prop:b1:poly:multiplicity}): een \omterm{def:b1:vspaces:free}{basis}. Voor $X^3$, Taylor in
$1$: $P = X^3$, $P(1) = 1$, $P'(1) = 3$, $P''(1) = 6$, $P'''(1) =
6$:
\[
X^3 = 1 + 3(X - 1) + 3(X-1)^2 + (X-1)^3 ,
\]
\omterm{prop:b1:vspaces:coordinates}{coĂ¶rdinaten} $(1, 3, 3, 1)$ (Pascals rij, zoals verwacht uit $X^3 =
((X-1)+1)^3$).
\end{solution}

\begin{solution}{exo:b1:vspaces:5}
$F \cap G$: de voorwaarden $x = y = z$ en $x = t = 0$ geven samen
$x = 0$, dus $y = z = 0$, en $t = 0$: de doorsnede is $\{0\}$. Som:
gegeven $(x,y,z,t)$, zoek $(a,a,a,b) \in F$ en $(0,c,d,0) \in G$ die
er samen op optellen: $a = x$, $b = t$, $c = y - x$, $d = z - x$:
altijd mogelijk. Dus $\R^4 = F \oplus G$, en
\[
(1,2,3,4) = (1,1,1,4) + (0,1,2,0) .
\]
\end{solution}

\begin{solution}{exo:b1:vspaces:6}
Een element van $G$ is $\lambda u$; als het convergeert, dan (daar
$\lambda u_n = \lambda(-1)^n$ de twee deelrijlimieten $\pm \lambda$
heeft) noodzakelijk $\lambda = 0$: $F \cap G = \{0\}$.

$F + G$ is niet alles: het bestaat uit rijen van de vorm $c_n +
\lambda(-1)^n$ met $(c_n)$ convergent. De rij $v_n = n$ is niet van
deze vorm ($v_n - \lambda(-1)^n$ is onbegrensd, nooit convergent).
Dus $F \oplus G \subsetneq$ (ruimte van alle rijen).
\end{solution}

\begin{solution}{exo:b1:vspaces:7}
($\supseteq$) Zowel $F \cap G$ als $F \cap H$ liggen in $F$, en hun
som ligt in $G + (F \cap H)$: de inclusie volgt aangezien het
linkerlid een \omterm{def:b1:vspaces:subspace}{deelruimte} is die beide stukken bevat --- concreet is
een element $g + h$ met $g \in F\cap G$, $h \in F \cap H$ in $F$
(som van twee elementen van $F$) en in $G + (F \cap H)$.

($\subseteq$) Zij $x \in F$ met $x = g + h$, $g \in G$, $h \in F
\cap H$. Dan $g = x - h \in F$ (verschil van elementen van $F$),
dus $g \in F \cap G$, en $x = g + h \in (F \cap G) + (F \cap H)$.

Tegenvoorbeeld voor volledige distributiviteit in $\R^2$: $F =
\operatorname{Vect}(1,1)$, $G = \operatorname{Vect}(1,0)$, $H =
\operatorname{Vect}(0,1)$. Dan $G + H = \R^2$, dus $F \cap (G+H) =
F$, terwijl $F \cap G = F \cap H = \{0\}$: het rechterlid is $\{0\}
\neq F$.
\end{solution}

\begin{solution}{exo:b1:vspaces:8}
\begin{enumerate}
  \item Stel $\sum_{i} \lambda_i \eu^{a_i x} = 0$ voor alle $x$.
        Vermenigvuldig met $\eu^{-a_p x}$: $\lambda_p + \sum_{i <
        p} \lambda_i \eu^{(a_i - a_p)x} \to \lambda_p$ als $x \to
        +\infty$ (elke exponent $a_i - a_p < 0$). Het linkerlid is
        identiek $0$, dus $\lambda_p = 0$; itereer naar beneden.
  \item Zij $a\cos x + b \sin x + c \cos 2x + d\sin 2x = 0$ voor
        alle $x$. Evalueer in $x = 0$: $a + c = 0$; in $x = \pi$:
        $-a + c = 0$; dus $a = c = 0$, en de relatie reduceert tot
        $b\sin x + d \sin 2x = 0$. Evalueer in $x = \frac\pi2$: $b
        = 0$; dan in $x = \frac\pi4$: $d = 0$.
  \item Stel $\sum \lambda_i \abs{x - a_i} = 0$ voor alle $x$. De
        functie $\sum_{i \neq j} \lambda_i\abs{x - a_i}$ is
        \omterm{def:b1:derivative:def}{afleidbaar} in $a_j$ (elke term is dat, weg van zijn eigen
        knik), dus $-\lambda_j \abs{x - a_j}$, hun verschil, moet
        ook \omterm{def:b1:derivative:def}{afleidbaar} zijn in $a_j$ --- wat $\lambda_j = 0$
        afdwingt ($\abs{\,\cdot\,}$ heeft een knik). Dit geldt voor
        elke $j$.
\end{enumerate}
\end{solution}

\begin{solution}{exo:b1:vspaces:9}
Zij $h \in H$. Aangezien $h \in H \subseteq F + H = F + G$, schrijf
$h = f + g$ met $f \in F$, $g \in G$. Dan $f = h - g \in H$ (beide
termen in $H$, gebruikmakend van $G \subseteq H$), dus $f \in F
\cap H = F \cap G \subseteq G$, en $h = f + g \in G$. Dus $H
\subseteq G$, en met de hypothese $G \subseteq H$: gelijkheid.

Tegenvoorbeeld zonder $G \subseteq H$: in $\R^2$, neem $F =
\operatorname{Vect}(1,0)$, $G = \operatorname{Vect}(0,1)$, $H =
\operatorname{Vect}(1,1)$: dan $F + G = F + H = \R^2$ en $F \cap G
= F \cap H = \{0\}$, en toch $G \neq H$.
\end{solution}

\begin{solution}{exo:b1:vspaces:10}
Beide \omterm{def:b1:logic:sets}{verzamelingen} zijn \omterm{def:b1:vspaces:subspace}{deelruimten} (de definiĂ«rende voorwaarden
zijn lineair en gelden voor $0$). Ontbinding: voor $P \in \R[X]$,
\[
P(X) = \underbrace{\frac{P(X) + P(-X)}{2}}_{\in\,\mathcal P}
+ \underbrace{\frac{P(X) - P(-X)}{2}}_{\in\,\mathcal I},
\]
en een \omterm{def:b1:poly:def}{veelterm} die zowel even als oneven is, voldoet aan $P = -P$,
dus $P = 0$: de som is direct en gelijk aan $\R[X]$.

Zij nu $P = \sum_k a_k X^k$ even. Dan is $P(X) - P(-X) = 2 \sum_{k
\text{ oneven}} a_k X^k$ de nulveelterm, dus elke coĂ«fficiĂ«nt van
oneven graad verdwijnt (\cref{def:b1:poly:def}): $P \in
\operatorname{Vect}(1, X^2, X^4, \dots)$, d.w.z.\ $P = Q(X^2)$ voor
een \omterm{def:b1:poly:def}{veelterm} $Q$. Omgekeerd is elke \omterm{def:b1:poly:def}{veelterm} in $X^2$ even.
\end{solution}

\begin{solution}{exo:b1:vspaces:11}
Stel $a(x_1 + x_2) + b(x_2 + x_3) + c(x_3 + x_1) = 0$.
Hergroeperend op de \omterm{def:b1:vspaces:free}{vrije familie} $(x_1, x_2, x_3)$:
\[
(a + c)\,x_1 + (a + b)\,x_2 + (b + c)\,x_3 = 0
\implies a + c = a + b = b + c = 0 .
\]
De eerste twee vergelijkingen aftrekken geeft $c = b$; de derde
geeft dan $2b = 0$, dus $b = c = 0$, dan $a = 0$: de familie is
\omterm{def:b1:vspaces:free}{vrij}.

Voor vier vectoren is de analoge familie \emph{altijd} gebonden:
\[
(x_1 + x_2) - (x_2 + x_3) + (x_3 + x_4) - (x_4 + x_1) = 0
\]
is een niet-triviale nulcombinatie (coĂ«fficiĂ«nten $1, -1, 1, -1$),
wat $(x_1, x_2, x_3, x_4)$ ook is. De pariteit van de cyclus\-lengte
beslist.
\end{solution}

\begin{solution}{exo:b1:vspaces:12}
\begin{enumerate}
  \item Als $F_1 \subseteq F_2$ of $F_2 \subseteq F_1$, dan is de
        vereniging een van de twee, dus eigenlijk. Anders kies $x
        \in F_1 \setminus F_2$ en $y \in F_2 \setminus F_1$, en
        beschouw $x + y$. Als $x + y \in F_1$, dan $y = (x + y) - x
        \in F_1$: tegenspraak. Als $x + y \in F_2$, dan $x \in
        F_2$: tegenspraak. Dus $x + y \notin F_1 \cup F_2$, en $E
        \neq F_1 \cup F_2$.
  \item Stel ter contradictie dat $E = F_1 \cup \dots \cup F_k$,
        met $k$ \emph{minimaal} gekozen onder al zulke
        overdekkingen. Minimaliteit verbiedt $F_1 \subseteq F_2
        \cup \dots \cup F_k$ (anders laat $F_1$ weg), dus is er een
        $x \in F_1$ met $x \notin F_i$ voor alle $i \geq 2$.
        Aangezien $F_1$ eigenlijk is, kies $y \notin F_1$. Voor
        elke scalair $t$ ligt de vector $y + t x$ in een of andere
        $F_i$. Ze ligt nooit in $F_1$: anders $y = (y + tx) - tx
        \in F_1$ (daar $x \in F_1$). Het \omterm{def:b1:structures:field}{lichaam} is oneindig, dus
        kies $k$ verschillende scalairen $t_1, \dots, t_k$: de $k$
        vectoren $y + t_j x$ vallen in de $k - 1$ \omterm{def:b1:vspaces:subspace}{deelruimten} $F_2,
        \dots, F_k$, en twee ervan, zeg $y + t x$ en $y + t' x$ met
        $t \neq t'$, liggen in dezelfde $F_i$ ($i \geq 2$). Dan hun
        verschil $(t - t')x \in F_i$, dus $x \in F_i$: tegenspraak.
        Dus bestaat er geen eindige overdekking door eigenlijke
        \omterm{def:b1:vspaces:subspace}{deelruimten}.
\end{enumerate}
\end{solution}

\begin{solution}{pb:b1:vspaces:1}
\textbf{1.} $L_i$ is een product van $n$ lineaire factoren gedeeld
door een niet-nul constante (de $x_i$ zijn verschillend), dus $\deg
L_i = n$. Evaluerend in $x_j$ met $j \neq i$: de factor $X - x_j$
van de teller verdwijnt, dus $L_i(x_j) = 0$. In $x_i$ vallen teller
en noemer samen: $L_i(x_i) = 1$.

\textbf{2.} Stel $\sum_i \lambda_i L_i = 0$. Evalueer in $x_j$:
alle termen sterven behalve $\lambda_j L_j(x_j) = \lambda_j$, dus
$\lambda_j = 0$ voor elke $j$: de familie is \omterm{def:b1:vspaces:free}{vrij}.

\textbf{3.} Zij $D = P - \sum_i P(x_i) L_i$. Dan $\deg D \leq n$
en, door vraag 1, $D(x_j) = P(x_j) - P(x_j) = 0$ voor de $n + 1$
verschillende punten $x_0, \dots, x_n$. Een niet-nul \omterm{def:b1:poly:def}{veelterm} van
graad $\leq n$ heeft hoogstens $n$ wortels
(\cref{cor:b1:poly:nroots}), dus $D = 0$. Dus is elke $P \in
\R_n[X]$ een combinatie van de $L_i$: de familie is voortbrengend,
en met vraag 2 een \omterm{def:b1:vspaces:free}{basis}.

\textbf{4.} Gegeven $y_0, \dots, y_n$, heeft de \omterm{def:b1:poly:def}{veelterm} $P =
\sum_i y_i L_i$ graad $\leq n$ en interpoleert. Uniciteit: een
interpolerende $P$ heeft, door vraag 3, \omterm{prop:b1:vspaces:coordinates}{coĂ¶rdinaten} $(P(x_0),
\dots, P(x_n)) = (y_0, \dots, y_n)$ in de \omterm{def:b1:vspaces:free}{basis} $(L_i)$, en
\omterm{prop:b1:vspaces:coordinates}{coĂ¶rdinaten} in een \omterm{def:b1:vspaces:free}{basis} zijn uniek
(\cref{prop:b1:vspaces:coordinates}). De \omterm{prop:b1:vspaces:coordinates}{coĂ¶rdinaten} van $P$ in de
\omterm{def:b1:vspaces:free}{Lagrange-basis} zijn zijn \emph{waarden in de knopen} --- dat is de
hele bedoeling van de \omterm{def:b1:vspaces:free}{basis}.

\textbf{5.} Pas vraag 3 toe op $P = X^k$ ($0 \leq k \leq n$):
\[
X^k = \sum_{i=0}^{n} x_i^{k} L_i ,
\]
en $k = 0$ geeft $\sum_i L_i = 1$.

\textbf{6.} $\deg N_k = k$ precies: de familie $(N_0, \dots, N_n)$
is een trap van graden in $\R_n[X]$, dus een \omterm{def:b1:vspaces:free}{basis} door
\cref{ex:b1:vspaces:staircase} (vrijheid uit
\cref{prop:b1:vspaces:freecriteria}~(1), voortbrenging via eindige
afdaling op de graad).

\textbf{7.} Uit de recurrentie,
\[
f[x_0, x_1] = \frac{f(x_1) - f(x_0)}{x_1 - x_0},
\qquad
f[x_0, x_1, x_2] = \frac{f[x_1, x_2] - f[x_0, x_1]}{x_2 - x_0}.
\]
Voor $f(x) = x^2$:
\[
f[x_0, x_1] = \frac{x_1^2 - x_0^2}{x_1 - x_0} = x_0 + x_1,
\]
en dan
\[
f[x_0, x_1, x_2]
= \frac{(x_1 + x_2) - (x_0 + x_1)}{x_2 - x_0}
= \frac{x_2 - x_0}{x_2 - x_0} = 1 .
\]

\textbf{8.} $\deg S \leq n$ aangezien $Q, R$ graad $\leq n - 1$
hebben. In $x_0$: $S(x_0) = \frac{-(x_0 - x_n) R(x_0)}{x_n - x_0} =
R(x_0) = f(x_0)$. In $x_n$: $S(x_n) = \frac{(x_n - x_0)
Q(x_n)}{x_n - x_0} = Q(x_n) = f(x_n)$. In een \omterm{def:b1:topology:closure}{inwendige} knoop $x_i$
($1 \leq i \leq n-1$) nemen zowel $Q$ als $R$ de waarde $f(x_i)$
aan, dus
\[
S(x_i) = \frac{(x_i - x_0) - (x_i - x_n)}{x_n - x_0}\, f(x_i)
= f(x_i) .
\]

\textbf{9.} Inductie op het aantal knopen. EĂ©n knoop: de
interpolant is de constante $f(x_0) = f[x_0]$. Neem de bewering aan
voor $k$ knopen en zij $S$ interpolant in $x_0, \dots, x_k$; door
uniciteit (vraag 4) is $S$ gegeven door het lemma van Aitken uit
$R$ (knopen $x_0, \dots, x_{k-1}$) en $Q$ (knopen $x_1, \dots,
x_k$). De coĂ«fficiĂ«nt van $X^{k}$ in $S$ is
\[
\frac{[X^{k-1}]\,Q - [X^{k-1}]\,R}{x_k - x_0}
= \frac{f[x_1, \dots, x_k] - f[x_0, \dots, x_{k-1}]}{x_k - x_0}
= f[x_0, \dots, x_k]
\]
door de inductiehypothese en de definiĂ«rende recurrentie.

\textbf{10.} Zij $P_k$ interpolant van $f$ in $x_0, \dots, x_k$.
Het verschil $P_k - P_{k-1}$ heeft graad $\leq k$ en verdwijnt in
$x_0, \dots, x_{k-1}$, dus door de factorstelling $k$ keer toegepast
(\cref{thm:b1:poly:factor}) is het gelijk aan $c\,N_k$ voor een
constante $c$; door de coĂ«fficiĂ«nten van $X^{k}$ te vergelijken en
vraag 9 te gebruiken, $c = f[x_0, \dots, x_k]$. Telescoperend vanaf
$P_0 = f(x_0) N_0$ geeft Newtons formule. Voor de gesloten vorm,
schrijf $P_k = \sum_{i \leq k} f(x_i) L_i$ (Lagrange, op de knopen
$x_0, \dots, x_k$) en lees de coĂ«fficiĂ«nt van $X^{k}$ af: elke $L_i$
draagt $\frac{1}{\prod_{j \neq i}(x_i - x_j)}$ bij, waaruit
\[
f[x_0, \dots, x_k] = \sum_{i=0}^{k}
\frac{f(x_i)}{\prod_{j \neq i,\, j \leq k}(x_i - x_j)} .
\]
Het rechterlid is invariant onder elke \omterm{def:b1:counting:objects}{permutatie} van de knopen,
dus de gedeelde differentie hangt niet af van hun ordening.

\textbf{11.} Als $P = a X^m + (\text{lagere graden})$, geeft het
binomium
\[
\Delta P = a\bigl((X+1)^m - X^m\bigr) + \dots
= a\,m\,X^{m-1} + (\text{lagere graden}),
\]
omdat $(X+1)^m - X^m = m X^{m-1} + \dots$ en het lagere-graad deel
van $P$ graad $\leq m - 2$ bijdraagt na $\Delta$ (of termen van
graad $\leq m-2$). Dus $\deg \Delta P = m - 1$ met leidende
coĂ«fficiĂ«nt $m a$. Een constante $c$ geeft $\Delta c = c - c = 0$.

\textbf{12.} $\deg B_k = k$: trap, dus een \omterm{def:b1:vspaces:free}{basis} van $\R_n[X]$
(\cref{ex:b1:vspaces:staircase}). Voor $\Delta B_k$ ($k \geq 1$),
factoriseer het gemeenschappelijke product:
\begin{align*}
k!\,\Delta B_k
&= (X+1)X\cdots(X-k+2) - X(X-1)\cdots(X-k+1) \\
&= X(X-1)\cdots(X-k+2)\,\bigl[(X+1) - (X-k+1)\bigr] \\
&= k\,X(X-1)\cdots(X-k+2),
\end{align*}
dus $\Delta B_k = \frac{X(X-1)\cdots(X-k+2)}{(k-1)!} = B_{k-1}$.

\textbf{13.} Schrijf $P = \sum_{k=0}^{n} c_k B_k$ (\omterm{def:b1:vspaces:free}{basis}, vraag
12). Pas $\Delta^{j}$ toe: door vraag 12, $\Delta^{j} P = \sum_{k
\geq j} c_k B_{k-j}$. Evalueer in $0$: $B_0(0) = 1$ en $B_m(0) = 0$
voor $m \geq 1$ (de factor $X$ verdwijnt), dus
$\bigl(\Delta^{j}P\bigr)(0) = c_j$. Dit is de
voorwaartse-verschilformule.

\textbf{14.} Inductie op $k$. Voor $k = 0$ luidt de identiteit
$P(0) = P(0)$. Neem haar aan voor $k$ en pas haar toe op $\Delta P$:
\[
\bigl(\Delta^{k+1} P\bigr)(0)
= \sum_{j=0}^{k} (-1)^{k-j}\binom kj \bigl(P(j+1) - P(j)\bigr).
\]
Verzamel de coĂ«fficiĂ«nt van $P(i)$: die is $(-1)^{k-i+1}\binom
k{i-1}\cdot(-1)^{0}$ uit de eerste som (verschoven) en
$-(-1)^{k-i}\binom ki$ uit de tweede --- samen
\[
(-1)^{k+1-i}\Bigl(\binom k{i-1} + \binom ki\Bigr)
= (-1)^{k+1-i}\binom{k+1}i
\]
door Pascals regel, wat de identiteit op rang $k + 1$ is.

\textbf{15.} Vraag 11 itererend vanaf graad $n$, leidende
coĂ«fficiĂ«nt $a_n$: na Ă©Ă©n $\Delta$, graad $n-1$ en leidende
coĂ«fficiĂ«nt $n a_n$; na twee, $n(n-1)a_n$; na $n$ stappen, graad
$0$ en waarde $n(n-1)\cdots 1\, a_n = n!\,a_n$, een constante. Nog
Ă©Ă©n $\Delta$ doodt ze: $\Delta^{n+1}P = 0$.

\textbf{16.} Als $m \geq k$: $B_k(m) = \binom mk \in \N$. Als $0
\leq m < k$: Ă©Ă©n factor van $m(m-1)\cdots(m-k+1)$ is nul, dus
$B_k(m) = 0$. Als $m = -q$ met $q \geq 1$:
\[
B_k(-q) = \frac{(-q)(-q-1)\cdots(-q-k+1)}{k!}
= (-1)^k\,\frac{q(q+1)\cdots(q+k-1)}{k!}
= (-1)^k \binom{q+k-1}{k},
\]
een geheel getal. Dus beeldt elke $B_k$ $\Z$ in $\Z$ af.

\textbf{17.} ($\Leftarrow$) Als $P = \sum_k c_k B_k$ met $c_k \in
\Z$, dan voor $m \in \Z$, $P(m) = \sum_k c_k B_k(m) \in \Z$ door
vraag 16. ($\Rightarrow$) Als $P$ geheelwaardig is, zijn zijn
\omterm{prop:b1:vspaces:coordinates}{coĂ¶rdinaten} $c_k = \bigl(\Delta^k P\bigr)(0) = \sum_{j=0}^k
(-1)^{k-j}\binom kj P(j)$ (vragen 13 en 14), een gehele combinatie
van de gehele getallen $P(0), \dots, P(k)$. Dit is PĂłlya's
karakterisatie van \omterm{pb:b1:vspaces:1}{geheelwaardige veeltermen}.

\textbf{18.} Stel $Q(X) = P(X + a)$, een \omterm{def:b1:poly:def}{veelterm} van graad $\leq
n$ met $Q(0), Q(1), \dots, Q(n) \in \Z$. Zijn \omterm{prop:b1:vspaces:coordinates}{coĂ¶rdinaten} in
$(B_k)_{k \leq n}$ zijn $c_k = \sum_{j \leq k}(-1)^{k-j}\binom kj
Q(j) \in \Z$ (vraag 14 gebruikt alleen de waarden in $0, \dots, k
\leq n$). Door vraag 17 ($\Leftarrow$) is $Q$ geheelwaardig op heel
$\Z$, dus ook $P(X) = Q(X - a)$.

\textbf{19.} Een product van $k$ opeenvolgende gehele getallen is
$m(m-1) \cdots(m-k+1) = k!\,B_k(m)$ voor een zekere $m \in \Z$, en
$B_k(m) \in \Z$ door vraag 16: het product is \omterm{def:b1:arith:divides}{deelbaar} door $k!$.

\textbf{20.} Waarden van $P = \frac{X(X+1)(2X+1)}{6}$ in $0,1,2,3$:
$0, 1, 5, 14$. Verschiltabel: $\Delta$-rij $1, 4, 9$; $\Delta^2$-rij
$3, 5$; $\Delta^3$-rij $2$. Dus, door vraag 13,
\[
P = 0\cdot B_0 + 1\cdot B_1 + 3\,B_2 + 2\,B_3 ,
\]
met gehele \omterm{prop:b1:vspaces:coordinates}{coĂ¶rdinaten}: $P$ is geheelwaardig (vraag 17), terwijl
zijn monomiale coĂ«fficiĂ«nten $\frac13, \frac12, \frac16$ geen gehele
getallen zijn. Directe berekening:
\begin{align*}
\Delta P &= \frac{(X+1)(X+2)(2X+3) - X(X+1)(2X+1)}{6} \\
&= \frac{(X+1)\bigl[(X+2)(2X+3) - X(2X+1)\bigr]}{6}
= \frac{(X+1)(6X+6)}{6} = (X+1)^2 .
\end{align*}
Telescoperend $P(m) = \sum_{j=0}^{m-1}\Delta P(j) = \sum_{j=1}^{m}
j^2$ (met $P(0) = 0$): de som-van-kwadraten-formule.

\textbf{21.} De waarden $2^i$ in $i = 0, \dots, n$ hebben een
verschiltabel die constant $1$ is op de linkerrand: $\Delta^k$ van
de rij $(2^i)$ is opnieuw $(2^i)$ (aangezien $2^{i+1} - 2^i = 2^i$),
dus $\bigl(\Delta^k P\bigr)(0) = 2^0 = 1$ voor alle $k \leq n$, en
$P = B_0 + B_1 + \dots + B_n$ door vraag 13. Dan
\[
P(n+1) = \sum_{k=0}^{n}\binom{n+1}{k}
= 2^{n+1} - \binom{n+1}{n+1} = 2^{n+1} - 1 \neq 2^{n+1}:
\]
het patroon breekt bij het eerste ongecontroleerde punt.

\textbf{22.} Door vraag 12, $B_k = \Delta B_{k+1}$, dus
\[
\sum_{j=0}^{m-1} B_k(j)
= \sum_{j=0}^{m-1}\bigl(B_{k+1}(j+1) - B_{k+1}(j)\bigr)
= B_{k+1}(m) - B_{k+1}(0) = B_{k+1}(m).
\]
Voor $j < k$ verdwijnen de termen $B_k(j)$, dus de som begint
werkelijk bij $j = k$: $\sum_{j=k}^{m-1}\binom jk = \binom m{k+1}$,
de hockeystick-identiteit.

\textbf{23.} Verschiltabellen (of directe ontwikkeling) geven
\[
X^2 = B_1 + 2 B_2, \qquad X^3 = B_1 + 6 B_2 + 6 B_3
\]
(controle: $B_1 + 2B_2 = X + X(X-1) = X^2$; in $X = 1, 2, 3$ geeft
de tweede $1, 8, 27$). Vraag 22 levert dan
\[
\sum_{j=0}^{m-1} j^2 = B_2(m) + 2B_3(m)
= \binom m2 + 2\binom m3 = \frac{m(m-1)(2m-1)}{6},
\]
\[
\sum_{j=0}^{m-1} j^3 = B_2(m) + 6B_3(m) + 6B_4(m)
= \binom m2 + 6\binom m3 + 6\binom m4 .
\]
De laatste uitdrukking ontwikkelend: $\binom m2 + 6\binom m3 +
6\binom m4 = \frac{m(m-1)}{2}\bigl[1 + 2(m-2) +
\frac{(m-2)(m-3)}{2}\bigr] = \frac{m^2(m-1)^2}{4} = \binom m2^2$.
$m$ vervangend door $m + 1$: $1^3 + \dots + m^3 =
\bigl(\frac{m(m+1)}2\bigr)^2 = (1 + \dots + m)^2$, Nicomachus'
identiteit.

\textbf{24.} Knopen $0, 1, 2$, \omterm{def:b1:poly:def}{veelterm} $X^2$. Monomiale \omterm{def:b1:vspaces:free}{basis}
$(1, X, X^2)$: \omterm{prop:b1:vspaces:coordinates}{coĂ¶rdinaten} $(0, 0, 1)$. \omterm{def:b1:vspaces:free}{Lagrange-basis}:
\omterm{prop:b1:vspaces:coordinates}{coĂ¶rdinaten} zijn de waarden $(0, 1, 4)$ (vraag 4). \omterm{def:b1:vspaces:free}{Newton-basis}
$(1, X, X(X-1))$: \omterm{pb:b1:vspaces:1}{gedeelde differenties} $f[0] = 0$, $f[0,1] = 1$,
$f[0,1,2] = \frac{3 - 1}{2} = 1$ (vraag 9), dus \omterm{prop:b1:vspaces:coordinates}{coĂ¶rdinaten} $(0, 1,
1)$ --- inderdaad $X + X(X-1) = X^2$. Drie bases, drie
coĂ¶rdinaatvectoren, Ă©Ă©n \omterm{def:b1:poly:def}{veelterm}.

\textbf{25.} (i) Vraag 4 is automatisch omdat $(L_i)$ een
\emph{\omterm{def:b1:vspaces:free}{basis}} is: bestaan en uniciteit van interpolatie zijn precies
bestaan en uniciteit van \omterm{prop:b1:vspaces:coordinates}{coĂ¶rdinaten}. (ii) De \omterm{def:b1:vspaces:free}{Newton-basis} is een

--- elke nieuwe knoop voegt Ă©Ă©n term toe zonder de vorige te
