\section*{Hoofdstuk \ref{ch:b1:topology} --- Topologie van de reële rechte}

\begin{solution}{exo:b1:topology:1}
$\intoo{0}{1} \cup \intoo{2}{3}$: \omterm{def:b1:topology:open}{open} (unie van \omterm{def:b1:topology:open}{open verzamelingen}),
niet \omterm{def:b1:topology:closed}{gesloten} ($\frac 1n \to 0$ erbuiten).

$\intco{0}{1}$: geen van beide. Niet \omterm{def:b1:topology:open}{open} (geen \omterm{prop:b1:reals:intervals}{interval} rond $0$
binnenin); niet \omterm{def:b1:topology:closed}{gesloten} ($1 - \frac1n \to 1 \notin$ \omterm{def:b1:logic:sets}{verzameling}).

$\{0\} \cup \intcc{1}{2}$: \omterm{def:b1:topology:closed}{gesloten} (eindige unie van \omterm{def:b1:topology:closed}{gesloten
verzamelingen}), niet \omterm{def:b1:topology:open}{open} (faalt in $0$).

$\R \setminus \Z$: \omterm{def:b1:topology:open}{open} ($\Z$ is \omterm{def:b1:topology:closed}{gesloten}), niet \omterm{def:b1:topology:closed}{gesloten}: de rij
$\bigl(\frac 1n\bigr)$ ligt erin, maar haar limiet $0$ behoort tot
$\Z$, d.w.z.\ ontsnapt aan de \omterm{def:b1:logic:sets}{verzameling}.

$\Q \cap \intoo{0}{1}$: geen van beide. Niet \omterm{def:b1:topology:open}{open}: elk \omterm{prop:b1:reals:intervals}{interval} rond
een rationaal getal bevat irrationale getallen. Niet \omterm{def:b1:topology:closed}{gesloten}: het
bevat rijen die streven naar het irrationale $\frac{\sqrt 2}{2}$
(\omterm{def:b1:topology:dense}{dichtheid}).
\end{solution}

\begin{solution}{exo:b1:topology:2}
$A = \intoc{0}{1} \cup \{2\}$: $\mathring A = \intoo{0}{1}$,
$\overline A = \intcc{0}{1} \cup \{2\}$, $\partial A = \{0, 1, 2\}$.

$A = \R \setminus \Q$: $\mathring A = \emptyset$ (elk \omterm{prop:b1:reals:intervals}{interval} bevat
rationale getallen), $\overline A = \R$ (\omterm{def:b1:topology:dense}{dichtheid} van de irrationale
getallen), $\partial A = \R$.

$A = \bigl\{\frac{(-1)^n n}{n+1}\bigr\}$: de even termen streven naar
$1$, de oneven termen naar $-1$, geen van beide behoort tot $A$.
$\mathring A = \emptyset$ (geïsoleerde punten), $\overline A = A \cup
\{-1, 1\}$, $\partial A = \overline A$.
\end{solution}

\begin{solution}{exo:b1:topology:3}
\emph{Complementen.} $F = \{a_1 < a_2 < \dots < a_k\}$: het complement
is de unie van de \omterm{def:b1:topology:open}{open} intervallen $\intoo{-\infty}{a_1}$,
$\intoo{a_i}{a_{i+1}}$, $\intoo{a_k}{+\infty}$ --- \omterm{def:b1:topology:open}{open} volgens
\cref{prop:b1:topology:openstable}.

\emph{Rijen.} Zij $u_n \in F$, $u_n \to \ell$. Met $\varepsilon =
\min\{\abs{a_i - a_j} : i \neq j\}/2 > 0$ (of eender welke
$\varepsilon$ als $F$ een singleton is): vanaf een zekere rang liggen
alle termen binnen $\varepsilon$ van $\ell$, dus binnen $2\varepsilon$
van elkaar, wat hen dwingt vanaf die rang één enkele $a_i$ te zijn;
dan $\ell = a_i \in F$.
\end{solution}

\begin{solution}{exo:b1:topology:4}
$\subseteq$: $\overline A \cup \overline B$ is \omterm{def:b1:topology:closed}{gesloten} (eindige unie)
en bevat $A \cup B$, dus bevat het de \emph{kleinste} \omterm{def:b1:topology:closed}{gesloten}
superverzameling $\overline{A \cup B}$. $\supseteq$: $A \subseteq A
\cup B$ geeft $\overline A \subseteq \overline{A \cup B}$ (\omterm{def:b1:topology:closure}{afsluiting}
is monotoon: aanklevende punten van $A$ zijn aanklevend aan de grotere
\omterm{def:b1:logic:sets}{verzameling}), en evenzo voor $B$.

Tegenvoorbeeld voor doorsneden: $A = \intoo{0}{1}$, $B =
\intoo{1}{2}$: $\overline{A \cap B} = \overline\emptyset = \emptyset$
maar $\overline A \cap \overline B = \{1\}$.
\end{solution}

\begin{solution}{exo:b1:topology:5}
Gebruik de sequentiële karakterisering
(\cref{thm:b1:topology:seqclosed}). Zij $S = \{u_n\} \cup \{\ell\}$ en
$(v_k)$ een rij in $S$ met $v_k \to m$; toon $m \in S$. Twee gevallen.
Als een zekere waarde $v \in S$ oneindig vaak door $(v_k)$ wordt
aangenomen, dan geeft een constante \omterm{def:b1:seq:subsequence}{deelrij} $m = v \in S$. Anders
wordt elke waarde eindig vaak aangenomen; in het bijzonder verschijnt
voor elke $n$ de term $u_n$ eindig vaak, en $\ell$ ook. Dan zijn voor
elke $N$ de indices $k$ met $v_k \in \{u_0, \dots, u_N, \ell\}$ eindig
in aantal: de overblijvende $v_k$ zijn termen $u_n$ met $n > N$.
Gegeven $\varepsilon > 0$, kies $N$ met $\abs{u_n - \ell} \leq
\varepsilon$ voor $n > N$: op eindig veel na voldoen alle $v_k$ aan
$\abs{v_k - \ell} \leq \varepsilon$. Dus $v_k \to \ell$, zodat $m =
\ell \in S$.
\end{solution}

\begin{solution}{exo:b1:topology:6}
$U + A = \bigcup_{a \in A} (U + a)$, en elke translatie $U + a$ is
\omterm{def:b1:topology:open}{open} (de intervalcertificaten worden getransleerd). Een unie van \omterm{def:b1:topology:open}{open
verzamelingen} is \omterm{def:b1:topology:open}{open} (\cref{prop:b1:topology:openstable}).

\omterm{def:b1:topology:closed}{Gesloten verzamelingen}: $\Z$ en $\sqrt 2\,\Z$ zijn \omterm{def:b1:topology:closed}{gesloten} (als
$\alpha\Z$: convergente rijen zijn uiteindelijk constant, vgl.\
\cref{ex:b1:topology:closed}). Hun som $\Z + \sqrt 2\,\Z$ is \omterm{def:b1:topology:dense}{dicht} in
$\R$ (\cref{exo:b1:reals:9}) maar is niet $\R$ (ze is aftelbaar, of
eenvoudigweg: $\frac{\sqrt 2}{2} \notin \Z + \sqrt2\Z$, anders zou
$\sqrt 2$ rationaal zijn --- door $\frac{\sqrt2}{2} = m + n\sqrt 2$ te
schrijven dwing je $(2n - 1)\sqrt 2 = -2m$ af, dus $\sqrt 2 \in \Q$
tenzij $n = \frac12$, onmogelijk). Een dichte echte deelverzameling is
niet \omterm{def:b1:topology:closed}{gesloten}: haar \omterm{def:b1:topology:closure}{afsluiting} is $\R \neq$ zichzelf.
\end{solution}

\begin{solution}{exo:b1:topology:7}
$\Z$: de \omterm{def:b1:topology:open}{omgeving} $\intoo{n - \frac12}{n + \frac12}$ van $n$ ontmoet
$\Z$ enkel in $n$.

$A = \{\frac1n : n \in \N^*\}$: rond $\frac 1n$ isoleert het \omterm{prop:b1:reals:intervals}{interval}
van straal $\frac{1}{n} - \frac{1}{n+1} = \frac{1}{n(n+1)}$ (gehalveerd,
zeg) het van zijn buren --- alle punten geïsoleerd. Toch $0 \in
\overline A \setminus A$: geïsoleerde punten verhinderen aanklevende
buitenstaanders niet.

Als alle punten van $A$ geïsoleerd zijn: geen enkel punt van $A$ is
inwendig, aangezien een inwendig punt een heel \omterm{prop:b1:reals:intervals}{interval} van
$A$-buren eromheen heeft (een \omterm{prop:b1:reals:intervals}{interval} is oneindig), in tegenspraak
met isolatie. Dus $\mathring A = \emptyset$.
\end{solution}

\begin{solution}{exo:b1:topology:8}
Als $A \subseteq \intcc{-M}{M}$, bevat de \omterm{def:b1:topology:closed}{gesloten verzameling}
$\intcc{-M}{M}$ $A$, dus bevat ze $\overline A$ (kleinste \omterm{def:b1:topology:closed}{gesloten}
superverzameling): $\overline A$ is begrensd.

Zij $s = \sup A$ (eindig). Aangezien $A \subseteq \overline A$, $\sup
\overline A \geq s$. Omgekeerd, $\overline A \subseteq
\intoc{-\infty}{s}$: de halve rechte is \omterm{def:b1:topology:closed}{gesloten} en bevat $A$; dus elk
element van $\overline A$ is $\leq s$, wat $\sup\overline A \leq s$
geeft. Gelijkheid.

$s \in \overline A$: volgens de $\varepsilon$-karakterisering
(\cref{prop:b1:reals:epsilon}) bevat elk \omterm{prop:b1:reals:intervals}{interval} $\intoo{s -
\varepsilon}{s + \varepsilon}$ een element van $A$: $s$ is
aanklevend.
\end{solution}

\begin{solution}{exo:b1:topology:9}
Stel dat $A$ \omterm{def:b1:topology:open}{open} en \omterm{def:b1:topology:closed}{gesloten} is, met $a \in A$ en $b \in \R
\setminus A$; zonder verlies van algemeenheid $a < b$. De \omterm{def:b1:logic:sets}{verzameling}
$B = A \cap \intcc{a}{b}$ is niet-leeg ($a$), begrensd: zij $s = \sup
B$. Volgens \cref{exo:b1:topology:8}, $s \in \overline B \subseteq
\overline A = A$ ($A$ \omterm{def:b1:topology:closed}{gesloten}). Merk op $s \leq b$, en aangezien $b
\notin A$: $s < b$. Nu is $A$ \omterm{def:b1:topology:open}{open}: een zeker \omterm{prop:b1:reals:intervals}{interval} $\intoo{s -
r}{s + r}$ ligt in $A$, en we mogen $r < b - s$ nemen. Dan behoort $s
+ \frac{r}{2}$ tot $A \cap \intcc{a}{b} = B$ en overtreft $s$ --- in
tegenspraak met $s = \sup B$. Dus bestaat er geen zulk paar $(a, b)$:
één van $A$, $\R \setminus A$ is leeg.
\end{solution}

\begin{solution}{exo:b1:topology:10}
$I_x$ is een unie van \omterm{def:b1:topology:open}{open} intervallen die allemaal $x$ bevatten: ze
is \omterm{def:b1:topology:open}{open}, en ze is een \omterm{prop:b1:reals:intervals}{interval}, aangezien ze convex is --- als $u < z
< v$ met $u, v \in I_x$, dan liggen $u$ en $v$ in \omterm{def:b1:topology:open}{open}
deelintervallen $J_u \ni x$, $J_v \ni x$ van $U$, en $J_u \cup J_v$ is
een \omterm{prop:b1:reals:intervals}{interval} (beide bevatten $x$) binnen $U$ dat $z$ bevat; dus $z \in
I_x$ (\cref{prop:b1:reals:intervals}).

Als $I_x \cap I_y \neq \emptyset$: $I_x \cup I_y$ is dan een \omterm{def:b1:topology:open}{open}
\omterm{prop:b1:reals:intervals}{interval} (convex: twee overlappende intervallen) bevat in $U$ dat $x$
en $y$ bevat, dus $I_x \cup I_y \subseteq I_x$ en $\subseteq I_y$
volgens de maximaliteit van elk: $I_x = I_y$.

Dus $U$ is de disjuncte unie van de verschillende \omterm{def:b1:logic:sets}{verzamelingen} $I_x$
(elke $x \in U$ ligt in haar eigen $I_x$). Aftelbaarheid: elk
niet-leeg \omterm{def:b1:topology:open}{open} \omterm{prop:b1:reals:intervals}{interval} $I$ van de familie bevat een rationaal getal
$q_I$ (\cref{thm:b1:reals:density}), en verschillende disjuncte
intervallen krijgen verschillende rationale getallen: de familie
injecteert in $\Q$, dat aftelbaar is (het is geïndexeerd door paren
gehele getallen). Dus ten hoogste aftelbaar veel intervallen.
\end{solution}

\begin{solution}{exo:b1:topology:11}
$\overline A = A \cup A'$. ($\supseteq$) $A \subseteq \overline A$
altijd; en als $x \in A'$, ontmoet elke \omterm{def:b1:topology:open}{omgeving} van $x$ $A \setminus
\{x\} \subseteq A$, dus $x$ is aanklevend. ($\subseteq$) Zij $x \in
\overline A$. Als $x \in A$, klaar. Als $x \notin A$, ontmoet elke
\omterm{def:b1:topology:open}{omgeving} van $x$ $A = A \setminus \{x\}$: $x \in A'$.

Bijgevolg is $A$ \omterm{def:b1:topology:closed}{gesloten} $\iff$ $A = \overline A = A \cup A'$ $\iff$
$A' \subseteq A$.

$A = \{\frac 1n\}$: $0$ is een ophopingspunt ($\frac 1n \to 0$, termen
$\neq 0$); elke $\frac 1n$ is geïsoleerd (\cref{exo:b1:topology:7}),
dus niet in $A'$; en een punt $x \notin A \cup \{0\}$ heeft een heel
\omterm{prop:b1:reals:intervals}{interval} dat $A$ vermijdt (tussen de twee buren van $x$ in $A \cup
\{0\}$, of voorbij $1$). Dus $A' = \{0\}$.

$\Z' = \emptyset$: elk geheel getal is geïsoleerd, elk niet-geheel
getal heeft een \omterm{def:b1:topology:open}{omgeving} binnen $\R \setminus \Z$.

$\Q' = \R$: elk \omterm{prop:b1:reals:intervals}{interval} rond eender welk reëel getal bevat oneindig
veel rationale getallen (\cref{thm:b1:reals:density}), in het
bijzonder eentje verschillend van het centrum.
\end{solution}

\begin{solution}{exo:b1:topology:12}
\begin{enumerate}
  \item Voor elke $a \in A$: $\abs{x - a} \leq \abs{x - y} + \abs{y -
        a}$, dus $d_A(x) \leq \abs{x - y} + \abs{y - a}$; door het
        \omterm{def:b1:reals:bounds}{infimum} over $a$ te nemen: $d_A(x) \leq \abs{x - y} + d_A(y)$.
        Door $x$ en $y$ te verwisselen krijg je de andere
        ongelijkheid: $\abs{d_A(x) - d_A(y)} \leq \abs{x - y}$.
  \item $d_A(x) = 0$ $\iff$ voor elke $\varepsilon > 0$ is er een $a
        \in A$ met $\abs{x - a} < \varepsilon$ $\iff$ elk \omterm{prop:b1:reals:intervals}{interval} rond
        $x$ ontmoet $A$ $\iff$ $x \in \overline A$. Als $F$ \omterm{def:b1:topology:closed}{gesloten} is
        en $x \notin F = \overline F$, dan $d_F(x) \neq 0$, d.w.z.\
        $d_F(x) > 0$.
  \item Als $d_F(x) < \varepsilon$, stel $r = \varepsilon - d_F(x) >
        0$: voor $\abs{y - x} < r$ geeft deel (1) $d_F(y) \leq d_F(x) +
        \abs{x - y} < \varepsilon$: het \omterm{prop:b1:reals:intervals}{interval} $\intoo{x - r}{x + r}$
        ligt in $V_\varepsilon$, dat dus \omterm{def:b1:topology:open}{open} is; het bevat $F$
        aangezien $d_F = 0$ daar. Ten slotte $x \in
        \bigcap_{\varepsilon>0} V_\varepsilon$ $\iff$ $d_F(x) <
        \varepsilon$ voor alle $\varepsilon$ $\iff$ $d_F(x) = 0$ $\iff$
        $x \in \overline F = F$. Aangezien $\bigcap_{\varepsilon > 0}
        V_\varepsilon = \bigcap_{n \geq 1} V_{1/n}$, is elke \omterm{def:b1:topology:closed}{gesloten
        verzameling} een aftelbare doorsnede van \omterm{def:b1:topology:open}{open verzamelingen}.
\end{enumerate}
\end{solution}

\begin{solution}{pb:b1:topology:1}
\textbf{1.} $C_1 = \intcc{0}{\frac13} \cup \intcc{\frac23}{1}$ en
\[
C_2 = \intcc{0}{\tfrac19} \cup \intcc{\tfrac29}{\tfrac13}
\cup \intcc{\tfrac23}{\tfrac79} \cup \intcc{\tfrac89}{1} .
\]
Inductie: als $C_n$ een disjuncte unie is van $2^n$ \omterm{def:b1:topology:closed}{gesloten} segmenten
van lengte $3^{-n}$, laat het verwijderen van de \omterm{def:b1:topology:open}{open} middelste derde
van elk twee \omterm{def:b1:topology:closed}{gesloten} segmenten van lengte $3^{-n-1}$ per ouder over:
$2^{n+1}$ segmenten, paarsgewijs disjunct (kinderen van
verschillende ouders zijn gescheiden omdat de ouders het waren;
kinderen van één ouder zijn gescheiden door het verwijderde gat).

\textbf{2.} Elke $C_n$ is een eindige unie van segmenten, dus
\omterm{def:b1:topology:closed}{gesloten}; $C = \bigcap C_n$ is een doorsnede van \omterm{def:b1:topology:closed}{gesloten
verzamelingen}: \omterm{def:b1:topology:closed}{gesloten} (\cref{def:b1:topology:closed}); begrensd
($\subseteq \intcc{0}{1}$): compact volgens
\cref{thm:b1:topology:compact}. Niet-leeg: $0$ ligt in het meest
linkse segment van elke $C_n$. Zij $a$ een eindpunt van een segment
$S$ van $C_n$. Voor $m \leq n$, $a \in C_n \subseteq C_m$. Voor de
latere stadia: de middelste-derde-verwijdering verwijdert nooit een
eindpunt, en $a$ is opnieuw een eindpunt van een van de twee kinderen
van $S$ (het kind dat $a$ raakt); door inductie $a \in C_m$ voor alle
$m \geq n$: $a \in C$.

\textbf{3.} Totale lengte van $C_n$: $2^n \cdot 3^{-n} = (\frac23)^n
\to 0$. Gegeven $\varepsilon > 0$, kies $n$ met $(\frac23)^n \leq
\varepsilon$: dan $C \subseteq C_n$, een unie van eindig veel
segmenten van totale lengte $\leq \varepsilon$.

\textbf{4.} Zij $I \subseteq C$ een \omterm{prop:b1:reals:intervals}{interval} met twee verschillende
punten. Voor elke $n$: $I \subseteq C_n$, en $I$, convex zijnd, moet
binnen een \emph{enkel} segment van $C_n$ liggen (het ontmoeten van
twee segmenten zou $I$ dwingen een punt van het gat ertussen te
bevatten, dat buiten $C_n$ ligt). Dus de lengte van $I$ is $\leq
3^{-n}$ voor alle $n$: tegenspraak. Dus de enige intervallen binnen
$C$ zijn leeg of singletons; in het bijzonder past geen enkele
$\intoo{x-r}{x+r}$ binnen $C$: $\mathring C = \emptyset$. Aangezien
$C$ \omterm{def:b1:topology:closed}{gesloten} is, heeft $\overline C = C$ leeg \omterm{def:b1:topology:closure}{inwendige}: $C$ is
nergens \omterm{def:b1:topology:dense}{dicht}.

\textbf{5.} Schrijf $\varphi_0(x) = \frac x3$ en $\varphi_2(x) =
\frac{2 + x}{3}$, stijgende affiene bijecties van $\intcc{0}{1}$ op
$\intcc{0}{\frac13}$ en $\intcc{\frac23}{1}$. Bewering: $C_{n+1} =
\varphi_0(C_n) \cup \varphi_2(C_n)$. Voor $n = 0$ is dit vraag 1.
Inductie: een stijgende affiene \omterm{def:b1:logic:map}{afbeelding} stuurt de middelste derde
van een segment naar de middelste derde van het beeldsegment, dus het
verwijderen van middelste derden commuteert met $\varphi_0$ en
$\varphi_2$; het toepassen van de verwijderingsstap op $C_{n+1} =
\varphi_0(C_n) \cup \varphi_2(C_n)$ levert $C_{n+2} =
\varphi_0(C_{n+1}) \cup \varphi_2(C_{n+1})$. Door over $n$ te
doorsnijden: voor $x \leq \frac13$, $x \in C \iff x \in
\varphi_0(C_n)$ voor alle $n$ $\iff 3x \in \bigcap C_n = C$; evenzo op
$\intcc{\frac23}{1}$; en geen punt van $\intoo{\frac13}{\frac23}$ ligt
in $C_1$. Dus $C = \varphi_0(C) \cup \varphi_2(C)$, disjunct.

\textbf{6.} Inductie op $n$; het geval $n = 0$ zegt dat elke $x \in
\intcc{0}{1}$ een code heeft, wat \cref{pb:b1:reals:1} is (vraag 9
voor $x < 1$; $1 = (0.\overline 2)_3$). Veronderstel de equivalentie
op rang $n$. Als $x \in C_{n+1}$: volgens vraag 5, $x = \varphi_i(z)$
met $z \in C_n$ en $i \in \{0, 2\}$; als $(e_k)$ een code van $z$ is
met de eerste $n$ cijfers $1$-vrij, dan heeft $(i, e_1, e_2, \dots)$
partiële sommen $\frac i3 + \frac13\sum_{k\leq m} e_k 3^{-k} \to
\varphi_i(z) = x$: een code van $x$ met de eerste $n + 1$ cijfers
$1$-vrij. Omgekeerd, als $x$ een code $(d_k)$ heeft met $d_1, \dots,
d_{n+1} \in \{0, 2\}$: de verschoven rij $(d_2, d_3, \dots)$ heeft een
zekere waarde $z \in \intcc{0}{1}$, haar eerste $n$ cijfers zijn
$1$-vrij, en de partiële-som-berekening achterwaarts gelezen geeft $x
= \varphi_{d_1}(z)$; door inductie $z \in C_n$, dus $x \in C_{n+1}$
volgens vraag 5. Ten slotte: een volledig $1$-vrije code plaatst $x$
in elke $C_n$, dus in $C$; omgekeerd, als $x \in C$, dan heeft voor
elke $n$ een van de ten hoogste twee codes van $x$
(\cref{pb:b1:reals:1}, vraag 11) haar eerste $n$ cijfers $1$-vrij; één
vaste code moet werken voor willekeurig grote $n$ (duiventil tussen
twee codes), en een code waarvan de eerste $n$ cijfers $1$-vrij zijn
voor willekeurig grote $n$ is $1$-vrij zonder meer.

\textbf{7.} $0 = (0.\overline 0)_3$, $1 = (0.\overline 2)_3$,
$\frac13 = (0.0\overline{2})_3$ (de oneigenlijke tweeling van
$(0.1)_3$), $\frac23 = (0.2\overline{0})_3$. Meetkundige sommen:
\[
(0.\overline{02})_3 = \sum_{j\geq1} \frac{2}{9^{\,j}}
= \frac{2/9}{1 - 1/9} = \frac14 ,
\qquad
(0.\overline{20})_3 = \sum_{j\geq1} \frac{2}{3\cdot 9^{\,j-1}}
= \frac{2/3}{1 - 1/9} = \frac34 ,
\]
beide $1$-vrij: $\frac14, \frac34 \in C$. (De oneindige sommen korten
suprema van partiële sommen af, zoals in \cref{pb:b1:reals:1}.)
Eindpunten van segmenten van $C_n$ zijn van de vorm $m/3^n$ (inductie:
kindeindpunten zijn oudereindpunten of verschillen van een ervan met
een veelvoud van $3^{-n-1}$). Als $\frac14 = \frac{m}{3^n}$ dan $3^n =
4m$, en $4 \nmid 3^n$: onmogelijk. Dus $\frac14 \in C$ zonder ooit een
eindpunt te zijn.

\textbf{8.} Veronderstel dat $x$ twee verschillende $1$-vrije codes
had. Überhaupt twee codes hebben betekent (\cref{pb:b1:reals:1}, vraag
11, grondtal $3$) dat $x = m/3^N \in \intoo{0}{1}$ en de twee codes
zijn: de afbrekende, met laatste niet-nul-cijfer $d_N \in \{1, 2\}$
gevolgd door $0$'s, en zijn tweeling, met $d_N - 1$ op positie $N$
gevolgd door $2$'s. Als $d_N = 1$ bevat de eerste een $1$; als $d_N =
2$ draagt de tweeling $d_N - 1 = 1$. Hoe dan ook is ten hoogste één
van het paar $1$-vrij: tegenspraak. Dus elke $x \in C$ heeft precies
één $1$-vrije code (bestaan volgens vraag 6), en verschillende
$\{0,2\}$-rijen hebben verschillende waarden. Elke $\{0,2\}$-rij heeft
een waarde in $\intcc{0}{1}$ (partiële sommen $\leq 1$) met alle
prefixen $1$-vrij, dus een waarde in elke $C_n$, d.w.z.\ in $C$: de
waardeafbeelding is een bijectie van $\{0,2\}$-rijen op $C$.

\textbf{9.} Zij $(d^{(k)})$ de $1$-vrije code van $x_k$ en stel $e_k =
2 - d^{(k)}_k \in \{0, 2\}$: een $\{0,2\}$-rij waarvan de waarde $y$
in $C$ ligt en $(e_k)$ als haar unieke $1$-vrije code heeft (vraag 8).
Voor elke $k$ verschillen de codes van $y$ en $x_k$ op positie $k$,
dus $y \neq x_k$: geen enkele \omterm{def:b1:logic:map}{afbeelding} $\N^* \to C$ is \omterm{def:b1:logic:inj}{surjectief}.
Een overaftelbare \omterm{def:b1:logic:sets}{verzameling} van lengte nul: grootheid in
\omterm{def:b1:counting:card}{kardinaliteit}, kleinheid in maat --- tegelijk.

\textbf{10.} Compactheid: geslotenheid van de oneindige doorsnede plus
begrensdheid (vraag 2) --- de ene eigenschap die niet door een enkele
inclusie gedragen wordt. Lengte nul: $C \subseteq C_n$ met totale
lengte $(\frac23)^n$ (vraag 3). Nergens \omterm{def:b1:topology:dense}{dichtheid}: $C \subseteq C_n$
dwingt intervallen binnen $C$ af om lengte $\leq 3^{-n}$ te hebben
(vraag 4). Overaftelbaarheid rijdt op helemaal geen inclusie: ze heeft
de volledige doorsnedestructuur nodig, gecodeerd in de bijectie van
vraag 8.

\textbf{11.} Keer $d_n$ om naar $2 - d_n$: de nieuwe rij is nog steeds
een $\{0,2\}$-rij, dus haar waarde $x_n$ ligt in $C$; de partiële
sommen voorbij rang $n$ verschillen met precies $2\cdot3^{-n}$, dus
$\abs{x_n - x} = 2\cdot3^{-n}$. Dus $x_n \neq x$ en $x_n \to x$: elk
punt van $C$ is een limiet van andere punten van $C$ --- $C$ is
\emph{perfect}, zonder geïsoleerd punt.

\textbf{12.} Volgens de inductie van vraag 5 zijn de segmenten van
$C_n$ precies de $\intcc{t}{t + 3^{-n}}$ waarbij $t$ de waarden van
lengte-$n$ $\{0,2\}$-rijen doorloopt. Gegeven $x \in C$ met code
$(d_k)$, is de afkapping $t_n$ (cijfers $d_1 \dots d_n$ dan $0$'s)
daarom een linker eindpunt, en $0 \leq x - t_n \leq 3^{-n}$:
eindpunten zijn \omterm{def:b1:topology:dense}{dicht} in $C$. Ze vormen een deelverzameling van
$\{m/3^n : m, n\}$, een \omterm{def:b1:logic:sets}{verzameling} geïndexeerd door paren gehele
getallen, dus aftelbaar (zoals voor $\Q$ in
\cref{exo:b1:topology:10}). Aangezien $C$ overaftelbaar is (vraag 9),
zijn op aftelbaar veel na alle punten van $C$ geen eindpunten ---
$\frac14$ (vraag 7) is de zichtbare top van die ijsberg.

\textbf{13.} Kies $n$ met $3^{-n} < y - x$. Beide $x, y \in C_n$, en
ze kunnen niet in hetzelfde segment liggen (lengte $3^{-n} < y - x$):
het verwijderde gat tussen hun segmenten levert $z$ met $x < z < y$ en
$z \notin C_n \supseteq C$. Dus elke twee punten van $C$ zijn
gescheiden door het complement: de enige convexe deelverzamelingen van
$C$ zijn singletons --- $C$ is volledig onsamenhangend.

\textbf{14.} Als $(d_k)$ de $1$-vrije code van $x$ is, is de rij $(2 -
d_k)$ opnieuw een $\{0,2\}$-rij, met partiële sommen
\[
\sum_{k=1}^{n} (2 - d_k)3^{-k} = (1 - 3^{-n}) - \sum_{k=1}^n
d_k 3^{-k} \longrightarrow 1 - x :
\]
dus $1 - x \in C$. Dus $1 - C \subseteq C$, en de \omterm{def:b1:logic:map}{afbeelding} tweemaal
toepassen geeft $1 - C = C$: de \omterm{pb:b1:topology:1}{Cantorverzameling} is symmetrisch rond
$\frac12$.

\textbf{15.} De partiële sommen $t_n \to x$ en $t'_n \to x'$
(stijgende rijen convergeren naar hun \omterm{def:b1:reals:bounds}{supremum}, d.w.z.\ de waarde),
dus $t_n + t'_n \to x + x'$ volgens \cref{thm:b1:seq:operations}.
Gegeven $y \in \intcc{0}{1}$ met code $(e_k)$, kies $(a_k, b_k) =
(0,0), (0,2), (2,2)$ naargelang $e_k = 0, 1, 2$: dan $a_k + b_k =
2e_k$, de rijen $(a_k)$, $(b_k)$ zijn $\{0,2\}$-rijen met waarden $x,
x' \in C$, en
\[
x + x' = \lim_n\,(t_n + t'_n) = \lim_n 2\sum_{k=1}^n e_k 3^{-k}
= 2y :
\]
elke $y \in \intcc{0}{1}$ is het middelpunt van twee punten van $C$.

\textbf{16.} Vraag 15 geeft $\intcc{0}{2} = 2\,\intcc{0}{1} \subseteq
C + C$, en $C + C \subseteq \intcc{0}{1} + \intcc{0}{1} =
\intcc{0}{2}$: gelijkheid. Dan, gebruikmakend van $1 - C = C$:
\[
C - C = C + (C - 1) = (C + C) - 1 = \intcc{-1}{1} .
\]
Concreet geval: $1 = \frac14 + \frac34$, een som van twee
niet-eindpuntleden van $C$.

\textbf{17.} Lengte meet hoeveel van de rechte de \omterm{def:b1:logic:sets}{verzameling} zelf
inneemt; ze zegt niets over de \omterm{def:b1:logic:sets}{verzameling} van \emph{sommen}, die het
beeld is van de tweeparameterfamilie $C \times C$ onder $(x, x')
\mapsto x + x'$ --- de twee cijferrijen worden onafhankelijk gekozen,
en die vrijheid is precies wat $\intcc{0}{2}$ vult. Geen enkele
stelling begrenst de lengte van een somverzameling door de lengten van
de sommanden, en $C$ is het bewijs dat er geen kan bestaan.

\textbf{18.} Volgens \cref{pb:b1:reals:1} (vraag 18) is $x$ rationaal
desda zijn eigenlijke ontwikkeling uiteindelijk periodiek is. De
$1$-vrije code van $x \in C$ is ofwel die eigenlijke ontwikkeling of
de oneigenlijke tweeling van een afbrekende; een afbrekende rij en
zijn tweeling (uiteindelijk constante $2$'s) zijn beide uiteindelijk
periodiek, dus periodiciteit van de $1$-vrije code is equivalent met
rationaliteit van $x$. Staartdeling van $\frac1{13}$ in grondtal $3$
($r_0 = 1$): $3 = 13\cdot0 + 3$, $9 = 13\cdot0 + 9$, $27 = 13\cdot2 +
1$, en de rest keert terug naar $1$: cijfers $\overline{002}$, dus
$\frac1{13} = (0.\overline{002})_3$, $1$-vrij en periodiek: een
rationaal lid van $C$. (Controle: $\frac{2/27}{1 - 1/27} = \frac{2}{26}
= \frac1{13}$.)

\textbf{19.} De rij met $d_k = 2$ op de driehoekige posities $k =
\frac{j(j+1)}{2}$ en $0$ elders is een $\{0,2\}$-rij, dus haar waarde
$x^*$ behoort tot $C$ (vraag 8). Ze heeft oneindig veel $2$'s met
gaten $j + 1 \to \infty$ tussen opeenvolgende, dus ze is niet
uiteindelijk periodiek (een periode $T$ zou uiteindelijk $2$'s op
gaten $\leq T$ afdwingen: het groeiende-gaten-argument van
\cref{pb:b1:reals:1}, vraag 20); volgens vraag 18, $x^* \notin \Q$. En
volgens vraag 9 plus de aftelbaarheid van $\Q$ zijn op aftelbaar veel
na alle leden van $C$ irrationaal: $x^*$ is de norm, niet de
uitzondering.

\textbf{20.} Zij $y \in \intcc{0}{1}$: het heeft een binaire code
$(c_k)$ met $c_k \in \{0, 1\}$ (\cref{pb:b1:reals:1}, vraag 9,
grondtal $2$; $y = 1$ neemt de allen-$1$-rij). Dan is $(2c_k)$ een
$\{0,2\}$-rij, haar waarde $x$ ligt in $C$, en $h(x)$ is de waarde van
$(c_k)$, namelijk $y$: $h$ beeldt $C$ op $\intcc{0}{1}$ af. Als $C$
het beeld was van een \omterm{def:b1:logic:map}{afbeelding} uit $\N^*$, zou het samenstellen met
$h$ heel $\intco{0}{1}$ oplijsten, in tegenspraak met de
diagonaalstelling van \cref{pb:b1:reals:1} (vraag 22): $C$ is
overaftelbaar, opnieuw. Een \omterm{def:b1:logic:sets}{verzameling} van lengte nul die op een vol
segment surjecteert.

\textbf{21.} Volgens vraag 5 is $C$ de disjuncte unie van
$\varphi_0(C)$ en $\varphi_2(C)$, elk een translatie van de geschaalde
kopie $\frac13 C$. Additiviteit, schaling en translatie-invariantie
geven
\[
L(C) = L(\varphi_0(C)) + L(\varphi_2(C))
= \tfrac13 L(C) + \tfrac13 L(C) = \tfrac23\,L(C),
\]
dus $\frac13 L(C) = 0$: $L(C) = 0$. Zelfgelijkvormigheid alleen
veroordeelt $C$ tot lengte nul --- vraag 3 voerde het vonnis slechts
uit.

\textbf{22.} Een segment van lengte $l_n$ verliest een centraal
\omterm{prop:b1:reals:intervals}{interval} van lengte $4^{-(n+1)}$, waarbij twee segmenten van lengte
$l_{n+1} = \frac{l_n - 4^{-(n+1)}}{2}$ overblijven; vanaf $l_0 = 1$
bevestigt inductie $l_n = \frac{2^n + 1}{2\cdot4^n}$: inderdaad
$\frac12\Bigl(\frac{2^n+1}{2\cdot4^n} - \frac{1}{4^{n+1}}\Bigr) =
\frac{2(2^n + 1) - 1}{2\cdot4^{n+1}} = \frac{2^{n+1} +
1}{2\cdot4^{n+1}}$, en $l_n > 0$ altijd: de constructie verhongert
nooit. $K = \bigcap K_n$ is \omterm{def:b1:topology:closed}{gesloten} en begrensd, dus compact; een
\omterm{prop:b1:reals:intervals}{interval} binnen $K$ ligt in één segment van $K_n$, van lengte $l_n \to
0$: leeg \omterm{def:b1:topology:closure}{inwendige}. Verwijderde lengte: $\sum_{n\geq0} 2^n \cdot
4^{-(n+1)} = \frac14\sum_{n\geq0}\bigl(\frac12\bigr)^n = \frac12$, en
elke $K_n$ heeft totale lengte $2^n l_n = \frac{2^n + 1}{2^{n+1}} >
\frac12$. Zij nu eindig veel \omterm{def:b1:topology:open}{open} intervallen met unie $U \supseteq
K$. Volgens de \omterm{def:b1:logic:statement}{metgezel-uitspraak} van \cref{ex:b1:topology:nested}, $U
\supseteq K_n$ voor een zekere $n$; toegevend de additiviteit van
lengte op eindige unies van intervallen, is de totale lengte van de
overdekkende intervallen minstens die van $K_n$, die $\frac12$
overtreft. Dus $K$ is nergens \omterm{def:b1:topology:dense}{dicht}, en toch bestaat er geen goedkope
overdekking: topologische kleinheid (nergens \omterm{def:b1:topology:dense}{dicht}) en metrische
kleinheid (lengte nul) zijn werkelijk verschillende begrippen, en $K$
scheidt ze.

\textbf{23.} Bereiking: zij $d = d(x, F)$ en kies $a_k \in F$ met
$\abs{x - a_k} \leq d + \frac1k$: de $a_k$ zijn begrensd, dus
Bolzano--Weierstrass (\cref{thm:b1:seq:bw}) extraheert $a_{\varphi(k)}
\to a$, met $a \in F$ ($F$ \omterm{def:b1:topology:closed}{gesloten},
\cref{thm:b1:topology:seqclosed}) en $\abs{x - a} = \lim \abs{x -
a_{\varphi(k)}} = d$. Nu het maximum: als $y \in C$, $d(y, C) = 0$;
anders ligt $y$ in een gat verwijderd in een zeker stadium $n \geq 1$,
een \omterm{def:b1:topology:open}{open} \omterm{prop:b1:reals:intervals}{interval} van lengte $3^{-n}$ waarvan de twee eindpunten tot
$C$ behoren (vraag 2), dus $d(y, C) \leq \frac{3^{-n}}{2} \leq
\frac16$, met gelijkheid die $n = 1$ vereist en $y$ in het centrum van
het gat $\intoo{\frac13}{\frac23}$, d.w.z.\ $y = \frac12$; en inderdaad
$d\bigl(\frac12, C\bigr) = \frac16$ aangezien $C \cap
\intoo{\frac13}{\frac23} = \emptyset$ en $\frac13, \frac23 \in C$. Dus
$\max_{y\in\intcc{0}{1}} d(y, C) = \frac16$, bereikt precies in
$\frac12$.

\textbf{24.} De eindpunten vormen een aftelbare dichte
deelverzameling van $C$ (vraag 12): lijst ze op als een enkele rij
$(e_j)_{j\geq1}$, een rij in $C$. Haar deelrijlimieten liggen allemaal
in $C$ ($C$ \omterm{def:b1:topology:closed}{gesloten}). Omgekeerd, fixeer $x \in C$: voor elke $n$
hebben de segmenten van de $C_m$ die $x$ bevatten ($m \geq n$) hun
eindpunten binnen $3^{-m} \leq 3^{-n}$ van $x$, dus liggen oneindig
veel verschillende eindpunten binnen $3^{-n}$ van $x$; kies indices
$j_1 < j_2 < \dots$ met $\abs{e_{j_n} - x} \leq 3^{-n}$: een \omterm{def:b1:seq:subsequence}{deelrij}
die naar $x$ convergeert. Dus de \omterm{def:b1:logic:sets}{verzameling} van deelrijlimieten van
$(e_j)$ is precies $C$ --- één rij die zich bij overaftelbaar veel
punten ophoopt, het tegengestelde uiterste van een convergente rij,
waarvan de ophopingsverzameling een singleton is.

\textbf{25.} (i) De constructie verbruikte: stabiliteit van \omterm{def:b1:topology:closed}{gesloten
verzamelingen} onder willekeurige doorsnede (bestaan van $C$ als een
\omterm{def:b1:topology:closed}{gesloten verzameling}), de compactheidsstelling
\cref{thm:b1:topology:compact} (vragen 2, 22, 23), en de sequentiële
karakteriseringen van geslotenheid en aankleving (het
geneste-compacten-argument van \cref{ex:b1:topology:nested} en vraag
23). (ii) De vier koppelingen: lengte nul en toch overaftelbaar
(vragen 3, 9); \omterm{def:b1:topology:closed}{gesloten} en toch met leeg \omterm{def:b1:topology:closure}{inwendige} (vraag 4); perfect
--- geen geïsoleerd punt --- en toch volledig onsamenhangend (vragen
11, 13); verwaarloosbaar en toch met $C + C = \intcc{0}{2}$ (vraag
16). (iii) De surjectie $h$ van vraag 20, continu en niet-dalend
gemaakt, wordt de duiveltrap in de theorie van continue functies; en
in de maattheorie van het boekdeel van jaar 3 is $C$ de standaardgetuige
dat ``verwaarloosbaar'' niet ``aftelbaar'' betekent, met zijn dikke
neef (vraag 22) die ``nergens \omterm{def:b1:topology:dense}{dicht}'' van ``verwaarloosbaar'' scheidt.
\end{solution}
