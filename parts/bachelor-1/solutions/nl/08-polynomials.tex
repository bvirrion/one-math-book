\section*{Hoofdstuk \ref{ch:b1:poly} --- Veeltermen}

\begin{solution}{exo:b1:poly:1}
$X^5 - 1 = (X^2 + X + 1)(X^3 - X^2 + 1) + (-X - 2)$. Stappen: trek
$X^3 B$ af, dan $-X^2 B$, dan $B$; de rest $-X - 2$ heeft graad
$1 < 2$. \emph{Controle in $X = 1$:} $\;0 = 3 \times 1 + (-3)$.

$2X^4 + X^3 - X + 3 = (X^2 - 2)(2X^2 + X + 4) + (X + 11)$. \emph{Controle
in $X = 0$:} $\;3 = (-2)(4) + 11$.
\end{solution}

\begin{solution}{exo:b1:poly:2}
$X^2 + X + 1 = (X - j)(X - j^2)$ met $j = \eu^{2\iu\pi/3}$, $j^3 =
1$. Het \omterm{def:b1:arith:divides}{deelt} $Q_n = X^{2n} + X^n + 1$ dan en slechts dan als $j$ en
$j^2$ wortels van $Q_n$ zijn; aangezien $Q_n$ reële coëfficiënten
heeft, is $Q_n(j^2) = \conj{Q_n(j)}$, dus de voorwaarde is gewoon
$Q_n(j) = 0$. Nu hangt $Q_n(j) = j^{2n} + j^n + 1$ af van $n$ mod $3$:
\begin{itemize}
  \item $n \equiv 0$: $Q_n(j) = 1 + 1 + 1 = 3 \neq 0$;
  \item $n \equiv 1$: $Q_n(j) = j^2 + j + 1 = 0$;
  \item $n \equiv 2$: $Q_n(j) = j^4 + j^2 + 1 = j + j^2 + 1 = 0$.
\end{itemize}
Dus $X^2 + X + 1 \mid X^{2n} + X^n + 1$ precies wanneer $3 \nmid n$.
\end{solution}

\begin{solution}{exo:b1:poly:3}
Volgens \cref{prop:b1:poly:multiplicity} geldt $(X-1)^2 \mid P$ dan en
slechts dan als $P(1) = P'(1) = 0$:
\[
P(1) = 2 + a + b = 0, \qquad P'(1) = 4 + 3a + 2b = 0 .
\]
Oplossen: $b = -a - 2$ en $4 + 3a - 2a - 4 = a = 0$, dus $a = 0$, $b =
-2$: $P = X^4 - 2X^2 + 1 = (X^2 - 1)^2 = (X-1)^2 (X+1)^2$, wat de
reële ontbinding is.
\end{solution}

\begin{solution}{exo:b1:poly:4}
$X^3 - 1 = (X - 1)(X - j)(X - j^2)$ over $\C$ ($j = \eu^{2\iu\pi/3}$),
en $(X - 1)(X^2 + X + 1)$ over $\R$.

$X^4 + X^2 + 1 = (X^2 + X + 1)(X^2 - X + 1)$ over $\R$ (werk uit, of
merk op dat $X^4 + X^2 + 1 = (X^2+1)^2 - X^2$); over $\C$ splitst elke
kwadratische: wortels $j, j^2$ en $-j, -j^2$, d.w.z.\ $\eu^{\pm
2\iu\pi/3}, \eu^{\pm\iu\pi/3}$.

$X^6 - 1 = \prod_{k=0}^{5} (X - \eu^{\iu k\pi/3})$ over $\C$, en over
$\R$:
\[
X^6 - 1 = (X-1)(X+1)(X^2 + X + 1)(X^2 - X + 1),
\]
door de toegevoegde paren $\eu^{\pm 2\iu\pi/3}$ en $\eu^{\pm
\iu\pi/3}$ te groeperen.
\end{solution}

\begin{solution}{exo:b1:poly:5}
\begin{enumerate}
  \item Laat $p/q$ (laagste termen) een wortel zijn van de monische
        gehele \omterm{def:b1:poly:def}{veelterm} $X^3 + \dots + c_0$: het wegwerken van de
        noemers in $P(p/q) = 0$ geeft $p^3 = -q\,(\text{geheel
        getal})$, dus $q \mid p^3$; onderlinge ondeelbaarheid dwingt
        $q = \pm 1$: de wortel is een geheel getal $p$, en $p \mid
        c_0$ (isoleer $c_0$). Hier delen de kandidaten $6$: bij het
        toetsen is $P(1) = 0$, $P(2) = 0$, $P(3) = 0$. Dus $P =
        (X-1)(X-2)(X-3)$.
  \item Vieta: $s_1 = 6$, $s_2 = 11$, $s_3 = 6$. Som van de
        kwadraten: $s_1^2 - 2s_2 = 36 - 22 = 14 = 1 + 4 + 9$, zoals
        verwacht. Som van de inversen: $\frac{s_2}{s_3} =
        \frac{11}{6} = 1 + \frac12 + \frac13$, zoals verwacht.
\end{enumerate}
\end{solution}

\begin{solution}{exo:b1:poly:6}
Eerste delingsstap van het \omterm{met:b1:arith:euclid}{algoritme van Euclides}:
\[
(X + 1)(X^3 - X^2 + X - 1)
= X^4 - X^3 + X^2 - X + X^3 - X^2 + X - 1 = X^4 - 1 ,
\]
dus de deling van $X^4 - 1$ door $X^3 - X^2 + X - 1$ is exact
(quotiënt $X + 1$, rest $0$), en het algoritme stopt onmiddellijk:
\[
\gcd(X^4 - 1,\; X^3 - X^2 + X - 1) = X^3 - X^2 + X - 1
\]
(al \omterm{def:b1:poly:def}{monisch}). De Bézout-relatie is de triviale: $\gcd = 0
\cdot (X^4 - 1) + 1 \cdot (X^3 - X^2 + X - 1)$. Consistentiecontrole
door ontbinding: $X^3 - X^2 + X - 1 = (X - 1)(X^2 + 1)$, wat inderdaad
het product is van de gemeenschappelijke irreducibele factoren van
$X^4 - 1 = (X-1)(X+1)(X^2+1)$.
\end{solution}

\begin{solution}{exo:b1:poly:7}
Aangezien $P \geq 0$ op $\R$, hebben zijn reële wortels even
\omterm{def:b1:poly:derivative}{multipliciteit} (in een wortel van oneven \omterm{def:b1:poly:derivative}{multipliciteit} verandert $P$
van teken). Met \cref{cor:b1:poly:factorization} en paring schrijven
we
\[
P = c \prod_i (X - a_i)^{2k_i} \prod_j \bigl((X - z_j)(X - \conj
z_j)\bigr)^{n_j},
\]
met $c > 0$ (gedrag in $+\infty$). Laat
\[
S = \sqrt c\, \prod_i (X - a_i)^{k_i} \prod_j (X - z_j)^{n_j}
\in \C[X],
\]
zodat $P = S\,\conj S$ waarbij $\conj S$ de toegevoegde coëfficiënten
heeft. Splits $S = A + \iu B$ met $A, B \in \R[X]$: dan
\[
P = (A + \iu B)(A - \iu B) = A^2 + B^2 .
\]
\end{solution}

\begin{solution}{exo:b1:poly:8}
Lagrange (\cref{thm:b1:poly:lagrange}) met knopen $0, 1, 2$:
\[
P = 1\cdot\frac{(X-1)(X-2)}{(0-1)(0-2)} + 3\cdot\frac{X(X-2)}{1\cdot(1-2)}
+ 2\cdot\frac{X(X-1)}{2\cdot 1}
= \frac{(X-1)(X-2)}{2} - 3X(X-2) + X(X-1).
\]
Uitwerken: $\frac{X^2 - 3X + 2}{2} - 3X^2 + 6X + X^2 - X =
-\frac{3}{2}X^2 + \frac{7}{2}X + 1$.

Stelsel: $P = aX^2 + bX + c$ met $c = 1$; $a + b + 1 = 3$; $4a + 2b +
1 = 2$. Twee keer de tweede van de derde aftrekken: $2a - 1 = -4$,
dus $a = -\frac32$, $b = \frac72$. Dezelfde \omterm{def:b1:poly:def}{veelterm}:
$P = -\frac32 X^2 + \frac72 X + 1$. (Controle $P(2) = -6 + 7 + 1 = 2$.)
\end{solution}

\begin{solution}{exo:b1:poly:9}
$P = X^{2n+1} - 1$: $P' = (2n+1)X^{2n} \geq 0$, dus de
veeltermfunctie is stijgend (strikt behalve in $0$), met limieten
$\mp\infty$: ze verdwijnt precies één keer op $\R$ (in $x = 1$).

Laat $E_n = \sum_{k=0}^{n} \frac{X^k}{k!}$. Dan $E_n' = E_{n-1} = E_n
- \frac{X^n}{n!}$. Een meervoudige wortel $a$ zou voldoen aan $E_n(a)
= E_n'(a) = 0$ (\cref{prop:b1:poly:multiplicity}), dus
$\frac{a^n}{n!} = E_n(a) - E_n'(a) = 0$, dus $a = 0$; maar $E_n(0) = 1
\neq 0$. Geen meervoudige wortel.
\end{solution}

\begin{solution}{exo:b1:poly:10}
\begin{enumerate}
  \item Inductie (twee basisgevallen gelden). Met $\cos(n+1)\theta +
        \cos(n-1)\theta = 2\cos\theta\cos n\theta$:
        \[
        T_{n+1}(\cos\theta) = 2\cos\theta \cos n\theta -
        \cos(n-1)\theta = \cos(n+1)\theta .
        \]
  \item $T_n(\cos\theta) = 0$ dan en slechts dan als $n\theta \equiv
        \frac\pi2 \pmod \pi$: de getallen
        \[
        x_k = \cos\Bigl(\frac{(2k+1)\pi}{2n}\Bigr),
        \qquad k = 0, 1, \dots, n-1,
        \]
        zijn $n$ verschillende punten van $\intoo{-1}{1}$ (de hoeken
        liggen in $\intoo{0}{\pi}$ waar $\cos$ \omterm{def:b1:logic:inj}{injectief} is), alle
        wortels van $T_n$; aangezien $\deg T_n = n$ (uit de
        recurrentie, met leidende coëfficiënt $2^{n-1}$ voor $n \geq
        1$, door inductie), zijn dit \emph{alle} wortels, elk
        enkelvoudig.
  \item Voor $x = \cos\theta \in \intcc{-1}{1}$: $\abs{T_n(x)} =
        \abs{\cos n\theta} \leq 1$, met gelijkheid dan en slechts dan
        als $n\theta \equiv 0 \pmod\pi$, d.w.z.\ in de $n+1$ punten
        $y_k = \cos\frac{k\pi}{n}$, $k = 0, \dots, n$, waar $T_n(y_k)
        = (-1)^k$. (Deze equioscillatie maakt $2^{1-n}T_n$ tot de
        \omterm{def:b1:poly:def}{monische veelterm} van graad $n$ met de kleinste supnorm op
        $\intcc{-1}{1}$ --- bewezen in het weekendprobleem van dit
        hoofdstuk.)
\end{enumerate}
\end{solution}

\begin{solution}{exo:b1:poly:11}
Schrijf $P = c\prod_i (X - a_i)^{m_i}$. De productregel (uitgebreid
tot meerdere factoren) geeft
\[
P' = c\sum_{i} m_i (X - a_i)^{m_i - 1} \prod_{k \neq i} (X -
a_k)^{m_k},
\]
en delen door $P$: $\frac{P'}{P} = \sum_i \frac{m_i}{X - a_i}$
(als rationale functies, d.w.z.\ weg van de wortels).

Laat $w$ een wortel van $P'$ zijn. Als $w$ een van de $a_i$ is, ligt
het triviaal in het convexe omhulsel. Anders, bij evaluatie in $w$:
\[
0 = \sum_i \frac{m_i}{w - a_i}
= \sum_i m_i\, \frac{\conj w - \conj a_i}{\abs{w - a_i}^2} .
\]
Toevoegen: $\sum_i \lambda_i (w - a_i) = 0$ waarbij $\lambda_i =
\frac{m_i}{\abs{w - a_i}^2} > 0$. Dus
\[
w = \frac{\sum_i \lambda_i a_i}{\sum_i \lambda_i} :
\]
een convexe combinatie (positieve gewichten die na normalisatie tot
$1$ sommeren) van de wortels $a_i$. Dus elke wortel van $P'$ ligt in
het convexe omhulsel van de wortels van $P$.
\end{solution}

\begin{solution}{exo:b1:poly:12}
Sommeer de evaluaties van $(1 + X)^n$ in de drie derdemachtseenheidswortels:
\[
2^n + (1 + j)^n + (1 + j^2)^n
= \sum_{k=0}^n \binom nk\,\bigl(1 + j^k + j^{2k}\bigr)
= 3\sum_{k\,:\,3\mid k}\binom nk ,
\]
aangezien $1 + j^k + j^{2k}$ een meetkundige som is die gelijk is aan
$3$ wanneer $3 \mid k$ en aan $\frac{j^{3k} - 1}{j^k - 1} = 0$ anders.
Nu is $1 + j = \frac12 + \iu\frac{\sqrt3}2 = \eu^{\iu\pi/3}$ en $1 +
j^2 = \conj{1 + j} = \eu^{-\iu\pi/3}$, dus $(1+j)^n + (1+j^2)^n =
2\cos\frac{n\pi}3$ en
\[
\sum_{k\geq0}\binom n{3k} = \frac{2^n + 2\cos\frac{n\pi}3}{3} .
\]
Controles: $n = 3$: $\frac{8 + 2\cos\pi}3 = 2 = \binom30 + \binom33$;
$n = 6$: $\frac{64 + 2}3 = 22 = 1 + 20 + 1$.
\end{solution}

\begin{solution}{pb:b1:poly:1}
\textbf{1.} $T_2 = 2X^2 - 1$; $T_3 = 2X(2X^2 - 1) - X = 4X^3 -
3X$; $T_4 = 2X\,T_3 - T_2 = 8X^4 - 8X^2 + 1$; $T_5 = 2X\,T_4 - T_3
= 16X^5 - 20X^3 + 5X$. De identiteit $T_3(\cos\theta) = \cos3\theta$
is precies $\cos3\theta = 4\cos^3\theta - 3\cos\theta$ uit
\cref{ex:b1:complex:cos3}.

\textbf{2.} Waar voor $n = 1, 2$. Als $T_{n-1}$, $T_n$ graden
$n-1$, $n$ en leidende coëfficiënten $2^{n-2}$, $2^{n-1}$ hebben, dan
heeft $2X\,T_n$ graad $n+1$ en leidende coëfficiënt $2^n$, terwijl
$T_{n-1}$ lagere graad heeft: $T_{n+1}$ heeft graad $n + 1$, leidende
coëfficiënt $2^n$. Pariteit: als $T_{n-1}$ de pariteit van $n - 1$
heeft en $T_n$ die van $n$, dan hebben $2X\,T_n$ en $T_{n-1}$ beide
de pariteit van $n + 1$, dus ook $T_{n+1}$.

\textbf{3.} Als $P(\cos\theta) = \cos n\theta$ voor alle $\theta$,
dan komen $P$ en $T_n$ in elk punt van $\intcc{-1}1$ overeen --- een
oneindige \omterm{def:b1:logic:sets}{verzameling} --- dus $P - T_n$ heeft oneindig veel wortels
en is de nulveelterm (\cref{cor:b1:poly:nroots}).

\textbf{4.} Voor $x = \cos\theta$: $T_m(T_n(\cos\theta)) =
T_m(\cos n\theta) = \cos(mn\theta) = T_{mn}(\cos\theta)$, en
$2T_mT_n(\cos\theta) = 2\cos m\theta\cos n\theta = \cos(m+n)\theta
+ \cos\abs{m - n}\theta$. Beide identiteiten gelden op $\intcc{-1}1$,
dus als veeltermidentiteiten volgens het argument van vraag 3.

\textbf{5.} De $x_k$ zijn $n$ verschillende enkelvoudige wortels en
de leidende coëfficiënt is $2^{n-1}$:
\[
T_n = 2^{n-1}\prod_{k=0}^{n-1}
\Bigl(X - \cos\frac{(2k+1)\pi}{2n}\Bigr) .
\]
Afwisseling: de hoeken $0 < \frac{\pi}{2n} < \frac\pi n <
\frac{3\pi}{2n} < \frac{2\pi}n < \dots < \pi$ wisselen af tussen
de $y$-hoeken $\frac{k\pi}n$ en de $x$-hoeken
$\frac{(2k+1)\pi}{2n}$; aangezien $\cos$ strikt dalend is op
$\intcc0\pi$, wisselen de waarden af in de omgekeerde volgorde: $y_n <
x_{n-1} < y_{n-1} < \dots < x_0 < y_0$. Tussen twee opeenvolgende
extrema zit precies één wortel, zoals een plaatje van $\cos n\theta$
suggereert.

\textbf{6.} Inductie met $2\cosh a\cosh b = \cosh(a + b) +
\cosh(a - b)$ (\cref{prop:b1:functions:hyprules}): $T_{n+1}(\cosh
t) = 2\cosh t\cosh nt - \cosh(n-1)t = \cosh(n+1)t$. Voor $x \geq
1$, schrijf $x = \cosh t$ met $t \geq 0$; dan $\eu^t = x +
\sqrt{x^2 - 1}$ en $\eu^{-t} = x - \sqrt{x^2 - 1}$, dus
\[
T_n(x) = \cosh(nt)
= \frac{(x + \sqrt{x^2-1})^n + (x - \sqrt{x^2-1})^n}2 .
\]
Voor $x > 1$ overschrijdt de eerste term strikt $\frac12(1)^n$ en
groeit meetkundig: $T_n(x) > 1$.

\textbf{7.} De Moivre: $\cos n\theta = \Re\bigl((\cos\theta +
\iu\sin\theta)^n\bigr) = \sum_{2j \leq n}\binom n{2j}
\cos^{n-2j}\theta\,(\iu\sin\theta)^{2j}$, en $(\iu\sin\theta)^{2j}
= (-\sin^2\theta)^j = (\cos^2\theta - 1)^j$. Substitutie van $x =
\cos\theta$ en beroep op vraag 3:
\[
T_n(x) = \sum_{0\leq 2j\leq n}\binom n{2j}x^{n-2j}(x^2 - 1)^j .
\]
Voor $n = 3$: $\binom30 x^3 + \binom32 x(x^2 - 1) = x^3 + 3x^3 -
3x = 4x^3 - 3x$, zoals in vraag 1.

\textbf{8.} $T_n(1) = \cos(n\cdot0) = 1$; $T_n(-1) = \cos(n\pi) =
(-1)^n$; $T_n(0) = \cos\frac{n\pi}2$, wat $0$ is voor oneven $n$ en
$(-1)^{n/2}$ voor even $n$.

\textbf{9.} Inductie voor $U_n(\cos\theta) =
\frac{\sin(n+1)\theta}{\sin\theta}$: waar voor $U_0 = 1$ en $U_1 =
2X$ ($\sin2\theta = 2\sin\theta\cos\theta$); de stap is de
som-naar-product-identiteit $\sin(n+2)\theta = 2\cos\theta\,
\sin(n+1)\theta - \sin n\theta$. Leid nu
$T_n(\cos\theta) = \cos n\theta$ af naar $\theta$:
$-\sin\theta\,T_n'(\cos\theta) = -n\sin n\theta$, dus voor $\theta
\notin \pi\Z$:
\[
T_n'(\cos\theta) = n\,\frac{\sin n\theta}{\sin\theta}
= n\,U_{n-1}(\cos\theta) ,
\]
en de \omterm{def:b1:poly:def}{veeltermen} $T_n'$ en $nU_{n-1}$, die op $\intoo{-1}1$
overeenkomen, zijn gelijk.

\textbf{10.} $\abs{\sin(n+1)\theta} = \abs{\sin n\theta\cos\theta
+ \cos n\theta\sin\theta} \leq \abs{\sin n\theta} +
\abs{\sin\theta}$, en inductie geeft $\abs{\sin n\theta} \leq
n\abs{\sin\theta}$. Dus $\abs{U_{n-1}} \leq n$ op $\intoo{-1}1$
en $\abs{T_n'} = n\abs{U_{n-1}} \leq n^2$ daar; in $\pm1$ strekt de
grens zich uit door limieten te nemen (of rechtstreeks: $U_{n-1}(1) =
n$ uit de recurrentie, $U_n(1) = n + 1$ door inductie, en de pariteit
geeft $U_{n-1}(-1) = (-1)^{n-1}n$). Dus $T_n'(1) = n^2$ en $T_n'(-1)
= (-1)^{n-1}n^2$: de grens $n^2$ wordt bereikt in de eindpunten.

\textbf{11.} Leid $\sin\theta\,T_n'(\cos\theta) = n\sin
n\theta$ (vraag 9) af naar $\theta$:
\[
\cos\theta\,T_n'(\cos\theta) - \sin^2\theta\,T_n''(\cos\theta)
= n^2\cos n\theta = n^2\,T_n(\cos\theta) .
\]
Met $x = \cos\theta$ en $\sin^2\theta = 1 - x^2$: $x\,T_n' - (1
- x^2)T_n'' = n^2T_n$ op $\intcc{-1}1$, dus overal:
$(1 - x^2)y'' - xy' + n^2y = 0$ voor $y = T_n$. Controle voor $T_2 =
2x^2 - 1$: $(1 - x^2)(4) - x(4x) + 4(2x^2 - 1) = 4 - 4x^2 - 4x^2
+ 8x^2 - 4 = 0$.

\textbf{12.} $\widetilde T_n$ is \omterm{def:b1:poly:def}{monisch} (vraag 2) en
$\abs{\widetilde T_n} = 2^{1-n}\abs{T_n} \leq 2^{1-n}$ op
$\intcc{-1}1$, met $\widetilde T_n(y_k) = (-1)^k2^{1-n}$ in de
$n + 1$ punten $y_k$ (\cref{exo:b1:poly:10}): de norm is precies
$2^{1-n}$, bereikt met afwisselende tekens.

\textbf{13.} $\widetilde T_n$ en $P$ zijn beide \omterm{def:b1:poly:def}{monisch} van graad
$n$, dus de leidende termen heffen elkaar op: $\deg D \leq n - 1$. In
$y_k$: $D(y_k) = (-1)^k2^{1-n} - P(y_k)$, en $\abs{P(y_k)} \leq \norm
P_\infty < 2^{1-n}$ dwingt het teken van $D(y_k)$ om dat van
$(-1)^k2^{1-n}$ te zijn, strikt.

\textbf{14.} $D$ verandert van teken tussen $y_{k+1}$ en $y_k$ voor
elke $k = 0, \dots, n-1$: volgens de tussenwaarde-eigenschap heeft $D$
een wortel in elk van deze $n$ paarsgewijs disjuncte open intervallen
--- $n$ verschillende wortels voor een niet-nulveelterm van graad
$\leq n - 1$, onmogelijk. En $D = 0$ is ook onmogelijk (de normen
verschillen). Tegenspraak: geen enkele monische $P$ van graad $n$
heeft $\norm P_\infty < 2^{1-n}$, wat de stelling van Tsjebysjev is.

\textbf{15.} Nu is $\abs{P(y_k)} \leq 2^{1-n}$ alleen, dus $(-1)^k
D(y_k) = 2^{1-n} - (-1)^kP(y_k) \geq 2^{1-n} - \abs{P(y_k)} \geq
0$. Veronderstel $D(y_k) = 0$ in een inwendig $y_k$ ($0 < k < n$):
dan $P(y_k) = (-1)^k2^{1-n}$, dus $\abs P$ bereikt zijn supremum
$2^{1-n}$ in het inwendige punt $y_k$, waaruit $P'(y_k) = 0$
(inwendig extremum); en $T_n'(y_k) = nU_{n-1}(y_k) = 0$ aangezien
$\sin(n\cdot\frac{k\pi}n) = 0$ --- dus $\widetilde T_n'(y_k) = 0$
eveneens, en $D'(y_k) = 0$: $y_k$ is een wortel van $D$ van
\omterm{def:b1:poly:derivative}{multipliciteit} ten minste $2$.

\textbf{16.} Tel de wortels van $D$ met \omterm{def:b1:poly:derivative}{multipliciteit}. Laat $z$ het
aantal inwendige punten $y_k$ zijn met $D(y_k) = 0$ (elk een dubbele
wortel, volgens vraag 15) en $e \in \{0, 1, 2\}$ het aantal
eindpunten ($y_0$ of $y_n$) met $D = 0$ (elk ten minste een
enkelvoudige wortel). Een tussenruimte $(y_{k+1}, y_k)$ waarvan beide
eindpunten $D \neq 0$ hebben draagt strikt afwisselende tekens, dus
een inwendige wortel. Elk verdwijnend inwendig punt bederft ten
hoogste zijn twee aangrenzende tussenruimtes, elk verdwijnend
eindpunt ten hoogste één tussenruimte: ten minste $n - 2z - e$
tussenruimtes dragen nog elk één wortel bij, alle verschillend van de
$y$-wortels. Totaal: ten minste $(n - 2z - e) + 2z + e = n$ wortels
met \omterm{def:b1:poly:derivative}{multipliciteit}, voor een \omterm{def:b1:poly:def}{veelterm} van graad $\leq n - 1$: dus
$D = 0$ en $P = \widetilde T_n$. De minimalisator is uniek.

\textbf{17.} De affiene \omterm{def:b1:logic:map}{afbeelding} $t \mapsto x = \frac{a+b}2 +
\frac{b-a}2\,t$ is een bijectie $\intcc{-1}1 \to \intcc ab$. Als $P$
\omterm{def:b1:poly:def}{monisch} van graad $n$ is, dan is $Q(t) = P(x(t))$ een \omterm{def:b1:poly:def}{veelterm} in
$t$ met leidende coëfficiënt $\bigl(\frac{b-a}2\bigr)^n$, en
$\sup_{\intcc ab}\abs P = \sup_{\intcc{-1}1}\abs Q$. De \omterm{def:b1:poly:def}{monische
veelterm} $Q/\bigl(\frac{b-a}2\bigr)^n$ heeft supnorm $\geq
2^{1-n}$ (vragen 13--14), dus
\[
\sup_{\intcc ab}\abs P \geq \Bigl(\frac{b-a}2\Bigr)^n 2^{1-n}
= 2\Bigl(\frac{b-a}4\Bigr)^n ,
\]
met gelijkheid precies voor $P(x) = \bigl(\frac{b-a}2\bigr)^n
\widetilde T_n\bigl(t(x)\bigr)$ (vraag 16).

\textbf{18.} $\widetilde T_3 = \frac{T_3}4 = X^3 - \frac34X$;
$\widetilde T_3{}' = 3X^2 - \frac34$ verdwijnt in $\pm\frac12$.
Waarden: $\widetilde T_3(-1) = -\frac14$, $\widetilde
T_3(-\tfrac12) = \frac14$, $\widetilde T_3(\tfrac12) = -\frac14$,
$\widetilde T_3(1) = \frac14$: vier afwisselende extrema van
absolute waarde $\frac14$ --- dus $\norm{\widetilde T_3}_\infty =
\frac14$, en volgens de stelling van Tsjebysjev heeft geen enkele
monische derdegraadsveelterm een kleinere supnorm op $\intcc{-1}1$.

\textbf{19.} $\omega$ is \omterm{def:b1:poly:def}{monisch} van graad $n + 1$, dus
$\norm\omega_\infty \geq 2^{-n}$ volgens de stelling van Tsjebysjev
(graad $n+1$), met gelijkheid dan en slechts dan als $\omega =
\widetilde T_{n+1} = 2^{-n}T_{n+1}$ (vraag 16), d.w.z.\ dan en slechts
dan als de knopen de $n + 1$ wortels van $T_{n+1}$ zijn. Met
Tsjebysjev-knopen is de foutfactor $\norm\omega_\infty$ gelijk aan
$2^{-n}$ --- de kleinst mogelijke.

\textbf{20.} $5 \times 36^\circ = 180^\circ$, dus $T_5(c) =
\cos180^\circ = -1$: $16c^5 - 20c^3 + 5c + 1 = 0$. Toetsen van $x =
-1$: $-16 + 20 - 5 + 1 = 0$, en uitwerken bevestigt
\[
16x^5 - 20x^3 + 5x + 1 = (x + 1)\bigl(4x^2 - 2x - 1\bigr)^2 .
\]
Aangezien $c = \cos36^\circ \neq -1$, is $c$ een wortel van $4x^2 - 2x -
1$, waarvan de wortels $\frac{1 \pm \sqrt5}4$ zijn; aangezien $c > 0$,
\[
\cos36^\circ = \frac{1 + \sqrt5}4 .
\]
Consistentie: $\cos72^\circ = T_2(c) = 2c^2 - 1 = 2\cdot\frac{3 +
\sqrt5}8 - 1 = \frac{\sqrt5 - 1}4$, de waarde gevonden in
\cref{exo:b1:complex:8}.

\textbf{21.} $\sqrt{1.1^2 - 1} = \sqrt{0.21} \approx 0.458$, dus
$x + \sqrt{x^2-1} \approx 1.558$ en $(1.558)^{10} \approx 84.5$,
terwijl $(1.1 - 0.458)^{10} \approx 0.01$: $T_{10}(1.1) \approx
\frac{84.5 + 0.01}2 \approx 42$. Een \omterm{def:b1:poly:def}{veelterm} gevangen in
$\intcc{-1}1$ op het interval is al voorbij $40$ gegroeid één tiende
buiten zijn rand: begrensdheid op een segment zegt niets een duimbreed
erbuiten.

\textbf{22.} In de formule van vraag 7 voor $T_p$ is de term $j = 0$
gelijk aan $X^p$; elke andere term draagt $\binom p{2j}$ met $0 < 2j <
p$ (merk op $2j \neq p$ aangezien $p$ oneven is), die \omterm{def:b1:arith:divides}{deelbaar} is door
$p$ volgens de eerste stap van het bewijs van
\cref{thm:b1:arith:fermat}. Dus elke coëfficiënt van $T_p - X^p$ is
een veelvoud van $p$. Controles: $T_3 - X^3 = 3X^3 - 3X = 3(X^3 -
X)$; $T_5 - X^5 = 15X^5 - 20X^3 + 5X = 5(3X^5 - 4X^3 + X)$.

\textbf{23.} Volgens vraag 17 met $\intcc ab = \intcc01$ en $n =
2$: minimale afwijking $2\bigl(\frac14\bigr)^2 = \frac18$,
bereikt door $\bigl(\frac12\bigr)^2\widetilde T_2(2x - 1) =
\frac14\bigl((2x-1)^2 - \frac12\bigr) = x^2 - x + \frac18$. De
monische kwadratische die het dichtst bij nul ligt op $\intcc01$ is
$x^2 - x + \frac18$, met supnorm $\frac18$.

\textbf{24.} (i) Starheid --- een \omterm{def:b1:poly:def}{veelterm} met meer wortels dan
zijn graad is nul --- voedde het uniciteitsprincipe (vraag 3), de
overdracht van trigonometrische identiteiten naar veeltermidentiteiten
(vragen 4, 7, 9, 11), en beide worteltellingsargumenten van het
extremaliteitsbewijs (vragen 14, 16). (ii) De trigonometrie van
\cref{ch:b1:complex} (de Moivre, som-naar-product) en de \omterm{def:b1:functions:hyperbolic}{hyperbolische
functies} van \cref{ch:b1:functions} leverden elke identiteit achter de
familie; de substitutie $x = \cos\theta$ is de brug. (iii) De
\omterm{def:b1:arith:divides}{deelbaarheid} $p \mid \binom p{2j}$ uit \cref{ch:b1:arith} zette de
coëfficiëntenformule om in de \omterm{def:b1:arith:congruence}{congruentie} van vraag 22.

\textbf{25.} De stelling van Tsjebysjev zet een optimalisatie over
een oneindig-dimensionale familie (alle \omterm{def:b1:poly:def}{monische veeltermen}) om in
eindige combinatoriek: een concurrent beter dan $\widetilde T_n$ zou
ervan verschillen door een \omterm{def:b1:poly:def}{veelterm} van lage graad die $n$ keer van
teken moet veranderen --- één wortel meer dan zijn graad toelaat. Het
equioscillatiepatroon is dus geen curiositeit maar het certificaat
zelf van optimaliteit, en het gelijkheidsgeval verscherpt de
worteltelling met \omterm{def:b1:poly:derivative}{multipliciteiten}. De substitutie $x = \cos\theta$
verdient het laatste woord: ze transporteert de starre, discrete
wereld van \omterm{def:b1:poly:def}{veeltermen} naar de periodieke wereld van de trigonometrie,
waar wortels en extrema van $T_n$ eenvoudig het regelmatige raster van
$\cos n\theta$ zijn. De twee ontleende analysefeiten --- de
tussenwaarde-eigenschap (vraag 14; bewezen in \cref{ch:b1:continuity})
en de verdwijnende afgeleide in een inwendig extremum (vraag 15;
bewezen in \cref{ch:b1:derivative}) --- zijn precies de gereedschappen
die die latere hoofdstukken zullen teruggeven, waarmee de lus gesloten
wordt.
\end{solution}
