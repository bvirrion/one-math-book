\section*{Hoofdstuk \ref{ch:b1:multivar} --- Functies van Twee Variabelen}

\begin{solution}{exo:b1:multivar:1}
$\dfrac{\partial f}{\partial x} = 2xy + y\,\eu^{xy}$,
$\dfrac{\partial f}{\partial y} = x^2 + x\,\eu^{xy}$.

$\dfrac{\partial g}{\partial x} = \dfrac{2x}{x^2+y^2}$,
$\dfrac{\partial g}{\partial y} = \dfrac{2y}{x^2+y^2}$.

$\dfrac{\partial h}{\partial x} = \dfrac{-y/x^2}{1 + y^2/x^2} =
\dfrac{-y}{x^2+y^2}$,
$\dfrac{\partial h}{\partial y} = \dfrac{x}{x^2+y^2}$.
\end{solution}

\begin{solution}{exo:b1:multivar:2}

$\abs{f} = \dfrac{\rho^3\abs{\cos^2\theta\sin\theta}}{\rho^2} \leq

\end{solution}

\begin{solution}{exo:b1:multivar:3}
Langs $y = mx$: $f(x, mx) = \dfrac{m x^3}{x^4 + m^2 x^2} =
\dfrac{mx}{x^2 + m^2} \to 0$ (voor $m \neq 0$; langs $y = 0$ en de
$y$-as is $f = 0$). Dus elke limiet langs een rechte lijn is $0$. Maar op de
parabool $y = x^2$:
\[
f(x, x^2) = \frac{x^4}{x^4 + x^4} = \frac12 :
\]
de rij $\bigl(\frac1n, \frac{1}{n^2}\bigr) \to (0,0)$ heeft
$f \to \frac12 \neq 0$. Niet \omterm{def:b1:continuity:continuous}{continu}: rechten volstaan niet om
tweevariabele-limieten te testen.
\end{solution}

\begin{solution}{exo:b1:multivar:4}
$\frac{\partial f}{\partial x} = 3x^2y^2 + y\cos(xy)$; dan
\[
\frac{\partial^2 f}{\partial y\,\partial x}
= 6x^2 y + \cos(xy) - xy\sin(xy) .
\]
$\frac{\partial f}{\partial y} = 2x^3 y + x\cos(xy)$; dan
\[
\frac{\partial^2 f}{\partial x\,\partial y}
= 6x^2 y + \cos(xy) - xy\sin(xy) :
\]
gelijk, zoals Schwarz belooft.
\end{solution}

\begin{solution}{exo:b1:multivar:5}
Volgens \cref{thm:b1:multivar:chain} met $(x, y) = (\cos t, \sin t)$:
\[
g'(t) = -\sin t\,\frac{\partial f}{\partial x}(\cos t, \sin t)
+ \cos t\,\frac{\partial f}{\partial y}(\cos t, \sin t)
= \bigl\langle \nabla f,\ (-\sin t, \cos t)\bigr\rangle .
\]
In een extremum van $g$ is $g'(t) = 0$: $\nabla f$ staat orthogonaal op de
raakvector $(-\sin t, \cos t)$ van de cirkel, dus evenwijdig met
de straalvector $(\cos t, \sin t)$ (het orthogonale complement in het
vlak van een eenheidsvector is de rechte die hij opspant, loodrecht
genomen). Dit is het eenvoudigste geval van een Lagrange-\-
multiplicator.
\end{solution}

\begin{solution}{exo:b1:multivar:6}
$f$: $\nabla f = (2x + y - 3,\; x + 2y) = 0$: $y = -\frac x2$ en
$2x - \frac x2 = 3$: $x = 2$, $y = -1$. Tweede orde: $r = 2$, $s =
1$, $t = 2$: $rt - s^2 = 3 > 0$, $r > 0$: lokaal (zelfs globaal ---
kwadratisch) minimum in $(2, -1)$, waarde $f(2,-1) = -3$.

$g$: $\nabla g = (2x,\; -2y + 4) = 0$: punt $(0, 2)$; $r = 2$, $s
= 0$, $t = -2$: $rt - s^2 = -4 < 0$: zadel.

$h$: $\nabla h = \bigl(4x^3 - 4(x - y),\; 4y^3 + 4(x-y)\bigr) = 0$.
Optellen: $x^3 + y^3 = 0$, dus $y = -x$; substitueren: $4x^3 - 8x = 0$:
$x \in \{0, \pm\sqrt2\}$. Kritieke punten: $(0,0)$,
$(\sqrt2, -\sqrt2)$, $(-\sqrt2, \sqrt2)$.
In $(\pm\sqrt2, \mp\sqrt2)$: $r = 12x^2 - 4 = 20$, $s = 4$, $t =
20$: $rt - s^2 > 0$, $r > 0$: lokale minima (waarde $h = 4 + 4 - 16 =
-8$). In $(0,0)$: $r = t = -4$, $s = 4$: $rt - s^2 = 0$: stil.
Onderzoek: $h(x, x) = 2x^4 > 0$ en $h(x, -x) = 2x^4 - 8x^2 < 0$ voor
kleine $x \neq 0$: beide tekens in elke \omterm{def:b1:topology:open}{omgeving} --- geen extremum
in de oorsprong.
\end{solution}

\begin{solution}{exo:b1:multivar:7}
$x^2 - y^2$: tweede \omterm{def:b1:multivar:partial}{partiĂ«le afgeleiden} $2$ en $-2$: som $0$. $xy$: beide zuivere
tweede \omterm{def:b1:multivar:partial}{partiĂ«le afgeleiden} verdwijnen. $\eu^x\cos y$: $\frac{\partial^2}{\partial
x^2} = \eu^x\cos y$, $\frac{\partial^2}{\partial y^2} =
-\eu^x\cos y$: som $0$. $\ln(x^2+y^2)$: uit
\cref{exo:b1:multivar:1},
\[
\frac{\partial^2}{\partial x^2}\ln(x^2+y^2)
= \frac{2(x^2+y^2) - 4x^2}{(x^2+y^2)^2}
= \frac{2(y^2 - x^2)}{(x^2+y^2)^2},
\]
en de $y$-versie is haar tegengestelde: som $0$.
\end{solution}

\begin{solution}{exo:b1:multivar:8}
\emph{\omterm{def:b1:linmaps:projection}{Projectie}:} het vlak $P$ heeft normaal $n = (1, 2, -1)$ en
gaat door $A = (4, 0, 0)$. Het dichtste punt bij de oorsprong is
$O$ geprojecteerd: $p = O + \frac{\langle A - O, n\rangle}{\norm
n^2}\,n = \frac{4}{6}(1,2,-1) = \bigl(\frac23, \frac43,
-\frac23\bigr)$, op afstand $\frac{4}{\sqrt 6}$. (Formule: afstand
van de oorsprong tot het vlak $\langle X, n \rangle = c$ is
$\frac{\abs c}{\norm n}$ met $c = 4$.)

\emph{Minimalisering:} $f(x,y) = x^2 + y^2 + (x + 2y - 4)^2$. Schrijven we
$w = x + 2y - 4$ voor de laatste factor:
\[
\nabla f = \bigl(2x + 2w,\; 2y + 4w\bigr) = 0
\iff x = -w \text{ en } y = -2w .
\]
Substitueren in de definitie van $w$: $w = -w - 4w - 4$, dus $w =
-\frac23$, wat geeft $x = \frac23$, $y = \frac43$, en $z = w =
-\frac23$. Zelfde punt als via \omterm{def:b1:linmaps:projection}{projectie}, op afstand
$\sqrt{\frac49 + \frac{16}{9} + \frac49} = \frac{2\sqrt 6}{3} =
\frac{4}{\sqrt6}$; het is een minimum aangezien $f \to \infty$ in het oneindige
(een positief definiete kwadratische vorm plus lineaire termen).
\end{solution}

\begin{solution}{exo:b1:multivar:9}
Met $\rho = \sqrt{x^2+y^2}$: $\frac{\partial \rho}{\partial x} =
\frac{x}{\rho}$, dus
\[
\frac{\partial f}{\partial x} = \varphi'(\rho)\,\frac{x}{\rho},
\qquad
\frac{\partial^2 f}{\partial x^2}
= \varphi''(\rho)\,\frac{x^2}{\rho^2}
+ \varphi'(\rho)\,\Bigl(\frac{1}{\rho} -
\frac{x^2}{\rho^3}\Bigr),
\]
(quotiĂ«nt- en kettingregel). Bij het optellen van de symmetrische $y$-uitdrukking:
verzamelen de $\varphi''$-termen $\frac{x^2 + y^2}{\rho^2} = 1$, de
$\varphi'$-termen verzamelen $\frac{2}{\rho} - \frac{x^2 +
y^2}{\rho^3} = \frac{1}{\rho}$:
\[
\Delta f = \varphi'' + \frac{\varphi'}{\rho} .
\]
Radiaal harmonisch: los $\varphi'' + \frac{\varphi'}{\rho} = 0$ op: de
vergelijking van eerste orde $u' + \frac u\rho = 0$ voor $u = \varphi'$
geeft $u = \frac{c}{\rho}$ (\cref{thm:b1:diffeq:homogeneous1}),
dan $\varphi = c\ln\rho + d$. De radiale harmonische functies zijn
$c\,\ln\sqrt{x^2 + y^2} + d$ --- de logaritmische potentiaal van
\cref{exo:b1:multivar:7} en de constanten, niets anders.
\end{solution}

\begin{solution}{exo:b1:multivar:10}
Voor $z = xy$ in $(1,1)$: \omterm{def:b1:multivar:partial}{partiĂ«le afgeleiden} $y = 1$ en $x = 1$, raak\-
vlak $z = 1 + (x - 1) + (y - 1) = x + y - 1$. Horizontaal
raakvlak betekent dat beide \omterm{def:b1:multivar:partial}{partiĂ«le afgeleiden} verdwijnen, d.w.z.\ $\nabla f = 0$:
volgens \cref{ex:b1:multivar:extrema} precies de kritieke punten
$(0, 0)$ en $(1, 1)$, met horizontale vlakken $z = 0$ en $z =
-1$. ``Horizontaal raakvlak'' en ``\omterm{thm:b1:multivar:critical}{kritiek punt}'' zijn hetzelfde
begrip, gezien op de grafiek en in de formule.
\end{solution}

\begin{solution}{exo:b1:multivar:11}
\begin{enumerate}
  \item Voor vaste $y$ heeft de functie in Ă©Ă©n variabele $x \mapsto
        f(x,y)$ een \omterm{def:b1:derivative:def}{afgeleide} nul op $\R$, dus is ze constant
        (\cref{thm:b1:derivative:mvt}): $f(x, y) = f(0, y)$ voor
        alle $x$: $f$ hangt alleen van $y$ af.
  \item Zij $g(u, v) = f\bigl(\frac{u+v}2, \frac{u-v}2\bigr)$.
        Via de kettingregel (\cref{thm:b1:multivar:chain}, toegepast
        in de variabele $v$ met $u$ bevroren):
        \[
        \frac{\partial g}{\partial v}
        = \frac12\,\frac{\partial f}{\partial x}
        - \frac12\,\frac{\partial f}{\partial y} = 0 .
        \]
        Via (1) hangt $g$ alleen van $u$ af: $g(u, v) = \varphi(u)$
        met $\varphi$ van klasse $C^1$, en door de verandering
        van veranderlijken om te keren ($u = x + y$, $v = x - y$),
        \[
        f(x, y) = \varphi(x + y).
        \]
        Omgekeerd voldoet elke zulke $f$ aan $f_x = f_y =
        \varphi'$: de oplossingen zijn precies de $C^1$-functies
        van $x + y$.
\end{enumerate}
\end{solution}

\begin{solution}{exo:b1:multivar:12}
Op de \omterm{def:b1:topology:open}{open} schijf zou een extremum \omterm{thm:b1:multivar:critical}{kritiek} zijn: $\nabla f = (y,
x) = 0$ alleen in $(0,0)$, waar $f = 0$; het is een zadel ($f(\pm
\varepsilon, \pm\varepsilon) = \varepsilon^2 > 0 > -\varepsilon^2
= f(\pm\varepsilon, \mp\varepsilon)$), dus daar geen extremum. Volgens
de aangenomen extremumwaardenstelling worden de grenzen bereikt,
noodzakelijkerwijs op de randcirkel. Daar, met
\cref{exo:b1:multivar:5},
\[
f(\cos t, \sin t) = \cos t\sin t = \frac{\sin 2t}{2}
\in \intcc{-\tfrac12}{\tfrac12},
\]
met maximum $\frac12$ in $t = \frac\pi4, \frac{5\pi}4$ (punten
$\pm\frac{1}{\sqrt2}(1,1)$) en minimum $-\frac12$ in $t =
\frac{3\pi}4, \frac{7\pi}4$ (punten $\pm\frac{1}{\sqrt2}(1,-1)$).
Globaal maximum $\frac12$, globaal minimum $-\frac12$.
\end{solution}

\begin{solution}{pb:b1:multivar:1}
\textbf{1.} Het rechterlid uitwerken:
\[
r\Bigl(h + \frac srk\Bigr)^{\!2} + \frac{rt - s^2}{r}k^2
= rh^2 + 2shk + \frac{s^2}{r}k^2 + \frac{rt - s^2}{r}k^2
= q(h,k) .
\]
Als $rt - s^2 > 0$: beide kwadraten dragen het factorteken van $r$,
en $q(h,k) = 0$ dwingt $k = 0$ dan $h = 0$: strikt teken van $r$
buiten de oorsprong. Als $rt - s^2 < 0$: $q(1, 0) = r$ terwijl
$q\bigl(-\frac sr, 1\bigr) = \frac{rt - s^2}{r}$ het
tegengestelde teken heeft: beide tekens komen voor.

\textbf{2.} Als $r = 0 \neq t$, verwissel de rollen van $h$ en $k$
(het kwadraat afgesplitst in $k$), met dezelfde conclusies; merk op dat
$rt - s^2 = -s^2 < 0$ zodra $s \neq 0$, en inderdaad neemt $q =
2shk + tk^2$ dan beide tekens aan ($k$ klein tegenover $h$). Als
$r = t = 0$: $q = 2shk$, beide tekens desda $s \neq 0$, d.w.z.\ desda
$rt - s^2 = -s^2 < 0$. Samenvatting: $q$ neemt beide tekens aan $\iff rt -
s^2 < 0$; $q$ verdwijnt alleen in de oorsprong (definiet) $\iff rt -
s^2 > 0$ (wat $r \neq 0$ dwingt, aangezien $r = 0$ geeft $q(1,0) =
0$).

\textbf{3.} Met $f_x = rx + sy + \beta$ en $f_y = sx + ty +
\gamma$ geeft rechtstreekse uitwerking
\[
f(a + h, b + k) = f(a, b) + h\,f_x(a,b) + k\,f_y(a,b)
+ \tfrac12 q(h, k),
\]
zonder hogere termen (de functie is een \omterm{def:b1:poly:def}{veelterm} van graad
$2$). In een \omterm{thm:b1:multivar:critical}{kritiek punt} verdwijnt het lineaire deel: $f(X_0 + H)
- f(X_0) = \frac12 q(H)$ \emph{exact}, zodat de tekenstudie van
vragen 1--2 classificeert: strikt globaal minimum ($rt - s^2 >
0$, $r > 0$), strikt globaal maximum ($rt - s^2 > 0$, $r < 0$),
zadel --- beide tekens in elke \omterm{def:b1:topology:open}{omgeving} --- wanneer $rt - s^2 <
0$. Dit bewijst \cref{met:b1:multivar:monge} voor kwadratische
functies.

\textbf{4.} De vorm $q - c(h^2 + k^2)$ heeft gegevens $r - c$, $s$,
$t - c$. Met $c = \frac{rt - s^2}{r + t}$ (merk op $t > 0$ aangezien
$rt > s^2 \geq 0$ en $r > 0$):
\[
(r - c)(t - c) - s^2 = rt - s^2 - c(r + t) + c^2 = c^2 \geq 0,
\]
en $r - c \geq 0$ ($c \leq r \iff rt - s^2 \leq r^2 + rt$,
waar). Als $r - c > 0$, toont het afgesplitste kwadraat van vraag 1 dat
$q - c(h^2 + k^2) \geq 0$; als $r - c = 0$, dan $s = 0$ (uit
$-s^2 = -c^2 \leq \dots$ die de weergegeven grootheid
$\geq 0$ dwingt met eerste factor $0$) en de vorm is $(t - c)k^2 \geq
0$. In beide gevallen is $q(h,k) \geq c(h^2 + k^2)$.

\textbf{5.} Volgens vragen 3--4 is $f(X) = f(X_0) + \frac12 q(X -
X_0) \geq f(X_0) + \frac c2\norm{X - X_0}^2 \to +\infty$ als
$\norm X \to \infty$: $f$ is coĂ«rcief, en de ongelijkheid is
strikt voor $X \neq X_0$: het kritieke punt is de unieke globale
minimaliseerder. Voor de derdegraadsfunctie $f = x^3 + y^3 - 3xy$ van
\cref{ex:b1:multivar:extrema} geldt zo'n conclusie niet: $f(x,0)
= x^3 \to -\infty$, en het lokale minimum in $(1,1)$ is niet
globaal --- het is de exactheid van de kwadratische ontwikkeling die faalde.

\textbf{6.} De gekwadrateerde norm uitwerken:
\[
f(u,v) = u^2\norm{C_1}^2 + 2uv\,\langle C_1, C_2\rangle +
v^2\norm{C_2}^2 - 2u\,\langle C_1, b\rangle - 2v\,\langle C_2,
b\rangle + \norm b^2 ,
\]
een kwadratische functie van $(u, v)$ waarvan het kwadratische deel
$\frac12(ru^2 + 2suv + tv^2)$ is met $r = 2\norm{C_1}^2$, $s =
2\langle C_1, C_2\rangle$, $t = 2\norm{C_2}^2$.

\textbf{7.} $rt - s^2 = 4\bigl(\norm{C_1}^2\norm{C_2}^2 -
\langle C_1, C_2\rangle^2\bigr) \geq 0$ volgens Cauchy--Schwarz
(\cref{thm:b1:euclid:cs}), met gelijkheid precies wanneer $C_1, C_2$
evenredig zijn (of Ă©Ă©n ervan nul is), d.w.z.\ wanneer het paar
gebonden is. Dus $rt - s^2 > 0 \iff (C_1, C_2)$ \omterm{def:b1:vspaces:free}{vrij}.

\textbf{8.} $\nabla f = 0$ luidt
\[
\norm{C_1}^2 u + \langle C_1, C_2\rangle v = \langle C_1,
b\rangle,
\qquad
\langle C_1, C_2\rangle u + \norm{C_2}^2 v = \langle C_2,
b\rangle,
\]
de normaalvergelijkingen met de Gram-matrix links; ze zeggen
$\langle uC_1 + vC_2 - b,\ C_i\rangle = 0$ voor $i = 1, 2$, d.w.z.\
$b - p \perp \operatorname{Vect}(C_1, C_2)$ voor $p = uC_1 +
vC_2$. Volgens Deel I (vragen 3, 5) is het unieke kritieke punt
de unieke globale minimaliseerder, en volgens
\cref{thm:b1:euclid:projection} identificeert de karakterisering ``$p \in
F$, $b - p \perp F$'' $p$ als de \omterm{thm:b1:euclid:projection}{orthogonale
projectie} van $b$ op $F = \operatorname{Vect}(C_1, C_2)$.

\textbf{9.} $\norm{b - p}^2 = \norm b^2 - 2\langle b, p\rangle +
\norm p^2$, en $\langle p, b - p\rangle = 0$ geeft $\norm p^2 =
\langle p, b\rangle$: het minimum is gelijk aan $\norm b^2 - \langle p,
b\rangle$. Pythagoras: $\norm b^2 = \norm p^2 + \norm{b - p}^2$
--- de gegevensvector splitst in zijn verklaarde deel $p$ en zijn
residuele deel $b - p$, orthogonaal op elkaar.

\textbf{10.} $E(\alpha, \beta) = \norm{\alpha C_1 + \beta C_2 -
b}^2$ met de gestelde $C_1, C_2, b$; de normaalvergelijkingen van
vraag 8 zijn, per component,
\[
n\alpha + \Bigl(\sum x_i\Bigr)\beta = \sum y_i,
\qquad
\Bigl(\sum x_i\Bigr)\alpha + \Bigl(\sum x_i^2\Bigr)\beta = \sum
x_i y_i .
\]

\textbf{11.} Delen door $n$: $\alpha + \beta\bar x = \bar y$
en $\alpha\bar x + \beta\bigl(v_x + \bar x^2\bigr) = c_{xy} +
\bar x\bar y$. $\alpha = \bar y - \beta\bar x$ in de tweede
substitueren: $\beta v_x = c_{xy}$, dus
\[
\beta = \frac{c_{xy}}{v_x}, \qquad \alpha = \bar y - \beta\bar
x ,
\]
en de eerste vergelijking zegt precies dat $(\bar x, \bar y)$
op de rechte ligt.

\textbf{12.} $v_x = \frac1n\sum(x_i - \bar x)^2 \geq 0$, nul
desda elke $x_i$ gelijk is aan $\bar x$. En $rt - s^2 = 4\bigl(n\sum
x_i^2 - (\sum x_i)^2\bigr) = 4n^2 v_x$: de vrijheidsvoorwaarde
van vraag 7 is $v_x > 0$, d.w.z.\ de $x_i$ niet alle gelijk.

\textbf{13.} Met $\alpha = \bar y - \beta\bar x$, centreer de
gegevens ($\tilde x_i = x_i - \bar x$, $\tilde y_i = y_i - \bar y$):
\[
E(\alpha, \beta) = \sum_i(\tilde y_i - \beta\tilde x_i)^2
= n\,v_y - 2\beta\,n\,c_{xy} + \beta^2 n\,v_x ,
\]
geminimaliseerd in $\beta = c_{xy}/v_x$ met waarde $E_{\min} =
n\bigl(v_y - \frac{c_{xy}^2}{v_x}\bigr) = n v_y(1 - \rho^2)$.
Aangezien $E_{\min} \geq 0$ en $v_y > 0$: $\rho^2 \leq 1$, en
$\abs\rho = 1$ desda $E_{\min} = 0$, d.w.z.\ desda elk residu
verdwijnt: de punten liggen exact op de rechte.

\textbf{14.} $n = 4$: $\bar x = \frac64 = 1.5$, $\bar y = 2$,
$\sum x_i^2 = 14$ dus $v_x = 3.5 - 2.25 = 1.25$, $\sum x_iy_i =
0 + 1 + 6 + 12 = 19$ dus $c_{xy} = 4.75 - 3 = 1.75$. Bijgevolg
\[
\beta = \frac{1.75}{1.25} = 1.4,
\qquad
\alpha = 2 - 1.4\times1.5 = -0.1 :
\qquad y = 1.4\,x - 0.1 .
\]
Gefitte waarden $-0.1,\ 1.3,\ 2.7,\ 4.1$; residu's $0.1,\ -0.3,\
0.3,\ -0.1$; $E_{\min} = 0.01 + 0.09 + 0.09 + 0.01 = 0.2$
(controle: $v_y = \frac{26}4 - 4 = 2.5$ en $4(2.5 -
\frac{1.75^2}{1.25}) = 4(2.5 - 2.45) = 0.2$).

\textbf{15.} De twee normaalvergelijkingen van vraag 10 zijn
precies $\sum_i\varepsilon_i = 0$ en $\sum_i x_i\varepsilon_i =
0$: de residuvector staat orthogonaal op $C_1 = (1, \dots, 1)$
en op $C_2 = (x_i)$ --- op de hele modelruimte. In het
bijzonder brengt de optimale rechte haar fouten altijd in evenwicht: ze
sommeren tot nul.

\textbf{16.} $\frac{\dd}{\dd\alpha}\sum(y_i - \alpha)^2 =
-2\sum(y_i - \alpha) = 0$ geeft $\alpha = \bar y$ (en de
tweede \omterm{def:b1:derivative:def}{afgeleide} $2n > 0$ maakt het tot het globale minimum, aangezien de
functie een coĂ«rcieve kwadratische functie in Ă©Ă©n variabele is). Minimale
waarde $\sum(y_i - \bar y)^2 = n v_y$: de variantie is de
onherleidbare kwadratische fout van een constant model.

\textbf{17.} $\frac{\dd}{\dd\beta}\sum(y_i - \beta x_i)^2 =
-2\sum x_i(y_i - \beta x_i) = 0$ geeft $\beta_0 = \frac{\sum
x_iy_i}{\sum x_i^2}$. Beide richtingscoĂ«fficiĂ«nten over gecentreerde
grootheden schrijven: $\beta_0 = \frac{c_{xy} + \bar x\bar y}{v_x + \bar
x^2}$ is gelijk aan $\frac{c_{xy}}{v_x}$ desda $c_{xy}\bar x^2 = \bar
x\bar y\,v_x$, d.w.z.\ desda $\bar x = 0$ (gecentreerde abscissen) of
$\bar y = \beta\bar x$ --- dit laatste betekent $\alpha = 0$: de
volledige regressierechte gaat al door de oorsprong.

\textbf{18.} $g(\tau) = \norm{M - P_0}^2 - 2\tau\langle M - P_0,
w\rangle + \tau^2$ is minimaal in $\tau^* = \langle M - P_0,
w\rangle$, met waarde $d(M,D)^2 = \norm{M - P_0}^2 - \langle M -
P_0, w\rangle^2$. In het vlak, vul $w$ aan tot een orthonormale
\omterm{def:b1:vspaces:free}{basis} $(w, n)$ met $n = \frac{(a, b)}{\sqrt{a^2+b^2}}$ (eenheids\-
normaal van $D$): dan $\norm{M - P_0}^2 = \langle M - P_0,
w\rangle^2 + \langle M - P_0, n\rangle^2$, dus $d(M, D) =
\abs{\langle M - P_0, n\rangle}$, en met $aP_{0x} + bP_{0y} =
-c$:
\[
d = \frac{\abs{a x_0 + b y_0 + c}}{\sqrt{a^2 + b^2}} .
\]

\textbf{19.} Uit $E_{\min}/n = v_y - \frac{c_{xy}^2}{v_x}$ en
$\beta^2 v_x = \frac{c_{xy}^2}{v_x}$:
\[
v_y = \beta^2 v_x + \frac{E_{\min}}n :
\]
totale variantie $=$ variantie langs de gefitte rechte $+$ residuele
variantie. (Delen door $v_y$: $1 = \rho^2 + (1 - \rho^2)$, het
aandeel van de variantie ``verklaard'' door de rechte is $\rho^2$.)

\textbf{20.} De modelruimte is
$\operatorname{Vect}\bigl(C_1, C_2, C_3\bigr)$ met $C_1 = (1)$,
$C_2 = (x_i)$, $C_3 = (x_i^2)$; het minimaliseren van $\norm{aC_1 + bC_2 +
cC_3 - b}^2$ leidt, precies als in vraag 8, tot het $3\times3$-
Gram-stelsel, waarvan de matrix elementen $\langle C_i, C_j\rangle =
\sum_k x_k^{\,i+j-2}$ heeft: de weergegeven momentmatrix. Ze is
inverteerbaar desda $(C_1, C_2, C_3)$ \omterm{def:b1:vspaces:free}{vrij} is
(\cref{exo:b1:euclid:11}); een relatie $a + bx_k + cx_k^2 = 0$
voor alle $k$ maakt elke $x_k$ een wortel van Ă©Ă©n \omterm{def:b1:poly:def}{veelterm} van
graad $\leq 2$, onmogelijk met drie verschillende waarden tenzij $a
= b = c = 0$. Vergelijk de momentmatrices van het weekendprobleem
van \cref{ch:b1:det}, waar hun determinanten gekwadrateerde
Vandermonde-waarden waren.

\textbf{21.} Eerste: $r = 2, s = 1, t = 2$, $rt - s^2 = 3 > 0$,
$r > 0$: strikt globaal minimum in de oorsprong. Tweede: $r = 2, s
= 3, t = 2$, $rt - s^2 = -5 < 0$: zadel. Derde: $r = s = t =
2$, $rt - s^2 = 0$: gedegenereerd --- maar $x^2 + 2xy + y^2 = (x +
y)^2 \geq 0$ verdwijnt op de hele rechte $y = -x$: een
(niet-strikt) globaal minimum bereikt langs een rechte, onzichtbaar voor
de determinanttest.

\textbf{22.} Nieuwe sommen ($n = 5$): $\bar x = 3.2$, $\bar y =

\textbf{23.} Stel $W = \sum w_i$ en definieer $\bar x_w =
\frac1W\sum w_ix_i$, $\bar y_w$ evenzo, $v_x^w = \frac1W\sum
w_ix_i^2 - \bar x_w^2$, $c^w_{xy} = \frac1W\sum w_ix_iy_i - \bar
x_w\bar y_w$. De \omterm{def:b1:logic:map}{afbeelding} $\langle u, v\rangle_w = \sum_i w_iu_iv_i$
is een \omterm{def:b1:euclid:def}{inwendig product} op $\R^n$ ($w_i > 0$ geeft definietheid), dus
Deel II geldt letterlijk en de normaalvergelijkingen delen door $W$
in
\[
\alpha + \beta\bar x_w = \bar y_w,
\qquad
\alpha\bar x_w + \beta(v^w_x + \bar x_w^2) = c^w_{xy} + \bar
x_w\bar y_w ,
\]
waaruit $\beta = c^w_{xy}/v^w_x$ en $\alpha = \bar y_w -
\beta\bar x_w$: dezelfde formules, met elk gemiddelde gewogen.

\textbf{24.} $E_{\min} = \sum\varepsilon_i^2 = 0$ dwingt elk
$\varepsilon_i = 0$: alle punten exact op de rechte; volgens vraag
13 is dit het geval $\abs\rho = 1$. Test: voor $(1,1), (2,3),
(3,5)$: $\bar x = 2$, $\bar y = 3$, $v_x = \frac{14}3 - 4 =
\frac23$, $c_{xy} = \frac{22}3 - 6 = \frac43$: $\beta = 2$,
$\alpha = -1$, en inderdaad $y_i = 2x_i - 1$ voor alle drie de punten;
$v_y = \frac{35}3 - 9 = \frac83$ en $\rho^2 =
\frac{(4/3)^2}{(2/3)(8/3)} = 1$.

\textbf{25.} (i) Voor kwadratische functies is de ontwikkeling van orde twee
een \emph{identiteit}, zodat de tekenstudie van de kwadratische vorm
--- pure algebra, vragen 1--2 --- kritieke punten classificeert
zonder aangenomen Taylor-stelling. (ii) De normaalvergelijkingen zeggen
``residu orthogonaal op de modelruimte'', zodat het analytische
minimum de \omterm{thm:b1:euclid:projection}{orthogonale projectie} \emph{is} van de gegevensvector:
calculus en euclidische geometrie berekenen hetzelfde object. (iii)
De correlatie $\rho = c_{xy}/\sqrt{v_xv_y}$ meet het aandeel
$\rho^2$ van de variantie verklaard door de rechte, en
Cauchy--Schwarz begrenst haar: $\abs\rho \leq 1$, met gelijkheid alleen
voor uitgelijnde gegevens. (iv) Bases en vrijheid (Hoofdstuk 18), Gram-
en momentmatrices en hun determinanten (Hoofdstukken 22--23),
en \omterm{thm:b1:euclid:projection}{orthogonale projectie} (Hoofdstuk 23) doken alle weer op als de
werkende onderdelen van Ă©Ă©n algoritme. Delen II--III ontwikkelen de
\emph{methode van de \omterm{pb:b1:multivar:1}{kleinste kwadraten}} (Legendre, Gauss) via haar
\emph{normaalvergelijkingen}.
