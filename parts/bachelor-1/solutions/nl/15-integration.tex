\section*{Hoofdstuk \ref{ch:b1:integration} --- Integratie op een segment}

\begin{solution}{exo:b1:integration:1}
$\dfrac{1}{x^2-4} = \dfrac{1/4}{x - 2} - \dfrac{1/4}{x+2}$
(afdekmethode), dus
\[
\int_0^1 \frac{\dd x}{x^2 - 4}
= \frac14\Bigl[\ln\abs{x-2} - \ln\abs{x+2}\Bigr]_0^1
= \frac14\Bigl(\ln\frac{1}{3} - \ln 1\Bigr) = -\frac{\ln 3}{4}.
\]

Met \omterm{thm:b1:integration:parts}{partiële integratie} ($u' = 1$, $v = \ln t$): $\int_1^{\eu} \ln
t\,\dd t = \bigl[t\ln t\bigr]_1^{\eu} - \int_1^{\eu} \dd t = \eu - (\eu
- 1) = 1$.

Met \omterm{thm:b1:integration:parts}{partiële integratie} ($u' = \sin t$, $v = t$): $\int_0^\pi t\sin
t\,\dd t = \bigl[-t\cos t\bigr]_0^\pi + \int_0^\pi \cos t\,\dd t = \pi
+ 0 = \pi$.
\end{solution}

\begin{solution}{exo:b1:integration:2}
Substitutie $u = t^2 + 1$, $\dd u = 2t\,\dd t$:
\[
\int_0^1 \frac{t\,\dd t}{(t^2+1)^2}
= \frac12 \int_1^2 \frac{\dd u}{u^2}
= \frac12\Bigl[-\frac1u\Bigr]_1^2 = \frac14 .
\]
Met $u = \sin t$: $\int_0^{\pi/2} \cos^3 t\,\dd t = \int_0^{\pi/2}
(1 - \sin^2 t)\cos t\,\dd t = \bigl[\sin t - \frac{\sin^3
t}{3}\bigr]_0^{\pi/2} = 1 - \frac13 = \frac23$.
\end{solution}

\begin{solution}{exo:b1:integration:3}
$u_n = \dfrac1n \sum_{k=1}^{n} \dfrac{1}{1 + (k/n)^2}$: een \omterm{thm:b1:integration:riemann}{Riemannsom}
van $x \mapsto \frac{1}{1+x^2}$ op $\intcc{0}{1}$, dus $u_n \to
\int_0^1 \frac{\dd x}{1+x^2} = \arctan 1 = \dfrac\pi4$.

$\ln v_n = \dfrac1n \sum_{k=1}^{n} \ln\dfrac{n+k}{n} = \dfrac 1n
\sum_{k=1}^n \ln\Bigl(1 + \dfrac kn\Bigr) \to \int_0^1 \ln(1+x)\,\dd
x = \bigl[(1+x)\ln(1+x) - x\bigr]_0^1 = 2\ln 2 - 1$. Bijgevolg $v_n \to
\eu^{2\ln 2 - 1} = \dfrac 4\eu$. (Controle van de identificatie:
$\frac{(2n)!}{n!\,n^n} = \prod_{k=1}^{n} \frac{n+k}{n}$.)
\end{solution}

\begin{solution}{exo:b1:integration:4}
De limiet is $0$. Zij $M = \sup \abs f$ en $\varepsilon \in
\intoo{0}{1}$. Snijd bij $1 - \varepsilon$:
\[
\Bigl| \int_0^1 x^n f \Bigr|
\leq \int_0^{1 - \varepsilon} x^n \abs f + \int_{1-\varepsilon}^1
x^n \abs f
\leq M\,(1-\varepsilon)^n + M\varepsilon .
\]
Aangezien $(1 - \varepsilon)^n \to 0$ (\cref{exo:b1:seq:3}), is de
limsup van het linkerlid $\leq M\varepsilon$ voor elke $\varepsilon$:
de \omterm{thm:b1:integration:def}{integraal} streeft naar $0$.
\end{solution}

\begin{solution}{exo:b1:integration:5}
$Q(\lambda) = \int_a^b (f + \lambda g)^2 = \int f^2 + 2\lambda \int
fg + \lambda^2 \int g^2 \geq 0$ voor alle $\lambda$. Als $\int g^2 =
0$, dan is $g = 0$ (strikte positiviteit,
\cref{thm:b1:integration:props}~(4)) en de ongelijkheid is $0 \leq
0$. Anders is $Q$ een echte kwadratische vergelijking, overal $\geq 0$:
haar discriminant is $\leq 0$, d.w.z.\ $\bigl(\int fg\bigr)^2 \leq \int
f^2 \int g^2$.

Gelijkheid dan en slechts dan als de discriminant verdwijnt dan en
slechts dan als $Q(\lambda_0) = 0$ voor een zekere $\lambda_0$, d.w.z.\
$\int (f + \lambda_0 g)^2 = 0$, d.w.z.\ (opnieuw strikte positiviteit)
$f = -\lambda_0 g$: gelijkheid geldt precies wanneer $f$ en $g$
evenredig zijn.
\end{solution}

\begin{solution}{exo:b1:integration:6}
Zij $\Phi(a) = \int_a^{a+T} f$. Volgens de hoofdstelling
(\cref{thm:b1:integration:ftc}) is $\Phi$ \omterm{def:b1:derivative:def}{afleidbaar} met
$\Phi'(a) = f(a + T) - f(a) = 0$: constant.

Voor $x > 0$, schrijf $x = nT + r$, $0 \leq r < T$ ($n = \lfloor x/T
\rfloor$). Chasles:
\[
\int_0^x f = n \int_0^T f + \int_{nT}^{nT + r} f,
\qquad
\Bigl| \int_{nT}^{nT+r} f \Bigr| \leq T \sup_{\intcc{0}{T}} \abs f
= C .
\]
Dan is $\frac 1x \int_0^x f = \frac{nT}{x}\cdot\frac 1T \int_0^T f +
O\bigl(\frac 1x\bigr)$, en $\frac{nT}{x} \to 1$: de limiet is
$\frac1T \int_0^T f$.
\end{solution}

\begin{solution}{exo:b1:integration:7}
Zij $g(x) = f(x) - x$: \omterm{def:b1:continuity:continuous}{continu}, met
\[
\int_0^1 g = \int_0^1 f - \frac12 = 0 .
\]
Als $g$ nooit verdween, zou de tussenwaardestelling een constant teken
afdwingen (een continue functie op een \omterm{prop:b1:reals:intervals}{interval} die beide tekens
aanneemt, verdwijnt); zeg $g > 0$. Dan, door strikte positiviteit
(\cref{thm:b1:integration:props}~(4) toegepast op $g > 0$, wat $\int g
> 0$ geeft): tegenspraak met $\int g = 0$. Dus $g(c) = 0$ voor een
zekere $c$: $f(c) = c$.
\end{solution}

\begin{solution}{exo:b1:integration:8}
\begin{enumerate}
  \item Met \omterm{thm:b1:integration:parts}{partiële integratie} met $u' = \sin t$, $v = \sin^{n-1} t$:
        \[
        W_n = \bigl[-\cos t\sin^{n-1}t\bigr]_0^{\pi/2}
        + (n-1)\int_0^{\pi/2} \cos^2 t\,\sin^{n-2} t\,\dd t
        = (n-1)(W_{n-2} - W_n),
        \]
        dus $nW_n = (n-1)W_{n-2}$. Uit $W_0 = \frac\pi2$, $W_1 = 1$:
        \[
        W_{2p} = \frac{(2p-1)(2p-3)\cdots 1}{(2p)(2p-2)\cdots 2}\,
        \frac{\pi}{2} = \frac{(2p)!}{4^p (p!)^2}\,\frac\pi2,
        \qquad
        W_{2p+1} = \frac{(2p)(2p-2)\cdots 2}{(2p+1)(2p-1)\cdots 3}
        = \frac{4^p (p!)^2}{(2p+1)!} .
        \]
  \item Op $\intoo{0}{\frac\pi2}$ is $0 < \sin t < 1$, dus $\sin^{n+1}
        < \sin^n$ en $(W_n)$ is (strikt) dalend, positief. Insluiten
        met de recurrentie:
        \[
        \frac{n}{n+1} = \frac{W_{n+1}}{W_{n-1}}
        \leq \frac{W_{n+1}}{W_n} \leq 1
        \quad\implies\quad \frac{W_{n+1}}{W_n} \to 1 .
        \]
        Invariant: $a_n = (n+1)W_{n+1}W_n$ voldoet aan $a_n = a_{n-1}$
        door de recurrentie $(n+1)W_{n+1} = nW_{n-1}$, dus $a_n = a_0 =
        1 \cdot W_1 W_0 = \frac\pi2$. Dan
        \[
        n W_n^2 \sim (n+1) W_{n+1} W_n = \frac\pi2
        \quad\implies\quad
        W_n \sim \sqrt{\frac{\pi}{2n}} .
        \]
\end{enumerate}
\end{solution}

\begin{solution}{exo:b1:integration:9}
\begin{enumerate}
  \item Op $\intoo{0}{\pi}$: $x > 0$, $a - bx = b(\frac ab - x) =
        b(\pi - x) > 0$ en $\sin x > 0$, dus de integrand is $> 0$
        en $I_n > 0$ (strikte positiviteit). Grens: op
        $\intcc{0}{\pi}$ is $x \leq \pi$ en $a - bx \leq a$, dus $P
        \leq \frac{\pi^n a^n}{n!}$ en $I_n \leq
        \pi\,\frac{(\pi a)^n}{n!}$, wat naar $0$ streeft (de
        faculteit verslaat de meetkundige term: het is de algemene
        term van de convergente exponentiële reeks, cf.\
        \cref{ex:b1:seq:e}); in het bijzonder $I_n < 1$ voor grote
        $n$.
  \item Ontwikkel $x^n(a - bx)^n = \sum_{j=0}^{n} \binom nj a^{n-j}
        (-b)^j x^{n+j}$: dus $P = \frac{1}{n!}\sum_j c_j x^{n+j}$
        met gehele $c_j$. Dan is $P^{(k)}(0) = 0$ voor $k < n$
        (valuatie) en, voor $n \leq k \leq 2n$, $P^{(k)}(0) =
        \frac{k!}{n!} c_{k-n}$, een geheel getal omdat $n! \mid k!$.
        Bovendien $P(\pi - x) = P(x)$ (substitueer: $\pi - x$
        verwisselt de factoren, met $a - b(\pi - x) = bx$), dus
        $P^{(k)}(\pi) = \pm P^{(k)}(0)$: eveneens gehele getallen.
  \item Met $Q = P - P'' + P^{(4)} - \dots$ (eindig: $P$ heeft graad
        $2n$): $Q + Q'' = P$, en
        \[
        \bigl(Q'\sin x - Q\cos x\bigr)' = (Q + Q'')\sin x = P\sin x .
        \]
        Bijgevolg $I_n = \bigl[Q'\sin x - Q\cos x\bigr]_0^{\pi} =
        Q(\pi) + Q(0)$, een som van waarden $P^{(2k)}$ in $0$ en
        $\pi$: een geheel getal volgens (2).
  \item Voor grote $n$ is $I_n$ een geheel getal met $0 < I_n < 1$:
        onmogelijk. De veronderstelling $\pi = \frac ab$ faalt: $\pi$
        is irrationaal.
\end{enumerate}
\end{solution}

\begin{solution}{exo:b1:integration:10}
$C^1$-geval: met \omterm{thm:b1:integration:parts}{partiële integratie},
\[
\int_a^b f(t)\sin\lambda t\,\dd t
= \Bigl[-f(t)\frac{\cos\lambda t}{\lambda}\Bigr]_a^b
+ \frac{1}{\lambda}\int_a^b f'(t)\cos\lambda t\,\dd t,
\]
in absolute waarde begrensd door $\frac{2\sup\abs f + (b -
a)\sup\abs{f'}}{\lambda} \to 0$.

\omterm{def:b1:continuity:continuous}{Continu} geval: zij $\varepsilon > 0$ en kies een \omterm{def:b1:integration:step}{trapfunctie}
$\varphi$ met $\abs{f - \varphi} \leq \varepsilon$
(\cref{thm:b1:integration:approx} levert $\varphi \leq f \leq
\psi$ met marge $\leq\varepsilon$; neem $\varphi$). Dan
\[
\Bigl|\int_a^b f\sin\lambda t\Bigr|
\leq \int_a^b \abs{f - \varphi}
+ \Bigl|\int_a^b \varphi \sin\lambda t\Bigr|
\leq (b-a)\varepsilon
+ \sum_i \abs{c_i}\,\Bigl|\int_{x_{i-1}}^{x_i} \sin\lambda t\,\dd
t\Bigr| ,
\]
en elke $\bigl|\int \sin \lambda t\bigr| = \bigl|\frac{\cos\lambda
x_{i-1} - \cos\lambda x_i}{\lambda}\bigr| \leq \frac{2}{\lambda}$:
de tweede term streeft naar $0$. Bijgevolg is de limsup $\leq
(b-a)\varepsilon$ voor elke $\varepsilon$: de limiet is $0$.
\end{solution}

\begin{solution}{exo:b1:integration:11}
Zij $m = \min f$ en $M = \max f$, bereikt volgens de
extremumstelling. Aangezien $g \geq 0$: $m\,g \leq fg \leq M\,g$, dus
door monotonie
\[
m \int_a^b g \;\leq\; \int_a^b fg \;\leq\; M \int_a^b g .
\]
Als $\int_a^b g = 0$: strikte positiviteit
(\cref{thm:b1:integration:props}~(4)) dwingt $g \equiv 0$ af, beide
leden verdwijnen, en elke $c$ voldoet. Anders ligt $t = \frac{\int
fg}{\int g}$ in $\intcc{m}{M} = f(\intcc{a}{b})$
(\cref{thm:b1:continuity:evt,thm:b1:continuity:ivt}), dus $t =
f(c)$ voor een zekere $c$.

Het teken doet ertoe: op $\intcc{-1}{1}$ met $f(t) = g(t) = t$: $\int
fg = \int_{-1}^1 t^2 = \frac23$, terwijl $f(c)\int_{-1}^1 t\,\dd t
= 0$ voor elke $c$.
\end{solution}

\begin{solution}{exo:b1:integration:12}
Als $f \equiv 0$ is de bewering triviaal (elk punt is een nulpunt).
Veronderstel dus $f \not\equiv 0$ en neem aan dat het hoogstens $n$
verschillende nulpunten heeft in $\intoo{a}{b}$; zij $z_1 < \dots <
z_m$ ($m \leq n$) die nulpunten waar $f$ \emph{van teken wisselt}
(mogelijk geen). Stel $P(t) = \prod_{i=1}^{m}(t - z_i)$ (leeg product
$= 1$), van graad $m \leq n$. Op elk deelinterval gesneden door de
$z_i$ hebben zowel $f$ als $P$ een constant teken, en beide keren van
teken bij het passeren van een zekere $z_i$: het product $fP$ heeft
één constant teken op heel $\intoo{a}{b}$. \omterm{def:b1:continuity:continuous}{Continu} zijnde, niet
identiek nul, van constant teken, heeft het $\bigl|\int_a^b fP\bigr| >
0$ (strikte positiviteit toegepast op $\abs{fP}$). Maar $\int f P$ is
een lineaire combinatie van de momenten $\int f\,t^k$, $k \leq n$, alle
nul: tegenspraak. Bijgevolg heeft $f$ minstens $n + 1$ verschillende
nulpunten in $\intoo{a}{b}$.
\end{solution}

\begin{solution}{pb:b1:integration:1}
\textbf{1.} Zij $N = \lceil 2c \rceil$. Voor $n \geq N$:
$\frac{c^{n+1}/(n+1)!}{c^n/n!} = \frac{c}{n+1} \leq \frac12$, dus
$0 < \frac{c^n}{n!} \leq \frac{c^N}{N!}\,2^{-(n - N)} \to 0$:
knijp.

\textbf{2.} Fixeer $k$; inductie op $m$. Voor $m = 0$: $\int_0^1
x^k = \frac{1}{k+1} = \frac{k!\,0!}{(k+1)!}$. Stap, met \omterm{thm:b1:integration:parts}{partiële
integratie} ($u = (1-x)^m$, $v' = x^k$):
\[
\int_0^1 x^k (1-x)^m \dd x
= \Bigl[\frac{x^{k+1}}{k+1}(1-x)^m\Bigr]_0^1
+ \frac{m}{k+1}\int_0^1 x^{k+1}(1-x)^{m-1}\dd x
= \frac{m}{k+1}\cdot\frac{(k+1)!\,(m-1)!}{(k+m+1)!} ,
\]
wat $\frac{k!\,m!}{(k+m+1)!}$ is.

\textbf{3.} $k = m = n$: $\int_0^1 (x(1-x))^n =
\frac{(n!)^2}{(2n+1)!} = \frac{1}{(2n+1)\binom{2n}{n}}$. Aangezien
$x(1-x) \leq \frac14$ op $\intcc{0}{1}$, is de \omterm{thm:b1:integration:def}{integraal} $\leq
4^{-n}$, waaruit $\binom{2n}{n} \geq \frac{4^n}{2n+1}$ ---
consistent met $\binom{2n}{n}^{1/n} \to 4$
(\cref{pb:b1:seq:1}).

\textbf{4.} De integrand is \omterm{def:b1:continuity:continuous}{continu}, $\geq 0$, en positief op
$\intoo{0}{1}$, dus niet identiek nul: zijn \omterm{thm:b1:integration:def}{integraal} is $> 0$
(\cref{thm:b1:integration:props}~(4)). \omterm{def:b1:reals:bounds}{Bovengrens}: $(x(1-x))^n \leq
4^{-n}$ en $g \leq \sup\abs g$, daarna monotonie.

\textbf{5.} $A_0 = \eu - 1$; $A_1 = [x\eu^x]_0^1 - \int_0^1
\eu^x = \eu - (\eu - 1) = 1$. Met \omterm{thm:b1:integration:parts}{partiële integratie}: $A_n = [x^n
\eu^x]_0^1 - n\int_0^1 x^{n-1}\eu^x = \eu - n A_{n-1}$. Grenzen: de
integrand is positief, dus $A_n > 0$; en $\eu^x \leq \eu$ geeft $A_n
\leq \eu\int_0^1 x^n = \frac{\eu}{n+1}$.

\textbf{6.} $A_0 = -1 + 1\cdot\eu$. Als $A_{n-1} = \alpha_{n-1} +
\beta_{n-1}\eu$ met gehele termen, dan
\[
A_n = \eu - n\alpha_{n-1} - n\beta_{n-1}\eu
= \underbrace{(-n\,\alpha_{n-1})}_{\alpha_n}
+ \underbrace{(1 - n\,\beta_{n-1})}_{\beta_n}\,\eu ,
\]
beide geheel.

\textbf{7.} Als $\eu = \frac pq$: $q A_n = q\alpha_n + p\beta_n
\in \Z$, en $0 < qA_n \leq \frac{q\eu}{n+1} < 1$ voor grote $n$:
een geheel getal strikt tussen $0$ en $1$ --- onmogelijk. Dus $\eu
\notin \Q$. In \cref{exo:b1:seq:9} was het gevangen geheel getal
$q!\,\frac pq - q!\,a_q$; hier is het $qA_n$: dezelfde val,
integraalbrandstof.

\textbf{8.} $n = 0$: $R_0 = \int_0^1 \eu^t = \eu - 1$, dus $\eu =
1 + R_0$. Met \omterm{thm:b1:integration:parts}{partiële integratie} ($u = \eu^t$, $v =
-\frac{(1-t)^{n+1}}{n+1}$):
\[
R_n = \frac{1}{n!}\Bigl(\Bigl[-\frac{(1-t)^{n+1}}{n+1}
\eu^t\Bigr]_0^1 + \frac{1}{n+1}\int_0^1 (1-t)^{n+1}\eu^t\Bigr)
= \frac{1}{(n+1)!} + R_{n+1} ,
\]
dus de formule plant zich voort van $n$ naar $n + 1$. Grenzen: $1 \leq
\eu^t \leq \eu$ op $\intcc{0}{1}$ en $\int_0^1 (1-t)^n =
\frac{1}{n+1}$ geven $\frac{1}{(n+1)!} \leq R_n \leq
\frac{\eu}{(n+1)!}$.

\textbf{9.} $\sum_{k=0}^{9} \frac{1}{k!} = \frac{986410}{362880}
= 2.71828152\dots$, en
\[
\frac{1}{10!} = 2.76\cdot10^{-7} \leq R_9 \leq
\frac{2.75}{10!} = 7.58\cdot10^{-7} ,
\]
dus $2.7182818 \leq \eu \leq 2.7182823$: met bewijs, $\eu =
2.718282$ tot op zes decimalen (werkelijke waarde $2.7182818\dots$).

\textbf{10.} $f(1 - x) = \frac{(1-x)^n x^n}{n!} = f(x)$. Op
$\intoo{0}{1}$: $0 < x(1-x) \leq \frac14$, dus $0 < f \leq
\frac{1}{4^n\,n!}$.

\textbf{11.} $x^n(1-x)^n = \sum_{j=0}^{n} (-1)^j\binom
nj\,x^{n+j}$, dus $f = \frac{1}{n!}\sum_j c_j\,x^{n+j}$ met
$c_j \in \Z$. Bijgevolg $f^{(k)}(0) = 0$ voor $k < n$ of $k > 2n$,
en voor $n \leq k \leq 2n$: $f^{(k)}(0) = \frac{k!}{n!}\,
c_{k-n}$, een geheel getal omdat $n! \mid k!$. De symmetrie geeft
$f^{(k)}(1) = (-1)^k f^{(k)}(0) \in \Z$.

\textbf{12.} $b^n \pi^{2n-2k} = b^n\bigl(\frac
ab\bigr)^{\,n-k} = a^{\,n-k}\,b^{\,k} \in \Z$, dus $G(0) =
\sum_k (-1)^k a^{n-k} b^k f^{(2k)}(0)$ en evenzo $G(1)$ zijn
gehele getallen volgens vraag 11.

\textbf{13.} In $\pi^2 G + G''$ draagt de term $k$ van $\pi^2 G$
$\pi^{2n-2k+2}f^{(2k)}$ en de term $j = k - 1$ van $G''$ draagt
$(-1)^{k-1}\pi^{2n-2k+2}f^{(2k)}$: alles heft zich op behalve $k = 0$
in de eerste som en $j = n$ in de tweede, d.w.z.
\[
G'' + \pi^2 G = b^n\bigl(\pi^{2n+2} f + (-1)^n f^{(2n+2)}\bigr)
= b^n \pi^{2n+2} f = \pi^2 a^n f
\]
($f$ heeft graad $2n$, dus $f^{(2n+2)} = 0$; en $b^n\pi^{2n} =
a^n$). Dan
\[
\bigl(G'\sin\pi x - \pi G\cos\pi x\bigr)'
= (G'' + \pi^2 G)\sin \pi x = \pi^2 a^n f\sin\pi x .
\]

\textbf{14.} Door te integreren over $\intcc{0}{1}$:
\[
\pi^2 a^n \int_0^1 f\sin(\pi x)\,\dd x
= \bigl[G'\sin\pi x - \pi G\cos\pi x\bigr]_0^1
= \pi\bigl(G(1) + G(0)\bigr) ,
\]
dus $\pi a^n \int_0^1 f\sin\pi x = G(0) + G(1) \in \Z$. Op
$\intoo{0}{1}$ is $f > 0$ en $\sin\pi x > 0$: het linkerlid is
positief, dus $G(0) + G(1) \geq 1$; en $\sin \leq 1$ met vraag 10
begrenst het door $\frac{\pi a^n}{4^n\,n!}$.

\textbf{15.} Volgens vraag 1 (met $c = \frac a4$),
$\frac{\pi a^n}{4^n n!} \to 0$: voor grote $n$ is het $< 1$,
in tegenspraak met $G(0) + G(1) \geq 1$. Dus geen enkele breuk
$\frac ab$ is gelijk aan $\pi^2$: Legendres stelling. Als $\pi$
rationaal was, zou $\pi^2$ dat ook zijn: dus $\pi \notin \Q$ --- en de
implicatie loopt enkel deze kant op ($\sqrt2$ is irrationaal met
rationaal kwadraat), wat de reden is waarom $\pi^2 \notin \Q$ strikt
sterker is dan \cref{exo:b1:integration:9}.

\textbf{16.} Geheeltalligheid: vragen 11--12 (eindpuntafgeleiden);
kleinheid: vragen 10 en 1; brug: vragen 13--14 (de getelescopeerde
dubbele \omterm{thm:b1:integration:parts}{partiële integratie}). Zonder $\frac{1}{n!}$ blijven de
eindpuntgegevens gehele getallen (zelfs gemakkelijker), maar de grens
wordt $\frac{\pi a^n}{4^n}$, die enkel naar $0$ streeft wanneer $a <
4$ --- en elke kandidaat heeft $a = b\,\pi^2 > 9$. De faculteit is
precies wat de meetkundige groei $a^n$ voorbijstreeft: geen faculteit,
geen stelling.

\textbf{17.} Voor $a \leq 10$ is het gehele getal $G(0) + G(1)$
positief en hoogstens $\pi\,(10/4)^n/n!$. Bij $n = 7$:
$2.5^7 = 610.35\dots$, dus de grens is $\frac{\pi \times
610.35}{5040} \approx 0.38 < 1$ (bij $n = 6$ is het nog steeds
$1.07$): de tegenspraak landt bij het zevende stadium, expliciet.

\textbf{18.} Twee \omterm{thm:b1:integration:parts}{partiële integraties}:
\[
\int_0^1 x(1-x)\sin\pi x\,\dd x
= \frac1\pi\int_0^1 (1 - 2x)\cos\pi x\,\dd x
= \frac{2}{\pi^2}\int_0^1 \sin\pi x\,\dd x
= \frac{2}{\pi^2}\cdot\frac{2}{\pi} = \frac{4}{\pi^3}
\]
(de haaktermen verdwijnen: $x(1-x)$ in $0, 1$, en $\sin\pi x$ in $0,
1$). Symbolisch luidt de $n = 1$-telescoop (geen aanname over $\pi$)
$\pi^3\int_0^1 f_1\sin\pi x = -(f_1''(0) + f_1''(1))$ met $f_1 = x(1
- x)$, $f_1'' = -2$: rechterlid $4$ --- de twee berekeningen komen
overeen.

\textbf{19.} $\eu^x$ lost $y' = y$ op en $\sin\pi x$ lost $y'' =
-\pi^2 y$ op: lineaire vergelijkingen met constante coëfficiënten,
zodat \omterm{thm:b1:integration:parts}{partiële integratie} de kern naar zichzelf terugvoert en alle
randgegevens binnen $\Z + \Z\eu$ houdt (resp.\ gehele \omterm{def:b1:poly:def}{veeltermen} in
$\pi^2$). Die geslotenheidseigenschap is wat de machine nodig heeft.
Verfijnd met kernen aangepast aan meerdere punten tegelijk levert
hetzelfde mechanisme Hermites stelling ($\eu$ \omterm{pb:b1:logic:1}{transcendent}, 1873) en
die van Lindemann ($\pi$ \omterm{pb:b1:logic:1}{transcendent}, 1882) --- voorbij dit volume.

\textbf{20.} Veeltermdeling (of vermenigvuldig terug en controleer):
\[
x^4(1-x)^4 = (x^6 - 4x^5 + 5x^4 - 4x^2 + 4)(1 + x^2) - 4 .
\]
Door de weergegeven identiteit gedeeld door $1 + x^2$ te integreren:
\[
\int_0^1 \frac{x^4(1-x)^4}{1+x^2}\dd x
= \Bigl(\frac17 - \frac46 + 1 - \frac43 + 4\Bigr)
- 4\arctan 1
= \frac{22}{7} - \pi ,
\]
met $\frac17 - \frac23 + 1 - \frac43 + 4 = 3 + \frac17$ en
$\arctan 1 = \frac\pi4$.

\textbf{21.} De integrand is \omterm{def:b1:continuity:continuous}{continu}, positief op
$\intoo{0}{1}$: de \omterm{thm:b1:integration:def}{integraal} is $> 0$, dus $\pi < \frac{22}{7}$.
Bovendien $\frac12 \leq \frac{1}{1+x^2} \leq 1$ op
$\intcc{0}{1}$ en $\int_0^1 (x(1-x))^4 = \frac{(4!)^2}{9!} =
\frac{1}{630}$ (vraag 3):
\[
\frac{1}{1260} \leq \frac{22}{7} - \pi \leq \frac{1}{630}
\quad\Longrightarrow\quad
3.14126 < \frac{22}{7} - \frac{1}{630} \leq \pi \leq
\frac{22}{7} - \frac{1}{1260} < 3.14207 .
\]

\textbf{22.} Modulo $x^2 + 1$: $x^2 \equiv -1$, dus $x^{4m} =
(x^2)^{2m} \equiv 1$ en $(1 - x)^2 = 1 - 2x + x^2 \equiv -2x$,
bijgevolg $(1-x)^{4m} \equiv (-2x)^{2m} = 4^m (x^2)^m \equiv
(-4)^m$: de rest is de constante $(-4)^m$, en het quotiënt $Q_m$
heeft gehele coëfficiënten (deling door een monische gehele
\omterm{def:b1:poly:def}{veelterm}). Door de identiteit door $1 + x^2$ te delen en te
integreren:
\[
J_m := \int_0^1 \frac{(x(1-x))^{4m}}{1+x^2}\dd x
= s_m + (-4)^m\,\frac{\pi}{4},
\qquad s_m = \int_0^1 Q_m \in \Q .
\]
Oplossen naar $\pi$: met $r_m = (-1)^{m+1}\,4^{\,1-m} s_m \in
\Q$, $\;\abs{\pi - r_m} = 4^{\,1-m} J_m \leq
4^{\,1-m}\cdot4^{-4m} = 4^{\,1-5m}$. Voor $m = 1$: $s_1 =
\frac{22}{7}$, $r_1 = \frac{22}{7}$, grens $4^{-4} =
\frac{1}{256}$ --- opnieuw vragen 20--21.

\textbf{23.} $\bigl|\pi - \frac{22}{7}\bigr| = \frac{22}{7} -
\pi \approx 1.26\cdot10^{-3}$, zestien keer beter dan de
orde-$2$-benchmark $\frac{1}{7^2} \approx 2.0\cdot10^{-2}$
gegarandeerd door Dirichlet (\cref{pb:b1:derivative:1}, vraag 4);
en $\bigl|\pi - \frac{355}{113}\bigr| \approx 2.7\cdot10^{-7}$
verslaat $\frac{1}{113^2} \approx 7.8\cdot10^{-5}$ met een factor
$\approx 300$. Geen tegenspraak met iets bewezen: Liouville-
ongelijkheden \emph{onderbegrenzen} benaderingsfouten enkel voor
algebraïsche getallen, en geen zulke grens voor $\pi$ is beschikbaar
op dit niveau --- $\pi$ is vrij om spectaculair goed benaderd te
worden.

\textbf{24.} Lemma: veronderstel $x = \frac pq$ en $0 < \abs{a_n +
b_n x} \to 0$. Dan is $\abs{a_n + b_n x} = \frac{\abs{q a_n + p
b_n}}{q}$, met $q a_n + p b_n$ een niet-nul geheel getal (niet-nul
omdat de absolute waarde $> 0$ is): dus $\abs{a_n + b_n x}
\geq \frac1q$ voor elke $n$, in tegenspraak met de convergentie naar
$0$. Gevallen: vraag 7 ($x = \eu$, $a_n = \alpha_n$, $b_n =
\beta_n$); \cref{exo:b1:seq:9} ($x = \eu$ opnieuw, met $a_q =
-q!\sum_{k \leq q}\frac{1}{k!}$, $b_q = q!$); en
\cref{pb:b1:derivative:1}, vraag 1, is haar meetkundige vorm.
In Deel III en in \cref{exo:b1:integration:9} draait de val
\emph{binnenin} de tegenspraak: rationaliteit aannemen zet een
uitdrukking om in een geheel getal, dat de analyse vervolgens
inknijpt in $\intoo{0}{1}$ --- hetzelfde principe, getransponeerd.

\textbf{25.} (i) Geheeltalligheid woont in de eindpuntrekenkunde van
\omterm{def:b1:poly:def}{veeltermen} (vragen 6, 11--12), kleinheid in de grenzen die
$\sup$-monotonie geeft (vragen 4, 10), de brug in \omterm{thm:b1:integration:parts}{partiële integratie}
(vragen 8, 13--14) --- alle drie zijn stellingen van dit hoofdstuk.
(ii) De \omterm{thm:b1:integration:def}{integraal} levert wat de middelwaardestelling niet kon: een
\emph{exacte} identiteit tussen het analytische object en de
rekenkundige gegevens (gelijkheid, niet enkel een ongelijkheid met een
onbekende $c$), wat de reden is waarom de machine $\pi^2$ bereikt
terwijl \cref{pb:b1:derivative:1} enkel benaderingsexponenten
bereikte. (iii) Geëxtraheerd: $\eu \notin \Q$, $\pi^2 \notin \Q$
(dus $\pi \notin \Q$), $\eu = 2.718282$ gecertificeerd,
$\frac{22}{7} - \frac{1}{630} \leq \pi \leq \frac{22}{7} -
\frac{1}{1260}$, en $\binom{2n}{n} \geq \frac{4^n}{2n+1}$. (iv)
Grens: Hermite en Lindemann duwen dezelfde machine naar
transcendentie; en Apéry (1979) liet de geheeltalligheidsval draaien
op $\zeta(3)$ --- de machine produceert nog steeds twintigste-eeuwse
wiskunde.
\end{solution}
