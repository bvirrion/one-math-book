\section*{Hoofdstuk \ref{ch:b1:linmaps} --- Lineaire afbeeldingen}

\begin{solution}{exo:b1:linmaps:1}
(1) \omterm{def:b1:linmaps:def}{Lineair}: de \omterm{prop:b1:vspaces:coordinates}{coördinaten} zijn lineaire uitdrukkingen. (2) Niet
\omterm{def:b1:linmaps:def}{lineair}: $u(2(1,1)) = 4 \neq 2 = 2u(1,1)$. (3) \omterm{def:b1:linmaps:def}{Lineair}: differentiëren en
vermenigvuldigen met $X$ zijn dat, en sommen van \omterm{def:b1:linmaps:def}{lineaire afbeeldingen}
zijn dat. (4) \omterm{def:b1:linmaps:def}{Lineair}: evaluatie respecteert de puntsgewijze bewerkingen.
\end{solution}

\begin{solution}{exo:b1:linmaps:2}
\omterm{def:b1:linmaps:kerim}{Kern}: los $x + y - z = 0$, $2x + y + z = 0$, $3x + 2y = 0$ op.
Uit de derde, $y = -\frac{3x}{2}$; de eerste geeft $z = x + y =
-\frac x2$; controle in de tweede: $2x - \frac{3x}{2} - \frac x2 = 0$:
voldaan. Dus $\ker u = \operatorname{Vect}\bigl((2, -3, -1)\bigr)$
(door $x = 2$ te nemen), dimensie $1$.

Dimensiestelling: $\operatorname{rk} u = 3 - 1 = 2$. \omterm{def:b1:linmaps:kerim}{Beeld}: opgespannen
door de \omterm{def:b1:linmaps:kerim}{beelden} van de canonieke \omterm{def:b1:vspaces:free}{basis}, $u(e_1) = (1,2,3)$, $u(e_2) =
(1,1,2)$, $u(e_3) = (-1,1,0)$; de eerste twee zijn \omterm{def:b1:vspaces:free}{vrij}, en de \omterm{def:b1:linmaps:rank}{rang}
is $2$: \omterm{def:b1:vspaces:free}{basis} $\bigl((1,2,3), (1,1,2)\bigr)$.

Niet \omterm{def:b1:logic:inj}{injectief} ($\ker \neq \{0\}$), niet \omterm{def:b1:logic:inj}{surjectief} (\omterm{def:b1:linmaps:rank}{rang} $2 < 3$):
consistent met \cref{cor:b1:linmaps:samedim}.
\end{solution}

\begin{solution}{exo:b1:linmaps:3}
\emph{\omterm{def:b1:linmaps:kerim}{Kern}:} $P = P'$ dwingt $\deg P = \deg P'$ af tenzij $P = 0$;
maar $\deg P' < \deg P$ voor $P \neq 0$: dus $\ker u = \{0\}$, en $u$,
een \omterm{def:b1:logic:inj}{injectief} endomorfisme van de eindigdimensionale $\R_n[X]$, is een
isomorfisme (\cref{cor:b1:linmaps:samedim}).

\emph{Inverse:} zij $v(P) = P + P' + P'' + \dots + P^{(n)}$ (een
eindige som op $\R_n[X]$). Dan
\[
v\bigl(u(P)\bigr) = \sum_{k=0}^{n} (P - P')^{(k)}
= \sum_{k=0}^{n} P^{(k)} - \sum_{k=0}^{n} P^{(k+1)}
= P - P^{(n+1)} = P ,
\]
telescoperend, aangezien $P^{(n+1)} = 0$. Dus $v = u^{-1}$.
\end{solution}

\begin{solution}{exo:b1:linmaps:4}
$\operatorname{im}(v \circ u) = v(\operatorname{im} u) \subseteq
\operatorname{im} v$: \omterm{def:b1:linmaps:rank}{rang} $\leq \operatorname{rk} v$. En $v$
beperkt tot $\operatorname{im} u$ heeft \omterm{def:b1:linmaps:kerim}{beeld}
$\operatorname{im}(vu)$ met de dimensiestelling binnen
$\operatorname{im} u$: $\operatorname{rk}(vu) \leq
\dim\operatorname{im} u = \operatorname{rk} u$.
\end{solution}

\begin{solution}{exo:b1:linmaps:5}
$u^2 = 0$ betekent $u(u(x)) = 0$ voor alle $x$: elke $u(x)$ ligt in
$\ker u$, d.w.z.\ $\operatorname{im} u \subseteq \ker u$. Dan geeft de
dimensiestelling:
\[
\dim E = \dim\ker u + \operatorname{rk} u \geq 2\operatorname{rk} u .
\]
Voorbeeld met gelijkheid in $\R^2$: $u(x, y) = (y, 0)$: $u^2 = 0$,
$\operatorname{rk} u = 1 = \frac{\dim E}{2}$.
\end{solution}

\begin{solution}{exo:b1:linmaps:6}
$(pq)^2 = pqpq = ppqq = pq$ (commutatie): een \omterm{def:b1:linmaps:projection}{projectie}
(\cref{thm:b1:linmaps:projchar}).

\omterm{def:b1:linmaps:kerim}{Beeld}: $\operatorname{im}(pq) \subseteq \operatorname{im} p$, en
$= \operatorname{im}(qp) \subseteq \operatorname{im} q$: bevat
in de doorsnede. Omgekeerd, als $x \in \operatorname{im} p \cap
\operatorname{im} q$, dan $p(x) = x$ en $q(x) = x$ (vaste punten
karakteriseren het \omterm{def:b1:linmaps:kerim}{beeld} van een \omterm{def:b1:linmaps:projection}{projectie}), dus $pq(x) = x$: $x \in
\operatorname{im}(pq)$.

\omterm{def:b1:linmaps:kerim}{Kern}: $\ker p \subseteq \ker(qp) = \ker(pq)$ en evenzo $\ker q
\subseteq \ker(pq)$: de som is bevat. Omgekeerd, zij $pq(x) =
0$, en schrijf
\[
x = \underbrace{q(x)}_{\in\, \ker p} +
\underbrace{(x - q(x))}_{\in\, \ker q} :
\]
de eerste term voldoet aan $p(q(x)) = 0$, dus ligt hij in $\ker p$; de
tweede ligt in $\ker q$ aangezien $q(x - q(x)) = q(x) - q^2(x) = 0$.
Bijgevolg $x \in \ker p + \ker q$.
\end{solution}

\begin{solution}{exo:b1:linmaps:7}
Merk eerst de algemene inclusies op $\ker u \subseteq \ker u^2$ en
$\operatorname{im} u^2 \subseteq \operatorname{im} u$, en via de
dimensiestelling, (2) $\iff$ (3) (gelijke \omterm{def:b1:linmaps:kerim}{kernen} $\iff$ gelijke \omterm{def:b1:linmaps:rank}{rangen}
$\iff$ gelijke \omterm{def:b1:linmaps:kerim}{beelden}, gegeven de inclusies).

(1 $\Rightarrow$ 2): zij $u^2(x) = 0$; dan $u(x) \in \ker u \cap
\operatorname{im} u = \{0\}$, dus $x \in \ker u$.

(2 $\Rightarrow$ 1): via Grassmann en de dimensiestelling, $\dim(\ker u +
\operatorname{im} u) = \dim\ker u + \operatorname{rk} u -
\dim(\ker u \cap \operatorname{im} u) = \dim E - \dim(\ker u \cap
\operatorname{im} u)$: de som is $E$ dan en slechts dan als de doorsnede
$\{0\}$ is. Zij $y \in \ker u \cap \operatorname{im} u$: $y = u(x)$ en
$u(y) = 0$, dus $u^2(x) = 0$, dus (via (2)) $u(x) = 0$: $y = 0$. Bijgevolg
$E = \ker u \oplus \operatorname{im} u$.
\end{solution}

\begin{solution}{exo:b1:linmaps:8}
Als $\varphi = 0$: dan dwingt $\ker\varphi = E \subseteq \ker\psi$
$\psi = 0 = 0\cdot\varphi$ af. Als $\varphi \neq 0$: $\ker\varphi$ is een
\omterm{def:b1:linmaps:forms}{hypervlak}; kies $a \notin \ker\varphi$. Stel $\lambda =
\frac{\psi(a)}{\varphi(a)}$. De vorm $\psi - \lambda\varphi$
verdwijnt op $\ker\varphi$ (beide doen dat, via de inclusie) en in $a$:
hij verdwijnt op $\ker\varphi \oplus Ka = E$. Dus $\psi =
\lambda\varphi$.
\end{solution}

\begin{solution}{exo:b1:linmaps:9}
Veronderstel $\lambda_0 x + \lambda_1 u(x) + \dots + \lambda_{n-1}
u^{n-1}(x) = 0$. Pas $u^{n-1}$ toe: alle termen met een factor $u^{\geq
n}$ verdwijnen, waarbij $\lambda_0 u^{n-1}(x) = 0$ overblijft, dus
$\lambda_0 = 0$. Pas $u^{n-2}$ toe op de resterende relatie: $\lambda_1
u^{n-1}(x) = 0$, dus $\lambda_1 = 0$; enzovoort. De familie is \omterm{def:b1:vspaces:free}{vrij};
aangezien ze grootte $n = \dim E$ heeft, is ze een \omterm{def:b1:vspaces:free}{basis}
(\cref{prop:b1:findim:twoofthree}). (In deze \omterm{def:b1:vspaces:free}{basis} werkt $u$ als een
shift --- het model van maximale nilpotentie.)
\end{solution}

\begin{solution}{exo:b1:linmaps:10}
\begin{enumerate}
  \item $f \circ f = -\mathrm{id}$ is \omterm{def:b1:logic:inj}{bijectief}, dus $f$ is dat
        (\cref{prop:b1:logic:comp} aangepast: $f$ heeft de
        tweezijdige inverse $-f$). Als $f(x) = \lambda x$ met $x \neq 0$:
        door $f$ toe te passen, $-x = \lambda^2 x$, dus $\lambda^2 = -1$:
        onmogelijk in $\R$.
  \item Bouw de familie gulzig op. Neem $x_1 \neq 0$: $(x_1,
        f(x_1))$ is \omterm{def:b1:vspaces:free}{vrij} via (1). Als $\operatorname{Vect}$ van de
        huidige familie $\bigl(x_1, f(x_1), \dots, x_k,
        f(x_k)\bigr)$, noem het $V_k$ --- een \omterm{def:b1:vspaces:subspace}{deelruimte} stabiel onder $f$
        (elke voortbrenger beeldt af op een andere voortbrenger of zijn
        tegengestelde: $f(f(x_i)) = -x_i$) --- niet heel $E$ is, kies
        $x_{k+1} \notin V_k$. \emph{Bewering: de vergrote familie is \omterm{def:b1:vspaces:free}{vrij}.}
        Veronderstel $\alpha x_{k+1} + \beta f(x_{k+1}) + v = 0$ met $v
        \in V_k$ en $(\alpha, \beta) \neq (0,0)$. Pas $f$ toe:
        $\alpha f(x_{k+1}) - \beta x_{k+1} + f(v) = 0$ met $f(v)
        \in V_k$. Elimineer $f(x_{k+1})$ tussen de twee relaties
        (vermenigvuldig de eerste met $\alpha$, de tweede met $-\beta$, en
        tel op):
        \[
        (\alpha^2 + \beta^2)\, x_{k+1} \in V_k ,
        \]
        en $\alpha^2 + \beta^2 \neq 0$ dwingt $x_{k+1} \in V_k$ af:
        tegenspraak. Dus de constructie gaat verder, waarbij ze vectoren
        \emph{twee per twee} toevoegt, tot $V_k = E$: de uiteindelijke
        familie is een \omterm{def:b1:vspaces:free}{basis} van even grootte, en $n$ is even.
\end{enumerate}
\end{solution}

\begin{solution}{exo:b1:linmaps:11}
\omterm{def:b1:reals:bounds}{Bovengrens}: voor elke $x$, $(u + v)(x) = u(x) + v(x) \in
\operatorname{im} u + \operatorname{im} v$, dus
\[
\operatorname{rk}(u + v)
\leq \dim(\operatorname{im} u + \operatorname{im} v)
\leq \operatorname{rk} u + \operatorname{rk} v
\]
(Grassmann, \cref{thm:b1:findim:grassmann}). Ondergrens: pas
de \omterm{def:b1:reals:bounds}{bovengrens} toe op het paar $(u + v, -v)$, waarvan de som $u$ is:
\[
\operatorname{rk} u \leq \operatorname{rk}(u + v) +
\operatorname{rk}(-v) = \operatorname{rk}(u + v) +
\operatorname{rk} v,
\]
dus $\operatorname{rk} u - \operatorname{rk} v \leq
\operatorname{rk}(u+v)$; door $u$ en $v$ te verwisselen krijgt men de
absolute waarde.
\end{solution}

\begin{solution}{exo:b1:linmaps:12}
\emph{Exacte formule.} Zij $v'$ de beperking van $v$ tot de
\omterm{def:b1:vspaces:subspace}{deelruimte} $\operatorname{im} w$. Haar \omterm{def:b1:linmaps:kerim}{beeld} is $v(w(F)) =
\operatorname{im}(v \circ w)$, en haar \omterm{def:b1:linmaps:kerim}{kern} is $\ker v \cap
\operatorname{im} w$. Dimensiestelling voor $v'$ op de ruimte
$\operatorname{im} w$:
\[
\operatorname{rk} w = \dim\operatorname{im} w
= \operatorname{rk}(v \circ w) + \dim(\ker v \cap
\operatorname{im} w) .
\]

\emph{Frobenius.} Pas de exacte formule tweemaal toe, op $w$ en op
$w \circ u$:
\[
\operatorname{rk} w - \operatorname{rk}(vw)
= \dim\bigl(\ker v \cap \operatorname{im} w\bigr),
\qquad
\operatorname{rk}(wu) - \operatorname{rk}(vwu)
= \dim\bigl(\ker v \cap \operatorname{im}(wu)\bigr) .
\]
Aangezien $\operatorname{im}(w \circ u) \subseteq \operatorname{im}
w$, is de tweede doorsnede bevat in de eerste, en is haar
dimensie niet groter:
\[
\operatorname{rk}(wu) - \operatorname{rk}(vwu)
\;\leq\; \operatorname{rk} w - \operatorname{rk}(vw) ,
\]
wat zich herschikt tot de ongelijkheid van Frobenius. Met $w =
\mathrm{id}_F$ (\omterm{def:b1:linmaps:rank}{rang} $\dim F$, en $\operatorname{im}\,
\mathrm{id}_F = F$): $\operatorname{rk} v + \operatorname{rk} u
\leq \dim F + \operatorname{rk}(vu)$, de ongelijkheid van Sylvester ---
opnieuw bewezen, in matrixvorm, in \cref{exo:b1:matrices:10}.
\end{solution}

\begin{solution}{pb:b1:linmaps:1}
\textbf{1.} $(\mathrm{id} - p)^2 = \mathrm{id} - 2p + p^2 =
\mathrm{id} - p$: een projector. Als $y = x - p(x)$, dan $p(y) =
p(x) - p^2(x) = 0$, en omgekeerd geeft $x \in \ker p$ $x =
(\mathrm{id} - p)(x)$: $\operatorname{im}(\mathrm{id} - p) = \ker
p$. En $(\mathrm{id} - p)(x) = 0 \iff p(x) = x \iff x \in
\operatorname{im} p$ (vaste punten,
\cref{thm:b1:linmaps:projchar}): $\ker(\mathrm{id} - p) =
\operatorname{im} p$.

\textbf{2.} $(\lambda\,\mathrm{id} + \mu p)^2 =
\lambda^2\,\mathrm{id} + (2\lambda\mu + \mu^2)\,p$. Het paar
$(\mathrm{id}, p)$ is \omterm{def:b1:vspaces:free}{vrij} in $\mathcal{L}(E)$: $p = c\,
\mathrm{id}$ zou $c^2 = c$ geven, dus $p = 0$ of $\mathrm{id}$,
uitgesloten. Door coëfficiënten te identificeren is de \omterm{def:b1:logic:map}{afbeelding} een
projector dan en slechts dan als $\lambda^2 = \lambda$ en $2\lambda\mu +
\mu^2 = \mu$. Voor $\lambda = 0$: $\mu \in \{0, 1\}$. Voor $\lambda = 1$:
$\mu^2 + \mu = 0$, $\mu \in \{0, -1\}$. Precies vier projectoren in het
vlak: $0$, $p$, $\mathrm{id}$, $\mathrm{id} - p$.

\textbf{3.} $(\lambda\,\mathrm{id} + \mu p)(\lambda'\,\mathrm{id}
+ \mu' p) = \lambda\lambda'\,\mathrm{id} + (\lambda\mu' +
\mu\lambda' + \mu\mu')\,p$: het vlak is stabiel onder
samenstelling. Aangezien $p^k = p$ voor elke $k \geq 1$, voor $Q =
\sum_k a_k X^k$:
\[
Q(p) = a_0\,\mathrm{id} + \Bigl(\sum_{k \geq 1} a_k\Bigr) p
= Q(0)\,\mathrm{id} + \bigl(Q(1) - Q(0)\bigr)\,p .
\]

\textbf{4.} Op $\operatorname{im} p$ (waar $p$ als de
identiteit werkt) vermenigvuldigt $\lambda\,\mathrm{id} + \mu p$ met
$\lambda + \mu$; op $\ker p$, met $\lambda$. Aangezien $E =
\operatorname{im} p \oplus \ker p$, is de \omterm{def:b1:logic:map}{afbeelding} \omterm{def:b1:logic:inj}{bijectief} dan en
slechts dan als $\lambda \neq 0$ en $\lambda + \mu \neq 0$. Door
$\lambda\alpha = 1$, $\lambda\beta + \mu\alpha + \mu\beta = 0$ op te
lossen in de samenstellingsregel van vraag 3:
\[
(\lambda\,\mathrm{id} + \mu p)^{-1}
= \frac1\lambda\,\mathrm{id} -
\frac{\mu}{\lambda(\lambda + \mu)}\,p ,
\]
waarvan de werking $1/\lambda$ is op $\ker p$ en $1/(\lambda +
\mu)$ op $\operatorname{im} p$, zoals het moet zijn.

\textbf{5.} Schrijf $F = \operatorname{im} p = \operatorname{im}
p'$. Voor elke $x$, $p'(x) \in F$ en $p$ houdt $F$ puntsgewijs vast:
$p(p'(x)) = p'(x)$, d.w.z.\ $p\,p' = p'$; symmetrisch $p'\,p =
p$. Wanneer twee \omterm{def:b1:linmaps:projection}{projecties} hun \omterm{def:b1:linmaps:kerim}{beeld} delen, beslist degene die
\emph{eerst} wordt toegepast: haar uitvoer ligt al in $F$, waar de
buitenste \omterm{def:b1:linmaps:projection}{projectie} als de identiteit werkt en niets verandert.

\textbf{6.} $(p + q)^2 = p^2 + pq + qp + q^2 = (p + q) + pq +
qp$, dus $p + q$ projector dwingt $pq + qp = 0$ af. Stel links samen
met $p$: $pq + pqp = 0$; rechts met $p$: $pqp + qp = 0$. Door af te
trekken, $pq = qp$; dan $pq + qp = 2pq = 0$ en de karakteristiek is
niet $2$: $pq = qp = 0$.

\textbf{7.} Met $pq = qp = 0$ geeft dezelfde ontwikkeling $(p +
q)^2 = p + q$. \omterm{def:b1:linmaps:kerim}{Beeld}: $\operatorname{im}(p + q) \subseteq
\operatorname{im} p + \operatorname{im} q$ altijd. Omgekeerd,
voor $x \in \operatorname{im} p$: $q(x) = q(p(x)) = 0$, dus $(p +
q)(x) = p(x) = x$ en $x \in \operatorname{im}(p+q)$; evenzo voor
$\operatorname{im} q$. Directheid: $x \in \operatorname{im} p
\cap \operatorname{im} q$ geeft $x = p(x) = p(q(x)) = 0$.
\omterm{def:b1:linmaps:kerim}{Kern}: als $p(x) + q(x) = 0$, geeft $p$ toepassen $p(x) +
p(q(x)) = p(x) = 0$, en $q$ toepassen geeft $q(x) = 0$: $\ker(p
+ q) = \ker p \cap \ker q$ (de omgekeerde inclusie is duidelijk).

\textbf{8.} $p - q$ is een projector dan en slechts dan als $\mathrm{id} - (p - q) =
(\mathrm{id} - p) + q$ er een is (vraag 1 tweemaal). Via vragen
6--7 toegepast op de projectoren $\mathrm{id} - p$ en $q$, geldt dit
dan en slechts dan als $(\mathrm{id} - p)q = q(\mathrm{id} - p) = 0$, d.w.z.\
dan en slechts dan als $pq = q$ en $qp = q$.

\textbf{9.} $pq = q$ betekent dat $p$ elke $q(x)$ vasthoudt, d.w.z.\
$\operatorname{im} q \subseteq \ker(p - \mathrm{id}) =
\operatorname{im} p$. En $qp = q$ betekent dat $q\bigl((\mathrm{id} -
p)(x)\bigr) = 0$ voor alle $x$, d.w.z.\ $q$ verdwijnt op
$\operatorname{im}(\mathrm{id} - p) = \ker p$: $\ker p \subseteq
\ker q$. Beide stappen zijn equivalenties: de orde $q \leq p$ zegt dat
$q$ projecteert op een kleiner \omterm{def:b1:linmaps:kerim}{beeld}, langs een grotere \omterm{def:b1:linmaps:kerim}{kern}.

\textbf{10.} Ontwikkelen, $(\mathrm{id} - p)(\mathrm{id} - q) =
\mathrm{id} - p - q + pq = \mathrm{id} - r$. De projectoren
$\mathrm{id} - p$ en $\mathrm{id} - q$ commuteren, dus via
\cref{exo:b1:linmaps:6} is hun product $\mathrm{id} - r$ de
projector op $\operatorname{im}(\mathrm{id} - p) \cap
\operatorname{im}(\mathrm{id} - q) = \ker p \cap \ker q$ langs
$\ker(\mathrm{id} - p) + \ker(\mathrm{id} - q) =
\operatorname{im} p + \operatorname{im} q$. Via vraag 1 is $r =
\mathrm{id} - (\mathrm{id} - r)$ dan de projector met
$\operatorname{im} r = \operatorname{im} p + \operatorname{im}
q$ en $\ker r = \ker p \cap \ker q$.

\textbf{11.} $p_i$ is welgedefinieerd (uniciteit van de
ontbinding) en \omterm{def:b1:linmaps:def}{lineair} (de ontbinding van $x + \lambda y$
is de som van de ontbindingen, opnieuw via uniciteit). Voor $x_i
\in F_i$ is de ontbinding $x_i$ zelf, dus $p_i(x_i) = x_i$:
$p_i^2 = p_i$, en $p_j(x_i) = 0$ voor $j \neq i$: $p_i p_j = 0$
($p_j(x) \in F_j$). Door de componenten op te tellen, $\sum_i p_i =
\mathrm{id}$. Ten slotte $\operatorname{im} p_i = F_i$ en $\ker
p_i = \bigoplus_{j \neq i} F_j$.

\textbf{12.} $p_i = p_i \circ \mathrm{id} = p_i\sum_j p_j =
p_i^2 + \sum_{j \neq i} p_i p_j = p_i^2$: elke $p_i$ is een
projector. Elke $x = \mathrm{id}(x) = \sum_i p_i(x)$ ligt in
$\sum_i \operatorname{im} p_i$: de \omterm{def:b1:linmaps:kerim}{beelden} sommeren tot $E$.
Directheid: veronderstel $y_1 + \dots + y_k = 0$ met $y_i \in
\operatorname{im} p_i$, dus $p_i(y_i) = y_i$. Pas $p_j$ toe: $p_j
(y_i) = p_j p_i (y_i) = 0$ voor $i \neq j$, dus $0 = p_j\bigl(\sum
y_i\bigr) = y_j$, voor elke $j$. Bijgevolg $E = \bigoplus_i
\operatorname{im} p_i$.

\textbf{13.} $q = \mathrm{id} - p$, en vraag 1 geeft $pq =
p - p^2 = 0 = qp$ rechtstreeks: voor twee projectoren dwingt
optellen tot de identiteit reeds de orthogonaliteit van het paar af.

\textbf{14.} $p + q = \mathrm{id} - r$ met $r$ een projector, en
$(\mathrm{id} - r)$ is een projector (vraag 1): dus $p + q$ is een
projector, en vraag 6 geeft $pq = qp = 0$. Via symmetrie ($q +
r = \mathrm{id} - p$ en $p + r = \mathrm{id} - q$) verdwijnen alle
paarsgewijze producten, en vraag 12 besluit: $E =
\operatorname{im} p \oplus \operatorname{im} q \oplus
\operatorname{im} r$, automatisch.

\textbf{15.} \emph{Lemma.} Via inductie met Grassmann
(\cref{thm:b1:findim:grassmann}):
\[
\dim(F_1 + \dots + F_k) \leq \dim(F_1 + \dots + F_{k-1}) + \dim
F_k \leq \dots \leq \sum_i \dim F_i .
\]
Als het totaal een gelijkheid is, is elke stap dat: $(F_1 + \dots +
F_{j-1}) \cap F_j = \{0\}$ voor elke $j$, en een relatie $y_1 +
\dots + y_k = 0$ ($y_i \in F_i$) stort in vanaf rechts: $y_k
\in (F_1 + \dots + F_{k-1}) \cap F_k = \{0\}$, dan $y_{k-1} =
0$, enz.: de som is direct. Omgekeerd heeft een \omterm{def:b1:vspaces:sum}{directe som}
additieve dimensies (concateneer bases). \emph{Toepassing:}
$x = \sum_i p_i(x)$ toont $E = \sum_i \operatorname{im} p_i$,
dus $n \leq \sum_i \operatorname{rk} p_i$; de hypothese geeft
gelijkheid, dus directheid. Producten: fixeer $j$ en $y \in
\operatorname{im} p_j$. Dan $y = \sum_i p_i(y)$ met $p_i(y)
\in \operatorname{im} p_i$, terwijl $y = y$ ook een
ontbinding is (component $j$ alleen); uniciteit dwingt $p_i(y)
= 0$ af voor $i \neq j$. Toegepast op $y = p_j(x)$: $p_i p_j = 0$.

\textbf{16.} Als $u^k(x) = 0$ dan $u^{k+1}(x) = u(0) = 0$. En
$\operatorname{im} u^{k+1} = u^k\bigl(u(E)\bigr) \subseteq
u^k(E) = \operatorname{im} u^k$.

\textbf{17.} Neem aan $\ker u^{r} = \ker u^{r+1}$ en zij $x \in
\ker u^{r+2}$: dan $u(x) \in \ker u^{r+1} = \ker u^{r}$, dus
$u^{r+1}(x) = 0$: $x \in \ker u^{r+1}$. Met vraag 16,
$\ker u^{r+1} = \ker u^{r+2}$, en via inductie vallen alle latere
\omterm{def:b1:linmaps:kerim}{kernen} samen met $\ker u^{r}$. Voor \omterm{def:b1:linmaps:kerim}{beelden}: de dimensiestelling
geeft $\dim\operatorname{im} u^k = n - \dim\ker u^k$, dus de
beelddimensies bevriezen precies wanneer de kerndimensies dat doen,
en met de inclusies van vraag 16 betekenen gelijke dimensies
gelijke \omterm{def:b1:vspaces:subspace}{deelruimten} (\cref{thm:b1:findim:subspaces}).

\textbf{18.} De rij $\bigl(\dim\ker u^k\bigr)_k$ is
niet-dalend met waarden in $\intint{0}{n}$; ze kan niet strikt
$n + 1$ keer stijgen, dus een zekere $\dim\ker u^{r} = \dim\ker
u^{r+1}$ met $r \leq n$, dus $\ker u^{r} = \ker u^{r+1}$
(inclusie plus gelijke dimensie). Neem $r$ het kleinst; vraag 17
bevriest alles vanaf $r$, \omterm{def:b1:linmaps:kerim}{beelden} inbegrepen.

\textbf{19.} Doorsnede: zij $x \in \ker u^{r} \cap
\operatorname{im} u^{r}$, zeg $x = u^{r}(y)$ met $u^{r}(x) =
0$. Dan $u^{2r}(y) = 0$, en $\ker u^{2r} = \ker u^{r}$
(vraag 17), dus $x = u^{r}(y) = 0$. Dimensies: de dimensiestelling
voor $u^{r}$ geeft $\dim\ker u^{r} + \dim\operatorname{im} u^{r}
= n$; met triviale doorsnede maakt Grassmann de som een
\omterm{def:b1:vspaces:subspace}{deelruimte} van dimensie $n$: $E = \ker u^{r} \oplus
\operatorname{im} u^{r}$.

\textbf{20.} Stabiliteit: $u^{r}(u(x)) = u(u^{r}(x)) = 0$ voor $x
\in \ker u^{r}$; en $u(u^{r}(y)) = u^{r}(u(y)) \in
\operatorname{im} u^{r}$. Op $N = \ker u^{r}$: $(u|_N)^{r} = 0$
per definitie van $N$: nilpotent. Op $I = \operatorname{im}
u^{r}$: $\ker(u|_I) = \ker u \cap I \subseteq \ker u^{r} \cap I
= \{0\}$, dus $u|_I$ is een \omterm{def:b1:logic:inj}{injectief} endomorfisme van de
eindigdimensionale $I$, dus \omterm{def:b1:logic:inj}{bijectief}
(\cref{cor:b1:linmaps:samedim}).

\textbf{21.} Zij $x = a + b$ met $a \in N$, $b \in I$. Dan
$u(x) = u(a) + u(b)$ met $u(a) \in N$ en $u(b) \in I$
(vraag 20): dit \emph{is} de ontbinding van $u(x)$, dus
$\pi(u(x)) = u(a) = u(\pi(x))$: $\pi u = u\pi$.

\textbf{22.} $u^2(x,y,z) = u(y, 0, z) = (0, 0, z)$ en
$u^3(x,y,z) = u(0,0,z) = (0,0,z) = u^2(x,y,z)$. \omterm{def:b1:linmaps:kerim}{Kernen}: $\ker u
= \{y = z = 0\} = \operatorname{Vect}(e_1)$, $\ker u^2 = \{z =
0\} = \operatorname{Vect}(e_1, e_2)$, $\ker u^3 = \ker u^2$:
stabilisatie bij $r = 2$. \omterm{def:b1:linmaps:kerim}{Beelden}: $\operatorname{im} u =
\operatorname{Vect}(e_1, e_3)$, $\operatorname{im} u^2 =
\operatorname{Vect}(e_3)$. Fitting: $\R^3 =
\operatorname{Vect}(e_1, e_2) \oplus \operatorname{Vect}(e_3)$,
en $\pi(x, y, z) = (x, y, 0)$. Controle: $\pi u(x,y,z) = \pi(y, 0,
z) = (y, 0, 0)$ en $u\pi(x,y,z) = u(x, y, 0) = (y, 0, 0)$:
gelijk. Op de eerste factor $u(x, y, 0) = (y, 0, 0)$, waarvan het
kwadraat $0$ is: nilpotent; op de tweede $u(0,0,z) = (0,0,z)$:
de identiteit, \omterm{def:b1:logic:inj}{bijectief}.

\textbf{23.} Als $u^m = 0$ dan $\ker u^m = E$; aangezien \omterm{def:b1:linmaps:kerim}{kernen} vanaf
$r$ bevroren zijn, $\ker u^{r} = \ker u^{\max(m, r)} = E$.
Omgekeerd betekent $\ker u^{r} = E$ dat $u^{r} = 0$. En $\ker u^{r} =
E \iff$ de Fitting-projector is op $E$ langs $\{0\}$, d.w.z.\
$\pi = \mathrm{id}$. Ten slotte geeft $r \leq n$ (vraag 18):
elk nilpotent endomorfisme voldoet aan $u^{n} = 0$ --- de
nilpotentie-index overschrijdt nooit de dimensie.

\textbf{24.} Zij $m$ een nilpotentie-index van $u|_A$: $A
\subseteq \ker u^{m} \subseteq \ker u^{\max(m,r)} = \ker u^{r}$.
Aangezien $u|_B$ \omterm{def:b1:logic:inj}{bijectief} is, $B = u(B) = u^{k}(B) \subseteq
\operatorname{im} u^{k}$ voor elke $k$, in het bijzonder $B
\subseteq \operatorname{im} u^{r}$. Dan
\[
n = \dim A + \dim B \leq \dim\ker u^{r} +
\dim\operatorname{im} u^{r} = n :
\]
beide inclusies zijn gelijkheden van dimensies, dus van
\omterm{def:b1:vspaces:subspace}{deelruimten}: $A = \ker u^{r}$, $B = \operatorname{im} u^{r}$.

\textbf{25.} (i) Deel III is een woordenboek: splitsingen $E = F_1
\oplus \dots \oplus F_k$ corresponderen precies met families van
projectoren met $\sum p_i = \mathrm{id}$ en $p_i p_j = 0$, waarbij de
$F_i$ de \omterm{def:b1:linmaps:kerim}{beelden} zijn. (ii) Voor $k = 3$ zijn de complementen
$\mathrm{id} - p_i$ zelf projectoren, wat het argument afsloot met geen
extra hypothese; voor algemene $k$ heeft men
$\sum_i \operatorname{rk} p_i \leq n$ nodig, een ongelijkheid die het
spoor van Jaar~2 gratis geeft ($\operatorname{tr} p = \operatorname{rk}
p$ voor een projector, en sporen sommeren tot $\operatorname{tr}
\mathrm{id} = n$). (iii) \cref{exo:b1:linmaps:7} is het lemma van
Fitting in het reeds-gestabiliseerde geval $r \leq 1$; in het algemeen
laat men de kern- en beeldketens bevriezen, wat hoogstens $n$
stappen kost. (iv) In het volume van Jaar~2, toegepast op $u - \lambda\,
\mathrm{id}$, wordt de nilpotente factor de veralgemeende
eigenruimte bij $\lambda$ en worden de projectoren van Deel III de
spectrale projectoren van de reductietheorie. De stelling van
Deel IV is \emph{het lemma van Fitting}.
\end{solution}
