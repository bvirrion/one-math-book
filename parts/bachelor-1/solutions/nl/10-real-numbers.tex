\section*{Hoofdstuk \ref{ch:b1:reals} --- Reële getallen}

\begin{solution}{exo:b1:reals:1}
$A$: elk element is $\leq 1$ en $1 \in A$: $\sup A = \max A = 1$.
Ondergrenzen: $0$ begrenst naar onder; voor $\varepsilon > 0$ levert
Archimedes een $n$ met $\frac 1n < \varepsilon$, dus geen positief getal
begrenst $A$ naar onder: $\inf A = 0$, niet bereikt (geen min).

$B$: termen $0, -\frac12, \frac23, -\frac34, \frac45, \dots$ De even
termen $\frac{n}{n+1}$ ($n$ even) stijgen naar $1$ zonder het te bereiken;
de oneven termen $-\frac{n}{n+1}$ dalen naar $-1$. Dus $\sup B = 1$ en
$\inf B = -1$, geen van beide bereikt: geen max, geen min. (Grenzen:
$\abs{b} < 1$ voor alle $b \in B$; en $\frac{n}{n+1} = 1 - \frac{1}{n+1}
> 1 - \varepsilon$ voor $n$ groot, analoog naar onder.)

$C = \intoo{-\sqrt 3}{\sqrt 3}$: $\sup C = \sqrt 3$, $\inf C =
-\sqrt 3$, geen van beide bereikt.
\end{solution}

\begin{solution}{exo:b1:reals:2}
Schrijf $x = \lfloor x \rfloor + u$, $y = \lfloor y \rfloor + v$ met
$u, v \in \intco{0}{1}$. Dan $x + y = \lfloor x \rfloor + \lfloor y
\rfloor + (u + v)$ met $u + v \in \intco{0}{2}$. Als $u + v < 1$,
$\lfloor x + y\rfloor = \lfloor x\rfloor + \lfloor y \rfloor$; als
$1 \leq u + v < 2$, $\lfloor x+y \rfloor = \lfloor x \rfloor +
\lfloor y \rfloor + 1$. Beide gevallen komen voor: $(x, y) = (0.2,\,
0.3)$ geeft de linkergelijkheid, $(0.7,\, 0.8)$ de rechtse.
\end{solution}

\begin{solution}{exo:b1:reals:3}
Zij $k = \lfloor x \rfloor$, dus $k \leq x < k + 1$. Vermenigvuldigen
met $n$: $nk \leq nx < nk + n$, en gehele delen nemen (een stijgende
bewerking aan de kant van de gehele getallen): $nk \leq \lfloor nx
\rfloor \leq nk + n - 1$. Delen door $n$: $k \leq \frac{\lfloor nx
\rfloor}{n} < k + 1$, dus het buitenste \omterm{thm:b1:reals:floor}{gehele deel} is $k$.
\end{solution}

\begin{solution}{exo:b1:reals:4}
Elk element van $A$ zit in $B$, dus $\sup B$ begrenst $A$ naar boven:
bijgevolg $\sup A \leq \sup B$ ($\sup A$ is de \emph{kleinste}
\omterm{def:b1:reals:bounds}{bovengrens}). Symmetrisch $\inf B \leq \inf A$. Ten slotte $\inf A \leq
\sup A$ omdat $A$ niet-leeg is: elke $a \in A$ zit ertussen.
\end{solution}

\begin{solution}{exo:b1:reals:5}
Zij $s = \sup A$, $t = \sup B$. Elke $a + b \leq s + t$: \omterm{def:b1:reals:bounds}{bovengrens}.
Voor $\varepsilon > 0$, kies $a > s - \frac\varepsilon2$ en
$b > t - \frac\varepsilon2$ (\cref{prop:b1:reals:epsilon}): dan $a +
b > s + t - \varepsilon$. Volgens de $\varepsilon$-karakterisering,
$\sup(A+B) = s + t$.

Voor $-A$: $m$ begrenst $-A$ naar boven $\iff$ $-m$ begrenst $A$ naar
onder; de kleinste \omterm{def:b1:reals:bounds}{bovengrens} van $-A$ correspondeert dus met de
grootste ondergrens van $A$: $\sup(-A) = -\inf A$.
\end{solution}

\begin{solution}{exo:b1:reals:6}
Voor elke $x$: $f(x) + g(x) \leq \sup f + \sup g$; het \omterm{def:b1:reals:bounds}{supremum} van het
linkerlid nemen geeft de ongelijkheid. Strikt voorbeeld: $E = \{0, 1\}$,
$f = \mathbf{1}_{\{0\}}$ (waarde $1$ in $0$, anders $0$), $g =
\mathbf{1}_{\{1\}}$: $\sup(f + g) = 1 < 2 = \sup f + \sup g$.

Geen tegenspraak met \cref{exo:b1:reals:5}: daar variëren $a \in A$ en
$b \in B$ \emph{onafhankelijk}; hier voedt dezelfde $x$ zowel $f$ als
$g$ --- de \omterm{def:b1:logic:sets}{verzameling} $\{f(x) + g(x) : x \in E\}$ is kleiner dan de
\omterm{def:b1:logic:sets}{verzameling} $\{f(x) + g(y) : x, y \in E\}$.
\end{solution}

\begin{solution}{exo:b1:reals:7}
$\sqrt 6$ is irrationaal: $6 = 2 \times 3$ is geen volkomen kwadraat,
en $v_2(6q^2) = 1 + 2v_2(q)$ oneven verhindert $6q^2 = r^2$ (zoals in
\cref{exo:b1:arith:7}). Stel nu $x = \sqrt 2 + \sqrt 3 \in \Q$.
Dan $x^2 = 5 + 2\sqrt 6 \in \Q$, dus $\sqrt 6 = \frac{x^2 - 5}{2} \in
\Q$: tegenspraak. Bijgevolg $\sqrt 2 + \sqrt 3 \notin \Q$.
\end{solution}

\begin{solution}{exo:b1:reals:8}
Zij $x < y$. Kies $n \in \N$ met $2^n > \frac{1}{y - x}$ (Archimedes:
$2^n \geq n + 1$ via een eenvoudige inductie, dus een zekere macht van
$2$ overtreft elk reëel getal). Dan, zoals in het bewijs van
\cref{thm:b1:reals:density} met $2^n$ in plaats van $n$: $m = \lfloor
2^n x \rfloor + 1$ voldoet aan $x < \frac{m}{2^n} < y$. Dus $D$ is
dicht.
\end{solution}

\begin{solution}{exo:b1:reals:9}
\begin{enumerate}
  \item Stel $\alpha > 0$. Ten eerste, $\alpha \in G$. Stel van niet: via
        de $\varepsilon$-karakterisering van het \omterm{def:b1:reals:bounds}{infimum} met
        $\varepsilon = \alpha$, bestaat er een $g \in G$ met $\alpha < g <
        2\alpha$ (strikt links aangezien $\alpha \notin G$); dan,
        met $\varepsilon = g - \alpha$, bestaat er een $h \in G$ met
        $\alpha < h < g$. Nu $g - h \in G$ en $0 < g - h < g -
        \alpha < \alpha$: een element van $G \cap \intoo{0}{+\infty}$
        onder zijn \omterm{def:b1:reals:bounds}{infimum}, absurd. Dus $\alpha \in G$, en $\alpha\Z
        \subseteq G$ ($G$ is een \omterm{def:b1:structures:group}{groep}). Omgekeerd, voor $x \in G$, zij
        $k = \lfloor x/\alpha \rfloor$: dan $x - k\alpha \in G$ en
        $0 \leq x - k\alpha < \alpha$, en de definitie van $\alpha$
        dwingt $x - k\alpha = 0$ af. Bijgevolg $G = \alpha\Z$.
  \item Stel $\alpha = 0$, en zij $x < y$. Er bestaat een $g \in G$
        met $0 < g < y - x$. Het veelvoud $kg$ met $k = \lfloor x/g
        \rfloor + 1$ voldoet aan $x < kg \leq x + g < y$, en $kg \in
        G$: dichtheid.
\end{enumerate}
$G = \Z + \sqrt 2\,\Z$ is een \omterm{def:b1:structures:subgroup}{deelgroep} van $(\R, +)$. Ze is niet van de
vorm $\alpha\Z$: anders zouden $1 = p\alpha$ en $\sqrt 2 = q\alpha$
($p, q \in \Z$) geven $\sqrt 2 = \frac qp \in \Q$,
tegenspraak. Volgens de dichotomie is $G$ dicht in $\R$.
\end{solution}

\begin{solution}{exo:b1:reals:10}
Zij $s = \sup A > 0$, $t = \sup B > 0$. Voor $a \in A$, $b \in B$: $ab
\leq st$ (vermenigvuldigen van ongelijkheden tussen \emph{positieve}
getallen). Voor $0 < \varepsilon < \min(s, t)$: kies $a > s - \varepsilon$
en $b > t - \varepsilon$; dan
\[
ab > (s - \varepsilon)(t - \varepsilon)
= st - \varepsilon(s + t) + \varepsilon^2
> st - \varepsilon (s + t),
\]
en $\varepsilon(s+t)$ kan willekeurig klein gemaakt worden: volgens de
$\varepsilon$-karakterisering (in de vorm: geen getal $< st$ begrenst
$AB$ naar boven), $\sup AB = st$.

Positiviteit is essentieel: $A = B = \{-1, 0\}$ geeft $AB = \{0, 1\}$,
$\sup AB = 1$, terwijl $\sup A \cdot \sup B = 0 \times 0 = 0$.
\end{solution}

\begin{solution}{exo:b1:reals:11}
Schrijf $s = \sup A$, $i = \inf A$. Voor $a, a' \in A$: $a \leq s$
en $a' \geq i$ geven $a - a' \leq s - i$; door symmetrie $\abs{a -
a'} \leq s - i$, dus $s - i$ begrenst de \omterm{def:b1:logic:sets}{verzameling} tussenruimten naar
boven. Voor $\varepsilon > 0$, kies $a > s - \frac\varepsilon2$ en $a' <
i + \frac\varepsilon2$ (\cref{prop:b1:reals:epsilon} en zijn spiegel
voor het \omterm{def:b1:reals:bounds}{infimum}): dan $\abs{a - a'} \geq a - a' > s - i -
\varepsilon$. Volgens de $\varepsilon$-karakterisering,
$\operatorname{diam} A = s - i$.

Elke $a \in A$ voldoet aan $i \leq a \leq s$, dus $A \subseteq
\intcc{i}{s}$, een gesloten \omterm{prop:b1:reals:intervals}{interval} van lengte $\operatorname{diam}
A$. Als een gesloten \omterm{prop:b1:reals:intervals}{interval} $\intcc{u}{v}$ de \omterm{def:b1:logic:sets}{verzameling} $A$ bevat,
dan is $v$ een \omterm{def:b1:reals:bounds}{bovengrens} en $u$ een ondergrens van $A$, dus $u \leq i$
en $v \geq s$: $\intcc{i}{s} \subseteq \intcc{u}{v}$. Bijgevolg is
$\intcc{i}{s}$ het kleinste.
\end{solution}

\begin{solution}{exo:b1:reals:12}
$0 \in E$, dus $E \neq \emptyset$. Als $x > \max(1, y)$ dan $x^2 >
x > y$, dus $E$ is naar boven begrensd door $\max(1, y)$: $s = \sup E$
bestaat (\cref{thm:b1:reals:sup}), en $s \geq \min(1, y) > 0$
omdat $\min(1, y) \in E$: inderdaad als $y \geq 1$ dan $1^2 = 1
\leq y$, en als $y < 1$ dan $y^2 < y$.

\emph{$s^2 < y$ is onmogelijk.} Kies $0 < h < 1$ met $h <
\frac{y - s^2}{2s + 1}$. Dan
\[
(s + h)^2 = s^2 + 2sh + h^2 \leq s^2 + (2s + 1)h < y ,
\]
dus $s + h \in E$, in tegenspraak met het feit dat $s$ de \omterm{def:b1:logic:sets}{verzameling}
$E$ naar boven begrenst.

\emph{$s^2 > y$ is onmogelijk.} Kies $0 < h < s$ met $h <
\frac{s^2 - y}{2s}$. Dan $(s - h)^2 = s^2 - 2sh + h^2 > s^2 -
2sh > y$; elke $x \in E$ voldoet aan $x^2 \leq y < (s - h)^2$,
dus $x < s - h$ (beide zijn $\geq 0$): $s - h$ is een \omterm{def:b1:reals:bounds}{bovengrens}
van $E$ kleiner dan $s$, in tegenspraak met de minimaliteit.

Bijgevolg $s^2 = y$. Eenduidigheid: als $0 < s < s'$ dan $s^2 <
s'^2$, dus twee verschillende positieve wortels kunnen niet beide $y$
als kwadraat hebben. Monotonie: als $0 < y < y'$, dan $\sqrt y \neq
\sqrt{y'}$, en $\sqrt y > \sqrt{y'}$ zou $y > y'$ geven door
kwadrateren: dus $\sqrt y < \sqrt{y'}$.
\end{solution}

\begin{solution}{pb:b1:reals:1}
\textbf{1.} Voor $\frac 18$: $10 = 8 \cdot 1 + 2$, $20 = 8 \cdot 2
+ 4$, $40 = 8 \cdot 5 + 0$; cijfers $1, 2, 5$, rest $0$, dan
enkel nullen: $\frac 18 = 0.125$. Voor $\frac 17$: $10 = 7 \cdot 1 +
3$, $30 = 7 \cdot 4 + 2$, $20 = 7 \cdot 2 + 6$, $60 = 7 \cdot 8 +
4$, $40 = 7 \cdot 5 + 5$, $50 = 7 \cdot 7 + 1$: cijfers $1, 4, 2,
8, 5, 7$, resten $3, 2, 6, 4, 5, 1$. De rest is
teruggekeerd naar $r = 1$, dus de zes stappen herhalen zich woordelijk
eeuwig: $\frac 17 = 0.(142857)$, resten die de cyclus $1, 3, 2, 6,
4, 5$ doorlopen.

\textbf{2.} $\frac 13$ in grondtal $2$ ($r_0 = 1$): $2 = 3 \cdot 0 +
2$, $4 = 3 \cdot 1 + 1$, en $r = 1$ keert terug: $\frac 13 =
(0.\overline{01})_2$. $\frac{5}{16}$ in grondtal $2$: $10 = 16 \cdot
0 + 10$, $20 = 16 \cdot 1 + 4$, $8 = 16 \cdot 0 + 8$, $16 = 16
\cdot 1 + 0$: $\frac{5}{16} = (0.0101)_2$, afbrekend. $\frac 12$
in grondtal $3$: $3 = 2 \cdot 1 + 1$, en $r = 1$ keert onmiddellijk
terug: $\frac 12 = (0.\overline{1})_3$.

\textbf{3.} De rest van het algoritme na $N$ stappen is $r_N =
b^N p \bmod q$ (formeel bewezen in vraag 14; hier is het de
observatie dat elke stap de rest met $b$ vermenigvuldigt en
reduceert mod $q$). Alle latere cijfers zijn $0$ als en slechts als een
zekere $r_N = 0$, d.w.z.\ $q \mid b^N p$; aangezien $\gcd(p, q) = 1$,
geeft het lemma van Gauss $q \mid b^N$. Als $q \mid b^N$, \omterm{def:b1:arith:divides}{deelt} elke
priemfactor van $q$ het getal $b^N$, en \omterm{def:b1:arith:divides}{deelt} dus $b$ (primaliteit).
Omgekeerd, als elke \omterm{def:b1:arith:prime}{priem} van $q = p_1^{a_1} \cdots p_r^{a_r}$ het getal
$b$ \omterm{def:b1:arith:divides}{deelt}, dan \omterm{def:b1:arith:divides}{deelt} met $A = \max_i a_i$ elke $p_i^{a_i}$ het getal
$b^A$, en de $p_i^{a_i}$ zijn paarsgewijs \omterm{cor:b1:arith:bezout}{relatief priem}, dus $q \mid
b^A$. Voor $q = 20 = 2^2 \cdot 5$: beide \omterm{def:b1:arith:prime}{priemen} delen $10$
($\frac{1}{20} = 0.05$), maar $2 \nmid 3$, dus $\frac{1}{20}$ herhaalt
zich eeuwig in grondtal $3$.

\textbf{4.} $1^2 = 1 < 2 < 4 = 2^2$ geeft $s_0 = 1$. Dan $1.4^2
= 1.96 < 2 < 2.25 = 1.5^2$: $\lfloor 10\sqrt 2 \rfloor = 14$,
$s_1 = 1.4$. Vervolgens $1.41^2 = 1.9881 < 2 < 2.0164 = 1.42^2$: $s_2 =
1.41$; $1.414^2 = 1.999396 < 2 < 2.002225 = 1.415^2$: $s_3 =
1.414$; $1.4142^2 = 1.99996164 < 2 < 2.00024449 = 1.4143^2$: $s_4
= 1.4142$. In elk geval zeggen de weergegeven ongelijkheden precies
$s_n \leq \sqrt 2 < s_n + 10^{-n}$, wat de definitie is van
het \omterm{thm:b1:reals:floor}{gehele deel} van $10^n \sqrt 2$.

\textbf{5.} $A_0 = \lfloor x \rfloor = 0$ aangezien $0 \leq x < 1$.
Uit $A_{n-1} \leq b^{n-1} x < A_{n-1} + 1$, vermenigvuldig met $b$:
\[
b\,A_{n-1} \leq b^n x < b\,A_{n-1} + b .
\]
Het gehele getal $b\,A_{n-1}$ is $\leq b^n x$, dus $b\,A_{n-1} \leq
A_n$; en $b^n x < b\,A_{n-1} + b$ met $b\,A_{n-1} + b$ een
geheel getal dwingt $A_n \leq b\,A_{n-1} + b - 1$ af. Bijgevolg $0 \leq
d_n = A_n - b\,A_{n-1} \leq b - 1$: een cijfer.

\textbf{6.} Telescoperen: $d_k b^{-k} = A_k b^{-k} - A_{k-1}
b^{-(k-1)}$, dus
\[
\sum_{k=1}^n d_k b^{-k} = A_n b^{-n} - A_0 = s_n .
\]
$A_n \leq b^n x < A_n + 1$ delen door $b^n$ geeft $s_n \leq x <
s_n + b^{-n}$.

\textbf{7.} Inductie: $b^0 = 1 \geq 1$, en $b^{n+1} = b \cdot
b^n \geq 2(n + 1) \geq n + 2$. Monotonie: $s_n - s_{n-1} = d_n
b^{-n} \geq 0$. Elke $s_n \leq x$ (vraag 6): $x$ is een \omterm{def:b1:reals:bounds}{bovengrens}
van $\{s_n\}$. Voor $\varepsilon > 0$ levert de \omterm{thm:b1:reals:archimedes}{archimedische
eigenschap} een $n$ met $n + 1 > \frac1\varepsilon$, dus
$b^{-n} < \varepsilon$, en dan $s_n > x - b^{-n} > x -
\varepsilon$ volgens vraag 6. Volgens \cref{prop:b1:reals:epsilon}, $x =
\sup_n s_n$.

\textbf{8.} Stel $d_k = b - 1$ voor alle $k > N$. Voor $n > N$
geeft de eindige meetkundige som
\[
s_n = s_N + (b - 1)\sum_{k=N+1}^{n} b^{-k}
= s_N + b^{-N} - b^{-n} .
\]
Dus $x \geq s_n = s_N + b^{-N} - b^{-n}$ voor elke $n$; door
de laatste term onder elke $\varepsilon$ te laten krimpen (vraag 7), $x
\geq s_N + b^{-N}$. Maar vraag 6 op rang $N$ zegt $x < s_N +
b^{-N}$: tegenspraak. De reeks $(d_n)$ is eigenlijk.

\textbf{9.} Begrensd: $t_n \leq (b-1)\sum_{k=1}^n b^{-k} = 1 -
b^{-n} < 1$, en $(t_n)$ is niet-dalend, dus $y = \sup t_n$
bestaat met $0 \leq y \leq 1$. Fixeer $n$. Voor de tweezijdige
schatting: $t_n \leq y$ is duidelijk. Kies via eigenlijkheid $m > n$ met
$e_m \leq b - 2$. Voor $p \geq m$:
\[
t_p - t_n = \sum_{k=n+1}^{p} e_k b^{-k}
\leq (b^{-n} - b^{-p}) - b^{-m} < b^{-n} - b^{-m},
\]
waarbij de middelste som ten minste $b^{-m}$ verliest tegen het
maximum met louter $(b-1)$; voor $p \leq m$, $t_p \leq t_m \leq t_n +
b^{-n} - b^{-m}$ eveneens (monotonie plus het geval $p = m$). Bijgevolg
is elke $t_p \leq t_n + b^{-n} - b^{-m}$, dus $y \leq t_n + b^{-n} -
b^{-m} < t_n + b^{-n}$. (Met $n = 0$: $y < 1$, dus $y \in
\intco{0}{1}$.) Nu is $b^n t_n = \sum_{k \leq n} e_k b^{n-k}$ een
geheel getal, en $b^n t_n \leq b^n y < b^n t_n + 1$: dus $\lfloor b^n
y \rfloor = b^n t_n$. Ten slotte de cijfers van $y$: $d_n(y) = b^n
t_n - b \cdot b^{n-1} t_{n-1} = b^n(t_n - t_{n-1}) = e_n$.

\textbf{10.} Vraag 9 zegt: (waarde van reeks) heeft cijfers (de
reeks); vragen 5--8 zeggen: (cijfers van $x$) vormen een eigenlijke
reeks wiens afkappingen \omterm{def:b1:reals:bounds}{supremum} $x$ hebben (vraag 7). Dus de twee
\omterm{def:b1:logic:map}{afbeeldingen} samengesteld geven de identiteit in beide volgorden: ze
zijn onderling inverse bijecties tussen $\intco{0}{1}$ en de eigenlijke
reeksen. Als twee eigenlijke reeksen gelijke waarde hadden, zou het
toepassen van de cijferafbeelding ze gelijk maken: eenduidigheid. Dit is
de \emph{$b$-adische ontwikkelingsstelling}.

\textbf{11.} Zij $e_n = b - 1$ voor $n > M$, $M \geq 0$ minimaal.
Zoals in vraag 8, $t_n = t_M + b^{-M} - b^{-n}$ voor $n \geq M$,
dus de waarde is $\sup t_n = t_M + b^{-M}$. Als $M = 0$ is de waarde
$0 + 1 = 1$: in grondtal $10$, $0.999\dots = 1$ exact --- niet
bij benadering. Als $M \geq 1$, geeft minimaliteit $e_M \leq b - 2$,
en de waarde is
\[
t_M + b^{-M} = \frac{b^M t_M + 1}{b^M} \in \intoo{0}{1},
\]
een $b$-adische breuk, wiens \emph{eigenlijke} ontwikkeling $e_1 \dots
e_{M-1}\,(e_M + 1)\,000\dots$ is (een afbrekende reeks is eigenlijk,
en haar waarde is hetzelfde getal). Omgekeerd moet een reëel getal met
twee representaties er één oneigenlijke hebben (eigenlijkheid pint de
representatie vast, vraag 10), en dus van deze vorm zijn. En elke
$m/b^N \in \intoo{0}{1}$, geschreven met laatste nullen-vrije cijfer
$d_N$, heeft inderdaad de oneigenlijke tweeling $d_1 \dots d_{N-1}(d_N -
1)(b-1)(b-1)\dots$: precies de $b$-adische breuken dragen twee
namen, alle andere reële getallen één.

\textbf{12.} Zeg dat de reeksen overeenstemmen tot $m - 1$, met
gemeenschappelijke afkapping $P = s_{m-1}$, en $d_m < e_m$. Volgens
vraag 9 (strikte bovenschatting op rang $m$), $x < P + d_m b^{-m} +
b^{-m} = P + (d_m + 1)b^{-m} \leq P + e_m b^{-m} \leq y$, de laatste
stap omdat $P + e_m b^{-m}$ de afkapping $t_m \leq y$ van $y$ is. Dus
$d_m < e_m \implies x < y$; door de rollen te verwisselen, $e_m < d_m
\implies y < x$; en aangezien de reeksen verschillen bij $m$, geldt één
van de twee. Beide richtingen volgen.

\textbf{13.} Zij $z = bx - A_1 \in \intco{0}{1}$ (inderdaad $A_1
\leq bx < A_1 + 1$). Voor $n \geq 0$: $b^n z = b^{n+1} x - b^n
A_1$ met $b^n A_1 \in \Z$, dus via $\lfloor u - K \rfloor =
\lfloor u \rfloor - K$ ($K$ geheel getal),
\[
A_n(z) = A_{n+1}(x) - b^n A_1(x) .
\]
Bijgevolg $d_n(z) = A_n(z) - b\,A_{n-1}(z) = A_{n+1} - b^n A_1 -
b\,A_n + b^n A_1 = d_{n+1}(x)$. Dus het fractionele deel van $bx$
draagt de verschoven cijfers; door $m$ keer te itereren, heeft het
fractionele deel van $b^m x$ cijfers $(d_{n+m})_{n \geq 1}$.

\textbf{14.} Euclidische deling: $b^n p = q\,Q_n + r_n$ met $0
\leq r_n < q$. Deel door $q$: $b^n x = Q_n + \frac{r_n}{q}$ met
$0 \leq \frac{r_n}{q} < 1$, dus $Q_n = \lfloor b^n x \rfloor =
A_n$, wat $A_n = \frac{b^n p - r_n}{q}$ geeft. Voor de recurrentie:
$b^n p = b(q\,A_{n-1} + r_{n-1}) = q\,(b\,A_{n-1}) +
b\,r_{n-1}$, dus $b^n p$ en $b\,r_{n-1}$ verschillen een veelvoud van
$q$: $r_n = (b\,r_{n-1}) \bmod q$.

\textbf{15.} Deel $b\,r_{n-1}$ door $q$: $b\,r_{n-1} = q\,c +
r_n$ met $c = \lfloor b\,r_{n-1}/q \rfloor$. Substitueren in
de weergave van vraag 14: $b^n p = q(b\,A_{n-1} + c) + r_n$,
en de eenduidigheid van de euclidische deling identificeert $A_n =
b\,A_{n-1} + c$, dat wil zeggen $d_n = c = \lfloor b\,r_{n-1}/q
\rfloor$. Cijfer $n$ hangt enkel af van $r_{n-1}$ --- de
staartdelingslus van Deel I, nu gecertificeerd.

\textbf{16.} De $q + 1$ resten $r_0, \dots, r_q$ nemen waarden aan
in de $q$-elementige \omterm{def:b1:logic:sets}{verzameling} $\intint{0}{q-1}$: volgens het
duiventilprincipe (\cref{cor:b1:counting:pigeonhole}) vallen twee samen,
zeg $r_N = r_{N+T}$ met $0 \leq N < N + T \leq q$. Aangezien $r_n$
$r_{n+1}$ bepaalt (vraag 14), geeft inductie $r_{n+T} =
r_n$ voor alle $n \geq N$; aangezien $r_{n-1}$ $d_n$ bepaalt
(vraag 15), $d_{n+T} = d_n$ voor alle $n \geq N + 1$. De
ontwikkeling van elk rationaal getal is uiteindelijk periodiek, met
voorperiode $\leq q$ en periode $\leq q$.

\textbf{17.} De cijfers van het fractionele deel van $b^T y$ zijn
$(d_{n+T}) = (d_n)$ (verschuivingslemma, dan zuivere periodiciteit):
dezelfde eigenlijke reeks als $y$. Volgens vraag 10 zijn de waarden
gelijk: $b^T y - \lfloor b^T y \rfloor = y$, dus $(b^T - 1)\,y = \lfloor
b^T y \rfloor = A_T \in \N$ en
\[
y = \frac{A_T}{b^T - 1} ,
\]
rationaal met noemer die $b^T - 1$ \omterm{def:b1:arith:divides}{deelt}; de teller $A_T$
is het gehele getal wiens cijfers in grondtal $b$ gelijk zijn aan $d_1
\dots d_T$. Controle: $0.(142857) = \frac{142857}{999999}$, en $142857
\times 7 = 999999$, dus dit is $\frac 17$.

\textbf{18.} Als $d_{n+T} = d_n$ voor $n > N$, heeft het fractionele
deel $z$ van $b^N x$ cijfers $(d_{N+n})_{n\geq1}$ (verschuivingslemma),
die zuiver periodiek zijn; volgens vraag 17, $z \in \Q$. Dan geeft $b^N
x = A_N + z$ dat $x = (A_N + z)/b^N \in \Q$. Met vraag 16:
$x$ rationaal $\iff$ ontwikkeling uiteindelijk periodiek. Het rechterlid
vermeldt het grondtal, het linkerlid niet: periodiciteit in één grondtal
is equivalent met rationaliteit, en dus met periodiciteit in elk
grondtal.

\textbf{19.} Voor $x = \frac 1q$, $r_n = b^n \bmod q$. Als
$\gcd(b, q) = 1$, dan $r_T = r_0 = 1$ als en slechts als $b^T \equiv
1 \pmod q$; zo'n $T$ bestaat (duiventil geeft $b^i \equiv b^j$,
$i < j$, en $b$ is inverteerbaar modulo $q$, dus $b^{j-i} \equiv
1$), en de kleinste --- de multiplicatieve orde --- maakt de
resten, en dus de cijfers, zuiver periodiek met periode $T$.
Geen kleinere periode is mogelijk: een periode $T'$ zou $(b^{T'}
- 1)\frac1q \in \N$ geven (vraag 17), d.w.z.\ $q \mid b^{T'} - 1$.
Voor $q = 7$, $b = 10$: $10 \equiv 3$, $10^2 \equiv 2$, $10^3
\equiv 6$, $10^4 \equiv 4$, $10^5 \equiv 5$, $10^6 \equiv 1
\pmod 7$: orde $6$, en inderdaad heeft $\frac 17$ periode zes.

\textbf{20.} De reeks heeft oneindig veel $0$'s (de cijfers van $x^*$
zijn meestal nul), dus ze is eigenlijk, en $x^*$ is goed gedefinieerd
(vraag 9). Stel de cijfers uiteindelijk periodiek met periode
$T$ voorbij $N$. Oneindig veel cijfers zijn gelijk aan $1$ (één per
driehoeksgetal), dus een zekere $1$ zit op een positie $j > N$; dan
plaatst periodiciteit een $1$ op elke positie $j + kT$: vanaf $j$
verder zijn de tussenruimten tussen opeenvolgende $1$'s ten hoogste $T$.
Maar de $1$'s zitten precies op de driehoeksgetallen, wiens
opeenvolgende tussenruimten $\frac{(j+1)(j+2)}{2} - \frac{j(j+1)}{2} = j
+ 1$ uiteindelijk $T$ overtreffen: tegenspraak. Niet uiteindelijk
periodiek, dus volgens vraag 18, $x^* \notin \Q$ --- irrationaliteit
afgelezen uit het ritme van de cijfers alleen.

\textbf{21.} Gegeven $x < y$, levert vraag 7 een $n$ met $b^{-n}
< y - x$; stel $m = \lfloor b^n x \rfloor + 1$. Dan $b^n x < m
\leq b^n x + 1 < b^n y$, dus $x < \frac{m}{b^n} < y$: dichtheid,
voor elk grondtal tegelijk ($b = 2$ herwint
\cref{exo:b1:reals:8}). Voor $\frac pq \in \intoo{0}{1}$ in grondtal
$b = q$: het eerste cijfer is $\lfloor q \cdot \frac pq \rfloor =
p$ en het fractionele deel van $q \cdot \frac pq = p$ is $0$: alle
latere cijfers verdwijnen, een afbrekende ontwikkeling $\frac pq =
(0.p)_q$. Afbreken hangt af van het grondtal; periodiciteit ---
rationaliteit --- niet (vraag 18).

\textbf{22.} Elke $e_k \in \{1, 2\}$ is een cijfer van grondtal $10$, en
de reeks eindigt nooit op louter $9$'s: eigenlijk. Haar waarde $y$ ligt in
$\intco{0}{1}$ en heeft precies cijfers $(e_k)$ (vraag 9). Fixeer
$k$: het $k$-de cijfer van $y$ is $e_k$, gekozen $\neq$ het $k$-de
cijfer van $x_k$, dus de eigenlijke reeksen van $y$ en $x_k$ verschillen,
dus $y \neq x_k$ (vraag 10: de codering is \omterm{def:b1:logic:inj}{injectief}). Dus $y$ zit
in geen enkele lijst: geen \omterm{def:b1:logic:map}{afbeelding} $\N^* \to \intco{0}{1}$ is
\omterm{def:b1:logic:inj}{surjectief}. De reële getallen kunnen, in tegenstelling tot de rationale,
niet worden opgesomd --- overaftelbaarheid, wiens theorie
\cref{ch:b1:topology} ontwikkelt.

\textbf{23.} Als de ontwikkelingen overeenstemmen tot $n$, dan hebben
$x$ en $y$ dezelfde afkapping $s_n$, en vraag 6 plaatst beide in
$\intco{s_n}{s_n + b^{-n}}$, een \omterm{prop:b1:reals:intervals}{interval} van lengte $b^{-n}$:
$\abs{x - y} < b^{-n}$. Omkering: $x = 0.1$ en $y = 0.0999$
(afbrekend, dus eigenlijk) voldoen aan $\abs{x - y} = 10^{-4} <
10^{-3}$, toch verschillen hun ontwikkelingen bij het allereerste cijfer.
De schuldigen zijn de $b$-adische breuken van vraag 11: in hun
buurt keert een minieme verplaatsing elk weergegeven cijfer om ($0.0999 \to
0.1000$), omdat het precies de reële getallen zijn waar de oneigenlijke
tweeling op de loer ligt.

\textbf{24.} $p = 1$, $q = 10$, $b = 2$, $r_0 = 1$: $2 = 10 \cdot
0 + 2$, $4 = 10 \cdot 0 + 4$, $8 = 10 \cdot 0 + 8$, $16 = 10
\cdot 1 + 6$, $12 = 10 \cdot 1 + 2$ --- en $r_5 = 2 = r_1$: de
resten doorlopen de cyclus $(2, 4, 8, 6)$ vanaf index $1$. Cijfers: $d_1 =
0$, dan het herhalende blok $d_2 d_3 d_4 d_5 = 0, 0, 1, 1$:
\[
\tfrac{1}{10} = (0.0\overline{0011})_2 ,
\]
voorperiode $1$, periode $4$. Volgens vraag 3 zou een afbrekende
ontwikkeling in grondtal $2$ vereisen dat elke \omterm{def:b1:arith:prime}{priem} van $10$ het getal
$2$ \omterm{def:b1:arith:divides}{deelt}; de \omterm{def:b1:arith:prime}{priem} $5$ weigert. Dus $0.1$ is \emph{niet}
representeerbaar door een eindige binaire reeks --- een computer die
eindig veel bits opslaat houdt slechts een afkapping bij, en de
opgestapelde afkappingsfouten zijn waarom de floating-point $0.1 + 0.2$
verschilt van $0.3$ in de laatste bits.

\textbf{25.} (i) Volledigheid produceerde de waarden: $x = \sup s_n$
en $y = \sup t_n$ (vragen 7 en 9) --- over $\Q$ alleen zou de
eigenlijke reeks van $\sqrt 2$ niets benoemen. (ii) De
\omterm{thm:b1:reals:archimedes}{archimedische eigenschap} maakte $b^{-n}$ uiteindelijk kleiner dan elke
$\varepsilon$, wat de afkappingen dwong hun \omterm{def:b1:reals:bounds}{supremum} te naderen
(vragen 7, 21). (iii) De eenduidigheidsclausule van
het \omterm{thm:b1:reals:floor}{gehele deel} identificeerde $Q_n = A_n$ in vraag 14 en legitimeerde
elke cijferextractie $\lfloor u - K \rfloor = \lfloor u \rfloor - K$
(vraag 13). (iv) Het duiventilprincipe, toegepast op eindig
veel resten, is de enige motor van periodiciteit (vraag 16).
Moraal: eigenlijke reeksen coderen $\intco{0}{1}$ getrouw en veranderen
rationaliteit in een zichtbaar ritme; maar het optellen van
cijferreeksen vereist overdrachten die van oneindig ver rechts
propageren, dus geen eindige-stapregel berekent zelfs maar het eerste
cijfer van een som --- terwijl de supremuminterface van
\cref{thm:b1:reals:sup} de hele analyse met één axioma afhandelt.
Cijfers zijn een prachtig \emph{beeld} van $\R$; het \omterm{def:b1:reals:bounds}{supremum} is haar
\emph{motor}.
\end{solution}
