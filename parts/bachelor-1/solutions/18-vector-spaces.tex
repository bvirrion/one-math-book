\begin{solution}{exo:b1:vspaces:1}
\begin{enumerate}
  \item Yes: contains $0$, and the defining equation is linear
        (stable under $x + \lambda y$).
  \item No: does not contain $(0,0,0)$.
  \item No: $(1, 0)$ and $(0, -1)$ belong ($xy = 0$), their sum $(1,
        -1)$ does not ($xy = -1 < 0$).
  \item Yes: $0$ vanishes at $1$; $(P + \lambda Q)(1) = P(1) +
        \lambda Q(1) = 0$.
  \item Yes: the zero function is bounded; if $\abs f \leq M$ and
        $\abs g \leq M'$, then $\abs{f + \lambda g} \leq M +
        \abs\lambda M'$.
\end{enumerate}
\end{solution}

\begin{solution}{exo:b1:vspaces:2}
$(1,2,1) = a(1,0,1) + b(1,1,0)$ requires $a + b = 1$, $b = 2$, $a =
1$: incompatible ($a + b = 3 \neq 1$): not in the span. $(2,1,1) =
a(1,0,1) + b(1,1,0)$: $b = 1$, $a = 1$, $a + b = 2$: consistent, so
$(2,1,1) = (1,0,1) + (1,1,0)$, in the span.

Equation: $(x, y, z) = (a + b, b, a)$ means $x = y + z$: the span is
the plane $\{x - y - z = 0\}$.
\end{solution}

\begin{solution}{exo:b1:vspaces:3}
First family: $\lambda(1,1,0) + \mu(1,0,1) + \nu(0,1,1) = 0$ gives
$\lambda + \mu = 0$, $\lambda + \nu = 0$, $\mu + \nu = 0$: adding,
$2(\lambda + \mu + \nu) = 0$, and subtracting each equation,
$\lambda = \mu = \nu = 0$: free.

Second: $(2,4,6) = 2(1,2,3)$: linked.

Third: four vectors in $\R^3$ --- necessarily linked once dimension
is available (\cref{ch:b1:findim}); directly: $(0,1,1) = -(1,0,0) +
0\cdot(1,1,0) + (1,1,1)$, indeed $(-1,0,0) + (1,1,1) = (0,1,1)$: a
nontrivial relation.
\end{solution}

\begin{solution}{exo:b1:vspaces:4}
Polynomials $1, (X-1), (X-1)^2, (X-1)^3$ have distinct degrees $0,
1, 2, 3$: free (\cref{prop:b1:vspaces:freecriteria}~(1)); four free
vectors generating (every $P \in \R_3[X]$ expands in powers of $X -
1$, e.g.\ by Taylor for polynomials, cf.\ the proof of
\cref{prop:b1:poly:multiplicity}): a basis. For $X^3$, Taylor at
$1$: $P = X^3$, $P(1) = 1$, $P'(1) = 3$, $P''(1) = 6$, $P'''(1) =
6$:
\[
X^3 = 1 + 3(X - 1) + 3(X-1)^2 + (X-1)^3 ,
\]
coordinates $(1, 3, 3, 1)$ (Pascal's row, as expected from $X^3 =
((X-1)+1)^3$).
\end{solution}

\begin{solution}{exo:b1:vspaces:5}
$F \cap G$: the conditions $x = y = z$ and $x = t = 0$ together give
$x = 0$, hence $y = z = 0$, and $t = 0$: the intersection is
$\{0\}$. Sum: given $(x,y,z,t)$, seek $(a,a,a,b) \in F$ and
$(0,c,d,0) \in G$ summing to it: $a = x$, $b = t$, $c = y - x$, $d =
z - x$: always possible. So $\R^4 = F \oplus G$, and
\[
(1,2,3,4) = (1,1,1,4) + (0,1,2,0) .
\]
\end{solution}

\begin{solution}{exo:b1:vspaces:6}
An element of $G$ is $\lambda u$; if it converges, then (as
$\lambda u_n = \lambda(-1)^n$ has the two subsequential limits $\pm
\lambda$) necessarily $\lambda = 0$: $F \cap G = \{0\}$.

$F + G$ is not everything: it consists of sequences of the form
$c_n + \lambda(-1)^n$ with $(c_n)$ convergent. The sequence $v_n =
n$ is not of this form ($v_n - \lambda(-1)^n$ is unbounded, never
convergent). So $F \oplus G \subsetneq$ (space of all sequences).
\end{solution}

\begin{solution}{exo:b1:vspaces:7}
($\supseteq$) Both $F \cap G$ and $F \cap H$ lie in $F$, and their
sum lies in $G + (F \cap H)$: inclusion follows since the left side
is a subspace containing both pieces --- concretely, an element $g +
h$ with $g \in F\cap G$, $h \in F \cap H$ is in $F$ (sum of two
elements of $F$) and in $G + (F \cap H)$.

($\subseteq$) Let $x \in F$ with $x = g + h$, $g \in G$, $h \in F
\cap H$. Then $g = x - h \in F$ (difference of elements of $F$), so
$g \in F \cap G$, and $x = g + h \in (F \cap G) + (F \cap H)$.

Counterexample to full distributivity in $\R^2$: $F =
\operatorname{Vect}(1,1)$, $G = \operatorname{Vect}(1,0)$, $H =
\operatorname{Vect}(0,1)$. Then $G + H = \R^2$, so $F \cap (G+H) =
F$, while $F \cap G = F \cap H = \{0\}$: the right side is $\{0\}
\neq F$.
\end{solution}

\begin{solution}{exo:b1:vspaces:8}
\begin{enumerate}
  \item Suppose $\sum_{i} \lambda_i \eu^{a_i x} = 0$ for all $x$.
        Multiply by $\eu^{-a_p x}$: $\lambda_p + \sum_{i < p}
        \lambda_i \eu^{(a_i - a_p)x} \to \lambda_p$ as $x \to
        +\infty$ (each exponent $a_i - a_p < 0$). The left side is
        identically $0$, so $\lambda_p = 0$; iterate downward.
  \item Let $a\cos x + b \sin x + c \cos 2x + d\sin 2x = 0$ for all
        $x$. Evaluate at $x = 0$: $a + c = 0$; at $x = \pi$: $-a + c
        = 0$; hence $a = c = 0$, and the relation reduces to $b\sin
        x + d \sin 2x = 0$. Evaluate at $x = \frac\pi2$: $b = 0$;
        then at $x = \frac\pi4$: $d = 0$.
  \item Suppose $\sum \lambda_i \abs{x - a_i} = 0$ for all $x$. The
        function $\sum_{i \neq j} \lambda_i\abs{x - a_i}$ is
        differentiable at $a_j$ (each term is, away from its own
        corner), so $-\lambda_j \abs{x - a_j}$, their difference,
        must be differentiable at $a_j$ too --- which forces
        $\lambda_j = 0$ ($\abs{\,\cdot\,}$ has a corner). This holds
        for every $j$.
\end{enumerate}
\end{solution}

\begin{solution}{exo:b1:vspaces:9}
Let $h \in H$. Since $h \in H \subseteq F + H = F + G$, write $h = f
+ g$ with $f \in F$, $g \in G$. Then $f = h - g \in H$ (both terms
in $H$, using $G \subseteq H$), so $f \in F \cap H = F \cap G
\subseteq G$, and $h = f + g \in G$. Hence $H \subseteq G$, and with
the hypothesis $G \subseteq H$: equality.

Counterexample without $G \subseteq H$: in $\R^2$, take $F =
\operatorname{Vect}(1,0)$, $G = \operatorname{Vect}(0,1)$, $H =
\operatorname{Vect}(1,1)$: then $F + G = F + H = \R^2$ and $F \cap G
= F \cap H = \{0\}$, yet $G \neq H$.
\end{solution}
