\begin{solution}{exo:b1:vspaces:1}
\begin{enumerate}
  \item Yes: contains $0$, and the defining equation is linear
        (stable under $x + \lambda y$).
  \item No: does not contain $(0,0,0)$.
  \item No: $(1, 0)$ and $(0, -1)$ belong ($xy = 0$), their sum $(1,
        -1)$ does not ($xy = -1 < 0$).
  \item Yes: $0$ vanishes at $1$; $(P + \lambda Q)(1) = P(1) +
        \lambda Q(1) = 0$.

\end{enumerate}

\begin{solution}{exo:b1:vspaces:2}
$(1,2,1) = a(1,0,1) + b(1,1,0)$ requires $a + b = 1$, $b = 2$, $a =
1$: incompatible ($a + b = 3 \neq 1$): not in the span. $(2,1,1) =
a(1,0,1) + b(1,1,0)$: $b = 1$, $a = 1$, $a + b = 2$: consistent, so
$(2,1,1) = (1,0,1) + (1,1,0)$, in the span.

Equation: $(x, y, z) = (a + b, b, a)$ means $x = y + z$: the span is
the plane $\{x - y - z = 0\}$.
\end{solution}

\begin{solution}{exo:b1:vspaces:3}
First family: $\lambda(1,1,0) + \mu(1,0,1) + \nu(0,1,1) = 0$ gives
$\lambda + \mu = 0$, $\lambda + \nu = 0$, $\mu + \nu = 0$: adding,
$2(\lambda + \mu + \nu) = 0$, and subtracting each equation,
$\lambda = \mu = \nu = 0$: free.

Second: $(2,4,6) = 2(1,2,3)$: linked.

Third: four vectors in $\R^3$ --- necessarily linked once dimension
is available (\cref{ch:b1:findim}); directly: $(0,1,1) = -(1,0,0) +
0\cdot(1,1,0) + (1,1,1)$, indeed $(-1,0,0) + (1,1,1) = (0,1,1)$: a
nontrivial relation.
\end{solution}

\begin{solution}{exo:b1:vspaces:4}
Polynomials $1, (X-1), (X-1)^2, (X-1)^3$ have distinct degrees $0,
1, 2, 3$: free (\cref{prop:b1:vspaces:freecriteria}~(1)); four free
vectors generating (every $P \in \R_3[X]$ expands in powers of $X -
1$, e.g.\ by Taylor for polynomials, cf.\ the proof of
\cref{prop:b1:poly:multiplicity}): a basis. For $X^3$, Taylor at
$1$: $P = X^3$, $P(1) = 1$, $P'(1) = 3$, $P''(1) = 6$, $P'''(1) =
6$:
\[
X^3 = 1 + 3(X - 1) + 3(X-1)^2 + (X-1)^3 ,
\]
coordinates $(1, 3, 3, 1)$ (Pascal's row, as expected from $X^3 =
((X-1)+1)^3$).
\end{solution}

\begin{solution}{exo:b1:vspaces:5}
$F \cap G$: the conditions $x = y = z$ and $x = t = 0$ together give
$x = 0$, hence $y = z = 0$, and $t = 0$: the intersection is
$\{0\}$. Sum: given $(x,y,z,t)$, seek $(a,a,a,b) \in F$ and
$(0,c,d,0) \in G$ summing to it: $a = x$, $b = t$, $c = y - x$, $d =
z - x$: always possible. So $\R^4 = F \oplus G$, and
\[
(1,2,3,4) = (1,1,1,4) + (0,1,2,0) .
\]
\end{solution}

\begin{solution}{exo:b1:vspaces:6}
An element of $G$ is $\lambda u$; if it converges, then (as
$\lambda u_n = \lambda(-1)^n$ has the two subsequential limits $\pm
\lambda$) necessarily $\lambda = 0$: $F \cap G = \{0\}$.

$F + G$ is not everything: it consists of sequences of the form
$c_n + \lambda(-1)^n$ with $(c_n)$ convergent. The sequence $v_n =
n$ is not of this form ($v_n - \lambda(-1)^n$ is unbounded, never
convergent). So $F \oplus G \subsetneq$ (space of all sequences).
\end{solution}

\begin{solution}{exo:b1:vspaces:7}
($\supseteq$) Both $F \cap G$ and $F \cap H$ lie in $F$, and their
sum lies in $G + (F \cap H)$: inclusion follows since the left side
is a subspace containing both pieces --- concretely, an element $g +
h$ with $g \in F\cap G$, $h \in F \cap H$ is in $F$ (sum of two
elements of $F$) and in $G + (F \cap H)$.

($\subseteq$) Let $x \in F$ with $x = g + h$, $g \in G$, $h \in F
\cap H$. Then $g = x - h \in F$ (difference of elements of $F$), so
$g \in F \cap G$, and $x = g + h \in (F \cap G) + (F \cap H)$.

Counterexample to full distributivity in $\R^2$: $F =
\operatorname{Vect}(1,1)$, $G = \operatorname{Vect}(1,0)$, $H =
\operatorname{Vect}(0,1)$. Then $G + H = \R^2$, so $F \cap (G+H) =
F$, while $F \cap G = F \cap H = \{0\}$: the right side is $\{0\}
\neq F$.
\end{solution}

\begin{solution}{exo:b1:vspaces:8}
\begin{enumerate}
  \item Suppose $\sum_{i} \lambda_i \eu^{a_i x} = 0$ for all $x$.
        Multiply by $\eu^{-a_p x}$: $\lambda_p + \sum_{i < p}
        \lambda_i \eu^{(a_i - a_p)x} \to \lambda_p$ as $x \to
        +\infty$ (each exponent $a_i - a_p < 0$). The left side is
        identically $0$, so $\lambda_p = 0$; iterate downward.
  \item Let $a\cos x + b \sin x + c \cos 2x + d\sin 2x = 0$ for all
        $x$. Evaluate at $x = 0$: $a + c = 0$; at $x = \pi$: $-a + c
        = 0$; hence $a = c = 0$, and the relation reduces to $b\sin
        x + d \sin 2x = 0$. Evaluate at $x = \frac\pi2$: $b = 0$;
        then at $x = \frac\pi4$: $d = 0$.
  \item Suppose $\sum \lambda_i \abs{x - a_i} = 0$ for all $x$. The
        function $\sum_{i \neq j} \lambda_i\abs{x - a_i}$ is
        differentiable at $a_j$ (each term is, away from its own
        corner), so $-\lambda_j \abs{x - a_j}$, their difference,
        must be differentiable at $a_j$ too --- which forces
        $\lambda_j = 0$ ($\abs{\,\cdot\,}$ has a corner). This holds
        for every $j$.
\end{enumerate}
\end{solution}

\begin{solution}{exo:b1:vspaces:9}
Let $h \in H$. Since $h \in H \subseteq F + H = F + G$, write $h = f
+ g$ with $f \in F$, $g \in G$. Then $f = h - g \in H$ (both terms
in $H$, using $G \subseteq H$), so $f \in F \cap H = F \cap G
\subseteq G$, and $h = f + g \in G$. Hence $H \subseteq G$, and with
the hypothesis $G \subseteq H$: equality.

Counterexample without $G \subseteq H$: in $\R^2$, take $F =
\operatorname{Vect}(1,0)$, $G = \operatorname{Vect}(0,1)$, $H =
\operatorname{Vect}(1,1)$: then $F + G = F + H = \R^2$ and $F \cap G
= F \cap H = \{0\}$, yet $G \neq H$.
\end{solution}

\begin{solution}{exo:b1:vspaces:10}
Both sets are subspaces (the defining conditions are linear and
hold for $0$). Decomposition: for $P \in \R[X]$,
\[
P(X) = \underbrace{\frac{P(X) + P(-X)}{2}}_{\in\,\mathcal P}
+ \underbrace{\frac{P(X) - P(-X)}{2}}_{\in\,\mathcal I},
\]
and a polynomial both even and odd satisfies $P = -P$, so $P = 0$:
the sum is direct and equals $\R[X]$.

Now let $P = \sum_k a_k X^k$ be even. Then $P(X) - P(-X) = 2
\sum_{k \text{ odd}} a_k X^k$ is the zero polynomial, so every
odd-degree coefficient vanishes (\cref{def:b1:poly:def}): $P \in
\operatorname{Vect}(1, X^2, X^4, \dots)$, i.e.\ $P = Q(X^2)$ for a
polynomial $Q$. Conversely every polynomial in $X^2$ is even.
\end{solution}

\begin{solution}{exo:b1:vspaces:11}
Suppose $a(x_1 + x_2) + b(x_2 + x_3) + c(x_3 + x_1) = 0$.
Regrouping on the free family $(x_1, x_2, x_3)$:
\[
(a + c)\,x_1 + (a + b)\,x_2 + (b + c)\,x_3 = 0
\implies a + c = a + b = b + c = 0 .
\]
Subtracting the first two equations gives $c = b$; the third then
gives $2b = 0$, so $b = c = 0$, then $a = 0$: the family is free.

For four vectors the analogous family is \emph{always} linked:
\[
(x_1 + x_2) - (x_2 + x_3) + (x_3 + x_4) - (x_4 + x_1) = 0
\]
is a nontrivial null combination (coefficients $1, -1, 1, -1$),
whatever $(x_1, x_2, x_3, x_4)$ is. The parity of the cycle length
decides.
\end{solution}

\begin{solution}{exo:b1:vspaces:12}
\begin{enumerate}
  \item If $F_1 \subseteq F_2$ or $F_2 \subseteq F_1$, the union
        is one of the two, hence proper. Otherwise pick $x \in F_1
        \setminus F_2$ and $y \in F_2 \setminus F_1$, and consider
        $x + y$. If $x + y \in F_1$, then $y = (x + y) - x \in
        F_1$: contradiction. If $x + y \in F_2$, then $x \in F_2$:
        contradiction. So $x + y \notin F_1 \cup F_2$, and $E \neq
        F_1 \cup F_2$.
  \item Suppose for contradiction that $E = F_1 \cup \dots \cup
        F_k$, with $k$ chosen \emph{minimal} among all such
        coverings. Minimality forbids $F_1 \subseteq F_2 \cup
        \dots \cup F_k$ (else drop $F_1$), so there is $x \in F_1$
        with $x \notin F_i$ for all $i \geq 2$. Since $F_1$ is
        proper, pick $y \notin F_1$. For each scalar $t$, the
        vector $y + t x$ lies in some $F_i$. It never lies in
        $F_1$: otherwise $y = (y + tx) - tx \in F_1$ (as $x \in
        F_1$). The field is infinite, so choose $k$ distinct
        scalars $t_1, \dots, t_k$: the $k$ vectors $y + t_j x$
        fall into the $k - 1$ subspaces $F_2, \dots, F_k$, and two
        of them, say $y + t x$ and $y + t' x$ with $t \neq t'$,
        lie in the same $F_i$ ($i \geq 2$). Then their difference
        $(t - t')x \in F_i$, so $x \in F_i$: contradiction. Hence
        no finite covering by proper subspaces exists.
\end{enumerate}
\end{solution}

\begin{solution}{pb:b1:vspaces:1}
\textbf{1.} $L_i$ is a product of $n$ linear factors divided by a
nonzero constant (the $x_i$ are distinct), so $\deg L_i = n$.
Evaluating at $x_j$ with $j \neq i$: the factor $X - x_j$ of the
numerator vanishes, so $L_i(x_j) = 0$. At $x_i$, numerator and
denominator coincide: $L_i(x_i) = 1$.

\textbf{2.} Suppose $\sum_i \lambda_i L_i = 0$. Evaluate at $x_j$:
all terms die except $\lambda_j L_j(x_j) = \lambda_j$, so
$\lambda_j = 0$ for every $j$: the family is free.

\textbf{3.} Let $D = P - \sum_i P(x_i) L_i$. Then $\deg D \leq n$
and, by question 1, $D(x_j) = P(x_j) - P(x_j) = 0$ for the $n + 1$
distinct points $x_0, \dots, x_n$. A nonzero polynomial of degree
$\leq n$ has at most $n$ roots (\cref{cor:b1:poly:nroots}), so $D
= 0$. Thus every $P \in \R_n[X]$ is a combination of the $L_i$:
the family is generating, and with question 2, a basis.

\textbf{4.} Given $y_0, \dots, y_n$, the polynomial $P = \sum_i
y_i L_i$ has degree $\leq n$ and interpolates. Uniqueness: an
interpolating $P$ has, by question 3, coordinates $(P(x_0), \dots,
P(x_n)) = (y_0, \dots, y_n)$ in the basis $(L_i)$, and coordinates
in a basis are unique (\cref{prop:b1:vspaces:coordinates}). The
coordinates of $P$ in the Lagrange basis are its \emph{values at
the nodes} --- this is the whole point of the basis.

\textbf{5.} Apply question 3 to $P = X^k$ ($0 \leq k \leq n$):
\[
X^k = \sum_{i=0}^{n} x_i^{k} L_i ,
\]
and $k = 0$ gives $\sum_i L_i = 1$.

\textbf{6.} $\deg N_k = k$ exactly: the family $(N_0, \dots, N_n)$
is a staircase of degrees in $\R_n[X]$, hence a basis by
\cref{ex:b1:vspaces:staircase} (freeness from
\cref{prop:b1:vspaces:freecriteria}~(1), generation by finite
descent on the degree).

\textbf{7.} From the recurrence,
\[
f[x_0, x_1] = \frac{f(x_1) - f(x_0)}{x_1 - x_0},
\qquad
f[x_0, x_1, x_2] = \frac{f[x_1, x_2] - f[x_0, x_1]}{x_2 - x_0}.
\]
For $f(x) = x^2$:
\[
f[x_0, x_1] = \frac{x_1^2 - x_0^2}{x_1 - x_0} = x_0 + x_1,
\]
and then
\[
f[x_0, x_1, x_2]
= \frac{(x_1 + x_2) - (x_0 + x_1)}{x_2 - x_0}
= \frac{x_2 - x_0}{x_2 - x_0} = 1 .
\]

\textbf{8.} $\deg S \leq n$ since $Q, R$ have degree $\leq n - 1$.
At $x_0$: $S(x_0) = \frac{-(x_0 - x_n) R(x_0)}{x_n - x_0} =
R(x_0) = f(x_0)$. At $x_n$: $S(x_n) = \frac{(x_n - x_0)
Q(x_n)}{x_n - x_0} = Q(x_n) = f(x_n)$. At an interior node $x_i$
($1 \leq i \leq n-1$), both $Q$ and $R$ take the value $f(x_i)$,
so
\[
S(x_i) = \frac{(x_i - x_0) - (x_i - x_n)}{x_n - x_0}\, f(x_i)
= f(x_i) .
\]

\textbf{9.} Induction on the number of nodes. One node: the
interpolant is the constant $f(x_0) = f[x_0]$. Assume the claim
for $k$ nodes and let $S$ interpolate at $x_0, \dots, x_k$; by
uniqueness (question 4), $S$ is given by Aitken's lemma from $R$
(nodes $x_0, \dots, x_{k-1}$) and $Q$ (nodes $x_1, \dots, x_k$).
The coefficient of $X^{k}$ in $S$ is
\[
\frac{[X^{k-1}]\,Q - [X^{k-1}]\,R}{x_k - x_0}
= \frac{f[x_1, \dots, x_k] - f[x_0, \dots, x_{k-1}]}{x_k - x_0}
= f[x_0, \dots, x_k]
\]
by the induction hypothesis and the defining recurrence.

\textbf{10.} Let $P_k$ interpolate $f$ at $x_0, \dots, x_k$. The
difference $P_k - P_{k-1}$ has degree $\leq k$ and vanishes at
$x_0, \dots, x_{k-1}$, so by the factor theorem applied $k$ times
(\cref{thm:b1:poly:factor}) it equals $c\,N_k$ for a constant $c$;
comparing coefficients of $X^{k}$ and using question 9, $c =
f[x_0, \dots, x_k]$. Telescoping from $P_0 = f(x_0) N_0$ gives
Newton's formula. For the closed form, write $P_k = \sum_{i \leq
k} f(x_i) L_i$ (Lagrange, on the nodes $x_0, \dots, x_k$) and read
the coefficient of $X^{k}$: each $L_i$ contributes
$\frac{1}{\prod_{j \neq i}(x_i - x_j)}$, whence
\[
f[x_0, \dots, x_k] = \sum_{i=0}^{k}
\frac{f(x_i)}{\prod_{j \neq i,\, j \leq k}(x_i - x_j)} .
\]
The right side is invariant under any permutation of the nodes, so
the divided difference does not depend on their ordering.

\textbf{11.} If $P = a X^m + (\text{lower degrees})$, the binomial
theorem gives
\[
\Delta P = a\bigl((X+1)^m - X^m\bigr) + \dots
= a\,m\,X^{m-1} + (\text{lower degrees}),
\]
because $(X+1)^m - X^m = m X^{m-1} + \dots$ and the lower-degree
part of $P$ contributes degree $\leq m - 2$ after $\Delta$ (or
degree $\leq m-2$ terms). So $\deg \Delta P = m - 1$ with leading
coefficient $m a$. A constant $c$ gives $\Delta c = c - c = 0$.

\textbf{12.} $\deg B_k = k$: staircase, hence a basis of
$\R_n[X]$ (\cref{ex:b1:vspaces:staircase}). For $\Delta B_k$
($k \geq 1$), factor the common product:
\begin{align*}
k!\,\Delta B_k
&= (X+1)X\cdots(X-k+2) - X(X-1)\cdots(X-k+1) \\
&= X(X-1)\cdots(X-k+2)\,\bigl[(X+1) - (X-k+1)\bigr] \\
&= k\,X(X-1)\cdots(X-k+2),
\end{align*}
so $\Delta B_k = \frac{X(X-1)\cdots(X-k+2)}{(k-1)!} = B_{k-1}$.

\textbf{13.} Write $P = \sum_{k=0}^{n} c_k B_k$ (basis, question
12). Apply $\Delta^{j}$: by question 12, $\Delta^{j} P = \sum_{k
\geq j} c_k B_{k-j}$. Evaluate at $0$: $B_0(0) = 1$ and $B_m(0) =
0$ for $m \geq 1$ (the factor $X$ vanishes), so
$\bigl(\Delta^{j}P\bigr)(0) = c_j$. This is the forward-difference
formula.

\textbf{14.} Induction on $k$. For $k = 0$ the identity reads
$P(0) = P(0)$. Assume it for $k$ and apply it to $\Delta P$:
\[
\bigl(\Delta^{k+1} P\bigr)(0)
= \sum_{j=0}^{k} (-1)^{k-j}\binom kj \bigl(P(j+1) - P(j)\bigr).
\]
Collect the coefficient of $P(i)$: it is $(-1)^{k-i+1}\binom
k{i-1}\cdot(-1)^{0}$ from the first sum (shifted) and
$-(-1)^{k-i}\binom ki$ from the second --- together
\[
(-1)^{k+1-i}\Bigl(\binom k{i-1} + \binom ki\Bigr)
= (-1)^{k+1-i}\binom{k+1}i
\]
by Pascal's rule, which is the identity at rank $k + 1$.

\textbf{15.} Iterating question 11 from degree $n$, leading
coefficient $a_n$: after one $\Delta$, degree $n-1$ and leading
coefficient $n a_n$; after two, $n(n-1)a_n$; after $n$ steps,
degree $0$ and value $n(n-1)\cdots 1\, a_n = n!\,a_n$, a constant.
One more $\Delta$ kills it: $\Delta^{n+1}P = 0$.

\textbf{16.} If $m \geq k$: $B_k(m) = \binom mk \in \N$. If $0
\leq m < k$: one factor of $m(m-1)\cdots(m-k+1)$ is zero, so
$B_k(m) = 0$. If $m = -q$ with $q \geq 1$:
\[
B_k(-q) = \frac{(-q)(-q-1)\cdots(-q-k+1)}{k!}
= (-1)^k\,\frac{q(q+1)\cdots(q+k-1)}{k!}
= (-1)^k \binom{q+k-1}{k},
\]
an integer. So every $B_k$ maps $\Z$ into $\Z$.

\textbf{17.} ($\Leftarrow$) If $P = \sum_k c_k B_k$ with $c_k \in
\Z$, then for $m \in \Z$, $P(m) = \sum_k c_k B_k(m) \in \Z$ by
question 16. ($\Rightarrow$) If $P$ is integer-valued, its
coordinates are $c_k = \bigl(\Delta^k P\bigr)(0) = \sum_{j=0}^k
(-1)^{k-j}\binom kj P(j)$ (questions 13 and 14), an integer
combination of the integers $P(0), \dots, P(k)$. This is
P\'olya's characterization of integer-valued polynomials.

\textbf{18.} Set $Q(X) = P(X + a)$, a polynomial of degree $\leq
n$ with $Q(0), Q(1), \dots, Q(n) \in \Z$. Its coordinates in
$(B_k)_{k \leq n}$ are $c_k = \sum_{j \leq k}(-1)^{k-j}\binom kj
Q(j) \in \Z$ (question 14 uses only the values at $0, \dots, k
\leq n$). By question 17 ($\Leftarrow$), $Q$ is integer-valued on
all of $\Z$, hence so is $P(X) = Q(X - a)$.

\textbf{19.} A product of $k$ consecutive integers is $m(m-1)
\cdots(m-k+1) = k!\,B_k(m)$ for some $m \in \Z$, and $B_k(m) \in
\Z$ by question 16: the product is divisible by $k!$.

\textbf{20.} Values of $P = \frac{X(X+1)(2X+1)}{6}$ at $0,1,2,3$:
$0, 1, 5, 14$. Difference table: $\Delta$ row $1, 4, 9$;
$\Delta^2$ row $3, 5$; $\Delta^3$ row $2$. Hence, by question 13,
\[
P = 0\cdot B_0 + 1\cdot B_1 + 3\,B_2 + 2\,B_3 ,
\]
with integer coordinates: $P$ is integer-valued (question 17),
while its monomial coefficients $\frac13, \frac12, \frac16$ are
not integers. Direct computation:
\begin{align*}
\Delta P &= \frac{(X+1)(X+2)(2X+3) - X(X+1)(2X+1)}{6} \\
&= \frac{(X+1)\bigl[(X+2)(2X+3) - X(2X+1)\bigr]}{6}
= \frac{(X+1)(6X+6)}{6} = (X+1)^2 .
\end{align*}
Telescoping $P(m) = \sum_{j=0}^{m-1}\Delta P(j) = \sum_{j=1}^{m}
j^2$ (with $P(0) = 0$): the sum-of-squares formula.

\textbf{21.} The values $2^i$ at $i = 0, \dots, n$ have difference
table constantly $1$ on the left edge: $\Delta^k$ of the sequence
$(2^i)$ is $(2^i)$ again (since $2^{i+1} - 2^i = 2^i$), so
$\bigl(\Delta^k P\bigr)(0) = 2^0 = 1$ for all $k \leq n$, and $P =
B_0 + B_1 + \dots + B_n$ by question 13. Then
\[
P(n+1) = \sum_{k=0}^{n}\binom{n+1}{k}
= 2^{n+1} - \binom{n+1}{n+1} = 2^{n+1} - 1 \neq 2^{n+1}:
\]
the pattern breaks at the first uncontrolled point.

\textbf{22.} By question 12, $B_k = \Delta B_{k+1}$, so
\[
\sum_{j=0}^{m-1} B_k(j)
= \sum_{j=0}^{m-1}\bigl(B_{k+1}(j+1) - B_{k+1}(j)\bigr)
= B_{k+1}(m) - B_{k+1}(0) = B_{k+1}(m).
\]
For $j < k$ the terms $B_k(j)$ vanish, so the sum really starts at
$j = k$: $\sum_{j=k}^{m-1}\binom jk = \binom m{k+1}$, the
hockey-stick identity.

\textbf{23.} Difference tables (or direct expansion) give
\[
X^2 = B_1 + 2 B_2, \qquad X^3 = B_1 + 6 B_2 + 6 B_3
\]
(check: $B_1 + 2B_2 = X + X(X-1) = X^2$; at $X = 1, 2, 3$ the
second gives $1, 8, 27$). Question 22 then yields
\[
\sum_{j=0}^{m-1} j^2 = B_2(m) + 2B_3(m)
= \binom m2 + 2\binom m3 = \frac{m(m-1)(2m-1)}{6},
\]
\[
\sum_{j=0}^{m-1} j^3 = B_2(m) + 6B_3(m) + 6B_4(m)
= \binom m2 + 6\binom m3 + 6\binom m4 .
\]
Expanding the last expression: $\binom m2 + 6\binom m3 + 6\binom
m4 = \frac{m(m-1)}{2}\bigl[1 + 2(m-2) +
\frac{(m-2)(m-3)}{2}\bigr] = \frac{m^2(m-1)^2}{4} = \binom m2^2$.
Replacing $m$ by $m + 1$: $1^3 + \dots + m^3 =
\bigl(\frac{m(m+1)}2\bigr)^2 = (1 + \dots + m)^2$, Nicomachus'
identity.

\textbf{24.} Nodes $0, 1, 2$, polynomial $X^2$. Monomial basis
$(1, X, X^2)$: coordinates $(0, 0, 1)$. Lagrange basis:
coordinates are the values $(0, 1, 4)$ (question 4). Newton basis
$(1, X, X(X-1))$: divided differences $f[0] = 0$, $f[0,1] = 1$,
$f[0,1,2] = \frac{3 - 1}{2} = 1$ (question 9), so coordinates
$(0, 1, 1)$ --- indeed $X + X(X-1) = X^2$. Three bases, three
coordinate vectors, one polynomial.

\textbf{25.} (i) Question 4 is automatic because $(L_i)$ is a
\emph{basis}: existence and uniqueness of interpolation are
exactly existence and uniqueness of coordinates. (ii) The Newton
basis is a staircase, so coordinates are computed by successive
divisions --- each new node adds one term without disturbing the
previous ones --- whereas the Lagrange coordinates of $P$ are the
values $P(x_i)$, available with no computation at all. (iii) Both
bases are free by the same two criteria of
\cref{prop:b1:vspaces:freecriteria}: distinct degrees for Newton,
evaluation at the nodes for Lagrange. (iv) P\'olya's theorem says
that ``$P(\Z) \subseteq \Z$'', a property of values, is equivalent
to integrality of the coordinates in the basis $(B_k)$ --- the
arithmetic of a polynomial becomes visible only in the basis
adapted to the question.
