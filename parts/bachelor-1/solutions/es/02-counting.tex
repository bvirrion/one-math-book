\begin{solution}{exo:b1:counting:5}
Para $1 \leq k \leq n$,
\[
k \binom nk = k\,\frac{n!}{k!\,(n-k)!}
= n\,\frac{(n-1)!}{(k-1)!\,(n-k)!} = n \binom{n-1}{k-1}.
\]
Suma y reindexación con $j = k - 1$:
\[
\sum_{k=0}^{n} k \binom nk = n \sum_{j=0}^{n-1} \binom{n-1}{j}
= n\, 2^{n-1}
\]
por \cref{prop:b1:counting:identities}. (Alternativa: diferenciar
$(1+x)^n = \sum_k \binom nk x^k$ y establezca $x = 1$).
\end{solution}

\begin{solution}{exo:b1:counting:6}
Una \omterm{def:b1:logic:map}{aplicación} $f \colon \intint{1}{k} \to \intint{1}{n}$ estrictamente creciente
está determinado por su imagen, un subconjunto $k$ de $\intint{1}{n}$(enumere el
subconjunto en orden creciente); por el contrario, cada subconjunto $k$ da exactamente
uno de esos \omterm{def:b1:logic:map}{aplicación}. Por lo tanto, $\binom nk$ aumenta estrictamente \omterm{def:b1:logic:map}{aplicaciones}.

Si $f$ simplemente aumenta, configure $g(i) = f(i) + i - 1$. Entonces $g$ es
estrictamente creciente (entre argumentos consecutivos, $f$ gana $\geq 0$
y $i - 1$ gana $1$) con valores en $\intint{1}{n + k - 1}$; y
$f(i) = g(i) - i + 1$ recupera $f$ de cualquier $g$ estrictamente creciente
en $\intint{1}{n+k-1}$. Esta es una biyección, por lo que hay
$\binom{n + k - 1}{k}$ aumentando \omterm{def:b1:logic:map}{aplicaciones}.
\end{solution}

\begin{solution}{exo:b1:counting:7}
Dividir un \omterm{def:b1:logic:sets}{conjunto} $E$ con elementos $m + n$ en bloques $M$(elementos $m$)
y $N$(elementos $n$). Un subconjunto $k$ de $E$ contiene algunos elementos $j$
de $M$($0 \leq j \leq k$) y $k - j$ de $N$; para $j$ fijo hay
$\binom mj \binom{n}{k-j}$ dichos subconjuntos, y en los casos $j = 0, \dots,
k $\omterm{thm:b1:logic:partition}{dividir} los subconjuntos $k$. La regla de la suma le da a Vandermonde
identidad\index{La identidad de Vandermonde}.

Con $m = n = k$:$\binom{2n}{n} = \sum_{j=0}^{n} \binom nj
\binom{n}{n-j} = \sum_{j=0}^{n} \binom nj^2$, usando $\binom{n}{n-j} =
\binom nj$.
\end{solution}

\begin{solution}{exo:b1:counting:10}
Una solución de $x_1 + \dots + x_n = k$ en $\N^n$ se codifica como una fila de
$k$ estrellas y $n - 1$ barras: escribe $x_1$ estrellas, una barra,$x_2$ estrellas, a
barra, \dots, que termina con estrellas $x_n$. Esta es una biyección sobre el
palabras de longitud $k + n - 1$ usando estrellas $k$ y barras $n - 1$, y
esas palabras están determinadas por las posiciones de las estrellas:
$\binom{n + k - 1}{k}$. Las selecciones con repetición corresponden a
soluciones de la ecuación ($x_i$= número de copias del objeto $i$), entonces
el conteo es el mismo.
\end{solution}

\begin{solution}{exo:b1:counting:11}
Con $A_i = \{\sigma : \sigma(i) = i\}$, un \omterm{def:b1:counting:objects}{permutación} en
$\bigcap_{i \in I} A_i$ arregla cada $i \in I$ y permuta el otro
$n - \abs I$ puntos libremente:$\abs{\bigcap_{i \in I} A_i} =
(n - \abs I)!$. Inclusión-exclusión:
\[
\Bigl|\bigcup_i A_i\Bigr|
= \sum_{k=1}^{n} (-1)^{k+1} \binom nk (n-k)!
= \sum_{k=1}^{n} (-1)^{k+1} \frac{n!}{k!} ,
\]
ya que hay $\binom nk$ subconjuntos $I$ de tamaño $k$. Por lo tanto
\[
D_n = n! - \Bigl|\bigcup_i A_i\Bigr|
= n!\Bigl(1 - \sum_{k=1}^{n} \frac{(-1)^{k+1}}{k!}\Bigr)
= n! \sum_{k=0}^{n} \frac{(-1)^k}{k!} .
\]
Para la segunda identidad: clasificar el \omterm{def:b1:counting:objects}{permutaciones} $\sigma$ de
$\intint{1}{n}$ por su punto fijo \omterm{def:b1:logic:sets}{conjunto}$F(\sigma)$. por un fijo
$k$-subconjunto $F$, el \omterm{def:b1:counting:objects}{permutaciones} con $F(\sigma) = F$ son exactamente los
Trastornos del complemento: $D_{n-k}$ de ellos. resumiendo el
$\binom nk$ opciones de $F$ para cada $k$:
$n! = \sum_{k=0}^{n} \binom nk D_{n-k}$.
\end{solution}

\begin{solution}{exo:b1:counting:12}
\emph{Primera identidad.} Cuente los pares $(A, a)$ donde $A \subseteq E$
($\abs E = n$) y $a \in A$. Por tamaño de $A$:$\sum_k \binom nk k$
pares. Eligiendo primero el elemento marcado: $n$ opciones para $a$, luego
cualquier subconjunto de los elementos $n - 1$ restantes para completar $A$:
$n\,2^{n-1}$ pares.

\emph{Segunda identidad.} Cuéntense los triples $(A,a,b)$ con
$a,b \in A$ (posiblemente $a=b$). Por tamaño:
$\sum_k k^2 \binom nk$. Directamente: o bien $a=b$
($n\,2^{n-1}$ triples, recuento anterior), o $a \neq b$
($n(n-1)$ opciones ordenadas, luego cualquier subconjunto de los
otros $n-2$ elementos: $n(n-1)\,2^{n-2}$). En total
\[
n\,2^{n-1} + n(n-1)\,2^{n-2} = n\,2^{n-2}\,(2 + n - 1)
= n(n+1)\,2^{n-2} .
\]
\end{solution}

\begin{solution}{pb:b1:counting:1}
\textbf{1.} $D_1 = 0$(el único \omterm{def:b1:counting:objects}{permutación} corrige $1$),$D_2 = 1$

\textbf{2.} Un \omterm{def:b1:counting:objects}{permutación} con exactamente $k$ puntos fijos es
determinado por la elección de su punto fijo \omterm{def:b1:logic:sets}{conjunto} $F$($\binom nk$
maneras) junto con su restricción al complemento, que debe
ser un \omterm{def:b1:counting:objects}{permutación} de puntos $n - k$ con punto fijo \emph{no}
($D_{n-k}$ formas). Las dos opciones son independientes y la
La correspondencia es \omterm{def:b1:logic:inj}{biyectivo}: $P_k(n) = \binom nk D_{n-k}$.

\textbf{3.}$P_0(4) = D_4 = 9$;$P_1(4) = \binom41 D_3 = 4 \times 2

\textbf{4.} Cuente los pares $(\sigma, i)$ con $\sigma(i) = i$. Para
fijo $i$, el \omterm{def:b1:counting:objects}{permutaciones} que fija $i$ son los \omterm{def:b1:counting:objects}{permutaciones} del
otros puntos $n - 1$:$(n-1)!$ de ellos. De ahí el número de pares.
es $n \cdot (n-1)! = n!$, y este número también es
$\sum_\sigma \abs{\mathrm{Fix}(\sigma)}$. Dividiendo por el número
$n!$ de \omterm{def:b1:counting:objects}{permutaciones}: el número medio de puntos fijos es
exactamente $1$, por cada $n \geq 1$.

\textbf{5.} Sea $\sigma$ un trastorno de $\intint1{n+1}$ y
$j = \sigma(n+1) \in \intint1n$:$n$ valores posibles. \emph{Case
$\sigma(j) = n+1$:} los puntos $j$ y $n+1$ de intercambio, y $\sigma$
restringido a los puntos $n - 1$ restantes es una opción arbitraria.
trastorno de ellos: $D_{n-1}$ posibilidades. \emph{Case $\sigma(j)
\neq n+1$:} deja $i_0 = \sigma^{-1}(n+1)$; aquí $i_0 \neq j $ y
$i_0 \leq n$. Defina $\tau$ en $\intint1n$ por $\tau(i) = \sigma(i)$
para $i \neq i_0$ y $\tau(i_0) = j$. Entonces $\tau$ es un \omterm{def:b1:counting:objects}{permutación}
de $\intint1n $(el valor $n+1$ ha sido reemplazado por el que falta
valor $j$), y es un trastorno:$\tau(i_0) = j \neq i_0$, y
$\tau(i) = \sigma(i) \neq i$ en otro lugar. Por el contrario, desde un
trastorno $\tau$ de $\intint1n$ y el valor $j$, se recupera
$\sigma$ configurando $\sigma(n+1) = j$,$\sigma(\tau^{-1}(j)) = n+1$
y $\sigma = \tau$ en otros lugares: una biyección, dando $D_n$
posibilidades. Resumiendo $j$:$D_{n+1} = n(D_n + D_{n-1})$.
Numéricamente: $D_5 = 4(9 + 2) = 44$,$D_6 = 5(44 + 9) = 265$.

\textbf{6.} De la pregunta 5, $D_{n+1} = nD_n + nD_{n-1}$, entonces

\textbf{8.} Revisión de trinomios, por factoriales:
\[
\binom nk \binom kj
= \frac{n!}{k!\,(n-k)!} \cdot \frac{k!}{j!\,(k-j)!}
= \frac{n!}{j!\,(n-j)!} \cdot \frac{(n-j)!}{(k-j)!\,(n-k)!}
= \binom nj \binom{n-j}{k-j} .
\]
Ahora sustituye $a_k = \sum_j \binom kj b_j$ e intercambia los dos.
\omterm{def:b1:counting:card}{finito} sumas:
\[
\sum_{k=0}^{n} (-1)^{n-k} \binom nk a_k
= \sum_{j=0}^{n} b_j \binom nj
\sum_{k=j}^{n} (-1)^{n-k} \binom{n-j}{k-j}
= \sum_{j=0}^{n} b_j \binom nj
\sum_{i=0}^{n-j} (-1)^{(n-j)-i} \binom{n-j}{i} .
\]
La suma interna es la expansión de $(1 + (-1))^{n-j} = 0^{n-j}$
(teorema del binomio, \cref{thm:b1:counting:binomial}): desaparece
para $j < n$ y es igual a$1$ para $j = n$. Sólo sobrevive $j = n$, y
el lado derecho es $b_n$, como se afirma.
