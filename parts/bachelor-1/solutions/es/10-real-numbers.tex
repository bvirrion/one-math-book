\begin{solution}{exo:b1:reals:1}
$A$: cada elemento es $\leq 1$ y $1 \in A$:$\sup A = \max A = 1$.
Límites inferiores: $0$ límites inferiores; para $\varepsilon > 0$, Arquímedes
proporciona $n$ con $\frac 1n < \varepsilon $, por lo que no hay un número positivo
límites $A$a continuación:$\inf A = 0$, no alcanzado (sin mínimo).

$B$: términos $0, -\frac12, \frac23, -\frac34, \frac45, \dots $ El par
los términos $\frac{n}{n+1}$($n$ par) aumentan a$1$ sin llegar a él;
los términos impares $-\frac{n}{n+1}$ disminuyen a$-1$. Entonces $\sup B = 1$ y
$\inf B = -1$, ninguno alcanzado: ni máximo, ni mínimo. (Límites:$\abs{b}
< 1$ para todos los $b \in B $; y $\frac{n}{n+1} = 1 - \frac{1}{n+1} > 1 -
\varepsilon$ para $n $ grandes, de manera similar a continuación).

$C = \intoo{-\sqrt 3}{\sqrt 3}$:$\sup C = \sqrt 3$,$\inf C =
-\sqrt 3$, ninguno alcanzado.
\end{solution}

\begin{solution}{exo:b1:reals:2}
Escribe $x = \lfloor x \rfloor + u$,$ y = \lfloor y \rfloor + v$ con
$u, v \in \intco{0}{1}$. Luego $x + y = \lfloor x \rfloor + \lfloor y
\rfloor + (u + v)$ con $ u + v \in \intco{0}{2}$. Si$u + v < 1$,
$\lfloor x + y\rfloor = \lfloor x\rfloor + \lfloor y \rfloor$; si
$1 \leq u + v < 2$,$\lfloor x+y \rfloor = \lfloor x \rfloor +
\lfloor y \rfloor + 1$. Se dan ambos casos:$(x, y) = (0.2,\, 0.3)$
da la igualdad izquierda, $(0.7,\, 0.8)$ la derecha.
\end{solution}

\begin{solution}{exo:b1:reals:3}
Sea $k = \lfloor x \rfloor$, entonces $k \leq x < k + 1$. Multiplicando por
$n$:$nk \leq nx < nk + n$, y tomando \omterm{thm:b1:reals:floor}{pisos} (un aumento
operación del lado de los números enteros): $nk \leq \lfloor nx \rfloor \leq nk + n
- 1$. Dividiendo por $n $:$k \leq \frac{\lfloor nx \rfloor}{n} < k + 1$,
entonces el \omterm{thm:b1:reals:floor}{piso} externo es $k$.
\end{solution}

\begin{solution}{exo:b1:reals:4}
Cada elemento de $A$ está en $B$, por lo que $\sup B$ limita a$A$ arriba:
por lo tanto $\sup A \leq \sup B$($\sup A$ es el \emph{el menos} \omterm{def:b1:reals:bounds}{superior
atado}). Simétricamente $\inf B \leq \inf A$. Finalmente $\inf A \leq
\sup A $porque $A $ no está vacío: cualquier $a \in A $ se encuentra entre ellos.
\end{solution}

\begin{solution}{exo:b1:reals:5}
Sea $s = \sup A$,$t = \sup B$. Cada $a + b \leq s + t$: \omterm{def:b1:reals:bounds}{superior
atado}. Para $\varepsilon > 0$, elija $a > s - \frac\varepsilon2$ y
$b > t - \frac\varepsilon2$(\cref{prop:b1:reals:epsilon}): luego $a +
b > s + t - \varepsilon $. Según la caracterización $\varepsilon$,
$\sup(A+B) = s + t$.

Para $-A$:$m$ limita$-A$ arriba $\iff$ $-m$ limita $A$ abajo; el
mínimo \omterm{def:b1:reals:bounds}{límite superior} de $-A$ corresponde por lo tanto al mayor inferior
cota de $A$:$\sup(-A) = -\inf A$.
\end{solution}

\begin{solution}{exo:b1:reals:6}
Por cada $x$:$f(x) + g(x) \leq \sup f + \sup g$; tomando la cena de
el lado izquierdo da la desigualdad. Ejemplo estricto: $E = \{0, 1\}$,
$f = \mathbf{1}_{\{0\}}$(valor $1$ en $0$, en caso contrario $0$),$g =
\mathbf{1}_{\{1\}}$:$\sup(f + g) = 1 < 2 = \sup f + \sup g $.

No hay contradicción con \cref{exo:b1:reals:5}: allí, $a \in A$ y $b
\in B $varían \emph{independientemente}; aquí el mismo $x $alimenta a ambos $f $
y $g$--- el \omterm{def:b1:logic:sets}{conjunto}$\{f(x) + g(x) : x \in E\}$ es más pequeño que el
establezca $\{f(x) + g(y) : x, y \in E\}$.
\end{solution}

\begin{solution}{exo:b1:reals:7}

\end{solution}

\begin{solution}{exo:b1:reals:9}
\begin{enumerate}
 \item Supongamos $\alpha > 0$. Primero,$\alpha \in G$. Supongamos que no: por
 la caracterización $\varepsilon$ del \omterm{def:b1:reals:bounds}{ínfimo} con
 $\varepsilon = \alpha$, hay $g \in G$ con $\alpha < g <
 2\alpha $(estricto a la izquierda desde $\alpha \notin G $); entonces,
 con $\varepsilon = g - \alpha$, hay $h \in G$ con
 $\alpha < h < g$. Ahora $g - h \in G$ y $0 < g - h < g -
 \alpha < \alpha $: un elemento de $G \cap \intoo{0}{+\infty}$
 debajo de su \omterm{def:b1:reals:bounds}{ínfimo}, absurdo. Entonces $\alpha \in G$ y $\alpha\Z
 \subseteq G $($G$ es \omterm{def:b1:structures:group}{grupo}). Por el contrario, para $x \in G$, sea
 $k = \lfloor x/\alpha \rfloor$: luego $x - k\alpha \in G$ y
 $0 \leq x - k\alpha < \alpha$ y la definición de $\alpha$
 fuerzas $x - k\alpha = 0$. Por lo tanto $G = \alpha\Z$.
 \item Supongamos $\alpha = 0$ y dejemos que $x < y $. Hay $g \in G$
 con $0 < g < y - x$. El múltiple$ kg $ con $k = \lfloor x/g
 \rfloor + 1$ satisface $ x < kg \leq x + g < y $ y $ kg \in
 G $: densidad.
\end{enumerate}
$G = \Z + \sqrt 2\,\Z$ es un \omterm{def:b1:structures:subgroup}{subgrupo} de $(\R, +)$. no es del
formulario $\alpha\Z$: en caso contrario $1 = p\alpha $ y $\sqrt 2 = q\alpha $
($p, q \in \Z$) daría $\sqrt 2 = \frac qp \in \Q$,
contradicción. Por dicotomía, $G$ es denso en $\R$.
\end{solution}

\begin{solution}{exo:b1:reals:10}
Sea $s = \sup A > 0$,$t = \sup B > 0$. Para $a \in A$,$b \in B$:$ab
\leq st $(multiplicando desigualdades entre números \emph{positivo}).
Para $0 < \varepsilon < \min(s, t)$: elija $a > s - \varepsilon $ y
$b > t - \varepsilon$; entonces
\[
ab > (s - \varepsilon)(t - \varepsilon)
= st - \varepsilon(s + t) + \varepsilon^2
> st - \varepsilon (s + t),
\]
y $\varepsilon(s+t)$ se pueden hacer arbitrariamente pequeños: por el
$\varepsilon$-caracterización (en la forma: sin número$< st $ límites
$AB$ arriba),$\sup AB = st $.

La positividad es fundamental: $A = B = \{-1, 0\}$ da $AB = \{0, 1\}$,
$\sup AB = 1$, mientras que $\sup A \cdot \sup B = 0 \times 0 = 0$.
\end{solution}

\begin{solution}{exo:b1:reals:11}
Escriba $s = \sup A$,$i = \inf A$. Para $a, a' \in A$:$a \leq s$
y $a' \geq i$ dan $a - a' \leq s - i$; por simetría $\abs{a -
a'} \leq s - i$, por lo que $s - i $ limita el \omterm{def:b1:logic:sets}{conjunto} de los espacios anteriores. Para
$\varepsilon > 0$, elija $a > s - \frac\varepsilon2$ y $a' < i
+ \frac\varepsilon2$(\cref{prop:b1:reals:epsilon} y su espejo
para \omterm{def:b1:reals:bounds}{ínfimo}): luego $\abs{a - a'} \geq a - a' > s - i -
\varepsilon$. Según la caracterización $\varepsilon$,
$\operatorname{diam} A = s - i$.

Cada $a \in A$ satisface $i \leq a \leq s$, por lo que $A \subseteq
\intcc{i}{s}$, un \omterm{prop:b1:reals:intervals}{intervalo} cerrado de longitud $\operatorname{diam}
A$. Si un \omterm{prop:b1:reals:intervals}{intervalo}$\intcc{u}{v}$ cerrado contiene $A $, entonces $v $ es
un \omterm{def:b1:reals:bounds}{límite superior} y $u$ un límite inferior de $A$, por lo que $u \leq i$ y $v
\geq s $:$\intcc{i}{s} \subseteq \intcc{u}{v}$. Por lo tanto
$\intcc{i}{s}$ es el más pequeño.
\end{solution}

\begin{solution}{exo:b1:reals:12}
$0 \in E$, entonces $E \neq \emptyset $. Si$x > \max(1, y)$ entonces $x^2 >
x > y $, entonces $E $ está limitado arriba por $\max(1, y)$:$s = \sup E $
existe (\cref{thm:b1:reals:sup}) y $s \geq \min(1, y) > 0$
porque $\min(1, y) \in E$: efectivamente si $ y \geq 1$ entonces $1^2 = 1
\leq y $, y si $ y < 1$ entonces $ y^2 < y $.

\emph{$s^2 < y $ es imposible.} Elija $0 < h < 1$ con $h <
\frac{y - s^2}{2s + 1}$. entonces
\[
(s + h)^2 = s^2 + 2sh + h^2 \leq s^2 + (2s + 1)h < y ,
\]
entonces $s + h \in E$, lo que contradice que $s$ limita a$E$ arriba.

\emph{$s^2 > y $ es imposible.} Elija $0 < h < s$ con $h <
\frac{s^2 - y}{2s}$. Entonces $(s - h)^2 = s^2 - 2sh + h^2 > s^2 -
2sh > y $; cada $x \in E $ satisface $x^2 \leq y < (s - h)^2$,
por lo tanto $x < s - h $(ambos son $\geq 0$):$s - h$ es un \omterm{def:b1:reals:bounds}{límite superior}
de $E$ más pequeño que $s$, lo que contradice la mínima.

Por lo tanto $s^2 = y $. Unicidad: si $0 < s < s'$ entonces $s^2 <
s'^2$, entonces dos raíces positivas distintas no pueden cuadrar ambas con $ y $.
Monotonicidad: si $0 < y < y'$, entonces $\sqrt y \neq \sqrt{y'}$, y
$\sqrt y > \sqrt{y'}$ daría$ y > y'$elevando al cuadrado: entonces $\sqrt
y < \sqrt{y'}$.
\end{solution}

\begin{solution}{pb:b1:reals:1}
\textbf{1.} Para $\frac 18$:$10 = 8 \cdot 1 + 2$,$20 = 8 \cdot 2
+ 4$,$40 = 8 \cdot 5 + 0$; dígitos $1, 2, 5$, resto $0$, luego
solo ceros: $\frac 18 = 0.125$. Para $\frac 17$:$10 = 7 \cdot 1 +
3$,$30 = 7 \cdot 4 + 2$,$20 = 7 \cdot 2 + 6$,$60 = 7 \cdot 8 +
4$,$40 = 7 \cdot 5 + 5$,$50 = 7 \cdot 7 + 1$: dígitos $1, 4, 2,
8, 5, 7$, restos $3, 2, 6, 4, 5, 1$. El resto tiene
regresó a $r = 1$, por lo que los seis pasos se repiten palabra por palabra para siempre:
$\frac 17 = 0.(142857)$, los restos pasan por $1, 3, 2, 6,
4, 5$.

\textbf{2.} $\frac 13$ en base $2$($r_0 = 1$):$2 = 3 \cdot 0 +
2$,$4 = 3 \cdot 1 + 1$ y $ r = 1$ recurrentes:$\frac 13 =
(0.\overline{01})_2$.$\frac{5}{16}$ en base $2$:$10 = 16 \cdot
0 + 10$,$20 = 16 \cdot 1 + 4$,$8 = 16 \cdot 0 + 8$,$16 = 16
\cdot 1 + 0$:$\frac{5}{16} = (0.0101)_2$, terminando.$\frac 12$
en base $3$:$3 = 2 \cdot 1 + 1$ y $r = 1$ se repite a la vez:
$\frac 12 = (0.\overline{1})_3$.

\textbf{3.} El resto del algoritmo después de los pasos $N$ es $r_N =
b^N p \bmod q $(probado formalmente en la pregunta 14; aquí está el
observación de que cada paso multiplica el resto por $b$ y
reduce el mod $q$). Todos los dígitos posteriores son $0$ si y sólo si algunos
$r_N = 0$, es decir $q \mid b^N p$; desde $\gcd(p, q) = 1$, Gauss
lema da $q \mid b^N$. Si$q \mid b^N$, cada factor \omterm{def:b1:arith:prime}{primo} de
$q$\omterm{def:b1:arith:divides}{divide}$b^N$, por lo tanto \omterm{def:b1:arith:divides}{divide}$b $(primalidad). Por el contrario, si
cada \omterm{def:b1:arith:prime}{primo} de $q = p_1^{a_1} \cdots p_r^{a_r}$\omterm{def:b1:arith:divides}{divide}$b$, entonces
con $A = \max_i a_i$ cada $p_i^{a_i}$\omterm{def:b1:arith:divides}{divide}$b^A$, y el
$p_i^{a_i}$ son \omterm{cor:b1:arith:bezout}{coprimo} en pares, por lo que $q \mid b^A$. Para $q = 20 =
2^2 \cdot 5$: ambos \omterm{def:b1:arith:prime}{primos} dividen $10$($\frac{1}{20} = 0.05$),
pero $2 \nmid 3$, entonces $\frac{1}{20}$ se repite para siempre en la base $3$.

\textbf{4.} $1^2 = 1 < 2 < 4 = 2^2$ da $s_0 = 1$. Entonces $1.4^2
= 1.96 < 2 < 2.25 = 1.5^2$:$\lfloor 10\sqrt 2 \rfloor = 14$,
$s_1 = 1.4$. Siguiente $1.41^2 = 1.9881 < 2 < 2.0164 = 1.42^2$:$s_2 =
1.41$;$1.414^2 = 1.999396 < 2 < 2.002225 = 1.415^2$:$s_3 =
1.414$;$1.4142^2 = 1.99996164 < 2 < 2.00024449 = 1.4143^2$:$s_4
= 1.4142$. En cada caso, las desigualdades mostradas dicen exactamente
$s_n \leq \sqrt 2 < s_n + 10^{-n}$, que es la definición de
el \omterm{thm:b1:reals:floor}{piso} de $10^n \sqrt 2$.

\textbf{5.} $A_0 = \lfloor x \rfloor = 0$ desde $0 \leq x < 1$.
De $A_{n-1} \leq b^{n-1} x < A_{n-1} + 1$, multiplicar por $b$:
\[
b\,A_{n-1} \leq b^n x < b\,A_{n-1} + b .
\]
El número entero $b\,A_{n-1}$ es $\leq b^n x$, por lo que $b\,A_{n-1} \leq
A_n $; y $b^n x < b\,A_{n-1} + b $ con $b\,A_{n-1} + b $ y
fuerzas enteras $A_n \leq b\,A_{n-1} + b - 1$. Por lo tanto $0 \leq d_n =
A_n - b\,A_{n-1} \leq b - 1$: un dígito.

\textbf{6.} Telescópico: $d_k b^{-k} = A_k b^{-k} - A_{k-1}
b^{-(k-1)}$, entonces
\[
\sum_{k=1}^n d_k b^{-k} = A_n b^{-n} - A_0 = s_n .
\]
Al dividir $A_n \leq b^n x < A_n + 1$ por $b^n$ se obtiene $s_n \leq x <
s_n + b^{-n}$.

\textbf{7.} Inducción: $b^0 = 1 \geq 1$ y $b^{n+1} = b \cdot
b^n \geq 2(n + 1) \geq n + 2$. Monotonicidad:$s_n - s_{n-1} = d_n
b^{-n} \geq 0$. Cada $s_n \leq x $(pregunta 6):$x $ es un \omterm{def:b1:reals:bounds}{superior
atado} de $\{s_n\}$. Para $\varepsilon > 0$, \omterm{thm:b1:reals:archimedes}{Arquímedes
propiedad} proporciona $n$ con $n + 1 > \frac1\varepsilon $, por lo tanto
$b^{-n} < \varepsilon$, y luego $s_n > x - b^{-n} > x -
\varepsilon $ por la pregunta 6. Por \cref{prop:b1:reals:epsilon},$x =
\sup_n s_n $.

\textbf{8.} Supongamos $d_k = b - 1$ para todos los $k > N$. Para $n > N$,
la suma geométrica finita da
\[
s_n = s_N + (b - 1)\sum_{k=N+1}^{n} b^{-k}
= s_N + b^{-N} - b^{-n} .
\]
Entonces $x \geq s_n = s_N + b^{-N} - b^{-n}$ por cada $n$; dejando
el último término se reduce por debajo de cualquier $\varepsilon $(pregunta 7),$x
\geq s_N + b^{-N}$. Pero la pregunta 6 en el rango $N $dice$ x < s_N +
b^{-N}$: contradicción. La cadena $(d_n)$ es correcta.

\textbf{9.} Acotado: $t_n \leq (b-1)\sum_{k=1}^n b^{-k} = 1 -
b^{-n} < 1$ y $(t_n)$ no es decreciente, por lo que $ y = \sup t_n $
existe con $0 \leq y \leq 1$. Reparar $n$. Para los dos lados
estimación: $t_n \leq y $ está claro. Por propiedad elija $m > n$ con
$e_m \leq b - 2$. Para $p \geq m$:
\[
t_p - t_n = \sum_{k=n+1}^{p} e_k b^{-k}
\leq (b^{-n} - b^{-p}) - b^{-m} < b^{-n} - b^{-m},
\]
la suma intermedia pierde al menos $b^{-m}$ contra todos-$(b-1)$
máximo; también para $p \leq m$,$t_p \leq t_m \leq t_n + b^{-n} -
b^{-m}$(monotonicidad más el caso $p = m $). Por lo tanto cada
$t_p \leq t_n + b^{-n} - b^{-m}$, entonces $ y \leq t_n + b^{-n} -
b^{-m} < t_n + b^{-n}$. (Con $n = 0$:$ y < 1$, entonces $ y \in
\intco{0}{1}$.) Ahora $b^n t_n = \sum_{k \leq n} e_k b^{n-k}$ es un
entero, y $b^n t_n \leq b^n y < b^n t_n + 1$: entonces $\lfloor b^n
y \rfloor = b^n t_n $. Finalmente los dígitos de $ y $:$d_n(y) = b^n
t_n - b \cdot b^{n-1} t_{n-1} = b^n(t_n - t_{n-1}) = e_n $.

\textbf{10.} La pregunta 9 dice: (valor de la cadena) tiene dígitos (el
cadena); las preguntas 5 a 8 dicen: (dígitos de $x$) forman una cadena adecuada
cuyos truncamientos tienen \omterm{def:b1:reals:bounds}{supremo} $x $(pregunta 7). Entonces los dos \omterm{def:b1:logic:map}{aplicaciones}
componen a la identidad en ambos órdenes: son mutuamente inversos
biyecciones entre $\intco{0}{1}$ y las cadenas adecuadas. si dos
las cadenas adecuadas tenían el mismo valor, aplicar el dígito \omterm{def:b1:logic:map}{aplicación} haría
ellos iguales: unicidad. Este es el \emph{$b$-expansión ádica
teorema}.

\textbf{11.} Sea $e_n = b - 1$ para $n > M$,$M \geq 0$ mínimo.
Como en la pregunta 8, $t_n = t_M + b^{-M} - b^{-n}$ para $n \geq M$,
entonces el valor es $\sup t_n = t_M + b^{-M}$. Si$M = 0$ el valor
es $0 + 1 = 1$: en base $10$,$0.999\dots = 1$ exactamente --- no
aproximadamente. Si $M \geq 1$, la minimalidad da $e_M \leq b - 2$,
y el valor es
\[
t_M + b^{-M} = \frac{b^M t_M + 1}{b^M} \in \intoo{0}{1},
\]
una fracción $b$-ádica, cuya expansión \emph{adecuado} es $e_1 \dots
e_{M-1}\,(e_M + 1)\,000\dots $(una cadena terminal es adecuada,
y su valor es el mismo número). Por el contrario, un real con dos
Las representaciones deben tener una impropia (la propiedad fija la
representación, pregunta 10), por lo tanto debe tener esta forma. y cada uno
$m/b^N \in \intoo{0}{1}$, escrito con el último dígito distinto de cero $d_N$,
tiene el gemelo impropio $d_1 \dots d_{N-1}(d_N -
1)(b-1)(b-1)\dots $: exactamente las fracciones ádicas $b $ llevan dos
nombres, todos los demás reales.

\textbf{12.} Digamos que las cadenas coinciden hasta $m - 1$, con común
truncamiento $P = s_{m-1}$ y $d_m < e_m$. Por la pregunta 9 (estricta
estimación superior en el rango $m$),$x < P + d_m b^{-m} + b^{-m} = P +
(d_m + 1)b^{-m} \leq P + e_m b^{-m} \leq y $, el último paso
porque $P + e_m b^{-m}$ es el truncamiento de $ y $ $t_m \leq y $. entonces
$d_m < e_m \implies x < y $; intercambiando roles,$e_m < d_m \implies
y < x $; y dado que las cadenas difieren en $m $, una de las dos
sostiene. Siguen ambas direcciones.

\textbf{13.} Dejemos $z = bx - A_1 \in \intco{0}{1}$(de hecho $A_1
\leq bx < A_1 + 1$). Para $n \geq 0$:$b^n z = b^{n+1} x - b^n
A_1$ con $ b^n A_1 \in \Z $, entonces por $\lfloor u - K \rfloor =
\lfloor u \rfloor - K $($K$ entero),
\[
A_n(z) = A_{n+1}(x) - b^n A_1(x) .
\]
Por lo tanto $d_n(z) = A_n(z) - b\,A_{n-1}(z) = A_{n+1} - b^n A_1 -
b\,A_n + b^n A_1 = d_{n+1}(x)$. Entonces la parte fraccionaria de $bx $
lleva los dígitos desplazados; iterando $m$ veces, el fraccionario
parte de $b^m x$ tiene dígitos $(d_{n+m})_{n \geq 1}$.

\textbf{14.} División euclidiana: $b^n p = q\,Q_n + r_n$ con $0
\leq r_n < q $. Dividir por $q $:$b^n x = Q_n + \frac{r_n}{q}$ con
$0 \leq \frac{r_n}{q} < 1$, entonces $Q_n = \lfloor b^n x \rfloor =
A_n $, dando $A_n = \frac{b^n p - r_n}{q}$. Para la recurrencia:
$b^n p = b(q\,A_{n-1} + r_{n-1}) = q\,(b\,A_{n-1}) +
b\,r_{n-1}$, por lo que $b^n p $ y $b\,r_{n-1}$ difieren en un múltiplo de
$q$:$r_n = (b\,r_{n-1}) \bmod q$.

\textbf{15.} Divide $b\,r_{n-1}$ por $q$:$b\,r_{n-1} = q\,c +
r_n $ con $c = \lfloor b\,r_{n-1}/q \rfloor $. Sustituyendo en
la visualización de la pregunta 14: $b^n p = q(b\,A_{n-1} + c) + r_n$,
y la unicidad de la división euclidiana identifica $A_n = b\,A_{n-1}
+ c $, es decir $d_n = c = \lfloor b\,r_{n-1}/q \rfloor $. dígito
$n$ depende sólo de $r_{n-1}$ --- el bucle de división largo de la Parte
Yo, ahora certificado.

\textbf{16.} Los restos $q + 1$ $r_0, \dots, r_q$ toman valores
en el elemento $q$\omterm{def:b1:logic:sets}{conjunto}$\intint{0}{q-1}$: junto al casillero
