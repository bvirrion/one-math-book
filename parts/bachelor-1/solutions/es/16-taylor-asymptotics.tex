\begin{solution}{exo:b1:taylor:4}
Taylor--Lagrange (\cref{thm:b1:taylor:lagrange}) para $\exp$ en $a =
0$,$x = 1$: el $(n+1)$-st \omterm{def:b1:derivative:def}{derivado} es $\eu^t \leq \eu < 3$ en
$\intcc{0}{1}$, entonces
\[
\Bigl|\eu - \sum_{k=0}^{n} \frac{1}{k!}\Bigr| \leq
\frac{3}{(n+1)!} .
\]
Para $6$ decimales exactos, desea $\frac{3}{(n+1)!} < 5\times 10^{-7}$,
es decir $(n+1)! > 6\times 10^{6}$: desde $10! = 3\,628\,800$ y $11! =
39\,916\,800$,$n + 1 = 11$, es decir $n = 10$ es suficiente.
\end{solution}

\begin{solution}{exo:b1:taylor:5}
$n\ln\bigl(1 + \frac1n\bigr) = n\Bigl(\frac1n - \frac{1}{2n^2} +
\frac{1}{3n^3} + o\bigl(\frac{1}{n^3}\bigr)\Bigr) = 1 - \frac{1}{2n}
+ \frac{1}{3n^2} + o\bigl(\frac{1}{n^2}\bigr)$. Exponenciando, con
$u = -\frac{1}{2n} + \frac{1}{3n^2}$ y $\eu^u = 1 + u + \frac{u^2}2
+ o(u^2)$:
\[
\Bigl(1 + \frac1n\Bigr)^n = \eu\cdot \eu^{u}
= \eu\Bigl(1 - \frac{1}{2n} + \frac{1}{3n^2} + \frac{1}{8n^2}
+ o\Bigl(\frac{1}{n^2}\Bigr)\Bigr)
= \eu\Bigl(1 - \frac{1}{2n} + \frac{11}{24n^2} +
o\Bigl(\frac{1}{n^2}\Bigr)\Bigr).
\]
Límite $\eu$; el error es $\sim \dfrac{\eu}{2n}$: lento (un dígito por
aumento diez veces mayor que $n$).
\end{solution}

\begin{solution}{exo:b1:taylor:9}
Taylor--Lagrange en el orden $1$ alrededor de $x$, en ambos lados:
\[
f(x + h) = f(x) + h f'(x) + R_+,\quad
f(x - h) = f(x) - h f'(x) + R_-,
\qquad \abs{R_\pm} \leq \frac{h^2}{2} \sup \abs{f''} .
\]
Restando: $f(x+h) - f(x-h) = 2h f'(x) + (R_+ - R_-)$, entonces
\[
\abs{f'(x)} \leq \frac{\abs{f(x+h) - f(x-h)}}{2h} +
\frac{h}{2}\sup_{\intcc{x-h}{x+h}}\abs{f''} .
\]
Con límites globales: $\abs{f'(x)} \leq \frac{M_0}{h} + \frac{M_2
h}{2}$ por cada $h > 0$. El lado derecho se minimiza en $h =
\sqrt{2M_0/M_2}$(\omterm{def:b1:derivative:def}{derivado} cero), con valor $\sqrt{2M_0M_2}$
--- por lo tanto $\abs{f'} \leq \sqrt{2M_0M_2}$. (Si$M_2 = 0$, sea $h \to
\infty $:$f' = 0$, consistente).
\end{solution}

\begin{solution}{exo:b1:taylor:10}
En $\intoo{0}{\pi}$:$0 < \sin u < u$, por lo que $(u_n)$ es estrictamente
decreciente, positivo, por tanto convergente; el límite es un punto fijo
de $\sin $ en $\intcc{0}{\pi}$, y $\sin \ell = \ell $fuerza a$\ell =
0$(desde $\sin x < x $ para $x > 0$).
\begin{enumerate}
 \item Usando $\sin u = u - \frac{u^3}{6} + o(u^3)$ como $u \to 0$:
 \[
 v_{n+1} - v_n = \frac{1}{\sin^2 u_n} - \frac{1}{u_n^2}
 = \frac{1}{u_n^2}\Bigl(\Bigl(1 - \frac{u_n^2}{6} +
 o(u_n^2)\Bigr)^{\!-2} - 1\Bigr)
 = \frac{1}{u_n^2}\Bigl(\frac{u_n^2}{3} + o(u_n^2)\Bigr)
 \longrightarrow \frac13 .
 \]
 \item Por \cref{exo:b1:seq:10}~(3) (Cesàro para diferencias),
 $\frac{v_n}{n} \to \frac13$, es decir $v_n \sim \frac n3$, es decir $u_n^2 \sim \frac 3n$: desde $u_n > 0$,
 \[
 u_n \sim \sqrt{\frac{3}{n}} .
 \]
\end{enumerate}
\end{solution}

\begin{solution}{exo:b1:taylor:12}
El $I_k = \intoo{k\pi - \frac\pi2}{k\pi + \frac\pi2}$, el
la función $g(x) = \tan x - x$ tiene \omterm{def:b1:derivative:def}{derivado}$\tan^2 x \geq 0$,
desapareciendo sólo en el único punto $k\pi$: $g$ es estrictamente
aumentando en $I_k $(\cref{cor:b1:derivative:monotone}~(2)),
con límites $-\infty$ y $+\infty$ en los extremos: exactamente uno
cero $x_k$. Para $k \geq 1$,$g(k\pi) = -k\pi < 0$, entonces $x_k \in
\intoo{k\pi}{k\pi + \frac\pi2}$: escriba $x_k = k\pi + \frac\pi2 -
\varepsilon_k $ con $\varepsilon_k \in \intoo{0}{\frac\pi2}$.
entonces
\[
x_k = \tan x_k = \tan\Bigl(\frac\pi2 - \varepsilon_k\Bigr)
= \frac{1}{\tan\varepsilon_k}
\quad\Longrightarrow\quad
\varepsilon_k = \arctan\frac{1}{x_k} ,
\]
utilizando $\tan\varepsilon_k = \frac{1}{x_k}$ y $\varepsilon_k \in
\intoo{0}{\frac\pi2}$. Desde $x_k \geq k\pi \to \infty $:
$\varepsilon_k \to 0$, y
\[
\varepsilon_k = \arctan\frac{1}{x_k} = \frac{1}{x_k} +
O\Bigl(\frac{1}{x_k^3}\Bigr)
= \frac{1}{k\pi + O(1)} + O\Bigl(\frac{1}{k^3}\Bigr)
= \frac{1}{k\pi} + O\Bigl(\frac{1}{k^2}\Bigr) ,
\]
de donde $x_k = k\pi + \frac\pi2 - \frac{1}{k\pi} +
o\bigl(\frac1k\bigr)$.
\end{solution}

\begin{solution}{pb:b1:taylor:1}
\textbf{1.} $S_{2n+1} - S_{2n-1} = a_{2n} - a_{2n+1} \geq 0$ y
$S_{2n+2} - S_{2n} = a_{2n+2} - a_{2n+1} \leq 0$, mientras que $S_{2n}
- S_{2n+1} = a_{2n+1} \to 0$: las secuencias $(S_{2n+1})$,
$(S_{2n})$ son adyacentes, convergiendo en un $S$ común
(\cref{thm:b1:seq:adjacent}) con $S_{2n+1} \leq S \leq S_{2n}$.
Incluso para $n$:$S_{n+1} \leq S \leq S_n$ da $-a_{n+1} \leq S -
S_n \leq 0$; para $n $impar:$0 \leq S - S_n \leq a_{n+1}$. en ambos
los casos $\abs{S - S_n} \leq a_{n+1}$ y $S - S_n$ tiene el signo de
$(-1)^{n+1}$, el primer término omitido. Una disminución estricta hace
cada uno muestra desigualdad estricta, en particular $0 < \abs{S -
S_n} < a_{n+1}$.

\textbf{2.} $a_k = \frac{1}{k!}$ disminuye estrictamente a$0$:
Se aplica la pregunta 1. Taylor--Lagrange
(\cref{thm:b1:taylor:lagrange}) para $\exp$ entre $-1$ y $0$:
$\abs{\eu^{-1} - T_n} \leq \frac{1}{(n+1)!}$(el \omterm{def:b1:derivative:def}{derivado}
$\eu^t$ es $\leq 1$ allí), por lo que $T_n \to \eu^{-1}$ y el límite
$S$ de la pregunta 1 \emph{is}$\eu^{-1}$, con los límites estrictos
$0 < \abs{\eu^{-1} - T_n} < \frac{1}{(n+1)!}$.

\textbf{3.} Integrando la identidad sobre $\intcc{0}{1}$: el
El lado izquierdo es $\arctan 1 = \frac\pi4$(teorema fundamental), el
$k$-ésimo término da $\frac{(-1)^k}{2k+1}$, y
\[
\rho_n = (-1)^{n+1}\int_0^1 \frac{t^{2n+2}}{1+t^2}\dd t,
\qquad
\abs{\rho_n} \leq \int_0^1 t^{2n+2}\dd t = \frac{1}{2n+3} .
\]

\textbf{4.} El error en $\pi$ es $4\abs{\rho_n} \leq
\frac{4}{2n+3}$: a continuación $10^{-6}$ requiere $2n + 3 > 4\cdot10^6$,
unos dos millones de términos. Mientras tanto $4S_4 = 4\bigl(1 - \frac13 +
\frac15 - \frac17 + \frac19\bigr) = 4 \times 0.834921 =
3.339683$, a casi $0.2$ de $\pi$: cinco términos, ni siquiera
un dígito.

\textbf{5.} Integración sobre $\intcc{0}{1}$:$\ln 2 =
\sum_{k=1}^{n} \frac{(-1)^{k-1}}{k} + (-1)^n R_n $ con $ R_n =
\int_0^1 \frac{t^n}{1+t}\dd t $; de $\frac12 \leq \frac{1}{1+t}
\leq 1$:$\frac{1}{2(n+1)} \leq R_n \leq \frac{1}{n+1}$. el
El error está atrapado entre dos múltiplos de $\frac1n$: seis.
los decimales cuestan alrededor de un millón de términos.

\textbf{6.} Integrando de $0$a$x$:$\frac12\ln\frac{1 +
x}{1 - x} = \sum_{k=0}^{n} \frac{x^{2k+1}}{2k+1} + \int_0^x
\frac{t^{2n+2}}{1-t^2}\dd t $. En$x = \frac13$:$\frac{1 +
1/3}{1 - 1/3} = 2$, y en $\intcc{0}{\frac13}$,$\frac{1}{1 -
t^2} \leq \frac98$:
\[
\ln 2 = 2\sum_{k=0}^{n} \frac{(1/3)^{2k+1}}{2k+1} +
\tilde\rho_n,
\qquad
0 < \tilde\rho_n \leq \frac{9}{4}\cdot
\frac{(1/3)^{2n+3}}{2n+3} :
\]
cada término adicional \omterm{def:b1:arith:divides}{divide} el error en aproximadamente $9$.

\textbf{7.} Inducción: para $n = 1$:$1 - \frac12 = \frac12 = H_2
- H_1$. Paso:
\[
H_{2n+2} - H_{n+1} = (H_{2n} - H_n) + \frac{1}{2n+1} +
\frac{1}{2n+2} - \frac{1}{n+1}
= (H_{2n} - H_n) + \frac{1}{2n+1} - \frac{1}{2n+2},

que es exactamente el incremento de la suma alterna. y
$H_{2n} - H_n = \sum_{k=1}^{n}\frac{1}{n+k}$ es la suma de Riemann

\textbf{14.} $a_k = \frac{1}{(2k)!}$ disminuye estrictamente a$0$;
por la pregunta 1 y \cref{prop:b1:taylor:standard} (Lagrange
cota como en la pregunta 2), $\sum_{k \leq n}\frac{(-1)^k}{(2k)!}
\to \cos 1$ con el corchete \emph{estricto}$0 < \bigl|\cos 1 -
S'_n\bigr| < \frac{1}{(2n+2)!}$. Supongamos $\cos 1 = \frac pq $ y
tomar $2n \geq q$: luego $(2n)!\,S'_n = \sum_{k\leq n} (-1)^k
\frac{(2n)!}{(2k)!} \in \Z $ y $(2n)!\,\frac pq \in \Z$, mientras
\[
0 < \Bigl|(2n)!\cos 1 - (2n)!S'_n\Bigr| <
\frac{(2n)!}{(2n+2)!} = \frac{1}{(2n+1)(2n+2)} < 1 :
\]
un número entero distinto de cero de valor absoluto $< 1$. Contradicción:$\cos
1 \notin \Q $.

\textbf{15.} Idénticamente con $a_k = \frac{1}{(2k+1)!}$,
multiplicando por $(2n+1)!$ con $2n + 1 \geq q$:$\sin 1 \notin
\Q $. Sin embargo $\cos^2 1 + \sin^2 1 = 1 \in \Q$: productos y sumas de
los irracionales pueden ser racionales --- la irracionalidad no pasa por ningún
operación algebraica libre.

\textbf{16.} Atado por la cola: para $m > n$,
\[
\sum_{k=n+1}^{m} \frac{1}{(2k)!}
\leq \frac{1}{(2n+2)!}\Bigl(1 + \frac12 + \frac14 +
\dots\Bigr) \leq \frac{2}{(2n+2)!} ,
\]
ya que cada relación sucesiva es $\frac{1}{(2k+1)(2k+2)} \leq
\frac12$; la cola es positiva (su primer término lo es). Entonces $0 <

y multiplicar por $(2n)!$ con $2n \geq q$ atrapa un valor distinto de cero
entero en $\intoo{0}{1}$ nuevamente:$\cosh 1 \notin \Q$.

\textbf{17.} $\cos\frac1m = \sum_k
\frac{(-1)^k}{m^{2k}(2k)!}$: los términos disminuyen estrictamente a

Para $\frac ab$ con $a \geq 2$: los denominadores de compensación se multiplican

El numerador $a^{2n+2}$ ya no se borra y la trampa se atasca.

\textbf{18.} En $t = x$ todos los términos de $g$ desaparecen excepto $f(x)
- f(x) = 0$:$g(x) = 0$; Se elige $A $ para que sea $g(a) = 0$.

\[
g'(t) = -\frac{f^{(n+1)}(t)}{n!}(x - t)^n +
A\,\frac{(x-t)^n}{n!}
= \frac{(x-t)^n}{n!}\bigl(A - f^{(n+1)}(t)\bigr) .

{(n+1)!}x^{n+1} > 0$: el exponencial excede cada uno de sus
