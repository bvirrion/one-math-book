\begin{solution}{exo:b1:linmaps:2}

\begin{solution}{exo:b1:linmaps:3}
\emph{\omterm{def:b1:linmaps:kerim}{Núcleo}:} $P = P'$fuerza a$\deg P = \deg P'$a menos que $P = 0$;
pero $\deg P' < \deg P$ para $P \neq 0$: entonces $\ker u = \{0\}$ y $u$,
un endomorfismo \omterm{def:b1:logic:inj}{inyectivo} del \omterm{def:b1:findim:def}{de dimensión finita} $\R_n[X]$, es un
isomorfismo (\cref{cor:b1:linmaps:samedim}).

\emph{Inverso:} deja que $v(P) = P + P' + P'' + \dots + P^{(n)}$(un
suma finita en $\R_n[X]$). entonces
\[
v\bigl(u(P)\bigr) = \sum_{k=0}^{n} (P - P')^{(k)}
= \sum_{k=0}^{n} P^{(k)} - \sum_{k=0}^{n} P^{(k+1)}
= P - P^{(n+1)} = P ,
\]
telescópico, desde $P^{(n+1)} = 0$. Entonces $v = u^{-1}$.
\end{solution}

\begin{solution}{exo:b1:linmaps:5}

\begin{solution}{exo:b1:linmaps:6}
$(pq)^2 = pqpq = ppqq = pq $(conmutación): a \omterm{def:b1:linmaps:projection}{proyección}
(\cref{thm:b1:linmaps:projchar}).

\omterm{def:b1:linmaps:kerim}{Imagen}: $\operatorname{im}(pq) \subseteq \operatorname{im} p$, y
$= \operatorname{im}(qp) \subseteq \operatorname{im} q$: contenido
en la intersección. Por el contrario, si $x \in \operatorname{im} p \cap
\operatorname{im} q $, entonces $p(x) = x $ y $q(x) = x $(puntos fijos
caracterizan la \omterm{def:b1:linmaps:kerim}{imagen} de un \omterm{def:b1:linmaps:projection}{proyección}), por lo que $pq(x) = x$:$x \in
\operatorname{im}(pq)$.

\omterm{def:b1:linmaps:kerim}{Núcleo}: $\ker p \subseteq \ker(qp) = \ker(pq)$ y también $\ker q
\subseteq \ker(pq)$: la suma está contenida. Por el contrario, deja$ pq(x) =
0$ y escribe
\[
x = \underbrace{q(x)}_{\in\, \ker p} +
\underbrace{(x - q(x))}_{\in\, \ker q} :
\]
el primer término satisface $p(q(x)) = 0$, por lo que se encuentra en $\ker p$; el
el segundo está en $\ker q$ desde $q(x - q(x)) = q(x) - q^2(x) = 0$. Por lo tanto
$x \in \ker p + \ker q$.
\end{solution}

\begin{solution}{exo:b1:linmaps:7}
Tenga en cuenta primero las inclusiones generales $\ker u \subseteq \ker u^2$ y
$\operatorname{im} u^2 \subseteq \operatorname{im} u$, y por
rango--nulidad, (2) $\iff $(3) (igual \omterm{def:b1:linmaps:kerim}{granos}$\iff$ igual \omterm{def:b1:linmaps:rank}{rangos}

(1 $\Rightarrow $2): deja $u^2(x) = 0$; luego $u(x) \in \ker u \cap

\end{solution}

\begin{solution}{exo:b1:linmaps:11}
\omterm{def:b1:reals:bounds}{límite superior}: por cada $x$,$(u + v)(x) = u(x) + v(x) \in
\operatorname{im} u + \operatorname{im} v $, entonces
\[
\operatorname{rk}(u + v)
\leq \dim(\operatorname{im} u + \operatorname{im} v)
\leq \operatorname{rk} u + \operatorname{rk} v
\]
(Grassmann, \cref{thm:b1:findim:grassmann}). Límite inferior: aplicar
el \omterm{def:b1:reals:bounds}{límite superior} al par $(u + v, -v)$, cuya suma es $u$:
\[
\operatorname{rk} u \leq \operatorname{rk}(u + v) +
\operatorname{rk}(-v) = \operatorname{rk}(u + v) +
\operatorname{rk} v,
\]
entonces $\operatorname{rk} u - \operatorname{rk} v \leq
\operatorname{rk}(u+v)$; intercambiar $u $ y $v$ da la absoluta
valor.
\end{solution}

\begin{solution}{exo:b1:linmaps:12}
\emph{Fórmula exacta.} Sea $v'$ la restricción de $v$ al
\omterm{def:b1:vspaces:subspace}{subespacio}$\operatorname{im} w$. Su \omterm{def:b1:linmaps:kerim}{imagen} es $v(w(F)) =
\operatorname{im}(v \circ w)$ y su \omterm{def:b1:linmaps:kerim}{núcleo} es $\ker v \cap
\operatorname{im} w $. Rango: nulidad para $v'$ en el espacio
$\operatorname{im} w$:
\[
\operatorname{rk} w = \dim\operatorname{im} w
= \operatorname{rk}(v \circ w) + \dim(\ker v \cap
\operatorname{im} w) .
\]

\emph{Frobenio.} Aplicar la fórmula exacta dos veces, a $w$ y a
$w \circ u$:
\[
\operatorname{rk} w - \operatorname{rk}(vw)
= \dim\bigl(\ker v \cap \operatorname{im} w\bigr),
\qquad
\operatorname{rk}(wu) - \operatorname{rk}(vwu)
= \dim\bigl(\ker v \cap \operatorname{im}(wu)\bigr) .
\]
Desde $\operatorname{im}(w \circ u) \subseteq \operatorname{im}
w$, la segunda intersección está contenida en la primera, y su
la dimensión no es mayor:
\[
\operatorname{rk}(wu) - \operatorname{rk}(vwu)
\;\leq\; \operatorname{rk} w - \operatorname{rk}(vw) ,
\]
que se reordena en la desigualdad de Frobenius. Con $w =
\mathrm{id}_F $(\omterm{def:b1:linmaps:rank}{rango}$\dim F$ y $\operatorname{im}\,
\mathrm{id}_F = F$):$\operatorname{rk} v + \operatorname{rk} u
\leq \dim F + \operatorname{rk}(vu)$, desigualdad de Sylvester ---
demostrado nuevamente, en forma matricial, en \cref{exo:b1:matrices:10}.
\end{solution}

\begin{solution}{pb:b1:linmaps:1}
\textbf{1.} $(\mathrm{id} - p)^2 = \mathrm{id} - 2p + p^2 =
\mathrm{id} - p $: un proyector. Si$ y = x - p(x)$, entonces $p(y) =
p(x) - p^2(x) = 0$, y viceversa $x \in \ker p $ da $ x =
(\mathrm{id} - p)(x)$:$\operatorname{im}(\mathrm{id} - p) = \ker
p$. Y$(\mathrm{id} - p)(x) = 0 \iff p(x) = x \iff x \in
\operatorname{im} p $(puntos fijos,
\cref{thm:b1:linmaps:projchar}): $\ker(\mathrm{id} - p) =
\operatorname{im} p $.

\textbf{2.}$(\lambda\,\mathrm{id} + \mu p)^2 =
\lambda^2\,\mathrm{id} + (2\lambda\mu + \mu^2)\,p $. la pareja
$(\mathrm{id}, p)$ es \omterm{def:b1:vspaces:free}{libre} en $\mathcal{L}(E)$:$p = c\,
\mathrm{id}$ daría$ c^2 = c $, entonces $p = 0$ o $\mathrm{id}$,
excluido. Coeficientes identificativos, la \omterm{def:b1:logic:map}{aplicación} es un proyector si y así
$\lambda^2 = \lambda$ y $2\lambda\mu + \mu^2 = \mu $. Para
$\lambda = 0$:$\mu \in \{0, 1\}$. Para $\lambda = 1$:$\mu^2 +
\mu = 0$,$\mu \in \{0, -1\}$. Exactamente cuatro proyectores en el
plano: $0$,$p$,$\mathrm{id}$,$\mathrm{id} - p$.

\textbf{3.} $(\lambda\,\mathrm{id} + \mu p)(\lambda'\,\mathrm{id}
+ \mu' p) = \lambda\lambda'\,\mathrm{id} + (\lambda\mu' +
\mu\lambda' + \mu\mu')\,p $: el avión está estable bajo
composición. Desde $p^k = p$ por cada $k \geq 1$, por $Q =
\sum_k a_k X^k $:
\[
Q(p) = a_0\,\mathrm{id} + \Bigl(\sum_{k \geq 1} a_k\Bigr) p
= Q(0)\,\mathrm{id} + \bigl(Q(1) - Q(0)\bigr)\,p .
\]

\textbf{4.} En $\operatorname{im} p $(donde $p$ actúa como
identidad), $\lambda\,\mathrm{id} + \mu p$ se multiplica por $\lambda
+ \mu $; en $\ker p $, por $\lambda$. Como $E = \operatorname{im} p
\oplus \ker p $, la \omterm{def:b1:logic:map}{aplicación} es \omterm{def:b1:logic:inj}{biyectivo} si es $\lambda \neq 0$ y
$\lambda + \mu \neq 0$. Resolviendo $\lambda\alpha = 1$,
$\lambda\beta + \mu\alpha + \mu\beta = 0$ en la regla de composición
de la pregunta 3:
\[
(\lambda\,\mathrm{id} + \mu p)^{-1}
= \frac1\lambda\,\mathrm{id} -
\frac{\mu}{\lambda(\lambda + \mu)}\,p ,
\]
cuya acción es por $1/\lambda$ sobre $\ker p$ y $1/(\lambda +
\mu)$ sobre $\operatorname{im} p$, como debe ser.

\textbf{5.} Escribe $F = \operatorname{im} p = \operatorname{im}
p'$. Para cada $x $,$p'(x) \in F$ y $p$ corrige $F$ puntualmente:
$p(p'(x)) = p'(x)$, es decir $p\,p' = p'$; simétricamente $p'\,p =
p $. Cuando dos \omterm{def:b1:linmaps:projection}{proyecciones} comparten su \omterm{def:b1:linmaps:kerim}{imagen}, el aplicado
\emph{primero} decide: su salida ya se encuentra en $F$, donde
\omterm{def:b1:linmaps:projection}{proyección} externo actúa como la identidad y no cambia nada.

\textbf{6.} $(p + q)^2 = p^2 + pq + qp + q^2 = (p + q) + pq +
qp $, por lo que el proyector $p + q $fuerza a$ pq + qp = 0$. componer en el
izquierda con $p$:$pq + pqp = 0$; a la derecha con $p$:$pqp + qp
= 0$. Restando,$pq = qp $; luego $pq + qp = 2pq = 0$ y el
la característica no es $2$:$pq = qp = 0$.

\textbf{7.} Con $pq = qp = 0$, la misma expansión da $(p +
q)^2 = p + q $. \omterm{def:b1:linmaps:kerim}{Imagen}:$\operatorname{im}(p + q) \subseteq
\operatorname{im} p + \operatorname{im} q$ siempre. Por el contrario,
para $x \in \operatorname{im} p$:$q(x) = q(p(x)) = 0$, entonces $(p +
q)(x) = p(x) = x $ y $x \in \operatorname{im}(p+q)$; lo mismo para
$\operatorname{im} q$. Directo:$x \in \operatorname{im} p
\cap \operatorname{im} q $ da $x = p(x) = p(q(x)) = 0$.
\omterm{def:b1:linmaps:kerim}{Núcleo}: si $p(x) + q(x) = 0$, aplicando $p$ da $p(x) +
p(q(x)) = p(x) = 0$, y aplicando $q $ da $q(x) = 0$:$\ker(p
+ q) = \ker p \cap \ker q $(la inclusión inversa es clara).

\textbf{8.} $p - q$ es un proyector si $\mathrm{id} - (p - q) =
(\mathrm{id} - p) + q$ lo es (pregunta 1 dos veces). Por preguntas
6--7 aplicado a los proyectores $\mathrm{id} - p$ y $q$, esto
contiene iff $(\mathrm{id} - p)q = q(\mathrm{id} - p) = 0$, es decir \
sif $pq = q$ y $qp = q$.

\textbf{9.} $pq = q$ significa que $p$ corrige cada $q(x)$, es decir $\operatorname{im} q \subseteq \ker(p - \mathrm{id}) =
\operatorname{im} p$. Y$ qp = q $ significa $ q\bigl((\mathrm{id} -
p)(x)\bigr) = 0$ para todos los $x $, es decir, $q$ desaparece en
$\operatorname{im}(\mathrm{id} - p) = \ker p$:$\ker p \subseteq
\ker q $. Ambos pasos son equivalencias: la orden $q \leq p $ dice
$q$ se proyecta sobre un \omterm{def:b1:linmaps:kerim}{imagen} más pequeño, junto con un \omterm{def:b1:linmaps:kerim}{núcleo} más grande.

\textbf{10.} Expandiendo, $(\mathrm{id} - p)(\mathrm{id} - q) =
\mathrm{id} - p - q + pq = \mathrm{id} - r $. los proyectores
$\mathrm{id} - p$ y $\mathrm{id} - q$ conmutan, por lo que por
\cref{exo:b1:linmaps:6} su producto $\mathrm{id} - r$ es el
proyector en $\operatorname{im}(\mathrm{id} - p) \cap
\operatorname{im}(\mathrm{id} - q) = \ker p \cap \ker q$ junto
$\ker(\mathrm{id} - p) + \ker(\mathrm{id} - q) =
\operatorname{im} p + \operatorname{im} q $. Por la pregunta 1,$r =
\mathrm{id} - (\mathrm{id} - r)$ es entonces el proyector con
$\operatorname{im} r = \operatorname{im} p + \operatorname{im}
q$ y $\ker r = \ker p \cap \ker q $.

\textbf{11.} $p_i$ está bien definido (singularidad del
descomposición) y \omterm{def:b1:linmaps:def}{lineal} (la descomposición de $x + \lambda y $
es la suma de las descomposiciones, nuevamente por unicidad). Para $x_i
\in F_i $ la descomposición es el propio $x_i $, por lo que $p_i(x_i) = x_i $:
$p_i^2 = p_i$ y $p_j(x_i) = 0$ para $j \neq i$:$p_i p_j = 0$
($p_j(x) \in F_j$). Sumando los componentes,$\sum_i p_i =
\mathrm{id}$. Finalmente $\operatorname{im} p_i = F_i$ y $\ker
p_i = \bigoplus_{j \neq i} F_j $.

\textbf{12.} $p_i = p_i \circ \mathrm{id} = p_i\sum_j p_j =
p_i^2 + \sum_{j \neq i} p_i p_j = p_i^2$: cada $p_i $ es un
proyector. Cada $x = \mathrm{id}(x) = \sum_i p_i(x)$ se encuentra en
$\sum_i \operatorname{im} p_i$: la suma \omterm{def:b1:linmaps:kerim}{imágenes} es $E$.
Directo: supongamos $y_1 + \dots + y_k = 0$ con $y_i \in
\operatorname{im} p_i $, entonces $p_i(y_i) = y_i $. Aplicar $p_j $:$p_j
(y_i) = p_j p_i (y_i) = 0$ para $ i \neq j $, es decir, $0 = p_j\bigl(\sum
y_i\bigr) = y_j $, para cada $j $. Por lo tanto $E = \bigoplus_i
\operatorname{im} p_i $.

La identidad ya fuerza la ortogonalidad del par.

\textbf{15.} \emph{Lema.} Por inducción con Grassmann
(\cref{thm:b1:findim:grassmann}):
\[

\]
Si el total es una igualdad, cada paso es: $(F_1 + \dots +
