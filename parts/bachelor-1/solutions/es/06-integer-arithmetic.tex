\begin{solution}{exo:b1:arith:1}
$1001 = 1 \times 777 + 224$;$777 = 3 \times 224 + 105$;$224 = 2
\times 105 + 14$;$105 = 7 \times 14 + 7$;$14 = 2 \times 7 + 0$. entonces
$\gcd(1001, 777) = 7$. al revés :
\[
7 = 105 - 7 \times 14
= 105 - 7(224 - 2\times 105) = 15 \times 105 - 7 \times 224
\]
\[
= 15(777 - 3\times 224) - 7\times 224 = 15 \times 777 - 52 \times 224
= 15 \times 777 - 52(1001 - 777) = 67 \times 777 - 52 \times 1001 .
\]
Verifique: $67 \times 777 = 52\,059$ y $52 \times 1001 = 52\,052$;
diferencia $7$. Par Bézout:$(u, v) = (-52, 67)$ para $1001u + 777v =
7$.
\end{solution}

\begin{solution}{exo:b1:arith:2}
Desde $10 \equiv 1 \pmod 9$:$10^k \equiv 1$, entonces $\sum_k d_k 10^k
\equiv \sum_k d_k \pmod 9$. Dado que $10 \equiv -1 \pmod{11}$:$10^k
\equiv (-1)^k $, el número entero es congruente con la suma alterna
$\sum_k (-1)^k d_k$ mod $11$(a partir del dígito \emph{unidades}
con signo $+$).

$123\,456\,789$: suma de dígitos $45 \equiv 0 \pmod 9$. Suma alterna de
las unidades: $9 - 8 + 7 - 6 + 5 - 4 + 3 - 2 + 1 = 5$, por lo que el número es
$\equiv 5 \pmod{11}$.
\end{solution}

\begin{solution}{exo:b1:arith:3}
Euclides: $237 = 2 \times 91 + 55$;$91 = 1 \times 55 + 36$;$55 = 1
\times 36 + 19$;$36 = 1 \times 19 + 17$;$19 = 1 \times 17 + 2$;$17
= 8 \times 2 + 1$. al revés :
\[
1 = 17 - 8\times 2 = 17 - 8(19 - 17) = 9\times 17 - 8\times 19
= 9(36 - 19) - 8\times 19 = 9\times 36 - 17\times 19
\]
\[
= 9\times 36 - 17(55 - 36) = 26\times 36 - 17\times 55
= 26(91 - 55) - 17\times 55 = 26\times 91 - 43\times 55
\]
\[
= 26\times 91 - 43(237 - 2\times 91) = 112 \times 91 - 43 \times 237.
\]
Entonces $91 \times 112 \equiv 1 \pmod{237}$: las soluciones son $x \equiv
112 \pmod{237}$. (Compruebe:$91 \times 112 = 10\,192 = 43 \times 237 +
1$.)
\end{solution}

\begin{solution}{exo:b1:arith:4}
$\gcd(17, 39) = 1$: Euclides da $39 = 2\times 17 + 5$,$17 = 3\times
5 + 2$,$5 = 2\times 2 + 1$ y al revés.
\[
1 = 5 - 2\times 2 = 5 - 2(17 - 3\times 5) = 7\times 5 - 2\times 17
= 7(39 - 2\times 17) - 2\times 17 = 7\times 39 - 16\times 17 .
\]
Solución particular $(x_0, y_0) = (-16, 7)$. solución general del
ecuación homogénea $17x + 39y = 0$:$x = 39k$,$ y = -17k $(ya que
$17 \mid 39y$ y $\gcd(17,39) = 1$ fuerza $17 \mid y $ ---
Lema de Gauss). Por lo tanto
\[
(x, y) = (-16 + 39k,\; 7 - 17k), \qquad k \in \Z .
\]
Para el lado derecho $5$, multiplica la solución particular por $5$:
$(x, y) = (-80 + 39k,\; 35 - 17k)$,$k \in \Z$.
\end{solution}

\begin{solution}{exo:b1:arith:5}
Por cada \omterm{def:b1:arith:prime}{primo} $p$, con $\alpha = v_p(a)$ y $\beta = v_p(b)$:
\[
v_p\bigl(\gcd(a,b)\bigr) + v_p\bigl(\operatorname{lcm}(a,b)\bigr)
= \min(\alpha, \beta) + \max(\alpha, \beta)
= \alpha + \beta = v_p(ab) .
\]
Dos números enteros positivos con la misma valoración en cada \omterm{def:b1:arith:prime}{primo} son iguales
(\cref{prop:b1:arith:valuation}), entonces $\gcd(a,b)\operatorname{lcm}(a,b)
= ab $.
\end{solution}

\begin{solution}{exo:b1:arith:7}
Supongamos $\sqrt p = \frac rq$ con $r, q \in \N^*$, es decir, $p q^2 =
r^2$. Aplicar $v_p $:$v_p(pq^2) = 1 + 2v_p(q)$ es impar, mientras que $v_p(r^2)
= 2 v_p(r)$ es par. Un número entero no puede tener pares ni impares.
$p$-valoración de una vez: contradicción. Entonces $\sqrt p \notin \Q$.
\end{solution}

\begin{solution}{exo:b1:arith:8}
\emph{Hecho generalizado.} Con $\gcd(m,n) = 1$, Bézout regala$ mu + nv = 1$.
Establezca $x_0 = b\,mu + a\,nv$. Luego $x_0 \equiv a\,nv \equiv a(1 - mu)
\equiv a \pmod m $ y de manera similar $x_0 \equiv b \pmod n $: existencia. si
$x$ y $x'$ son dos soluciones,$m$ y $n$ dividen $x - x'$, entonces
$mn \mid x - x'$(\cref{thm:b1:arith:gauss}~(2)): mod de unicidad
$mn$.

\emph{Numéricamente:} $m = 7$,$n = 11$:$7 \times (-3) + 11 \times 2 =
1$. Entonces $x_0 = 5 \times 7 \times (-3) + 2 \times 11 \times 2 = -105 +
44 = -61 \equiv 16 \pmod{77}$. Verificar:$16 = 2\times 7 + 2 \equiv 2
\pmod 7$;$16 = 11 + 5 \equiv 5 \pmod{11}$. Soluciones:$x \equiv 16
\pmod{77}$.
\end{solution}

\begin{solution}{exo:b1:arith:10}
Escriba $m = nq + r$,$0 \leq r < n$. entonces
\[
2^m - 1 = 2^r\bigl(2^{nq} - 1\bigr) + 2^r - 1,
\]
y $2^n - 1$\omterm{def:b1:arith:divides}{divide}$2^{nq} - 1 = (2^n - 1)(2^{n(q-1)} + \dots +
1)$. Entonces mod $2^n - 1$,$\;2^m - 1 \equiv 2^r - 1$, y desde $0 \leq
2^r - 1 < 2^n - 1$, este \emph{is} es el resto euclidiano.

Por lo tanto el \omterm{met:b1:arith:euclid}{algoritmo euclidiano} en el par $(2^m - 1, 2^n - 1)$
refleja, exponente por exponente, el algoritmo de $(m, n)$: cada uno
El paso de división reemplaza $(m, n)$ por $(n, r)$ arriba y $(2^m - 1,
2^n - 1)$ por $(2^n - 1, 2^r - 1)$ abajo. El algoritmo de arriba
termina en $\gcd(m,n)$, por lo que abajo termina en
$2^{\gcd(m,n)} - 1$.
\end{solution}

\begin{solution}{exo:b1:arith:11}
Primero resuelva $x^2 \equiv 1 \pmod p$:$p \mid (x-1)(x+1)$, entonces por
Lema de Euclides $x \equiv 1$ o $x \equiv -1 \pmod p$.

En el producto $(p-1)! = 1 \times 2 \times \dots \times (p-1)$, cada
El factor $a$ es mod $p$ invertible, y su inverso $a^{-1}$ es nuevamente
uno de los factores (\cref{prop:b1:arith:invmod}). Empareje cada $a$ con
$a^{-1}$: los pares se multiplican a$1$, excepto que los factores autoemparejados
($a = a^{-1}$, es decir, $a^2 \equiv 1$) son independientes --- y estos son
exactamente $1$ y $p - 1$. Por lo tanto
\[
(p-1)! \equiv 1 \times (p - 1) \equiv -1 \pmod p .
\]
(Para $p = 2$:$1! = 1 \equiv -1 \pmod 2$; el argumento de emparejamiento
degenera pero el resultado se mantiene.)

\emph{Conversar.} Sea $n \geq 2$ compuesto,$n = ab $ con $1 < a
\leq b < n $. Si$a < b $, ambos aparecen como factores distintos de $(n-1)!$,
entonces $n \mid (n-1)!$ y $(n-1)! \equiv 0 \not\equiv -1$. Si$a = b$
(es decir,\ $n = a^2$): para $a \geq 3$, tanto $a$ como $2a$ son $< n$, por lo que
$n = a^2 \mid a \times 2a \mid (n-1)!$, misma conclusión; para $n = 4$,
$(n-1)! = 6 \equiv 2 \not\equiv -1 \pmod 4$.
\end{solution}

\begin{solution}{pb:b1:arith:1}
\textbf{1.} $10! = 3\,628\,800$: dos ceros finales. Valoraciones
factor por factor: las potencias de $2$ provienen de $2, 4 = 2^2, 6, 8 = 2^3,
10$, totalizando $v_2(10!) = 1 + 2 + 1 + 3 + 1 = 8$; poderes de $5$
de $5$ y $10$:$v_5(10!) = 2$. Ceros finales$=
\min(v_2, v_5) = 2$, consistentes.

\textbf{2.} Escribe la división euclidiana $\lfloor x\rfloor = nq +
r $,$0 \leq r \leq n - 1$. Luego $x = nq + r + \{x\}$ con $0 \leq
r + \{x\} < n $, entonces $\lfloor x/n \rfloor = q = \bigl\lfloor \lfloor
x \rfloor / n \bigr\rfloor $.

\textbf{3.} Sea $\alpha = v_p(a) \leq \beta = v_p(b)$(intercambiar si
necesario) y escriba $a = p^\alpha a'$,$b = p^\beta b'$ con $p
\nmid a', b'$. Luego $a + b = p^\alpha\bigl(a' + p^{\beta -
\alpha}b'\bigr)$, entonces $v_p(a + b) \geq \alpha = \min $. Si$\alpha <
\beta $, el paréntesis es $a' + p^{\beta-\alpha}b' \equiv a'
\not\equiv 0 \pmod p $: la valoración es exactamente $\alpha$.

\textbf{4.} Los múltiplos de $m$ en $\intint1n$ son $m, 2m, \dots,
qm $ donde $q $ es el entero más grande con $qm \leq n $, es decir, $q =
\lfloor n/m \rfloor $.

\textbf{5.} Por factorización única, $v_p(n!) = \sum_{j=1}^{n}
v_p(j)$. Cuente de manera diferente: cada $j $contribuye con $v_p(j) =
\#\{k \geq 1 : p^k \mid j\}$, por lo que
\[
v_p(n!) = \sum_{j=1}^n \#\{k : p^k \mid j\}
= \sum_{k\geq1} \#\{j \leq n : p^k \mid j\}
= \sum_{k\geq1} \Bigl\lfloor \frac n{p^k} \Bigr\rfloor
\]
por la pregunta 4 --- Fórmula de Legendre. La suma es finita: términos
con $p^k > n$ desaparecen.

\textbf{6.} $v_5(1000!) = 200 + 40 + 8 + 1 = 249$(divisiones por
$5, 25, 125, 625$);$v_2(1000!) = 500 + 250 + 125 + 62 + 31 + 15 +
7 + 3 + 1 = 994$. Ceros finales de $1000!$: cada cero consume
un $2$ y un $5$, por lo que hay $\min(994, 249) = 249$ de ellos.

\textbf{7.} Con $n = \sum_i a_ip^i$, la pregunta 2 da $\lfloor
n/p^k \rfloor = \sum_{i \geq k} a_ip^{i-k}$(trunca la base-$p$
expansión). Sumando $k \geq 1$ e intercambiando los dos finitos
sumas:
\[
v_p(n!) = \sum_{i\geq1} a_i \sum_{k=1}^{i} p^{i-k}
= \sum_{i\geq0} a_i\,\frac{p^i - 1}{p - 1}
= \frac{n - s_p(n)}{p - 1} .
\]

\textbf{8.} Para $p = 2$:$v_2(n!) = n - s_2(n)$. Desde $n \geq 1$
tiene $s_2(n) \geq 1$, siempre $v_2(n!) \leq n - 1 < n$:$2^n \nmid
n!$. Y$v_2(n!) = n - 1$ si $ s_2(n) = 1$ si $ n $ es un poder de
$2$.

\textbf{9.} $n$ tiene $\lfloor \log_p n \rfloor + 1$ base-$p$
dígitos, cada uno como máximo $p - 1$, por lo que $1 \leq s_p(n) \leq
(p-1)\bigl(\log_p(n) + 1\bigr)$. Sustituyendo en la pregunta 7:
\[
\frac n{p-1} - \log_p(n) - 1 \;\leq\; v_p(n!) \;<\; \frac n{p-1},
\]
y dividiendo por $n$:$\frac{v_p(n!)}n \to \frac1{p-1}$.

\textbf{10.} $Z(n) - Z(n-1) = v_5(n!/(n-1)!) = v_5(n)$: el conteo
de ceros finales salta $v_5(n)$ en cada múltiplo de $5$ y es
constante en el medio. $Z(24) = \lfloor24/5\rfloor = 4$ y $Z(25) =
5 + 1 = 6$: en $n = 25$ el conteo salta de $4$ directamente a$6$
($v_5(25) = 2$), y dado que $Z$ no es decreciente con $Z \leq 4$
antes y $Z \geq 6$ después, el valor $5$ nunca se alcanza: no
El factorial termina exactamente en cinco ceros.

\textbf{11.} Escribe $x = \lfloor x\rfloor + \{x\}$:$\lfloor x +
y\rfloor = \lfloor x\rfloor + \lfloor y\rfloor + \lfloor \{x\} +
\{y\}\rfloor $, y $0 \leq \{x\} + \{y\} < 2$ hace el último piso.
$0$ o $1$. Luego, por Legendre aplicado tres veces,
\[
v_p\binom{m+n}m = v_p\bigl((m{+}n)!\bigr) - v_p(m!) - v_p(n!)
= \sum_{k\geq1}\Bigl(
\Bigl\lfloor\frac{m+n}{p^k}\Bigr\rfloor
- \Bigl\lfloor\frac{m}{p^k}\Bigr\rfloor
- \Bigl\lfloor\frac{n}{p^k}\Bigr\rfloor\Bigr),
\]
una suma finita de $0$s y $1$s (aplicar la primera afirmación a$x =
m/p^k $,$ y = n/p^k$).

\textbf{12.} Corrija $k \geq 1$ y escriba $m = p^km_1 + m_0$,$n =
p^kn_1 + n_0$ con $0 \leq m_0, n_0 < p^k $(división euclidiana:
$m_0$ es el número formado por los dígitos bajos $k$ de $m$). entonces
\[
\Bigl\lfloor\frac{m+n}{p^k}\Bigr\rfloor
- \Bigl\lfloor\frac m{p^k}\Bigr\rfloor
- \Bigl\lfloor\frac n{p^k}\Bigr\rfloor
= \Bigl\lfloor\frac{m_0 + n_0}{p^k}\Bigr\rfloor ,
\]
que es $1$ si es $m_0 + n_0 \geq p^k$ y $0$ en caso contrario. Pero $m_0 +
n_0 \geq p^k $dice precisamente que agregar los dígitos bajos $k $ de $m $
y $n$ se desborda a la posición $k$--- un acarreo a la posición $k$
en el algoritmo de suma de libros escolares. Resumiendo $k$:
$v_p\binom{m+n}m$ es el número de acarreos en la base-$p$
Además $m + n$. (Kummer, 1852.)

\textbf{13.} Aplicar Kummer a $m = j$,$n = p^k - j$, suma $p^k =
(1\underbrace{0\cdots0}_{k})_p $. Sea $a = v_p(j)$, entonces la base-$p$
Los dígitos de $j$ en las posiciones $0, \dots, a-1$ son $0$ y el dígito
en la posición $a$ es distinto de cero. Los dígitos de $p^k - j$ debajo de la posición
$a$ también son $0$($p^k - j = p^a(p^{k-a} - j/p^a)$). en la posición
$a$, los dos dígitos distintos de cero deben sumar $p $(dígito de resultado $0$):
uno lleva; en cada posición $a+1, \dots, k-1$, dígitos más el
suma de acarreo entrante a $p $(dígito de resultado $0$ nuevamente): el acarreo
se propaga. Total: $k - a$ lleva, por lo que $v_p\binom{p^k}j = k -
v_p(j)$. Para $k = 1$:$v_p\binom pj = 1$ para $0 < j < p $, el
\omterm{def:b1:arith:divides}{divisibilidad} utilizado en \cref{thm:b1:arith:fermat}.

\textbf{14.} Mediante la forma digital (pregunta 7), utilizando $s_2(2n) =
s_2(n)$(agregando un dígito cero):
\[
v_2\binom{2n}n = \bigl(2n - s_2(2n)\bigr) - 2\bigl(n -
s_2(n)\bigr) = 2s_2(n) - s_2(2n) = s_2(n) \geq 1 :
\]
$\binom{2n}n$ es siempre par, y $v_2 = 1$(es decir,$\binom{2n}n
\equiv 2 \pmod 4$) exactamente cuando $s_2(n) = 1$, es decir,\ cuando $n $ es un
potencia de $2$.

\textbf{15.} Vandermonde con $m = n = k = p$:$\binom{2p}p =
\sum_{j=0}^p \binom pj\binom p{p-j} = \sum_{j=0}^p \binom pj^2$.
Para $0 < j < p$,$p \mid \binom pj $(pregunta 13), entonces $\binom
pj^2 \equiv 0 \pmod p $; los términos finales dan $1 + 1$:
$\binom{2p}p \equiv 2 \pmod p$.

\textbf{16.} Base $3$:$500 = 486 + 9 + 3 + 2$, dígitos (bajo a
alto) $(2, 1, 1, 0, 0, 2)$, entonces $s_3(500) = 6$; y $1000 = 729 +
243 + 27 + 1$, dígitos $(1, 0, 0, 1, 0, 1, 1)$, entonces $s_3(1000) = 4$.
\emph{Kummer:} agrega $500 + 500$ en la base $3$: posición $0$:$2 + 2 =
4$, dígito $1$ lleva $1$; posición $1$:$1 + 1 + 1 = 3$, dígito $0$
llevar $1$; posición $2$:$1 + 1 + 1 = 3$, dígito $0$ lleva $1$;
posición $3$:$0 + 0 + 1 = 1$, sin acarreo; posición $4$:$0$;
posición $5$:$2 + 2 = 4$, dígito $1$ lleva $1$; posición $6$:
tierras de acarreo: dígito $1$. Cuatro acarreos:$v_3\binom{1000}{500} = 4$.
\emph{Leyenda:} $v_3(1000!) = \frac{1000 - 4}2 = 498$ y
$v_3(500!) = \frac{500 - 6}2 = 247$, entonces $v_3\binom{1000}{500} =
498 - 2\times247 = 4$. Los dos cálculos concuerdan --- y el
Los dígitos de suma $(1, 0, 0, 1, 0, 1, 1)$ reproducen $1000$, tal como
debe.

\textbf{17.} Por Kummer ($p = 2$,$m = k$,$n' = n - k$):
$\binom nk$ es impar si la suma $k + (n - k)$ en la base $2$ tiene
sin acarreo, si en cada posición los dígitos satisfacen $k_i + (n -
k)_i = n_i $; en ese caso $k_i \leq n_i $ para todos los $i $. Por el contrario,
si $k_i \leq n_i$ para todos los $i$, entonces el número con dígitos $n_i -
k_i $ es $n - k $ y la suma no tiene acarreo. Misma prueba en base.
$p$:$p \nmid \binom nk $ si cada dígito de base $p$ de $k$ está en
más el dígito correspondiente de $n$.

\textbf{18.} Contando el $k \in \intint0n$ cuyos dígitos obedecen
$k_i \leq n_i$: cada dígito de $k$ se elige de forma independiente entre
Valores $n_i + 1$, que dan opciones a$\prod_i (n_i + 1)$; en la base $2$
Este es $2^{\#\{i : n_i = 1\}} = 2^{s_2(n)}$. Fila $4 =
(100)_2$:$2^1 = 2$ entradas impares --- de hecho $1, 4, 6, 4, 1$ tiene entradas impares
entradas sólo en los extremos. Fila $5 = (101)_2$:$2^2 = 4$ --- de hecho
$1, 5, 10, 10, 5, 1$.

\textbf{19.} Todas las entradas interiores, incluso $\iff$, la fila tiene exactamente
$2$ entradas impares (los dos extremos siempre son impares)$\iff 2^{s_2(n)} =
2 \iff s_2(n) = 1 \iff n $ es una potencia de $2$.

\textbf{20.} En la suma de la pregunta 11, el término $k$ desaparece como
tan pronto como $p^k > m + n $(los tres pisos son entonces iguales, de hecho el
primero es $0$ cuando $p^k > m+n$; más simplemente cada término es $0$).
Por lo tanto, como máximo los términos $\lfloor \log_p(m+n)\rfloor$ son distintos de cero, cada uno
vale $1$:$a = v_p\binom{m+n}m \leq \log_p(m+n)$, es decir $p^a \leq
m + n$.

\textbf{21.} Por cada \omterm{def:b1:arith:prime}{primo} $p$,$v_p\bigl(\operatorname{lcm}(1,
\dots, 2n)\bigr) = \lfloor\log_p(2n)\rfloor $(la mayor potencia de
$p$ no superior a$2n$ aparece entre $1, \dots, 2n$). Pregunta 20
con $m = n$ da $v_p\binom{2n}n \leq \lfloor\log_p(2n)\rfloor $
por cada $p$: por \cref{prop:b1:arith:valuation},$\binom{2n}n
\mid \operatorname{lcm}(1, \dots, 2n)$. Para el tamaño: la proporción
$\binom{2n}{k+1}/\binom{2n}k = \frac{2n-k}{k+1} \geq 1$ exactamente
para $k < n$, por lo que la entrada central es la más grande de $2n + 1$
entradas de la fila $2n$, de donde $4^n = \sum_k \binom{2n}k \leq
(2n+1)\binom{2n}n $. Combinando:
\[
\operatorname{lcm}(1, \dots, 2n) \geq \binom{2n}n \geq
\frac{4^n}{2n + 1} .
\]
