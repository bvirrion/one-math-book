\begin{solution}{exo:b1:series:1}
$\dfrac{1}{n(n+1)} = \dfrac1n - \dfrac{1}{n+1}$: telescópico,$S_N =
1 - \frac{1}{N+1} \to 1$. Convergente, suma $1$.

$\ln\bigl(1 - \frac{1}{n^2}\bigr) = \ln\frac{(n-1)(n+1)}{n^2} =
\ln\frac{n-1}{n} - \ln\frac{n}{n+1}$: telescópico nuevamente,$S_N =
\ln\frac12 - \ln\frac{N}{N+1} \to -\ln 2$. Convergente, suma$-\ln 2$.

$\dfrac{3^n + 4^n}{5^n} = \bigl(\frac35\bigr)^n +
\bigl(\frac45\bigr)^n$: dos \omterm{ex:b1:series:geometric}{serie geométrica} convergentes, suma
$\frac{1}{1 - 3/5} + \frac{1}{1 - 4/5} = \frac52 + 5 = \frac{15}{2}$.
\end{solution}

\begin{solution}{exo:b1:series:2}
\omterm{thm:b1:series:ratio}{prueba de relación} en todas partes.

$\frac{u_{n+1}}{u_n} = \frac{(n+1)^2}{2n^2} \to \frac12 < 1$:
convergente.

$\frac{u_{n+1}}{u_n} = \frac{(n+1)!\,n^n}{n!\,(n+1)^{n+1}} =
\bigl(\frac{n}{n+1}\bigr)^n = \bigl(1 + \frac1n\bigr)^{-n} \to
\frac1\eu < 1$: convergente.

Con el factor $2^n$: ratio $\to \frac2\eu < 1$: convergente.

Con $3^n$: relación $\to \frac3\eu > 1$: divergente (los términos tienden a
$+\infty$).
\end{solution}

\begin{solution}{exo:b1:series:3}
Todos los términos no negativos; usar equivalentes
(\cref{thm:b1:series:positive}).

$\sin\frac{1}{n^2} \sim \frac{1}{n^2}$: convergente (Riemann $\alpha
= 2$).

$1 - \cos\frac1n \sim \frac{1}{2n^2}$: convergente.

$\frac{1}{\sqrt{n(n+1)}} \sim \frac1n$: divergente.

$\frac{\ln n}{n^2} = \frac{1}{n^{3/2}}\cdot\frac{\ln n}{n^{1/2}}$
y $\frac{\ln n}{\sqrt n} \to 0$
(\cref{prop:b1:functions:powerrules}): entonces $\frac{\ln n}{n^2} \leq
\frac{1}{n^{3/2}}$ para $ n $ grande: convergente.
\end{solution}

\begin{solution}{exo:b1:series:4}
Para $n \geq 2$:$\frac{1}{n^2} \leq \frac{1}{n(n-1)} =
\frac{1}{n-1} - \frac1n$. Por lo tanto
\[
\sum_{n=1}^{N} \frac{1}{n^2} \leq 1 + \sum_{n=2}^{N}
\Bigl(\frac{1}{n-1} - \frac1n\Bigr) = 1 + 1 - \frac1N < 2 :
\]
sumas parciales crecientes y acotadas por $2$: convergencia
(\cref{thm:b1:series:positive}), suma $\leq 2$. (El valor exacto
$\frac{\pi^2}{6}$ es una celebración del segundo año).
\end{solution}

\begin{solution}{exo:b1:series:5}
$\sum \frac{(-1)^n}{\sqrt n}$: alternando con $\frac{1}{\sqrt n}
\downarrow 0$: convergente (\cref{thm:b1:series:alternating}); no
absolutamente ($\alpha = \frac12 \leq 1$).

$\sum \frac{(-1)^n}{n + (-1)^n}$: la secuencia $\frac{1}{n +
(-1)^n}$ es \emph{no} decreciente ($\frac{1}{n+1}$ luego
$\frac{1}{n}$ alternan mal), por lo que la prueba no se aplica directamente.
Ampliar:
\[
\frac{(-1)^n}{n + (-1)^n}
= \frac{(-1)^n}{n}\cdot\frac{1}{1 + \frac{(-1)^n}{n}}
= \frac{(-1)^n}{n} - \frac{1}{n^2} +
O\Bigl(\frac{1}{n^3}\Bigr):
\]
el primer \omterm{def:b1:series:def}{serie} converge (alternando), $\sum \frac{1}{n^2}$
converge, el $O(n^{-3})$ converge absolutamente: la suma de tres
convergente \omterm{def:b1:series:def}{serie} converge.

$\sin\bigl(\pi\sqrt{n^2+1}\bigr)$: escriba $\sqrt{n^2 + 1} = n +
\frac{1}{2n} + \varepsilon_n$ con $\varepsilon_n = O(n^{-3})$;
luego, por $\pi$-periodicidad de $\sin $ hasta firmar,
\[
\sin\bigl(\pi\sqrt{n^2+1}\bigr)
= (-1)^n \sin\Bigl(\frac{\pi}{2n} + \pi\varepsilon_n\Bigr) .
\]
Configure $\theta_n = \frac{\pi}{2n} + \pi\varepsilon_n$ y $a_n =
\sin\theta_n $. Para grandes $n $,$\theta_n \in \intoo{0}{\frac\pi2}$
y
\[
\theta_n - \theta_{n+1} = \frac{\pi}{2n(n+1)} +
\pi(\varepsilon_n - \varepsilon_{n+1})
= \frac{\pi}{2n^2} + O\Bigl(\frac{1}{n^3}\Bigr) > 0
\]
eventualmente, $(\theta_n)$ disminuye a$0$; ya que $\sin $ es
aumentando en $\intcc{0}{\frac\pi2}$,$(a_n)$ disminuye a$0$ a medida
bueno. Se aplica la prueba de alternancia: convergente --- no absolutamente,
desde $a_n \sim \frac{\pi}{2n}$.
\end{solution}

\begin{solution}{exo:b1:series:6}
$f(t) = \frac{1}{t(\ln t)^\alpha}$ es positivo, \omterm{def:b1:continuity:continuous}{continuo},
disminuyendo en $\intco{2}{+\infty}$. Sustituyendo $u = \ln t$:
\[
\int_2^x \frac{\dd t}{t(\ln t)^\alpha}
= \int_{\ln 2}^{\ln x} \frac{\dd u}{u^\alpha},
\]
delimitado como $x \to \infty$ y si $\alpha > 1$
(cálculo de \cref{thm:b1:series:riemann}). Por \omterm{thm:b1:integration:def}{integral}
comparación: convergencia si $\alpha > 1$. (Estos \emph{tipo Bertrand}
\omterm{def:b1:series:def}{serie} muestra qué tan fino es el \omterm{def:b1:topology:closure}{límite} de convergencia: $n\ln n$
diverge, $n(\ln n)^{1.01}$ converge).
\end{solution}

\begin{solution}{exo:b1:series:7}
Por los límites de la línea tangente de \cref{exo:b1:derivative:3} reescrito
mediante ampliaciones: para $x = \frac1n \in \intoc{0}{1}$, Taylor--Lagrange
para $\ln(1+x)$ en el pedido $1$ da $\ln(1 + x) = x - \frac{x^2}{2(1 +
c)^2}$ para algunos $c \in \intoo{0}{x}$, por lo que
\[
0 \leq u_n = \frac1n - \ln\Bigl(1 + \frac1n\Bigr) \leq
\frac{1}{2n^2} .
\]
Comparación con el \omterm{thm:b1:series:riemann}{serie de riemann}: $\sum u_n$ converge. es parcial
la suma telescopio los logaritmos:
\[
\sum_{n=1}^{N} u_n = H_N - \ln(N+1)
\]
(desde $\sum_{n\leq N} \ln\frac{n+1}{n} = \ln(N+1)$). Entonces $H_N -
\ln(N+1)$ converge; agregando $\ln\frac{N+1}{N} \to 0$, la secuencia
$H_N - \ln N$ converge. Su límite es $\gamma \approx 0.5772$.
\end{solution}

\begin{solution}{exo:b1:series:8}
$\dfrac{1}{n(n+2)} = \dfrac{1/2}{n} - \dfrac{1/2}{n+2}$: el parcial
telescopios de suma con un retraso de $2$,
\[
S_N = \frac12\Bigl(1 + \frac12 - \frac{1}{N+1} -
\frac{1}{N+2}\Bigr) \longrightarrow \frac34 .
\]

$\sum \frac{n}{2^n}$: para $\abs x < 1$, diferenciando lo finito
suma geométrica y pasando al límite (todos \omterm{def:b1:series:def}{serie} aquí convergen
absolutamente, \omterm{thm:b1:series:ratio}{prueba de relación}): de $\sum_{n\geq0} x^n = \frac{1}{1-x}$,
se obtiene mediante cálculo directo con sumas parciales
\[
\sum_{n=1}^{N} n x^{n-1}
= \frac{1 - (N+1)x^N + N x^{N+1}}{(1 - x)^2}
\xrightarrow[N\to\infty]{} \frac{1}{(1-x)^2}
\quad (\abs x < 1),
\]
(los términos \omterm{def:b1:topology:closure}{límite} $N x^N \to 0$). En$x = \frac12$:
$\sum_{n\geq1} n\bigl(\frac12\bigr)^{n-1} = 4$, entonces $\sum_{n\geq0}
\frac{n}{2^n} = \frac12 \times 4 = 2$.
\end{solution}

\begin{solution}{exo:b1:series:9}
Agrupe los términos de $\sum u_n$ en paquetes entre potencias de $2$.
Paquetes superiores: para $2^k \leq n < 2^{k+1}$ hay términos $2^k$,
cada $\leq u_{2^k}$:
\[
\sum_{n=1}^{2^{K+1}-1} u_n
= \sum_{k=0}^{K} \sum_{n=2^k}^{2^{k+1}-1} u_n
\leq \sum_{k=0}^{K} 2^k u_{2^k} .
\]
Paquetes inferiores: cada término del mismo paquete es $\geq u_{2^{k+1}}$,
entonces $\sum_{n=2^k}^{2^{k+1}-1} u_n \geq 2^k u_{2^{k+1}} = \frac12
\cdot 2^{k+1} u_{2^{k+1}}$, de donde
\[
\sum_{n=1}^{2^{K+1}-1} u_n \geq \frac12 \sum_{k=1}^{K+1} 2^{k}
u_{2^{k}} .
\]
Ambas comparaciones de suma parcial van en ambos sentidos (términos no negativos,
\cref{thm:b1:series:positive}): los dos \omterm{def:b1:series:def}{serie} tienen la misma naturaleza.

Riemann: $u_n = n^{-\alpha}$ da $2^k u_{2^k} =
2^{k(1-\alpha)}$, un \omterm{ex:b1:series:geometric}{serie geométrica}, convergente si $2^{1 - \alpha}
< 1$ si $\alpha > 1$. Bertrand
(\cref{exo:b1:series:6}): $u_n = \frac{1}{n(\ln n)^\alpha}$ da
$2^k u_{2^k} = \frac{1}{(k\ln 2)^\alpha}$, un \omterm{thm:b1:series:riemann}{serie de riemann} en $k$:
convergente si $\alpha > 1$.
\end{solution}

\begin{solution}{exo:b1:series:10}
Identidad geométrica finita con relación $-t^2$:
\[
\frac{1}{1 + t^2} = \sum_{k=0}^{n-1} (-1)^k t^{2k}
+ \frac{(-1)^n t^{2n}}{1 + t^2} .
\]
Integrar sobre $\intcc{0}{1}$(el lado izquierdo se integra a$\arctan
1 = \frac\pi4$, \cref{prop:b1:functions:arcderiv}):
\[
\frac{\pi}{4} = \sum_{k=0}^{n-1} \frac{(-1)^k}{2k+1}
+ (-1)^n \int_0^1 \frac{t^{2n}}{1+t^2}\,\dd t,
\qquad
0 \leq \int_0^1 \frac{t^{2n}}{1+t^2}\,\dd t \leq \int_0^1 t^{2n}\dd
t = \frac{1}{2n+1} .
\]
Dejar que $n \to \infty$ pruebe la fórmula y el límite mostrado
en el \omterm{thm:b1:integration:def}{integral} es exactamente el límite restante: después de sumar hasta
$N$(es decir, \ términos $n = N + 1$),$\abs{R_N} \leq \frac{1}{2N + 3}$.
\end{solution}

\begin{solution}{exo:b1:series:11}
$n^{1/n} = \eu^{\frac{\ln n}{n}} \to \eu^0 = 1$
(\cref{prop:b1:functions:powerrules}). Por lo tanto
\[
\frac{1}{n^{1 + 1/n}} = \frac{1}{n}\,\eu^{-\frac{\ln n}{n}}
\sim \frac{1}{n} ,
\]
y la prueba de equivalentes (\cref{thm:b1:series:positive})
se compara con el armónico divergente \omterm{def:b1:series:def}{serie}: \emph{divergente},
aunque cada exponente $1 + \frac1n$ excede a$1$. El Riemann
el criterio se refiere a un exponente \emph{fijado} $\alpha$; un exponente
deslizarse hacia $1$ puede perder todo su margen, como aquí.
\end{solution}

\begin{solution}{exo:b1:series:12}
Sea $\varepsilon > 0$. Según el criterio de Cauchy para la
\omterm{def:b1:series:def}{serie} convergente (\cref{thm:b1:seq:complete} aplicado al
sumas parciales), hay $N$ con $\sum_{k=n+1}^{2n} u_k \leq
\varepsilon$ para $n \geq N $. Por monotonicidad cada uno de estos $n $
términos es $\geq u_{2n}$:
\[
n\,u_{2n} \leq \sum_{k=n+1}^{2n} u_k \leq \varepsilon
\quad\Longrightarrow\quad 2n\,u_{2n} \leq 2\varepsilon ,
\]
y para índices impares $(2n+1)\,u_{2n+1} \leq (2n+1)\,u_{2n} \leq
2\bigl(2n\,u_{2n}\bigr) \leq 4\varepsilon $ para $n \geq N $: en
ambas paridades, $n u_n \to 0$.

Inverso falso: $u_n = \frac{1}{n\ln n}$ tiene $n u_n =
\frac{1}{\ln n} \to 0$, pero \omterm{def:b1:series:def}{serie} diverge
(\cref{exo:b1:series:6}, $\alpha = 1$). Monotonicidad necesaria:
sea $u_n = \frac1n$ cuando $n$ es un cuadrado perfecto y $u_n =
2^{-n}$ en caso contrario: el \omterm{def:b1:series:def}{serie} converge (los términos del cuadrado suman
como $\sum \frac{1}{k^2}$, el resto geométricamente), pero $n u_n =
1$ a lo largo de los cuadrados.
\end{solution}

\begin{solution}{pb:b1:series:1}
\textbf{1.} $a_{n+1} - a_n = \frac{1}{n+1} - \ln\frac{n+1}{n}
\leq 0$ porque $\ln(1 + \frac1n) \geq \frac{1/n}{1 + 1/n} =
\frac{1}{n+1}$; y $b_{n+1} - b_n = \frac{1}{n+1} -
\ln\frac{n+2}{n+1} \geq 0$ porque $\ln(1 + \frac{1}{n+1}) \leq
\frac{1}{n+1}$. Su brecha $a_n - b_n = \ln(1 + \frac1n) \to 0$:
adyacente (\cref{thm:b1:seq:adjacent}), con límite común $\lim
a_n = \gamma $(\cref{exo:b1:series:7}) y $b_n \leq \gamma
\leq a_n $.

\textbf{2.} $b_{10} = 2.928968 - \ln 11 = 0.5311$ y $a_{10} =
2.928968 - \ln 10 = 0.6264$: entonces $\gamma \in
\intcc{0.5311}{0.6264}$. La brecha es $\ln 1.1 \approx 0.095$ y
se reduce como $\frac1n$: cuatro decimales ($\text{gap} \leq
10^{-4}$) necesitarían $n \approx 10^4$ --- los corchetes son
correcto pero lento.

\textbf{3.} Telescópico $a_n - a_{m+1} = \sum_{k=n}^{m} w_k$ y
dejando $m \to \infty$: $a_n - \gamma = \sum_{k\geq n} w_k$
(límite de sumas parciales). Además
\[
w_k = \int_k^{k+1} \frac{\dd t}{t} - \frac{1}{k+1}
= \int_k^{k+1} \Bigl(\frac1t - \frac{1}{k+1}\Bigr)\dd t
= \int_k^{k+1} \frac{k + 1 - t}{t\,(k+1)}\,\dd t .
\]
En $\intcc{k}{k+1}$:$\frac{1}{(k+1)^2} \leq \frac{1}{t(k+1)}
\leq \frac{1}{k(k+1)}$ y $\int_k^{k+1}(k + 1 - t)\dd t =
\frac12$: por lo tanto $\frac{1}{2(k+1)^2} \leq w_k \leq
\frac{1}{2k(k+1)}$.

\textbf{4.} Superior: $\sum_{k \geq n} \frac{1}{2k(k+1)} =
\frac12\sum_{k\geq n}\bigl(\frac1k - \frac{1}{k+1}\bigr) =
\frac{1}{2n}$(telescópico). Inferior:$\frac{1}{2(k+1)^2} \geq
\frac{1}{2(k+1)(k+2)}$, cuya suma se reduce a
$\frac{1}{2(n+1)}$. Con la pregunta 3:
\[
\frac{1}{2(n+1)} \leq H_n - \ln n - \gamma \leq \frac{1}{2n} .
\]

\textbf{5.} Resta $\frac{1}{2n}$:$\gamma_n - \gamma \in
\intcc{\frac{1}{2(n+1)} - \frac{1}{2n}}{0} =
\intcc{-\frac{1}{2n(n+1)}}{0}$: la estimación corregida es exacta
a $O\bigl(\frac{1}{n^2}\bigr)$, y siempre desde abajo.

\textbf{6.} $\gamma_{100} = 5.1873775 - \ln 100 - 0.005 =
0.5772073$, con $0 \leq \gamma - \gamma_{100} \leq
\frac{1}{20200} < 5\cdot10^{-5}$: por lo tanto $0.577207 \leq \gamma
\leq 0.577257$, es decir $\gamma = 0.5772 \pm 5\cdot10^{-5}$(verdadero
valor $0.5772156\dots$) --- cuatro decimales certificados de un
cien términos, frente a diez mil de la pregunta 2.

\textbf{7.} Primer dividendo:
\[
H_{2m} - H_m = \Bigl(\ln 2m + \gamma + \frac{1}{4m}\Bigr)
- \Bigl(\ln m + \gamma + \frac{1}{2m}\Bigr) +

= \ln 2 - \frac{1}{4m} + O\Bigl(\frac{1}{m^2}\Bigr).

Segundo: los recíprocos pares hasta $2m$ suman $\frac12 H_m$, por lo que

\ln m + \ln 2 + \frac\gamma2 + o(1)$ y $\sum_{j=1}^m
\frac{1}{2j} = \frac12\ln m + \frac\gamma2 + o(1)$.

\textbf{8.} Dividiendo los términos pares dos veces:
$\sum_{k=1}^{2m}\frac{(-1)^{k-1}}{k} = H_{2m} - 2\cdot\frac12

\textbf{10.} El promedio mata el término principal oscilante:
\[
S - \tilde S_N = \frac{(S - S_N) + (S - S_{N+1})}{2}

\textbf{11.} El error alterno cambia de signo en cada paso,
por lo que sumas parciales consecutivas se encuentran entre el límite y su media
cancela el término de primer orden; el error de corchete $H_n - \ln n -

(pregunta 5 y $\ln(1 + \frac{1}{2n}) = \frac{1}{2n} -

\textbf{12.} $\sum_{j\leq m} \frac{1}{2j-1} \geq \sum_{j \leq m}
\frac{1}{2j} = \frac12 H_m \to +\infty$: tanto el positivo como el
la parte negativa del armónico alterno \omterm{def:b1:series:def}{serie} diverge.

\textbf{13.} Escriba $u_n^\pm = \frac{\abs{u_n} \pm u_n}{2} \geq
0$, por lo que $u_n = u_n^+ - u_n^-$ y $\abs{u_n} = u_n^+ + u_n^-$. si
$\sum u_n^+$convergió, luego $\sum u_n^- = \sum (u_n^+ - u_n)$
convergería (diferencia de \omterm{def:b1:series:def}{serie} convergente), de ahí también $\sum
\abs{u_n}$: contradicción con convergencia condicional. Por
simetría ambos $\sum u_n^\pm$ divergen (a $+\infty$): un infinito
reservorio de masa positiva y negativa.

\textbf{14.} Sea $S = \sum u_n$,$\varepsilon > 0$ y $N$ con
$\sum_{n > N} \abs{u_n} \leq \varepsilon $(criterio de Cauchy para
$\sum\abs{u_n}$). Sea $M_0$ lo suficientemente grande como para que $\sigma(\{0,
\dots, M_0\}) \supseteq \{0, \dots, N\}$. Para $M \geq M_0$, el
diferencia $\sum_{m \leq M} u_{\sigma(m)} - \sum_{n \leq N} u_n$
es una suma finita de \emph{distinto} términos $u_n$ con $n > N$,
por tanto de valor absoluto $\leq \varepsilon$; y $\abs{S -
\sum_{n\leq N} u_n} \leq \varepsilon$ también. Entonces el reordenado
las sumas parciales están dentro de $2\varepsilon$ de $S$ eventualmente:
$\sum u_{\sigma(n)} = S$. \omterm{thm:b1:series:absolute}{Convergencia absoluta} es
a prueba de reordenamientos.
