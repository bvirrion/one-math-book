\begin{solution}{exo:b1:poly:1}
$X^5 - 1 = (X^2 + X + 1)(X^3 - X^2 + 1) + (-X - 2)$. Pasos: restar
$X^3 B$, luego $-X^2 B$, luego $B$; el resto$-X - 2$ tiene titulación
$1 < 2$. \emph{Consultar en $X = 1$:}$\;0 = 3 \times 1 + (-3)$.

$2X^4 + X^3 - X + 3 = (X^2 - 2)(2X^2 + X + 4) + (X + 11)$. \emph{comprobar
en $X = 0$:}$\;3 = (-2)(4) + 11$.
\end{solution}

\begin{solution}{exo:b1:poly:2}
$X^2 + X + 1 = (X - j)(X - j^2)$ con $j = \eu^{2\iu\pi/3}$,$j^3 =
1$. Es \omterm{def:b1:arith:divides}{divide}$Q_n = X^{2n} + X^n + 1$ si $ j $ y $j^2$ son raíces de
$Q_n$; dado que $Q_n$ tiene coeficientes reales,$Q_n(j^2) =
\conj{Q_n(j)}$, la condición es solo $Q_n(j) = 0$. Ahora $Q_n(j) =
j^{2n} + j^n + 1$ depende de $n $mod $3$:
\begin{itemize}
 \item $n \equiv 0$:$Q_n(j) = 1 + 1 + 1 = 3 \neq 0$;
 \item $n \equiv 1$:$Q_n(j) = j^2 + j + 1 = 0$;
 \item $n \equiv 2$:$Q_n(j) = j^4 + j^2 + 1 = j + j^2 + 1 = 0$.
\end{itemize}
Entonces $X^2 + X + 1 \mid X^{2n} + X^n + 1$ exactamente cuando $3 \nmid n$.
\end{solution}

\begin{solution}{exo:b1:poly:3}
Por \cref{prop:b1:poly:multiplicity}, $(X-1)^2 \mid P$ y si $P(1) =
P'(1) = 0$:
\[
P(1) = 2 + a + b = 0, \qquad P'(1) = 4 + 3a + 2b = 0 .
\]
Resolviendo: $b = -a - 2$ y $4 + 3a - 2a - 4 = a = 0$, entonces $a = 0$,$b =
-2$:$P = X^4 - 2X^2 + 1 = (X^2 - 1)^2 = (X-1)^2 (X+1)^2$, que es
la factorización real.
\end{solution}

\begin{solution}{exo:b1:poly:4}
$X^3 - 1 = (X - 1)(X - j)(X - j^2)$ sobre $\C$($j = \eu^{2\iu\pi/3}$),
y $(X - 1)(X^2 + X + 1)$ sobre $\R$.

$X^4 + X^2 + 1 = (X^2 + X + 1)(X^2 - X + 1)$ sobre $\R$(multiplicar,
o nota $X^4 + X^2 + 1 = (X^2+1)^2 - X^2$); sobre $\C$, cada cuadrático
divisiones: raíces $j, j^2$ y $-j, -j^2$, es decir,$\eu^{\pm 2\iu\pi/3},
\eu^{\pm\iu\pi/3}$.

$X^6 - 1 = \prod_{k=0}^{5} (X - \eu^{\iu k\pi/3})$ sobre $\C$ y más
$\R$:
\[
X^6 - 1 = (X-1)(X+1)(X^2 + X + 1)(X^2 - X + 1),
\]
agrupando los pares conjugados $\eu^{\pm 2\iu\pi/3}$ y $\eu^{\pm
\iu\pi/3}$.
\end{solution}

\begin{solution}{exo:b1:poly:6}
Paso de primera división del \omterm{met:b1:arith:euclid}{algoritmo euclidiano}:
\[
(X + 1)(X^3 - X^2 + X - 1)
= X^4 - X^3 + X^2 - X + X^3 - X^2 + X - 1 = X^4 - 1 ,
\]
entonces la división de $X^4 - 1$ entre $X^3 - X^2 + X - 1$ es exacta
(cociente $X + 1$, resto $0$), y el algoritmo se detiene de inmediato:
\[
\gcd(X^4 - 1,\; X^3 - X^2 + X - 1) = X^3 - X^2 + X - 1
\]
(ya \omterm{def:b1:poly:def}{mónico}). La relación de Bézout es la trivial: $\gcd = 0
\cdot (X^4 - 1) + 1 \cdot (X^3 - X^2 + X - 1)$. Verificación de consistencia por
factorización: $X^3 - X^2 + X - 1 = (X - 1)(X^2 + 1)$, que de hecho es
el producto de los factores irreducibles comunes de $X^4 - 1 =
(X-1)(X+1)(X^2+1)$.
\end{solution}

\begin{solution}{exo:b1:poly:7}
Desde $P \geq 0$ en $\R$, sus raíces reales tienen incluso \omterm{def:b1:poly:derivative}{multiplicidad} (en un
raíz del impar \omterm{def:b1:poly:derivative}{multiplicidad}, $P$ cambia de signo). Usando
\cref{cor:b1:poly:factorization} y emparejamiento, escribir
\[
P = c \prod_i (X - a_i)^{2k_i} \prod_j \bigl((X - z_j)(X - \conj
z_j)\bigr)^{n_j},
\]
con $c > 0$(comportamiento en $+\infty $). dejar
\[
S = \sqrt c\, \prod_i (X - a_i)^{k_i} \prod_j (X - z_j)^{n_j}
\in \C[X],
\]
de modo que $P = S\,\conj S$ donde $\conj S$ tiene el conjugado
coeficientes. Dividir $S = A + \iu B$ con $A, B \in \R[X]$: luego
\[
P = (A + \iu B)(A - \iu B) = A^2 + B^2 .
\]
\end{solution}

\begin{solution}{exo:b1:poly:8}
Lagrange (\cref{thm:b1:poly:lagrange}) con nodos $0, 1, 2$:
\[
P = 1\cdot\frac{(X-1)(X-2)}{(0-1)(0-2)} + 3\cdot\frac{X(X-2)}{1\cdot(1-2)}
+ 2\cdot\frac{X(X-1)}{2\cdot 1}
= \frac{(X-1)(X-2)}{2} - 3X(X-2) + X(X-1).
\]
Ampliando: $\frac{X^2 - 3X + 2}{2} - 3X^2 + 6X + X^2 - X =
-\frac{3}{2}X^2 + \frac{7}{2}X + 1$.

Sistema: $P = aX^2 + bX + c$ con $c = 1$;$a + b + 1 = 3$;$4a + 2b +
1 = 2$. Restando el doble del segundo del tercero:$2a - 1 = -4$,
entonces $a = -\frac32$,$b = \frac72$. Mismo \omterm{def:b1:poly:def}{polinomio}:
$P = -\frac32 X^2 + \frac72 X + 1$. (Consulte $P(2) = -6 + 7 + 1 = 2$).
\end{solution}

\begin{solution}{exo:b1:poly:9}
$P = X^{2n+1} - 1$:$P' = (2n+1)X^{2n} \geq 0$, entonces el \omterm{def:b1:poly:def}{polinomio}
La función es creciente (estrictamente excepto en $0$), con límites.
$\mp\infty$: desaparece exactamente una vez en $\R$(en $x = 1$).

Sea $E_n = \sum_{k=0}^{n} \frac{X^k}{k!}$. Entonces $E_n' = E_{n-1} = E_n

\begin{solution}{exo:b1:poly:10}
\begin{enumerate}
 \item Inducción (se mantienen dos casos base). Usando $\cos(n+1)\theta +
 \cos(n-1)\theta = 2\cos\theta\cos n\theta $:
 \[
 T_{n+1}(\cos\theta) = 2\cos\theta \cos n\theta -
 \cos(n-1)\theta = \cos(n+1)\theta .
 \]
 \item $T_n(\cos\theta) = 0$ y si $n\theta \equiv \frac\pi2 \pmod

 \[
 x_k = \cos\Bigl(\frac{(2k+1)\pi}{2n}\Bigr),
 \qquad k = 0, 1, \dots, n-1,
 \]

 estas son \emph{todo} las raíces, cada una simple.
 \item Para $x = \cos\theta \in \intcc{-1}{1}$:$\abs{T_n(x)} =

\begin{solution}{exo:b1:poly:11}
Escribe $P = c\prod_i (X - a_i)^{m_i}$. La regla del producto (ampliada a
varios factores) da
\[
P' = c\sum_{i} m_i (X - a_i)^{m_i - 1} \prod_{k \neq i} (X -
a_k)^{m_k},
\]
y dividiendo por $P$:$\frac{P'}{P} = \sum_i \frac{m_i}{X - a_i}$
(como funciones racionales, es decir, lejos de las raíces).

Sea $w$ una raíz de $P'$. Si$w$ es uno de los $a_i$, se encuentra en el
casco convexo trivialmente. De lo contrario, evaluando en $w$:
\[
0 = \sum_i \frac{m_i}{w - a_i}
= \sum_i m_i\, \frac{\conj w - \conj a_i}{\abs{w - a_i}^2} .
\]
Conjugando: $\sum_i \lambda_i (w - a_i) = 0$ donde $\lambda_i =
\frac{m_i}{\abs{w - a_i}^2} > 0$. Por lo tanto
\[
w = \frac{\sum_i \lambda_i a_i}{\sum_i \lambda_i} :
\]
una combinación convexa (pesos positivos que suman $1$ después
normalización) de las raíces $a_i$. Entonces cada raíz de $P'$ se encuentra en el
casco convexo de las raíces de $P$.
\end{solution}

\begin{solution}{exo:b1:poly:12}
Sume las evaluaciones de $(1 + X)^n$ en los tres cubos \omterm{def:b1:complex:unity}{raíces de
unidad}:
\[
2^n + (1 + j)^n + (1 + j^2)^n
= \sum_{k=0}^n \binom nk\,\bigl(1 + j^k + j^{2k}\bigr)
= 3\sum_{k\,:\,3\mid k}\binom nk ,
\]
ya que $1 + j^k + j^{2k}$ es una suma geométrica igual a$3$ cuando $3
\mid k $ y a$\frac{j^{3k} - 1}{j^k - 1} = 0$ en caso contrario. Ahora $1 +
j = \frac12 + \iu\frac{\sqrt3}2 = \eu^{\iu\pi/3}$ y $1 + j^2 =
\conj{1 + j} = \eu^{-\iu\pi/3}$, entonces $(1+j)^n + (1+j^2)^n =
2\cos\frac{n\pi}3$ y
\[
\sum_{k\geq0}\binom n{3k} = \frac{2^n + 2\cos\frac{n\pi}3}{3} .
\]
Comprobaciones: $n = 3$:$\frac{8 + 2\cos\pi}3 = 2 = \binom30 + \binom33$;
$n = 6$:$\frac{64 + 2}3 = 22 = 1 + 20 + 1$.
\end{solution}

\begin{solution}{pb:b1:poly:1}
\textbf{1.}$T_2 = 2X^2 - 1$;$T_3 = 2X(2X^2 - 1) - X = 4X^3 -
3X $;$T_4 = 2X\,T_3 - T_2 = 8X^4 - 8X^2 + 1$;$T_5 = 2X\,T_4 - T_3
= 16X^5 - 20X^3 + 5X $. La identidad $T_3(\cos\theta) = \cos3\theta $
es exactamente $\cos3\theta = 4\cos^3\theta - 3\cos\theta$ de
\cref{ex:b1:complex:cos3}.

\textbf{2.} Verdadero para $n = 1, 2$. Si$T_{n-1}$,$T_n$ tienen grados
$n-1$,$n$ y coeficientes principales $2^{n-2}$,$2^{n-1}$, luego
$2X\,T_n$ tiene grado $n+1$ y coeficiente principal $2^n$, mientras que
$T_{n-1}$ tiene grado inferior:$T_{n+1}$ tiene grado $n + 1$, líder
coeficiente $2^n$. Paridad: si $T_{n-1}$ tiene la paridad de $n - 1$
y $T_n$ el de $n$, entonces $2X\,T_n$ y $T_{n-1}$ tienen ambos el
paridad de $n + 1$, por lo tanto también lo hace $T_{n+1}$.

\textbf{3.} Si $P(\cos\theta) = \cos n\theta$ para todos los $\theta$,
entonces $P$ y $T_n$ coinciden en cada punto de $\intcc{-1}1$ --- un
infinito \omterm{def:b1:logic:sets}{conjunto} --- entonces $P - T_n$ tiene infinitas raíces y es el
cero \omterm{def:b1:poly:def}{polinomio} (\cref{cor:b1:poly:nroots}).

\textbf{4.} Para $x = \cos\theta$: $T_m(T_n(\cos\theta)) =
T_m(\cos n\theta) = \cos(mn\theta) = T_{mn}(\cos\theta)$, y
$2T_mT_n(\cos\theta) = 2\cos m\theta\cos n\theta = \cos(m+n)\theta
+ \cos\abs{m - n}\theta $. Ambas identidades se mantienen en $\intcc{-1}1$,
por lo tanto, como identidades \omterm{def:b1:poly:def}{polinomio} según el argumento de la pregunta 3.

\textbf{5.} Las $x_k$ son $n$ raíces simples distintas y las
El coeficiente principal es $2^{n-1}$:
\[
T_n = 2^{n-1}\prod_{k=0}^{n-1}
\Bigl(X - \cos\frac{(2k+1)\pi}{2n}\Bigr) .
\]
Entrelazado: los ángulos $0 < \frac{\pi}{2n} < \frac\pi n <
\frac{3\pi}{2n} < \frac{2\pi}n < \dots < \pi $ se alternan entre
los ángulos $ y $ $\frac{k\pi}n$ y los ángulos $x$
$\frac{(2k+1)\pi}{2n}$; ya que $\cos$ es estrictamente decreciente en
$\intcc0\pi$, los valores se entrelazan en orden inverso: $y_n <
x_{n-1} < y_{n-1} < \dots < x_0 < y_0$. Entre dos consecutivos
extrema se encuentra exactamente en una raíz, como en la imagen de $\cos n\theta$
sugiere.

\textbf{6.} Inducción con $2\cosh a\cosh b = \cosh(a + b) +
\cosh(a - b)$(\cref{prop:b1:functions:hyprules}):$T_{n+1}(\cosh
t) = 2\cosh t\cosh nt - \cosh(n-1)t = \cosh(n+1)t $. Para $x \geq
1$, escriba $x = \cosh t $ con $t \geq 0$; luego $\eu^t = x +
\sqrt{x^2 - 1}$ y $\eu^{-t} = x - \sqrt{x^2 - 1}$, entonces
\[
T_n(x) = \cosh(nt)
= \frac{(x + \sqrt{x^2-1})^n + (x - \sqrt{x^2-1})^n}2 .
\]
Para $x > 1$ el primer término excede $\frac12(1)^n$ estrictamente y
crece geométricamente: $T_n(x) > 1$.

\textbf{7.} De Moivre: $\cos n\theta = \Re\bigl((\cos\theta +
\iu\sin\theta)^n\bigr) = \sum_{2j \leq n}\binom n{2j}
\cos^{n-2j}\theta\,(\iu\sin\theta)^{2j}$ y $(\iu\sin\theta)^{2j}
= (-\sin^2\theta)^j = (\cos^2\theta - 1)^j $. Sustituyendo $x =
\cos\theta $ e invocando la pregunta 3:
\[
T_n(x) = \sum_{0\leq 2j\leq n}\binom n{2j}x^{n-2j}(x^2 - 1)^j .
\]
Para $n = 3$:$\binom30 x^3 + \binom32 x(x^2 - 1) = x^3 + 3x^3 -
3x = 4x^3 - 3x $, como en la pregunta 1.

\textbf{8.}$T_n(1) = \cos(n\cdot0) = 1$;$T_n(-1) = \cos(n\pi) =
(-1)^n $;$T_n(0) = \cos\frac{n\pi}2$, que es $0$ para $n $ impar y
$(-1)^{n/2}$ para incluso $n$.

\textbf{9.} Inducción para $U_n(\cos\theta) =
\frac{\sin(n+1)\theta}{\sin\theta}$: verdadero para $U_0 = 1$ y $ U_1 =
2X $($\sin2\theta = 2\sin\theta\cos\theta$); el paso es el
identidad de suma a producto $\sin(n+2)\theta = 2\cos\theta\,
\sin(n+1)\theta - \sin n\theta $. Ahora diferencia
$T_n(\cos\theta) = \cos n\theta$ en $\theta$:
$-\sin\theta\,T_n'(\cos\theta) = -n\sin n\theta$, entonces para $\theta
\notin \pi\Z $:
\[
T_n'(\cos\theta) = n\,\frac{\sin n\theta}{\sin\theta}
= n\,U_{n-1}(\cos\theta) ,
\]
y el \omterm{def:b1:poly:def}{polinomios} $T_n'$ y $nU_{n-1}$, acordando
$\intoo{-1}1$, son iguales.

\textbf{10.} $\abs{\sin(n+1)\theta} = \abs{\sin n\theta\cos\theta
+ \cos n\theta\sin\theta} \leq \abs{\sin n\theta} +
\abs{\sin\theta}$, y la inducción da $\abs{\sin n\theta} \leq
n\abs{\sin\theta}$. Por lo tanto $\abs{U_{n-1}} \leq n$ en $\intoo{-1}1$
y $\abs{T_n'} = n\abs{U_{n-1}} \leq n^2$ allí; en $\pm1$ el
límite se extiende tomando límites (o directamente: $U_{n-1}(1) = n$ de
la recurrencia, $U_n(1) = n + 1$ por inducción, y la paridad da
$U_{n-1}(-1) = (-1)^{n-1}n$). Así,$T_n'(1) = n^2$ y $T_n'(-1)
= (-1)^{n-1}n^2$: el límite $n^2$ se alcanza en los puntos finales.

\textbf{11.} Diferenciar $\sin\theta\,T_n'(\cos\theta) = n\sin
n\theta $(pregunta 9) respecto de $\theta$:
\[
\cos\theta\,T_n'(\cos\theta) - \sin^2\theta\,T_n''(\cos\theta)
= n^2\cos n\theta = n^2\,T_n(\cos\theta) .
\]
Con $x = \cos\theta$ y $\sin^2\theta = 1 - x^2$:$x\,T_n' - (1
- x^2)T_n'' = n^2T_n $ en $\intcc{-1}1$, por lo tanto en todas partes:
$(1 - x^2)y'' - xy' + n^2y = 0$ para $ y = T_n$. Verifique $T_2 =
2x^2 - 1$:$(1 - x^2)(4) - x(4x) + 4(2x^2 - 1) = 4 - 4x^2 - 4x^2
+ 8x^2 - 4 = 0$.

\textbf{12.} $\widetilde T_n$ es \omterm{def:b1:poly:def}{mónico} (pregunta 2) y
$\abs{\widetilde T_n} = 2^{1-n}\abs{T_n} \leq 2^{1-n}$ en
$\intcc{-1}1$, con $\widetilde T_n(y_k) = (-1)^k2^{1-n}$ en el
$n + 1$ puntos $y_k $(\cref{exo:b1:poly:10}): la norma es exactamente
$2^{1-n}$, obtenido con signos alternos.

\textbf{13.} $\widetilde T_n$ y $P$ son ambos \omterm{def:b1:poly:def}{mónico} de grado
$n$, por lo que los términos principales se cancelan:$\deg D \leq n - 1$. En$y_k$:
$D(y_k) = (-1)^k2^{1-n} - P(y_k)$ y $\abs{P(y_k)} \leq \norm
P_\infty < 2^{1-n}$ obligan a que el signo de $D(y_k)$ sea el de
$(-1)^k2^{1-n}$, estrictamente.

\textbf{14.} $D$ cambia de signo entre $y_{k+1}$ y $y_k$ para cada
$k = 0, \dots, n-1$: por la propiedad de valor intermedio,$D$ tiene un
raíz en cada uno de estos intervalos abiertos disjuntos por pares $n$---$n$
raíces distintas para un \omterm{def:b1:poly:def}{polinomio} distinto de cero de grado $\leq n - 1$,
imposible. Y $D = 0$ también es imposible (las normas difieren).
Contradicción: ningún \omterm{def:b1:poly:def}{mónico} $P$ de grado $n$ tiene $\norm P_\infty <
2^{1-n}$, que es el teorema de Chebyshev.

\textbf{15.} Ahora solo $\abs{P(y_k)} \leq 2^{1-n}$, por lo que $(-1)^k
D(y_k) = 2^{1-n} - (-1)^kP(y_k) \geq 2^{1-n} - \abs{P(y_k)} \geq
0$. Supongamos $D(y_k) = 0$ en un interior $y_k $($0 < k < n$): entonces
$P(y_k) = (-1)^k2^{1-n}$, por lo que $\abs P$ alcanza su supremo
$2^{1-n}$ en el punto interior $y_k$, de donde $P'(y_k) = 0$
(extremo interior); y $T_n'(y_k) = nU_{n-1}(y_k) = 0$ desde
$\sin(n\cdot\frac{k\pi}n) = 0$--- entonces $\widetilde T_n'(y_k) = 0$
también, y $D'(y_k) = 0$:$y_k$ es una raíz de $D$ de \omterm{def:b1:poly:derivative}{multiplicidad} en
menos $2$.

\textbf{16.} Cuente las raíces de $D$ con \omterm{def:b1:poly:derivative}{multiplicidad}. Sea $z$ el
número de puntos interiores $y_k$ con $D(y_k) = 0$(cada uno un doble
raíz, por la pregunta 15) y $e \in \{0, 1, 2\}$ el número de
puntos finales ($y_0$ o $y_n$) con $D = 0$(cada uno de ellos una raíz simple en
menos). Un espacio $(y_{k+1}, y_k)$ cuyos dos puntos finales tienen $D
\neq 0$ lleva signos estrictamente alternos, por lo tanto, un interior
raíz. Cada punto de fuga interior estropea como máximo sus dos
espacios adyacentes, cada punto final de fuga como máximo un espacio: al menos
$n - 2z - e$ los espacios todavía contribuyen con una raíz cada uno, todos distintos
de las raíces $ y $. Total: al menos $(n - 2z - e) + 2z + e = n$
raíces con \omterm{def:b1:poly:derivative}{multiplicidad}, para un \omterm{def:b1:poly:def}{polinomio} de grado $\leq n - 1$:
entonces $D = 0$ y $P = \widetilde T_n$. El minimizador es único.

\textbf{17.} El afín \omterm{def:b1:logic:map}{aplicación} $t \mapsto x = \frac{a+b}2 +
\frac{b-a}2\,t $ es una biyección $\intcc{-1}1 \to \intcc ab$. Si$P $
es \omterm{def:b1:poly:def}{mónico} de grado $n$, entonces $Q(t) = P(x(t))$ es un \omterm{def:b1:poly:def}{polinomio} en
$t$ con coeficiente principal $\bigl(\frac{b-a}2\bigr)^n$, y

se diferencian de él por un grado bajo \omterm{def:b1:poly:def}{polinomio} obligado a cambiar de signo
$n$ veces --- una raíz más de la que permite su grado. el
Por lo tanto, el patrón de equioscilación no es una curiosidad sino la
certificado de optimidad, y el caso de igualdad se agudiza
conteo de raíces con multiplicidades. La sustitución $x =
\cos\theta $ merece la última palabra: transporta lo rígido,
mundo discreto de \omterm{def:b1:poly:def}{polinomios} en el mundo periódico de

cerrando el bucle.
