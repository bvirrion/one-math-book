\begin{solution}{exo:b1:matrices:9}
$J^2 = nJ$(cada entrada de $J^2$ suma $n$ unos). Buscar $M^{-1} =
\alpha I + \beta J $:
\[
(aI + bJ)(\alpha I + \beta J)
= a\alpha\, I + (a\beta + b\alpha + nb\beta)\, J .
\]
Esto es igual a $I$ y si $a\alpha = 1$ y $a\beta + b\alpha + nb\beta =
0$, es decir, $\alpha = \frac1a$ y $\beta(a + nb) = -\frac ba$. Si $a
\neq 0$ y $ a + nb \neq 0$:
\[
M^{-1} = \frac 1a I - \frac{b}{a(a + nb)}\, J .
\]
Por el contrario, si $a = 0$:$M = bJ$ tiene rango $\leq 1 < n $(para $n \geq
2$): no invertible ($n = 1$ es el caso escalar). Si$a + nb = 0$:
el vector $v = (1, \dots, 1)^{\mathsf T}$ satisface $Mv = (a +
nb)v = 0$ con $ v \neq 0$: no invertible. Entonces $M \in GL_n \iff a
\neq 0$ y $ a + nb \neq 0$.
\end{solution}

\begin{solution}{exo:b1:matrices:11}

 \item Entrada, $(AD)_{ij} = a_{ij}\,d_j$ y $(DA)_{ij} =

\begin{solution}{pb:b1:matrices:1}
\textbf{1.} DivisiĂłn euclidiana de $X^n$ por el grado-$2$\omterm{def:b1:poly:def}{mĂłnico}
$D$(\cref{thm:b1:poly:division}): existen cociente y resto
y son Ăşnicos, y el resto tiene grado $\leq 1$:$X^n =
Q_n D + a_n X + b_n $. Para $n = 0$:$Q_0 = 0$,$(a_0, b_0) = (0,
1)$; para $n = 1$:$(a_1, b_1) = (1, 0)$.

\textbf{2.} Multiplica por $X$ y reduce $X^2 = D + sX - p$:
\[
X^{n+1} = X Q_n D + a_n X^2 + b_n X
= (X Q_n + a_n)\,D + (s\,a_n + b_n)\,X - p\,a_n .
\]
La Ăşltima expresiĂłn tiene forma de resto (grado $\leq 1$), por lo que por
unicidad $a_{n+1} = s a_n + b_n$ y $b_{n+1} = -p a_n$.
Al sustituir $b_{n+1} = -pa_n$ en $a_{n+2} = s a_{n+1} +
b_{n+1}$ se obtiene $a_{n+2} = s\,a_{n+1} - p\,a_n $.

\textbf{3.} EvalĂşe $X^n = Q_n D + a_n X + b_n$ en las raĂ­ces:
$\lambda^n = a_n\lambda + b_n$ y $\mu^n = a_n\mu + b_n$.
Restar y dividir por $\lambda - \mu \neq 0$:
\[
a_n = \frac{\lambda^n - \mu^n}{\lambda - \mu},
\qquad
b_n = \lambda^n - a_n\lambda
= \frac{\lambda\mu^n - \mu\lambda^n}{\lambda - \mu} .
\]

\textbf{4.} En la doble raĂ­z: $\lambda^n = a_n\lambda + b_n$.
Diferenciando la identidad, $nX^{n-1} = Q_n'\,(X - \lambda)^2
+ 2Q_n\,(X - \lambda) + a_n $, y evaluando en $\lambda$:$a_n
= n\lambda^{n-1}$; luego $b_n = \lambda^n - n\lambda^{n} = (1 -
n)\lambda^{n}$.

\textbf{5.} Para $P = \sum_i p_i X^i$ y $Q = \sum_j q_j X^j$,
\[
P(M)\,Q(M) = \sum_{i,j} p_i q_j M^{i+j} = (PQ)(M),
\]
porque las potencias de la matriz Ăşnica $M$ conmutan entre sĂ­
(las sumas son claras por linealidad). Si $D(M) = 0$, sustituyendo $M$
en $X^n = Q_n D + a_n X + b_n$ da $M^n = Q_n(M)\,D(M) + a_n
M + b_n I = a_n M + b_n I $.

\textbf{6.} Productos directos:
\[
A^2 = \begin{pmatrix}
a^2 + bc & b(a + d)\\
c(a + d) & d^2 + bc
\end{pmatrix},
\qquad
s A = \begin{pmatrix}
a(a+d) & b(a+d)\\
c(a+d) & d(a+d)
\end{pmatrix},
\]
entonces $A^2 - sA$ tiene cero entradas fuera de la diagonal y entradas diagonales
$a^2 + bc - a^2 - ad = bc - ad = -p$:$A^2 - sA + pI_2 = 0$.

\textbf{7.} Con $A' = \begin{pmatrix} a' & b'\\ c' &
d'\end{pmatrix}$, ampliando $p(AA') = (aa' + bc')(cb' + dd') -
(ab' + bd')(ca' + dc')$: los tĂ©rminos$ aa'cb'$ y $ab'ca'$ se cancelan,
los tĂ©rminos $bc'dd'$ y $bd'dc'$ se cancelan, y lo que queda es
\[
aa'dd' - bca'd' + bcb'c' - adb'c'
= (ad - bc)(a'd' - b'c') = p(A)\,p(A').
\]
Si $p \neq 0$, Cayley--Hamilton da $A\,\bigl(\tfrac1p(sI_2 -
A)\bigr) = \tfrac1p(sA - A^2) = I_2$, de donde lo inverso (y
\cref{prop:b1:matrices:ring} lo hace de dos caras). Si $p = 0$ y
$A$ eran invertibles, la multiplicatividad da $1 = p(I_2) =
p(A)\,p(A^{-1}) = 0$: imposible. Entonces $A \in GL_2 \iff p \neq 0$.

\textbf{8.} $s = 3$,$p = 2$,$D = X^2 - 3X + 2 = (X - 1)(X -
2)$:$\lambda = 2$,$\mu = 1$, entonces $a_n = 2^n - 1$ y $ b_n = 2 -
2^n $(pregunta 3). Por lo tanto
\[
A^n = (2^n - 1)A + (2 - 2^n)I
= \begin{pmatrix} 1 & 2^n - 1\\ 0 & 2^n \end{pmatrix}.
\]
Verificar: $A^2 = \begin{pmatrix} 1 & 3\\ 0 & 4\end{pmatrix}$ ambos
por la fĂłrmula y elevando al cuadrado directamente.

\textbf{9.} $s = 4$,$p = 3\cdot1 - 1\cdot(-1) = 4$:$D = X^2 -
4X + 4 = (X - 2)^2$, doble raĂ­z $\lambda = 2$. Pregunta 4:$a_n
= n\,2^{n-1}$,$b_n = (1 - n)2^n$, entonces
\[
A^n = n\,2^{n-1}A + (1 - n)2^n I
= 2^{n-1}\begin{pmatrix} n + 2 & n\\ -n & 2 - n
\end{pmatrix}.
\]
En $n = 2$:$2\begin{pmatrix} 4 & 2\\ -2 & 0\end{pmatrix} =
\begin{pmatrix} 8 & 4\\ -4 & 0 \end{pmatrix}$, que es $A^2$
calculado directamente.

\textbf{10.} InducciĂłn: $F^1 = \begin{pmatrix} F_2 & F_1\\ F_1 &
F_0\end{pmatrix}$, y
\[

AquĂ­ $s = 1$,$p = -1$,$D = X^2 - X - 1$ con raĂ­ces $\varphi,
\psi $($\varphi - \psi = \sqrt5$,$\varphi\psi = -1$). el
