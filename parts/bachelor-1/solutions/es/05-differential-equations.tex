\begin{solution}{exo:b1:diffeq:1}
Homogéneo: $y_h = \lambda\,\eu^{-2x}$. Particular: pruebe $y_p =
c\,\eu^{3x}$($3$ no es una raíz de $r + 2$):$3c + 2c = 1$,$ c =
\frac15$. Solución general:$ y = \frac{\eu^{3x}}{5} +
\lambda\,\eu^{-2x}$. Con$ y(0) = 1$:$\frac15 + \lambda = 1$,
$\lambda = \frac45$, entonces $ y = \frac{\eu^{3x} + 4\,\eu^{-2x}}{5}$.
\end{solution}

\begin{solution}{exo:b1:diffeq:2}
En $\intoo{0}{+\infty}$, divida entre $x$:$ y' - \frac{2}{x}\,y = x^2$.
Aquí $A(x) = -2\ln x$,$\eu^{-A(x)} = x^2$: soluciones homogéneas
$\lambda x^2$. \omterm{thm:b1:diffeq:voc}{Variación de constantes}:$\mu'(x) = x^2 \cdot x^{-2} =
1$, entonces $\mu = x + \lambda $ y
\[
y(x) = x^3 + \lambda x^2, \qquad \lambda \in \R .
\]
\emph{Controlar:}$x(3x^2 + 2\lambda x) - 2(x^3 + \lambda x^2) = x^3$.
\end{solution}

\begin{solution}{exo:b1:diffeq:4}
Homogéneo: raíces $\pm 1$,$y_h = \lambda\,\eu^x + \mu\,\eu^{-x}$.
Particular con polinomio en el lado derecho ($\gamma = 0$ no es raíz):
$y_p = ax^2 + bx + c$; sustituyendo,$2a - (ax^2 + bx + c) = x^2$
da $a = -1$,$b = 0$,$c = 2a = -2$:$y_p = -x^2 - 2$. generales
solución $ y = -x^2 - 2 + \lambda\eu^x + \mu\eu^{-x}$.

Cauchy: $ y(0) = -2 + \lambda + \mu = 0$ y $ y'(0) = \lambda - \mu =
1$:$\lambda = \frac32$,$\mu = \frac12$. entonces
$ y = -x^2 - 2 + \frac{3\eu^x + \eu^{-x}}{2}$.
\end{solution}

\begin{solution}{exo:b1:diffeq:5}
$a(x) = \tan x$,$A(x) = -\ln(\cos x)$(válido:$\cos > 0$ en el
intervalo), $\eu^{-A} = \cos x$: soluciones homogéneas $\lambda\cos x$.
\omterm{thm:b1:diffeq:voc}{Variación de constantes}: $\mu'(x) = \sin 2x \cdot \frac{1}{\cos x} =
2\sin x $, entonces $\mu = -2\cos x + \lambda $ y
\[
y(x) = -2\cos^2 x + \lambda \cos x .
\]
\emph{Controlar:} $ y' = 4\cos x \sin x - \lambda\sin x$ y $ y\tan x =
-2\cos x\sin x + \lambda \sin x $; su suma es $2\cos x\sin x =
\sin 2x $, según sea necesario.
\end{solution}

\begin{solution}{exo:b1:diffeq:6}
$\chi(r) = r^2 - 4r + 3 = (r-1)(r-3)$:$\gamma = 1$ es una raíz simple
($m = 1$). Pruebe $y_p = x(ax + b)\,\eu^x$. Con $u = ax^2 + bx $,
\[
y_p'' - 4y_p' + 3y_p = \bigl(u'' + (2 - 4)u' + \chi(1) u\bigr)\eu^x
= \bigl(2a - 2(2ax + b)\bigr)\eu^x .
\]
Identificarse con $(2x + 1)\eu^x$:$-4a = 2$ y $2a - 2b = 1$, por lo que $a =
-\frac12$,$b = -1$. Solución general:
\[
y = -\Bigl(\frac{x^2}{2} + x\Bigr)\eu^{x} + \lambda\,\eu^{x} +
\mu\,\eu^{3x}, \qquad (\lambda, \mu) \in \R^2 .
\]
\end{solution}

\begin{solution}{exo:b1:diffeq:7}
Homogéneo: $y_h = \lambda\cos 2x + \mu\sin 2x$.

Lado derecho $x $($\gamma = 0$ no es raíz):$y_1 = ax + b$ con
$4(ax + b) = x$:$y_1 = \frac x4$.

Lado derecho $\sin 2x = \Im(\eu^{2\iu x})$,$2\iu $ raíz simple de
$r^2 + 4$: prueba $z = c\,x\,\eu^{2\iu x}$; luego $z'' + 4z =
4\iu c\,\eu^{2\iu x}$, igual a$\eu^{2\iu x}$ para $ c =
\frac{1}{4\iu} = -\frac{\iu}{4}$. Entonces $z = -\frac{\iu x}{4}(\cos 2x +
\iu \sin 2x)$ y $ y_2 = \Im(z) = -\frac{x\cos 2x}{4}$.

Por superposición:
\[
y = \frac{x}{4} - \frac{x\cos 2x}{4} + \lambda\cos 2x + \mu\sin 2x .
\]
\end{solution}

\begin{solution}{exo:b1:diffeq:8}
La ecuación $T' + kT = 20k$ tiene solución particular constante $20$ y
soluciones homogéneas $\lambda\eu^{-kt}$:$T(t) = 20 +
\lambda\,\eu^{-kt}$, y $T(0) = 80$ da $\lambda = 60$:
\[
T(t) = 20 + 60\,\eu^{-kt} .
\]
$T(10) = 50$:$\eu^{-10k} = \frac12$, entonces $k = \frac{\ln 2}{10}$. entonces
$T(t) = 25$ requiere $\eu^{-kt} = \frac{5}{60} = \frac{1}{12}$, es decir
\[
t = \frac{\ln 12}{k} = 10\,\frac{\ln 12}{\ln 2} \approx 35.8
\text{ minutes.}
\]
\end{solution}

\begin{solution}{exo:b1:diffeq:9}
\begin{enumerate}
 \item $\chi(r) = r^2 + 2\varepsilon r + 1$,$\Delta = 4(\varepsilon^2
 - 1)$.
 Para $\varepsilon \in \intco{0}{1}$: raíces$-\varepsilon \pm
 \iu\sqrt{1 - \varepsilon^2}$, entonces
 $ y = \eu^{-\varepsilon t}\bigl(\lambda\cos\omega t +
 \mu\sin\omega t\bigr)$ con $\omega = \sqrt{1 -
 \varepsilon^2}$.
 Para $\varepsilon = 1$: doble raíz$-1$,$ y = (\lambda + \mu
 t)\,\eu^{-t}$.
 Para $\varepsilon > 1$: raíces reales $r_\pm = -\varepsilon \pm
 \sqrt{\varepsilon^2 - 1}$, tanto$< 0$ como $ y =
 \lambda\eu^{r_+t} + \mu\eu^{r_-t}$.
 \item Para $\varepsilon \in \intoo{0}{1}$:$\abs y \leq
 \eu^{-\varepsilon t}(\abs\lambda + \abs\mu) \to 0$. Para
 $\varepsilon = 1$:$(\lambda + \mu t)\eu^{-t} \to 0$
 (exponencial vence al polinomio,
 \cref{prop:b1:functions:powerrules}). Para $\varepsilon > 1$:
 ambos exponenciales decaen desde $r_\pm < 0$(de hecho
 $\sqrt{\varepsilon^2 - 1} < \varepsilon$). Para $\varepsilon =
 0$:$ y = \lambda\cos t + \mu\sin t $ tiene amplitud constante
 $\sqrt{\lambda^2 + \mu^2} \neq 0$a menos que $ y = 0$.
 \item Escriba $\lambda\cos\omega t + \mu\sin\omega t =
 R\cos(\omega t - \varphi)$ con $ R = \sqrt{\lambda^2 + \mu^2}
 > 0$. Los ceros de $ y $ son los de $\cos(\omega t -
 \varphi)$(el factor $\eu^{-\varepsilon t}$ nunca
 desaparece): $\omega t - \varphi \equiv \frac\pi2 \pmod \pi$,
 una progresión aritmética con espacio $\frac{\pi}{\omega} =
 \frac{\pi}{\sqrt{1 - \varepsilon^2}}$.
\end{enumerate}
\end{solution}

\begin{solution}{exo:b1:diffeq:10}
Configure $x = y = 0$:$2f(0) = 2f(0)^2$ y $f(0) \neq 0$ da $f(0) =
1$. Arreglar $x $ y diferenciar la ecuación dos veces con respecto a$ y $:
\[
f''(x+y) + f''(x-y) = 2 f(x) f''(y) .
\]
Configuración $ y = 0$:$\;2f''(x) = 2 f(x) f''(0)$, es decir
\[
f''(x) = c\,f(x), \qquad c = f''(0).
\]
\emph{Case $c = \omega^2 > 0$:}$f(x) = \lambda\cosh\omega x +
\mu\sinh\omega x $;$f(0) = 1$ da $\lambda = 1$. Enchufándose al
ecuación funcional y usando las fórmulas de suma
(\cref{prop:b1:functions:hyprules}), la ecuación fuerza a $\mu = 0$
(compare los coeficientes de $\sinh\omega x \sinh\omega y $ o
evaluar en $x = y $):$f = \cosh\omega x$, que sí satisface
$\cosh(x+y) + \cosh(x-y) = 2\cosh x\cosh y $.

\emph{Case $c = -\omega^2 < 0$:} de manera similar $f(x) = \cos\omega x$
($\omega \neq 0$), que satisface la ecuación.

\emph{Caso $c = 0$:}$f$ afín con $f(0) = 1$:$f(x) = 1 + \mu x$;
la ecuación fuerza $\mu = 0$, excluida ($f$ no constante).

Conclusión: las soluciones son $f(x) = \cos\omega x$ y $f(x) =
\cosh\omega x $,$\omega > 0$.
\end{solution}

\begin{solution}{exo:b1:diffeq:11}
Configure $z(t) = y(\eu^t)$, de modo que $ y(x) = z(\ln x)$ sea $x > 0$. entonces
\[
y'(x) = \frac{z'(\ln x)}x,
\qquad
y''(x) = \frac{z''(\ln x) - z'(\ln x)}{x^2} ,
\]
y sustituyendo en la ecuación:
\[
x^2y'' - xy' + y = \bigl(z'' - z'\bigr) - z' + z
= z'' - 2z' + z = 0 .
\]
\omterm{def:b1:diffeq:linear2}{Polinomio característico} $(r - 1)^2$: doble raíz $1$, entonces $z(t) =
(\lambda + \mu t)\,\eu^t $ y, de nuevo en la variable $x = \eu^t $:
\[
y(x) = (\lambda + \mu\ln x)\,x,
\qquad \lambda, \mu \in \R .
\]
\end{solution}

\begin{solution}{exo:b1:diffeq:12}
\begin{enumerate}
 \item En forma normalizada $ y' - \frac2x\,y = 0$ en cada intervalo:
 $A(x) = -2\ln\abs x$, por lo que las soluciones son $ y = a x^2$ en
 $\intoo0{+\infty}$ y $ y = b x^2$ en $\intoo{-\infty}0$,
 con constantes independientes
 (\cref{thm:b1:diffeq:homogeneous1}).
 \item Sea $ y = ax^2$ para $x \geq 0$ y $bx^2$ para $x < 0$. encendido
 cada media línea abierta $ y $ es diferenciable con $xy' = 2y$.
 En $0$: los cocientes de diferencias $\frac{y(h) - y(0)}h =
 ah$ o $bh$ tienden a $0$, por lo que $ y'(0) = 0$ existe, y el
 La ecuación en $x = 0$ lee $0 \cdot y'(0) = 2y(0) = 0$:
 satisfecho. Entonces $ y $resuelve la ecuación en todo $\R$.
 \item Las soluciones en $\R$ son exactamente estas funciones pegadas: a
 \emph{dos} familia de parámetros para una ecuación de primer orden.
 No hay contradicción con \cref{thm:b1:diffeq:voc},
 cuyas hipótesis fallan aquí: escrito como $ y' + a(x)y = 0$,
 el coeficiente $a(x) = -\frac2x$ no es continuo en $0$
 --- de hecho no está definido --- entonces $\R$ no es un intervalo en
 que se aplica el teorema. La singularidad en $0$
 desconecta las dos medias líneas, y el valor $ y(0) = 0$
 es forzado y no transmite ninguna información. cada cauchy
 El dato en $x_0 \neq 0$ determina la solución sólo en el
 media línea que contiene $x_0$.
\end{enumerate}
\end{solution}

\begin{solution}{pb:b1:diffeq:1}
\textbf{1.} $\chi(r) = r^2 + 2\lambda r + \omega_0^2$,$\Delta =
4(\lambda^2 - \omega_0^2) < 0$ para $0 < \lambda < \omega_0$: raíces
$-\lambda \pm \iu\omega_d$, entonces por
\cref{thm:b1:diffeq:homogeneous2}
\[
x(t) = \eu^{-\lambda t}\bigl(\lambda_1\cos\omega_d t +
\mu_1\sin\omega_d t\bigr), \qquad (\lambda_1, \mu_1) \in \R^2 .
\]
Para $\lambda = 0$:$x = \lambda_1\cos\omega_0 t +
\mu_1\sin\omega_0 t $. El crítico ($\lambda = \omega_0$) y
Los regímenes sobreamortiguados ($\lambda > \omega_0$) son los de
\cref{exo:b1:diffeq:9} (después del tiempo de reescalado): $(\lambda_1 +
\mu_1 t)\eu^{-\lambda t}$, respectivamente\ combinaciones de $\eu^{r_\pm t}$
con $r_\pm = -\lambda \pm \sqrt{\lambda^2 - \omega_0^2}$.

\textbf{2.} Subamortiguado: $\abs x \leq \eu^{-\lambda
t}(\abs{\lambda_1} + \abs{\mu_1}) \to 0$. Crítico:$(\lambda_1 +
\mu_1 t)\eu^{-\lambda t} \to 0$ ya que los exponenciales superan
polinomios (\cref{prop:b1:functions:powerrules}). Sobreamortiguado:
$r_- < r_+ = -\lambda + \sqrt{\lambda^2 - \omega_0^2} < 0$
porque $\sqrt{\lambda^2 - \omega_0^2} < \lambda$; ambos
decaimiento exponencial.

\textbf{3.} A lo largo de una solución de $(H)$, usando $x'' = -2\lambda x' -
\omega_0^2 x $:
\[
\mathcal E'(t) = x'x'' + \omega_0^2 x x'
= x'\bigl(-2\lambda x' - \omega_0^2 x\bigr) + \omega_0^2 xx'
= -2\lambda\,x'^2 \leq 0 .
\]
Si $x(t_0) = x'(t_0) = 0$ entonces $\mathcal E(t_0) = 0$;$\mathcal E$
es no negativo y no creciente, por lo que $\mathcal E \equiv 0$ en
$\intco{t_0}{+\infty}$, forzando a$x \equiv 0$ allí; para $t \leq
t_0$, ejecute el mismo argumento en $\tilde x(t) = x(2t_0 - t)$, que
resuelve la ecuación con amortiguación $-\lambda$ pero aún tiene
$\tilde{\mathcal E}(t_0) = 0$ y $\tilde{\mathcal E}' =
+2\lambda\tilde x'^2 \geq 0$ con $\tilde{\mathcal E} \geq 0$;
no negativo, no decreciente y cero en el extremo derecho de
$\intoc{-\infty}{t_0}$ significa cero en todo momento. Entonces $x \equiv 0$ en adelante
$\R$--- una prueba energética de unicidad, válida para todos los $\lambda
\geq 0$.

\textbf{4.}$x(t + T_d) = R\,\eu^{-\lambda t}\eu^{-\lambda
T_d}\cos(\omega_d t + 2\pi - \varphi) = \eu^{-\lambda T_d}x(t)$.
El factor de reducción por pseudoperíodo es $\eu^{-\delta}$ con
$\delta = \lambda T_d = \frac{2\pi\lambda}{\omega_d}$. Para
$\omega_0 = 1$,$\lambda = 0.1$:$\omega_d = \sqrt{0.99} =
0.99499$, entonces $\delta = \frac{0.62832}{0.99499} = 0.6315$: cada uno
swing mantiene $\eu^{-0.63} \approx 53\%$ de su amplitud.

\textbf{5.} El factor de amplitud es $\eu^{-\lambda t}$, que
decae en $\eu$ sobre $t = \frac1\lambda $. Ese intervalo contiene
$\frac{1/\lambda}{T_d} = \frac{\omega_d}{2\pi\lambda}$
pseudoperíodos. Para $\lambda \ll \omega_0$,$\omega_d \approx
\omega_0$ y este es $\approx \frac{\omega_0}{2\pi\lambda} =
\frac Q\pi $. Una cuerda de guitarra $Q = 300$ suena durante aproximadamente cien
períodos; un amortiguador de puerta $Q = 1$ no completa uno.

\textbf{6.} Sustituyendo $z\,\eu^{\iu\Omega t}$ en el lado izquierdo
lado da $z\,(-\Omega^2 + 2\iu\lambda\Omega +
\omega_0^2)\,\eu^{\iu\Omega t}$, que es igual a$A\,\eu^{\iu\Omega
t}$ exactamente para $z = \frac{A}{\omega_0^2 - \Omega^2 +
2\iu\lambda\Omega}$(el denominador es distinto de cero: es imaginario
parte es $2\lambda\Omega > 0$). Como los coeficientes son reales,
la parte real $x_p = \Re\bigl(z\eu^{\iu\Omega t}\bigr)$ resuelve el
ecuación con el lado derecho $\Re\bigl(A\eu^{\iu\Omega t}\bigr) =
A\cos\Omega t $.

\textbf{7.} Escribe $\omega_0^2 - \Omega^2 + 2\iu\lambda\Omega =
\sqrt{D}\,\eu^{\iu\varphi}$ con $ D = (\omega_0^2 - \Omega^2)^2 +
4\lambda^2\Omega^2$ y $\varphi \in \intoo0\pi $(el imaginario
la parte $2\lambda\Omega$ es positiva), por lo que $\tan\varphi =
\frac{2\lambda\Omega}{\omega_0^2 - \Omega^2}$. Entonces $z =
\frac{A}{\sqrt D}\eu^{-\iu\varphi}$ y
\[
x_p(t) = \Re\Bigl(\frac A{\sqrt D}\,\eu^{\iu(\Omega t -
\varphi)}\Bigr) = R\cos(\Omega t - \varphi),
\qquad R = \frac A{\sqrt D} .
\]

\textbf{8.} Como $\Omega \to 0^+$:$D \to \omega_0^4$, entonces $R \to
A/\omega_0^2$ y $\tan\varphi \to 0^+$ con $\varphi \in
\intoo0{\frac\pi2}$:$\varphi \to 0$. La masa sigue a la fuerza.
casi estáticamente, desplazado por fuerza/rigidez. Como $\Omega \to
+\infty $:$D \sim \Omega^4$, entonces $R \sim A/\Omega^2 \to 0$, y
$\varphi \to \pi $(el número complejo $\omega_0^2 - \Omega^2 +
2\iu\lambda\Omega$ pasa al segundo cuadrante con el argumento $\to
\pi $): la masa apenas se mueve, y en oposición de fase ---
domina la inercia.

\textbf{9.} Por \cref{thm:b1:diffeq:structure2}~(1), cada solución
es $x = x_p + x_h$ con $x_h$ resolviendo $(H)$; por la pregunta 2,$x_h(t)
\to 0$, entonces $x(t) - x_p(t) \to 0$: todas las soluciones convergen a la
mismo estado estacionario. Las condiciones iniciales sólo dan forma a lo transitorio.

\textbf{10.} Si $x$ es una solución periódica,$x - x_p = x_h$ es una
solución periódica de $(H)$ que tiende a$0$ en $+\infty $; un
función periódica con límite $0$ es idénticamente $0$(sus valores en
un período se repite para siempre, por lo que cada valor es un límite de un
subsecuencia que tiende a $0$). Por lo tanto $x = x_p$.

\textbf{11.} Aquí $\lambda = 1$,$\omega_0^2 = 2$,$\Omega = 1$,
$A = 1$:$z = \frac1{2 - 1 + 2\iu} = \frac{1 - 2\iu}5$, entonces
\[
x_p = \Re\Bigl(\frac{(1 - 2\iu)(\cos t + \iu\sin t)}5\Bigr)
= \frac{\cos t + 2\sin t}5 .
\]
Homogéneo: las raíces de $r^2 + 2r + 2$ son $-1 \pm \iu $:$x_h =
\eu^{-t}(C\cos t + S\sin t)$. Condiciones:$x(0) = \frac15 + C = 0$
da $C = -\frac15$; diferenciando,$x'(0) = \frac25 - C + S =
0$ da $S = C - \frac25 = -\frac35$. Por lo tanto
\[
x(t) = \underbrace{\frac{\cos t + 2\sin t}5}_{\text{steady}}
- \underbrace{\eu^{-t}\,\frac{\cos t + 3\sin
t}5}_{\text{transient}} ,
\]
el transitorio muriendo como $\eu^{-t}$.

\textbf{12.} Expandiendo, $g(u) = u^2 - 2(\omega_0^2 - 2\lambda^2)u
+ \omega_0^4 = (u - u_r)^2 + g(u_r)$ con $ u_r = \omega_0^2 -
2\lambda^2$ y
\[
g(u_r) = \omega_0^4 - u_r^2 = (\omega_0^2 - u_r)(\omega_0^2 +
u_r) = 2\lambda^2\bigl(2\omega_0^2 - 2\lambda^2\bigr) =
4\lambda^2(\omega_0^2 - \lambda^2) .
\]
Si $2\lambda^2 < \omega_0^2$, entonces $u_r > 0$ es un valor admisible.
Frecuencia al cuadrado: $g$ tiene un mínimo estricto allí, por lo que $R = A/\sqrt
g $ tiene un máximo estricto en $\Omega_r = \sqrt{u_r} =
\sqrt{\omega_0^2 - 2\lambda^2}$, con $R_{\max} = A/\sqrt{g(u_r)}
= \frac{A}{2\lambda\sqrt{\omega_0^2 - \lambda^2}}$.

\textbf{13.} $R(0) = A/\omega_0^2$, entonces
\[
\frac{R_{\max}}{R(0)}
= \frac{\omega_0^2}{2\lambda\sqrt{\omega_0^2 - \lambda^2}}
= \frac{\omega_0}{2\lambda}\cdot
\frac{\omega_0}{\sqrt{\omega_0^2 - \lambda^2}}
= Q\,\Bigl(1 - \frac{\lambda^2}{\omega_0^2}\Bigr)^{-1/2}
\geq Q .
\]
Para una amortiguación débil, el factor de corrección está cerca de $1$: \omterm{ex:b1:diffeq:oscillation}{resonancia}
multiplica el desplazamiento estático por esencialmente $Q$.

\textbf{14.} $V(\Omega)^2 = \frac{A^2 u}{g(u)}$ con $u =
\Omega^2$. Su derivada tiene el signo de $g(u) - u\,g'(u) =
(\omega_0^2 - u)^2 + 4\lambda^2 u - u\bigl(2(u - \omega_0^2) +
4\lambda^2\bigr) = (\omega_0^2 - u)^2 + 2u(\omega_0^2 - u) =
(\omega_0^2 - u)(\omega_0^2 + u)$, positivo para $u < \omega_0^2$
y negativo más allá: máximo estricto exactamente en $\Omega =
\omega_0$, para cada amortiguación. Allí $\tan\varphi $ explota con
$\varphi \in \intoo0\pi$: $\varphi = \frac\pi2$ y $x_p'(t) =
-R\Omega\sin(\Omega t - \frac\pi2) = R\Omega\cos(\Omega t)$ es
exactamente en sintonía con la fuerza: transferencia de potencia óptima.

\textbf{15.} $R = R_{\max}/\sqrt2 \iff g(u) = 2g(u_r) \iff (u -
u_r)^2 = g(u_r) = 4\lambda^2(\omega_0^2 - \lambda^2)$, dando
\[
u_\pm = u_r \pm 2\lambda\sqrt{\omega_0^2 - \lambda^2},
\qquad
\Omega_+^2 - \Omega_-^2 = 4\lambda\sqrt{\omega_0^2 - \lambda^2} .
\]
Luego $\Omega_+ - \Omega_- = \frac{\Omega_+^2 -
\Omega_-^2}{\Omega_+ + \Omega_-}$, y para $\lambda \ll \omega_0$
ambos $\Omega_\pm \approx \omega_0$:$\Omega_+ - \Omega_- \approx
\frac{4\lambda\omega_0}{2\omega_0} = 2\lambda$, entonces
$\frac{\omega_0}{\Omega_+ - \Omega_-} \approx
\frac{\omega_0}{2\lambda} = Q$. Medición del ancho de un pico \omterm{ex:b1:diffeq:oscillation}{resonancia}
mide su \omterm{pb:b1:diffeq:1}{factor de calidad}.

\textbf{16.}$Q = 10$;$\Omega_r = \sqrt{1 - 2(0.05)^2} =
\sqrt{0.995} = 0.9975$;$R_{\max} = \frac1{2 \times 0.05
\sqrt{1 - 0.0025}} = \frac1{0.1 \times 0.99875} = 10.01$; estático
respuesta $R(0) = 1$; ancho de banda $\approx 2\lambda = 0.1$. un alto
pico delgado de altura $\approx Q$ sobre una meseta de altura $1$.

\textbf{17.} Si $2\lambda^2 \geq \omega_0^2$ entonces $u_r \leq 0$ y
$g'(u) = 2(u - u_r) > 0$ para todos $u > 0$:$g$ aumenta estrictamente
en $\intoo0{+\infty}$, por lo que $R = A/\sqrt g$ disminuye estrictamente desde
$R(0) = A/\omega_0^2$: la respuesta es mayor en frecuencia cero
y no hay pico.

\textbf{18.} $\gamma = \iu\Omega$ no es una raíz de $r^2 +
\omega_0^2$(como $\Omega \neq \omega_0$), por lo que
\cref{met:b1:diffeq:particular} con $m = 0$ da $x_p =
\frac{A\cos\Omega t}{\omega_0^2 - \Omega^2}$(sustituto y
comprobar: $-\Omega^2 + \omega_0^2$ veces el coseno). generales
solución:
\[
x(t) = \frac{A\cos\Omega t}{\omega_0^2 - \Omega^2}
+ \lambda_1\cos\omega_0 t + \mu_1\sin\omega_0 t .
\]

\textbf{19.} $x(0) = 0$ fuerza a$\lambda_1 = -\frac A{\omega_0^2 -

\textbf{21.} Reparar $t$. Como $\Omega \to \omega_0$:
\[
\frac{2A}{\omega_0^2 - \Omega^2}\sin\Bigl(\frac{(\omega_0 -
\Omega)t}2\Bigr)
= \frac{2A}{\omega_0 + \Omega}\cdot
\frac{\sin\bigl(\frac{(\omega_0 - \Omega)t}2\bigr)}

= \frac{At}{2\omega_0} ,
\]
mientras que $\sin\bigl(\frac{(\omega_0 + \Omega)t}2\bigr) \to
\sin(\omega_0 t)$: el límite es $x_\infty(t) =
\frac{At\sin\omega_0 t}{2\omega_0}$. Comprobación directa: con $C =

\textbf{22.} Sin amortiguación la amplitud resonante crece linealmente
y sin límites; con amortiguación $\lambda > 0$ el crecimiento se satura
en $R_{\max} \approx Q\,\frac A{\omega_0^2}$. El mecanismo:
La disipación elimina energía a un ritmo que crece con la amplitud.
(pregunta 23), por lo que la acumulación se detiene exactamente cuando la amortiguación
burns energy as fast as the forcing supplies it.

\textbf{23.} Con $x_p' = -R\Omega\sin(\Omega t - \varphi)$, terminado
un periodo los promedios $\langle\cos^2\rangle =
\langle\sin^2\rangle = \frac12$ y $\langle\sin\cos\rangle = 0$
dar:
\[
\langle P_{\mathrm{diss}}\rangle
= 2\lambda\,R^2\Omega^2\,\langle\sin^2(\Omega t -
\varphi)\rangle = \lambda R^2\Omega^2 ;
\]
y expandiendo $\sin(\Omega t - \varphi) = \sin\Omega t\cos\varphi
- \cos\Omega t\sin\varphi $:
\[
\langle P_{\mathrm{in}}\rangle
= -AR\Omega\,\bigl\langle\cos\Omega t\,\sin(\Omega t -
\varphi)\bigr\rangle
= AR\Omega\,\frac{\sin\varphi}2 .
\]
Desde $\sin\varphi = \frac{2\lambda\Omega}{\sqrt D} =
\frac{2\lambda\Omega R}A $, esto es $\frac{AR\Omega}2 \cdot
\frac{2\lambda\Omega R}A = \lambda R^2\Omega^2$: inyectado y
Los poderes disipados se equilibran exactamente: la propiedad definitoria de un
régimen estable.
