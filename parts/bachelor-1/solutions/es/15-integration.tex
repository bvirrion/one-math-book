\section*{Capítulo \ref{ch:b1:integration} --- Integración en un segmento}
\begin{solution}{exo:b1:integration:1}
$\dfrac{1}{x^2-4} = \dfrac{1/4}{x - 2} - \dfrac{1/4}{x+2}$
(encubrimiento), entonces
\[
\int_0^1 \frac{\dd x}{x^2 - 4}
= \frac14\Bigl[\ln\abs{x-2} - \ln\abs{x+2}\Bigr]_0^1
= \frac14\Bigl(\ln\frac{1}{3} - \ln 1\Bigr) = -\frac{\ln 3}{4}.
\]

Por partes ($u' = 1$,$v = \ln t$):$\int_1^{\eu} \ln t\,\dd t =
\bigl[t\ln t\bigr]_1^{\eu} - \int_1^{\eu} \dd t = \eu - (\eu - 1) =
1$.

Por partes ($u' = \sin t$,$v = t$):$\int_0^\pi t\sin t\,\dd t =
\bigl[-t\cos t\bigr]_0^\pi + \int_0^\pi \cos t\,\dd t = \pi + 0 =
\pi$.
\end{solution}

\begin{solution}{exo:b1:integration:2}
Sustitución $u = t^2 + 1$,$\dd u = 2t\,\dd t$:
\[
\int_0^1 \frac{t\,\dd t}{(t^2+1)^2}
= \frac12 \int_1^2 \frac{\dd u}{u^2}
= \frac12\Bigl[-\frac1u\Bigr]_1^2 = \frac14 .
\]
Con $u = \sin t$:$\int_0^{\pi/2} \cos^3 t\,\dd t = \int_0^{\pi/2}
(1 - \sin^2 t)\cos t\,\dd t = \bigl[\sin t - \frac{\sin^3
t}{3}\bigr]_0^{\pi/2} = 1 - \frac13 = \frac23$.
\end{solution}

\begin{solution}{exo:b1:integration:3}
$u_n = \dfrac1n \sum_{k=1}^{n} \dfrac{1}{1 + (k/n)^2}$: una suma de Riemann
de $x \mapsto \frac{1}{1+x^2}$ en $\intcc{0}{1}$, entonces $u_n \to
\int_0^1 \frac{\dd x}{1+x^2} = \arctan 1 = \dfrac\pi4$.

$\ln v_n = \dfrac1n \sum_{k=1}^{n} \ln\dfrac{n+k}{n} = \dfrac 1n
\sum_{k=1}^n \ln\Bigl(1 + \dfrac kn\Bigr) \to \int_0^1 \ln(1+x)\,\dd
x = \bigl[(1+x)\ln(1+x) - x\bigr]_0^1 = 2\ln 2 - 1$. Por lo tanto $v_n \to
\eu^{2\ln 2 - 1} = \dfrac 4\eu $. (Comprobación de la identificación:
$\frac{(2n)!}{n!\,n^n} = \prod_{k=1}^{n} \frac{n+k}{n}$.)
\end{solution}

\begin{solution}{exo:b1:integration:4}
El límite es $0$. Sean $M = \sup \abs f$ y $\varepsilon \in
\intoo{0}{1}$. Corte en $1 - \varepsilon $:
\[
\Bigl| \int_0^1 x^n f \Bigr|
\leq \int_0^{1 - \varepsilon} x^n \abs f + \int_{1-\varepsilon}^1
x^n \abs f
\leq M\,(1-\varepsilon)^n + M\varepsilon .
\]
Desde $(1 - \varepsilon)^n \to 0$(\cref{exo:b1:seq:3}), el limsup
del lado izquierdo es $\leq M\varepsilon$ por cada $\varepsilon$:
el \omterm{thm:b1:integration:def}{integral} tiende a $0$.
\end{solution}

\begin{solution}{exo:b1:integration:5}
$Q(\lambda) = \int_a^b (f + \lambda g)^2 = \int f^2 + 2\lambda \int
fg + \lambda^2 \int g^2 \geq 0$ para todos los $\lambda$. Si$\int g^2 =
0$, entonces $g = 0$(positividad estricta,
\cref{thm:b1:integration:props}~(4)) y la desigualdad es $0 \leq
0$. De lo contrario,$Q $ es una cuadrática genuina, en todas partes $\geq 0$: es
discriminante es $\leq 0$, es decir $\bigl(\int fg\bigr)^2 \leq \int
f^2 \int g^2$.

Igualdad si el discriminante desaparece si $Q(\lambda_0) = 0$ para
algunos $\lambda_0$, es decir $\int (f + \lambda_0 g)^2 = 0$, es decir (estricto
positividad nuevamente) $f = -\lambda_0 g$: la igualdad se cumple exactamente cuando
$f$ y $g$ son proporcionales.
\end{solution}

\begin{solution}{exo:b1:integration:6}
Sea $\Phi(a) = \int_a^{a+T} f$. Por el teorema fundamental
(\cref{thm:b1:integration:ftc}), $\Phi$ es \omterm{def:b1:derivative:def}{diferenciable} con
$\Phi'(a) = f(a + T) - f(a) = 0$: constante.

Para $x > 0$, escriba $x = nT + r$,$0 \leq r < T$($n = \lfloor x/T
\rfloor $). Chasles:
\[
\int_0^x f = n \int_0^T f + \int_{nT}^{nT + r} f,
\qquad
\Bigl| \int_{nT}^{nT+r} f \Bigr| \leq T \sup_{\intcc{0}{T}} \abs f
= C .
\]
Luego $\frac 1x \int_0^x f = \frac{nT}{x}\cdot\frac 1T \int_0^T f +
O\bigl(\frac 1x\bigr)$ y $\frac{nT}{x} \to 1$: el límite es
$\frac1T \int_0^T f$.
\end{solution}

\begin{solution}{exo:b1:integration:7}
Sea $g(x) = f(x) - x$: \omterm{def:b1:continuity:continuous}{continuo}, con
\[
\int_0^1 g = \int_0^1 f - \frac12 = 0 .
\]
Si $g$ nunca desapareciera, el teorema del valor intermedio forzaría una
signo constante (una función \omterm{def:b1:continuity:continuous}{continuo} en un \omterm{prop:b1:reals:intervals}{intervalo} que toma ambos signos
desaparece); diga $g > 0$. Luego, por estricta positividad
(\cref{thm:b1:integration:props}~(4) aplicado a $g > 0$, dando
$\int g > 0$): contradicción con $\int g = 0$. Entonces $g(c) = 0$ para
algunos $c$:$f(c) = c$.
\end{solution}

\begin{solution}{exo:b1:integration:8}
\begin{enumerate}
 \item Por partes con $u' = \sin t$,$v = \sin^{n-1} t$:
 \[
 W_n = \bigl[-\cos t\sin^{n-1}t\bigr]_0^{\pi/2}
 + (n-1)\int_0^{\pi/2} \cos^2 t\,\sin^{n-2} t\,\dd t
 = (n-1)(W_{n-2} - W_n),
 \]
 Entonces $nW_n = (n-1)W_{n-2}$. De$W_0 = \frac\pi2$,$W_1 = 1$:
 \[
 W_{2p} = \frac{(2p-1)(2p-3)\cdots 1}{(2p)(2p-2)\cdots 2}\,
 \frac{\pi}{2} = \frac{(2p)!}{4^p (p!)^2}\,\frac\pi2,
 \qquad
 W_{2p+1} = \frac{(2p)(2p-2)\cdots 2}{(2p+1)(2p-1)\cdots 3}
 = \frac{4^p (p!)^2}{(2p+1)!} .
 \]
 \item En $\intoo{0}{\frac\pi2}$,$0 < \sin t < 1$, por lo que $\sin^{n+1} <
 \sin^n$ y $(W_n)$ son (estrictamente) decrecientes, positivos.
 Intercalado con la recurrencia:
 \[
 \frac{n}{n+1} = \frac{W_{n+1}}{W_{n-1}}
 \leq \frac{W_{n+1}}{W_n} \leq 1
 \quad\implies\quad \frac{W_{n+1}}{W_n} \to 1 .
 \]
 Invariante: $a_n = (n+1)W_{n+1}W_n$ satisface $a_n = a_{n-1}$
 por la recurrencia $(n+1)W_{n+1} = nW_{n-1}$, entonces $a_n = a_0 =
 1 \cdot W_1 W_0 = \frac\pi2$. entonces
 \[
 n W_n^2 \sim (n+1) W_{n+1} W_n = \frac\pi2
 \quad\implies\quad
 W_n \sim \sqrt{\frac{\pi}{2n}} .
 \]
\end{enumerate}
\end{solution}

\begin{solution}{exo:b1:integration:9}
\begin{enumerate}
 \item En $\intoo{0}{\pi}$:$x > 0$,$a - bx = b(\frac ab - x) =
 b(\pi - x) > 0$ y $\sin x > 0$, por lo que el integrando es $> 0$
 y $I_n > 0$(positividad estricta). Cota: encendido
 $\intcc{0}{\pi}$,$x \leq \pi $ y $a - bx \leq a$, por lo que $P
 \leq \frac{\pi^n a^n}{n!}$ y $ I_n \leq
 \pi\,\frac{(\pi a)^n}{n!}$, que tiende a$0$(el
 factorial supera al término geométrico: es el término general
 de la serie exponencial convergente, cf.\
 \cref{ex:b1:seq:e}); en particular $I_n < 1$ para $n$ grande.
 \item Expandir $x^n(a - bx)^n = \sum_{j=0}^{n} \binom nj a^{n-j}
 (-b)^j x^{n+j}$: entonces $P = \frac{1}{n!}\sum_j c_j x^{n+j}$
 con número entero $c_j$. Entonces $P^{(k)}(0) = 0$ para $k < n$
 (valoración) y, para $n \leq k \leq 2n$,$P^{(k)}(0) =
 \frac{k!}{n!} c_{k-n}$, un número entero desde $n! \mid k!$.
 Además $P(\pi - x) = P(x)$(sustituto:$\pi - x$ swaps
 los factores, usando $a - b(\pi - x) = bx$), por lo que $P^{(k)}(\pi)
 = \pm P^{(k)}(0)$: también son números enteros.
 \item Con $Q = P - P'' + P^{(4)} - \dots $(finito: $P$ tiene grado
 $2n$):$Q + Q'' = P$, y
 \[
 \bigl(Q'\sin x - Q\cos x\bigr)' = (Q + Q'')\sin x = P\sin x .
 \]
 Por lo tanto $I_n = \bigl[Q'\sin x - Q\cos x\bigr]_0^{\pi} = Q(\pi)
 + Q(0)$, una suma de valores $P^{(2k)}$ en $0$ y $\pi$: un
 entero por (2).
 \item Para $n$ grande,$I_n$ es un número entero con $0 < I_n < 1$:
 imposible. La suposición $\pi = \frac ab$ falla: $\pi$ es
 irracional.
\end{enumerate}
\end{solution}

\begin{solution}{exo:b1:integration:10}
Caso $C^1$: por partes,
\[
\int_a^b f(t)\sin\lambda t\,\dd t
= \Bigl[-f(t)\frac{\cos\lambda t}{\lambda}\Bigr]_a^b
+ \frac{1}{\lambda}\int_a^b f'(t)\cos\lambda t\,\dd t,
\]
acotado en valor absoluto por $\frac{2\sup\abs f + (b - a)\sup\abs{f'}}
{\lambda} \to 0$.

Caso \omterm{def:b1:continuity:continuous}{Continuo}: deje $\varepsilon > 0$ y elija un \omterm{def:b1:integration:step}{función de paso}
$\varphi$ con $\abs{f - \varphi} \leq \varepsilon$
(\cref{thm:b1:integration:approx} proporciona a $\varphi \leq f \leq
\psi $ el espacio $\leq\varepsilon $; tome $\varphi$). entonces
\[
\Bigl|\int_a^b f\sin\lambda t\Bigr|
\leq \int_a^b \abs{f - \varphi}
+ \Bigl|\int_a^b \varphi \sin\lambda t\Bigr|
\leq (b-a)\varepsilon
+ \sum_i \abs{c_i}\,\Bigl|\int_{x_{i-1}}^{x_i} \sin\lambda t\,\dd
t\Bigr| ,
\]
y cada $\bigl|\int \sin \lambda t\bigr| = \bigl|\frac{\cos\lambda
x_{i-1} - \cos\lambda x_i}{\lambda}\bigr| \leq \frac{2}{\lambda}$:
el segundo término tiende a $0$. Por lo tanto, el límite es $\leq
(b-a)\varepsilon $ para cada $\varepsilon$: el límite es $0$.
\end{solution}

\begin{solution}{exo:b1:integration:11}
Sean $m = \min f$ y $M = \max f$, alcanzados por el valor extremo
teorema. Desde $g \geq 0$:$m\,g \leq fg \leq M\,g$, entonces por
monotonicidad
\[
m \int_a^b g \;\leq\; \int_a^b fg \;\leq\; M \int_a^b g .
\]
Si $\int_a^b g = 0$: positividad estricta
(\cref{thm:b1:integration:props}~(4)) fuerza a $g \equiv 0$, ambos
Los lados desaparecen y cualquier $c$ funciona. De lo contrario,$t = \frac{\int
fg}{\int g}$ se encuentra en $\intcc{m}{M} = f(\intcc{a}{b})$.
(\cref{thm:b1:continuity:evt,thm:b1:continuity:ivt}), entonces $t =
f(c)$ para algunos $c $.

El signo importa: en $\intcc{-1}{1}$ con $f(t) = g(t) = t$:$\int
fg = \int_{-1}^1 t^2 = \frac23$, mientras que $f(c)\int_{-1}^1 t\,\dd t
= 0$ para cada $c $.
\end{solution}

\begin{solution}{exo:b1:integration:12}
Si $f \equiv 0$ el reclamo es vacío (cada punto es un cero). entonces
supongamos $f \not\equiv 0$ y supongamos que tiene como máximo $n$ distintos
ceros en $\intoo{a}{b}$; sea $z_1 < \dots < z_m $($m \leq n$)
esos ceros donde $f$\emph{cambia de signo} (posiblemente ninguno). conjunto
$P(t) = \prod_{i=1}^{m}(t - z_i)$(producto vacío$= 1$), de
grado $m \leq n$. En cada subintervalo cortado por el $z_i$, ambos
$f$ y $P$ tienen signo constante y ambos invierten el signo al cruzar
algunos $z_i$: el producto $fP$ tiene un signo constante en todos
$\intoo{a}{b}$. Siendo \omterm{def:b1:continuity:continuous}{continuo}, no idénticamente cero, de
signo constante, tiene $\bigl|\int_a^b fP\bigr| > 0$(estricta
positividad aplicada a $\abs{fP}$). Pero $\int f P$ es lineal.
combinación de los momentos $\int f\,t^k$,$k \leq n$, todos cero:
contradicción. Por lo tanto, $f$ tiene al menos $n + 1$ ceros distintos en
$\intoo{a}{b}$.
\end{solution}

\begin{solution}{pb:b1:integration:1}
\textbf{1.} Vamos $N = \lceil 2c \rceil$. Para $n \geq N$:
$\frac{c^{n+1}/(n+1)!}{c^n/n!} = \frac{c}{n+1} \leq \frac12$, entonces
$0 < \frac{c^n}{n!} \leq \frac{c^N}{N!}\,2^{-(n - N)} \to 0$:
apretar.

\textbf{2.} Reparar $k$; inducción en $m$. Para $m = 0$:$\int_0^1
x^k = \frac{1}{k+1} = \frac{k!\,0!}{(k+1)!}$. Paso, por partes
($u = (1-x)^m$,$v' = x^k$):
\[
\int_0^1 x^k (1-x)^m \dd x
= \Bigl[\frac{x^{k+1}}{k+1}(1-x)^m\Bigr]_0^1
+ \frac{m}{k+1}\int_0^1 x^{k+1}(1-x)^{m-1}\dd x
= \frac{m}{k+1}\cdot\frac{(k+1)!\,(m-1)!}{(k+m+1)!} ,
\]
que es $\frac{k!\,m!}{(k+m+1)!}$.

\textbf{3.} $k = m = n$:$\int_0^1 (x(1-x))^n =
\frac{(n!)^2}{(2n+1)!} = \frac{1}{(2n+1)\binom{2n}{n}}$. desde
$x(1-x) \leq \frac14$ en $\intcc{0}{1}$, el \omterm{thm:b1:integration:def}{integral} es $\leq
4^{-n}$, de donde $\binom{2n}{n} \geq \frac{4^n}{2n+1}$ ---
consistente con $\binom{2n}{n}^{1/n} \to 4$
(\cref{pb:b1:seq:1}).

\textbf{4.} El integrando es \omterm{def:b1:continuity:continuous}{continuo}, $\geq 0$ y positivo.
en $\intoo{0}{1}$, por lo tanto no es idénticamente cero: su \omterm{thm:b1:integration:def}{integral} es
$> 0$(\cref{thm:b1:integration:props}~(4)). \omterm{def:b1:reals:bounds}{límite superior}:
$(x(1-x))^n \leq 4^{-n}$ y $g \leq \sup\abs g$, entonces
monotonía.

\textbf{5.} $A_0 = \eu - 1$;$A_1 = [x\eu^x]_0^1 - \int_0^1
\eu^x = \eu - (\eu - 1) = 1$. Por partes:$A_n = [x^n \eu^x]_0^1 -
n\int_0^1 x^{n-1}\eu^x = \eu - n A_{n-1}$. Límites: el integrando
es positivo, entonces $A_n > 0$; y $\eu^x \leq \eu$ da $A_n \leq
\eu\int_0^1 x^n = \frac{\eu}{n+1}$.

\textbf{6.} $A_0 = -1 + 1\cdot\eu$. Si $A_{n-1} = \alpha_{n-1} +
\beta_{n-1}\eu $ tiene entradas enteras, entonces
\[
A_n = \eu - n\alpha_{n-1} - n\beta_{n-1}\eu
= \underbrace{(-n\,\alpha_{n-1})}_{\alpha_n}
+ \underbrace{(1 - n\,\beta_{n-1})}_{\beta_n}\,\eu ,
\]
ambos números enteros.

\textbf{7.} Si $\eu = \frac pq$: $q A_n = q\alpha_n + p\beta_n
\in \Z $ y $0 < qA_n \leq \frac{q\eu}{n+1} < 1$ para $ n $ grande:
un número entero estrictamente entre $0$ y $1$--- imposible. Entonces $\eu
\notin \Q $. En \cref{exo:b1:seq:9} el entero atrapado era
$q!\,\frac pq - q!\,a_q$; aquí está $qA_n$: la misma trampa,
\omterm{thm:b1:integration:def}{integral} combustible.

\textbf{8.} $n = 0$:$R_0 = \int_0^1 \eu^t = \eu - 1$, entonces $\eu =
1 + R_0$. Por partes ($u = \eu^t$,$v = -\frac{(1-t)^{n+1}}{n+1}$):
\[
R_n = \frac{1}{n!}\Bigl(\Bigl[-\frac{(1-t)^{n+1}}{n+1}
\eu^t\Bigr]_0^1 + \frac{1}{n+1}\int_0^1 (1-t)^{n+1}\eu^t\Bigr)
= \frac{1}{(n+1)!} + R_{n+1} ,
\]
entonces la fórmula se propaga de $n$a$n + 1$. Límites:$1 \leq
\eu^t \leq \eu $ en $\intcc{0}{1}$ y $\int_0^1 (1-t)^n =
\frac{1}{n+1}$ dan $\frac{1}{(n+1)!} \leq R_n \leq
\frac{\eu}{(n+1)!}$.

\textbf{9.} $\sum_{k=0}^{9} \frac{1}{k!} = \frac{986410}{362880}
= 2.71828152\dots$, y
\[
\frac{1}{10!} = 2.76\cdot10^{-7} \leq R_9 \leq
\frac{2.75}{10!} = 7.58\cdot10^{-7} ,
\]
entonces $2.7182818 \leq \eu \leq 2.7182823$: con prueba,$\eu =
2.718282$ con seis decimales (valor verdadero $2.7182818\dots $).

\textbf{10.} $f(1 - x) = \frac{(1-x)^n x^n}{n!} = f(x)$. On
$\intoo{0}{1}$:$0 < x(1-x) \leq \frac14$, so $0 < f \leq
\frac{1}{4^n\,n!}$.

\textbf{11.} $x^n(1-x)^n = \sum_{j=0}^{n} (-1)^j\binom
nj\,x^{n+j}$, entonces $f = \frac{1}{n!}\sum_j c_j\,x^{n+j}$ con
$c_j \in \Z$. Por lo tanto,$f^{(k)}(0) = 0$ para $k < n$ o $k > 2n$,
y para $n \leq k \leq 2n$:$f^{(k)}(0) = \frac{k!}{n!}\,
c_{k-n}$, un número entero porque $n! \mid k!$. La simetría da
$f^{(k)}(1) = (-1)^k f^{(k)}(0) \in \Z$.

\textbf{12.} $b^n \pi^{2n-2k} = b^n\bigl(\frac
ab\bigr)^{\,n-k} = a^{\,n-k}\,b^{\,k} \in \Z $, por lo que $G(0) =
\sum_k (-1)^k a^{n-k} b^k f^{(2k)}(0)$ y también $G(1)$ son
números enteros por la pregunta 11.

\textbf{13.} En $\pi^2 G + G''$, el término $k$ de $\pi^2 G$
lleva $\pi^{2n-2k+2}f^{(2k)}$ y el término $j = k - 1$ de
$G''$lleva $(-1)^{k-1}\pi^{2n-2k+2}f^{(2k)}$: todo
cancela excepto $k = 0$ en la primera suma y $j = n$ en la
segundo, es decir
\[
G'' + \pi^2 G = b^n\bigl(\pi^{2n+2} f + (-1)^n f^{(2n+2)}\bigr)
= b^n \pi^{2n+2} f = \pi^2 a^n f
\]
($f$ tiene grado $2n$, por lo tanto $f^{(2n+2)} = 0$; y $b^n\pi^{2n} =
a^n $). entonces
\[
\bigl(G'\sin\pi x - \pi G\cos\pi x\bigr)'
= (G'' + \pi^2 G)\sin \pi x = \pi^2 a^n f\sin\pi x .
\]

\textbf{14.} Integración sobre $\intcc{0}{1}$:
\[
\pi^2 a^n \int_0^1 f\sin(\pi x)\,\dd x
= \bigl[G'\sin\pi x - \pi G\cos\pi x\bigr]_0^1
= \pi\bigl(G(1) + G(0)\bigr) ,
\]
entonces $\pi a^n \int_0^1 f\sin\pi x = G(0) + G(1) \in \Z$. encendido
$\intoo{0}{1}$,$f > 0$ y $\sin\pi x > 0$: el lado izquierdo es
positivo, entonces $G(0) + G(1) \geq 1$; y $\sin \leq 1$ con
la pregunta 10 lo limita a $\frac{\pi a^n}{4^n\,n!}$.

\textbf{15.} Por pregunta 1 (con $c = \frac a4$),
$\frac{\pi a^n}{4^n n!} \to 0$: para $n$ grande es $< 1$,
contradiciendo $G(0) + G(1) \geq 1$. Entonces no hay fracción $\frac ab $
es igual a $\pi^2$: teorema de Legendre. Si$\pi$ fuera racional,
$\pi^2$ también lo sería: entonces $\pi \notin \Q$ --- y la implicación
solo funciona de esta manera ($\sqrt2$ es irracional con racional
cuadrado), razón por la cual $\pi^2 \notin \Q$ es estrictamente más fuerte
que \cref{exo:b1:integration:9}.

\textbf{16.} Integralidad: preguntas 11--12 (criterio de valoración
\omterm{def:b1:derivative:def}{derivados}); pequeñez: preguntas 10 y 1; puente: preguntas
13--14 (el doble telescópico \omterm{thm:b1:integration:parts}{integración por partes}). sin
$\frac{1}{n!}$, los datos del punto final siguen siendo números enteros (aún más
fácilmente), pero el límite se convierte en $\frac{\pi a^n}{4^n}$, lo que
tiende a $0$ solo cuando $a < 4$--- y cada candidato tiene $a =
b\,\pi^2 > 9$. El factorial es exactamente lo que supera al
crecimiento geométrico $a^n$: sin factorial, sin teorema.

\textbf{17.} Para $a \leq 10$, el número entero $G(0) + G(1)$ es
positivo y como máximo $\pi\,(10/4)^n/n!$. En$n = 7$:
$2.5^7 = 610.35\dots$, por lo que el límite es $\frac{\pi \times
610.35}{5040} \approx 0.38 < 1$(en $n = 6$ todavía es
$1.07$): la contradicción aterriza en la séptima etapa,
explícitamente.

\textbf{18.} Dos integraciones por partes:
\[
\int_0^1 x(1-x)\sin\pi x\,\dd x
= \frac1\pi\int_0^1 (1 - 2x)\cos\pi x\,\dd x
= \frac{2}{\pi^2}\int_0^1 \sin\pi x\,\dd x
= \frac{2}{\pi^2}\cdot\frac{2}{\pi} = \frac{4}{\pi^3}
\]
(los términos entre corchetes desaparecen: $x(1-x)$ en $0, 1$ y $\sin\pi x$
en $0, 1$). Simbólicamente, el telescopio $n = 1$(sin suposiciones
en $\pi$) lee $\pi^3\int_0^1 f_1\sin\pi x = -(f_1''(0) +
f_1''(1))$ con $ f_1 = x(1 - x)$,$f_1'' = -2$: lado derecho $4$
--- los dos cálculos concuerdan.

\textbf{19.} $\eu^x$ resuelve$ y' = y $ y $\sin\pi x$ resuelve
$ y'' = -\pi^2 y $: ecuaciones lineales con coeficientes constantes,
entonces \omterm{thm:b1:integration:parts}{integración por partes} hace que el kernel vuelva a sí mismo y
mantiene todos los datos \omterm{def:b1:topology:closure}{límite} dentro de $\Z + \Z\eu $(resp.\ entero
\omterm{def:b1:poly:def}{polinomios} en $\pi^2$). Esa propiedad \omterm{def:b1:topology:closure}{cierre} es lo que
necesidades de la máquina. Refinado con pepitas adaptadas a varios puntos del
una vez, el mismo mecanismo produce el teorema de Hermite ($\eu$
trascendental, 1873) y el de Lindemann ($\pi$ trascendental,
1882) --- más allá de este volumen.

\textbf{20.} \omterm{def:b1:poly:def}{Polinomio} división (o multiplicar y verificar):
\[
x^4(1-x)^4 = (x^6 - 4x^5 + 5x^4 - 4x^2 + 4)(1 + x^2) - 4 .
\]
Integrando la identidad mostrada dividida por $1 + x^2$:
\[
\int_0^1 \frac{x^4(1-x)^4}{1+x^2}\dd x
= \Bigl(\frac17 - \frac46 + 1 - \frac43 + 4\Bigr)
- 4\arctan 1
= \frac{22}{7} - \pi ,
\]
usando $\frac17 - \frac23 + 1 - \frac43 + 4 = 3 + \frac17$ y
$\arctan 1 = \frac\pi4$.

\textbf{21.} El integrando es \omterm{def:b1:continuity:continuous}{continuo}, positivo en
$\intoo{0}{1}$: el \omterm{thm:b1:integration:def}{integral} es $> 0$, entonces $\pi < \frac{22}{7}$.
Además $\frac12 \leq \frac{1}{1+x^2} \leq 1$ en
$\intcc{0}{1}$ y $\int_0^1 (x(1-x))^4 = \frac{(4!)^2}{9!} =
\frac{1}{630}$(pregunta 3):
\[
\frac{1}{1260} \leq \frac{22}{7} - \pi \leq \frac{1}{630}
\quad\Longrightarrow\quad
3.14126 < \frac{22}{7} - \frac{1}{630} \leq \pi \leq
\frac{22}{7} - \frac{1}{1260} < 3.14207 .
\]

\textbf{22.} Módulo $x^2 + 1$:$x^2 \equiv -1$, entonces $x^{4m} =
(x^2)^{2m} \equiv 1$ y $(1 - x)^2 = 1 - 2x + x^2 \equiv -2x$,
por lo tanto $(1-x)^{4m} \equiv (-2x)^{2m} = 4^m (x^2)^m \equiv
(-4)^m $: el resto es la constante $(-4)^m $, y el
el cociente $Q_m$ tiene coeficientes enteros (división por un \omterm{def:b1:poly:def}{mónico}
entero \omterm{def:b1:poly:def}{polinomio}). Dividiendo la identidad por $1 + x^2$ y
integrando:
\[
J_m := \int_0^1 \frac{(x(1-x))^{4m}}{1+x^2}\dd x
= s_m + (-4)^m\,\frac{\pi}{4},
\qquad s_m = \int_0^1 Q_m \in \Q .
\]
Resolviendo para $\pi$: con $r_m = (-1)^{m+1}\,4^{\,1-m} s_m \in
\Q $,$\;\abs{\pi - r_m} = 4^{\,1-m} J_m \leq
4^{\,1-m}\cdot4^{-4m} = 4^{\,1-5m}$. Para $m = 1$:$s_1 =
\frac{22}{7}$,$r_1 = \frac{22}{7}$, cota $4^{-4} =
\frac{1}{256}$ --- preguntas 20--21 nuevamente.

\textbf{23.} $\bigl|\pi - \frac{22}{7}\bigr| = \frac{22}{7} -
\pi \approx 1.26\cdot10^{-3}$, dieciséis veces mejor que el
orden-$2$ punto de referencia $\frac{1}{7^2} \approx 2.0\cdot10^{-2}$
garantizado por Dirichlet (\cref{pb:b1:derivative:1}, pregunta 4);
y $\bigl|\pi - \frac{355}{113}\bigr| \approx 2.7\cdot10^{-7}$
supera a $\frac{1}{113^2} \approx 7.8\cdot10^{-5}$ por un factor
$\approx 300$. No hay contradicción con nada demostrado: Liouville
desigualdades \emph{límite inferior} errores de aproximación solo para
\omterm{pb:b1:logic:1}{números algebraicos}, y no hay ningún enlace para $\pi$ disponible en
este nivel --- $\pi$ es libre de aproximarse espectacularmente
bueno.

\textbf{24.} Lema: supongamos $x = \frac pq$ y $0 < \abs{a_n +
b_n x} \to 0$. Luego $\abs{a_n + b_n x} = \frac{\abs{q a_n + p
b_n}}{q}$, con $q a_n + p b_n $ un entero distinto de cero (distinto de cero
porque el valor absoluto es $> 0$): entonces $\abs{a_n + b_n x}
\geq \frac1q$ para cada $n $, contradiciendo la convergencia a
$0$. Instancias: pregunta 7 ($x = \eu$, $a_n = \alpha_n$,$b_n =
\beta_n $); \cref{exo:b1:seq:9} ($x = \eu$ nuevamente, con $a_q =
-q!\sum_{k \leq q}\frac{1}{k!}$,$b_q = q!$); y
\cref{pb:b1:derivative:1}, pregunta 1, es su forma geométrica.
En la Parte III y en \cref{exo:b1:integration:9} se ejecuta la trampa
\emph{adentro} la contradicción: asumir la racionalidad convierte
una expresión en un número entero, que luego el análisis comprime
en $\intoo{0}{1}$ --- el mismo principio, transpuesto.

\textbf{25.} (i) La integralidad vive en el cálculo del punto final de
\omterm{def:b1:poly:def}{polinomios} (preguntas 6, 11--12), pequeñez en los límites
$\sup$-monotonicidad da (preguntas 4, 10), el puente en
\omterm{thm:b1:integration:parts}{integración por partes} (preguntas 8, 13--14) --- las tres son
teoremas de este capítulo. (ii) El \omterm{thm:b1:integration:def}{integral} proporciona lo que el
El teorema del valor medio no pudo: una identidad \emph{exacto} entre
el objeto analítico y los datos aritméticos (igualdad, no sólo
una desigualdad con una incógnita $c$), por lo que la máquina
llega a $\pi^2$ mientras que \cref{pb:b1:derivative:1} alcanza solo
exponentes de aproximación. (iii) Extraído: $\eu \notin \Q$,
$\pi^2 \notin \Q$(de ahí $\pi \notin \Q$),$\eu = 2.718282$
certificados, $\frac{22}{7} - \frac{1}{630} \leq \pi \leq
\frac{22}{7} - \frac{1}{1260}$ y $\binom{2n}{n} \geq
\frac{4^n}{2n+1}$. (iv) Frontera: Hermite y Lindemann empujan
la misma máquina a la trascendencia; y Apéry (1979) ejecutaron el
trampa de enteros en $\zeta(3)$ --- la máquina todavía está produciendo
Matemáticas del siglo XX.
