\section*{Capítulo \ref{ch:b1:fractions} --- fracciones racionales}
\begin{solution}{exo:b1:fractions:1}
$\dfrac{1}{X^2 - 1}$: simple \omterm{def:b1:fractions:field}{polos}$\pm 1$; encubrimiento:
$\dfrac{1/2}{X-1} - \dfrac{1/2}{X+1}$.

$\dfrac{X}{X^2 - 3X + 2} = \dfrac{X}{(X-1)(X-2)}$: encubrimiento da
$\frac{1}{1-2} = -1$ en $1$ y $\frac{2}{2-1} = 2$ en $2$:
$\dfrac{-1}{X-1} + \dfrac{2}{X-2}$.

$\dfrac{X^2+1}{X(X-1)}$: el grado es $0$, por lo que es un número entero
parte: dividir, $X^2 + 1 = (X^2 - X) + (X + 1)$, entonces $F = 1 +
\frac{X+1}{X(X-1)}$. Encubrimiento del resto:$\frac{1}{-1} = -1$
en $0$,$\frac{2}{1} = 2$ en $1$:
\[
F = 1 - \frac{1}{X} + \frac{2}{X-1} .
\]
\end{solution}

\begin{solution}{exo:b1:fractions:2}
$\dfrac{1}{X(X^2+1)}$: forma $\frac aX + \frac{bX + c}{X^2 + 1}$.
Encubrimiento en $0$:$a = 1$. Límite de $xF$:$0 = a + b$, entonces $b = -1$.
Evaluación en $X = 1$:$\frac12 = 1 + \frac{c - 1}{2}$, entonces $c = 0$:
\[
\frac{1}{X(X^2+1)} = \frac 1X - \frac{X}{X^2+1} .
\]

$\dfrac{X^3}{X^2+X+1}$: división:$X^3 = (X^2+X+1)(X - 1) + 1$, entonces
\[
\frac{X^3}{X^2+X+1} = X - 1 + \frac{1}{X^2 + X + 1} ,
\]
ya en forma real descompuesta (la cuadrática tiene valores negativos)
discriminante).
\end{solution}

\begin{solution}{exo:b1:fractions:3}
$\dfrac{1}{X^2(X-1)}$: forma $\frac{a}{X^2} + \frac bX +
\frac{c}{X-1}$. Encubrimiento en el doble \omterm{def:b1:fractions:field}{polo}$0$:
$a = \bigl[\frac{1}{X-1}\bigr]_{0} = -1$. Encubrimiento en $1$:$c = 1$.
Límite de $xF$:$0 = b + c$, por lo que $b = -1$:
\[
\frac{1}{X^2(X-1)} = -\frac{1}{X^2} - \frac1X + \frac{1}{X-1} .
\]

$\dfrac{X+1}{(X-1)^3}$: con $Y = X - 1$, el numerador es $Y + 2$:
\[
\frac{Y + 2}{Y^3} = \frac{1}{Y^2} + \frac{2}{Y^3}
= \frac{1}{(X-1)^2} + \frac{2}{(X-1)^3} .
\]
\end{solution}

\begin{solution}{exo:b1:fractions:4}
Los \omterm{def:b1:fractions:field}{polos} son los cuartos \omterm{def:b1:complex:unity}{raíces de la unidad} $1, \iu, -1, -\iu$, todos
sencillo. Fórmula \omterm{def:b1:fractions:field}{polo simple} con $B' = 4X^3$: el coeficiente en $a$ es
$\frac{1}{4a^3} = \frac{a}{4a^4} = \frac a4$(usando $a^4 = 1$). Entonces,
sobre $\C$:
\[
\frac{1}{X^4 - 1}
= \frac{1/4}{X - 1} - \frac{1/4}{X + 1}
+ \frac{\iu/4}{X - \iu} - \frac{\iu/4}{X + \iu} .
\]
Agrupando el par conjugado (denominador común $X^2 + 1$):
$\frac{\iu}{4}\bigl(\frac{1}{X-\iu} - \frac{1}{X+\iu}\bigr) =
\frac{\iu}{4}\cdot\frac{2\iu}{X^2+1} = \frac{-1/2}{X^2+1}$. Más
$\R$:
\[
\frac{1}{X^4 - 1}
= \frac{1/4}{X-1} - \frac{1/4}{X+1} - \frac{1/2}{X^2 + 1} .
\]
\emph{Verificar en $X = 0$:}$-1 = -\frac14 - \frac14 - \frac12$.
\end{solution}

\begin{solution}{exo:b1:fractions:5}
$\dfrac{X^2}{(X^2+1)^2} = \dfrac{(X^2 + 1) - 1}{(X^2+1)^2} =
\dfrac{1}{X^2+1} - \dfrac{1}{(X^2+1)^2}$.

De ahí una primitiva:
\[
\int \frac{x^2\,\dd x}{(x^2+1)^2}
= \arctan x - \frac12\Bigl(\arctan x + \frac{x}{x^2+1}\Bigr) + C
= \frac12\arctan x - \frac{x}{2(x^2+1)} + C .
\]
\end{solution}

\begin{solution}{exo:b1:fractions:6}
Los \omterm{def:b1:fractions:field}{polos} $0, -1, \dots, -n$ son simples. Encubrimiento en $-k$:
\[
c_k = \frac{n!}{\prod_{j \neq k} (-k + j)}
= \frac{n!}{\bigl(\prod_{j=0}^{k-1}(j - k)\bigr)
\bigl(\prod_{j=k+1}^{n}(j-k)\bigr)}
= \frac{n!}{(-1)^k k!\,(n-k)!} = (-1)^k \binom nk .
\]
entonces
\[
\frac{n!}{X(X+1)\cdots(X+n)}
= \sum_{k=0}^{n} \frac{(-1)^k \binom nk}{X + k} .
\]
(Comprobación de cordura para $n = 1$:$\frac{1}{X(X+1)} = \frac1X -
\frac{1}{X+1}$).
\end{solution}

\begin{solution}{exo:b1:fractions:7}
$P = X^n - 1$ tiene las raíces simples $n$ $\omega^k$
(\cref{thm:b1:complex:roots}), por lo que la identidad de la derivada logarítmica
de lecturas \cref{exo:b1:poly:11}
\[
\sum_{k=0}^{n-1} \frac{1}{X - \omega^k}
= \frac{P'(X)}{P(X)} = \frac{n X^{n-1}}{X^n - 1} .
\]
En $X = 2$,$n = 4$: lado derecho$= \frac{4 \times 8}{15} =
\frac{32}{15}$. Lado izquierdo:$\frac{1}{2-1} + \frac{1}{2+1} +
\frac{1}{2 - \iu} + \frac{1}{2 + \iu} = 1 + \frac13 + \frac{4}{5} =
\frac{15 + 5 + 12}{15} = \frac{32}{15}$, usando $\frac{1}{2-\iu} +
\frac{1}{2+\iu} = \frac{4}{5}$.
\end{solution}

\begin{solution}{exo:b1:fractions:8}
Encubrimiento: $\dfrac{1}{k(k+1)(k+2)} = \dfrac{1/2}{k} - \dfrac{1}{k+1} +
\dfrac{1/2}{k+2}$. Reescribir como una diferencia telescópica:
\[
\frac{1}{k(k+1)(k+2)}
= \frac12\Bigl(\frac{1}{k(k+1)} - \frac{1}{(k+1)(k+2)}\Bigr),
\]
(expandir para comprobar --- o restar las dos descomposiciones). Resumiendo:
\[
S_n = \frac12\Bigl(\frac{1}{1 \times 2} - \frac{1}{(n+1)(n+2)}\Bigr)
= \frac14 - \frac{1}{2(n+1)(n+2)}
\xrightarrow[n \to \infty]{} \frac14 .
\]
\end{solution}

\begin{solution}{exo:b1:fractions:9}
Descomponer, para $0 \leq m \leq n - 1$, la fracción $\frac{X^m}{P}$
(grado $m - n \leq -1$, \omterm{def:b1:fractions:field}{polos} simple): la fórmula \omterm{def:b1:fractions:field}{polo simple} da
\[
\frac{X^m}{P} = \sum_{i=1}^{n} \frac{x_i^m / P'(x_i)}{X - x_i} .
\]
Multiplica por $X$ y deja que $X \to +\infty $: el lado izquierdo tiende al
límite de $X^{m+1}/P$, que es $0$ si $m \leq n - 2$ y $1$ si $m =
n - 1$($P$ es \omterm{def:b1:poly:def}{mónico} de grado $n$); el lado derecho tiende a
$\sum_i \frac{x_i^m}{P'(x_i)}$. Por lo tanto
\[
\sum_{i=1}^{n} \frac{x_i^{m}}{P'(x_i)} =
\begin{cases}
0 & \text{for } 0 \leq m \leq n-2,\\
1 & \text{for } m = n - 1,
\end{cases}
\]
que contiene ambas identidades anunciadas ($m = 0$ requiere $n \geq
2$). (Interpretación vía \cref{thm:b1:poly:lagrange}: estas sumas son
los coeficientes principales de los interpolantes de Lagrange de $X^m$, y
interpolando un \omterm{def:b1:poly:def}{polinomio} de grado $\leq n-1$ en puntos $n$
lo reproduce exactamente.)
\end{solution}

\begin{solution}{exo:b1:fractions:10}
Forma $\frac aX + \frac b{X+1} + \frac c{(X+1)^2}$. Encubrimiento en
$0$:$a = 1$; encubrimiento en el doble \omterm{def:b1:fractions:field}{polo}:$c = \bigl[\frac1X
\bigr]_{X=-1} = -1$; límite de $xF(x)$ en el infinito:$0 = a + b $, entonces
$b = -1$:
\[
\frac{1}{X(X+1)^2} = \frac1X - \frac1{X+1} - \frac1{(X+1)^2} .
\]
Resumen de $n = 1, \dots, N$: el primer telescopio de dos ladrillos que
$1 - \frac1{N+1}$, y el tercero aporta$-\sum_{k=2}^{N+1}
\frac1{k^2}$. Dejar $N \to \infty $ y usar $\sum_{k\geq1}
\frac1{k^2} = \frac{\pi^2}6$:
\[
\sum_{n=1}^{\infty}\frac{1}{n(n+1)^2}
= 1 - \Bigl(\frac{\pi^2}6 - 1\Bigr) = 2 - \frac{\pi^2}6
\approx 0.355 .
\]
\end{solution}

\begin{solution}{exo:b1:fractions:11}
En la variable $Y$:$\frac1{(Y+1)(Y+4)} = \frac{1/3}{Y+1} -
\frac{1/3}{Y+4}$(encubrimiento en $-1$ y $-4$), entonces
\[
\frac{1}{(X^2+1)(X^2+4)}
= \frac13\,\frac1{X^2+1} - \frac13\,\frac1{X^2+4} .
\]
Asimismo $\frac{Y}{(Y+1)(Y+4)} = \frac{-1/3}{Y+1} +
\frac{4/3}{Y+4}$, entonces
\[
\frac{X^2}{(X^2+1)(X^2+4)}
= -\frac13\,\frac1{X^2+1} + \frac43\,\frac1{X^2+4} .
\]
(Consulte en $X = 0$:$0 = \frac13(-1 + 1)$). Estos ya son los
Descomposiciones reales: los numeradores sobre las cuadráticas irreducibles.
resultan ser constantes.
\end{solution}

\begin{solution}{exo:b1:fractions:12}
\begin{enumerate}
 \item En cada intervalo evitando \omterm{def:b1:fractions:field}{polos}, $F'(x) = -\sum_i
 \frac{c_i}{(x - p_i)^2} < 0$: estrictamente decreciente. Como $x
 \to \pm\infty $, cada ladrillo tiende a$0$:$F \to 0$, de
 arriba en $+\infty $(todos los ladrillos positivos allí) y desde
 a continuación en $-\infty$. Como $x \to p_i^+$, el ladrillo
 $\frac{c_i}{x - p_i}$ explota contra$+\infty $ y los demás
 permanecer limitado: $F \to +\infty$; igualmente $F \to -\infty$ como
 $x \to p_i^-$.
 \item Reparar $\lambda > 0$. En$\intoo{-\infty}{p_1}$:$F$
 disminuye de $0^-$a$-\infty $, por lo que $F < 0 < \lambda $:
 ninguna solución. En cada $\intoo{p_i}{p_{i+1}}$($1 \leq i
 \leq r-1$):$F $ disminuye de $+\infty $a$-\infty$,
 por lo tanto toma el valor $\lambda$ exactamente una vez (intermedio
 propiedad de valor más estricta monotonicidad). encendido
 $\intoo{p_r}{+\infty}$:$F$ disminuye de $+\infty $ a
 $0^+$, nuevamente exactamente una solución. Total: exactamente $r$
 soluciones, entrelazadas con el \omterm{def:b1:fractions:field}{polos}.
\end{enumerate}
\end{solution}

\begin{solution}{pb:b1:fractions:1}
\textbf{1.} Encubrimiento: $\frac1{X(X+1)} = \frac1X - \frac1{X+1}$.
La suma telescopios:
\[
\sum_{n=1}^{N}\Bigl(\frac1n - \frac1{n+1}\Bigr)
= 1 - \frac1{N+1} \longrightarrow 1 .
\]

\textbf{2.} $\frac1{X(X+2)} = \frac{1/2}X - \frac{1/2}{X+2}$.
En resumen, los términos $\frac1n$ sobreviven para $n = 1, 2$ y los términos
$-\frac1{n+2}$ sobrevive para $n = N-1, N$:
\[
\sum_{n=1}^{N}\frac1{n(n+2)}
= \frac12\Bigl(1 + \frac12 - \frac1{N+1} - \frac1{N+2}\Bigr)
\longrightarrow \frac34 .
\]

\textbf{3.} Si $F(X) = G(X) - G(X+1)$, entonces $\sum_{n=1}^N F(n) =
\sum_{n=1}^N\bigl(G(n) - G(n+1)\bigr) = G(1) - G(N+1)$: todos
los valores intermedios se cancelan en pares. Para $F =
\frac1{X(X+1)(X+2)}$, el testigo es $G(X) = \frac1{2X(X+1)}$:
\[
G(X) - G(X+1)
= \frac{(X+2) - X}{2X(X+1)(X+2)} = F(X) ,
\]
entonces $\sum_{n=1}^N F(n) = \frac14 - \frac1{2(N+1)(N+2)} \to
\frac14$, el valor de \cref{exo:b1:fractions:8}.

\textbf{4.} Pon el lado derecho sobre el denominador común.
$X(X+1)\cdots(X+k)$:
\[
\frac1k\cdot\frac{(X + k) - X}{X(X+1)\cdots(X+k)}
= \frac{1}{X(X+1)\cdots(X+k)} ,
\]
cual es la identidad. Entonces $F_k(X) = \frac1k\bigl(G_k(X) -
G_k(X+1)\bigr)$ con $ G_k(X) = \frac1{X(X+1)\cdots(X+k-1)}$, y
El mecanismo de la pregunta 3 da
\[
\sum_{n=1}^{N}\frac1{n(n+1)\cdots(n+k)}
= \frac1k\Bigl(\frac1{k!} - G_k(N+1)\Bigr)
\longrightarrow \frac1{k\cdot k!} ,
\]
desde $G_k(1) = \frac1{k!}$ y $G_k(N+1) \to 0$. Para $k = 2$:
$\frac1{2\cdot2} = \frac14$, pregunta coincidente 3.

\textbf{5.} \cref{exo:b1:fractions:6} da
$\frac{n!}{X(X+1)\cdots(X+n)} = \sum_{j=0}^n
\frac{(-1)^j\binom nj}{X + j}$. Evaluar en $X = 1$: el lado izquierdo
es $\frac{n!}{(n+1)!} = \frac1{n+1}$, el lado derecho es
$\sum_j(-1)^j\binom nj\frac1{1+j}$: primera identidad. En$X = 2$:
el lado izquierdo es $\frac{n!}{2\cdot3\cdots(n+2)} =
\frac{n!\cdot1}{(n+2)!} = \frac1{(n+1)(n+2)}$, el lado derecho
$\sum_j(-1)^j\binom nj\frac1{2+j}$: segunda identidad.

\textbf{6.} Los \omterm{def:b1:fractions:field}{polos} $\omega^k$ son simples, y los \omterm{def:b1:fractions:field}{polo simple}
la fórmula de \cref{met:b1:fractions:compute} da el coeficiente
\[
\frac{1}{\bigl(nX^{n-1}\bigr)_{X = \omega^k}}
= \frac{1}{n\,\omega^{k(n-1)}}
= \frac{\omega^k}{n\,\omega^{kn}} = \frac{\omega^k}n ,
\]
utilizando $\omega^{kn} = 1$. Por lo tanto $\frac1{X^n-1} = \frac1n\sum_k
\frac{\omega^k}{X - \omega^k}$.

\textbf{7.} Para $n = 2$($\omega = -1$):$\frac12\bigl(
\frac1{X-1} - \frac1{X+1}\bigr) = \frac12\cdot\frac{2}{X^2-1} =
\frac1{X^2-1}$: correcto. Los coeficientes suman
$\frac1n\sum_k\omega^k = 0$ para $n \geq 2$
(\cref{prop:b1:complex:sumroots}). Deben: $x\,F(x) \to
\sum_k c_k $ como $x \to \infty $ para cualquier descomposición con simple
\omterm{def:b1:fractions:field}{polos}, mientras aquí $xF(x) = \frac{x}{x^n-1} \to 0$ desde $n \geq
2$.

\textbf{8.} Encubrimiento para $\frac{X^{n-1}}{X^n - 1}$ en $\omega^k$:
$\frac{A(\omega^k)}{B'(\omega^k)} =
\frac{\omega^{k(n-1)}}{n\omega^{k(n-1)}} = \frac1n$, entonces
$\frac{X^{n-1}}{X^n-1} = \frac1n\sum_k\frac1{X - \omega^k}$ ---
la identidad de \cref{exo:b1:fractions:7} nuevamente.

\textbf{9.} Con $c = \omega^k$,$\conj c = \omega^{n-k}$ y
$c\conj c = 1$,$c + \conj c = 2\cos\theta_k$:
\[
\frac{c}{X - c} + \frac{\conj c}{X - \conj c}
= \frac{c(X - \conj c) + \conj c(X - c)}
{(X - c)(X - \conj c)}
= \frac{2\cos\theta_k\,X - 2}{X^2 - 2\cos\theta_k\,X + 1} .
\]
Agrupando $k$ con $n - k$ en la pregunta 6: para $n$ impar,
\[
\frac1{X^n - 1} = \frac1n\Biggl(\frac1{X-1} +
\sum_{k=1}^{(n-1)/2}
\frac{2\cos\theta_k X - 2}{X^2 - 2\cos\theta_k X + 1}\Biggr) ;
\]
incluso para $n$, el \omterm{def:b1:fractions:field}{polo}$\omega^{n/2} = -1$ extra emparejado
aporta $\frac{-1}{X+1}$ dentro del paréntesis y el par
La suma llega a $\frac n2 - 1$.

\textbf{10.} $n = 4$:$\theta_1 = \frac\pi2$,$\cos\theta_1 = 0$,
entonces el término del par es $\frac{-2}{X^2+1}$ y
\[
\frac1{X^4-1} = \frac14\Bigl(\frac1{X-1} - \frac1{X+1}
- \frac2{X^2+1}\Bigr) ,
\]
la descomposición de \cref{exo:b1:fractions:4}.

\textbf{11.} $P = \prod_{k=1}^{n-1}(X - \omega^k)$(dividir $X^n -
1$ por $ X - 1$), entonces por \cref{exo:b1:poly:11},$\frac{P'}P =
\sum_{k=1}^{n-1}\frac1{X-\omega^k}$. En$X = 1$:$P(1) = n $ y
$P'(1) = \sum_{j=1}^{n-1}j = \frac{n(n-1)}2$, de donde
\[
\sum_{k=1}^{n-1}\frac1{1 - \omega^k}
= \frac{P'(1)}{P(1)} = \frac{n-1}2 .
\]

\textbf{12.} Medio ángulo: $1 - \eu^{\iu\theta} =
-2\iu\sin\frac\theta2\,\eu^{\iu\theta/2}$, entonces
\[
\frac1{1 - \eu^{\iu\theta}}
= \frac{\eu^{-\iu\theta/2}}{-2\iu\sin\frac\theta2}
= \frac{\iu\cos\frac\theta2 + \sin\frac\theta2}
{2\sin\frac\theta2}
= \frac12 + \frac\iu2\,
\frac{\cos\frac\theta2}{\sin\frac\theta2} .
\]
Con $\theta = \theta_k = \frac{2k\pi}n$:$\frac1{1-\omega^k} =
\frac12 + \frac\iu2\cot\frac{k\pi}n $. Sumando $k = 1, \dots,
n-1$ y comparando con el valor real $\frac{n-1}2$ de
pregunta 11: las partes reales ya dan cuenta de todo, así que
$\sum_k\cot\frac{k\pi}n = 0$--- como también muestra la simetría $\cot\frac{(n-k)
\pi}n = -\cot\frac{k\pi}n $.

\textbf{13.} Diferenciando $\frac{P'}P = \sum_k\frac1{X -
\omega^k}$:
\[
\frac{P''}P - \Bigl(\frac{P'}P\Bigr)^2
= -\sum_{k=1}^{n-1}\frac1{(X - \omega^k)^2} .
\]
En $X = 1$:$P''(1) = \sum_{j=2}^{n-1}j(j-1) = 2\binom n3 =
\frac{n(n-1)(n-2)}3$(identidad de palo de hockey o inducción), entonces
\[
\sum_{k=1}^{n-1}\frac1{(1 - \omega^k)^2}
= \Bigl(\frac{n-1}2\Bigr)^2 - \frac{(n-1)(n-2)}3
= \frac{(n-1)\bigl(3(n-1) - 4(n-2)\bigr)}{12}
= \frac{(n-1)(5-n)}{12} .
\]

\textbf{14.} Elevando la fórmula de la pregunta 12, con $c_k =
\cot\frac{k\pi}n $:
\[
\frac1{(1-\omega^k)^2}
= \Bigl(\frac12 + \frac\iu2 c_k\Bigr)^2
= \frac14 - \frac{c_k^2}4 + \frac\iu2\,c_k .
\]
Sumando y usando $\sum c_k = 0$(pregunta 12) y pregunta 13:
$\frac{n-1}4 - \frac14\sum_k c_k^2 = \frac{(n-1)(5-n)}{12}$, entonces
\[
\sum_{k=1}^{n-1}\cot^2\frac{k\pi}n
= (n-1) - \frac{(n-1)(5-n)}3 = \frac{(n-1)(n-2)}3 .
\]
Entonces $\frac1{\sin^2t} = 1 + \cot^2t$ da
$\sum_k\frac1{\sin^2\frac{k\pi}n} = (n-1) +
\frac{(n-1)(n-2)}3 = \frac{n^2-1}3$.

\textbf{15.} $n = 3$:$\cot^2\frac\pi3 + \cot^2\frac{2\pi}3 =
\frac13 + \frac13 = \frac23 = \frac{2\cdot1}3$; y
$\frac1{\sin^2}$ suma $\frac43 + \frac43 = \frac83 =
\frac{9-1}3$.$n = 4$:$1 + 0 + 1 = 2 = \frac{3\cdot2}3$; y $2
+ 1 + 2 = 5 = \frac{16-1}3$. Ambas fórmulas lo comprueban.

\textbf{16.} Para $t \in \intoo0{\frac\pi2}$, el clásico
comparación $\sin t < t < \tan t $(argumento de área o convexidad,
familiar de la escuela secundaria) produce, tomando recíprocos, $\cot t <
\frac1t < \frac1{\sin t}$, los tres positivos allí; elevar al cuadrado
conserva el orden. Con $t = \frac{k\pi}n$,$1 \leq k \leq m$,
$n = 2m+1$(entonces $t < \frac\pi2$):
\[
\cot^2\frac{k\pi}n < \frac{n^2}{k^2\pi^2} <
\frac1{\sin^2\frac{k\pi}n} .
\]

\textbf{17.} Por las simetrías $\cot^2\frac{(n-k)\pi}n =
\cot^2\frac{k\pi}n$ y de igual manera para $\sin^2$, las sumas de la pregunta 14
reducir a la mitad: $\sum_{k=1}^{m}\cot^2\frac{k\pi}n = \frac{(n-1)(n-2)}6 =
\frac{m(2m-1)}3$ y $\sum_{k=1}^m\frac1{\sin^2\frac{k\pi}n} =
\frac{n^2-1}6 = \frac{2m(m+1)}3$. Sumando la pregunta 16 sobre $k =
1, \dots, m $ y multiplicando por $\frac{\pi^2}{n^2}$:
\[
\frac{\pi^2}{(2m+1)^2}\cdot\frac{m(2m-1)}3
\;<\; \sum_{k=1}^{m}\frac1{k^2} \;<\;
\frac{\pi^2}{(2m+1)^2}\cdot\frac{2m(m+1)}3 .
\]
Ambos límites tienden a $\frac{\pi^2}6$ como $m \to \infty $(las razones
$\frac{m(2m-1)}{(2m+1)^2}$ y $\frac{2m(m+1)}{(2m+1)^2}$ ambos
tienden a $\frac12$), por lo que al apretar las sumas parciales crecientes
converger y
\[
\sum_{k=1}^{\infty}\frac1{k^2} = \frac{\pi^2}6 .
\]

\textbf{18.} Cuadrado $\frac1{X^2-1} = \frac12\bigl(\frac1{X-1} -
\frac1{X+1}\bigr)$:
\[
\frac1{(X^2-1)^2} = \frac14\Bigl(\frac1{(X-1)^2} +
\frac1{(X+1)^2}\Bigr) - \frac12\cdot\frac1{(X-1)(X+1)} ,
\]
y vuelva a descomponer el término cruzado $\frac1{(X-1)(X+1)} =
\frac12\bigl(\frac1{X-1} - \frac1{X+1}\bigr)$ para obtener lo indicado
forma. En $X = 0$: lado izquierdo $1$; lado derecho $\frac14(1 + 1) -
\frac14(-1 - 1) = \frac12 + \frac12 = 1$.

\textbf{19.} Suma la pregunta 18 sobre $n \geq 2$. Con $S =
\sum_{k\geq1}\frac1{k^2} = \frac{\pi^2}6$:$\sum_{n\geq2}
\frac1{(n-1)^2} = S$;$\sum_{n\geq2}\frac1{(n+1)^2} = S - 1 -
\frac14$; y el telescopio Gap-dos $\sum_{n\geq2}\bigl(
\frac1{n-1} - \frac1{n+1}\bigr) = 1 + \frac12 = \frac32$. Por lo tanto
\[
\sum_{n=2}^{\infty}\frac1{(n^2-1)^2}
= \frac14\Bigl(2S - \frac54\Bigr) - \frac14\cdot\frac32
= \frac S2 - \frac{11}{16}
= \frac{\pi^2}{12} - \frac{11}{16} \approx 0.135 .
\]
Numéricamente: $\frac19 + \frac1{64} + \frac1{225} + \frac1{576} +
\frac1{1225} + \dots \approx 0.1111 + 0.0156 + 0.0044 + 0.0017 +
0.0008 + \dots \approx 0.135$: consistente.

\textbf{20.} Diferenciar la identidad de la pregunta 8
$\sum_k\frac1{X-\omega^k} = \frac{nX^{n-1}}{X^n-1}$:
\[
\sum_{k=0}^{n-1}\frac1{(X - \omega^k)^2}
= \frac{n^2X^{2n-2} - n(n-1)X^{n-2}(X^n - 1)}{(X^n - 1)^2} .
\]
Verifique en $n = 2$,$X = 2$: lado derecho $\frac{4\cdot4 -
2\cdot1\cdot3}{9} = \frac{10}9$; lado izquierdo $\frac1{(2-1)^2} +
\frac1{(2+1)^2} = 1 + \frac19 = \frac{10}9$.

\textbf{21.} $\frac1{\binom nk} = \frac{k!\,(n-k)!}{n!} =
\frac{k!}{(n-k+1)(n-k+2)\cdots n}$, un producto de $k $ consecutivo
números enteros abajo. Sustituyendo $j = n - k + 1$(para que se ejecute $j$
sobre $\N^*$ya que $n$ se ejecuta desde $k$):
\[
\sum_{n=k}^{\infty}\frac1{\binom nk}
= k!\sum_{j=1}^{\infty}\frac1{j(j+1)\cdots(j+k-1)}
= \frac{k!}{(k-1)\,(k-1)!} = \frac{k}{k-1} ,
\]
por la pregunta 4 aplicada con $k - 1$ en lugar de $k $(válido desde
$k - 1 \geq 1$).

\textbf{22.} Para $k = 3$:$\frac1{\binom n3} =
\frac6{(n-2)(n-1)n}$, entonces por la suma parcial de la pregunta 3 (desplazada),
\[
\sum_{n=3}^{N}\frac1{\binom n3}
= 6\sum_{j=1}^{N-2}\frac1{j(j+1)(j+2)}
= 6\Bigl(\frac14 - \frac1{2(N-1)N}\Bigr)
= \frac32 - \frac3{(N-1)N}
\longrightarrow \frac32 ,
\]
el valor $\frac k{k-1} = \frac32$ de la pregunta 21.

\textbf{23.} $n = 6$: pares $k = 1$($\theta_1 = \frac\pi3$,
$2\cos\theta_1 = 1$) y $k = 2$($\theta_2 = \frac{2\pi}3$,
$2\cos\theta_2 = -1$), más el \omterm{def:b1:fractions:field}{polos}$\pm1$ real:

\frac1{X^6-1} = \frac16\Bigl(\frac1{X-1} - \frac1{X+1}
+ \frac{X - 2}{X^2 - X + 1}
+ \frac{-X - 2}{X^2 + X + 1}\Bigr) .
\]
En $X = 0$: lado izquierdo$-1$; lado derecho $\frac16(-1 - 1 - 2 - 2)
= -1$: correcto.

\textbf{25.} La descomposición es una identidad puramente algebraica,
verdadero para cada valor de la variable a la vez; eso es lo que hace
es un motor. Sustituir números enteros y sumar lo convirtió en
telescopios (Parte I); sustituyendo \omterm{def:b1:complex:unity}{raíces de la unidad} y agrupando
los conjugados lo convirtieron en identidades trigonométricas (Partes II y
III); y sólo en el último paso entró el análisis: un
apretar entre dos formas cerradas --- para entregar $\frac{\pi^2}6$,
a \omterm{def:b1:logic:statement}{enunciado} no se pudo alcanzar ninguna sustitución finita. Esta división de
trabajo (el álgebra produce identidades finitas exactas, el análisis pasa
al límite) es la plantilla para \cref{ch:b1:series}, donde
el telescopio y la comparación se vuelven sistemáticos, y por
\cref{ch:b1:integration}, donde cada ladrillo adquiere una primitiva
y las mismas descomposiciones calculan integrales en lugar de sumas.
