\begin{solution}{exo:b1:functions:2}
$f(x) = \arcsin(2x - 1)$ requiere $-1 \leq 2x - 1 \leq 1$, es decir,$x
\in \intcc{0}{1}$. En$\intoo{0}{1}$, la regla de la cadena y
\cref{prop:b1:functions:arcderiv} dar
\[
f'(x) = \frac{2}{\sqrt{1 - (2x-1)^2}} = \frac{2}{\sqrt{4x - 4x^2}}
= \frac{1}{\sqrt{x(1 - x)}} .
\]
$g(x) = \arctan\sqrt x$ está definido para $x \geq 0$ y para $x > 0$:
\[
g'(x) = \frac{1}{1 + x} \cdot \frac{1}{2\sqrt x}
= \frac{1}{2\sqrt x\,(1 + x)} .
\]
\end{solution}

\begin{solution}{exo:b1:functions:3}
$\cosh x + \sinh x = \frac{\eu^x + \eu^{-x}}{2} + \frac{\eu^x -
\eu^{-x}}{2} = \eu^x $. Por lo tanto, para $n \in \Z $:
\[
(\cosh x + \sinh x)^n = (\eu^x)^n = \eu^{nx}
= \cosh nx + \sinh nx .
\]
\end{solution}

\begin{solution}{exo:b1:functions:4}
Con $ y = \arcsin x$:$\cos 2y = 1 - 2\sin^2 y = 1 - 2x^2$, entonces
$\cos(2\arcsin x) = 1 - 2x^2$.

Con $ y = \arctan x$:$\sin 2y = 2 \sin y \cos y = 2 \tan y \cos^2 y =

\begin{solution}{exo:b1:functions:5}
De más lento a más rápido:
\[
(\ln x)^{1000} \;\ll\; x^{100} \;\ll\; x^{\ln x} \;\ll\;
\eu^{\sqrt x} \;\ll\; \eu^{x/100} .
\]
Justificaciones: $(\ln x)^{1000}/x^{100} \to 0$ por
\cref{prop:b1:functions:powerrules} (los registros pierden poderes).
$x^{100} = \eu^{100\ln x}$ y $x^{\ln x} = \eu^{(\ln x)^2}$: desde
$(\ln x)^2 - 100\ln x \to +\infty$, gana el segundo. $x^{\ln x} =
\eu^{(\ln x)^2} \ll \eu^{\sqrt x}$ porque $(\ln x)^2/\sqrt x \to 0$
(los registros pierden elevado a la potencia $x^{1/4}$, al cuadrado). Finalmente $\sqrt x - x/100
\to -\infty $, entonces $\eu^{\sqrt x} \ll \eu^{x/100}$.
\end{solution}

\begin{solution}{exo:b1:functions:6}
Dominio: $x \neq \pm 1$, tres intervalos. En cada uno,
\[
f'(x) = \frac{\bigl(\frac{2x}{1-x^2}\bigr)'}{1 +
\frac{4x^2}{(1-x^2)^2}}
= \frac{\frac{2(1-x^2) + 4x^2}{(1-x^2)^2}}
{\frac{(1-x^2)^2 + 4x^2}{(1-x^2)^2}}
= \frac{2(1 + x^2)}{(1 + x^2)^2} = \frac{2}{1 + x^2}
= (2\arctan x)' .
\]
Entonces $f(x) - 2\arctan x$ es constante en cada intervalo. Valores: en $x =
0$,$f(0) = 0$: la constante es $0$ en $\intoo{-1}{1}$. Como $x \to
+\infty $,$\frac{2x}{1-x^2} \to 0^-$ entonces $f \to 0$, mientras que $2\arctan x
\to \pi $: la constante es $-\pi $ en $\intoo{1}{+\infty}$. Por rareza,
es $+\pi$ en $\intoo{-\infty}{-1}$. Resumen:$f = 2\arctan x$ en
$\intoo{-1}{1}$,$f = 2\arctan x - \pi $ para $x > 1$,$f = 2\arctan x +
\pi $ para $x < -1$. (Esta es la fórmula del doble ángulo para la tangente,
lea $\arctan$.)
\end{solution}

\begin{solution}{exo:b1:functions:7}
$\cosh x = 2$: con $u = \eu^x > 0$,$u + u^{-1} = 4$, entonces $u^2 - 4u +
1 = 0$,$u = 2 \pm \sqrt 3$:$x = \ln(2 + \sqrt 3)$ o $ x = \ln(2 -
\sqrt 3) = -\ln(2 + \sqrt 3)$(las dos soluciones son opuestas, como
$\cosh$ es par; ambos son válidos). Equivalentemente $x = \pm
\operatorname{arcosh} 2$.

$5\cosh x - 4\sinh x = 3$: sustituyendo las definiciones exponenciales,
$\frac{5(u + u^{-1}) - 4(u - u^{-1})}{2} = 3$, es decir $u + 9u^{-1} =

\ln 3$.
\end{solution}

\begin{solution}{exo:b1:functions:8}
Sea $\alpha = \arctan\frac12 + \arctan\frac13$. Fórmula de suma:
\[
\tan\alpha = \frac{\frac12 + \frac13}{1 - \frac12\cdot\frac13}
= \frac{5/6}{5/6} = 1 .
\]
Ambos \omterm{def:b1:functions:arc}{arctangentes} se encuentran en $\intoo{0}{\frac\pi4}$(sus argumentos están en
$\intoo{0}{1}$), entonces $\alpha \in \intoo{0}{\frac\pi2}$; el único ángulo
allí con tangente $1$ es $\frac\pi4$.

Máquina: let $\beta = \arctan\frac15$. Doble ángulo dos veces:
\[
\tan 2\beta = \frac{2/5}{1 - 1/25} = \frac{5}{12},
\qquad
\tan 4\beta = \frac{2 \cdot 5/12}{1 - 25/144} = \frac{120}{119}.
\]
entonces
\[
\tan\Bigl(4\beta - \frac\pi4\Bigr)
= \frac{\frac{120}{119} - 1}{1 + \frac{120}{119}}
= \frac{1/119}{239/119} = \frac{1}{239}.
\]
Ubicación: $\beta < \arctan 1 = \frac\pi4$ y de hecho $\tan 4\beta =
\frac{120}{119}$ cerca de $1$ con $4\beta \in \intoo{0}{\frac\pi2}$
(desde $\beta < \frac\pi8$, como $\tan\frac\pi8 = \sqrt 2 - 1 > \frac15$);
entonces $4\beta - \frac\pi4 \in \intoo{-\frac\pi4}{\frac\pi4}$, donde
$\arctan$ invierte $\tan $:$4\beta - \frac\pi4 =
\arctan\frac{1}{239}$, que es la fórmula de Machin.
\end{solution}

\begin{solution}{exo:b1:functions:9}
Sea $f(x) = x - \sin x$:$f(0) = 0$ y $f'(x) = 1 - \cos x \geq 0$,
entonces $f \geq 0$ en $\R_+$:$\sin x \leq x$.

Sea $g(x) = \sin x - x + \frac{x^3}{6}$:$g(0) = 0$,$g'(x) = \cos x -
1 + \frac{x^2}{2}$,$g'(0) = 0$,$g''(x) = -\sin x + x = f(x) \geq 0$
en $\R_+$. Entonces $g'$ aumenta con $g'(0) = 0$, por lo tanto $g' \geq 0$,
por lo tanto, $g$ aumenta con $g(0) = 0$:$g \geq 0$, es decir, $x -
\frac{x^3}{6} \leq \sin x $ en $\R_+$.

En consecuencia, $0 \leq \frac{x - \sin x}{x^3} \leq \frac 16$ para $x >
0$: la relación está atrapada cerca de la escala de $\frac16$, y
\cref{ch:b1:taylor} muestra que su límite es exactamente $\frac 16$.
\end{solution}

\begin{solution}{exo:b1:functions:10}
Escriba $ y = \arcsin x$, por lo que $2x\sqrt{1 - x^2} = 2\sin y\cos y = \sin
2y $ y $h(x) = \arcsin(\sin 2y) - 2y$.

Para $\abs x \leq \frac{\sqrt 2}{2}$:$2y \in
\intcc{-\frac\pi2}{\frac\pi2}$, entonces $\arcsin(\sin 2y) = 2y $ y $ h =
0$.

Para $x \in \intoc{\frac{\sqrt 2}{2}}{1}$:$2y \in
\intoc{\frac\pi2}{\pi}$, y el ángulo de
$\intcc{-\frac\pi2}{\frac\pi2}$ con seno $\sin 2y$ es $\pi - 2y$:
entonces $h(x) = \pi - 4\arcsin x$. (Compruebe por derivada: hay
$h'(x) = \frac{2(1 - 2x^2)}{\abs{1 - 2x^2}\sqrt{1 - x^2}} -
\frac{2}{\sqrt{1-x^2}} = \frac{-4}{\sqrt{1 - x^2}}$, la derivada
de $-4\arcsin x$; y en $x = 1$,$h(1) = \arcsin 0 - 2\cdot\frac\pi2
= -\pi = \pi - 4\cdot\frac\pi2$.)

Para $x \in \intco{-1}{-\frac{\sqrt 2}{2}}$, por imparidad de $h$:
$h(x) = -\pi - 4\arcsin x$.
\end{solution}

\begin{solution}{exo:b1:functions:11}
$g = \arctan \circ \sinh$ es una composición de impar, estrictamente
funciones crecientes, por lo que es impar y estrictamente creciente; como $x \to
+\infty $,$\sinh x \to +\infty $, por lo que $g(x) \to \frac\pi2$ y $ g$ son
biyección sobre $\intoo{-\frac\pi2}{\frac\pi2}$(continuidad más la
teorema del valor intermedio). Regla de la cadena con
\cref{prop:b1:functions:hyprules}~(1):
\[
g'(x) = \frac{\cosh x}{1 + \sinh^2 x} = \frac{\cosh x}{\cosh^2 x}
= \frac{1}{\cosh x} .
\]
Por construcción $\tan g(x) = \sinh x$. Entonces, dado que $g(x) \in
\intoo{-\frac\pi2}{\frac\pi2}$ tiene coseno positivo,
\begin{align*}
\cos g(x) &= \frac{1}{\sqrt{1 + \tan^2 g(x)}}
= \frac{1}{\sqrt{1 + \sinh^2 x}} = \frac{1}{\cosh x},\\
\sin g(x) &= \tan g(x) \cos g(x) = \frac{\sinh x}{\cosh x} = \tanh x .
\end{align*}
\end{solution}

\begin{solution}{exo:b1:functions:12}
Establecer $u = \arctan 2$ y $v = \arctan 3$; ambos se encuentran en
$\intoo{\frac\pi4}{\frac\pi2}$(sus argumentos exceden $1$), entonces $u
+ v \in \intoo{\frac\pi2}{\pi}$. La fórmula de la suma da
\[
\tan(u + v) = \frac{2 + 3}{1 - 6} = -1 ,
\]
y el único ángulo de $\intoo{\frac\pi2}{\pi}$ con la tangente$-1$
es $\frac{3\pi}4$: entonces $\arctan 2 + \arctan 3 = \frac{3\pi}4$.
(Aplicar a ciegas $\arctan$ a la tangente daría
$-\frac\pi4$, apagado por $\pi$ --- la localización es lo que salva el
cálculo, y el denominador negativo $1 - 6$ es precisamente el
señal de que la suma ha salido
$\intoo{-\frac\pi2}{\frac\pi2}$.) Añadiendo $\arctan 1 = \frac\pi4$:
\[
\arctan 1 + \arctan 2 + \arctan 3 = \frac\pi4 + \frac{3\pi}4 = \pi .
\]
\end{solution}

\begin{solution}{pb:b1:functions:1}
\textbf{1.} $f$ es un cociente de funciones diferenciables con
denominador que no desaparece en $\intoo0{+\infty}$, y
\[
f'(t) = \frac{\frac1t \cdot t - \ln t}{t^2} = \frac{1 - \ln
t}{t^2} ,
\]
positivo para $t < \eu$, cero en $t = \eu$, negativo para $t > \eu$:
$f$ aumenta estrictamente en $\intoc0\eu $, disminuye estrictamente en
$\intco\eu{+\infty}$, con máximo $f(\eu) = \frac1\eu $.

\textbf{2.} Como $t \to 0^+$:$\ln t \to -\infty $ y $\frac1t \to
+\infty $, entonces $f(t) \to -\infty $. Como $t \to +\infty $:$f(t) \to 0$
por el \omterm{prop:b1:functions:powerrules}{comparación de crecimiento} de \cref{prop:b1:functions:powerrules}
($\beta = \alpha = 1$). Signo: el de $\ln t$, por lo que $f < 0$ sigue
$\intoo01$,$f(1) = 0$,$f > 0$ en $\intoo1{+\infty}$. el grafico
sube desde $-\infty$, cruza cero en $1$, alcanza su punto máximo en $(\eu,
\frac1\eu)$ y luego decae a$0^+$.

\textbf{3.} $f(4) = \frac{\ln 4}4 = \frac{2\ln 2}4 = \frac{\ln 2}2
= f(2)$.

\textbf{4.} Por la tabla de variación y el valor intermedio
teorema (usado en el nivel de escuela secundaria; formalizado en
\cref{ch:b1:continuity}). Para $c < 0$: las soluciones existen sólo donde
$f < 0$, es decir en $\intoo01$ donde $f$ es estrictamente creciente
biyección sobre $\intoo{-\infty}0$: exactamente uno. Para $c = 0$: sólo
$t = 1$. Para $0 < c < \frac1\eu $: en $\intoo1\eu $,$f$ aumenta
de $0$a$\frac1\eu $: una solución; en $\intoo\eu{+\infty}$,
$f$ disminuye de $\frac1\eu $a$0$: uno más; total dos. Para
$c = \frac1\eu$: sólo el punto máximo $t = \eu$. Para $c >
\frac1\eu $: ninguno.

\textbf{5.} Para $x \neq \eu$, la maximalidad estricta da $f(x) <
f(\eu)$, es decir $\frac{\ln x}x < \frac1\eu$, es decir $\eu\ln x < x $,
es decir,\ $\ln(x^\eu) < \ln(\eu^x)$:$x^\eu < \eu^x$. Con $x = \pi $:
$\eu^\pi > \pi^\eu $(numéricamente $23.14 > 22.46$).

\textbf{6.} Para $x, y > 0$, ambos lados son positivos, por lo que
\[
x^y = y^x \iff y\ln x = x\ln y \iff \frac{\ln x}x = \frac{\ln y}y
\iff f(x) = f(y) ,
\]
dividiendo por $xy > 0$.

\textbf{7.} Sea $f(x) = f(y) = c$ con $x \neq y $. Si$c \leq 0$ o
$c = \frac1\eu$, la pregunta 4 dice que la ecuación $f = c$ tiene un solo
solución: imposible. Entonces $0 < c < \frac1\eu$, y nuevamente por
pregunta 4 las dos soluciones son un punto de $\intoo1\eu$ y un
de $\intoo\eu{+\infty}$: ambas coordenadas exceden $1$ y
a horcajadas sobre $\eu$.

\textbf{8.} Sustituyendo $ y = tx$ en $ y\ln x = x\ln y $:
\[
tx\ln x = x(\ln t + \ln x)
\iff (t - 1)\ln x = \ln t
\iff \ln x = \frac{\ln t}{t - 1} ,
\]
entonces $x = t^{1/(t-1)}$ y $ y = tx = t^{1 + \frac1{t-1}} =
t^{t/(t-1)}$. Por el contrario, para estos valores,$\ln y = \frac{t\ln
t}{t-1} = t\ln x $ y $ y = tx$, entonces $\frac{\ln y}y = \frac{t\ln
x}{tx} = \frac{\ln x}x$: una solución. Toda solución no trivial tiene
alguna proporción $t = y/x \in \intoo0{+\infty} \setminus \{1\}$, por lo que el
la parametrización está completa.

\textbf{9.} $t = 2$:$x = 2^{1/1} = 2$,$ y = 2^{2/1} = 4$. y
\[
x(1/t) = (1/t)^{\frac1{\frac1t - 1}}
= (t^{-1})^{\frac{t}{1 - t}}
= t^{\frac{t}{t - 1}} = y(t) ,
\]
entonces $ y(1/t) = \frac1t\,x(1/t) = \frac{y(t)}t = x(t)$: el
cambio de parámetro $t \mapsto 1/t$ intercambia las coordenadas, ya que
la simetría de la ecuación exige.

\textbf{10.} Como $t \to 1$:$\frac{\ln t}{t-1} \to 1$(es el
cociente de diferencia de $\ln$ en $1$), por lo que $x \to \eu^1 = \eu $ y
$ y = tx \to \eu$. Como $t \to +\infty$: $\ln x = \frac{\ln t}{t-1}
\to 0$ entonces $ x \to 1$, mientras que $\ln y = \frac{t\ln t}{t-1} \to
+\infty $ entonces $ y \to +\infty $. Como $t \to 0^+$:$\ln x = \frac{\ln
t}{t-1} \to \frac{-\infty}{-1} = +\infty $ entonces $x \to +\infty $,
mientras que $\ln y = \frac{t\ln t}{t-1} \to \frac{0}{-1} = 0$ entonces $ y \to
1$(usando $t\ln t \to 0$, \cref{ex:b1:functions:xx}). la sucursal
por lo tanto va desde la asíntota $ y = 1$(extrema derecha), hasta
$(\eu, \eu)$ en la diagonal y fuera a lo largo de la asíntota $x = 1$
(arriba) --- simétrico con respecto a la diagonal según la pregunta 9.

\textbf{11.} $\ln x(t) = g(t) = \frac{\ln t}{t-1}$, y
\[
g'(t) = \frac{\frac{t-1}t - \ln t}{(t-1)^2}
= \frac{h(t)}{(t-1)^2},
\qquad h(t) = 1 - \frac1t - \ln t .
\]
$h(1) = 0$ y $h'(t) = \frac1{t^2} - \frac1t = \frac{1 -
t}{t^2} < 0$ para $ t > 1$:$h < 0$ en $\intoo1{+\infty}$, por lo que $g' <
0$ y $ x = \eu^g$ disminuyen estrictamente allí (de $\eu$ a
$1$, por pregunta 10).

\textbf{12.} Según la pregunta 11, $t \mapsto x(t)$ es estrictamente
biyección decreciente de $\intoo1{+\infty}$a$\intoo1\eu $;
escriba $t(x)$ para su inversa (también estrictamente decreciente) y establezca
$\varphi(x) = y(t(x))$. Nota $h < 0$ también en $\intoo01$(hay
$h' > 0$ y $h(1) = 0$), por lo que $x(\cdot)$ está disminuyendo en
$\intoo01$ también; por lo tanto$ y(t) = x(1/t)$(pregunta 9) es
\emph{creciente} en $t$ en $\intoo1{+\infty}$, de $\eu$ a
$+\infty$. Redacción: $\varphi = y \circ t(\cdot)$ es estrictamente
disminuyendo de $\intoo1\eu $a $\intoo\eu{+\infty}$, y
$x^{\varphi(x)} = \varphi(x)^x$ por la pregunta 8. Finalmente, por un
valor común $c = f(x) \in \intoo0{\frac1\eu}$, el par $\{x,
\varphi(x)\}$ es \emph{el} solución de dos elementos \omterm{def:b1:logic:sets}{conjunto} de $f = c $
(pregunta 4); extendiendo $\varphi $a$\intoo\eu{+\infty}$ como el
inverso \omterm{def:b1:logic:map}{aplicación}, $\varphi(\varphi(x))$ regresa al otro (es decir,\
original) elemento: $\varphi \circ \varphi = \mathrm{id}$.

\textbf{13.} Si $(x, y)$ es una solución entera no trivial con $x
< y $, la pregunta 7 pone $x \in \intoo1\eu $: el único número entero allí
es $x = 2$. Luego $f(y) = f(2) = \frac{\ln2}2$ con $ y > \eu $; por
pregunta 4 la ecuación $f = \frac{\ln2}2$ tiene exactamente uno
solución más allá de $\eu$, y la pregunta 3 la muestra:$ y = 4$. Por lo tanto
$(2, 4)$ y su swap son los únicos.

\textbf{18.} Los parámetros $t_n = 1 + \frac1n$ disminuyen estrictamente

\textbf{19.} Si $\eu \leq a < b$:$f$ estrictamente decreciente en
$\intco\eu{+\infty}$ da $f(a) > f(b)$, es decir $b\ln a > a\ln

$2 < \eu < 3$ y $2^3 = 8 < 9 = 3^2$; pero $2 < \eu < 5$ y

del ramal cae el punto $(a, b)$.
