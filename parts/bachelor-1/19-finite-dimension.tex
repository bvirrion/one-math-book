\chapter{Finite Dimension}\label{ch:b1:findim}

A space spanned by finitely many vectors carries a well-defined
\emph{dimension} --- the common size of all its bases. The proofs
below all flow from one combinatorial engine, the exchange lemma:
a free family can never outnumber a generating one. With dimension
come the tools used everywhere afterwards: the incomplete basis
theorem, the rank of a family, Grassmann's formula.

\section{Existence of bases}

\begin{definition}\label{def:b1:findim:def}
$E$ is \emph{finite-dimensional}\index{finite dimension} when it has
a finite generating family. (Otherwise \emph{infinite-dimensional}:
so is $K[X]$, whose finite families only span polynomials of bounded
degree.)
\end{definition}

\begin{theorem}[Exchange lemma]\label{thm:b1:findim:exchange}
Let $(g_1, \dots, g_n)$ generate $E$ and $(f_1, \dots, f_p)$ be free
in $E$. Then $p \leq n$.
\end{theorem}

\begin{proof}
We prove by induction on $k \leq p$: \emph{after renumbering the
$g_j$, the family $(f_1, \dots, f_k, g_{k+1}, \dots, g_n)$ generates
$E$} --- which forces $k \leq n$ at each stage, and $p \leq n$ at
the end.

$k = 0$: the hypothesis. Step: assume it for $k - 1$; then $f_k$ is
a combination of $(f_1, \dots, f_{k-1},$ $g_k, \dots, g_n)$. In
this combination some $g_j$ ($j \geq k$) has a nonzero coefficient
---
otherwise $f_k$ would be a combination of $f_1, \dots, f_{k-1}$,
contradicting freeness. Renumber so that $j = k$, and solve for
$g_k$: $g_k$ is a combination of $(f_1, \dots, f_k, g_{k+1}, \dots,
g_n)$. Every vector of $E$, expressed through the $(k-1)$-family,
can then be re-expressed through the $k$-family: it generates.
(If $k - 1 = n$, no $g$ remains and $f_k$ would be a combination of
the $f_i$'s alone: impossible; so $k \leq n$.)
\end{proof}

\begin{theorem}[Bases in finite dimension]\label{thm:b1:findim:bases}
Let $E \neq \{0\}$ be finite-dimensional.
\begin{enumerate}
  \item From every finite generating family one can extract a basis.
  \item (Incomplete basis theorem\index{incomplete basis theorem})
        Every free family extends to a basis, using vectors of any
        chosen generating family.
  \item All bases of $E$ are finite, with the same number of
        elements: the \emph{dimension} $\dim E$. (Convention:
        $\dim\{0\} = 0$.)
\end{enumerate}
\end{theorem}

\begin{proof}
(1) Discard, one at a time, any vector that is a combination of the
others (the family stays generating: replace the discarded vector by
its expression); stop when no vector can be discarded. The result is
generating and free: a null combination with some nonzero
coefficient would let one solve for that vector, making it
discardable.

(2) Let $(f_1, \dots, f_p)$ be free, $(g_1, \dots, g_n)$ generating.
Run through $g_1, \dots, g_n$, appending $g_j$ to the current family
whenever it is not already in its span
(\cref{prop:b1:vspaces:freecriteria}~(2) keeps the family free).
The final family is free, and generating: every $g_j$ lies in its
span --- either it was appended, or it was already a combination.

(3) Two bases are each free and each generating: the exchange lemma
gives both inequalities between their cardinalities. (Finiteness:
a basis is free, hence by exchange no larger than a finite
generating family.)
\end{proof}

\begin{example}\label{ex:b1:findim:dims}
$\dim K^n = n$ (canonical basis); $\dim K_n[X] = n + 1$ (monomials);
$\dim_\R \C = 2$; the solution space of $y'' + ay' + by = 0$ has
dimension $2$ (\cref{thm:b1:diffeq:homogeneous2}: the solutions are
parametrized bijectively and linearly by $(\lambda, \mu) \in K^2$).
\end{example}

\begin{proposition}[The two-out-of-three rule]\label{prop:b1:findim:twoofthree}
Let $\dim E = n$ and $\mathcal{F}$ a family of exactly $n$ vectors
of $E$. Then
\[
\mathcal{F} \text{ free} \iff \mathcal{F} \text{ generating} \iff
\mathcal{F} \text{ basis}.
\]
Moreover any free family has $\leq n$ vectors, any generating family
$\geq n$.
\end{proposition}

\begin{proof}
The cardinality bounds are the exchange lemma against a basis. If
$\mathcal{F}$ (size $n$) is free but not generating, some $x$ lies
outside its span; appending $x$ gives a free family of $n + 1$
vectors: impossible. If $\mathcal{F}$ is generating but not free,
extracting a basis (\cref{thm:b1:findim:bases}~(1)) gives a basis of
$< n$ vectors: impossible.
\end{proof}

\section{Subspaces, rank, Grassmann}

\begin{theorem}[Subspaces]\label{thm:b1:findim:subspaces}
Let $E$ be finite-dimensional and $F$ a subspace. Then $F$ is
finite-dimensional, $\dim F \leq \dim E$, with equality if and only
if $F = E$. Moreover every subspace has a supplementary subspace.
\end{theorem}

\begin{proof}
Free families of $F$ have at most $\dim E$ vectors (exchange lemma
in $E$). Take a free family of $F$ of \emph{maximal} size $p$: it
generates $F$ (a vector of $F$ outside its span would extend it,
contradicting maximality), so it is a basis of $F$ and $\dim F = p
\leq \dim E$. If $p = \dim E = n$: a free family of $n$ vectors of
$E$ is a basis of $E$ (\cref{prop:b1:findim:twoofthree}), so $F
\supseteq \operatorname{span} = E$. Supplementary: complete a basis
$(f_1, \dots, f_p)$ of $F$ into a basis $(f_1, \dots, f_p, g_{p+1},
\dots, g_n)$ of $E$ (incomplete basis theorem); then $G =
\operatorname{Vect}(g_{p+1}, \dots, g_n)$ satisfies $E = F \oplus G$
(existence and uniqueness of decompositions = coordinates in the
big basis).
\end{proof}

\begin{definition}[Rank of a family]\label{def:b1:findim:rank}
The \emph{rank}\index{rank!of a family} of a finite family of
vectors is the dimension of its span:
$\operatorname{rk}(x_1, \dots, x_p) = \dim
\operatorname{Vect}(x_1, \dots, x_p) \leq \min(p, \dim E)$,
with equality to $p$ iff the family is free.
\end{definition}

\begin{theorem}[Grassmann's formula]\label{thm:b1:findim:grassmann}
For finite-dimensional subspaces $F, G$ of
$E$:\index{Grassmann's formula}
\[
\dim(F + G) = \dim F + \dim G - \dim(F \cap G) .
\]
In particular $F + G$ is direct iff $\dim(F + G) = \dim F + \dim G$.
\end{theorem}

\begin{proof}
Start from a basis $(e_1, \dots, e_r)$ of $F \cap G$; complete it
into a basis $(e_1, \dots, e_r, f_1, \dots, f_s)$ of $F$ and into a
basis $(e_1, \dots, e_r, g_1, \dots, g_t)$ of $G$
(\cref{thm:b1:findim:bases}~(2)). We claim
\[
\mathcal{B} = (e_1, \dots, e_r, f_1, \dots, f_s, g_1, \dots, g_t)
\]
is a basis of $F + G$; the formula follows by counting: $(r + s) +
(r + t) - r = r + s + t$.

$\mathcal{B}$ generates $F + G$: any $u + v$ ($u \in F$, $v \in G$)
expands through it. Freeness: suppose $\sum \alpha_i e_i + \sum
\beta_j f_j + \sum \gamma_k g_k = 0$. The vector $w = \sum \gamma_k
g_k = -\sum\alpha_i e_i - \sum\beta_j f_j$ lies in $G \cap F$, so it
expands on $(e_i)$ alone; but $w$ also expands on $(g_k)$, and in
the basis of $G$ these two expressions must coincide: all
$\gamma_k = 0$ (and the $e$-coordinates match). The relation reduces
to $\sum\alpha_i e_i + \sum\beta_j f_j = 0$, a relation in the basis
of $F$: all remaining coefficients vanish.
\end{proof}

\begin{example}\label{ex:b1:findim:planes}
Two distinct planes $F, G$ (dimension $2$) of $\R^3$ satisfy $F + G
= \R^3$ (their sum strictly contains a plane), so $\dim(F \cap G) =
2 + 2 - 3 = 1$: they always intersect along a line --- no
``parallel planes'' through the origin.
\end{example}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:b1:findim:1}
Give a basis and the dimension of:
\begin{enumerate}
  \item $F = \{(x,y,z) \in \R^3 : x + y + z = 0\}$;
  \item $G = \{(x,y,z,t) \in \R^4 : x = y,\ z = 2t\}$;
  \item $H = \{P \in \R_3[X] : P(1) = P'(1) = 0\}$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:findim:2}
Compute the rank of the family
$\bigl((1,1,1), (1,2,3), (3,5,7), (0,1,2)\bigr)$ in $\R^3$, and
extract a basis of its span.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:findim:3}
Complete the free family $\bigl((1,1,0,0), (0,0,1,1)\bigr)$ into a
basis of $\R^4$ using canonical vectors, and justify.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:findim:4}
Prove that $\bigl(1, X, X(X-1), X(X-1)(X-2)\bigr)$ is a basis of
$\R_3[X]$, and find the coordinates of $X^3$ in it.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:findim:5}
Let $F$ and $G$ be subspaces of dimensions $4$ and $5$ of a space
$E$ with $\dim E = 7$. What are the possible values of $\dim(F \cap
G)$? Give an instance realizing each value with $E = \R^7$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:findim:6}
Let $H = \{P \in \R_n[X] : P(1) = 0\}$. Prove that $H$ is a
hyperplane of $\R_n[X]$ (a subspace of dimension $n$), exhibit a
basis of $H$ \emph{(think of the factor theorem: $P = (X-1)Q$)}, and
give a supplementary line.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:findim:7}
Let $u_1, \dots, u_p$ be vectors of rank $r$. Prove that removing
one vector yields a family of rank $r$ or $r - 1$, and that
appending one vector yields rank $r$ or $r + 1$. Deduce that rank
changes by at most $1$ under any single insertion or deletion.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:findim:8}
Let $F_1 \subseteq F_2 \subseteq \dots \subseteq F_k$ be subspaces
of $E$ ($\dim E = n$) with $F_i \neq F_{i+1}$ for all $i$. Prove $k
\leq n + 1$. Deduce that a strictly increasing chain of subspaces
of $\R^n$ has length at most $n + 1$, and exhibit one of maximal
length.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:findim:9}
Let $E$ be of dimension $n$ and $F$, $G$ two hyperplanes (dimension
$n - 1$), $F \neq G$. Compute $\dim(F \cap G)$. Generalize: the
intersection of $k$ hyperplanes has dimension $\geq n - k$.
\end{exercise}
