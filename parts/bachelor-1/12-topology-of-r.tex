\chapter{Topology of the Real Line}\label{ch:b1:topology}

Limits keep referring to the same geometric vocabulary: points
``close to'' a set, sets ``without boundary leaks'', intervals in
which sequences cannot escape. This chapter fixes that vocabulary ---
open and closed sets, interior and closure, density --- on the real
line, and proves the compactness of segments in its sequential form.
The same notions, in normed vector spaces, are second-year material;
on $\R$ they are within reach and immediately useful for
\cref{ch:b1:continuity}.

\section{Open sets, closed sets}

\begin{definition}[Neighborhood, open set]\label{def:b1:topology:open}
A set $V \subseteq \R$ is a \emph{neighborhood}\index{neighborhood}
of $x \in \R$ when it contains an interval $\intoo{x - r}{x + r}$ for
some $r > 0$. A set $U \subseteq \R$ is \emph{open}\index{open set}
when it is a neighborhood of each of its points:
\[
\forall x \in U,\ \exists r > 0, \quad \intoo{x - r}{x + r}
\subseteq U .
\]
\end{definition}

\begin{example}\label{ex:b1:topology:open}
Open intervals are open: for $x \in \intoo{a}{b}$, take $r = \min(x -
a,\, b - x) > 0$. Half-lines $\intoo{a}{+\infty}$ are open; $\R$ and
$\emptyset$ are open (the latter vacuously). $\intcc{0}{1}$ is
\emph{not} open: no interval around $0$ stays inside.
\end{example}

\begin{proposition}[Stability of open sets]\label{prop:b1:topology:openstable}
Any union of open sets is open; a \emph{finite} intersection of open
sets is open. Infinite intersections may fail:
$\bigcap_{n \geq 1} \intoo{-\frac 1n}{\frac 1n} = \{0\}$, not open.
\end{proposition}

\begin{proof}
Union: if $x \in \bigcup_i U_i$, then $x \in U_{i_0}$ for some $i_0$,
and the interval provided by $U_{i_0}$ sits inside the union. Finite
intersection: if $x \in U_1 \cap \dots \cap U_k$, take $r = \min(r_1,
\dots, r_k) > 0$ of the radii provided by each $U_j$. For the
counterexample: any interval around $0$ contains some $\frac{1}{n}$
(Archimedes), hence leaves the intersection.
\end{proof}

\begin{definition}[Closed set]\label{def:b1:topology:closed}
A set $F \subseteq \R$ is \emph{closed}\index{closed set} when its
complement $\R \setminus F$ is open. By de Morgan and
\cref{prop:b1:topology:openstable}: any intersection of closed sets
is closed, finite unions of closed sets are closed.
\end{definition}

\begin{theorem}[Sequential characterization of closed sets]\label{thm:b1:topology:seqclosed}
$F$ is closed if and only if: for every sequence $(u_n)$ of points of
$F$ converging to some $\ell \in \R$, the limit $\ell$ belongs to
$F$. (``Closed'' $=$ ``stable under limits''.)
\end{theorem}

\begin{proof}
($\Rightarrow$) Let $F$ be closed, $u_n \in F$, $u_n \to \ell$, and
suppose $\ell \notin F$. The complement is open: some
$\intoo{\ell - r}{\ell + r}$ avoids $F$. But convergence puts $u_n$
in that interval for large $n$: contradiction with $u_n \in F$.

($\Leftarrow$) Suppose $F$ is not closed: the complement is not open,
so some $x \notin F$ has no interval $\intoo{x - r}{x + r}$ inside
the complement; taking $r = \frac{1}{n+1}$, choose $u_n \in F$ with
$\abs{u_n - x} < \frac{1}{n+1}$. Then $u_n \in F$, $u_n \to x \notin
F$: the sequential property fails.
\end{proof}

\begin{example}\label{ex:b1:topology:closed}
Segments $\intcc{a}{b}$, half-lines $\intco{a}{+\infty}$, finite
sets, $\Z$ (a convergent sequence of integers is eventually constant)
are closed. $\intoc{0}{1}$ is neither open (fails at $1$) nor closed
($\frac 1n \to 0 \notin$ the set): most sets are neither. $\R$ and
$\emptyset$ are both open and closed --- and they are the only such
subsets of $\R$ (\cref{exo:b1:topology:9}).
\end{example}

\section{Interior, closure, density}

\begin{definition}[Interior, closure, boundary]\label{def:b1:topology:closure}
Let $A \subseteq \R$.
\begin{itemize}
  \item A point $x$ is \emph{interior} to $A$ when $A$ is a
        neighborhood of $x$; the \emph{interior}
        $\mathring{A}$\index{interior} is the set of interior points.
  \item A point $x$ is \emph{adherent} to $A$ when every neighborhood
        of $x$ meets $A$; the \emph{closure}
        $\overline{A}$\index{closure} is the set of adherent points.
  \item The \emph{boundary} is $\partial A = \overline A \setminus
        \mathring A$.\index{boundary}
\end{itemize}
Then $\mathring A \subseteq A \subseteq \overline A$.
\end{definition}

\begin{proposition}[Main properties]\label{prop:b1:topology:closureprops}
\begin{enumerate}
  \item $\mathring A$ is the largest open set contained in $A$; $A$
        is open iff $A = \mathring A$.
  \item $\overline A$ is the smallest closed set containing $A$; $A$
        is closed iff $A = \overline A$.
  \item (Sequential characterization of adherence) $x \in \overline
        A$ if and only if $x$ is the limit of a sequence of points of
        $A$.
  \item Complementation exchanges the notions: $\R \setminus
        \overline A = \bigl(\R \setminus A\bigr)^{\!\circ}$.
\end{enumerate}
\end{proposition}

\begin{proof}
(4) $x \notin \overline A$ $\iff$ some neighborhood of $x$ avoids $A$
$\iff$ some interval around $x$ lies in $\R \setminus A$ $\iff$ $x$
is interior to $\R \setminus A$.

(1) $\mathring A$ is open: if $x \in \mathring A$, some
$\intoo{x-r}{x+r} \subseteq A$; every point $y$ of that interval has
a smaller interval around it inside it, hence inside $A$: the whole
interval is in $\mathring A$. Any open $U \subseteq A$ consists of
interior points of $A$, so $U \subseteq \mathring A$: largest. The
characterization of openness follows.

(2) By (4), $\overline A$ is the complement of an open set: closed;
minimality and the characterization follow from (1) applied to the
complement.

(3) If $u_n \in A$, $u_n \to x$: every neighborhood of $x$ contains
some $u_n \in A$, so $x \in \overline A$. Conversely, if $x \in
\overline A$: each interval $\intoo{x - \frac{1}{n+1}}{x +
\frac{1}{n+1}}$ meets $A$ at some $u_n$, and $u_n \to x$.
\end{proof}

\begin{example}\label{ex:b1:topology:closureexamples}
$\overline{\intoo{0}{1}} = \intcc{0}{1}$;
$\mathring{\intcc{0}{1}} = \intoo{0}{1}$; $\partial\intoo{0}{1} =
\{0, 1\}$. For $A = \{\frac 1n : n \in \N^*\}$: $\overline A = A
\cup \{0\}$, $\mathring A = \emptyset$, $\partial A = A \cup \{0\}$.
For $\Q$: by density (\cref{thm:b1:reals:density}) every real is
adherent to $\Q$, so $\overline{\Q} = \R$ while $\mathring{\Q} =
\emptyset$ (every interval contains irrationals): the boundary of
$\Q$ is all of $\R$.
\end{example}

\begin{definition}[Density, topological form]\label{def:b1:topology:dense}
$A$ is \emph{dense}\index{density} in $\R$ when $\overline A = \R$
--- equivalently, every nonempty open interval meets $A$;
equivalently (by \cref{prop:b1:topology:closureprops}~(3)), every
real is a limit of elements of $A$. Examples: $\Q$, $\R \setminus
\Q$, the dyadics (\cref{exo:b1:reals:8}), dense subgroups
(\cref{exo:b1:reals:9}).
\end{definition}

\section{Compactness of segments}

\begin{theorem}[Segments are sequentially compact]\label{thm:b1:topology:compact}
Let $a \leq b$. Every sequence of points of $\intcc{a}{b}$ has a
subsequence converging to a point \emph{of}
$\intcc{a}{b}$.\index{compactness}

More generally, the subsets of $\R$ with this property (every
sequence has a subsequence converging in the set) are exactly the
\emph{closed and bounded} sets.
\end{theorem}

\begin{proof}
A sequence in $\intcc{a}{b}$ is bounded, so Bolzano--Weierstrass
(\cref{thm:b1:seq:bw}) extracts a convergent subsequence; its limit
stays in $\intcc{a}{b}$ because segments are closed
(\cref{thm:b1:topology:seqclosed}).

General case. \emph{(Closed bounded $\Rightarrow$ compact)}: same
argument verbatim. \emph{(Compact $\Rightarrow$ closed and
bounded)}: if $F$ is unbounded, pick $u_n \in F$ with $\abs{u_n} \geq
n$; every subsequence is unbounded, hence divergent
(\cref{prop:b1:seq:first}): no convergent subsequence at all. If $F$
is not closed, take $u_n \in F$ with $u_n \to \ell \notin F$
(\cref{thm:b1:topology:seqclosed}): every subsequence converges to
$\ell \notin F$, so no subsequence converges \emph{in} $F$.
\end{proof}

\begin{remark}
This is the engine behind the extreme value theorem
(\cref{ch:b1:continuity}) and Heine's theorem on uniform continuity.
The name ``compact'' will acquire its general (covering) definition
in the second year; on $\R$, sequential compactness is all we need,
and ``compact $=$ closed $+$ bounded'' is the statement to remember.
\end{remark}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:b1:topology:1}
For each set, say whether it is open, closed, both, or neither (with
justification): $\intoo{0}{1} \cup \intoo{2}{3}$; $\;\intco{0}{1}$;
$\;\{0\} \cup \intcc{1}{2}$; $\;\R \setminus \Z$; $\;\Q \cap
\intoo{0}{1}$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:topology:2}
Determine $\mathring A$, $\overline A$ and $\partial A$ for:
$A = \intoc{0}{1} \cup \{2\}$; $\;A = \R \setminus \Q$;
$\;A = \bigl\{\frac{(-1)^n n}{n+1} : n \in \N\bigr\}$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:topology:3}
Prove that a finite set is closed, first via complements, then via
the sequential characterization.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:topology:4}
Prove that for any $A, B \subseteq \R$: $\overline{A \cup B} =
\overline A \cup \overline B$. Show by example that
$\overline{A \cap B}$ can differ from $\overline A \cap \overline
B$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:topology:5}
Let $u_n \to \ell$ in $\R$. Prove that the set $\{u_n : n \in \N\}
\cup \{\ell\}$ is closed (hence compact if we add that it is
bounded --- which it is).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:topology:6}
Let $U$ be open and $A$ arbitrary. Prove that $U + A = \{u + a\}$ is
open. Deduce that the sum of an open set and any set is open, and
contrast: exhibit two \emph{closed} sets whose sum is not closed.
\emph{(Try $\Z$ and $\sqrt 2\,\Z$, with \cref{exo:b1:reals:9}.)}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:topology:7}
A point $x \in A$ is \emph{isolated} in $A$ when some neighborhood of
$x$ meets $A$ only at $x$. Prove that every point of $\Z$ is isolated
in $\Z$, that $A = \{\frac1n\}$ has all its points isolated yet
$\overline A \neq A$, and that a set whose points are all isolated
has empty interior.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:topology:8}
Prove that the closure of a bounded set is bounded, and that
$\sup \overline A = \sup A$ for $A$ nonempty bounded above. Deduce
that $\sup A \in \overline A$: the supremum is always adherent.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:topology:9}
Prove that the only subsets of $\R$ that are both open and closed are
$\emptyset$ and $\R$. \emph{Hint: suppose $A$ is open, closed, with
$A \neq \emptyset$ and $\R \setminus A \neq \emptyset$; pick $a \in
A$, $b \notin A$, say $a < b$, and consider $s = \sup\,(A \cap
\intcc{a}{b})$; decide whether $s$ can belong to $A$ or to its
complement.}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:topology:10}
(Structure of open sets) Let $U \subseteq \R$ be open, nonempty. For
$x \in U$, let $I_x$ be the union of all open intervals containing
$x$ and contained in $U$. Prove that $I_x$ is an open interval, that
two sets $I_x$, $I_y$ are equal or disjoint, and that $U$ is a union
of \emph{countably many} pairwise disjoint open intervals \emph{(pick
a rational in each)}.
\end{exercise}
