\chapter{Plane Curves}\label{ch:b1:curves}

The calculus of \cref{ch:b1:derivative,ch:b1:taylor} was built for
functions $y = f(x)$; most curves of geometry and mechanics ---
trajectories, circles rolled on circles, orbits --- refuse that form
and come instead as \emph{parametrized} curves $t \mapsto (x(t),
y(t))$ or as \emph{polar} curves $r = r(\theta)$. This classical
chapter studies both, and closes with the \omterm{def:b1:curves:conics}{conics}.

\section{Parametrized curves}

\begin{definition}\label{def:b1:curves:def}
A \emph{parametrized curve}\index{parametrized curve} is a \omterm{def:b1:logic:map}{map} $f
\colon I \to \R^2$, $t \mapsto M(t) = (x(t), y(t))$, with $x, y$ of
class $C^1$ (at least) on the \omterm{prop:b1:reals:intervals}{interval} $I$. The \emph{velocity
vector} is $f'(t) = (x'(t), y'(t))$; the point $M(t_0)$ is
\emph{regular} when $f'(t_0) \neq (0,0)$, and the \emph{tangent}
there is the line through $M(t_0)$ directed by $f'(t_0)$.
\end{definition}

\begin{proof}[Why $f'$ directs the tangent]
By Taylor--Young componentwise, $M(t) = M(t_0) + (t - t_0) f'(t_0) +
o(t - t_0)$: the chord direction $\frac{M(t) - M(t_0)}{t - t_0}$
tends to $f'(t_0)$. The regularity hypothesis is what makes the
limit a \emph{direction}: if $f'(t_0) = 0$, the display collapses
to $M(t) = M(t_0) + o(t - t_0)$ and says nothing about how the
curve leaves the point --- whence the separate treatment of
singular points in the method below, where the first
\emph{nonvanishing} \omterm{def:b1:derivative:def}{derivative} takes over the role of $f'$.
Note also that the tangent line is a geometric object: any
reparametrization changes $f'$ by a nonzero scalar factor and
leaves the line unchanged.
\end{proof}

\begin{method}[Studying a parametrized curve]\label{met:b1:curves:study}
\begin{enumerate}
  \item \emph{Reduce the domain} using symmetries: relations between
        $M(-t)$, $M(t + T)$, \dots\ and $M(t)$ (reflections,
        translations, periodicity), then draw only the reduced part.
  \item \emph{Variations}: tabulate signs of $x'$ and $y'$ jointly;
        the curve moves right/left as $x'$, up/down as $y'$.
  \item \emph{Remarkable points}: horizontal tangent ($y' = 0 \neq
        x'$), vertical tangent ($x' = 0 \neq y'$); at a
        \emph{singular} point ($f' = 0$), examine higher \omterm{def:b1:derivative:def}{derivatives}
        for the tangent direction.
  \item \emph{Asymptotic behavior} at the ends of $I$, then sketch.
\end{enumerate}
\end{method}

\begin{example}[Asymptotic branches, worked]
\label{ex:b1:curves:asymptotes}
$x(t) = t$, $y(t) = t + \dfrac1t$ on $\intoo{0}{+\infty}$. As $t
\to 0^{+}$: $x \to 0$ while $y \to +\infty$ --- the curve climbs
along the \emph{vertical asymptote} $x = 0$ (finite limit for
one coordinate, infinite for the other). As $t \to +\infty$:
both \omterm{prop:b1:vspaces:coordinates}{coordinates} blow up, so test a line: the difference
\[
y(t) - x(t) = \frac1t \longrightarrow 0^{+}
\]
exhibits the \emph{oblique asymptote} $y = x$, approached from
above. In between, $y' = 1 - \frac1{t^2}$ vanishes at $t = 1$:
the point $(1, 2)$ is the low point of the branch, and $x' = 1 >
0$ throughout, so the curve always advances rightward. One arc,
two asymptotes, one minimum: a complete picture from three
computations --- with the general recipe visible underneath:
when $x, y \to \infty$ together, examine $y/x$ for a candidate
slope (here $\to 1$), then $y - (\text{slope})\,x$ for the
intercept and the side of approach.
\end{example}

\begin{example}[The astroid]\label{ex:b1:curves:astroid}
$x(t) = \cos^3 t$, $y(t) = \sin^3 t$.\index{astroid} Symmetries:
$M(t + 2\pi) = M(t)$ (study a period); $M(-t)$ is the reflection of
$M(t)$ in the $x$-axis; $M(\pi - t)$ in the $y$-axis; $M(\frac\pi2 -
t)$ in the diagonal $y = x$: it suffices to study $t \in
\intcc{0}{\frac\pi4}$ and unfold.

Velocity: $f'(t) = 3\sin t\cos t\,(-\cos t, \sin t)$. On
$\intoo{0}{\frac\pi2}$ all points are regular with tangent directed
by $(-\cos t, \sin t)$; at $t = 0$ (the point $(1,0)$) the velocity
vanishes: a \emph{cusp}\index{cusp}, where the curve reverses along
the tangent direction $(-1, 0)\cdot(\pm)$ --- by symmetry the four
\omterm{ex:b1:curves:astroid}{cusps} sit at $(\pm1, 0), (0, \pm1)$.
\end{example}

\begin{example}[A figure-eight, studied in full]
\label{ex:b1:curves:lissajous}
$x(t) = \sin t$, $y(t) = \sin 2t$ (a \emph{Lissajous curve}).
\emph{Symmetries}: $M(t + \pi) = (-x(t), y(t))$ (reflection in
the $y$-axis), $M(-t) = (-x(t), -y(t))$ (central symmetry),
$M(\pi - t) = (x(t), -y(t))$ (reflection in the $x$-axis): it
suffices to study $t \in \intcc{0}{\frac\pi2}$ and unfold.
\emph{Variations}: $x' = \cos t \geq 0$ throughout, while $y' =
2\cos 2t$ is positive before $t = \frac\pi4$ and negative after:
the arc climbs rightward to the summit
$\bigl(\frac{\sqrt2}2,\ 1\bigr)$ at $t = \frac\pi4$ (horizontal
tangent), then descends rightward to $(1, 0)$ at $t =
\frac\pi2$, where $x' = 0 \neq y'$: vertical tangent.
\emph{Double point}: $M(0) = M(\pi) = (0,0)$, with two different
velocities
\[
f'(0) = (1,\ 2), \qquad f'(\pi) = (-1,\ 2) :
\]
two \emph{regular} branches crossing at the origin at distinct
angles --- a double point, not a singular point: each passage is
perfectly smooth, the two passages merely share their location.
The whole curve is the figure-eight below.
\end{example}

\section{Polar curves}

\begin{definition}\label{def:b1:curves:polar}
A \emph{polar curve}\index{polar curve} is given by $r = r(\theta)$:
the point of parameter $\theta$ is
\[
M(\theta) = r(\theta)\,\vec u(\theta),
\qquad \vec u(\theta) = (\cos\theta, \sin\theta).
\]
With $\vec v(\theta) = (-\sin\theta, \cos\theta) = \vec
u\,'(\theta)$, the velocity is
\[
M'(\theta) = r'(\theta)\, \vec u(\theta) + r(\theta)\, \vec
v(\theta) .
\]
Consequences: where $r \neq 0$, the point is regular, and the
tangent makes with the ray the angle $V$ given by $\tan V =
\frac{r}{r'}$ (angle between $M'$ and $\vec u$); where $r(\theta_0)
= 0$, the curve passes through the origin \emph{with tangent the ray
$\theta = \theta_0$} (direction $\vec u(\theta_0)$, read off
$M'(\theta_0) = r'(\theta_0)\vec u(\theta_0)$, or from the limit
chord: the chord from $O$ to $M(\theta)$ is carried by $\vec
u(\theta)$ itself, which tends to $\vec u(\theta_0)$ --- so the
rule holds even when $r'(\theta_0) = 0$ and the point is
singular, which is why polar \omterm{ex:b1:curves:astroid}{cusps} at the origin, like the
cardioid's, get their tangent for free).
\end{definition}

\begin{example}[The cardioid]\label{ex:b1:curves:cardioid}
$r(\theta) = 1 + \cos\theta$.\index{cardioid} Symmetry: $r(-\theta)
= r(\theta)$: reflection in the $x$-axis; study $\theta \in
\intcc{0}{\pi}$. $r$ decreases from $2$ to $0$; at $\theta = \pi$,
$r = 0$: the curve reaches the origin tangentially to the ray
$\theta = \pi$ (the $x$-axis), forming a \omterm{ex:b1:curves:astroid}{cusp} there --- the heart's
point. Tangent at $\theta = 0$: $r' = 0$, so $\tan V = \infty$:
perpendicular to the axis.

The angle $V$ deserves one more reading. At $\theta =
\frac\pi2$: $r = 1$ and $r' = -1$, so $\tan V = \frac{r}{r'} =
-1$: the tangent makes three-quarters of a right angle with the
outgoing ray --- the curve is already bending back toward its
\omterm{ex:b1:curves:astroid}{cusp}. The formula $\tan V = r/r'$ delivers tangent directions
along the whole curve with no computation of $M'(\theta)$
whatsoever: it is the polar analogue of reading a slope.
\end{example}

\begin{example}[Polar to cartesian: a hidden circle]
\label{ex:b1:curves:polarcircle}
What is the \omterm{def:b1:curves:polar}{polar curve} $r = 2\cos\theta$? Multiply by $r$:
$r^2 = 2r\cos\theta$, i.e.\ $x^2 + y^2 = 2x$, i.e.
\[
(x - 1)^2 + y^2 = 1 :
\]
the circle of center $(1, 0)$ and radius $1$, passing through
the origin. Bookkeeping matters: as $\theta$ runs over
$\intoc{-\frac\pi2}{\frac\pi2}$ the whole circle is swept
exactly once ($r$ vanishes at both ends), and at $\theta =
\pm\frac\pi2$ the rule ``tangent at the origin along the ray
$\theta = \theta_0$'' gives a \emph{vertical} tangent there ---
matching the geometry, since the vertical axis is indeed tangent
to this circle at $O$. For $\theta$ beyond that range, $r < 0$
retraces the same circle: a reminder that a \omterm{def:b1:curves:polar}{polar curve} is a
\emph{parametrized} object, allowed to pass over itself.
\end{example}

\section{Arc length}

\begin{definition}[Length of an arc]\label{def:b1:curves:arclength}
The \emph{length}\index{arc length} of a $C^1$ arc $f \colon
\intcc{a}{b} \to \R^2$ is the \omterm{thm:b1:integration:def}{integral} of the speed:
\[
L = \int_a^b \norm{f'(t)}\,\dd t
= \int_a^b \sqrt{x'(t)^2 + y'(t)^2}\;\dd t .
\]
(Motivation: on a small \omterm{prop:b1:reals:intervals}{interval}, $M(t + h) \approx M(t) + h
f'(t)$, so the arc is close to a polygon whose segment lengths sum
to a Riemann sum of $\norm{f'}$,
\cref{thm:b1:integration:riemann}.) For a \omterm{def:b1:curves:polar}{polar curve} $r =
r(\theta)$, the velocity $r'\vec u + r\vec v$ has \omterm{def:b1:euclid:orthogonal}{orthogonal}
components, so
\[
L = \int_{\theta_1}^{\theta_2}
\sqrt{r'(\theta)^2 + r(\theta)^2}\;\dd\theta .
\]
The length does not depend on the (monotone, $C^1$)
parametrization chosen: substituting $t = \varphi(s)$ in the
\omterm{thm:b1:integration:def}{integral} (\cref{thm:b1:integration:parts}) multiplies $f'$ by
$\varphi'$ and $\dd t$ by $\varphi'^{-1}$.
\end{definition}

\begin{example}[Sanity check: the circle]\label{ex:b1:curves:circlelength}
For $f(t) = (R\cos t, R\sin t)$ on $\intcc{0}{2\pi}$: $\norm{f'} =
R$, so $L = 2\pi R$ --- the definition returns the circumference.
Reparametrization test: the \omterm{def:b1:logic:map}{map} $g(t) = (R\cos 2t, R\sin 2t)$ on
$\intcc{0}{\pi}$ draws the \emph{same} circle at doubled speed
$\norm{g'} = 2R$, and
\[
\int_0^{\pi} 2R\,\dd t = 2\pi R
\]
again: half the time, twice the speed, same \omterm{def:b1:curves:arclength}{length} --- the
invariance promised in the definition, watched once on numbers.
(Running $g$ on all of $\intcc{0}{2\pi}$ would give $4\pi R$: a
curve traversed twice is twice as long \emph{as a journey};
\omterm{def:b1:curves:arclength}{length} measures the parametrized path, and honest bookkeeping of
the \omterm{prop:b1:reals:intervals}{interval} is part of the computation.)
For the astroid $(\cos^3t, \sin^3t)$: $\norm{f'} = 3\abs{\sin
t\cos t} = \tfrac32\abs{\sin 2t}$, and by symmetry $L = 4\int_0^{\pi/2}
\tfrac32\sin 2t\,\dd t = 6$: a curve drawn inside the unit circle,
of \omterm{def:b1:curves:arclength}{length} $6 < 2\pi$. The weekend problem measures the most famous
arch of all.
\end{example}

\section{Conics}

\begin{definition}[Focus--directrix definition]\label{def:b1:curves:conics}
Fix a point $F$ (\emph{focus}), a line $D$ not through $F$
(\emph{directrix}) and $e > 0$ (\emph{eccentricity}). The
\emph{conic}\index{conic} of these data is
\[
\mathcal{C} = \{M : d(M, F) = e\; d(M, D)\}:
\]
an \emph{ellipse} for $e < 1$, a \emph{parabola} for $e = 1$, a
\emph{hyperbola} for $e > 1$. (The circle appears as a degenerate
limit $e \to 0$.)
\end{definition}

\begin{example}[The definition, checked on a parabola]
\label{ex:b1:curves:parabolacheck}
Take the parabola $y^2 = 4x$, i.e.\ $2p = 4$: focus $F = (1, 0)$
and directrix $D : x = -1$ (the reduced form
\cref{thm:b1:curves:reduced} puts them at $\pm\frac p2$). At the
point $M = (1, 2)$ of the curve:
\[
MF = \sqrt{(1-1)^2 + 2^2} = 2,
\qquad
d(M, D) = 1 - (-1) = 2 :
\]
equal, as $e = 1$ demands. At $M' = (4, 4)$: $MF' = \sqrt{9 +
16} = 5$ and $d(M', D) = 5$ again. The focus--directrix
definition is not an abstraction: it is a pair of distances one
can measure on any point, and the algebra of the reduced
equations is nothing but this measurement done once and for
all.
\end{example}

\begin{theorem}[Reduced equations]\label{thm:b1:curves:reduced}
In a well-chosen \omterm{def:b1:euclid:orthogonal}{orthonormal} frame:
\[
\text{ellipse: } \frac{x^2}{a^2} + \frac{y^2}{b^2} = 1
\quad (a \geq b > 0),
\qquad
\text{hyperbola: } \frac{x^2}{a^2} - \frac{y^2}{b^2} = 1,
\qquad
\text{parabola: } y^2 = 2px,
\]
with, for the ellipse: foci at $(\pm c, 0)$, $c = \sqrt{a^2 - b^2}$,
$e = \frac ca$, and the bifocal characterization $MF + MF' = 2a$;
for the hyperbola: $c = \sqrt{a^2 + b^2}$, $e = \frac ca$,
$\abs{MF - MF'} = 2a$, asymptotes $y = \pm\frac ba x$.
\end{theorem}

\begin{proof}
Take the focus at the origin and the directrix vertical, $x = -h$
($h > 0$): the condition $MF^2 = e^2 d(M, D)^2$ reads $x^2 + y^2 =
e^2 (x + h)^2$. For $e = 1$: $y^2 = 2hx + h^2$, a parabola after the
shift $x \mapsto x - \frac h2$ (so $p = h$). For $e \neq 1$:
completing the square in $x$,
\[
(1 - e^2)\Bigl(x - \frac{e^2 h}{1 - e^2}\Bigr)^{\!2} + y^2
= e^2h^2 + \frac{e^4 h^2}{1 - e^2}
= \frac{e^2 h^2}{1 - e^2} .
\]
Set $X = x - \frac{e^2h}{1-e^2}$ (a shift of frame). When $e < 1$,
divide by the positive right-hand side:
$\frac{X^2}{a^2} + \frac{y^2}{b^2} = 1$ with $a = \frac{eh}{1 -
e^2}$, $b = \frac{eh}{\sqrt{1-e^2}}$; when $e > 1$, both sides of
the display flip sign appropriately and the same division gives
$\frac{X^2}{a^2} - \frac{y^2}{b^2} = 1$ with $a = \frac{eh}{e^2 -
1}$, $b = \frac{eh}{\sqrt{e^2 - 1}}$. The stated values of $c$
follow ($c^2 = a^2 - b^2$ or $a^2 + b^2$ gives $c = ea$ in both
cases, placing the focus correctly), and the bifocal properties are
direct verifications on the reduced equations. In detail for the
ellipse, with $F' = (c, 0)$ and $M = (X, y)$ on the curve:
\[
MF'^2 = (X - c)^2 + y^2
= X^2 - 2cX + c^2 + b^2\Bigl(1 - \frac{X^2}{a^2}\Bigr)
= \frac{c^2}{a^2}X^2 - 2cX + a^2
= (a - eX)^2 ,
\]
using $c^2 + b^2 = a^2$ and $e = \frac ca$; since $\abs X \leq
a$ and $e < 1$, $a - eX > 0$, so $MF' = a - eX$ with no square
root ever extracted. The mirror computation gives $MF = a + eX$,
whence $MF + MF' = 2a$, constant. For the hyperbola the same
algebra yields $MF = \abs{a + eX}$ and $MF' = \abs{eX - a}$,
with difference $\pm2a$ according to the branch.
\end{proof}

\begin{example}\label{ex:b1:curves:conicexample}
$\frac{x^2}{25} + \frac{y^2}{9} = 1$: ellipse, $a = 5$, $b = 3$, $c
= 4$: foci $(\pm4, 0)$, eccentricity $\frac45$. Its parametrization:
$(5\cos t, 3\sin t)$ --- a circle stretched anisotropically; the
sum of distances to the foci of any of its points is $10$.
\end{example}

\begin{example}[Reading an orbit from its polar equation]
\label{ex:b1:curves:orbit}
The polar \omterm{def:b1:curves:conics}{conic} $r = \dfrac{1}{1 + \frac12\cos\theta}$
(\cref{exo:b1:curves:7} with $p = 1$, $e = \frac12$) is an
ellipse with a \emph{focus at the origin} --- the geometry of a
planetary orbit with the sun at $O$. Extract everything from
$p$ and $e$: from the reduction of
\cref{thm:b1:curves:reduced}, $a = \dfrac{p}{1 - e^2} =
\dfrac{1}{3/4} = \dfrac43$ and $c = ea = \dfrac23$. The two
apsides check it with no theory at all:
\[
r(0) = \frac{1}{3/2} = \frac23 = a - c
\quad (\text{perihelion}),
\qquad
r(\pi) = \frac{1}{1/2} = 2 = a + c
\quad (\text{aphelion}),
\]
and their sum $\frac23 + 2 = \frac83 = 2a$ recovers the major
axis. The polar form is the natural one whenever a focus is
physically distinguished; the reduced cartesian form, whenever
the symmetry axes are. Converting between the two is exactly
what the completed-square computation of the theorem does.

Eccentricity as a dial, finally: keep $p = 1$ and turn $e$ in $r
= \frac{1}{1 + e\cos\theta}$. At $e = 0$: the circle $r = 1$. At
$e = \frac12$: the ellipse just studied, $r$ oscillating between
$\frac23$ and $2$. At $e = 1$: $r(\theta) \to \infty$ as $\theta
\to \pi$ --- the curve no longer closes: a parabola, its
farthest point pushed to infinity. At $e = 2$: the denominator
vanishes at $\cos\theta = -\frac12$, and only $\theta \in
\intoo{-\frac{2\pi}3}{\frac{2\pi}3}$ survives: one branch of a
hyperbola, escaping along two asymptotic directions. One
formula, the entire \omterm{def:b1:curves:conics}{conic} family, and the transition points $e =
1$ visible as the moment the denominator first reaches zero.
\end{example}

\begin{remark}[Common pitfalls]
\label{rem:b1:curves:pitfalls}
\emph{Double points are not singular points}: at a
self-crossing, each branch is regular; ``singular'' refers to
$f'(t_0) = 0$ for one value of the parameter
(\cref{ex:b1:curves:lissajous} vs the astroid's \omterm{ex:b1:curves:astroid}{cusps}).
\emph{Vertical tangent versus \omterm{ex:b1:curves:astroid}{cusp}}: $x'(t_0) = 0 \neq y'(t_0)$
is a regular point with vertical tangent; only $x' = y' = 0$
demands the higher-order expansion.
\emph{Negative radii plot on the opposite ray}: for $r(\theta) <
0$ the point is $-\abs{r}\,\vec u(\theta)$, at angle $\theta +
\pi$ (\cref{exo:b1:curves:5}); forgetting this loses inner loops
or doubles curves.
\emph{\omterm{def:b1:curves:arclength}{Arc length} integrates $\norm{f'}$, not $f'$}: split the
\omterm{thm:b1:integration:def}{integral} at the zeros of the speed --- for the astroid,
integrating $\frac32\sin 2t$ over a full period without absolute
values gives $0$, not $6$.
\emph{\omterm{def:b1:curves:conics}{Conics} must be reduced before being read}: in $9x^2 +
25y^2 - 36x - 50y - 164 = 0$, neither the axes, nor the center,
nor the eccentricity is visible until the squares are completed
(\cref{exo:b1:curves:6}); and a vanishing quadratic part on one
variable means parabola, not ``degenerate ellipse''.
\end{remark}

\begin{remark}[Where these curves go]
\label{rem:b1:curves:whereused}
\omterm{def:b1:curves:def}{Parametrized curves} are the language of mechanics: trajectories
are curves, velocity vectors are tangent vectors, and the \omterm{def:b1:curves:arclength}{arc
length} of \cref{def:b1:curves:arclength} is the distance run. The
\omterm{def:b1:curves:conics}{conics} reappear wherever an inverse-square law acts --- planetary
orbits are ellipses with the sun at a focus. In
\cref{ch:b1:multivar}, curves become the level \omterm{def:b1:logic:sets}{sets} of functions
of two variables, and the tangent of this chapter meets the
gradient of the next. The Year~2 volume adds curvature and the
local canonical form of a curve; the weekend problem below
already extracts, with this year's tools only, everything the
seventeenth century knew about its most celebrated curve.
\end{remark}

\begin{remark}[Perspectives inside Book 3]
\label{rem:b1:curves:perspectives}
This chapter consumes the whole first half of the volume and
feeds the last chapter. \emph{Consumed}: tangent vectors are
\omterm{def:b1:derivative:def}{derivatives} (\cref{ch:b1:derivative}), \omterm{def:b1:curves:arclength}{arc length} and areas are
\omterm{thm:b1:integration:def}{integrals} (\cref{ch:b1:integration}), \omterm{ex:b1:curves:astroid}{cusps} are settled by
Taylor expansions (\cref{ch:b1:taylor}), the tautochrone is a
\omterm{def:b1:linmaps:def}{linear} differential equation (\cref{ch:b1:diffeq}), and every
distance and angle is Euclidean (\cref{ch:b1:euclid}).
\emph{Fed}: in \cref{ch:b1:multivar}, a curve $t \mapsto (x(t),
y(t))$ drawn inside a level \omterm{def:b1:logic:sets}{set} $\{f = c\}$ is differentiated by
the chain rule, and the resulting identity $\langle \nabla f,\
f'(t)\rangle = 0$ marries this chapter's tangent vectors to the
next one's gradients --- the tangent line of the curve and the
normal direction of the surface are the same computation seen
from both banks.
\end{remark}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:b1:curves:1}
Study and sketch the curve $x(t) = t^2$, $y(t) = t^3$ (symmetries,
variations, behavior at the singular point $t = 0$).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:curves:2}
For the cycloid $x(t) = t - \sin t$, $y(t) = 1 - \cos t$ (the
trajectory of a point of a rolling wheel of radius $1$): identify
the period-translation symmetry, the singular points, and the
tangent direction at $t = 0$ \emph{(expand $x$ and $y$ to the first
nonzero orders)}.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:curves:3}
Give the tangent line to the curve $(\cos^3 t, \sin^3 t)$ at $t =
\frac\pi4$, and check that the segment of this tangent cut by the
axes has \omterm{def:b1:curves:arclength}{length} $1$ --- a famous property of the astroid (true at
every regular point).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:curves:4}
Sketch the \omterm{def:b1:curves:polar}{polar curves} $r = \cos 2\theta$ (four-petaled rose) and
$r = \frac{1}{\cos\theta}$ on $\intoo{-\frac\pi2}{\frac\pi2}$
(recognize a line).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:curves:5}
Study the \omterm{def:b1:curves:polar}{polar curve} $r = 1 + 2\cos\theta$ (a lima\c{c}on): domain
where $r \geq 0$ vs $r < 0$ (points plotted with negative radius
sit on the opposite ray), passage through the origin, inner loop,
sketch.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:curves:6}
Identify the \omterm{def:b1:curves:conics}{conic} $9x^2 + 25y^2 - 36x - 50y - 164 = 0$: reduce by
completing squares, give center, semi-axes, foci, eccentricity.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:curves:7}
Prove that the polar equation of a \omterm{def:b1:curves:conics}{conic} with focus at the origin
is
\[
r = \frac{p}{1 + e\cos\theta}
\]
(directrix vertical at distance $\frac pe$ right of the focus).
Which values of $\theta$ are allowed when $e > 1$?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:curves:8}
From the bifocal property $MF + MF' = 2a$, deduce the ``gardener's
construction'' of the ellipse, and prove that the tangent at $M$
makes equal angles with $MF$ and $MF'$ \emph{(reflection property;
differentiate $\norm{M(t) - F} + \norm{M(t) - F'} = 2a$ and
interpret the vanishing sum of unit-vector products)}.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:curves:9}
The \emph{lemniscate of Bernoulli} is the \omterm{def:b1:curves:polar}{polar curve} $r^2 = \cos
2\theta$ (take $r = \sqrt{\cos2\theta}$ where defined).
\begin{enumerate}
  \item Give its domain, symmetries, tangents at the origin, and
        sketch it.
  \item Prove that it is the locus of points $M$ with $MF \cdot MF'
        = \frac12$ where $F, F' = \bigl(\pm\frac{1}{\sqrt2},
        0\bigr)$. \emph{(Compute $MF^2\,MF'^2$ in polar
        \omterm{prop:b1:vspaces:coordinates}{coordinates}.)}
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:curves:10}
Compute the total \omterm{def:b1:curves:arclength}{length} of the cardioid $r = 1 + \cos\theta$
using the polar formula of \cref{def:b1:curves:arclength}
\emph{(the identity $1 + \cos\theta = 2\cos^2\frac\theta2$ turns
the square root into $2\abs{\cos\frac\theta2}$)}.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:curves:11}
Show that the tangent to the ellipse $(a\cos t, b\sin t)$ at the
point of parameter $t$ has equation
\[
\frac{x\cos t}{a} + \frac{y\sin t}{b} = 1 ,
\]
and deduce the tangent to $\frac{x^2}{a^2} + \frac{y^2}{b^2} = 1$
at a point $(x_0, y_0)$ of the ellipse: $\frac{x\,x_0}{a^2} +
\frac{y\,y_0}{b^2} = 1$ (the ``splitting of the squares'' rule).
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:curves:12}
The \emph{logarithmic spiral} is the \omterm{def:b1:curves:polar}{polar curve} $r =
\eu^{k\theta}$ ($k > 0$ fixed, $\theta \in \R$).
\begin{enumerate}
  \item Show that the angle $V$ between the radius and the
        tangent is \emph{constant} ($\tan V = \frac1k$) --- the
        spiral crosses every ray at the same angle.
  \item Show that rotating the spiral by an angle $c$ \omterm{def:b1:logic:map}{maps} it
        onto its scaling by the factor $\eu^{-kc}$: every
        rotation of the spiral is a magnification of it
        (self-similarity).
  \item Compute the \omterm{def:b1:curves:arclength}{length} of the arc $\theta \in
        \intoc{-\infty}{\theta_0}$ (as a limit of \omterm{def:b1:curves:arclength}{lengths} on
        $\intcc{A}{\theta_0}$, $A \to -\infty$) and observe that
        it is finite: a curve spiraling infinitely many times
        around the origin, of finite \omterm{def:b1:curves:arclength}{length}.
\end{enumerate}
\end{exercise}

\section{Problem: the cycloid, queen of curves}

\begin{problem}\label{pb:b1:curves:1}
A wheel of radius $R$ rolls without slipping along the $x$-axis;
the point of the rim initially at the origin \omterm{def:b1:matrices:transpose}{traces} the
\emph{cycloid}. The seventeenth century fought over this curve
--- Galileo weighed paper cutouts of it, Wren measured it,
Roberval computed its area, Huygens built clocks on it --- and
every one of their results is within reach of this chapter. We
prove the four classics: the tangent construction, Wren's \omterm{def:b1:curves:arclength}{length}
$8R$, the area $3\pi R^2$, and Huygens' tautochrone
property.\index{cycloid}\index{tautochrone}

\medskip
\noindent\textbf{Part I --- Rolling and the tangent.}
\begin{enumerate}
  \item After the wheel has turned by an angle $t$, its center
        sits at $\Omega(t) = (Rt, R)$ (rolling without slipping:
        contact distance $=$ arc rolled). Show that the marked
        point is at
        \[
        M(t) = \bigl(R(t - \sin t),\; R(1 - \cos t)\bigr).
        \]
  \item Compute $f'(t)$ and show $\norm{f'(t)} =
        2R\,\abs{\sin\frac t2}$; locate the singular points
        (\omterm{ex:b1:curves:astroid}{cusps} --- cf.\ \cref{exo:b1:curves:2}) and the top of
        each arch.
  \item Let $C(t) = (Rt, 0)$ be the contact point and $T(t) =
        (Rt, 2R)$ the top of the wheel. Prove that for $0 < t <
        2\pi$ the vector $M - C$ is \emph{normal} to the curve at
        $M(t)$ and the vector $T - M$ is \emph{tangent}: to draw
        the tangent to a cycloid, join the point to the top of
        its rolling circle. \emph{(Factor everything through
        $\sin\frac t2$ and $\cos\frac t2$.)}
  \item Interpret question 3 kinematically: the contact point is
        the \emph{instantaneous center of rotation}, and the
        speed of $M$ equals its distance to $C$ (for unit angular
        velocity). Verify $\norm{M - C} = 2R\abs{\sin\frac t2}$.
  \item Prove the height--speed relation
        \[
        \norm{f'(t)}^2 = 2R\,y(t) :
        \]
        on a cycloid traversed at unit angular velocity, the
        speed at each point is exactly the free-fall speed for a
        drop equal to the current height. (Keep this for
        Part IV.)
\end{enumerate}

\medskip
\noindent\textbf{Part II --- Wren's theorem: the arch has \omterm{def:b1:curves:arclength}{length}
$8R$.}
\begin{enumerate}[resume]
  \item Using \cref{def:b1:curves:arclength}, compute the \omterm{def:b1:curves:arclength}{length}
        of one arch:
        \[
        L = \int_0^{2\pi} 2R\sin\frac t2\,\dd t = 8R
        \]
        (Wren's theorem, 1658). Four wheel diameters, and no
        $\pi$ anywhere.
  \item Compute the \omterm{def:b1:curves:arclength}{arc length} from the \omterm{ex:b1:curves:astroid}{cusp}: $s(t) =
        4R\bigl(1 - \cos\frac t2\bigr)$, and check $s(2\pi) =
        8R$.
  \item Now measure the arc from the \emph{apex} $t = \pi$:
        $\sigma(t) = \abs{4R\cos\frac t2}$. Prove the intrinsic
        relation
        \[
        \sigma^2 = 8R\,\bigl(2R - y\bigr) :
        \]
        the squared arc distance from the top is proportional to
        the height drop below the top.
  \item Sanity checks: recover from question 8 that the half-arch
        from apex to \omterm{ex:b1:curves:astroid}{cusp} has \omterm{def:b1:curves:arclength}{length} $4R$, and compare with the
        astroid computation of \cref{ex:b1:curves:circlelength}
        --- both curves have algebraic, $\pi$-free \omterm{def:b1:curves:arclength}{lengths};
        explain what makes this possible even though both are
        built from circles. \emph{(Look at the form of
        $\norm{f'}$.)}
\end{enumerate}

\medskip
\noindent\textbf{Part III --- Roberval's area: $3\pi R^2$.}
\begin{enumerate}[resume]
  \item Justify that the area between one arch and the ground is
        $A = \int_0^{2\pi} y(t)\,x'(t)\,\dd t$ \emph{(the
        substitution $x = x(t)$ in $\int y\,\dd x$,
        \cref{thm:b1:integration:parts}; $x$ is increasing)}.
  \item Compute
        \[
        A = R^2\int_0^{2\pi}(1 - \cos t)^2\,\dd t = 3\pi R^2 :
        \]
        exactly \emph{three} times the area of the wheel ---
        the ratio Galileo had guessed by weighing.
  \item Same method for the astroid $(\cos^3t, \sin^3t)$: show
        that the enclosed area is $\frac{3\pi}8$
        \emph{(linearize $\sin^4 t\cos^2 t$; only the constant
        term survives over a full period)}.
  \item Sanity check the formula of question 10 on the upper unit
        semicircle $(\cos t, \sin t)$, $t$ from $\pi$ to $0$:
        does it return $\frac\pi2$?
\end{enumerate}

\medskip
\noindent\textbf{Part IV --- Huygens' tautochrone.} Flip the
arch: a frictionless bead slides, under gravity $g$, inside the
cycloidal bowl
\[
x(t) = R(t + \sin t), \qquad y(t) = R(1 - \cos t)
\qquad (t \in \intcc{-\pi}{\pi}),
\]
whose lowest point is the origin ($y$ measured upward).
\begin{enumerate}[resume]
  \item Compute the speed $\norm{f'(t)} = 2R\cos\frac t2$, the
        \omterm{def:b1:curves:arclength}{arc length} from the bottom $s(t) = 4R\sin\frac t2$, and
        prove the key identity
        \[
        y = \frac{s^2}{8R} .
        \]
  \item The bead released at rest from the point of parameter
        $t_0 > 0$ obeys energy conservation: if $s(\tau)$ denotes
        its arc position at time $\tau$, then $\frac12\bigl(
        \frac{\dd s}{\dd\tau}\bigr)^2 + g\,y = g\,y_0$. Rewrite
        this, using question 14, as
        \[
        \Bigl(\frac{\dd s}{\dd\tau}\Bigr)^{\!2}
        = \frac{g}{4R}\,\bigl(s_0^2 - s^2\bigr),
        \qquad s_0 = s(t_0).
        \]
  \item Differentiate with respect to $\tau$ and obtain the
        harmonic oscillator
        \[
        \frac{\dd^2 s}{\dd\tau^2} = -\frac{g}{4R}\,s ;
        \]
        solve it with \cref{thm:b1:diffeq:homogeneous2} and the
        initial conditions: $s(\tau) = s_0\cos(\omega\tau)$,
        $\omega = \sqrt{g/4R}$.
  \item Deduce the \emph{tautochrone property} (Huygens, 1659):
        the time to reach the bottom,
        \[
        T_{\downarrow} = \frac{\pi}{2\omega}
        = \pi\sqrt{\frac Rg}\,,
        \]
        does not depend on the release point --- beads released
        together from any two heights of the bowl arrive
        together.
  \item Verify by substitution that $s(\tau) =
        s_0\cos\omega\tau$ satisfies the first-order energy
        equation of question 15 exactly (not only the
        differentiated one), and explain in one sentence why a
        circular pendulum is only \emph{approximately}
        isochronous while the cycloid is exactly so.
  \item Numerically: what radius $R$ makes the descent time
        exactly one second ($g = 9.81$)? Note how close the
        answer is to one meter, and how the full oscillation
        period $2\pi\sqrt{4R/g}$ compares with the small-angle
        pendulum formula $2\pi\sqrt{\ell/g}$ for $\ell = 4R$.
\end{enumerate}

\medskip
\noindent\textbf{Part V --- Dividends, and synthesis.}
\begin{enumerate}[resume]
  \item Apply the tangent rule of question 3 at $t =
        \frac\pi2$ (take $R = 1$): compute $M$, $T$, the
        direction of $MT$, and check it against
        $f'(\frac\pi2)$.
  \item (Trochoids) Mark instead a point at distance $d$ from
        the center ($d \neq R$): the curve is $x = Rt -
        d\sin t$, $y = R - d\cos t$. Show that for $d < R$ the
        curve is regular everywhere and is the graph of no
        singular behavior ($x' > 0$: it advances), while for $d
        > R$ the abscissa $x'$ changes sign and the curve makes
        loops --- the flanged railway wheel whose rim points
        travel \emph{backwards}.
  \item (Brachistochrone teaser) From the \omterm{ex:b1:curves:astroid}{cusp} $(\pi R, 2R)$ of
        the bowl to the bottom, compare the cycloid descent time
        $\pi\sqrt{R/g}$ with the time along the straight chute
        joining the same points \emph{(constant acceleration
        $g\sin\alpha$ along the chord)}: show the chord takes
        $\sqrt{\pi^2 + 4}\,\sqrt{R/g} \approx 3.72\sqrt{R/g}$.
        The curve beats the line --- it is, in fact, the fastest
        of all curves, a result of the calculus of variations.
  \item Show that on the arch ($0 < t < 2\pi$),
        \[
        \frac{\dd y}{\dd x} = \cot\frac t2,
        \qquad
        \frac{\dd^2 y}{\dd x^2} =
        -\frac{1}{4R\sin^4\frac t2} < 0 :
        \]
        the arch is concave, with vertical tangents exactly at
        the \omterm{ex:b1:curves:astroid}{cusps}.
  \item Recover the vertical \omterm{ex:b1:curves:astroid}{cusp} tangent of
        \cref{exo:b1:curves:2} geometrically: compute the limit
        direction of the chord $MT$ of question 3 as $t \to
        0^{+}$, with no expansions at all.
  \item Synthesis, in four sentences: which three named theorems
        this problem proved (with their numbers $8R$, $3\pi
        R^2$, $\pi\sqrt{R/g}$ and their authors); which single
        computational device ($\sin\frac t2$, $\cos\frac t2$
        factorizations) powered all of Parts I, II and IV; how
        the intrinsic relation $y = s^2/8R$ converted geometry
        into a \omterm{def:b1:linmaps:def}{linear} differential equation; and which chapter
        of this book each Part leaned on.
\end{enumerate}
\end{problem}
