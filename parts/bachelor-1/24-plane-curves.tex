\chapter{Plane Curves}\label{ch:b1:curves}

The calculus of \cref{ch:b1:derivative,ch:b1:taylor} was built for
functions $y = f(x)$; most curves of geometry and mechanics ---
trajectories, circles rolled on circles, orbits --- refuse that form
and come instead as \emph{parametrized} curves $t \mapsto (x(t),
y(t))$ or as \emph{polar} curves $r = r(\theta)$. This classical
chapter studies both, and closes with the conics.

\section{Parametrized curves}

\begin{definition}\label{def:b1:curves:def}
A \emph{parametrized curve}\index{parametrized curve} is a map $f
\colon I \to \R^2$, $t \mapsto M(t) = (x(t), y(t))$, with $x, y$ of
class $C^1$ (at least) on the interval $I$. The \emph{velocity
vector} is $f'(t) = (x'(t), y'(t))$; the point $M(t_0)$ is
\emph{regular} when $f'(t_0) \neq (0,0)$, and the \emph{tangent}
there is the line through $M(t_0)$ directed by $f'(t_0)$.
\end{definition}

\begin{proof}[Why $f'$ directs the tangent]
By Taylor--Young componentwise, $M(t) = M(t_0) + (t - t_0) f'(t_0) +
o(t - t_0)$: the chord direction $\frac{M(t) - M(t_0)}{t - t_0}$
tends to $f'(t_0)$.
\end{proof}

\begin{method}[Studying a parametrized curve]\label{met:b1:curves:study}
\begin{enumerate}
  \item \emph{Reduce the domain} using symmetries: relations between
        $M(-t)$, $M(t + T)$, \dots\ and $M(t)$ (reflections,
        translations, periodicity), then draw only the reduced part.
  \item \emph{Variations}: tabulate signs of $x'$ and $y'$ jointly;
        the curve moves right/left as $x'$, up/down as $y'$.
  \item \emph{Remarkable points}: horizontal tangent ($y' = 0 \neq
        x'$), vertical tangent ($x' = 0 \neq y'$); at a
        \emph{singular} point ($f' = 0$), examine higher derivatives
        for the tangent direction.
  \item \emph{Asymptotic behavior} at the ends of $I$, then sketch.
\end{enumerate}
\end{method}

\begin{example}[The astroid]\label{ex:b1:curves:astroid}
$x(t) = \cos^3 t$, $y(t) = \sin^3 t$.\index{astroid} Symmetries:
$M(t + 2\pi) = M(t)$ (study a period); $M(-t)$ is the reflection of
$M(t)$ in the $x$-axis; $M(\pi - t)$ in the $y$-axis; $M(\frac\pi2 -
t)$ in the diagonal $y = x$: it suffices to study $t \in
\intcc{0}{\frac\pi4}$ and unfold.

Velocity: $f'(t) = 3\sin t\cos t\,(-\cos t, \sin t)$. On
$\intoo{0}{\frac\pi2}$ all points are regular with tangent directed
by $(-\cos t, \sin t)$; at $t = 0$ (the point $(1,0)$) the velocity
vanishes: a \emph{cusp}\index{cusp}, where the curve reverses along
the tangent direction $(-1, 0)\cdot(\pm)$ --- by symmetry the four
cusps sit at $(\pm1, 0), (0, \pm1)$.
\end{example}

\section{Polar curves}

\begin{definition}\label{def:b1:curves:polar}
A \emph{polar curve}\index{polar curve} is given by $r = r(\theta)$:
the point of parameter $\theta$ is
\[
M(\theta) = r(\theta)\,\vec u(\theta),
\qquad \vec u(\theta) = (\cos\theta, \sin\theta).
\]
With $\vec v(\theta) = (-\sin\theta, \cos\theta) = \vec
u\,'(\theta)$, the velocity is
\[
M'(\theta) = r'(\theta)\, \vec u(\theta) + r(\theta)\, \vec
v(\theta) .
\]
Consequences: where $r \neq 0$, the point is regular, and the
tangent makes with the ray the angle $V$ given by $\tan V =
\frac{r}{r'}$ (angle between $M'$ and $\vec u$); where $r(\theta_0)
= 0$, the curve passes through the origin \emph{with tangent the ray
$\theta = \theta_0$} (direction $\vec u(\theta_0)$, read off
$M'(\theta_0) = r'(\theta_0)\vec u(\theta_0)$, or from the limit
chord).
\end{definition}

\begin{example}[The cardioid]\label{ex:b1:curves:cardioid}
$r(\theta) = 1 + \cos\theta$.\index{cardioid} Symmetry: $r(-\theta)
= r(\theta)$: reflection in the $x$-axis; study $\theta \in
\intcc{0}{\pi}$. $r$ decreases from $2$ to $0$; at $\theta = \pi$,
$r = 0$: the curve reaches the origin tangentially to the ray
$\theta = \pi$ (the $x$-axis), forming a cusp there --- the heart's
point. Tangent at $\theta = 0$: $r' = 0$, so $\tan V = \infty$:
perpendicular to the axis.
\end{example}

\section{Conics}

\begin{definition}[Focus--directrix definition]\label{def:b1:curves:conics}
Fix a point $F$ (\emph{focus}), a line $D$ not through $F$
(\emph{directrix}) and $e > 0$ (\emph{eccentricity}). The
\emph{conic}\index{conic} of these data is
\[
\mathcal{C} = \{M : d(M, F) = e\; d(M, D)\}:
\]
an \emph{ellipse} for $e < 1$, a \emph{parabola} for $e = 1$, a
\emph{hyperbola} for $e > 1$. (The circle appears as a degenerate
limit $e \to 0$.)
\end{definition}

\begin{theorem}[Reduced equations]\label{thm:b1:curves:reduced}
In a well-chosen orthonormal frame:
\[
\text{ellipse: } \frac{x^2}{a^2} + \frac{y^2}{b^2} = 1
\quad (a \geq b > 0),
\qquad
\text{hyperbola: } \frac{x^2}{a^2} - \frac{y^2}{b^2} = 1,
\qquad
\text{parabola: } y^2 = 2px,
\]
with, for the ellipse: foci at $(\pm c, 0)$, $c = \sqrt{a^2 - b^2}$,
$e = \frac ca$, and the bifocal characterization $MF + MF' = 2a$;
for the hyperbola: $c = \sqrt{a^2 + b^2}$, $e = \frac ca$,
$\abs{MF - MF'} = 2a$, asymptotes $y = \pm\frac ba x$.
\end{theorem}

\begin{proof}
Take the focus at the origin and the directrix vertical, $x = -h$
($h > 0$): the condition $MF^2 = e^2 d(M, D)^2$ reads $x^2 + y^2 =
e^2 (x + h)^2$. For $e = 1$: $y^2 = 2hx + h^2$, a parabola after the
shift $x \mapsto x - \frac h2$ (so $p = h$). For $e \neq 1$:
completing the square in $x$,
\[
(1 - e^2)\Bigl(x - \frac{e^2 h}{1 - e^2}\Bigr)^{\!2} + y^2
= e^2h^2 + \frac{e^4 h^2}{1 - e^2}
= \frac{e^2 h^2}{1 - e^2} .
\]
Set $X = x - \frac{e^2h}{1-e^2}$ (a shift of frame). When $e < 1$,
divide by the positive right-hand side:
$\frac{X^2}{a^2} + \frac{y^2}{b^2} = 1$ with $a = \frac{eh}{1 -
e^2}$, $b = \frac{eh}{\sqrt{1-e^2}}$; when $e > 1$, both sides of
the display flip sign appropriately and the same division gives
$\frac{X^2}{a^2} - \frac{y^2}{b^2} = 1$ with $a = \frac{eh}{e^2 -
1}$, $b = \frac{eh}{\sqrt{e^2 - 1}}$. The stated values of $c$
follow ($c^2 = a^2 - b^2$ or $a^2 + b^2$ gives $c = ea$ in both
cases, placing the focus correctly), and the bifocal properties are
direct verifications on the reduced equations: computing $MF^2$ for
$M$ on the ellipse, one finds $MF = \abs{a + eX}$ and $MF' = \abs{a
- eX}$, whence the sum $2a$ (difference $2a$ for the hyperbola).
\end{proof}

\begin{example}\label{ex:b1:curves:conicexample}
$\frac{x^2}{25} + \frac{y^2}{9} = 1$: ellipse, $a = 5$, $b = 3$, $c
= 4$: foci $(\pm4, 0)$, eccentricity $\frac45$. Its parametrization:
$(5\cos t, 3\sin t)$ --- a circle stretched anisotropically; the
sum of distances to the foci of any of its points is $10$.
\end{example}

\section{Exercises}

\begin{exercise}[$\star$]\label{exo:b1:curves:1}
Study and sketch the curve $x(t) = t^2$, $y(t) = t^3$ (symmetries,
variations, behavior at the singular point $t = 0$).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:curves:2}
For the cycloid $x(t) = t - \sin t$, $y(t) = 1 - \cos t$ (the
trajectory of a point of a rolling wheel of radius $1$): identify
the period-translation symmetry, the singular points, and the
tangent direction at $t = 0$ \emph{(expand $x$ and $y$ to the first
nonzero orders)}.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:curves:3}
Give the tangent line to the curve $(\cos^3 t, \sin^3 t)$ at $t =
\frac\pi4$, and check that the segment of this tangent cut by the
axes has length $1$ --- a famous property of the astroid (true at
every regular point).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:curves:4}
Sketch the polar curves $r = \cos 2\theta$ (four-petaled rose) and
$r = \frac{1}{\cos\theta}$ on $\intoo{-\frac\pi2}{\frac\pi2}$
(recognize a line).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:curves:5}
Study the polar curve $r = 1 + 2\cos\theta$ (a limaĂ§on): domain
where $r \geq 0$ vs $r < 0$ (points plotted with negative radius
sit on the opposite ray), passage through the origin, inner loop,
sketch.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:curves:6}
Identify the conic $9x^2 + 25y^2 - 36x - 50y - 164 = 0$: reduce by
completing squares, give center, semi-axes, foci, eccentricity.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:curves:7}
Prove that the polar equation of a conic with focus at the origin
is
\[
r = \frac{p}{1 + e\cos\theta}
\]
(directrix vertical at distance $\frac pe$ right of the focus).
Which values of $\theta$ are allowed when $e > 1$?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:curves:8}
From the bifocal property $MF + MF' = 2a$, deduce the ``gardener's
construction'' of the ellipse, and prove that the tangent at $M$
makes equal angles with $MF$ and $MF'$ \emph{(reflection property;
differentiate $\norm{M(t) - F} + \norm{M(t) - F'} = 2a$ and
interpret the vanishing sum of unit-vector products)}.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:curves:9}
The \emph{lemniscate of Bernoulli} is the polar curve $r^2 = \cos
2\theta$ (take $r = \sqrt{\cos2\theta}$ where defined).
\begin{enumerate}
  \item Give its domain, symmetries, tangents at the origin, and
        sketch it.
  \item Prove that it is the locus of points $M$ with $MF \cdot MF'
        = \frac12$ where $F, F' = \bigl(\pm\frac{1}{\sqrt2},
        0\bigr)$. \emph{(Compute $MF^2\,MF'^2$ in polar
        coordinates.)}
\end{enumerate}
\end{exercise}
