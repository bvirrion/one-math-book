\chapter{Lineaire afbeeldingen}\label{ch:b1:linmaps}

De \omterm{def:b1:logic:map}{afbeeldingen} tussen \omterm{def:b1:vspaces:def}{vectorruimten} die de moeite waard zijn om te
bestuderen, zijn die welke verenigbaar zijn met de structuur:
\emph{\omterm{def:b1:linmaps:def}{lineaire afbeeldingen}}. Hun twee fundamentele \omterm{def:b1:vspaces:subspace}{deelruimten} ---
\omterm{def:b1:linmaps:kerim}{kern} en \omterm{def:b1:linmaps:kerim}{beeld} --- meten injectiviteit en surjectiviteit, en in \omterm{def:b1:findim:def}{eindige
dimensie} bindt de dimensiestelling hun groottes samen tot één
behoudswet. \omterm{def:b1:linmaps:projection}{Projecties} en symmetrieën, en vervolgens \omterm{def:b1:linmaps:forms}{lineaire vormen} en
hypervlakken, sluiten het hoofdstuk af.

\section{Definities en eerste eigenschappen}

\begin{definition}\label{def:b1:linmaps:def}
Zij $E, F$ $K$-vectorruimten. Een \omterm{def:b1:logic:map}{afbeelding} $u \colon E \to F$ is
\emph{lineair}\index{lineaire afbeelding} wanneer
\[
\forall x, y \in E,\ \forall \lambda \in K, \qquad
u(x + \lambda y) = u(x) + \lambda u(y).
\]
Dan geldt $u(0) = 0$ en $u(\sum \lambda_i x_i) = \sum \lambda_i u(x_i)$.
De \omterm{def:b1:logic:sets}{verzameling} $\mathcal{L}(E, F)$ van lineaire \omterm{def:b1:logic:map}{afbeeldingen} is zelf een
\omterm{def:b1:vspaces:def}{vectorruimte}; de samenstelling van lineaire \omterm{def:b1:logic:map}{afbeeldingen} is lineair, en
bilineair in het paar. Een \emph{endomorfisme} is een lineaire $E \to
E$; een \emph{isomorfisme} is een bijectieve lineaire \omterm{def:b1:logic:map}{afbeelding} (haar
inverse is dan automatisch lineair); $u^{-1}$ van een isomorfisme, en
samenstellingen van isomorfismen, zijn isomorfismen.
\end{definition}

\begin{proof}[Bewijs dat de inverse lineair is]
Zij $u$ \omterm{def:b1:linmaps:def}{lineair} en \omterm{def:b1:logic:inj}{bijectief}, $y, y' \in F$ en $\lambda \in K$.
Stel $x = u^{-1}(y)$ en $x' = u^{-1}(y')$. Dan geldt
\[
u\bigl(x + \lambda x'\bigr) = u(x) + \lambda u(x')
= y + \lambda y' ,
\]
en door $u^{-1}$ op beide uiteinden toe te passen: $u^{-1}(y + \lambda
y') = x + \lambda x' = u^{-1}(y) + \lambda\,u^{-1}(y')$. Er werd niets
over $u^{-1}$ berekend: lineariteit gaat over via de definiërende
eigenschap van $u$ alleen --- een patroon dat de moeite waard is om te
onthouden, aangezien structuur vaak gratis meelift over bijecties.
\end{proof}

\begin{proposition}[Een lineaire afbeelding is bekend op een basis]\label{prop:b1:linmaps:basis}
Zij $(e_1, \dots, e_n)$ een \omterm{def:b1:vspaces:free}{basis} van $E$ en $(v_1, \dots, v_n)$
willekeurige vectoren van $F$. Er is precies één \omterm{def:b1:linmaps:def}{lineaire afbeelding} $u
\colon E \to F$ met $u(e_i) = v_i$ voor alle $i$. Bovendien:
\[
u \text{ injectief} \iff (v_i) \text{ vrij};
\qquad
u \text{ surjectief} \iff (v_i) \text{ brengt } F \text{ voort} .
\]
\end{proposition}

\begin{proof}
Bestaan/uniciteit: elke $x$ heeft unieke \omterm{prop:b1:vspaces:coordinates}{coördinaten} $x = \sum
\lambda_i e_i$ (\cref{prop:b1:vspaces:coordinates}); lineariteit
dwingt $u(x) = \sum \lambda_i v_i$ af, en deze formule definieert
inderdaad een \omterm{def:b1:linmaps:def}{lineaire afbeelding}.

Injectiviteit: volgens \cref{prop:b1:linmaps:kernel} hieronder is $u$
\omterm{def:b1:logic:inj}{injectief} dan en slechts dan als haar \omterm{def:b1:linmaps:kerim}{kern} triviaal is. Nu betekent
$u\bigl(\sum\lambda_i e_i\bigr) = 0$ precies $\sum\lambda_i v_i = 0$.
Als $(v_i)$ \omterm{def:b1:vspaces:free}{vrij} is, dwingt dit elke $\lambda_i = 0$ af, d.w.z.\ de \omterm{def:b1:linmaps:kerim}{kern}
reduceert tot $0$: \omterm{def:b1:logic:inj}{injectief}. Als $(v_i)$ gebonden is, produceert een
niet-triviale relatie $\sum\lambda_i v_i = 0$ de niet-nulvector
$\sum\lambda_i e_i$ in de \omterm{def:b1:linmaps:kerim}{kern} ($(e_i)$ is \omterm{def:b1:vspaces:free}{vrij}): niet \omterm{def:b1:logic:inj}{injectief}. De
twee voorwaarden komen term voor term overeen.

Surjectiviteit: het \omterm{def:b1:linmaps:kerim}{beeld} van $u$ is de \omterm{def:b1:logic:sets}{verzameling} van alle
$\sum\lambda_i v_i$, d.w.z.\ precies $\operatorname{Vect}(v_1, \dots,
v_n)$, die gelijk is aan $F$ dan en slechts dan als de familie
voortbrengt.
\end{proof}

\begin{definition}[Kern en beeld]\label{def:b1:linmaps:kerim}
Voor $u \in \mathcal{L}(E, F)$:
\[
\ker u = \{x \in E : u(x) = 0\} \subseteq E,
\qquad
\operatorname{im} u = u(E) \subseteq F ,
\]
beide \omterm{def:b1:vspaces:subspace}{deelruimten} (rechtstreekse verificatie met het
criterium).\index{kern}\index{beeld}
\end{definition}

\begin{method}[Kern en beeld, in de praktijk]
\label{met:b1:linmaps:kerim}
\emph{\omterm{def:b1:linmaps:kerim}{Kern}}: schrijf $u(x) = 0$ als een stelsel in de \omterm{prop:b1:vspaces:coordinates}{coördinaten}
(of coëfficiënten) van $x$, los op, en parametriseer --- de \omterm{def:b1:linmaps:kerim}{kern}
komt tevoorschijn met een \omterm{def:b1:vspaces:free}{basis} erbij
(\cref{met:b1:findim:computedim}). \emph{\omterm{def:b1:linmaps:kerim}{Beeld}}: het is het \omterm{def:b1:vspaces:span}{opspansel}
van de \omterm{def:b1:linmaps:kerim}{beelden} van \emph{om het even welke} voortbrengende familie van
$E$ --- gewoonlijk een \omterm{def:b1:vspaces:free}{basis}, dus $\operatorname{im} u =
\operatorname{Vect}\bigl(u(e_1), \dots, u(e_n)\bigr)$; elimineer
vervolgens overbodige \omterm{def:b1:linmaps:kerim}{beelden} om een \omterm{def:b1:vspaces:free}{basis} te verkrijgen.
\emph{Kortere weg}: bereken degene van de twee die het gemakkelijkst is
en verkrijg de dimensie van de andere gratis via de dimensiestelling
(\cref{thm:b1:linmaps:ranknullity}); wanneer een aannemelijke
\omterm{def:b1:vspaces:subspace}{kandidaat-deelruimte} voor het \omterm{def:b1:linmaps:kerim}{beeld} bekend is, verheft het vergelijken
van dimensies de gemakkelijke inclusie tot een gelijkheid
(\cref{thm:b1:findim:subspaces}). Beide kortere wegen worden gebruikt in
\cref{ex:b1:linmaps:delta} hieronder.
\end{method}

\begin{proposition}\label{prop:b1:linmaps:kernel}
$u$ is \omterm{def:b1:logic:inj}{injectief} $\iff$ $\ker u = \{0\}$; $u$ is \omterm{def:b1:logic:inj}{surjectief} $\iff$
$\operatorname{im} u = F$.
\end{proposition}

\begin{proof}
Zoals voor \omterm{def:b1:structures:group}{groepen} (\cref{prop:b1:structures:kernel}): $u(x) = u(y) \iff
u(x - y) = 0 \iff x - y \in \ker u$. Het tweede punt is de
definitie.
\end{proof}

\section{De dimensiestelling}

\begin{definition}\label{def:b1:linmaps:rank}
De \emph{rang}\index{rang!van een lineaire afbeelding} van $u \in
\mathcal{L}(E, F)$ (met $E$ \omterm{def:b1:findim:def}{eindigdimensionaal}) is $\operatorname{rk} u
= \dim \operatorname{im} u$ --- tevens de rang van de familie
$\bigl(u(e_1), \dots, u(e_n)\bigr)$ voor om het even welke \omterm{def:b1:vspaces:free}{basis} $(e_i)$
van $E$.
\end{definition}

\begin{theorem}[Dimensiestelling]\label{thm:b1:linmaps:ranknullity}
Zij $E$ \omterm{def:b1:findim:def}{eindigdimensionaal} en $u \in \mathcal{L}(E, F)$.
Dan\index{dimensiestelling}
\[
\dim E = \dim \ker u + \operatorname{rk} u .
\]
Preciezer: als $S$ een willekeurige complementaire \omterm{def:b1:vspaces:subspace}{deelruimte} van $\ker
u$ in $E$ is, dan beperkt $u$ zich tot een \emph{isomorfisme} van $S$
op $\operatorname{im} u$.
\end{theorem}

\begin{proof}
Zij $S$ zodanig dat $E = \ker u \oplus S$
(\cref{thm:b1:findim:subspaces}), en zij $v \colon S \to
\operatorname{im} u$ de beperking van $u$.

$v$ is \omterm{def:b1:logic:inj}{injectief}: $\ker v = S \cap \ker u = \{0\}$.

$v$ is \omterm{def:b1:logic:inj}{surjectief}: elke $u(x)$ met $x = k + s$ ($k \in \ker u$, $s
\in S$) is gelijk aan $u(s) = v(s)$.

Dus $v$ is een isomorfisme; een isomorfisme stuurt een \omterm{def:b1:vspaces:free}{basis} naar een
\omterm{def:b1:vspaces:free}{basis} (\cref{prop:b1:linmaps:basis}), dus $\dim S =
\dim\operatorname{im} u$, en $\dim E = \dim\ker u + \dim S$ besluit.
\end{proof}

\begin{example}[Een afbeelding op specificatie bouwen]
\label{ex:b1:linmaps:tospec}
Construeer $u \in \mathcal{L}(\R^3)$ met $\ker u =
\operatorname{Vect}(1,1,1)$ en $\operatorname{im} u = \{z =
0\}$. Eerst een controle: de dimensiestelling eist $1 + 2 = 3$ ---
consistent, dus een oplossing kan bestaan. Kies een \omterm{def:b1:vspaces:free}{basis} aangepast aan
de \omterm{def:b1:linmaps:kerim}{kern}, zeg $\bigl((1,1,1),\ e_1,\ e_2\bigr)$
(\cref{ex:b1:findim:completion}), en schrijf de \omterm{def:b1:linmaps:kerim}{beelden} voor
(\cref{prop:b1:linmaps:basis}):
\[
u(1,1,1) = 0, \qquad u(e_1) = e_1, \qquad u(e_2) = e_2 .
\]
Dan geldt $\ker u \supseteq \operatorname{Vect}(1,1,1)$ en
$\operatorname{im} u = \operatorname{Vect}(e_1, e_2) = \{z =
0\}$; de dimensiestelling dwingt $\dim\ker u = 1$ af, dus de \omterm{def:b1:linmaps:kerim}{kern} is
precies de voorgeschreven rechte. Expliciet, door $(x, y, z)
= z(1,1,1) + (x - z)e_1 + (y - z)e_2$ te ontbinden:
\[
u(x, y, z) = (x - z,\ y - z,\ 0).
\]
Het recept veralgemeent: een \omterm{def:b1:linmaps:def}{lineaire afbeelding} met voorgeschreven
\omterm{def:b1:linmaps:kerim}{kern} $N$ en \omterm{def:b1:linmaps:kerim}{beeld} $I$ bestaat precies wanneer $\dim N + \dim I = \dim E$
--- de noodzaak is de dimensiestelling, de voldoendheid is deze
constructie.
\end{example}

\begin{corollary}\label{cor:b1:linmaps:samedim}
Als $\dim E = \dim F$ (eindig), dan geldt voor $u \in \mathcal{L}(E, F)$:
\[
u \text{ injectief} \iff u \text{ surjectief} \iff u \text{
bijectief}.
\]
In het bijzonder geldt dit voor endomorfismen in \omterm{def:b1:findim:def}{eindige dimensie}. (Het
faalt in oneindige dimensie: op $K[X]$ is de \omterm{def:b1:derivative:def}{afgeleide} \omterm{def:b1:logic:inj}{surjectief} maar
niet \omterm{def:b1:logic:inj}{injectief}, en $P \mapsto XP$ is \omterm{def:b1:logic:inj}{injectief} maar niet
\omterm{def:b1:logic:inj}{surjectief}.)
\end{corollary}

\begin{proof}
\omterm{def:b1:logic:inj}{Injectief} $\iff \dim\ker u = 0 \iff \operatorname{rk} u = \dim E =
\dim F \iff \operatorname{im} u = F$ (een \omterm{def:b1:vspaces:subspace}{deelruimte} van volle dimensie
is alles, \cref{thm:b1:findim:subspaces}) $\iff$ \omterm{def:b1:logic:inj}{surjectief}.
\end{proof}

\begin{example}[Interpolatie, structureel]\label{ex:b1:linmaps:interpolation}
Fixeer verschillende $x_0, \dots, x_n$ en zij $u \colon \R_n[X] \to
\R^{n+1}$, $P \mapsto (P(x_0), \dots, P(x_n))$: \omterm{def:b1:linmaps:def}{lineair}. Haar \omterm{def:b1:linmaps:kerim}{kern}
is $\{P : \deg P \leq n,\ n+1 \text{ wortels}\} = \{0\}$
(\cref{cor:b1:poly:nroots}). Gelijke dimensies $n + 1$: $u$ is een
isomorfisme --- bestaan \emph{en} uniciteit van de
Lagrange-interpolant (\cref{thm:b1:poly:lagrange}) in één regel.

Hetzelfde éénregelige patroon behandelt data die waarden en \omterm{def:b1:derivative:def}{afgeleiden}
vermengen: $v \colon \R_3[X] \to \R^4$, $P \mapsto \bigl(P(0),
P'(0), P(1), P'(1)\bigr)$ is \omterm{def:b1:linmaps:def}{lineair}, en haar \omterm{def:b1:linmaps:kerim}{kern} bestaat uit
\omterm{def:b1:poly:def}{veeltermen} van graad $\leq 3$ met dubbele wortels in $0$ \emph{en}
$1$, d.w.z.\ \omterm{def:b1:arith:divides}{deelbaar} door $X^2(X-1)^2$ van graad $4$: enkel $P =
0$. Opnieuw gelijke dimensies: elk viertal data $(P(0),
P'(0), P(1), P'(1))$ wordt gerealiseerd door precies één derdegraads
\omterm{def:b1:poly:def}{veelterm} --- Hermite-interpolatie, toegekend door een kernberekening
voordat enige formule wordt opgeschreven (het weekendprobleem van
\cref{ch:b1:det} ontmoet zijn determinant).
\end{example}

\begin{example}[De dimensiestelling aan het werk: de verschiloperator]
\label{ex:b1:linmaps:delta}
Zij $\Delta \colon \R_n[X] \to \R_n[X]$, $P \mapsto P(X+1) -
P(X)$: \omterm{def:b1:linmaps:def}{lineair}. \omterm{def:b1:linmaps:kerim}{Kern}: als $\Delta P = 0$, dan $P(0) = P(1) =
P(2) = \dots$, dus $P - P(0)$ heeft oneindig veel wortels en
verdwijnt (\cref{cor:b1:poly:nroots}): $\ker\Delta$ is de rechte van
de constanten. Dimensiestelling: $\operatorname{rk}\Delta = (n + 1) - 1
= n$. Aangezien $\deg \Delta P < \deg P$ voor niet-constante $P$ (de
hoogste termen heffen elkaar op), geldt $\operatorname{im}\Delta
\subseteq \R_{n-1}[X]$, die precies dimensie $n$ heeft: de inclusie is
een gelijkheid. Conclusie, zonder enige berekening van originelen:
\emph{elke} \omterm{def:b1:poly:def}{veelterm} $Q$ van graad $\leq n - 1$ is een verschil $Q =
P(X+1) - P(X)$ --- de discrete primitieve bestaat. (Vergelijk het
weekendprobleem van \cref{ch:b1:vspaces}, waar $\Delta$ expliciet werd
geïnverteerd in de binomiaalbasis.)
\end{example}

\begin{example}[Rangboekhouding langs een samenstelling]
\label{ex:b1:linmaps:rankbookkeeping}
Stel op $\R_2[X]$ de \omterm{def:b1:derivative:def}{afgeleide} $D(P) = P'$ (\omterm{def:b1:linmaps:rank}{rang} $2$: \omterm{def:b1:linmaps:kerim}{beeld}
$\R_1[X]$, \omterm{def:b1:linmaps:kerim}{kern} de constanten) samen met zichzelf. Dan beeldt $D
\circ D = D^2$ af $P \mapsto P''$, met \omterm{def:b1:linmaps:kerim}{beeld} $\R_0[X]$: \omterm{def:b1:linmaps:rank}{rang}
$1$. Vergelijk met de algemene grenzen: de grove geeft
$\operatorname{rk} D^2 \leq \min(2, 2) = 2$; de exacte formule
van \cref{exo:b1:linmaps:12} verantwoordt het verlies nauwkeurig,
\[
\operatorname{rk} D^2 = \operatorname{rk} D -
\dim\bigl(\ker D \cap \operatorname{im} D\bigr) = 2 - 1 = 1 ,
\]
aangezien de constanten (\omterm{def:b1:linmaps:kerim}{kern} van de buitenste $D$) binnen
$\R_1[X]$ (\omterm{def:b1:linmaps:kerim}{beeld} van de binnenste $D$) liggen met dimensie $1$. \omterm{def:b1:linmaps:rank}{Rang}
gaat precies daar verloren waar de buitenste \omterm{def:b1:linmaps:kerim}{kern} het binnenste \omterm{def:b1:linmaps:kerim}{beeld}
overvalt --- de zin om te onthouden wanneer samenstellingsrangen zich
misdragen.
\end{example}

\begin{example}[Kern en beeld van de Euler-operator]
\label{ex:b1:linmaps:euler}
Zij op $\R_n[X]$ $u(P) = X\,P'$ (\omterm{def:b1:linmaps:def}{lineair}: differentiëren en
vermenigvuldigen met $X$ zijn dat). \emph{\omterm{def:b1:linmaps:kerim}{Kern}}: $XP' = 0$ dwingt $P' =
0$ af (een product van \omterm{def:b1:poly:def}{veeltermen} verdwijnt enkel als een van de
factoren dat doet), dus $\ker u$ is de rechte van de constanten.
\emph{\omterm{def:b1:linmaps:kerim}{Beeld}}: op de monoombasis,
\[
u(X^k) = k\,X^{k} \qquad (k = 0, 1, \dots, n),
\]
dus $\operatorname{im} u = \operatorname{Vect}(X, 2X^2, \dots,
nX^n) = \operatorname{Vect}(X, X^2, \dots, X^n)$: de \omterm{def:b1:poly:def}{veeltermen} met
nulconstante term. Controleer aan de hand van de dimensiestelling:
$\operatorname{rk} u = (n + 1) - 1 = n$, wat inderdaad de gevonden
dimensie is. Twee opmerkingen die de moeite waard zijn om te bewaren.
Ten eerste, hier geldt $\operatorname{im} u \oplus \ker u = \R_n[X]$ ---
maar dat is een \emph{gelukkig toeval} van deze operator, geen stelling:
voor de shift-achtige $v(P) = P'$ op $\R_1[X]$ geldt $\ker v =
\operatorname{im} v = \R_0[X]$ en is de som niet direct. Ten tweede zegt
de relatie $u(X^k) = kX^k$ dat elk monoom door $u$ enkel herschaald
wordt --- een \omterm{def:b1:vspaces:free}{basis} aangepast aan de \omterm{def:b1:logic:map}{afbeelding}, de kiem van het
eigenwaarde-idee dat ontwikkeld wordt in het volume van Jaar~2.
\end{example}

\section{Projecties en symmetrieën}

\begin{definition}\label{def:b1:linmaps:projection}
Zij $E = F \oplus G$. De \emph{projectie op $F$ langs
$G$}\index{projectie} beeldt $x = f + g$ (unieke ontbinding) af op
$p(x) = f$; de bijbehorende \emph{symmetrie}\index{symmetrie} is $s(x)
= f - g$. Beide zijn \omterm{def:b1:linmaps:def}{lineair}, en $s = 2p - \mathrm{id}$.
\end{definition}

\begin{theorem}[Algebraïsche karakterisering]\label{thm:b1:linmaps:projchar}
\begin{enumerate}
  \item Een endomorfisme $p$ is een \omterm{def:b1:linmaps:projection}{projectie} (op een zekere $F$ langs
        een zekere $G$) dan en slechts dan als $p \circ p = p$; dan geldt
        $F = \operatorname{im} p = \ker(p - \mathrm{id})$ en $G = \ker
        p$.
  \item Een endomorfisme $s$ is een symmetrie dan en slechts dan als $s
        \circ s = \mathrm{id}$; dan geldt $E = \ker(s - \mathrm{id})
        \oplus \ker(s + \mathrm{id})$.
\end{enumerate}
\end{theorem}

\begin{proof}
(1) Een \omterm{def:b1:linmaps:projection}{projectie} voldoet aan $p(f + g) = f$ en $p(f) = f$: $p^2 = p$.
Omgekeerd, zij $p^2 = p$; stel $F = \operatorname{im} p$, $G = \ker
p$. Elke $x$ schrijft $x = p(x) + (x - p(x))$ met $p(x) \in F$ en
$p\bigl(x - p(x)\bigr) = p(x) - p^2(x) = 0$: $E = F + G$. Als $y \in
F \cap G$: $y = p(z)$ en $p(y) = 0$, dus $y = p(z) = p^2(z) = p(y) =
0$: \omterm{def:b1:vspaces:sum}{directe som}, en $p$ is de \omterm{def:b1:linmaps:projection}{projectie} op $F$ langs $G$.
Ten slotte op $F$: $y = p(z)$ geeft $p(y) = y$, dus $F \subseteq
\ker(p - \mathrm{id})$, en omgekeerd plaatst $p(y) = y$ $y$ in het
\omterm{def:b1:linmaps:kerim}{beeld}.

(2) De correspondentie $s = 2p - \mathrm{id}$, $p = \frac{s +
\mathrm{id}}2$ is een bijectie tussen endomorfismen, en daaronder
\[
s^2 = 4p^2 - 4p + \mathrm{id} = \mathrm{id}
\iff 4p^2 = 4p \iff p^2 = p :
\]
symmetrieën corresponderen precies met \omterm{def:b1:linmaps:projection}{projecties}. Vertaling van de
\omterm{def:b1:vspaces:subspace}{deelruimten}: $s(x) = x \iff p(x) = x$, dus $\ker(s - \mathrm{id}) =
\operatorname{im} p = F$; en $s(x) = -x \iff 2p(x) = 0 \iff x
\in \ker p = G$, dus $\ker(s + \mathrm{id}) = G$. De \omterm{def:b1:vspaces:sum}{directe som}
$E = F \oplus G$ van punt (1) wordt de aangekondigde ontbinding
in de vaste vectoren en de omgekeerde vectoren van $s$.
\end{proof}

\begin{example}[Een projectie en haar symmetrie, expliciet]
\label{ex:b1:linmaps:projexplicit}
Projecteer in $\R^2$ op $F = \operatorname{Vect}(1,1)$ langs $G =
\operatorname{Vect}(0,1)$. Ontbind $(x, y) = a(1,1) + b(0,1)$:
de eerste coördinaat geeft $a = x$, de tweede $b = y - x$.
Bijgevolg
\[
p(x, y) = (x, x),
\qquad
s(x, y) = 2p(x,y) - (x,y) = (x,\ 2x - y).
\]
Controleer de algebra: $p(p(x,y)) = p(x,x) = (x,x)$, en $s(s(x,y)) =
s(x, 2x - y) = (x, 2x - (2x - y)) = (x, y)$. Meetkundig is $s$
de ``scheve spiegeling'' over de rechte $y = x$ in de
verticale richting: ze houdt $F$ puntsgewijs vast en keert $G$ om. Hadden
we op dezelfde $F$ langs $G' =
\operatorname{Vect}(1,-1)$ geprojecteerd, dan zou de formule veranderen
in $p'(x,y) = \bigl(\frac{x+y}2, \frac{x+y}2\bigr)$: een \omterm{def:b1:linmaps:projection}{projectie} wordt
bepaald door haar \omterm{def:b1:linmaps:kerim}{beeld} \emph{en} haar \omterm{def:b1:linmaps:kerim}{kern}, nooit door het \omterm{def:b1:linmaps:kerim}{beeld}
alleen.
\end{example}

\section{Lineaire vormen en hypervlakken}

\begin{definition}\label{def:b1:linmaps:forms}
Een \emph{lineaire vorm}\index{lineaire vorm} op $E$ is een \omterm{def:b1:linmaps:def}{lineaire
afbeelding} $\varphi \colon E \to K$. Een \emph{hypervlak}\index{hypervlak}
van $E$ ($\dim E = n$) is een \omterm{def:b1:vspaces:subspace}{deelruimte} van dimensie $n - 1$.
\end{definition}

\begin{example}[Een evaluatievorm en zijn hypervlak]
\label{ex:b1:linmaps:evaluationform}
Op $\R_2[X]$ is de evaluatie $\varphi(P) = P(2)$ een \omterm{def:b1:linmaps:forms}{lineaire
vorm}, niet-nul ($\varphi(1) = 1$). Zijn \omterm{def:b1:linmaps:kerim}{kern} is het \omterm{def:b1:linmaps:forms}{hypervlak}
van \omterm{def:b1:poly:def}{veeltermen} die in $2$ verdwijnen, d.w.z.\ (factorstelling,
\cref{thm:b1:poly:factor}) de veelvouden van $X - 2$ binnen
$\R_2[X]$:
\[
\ker\varphi = \operatorname{Vect}\bigl(X - 2,\ X(X - 2)\bigr),
\qquad \dim = 2 .
\]
In \omterm{prop:b1:vspaces:coordinates}{coördinaten} op $(1, X, X^2)$, $\varphi(a + bX + cX^2) = a +
2b + 4c$: elke \omterm{def:b1:linmaps:forms}{lineaire vorm} op een eindigdimensionale ruimte is,
zodra een \omterm{def:b1:vspaces:free}{basis} is vastgelegd, een vaste lineaire uitdrukking in de
\omterm{prop:b1:vspaces:coordinates}{coördinaten} --- vormen zijn ``rijvectoren'', zoals
\cref{ch:b1:matrices} letterlijk zal maken, en de coëfficiëntenrij
hier, $(1, 2, 4)$, is een Vandermonde-rij: evaluatievormen
zijn de manier waarop de interpolatietheorie van het weekendprobleem
van \cref{ch:b1:det} de lineaire algebra binnenkomt.
\end{example}

\begin{theorem}\label{thm:b1:linmaps:hyperplanes}
De hypervlakken van $E$ zijn precies de \omterm{def:b1:linmaps:kerim}{kernen} van niet-nul \omterm{def:b1:linmaps:forms}{lineaire
vormen}. Twee niet-nul vormen hebben dezelfde \omterm{def:b1:linmaps:kerim}{kern} dan en slechts dan als
ze evenredig zijn.
\end{theorem}

\begin{proof}
Als $\varphi \neq 0$: $\operatorname{rk}\varphi = 1$ (\omterm{def:b1:linmaps:kerim}{beeld} een niet-nul
\omterm{def:b1:vspaces:subspace}{deelruimte} van $K$), dus $\dim\ker\varphi = n - 1$: een \omterm{def:b1:linmaps:forms}{hypervlak}.
Omgekeerd, zij $H$ een \omterm{def:b1:linmaps:forms}{hypervlak}, $(e_1, \dots, e_{n-1})$ een \omterm{def:b1:vspaces:free}{basis}
van $H$ aangevuld met $e_n$: de vorm ``laatste coördinaat'' heeft \omterm{def:b1:linmaps:kerim}{kern}
$H$.

Evenredige vormen delen hun \omterm{def:b1:linmaps:kerim}{kern}. Omgekeerd, veronderstel
$\ker\varphi = \ker\psi = H$ en kies $a \notin H$: elke $x$ schrijft
$x = h + \lambda a$ (aangezien $E = H \oplus Ka$), en
\[
\varphi(x) = \lambda \varphi(a), \qquad \psi(x) = \lambda\psi(a):
\]
dus $\varphi = \frac{\varphi(a)}{\psi(a)}\,\psi$.
\end{proof}

\begin{example}\label{ex:b1:linmaps:hyperplane}
In $K^n$ is een \omterm{def:b1:linmaps:forms}{hypervlak} een oplossingsverzameling $\{a_1 x_1 + \dots +
a_n x_n = 0\}$ met niet alle $a_i$ nul --- de vertrouwde vergelijking van
een vlak door de oorsprong in $\R^3$. In functieruimten definiëren
evaluatievormen $P \mapsto P(1)$ of $f \mapsto \int_0^1 f$ hypervlakken
van $\R_n[X]$, van $C(\intcc{0}{1})$ (vgl.\ \cref{exo:b1:findim:6}).
\end{example}

\begin{example}[Eén hypervlak, op drie manieren uitgewerkt]
\label{ex:b1:linmaps:hyperplanerun}
Neem $\varphi(x, y, z) = x - 2y + 3z$ op $\R^3$ en $H =
\ker\varphi$. \emph{\omterm{def:b1:vspaces:free}{Basis}}: los $x = 2y - 3z$ op:
\[
(2y - 3z,\ y,\ z) = y\,(2, 1, 0) + z\,(-3, 0, 1),
\]
twee vrije vectoren: $\dim H = 2$, een \omterm{def:b1:linmaps:forms}{hypervlak}, zoals
\cref{thm:b1:linmaps:hyperplanes} voorspelt uit $\varphi \neq
0$. \emph{Complementaire rechte}: elke vector buiten $H$ brengt er een
voort, bijv.\ $a = (1, 0, 0)$ ($\varphi(a) = 1 \neq 0$); de
ontbinding van een willekeurige $v$ is expliciet:
\[
v = \underbrace{\bigl(v - \varphi(v)\,a\bigr)}_{\in\,H}
+ \underbrace{\varphi(v)\,a}_{\in\,\operatorname{Vect}(a)},
\]
aangezien $\varphi\bigl(v - \varphi(v)a\bigr) = \varphi(v) -
\varphi(v)\varphi(a) = 0$. \emph{Evenredigheid}: als $\psi(x,y,z)
= -2x + 4y - 6z$, dan geldt $\psi = -2\varphi$ en beide hebben \omterm{def:b1:linmaps:kerim}{kern}
$H$; omgekeerd is elke vorm die verdwijnt op $H$ een veelvoud van
$\varphi$ (\cref{exo:b1:linmaps:8}) --- de vergelijking van een
\omterm{def:b1:linmaps:forms}{hypervlak} is uniek op een schaalfactor na, een feit dat voortdurend wordt
gebruikt voor vlakken in de meetkunde.
\end{example}

\begin{remark}[Veelvoorkomende valkuilen]
\label{rem:b1:linmaps:pitfalls}
\emph{\omterm{def:b1:linmaps:kerim}{Kern} en \omterm{def:b1:linmaps:kerim}{beeld} leven in verschillende ruimten}: $\ker u
\subseteq E$, $\operatorname{im} u \subseteq F$; de som $\ker u
+ \operatorname{im} u$ heeft enkel zin voor endomorfismen, en
zelfs dan hoeft ze niet direct te zijn ($u(x, y) = (y, 0)$ heeft $\ker u
= \operatorname{im} u$; \cref{exo:b1:linmaps:7} karakteriseert
wanneer directheid geldt).
\emph{$u^2 = 0$ betekent niet $u = 0$}: dezelfde $u(x,y) =
(y, 0)$ kwadrateert tot nul zonder te verdwijnen --- wat $u^2 = 0$
werkelijk zegt is $\operatorname{im} u \subseteq \ker u$
(\cref{exo:b1:linmaps:5}).
\emph{\omterm{def:b1:logic:inj}{Injectief} $\iff$ \omterm{def:b1:logic:inj}{surjectief} vereist gelijke \omterm{def:b1:findim:def}{eindige
dimensies}}: op $K[X]$ is de \omterm{def:b1:derivative:def}{afgeleide} \omterm{def:b1:logic:inj}{surjectief} en niet
\omterm{def:b1:logic:inj}{injectief}, $P \mapsto XP$ \omterm{def:b1:logic:inj}{injectief} en niet \omterm{def:b1:logic:inj}{surjectief}
(\cref{cor:b1:linmaps:samedim}); en tussen ruimten van
\emph{verschillende} dimensies is de ene implicatie eenvoudigweg
onmogelijk ($\operatorname{rk} u \leq \min(\dim E, \dim F)$).
\emph{\omterm{def:b1:linmaps:kerim}{Beelden} voorschrijven werkt op een \omterm{def:b1:vspaces:free}{basis}, niet op een
willekeurige familie}: eisen dat $u(1, 0) = a$, $u(0, 1) = b$, $u(1, 1) = c$
overbepaalt $u$ tenzij $c = a + b$; een \omterm{def:b1:linmaps:def}{lineaire afbeelding} is \omterm{def:b1:vspaces:free}{vrij} op een
\omterm{def:b1:vspaces:free}{basis}, geketend overal elders.
\emph{\omterm{def:b1:linmaps:rank}{Rang} wordt niet behouden door samenstelling}: ze kan enkel dalen,
$\operatorname{rk}(vu) \leq \min(\operatorname{rk} u,
\operatorname{rk} v)$ (\cref{exo:b1:linmaps:4}), met exact verlies
gemeten in \cref{exo:b1:linmaps:12}.
\end{remark}

\begin{remark}[Waar deze afbeeldingen naartoe gaan]
\label{rem:b1:linmaps:whereused}
\omterm{def:b1:linmaps:def}{Lineaire afbeeldingen} staan op het punt \emph{matrices} te worden: zodra
bases zijn vastgelegd, codeert \cref{ch:b1:matrices} elke $u \in
\mathcal{L}(E, F)$ door een rechthoekige tabel, en wordt de
samenstelling het matrixproduct --- de dimensiestelling stuwt dan de
theorie van lineaire stelsels in \cref{ch:b1:det}. \omterm{def:b1:linmaps:projection}{Projecties} keren
terug in \cref{ch:b1:euclid} in hun nuttigste bijzondere geval, de
\emph{orthogonale} \omterm{def:b1:linmaps:projection}{projectie}, waar de \omterm{def:b1:linmaps:kerim}{kern} loodrecht op het \omterm{def:b1:linmaps:kerim}{beeld} wordt
gekozen. Het weekendprobleem hieronder drijft de projectoralgebra zo ver
als de gereedschappen van het eerste jaar reiken, tot aan het lemma van
Fitting; het volume van Jaar~2 gaat verder met het spoor en met de
eigenwaardetheorie, waarvoor projectoren op stabiele \omterm{def:b1:vspaces:subspace}{deelruimten} de
basisbouwstenen zijn.
\end{remark}

\begin{remark}[Perspectieven binnen Boek 3: de dimensiestelling nog drie keer]
\label{rem:b1:linmaps:perspectives}
De behoudswet $\dim E = \dim\ker u + \operatorname{rk} u$
zal nog drie keer herlezen worden voor het volume eindigt. In
\cref{ch:b1:det} wordt ze de vorm van oplossingsverzamelingen: een
verenigbaar stelsel met $p$ onbekenden en \omterm{def:b1:linmaps:rank}{rang} $r$ heeft een
oplossingsverzameling van dimensie $p - r$ --- kerndimensie in vermomming.
In \cref{ch:b1:euclid} splitst ze orthogonaal, $\dim F + \dim
F^\perp = \dim E$, en drijft ze elke afstandsberekening aan. In
het weekendprobleem van \cref{ch:b1:multivar} is ze de boekhouder van de
kleinste kwadraten: $n$ waarnemingen, $2$ gefitte parameters, $n - 2$
residuele dimensies, en de pythagoreïsche identiteit $\norm b^2 = \norm
p^2 + \norm{b - p}^2$ is de euclidische schaduw van de dimensiestelling.
Eén stelling, vier gedaanten.
\end{remark}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:b1:linmaps:1}
Welke \omterm{def:b1:logic:map}{afbeeldingen} zijn \omterm{def:b1:linmaps:def}{lineair}?
\begin{enumerate}
  \item $\R^2 \to \R^2$, $(x, y) \mapsto (x + y, x - 2y)$;
  \item $\R^2 \to \R$, $(x, y) \mapsto xy$;
  \item $\R[X] \to \R[X]$, $P \mapsto P' + XP$;
  \item $\mathcal{F}(\R,\R) \to \R$, $f \mapsto f(3)$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:linmaps:2}
Zij $u \colon \R^3 \to \R^3$, $(x,y,z) \mapsto (x + y - z,\; 2x + y
+ z,\; 3x + 2y)$. Bepaal $\ker u$ (\omterm{def:b1:vspaces:free}{basis}, dimensie),
$\operatorname{rk} u$, en een \omterm{def:b1:vspaces:free}{basis} van $\operatorname{im} u$. Is $u$
\omterm{def:b1:logic:inj}{injectief}? \omterm{def:b1:logic:inj}{surjectief}?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:linmaps:3}
Zij $u \colon \R_n[X] \to \R_n[X]$, $P \mapsto P - P'$. Bewijs dat
$u$ een isomorfisme is: eerst via $\ker u$, en daarna door de inverse
te tonen \emph{(beschouw $P + P' + P'' + \dots$)}.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:linmaps:4}
Zij $u \in \mathcal{L}(E, F)$ en $v \in \mathcal{L}(F, G)$, ruimten
\omterm{def:b1:findim:def}{eindigdimensionaal}. Bewijs:
\[
\operatorname{rk}(v \circ u) \leq
\min\bigl(\operatorname{rk} u,\ \operatorname{rk} v\bigr).
\]
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:linmaps:5}
Zij $u$ een endomorfisme van $E$ (\omterm{def:b1:findim:def}{eindigdimensionaal}) met $u^2 =
0$. Bewijs dat $\operatorname{im} u \subseteq \ker u$, dus
$\operatorname{rk} u \leq \frac{\dim E}{2}$. Geef voor $E = \R^2$ een
voorbeeld met gelijkheid.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:linmaps:6}
Zij $p, q$ \omterm{def:b1:linmaps:projection}{projecties} van $E$ met $p \circ q = q \circ p$.
Bewijs dat $p \circ q$ een \omterm{def:b1:linmaps:projection}{projectie} is, met
\[
\operatorname{im}(pq) = \operatorname{im} p \cap \operatorname{im}
q ,
\qquad
\ker (pq) = \ker p + \ker q .
\]
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:linmaps:7}
Zij $u \in \mathcal{L}(E)$, $E$ \omterm{def:b1:findim:def}{eindigdimensionaal}. Bewijs de
equivalentie van:
\begin{enumerate}
  \item $E = \ker u \oplus \operatorname{im} u$;
  \item $\ker u = \ker u^2$;
  \item $\operatorname{im} u = \operatorname{im} u^2$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:linmaps:8}
Zij $\varphi, \psi$ \omterm{def:b1:linmaps:forms}{lineaire vormen} op $E$ met $\ker\varphi
\subseteq \ker\psi$. Bewijs $\psi = \lambda\varphi$ voor een zekere
$\lambda \in K$ (met inbegrip van de gedegenereerde gevallen).
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:linmaps:9}
Zij $u \in \mathcal{L}(E)$ met $\dim E = n$, en veronderstel $u^n = 0$
maar $u^{n-1} \neq 0$ (een \emph{maximaal nilpotent} endomorfisme).
Kies $x$ met $u^{n-1}(x) \neq 0$; bewijs dat $\bigl(x, u(x), \dots,
u^{n-1}(x)\bigr)$ een \omterm{def:b1:vspaces:free}{basis} van $E$ is. \emph{(Pas machten van $u$ toe op
een nulcombinatie, beginnend met $u^{n-1}$.)}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:linmaps:10}
Zij $f \in \mathcal{L}(\R^n)$ met $f \circ f = -\mathrm{id}$.
\begin{enumerate}
  \item Bewijs dat $f$ een isomorfisme is en dat geen enkele $x \neq 0$
        voldoet aan $f(x) = \lambda x$ met $\lambda \in \R$.
  \item Bewijs dat $n$ even is. \emph{Hint: kies $x_1 \neq 0$; toon aan dat
        $\operatorname{Vect}(x_1, f(x_1))$ een vlak is dat stabiel is onder
        $f$; kies $x_2$ erbuiten en herhaal, waarbij je aantoont dat
        $\bigl(x_1, f(x_1), x_2, f(x_2), \dots\bigr)$ \omterm{def:b1:vspaces:free}{vrij} blijft.}
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:linmaps:11}
Zij $u, v \in \mathcal{L}(E, F)$, ruimten \omterm{def:b1:findim:def}{eindigdimensionaal}.
Bewijs de tweezijdige grens
\[
\abs{\operatorname{rk} u - \operatorname{rk} v}
\;\leq\; \operatorname{rk}(u + v)
\;\leq\; \operatorname{rk} u + \operatorname{rk} v .
\]
\emph{(Vergelijk voor de \omterm{def:b1:reals:bounds}{bovengrens} $\operatorname{im}(u+v)$ met
$\operatorname{im} u + \operatorname{im} v$; pas voor de ondergrens
de \omterm{def:b1:reals:bounds}{bovengrens} op een slimme manier toe.)}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:linmaps:12}
(Ongelijkheid van Frobenius) Zij $u \in \mathcal{L}(E, F)$, $w \in
\mathcal{L}(F, G)$ en $v \in \mathcal{L}(G, H)$, alle ruimten
\omterm{def:b1:findim:def}{eindigdimensionaal}. Bewijs de exacte formule
\[
\operatorname{rk}(v \circ w) = \operatorname{rk} w -
\dim\bigl(\ker v \cap \operatorname{im} w\bigr),
\]
en leid de ongelijkheid van Frobenius af
\[
\operatorname{rk}(v \circ w) + \operatorname{rk}(w \circ u)
\;\leq\; \operatorname{rk} w + \operatorname{rk}(v \circ w \circ
u) .
\]
Ga na dat het geval $w = \mathrm{id}_F$ de ongelijkheid van Sylvester
is, in matrixvorm bewezen in \cref{exo:b1:matrices:10}.
\end{exercise}

\section{Probleem: projectorrekening en het lemma van Fitting}

\begin{problem}\label{pb:b1:linmaps:1}
\omterm{def:b1:linmaps:projection}{Projecties} zijn de endomorfismen die directe sommen voortbrengen, en
omgekeerd: elke identiteit $E = F_1 \oplus \dots \oplus F_k$ is
in het geheim een familie van projectoren die optellen tot de identiteit.
Dit probleem ontwikkelt dat woordenboek --- de algebra van één
projector, van twee, van $k$ --- en past vervolgens dezelfde
stabilisatie-ideeën toe op een willekeurig endomorfisme en bewijst
\emph{het lemma van Fitting}: elk endomorfisme van een
eindigdimensionale ruimte splitst in een nilpotent deel en een
inverteerbaar deel. Overal is $E$ een $K$-vectorruimte van
dimensie $n$, en \emph{projector} betekent $p \in \mathcal{L}(E)$
met $p^2 = p$ (\cref{thm:b1:linmaps:projchar}).
\index{projectie}\index{lemma van Fitting}\index{nilpotent}

\medskip
\noindent\textbf{Deel I --- De algebra rond één projector.}
Zij $p$ een projector, $p \neq 0$, $p \neq \mathrm{id}$.
\begin{enumerate}
  \item Toon aan dat $\mathrm{id} - p$ een projector is en identificeer
        $\operatorname{im}(\mathrm{id} - p)$ en
        $\ker(\mathrm{id} - p)$.
  \item Bereken $(\lambda\,\mathrm{id} + \mu\,p)^2$ en bepaal
        alle paren $(\lambda, \mu) \in K^2$ waarvoor
        $\lambda\,\mathrm{id} + \mu\,p$ een projector is.
  \item Toon aan dat het vlak $\operatorname{Vect}(\mathrm{id}, p)$
        van $\mathcal{L}(E)$ stabiel is onder samenstelling, en dat
        voor elke \omterm{def:b1:poly:def}{veelterm} $Q \in K[X]$,
        \[
        Q(p) = Q(0)\,\mathrm{id} + \bigl(Q(1) -
        Q(0)\bigr)\,p .
        \]
  \item Bepaal voor welke $(\lambda, \mu)$ de \omterm{def:b1:logic:map}{afbeelding}
        $\lambda\,\mathrm{id} + \mu\,p$ inverteerbaar is, en geef
        haar inverse in de vorm $\alpha\,\mathrm{id} + \beta\,p$.
        Interpreteer het antwoord via de werking van
        $\lambda\,\mathrm{id} + \mu\,p$ op $\operatorname{im} p$
        en op $\ker p$.
  \item Zij $p'$ een andere projector met hetzelfde \emph{\omterm{def:b1:linmaps:kerim}{beeld}}
        $\operatorname{im} p' = \operatorname{im} p$. Toon aan dat
        $p\,p' = p'$ en $p'\,p = p$. Wat zeggen deze identiteiten
        over het samenstellen van \omterm{def:b1:linmaps:projection}{projecties} op dezelfde \omterm{def:b1:vspaces:subspace}{deelruimte}
        langs verschillende \omterm{def:b1:linmaps:kerim}{kernen}?
\end{enumerate}

\medskip
\noindent\textbf{Deel II --- Twee projectoren.} Zij $p, q$
projectoren van $E$; neem aan dat de karakteristiek niet $2$ is (waar
voor $K = \R, \C$).
\begin{enumerate}[resume]
  \item Veronderstel dat $p + q$ een projector is. Door $(p + q)^2$ te
        ontwikkelen, toon aan dat $pq + qp = 0$; door links met $p$
        samen te stellen, en daarna rechts, leid $pq = qp$ af, en
        besluit $pq = qp = 0$.
  \item Veronderstel omgekeerd $pq = qp = 0$. Toon aan dat $p + q$ een
        projector is, met
        \[
        \operatorname{im}(p + q) = \operatorname{im} p \oplus
        \operatorname{im} q,
        \qquad
        \ker(p + q) = \ker p \cap \ker q .
        \]
  \item Toon aan dat $p - q$ een projector is dan en slechts dan als $pq = qp
        = q$. \emph{(Pas vragen 6--7 toe op $\mathrm{id} - p$
        en $q$.)}
  \item Toon de meetkundige betekenis van $pq = qp = q$: het geldt dan
        en slechts dan als $\operatorname{im} q \subseteq
        \operatorname{im} p$ en $\ker p \subseteq \ker q$. (Men
        schrijft dan $q \leq p$: ``$q$ projecteert op minder, langs
        meer''.)
  \item Zij nu $p, q$ commuterend. Herinner uit
        \cref{exo:b1:linmaps:6} dat $pq$ de projector is op
        $\operatorname{im} p \cap \operatorname{im} q$ langs
        $\ker p + \ker q$. Toon aan dat $r = p + q - pq$ een
        projector is met
        \[
        \operatorname{im} r = \operatorname{im} p +
        \operatorname{im} q,
        \qquad
        \ker r = \ker p \cap \ker q .
        \]
        \emph{(Beschouw $\mathrm{id} - r = (\mathrm{id} -
        p)(\mathrm{id} - q)$.)}
\end{enumerate}

\medskip
\noindent\textbf{Deel III --- Ontbindingen van de identiteit.}
\begin{enumerate}[resume]
  \item Zij $E = F_1 \oplus \dots \oplus F_k$ en, voor $x = x_1 +
        \dots + x_k$ (unieke ontbinding, $x_i \in F_i$), stel
        $p_i(x) = x_i$. Toon aan dat elke $p_i$ een projector is, dat
        $p_i p_j = 0$ voor $i \neq j$, en dat $p_1 + \dots + p_k
        = \mathrm{id}$; identificeer $\operatorname{im} p_i$ en
        $\ker p_i$.
  \item Zij omgekeerd $p_1, \dots, p_k \in \mathcal{L}(E)$
        zodanig dat $p_1 + \dots + p_k = \mathrm{id}$ en $p_i p_j =
        0$ voor alle $i \neq j$. Toon aan dat elke $p_i$ een
        projector is en dat $E = \operatorname{im} p_1 \oplus
        \dots \oplus \operatorname{im} p_k$.
  \item Twee projectoren met $p + q = \mathrm{id}$: toon aan dat $pq
        = qp = 0$ automatisch geldt.
  \item Drie projectoren met $p + q + r = \mathrm{id}$: toon aan
        dat $p + q$ een projector is, en leid uit vraag 6
        af dat \emph{alle} paarsgewijze producten verdwijnen --- dus $E =
        \operatorname{im} p \oplus \operatorname{im} q \oplus
        \operatorname{im} r$, zonder enige hypothese op de producten.
  \item Voor $k$ projectoren met $p_1 + \dots + p_k =
        \mathrm{id}$: toon eerst dat voor willekeurige \omterm{def:b1:vspaces:subspace}{deelruimten} $\dim
        (F_1 + \dots + F_k) \leq \dim F_1 + \dots + \dim F_k$,
        met gelijkheid dan en slechts dan als de som direct is; toon
        vervolgens $E = \operatorname{im} p_1 + \dots +
        \operatorname{im} p_k$, en bewijs dat \emph{als}
        bovendien $\sum_i \operatorname{rk} p_i \leq n$, de som
        direct is en $p_i p_j = 0$ voor $i \neq j$.
\end{enumerate}

\medskip
\noindent\textbf{Deel IV --- Herhaalde \omterm{def:b1:linmaps:kerim}{kernen}: het lemma van Fitting.}
Zij $u \in \mathcal{L}(E)$, $\dim E = n$.
\begin{enumerate}[resume]
  \item Toon de twee ketens aan, geldig voor elke $k \geq 0$:
        \[
        \ker u^k \subseteq \ker u^{k+1},
        \qquad
        \operatorname{im} u^{k+1} \subseteq \operatorname{im}
        u^k .
        \]
  \item Toon aan dat als $\ker u^{r} = \ker u^{r+1}$ voor een zekere $r$,
        dan $\ker u^{k} = \ker u^{r}$ voor alle $k \geq r$; formuleer
        en bewijs de analoge stabilisatie voor \omterm{def:b1:linmaps:kerim}{beelden}.
  \item Leid af dat er een kleinste geheel getal $r$ is met $\ker
        u^{r} = \ker u^{r+1}$, dat $r \leq n$, en dat \omterm{def:b1:linmaps:kerim}{beelden}
        stabiliseren bij dezelfde $r$.
  \item (Lemma van Fitting) Bewijs dat
        \[
        E \;=\; \ker u^{r} \,\oplus\, \operatorname{im} u^{r} .
        \]
  \item Toon aan dat beide \omterm{def:b1:vspaces:subspace}{deelruimten} stabiel zijn onder $u$, dat de
        beperking van $u$ tot $\ker u^{r}$ nilpotent is, en dat
        de beperking van $u$ tot $\operatorname{im} u^{r}$ een
        isomorfisme van $\operatorname{im} u^{r}$ is: elk
        endomorfisme is, op een canonieke \omterm{def:b1:vspaces:sum}{directe som}, ``nilpotent
        plus inverteerbaar''.
  \item Zij $\pi$ de projector op $\ker u^{r}$ langs
        $\operatorname{im} u^{r}$. Toon aan dat $\pi \circ u = u
        \circ \pi$.
\end{enumerate}

\medskip
\noindent\textbf{Deel V --- Een uitgewerkt geval, en synthese.}
\begin{enumerate}[resume]
  \item Zij $u(x, y, z) = (y, 0, z)$ op $\R^3$. Bereken $u^2$ en
        $u^3$, bepaal de stabilisatie-index $r$, de
        \omterm{def:b1:vspaces:subspace}{deelruimten} $\ker u^{r}$ en $\operatorname{im} u^{r}$, de
        Fitting-projector $\pi$, en verifieer op de formules dat
        $\pi u = u\pi$ en dat $u$ nilpotent is op de ene factor,
        \omterm{def:b1:logic:inj}{bijectief} op de andere.
  \item Toon de equivalenties: $u$ nilpotent $\iff$ $\ker u^{r} =
        E$ $\iff$ $\pi = \mathrm{id}$; en leid af dat een
        nilpotent endomorfisme van een $n$-dimensionale ruimte
        altijd voldoet aan $u^{n} = 0$ (de nilpotentie-index nooit
        de dimensie overschrijdt).
  \item (Uniciteit) Veronderstel $E = A \oplus B$ met $A, B$ stabiel
        onder $u$, de beperking $u|_A$ nilpotent en $u|_B$
        \omterm{def:b1:logic:inj}{bijectief}. Bewijs $A = \ker u^{r}$ en $B =
        \operatorname{im} u^{r}$: de Fitting-ontbinding is
        uniek.
  \item Synthese, in vier zinnen: welk woordenboek Deel III
        vestigt tussen directe sommen en families van
        projectoren; waarom vraag 14 geen producthypothesen
        nodig had terwijl vraag 15 een ranghypothese nodig had
        (en welk gereedschap van Jaar~2, het spoor, ze verwijdert); in welke
        zin het lemma van Fitting de gestabiliseerde versie is van
        \cref{exo:b1:linmaps:7}; en wat de twee factoren van
        Fitting worden in de eigenwaardetheorie van het volume van
        Jaar~2. Noem de stelling die in Deel IV bewezen wordt.
\end{enumerate}
\end{problem}
