\chapter{Lineaire differentiaalvergelijkingen}\label{ch:b1:diffeq}

Voor het eerst tegengekomen in het deel over de middelbare school, worden
differentiaalvergelijkingen hier behandeld met volledige bewijzen en in
grotere algemeenheid: lineaire vergelijkingen van de eerste orde met
variabele coëfficiënten (volledig opgelost door de methode van \omterm{thm:b1:diffeq:voc}{variatie
van constanten}) en lineaire vergelijkingen van de tweede orde met
constante coëfficiënten, het model voor oscillaties. Beide gevallen
vertonen dezelfde structuur: \emph{algemene oplossing $=$ één particuliere
oplossing $+$ algemene oplossing van de homogene vergelijking}.

\section{Lineaire vergelijkingen van de eerste orde}

\begin{definition}\label{def:b1:diffeq:linear1}
Zij $I$ een interval en $a, b \colon I \to \R$ (of $\C$)
continu. De vergelijking
\[
(E)\colon\quad y' + a(x)\,y = b(x),
\]
in de onbekende afleidbare functie $y \colon I \to \R$ (of $\C$),
is een \emph{lineaire differentiaalvergelijking van de eerste
orde}\index{differentiaalvergelijking!lineair van de eerste orde}. De
vergelijking $(H)\colon y' + a(x) y = 0$ is haar \emph{homogene}
vergelijking.
\end{definition}

\begin{theorem}[De homogene vergelijking oplossen]\label{thm:b1:diffeq:homogeneous1}
Zij $A$ een primitieve van $a$ op $I$ (die bestaat:
\cref{ch:b1:integration}). De oplossingen van $(H)$ op $I$ zijn precies
de functies
\[
y(x) = \lambda\, \eu^{-A(x)}, \qquad \lambda \in \R \text{ (of } \C).
\]
\end{theorem}

\begin{proof}
Deze functies zijn oplossingen: $y' = -\lambda A' \eu^{-A} = -a y$.
Omgekeerd, zij $y$ een oplossing van $(H)$ en stel $z(x) = y(x)\, \eu^{A(x)}$.
Dan is
\[
z' = y' \eu^{A} + y\, a\, \eu^{A} = (y' + ay)\, \eu^{A} = 0 ,
\]
dus $z$ is constant op het interval $I$, zeg $z = \lambda$: $y =
\lambda \eu^{-A}$. (Let op de logica: geen enkele oplossing gaat verloren,
want \emph{elke} oplossing is in de aangekondigde vorm geschreven.)
\end{proof}

\begin{example}[Een homogene vergelijking met variabele coëfficiënt]\label{ex:b1:diffeq:varcoeff}
Los $y' + (\cos x)\,y = 0$ op op $\R$. Een primitieve van $a(x) =
\cos x$ is $A(x) = \sin x$, dus de oplossingen zijn
\[
y(x) = \lambda\,\eu^{-\sin x}, \qquad \lambda \in \R .
\]
Twee lezingen. Elke oplossing is periodiek (periode $2\pi$) en wordt
nooit nul tenzij $\lambda = 0$ --- het teken van $\lambda$ is het
teken van $y$ voor altijd, aangezien een exponentiële functie geen nul
kan overschrijden. En de oplossing door $y(0) = y_0$ is $y_0\eu^{-\sin x}$:
precies één kromme van de familie door elk beginpunt, het
eendimensionale beeld van \cref{thm:b1:diffeq:voc}~(2).
\end{example}

\begin{theorem}[Variatie van constanten; Cauchy-probleem]\label{thm:b1:diffeq:voc}
Met de notatie hierboven:
\begin{enumerate}
  \item De oplossingen van $(E)$ op $I$ zijn precies
        \[
        y(x) = \Bigl(\lambda + \int_{x_0}^{x} b(t)\, \eu^{A(t)} \dd t
        \Bigr)\, \eu^{-A(x)},
        \qquad \lambda \in \R,
        \]
        waar $x_0 \in I$ vast is. Equivalent: algemene oplossing van
        $(H)$ plus één particuliere oplossing van $(E)$.
  \item Voor elke $x_0 \in I$ en $y_0$ heeft het \emph{Cauchy-probleem}\index{Cauchy-probleem}
        ``$(E)$ en $y(x_0) = y_0$''
        precies één oplossing op $I$.
\end{enumerate}
\end{theorem}

\begin{proof}
(1) Volgens de methode die \emph{variatie van
constanten}\index{variatie van constanten} heet, zoeken we oplossingen van
de vorm $y = \mu(x)\, \eu^{-A(x)}$ met $\mu$ afleidbaar --- er gaat geen
algemeenheid verloren, aangezien elke functie op $I$ zo geschreven kan
worden ($\mu = y\,\eu^{A}$). Door substitutie,
\[
y' + ay = \mu' \eu^{-A} - \mu a \eu^{-A} + a\mu\eu^{-A}
= \mu'\, \eu^{-A},
\]
dus $y$ is een oplossing van $(E)$ dan en slechts dan als $\mu'(x) = b(x)\,\eu^{A(x)}$,
dan en slechts dan als $\mu(x) = \lambda + \int_{x_0}^x b(t)\eu^{A(t)}\dd t$ voor
een zekere constante $\lambda$ (twee primitieven van dezelfde continue
functie op een interval verschillen een constante).

(2) In de formule is $y(x_0) = \lambda\,\eu^{-A(x_0)}$: de voorwaarde
$y(x_0) = y_0$ bepaalt $\lambda = y_0 \eu^{A(x_0)}$ eenduidig.
\end{proof}

\begin{example}\label{ex:b1:diffeq:order1}
Los $y' + \dfrac{y}{x} = x^2$ op op $I = \intoo{0}{+\infty}$. Hier is
$a(x) = \frac 1x$, $A(x) = \ln x$, $\eu^{-A(x)} = \frac 1x$.
Homogene oplossingen: $\frac{\lambda}{x}$. \omterm{thm:b1:diffeq:voc}{Variatie van constanten}:
$\mu'(x) = x^2 \cdot x = x^3$, dus $\mu = \frac{x^4}{4} + \lambda$, en
\[
y(x) = \frac{x^3}{4} + \frac{\lambda}{x}, \qquad \lambda \in \R .
\]
Met de beginvoorwaarde $y(1) = 0$: $\lambda = -\frac14$.
\emph{Controle:} $y' + \frac yx = \frac{3x^2}{4} - \frac{\lambda}{x^2} +
\frac{x^2}{4} + \frac{\lambda}{x^2} = x^2$.
\end{example}

\begin{example}[Gokken verslaat integreren]\label{ex:b1:diffeq:guess}
Los $y' + 2x\,y = x$ op op $\R$. \omterm{thm:b1:diffeq:voc}{Variatie van constanten} werkt ($A =
x^2$, $\mu' = x\,\eu^{x^2}$, $\mu = \frac12\eu^{x^2} + \lambda$),
maar opmerken dat de \emph{constante} $y_p = \frac12$ de
vergelijking oplost ($0 + 2x\cdot\frac12 = x$) is sneller. Met de
homogene oplossingen $\lambda\,\eu^{-x^2}$:
\[
y(x) = \frac12 + \lambda\,\eu^{-x^2}, \qquad \lambda \in \R .
\]
Elke oplossing convergeert bijzonder snel naar $\frac12$ als $x \to
\pm\infty$: de constante particuliere oplossing is een
\emph{evenwicht} waar alle oplossingen naartoe gaan. Het inzicht: voordat
je de algemene methode inzet, besteed tien seconden aan het zoeken naar een
voor de hand liggende particuliere oplossing (constante, monoom, veelvoud
van het rechterlid); de structuurstelling maakt het werk dan af.
\end{example}

\begin{remark}[Intervallen doen ertoe]
De stelling leeft op een \emph{interval} waar $a$ en $b$ continu
zijn. Voor $y' + \frac yx = 0$ op $\R^*$ zijn de oplossingen
$\frac{\lambda}{x}$ op $\intoo{0}{+\infty}$ en
$\frac{\mu}{x}$ op $\intoo{-\infty}{0}$ met \emph{onafhankelijke}
constanten: er is geen reden waarom één enkele formule zou aansluiten over
de singulariteit in $0$.
\end{remark}

\begin{example}[Een complex rechterlid, twee reële antwoorden]\label{ex:b1:diffeq:complexrhs}
Los $y' - y = \cos x$ en $y' - y = \sin x$ in één beweging op. Werk
in $\C$ met het rechterlid $\eu^{\iu x}$: het proberen van $y_p =
c\,\eu^{\iu x}$ geeft $c(\iu - 1)\eu^{\iu x} = \eu^{\iu x}$, dus
\[
c = \frac1{\iu - 1} = \frac{-1 - \iu}2,
\qquad
y_p = -\frac{(1 + \iu)(\cos x + \iu\sin x)}2
= \frac{\sin x - \cos x}2 + \iu\,\frac{-\sin x - \cos x}2 .
\]
Aangezien de vergelijking reële coëfficiënten heeft, splitsen reëel en
imaginair deel zich: $\frac{\sin x - \cos x}2$ lost $y' - y = \cos x$ op,
en $-\frac{\sin x + \cos x}2$ lost $y' - y = \sin x$ op (controleer
het eerste: afgeleide $\frac{\cos x + \sin x}2$, min de
functie, geeft $\cos x$). Eén complexe regel verving twee toepassingen van
\omterm{thm:b1:diffeq:voc}{variatie van constanten} --- dezelfde besparing die
\cref{met:b1:diffeq:particular} systematiseert voor de tweede orde,
en een terugkerend rendement van \cref{ch:b1:complex}.
\end{example}

\section{Lineaire vergelijkingen van de tweede orde met constante coëfficiënten}

\begin{definition}\label{def:b1:diffeq:linear2}
Zij $a, b \in \R$ en $f \colon I \to \R$ continu. De vergelijking
\[
(E)\colon\quad y'' + a\,y' + b\,y = f(x)
\]
is een \emph{lineaire vergelijking van de tweede orde met constante
coëfficiënten}\index{differentiaalvergelijking!lineair van de tweede orde};
$(H)\colon y'' + ay' + by = 0$ is haar homogene vergelijking, en
$\chi(r) = r^2 + ar + b$ haar \emph{karakteristieke
veelterm}\index{karakteristieke veelterm}.
\end{definition}

\begin{theorem}[Homogene oplossingen]\label{thm:b1:diffeq:homogeneous2}
Zij $\Delta = a^2 - 4b$ de discriminant van $\chi$. De reële
oplossingen van $(H)$ op $\R$ zijn:
\begin{enumerate}
  \item als $\Delta > 0$, met $r_1 \neq r_2$ de twee reële wortels:
        $\;y = \lambda\, \eu^{r_1 x} + \mu\, \eu^{r_2 x}$;
  \item als $\Delta = 0$, met $r_0$ de dubbele wortel:
        $\;y = (\lambda + \mu x)\, \eu^{r_0 x}$;
  \item als $\Delta < 0$, met wortels $\alpha \pm \iu\omega$
        ($\omega > 0$):
        $\;y = \eu^{\alpha x} (\lambda \cos\omega x + \mu
        \sin\omega x)$;
\end{enumerate}
in elk geval met $(\lambda, \mu)$ die $\R^2$ doorloopt.
\end{theorem}

\begin{proof}
Merk eerst op dat voor $r \in \C$, $x \mapsto \eu^{rx}$ een oplossing van $(H)$ is
dan en slechts dan als $\chi(r) = 0$ (substitueer: $(r^2 + ar + b)\eu^{rx} =
0$). Dit is waarom exponentiële functies de natuurlijke eerste gok zijn:
differentiëren werkt op $\eu^{rx}$ als vermenigvuldiging met het getal
$r$, zodat de differentiaalvergelijking de numerieke vergelijking
$\chi(r) = 0$ wordt --- het hele analytische probleem wordt samengeperst tot
het vinden van de wortels van één kwadratische vergelijking.

De sleutelstap is een verandering van onbekende die de orde verlaagt. Zij $r$
een (mogelijk complexe) wortel van $\chi$ en schrijf $y = z\, \eu^{rx}$,
wat geen algemeenheid verliest. Dan is
\[
y'' + ay' + by
= \bigl(z'' + (2r + a) z' + \chi(r) z\bigr)\eu^{rx}
= \bigl(z'' + (2r + a) z'\bigr)\eu^{rx},
\]
dus $(H)$ wordt de vergelijking \emph{van de eerste orde} $u' + (2r + a) u = 0$
voor $u = z'$.

\emph{Geval $\Delta \neq 0$:} kies $r = r_1$; dan is $2r_1 + a = r_1 -
r_2$ (aangezien $r_1 + r_2 = -a$). Volgens
\cref{thm:b1:diffeq:homogeneous1} is $z' = c\,\eu^{(r_2 - r_1)x}$ voor
een zekere constante $c$; door te integreren op $\R$, $z = \mu\, \eu^{(r_2 - r_1)x}
+ \lambda$ met $\mu = \frac{c}{r_2 - r_1}$, en dus $y = z\,
\eu^{r_1 x} = \lambda\,\eu^{r_1x} + \mu\,\eu^{r_2x}$.
Wanneer $\Delta < 0$, zijn de wortels $\alpha \pm \iu\omega$ en de
complexe oplossingen zijn $y = c_1\eu^{(\alpha+\iu\omega)x} +
c_2\eu^{(\alpha-\iu\omega)x}$ met $c_1, c_2 \in \C$. Welke van
deze zijn reëelwaardig? Aangezien $\conj{\eu^{(\alpha+\iu\omega)x}} =
\eu^{(\alpha-\iu\omega)x}$, is de complex toegevoegde van $y$ gelijk aan $\conj{c_2}\,
\eu^{(\alpha+\iu\omega)x} + \conj{c_1}\,\eu^{(\alpha-\iu\omega)x}$,
en $y = \conj y$ voor alle $x$ dwingt $c_2 = \conj{c_1}$ af (de twee
exponentiële functies zijn lineair onafhankelijk: evalueer in twee punten, of
vergelijk in $x=0$ na deling door $\eu^{\alpha x}$). Schrijvend $c_1
= \frac{\lambda - \iu\mu}2$ met $\lambda, \mu$ reëel:
\[
y = 2\,\Re\Bigl(c_1\,\eu^{\alpha x}(\cos\omega x +
\iu\sin\omega x)\Bigr)
= \eu^{\alpha x}\bigl(\lambda\cos\omega x + \mu\sin\omega
x\bigr),
\]
en omgekeerd is elke zodanige functie een oplossing (reëel deel van een
complexe oplossing van een reële vergelijking): de reële oplossingsruimte is
zoals aangekondigd.

\emph{Geval $\Delta = 0$:} $r = r_0$, $2r_0 + a = 0$, dus $z'' = 0$:
$z = \lambda + \mu x$ en $y = (\lambda + \mu x)\eu^{r_0 x}$.
\end{proof}

\begin{example}[Een Cauchy-probleem, van begin tot eind]\label{ex:b1:diffeq:cauchy2}
Los $y'' - 3y' + 2y = 0$ met $y(0) = 0$, $y'(0) = 1$ op. De
\omterm{def:b1:diffeq:linear2}{karakteristieke veelterm} $r^2 - 3r + 2 = (r - 1)(r - 2)$ heeft de
reële wortels $1$ en $2$: algemene oplossing $y = \lambda\eu^{x} +
\mu\eu^{2x}$. De twee voorwaarden geven het lineaire stelsel
\[
\lambda + \mu = 0, \qquad \lambda + 2\mu = 1 ,
\]
dus $\mu = 1$, $\lambda = -1$:
\[
y(x) = \eu^{2x} - \eu^{x} .
\]
Controle: $y(0) = 0$; $y' = 2\eu^{2x} - \eu^x$ heeft $y'(0) = 1$; en
$y'' - 3y' + 2y = (4 - 6 + 2)\eu^{2x} + (-1 + 3 - 2)\eu^x = 0$.
Merk de vorm van het antwoord op: nabij $-\infty$ domineert de trage mode
$-\eu^x$; nabij $+\infty$ doet de snelle mode $\eu^{2x}$ dat.
Het lezen van oplossingen als superposities van modes met verschillende afname-
of groeisnelheden is de winstgevende gewoonte --- zo is de
transiënt/steady-state-splitsing van het weekendprobleem georganiseerd.
\end{example}

\begin{theorem}[Structuur en Cauchy-probleem]\label{thm:b1:diffeq:structure2}
\begin{enumerate}
  \item Als $y_p$ één particuliere oplossing van $(E)$ is, zijn de oplossingen van
        $(E)$ precies $y_p + y_h$, met $y_h$ die de
        oplossingen van $(H)$ doorloopt.
  \item (Superpositie) Als $y_1$ een oplossing is van $y'' + ay' + by = f_1$ en
        $y_2$ een oplossing van $y'' + ay' + by = f_2$, dan is $y_1 + y_2$ een oplossing van
        de vergelijking met rechterlid $f_1 + f_2$.
  \item Voor alle $x_0 \in I$ en $(y_0, y_0')$ heeft het \omterm{thm:b1:diffeq:voc}{Cauchy-probleem}
        ``$(E)$, $y(x_0) = y_0$, $y'(x_0) = y_0'$'' precies één
        oplossing op $I$. \emph{(Bestaan mits een particuliere
        oplossing; eenduidigheid volledig.)}
\end{enumerate}
\end{theorem}

\begin{proof}
(1) $y$ is een oplossing van $(E)$ dan en slechts dan als $y - y_p$ een oplossing is van $(H)$, door lineariteit van $y
\mapsto y'' + ay' + by$. (2) is dezelfde lineariteit.

(3) Volgens (1) volstaat het te bewijzen dat de constanten $(\lambda,
\mu)$ altijd, eenduidig, aangepast kunnen worden aan willekeurige data $(y_0, y_0')$.
Door de variabele te verschuiven, neem $x_0 = 0$. In geval (1) van
\cref{thm:b1:diffeq:homogeneous2} geeft $y = \lambda\eu^{r_1x} +
\mu\eu^{r_2x}$
\[
y(0) = \lambda + \mu, \qquad y'(0) = r_1\lambda + r_2\mu :
\]
een lineair stelsel in $(\lambda, \mu)$ waarvan de determinant $r_2 -
r_1 \neq 0$ is; door het expliciet op te lossen, $\mu = \frac{y_0' -
r_1y_0}{r_2 - r_1}$ en $\lambda = y_0 - \mu$: precies één
oplossing. In geval (2), $y(0) = \lambda$ en $y'(0) = r_0\lambda +
\mu$: het stelsel is driehoekig met determinant $1$, opgelost door
$\lambda = y_0$, $\mu = y_0' - r_0y_0$. In geval (3), $y(0) =
\lambda$ en $y'(0) = \alpha\lambda + \omega\mu$: determinant
$\omega \neq 0$, opgelost door $\lambda = y_0$, $\mu = \frac{y_0' -
\alpha y_0}\omega$. In elk geval is de \omterm{def:b1:logic:map}{afbeelding} $(\lambda, \mu) \mapsto
(y(x_0), y'(x_0))$ een lineaire bijectie --- de taal van
\cref{ch:b1:linmaps} zal deze gevalscontrole comprimeren tot één
zin.
\end{proof}

\begin{method}[Particuliere oplossing voor $f(x) = P(x)\,\eu^{\gamma x}$]\label{met:b1:diffeq:particular}
Wanneer het rechterlid $P(x)\,\eu^{\gamma x}$ is met $P$ een
veelterm en $\gamma \in \R$ (dit dekt veeltermen, exponentiële functies,
en via complexe $\gamma$ of superpositie, $\cos$ en $\sin$): zoek
een particuliere oplossing van de vorm
\[
y_p(x) = x^{m}\, Q(x)\, \eu^{\gamma x},
\qquad
m = \text{multipliciteit van } \gamma \text{ als wortel van } \chi
\ (m = 0, 1 \text{ of } 2),
\]
met $Q$ een veelterm van dezelfde graad als $P$, waarvan de coëfficiënten
gevonden worden door substitutie en identificatie. Voor $f = K\cos\omega x$
(of $\sin$), los op met rechterlid $K\eu^{\iu\omega x}$ en neem
het reële (resp.\ imaginaire) deel.
\end{method}

\begin{example}[Superpositie in actie]\label{ex:b1:diffeq:superposition}
Los $y'' - y = \eu^{x} + 4$ op op $\R$. Homogeen: $\chi(r) = r^2
- 1$, wortels $\pm1$, dus $y_h = \lambda\eu^x + \mu\eu^{-x}$. Splits
het rechterlid en behandel elk stuk met de methodebox.
\emph{Stuk $\eu^x$:} hier is $\gamma = 1$ een enkelvoudige wortel van
$\chi$, dus probeer $y_1 = c\,x\,\eu^x$: dan $y_1'' - y_1 = c(x +
2)\eu^x - cx\eu^x = 2c\,\eu^x$, wat $c = \frac12$ geeft.
\emph{Stuk $4$:} $\gamma = 0$ is geen wortel; de constante $y_2 =
-4$ werkt. Door superpositie
(\cref{thm:b1:diffeq:structure2}~(2)):
\[
y = \frac{x\,\eu^x}2 - 4 + \lambda\,\eu^x + \mu\,\eu^{-x},
\qquad (\lambda, \mu) \in \R^2 .
\]
Merk op hoe de twee stukken \emph{verschillende} vormen vereisten ($m = 1$
tegenover $m = 0$): de multipliciteitstest wordt op elke
exponent apart toegepast, wat het hele punt is van het splitsen van het
rechterlid voordat er gegokt wordt.
\end{example}

\begin{example}[De multipliciteitsregel aan het werk]\label{ex:b1:diffeq:multiplicity}
Los $y'' + y' = x$ op op $\R$. Het rechterlid is $P(x)\eu^{0
\cdot x}$ met $P(x) = x$, en $\gamma = 0$ is een \emph{enkelvoudige} wortel
van $\chi(r) = r^2 + r = r(r + 1)$: dus $m = 1$, en de juiste
gok is $y_p = x\,(\alpha x + \beta) = \alpha x^2 + \beta x$, één
graad hoger dan $P$. Door substitutie:
\[
y_p'' + y_p' = 2\alpha + (2\alpha x + \beta)
= 2\alpha x + (2\alpha + \beta) ,
\]
en identificatie met $x$ geeft $\alpha = \frac12$, $\beta = -1$:
$y_p = \frac{x^2}2 - x$. Algemene oplossing: $y = \frac{x^2}2 - x +
\lambda + \mu\,\eu^{-x}$. Hadden we $y_p = \alpha x + \beta$ gegokt
(met verwaarlozing van de multipliciteit), dan zou substitutie $y_p'' + y_p'
= \alpha$ geven, een constante --- geen enkele keuze van $\alpha, \beta$ kan
$x$ matchen, en de mislukking is structureel: constanten lossen de
homogene vergelijking al op, dus ze zijn onzichtbaar voor het linkerlid.
De factor $x^m$ bestaat juist om uit de homogene
oplossingsruimte te klimmen.
\end{example}

\begin{remark}[De verzekeringspolis van dertig seconden]\label{rem:b1:diffeq:check}
Elke opgeloste vergelijking in dit hoofdstuk eindigt met een
substitutiecontrole, en dit is niet decoratief. Een berekening van een
differentiaalvergelijking rijgt vele kleine stappen aaneen (een primitieve, een productregel,
twee constanten), en één enkele tekenfout plant zich onzichtbaar voort;
de uiteindelijke formule terugsubstitueren in de vergelijking vangt
in wezen ze allemaal op tegen de kostprijs van één differentiatie.
Kweek de reflex in drie lagen: controleer de \emph{particuliere
oplossing} alleen (het homogene deel valt toch weg), controleer de
\emph{beginvoorwaarden} op de volledige oplossing, en wanneer een
parameter aanwezig is, controleer een \emph{gedegenereerde waarde} (geeft de
formule voor algemene $\Omega$ het bekende antwoord in
$\Omega = 0$ terug?). De gewoonte kost een halve minuut; ze zet
``waarschijnlijk juist'' om in ``geverifieerd''.
\end{remark}

\begin{remark}[Veelvoorkomende valkuilen]\label{rem:b1:diffeq:pitfalls}
\begin{enumerate}
  \item \emph{Normaliseer eerst.} De formules veronderstellen dat de vergelijking
        $y' + a(x)y = b(x)$ luidt --- coëfficiënt $1$ op $y'$.
        Voor $xy' - 2y = x^3$, deel door $x$ (op een interval
        dat $0$ vermijdt) voordat je $a$ en $b$ identificeert, zoals in
        \cref{exo:b1:diffeq:2}.
  \item \emph{Eén constante per dimensie, aan het eind vastgelegd.} De
        algemene oplossing van de eerste orde draagt één constante, die van de
        tweede orde twee; beginvoorwaarden worden opgelegd aan de
        \emph{volledige} oplossing $y_p + y_h$, nooit aan $y_h$
        alleen --- ze opleggen voordat $y_p$ is toegevoegd is de meest
        frequente structurele fout.
  \item \emph{Let op de multipliciteit.} Een gok voor de particuliere oplossing
        die de homogene vergelijking oplost is onzichtbaar voor het
        linkerlid; de factor $x^m$ van
        \cref{met:b1:diffeq:particular} is niet optioneel
        (\cref{ex:b1:diffeq:multiplicity}).
  \item \emph{Intervallen zijn deel van het antwoord.} Oplossingen leven
        op intervallen waar de coëfficiënten continu zijn;
        aansluiten over een singulariteit kan valse constanten creëren
        (\cref{exo:b1:diffeq:12}) of eenduidigheid vernietigen. ``Los
        op op $\R^*$'' betekent twee onafhankelijke problemen.
\end{enumerate}
\end{remark}

\begin{example}[Aandrijving buiten resonantie]\label{ex:b1:diffeq:offresonance}
Los $y'' + 4y = \sin x$ op op $\R$. Natuurlijke frequentie $2$, aandrijvings-
frequentie $1$: aangezien $\iu$ \emph{geen} wortel van $\chi(r) = r^2
+ 4$ is, is de multipliciteit $m = 0$ en volstaat een gewone sinusoïde.
Het proberen van $y_p = \alpha\sin x$ (geen cosinus nodig: de vergelijking heeft
geen $y'$-term, en $\sin$ regenereert $\sin$):
\[
y_p'' + 4y_p = -\alpha\sin x + 4\alpha\sin x = 3\alpha\sin x ,
\]
dus $\alpha = \frac13$ en de algemene oplossing is
\[
y = \frac{\sin x}3 + \lambda\cos 2x + \mu\sin 2x .
\]
Elke oplossing blijft begrensd: een superpositie van twee oscillaties
op de frequenties $1$ (geforceerd) en $2$ (natuurlijk). Vergelijk met
het volgende voorbeeld, waar aandrijving \emph{op} de natuurlijke frequentie
de vorm van het antwoord zelf verandert.
\end{example}

\begin{example}[Een geforceerde oscillatie]\label{ex:b1:diffeq:oscillation}
Los $y'' + y = \cos x$, $y(0) = 0$, $y'(0) = 0$ op.

\emph{Homogeen:} $\chi(r) = r^2 + 1$, wortels $\pm\iu$: $y_h =
\lambda\cos x + \mu \sin x$.

\emph{Particulier:} rechterlid $\Re(\eu^{\iu x})$ met $\gamma =
\iu$ een enkelvoudige wortel van $\chi$: probeer $z_p = c\, x\, \eu^{\iu x}$
($c \in \C$). Dan $z_p'' + z_p = c\,(2\iu)\eu^{\iu x}$, wat gelijk is aan
$\eu^{\iu x}$ voor $c = \frac{1}{2\iu} = -\frac\iu2$. Dus $z_p =
-\frac{\iu}{2} x (\cos x + \iu \sin x)$ en $y_p = \Re(z_p) =
\frac{x \sin x}{2}$.

\emph{Algemene oplossing:} $y = \frac{x\sin x}{2} + \lambda\cos x +
\mu\sin x$. Voorwaarden: $y(0) = \lambda = 0$; $y' = \frac{\sin x +
x\cos x}{2} + \mu\cos x$, dus $y'(0) = \mu = 0$. Antwoord: $y =
\frac{x\sin x}{2}$ --- een oscillatie waarvan de amplitude lineair groeit:
het \emph{resonantie}\index{resonantie}verschijnsel, veroorzaakt door het
aandrijven van het systeem op zijn natuurlijke frequentie.
\end{example}

\begin{remark}[Intermezzo: lineariteit is een meetkunde]\label{rem:b1:diffeq:interlude}
Kijk terug naar de vorm van elke oplossingsverzameling in dit hoofdstuk: een
speciale oplossing plus een ruimte van homogene oplossingen met één
vrije constante (eerste orde) of twee (tweede orde). De hoofdstukken over
lineaire algebra (\cref{ch:b1:vspaces,ch:b1:findim,ch:b1:linmaps})
zullen de precieze
vocabulaire leveren: de \omterm{def:b1:logic:map}{afbeelding} $L(y) = y'' + ay' + by$ is \emph{lineair}, haar
homogene oplossingen vormen de \emph{kern} van $L$, een vector-
ruimte waarvan de \emph{dimensie} gelijk is aan de orde van de vergelijking ---
dat is de eerlijke inhoud van ``één constante per orde'' --- en
de oplossingsverzameling van $L(y) = f$ is een \emph{affiene deelruimte}, een
verschuiving van de kern. Zelfs de \omterm{def:b1:logic:map}{Cauchy-afbeelding} $(\lambda, \mu)
\mapsto (y(x_0), y'(x_0))$ van \cref{thm:b1:diffeq:structure2} is een
lineaire bijectie tussen twee vlakken, d.w.z.\ een inverteerbaar $2
\times 2$-stelsel (\cref{ch:b1:matrices}). Niets in dit hoofdstuk
zal overgedaan hoeven te worden --- het zal alleen hernoemd hoeven te worden, en
het hernoemen is de best mogelijke opwarming voor lineaire algebra:
elke abstracte definitie daar heeft haar bestaansrecht hier al
verdiend.
\end{remark}

\begin{remark}[Waar dit hoofdstuk gebruikt wordt]\label{rem:b1:diffeq:whereused}
De structuurstelling --- oplossingen van $(E)$ vormen ``een particuliere
oplossing plus de oplossingen van $(H)$'' --- is de eerste verschijning
van een patroon dat \cref{ch:b1:vspaces,ch:b1:linmaps} zullen benoemen: de
oplossingsverzameling van $(H)$ is de \emph{kern} van de lineaire \omterm{def:b1:logic:map}{afbeelding} $y
\mapsto y'' + ay' + by$, en de oplossingsverzameling van $(E)$ is een
affiene verschuiving ervan. De \omterm{def:b1:diffeq:linear2}{karakteristieke veelterm} verschijnt opnieuw als
de \omterm{def:b1:diffeq:linear2}{karakteristieke veelterm} van een matrix in \cref{ch:b1:matrices}:
een vergelijking van de tweede orde is een $2 \times 2$-stelsel van de eerste orde in
vermomming, een gezichtspunt dat het deel van jaar 2 systematiseert. De
integralen die \omterm{thm:b1:diffeq:voc}{variatie van constanten} vereist worden geleverd door
\cref{ch:b1:integration}, en het weekendprobleem hieronder --- de
aangedreven gedempte oscillator --- is het modelgeval voor elke
oscillatievraag in de wetenschappen, van schakelingen tot hangbruggen.
\end{remark}

Een kop koffie op temperatuur $T_0 = 80\,^\circ$C staat in een kamer op

\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:diffeq:9}
(Gedempte oscillator) Voor $\varepsilon \geq 0$, beschouw
$y'' + 2\varepsilon y' + y = 0$.
\begin{enumerate}
  \item Los op voor $\varepsilon \in \intco{0}{1}$, $\varepsilon = 1$,
        en $\varepsilon > 1$.
  \item Toon aan dat voor $\varepsilon > 0$ elke oplossing naar $0$ nadert
        in $+\infty$, en dat voor $\varepsilon = 0$ de niet-nul
        oplossingen dat niet doen.
  \item Voor $\varepsilon \in \intoo{0}{1}$, toon aan dat de nulpunten van een
        niet-nul oplossing regelmatig gespreid liggen, met afstand
        $\frac{\pi}{\sqrt{1 - \varepsilon^2}}$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:diffeq:10}
Vind alle functies $f \colon \R \to \R$, tweemaal afleidbaar, zodat
\[
\forall x, y \in \R, \qquad f(x + y) + f(x - y) = 2 f(x) f(y),
\]
met $f(0) \ne 0$ en $f$ niet constant.
\emph{Tip: houd $y$ vast, differentieer tweemaal naar $x$ in $0$;
toon aan dat $f(0) = 1$ en $f'' = c f$ voor een zekere constante $c$; los dan op
naar het teken van $c$ en controleer welke oplossingen aan de
functievergelijking voldoen.}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:diffeq:11}
(Eulervergelijking) Los $x^2 y'' - x y' + y = 0$ op op
$\intoo{0}{+\infty}$. \emph{Tip: stel $z(t) = y(\eu^t)$, d.w.z.\
substitueer $x = \eu^t$, en toon aan dat $z$ voldoet aan een lineaire
vergelijking met constante coëfficiënten.}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:diffeq:12}
Beschouw de vergelijking $x\,y' = 2y$ op de hele reële rechte, in de
onbekende afleidbare functie $y \colon \R \to \R$.
\begin{enumerate}
  \item Los op op $\intoo{0}{+\infty}$ en op $\intoo{-\infty}{0}$.
  \item Toon aan dat voor \emph{willekeurige} constanten $a, b \in \R$, de
        functie die gelijk is aan $ax^2$ voor $x \geq 0$ en aan $bx^2$ voor
        $x < 0$ afleidbaar is op $\R$ en de vergelijking overal
        oplost.
  \item Concludeer dat de oplossingsverzameling op $\R$ een familie met twee
        parameters is, en leg uit waarom dit niet in strijd is met de
        eenduidigheid in \cref{thm:b1:diffeq:voc}.
\end{enumerate}
\end{exercise}

\section{Probleem: De aangedreven gedempte oscillator}

\begin{problem}\label{pb:b1:diffeq:1}
Eén vergelijking beheerst een massa aan een veer in een viskeus medium, de
lading in een RLC-schakeling, en een gebouw dat wiegt in de wind:
\[
(E_\Omega)\colon\quad
x'' + 2\lambda x' + \omega_0^2\,x = A\cos(\Omega t),
\]
met $\lambda \geq 0$ de demping, $\omega_0 > 0$ de natuurlijke
frequentie, en $A > 0$, $\Omega > 0$ de amplitude en frequentie van
de aandrijving. Dit probleem haalt haar volledige gedrag eruit: de afname
van transiënten, de unieke periodieke steady state, de resonantie-
kromme en haar scherpte (de \emph{kwaliteitsfactor}), de \omterm{pb:b1:diffeq:1}{zwevingen} van
het ongedempte geval, en de energiebalans die de oscillatie
in stand houdt.\index{resonantie}\index{oscillator}
Tenzij anders vermeld, $0 < \lambda < \omega_0$ (ondergedempt
regime) en we schrijven $\omega_d = \sqrt{\omega_0^2 - \lambda^2}$.

\medskip
\noindent\textbf{Deel I --- De vrije oscillator.} Hier is $A = 0$.
\begin{enumerate}
  \item Los de homogene vergelijking $(H)$ op voor $0 < \lambda <
        \omega_0$, en voor $\lambda = 0$. (De regimes $\lambda
        \geq \omega_0$ werden behandeld in \cref{exo:b1:diffeq:9};
        citeer ze.)
  \item Toon aan dat voor elke $\lambda > 0$, alle oplossingen van $(H)$
        naar $0$ naderen in $+\infty$ --- in alle drie de regimes.
  \item Definieer de energie $\mathcal E(t) = \frac12 x'(t)^2 +
        \frac12\omega_0^2\,x(t)^2$ langs een oplossing van $(H)$. Toon aan dat
        $\mathcal E'(t) = -2\lambda\,x'(t)^2 \leq 0$, en leid af
        (zonder iets op te lossen) dat het \omterm{thm:b1:diffeq:voc}{Cauchy-probleem}
        ``$(H)$, $x(t_0) = x'(t_0) = 0$'' alleen de nul-
        oplossing heeft, voor elke $\lambda \geq 0$.
  \item Voor $0 < \lambda < \omega_0$, schrijf de niet-nul oplossing als
        $x(t) = R\,\eu^{-\lambda t}\cos(\omega_d t - \varphi)$ en
        zij $T_d = \frac{2\pi}{\omega_d}$ de pseudoperiode.
        Toon aan dat $x(t + T_d) = \eu^{-\lambda T_d}\,x(t)$: elke
        zwaai is de vorige, verkleind met de constante factor
        $\eu^{-\delta}$, $\delta = \frac{2\pi\lambda}{\omega_d}$
        (het \emph{logaritmische decrement}). Bereken $\delta$ voor
        $\omega_0 = 1$, $\lambda = 0.1$.
  \item Definieer de \emph{kwaliteitsfactor} $Q =
        \dfrac{\omega_0}{2\lambda}$\index{kwaliteitsfactor}. Toon aan
        dat na de tijd $\frac1\lambda$ (één amplitude-
        $\eu$-vouwing), de oscillator
        $\frac{\omega_d}{2\pi\lambda}$ pseudoperioden heeft voltooid,
        wat voor zwakke demping ($\lambda \ll \omega_0$) ongeveer
        $\frac Q\pi$ is: de kwaliteitsfactor telt, op $\pi$ na, de
        oscillaties die overleven voordat de amplitude afneemt met
        $\eu$.
\end{enumerate}

\medskip
\noindent\textbf{Deel II --- De steady state.} Nu $A > 0$ en
$\lambda > 0$.
\begin{enumerate}[resume]
  \item Zoek een particuliere oplossing als het reële deel van
        $z\,\eu^{\iu\Omega t}$ met $z \in \C$
        (\cref{met:b1:diffeq:particular}). Toon aan dat dit werkt
        met
        \[
        z = \frac{A}{\omega_0^2 - \Omega^2 + 2\iu\lambda\Omega} .
        \]
  \item Leid de steady state af in amplitude--fasevorm:
        $x_p(t) = R(\Omega)\cos\bigl(\Omega t -
        \varphi(\Omega)\bigr)$ met
        \[
        R(\Omega) = \frac{A}{\sqrt{(\omega_0^2 - \Omega^2)^2 +
        4\lambda^2\Omega^2}},
        \qquad
        \tan\varphi = \frac{2\lambda\Omega}{\omega_0^2 -
        \Omega^2},
        \quad \varphi \in \intoo0\pi .
        \]
  \item Interpreteer de twee extreme regimes: bereken de limieten van
        $R$ en $\varphi$ als $\Omega \to 0^+$ (quasistatische
        respons $A/\omega_0^2$, fase $0$) en als $\Omega \to
        +\infty$ ($R \sim A/\Omega^2 \to 0$, fase $\to \pi$: de
        massa beweegt tegengesteld aan een te snelle aandrijving).
  \item Toon aan dat \emph{elke} oplossing van $(E_\Omega)$ gelijk is aan $x_p$
        plus een oplossing van $(H)$, dus convergeert naar de steady
        state $x_p$ als $t \to +\infty$, wat de beginvoorwaarden
        ook zijn: nadat de transiënt is gestorven, heeft de oscillator
        geen geheugen van hoe hij begon.
  \item Toon aan dat $x_p$ de \emph{enige} periodieke oplossing van
        $(E_\Omega)$ is.
  \item Werk één \omterm{thm:b1:diffeq:voc}{Cauchy-probleem} tot het einde uit: voor $x'' + 2x' + 2x =
        \cos t$ met $x(0) = x'(0) = 0$, toon aan dat de oplossing
        \[
        x(t) = \frac{\cos t + 2\sin t}5
        - \eu^{-t}\,\frac{\cos t + 3\sin t}5
        \]
        is, en identificeer transiënt en steady deel.
\end{enumerate}

\medskip
\noindent\textbf{Deel III --- De resonantiekromme.} Studie van
$\Omega \mapsto R(\Omega)$ op $\intoo0{+\infty}$.
\begin{enumerate}[resume]
  \item Stel $u = \Omega^2$ en $g(u) = (\omega_0^2 - u)^2 +
        4\lambda^2 u$, toon aan: als $2\lambda^2 < \omega_0^2$, dan bereikt $R$
        een strikt maximum in de \emph{resonantiefrequentie}
        $\Omega_r = \sqrt{\omega_0^2 - 2\lambda^2}$, met
        \[
        R_{\max} = R(\Omega_r)
        = \frac{A}{2\lambda\sqrt{\omega_0^2 - \lambda^2}} .
        \]
  \item Toon aan dat $\dfrac{R_{\max}}{R(0)} = Q\,\bigl(1 -
        \tfrac{\lambda^2}{\omega_0^2}\bigr)^{-1/2} \geq Q$: bij
        \omterm{ex:b1:diffeq:oscillation}{resonantie} wordt de aandrijving versterkt met (in wezen) de
        kwaliteitsfactor.
  \item Bewijs dat de amplitude van de \emph{snelheid} $V(\Omega) =
        \Omega\,R(\Omega)$ maximaal is precies in $\Omega =
        \omega_0$ (niet in $\Omega_r$), en dat de fase daar
        $\varphi(\omega_0) = \frac\pi2$ is: in $\Omega = \omega_0$
        is de snelheid exact in fase met de kracht.
  \item (Bandbreedte) Los $g(u) = 2\,g(u_r)$ exact op, waar $u_r =
        \omega_0^2 - 2\lambda^2$, en leid af dat de twee
        frequenties $\Omega_\pm$ waar $R = R_{\max}/\sqrt2$
        voldoen aan $\Omega_+^2 - \Omega_-^2 =
        4\lambda\sqrt{\omega_0^2 - \lambda^2}$; concludeer dat voor
        zwakke demping de bandbreedte $\Omega_+ - \Omega_-
        \approx 2\lambda$ is, d.w.z.\ $Q \approx
        \frac{\omega_0}{\Omega_+ - \Omega_-}$: scherpe resonantie-
        pieken zijn systemen met hoge $Q$.
  \item Numeriek portret voor $\omega_0 = 1$, $\lambda = 0.05$
        ($Q = 10$), $A = 1$: bereken $\Omega_r$, $R_{\max}$, de
        statische respons $R(0)$, en de benaderende bandbreedte.
  \item Toon aan dat als $2\lambda^2 \geq \omega_0^2$, dan $R$
        strikt dalend is op $\intoo0{+\infty}$: sterk gedempte
        systemen hebben helemaal geen resonantiepiek.
\end{enumerate}

\medskip
\noindent\textbf{Deel IV --- Geen demping: \omterm{pb:b1:diffeq:1}{zwevingen} en \omterm{ex:b1:diffeq:oscillation}{resonantie}.}
Hier is $\lambda = 0$.
\begin{enumerate}[resume]
  \item Voor $\Omega \neq \omega_0$, vind de algemene oplossing van
        $x'' + \omega_0^2 x = A\cos(\Omega t)$.
  \item Los het \omterm{thm:b1:diffeq:voc}{Cauchy-probleem} $x(0) = x'(0) = 0$ op en transformeer
        het antwoord naar de productvorm
        \[
        x(t) = \frac{2A}{\omega_0^2 - \Omega^2}\,
        \sin\Bigl(\frac{(\omega_0 - \Omega)t}2\Bigr)
        \sin\Bigl(\frac{(\omega_0 + \Omega)t}2\Bigr) .
        \]
  \item Voor $\Omega$ dicht bij $\omega_0$, lees het product als een
        snelle oscillatie op frequentie $\frac{\omega_0 + \Omega}2$
        gemoduleerd door een trage omhullende op frequentie
        $\frac{\abs{\omega_0 - \Omega}}2$: de \emph{zwevingen}
        \index{zwevingen}. Geef de periode van de omhullende en de
        maximale amplitude, en merk op hoe beide exploderen als $\Omega
        \to \omega_0$.
  \item Houd $t$ vast en laat $\Omega \to \omega_0$ in de formule van vraag 19:
        toon aan dat de limiet
        \[
        x_\infty(t) = \frac{A\,t\,\sin(\omega_0 t)}{2\omega_0}
        \]
        is, en controleer rechtstreeks dat $x_\infty$ de resonante
        vergelijking $x'' + \omega_0^2 x = A\cos(\omega_0 t)$ met
        $x(0) = x'(0) = 0$ oplost (vergelijk
        \cref{ex:b1:diffeq:oscillation}): \omterm{ex:b1:diffeq:oscillation}{resonantie} is de limiet
        van steeds tragere, steeds grotere \omterm{pb:b1:diffeq:1}{zwevingen}.
  \item Contrasteer de twee lotgevallen van \omterm{ex:b1:diffeq:oscillation}{resonantie}: lineaire groei
        $\frac{At}{2\omega_0}$ zonder demping, tegenover verzadiging
        bij $R_{\max} \approx Q\,\frac{A}{\omega_0^2}$ met zwakke
        demping. In één zin: welk fysisch mechanisme zet het
        eerste om in het tweede?
\end{enumerate}

\medskip
\noindent\textbf{Deel V --- Energiebalans en synthese.}
\begin{enumerate}[resume]
  \item In de steady state van Deel II, bereken het gemiddelde over
        één periode $\frac{2\pi}\Omega$ van (a) het vermogen geïnjecteerd
        door de aandrijving, $P_{\mathrm{in}}(t) = A\cos(\Omega t)
        \cdot x_p'(t)$, en (b) het vermogen gedissipeerd door de
        demping, $P_{\mathrm{diss}}(t) = 2\lambda\,x_p'(t)^2$.
        Toon aan dat beide gemiddelden gelijk zijn aan $\lambda\,R^2\Omega^2$: de
        aandrijving voedt precies in wat de demping verbrandt --- dit
        is waarom de steady state steady is.
  \item Waar precies gebruikte het probleem: (i) de structuur-
        stelling \cref{thm:b1:diffeq:structure2}; (ii) de complexe
        exponentiële methode; (iii) een functiestudie van een reële
        variabele in de stijl van \cref{ch:b1:functions}? Eén zin
        elk.
  \item Synthese: beschrijf de volledige gedragskaart van $(E_\Omega)$
        --- vrij tegenover geforceerd, gedempt tegenover ongedempt, de rol
        van $Q$ als de enige dimensieloze knop die piekhoogte,
        bandbreedte en transiëntlevensduur afstemt --- en vermeld
        waar het verhaal verdergaat: $2 \times 2$-stelsels van de eerste
        orde (\cref{ch:b1:matrices} en het deel van jaar 2) en
        de ontbinding van een algemene periodieke aandrijving in
        sinusoïden (Fourierreeksen, in het deel van jaar 3), waarvoor
        het sinusoïdale geval van dit probleem de fundamentele
        bouwsteen is.
\end{enumerate}
