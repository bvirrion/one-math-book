\chapter{Limieten en continuïteit}\label{ch:b1:continuity}

De tussenwaardestelling en de extremumstelling werden op middelbareschoolniveau
op visueel vertrouwen gebruikt. Met rijen
(\cref{ch:b1:seq}) en de topologie van $\R$ (\cref{ch:b1:topology})
in handen bewijst dit hoofdstuk ze --- en vervolledigt het de theorie met
de stelling van de monotone bijectie (die $\arcsin$,
$\operatorname{arcosh}$ en hun soortgenoten uit \cref{ch:b1:functions}
legitimeert) en de stelling van Heine over \omterm{def:b1:continuity:uniform}{uniforme continuïteit}.

Overal is $I$ een \omterm{prop:b1:reals:intervals}{interval} en $f \colon I \to \R$; ``$x_0 \in
\overline I$'' laat limieten in de eindpunten toe.

\section{Limieten van functies}

\begin{definition}[Limiet in een punt]\label{def:b1:continuity:limit}
Zij $x_0 \in \overline{I}$ en $\ell \in \R$. Dan geldt $f(x) \to \ell$ als
$x \to x_0$ wanneer\index{limiet!van een functie}
\[
\forall \varepsilon > 0,\ \exists \delta > 0,\ \forall x \in I,
\qquad \abs{x - x_0} \leq \delta \implies \abs{f(x) - \ell} \leq
\varepsilon .
\]
Limieten in $\pm\infty$ en oneindige limieten worden volgens hetzelfde
patroon gedefinieerd ($\abs{x - x_0} \leq \delta$ wordt $x \geq M$; $\abs{f(x) -
\ell} \leq \varepsilon$ wordt $f(x) \geq M'$). Eenzijdige limieten
beperken $x$ tot $x > x_0$ (geschreven $x \to x_0^+$) of $x < x_0$. De
limiet is uniek wanneer hij bestaat (zelfde bewijs als
\cref{prop:b1:seq:first}).
\end{definition}

\begin{example}[Een limiet via de insluiting]\label{ex:b1:continuity:floorlimit}
Bereken $\lim_{x \to 0} x\,\bigl\lfloor \frac1x \bigr\rfloor$.
De insluiting door het \omterm{thm:b1:reals:floor}{gehele deel} $\frac1x - 1 < \lfloor \frac1x \rfloor \leq
\frac1x$ geeft, na vermenigvuldiging met $x$ (let op het teken!):
\[
1 - x < x\Bigl\lfloor \frac1x \Bigr\rfloor \leq 1 \quad (x > 0),
\qquad
1 \leq x\Bigl\lfloor \frac1x \Bigr\rfloor < 1 - x \quad (x < 0),
\]
en beide eenzijdige insluitingen sluiten op $1$: de limiet is $1$. Merk op
wat er gebeurde: $\lfloor \frac1x\rfloor$ alleen heeft wilde sprongen nabij
$0$, maar de factor $x$ temt elke sprong ($x$ maal een eenheidssprong is
klein), en alleen de insluiting overleeft. Het afsluitende inzicht:
limieten van producten van een kleine factor met een factor van begrensde
oscillatie zijn insluitingsproblemen, nooit problemen voor de
bewerkingsstelling --- de bewerkingsstelling vereist dat \emph{beide} factoren
convergeren.
\end{example}

\begin{theorem}[Sequentiële karakterisering]\label{thm:b1:continuity:seqchar}
$f(x) \to \ell$ als $x \to x_0$ dan en slechts dan als: voor \emph{elke}
rij $(u_n)$ van punten van $I$ met $u_n \to x_0$ geldt
$f(u_n) \to \ell$.
\end{theorem}

\begin{proof}
($\Rightarrow$) Zij $u_n \to x_0$ en $\varepsilon > 0$. Neem
$\delta$ uit de definitie, dan $N$ met $\abs{u_n - x_0} \leq
\delta$ voor $n \geq N$: vanaf $N$ geldt $\abs{f(u_n) - \ell} \leq
\varepsilon$.

($\Leftarrow$) Contrapositie. Als $f \not\to \ell$: dan verslaat een
$\varepsilon_0 > 0$ elke $\delta$; door $\delta =
\frac{1}{n+1}$ te kiezen ontstaan $u_n \in I$ met $\abs{u_n - x_0} \leq
\frac{1}{n+1}$ en $\abs{f(u_n) - \ell} > \varepsilon_0$. Dan geldt $u_n
\to x_0$ maar $f(u_n) \not\to \ell$.
\end{proof}

\begin{corollary}[Bewerkingen, samenstelling, orde]\label{cor:b1:continuity:operations}
Sommen, producten, quotiënten (limiet in de noemer niet nul) van limieten gedragen
zich zoals bij rijen; als $f \to \ell$ in $x_0$ en $g \to m$ in $\ell$,
en $g$ is gedefinieerd rond $\ell$ met $g(\ell) = m$ of $f \neq \ell$
nabij $x_0$, dan geldt $g \circ f \to m$ in $x_0$; limieten behouden ruime
ongelijkheden, en de insluitstelling geldt.
\end{corollary}

\begin{proof}
Elke \omterm{def:b1:logic:statement}{uitspraak} wordt via
\cref{thm:b1:continuity:seqchar} overgedragen naar haar rijanaloog
(\cref{thm:b1:seq:operations,thm:b1:seq:order}). Voor de samenstelling
is de directe ketening het waard om één keer op te schrijven: gegeven $\varepsilon >
0$ levert de limiet van $g$ in $\ell$ een $\eta > 0$ met
\[
\abs{y - \ell} \leq \eta \implies \abs{g(y) - m} \leq
\varepsilon \quad (y \text{ in het domein van } g),
\]
waarbij het geval $y = \ell$ gedekt is omdat $g(\ell) = m$; vervolgens
levert de limiet van $f$ in $x_0$ een $\delta > 0$ met $\abs{x -
x_0} \leq \delta \implies \abs{f(x) - \ell} \leq \eta$; door ketening
geeft $\abs{x - x_0} \leq \delta$ dat $\abs{g(f(x)) - m} \leq
\varepsilon$. In de alternatieve hypothese ($f \neq \ell$ nabij
$x_0$) wordt de waarde $y = \ell$ nooit aan $g$ toegevoerd en is haar waarde
daar irrelevant.
\end{proof}

\begin{example}[Waarom de voorwaarde bij de samenstelling bestaat]\label{ex:b1:continuity:compositiontrap}
Zij $g(y) = 0$ voor $y \neq 0$ en $g(0) = 1$, en zij $f$
identiek $0$. Dan geldt $f(x) \to 0$ als $x \to 0$, en $g(y) \to 0$
als $y \to 0$; toch is $g(f(x)) = g(0) = 1$ voor elke $x$, dus $g
\circ f \to 1 \neq 0$. De binnenste functie zit \emph{precies op}
de verboden waarde $\ell = 0$ voor altijd, en de limiet van $g$ in $0$
negeert wat $g$ doet in $0$. De voorwaarde van
\cref{cor:b1:continuity:operations} --- ofwel $g(\ell) = m$
(d.w.z.\ $g$ \omterm{def:b1:continuity:continuous}{continu} in $\ell$), ofwel $f \neq \ell$ nabij $x_0$ ---
is precies wat dit uitsluit. Het afsluitende inzicht: in
de praktijk stelt men \emph{continue} functies samen en is de
voorwaarde gratis; ze bijt alleen wanneer limieten worden genomen langs
doorprikte \omterm{def:b1:topology:open}{omgevingen}, en daarom bevat de definitie van
$\lim_{x \to x_0}$ die in dit boek gebruikt wordt het punt zelf wanneer het
in het domein ligt.
\end{example}

\section{Continuïteit}

\begin{definition}\label{def:b1:continuity:continuous}
$f$ is \emph{continu in $x_0 \in I$}\index{continuïteit} wanneer $f(x)
\to f(x_0)$ als $x \to x_0$; \emph{continu op $I$} wanneer het
continu is in elk punt. Volgens
\cref{thm:b1:continuity:seqchar}: $f$ is continu in $x_0$ dan en slechts dan als
$f(u_n) \to f(x_0)$ voor elke rij $u_n \to x_0$ in $I$.
\end{definition}

\begin{example}[Een taxonomie van discontinuïteiten]\label{ex:b1:continuity:taxonomy}
Drie manieren om in een punt te falen, in oplopende ernst.
\emph{Ophefbaar}: $f(x) = \frac{\sin x}{x}$ op $\R^*$ heeft limiet
$1$ in $0$; door $f(0) = 1$ te definiëren wordt het hersteld --- de discontinuïteit
was een gat, geen eigenschap. \emph{Sprong}: $\lfloor x \rfloor$ in
een geheel getal heeft verschillende eenzijdige limieten ($n - 1$ en $n$); geen
keuze van waarde kan ze verzoenen, maar beide halve limieten bestaan.
\emph{Essentieel}: $\sin\frac1x$ in $0$ heeft helemaal geen eenzijdige limiet
(\cref{exo:b1:continuity:1}) --- oscillatie zonder
uitdemping. Monotone functies kunnen alleen het middelste soort produceren
(hun eenzijdige limieten bestaan altijd, als suprema en infima),
en daarom zijn hun \omterm{def:b1:logic:sets}{verzamelingen} van discontinuïteiten hoogstens aftelbaar ---
één rationaal getal per sprong. Afgeleiden kunnen, volgens de stelling van Darboux
(\cref{exo:b1:derivative:10}), alleen het laatste soort produceren:
een functie met een sprongdiscontinuïteit is nooit de afgeleide van
iets.
\end{example}

\begin{proposition}\label{prop:b1:continuity:algebra}
Sommen, producten, quotiënten (waar gedefinieerd) en samenstellingen van
continue functies zijn \omterm{def:b1:continuity:continuous}{continu}. \omterm{def:b1:poly:def}{Veeltermen}, \omterm{def:b1:fractions:field}{rationale breuken}
(buiten hun \omterm{def:b1:fractions:field}{polen}), $\abs{\,\cdot\,}$, $\exp$, $\ln$, de
goniometrische en \omterm{def:b1:functions:hyperbolic}{hyperbolische functies} en hun inversen
(\cref{ch:b1:functions}) zijn \omterm{def:b1:continuity:continuous}{continu} op hun domeinen.
\end{proposition}

\begin{proof}
Bewerkingen: \cref{cor:b1:continuity:operations}. Constanten en
de identiteit zijn rechtstreeks \omterm{def:b1:continuity:continuous}{continu} vanuit de definitie
($\delta = \varepsilon$ volstaat voor de identiteit, elke $\delta$ voor de
constanten); aangezien producten van continue functies \omterm{def:b1:continuity:continuous}{continu} zijn,
volgt elke monoom $a_k x^k$ door inductie op $k$, en sommen
maken de \omterm{def:b1:poly:def}{veeltermen} af; een \omterm{def:b1:fractions:field}{rationale breuk} is een quotiënt van twee
\omterm{def:b1:poly:def}{veeltermen}, \omterm{def:b1:continuity:continuous}{continu} overal waar de noemer niet
nul is. $\abs{\,\cdot\,}$: omgekeerde driehoeksongelijkheid,
$\bigl|\abs{f(x)} - \abs{f(x_0)}\bigr| \leq \abs{f(x) -
f(x_0)}$, dus dezelfde $\delta$ werkt. Voor de klassieke
functies verlenen we hier \omterm{def:b1:continuity:continuous}{continuïteit}; afleidbaarheid (bewezen in
\cref{ch:b1:derivative}) is sterker.
\end{proof}

\begin{example}[Maximum en minimum van continue functies]\label{ex:b1:continuity:maxmin}
Als $f, g$ \omterm{def:b1:continuity:continuous}{continu} zijn, dan zijn $\max(f, g)$ en $\min(f, g)$ dat ook:
geen gevalsonderscheid nodig, dankzij de identiteiten
\[
\max(f, g) = \frac{f + g + \abs{f - g}}{2},
\qquad
\min(f, g) = \frac{f + g - \abs{f - g}}{2},
\]
en de \omterm{def:b1:continuity:continuous}{continuïteit} van sommen en van $\abs{\,\cdot\,}$
(\cref{prop:b1:continuity:algebra}). In het bijzonder zijn $f^+ =
\max(f, 0)$ en $f^- = \max(-f, 0)$ \omterm{def:b1:continuity:continuous}{continu} met $f =
f^+ - f^-$: de tekensplitsing die gebruikt wordt voor reeksen
(\cref{ch:b1:series}) en, op volle schaal, in de integratietheorie
van het deel van het derde jaar, kost niets aan regelmaat.
\end{example}

\begin{theorem}[Tussenwaardestelling]\label{thm:b1:continuity:ivt}
Zij $f$ \omterm{def:b1:continuity:continuous}{continu} op $\intcc{a}{b}$ met $f(a) \leq 0 \leq
f(b)$. Dan geldt $f(c) = 0$ voor een zekere $c \in \intcc{a}{b}$.
Bijgevolg neemt een continue functie op een \omterm{prop:b1:reals:intervals}{interval} elke waarde
tussen twee van haar waarden aan: $f(I)$ is een
\omterm{prop:b1:reals:intervals}{interval}.\index{tussenwaardestelling}
\end{theorem}

\begin{proof}
Dichotomie. Stel $a_0 = a$, $b_0 = b$. Gegeven $\intcc{a_n}{b_n}$ met
$f(a_n) \leq 0 \leq f(b_n)$, zij $m$ het middelpunt: als $f(m) \leq
0$ behoud $\intcc{m}{b_n}$, anders behoud $\intcc{a_n}{m}$; de
tekenvoorwaarden blijven behouden. De rijen $(a_n), (b_n)$ zijn ingesloten ($b_n -
a_n = \frac{b-a}{2^n}$), met gemeenschappelijke limiet $c$
(\cref{thm:b1:seq:adjacent}). Door \omterm{def:b1:continuity:continuous}{continuïteit} en
\cref{thm:b1:seq:order}: $f(c) = \lim f(a_n) \leq 0$ en $f(c) =
\lim f(b_n) \geq 0$, dus $f(c) = 0$.

Voor het gevolg: gegeven waarden $f(u) < v < f(w)$, pas het bovenstaande
toe op $x \mapsto f(x) - v$ op het segment met eindpunten $u$ en $w$;
dus $f(I)$ is convex, d.w.z.\ een \omterm{prop:b1:reals:intervals}{interval}
(\cref{prop:b1:reals:intervals}).
\end{proof}

\begin{example}[Eén vergelijking, het volledige protocol]\label{ex:b1:continuity:protocol}
Los $\eu^x = 3 - x$ op over $\R$: bestaan, uniciteit,
lokalisatie. Stel $g(x) = \eu^x + x - 3$, \omterm{def:b1:continuity:continuous}{continu}. Lokalisatie en
bestaan: $g(0) = -2 < 0$ en $g(1) = \eu - 2 > 0$, dus
plant de tussenwaardestelling een oplossing in
$\intoo{0}{1}$. Uniciteit: $g$ is een som van de strikt
stijgende $\eu^x$ en $x - 3$, dus strikt stijgend op
$\R$; een strikt monotone functie neemt elke waarde hoogstens
één keer aan, dus is de oplossing uniek op heel $\R$ (niet alleen in het
onderzochte \omterm{prop:b1:reals:intervals}{interval}). Het protocol --- herschik naar $g = 0$, teken\-
verandering voor bestaan, monotonie voor uniciteit --- beslecht
de meeste ``hoeveel oplossingen''-vragen in drie regels, en
de variatietabellen van \cref{ch:b1:derivative}
breiden het uit tot niet-monotone $g$ door $\R$ op te delen in monotone
takken.
\end{example}

\begin{example}[Dichotomie als algoritme]\label{ex:b1:continuity:dichotomy}
Het bewijs van \cref{thm:b1:continuity:ivt} rekent. Neem $f(x) =
x^3 + x - 1$: $f(0) = -1 < 0 < 1 = f(1)$, dus ligt een wortel in
$\intoo{0}{1}$. Halveren:
\[
f(0.5) = -0.375 < 0, \qquad
f(0.75) = 0.171875 > 0, \qquad
f(0.625) = -0.130859375 < 0 ,
\]
dus wordt de wortel achtereenvolgens ingesloten in $\intoo{0.5}{1}$, dan
$\intoo{0.5}{0.75}$, dan $\intoo{0.625}{0.75}$ (echte waarde:
$c \approx 0.6823$). Na $n$ stappen is de fout hoogstens
$\frac{b - a}{2^n}$: tien stappen geven drie decimalen, twintig geven
zes. Het afsluitende inzicht: de tussenwaardestelling is niet
alleen een bestaansuitspraak --- haar dichotomiebewijs is een
gegarandeerd, zij het traag, wortelzoekend algoritme, waartegen de
snelle maar lokale methode van Newton uit \cref{ch:b1:derivative}
gemeten moet worden.
\end{example}

\begin{theorem}[Extremumstelling]\label{thm:b1:continuity:evt}
Een continue functie op een segment $\intcc{a}{b}$ is begrensd en
bereikt haar grenzen: er zijn $c, d \in \intcc{a}{b}$ met
\[
f(c) = \inf_{\intcc{a}{b}} f,
\qquad
f(d) = \sup_{\intcc{a}{b}} f .
\]
Gecombineerd met \cref{thm:b1:continuity:ivt}: \emph{het continue
beeld van een segment is een segment} $\intcc{f(c)}{f(d)}$.
\index{extremumstelling}
\end{theorem}

\begin{proof}
\emph{Naar boven begrensd:} anders kies $u_n$ met $f(u_n) \geq n$. Door
compactheid van het segment (\cref{thm:b1:topology:compact}) geldt voor een
\omterm{def:b1:seq:subsequence}{deelrij} $u_{\varphi(n)} \to x \in \intcc{a}{b}$; \omterm{def:b1:continuity:continuous}{continuïteit} geeft
$f(u_{\varphi(n)}) \to f(x)$, maar $f(u_{\varphi(n)}) \geq \varphi(n)
\to +\infty$: tegenspraak.

\emph{\omterm{def:b1:reals:bounds}{Supremum} bereikt:} zij $M = \sup f$ en kies $v_n$ met $f(v_n) >
M - \frac{1}{n+1}$ (\cref{prop:b1:reals:epsilon}). Trek
$v_{\varphi(n)} \to d \in \intcc{a}{b}$ eruit: dan geldt $f(d) = \lim
f(v_{\varphi(n)}) = M$ door de insluiting. Het \omterm{def:b1:reals:bounds}{infimum} wordt behandeld via
$-f$.
\end{proof}

\begin{example}[Positief minimum op een segment]\label{ex:b1:continuity:posmin}
Zij $f$ \omterm{def:b1:continuity:continuous}{continu} op $\intcc{0}{1}$ met $f(x) > 0$ voor
elke $x$. Dan geldt $\inf f > 0$: door de extremumstelling is het
\omterm{def:b1:reals:bounds}{infimum} een \emph{waarde} $f(c)$, en $f(c) > 0$ volgens hypothese.
Een continue positieve functie op een segment blijft dus weg
van $0$ --- een tweeregelig argument dat een dozijn keer gebruikt wordt in de
komende hoofdstukken (noemers onder controle, insluitingen met trapfuncties,
foutafschattingen). Op een niet-compact \omterm{prop:b1:reals:intervals}{interval} faalt dit
spectaculair: $f(x) = x$ op $\intoc{0}{1}$ is \omterm{def:b1:continuity:continuous}{continu} en
positief met $\inf f = 0$, niet bereikt. Het afsluitende inzicht:
``positief'' promoveert tot ``uniform positief'' precies wanneer het
domein compact is; elke hypothese van de extremumstelling
is dragend.
\end{example}

\begin{remark}[Veelvoorkomende valkuilen rond de drie stellingen]
(i) \emph{Continue beelden}: alleen \emph{segmenten} zijn robuust.
Het continue beeld van een \omterm{def:b1:topology:open}{open} \omterm{prop:b1:reals:intervals}{interval} hoeft niet \omterm{def:b1:topology:open}{open} te zijn
($x \mapsto x^2$ beeldt $\intoo{-1}{1}$ af op $\intco{0}{1}$), het
beeld van een \omterm{def:b1:topology:closed}{gesloten verzameling} hoeft niet \omterm{def:b1:topology:closed}{gesloten} te zijn ($\arctan$ beeldt het
\omterm{def:b1:topology:closed}{gesloten} $\R$ af op het \omterm{def:b1:topology:open}{open}
$\intoo{-\frac\pi2}{\frac\pi2}$); maar het beeld van een segment is
een segment (\cref{thm:b1:continuity:evt}). (ii) \emph{De
tussenwaardestelling vereist een \omterm{prop:b1:reals:intervals}{interval}}: de functie $x
\mapsto \frac1x$, \omterm{def:b1:continuity:continuous}{continu} op $\intco{-1}{0} \cup
\intoc{0}{1}$, neemt de waarden $-1$ en $1$ aan maar wordt nooit
nul --- haar domein bestaat uit twee disjuncte stukken, en de waarde
$0$ valt in het gat; benoem altijd het \omterm{prop:b1:reals:intervals}{interval} waarop de
stelling wordt toegepast. (iii) \emph{\omterm{def:b1:continuity:uniform}{Uniforme continuïteit} is een eigenschap van het
paar (functie, \omterm{def:b1:logic:sets}{verzameling})}: $x^2$ is \omterm{def:b1:continuity:uniform}{uniform continu} op elk
segment maar niet op $\R$
(\cref{ex:b1:continuity:notuniform}) --- de woorden ``\omterm{def:b1:continuity:uniform}{uniform
continu}'' zonder een domein zijn betekenisloos. (iv)
\emph{\omterm{def:b1:continuity:continuous}{Continuïteit} van de inverse is niet formeel}: ze geldt op
intervallen via monotonie
(\cref{thm:b1:continuity:bijection}), maar een continue bijectie
tussen verenigingen van intervallen kan een discontinue inverse hebben ---
de opmerking na die stelling staat er omdat studenten ze
citeren zonder de intervalhypothese.
\end{remark}

\section{Monotone functies en inverse functies}

\begin{theorem}[Stelling van de monotone bijectie]\label{thm:b1:continuity:bijection}
Zij $f$ \omterm{def:b1:continuity:continuous}{continu} en strikt monotoon op een \omterm{prop:b1:reals:intervals}{interval} $I$. Dan
\begin{enumerate}
  \item is $f$ een bijectie van $I$ op het \omterm{prop:b1:reals:intervals}{interval} $J = f(I)$;
  \item is de inverse $f^{-1} \colon J \to I$ strikt monotoon (in
        dezelfde richting) en \emph{\omterm{def:b1:continuity:continuous}{continu}}.
\end{enumerate}
\end{theorem}

\begin{proof}
Zeg dat $f$ strikt stijgend is. (1) Injectiviteit volgt onmiddellijk uit
strikte monotonie; surjectiviteit op $f(I)$ is triviaal, en $f(I)$
is een \omterm{prop:b1:reals:intervals}{interval} volgens \cref{thm:b1:continuity:ivt}.

(2) $f^{-1}$ is strikt stijgend: als $y < y'$ in $J$ maar $f^{-1}(y)
\geq f^{-1}(y')$, dan geeft het toepassen van de stijgende $f$ dat $y \geq y'$,
absurd. \omterm{def:b1:continuity:continuous}{Continuïteit} van $f^{-1}$ in $y_0 = f(x_0) \in J$: zij
$\varepsilon > 0$. Veronderstel eerst dat $x_0$ inwendig ligt in $I$, en
verklein $\varepsilon$ zodat $x_0 \pm \varepsilon \in I$: hun
beelden voldoen aan $f(x_0 - \varepsilon) < y_0 < f(x_0 +
\varepsilon)$. Neem
$\delta = \min\bigl(y_0 - f(x_0 - \varepsilon),\, f(x_0 +
\varepsilon) - y_0\bigr) > 0$: voor $\abs{y - y_0} \leq \delta$
sluit de monotonie van $f^{-1}$ de waarde $f^{-1}(y)$ in tussen $x_0 -
\varepsilon$ en $x_0 + \varepsilon$. Als $x_0$ bijvoorbeeld het linker\-
eindpunt van $I$ is, is alleen $x_0 + \varepsilon$ beschikbaar: dan is
$y_0 = \min J$ (monotonie), voldoet elke $y \in J$ met $y - y_0
\leq f(x_0 + \varepsilon) - y_0$ aan $x_0 \leq f^{-1}(y)
\leq x_0 + \varepsilon$, en de eenzijdige afschatting is precies
\omterm{def:b1:continuity:continuous}{continuïteit} in een eindpunt; het rechtereindpunt is symmetrisch.
(Merk op: de \omterm{def:b1:continuity:continuous}{continuïteit} van $f^{-1}$
werd \emph{niet} afgeleid uit die van $f$ door symmetrie --- het is
de monotonie op een \omterm{prop:b1:reals:intervals}{interval} die het werk doet.)
\end{proof}

\begin{remark}
Deze stelling is wat \cref{def:b1:functions:arc} en
\cref{prop:b1:functions:invhyp} stilzwijgend gebruikten: $\arcsin$, $\arccos$,
$\arctan$, $\operatorname{arsinh}$, \dots\ zijn \omterm{def:b1:continuity:continuous}{continu}. Een
aanvulling (\cref{exo:b1:continuity:10}): een continue
\emph{injectieve} functie op een \omterm{prop:b1:reals:intervals}{interval} is automatisch strikt
monotoon, dus kost de monotoniehypothese niets.
\end{remark}

\begin{example}[Wortels van alle ordes]\label{ex:b1:continuity:nthroot}
Voor $n \in \N^*$ is de functie $f(x) = x^n$ \omterm{def:b1:continuity:continuous}{continu} en
strikt stijgend op $\intco{0}{+\infty}$, met $f(0) = 0$ en
$f(x) \to +\infty$: haar beeld is $\intco{0}{+\infty}$
(\cref{thm:b1:continuity:ivt} voor de intervalstructuur). De
stelling van de monotone bijectie levert dan, in één keer, een
strikt stijgende \emph{continue} inverse
\[
x \mapsto x^{1/n} \colon \intco{0}{+\infty} \to
\intco{0}{+\infty} :
\]
bestaan, uniciteit en \omterm{def:b1:continuity:continuous}{continuïteit} van $n$-de wortels, zonder
berekening. Vergelijk met \cref{exo:b1:reals:12}, die
$\sqrt y$ met de hand opbouwde uit het \omterm{def:b1:reals:bounds}{supremum}: één hoofdstuk theorie heeft
die bladzijde werk samengedrukt tot twee regels, en dezelfde twee
regels legitimeerden $\arcsin$, $\arctan$ en
$\operatorname{arsinh}$ ervoor. Het afsluitende inzicht: een goede
stelling is opgeslagen arbeid.
\end{example}

\section{Uniforme continuïteit}

\begin{definition}\label{def:b1:continuity:uniform}
$f \colon I \to \R$ is \emph{uniform continu}\index{uniforme
continuïteit} wanneer
\[
\forall \varepsilon > 0,\ \exists \delta > 0,\ \forall x, y \in I,
\qquad \abs{x - y} \leq \delta \implies \abs{f(x) - f(y)} \leq
\varepsilon .
\]
De kern: $\delta$ hangt alleen af van $\varepsilon$, niet van de
plaats in $I$. Uniforme \omterm{def:b1:continuity:continuous}{continuïteit} impliceert \omterm{def:b1:continuity:continuous}{continuïteit}; een
\emph{lipschitz}functie ($\abs{f(x) - f(y)} \leq k \abs{x - y}$)
is uniform \omterm{def:b1:continuity:continuous}{continu} ($\delta = \varepsilon/k$).
\end{definition}

\begin{example}\label{ex:b1:continuity:notuniform}
$x \mapsto x^2$ is \omterm{def:b1:continuity:continuous}{continu} op $\R$ maar \emph{niet} \omterm{def:b1:continuity:uniform}{uniform
continu}: $\abs{(n + \frac 1n)^2 - n^2} = 2 + \frac{1}{n^2} \geq 2$
hoewel de argumenten op afstand $\frac 1n \to 0$ liggen. Op elk
begrensd \omterm{prop:b1:reals:intervals}{interval} is ze lipschitz, dus \omterm{def:b1:continuity:uniform}{uniform continu} ---
in overeenstemming met de stelling van Heine hieronder.
\end{example}

\begin{example}[Uniforme moduli, expliciet]\label{ex:b1:continuity:moduli}
Op een segment garandeert Heine een uniforme $\delta$; vaak kan men
die ook \emph{berekenen}. Voor $f(x) = x^2$ op $\intcc{0}{10}$:
\[
\abs{x^2 - y^2} = \abs{x + y}\,\abs{x - y} \leq 20\,\abs{x - y},
\]
dus $\delta = \frac{\varepsilon}{20}$ werkt uniform (een
lipschitzmodulus, lineair in $\varepsilon$). Voor $\sqrt x$ op
$\intcc{0}{1}$: $\abs{\sqrt x - \sqrt y} \leq \sqrt{\abs{x -
y}}$ (\cref{exo:b1:continuity:9}), dus $\delta = \varepsilon^2$
werkt --- uniform maar \emph{niet} lineair: nabij $0$ is de
vierkantswortel steil, en de prijs verschijnt in de exponent van
$\varepsilon$, niet in een falen van de uniformiteit. Het afsluitende
inzicht: \omterm{def:b1:continuity:uniform}{uniforme continuïteit} is een spectrum, geen ja/nee ---
de functie $\delta(\varepsilon)$, de \omterm{def:b1:complex:field}{modulus} genoemd, meet
hoe duur de uniformiteit is, en lipschitz is simpelweg haar
beste graad.
\end{example}

\begin{theorem}[Heine]\label{thm:b1:continuity:heine}
Een continue functie op een \emph{segment} $\intcc{a}{b}$ is \omterm{def:b1:continuity:uniform}{uniform
continu}.\index{stelling van Heine}
\end{theorem}

\begin{proof}
Uit het ongerijmde: veronderstel dat een $\varepsilon_0 > 0$ elke
$\delta$ verslaat. Met $\delta = \frac{1}{n+1}$ kies $x_n, y_n \in
\intcc{a}{b}$ met $\abs{x_n - y_n} \leq \frac{1}{n+1}$ en
$\abs{f(x_n) - f(y_n)} > \varepsilon_0$. Compactheid
(\cref{thm:b1:topology:compact}) trekt $x_{\varphi(n)} \to c \in
\intcc{a}{b}$ eruit; dan geldt $y_{\varphi(n)} \to c$ ook (insluiting op $\abs{x -
y}$). \omterm{def:b1:continuity:continuous}{Continuïteit} in $c$ geeft $f(x_{\varphi(n)}) \to f(c)$ en
$f(y_{\varphi(n)}) \to f(c)$, dus $\abs{f(x_{\varphi(n)}) -
f(y_{\varphi(n)})} \to 0 < \varepsilon_0$: tegenspraak.
\end{proof}

\begin{example}[Begrensd, continu, maar toch niet uniform]\label{ex:b1:continuity:sinx2}
De functie $f(x) = \sin(x^2)$ is \omterm{def:b1:continuity:continuous}{continu} en begrensd op
$\R$, maar niet \omterm{def:b1:continuity:uniform}{uniform continu}. Neem
\[
x_n = \sqrt{2\pi n}, \qquad y_n = \sqrt{2\pi n + \tfrac\pi2}:
\qquad
y_n - x_n = \frac{\pi/2}{\sqrt{2\pi n + \frac\pi2} +
\sqrt{2\pi n}} \longrightarrow 0 ,
\]
maar $f(y_n) - f(x_n) = \sin\bigl(2\pi n + \frac\pi2\bigr) -
\sin(2\pi n) = 1 - 0 = 1$ voor elke $n$: geen enkele $\delta$ kan
overal $\varepsilon = \frac12$ bedienen. Meetkundig \emph{versnellen}
de oscillaties van $\sin(x^2)$: de grafiek
voltooit een volledige golf over steeds kortere vensters, dus de
horizontale schaal die een gegeven $\varepsilon$ vereist krimpt naar nul
naarmate $x$ groeit. Het afsluitende inzicht: begrensdheid koopt geen
uniformiteit (dit voorbeeld), en onbegrensdheid sluit ze niet uit
($\sqrt x$, \cref{exo:b1:continuity:9}); wat beslist is de
\emph{oscillatiemodulus}, en de stelling van Heine zegt dat compacte
domeinen hem automatisch disciplineren.
\end{example}

\begin{example}[Het rekenmachine-experiment, verklaard]\label{ex:b1:continuity:cositeration}
Tik een willekeurig getal in een rekenmachine en druk herhaaldelijk op $\cos$:
het scherm zet zich vast op $0.7390851\dots$ Waarom? Na één druk ligt de
waarde in $\intcc{-1}{1}$, na twee in $\intcc{\cos 1}{1}
\subseteq \intcc{0.54}{1}$, een stabiel \omterm{prop:b1:reals:intervals}{interval} voor $\cos$. Daarop geldt
$\abs{\cos a - \cos b} \leq \sin(1)\,\abs{a - b}$ met $\sin 1 =
0.841\dots < 1$ (de product-naar-somgrens van
\cref{pb:b1:seq:1}, of de middelwaardeongelijkheid van
\cref{ch:b1:derivative}): de iteratie is contraherend, dus door de
foutbeheersingsstap van \cref{met:b1:seq:recurrent},
\[
\abs{u_n - c} \leq (0.842)^{\,n-2}\,\abs{u_2 - c}
\longrightarrow 0 ,
\]
waarbij $c$ het unieke vaste punt $\cos c = c$ is
(\cref{exo:b1:continuity:6}). Ongeveer $40$ drukken kopen drie
decimalen ($0.842^{40} \approx 10^{-3}$) --- een meetkundig tempo,
zachter dan de $2^{-n}$ per stap van de dichotomie, maar elke druk kost
één toetsaanslag terwijl elke dichotomiestap een volledige teken\-
evaluatie kost. Het afsluitende inzicht: het vastepuntbeeld van
\cref{ch:b1:seq} en de bestaansstellingen van dit hoofdstuk zijn
twee helften van één verhaal --- de tussenwaardestelling vindt $c$, de contractie bereikt
het.
\end{example}

\begin{remark}[Waar deze stellingen vervolgens werken]
De drie pijlers van dit hoofdstuk voeden elk een latere. De
tussenwaardestelling voedt elk argument voor het bestaan van oplossingen
en de stelling van de monotone bijectie; de extremum\-
stelling maakt van optimalisatieproblemen stellingen (Rolle en de
middelwaardestelling in \cref{ch:b1:derivative} beginnen precies
daar); de stelling van Heine is de reden dat continue functies op
segmenten geïntegreerd kunnen worden in \cref{ch:b1:integration} --- de
uniforme $\delta$ is wat de Riemannsommen doet convergeren. In het deel van het
tweede jaar verschijnt hetzelfde trio opnieuw in genormeerde vectorruimten, met
compactheid die het werk doet dat segmenten hier doen.
\end{remark}

\begin{remark}[Perspectieven binnen dit deel]
\omterm{def:b1:continuity:continuous}{Continuïteit} staat op het punt overtroffen te worden maar nooit met pensioen te gaan.
\Cref{ch:b1:derivative} versterkt haar tot afleidbaarheid en
bewijst een wederdienst (afleidbaar impliceert \omterm{def:b1:continuity:continuous}{continu});
\cref{ch:b1:integration} berust er tweemaal op, via Heine voor
de constructie en via de hoofdstelling, waarvan het
centrale object $x \mapsto \int_a^x f$ een louter
continue $f$ opwaardeert tot een $C^1$-primitieve. In
\cref{ch:b1:multivar} bevat \omterm{def:b1:continuity:continuous}{continuïteit} in twee variabelen een valkuil
die het waard is om vooruit te bekijken: de functie $\frac{xy}{x^2 + y^2}$ (uitgebreid
door $0$) is \omterm{def:b1:continuity:continuous}{continu} in $x$ voor elke vaste $y$ en in $y$ voor
elke vaste $x$, maar toch niet \omterm{def:b1:continuity:continuous}{continu} in de oorsprong --- langs
de diagonaal $x = y$ is ze constant $\frac12$. Afzonderlijke \omterm{def:b1:continuity:continuous}{continuïteit}
is strikt zwakker dan \omterm{def:b1:continuity:continuous}{continuïteit}: de sequentiële
karakterisering overleeft de overgang naar $\R^2$, maar rijen moeten
uit \emph{elke} richting mogen naderen, niet alleen
langs de assen.
\end{remark}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:b1:continuity:1}
Bewijs met de sequentiële karakterisering dat $x \mapsto
\sin\frac 1x$ geen limiet heeft in $0^+$ \emph{(geef twee rijen)}.
Heeft $x \mapsto x \sin\frac 1x$ er een?
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:continuity:2}
Bestudeer de \omterm{def:b1:continuity:continuous}{continuïteit} op $\R$ van $f(x) = \lfloor x \rfloor$ en van
$g(x) = x - \lfloor x \rfloor$, en van $h(x) = \lfloor x \rfloor +
(x - \lfloor x\rfloor)^2$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:continuity:3}
Bewijs dat de vergelijking $x^5 - 3x + 1 = 0$ minstens drie reële
oplossingen heeft \emph{(evalueer in goedgekozen punten en pas
\cref{thm:b1:continuity:ivt} toe op drie disjuncte segmenten)}.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:continuity:4}
Bewijs dat elke \omterm{def:b1:poly:def}{veelterm} van oneven graad een reële wortel heeft.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:continuity:5}
(Vast punt) Zij $f \colon \intcc{0}{1} \to \intcc{0}{1}$
\omterm{def:b1:continuity:continuous}{continu}. Bewijs dat $f$ een vast punt heeft: $f(c) = c$ voor een zekere
$c$. Illustreer dat noch de \omterm{def:b1:continuity:continuous}{continuïteit} noch het segment
weggelaten kan worden.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:continuity:6}
Bewijs dat de vergelijking $\cos x = x$ precies één reële oplossing heeft,
en dat deze in $\intoo{0}{1}$ ligt.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:continuity:7}
Zij $f \colon \R \to \R$ \omterm{def:b1:continuity:continuous}{continu} met $f(x) \to +\infty$ als $x
\to \pm\infty$. Bewijs dat $f$ een globaal minimum bereikt op $\R$.
\emph{(Herleid tot een segment dat een subniveauverzameling bevat.)}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:continuity:8}
Zij $f \colon \R \to \R$ \omterm{def:b1:continuity:continuous}{continu} en periodiek (met periode $T >
0$). Bewijs dat $f$ begrensd is en haar grenzen bereikt, en dat er
een $c$ bestaat met $f(c + \frac T2) = f(c)$. \emph{(Bestudeer voor het tweede
punt $g(x) = f(x + \frac T2) - f(x)$ over één periode.)}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:continuity:9}
Bewijs dat $x \mapsto \sqrt x$ \omterm{def:b1:continuity:uniform}{uniform continu} is op
$\intco{0}{+\infty}$, hoewel het nabij $0$ niet lipschitz is.
\emph{(Bewijs en gebruik $\abs{\sqrt x - \sqrt y} \leq \sqrt{\abs{x -
y}}$.)}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:continuity:10}
Zij $f$ \omterm{def:b1:continuity:continuous}{continu} en \omterm{def:b1:logic:inj}{injectief} op een \omterm{prop:b1:reals:intervals}{interval} $I$. Bewijs dat
$f$ strikt monotoon is. \emph{Aanwijzing: zo niet, dan zijn er $a < b < c$
met bijvoorbeeld $f(b) > f(a)$ en $f(b) > f(c)$; pas de tussenwaarde\-
stelling toe op een waarde tussen $\max(f(a), f(c))$ en $f(b)$ aan
beide zijden van $b$.}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:continuity:11}
(Functievergelijking van Cauchy, continue geval) Zij $f \colon \R \to
\R$ \omterm{def:b1:continuity:continuous}{continu} met $f(x + y) = f(x) + f(y)$ voor alle $x, y$.
Bewijs dat $f(x) = f(1)\,x$ voor alle $x$: eerst op $\N$, $\Z$, $\Q$
(alleen met additiviteit), dan op $\R$ door \omterm{def:b1:continuity:continuous}{continuïteit} en \omterm{def:b1:topology:dense}{dichtheid}
(\cref{thm:b1:reals:density}).
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:continuity:12}
Zij $f \colon \intco{0}{+\infty} \to \R$ \omterm{def:b1:continuity:continuous}{continu} met
$f(x) \to \ell \in \R$ als $x \to +\infty$. Bewijs dat $f$
\emph{uniform} \omterm{def:b1:continuity:continuous}{continu} is op $\intco{0}{+\infty}$. \emph{(Knip bij een
grote $A$: Heine op $\intcc{0}{A+1}$, de limiet voorbij $A$; laat de
twee regimes overlappen.)}
\end{exercise}

\section{Probleem: de functievergelijking van Cauchy en haar zusters}

\begin{problem}[{Weekendprobleem --- $f(x + y) = f(x) + f(y)$:
regelmaat dwingt lineariteit af, en het portret van de monsters}]
\label{pb:b1:continuity:1}
Welke functies voldoen aan $f(x + y) = f(x) + f(y)$ voor alle reële $x,
y$? Cauchy stelde de vraag in 1821; het antwoord is een paradigma.
\Cref{exo:b1:continuity:11} toont dat zo'n \emph{additieve}
functie lineair is op $\Q$ en dat volledige \omterm{def:b1:continuity:continuous}{continuïteit} $f(x)
= cx$ afdwingt. Dit probleem verscherpt de hypothese dramatisch ---
\omterm{def:b1:continuity:continuous}{continuïteit} in een \emph{enkel} punt, of monotonie, of louter
begrensdheid op één klein \omterm{prop:b1:reals:intervals}{interval}, elk volstaat --- en schildert dan
het portret van een hypothetische niet-lineaire oplossing (haar grafiek
vult het vlak), lost de zusterverwante vergelijkingen op die
$\eu^{cx}$, $c\ln x$, $x^c$ en $cx^2$ karakteriseren, en sluit af met de vergelijking van Jensen
en de stelling \emph{middelpuntsconvex $+$ \omterm{def:b1:continuity:continuous}{continu}
$\implies$ convex}.\index{functievergelijking van
Cauchy}\index{functievergelijking} Overal betekent \emph{additief}:
$f(x + y) = f(x) + f(y)$ voor alle $x, y \in \R$.

\medskip
\noindent\textbf{Deel I --- $\Q$-lineariteit, en één punt van
\omterm{def:b1:continuity:continuous}{continuïteit}.}
\begin{enumerate}
  \item Zij $f$ additief. Uit \cref{exo:b1:continuity:11} geldt
        $f(r) = f(1)\,r$ voor rationale $r$. Bewijs de fijnere
        \omterm{def:b1:logic:statement}{uitspraak} die hieronder gebruikt wordt: voor \emph{elke} $x \in \R$ en $r
        \in \Q$ geldt $f(rx) = r\,f(x)$ ($f$ is
        $\Q$-\emph{lineair}).
  \item Veronderstel dat de additieve $f$ \omterm{def:b1:continuity:continuous}{continu} is in één enkel
        punt $x_0$. Toon aan dat $f$ overal \omterm{def:b1:continuity:continuous}{continu} is
        \emph{(bereken $f(x + h) - f(x)$ in termen van $f(x_0 + h)
        - f(x_0)$)}, dus $f(x) = f(1)\,x$.
  \item Toon aan dat een additieve functie bepaald wordt door haar
        beperking tot elke dichte \omterm{def:b1:structures:subgroup}{deelgroep}: als twee additieve
        functies overeenstemmen op $\Z + \sqrt2\,\Z$ en beide
        \omterm{def:b1:continuity:continuous}{continu} zijn, dan zijn ze gelijk --- terwijl zonder \omterm{def:b1:continuity:continuous}{continuïteit}
        het voorschrijven van $f(1) = 0$ en $f(\sqrt 2) = 1$ verenigbaar
        is met $\Q$-lineariteit op de \omterm{def:b1:structures:subgroup}{deelgroep}. Bereken $f(m +
        n\sqrt2)$ voor dit voorschrift.
  \item Zij $f$ additief en naar boven begrensd door $M$ op een zeker
        \omterm{prop:b1:reals:intervals}{interval} $\intcc{a}{b}$ met $a < b$. Toon aan dat $f$
        naar boven begrensd is op $\intcc{0}{\ell}$, $\ell = b - a$
        \emph{(transleer met $a$)}.
\end{enumerate}

\medskip
\noindent\textbf{Deel II --- De regelmaatladder.}
\begin{enumerate}[resume]
  \item Vervolg op vraag 4: gebruik $f(\ell - t) + f(t) =
        f(\ell)$ om aan te tonen dat $f$ ook naar \emph{beneden} begrensd is op
        $\intcc{0}{\ell}$: $\abs f \leq C$ daar.
  \item Toon aan dat $\abs{f(t)} \leq \frac{C}{n}$ voor $t \in
        \intcc{0}{\ell/n}$, en leid af dat $f$ \omterm{def:b1:continuity:continuous}{continu} is in
        $0$ \emph{(oneven zijn regelt de linkerkant)}, dus
        overal (vraag 2): een additieve functie die begrensd is op
        één \omterm{prop:b1:reals:intervals}{interval} is lineair.
  \item Leid het monotone geval af: een additieve functie
        niet-dalend op een zeker $\intcc{a}{b}$ ($a < b$) is $f(x)
        = cx$ met $c \geq 0$.
  \item Stel de \emph{regelmaatladder} samen: voor een additieve
        $f$ zijn de volgende gelijkwaardig --- (a) $f(x) = cx$;
        (b) $f$ \omterm{def:b1:continuity:continuous}{continu}; (c) $f$ \omterm{def:b1:continuity:continuous}{continu} in één punt; (d)
        $f$ monotoon op een zeker niet-ontaard \omterm{prop:b1:reals:intervals}{interval}; (e) $f$
        begrensd op een zeker niet-ontaard \omterm{prop:b1:reals:intervals}{interval}. Rangschik de
        implicaties zo dat elk ofwel triviaal ofwel reeds
        bewezen is.
  \item Ga na dat de vragen 4--6 alleen een grens naar
        \emph{boven} verbruikten: een additieve functie die naar boven begrensd is op één
        niet-ontaard \omterm{prop:b1:reals:intervals}{interval} is al lineair. Leid de
        spiegeluitspraak voor een grens naar beneden af, en noteer de
        sterkste vorm van sport (e) van de ladder die zo verkregen wordt.
\end{enumerate}

\medskip
\noindent\textbf{Deel III --- Portret van een monster.} Veronderstel
nu dat $f$ additief maar \emph{niet} lineair is.
\begin{enumerate}[resume]
  \item Toon aan dat er van nul verschillende reële getallen $u, v$ zijn met
        $\dfrac{f(u)}{u} \neq \dfrac{f(v)}{v}$, en dat de
        vectoren $(u, f(u))$ en $(v, f(v))$ het vlak opspannen (hun
        determinant $u f(v) - v f(u)$ is van nul verschillend).
  \item Toon aan dat de grafiek van $f$ alle punten bevat
        \[
        r\,(u, f(u)) + s\,(v, f(v)), \qquad r, s \in \Q ,
        \]
        en leid af dat de grafiek \emph{\omterm{def:b1:topology:dense}{dicht} is in $\R^2$}: voor
        elk punt $(x_0, y_0)$ van het vlak en elke
        $\varepsilon > 0$ ligt een zeker $(x, f(x))$ binnen
        $\varepsilon$ ervan \emph{(los het reële $2\times2$
        stelsel op, en benader dan de reële coëfficiënten met
        rationale getallen)}.
  \item Leid uit vraag 11 het volledige portret af: een niet-lineaire
        additieve functie is onbegrensd op elk niet-ontaard
        \omterm{prop:b1:reals:intervals}{interval}, discontinu in elk punt, op geen enkel
        \omterm{prop:b1:reals:intervals}{interval} monotoon, en haar beeld van elk \omterm{prop:b1:reals:intervals}{interval} is \omterm{def:b1:topology:dense}{dicht} in
        $\R$. Verzoen dit met vraag 8.
  \item Monsters bestaan --- op een dichte \omterm{def:b1:structures:subgroup}{deelgroep}, constructief:
        definieer op $G = \Z + \sqrt2\,\Z$ de functie $f(m + n\sqrt2) = n$.
        Toon aan dat $f$ welgedefinieerd en additief is op $G$, en dat
        $f$ onbegrensd is op $G \cap \intoo{0}{\varepsilon}$ voor
        elke $\varepsilon > 0$ \emph{(voor vaste $N$ hebben slechts
        eindig veel $g = m + n\sqrt2 \in \intoo{0}{1}$ dat
        $\abs n \leq N$; toch is $G \cap \intoo{0}{\varepsilon}$
        oneindig)}. Leg in één alinea uit waarom het uitbreiden van zo'n
        $f$ tot heel $\R$ een basis van $\R$ als
        $\Q$-vectorruimte vereist (een \emph{Hamelbasis}), waarvan de
        existentie een keuzeaxioma-kwestie is die buiten dit
        deel valt.
\end{enumerate}

\medskip
\noindent\textbf{Deel IV --- De zusterverwante vergelijkingen.} Alle
functies hier zijn \omterm{def:b1:continuity:continuous}{continu}.
\begin{enumerate}[resume]
  \item Zij $f \colon \R \to \R$ \omterm{def:b1:continuity:continuous}{continu}, niet identiek
        $0$, met $f(x + y) = f(x)f(y)$. Toon aan dat $f(x) =
        f\bigl(\frac x2\bigr)^2 \geq 0$, dan $f > 0$ overal,
        dan $f(x) = \eu^{cx}$ voor een zekere $c$: de exponentiële functies
        zijn precies de continue \omterm{def:b1:structures:morphism}{morfismen} van $(\R, +)$ naar
        $(\R^*, \times)$.
  \item Zij $f \colon \intoo{0}{+\infty} \to \R$ \omterm{def:b1:continuity:continuous}{continu}
        met $f(xy) = f(x) + f(y)$. Toon aan dat $f(x) = c\ln x$
        \emph{(transporteer via $\exp$)}.
  \item Zij $f \colon \intoo{0}{+\infty} \to \intoo{0}{+\infty}$
        \omterm{def:b1:continuity:continuous}{continu} met $f(xy) = f(x)f(y)$. Toon aan dat $f(x) = x^c$.
  \item Vind alle continue $f \colon \R \to \R$ met
        \[
        f(x + y) = f(x) + f(y) + f(x)f(y) .
        \]
        \emph{(Bestudeer $h = 1 + f$; behandel het ontaarde geval
        apart.)}
  \item (Parallellogram) Vind alle continue $f \colon \R \to \R$
        met $f(x + y) + f(x - y) = 2f(x) + 2f(y)$: toon aan dat $f$
        even is, $f(0) = 0$, $f(nx) = n^2 f(x)$ door inductie, dan
        $f(x) = f(1)\,x^2$. (Deze vergelijking is de vingerafdruk van
        kwadratische vormen --- de parallellogramwet die, in het deel van het
        tweede jaar, detecteert welke normen van een inwendig
        product komen.)
\end{enumerate}

\medskip
\noindent\textbf{Deel V --- Jensen en middelpuntsconvexiteit.}
\begin{enumerate}[resume]
  \item (Vergelijking van Jensen) Zij $f \colon \R \to \R$
        \omterm{def:b1:continuity:continuous}{continu} met $f\bigl(\frac{x+y}{2}\bigr) = \frac{f(x)
        + f(y)}{2}$. Toon aan dat $g = f - f(0)$ voldoet aan
        $g\bigl(\frac x2\bigr) = \frac{g(x)}{2}$, leid af dat
        $g$ additief is, en besluit $f(x) = cx + d$.
  \item Veronderstel nu alleen de \emph{ongelijkheid}: $f$ \omterm{def:b1:continuity:continuous}{continu}
        met
        \[
        f\Bigl(\frac{x + y}{2}\Bigr) \leq \frac{f(x) +
        f(y)}{2} \qquad (x, y \in \R).
        \]
        Bewijs door inductie op $n$ dat voor alle dyadische gewichten
        $\lambda = \frac{k}{2^n} \in \intcc{0}{1}$:
        \[
        f\bigl(\lambda x + (1 - \lambda)y\bigr) \leq \lambda
        f(x) + (1 - \lambda) f(y) .
        \]
  \item Breid uit door \omterm{def:b1:continuity:continuous}{continuïteit} en \omterm{def:b1:topology:dense}{dichtheid} van de dyadische getallen
        (\cref{exo:b1:reals:8}) tot elke $\lambda \in
        \intcc{0}{1}$: een continue middelpuntsconvexe functie
        voldoet aan de volledige convexiteitsongelijkheid (het begrip
        dat systematisch bestudeerd wordt in \cref{ch:b1:derivative}).
  \item Toon aan dat \omterm{def:b1:continuity:continuous}{continuïteit} niet weggelaten kan worden: een niet-lineaire additieve
        $f$ voldoet aan de \emph{middelpuntsgelijkheid} van vraag
        19 maar aan geen enkele convexiteitsongelijkheid op enig \omterm{prop:b1:reals:intervals}{interval} (vraag
        12). Moraal: middelpuntsconvexiteit is een aftelbare-fase\-
        eigenschap (dyadische getallen), convexiteit een continue;
        \omterm{def:b1:continuity:continuous}{continuïteit} is de brug --- precies als in de Delen I--II.
\end{enumerate}

\medskip
\noindent\textbf{Deel VI --- Laatste varianten en synthese.}
\begin{enumerate}[resume]
  \item Vind alle continue $f \colon \R \to \R$ met $f(x + y)
        = f(x) + f(y) + xy$ \emph{(trek de particuliere
        oplossing $\frac{x^2}{2}$ af)}.
  \item Bewijs: als $f \colon \R \to \R$ \omterm{def:b1:continuity:continuous}{continu} is en
        alleen additief op een \emph{dichte \omterm{def:b1:structures:subgroup}{deelgroep}} $G$ (d.w.z.\
        $f(g + g') = f(g) + f(g')$ voor $g, g' \in G$), dan is $f$
        additief op $\R$. Algemener zijn twee continue
        functies die overeenstemmen op een dichte deelverzameling van $\R$
        gelijk.
  \item Synthese, telkens één zin: (i) formuleer de regelmaat\-
        ladder van vraag 8 uit het hoofd; (ii) leg uit waarom
        ``grafiek \omterm{def:b1:topology:dense}{dicht} in het vlak'' het juiste mentale beeld
        is voor het falen van regelmaat; (iii) som de vijf
        klassieke functies op die gekarakteriseerd worden in Deel IV--V en de
        enkele methode die ze allemaal ving; (iv) noem de twee
        plaatsen waar \omterm{def:b1:topology:dense}{dichtheid} van $\Q$ (of van de dyadische getallen) in $\R$
        het argument droeg, en de plaats waar dat niet kon
        (vraag 13).
\end{enumerate}
\end{problem}
