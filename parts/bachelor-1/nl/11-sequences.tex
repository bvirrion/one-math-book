\chapter{Rijen}\label{ch:b1:seq}

Rijen werden in het deel Secundair onderwijs behandeld met het
limietbegrip half op vertrouwen aangenomen. Hier wordt de theorie
heropgebouwd op de volledigheid van $\R$ (\cref{ch:b1:reals}): elke
klassieke stelling --- monotone convergentie, \omterm{thm:b1:seq:adjacent}{ingesloten rijen},
Bolzano--Weierstrass, het Cauchycriterium --- is een gezicht van dat
ene axioma. Het hoofdstuk eindigt met de praktische studie van rijen
gedefinieerd door $u_{n+1} = f(u_n)$.

\section{Convergentie}

\begin{definition}[Limiet van een rij]\label{def:b1:seq:limit}
Een rij $(u_n)$ van reële getallen \emph{convergeert} naar $\ell \in
\R$ wanneer\index{limiet!van een rij}
\[
\forall \varepsilon > 0,\ \exists N \in \N,\ \forall n \geq N,
\qquad \abs{u_n - \ell} \leq \varepsilon .
\]
Men schrijft $u_n \to \ell$ of $\lim u_n = \ell$. Een rij die niet
convergeert (naar enig reëel getal) \emph{divergeert}. Divergentie
\emph{naar $+\infty$}: $\forall M,\ \exists N,\ \forall n \geq N,\ u_n
\geq M$ (analoog $-\infty$).
\end{definition}

\begin{example}[Een $\varepsilon$--$N$-bewijs, één keer uitgeschreven]\label{ex:b1:seq:epsilonN}
Bewering: $u_n = \dfrac{n^2 + 1}{2n^2 - 3} \to \dfrac12$. Isoleer
eerst de fout:
\[
\Bigl| u_n - \frac12 \Bigr|
= \Bigl| \frac{2(n^2 + 1) - (2n^2 - 3)}{2(2n^2 - 3)} \Bigr|
= \frac{5}{2\,\abs{2n^2 - 3}}
= \frac{5}{2\,(2n^2 - 3)} \quad (n \geq 2).
\]
Domineer ze dan met iets eenvoudigs: voor $n \geq 2$ is $2n^2 - 3
\geq n^2$, dus de fout is $\leq \frac{5}{2n^2} \leq \frac 5{2n}$.
Gegeven $\varepsilon > 0$ levert de \omterm{thm:b1:reals:archimedes}{archimedische eigenschap} een $N
\geq \max\bigl(2, \frac{5}{2\varepsilon}\bigr)$; voor $n \geq N$ is de
fout $\leq \varepsilon$. Klaar. Het afsluitende inzicht: een
$\varepsilon$--$N$-bewijs heeft precies drie stappen --- bereken de
fout, begrens ze door een dalende elementaire uitdrukking, los op naar
de drempel --- en na de stellingen van dit hoofdstuk (bewerkingen,
insluiting) schrijft men zo'n bewijs bijna nooit meer: de stellingen
verpakken de drie stappen eens en voor altijd.
\end{example}

\begin{example}[Divergentie naar oneindig, gecertificeerd]\label{ex:b1:seq:toinfinity}
Bewering: $u_n = n^2 - 100n \to +\infty$. Ontbind de dominante term:
$u_n = n^2\bigl(1 - \frac{100}{n}\bigr) \geq \frac{n^2}{2}$ voor $n
\geq 200$. Gegeven $M$, neem $N = \max\bigl(200,
\lceil\sqrt{2M}\rceil\bigr)$: voor $n \geq N$ is $u_n \geq
\frac{n^2}{2} \geq M$. Twee gewoontes zijn hier zichtbaar: het
ontbinden van de dominante term zet een wedstrijd ($n^2$ tegen
$-100n$) om in één schaal maal een factor die naar $1$ gaat; en de
drempel mag enorm zijn ($u_{100} = 0$, de rij is zelfs negatief vóór
$n = 100$) --- divergentie naar $+\infty$ is een \omterm{def:b1:logic:statement}{uitspraak} over de
staart, ongevoelig voor eender welke eindige hoeveelheid wangedrag.
\end{example}

\begin{proposition}[Eerste eigenschappen]\label{prop:b1:seq:first}
\begin{enumerate}
  \item De limiet is, indien hij bestaat, uniek.
  \item Een convergente rij is begrensd.
  \item Als $u_n \to \ell$, dan laat elke wijziging van eindig veel
        termen de convergentie en de limiet ongewijzigd.
\end{enumerate}
\end{proposition}

\begin{proof}
(1) Als $u_n \to \ell$ en $u_n \to \ell'$ met $\ell \neq \ell'$, neem
$\varepsilon = \frac{\abs{\ell - \ell'}}{3}$: voorbij de twee drempels
geldt $\abs{\ell - \ell'} \leq \abs{\ell - u_n} + \abs{u_n - \ell'}
\leq 2\varepsilon = \frac23 \abs{\ell - \ell'}$, absurd.

(2) Met $\varepsilon = 1$: voorbij $N$ is $\abs{u_n} \leq \abs\ell +
1$; de eindig vele eerdere termen zijn eveneens begrensd, dus
$\abs{u_n} \leq \max(\abs{u_0}, \dots, \abs{u_{N-1}}, \abs\ell + 1)$.

(3) In detail: stel $v_n = u_n$ voor $n \geq n_0$ en $u_n \to \ell$.
Gegeven $\varepsilon > 0$, neem de drempel $N$ voor $(u_n)$: voor $n
\geq \max(N, n_0)$ is $\abs{v_n - \ell} = \abs{u_n - \ell} \leq
\varepsilon$. Dus $v_n \to \ell$: de definitie kwantificeert enkel
over $n \geq N$, en elk eindig beginstuk kan overschreven worden ten
koste van een vergroting van de drempel. (Daarom volstaan
hypothesen ``voor alle grote $n$'' overal in dit hoofdstuk.)
\end{proof}

\begin{theorem}[Bewerkingen met limieten]\label{thm:b1:seq:operations}
Als $u_n \to \ell$ en $v_n \to m$, dan
\[
u_n + v_n \to \ell + m, \qquad u_n v_n \to \ell m, \qquad
\frac{u_n}{v_n} \to \frac{\ell}{m} \ (\text{als } m \neq 0),
\qquad \abs{u_n} \to \abs\ell .
\]
\end{theorem}

\begin{proof}
\emph{Som:} $\abs{(u_n + v_n) - (\ell + m)} \leq \abs{u_n - \ell} +
\abs{v_n - m} \leq 2\varepsilon$ voorbij de grootste drempel.
\emph{Product:} schrijf
\[
u_n v_n - \ell m = (u_n - \ell)\,v_n + \ell\,(v_n - m);
\]
$(v_n)$ is begrensd door een zekere $B$ (\cref{prop:b1:seq:first}),
dus het rechterlid is $\leq B\abs{u_n - \ell} + \abs{\ell}\,\abs{v_n -
m}$, willekeurig klein. \emph{Quotiënt:} het volstaat $\frac 1{v_n}$
te behandelen. Met $\varepsilon = \frac{\abs m}{2}$: voorbij een
zekere $N_0$ is $\abs{v_n} \geq \frac{\abs m}{2}$, dus
\[
\Bigl| \frac{1}{v_n} - \frac 1m \Bigr|
= \frac{\abs{m - v_n}}{\abs{v_n m}}
\leq \frac{2}{m^2}\,\abs{v_n - m} \longrightarrow 0 .
\]
\emph{Absolute waarde:} $\bigl|\abs{u_n} - \abs\ell\bigr| \leq
\abs{u_n - \ell}$ (omgekeerde driehoeksongelijkheid,
\cref{prop:b1:complex:rules}).
\end{proof}

\begin{example}[Bewerkingen plus één algebraïsche truc]\label{ex:b1:seq:conjugatetrick}
Bereken $\lim\,\bigl(\sqrt{n^2 + n} - n\bigr)$. De twee stukken gaan
afzonderlijk naar $+\infty$: de bewerkingsstelling zegt niets over hun
verschil (een \emph{onbepaalde vorm}). Vermenigvuldig met de
toegevoegde:
\[
\sqrt{n^2 + n} - n
= \frac{(n^2 + n) - n^2}{\sqrt{n^2 + n} + n}
= \frac{n}{\sqrt{n^2+n} + n}
= \frac{1}{\sqrt{1 + \frac1n} + 1} .
\]
Nu convergeert alles: $\sqrt{1 + \frac1n} \to 1$, want $0 \leq
\sqrt{1 + h} - 1 = \frac{h}{\sqrt{1+h} + 1} \leq h$ (opnieuw de
toegevoegde, dan insluiting met $h = \frac1n$); vervolgens geeft de
bewerkingsstelling de limiet $\frac{1}{1 + 1} = \frac12$. Het
afsluitende inzicht: de bewerkingsstelling is geen rekenmachine voor
alle limieten --- onbepaalde vormen ($\infty - \infty$, $\frac00$, $0
\times \infty$, $1^\infty$) moeten eerst \emph{omgevormd} worden door
algebra (toegevoegden, ontbinden van de dominante term) tot elk stuk
convergeert; de systematische machine voor hardnekkige gevallen is de
asymptotische ontwikkeling van \cref{ch:b1:taylor}.
\end{example}

\begin{theorem}[Limieten en orde]\label{thm:b1:seq:order}
\begin{enumerate}
  \item Als $u_n \leq v_n$ voor alle grote $n$, en beide convergeren,
        dan $\lim u_n \leq \lim v_n$. (Strikte ongelijkheden gaan
        \emph{niet} over op de limiet: $\frac 1n > 0$ maar $\lim =
        0$.)
  \item (Insluitstelling\index{insluitstelling}) Als $u_n \leq w_n
        \leq v_n$ voor alle grote $n$ en $u_n, v_n \to \ell$, dan
        $w_n \to \ell$.
  \item Als $u_n \to \ell > 0$, dan $u_n > \frac\ell2 > 0$ voor alle
        grote $n$.
\end{enumerate}
\end{theorem}

\begin{proof}
(1) Stel $\ell = \lim u_n > m = \lim v_n$; met $\varepsilon =
\frac{\ell - m}{3}$ voldoen grote termen aan $v_n \leq m + \varepsilon
< \ell - \varepsilon \leq u_n$, in tegenspraak met $u_n \leq v_n$.

(2) Voorbij de drempels: $\ell - \varepsilon \leq u_n \leq w_n \leq
v_n \leq \ell + \varepsilon$.

(3) is \cref{def:b1:seq:limit} met $\varepsilon = \frac\ell2$.
\end{proof}

\begin{example}[Twee insluitingen]\label{ex:b1:seq:squeezes}
(i) $\dfrac{\sin n}{n} \to 0$: uit $-\frac1n \leq \frac{\sin n}{n}
\leq \frac1n$, waarbij beide wanden op $0$ instorten --- zonder ook
maar iets van de grillige teller te hoeven begrijpen. (ii) $(2^n +
3^n)^{1/n} \to 3$: omkader de binnenkant,
\[
3^n \leq 2^n + 3^n \leq 2\cdot3^n
\quad\Longrightarrow\quad
3 \leq (2^n + 3^n)^{1/n} \leq 3\cdot 2^{1/n} ,
\]
en $2^{1/n} = \eu^{\frac{\ln 2}{n}} \to 1$ (zoals voor $5^{1/n}$ in
\cref{exo:b1:seq:2}): de insluiting levert $3$. Het afsluitende
inzicht: een som van concurrerende exponentiëlen gedraagt zich als
haar \emph{grootste} term --- de kleinere worden opgeslokt door een
onschadelijke constante factor, die de $n$-de wortel vervolgens
uitwist.
\end{example}

\section{Monotone rijen}

\begin{theorem}[Stelling van de monotone limiet]\label{thm:b1:seq:monotone}
Een stijgende, naar boven begrensde rij convergeert, naar $\sup\{u_n :
n \in \N\}$; een stijgende, niet naar boven begrensde rij divergeert
naar $+\infty$. (Spiegelbeeldige uitspraken voor dalende
rijen.)\index{stelling van de monotone limiet}
\end{theorem}

\begin{proof}
Zij $s = \sup\{u_n\}$ (\cref{thm:b1:reals:sup}). Gegeven $\varepsilon
> 0$ levert de $\varepsilon$-karakterisering
(\cref{prop:b1:reals:epsilon}) een $N$ met $u_N > s - \varepsilon$;
door monotonie geldt $s - \varepsilon < u_N \leq u_n \leq s$ voor alle
$n \geq N$: convergentie naar $s$. Indien onbegrensd: voor elke $M$ is
er een $u_N > M$, en monotonie houdt alle latere termen boven $M$.
\end{proof}

\begin{example}[De monotone stelling als bestaansmachine]\label{ex:b1:seq:existencemachine}
Zij $u_n = \prod_{k=1}^{n} \bigl(1 + \frac{1}{2^k}\bigr)$. Elke factor
is groter dan $1$, dus $(u_n)$ is stijgend. Naar boven begrensd? Neem
logaritmen en gebruik $\ln(1 + x) \leq x$
(\cref{ex:b1:derivative:convexineq} anticipeert erop; of het grove
$1 + x \leq \eu^x$ uit het deel Secundair onderwijs):
\[
\ln u_n = \sum_{k=1}^{n} \ln\Bigl(1 + \frac{1}{2^k}\Bigr)
\leq \sum_{k=1}^{n} \frac{1}{2^k} < 1 ,
\]
dus $u_n < \eu$. Stijgend en begrensd: $(u_n)$ convergeert naar een
zekere $\ell \in \intoc{u_1}{\eu}$ --- een perfect welgedefinieerd
reëel getal zonder gesloten vorm in zicht ($\ell = 2.384\dots$). Het
afsluitende inzicht: de stelling van de monotone limiet is de
goedkoopste bestaansmachine in de analyse; ze benoemde $\eu$ zelf
(\cref{ex:b1:seq:e} hieronder), en in \cref{ch:b1:series} zal ze de
convergentie van elke positieve reeks beslissen door louter
begrensdheid.
\end{example}

\begin{theorem}[Ingesloten rijen]\label{thm:b1:seq:adjacent}
Zij $(a_n)$ stijgend, $(b_n)$ dalend, met $b_n - a_n \to 0$. Dan
convergeren beide, naar een \emph{gemeenschappelijke} limiet $\ell$,
en $a_n \leq \ell \leq b_n$ voor alle $n$.\index{ingesloten rijen}
\end{theorem}

\begin{proof}
Eerst $a_n \leq b_n$ voor alle $n$: de rij $(b_n - a_n)$ is dalend en
gaat naar $0$, dus is ze $\geq 0$ (een negatieve term zou ze onder $0$
vastzetten). Vervolgens is $(a_n)$ stijgend en naar boven begrensd
door $b_0$: ze convergeert naar een zekere $\ell$
(\cref{thm:b1:seq:monotone}); evenzo $(b_n) \to \ell'$; en $\ell' -
\ell = \lim (b_n - a_n) = 0$. De ongelijkheden $a_n \leq \ell \leq
b_n$ volgen uit monotonie ($\ell = \sup a_k \geq a_n$, enz.).
\end{proof}

\begin{example}[Het getal $\eu$]\label{ex:b1:seq:e}
Stel $a_n = \sum_{k=0}^{n} \frac{1}{k!}$ en $b_n = a_n + \frac{1}{n
\cdot n!}$ ($n \geq 1$). Dan stijgt $(a_n)$; en
\[
b_{n+1} - b_n = \frac{1}{(n+1)!} + \frac{1}{(n+1)(n+1)!} -
\frac{1}{n\,n!}
= \frac{n(n+1) + n - (n+1)^2}{n(n+1)(n+1)!}
= \frac{-1}{n(n+1)(n+1)!} < 0 ,
\]
dus $(b_n)$ daalt, en $b_n - a_n \to 0$: ingesloten. Hun
gemeenschappelijke limiet is (per definitie hier) het getal $\eu
\approx 2.71828$; de ongelijkheden $a_n < \eu < b_n$ zijn scherp
genoeg om $\eu \notin \Q$ te bewijzen (\cref{exo:b1:seq:9}).
\end{example}

\section{Deelrijen en Bolzano--Weierstrass}

\begin{definition}[Deelrij]\label{def:b1:seq:subsequence}
Een \emph{deelrij}\index{deelrij} van $(u_n)$ is een rij
$(u_{\varphi(n)})$ waarbij $\varphi \colon \N \to \N$ strikt stijgend
is (merk op dat $\varphi(n) \geq n$, door inductie).
\end{definition}

\begin{proposition}\label{prop:b1:seq:subsequences}
Als $u_n \to \ell$ ($\ell \in \R$ of $\pm\infty$), dan gaat elke
\omterm{def:b1:seq:subsequence}{deelrij} naar $\ell$. Bijgevolg divergeert een rij met twee \omterm{def:b1:seq:subsequence}{deelrijen}
van verschillende limieten. Omgekeerd, als $(u_{2n})$ en $(u_{2n+1})$
beide naar \emph{dezelfde} $\ell$ convergeren, dan $u_n \to \ell$.
\end{proposition}

\begin{proof}
Voorbij de drempel $N$ voor $(u_n)$ voldoen alle indices $\varphi(n)
\geq n \geq N$ (de ongelijkheid $\varphi(n) \geq n$ is de inductie
opgemerkt in \cref{def:b1:seq:subsequence}: $\varphi(0) \geq 0$, en
$\varphi(n+1) > \varphi(n) \geq n$ dwingt $\varphi(n+1) \geq n + 1$
af). Voor de omkering: gegeven $\varepsilon$, neem de twee drempels
$N_0$ (even) en $N_1$ (oneven); een willekeurige index $n \geq
\max(2N_0, 2N_1 + 1)$ is ofwel even, $n = 2k$ met $k \geq N_0$, ofwel
oneven, $n = 2k+1$ met $k \geq N_1$ --- in beide gevallen $\abs{u_n -
\ell} \leq \varepsilon$: elke index wordt gedekt door een van de twee
\omterm{def:b1:seq:subsequence}{deelrijen}, en dat is het hele punt.
\end{proof}

\begin{example}[Deelrijlimieten]\label{ex:b1:seq:sublimits}
Voor $u_n = (-1)^n \frac{n}{n+1}$: de even \omterm{def:b1:seq:subsequence}{deelrij} gaat naar $1$, de
oneven naar $-1$, dus de rij divergeert --- maar ze doet dat op een
georganiseerde manier, clusterend rond de twee waarden $\pm 1$. Voor
$u_n = \cos\frac{2\pi n}{3}$: de drie \omterm{def:b1:seq:subsequence}{deelrijen} van indices $3k$, $3k
+ 1$, $3k + 2$ zijn constant, gelijk aan $1$, $-\frac12$, $-\frac12$;
de \omterm{def:b1:logic:sets}{verzameling} deelrijlimieten is $\{1, -\frac12\}$. Het afsluitende
inzicht: een \emph{begrensde} rij convergeert precies wanneer ze één
enkele deelrijlimiet heeft (\cref{exo:b1:seq:8}); divergentie van een
begrensde rij betekent altijd minstens twee clusters, en
Bolzano--Weierstrass hieronder garandeert dat er minstens één is.
\end{example}

\begin{theorem}[Bolzano--Weierstrass]\label{thm:b1:seq:bw}
Elke begrensde rij van reële getallen heeft een convergente
\omterm{def:b1:seq:subsequence}{deelrij}.\index{stelling van Bolzano--Weierstrass}
\end{theorem}

\begin{proof}
Zij $u_n \in \intcc{a}{b}$ voor alle $n$. Bouw geneste segmenten door
dichotomie: stel $\intcc{a_0}{b_0} = \intcc{a}{b}$; gegeven
$\intcc{a_k}{b_k}$ dat $u_n$ bevat voor oneindig veel $n$, bevat een
van de twee helften nog steeds $u_n$ voor oneindig veel $n$ --- noem
ze $\intcc{a_{k+1}}{b_{k+1}}$. De rijen $(a_k)$, $(b_k)$ zijn
ingesloten ($b_k - a_k = \frac{b-a}{2^k} \to 0$), met
gemeenschappelijke limiet $\ell$ (\cref{thm:b1:seq:adjacent}).

Extraheer: kies $\varphi(0)$ met $u_{\varphi(0)} \in
\intcc{a_0}{b_0}$, dan, door inductie, $\varphi(k+1) > \varphi(k)$ met
$u_{\varphi(k+1)} \in \intcc{a_{k+1}}{b_{k+1}}$ --- mogelijk omdat dat
segment oneindig veel termen bevat. Dan $a_k \leq u_{\varphi(k)} \leq
b_k$, en de insluitstelling geeft $u_{\varphi(k)} \to \ell$.
\end{proof}

\begin{remark}[Wat Bolzano--Weierstrass wel, en niet, zegt]
Ze zegt \emph{wel}: uit louter begrensdheid convergeert een \omterm{def:b1:seq:subsequence}{deelrij}
--- bestaan zonder formule, zoals het dichotomiebewijs duidelijk maakt
(niets vertelt ons \emph{welke} indices overleven). Ze zegt
\emph{niet} dat de limiet uniek is: $((-1)^n)$ heeft \omterm{def:b1:seq:subsequence}{deelrijen} die
naar $1$ en naar $-1$ convergeren, en de \omterm{def:b1:logic:sets}{verzameling} deelrijlimieten
kan zelfs oneindig zijn (\cref{ex:b1:seq:sublimits}, en heel de
Cantorverzameling in \cref{pb:b1:topology:1}). Ze overleeft
onbegrensdheid niet: $(n)$ heeft helemaal geen convergente \omterm{def:b1:seq:subsequence}{deelrij} ---
al kan men uit elke onbegrensde rij altijd een \omterm{def:b1:seq:subsequence}{deelrij} extraheren die
naar $+\infty$ of $-\infty$ gaat (kies $\varphi(k)$ met
$u_{\varphi(k)} \geq k$, bijvoorbeeld). Correct gebruikt is de
stelling een \emph{bestaanspomp}: ze verschijnt op het kruispunt van
het Cauchycriterium hieronder, van de stelling van Heine, en van de
extremumstelling --- steeds om een punt voort te brengen dat geen
enkele expliciete constructie aanbiedt.
\end{remark}

\section{Cauchyrijen en volledigheid}

\begin{definition}\label{def:b1:seq:cauchy}
Een rij $(u_n)$ is een \emph{Cauchyrij}\index{Cauchyrij} wanneer haar
termen willekeurig dicht \emph{bij elkaar} komen:
\[
\forall \varepsilon > 0,\ \exists N,\ \forall p, q \geq N,
\qquad \abs{u_p - u_q} \leq \varepsilon .
\]
\end{definition}

\begin{example}[De Cauchy-eigenschap met de hand nagaan]\label{ex:b1:seq:cauchyhand}
Zij $u_n = \sum_{k=0}^{n} \frac{\cos k}{2^k}$ --- geen monotonie, geen
raadbare limiet. Voor $p > q$:
\[
\abs{u_p - u_q}
= \Bigl| \sum_{k=q+1}^{p} \frac{\cos k}{2^k} \Bigr|
\leq \sum_{k=q+1}^{p} \frac{1}{2^k}
< \frac{1}{2^{q}} ,
\]
door de driehoeksongelijkheid, $\abs{\cos k} \leq 1$ en een eindige
meetkundige som. Gegeven $\varepsilon > 0$, kies $N$ met $2^{-N} \leq
\varepsilon$: alle afstanden voorbij $N$ zijn $\leq \varepsilon$, de
rij is Cauchy, dus convergeert ze --- naar een limiet die niemand in
gesloten vorm kan benoemen, wat precies het punt is. Het afsluitende
inzicht: meetkundige dominantie van de aangroeiingen is de standaard
manier om de Cauchy-eigenschap te verdienen, en \cref{ch:b1:series}
zal het argument bottelen als ``absolute convergentie impliceert
convergentie''.
\end{example}

\begin{theorem}[Volledigheid van $\R$]\label{thm:b1:seq:complete}
Een rij van reële getallen convergeert dan en slechts dan als ze een
\omterm{def:b1:seq:cauchy}{Cauchyrij} is.\index{volledigheid van R@volledigheid van $\R$}
\end{theorem}

\begin{proof}
($\Rightarrow$) Als $u_n \to \ell$: voorbij de drempel voor
$\frac\varepsilon2$ is $\abs{u_p - u_q} \leq \abs{u_p - \ell} +
\abs{\ell - u_q} \leq \varepsilon$.

($\Leftarrow$) Zij $(u_n)$ Cauchy. \emph{Ze is begrensd}: met
$\varepsilon = 1$ liggen voorbij $N$ alle termen binnen $1$ van $u_N$,
en de kop is eindig. \emph{Extraheer}: door \cref{thm:b1:seq:bw} gaat
een zekere \omterm{def:b1:seq:subsequence}{deelrij} $u_{\varphi(n)} \to \ell$. \emph{Besluit}: gegeven
$\varepsilon > 0$, neem $N$ (Cauchy, voor $\frac\varepsilon2$) en $n
\geq N$ met $\abs{u_{\varphi(n)} - \ell} \leq \frac\varepsilon2$ en
$\varphi(n) \geq N$; dan geldt voor elke $p \geq N$:
\[
\abs{u_p - \ell} \leq \abs{u_p - u_{\varphi(n)}} +
\abs{u_{\varphi(n)} - \ell} \leq \varepsilon . \qedhere
\]
\end{proof}

\begin{remark}
De waarde van het criterium: het certificeert convergentie \emph{zonder
de limiet te benoemen}. Het faalt over $\Q$ (de decimale
afkappingen van $\sqrt 2$ vormen een \omterm{def:b1:seq:cauchy}{Cauchyrij} van rationale getallen
zonder rationale limiet): volledigheid is een eigenschap van $\R$,
equivalent met het bovengrensaxioma. Het is ook het werkpaard achter
de convergentie van reeksen (\cref{ch:b1:series}).
\end{remark}

\begin{example}[Een Cauchyrij met een onzichtbare limiet]\label{ex:b1:seq:basel}
Zij $S_n = \sum_{k=1}^{n} \frac{1}{k^2}$. Voor $p > q \geq 1$:
\[
S_p - S_q = \sum_{k=q+1}^{p} \frac{1}{k^2}
\leq \sum_{k=q+1}^{p} \frac{1}{k(k-1)}
= \sum_{k=q+1}^{p} \Bigl(\frac{1}{k-1} - \frac 1k\Bigr)
= \frac 1q - \frac 1p < \frac 1q ,
\]
dus voorbij $N > \frac1\varepsilon$ zijn alle afstanden $\leq
\varepsilon$: $(S_n)$ is Cauchy, dus convergeert ze. Merk op wat
zojuist gebeurde: we bewezen dat een bepaald reëel getal bestaat
zonder er enige naam voor te hebben. (Het is $\frac{\pi^2}{6}$ --- een
gevierde identiteit van Euler, bewezen in het deel Jaar 2; niets in
dit hoofdstuk kon ons dat vertellen.) Deze arbeidsverdeling ---
bestaan nu, identificatie later, indien ooit --- is het hele punt van
het Cauchycriterium, en de motor van de theorie van reeksen in
\cref{ch:b1:series}.
\end{example}

\section{Recurrente rijen}

\begin{method}[Studie van $u_{n+1} = f(u_n)$]\label{met:b1:seq:recurrent}
Gegeven $f$ en een startpunt $u_0$:\index{recurrente rij}
\begin{enumerate}
  \item \emph{Stabiel \omterm{prop:b1:reals:intervals}{interval}:} vind een \omterm{prop:b1:reals:intervals}{interval} $I$ met $f(I)
        \subseteq I$ dat $u_0$ bevat: dan zijn alle $u_n \in I$ (door
        inductie).
  \item \emph{Kandidaat-limieten:} als $u_n \to \ell \in I$ en $f$
        continu is in $\ell$ (\cref{ch:b1:continuity}), dan is $\ell$
        een \emph{vast punt}: $f(\ell) = \ell$. Los $f(x) = x$ op.
  \item \emph{Monotonie:} als $f$ stijgend is op $I$, dan is $(u_n)$
        monotoon (stijgend als $u_1 \geq u_0$, anders dalend);
        gecombineerd met begrensdheid besluit
        \cref{thm:b1:seq:monotone}. Als $f$ dalend is, bestudeer de
        twee \omterm{def:b1:seq:subsequence}{deelrijen} $(u_{2n})$ en $(u_{2n+1})$, die monotoon zijn
        voor $f \circ f$.
  \item \emph{Foutbeheersing:} een ongelijkheid $\abs{f(x) - \ell}
        \leq k\abs{x - \ell}$ met $k < 1$ geeft rechtstreeks
        $\abs{u_n - \ell} \leq k^n \abs{u_0 - \ell} \to 0$.
\end{enumerate}
\end{method}

\begin{example}[Methode van Heron]\label{ex:b1:seq:heron}
Zij $u_0 = 2$ en $u_{n+1} = \dfrac12\Bigl(u_n +
\dfrac{2}{u_n}\Bigr)$: het aloude algoritme voor
$\sqrt 2$.\index{methode van Heron}
\begin{itemize}
  \item \emph{Stabiliteit:} voor $x > 0$ geeft de
        rekenkundig-meetkundige middelongelijkheid $\frac12(x +
        \frac2x) \geq \sqrt{x \cdot \frac 2x} = \sqrt 2$; dus $I =
        \intco{\sqrt 2}{+\infty}$ is stabiel en bevat $u_1$
        (inderdaad $u_1 = \frac32 \geq \sqrt 2$).
  \item \emph{Monotonie:} voor $x \geq \sqrt 2$ is $\;x - f(x) =
        \frac{x^2 - 2}{2x} \geq 0$: de rij daalt vanaf $u_1$, en is
        naar onder begrensd door $\sqrt 2$: ze convergeert.
  \item \emph{Limiet:} de vaste punten lossen $x = \frac12(x +
        \frac2x)$ op, d.w.z.\ $x^2 = 2$: op $I$ is $\ell = \sqrt 2$.
  \item \emph{Snelheid:} $u_{n+1} - \sqrt 2 = \frac{(u_n -
        \sqrt2)^2}{2u_n} \leq \frac{(u_n - \sqrt 2)^2}{2\sqrt 2}$: het
        aantal juiste cijfers \emph{verdubbelt} ruwweg bij elke stap
        (kwadratische convergentie).
\end{itemize}
\end{example}

\begin{remark}[Veelvoorkomende valkuilen met limieten]
Vier klassiekers. (i) \emph{Kleine stappen impliceren geen
convergentie}: $u_{n+1} - u_n \to 0$ is veel zwakker dan de
Cauchy-eigenschap --- de harmonische sommen $H_n$ hebben stappen
$\frac{1}{n+1} \to 0$ en divergeren toch naar $+\infty$
(\cref{exo:b1:seq:5}); de Cauchy-voorwaarde beheerst $\abs{u_p - u_q}$
voor \emph{alle} grote paren, niet voor opeenvolgende. (ii)
\emph{Strikte ongelijkheden sterven in de limiet}: uit $u_n < v_n$
voor alle $n$ krijgt men enkel $\lim u_n \leq \lim v_n$
(\cref{thm:b1:seq:order}); $\frac1n > 0$ en toch $\lim = 0$. (iii)
\emph{Begrensd is niet convergent}: $((-1)^n)$ is begrensd en
divergeert; begrensdheid plus \emph{monotonie} convergeert,
begrensdheid alleen garandeert enkel een convergente \omterm{def:b1:seq:subsequence}{deelrij}
(\cref{thm:b1:seq:bw}). (iv) \emph{De vastepuntsvergelijking komt op
de tweede plaats, niet de eerste}: voor $u_{n+1} = f(u_n)$
identificeert het oplossen van $f(\ell) = \ell$ de limiet \emph{pas
nadat} de convergentie bewezen is. De recurrentie $u_{n+1} = 2u_n$
heeft het unieke vaste punt $\ell = 0$, maar vanaf $u_0 = 1$ loopt de
rij naar $+\infty$: de vergelijking $\ell = 2\ell$ had nooit recht op
een limiet. Volgorde van bewerkingen, altijd: bestaan eerst
(\cref{met:b1:seq:recurrent}, stappen 1--3), identificatie tweede.
\end{remark}

\begin{example}[Een dalende $f$: de gulden recurrentie]\label{ex:b1:seq:golden}
Zij $u_0 = 1$ en $u_{n+1} = \dfrac{1}{1 + u_n}$. Hier is $f(x) =
\frac{1}{1+x}$ \emph{dalend}, dus de rij is niet monotoon (ze
alterneert rond haar limiet); de contractiestap van
\cref{met:b1:seq:recurrent} is het juiste gereedschap. Stabiliteit:
als $x \in \intcc{\frac12}{1}$ dan $1 + x \in \intcc{\frac32}{2}$, dus
$f(x) \in \intcc{\frac12}{\frac23} \subseteq \intcc{\frac12}{1}$, en
$u_1 = \frac12$ plaatst de hele rij daar. Vast punt: $\ell =
\frac{1}{1+\ell}$ met $\ell > 0$ geeft $\ell^2 + \ell - 1 = 0$,
d.w.z.
\[
\ell = \frac{\sqrt5 - 1}{2} = 0.6180\dots
\]
(de inverse van de gulden snede). Contractie: voor $x, y \in
\intcc{\frac12}{1}$,
\[
\abs{f(x) - f(y)} = \frac{\abs{x - y}}{(1+x)(1+y)}
\leq \frac{\abs{x-y}}{(3/2)^2} = \frac49\,\abs{x - y} ,
\]
dus $\abs{u_n - \ell} \leq \bigl(\frac49\bigr)^{n-1}\abs{u_1 - \ell}
\to 0$: convergentie, met meetkundige snelheid, zonder monotonie
nodig te hebben. Het afsluitende inzicht: monotone methoden en
contractiemethoden verdelen de recurrente wereld onder elkaar ---
stijgende $f$ geeft monotone banen, dalende $f$ geeft alternerende
banen getemd door een Lipschitz-constante $< 1$ (de systematische
theorie is \cref{exo:b1:derivative:11}).
\end{example}

\begin{remark}[Perspectieven binnen dit deel]
Rijen zijn het meetinstrument dat de rest van het deel tegen elk
object aanhoudt. In \cref{ch:b1:topology} \emph{karakteriseren} ze
geslotenheid en compactheid; in \cref{ch:b1:continuity} transporteren
ze limieten van functies; in \cref{ch:b1:integration} zijn
Riemannsommen rijen die naar de integraal convergeren;
\cref{ch:b1:series} \emph{is} de theorie van één speciale klasse van
rijen, de partiële sommen. Zelfs de algebrahoofdstukken verbruiken
ze: de iteraten van een matrix in \cref{ch:b1:matrices} vormen rijen
waarvan het gedrag (convergentie van $A^n$) een lineaire-algebravraag
is met de woordenschat van dit hoofdstuk. De twee stellingen om overal
mee te dragen: monotone limiet (bestaan uit orde) en
Bolzano--Weierstrass (bestaan uit begrensdheid) --- tussen hen beiden
wordt bijna elke limiet in dit boek geboren.
\end{remark}

\begin{remark}[Complexe rijen]
Een rij $(z_n)$ van complexe getallen convergeert naar $\ell$ wanneer
$\abs{z_n - \ell} \to 0$; equivalent, wanneer $\Re(z_n) \to \Re(\ell)$
en $\Im(z_n) \to \Im(\ell)$ (vergelijk $\abs{z}$ met $\abs{\Re z} +
\abs{\Im z}$). De stellingen die geen orde betrekken --- bewerkingen,
Bolzano--Weierstrass (extraheer tweemaal), Cauchycriterium --- gaan
woordelijk over.
\end{remark}

\begin{exercise}[$\star\star\star$]\label{exo:b1:seq:10}

\end{enumerate}
\end{exercise}

\section{Probleem: Cesàro, Stolz, en de trage val van de sinus}

\begin{problem}[{Weekendprobleem --- de stelling van Cesàro--Stolz
en de asymptotiek $u_n \sim \sqrt{3/n}$ voor $u_{n+1} = \sin
u_n$}]
\label{pb:b1:seq:1}
De stelling van Cesàro--Stolz is de discrete regel van l'Hôpital: om
de limiet van een quotiënt $a_n/b_n$ te vinden, volstaat het de
limiet te vinden van het quotiënt van \emph{aangroeiingen} $(a_{n+1} -
a_n)/(b_{n+1} - b_n)$. Dit probleem bewijst de stelling, oogst er
klassieke limieten mee, en richt ze dan op een beroemd doelwit: de
rij $u_{n+1} = \sin u_n$, die naar $0$ kruipt met de exact
berekenbare snelheid $u_n \sim \sqrt{3/n}$.\index{Cesaro--Stolz
theorem@stelling van Cesàro--Stolz} Twee feiten uit het deel
Secundair onderwijs worden hier verleend en later in dit deel eerlijk
opnieuw bewezen: de raaklijnongelijkheid
\[
\tag{G1} \eu^{u} \geq 1 + u \quad (u \in \R),
\]
opnieuw bewezen door convexiteit in \cref{ch:b1:derivative}, en de
sinusinsluiting
\[
\tag{G2} x - \frac{x^3}{6} \;\leq\; \sin x \;\leq\; x -
\frac{x^3}{6} + \frac{x^5}{120} \quad (0 \leq x \leq 1), \qquad
\abs{\sin x} \leq \abs{x} \quad (x \in \R),
\]
opnieuw bewezen door de formule van Taylor in \cref{ch:b1:taylor}.

\medskip
\noindent\textbf{Deel I --- Sommen zonder gesloten formules.}
\begin{enumerate}
  \item Gebruik $1 + 2 + \dots + n = \frac{n(n+1)}{2}$ en $1^2 +
        \dots + n^2 = \frac{n(n+1)(2n+1)}{6}$ om $\lim \frac{1 + 2 +
        \dots + n}{n^2}$ en $\lim \frac{1^2 + \dots + n^2}{n^3}$ te
        berekenen.
  \item Zij $T_n = \sum_{k=1}^n \sqrt k$, waarvoor geen gesloten
        formule bestaat. Bewijs de insluiting
        \[
        \frac{1}{2\sqrt 2}\,n^{3/2} \;\leq\; T_n \;\leq\;
        n^{3/2}
        \]
        \emph{(behoud enkel de termen $k > \frac n2$ voor de
        ondergrens)}. Dus $T_n$ heeft de orde $n^{3/2}$ --- maar met
        welke constante? Houd de vraag aan tot vraag 8.
  \item (Telescoperingslemma) Zij $(b_n)$ strikt stijgend en stel dat
        voor alle $k \geq N$,
        \[
        m \;\leq\; \frac{a_{k+1} - a_k}{b_{k+1} - b_k} \;\leq\;
        M .
        \]
        Bewijs dat $m \leq \dfrac{a_n - a_N}{b_n - b_N} \leq M$ voor
        elke $n > N$.
\end{enumerate}

\medskip
\noindent\textbf{Deel II --- De stelling van Cesàro--Stolz.} Zij
$(b_n)$ strikt stijgend met $b_n \to +\infty$, en stel $\dfrac{a_{n+1}
- a_n}{b_{n+1} - b_n} \to \ell \in \R$.
\begin{enumerate}[resume]
  \item Fixeer $\varepsilon > 0$. Toon aan dat er een $N$ is zodat
        $\ell - \varepsilon \leq \dfrac{a_n - a_N}{b_n - b_N} \leq
        \ell + \varepsilon$ voor alle $n > N$.
  \item Stel, voor $n > N$, de identiteit op
        \[
        \frac{a_n}{b_n} - \ell
        = \frac{a_N - \ell\,b_N}{b_n}
        + \Bigl(1 - \frac{b_N}{b_n}\Bigr)
        \Bigl(\frac{a_n - a_N}{b_n - b_N} - \ell\Bigr),
        \]
        en besluit de stelling: $\dfrac{a_n}{b_n} \to \ell$.
  \item Bewijs de $+\infty$-variant: als $\dfrac{a_{n+1} -
        a_n}{b_{n+1} - b_n} \to +\infty$ (dezelfde hypothesen op
        $(b_n)$), dan $\dfrac{a_n}{b_n} \to +\infty$.
  \item Neem $b_n = n$: herwin de stelling van het Cesàro-gemiddelde
        van \cref{exo:b1:seq:10}. Toon dan aan dat de omkering van
        Cesàro--Stolz faalt: voor $a_n = (-1)^n$, $b_n = n$
        convergeert het quotiënt $a_n/b_n$ terwijl het quotiënt van
        aangroeiingen dat niet doet. Stolz is een eenrichtingsstraat.
\end{enumerate}

\medskip
\noindent\textbf{Deel III --- Eerste dividenden.}
\begin{enumerate}[resume]
  \item Bewijs $(1+h)^{3/2} - 1 = \dfrac{3h + 3h^2 + h^3}{(1+h)^{3/2}
        + 1}$ door toevoeging, leid $n\bigl((1 + \tfrac1n)^{3/2} -
        1\bigr) \to \tfrac32$ af, en besluit met Cesàro--Stolz:
        \[
        T_n = \sum_{k=1}^{n} \sqrt k \;\sim\; \tfrac23\,
        n^{3/2} ,
        \]
        wat de cliffhanger van vraag 2 oplost.
  \item Leid uit (G1) alleen de logaritme-insluiting af
        \[
        \frac{t}{1 + t} \;\leq\; \ln(1 + t) \;\leq\; t
        \qquad (t > -1)
        \]
        \emph{(pas (G1) toe in $u = \ln(1+t)$ en in $u =
        -t/(1+t)$)}.
  \item Toon aan dat $b_n = \ln n$ strikt stijgend is met $\ln n \to
        +\infty$, en bewijs met Cesàro--Stolz en vraag 9 dat
        \[
        H_n = \sum_{k=1}^{n} \frac 1k \;\sim\; \ln n .
        \]
        (De fijnere structuur $H_n = \ln n + \gamma + o(1)$ is het
        weekendprobleem van \cref{ch:b1:series}.)
  \item (Van verhoudingen naar wortels) Zij $u_n > 0$ met
        $\frac{u_{n+1}}{u_n} \to L > 0$. Toon met vraag 9 aan dat
        $\ln\frac{u_{n+1}}{u_n} \to \ln L$; pas Cesàro toe om te
        besluiten dat $\frac{\ln u_n}{n} \to \ln L$, en dan, met
        (G1), dat $u_n^{1/n} \to L$. Toepassing: bereken
        $\lim\,\binom{2n}{n}^{1/n}$.
\end{enumerate}

\medskip
\noindent\textbf{Deel IV --- De trage val van de sinus.} Zij $u_0 \in
\R$ en $u_{n+1} = \sin u_n$.
\begin{enumerate}[resume]
  \item Toon uit (G2) aan dat $0 < \sin x < x$ voor $0 < x \leq 1$.
        Leid af: $u_1 \in \intcc{-1}{1}$; als $u_1 = 0$ is de rij nul
        vanaf rang $1$; en als $u_1 > 0$ (het geval $u_1 < 0$ is
        symmetrisch, $\sin$ oneven), dan is $(u_n)_{n \geq 1}$ strikt
        dalend, positief, en convergeert naar $0$ \emph{(identificeer
        de limiet via $\ell = \sin \ell$, met $\abs{\sin a - \sin b}
        \leq \abs{a - b}$, zelf een gevolg van (G2) en de
        product-naar-som-formule)}.
  \item Neem vanaf nu aan $u_1 \in \intoc{0}{1}$. Toon door
        insluiting, met (G2):
        \[
        \frac{\sin u_n}{u_n} \to 1
        \qquad\text{en}\qquad
        \frac{u_n - \sin u_n}{u_n^{3}} \to \frac16 .
        \]
  \item Bewijs de ontbinding
        \[
        w_n := \frac{1}{u_{n+1}^{2}} - \frac{1}{u_n^{2}}
        = \frac{u_n - \sin u_n}{u_n^{3}} \cdot
        \frac{u_n + \sin u_n}{u_n} \cdot
        \Bigl(\frac{u_n}{\sin u_n}\Bigr)^{2},
        \]
        en leid $w_n \to \frac13$ af.
  \item Besluit met \cref{exo:b1:seq:10} (aangroeiingsversie) dat
        $\frac{1}{n\,u_n^{2}} \to \frac13$, en dan, door een
        toevoegingsargument voor de vierkantswortel, de kop:
        \[
        \sqrt n\;u_n \longrightarrow \sqrt 3 ,
        \qquad\text{d.w.z.}\qquad
        u_n \sim \sqrt{\frac 3n} .
        \]
  \item Kwantificeer de traagheid: toon aan dat uiteindelijk
        $\sqrt{2/n} \leq u_n \leq 2/\sqrt n$, zodat het bereiken van
        $u_n \leq 10^{-2}$ meer dan $20\,000$ iteraties vereist
        (ongeveer $30\,000$, volgens de asymptotiek). Contrasteer met
        de \omterm{ex:b1:seq:heron}{methode van Heron} (\cref{ex:b1:seq:heron}) en verklaar de
        structurele reden: in het vaste punt $0$ is de helling van
        $\sin$ gelijk aan $1$ (een \emph{neutraal} vast punt), terwijl
        fouthalverende iteraties een helling van \omterm{def:b1:complex:field}{modulus} $< 1$ nodig
        hebben.
  \item Toon aan dat voor \emph{elk} startpunt $u_0 \in \R$ ofwel
        $u_n = 0$ vanaf rang $1$, ofwel $\abs{u_n} \sim \sqrt{3/n}$
        --- de val is universeel, alleen het teken onthoudt $u_0$.
\end{enumerate}

\medskip
\noindent\textbf{Deel V --- Het algemene principe.} De sinus is één
exemplaar van een machine.
\begin{enumerate}[resume]
  \item Zij $u_n > 0$, $u_n \to 0$, en $\dfrac{u_n - u_{n+1}}{u_n^{2}}
        \to a > 0$. Bewijs achtereenvolgens: $\frac{u_{n+1}}{u_n} \to
        1$; dan $\frac{1}{u_{n+1}} - \frac{1}{u_n} \to a$; dan
        $n\,u_n \to \frac1a$.
  \item (Exact model) Voor $u_{n+1} = \dfrac{u_n}{1 + u_n}$, $u_0 >
        0$: toon aan dat $\frac{1}{u_n}$ rekenkundig is, los exact op,
        en controleer het besluit van vraag 18 tegen de exacte
        formule.
  \item Voor $u_{n+1} = u_n \eu^{-u_n}$, $u_0 > 0$: toon aan dat $u_n
        \to 0$, gebruik (G1) om $\frac{1 - \eu^{-t}}{t}$ in te sluiten
        tussen $\frac{1}{1+t}$ en $1$ voor $t > 0$, en besluit $u_n
        \sim \frac 1n$.
  \item (Kubisch contact, gekwadrateerde telescoop) Zij $u_n > 0$,
        $u_n \to 0$, $\dfrac{u_n - u_{n+1}}{u_n^{3}} \to a > 0$. Pas
        de ontbinding van vraag 14 aan om aan te tonen dat
        $\frac{1}{u_{n+1}^2} - \frac{1}{u_n^2} \to 2a$, en besluit
        $n\,u_n^{2} \to \frac{1}{2a}$. Controleer dat $a = \frac16$
        Deel IV herwint.
\end{enumerate}

\medskip
\noindent\textbf{Deel VI --- Grenzen van de methode, en moralen.}
\begin{enumerate}[resume]
  \item Toon aan dat de hypothese $b_n \to +\infty$ niet kan worden
        weggelaten: voor $a_n = 2 - 2^{-n}$ en $b_n = 1 - 2^{-n}$
        gaat het aangroeiingsquotiënt naar $1$ terwijl $\frac{a_n}
        {b_n} \to 2$. Wijs de precieze regel van het bewijs van vraag
        5 aan die breekt.
  \item (Stolz tweemaal) Bewijs $\sum_{k=1}^{n} H_k \sim n \ln n$
        \emph{(één toepassing van Cesàro--Stolz, dan vraag 10;
        begrens $(n+1)\ln(n+1) - n\ln n$ met vraag 9)}.
  \item (Meetkundige gemiddelden) Als $u_n > 0$ en $u_n \to \ell >
        0$, toon aan dat $(u_1 u_2 \cdots u_n)^{1/n} \to \ell$; als
        $u_n \to +\infty$, toon aan dat $(u_1 \cdots u_n)^{1/n} \to
        +\infty$. Leid af $(n!)^{1/n} \to +\infty$.
  \item Synthese, één zin elk: (i) waar precies deed volledigheid haar
        intrede in dit probleem; (ii) in welke zin is Cesàro--Stolz
        een discrete regel van l'Hôpital (haar differentiële tweeling
        rust op de middelwaardestelling van \cref{ch:b1:derivative});
        (iii) formuleer de heuristiek die de contactorde van $f$ in
        een neutraal vast punt verbindt met de vervalexponent van
        $u_{n+1} = f(u_n)$; (iv) traceer de constante $3$ van
        $\sqrt{3/n}$ terug door de pijplijn $\frac16 \to \frac13 \to
        3$.
\end{enumerate}
\end{problem}
