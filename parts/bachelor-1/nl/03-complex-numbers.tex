\chapter{Complexe getallen}\label{ch:b1:complex}

Complexe getallen kwamen in het deel Middelbaar onderwijs aan bod als
een rekenkundig hulpmiddel voor kwadratische vergelijkingen. Dit
hoofdstuk behandelt ze als een centraal object: de exponentiële vorm en
haar gevolgen (de Moivre, $n$-de wortels, \omterm{def:b1:complex:unity}{eenheidswortels}), de
systematische vertaling tussen $\C$ en de vlakke meetkunde, en het
gebruik van $\eu^{\iu\theta}$ als een machine om trigonometrische
identiteiten te bewijzen.

\section{Het lichaam $\C$, modulus en toegevoegde}

\begin{definition}[Het complexe lichaam]\label{def:b1:complex:field}
$\C = \{a + \iu b : a, b \in \R\}$ met de gebruikelijke optelling en de
vermenigvuldiging bepaald door $\iu^2 = -1$. Elke $z = a + \iu
b \neq 0$ heeft een inverse: $z^{-1} = \frac{a - \iu b}{a^2 + b^2}$ ---
in de taal van \cref{ch:b1:structures} is $\C$ een lichaam. Men schrijft
$a = \Re(z)$, $b = \Im(z)$, $\conj{z} = a - \iu b$ (de
\emph{toegevoegde}\index{toegevoegde}) en $\abs{z} = \sqrt{a^2 + b^2}$
(de \emph{modulus}\index{modulus}).
\end{definition}

\begin{proposition}[Regels voor toegevoegde en modulus]\label{prop:b1:complex:rules}
Voor $z, w \in \C$:
\begin{enumerate}
  \item $\conj{z + w} = \conj z + \conj w$, $\conj{zw} = \conj z\,
        \conj w$, $\conj{\conj z} = z$;
  \item $z \conj z = \abs z^2$; $\;\Re(z) = \frac{z + \conj z}{2}$,
        $\Im(z) = \frac{z - \conj z}{2\iu}$;
  \item $\abs{zw} = \abs z\, \abs w$, en $\abs{z^{-1}} =
        \abs{z}^{-1}$ voor $z \neq 0$;
  \item (driehoeksongelijkheid\index{driehoeksongelijkheid}) $\abs{z + w}
        \leq \abs z + \abs w$, met gelijkheid dan en slechts dan als $z$ en
        $w$ op een gemeenschappelijke halfrechte vanuit $0$ liggen ($w = \lambda z$ of
        $z = \lambda w$ met $\lambda \geq 0$);
  \item (omgekeerde driehoeksongelijkheid) $\bigl|\abs z - \abs w\bigr|
        \leq \abs{z - w}$.
\end{enumerate}
\end{proposition}

\begin{proof}
(1) en (2) zijn directe berekeningen op reële en imaginaire delen. (3):
$\abs{zw}^2 = zw\,\conj{zw} = z\conj z\, w \conj w = \abs z^2 \abs
w^2$, neem dan vierkantswortels; pas dit toe op $z \cdot z^{-1} = 1$ voor de
inverse.

(4) Beide leden zijn niet-negatief, dus vergelijk kwadraten:
\[
\abs{z+w}^2 = (z+w)(\conj z + \conj w)
= \abs z^2 + \abs w^2 + 2\,\Re(z \conj w),
\]
en $\Re(z\conj w) \leq \abs{z \conj w} = \abs z \abs w$ geeft
$\abs{z+w}^2 \leq (\abs z + \abs w)^2$. Gelijkheid dwingt $\Re(z \conj
w) = \abs{z \conj w}$ af, d.w.z.\ $z \conj w \in \R_+$; als $w \neq 0$ geeft dit
$z = \frac{z\conj w}{\abs w^2}\, w = \lambda w$ met $\lambda
\geq 0$ (en het geval $w = 0$ is triviaal).

(5) $\abs z = \abs{(z - w) + w} \leq \abs{z-w} + \abs w$ geeft $\abs z
- \abs w \leq \abs{z - w}$; verwissel $z$ en $w$ voor het andere teken.
\end{proof}

\begin{example}[Een volledige oefening met modulus en argument]\label{ex:b1:complex:workout}
Schrijf $w = \dfrac{3 + 4\iu}{1 - 2\iu}$ in algebraïsche vorm en bereken
haar \omterm{def:b1:complex:field}{modulus} tweemaal. Vermenigvuldigen met de \omterm{def:b1:complex:field}{toegevoegde} van de
noemer:
\[
w = \frac{(3 + 4\iu)(1 + 2\iu)}{(1 - 2\iu)(1 + 2\iu)}
= \frac{3 + 6\iu + 4\iu - 8}{1 + 4}
= \frac{-5 + 10\iu}{5} = -1 + 2\iu .
\]
Rechtstreeks: $\abs w = \sqrt{1 + 4} = \sqrt5$. Via de quotiëntregel
(\cref{prop:b1:complex:rules}~(3)): $\abs w = \frac{\abs{3 +
4\iu}}{\abs{1 - 2\iu}} = \frac5{\sqrt5} = \sqrt5$ --- hetzelfde
antwoord, geen algebraïsche vorm nodig. De les veralgemeent: moduli en
argumenten reizen goed door producten en quotiënten, reële en
imaginaire delen reizen goed door sommen. Kies de
voorstelling die past bij de bewerkingen die voorhanden zijn, en zet
alleen om wanneer je gedwongen wordt.
\end{example}

\begin{example}[Vergelijkingen met de toegevoegde]\label{ex:b1:complex:conjeq}
Los op in $\C$: $\;z + 2\conj z = 6 + 2\iu$. Een vergelijking die $z$
en $\conj z$ mengt, is \emph{geen} veelterm in $z$; de betrouwbare zet
is haar op te splitsen in reële coördinaten. Met $z = x + \iu y$:
\[
z + 2\conj z = 3x - \iu y ,
\]
dus de vergelijking luidt $3x = 6$ en $-y = 2$: de enige oplossing
is $z = 2 - 2\iu$. (Controle: $(2 - 2\iu) + 2(2 + 2\iu) = 6 +
2\iu$.) Als alternatief kun je de hele vergelijking toevoegen om
$\conj z + 2z = 6 - 2\iu$ te krijgen en het lineaire stelsel in de
onbekenden $z, \conj z$ op te lossen --- hetzelfde antwoord, en een
nuttige truc wanneer de coëfficiënten complex zijn. Vergelijkingen in $z$ en $\conj z$ zijn
in werkelijkheid stelsels van twee reële vergelijkingen; verwachten dat ``graad $1$, één
oplossing'' is hier veilig, maar $z\conj z = -1$ (geen oplossing) toont
dat de veelterm-intuïtie faalt zodra er producten opduiken.
\end{example}

\section{Exponentiële vorm}

\begin{definition}[Complexe exponentiële van een imaginair argument]\label{def:b1:complex:expi}
Voor $\theta \in \R$ definieert men
\[
\eu^{\iu\theta} = \cos\theta + \iu \sin\theta .
\]
Elke $z \neq 0$ kan geschreven worden als $z = r\,\eu^{\iu\theta}$ met $r =
\abs z > 0$; $\theta$ is een \emph{argument}\index{argument} van $z$,
bepaald op een veelvoud van $2\pi$ na. De waarde in
$\intoc{-\pi}{\pi}$ is het \emph{hoofdargument}, geschreven
$\arg z$.
\end{definition}

\begin{example}[Eerste waarden, en één beroemde identiteit]\label{ex:b1:complex:eulerid}
De definitie aflezen bij de kardinale hoeken:
\[
\eu^{\iu\pi/2} = \iu, \qquad
\eu^{\iu\pi} = -1, \qquad
\eu^{2\iu\pi} = 1, \qquad
\eu^{\iu\pi/4} = \frac{\sqrt2}2\,(1 + \iu) .
\]
De tweede, herschikt als $\eu^{\iu\pi} + 1 = 0$, is Eulers
gevierde identiteit die $\eu$, $\iu$, $\pi$, $1$ en $0$ verbindt; in
dit stadium van het boek is het eerder een definitie die zich ontrolt dan een
stelling, en de echte inhoud ervan --- waarom de analytische
exponentiële functie van \cref{ch:b1:functions}, uitgebreid tot imaginaire
argumenten, dezelfde naam verdient --- wordt vastgelegd door de
machtreeksen van \cref{ch:b1:series}. Intussen is de bovenstaande
weergave het waard om uit het hoofd te leren als een omzettingstabel: ze wordt stil gebruikt telkens
wanneer een argument van een tekening wordt afgelezen.
\end{example}

\begin{theorem}[De functionaalvergelijking]\label{thm:b1:complex:funceq}
Voor alle $\theta, \varphi \in \R$:
\[
\eu^{\iu\theta}\, \eu^{\iu\varphi} = \eu^{\iu(\theta + \varphi)},
\qquad
\abs{\eu^{\iu\theta}} = 1,
\qquad
\conj{\eu^{\iu\theta}} = \eu^{-\iu\theta} = (\eu^{\iu\theta})^{-1}.
\]
Bijgevolg draagt $\abs{zw} = \abs z \abs w$ ook de argumenten:
$\arg(zw) \equiv \arg z + \arg w \pmod{2\pi}$.
\end{theorem}

\begin{proof}
Ontwikkel het product en gebruik de somformules:
\[
(\cos\theta + \iu\sin\theta)(\cos\varphi + \iu\sin\varphi)
= (\cos\theta\cos\varphi - \sin\theta\sin\varphi)
+ \iu\,(\sin\theta\cos\varphi + \cos\theta\sin\varphi),
\]
wat $\cos(\theta+\varphi) + \iu\sin(\theta+\varphi)$ is. De \omterm{def:b1:complex:field}{modulus}
is $\sqrt{\cos^2\theta + \sin^2\theta} = 1$, en de formule voor de \omterm{def:b1:complex:field}{toegevoegde}
is de pariteit van cosinus en sinus; ze inverteert $\eu^{\iu\theta}$ aangezien
$\eu^{\iu\theta}\eu^{-\iu\theta} = \eu^0 = 1$.
\end{proof}

\begin{corollary}[Formule van de Moivre]\label{cor:b1:complex:demoivre}
Voor $\theta \in \R$ en $n \in \Z$:
$\;(\cos\theta + \iu\sin\theta)^n = \cos(n\theta) +
\iu\sin(n\theta)$.\index{formule van de Moivre}
\end{corollary}

\begin{proof}
Voor $n \geq 0$, inductie. Het geval $n = 0$ luidt $1 = 1$. Als de
formule geldt voor $n$, dan geeft de functionaalvergelijking
(\cref{thm:b1:complex:funceq})
\[
\bigl(\eu^{\iu\theta}\bigr)^{n+1}
= \bigl(\eu^{\iu\theta}\bigr)^{n}\,\eu^{\iu\theta}
= \eu^{\iu n\theta}\,\eu^{\iu\theta}
= \eu^{\iu(n+1)\theta} ,
\]
wat de formule op rang $n + 1$ is. Voor $n < 0$, schrijf $n = -m$
met $m > 0$: aangezien $\eu^{\iu m\theta}$ de inverse
$\eu^{-\iu m\theta}$ heeft (\cref{thm:b1:complex:funceq} opnieuw),
\[
\bigl(\eu^{\iu\theta}\bigr)^{-m}
= \bigl(\eu^{\iu m\theta}\bigr)^{-1}
= \eu^{\iu(-m)\theta} . \qedhere
\]
\end{proof}

\begin{example}[$\cos 3\theta$ ontwikkelen met de Moivre]\label{ex:b1:complex:cos3}
Schrijf $c = \cos\theta$, $s = \sin\theta$. De Moivre en het
binomium van Newton geven
\[
\cos 3\theta + \iu \sin 3\theta = (c + \iu s)^3
= c^3 - 3cs^2 + \iu\,(3c^2 s - s^3),
\]
en door de reële delen te identificeren en vervolgens $s^2 = 1 - c^2$ te substitueren:
\[
\cos 3\theta = c^3 - 3c(1 - c^2) = 4\cos^3\theta - 3\cos\theta .
\]
Het imaginaire deel levert gratis $\sin 3\theta = 3\sin\theta -
4\sin^3\theta$ op: één complexe identiteit draagt altijd
\emph{twee} reële met zich mee. Achterstevoren gelezen is de omkaderde
identiteit de sleutel tot de klassieke driedelingsvergelijking:
$\cos(\theta/3)$ construeren uit $\cos\theta$ betekent de derdegraadsvergelijking $4x^3 -
3x = \cos\theta$ oplossen, en daar neemt de algebra van
\cref{ch:b1:poly} het over.
\end{example}

\begin{proposition}[Formules van Euler]\label{prop:b1:complex:euler}
\[
\cos\theta = \frac{\eu^{\iu\theta} + \eu^{-\iu\theta}}{2},
\qquad
\sin\theta = \frac{\eu^{\iu\theta} - \eu^{-\iu\theta}}{2\iu}.
\]
\end{proposition}

\begin{proof}
Tel op, respectievelijk trek af, $\eu^{\iu\theta} = \cos\theta +
\iu\sin\theta$ en $\eu^{-\iu\theta} = \cos\theta - \iu\sin\theta$.
\end{proof}

\begin{method}[Trigonometrie via exponentiëlen]\label{met:b1:complex:trig}
\begin{enumerate}
  \item \emph{Lineariseer} $\cos^p\theta \sin^q\theta$ (zet machten om in
        een som van $\cos k\theta$, $\sin k\theta$): substitueer de
        formules van Euler, ontwikkel met het binomium van Newton
        (\cref{thm:b1:counting:binomial}), hergroepeer \omterm{def:b1:complex:field}{toegevoegde} termen.
  \item \emph{Ontwikkel} $\cos n\theta$ als een veelterm in $\cos\theta$:
        schrijf $\cos n\theta = \Re\bigl((\cos\theta +
        \iu\sin\theta)^n\bigr)$, ontwikkel, en zet even machten van
        $\sin$ om via $\sin^2 = 1 - \cos^2$.
  \item \emph{Sommeer trigonometrische reeksen} zoals $\sum_k \cos k\theta$:
        herken het reële deel van een meetkundige som $\sum_k
        (\eu^{\iu\theta})^k$.
  \item De \emph{halvehoekontbinding}: voor alle $p, q$,
        \[
        \eu^{\iu p} + \eu^{\iu q}
        = 2 \cos\tfrac{p - q}{2}\; \eu^{\iu \frac{p+q}{2}},
        \qquad
        \eu^{\iu p} - \eu^{\iu q}
        = 2\iu \sin\tfrac{p - q}{2}\; \eu^{\iu \frac{p+q}{2}} .
        \]
\end{enumerate}
\end{method}

\begin{example}[Linearisatie]\label{ex:b1:complex:linearize}
\[
\cos^3\theta
= \Bigl(\frac{\eu^{\iu\theta} + \eu^{-\iu\theta}}{2}\Bigr)^{\!3}
= \frac{\eu^{3\iu\theta} + 3\eu^{\iu\theta} + 3\eu^{-\iu\theta} +
\eu^{-3\iu\theta}}{8}
= \frac{\cos 3\theta + 3\cos\theta}{4}.
\]
Deze vorm integreert onmiddellijk --- precies de reden waarom linearisatie
ertoe doet in \cref{ch:b1:integration}. Een gemengd product werkt
op dezelfde manier, alleen met beide formules van Euler tegelijk:
\[
\sin^2\theta\cos^2\theta
= \Bigl(\frac{\sin2\theta}2\Bigr)^{\!2}
= \frac{1}{4}\cdot
\Bigl(\frac{\eu^{2\iu\theta} - \eu^{-2\iu\theta}}{2\iu}
\Bigr)^{\!2}
= \frac{2 - \eu^{4\iu\theta} - \eu^{-4\iu\theta}}{16}
= \frac{1 - \cos4\theta}{8} ,
\]
waarbij de dubbelehoek-verkorting in de eerste stap één
binomiale ontwikkeling bespaarde --- daar loont het altijd om naar uit te
kijken vooraleer je gaat mechaniseren.
\end{example}

\begin{example}[Een binomiale trigonometrische som]\label{ex:b1:complex:binomsum}
Voor $n \in \N$ en $\theta \in \R$, bereken $S =
\sum_{k=0}^{n}\binom nk \cos k\theta$. Herken het reële deel van
een binomiale ontwikkeling:
\[
S = \Re\sum_{k=0}^n \binom nk \bigl(\eu^{\iu\theta}\bigr)^k
= \Re\bigl(1 + \eu^{\iu\theta}\bigr)^n ,
\]
ontbind dan de halve hoek (\cref{met:b1:complex:trig}~(4)): $1 +
\eu^{\iu\theta} = 2\cos\frac\theta2\,\eu^{\iu\theta/2}$, dus
\[
S = \Re\Bigl(2^n\cos^n\frac\theta2\;\eu^{\iu n\theta/2}\Bigr)
= 2^n \cos^n\frac\theta2\,\cos\frac{n\theta}2 .
\]
Het imaginaire deel levert gratis $\sum_k\binom nk\sin k\theta =
2^n\cos^n\frac\theta2\sin\frac{n\theta}2$ op. Controles:
$\theta = 0$ herwint $\sum\binom nk = 2^n$, en $\theta = \pi$
geeft $S = 0$ voor $n \geq 1$ (elke factor $\cos\frac\pi2$
verdwijnt), d.w.z.\ de alternerende rijsom van
\cref{ex:b1:counting:binomial}. De methode --- ``zie de cosinus-
som als de schaduw van een complexe macht, ontbind dan halve hoeken''
--- is precies die van \cref{exo:b1:complex:6}, met het binomium
van Newton dat de meetkundige reeks vervangt.
\end{example}

\section{Wortels van complexe getallen}

\begin{theorem}[$n$-de wortels]\label{thm:b1:complex:roots}
Zij $a = r\,\eu^{\iu\alpha} \neq 0$ en $n \in \N^*$. De vergelijking
$z^n = a$ heeft precies $n$ oplossingen:
\[
z_k = r^{1/n}\, \eu^{\iu\left(\frac{\alpha}{n} +
\frac{2k\pi}{n}\right)},
\qquad k = 0, 1, \dots, n - 1 .
\]
\end{theorem}

\begin{proof}
Schrijf $z = \rho\,\eu^{\iu\theta}$ ($\rho > 0$). Dan geldt $z^n =
\rho^n \eu^{\iu n\theta} = r \eu^{\iu\alpha}$ dan en slechts dan als $\rho^n
= r$ (moduli) en $n\theta \equiv \alpha \pmod{2\pi}$ (argumenten),
d.w.z.\ $\rho = r^{1/n}$ en $\theta = \frac{\alpha}{n} +
\frac{2k\pi}{n}$ voor zekere $k \in \Z$. Het blijft na te gaan wanneer twee
gehele getallen $k, k'$ hetzelfde getal opleveren: dit gebeurt precies wanneer
de hoeken verschillen door een veelvoud van $2\pi$,
\[
\frac{2k\pi}n - \frac{2k'\pi}n \in 2\pi\Z
\iff \frac{k - k'}n \in \Z
\iff n \mid k - k' .
\]
Door euclidische deling is elke $k \in \Z$ congruent mod $n$ met
precies één element van $\{0, 1, \dots, n-1\}$, dus dit bereik somt
elke oplossing eenmaal op en het aantal is precies $n$. (Hetzelfde
argument, uitgevoerd binnen $\mathbb U_n$, toont aan dat de wortels een regelmatige
$n$-hoek vormen: opeenvolgende waarden van $k$ roteren over de vaste hoek
$\frac{2\pi}n$.)
\end{proof}

\begin{example}[Derdemachtswortels van $-27$]\label{ex:b1:complex:cuberoots}
Los $z^3 = -27$ op. Exponentiële vorm van het rechterlid: $-27 =
27\,\eu^{\iu\pi}$, dus de drie wortels zijn
\[
z_k = 3\,\eu^{\iu(\frac\pi3 + \frac{2k\pi}3)}, \quad k = 0, 1, 2 :
\qquad
z_0 = 3\eu^{\iu\pi/3} = \frac32 + \frac{3\sqrt3}2\,\iu,
\quad
z_1 = -3,
\quad
z_2 = \conj{z_0} .
\]
Twee controles. Ten eerste is de reële wortel $-3$ de voor de hand liggende, en de
andere twee zijn zijn rotaties over $\pm\frac{2\pi}3$ --- equivalent
$-3j$ en $-3j^2$. Ten tweede bevestigt de algebra: $z^3 + 27 = (z +
3)(z^2 - 3z + 9)$, en de kwadratische heeft discriminant $9 - 36 =
-27 < 0$ met wortels $\frac{3 \pm 3\iu\sqrt3}2 = z_0, \conj{z_0}$.
Het inzicht: voor reële rechterleden komen de niet-reële wortels altijd
in \omterm{def:b1:complex:field}{toegevoegde} paren, zodat een tekening van de oplossingsverzameling
symmetrisch is om de reële as --- een voorproefje van de reële
factorisatiestelling van \cref{ch:b1:poly}.
\end{example}

\begin{example}[Een niet-reëel rechterlid]\label{ex:b1:complex:fourthroots}
Los $z^4 = -8 + 8\iu\sqrt3$ op. Exponentiële vorm van het rechterlid:
\omterm{def:b1:complex:field}{modulus} $\sqrt{64 + 192} = 16$, argument $\theta$ met
$\cos\theta = -\frac12$, $\sin\theta = \frac{\sqrt3}2$, d.w.z.\
$\theta = \frac{2\pi}3$. De vier wortels zijn
\[
z_k = 2\,\eu^{\iu(\frac\pi6 + \frac{k\pi}2)}, \quad k = 0, 1, 2,
3 :
\qquad
z_0 = \sqrt3 + \iu,\quad z_1 = \iu z_0 = -1 + \iu\sqrt3,
\]
\[
z_2 = -z_0 = -\sqrt3 - \iu,\qquad z_3 = -\iu z_0 = 1 - \iu\sqrt3 .
\]
(Controle: $z_0^2 = 2 + 2\iu\sqrt3$, dus $z_0^4 = (2 +
2\iu\sqrt3)^2 = 4 - 12 + 8\iu\sqrt3 = -8 + 8\iu\sqrt3$.) Zodra
\emph{één} wortel gevonden is, komen de andere drie gratis: het zijn zijn
opeenvolgende rotaties over $\frac\pi2$, d.w.z.\ zijn producten met de
vierde \omterm{def:b1:complex:unity}{eenheidswortels} --- de algemene structuur achter
\cref{thm:b1:complex:roots}, het waard om te benutten vooraleer je
elke wortel opnieuw van nul af aan berekent. Geen \omterm{def:b1:complex:field}{toegevoegde} symmetrie deze keer: het
rechterlid is niet reëel.
\end{example}

\begin{definition}[Eenheidswortels]\label{def:b1:complex:unity}
De \emph{$n$-de eenheidswortels}\index{eenheidswortels} zijn de
oplossingen van $z^n = 1$:
\[
\mathbb{U}_n = \bigl\{\, \omega^k : k = 0, \dots, n-1 \,\bigr\},
\qquad \omega = \eu^{2\iu\pi/n}.
\]
Ze vormen een groep onder de vermenigvuldiging (\cref{ch:b1:structures}) en
liggen op de hoekpunten van een regelmatige $n$-hoek ingeschreven in de eenheidscirkel.
\end{definition}

\begin{proposition}[Som van de eenheidswortels]\label{prop:b1:complex:sumroots}
Voor $n \geq 2$ sommeren de $n$-de \omterm{def:b1:complex:unity}{eenheidswortels} tot nul:
$\sum_{k=0}^{n-1} \omega^k = 0$.
\end{proposition}

\begin{proof}
Meetkundige som met reden $\omega \neq 1$:
$\sum_{k=0}^{n-1} \omega^k = \frac{\omega^n - 1}{\omega - 1} = 0$
aangezien $\omega^n = 1$.
\end{proof}

\begin{example}[Reële en imaginaire delen aflezen]\label{ex:b1:complex:sumrootsparts}
Het opsplitsen van $\sum_{k=0}^{n-1}\omega^k = 0$ in reële en imaginaire
delen levert gratis twee trigonometrische identiteiten op:
\[
\sum_{k=0}^{n-1}\cos\frac{2k\pi}n = 0,
\qquad
\sum_{k=0}^{n-1}\sin\frac{2k\pi}n = 0
\qquad (n \geq 2).
\]
Meetkundig: het zwaartepunt van een regelmatige $n$-hoek ingeschreven in de
eenheidscirkel is haar middelpunt --- de hoekpunten heffen elkaar precies op.
Voor $n = 5$ geeft de eerste identiteit $1 + 2\cos\frac{2\pi}5 +
2\cos\frac{4\pi}5 = 0$ (door $k$ met $n - k$ te paren), het uitgangs-
punt van de berekening van $\cos\frac{2\pi}5$ in \cref{exo:b1:complex:8}.
\end{example}

\begin{example}[Vierkantswortels in algebraïsche vorm]\label{ex:b1:complex:sqrt}
Om $z^2 = 3 + 4\iu$ zonder trigonometrie op te lossen, stel $z = x + \iu y$:
\[
x^2 - y^2 = 3, \qquad 2xy = 4, \qquad x^2 + y^2 = \abs{3 + 4\iu} = 5 .
\]
Optellen van de eerste en de laatste: $x^2 = 4$, dus $x = \pm 2$, dan $y = 2/x =
\pm 1$ met \emph{dezelfde} tekenparing ($xy = 2 > 0$): $z = \pm(2 +
\iu)$. Gecombineerd met de gebruikelijke formule lost dit elke kwadratische
vergelijking met complexe coëfficiënten op
(\cref{exo:b1:complex:7}).
\end{example}

\section{Complexe getallen en vlakke meetkunde}

\begin{proposition}[Meetkundig woordenboek]\label{prop:b1:complex:geometry}
Identificeer het punt $M(x, y)$ van het vlak met zijn \emph{affix} $z = x
+ \iu y$\index{affix}. Voor verschillende punten $A, B, C$ met affixen $a, b,
c$:
\begin{enumerate}
  \item $\abs{b - a}$ is de afstand $AB$;
  \item $\arg\dfrac{c - a}{b - a}$ is de hoek tussen de vectoren
        $\vect{AB}$ en $\vect{AC}$ (mod $2\pi$);
  \item $A, B, C$ liggen op één rechte dan en slechts dan als $\dfrac{c - a}{b - a} \in
        \R$; de rechten $AB$ en $AC$ staan loodrecht dan en slechts dan als
        $\dfrac{c - a}{b - a} \in \iu\R$.
\end{enumerate}
\end{proposition}

\begin{proof}
(1) is de definitie van de \omterm{def:b1:complex:field}{modulus} toegepast op $b - a$, de affix van
$\vect{AB}$. (2): schrijf $b - a = r\eu^{\iu\theta}$, $c - a =
s\eu^{\iu\varphi}$; dan heeft $\frac{c-a}{b-a} = \frac sr
\eu^{\iu(\varphi - \theta)}$ argument $\varphi - \theta$, de
hoek van $\vect{AB}$ naar $\vect{AC}$. (3): op één rechte liggen betekent hoek $0$
of $\pi$, d.w.z.\ argument in $\pi\Z$, d.w.z.\ het quotiënt is reëel;
loodrechtheid betekent hoek $\pm\frac\pi2$, d.w.z.\ het quotiënt is
zuiver imaginair. (Het quotiënt is verschillend van nul aangezien $C \neq A$.)
\end{proof}

\begin{remark}[Intermezzo: $\C$ is het vlak met een vermenigvuldiging]\label{rem:b1:complex:interlude}
Het is de moeite waard even stil te staan bij wat dit hoofdstuk überhaupt mogelijk maakt:
het vlak $\R^2$ draagt optellingen in elke richting, maar geen
door de hemel gegeven vermenigvuldiging --- en $\C$ \emph{is} het vlak
uitgerust met één, waarin vermenigvuldigen met een vast getal
roteert en schaalt. Deze ene structuur zal in dit deel nog drie keer
uitgemolken worden. In \cref{ch:b1:matrices} duikt vermenigvuldiging met
$a + \iu b$ opnieuw op als de $2 \times 2$-matrix met rijen $(a,
-b)$ en $(b, a)$: de complexe rekenkunde is een eerste, volledig
concrete familie van matrixproducten. In \cref{ch:b1:euclid} blijkt de
formule $\Re(\conj z\,w)$ het inwendig product te zijn, en
$\abs z$ de euclidische norm: de driehoeksongelijkheid die hier bewezen is
is het model voor het verhaal van Cauchy--Schwarz daar. En in
\cref{ch:b1:curves} wordt een bewegend punt het best geschreven als $t \mapsto
z(t)$, zodat snelheid en versnelling complexwaardige
afgeleiden worden --- de cirkelbeweging bijvoorbeeld is eenvoudigweg $z(t) =
R\,\eu^{\iu\omega t}$. Eén goede vermenigvuldiging, vier hoofdstukken
dividend.
\end{remark}

\begin{proposition}[Afbeeldingen $z \mapsto az + b$]\label{prop:b1:complex:similitude}
Zij $a \in \C^*$, $b \in \C$. De transformatie $f(z) = az + b$ van
het vlak:
\begin{itemize}
  \item is een translatie als $a = 1$;
  \item heeft anders een uniek vast punt $\zeta = \frac{b}{1-a}$,
        en $f(z) - \zeta = a\,(z - \zeta)$: $f$ is de rotatie met
        middelpunt $\zeta$ en hoek $\arg a$, samengesteld met de schaling
        met middelpunt $\zeta$ en verhouding $\abs a$.
\end{itemize}
In het bijzonder is $z \mapsto \eu^{\iu\theta} z$ de rotatie over hoek
$\theta$ om de oorsprong, en $\conj z$ de spiegeling in de reële
as.
\end{proposition}

\begin{proof}
Als $a = 1$ translateert $f(z) = z + b$ over de vector met affix $b$. Als
$a \neq 1$ heeft de vastpuntvergelijking $z = az + b$ de unieke
oplossing $\zeta = \frac{b}{1-a}$, en dan $f(z) - \zeta = az + b -
(a\zeta + b) = a(z - \zeta)$. Schrijf $a = \abs a\, \eu^{\iu\arg a}$;
vermenigvuldiging met $a$ schaalt afstanden tot $\zeta$ met $\abs a$ en telt
$\arg a$ op bij hoeken in $\zeta$, wat de aangekondigde samenstelling is.
\end{proof}

\begin{example}[Een afbeelding $z \mapsto az+b$ classificeren]\label{ex:b1:complex:classify}
Neem $f(z) = \iu z + 1$. Hier is $a = \iu \neq 1$: het vaste punt is
\[
\zeta = \frac{b}{1 - a} = \frac1{1 - \iu}
= \frac{1 + \iu}{2},
\]
en aangezien $\abs a = 1$ is er helemaal geen schaling: $f$ is de zuivere
rotatie met middelpunt $\frac{1+\iu}2$ en hoek $\arg \iu = \frac\pi2$.
Controle: $f(\zeta) = \iu\,\frac{1+\iu}2 + 1 = \frac{\iu - 1}2 + 1 =
\frac{1 + \iu}2 = \zeta$, en $f(0) = 1$, $f(1) = 1 + \iu$, $f(1 +
\iu) = \iu\,(1 + \iu) + 1 = \iu$: de vier punten $0, 1, 1+\iu, \iu$
van het eenheidsvierkant draaien rond hun middelpunt $\zeta$, telkens een kwart-
slag --- precies wat een rotatie over $\frac\pi2$ om het
middelpunt van het vierkant moet doen.
\end{example}

\begin{remark}[Veelvoorkomende valkuilen met moduli en argumenten]\label{rem:b1:complex:pitfalls}
\begin{enumerate}
  \item \emph{Geen ongelijkheden in $\C$.} $z \leq w$ schrijven voor
        niet-reële getallen is betekenisloos; alleen moduli, reële delen
        en imaginaire delen kunnen vergeleken worden.
  \item \emph{$\sqrt{\phantom z}$ is voorbehouden voor $\R_+$.} Elk
        complex getal verschillend van nul heeft \emph{twee} vierkantswortels en
        geen van beide is bevoorrecht: schrijf ``zij $\delta$ een vierkants-
        wortel van $\Delta$'' (berekend zoals in
        \cref{ex:b1:complex:sqrt}), nooit $\sqrt\Delta$ ---
        de regel $\sqrt{ab} = \sqrt a\sqrt b$ faalt al bij
        $a = b = -1$.
  \item \emph{Argumenten leven modulo $2\pi$.} Uit
        $\eu^{\iu\alpha} = \eu^{\iu\beta}$ besluit je $\alpha
        \equiv \beta \pmod{2\pi}$, niet $\alpha = \beta$;
        de $2k\pi$ vergeten is hoe oplossingsverzamelingen van $z^n = a$
        $n - 1$ van hun $n$ elementen verliezen.
  \item \emph{$\abs{z + w}$ is niet $\abs z + \abs w$.} Gelijkheid
        in de driehoeksongelijkheid is het uitzonderlijke uitgelijnde
        geval (\cref{prop:b1:complex:rules}~(4)); in het algemeen moet de
        \omterm{def:b1:complex:field}{modulus} van een som geschat worden, niet berekend.
\end{enumerate}
\end{remark}

\begin{example}\label{ex:b1:complex:equilateral}
$A, B, C$ (affixen $a, b, c$, paarsgewijs verschillend) vormen een gelijkzijdige
driehoek met de hoekpunten in directe (tegenwijzerzin) volgorde dan en slechts
dan als $\frac{c - a}{b - a} = \eu^{\iu\pi/3}$: de rotatie met
middelpunt $A$ en hoek $\frac\pi3$ stuurt $B$ naar $C$. Beide oriëntaties
samen worden gevat door de symmetrische vergelijking $a^2 + b^2 + c^2 = ab
+ bc + ca$ (\cref{exo:b1:complex:10}).
\end{example}

\begin{remark}[Waar dit hoofdstuk gebruikt wordt]\label{rem:b1:complex:whereused}
De exponentiële vorm is verreweg het meest hergebruikte rekenkundige
hulpmiddel van het deel. \omterm{def:b1:complex:unity}{Eenheidswortels} worden het standaardvoorbeeld van een
cyclische groep in \cref{ch:b1:structures} en drijven de factorisatie
van $X^n - 1$ in \cref{ch:b1:poly}; het groeperen van \omterm{def:b1:complex:field}{toegevoegde} wortels daar
produceert de reële factorisaties die door de partiële breuken in
\cref{ch:b1:fractions} gebruikt worden. Linearisatie (\cref{met:b1:complex:trig})
is de standaardvoorbereiding voor het integreren van trigonometrische machten in
\cref{ch:b1:integration}, en de karakteristieke vergelijkingen van
\cref{ch:b1:diffeq} hebben complexe wortels waarvan de reële en imaginaire
delen de oscillerende oplossingen $\eu^{\lambda t}\cos\omega
t$ produceren. Het meetkundige woordenboek keert terug in matrixgewaad: rotaties
en gelijkvormigheden worden de orthogonale matrices van
\cref{ch:b1:matrices,ch:b1:euclid}, en geparametriseerde krommen in
\cref{ch:b1:curves} worden vaak het best geschreven als \omterm{def:b1:logic:map}{afbeeldingen} $t \mapsto z(t)
\in \C$.
\end{remark}

Zet in exponentiële vorm: $1 + \iu$; $\;\sqrt 3 - \iu$; $\;-5$;

Beschrijf meetkundig de \omterm{def:b1:logic:sets}{verzameling} van $z \in \C$ zodat:
$\abs{z - 2} = \abs{z + \iu}$; $\;\abs{z - 1} = 2$;
$\;\dfrac{z - 1}{z + 1} \in \iu\R$ (voor $z \neq -1$).
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:complex:4}
Lineariseer $\sin^4\theta$, en ontwikkel $\cos 4\theta$ als een veelterm in
$\cos\theta$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:complex:5}
Los $z^3 = 8\iu$ op en plaats de oplossingen op een tekening. Los
$z^4 = -4$ op en factoriseer $X^4 + 4$ in twee reële kwadratische veeltermen.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:complex:6}
Voor $\theta \in \R$ met $\eu^{\iu\theta} \neq 1$ en $n \in \N$,
bereken
\[
C_n = \sum_{k=0}^{n} \cos k\theta
\qquad\text{en}\qquad
S_n = \sum_{k=0}^{n} \sin k\theta
\]
door de meetkundige reeks $\sum_k \eu^{\iu k\theta}$ te sommeren en de
halvehoekontbinding te gebruiken.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:complex:7}
Los op in $\C$: $z^2 - (3 + 4\iu) z + (-1 + 5\iu) = 0$. \emph{(Bereken
de discriminant, trek zijn vierkantswortels zoals in
\cref{ex:b1:complex:sqrt}.)}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:complex:8}
Zij $\omega = \eu^{2\iu\pi/5}$.
\begin{enumerate}
  \item Verantwoord $1 + \omega + \omega^2 + \omega^3 + \omega^4 = 0$.
  \item Stel $u = \omega + \omega^4$ en $v = \omega^2 + \omega^3$.
        Bereken $u + v$ en $uv$, en leid af dat $u$ en $v$ de
        wortels zijn van $X^2 + X - 1$.
  \item Besluit dat $\cos\frac{2\pi}{5} = \frac{\sqrt 5 - 1}{4}$.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:complex:9}
Bewijs dat voor alle $z, w \in \C$ (parallellogramidentiteit):
\[
\abs{z + w}^2 + \abs{z - w}^2 = 2\abs z^2 + 2\abs w^2 ,
\]
en interpreteer haar meetkundig in het parallellogram met hoekpunten $0,
z, w, z + w$.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:complex:10}
Bewijs dat drie paarsgewijs verschillende punten met affixen $a, b, c$ een
gelijkzijdige driehoek (beide oriëntaties) vormen dan en slechts dan als
\[
a^2 + b^2 + c^2 = ab + bc + ca .
\]
\emph{Hint: het directe geval is $\frac{c-a}{b-a} = -j^2$ en het
indirecte geval $\frac{c-a}{b-a} = -j$, waarbij $j = \eu^{2\iu\pi/3}$
voldoet aan $j^2 + j + 1 = 0$; of factoriseer $a + jb + j^2c$ en $a + j^2 b
+ jc$.}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:complex:11}
Voor $n \in \N^*$, bereken $P = \prod_{k=1}^{n-1}
\bigl(1 - \omega^k\bigr)$ waarbij $\omega = \eu^{2\iu\pi/n}$.
\emph{Hint: $X^n - 1 = \prod_{k=0}^{n-1} (X - \omega^k)$; deel door
$X - 1$ en evalueer in $X = 1$.} Leid
$\prod_{k=1}^{n-1} \sin\frac{k\pi}{n} = \dfrac{n}{2^{n-1}}$ af.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:complex:12}
Zij $n \geq 2$. Los de vergelijking $(z + 1)^n = (z - 1)^n$ op in $\C$:
toon dat ze precies $n - 1$ oplossingen heeft, alle zuiver imaginair,
namelijk
\[
z_k = -\iu\,\frac{\cos\frac{k\pi}{n}}{\sin\frac{k\pi}{n}},
\qquad k = 1, \dots, n - 1 .
\]
\emph{Hint: $z = 1$ is geen oplossing, dus deel en gebruik de eenheids-
wortels; pas dan de halvehoekontbinding van
\cref{met:b1:complex:trig} toe.}
\end{exercise}

\section{Probleem: de stelling van Napoleon}

\begin{problem}\label{pb:b1:complex:1}
Richt op elke zijde van een \emph{willekeurige} driehoek een gelijkzijdige
driehoek naar buiten op: de drie middelpunten van die driehoeken
vormen altijd een gelijkzijdige driehoek. Dit is de stelling van Napoleon
\index{stelling van Napoleon} --- een \omterm{def:b1:logic:statement}{uitspraak} zonder zichtbare reden om
waar te zijn, die de algebra van $j = \eu^{2\iu\pi/3}$ bewijst in drie
regels berekening. Dit probleem bouwt de volledige gereedschapskist
(rotaties, directe gelijkvormigheden, het $j$-criterium van
\cref{exo:b1:complex:10}), bewijst de stelling van Napoleon binnen en
buiten, lokaliseert het ontaarde geval, en sluit af met een tweede juweel
van dezelfde school: de stelling van van Schooten over gelijkzijdige driehoeken
en de ongelijkheid van Ptolemaeus. Overal worden punten van het vlak
geïdentificeerd met hun affixen.

\medskip
\noindent\textbf{Deel I --- Het getal $j$ en rotaties.}
\begin{enumerate}
  \item Bereken de algebraïsche vorm van $j$, dan $j^2$, $j^3$, $1 + j
        + j^2$, $\conj j$ en $j^{-1}$. Plaats $1$, $j$, $j^2$ op een
        schets van de eenheidscirkel.
  \item Toon dat de rotatie met middelpunt $a$ en hoek $\theta$
        $r(z) = a + \eu^{\iu\theta}(z - a)$ is, en dat de
        samenstelling van twee rotaties, met hoeken $\theta$ en
        $\theta'$, een rotatie met hoek $\theta + \theta'$ is als
        $\theta + \theta' \notin 2\pi\Z$, en anders een translatie.
        (Gebruik \cref{prop:b1:complex:similitude}.)
  \item Zij $b \neq c$. Toon dat er precies twee punten $p$ zijn
        die $(b, c, p)$ gelijkzijdig maken, namelijk $p = b +
        \eu^{\pm\iu\pi/3}(c - b)$. Ga voor een \emph{directe} driehoek
        $(a, b, c)$ op het voorbeeld $a = 0$, $b = 1$, $c =
        \iu$ na dat de keuze $\eu^{-\iu\pi/3}$ diegene is die aan de
        overkant van de rechte $BC$ ligt ten opzichte van $a$ --- de
        \emph{naar buiten gerichte} top.
  \item Uit de oplossing van \cref{exo:b1:complex:10} is $(a, b, c)$
        direct gelijkzijdig d.e.s.d.a.\ $a + jb + j^2c = 0$. Bewijs de twee
        rotatie-identiteiten
        \[
        b + jc + j^2a = j^2\,(a + jb + j^2c),
        \qquad
        c + ja + j^2b = j\,(a + jb + j^2c),
        \]
        en leid af dat het criterium invariant is onder cyclische
        \omterm{def:b1:counting:objects}{permutatie} van $(a, b, c)$.
  \item Toon dat als $a + jb + j^2c = 0$ en twee van de drie punten
        samenvallen, dan alle drie samenvallen. (Dus het criterium karakteriseert
        precies: directe gelijkzijdige driehoek, of één enkel
        punt.)
\end{enumerate}

\medskip
\noindent\textbf{Deel II --- Directe gelijkvormigheden.}
\begin{enumerate}[resume]
  \item Toon dat de \omterm{def:b1:logic:map}{afbeeldingen} $f(z) = az + b$ met $a \in \C^*$
        (\emph{directe gelijkvormigheden}) stabiel zijn onder samenstelling
        en inversie: de samenstelling van twee ervan, en de
        inverse van elke ervan, is opnieuw van deze vorm. (In de
        taal van \cref{ch:b1:structures}: ze vormen een groep.)
  \item Gegeven $z_1 \neq z_2$ en $w_1 \neq w_2$, toon dat er een
        \emph{unieke} directe gelijkvormigheid $f$ bestaat met $f(z_1) = w_1$ en
        $f(z_2) = w_2$, en geef $a$ en $b$ expliciet.
  \item Toon dat een directe gelijkvormigheid $f(z) = az + b$ alle
        afstanden vermenigvuldigt met $\abs a$ en de \emph{vorm}
        $\frac{c - a'}{b' - a'}$ van elke driehoek $(a', b', c')$
        bewaart \emph{(teller en noemer worden beide vermenigvuldigd met
        $a$)}. Leid af dat twee driehoeken direct gelijkvormig zijn
        precies wanneer hun vormen gelijk zijn.
  \item Bepaal volledig de directe gelijkvormigheid met $f(0) = 1$
        en $f(1) = \iu$: geef $a$, $b$, het vaste punt, de verhouding
        en de hoek.
  \item Zij $\alpha + \beta = 1$. Toon dat het ``gewogen punt''
        $g(a', b') = \alpha a' + \beta b'$ commuteert met elke
        directe gelijkvormigheid: $f(\alpha a' + \beta b') = \alpha f(a')
        + \beta f(b')$. Leid af dat zwaartepunten, middens en de
        hieronder gedefinieerde Napoleon-middelpunten allemaal getransporteerd worden door gelijkvormigheden
        --- de algebraïsche vrijbrief achter elk ``zonder verlies van
        algemeenheid, plaats de omgeschreven cirkel op de eenheidscirkel''-
        argument.
\end{enumerate}

\medskip
\noindent\textbf{Deel III --- De stelling van Napoleon.} Zij $(a, b, c)$
een directe driehoek. Richt op elke zijde de naar buiten gerichte gelijkzijdige
driehoek op (vraag 3) en zij $n_a$, $n_b$, $n_c$ de middelpunten
(zwaartepunten) van de driehoeken opgericht op respectievelijk $[b, c]$, $[c, a]$,
$[a, b]$.
\begin{enumerate}[resume]
  \item Toon dat
        \[
        n_a = \alpha b + \beta c, \qquad
        n_b = \alpha c + \beta a, \qquad
        n_c = \alpha a + \beta b,
        \qquad\text{met}\quad
        \alpha = \frac{3 + \iu\sqrt3}{6},\ \beta = \conj\alpha .
        \]
  \item Controleer $\alpha + \beta = 1$, en leid af dat de driehoek
        $(n_a, n_b, n_c)$ \emph{hetzelfde zwaartepunt} heeft als
        $(a, b, c)$.
  \item Toon met behulp van de identiteiten van vraag 4 dat
        \[
        n_a + j\,n_b + j^2 n_c
        = (\alpha j^2 + \beta j)\,(a + jb + j^2c) .
        \]
  \item Bereken $\alpha j + \beta$ en besluit: $\alpha j^2 +
        \beta j = j(\alpha j + \beta) = 0$, dus $n_a + j n_b + j^2
        n_c = 0$ voor \emph{elke} driehoek: de uitwendige Napoleon-
        driehoek is direct gelijkzijdig (of een punt). De stelling van Napoleon
        is bewezen.
  \item Richt de gelijkzijdige driehoeken in plaats daarvan \emph{naar binnen} op
        (keuze $\eu^{+\iu\pi/3}$ in vraag 3) en zij $m_a, m_b,
        m_c$ hun middelpunten. Toon $m_a = \beta b + \alpha c$ (en
        cyclisch), bewijs dan $m_a + j^2 m_b + j\,m_c = 0$: de
        inwendige Napoleon-driehoek is ook gelijkzijdig, met de
        tegengestelde oriëntatie.
  \item Lokaliseer de ontaarding: toon dat $n_a = n_b = n_c$ zich
        precies voordoet wanneer $a + j^2 b + jc = 0$, d.w.z.\ wanneer $(a, b, c)$
        een \emph{indirecte} gelijkzijdige driehoek is. \emph{(Twee
        punten onder $n_a, n_b, n_c$ vallen samen d.e.s.d.a.\ alle drie samenvallen, volgens
        vraag 5; gebruik dan $n_a + j^2 n_b + j n_c = j(\alpha +
        \beta j)(a + j^2b + jc)$ en controleer $\alpha + \beta j \neq
        0$.)}
  \item Voer de hele berekening uit op de driehoek $a = 0$,
        $b = 1$, $c = \iu$: geef $n_a, n_b, n_c$ exact, en
        verifieer door directe berekening van de drie gekwadrateerde
        zijdelengten dat de driehoek gelijkzijdig is, met gekwadrateerde zijde
        $\frac{2 + \sqrt3}{3}$.
\end{enumerate}

\medskip
\noindent\textbf{Deel IV --- Ptolemaeus en van Schooten.}
\begin{enumerate}[resume]
  \item Bewijs de identiteit, geldig voor alle complexe $a, b, c, d$:
        \[
        (a - b)(c - d) + (a - d)(b - c) = (a - c)(b - d) .
        \]
  \item Leid de \emph{ongelijkheid van Ptolemaeus}\index{ongelijkheid van
        Ptolemaeus} af: voor elke vier punten $A, B, C, D$,
        \[
        AC \cdot BD \;\leq\; AB \cdot CD + AD \cdot BC ,
        \]
        met gelijkheid dan en slechts dan als $(a-b)(c-d)$ en
        $(a-d)(b-c)$ op een gemeenschappelijke halfrechte vanuit $0$ liggen.
  \item Toon dat voor $\theta, \varphi \in \R$,
        $\abs{\eu^{\iu\theta} - \eu^{\iu\varphi}} =
        2\,\abs{\sin\frac{\theta - \varphi}2}$.
  \item (stelling van van Schooten\index{stelling van van Schooten}) Zij
        $(a, b, c) = (1, j, j^2)$ --- volgens Deel II verliest dit geen
        algemeenheid onder de directe gelijkzijdige driehoeken --- en zij
        $p = \eu^{\iu\theta}$ met $\theta \in
        \intoo{2\pi/3}{4\pi/3}$, een punt van de omgeschreven cirkel op de
        boog $BC$ die $A$ niet bevat. Bewijs
        \[
        PA = PB + PC .
        \]
        \emph{(Druk de drie afstanden uit met vraag 20 en gebruik
        een som-naar-product-formule.)}
  \item Verifieer bij $p = -1$ dat dit precies het gelijkheidsgeval
        is van de ongelijkheid van Ptolemaeus voor de cyclische volgorde $A, B, P, C$:
        bereken $(a - b)(p - c)$ en $(a - c)(b - p)$ en controleer
        dat hun verhouding een positief reëel getal is.
\end{enumerate}

\medskip
\noindent\textbf{Deel V --- Synthese.}
\begin{enumerate}[resume]
  \item Bereken de uitwendige Napoleon-driehoek van de gelijkzijdige
        driehoek $(1, j, j^2)$ zelf, en beschrijf het resultaat
        meetkundig.
  \item Waar precies gebruikte het probleem: (i) vermenigvuldiging als
        rotatie; (ii) de groepsstructuur van gelijkvormigheden en de
        vorminvariant; (iii) de halvehoekontbinding van
        \cref{met:b1:complex:trig}? Eén zin elk.
  \item Formuleer in een korte paragraaf de moraal van het probleem:
        wat koopt het woordenboek tussen vlakke meetkunde en de
        algebra van $\C$ je op, wat kost het, en welke van de
        twee bewijsstappen --- de identiteit $\alpha j + \beta = 0$ of
        de klassieke tekening --- \emph{verklaart} de stelling van Napoleon
        naar jouw mening beter? Vermeld één plaats waar het
        woordenboek later in dit deel in matrixvorm zal terugkeren.
\end{enumerate}
\end{problem}
