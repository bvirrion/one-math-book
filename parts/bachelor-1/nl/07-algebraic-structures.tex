Dezelfde rekenregels duiken telkens weer op: gehele getallen, reële
getallen, complexe getallen, congruentieklassen, en straks veeltermen

\section{Bewerkingen}

\begin{definition}\label{def:b1:structures:law}
Een \emph{bewerking}\index{bewerking} op een \omterm{def:b1:logic:sets}{verzameling} $E$ is een
\omterm{def:b1:logic:map}{afbeelding} $E \times E \to E$, geschreven $(x, y) \mapsto x * y$. Ze is
\emph{associatief} wanneer steeds $(x*y)*z = x*(y*z)$, \emph{commutatief}
wanneer steeds $x * y = y * x$. Een element $e$ is een \emph{neutraal
element} wanneer $e * x = x * e = x$ voor alle $x$; dan is $x'$ een
\emph{inverse} van $x$ wanneer $x * x' = x' * x = e$.
\end{definition}

\begin{proposition}[Uniciteit]\label{prop:b1:structures:unique}
Een bewerking heeft hoogstens één neutraal element; voor een
associatieve bewerking met neutraal element heeft elk element hoogstens
één inverse.
\end{proposition}

\begin{proof}
Als $e$ en $e'$ neutrale elementen zijn: $e = e * e' = e'$. Als $x'$ en
$x''$ inversen van $x$ zijn: $x' = x' * e = x' * (x * x'') = (x' * x) *
x'' = e * x'' = x''$.
\end{proof}

\section{Groepen}

\begin{definition}[Groep]\label{def:b1:structures:group}
Een \emph{groep}\index{groep} $(G, *)$ is een \omterm{def:b1:logic:sets}{verzameling} met een
associatieve bewerking die een neutraal element bezit en waarin elk
element een inverse heeft. De groep is \emph{abels}\index{abelse groep}
wanneer de bewerking commutatief is.
\end{definition}

\begin{example}\label{ex:b1:structures:groups}
$(\Z, +)$, $(\Q, +)$, $(\R, +)$, $(\C, +)$; $(\Q^*, \times)$,
$(\R^*, \times)$, $(\C^*, \times)$, $(\mathbb{U}_n, \times)$
(\omterm{def:b1:complex:unity}{eenheidswortels}, \cref{def:b1:complex:unity}); de \omterm{def:b1:logic:sets}{verzameling}
$\mathfrak{S}(E)$ van bijecties van een \omterm{def:b1:logic:sets}{verzameling} $E$ op zichzelf,
onder samenstelling --- de \emph{symmetrische groep}\index{symmetrische
groep} van $E$, \omterm{def:b1:structures:group}{niet-abels} zodra $\abs E \geq 3$. Geen \omterm{def:b1:structures:group}{groepen}:
$(\N, +)$ (geen inversen), $(\Z, \times)$ (enkel $\pm 1$ inverteerbaar).
\end{example}

\begin{proposition}[Rekenregels]\label{prop:b1:structures:rules}
In een \omterm{def:b1:structures:group}{groep} $G$ (multiplicatief genoteerd, neutraal element $e$):
\begin{enumerate}
  \item schrappingsregel: $ax = ay \implies x = y$ en $xa = ya \implies
        x = y$;
  \item $(ab)^{-1} = b^{-1} a^{-1}$ en $(a^{-1})^{-1} = a$;
  \item voor $a, b \in G$ heeft elke vergelijking $ax = b$ en $xa = b$
        een unieke oplossing ($x = a^{-1}b$, resp.\ $x = b a^{-1}$).
\end{enumerate}
\end{proposition}

\begin{proof}
(1) Vermenigvuldig met $a^{-1}$ aan de gepaste kant, met gebruik van
associativiteit. (2) $(b^{-1}a^{-1})(ab) = b^{-1}(a^{-1}a)b = b^{-1}b =
e$ en symmetrisch; de uniciteit van de inverse besluit; het tweede punt
is \cref{prop:b1:structures:unique} toegepast op $a^{-1}$. (3)
Substitueer en gebruik (1) voor de uniciteit.
\end{proof}

\begin{example}[De symmetrieën van een rechthoek]\label{ex:b1:structures:klein}
Een (niet-vierkante) rechthoek laat precies vier isometrieën op
zichzelf toe: de identiteit $e$, de spiegeling $h$ om de horizontale
as, de spiegeling $v$ om de verticale as, en de halve draai $r$ om
het middelpunt. Samenstelling maakt van deze \omterm{def:b1:logic:sets}{verzameling} met vier
elementen een \omterm{def:b1:structures:group}{groep}: elk element is zijn eigen inverse ($h^2 = v^2 =
r^2 = e$), en het product van twee verschillende niet-neutrale
elementen is het derde ($hv = vh = r$: spiegelen om beide assen is de
halve draai). De volledige tabel is symmetrisch, dus de \omterm{def:b1:structures:group}{groep} is
\omterm{def:b1:structures:group}{abels} --- toch is het \emph{niet} dezelfde \omterm{def:b1:structures:group}{groep} als de rotaties
$\mathbb U_4$ uit \cref{ex:b1:structures:order}: daar heeft $\iu$
\omterm{def:b1:structures:order}{orde} $4$, terwijl hier elk element \omterm{def:b1:structures:order}{orde} $\leq 2$ heeft. Twee \omterm{def:b1:structures:group}{groepen}
van dezelfde grootte kunnen dus werkelijk verschillende
vermenigvuldigingsstructuren hebben --- de figuur hieronder toont
beide tabellen naast elkaar. Deze \omterm{def:b1:structures:group}{groep} met vier elementen keert
terug als $\{\pm1\} \times \{\pm1\}$, en
\cref{exo:b1:structures:7} verklaart waarom elke \omterm{def:b1:structures:group}{groep} waarin alle
kwadraten triviaal zijn, net als deze, \omterm{def:b1:structures:group}{abels} moet zijn.
\end{example}

\begin{definition}[Deelgroep]\label{def:b1:structures:subgroup}
Een deelverzameling $H$ van een \omterm{def:b1:structures:group}{groep} $G$ is een
\emph{deelgroep}\index{deelgroep} (genoteerd $H \leq G$) wanneer ze $e$
bevat, stabiel is onder de bewerking en onder inversie. Dan is $H$ zelf
een \omterm{def:b1:structures:group}{groep}.

\emph{Criterium:} een niet-lege $H \subseteq G$ is een deelgroep als en
slechts als
\[
\forall x, y \in H, \quad x y^{-1} \in H .
\]
\end{definition}

\begin{proof}[Bewijs van het criterium]
Een \omterm{def:b1:structures:subgroup}{deelgroep} voldoet er uiteraard aan. Omgekeerd, zij $H \neq \emptyset$
die eraan voldoet, en kies $x_0 \in H$. Dan $e = x_0 x_0^{-1} \in H$;
voor $y \in H$ geldt $y^{-1} = e\,y^{-1} \in H$; en voor $x, y \in H$
geldt $xy = x (y^{-1})^{-1} \in H$.
\end{proof}

\begin{example}\label{ex:b1:structures:subgroups}
$\mathbb{U}_n \leq (\C^*, \times)$: niet-leeg, en voor $z, w \in
\mathbb{U}_n$ geldt $(zw^{-1})^n = z^n (w^n)^{-1} = 1$. De \omterm{def:b1:structures:subgroup}{deelgroepen}
van $(\Z, +)$ zijn precies de $n\Z$ (bewezen in
\cref{thm:b1:arith:gcd}). Een doorsnede van \omterm{def:b1:structures:subgroup}{deelgroepen} is altijd een
\omterm{def:b1:structures:subgroup}{deelgroep}, maar een unie bijna nooit
(\cref{exo:b1:structures:6}).
\end{example}

\begin{remark}[Veelgemaakte fouten met structuren]\label{rem:b1:structures:pitfalls}
\begin{enumerate}
  \item \emph{Stabiliteit onder de bewerking volstaat niet.} $\N$ is
        stabiel onder optelling binnen $\Z$, bevat $0$, en is toch
        geen \omterm{def:b1:structures:subgroup}{deelgroep}: de inversen ontbreken. Het criterium
        $xy^{-1} \in H$ test alles tegelijk --- maar pas nadat men
        $H \neq \emptyset$ heeft nagegaan.
  \item \emph{Niet-abelse reflexen.} In een algemene \omterm{def:b1:structures:group}{groep} is $(ab)^2 =
        abab$, wat enkel $a^2b^2$ is wanneer $a$ en $b$ commuteren;
        evenzo $(ab)^{-1} = b^{-1}a^{-1}$, met omgekeerde volgorde.
        Elke identiteit die uit de schoolalgebra is overgenomen, moet
        opnieuw uit de axioma's worden afgeleid of als commutatief
        worden gemarkeerd.
  \item \emph{\omterm{def:b1:structures:morphism}{Kern} versus beeld.} $\ker f$ leeft in de \emph{bron},
        $\operatorname{im} f$ in het doel;
        ``$f$ \omterm{def:b1:logic:inj}{injectief} als en slechts als $\ker f$ triviaal''
        (\cref{prop:b1:structures:kernel}) heeft geen analogon met
        het beeld ($\operatorname{im} f = G'$ is surjectiviteit).
  \item \emph{\omterm{def:b1:structures:ring}{Ringen} zijn geen \omterm{def:b1:structures:group}{groepen} voor $\times$.} In een \omterm{def:b1:structures:ring}{ring}
        hoeven de meeste elementen niet inverteerbaar te zijn, en
        schrappen door $a$ vereist dat $a$ een eenheid is of dat de
        \omterm{def:b1:structures:ring}{ring} een \omterm{def:b1:structures:field}{integriteitsdomein} is: in $\Z/12\Z$ geldt
        $\overline3\,\overline2 = \overline3\,\overline6$ terwijl
        $\overline2 \neq \overline6$
        (\cref{ex:b1:structures:zncomputation}).
\end{enumerate}
\end{remark}

\begin{definition}[Groepsmorfisme]\label{def:b1:structures:morphism}
Zij $(G, *)$ en $(G', \star)$ \omterm{def:b1:structures:group}{groepen}. Een \omterm{def:b1:logic:map}{afbeelding} $f \colon G \to G'$
is een \emph{morfisme}\index{morfisme!van groepen} wanneer
\[
\forall x, y \in G, \qquad f(x * y) = f(x) \star f(y).
\]
Dan is $f(e_G) = e_{G'}$ en $f(x^{-1}) = f(x)^{-1}$. De
\emph{kern}\index{kern} en het \emph{beeld} van $f$ zijn
\[
\ker f = f^{-1}(\{e_{G'}\}) \leq G,
\qquad
\operatorname{im} f = f(G) \leq G' .
\]
Een \omterm{def:b1:logic:inj}{bijectief} morfisme is een \emph{isomorfisme}; de inverse \omterm{def:b1:logic:map}{afbeelding}
is dan automatisch een morfisme.
\end{definition}

\begin{proof}[Bewijs van de beweringen]
$f(e) = f(e * e) = f(e)\star f(e)$, en $f(e)$ schrappen geeft
$e_{G'} = f(e)$. Vervolgens identificeert $f(x)\star f(x^{-1}) = f(x
x^{-1}) = e_{G'}$ het element $f(x^{-1})$ als de inverse. \omterm{def:b1:structures:morphism}{Kern}: $e \in
\ker f$; als $x, y \in \ker f$, dan $f(xy^{-1}) = f(x)f(y)^{-1} = e$;
het criterium is van toepassing. Beeld: hetzelfde criterium met
$f(x)f(y)^{-1} = f(xy^{-1})$. Inverse van een isomorfisme: voor $u, v
\in G'$ schrijf $u = f(x)$, $v = f(y)$; dan $f^{-1}(u \star v) =
f^{-1}(f(xy)) = xy = f^{-1}(u) f^{-1}(v)$.
\end{proof}

\begin{proposition}[Injectiviteit via de kern]\label{prop:b1:structures:kernel}
Een groepsmorfisme $f$ is \omterm{def:b1:logic:inj}{injectief} als en slechts als $\ker f =
\{e\}$.
\end{proposition}

\begin{proof}
Als $f$ \omterm{def:b1:logic:inj}{injectief} is, kan $\ker f$ enkel het ene \omterm{def:b1:logic:map}{origineel} van
$e_{G'}$ bevatten, namelijk $e$. Omgekeerd, als $\ker f = \{e\}$ en
$f(x) = f(y)$, dan $f(xy^{-1}) = f(x) f(y)^{-1} = e_{G'}$, dus $xy^{-1}
= e$, d.w.z.\ $x = y$.
\end{proof}

\begin{example}\label{ex:b1:structures:expmorphism}
$\exp \colon (\R, +) \to (\R_+^*, \times)$ is een \omterm{def:b1:structures:morphism}{morfisme}
($\eu^{x+y} = \eu^x \eu^y$), \omterm{def:b1:logic:inj}{bijectief}
(\cref{prop:b1:functions:expln}): de additieve en multiplicatieve
structuren zijn isomorf --- de historische raison d'être van de
logaritmen. Een ander \omterm{def:b1:structures:morphism}{morfisme}: $\theta \mapsto \eu^{\iu\theta}$ van
$(\R, +)$ op de eenheidscirkel $(\mathbb{U}, \times)$, met \omterm{def:b1:structures:morphism}{kern}
$2\pi\Z$.
\end{example}

\begin{example}[Het tekenmorfisme]\label{ex:b1:structures:signmorphism}
De \omterm{def:b1:logic:map}{afbeelding} $s \colon (\R^*, \times) \to (\{\pm1\}, \times)$ die
$x$ naar zijn teken stuurt, is een \omterm{def:b1:structures:morphism}{morfisme}: het teken van een
product is het product van de tekens. De \omterm{def:b1:structures:morphism}{kern} is $\intoo0{+\infty}$
(een \omterm{def:b1:structures:subgroup}{deelgroep}, zoals \cref{def:b1:structures:morphism} belooft), het
beeld heel $\{\pm1\}$: \omterm{def:b1:logic:inj}{surjectief}, massaal \omterm{def:b1:logic:inj}{niet-injectief}.
Twee algemene lessen in het klein. Ten eerste kan een \omterm{def:b1:structures:morphism}{morfisme}
informatie pletten: $s$ onthoudt van $x$ niets dan één bit, en dat
is zijn deugd --- tekenargumenten zijn precies de berekeningen
die door $s$ factoriseren. Ten tweede zijn \omterm{def:b1:structures:morphism}{morfismen} naar $\{\pm1\}$
de eenvoudigste ``invarianten'': de signatuur van \omterm{def:b1:counting:objects}{permutaties},
opgebouwd in het weekendprobleem van dit hoofdstuk, is hetzelfde
fenomeen op de \omterm{def:b1:structures:group}{groep} $\mathfrak S_n$, en de pariteitsargumenten die
ze aandrijft, dalen allemaal af via zo'n tweewaardig \omterm{def:b1:structures:morphism}{morfisme}.
\end{example}

\begin{definition}[Machten, orde van een element]\label{def:b1:structures:order}
In een \omterm{def:b1:structures:group}{groep} $G$ (multiplicatieve notatie), stel $x^0 = e$, $x^{k+1} =
x^k x$ en $x^{-k} = (x^k)^{-1}$ voor $k \in \N$; dan geldt $x^{k+l} =
x^k x^l$ voor alle $k, l \in \Z$, dus $k \mapsto x^k$ is een \omterm{def:b1:structures:morphism}{morfisme}
$(\Z, +) \to G$ waarvan het beeld $\langle x \rangle = \{x^k : k \in
\Z\}$ een \omterm{def:b1:structures:subgroup}{deelgroep} is, de \omterm{def:b1:structures:subgroup}{deelgroep} \emph{voortgebracht} door $x$. De
\emph{orde}\index{orde van een element} van $x$ is de kleinste $m \geq
1$ met $x^m = e$ als die bestaat (dan heeft $\langle x\rangle = \{e, x,
\dots, x^{m-1}\}$ precies $m$ elementen, en $x^k = e \iff m \mid k$), en
$\infty$ anders.
\end{definition}

\begin{example}\label{ex:b1:structures:order}
In $(\C^*, \times)$: $\iu$ heeft \omterm{def:b1:structures:order}{orde} $4$, met $\langle \iu \rangle
= \{1, \iu, -1, -\iu\} = \mathbb{U}_4$; algemener heeft $\omega =
\eu^{2\iu\pi/n}$ \omterm{def:b1:structures:order}{orde} $n$ en $\langle\omega\rangle =
\mathbb{U}_n$. In $(\Z, +)$ heeft elke $x \neq 0$ oneindige \omterm{def:b1:structures:order}{orde}.
Waarom de beweringen in de definitie gelden: als $x$ \omterm{def:b1:structures:order}{orde} $m$ heeft,
deel dan een willekeurige $k$ door $m$ ($k = mq + r$, $0 \leq r < m$,
\cref{thm:b1:arith:division}): $x^k = (x^m)^q x^r = x^r$, dus de
machten doorlopen een cyclus met periode $m$, de opgesomde elementen
zijn paarsgewijs verschillend wegens de minimaliteit van $m$, en $x^k
= e$ dwingt $r = 0$ af. \omterm{def:b1:structures:order}{Orden} van \omterm{def:b1:counting:objects}{permutaties} worden berekend in het
weekendprobleem hieronder.
\end{example}

\begin{example}[Orden binnen $\mathbb U_{12}$]\label{ex:b1:structures:orderU12}
Wat is de \omterm{def:b1:structures:order}{orde} van $\omega^k$ in $\mathbb U_n$, voor $\omega =
\eu^{2\iu\pi/n}$? Men heeft $(\omega^k)^m = 1$ als en slechts als $n
\mid km$, en met $d = \gcd(n, k)$, $n = dn'$, $k = dk'$ met $\gcd(n',
k') = 1$: $n \mid km \iff n' \mid k'm \iff n' \mid m$ (lemma van
Gauss, \cref{thm:b1:arith:gauss}). De kleinste zulke $m \geq 1$ is
$n' = \frac{n}{\gcd(n,k)}$. In $\mathbb U_{12}$ bijvoorbeeld heeft
$\omega^8$ \omterm{def:b1:structures:order}{orde} $\frac{12}{\gcd(12,8)} = 3$ (inderdaad
$\omega^8 = \eu^{4\iu\pi/3} \in \mathbb U_3$), terwijl $\omega^5$
\omterm{def:b1:structures:order}{orde} $12$ heeft: ze brengt de hele \omterm{def:b1:structures:group}{groep} voort, hoewel ze niet
de ``standaard'' voortbrenger is. Het tellen van de voortbrengers ---
de $k$ met $\gcd(k, n) = 1$ --- geeft de tellingen van \omterm{cor:b1:arith:bezout}{relatief
priemen} uit \cref{ex:b1:counting:coprime} terug: groepentheorie en
tellen ontmoeten elkaar.
\end{example}

\section{Ringen en lichamen}

\begin{definition}[Ring]\label{def:b1:structures:ring}
Een \emph{ring}\index{ring} $(A, +, \times)$ is een \omterm{def:b1:logic:sets}{verzameling} met twee
bewerkingen zodat: $(A, +)$ een \omterm{def:b1:structures:group}{abelse groep} is (neutraal element $0$);
$\times$ associatief is met een neutraal element $1$; en $\times$ langs
beide kanten distribueert over $+$. De ring is \emph{commutatief}
wanneer $\times$ dat is. Een element $a$ is \emph{inverteerbaar} (een
\emph{eenheid}) wanneer $ab = ba = 1$ voor een zekere $b$; de eenheden
vormen een \omterm{def:b1:structures:group}{groep} $(A^\times, \times)$.
\end{definition}

\begin{proof}[Bewijs dat de eenheden een groep vormen]
Stabiliteit: als $a, a'$ eenheden zijn met inversen $b, b'$, dan
\[
(aa')(b'b) = a(a'b')b = a\,1\,b = ab = 1,
\qquad (b'b)(aa') = 1
\]
symmetrisch, dus $aa'$ is een eenheid. Het element $1$ is een eenheid
(zijn eigen inverse), de associativiteit wordt overgeërfd van $A$, en
de inverse $b$ van een eenheid $a$ is zelf een eenheid (met inverse
$a$). Dus $(A^\times, \times)$ voldoet aan alle groepsaxioma's. Elke
\omterm{def:b1:structures:group}{groep} in dit boek die niet uit \omterm{def:b1:counting:objects}{permutaties} is opgebouwd, ontstaat op
deze manier: $\Q^* = \Q^\times$, $\R^*$, $\C^*$, de eenheden van
$\Z/n\Z$ hieronder, en later de inverteerbare matrices
(\cref{ch:b1:matrices}).
\end{proof}

\begin{example}\label{ex:b1:structures:rings}
$\Z, \Q, \R, \C$ zijn commutatieve \omterm{def:b1:structures:ring}{ringen}; $\Z^\times = \{1, -1\}$,
$\Q^\times = \Q^*$. Later: veeltermringen $K[X]$
(\cref{ch:b1:poly}), matrixringen (niet-commutatief,
\cref{ch:b1:matrices}), en $\Z/n\Z$ hieronder. In elke \omterm{def:b1:structures:ring}{ring} geldt $0
\times a = 0$ (uit de distributiviteit: $0a = (0+0)a = 0a + 0a$), en
$(-1)a = -a$.
\end{example}

\begin{example}[Idempotenten: nieuwe fenomenen in nieuwe ringen]\label{ex:b1:structures:idempotents}
In $\Z$ heeft de vergelijking $x^2 = x$, d.w.z.\ $x(x - 1) = 0$, enkel
de oplossingen $0$ en $1$. In $\Z/6\Z$, met alle klassen getest:
$\overline0^2 = \overline0$, $\overline1^2 = \overline1$,
$\overline3^2 = \overline9 = \overline3$ en $\overline4^2 =
\overline{16} = \overline4$ --- \emph{vier} idempotenten. De twee
exotische komen van nuldelers: $\overline3\,(\overline3 -
\overline1) = \overline3 \times \overline2 = \overline6 =
\overline0$ zonder dat een van beide factoren nul is. Zulke
berekeningen ijken de intuïtie: vertrouwde feiten over vergelijkingen
overleven in \omterm{def:b1:structures:field}{integriteitsdomeinen} en lichamen, maar een algemene \omterm{def:b1:structures:ring}{ring}
kan zich anders gedragen, en doet dat ook --- zie ook de Booleaanse
\omterm{def:b1:structures:ring}{ringen} van \cref{exo:b1:structures:10}, waar \emph{elk} element
idempotent is.
\end{example}

\begin{proposition}[Binomium van Newton in een commutatieve ring]\label{prop:b1:structures:binomial}
Als $a, b$ elementen zijn van een commutatieve \omterm{def:b1:structures:ring}{ring} (algemener, als
$ab = ba$), dan geldt voor $n \in \N$:
\[
(a+b)^n = \sum_{k=0}^n \binom nk a^k b^{n-k},
\qquad
a^n - b^n = (a - b) \sum_{k=0}^{n-1} a^k b^{\,n-1-k} .
\]
\end{proposition}

\begin{proof}
De bewijzen van \cref{thm:b1:counting:binomial} en van de meetkundige
identiteit gebruiken enkel associativiteit, commutativiteit van de twee
elementen, en distributiviteit --- ze zijn woordelijk van toepassing.
\end{proof}

\begin{example}[Het binomium van Newton in een onvertrouwde ring]\label{ex:b1:structures:binomialring}
Twee snelle opbrengsten van de algemeenheid. In $\Z/p\Z$ ($p$ \omterm{def:b1:arith:prime}{priem})
verdwijnen de middelste \omterm{def:b1:counting:objects}{binomiaalcoëfficiënten}
(de eerste stap van \cref{thm:b1:arith:fermat}), zodat de stelling
ineenstort tot de \emph{freshman's dream}
\[
(a + b)^p = a^p + b^p \qquad \text{in } \Z/p\Z ,
\]
daar een echte identiteit, hoe misdadig ze ook oogt over $\R$.
En in elke commutatieve \omterm{def:b1:structures:ring}{ring} die een element $\varepsilon$
met $\varepsilon^2 = 0$ bevat, wordt de stelling afgekapt:
$(a + \varepsilon)^n = a^n + n\,a^{n-1}\varepsilon$, waarbij alle
hogere termen een factor $\varepsilon^2 = 0$ dragen. De coëfficiënt
$n\,a^{n-1}$ van $\varepsilon$ is de afgeleide van $x^n$ ---
geen toeval, en een eerste hint dat afgeleiden evenzeer algebra als
analyse zijn (vergelijk de formele afgeleide van
\cref{ch:b1:poly}).
\end{example}

\begin{definition}[Integriteitsdomein, lichaam]\label{def:b1:structures:field}
Een commutatieve \omterm{def:b1:structures:ring}{ring} $A \neq \{0\}$ is een \emph{integriteitsdomein}\index{integriteitsdomein}
wanneer ze geen nuldelers heeft: $ab = 0
\implies a = 0$ of $b = 0$. Ze is een \emph{lichaam}\index{lichaam}
wanneer elk niet-nul element inverteerbaar is. Elk lichaam is een
integriteitsdomein ($ab = 0$ en $a \neq 0$ geven $b = a^{-1}ab = 0$).
\end{definition}

\begin{example}
$\Q$, $\R$, $\C$ zijn lichamen; $\Z$ is een \omterm{def:b1:structures:field}{integriteitsdomein} maar
geen \omterm{def:b1:structures:field}{lichaam}. In een \omterm{def:b1:structures:field}{integriteitsdomein} geldt de schrappingsregel voor
$\times$: $ab = ac$ en $a \neq 0$ impliceren $b = c$.
\end{example}

\section{De ring $\Z/n\Z$}

\begin{definition}\label{def:b1:structures:zn}
Fixeer $n \in \N^*$. De congruentieklassen mod $n$
(\cref{ex:b1:logic:congruence}) vormen een \omterm{def:b1:logic:sets}{verzameling}
$\Z/n\Z$\index{Z/nZ@$\Z/n\Z$} van $n$ elementen, geschreven $\overline 0,
\overline 1, \dots, \overline{n-1}$. De bewerkingen
\[
\overline a + \overline b = \overline{a + b},
\qquad
\overline a \times \overline b = \overline{ab}
\]
zijn goed gedefinieerd --- de klassen van de resultaten hangen niet af
van de representanten, juist omdat \omterm{def:b1:arith:congruence}{congruentie} verenigbaar is met $+$
en $\times$ (\cref{def:b1:arith:congruence}) --- en maken van $\Z/n\Z$
een commutatieve \omterm{def:b1:structures:ring}{ring}.
\end{definition}

\begin{theorem}[Eenheden van $\Z/n\Z$; de lichamen $\Z/p\Z$]\label{thm:b1:structures:zp}
\begin{enumerate}
  \item $\overline a$ is inverteerbaar in $\Z/n\Z$ als en slechts als
        $\gcd(a, n) = 1$.
  \item $\Z/n\Z$ is een \omterm{def:b1:structures:field}{lichaam} als en slechts als $n$ \omterm{def:b1:arith:prime}{priem} is.
\end{enumerate}
\end{theorem}

\begin{proof}
(1) is \cref{prop:b1:arith:invmod} herschreven met klassen.

(2) Als $n = p$ \omterm{def:b1:arith:prime}{priem} is, heeft elke $\overline a \neq \overline 0$ dat
$p \nmid a$, dus $\gcd(a, p) = 1$: inverteerbaar volgens (1) --- een
\omterm{def:b1:structures:field}{lichaam}. Als $n = ab$ met $1 < a, b < n$, dan $\overline a\, \overline b
= \overline n = \overline 0$ met $\overline a, \overline b \neq
\overline 0$: nuldelers, dus zelfs geen \omterm{def:b1:structures:field}{integriteitsdomein}; en $n = 1$
geeft de nulring, uitgesloten.
\end{proof}

\begin{example}[Hoeveel vierkantswortels van $1$?]\label{ex:b1:structures:sqrtone}
Los $x^2 = \overline 1$ op in $\Z/8\Z$ en in $\Z/7\Z$. Met de acht
klassen mod $8$ getest: $1^2 = 1$, $3^2 = 9 \equiv 1$, $5^2 = 25
\equiv 1$, $7^2 = 49 \equiv 1$ --- \emph{vier} oplossingen
$\{\overline1, \overline3, \overline5, \overline7\}$, hoewel de
veelterm $X^2 - 1$ graad $2$ heeft. In het \omterm{def:b1:structures:field}{lichaam} $\Z/7\Z$
daarentegen betekent $x^2 = \overline1$ dat $(x - \overline1)(x +
\overline1) = \overline0$, en een \omterm{def:b1:structures:field}{lichaam} heeft geen nuldelers: $x =
\pm\overline1$, slechts twee oplossingen. Het falen mod $8$ is
naspeurbaar: $(3-1)(3+1) = 2 \times 4 = 8 \equiv 0$ zonder dat een
van beide factoren verdwijnt. Moraal: de vertrouwde regel ``een
vergelijking van graad $d$ heeft hoogstens $d$ wortels'' is een
stelling over \emph{\omterm{def:b1:structures:field}{integriteitsdomeinen}} (\cref{cor:b1:poly:nroots}
bewijst ze over lichamen); in \omterm{def:b1:structures:ring}{ringen} met nuldelers faalt ze
stilzwijgend --- wat precies de reden is waarom het koppelingsbewijs
van de stelling van Wilson (\cref{exo:b1:arith:11}) $p$ \omterm{def:b1:arith:prime}{priem} nodig
had.
\end{example}

\begin{example}[Rekenen in $\Z/n\Z$]\label{ex:b1:structures:zncomputation}
In $\Z/12\Z$: de eenheden zijn $\overline 1, \overline 5, \overline 7,
\overline{11}$ (de klassen \omterm{cor:b1:arith:bezout}{relatief priem} met $12$), en elk is zijn
eigen inverse ($5^2 = 25 \equiv 1$, $7^2 = 49 \equiv 1$, $11^2 = 121
\equiv 1$). De vergelijking $\overline 3\, x = \overline 6$ heeft
\emph{drie} oplossingen ($x \in \{\overline 2, \overline 6,
\overline{10}\}$): zonder inverteerbaarheid geen schrapping. In
$\Z/11\Z$ daarentegen heeft elke vergelijking $\overline a x =
\overline b$ met $\overline a \neq \overline 0$ precies één oplossing.
\end{example}

\begin{example}[Groepsaxioma's als vergunning om op te lossen]\label{ex:b1:structures:solving}
In de \omterm{def:b1:structures:group}{groep} $\bigl((\Z/7\Z)^*, \times\bigr)$, los $\overline
3\,x = \overline 5$ op. Volgens \cref{prop:b1:structures:rules}~(3)
bestaat de oplossing, is ze uniek, en gelijk aan $\overline3^{-1}\,
\overline5$; omdat $\overline3 \times \overline5 = \overline{15}
= \overline1$, is de inverse van $\overline 3$ gelijk aan $\overline
5$, dus
\[
x = \overline5 \times \overline5 = \overline{25} = \overline4,
\qquad\text{controle: } \overline3 \times \overline4 =
\overline{12} = \overline5 .
\]
Het gaat minder om het antwoord dan om de garantie: in een \omterm{def:b1:structures:group}{groep} is
elke zulke vergelijking uniek oplosbaar \emph{voordat} men iets
berekent, zodat een oplossingsprocedure nooit op ``geen oplossing''
of ``verscheidene'' kan stuiten. Vergelijk $\overline3\,x = \overline6$
in $\Z/12\Z$ hierboven, waar de garantie faalt --- weten in welke
structuur men zich bevindt, is weten wat men als vanzelfsprekend mag
aannemen.
\end{example}

\begin{example}[Directe producten]\label{ex:b1:structures:product}
Als $G$ en $H$ \omterm{def:b1:structures:group}{groepen} zijn, is de productverzameling $G \times H$ met
de componentsgewijze bewerking $(g, h)(g', h') = (gg', hh')$ een \omterm{def:b1:structures:group}{groep}:
de axioma's worden coördinaat per coördinaat nagegaan, met neutraal
element $(e_G, e_H)$ en inversen $(g^{-1}, h^{-1})$. \omterm{def:b1:structures:order}{Orden} combineren
via het kgv: $(g, h)^m = (g^m, h^m)$ is het neutrale element als en
slechts als de \omterm{def:b1:structures:order}{orde} van $g$ en de \omterm{def:b1:structures:order}{orde} van $h$ beide $m$ delen. Zo
heeft in $\Z/2\Z \times \Z/2\Z$ (additief) elk niet-nul element \omterm{def:b1:structures:order}{orde} $2$
--- dit is precies de rechthoekgroep van
\cref{ex:b1:structures:klein} in coördinaten --- terwijl $\Z/4\Z$
een element van \omterm{def:b1:structures:order}{orde} $4$ heeft: een tweede, berekeningsvrij bewijs
dat de twee \omterm{def:b1:structures:group}{groepen} van grootte $4$ niet isomorf zijn (een
isomorfisme bewaart \omterm{def:b1:structures:order}{orden}). Producten zijn de eenvoudigste manier om
nieuwe \omterm{def:b1:structures:group}{groepen} uit oude te vervaardigen, en het vlak $\R^2 = \R \times
\R$ uit \cref{ch:b1:vspaces} is het belangrijkste voorbeeld van deze
constructie.
\end{example}

\begin{remark}[Fermat, structureel]
In het \omterm{def:b1:structures:field}{lichaam} $\Z/p\Z$ vormen de niet-nul klassen een multiplicatieve
\omterm{def:b1:structures:group}{groep} met $p - 1$ elementen, en de kleine stelling van Fermat
(\cref{thm:b1:arith:fermat}) zegt: elk element $x$ van deze \omterm{def:b1:structures:group}{groep}
voldoet aan $x^{p-1} = \overline 1$. Dit is een bijzonder geval van een
algemeen feit over eindige \omterm{def:b1:structures:group}{groepen} (stelling van Lagrange), bewezen in
het tweede jaar; het koppelingsbewijs van de stelling van Wilson
(\cref{exo:b1:arith:11}) had al deze groepentheoretische inslag.
\end{remark}

\begin{remark}[Intermezzo: wat abstractie oplevert]\label{rem:b1:structures:interlude}
Men mag zich afvragen wat er gewonnen werd door, zeg,
\cref{prop:b1:structures:unique} voor een abstracte bewerking te
bewijzen in plaats van voor getallen. Het antwoord is hefboomwerking.
Dat argument van twee regels dekt nu, in één klap: inversen van
functies onder samenstelling
(\cref{thm:b1:logic:inverse}, waarvan het het uniciteitsbewijs woord
voor woord herhaalt), inversen mod $n$
(\cref{prop:b1:arith:invmod}), inversen van niet-nul reële getallen,
van eenheden in eender welke \omterm{def:b1:structures:ring}{ring}, en --- ongezien --- van de
inverteerbare matrices van \cref{ch:b1:matrices}, waar de uniciteit
van $A^{-1}$ geen enkele regel bewijs zal vergen. Dezelfde zuinigheid
geldt voor \cref{prop:b1:structures:kernel} (één injectiviteitscriterium,
hergebruikt voor lineaire \omterm{def:b1:logic:map}{afbeeldingen} in \cref{ch:b1:linmaps}) en voor
het deelgroepcriterium. Abstractie is hier geen algemeenheid om
haarzelf: het is de weigering om hetzelfde lemma vijf keer onder vijf
namen te bewijzen. De prijs --- bijhouden welke axioma's elke \omterm{def:b1:logic:statement}{uitspraak}
werkelijk gebruikte --- is precies wat de oefeningen van dit hoofdstuk
trainen.
\end{remark}

\begin{remark}[Waar dit hoofdstuk gebruikt wordt]\label{rem:b1:structures:whereused}
De woordenschat van dit hoofdstuk is de grammatica van de rest van dit
deel. \omterm{def:b1:structures:ring}{Ringen} en lichamen ordenen \cref{ch:b1:poly} ($K[X]$ is een
\omterm{def:b1:structures:ring}{ring} die $\Z$ nabootst) en \cref{ch:b1:fractions} ($K(X)$ is haar
breukenlichaam); vectorruimten (\cref{ch:b1:vspaces}) zijn \omterm{def:b1:structures:group}{abelse
groepen} waarop een \omterm{def:b1:structures:field}{lichaam} werkt; matrices
(\cref{ch:b1:matrices}) vormen de eerste ernstig niet-commutatieve
\omterm{def:b1:structures:ring}{ring} van dit deel, en hun inverteerbare elementen een \omterm{def:b1:structures:group}{groep} waarvan de
studie de lineaire algebra zelf is. \omterm{def:b1:structures:morphism}{Morfismen} en \omterm{def:b1:structures:morphism}{kernen} keren terug als
lineaire \omterm{def:b1:logic:map}{afbeeldingen} en nulruimten in \cref{ch:b1:linmaps} ---
\cref{prop:b1:structures:kernel} \emph{is} het injectiviteitscriterium
van dat hoofdstuk, hier eens en voor altijd bewezen. De \omterm{ex:b1:structures:groups}{symmetrische
groep}, ster van het weekendprobleem hieronder, levert de signatuur
waarop determinanten worden gebouwd in \cref{ch:b1:det}.
\end{remark}

\begin{exercise}[$\star$]\label{exo:b1:structures:1}
Op $E = \R \setminus \{1\}$, definieer $x * y = x + y - xy$. Bewijs dat
$(E, *)$ een \omterm{def:b1:structures:group}{abelse groep} is. \emph{(Bepaal het neutrale element en de

\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:structures:2}
Welke van de volgende zijn \omterm{def:b1:structures:group}{groepen}?
\begin{enumerate}
  \item $(\intoo{0}{+\infty}, \times)$;
  \item $(\{-1, 0, 1\}, +)$;
  \item $(\Q^*, \times)$;
  \item de \omterm{def:b1:logic:sets}{verzameling} van oneven gehele getallen onder optelling.
\end{enumerate}
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:structures:3}
Schrijf de samenstellingstabel van de \omterm{ex:b1:structures:groups}{symmetrische groep} $\mathfrak{S}_3$
van $\{1,2,3\}$ (zes bijecties: identiteit, drie transposities, twee
$3$-cykels), en geef twee elementen die niet commuteren.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:structures:4}
Bewijs dat $H = \{z \in \C^* : \abs z = 1\}$ een \omterm{def:b1:structures:subgroup}{deelgroep} is van
$(\C^*, \times)$, en dat $\R_+^*$ er een andere is; is $H \cup \R_+^*$
een \omterm{def:b1:structures:subgroup}{deelgroep}?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:structures:5}
Zij $f \colon (\R, +) \to (\C^*, \times)$, $\theta \mapsto
\eu^{\iu\theta}$. Bewijs dat $f$ een \omterm{def:b1:structures:morphism}{morfisme} is, bereken $\ker f$ en
$\operatorname{im} f$, en leid uit
\cref{prop:b1:structures:kernel} af dat $f$ niet \omterm{def:b1:logic:inj}{injectief} is. Beperk
het domein om het \omterm{def:b1:logic:inj}{injectief} te maken op een zo groot mogelijk interval.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:structures:6}
Zij $H, K$ \omterm{def:b1:structures:subgroup}{deelgroepen} van $G$. Bewijs dat $H \cap K$ een \omterm{def:b1:structures:subgroup}{deelgroep} is,
en dat $H \cup K$ \emph{enkel} een \omterm{def:b1:structures:subgroup}{deelgroep} is wanneer $H \subseteq K$
of $K \subseteq H$. \emph{(Als $h \in H \setminus K$ en $k \in K
\setminus H$, waar kan $hk$ dan leven?)}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:structures:7}
Een \omterm{def:b1:structures:group}{groep} $G$ voldoet aan $x^2 = e$ voor alle $x \in G$. Bewijs dat $G$
\omterm{def:b1:structures:group}{abels} is. \emph{(Werk $(xy)^2$ uit.)}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:structures:8}
In $\Z/18\Z$: som de eenheden op en vind de inverse van $\overline 5$;
los $\overline 5\, x = \overline 7$ op; los $\overline 6\, x =
\overline 3$ en $\overline 6\, x = \overline{12}$ op.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:structures:9}
Bewijs dat de \omterm{def:b1:logic:sets}{verzameling} $\Z[\sqrt 2] = \{a + b\sqrt 2 : a, b \in \Z\}$
een \omterm{def:b1:structures:ring}{ring} is (een deelring van $\R$), en dat $1 + \sqrt 2$ er een eenheid
van is met oneindig veel verschillende machten --- zodat $\Z[\sqrt
2]^\times$ oneindig is, anders dan $\Z^\times$.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:structures:10}
(Booleaanse \omterm{def:b1:structures:ring}{ringen}) Zij $A$ een \omterm{def:b1:structures:ring}{ring} waarin $x^2 = x$ voor elke $x$.
Bewijs dat $x + x = 0$ voor alle $x$, en dat $A$ commutatief is.
\emph{(Werk $(x+x)^2$ en $(x+y)^2$ uit.)} Geef een voorbeeld van zo'n
\omterm{def:b1:structures:ring}{ring} met $\mathcal{P}(E)$, waarbij het symmetrisch verschil als
optelling en de doorsnede als vermenigvuldiging wordt genomen.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:structures:11}
Zij $G$ een \omterm{def:b1:structures:group}{groep} waarin, voor een zekere vaste $n \geq 1$, $(xy)^n =
x^n y^n$, $(xy)^{n+1} = x^{n+1}y^{n+1}$ en $(xy)^{n+2} =
x^{n+2}y^{n+2}$ voor alle $x, y$. Bewijs dat $G$ \omterm{def:b1:structures:group}{abels} is.
\emph{(Leid uit de drie identiteiten eerst $y^n x = x y^n$ af, dan
$y^{n+1} x = x y^{n+1}$, en besluit.)}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:structures:12}
\begin{enumerate}
  \item Bepaal alle groepsmorfismen van $(\Z, +)$ naar $(\Z, +)$.
  \item Bewijs dat het enige groepsmorfisme van $(\Q, +)$ naar
        $(\Z, +)$ het nulmorfisme is. \emph{(Voor $x \in \Q$ en
        $n \in \N^*$, vergelijk $f(x)$ en $n\,f(x/n)$.)}
\end{enumerate}
\end{exercise}

\section{Probleem: De symmetrische groep en de 8-puzzel}

\begin{problem}\label{pb:b1:structures:1}
De \omterm{def:b1:structures:group}{groep} $\mathfrak S_n$ van \omterm{def:b1:counting:objects}{permutaties} van $\intint1n$ is de
oudste \omterm{def:b1:structures:group}{groep} in de wiskunde en nog steeds de meest leerzame. Dit
probleem bouwt haar structuurtheorie vanaf nul op --- cykels,
voortbrenging door transposities, het \emph{signatuur}\omterm{def:b1:structures:morphism}{morfisme}
$\varepsilon \colon \mathfrak S_n \to \{\pm1\}$ (waarvan het bestaan
werkelijk niet-triviaal is), en de \omterm{pb:b1:structures:1}{alternerende groep} $\mathfrak A_n$
voortgebracht door $3$-cykels --- en verzilvert dat vervolgens op een
klassieke puzzel: in het $3 \times 3$ schuifspel kan geen enkele reeks
zetten twee tegels verwisselen en al de rest op zijn plaats laten.
\index{symmetrische groep}\index{signatuur}\index{alternerende groep}
\omterm{def:b1:counting:objects}{Permutaties} werken op $\intint1n$; producten $\sigma\tau$ betekenen
``pas eerst $\tau$ toe''; $[\,v_1, \dots, v_n]$ noteert de
\omterm{def:b1:counting:objects}{permutatie} die $i$ naar $v_i$ stuurt.

\medskip
\noindent\textbf{Deel I --- Cykels en transposities.}
\begin{enumerate}
  \item Verantwoord $\abs{\mathfrak S_n} = n!$
        (\cref{thm:b1:counting:counts}). In $\mathfrak S_3$,
        bereken beide producten van $\sigma = [2, 3, 1]$ en $\tau =
        [1, 3, 2]$, en besluit dat $\mathfrak S_3$ niet
        \omterm{def:b1:structures:group}{abels} is.
  \item Een \emph{$k$-cykel} $(a_1\ a_2\ \dots\ a_k)$ ($k \geq 2$,
        de $a_i$ paarsgewijs verschillend) stuurt $a_1 \mapsto a_2
        \mapsto \dots \mapsto a_k \mapsto a_1$ en laat al de rest
        vast; haar \emph{drager} is $\{a_1, \dots,
        a_k\}$. Bewijs dat twee cykels met disjuncte dragers
        commuteren.
  \item Bewijs dat elke $\sigma \in \mathfrak S_n$ een product
        is van cykels met paarsgewijs disjuncte dragers, en dat deze
        ontbinding uniek is op de volgorde van de factoren na.
        \emph{(Beschouw, voor elke $i$, de rij $i, \sigma(i),
        \sigma^2(i), \dots$: ze moet naar $i$ terugkeren; de
        resulterende \emph{banen} partitioneren $\intint1n$, en
        $\sigma$ werkt op elke als een cykel.)}
  \item Ontbind $\sigma = [4, 1, 5, 2, 3, 7, 8, 6] \in
        \mathfrak S_8$ in disjuncte cykels. Definieer de
        \emph{\omterm{def:b1:structures:order}{orde}} van $\sigma$ als in
        \cref{def:b1:structures:order}, en bewijs dat de \omterm{def:b1:structures:order}{orde} van een
        product van disjuncte cykels het kgv van hun lengten is, en
        bereken de \omterm{def:b1:structures:order}{orde} van deze $\sigma$.
  \item Bewijs de telescoperende identiteit
        \[
        (a_1\ a_2\ \dots\ a_k)
        = (a_1\ a_k)(a_1\ a_{k-1})\cdots(a_1\ a_2) ,
        \]
        en besluit dat elke \omterm{def:b1:counting:objects}{permutatie} een product van
        transposities is. Schrijf de $\sigma$ van vraag 4 als zo'n
        product.
  \item Toon verder aan dat de \emph{aangrenzende} transposities
        $(i\ \ i{+}1)$ volstaan: voor $a < b$,
        \[
        (a\ b) = (a\ \ a{+}1)(a{+}1\ \ a{+}2)\cdots(b{-}1\ \ b)
        \cdots(a{+}1\ \ a{+}2)(a\ \ a{+}1),
        \]
        een product van $2(b - a) - 1$ aangrenzende transposities ---
        een \emph{oneven} aantal (deze pariteit zal hieronder tweemaal
        van belang zijn).
\end{enumerate}

\medskip
\noindent\textbf{Deel II --- De signatuur bestaat.} Voor $\sigma
\in \mathfrak S_n$, zij
\[
N(\sigma) = \#\bigl\{(i, j) : i < j,\ \sigma(i) >
\sigma(j)\bigr\}
\]
haar aantal \emph{inversies}, en stel $\varepsilon(\sigma) =
(-1)^{N(\sigma)}$.
\begin{enumerate}[resume]
  \item Bereken $N$ en $\varepsilon$ voor de identiteit, voor een
        transpositie $(i\ \ i{+}1)$, en voor $[2, 3, 1]$.
  \item Bewijs dat voor elke $\sigma$ en elke aangrenzende
        transpositie $\tau = (i\ \ i{+}1)$:
        $N(\sigma\tau) = N(\sigma) \pm 1$. \emph{(Rechts samenstellen
        met $\tau$ verwisselt de waarden op posities $i$ en
        $i + 1$; precies één paar verandert van inversiestatus.)}
  \item Leid, met vraag 6, af dat voor \emph{elke}
        transpositie $\tau$ geldt $\varepsilon(\sigma\tau) =
        -\varepsilon(\sigma)$; besluit dat als $\sigma$ een
        product is van $p$ transposities, dan $\varepsilon(\sigma)
        = (-1)^p$ --- in het bijzonder hangt de pariteit van $p$ enkel
        van $\sigma$ af, niet van de gekozen ontbinding --- en dat
        $\varepsilon \colon \mathfrak S_n \to \{\pm 1\}$ een
        groepsmorfisme is.
  \item Toon aan dat een $k$-cykel signatuur $(-1)^{k-1}$ heeft, en
        dat in het algemeen $\varepsilon(\sigma) = (-1)^{n -
        c(\sigma)}$, waarbij $c(\sigma)$ het aantal banen van
        $\sigma$ is (vaste punten inbegrepen).
  \item De \emph{\omterm{pb:b1:structures:1}{alternerende groep}} is $\mathfrak A_n =
        \ker\varepsilon$. Verantwoord dat ze een \omterm{def:b1:structures:subgroup}{deelgroep} is, en bewijs
        $\abs{\mathfrak A_n} = \frac{n!}2$ voor $n \geq 2$.
        \emph{(Fixeer een transpositie $\tau_0$ en beschouw $\sigma
        \mapsto \sigma\tau_0$.)}
  \item Consistentiecontrole op $\sigma = [4, 1, 5, 2, 3, 7, 8, 6]$:
        bereken $\varepsilon(\sigma)$ op drie manieren --- door
        inversies te tellen, uit het cykeltype via vraag 10, en uit
        je transpositietelling in vraag 5.
\end{enumerate}

\medskip
\noindent\textbf{Deel III --- $\mathfrak A_n$ wordt voortgebracht door
$3$-cykels.}
\begin{enumerate}[resume]
  \item Zij $a, b, c, d$ paarsgewijs verschillend. Ga de twee
        identiteiten na
        \[
        (a\ b)(a\ c) = (a\ c\ b),
        \qquad
        (a\ b)(c\ d) = (a\ c\ b)(a\ c\ d) .
        \]
  \item Bewijs dat voor $n \geq 3$ elk element van $\mathfrak
        A_n$ een product van $3$-cykels is. \emph{(Een even
        \omterm{def:b1:counting:objects}{permutatie} is een product van een even aantal
        transposities; absorbeer ze twee per twee.)}
  \item Schrijf $(1\ 2)(3\ 4)$ en de $5$-cykel $(1\ 2\ 3\ 4\ 5)$
        expliciet als producten van $3$-cykels.
  \item Bewijs de conjugatieformule: voor elke $\sigma \in
        \mathfrak S_n$,
        \[
        \sigma\,(a_1\ \dots\ a_k)\,\sigma^{-1}
        = \bigl(\sigma(a_1)\ \dots\ \sigma(a_k)\bigr) .
        \]
\end{enumerate}

\medskip
\noindent\textbf{Deel IV --- De 8-puzzel.} Tegels $1, \dots, 8$
schuiven in een $3 \times 3$ raam met één lege cel; een \emph{zet}
schuift een tegel die grenst aan de lege cel daarin. Nummer de cellen
$1, \dots, 9$ (rij per rij; de opgeloste positie heeft tegel $i$ in
cel $i$ en de lege cel in cel $9$). Behandel de lege cel als
een negende tegel, zodat een positie een \omterm{def:b1:counting:objects}{permutatie} $\sigma \in \mathfrak
S_9$ is (tegel $\sigma(i)$ ligt in cel $i$).
\begin{enumerate}[resume]
  \item Toon aan dat een zet $\sigma$ vervangt door $\sigma \circ \tau$
        waarbij $\tau$ de transpositie is van de twee betrokken
        cellen; leid af dat elke zet
        $\varepsilon(\sigma)$ omklapt.
  \item Zij $d(\sigma)$ de taxi-afstand (rijen plus
        kolommen) tussen de huidige cel van de lege cel en haar
        thuiscel $9$. Toon aan dat elke zet $d$ met $\pm1$ verandert,
        zodat elke zet ook $(-1)^{d(\sigma)}$ omklapt. Besluit dat
        \[
        I(\sigma) = \varepsilon(\sigma)\cdot(-1)^{d(\sigma)}
        \]
        \emph{invariant} is onder elke zet.
  \item Bewijs de klassieke onmogelijkheid van de puzzel: de positie
        die tegels $7$ en $8$ verwisselt en al de rest
        (inclusief de lege cel) op zijn plaats laat, kan niet
        vanuit de opgeloste positie worden bereikt.
  \item We nemen het omgekeerde aan (het bewijs is een leerzame maar
        langdradige inductie): elke positie met $I = +1$ is
        bereikbaar. Leid af dat precies de helft van de $8!$ posities
        met de lege cel thuis oplosbaar is, d.w.z.\
        $\frac{8!}2 = 20\,160$.
  \item Leid uit vraag 20 af dat de bereikbare tegelschikkingen
        met de lege cel thuis precies de
        \omterm{def:b1:structures:subgroup}{deelgroep} $\mathfrak A_8 \leq \mathfrak S_8$ vormen.
  \item Toepassingen van de invariant: kan men (a) de
        positie bereiken waar tegels $1, 2, 3$ cyclisch gepermuteerd
        zijn en al de rest, lege cel inbegrepen, thuis is?
        (b) de positie waar tegel $5$ en de lege cel van
        plaats zijn gewisseld en alle andere tegels thuis zijn? Verantwoord
        beide antwoorden met $I$.
\end{enumerate}

\medskip
\noindent\textbf{Deel V --- Synthese.}
\begin{enumerate}[resume]
  \item Bewijs dat voor $n \geq 3$ de enige groepsmorfismen $f
        \colon \mathfrak S_n \to \{\pm 1\}$ het constante
        \omterm{def:b1:structures:morphism}{morfisme} en $\varepsilon$ zijn. \emph{(Gebruik vraag 16 en de
        commutativiteit van $\{\pm1\}$ om aan te tonen dat $f$ dezelfde
        waarde aanneemt op alle transposities.)}
  \item Waar precies gebruikte het probleem: (i) het begrip
        \omterm{def:b1:structures:morphism}{morfisme} en \cref{prop:b1:structures:kernel}; (ii) de
        telprincipes van \cref{ch:b1:counting}; (iii) de
        kwestie van goede definitie die vragen 8--9 oplossen? Eén
        zin per punt.
  \item Synthese, in een korte paragraaf: één pariteitsfunctie,
        één keer bewezen goed gedefinieerd te zijn, ordent tegelijk de
        interne structuur van $\mathfrak S_n$ (de \omterm{def:b1:structures:subgroup}{deelgroep}
        $\mathfrak A_n$), beslist een fysieke puzzel, en ---
        via de formule $\det = \sum_\sigma
        \varepsilon(\sigma)\cdots$ --- zal determinanten
        definiëren in \cref{ch:b1:det}. Becommentarieer het
        terugkerende patroon:
        invarianten veranderen ``probeer alle reeksen zetten'' in één
        berekening.
\end{enumerate}
\end{problem}
