\chapter{Afleiden}\label{ch:b1:derivative}

\omterm{def:b1:derivative:def}{Afgeleiden} werden overal in het deel over de middelbare school berekend;
wat ontbrak is de keten van stellingen die berekening omzet in
informatie over functies: de stelling van Rolle, de middelwaardestelling
en hun gevolgen --- monotoniecriteria, Lipschitz-grenzen, convexiteit.
Alles in dit hoofdstuk betreft functies gedefinieerd op een \omterm{prop:b1:reals:intervals}{interval} $I$.

\section{De afgeleide}

\begin{definition}\label{def:b1:derivative:def}
$f \colon I \to \R$ is \emph{afleidbaar in $x_0 \in
I$}\index{afgeleide} wanneer het differentiequotiënt
$\frac{f(x) - f(x_0)}{x - x_0}$ een (eindige) limiet heeft als $x \to x_0$;
de limiet wordt geschreven als $f'(x_0)$. Equivalent:
\[
f(x_0 + h) = f(x_0) + f'(x_0)\,h + h\,\varepsilon(h),
\qquad \varepsilon(h) \xrightarrow[h \to 0]{} 0 ,
\]
waarbij de grafiek dan de raaklijn $y = f(x_0) + f'(x_0)(x -
x_0)$ toelaat. Afleidbaarheid in $x_0$ impliceert \omterm{def:b1:continuity:continuous}{continuïteit} in $x_0$ (lees
de formule). $f$ is afleidbaar op $I$ wanneer het dat in elk punt is;
$f$ is van klasse $C^1$ wanneer bovendien $f'$ \omterm{def:b1:continuity:continuous}{continu} is, en van
klasse $C^k$\index{Ck function@$C^k$-functie} wanneer $f$ $k$ keer
kan worden afgeleid met $f^{(k)}$ \omterm{def:b1:continuity:continuous}{continu}.
\end{definition}

\begin{example}\label{ex:b1:derivative:notc1}
De omkering van ``\omterm{def:b1:derivative:def}{afleidbaar} $\Rightarrow$ \omterm{def:b1:continuity:continuous}{continu}'' faalt:
$\abs{\,\cdot\,}$ in $0$. Verrassender is dat \omterm{def:b1:derivative:def}{afleidbaar} niet
$C^1$ impliceert: de functie $f(x) = x^2 \sin\frac 1x$ ($f(0) = 0$) is
overal \omterm{def:b1:derivative:def}{afleidbaar}, met $f'(0) = 0$, maar $f'(x) = 2x
\sin\frac1x - \cos\frac 1x$ heeft geen limiet in $0$
(\cref{exo:b1:derivative:2}).
\end{example}

\begin{example}[Afleidbaar in precies één punt]\label{ex:b1:derivative:onepoint}
Zij $f(x) = x^2$ voor $x \in \Q$ en $f(x) = 0$ voor $x \notin
\Q$. In $0$: $\bigl|\frac{f(h) - 0}{h}\bigr| \leq \abs h \to 0$,
dus $f$ is \omterm{def:b1:derivative:def}{afleidbaar} in $0$ met $f'(0) = 0$. In elke $x_0
\neq 0$ is $f$ zelfs niet \omterm{def:b1:continuity:continuous}{continu}: rationale en irrationale
rijen die naar $x_0$ convergeren sturen $f$ naar respectievelijk $x_0^2 \neq 0$ en
$0$ (\omterm{def:b1:topology:dense}{dichtheid}, \cref{thm:b1:reals:density}). Dus
is \omterm{def:b1:derivative:def}{afleidbaarheid} een werkelijk \emph{puntsgewijs} begrip: het kan
gelden in één punt van $\R$ en nergens anders. De moraal voor
de praktijk: uitspraken zoals het monotoniecriterium of Rolle
vereisen de \omterm{def:b1:derivative:def}{afgeleide} \emph{op een \omterm{prop:b1:reals:intervals}{interval}} --- het bezitten van
$f'(x_0)$ in geïsoleerde punten, hoeveel ook, ondersteunt geen enkele globale
conclusie hoegenaamd.
\end{example}

\begin{theorem}[Bewerkingen]\label{thm:b1:derivative:operations}
Als $f, g$ \omterm{def:b1:derivative:def}{afleidbaar} zijn in $x_0$ (en waar de formules zin
hebben):
\[
(f + g)' = f' + g', \qquad
(fg)' = f'g + fg', \qquad
\Bigl(\frac fg\Bigr)' = \frac{f'g - fg'}{g^2},
\]
en als $g$ \omterm{def:b1:derivative:def}{afleidbaar} is in $f(x_0)$:
$\;(g \circ f)'(x_0) = g'\bigl(f(x_0)\bigr)\, f'(x_0)$ (ketting\-regel\index{kettingregel}).
\end{theorem}

\begin{proof}
Som: onmiddellijk. Product: schrijf
\[
f(x)g(x) - f(x_0)g(x_0)
= \bigl(f(x) - f(x_0)\bigr) g(x) + f(x_0)\bigl(g(x) - g(x_0)\bigr),
\]
deel door $x - x_0$ en laat $x \to x_0$ ($g$ is \omterm{def:b1:continuity:continuous}{continu} in $x_0$).
Quotiënt: behandel $\frac 1g$ via $\frac{1/g(x) - 1/g(x_0)}{x - x_0} =
\frac{-1}{g(x)g(x_0)}\cdot\frac{g(x) - g(x_0)}{x - x_0}$, en pas dan
de productregel toe. Kettingregel: met $y_0 = f(x_0)$, definieer
$\theta(y) = \frac{g(y) - g(y_0)}{y - y_0}$ voor $y \neq y_0$ en
$\theta(y_0) = g'(y_0)$: $\theta$ is \omterm{def:b1:continuity:continuous}{continu} in $y_0$, en voor
$x \neq x_0$,
\[
\frac{g(f(x)) - g(f(x_0))}{x - x_0}
= \theta\bigl(f(x)\bigr)\cdot \frac{f(x) - f(x_0)}{x - x_0}
\longrightarrow g'(y_0)\, f'(x_0),
\]
de eerste factor door samenstelling van limieten (deze truc behandelt het
geval $f(x) = f(x_0)$ netjes, waar het naïeve ``vermenigvuldig en deel
door $f(x) - f(x_0)$'' breekt).
\end{proof}

\begin{theorem}[Afgeleide van een inverse functie]\label{thm:b1:derivative:inverse}
Zij $f$ \omterm{def:b1:continuity:continuous}{continu} en strikt monotoon op $I$, \omterm{def:b1:derivative:def}{afleidbaar}
in $x_0$ met $f'(x_0) \neq 0$. Dan is $f^{-1}$
(\cref{thm:b1:continuity:bijection}) \omterm{def:b1:derivative:def}{afleidbaar} in $y_0 =
f(x_0)$, met
\[
(f^{-1})'(y_0) = \frac{1}{f'(x_0)} = \frac{1}{f'\bigl(f^{-1}(y_0)\bigr)} .
\]
Als $f'(x_0) = 0$, heeft de inverse een verticale raaklijn in $y_0$.
\end{theorem}

\begin{proof}
Voor $y \to y_0$, stel $x = f^{-1}(y)$: \omterm{def:b1:continuity:continuous}{continuïteit} van $f^{-1}$ geeft $x
\to x_0$, en
\[
\frac{f^{-1}(y) - f^{-1}(y_0)}{y - y_0}
= \frac{x - x_0}{f(x) - f(x_0)}
= \frac{1}{\dfrac{f(x) - f(x_0)}{x - x_0}}
\longrightarrow \frac{1}{f'(x_0)} .
\]
Bewering over de verticale raaklijn: als $f'(x_0) = 0$, is het weergegeven quotiënt
de inverse van een grootheid die naar $0$ neigt terwijl ze
één constant teken behoudt (voor $f$ strikt stijgend, $\frac{f(x) -
f(x_0)}{x - x_0} > 0$ voor alle $x \neq x_0$): het differentie\-quotiënt
van $f^{-1}$ neigt daarom naar $+\infty$ (naar $-\infty$
voor $f$ dalend). De inverse blijft \omterm{def:b1:continuity:continuous}{continu} maar is niet
\omterm{def:b1:derivative:def}{afleidbaar} in $y_0$ --- haar grafiek, de spiegeling van die van $f$
over de diagonaal, staat verticaal precies waar die van $f$
horizontaal liep, zoals $x^{1/3}$ in $0$ illustreert tegenover $x^3$.
\end{proof}

\begin{example}[Inverse afgeleiden, tweemaal]\label{ex:b1:derivative:inverseaction}
De stelling herberekent de klassieke \omterm{def:b1:derivative:def}{afgeleiden} zonder enig limiet\-werk. Voor
$\ln = \exp^{-1}$: in $y = \eu^x$,
\[
(\ln)'(y) = \frac{1}{\exp'(x)} = \frac{1}{\eu^{x}} = \frac1y ,
\]
geldig voor elke $y > 0$ aangezien $\exp' = \exp$ nooit nul wordt. Voor
$\arctan = \tan^{-1}$: in $y = \tan x$,
\[
(\arctan)'(y) = \frac{1}{1 + \tan^2 x} = \frac{1}{1 + y^2} ,
\]
gebruikmakend van $\tan' = 1 + \tan^2 > 0$. Het afsluitende inzicht: de formule
zet kennis over een functie om in kennis over haar
inverse tegen de prijs van één substitutie --- en de
substitutie ($x = \ln y$, $x = \arctan y$) is precies de
\omterm{def:b1:logic:statement}{uitspraak} dat de twee variabelen aan tegenovergestelde zijden van de
bijectie leven.
\end{example}

\section{Rolle en de middelwaardestelling}

\begin{proposition}[Inwendig extremum]\label{prop:b1:derivative:fermat}
Als $f$ \omterm{def:b1:derivative:def}{afleidbaar} is in een \emph{inwendig} punt $x_0$ van $I$ en
daar een lokaal extremum heeft, dan is $f'(x_0) = 0$.
\end{proposition}

\begin{proof}
Zeg een lokaal maximum: er is een $r > 0$ met $f(x) \leq f(x_0)$
voor $\abs{x - x_0} \leq r$, en inwendigheid garandeert dat beide
zijden van $x_0$ beschikbaar zijn binnen $I$. Voor $0 < h \leq r$ heeft het
quotiënt $\frac{f(x_0 + h) - f(x_0)}{h}$ teller $\leq 0$
en noemer $> 0$: het is $\leq 0$, en zijn limiet $f'(x_0)$
erft $\leq 0$ (brede ongelijkheden gaan over naar limieten,
\cref{thm:b1:seq:order}); voor $-r \leq h < 0$ is het quotiënt
$\geq 0$, wat $f'(x_0) \geq 0$ geeft. Dus $f'(x_0) = 0$. (In een
eindpunt is slechts één teken beschikbaar: de conclusie faalt daar ---
denk aan $x$ op $\intcc{0}{1}$, maximaal in $1$ met \omterm{def:b1:derivative:def}{afgeleide}
$1$.)
\end{proof}

\begin{theorem}[Rolle]\label{thm:b1:derivative:rolle}
Zij $f$ \omterm{def:b1:continuity:continuous}{continu} op $\intcc{a}{b}$, \omterm{def:b1:derivative:def}{afleidbaar} op
$\intoo{a}{b}$, met $f(a) = f(b)$. Dan is $f'(c) = 0$ voor een zekere $c \in
\intoo{a}{b}$.\index{stelling van Rolle}
\end{theorem}

\begin{proof}
Volgens de extremumstelling (\cref{thm:b1:continuity:evt}) bereikt $f$
zijn maximum en minimum op $\intcc{a}{b}$. Als beide worden
bereikt in eindpunten, dan (aangezien $f(a) = f(b)$) is max $=$ min en $f$
constant: elke \omterm{def:b1:topology:closure}{inwendige} $c$ voldoet. Anders wordt een extremum
bereikt in een inwendig punt $c$, en
\cref{prop:b1:derivative:fermat} geeft $f'(c) = 0$.
\end{proof}

\begin{theorem}[Middelwaardestelling]\label{thm:b1:derivative:mvt}
Zij $f$ \omterm{def:b1:continuity:continuous}{continu} op $\intcc{a}{b}$, \omterm{def:b1:derivative:def}{afleidbaar} op
$\intoo{a}{b}$. Er bestaat een $c \in \intoo{a}{b}$ met
\[
f(b) - f(a) = f'(c)\,(b - a) .
\]
\emph{Middelwaarde-ongelijkheid:} als bovendien $m \leq f' \leq M$ op
$\intoo{a}{b}$, dan $m(b-a) \leq f(b) - f(a) \leq M(b-a)$; in het
bijzonder impliceert $\abs{f'} \leq K$ dat $f$ $K$-Lipschitz is.
\index{middelwaardestelling}
\end{theorem}

\begin{proof}
Pas Rolle toe op $g(x) = f(x) - \frac{f(b) - f(a)}{b - a}(x - a)$: $g$
is \omterm{def:b1:continuity:continuous}{continu} op $\intcc{a}{b}$, \omterm{def:b1:derivative:def}{afleidbaar} binnenin, en $g(a) =
f(a) = g(b)$. In het punt $c$ waar $g'(c) = 0$: $f'(c) =
\frac{f(b)-f(a)}{b-a}$. De ongelijkheid volgt door $f'(c)$ te begrenzen;
de \omterm{def:b1:logic:statement}{Lipschitz-uitspraak} past dit toe op elk paar punten.
\end{proof}

\begin{example}[Newtons methode is die van Heron]\label{ex:b1:derivative:newton}
Newtons methode voor het oplossen van $f(x) = 0$ vervangt de kromme door
haar raaklijn in de huidige schatting $x_n$ en neemt de wortel van de raaklijn
als volgende schatting:
\[
0 = f(x_n) + f'(x_n)(x_{n+1} - x_n)
\quad\Longrightarrow\quad
x_{n+1} = x_n - \frac{f(x_n)}{f'(x_n)} .
\]
Pas het toe op $f(x) = x^2 - 2$:
\[
x_{n+1} = x_n - \frac{x_n^2 - 2}{2x_n}
= \frac{x_n}{2} + \frac{1}{x_n}
= \frac12\Bigl(x_n + \frac{2}{x_n}\Bigr) :
\]
precies de iteratie van Heron (\cref{ex:b1:seq:heron}), twee
millennia te vroeg. De kwadratische snelheid die daar werd waargenomen wordt nu
verklaard door het raaklijnbeeld: nabij een enkelvoudige wortel verschillen kromme en
raaklijn door een fout van tweede orde, dus verdubbelt elke stap ruwweg
het aantal correcte cijfers --- de algemene \omterm{def:b1:logic:statement}{uitspraak} volgt uit de
Taylor-grenzen van \cref{ch:b1:taylor}. Het afsluitende inzicht:
waar dichotomie (\cref{ex:b1:continuity:dichotomy}) alleen
\omterm{def:b1:continuity:continuous}{continuïteit} gebruikt en één bit per stap wint, besteedt Newton een
\omterm{def:b1:derivative:def}{afgeleide} om het aantal correcte cijfers per stap te verdubbelen.
\end{example}

\begin{example}[De middelwaardestelling als schatter]\label{ex:b1:derivative:mvtestimate}
Hoe groot is $\sqrt{101}$? Pas de stelling toe op $f(t) = \sqrt t$
op $\intcc{100}{101}$: voor een zekere $c \in \intoo{100}{101}$,
\[
\sqrt{101} - 10 = \frac{1}{2\sqrt c},
\qquad\text{dus}\qquad
\frac{1}{2\sqrt{101}} < \sqrt{101} - 10 < \frac{1}{20} = 0.05 ,
\]
en aangezien $\sqrt{101} < 10.05$, overtreft de linkergrens
$\frac{1}{20.1} > 0.0497$: dus $10.0497 < \sqrt{101} < 10.05$
(werkelijke waarde $10.049875\dots$) --- drie correcte decimalen uit één
afgeleide-evaluatie. Evenzo $\abs{\sin a - \sin b} \leq
\abs{a - b}$ (grens $\abs{\cos}\leq 1$): de Lipschitz-schattingen
die sinds \cref{ch:b1:seq} gebruikt worden zijn allemaal deze stelling. Het afsluitende
inzicht: de middelwaardestelling is een Taylor-formule van orde nul
--- ze ruilt één onbekend punt $c$ voor een moeilijke ongelijkheid, en
\cref{ch:b1:taylor} zal precies deze ruil itereren.
\end{example}

\begin{corollary}[Monotoniecriterium]\label{cor:b1:derivative:monotone}
Zij $f$ \omterm{def:b1:continuity:continuous}{continu} op $I$, \omterm{def:b1:derivative:def}{afleidbaar} op het \omterm{def:b1:topology:closure}{inwendige}.
\begin{enumerate}
  \item $f' \geq 0$ op het \omterm{def:b1:topology:closure}{inwendige} $\iff$ $f$ is stijgend; $f' = 0$
        $\iff$ $f$ constant.
  \item Als $f' > 0$ behalve in eindig veel punten waar het nul wordt,
        is $f$ \emph{strikt} stijgend.
\end{enumerate}
\end{corollary}

\begin{proof}
Als $f' \geq 0$: voor $x < y$ in $I$ geeft de middelwaardestelling op
$\intcc{x}{y}$ dat $f(y) - f(x) = f'(c)(y - x) \geq 0$. Omgekeerd zijn
differentiequotiënten van een stijgende functie $\geq 0$, dus hun
limieten ook. Het constante geval: pas het voorgaande toe op $f'$ en
$-f' \geq 0$. Strikte versie: $f$ is stijgend; gelijkheid $f(x) =
f(y)$ voor $x < y$ zou $f$ bevriezen op $\intcc{x}{y}$, wat $f' =
0$ daar afdwingt --- oneindig veel punten.
\end{proof}

\begin{example}[Gelijke afgeleiden, ongelijke functies]\label{ex:b1:derivative:intervalmatters}
Op $\R^* = \intoo{-\infty}{0} \cup \intoo{0}{+\infty}$ voldoen zowel
$f(x) = \ln\abs x$ als $g(x) = \ln\abs x + \mathbf{1}_{x>0}$
(voeg $1$ alleen op de rechterhalflijn toe) aan $f' = g' =
\frac1x$. Ze verschillen niet met een constante: het criterium
``$f' = 0 \implies f$ constant'' is een \emph{\omterm{prop:b1:reals:intervals}{interval}}-uitspraak
--- het bewijs ervan draait de middelwaardestelling tussen twee punten,
wat vereist dat het hele segment dat ze verbindt in het
domein ligt. Op elke halflijn afzonderlijk zijn de primitieven van
$\frac1x$ gelijk aan $\ln\abs x + c$ met één constante per halflijn,
twee onafhankelijke constanten in totaal. \Cref{ch:b1:integration}
erft deze kleine lettertjes: ``de'' primitieve van een functie is
welgedefinieerd op een constante na \emph{op elk \omterm{prop:b1:reals:intervals}{interval} van haar
domein}, en tabellen van primitieven veronderstellen stilzwijgend samenhang.
\end{example}

\begin{example}[Striktheid gratis]\label{ex:b1:derivative:strictness}
$x \mapsto x^3$ is \emph{strikt} stijgend op $\R$ ook al
wordt haar \omterm{def:b1:derivative:def}{afgeleide} nul in $0$: de clausule van het criterium
``$f' > 0$ behalve in eindig veel punten'' is precies ontworpen
voor zulke vlakke punten. Daarentegen geeft $f' \geq 0$ alleen
stijging in de brede zin (een constante functie kwalificeert), en
een \omterm{def:b1:derivative:def}{afgeleide} die nul wordt op een heel deelinterval bevriest de
functie daar wel. De praktische regel: om strikte
monotonie te beweren, som de nulpunten van $f'$ op; eindig veel (of meer
algemeen, geen enkel op enig deelinterval) is onschadelijk, een \omterm{prop:b1:reals:intervals}{interval}
ervan is fataal.
\end{example}

\begin{example}[Een volledige verloopstudie]\label{ex:b1:derivative:variations}
Bestudeer $f(x) = x^3 - 3x + 1$ op $\R$. \omterm{def:b1:derivative:def}{Afgeleide}: $f'(x) =
3(x^2 - 1)$, positief op $\intoo{-\infty}{-1}$, negatief op
$\intoo{-1}{1}$, positief op $\intoo{1}{+\infty}$: volgens het
monotoniecriterium stijgt $f$, daalt dan, stijgt dan,
met een lokaal maximum $f(-1) = 3$ en een lokaal minimum
$f(1) = -1$. Limieten: $\mp\infty$ in $\mp\infty$. Gevolgen,
afgelezen van de verlooptabel met de tussenwaardestelling
op elke monotone tak: $f$ wordt precies eenmaal nul in
elk van
\[
\intoo{-\infty}{-1}, \qquad \intoo{-1}{1}, \qquad
\intoo{1}{+\infty}
\]
(de waarden in de knooppunten hebben tegengestelde tekens: $3 > 0 > -1$),
dus heeft de vergelijking $x^3 - 3x + 1 = 0$ precies drie reële
wortels; numeriek liggen ze nabij $-1.88$, $0.35$, $1.53$. Het
afsluitende inzicht: een verlooptabel is een \emph{bewijsmiddel},
geen schets --- monotone tak plus tekenwissel is gelijk aan
precies één wortel, en de tabel somt de takken
uitputtend op.
\end{example}

\begin{theorem}[Formule van Leibniz]\label{thm:b1:derivative:leibniz}
Als $f, g$ $n$ keer \omterm{def:b1:derivative:def}{afleidbaar} zijn, dan ook $fg$,
en\index{formule van Leibniz}
\[
(fg)^{(n)} = \sum_{k=0}^{n} \binom nk f^{(k)}\, g^{(n-k)} .
\]
\end{theorem}

\begin{proof}
Inductie op $n$, precies parallel aan het binomium van Newton. Het
geval $n = 1$ is de productregel. Aannemend dat de formule op rang
$n$ geldt, leid nog eens af:
\[
(fg)^{(n+1)} = \sum_{k=0}^{n} \binom nk
\Bigl( f^{(k+1)} g^{(n-k)} + f^{(k)} g^{(n-k+1)} \Bigr),
\]
herindexeer dan de eerste som met $j = k + 1$ en verzamel de
coëfficiënt van $f^{(j)} g^{(n+1-j)}$: die is $\binom{n}{j-1} +
\binom nj = \binom{n+1}{j}$ volgens de regel van Pascal
(\cref{prop:b1:counting:identities}), waarbij de randtermen $j = 0$
en $j = n + 1$ $\binom{n+1}{0} = \binom{n+1}{n+1} =
1$ dragen zoals het hoort.
\end{proof}

\begin{example}[Leibniz in actie]\label{ex:b1:derivative:leibnizaction}
Bereken $\bigl(x^2 \eu^x\bigr)^{(n)}$ voor $n \geq 2$. Neem $f =
x^2$, waarvan de \omterm{def:b1:derivative:def}{afgeleiden} snel uitsterven ($f' = 2x$, $f'' = 2$,
$f^{(k)} = 0$ voor $k \geq 3$), en $g = \eu^x$: slechts drie termen
van de Leibniz-som overleven,
\[
\bigl(x^2\eu^x\bigr)^{(n)}
= \binom n0 x^2 \eu^x + \binom n1 (2x)\,\eu^x + \binom n2\,
2\,\eu^x
= \eu^x\bigl(x^2 + 2nx + n(n-1)\bigr).
\]
Controle bij $n = 1$: $\eu^x(x^2 + 2x)$, wat inderdaad
$(x^2\eu^x)'$ is. Het afsluitende inzicht: gebruik Leibniz wanneer één factor
een \omterm{def:b1:poly:def}{veelterm} is --- de som heeft dan slechts $\deg + 1$ termen, en
de formule is een gesloten vorm, geen abstracte identiteit. (Voor twee
oneindig levendige factoren zoals $\eu^x\sin x$ zijn complexe
exponentiëlen uit \cref{ch:b1:complex} het betere hulpmiddel.)
\end{example}

\section{Convexiteit}

\begin{definition}\label{def:b1:derivative:convex}
$f \colon I \to \R$ is \emph{convex}\index{convexe functie} wanneer
elke koorde boven de grafiek ligt:
\[
\forall x, y \in I,\ \forall t \in \intcc{0}{1}, \quad
f\bigl(tx + (1-t)y\bigr) \leq t f(x) + (1-t) f(y).
\]
($f$ is \emph{concaaf} wanneer $-f$ convex is.)
\end{definition}

\begin{theorem}[Differentiële karakteriseringen]\label{thm:b1:derivative:convexchar}
Zij $f$ \omterm{def:b1:derivative:def}{afleidbaar} op $I$. De volgende zijn equivalent:
\begin{enumerate}
  \item $f$ is \omterm{def:b1:derivative:convex}{convex};
  \item $f'$ is stijgend op $I$;
  \item de grafiek ligt boven elke raaklijn: $f(y) \geq f(x) +
        f'(x)(y - x)$ voor alle $x, y \in I$.
\end{enumerate}
Als $f$ twee keer \omterm{def:b1:derivative:def}{afleidbaar} is: $f$ \omterm{def:b1:derivative:convex}{convex} $\iff f'' \geq 0$.
\end{theorem}

\begin{proof}
(1 $\Rightarrow$ 3) Convexiteit geschreven als $\frac{f(x + t(y-x)) -
f(x)}{t} \leq f(y) - f(x)$ voor $t \in \intoc{0}{1}$; laat $t \to 0^+$:
$f'(x)(y - x) \leq f(y) - f(x)$.

(3 $\Rightarrow$ 2) Voor $x < y$ geven de twee raaklijnongelijkheden in $x$
en in $y$ dat $f'(x)(y-x) \leq f(y) - f(x) \leq f'(y)(y - x)$,
dus $f'(x) \leq f'(y)$.

(2 $\Rightarrow$ 1) Fixeer $x < y$ en $t \in \intoo{0}{1}$, en zij $z
= tx + (1-t)y \in \intoo{x}{y}$. Volgens de middelwaardestelling op
$\intcc{x}{z}$ en $\intcc{z}{y}$: er zijn $c_1 < z < c_2$ met
\[
\frac{f(z) - f(x)}{z - x} = f'(c_1) \leq f'(c_2)
= \frac{f(y) - f(z)}{y - z} ,
\]
en het wegwerken van noemers ($z - x = (1-t)(y-x)$, $y - z = t(y-x)$)
herschikt precies tot de convexiteitsongelijkheid.

Tweemaal \omterm{def:b1:derivative:def}{afleidbaar} geval: $f'' \geq 0 \iff f'$ stijgend
(\cref{cor:b1:derivative:monotone}).
\end{proof}

\begin{example}[Klassieke convexiteitsongelijkheden]\label{ex:b1:derivative:convexineq}
$\exp$ is \omterm{def:b1:derivative:convex}{convex} ($\exp'' = \exp > 0$): haar raaklijn in $0$ geeft
$\eu^x \geq 1 + x$ voor alle $x$. $\ln$ is concaaf: haar raaklijn in $1$
geeft $\ln x \leq x - 1$; haar koorden geven, voor $0 < a \leq b$, de
ongelijkheid tussen meetkundig en rekenkundig gemiddelde: door $t =
\frac12$ te nemen in concaviteit,
\[
\ln\frac{a + b}{2} \geq \frac{\ln a + \ln b}{2} = \ln\sqrt{ab},
\qquad\text{dus}\qquad
\sqrt{ab} \leq \frac{a+b}{2} .
\]
De algemene rekenkundig-meetkundige ongelijkheid is
\cref{exo:b1:derivative:9}.
\end{example}

\begin{example}[Een convexiteitsongelijkheid vanaf nul]\label{ex:b1:derivative:xlnx}
De functie $f(t) = t\ln t$ is \omterm{def:b1:derivative:convex}{convex} op $\intoo{0}{+\infty}$:
$f''(t) = \frac1t > 0$. Haar middelpuntsongelijkheid, vermenigvuldigd met
$2$, luidt: voor alle $a, b > 0$,
\[
a\ln a + b\ln b \;\geq\; (a + b)\,\ln\frac{a + b}{2} ,
\]
met gelijkheid dan en slechts dan als $a = b$ (strikte convexiteit). Test: $a =
1$, $b = 3$ geeft $3\ln 3 = 3.296$ tegenover $4\ln 2 = 2.773$.
Deze onschuldige ongelijkheid is het tweepuntsgeval van de
\emph{entropie}-vergelijking die terugkeert met Jensens
ongelijkheid (\cref{exo:b1:derivative:9}) en in de
informatietheoretische asymptotiek van het deel over Jaar 3. Het
afsluitende inzicht: om een ongelijkheid te vervaardigen, vind een functie
waarvan de tweede \omterm{def:b1:derivative:def}{afgeleide} een teken heeft en schrijf op wat convexiteit
zegt --- de differentiële karakterisering zet één tekencontrole
om in oneindig veel ongelijkheden.
\end{example}

\begin{remark}[Veelvoorkomende valkuilen met afgeleiden]
(i) \emph{Een positieve \omterm{def:b1:derivative:def}{afgeleide} in één punt geeft geen
monotonie in de \omterm{def:b1:topology:open}{omgeving} ervan}: $f(x) = \frac x2 + x^2\sin\frac1x$ (met
$f(0) = 0$) heeft $f'(0) = \frac12 > 0$, en toch is
\[
f'(x) = \frac12 + 2x\sin\frac1x - \cos\frac1x
\]
gelijk aan $-\frac12$ in elke $x_n = \frac{1}{2\pi n}$: elke
\omterm{def:b1:topology:open}{omgeving} van $0$ bevat dalingen. Monotonie vereist $f'
\geq 0$ \emph{op een \omterm{prop:b1:reals:intervals}{interval}}
(\cref{cor:b1:derivative:monotone}); het puntsgewijze teken beheerst
alleen de kruising van de raaklijn. (ii) \emph{Rolles
drie hypothesen zijn allemaal actief}: $\abs x$ op $\intcc{-1}{1}$
(geen \omterm{def:b1:topology:closure}{inwendige} \omterm{def:b1:derivative:def}{afleidbaarheid}), $x$ op $\intcc{0}{1}$ (uiteinden niet
gelijk), en $x - \lfloor x\rfloor$ op $\intcc{0}{1}$ (\omterm{def:b1:continuity:continuous}{continuïteit}
faalt in $1$) breken elk precies één hypothese en de
conclusie. (iii) \emph{\omterm{def:b1:derivative:def}{Afgeleiden} mogen discontinu zijn, maar
niet willekeurig}: $f'$ kan oscilleren
(\cref{ex:b1:derivative:notc1}) en voldoet toch altijd aan de
tussenwaarde-eigenschap (Darboux,
\cref{exo:b1:derivative:10}): een \omterm{def:b1:derivative:def}{afgeleide} springt nooit --- als
je een eenzijdige ``afgeleidelimiet'' met een sprong berekent, heb
je een niet-afleidbare functie afgeleid. (iv) \emph{De
inverseformule vereist $f' \neq 0$}: $x \mapsto x^3$ is een gladde
strikt stijgende bijectie waarvan de inverse $x^{1/3}$ een
verticale raaklijn heeft in $0$ --- \omterm{def:b1:derivative:def}{afleidbaarheid} van de inverse gaat
verloren precies waar $f'$ nul wordt
(\cref{thm:b1:derivative:inverse}).
\end{remark}

\begin{remark}[Waar de middelwaardestelling hierna werkt]
Bijna elke kwantitatieve \omterm{def:b1:logic:statement}{uitspraak} van de volgende hoofdstukken is dit
hoofdstuks middelwaardestelling in vermomming: de hoofdstelling
van de integraalrekening (\cref{ch:b1:integration}) leidt de oppervlakte\-functie
af en besluit met het monotoniecriterium; de
Taylor--Lagrange-formule (\cref{ch:b1:taylor}) is de middelwaarde\-stelling
$n$ keer geïtereerd; de foutenanalyse van Newtons methode
en van vaste-puntiteraties (\cref{exo:b1:derivative:11}) is
de Lipschitz-vorm; en dit hoofdstuks weekendprobleem
(\cref{pb:b1:derivative:1}) maakt van dezelfde Lipschitz-grens
getaltheorie --- een afstotingsongelijkheid tussen algebraïsche
getallen en rationale getallen, die het eerste transcendente getal
in de geschiedenis oplevert. In het deel over Jaar 2 overleeft de middelwaarde\-\emph{ongelijkheid}
in meerdere variabelen wanneer de gelijkheid dat niet doet.
\end{remark}

\begin{example}[Ongelijkheid van Young uit concaviteit]\label{ex:b1:derivative:young}
Zij $p, q > 1$ met $\frac1p + \frac1q = 1$. Voor alle $a, b > 0$:
\[
ab \;\leq\; \frac{a^p}{p} + \frac{b^q}{q} .
\]
Bewijs door één toepassing van de concaviteit van $\ln$ met gewichten
$\frac1p, \frac1q$ (de tweepunts-Jensen-ongelijkheid, zoals in
\cref{exo:b1:derivative:9}):
\[
\ln\Bigl(\frac{a^p}{p} + \frac{b^q}{q}\Bigr)
\;\geq\; \frac1p \ln(a^p) + \frac1q \ln(b^q)
= \ln a + \ln b = \ln(ab),
\]
en $\ln$ stijgend zet de ongelijkheid van logaritmen om in
de bewering; gelijkheid dan en slechts dan als $a^p = b^q$ (strikte concaviteit). Het geval
$p = q = 2$ is de rekenkundig-meetkundige ongelijkheid $ab \leq
\frac{a^2 + b^2}{2}$ in vermomming. Het afsluitende inzicht: Youngs
ongelijkheid is de algebraïsche kiem van de ongelijkheden van Hölder en Minkowski
uit het deel over Jaar 2 --- één concaviteitsuitspraak
over $\ln$, geoogst voor normen.
\end{example}

\begin{remark}[Perspectieven binnen dit deel]
De \omterm{def:b1:derivative:def}{afgeleide} krijgt drie nieuwe levens voordat het deel eindigt.
In \cref{ch:b1:taylor} itereert ze: $n$ \omterm{def:b1:derivative:def}{afgeleiden} in een punt
worden samengeperst tot één \omterm{def:b1:poly:def}{veelterm} plus een beheerste fout, en
de middelwaardestelling wordt de restterm van Lagrange. In
\cref{ch:b1:curves} wordt afleiden meetkundig: voor een
geparametriseerde kromme $t \mapsto (x(t), y(t))$ is het paar $(x'(t),
y'(t))$ een snelheids\-\emph{vector}, wordt raking
collineariteit, en worden kritieke punten keerpunten om te classificeren. In
\cref{ch:b1:multivar} wordt één variabele tegelijk bevroren: partiële
\omterm{def:b1:derivative:def}{afgeleiden} herhalen dit hoofdstuk tweemaal, en de raaklijn
groeit uit tot een raakvlak. Alle drie hoofdstukken erven dezelfde
grammatica --- lokale lineaire benadering plus een foutterm ---
hier voor het eerst uitgesproken.
\end{remark}

Vervolledig \cref{ex:b1:derivative:notc1}: bewijs dat $f(x) = x^2

\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:derivative:3}
Gebruik de middelwaardestelling of de raaklijnongelijkheden om te bewijzen dat
voor alle $x > 0$:
\[
\frac{x}{1 + x} < \ln(1 + x) < x .
\]
Leid $\lim_{n\to\infty} \bigl(1 + \frac xn\bigr)^n = \eu^x$ af voor
elke $x > 0$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:derivative:4}
Zij $P$ een reële \omterm{def:b1:poly:def}{veelterm} met $k$ verschillende reële wortels. Bewijs dat
$P'$ ten minste $k - 1$ verschillende reële wortels heeft, verweven met die
van $P$. Leid af dat als $P$ al zijn wortels reëel heeft, dan ook $P'$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:derivative:5}
Zij $f$ \omterm{def:b1:derivative:def}{afleidbaar} op $\R$ met $f' (x)\to \ell$ als $x \to
+\infty$. Bewijs dat $\frac{f(x)}{x} \to \ell$ \emph{(middelwaarde\-stelling
op $\intcc{A}{x}$)}. Geldt $f(x+1) - f(x) \to \ell$ ook?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:derivative:6}
(Een discrete Rolle) Zij $f$ $n$ keer \omterm{def:b1:derivative:def}{afleidbaar} op $I$ en
nul in $n + 1$ verschillende punten. Bewijs dat $f^{(n)}$ ten
minste eenmaal nul wordt. Toepassing: een \omterm{def:b1:poly:def}{veelterm} van graad $\leq n$ die nul is in
$n+1$ punten is nul (opnieuw).
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:derivative:7}
Zij $f$ twee keer \omterm{def:b1:derivative:def}{afleidbaar} op $\intcc{a}{b}$ met $f(a) = f(b)
= 0$ en $f(x_0) > 0$ voor een zeker inwendig $x_0$. Bewijs dat $f''(c) <
0$ voor een zekere $c \in \intoo{a}{b}$. \emph{(Twee middelwaarde\-stellingen en
een vergelijking van hellingen.)}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:derivative:8}
Bestudeer de functie $f(x) = \dfrac{\ln x}{x}$ op
$\intoo{0}{+\infty}$: verloop, limieten, maximum. Leid af dat $a^b
> b^a$ voor alle reële getallen $\eu \leq a < b$, en beslecht het beroemde speciale
geval: welke van $\eu^\pi$, $\pi^\eu$ is groter? Controleer tegen de
kleine gehele paren $(2,3)$ en $(2,4)$: waarom gedragen ze zich
verschillend?
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:derivative:9}
(Rekenkundig-meetkundige ongelijkheid) Toon met behulp van de concaviteit van $\ln$ met
algemene gewichten (Jensens ongelijkheid voor $n$ punten, te bewijzen door
inductie op $n$) aan dat voor positieve reële getallen $a_1, \dots, a_n$:
\[
\sqrt[n]{a_1 a_2 \cdots a_n} \leq \frac{a_1 + \dots + a_n}{n},
\]
met gelijkheid dan en slechts dan als alle $a_i$ gelijk zijn.
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:derivative:10}
(Darboux: \omterm{def:b1:derivative:def}{afgeleiden} nemen tussenwaarden aan) Zij $f$
\omterm{def:b1:derivative:def}{afleidbaar} op $I$ en $a < b$ in $I$ met $f'(a) < v < f'(b)$.
Bewijs, door $g(x) = f(x) - vx$ te beschouwen en het punt waar $g$
zijn minimum op $\intcc{a}{b}$ bereikt, dat $f'(c) = v$ voor een zekere $c
\in \intoo{a}{b}$ --- ook al hoeft $f'$ niet \omterm{def:b1:continuity:continuous}{continu} te zijn
(\cref{exo:b1:derivative:2}).
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:derivative:11}
Zij $f \colon \R \to \R$ \omterm{def:b1:derivative:def}{afleidbaar} met $\abs{f'(x)} \leq k
< 1$ voor alle $x$ (een \emph{contractie}). Bewijs dat $f$ precies
één vast punt $\ell$ heeft, en dat elke rij $u_{n+1} = f(u_n)$
convergeert naar $\ell$ met $\abs{u_n - \ell} \leq k^n\abs{u_0 -
\ell}$. \emph{(Bestaan: pas de tussenwaardestelling toe op
$f(x) - x$ op een groot segment, gebruikmakend van de Lipschitz-grens; of gebruik
volledigheid met het Cauchy-criterium.)}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:derivative:12}
(Cauchy's middelwaardestelling en de regel van l'Hôpital)
\begin{enumerate}
  \item Zij $f, g$ \omterm{def:b1:continuity:continuous}{continu} op $\intcc{a}{b}$,
        \omterm{def:b1:derivative:def}{afleidbaar} op $\intoo{a}{b}$, met $g'$ daar nooit nul.
        Bewijs dat $g(b) \neq g(a)$ en dat een zekere $c \in
        \intoo{a}{b}$ voldoet aan
        \[
        \frac{f(b) - f(a)}{g(b) - g(a)} = \frac{f'(c)}{g'(c)}
        \]
        \emph{(pas Rolle toe op $h = f - \lambda g$ voor de juiste
        constante $\lambda$)}.
  \item Leid de regel van l'Hôpital in de $\frac00$-vorm in een
        punt af: als $f(a) = g(a) = 0$ en
        $\frac{f'(x)}{g'(x)} \to \ell$ als $x \to a^+$, dan
        $\frac{f(x)}{g(x)} \to \ell$.
  \item Toon dat de omkering faalt: voor $f(x) = x^2\sin\frac1x$
        ($f(0) = 0$) en $g(x) = x$ heeft het quotiënt
        $\frac{f}{g}$ een limiet in $0$ maar
        $\frac{f'}{g'}$ geen.
\end{enumerate}
\end{exercise}

\section{Probleem: de ongelijkheid van Liouville en het eerste
transcendente getal}

\begin{problem}[{Weekendprobleem --- algebraïsche getallen stoten
rationale getallen af: $\abs{x - p/q} \geq C/q^d$, en de transcendentie van
$\sum 10^{-n!}$}]
\label{pb:b1:derivative:1}
Een reëel getal is \emph{algebraïsch} wanneer het een wortel is van een niet-nul
\omterm{def:b1:poly:def}{veelterm} met gehele coëfficiënten, en \emph{\omterm{pb:b1:logic:1}{transcendent}}
anders. In 1844 produceerde Liouville het eerste getal ooit
\emph{bewezen} \omterm{pb:b1:logic:1}{transcendent}, en de motor van zijn bewijs is dit
hoofdstuks middelwaardestelling: een \omterm{pb:b1:logic:1}{algebraïsch getal} van graad $d$
kan niet beter door rationale getallen worden benaderd dan $C/q^d$ --- dus een
getal dat \emph{sneller dan elke macht} benaderbaar is kan niet
algebraïsch zijn. Dit probleem bouwt de ongelijkheid op, construeert
Liouvilles getal $L = 0.110001000\dots$ (enen op de faculteits\-posities,
via de cijfermachinerie van \cref{pb:b1:reals:1}),
bewijst zijn transcendentie, en eindigt met Cantors rivaliserende bewijs en
effectieve grenzen voor $\sqrt2$ en
$2^{1/3}$.\index{transcendent getal}\index{Liouville's
inequality}\index{algebraïsch getal}

\medskip
\noindent\textbf{Deel I --- Hoe goed kunnen rationale getallen worden
benaderd?}
\begin{enumerate}
  \item Toon aan dat twee verschillende rationale getallen $\frac ab \neq \frac pq$
        (geschreven met $b, q \geq 1$) voldoen aan $\bigl|\frac ab -
        \frac pq\bigr| \geq \frac{1}{bq}$. Leid af: als $x =
        \frac ab$ en $0 < \bigl|x - \frac pq\bigr| <
        \frac{1}{bq}$, bestaat zo'n $\frac pq$ niet --- een rationaal getal
        stoot alle andere rationale getallen af op schaal $\frac 1q$.
  \item Bewijs dat voor \emph{elk} rationaal getal $\frac pq$ ($q \geq
        1$): $\bigl|\sqrt2 - \frac pq\bigr| \geq
        \frac{1}{4q^2}$ \emph{(als de afstand groter is dan $1$ is dit
        duidelijk; begrens anders $\abs{\sqrt2 + p/q} < 4$ en
        gebruik het niet-nulle gehele getal $\abs{p^2 - 2q^2} \geq 1$)}.
  \item In de andere richting: controleer dat $(p, q) \mapsto (p +
        2q, p + q)$ $\abs{p^2 - 2q^2} = 1$ behoudt, genereer
        vanaf $(1,1)$ de paren $(3,2)$, $(7,5)$, $(17,12)$,
        $(41,29)$, $(99,70)$, en toon dat elk voldoet aan
        \[
        \Bigl|\sqrt2 - \frac pq\Bigr| =
        \frac{1}{q^2\,(\sqrt2 + p/q)} < \frac{1}{2q^2} :
        \]
        oneindig veel benaderingen van orde $2$. Met
        vraag 2: de benaderingsexponent van $\sqrt 2$ is
        \emph{precies} $2$.
  \item (Dirichlet) Zij $x$ irrationaal en $N \in \N^*$.
        Beschouw de $N + 1$ gebroken delen van $0, x, 2x,
        \dots, Nx$ in de $N$ vakjes $\intco{\frac kN}{\frac{k +
        1}{N}}$: volgens het duiventilprincipe
        (\cref{cor:b1:counting:pigeonhole}) vallen er twee in één vakje.
        Leid $q \leq N$ en $p$ af met $\abs{qx - p} <
        \frac 1N$, dus oneindig veel rationale getallen met
        $\bigl|x - \frac pq\bigr| < \frac{1}{q^2}$: \emph{elk}
        irrationaal getal is benaderbaar tot orde $2$.
\end{enumerate}

\medskip
\noindent\textbf{Deel II --- De ongelijkheid van Liouville.} Zij $x$
irrationaal en algebraïsch.
\begin{enumerate}[resume]
  \item Toon aan dat er onder de niet-nulle gehele \omterm{def:b1:poly:def}{veeltermen}
        die nul zijn in $x$ er één is, zeg $P$ van graad $d$,
        \emph{zonder rationale wortel}; en controleer $d \geq 2$
        \emph{(deel een factor $X - \frac ab$ uit over $\Q$ en
        werk noemers weg; graad $1$ zou $x$
        rationaal maken)}.
  \item Toon aan dat voor elk rationaal getal $\frac pq$ ($q \geq 1$):
        $\bigl|P\bigl(\frac pq\bigr)\bigr| \geq \frac{1}{q^d}$
        \emph{($q^d P(p/q)$ is een niet-nul geheel getal)}.
  \item Zij $M = \max_{\intcc{x-1}{x+1}} \abs{P'}$
        (\cref{thm:b1:continuity:evt}). Bewijs met de middelwaarde\-stelling
        tussen $x$ en $\frac pq$ de
        \emph{ongelijkheid van Liouville}: met $C = \min\bigl(1,
        \frac 1M\bigr) > 0$,
        \[
        \Bigl| x - \frac pq \Bigr| \geq \frac{C}{q^{\,d}}
        \qquad\text{voor elk rationaal getal } \frac pq,\ q \geq 1 .
        \]
  \item Noem $x$ een \emph{Liouville-getal} wanneer er voor elke $n \in
        \N$ een rationaal getal $\frac pq$ is met $q \geq 2$ en
        $0 < \bigl|x - \frac pq\bigr| < q^{-n}$. Bewijs dat een
        Liouville-getal irrationaal is \emph{(vraag 1: kies
        $n$ met $2^{\,n-1} > b$)}.
  \item Bewijs de stelling van Liouville: \emph{een Liouville-getal is
        \omterm{pb:b1:logic:1}{transcendent}} \emph{(combineer de vragen 7 en 8: de
        ongelijkheid $C < q^{\,d-n}$ faalt voor grote $n$)}.
\end{enumerate}

\medskip
\noindent\textbf{Deel III --- Het getal $L$.}
\begin{enumerate}[resume]
  \item Zij $L$ de waarde (in de zin van
        \cref{pb:b1:reals:1}) van de decimale cijferreeks met
        cijfer $1$ op de posities $n!$ ($n = 1, 2, 3, \dots$)
        en $0$ elders, d.w.z.\ $L = \sup_k t_k$ met $t_k =
        \sum_{n=1}^{k} 10^{-n!}$. Schrijf de eerste $25$
        cijfers van $L$ uit.
  \item Bewijs de staartinsluiting, voor elke $k \geq 1$:
        \[
        10^{-(k+1)!} \;\leq\; L - t_k \;\leq\;
        \frac{10}{9}\,10^{-(k+1)!} \;<\; 2\cdot 10^{-(k+1)!}
        \]
        \emph{(begrens elke partiële som voorbij $t_k$ door een eindige
        meetkundige som)}.
  \item Schrijf $t_k = \frac{p_k}{q_k}$ met $q_k = 10^{k!}$.
        Toon $0 < L - \frac{p_k}{q_k} < \frac{2}{q_k^{\,k+1}}$,
        en besluit dat $L$ een Liouville-getal is in de zin
        van vraag 8.
  \item Besluit: $L$ is \omterm{pb:b1:logic:1}{transcendent} --- het eerste expliciete
        voorbeeld in de geschiedenis (Liouville, 1844). Controleer zijn
        irrationaliteit rechtstreeks: zijn cijfers zijn niet uiteindelijk
        periodiek (groeiende gaten, zoals in \cref{pb:b1:reals:1},
        vraag 20).
  \item Veralgemeen: vervang elk cijfer $1$ door een willekeurig
        niet-nul cijfer $d_n \in \intint{1}{9}$. Toon dat de waarde
        nog steeds een Liouville-getal is, en leid af --- via het diagonaal\-argument
        van \cref{pb:b1:reals:1} (vraag 22) toegepast
        op deze cijferkeuzes --- dat er overaftelbaar
        veel transcendente getallen van deze vorm zijn.
\end{enumerate}

\medskip
\noindent\textbf{Deel IV --- De hiërarchie van benaderings\-ordes.}
Zeg dat $x$ \emph{benaderbaar tot orde $\mu$} is wanneer voor
een zekere constante $c > 0$ oneindig veel rationale getallen voldoen aan
$\bigl|x - \frac pq\bigr| < \frac{c}{q^{\mu}}$.
\begin{enumerate}[resume]
  \item Stel de hiërarchie samen uit de delen I--III: rationale getallen zijn
        benaderbaar tot orde $1$ en niet beter; $\sqrt 2$ tot
        orde $2$ en niet beter; elk irrationaal getal tot orde
        ten minste $2$; een \omterm{pb:b1:logic:1}{algebraïsch getal} van graad $d$ tot geen orde
        voorbij $d$; Liouville-getallen tot elke orde. Rechtvaardig
        elke bewering door de relevante vraag aan te halen.
  \item Toon dat $L + r$ een Liouville-getal is voor elk
        rationaal getal $r = \frac ab$ \emph{(verschuif de
        benaderingen: de nieuwe noemers zijn $b\,q_k$)}.
        Besluit dat Liouville-getallen --- en dus transcendente ---
        getallen \omterm{def:b1:topology:dense}{dicht} liggen in $\R$.
  \item (Cantor, 1874) Bewijs dat de \omterm{def:b1:logic:sets}{verzameling} van algebraïsche getallen
        aftelbaar is: er zijn eindig veel gehele
        \omterm{def:b1:poly:def}{veeltermen} met graad plus som van $\abs{\text{coëfficiënten}}$
        begrensd door $h$, elk met ten hoogste $\deg$ wortels; een
        aftelbare vereniging van \omterm{def:b1:counting:card}{eindige verzamelingen} is aftelbaar. Aangezien geen
        rij $\R$ uitput (\cref{pb:b1:reals:1}, vraag
        22), bestaan er transcendente getallen --- ze vormen zelfs
        een overaftelbare \omterm{def:b1:logic:sets}{verzameling}. Vergelijk de twee bewijzen: wat geeft
        dat van Liouville dat dat van Cantor niet kan?
  \item Bewijs rechtstreeks uit vraag 2 dat $\sqrt 2$
        \emph{geen} Liouville-getal is \emph{(voor $n \geq 3$
        begrenst de ongelijkheid $q^{-n} > \frac{1}{4q^2}$ $q$;
        dan blijven er slechts eindig veel kandidaat-rationale getallen over, alle
        op positieve afstand van $\sqrt2$)}. Veralgemeen:
        geen \omterm{pb:b1:logic:1}{algebraïsch getal} is een Liouville-getal.
\end{enumerate}

\medskip
\noindent\textbf{Deel V --- Effectieve constanten.}
\begin{enumerate}[resume]
  \item Voor het Pell-paar $(99, 70)$: verifieer $99^2 - 2\cdot70^2
        = 1$ en evalueer de exacte fout
        \[
        \sqrt2 - \frac{99}{70}
        = \frac{-1}{70^2\,\bigl(\sqrt2 + \frac{99}{70}\bigr)},
        \qquad
        \Bigl|\sqrt 2 - \frac{99}{70}\Bigr| \approx 7.2\cdot
        10^{-5} :
        \]
        vijf correcte cijfers uit een breuk met drie cijfers.
  \item Voer Deel II uit op $x = 2^{1/3}$, $P = X^3 - 2$: controleer dat $P$
        geen rationale wortel heeft, begrens $M = \max_{\intcc{x-1}{x+1}}
        3t^2 \leq 3\,(1 + 2^{1/3})^2 < 16$, en besluit de
        effectieve ongelijkheid
        \[
        \Bigl| 2^{1/3} - \frac pq \Bigr| \geq
        \frac{1}{16\,q^3} \qquad \text{voor alle } \frac pq .
        \]
  \item Opbrengst: toon dat elk rationaal getal dat $2^{1/3}$
        tot op $10^{-6}$ benadert een noemer $q \geq 40$ moet hebben.
  \item Toon dat het grondtal $10$ irrelevant is: het binaire
        analoog $\sum_{n\geq1} 2^{-n!}$ (waarde van de binaire
        reeks met enen op faculteitsposities) is ook een
        Liouville-getal, dus \omterm{pb:b1:logic:1}{transcendent}.
\end{enumerate}

\medskip
\noindent\textbf{Deel VI --- Grenzen en synthese.}
\begin{enumerate}[resume]
  \item Zij $x^\dagger$ de waarde van de decimale reeks met
        enen precies op de posities $3^k$ ($k \geq 0$). Toon dat
        $x^\dagger$ benaderbaar is tot orde $3$, en leid
        uit de ongelijkheid van Liouville af dat $x^\dagger$ noch
        rationaal noch een kwadratisch irrationaal getal is. Leg uit waarom de
        methode daar vastloopt: orde $3$ is verenigbaar met
        algebraïciteit van graad $\geq 3$, en dat gat sluiten
        (elke exponent $> 2$ volstaat, voor elk \omterm{pb:b1:logic:1}{algebraïsch
        getal}) is de stelling van Roth, ver voorbij dit deel.
  \item Kwantificeer Cantor: toon dat de algebraïsche getallen van
        graad $\leq d$ gegeven door \omterm{def:b1:poly:def}{veeltermen} met coëfficiënten
        in $\intint{-H}{H}$ er ten hoogste $d\,(2H + 1)^{d+1}$ zijn.
        (Deze eindigheid is wat vraag 17 liet werken.)
  \item Synthese, één zin elk: (i) lokaliseer het enige
        analytische ingrediënt van Liouvilles bewijs (welke stelling
        van dit hoofdstuk, waar gebruikt); (ii) formuleer de spanning
        die het aandrijft (integraliteit dwingt $\abs{P(p/q)} \geq
        q^{-d}$ af, gladheid verbiedt $\abs{P(p/q)} > M\abs{x -
        p/q}$); (iii) contrasteer Liouvilles en Cantors bewijzen
        van het bestaan van transcendente getallen; (iv) noem
        waar dit deel het thema opnieuw ontmoet --- het weekend\-probleem
        van \cref{ch:b1:integration} bewijst dat $\pi$
        irrationaal is via dezelfde integraliteit-tegen-kleinheid
        klem, met integralen in plaats van \omterm{def:b1:derivative:def}{afgeleiden}.
\end{enumerate}
\end{problem}
