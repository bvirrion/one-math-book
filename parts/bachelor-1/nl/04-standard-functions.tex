\chapter{Standaardfuncties}\label{ch:b1:functions}

Analyse is slechts zo nuttig als de voorraad functies die men beheerst. Aan
de \omterm{def:b1:logic:sets}{verzameling} geërfd van het secundair onderwijs --- machten, exponentiële,
logaritme, trigonometrische functies --- voegt dit hoofdstuk hun inverse
functies toe ($\arcsin$, $\arccos$, $\arctan$) en de hyperbolische familie.
Afgeleiden worden vrij gebruikt op middelbareschoolniveau; de
theorie achter inverse functies wordt vervolledigd in
\cref{ch:b1:continuity,ch:b1:derivative}.

\section{Exponentiële, logaritme, machten}

\begin{proposition}[Samenvatting en karakterisering]\label{prop:b1:functions:expln}
$\exp \colon \R \to \intoo{0}{+\infty}$ en $\ln \colon
\intoo{0}{+\infty} \to \R$ zijn wederzijds inverse bijecties, strikt
stijgend, met
\[
\exp(x + y) = \exp x \exp y, \qquad \ln(xy) = \ln x + \ln y,
\qquad \exp' = \exp, \qquad \ln'(x) = \frac 1x .
\]
Bovendien is $\exp$ de \emph{enige} afleidbare functie $f \colon
\R \to \R$ met $f' = f$ en $f(0) = 1$.
\end{proposition}

\begin{proof}
De samenvatting is middelbareschoolstof. Voor de uniciteit, zij $f' = f$,
$f(0) = 1$, en stel $g(x) = f(x)\,\eu^{-x}$. Dan is $g' = f'\eu^{-x} -
f\eu^{-x} = 0$, dus $g$ is constant gelijk aan $g(0) = 1$: $f = \exp$.
\end{proof}

\begin{definition}[Algemene machten]\label{def:b1:functions:powers}
Voor $x > 0$ en $\alpha \in \R$:
$\;x^\alpha = \eu^{\alpha \ln x}$\index{machtsfunctie}. Voor $a > 0$,
$a \neq 1$, is de \emph{logaritme met grondtal $a$} $\log_a x =
\frac{\ln x}{\ln a}$, de inverse van $x \mapsto a^x$.
\end{definition}

\begin{example}[Exponentiële vergelijkingen oplossen]\label{ex:b1:functions:expequations}
Los $2^x = 5^{\,x-1}$ op in $\R$. Beide leden zijn positief, dus neem
logaritmen --- een omkeerbare stap:
\[
x\ln2 = (x - 1)\ln5
\iff x(\ln2 - \ln5) = -\ln5
\iff x = \frac{\ln5}{\ln5 - \ln2} = \frac{\ln 5}{\ln\frac52}
\approx 1.756 .
\]
Los vervolgens $x^{\sqrt2} = 3$ op voor $x > 0$: verhef tot de macht
$\frac1{\sqrt2}$ (d.w.z.\ pas de inverse bijectie toe):
$x = 3^{1/\sqrt2} = \eu^{(\ln 3)/\sqrt2} \approx 2.175$. Het
inzicht: elke vergelijking waarin machten verweven zijn, wordt ontrafeld via $\ln$ en
$\exp$, omdat de definitie $x^\alpha = \eu^{\alpha\ln x}$
alle manipulaties met machten reduceert tot rekenen met exponenten ---
maar enkel op het domein $x > 0$ waar die definitie leeft.
\end{example}

\begin{example}[Verdubbelingstijden]\label{ex:b1:functions:doubling}
Een grootheid groeit met $3\%$ per stap: na $n$ stappen is ze
vermenigvuldigd met $(1.03)^n$. Wanneer verdubbelt ze? Los $(1.03)^n
\geq 2$ op:
\[
n\ln(1.03) \geq \ln2
\iff n \geq \frac{\ln 2}{\ln 1.03}
= \frac{0.6931}{0.02956} \approx 23.4 ,
\]
dus de eerste verdubbeling treedt op bij stap $24$. (De ``regel van
$72$'' der financiers, die de verdubbelingstijd schat als
$72$ gedeeld door het percentage, is deze berekening met
de benadering $\ln(1 + x) \approx x$, gekwantificeerd in
\cref{ch:b1:taylor}.) Exponentiële
processen redeneert men het best via hun logaritmen: op
die schaal is de groei lineair en worden vragen delingen.
\end{example}

\begin{example}[Hoe lang is $2^{2026}$?]\label{ex:b1:functions:digits}
Het aantal decimale cijfers van een geheel getal $N \geq 1$ is
$\floor{\log_{10} N} + 1$ (inderdaad heeft $N$ precies $d$ cijfers wanneer
$10^{d-1} \leq N < 10^d$, d.w.z.\ $d - 1 \leq \log_{10}N < d$). Voor
$N = 2^{2026}$:
\[
\log_{10} 2^{2026} = 2026\,\log_{10}2
= 2026 \times 0.301030 = 609.887 ,
\]
dus $2^{2026}$ heeft $610$ cijfers. Het decimale deel levert een
bonus: $10^{0.887} \approx 7.7$, dus het getal \emph{begint} met
$7$. Eén vermenigvuldiging beantwoordde een vraag over een getal dat niemand
ooit voluit zal schrijven --- logaritmen comprimeren multiplicatieve grootte
tot additieve grootte, wat hun hele historische bestaansreden is
(\cref{ex:b1:structures:expmorphism} maakt die zin precies).
\end{example}

\begin{proposition}[Machtsregels en groeivergelijking]\label{prop:b1:functions:powerrules}
Voor $x, y > 0$ en $\alpha, \beta \in \R$:
\[
x^{\alpha+\beta} = x^\alpha x^\beta, \quad
(x^\alpha)^\beta = x^{\alpha\beta}, \quad
(xy)^\alpha = x^\alpha y^\alpha, \quad
(x^\alpha)' = \alpha\, x^{\alpha - 1}.
\]
Groeischaal\index{groeivergelijking} als $x \to +\infty$, voor elke
$\alpha > 0$ en $\beta > 0$:
\[
\frac{(\ln x)^\beta}{x^\alpha} \longrightarrow 0,
\qquad
\frac{x^\alpha}{\eu^{\beta x}} \longrightarrow 0 .
\]
\end{proposition}

\begin{proof}
De identiteiten schrijven die van $\exp$ en $\ln$ over via de
definitie. In detail voor de tweede (de minst voor de hand liggende): $x^\alpha
> 0$ en $\ln(x^\alpha) = \alpha\ln x$ (pas $\ln$ toe op de
definitie), dus
\[
(x^\alpha)^\beta = \eu^{\beta\ln(x^\alpha)}
= \eu^{\beta\alpha\ln x} = x^{\alpha\beta} ;
\]
de eerste en de derde zijn dezelfde ontrafelingen van twee regels via
$\exp(u + v) = \exp u\exp v$ en $\ln(xy) = \ln x + \ln y$. De
afgeleide is de kettingregel: $(\eu^{\alpha\ln x})' =
\frac{\alpha}{x} \eu^{\alpha\ln x}$.

Vergelijkingen: uit $\frac{\ln t}{t} \to 0$ (bewezen in het
middelbareschoolvolume), substitueer $t = x^{\alpha/\beta}$:
$\frac{\ln x}{x^{\alpha/\beta}} \to 0$, verhef vervolgens tot de macht
$\beta$. Voor de tweede, substitueer $x = \ln u$ in de eerste.
\end{proof}

\begin{example}[Twee limieten die elke lezer zich eigen moet maken]\label{ex:b1:functions:xx}
Wat zijn $\lim_{x \to 0^+} x^x$ en $\lim_{x \to +\infty}
x^{1/x}$? Beide machten zijn gedefinieerd via de exponentiële, dus
regel eerst de exponent af:
\[
x^x = \eu^{x\ln x}, \qquad
x\ln x = -\frac{\ln(1/x)}{1/x} \longrightarrow 0
\quad (x \to 0^+),
\qquad\text{dus } x^x \longrightarrow \eu^0 = 1 ;
\]
en $x^{1/x} = \eu^{(\ln x)/x} \to \eu^0 = 1$ als $x \to +\infty$,
rechtstreeks via de \omterm{prop:b1:functions:powerrules}{groeivergelijking}. Het inzicht: een onbepaalde
macht (van de vorm $0^0$ of $\infty^0$) wordt \emph{altijd} behandeld door
$u^v = \eu^{v\ln u}$ te herschrijven en het product $v \ln u$ te analyseren
--- nooit door te gissen op basis van grondtal en exponent afzonderlijk. De
functie $x \mapsto \frac{\ln x}{x}$ die beide limieten besliste, wordt
uitputtend bestudeerd in het weekendprobleem van dit hoofdstuk.
\end{example}

\begin{remark}[Veelvoorkomende valkuilen met machten en logaritmen]\label{rem:b1:functions:pitfalls}
\begin{enumerate}
  \item \emph{Domeinen.} $x^\alpha$ voor irrationale $\alpha$ vereist
        $x > 0$; $(-8)^{1/3}$ vermijdt men het best ten voordele van ``de
        reële derdemachtswortel van $-8$'', omdat de regel
        $(x^\alpha)^\beta = x^{\alpha\beta}$ stilzwijgend faalt bij
        negatieve getallen: $\bigl((-8)^2\bigr)^{1/6} = 2 \neq -2$.
  \item \emph{$\sqrt{x^2} = \abs x$}, niet $x$: de absolute
        waarde vergeten is de klassieke bron van verloren negatieve
        oplossingen.
  \item \emph{$\ln(xy) = \ln x + \ln y$ vereist $x, y > 0$.} Voor
        negatieve argumenten kan $\ln(xy)$ gedefinieerd zijn terwijl het
        rechterlid dat niet is.
  \item \emph{Welke functie groeit?} In $x^\alpha$ varieert het
        \emph{grondtal}; in $a^x$ de \emph{exponent}.
        Hun groei is wild verschillend
        (\cref{prop:b1:functions:powerrules}), en hybride
        uitdrukkingen zoals $x^{\ln x}$ of $x^{1/x}$ moeten
        herschreven worden als $\eu^{(\cdot)}$ vooraleer men redeneert ---
        zoals in \cref{ex:b1:functions:xx}.
\end{enumerate}
\end{remark}

\section{Inverse trigonometrische functies}

\begin{definition}[$\arcsin$, $\arccos$, $\arctan$]\label{def:b1:functions:arc}
De beperkingen
\[
\sin \colon \intcc{-\tfrac\pi2}{\tfrac\pi2} \to \intcc{-1}{1},
\qquad
\cos \colon \intcc{0}{\pi} \to \intcc{-1}{1},
\qquad
\tan \colon \intoo{-\tfrac\pi2}{\tfrac\pi2} \to \R
\]
zijn strikt monotone bijecties. Hun inverse \omterm{def:b1:logic:map}{afbeeldingen} worden geschreven
$\arcsin$\index{boogsinus}, $\arccos$\index{boogcosinus} en
$\arctan$\index{boogtangens}. Zo is bijvoorbeeld $y = \arcsin x$
($x \in \intcc{-1}{1}$) \emph{de} hoek in
$\intcc{-\frac\pi2}{\frac\pi2}$ waarvan de sinus $x$ is.
\end{definition}

\begin{proposition}[Afgeleiden]\label{prop:b1:functions:arcderiv}
Op het inwendige van hun domeinen:
\[
\arcsin' x = \frac{1}{\sqrt{1 - x^2}},
\qquad
\arccos' x = \frac{-1}{\sqrt{1 - x^2}},
\qquad
\arctan' x = \frac{1}{1 + x^2}.
\]
\end{proposition}

\begin{proof}
Anticiperend op de regel voor inverse functies, bewezen in
\cref{ch:b1:derivative}: als $f$ een afleidbare bijectie is en
$f' \neq 0$, dan is $(f^{-1})'(x) = \frac{1}{f'(f^{-1}(x))}$. Voor
$\arcsin$: met $y = \arcsin x$,
\[
\arcsin' x = \frac{1}{\cos y}
= \frac{1}{\sqrt{1 - \sin^2 y}} = \frac{1}{\sqrt{1 - x^2}},
\]
waar $\cos y = +\sqrt{1 - \sin^2 y}$ omdat $y \in
\intoo{-\frac\pi2}{\frac\pi2}$ oplegt dat $\cos y > 0$. Analoog is
$\arccos' x = \frac{1}{-\sin y}$ met $\sin y > 0$ op
$\intoo{0}{\pi}$, en $\arctan' x = \frac{1}{1 + \tan^2 y} =
\frac{1}{1 + x^2}$ gebruikmakend van $\tan' = 1 + \tan^2$.
\end{proof}

\begin{example}[Een verborgen constante herkennen]\label{ex:b1:functions:hiddenconst}
Bestudeer $g(x) = \arctan\dfrac{1 - x}{1 + x}$ op $\intoo{-1}
{+\infty}$. Kettingregel en een korte berekening:
\[
g'(x) = \frac{1}{1 + \bigl(\frac{1-x}{1+x}\bigr)^2}\cdot
\frac{-(1+x) - (1-x)}{(1+x)^2}
= \frac{-2}{(1+x)^2 + (1-x)^2}
= \frac{-1}{1 + x^2} ,
\]
aangezien $(1+x)^2 + (1-x)^2 = 2 + 2x^2$. Dus $g' = -\arctan'$: de
functie $g + \arctan$ heeft nulafgeleide op het \emph{interval}
$\intoo{-1}{+\infty}$, en is er dus constant; haar waarde in $x =
0$ is $\arctan 1 + \arctan 0 = \frac\pi4$. Besluit:
\[
\arctan\frac{1 - x}{1 + x} = \frac\pi4 - \arctan x
\qquad (x > -1) .
\]
Op $\intoo{-\infty}{-1}$ geldt dezelfde berekening van de afgeleide, maar
de constante is anders ($-\frac{3\pi}4$: bereken de limiet als
$x \to -\infty$). Het inzicht: ``nulafgeleide impliceert
constant'' is een \omterm{def:b1:logic:statement}{uitspraak} per interval --- precies de
subtiliteit die uitgebuit wordt in \cref{exo:b1:functions:6} en
\cref{exo:b1:functions:10}.
\end{example}

\begin{proposition}[Standaardidentiteiten]\label{prop:b1:functions:arcidentities}
\begin{enumerate}
  \item Voor $x \in \intcc{-1}{1}$: $\arcsin x + \arccos x =
        \dfrac{\pi}{2}$.
  \item Voor $x > 0$: $\arctan x + \arctan\dfrac 1x = \dfrac{\pi}{2}$
        (en $-\frac\pi2$ voor $x < 0$).
  \item $\sin(\arccos x) = \cos(\arcsin x) = \sqrt{1 - x^2}$;
        $\;\tan(\arcsin x) = \dfrac{x}{\sqrt{1 - x^2}}$ voor
        $\abs x < 1$.
\end{enumerate}
\end{proposition}

\begin{proof}
(1) De afgeleide van $x \mapsto \arcsin x + \arccos x$ is nul op
$\intoo{-1}{1}$ (\cref{prop:b1:functions:arcderiv}), dus de functie
is er constant, gelijk aan haar waarde $\frac\pi2$ in $0$; de
randwaarden $\pm 1$ worden rechtstreeks nagegaan ($\frac\pi2 + 0$ en $-\frac\pi2 +
\pi$).

(2) Dezelfde methode op $\intoo{0}{+\infty}$: de afgeleide is
$\frac{1}{1+x^2} + \frac{-1/x^2}{1 + 1/x^2} = \frac{1}{1+x^2} -
\frac{1}{x^2 + 1} = 0$, en in $x = 1$ is de som $2\arctan 1 =
\frac\pi2$. Voor $x < 0$, gebruik de oneven pariteit van $\arctan$.

(3) Met $y = \arccos x \in \intcc{0}{\pi}$: $\sin y \geq 0$, dus
$\sin y = \sqrt{1 - \cos^2 y} = \sqrt{1 - x^2}$; evenzo voor
$\cos(\arcsin x)$, en de tangensformule is het quotiënt.
\end{proof}

\begin{remark}
$\arcsin(\sin\theta) = \theta$ geldt \emph{enkel} voor $\theta \in
\intcc{-\frac\pi2}{\frac\pi2}$: bijvoorbeeld is
$\arcsin(\sin\pi) = 0 \neq \pi$. De samenstelling in de andere
richting, $\sin(\arcsin x) = x$, is geldig op heel
$\intcc{-1}{1}$.
\end{remark}

\begin{example}[Terugvouwen naar het hoofdinterval]
\label{ex:b1:functions:wrap}
Bereken
\[
\arctan\Bigl(\tan\frac{3\pi}4\Bigr)
\qquad\text{en}\qquad
\arctan\Bigl(\tan\frac{17\pi}5\Bigr).
\]
Het recept: vervang de
hoek door de unieke hoek van $\intoo{-\frac\pi2}{\frac\pi2}$ met
dezelfde tangens, d.w.z.\ trek het juiste veelvoud van $\pi$ af
(de periode van $\tan$). Eerst: $\frac{3\pi}4 - \pi =
-\frac\pi4$, dus het antwoord is $-\frac\pi4$. Tweede: $\frac{17\pi}
5 - 3\pi = \frac{2\pi}5 \in \intoo{-\frac\pi2}{\frac\pi2}$, dus
het antwoord is $\frac{2\pi}5$. De berekening is een euclidische
deling van de hoek door $\pi$ in vermomming --- en de analoge
recepten voor $\arcsin$ (spiegel naar
$\intcc{-\frac\pi2}{\frac\pi2}$, periode $2\pi$) en $\arccos$
(spiegel naar $\intcc0\pi$) sturen \cref{exo:b1:functions:1} en
het stuksgewijze antwoord van \cref{exo:b1:functions:10}.
\end{example}

\begin{example}[Boogtangensen veilig optellen]\label{ex:b1:functions:arctansum}
Laten we bewijzen dat
\[
\arctan\frac12 + \arctan\frac15 + \arctan\frac18 = \frac\pi4 .
\]
Twee ingrediënten: de optelformule voor de tangens, en --- de stap
die beginners vergeten --- een \emph{lokalisering} van de som. Eerst geeft
$\tan(u + v) = \frac{\tan u + \tan v}{1 - \tan u\tan v}$ met
$u = \arctan\frac12$, $v = \arctan\frac15$
\[
\tan(u + v) = \frac{\frac12 + \frac15}{1 - \frac1{10}}
= \frac{7/10}{9/10} = \frac79,
\qquad\text{vervolgens}\qquad
\tan\Bigl(u + v + \arctan\frac18\Bigr)
= \frac{\frac79 + \frac18}{1 - \frac7{72}}
= \frac{65/72}{65/72} = 1 .
\]
Ten tweede ligt elk van de drie hoeken in $\intoo0{\frac\pi4}$
(hun argumenten zijn kleiner dan $1$), dus de som ligt in
$\intoo0{\frac{3\pi}4}$; de enige hoek daar met tangens $1$ is
$\frac\pi4$. Zonder de lokalisering zou het besluit
``$\frac\pi4$ op een veelvoud van $\pi$ na'' zijn --- een half bewijs. Dezelfde
tweestapsdiscipline stuurt \cref{exo:b1:functions:8} en
\cref{exo:b1:functions:12}.
\end{example}

\begin{example}[Een boogtangensvergelijking]\label{ex:b1:functions:arctaneq}
Los $\arctan x + \arctan 2x = \dfrac\pi4$ op. Lokaliseer eerst: het
linkerlid heeft het teken van $x$ (beide termen hebben dat), dus elke oplossing heeft
$x > 0$, en dan ligt de som in $\intoo0\pi$. Neem tangensen
(\omterm{def:b1:logic:inj}{injectief} op geen enkel interval van lengte $\pi$, maar gecombineerd met
de lokalisering volstaat dit): de optelformule geeft
\[
\tan\bigl(\arctan x + \arctan 2x\bigr)
= \frac{3x}{1 - 2x^2} = 1
\iff 2x^2 + 3x - 1 = 0
\iff x = \frac{-3 \pm \sqrt{17}}4 .
\]
De negatieve wortel wordt verworpen door de lokalisering; voor de
positieve kandidaat $x_0 = \frac{\sqrt{17} - 3}4 \approx 0.28$
ligt de som in $\intoo0\pi$ met tangens $1$, en de enige zulke
hoek is $\frac\pi4$ (in $\intoo{\frac\pi2}\pi$ is de tangens
negatief): $x_0$ is de unieke oplossing. Merk de vorm van het
argument op: tangensen nemen kan oplossingen creëren, nooit verliezen,
dus men lost de veeltermvergelijking op \emph{en filtert dan} via
lokalisering --- dezelfde discipline van voorwaartse controle als bij het kwadrateren
van een vergelijking.
\end{example}

\section{Hyperbolische functies}

\begin{definition}[$\cosh$, $\sinh$, $\tanh$]\label{def:b1:functions:hyperbolic}
Voor $x \in \R$:
\[
\cosh x = \frac{\eu^x + \eu^{-x}}{2},
\qquad
\sinh x = \frac{\eu^x - \eu^{-x}}{2},
\qquad
\tanh x = \frac{\sinh x}{\cosh x}
\]
(\emph{hyperbolische cosinus, sinus, tangens}\index{hyperbolische
functies}). $\cosh$ is even, $\sinh$ en $\tanh$ zijn oneven.
\end{definition}

\begin{proposition}[Basiseigenschappen]\label{prop:b1:functions:hyprules}
Voor alle $x, y \in \R$:
\begin{enumerate}
  \item $\cosh^2 x - \sinh^2 x = 1$;
  \item $\cosh' = \sinh$, $\;\sinh' = \cosh$, $\;\tanh' = 1 - \tanh^2
        = \dfrac{1}{\cosh^2}$;
  \item $\cosh(x+y) = \cosh x\cosh y + \sinh x \sinh y$ en
        $\sinh(x+y) = \sinh x \cosh y + \cosh x \sinh y$;
  \item $\sinh$ is een strikt stijgende bijectie van $\R$ op $\R$;
        $\cosh$ beperkt tot $\R_+$ is een strikt stijgende
        bijectie op $\intco{1}{+\infty}$; $\tanh$ is een strikt
        stijgende bijectie van $\R$ op $\intoo{-1}{1}$.
\end{enumerate}
\end{proposition}

\begin{proof}
(1) $\bigl(\frac{\eu^x + \eu^{-x}}{2}\bigr)^2 - \bigl(\frac{\eu^x -
\eu^{-x}}{2}\bigr)^2 = \frac{(\eu^{2x} + 2 + \eu^{-2x}) - (\eu^{2x} -
2 + \eu^{-2x})}{4} = 1$.

(2) Leid de definiërende formules af; voor $\tanh$ geeft de
quotiëntregel $\frac{\cosh^2 - \sinh^2}{\cosh^2}$, wat zowel
$1 - \tanh^2$ als $\frac{1}{\cosh^2}$ is via (1).

(3) Werk de rechterleden uit met de definities. In detail
voor de eerste formule:
\[
\cosh x\cosh y + \sinh x\sinh y
= \frac{(\eu^x + \eu^{-x})(\eu^y + \eu^{-y})
+ (\eu^x - \eu^{-x})(\eu^y - \eu^{-y})}{4} ;
\]
van de acht producten heffen de vier ``gemengde''
($\eu^{x-y}$, $\eu^{y-x}$) elkaar paarsgewijs op, terwijl
$\eu^{x+y}$ en $\eu^{-(x+y)}$ elk tweemaal verschijnen:
het totaal is $\frac{2\eu^{x+y} + 2\eu^{-(x+y)}}{4} = \cosh(x+y)$.
De sinusformule is identiek, met de gemengde termen die overleven
in plaats daarvan.

(4) $\sinh' = \cosh \geq 1 > 0$, dus $\sinh$ is strikt stijgend,
met limieten $\pm\infty$ (dominante term $\pm\frac{\eu^{\abs
x}}{2}$); de \omterm{def:b1:logic:statement}{bijectie-uitspraak} volgt dan uit de tussenwaarde\-stelling
(gebruikt op middelbareschoolniveau; systematische behandeling in
\cref{ch:b1:continuity}). Op $\R_+$ is $\cosh' = \sinh \geq 0$,
enkel nul in $0$: strikt stijgend van $\cosh 0 = 1$ tot
$+\infty$. En $\tanh' > 0$ met limieten $\pm 1$ in $\pm\infty$: in
detail,
\[
\tanh x = \frac{\eu^x - \eu^{-x}}{\eu^x + \eu^{-x}}
= \frac{1 - \eu^{-2x}}{1 + \eu^{-2x}}
\xrightarrow[x \to +\infty]{} 1 ,
\]
na deling van teller en noemer door $\eu^x$, en de oneven pariteit
geeft de limiet $-1$ in $-\infty$; de strikt stijgende
functie $\tanh$ beeldt dus $\R$ af op $\intoo{-1}1$.
\end{proof}

\begin{remark}[Waarom ``hyperbolisch''?]
Het punt $(\cosh t, \sinh t)$ loopt langs de tak $x > 0$ van de
hyperbool $x^2 - y^2 = 1$ (via
\cref{prop:b1:functions:hyprules}~(1)), precies zoals $(\cos t, \sin
t)$ langs de cirkel $x^2 + y^2 = 1$ loopt. Elke cirkelidentiteit heeft
een hyperbolische tegenhanger, met tekenveranderingen bestuurd door
$\sinh^2 \leftrightarrow -\sin^2$.
\end{remark}

\begin{example}[De optelformule voor $\tanh$]\label{ex:b1:functions:tanhadd}
Door de twee optelformules van
\cref{prop:b1:functions:hyprules}~(3) te delen door $\cosh x\cosh y$:
\[
\tanh(x + y)
= \frac{\sinh x\cosh y + \cosh x\sinh y}
{\cosh x\cosh y + \sinh x\sinh y}
= \frac{\tanh x + \tanh y}{1 + \tanh x\,\tanh y} ,
\]
de hyperbolische tegenhanger van de optelformule voor de tangens --- met een
$+$ waar de trigonometrie een $-$ heeft. Een dividend: aangezien $\abs{\tanh}
< 1$, is het rechterlid een ``snelheidsoptel''-regel die
$\intoo{-1}1$ nooit verlaat: als $u, v \in \intoo{-1}1$ dan is
$\frac{u + v}{1 + uv} \in \intoo{-1}1$ eveneens (schrijf $u = \tanh
a$, $v = \tanh b$, mogelijk door bijectiviteit, en lees de formule
achterstevoren). Dit algebraïsch nagaan, zonder \omterm{def:b1:functions:hyperbolic}{hyperbolische
functies}, is een licht pijnlijke oefening; parametriseren via
$\tanh$ maakt er één regel van --- dezelfde strategie die cirkel-
functies bieden voor de eenheidscirkel.
\end{example}

\begin{proposition}[Inverse hyperbolische functies]\label{prop:b1:functions:invhyp}
De inversen verleend door \cref{prop:b1:functions:hyprules}~(4) hebben
gesloten vormen:
\[
\operatorname{arsinh} x = \ln\bigl(x + \sqrt{x^2 + 1}\bigr)
\ (x \in \R),
\qquad
\operatorname{arcosh} x = \ln\bigl(x + \sqrt{x^2 - 1}\bigr)
\ (x \geq 1),
\]
\[
\operatorname{artanh} x = \frac 12 \ln\frac{1 + x}{1 - x}
\ (\abs x < 1),
\]
met afgeleiden $\frac{1}{\sqrt{x^2+1}}$, $\frac{1}{\sqrt{x^2-1}}$
($x > 1$) en $\frac{1}{1 - x^2}$ respectievelijk.
\end{proposition}

\begin{proof}
Voor $\operatorname{arsinh}$: los $x = \sinh y = \frac{\eu^y -
\eu^{-y}}{2}$ op. Met $u = \eu^y > 0$: $u^2 - 2xu - 1 = 0$, dus $u = x
+ \sqrt{x^2 + 1}$ (de wortel $x - \sqrt{x^2+1}$ is negatief), en $y =
\ln(x + \sqrt{x^2+1})$. De andere twee zijn identieke berekeningen
($u^2 - 2xu + 1 = 0$ voor $\operatorname{arcosh}$, met behoud van de wortel
$\geq 1$; een herschikking van twee regels voor $\operatorname{artanh}$).
Afgeleiden: leid de logaritmische uitdrukkingen af, bv.
\[
\bigl(\ln(x + \sqrt{x^2+1})\bigr)'
= \frac{1 + \frac{x}{\sqrt{x^2+1}}}{x + \sqrt{x^2+1}}
= \frac{1}{\sqrt{x^2 + 1}} . \qedhere
\]
\end{proof}

\begin{example}[Gesloten vormen aan het werk]\label{ex:b1:functions:arcoshtwo}
De oplossing van $\cosh t = 2$ met $t \geq 0$ is, via de gesloten
vorm, $t = \operatorname{arcosh} 2 = \ln(2 + \sqrt3) \approx
1.317$; de andere oplossing is $-t$, door de even pariteit --- en inderdaad
is $\ln(2 - \sqrt3) = \ln\frac1{2 + \sqrt3} = -\ln(2 + \sqrt3)$: de
twee wortels $u = \eu^t$ van de vierkantsvergelijking $u^2 - 4u + 1 = 0$ zijn
elkaars inverse, zoals hun product $1$ (Vieta) eist. Deze kleine
berekening toont het algemene patroon: hyperbolische vergelijkingen
worden vierkantsvergelijkingen in $\eu^t$, en de symmetrie $t \mapsto -t$
verschijnt als de symmetrie $u \mapsto 1/u$ van de vierkantsvergelijking ---
het onthouden waard bij het oplossen van \cref{exo:b1:functions:7}.
\end{example}

\begin{method}[De juiste primitieve vorm kiezen]\label{met:b1:functions:primitive}
De drie afgeleidepatronen om te onthouden voor integratie
(\cref{ch:b1:integration}):
\[
\int \frac{\dd x}{1 + x^2} = \arctan x + C,
\qquad
\int \frac{\dd x}{\sqrt{1 - x^2}} = \arcsin x + C,
\qquad
\int \frac{\dd x}{\sqrt{x^2 + 1}} = \operatorname{arsinh} x + C,
\]
en voor een algemene vierkantsuitdrukking herleidt men tot deze door kwadraat- afsplitsing en herschaling.
\end{method}

\begin{remark}[Waar dit hoofdstuk gebruikt wordt]\label{rem:b1:functions:whereused}
Dit hoofdstuk is de werkwoordenschat van alle komende analyse.
De groeischaal van \cref{prop:b1:functions:powerrules} beslist
convergentievragen doorheen
\cref{ch:b1:seq,ch:b1:series}; de afgeleideformules van
\cref{prop:b1:functions:arcderiv} en
\cref{prop:b1:functions:invhyp} zijn de primitieven die het vaakst
nodig zijn in \cref{ch:b1:integration}, via
\cref{met:b1:functions:primitive}; \omterm{def:b1:functions:hyperbolic}{hyperbolische functies}
parametriseren de oplossingen van de vergelijking $y'' = y$ in
\cref{ch:b1:diffeq} precies zoals cirkelfuncties die
van $y'' = -y$ parametriseren. De karakterisering van $\exp$ via $f' = f$,
$f(0) = 1$ (\cref{prop:b1:functions:expln}) is het eendimensionale
zaad van de theorie der lineaire differentiaalvergelijkingen, en de
kettinglijn $y = \cosh x$ keert terug bij de vlakke krommen van
\cref{ch:b1:curves}.
\end{remark}

\begin{remark}[Intermezzo: het programma van de inverse functies]\label{rem:b1:functions:interlude}
Dit hoofdstuk voerde één programma viermaal uit: beperk een functie
tot ze een strikt monotone bijectie wordt, benoem de inverse,
transporteer elke formule erdoorheen. Wat op elke stap \emph{gebruikt}
werd --- dat een continue strikt monotone functie op een
interval een bijectie is op een interval, en dat haar inverse
continu is, en dan afleidbaar weg van kritieke punten --- werd
op middelbareschoolkrediet geleend. De schuld wordt afbetaald binnen dit
volume: \cref{ch:b1:continuity} bewijst de \omterm{def:b1:logic:statement}{bijectie-uitspraak}
(de stelling van de monotone inverse, steunend op de tussenwaarde\-stelling),
en \cref{ch:b1:derivative} bewijst de afgeleideregel
$(f^{-1})' = 1/(f' \circ f^{-1})$ die stilzwijgend elke
formule van \cref{prop:b1:functions:arcderiv} en
\cref{prop:b1:functions:invhyp} voortbracht. Die hoofdstukken lezen met
$\arcsin$ en $\operatorname{arsinh}$ in gedachten --- als de uitgewerkte
voorbeelden die de theorie moest rechtvaardigen --- is de bedoelde weg
rond de schijnbare cirkelredenering.
\end{remark}

\section{Probleem: De vergelijking $x^y = y^x$}

\begin{problem}\label{pb:b1:functions:1}
Welke paren van positieve getallen voldoen aan $x^y = y^x$? Iedereen kent
één toevallig ogend voorbeeld, $2^4 = 4^2 = 16$; dit probleem toont
dat er niets toevalligs aan is. De hele vergelijking wordt bestuurd
door de variaties van de enkele functie
\[
f(t) = \frac{\ln t}{t} \qquad (t > 0),
\]
waarvan de studie oplevert: de volledige oplossingsverzameling (een diagonaal plus één
gekromde tak door $(\eu, \eu)$), een rationale parametrisatie van
de tak, het feit dat $(2, 4)$ haar \emph{enige} geheeltallige punt is
en de classificatie van al haar rationale punten --- plus, als
dividenden, de vergelijking van $\eu^\pi$ met $\pi^\eu$ en de
monotone convergentie van $\bigl(1 + \frac1n\bigr)^n$ naar $\eu$.
\index{machtsfunctie}

\medskip
\noindent\textbf{Deel I --- De functie $f(t) = \ln t / t$.}
\begin{enumerate}
  \item Verantwoord dat $f$ afleidbaar is op $\intoo0{+\infty}$,
        bereken $f'$, en stel de variatietabel op: $f$
        stijgt strikt op $\intoc0\eu$, daalt strikt op
        $\intco\eu{+\infty}$, met maximum $f(\eu) = \frac1\eu$.
  \item Bepaal de limieten van $f$ in $0^+$ en $+\infty$
        (\cref{prop:b1:functions:powerrules}), het teken van $f$
        (negatief op $\intoo01$, nul in $1$, positief daarboven),
        en schets de grafiek.
  \item Ga door directe berekening na dat $f(2) = f(4)$. (Houd deze
        gelijkheid in gedachten: het hele probleem groeit eruit.)
  \item Zij $c \in \R$. Bespreek, naargelang de waarde van $c$, het
        aantal oplossingen van $f(t) = c$: precies één voor $c \leq
        0$; precies twee (één in $\intoo1\eu$, één in
        $\intoo\eu{+\infty}$) voor $0 < c < \frac1\eu$; precies één
        voor $c = \frac1\eu$; geen voor $c > \frac1\eu$.
  \item Leid af dat $\eu^x > x^\eu$ voor elke $x > 0$ met $x \neq
        \eu$, en beslis in het bijzonder welke van $\eu^\pi$ en
        $\pi^\eu$ groter is.
\end{enumerate}

\medskip
\noindent\textbf{Deel II --- De vergelijking en haar kromme.} In dit
deel is $x, y > 0$.
\begin{enumerate}[resume]
  \item Toon aan dat $x^y = y^x \iff f(x) = f(y)$.
  \item Leid de structuur van de oplossingsverzameling af: alle diagonale
        paren $(x, x)$; en de \emph{niet-triviale} paren ($x \neq
        y$), die voldoen aan: beide coördinaten zijn $> 1$, en één
        ligt in $\intoo1\eu$ terwijl de andere in
        $\intoo\eu{+\infty}$ ligt.
  \item Parametriseer de niet-triviale paren: door $y = tx$ te schrijven met $t
        > 0$, $t \neq 1$, toon aan dat $x^y = y^x$ oplegt dat
        \[
        x(t) = t^{\frac1{t-1}},
        \qquad
        y(t) = t\,x(t) = t^{\frac{t}{t-1}},
        \]
        en dat omgekeerd elk zulk paar een oplossing is.
  \item Ga na dat $t = 2$ het paar $(2, 4)$ geeft, en bewijs de symmetrie
        $x(1/t) = y(t)$, $y(1/t) = x(t)$: het inverteren van de parameter
        verwisselt de twee coördinaten.
  \item Bepaal de limieten van $x(t)$ en $y(t)$ als $t \to 1$
        (beide streven naar $\eu$), als $t \to +\infty$ ($x \to 1$, $y
        \to +\infty$) en als $t \to 0^+$ ($x \to +\infty$, $y \to
        1$). Beschrijf de resulterende tak: een kromme asymptotisch aan
        de rechten $x = 1$ en $y = 1$, die de diagonaal kruist in
        $(\eu, \eu)$.
  \item Toon aan dat $t \mapsto x(t)$ strikt dalend is op
        $\intoo1{+\infty}$. \emph{(Bestudeer $h(t) = 1 - \frac1t - \ln
        t$, waarvan het teken de afgeleide van $\ln x(t) =
        \frac{\ln t}{t - 1}$ bestuurt.)}
  \item Besluit dat de niet-triviale oplossingen een strikt
        dalende bijectie $\varphi \colon \intoo1\eu \to
        \intoo\eu{+\infty}$ definiëren met $x^{\varphi(x)} = \varphi(x)^x$,
        en dat $\varphi$ een involutie van de tak is:
        $\varphi(\varphi(x)) = x$ overal waar beide leden gedefinieerd zijn.
\end{enumerate}

\medskip
\noindent\textbf{Deel III --- Geheeltallige en rationale punten.}
\begin{enumerate}[resume]
  \item Bewijs dat $(2, 4)$ en $(4, 2)$ de enige niet-triviale
        \emph{geheeltallige} oplossingen van $x^y = y^x$ zijn.
  \item Voor $n \in \N^*$, pas de parametrisatie toe met $t = 1 +
        \frac1n$ en toon aan dat
        \[
        x_n = \Bigl(1 + \frac1n\Bigr)^{n},
        \qquad
        y_n = \Bigl(1 + \frac1n\Bigr)^{n+1}
        \]
        een niet-triviale \emph{rationale} oplossing is voor elke $n$,
        met $(x_1, y_1) = (2, 4)$.
  \item Omgekeerd, zij $(x, y)$ een niet-triviale rationale oplossing
        met $y > x$, en schrijf $t = y/x = 1 + \frac rs$ in
        onvereenvoudigbare vorm ($r, s \in \N^*$). Gebruikmakend van $x = t^{s/r}$ en aannemend
        de uniciteit van de priemontbinding (vertrouwd van
        school; bewezen in \cref{ch:b1:arith}), toon aan dat $x$
        rationaal oplegt dat zowel $s$ als $s + r$ $r$-de machten
        van gehele getallen zijn.
  \item Toon met het binomium van Newton aan dat $a^r - b^r = r$ geen
        geheeltallige oplossingen $a > b \geq 1$ heeft wanneer $r \geq 2$, en
        besluit: de rationale oplossingen van $x^y = y^x$ zijn
        precies de paren $(x_n, y_n)$ van vraag 14 (en hun
        verwisselingen).
  \item Ga het geval $n = 2$ numeriek na: bereken $f(9/4)$ en
        $f(27/8)$ tot op vier decimalen en controleer dat ze overeenkomen.
\end{enumerate}

\medskip
\noindent\textbf{Deel IV --- Dividenden.}
\begin{enumerate}[resume]
  \item Gebruik vragen 7 en 11 om, zonder verdere
        berekening, te bewijzen dat de rij $x_n = \bigl(1 +
        \frac1n\bigr)^n$ strikt stijgend is met $x_n < \eu$,
        dat $y_n = \bigl(1 + \frac1n\bigr)^{n+1}$ strikt
        dalend is met $y_n > \eu$, en dat beide convergeren naar
        $\eu$. (De parameter $t_n = 1 + \frac1n$ daalt naar
        $1$.)
  \item Stel de algemene vergelijkingsregel op voor $1 < a < b$:
        als $\eu \leq a$ dan $a^b > b^a$; als $b \leq \eu$ dan
        $a^b < b^a$; en toon met de twee voorbeelden $(2, 3)$ en
        $(2, 5)$ aan dat in het gemengde geval $a < \eu < b$ beide
        uitkomsten werkelijk optreden.
  \item Veronderstel dat de involutie $\varphi$ van vraag 12
        afleidbaar is in $\eu$ (dat is ze). Leid de
        identiteit $\varphi(\varphi(x)) = x$ af in $x = \eu$ en leid
        $\varphi'(\eu) = -1$ af: de tak kruist de diagonaal
        loodrecht erop.
  \item Vind het unieke oplossingspaar met $y = 3x$, in gesloten
        vorm, en controleer het numeriek tot op vier decimalen via $f$.
  \item Bepaal onder de getallen $\sqrt2$, $\sqrt[3]3$, $\sqrt[4]4$,
        $\sqrt[5]5$, het grootste en identificeer de twee
        die gelijk zijn. \emph{(Vergelijk $n^{1/n} =
        \eu^{f(n)}$.)}
\end{enumerate}

\medskip
\noindent\textbf{Deel V --- Synthese.}
\begin{enumerate}[resume]
  \item Beschrijf de volledige oplossingsverzameling van $x^y = y^x$ in het
        kwartvlak $x, y > 0$ --- diagonaal plus tak, hun
        snijpunt $(\eu, \eu)$, de asymptoten, het geheeltallige
        punt $(2, 4)$, de rationale punten die accumuleren in
        $(\eu, \eu)$ --- in een vorm die je uit het hoofd zou kunnen schetsen.
  \item Waar precies gebruikte het probleem: (i) de groei-
        vergelijkingen van \cref{prop:b1:functions:powerrules}; (ii)
        de tussenwaardestelling (via de bijectie-
        uitspraken); (iii) de aangenomen uniciteit van de priem-
        ontbinding? Eén zin elk.
  \item Moraal, in een korte alinea: een enkele variatietabel
        loste een vergelijking in twee onbekenden op, classificeerde haar
        rationale punten, en bewees de monotone convergentie van
        $\bigl(1 + \frac1n\bigr)^n$ --- becommentarieer deze zuinigheid,
        en benoem waar elke draad later in het
        volume geïndustrialiseerd wordt (\cref{ch:b1:seq} voor de rij,
        \cref{ch:b1:derivative} voor variatietabellen,
        \cref{ch:b1:taylor} voor de precisie die de tabel mist).
\end{enumerate}
