\chapter{Rationale breuken}\label{ch:b1:fractions}

Een \omterm{def:b1:fractions:field}{rationale breuk} is een quotiënt van \omterm{def:b1:poly:def}{veeltermen}. De centrale
stelling van dit korte hoofdstuk --- de partieelbreuksplitsing\index{partieelbreuksplitsing} --- ontbindt elk
zulk quotiënt in een som van elementaire bouwstenen
$\frac{c}{(X - a)^k}$. Naast haar algebraïsche belang is zij de
standaardmachine om rationale functies te integreren
(\cref{ch:b1:integration}) en om bepaalde reeksen te sommeren
(\cref{ch:b1:series}).

\section{Het lichaam $K(X)$}

\begin{definition}\label{def:b1:fractions:field}
Een \emph{rationale breuk}\index{rationale breuk} over $K$ ($= \R$
of $\C$) is een quotiënt $F = \frac{A}{B}$ met $A, B \in K[X]$, $B \neq
0$; twee quotiënten $\frac AB$ en $\frac{A'}{B'}$ worden geïdentificeerd
wanneer $AB' = A'B$. Elke breuk heeft een \emph{gereduceerde vorm} met
$\gcd(A, B) = 1$, uniek op constanten na. Met de natuurlijke bewerkingen
is de \omterm{def:b1:logic:sets}{verzameling} $K(X)$ van rationale breuken een \omterm{def:b1:structures:field}{lichaam}.

De \emph{polen}\index{pool} van $F$ (in gereduceerde vorm) zijn de
wortels van $B$; de \emph{orde} van een pool is haar \omterm{def:b1:poly:derivative}{multipliciteit} als
wortel van $B$. De \emph{graad} van $F$ is $\deg F = \deg A - \deg B \in
\Z \cup \{-\infty\}$.
\end{definition}

\begin{example}[Polen, orden en graad aflezen]\label{ex:b1:fractions:vocab}
Zij $F = \dfrac{X^3 - X}{X^4 - 2X^3 + X^2}$. Ontbind beide lagen:
teller $X(X-1)(X+1)$, noemer $X^2(X-1)^2$; schrap de gemeenschappelijke
factor $X(X-1)$:
\[
F = \frac{X + 1}{X(X - 1)} \quad\text{(gereduceerde vorm).}
\]
\omterm{def:b1:fractions:field}{Polen}: $0$ en $1$, beide \emph{enkelvoudig} --- de orden worden op
de gereduceerde noemer afgelezen, dus de schijnbare dubbele wortels
van de oorspronkelijke noemer zijn irrelevant. Graad: $\deg F = 1 - 2
= -1$, asymptotisch zichtbaar ($xF(x) \to 1$ als $x \to \infty$). De
graad gedraagt zich als de graad van een \omterm{def:b1:poly:def}{veelterm} ($\deg FG = \deg F +
\deg G$, $\deg(F + G) \leq \max$), een boekhoudregel die hieronder
voortdurend gebruikt wordt in de zoektocht naar coëfficiënten: elk
argument met de ``limiet van $xF(x)$'' is een verkapte graadtelling.
\end{example}

\begin{proposition}[Geheel deel]\label{prop:b1:fractions:integerpart}
Elke $F = \frac AB \in K(X)$ is op unieke wijze $F = E + \frac RB$ met $E
\in K[X]$ (het \emph{geheel deel}, of veeltermdeel, van $F$) en
$\deg R < \deg B$. Men heeft $E \neq 0$ dan en slechts dan als $\deg F
\geq 0$.
\end{proposition}

\begin{proof}
Euclidische deling $A = BE + R$ (\cref{thm:b1:poly:division}), gedeeld
door $B$. Uniciteit: als $E + \frac RB = E' + \frac{R'}{B}$, dan is $(E -
E')B = R' - R$ met $\deg(R' - R) < \deg B$, wat $E = E'$ afdwingt en
vervolgens $R = R'$.
\end{proof}

\begin{example}[Eerst reduceren, dan delen]\label{ex:b1:fractions:integerpart}
Bepaal het geheel deel van $F = \dfrac{X^3 + 1}{X^2 - 1}$. Blind delen:
$X^3 + 1 = (X^2 - 1)X + (X + 1)$, dus $F = X +
\frac{X + 1}{X^2 - 1}$. Maar de breuk was niet gereduceerd: $X^3 + 1
= (X + 1)(X^2 - X + 1)$ en $X^2 - 1 = (X+1)(X-1)$ delen de factor
$X + 1$, en
\[
F = \frac{X^2 - X + 1}{X - 1} = X + \frac{1}{X - 1} :
\]
hetzelfde geheel deel $X$, maar het breukgedeelte valt samen tot één
enkele bouwsteen, en de ``\omterm{def:b1:fractions:field}{pool}'' in $-1$ was nooit een \omterm{def:b1:fractions:field}{pool}. Reduceer
altijd tot de laagste termen voordat je \omterm{def:b1:fractions:field}{polen} zoekt: de \omterm{def:b1:fractions:field}{polen} van $F$
zijn de wortels van de \emph{gereduceerde} noemer. (Het geheel deel is
ongevoelig voor de vereenvoudiging, zoals de uniciteit in
\cref{prop:b1:fractions:integerpart} garandeert.)
\end{example}

\section{Partieelbreuksplitsing over $\C$}

\begin{theorem}[Splitsing over $\C$]\label{thm:b1:fractions:complex}
Zij $F = \frac AB \in \C(X)$ in gereduceerde vorm, met $B = c\,(X -
a_1)^{m_1} \cdots (X - a_r)^{m_r}$. Dan is $F$, op unieke wijze,
\[
F = E + \sum_{i=1}^{r} \sum_{k=1}^{m_i}
\frac{c_{i,k}}{(X - a_i)^k},
\qquad E \in \C[X],\ c_{i,k} \in \C ,
\]
waarbij $E$ het geheel deel van $F$ is.
\end{theorem}

\begin{proof}
Volgens \cref{prop:b1:fractions:integerpart} mogen we aannemen dat
$\deg A < \deg B$ en de somsplitsing bewijzen met $E = 0$.

\emph{De \omterm{def:b1:fractions:field}{polen} scheiden.} Schrijf $B = (X - a_1)^{m_1} B_1$ met
$B_1(a_1) \neq 0$. De \omterm{def:b1:poly:def}{veeltermen} $(X-a_1)^{m_1}$ en $B_1$ zijn
\omterm{cor:b1:arith:bezout}{relatief priem} (geen gemeenschappelijke wortel), dus door Bézout in
$\C[X]$ (zie de opmerking in \cref{ch:b1:poly}) bestaan er $U_0, V_0$
met $U_0 (X-a_1)^{m_1} + V_0 B_1 = 1$; door met $A$ te vermenigvuldigen
en $U = AU_0$, $V = AV_0$ te stellen, en vervolgens door $B$ te delen:
\[
\frac AB = \frac{V}{(X - a_1)^{m_1}} + \frac{U}{B_1} .
\]
De graadvoorwaarden kunnen worden afgedwongen: deel $V$ door
$(X-a_1)^{m_1}$, zeg $V = (X-a_1)^{m_1}Q + V_1$ met $\deg V_1 <
m_1$, en neem het quotiënt op in de tweede term ($U_1 = U +
QB_1$):
\[
\frac AB = \frac{V_1}{(X - a_1)^{m_1}} + \frac{U_1}{B_1},
\qquad \deg V_1 < m_1 ;
\]
het vergelijken van graden ($\deg A < \deg B$ en $\deg V_1 < m_1$)
dwingt ook $\deg U_1 < \deg B_1$ af. Iteratie op $\frac{U_1}{B_1}$,
\omterm{def:b1:fractions:field}{pool} na \omterm{def:b1:fractions:field}{pool}, herleidt alles tot het geval met één \omterm{def:b1:fractions:field}{pool} hieronder.

\emph{Eén \omterm{def:b1:fractions:field}{pool}.} Voor $\frac{A}{(X-a)^m}$ met $\deg A < m$: ontwikkel $A$
in machten van $(X - a)$, $A = \sum_{j=0}^{m-1} \alpha_j (X - a)^j$
(Taylor-ontwikkeling van een \omterm{def:b1:poly:def}{veelterm}, zoals in het bewijs van
\cref{prop:b1:poly:multiplicity}); delen geeft precies de bouwstenen
$\frac{\alpha_j}{(X-a)^{m-j}}$. Concreet, voor $\frac{X^2 +
1}{(X+2)^3}$: door $X = Y - 2$ te substitueren,
\[
X^2 + 1 = (Y - 2)^2 + 1 = Y^2 - 4Y + 5 ,
\qquad\text{dus}\qquad
\frac{X^2+1}{(X+2)^3} = \frac1{Y} - \frac4{Y^2} + \frac5{Y^3}
\]
met $Y = X + 2$: de drie bouwstenen verschijnen door een eenvoudige
deling van de verschoven ontwikkeling --- de snelste weg wanneer een
enkele \omterm{def:b1:fractions:field}{pool} van hoge orde in het spel is, en de aanbevolen weg voor
\cref{exo:b1:fractions:3}.

\emph{Uniciteit.} Stel dat twee splitsingen samenvallen; hun verschil
is een identiteit $0 = \sum_{i,k} \frac{d_{i,k}}{(X -
a_i)^k}$. Vermenigvuldig door met $(X - a_1)^{m_1}$: elke term krijgt
een factor die nul wordt in $a_1$ \emph{behalve} die met
$i = 1$, $k = m_1$, waarvan de coëfficiënt $d_{1,m_1}$ wordt plus
termen die ten minste één factor $(X - a_1)$ dragen. Invullen in
$a_1$ (geoorloofd: na de vermenigvuldiging blijft er geen \omterm{def:b1:fractions:field}{pool} over in
$a_1$) geeft $d_{1,m_1} = 0$. Nu de hoogste coëfficiënt weg is,
herhaal met $(X - a_1)^{m_1 - 1}$, enzovoort naar beneden tot $k = 1$;
ga dan over naar de volgende \omterm{def:b1:fractions:field}{pool}. Alle $d_{i,k}$ worden nul: de
splitsing is uniek.
\end{proof}

\begin{method}[De coëfficiënten berekenen]\label{met:b1:fractions:compute}
In de praktijk vermijd je Bézout; combineer:
\begin{enumerate}
  \item \emph{afdekken voor de hoogste macht}: de coëfficiënt van
        $\frac{1}{(X-a)^m}$ (met $m$ de orde van de \omterm{def:b1:fractions:field}{pool} $a$) is
        \[
        c_{a,m} = \Bigl[\,(X - a)^m F\,\Bigr]_{X = a};
        \]
        voor een \emph{enkelvoudige} \omterm{def:b1:fractions:field}{pool} van $F = \frac AB$ is dit
        $\frac{A(a)}{B'(a)}$;
  \item \emph{evaluaties} in geschikte punten en \emph{limieten} van
        $xF(x)$ als $x \to \infty$ om lineaire relaties voor de
        overige coëfficiënten te verzamelen;
  \item \emph{pariteit}- of conjugatiesymmetrieën, indien aanwezig, om
        het werk te halveren.
\end{enumerate}
\end{method}

\begin{remark}[Veelvoorkomende valkuilen bij partieelbreuken]\label{rem:b1:fractions:pitfalls}
\begin{enumerate}
  \item \emph{Het geheel deel overslaan.} De ontbinding in bouwstenen
        geldt voor breuken van graad $< 0$; wanneer $\deg A \geq
        \deg B$, deel dan eerst
        (\cref{prop:b1:fractions:integerpart}), anders levert de
        zoektocht naar coëfficiënten tegenspraken op.
  \item \emph{Afdekken buiten zijn geldigheidsgebied.} Vermenigvuldigen
        met $(X - a)^k$ en invullen in $a$ levert de coëfficiënt
        alleen voor $k = m$, de \emph{volledige} orde van de \omterm{def:b1:fractions:field}{pool}; de
        coëfficiënten van lagere orde vereisen andere relaties
        (limieten, evaluaties) --- zie \cref{ex:b1:fractions:multiple}.
  \item \emph{Vergeten te reduceren.} \omterm{def:b1:fractions:field}{Polen} worden op de
        \emph{gereduceerde} vorm afgelezen; een gemeenschappelijke
        factor tussen teller en noemer schept spookpolen
        (\cref{ex:b1:fractions:integerpart}).
  \item \emph{Verkeerde vormen van de bouwstenen over $\R$.} Boven een
        onontbindbare kwadratische \omterm{def:b1:poly:def}{veelterm} zijn de tellers
        \emph{affien} ($\alpha X + \beta$), niet constant; alleen
        $\frac{c}{X^2 + 1}$ schrijven verliest oplossingen --- de juiste
        vormen worden voorgeschreven door
        \cref{thm:b1:fractions:real}, nooit geïmproviseerd.
\end{enumerate}
\end{remark}

\begin{proof}[Bewijs van de formule voor een enkelvoudige pool]
Nabij een enkelvoudige \omterm{def:b1:fractions:field}{pool} $a$: $B = (X - a) Q$ met $Q(a) \neq 0$, en
$B' = Q + (X - a) Q'$, dus $B'(a) = Q(a)$. De afdekwaarde is
$\frac{A(a)}{Q(a)} = \frac{A(a)}{B'(a)}$.
\end{proof}

\begin{example}\label{ex:b1:fractions:simple}
Splits $F = \dfrac{1}{X(X-1)(X-2)}$. Drie enkelvoudige \omterm{def:b1:fractions:field}{polen}; afdekken
in elk:
\[
c_0 = \frac{1}{(0-1)(0-2)} = \frac12,
\quad
c_1 = \frac{1}{1 \times (1 - 2)} = -1,
\quad
c_2 = \frac{1}{2 \times 1} = \frac12,
\]
dus $F = \dfrac{1/2}{X} - \dfrac{1}{X-1} + \dfrac{1/2}{X-2}$.
\emph{Controle in $X = 3$:} rechtstreeks, $F(3) = \frac{1}{3\cdot2\cdot1} =
\frac16$; uit de splitsing, $\frac{1/2}{3} - \frac{1}{2} +
\frac{1/2}{1} = \frac16 - \frac12 + \frac12 = \frac16$.
\end{example}

\begin{example}[Meervoudige pool]\label{ex:b1:fractions:multiple}
Splits $F = \dfrac{X}{(X-1)^2 (X+1)}$. Vorm:
$\frac{a}{(X-1)^2} + \frac{b}{X-1} + \frac{c}{X+1}$.
\begin{itemize}
  \item Afdekken in de dubbele \omterm{def:b1:fractions:field}{pool}: $a = \bigl[\frac{X}{X+1}\bigr]_{X=1}
        = \frac12$.
  \item Afdekken in $-1$: $c = \bigl[\frac{X}{(X-1)^2}\bigr]_{X=-1} =
        -\frac14$.
  \item Limiet van $xF(x)$ in $\infty$: $0 = b + c$, dus $b = \frac14$.
\end{itemize}
\[
F = \frac{1/2}{(X-1)^2} + \frac{1/4}{X-1} - \frac{1/4}{X+1} .
\]
\emph{Controle in $X = 0$:} $F(0) = 0$ en $\frac12 - \frac14 - \frac14
= 0$.
\end{example}

\section{Splitsing over $\R$}

\begin{theorem}[Splitsing over $\R$]\label{thm:b1:fractions:real}
Zij $F \in \R(X)$ in gereduceerde vorm met noemer
\[
B = c \prod_i (X - a_i)^{m_i} \prod_j (X^2 + p_j X + q_j)^{n_j}
\qquad (p_j^2 - 4q_j < 0).
\]
Dan splitst $F$ uniek als zijn geheel deel plus termen
\[
\frac{c_{i,k}}{(X - a_i)^k} \quad (1 \leq k \leq m_i),
\qquad
\frac{\alpha_{j,l}\, X + \beta_{j,l}}{(X^2 + p_j X + q_j)^{l}}
\quad (1 \leq l \leq n_j),
\]
met reële coëfficiënten.
\end{theorem}

\begin{proof}
Splits over $\C$ (\cref{thm:b1:fractions:complex}). Aangezien $F$
reëel is, is de coëfficiënt boven de \omterm{def:b1:fractions:field}{pool} $\conj a$ (bij elke orde) de
toegevoegd complexe van de coëfficiënt boven $a$ (pas conjugatie toe op
de splitsing en beroep je op de uniciteit). Groepeer elk toegevoegd
complex paar:
\[
\frac{c}{(X - z)^l} + \frac{\conj c}{(X - \conj z)^l}
= \frac{c\,(X - \conj z)^l + \conj c\,(X - z)^l}{\bigl(X^2 - 2\Re(z)X
+ \abs z^2\bigr)^{l}},
\]
waarvan de teller zijn eigen toegevoegd complexe is, dus reëel, van
graad $\leq l$; het afsplitsen van veelvouden van de reële
kwadratische \omterm{def:b1:poly:def}{veelterm} verlaagt hem tot graad $\leq 1$ bij elk niveau
$l$ (een kleine dalende inductie). Reële \omterm{def:b1:fractions:field}{polen} behouden hun reële
coëfficiënten (conjugatie laat ze vast). De uniciteit volgt uit de
uniciteit over $\C$.
\end{proof}

\begin{example}[De koppeling van de toegevoegd complexe polen bekeken]\label{ex:b1:fractions:pairing}
Het mechanisme van het bewijs, op het kleinste geval: over $\C$ zijn
de \omterm{def:b1:fractions:field}{polen} van $\frac1{X^2+1}$ de punten $\pm\iu$, met afdekcoëfficiënten
$\frac1{2\iu}$ in $\iu$ en $\frac1{-2\iu}$ in $-\iu$ --- toegevoegd
complexe van elkaar, zoals de stelling voorspelt:
\[
\frac{1}{X^2 + 1}
= \frac{1/(2\iu)}{X - \iu} - \frac{1/(2\iu)}{X + \iu} .
\]
Herschikken over de gemeenschappelijke noemer:
\[
\frac{1}{2\iu}\cdot\frac{(X + \iu) - (X - \iu)}{X^2 + 1}
= \frac{1}{2\iu}\cdot\frac{2\iu}{X^2 + 1} = \frac{1}{X^2+1} :
\]
de imaginaire delen heffen op en de reële bouwsteen verschijnt
onbeschadigd opnieuw. Voor reële integranden verlaat men gewoonlijk
nooit $\R$ --- maar bij het evalueren van \emph{sommen} in complexe
punten (zoals het weekendprobleem doet met \omterm{def:b1:complex:unity}{eenheidswortels}) zijn de
complexe bouwstenen de natuurlijke munt, en deze koppeling is de
wisselkoers tussen de twee splitsingen.
\end{example}

\begin{example}\label{ex:b1:fractions:real}
Splits $F = \dfrac{4}{(X^2+1)(X-1)^2}$ over $\R$. Vorm:
$\frac{aX + b}{X^2 + 1} + \frac{c}{(X-1)^2} + \frac{d}{X - 1}$.
Afdekken in de dubbele \omterm{def:b1:fractions:field}{pool}: $c = \bigl[\frac{4}{X^2+1}\bigr]_{X=1} =
2$. Afdekken in de complexe \omterm{def:b1:fractions:field}{pool} $\iu$ (teller boven $X^2 + 1$
geëvalueerd via de complexe splitsing, of rechtstreeks): vermenigvuldig
met $X^2 + 1$ en stel $X = \iu$:
\[
a\iu + b = \frac{4}{(\iu - 1)^2} = \frac{4}{-2\iu} = 2\iu ,
\]
dus $a = 2$, $b = 0$. Limiet van $xF(x)$ in oneindig: $0 = a + d$, dus
$d = -2$. Bijgevolg
\[
F = \frac{2X}{X^2+1} + \frac{2}{(X-1)^2} - \frac{2}{X-1} .
\]
\emph{Controle in $X = 0$:} $F(0) = 4$ en $0 + 2 + 2 = 4$.
\end{example}

\begin{example}[Een kubische noemer, van begin tot eind]\label{ex:b1:fractions:cubic}
Splits $F = \dfrac{1}{X^3 + 1}$ over $\R$. Ontbind eerst: $X^3 +
1 = (X + 1)(X^2 - X + 1)$, waarbij de kwadratische \omterm{def:b1:poly:def}{veelterm}
discriminant $-3 < 0$ heeft. Vorm: $\frac{a}{X+1} + \frac{bX +
c}{X^2 - X + 1}$. Afdekken in de enkelvoudige \omterm{def:b1:fractions:field}{pool} $-1$: $a =
\bigl[\frac1{X^2 - X + 1}\bigr]_{X=-1} = \frac13$. Limiet van $xF(x)$
in oneindig: $0 = a + b$, dus $b = -\frac13$. Evaluatie in $X = 0$:
$1 = a + c$, dus $c = \frac23$. Bijgevolg
\[
\frac{1}{X^3+1} = \frac13\Bigl(\frac{1}{X+1}
+ \frac{-X + 2}{X^2 - X + 1}\Bigr) ,
\]
bevestigd in $X = 1$: linkerlid $\frac12$, rechterlid
$\frac13\bigl(\frac12 + 1\bigr) = \frac12$. Merk de zuinigheid op:
drie onbekenden, drie goedkope lineaire feiten (één afdekking, één
limiet, één evaluatie), geen enkele ontwikkeling --- de werkwijze
van \cref{met:b1:fractions:compute} in haar zuivere vorm.
\end{example}

\begin{remark}[Waar het voor dient]
Eenmaal gesplitst, integreert een rationale functie term voor term:
bouwstenen $\frac{1}{(x-a)^k}$ hebben elementaire primitieven, en
bouwstenen $\frac{\alpha x + \beta}{(x^2 + px + q)^l}$ herleiden tot
$\ln$ en $\arctan$ (\cref{ch:b1:integration}). Telescoperende sommen
zijn de andere standaardtoepassing (\cref{exo:b1:fractions:8}).
\end{remark}

\begin{remark}[Waar dit hoofdstuk gebruikt wordt]\label{rem:b1:fractions:whereused}
Partieelbreuken zijn vóór alles een \emph{voorbewerkings}stap: het
integratiehoofdstuk (\cref{ch:b1:integration}) voert elke rationale
integrand door \cref{thm:b1:fractions:real} vóór het integreren, en
het reeksenhoofdstuk (\cref{ch:b1:series}) telescopeert rationale
termen precies zoals in \cref{exo:b1:fractions:8} en in het
weekendprobleem hieronder --- dat de techniek helemaal doorvoert tot
$\sum 1/k^2 = \pi^2/6$. De logaritmische afgeleide $P'/P = \sum
m_i/(X - a_i)$ (\cref{exo:b1:poly:11}) duikt weer op wanneer de ligging
van wortels bestudeerd wordt. Voorbij dit deel ligt de splitsing van
$1/\chi(X)$ voor een \omterm{def:b1:diffeq:linear2}{karakteristieke veelterm} $\chi$ aan de basis van
de berekening van matrixmachten en van Laplace-transformaties in het
deel van jaar 2: de bouwstenen $\frac{c}{(X - a)^k}$ zijn de
algebraïsche schaduw van de oplossingen $t^{k-1}\eu^{at}$ die men
ontmoet in \cref{ch:b1:diffeq}.
\end{remark}

(gemarkeerde punten): de oplossingen van $F = \lambda$ verstrengelen
met de \omterm{def:b1:fractions:field}{polen}.}
\end{omfigure}

\section{Oefeningen}

\begin{exercise}[$\star$]\label{exo:b1:fractions:1}
Splits over $\R$: $\dfrac{1}{X^2 - 1}$;
$\;\dfrac{X}{X^2 - 3X + 2}$; $\;\dfrac{X^2 + 1}{X(X-1)}$ \emph{(let op
het geheel deel)}.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:fractions:2}
Splits over $\R$: $\dfrac{1}{X(X^2 + 1)}$ en
$\dfrac{X^3}{X^2 + X + 1}$.
\end{exercise}

\begin{exercise}[$\star$]\label{exo:b1:fractions:3}
Splits $\dfrac{1}{X^2(X - 1)}$ en $\dfrac{X + 1}{(X - 1)^3}$
\emph{(substitueer voor de tweede $Y = X - 1$)}.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:fractions:4}
Splits over $\C$, daarna over $\R$: $\dfrac{1}{X^4 - 1}$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:fractions:5}
Splits over $\R$: $\dfrac{X^2}{(X^2 + 1)^2}$, en leid daaruit een
primitieve af van $x \mapsto \dfrac{x^2}{(x^2+1)^2}$, gegeven dat
$\int \frac{\dd x}{(x^2+1)^2} = \frac12\bigl(\arctan x +
\frac{x}{x^2+1}\bigr) + C$.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:fractions:6}
Splits voor $n \in \N^*$ de breuk $F_n = \dfrac{n!}{X(X+1)\cdots(X+n)}$
\emph{(enkelvoudige \omterm{def:b1:fractions:field}{polen} in $0, -1, \dots, -n$; gebruik de
afdekformule en herken \omterm{def:b1:counting:objects}{binomiaalcoëfficiënten})}.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:fractions:7}
Bewijs, met behulp van de identiteit van \cref{exo:b1:poly:11} voor $P
= X^n - 1$, dat
\[
\sum_{k=0}^{n-1} \frac{1}{X - \omega^k} = \frac{n X^{n-1}}{X^n - 1},
\qquad \omega = \eu^{2\iu\pi/n},
\]
en evalueer beide leden in $X = 2$ voor $n = 4$ als controle.
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:fractions:8}
Splits $\dfrac{1}{k(k+1)(k+2)}$ en bereken
\[
S_n = \sum_{k=1}^{n} \frac{1}{k(k+1)(k+2)},
\qquad\text{en vervolgens}\qquad
\lim_{n \to \infty} S_n .
\]
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:fractions:9}
Zij $P \in \R[X]$ \omterm{def:b1:poly:def}{monisch} van graad $n$ met $n$ verschillende reële
wortels $x_1 < \dots < x_n$. Bewijs dat
\[
\sum_{i=1}^{n} \frac{1}{P'(x_i)} = 0 \quad (n \geq 2),
\qquad
\sum_{i=1}^{n} \frac{x_i^{\,n-1}}{P'(x_i)} = 1 .
\]
\emph{Aanwijzing: splits $\frac{X^m}{P}$ voor $m \leq n - 1$ en kijk
naar het afnemen van de coëfficiënt in oneindig --- of gebruik
\omterm{thm:b1:poly:lagrange}{Lagrange-interpolatie} (\cref{thm:b1:poly:lagrange}) van de monoom $X^m$
in de knopen $x_i$.}
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:fractions:10}
Splits $\dfrac{1}{X(X+1)^2}$ over $\R$. Leid, aannemend dat
$\sum_{k \geq 1} \frac1{k^2} = \frac{\pi^2}6$ (bewezen in het
weekendprobleem van dit hoofdstuk), af dat
\[
\sum_{n=1}^{\infty} \frac{1}{n(n+1)^2} = 2 - \frac{\pi^2}{6} .
\]
\end{exercise}

\begin{exercise}[$\star\star$]\label{exo:b1:fractions:11}
Splits over $\R$: $\dfrac{1}{(X^2+1)(X^2+4)}$, daarna
$\dfrac{X^2}{(X^2+1)(X^2+4)}$. \emph{Aanwijzing: beide noemers zijn
\omterm{def:b1:poly:def}{veeltermen} in $X^2$: splits eerst $\frac1{(Y+1)(Y+4)}$.}
\end{exercise}

\begin{exercise}[$\star\star\star$]\label{exo:b1:fractions:12}
(Seculaire vergelijkingen) Zij $F = \sum_{i=1}^{r} \frac{c_i}{X - p_i}$
met $p_1 < p_2 < \dots < p_r$ reëel en alle $c_i > 0$.
\begin{enumerate}
  \item Toon aan dat $F$ strikt dalend is op elk interval van zijn
        domein, en geef de limieten in $\pm\infty$ en aan beide
        zijden van elke \omterm{def:b1:fractions:field}{pool}.
  \item Leid af dat voor elke $\lambda > 0$ de vergelijking $F(x) =
        \lambda$ precies $r$ reële oplossingen heeft, één in elk
        interval $\intoo{p_i}{p_{i+1}}$ en één voorbij $p_r$.
        (Zulke vergelijkingen sturen eigenwaardeverstoringen; de
        tussenwaardestelling wordt hier op middelbareschoolniveau
        gebruikt en bewezen in \cref{ch:b1:continuity}.)
\end{enumerate}
\end{exercise}

\section{Probleem: partieelbreuken als motor}

\begin{problem}\label{pb:b1:fractions:1}
Partieelbreuksplitsing lijkt op boekhouden; dit probleem toont dat het
een motor is. Gevoed met de breuk $\frac1{X(X+1)\cdots(X+k)}$
telescopeert zij hele families van sommen in gesloten vorm; gevoed met
$\frac1{X^n - 1}$ produceert zij goniometrische identiteiten zoals
\[
\sum_{k=1}^{n-1}\frac{1}{\sin^2\frac{k\pi}{n}} = \frac{n^2-1}{3} ;
\]
en, één stap verder gedreven, perst die identiteit een van de meest
gevierde formules in de wiskunde uit, die van Euler
\[
\sum_{k=1}^{\infty}\frac1{k^2} = \frac{\pi^2}{6}
\]
--- hier verkregen met niets meer dan de algebra van dit hoofdstuk en
middelbareschoolgoniometrie.\index{telescoperen}\index{Baselprobleem}
Overal is $\omega = \eu^{2\iu\pi/n}$; limieten van rijen worden op
middelbareschoolniveau gebruikt (\cref{ch:b1:seq} formaliseert ze).

\medskip
\noindent\textbf{Deel I --- De telescoop.}
\begin{enumerate}
  \item Splits $\frac1{X(X+1)}$ en bereken $\sum_{n=1}^{N}
        \frac1{n(n+1)}$ exact; besluit dat de som naar
        $1$ neigt.
  \item Idem voor $\frac1{X(X+2)}$: toon aan dat $\sum_{n=1}^{N}
        \frac1{n(n+2)} = \frac12\bigl(\frac32 - \frac1{N+1} -
        \frac1{N+2}\bigr) \to \frac34$. (Met een tussenruimte
        overleven \emph{twee} randtermen aan elk uiteinde.)
  \item Formaliseer het mechanisme: als $F(X) = G(X) - G(X+1)$ voor
        een of andere rationale $G$ zonder \omterm{def:b1:fractions:field}{polen} in $\intco1{+\infty}$,
        dan is $\sum_{n=1}^N F(n) = G(1) - G(N+1)$. Vind de
        waarde $\frac14$ van \cref{exo:b1:fractions:8} terug door de
        getuige $G$ voor $F = \frac1{X(X+1)(X+2)}$ te vertonen.
  \item Bewijs de algemene faculteitstelescoop: voor $k \geq 1$,
        \[
        \frac{1}{X(X+1)\cdots(X+k)}
        = \frac1k\biggl(\frac{1}{X(X+1)\cdots(X+k-1)}
        - \frac{1}{(X+1)\cdots(X+k)}\biggr),
        \]
        en leid af dat
        \[
        \sum_{n=1}^{\infty}\frac{1}{n(n+1)\cdots(n+k)}
        = \frac{1}{k \cdot k!} .
        \]
        Controleer het geval $k = 2$ tegen vraag 3.
  \item Evalueer de splitsing van \cref{exo:b1:fractions:6}
        in goedgekozen punten om te bewijzen dat
        \[
        \sum_{j=0}^{n}(-1)^j\binom nj\,\frac1{j+1}
        = \frac1{n+1},
        \qquad
        \sum_{j=0}^{n}(-1)^j\binom nj\,\frac1{j+2}
        = \frac1{(n+1)(n+2)} .
        \]
\end{enumerate}

\medskip
\noindent\textbf{Deel II --- De breuk $1/(X^n - 1)$.}
\begin{enumerate}[resume]
  \item Toon met de afdekformule aan dat
        \[
        \frac{1}{X^n - 1}
        = \frac1n\sum_{k=0}^{n-1}\frac{\omega^k}{X - \omega^k} .
        \]
  \item Twee plausibiliteitscontroles: verifieer de formule
        rechtstreeks voor $n = 2$, en toon aan dat de som van de $n$
        coëfficiënten nul is voor $n \geq 2$ --- verklaar waarom dat
        zo moet zijn (beschouw $xF(x)$ als $x \to \infty$).
  \item Leid met afdekken de identiteit van
        \cref{exo:b1:fractions:7} opnieuw af:
        $\frac{nX^{n-1}}{X^n-1} = \sum_k \frac1{X - \omega^k}$.
  \item Groepeer toegevoegd complexe \omterm{def:b1:fractions:field}{polen} om de reële splitsing te
        bewijzen: met $\theta_k = \frac{2k\pi}n$,
        \[
        \frac{\omega^k}{X - \omega^k} +
        \frac{\omega^{n-k}}{X - \omega^{n-k}}
        = \frac{2\cos\theta_k\,X - 2}
        {X^2 - 2\cos\theta_k\,X + 1} ,
        \]
        en schrijf de volledige reële splitsing van
        $\frac1{X^n-1}$ op (onderscheid $n$ oneven en $n$ even).
  \item Specialiseer naar $n = 4$ en controleer tegen
        \cref{exo:b1:fractions:4}.
\end{enumerate}

\medskip
\noindent\textbf{Deel III --- Goniometrische sommen, en Eulers
$\pi^2/6$.} Zij $P = 1 + X + \dots + X^{n-1}$, waarvan de wortels
$\omega, \omega^2, \dots, \omega^{n-1}$ zijn (alle enkelvoudig).
\begin{enumerate}[resume]
  \item Bewijs, met behulp van $\frac{P'}P = \sum_{k=1}^{n-1}\frac1{X
        - \omega^k}$ (\cref{exo:b1:poly:11}), dat
        \[
        \sum_{k=1}^{n-1}\frac{1}{1 - \omega^k} = \frac{n-1}2 .
        \]
  \item Bewijs $\dfrac1{1 - \eu^{\iu\theta}} = \dfrac12 +
        \dfrac\iu2\,\frac{\cos(\theta/2)}{\sin(\theta/2)}$ voor
        $\theta \notin 2\pi\Z$ (halvehoekontbinding,
        \cref{met:b1:complex:trig}), en leid uit vraag 11
        af dat $\sum_{k=1}^{n-1}
        \frac{\cos(k\pi/n)}{\sin(k\pi/n)} = 0$ --- ook zichtbaar
        via de symmetrie $k \leftrightarrow n - k$.
  \item Bewijs, door de identiteit van vraag 11 te differentiëren
        (d.w.z.\ door $\bigl(\frac{P'}P\bigr)' = \frac{P''}P -
        \bigl(\frac{P'}P\bigr)^2$ te evalueren in $X = 1$), dat
        \[
        \sum_{k=1}^{n-1}\frac{1}{(1 - \omega^k)^2}
        = \frac{(n-1)(5-n)}{12} .
        \]
  \item Leid, door $\cot t = \frac{\cos t}{\sin t}$ te schrijven, uit
        de vragen 12--13 de twee gesloten vormen af
        \[
        \sum_{k=1}^{n-1}\cot^2\frac{k\pi}n = \frac{(n-1)(n-2)}3,
        \qquad
        \sum_{k=1}^{n-1}\frac1{\sin^2\frac{k\pi}n}
        = \frac{n^2-1}3 .
        \]
  \item Verifieer beide formules met de hand voor $n = 3$ en $n = 4$.
  \item Bewijs de ongelijkheden $\cot t < \frac1t < \frac1{\sin t}$
        voor $t \in \intoo0{\frac\pi2}$ (uit $\sin t < t < \tan
        t$), en leid, voor $n = 2m + 1$ en $1 \leq k \leq m$, af:
        \[
        \cot^2\frac{k\pi}n \;<\; \frac{n^2}{k^2\pi^2}
        \;<\; \frac1{\sin^2\frac{k\pi}n} .
        \]
  \item Sommeer deze ongelijkheden voor $k = 1, \dots, m$ (gebruik de
        symmetrie $k \leftrightarrow n - k$ om de formules van
        vraag 14 te halveren) en knel in:
        \[
        \sum_{k=1}^{\infty}\frac1{k^2} = \frac{\pi^2}6 .
        \]
\end{enumerate}

\medskip
\noindent\textbf{Deel IV --- Hogere bouwstenen.}
\begin{enumerate}[resume]
  \item Bewijs, door de splitsing van $\frac1{X^2-1}$ te kwadrateren
        en de kruisterm opnieuw te splitsen, dat
        \[
        \frac1{(X^2-1)^2}
        = \frac14\Bigl(\frac1{(X-1)^2} + \frac1{(X+1)^2}\Bigr)
        - \frac14\Bigl(\frac1{X-1} - \frac1{X+1}\Bigr),
        \]
        en controleer het in $X = 0$.
  \item Combineer vraag 18, de telescoop en Eulers waarde
        (vraag 17) om te bewijzen dat
        \[
        \sum_{n=2}^{\infty}\frac1{(n^2-1)^2}
        = \frac{\pi^2}{12} - \frac{11}{16} ,
        \]
        en bevestig de waarde numeriek tot op drie decimalen.
  \item Differentieer de identiteit van vraag 8 om een gesloten vorm
        te bekomen voor $\sum_{k=0}^{n-1}\frac1{(X -
        \omega^k)^2}$, en controleer die in $X = 2$, $n = 2$.
  \item Bewijs dat voor $k \geq 2$,
        \[
        \sum_{n=k}^{\infty}\frac{1}{\binom nk} = \frac{k}{k-1} .
        \]
        \emph{(Herleid tot vraag 4 door $1/\binom nk$ met faculteiten
        te schrijven.)}
  \item Geef voor $k = 3$ de exacte deelsom
        $\sum_{n=3}^{N}\frac1{\binom n3}$ en haar limiet.
\end{enumerate}

\medskip
\noindent\textbf{Deel V --- Synthese.}
\begin{enumerate}[resume]
  \item Schrijf als slotberekening de volledige reële splitsing van
        $\dfrac1{X^6 - 1}$ uit en controleer die in $X =
        0$.
  \item Waar precies gebruikte het probleem: (i) de uniciteit van
        de splitsing; (ii) de \omterm{def:b1:complex:unity}{eenheidswortels} uit
        \cref{ch:b1:complex}; (iii) de logaritmische afgeleide
        uit \cref{exo:b1:poly:11}? Eén zin voor elk.
  \item Synthese, in een korte alinea: één algebraïsche identiteit
        --- het breken van een breuk in bouwstenen --- genereerde
        exacte sommen, goniometrische identiteiten, en $\pi^2/6$.
        Becommentarieer de arbeidsverdeling tussen algebra (exacte
        splitsingen, overal geldig) en analyse (limieten,
        inknellen), en wijs aan waar elke draad
        geïndustrialiseerd wordt: telescoperen en vergelijken in
        \cref{ch:b1:series}, integratie van de bouwstenen in
        \cref{ch:b1:integration}.
\end{enumerate}
\end{problem}
